// The input only specifies vertices, so the container is interpreted as
// the convex hull of those vertices. Packing uses an FCC lattice of
// tetrahedron circumcenters, which guarantees non-overlap via sphere spacing.

struct Vec3 {
    double x = 0.0;
    double y = 0.0;
    double z = 0.0;
};

static Vec3 operator+(const Vec3& a, const Vec3& b) {
    return {a.x + b.x, a.y + b.y, a.z + b.z};
}

static Vec3 operator-(const Vec3& a, const Vec3& b) {
    return {a.x - b.x, a.y - b.y, a.z - b.z};
}

static Vec3 operator*(const Vec3& a, double s) {
    return {a.x * s, a.y * s, a.z * s};
}

static Vec3 operator*(double s, const Vec3& a) {
    return a * s;
}

static Vec3 operator/(const Vec3& a, double s) {
    return {a.x / s, a.y / s, a.z / s};
}

static double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

static double norm2(const Vec3& v) {
    return dot(v, v);
}

static double norm(const Vec3& v) {
    return std::sqrt(norm2(v));
}

static Vec3 rotate_xyz(const Vec3& p, double rx, double ry, double rz) {
    const double cx = std::cos(rx);
    const double sx = std::sin(rx);
    const double cy = std::cos(ry);
    const double sy = std::sin(ry);
    const double cz = std::cos(rz);
    const double sz = std::sin(rz);

    Vec3 v = {p.x, cx * p.y - sx * p.z, sx * p.y + cx * p.z};
    Vec3 w = {cy * v.x + sy * v.z, v.y, -sy * v.x + cy * v.z};
    return {cz * w.x - sz * w.y, sz * w.x + cz * w.y, w.z};
}

struct Face {
    int a = -1;
    int b = -1;
    int c = -1;
    Vec3 n{};
    double d = 0.0;
    bool alive = false;
};

struct PlaneSet {
    std::vector<double> nx;
    std::vector<double> ny;
    std::vector<double> nz;
    std::vector<double> d;

    std::size_t size() const {
        return d.size();
    }
};

struct Hull {
    bool valid = false;
    std::vector<Vec3> points;
    std::vector<Face> faces;
    Vec3 inside{};
    Vec3 bmin{};
    Vec3 bmax{};
    double eps = 1e-9;
    double volume = 0.0;
};

struct Orientation {
    double rx = 0.0;
    double ry = 0.0;
    double rz = 0.0;
};

struct SearchResult {
    long long count = 0;
    int orientation_index = -1;
    int phase_index = -1;
    bool single_fallback = false;
    Vec3 single_center{};
};

static constexpr double PI = 3.141592653589793238462643383279502884;

static bool lex_less(const Vec3& a, const Vec3& b) {
    if (a.x != b.x) {
        return a.x < b.x;
    }
    if (a.y != b.y) {
        return a.y < b.y;
    }
    return a.z < b.z;
}

static Face make_face(int a, int b, int c, const std::vector<Vec3>& pts, const Vec3& inside, double eps) {
    Face f;
    f.a = a;
    f.b = b;
    f.c = c;
    f.alive = false;

    Vec3 n = cross(pts[b] - pts[a], pts[c] - pts[a]);
    const double len = norm(n);
    if (len <= eps) {
        return f;
    }
    n = n / len;
    double d = dot(n, pts[a]);
    if (dot(n, inside) > d) {
        std::swap(f.b, f.c);
        n = n * -1.0;
        d = -d;
    }
    f.n = n;
    f.d = d;
    f.alive = true;
    return f;
}

static std::uint64_t edge_key(int u, int v) {
    const std::uint32_t a = static_cast<std::uint32_t>(std::min(u, v));
    const std::uint32_t b = static_cast<std::uint32_t>(std::max(u, v));
    return (static_cast<std::uint64_t>(a) << 32) | b;
}

static Hull build_convex_hull(std::vector<Vec3> pts) {
    Hull hull;
    if (pts.size() < 4) {
        return hull;
    }

    std::sort(pts.begin(), pts.end(), lex_less);
    pts.erase(std::unique(pts.begin(), pts.end(), [](const Vec3& a, const Vec3& b) {
        return a.x == b.x && a.y == b.y && a.z == b.z;
    }), pts.end());
    if (pts.size() < 4) {
        return hull;
    }

    Vec3 bmin = pts.front();
    Vec3 bmax = pts.front();
    for (const Vec3& p : pts) {
        bmin.x = std::min(bmin.x, p.x);
        bmin.y = std::min(bmin.y, p.y);
        bmin.z = std::min(bmin.z, p.z);
        bmax.x = std::max(bmax.x, p.x);
        bmax.y = std::max(bmax.y, p.y);
        bmax.z = std::max(bmax.z, p.z);
    }
    const double diag = norm(bmax - bmin);
    const double eps = std::max(1e-9, diag * 1e-9);

    const int n = static_cast<int>(pts.size());
    int i0 = 0;
    for (int i = 1; i < n; ++i) {
        if (lex_less(pts[i], pts[i0])) {
            i0 = i;
        }
    }

    int i1 = -1;
    double best_d2 = -1.0;
    for (int i = 0; i < n; ++i) {
        if (i == i0) {
            continue;
        }
        const double d2 = norm2(pts[i] - pts[i0]);
        if (d2 > best_d2) {
            best_d2 = d2;
            i1 = i;
        }
    }
    if (i1 < 0 || best_d2 <= eps * eps) {
        return hull;
    }

    int i2 = -1;
    double best_line_d2 = -1.0;
    const Vec3 line = pts[i1] - pts[i0];
    for (int i = 0; i < n; ++i) {
        if (i == i0 || i == i1) {
            continue;
        }
        const Vec3 c = cross(line, pts[i] - pts[i0]);
        const double d2 = norm2(c);
        if (d2 > best_line_d2) {
            best_line_d2 = d2;
            i2 = i;
        }
    }
    if (i2 < 0 || best_line_d2 <= eps * eps) {
        return hull;
    }

    int i3 = -1;
    double best_plane_dist = -1.0;
    Vec3 base_normal = cross(pts[i1] - pts[i0], pts[i2] - pts[i0]);
    const double base_normal_len = norm(base_normal);
    if (base_normal_len <= eps) {
        return hull;
    }
    base_normal = base_normal / base_normal_len;
    for (int i = 0; i < n; ++i) {
        if (i == i0 || i == i1 || i == i2) {
            continue;
        }
        const double dist = std::abs(dot(base_normal, pts[i] - pts[i0]));
        if (dist > best_plane_dist) {
            best_plane_dist = dist;
            i3 = i;
        }
    }
    if (i3 < 0 || best_plane_dist <= eps) {
        return hull;
    }

    const Vec3 inside = (pts[i0] + pts[i1] + pts[i2] + pts[i3]) / 4.0;
    std::vector<Face> faces;
    faces.reserve(static_cast<std::size_t>(pts.size()) * 4);

    auto push_face = [&](int a, int b, int c) {
        Face f = make_face(a, b, c, pts, inside, eps);
        if (f.alive) {
            faces.push_back(f);
        }
    };

    push_face(i0, i1, i2);
    push_face(i0, i3, i1);
    push_face(i0, i2, i3);
    push_face(i1, i3, i2);
    if (faces.size() < 4) {
        return hull;
    }

    std::vector<int> order;
    order.reserve(pts.size() - 4);
    for (int i = 0; i < n; ++i) {
        if (i != i0 && i != i1 && i != i2 && i != i3) {
            order.push_back(i);
        }
    }

    std::mt19937_64 rng(0xC0FFEEULL);
    std::shuffle(order.begin(), order.end(), rng);

    const double visible_eps = eps * 4.0;
    for (int idx : order) {
        std::vector<int> visible;
        visible.reserve(faces.size());
        for (int fi = 0; fi < static_cast<int>(faces.size()); ++fi) {
            const Face& f = faces[fi];
            if (!f.alive) {
                continue;
            }
            if (dot(f.n, pts[idx]) - f.d > visible_eps) {
                visible.push_back(fi);
            }
        }
        if (visible.empty()) {
            continue;
        }

        struct EdgeInfo {
            int u = -1;
            int v = -1;
            int count = 0;
        };

        std::unordered_map<std::uint64_t, EdgeInfo> edges;
        edges.reserve(visible.size() * 3 + 8);

        auto add_edge = [&](int u, int v) {
            const std::uint64_t key = edge_key(u, v);
            auto it = edges.find(key);
            if (it == edges.end()) {
                edges.emplace(key, EdgeInfo{u, v, 1});
            } else {
                ++it->second.count;
            }
        };

        for (int fi : visible) {
            Face& f = faces[fi];
            if (!f.alive) {
                continue;
            }
            add_edge(f.a, f.b);
            add_edge(f.b, f.c);
            add_edge(f.c, f.a);
            f.alive = false;
        }

        for (const auto& kv : edges) {
            const EdgeInfo& e = kv.second;
            if (e.count != 1) {
                continue;
            }
            if (norm(cross(pts[e.v] - pts[e.u], pts[idx] - pts[e.u])) <= eps) {
                continue;
            }
            Face f = make_face(e.u, e.v, idx, pts, inside, eps);
            if (f.alive) {
                faces.push_back(f);
            }
        }
    }

    std::vector<Face> alive_faces;
    alive_faces.reserve(faces.size());
    for (const Face& f : faces) {
        if (f.alive) {
            alive_faces.push_back(f);
        }
    }
    if (alive_faces.size() < 4) {
        return hull;
    }

    long double volume = 0.0L;
    for (const Face& f : alive_faces) {
        const Vec3& a = pts[f.a];
        const Vec3& b = pts[f.b];
        const Vec3& c = pts[f.c];
        volume += static_cast<long double>(dot(a, cross(b, c)));
    }
    volume = std::abs(volume) / 6.0L;
    if (volume <= static_cast<long double>(eps) * eps * eps) {
        return hull;
    }

    hull.valid = true;
    hull.points = std::move(pts);
    hull.faces = std::move(alive_faces);
    hull.inside = inside;
    hull.bmin = bmin;
    hull.bmax = bmax;
    hull.eps = eps;
    hull.volume = static_cast<double>(volume);
    return hull;
}

static PlaneSet make_plane_set(const std::vector<Face>& faces) {
    PlaneSet planes;
    planes.nx.reserve(faces.size());
    planes.ny.reserve(faces.size());
    planes.nz.reserve(faces.size());
    planes.d.reserve(faces.size());
    for (const Face& f : faces) {
        planes.nx.push_back(f.n.x);
        planes.ny.push_back(f.n.y);
        planes.nz.push_back(f.n.z);
        planes.d.push_back(f.d);
    }
    return planes;
}

static double halton(std::uint64_t index, int base) {
    double f = 1.0;
    double r = 0.0;
    while (index > 0) {
        f /= static_cast<double>(base);
        r += f * static_cast<double>(index % static_cast<std::uint64_t>(base));
        index /= static_cast<std::uint64_t>(base);
    }
    return r;
}

static bool contains_shifted(const PlaneSet& hull, const std::vector<double>& shifted_d, const Vec3& p, double eps) {
    for (std::size_t i = 0; i < shifted_d.size(); ++i) {
        const double side = hull.nx[i] * p.x + hull.ny[i] * p.y + hull.nz[i] * p.z;
        if (side > shifted_d[i] + eps) {
            return false;
        }
    }
    return true;
}

template <class F>
static long long enumerate_fcc(
    const PlaneSet& hull,
    const std::vector<double>& shifted_d,
    const Vec3& cmin,
    const Vec3& cmax,
    double cell,
    const Vec3& phase,
    double eps,
    F&& on_point
) {
    if (cmin.x > cmax.x || cmin.y > cmax.y || cmin.z > cmax.z) {
        return 0;
    }

    const double half = 0.5 * cell;
    const double xmin = cmin.x - eps;
    const double ymin = cmin.y - eps;
    const double zmin = cmin.z - eps;
    const double xmax = cmax.x + eps;
    const double ymax = cmax.y + eps;
    const double zmax = cmax.z + eps;

    const long long ix0 = static_cast<long long>(std::floor((xmin - phase.x) / cell)) - 1;
    const long long iy0 = static_cast<long long>(std::floor((ymin - phase.y) / cell)) - 1;
    const long long iz0 = static_cast<long long>(std::floor((zmin - phase.z) / cell)) - 1;
    const long long ix1 = static_cast<long long>(std::ceil((xmax - phase.x) / cell)) + 1;
    const long long iy1 = static_cast<long long>(std::ceil((ymax - phase.y) / cell)) + 1;
    const long long iz1 = static_cast<long long>(std::ceil((zmax - phase.z) / cell)) + 1;

    long long count = 0;
    auto test_point = [&](double x, double y, double z) {
        if (x < xmin || x > xmax || y < ymin || y > ymax || z < zmin || z > zmax) {
            return;
        }
        Vec3 p{x, y, z};
        if (contains_shifted(hull, shifted_d, p, eps)) {
            on_point(p);
            ++count;
        }
    };

    for (long long ix = ix0; ix <= ix1; ++ix) {
        const double x0 = phase.x + cell * static_cast<double>(ix);
        const double x1 = x0 + half;
        for (long long iy = iy0; iy <= iy1; ++iy) {
            const double y0 = phase.y + cell * static_cast<double>(iy);
            const double y1 = y0 + half;
            for (long long iz = iz0; iz <= iz1; ++iz) {
                const double z0 = phase.z + cell * static_cast<double>(iz);
                const double z1 = z0 + half;
                test_point(x0, y0, z0);
                test_point(x0, y1, z1);
                test_point(x1, y0, z1);
                test_point(x1, y1, z0);
            }
        }
    }

    return count;
}

static std::vector<double> shifted_planes(
    const PlaneSet& hull,
    const std::array<Vec3, 4>& offsets
) {
    std::vector<double> shifted(hull.size(), 0.0);
    for (std::size_t i = 0; i < hull.size(); ++i) {
        double support = -std::numeric_limits<double>::infinity();
        for (const Vec3& o : offsets) {
            support = std::max(support, hull.nx[i] * o.x + hull.ny[i] * o.y + hull.nz[i] * o.z);
        }
        shifted[i] = hull.d[i] - support;
    }
    return shifted;
}

static std::array<Vec3, 4> rotated_tetra_offsets(double edge, const Orientation& ori) {
    const double s = edge / (2.0 * std::sqrt(2.0));
    std::array<Vec3, 4> base = {
        Vec3{ s,  s,  s},
        Vec3{ s, -s, -s},
        Vec3{-s,  s, -s},
        Vec3{-s, -s,  s}
    };
    for (Vec3& v : base) {
        v = rotate_xyz(v, ori.rx, ori.ry, ori.rz);
    }
    return base;
}

static Vec3 average_point(const std::vector<Vec3>& pts) {
    Vec3 acc{};
    for (const Vec3& p : pts) {
        acc = acc + p;
    }
    return acc / static_cast<double>(pts.size());
}

static SearchResult solve_with_lattices(
    const Hull& hull,
    const PlaneSet& planes,
    double edge
) {
    SearchResult best;

    const double cell = std::sqrt(3.0) * edge;
    const double point_cell_volume = (cell * cell * cell) / 4.0;
    const double estimated_count = hull.volume / std::max(point_cell_volume, 1e-18);

    const std::vector<Orientation> all_orientations = {
        {0.0, 0.0, 0.0},
        {PI * 0.5, 0.0, 0.0},
        {PI, 0.0, 0.0},
        {PI * 1.5, 0.0, 0.0},
        {0.0, PI * 0.5, 0.0},
        {0.0, PI, 0.0},
        {0.0, PI * 1.5, 0.0},
        {0.0, 0.0, PI * 0.5},
        {0.0, 0.0, PI},
        {0.0, 0.0, PI * 1.5}
    };

    std::vector<std::pair<int, Orientation>> orientations;
    if (estimated_count > 200000.0) {
        orientations.push_back({0, all_orientations[0]});
    } else if (estimated_count > 20000.0) {
        orientations.push_back({0, all_orientations[0]});
        orientations.push_back({1, all_orientations[1]});
        orientations.push_back({4, all_orientations[4]});
        orientations.push_back({7, all_orientations[7]});
    } else {
        for (int i = 0; i < static_cast<int>(all_orientations.size()); ++i) {
            orientations.push_back({i, all_orientations[i]});
        }
    }

    int phase_count = 4;
    if (estimated_count < 1000.0) {
        phase_count = 32;
    } else if (estimated_count < 20000.0) {
        phase_count = 16;
    } else if (estimated_count < 200000.0) {
        phase_count = 8;
    }

    std::vector<Vec3> phases;
    phases.reserve(phase_count);
    phases.push_back({0.0, 0.0, 0.0});
    for (int i = 1; i < phase_count; ++i) {
        phases.push_back({
            halton(static_cast<std::uint64_t>(i), 2) * cell,
            halton(static_cast<std::uint64_t>(i), 3) * cell,
            halton(static_cast<std::uint64_t>(i), 5) * cell
        });
    }

    for (int oi = 0; oi < static_cast<int>(orientations.size()); ++oi) {
        const auto offsets = rotated_tetra_offsets(edge, orientations[oi].second);
        Vec3 pad{};
        for (const Vec3& o : offsets) {
            pad.x = std::max(pad.x, std::abs(o.x));
            pad.y = std::max(pad.y, std::abs(o.y));
            pad.z = std::max(pad.z, std::abs(o.z));
        }
        const Vec3 cmin = {hull.bmin.x + pad.x, hull.bmin.y + pad.y, hull.bmin.z + pad.z};
        const Vec3 cmax = {hull.bmax.x - pad.x, hull.bmax.y - pad.y, hull.bmax.z - pad.z};
        if (cmin.x > cmax.x || cmin.y > cmax.y || cmin.z > cmax.z) {
            continue;
        }

        const std::vector<double> shifted_d = shifted_planes(planes, offsets);
        for (int pi = 0; pi < static_cast<int>(phases.size()); ++pi) {
            const long long count = enumerate_fcc(
                planes,
                shifted_d,
                cmin,
                cmax,
                cell,
                phases[pi],
                hull.eps * 8.0,
                [](const Vec3&) {}
            );
            if (count > best.count) {
                best.count = count;
                best.orientation_index = orientations[oi].first;
                best.phase_index = pi;
            }
        }
    }

    if (best.count > 0) {
        return best;
    }

    const Vec3 avg = average_point(hull.points);
    const Vec3 mid = (hull.bmin + hull.bmax) * 0.5;
    const std::vector<Vec3> seed_points = {avg, mid, hull.inside};

    for (int oi = 0; oi < static_cast<int>(orientations.size()); ++oi) {
        const auto offsets = rotated_tetra_offsets(edge, orientations[oi].second);
        Vec3 pad{};
        for (const Vec3& o : offsets) {
            pad.x = std::max(pad.x, std::abs(o.x));
            pad.y = std::max(pad.y, std::abs(o.y));
            pad.z = std::max(pad.z, std::abs(o.z));
        }
        const Vec3 cmin = {hull.bmin.x + pad.x, hull.bmin.y + pad.y, hull.bmin.z + pad.z};
        const Vec3 cmax = {hull.bmax.x - pad.x, hull.bmax.y - pad.y, hull.bmax.z - pad.z};
        if (cmin.x > cmax.x || cmin.y > cmax.y || cmin.z > cmax.z) {
            continue;
        }

        const std::vector<double> shifted_d = shifted_planes(planes, offsets);

        for (const Vec3& seed : seed_points) {
            if (seed.x < cmin.x || seed.x > cmax.x ||
                seed.y < cmin.y || seed.y > cmax.y ||
                seed.z < cmin.z || seed.z > cmax.z) {
                continue;
            }
            if (contains_shifted(planes, shifted_d, seed, hull.eps * 8.0)) {
                best.count = 1;
                best.orientation_index = orientations[oi].first;
                best.phase_index = -1;
                best.single_fallback = true;
                best.single_center = seed;
                return best;
            }
        }

        constexpr int sample_budget = 4096;
        for (int i = 1; i <= sample_budget; ++i) {
            const Vec3 p = {
                cmin.x + (cmax.x - cmin.x) * halton(static_cast<std::uint64_t>(i), 2),
                cmin.y + (cmax.y - cmin.y) * halton(static_cast<std::uint64_t>(i), 3),
                cmin.z + (cmax.z - cmin.z) * halton(static_cast<std::uint64_t>(i), 5)
            };
            if (contains_shifted(planes, shifted_d, p, hull.eps * 8.0)) {
                best.count = 1;
                best.orientation_index = orientations[oi].first;
                best.phase_index = -1;
                best.single_fallback = true;
                best.single_center = p;
                return best;
            }
        }
    }

    return best;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int v = 0;
    double edge = 0.0;
    if (!(std::cin >> v >> edge)) {
        return 0;
    }

    std::vector<Vec3> pts;
    pts.reserve(std::max(v, 0));
    for (int i = 0; i < v; ++i) {
        Vec3 p;
        std::cin >> p.x >> p.y >> p.z;
        pts.push_back(p);
    }

    if (v < 4 || edge <= 0.0) {
        std::cout << 0 << '\n';
        return 0;
    }

    Hull hull = build_convex_hull(std::move(pts));
    if (!hull.valid) {
        std::cout << 0 << '\n';
        return 0;
    }

    const PlaneSet planes = make_plane_set(hull.faces);
    if (planes.size() < 4) {
        std::cout << 0 << '\n';
        return 0;
    }

    const std::vector<Orientation> all_orientations = {
        {0.0, 0.0, 0.0},
        {PI * 0.5, 0.0, 0.0},
        {PI, 0.0, 0.0},
        {PI * 1.5, 0.0, 0.0},
        {0.0, PI * 0.5, 0.0},
        {0.0, PI, 0.0},
        {0.0, PI * 1.5, 0.0},
        {0.0, 0.0, PI * 0.5},
        {0.0, 0.0, PI},
        {0.0, 0.0, PI * 1.5}
    };

    SearchResult best = solve_with_lattices(hull, planes, edge);
    std::cout << best.count << '\n';
    if (best.count <= 0 || best.orientation_index < 0) {
        return 0;
    }

    const Orientation& ori = all_orientations[best.orientation_index];
    std::cout << std::setprecision(12);

    if (best.single_fallback) {
        std::cout
            << best.single_center.x << ' '
            << best.single_center.y << ' '
            << best.single_center.z << ' '
            << ori.rx << ' '
            << ori.ry << ' '
            << ori.rz << '\n';
        return 0;
    }

    const double cell = std::sqrt(3.0) * edge;
    const int phase_count =
        (hull.volume / std::max((cell * cell * cell) / 4.0, 1e-18) < 1000.0) ? 32 :
        (hull.volume / std::max((cell * cell * cell) / 4.0, 1e-18) < 20000.0) ? 16 :
        (hull.volume / std::max((cell * cell * cell) / 4.0, 1e-18) < 200000.0) ? 8 : 4;

    std::vector<Vec3> phases;
    phases.reserve(phase_count);
    phases.push_back({0.0, 0.0, 0.0});
    for (int i = 1; i < phase_count; ++i) {
        phases.push_back({
            halton(static_cast<std::uint64_t>(i), 2) * cell,
            halton(static_cast<std::uint64_t>(i), 3) * cell,
            halton(static_cast<std::uint64_t>(i), 5) * cell
        });
    }

    const auto offsets = rotated_tetra_offsets(edge, ori);
    Vec3 pad{};
    for (const Vec3& o : offsets) {
        pad.x = std::max(pad.x, std::abs(o.x));
        pad.y = std::max(pad.y, std::abs(o.y));
        pad.z = std::max(pad.z, std::abs(o.z));
    }
    const Vec3 cmin = {hull.bmin.x + pad.x, hull.bmin.y + pad.y, hull.bmin.z + pad.z};
    const Vec3 cmax = {hull.bmax.x - pad.x, hull.bmax.y - pad.y, hull.bmax.z - pad.z};
    const std::vector<double> shifted_d = shifted_planes(planes, offsets);

    enumerate_fcc(
        planes,
        shifted_d,
        cmin,
        cmax,
        cell,
        phases[best.phase_index],
        hull.eps * 8.0,
        [&](const Vec3& p) {
            std::cout
                << p.x << ' '
                << p.y << ' '
                << p.z << ' '
                << ori.rx << ' '
                << ori.ry << ' '
                << ori.rz << '\n';
        }
    );

    return 0;
}
