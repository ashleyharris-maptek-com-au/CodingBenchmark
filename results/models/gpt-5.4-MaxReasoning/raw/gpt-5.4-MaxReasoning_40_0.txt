#include <algorithm>
#include <cstdint>
#include <iostream>
#include <numeric>
#include <queue>
#include <utility>
#include <vector>
using namespace std;

using ll = long long;
using U64 = unsigned long long;

struct MaxWeightClique {
    int n;
    int words;
    vector<ll> weight;
    vector<vector<U64>> conflict;
    vector<int> compatible_degree;
    vector<int> best_clique;
    vector<int> current_clique;
    ll best_weight = 0;
    ll current_weight = 0;

    static int popcount64(U64 x) {
        int count = 0;
        while (x != 0) {
            x &= (x - 1);
            ++count;
        }
        return count;
    }

    MaxWeightClique(vector<ll> w, vector<vector<U64>> conflict_bits)
        : n(static_cast<int>(w.size())),
          words((n + 63) / 64),
          weight(move(w)),
          conflict(move(conflict_bits)),
          compatible_degree(n, 0) {
        for (int v = 0; v < n; ++v) {
            int conflicts = 0;
            for (U64 block : conflict[v]) {
                conflicts += popcount64(block);
            }
            compatible_degree[v] = n - 1 - conflicts;
        }
    }

    bool compatible(int a, int b) const {
        return ((conflict[a][b >> 6] >> (b & 63)) & 1ULL) == 0ULL;
    }

    void update_best(const vector<int>& clique, ll total_weight) {
        if (total_weight > best_weight) {
            best_weight = total_weight;
            best_clique = clique;
        }
    }

    auto heuristic_order() const {
        return [&](int a, int b) {
            if (weight[a] != weight[b]) {
                return weight[a] > weight[b];
            }
            if (compatible_degree[a] != compatible_degree[b]) {
                return compatible_degree[a] > compatible_degree[b];
            }
            return a < b;
        };
    }

    void greedy_seed() {
        if (n == 0) {
            return;
        }

        vector<int> verts(n);
        iota(verts.begin(), verts.end(), 0);
        auto cmp = heuristic_order();
        sort(verts.begin(), verts.end(), cmp);

        auto run_from = [&](int start_pos) {
            int start = verts[start_pos];
            ll total = weight[start];
            vector<int> clique{start};
            vector<int> cand;
            cand.reserve(n);
            for (int u : verts) {
                if (u != start && compatible(start, u)) {
                    cand.push_back(u);
                }
            }
            sort(cand.begin(), cand.end(), cmp);

            while (!cand.empty()) {
                int v = cand.front();
                clique.push_back(v);
                total += weight[v];

                vector<int> next;
                next.reserve(cand.size());
                for (int u : cand) {
                    if (u != v && compatible(v, u)) {
                        next.push_back(u);
                    }
                }
                sort(next.begin(), next.end(), cmp);
                cand.swap(next);
            }

            update_best(clique, total);
        };

        int tries = min(n, 24);
        for (int i = 0; i < tries; ++i) {
            run_from(i);
        }
    }

    void color_sort(const vector<int>& p, vector<int>& order, vector<ll>& bounds) const {
        vector<int> verts = p;
        auto cmp = heuristic_order();
        sort(verts.begin(), verts.end(), cmp);

        vector<vector<int>> color_classes;
        vector<vector<U64>> color_masks;
        vector<ll> color_max_weight;

        for (int v : verts) {
            int chosen = -1;
            for (int c = 0; c < static_cast<int>(color_classes.size()); ++c) {
                bool ok = true;
                for (int i = 0; i < words; ++i) {
                    if (color_masks[c][i] & ~conflict[v][i]) {
                        ok = false;
                        break;
                    }
                }
                if (ok) {
                    chosen = c;
                    break;
                }
            }

            if (chosen == -1) {
                chosen = static_cast<int>(color_classes.size());
                color_classes.push_back({});
                color_masks.push_back(vector<U64>(words, 0));
                color_max_weight.push_back(0);
            }

            color_classes[chosen].push_back(v);
            color_masks[chosen][v >> 6] |= 1ULL << (v & 63);
            color_max_weight[chosen] = max(color_max_weight[chosen], weight[v]);
        }

        order.clear();
        bounds.clear();
        order.reserve(p.size());
        bounds.reserve(p.size());

        ll prefix = 0;
        for (int c = 0; c < static_cast<int>(color_classes.size()); ++c) {
            prefix += color_max_weight[c];
            for (auto it = color_classes[c].rbegin(); it != color_classes[c].rend(); ++it) {
                order.push_back(*it);
                bounds.push_back(prefix);
            }
        }
    }

    void expand(const vector<int>& p) {
        if (p.empty()) {
            update_best(current_clique, current_weight);
            return;
        }

        vector<int> order;
        vector<ll> bounds;
        color_sort(p, order, bounds);

        for (int i = static_cast<int>(order.size()) - 1; i >= 0; --i) {
            if (current_weight + bounds[i] <= best_weight) {
                return;
            }

            int v = order[i];
            current_clique.push_back(v);
            current_weight += weight[v];

            vector<int> next;
            next.reserve(i);
            for (int j = 0; j < i; ++j) {
                int u = order[j];
                if (compatible(v, u)) {
                    next.push_back(u);
                }
            }

            if (next.empty()) {
                update_best(current_clique, current_weight);
            } else {
                expand(next);
            }

            current_weight -= weight[v];
            current_clique.pop_back();
        }
    }

    pair<ll, vector<int>> solve() {
        greedy_seed();
        vector<int> p(n);
        iota(p.begin(), p.end(), 0);
        expand(p);
        return {best_weight, best_clique};
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) {
        return 0;
    }

    vector<ll> c(n);
    for (int i = 0; i < n; ++i) {
        cin >> c[i];
    }

    vector<char> forbidden(n, false);
    vector<vector<int>> graph(n);

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u == v) {
            forbidden[u] = true;
            continue;
        }
        graph[u].push_back(v);
        graph[v].push_back(u);
    }

    vector<int> answer(n, 0);
    vector<char> active(n, false);
    for (int i = 0; i < n; ++i) {
        active[i] = (c[i] > 0 && !forbidden[i]);
    }

    ll objective = 0;
    vector<char> visited(n, false);
    vector<int> local_index(n, -1);

    for (int start = 0; start < n; ++start) {
        if (!active[start] || visited[start]) {
            continue;
        }

        vector<int> component;
        queue<int> q;
        q.push(start);
        visited[start] = true;

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            component.push_back(v);
            for (int u : graph[v]) {
                if (active[u] && !visited[u]) {
                    visited[u] = true;
                    q.push(u);
                }
            }
        }

        if (component.size() == 1) {
            int v = component[0];
            answer[v] = 1;
            objective += c[v];
            continue;
        }

        int k = static_cast<int>(component.size());
        int words = (k + 63) / 64;
        vector<ll> local_weight(k);
        vector<vector<U64>> conflict(k, vector<U64>(words, 0));

        for (int i = 0; i < k; ++i) {
            local_index[component[i]] = i;
            local_weight[i] = c[component[i]];
        }

        for (int i = 0; i < k; ++i) {
            int gv = component[i];
            for (int gu : graph[gv]) {
                int j = local_index[gu];
                if (j != -1) {
                    conflict[i][j >> 6] |= 1ULL << (j & 63);
                }
            }
        }

        MaxWeightClique solver(move(local_weight), move(conflict));
        auto result = solver.solve();
        objective += result.first;
        for (int lv : result.second) {
            answer[component[lv]] = 1;
        }

        for (int gv : component) {
            local_index[gv] = -1;
        }
    }

    cout << objective << '\n';
    for (int i = 0; i < n; ++i) {
        if (i) {
            cout << ' ';
        }
        cout << answer[i];
    }
    cout << '\n';

    return 0;
}
