use std::cmp::Ordering;
use std::collections::VecDeque;
use std::io::{self, Read};

#[derive(Clone)]
struct Graph {
    n: usize,
    m: usize,
    adj: Vec<Vec<usize>>,
    adj_mat: Vec<Vec<bool>>,
    edges: Vec<(usize, usize)>,
    degree: Vec<usize>,
}

#[derive(Clone)]
struct State {
    side: Vec<i8>, // 1 = A, -1 = B, 0 = free
    to_a: Vec<i32>,
    to_b: Vec<i32>,
    free_deg: Vec<i32>,
    a_count: usize,
    b_count: usize,
    fixed_cut: i64,
}

#[derive(Clone)]
struct Best {
    cut: i64,
    in_a: Vec<bool>,
}

#[derive(Clone)]
struct RelaxedSol {
    lambda: i64,
    cost: i64,
    count_a: usize,
    unassigned: Vec<usize>,
    in_a_local: Vec<bool>,
}

struct BoundInfo {
    lb: i64,
    exact: Option<RelaxedSol>,
    over: Option<RelaxedSol>,
    under: Option<RelaxedSol>,
    guide: Option<RelaxedSol>,
}

#[derive(Clone)]
struct Edge {
    to: usize,
    rev: usize,
    cap: i64,
}

struct Dinic {
    g: Vec<Vec<Edge>>,
    level: Vec<i32>,
    it: Vec<usize>,
}

impl Dinic {
    fn new(n: usize) -> Self {
        Self {
            g: vec![Vec::new(); n],
            level: vec![0; n],
            it: vec![0; n],
        }
    }

    fn add_edge(&mut self, fr: usize, to: usize, cap: i64) {
        let rev_to = self.g[to].len();
        let rev_fr = self.g[fr].len();
        self.g[fr].push(Edge {
            to,
            rev: rev_to,
            cap,
        });
        self.g[to].push(Edge {
            to: fr,
            rev: rev_fr,
            cap: 0,
        });
    }

    fn add_undirected_cut_edge(&mut self, u: usize, v: usize, cap: i64) {
        self.add_edge(u, v, cap);
        self.add_edge(v, u, cap);
    }

    fn bfs(&mut self, s: usize, t: usize) -> bool {
        self.level.fill(-1);
        let mut q = VecDeque::new();
        self.level[s] = 0;
        q.push_back(s);
        while let Some(v) = q.pop_front() {
            for e in &self.g[v] {
                if e.cap > 0 && self.level[e.to] < 0 {
                    self.level[e.to] = self.level[v] + 1;
                    q.push_back(e.to);
                }
            }
        }
        self.level[t] >= 0
    }

    fn dfs(&mut self, v: usize, t: usize, f: i64) -> i64 {
        if v == t {
            return f;
        }
        while self.it[v] < self.g[v].len() {
            let i = self.it[v];
            let to = self.g[v][i].to;
            let cap = self.g[v][i].cap;
            if cap > 0 && self.level[v] < self.level[to] {
                let d = self.dfs(to, t, f.min(cap));
                if d > 0 {
                    let rev = self.g[v][i].rev;
                    self.g[v][i].cap -= d;
                    self.g[to][rev].cap += d;
                    return d;
                }
            }
            self.it[v] += 1;
        }
        0
    }

    fn max_flow(&mut self, s: usize, t: usize) -> i64 {
        let mut flow = 0;
        while self.bfs(s, t) {
            self.it.fill(0);
            loop {
                let f = self.dfs(s, t, i64::MAX / 4);
                if f == 0 {
                    break;
                }
                flow += f;
            }
        }
        flow
    }

    fn source_side(&self, s: usize) -> Vec<bool> {
        let mut seen = vec![false; self.g.len()];
        let mut q = VecDeque::new();
        seen[s] = true;
        q.push_back(s);
        while let Some(v) = q.pop_front() {
            for e in &self.g[v] {
                if e.cap > 0 && !seen[e.to] {
                    seen[e.to] = true;
                    q.push_back(e.to);
                }
            }
        }
        seen
    }
}

fn cut_of_partition(graph: &Graph, in_a: &[bool]) -> i64 {
    graph
        .edges
        .iter()
        .filter(|&&(u, v)| in_a[u] != in_a[v])
        .count() as i64
}

fn build_graph(input: &str) -> Graph {
    let mut it = input.split_whitespace();
    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();
    let mut adj = vec![Vec::new(); n];
    let mut adj_mat = vec![vec![false; n]; n];
    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if !adj_mat[u][v] {
            adj[u].push(v);
            adj[v].push(u);
            adj_mat[u][v] = true;
            adj_mat[v][u] = true;
            edges.push((u, v));
        }
    }
    let degree = adj.iter().map(|nbrs| nbrs.len()).collect::<Vec<_>>();
    Graph {
        n,
        m: edges.len(),
        adj,
        adj_mat,
        edges,
        degree,
    }
}

fn initial_state(graph: &Graph) -> State {
    let n = graph.n;
    let mut side = vec![0; n];
    let mut to_a = vec![0; n];
    let to_b = vec![0; n];
    let mut free_deg = graph.degree.iter().map(|&d| d as i32).collect::<Vec<_>>();
    if n > 0 {
        side[0] = 1;
        free_deg[0] = 0;
        for &u in &graph.adj[0] {
            to_a[u] += 1;
            free_deg[u] -= 1;
        }
    }
    State {
        side,
        to_a,
        to_b,
        free_deg,
        a_count: usize::from(n > 0),
        b_count: 0,
        fixed_cut: 0,
    }
}

fn assign_vertex(state: &State, graph: &Graph, v: usize, side_value: i8) -> State {
    let mut next = state.clone();
    next.side[v] = side_value;
    if side_value == 1 {
        next.a_count += 1;
        next.fixed_cut += next.to_b[v] as i64;
    } else {
        next.b_count += 1;
        next.fixed_cut += next.to_a[v] as i64;
    }
    next.free_deg[v] = 0;
    for &u in &graph.adj[v] {
        if next.side[u] == 0 {
            next.free_deg[u] -= 1;
            if side_value == 1 {
                next.to_a[u] += 1;
            } else {
                next.to_b[u] += 1;
            }
        }
    }
    next
}

fn full_assignment_from_state(state: &State, force_rest_to_a: Option<bool>) -> Vec<bool> {
    let mut in_a = vec![false; state.side.len()];
    for (u, &s) in state.side.iter().enumerate() {
        in_a[u] = match s {
            1 => true,
            -1 => false,
            _ => force_rest_to_a.unwrap_or(false),
        };
    }
    in_a
}

fn full_assignment_from_relaxed(state: &State, sol: &RelaxedSol) -> Vec<bool> {
    let mut in_a = vec![false; state.side.len()];
    for (u, &s) in state.side.iter().enumerate() {
        if s == 1 {
            in_a[u] = true;
        }
    }
    for (i, &u) in sol.unassigned.iter().enumerate() {
        in_a[u] = sol.in_a_local[i];
    }
    in_a
}

fn move_delta(graph: &Graph, in_a: &[bool], u: usize) -> i64 {
    let current = in_a[u];
    let mut same = 0i64;
    let mut opp = 0i64;
    for &v in &graph.adj[u] {
        if in_a[v] == current {
            same += 1;
        } else {
            opp += 1;
        }
    }
    same - opp
}

fn build_candidate(
    graph: &Graph,
    state: &State,
    target_a_total: usize,
    guide: Option<&RelaxedSol>,
    do_local_search: bool,
) -> Vec<bool> {
    let mut in_a = vec![false; graph.n];
    let mut free_vertices = Vec::new();
    for u in 0..graph.n {
        match state.side[u] {
            1 => in_a[u] = true,
            -1 => in_a[u] = false,
            _ => free_vertices.push(u),
        }
    }

    let rem_a = target_a_total - state.a_count;
    let mut free_a_count = 0usize;

    if let Some(sol) = guide {
        for (i, &u) in sol.unassigned.iter().enumerate() {
            in_a[u] = sol.in_a_local[i];
            if in_a[u] {
                free_a_count += 1;
            }
        }
    } else {
        let mut order = free_vertices.clone();
        order.sort_by_key(|&u| (state.to_b[u] - state.to_a[u]) as i64);
        for &u in order.iter().take(rem_a) {
            in_a[u] = true;
            free_a_count += 1;
        }
    }

    while free_a_count > rem_a {
        let mut best_u = None;
        let mut best_delta = i64::MAX;
        for &u in &free_vertices {
            if in_a[u] {
                let delta = move_delta(graph, &in_a, u);
                if delta < best_delta {
                    best_delta = delta;
                    best_u = Some(u);
                }
            }
        }
        let u = best_u.unwrap();
        in_a[u] = false;
        free_a_count -= 1;
    }

    while free_a_count < rem_a {
        let mut best_u = None;
        let mut best_delta = i64::MAX;
        for &u in &free_vertices {
            if !in_a[u] {
                let delta = move_delta(graph, &in_a, u);
                if delta < best_delta {
                    best_delta = delta;
                    best_u = Some(u);
                }
            }
        }
        let u = best_u.unwrap();
        in_a[u] = true;
        free_a_count += 1;
    }

    if do_local_search {
        loop {
            let mut best_pair = None;
            let mut best_delta = 0i64;
            for &a in &free_vertices {
                if !in_a[a] {
                    continue;
                }
                let da = move_delta(graph, &in_a, a);
                for &b in &free_vertices {
                    if in_a[b] {
                        continue;
                    }
                    let db = move_delta(graph, &in_a, b);
                    let mut delta = da + db;
                    if graph.adj_mat[a][b] {
                        delta += 2;
                    }
                    if delta < best_delta {
                        best_delta = delta;
                        best_pair = Some((a, b));
                    }
                }
            }
            if let Some((a, b)) = best_pair {
                in_a[a] = false;
                in_a[b] = true;
            } else {
                break;
            }
        }
    }

    in_a
}

fn consider_best(best: &mut Best, graph: &Graph, in_a: Vec<bool>) {
    if in_a.iter().filter(|&&x| x).count() != graph.n / 2 {
        return;
    }
    let cut = cut_of_partition(graph, &in_a);
    if cut < best.cut {
        best.cut = cut;
        best.in_a = in_a;
    }
}

fn evaluate_lambda(
    state: &State,
    unassigned: &[usize],
    residual_edges: &[(usize, usize)],
    rem_a: usize,
    lambda: i64,
) -> RelaxedSol {
    let free_n = unassigned.len();
    let src = free_n;
    let snk = free_n + 1;
    let mut dinic = Dinic::new(free_n + 2);
    let mut base = 0i64;

    for (i, &u) in unassigned.iter().enumerate() {
        let e_a = state.to_b[u] as i64 + lambda;
        let e_b = state.to_a[u] as i64;
        match e_a.cmp(&e_b) {
            Ordering::Less | Ordering::Equal => {
                base += e_a;
                let diff = e_b - e_a;
                if diff > 0 {
                    dinic.add_edge(src, i, diff);
                }
            }
            Ordering::Greater => {
                base += e_b;
                let diff = e_a - e_b;
                if diff > 0 {
                    dinic.add_edge(i, snk, diff);
                }
            }
        }
    }

    for &(a, b) in residual_edges {
        dinic.add_undirected_cut_edge(a, b, 1);
    }

    let flow = dinic.max_flow(src, snk);
    let reach = dinic.source_side(src);
    let mut in_a_local = vec![false; free_n];
    let mut count_a = 0usize;
    for i in 0..free_n {
        if reach[i] {
            in_a_local[i] = true;
            count_a += 1;
        }
    }

    RelaxedSol {
        lambda,
        cost: state.fixed_cut + base + flow - lambda * rem_a as i64,
        count_a,
        unassigned: unassigned.to_vec(),
        in_a_local,
    }
}

fn compute_bound(graph: &Graph, state: &State, target_a_total: usize) -> BoundInfo {
    let rem_a = target_a_total - state.a_count;
    let mut unassigned = Vec::new();
    for u in 0..graph.n {
        if state.side[u] == 0 {
            unassigned.push(u);
        }
    }

    if unassigned.is_empty() {
        return BoundInfo {
            lb: state.fixed_cut,
            exact: Some(RelaxedSol {
                lambda: 0,
                cost: state.fixed_cut,
                count_a: 0,
                unassigned,
                in_a_local: Vec::new(),
            }),
            over: None,
            under: None,
            guide: None,
        };
    }

    let mut pos = vec![usize::MAX; graph.n];
    for (i, &u) in unassigned.iter().enumerate() {
        pos[u] = i;
    }
    let mut residual_edges = Vec::new();
    for &u in &unassigned {
        for &v in &graph.adj[u] {
            if u < v && state.side[v] == 0 {
                residual_edges.push((pos[u], pos[v]));
            }
        }
    }

    let limit = 2 * graph.m as i64 + graph.n as i64 + 5;
    let mut samples = Vec::new();
    let mut best_lb = i64::MIN;
    let mut exact = None;

    for &lambda in &[-limit, 0, limit] {
        let sol = evaluate_lambda(state, &unassigned, &residual_edges, rem_a, lambda);
        best_lb = best_lb.max(sol.cost);
        if sol.count_a == rem_a && exact.is_none() {
            exact = Some(sol.clone());
        }
        samples.push(sol);
    }

    if exact.is_none() {
        let low = samples[0].clone();
        let high = samples[2].clone();
        if low.count_a >= rem_a && high.count_a <= rem_a {
            let mut l = low.lambda;
            let mut r = high.lambda;
            for _ in 0..32 {
                if r - l <= 1 {
                    break;
                }
                let mid = l + (r - l) / 2;
                let sol = evaluate_lambda(state, &unassigned, &residual_edges, rem_a, mid);
                best_lb = best_lb.max(sol.cost);
                if sol.count_a == rem_a {
                    exact = Some(sol.clone());
                    samples.push(sol);
                    break;
                }
                if sol.count_a > rem_a {
                    l = mid;
                } else {
                    r = mid;
                }
                samples.push(sol);
            }
        }
    }

    let mut over = None;
    let mut under = None;
    let mut guide = None;

    for sol in samples {
        let diff = sol.count_a.abs_diff(rem_a);
        if guide
            .as_ref()
            .map_or(true, |g: &RelaxedSol| diff < g.count_a.abs_diff(rem_a) || (diff == g.count_a.abs_diff(rem_a) && sol.cost < g.cost))
        {
            guide = Some(sol.clone());
        }
        if sol.count_a >= rem_a
            && over.as_ref().map_or(true, |g: &RelaxedSol| {
                sol.count_a < g.count_a || (sol.count_a == g.count_a && sol.cost < g.cost)
            })
        {
            over = Some(sol.clone());
        }
        if sol.count_a <= rem_a
            && under.as_ref().map_or(true, |g: &RelaxedSol| {
                sol.count_a > g.count_a || (sol.count_a == g.count_a && sol.cost < g.cost)
            })
        {
            under = Some(sol);
        }
    }

    BoundInfo {
        lb: best_lb,
        exact,
        over,
        under,
        guide,
    }
}

fn choose_branch_vertex(graph: &Graph, state: &State, bound: &BoundInfo) -> usize {
    if let (Some(over), Some(under)) = (&bound.over, &bound.under) {
        let mut best_v = None;
        let mut best_score = i64::MIN;
        for i in 0..over.unassigned.len() {
            if over.in_a_local[i] != under.in_a_local[i] {
                let u = over.unassigned[i];
                let score = 4 * graph.degree[u] as i64
                    + 3 * state.free_deg[u] as i64
                    + (state.to_a[u] - state.to_b[u]).abs() as i64;
                if score > best_score {
                    best_score = score;
                    best_v = Some(u);
                }
            }
        }
        if let Some(v) = best_v {
            return v;
        }
    }

    let mut best_v = None;
    let mut best_score = i64::MIN;
    for u in 0..graph.n {
        if state.side[u] != 0 {
            continue;
        }
        let score =
            4 * graph.degree[u] as i64 + 3 * state.free_deg[u] as i64 + (state.to_a[u] - state.to_b[u]).abs() as i64;
        if score > best_score {
            best_score = score;
            best_v = Some(u);
        }
    }
    best_v.unwrap()
}

fn dfs(graph: &Graph, state: State, target_a_total: usize, best: &mut Best, depth: usize) {
    let unassigned_count = graph.n - state.a_count - state.b_count;
    if state.a_count > target_a_total || state.b_count > target_a_total {
        return;
    }
    if state.a_count + unassigned_count < target_a_total || state.b_count + unassigned_count < target_a_total {
        return;
    }

    if unassigned_count == 0 {
        let in_a = full_assignment_from_state(&state, None);
        consider_best(best, graph, in_a);
        return;
    }

    if state.a_count == target_a_total {
        let in_a = full_assignment_from_state(&state, Some(false));
        consider_best(best, graph, in_a);
        return;
    }

    if state.b_count == target_a_total {
        let in_a = full_assignment_from_state(&state, Some(true));
        consider_best(best, graph, in_a);
        return;
    }

    let bound = compute_bound(graph, &state, target_a_total);
    if bound.lb >= best.cut {
        return;
    }

    if let Some(sol) = &bound.exact {
        let in_a = full_assignment_from_relaxed(&state, sol);
        consider_best(best, graph, in_a);
        return;
    }

    let rem_total = graph.n - state.a_count - state.b_count;
    let do_local_search = depth <= 3 || rem_total <= 18;
    let candidate = build_candidate(graph, &state, target_a_total, bound.guide.as_ref(), do_local_search);
    consider_best(best, graph, candidate.clone());

    let v = choose_branch_vertex(graph, &state, &bound);
    let first_side_is_a = candidate[v];

    let first_side = if first_side_is_a { 1 } else { -1 };
    let second_side = -first_side;

    dfs(graph, assign_vertex(&state, graph, v, first_side), target_a_total, best, depth + 1);
    dfs(
        graph,
        assign_vertex(&state, graph, v, second_side),
        target_a_total,
        best,
        depth + 1,
    );
}

fn solve(graph: &Graph) -> Best {
    if graph.n == 0 {
        return Best {
            cut: 0,
            in_a: Vec::new(),
        };
    }

    let target_a_total = graph.n / 2;
    let init = initial_state(graph);
    let seed = build_candidate(graph, &init, target_a_total, None, true);
    let mut best = Best {
        cut: cut_of_partition(graph, &seed),
        in_a: seed,
    };
    dfs(graph, init, target_a_total, &mut best, 0);
    best
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let graph = build_graph(&input);
    let best = solve(&graph);

    println!("{}", best.cut);
    let mut first = true;
    for u in 0..graph.n {
        if best.in_a[u] {
            if !first {
                print!(" ");
            }
            first = false;
            print!("{}", u);
        }
    }
    println!();
}
