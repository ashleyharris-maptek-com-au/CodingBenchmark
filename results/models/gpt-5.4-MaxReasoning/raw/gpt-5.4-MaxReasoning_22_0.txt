using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private static readonly Dictionary<char, RotationShape[]> Pieces = BuildPieces();
    private static readonly char[] NextPieceTypes = { 'I', 'O', 'T', 'S', 'Z', 'J', 'L' };

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        string widthToken = scanner.Next();
        if (widthToken == null)
        {
            return;
        }

        int width = int.Parse(widthToken);
        int height = int.Parse(scanner.Next() ?? "0");
        int totalPieces = int.Parse(scanner.Next() ?? "0");

        var board = new BoardState(width, height);

        for (int pieceIndex = 0; pieceIndex < totalPieces; pieceIndex++)
        {
            string pieceToken = scanner.Next();
            if (string.IsNullOrEmpty(pieceToken))
            {
                break;
            }

            char pieceType = char.ToUpperInvariant(pieceToken[0]);
            Move bestMove = ChooseMove(board, pieceType, totalPieces - pieceIndex);

            Console.WriteLine(bestMove.Rotation + " " + bestMove.Column);
            Console.Out.Flush();

            string result = scanner.Next();
            if (result == null)
            {
                break;
            }

            if (string.Equals(result, "gameover", StringComparison.OrdinalIgnoreCase))
            {
                break;
            }

            board.Apply(bestMove.Shape, bestMove.Column);
        }
    }

    private static Move ChooseMove(BoardState board, char pieceType, int remainingPieces)
    {
        BoardMetrics current = Analyze(board.Rows, board.Heights, board.Width, board.Height, true);
        StrategyProfile profile = ChooseProfile(board, current, remainingPieces);

        int[] scratchRows = new int[board.Height];
        int[] scratchHeights = new int[board.Width];
        int[] replyRows = profile.UseLookahead ? new int[board.Height] : Array.Empty<int>();
        int[] replyHeights = profile.UseLookahead ? new int[board.Width] : Array.Empty<int>();

        double bestScore = double.NegativeInfinity;
        RotationShape bestShape = Pieces[pieceType][0];
        int bestColumn = 0;

        RotationShape[] rotations = Pieces[pieceType];
        for (int i = 0; i < rotations.Length; i++)
        {
            RotationShape shape = rotations[i];
            int lastColumn = board.Width - shape.Width;
            for (int column = 0; column <= lastColumn; column++)
            {
                SimulationResult sim;
                if (!TrySimulate(board.Width, board.Height, board.FullMask, board.Rows, board.Heights, shape, column, scratchRows, scratchHeights, profile.UseDetailed, out sim))
                {
                    continue;
                }

                double score = ScoreMove(board, pieceType, profile, sim);
                if (profile.UseLookahead)
                {
                    score += profile.FutureWeight * EvaluateExpectedReply(board.Width, board.Height, board.FullMask, scratchRows, scratchHeights, profile, replyRows, replyHeights);
                }

                score -= shape.Rotation * 0.0001;
                score -= column * 0.00001;

                if (score > bestScore)
                {
                    bestScore = score;
                    bestShape = shape;
                    bestColumn = column;
                }
            }
        }

        return new Move(bestShape.Rotation, bestColumn, bestShape);
    }

    private static StrategyProfile ChooseProfile(BoardState board, BoardMetrics current, int remainingPieces)

        bool endgame = remainingPieces <= Math.Max(board.Width, 16);
        bool wideBoard = board.Width >= 14;
        bool marathon = remainingPieces >= 200;

                RowTransitionsWeight = -0.20,
                ColumnTransitionsWeight = -0.18,
                NearCompleteRowsWeight = 0.45,
                LandingHeightWeight = -0.06,
                RightWellReward = 0.0,
                RightWellPenalty = 0.0,
                TetrisBonus = 1.5,
                PerfectClearBonus = 25.0
            };
        }

        if (endgame)
        {
            return new StrategyProfile
            {
                UseDetailed = true,
                UseLookahead = true,
                KeepRightWell = false,
                FutureWeight = 0.32,
                ClearReward = new[] { 0.0, 2.2, 6.0, 10.5, 15.5 },
                AggregateHeightWeight = -0.30,
                MaxHeightWeight = -0.72,
                HolesWeight = -5.1,
                CoveredHolesWeight = -1.10,
                BumpinessWeight = -0.18,
                WellWeight = -0.12,
                RowTransitionsWeight = -0.13,
                ColumnTransitionsWeight = -0.12,
                NearCompleteRowsWeight = 0.55,
                LandingHeightWeight = -0.03,
                RightWellReward = 0.0,
                RightWellPenalty = 0.0,
                TetrisBonus = 4.0,
                PerfectClearBonus = 40.0
            };
        }

        if (complex)
        {
            return new StrategyProfile
            {
                UseDetailed = false,
                UseLookahead = false,
                KeepRightWell = board.Width >= 8 && current.MaxHeight * 10 < board.Height * 6,
                FutureWeight = 0.0,
                ClearReward = new[] { 0.0, 1.0, 3.2, 6.2, 10.8 },
                AggregateHeightWeight = -0.28,
                MaxHeightWeight = -0.46,
                HolesWeight = -4.3,
                CoveredHolesWeight = -0.80,
                BumpinessWeight = -0.17,

                RowTransitionsWeight = 0.0,

                NearCompleteRowsWeight = 0.25,

                RightWellReward = 0.30,

            };

        return new StrategyProfile
        {
            UseDetailed = true,
            UseLookahead = useLookahead,
            KeepRightWell = board.Width >= 8 && current.MaxHeight * 10 < board.Height * 6,
            FutureWeight = 0.38,
            ClearReward = new[] { 0.0, 1.4, 4.1, 7.8, 12.8 },
            AggregateHeightWeight = -0.31,
            MaxHeightWeight = -0.60,
            HolesWeight = -4.8,
            CoveredHolesWeight = -1.00,
            BumpinessWeight = -0.19,
            WellWeight = -0.11,
            RowTransitionsWeight = -0.14,
            ColumnTransitionsWeight = -0.13,
            NearCompleteRowsWeight = 0.38,
            LandingHeightWeight = -0.03,

    private static double EvaluateExpectedReply(
        int width,
        int height,
        int fullMask,
        int[] sourceRows,
        int[] sourceHeights,
        StrategyProfile profile,
        int[] scratchRows,
        int[] scratchHeights)
    {
        double total = 0.0;
        for (int i = 0; i < NextPieceTypes.Length; i++)
        {
            char nextPiece = NextPieceTypes[i];
            double best = double.NegativeInfinity;
            RotationShape[] rotations = Pieces[nextPiece];

            for (int r = 0; r < rotations.Length; r++)
            {
                RotationShape shape = rotations[r];
                int lastColumn = width - shape.Width;
                for (int column = 0; column <= lastColumn; column++)
                {
                    SimulationResult sim;
                    if (!TrySimulate(width, height, fullMask, sourceRows, sourceHeights, shape, column, scratchRows, scratchHeights, profile.UseDetailed, out sim))
                    {
                        continue;
                    }

                    double score = ScoreMove(width, height, nextPiece, profile, sim);
                    if (score > best)
                    {
                        best = score;
                    }
                }
            }

            if (double.IsNegativeInfinity(best))
            {
                best = -10000.0;
            }

            total += best;
        }

        return total / NextPieceTypes.Length;
    }

    private static double ScoreMove(BoardState board, char pieceType, StrategyProfile profile, SimulationResult sim)
    {
        return ScoreMove(board.Width, board.Height, pieceType, profile, sim);
    }

    private static double ScoreMove(int width, int height, char pieceType, StrategyProfile profile, SimulationResult sim)
    {
        double score = profile.ClearReward[sim.ClearedLines];
        score += profile.AggregateHeightWeight * sim.Metrics.AggregateHeight;
        score += profile.MaxHeightWeight * sim.Metrics.MaxHeight;
        score += profile.HolesWeight * sim.Metrics.Holes;
        score += profile.CoveredHolesWeight * sim.Metrics.CoveredHoles;
        score += profile.BumpinessWeight * sim.Metrics.Bumpiness;
        score += profile.WellWeight * sim.Metrics.WellSum;
        score += profile.NearCompleteRowsWeight * sim.Metrics.NearCompleteRows;
        score += profile.LandingHeightWeight * sim.LandingHeight;

        if (profile.UseDetailed)
        {
            score += profile.RowTransitionsWeight * sim.Metrics.RowTransitions;
            score += profile.ColumnTransitionsWeight * sim.Metrics.ColumnTransitions;
        }

        if (profile.KeepRightWell && width >= 2)
        {
            int safeDepth = sim.Metrics.MaxHeight * 10 < height * 6 ? Math.Min(sim.Metrics.RightWellDepth, 4) : 0;
            score += profile.RightWellReward * safeDepth;
            if (pieceType != 'I')
            {
                score += profile.RightWellPenalty * sim.CellsPlacedInRightmostColumn;
            }
        }

        if (pieceType == 'I' && sim.ClearedLines == 4)
        {
            score += profile.TetrisBonus;
        }

        if (sim.PerfectClear)
        {
            score += profile.PerfectClearBonus;
        }

        return score;
    }

    private static bool TrySimulate(
        int width,
        int height,
        int fullMask,
        int[] sourceRows,
        int[] sourceHeights,
        RotationShape shape,
        int column,
        int[] targetRows,
        int[] targetHeights,
        bool detailed,
        out SimulationResult sim)
    {
        sim = default;
        int placementRow = GetPlacementRow(width, height, sourceHeights, shape, column);
        if (placementRow < 0 || placementRow + shape.Height > height)
        {
            return false;
        }

        Array.Copy(sourceRows, targetRows, height);
        int rightColumn = width - 1;
        int placedInRightmostColumn = 0;

        for (int i = 0; i < shape.Cells.Length; i++)
        {
            Cell cell = shape.Cells[i];
            int boardX = column + cell.X;
            int boardY = placementRow + cell.Y;
            int bit = 1 << boardX;

            if ((targetRows[boardY] & bit) != 0)
            {
                return false;
            }

            targetRows[boardY] |= bit;
            if (boardX == rightColumn)
            {
                placedInRightmostColumn++;
            }
        }

        int cleared = 0;
        int writeRow = 0;
        for (int readRow = 0; readRow < height; readRow++)
        {
            int rowBits = targetRows[readRow];
            if (rowBits == fullMask)
            {
                cleared++;
                continue;
            }

            targetRows[writeRow++] = rowBits;
        }

        while (writeRow < height)
        {
            targetRows[writeRow++] = 0;
        }

        RecomputeHeights(targetRows, targetHeights, width, height);
        BoardMetrics metrics = Analyze(targetRows, targetHeights, width, height, detailed);

        sim = new SimulationResult
        {
            Metrics = metrics,
            ClearedLines = cleared,
            LandingHeight = placementRow + shape.Height,
            CellsPlacedInRightmostColumn = placedInRightmostColumn,
            PerfectClear = metrics.FilledCells == 0
        };

        return true;
    }

    private static int GetPlacementRow(int width, int height, int[] heights, RotationShape shape, int column)
    {
        if (column < 0 || column + shape.Width > width)
        {
            return -1;
        }

        int row = 0;
        for (int x = 0; x < shape.Width; x++)
        {
            int candidate = heights[column + x] - shape.Bottom[x];
            if (candidate > row)
            {
                row = candidate;
            }
        }

        if (row + shape.Height > height)
        {
            return -1;
        }

        return row;
    }

    private static void RecomputeHeights(int[] rows, int[] heights, int width, int height)
    {
        Array.Clear(heights, 0, width);
        for (int y = 0; y < height; y++)
        {
            int rowBits = rows[y];
            if (rowBits == 0)
            {
                continue;
            }

            for (int x = 0; x < width; x++)
            {
                if ((rowBits & (1 << x)) != 0)
                {
                    heights[x] = y + 1;
                }
            }
        }
    }

    private static BoardMetrics Analyze(int[] rows, int[] heights, int width, int height, bool detailed)
    {
        BoardMetrics metrics = default;

        for (int x = 0; x < width; x++)
        {
            int columnHeight = heights[x];
            metrics.AggregateHeight += columnHeight;
            if (columnHeight > metrics.MaxHeight)
            {
                metrics.MaxHeight = columnHeight;
            }

            int filledAbove = 0;
            for (int y = columnHeight - 1; y >= 0; y--)
            {
                if ((rows[y] & (1 << x)) != 0)
                {
                    filledAbove++;
                }
                else
                {
                    metrics.Holes++;
                    metrics.CoveredHoles += filledAbove;
                }
            }

            if (x + 1 < width)
            {
                metrics.Bumpiness += Math.Abs(columnHeight - heights[x + 1]);
            }

            int left = x == 0 ? height : heights[x - 1];
            int right = x == width - 1 ? height : heights[x + 1];
            int limit = Math.Min(left, right);
            if (limit > columnHeight)
            {
                metrics.WellSum += limit - columnHeight;
            }
        }

        if (width >= 2)
        {
            metrics.RightWellDepth = Math.Max(0, heights[width - 2] - heights[width - 1]);
        }

        for (int y = 0; y < height; y++)
        {
            int rowBits = rows[y];
            int filled = PopCount(rowBits);
            metrics.FilledCells += filled;
            if (filled == width - 1)
            {
                metrics.NearCompleteRows++;
            }
        }

        if (!detailed)
        {
            return metrics;
        }

        int analysisHeight = Math.Max(metrics.MaxHeight, 1);
        for (int y = 0; y < analysisHeight; y++)
        {
            bool previousFilled = true;
            for (int x = 0; x < width; x++)
            {
                bool filled = (rows[y] & (1 << x)) != 0;
                if (filled != previousFilled)
                {
                    metrics.RowTransitions++;
                }

                previousFilled = filled;
            }

            if (!previousFilled)
            {
                metrics.RowTransitions++;
            }
        }

        for (int x = 0; x < width; x++)
        {
            bool previousFilled = true;
            for (int y = 0; y < analysisHeight; y++)
            {
                bool filled = (rows[y] & (1 << x)) != 0;
                if (filled != previousFilled)
                {
                    metrics.ColumnTransitions++;
                }

                previousFilled = filled;
            }

            if (!previousFilled)
            {
                metrics.ColumnTransitions++;
            }
        }

        return metrics;
    }

    private static int PopCount(int value)
    {
        uint x = unchecked((uint)value);
        x -= (x >> 1) & 0x55555555u;
        x = (x & 0x33333333u) + ((x >> 2) & 0x33333333u);
        x = (x + (x >> 4)) & 0x0F0F0F0Fu;
        return (int)((x * 0x01010101u) >> 24);
    }

    private static Dictionary<char, RotationShape[]> BuildPieces()
    {
        return new Dictionary<char, RotationShape[]>
        {
            ['I'] = BuildRotations(new[] { new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(3, 0) }),
            ['O'] = BuildRotations(new[] { new Cell(0, 0), new Cell(1, 0), new Cell(0, 1), new Cell(1, 1) }),
            ['T'] = BuildRotations(new[] { new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(1, 1) }),
            ['S'] = BuildRotations(new[] { new Cell(0, 0), new Cell(1, 0), new Cell(1, 1), new Cell(2, 1) }),
            ['Z'] = BuildRotations(new[] { new Cell(1, 0), new Cell(2, 0), new Cell(0, 1), new Cell(1, 1) }),
            ['J'] = BuildRotations(new[] { new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(0, 1) }),
            ['L'] = BuildRotations(new[] { new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(2, 1) })
        };
    }

    private static RotationShape[] BuildRotations(Cell[] baseCells)
    {
        List<RotationShape> rotations = new List<RotationShape>();
        HashSet<string> seen = new HashSet<string>(StringComparer.Ordinal);
        Cell[] current = Normalize(baseCells, out int width, out int height);

        for (int rotation = 0; rotation < 4; rotation++)
        {
            Cell[] normalized = Normalize(current, out width, out height);
            string key = GetShapeKey(normalized);
            if (seen.Add(key))
            {
                rotations.Add(CreateRotationShape(rotation, normalized, width, height));
            }

            current = RotateClockwise(normalized, width, height);
        }

        return rotations.ToArray();
    }

    private static RotationShape CreateRotationShape(int rotation, Cell[] cells, int width, int height)
    {
        int[] bottom = new int[width];
        for (int i = 0; i < width; i++)
        {
            bottom[i] = int.MaxValue;
        }

        for (int i = 0; i < cells.Length; i++)
        {
            Cell cell = cells[i];
            if (cell.Y < bottom[cell.X])
            {
                bottom[cell.X] = cell.Y;
            }
        }

        return new RotationShape(rotation, width, height, cells, bottom);
    }

    private static Cell[] Normalize(Cell[] cells, out int width, out int height)
    {
        int minX = int.MaxValue;
        int minY = int.MaxValue;
        int maxX = int.MinValue;
        int maxY = int.MinValue;

        for (int i = 0; i < cells.Length; i++)
        {
            Cell cell = cells[i];
            if (cell.X < minX) minX = cell.X;
            if (cell.Y < minY) minY = cell.Y;
            if (cell.X > maxX) maxX = cell.X;
            if (cell.Y > maxY) maxY = cell.Y;
        }

        Cell[] normalized = new Cell[cells.Length];
        for (int i = 0; i < cells.Length; i++)
        {
            normalized[i] = new Cell(cells[i].X - minX, cells[i].Y - minY);
        }

        Array.Sort(normalized, CompareCells);
        width = maxX - minX + 1;
        height = maxY - minY + 1;
        return normalized;
    }

    private static int CompareCells(Cell a, Cell b)
    {
        int byX = a.X.CompareTo(b.X);
        return byX != 0 ? byX : a.Y.CompareTo(b.Y);
    }

    private static string GetShapeKey(Cell[] cells)
    {
        StringBuilder builder = new StringBuilder(cells.Length * 4);
        for (int i = 0; i < cells.Length; i++)
        {
            if (i != 0)
            {
                builder.Append(';');
            }

            builder.Append(cells[i].X);
            builder.Append(',');
            builder.Append(cells[i].Y);
        }

        return builder.ToString();
    }

    private static Cell[] RotateClockwise(Cell[] cells, int width, int height)
    {
        Cell[] rotated = new Cell[cells.Length];
        for (int i = 0; i < cells.Length; i++)
        {
            Cell cell = cells[i];
            rotated[i] = new Cell(height - 1 - cell.Y, cell.X);
        }

        return rotated;
    }

    private struct Cell
    {
        public readonly int X;
        public readonly int Y;

        public Cell(int x, int y)
        {
            X = x;
            Y = y;
        }
    }

    private struct Move
    {
        public readonly int Rotation;
        public readonly int Column;
        public readonly RotationShape Shape;

        public Move(int rotation, int column, RotationShape shape)
        {
            Rotation = rotation;
            Column = column;
            Shape = shape;
        }
    }

    private sealed class RotationShape
    {
        public RotationShape(int rotation, int width, int height, Cell[] cells, int[] bottom)
        {
            Rotation = rotation;
            Width = width;
            Height = height;
            Cells = cells;
            Bottom = bottom;
        }

        public int Rotation { get; }
        public int Width { get; }
        public int Height { get; }
        public Cell[] Cells { get; }
        public int[] Bottom { get; }
    }

    private sealed class BoardState
    {
        public BoardState(int width, int height)
        {
            Width = width;
            Height = height;
            FullMask = (1 << width) - 1;
            Rows = new int[height];
            Heights = new int[width];
        }

        public int Width { get; }
        public int Height { get; }
        public int FullMask { get; }
        public int[] Rows { get; }
        public int[] Heights { get; }

        public bool Apply(RotationShape shape, int column)
        {
            int row = GetPlacementRow(Width, Height, Heights, shape, column);
            if (row < 0)
            {
                return false;
            }

            for (int i = 0; i < shape.Cells.Length; i++)
            {
                Cell cell = shape.Cells[i];
                int boardX = column + cell.X;
                int boardY = row + cell.Y;
                Rows[boardY] |= 1 << boardX;
            }

            int writeRow = 0;
            for (int readRow = 0; readRow < Height; readRow++)
            {
                int rowBits = Rows[readRow];
                if (rowBits == FullMask)
                {
                    continue;
                }

                Rows[writeRow++] = rowBits;
            }

            while (writeRow < Height)
            {
                Rows[writeRow++] = 0;
            }

            RecomputeHeights(Rows, Heights, Width, Height);
            return true;
        }
    }

    private struct BoardMetrics
    {
        public int AggregateHeight;
        public int MaxHeight;
        public int Holes;
        public int CoveredHoles;
        public int Bumpiness;
        public int WellSum;
        public int RowTransitions;
        public int ColumnTransitions;
        public int NearCompleteRows;
        public int RightWellDepth;
        public int FilledCells;
    }

    private struct StrategyProfile
    {
        public bool UseDetailed;
        public bool UseLookahead;
        public bool KeepRightWell;
        public double FutureWeight;
        public double[] ClearReward;
        public double AggregateHeightWeight;
        public double MaxHeightWeight;
        public double HolesWeight;
        public double CoveredHolesWeight;
        public double BumpinessWeight;
        public double WellWeight;
        public double RowTransitionsWeight;
        public double ColumnTransitionsWeight;
        public double NearCompleteRowsWeight;
        public double LandingHeightWeight;
        public double RightWellReward;
        public double RightWellPenalty;
        public double TetrisBonus;
        public double PerfectClearBonus;
    }

    private struct SimulationResult
    {
        public BoardMetrics Metrics;
        public int ClearedLines;
        public int LandingHeight;
        public int CellsPlacedInRightmostColumn;
        public bool PerfectClear;
    }

    private sealed class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int length;
        private int index;

        public FastScanner(Stream stream)
        {
            this.stream = stream;
        }

        public string Next()
        {
            int c;
            do
            {
                c = Read();
                if (c < 0)
                {
                    return null;
                }
            } while (c <= 32);

            StringBuilder builder = new StringBuilder();
            while (c > 32)
            {
                builder.Append((char)c);
                c = Read();
            }

            return builder.ToString();
        }

        private int Read()
        {
            if (index >= length)
            {
                length = stream.Read(buffer, 0, buffer.Length);
                index = 0;
                if (length == 0)
                {
                    return -1;
                }
            }

            return buffer[index++];
        }
    }
