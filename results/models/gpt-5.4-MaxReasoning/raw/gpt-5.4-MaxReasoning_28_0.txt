use std::cmp::Reverse;
use std::collections::{BinaryHeap, VecDeque};
use std::io::{self, Read};

const NONE: usize = usize::MAX;
const EXACT_THRESHOLD: usize = 64;

struct Scanner {
    data: Vec<u8>,
    pos: usize,
}

impl Scanner {
    fn new() -> io::Result<Self> {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data)?;
        Ok(Self { data, pos: 0 })
    }

    fn next_usize(&mut self) -> usize {
        while self.pos < self.data.len() && self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        let mut value = 0usize;
        while self.pos < self.data.len() {
            let b = self.data[self.pos];
            if !b.is_ascii_digit() {
                break;
            }
            value = value * 10 + usize::from(b - b'0');
            self.pos += 1;
        }
        value
    }
}

struct Graph {
    offsets: Vec<usize>,
    neighbors: Vec<u32>,
}

impl Graph {
    fn neighbors(&self, u: usize) -> &[u32] {
        &self.neighbors[self.offsets[u]..self.offsets[u + 1]]
    }
}

fn full_mask(n: usize) -> u64 {
    if n == 64 {
        u64::MAX
    } else {
        (1u64 << n) - 1
    }
}

fn include_vertex(
    u: usize,
    graph: &Graph,
    active: &mut [u8],
    degree: &mut [u32],
    in_cover: &mut [u8],
    queue: &mut VecDeque<usize>,
) {
    if active[u] == 0 {
        return;
    }
    active[u] = 0;
    in_cover[u] = 1;
    for &v in graph.neighbors(u) {
        let v = v as usize;
        if active[v] != 0 {
            degree[v] -= 1;
            if degree[v] <= 1 {
                queue.push_back(v);
            }
        }
    }
}

fn discard_vertex(
    u: usize,
    graph: &Graph,
    active: &mut [u8],
    degree: &mut [u32],
    queue: &mut VecDeque<usize>,
) {
    if active[u] == 0 {
        return;
    }
    active[u] = 0;
    for &v in graph.neighbors(u) {
        let v = v as usize;
        if active[v] != 0 {
            degree[v] -= 1;
            if degree[v] <= 1 {
                queue.push_back(v);
            }
        }
    }
}

fn bit(v: usize) -> u64 {
    1u64 << v
}

fn has_edges(rem: u64, adj: &[u64]) -> bool {
    let mut x = rem;
    while x != 0 {
        let u = x.trailing_zeros() as usize;
        x &= x - 1;
        if (adj[u] & rem) != 0 {
            return true;
        }
    }
    false
}

fn matching_lower_bound(rem: u64, adj: &[u64]) -> u32 {
    let mut remaining = rem;
    let mut matched = 0u32;
    while remaining != 0 {
        let u = remaining.trailing_zeros() as usize;
        let ub = bit(u);
        remaining &= !ub;
        let nbrs = adj[u] & remaining;
        if nbrs != 0 {
            let v = nbrs.trailing_zeros() as usize;
            remaining &= !bit(v);
            matched += 1;
        }
    }
    matched
}

fn greedy_upper_bound(adj: &[u64], n: usize) -> u64 {
    let mut rem = full_mask(n);
    let mut cover = 0u64;
    loop {
        let mut best_u = NONE;
        let mut best_deg = 0u32;
        let mut x = rem;
        while x != 0 {
            let u = x.trailing_zeros() as usize;
            x &= x - 1;
            let deg = (adj[u] & rem).count_ones();
            if deg > best_deg {
                best_deg = deg;
                best_u = u;
            }
        }
        if best_deg == 0 {
            break;
        }
        let ub = bit(best_u);
        cover |= ub;
        rem &= !ub;
    }
    cover
}

fn prune_cover_bitset(adj: &[u64], n: usize, mut cover: u64) -> u64 {
    let all = full_mask(n);
    let mut i = 0usize;
    while i < n {
        let ib = bit(i);
        if (cover & ib) != 0 && (adj[i] & (all ^ cover)) == 0 {
            cover &= !ib;
        }
        i += 1;
    }
    cover
}

fn select_branch_vertex(rem: u64, adj: &[u64]) -> usize {
    let mut best_u = 0usize;
    let mut best_deg = 0u32;
    let mut x = rem;
    while x != 0 {
        let u = x.trailing_zeros() as usize;
        x &= x - 1;
        let deg = (adj[u] & rem).count_ones();
        if deg > best_deg {
            best_deg = deg;
            best_u = u;
        }
    }
    best_u
}

fn exact_cover_dfs(
    adj: &[u64],
    mut rem: u64,
    mut current: u64,
    mut current_count: u32,
    best_mask: &mut u64,
    best_count: &mut u32,
) {
    loop {
        let mut changed = false;
        let mut x = rem;
        while x != 0 {
            let u = x.trailing_zeros() as usize;
            x &= x - 1;
            let nbrs = adj[u] & rem;
            let deg = nbrs.count_ones();
            if deg == 0 {
                rem &= !bit(u);
                changed = true;
            } else if deg == 1 {
                let v = nbrs.trailing_zeros() as usize;
                let vb = bit(v);
                current |= vb;
                current_count += 1;
                if current_count >= *best_count {
                    return;
                }
                rem &= !vb;
                changed = true;
                break;
            }
        }
        if !changed {
            break;
        }
    }

    if !has_edges(rem, adj) {
        if current_count < *best_count {
            *best_count = current_count;
            *best_mask = current;
        }
        return;
    }

    let lower_bound = matching_lower_bound(rem, adj);
    if current_count + lower_bound >= *best_count {
        return;
    }

    let u = select_branch_vertex(rem, adj);
    let ub = bit(u);
    exact_cover_dfs(
        adj,
        rem & !ub,
        current | ub,
        current_count + 1,
        best_mask,
        best_count,
    );

    let nbrs = adj[u] & rem;
    let add_count = nbrs.count_ones();
    if current_count + add_count < *best_count {
        exact_cover_dfs(
            adj,
            rem & !nbrs,
            current | nbrs,
            current_count + add_count,
            best_mask,
            best_count,
        );
    }
}

fn exact_small_component(graph: &Graph, comp: &[usize], map: &mut [usize]) -> Vec<usize> {
    for (i, &u) in comp.iter().enumerate() {
        map[u] = i;
    }

    let mut adj = vec![0u64; comp.len()];
    for (i, &u) in comp.iter().enumerate() {
        let mut mask = 0u64;
        for &v in graph.neighbors(u) {
            let j = map[v as usize];
            if j != NONE {
                mask |= bit(j);
            }
        }
        adj[i] = mask;
    }

    let mut best_mask = prune_cover_bitset(&adj, comp.len(), greedy_upper_bound(&adj, comp.len()));
    let mut best_count = best_mask.count_ones();
    exact_cover_dfs(
        &adj,
        full_mask(comp.len()),
        0,
        0,
        &mut best_mask,
        &mut best_count,
    );

    let mut result = Vec::with_capacity(best_count as usize);
    for (i, &u) in comp.iter().enumerate() {
        if (best_mask & bit(i)) != 0 {
            result.push(u);
        }
        map[u] = NONE;
    }
    result
}

fn hopcroft_karp_dfs(
    start: usize,
    offsets: &[usize],
    adj: &[usize],
    pair_u: &mut [usize],
    pair_v: &mut [usize],
    dist: &mut [u32],
) -> bool {
    let mut stack_u = Vec::new();
    let mut stack_idx = Vec::new();
    let mut chosen_v = Vec::new();

    stack_u.push(start);
    stack_idx.push(offsets[start]);

    while let Some(&u) = stack_u.last() {
        let end = offsets[u + 1];
        let idx_pos = stack_idx.len() - 1;
        let mut advanced = false;

        while stack_idx[idx_pos] < end {
            let v = adj[stack_idx[idx_pos]];
            stack_idx[idx_pos] += 1;
            let next_u = pair_v[v];
            if next_u == NONE {
                pair_u[u] = v;
                pair_v[v] = u;
                for depth in (0..chosen_v.len()).rev() {
                    let parent_u = stack_u[depth];
                    let chosen = chosen_v[depth];
                    pair_u[parent_u] = chosen;
                    pair_v[chosen] = parent_u;
                }
                return true;
            }
            if dist[next_u] == dist[u] + 1 {
                chosen_v.push(v);
                stack_u.push(next_u);
                stack_idx.push(offsets[next_u]);
                advanced = true;
                break;
            }
        }

        if !advanced {
            dist[u] = u32::MAX;
            stack_u.pop();
            stack_idx.pop();
            if !chosen_v.is_empty() && chosen_v.len() == stack_u.len() {
                chosen_v.pop();
            }
        }
    }

    false
}

fn exact_bipartite_component(
    graph: &Graph,
    comp: &[usize],
    colors: &[u8],
    map: &mut [usize],
) -> Vec<usize> {
    let mut count_zero = 0usize;
    for &u in comp {
        if colors[u] == 0 {
            count_zero += 1;
        }
    }
    let left_color = if count_zero * 2 <= comp.len() { 0 } else { 1 };

    let mut left = Vec::new();
    let mut right = Vec::new();
    for &u in comp {
        if colors[u] == left_color {
            map[u] = left.len();
            left.push(u);
        } else {
            map[u] = right.len();
            right.push(u);
        }
    }

    let mut left_degree = vec![0usize; left.len()];
    for &u in &left {
        let lu = map[u];
        for &v in graph.neighbors(u) {
            let v = v as usize;
            if colors[v] != left_color {
                left_degree[lu] += 1;
            }
        }
    }

    let mut offsets = vec![0usize; left.len() + 1];
    for i in 0..left.len() {
        offsets[i + 1] = offsets[i] + left_degree[i];
    }
    let mut next = offsets[..left.len()].to_vec();
    let mut adj = vec![0usize; offsets[left.len()]];
    for &u in &left {
        let lu = map[u];
        for &v in graph.neighbors(u) {
            let v = v as usize;
            if colors[v] != left_color {
                let rv = map[v];
                adj[next[lu]] = rv;
                next[lu] += 1;
            }
        }
    }

    let mut pair_u = vec![NONE; left.len()];
    let mut pair_v = vec![NONE; right.len()];
    let mut dist = vec![u32::MAX; left.len()];
    let mut queue = VecDeque::new();

    loop {
        queue.clear();
        for u in 0..left.len() {
            if pair_u[u] == NONE {
                dist[u] = 0;
                queue.push_back(u);
            } else {
                dist[u] = u32::MAX;
            }
        }

        while let Some(u) = queue.pop_front() {
            let next_dist = dist[u] + 1;
            for idx in offsets[u]..offsets[u + 1] {
                let v = adj[idx];
                let mate = pair_v[v];
                if mate != NONE && dist[mate] == u32::MAX {
                    dist[mate] = next_dist;
                    queue.push_back(mate);
                }
            }
        }

        let mut augmented = 0usize;
        for u in 0..left.len() {
            if pair_u[u] == NONE
                && hopcroft_karp_dfs(u, &offsets, &adj, &mut pair_u, &mut pair_v, &mut dist)
            {
                augmented += 1;
            }
        }
        if augmented == 0 {
            break;
        }
    }

    let mut visit_left = vec![0u8; left.len()];
    let mut visit_right = vec![0u8; right.len()];
    queue.clear();
    for u in 0..left.len() {
        if pair_u[u] == NONE {
            visit_left[u] = 1;
            queue.push_back(u);
        }
    }

    while let Some(u) = queue.pop_front() {
        for idx in offsets[u]..offsets[u + 1] {
            let v = adj[idx];
            if pair_u[u] != v && visit_right[v] == 0 {
                visit_right[v] = 1;
                let mate = pair_v[v];
                if mate != NONE && visit_left[mate] == 0 {
                    visit_left[mate] = 1;
                    queue.push_back(mate);
                }
            }
        }
    }

    let mut result = Vec::new();
    for i in 0..left.len() {
        if visit_left[i] == 0 {
            result.push(left[i]);
        }
    }
    for i in 0..right.len() {
        if visit_right[i] != 0 {
            result.push(right[i]);
        }
    }

    for &u in comp {
        map[u] = NONE;
    }
    result
}

fn prune_cover_local(graph: &Graph, comp: &[usize], map: &[usize], cover: &mut [u8]) {
    for &u in comp {
        let i = map[u];
        if cover[i] == 0 {
            continue;
        }
        let mut removable = true;
        for &v in graph.neighbors(u) {
            let j = map[v as usize];
            if j != NONE && cover[j] == 0 {
                removable = false;
                break;
            }
        }
        if removable {
            cover[i] = 0;
        }
    }
}

fn maximal_matching_cover(graph: &Graph, comp: &[usize], map: &[usize]) -> Vec<u8> {
    let mut matched = vec![0u8; comp.len()];
    let mut cover = vec![0u8; comp.len()];
    for &u in comp {
        let iu = map[u];
        if matched[iu] != 0 {
            continue;
        }
        for &v in graph.neighbors(u) {
            let iv = map[v as usize];
            if iv != NONE && matched[iv] == 0 {
                matched[iu] = 1;
                matched[iv] = 1;
                cover[iu] = 1;
                cover[iv] = 1;
                break;
            }
        }
    }
    cover
}

fn greedy_independent_set_cover(
    graph: &Graph,
    comp: &[usize],
    degree: &[u32],
    map: &[usize],
) -> Vec<u8> {
    let mut current_degree = vec![0u32; comp.len()];
    for (i, &u) in comp.iter().enumerate() {
        current_degree[i] = degree[u];
    }

    let mut removed = vec![0u8; comp.len()];
    let mut independent = vec![0u8; comp.len()];
    let mut heap = BinaryHeap::new();
    for i in 0..comp.len() {
        heap.push((Reverse(current_degree[i]), i));
    }

    let mut removal_list = Vec::new();

    while let Some((Reverse(d), i)) = heap.pop() {
        if removed[i] != 0 || current_degree[i] != d {
            continue;
        }

        independent[i] = 1;
        removal_list.clear();
        removed[i] = 1;
        removal_list.push(i);

        let u = comp[i];
        for &v in graph.neighbors(u) {
            let j = map[v as usize];
            if j != NONE && removed[j] == 0 {
                removed[j] = 1;
                removal_list.push(j);
            }
        }

        for &idx in &removal_list {
            let x = comp[idx];
            for &y in graph.neighbors(x) {
                let jy = map[y as usize];
                if jy != NONE && removed[jy] == 0 {
                    current_degree[jy] -= 1;
                    heap.push((Reverse(current_degree[jy]), jy));
                }
            }
        }
    }

    let mut cover = vec![1u8; comp.len()];
    for i in 0..comp.len() {
        if independent[i] != 0 {
            cover[i] = 0;
        }
    }
    cover
}

fn heuristic_component(
    graph: &Graph,
    comp: &[usize],
    degree: &[u32],
    map: &mut [usize],
) -> Vec<usize> {
    for (i, &u) in comp.iter().enumerate() {
        map[u] = i;
    }

    let mut cover_match = maximal_matching_cover(graph, comp, map);
    prune_cover_local(graph, comp, map, &mut cover_match);
    let count_match = cover_match.iter().filter(|&&x| x != 0).count();

    let mut cover_mis = greedy_independent_set_cover(graph, comp, degree, map);
    prune_cover_local(graph, comp, map, &mut cover_mis);
    let count_mis = cover_mis.iter().filter(|&&x| x != 0).count();

    let chosen = if count_match <= count_mis {
        cover_match
    } else {
        cover_mis
    };

    let mut result = Vec::new();
    for (i, &u) in comp.iter().enumerate() {
        if chosen[i] != 0 {
            result.push(u);
        }
        map[u] = NONE;
    }
    result
}

fn main() -> io::Result<()> {
    let mut scanner = Scanner::new()?;
    let n = scanner.next_usize();
    let m = scanner.next_usize();

    let mut build_degree = vec![0usize; n];
    let mut forced = vec![0u8; n];
    let mut edges = Vec::with_capacity(m);

    for _ in 0..m {
        let u = scanner.next_usize();
        let v = scanner.next_usize();
        if u == v {
            forced[u] = 1;
        } else {
            build_degree[u] += 1;
            build_degree[v] += 1;
            edges.push((u as u32, v as u32));
        }
    }

    let mut offsets = vec![0usize; n + 1];
    for i in 0..n {
        offsets[i + 1] = offsets[i] + build_degree[i];
    }
    let mut next = offsets[..n].to_vec();
    let mut neighbors = vec![0u32; offsets[n]];
    for (u, v) in edges {
        let uu = u as usize;
        let vv = v as usize;
        neighbors[next[uu]] = v;
        next[uu] += 1;
        neighbors[next[vv]] = u;
        next[vv] += 1;
    }

    let graph = Graph { offsets, neighbors };
    let mut active = vec![1u8; n];
    let mut in_cover = vec![0u8; n];
    let mut degree = vec![0u32; n];
    for i in 0..n {
        degree[i] = (graph.offsets[i + 1] - graph.offsets[i]) as u32;
    }

    let mut queue = VecDeque::new();
    for u in 0..n {
        if forced[u] != 0 {
            include_vertex(u, &graph, &mut active, &mut degree, &mut in_cover, &mut queue);
        }
    }
    for u in 0..n {
        if active[u] != 0 && degree[u] <= 1 {
            queue.push_back(u);
        }
    }

    while let Some(u) = queue.pop_front() {
        if active[u] == 0 {
            continue;
        }
        if degree[u] == 0 {
            discard_vertex(u, &graph, &mut active, &mut degree, &mut queue);
            continue;
        }
        if degree[u] != 1 {
            continue;
        }
        let mut neighbor = NONE;
        for &v in graph.neighbors(u) {
            let v = v as usize;
            if active[v] != 0 {
                neighbor = v;
                break;
            }
        }
        if neighbor == NONE {
            discard_vertex(u, &graph, &mut active, &mut degree, &mut queue);
        } else {
            include_vertex(
                neighbor,
                &graph,
                &mut active,
                &mut degree,
                &mut in_cover,
                &mut queue,
            );
        }
    }

    let mut visited = vec![0u8; n];
    let mut colors = vec![2u8; n];
    let mut bfs = VecDeque::new();
    let mut map = vec![NONE; n];

    for start in 0..n {
        if active[start] == 0 || degree[start] == 0 || visited[start] != 0 {
            continue;
        }

        let mut component = Vec::new();
        let mut is_bipartite = true;
        visited[start] = 1;
        colors[start] = 0;
        bfs.push_back(start);

        while let Some(u) = bfs.pop_front() {
            component.push(u);
            let color_u = colors[u];
            for &v in graph.neighbors(u) {
                let v = v as usize;
                if active[v] == 0 {
                    continue;
                }
                if visited[v] == 0 {
                    visited[v] = 1;
                    colors[v] = color_u ^ 1;
                    bfs.push_back(v);
                } else if colors[v] == color_u {
                    is_bipartite = false;
                }
            }
        }

        let chosen = if component.len() <= EXACT_THRESHOLD {
            exact_small_component(&graph, &component, &mut map)
        } else if is_bipartite {
            exact_bipartite_component(&graph, &component, &colors, &mut map)
        } else {
            heuristic_component(&graph, &component, &degree, &mut map)
        };

        for u in chosen {
            in_cover[u] = 1;
        }
    }

    let mut answer = Vec::new();
    for (u, &selected) in in_cover.iter().enumerate() {
        if selected != 0 {
            answer.push(u);
        }
    }
    answer.sort_unstable();

    let mut out = String::new();
    out.push_str(&answer.len().to_string());
    out.push('\n');
    for (i, u) in answer.iter().enumerate() {
        if i != 0 {
            out.push(' ');
        }
        out.push_str(&u.to_string());
    }
    out.push('\n');
    print!("{out}");

    Ok(())
}
