OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main Invocations 1
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0
%float_h = OpConstant %float 0.05
%float_nh = OpConstant %float -0.05
%delta_top = OpConstantComposite %v4float %float_0 %float_h %float_0 %float_0
%delta_right = OpConstantComposite %v4float %float_h %float_0 %float_0 %float_0
%delta_left = OpConstantComposite %v4float %float_nh %float_0 %float_0 %float_0
%delta_bottom = OpConstantComposite %v4float %float_0 %float_nh %float_0 %float_0
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_1 = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_Input_arr_gl_PerVertex_in_1 = OpTypePointer Input %arr_gl_PerVertex_in_1
%ptr_Input_v4float = OpTypePointer Input %v4float
%arr_v3float_1 = OpTypeArray %v3float %uint_1
%ptr_Input_arr_v3float_1 = OpTypePointer Input %arr_v3float_1
%ptr_Input_v3float = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_1 Input
%worldPos = OpVariable %ptr_Input_arr_v3float_1 Input
%normal = OpVariable %ptr_Input_arr_v3float_1 Input
%color = OpVariable %ptr_Input_arr_v3float_1 Input
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %ptr_Output_v3float Output
%main = OpFunction %void None %func
%entry = OpLabel
%pos_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr
%color_ptr = OpAccessChain %ptr_Input_v3float %color %int_0
%in_color = OpLoad %v3float %color_ptr
%gl_out_pos = OpAccessChain %ptr_Output_v4float %gl_out %int_0
%top = OpFAdd %v4float %pos %delta_top
OpStore %gl_out_pos %top
OpStore %outColor %in_color
OpEmitVertex
%right = OpFAdd %v4float %pos %delta_right
OpStore %gl_out_pos %right
OpStore %outColor %in_color
OpEmitVertex
%left = OpFAdd %v4float %pos %delta_left
OpStore %gl_out_pos %left
OpStore %outColor %in_color
OpEmitVertex
%bottom = OpFAdd %v4float %pos %delta_bottom
OpStore %gl_out_pos %bottom
OpStore %outColor %in_color
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
