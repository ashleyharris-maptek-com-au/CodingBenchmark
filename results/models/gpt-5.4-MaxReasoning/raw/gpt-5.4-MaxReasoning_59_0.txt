const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = input.trim() ? JSON.parse(input) : {};

const probability = Array.isArray(data.probability) ? data.probability : [];
const drones = Array.isArray(data.drones) ? data.drones : [];
const maxStepsPerDrone = Math.max(0, Number.isFinite(data.maxStepsPerDrone) ? Math.floor(data.maxStepsPerDrone) : 0);

const height = probability.length;
const width = probability.reduce(
  (best, row) => Math.max(best, Array.isArray(row) ? row.length : 0),
  0
);

function emit(paths) {
  process.stdout.write(JSON.stringify({ paths }));
}

function clampInt(value, min, max) {
  if (value < min) return min;
  if (value > max) return max;
  return value;
}

if (height === 0 || width === 0) {
  const paths = drones.map((drone) => {
    const x = Number.isFinite(drone && drone.x) ? Math.round(drone.x) : 0;
    const y = Number.isFinite(drone && drone.y) ? Math.round(drone.y) : 0;
    const cell = [x, y];
    const path = new Array(maxStepsPerDrone + 1);
    for (let i = 0; i < path.length; i += 1) path[i] = cell;
    return path;
  });
  emit(paths);
  process.exit(0);
}

const cellCount = width * height;
const values = new Float64Array(cellCount);
const passable = new Uint8Array(cellCount);

let totalPositive = 0;
let positiveCount = 0;

for (let y = 0; y < height; y += 1) {
  const row = Array.isArray(probability[y]) ? probability[y] : [];
  for (let x = 0; x < width; x += 1) {
    const idx = y * width + x;
    let raw = Number(row[x]);
    if (!Number.isFinite(raw)) raw = -1;
    if (raw >= 0) {
      passable[idx] = 1;
      values[idx] = raw;
      if (raw > 0) {
        totalPositive += raw;
        positiveCount += 1;
      }
    }
  }
}

const avgPositive = positiveCount > 0 ? totalPositive / positiveCount : 1;

function indexToCoord(idx) {
  return [idx % width, Math.floor(idx / width)];
}

function nearestPassableIndex(rawX, rawY) {
  const startX = clampInt(Number.isFinite(rawX) ? Math.round(rawX) : 0, 0, width - 1);
  const startY = clampInt(Number.isFinite(rawY) ? Math.round(rawY) : 0, 0, height - 1);
  const startIdx = startY * width + startX;
  if (passable[startIdx]) return startIdx;

  const seen = new Uint8Array(cellCount);
  const queue = new Int32Array(cellCount);
  let head = 0;
  let tail = 0;
  queue[tail++] = startIdx;
  seen[startIdx] = 1;

  while (head < tail) {
    const idx = queue[head++];
    if (passable[idx]) return idx;
    const x = idx % width;

    const up = idx - width;
    if (up >= 0 && !seen[up]) {
      seen[up] = 1;
      queue[tail++] = up;
    }
    const down = idx + width;
    if (down < cellCount && !seen[down]) {
      seen[down] = 1;
      queue[tail++] = down;
    }
    if (x > 0) {
      const left = idx - 1;
      if (!seen[left]) {
        seen[left] = 1;
        queue[tail++] = left;
      }
    }
    if (x + 1 < width) {
      const right = idx + 1;
      if (!seen[right]) {
        seen[right] = 1;
        queue[tail++] = right;
      }
    }
  }

  for (let idx = 0; idx < cellCount; idx += 1) {
    if (passable[idx]) return idx;
  }
  return 0;
}

const startIndices = drones.map((drone) => nearestPassableIndex(drone && drone.x, drone && drone.y));

const owner = new Int32Array(cellCount);
owner.fill(-1);
const ownerDist = new Int32Array(cellCount);
ownerDist.fill(-1);

const ownerQueue = new Int32Array(cellCount);
let ownerHead = 0;
let ownerTail = 0;

for (let i = 0; i < startIndices.length; i += 1) {
  const idx = startIndices[i];
  if (!passable[idx]) continue;
  if (ownerDist[idx] === -1) {
    ownerDist[idx] = 0;
    owner[idx] = i;
    ownerQueue[ownerTail++] = idx;
  } else if (i < owner[idx]) {
    owner[idx] = i;
  }
}

while (ownerHead < ownerTail) {
  const idx = ownerQueue[ownerHead++];
  const nextDist = ownerDist[idx] + 1;
  const x = idx % width;

  const up = idx - width;
  if (up >= 0 && passable[up] && ownerDist[up] === -1) {
    ownerDist[up] = nextDist;
    owner[up] = owner[idx];
    ownerQueue[ownerTail++] = up;
  }
  const down = idx + width;
  if (down < cellCount && passable[down] && ownerDist[down] === -1) {
    ownerDist[down] = nextDist;
    owner[down] = owner[idx];
    ownerQueue[ownerTail++] = down;
  }
  if (x > 0) {
    const left = idx - 1;
    if (passable[left] && ownerDist[left] === -1) {
      ownerDist[left] = nextDist;
      owner[left] = owner[idx];
      ownerQueue[ownerTail++] = left;
    }
  }
  if (x + 1 < width) {
    const right = idx + 1;
    if (passable[right] && ownerDist[right] === -1) {
      ownerDist[right] = nextDist;
      owner[right] = owner[idx];
      ownerQueue[ownerTail++] = right;
    }
  }
}

const potentialOffsets = [
  [0, 0, 1.0],
  [1, 0, 0.6],
  [-1, 0, 0.6],
  [0, 1, 0.6],
  [0, -1, 0.6],
  [2, 0, 0.28],
  [-2, 0, 0.28],
  [0, 2, 0.28],
  [0, -2, 0.28],
  [1, 1, 0.28],
  [1, -1, 0.28],
  [-1, 1, 0.28],
  [-1, -1, 0.28],
];

const potential = new Float64Array(cellCount);
for (let y = 0; y < height; y += 1) {
  for (let x = 0; x < width; x += 1) {
    const idx = y * width + x;
    if (!passable[idx]) continue;
    let sum = 0;
    for (let i = 0; i < potentialOffsets.length; i += 1) {
      const offset = potentialOffsets[i];
      const nx = x + offset[0];
      const ny = y + offset[1];
      if (nx < 0 || ny < 0 || nx >= width || ny >= height) continue;
      const nidx = ny * width + nx;
      if (passable[nidx] && values[nidx] > 0) {
        sum += values[nidx] * offset[2];
      }
    }
    potential[idx] = sum;
  }
}

const covered = new Uint8Array(cellCount);
const bfsSeen = new Int32Array(cellCount);
const bfsDist = new Int32Array(cellCount);
const bfsParent = new Int32Array(cellCount);
const bfsQueue = new Int32Array(cellCount);
let bfsToken = 1;

function markCovered(idx) {
  if (idx >= 0 && idx < cellCount) covered[idx] = 1;
}

function uncoveredNeighborhoodScore(idx) {
  const x = idx % width;
  const y = Math.floor(idx / width);
  let sum = 0;
  for (let i = 0; i < potentialOffsets.length; i += 1) {
    const offset = potentialOffsets[i];
    const nx = x + offset[0];
    const ny = y + offset[1];
    if (nx < 0 || ny < 0 || nx >= width || ny >= height) continue;
    const nidx = ny * width + nx;
    if (!covered[nidx] && values[nidx] > 0) {
      sum += values[nidx] * offset[2];
    }
  }
  return sum;
}

function localMoveReward(candidateIdx, currentIdx, previousIdx, droneIdx) {
  if (!passable[candidateIdx]) return -Infinity;

  let reward = uncoveredNeighborhoodScore(candidateIdx) * 0.9;

  if (!covered[candidateIdx] && values[candidateIdx] > 0) {
    reward += values[candidateIdx] * 2.4;
  }
  if (candidateIdx === currentIdx) {
    reward -= avgPositive * 0.9;
  } else {
    reward += avgPositive * 0.1;
  }
  if (candidateIdx === previousIdx) {
    reward -= avgPositive * 1.05;
  }
  if (covered[candidateIdx]) {
    reward -= avgPositive * 0.5;
  }
  if (owner[candidateIdx] === droneIdx) {
    reward += avgPositive * 0.18;
  } else if (owner[candidateIdx] !== -1) {
    reward -= avgPositive * 0.08;
  }

  return reward;
}

function bestImmediateMove(currentIdx, previousIdx, droneIdx) {
  const candidates = [currentIdx];
  const x = currentIdx % width;
  const up = currentIdx - width;
  if (up >= 0) candidates.push(up);
  const down = currentIdx + width;
  if (down < cellCount) candidates.push(down);
  if (x > 0) candidates.push(currentIdx - 1);
  if (x + 1 < width) candidates.push(currentIdx + 1);

  let bestIdx = currentIdx;
  let bestScore = -Infinity;
  for (let i = 0; i < candidates.length; i += 1) {
    const candidateIdx = candidates[i];
    if (!passable[candidateIdx]) continue;
    const score = localMoveReward(candidateIdx, currentIdx, previousIdx, droneIdx);
    if (score > bestScore) {
      bestScore = score;
      bestIdx = candidateIdx;
    }
  }

  return { bestIdx, bestScore };
}

function walkLocalCluster(path, currentIdx, previousIdx, droneIdx, remainingSteps) {
  let current = currentIdx;
  let previous = previousIdx;
  let stepsUsed = 0;
  const localLimit = Math.min(remainingSteps, 12);
  const rewardFloor = Math.max(0.001, avgPositive * 0.45);

  while (stepsUsed < localLimit) {
    const choice = bestImmediateMove(current, previous, droneIdx);
    if (choice.bestIdx === current || choice.bestScore < rewardFloor) break;
    previous = current;
    current = choice.bestIdx;
    markCovered(current);
    path.push(indexToCoord(current));
    stepsUsed += 1;
  }

  return { currentIdx: current, previousIdx: previous, stepsUsed };
}

function findBestRoute(startIdx, stepBudget, droneIdx) {
  bfsToken += 1;
  if (bfsToken === 0x7fffffff) {
    bfsSeen.fill(0);
    bfsToken = 1;
  }
  const token = bfsToken;

  let head = 0;
  let tail = 0;
  bfsQueue[tail++] = startIdx;
  bfsSeen[startIdx] = token;
  bfsDist[startIdx] = 0;
  bfsParent[startIdx] = -1;

  let bestIdx = -1;
  let bestScore = -Infinity;
  let bestDist = 0;

  while (head < tail) {
    const idx = bfsQueue[head++];
    const dist = bfsDist[idx];

    if (idx !== startIdx) {
      const uncovered = covered[idx] ? 0 : values[idx];
      const region = potential[idx];
      if (uncovered > 0 || region > 0) {
        const ownFactor =
          owner[idx] === droneIdx ? 1.15 : owner[idx] === -1 ? 0.9 : 0.68;
        const leftover = stepBudget - dist;
        const reachFactor = 0.65 + 0.35 * Math.min(1, leftover / 6);
        let score =
          (ownFactor * (uncovered * 2.4 + region * 0.62 * reachFactor)) /
          Math.pow(dist + 1, 0.9);
        if (covered[idx]) score *= 0.92;
        if (
          score > bestScore + 1e-12 ||
          (Math.abs(score - bestScore) <= 1e-12 && dist < bestDist)
        ) {
          bestScore = score;
          bestIdx = idx;
          bestDist = dist;
        }
      }
    }

    if (dist >= stepBudget) continue;
    const x = idx % width;
    const nextDist = dist + 1;

    const up = idx - width;
    if (up >= 0 && passable[up] && bfsSeen[up] !== token) {
      bfsSeen[up] = token;
      bfsDist[up] = nextDist;
      bfsParent[up] = idx;
      bfsQueue[tail++] = up;
    }
    const down = idx + width;
    if (down < cellCount && passable[down] && bfsSeen[down] !== token) {
      bfsSeen[down] = token;
      bfsDist[down] = nextDist;
      bfsParent[down] = idx;
      bfsQueue[tail++] = down;
    }
    if (x > 0) {
      const left = idx - 1;
      if (passable[left] && bfsSeen[left] !== token) {
        bfsSeen[left] = token;
        bfsDist[left] = nextDist;
        bfsParent[left] = idx;
        bfsQueue[tail++] = left;
      }
    }
    if (x + 1 < width) {
      const right = idx + 1;
      if (passable[right] && bfsSeen[right] !== token) {
        bfsSeen[right] = token;
        bfsDist[right] = nextDist;
        bfsParent[right] = idx;
        bfsQueue[tail++] = right;
      }
    }
  }

  const usefulFloor = Math.max(0.001, avgPositive * 0.22);
  if (bestIdx === -1 || bestScore < usefulFloor) return null;

  const route = [];
  let cursor = bestIdx;
  while (cursor !== startIdx && cursor !== -1) {
    route.push(cursor);
    cursor = bfsParent[cursor];
  }
  route.reverse();
  return route;
}

function planDronePath(droneIdx) {
  let currentIdx = startIndices[droneIdx];
  let previousIdx = -1;
  let remainingSteps = maxStepsPerDrone;
  const path = [indexToCoord(currentIdx)];
  markCovered(currentIdx);

  while (remainingSteps > 0) {
    const localWalk = walkLocalCluster(
      path,
      currentIdx,
      previousIdx,
      droneIdx,
      remainingSteps
    );
    currentIdx = localWalk.currentIdx;
    previousIdx = localWalk.previousIdx;
    remainingSteps -= localWalk.stepsUsed;
    if (remainingSteps <= 0) break;

    const route = findBestRoute(currentIdx, remainingSteps, droneIdx);
    if (!route || route.length === 0) break;

    for (let i = 0; i < route.length && remainingSteps > 0; i += 1) {
      previousIdx = currentIdx;
      currentIdx = route[i];
      markCovered(currentIdx);
      path.push(indexToCoord(currentIdx));
      remainingSteps -= 1;
    }
  }

  const stayCell = indexToCoord(currentIdx);
  while (path.length < maxStepsPerDrone + 1) {
    path.push(stayCell);
  }

  return path;
}

const paths = drones.map((_, idx) => planDronePath(idx));
emit(paths);
