; SPIR-V
; Version: 1.0
OpCapability Shader
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
%void = OpTypeVoid
%fn = OpTypeFunction %void
%bool = OpTypeBool
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3bool = OpTypeVector %bool 3
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%int = OpTypeInt 32 1
%float_0 = OpConstant %float 0
%float_0_1 = OpConstant %float 0.1
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_7 = OpConstant %float 0.7
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1
%float_6 = OpConstant %float 6
%int_3 = OpConstant %int 3
%v2_6 = OpConstantComposite %v2float %float_6 %float_6
%v2_0_5 = OpConstantComposite %v2float %float_0_5 %float_0_5
%yellow = OpConstantComposite %v3float %float_1 %float_0_8 %float_0
%darkBlue = OpConstantComposite %v3float %float_0_1 %float_0_1 %float_0_3
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%inWorldPos = OpVariable %_ptr_Input_v3float Input
%inNormal = OpVariable %_ptr_Input_v3float Input
%inUV = OpVariable %_ptr_Input_v2float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%main = OpFunction %void None %fn
%entry = OpLabel
%worldPos = OpLoad %v3float %inWorldPos
%normal = OpLoad %v3float %inNormal
%uv = OpLoad %v2float %inUV
%lightPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%light4 = OpLoad %v4float %lightPtr
%lightPos = OpVectorShuffle %v3float %light4 %light4 0 1 2
%lightVec = OpFSub %v3float %lightPos %worldPos
%n = OpExtInst %v3float %glsl450 Normalize %normal
%l = OpExtInst %v3float %glsl450 Normalize %lightVec
%ndotlRaw = OpDot %float %n %l
%isLit = OpFOrdGreaterThan %bool %ndotlRaw %float_0
%diffuse = OpSelect %float %isLit %ndotlRaw %float_0
%shadeScaled = OpFMul %float %float_0_7 %diffuse
%shade = OpFAdd %float %float_0_3 %shadeScaled
%uvScaled = OpFMul %v2float %uv %v2_6
%uvCell = OpExtInst %v2float %glsl450 Fract %uvScaled
%uvCentered = OpFSub %v2float %uvCell %v2_0_5
%dist = OpExtInst %float %glsl450 Length %uvCentered
%isDot = OpFOrdLessThanEqual %bool %dist %float_0_3
%isDot3 = OpCompositeConstruct %v3bool %isDot %isDot %isDot
%baseRgb = OpSelect %v3float %isDot3 %yellow %darkBlue
%litRgb = OpVectorTimesScalar %v3float %baseRgb %shade
%r = OpCompositeExtract %float %litRgb 0
%g = OpCompositeExtract %float %litRgb 1
%b = OpCompositeExtract %float %litRgb 2
%rgba = OpCompositeConstruct %v4float %r %g %b %float_1
OpStore %outColor %rgba
OpReturn
OpFunctionEnd
