declare const require: any;
declare const process: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type DroneInput = {
  id: string | number;
  x: number;
  y: number;
};

type InputData = {
  probability: number[][];
  drones: DroneInput[];
  maxStepsPerDrone: number;
};

type DroneState = {
  index: number;
  current: number;
  previous: number;
  recent: number[];
  path: number[];
};

const parsed: InputData = {
  probability: Array.isArray(data?.probability) ? data.probability : [],
  drones: Array.isArray(data?.drones) ? data.drones : [],
  maxStepsPerDrone: Number.isFinite(data?.maxStepsPerDrone)
    ? Math.max(0, Math.floor(data.maxStepsPerDrone))
    : 0,
};

const grid = parsed.probability;
const height = grid.length;
let width = 0;
for (let y = 0; y < height; y++) {
  const row = Array.isArray(grid[y]) ? grid[y] : [];
  if (row.length > width) width = row.length;
}
const droneCount = parsed.drones.length;

if (height === 0 || width === 0 || droneCount === 0) {
  const emptyPaths = parsed.drones.map(() => []);
  process.stdout.write(JSON.stringify({ paths: emptyPaths }));
  process.exit(0);
}

const cellCount = width * height;
const blocked = new Uint8Array(cellCount);
const remainingValue = new Float64Array(cellCount);

let freeCellCount = 0;
for (let y = 0; y < height; y++) {
  const row = Array.isArray(grid[y]) ? grid[y] : [];
  const rowBase = y * width;
  for (let x = 0; x < width; x++) {
    const raw = x < row.length ? Number(row[x]) : -1;
    const idx = rowBase + x;
    if (!Number.isFinite(raw) || raw === -1) {
      blocked[idx] = 1;
      continue;
    }
    freeCellCount++;
    if (raw > 0) {
      remainingValue[idx] = raw;
    }
  }
}

if (freeCellCount === 0) {
  const impossiblePaths = parsed.drones.map(() => []);
  process.stdout.write(JSON.stringify({ paths: impossiblePaths }));
  process.exit(0);
}

const indexOf = (x: number, y: number): number => y * width + x;
const xOf = (idx: number): number => idx % width;
const yOf = (idx: number): number => Math.floor(idx / width);

function clamp(value: number, min: number, max: number): number {
  if (value < min) return min;
  if (value > max) return max;
  return value;
}

function inside(x: number, y: number): boolean {
  return x >= 0 && x < width && y >= 0 && y < height;
}

function findNearestFreeIndex(x: number, y: number): number {
  let bestIdx = -1;
  let bestDist = Number.POSITIVE_INFINITY;
  for (let idx = 0; idx < cellCount; idx++) {
    if (blocked[idx] !== 0) continue;
    const dist = Math.abs(xOf(idx) - x) + Math.abs(yOf(idx) - y);
    if (dist < bestDist) {
      bestDist = dist;
      bestIdx = idx;
      if (dist === 0) break;
    }
  }
  return bestIdx;
}

const startIndices = new Int32Array(droneCount);
for (let i = 0; i < droneCount; i++) {
  const drone = parsed.drones[i] ?? ({ x: 0, y: 0 } as DroneInput);
  const rawX = Number.isFinite(drone.x) ? Math.trunc(drone.x) : 0;
  const rawY = Number.isFinite(drone.y) ? Math.trunc(drone.y) : 0;
  const x = clamp(rawX, 0, width - 1);
  const y = clamp(rawY, 0, height - 1);
  let idx = indexOf(x, y);
  if (blocked[idx] !== 0) {
    idx = findNearestFreeIndex(x, y);
  }
  startIndices[i] = idx;
}

const owner = new Int32Array(cellCount);
owner.fill(-1);
const bfsQueue = new Int32Array(cellCount);
let queueHead = 0;
let queueTail = 0;

for (let i = 0; i < droneCount; i++) {
  const start = startIndices[i];
  if (start < 0) continue;
  if (owner[start] === -1) {
    owner[start] = i;
    bfsQueue[queueTail++] = start;
  }
}

while (queueHead < queueTail) {
  const idx = bfsQueue[queueHead++];
  const x = xOf(idx);
  const y = yOf(idx);
  const currentOwner = owner[idx];

  const nx0 = x + 1;
  if (nx0 < width) {
    const next = idx + 1;
    if (blocked[next] === 0 && owner[next] === -1) {
      owner[next] = currentOwner;
      bfsQueue[queueTail++] = next;
    }
  }

  const nx1 = x - 1;
  if (nx1 >= 0) {
    const next = idx - 1;
    if (blocked[next] === 0 && owner[next] === -1) {
      owner[next] = currentOwner;
      bfsQueue[queueTail++] = next;
    }
  }

  const ny0 = y + 1;
  if (ny0 < height) {
    const next = idx + width;
    if (blocked[next] === 0 && owner[next] === -1) {
      owner[next] = currentOwner;
      bfsQueue[queueTail++] = next;
    }
  }

  const ny1 = y - 1;
  if (ny1 >= 0) {
    const next = idx - width;
    if (blocked[next] === 0 && owner[next] === -1) {
      owner[next] = currentOwner;
      bfsQueue[queueTail++] = next;
    }
  }
}

let reachablePositiveCount = 0;
let reachableValueTotal = 0;
for (let idx = 0; idx < cellCount; idx++) {
  if (owner[idx] === -1) {
    remainingValue[idx] = 0;
    continue;
  }
  const value = remainingValue[idx];
  if (value > 0) {
    reachablePositiveCount++;
    reachableValueTotal += value;
  }
}

const maxSteps = parsed.maxStepsPerDrone;

if (reachablePositiveCount === 0) {
  const stayPaths = parsed.drones.map((_, i) => {
    const start = startIndices[i];
    if (start < 0) return [];
    const x = xOf(start);
    const y = yOf(start);
    const path: [number, number][] = [];
    for (let step = 0; step <= maxSteps; step++) {
      path.push([x, y]);
    }
    return path;
  });
  process.stdout.write(JSON.stringify({ paths: stayPaths }));
  process.exit(0);
}

const valueScale = Math.max(1e-9, reachableValueTotal / reachablePositiveCount);

const freeDegree = new Uint8Array(cellCount);
for (let idx = 0; idx < cellCount; idx++) {
  if (owner[idx] === -1) continue;
  const x = xOf(idx);
  const y = yOf(idx);
  let degree = 0;
  if (x + 1 < width && owner[idx + 1] !== -1) degree++;
  if (x - 1 >= 0 && owner[idx - 1] !== -1) degree++;
  if (y + 1 < height && owner[idx + width] !== -1) degree++;
  if (y - 1 >= 0 && owner[idx - width] !== -1) degree++;
  freeDegree[idx] = degree;
}

const prefixWidth = width + 1;
const prefix = new Float64Array((height + 1) * prefixWidth);
for (let y = 0; y < height; y++) {
  let rowSum = 0;
  const rowBase = y * width;
  const prefixRow = (y + 1) * prefixWidth;
  const prefixPrev = y * prefixWidth;
  for (let x = 0; x < width; x++) {
    rowSum += remainingValue[rowBase + x];
    prefix[prefixRow + x + 1] = prefix[prefixPrev + x + 1] + rowSum;
  }
}

function boxSum(minX: number, minY: number, maxX: number, maxY: number): number {
  const x0 = clamp(minX, 0, width - 1);
  const y0 = clamp(minY, 0, height - 1);
  const x1 = clamp(maxX, 0, width - 1);
  const y1 = clamp(maxY, 0, height - 1);
  if (x0 > x1 || y0 > y1) return 0;
  const a = y0 * prefixWidth + x0;
  const b = y0 * prefixWidth + (x1 + 1);
  const c = (y1 + 1) * prefixWidth + x0;
  const d = (y1 + 1) * prefixWidth + (x1 + 1);
  return prefix[d] - prefix[b] - prefix[c] + prefix[a];
}

const boxRadius = 4;
const staticDensity = new Float64Array(cellCount);
for (let idx = 0; idx < cellCount; idx++) {
  if (owner[idx] === -1) continue;
  const x = xOf(idx);
  const y = yOf(idx);
  staticDensity[idx] = boxSum(x - boxRadius, y - boxRadius, x + boxRadius, y + boxRadius);
}

type Offset = { dx: number; dy: number; weight: number };
const influenceRadius = 3;
const influenceOffsets: Offset[] = [];
for (let dy = -influenceRadius; dy <= influenceRadius; dy++) {
  for (let dx = -influenceRadius; dx <= influenceRadius; dx++) {
    const distance = Math.abs(dx) + Math.abs(dy);
    if (distance > influenceRadius) continue;
    influenceOffsets.push({ dx, dy, weight: influenceRadius + 1 - distance });
  }
}

const potential = new Float64Array(cellCount);
for (let idx = 0; idx < cellCount; idx++) {
  const value = remainingValue[idx];
  if (value <= 0) continue;
  const x = xOf(idx);
  const y = yOf(idx);
  for (let i = 0; i < influenceOffsets.length; i++) {
    const offset = influenceOffsets[i];
    const nx = x + offset.dx;
    const ny = y + offset.dy;
    if (!inside(nx, ny)) continue;
    const next = indexOf(nx, ny);
    if (owner[next] === -1) continue;
    potential[next] += value * offset.weight;
  }
}

function removeCellValue(idx: number): void {
  const value = remainingValue[idx];
  if (value <= 0) return;
  remainingValue[idx] = 0;
  const x = xOf(idx);
  const y = yOf(idx);
  for (let i = 0; i < influenceOffsets.length; i++) {
    const offset = influenceOffsets[i];
    const nx = x + offset.dx;
    const ny = y + offset.dy;
    if (!inside(nx, ny)) continue;
    const next = indexOf(nx, ny);
    if (owner[next] === -1) continue;
    potential[next] -= value * offset.weight;
  }
}

const visitCount = new Uint16Array(cellCount);
const recentWindow = 8;

const drones: DroneState[] = [];
for (let i = 0; i < droneCount; i++) {
  const start = startIndices[i];
  if (start >= 0) {
    visitCount[start] = Math.min(65535, visitCount[start] + 1);
    removeCellValue(start);
  }
  drones.push({
    index: i,
    current: start,
    previous: -1,
    recent: start >= 0 ? [start] : [],
    path: start >= 0 ? [start] : [],
  });
}

function hasRecent(drone: DroneState, idx: number): boolean {
  const recent = drone.recent;
  for (let i = 0; i < recent.length; i++) {
    if (recent[i] === idx) return true;
  }
  return false;
}

function frontierScore(idx: number): number {
  const x = xOf(idx);
  const y = yOf(idx);
  let score = 0;
  if (x + 1 < width) {
    const next = idx + 1;
    if (owner[next] !== -1) score += remainingValue[next] > 0 ? 1.5 : potential[next] > 0 ? 0.35 : 0;
  }
  if (x - 1 >= 0) {
    const next = idx - 1;
    if (owner[next] !== -1) score += remainingValue[next] > 0 ? 1.5 : potential[next] > 0 ? 0.35 : 0;
  }
  if (y + 1 < height) {
    const next = idx + width;
    if (owner[next] !== -1) score += remainingValue[next] > 0 ? 1.5 : potential[next] > 0 ? 0.35 : 0;
  }
  if (y - 1 >= 0) {
    const next = idx - width;
    if (owner[next] !== -1) score += remainingValue[next] > 0 ? 1.5 : potential[next] > 0 ? 0.35 : 0;
  }
  return score;
}

function localScore(drone: DroneState, candidate: number, reserved: Set<number>): number {
  if (candidate < 0) return Number.NEGATIVE_INFINITY;
  if (reserved.has(candidate)) return Number.NEGATIVE_INFINITY;

  const current = drone.current;
  const immediate = remainingValue[candidate] / valueScale;
  const localPotential = potential[candidate] / valueScale;
  const boxPotential = staticDensity[candidate] / valueScale;
  const currentBoxPotential = staticDensity[current] / valueScale;
  const frontier = frontierScore(candidate);
  let score = 0;

  score += immediate * 18;
  score += localPotential * 1.9;
  score += boxPotential * 0.08;
  score += (boxPotential - currentBoxPotential) * 0.14;
  score += frontier;
  score += freeDegree[candidate] * 0.18;

  if (owner[candidate] === drone.index) {
    score += 2.5;
  } else if (owner[candidate] !== -1) {
    score -= 1.5;
  }

  score -= visitCount[candidate] * 4.0;

  if (candidate === current) score -= 2.8;
  if (candidate === drone.previous) score -= 3.5;
  if (hasRecent(drone, candidate)) score -= 6.0;

  return score;
}

const searchSeen = new Int32Array(cellCount);
const searchParent = new Int32Array(cellCount);
const searchDepth = new Int16Array(cellCount);
const searchQueue = new Int32Array(cellCount);
let searchToken = 1;

function chooseEscapeStep(drone: DroneState, reserved: Set<number>, baseline: number, stepsLeft: number): number {
  const start = drone.current;
  const maxDepth = Math.min(12, stepsLeft);
  if (start < 0 || maxDepth <= 1) return -1;

  searchToken++;
  if (searchToken === 0) {
    searchSeen.fill(0);
    searchToken = 1;
  }

  let head = 0;
  let tail = 0;
  searchSeen[start] = searchToken;
  searchParent[start] = -1;
  searchDepth[start] = 0;
  searchQueue[tail++] = start;

  let bestTarget = -1;
  let bestScore = baseline + 2.0;

  while (head < tail) {
    const idx = searchQueue[head++];
    const depth = searchDepth[idx];

    if (depth > 0) {
      let score = 0;
      score += (remainingValue[idx] / valueScale) * 18;
      score += (potential[idx] / valueScale) * 2.0;
      score += (staticDensity[idx] / valueScale) * 0.08;
      score -= depth * 2.25;
      score -= visitCount[idx] * 2.5;
      if (owner[idx] === drone.index) {
        score += 2.0;
      } else if (owner[idx] !== -1) {
        score -= 1.25;
      }
      if (hasRecent(drone, idx)) score -= 5.0;

      if (score > bestScore) {
        bestScore = score;
        bestTarget = idx;
      }
    }

    if (depth >= maxDepth) continue;

    const x = xOf(idx);
    const y = yOf(idx);

    const neighbors: number[] = [];
    if (x + 1 < width) neighbors.push(idx + 1);
    if (x - 1 >= 0) neighbors.push(idx - 1);
    if (y + 1 < height) neighbors.push(idx + width);
    if (y - 1 >= 0) neighbors.push(idx - width);

    for (let i = 0; i < neighbors.length; i++) {
      const next = neighbors[i];
      if (owner[next] === -1) continue;
      if (searchSeen[next] === searchToken) continue;
      searchSeen[next] = searchToken;
      searchParent[next] = idx;
      searchDepth[next] = (depth + 1) as number;
      searchQueue[tail++] = next;
    }
  }

  if (bestTarget === -1) return -1;

  let step = bestTarget;
  while (searchParent[step] !== start && searchParent[step] !== -1) {
    step = searchParent[step];
  }

  if (reserved.has(step)) return -1;
  return step;
}

function chooseNextStep(drone: DroneState, reserved: Set<number>, stepsLeft: number): number {
  const current = drone.current;
  if (current < 0) return -1;

  const candidates: number[] = [current];
  const x = xOf(current);
  const y = yOf(current);
  if (x + 1 < width && owner[current + 1] !== -1) candidates.push(current + 1);
  if (x - 1 >= 0 && owner[current - 1] !== -1) candidates.push(current - 1);
  if (y + 1 < height && owner[current + width] !== -1) candidates.push(current + width);
  if (y - 1 >= 0 && owner[current - width] !== -1) candidates.push(current - width);

  let best = current;
  let bestScore = Number.NEGATIVE_INFINITY;
  for (let i = 0; i < candidates.length; i++) {
    const candidate = candidates[i];
    const score = localScore(drone, candidate, reserved);
    if (score > bestScore) {
      bestScore = score;
      best = candidate;
    }
  }

  if (
    stepsLeft > 1 &&
    (best === current || bestScore < 3.0 || (remainingValue[best] <= 0 && potential[best] / valueScale < 1.2))
  ) {
    const escape = chooseEscapeStep(drone, reserved, bestScore, stepsLeft);
    if (escape >= 0) return escape;
  }

  return best;
}

for (let step = 0; step < maxSteps; step++) {
  const reserved = new Set<number>();
  for (let turn = 0; turn < droneCount; turn++) {
    const droneIndex = (step + turn) % droneCount;
    const drone = drones[droneIndex];
    if (drone.current < 0) continue;

    const next = chooseNextStep(drone, reserved, maxSteps - step);
    const destination = next >= 0 ? next : drone.current;

    drone.previous = drone.current;
    drone.current = destination;
    drone.path.push(destination);
    drone.recent.push(destination);
    if (drone.recent.length > recentWindow) {
      drone.recent.shift();
    }

    reserved.add(destination);
    visitCount[destination] = Math.min(65535, visitCount[destination] + 1);
    removeCellValue(destination);
  }
}

const paths = drones.map((drone) =>
  drone.path.map((idx) => [xOf(idx), yOf(idx)] as [number, number]),
);

process.stdout.write(JSON.stringify({ paths }));
