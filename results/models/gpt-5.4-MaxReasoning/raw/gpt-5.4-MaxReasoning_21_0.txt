#include <algorithm>
#include <array>
#include <cstdint>
#include <deque>
#include <iostream>
#include <limits>
#include <optional>
#include <queue>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using Id = std::uint64_t;
constexpr int MAX_D = 24;

struct Move {
    int axis = 0;
    int dir = 1;
};

struct Board {
    int dims = 0;
    std::array<int, MAX_D> bounds{};
    std::array<Id, MAX_D> stride{};
};

struct Problem {
    Board board;
    int maxTurns = 0;
    std::unordered_set<Id> obstacles;
};

struct SnakeState {
    std::deque<Id> body;                 // head at front, tail at back
    std::unordered_set<Id> occupied;     // mirrors body for O(1) collision checks
    std::unordered_set<Id> foods;
};

struct SearchNode {
    int f = 0;
    int g = 0;
    int h = 0;
    Id id = 0;

    bool operator<(const SearchNode& other) const {
        if (f != other.f) return f > other.f;
        if (h != other.h) return h > other.h;
        return g < other.g;
    }
};

static std::array<int, MAX_D> decode(Id id, const Board& board) {
    std::array<int, MAX_D> point{};
    for (int axis = board.dims - 1; axis >= 0; --axis) {
        point[axis] = static_cast<int>((id / board.stride[axis]) % static_cast<Id>(board.bounds[axis]));
    }
    return point;
}

static Id encode(const std::array<int, MAX_D>& point, const Board& board) {
    Id id = 0;
    for (int axis = 0; axis < board.dims; ++axis) {
        id += static_cast<Id>(point[axis]) * board.stride[axis];
    }
    return id;
}

static int manhattan(Id a, Id b, const Board& board) {
    auto pa = decode(a, board);
    auto pb = decode(b, board);
    int dist = 0;
    for (int axis = 0; axis < board.dims; ++axis) {
        int delta = pa[axis] - pb[axis];
        dist += delta >= 0 ? delta : -delta;
    }
    return dist;
}

static int manhattanToPoint(Id id, const std::array<int, MAX_D>& goal, const Board& board) {
    auto point = decode(id, board);
    int dist = 0;
    for (int axis = 0; axis < board.dims; ++axis) {
        int delta = point[axis] - goal[axis];
        dist += delta >= 0 ? delta : -delta;
    }
    return dist;
}

template <class Func>
static void forEachNeighbor(Id id, const Board& board, Func&& func) {
    auto point = decode(id, board);
    for (int axis = 0; axis < board.dims; ++axis) {
        if (point[axis] + 1 < board.bounds[axis]) {
            func(id + board.stride[axis], axis, +1);
        }
        if (point[axis] > 0) {
            func(id - board.stride[axis], axis, -1);
        }
    }
}

static bool cellBlockedForSearch(const Problem& problem,
                                 const SnakeState& state,
                                 Id cell,
                                 Id start,
                                 Id goal,
                                 bool avoidFoods) {
    if (problem.obstacles.contains(cell)) return true;
    if (cell != start && state.occupied.contains(cell) && cell != state.body.back() && cell != goal) return true;
    if (avoidFoods && cell != goal && state.foods.contains(cell)) return true;
    return false;
}

static std::optional<std::vector<Move>> findPath(const Problem& problem,
                                                 const SnakeState& state,
                                                 Id start,
                                                 Id goal,
                                                 std::size_t budget,
                                                 bool avoidFoods) {
    if (start == goal) return std::vector<Move>{};

    const auto goalPoint = decode(goal, problem.board);
    auto heuristic = [&](Id id) { return manhattanToPoint(id, goalPoint, problem.board); };

    std::priority_queue<SearchNode> open;
    std::unordered_map<Id, int> bestG;
    std::unordered_map<Id, Id> parent;
    bestG.reserve(budget * 2 + 32);
    parent.reserve(budget * 2 + 32);

    int startH = heuristic(start);
    open.push(SearchNode{startH * 3, 0, startH, start});
    bestG[start] = 0;

    std::size_t expanded = 0;
    while (!open.empty() && expanded < budget) {
        SearchNode cur = open.top();
        open.pop();

        auto gIt = bestG.find(cur.id);
        if (gIt == bestG.end() || gIt->second != cur.g) continue;
        if (cur.id == goal) break;

        ++expanded;

        struct Candidate {
            int h;
            Id id;
        };

        std::vector<Candidate> neighbors;
        neighbors.reserve(problem.board.dims * 2);

        forEachNeighbor(cur.id, problem.board, [&](Id next, int, int) {
            if (cellBlockedForSearch(problem, state, next, start, goal, avoidFoods)) return;
            neighbors.push_back(Candidate{heuristic(next), next});
        });

        std::sort(neighbors.begin(), neighbors.end(), [](const Candidate& lhs, const Candidate& rhs) {
            if (lhs.h != rhs.h) return lhs.h < rhs.h;
            return lhs.id < rhs.id;
        });

        for (const Candidate& next : neighbors) {
            int nextG = cur.g + 1;
            auto bestIt = bestG.find(next.id);
            if (bestIt != bestG.end() && bestIt->second <= nextG) continue;
            bestG[next.id] = nextG;
            parent[next.id] = cur.id;
            open.push(SearchNode{nextG + next.h * 3, nextG, next.h, next.id});
        }
    }

    if (!bestG.contains(goal)) return std::nullopt;

    std::vector<Id> idPath;
    for (Id cur = goal; cur != start; cur = parent[cur]) {
        idPath.push_back(cur);
    }
    std::reverse(idPath.begin(), idPath.end());

    std::vector<Move> moves;
    moves.reserve(idPath.size());

    Id prev = start;
    for (Id cur : idPath) {
        Id diff = cur >= prev ? cur - prev : prev - cur;
        int dir = cur >= prev ? +1 : -1;
        int axis = 0;
        while (axis < problem.board.dims && problem.board.stride[axis] != diff) ++axis;
        if (axis == problem.board.dims) return std::nullopt;
        moves.push_back(Move{axis, dir});
        prev = cur;
    }

    return moves;
}

static bool applyMove(const Problem& problem, SnakeState& state, const Move& move, bool* ateFood = nullptr) {
    if (state.body.empty()) return false;

    Id head = state.body.front();
    auto point = decode(head, problem.board);

    if (move.dir > 0) {
        if (point[move.axis] + 1 >= problem.board.bounds[move.axis]) return false;
        head += problem.board.stride[move.axis];
    } else {
        if (point[move.axis] == 0) return false;
        head -= problem.board.stride[move.axis];
    }

    if (problem.obstacles.contains(head)) return false;

    bool eat = state.foods.contains(head);
    Id tail = state.body.back();
    if (state.occupied.contains(head) && !(head == tail && !eat)) return false;

    state.body.push_front(head);
    state.occupied.insert(head);
    if (eat) {
        state.foods.erase(head);
    } else {
        state.occupied.erase(tail);
        state.body.pop_back();
    }

    if (ateFood) *ateFood = eat;
    return true;
}

static std::optional<SnakeState> simulateMoves(const Problem& problem,
                                               const SnakeState& initial,
                                               const std::vector<Move>& moves,
                                               bool* ateAnyFood = nullptr) {
    SnakeState state = initial;
    bool ate = false;
    for (const Move& move : moves) {
        bool ateThisStep = false;
        if (!applyMove(problem, state, move, &ateThisStep)) return std::nullopt;
        ate = ate || ateThisStep;
    }
    if (ateAnyFood) *ateAnyFood = ate;
    return state;
}

static std::size_t reachableSpaceEstimate(const Problem& problem,
                                          const SnakeState& state,
                                          Id start,
                                          std::size_t cap) {
    if (cap == 0) return 0;

    const Id tail = state.body.back();
    std::queue<Id> q;
    std::unordered_set<Id> visited;
    visited.reserve(cap * 2 + 32);

    q.push(start);
    visited.insert(start);

    std::size_t count = 0;
    while (!q.empty() && count < cap) {
        Id cur = q.front();
        q.pop();
        ++count;

        forEachNeighbor(cur, problem.board, [&](Id next, int, int) {
            if (visited.contains(next)) return;
            if (problem.obstacles.contains(next)) return;
            if (next != start && state.occupied.contains(next) && next != tail) return;
            visited.insert(next);
            q.push(next);
        });
    }

    return count;
}

static std::size_t searchBudget(const SnakeState& state, int dims, bool tailSearch) {
    std::size_t budget = 3000u + static_cast<std::size_t>(dims) * 350u + state.body.size() * (tailSearch ? 6u : 4u);
    if (tailSearch) return std::clamp<std::size_t>(budget, 8000u, 50000u);
    return std::clamp<std::size_t>(budget, 5000u, 30000u);
}

static bool hasTailEscape(const Problem& problem, const SnakeState& state) {
    if (state.body.size() <= 1) return true;

    const Id head = state.body.front();
    const Id tail = state.body.back();
    auto path = findPath(problem, state, head, tail, searchBudget(state, problem.board.dims, true), true);
    if (path) return true;

    std::size_t cap = std::min<std::size_t>(5000u, state.body.size() + 64u);
    return reachableSpaceEstimate(problem, state, head, cap) >= state.body.size();
}

static int nearestFoodDistance(const Problem& problem, const SnakeState& state) {
    if (state.foods.empty()) return 0;
    Id head = state.body.front();
    int best = std::numeric_limits<int>::max();
    for (Id food : state.foods) {
        best = std::min(best, manhattan(head, food, problem.board));
    }
    return best;
}

static std::vector<Id> nearestFoods(const Problem& problem, const SnakeState& state, std::size_t limit) {
    std::vector<std::pair<int, Id>> ranked;
    ranked.reserve(state.foods.size());

    const Id head = state.body.front();
    for (Id food : state.foods) {
        ranked.push_back({manhattan(head, food, problem.board), food});
    }

    if (ranked.size() > limit) {
        std::nth_element(ranked.begin(), ranked.begin() + static_cast<std::ptrdiff_t>(limit), ranked.end(),
                         [](const auto& lhs, const auto& rhs) {
                             if (lhs.first != rhs.first) return lhs.first < rhs.first;
                             return lhs.second < rhs.second;
                         });
        ranked.resize(limit);
    }

    std::sort(ranked.begin(), ranked.end(), [](const auto& lhs, const auto& rhs) {
        if (lhs.first != rhs.first) return lhs.first < rhs.first;
        return lhs.second < rhs.second;
    });

    std::vector<Id> foods;
    foods.reserve(ranked.size());
    for (const auto& item : ranked) foods.push_back(item.second);
    return foods;
}

static std::optional<std::vector<Move>> findSafeFoodPath(const Problem& problem, const SnakeState& state) {
    if (state.foods.empty()) return std::nullopt;

    const Id head = state.body.front();
    const std::size_t budget = searchBudget(state, problem.board.dims, false);
    const std::size_t candidateCount = std::min<std::size_t>(state.foods.size(), 8u + static_cast<std::size_t>(problem.board.dims / 2));

    for (Id food : nearestFoods(problem, state, candidateCount)) {
        auto path = findPath(problem, state, head, food, budget, false);
        if (!path || path->empty()) continue;

        bool ateAnyFood = false;
        auto simulated = simulateMoves(problem, state, *path, &ateAnyFood);
        if (!simulated || !ateAnyFood) continue;

        if (simulated->foods.empty() || hasTailEscape(problem, *simulated)) {
            return path;
        }
    }

    return std::nullopt;
}

static std::optional<std::vector<Move>> findSafeTailPath(const Problem& problem, const SnakeState& state) {
    if (state.body.size() <= 1) return std::nullopt;

    Id head = state.body.front();
    Id tail = state.body.back();
    auto path = findPath(problem, state, head, tail, searchBudget(state, problem.board.dims, true), true);
    if (!path || path->empty()) return std::nullopt;

    auto simulated = simulateMoves(problem, state, *path, nullptr);
    if (!simulated) return std::nullopt;

    return path;
}

static std::vector<Move> legalMoves(const Problem& problem, const SnakeState& state) {
    std::vector<Move> moves;
    moves.reserve(problem.board.dims * 2);

    const Id head = state.body.front();
    const Id tail = state.body.back();
    auto point = decode(head, problem.board);

    for (int axis = 0; axis < problem.board.dims; ++axis) {
        for (int dir : {-1, +1}) {
            if (dir > 0 && point[axis] + 1 >= problem.board.bounds[axis]) continue;
            if (dir < 0 && point[axis] == 0) continue;

            Id next = dir > 0 ? head + problem.board.stride[axis] : head - problem.board.stride[axis];
            if (problem.obstacles.contains(next)) continue;

            bool eat = state.foods.contains(next);
            if (state.occupied.contains(next) && !(next == tail && !eat)) continue;

            moves.push_back(Move{axis, dir});
        }
    }

    return moves;
}

static std::optional<Move> chooseEmergencyMove(const Problem& problem, const SnakeState& state) {
    auto candidates = legalMoves(problem, state);
    if (candidates.empty()) return std::nullopt;

    double bestScore = -1e300;
    Move best = candidates.front();

    for (const Move& move : candidates) {
        SnakeState next = state;
        bool ate = false;
        if (!applyMove(problem, next, move, &ate)) continue;

        bool tailEscape = hasTailEscape(problem, next);
        std::size_t cap = std::min<std::size_t>(5000u, next.body.size() * 4u + 128u);
        std::size_t space = reachableSpaceEstimate(problem, next, next.body.front(), cap);
        int foodDist = nearestFoodDistance(problem, next);

        double score = 0.0;
        if (tailEscape) score += 1.0e12;
        if (ate) score += 1.0e9;
        score += static_cast<double>(space) * 1000.0;
        score -= static_cast<double>(foodDist) * 10.0;

        if (score > bestScore) {
            bestScore = score;
            best = move;
        }
    }

    return best;
}

static int localFreeDegree(const Problem& problem, Id cell, const std::unordered_set<Id>& used) {
    int degree = 0;
    forEachNeighbor(cell, problem.board, [&](Id next, int, int) {
        if (!problem.obstacles.contains(next) && !used.contains(next)) ++degree;
    });
    return degree;
}

// The prompt gives only the head position and initial length. We deterministically
// reconstruct a plausible body by preferring a straight segment and falling back to
// a degree-aware DFS path when the straight placement does not fit.
static std::deque<Id> reconstructInitialBody(const Problem& problem, Id head, int length) {
    length = std::max(length, 1);

    std::deque<Id> bestStraight;
    for (int axis = 0; axis < problem.board.dims; ++axis) {
        for (int dir : {-1, +1}) {
            std::deque<Id> candidate;
            candidate.push_back(head);
            auto point = decode(head, problem.board);
            Id cur = head;

            for (int step = 1; step < length; ++step) {
                if (dir > 0) {
                    if (point[axis] + 1 >= problem.board.bounds[axis]) break;
                    ++point[axis];
                    cur += problem.board.stride[axis];
                } else {
                    if (point[axis] == 0) break;
                    --point[axis];
                    cur -= problem.board.stride[axis];
                }
                if (problem.obstacles.contains(cur)) break;
                candidate.push_back(cur);
            }

            if (candidate.size() > bestStraight.size()) bestStraight = candidate;
            if (static_cast<int>(candidate.size()) == length) return candidate;
        }
    }

    std::vector<Id> path;
    std::unordered_set<Id> used;
    path.reserve(length);
    used.reserve(static_cast<std::size_t>(length) * 2u + 32u);
    path.push_back(head);
    used.insert(head);

    int expansions = 0;
    const int budget = 150000;

    auto dfs = [&](auto&& self) -> bool {
        if (static_cast<int>(path.size()) == length) return true;
        if (++expansions > budget) return false;

        Id cur = path.back();
        struct Choice {
            int degree;
            int heuristic;
            Id id;
        };

        std::vector<Choice> choices;
        choices.reserve(problem.board.dims * 2);

        forEachNeighbor(cur, problem.board, [&](Id next, int, int) {
            if (problem.obstacles.contains(next) || used.contains(next)) return;
            int degree = localFreeDegree(problem, next, used);
            choices.push_back(Choice{degree, manhattan(next, head, problem.board), next});
        });

        std::sort(choices.begin(), choices.end(), [](const Choice& lhs, const Choice& rhs) {
            if (lhs.degree != rhs.degree) return lhs.degree < rhs.degree;
            if (lhs.heuristic != rhs.heuristic) return lhs.heuristic > rhs.heuristic;
            return lhs.id < rhs.id;
        });

        for (const Choice& choice : choices) {
            path.push_back(choice.id);
            used.insert(choice.id);
            if (self(self)) return true;
            used.erase(choice.id);
            path.pop_back();
        }

        return false;
    };

    if (dfs(dfs)) {
        return std::deque<Id>(path.begin(), path.end());
    }

    if (!bestStraight.empty()) return bestStraight;
    return std::deque<Id>{head};
}

static std::optional<Move> chooseNextMove(const Problem& problem, const SnakeState& state) {
    if (auto foodPath = findSafeFoodPath(problem, state); foodPath && !foodPath->empty()) {
        return foodPath->front();
    }

    if (auto tailPath = findSafeTailPath(problem, state); tailPath && !tailPath->empty()) {
        return tailPath->front();
    }

    return chooseEmergencyMove(problem, state);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    Problem problem;
    if (!(std::cin >> problem.board.dims >> problem.maxTurns)) return 0;

    if (problem.board.dims <= 0 || problem.board.dims > MAX_D) return 0;

    for (int axis = 0; axis < problem.board.dims; ++axis) {
        std::cin >> problem.board.bounds[axis];
        if (problem.board.bounds[axis] <= 0) return 0;
    }

    problem.board.stride[0] = 1;
    for (int axis = 1; axis < problem.board.dims; ++axis) {
        problem.board.stride[axis] = problem.board.stride[axis - 1] * static_cast<Id>(problem.board.bounds[axis - 1]);
    }

    int snakeLength = 1;
    std::array<int, MAX_D> headPoint{};
    std::cin >> snakeLength;
    for (int axis = 0; axis < problem.board.dims; ++axis) {
        std::cin >> headPoint[axis];
    }

    int foodCount = 0;
    std::cin >> foodCount;
    std::vector<Id> initialFoods;
    initialFoods.reserve(static_cast<std::size_t>(std::max(foodCount, 0)));
    for (int i = 0; i < foodCount; ++i) {
        std::array<int, MAX_D> point{};
        for (int axis = 0; axis < problem.board.dims; ++axis) {
            std::cin >> point[axis];
        }
        initialFoods.push_back(encode(point, problem.board));
    }

    int obstacleCount = 0;
    std::cin >> obstacleCount;
    problem.obstacles.reserve(static_cast<std::size_t>(obstacleCount) * 2u + 32u);
    for (int i = 0; i < obstacleCount; ++i) {
        std::array<int, MAX_D> point{};
        for (int axis = 0; axis < problem.board.dims; ++axis) {
            std::cin >> point[axis];
        }
        problem.obstacles.insert(encode(point, problem.board));
    }

    SnakeState state;
    const Id head = encode(headPoint, problem.board);
    state.body = reconstructInitialBody(problem, head, snakeLength);
    state.occupied.insert(state.body.begin(), state.body.end());
    for (Id food : initialFoods) {
        if (!state.occupied.contains(food) && !problem.obstacles.contains(food)) {
            state.foods.insert(food);
        }
    }

    std::vector<Move> answer;
    answer.reserve(problem.maxTurns);

    for (int turn = 0; turn < problem.maxTurns && !state.foods.empty(); ++turn) {
        auto nextMove = chooseNextMove(problem, state);
        if (!nextMove) break;
        if (!applyMove(problem, state, *nextMove, nullptr)) break;
        answer.push_back(*nextMove);
    }

    for (const Move& move : answer) {
        std::cout << move.axis << ' ' << move.dir << '\n';
    }

    return 0;
}
