const fs = require('fs');

function solve(data) {
  const width = Math.max(0, Math.floor(Number(data && data.width) || 0));
  const height = Math.max(0, Math.floor(Number(data && data.height) || 0));
  const grid = Array.isArray(data && data.grid) ? data.grid : [];
  const cutBudget = Math.max(0, Math.floor(Number(data && data.cutBudget) || 0));
  const n = width * height;

  if (width <= 0 || height <= 0 || n <= 0 || cutBudget <= 0) {
    return { firebreaks: [] };
  }

  const burnable = new Uint8Array(n);
  const neighbors = new Int32Array(n * 4);
  neighbors.fill(-1);

  for (let y = 0; y < height; y++) {
    const row = typeof grid[y] === 'string' ? grid[y] : '';
    for (let x = 0; x < width; x++) {
      const idx = y * width + x;
      if (x > 0) neighbors[idx * 4] = idx - 1;
      if (x + 1 < width) neighbors[idx * 4 + 1] = idx + 1;
      if (y > 0) neighbors[idx * 4 + 2] = idx - width;
      if (y + 1 < height) neighbors[idx * 4 + 3] = idx + width;
      if (row.charCodeAt(x) === 46) {
        burnable[idx] = 1;
      }
    }
  }

  const townValueByCell = new Float64Array(n);
  const isTownCell = new Uint8Array(n);
  const townCells = [];
  const towns = Array.isArray(data && data.towns) ? data.towns : [];
  for (const town of towns) {
    if (!town || !Number.isFinite(town.x) || !Number.isFinite(town.y)) continue;
    const x = Math.floor(town.x);
    const y = Math.floor(town.y);
    if (x < 0 || x >= width || y < 0 || y >= height) continue;
    const idx = y * width + x;
    if (!burnable[idx]) continue;
    const value = Number(town.value);
    if (!(value > 0)) continue;
    townValueByCell[idx] += value;
    if (!isTownCell[idx]) {
      isTownCell[idx] = 1;
      townCells.push(idx);
    }
  }

  const ignitionSet = new Uint8Array(n);
  const ignitionCells = [];
  const ignitions = Array.isArray(data && data.ignitions) ? data.ignitions : [];
  for (const ignition of ignitions) {
    if (!Array.isArray(ignition) || ignition.length < 2) continue;
    const x = Math.floor(Number(ignition[0]));
    const y = Math.floor(Number(ignition[1]));
    if (x < 0 || x >= width || y < 0 || y >= height) continue;
    const idx = y * width + x;
    if (!burnable[idx] || ignitionSet[idx]) continue;
    ignitionSet[idx] = 1;
    ignitionCells.push(idx);
  }

  if (ignitionCells.length === 0 || townCells.length === 0) {
    return { firebreaks: [] };
  }

  const cut = new Uint8Array(n);
  const selectedCuts = [];

  const currentReach = new Uint8Array(n);
  const dist = new Int32Array(n);
  const parent = new Int32Array(n);
  const queue = new Int32Array(n);

  const atRiskStamp = new Int32Array(n);
  let atRiskToken = 1;

  const evalVisitStamp = new Int32Array(n);
  let evalVisitToken = 1;
  const extraBlockedStamp = new Int32Array(n);
  let extraBlockedToken = 1;

  const disc = new Int32Array(n);
  const low = new Int32Array(n);
  const dfsParent = new Int32Array(n);
  const nextDir = new Uint8Array(n);
  const townSub = new Float64Array(n);
  const ignSub = new Int32Array(n);
  const savedByArt = new Float64Array(n);
  const artTouchedStamp = new Int32Array(n);
  let artTouchedToken = 1;
  const dfsStack = new Int32Array(n);

  const townCompSeen = new Int32Array(n);
  let townCompToken = 1;
  const boundarySeen = new Int32Array(n);
  let boundaryToken = 1;

  const boxStamp = new Int32Array(n);
  let boxToken = 1;
  const localStamp = new Int32Array(n);
  const localId = new Int32Array(n);
  let localToken = 1;
  const outsideVisitStamp = new Int32Array(n);
  let outsideVisitToken = 1;
  const exposedStamp = new Int32Array(n);
  let exposedToken = 1;

  const INF = 1e9;
  const maxIterations = Math.min(20, Math.max(6, cutBudget));

  function isCuttableCell(idx) {
    return burnable[idx] && !cut[idx] && !isTownCell[idx] && !ignitionSet[idx];
  }

  function candidateScore(saved, cost) {
    return saved / Math.pow(cost, 0.65);
  }

  function compareCandidates(a, b) {
    if (a.score !== b.score) return b.score - a.score;
    if (a.saved !== b.saved) return b.saved - a.saved;
    if (a.cost !== b.cost) return a.cost - b.cost;
    return 0;
  }

  function makeCandidate(kind, cutsList, saved) {
    if (!(saved > 0)) return null;
    const cuts = [];
    for (const idx of cutsList) {
      if (isCuttableCell(idx)) cuts.push(idx);
    }
    if (cuts.length === 0) return null;
    return {
      kind,
      cuts,
      cost: cuts.length,
      saved,
      score: candidateScore(saved, cuts.length),
    };
  }

  function computeCurrentReach() {
    currentReach.fill(0);
    dist.fill(-1);
    parent.fill(-1);

    let head = 0;
    let tail = 0;
    for (const src of ignitionCells) {
      if (cut[src] || currentReach[src]) continue;
      currentReach[src] = 1;
      dist[src] = 0;
      queue[tail++] = src;
    }

    while (head < tail) {
      const u = queue[head++];
      const base = u * 4;
      const nextDist = dist[u] + 1;
      for (let k = 0; k < 4; k++) {
        const v = neighbors[base + k];
        if (v < 0 || !burnable[v] || cut[v] || currentReach[v]) continue;
        currentReach[v] = 1;
        dist[v] = nextDist;
        parent[v] = u;
        queue[tail++] = v;
      }
    }
    return tail;
  }

  function buildAtRiskList() {
    atRiskToken++;
    const atRiskCells = [];
    let totalAtRiskValue = 0;
    for (const idx of townCells) {
      if (!currentReach[idx]) continue;
      atRiskStamp[idx] = atRiskToken;
      atRiskCells.push(idx);
      totalAtRiskValue += townValueByCell[idx];
    }
    return { atRiskCells, totalAtRiskValue };
  }

  function simulateSavedValue(extraCuts, totalAtRiskValue) {
    extraBlockedToken++;
    for (const idx of extraCuts) {
      extraBlockedStamp[idx] = extraBlockedToken;
    }

    evalVisitToken++;
    let head = 0;
    let tail = 0;
    let burnedValue = 0;

    for (const src of ignitionCells) {
      if (cut[src] || extraBlockedStamp[src] === extraBlockedToken) continue;
      if (evalVisitStamp[src] === evalVisitToken) continue;
      evalVisitStamp[src] = evalVisitToken;
      queue[tail++] = src;
      if (atRiskStamp[src] === atRiskToken) burnedValue += townValueByCell[src];
    }

    while (head < tail) {
      const u = queue[head++];
      const base = u * 4;
      for (let k = 0; k < 4; k++) {
        const v = neighbors[base + k];
        if (
          v < 0 ||
          !burnable[v] ||
          cut[v] ||
          extraBlockedStamp[v] === extraBlockedToken ||
          evalVisitStamp[v] === evalVisitToken
        ) {
          continue;
        }
        evalVisitStamp[v] = evalVisitToken;
        queue[tail++] = v;
        if (atRiskStamp[v] === atRiskToken) burnedValue += townValueByCell[v];
      }
    }

    return totalAtRiskValue - burnedValue;
  }

  function buildArticulationCandidates() {
    disc.fill(0);
    dfsParent.fill(-1);
    artTouchedToken++;
    const touched = [];
    let time = 0;

    for (const root of ignitionCells) {
      if (!currentReach[root] || disc[root] !== 0) continue;
      disc[root] = low[root] = ++time;
      townSub[root] = townValueByCell[root];
      ignSub[root] = 1;
      nextDir[root] = 0;
      let top = 0;
      dfsStack[top++] = root;

      while (top > 0) {
        const u = dfsStack[top - 1];
        const dir = nextDir[u];
        if (dir < 4) {
          nextDir[u] = dir + 1;
          const v = neighbors[u * 4 + dir];
          if (v < 0 || !currentReach[v]) continue;
          if (disc[v] === 0) {
            dfsParent[v] = u;
            disc[v] = low[v] = ++time;
            townSub[v] = townValueByCell[v];
            ignSub[v] = ignitionSet[v] ? 1 : 0;
            nextDir[v] = 0;
            dfsStack[top++] = v;
          } else if (v !== dfsParent[u] && disc[v] < low[u]) {
            low[u] = disc[v];
          }
        } else {
          top--;
          const p = dfsParent[u];
          if (p !== -1) {
            if (low[u] < low[p]) low[p] = low[u];
            townSub[p] += townSub[u];
            ignSub[p] += ignSub[u];
            if (low[u] >= disc[p] && ignSub[u] === 0) {
              if (artTouchedStamp[p] !== artTouchedToken) {
                artTouchedStamp[p] = artTouchedToken;
                savedByArt[p] = 0;
                touched.push(p);
              }
              savedByArt[p] += townSub[u];
            }
          }
        }
      }
    }

    const candidates = [];
    for (const idx of touched) {
      const saved = savedByArt[idx];
      savedByArt[idx] = 0;
      if (!isCuttableCell(idx) || !(saved > 0)) continue;
      const candidate = makeCandidate('articulation', [idx], saved);
      if (candidate) candidates.push(candidate);
    }
    candidates.sort(compareCandidates);
    if (candidates.length > 12) candidates.length = 12;
    return candidates;
  }

  function buildRingCandidates(atRiskCells, remainingBudget) {
    townCompToken++;
    const candidates = [];

    for (const start of atRiskCells) {
      if (townCompSeen[start] === townCompToken) continue;
      boundaryToken++;
      const stack = [start];
      townCompSeen[start] = townCompToken;
      const cutsList = [];
      let savedValue = 0;
      let blockedByIgnition = ignitionSet[start] ? 1 : 0;

      while (stack.length > 0) {
        const u = stack.pop();
        if (atRiskStamp[u] === atRiskToken) savedValue += townValueByCell[u];
        const base = u * 4;
        for (let k = 0; k < 4; k++) {
          const v = neighbors[base + k];
          if (v < 0) continue;
          if (ignitionSet[v]) {
            blockedByIgnition = 1;
            continue;
          }
          if (!burnable[v] || cut[v]) continue;
          if (isTownCell[v]) {
            if (townCompSeen[v] !== townCompToken) {
              townCompSeen[v] = townCompToken;
              stack.push(v);
            }
          } else if (boundarySeen[v] !== boundaryToken) {
            boundarySeen[v] = boundaryToken;
            cutsList.push(v);
          }
        }
      }

      if (blockedByIgnition || !(savedValue > 0) || cutsList.length === 0 || cutsList.length > remainingBudget) {
        continue;
      }
      const candidate = makeCandidate('ring', cutsList, savedValue);
      if (candidate) candidates.push(candidate);
    }

    candidates.sort(compareCandidates);
    if (candidates.length > 12) candidates.length = 12;
    return candidates;
  }

  function idxToXY(idx) {
    const y = Math.floor(idx / width);
    return [idx - y * width, y];
  }

  function clampBox(x0, y0, x1, y1) {
    if (x0 < 0) x0 = 0;
    if (y0 < 0) y0 = 0;
    if (x1 >= width) x1 = width - 1;
    if (y1 >= height) y1 = height - 1;
    return { x0, y0, x1, y1 };
  }

  function squareBox(idx, radius) {
    const [x, y] = idxToXY(idx);
    return clampBox(x - radius, y - radius, x + radius, y + radius);
  }

  function pathMidBox(seedIdx, stepTowardSource, pad) {
    let cur = seedIdx;
    let steps = stepTowardSource;
    while (steps > 0 && parent[cur] !== -1) {
      cur = parent[cur];
      steps--;
    }
    const [sx, sy] = idxToXY(seedIdx);
    const [mx, my] = idxToXY(cur);
    return clampBox(
      Math.min(sx, mx) - pad,
      Math.min(sy, my) - pad,
      Math.max(sx, mx) + pad,
      Math.max(sy, my) + pad
    );
  }

  function boxKey(box) {
    return `${box.x0},${box.y0},${box.x1},${box.y1}`;
  }

  function addEdge(graph, u, v, cap) {
    const a = { to: v, rev: graph[v].length, cap };
    const b = { to: u, rev: graph[u].length, cap: 0 };
    graph[u].push(a);
    graph[v].push(b);
  }

  function limitedMaxFlow(graph, source, sink, limit) {
    const nodeCount = graph.length;
    const parentNode = new Int32Array(nodeCount);
    const parentEdge = new Int32Array(nodeCount);
    const localQueue = new Int32Array(nodeCount);
    let flow = 0;

    while (flow <= limit) {
      parentNode.fill(-1);
      let head = 0;
      let tail = 0;
      parentNode[source] = source;
      localQueue[tail++] = source;

      while (head < tail && parentNode[sink] === -1) {
        const u = localQueue[head++];
        const edges = graph[u];
        for (let i = 0; i < edges.length; i++) {
          const e = edges[i];
          if (e.cap <= 0 || parentNode[e.to] !== -1) continue;
          parentNode[e.to] = u;
          parentEdge[e.to] = i;
          localQueue[tail++] = e.to;
          if (e.to === sink) break;
        }
      }

      if (parentNode[sink] === -1) break;

      let add = limit + 1 - flow;
      let v = sink;
      while (v !== source) {
        const u = parentNode[v];
        const e = graph[u][parentEdge[v]];
        if (e.cap < add) add = e.cap;
        v = u;
      }

      v = sink;
      while (v !== source) {
        const u = parentNode[v];
        const ei = parentEdge[v];
        const e = graph[u][ei];
        e.cap -= add;
        graph[v][e.rev].cap += add;
        v = u;
      }

      flow += add;
      if (flow > limit) return { flow, reachable: null };
    }

    const reachable = new Uint8Array(nodeCount);
    let head = 0;
    let tail = 0;
    localQueue[tail++] = source;
    reachable[source] = 1;

    while (head < tail) {
      const u = localQueue[head++];
      for (const e of graph[u]) {
        if (e.cap <= 0 || reachable[e.to]) continue;
        reachable[e.to] = 1;
        localQueue[tail++] = e.to;
      }
    }

    return { flow, reachable };
  }

  function buildMinCutCandidate(box, atRiskCells, remainingBudget, totalAtRiskValue) {
    boxToken++;
    const cells = [];
    const availableInside = [];

    for (let y = box.y0; y <= box.y1; y++) {
      let idx = y * width + box.x0;
      for (let x = box.x0; x <= box.x1; x++, idx++) {
        boxStamp[idx] = boxToken;
        if (burnable[idx] && !cut[idx]) availableInside.push(idx);
      }
    }

    const maxBoxCells = n <= 50000 ? 7000 : 5000;
    if (availableInside.length === 0 || availableInside.length > maxBoxCells) return null;

    const targetCells = [];
    let targetValue = 0;
    for (const idx of atRiskCells) {
      if (boxStamp[idx] !== boxToken || cut[idx]) continue;
      targetCells.push(idx);
      targetValue += townValueByCell[idx];
    }
    if (targetCells.length === 0 || !(targetValue > 0)) return null;

    exposedToken++;
    const exposedCells = [];
    for (const src of ignitionCells) {
      if (cut[src]) continue;
      if (boxStamp[src] === boxToken && exposedStamp[src] !== exposedToken) {
        exposedStamp[src] = exposedToken;
        exposedCells.push(src);
      }
    }

    outsideVisitToken++;
    let head = 0;
    let tail = 0;
    for (const src of ignitionCells) {
      if (cut[src] || boxStamp[src] === boxToken || outsideVisitStamp[src] === outsideVisitToken) continue;
      outsideVisitStamp[src] = outsideVisitToken;
      queue[tail++] = src;
    }

    while (head < tail) {
      const u = queue[head++];
      const base = u * 4;
      for (let k = 0; k < 4; k++) {
        const v = neighbors[base + k];
        if (v < 0 || !burnable[v] || cut[v]) continue;
        if (boxStamp[v] === boxToken) {
          if (exposedStamp[v] !== exposedToken) {
            exposedStamp[v] = exposedToken;
            exposedCells.push(v);
          }
        } else if (outsideVisitStamp[v] !== outsideVisitToken) {
          outsideVisitStamp[v] = outsideVisitToken;
          queue[tail++] = v;
        }
      }
    }

    if (exposedCells.length === 0) return null;

    localToken++;
    for (const idx of availableInside) {
      localStamp[idx] = localToken;
      localId[idx] = cells.length;
      cells.push(idx);
    }

    const nodeCount = cells.length * 2 + 2;
    const source = nodeCount - 2;
    const sink = nodeCount - 1;
    const graph = Array.from({ length: nodeCount }, () => []);

    for (let i = 0; i < cells.length; i++) {
      const idx = cells[i];
      const inNode = i * 2;
      const outNode = inNode + 1;
      const cap = isTownCell[idx] || ignitionSet[idx] ? INF : 1;
      addEdge(graph, inNode, outNode, cap);
      if (exposedStamp[idx] === exposedToken) addEdge(graph, source, inNode, INF);
      if (atRiskStamp[idx] === atRiskToken) addEdge(graph, outNode, sink, INF);
    }

    for (let i = 0; i < cells.length; i++) {
      const idx = cells[i];
      const outNode = i * 2 + 1;
      const base = idx * 4;
      for (let k = 0; k < 4; k++) {
        const v = neighbors[base + k];
        if (v < 0 || localStamp[v] !== localToken) continue;
        addEdge(graph, outNode, localId[v] * 2, INF);
      }
    }

    const result = limitedMaxFlow(graph, source, sink, remainingBudget);
    if (!(result.flow > 0) || result.flow > remainingBudget) return null;

    const cutsList = [];
    for (let i = 0; i < cells.length; i++) {
      const idx = cells[i];
      if (!isCuttableCell(idx)) continue;
      if (result.reachable[i * 2] && !result.reachable[i * 2 + 1]) {
        cutsList.push(idx);
      }
    }
    if (cutsList.length === 0 || cutsList.length > remainingBudget) return null;

    const saved = simulateSavedValue(cutsList, totalAtRiskValue);
    return makeCandidate('mincut', cutsList, saved);
  }

  function buildMinCutCandidates(atRiskCells, totalAtRiskValue, remainingBudget) {
    const seedLimit = n < 50000 ? 8 : n < 200000 ? 6 : 4;
    const seeds = atRiskCells
      .slice()
      .sort((a, b) => {
        const pa = townValueByCell[a] / (dist[a] + 1);
        const pb = townValueByCell[b] / (dist[b] + 1);
        if (pa !== pb) return pb - pa;
        return townValueByCell[b] - townValueByCell[a];
      })
      .slice(0, seedLimit);

    const candidates = [];
    const seenBoxes = new Set();
    const baseRadius = Math.min(n < 50000 ? 60 : 48, Math.max(10, remainingBudget * 2 + 8));
    const maxBoxArea = n < 50000 ? 12000 : 9000;

    for (const seed of seeds) {
      const firstBox = squareBox(seed, baseRadius);
      let key = boxKey(firstBox);
      if (!seenBoxes.has(key)) {
        seenBoxes.add(key);
        const area = (firstBox.x1 - firstBox.x0 + 1) * (firstBox.y1 - firstBox.y0 + 1);
        if (area <= maxBoxArea) {
          const candidate = buildMinCutCandidate(firstBox, atRiskCells, remainingBudget, totalAtRiskValue);
          if (candidate) candidates.push(candidate);
        }
      }

      if (dist[seed] > baseRadius + 8) {
        const stepTowardSource = Math.floor(dist[seed] * 0.5);
        const pad = Math.max(6, Math.min(remainingBudget + 4, 18));
        const secondBox = pathMidBox(seed, stepTowardSource, pad);
        key = boxKey(secondBox);
        if (!seenBoxes.has(key)) {
          seenBoxes.add(key);
          const area = (secondBox.x1 - secondBox.x0 + 1) * (secondBox.y1 - secondBox.y0 + 1);
          if (area <= maxBoxArea) {
            const candidate = buildMinCutCandidate(secondBox, atRiskCells, remainingBudget, totalAtRiskValue);
            if (candidate) candidates.push(candidate);
          }
        }
      }
    }

    candidates.sort(compareCandidates);
    if (candidates.length > 8) candidates.length = 8;
    return candidates;
  }

  function applyCuts(cutsList) {
    for (const idx of cutsList) {
      if (!isCuttableCell(idx)) continue;
      cut[idx] = 1;
      selectedCuts.push(idx);
    }
  }

  let remainingBudget = cutBudget;

  for (let iter = 0; iter < maxIterations && remainingBudget > 0; iter++) {
    computeCurrentReach();
    const { atRiskCells, totalAtRiskValue } = buildAtRiskList();
    if (atRiskCells.length === 0 || !(totalAtRiskValue > 0)) break;

    const candidates = [];
    candidates.push(...buildArticulationCandidates());
    candidates.push(...buildRingCandidates(atRiskCells, remainingBudget));
    if (remainingBudget > 1) {
      candidates.push(...buildMinCutCandidates(atRiskCells, totalAtRiskValue, remainingBudget));
    }

    if (candidates.length === 0) break;
    candidates.sort(compareCandidates);
    const best = candidates[0];
    if (!best || best.cost > remainingBudget || !(best.saved > 0)) break;

    applyCuts(best.cuts);
    remainingBudget -= best.cost;
  }

  if (remainingBudget > 0) {
    computeCurrentReach();
    const { atRiskCells } = buildAtRiskList();
    const rings = buildRingCandidates(atRiskCells, remainingBudget);
    rings.sort(compareCandidates);
    for (const ring of rings) {
      if (remainingBudget <= 0) break;
      const freshCuts = [];
      for (const idx of ring.cuts) {
        if (isCuttableCell(idx)) freshCuts.push(idx);
      }
      if (freshCuts.length === 0 || freshCuts.length > remainingBudget) continue;
      applyCuts(freshCuts);
      remainingBudget -= freshCuts.length;
    }
  }

  selectedCuts.sort((a, b) => a - b);
  const firebreaks = selectedCuts.map((idx) => {
    const y = Math.floor(idx / width);
    return [idx - y * width, y];
  });
  return { firebreaks };
}

const input = fs.readFileSync(0, 'utf8');
const data = input.trim() ? JSON.parse(input) : {};
const result = solve(data);
process.stdout.write(JSON.stringify(result));
