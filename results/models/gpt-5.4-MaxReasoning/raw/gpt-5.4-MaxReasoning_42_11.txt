OpCapability Shader
OpCapability Geometry
%GLSL450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %arr_v3_in ArrayStride 16
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpDecorate %outColor Location 0
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_n1 = OpConstant %float -1
%float_1 = OpConstant %float 1
%float_0_02 = OpConstant %float 0.02
%float_0_33333334 = OpConstant %float 0.33333334
%float_0_66666669 = OpConstant %float 0.66666669
%float_0_000001 = OpConstant %float 0.000001
%neg_y_pos_x = OpConstantComposite %v2float %float_n1 %float_1
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%fn = OpTypeFunction %void
%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output
%main = OpFunction %void None %fn
%entry = OpLabel
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr
%p3 = OpLoad %v4float %p3_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c1 = OpLoad %v3float %c1_ptr
%c2 = OpLoad %v3float %c2_ptr
%p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p2xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p3xy = OpVectorShuffle %v2float %p3 %p3 0 1
%d0 = OpFSub %v2float %p1xy %p0xy
%d1 = OpFSub %v2float %p2xy %p1xy
%d2 = OpFSub %v2float %p3xy %p2xy
%dir0 = OpExtInst %v2float %GLSL450 Normalize %d0
%dir1 = OpExtInst %v2float %GLSL450 Normalize %d1
%dir2 = OpExtInst %v2float %GLSL450 Normalize %d2
%tan_start_sum = OpFAdd %v2float %dir0 %dir1
%tan_end_sum = OpFAdd %v2float %dir1 %dir2
%tan_start = OpExtInst %v2float %GLSL450 Normalize %tan_start_sum
%tan_end = OpExtInst %v2float %GLSL450 Normalize %tan_end_sum
%dir1_yx = OpVectorShuffle %v2float %dir1 %dir1 1 0
%seg_perp = OpFMul %v2float %dir1_yx %neg_y_pos_x
%tan_start_yx = OpVectorShuffle %v2float %tan_start %tan_start 1 0
%tan_end_yx = OpVectorShuffle %v2float %tan_end %tan_end 1 0
%perp_start_unit = OpFMul %v2float %tan_start_yx %neg_y_pos_x
%perp_end_unit = OpFMul %v2float %tan_end_yx %neg_y_pos_x
%start_dot = OpDot %float %perp_start_unit %seg_perp
%end_dot = OpDot %float %perp_end_unit %seg_perp
%start_abs = OpExtInst %float %GLSL450 FAbs %start_dot
%end_abs = OpExtInst %float %GLSL450 FAbs %end_dot
%start_denom = OpExtInst %float %GLSL450 FMax %start_abs %float_0_000001
%end_denom = OpExtInst %float %GLSL450 FMax %end_abs %float_0_000001
%start_scale = OpFDiv %float %float_0_02 %start_denom
%end_scale = OpFDiv %float %float_0_02 %end_denom
%offset_start = OpVectorTimesScalar %v2float %perp_start_unit %start_scale
%offset_end = OpVectorTimesScalar %v2float %perp_end_unit %end_scale
%delta_pos = OpFSub %v4float %p2 %p1
%delta_off = OpFSub %v2float %offset_end %offset_start
%delta_col = OpFSub %v3float %c2 %c1
%pos_1_3_delta = OpVectorTimesScalar %v4float %delta_pos %float_0_33333334
%pos_2_3_delta = OpVectorTimesScalar %v4float %delta_pos %float_0_66666669
%base_1_3 = OpFAdd %v4float %p1 %pos_1_3_delta
%base_2_3 = OpFAdd %v4float %p1 %pos_2_3_delta
%off_1_3_delta = OpVectorTimesScalar %v2float %delta_off %float_0_33333334
%off_2_3_delta = OpVectorTimesScalar %v2float %delta_off %float_0_66666669
%offset_1_3 = OpFAdd %v2float %offset_start %off_1_3_delta
%offset_2_3 = OpFAdd %v2float %offset_start %off_2_3_delta
%col_1_3_delta = OpVectorTimesScalar %v3float %delta_col %float_0_33333334
%col_2_3_delta = OpVectorTimesScalar %v3float %delta_col %float_0_66666669
%color_1_3 = OpFAdd %v3float %c1 %col_1_3_delta
%color_2_3 = OpFAdd %v3float %c1 %col_2_3_delta
%p1zw = OpVectorShuffle %v2float %p1 %p1 2 3
%p1_plus_xy = OpFAdd %v2float %p1xy %offset_start
%p1_minus_xy = OpFSub %v2float %p1xy %offset_start
%p1_plus = OpVectorShuffle %v4float %p1_plus_xy %p1zw 0 1 2 3
%p1_minus = OpVectorShuffle %v4float %p1_minus_xy %p1zw 0 1 2 3
%base_1_3_xy = OpVectorShuffle %v2float %base_1_3 %base_1_3 0 1
%base_1_3_zw = OpVectorShuffle %v2float %base_1_3 %base_1_3 2 3
%base_1_3_plus_xy = OpFAdd %v2float %base_1_3_xy %offset_1_3
%base_1_3_minus_xy = OpFSub %v2float %base_1_3_xy %offset_1_3
%pos_1_3_plus = OpVectorShuffle %v4float %base_1_3_plus_xy %base_1_3_zw 0 1 2 3
%pos_1_3_minus = OpVectorShuffle %v4float %base_1_3_minus_xy %base_1_3_zw 0 1 2 3
%base_2_3_xy = OpVectorShuffle %v2float %base_2_3 %base_2_3 0 1
%base_2_3_zw = OpVectorShuffle %v2float %base_2_3 %base_2_3 2 3
%base_2_3_plus_xy = OpFAdd %v2float %base_2_3_xy %offset_2_3
%base_2_3_minus_xy = OpFSub %v2float %base_2_3_xy %offset_2_3
%pos_2_3_plus = OpVectorShuffle %v4float %base_2_3_plus_xy %base_2_3_zw 0 1 2 3
%pos_2_3_minus = OpVectorShuffle %v4float %base_2_3_minus_xy %base_2_3_zw 0 1 2 3
%p2zw = OpVectorShuffle %v2float %p2 %p2 2 3
%p2_plus_xy = OpFAdd %v2float %p2xy %offset_end
%p2_minus_xy = OpFSub %v2float %p2xy %offset_end
%p2_plus = OpVectorShuffle %v4float %p2_plus_xy %p2zw 0 1 2 3
%p2_minus = OpVectorShuffle %v4float %p2_minus_xy %p2zw 0 1 2 3
OpStore %out_pos_ptr %p1_plus
OpStore %outColor %c1
OpEmitVertex
OpStore %out_pos_ptr %p1_minus
OpStore %outColor %c1
OpEmitVertex
OpStore %out_pos_ptr %pos_1_3_plus
OpStore %outColor %color_1_3
OpEmitVertex
OpStore %out_pos_ptr %pos_1_3_minus
OpStore %outColor %color_1_3
OpEmitVertex
OpEndPrimitive
OpStore %out_pos_ptr %pos_2_3_plus
OpStore %outColor %color_2_3
OpEmitVertex
OpStore %out_pos_ptr %pos_2_3_minus
OpStore %outColor %color_2_3
OpEmitVertex
OpStore %out_pos_ptr %p2_plus
OpStore %outColor %c2
OpEmitVertex
OpStore %out_pos_ptr %p2_minus
OpStore %outColor %c2
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
