#include <algorithm>
#include <array>
#include <chrono>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <random>
#include <utility>
#include <vector>

struct Graph {
    int n = 0;
    std::vector<int> degree;
    std::vector<std::size_t> offset;
    std::vector<int> adj;
    bool has_self_loop = false;
};

static inline int popcount64(std::uint64_t x) {
    int count = 0;
    while (x != 0) {
        x &= (x - 1);
        ++count;
    }
    return count;
}

static inline int ctz64(std::uint64_t x) {
    int bit = 0;
    while ((x & 1ULL) == 0ULL) {
        x >>= 1U;
        ++bit;
    }
    return bit;
}

static inline std::uint64_t full_mask_for(int k) {
    if (k <= 0) {
        return 0ULL;
    }
    if (k >= 64) {
        return ~0ULL;
    }
    return (1ULL << k) - 1ULL;
}

class Solver {
public:
    using Clock = std::chrono::steady_clock;

    Solver(Graph graph, int k)
        : g_(std::move(graph)),
          k_(k),
          full_mask_(full_mask_for(k)),
          rng_(static_cast<std::uint64_t>(
              std::chrono::steady_clock::now().time_since_epoch().count())),
          start_(Clock::now()),
          deadline_(start_ + std::chrono::seconds(290)) {}

    std::vector<int> solve() {
        if (g_.n == 0) {
            return {};
        }

        if (k_ <= 0 || g_.has_self_loop) {
            return std::vector<int>(g_.n, 0);
        }

        if (k_ == 1) {
            return std::vector<int>(g_.n, 0);
        }

        if (k_ > 62) {
            std::vector<int> color = solve_large_k_fallback();
            ensure_exact_k(color);
            return color;
        }

        std::vector<int> color;
        const double avg_degree =
            g_.n == 0 ? 0.0 : static_cast<double>(g_.adj.size()) / static_cast<double>(g_.n);

        if (g_.n <= 90) {
            const Clock::time_point local_deadline =
                std::min(deadline_, start_ + std::chrono::seconds(15));
            if (exact_dsatur(color, local_deadline)) {
                ensure_exact_k(color);
                if (validate(color)) {
                    return color;
                }
            }
        }

        if (g_.n <= 20000 || (g_.n <= 50000 && avg_degree >= 32.0)) {
            if (dsatur_greedy(color)) {
                ensure_exact_k(color);
                if (validate(color)) {
                    return color;
                }
            }
        }

        std::vector<int> degree_order = degree_order_desc();
        if (try_constructive_pass(degree_order, color)) {
            return color;
        }

        if (!time_up() && g_.n <= 250000 && g_.adj.size() <= static_cast<std::size_t>(10000000)) {
            std::vector<int> sl_order = smallest_last_order();
            std::reverse(sl_order.begin(), sl_order.end());
            if (try_constructive_pass(sl_order, color)) {
                return color;
            }
        }

        if (!time_up()) {
            std::vector<int> natural = natural_order();
            if (try_constructive_pass(natural, color)) {
                return color;
            }
        }

        for (int attempt = 0; attempt < 3 && !time_up(); ++attempt) {
            std::vector<int> order = degree_order;
            perturb_order(order);
            if (try_constructive_pass(order, color)) {
                return color;
            }
        }

        std::vector<int> fallback;
        greedy_color_order(degree_order, fallback, true);
        repair_min_conflicts(fallback);
        ensure_exact_k(fallback);
        if (validate(fallback)) {
            return fallback;
        }

        if (fallback.size() != static_cast<std::size_t>(g_.n)) {
            fallback.assign(g_.n, 0);
        }
        for (int i = 0; i < g_.n; ++i) {
            if (fallback[i] < 0) {
                fallback[i] = i % k_;
            } else {
                fallback[i] %= k_;
            }
        }
        return fallback;
    }

private:
    Graph g_;
    int k_;
    std::uint64_t full_mask_;
    std::mt19937_64 rng_;
    Clock::time_point start_;
    Clock::time_point deadline_;

    bool time_up() const {
        return Clock::now() >= deadline_;
    }

    std::uint64_t neighbor_mask(const std::vector<int>& color, int v) const {
        std::uint64_t mask = 0ULL;
        for (std::size_t i = g_.offset[v]; i < g_.offset[v + 1]; ++i) {
            const int c = color[g_.adj[i]];
            if (c >= 0) {
                mask |= (1ULL << c);
            }
        }
        return mask;
    }

    int choose_available_color(std::uint64_t avail, const std::vector<int>& color_use) const {
        int best = -1;
        int best_load = std::numeric_limits<int>::max();
        while (avail != 0ULL) {
            const int c = ctz64(avail);
            if (color_use[c] < best_load) {
                best = c;
                best_load = color_use[c];
            }
            avail &= (avail - 1ULL);
        }
        return best;
    }

    bool try_single_recolor(int v, std::vector<int>& color, std::vector<int>& color_use) const {
        std::array<int, 64> owner;
        std::array<unsigned char, 64> count{};
        owner.fill(-1);

        for (std::size_t i = g_.offset[v]; i < g_.offset[v + 1]; ++i) {
            const int u = g_.adj[i];
            const int c = color[u];
            if (c < 0) {
                continue;
            }
            if (count[c] == 0) {
                owner[c] = u;
                count[c] = 1;
            } else if (count[c] == 1) {
                owner[c] = -2;
                count[c] = 2;
            }
        }

        for (int c = 0; c < k_; ++c) {
            if (count[c] != 1 || owner[c] < 0) {
                continue;
            }
            const int u = owner[c];
            const std::uint64_t used_u = neighbor_mask(color, u);
            const std::uint64_t avail_u = full_mask_ & ~used_u & ~(1ULL << c);
            if (avail_u == 0ULL) {
                continue;
            }
            const int new_c = choose_available_color(avail_u, color_use);
            --color_use[color[u]];
            color[u] = new_c;
            ++color_use[new_c];
            color[v] = c;
            ++color_use[c];
            return true;
        }
        return false;
    }

    bool greedy_color_order(const std::vector<int>& order,
                            std::vector<int>& color,
                            bool allow_conflicts) const {
        if (k_ <= 0) {
            return false;
        }

        color.assign(g_.n, -1);
        std::vector<int> color_use(k_, 0);
        std::array<int, 64> counts{};

        for (int v : order) {
            const std::uint64_t used = neighbor_mask(color, v);
            const std::uint64_t avail = full_mask_ & ~used;
            if (avail != 0ULL) {
                const int c = choose_available_color(avail, color_use);
                color[v] = c;
                ++color_use[c];
                continue;
            }

            if (try_single_recolor(v, color, color_use)) {
                continue;
            }

            if (!allow_conflicts) {
                return false;
            }

            counts.fill(0);
            for (std::size_t i = g_.offset[v]; i < g_.offset[v + 1]; ++i) {
                const int u = g_.adj[i];
                if (color[u] >= 0) {
                    ++counts[color[u]];
                }
            }

            int best = 0;
            int best_score = counts[0];
            for (int c = 1; c < k_; ++c) {
                if (counts[c] < best_score ||
                    (counts[c] == best_score && color_use[c] < color_use[best])) {
                    best = c;
                    best_score = counts[c];
                }
            }
            color[v] = best;
            ++color_use[best];
        }

        return true;
    }

    bool validate(const std::vector<int>& color) const {
        if (color.size() != static_cast<std::size_t>(g_.n)) {
            return false;
        }
        for (int v = 0; v < g_.n; ++v) {
            if (color[v] < 0 || color[v] >= k_) {
                return false;
            }
            for (std::size_t i = g_.offset[v]; i < g_.offset[v + 1]; ++i) {
                const int u = g_.adj[i];
                if (u > v && color[u] == color[v]) {
                    return false;
                }
            }
        }
        return true;
    }

    void ensure_exact_k(std::vector<int>& color) const {
        if (k_ <= 0 || g_.n < k_ || color.size() != static_cast<std::size_t>(g_.n)) {
            return;
        }

        std::vector<int> representative(k_, -1);
        std::vector<int> movable;
        movable.reserve(g_.n);

        for (int v = 0; v < g_.n; ++v) {
            const int c = color[v];
            if (c < 0 || c >= k_) {
                return;
            }
            if (representative[c] == -1) {
                representative[c] = v;
            } else {
                movable.push_back(v);
            }
        }

        std::vector<int> missing;
        for (int c = 0; c < k_; ++c) {
            if (representative[c] == -1) {
                missing.push_back(c);
            }
        }

        if (missing.size() > movable.size()) {
            return;
        }

        for (std::size_t i = 0; i < missing.size(); ++i) {
            color[movable[i]] = missing[i];
        }
    }

    bool exact_dsatur(std::vector<int>& color, Clock::time_point local_deadline) {
        const int n = g_.n;
        color.assign(n, -1);
        if (n == 0) {
            return true;
        }

        std::vector<std::uint64_t> sat_mask(n, 0ULL);
        int seed = 0;
        for (int v = 1; v < n; ++v) {
            if (g_.degree[v] > g_.degree[seed]) {
                seed = v;
            }
        }

        color[seed] = 0;
        for (std::size_t i = g_.offset[seed]; i < g_.offset[seed + 1]; ++i) {
            sat_mask[g_.adj[i]] |= 1ULL;
        }

        auto dfs = [&](auto&& self, int colored, int used_colors) -> bool {
            if (Clock::now() >= local_deadline) {
                return false;
            }
            if (colored == n) {
                return true;
            }

            int best_v = -1;
            int best_sat = -1;
            int best_deg = -1;
            for (int v = 0; v < n; ++v) {
                if (color[v] != -1) {
                    continue;
                }
                const int sat = popcount64(sat_mask[v]);
                if (sat > best_sat || (sat == best_sat && g_.degree[v] > best_deg)) {
                    best_v = v;
                    best_sat = sat;
                    best_deg = g_.degree[v];
                }
            }

            if (best_v == -1) {
                return true;
            }

            const std::uint64_t avail = full_mask_ & ~sat_mask[best_v];
            if (avail == 0ULL) {
                return false;
            }

            std::array<int, 64> candidates{};
            int candidate_count = 0;

            if (used_colors > 0) {
                const std::uint64_t used_mask = (1ULL << used_colors) - 1ULL;
                std::uint64_t existing = avail & used_mask;
                while (existing != 0ULL) {
                    candidates[candidate_count++] = ctz64(existing);
                    existing &= (existing - 1ULL);
                }
            }
            if (used_colors < k_ && ((avail >> used_colors) & 1ULL) != 0ULL) {
                candidates[candidate_count++] = used_colors;
            }

            for (int idx = 0; idx < candidate_count; ++idx) {
                const int c = candidates[idx];
                color[best_v] = c;
                const std::uint64_t bit = 1ULL << c;
                std::vector<int> changed;
                changed.reserve(g_.degree[best_v]);
                for (std::size_t i = g_.offset[best_v]; i < g_.offset[best_v + 1]; ++i) {
                    const int u = g_.adj[i];
                    if (color[u] == -1 && (sat_mask[u] & bit) == 0ULL) {
                        sat_mask[u] |= bit;
                        changed.push_back(u);
                    }
                }

                const int next_used = std::max(used_colors, c + 1);
                if (self(self, colored + 1, next_used)) {
                    return true;
                }

                for (int u : changed) {
                    sat_mask[u] &= ~bit;
                }
                color[best_v] = -1;
            }
            return false;
        };

        if (dfs(dfs, 1, 1)) {
            return true;
        }
        color.clear();
        return false;
    }

    bool dsatur_greedy(std::vector<int>& color) const {
        struct State {
            int sat;
            int deg;
            int v;
        };

        auto cmp = [](const State& a, const State& b) {
            if (a.sat != b.sat) {
                return a.sat < b.sat;
            }
            if (a.deg != b.deg) {
                return a.deg < b.deg;
            }
            return a.v > b.v;
        };

        const int n = g_.n;
        color.assign(n, -1);
        std::vector<std::uint64_t> sat_mask(n, 0ULL);
        std::vector<int> sat_count(n, 0);
        std::vector<int> color_use(k_, 0);
        std::priority_queue<State, std::vector<State>, decltype(cmp)> pq(cmp);

        for (int v = 0; v < n; ++v) {
            pq.push(State{0, g_.degree[v], v});
        }

        for (int colored = 0; colored < n; ++colored) {
            if (time_up()) {
                return false;
            }

            int v = -1;
            while (!pq.empty()) {
                const State cur = pq.top();
                pq.pop();
                if (color[cur.v] == -1 && sat_count[cur.v] == cur.sat) {
                    v = cur.v;
                    break;
                }
            }
            if (v == -1) {
                return false;
            }

            const std::uint64_t avail = full_mask_ & ~sat_mask[v];
            if (avail == 0ULL) {
                return false;
            }

            const int c = choose_available_color(avail, color_use);
            color[v] = c;
            ++color_use[c];
            const std::uint64_t bit = 1ULL << c;

            for (std::size_t i = g_.offset[v]; i < g_.offset[v + 1]; ++i) {
                const int u = g_.adj[i];
                if (color[u] == -1 && (sat_mask[u] & bit) == 0ULL) {
                    sat_mask[u] |= bit;
                    ++sat_count[u];
                    pq.push(State{sat_count[u], g_.degree[u], u});
                }
            }
        }

        return true;
    }

    std::vector<int> degree_order_desc() const {
        std::vector<int> order(g_.n);
        std::iota(order.begin(), order.end(), 0);
        std::sort(order.begin(), order.end(), [&](int a, int b) {
            if (g_.degree[a] != g_.degree[b]) {
                return g_.degree[a] > g_.degree[b];
            }
            return a < b;
        });
        return order;
    }

    std::vector<int> natural_order() const {
        std::vector<int> order(g_.n);
        std::iota(order.begin(), order.end(), 0);
        return order;
    }

    std::vector<int> smallest_last_order() const {
        using Item = std::pair<int, int>;
        auto cmp = [](const Item& a, const Item& b) {
            if (a.first != b.first) {
                return a.first > b.first;
            }
            return a.second > b.second;
        };

        std::vector<int> order;
        order.reserve(g_.n);
        std::vector<int> current_degree = g_.degree;
        std::vector<unsigned char> removed(g_.n, 0);
        std::priority_queue<Item, std::vector<Item>, decltype(cmp)> pq(cmp);

        for (int v = 0; v < g_.n; ++v) {
            pq.push(Item{current_degree[v], v});
        }

        while (order.size() < static_cast<std::size_t>(g_.n)) {
            const Item cur = pq.top();
            pq.pop();
            const int deg = cur.first;
            const int v = cur.second;
            if (removed[v] != 0 || current_degree[v] != deg) {
                continue;
            }
            removed[v] = 1;
            order.push_back(v);
            for (std::size_t i = g_.offset[v]; i < g_.offset[v + 1]; ++i) {
                const int u = g_.adj[i];
                if (removed[u] == 0) {
                    --current_degree[u];
                    pq.push(Item{current_degree[u], u});
                }
            }
        }

        return order;
    }

    void perturb_order(std::vector<int>& order) {
        if (order.size() < 2U) {
            return;
        }
        const std::size_t swaps = std::min<std::size_t>(order.size() / 8U + 1U, 50000U);
        for (std::size_t i = 0; i < swaps; ++i) {
            const std::size_t a = static_cast<std::size_t>(rng_() % order.size());
            const std::size_t b = static_cast<std::size_t>(rng_() % order.size());
            std::swap(order[a], order[b]);
        }
    }

    bool repair_min_conflicts(std::vector<int>& color) {
        const int n = g_.n;
        if (color.size() != static_cast<std::size_t>(n)) {
            return false;
        }

        std::vector<int> same(n, 0);
        long long total_conflicts = 0;
        for (int v = 0; v < n; ++v) {
            for (std::size_t i = g_.offset[v]; i < g_.offset[v + 1]; ++i) {
                const int u = g_.adj[i];
                if (u > v && color[u] == color[v]) {
                    ++same[v];
                    ++same[u];
                    ++total_conflicts;
                }
            }
        }

        if (total_conflicts == 0) {
            return true;
        }

        std::vector<int> conflicted;
        conflicted.reserve(g_.n);
        for (int v = 0; v < n; ++v) {
            if (same[v] > 0) {
                conflicted.push_back(v);
            }
        }

        std::array<int, 64> counts{};
        long long stagnation = 0;
        const long long max_steps = std::max<long long>(50000LL, 20LL * n);

        for (long long step = 0; step < max_steps && !time_up(); ++step) {
            if (total_conflicts == 0) {
                return true;
            }
            if (conflicted.empty()) {
                for (int v = 0; v < n; ++v) {
                    if (same[v] > 0) {
                        conflicted.push_back(v);
                    }
                }
                if (conflicted.empty()) {
                    break;
                }
            }

            const int v = conflicted[static_cast<std::size_t>(rng_() % conflicted.size())];
            if (same[v] == 0) {
                continue;
            }

            counts.fill(0);
            for (std::size_t i = g_.offset[v]; i < g_.offset[v + 1]; ++i) {
                ++counts[color[g_.adj[i]]];
            }

            const int old_c = color[v];
            const int old_score = counts[old_c];

            std::array<int, 64> best_colors{};
            int best_count = 0;
            int best_score = std::numeric_limits<int>::max();
            for (int c = 0; c < k_; ++c) {
                if (counts[c] < best_score) {
                    best_score = counts[c];
                    best_count = 0;
                    best_colors[best_count++] = c;
                } else if (counts[c] == best_score) {
                    best_colors[best_count++] = c;
                }
            }

            int new_c = old_c;
            if (best_score < old_score) {
                std::array<int, 64> filtered{};
                int filtered_count = 0;
                for (int i = 0; i < best_count; ++i) {
                    if (best_colors[i] != old_c) {
                        filtered[filtered_count++] = best_colors[i];
                    }
                }
                if (filtered_count > 0) {
                    new_c = filtered[static_cast<std::size_t>(rng_() % filtered_count)];
                }
                stagnation = 0;
            } else {
                ++stagnation;
                std::array<int, 64> filtered{};
                int filtered_count = 0;
                for (int i = 0; i < best_count; ++i) {
                    if (best_colors[i] != old_c) {
                        filtered[filtered_count++] = best_colors[i];
                    }
                }
                if (filtered_count > 0 && (stagnation % 16LL == 0LL)) {
                    new_c = filtered[static_cast<std::size_t>(rng_() % filtered_count)];
                } else if (k_ > 1 && (stagnation % 512LL == 0LL)) {
                    do {
                        new_c = static_cast<int>(rng_() % k_);
                    } while (new_c == old_c);
                } else {
                    continue;
                }
            }

            if (new_c == old_c) {
                continue;
            }

            int old_conflicts = 0;
            int new_conflicts = 0;
            for (std::size_t i = g_.offset[v]; i < g_.offset[v + 1]; ++i) {
                const int u = g_.adj[i];
                if (color[u] == old_c) {
                    --same[u];
                    ++old_conflicts;
                    if (same[u] > 0) {
                        conflicted.push_back(u);
                    }
                } else if (color[u] == new_c) {
                    ++same[u];
                    ++new_conflicts;
                    conflicted.push_back(u);
                }
            }

            color[v] = new_c;
            same[v] = new_conflicts;
            total_conflicts += static_cast<long long>(new_conflicts) - old_conflicts;
            if (same[v] > 0) {
                conflicted.push_back(v);
            }

            if ((step & 4095LL) == 0LL &&
                conflicted.size() > static_cast<std::size_t>(6) * static_cast<std::size_t>(n)) {
                conflicted.clear();
                for (int u = 0; u < n; ++u) {
                    if (same[u] > 0) {
                        conflicted.push_back(u);
                    }
                }
            }
        }

        return total_conflicts == 0;
    }

    bool try_constructive_pass(const std::vector<int>& order, std::vector<int>& out) {
        if (k_ <= 0) {
            return false;
        }

        std::vector<int> color;
        if (greedy_color_order(order, color, false)) {
            ensure_exact_k(color);
            if (validate(color)) {
                out.swap(color);
                return true;
            }
        }

        if (!greedy_color_order(order, color, true)) {
            return false;
        }

        repair_min_conflicts(color);
        ensure_exact_k(color);
        if (validate(color)) {
            out.swap(color);
            return true;
        }
        return false;
    }

    std::vector<int> solve_large_k_fallback() const {
        std::vector<int> order = degree_order_desc();
        std::vector<int> color(g_.n, -1);
        std::vector<int> color_use(k_, 0);
        std::vector<int> seen(k_, -1);
        std::vector<int> count(k_, 0);
        std::vector<int> touched;
        touched.reserve(128);
        int stamp = 1;

        for (int v : order) {
            ++stamp;
            if (stamp == std::numeric_limits<int>::max()) {
                std::fill(seen.begin(), seen.end(), -1);
                stamp = 1;
            }

            touched.clear();
            for (std::size_t i = g_.offset[v]; i < g_.offset[v + 1]; ++i) {
                const int u = g_.adj[i];
                if (color[u] >= 0) {
                    const int c = color[u];
                    seen[c] = stamp;
                    if (count[c] == 0) {
                        touched.push_back(c);
                    }
                    ++count[c];
                }
            }

            int best = -1;
            int best_load = std::numeric_limits<int>::max();
            for (int c = 0; c < k_; ++c) {
                if (seen[c] != stamp && color_use[c] < best_load) {
                    best = c;
                    best_load = color_use[c];
                }
            }

            if (best == -1) {
                best = 0;
                int best_score = count[0];
                for (int c = 1; c < k_; ++c) {
                    if (count[c] < best_score ||
                        (count[c] == best_score && color_use[c] < color_use[best])) {
                        best = c;
                        best_score = count[c];
                    }
                }
            }

            color[v] = best;
            ++color_use[best];

            for (int c : touched) {
                count[c] = 0;
            }
        }

        return color;
    }
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n = 0;
    long long m = 0;
    int k = 0;
    if (!(std::cin >> n >> m >> k)) {
        return 0;
    }

    Graph g;
    g.n = n;
    g.degree.assign(n, 0);

    std::vector<int> eu;
    std::vector<int> ev;
    eu.reserve(static_cast<std::size_t>(m));
    ev.reserve(static_cast<std::size_t>(m));

    for (long long i = 0; i < m; ++i) {
        int u = 0;
        int v = 0;
        std::cin >> u >> v;
        eu.push_back(u);
        ev.push_back(v);
        if (u == v) {
            g.has_self_loop = true;
            continue;
        }
        if (u >= 0 && u < n && v >= 0 && v < n) {
            ++g.degree[u];
            ++g.degree[v];
        }
    }

    g.offset.assign(static_cast<std::size_t>(n) + 1U, 0U);
    for (int v = 0; v < n; ++v) {
        g.offset[static_cast<std::size_t>(v) + 1U] =
            g.offset[static_cast<std::size_t>(v)] + static_cast<std::size_t>(g.degree[v]);
    }

    g.adj.assign(g.offset.back(), 0);
    std::vector<std::size_t> next = g.offset;
    for (std::size_t i = 0; i < eu.size(); ++i) {
        const int u = eu[i];
        const int v = ev[i];
        if (u == v || u < 0 || u >= n || v < 0 || v >= n) {
            continue;
        }
        g.adj[next[u]++] = v;
        g.adj[next[v]++] = u;
    }

    Solver solver(std::move(g), k);
    std::vector<int> color = solver.solve();
    if (color.size() != static_cast<std::size_t>(n)) {
        color.assign(n, 0);
    }

    for (int i = 0; i < n; ++i) {
        if (i != 0) {
            std::cout << ' ';
        }
        const int value = (k > 0 && color[i] >= 0) ? (color[i] % k) : 0;
        std::cout << value;
    }
    std::cout << '\n';
    return 0;
}
