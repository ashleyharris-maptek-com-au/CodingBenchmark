using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

public static class Program
{
    private const double G = 6.67430e-11;
    private const double DaySeconds = 86400.0;
    private const double EarthRadius = 6_378_137.0;
    private const double RequiredClearance = EarthRadius + 1_000_000.0;
    private const double ImpactEventToleranceDays = 1e-6;
    private const double OutputBurnEpsilon = 1e-6;

    public static void Main()
    {
        try
        {
            Solve();
        }
        catch
        {
            Console.WriteLine("2");
            Console.WriteLine("0 0 0 0");
            Console.WriteLine("1 0 0 0");
        }
    }

    private static void Solve()
    {
        string input = Console.In.ReadToEnd();
        var reader = new TokenReader(input);

        int numBodies = reader.ReadInt();
        double warningDays = reader.ReadDouble();
        double impactorMass = reader.ReadDouble();
        double dvBudget = reader.ReadDouble();
        double asteroidMass = reader.ReadDouble();
        double t0UnixUtc = reader.ReadDouble();
        _ = t0UnixUtc;

        var bodies = new Body[numBodies];
        for (int i = 0; i < numBodies; i++)
        {
            string name = reader.ReadString();
            double mass = reader.ReadDouble();
            double x = reader.ReadDouble();
            double y = reader.ReadDouble();
            double z = reader.ReadDouble();
            double vx = reader.ReadDouble();
            double vy = reader.ReadDouble();
            double vz = reader.ReadDouble();
            bodies[i] = new Body(name, mass, new Vec3(x, y, z), new Vec3(vx, vy, vz));
        }

        string bspPath = reader.ReadRemaining().Trim();
        _ = bspPath;

        int earthIndex = FindBodyIndex(bodies, "Earth");
        int asteroidIndex = FindBodyIndex(bodies, "Asteroid");
        int sunIndex = FindSunIndex(bodies);

        if (earthIndex < 0 || asteroidIndex < 0 || sunIndex < 0)
        {
            WritePlan(BuildFallbackMissionSimple(warningDays, dvBudget));
            return;
        }

        double ephemerisEndDays = Math.Max(2.0, warningDays + 5.0);
        var ephemeris = BuildEphemeris(bodies, ephemerisEndDays);

        double nominalImpactDay = FindNominalImpactTime(ephemeris, earthIndex, asteroidIndex, warningDays);
        if (!IsFinite(nominalImpactDay) || nominalImpactDay <= 0.0)
        {
            nominalImpactDay = Math.Max(0.5, warningDays);
        }

        List<QuickCandidate> candidates = SearchCandidates(
            ephemeris,
            bodies,
            sunIndex,
            earthIndex,
            asteroidIndex,
            warningDays,
            nominalImpactDay,
            dvBudget,
            impactorMass,
            asteroidMass);

        MissionPlan bestPlan = null;
        foreach (QuickCandidate candidate in candidates)
        {
            MissionPlan plan = BuildMission(
                ephemeris,
                bodies,
                sunIndex,
                earthIndex,
                asteroidIndex,
                candidate,
                nominalImpactDay,
                warningDays,
                dvBudget,
                impactorMass,
                asteroidMass);

            if (plan == null)
            {
                continue;
            }

            if (bestPlan == null || plan.Score > bestPlan.Score)
            {
                bestPlan = plan;
            }
        }

        if (bestPlan == null)
        {
            bestPlan = BuildFallbackMission(
                ephemeris,
                earthIndex,
                asteroidIndex,
                nominalImpactDay,
                warningDays,
                dvBudget);
        }

        WritePlan(bestPlan);
    }

    private static int FindBodyIndex(Body[] bodies, string name)
    {
        for (int i = 0; i < bodies.Length; i++)
        {
            if (string.Equals(bodies[i].Name, name, StringComparison.OrdinalIgnoreCase))
            {
                return i;
            }
        }

        return -1;
    }

    private static int FindSunIndex(Body[] bodies)
    {
        int namedSun = FindBodyIndex(bodies, "Sun");
        if (namedSun >= 0)
        {
            return namedSun;
        }

        int index = -1;
        double maxMass = double.NegativeInfinity;
        for (int i = 0; i < bodies.Length; i++)
        {
            if (bodies[i].Mass > maxMass)
            {
                maxMass = bodies[i].Mass;
                index = i;
            }
        }

        return index;
    }

    private static Ephemeris BuildEphemeris(Body[] bodies, double endDays)
    {
        double stepDays = endDays <= 800.0 ? 0.125 : 0.25;
        double stepSeconds = stepDays * DaySeconds;
        int steps = Math.Max(8, (int)Math.Ceiling(endDays / stepDays) + 2);
        int bodyCount = bodies.Length;

        var positions = new Vec3[bodyCount][];
        var velocities = new Vec3[bodyCount][];
        var masses = new double[bodyCount];
        var currentPositions = new Vec3[bodyCount];
        var currentVelocities = new Vec3[bodyCount];

        for (int i = 0; i < bodyCount; i++)
        {
            positions[i] = new Vec3[steps + 1];
            velocities[i] = new Vec3[steps + 1];
            masses[i] = bodies[i].Mass;
            currentPositions[i] = bodies[i].Position0;
            currentVelocities[i] = bodies[i].Velocity0;
            positions[i][0] = currentPositions[i];
            velocities[i][0] = currentVelocities[i];
        }

        var accel = new Vec3[bodyCount];
        var accelNext = new Vec3[bodyCount];
        ComputeMutualAccelerations(currentPositions, masses, accel);

        for (int step = 1; step <= steps; step++)
        {
            var halfVelocities = new Vec3[bodyCount];
            for (int i = 0; i < bodyCount; i++)
            {
                halfVelocities[i] = currentVelocities[i] + accel[i] * (0.5 * stepSeconds);
                currentPositions[i] += halfVelocities[i] * stepSeconds;
            }

            ComputeMutualAccelerations(currentPositions, masses, accelNext);

            for (int i = 0; i < bodyCount; i++)
            {
                currentVelocities[i] = halfVelocities[i] + accelNext[i] * (0.5 * stepSeconds);
                positions[i][step] = currentPositions[i];
                velocities[i][step] = currentVelocities[i];
                accel[i] = accelNext[i];
            }
        }

        return new Ephemeris(bodies, positions, velocities, stepDays, steps);
    }

    private static void ComputeMutualAccelerations(Vec3[] positions, double[] masses, Vec3[] accelerations)
    {
        for (int i = 0; i < accelerations.Length; i++)
        {
            accelerations[i] = Vec3.Zero;
        }

        for (int i = 0; i < positions.Length; i++)
        {
            for (int j = i + 1; j < positions.Length; j++)
            {
                Vec3 delta = positions[j] - positions[i];
                double distanceSquared = delta.LengthSquared();
                if (distanceSquared < 1.0)
                {
                    distanceSquared = 1.0;
                }

                double distance = Math.Sqrt(distanceSquared);
                double invCube = 1.0 / (distanceSquared * distance);
                Vec3 directionScaled = delta * invCube;
                accelerations[i] += directionScaled * (G * masses[j]);
                accelerations[j] -= directionScaled * (G * masses[i]);
            }
        }
    }

    private static double FindNominalImpactTime(Ephemeris ephemeris, int earthIndex, int asteroidIndex, double warningDays)
    {
        int sampleCount = Math.Min(ephemeris.StepCount, Math.Max(2, (int)Math.Ceiling(warningDays / ephemeris.StepDays)));
        double bestDay = 0.0;
        double bestDistanceSq = double.PositiveInfinity;

        for (int i = 0; i <= sampleCount; i++)
        {
            double day = Math.Min(warningDays, i * ephemeris.StepDays);
            Vec3 earth = ephemeris.GetState(earthIndex, day).Position;
            Vec3 asteroid = ephemeris.GetState(asteroidIndex, day).Position;
            double d2 = (asteroid - earth).LengthSquared();
            if (d2 < bestDistanceSq)
            {
                bestDistanceSq = d2;
                bestDay = day;
            }
        }

        double window = Math.Max(ephemeris.StepDays * 2.0, 0.5);
        double left = Math.Max(0.0, bestDay - window);
        double right = Math.Min(warningDays, bestDay + window);

        for (int iter = 0; iter < 40; iter++)
        {
            double m1 = left + (right - left) / 3.0;
            double m2 = right - (right - left) / 3.0;
            double d1 = EarthAsteroidDistanceSquared(ephemeris, earthIndex, asteroidIndex, m1);
            double d2 = EarthAsteroidDistanceSquared(ephemeris, earthIndex, asteroidIndex, m2);
            if (d1 < d2)
            {
                right = m2;
            }
            else
            {
                left = m1;
            }
        }

        return 0.5 * (left + right);
    }

    private static double EarthAsteroidDistanceSquared(Ephemeris ephemeris, int earthIndex, int asteroidIndex, double day)
    {
        Vec3 earth = ephemeris.GetState(earthIndex, day).Position;
        Vec3 asteroid = ephemeris.GetState(asteroidIndex, day).Position;
        return (asteroid - earth).LengthSquared();
    }

    private static List<QuickCandidate> SearchCandidates(
        Ephemeris ephemeris,
        Body[] bodies,
        int sunIndex,
        int earthIndex,
        int asteroidIndex,
        double warningDays,
        double nominalImpactDay,
        double dvBudget,
        double impactorMass,
        double asteroidMass)
    {
        var all = new List<QuickCandidate>();
        double muSun = G * bodies[sunIndex].Mass;
        double latestIntercept = Math.Min(warningDays, nominalImpactDay) - 0.05;
        if (latestIntercept <= 0.5)
        {
            return all;
        }

        double minTof = Math.Max(1.0, Math.Min(10.0, latestIntercept / 50.0));
        int launchSamples = latestIntercept < 200.0 ? 18 : latestIntercept < 3000.0 ? 24 : 30;
        int impactSamples = latestIntercept < 200.0 ? 24 : latestIntercept < 3000.0 ? 32 : 40;

        double launchLatest = Math.Max(0.0, latestIntercept - minTof);
        for (int i = 0; i < launchSamples; i++)
        {
            double uLaunch = launchSamples == 1 ? 0.0 : (double)i / (launchSamples - 1);
            double launchDay = launchLatest * (0.35 * uLaunch + 0.65 * uLaunch * uLaunch);
            double localLatestIntercept = latestIntercept;
            if (localLatestIntercept - launchDay <= minTof)
            {
                continue;
            }

            for (int j = 0; j < impactSamples; j++)
            {
                double uImpact = impactSamples == 1 ? 0.0 : (double)j / (impactSamples - 1);
                double interceptDay = launchDay + minTof + (localLatestIntercept - launchDay - minTof) * uImpact;
                AddQuickCandidates(
                    all,
                    ephemeris,
                    sunIndex,
                    earthIndex,
                    asteroidIndex,
                    launchDay,
                    interceptDay,
                    nominalImpactDay,
                    dvBudget,
                    muSun,
                    impactorMass,
                    asteroidMass);
            }
        }

        all.Sort((a, b) => b.QuickScore.CompareTo(a.QuickScore));
        int seedCount = Math.Min(8, all.Count);
        double launchStep = launchSamples > 1 ? launchLatest / (launchSamples - 1) : 1.0;
        double interceptStep = impactSamples > 1 ? latestIntercept / (impactSamples - 1) : 1.0;

        for (int k = 0; k < seedCount; k++)
        {
            QuickCandidate seed = all[k];
            for (int dl = -1; dl <= 1; dl++)
            {
                for (int di = -1; di <= 1; di++)
                {
                    if (dl == 0 && di == 0)
                    {
                        continue;
                    }

                    double launchDay = Clamp(seed.LaunchDay + 0.5 * launchStep * dl, 0.0, launchLatest);
                    double interceptDay = Clamp(seed.InterceptDay + 0.5 * interceptStep * di, launchDay + minTof, latestIntercept);
                    AddQuickCandidates(
                        all,
                        ephemeris,
                        sunIndex,
                        earthIndex,
                        asteroidIndex,
                        launchDay,
                        interceptDay,
                        nominalImpactDay,
                        dvBudget,
                        muSun,
                        impactorMass,
                        asteroidMass);
                }
            }
        }

        all.Sort((a, b) => b.QuickScore.CompareTo(a.QuickScore));
        var distinct = new List<QuickCandidate>();
        foreach (QuickCandidate candidate in all)
        {
            bool duplicate = false;
            for (int i = 0; i < distinct.Count; i++)
            {
                if (Math.Abs(distinct[i].LaunchDay - candidate.LaunchDay) < 0.05 &&
                    Math.Abs(distinct[i].InterceptDay - candidate.InterceptDay) < 0.05)
                {
                    duplicate = true;
                    break;
                }
            }

            if (!duplicate)
            {
                distinct.Add(candidate);
                if (distinct.Count >= 24)
                {
                    break;
                }
            }
        }

        return distinct;
    }

    private static void AddQuickCandidates(
        List<QuickCandidate> all,
        Ephemeris ephemeris,
        int sunIndex,
        int earthIndex,
        int asteroidIndex,
        double launchDay,
        double interceptDay,
        double nominalImpactDay,
        double dvBudget,
        double muSun,
        double impactorMass,
        double asteroidMass)
    {
        if (interceptDay <= launchDay)
        {
            return;
        }

        State sunLaunch = ephemeris.GetState(sunIndex, launchDay);
        State sunIntercept = ephemeris.GetState(sunIndex, interceptDay);
        State earthLaunch = ephemeris.GetState(earthIndex, launchDay);
        State asteroidIntercept = ephemeris.GetState(asteroidIndex, interceptDay);

        Vec3 r1 = earthLaunch.Position - sunLaunch.Position;
        Vec3 r2 = asteroidIntercept.Position - sunIntercept.Position;
        double tofSeconds = (interceptDay - launchDay) * DaySeconds;

        Vec3 v1;
        Vec3 v2;
        if (TrySolveLambert(r1, r2, tofSeconds, muSun, false, out v1, out v2))
        {
            TryAddQuickCandidate(all, launchDay, interceptDay, earthLaunch, asteroidIntercept, sunLaunch, sunIntercept, v1, v2, nominalImpactDay, dvBudget, impactorMass, asteroidMass);
        }

        if (TrySolveLambert(r1, r2, tofSeconds, muSun, true, out v1, out v2))
        {
            TryAddQuickCandidate(all, launchDay, interceptDay, earthLaunch, asteroidIntercept, sunLaunch, sunIntercept, v1, v2, nominalImpactDay, dvBudget, impactorMass, asteroidMass);
        }
    }

    private static void TryAddQuickCandidate(
        List<QuickCandidate> all,
        double launchDay,
        double interceptDay,
        State earthLaunch,
        State asteroidIntercept,
        State sunLaunch,
        State sunIntercept,
        Vec3 lambertDepartureSun,
        Vec3 lambertArrivalSun,
        double nominalImpactDay,
        double dvBudget,
        double impactorMass,
        double asteroidMass)
    {
        Vec3 departureBary = sunLaunch.Velocity + lambertDepartureSun;
        Vec3 arrivalBary = sunIntercept.Velocity + lambertArrivalSun;
        Vec3 launchBurn = departureBary - earthLaunch.Velocity;
        double launchDv = launchBurn.Length();
        if (!IsFinite(launchDv) || launchDv > dvBudget * 0.95)
        {
            return;
        }

        Vec3 relativeImpactVelocity = arrivalBary - asteroidIntercept.Velocity;
        double relSpeed = relativeImpactVelocity.Length();
        double momentumRatio = impactorMass / Math.Max(1.0, asteroidMass + impactorMass);
        double leadSeconds = Math.Max(0.0, (nominalImpactDay - interceptDay) * DaySeconds);
        double approximateMissGain = momentumRatio * relSpeed * leadSeconds;
        if (!IsFinite(approximateMissGain) || approximateMissGain <= 0.0)
        {
            return;
        }

        double quickScore = approximateMissGain / (1.0 + launchDv / Math.Max(100.0, dvBudget));
        all.Add(new QuickCandidate(launchDay, interceptDay, launchBurn, quickScore));
    }

    private static MissionPlan BuildMission(
        Ephemeris ephemeris,
        Body[] bodies,
        int sunIndex,
        int earthIndex,
        int asteroidIndex,
        QuickCandidate candidate,
        double nominalImpactDay,
        double warningDays,
        double dvBudget,
        double impactorMass,
        double asteroidMass)
    {
        double launchDay = candidate.LaunchDay;
        double interceptDay = candidate.InterceptDay;
        if (interceptDay <= launchDay || interceptDay > nominalImpactDay || launchDay > warningDays)
        {
            return null;
        }

        State earthLaunch = ephemeris.GetState(earthIndex, launchDay);
        var burns = new List<Burn>();
        double ignoreEarth = earthIndex;
        double ignoreAsteroid = asteroidIndex;

        BurnSolveResult launchSolve = SolveBurnWithGuesses(
            ephemeris,
            earthLaunch,
            launchDay,
            interceptDay,
            earthIndex,
            asteroidIndex,
            new[] { candidate.LaunchBurnGuess, candidate.LaunchBurnGuess * 0.9, Vec3.Zero });

        if (!launchSolve.Success)
        {
            return null;
        }

        burns.Add(new Burn(launchDay, launchSolve.DeltaV));

        double tof = interceptDay - launchDay;
        var correctionTimes = new List<double>();
        if (tof > 10.0)
        {
            correctionTimes.Add(Math.Min(interceptDay - 1.0, launchDay + 0.60 * tof));
        }

        if (tof > 40.0)
        {
            double lateTime = Math.Min(interceptDay - 0.25, launchDay + 0.90 * tof);
            if (lateTime - correctionTimes[correctionTimes.Count - 1] > 0.2)
            {
                correctionTimes.Add(lateTime);
            }
        }

        State currentState = new State(earthLaunch.Position, earthLaunch.Velocity + launchSolve.DeltaV);
        double currentDay = launchDay;
        double totalDv = launchSolve.DeltaV.Length();

        foreach (double correctionDay in correctionTimes)
        {
            currentState = PropagateMassless(ephemeris, currentState, currentDay, correctionDay, earthIndex, asteroidIndex);
            BurnSolveResult correction = SolveBurnWithGuesses(
                ephemeris,
                currentState,
                correctionDay,
                interceptDay,
                earthIndex,
                asteroidIndex,
                new[] { Vec3.Zero, currentState.Velocity * 0.0 });

            if (!correction.Success)
            {
                return null;
            }

            burns.Add(new Burn(correctionDay, correction.DeltaV));
            currentState = new State(currentState.Position, currentState.Velocity + correction.DeltaV);
            currentDay = correctionDay;
            totalDv += correction.DeltaV.Length();
            if (totalDv > dvBudget * 1.0001)
            {
                return null;
            }
        }

        State spacecraftImpact = PropagateMassless(ephemeris, currentState, currentDay, interceptDay, earthIndex, asteroidIndex);
        State asteroidImpact = ephemeris.GetState(asteroidIndex, interceptDay);
        double interceptMiss = (spacecraftImpact.Position - asteroidImpact.Position).Length();
        if (!IsFinite(interceptMiss) || interceptMiss > 1_000_000.0)
        {
            return null;
        }

        Vec3 relativeVelocity = spacecraftImpact.Velocity - asteroidImpact.Velocity;
        double impactSpeed = relativeVelocity.Length();
        double momentumRatio = impactorMass / Math.Max(1.0, asteroidMass + impactorMass);
        Vec3 asteroidDeltaV = relativeVelocity * momentumRatio;

        double postImpactMiss = EvaluateDeflection(
            ephemeris,
            asteroidIndex,
            earthIndex,
            new State(asteroidImpact.Position, asteroidImpact.Velocity + asteroidDeltaV),
            interceptDay,
            nominalImpactDay);

        double score = postImpactMiss - 750.0 * totalDv - 5.0 * interceptMiss + 0.02 * impactSpeed * Math.Max(0.0, (nominalImpactDay - interceptDay) * DaySeconds * momentumRatio);
        if (postImpactMiss >= RequiredClearance)
        {
            score += 1.0e9;
        }

        return new MissionPlan(burns, interceptDay, totalDv, interceptMiss, postImpactMiss, score);
    }

    private static BurnSolveResult SolveBurnWithGuesses(
        Ephemeris ephemeris,
        State startState,
        double startDay,
        double targetDay,
        int ignoreEarthIndex,
        int ignoreAsteroidIndex,
        Vec3[] guesses)
    {
        BurnSolveResult best = BurnSolveResult.Failure;
        foreach (Vec3 guess in guesses)
        {
            BurnSolveResult candidate = SolveTargetedBurn(ephemeris, startState, startDay, targetDay, ignoreEarthIndex, ignoreAsteroidIndex, guess);
            if (candidate.BetterThan(best))
            {
                best = candidate;
            }
        }

        return best;
    }

    private static BurnSolveResult SolveTargetedBurn(
        Ephemeris ephemeris,
        State startState,
        double startDay,
        double targetDay,
        int ignoreEarthIndex,
        int ignoreAsteroidIndex,
        Vec3 initialGuess)
    {
        Vec3 burn = initialGuess;
        State bestState = default(State);
        double bestMiss = double.PositiveInfinity;
        bool converged = false;
        double durationSeconds = Math.Max(1.0, (targetDay - startDay) * DaySeconds);
        double epsilon = Math.Max(0.05, Math.Min(2.0, 1.0e7 / durationSeconds));

        for (int iter = 0; iter < 5; iter++)
        {
            State baseline = PropagateMassless(ephemeris, new State(startState.Position, startState.Velocity + burn), startDay, targetDay, ignoreEarthIndex, ignoreAsteroidIndex);
            Vec3 targetPosition = ephemeris.GetState(ignoreAsteroidIndex, targetDay).Position;
            Vec3 missVector = targetPosition - baseline.Position;
            double miss = missVector.Length();

            if (miss < bestMiss)
            {
                bestMiss = miss;
                bestState = baseline;
            }

            if (miss < 50_000.0)
            {
                converged = true;
                bestState = baseline;
                break;
            }

            Vec3 columnX = FiniteDifferenceColumn(ephemeris, startState, startDay, targetDay, ignoreEarthIndex, ignoreAsteroidIndex, burn, new Vec3(epsilon, 0.0, 0.0), baseline.Position);
            Vec3 columnY = FiniteDifferenceColumn(ephemeris, startState, startDay, targetDay, ignoreEarthIndex, ignoreAsteroidIndex, burn, new Vec3(0.0, epsilon, 0.0), baseline.Position);
            Vec3 columnZ = FiniteDifferenceColumn(ephemeris, startState, startDay, targetDay, ignoreEarthIndex, ignoreAsteroidIndex, burn, new Vec3(0.0, 0.0, epsilon), baseline.Position);

            Vec3 delta;
            if (!SolveLeastSquares(columnX, columnY, columnZ, missVector, out delta))
            {
                break;
            }

            double maxStep = Math.Max(200.0, burn.Length() * 0.75 + 200.0);
            if (delta.Length() > maxStep)
            {
                delta *= maxStep / delta.Length();
            }

            burn += delta;
            if (!IsFinite(burn.X) || !IsFinite(burn.Y) || !IsFinite(burn.Z))
            {
                break;
            }
        }

        if (!converged)
        {
            State baseline = PropagateMassless(ephemeris, new State(startState.Position, startState.Velocity + burn), startDay, targetDay, ignoreEarthIndex, ignoreAsteroidIndex);
            Vec3 targetPosition = ephemeris.GetState(ignoreAsteroidIndex, targetDay).Position;
            double miss = (targetPosition - baseline.Position).Length();
            if (miss < bestMiss)
            {
                bestMiss = miss;
                bestState = baseline;
            }

            converged = miss < 250_000.0;
        }

        return new BurnSolveResult(converged, burn, bestMiss, bestState);
    }

    private static Vec3 FiniteDifferenceColumn(
        Ephemeris ephemeris,
        State startState,
        double startDay,
        double targetDay,
        int ignoreEarthIndex,
        int ignoreAsteroidIndex,
        Vec3 burn,
        Vec3 deltaBurn,
        Vec3 baselinePosition)
    {
        State perturbed = PropagateMassless(
            ephemeris,
            new State(startState.Position, startState.Velocity + burn + deltaBurn),
            startDay,
            targetDay,
            ignoreEarthIndex,
            ignoreAsteroidIndex);

        double scale = 1.0 / Math.Max(deltaBurn.Length(), 1e-9);
        return (perturbed.Position - baselinePosition) * scale;
    }

    private static bool SolveLeastSquares(Vec3 c0, Vec3 c1, Vec3 c2, Vec3 rhs, out Vec3 solution)
    {
        double a00 = Vec3.Dot(c0, c0);
        double a01 = Vec3.Dot(c0, c1);
        double a02 = Vec3.Dot(c0, c2);
        double a11 = Vec3.Dot(c1, c1);
        double a12 = Vec3.Dot(c1, c2);
        double a22 = Vec3.Dot(c2, c2);
        double trace = a00 + a11 + a22;
        double lambda = Math.Max(1e-6, trace * 1e-12);

        a00 += lambda;
        a11 += lambda;
        a22 += lambda;

        double b0 = Vec3.Dot(c0, rhs);
        double b1 = Vec3.Dot(c1, rhs);
        double b2 = Vec3.Dot(c2, rhs);

        return SolveLinearSystem3x3(
            a00, a01, a02,
            a01, a11, a12,
            a02, a12, a22,
            b0, b1, b2,
            out solution);
    }

    private static bool SolveLinearSystem3x3(
        double a00, double a01, double a02,
        double a10, double a11, double a12,
        double a20, double a21, double a22,
        double b0, double b1, double b2,
        out Vec3 solution)
    {
        double[,] m =
        {
            { a00, a01, a02, b0 },
            { a10, a11, a12, b1 },
            { a20, a21, a22, b2 }
        };

        for (int pivot = 0; pivot < 3; pivot++)
        {
            int bestRow = pivot;
            double bestValue = Math.Abs(m[pivot, pivot]);
            for (int row = pivot + 1; row < 3; row++)
            {
                double value = Math.Abs(m[row, pivot]);
                if (value > bestValue)
                {
                    bestValue = value;
                    bestRow = row;
                }
            }

            if (bestValue < 1e-18)
            {
                solution = Vec3.Zero;
                return false;
            }

            if (bestRow != pivot)
            {
                for (int col = pivot; col < 4; col++)
                {
                    double tmp = m[pivot, col];
                    m[pivot, col] = m[bestRow, col];
                    m[bestRow, col] = tmp;
                }
            }

            double invPivot = 1.0 / m[pivot, pivot];
            for (int col = pivot; col < 4; col++)
            {
                m[pivot, col] *= invPivot;
            }

            for (int row = 0; row < 3; row++)
            {
                if (row == pivot)
                {
                    continue;
                }

                double factor = m[row, pivot];
                if (factor == 0.0)
                {
                    continue;
                }

                for (int col = pivot; col < 4; col++)
                {
                    m[row, col] -= factor * m[pivot, col];
                }
            }
        }

        solution = new Vec3(m[0, 3], m[1, 3], m[2, 3]);
        return IsFinite(solution.X) && IsFinite(solution.Y) && IsFinite(solution.Z);
    }

    private static State PropagateMassless(
        Ephemeris ephemeris,
        State initialState,
        double startDay,
        double endDay,
        int ignoreEarthIndex,
        int ignoreAsteroidIndex)
    {
        if (endDay <= startDay)
        {
            return initialState;
        }

        double startSeconds = startDay * DaySeconds;
        double endSeconds = endDay * DaySeconds;
        double totalDuration = endSeconds - startSeconds;
        double currentTime = startSeconds;
        Vec3 position = initialState.Position;
        Vec3 velocity = initialState.Velocity;

        while (currentTime < endSeconds - 1e-9)
        {
            double remaining = endSeconds - currentTime;
            double step = ChoosePropagationStep(totalDuration, remaining);
            if (step > remaining)
            {
                step = remaining;
            }

            RK4Step(ephemeris, ref position, ref velocity, currentTime, step, ignoreEarthIndex, ignoreAsteroidIndex);
            currentTime += step;
        }

        return new State(position, velocity);
    }

    private static double ChoosePropagationStep(double totalDuration, double remaining)
    {
        double baseStep = Math.Min(43_200.0, Math.Max(1_800.0, totalDuration / 3_000.0));
        if (remaining < 5.0 * DaySeconds)
        {
            return Math.Min(baseStep, 1_800.0);
        }

        if (remaining < 20.0 * DaySeconds)
        {
            return Math.Min(baseStep, 7_200.0);
        }

        if (remaining < 60.0 * DaySeconds)
        {
            return Math.Min(baseStep, 21_600.0);
        }

        return baseStep;
    }

    private static void RK4Step(
        Ephemeris ephemeris,
        ref Vec3 position,
        ref Vec3 velocity,
        double timeSeconds,
        double stepSeconds,
        int ignoreEarthIndex,
        int ignoreAsteroidIndex)
    {
        double t0 = timeSeconds / DaySeconds;
        double h = stepSeconds;
        double tHalf = (timeSeconds + 0.5 * h) / DaySeconds;
        double t1 = (timeSeconds + h) / DaySeconds;

        Vec3 a1 = ExternalAcceleration(ephemeris, position, t0, ignoreEarthIndex, ignoreAsteroidIndex);
        Vec3 k1r = velocity;
        Vec3 k1v = a1;

        Vec3 p2 = position + k1r * (0.5 * h);
        Vec3 v2 = velocity + k1v * (0.5 * h);
        Vec3 k2r = v2;
        Vec3 k2v = ExternalAcceleration(ephemeris, p2, tHalf, ignoreEarthIndex, ignoreAsteroidIndex);

        Vec3 p3 = position + k2r * (0.5 * h);
        Vec3 v3 = velocity + k2v * (0.5 * h);
        Vec3 k3r = v3;
        Vec3 k3v = ExternalAcceleration(ephemeris, p3, tHalf, ignoreEarthIndex, ignoreAsteroidIndex);

        Vec3 p4 = position + k3r * h;
        Vec3 v4 = velocity + k3v * h;
        Vec3 k4r = v4;
        Vec3 k4v = ExternalAcceleration(ephemeris, p4, t1, ignoreEarthIndex, ignoreAsteroidIndex);

        position += (k1r + 2.0 * k2r + 2.0 * k3r + k4r) * (h / 6.0);
        velocity += (k1v + 2.0 * k2v + 2.0 * k3v + k4v) * (h / 6.0);
    }

    private static Vec3 ExternalAcceleration(
        Ephemeris ephemeris,
        Vec3 position,
        double day,
        int ignoreEarthIndex,
        int ignoreAsteroidIndex)
    {
        Vec3 total = Vec3.Zero;
        for (int i = 0; i < ephemeris.BodyCount; i++)
        {
            if (i == ignoreEarthIndex || i == ignoreAsteroidIndex)
            {
                continue;
            }

            State bodyState = ephemeris.GetState(i, day);
            Vec3 delta = bodyState.Position - position;
            double distanceSquared = delta.LengthSquared();
            if (distanceSquared < 1.0)
            {
                continue;
            }

            double distance = Math.Sqrt(distanceSquared);
            double invCube = 1.0 / (distanceSquared * distance);
            total += delta * (G * ephemeris.Bodies[i].Mass * invCube);
        }

        return total;
    }

    private static double EvaluateDeflection(
        Ephemeris ephemeris,
        int asteroidIndex,
        int earthIndex,
        State deflectedAsteroidState,
        double impactDay,
        double evaluationDay)
    {
        if (evaluationDay <= impactDay)
        {
            State earth = ephemeris.GetState(earthIndex, impactDay);
            return (deflectedAsteroidState.Position - earth.Position).Length();
        }

        State asteroidAtEncounter = PropagateMassless(ephemeris, deflectedAsteroidState, impactDay, evaluationDay, asteroidIndex, -1);
        State earthAtEncounter = ephemeris.GetState(earthIndex, evaluationDay);
        return (asteroidAtEncounter.Position - earthAtEncounter.Position).Length();
    }

    private static bool TrySolveLambert(
        Vec3 r1,
        Vec3 r2,
        double timeOfFlightSeconds,
        double mu,
        bool longWay,
        out Vec3 v1,
        out Vec3 v2)
    {
        v1 = Vec3.Zero;
        v2 = Vec3.Zero;

        double r1Mag = r1.Length();
        double r2Mag = r2.Length();
        if (r1Mag < 1.0 || r2Mag < 1.0 || timeOfFlightSeconds <= 0.0 || mu <= 0.0)
        {
            return false;
        }

        double cosDeltaNu = Clamp(Vec3.Dot(r1, r2) / (r1Mag * r2Mag), -1.0, 1.0);
        double sinDeltaNu = Vec3.Cross(r1, r2).Length() / (r1Mag * r2Mag);
        if (longWay)
        {
            sinDeltaNu = -sinDeltaNu;
        }

        double denominator = 1.0 - cosDeltaNu;
        if (Math.Abs(denominator) < 1e-12)
        {
            return false;
        }

        double a = sinDeltaNu * Math.Sqrt(r1Mag * r2Mag / denominator);
        if (Math.Abs(a) < 1e-12)
        {
            return false;
        }

        double zLow = -16.0 * Math.PI * Math.PI;
        double zHigh = 16.0 * Math.PI * Math.PI;
        if (!BracketLambertRoot(r1Mag, r2Mag, a, mu, timeOfFlightSeconds, ref zLow, ref zHigh))
        {
            return false;
        }

        double z = 0.0;
        double timeTolerance = Math.Max(1e-3, timeOfFlightSeconds * 1e-8);
        for (int iter = 0; iter < 100; iter++)
        {
            z = 0.5 * (zLow + zHigh);
            double tof = LambertTimeOfFlight(r1Mag, r2Mag, a, z, mu, out _);
            if (!IsFinite(tof))
            {
                zLow = z;
                continue;
            }

            double error = tof - timeOfFlightSeconds;
            if (Math.Abs(error) <= timeTolerance)
            {
                break;
            }

            if (error > 0.0)
            {
                zHigh = z;
            }
            else
            {
                zLow = z;
            }
        }

        double y;
        double finalTof = LambertTimeOfFlight(r1Mag, r2Mag, a, z, mu, out y);
        if (!IsFinite(finalTof) || y <= 0.0)
        {
            return false;
        }

        double f = 1.0 - y / r1Mag;
        double g = a * Math.Sqrt(y / mu);
        double gDot = 1.0 - y / r2Mag;
        if (Math.Abs(g) < 1e-12)
        {
            return false;
        }

        v1 = (r2 - r1 * f) / g;
        v2 = (r2 * gDot - r1) / g;
        return IsFinite(v1.X) && IsFinite(v1.Y) && IsFinite(v1.Z) &&
               IsFinite(v2.X) && IsFinite(v2.Y) && IsFinite(v2.Z);
    }

    private static bool BracketLambertRoot(double r1Mag, double r2Mag, double a, double mu, double targetTof, ref double zLow, ref double zHigh)
    {
        double fLow = LambertTimeOfFlight(r1Mag, r2Mag, a, zLow, mu, out _) - targetTof;
        double fHigh = LambertTimeOfFlight(r1Mag, r2Mag, a, zHigh, mu, out _) - targetTof;

        for (int iter = 0; iter < 20 && (!IsFinite(fLow) || !IsFinite(fHigh) || fLow > 0.0 || fHigh < 0.0); iter++)
        {
            if (!IsFinite(fLow) || fLow > 0.0)
            {
                zLow *= 0.5;
                fLow = LambertTimeOfFlight(r1Mag, r2Mag, a, zLow, mu, out _) - targetTof;
            }

            if (!IsFinite(fHigh) || fHigh < 0.0)
            {
                zHigh *= 2.0;
                fHigh = LambertTimeOfFlight(r1Mag, r2Mag, a, zHigh, mu, out _) - targetTof;
            }
        }

        return IsFinite(fLow) && IsFinite(fHigh) && fLow <= 0.0 && fHigh >= 0.0;
    }

    private static double LambertTimeOfFlight(double r1Mag, double r2Mag, double a, double z, double mu, out double y)
    {
        double c = StumpffC(z);
        double s = StumpffS(z);
        if (c <= 0.0)
        {
            y = double.NaN;
            return double.NaN;
        }

        y = r1Mag + r2Mag + a * ((z * s - 1.0) / Math.Sqrt(c));
        if (y <= 0.0)
        {
            return double.NaN;
        }

        double x = Math.Sqrt(y / c);
        return (x * x * x * s + a * Math.Sqrt(y)) / Math.Sqrt(mu);
    }

    private static double StumpffC(double z)
    {
        if (z > 1e-8)
        {
            double root = Math.Sqrt(z);
            return (1.0 - Math.Cos(root)) / z;
        }

        if (z < -1e-8)
        {
            double root = Math.Sqrt(-z);
            return (1.0 - Math.Cosh(root)) / z;
        }

        double z2 = z * z;
        double z3 = z2 * z;
        return 0.5 - z / 24.0 + z2 / 720.0 - z3 / 40320.0;
    }

    private static double StumpffS(double z)
    {
        if (z > 1e-8)
        {
            double root = Math.Sqrt(z);
            return (root - Math.Sin(root)) / (root * root * root);
        }

        if (z < -1e-8)
        {
            double root = Math.Sqrt(-z);
            return (Math.Sinh(root) - root) / (root * root * root);
        }

        double z2 = z * z;
        double z3 = z2 * z;
        return (1.0 / 6.0) - z / 120.0 + z2 / 5040.0 - z3 / 362880.0;
    }

    private static MissionPlan BuildFallbackMission(
        Ephemeris ephemeris,
        int earthIndex,
        int asteroidIndex,
        double nominalImpactDay,
        double warningDays,
        double dvBudget)
    {
        double impactDay = Clamp(Math.Min(nominalImpactDay * 0.75, warningDays - 0.1), 1.0, Math.Max(1.0, warningDays - 0.1));
        State earth = ephemeris.GetState(earthIndex, 0.0);
        State asteroid = ephemeris.GetState(asteroidIndex, impactDay);
        Vec3 lineOfSight = (asteroid.Position - earth.Position).Normalized();
        Vec3 burn = lineOfSight * Math.Max(1.0, Math.Min(dvBudget, dvBudget * 0.8));
        var burns = new List<Burn> { new Burn(0.0, burn) };
        return new MissionPlan(burns, impactDay, burn.Length(), (asteroid.Position - earth.Position).Length(), 0.0, -1e30);
    }

    private static MissionPlan BuildFallbackMissionSimple(double warningDays, double dvBudget)
    {
        var burns = new List<Burn> { new Burn(0.0, new Vec3(Math.Max(1.0, dvBudget * 0.5), 0.0, 0.0)) };
        return new MissionPlan(burns, Math.Max(1.0, warningDays * 0.5), burns[0].DeltaV.Length(), double.PositiveInfinity, 0.0, -1e30);
    }

    private static void WritePlan(MissionPlan plan)
    {
        var outputBurns = new List<Burn>(plan.Burns.Count + 1);
        foreach (Burn burn in plan.Burns)
        {
            outputBurns.Add(burn);
        }

        outputBurns.Sort((a, b) => a.TimeDay.CompareTo(b.TimeDay));
        if (outputBurns.Count == 0 || Math.Abs(outputBurns[outputBurns.Count - 1].TimeDay - plan.ImpactDay) > ImpactEventToleranceDays)
        {
            outputBurns.Add(new Burn(plan.ImpactDay, Vec3.Zero));
        }

        Console.WriteLine(outputBurns.Count.ToString(CultureInfo.InvariantCulture));
        foreach (Burn burn in outputBurns)
        {

            Vec3 dv = new Vec3(

                FormatDouble(dv.Z));
        }
    }

    private static string FormatDouble(double value)
    {
        return value.ToString("G17", CultureInfo.InvariantCulture);
    }

    private static bool IsFinite(double value)
    {
        return !(double.IsNaN(value) || double.IsInfinity(value));
    }

    private static double Clamp(double value, double min, double max)
    {
        if (value < min) return min;
        if (value > max) return max;
        return value;
    }

public readonly struct Vec3
{
    public readonly double X;
    public readonly double Y;
    public readonly double Z;

    public Vec3(double x, double y, double z)
    {
        X = x;
        Y = y;
        Z = z;
    }

    public static Vec3 Zero => new Vec3(0.0, 0.0, 0.0);

    public double Length()
    {
        return Math.Sqrt(X * X + Y * Y + Z * Z);
    }

    public double LengthSquared()
    {
        return X * X + Y * Y + Z * Z;
    }

    public Vec3 Normalized()
    {
        double len = Length();
        return len > 0.0 ? this / len : Zero;
    }

    public static Vec3 operator +(Vec3 a, Vec3 b)
    {
        return new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    }

    public static Vec3 operator -(Vec3 a, Vec3 b)
    {
        return new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    }

    public static Vec3 operator -(Vec3 a)
    {
        return new Vec3(-a.X, -a.Y, -a.Z);
    }

    public static Vec3 operator *(Vec3 a, double scalar)
    {
        return new Vec3(a.X * scalar, a.Y * scalar, a.Z * scalar);
    }

    public static Vec3 operator *(double scalar, Vec3 a)
    {
        return a * scalar;
    }

    public static Vec3 operator /(Vec3 a, double scalar)
    {
        return new Vec3(a.X / scalar, a.Y / scalar, a.Z / scalar);
    }

    public static double Dot(Vec3 a, Vec3 b)
    {
        return a.X * b.X + a.Y * b.Y + a.Z * b.Z;
    }

    public static Vec3 Cross(Vec3 a, Vec3 b)
    {
        return new Vec3(
            a.Y * b.Z - a.Z * b.Y,
            a.Z * b.X - a.X * b.Z,
            a.X * b.Y - a.Y * b.X);
    }
}

public sealed class Ephemeris
{
    private readonly Vec3[][] _positions;
    private readonly Vec3[][] _velocities;

    public Ephemeris(Body[] bodies, Vec3[][] positions, Vec3[][] velocities, double stepDays, int stepCount)
    {
        Bodies = bodies;
        _positions = positions;
        _velocities = velocities;
        StepDays = stepDays;
        StepCount = stepCount;
    }

    public Body[] Bodies { get; }
    public int BodyCount => Bodies.Length;
    public double StepDays { get; }
    public int StepCount { get; }

    public State GetState(int bodyIndex, double day)
    {
        if (day <= 0.0)
        {
            return new State(_positions[bodyIndex][0], _velocities[bodyIndex][0]);
        }

        double maxDay = StepCount * StepDays;
        if (day >= maxDay)
        {
            return new State(_positions[bodyIndex][StepCount], _velocities[bodyIndex][StepCount]);
        }

        double scaled = day / StepDays;
        int index = (int)Math.Floor(scaled);
        double u = scaled - index;
        if (index >= StepCount)
        {
            index = StepCount - 1;
            u = 1.0;
        }

        Vec3 p0 = _positions[bodyIndex][index];
        Vec3 p1 = _positions[bodyIndex][index + 1];
        Vec3 v0 = _velocities[bodyIndex][index];
        Vec3 v1 = _velocities[bodyIndex][index + 1];
        double dt = StepDays * 86400.0;

        double h00 = 2.0 * u * u * u - 3.0 * u * u + 1.0;
        double h10 = u * u * u - 2.0 * u * u + u;
        double h01 = -2.0 * u * u * u + 3.0 * u * u;
        double h11 = u * u * u - u * u;

        Vec3 position = p0 * h00 + v0 * (h10 * dt) + p1 * h01 + v1 * (h11 * dt);

        double dh00 = 6.0 * u * u - 6.0 * u;
        double dh10 = 3.0 * u * u - 4.0 * u + 1.0;
        double dh01 = -6.0 * u * u + 6.0 * u;
        double dh11 = 3.0 * u * u - 2.0 * u;
        Vec3 velocity = (p0 * dh00 + v0 * (dh10 * dt) + p1 * dh01 + v1 * (dh11 * dt)) / dt;

        return new State(position, velocity);
    }
}

    {

        FinalState = finalState;
    }

    public bool Success { get; }

    {
        if (Success != other.Success)
        {
            return Success;

            return MissDistance < other.MissDistance;
        }

        return DeltaV.Length() < other.DeltaV.Length();
    }

public sealed class TokenReader
{
    private readonly string _text;
    private int _index;

    public TokenReader(string text)
    {
        _text = text ?? string.Empty;
        _index = 0;
    }

    public int ReadInt()
    {
        return int.Parse(ReadString(), CultureInfo.InvariantCulture);
    }

    public double ReadDouble()
    {
        return double.Parse(ReadString(), CultureInfo.InvariantCulture);
    }

    public string ReadString()
    {
        SkipWhitespace();
        if (_index >= _text.Length)
        {
            throw new EndOfStreamException();
        }

        int start = _index;
        while (_index < _text.Length && !char.IsWhiteSpace(_text[_index]))
        {
            _index++;
        }

        return _text.Substring(start, _index - start);
    }

    public string ReadRemaining()
    {
        SkipWhitespace();
        if (_index >= _text.Length)
        {
            return string.Empty;
        }

        return _text.Substring(_index);
    }

    private void SkipWhitespace()
    {
        while (_index < _text.Length && char.IsWhiteSpace(_text[_index]))
        {
            _index++;
        }
    }
