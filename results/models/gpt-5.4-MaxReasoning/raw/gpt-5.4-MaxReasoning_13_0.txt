string trim_cr(string line) {
    if (!line.empty() && line.back() == '\r') {
        line.pop_back();
    }
    return line;
}

string solve_bruteforce(const vector<string>& strings) {
    size_t base_index = 0;
    for (size_t i = 1; i < strings.size(); ++i) {
        if (strings[i].size() < strings[base_index].size()) {
            base_index = i;
        }
    }

    const string& base = strings[base_index];
    for (size_t len = base.size(); len > 0; --len) {
        for (size_t start = 0; start + len <= base.size(); ++start) {
            const string candidate = base.substr(start, len);
            bool ok = true;
            for (size_t i = 0; i < strings.size(); ++i) {
                if (i != base_index && strings[i].find(candidate) == string::npos) {
                    ok = false;
                    break;
                }
            }
            if (ok) {
                return candidate;
            }
        }
    }
    return "";
}

string solve_two_strings_dp(const string& a, const string& b) {
    vector<int> dp(b.size() + 1, 0);
    size_t best_len = 0;
    size_t best_end = 0;

    for (size_t i = 1; i <= a.size(); ++i) {
        for (size_t j = b.size(); j > 0; --j) {
            if (a[i - 1] == b[j - 1]) {
                dp[j] = dp[j - 1] + 1;
                if (static_cast<size_t>(dp[j]) > best_len) {
                    best_len = static_cast<size_t>(dp[j]);
                    best_end = i;
                }
            } else {
                dp[j] = 0;
            }
        }
    }

    return a.substr(best_end - best_len, best_len);
}

struct SuffixAutomaton {
    struct State {
        int link = -1;
        int len = 0;
        int head = -1;
        int first_pos = -1;
        int best = 0;
    };

    struct Edge {
        int to = 0;
        int next = -1;
        unsigned char ch = 0;
    };

    vector<State> states;
    vector<Edge> edges;
    int last = 0;

    explicit SuffixAutomaton(size_t base_length) {
        const size_t state_cap = max<size_t>(1, base_length * 2);
        const size_t edge_cap = max<size_t>(1, base_length * 3);
        states.reserve(state_cap);
        edges.reserve(edge_cap);
        states.push_back(State{});
    }

    int add_state(int len, int first_pos) {
        states.push_back(State{-1, len, -1, first_pos, 0});
        return static_cast<int>(states.size()) - 1;
    }

    int find_edge(int state, unsigned char ch) const {
        for (int edge = states[state].head; edge != -1; edge = edges[edge].next) {
            if (edges[edge].ch == ch) {
                return edge;
            }
        }
        return -1;
    }

    int next_state(int state, unsigned char ch) const {
        const int edge = find_edge(state, ch);
        return edge == -1 ? -1 : edges[edge].to;
    }

    void add_transition(int state, unsigned char ch, int to) {
        edges.push_back(Edge{to, states[state].head, ch});
        states[state].head = static_cast<int>(edges.size()) - 1;
    }

    int clone_state(int source, int new_len) {
        const int clone = add_state(new_len, states[source].first_pos);
        states[clone].link = states[source].link;
        for (int edge = states[source].head; edge != -1; edge = edges[edge].next) {
            add_transition(clone, edges[edge].ch, edges[edge].to);
        }
        return clone;
    }

    void extend(unsigned char ch, int pos) {
        const int current = add_state(states[last].len + 1, pos);
        int p = last;

        while (p != -1 && find_edge(p, ch) == -1) {
            add_transition(p, ch, current);
            p = states[p].link;
        }

        if (p == -1) {
            states[current].link = 0;
            last = current;
            return;
        }

        const int edge = find_edge(p, ch);
        const int q = edges[edge].to;
        if (states[p].len + 1 == states[q].len) {
            states[current].link = q;
            last = current;
            return;
        }

        const int clone = clone_state(q, states[p].len + 1);
        while (p != -1) {
            const int pe = find_edge(p, ch);
            if (pe == -1 || edges[pe].to != q) {
                break;
            }
            edges[pe].to = clone;
            p = states[p].link;
        }

        states[q].link = clone;
        states[current].link = clone;
        last = current;
    }

    vector<int> order_by_length() const {
        int max_len = 0;
        for (const State& state : states) {
            if (state.len > max_len) {
                max_len = state.len;
            }
        }

        vector<int> count(static_cast<size_t>(max_len) + 1, 0);
        vector<int> order(states.size(), 0);
        for (const State& state : states) {
            ++count[state.len];
        }
        for (int i = 1; i <= max_len; ++i) {
            count[i] += count[i - 1];
        }
        for (int i = static_cast<int>(states.size()) - 1; i >= 0; --i) {
            order[--count[states[i].len]] = i;
        }
        return order;
    }
};

string solve_suffix_automaton(vector<string> strings) {
    size_t base_index = 0;
    for (size_t i = 1; i < strings.size(); ++i) {
        if (strings[i].size() < strings[base_index].size()) {
            base_index = i;
        }
    }

    swap(strings[0], strings[base_index]);
    sort(strings.begin() + 1, strings.end(),
         [](const string& lhs, const string& rhs) { return lhs.size() < rhs.size(); });

    const string& base = strings[0];
    SuffixAutomaton sam(base.size());
    for (int i = 0; i < static_cast<int>(base.size()); ++i) {
        sam.extend(static_cast<unsigned char>(base[i]), i);
    }

    for (auto& state : sam.states) {
        state.best = state.len;
    }

    const vector<int> order = sam.order_by_length();
    vector<int> match(sam.states.size(), 0);

    for (size_t idx = 1; idx < strings.size(); ++idx) {
        fill(match.begin(), match.end(), 0);

        int state = 0;
        int length = 0;
        for (unsigned char ch : strings[idx]) {
            int next = sam.next_state(state, ch);
            while (state != 0 && next == -1) {
                state = sam.states[state].link;
                length = sam.states[state].len;
                next = sam.next_state(state, ch);
            }

            if (next == -1) {
                state = 0;
                length = 0;
                continue;
            }

            state = next;
            ++length;
            if (length > match[state]) {
                match[state] = length;
            }
        }

        int round_best = 0;
        for (int pos = static_cast<int>(order.size()) - 1; pos > 0; --pos) {
            const int v = order[pos];
            sam.states[v].best = min(sam.states[v].best, match[v]);
            if (sam.states[v].best > round_best) {
                round_best = sam.states[v].best;
            }

            const int parent = sam.states[v].link;
            if (parent >= 0) {
                const int propagated = min(match[v], sam.states[parent].len);
                if (propagated > match[parent]) {
                    match[parent] = propagated;
                }
            }
        }

        if (round_best == 0) {
            return "";
        }
    }

    int best_state = 0;
    int best_len = 0;
    for (int i = 1; i < static_cast<int>(sam.states.size()); ++i) {
        if (sam.states[i].best > best_len) {
            best_len = sam.states[i].best;
            best_state = i;
        }
    }

    if (best_len == 0) {
        return "";
    }

    const int start = sam.states[best_state].first_pos - best_len + 1;
    return base.substr(static_cast<size_t>(start), static_cast<size_t>(best_len));
}

}  // namespace

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n = 0;
    if (!(cin >> n)) {
        return 0;
    }

    string dummy;
    getline(cin, dummy);

    if (n <= 0) {
        cout << '\n';
        return 0;
    }

    vector<string> strings;
    strings.reserve(static_cast<size_t>(n));

    size_t total_length = 0;
    for (int i = 0; i < n; ++i) {
        string line;
        getline(cin, line);
        line = trim_cr(line);
        total_length += line.size();
        strings.push_back(move(line));
    }

    for (const string& s : strings) {
        if (s.empty()) {
            cout << '\n';
            return 0;
        }
    }

    if (n == 1) {
        cout << strings[0] << '\n';
        return 0;
    }

    if (total_length <= 2048) {
        cout << solve_bruteforce(strings) << '\n';
        return 0;
    }

    const unsigned long long dp_work =
        static_cast<unsigned long long>(strings[0].size()) *
        static_cast<unsigned long long>(strings[1].size());
    if (n == 2 && dp_work <= 50000000ULL) {
        cout << solve_two_strings_dp(strings[0], strings[1]) << '\n';
        return 0;
    }

    cout << solve_suffix_automaton(move(strings)) << '\n';
    return 0;
}
