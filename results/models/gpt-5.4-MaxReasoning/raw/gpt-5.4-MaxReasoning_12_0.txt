using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        if (!scanner.TryReadInt(out int vertexCount))
        {
            return;
        }

        int faceCount = scanner.NextInt();
        var vertices = new Vec3[vertexCount];
        for (int i = 0; i < vertexCount; i++)
        {
            vertices[i] = new Vec3(scanner.NextDouble(), scanner.NextDouble(), scanner.NextDouble());
        }

        var rawFaces = new Triangle[faceCount];
        int minIndex = int.MaxValue;
        int maxIndex = int.MinValue;
        for (int i = 0; i < faceCount; i++)
        {
            int a = scanner.NextInt();
            int b = scanner.NextInt();
            int c = scanner.NextInt();
            rawFaces[i] = new Triangle(a, b, c);
            minIndex = Math.Min(minIndex, Math.Min(a, Math.Min(b, c)));
            maxIndex = Math.Max(maxIndex, Math.Max(a, Math.Max(b, c)));
        }

        int faceOffset = 0;
        if (minIndex >= 1 && maxIndex <= vertexCount)
        {
            faceOffset = 1;
        }

        var faces = new Triangle[faceCount];
        bool validFaces = true;
        for (int i = 0; i < faceCount; i++)
        {
            int a = rawFaces[i].A - faceOffset;
            int b = rawFaces[i].B - faceOffset;
            int c = rawFaces[i].C - faceOffset;
            if (a < 0 || a >= vertexCount || b < 0 || b >= vertexCount || c < 0 || c >= vertexCount)
            {
                validFaces = false;
                break;
            }

            faces[i] = new Triangle(a, b, c);
        }

        double width = scanner.NextDouble();
        double height = scanner.NextDouble();
        double depth = scanner.NextDouble();

        using var writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        if (!validFaces)
        {
            writer.WriteLine(0);
            return;
        }

        var solver = new Solver(vertices, faces, new Vec3(width, height, depth));
        solver.Solve(writer);
    }
}

internal sealed class Solver
{
    private const double Eps = 1e-9;

    private static readonly int[][] AxisOrders =
    {
        new[] { 0, 1, 2 },
        new[] { 0, 2, 1 },
        new[] { 1, 0, 2 },
        new[] { 1, 2, 0 },
        new[] { 2, 0, 1 },
        new[] { 2, 1, 0 }
    };

    private readonly Vec3[] _vertices;
    private readonly Triangle[] _faces;
    private readonly Vec3 _container;
    private readonly double _pieceVolume;
    private readonly StrategyConfig _strategy;
    private readonly List<OrientationInfo> _orientations;
    private readonly Dictionary<StateKey, PlanTemplate> _cache = new Dictionary<StateKey, PlanTemplate>();

    public Solver(Vec3[] vertices, Triangle[] faces, Vec3 container)
    {
        _vertices = vertices;
        _faces = faces;
        _container = container;
        _pieceVolume = ComputeAbsoluteVolume(vertices, faces);
        _strategy = ChooseStrategy(vertices.Length + faces.Length, container);
        _orientations = BuildOrientations();
    }

    public void Solve(TextWriter writer)
    {
        if (_vertices.Length == 0 || _faces.Length == 0 || _container.X <= Eps || _container.Y <= Eps || _container.Z <= Eps)
        {
            writer.WriteLine(0);
            writer.Flush();
            return;
        }

        if (_orientations.Count == 0)
        {
            writer.WriteLine(0);
            writer.Flush();
            return;
        }

        var plan = SolveSpace(_container, _strategy.Depth);
        writer.WriteLine(plan.Count.ToString(CultureInfo.InvariantCulture));
        plan.Write(writer, Vec3.Zero);
        writer.Flush();
    }

    private static StrategyConfig ChooseStrategy(int complexity, Vec3 container)
    {
        double maxDim = Math.Max(container.X, Math.Max(container.Y, container.Z));
        double volume = Math.Max(0.0, container.X * container.Y * container.Z);

        StrategyConfig config;
        if (maxDim <= 8.0 && volume <= 4_096.0 && complexity <= 600)
        {
            config = new StrategyConfig("trivial", faceSamples: 64, rawOrientationCap: 256, keepOrientations: 48, branchWidth: 10, depth: 6, partitionOrders: 6);
        }
        else if (maxDim <= 30.0 && volume <= 200_000.0 && complexity <= 5_000)
        {
            config = new StrategyConfig("medium", faceSamples: 48, rawOrientationCap: 192, keepOrientations: 32, branchWidth: 8, depth: 5, partitionOrders: 6);
        }
        else if (maxDim <= 120.0 && volume <= 5_000_000.0)
        {
            config = new StrategyConfig("large", faceSamples: 24, rawOrientationCap: 96, keepOrientations: 18, branchWidth: 5, depth: 3, partitionOrders: 4);
        }
        else
        {
            config = new StrategyConfig("extreme", faceSamples: 12, rawOrientationCap: 48, keepOrientations: 10, branchWidth: 3, depth: 2, partitionOrders: 2);
        }

        if (complexity > 20_000)
        {
            config = config.With(faceSamples: Math.Min(config.FaceSamples, 16), rawOrientationCap: Math.Min(config.RawOrientationCap, 64), keepOrientations: Math.Min(config.KeepOrientations, 14), branchWidth: Math.Min(config.BranchWidth, 4), depth: Math.Min(config.Depth, 3));
        }

        if (complexity > 80_000)
        {
            config = config.With(faceSamples: Math.Min(config.FaceSamples, 8), rawOrientationCap: Math.Min(config.RawOrientationCap, 24), keepOrientations: Math.Min(config.KeepOrientations, 8), branchWidth: Math.Min(config.BranchWidth, 3), depth: Math.Min(config.Depth, 2));
        }

        return config;
    }

    private List<OrientationInfo> BuildOrientations()
    {
        var result = new List<OrientationInfo>();
        var seen = new HashSet<string>(StringComparer.Ordinal);

        void TryAdd(Mat3 matrix)
        {
            if (matrix.Determinant() < 0.5)
            {
                return;
            }

            var quaternion = QuaternionD.FromMatrix(matrix).Canonicalized();
            string key = quaternion.QuantizedKey(6);
            if (!seen.Add(key))
            {
                return;
            }

            var bounds = ComputeBounds(matrix);
            if (bounds.Size.X <= Eps || bounds.Size.Y <= Eps || bounds.Size.Z <= Eps)
            {
                return;
            }

            if (!Fits(_container.X, bounds.Size.X) || !Fits(_container.Y, bounds.Size.Y) || !Fits(_container.Z, bounds.Size.Z))
            {
                return;
            }

            long nx = FloorFit(_container.X, bounds.Size.X);
            long ny = FloorFit(_container.Y, bounds.Size.Y);
            long nz = FloorFit(_container.Z, bounds.Size.Z);
            long fullCount = SaturatingMul(nx, SaturatingMul(ny, nz));
            if (fullCount <= 0)
            {
                return;
            }

            double aabbVolume = bounds.Size.X * bounds.Size.Y * bounds.Size.Z;
            double efficiency = (_pieceVolume > Eps && aabbVolume > Eps) ? _pieceVolume / aabbVolume : 0.0;
            result.Add(new OrientationInfo(matrix, quaternion, bounds.Min, bounds.Max, bounds.Size, aabbVolume, efficiency, fullCount));
        }

        AddPermutationsOfBasis(Vec3.UnitX, Vec3.UnitY, Vec3.UnitZ, TryAdd);

        if (TryComputePrincipalBasis(_vertices, out Vec3 px, out Vec3 py, out Vec3 pz))
        {
            AddPermutationsOfBasis(px, py, pz, TryAdd);
        }

        var faceSeeds = CollectFaceEdgeBases(_strategy.FaceSamples);
        foreach (var basis in faceSeeds)
        {
            AddPermutationsOfBasis(basis.X, basis.Y, basis.Z, TryAdd);
            if (result.Count >= _strategy.RawOrientationCap * 2)
            {
                break;
            }
        }

        result.Sort(static (a, b) =>
        {
            int cmp = b.FullContainerCount.CompareTo(a.FullContainerCount);
            if (cmp != 0)
            {
                return cmp;
            }

            cmp = b.Efficiency.CompareTo(a.Efficiency);
            if (cmp != 0)
            {
                return cmp;
            }

            cmp = a.AabbVolume.CompareTo(b.AabbVolume);
            if (cmp != 0)
            {
                return cmp;
            }

            cmp = a.Size.MaxComponent.CompareTo(b.Size.MaxComponent);
            if (cmp != 0)
            {
                return cmp;
            }

            return a.Quaternion.W.CompareTo(b.Quaternion.W);
        });

        if (result.Count > _strategy.KeepOrientations)
        {
            result.RemoveRange(_strategy.KeepOrientations, result.Count - _strategy.KeepOrientations);
        }

        return result;
    }

    private Bounds ComputeBounds(Mat3 matrix)
    {
        Vec3 min = new Vec3(double.PositiveInfinity, double.PositiveInfinity, double.PositiveInfinity);
        Vec3 max = new Vec3(double.NegativeInfinity, double.NegativeInfinity, double.NegativeInfinity);

        for (int i = 0; i < _vertices.Length; i++)
        {
            Vec3 rotated = matrix.Transform(_vertices[i]);
            min = Vec3.Min(min, rotated);
            max = Vec3.Max(max, rotated);
        }

        return new Bounds(min, max);
    }

    private List<BasisFrame> CollectFaceEdgeBases(int faceBudget)
    {
        var samples = new List<FaceSample>(_faces.Length);
        for (int i = 0; i < _faces.Length; i++)
        {
            Triangle t = _faces[i];
            Vec3 a = _vertices[t.A];
            Vec3 b = _vertices[t.B];
            Vec3 c = _vertices[t.C];
            Vec3 n = Vec3.Cross(b - a, c - a);
            double area2 = n.Length;
            if (area2 > Eps)
            {
                samples.Add(new FaceSample(i, area2));
            }
        }

        samples.Sort(static (a, b) => b.Area2.CompareTo(a.Area2));
        if (samples.Count > faceBudget)
        {
            samples.RemoveRange(faceBudget, samples.Count - faceBudget);
        }

        var result = new List<BasisFrame>(samples.Count);
        for (int i = 0; i < samples.Count; i++)
        {
            Triangle t = _faces[samples[i].FaceIndex];
            Vec3 a = _vertices[t.A];
            Vec3 b = _vertices[t.B];
            Vec3 c = _vertices[t.C];
            Vec3 normal = Vec3.Cross(b - a, c - a).NormalizedOrZero();
            if (normal.LengthSquared <= Eps)
            {
                continue;
            }

            TryAddEdgeBasis(a, b, normal, result);
            TryAddEdgeBasis(b, c, normal, result);
            TryAddEdgeBasis(c, a, normal, result);
        }

        return result;
    }

    private static void TryAddEdgeBasis(Vec3 start, Vec3 end, Vec3 normal, List<BasisFrame> result)
    {
        Vec3 edge = end - start;
        Vec3 x = (edge - normal * Vec3.Dot(edge, normal)).NormalizedOrZero();
        if (x.LengthSquared <= Eps)
        {
            return;
        }

        Vec3 y = Vec3.Cross(normal, x).NormalizedOrZero();
        if (y.LengthSquared <= Eps)
        {
            return;
        }

        x = Vec3.Cross(y, normal).NormalizedOrZero();
        if (x.LengthSquared <= Eps)
        {
            return;
        }

        result.Add(new BasisFrame(x, y, normal));
    }

    private PlanTemplate SolveSpace(Vec3 size, int depth)
    {
        if (size.X <= Eps || size.Y <= Eps || size.Z <= Eps)
        {
            return EmptyTemplate.Instance;
        }

        var key = new StateKey(size, depth);
        if (_cache.TryGetValue(key, out PlanTemplate cached))
        {
            return cached;
        }

        var candidates = SelectCandidates(size);
        if (candidates.Count == 0)
        {
            _cache[key] = EmptyTemplate.Instance;
            return EmptyTemplate.Instance;
        }

        long bestCount = 0;
        PlanTemplate bestPlan = EmptyTemplate.Instance;

        for (int i = 0; i < candidates.Count; i++)
        {
            var candidate = candidates[i];
            var block = new BlockTemplate(-candidate.Orientation.Min, candidate.Orientation.Size, candidate.Nx, candidate.Ny, candidate.Nz, candidate.Orientation.QuaternionText);
            long baseCount = candidate.Count;

            if (baseCount > bestCount)
            {
                bestCount = baseCount;
                bestPlan = block;
            }

            if (depth <= 0)
            {
                continue;
            }

            int orderLimit = Math.Min(_strategy.PartitionOrders, AxisOrders.Length);
            for (int orderIndex = 0; orderIndex < orderLimit; orderIndex++)
            {
                var order = AxisOrders[orderIndex];
                List<PlanTemplate>? children = null;
                long total = baseCount;

                foreach (var part in CreatePartitions(size, candidate.UsedSize, order))
                {
                    if (part.Size.X <= Eps || part.Size.Y <= Eps || part.Size.Z <= Eps)
                    {
                        continue;
                    }

                    var child = SolveSpace(part.Size, depth - 1);
                    if (child.Count <= 0)
                    {
                        continue;
                    }

                    total = SaturatingAdd(total, child.Count);
                    children ??= new List<PlanTemplate>(3);
                    children.Add(OffsetTemplate.Create(child, part.Origin));
                }

                if (total > bestCount)
                {
                    bestCount = total;
                    bestPlan = Compose(block, children);
                }
            }
        }

        _cache[key] = bestPlan;
        return bestPlan;
    }

    private List<CandidatePlacement> SelectCandidates(Vec3 size)
    {
        var candidates = new List<CandidatePlacement>(_orientations.Count);
        double spaceVolume = Math.Max(Eps, size.X * size.Y * size.Z);

        for (int i = 0; i < _orientations.Count; i++)
        {
            var orientation = _orientations[i];
            long nx = FloorFit(size.X, orientation.Size.X);
            long ny = FloorFit(size.Y, orientation.Size.Y);
            long nz = FloorFit(size.Z, orientation.Size.Z);
            if (nx <= 0 || ny <= 0 || nz <= 0)
            {
                continue;
            }

            long count = SaturatingMul(nx, SaturatingMul(ny, nz));
            Vec3 used = new Vec3(nx * orientation.Size.X, ny * orientation.Size.Y, nz * orientation.Size.Z);
            double usedVolume = Math.Max(0.0, used.X * used.Y * used.Z);
            double fillRatio = usedVolume / spaceVolume;
            double remnant = (size.X - used.X) + (size.Y - used.Y) + (size.Z - used.Z);
            candidates.Add(new CandidatePlacement(orientation, nx, ny, nz, count, used, fillRatio, remnant));
        }

        candidates.Sort(static (a, b) =>
        {
            int cmp = b.Count.CompareTo(a.Count);
            if (cmp != 0)
            {
                return cmp;
            }

            cmp = b.FillRatio.CompareTo(a.FillRatio);
            if (cmp != 0)
            {
                return cmp;
            }

            cmp = a.Remnant.CompareTo(b.Remnant);
            if (cmp != 0)
            {
                return cmp;
            }

            cmp = b.Orientation.Efficiency.CompareTo(a.Orientation.Efficiency);
            if (cmp != 0)
            {
                return cmp;
            }

            return a.Orientation.AabbVolume.CompareTo(b.Orientation.AabbVolume);
        });

        if (candidates.Count > _strategy.BranchWidth)
        {
            candidates.RemoveRange(_strategy.BranchWidth, candidates.Count - _strategy.BranchWidth);
        }

        return candidates;
    }

    private static IEnumerable<Subspace> CreatePartitions(Vec3 size, Vec3 used, int[] order)
    {
        int[] position = new int[3];
        for (int i = 0; i < 3; i++)
        {
            position[order[i]] = i;
        }

        for (int step = 0; step < 3; step++)
        {
            int axis = order[step];
            double remainder = size.Get(axis) - used.Get(axis);
            if (remainder <= Eps)
            {
                continue;
            }

            double ox = 0.0;
            double oy = 0.0;
            double oz = 0.0;
            double sx = 0.0;
            double sy = 0.0;
            double sz = 0.0;

            for (int currentAxis = 0; currentAxis < 3; currentAxis++)
            {
                double span;
                if (currentAxis == axis)
                {
                    span = remainder;
                    if (currentAxis == 0)
                    {
                        ox = used.X;
                    }
                    else if (currentAxis == 1)
                    {
                        oy = used.Y;
                    }
                    else
                    {
                        oz = used.Z;
                    }
                }
                else if (position[currentAxis] < step)
                {
                    span = used.Get(currentAxis);
                }
                else
                {
                    span = size.Get(currentAxis);
                }

                if (currentAxis == 0)
                {
                    sx = span;
                }
                else if (currentAxis == 1)
                {
                    sy = span;
                }
                else
                {
                    sz = span;
                }
            }

            yield return new Subspace(new Vec3(ox, oy, oz), new Vec3(sx, sy, sz));
        }
    }

    private static PlanTemplate Compose(BlockTemplate block, List<PlanTemplate>? children)
    {
        if (children == null || children.Count == 0)
        {
            return block;
        }

        var all = new PlanTemplate[children.Count + 1];
        all[0] = block;
        for (int i = 0; i < children.Count; i++)
        {
            all[i + 1] = children[i];
        }

        return new CompositeTemplate(all);
    }

    private static long FloorFit(double space, double item)
    {
        if (item <= Eps || space + 1e-10 < item)
        {
            return 0;
        }

        double adjusted = (space + Math.Max(1.0, space) * 1e-10) / item;
        if (adjusted <= 0.0)
        {
            return 0;
        }

        double floored = Math.Floor(adjusted);
        if (floored >= long.MaxValue)
        {
            return long.MaxValue;
        }

        return (long)floored;
    }

    private static bool Fits(double outer, double inner)
    {
        return inner <= outer + Math.Max(1.0, outer) * 1e-8;
    }

    private static long SaturatingMul(long a, long b)
    {
        if (a <= 0 || b <= 0)
        {
            return 0;
        }

        if (a > long.MaxValue / b)
        {
            return long.MaxValue;
        }

        return a * b;
    }

    private static long SaturatingAdd(long a, long b)
    {
        if (a >= long.MaxValue - b)
        {
            return long.MaxValue;
        }

        return a + b;
    }

    private static void AddPermutationsOfBasis(Vec3 x, Vec3 y, Vec3 z, Action<Mat3> sink)
    {
        Vec3[] axes = { x, y, z };
        int[][] perms =
        {
            new[] { 0, 1, 2 },
            new[] { 0, 2, 1 },
            new[] { 1, 0, 2 },
            new[] { 1, 2, 0 },
            new[] { 2, 0, 1 },
            new[] { 2, 1, 0 }
        };

        int[] signs = { -1, 1 };
        for (int p = 0; p < perms.Length; p++)
        {
            for (int sx = 0; sx < signs.Length; sx++)
            {
                for (int sy = 0; sy < signs.Length; sy++)
                {
                    for (int sz = 0; sz < signs.Length; sz++)
                    {
                        Vec3 r0 = axes[perms[p][0]] * signs[sx];
                        Vec3 r1 = axes[perms[p][1]] * signs[sy];
                        Vec3 r2 = axes[perms[p][2]] * signs[sz];
                        if (Vec3.Dot(Vec3.Cross(r0, r1), r2) <= 0.0)
                        {
                            continue;
                        }

                        sink(new Mat3(r0, r1, r2));
                    }
                }
            }
        }
    }

    private static bool TryComputePrincipalBasis(Vec3[] vertices, out Vec3 x, out Vec3 y, out Vec3 z)
    {
        x = Vec3.UnitX;
        y = Vec3.UnitY;
        z = Vec3.UnitZ;

        if (vertices.Length == 0)
        {
            return false;
        }

        double meanX = 0.0;
        double meanY = 0.0;
        double meanZ = 0.0;
        for (int i = 0; i < vertices.Length; i++)
        {
            meanX += vertices[i].X;
            meanY += vertices[i].Y;
            meanZ += vertices[i].Z;
        }

        double invCount = 1.0 / vertices.Length;
        meanX *= invCount;
        meanY *= invCount;
        meanZ *= invCount;

        double[,] a = new double[3, 3];
        for (int i = 0; i < vertices.Length; i++)
        {
            double dx = vertices[i].X - meanX;
            double dy = vertices[i].Y - meanY;
            double dz = vertices[i].Z - meanZ;
            a[0, 0] += dx * dx;
            a[0, 1] += dx * dy;
            a[0, 2] += dx * dz;
            a[1, 1] += dy * dy;
            a[1, 2] += dy * dz;
            a[2, 2] += dz * dz;
        }

        a[1, 0] = a[0, 1];
        a[2, 0] = a[0, 2];
        a[2, 1] = a[1, 2];

        double[,] v = new double[3, 3];
        v[0, 0] = 1.0;
        v[1, 1] = 1.0;
        v[2, 2] = 1.0;

        for (int iter = 0; iter < 24; iter++)
        {
            int p = 0;
            int q = 1;
            double max = Math.Abs(a[0, 1]);

            double a02 = Math.Abs(a[0, 2]);
            if (a02 > max)
            {
                max = a02;
                p = 0;
                q = 2;
            }

            double a12 = Math.Abs(a[1, 2]);
            if (a12 > max)
            {
                max = a12;
                p = 1;
                q = 2;
            }

            if (max <= 1e-12)
            {
                break;
            }

            double app = a[p, p];
            double aqq = a[q, q];
            double apq = a[p, q];
            double phi = 0.5 * Math.Atan2(2.0 * apq, aqq - app);
            double c = Math.Cos(phi);
            double s = Math.Sin(phi);

            for (int k = 0; k < 3; k++)
            {
                double aik = a[k, p];
                double aiq = a[k, q];
                a[k, p] = c * aik - s * aiq;
                a[k, q] = s * aik + c * aiq;
            }

            for (int k = 0; k < 3; k++)
            {
                double apk = a[p, k];
                double aqk = a[q, k];
                a[p, k] = c * apk - s * aqk;
                a[q, k] = s * apk + c * aqk;
            }

            for (int k = 0; k < 3; k++)
            {
                double vkp = v[k, p];
                double vkq = v[k, q];
                v[k, p] = c * vkp - s * vkq;
                v[k, q] = s * vkp + c * vkq;
            }
        }

        double[] eval = { a[0, 0], a[1, 1], a[2, 2] };
        int[] order = { 0, 1, 2 };
        Array.Sort(eval, order);
        Array.Reverse(order);

        x = new Vec3(v[0, order[0]], v[1, order[0]], v[2, order[0]]).NormalizedOrZero();
        y = new Vec3(v[0, order[1]], v[1, order[1]], v[2, order[1]]).NormalizedOrZero();
        if (x.LengthSquared <= Eps)
        {
            return false;
        }

        y = (y - x * Vec3.Dot(y, x)).NormalizedOrZero();
        if (y.LengthSquared <= Eps)
        {
            y = Vec3.PerpendicularUnit(x);
        }

        z = Vec3.Cross(x, y).NormalizedOrZero();
        if (z.LengthSquared <= Eps)
        {
            return false;
        }

        if (Vec3.Dot(Vec3.Cross(x, y), z) < 0.0)
        {
            z = -z;
        }

        return true;
    }

    private static double ComputeAbsoluteVolume(Vec3[] vertices, Triangle[] faces)
    {
        double volume6 = 0.0;
        for (int i = 0; i < faces.Length; i++)
        {
            Vec3 a = vertices[faces[i].A];
            Vec3 b = vertices[faces[i].B];
            Vec3 c = vertices[faces[i].C];
            volume6 += Vec3.Dot(a, Vec3.Cross(b, c));
        }

        return Math.Abs(volume6) / 6.0;
    }
}

internal readonly struct StrategyConfig
{
    public StrategyConfig(string name, int faceSamples, int rawOrientationCap, int keepOrientations, int branchWidth, int depth, int partitionOrders)
    {
        Name = name;
        FaceSamples = faceSamples;
        RawOrientationCap = rawOrientationCap;
        KeepOrientations = keepOrientations;
        BranchWidth = branchWidth;
        Depth = depth;
        PartitionOrders = partitionOrders;
    }

    public string Name { get; }
    public int FaceSamples { get; }
    public int RawOrientationCap { get; }
    public int KeepOrientations { get; }
    public int BranchWidth { get; }
    public int Depth { get; }
    public int PartitionOrders { get; }

    public StrategyConfig With(int? faceSamples = null, int? rawOrientationCap = null, int? keepOrientations = null, int? branchWidth = null, int? depth = null, int? partitionOrders = null)
    {
        return new StrategyConfig(
            Name,
            faceSamples ?? FaceSamples,
            rawOrientationCap ?? RawOrientationCap,
            keepOrientations ?? KeepOrientations,
            branchWidth ?? BranchWidth,
            depth ?? Depth,

internal readonly struct StateKey : IEquatable<StateKey>
{
    private const double QuantizeScale = 1_000_000.0;

    private readonly long _x;
    private readonly long _y;
    private readonly long _z;
    private readonly int _depth;

    public StateKey(Vec3 size, int depth)
    {
        _x = Quantize(size.X);
        _y = Quantize(size.Y);
        _z = Quantize(size.Z);
        _depth = depth;
    }

    public bool Equals(StateKey other)
    {
        return _x == other._x && _y == other._y && _z == other._z && _depth == other._depth;
    }

    public override bool Equals(object? obj)
    {
        return obj is StateKey other && Equals(other);
    }

    public override int GetHashCode()
    {
        return HashCode.Combine(_x, _y, _z, _depth);
    }

    private static long Quantize(double value)
    {
        return (long)Math.Round(value * QuantizeScale, MidpointRounding.AwayFromZero);
    }
}

internal sealed class CompositeTemplate : PlanTemplate
{
    private readonly PlanTemplate[] _children;

    public CompositeTemplate(PlanTemplate[] children) : base(SumCounts(children))
    {
        _children = children;
    }

    public override void Write(TextWriter writer, Vec3 offset)
    {
        for (int i = 0; i < _children.Length; i++)
        {
            _children[i].Write(writer, offset);
        }
    }

    private static long SumCounts(PlanTemplate[] children)
    {
        long total = 0;
        for (int i = 0; i < children.Length; i++)
        {
            total = AddSaturating(total, children[i].Count);
        }

        return total;
    }

    private static long AddSaturating(long a, long b)
    {
        if (a >= long.MaxValue - b)
        {
            return long.MaxValue;
        }

        return a + b;
    }
}

internal sealed class OffsetTemplate : PlanTemplate
{
    private readonly PlanTemplate _child;
    private readonly Vec3 _offset;

    private OffsetTemplate(PlanTemplate child, Vec3 offset) : base(child.Count)
    {
        _child = child;
        _offset = offset;
    }

    public static PlanTemplate Create(PlanTemplate child, Vec3 offset)
    {
        if (child.Count == 0)
        {
            return EmptyTemplate.Instance;
        }

        if (offset.LengthSquared <= 1e-24)
        {
            return child;
        }

        return new OffsetTemplate(child, offset);
    }

    public override void Write(TextWriter writer, Vec3 offset)
    {
        _child.Write(writer, offset + _offset);
    }
}

internal sealed class BlockTemplate : PlanTemplate
{
    private readonly Vec3 _baseTranslation;
    private readonly Vec3 _step;
    private readonly long _nx;
    private readonly long _ny;
    private readonly long _nz;
    private readonly string _quaternionText;

    public BlockTemplate(Vec3 baseTranslation, Vec3 step, long nx, long ny, long nz, string quaternionText)
        : base(MultiplyCount(nx, ny, nz))
    {
        _baseTranslation = baseTranslation;
        _step = step;
        _nx = nx;
        _ny = ny;
        _nz = nz;
        _quaternionText = quaternionText;
    }

    public override void Write(TextWriter writer, Vec3 offset)
    {
        var sb = new StringBuilder(1 << 16);
        double baseX = offset.X + _baseTranslation.X;
        double baseY = offset.Y + _baseTranslation.Y;
        double baseZ = offset.Z + _baseTranslation.Z;

        for (long iz = 0; iz < _nz; iz++)
        {
            double tz = baseZ + iz * _step.Z;
            for (long iy = 0; iy < _ny; iy++)
            {
                double ty = baseY + iy * _step.Y;
                for (long ix = 0; ix < _nx; ix++)
                {
                    double tx = baseX + ix * _step.X;
                    NumberFormatter.AppendDouble(sb, tx);
                    sb.Append(' ');
                    NumberFormatter.AppendDouble(sb, ty);
                    sb.Append(' ');
                    NumberFormatter.AppendDouble(sb, tz);
                    sb.Append(' ');
                    sb.Append(_quaternionText);
                    sb.Append('\n');

                    if (sb.Length >= 1 << 18)
                    {
                        writer.Write(sb.ToString());
                        sb.Clear();
                    }
                }
            }
        }

        if (sb.Length > 0)
        {
            writer.Write(sb.ToString());
        }
    }

    private static long MultiplyCount(long nx, long ny, long nz)
    {
        if (nx <= 0 || ny <= 0 || nz <= 0)
        {
            return 0;
        }

        if (nx > long.MaxValue / ny)
        {
            return long.MaxValue;
        }

        long xy = nx * ny;
        if (xy > long.MaxValue / nz)
        {
            return long.MaxValue;
        }

        return xy * nz;
    }
}

internal static class NumberFormatter
{
    private static readonly CultureInfo Invariant = CultureInfo.InvariantCulture;

    public static void AppendDouble(StringBuilder sb, double value)
    {
        if (Math.Abs(value) < 1e-12)
        {
            value = 0.0;
        }

        Span<char> buffer = stackalloc char[32];
        if (value.TryFormat(buffer, out int written, "G17".AsSpan(), Invariant))
        {
            sb.Append(buffer.Slice(0, written));
        }
        else
        {
            sb.Append(value.ToString("G17", Invariant));
        }
    }

    public static string Join(double a, double b, double c, double d)
    {
        var sb = new StringBuilder(96);
        AppendDouble(sb, a);
        sb.Append(' ');
        AppendDouble(sb, b);
        sb.Append(' ');
        AppendDouble(sb, c);
        sb.Append(' ');
        AppendDouble(sb, d);
        return sb.ToString();
    }
}

internal readonly struct QuaternionD
{
    public QuaternionD(double w, double x, double y, double z)
    {
        W = w;
        X = x;
        Y = y;
        Z = z;
    }

    public double W { get; }
    public double X { get; }
    public double Y { get; }
    public double Z { get; }

    public QuaternionD Canonicalized()
    {
        QuaternionD q = Normalized();
        if (q.W < 0.0)
        {
            return new QuaternionD(-q.W, -q.X, -q.Y, -q.Z);
        }

        if (Math.Abs(q.W) <= 1e-12)
        {
            if (q.X < 0.0 || (Math.Abs(q.X) <= 1e-12 && (q.Y < 0.0 || (Math.Abs(q.Y) <= 1e-12 && q.Z < 0.0))))
            {
                return new QuaternionD(-q.W, -q.X, -q.Y, -q.Z);
            }
        }

        return q;
    }

    public QuaternionD Normalized()
    {
        double norm = Math.Sqrt(W * W + X * X + Y * Y + Z * Z);
        if (norm <= 1e-18)
        {
            return new QuaternionD(1.0, 0.0, 0.0, 0.0);
        }

        double inv = 1.0 / norm;
        return new QuaternionD(W * inv, X * inv, Y * inv, Z * inv);
    }

    public string QuantizedKey(int digits)
    {
        double scale = Math.Pow(10.0, digits);
        long w = (long)Math.Round(W * scale, MidpointRounding.AwayFromZero);
        long x = (long)Math.Round(X * scale, MidpointRounding.AwayFromZero);
        long y = (long)Math.Round(Y * scale, MidpointRounding.AwayFromZero);
        long z = (long)Math.Round(Z * scale, MidpointRounding.AwayFromZero);
        return string.Concat(w.ToString(CultureInfo.InvariantCulture), "|", x.ToString(CultureInfo.InvariantCulture), "|", y.ToString(CultureInfo.InvariantCulture), "|", z.ToString(CultureInfo.InvariantCulture));
    }

    public static QuaternionD FromMatrix(Mat3 m)
    {
        double trace = m.M00 + m.M11 + m.M22;
        if (trace > 0.0)
        {
            double s = Math.Sqrt(trace + 1.0) * 2.0;
            return new QuaternionD(
                0.25 * s,
                (m.M21 - m.M12) / s,
                (m.M02 - m.M20) / s,
                (m.M10 - m.M01) / s);
        }

        if (m.M00 > m.M11 && m.M00 > m.M22)
        {
            double s = Math.Sqrt(1.0 + m.M00 - m.M11 - m.M22) * 2.0;
            return new QuaternionD(
                (m.M21 - m.M12) / s,
                0.25 * s,
                (m.M01 + m.M10) / s,
                (m.M02 + m.M20) / s);
        }

        if (m.M11 > m.M22)
        {
            double s = Math.Sqrt(1.0 + m.M11 - m.M00 - m.M22) * 2.0;
            return new QuaternionD(
                (m.M02 - m.M20) / s,
                (m.M01 + m.M10) / s,
                0.25 * s,
                (m.M12 + m.M21) / s);
        }

        {
            double s = Math.Sqrt(1.0 + m.M22 - m.M00 - m.M11) * 2.0;
            return new QuaternionD(
                (m.M10 - m.M01) / s,
                (m.M02 + m.M20) / s,
                (m.M12 + m.M21) / s,
                0.25 * s);
        }
    }
}

internal readonly struct Vec3
{
    public Vec3(double x, double y, double z)
    {
        X = x;
        Y = y;
        Z = z;
    }

    public static Vec3 Zero => new Vec3(0.0, 0.0, 0.0);
    public static Vec3 UnitX => new Vec3(1.0, 0.0, 0.0);
    public static Vec3 UnitY => new Vec3(0.0, 1.0, 0.0);
    public static Vec3 UnitZ => new Vec3(0.0, 0.0, 1.0);

    public double X { get; }
    public double Y { get; }
    public double Z { get; }

    public double LengthSquared => X * X + Y * Y + Z * Z;
    public double Length => Math.Sqrt(LengthSquared);
    public double MaxComponent => Math.Max(X, Math.Max(Y, Z));

    public double Get(int axis)
    {
        return axis switch
        {
            0 => X,
            1 => Y,
            _ => Z
        };
    }

    public Vec3 NormalizedOrZero()
    {
        double len = Length;
        if (len <= 1e-18)
        {
            return Zero;
        }

        double inv = 1.0 / len;
        return new Vec3(X * inv, Y * inv, Z * inv);
    }

    public static Vec3 Min(Vec3 a, Vec3 b)
    {
        return new Vec3(Math.Min(a.X, b.X), Math.Min(a.Y, b.Y), Math.Min(a.Z, b.Z));
    }

    public static Vec3 Max(Vec3 a, Vec3 b)
    {
        return new Vec3(Math.Max(a.X, b.X), Math.Max(a.Y, b.Y), Math.Max(a.Z, b.Z));
    }

    public static double Dot(Vec3 a, Vec3 b)
    {
        return a.X * b.X + a.Y * b.Y + a.Z * b.Z;
    }

    public static Vec3 Cross(Vec3 a, Vec3 b)
    {
        return new Vec3(
            a.Y * b.Z - a.Z * b.Y,
            a.Z * b.X - a.X * b.Z,
            a.X * b.Y - a.Y * b.X);
    }

    public static Vec3 PerpendicularUnit(Vec3 v)
    {
        Vec3 basis = Math.Abs(v.X) < 0.5 ? UnitX : UnitY;
        Vec3 perp = Cross(v, basis).NormalizedOrZero();
        if (perp.LengthSquared <= 1e-18)
        {
            perp = Cross(v, UnitZ).NormalizedOrZero();
        }

        return perp.LengthSquared <= 1e-18 ? UnitX : perp;
    }

    public static Vec3 operator +(Vec3 a, Vec3 b)
    {
        return new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    }

    public static Vec3 operator -(Vec3 a, Vec3 b)
    {
        return new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    }

    public static Vec3 operator -(Vec3 v)
    {
        return new Vec3(-v.X, -v.Y, -v.Z);
    }

    public static Vec3 operator *(Vec3 v, double s)
    {
        return new Vec3(v.X * s, v.Y * s, v.Z * s);
    }

    public static Vec3 operator *(double s, Vec3 v)
    {
        return v * s;
    }

    public static Vec3 operator /(Vec3 v, double s)
    {
        return new Vec3(v.X / s, v.Y / s, v.Z / s);
    }
}

internal sealed class FastScanner
{
    private readonly Stream _stream;
    private readonly byte[] _buffer = new byte[1 << 16];
    private int _length;
    private int _position;

    public FastScanner(Stream stream)
    {
        _stream = stream;
    }

    public bool TryReadInt(out int value)
    {
        string? token = NextToken();
        if (token == null)
        {
            value = 0;
            return false;
        }

        value = int.Parse(token, CultureInfo.InvariantCulture);
        return true;
    }

    public int NextInt()
    {
        string? token = NextToken();
        if (token == null)
        {
            throw new EndOfStreamException();
        }

        return int.Parse(token, CultureInfo.InvariantCulture);
    }

    public double NextDouble()
    {
        string? token = NextToken();
        if (token == null)
        {
            throw new EndOfStreamException();
        }

        return double.Parse(token, CultureInfo.InvariantCulture);
    }

    private string? NextToken()
    {
        int c;
        do
        {
            c = Read();
            if (c < 0)
            {
                return null;
            }
        }
        while (char.IsWhiteSpace((char)c));

        var sb = new StringBuilder(32);
        do
        {
            sb.Append((char)c);
            c = Read();
        }
        while (c >= 0 && !char.IsWhiteSpace((char)c));

        return sb.ToString();
    }

    private int Read()
    {
        if (_position >= _length)
        {
            _length = _stream.Read(_buffer, 0, _buffer.Length);
            _position = 0;
            if (_length == 0)
            {
                return -1;
            }
        }

        return _buffer[_position++];
    }
}
