use std::collections::{HashSet, VecDeque};
use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }

    let mut it = input.split_whitespace();
    let width: usize = it.next().unwrap().parse().unwrap();
    let height: usize = it.next().unwrap().parse().unwrap();
    let obstacle_count: usize = it.next().unwrap().parse().unwrap();
    let require_cycle = it.next().unwrap().parse::<u32>().unwrap() != 0;

    if width == 0 || height == 0 {
        println!("0");
        return;
    }

    let mut blocked = HashSet::with_capacity(obstacle_count.saturating_mul(2) + 1);
    for _ in 0..obstacle_count {
        let x: usize = it.next().unwrap().parse().unwrap();
        let y: usize = it.next().unwrap().parse().unwrap();
        if x < width && y < height {
            blocked.insert(y * width + x);
        }
    }

    if blocked.contains(&0) {
        println!("0");
        return;
    }

    if blocked.is_empty() {
        if require_cycle {
            if let Some(path) = rectangle_returnable_path(width, height) {
                print_path(&path);
            } else {
                println!("0");
            }
        } else {
            let path = rectangle_path(width, height);
            print_path(&path);
        }
        return;
    }

    if width == 1 || height == 1 {
        if let Some(path) = line_with_obstacles(width, height, &blocked, require_cycle) {
            print_path(&path);
        } else {
            println!("0");
        }
        return;
    }

    let area = match width.checked_mul(height) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let graph = match Graph::from_grid(width, height, area, &blocked) {
        Some(g) => g,
        None => {
            println!("0");
            return;
        }
    };

    let mut solver = Solver::new(graph, require_cycle);
    if let Some(path_ids) = solver.solve() {
        let path: Vec<(usize, usize)> = path_ids
            .into_iter()
            .map(|id| solver.graph.coords[id])
            .collect();
        print_path(&path);
    } else {
        println!("0");
    }
}

fn print_path(path: &[(usize, usize)]) {
    let mut out = String::new();
    out.push_str(&format!("{}\n", path.len()));
    for &(x, y) in path {
        out.push_str(&format!("{x} {y}\n"));
    }
    print!("{out}");
}

fn rectangle_path(width: usize, height: usize) -> Vec<(usize, usize)> {
    let mut path = Vec::with_capacity(width.saturating_mul(height));
    for y in 0..height {
        if y % 2 == 0 {
            for x in 0..width {
                path.push((x, y));
            }
        } else {
            for x in (0..width).rev() {
                path.push((x, y));
            }
        }
    }
    path
}

fn rectangle_returnable_path(width: usize, height: usize) -> Option<Vec<(usize, usize)>> {
    let cells = width.checked_mul(height)?;
    if cells == 2 {
        return Some(rectangle_path(width, height));
    }
    if width == 1 || height == 1 || cells % 2 == 1 {
        return None;
    }

    let mut path = Vec::with_capacity(cells);
    if height % 2 == 0 {
        for x in 0..width {
            path.push((x, 0));
        }
        for y in 1..height {
            if y % 2 == 1 {
                for x in (1..width).rev() {
                    path.push((x, y));
                }
            } else {
                for x in 1..width {
                    path.push((x, y));
                }
            }
        }
        for y in (1..height).rev() {
            path.push((0, y));
        }
    } else {
        for y in 0..height {
            path.push((0, y));
        }
        for x in 1..width {
            if x % 2 == 1 {
                for y in (1..height).rev() {
                    path.push((x, y));
                }
            } else {
                for y in 1..height {
                    path.push((x, y));
                }
            }
        }
        for x in (1..width).rev() {
            path.push((x, 0));
        }
    }

    if is_adjacent(path[0], *path.last().unwrap()) {
        Some(path)
    } else {
        None
    }
}

fn line_with_obstacles(
    width: usize,
    height: usize,
    blocked: &HashSet<usize>,
    require_cycle: bool,
) -> Option<Vec<(usize, usize)>> {
    let len = width.checked_mul(height)?;
    let total_free = len.checked_sub(blocked.len())?;
    if total_free == 0 {
        return None;
    }

    let mut path = Vec::with_capacity(total_free);
    if height == 1 {
        for x in 0..width {
            if blocked.contains(&x) {
                break;
            }
            path.push((x, 0));
        }
    } else {
        for y in 0..height {
            let idx = y * width;
            if blocked.contains(&idx) {
                break;
            }
            path.push((0, y));
        }
    }

    if path.len() != total_free {
        return None;
    }
    if require_cycle {
        if path.len() == 2 && is_adjacent(path[0], path[1]) {
            Some(path)
        } else {
            None
        }
    } else {
        Some(path)
    }
}

fn is_adjacent(a: (usize, usize), b: (usize, usize)) -> bool {
    let dx = a.0.max(b.0) - a.0.min(b.0);
    let dy = a.1.max(b.1) - a.1.min(b.1);
    dx + dy == 1
}

struct Graph {
    coords: Vec<(usize, usize)>,
    adj: Vec<Vec<usize>>,
    colors: Vec<usize>,
    start: usize,
}

impl Graph {
    fn from_grid(
        width: usize,
        height: usize,
        area: usize,
        blocked: &HashSet<usize>,
    ) -> Option<Self> {
        let mut id_at = vec![usize::MAX; area];
        let mut coords = Vec::with_capacity(area.saturating_sub(blocked.len()));

        for y in 0..height {
            for x in 0..width {
                let idx = y * width + x;
                if !blocked.contains(&idx) {
                    id_at[idx] = coords.len();
                    coords.push((x, y));
                }
            }
        }

        let start = id_at[0];
        if start == usize::MAX {
            return None;
        }

        let mut adj = vec![Vec::with_capacity(4); coords.len()];
        for y in 0..height {
            for x in 0..width {
                let idx = y * width + x;
                let id = id_at[idx];
                if id == usize::MAX {
                    continue;
                }
                if x + 1 < width {
                    let right = idx + 1;
                    let rid = id_at[right];
                    if rid != usize::MAX {
                        adj[id].push(rid);
                        adj[rid].push(id);
                    }
                }
                if y + 1 < height {
                    let down = idx + width;
                    let did = id_at[down];
                    if did != usize::MAX {
                        adj[id].push(did);
                        adj[did].push(id);
                    }
                }
            }
        }

        let colors = coords
            .iter()
            .map(|&(x, y)| (x + y) & 1)
            .collect::<Vec<_>>();

        Some(Self {
            coords,
            adj,
            colors,
            start,
        })
    }
}

struct Solver {
    graph: Graph,
    require_cycle: bool,
    adj_to_start: Vec<bool>,
    visited: Vec<u8>,
    path: Vec<usize>,
    rem_color: [usize; 2],
    rem_target_color: [usize; 2],
    bfs_seen: Vec<u32>,
    bfs_token: u32,
    active_list: Vec<usize>,
    active_degree: Vec<u8>,
    disc: Vec<i32>,
    low: Vec<i32>,
    sep_children: Vec<usize>,
    root_children: Vec<usize>,
}

impl Solver {
    fn new(graph: Graph, require_cycle: bool) -> Self {
        let n = graph.coords.len();
        let start = graph.start;
        let mut adj_to_start = vec![false; n];
        for &v in &graph.adj[start] {
            adj_to_start[v] = true;
        }

        let mut rem_color = [0usize; 2];
        let mut rem_target_color = [0usize; 2];
        for v in 0..n {
            if v == start {
                continue;
            }
            let c = graph.colors[v];
            rem_color[c] += 1;
            if adj_to_start[v] {
                rem_target_color[c] += 1;
            }
        }

        let mut visited = vec![0u8; n];
        visited[start] = 1;

        Self {
            graph,
            require_cycle,
            adj_to_start,
            visited,
            path: vec![start],
            rem_color,
            rem_target_color,
            bfs_seen: vec![0; n],
            bfs_token: 0,
            active_list: Vec::with_capacity(n),
            active_degree: vec![0; n],
            disc: vec![-1; n],
            low: vec![0; n],
            sep_children: vec![0; n],
            root_children: vec![0; n],
        }
    }

    fn solve(&mut self) -> Option<Vec<usize>> {
        if !self.quick_check(self.graph.start) {
            return None;
        }
        if self.dfs(self.graph.start) {
            Some(self.path.clone())
        } else {
            None
        }
    }

    fn dfs(&mut self, cur: usize) -> bool {
        if self.path.len() == self.graph.coords.len() {
            return !self.require_cycle || self.adj_to_start[cur];
        }

        let candidates = self.next_candidates(cur);
        for next in candidates {
            self.visit(next);
            let ok = self.quick_check(next) && self.dfs(next);
            if ok {
                return true;
            }
            self.unvisit(next);
        }
        false
    }

    fn visit(&mut self, v: usize) {
        self.visited[v] = 1;
        self.path.push(v);
        let c = self.graph.colors[v];
        self.rem_color[c] -= 1;
        if self.adj_to_start[v] {
            self.rem_target_color[c] -= 1;
        }
    }

    fn unvisit(&mut self, v: usize) {
        let popped = self.path.pop().unwrap();
        debug_assert_eq!(popped, v);
        let c = self.graph.colors[v];
        self.rem_color[c] += 1;
        if self.adj_to_start[v] {
            self.rem_target_color[c] += 1;
        }
        self.visited[v] = 0;
    }

    fn next_candidates(&self, cur: usize) -> Vec<usize> {
        let remaining_before_move = self.graph.coords.len() - self.path.len();
        let mut scored = Vec::new();

        for &next in &self.graph.adj[cur] {
            if self.visited[next] != 0 {
                continue;
            }

            let onward = self.graph.adj[next]
                .iter()
                .filter(|&&u| self.visited[u] == 0 && u != cur)
                .count();

            if remaining_before_move > 1 && onward == 0 {
                continue;
            }

            let local_tightness = self.graph.adj[next]
                .iter()
                .filter(|&&u| self.visited[u] == 0 && u != cur)
                .map(|&u| {
                    self.graph.adj[u]
                        .iter()
                        .filter(|&&w| self.visited[w] == 0 && w != next)
                        .count()
                })
                .min()
                .unwrap_or(usize::MAX / 4);

            let postpone_target = if self.require_cycle
                && remaining_before_move > 1
                && self.adj_to_start[next]
            {
                1usize
            } else {
                0usize
            };

            scored.push((
                onward,
                local_tightness,
                postpone_target,
                self.graph.adj[next].len(),
                next,
            ));
        }

        scored.sort_unstable();
        scored.into_iter().map(|t| t.4).collect()
    }

    fn quick_check(&mut self, cur: usize) -> bool {
        let active_size = self.graph.coords.len() - self.path.len() + 1;
        let cur_color = self.graph.colors[cur];
        let black = self.rem_color[0] + usize::from(cur_color == 0);
        let white = self.rem_color[1] + usize::from(cur_color == 1);

        if active_size % 2 == 0 {
            if black != white {
                return false;
            }
        } else if cur_color == 0 {
            if black != white + 1 {
                return false;
            }
        } else if white != black + 1 {
            return false;
        }

        if self.require_cycle {
            if active_size == 1 {
                if !self.adj_to_start[cur] {
                    return false;
                }
            } else {
                let target_color = if active_size % 2 == 0 {
                    1 - cur_color
                } else {
                    cur_color
                };
                if self.rem_target_color[target_color] == 0 {
                    return false;
                }
            }
        }

        self.bfs_token = self.bfs_token.wrapping_add(1);
        if self.bfs_token == 0 {
            self.bfs_seen.fill(0);
            self.bfs_token = 1;
        }
        let token = self.bfs_token;

        self.active_list.clear();
        let mut q = VecDeque::new();
        self.bfs_seen[cur] = token;
        q.push_back(cur);

        let mut reached = 0usize;
        let mut deg1_count = 0usize;
        let mut cur_deg = 0usize;

        while let Some(v) = q.pop_front() {
            reached += 1;
            self.active_list.push(v);

            let mut deg = 0usize;
            for i in 0..self.graph.adj[v].len() {
                let to = self.graph.adj[v][i];
                if to == cur || self.visited[to] == 0 {
                    deg += 1;
                    if self.bfs_seen[to] != token {
                        self.bfs_seen[to] = token;
                        q.push_back(to);
                    }
                }
            }

            if active_size > 1 && deg == 0 {
                return false;
            }
            if deg == 1 {
                deg1_count += 1;
            }
            if v == cur {
                cur_deg = deg;
            }
            self.active_degree[v] = deg as u8;
        }

        if reached != active_size {
            return false;
        }
        if deg1_count > 2 {
            return false;
        }
        if deg1_count == 2 && cur_deg != 1 {
            return false;
        }

        if self.require_cycle && active_size > 1 {
            if deg1_count == 1 && cur_deg != 1 {
                let end = self
                    .active_list
                    .iter()
                    .copied()
                    .find(|&v| self.active_degree[v] == 1)
                    .unwrap();
                if !self.adj_to_start[end] {
                    return false;
                }
            } else if deg1_count == 2 && cur_deg == 1 {
                let end = self
                    .active_list
                    .iter()
                    .copied()
                    .find(|&v| v != cur && self.active_degree[v] == 1)
                    .unwrap();
                if !self.adj_to_start[end] {
                    return false;
                }
            }
        }

        if !self.articulation_check(cur, active_size) {
            return false;
        }

        true
    }

    fn articulation_check(&mut self, root: usize, active_size: usize) -> bool {
        for &v in &self.active_list {
            self.disc[v] = -1;
            self.low[v] = 0;
            self.sep_children[v] = 0;
            self.root_children[v] = 0;
        }

        let mut time = 0i32;
        self.tarjan(root, usize::MAX, root, &mut time);

        for &v in &self.active_list {
            let comps_after_removal = if v == root {
                if active_size == 1 {
                    0
                } else {
                    self.root_children[v]
                }
            } else {
                1 + self.sep_children[v]
            };

            let allowed = if v == root { 1 } else { 2 };
            if comps_after_removal > allowed {
                return false;
            }
        }

        true
    }

    fn tarjan(&mut self, v: usize, parent: usize, root: usize, time: &mut i32) {
        self.disc[v] = *time;
        self.low[v] = *time;
        *time += 1;

        let mut child_count = 0usize;
        let mut sep = 0usize;

        for i in 0..self.graph.adj[v].len() {
            let to = self.graph.adj[v][i];
            if to == parent {
                continue;
            }
            if to != root && self.visited[to] != 0 {
                continue;
            }

            if self.disc[to] == -1 {
                child_count += 1;
                self.tarjan(to, v, root, time);
                self.low[v] = self.low[v].min(self.low[to]);
                if parent != usize::MAX && self.low[to] >= self.disc[v] {
                    sep += 1;
                }
            } else {
                self.low[v] = self.low[v].min(self.disc[to]);
            }
        }

        if parent == usize::MAX {
            self.root_children[v] = child_count;
        } else {
            self.sep_children[v] = sep;
        }
    }
}
