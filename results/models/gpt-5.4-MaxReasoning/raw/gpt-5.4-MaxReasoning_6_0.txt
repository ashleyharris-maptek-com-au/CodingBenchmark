using System;
using System.Globalization;
using System.IO;

public static class Program
{
    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        if (!scanner.TryNextInt(out int stationCount))
        {
            return;
        }

        double mu = scanner.NextDouble();
        Orbit start = Orbit.Read(scanner, mu);
        Orbit[] stations = new Orbit[stationCount];
        for (int i = 0; i < stationCount; i++)
        {
            stations[i] = Orbit.Read(scanner, mu);
        }

        if (stationCount == 0)
        {
            Console.WriteLine();
            return;
        }

        double[] startCost = new double[stationCount];
        double[,] cost = new double[stationCount, stationCount];

        for (int i = 0; i < stationCount; i++)
        {
            startCost[i] = TransferCost(start, stations[i], mu);
        }

        for (int i = 0; i < stationCount; i++)
        {
            for (int j = i + 1; j < stationCount; j++)
            {
                double c = TransferCost(stations[i], stations[j], mu);
                cost[i, j] = c;
                cost[j, i] = c;
            }
        }

        int[] route = Solve(stations, startCost, cost);
        var writer = new StringWriter(CultureInfo.InvariantCulture);
        for (int i = 0; i < route.Length; i++)
        {
            if (i > 0)
            {
                writer.Write(' ');
            }

            writer.Write(route[i]);
        }

        Console.WriteLine(writer.ToString());
    }

    private static int[] Solve(Orbit[] stations, double[] startCost, double[,] cost)
    {
        int n = stations.Length;
        if (n == 1)
        {
            return new[] { 0 };
        }

        if (n <= 17)
        {
            return SolveExact(startCost, cost);
        }

        double[] nearest = ComputeNearestNeighborBaseline(startCost, cost);
        int[] seedOrder = SortedIndicesBy(startCost);

        int[]? bestRoute = null;
        double bestScore = double.PositiveInfinity;

        int topSeeds = Math.Min(n, 6);
        for (int s = 0; s < topSeeds; s++)
        {
            int first = seedOrder[s];
            ConsiderCandidate(ref bestRoute, ref bestScore, BuildNearestNeighbor(startCost, cost, nearest, first, 0.00), startCost, cost);
            ConsiderCandidate(ref bestRoute, ref bestScore, BuildNearestNeighbor(startCost, cost, nearest, first, 0.35), startCost, cost);

            if (s < 3)
            {
                ConsiderCandidate(ref bestRoute, ref bestScore, BuildCheapestInsertionWithFirst(startCost, cost, first), startCost, cost);
            }
        }

        ConsiderCandidate(ref bestRoute, ref bestScore, BuildBestPairInsertion(startCost, cost), startCost, cost);
        ConsiderCandidate(ref bestRoute, ref bestScore, BuildNearestNeighbor(startCost, cost, nearest, -1, 0.75), startCost, cost);

        if (n >= 24)
        {
            ConsiderCandidate(ref bestRoute, ref bestScore, BuildFarthestInsertion(startCost, cost, seedOrder[0]), startCost, cost);
        }

        if (bestRoute == null)
        {
            bestRoute = BuildNearestNeighbor(startCost, cost, nearest, -1, 0.25);
            ImproveRoute(bestRoute, startCost, cost);
            bestScore = RouteCost(bestRoute, startCost, cost);
        }

        if (n >= 24)
        {
            var rng = new Random(123456789);
            int attempts = Math.Min(8, 3 + n / 20);
            for (int i = 0; i < attempts; i++)
            {
                int[] perturbed = SegmentKick(bestRoute, rng);
                ImproveRoute(perturbed, startCost, cost);
                double score = RouteCost(perturbed, startCost, cost);
                if (score < bestScore)
                {
                    bestScore = score;
                    bestRoute = perturbed;
                }
            }
        }

        return bestRoute!;
    }

    private static void ConsiderCandidate(ref int[]? bestRoute, ref double bestScore, int[] candidate, double[] startCost, double[,] cost)
    {
        if (candidate == null || candidate.Length == 0)
        {
            return;
        }

        ImproveRoute(candidate, startCost, cost);
        double score = RouteCost(candidate, startCost, cost);
        if (score < bestScore)
        {
            bestScore = score;
            bestRoute = (int[])candidate.Clone();
        }
    }

    private static int[] SolveExact(double[] startCost, double[,] cost)
    {
        int n = startCost.Length;
        int stateCount = 1 << n;
        double[] dp = new double[stateCount * n];
        int[] parent = new int[stateCount * n];

        for (int i = 0; i < dp.Length; i++)
        {
            dp[i] = double.PositiveInfinity;
            parent[i] = -1;
        }

        for (int i = 0; i < n; i++)
        {
            dp[(1 << i) * n + i] = startCost[i];
        }

        for (int mask = 1; mask < stateCount; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                if ((mask & (1 << last)) == 0)
                {
                    continue;
                }

                double current = dp[mask * n + last];
                if (double.IsPositiveInfinity(current))
                {
                    continue;
                }

                for (int next = 0; next < n; next++)
                {
                    int bit = 1 << next;
                    if ((mask & bit) != 0)
                    {
                        continue;
                    }

                    int nextMask = mask | bit;
                    int index = nextMask * n + next;
                    double candidate = current + cost[last, next];
                    if (candidate < dp[index])
                    {
                        dp[index] = candidate;
                        parent[index] = last;
                    }
                }
            }
        }

        int fullMask = stateCount - 1;
        int bestLast = 0;
        double bestScore = double.PositiveInfinity;
        for (int last = 0; last < n; last++)
        {
            double score = dp[fullMask * n + last];
            if (score < bestScore)
            {
                bestScore = score;
                bestLast = last;
            }
        }

        int[] route = new int[n];
        int maskCursor = fullMask;
        int lastCursor = bestLast;
        route[n - 1] = bestLast;

        for (int position = n - 1; position >= 1; position--)
        {
            int previous = parent[maskCursor * n + lastCursor];
            route[position - 1] = previous;
            maskCursor ^= 1 << lastCursor;
            lastCursor = previous;
        }

        return route;
    }

    private static double[] ComputeNearestNeighborBaseline(double[] startCost, double[,] cost)
    {
        int n = startCost.Length;
        double[] nearest = new double[n];
        for (int i = 0; i < n; i++)
        {
            double best = startCost[i];
            for (int j = 0; j < n; j++)
            {
                if (i == j)
                {
                    continue;
                }

                if (cost[i, j] < best)
                {
                    best = cost[i, j];
                }
            }

            nearest[i] = best;
        }

        return nearest;
    }

    private static int[] SortedIndicesBy(double[] values)
    {
        int[] indices = new int[values.Length];
        for (int i = 0; i < indices.Length; i++)
        {
            indices[i] = i;
        }

        Array.Sort(indices, (a, b) => values[a].CompareTo(values[b]));
        return indices;
    }

    private static int[] BuildNearestNeighbor(double[] startCost, double[,] cost, double[] nearest, int forcedFirst, double lookaheadWeight)
    {
        int n = startCost.Length;
        bool[] used = new bool[n];
        int[] route = new int[n];
        int count = 0;

        int current = forcedFirst;
        if (current < 0)
        {
            double bestScore = double.PositiveInfinity;
            for (int i = 0; i < n; i++)
            {
                double score = startCost[i] + lookaheadWeight * nearest[i];
                if (score < bestScore)
                {
                    bestScore = score;
                    current = i;
                }
            }
        }

        route[count++] = current;
        used[current] = true;

        while (count < n)
        {
            int nextNode = -1;
            double bestScore = double.PositiveInfinity;
            for (int candidate = 0; candidate < n; candidate++)
            {
                if (used[candidate])
                {
                    continue;
                }

                double score = cost[current, candidate] + lookaheadWeight * nearest[candidate];
                if (score < bestScore)
                {
                    bestScore = score;
                    nextNode = candidate;
                }
            }

            route[count++] = nextNode;
            used[nextNode] = true;
            current = nextNode;
        }

        return route;
    }

    private static int[] BuildBestPairInsertion(double[] startCost, double[,] cost)
    {
        int n = startCost.Length;
        if (n == 1)
        {
            return new[] { 0 };
        }

        int first = 0;
        int second = 1;
        double bestPair = double.PositiveInfinity;
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (i == j)
                {
                    continue;
                }

                double score = startCost[i] + cost[i, j];
                if (score < bestPair)
                {
                    bestPair = score;
                    first = i;
                    second = j;
                }
            }
        }

        bool[] used = new bool[n];
        int[] route = new int[n];
        route[0] = first;
        route[1] = second;
        used[first] = true;
        used[second] = true;
        CompleteByCheapestInsertion(route, 2, used, startCost, cost);
        return route;
    }

    private static int[] BuildCheapestInsertionWithFirst(double[] startCost, double[,] cost, int first)
    {
        int n = startCost.Length;
        int[] route = new int[n];
        bool[] used = new bool[n];
        route[0] = first;
        used[first] = true;

        int count = 1;
        if (n > 1)
        {
            int second = -1;
            double best = double.PositiveInfinity;
            for (int i = 0; i < n; i++)
            {
                if (used[i])
                {
                    continue;
                }

                if (cost[first, i] < best)
                {
                    best = cost[first, i];
                    second = i;
                }
            }

            route[1] = second;
            used[second] = true;
            count = 2;
        }

        CompleteByCheapestInsertion(route, count, used, startCost, cost);
        return route;
    }

    private static int[] BuildFarthestInsertion(double[] startCost, double[,] cost, int seedFirst)
    {
        int n = startCost.Length;
        int[] route = new int[n];
        bool[] used = new bool[n];
        route[0] = seedFirst;
        used[seedFirst] = true;
        int count = 1;

        if (n > 1)
        {
            int second = -1;
            double farthest = double.NegativeInfinity;
            for (int i = 0; i < n; i++)
            {
                if (used[i])
                {
                    continue;
                }

                double score = Math.Min(startCost[i], cost[seedFirst, i]);
                if (score > farthest)
                {
                    farthest = score;
                    second = i;
                }
            }

            route[1] = second;
            used[second] = true;
            count = 2;
        }

        while (count < n)
        {
            int chosen = -1;
            double farthestDistance = double.NegativeInfinity;

            for (int candidate = 0; candidate < n; candidate++)
            {
                if (used[candidate])
                {
                    continue;
                }

                double distance = startCost[candidate];
                for (int i = 0; i < count; i++)
                {
                    double c = cost[candidate, route[i]];
                    if (c < distance)
                    {
                        distance = c;
                    }
                }

                if (distance > farthestDistance)
                {
                    farthestDistance = distance;
                    chosen = candidate;
                }
            }

            InsertAtBestPosition(route, ref count, chosen, startCost, cost);
            used[chosen] = true;
        }

        return route;
    }

    private static void CompleteByCheapestInsertion(int[] route, int count, bool[] used, double[] startCost, double[,] cost)
    {
        int n = route.Length;
        while (count < n)
        {
            int chosenNode = -1;
            int chosenPosition = 0;
            double bestIncrease = double.PositiveInfinity;

            for (int candidate = 0; candidate < n; candidate++)
            {
                if (used[candidate])
                {
                    continue;
                }

                double increase = startCost[candidate] + cost[candidate, route[0]] - startCost[route[0]];
                if (increase < bestIncrease)
                {
                    bestIncrease = increase;
                    chosenNode = candidate;
                    chosenPosition = 0;
                }

                for (int position = 1; position < count; position++)
                {
                    int left = route[position - 1];
                    int right = route[position];
                    increase = cost[left, candidate] + cost[candidate, right] - cost[left, right];
                    if (increase < bestIncrease)
                    {
                        bestIncrease = increase;
                        chosenNode = candidate;
                        chosenPosition = position;
                    }
                }

                increase = cost[route[count - 1], candidate];
                if (increase < bestIncrease)
                {
                    bestIncrease = increase;
                    chosenNode = candidate;
                    chosenPosition = count;
                }
            }

            used[chosenNode] = true;
            for (int i = count; i > chosenPosition; i--)
            {
                route[i] = route[i - 1];
            }

            route[chosenPosition] = chosenNode;
            count++;
        }
    }

    private static void InsertAtBestPosition(int[] route, ref int count, int node, double[] startCost, double[,] cost)
    {
        int bestPosition = 0;
        double bestIncrease = startCost[node] + cost[node, route[0]] - startCost[route[0]];

        for (int position = 1; position < count; position++)
        {
            int left = route[position - 1];
            int right = route[position];
            double increase = cost[left, node] + cost[node, right] - cost[left, right];
            if (increase < bestIncrease)
            {
                bestIncrease = increase;
                bestPosition = position;
            }
        }

        double endIncrease = cost[route[count - 1], node];
        if (endIncrease < bestIncrease)
        {
            bestPosition = count;
        }

        for (int i = count; i > bestPosition; i--)
        {
            route[i] = route[i - 1];
        }

        route[bestPosition] = node;
        count++;
    }

    private static void ImproveRoute(int[] route, double[] startCost, double[,] cost)
    {
        int guard = Math.Max(16, route.Length * 4);
        for (int iteration = 0; iteration < guard; iteration++)
        {
            bool improved = false;
            if (RelocateBestImprovement(route, startCost, cost))
            {
                improved = true;
            }

            if (TwoOptBestImprovement(route, startCost, cost))
            {
                improved = true;
            }

            if (!improved)
            {
                break;
            }
        }
    }

    private static bool RelocateBestImprovement(int[] route, double[] startCost, double[,] cost)
    {
        int n = route.Length;
        if (n < 3)
        {
            return false;
        }

        double bestDelta = -1e-9;
        int bestFrom = -1;
        int bestPos = -1;

        for (int from = 0; from < n; from++)
        {
            int node = route[from];
            int prev = from == 0 ? -1 : route[from - 1];
            int next = from + 1 < n ? route[from + 1] : -1;
            double removeDelta = LinkCost(prev, next, startCost, cost)
                               - LinkCost(prev, node, startCost, cost)
                               - LinkCost(node, next, startCost, cost);

            int reducedLength = n - 1;
            for (int pos = 0; pos <= reducedLength; pos++)
            {
                if (pos == from)
                {
                    continue;
                }

                int left = pos == 0 ? -1 : ReducedAt(route, from, pos - 1);
                int right = pos == reducedLength ? -1 : ReducedAt(route, from, pos);
                double insertDelta = LinkCost(left, node, startCost, cost)
                                   + LinkCost(node, right, startCost, cost)
                                   - LinkCost(left, right, startCost, cost);

                double totalDelta = removeDelta + insertDelta;
                if (totalDelta < bestDelta)
                {
                    bestDelta = totalDelta;
                    bestFrom = from;
                    bestPos = pos;
                }
            }
        }

        if (bestFrom < 0)
        {
            return false;
        }

        ApplyRelocate(route, bestFrom, bestPos);
        return true;
    }

    private static bool TwoOptBestImprovement(int[] route, double[] startCost, double[,] cost)
    {
        int n = route.Length;
        if (n < 3)
        {
            return false;
        }

        double bestDelta = -1e-9;
        int bestI = -1;
        int bestJ = -1;

        for (int i = 0; i < n - 1; i++)
        {
            int left = i == 0 ? -1 : route[i - 1];
            int first = route[i];
            for (int j = i + 1; j < n; j++)
            {
                int last = route[j];
                int right = j + 1 < n ? route[j + 1] : -1;

                double before = LinkCost(left, first, startCost, cost) + LinkCost(last, right, startCost, cost);
                double after = LinkCost(left, last, startCost, cost) + LinkCost(first, right, startCost, cost);
                double delta = after - before;

                if (delta < bestDelta)
                {
                    bestDelta = delta;
                    bestI = i;
                    bestJ = j;
                }
            }
        }

        if (bestI < 0)
        {
            return false;
        }

        Array.Reverse(route, bestI, bestJ - bestI + 1);
        return true;
    }

    private static int ReducedAt(int[] route, int removedIndex, int reducedIndex)
    {
        return reducedIndex < removedIndex ? route[reducedIndex] : route[reducedIndex + 1];
    }

    private static void ApplyRelocate(int[] route, int from, int pos)
    {
        int node = route[from];
        if (pos < from)
        {
            for (int i = from; i > pos; i--)
            {
                route[i] = route[i - 1];
            }

            route[pos] = node;
        }
        else
        {
            for (int i = from; i < pos; i++)
            {
                route[i] = route[i + 1];
            }

            route[pos] = node;
        }
    }

    private static double RouteCost(int[] route, double[] startCost, double[,] cost)
    {
        if (route.Length == 0)
        {
            return 0.0;
        }

        double total = startCost[route[0]];
        for (int i = 1; i < route.Length; i++)
        {
            total += cost[route[i - 1], route[i]];
        }

        return total;
    }

    private static double LinkCost(int left, int right, double[] startCost, double[,] cost)
    {
        if (right < 0)
        {
            return 0.0;
        }

        if (left < 0)
        {
            return startCost[right];
        }

        return cost[left, right];
    }

    private static int[] SegmentKick(int[] route, Random rng)
    {
        int n = route.Length;
        int[] clone = (int[])route.Clone();
        if (n < 8)
        {
            return clone;
        }

        int a = 1 + rng.Next(Math.Max(1, n / 4));
        int b = a + 1 + rng.Next(Math.Max(1, (n - a) / 3));
        int c = b + 1 + rng.Next(Math.Max(1, (n - b) / 2));

        if (b >= n || c >= n || a >= b || b >= c)
        {
            return clone;
        }

        int[] kicked = new int[n];
        int index = 0;
        CopySlice(clone, 0, a, kicked, ref index);
        CopySlice(clone, b, c, kicked, ref index);
        CopySlice(clone, a, b, kicked, ref index);
        CopySlice(clone, c, n, kicked, ref index);
        return kicked;
    }

    private static void CopySlice(int[] source, int start, int end, int[] destination, ref int destinationIndex)
    {
        for (int i = start; i < end; i++)
        {
            destination[destinationIndex++] = source[i];
        }
    }

    private static double TransferCost(Orbit from, Orbit to, double mu)
    {
        double r1 = from.ReferenceRadius;
        double r2 = to.ReferenceRadius;
        double v1 = from.ReferenceSpeed;
        double v2 = to.ReferenceSpeed;
        double averageSpeed = 0.5 * (v1 + v2);

        double hohmann = HohmannTransfer(r1, r2, mu);
        double planeAngle = Vec3.AngleBetween(from.PlaneNormal, to.PlaneNormal);
        double planeFactor = 0.75 + 0.25 / (1.0 + 4.0 * Math.Abs(r1 - r2) / Math.Max(r1, r2));
        double planeChange = 2.0 * Math.Min(v1, v2) * Math.Sin(0.5 * planeAngle) * planeFactor;

        double eccentricityVectorGap = (from.EccentricityVector - to.EccentricityVector).Length;
        double shapeChange = averageSpeed * 0.55 * eccentricityVectorGap;

        double radialGap = 0.0;
        if (from.ApoapsisRadius < to.PeriapsisRadius)
        {
            radialGap = to.PeriapsisRadius - from.ApoapsisRadius;
        }
        else if (to.ApoapsisRadius < from.PeriapsisRadius)
        {
            radialGap = from.PeriapsisRadius - to.ApoapsisRadius;
        }

        double nonOverlapPenalty = averageSpeed * 0.12 * radialGap / Math.Max(r1, r2);
        double envelopePenalty = averageSpeed * 0.05
                               * (Math.Abs(from.PeriapsisRadius - to.PeriapsisRadius) + Math.Abs(from.ApoapsisRadius - to.ApoapsisRadius))
                               / Math.Max(1.0, from.ReferenceRadius + to.ReferenceRadius);

        double phaseAngle = Vec3.AngleBetween(from.PositionDirection, to.PositionDirection);
        double meanMotionDiff = Math.Abs(from.MeanMotion - to.MeanMotion)
                              / Math.Max(1e-9, 0.5 * (from.MeanMotion + to.MeanMotion));
        double phaseWeight = 0.035 * Math.Exp(-2.25 * planeAngle) / (1.0 + 20.0 * meanMotionDiff);
        double phasePenalty = averageSpeed * phaseWeight * phaseAngle;

        return hohmann + planeChange + shapeChange + nonOverlapPenalty + envelopePenalty + phasePenalty;
    }

    private static double HohmannTransfer(double r1, double r2, double mu)
    {
        r1 = Math.Max(1.0, r1);
        r2 = Math.Max(1.0, r2);
        if (Math.Abs(r1 - r2) < 1e-9)
        {
            return 0.0;
        }

        double v1 = Math.Sqrt(mu / r1);
        double v2 = Math.Sqrt(mu / r2);
        double burn1 = v1 * (Math.Sqrt(2.0 * r2 / (r1 + r2)) - 1.0);
        double burn2 = v2 * (1.0 - Math.Sqrt(2.0 * r1 / (r1 + r2)));
        return Math.Abs(burn1) + Math.Abs(burn2);
    }
}

public sealed class Orbit
{
    public Vec3 Position { get; }
    public Vec3 Velocity { get; }
    public Vec3 PositionDirection { get; }
    public Vec3 PlaneNormal { get; }
    public Vec3 EccentricityVector { get; }
    public double ReferenceRadius { get; }
    public double ReferenceSpeed { get; }
    public double PeriapsisRadius { get; }
    public double ApoapsisRadius { get; }
    public double MeanMotion { get; }

    private Orbit(
        Vec3 position,
        Vec3 velocity,
        Vec3 positionDirection,
        Vec3 planeNormal,
        Vec3 eccentricityVector,
        double referenceRadius,
        double referenceSpeed,
        double periapsisRadius,
        double apoapsisRadius,
        double meanMotion)
    {
        Position = position;
        Velocity = velocity;
        PositionDirection = positionDirection;
        PlaneNormal = planeNormal;
        EccentricityVector = eccentricityVector;
        ReferenceRadius = referenceRadius;
        ReferenceSpeed = referenceSpeed;
        PeriapsisRadius = periapsisRadius;
        ApoapsisRadius = apoapsisRadius;
        MeanMotion = meanMotion;
    }

    public static Orbit Read(FastScanner scanner, double mu)
    {
        Vec3 position = new Vec3(
            scanner.NextDouble(),
            scanner.NextDouble(),
            scanner.NextDouble());

        Vec3 velocity = new Vec3(
            scanner.NextDouble(),
            scanner.NextDouble(),
            scanner.NextDouble());

        double radius = position.Length;
        if (radius < 1e-9)
        {
            radius = 1.0;
        }

        Vec3 positionDirection = position / radius;
        Vec3 angularMomentum = Vec3.Cross(position, velocity);
        double h = angularMomentum.Length;
        Vec3 planeNormal = h > 1e-12 ? angularMomentum / h : new Vec3(0.0, 0.0, 1.0);

        Vec3 eccentricityVector = Vec3.Cross(velocity, angularMomentum) / mu - positionDirection;
        double eccentricity = eccentricityVector.Length;

        double speedSquared = velocity.LengthSquared;
        double specificEnergy = 0.5 * speedSquared - mu / radius;
        double semiMajorAxis = specificEnergy < -1e-10 ? -mu / (2.0 * specificEnergy) : radius;
        if (!double.IsFinite(semiMajorAxis) || semiMajorAxis < 1.0)
        {
            semiMajorAxis = radius;
        }

        double periapsisRadius;
        double apoapsisRadius;
        if (eccentricity < 0.999999)
        {
            periapsisRadius = Math.Max(1.0, semiMajorAxis * (1.0 - eccentricity));
            apoapsisRadius = Math.Max(periapsisRadius, semiMajorAxis * (1.0 + eccentricity));
        }
        else
        {
            periapsisRadius = radius;
            apoapsisRadius = Math.Max(radius, semiMajorAxis);
        }

        double referenceRadius = semiMajorAxis;
        double referenceSpeed = Math.Sqrt(mu / referenceRadius);
        double meanMotion = Math.Sqrt(mu / (referenceRadius * referenceRadius * referenceRadius));

        return new Orbit(
            position,
            velocity,
            positionDirection,
            planeNormal,
            eccentricityVector,
            referenceRadius,
            referenceSpeed,
            periapsisRadius,
            apoapsisRadius,
            meanMotion);
    }
}

public readonly struct Vec3
{
    public readonly double X;
    public readonly double Y;
    public readonly double Z;

    public Vec3(double x, double y, double z)
    {
        X = x;
        Y = y;
        Z = z;
    }

    public double LengthSquared => X * X + Y * Y + Z * Z;
    public double Length => Math.Sqrt(LengthSquared);

    public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    public static Vec3 operator *(Vec3 value, double scale) => new Vec3(value.X * scale, value.Y * scale, value.Z * scale);
    public static Vec3 operator /(Vec3 value, double scale) => new Vec3(value.X / scale, value.Y / scale, value.Z / scale);

    public static double Dot(Vec3 a, Vec3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;

    public static Vec3 Cross(Vec3 a, Vec3 b)
    {
        return new Vec3(
            a.Y * b.Z - a.Z * b.Y,
            a.Z * b.X - a.X * b.Z,
            a.X * b.Y - a.Y * b.X);
    }

    public static double AngleBetween(Vec3 a, Vec3 b)
    {
        double denom = a.Length * b.Length;
        if (denom < 1e-12)
        {
            return 0.0;
        }

        double cosine = Dot(a, b) / denom;
        if (cosine > 1.0)
        {
            cosine = 1.0;
        }
        else if (cosine < -1.0)
        {
            cosine = -1.0;
        }

        return Math.Acos(cosine);
    }
}

public sealed class FastScanner
{
    private readonly Stream _stream;
    private readonly byte[] _buffer = new byte[1 << 16];
    private int _bufferLength;
    private int _bufferIndex;

    public FastScanner(Stream stream)
    {
        _stream = stream;
    }

    public bool TryNextInt(out int value)
    {
        string? token = NextToken();
        if (token == null)
        {
            value = 0;
            return false;
        }

        value = int.Parse(token, CultureInfo.InvariantCulture);
        return true;
    }

    public double NextDouble()
    {
        string? token = NextToken();
        if (token == null)
        {
            throw new EndOfStreamException("Unexpected end of input.");
        }

        return double.Parse(token, NumberStyles.Float, CultureInfo.InvariantCulture);
    }

    private string? NextToken()
    {
        int b;
        do
        {
            b = ReadByte();
            if (b < 0)
            {
                return null;
            }
        }
        while (char.IsWhiteSpace((char)b));

        char[] chars = new char[32];
        int length = 0;
        while (b >= 0 && !char.IsWhiteSpace((char)b))
        {
            if (length == chars.Length)
            {
                Array.Resize(ref chars, chars.Length * 2);
            }

            chars[length++] = (char)b;
            b = ReadByte();
        }

        return new string(chars, 0, length);
    }

    private int ReadByte()
    {
        if (_bufferIndex >= _bufferLength)
        {
            _bufferLength = _stream.Read(_buffer, 0, _buffer.Length);
            _bufferIndex = 0;
            if (_bufferLength == 0)
            {
                return -1;
            }
        }

        return _buffer[_bufferIndex++];
    }
}
