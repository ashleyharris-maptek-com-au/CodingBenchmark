use std::collections::{HashMap, HashSet};
use std::f64::consts::{PI, TAU};
use std::io::{self, Read};
use std::time::{Duration, Instant};

const EPS: f64 = 1.0e-8;

#[derive(Clone, Copy, Debug, Default)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }

    fn cross(self, other: Pt) -> f64 {
        self.x * other.y - self.y * other.x
    }
}

impl std::ops::Add for Pt {
    type Output = Pt;

    fn add(self, rhs: Pt) -> Pt {
        Pt::new(self.x + rhs.x, self.y + rhs.y)
    }
}

impl std::ops::Sub for Pt {
    type Output = Pt;

    fn sub(self, rhs: Pt) -> Pt {
        Pt::new(self.x - rhs.x, self.y - rhs.y)
    }
}

impl std::ops::Mul<f64> for Pt {
    type Output = Pt;

    fn mul(self, rhs: f64) -> Pt {
        Pt::new(self.x * rhs, self.y * rhs)
    }
}

impl std::ops::Div<f64> for Pt {
    type Output = Pt;

    fn div(self, rhs: f64) -> Pt {
        Pt::new(self.x / rhs, self.y / rhs)
    }
}

#[derive(Clone, Copy, Debug)]
struct BBox {
    min: Pt,
    max: Pt,
}

impl BBox {
    fn from_points(points: &[Pt]) -> Self {
        let mut min_x = f64::INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut max_y = f64::NEG_INFINITY;
        for p in points {
            min_x = min_x.min(p.x);
            min_y = min_y.min(p.y);
            max_x = max_x.max(p.x);
            max_y = max_y.max(p.y);
        }
        Self {
            min: Pt::new(min_x, min_y),
            max: Pt::new(max_x, max_y),
        }
    }

    fn width(&self) -> f64 {
        self.max.x - self.min.x
    }

    fn height(&self) -> f64 {
        self.max.y - self.min.y
    }

    fn area(&self) -> f64 {
        self.width().max(0.0) * self.height().max(0.0)
    }

    fn translate(&self, delta: Pt) -> Self {
        Self {
            min: self.min + delta,
            max: self.max + delta,
        }
    }

    fn intersects(&self, other: &Self) -> bool {
        self.min.x <= other.max.x + EPS
            && self.max.x + EPS >= other.min.x
            && self.min.y <= other.max.y + EPS
            && self.max.y + EPS >= other.min.y
    }
}

#[derive(Clone)]
struct Polygon {
    vertices: Vec<Pt>,
    bbox: BBox,
    area: f64,
    probe: Pt,
}

#[derive(Clone)]
struct Piece {
    original_index: usize,
    area: f64,
    vertex_count: usize,
    variants: Vec<PieceVariant>,
}

#[derive(Clone)]
struct PieceVariant {
    angle: f64,
    vertices: Vec<Pt>,
    bbox: BBox,
    probe: Pt,
    anchor_vertices: Vec<Pt>,
    score: f64,
}

#[derive(Clone)]
struct PlacedPiece {
    bbox: BBox,
    vertices: Vec<Pt>,
    probe: Pt,
}

#[derive(Clone, Copy)]
struct OutputPlacement {
    stock_index: usize,
    translation: Pt,
    rotation: f64,
}

struct Candidate {
    stock_index: usize,
    variant_index: usize,
    translation: Pt,
    bbox: BBox,
    rank: CandidateRank,
}

#[derive(Clone, Copy)]
struct CandidateRank {
    primary: f64,
    secondary: f64,
    tertiary: f64,
}

impl CandidateRank {
    fn better_than(&self, other: &Self) -> bool {
        self.primary.total_cmp(&other.primary).is_lt()
            || (self.primary.total_cmp(&other.primary).is_eq()
                && (self.secondary.total_cmp(&other.secondary).is_lt()
                    || (self.secondary.total_cmp(&other.secondary).is_eq()
                        && self.tertiary.total_cmp(&other.tertiary).is_lt())))
    }
}

struct StockState {
    index: usize,
    placed: Vec<PlacedPiece>,
    used_area: f64,
    frontier: Vec<Pt>,
    frontier_seen: HashSet<(i64, i64)>,
    seed_anchors: Vec<Pt>,
    cell_size: f64,
    anchor_quantum: f64,
    frontier_limit: usize,
    grid: HashMap<(i32, i32), Vec<usize>>,
}

impl StockState {
    fn new(index: usize, seed_anchors: &[Pt], settings: &Settings, stock: &Polygon) -> Self {
        let mut state = Self {
            index,
            placed: Vec::new(),
            used_area: 0.0,
            frontier: Vec::new(),
            frontier_seen: HashSet::new(),
            seed_anchors: seed_anchors.to_vec(),
            cell_size: settings.cell_size_hint.max(
                (stock.bbox.width().max(stock.bbox.height()) / 48.0).max(1.0e-3),
            ),
            anchor_quantum: settings.anchor_quantum.max(1.0e-4),
            frontier_limit: settings.frontier_limit,
            grid: HashMap::new(),
        };
        state.push_anchor(stock.bbox.min);
        for seed in seed_anchors.iter().take(settings.seed_refresh) {
            state.push_anchor(*seed);
        }
        state
    }

    fn remaining_area(&self, stock_area: f64) -> f64 {
        stock_area - self.used_area
    }

    fn quantize(&self, p: Pt) -> (i64, i64) {
        (
            (p.x / self.anchor_quantum).round() as i64,
            (p.y / self.anchor_quantum).round() as i64,
        )
    }

    fn push_anchor(&mut self, p: Pt) {
        let key = self.quantize(p);
        if self.frontier_seen.insert(key) {
            self.frontier.push(p);
            if self.frontier.len() > self.frontier_limit * 2 {
                let drop_count = self.frontier.len() - self.frontier_limit;
                self.frontier.drain(0..drop_count);
            }
        }
    }

    fn gather_anchors(&self, settings: &Settings) -> Vec<Pt> {
        let low_limit = settings.anchor_search_limit / 2;
        let recent_limit = settings.anchor_search_limit - low_limit;
        let mut sorted_frontier = self.frontier.clone();
        sorted_frontier.sort_by(|a, b| a.y.total_cmp(&b.y).then_with(|| a.x.total_cmp(&b.x)));

        let mut anchors = Vec::with_capacity(settings.anchor_search_limit + settings.seed_refresh);
        let mut seen = HashSet::new();
        let mut push_unique = |anchor: Pt, anchors: &mut Vec<Pt>| {
            let key = self.quantize(anchor);
            if seen.insert(key) {
                anchors.push(anchor);
            }
        };

        for &anchor in sorted_frontier.iter().take(low_limit) {
            push_unique(anchor, &mut anchors);
        }
        for &anchor in self.frontier.iter().rev().take(recent_limit) {
            push_unique(anchor, &mut anchors);
        }
        let refresh = if self.placed.is_empty() {
            self.seed_anchors.len()
        } else {
            settings.seed_refresh.min(self.seed_anchors.len())
        };
        for &anchor in self.seed_anchors.iter().take(refresh) {
            push_unique(anchor, &mut anchors);
        }
        anchors
    }

    fn insert_placed(&mut self, placed: PlacedPiece) {
        let idx = self.placed.len();
        for cell in cells_for_bbox(&placed.bbox, self.cell_size) {
            self.grid.entry(cell).or_default().push(idx);
        }
        self.used_area += polygon_area(&placed.vertices).abs();
        let bbox = placed.bbox;
        self.push_anchor(Pt::new(bbox.max.x, bbox.min.y));
        self.push_anchor(Pt::new(bbox.min.x, bbox.max.y));
        self.push_anchor(Pt::new(bbox.max.x, bbox.max.y));
        self.placed.push(placed);
    }

    fn overlaps_any(&self, vertices: &[Pt], translation: Pt, probe: Pt, bbox: &BBox) -> bool {
        let mut seen = HashSet::new();
        for cell in cells_for_bbox(bbox, self.cell_size) {
            if let Some(list) = self.grid.get(&cell) {
                for &idx in list {
                    if seen.insert(idx)
                        && polygons_overlap_translated(vertices, translation, probe, bbox, &self.placed[idx])
                    {
                        return true;
                    }
                }
            }
        }
        false
    }
}

#[derive(Clone, Copy, PartialEq, Eq)]
enum Scale {
    Small,
    Medium,
    Large,
}

struct Settings {
    scale: Scale,
    max_rotations: usize,
    piece_edge_refs: usize,
    stock_edge_refs: usize,
    initial_grid: usize,
    dense_grid: usize,
    anchor_search_limit: usize,
    frontier_limit: usize,
    seed_refresh: usize,
    cell_size_hint: f64,
    anchor_quantum: f64,
    soft_deadline: Duration,
}

fn main() {
    let start = Instant::now();
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut scanner = Scanner::new(&input);

    let stock_vertex_count = scanner.next_usize();
    let piece_count = scanner.next_usize();
    let mut stock_vertices = Vec::with_capacity(stock_vertex_count);
    for _ in 0..stock_vertex_count {
        stock_vertices.push(Pt::new(scanner.next_f64(), scanner.next_f64()));
    }
    normalize_polygon(&mut stock_vertices);
    let stock = Polygon::new(stock_vertices);

    let mut raw_pieces = Vec::with_capacity(piece_count);
    let mut total_piece_vertices = 0usize;
    let mut total_piece_area = 0.0;
    let mut max_piece_min_dim: f64 = 0.0;
    let mut min_piece_min_dim: f64 = f64::INFINITY;

    for piece_index in 0..piece_count {
        let vertex_count = scanner.next_usize();
        let mut vertices = Vec::with_capacity(vertex_count);
        for _ in 0..vertex_count {
            vertices.push(Pt::new(scanner.next_f64(), scanner.next_f64()));
        }
        normalize_polygon(&mut vertices);
        let polygon = Polygon::new(vertices);
        let min_dim = polygon.bbox.width().min(polygon.bbox.height());
        max_piece_min_dim = max_piece_min_dim.max(min_dim);
        min_piece_min_dim = min_piece_min_dim.min(min_dim);
        total_piece_area += polygon.area;
        total_piece_vertices += vertex_count;
        raw_pieces.push((piece_index, polygon));
    }

    let settings = build_settings(
        piece_count,
        total_piece_vertices,
        stock.area,
        total_piece_area,
        min_piece_min_dim,
        max_piece_min_dim,
        start,
    );

    let stock_edge_angles = top_edge_angles(&stock.vertices, settings.stock_edge_refs);
    let seed_anchors = build_seed_anchors(&stock, settings.initial_grid);

    let mut pieces = Vec::with_capacity(raw_pieces.len());
    for (original_index, polygon) in raw_pieces {
        let variants = build_piece_variants(&polygon, &stock_edge_angles, &settings);
        pieces.push(Piece {
            original_index,
            area: polygon.area,
            vertex_count: polygon.vertices.len(),
            variants,
        });
    }

    pieces.sort_by(|a, b| {
        b.area
            .total_cmp(&a.area)
            .then_with(|| b.vertex_count.cmp(&a.vertex_count))
            .then_with(|| a.original_index.cmp(&b.original_index))
    });

    let mut placements = vec![
        OutputPlacement {
            stock_index: 0,
            translation: Pt::new(0.0, 0.0),
            rotation: 0.0,
        };
        piece_count
    ];
    let mut stocks: Vec<StockState> = Vec::new();

    for piece in &pieces {
        let elapsed = start.elapsed();
        let candidate = find_existing_stock_candidate(
            &stocks,
            &stock,
            piece,
            &settings,
            elapsed >= settings.soft_deadline,
        );
        let chosen = if let Some(candidate) = candidate {
            candidate
        } else {
            let new_index = stocks.len();
            let empty_state = StockState::new(new_index, &seed_anchors, &settings, &stock);
            search_empty_stock(&empty_state, &stock, piece, &settings).unwrap_or_else(|| {
                panic!(
                    "Failed to place piece {} in an empty stock polygon",
                    piece.original_index
                )
            })
        };

        while stocks.len() <= chosen.stock_index {
            let index = stocks.len();
            stocks.push(StockState::new(index, &seed_anchors, &settings, &stock));
        }

        let variant = &piece.variants[chosen.variant_index];
        let placed_vertices = translate_points(&variant.vertices, chosen.translation);
        let placed_probe = variant.probe + chosen.translation;
        let placed = PlacedPiece {
            bbox: chosen.bbox,
            vertices: placed_vertices,
            probe: placed_probe,
        };
        placements[piece.original_index] = OutputPlacement {
            stock_index: chosen.stock_index,
            translation: chosen.translation,
            rotation: variant.angle,
        };
        stocks[chosen.stock_index].insert_placed(placed);
    }

    println!("{}", stocks.len());
    for placement in placements {
        println!(
            "{} {:.12} {:.12} {:.12}",
            placement.stock_index,
            placement.translation.x,
            placement.translation.y,
            placement.rotation
        );
    }
}

impl Polygon {
    fn new(mut vertices: Vec<Pt>) -> Self {
        normalize_polygon(&mut vertices);
        let area = polygon_area(&vertices).abs();
        let bbox = BBox::from_points(&vertices);
        let probe = find_interior_probe(&vertices);
        Self {
            vertices,
            bbox,
            area,
            probe,
        }
    }
}

fn build_settings(
    piece_count: usize,
    total_piece_vertices: usize,
    stock_area: f64,
    total_piece_area: f64,
    min_piece_min_dim: f64,
    max_piece_min_dim: f64,
    _start: Instant,
) -> Settings {
    let scale = if piece_count <= 80 && total_piece_vertices <= 6_000 {
        Scale::Small
    } else if piece_count <= 900 && total_piece_vertices <= 80_000 {
        Scale::Medium
    } else {
        Scale::Large
    };

    let density_ratio = if stock_area > EPS {
        (total_piece_area / stock_area).max(1.0)
    } else {
        1.0
    };
    let span_hint = if min_piece_min_dim.is_finite() {
        min_piece_min_dim.max(max_piece_min_dim * 0.1).max(1.0e-3)
    } else {
        1.0
    };

    match scale {
        Scale::Small => Settings {
            scale,
            max_rotations: 16,
            piece_edge_refs: 4,
            stock_edge_refs: 8,
            initial_grid: 10,
            dense_grid: 18,
            anchor_search_limit: 160,
            frontier_limit: 2048,
            seed_refresh: 40,
            cell_size_hint: span_hint,
            anchor_quantum: span_hint * 0.08,
            soft_deadline: Duration::from_secs(250),
        },
        Scale::Medium => Settings {
            scale,
            max_rotations: 10,
            piece_edge_refs: 3,
            stock_edge_refs: 6,
            initial_grid: 8,
            dense_grid: 14,
            anchor_search_limit: 96,
            frontier_limit: 1024,
            seed_refresh: 24,
            cell_size_hint: span_hint * (1.0 + density_ratio.sqrt() * 0.02),
            anchor_quantum: span_hint * 0.12,
            soft_deadline: Duration::from_secs(235),
        },
        Scale::Large => Settings {
            scale,
            max_rotations: 6,
            piece_edge_refs: 2,
            stock_edge_refs: 4,
            initial_grid: 6,
            dense_grid: 10,
            anchor_search_limit: 120,
            frontier_limit: 1536,
            seed_refresh: 24,
            cell_size_hint: span_hint * (1.0 + density_ratio.sqrt() * 0.04),
            anchor_quantum: span_hint * 0.16,
            soft_deadline: Duration::from_secs(210),
        },
    }
}

fn build_piece_variants(piece: &Polygon, stock_edges: &[(f64, f64)], settings: &Settings) -> Vec<PieceVariant> {
    let piece_edges = top_edge_angles(&piece.vertices, settings.piece_edge_refs);
    let mut angles = vec![0.0, 0.5 * PI, PI, 1.5 * PI];
    if let Some((angle, _)) = piece_edges.first().copied() {
        angles.push(normalize_angle(-angle));
        angles.push(normalize_angle(0.5 * PI - angle));
    }
    for &(piece_angle, _) in &piece_edges {
        for &(stock_angle, _) in stock_edges {
            angles.push(normalize_angle(stock_angle - piece_angle));
            angles.push(normalize_angle(stock_angle - piece_angle + PI));
        }
    }

    let mut unique = Vec::new();
    for angle in angles {
        let norm = normalize_angle(angle);
        if unique.iter().all(|&existing| angular_distance(existing, norm) > 1.0e-6) {
            unique.push(norm);
        }
    }

    let mut variants = Vec::new();
    for angle in unique {
        let rotated_vertices = rotate_points(&piece.vertices, angle);
        let bbox = BBox::from_points(&rotated_vertices);
        let probe = rotate_point(piece.probe, angle);
        let anchor_vertices = extreme_vertices(&rotated_vertices);
        let score = bbox.area() + bbox.height() * 0.15 + bbox.width() * 0.1;
        variants.push(PieceVariant {
            angle,
            vertices: rotated_vertices,
            bbox,
            probe,
            anchor_vertices,
            score,
        });
    }

    variants.sort_by(|a, b| a.score.total_cmp(&b.score));
    if variants.len() > settings.max_rotations {
        variants.truncate(settings.max_rotations);
    }
    variants
}

fn build_seed_anchors(stock: &Polygon, grid_n: usize) -> Vec<Pt> {
    let mut anchors = Vec::new();
    anchors.push(stock.bbox.min);
    anchors.extend(stock.vertices.iter().copied());
    let gx = grid_n.max(2);
    let gy = grid_n.max(2);
    let width = stock.bbox.width();
    let height = stock.bbox.height();
    for iy in 0..gy {
        let fy = if gy == 1 {
            0.0
        } else {
            iy as f64 / (gy as f64 - 1.0)
        };
        let y = stock.bbox.min.y + height * fy;
        for ix in 0..gx {
            let fx = if gx == 1 {
                0.0
            } else {
                ix as f64 / (gx as f64 - 1.0)
            };
            let x = stock.bbox.min.x + width * fx;
            let point = Pt::new(x, y);
            if point_location(&stock.vertices, point) >= 0 {
                anchors.push(point);
            }
        }
    }
    anchors.sort_by(|a, b| a.y.total_cmp(&b.y).then_with(|| a.x.total_cmp(&b.x)));
    anchors.dedup_by(|a, b| (a.x - b.x).abs() <= 1.0e-8 && (a.y - b.y).abs() <= 1.0e-8);
    anchors
}

fn find_existing_stock_candidate(
    stocks: &[StockState],
    stock: &Polygon,
    piece: &Piece,
    settings: &Settings,
    deadline_mode: bool,
) -> Option<Candidate> {
    let mut best: Option<Candidate> = None;
    for state in stocks {
        if state.remaining_area(stock.area) + EPS < piece.area {
            continue;
        }
        let candidate = search_stock(state, stock, piece, settings, deadline_mode);
        if let Some(candidate) = candidate {
            let replace = match &best {
                None => true,
                Some(current) => candidate.rank.better_than(&current.rank),
            };
            if replace {
                best = Some(candidate);
            }
        }
    }
    best
}

fn search_stock(
    state: &StockState,
    stock: &Polygon,
    piece: &Piece,
    settings: &Settings,
    deadline_mode: bool,
) -> Option<Candidate> {
    let anchors = state.gather_anchors(settings);
    let variant_limit = if deadline_mode && settings.scale == Scale::Large {
        piece.variants.len().min(3)
    } else {
        piece.variants.len()
    };
    let mut best: Option<Candidate> = None;

    for anchor in anchors {
        for (variant_index, variant) in piece.variants.iter().take(variant_limit).enumerate() {
            let translation = anchor - variant.bbox.min;
            let bbox = variant.bbox.translate(translation);
            if !bbox_fits_within_stock_bbox(&bbox, &stock.bbox) {
                continue;
            }
            if !polygon_inside_stock_translated(&variant.vertices, translation, stock) {
                continue;
            }
            if state.overlaps_any(&variant.vertices, translation, variant.probe, &bbox) {
                continue;
            }
            let remaining = (state.remaining_area(stock.area) - piece.area).max(0.0);
            let rank = CandidateRank {
                primary: remaining / stock.area.max(EPS),
                secondary: (bbox.max.y - stock.bbox.min.y) / stock.bbox.height().max(1.0),
                tertiary: (bbox.max.x - stock.bbox.min.x) / stock.bbox.width().max(1.0),
            };
            let candidate = Candidate {
                stock_index: state.index,
                variant_index,
                translation,
                bbox,
                rank,
            };
            let replace = match &best {
                None => true,
                Some(current) => candidate.rank.better_than(&current.rank),
            };
            if replace {
                best = Some(candidate);
            }
        }
    }

    best
}

fn search_empty_stock(
    state: &StockState,
    stock: &Polygon,
    piece: &Piece,
    settings: &Settings,
) -> Option<Candidate> {
    let mut anchors = build_seed_anchors(stock, settings.dense_grid);
    anchors.extend(state.seed_anchors.iter().copied());
    anchors.sort_by(|a, b| a.y.total_cmp(&b.y).then_with(|| a.x.total_cmp(&b.x)));
    anchors.dedup_by(|a, b| (a.x - b.x).abs() <= state.anchor_quantum && (a.y - b.y).abs() <= state.anchor_quantum);

    let mut best: Option<Candidate> = None;
    for (variant_index, variant) in piece.variants.iter().enumerate() {
        for anchor in &anchors {
            let translation = *anchor - variant.bbox.min;
            let bbox = variant.bbox.translate(translation);
            if !bbox_fits_within_stock_bbox(&bbox, &stock.bbox) {
                continue;
            }
            if !polygon_inside_stock_translated(&variant.vertices, translation, stock) {
                continue;
            }
            let candidate = Candidate {
                stock_index: state.index,
                variant_index,
                translation,
                bbox,
                rank: CandidateRank {
                    primary: (bbox.max.y - stock.bbox.min.y) / stock.bbox.height().max(1.0),
                    secondary: (bbox.max.x - stock.bbox.min.x) / stock.bbox.width().max(1.0),
                    tertiary: variant.score,
                },
            };
            let replace = match &best {
                None => true,
                Some(current) => candidate.rank.better_than(&current.rank),
            };
            if replace {
                best = Some(candidate);
            }
        }

        for stock_vertex in &stock.vertices {
            for piece_vertex in &variant.anchor_vertices {
                let translation = *stock_vertex - *piece_vertex;
                let bbox = variant.bbox.translate(translation);
                if !bbox_fits_within_stock_bbox(&bbox, &stock.bbox) {
                    continue;
                }
                if !polygon_inside_stock_translated(&variant.vertices, translation, stock) {
                    continue;
                }
                let candidate = Candidate {
                    stock_index: state.index,
                    variant_index,
                    translation,
                    bbox,
                    rank: CandidateRank {
                        primary: (bbox.max.y - stock.bbox.min.y) / stock.bbox.height().max(1.0),
                        secondary: (bbox.max.x - stock.bbox.min.x) / stock.bbox.width().max(1.0),
                        tertiary: variant.score * 0.5,
                    },
                };
                let replace = match &best {
                    None => true,
                    Some(current) => candidate.rank.better_than(&current.rank),
                };
                if replace {
                    best = Some(candidate);
                }
            }
        }
    }
    best
}

fn bbox_fits_within_stock_bbox(piece_bbox: &BBox, stock_bbox: &BBox) -> bool {
    piece_bbox.min.x + EPS >= stock_bbox.min.x
        && piece_bbox.min.y + EPS >= stock_bbox.min.y
        && piece_bbox.max.x <= stock_bbox.max.x + EPS
        && piece_bbox.max.y <= stock_bbox.max.y + EPS
}

fn top_edge_angles(vertices: &[Pt], limit: usize) -> Vec<(f64, f64)> {
    let mut edges = Vec::new();
    for i in 0..vertices.len() {
        let a = vertices[i];
        let b = vertices[(i + 1) % vertices.len()];
        let dx = b.x - a.x;
        let dy = b.y - a.y;
        let len = (dx * dx + dy * dy).sqrt();
        if len > EPS {
            edges.push((normalize_angle(dy.atan2(dx)), len));
        }
    }
    edges.sort_by(|a, b| b.1.total_cmp(&a.1));
    if edges.len() > limit {
        edges.truncate(limit);
    }
    edges
}

fn extreme_vertices(vertices: &[Pt]) -> Vec<Pt> {
    let mut out = Vec::new();
    let mut push_unique = |p: Pt| {
        if out.iter().all(|q: &Pt| (q.x - p.x).abs() > 1.0e-8 || (q.y - p.y).abs() > 1.0e-8) {
            out.push(p);
        }
    };

    if let Some(p) = vertices.iter().min_by(|a, b| a.x.total_cmp(&b.x)) {
        push_unique(*p);
    }
    if let Some(p) = vertices.iter().min_by(|a, b| a.y.total_cmp(&b.y)) {
        push_unique(*p);
    }
    if let Some(p) = vertices.iter().min_by(|a, b| (a.x + a.y).total_cmp(&(b.x + b.y))) {
        push_unique(*p);
    }
    if let Some(p) = vertices
        .iter()
        .min_by(|a, b| (a.x - a.y).total_cmp(&(b.x - b.y)))
    {
        push_unique(*p);
    }
    out
}

fn normalize_polygon(vertices: &mut Vec<Pt>) {
    if polygon_area(vertices) < 0.0 {
        vertices.reverse();
    }
}

fn polygon_area(vertices: &[Pt]) -> f64 {
    let mut area = 0.0;
    for i in 0..vertices.len() {
        area += vertices[i].cross(vertices[(i + 1) % vertices.len()]);
    }
    area * 0.5
}

fn rotate_point(p: Pt, angle: f64) -> Pt {
    let (s, c) = angle.sin_cos();
    Pt::new(c * p.x - s * p.y, s * p.x + c * p.y)
}

fn rotate_points(points: &[Pt], angle: f64) -> Vec<Pt> {
    points.iter().copied().map(|p| rotate_point(p, angle)).collect()
}

fn translate_points(points: &[Pt], delta: Pt) -> Vec<Pt> {
    points.iter().copied().map(|p| p + delta).collect()
}

fn angular_distance(a: f64, b: f64) -> f64 {
    let diff = (a - b).abs();
    diff.min(TAU - diff)
}

fn normalize_angle(mut angle: f64) -> f64 {
    while angle < 0.0 {
        angle += TAU;
    }
    while angle >= TAU {
        angle -= TAU;
    }
    angle
}

fn find_interior_probe(vertices: &[Pt]) -> Pt {
    let n = vertices.len();
    if n < 3 {
        return vertices[0];
    }
    for i in 0..n {
        let prev = vertices[(i + n - 1) % n];
        let curr = vertices[i];
        let next = vertices[(i + 1) % n];
        if orient(prev, curr, next) <= EPS {
            continue;
        }
        let mut ear = true;
        for (j, &p) in vertices.iter().enumerate() {
            if j == i || j == (i + 1) % n || j == (i + n - 1) % n {
                continue;
            }
            if point_in_triangle(p, prev, curr, next) {
                ear = false;
                break;
            }
        }
        if ear {
            return (prev + curr + next) / 3.0;
        }
    }
    let mut sum = Pt::new(0.0, 0.0);
    for &p in vertices {
        sum = sum + p;
    }
    sum / n as f64
}

fn point_in_triangle(p: Pt, a: Pt, b: Pt, c: Pt) -> bool {
    let ab = orient(a, b, p);
    let bc = orient(b, c, p);
    let ca = orient(c, a, p);
    (ab >= -EPS && bc >= -EPS && ca >= -EPS) || (ab <= EPS && bc <= EPS && ca <= EPS)
}

fn orient(a: Pt, b: Pt, c: Pt) -> f64 {
    (b - a).cross(c - a)
}

fn point_on_segment(a: Pt, b: Pt, p: Pt) -> bool {
    orient(a, b, p).abs() <= EPS
        && p.x >= a.x.min(b.x) - EPS
        && p.x <= a.x.max(b.x) + EPS
        && p.y >= a.y.min(b.y) - EPS
        && p.y <= a.y.max(b.y) + EPS
}

fn point_location(vertices: &[Pt], p: Pt) -> i8 {
    let mut inside = false;
    for i in 0..vertices.len() {
        let a = vertices[i];
        let b = vertices[(i + 1) % vertices.len()];
        if point_on_segment(a, b, p) {
            return 0;
        }
        let crosses = (a.y > p.y) != (b.y > p.y);
        if crosses {
            let x = a.x + (b.x - a.x) * (p.y - a.y) / (b.y - a.y);
            if x > p.x + EPS {
                inside = !inside;
            }
        }
    }
    if inside { 1 } else { -1 }
}

fn point_location_translated(vertices: &[Pt], translation: Pt, p: Pt) -> i8 {
    let mut inside = false;
    for i in 0..vertices.len() {
        let a = vertices[i] + translation;
        let b = vertices[(i + 1) % vertices.len()] + translation;
        if point_on_segment(a, b, p) {
            return 0;
        }
        let crosses = (a.y > p.y) != (b.y > p.y);
        if crosses {
            let x = a.x + (b.x - a.x) * (p.y - a.y) / (b.y - a.y);
            if x > p.x + EPS {
                inside = !inside;
            }
        }
    }
    if inside { 1 } else { -1 }
}

fn segments_properly_intersect(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    if !ranges_overlap(a.x, b.x, c.x, d.x) || !ranges_overlap(a.y, b.y, c.y, d.y) {
        return false;
    }
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);

    let s1 = sign_eps(o1);
    let s2 = sign_eps(o2);
    let s3 = sign_eps(o3);
    let s4 = sign_eps(o4);

    s1 != 0 && s2 != 0 && s3 != 0 && s4 != 0 && s1 != s2 && s3 != s4
}

fn sign_eps(v: f64) -> i32 {
    if v > EPS {
        1
    } else if v < -EPS {
        -1
    } else {
        0
    }
}

fn ranges_overlap(a1: f64, a2: f64, b1: f64, b2: f64) -> bool {
    a1.min(a2) <= b1.max(b2) + EPS && b1.min(b2) <= a1.max(a2) + EPS
}

fn polygon_inside_stock_translated(vertices: &[Pt], translation: Pt, stock: &Polygon) -> bool {
    for &vertex in vertices {
        if point_location(&stock.vertices, vertex + translation) < 0 {
            return false;
        }
    }
    for i in 0..vertices.len() {
        let a = vertices[i] + translation;
        let b = vertices[(i + 1) % vertices.len()] + translation;
        let midpoint = (a + b) * 0.5;
        if point_location(&stock.vertices, midpoint) < 0 {
            return false;
        }
        for j in 0..stock.vertices.len() {
            let c = stock.vertices[j];
            let d = stock.vertices[(j + 1) % stock.vertices.len()];
            if segments_properly_intersect(a, b, c, d) {
                return false;
            }
        }
    }
    true
}

fn polygons_overlap_translated(
    vertices: &[Pt],
    translation: Pt,
    probe: Pt,
    bbox: &BBox,
    other: &PlacedPiece,
) -> bool {
    if !bbox.intersects(&other.bbox) {
        return false;
    }

    for i in 0..vertices.len() {
        let a = vertices[i] + translation;
        let b = vertices[(i + 1) % vertices.len()] + translation;
        let midpoint = (a + b) * 0.5;
        if point_location(&other.vertices, midpoint) > 0 {
            return true;
        }
        for j in 0..other.vertices.len() {
            let c = other.vertices[j];
            let d = other.vertices[(j + 1) % other.vertices.len()];
            if segments_properly_intersect(a, b, c, d) {
                return true;
            }
        }
    }

    for j in 0..other.vertices.len() {
        let c = other.vertices[j];
        let d = other.vertices[(j + 1) % other.vertices.len()];
        let midpoint = (c + d) * 0.5;
        if point_location_translated(vertices, translation, midpoint) > 0 {
            return true;
        }
    }

    if point_location(&other.vertices, probe + translation) > 0 {
        return true;
    }
    if point_location_translated(vertices, translation, other.probe) > 0 {
        return true;
    }
    false
}

fn cells_for_bbox(bbox: &BBox, cell_size: f64) -> Vec<(i32, i32)> {
    let min_x = (bbox.min.x / cell_size).floor() as i32;
    let max_x = (bbox.max.x / cell_size).floor() as i32;
    let min_y = (bbox.min.y / cell_size).floor() as i32;
    let max_y = (bbox.max.y / cell_size).floor() as i32;
    let mut out = Vec::with_capacity(((max_x - min_x + 1) * (max_y - min_y + 1)).max(0) as usize);
    for ix in min_x..=max_x {
        for iy in min_y..=max_y {
            out.push((ix, iy));
        }
    }
    out
}

struct Scanner<'a> {
    data: std::str::SplitWhitespace<'a>,
}

impl<'a> Scanner<'a> {
    fn new(input: &'a str) -> Self {
        Self {
            data: input.split_whitespace(),
        }
    }

    fn next_usize(&mut self) -> usize {
        self.data.next().unwrap().parse().unwrap()
    }

    fn next_f64(&mut self) -> f64 {
        self.data.next().unwrap().parse().unwrap()
    }
}
