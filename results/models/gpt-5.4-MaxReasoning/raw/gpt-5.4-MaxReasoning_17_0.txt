using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;

public static class Program
{
    private static readonly int[][] SplitOrders =
    {
        new[] { 0, 1, 2 },
        new[] { 0, 2, 1 },
        new[] { 1, 0, 2 },
        new[] { 1, 2, 0 },
        new[] { 2, 0, 1 },
        new[] { 2, 1, 0 },
    };

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n = scanner.NextInt();
        if (!scanner.HasValue)
        {
            return;
        }

        int containerW = scanner.NextInt();
        int containerH = scanner.NextInt();
        int containerD = scanner.NextInt();

        var boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            int w = scanner.NextInt();
            int h = scanner.NextInt();
            int d = scanner.NextInt();
            boxes[i] = new Box(i, w, h, d);
        }

        var solver = new Solver(boxes, new Space(0, 0, 0, containerW, containerH, containerD));
        Solution best = solver.Solve();

        using var writer = new StreamWriter(Console.OpenStandardOutput());
        writer.WriteLine(best.Count);

        // The prompt allows axis reordering but does not expose orientation in the
        // stated output format. The solver still tracks the chosen orientation
        // internally; only the required fields are printed here.
        for (int i = 0; i < best.Count; i++)
        {
            var p = best.Placements[i];
            writer.WriteLine($"{p.BoxIndex} {p.X} {p.Y} {p.Z}");
        }
    }

    private sealed class Solver
    {
        private readonly Box[] _boxes;
        private readonly Space _container;
        private readonly Stopwatch _clock;
        private readonly long _timeBudgetMs;
        private readonly Dictionary<int, int[]> _orderCache;

        public Solver(Box[] boxes, Space container)
        {
            _boxes = boxes;
            _container = container;
            _clock = Stopwatch.StartNew();
            _timeBudgetMs = 28_500;
            _orderCache = new Dictionary<int, int[]>();
        }

        public Solution Solve()
        {
            var best = new Solution();

            if (_boxes.Length <= 11)
            {
                long exactBudget = Math.Min(1_500L, 450L + 120L * _boxes.Length);
                var exact = TryExact(exactBudget);
                if (IsBetter(exact, best))
                {
                    best = exact;
                }

                if (best.Count == _boxes.Length)
                {
                    return best;
                }
            }

            RunConfig[] configs = BuildRunConfigs(_boxes.Length);
            for (int i = 0; i < configs.Length; i++)
            {
                if (TimeExpired())
                {
                    break;
                }

                var candidate = RunGreedy(configs[i]);
                if (IsBetter(candidate, best))
                {
                    best = candidate;
                }

                if (best.Count == _boxes.Length)
                {
                    break;
                }
            }

            return best;
        }

        private Solution TryExact(long budgetMs)
        {
            long deadline = Math.Min(_timeBudgetMs, _clock.ElapsedMilliseconds + budgetMs);
            int[] order = GetOrder(0);

            var search = new ExactSearch(_boxes, _container, order, deadline, _clock);
            return search.Run();
        }

        private Solution RunGreedy(RunConfig config)
        {
            int[] order = GetOrder(config.BoxOrderMode);
            int n = _boxes.Length;

            var active = new bool[n];
            var posByBox = new int[n];
            int[] next = new int[n];
            int[] prev = new int[n];

            for (int i = 0; i < n; i++)
            {
                active[i] = true;
                int boxIndex = order[i];
                posByBox[boxIndex] = i;
                next[i] = i + 1 < n ? i + 1 : -1;
                prev[i] = i - 1;
            }

            int head = n > 0 ? 0 : -1;
            int remaining = n;
            long packedVolume = 0;
            var placements = new List<Placement>(n);
            var spaces = new PriorityQueue<Space, SpacePriority>();
            spaces.Enqueue(_container, CreateSpacePriority(_container, config.SpaceMode));

            int fastScanLimit = GetFastScanLimit(n);
            int targetFits = GetTargetFitCount(n);
            int fallbackFits = n <= 1_000 ? 6 : 3;

            while (spaces.Count > 0 && head != -1 && !TimeExpired())
            {
                Space space = spaces.Dequeue();
                Candidate bestCandidate = default;
                bool found = false;
                int fitsFound = 0;
                int checkedBoxes = 0;
                int pos = head;

                for (; pos != -1 && checkedBoxes < fastScanLimit; pos = next[pos])
                {
                    int boxIndex = order[pos];
                    checkedBoxes++;
                    if (!TryBestPlacementForBox(space, _boxes[boxIndex], out Candidate candidate))
                    {
                        continue;
                    }

                    if (!found || candidate.Score.CompareTo(bestCandidate.Score) < 0)
                    {
                        bestCandidate = candidate;
                        found = true;
                    }

                    fitsFound++;
                    if (fitsFound >= targetFits)
                    {
                        pos = next[pos];
                        break;
                    }
                }

                if (!found)
                {
                    int lateFits = 0;
                    for (; pos != -1; pos = next[pos])
                    {
                        int boxIndex = order[pos];
                        if (!TryBestPlacementForBox(space, _boxes[boxIndex], out Candidate candidate))
                        {
                            continue;
                        }

                        if (!found || candidate.Score.CompareTo(bestCandidate.Score) < 0)
                        {
                            bestCandidate = candidate;
                            found = true;
                        }

                        lateFits++;
                        if (lateFits >= fallbackFits)
                        {
                            break;
                        }
                    }
                }

                if (!found)
                {
                    continue;
                }

                int removePos = posByBox[bestCandidate.BoxIndex];
                int previous = prev[removePos];
                int following = next[removePos];

                if (previous != -1)
                {
                    next[previous] = following;
                }
                else
                {
                    head = following;
                }

                if (following != -1)
                {
                    prev[following] = previous;
                }

                active[bestCandidate.BoxIndex] = false;
                remaining--;
                packedVolume += bestCandidate.Dims.Volume;

                placements.Add(new Placement(
                    bestCandidate.BoxIndex,
                    space.X,
                    space.Y,
                    space.Z,
                    bestCandidate.Dims.W,
                    bestCandidate.Dims.H,
                    bestCandidate.Dims.D));

                AddSplitSpaces(spaces, space, bestCandidate.Dims, bestCandidate.SplitOrder, config.SpaceMode);

                if (remaining == 0)
                {
                    break;
                }
            }

            return new Solution(placements, packedVolume);
        }

        private int[] GetOrder(int mode)
        {
            if (_orderCache.TryGetValue(mode, out int[] cached))
            {
                return cached;
            }

            int n = _boxes.Length;
            int[] order = new int[n];
            for (int i = 0; i < n; i++)
            {
                order[i] = i;
            }

            Array.Sort(order, (leftIndex, rightIndex) => CompareBoxes(_boxes[leftIndex], _boxes[rightIndex], mode));
            _orderCache[mode] = order;
            return order;
        }

        private static int CompareBoxes(Box a, Box b, int mode)
        {
            int cmp;
            switch (mode)
            {
                case 1:
                    cmp = a.MaxDim.CompareTo(b.MaxDim);
                    if (cmp != 0) return cmp;
                    cmp = a.MidDim.CompareTo(b.MidDim);
                    if (cmp != 0) return cmp;
                    cmp = a.MinDim.CompareTo(b.MinDim);
                    if (cmp != 0) return cmp;
                    cmp = a.Volume.CompareTo(b.Volume);
                    if (cmp != 0) return cmp;
                    break;
                case 2:
                    cmp = a.MinFaceArea.CompareTo(b.MinFaceArea);
                    if (cmp != 0) return cmp;
                    cmp = a.Volume.CompareTo(b.Volume);
                    if (cmp != 0) return cmp;
                    cmp = a.MaxDim.CompareTo(b.MaxDim);
                    if (cmp != 0) return cmp;
                    cmp = a.MidDim.CompareTo(b.MidDim);
                    if (cmp != 0) return cmp;
                    cmp = a.MinDim.CompareTo(b.MinDim);
                    if (cmp != 0) return cmp;
                    break;
                case 3:
                    cmp = a.SurfaceArea.CompareTo(b.SurfaceArea);
                    if (cmp != 0) return cmp;
                    cmp = a.Volume.CompareTo(b.Volume);
                    if (cmp != 0) return cmp;
                    cmp = a.MaxDim.CompareTo(b.MaxDim);
                    if (cmp != 0) return cmp;
                    cmp = a.MidDim.CompareTo(b.MidDim);
                    if (cmp != 0) return cmp;
                    cmp = a.MinDim.CompareTo(b.MinDim);
                    if (cmp != 0) return cmp;
                    break;
                default:
                    cmp = a.Volume.CompareTo(b.Volume);
                    if (cmp != 0) return cmp;
                    cmp = a.MaxDim.CompareTo(b.MaxDim);
                    if (cmp != 0) return cmp;
                    cmp = a.MidDim.CompareTo(b.MidDim);
                    if (cmp != 0) return cmp;
                    cmp = a.MinDim.CompareTo(b.MinDim);
                    if (cmp != 0) return cmp;
                    break;
            }

            return a.Index.CompareTo(b.Index);
        }

        private static RunConfig[] BuildRunConfigs(int n)
        {
            if (n <= 64)
            {
                return new[]
                {
                    new RunConfig(0, 0),
                    new RunConfig(1, 0),
                    new RunConfig(0, 1),
                    new RunConfig(2, 0),
                    new RunConfig(3, 1),
                };
            }

            if (n <= 2_000)
            {
                return new[]
                {
                    new RunConfig(0, 0),
                    new RunConfig(1, 0),
                    new RunConfig(0, 1),
                    new RunConfig(2, 0),
                };
            }

            if (n <= 6_000)
            {
                return new[]
                {
                    new RunConfig(0, 0),
                    new RunConfig(1, 0),
                    new RunConfig(0, 1),
                };
            }

            return new[]
            {
                new RunConfig(0, 0),
                new RunConfig(1, 0),
            };
        }

        private bool TryBestPlacementForBox(Space space, Box box, out Candidate candidate)
        {
            candidate = default;
            bool found = false;
            FitScore bestScore = default;
            Dims bestDims = default;
            int bestSplitOrder = 0;

            for (int i = 0; i < box.Orientations.Length; i++)
            {
                Dims dims = box.Orientations[i];
                if (dims.W > space.W || dims.H > space.H || dims.D > space.D)
                {
                    continue;
                }

                int rx = space.W - dims.W;
                int ry = space.H - dims.H;
                int rz = space.D - dims.D;
                int exactMatches = 0;
                if (rx == 0) exactMatches++;
                if (ry == 0) exactMatches++;
                if (rz == 0) exactMatches++;

                Sort3(ref rx, ref ry, ref rz);
                FitScore score = new FitScore(
                    3 - exactMatches,
                    rz,
                    ry,
                    rx,
                    box.Volume);

                if (!found || score.CompareTo(bestScore) < 0)
                {
                    found = true;
                    bestScore = score;
                    bestDims = dims;
                    bestSplitOrder = ChooseSplitOrder(space.W - dims.W, space.H - dims.H, space.D - dims.D);
                }
            }

            if (!found)
            {
                return false;
            }

            candidate = new Candidate(box.Index, bestDims, bestSplitOrder, bestScore);
            return true;
        }

        private static int ChooseSplitOrder(int rx, int ry, int rz)
        {
            if (rx >= ry)
            {
                if (ry >= rz) return 0;
                if (rx >= rz) return 1;
                return 4;
            }

            if (rx >= rz) return 2;
            if (ry >= rz) return 3;
            return 5;
        }

        private static void AddSplitSpaces(
            PriorityQueue<Space, SpacePriority> spaces,
            Space source,
            Dims dims,
            int splitOrder,
            int spaceMode)
        {
            int[] order = SplitOrders[splitOrder];
            var processed = new bool[3];

            for (int step = 0; step < 3; step++)
            {
                int axis = order[step];
                Space child = CreateChildSpace(source, dims, axis, processed);
                processed[axis] = true;
                if (child.IsValid)
                {
                    spaces.Enqueue(child, CreateSpacePriority(child, spaceMode));
                }
            }
        }

        private static Space CreateChildSpace(Space source, Dims dims, int axis, bool[] processed)
        {
            int[] pos = { source.X, source.Y, source.Z };
            int[] extents = { source.W, source.H, source.D };
            int[] boxExtents = { dims.W, dims.H, dims.D };
            int[] childPos = { pos[0], pos[1], pos[2] };
            int[] childExtents = new int[3];

            childPos[axis] += boxExtents[axis];
            childExtents[axis] = extents[axis] - boxExtents[axis];

            if (childExtents[axis] <= 0)
            {
                return default;
            }

            for (int other = 0; other < 3; other++)
            {
                if (other == axis)
                {
                    continue;
                }

                childExtents[other] = processed[other] ? boxExtents[other] : extents[other];
                if (childExtents[other] <= 0)
                {
                    return default;
                }
            }

            return new Space(childPos[0], childPos[1], childPos[2], childExtents[0], childExtents[1], childExtents[2]);
        }

        private static SpacePriority CreateSpacePriority(Space space, int mode)
        {
            return mode switch
            {
                1 => new SpacePriority(space.Volume, space.Z, space.Y, space.X),
                _ => new SpacePriority(space.Z, space.Y, space.X, space.Volume),
            };
        }

        private static int GetFastScanLimit(int n)
        {
            if (n <= 64) return n;
            if (n <= 512) return 160;
            if (n <= 2_000) return 96;
            return 48;
        }

        private static int GetTargetFitCount(int n)
        {
            if (n <= 128) return 24;
            if (n <= 2_000) return 12;
            return 6;
        }

        private bool TimeExpired()
        {
            return _clock.ElapsedMilliseconds >= _timeBudgetMs;
        }
    }

    private sealed class ExactSearch
    {
        private readonly Box[] _boxes;
        private readonly Space _container;
        private readonly int[] _order;
        private readonly long _deadlineMs;
        private readonly Stopwatch _clock;
        private readonly bool[] _used;
        private readonly List<Placement> _current;
        private Solution _best;

        public ExactSearch(Box[] boxes, Space container, int[] order, long deadlineMs, Stopwatch clock)
        {
            _boxes = boxes;
            _container = container;
            _order = order;
            _deadlineMs = deadlineMs;
            _clock = clock;
            _used = new bool[boxes.Length];
            _current = new List<Placement>(boxes.Length);
            _best = new Solution();
        }

        public Solution Run()
        {
            var spaces = new List<Space>(1) { _container };
            Search(spaces, 0L);
            return _best;
        }

        private void Search(List<Space> spaces, long packedVolume)
        {
            if (_clock.ElapsedMilliseconds >= _deadlineMs)
            {
                return;
            }

            int placed = _current.Count;
            int remaining = _boxes.Length - placed;
            if (placed + remaining <= _best.Count)
            {
                return;
            }

            long freeVolume = 0;
            for (int i = 0; i < spaces.Count; i++)
            {
                freeVolume += spaces[i].Volume;
            }

            long minRemainingVolume = long.MaxValue;
            for (int i = 0; i < _boxes.Length; i++)
            {
                if (!_used[i] && _boxes[i].Volume < minRemainingVolume)
                {
                    minRemainingVolume = _boxes[i].Volume;
                }
            }

            if (minRemainingVolume != long.MaxValue)
            {
                int volumeBound = placed + (int)Math.Min((long)remaining, freeVolume / minRemainingVolume);
                if (volumeBound <= _best.Count)
                {
                    return;
                }
            }

            if (spaces.Count == 0)
            {
                MaybeUpdateBest(packedVolume);
                return;
            }

            int bestSpaceIndex = SelectSpace(spaces);
            Space space = spaces[bestSpaceIndex];

            var tailSpaces = new List<Space>(spaces.Count - 1);
            for (int i = 0; i < spaces.Count; i++)
            {
                if (i != bestSpaceIndex)
                {
                    tailSpaces.Add(spaces[i]);
                }
            }

            var candidates = new List<Candidate>();
            for (int orderPos = 0; orderPos < _order.Length; orderPos++)
            {
                int boxIndex = _order[orderPos];
                if (_used[boxIndex])
                {
                    continue;
                }

                if (TryExactCandidate(space, _boxes[boxIndex], out Candidate candidate))
                {
                    candidates.Add(candidate);
                }
            }

            candidates.Sort((a, b) => a.Score.CompareTo(b.Score));

            for (int i = 0; i < candidates.Count; i++)
            {
                if (_clock.ElapsedMilliseconds >= _deadlineMs)
                {
                    return;
                }

                Candidate candidate = candidates[i];
                _used[candidate.BoxIndex] = true;
                _current.Add(new Placement(candidate.BoxIndex, space.X, space.Y, space.Z, candidate.Dims.W, candidate.Dims.H, candidate.Dims.D));

                var nextSpaces = new List<Space>(tailSpaces.Count + 3);
                nextSpaces.AddRange(tailSpaces);
                AddSplitSpaces(nextSpaces, space, candidate.Dims, candidate.SplitOrder);

                Search(nextSpaces, packedVolume + candidate.Dims.Volume);

                _current.RemoveAt(_current.Count - 1);
                _used[candidate.BoxIndex] = false;

                if (_best.Count == _boxes.Length)
                {
                    return;
                }
            }

            Search(tailSpaces, packedVolume);
        }

        private static int SelectSpace(List<Space> spaces)
        {
            int bestIndex = 0;
            for (int i = 1; i < spaces.Count; i++)
            {
                if (CompareSpace(spaces[i], spaces[bestIndex]) < 0)
                {
                    bestIndex = i;
                }
            }

            return bestIndex;
        }

        private static int CompareSpace(Space a, Space b)
        {
            int cmp = a.Z.CompareTo(b.Z);
            if (cmp != 0) return cmp;
            cmp = a.Y.CompareTo(b.Y);
            if (cmp != 0) return cmp;
            cmp = a.X.CompareTo(b.X);
            if (cmp != 0) return cmp;
            return a.Volume.CompareTo(b.Volume);
        }

        private static bool TryExactCandidate(Space space, Box box, out Candidate candidate)
        {
            candidate = default;
            bool found = false;
            FitScore bestScore = default;
            Dims bestDims = default;
            int bestSplitOrder = 0;

            for (int i = 0; i < box.Orientations.Length; i++)
            {
                Dims dims = box.Orientations[i];
                if (dims.W > space.W || dims.H > space.H || dims.D > space.D)
                {
                    continue;
                }

                int rx = space.W - dims.W;
                int ry = space.H - dims.H;
                int rz = space.D - dims.D;
                int exactMatches = 0;
                if (rx == 0) exactMatches++;
                if (ry == 0) exactMatches++;
                if (rz == 0) exactMatches++;

                Sort3(ref rx, ref ry, ref rz);
                FitScore score = new FitScore(3 - exactMatches, rz, ry, rx, box.Volume);

                if (!found || score.CompareTo(bestScore) < 0)
                {
                    found = true;
                    bestScore = score;
                    bestDims = dims;
                    bestSplitOrder = ChooseSplitOrder(space.W - dims.W, space.H - dims.H, space.D - dims.D);
                }
            }

            if (!found)
            {
                return false;
            }

            candidate = new Candidate(box.Index, bestDims, bestSplitOrder, bestScore);
            return true;
        }

        private static int ChooseSplitOrder(int rx, int ry, int rz)
        {
            if (rx >= ry)
            {
                if (ry >= rz) return 0;
                if (rx >= rz) return 1;
                return 4;
            }

            if (rx >= rz) return 2;
            if (ry >= rz) return 3;
            return 5;
        }

        private static void AddSplitSpaces(List<Space> spaces, Space source, Dims dims, int splitOrder)
        {
            int[] order = SplitOrders[splitOrder];
            var processed = new bool[3];

            for (int step = 0; step < 3; step++)
            {
                int axis = order[step];
                Space child = CreateChildSpace(source, dims, axis, processed);
                processed[axis] = true;
                if (child.IsValid)
                {
                    spaces.Add(child);
                }
            }
        }

        private static Space CreateChildSpace(Space source, Dims dims, int axis, bool[] processed)
        {
            int[] pos = { source.X, source.Y, source.Z };
            int[] extents = { source.W, source.H, source.D };
            int[] boxExtents = { dims.W, dims.H, dims.D };
            int[] childPos = { pos[0], pos[1], pos[2] };
            int[] childExtents = new int[3];

            childPos[axis] += boxExtents[axis];
            childExtents[axis] = extents[axis] - boxExtents[axis];

            if (childExtents[axis] <= 0)
            {
                return default;
            }

            for (int other = 0; other < 3; other++)
            {
                if (other == axis)
                {
                    continue;
                }

                childExtents[other] = processed[other] ? boxExtents[other] : extents[other];
                if (childExtents[other] <= 0)
                {
                    return default;
                }
            }

            return new Space(childPos[0], childPos[1], childPos[2], childExtents[0], childExtents[1], childExtents[2]);
        }

        private void MaybeUpdateBest(long packedVolume)
        {
            var candidate = new Solution(_current, packedVolume);
            if (IsBetter(candidate, _best))
            {
                _best = candidate;
            }
        }
    }

    private readonly struct RunConfig
    {
        public readonly int BoxOrderMode;
        public readonly int SpaceMode;

        public RunConfig(int boxOrderMode, int spaceMode)
        {
            BoxOrderMode = boxOrderMode;
            SpaceMode = spaceMode;
        }
    }

    private sealed class Box
    {
        public readonly int Index;
        public readonly long Volume;
        public readonly long SurfaceArea;
        public readonly long MinFaceArea;
        public readonly int MinDim;
        public readonly int MidDim;
        public readonly int MaxDim;
        public readonly Dims[] Orientations;

        public Box(int index, int w, int h, int d)
        {
            Index = index;
            Volume = (long)w * h * d;
            SurfaceArea = 2L * (long)(w * h + w * d + h * d);

            long face1 = (long)w * h;
            long face2 = (long)w * d;
            long face3 = (long)h * d;
            MinFaceArea = Math.Min(face1, Math.Min(face2, face3));

            int a = w;
            int b = h;
            int c = d;
            Sort3(ref a, ref b, ref c);
            MinDim = a;
            MidDim = b;
            MaxDim = c;

            Orientations = BuildOrientations(w, h, d);
        }
    }

    private readonly struct Dims
    {
        public readonly int W;
        public readonly int H;
        public readonly int D;

        public Dims(int w, int h, int d)
        {
            W = w;
            H = h;
            D = d;
        }

        public long Volume => (long)W * H * D;
    }

    private readonly struct Placement
    {
        public readonly int BoxIndex;
        public readonly int X;
        public readonly int Y;
        public readonly int Z;
        public readonly int W;
        public readonly int H;
        public readonly int D;

        public Placement(int boxIndex, int x, int y, int z, int w, int h, int d)
        {
            BoxIndex = boxIndex;
            X = x;
            Y = y;
            Z = z;
            W = w;
            H = h;
            D = d;
        }
    }

    private sealed class Solution
    {
        public static readonly Solution Empty = new Solution();

        public readonly Placement[] Placements;
        public readonly int Count;
        public readonly long PackedVolume;

        public Solution()
        {
            Placements = Array.Empty<Placement>();
            Count = 0;
            PackedVolume = 0;
        }

        public Solution(List<Placement> placements, long packedVolume)
        {
            Placements = placements.ToArray();
            Count = Placements.Length;
            PackedVolume = packedVolume;
        }

        public Solution(List<Placement> placements, long packedVolume, bool copy)
        {
            if (copy)
            {
                Placements = placements.ToArray();
                Count = Placements.Length;
                PackedVolume = packedVolume;
            }
            else
            {
                Placements = Array.Empty<Placement>();
                Count = 0;
                PackedVolume = 0;
            }
        }
    }

    private readonly struct Space
    {
        public readonly int X;
        public readonly int Y;
        public readonly int Z;
        public readonly int W;
        public readonly int H;
        public readonly int D;

        public Space(int x, int y, int z, int w, int h, int d)
        {
            X = x;
            Y = y;
            Z = z;
            W = w;
            H = h;
            D = d;
        }

        public bool IsValid => W > 0 && H > 0 && D > 0;
        public long Volume => (long)W * H * D;
    }

    private readonly struct Candidate
    {
        public readonly int BoxIndex;
        public readonly Dims Dims;
        public readonly int SplitOrder;
        public readonly FitScore Score;

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _length;
        private int _position;

        public FastScanner(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 16];
            _length = 0;
            _position = 0;
            HasValue = true;
        }

        public bool HasValue { get; private set; }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1)
                {
                    HasValue = false;
                    return 0;
                }
            }
            while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length == 0)
                {
                    return -1;
                }
            }

            return _buffer[_position++];
        }
    }

    private static Dims[] BuildOrientations(int a, int b, int c)
    {
        var candidates = new[]
        {
            new Dims(a, b, c),
            new Dims(a, c, b),
            new Dims(b, a, c),
            new Dims(b, c, a),
            new Dims(c, a, b),
            new Dims(c, b, a),
        };

        var unique = new Dims[6];
        int count = 0;
        for (int i = 0; i < candidates.Length; i++)
        {
            bool duplicate = false;
            for (int j = 0; j < count; j++)
            {
                if (SameDims(unique[j], candidates[i]))
                {
                    duplicate = true;
                    break;
                }
            }

            if (!duplicate)
            {
                unique[count++] = candidates[i];
            }
        }

        var result = new Dims[count];
        Array.Copy(unique, result, count);
        return result;
    }

    private static bool SameDims(Dims a, Dims b)
    {
        return a.W == b.W && a.H == b.H && a.D == b.D;
    }

    private static bool IsBetter(Solution candidate, Solution best)
    {
        if (candidate.Count != best.Count)
        {
            return candidate.Count > best.Count;
        }

        return candidate.PackedVolume > best.PackedVolume;
    }

    private static void Sort3(ref int a, ref int b, ref int c)
    {
        if (a > b)
        {
            (a, b) = (b, a);
        }

        if (b > c)
        {
            (b, c) = (c, b);
        }

        if (a > b)
        {
            (a, b) = (b, a);
        }
    }
