using System;
using System.Collections.Generic;
using System.IO;
using System.Numerics;
using System.Text;

public static class Program
{
    private const int SmallDpThreshold = 22;

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int m = fs.NextInt();

        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        var from = new int[m];
        var to = new int[m];
        var forcedInitial = new bool[n];

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            from[i] = u;
            to[i] = v;
            if (u == v)
            {
                forcedInitial[u] = true;
            }
        }

        var answer = SolveAll(n, from, to, forcedInitial);
        answer.Sort();

        Console.WriteLine(answer.Count);
        if (answer.Count == 0)
        {
            Console.WriteLine();
            return;
        }

        var sb = new StringBuilder();
        for (int i = 0; i < answer.Count; i++)
        {
            if (i > 0)
            {
                sb.Append(' ');
            }

            sb.Append(answer[i]);
        }

        Console.WriteLine(sb.ToString());
    }

    private static List<int> SolveAll(int n, int[] from, int[] to, bool[] forcedInitial)
    {
        var result = new List<int>();
        var alive = new bool[n];
        var outAdj = new List<int>[n];
        var inAdj = new List<int>[n];
        var indeg = new int[n];
        var outdeg = new int[n];

        for (int v = 0; v < n; v++)
        {
            alive[v] = !forcedInitial[v];
            if (forcedInitial[v])
            {
                result.Add(v);
            }
        }

        for (int i = 0; i < from.Length; i++)
        {
            int u = from[i];
            int v = to[i];
            if (u == v || !alive[u] || !alive[v])
            {
                continue;
            }

            (outAdj[u] ??= new List<int>()).Add(v);
            (inAdj[v] ??= new List<int>()).Add(u);
            outdeg[u]++;
            indeg[v]++;
        }

        var queue = new Queue<int>();
        for (int v = 0; v < n; v++)
        {
            if (alive[v] && (indeg[v] == 0 || outdeg[v] == 0))
            {
                queue.Enqueue(v);
            }
        }

        while (queue.Count > 0)
        {
            int v = queue.Dequeue();
            if (!alive[v] || (indeg[v] != 0 && outdeg[v] != 0))
            {
                continue;
            }

            alive[v] = false;

            if (outAdj[v] != null)
            {
                foreach (int w in outAdj[v])
                {
                    if (!alive[w])
                    {
                        continue;
                    }

                    indeg[w]--;
                    if (indeg[w] == 0 || outdeg[w] == 0)
                    {
                        queue.Enqueue(w);
                    }
                }
            }

            if (inAdj[v] != null)
            {
                foreach (int u in inAdj[v])
                {
                    if (!alive[u])
                    {
                        continue;
                    }

                    outdeg[u]--;
                    if (indeg[u] == 0 || outdeg[u] == 0)
                    {
                        queue.Enqueue(u);
                    }
                }
            }
        }

        var sccs = GetGlobalSccs(n, alive, outAdj, inAdj);
        foreach (var comp in sccs)
        {
            if (comp.Count <= 1)
            {
                continue;
            }

            var state = BuildInitialState(comp, outAdj);
            var part = Solve(state, int.MaxValue / 4);
            if (part == null)
            {
                throw new InvalidOperationException("Exact solver failed unexpectedly.");
            }

            result.AddRange(part);
        }

        return result;
    }

    private static GraphState BuildInitialState(List<int> component, List<int>[] outAdj)
    {
        int k = component.Count;
        var orig = new int[k];
        var map = new Dictionary<int, int>(k);

        for (int i = 0; i < k; i++)
        {
            int v = component[i];
            orig[i] = v;
            map[v] = i;
        }

        var state = new GraphState(orig);
        for (int i = 0; i < k; i++)
        {
            int uOrig = orig[i];
            var list = outAdj[uOrig];
            if (list == null)
            {
                continue;
            }

            foreach (int vOrig in list)
            {
                if (map.TryGetValue(vOrig, out int j))
                {
                    state.AddEdge(i, j);
                }
            }
        }

        return state;
    }

    private static List<int>? Solve(GraphState state, int limit)
    {
        var forced = new List<int>();
        Reduce(state, forced);

        if (forced.Count >= limit)
        {
            return null;
        }

        if (state.ActiveCount == 0)
        {
            return forced;
        }

        if (state.ActiveCount <= SmallDpThreshold)
        {
            var tail = SolveSmallDp(state);
            if (forced.Count + tail.Count >= limit)
            {
                return null;
            }

            forced.AddRange(tail);
            return forced;
        }

        var sccs = GetStateSccs(state);
        if (sccs.Count == 0)
        {
            return forced;
        }

        if (sccs.Count > 1)
        {
            sccs.Sort((a, b) => b.Count.CompareTo(a.Count));
            var total = new List<int>(forced);
            int remaining = limit - forced.Count;

            foreach (var comp in sccs)
            {
                var sub = ExtractComponent(state, comp);
                var part = Solve(sub, remaining);
                if (part == null)
                {
                    return null;
                }

                total.AddRange(part);
                remaining = limit - total.Count;
            }

            return total;
        }

        int lb = CyclePackingLowerBound(state);
        if (forced.Count + lb >= limit)
        {
            return null;
        }

        List<int>? best = null;
        int bestLimit = limit;

        var greedy = GreedyUpperBound(state);
        if (forced.Count + greedy.Count < bestLimit)
        {
            bestLimit = forced.Count + greedy.Count;
            best = new List<int>(bestLimit);
            best.AddRange(forced);
            best.AddRange(greedy);
            if (bestLimit == forced.Count + lb)
            {
                return best;
            }
        }

        var cycle = FindBranchCycle(state, sccs[0]);
        cycle.Sort((a, b) => VertexScore(state, b).CompareTo(VertexScore(state, a)));

        foreach (int v in cycle)
        {
            if (forced.Count + 1 >= bestLimit)
            {
                continue;
            }

            var child = state.Clone();
            child.RemoveVertex(v);

            var tail = Solve(child, bestLimit - forced.Count - 1);
            if (tail == null)
            {
                continue;
            }

            int totalSize = forced.Count + 1 + tail.Count;
            if (totalSize < bestLimit)
            {
                bestLimit = totalSize;
                best = new List<int>(totalSize);
                best.AddRange(forced);
                best.Add(state.Orig[v]);
                best.AddRange(tail);

                if (bestLimit == forced.Count + lb)
                {
                    break;
                }
            }
        }

        return best;
    }

    private static void Reduce(GraphState state, List<int> forced)
    {
        bool changed;
        do
        {
            changed = false;
            for (int v = 0; v < state.N; v++)
            {
                if (!state.Active[v])
                {
                    continue;
                }

                if (state.HasEdge(v, v))
                {
                    forced.Add(state.Orig[v]);
                    state.RemoveVertex(v);
                    changed = true;
                    continue;
                }

                if (state.InDeg[v] == 0 || state.OutDeg[v] == 0)
                {
                    state.RemoveVertex(v);
                    changed = true;
                    continue;
                }

                if (state.InDeg[v] == 1)
                {
                    int p = state.GetOnlyInNeighbor(v);
                    state.BypassIn(v, p);
                    changed = true;
                    continue;
                }

                if (state.OutDeg[v] == 1)
                {
                    int s = state.GetOnlyOutNeighbor(v);
                    state.BypassOut(v, s);
                    changed = true;
                }
            }
        }
        while (changed);
    }

    private static List<int> SolveSmallDp(GraphState state)
    {
        var verts = new List<int>(state.ActiveCount);
        for (int v = 0; v < state.N; v++)
        {
            if (state.Active[v])
            {
                verts.Add(v);
            }
        }

        int k = verts.Count;
        if (k == 0)
        {
            return new List<int>();
        }

        var map = new int[state.N];
        Array.Fill(map, -1);
        for (int i = 0; i < k; i++)
        {
            map[verts[i]] = i;
        }

        var inMask = new ulong[k];
        var outMask = new ulong[k];

        for (int i = 0; i < k; i++)
        {
            int old = verts[i];
            for (int wi = 0; wi < state.Words; wi++)
            {
                ulong word = state.Out[old][wi];
                while (word != 0)
                {
                    int bit = BitOperations.TrailingZeroCount(word);
                    int to = (wi << 6) + bit;
                    word &= word - 1;

                    int j = map[to];
                    if (j >= 0)
                    {
                        outMask[i] |= 1UL << j;
                        inMask[j] |= 1UL << i;
                    }
                }
            }
        }

        int totalMasks = 1 << k;
        var good = new bool[totalMasks];
        good[0] = true;

        int bestMask = 0;
        int bestSize = 0;

        for (int mask = 1; mask < totalMasks; mask++)
        {
            int bits = mask;
            while (bits != 0)
            {
                int v = BitOperations.TrailingZeroCount((uint)bits);
                bits &= bits - 1;

                int prev = mask & ~(1 << v);
                if (!good[prev])
                {
                    continue;
                }

                ulong prevMask = (uint)prev;
                if ((inMask[v] & prevMask) == 0UL || (outMask[v] & prevMask) == 0UL)
                {
                    good[mask] = true;
                    break;
                }
            }

            if (good[mask])
            {
                int count = BitOperations.PopCount((uint)mask);
                if (count > bestSize)
                {
                    bestSize = count;
                    bestMask = mask;
                }
            }
        }

        var removed = new List<int>(k - bestSize);
        for (int i = 0; i < k; i++)
        {
            if ((bestMask & (1 << i)) == 0)
            {
                removed.Add(state.Orig[verts[i]]);
            }
        }

        return removed;
    }

    private static List<int> GreedyUpperBound(GraphState state)
    {
        var temp = state.Clone();
        var result = new List<int>();

        while (true)
        {
            Reduce(temp, result);
            if (temp.ActiveCount == 0)
            {
                return result;
            }

            var sccs = GetStateSccs(temp);
            if (sccs.Count == 0)
            {
                return result;
            }

            int bestVertex = -1;
            long bestScore = long.MinValue;

            foreach (var comp in sccs)
            {
                foreach (int v in comp)
                {
                    long score = VertexScore(temp, v);
                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestVertex = v;
                    }
                }
            }

            result.Add(temp.Orig[bestVertex]);
            temp.RemoveVertex(bestVertex);
        }
    }

    private static int CyclePackingLowerBound(GraphState state)
    {
        var temp = state.Clone();
        int lb = 0;

        while (true)
        {
            var scratch = new List<int>();
            Reduce(temp, scratch);
            if (temp.ActiveCount == 0)
            {
                return lb;
            }

            var sccs = GetStateSccs(temp);
            if (sccs.Count == 0)
            {
                return lb;
            }

            var cycle = FindBranchCycle(temp, sccs[0]);
            if (cycle.Count == 0)
            {
                return lb;
            }

            lb++;
            foreach (int v in cycle)
            {
                temp.RemoveVertex(v);
            }
        }
    }

    private static long VertexScore(GraphState state, int v)
    {
        return (long)state.InDeg[v] * state.OutDeg[v] + state.InDeg[v] + state.OutDeg[v];
    }

    private static List<int> FindBranchCycle(GraphState state, List<int>? preferred)
    {
        var twoCycle = FindTwoCycle(state);
        if (twoCycle != null)
        {
            return twoCycle;
        }

        var threeCycle = FindThreeCycle(state);
        if (threeCycle != null)
        {
            return threeCycle;
        }

        var pivots = SelectTopPivots(state, preferred, 3);
        List<int>? best = null;

        foreach (int pivot in pivots)
        {
            if (pivot < 0)
            {
                continue;
            }

            var cycle = ShortestCycleThroughPivot(state, pivot);
            if (cycle != null && (best == null || cycle.Count < best.Count))
            {
                best = cycle;
            }
        }

        if (best != null)
        {
            return best;
        }

        for (int v = 0; v < state.N; v++)
        {
            if (!state.Active[v])
            {
                continue;
            }

            var cycle = ShortestCycleThroughPivot(state, v);
            if (cycle != null)
            {
                return cycle;
            }
        }

        return new List<int>();
    }

    private static int[] SelectTopPivots(GraphState state, List<int>? preferred, int count)
    {
        var bestVertices = new int[count];
        var bestScores = new long[count];
        for (int i = 0; i < count; i++)
        {
            bestVertices[i] = -1;
            bestScores[i] = long.MinValue;
        }

        void Consider(int v)
        {
            if (!state.Active[v])
            {
                return;
            }

            long score = VertexScore(state, v);
            for (int i = 0; i < count; i++)
            {
                if (score <= bestScores[i])
                {
                    continue;
                }

                for (int j = count - 1; j > i; j--)
                {
                    bestScores[j] = bestScores[j - 1];
                    bestVertices[j] = bestVertices[j - 1];
                }

                bestScores[i] = score;
                bestVertices[i] = v;
                break;
            }
        }

        if (preferred != null)
        {
            foreach (int v in preferred)
            {
                Consider(v);
            }
        }
        else
        {
            for (int v = 0; v < state.N; v++)
            {
                Consider(v);
            }
        }

        return bestVertices;
    }

    private static List<int>? FindTwoCycle(GraphState state)
    {
        for (int u = 0; u < state.N; u++)
        {
            if (!state.Active[u])
            {
                continue;
            }

            for (int wi = 0; wi < state.Words; wi++)
            {
                ulong word = state.Out[u][wi];
                while (word != 0)
                {
                    int bit = BitOperations.TrailingZeroCount(word);
                    int v = (wi << 6) + bit;
                    word &= word - 1;

                    if (u < v && state.Active[v] && state.HasEdge(v, u))
                    {
                        return new List<int> { u, v };
                    }
                }
            }
        }

        return null;
    }

    private static List<int>? FindThreeCycle(GraphState state)
    {
        for (int u = 0; u < state.N; u++)
        {
            if (!state.Active[u])
            {
                continue;
            }

            for (int wi = 0; wi < state.Words; wi++)
            {
                ulong word = state.Out[u][wi];
                while (word != 0)
                {
                    int bit = BitOperations.TrailingZeroCount(word);
                    int v = (wi << 6) + bit;
                    word &= word - 1;

                    if (!state.Active[v])
                    {
                        continue;
                    }

                    for (int wj = 0; wj < state.Words; wj++)
                    {
                        ulong inter = state.Out[v][wj] & state.In[u][wj];
                        while (inter != 0)
                        {
                            int b = BitOperations.TrailingZeroCount(inter);
                            int w = (wj << 6) + b;
                            inter &= inter - 1;

                            if (w != u && w != v && state.Active[w])
                            {
                                return new List<int> { u, v, w };
                            }
                        }
                    }
                }
            }
        }

        return null;
    }

    private static List<int>? ShortestCycleThroughPivot(GraphState state, int pivot)
    {
        if (!state.Active[pivot])
        {
            return null;
        }

        var dist = new int[state.N];
        var parent = new int[state.N];
        Array.Fill(dist, -1);
        Array.Fill(parent, -1);

        var q = new int[state.N];
        int head = 0;
        int tail = 0;
        dist[pivot] = 0;
        q[tail++] = pivot;

        while (head < tail)
        {
            int cur = q[head++];
            for (int wi = 0; wi < state.Words; wi++)
            {
                ulong word = state.In[cur][wi];
                while (word != 0)
                {
                    int bit = BitOperations.TrailingZeroCount(word);
                    int pred = (wi << 6) + bit;
                    word &= word - 1;

                    if (!state.Active[pred] || dist[pred] >= 0)
                    {
                        continue;
                    }

                    dist[pred] = dist[cur] + 1;
                    parent[pred] = cur;
                    q[tail++] = pred;
                }
            }
        }

        int bestNext = -1;
        int bestDist = int.MaxValue;
        for (int wi = 0; wi < state.Words; wi++)
        {
            ulong word = state.Out[pivot][wi];
            while (word != 0)
            {
                int bit = BitOperations.TrailingZeroCount(word);
                int next = (wi << 6) + bit;
                word &= word - 1;

                if (!state.Active[next] || dist[next] < 0 || dist[next] >= bestDist)
                {
                    continue;
                }

                bestDist = dist[next];
                bestNext = next;
            }
        }

        if (bestNext < 0)
        {
            return null;
        }

        var cycle = new List<int> { pivot, bestNext };
        int curNode = bestNext;
        while (parent[curNode] != pivot)
        {
            curNode = parent[curNode];
            if (curNode < 0)
            {
                return null;
            }

            cycle.Add(curNode);
        }

        return cycle;
    }

    private static List<List<int>> GetGlobalSccs(int n, bool[] alive, List<int>[] outAdj, List<int>[] inAdj)
    {
        var visited = new bool[n];
        var order = new List<int>();
        var stack = new Stack<(int Node, bool Exit)>();

        for (int v = 0; v < n; v++)
        {
            if (!alive[v] || visited[v])
            {
                continue;
            }

            stack.Push((v, false));
            while (stack.Count > 0)
            {
                var frame = stack.Pop();
                int u = frame.Node;

                if (frame.Exit)
                {
                    order.Add(u);
                    continue;
                }

                if (visited[u])
                {
                    continue;
                }

                visited[u] = true;
                stack.Push((u, true));

                var list = outAdj[u];
                if (list == null)
                {
                    continue;
                }

                foreach (int w in list)
                {
                    if (alive[w] && !visited[w])
                    {
                        stack.Push((w, false));
                    }
                }
            }
        }

        Array.Fill(visited, false);
        var components = new List<List<int>>();
        var stack2 = new Stack<int>();

        for (int idx = order.Count - 1; idx >= 0; idx--)
        {
            int start = order[idx];
            if (!alive[start] || visited[start])
            {
                continue;
            }

            var comp = new List<int>();
            visited[start] = true;
            stack2.Push(start);

            while (stack2.Count > 0)
            {
                int u = stack2.Pop();
                comp.Add(u);

                var list = inAdj[u];
                if (list == null)
                {
                    continue;
                }

                foreach (int p in list)
                {
                    if (alive[p] && !visited[p])
                    {
                        visited[p] = true;
                        stack2.Push(p);
                    }
                }
            }

            if (comp.Count > 1)
            {
                components.Add(comp);
            }
        }

        return components;
    }

    private static List<List<int>> GetStateSccs(GraphState state)
    {
        var visited = new bool[state.N];
        var order = new List<int>(state.ActiveCount);
        var stack = new Stack<(int Node, bool Exit)>();

        for (int v = 0; v < state.N; v++)
        {
            if (!state.Active[v] || visited[v])
            {
                continue;
            }

            stack.Push((v, false));
            while (stack.Count > 0)
            {
                var frame = stack.Pop();
                int u = frame.Node;

                if (frame.Exit)
                {
                    order.Add(u);
                    continue;
                }

                if (visited[u])
                {
                    continue;
                }

                visited[u] = true;
                stack.Push((u, true));

                for (int wi = 0; wi < state.Words; wi++)
                {
                    ulong word = state.Out[u][wi];
                    while (word != 0)
                    {
                        int bit = BitOperations.TrailingZeroCount(word);
                        int w = (wi << 6) + bit;
                        word &= word - 1;

                        if (state.Active[w] && !visited[w])
                        {
                            stack.Push((w, false));
                        }
                    }
                }
            }
        }

        Array.Fill(visited, false);
        var components = new List<List<int>>();
        var stack2 = new Stack<int>();

        for (int idx = order.Count - 1; idx >= 0; idx--)
        {
            int start = order[idx];
            if (!state.Active[start] || visited[start])
            {
                continue;
            }

            var comp = new List<int>();
            visited[start] = true;
            stack2.Push(start);

            while (stack2.Count > 0)
            {
                int u = stack2.Pop();
                comp.Add(u);

                for (int wi = 0; wi < state.Words; wi++)
                {
                    ulong word = state.In[u][wi];
                    while (word != 0)
                    {
                        int bit = BitOperations.TrailingZeroCount(word);
                        int p = (wi << 6) + bit;
                        word &= word - 1;

                        if (state.Active[p] && !visited[p])
                        {
                            visited[p] = true;
                            stack2.Push(p);
                        }
                    }
                }
            }

            if (comp.Count > 1)
            {
                components.Add(comp);
            }
        }

        return components;
    }

    private static GraphState ExtractComponent(GraphState state, List<int> comp)
    {
        int k = comp.Count;
        var orig = new int[k];
        var map = new int[state.N];
        Array.Fill(map, -1);

        for (int i = 0; i < k; i++)
        {
            int old = comp[i];
            map[old] = i;
            orig[i] = state.Orig[old];
        }

        var sub = new GraphState(orig);
        for (int i = 0; i < k; i++)
        {
            int old = comp[i];
            for (int wi = 0; wi < state.Words; wi++)
            {
                ulong word = state.Out[old][wi];
                while (word != 0)
                {
                    int bit = BitOperations.TrailingZeroCount(word);
                    int to = (wi << 6) + bit;
                    word &= word - 1;

                    int j = map[to];
                    if (j >= 0)
                    {
                        sub.AddEdge(i, j);
                    }
                }
            }
        }

        return sub;
    }

    private sealed class GraphState
    {
        public int N { get; }
        public int Words { get; }
        public int[] Orig { get; }
        public ulong[][] Out { get; }
        public ulong[][] In { get; }
        public bool[] Active { get; }
        public int[] OutDeg { get; }
        public int[] InDeg { get; }
        public int ActiveCount { get; private set; }

        public GraphState(int[] orig)
        {
            N = orig.Length;
            Words = (N + 63) >> 6;
            Orig = orig;
            Out = new ulong[N][];
            In = new ulong[N][];
            Active = new bool[N];
            OutDeg = new int[N];
            InDeg = new int[N];
            ActiveCount = N;

            for (int i = 0; i < N; i++)
            {
                Out[i] = new ulong[Words];
                In[i] = new ulong[Words];
                Active[i] = true;
            }
        }

        private GraphState(
            int n,
            int words,
            int[] orig,
            ulong[][] outAdj,
            ulong[][] inAdj,
            bool[] active,
            int[] outDeg,
            int[] inDeg,
            int activeCount)
        {
            N = n;
            Words = words;
            Orig = orig;
            Out = outAdj;
            In = inAdj;
            Active = active;
            OutDeg = outDeg;
            InDeg = inDeg;
            ActiveCount = activeCount;
        }

        public GraphState Clone()
        {
            var outCopy = new ulong[N][];
            var inCopy = new ulong[N][];
            for (int i = 0; i < N; i++)
            {
                outCopy[i] = (ulong[])Out[i].Clone();
                inCopy[i] = (ulong[])In[i].Clone();
            }

            return new GraphState(
                N,
                Words,
                Orig,
                outCopy,
                inCopy,
                (bool[])Active.Clone(),
                (int[])OutDeg.Clone(),
                (int[])InDeg.Clone(),
                ActiveCount);
        }

        public bool HasEdge(int u, int v)
        {
            int wi = v >> 6;
            ulong mask = 1UL << (v & 63);
            return (Out[u][wi] & mask) != 0UL;
        }

        public void AddEdge(int u, int v)
        {
            if (!Active[u] || !Active[v])
            {
                return;
            }

            int wi = v >> 6;
            ulong maskV = 1UL << (v & 63);
            if ((Out[u][wi] & maskV) != 0UL)
            {
                return;
            }

            Out[u][wi] |= maskV;
            int wj = u >> 6;
            ulong maskU = 1UL << (u & 63);
            In[v][wj] |= maskU;
            OutDeg[u]++;
            InDeg[v]++;
        }

        public void RemoveVertex(int v)
        {
            if (!Active[v])
            {
                return;
            }

            int vWord = v >> 6;
            ulong vMask = 1UL << (v & 63);

            for (int wi = 0; wi < Words; wi++)
            {
                ulong word = In[v][wi];
                while (word != 0)
                {
                    int bit = BitOperations.TrailingZeroCount(word);
                    int u = (wi << 6) + bit;
                    word &= word - 1;

                    if (u == v || !Active[u])
                    {
                        continue;
                    }

                    if ((Out[u][vWord] & vMask) != 0UL)
                    {
                        Out[u][vWord] &= ~vMask;
                        OutDeg[u]--;
                    }
                }
            }

            for (int wi = 0; wi < Words; wi++)
            {
                ulong word = Out[v][wi];
                while (word != 0)
                {
                    int bit = BitOperations.TrailingZeroCount(word);
                    int w = (wi << 6) + bit;
                    word &= word - 1;

                    if (w == v || !Active[w])
                    {
                        continue;
                    }

                    if ((In[w][vWord] & vMask) != 0UL)
                    {
                        In[w][vWord] &= ~vMask;
                        InDeg[w]--;
                    }
                }
            }

            Array.Clear(Out[v], 0, Words);
            Array.Clear(In[v], 0, Words);
            Active[v] = false;
            OutDeg[v] = 0;
            InDeg[v] = 0;
            ActiveCount--;
        }

        public int GetOnlyInNeighbor(int v)
        {
            for (int wi = 0; wi < Words; wi++)
            {
                ulong word = In[v][wi];
                if (word != 0)
                {
                    return (wi << 6) + BitOperations.TrailingZeroCount(word);
                }
            }

            return -1;
        }

        public int GetOnlyOutNeighbor(int v)
        {
            for (int wi = 0; wi < Words; wi++)
            {
                ulong word = Out[v][wi];
                if (word != 0)
                {
                    return (wi << 6) + BitOperations.TrailingZeroCount(word);
                }
            }

            return -1;
        }

        public void BypassIn(int v, int predecessor)
        {
            for (int wi = 0; wi < Words; wi++)
            {
                ulong word = Out[v][wi];
                while (word != 0)
                {
                    int bit = BitOperations.TrailingZeroCount(word);
                    int w = (wi << 6) + bit;
                    word &= word - 1;

                    if (w != v && Active[w])
                    {
                        AddEdge(predecessor, w);
                    }
                }
            }

            RemoveVertex(v);
        }

        public void BypassOut(int v, int successor)
        {
            for (int wi = 0; wi < Words; wi++)
            {
                ulong word = In[v][wi];
                while (word != 0)
                {
                    int bit = BitOperations.TrailingZeroCount(word);
                    int u = (wi << 6) + bit;
                    word &= word - 1;

                    if (u != v && Active[u])
                    {
                        AddEdge(u, successor);
                    }
                }
            }

            RemoveVertex(v);
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _input = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _ptr;

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            }
            while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (_ptr >= _length)
            {
                _length = _input.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_length <= 0)
                {
                    return 0;
                }
            }

            return _buffer[_ptr++];
        }
    }
