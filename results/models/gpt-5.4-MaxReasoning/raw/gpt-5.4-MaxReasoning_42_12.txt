OpCapability Shader
OpCapability Geometry
%GLSL450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1
OpSource GLSL 450

OpName %main "main"
OpName %gl_in "gl_in"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpMemberName %gl_PerVertex_out 0 "gl_Position"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%uint_6 = OpConstant %uint 6
%float_0 = OpConstant %float 0
%float_0_3 = OpConstant %float 0.300000012
%float_0_015 = OpConstant %float 0.015
%float_1 = OpConstant %float 1
%white = OpConstantComposite %v3float %float_1 %float_1 %float_1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_6 = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_Input_arr_gl_PerVertex_in_6 = OpTypePointer Input %arr_gl_PerVertex_in_6
%ptr_Input_v4float = OpTypePointer Input %v4float

%arr_v3float_6 = OpTypeArray %v3float %uint_6
%ptr_Input_arr_v3float_6 = OpTypePointer Input %arr_v3float_6
%ptr_Input_v3float = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_6 Input
%color = OpVariable %ptr_Input_arr_v3float_6 Input
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %ptr_Output_v3float Output

%main = OpFunction %void None %fn
%entry = OpLabel
%out_pos = OpAccessChain %ptr_Output_v4float %gl_out %int_0

%p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p2_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%p4_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_4 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p2 = OpLoad %v4float %p2_ptr
%p4 = OpLoad %v4float %p4_ptr

%c0_ptr = OpAccessChain %ptr_Input_v3float %color %int_0
%c2_ptr = OpAccessChain %ptr_Input_v3float %color %int_2
%c4_ptr = OpAccessChain %ptr_Input_v3float %color %int_4
%c0 = OpLoad %v3float %c0_ptr
%c2 = OpLoad %v3float %c2_ptr
%c4 = OpLoad %v3float %c4_ptr

%c0_dim = OpVectorTimesScalar %v3float %c0 %float_0_3
%c2_dim = OpVectorTimesScalar %v3float %c2 %float_0_3
%c4_dim = OpVectorTimesScalar %v3float %c4 %float_0_3

OpStore %out_pos %p0
OpStore %outColor %c0_dim
OpEmitVertex
OpStore %out_pos %p2
OpStore %outColor %c2_dim
OpEmitVertex
OpStore %out_pos %p4
OpStore %outColor %c4_dim
OpEmitVertex
OpEndPrimitive

%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p4_xy = OpVectorShuffle %v2float %p4 %p4 0 1

%edge02 = OpFSub %v2float %p2_xy %p0_xy
%dir02 = OpExtInst %v2float %GLSL450 Normalize %edge02
%dir02_x = OpCompositeExtract %float %dir02 0
%dir02_y = OpCompositeExtract %float %dir02 1
%neg_dir02_y = OpFNegate %float %dir02_y
%perp02 = OpCompositeConstruct %v2float %neg_dir02_y %dir02_x
%off02 = OpVectorTimesScalar %v2float %perp02 %float_0_015
%off02_x = OpCompositeExtract %float %off02 0
%off02_y = OpCompositeExtract %float %off02 1
%off02_4 = OpCompositeConstruct %v4float %off02_x %off02_y %float_0 %float_0
%p0_plus = OpFAdd %v4float %p0 %off02_4
%p0_minus = OpFSub %v4float %p0 %off02_4
%p2_plus = OpFAdd %v4float %p2 %off02_4
%p2_minus = OpFSub %v4float %p2 %off02_4

OpStore %out_pos %p0_plus
OpStore %outColor %white
OpEmitVertex
OpStore %out_pos %p0_minus
OpStore %outColor %white
OpEmitVertex
OpStore %out_pos %p2_plus
OpStore %outColor %white
OpEmitVertex
OpStore %out_pos %p2_minus
OpStore %outColor %white
OpEmitVertex
OpEndPrimitive

%edge24 = OpFSub %v2float %p4_xy %p2_xy
%dir24 = OpExtInst %v2float %GLSL450 Normalize %edge24
%dir24_x = OpCompositeExtract %float %dir24 0
%dir24_y = OpCompositeExtract %float %dir24 1
%neg_dir24_y = OpFNegate %float %dir24_y
%perp24 = OpCompositeConstruct %v2float %neg_dir24_y %dir24_x
%off24 = OpVectorTimesScalar %v2float %perp24 %float_0_015
%off24_x = OpCompositeExtract %float %off24 0
%off24_y = OpCompositeExtract %float %off24 1
%off24_4 = OpCompositeConstruct %v4float %off24_x %off24_y %float_0 %float_0
%p2_plus_e = OpFAdd %v4float %p2 %off24_4
%p2_minus_e = OpFSub %v4float %p2 %off24_4
%p4_plus = OpFAdd %v4float %p4 %off24_4
%p4_minus = OpFSub %v4float %p4 %off24_4

OpStore %out_pos %p2_plus_e
OpStore %outColor %white
OpEmitVertex
OpStore %out_pos %p2_minus_e
OpStore %outColor %white
OpEmitVertex
OpStore %out_pos %p4_plus
OpStore %outColor %white
OpEmitVertex
OpStore %out_pos %p4_minus
OpStore %outColor %white
OpEmitVertex
OpEndPrimitive

%edge40 = OpFSub %v2float %p0_xy %p4_xy
%dir40 = OpExtInst %v2float %GLSL450 Normalize %edge40
%dir40_x = OpCompositeExtract %float %dir40 0
%dir40_y = OpCompositeExtract %float %dir40 1
%neg_dir40_y = OpFNegate %float %dir40_y
%perp40 = OpCompositeConstruct %v2float %neg_dir40_y %dir40_x
%off40 = OpVectorTimesScalar %v2float %perp40 %float_0_015
%off40_x = OpCompositeExtract %float %off40 0
%off40_y = OpCompositeExtract %float %off40 1
%off40_4 = OpCompositeConstruct %v4float %off40_x %off40_y %float_0 %float_0
%p4_plus_e = OpFAdd %v4float %p4 %off40_4
%p4_minus_e = OpFSub %v4float %p4 %off40_4
%p0_plus_e = OpFAdd %v4float %p0 %off40_4
%p0_minus_e = OpFSub %v4float %p0 %off40_4

OpStore %out_pos %p4_plus_e
OpStore %outColor %white
OpEmitVertex
OpStore %out_pos %p4_minus_e
OpStore %outColor %white
OpEmitVertex
OpStore %out_pos %p0_plus_e
OpStore %outColor %white
OpEmitVertex
OpStore %out_pos %p0_minus_e
OpStore %outColor %white
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
