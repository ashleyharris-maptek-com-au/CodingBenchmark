import math
import sys


def main():
    data = sys.stdin.buffer.read().split()
    if not data:
        return

    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    capacity = int(next(it))
    depot_x = float(next(it))
    depot_y = float(next(it))

    xs = [0.0] * n
    ys = [0.0] * n
    demand = [0] * n
    for i in range(n):
        xs[i] = float(next(it))
        ys[i] = float(next(it))
        demand[i] = int(next(it))

    if n == 0:
        sys.stdout.write("0.0000000000\n0\n")
        return

    total_demand = sum(demand)
    if any(d > capacity for d in demand) or total_demand > k * capacity:
        sys.stdout.write("-1\n0\n")
        return

    max_routes = min(k, n)
    depot_dist = [
        math.hypot(xs[i] - depot_x, ys[i] - depot_y)
        for i in range(n)
    ]
    angles = [
        math.atan2(ys[i] - depot_y, xs[i] - depot_x)
        for i in range(n)
    ]
    pair_cache = {}

    def dist_pair(a, b):
        if a > b:
            a, b = b, a
        key = (a, b)
        value = pair_cache.get(key)
        if value is None:
            value = math.hypot(xs[a] - xs[b], ys[a] - ys[b])
            pair_cache[key] = value
        return value

    def edge_dist(a, b):
        if a == -1:
            return depot_dist[b]
        if b == -1:
            return depot_dist[a]
        return dist_pair(a, b)

    def route_load(route):
        return sum(demand[c] for c in route)

    def route_distance(route):
        if not route:
            return 0.0
        total = edge_dist(-1, route[0]) + edge_dist(route[-1], -1)
        for i in range(len(route) - 1):
            total += dist_pair(route[i], route[i + 1])
        return total

    def total_distance(routes):
        return sum(route_distance(route) for route in routes)

    def two_opt(route):
        m = len(route)
        if m < 4:
            return route

        improved = True
        while improved:
            improved = False
            for i in range(m - 1):
                a = -1 if i == 0 else route[i - 1]
                b = route[i]
                for j in range(i + 1, m):
                    c = route[j]
                    d = -1 if j == m - 1 else route[j + 1]
                    old_cost = edge_dist(a, b) + edge_dist(c, d)
                    new_cost = edge_dist(a, c) + edge_dist(b, d)
                    if new_cost + 1e-12 < old_cost:
                        route[i:j + 1] = reversed(route[i:j + 1])
                        improved = True
                        break
                if improved:
                    break
        return route

    def build_route(customers):
        if len(customers) <= 1:
            return list(customers)

        order = sorted(
            customers,
            key=lambda c: (depot_dist[c], demand[c], c),
            reverse=True,
        )
        route = [order[0]]
        for customer in order[1:]:
            best_pos = 0
            best_delta = float("inf")
            m = len(route)
            for pos in range(m + 1):
                left = -1 if pos == 0 else route[pos - 1]
                right = -1 if pos == m else route[pos]
                delta = (
                    edge_dist(left, customer)
                    + edge_dist(customer, right)
                    - edge_dist(left, right)
                )
                if delta + 1e-12 < best_delta:
                    best_delta = delta
                    best_pos = pos
            route.insert(best_pos, customer)
        return two_opt(route)

    def removal_delta(route, pos):
        left = -1 if pos == 0 else route[pos - 1]
        customer = route[pos]
        right = -1 if pos == len(route) - 1 else route[pos + 1]
        return (
            edge_dist(left, right)
            - edge_dist(left, customer)
            - edge_dist(customer, right)
        )

    def best_insertion(route, customer):
        if not route:
            return edge_dist(-1, customer) * 2.0, 0

        best_delta = float("inf")
        best_pos = 0
        m = len(route)
        for pos in range(m + 1):
            left = -1 if pos == 0 else route[pos - 1]
            right = -1 if pos == m else route[pos]
            delta = (
                edge_dist(left, customer)
                + edge_dist(customer, right)
                - edge_dist(left, right)
            )
            if delta + 1e-12 < best_delta:
                best_delta = delta
                best_pos = pos
        return best_delta, best_pos

    def relocate_between_routes(routes):
        if len(routes) < 2 or n > 250:
            return routes

        loads = [route_load(route) for route in routes]
        move_limit = max(10, n)
        moves = 0
        while moves < move_limit:
            best_move = None
            best_delta = -1e-9

            for r1, route1 in enumerate(routes):
                for pos, customer in enumerate(route1):
                    remove_gain = removal_delta(route1, pos)
                    for r2, route2 in enumerate(routes):
                        if r1 == r2 or loads[r2] + demand[customer] > capacity:
                            continue
                        insert_cost, insert_pos = best_insertion(route2, customer)
                        delta = remove_gain + insert_cost
                        if delta < best_delta:
                            best_delta = delta
                            best_move = (r1, pos, r2, insert_pos, customer)

            if best_move is None:
                break

            r1, pos, r2, insert_pos, customer = best_move
            route1 = routes[r1]
            route2 = routes[r2]
            del route1[pos]
            route2.insert(insert_pos, customer)
            loads[r1] -= demand[customer]
            loads[r2] += demand[customer]

            if route1:
                two_opt(route1)
            two_opt(route2)

            if not route1:
                routes.pop(r1)
                loads.pop(r1)

            moves += 1

        return routes

    def split_order(order):
        m = len(order)
        inf = float("inf")
        dp = [inf] * (m + 1)
        dp[0] = 0.0
        parents = [None] * (max_routes + 1)
        best_cost = inf
        best_count = -1

        for used in range(1, max_routes + 1):
            nxt = [inf] * (m + 1)
            parent = [-1] * (m + 1)
            for start in range(m):
                base = dp[start]
                if base == inf:
                    continue

                first = order[start]
                load = 0
                path = 0.0
                prev = first
                first_leg = depot_dist[first]

                for end in range(start, m):
                    customer = order[end]
                    load += demand[customer]
                    if load > capacity:
                        break
                    if end > start:
                        path += dist_pair(prev, customer)
                        prev = customer
                    cost = first_leg + path + depot_dist[customer]
                    idx = end + 1
                    value = base + cost
                    if value + 1e-12 < nxt[idx]:
                        nxt[idx] = value
                        parent[idx] = start

            parents[used] = parent
            dp = nxt
            if dp[m] < best_cost:
                best_cost = dp[m]
                best_count = used

        if best_count == -1:
            return None

        routes = []
        end = m
        used = best_count
        while used > 0 and end > 0:
            start = parents[used][end]
            if start < 0:
                return None
            routes.append(order[start:end])
            end = start
            used -= 1
        routes.reverse()
        return routes

    def greedy_packing(ordering):
        routes = []
        loads = []

        for customer in ordering:
            best_route = -1
            best_slack = float("inf")
            customer_angle = angles[customer]
            for ridx, current_load in enumerate(loads):
                slack = capacity - current_load
                if slack < demand[customer]:
                    continue
                new_slack = slack - demand[customer]
                if new_slack + 1e-12 < best_slack:
                    best_slack = new_slack
                    best_route = ridx
                elif abs(new_slack - best_slack) <= 1e-12 and best_route != -1:
                    current_center = sum(angles[c] for c in routes[ridx]) / len(routes[ridx])
                    best_center = sum(angles[c] for c in routes[best_route]) / len(routes[best_route])
                    if abs(customer_angle - current_center) < abs(customer_angle - best_center):
                        best_route = ridx

            if best_route == -1:
                if len(routes) == max_routes:
                    return None
                routes.append([customer])
                loads.append(demand[customer])
            else:
                routes[best_route].append(customer)
                loads[best_route] += demand[customer]

        return routes

    def nearest_neighbor_order():
        remaining = set(range(n))
        start = max(range(n), key=lambda c: (depot_dist[c], demand[c], c))
        order = [start]
        remaining.remove(start)
        current = start
        while remaining:
            nxt = min(remaining, key=lambda c: (dist_pair(current, c), depot_dist[c], c))
            order.append(nxt)
            remaining.remove(nxt)
            current = nxt
        return order

    def candidate_orders():
        if n == 1:
            return [[0]]

        base = sorted(range(n), key=lambda c: (angles[c], depot_dist[c], c))
        sorted_angles = [angles[c] for c in base]
        gaps = []
        for i in range(n):
            current = sorted_angles[i]
            nxt = sorted_angles[(i + 1) % n]
            gap = nxt - current if i + 1 < n else (sorted_angles[0] + 2.0 * math.pi - current)
            gaps.append((gap, (i + 1) % n))

        starts = [0]
        for _, start in sorted(gaps, reverse=True)[:3]:
            starts.append(start)
        starts.append(n // 2)

        seen = set()
        result = []

        def add_order(seq):
            key = tuple(seq)
            if key not in seen:
                seen.add(key)
                result.append(list(seq))

        for start in starts:
            start %= n
            add_order(base[start:] + base[:start])

        rev = list(reversed(base))
        for start in starts:
            start %= n
            add_order(rev[start:] + rev[:start])

        if n <= 250:
            nn = nearest_neighbor_order()
            add_order(nn)
            add_order(list(reversed(nn)))

        return result

    best_routes = None
    best_cost = float("inf")

    for order in candidate_orders():
        segmented = split_order(order)
        if segmented is None:
            continue
        routes = [build_route(group) for group in segmented]
        routes = relocate_between_routes(routes)
        for idx in range(len(routes)):
            two_opt(routes[idx])
        cost = total_distance(routes)
        if cost + 1e-9 < best_cost:
            best_cost = cost
            best_routes = [list(route) for route in routes]

    if best_routes is None:
        fallback_orders = [
            sorted(range(n), key=lambda c: (-demand[c], angles[c], depot_dist[c], c)),
            sorted(range(n), key=lambda c: (-demand[c], -depot_dist[c], angles[c], c)),
            sorted(range(n), key=lambda c: (angles[c], -demand[c], depot_dist[c], c)),
        ]
        for ordering in fallback_orders:
            packed = greedy_packing(ordering)
            if packed is None:
                continue
            routes = [build_route(group) for group in packed]
            routes = relocate_between_routes(routes)
            for idx in range(len(routes)):
                two_opt(routes[idx])
            cost = total_distance(routes)
            if cost + 1e-9 < best_cost:
                best_cost = cost
                best_routes = [list(route) for route in routes]

    if best_routes is None:
        sys.stdout.write("-1\n0\n")
        return

    lines = [f"{best_cost:.10f}", str(len(best_routes))]
    for route in best_routes:
        lines.append(" ".join(str(customer) for customer in route))
    sys.stdout.write("\n".join(lines))


if __name__ == "__main__":
    main()
