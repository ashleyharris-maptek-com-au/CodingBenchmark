%dir = OpFSub %v2float %p1_xy %p0_xy
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%raw_perp = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp = OpExtInst %v2float %1 Normalize %raw_perp
%offset0 = OpVectorTimesScalar %v2float %perp %float_0_04
%offset1 = OpVectorTimesScalar %v2float %perp %float_0_005
%p0_plus_xy = OpFAdd %v2float %p0_xy %offset0
%p0_minus_xy = OpFSub %v2float %p0_xy %offset0
%p1_plus_xy = OpFAdd %v2float %p1_xy %offset1
%p1_minus_xy = OpFSub %v2float %p1_xy %offset1
%p0_z = OpCompositeExtract %float %p0 2
%p1_z = OpCompositeExtract %float %p1 2

%p1_plus_y = OpCompositeExtract %float %p1_plus_xy 1
%p1_minus_x = OpCompositeExtract %float %p1_minus_xy 0
%p1_minus_y = OpCompositeExtract %float %p1_minus_xy 1
