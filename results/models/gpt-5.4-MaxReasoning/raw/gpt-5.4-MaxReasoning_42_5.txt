OpCapability Shader
OpCapability Geometry
%GLSL450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main Invocations 1
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75
%float_1 = OpConstant %float 1.0
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output
%main = OpFunction %void None %func
%entry = OpLabel
%p0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p1_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_1
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%p0 = OpLoad %v3float %p0_ptr
%p1 = OpLoad %v3float %p1_ptr
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr
%dir3 = OpFSub %v3float %p1 %p0
%dir = OpVectorShuffle %v2float %dir3 %dir3 0 1
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_raw = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp = OpExtInst %v2float %GLSL450 Normalize %perp_raw
%shaft_off = OpVectorTimesScalar %v2float %perp %float_0_015
%head_off = OpVectorTimesScalar %v2float %perp %float_0_05
%dir3_75 = OpVectorTimesScalar %v3float %dir3 %float_0_75
%mid = OpFAdd %v3float %p0 %dir3_75
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%mid_xy = OpVectorShuffle %v2float %mid %mid 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p0_z = OpCompositeExtract %float %p0 2
%mid_z = OpCompositeExtract %float %mid 2
%p1_z = OpCompositeExtract %float %p1 2
%shaft0_xy = OpFAdd %v2float %p0_xy %shaft_off
%shaft0_x = OpCompositeExtract %float %shaft0_xy 0
%shaft0_y = OpCompositeExtract %float %shaft0_xy 1
%shaft0_pos = OpCompositeConstruct %v4float %shaft0_x %shaft0_y %p0_z %float_1
%gl_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos %shaft0_pos
OpStore %outColor %c0
OpEmitVertex
%shaft1_xy = OpFSub %v2float %p0_xy %shaft_off
%shaft1_x = OpCompositeExtract %float %shaft1_xy 0
%shaft1_y = OpCompositeExtract %float %shaft1_xy 1
%shaft1_pos = OpCompositeConstruct %v4float %shaft1_x %shaft1_y %p0_z %float_1
OpStore %gl_out_pos %shaft1_pos
OpStore %outColor %c0
OpEmitVertex
%shaft2_xy = OpFAdd %v2float %mid_xy %shaft_off
%shaft2_x = OpCompositeExtract %float %shaft2_xy 0
%shaft2_y = OpCompositeExtract %float %shaft2_xy 1
%shaft2_pos = OpCompositeConstruct %v4float %shaft2_x %shaft2_y %mid_z %float_1
OpStore %gl_out_pos %shaft2_pos
OpStore %outColor %c0
OpEmitVertex
%shaft3_xy = OpFSub %v2float %mid_xy %shaft_off
%shaft3_x = OpCompositeExtract %float %shaft3_xy 0
%shaft3_y = OpCompositeExtract %float %shaft3_xy 1
%shaft3_pos = OpCompositeConstruct %v4float %shaft3_x %shaft3_y %mid_z %float_1
OpStore %gl_out_pos %shaft3_pos
OpStore %outColor %c0
OpEmitVertex
OpEndPrimitive
%head0_xy = OpFAdd %v2float %mid_xy %head_off
%head0_x = OpCompositeExtract %float %head0_xy 0
%head0_y = OpCompositeExtract %float %head0_xy 1
%head0_pos = OpCompositeConstruct %v4float %head0_x %head0_y %mid_z %float_1
OpStore %gl_out_pos %head0_pos
OpStore %outColor %c1
OpEmitVertex
%head1_xy = OpFSub %v2float %mid_xy %head_off
%head1_x = OpCompositeExtract %float %head1_xy 0
%head1_y = OpCompositeExtract %float %head1_xy 1
%head1_pos = OpCompositeConstruct %v4float %head1_x %head1_y %mid_z %float_1
OpStore %gl_out_pos %head1_pos
OpStore %outColor %c1
OpEmitVertex
%tip_x = OpCompositeExtract %float %p1_xy 0
%tip_y = OpCompositeExtract %float %p1_xy 1
%tip_pos = OpCompositeConstruct %v4float %tip_x %tip_y %p1_z %float_1
OpStore %gl_out_pos %tip_pos
OpStore %outColor %c1
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
