declare const require: any;
declare const process: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Town {
  id?: string | number;
  x: number;
  y: number;
  value: number;
}

interface InputData {
  width: number;
  height: number;
  grid: string[];
  ignitions: number[][];
  towns: Town[];
  cutBudget: number;
}

interface Region {
  cells: Int32Array;
  pairU: Int32Array;
  pairV: Int32Array;
  townValues: Float64Array;
  townLocals: Int32Array;
  sortedTownLocals: Int32Array;
  ignitionLocals: Int32Array;
  cuttable: Uint8Array;
  baselineValue: number;
  cuttableCount: number;
}

interface Candidate {
  cutGlobals: number[];
  cutCount: number;
  savedValue: number;
  kind: string;
}

class PushRelabel {
  private static readonly EPS = 1e-9;

  readonly n: number;
  readonly head: Int32Array;
  readonly to: Int32Array;
  readonly next: Int32Array;
  readonly cap: Float64Array;
  edgeCount: number;

  constructor(n: number, edgeSlots: number) {
    this.n = n;
    this.head = new Int32Array(n);
    this.head.fill(-1);
    this.to = new Int32Array(edgeSlots);
    this.next = new Int32Array(edgeSlots);
    this.cap = new Float64Array(edgeSlots);
    this.edgeCount = 0;
  }

  addEdge(u: number, v: number, c: number): void {
    const e = this.edgeCount;
    this.to[e] = v;
    this.cap[e] = c;
    this.next[e] = this.head[u];
    this.head[u] = e;

    this.to[e + 1] = u;
    this.cap[e + 1] = 0;
    this.next[e + 1] = this.head[v];
    this.head[v] = e + 1;

    this.edgeCount += 2;
  }

  private bfs(s: number, t: number, level: Int32Array): boolean {
    const q = new Int32Array(this.n);
    let qHead = 0;
    let qTail = 0;

    level.fill(-1);
    level[s] = 0;
    q[qTail++] = s;

    while (qHead < qTail) {
      const u = q[qHead++];
      for (let e = this.head[u]; e !== -1; e = this.next[e]) {
        const v = this.to[e];
        if (level[v] === -1 && this.cap[e] > PushRelabel.EPS) {
          level[v] = level[u] + 1;
          if (v === t) {
            return true;
          }
          q[qTail++] = v;
        }
      }
    }

    return level[t] !== -1;
  }

  maxFlow(s: number, t: number): number {
    if (s === t) {
      return 0;
    }

    let flow = 0;
    const level = new Int32Array(this.n);
    const ptr = new Int32Array(this.n);
    const nodeStack = new Int32Array(this.n);
    const edgeStack = new Int32Array(this.n);
    const minStack = new Float64Array(this.n);

    while (this.bfs(s, t, level)) {
      ptr.set(this.head);
      let depth = 0;
      nodeStack[0] = s;
      minStack[0] = Number.POSITIVE_INFINITY;

      while (depth >= 0) {
        const u = nodeStack[depth];

        if (u === t) {
          const pushed = minStack[depth];
          flow += pushed;

          let rewindDepth = -1;
          for (let i = 1; i <= depth; i++) {
            const e = edgeStack[i];
            this.cap[e] -= pushed;
            this.cap[e ^ 1] += pushed;
            if (rewindDepth === -1 && this.cap[e] <= PushRelabel.EPS) {
              rewindDepth = i - 1;
            }
          }

          if (rewindDepth === -1) {
            rewindDepth = depth - 1;
          }

          if (rewindDepth >= 0) {
            const saturatedEdge = edgeStack[rewindDepth + 1];
            ptr[nodeStack[rewindDepth]] = this.next[saturatedEdge];
          }

          depth = rewindDepth;
          continue;
        }

        let e = ptr[u];
        while (e !== -1) {
          const v = this.to[e];
          if (this.cap[e] > PushRelabel.EPS && level[v] === level[u] + 1) {
            break;
          }
          e = this.next[e];
        }
        ptr[u] = e;

        if (e === -1) {
          level[u] = -1;
          if (depth === 0) {
            break;
          }
          const parent = nodeStack[depth - 1];
          ptr[parent] = this.next[edgeStack[depth]];
          depth--;
          continue;
        }

        const v = this.to[e];
        depth++;
        nodeStack[depth] = v;
        edgeStack[depth] = e;
        minStack[depth] = Math.min(minStack[depth - 1], this.cap[e]);
      }
    }

    return flow;
  }

  minCutSourceSide(s: number): Uint8Array {
    const seen = new Uint8Array(this.n);
    const q = new Int32Array(this.n);
    let qHead = 0;
    let qTail = 0;

    seen[s] = 1;
    q[qTail++] = s;

    while (qHead < qTail) {
      const u = q[qHead++];
      for (let e = this.head[u]; e !== -1; e = this.next[e]) {
        const v = this.to[e];
        if (seen[v] === 0 && this.cap[e] > PushRelabel.EPS) {
          seen[v] = 1;
          q[qTail++] = v;
        }
      }
    }

    return seen;
  }
}

function buildRegion(
  width: number,
  height: number,
  burnable: Uint8Array,
  blocked: Uint8Array,
  ignitionMarks: Uint8Array,
  towns: Town[],
  hasTown: Uint8Array
): Region | null {
  const total = width * height;
  const reachable = new Uint8Array(total);
  const queue = new Int32Array(total);
  let tail = 0;

  for (let idx = 0; idx < total; idx++) {
    if (ignitionMarks[idx] !== 0 && burnable[idx] !== 0 && blocked[idx] === 0 && reachable[idx] === 0) {
      reachable[idx] = 1;
      queue[tail++] = idx;
    }
  }

  for (let qHead = 0; qHead < tail; qHead++) {
    const idx = queue[qHead];
    const x = idx % width;

    if (x > 0) {
      const n = idx - 1;
      if (burnable[n] !== 0 && blocked[n] === 0 && reachable[n] === 0) {
        reachable[n] = 1;
        queue[tail++] = n;
      }
    }
    if (x + 1 < width) {
      const n = idx + 1;
      if (burnable[n] !== 0 && blocked[n] === 0 && reachable[n] === 0) {
        reachable[n] = 1;
        queue[tail++] = n;
      }
    }
    if (idx >= width) {
      const n = idx - width;
      if (burnable[n] !== 0 && blocked[n] === 0 && reachable[n] === 0) {
        reachable[n] = 1;
        queue[tail++] = n;
      }
    }
    if (idx + width < total) {
      const n = idx + width;
      if (burnable[n] !== 0 && blocked[n] === 0 && reachable[n] === 0) {
        reachable[n] = 1;
        queue[tail++] = n;
      }
    }
  }

  if (tail === 0) {
    return null;
  }

  const threatenedByCell = new Map<number, number>();
  let baselineValue = 0;

  for (const town of towns) {
    const x = town.x | 0;
    const y = town.y | 0;
    const value = Number(town.value);
    if (!Number.isFinite(value) || value <= 0) {
      continue;
    }
    if (x < 0 || x >= width || y < 0 || y >= height) {
      continue;
    }
    const idx = y * width + x;
    if (burnable[idx] === 0 || blocked[idx] !== 0 || reachable[idx] === 0) {
      continue;
    }
    baselineValue += value;
    threatenedByCell.set(idx, (threatenedByCell.get(idx) ?? 0) + value);
  }

  if (baselineValue <= 0) {
    return null;
  }

  const cells = queue.slice(0, tail);
  const localIndex = new Int32Array(total);
  localIndex.fill(-1);
  for (let i = 0; i < cells.length; i++) {
    localIndex[cells[i]] = i;
  }

  const townValues = new Float64Array(cells.length);
  const townLocals: number[] = [];
  threatenedByCell.forEach((value, globalIdx) => {
    const local = localIndex[globalIdx];
    if (local >= 0) {
      townValues[local] = value;
      townLocals.push(local);
    }
  });

  const sortedTownLocals = townLocals.slice().sort((a, b) => townValues[b] - townValues[a]);
  const ignitionLocals: number[] = [];
  const cuttable = new Uint8Array(cells.length);
  let cuttableCount = 0;

  for (let i = 0; i < cells.length; i++) {
    const globalIdx = cells[i];
    if (ignitionMarks[globalIdx] !== 0) {
      ignitionLocals.push(i);
    }
    if (ignitionMarks[globalIdx] === 0 && hasTown[globalIdx] === 0) {
      cuttable[i] = 1;
      cuttableCount++;
    }
  }

  const pairU: number[] = [];
  const pairV: number[] = [];

  for (let i = 0; i < cells.length; i++) {
    const globalIdx = cells[i];
    const x = globalIdx % width;

    if (x + 1 < width) {
      const right = localIndex[globalIdx + 1];
      if (right !== -1) {
        pairU.push(i);
        pairV.push(right);
      }
    }
    if (globalIdx + width < total) {
      const down = localIndex[globalIdx + width];
      if (down !== -1) {
        pairU.push(i);
        pairV.push(down);
      }
    }
  }

  return {
    cells,
    pairU: Int32Array.from(pairU),
    pairV: Int32Array.from(pairV),
    townValues,
    townLocals: Int32Array.from(townLocals),
    sortedTownLocals: Int32Array.from(sortedTownLocals),
    ignitionLocals: Int32Array.from(ignitionLocals),
    cuttable,
    baselineValue,
    cuttableCount,
  };
}

function candidateFromCut(
  region: Region,
  sourceSide: Uint8Array,
  kind: string,
  requiredTownLocals: Int32Array | null
): Candidate | null {
  if (requiredTownLocals !== null) {
    for (let i = 0; i < requiredTownLocals.length; i++) {
      const local = requiredTownLocals[i];
      if (sourceSide[2 * local + 1] !== 0) {
        return null;
      }
    }
  }

  let savedValue = 0;
  for (let i = 0; i < region.townLocals.length; i++) {
    const local = region.townLocals[i];
    if (sourceSide[2 * local + 1] === 0) {
      savedValue += region.townValues[local];
    }
  }

  if (savedValue <= 0) {
    return null;
  }

  const cutGlobals: number[] = [];
  for (let local = 0; local < region.cells.length; local++) {
    if (region.cuttable[local] !== 0 && sourceSide[2 * local] !== 0 && sourceSide[2 * local + 1] === 0) {
      cutGlobals.push(region.cells[local]);
    }
  }

  if (cutGlobals.length === 0) {
    return null;
  }

  cutGlobals.sort((a, b) => a - b);
  return {
    cutGlobals,
    cutCount: cutGlobals.length,
    savedValue,
    kind,
  };
}

function runPrizeCut(region: Region, lambda: number): Candidate | null {
  const cellCount = region.cells.length;
  const source = 2 * cellCount;
  const sink = source + 1;
  const nodeCount = sink + 1;
  const edgeSlots =
    2 * cellCount +
    4 * region.pairU.length +
    2 * region.ignitionLocals.length +
    2 * region.townLocals.length;

  const graph = new PushRelabel(nodeCount, edgeSlots);
  const inf = region.baselineValue + lambda * (cellCount + 1) + 1;

  for (let local = 0; local < cellCount; local++) {
    const inNode = 2 * local;
    const outNode = inNode + 1;
    const cutCost = region.cuttable[local] !== 0 ? lambda : inf;
    graph.addEdge(inNode, outNode, cutCost);
  }

  for (let i = 0; i < region.ignitionLocals.length; i++) {
    graph.addEdge(source, 2 * region.ignitionLocals[i], inf);
  }

  for (let i = 0; i < region.townLocals.length; i++) {
    const local = region.townLocals[i];
    graph.addEdge(2 * local + 1, sink, region.townValues[local]);
  }

  for (let i = 0; i < region.pairU.length; i++) {
    const a = region.pairU[i];
    const b = region.pairV[i];
    graph.addEdge(2 * a + 1, 2 * b, inf);
    graph.addEdge(2 * b + 1, 2 * a, inf);
  }

  graph.maxFlow(source, sink);
  const side = graph.minCutSourceSide(source);
  return candidateFromCut(region, side, `global-${lambda.toFixed(3)}`, null);
}

function runForcedSubsetCut(region: Region, selectedLocals: Int32Array): Candidate | null {
  const cellCount = region.cells.length;
  const source = 2 * cellCount;
  const sink = source + 1;
  const nodeCount = sink + 1;
  const edgeSlots =
    2 * cellCount +
    4 * region.pairU.length +
    2 * region.ignitionLocals.length +
    2 * selectedLocals.length;

  const graph = new PushRelabel(nodeCount, edgeSlots);
  const inf = cellCount + 1;

  for (let local = 0; local < cellCount; local++) {
    const inNode = 2 * local;
    const outNode = inNode + 1;
    const cutCost = region.cuttable[local] !== 0 ? 1 : inf;
    graph.addEdge(inNode, outNode, cutCost);
  }

  for (let i = 0; i < region.ignitionLocals.length; i++) {
    graph.addEdge(source, 2 * region.ignitionLocals[i], inf);
  }

  for (let i = 0; i < selectedLocals.length; i++) {
    const local = selectedLocals[i];
    graph.addEdge(2 * local + 1, sink, inf);
  }

  for (let i = 0; i < region.pairU.length; i++) {
    const a = region.pairU[i];
    const b = region.pairV[i];
    graph.addEdge(2 * a + 1, 2 * b, inf);
    graph.addEdge(2 * b + 1, 2 * a, inf);
  }

  graph.maxFlow(source, sink);
  const side = graph.minCutSourceSide(source);
  return candidateFromCut(region, side, `subset-${selectedLocals.length}`, selectedLocals);
}

function chooseBetter(a: Candidate | null, b: Candidate | null): Candidate | null {
  if (a === null) {
    return b;
  }
  if (b === null) {
    return a;
  }
  if (Math.abs(a.savedValue - b.savedValue) > 1e-9) {
    return a.savedValue > b.savedValue ? a : b;
  }
  if (a.cutCount !== b.cutCount) {
    return a.cutCount < b.cutCount ? a : b;
  }
  const aDensity = a.savedValue / a.cutCount;
  const bDensity = b.savedValue / b.cutCount;
  if (Math.abs(aDensity - bDensity) > 1e-9) {
    return aDensity > bDensity ? a : b;
  }
  return a.kind <= b.kind ? a : b;
}

function collectCandidates(region: Region, budget: number): Candidate[] {
  const collected = new Map<string, Candidate>();

  const keep = (candidate: Candidate | null): void => {
    if (candidate === null || candidate.cutCount <= 0 || candidate.cutCount > budget) {
      return;
    }
    const key = candidate.cutGlobals.join(',');
    const prev = collected.get(key) ?? null;
    const best = chooseBetter(prev, candidate);
    if (best !== null) {
      collected.set(key, best);
    }
  };

  const lambdaMemo = new Map<string, Candidate | null>();
  const evalLambda = (lambda: number): Candidate | null => {
    const safeLambda = lambda < 0 ? 0 : lambda;
    const key = safeLambda.toFixed(6);
    if (!lambdaMemo.has(key)) {
      lambdaMemo.set(key, runPrizeCut(region, safeLambda));
    }
    return lambdaMemo.get(key) ?? null;
  };

  const upper = Math.max(1, region.baselineValue + 1);
  const samples: number[] = [0, upper];
  samples.push(region.baselineValue / Math.max(1, budget));
  samples.push(region.baselineValue / Math.max(1, 2 * budget));

  let low = 0;
  let high = upper;
  for (let i = 0; i < 8; i++) {
    const mid = (low + high) * 0.5;
    samples.push(mid);
    const candidate = evalLambda(mid);
    if (candidate !== null && candidate.cutCount > budget) {
      low = mid;
    } else {
      high = mid;
    }
  }
  samples.push(low);
  samples.push(high);
  samples.push((low + high) * 0.5);

  for (const lambda of samples) {
    keep(evalLambda(lambda));
  }

  const townCount = region.sortedTownLocals.length;
  const singleLimit = Math.min(region.cells.length <= 60000 ? 6 : 4, townCount);

  for (let i = 0; i < singleLimit; i++) {
    keep(runForcedSubsetCut(region, Int32Array.of(region.sortedTownLocals[i])));
  }

  const prefixSizes = region.cells.length <= 60000 ? [2, 4, 8] : [2, 4];
  for (const size of prefixSizes) {
    if (size <= townCount) {
      keep(runForcedSubsetCut(region, region.sortedTownLocals.slice(0, size)));
    }
  }

  return Array.from(collected.values());
}

function solveFirebreaks(problem: InputData): number[] {
  const width = Math.max(0, problem.width | 0);
  const height = Math.max(0, problem.height | 0);
  const total = width * height;
  const budget = Math.max(0, Math.floor(Number(problem.cutBudget) || 0));

  if (width === 0 || height === 0 || total === 0 || budget === 0) {
    return [];
  }

  const burnable = new Uint8Array(total);
  for (let y = 0; y < height; y++) {
    const row = typeof problem.grid[y] === 'string' ? problem.grid[y] : '';
    for (let x = 0; x < width; x++) {
      if (row.charCodeAt(x) === 46) {
        burnable[y * width + x] = 1;
      }
    }
  }

  const ignitionMarks = new Uint8Array(total);
  for (const p of problem.ignitions ?? []) {
    if (!Array.isArray(p) || p.length < 2) {
      continue;
    }
    const x = p[0] | 0;
    const y = p[1] | 0;
    if (x < 0 || x >= width || y < 0 || y >= height) {
      continue;
    }
    const idx = y * width + x;
    if (burnable[idx] !== 0) {
      ignitionMarks[idx] = 1;
    }
  }

  const towns: Town[] = Array.isArray(problem.towns) ? problem.towns : [];
  const hasTown = new Uint8Array(total);
  for (const town of towns) {
    const x = town.x | 0;
    const y = town.y | 0;
    if (x >= 0 && x < width && y >= 0 && y < height) {
      hasTown[y * width + x] = 1;
    }
  }

  const blocked = new Uint8Array(total);
  const chosen: number[] = [];
  let remaining = budget;
  const maxRounds = Math.min(20, Math.max(6, Math.ceil(Math.log2(budget + 1)) * 4));

  for (let round = 0; round < maxRounds && remaining > 0; round++) {
    const region = buildRegion(width, height, burnable, blocked, ignitionMarks, towns, hasTown);
    if (region === null || region.cuttableCount === 0) {
      break;
    }

    const candidates = collectCandidates(region, remaining);
    let best: Candidate | null = null;
    for (const candidate of candidates) {
      best = chooseBetter(best, candidate);
    }

    if (best === null) {
      break;
    }

    let applied = 0;
    for (const idx of best.cutGlobals) {
      if (applied >= remaining) {
        break;
      }
      if (blocked[idx] === 0) {
        blocked[idx] = 1;
        chosen.push(idx);
        applied++;
      }
    }

    if (applied === 0) {
      break;
    }
    remaining -= applied;
  }

  chosen.sort((a, b) => a - b);
  return chosen;
}

const problem = data as InputData;
const firebreaks = solveFirebreaks(problem).map((idx) => [idx % problem.width, Math.floor(idx / problem.width)]);
process.stdout.write(JSON.stringify({ firebreaks }));
