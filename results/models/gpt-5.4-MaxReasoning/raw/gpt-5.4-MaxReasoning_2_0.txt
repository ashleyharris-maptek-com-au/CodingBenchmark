#nullable disable
using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    private const long INF = long.MaxValue / 4;
    private const int ExactOddThreshold = 20;
    private const int GreedyOddThreshold = 128;
    private const int GreedyNodeThreshold = 3000;
    private const int GreedyEdgeThreshold = 30000;

    private static int _n;
    private static int _m;
    private static int[] _head = Array.Empty<int>();
    private static int[] _to = Array.Empty<int>();
    private static int[] _next = Array.Empty<int>();
    private static int[] _edgeId = Array.Empty<int>();
    private static int[] _u = Array.Empty<int>();
    private static int[] _v = Array.Empty<int>();
    private static int[] _degree = Array.Empty<int>();
    private static long[] _w = Array.Empty<long>();
    private static int _arcCount;

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        _n = fs.NextInt();
        _m = fs.NextInt();

        if (_n <= 0)
        {
            return;
        }

        _head = new int[_n];
        Array.Fill(_head, -1);
        _to = new int[_m * 2];
        _next = new int[_m * 2];
        _edgeId = new int[_m * 2];
        _u = new int[_m];
        _v = new int[_m];
        _w = new long[_m];
        _degree = new int[_n];
        _arcCount = 0;

        bool allNonNegative = true;
        for (int i = 0; i < _m; i++)
        {
            int a = fs.NextInt();
            int b = fs.NextInt();
            long w = fs.NextLong();
            if (w < 0)
            {
                allNonNegative = false;
            }

            _u[i] = a;
            _v[i] = b;
            _w[i] = w;
            _degree[a]++;
            _degree[b]++;
            AddArc(a, b, i);
            AddArc(b, a, i);
        }

        if (_m == 0)
        {
            WriteRoute(new List<int> { 0 });
            return;
        }

        if (!AllUsedEdgesReachableFromZero())
        {
            WriteRoute(new List<int> { 0 });
            return;
        }

        var oddVertices = new List<int>();
        for (int i = 0; i < _n; i++)
        {
            if ((_degree[i] & 1) != 0)
            {
                oddVertices.Add(i);
            }
        }

        int[] multiplicity = new int[_m];
        if (oddVertices.Count == 0)
        {
            Array.Fill(multiplicity, 1);
        }
        else
        {
            bool augmented = false;
            bool canUseShortestPathAugmentation =
                allNonNegative &&
                oddVertices.Count <= GreedyOddThreshold &&
                _n <= GreedyNodeThreshold &&
                _m <= GreedyEdgeThreshold;

            if (canUseShortestPathAugmentation)
            {
                if (TryBuildShortestPathAugmentation(oddVertices, multiplicity))
                {
                    augmented = true;
                }
            }

            if (!augmented)
            {
                Array.Fill(multiplicity, 2);
            }
        }

        var route = BuildEulerTour(multiplicity);
        if (route == null || route.Count == 0)
        {
            WriteRoute(new List<int> { 0 });
            return;
        }

        WriteRoute(route);
    }

    private static void AddArc(int from, int to, int edge)
    {
        _to[_arcCount] = to;
        _edgeId[_arcCount] = edge;
        _next[_arcCount] = _head[from];
        _head[from] = _arcCount;
        _arcCount++;
    }

    private static bool AllUsedEdgesReachableFromZero()
    {
        var seen = new bool[_n];
        var stack = new int[_n];
        int sp = 0;
        seen[0] = true;
        stack[sp++] = 0;

        while (sp > 0)
        {
            int node = stack[--sp];
            for (int arc = _head[node]; arc != -1; arc = _next[arc])
            {
                int nxt = _to[arc];
                if (!seen[nxt])
                {
                    seen[nxt] = true;
                    stack[sp++] = nxt;
                }
            }
        }

        for (int i = 0; i < _n; i++)
        {
            if (_degree[i] > 0 && !seen[i])
            {
                return false;
            }
        }

        return true;
    }

    private static bool TryBuildShortestPathAugmentation(List<int> oddVertices, int[] multiplicity)
    {
        Array.Fill(multiplicity, 1);

        long[][] pairDist;
        int[][] prevEdgeBySource;
        if (!ComputePairwiseOddData(oddVertices, out pairDist, out prevEdgeBySource))
        {
            return false;
        }

        List<(int A, int B)> pairs;
        if (oddVertices.Count <= ExactOddThreshold)
        {
            pairs = ExactMinimumMatching(pairDist);
        }
        else
        {
            pairs = GreedyMatching(pairDist);
        }

        if (pairs == null)
        {
            return false;
        }

        foreach (var pair in pairs)
        {
            int sourceIndex = pair.A;
            int sourceNode = oddVertices[sourceIndex];
            int current = oddVertices[pair.B];

            while (current != sourceNode)
            {
                int edge = prevEdgeBySource[sourceIndex][current];
                if (edge < 0)
                {
                    return false;
                }

                multiplicity[edge]++;
                current = OtherEndpoint(edge, current);
            }
        }

        return true;
    }

    private static bool ComputePairwiseOddData(
        List<int> oddVertices,
        out long[][] pairDist,
        out int[][] prevEdgeBySource)
    {
        int k = oddVertices.Count;
        pairDist = new long[k][];
        prevEdgeBySource = new int[k][];

        for (int i = 0; i < k; i++)
        {
            long[] dist = new long[_n];
            Array.Fill(dist, INF);
            int[] prevEdge = new int[_n];
            Array.Fill(prevEdge, -1);

            int source = oddVertices[i];
            var pq = new PriorityQueue<int, long>();
            dist[source] = 0;
            pq.Enqueue(source, 0);

            while (pq.Count > 0)
            {
                pq.TryDequeue(out int node, out long curDist);
                if (curDist != dist[node])
                {
                    continue;
                }

                for (int arc = _head[node]; arc != -1; arc = _next[arc])
                {
                    int edge = _edgeId[arc];
                    int nxt = _to[arc];
                    long nextDist = curDist + _w[edge];
                    if (nextDist < dist[nxt])
                    {
                        dist[nxt] = nextDist;
                        prevEdge[nxt] = edge;
                        pq.Enqueue(nxt, nextDist);
                    }
                }
            }

            var row = new long[k];
            for (int j = 0; j < k; j++)
            {
                row[j] = dist[oddVertices[j]];
                if (row[j] >= INF)
                {
                    return false;
                }
            }

            pairDist[i] = row;
            prevEdgeBySource[i] = prevEdge;
        }

        return true;
    }

    private static List<(int A, int B)> ExactMinimumMatching(long[][] pairDist)
    {
        int k = pairDist.Length;
        int fullMask = (1 << k) - 1;
        long[] memo = new long[1 << k];
        Array.Fill(memo, -1);
        int[] choice = new int[1 << k];

        long Solve(int mask)
        {
            if (mask == fullMask)
            {
                return 0;
            }

            long cached = memo[mask];
            if (cached != -1)
            {
                return cached;
            }

            int i = FirstFree(mask, k);
            int nextMaskBase = mask | (1 << i);
            long best = INF;
            int bestJ = -1;

            for (int j = i + 1; j < k; j++)
            {
                if ((mask & (1 << j)) != 0)
                {
                    continue;
                }

                long d = pairDist[i][j];
                if (d >= INF)
                {
                    continue;
                }

                long candidate = d + Solve(nextMaskBase | (1 << j));
                if (candidate < best)
                {
                    best = candidate;
                    bestJ = j;
                }
            }

            choice[mask] = bestJ;
            memo[mask] = best;
            return best;
        }

        if (Solve(0) >= INF)
        {
            return null;
        }

        var result = new List<(int A, int B)>(k / 2);
        int currentMask = 0;
        while (currentMask != fullMask)
        {
            int i = FirstFree(currentMask, k);
            int j = choice[currentMask];
            if (j < 0)
            {
                return null;
            }

            result.Add((i, j));
            currentMask |= (1 << i) | (1 << j);
        }

        return result;
    }

    private static List<(int A, int B)> GreedyMatching(long[][] pairDist)
    {
        int k = pairDist.Length;
        var used = new bool[k];
        var result = new List<(int A, int B)>(k / 2);

        for (int i = 0; i < k; i++)
        {
            if (used[i])
            {
                continue;
            }

            used[i] = true;
            long best = INF;
            int bestJ = -1;

            for (int j = i + 1; j < k; j++)
            {
                if (used[j])
                {
                    continue;
                }

                long d = pairDist[i][j];
                if (d < best)
                {
                    best = d;
                    bestJ = j;
                }
            }

            if (bestJ < 0 || best >= INF)
            {
                return null;
            }

            used[bestJ] = true;
            result.Add((i, bestJ));
        }

        return result;
    }

    private static List<int> BuildEulerTour(int[] multiplicity)
    {
        long totalTraversals = 0;
        for (int i = 0; i < multiplicity.Length; i++)
        {
            totalTraversals += multiplicity[i];
        }

        if (totalTraversals == 0)
        {
            return new List<int> { 0 };
        }

        int limit = checked((int)(totalTraversals + 1));
        var remaining = (int[])multiplicity.Clone();
        var cursor = (int[])_head.Clone();
        var stack = new int[limit];
        var circuit = new List<int>(limit);

        int sp = 0;
        stack[sp++] = 0;

        while (sp > 0)
        {
            int node = stack[sp - 1];
            int arc = cursor[node];

            while (arc != -1 && remaining[_edgeId[arc]] == 0)
            {
                arc = _next[arc];
                cursor[node] = arc;
            }

            if (arc == -1)
            {
                circuit.Add(node);
                sp--;
            }
            else
            {
                int edge = _edgeId[arc];
                remaining[edge]--;
                stack[sp++] = _to[arc];
            }
        }

        if (circuit.Count != limit)
        {
            return null;
        }

        circuit.Reverse();
        return circuit;
    }

    private static int OtherEndpoint(int edge, int node)
    {
        return _u[edge] ^ _v[edge] ^ node;
    }

    private static int FirstFree(int mask, int bits)
    {
        for (int i = 0; i < bits; i++)
        {
            if ((mask & (1 << i)) == 0)
            {
                return i;
            }
        }

        return -1;
    }

    private static void WriteRoute(List<int> route)
    {
        using var writer = new StreamWriter(Console.OpenStandardOutput(), System.Text.Encoding.ASCII, 1 << 16, false);
        for (int i = 0; i < route.Count; i++)
        {
            if (i > 0)
            {
                writer.Write(' ');
            }

            writer.Write(route[i]);
        }

        writer.WriteLine();
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _index;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        public int NextInt()
        {
            int c = SkipWhitespace();
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        public long NextLong()
        {
            int c = SkipWhitespace();
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            long value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int SkipWhitespace()
        {
            int c;
            do
            {
                c = Read();
            } while (c != -1 && c <= 32);

            return c;
        }

        private int Read()
        {
            if (_index >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _index = 0;
                if (_length == 0)
                {
                    return -1;
                }
            }

            return _buffer[_index++];
        }
    }
}
