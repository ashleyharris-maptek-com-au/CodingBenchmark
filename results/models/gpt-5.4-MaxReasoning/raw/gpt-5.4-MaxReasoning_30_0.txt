import sys
import tempfile
from array import array


def fast_ints():
    data = sys.stdin.buffer
    num = 0
    sign = 1
    in_num = False
    while True:
        chunk = data.read(1 << 20)
        if not chunk:
            break
        for b in chunk:
            if 48 <= b <= 57:
                num = num * 10 + (b - 48)
                in_num = True
            elif b == 45:
                sign = -1
            elif in_num:
                yield sign * num
                num = 0
                sign = 1
                in_num = False
            else:
                sign = 1
    if in_num:
        yield sign * num


class XorShift32:
    __slots__ = ("state",)

    def __init__(self, seed=2463534242):
        self.state = seed & 0xFFFFFFFF or 1

    def next(self):
        x = self.state
        x ^= (x << 13) & 0xFFFFFFFF
        x ^= (x >> 17) & 0xFFFFFFFF
        x ^= (x << 5) & 0xFFFFFFFF
        self.state = x & 0xFFFFFFFF or 1
        return self.state


def lit_true(lit, assign):
    v = -lit if lit < 0 else lit
    val = assign[v]
    return (lit > 0 and val == 1) or (lit < 0 and val == 0)


def build_initial_assignment(n, pos_count, neg_count, rng, restart):
    assign = bytearray(n + 1)
    if restart == 0:
        for v in range(1, n + 1):
            assign[v] = 1 if pos_count[v] >= neg_count[v] else 0
        return assign
    for v in range(1, n + 1):
        p = pos_count[v]
        q = neg_count[v]
        if p > q:
            val = 1
        elif p < q:
            val = 0
        else:
            val = rng.next() & 1
        if (rng.next() & 15) == 0:
            val ^= 1
        assign[v] = val
    return assign


def exact_dpll(n, m, a_arr, b_arr, c_arr, pos_count, neg_count, preferred):
    sys.setrecursionlimit(max(1000, n + 100))
    order = list(range(1, n + 1))
    order.sort(key=lambda v: pos_count[v] + neg_count[v], reverse=True)
    assign = [-1] * (n + 1)
    trail = []

    def set_var(v, val):
        cur = assign[v]
        if cur == -1:
            assign[v] = val
            trail.append(v)
            return True
        return cur == val

    def undo(snap):
        while len(trail) > snap:
            assign[trail.pop()] = -1

    def propagate():
        while True:
            changed = False
            for i in range(m):
                a = a_arr[i]
                b = b_arr[i]
                c = c_arr[i]

                unassigned = 0
                last = 0

                for lit in (a, b, c):
                    v = assign[-lit if lit < 0 else lit]
                    if v == -1:
                        unassigned += 1
                        last = lit
                    elif (lit > 0 and v == 1) or (lit < 0 and v == 0):
                        unassigned = -1
                        break

                if unassigned == -1:
                    continue
                if unassigned == 0:
                    return False
                if unassigned == 1:
                    v = -last if last < 0 else last
                    val = 0 if last < 0 else 1
                    if not set_var(v, val):
                        return False
                    changed = True
            if not changed:
                return True

    def dfs():
        snap = len(trail)
        if not propagate():
            undo(snap)
            return False

        var = 0
        for v in order:
            if assign[v] == -1:
                var = v
                break
        if var == 0:
            return True

        after_prop = len(trail)
        pref = preferred[var]
        if set_var(var, pref) and dfs():
            return True
        undo(after_prop)
        if set_var(var, 1 - pref) and dfs():
            return True
        undo(after_prop)
        undo(snap)
        return False

    if dfs():
        out = bytearray(n + 1)
        for v in range(1, n + 1):
            val = assign[v]
            out[v] = 0 if val <= 0 else 1
        return out
    return None


def build_occurrence_arrays(n, m, a_arr, b_arr, c_arr, pos_count, neg_count):
    pos_off = array("I", [0]) * (n + 2)
    neg_off = array("I", [0]) * (n + 2)
    for v in range(1, n + 1):
        pos_off[v + 1] = pos_off[v] + pos_count[v]
        neg_off[v + 1] = neg_off[v] + neg_count[v]

    pos_occ = array("I", [0]) * pos_off[n + 1]
    neg_occ = array("I", [0]) * neg_off[n + 1]
    pos_ptr = array("I", pos_off)
    neg_ptr = array("I", neg_off)

    for ci in range(m):
        for lit in (a_arr[ci], b_arr[ci], c_arr[ci]):
            if lit > 0:
                p = pos_ptr[lit]
                pos_occ[p] = ci
                pos_ptr[lit] = p + 1
            else:
                v = -lit
                p = neg_ptr[v]
                neg_occ[p] = ci
                neg_ptr[v] = p + 1
    return pos_off, neg_off, pos_occ, neg_occ


def walksat_solver(n, m, a_arr, b_arr, c_arr, pos_count, neg_count):
    rng = XorShift32()
    pos_off, neg_off, pos_occ, neg_occ = build_occurrence_arrays(
        n, m, a_arr, b_arr, c_arr, pos_count, neg_count
    )

    best_assign = None
    best_unsat = m + 1
    restarts = 4 if m <= 500000 else 2

    for restart in range(restarts):
        assign = build_initial_assignment(n, pos_count, neg_count, rng, restart)
        sat_count = bytearray(m)
        where = array("i", [-1]) * m
        unsat = []

        for ci in range(m):
            cnt = 0
            a = a_arr[ci]
            b = b_arr[ci]
            c = c_arr[ci]
            if lit_true(a, assign):
                cnt += 1
            if lit_true(b, assign):
                cnt += 1
            if lit_true(c, assign):
                cnt += 1
            sat_count[ci] = cnt
            if cnt == 0:
                where[ci] = len(unsat)
                unsat.append(ci)

        if not unsat:
            return assign

        if len(unsat) < best_unsat:
            best_unsat = len(unsat)
            best_assign = assign[:]

        def remove_unsat(ci):
            pos = where[ci]
            if pos < 0:
                return
            last = unsat.pop()
            if pos < len(unsat):
                unsat[pos] = last
                where[last] = pos
            where[ci] = -1

        def add_unsat(ci):
            if where[ci] >= 0:
                return
            where[ci] = len(unsat)
            unsat.append(ci)

        def break_count(v):
            total = 0
            if assign[v]:
                start = pos_off[v]
                end = pos_off[v + 1]
                opposite = -v
                occ = pos_occ
            else:
                start = neg_off[v]
                end = neg_off[v + 1]
                opposite = v
                occ = neg_occ

            for p in range(start, end):
                ci = occ[p]
                if sat_count[ci] != 1:
                    continue
                if (
                    a_arr[ci] != opposite
                    and b_arr[ci] != opposite
                    and c_arr[ci] != opposite
                ):
                    total += 1
            return total

        def flip(v):
            if assign[v]:
                start = pos_off[v]
                end = pos_off[v + 1]
                for p in range(start, end):
                    ci = pos_occ[p]
                    sc = sat_count[ci] - 1
                    sat_count[ci] = sc
                    if sc == 0:
                        add_unsat(ci)

                start = neg_off[v]
                end = neg_off[v + 1]
                for p in range(start, end):
                    ci = neg_occ[p]
                    if sat_count[ci] == 0:
                        remove_unsat(ci)
                    sat_count[ci] += 1
                assign[v] = 0
            else:
                start = neg_off[v]
                end = neg_off[v + 1]
                for p in range(start, end):
                    ci = neg_occ[p]
                    sc = sat_count[ci] - 1
                    sat_count[ci] = sc
                    if sc == 0:
                        add_unsat(ci)

                start = pos_off[v]
                end = pos_off[v + 1]
                for p in range(start, end):
                    ci = pos_occ[p]
                    if sat_count[ci] == 0:
                        remove_unsat(ci)
                    sat_count[ci] += 1
                assign[v] = 1

        max_flips = min(8000000, max(200000, m + 20 * len(unsat)))

        for _ in range(max_flips):
            if not unsat:
                return assign

            ci = unsat[rng.next() % len(unsat)]
            a = a_arr[ci]
            b = b_arr[ci]
            c = c_arr[ci]

            va = -a if a < 0 else a
            vb = -b if b < 0 else b
            vc = -c if c < 0 else c

            if (rng.next() & 7) == 0:
                if va == vb == vc:
                    choice = va
                elif va == vb:
                    choice = va if (rng.next() & 1) == 0 else vc
                elif va == vc:
                    choice = va if (rng.next() & 1) == 0 else vb
                elif vb == vc:
                    choice = vb if (rng.next() & 1) == 0 else va
                else:
                    r = rng.next() % 3
                    choice = va if r == 0 else (vb if r == 1 else vc)
                flip(choice)
                continue

            candidates = [va]
            if vb != va:
                candidates.append(vb)
            if vc != va and vc != vb:
                candidates.append(vc)

            best_v = candidates[0]
            best_b = break_count(best_v)
            if best_b != 0:
                for cand in candidates[1:]:
                    cur_b = break_count(cand)
                    if cur_b < best_b or (
                        cur_b == best_b and (rng.next() & 1) == 0
                    ):
                        best_b = cur_b
                        best_v = cand
                    if best_b == 0:
                        break

            flip(best_v)

        if len(unsat) < best_unsat:
            best_unsat = len(unsat)
            best_assign = assign[:]

    if n <= 160 and m <= 4000:
        exact = exact_dpll(n, m, a_arr, b_arr, c_arr, pos_count, neg_count, best_assign)
        if exact is not None:
            return exact
    return best_assign


def verify_stream(tmp, assign):
    tmp.seek(0)
    block_bytes = 12 * 262144
    unsat = 0
    while True:
        data = tmp.read(block_bytes)
        if not data:
            break
        chunk = array("i")
        chunk.frombytes(data)
        for i in range(0, len(chunk), 3):
            a = chunk[i]
            b = chunk[i + 1]
            c = chunk[i + 2]
            if not (lit_true(a, assign) or lit_true(b, assign) or lit_true(c, assign)):
                unsat += 1
    return unsat


def greedy_stream_sweep(tmp, assign, pos_count, neg_count, rng):
    tmp.seek(0)
    block_bytes = 12 * 262144
    unsat = 0
    changes = 0

    while True:
        data = tmp.read(block_bytes)
        if not data:
            break
        chunk = array("i")
        chunk.frombytes(data)
        for i in range(0, len(chunk), 3):
            a = chunk[i]
            b = chunk[i + 1]
            c = chunk[i + 2]
            if lit_true(a, assign) or lit_true(b, assign) or lit_true(c, assign):
                continue

            unsat += 1
            best_v = 0
            best_val = 0
            best_deg = 0
            best_bias = 0

            for lit in (a, b, c):
                v = -lit if lit < 0 else lit
                desired = 0 if lit < 0 else 1
                deg = pos_count[v] + neg_count[v]
                bias = (pos_count[v] - neg_count[v]) if desired else (neg_count[v] - pos_count[v])

                if (
                    best_v == 0
                    or deg < best_deg
                    or (deg == best_deg and bias > best_bias)
                    or (deg == best_deg and bias == best_bias and (rng.next() & 1) == 0)
                ):
                    best_v = v
                    best_val = desired
                    best_deg = deg
                    best_bias = bias

            if assign[best_v] != best_val:
                assign[best_v] = best_val
                changes += 1

    return unsat, changes


def vote_stream_sweep(tmp, assign, rng, limit_clauses=400000):
    tmp.seek(0)
    block_bytes = 12 * 262144
    votes = {}
    unsat = 0
    sampled = 0

    while True:
        data = tmp.read(block_bytes)
        if not data:
            break
        chunk = array("i")
        chunk.frombytes(data)
        for i in range(0, len(chunk), 3):
            a = chunk[i]
            b = chunk[i + 1]
            c = chunk[i + 2]
            if lit_true(a, assign) or lit_true(b, assign) or lit_true(c, assign):
                continue
            unsat += 1
            if sampled >= limit_clauses:
                continue
            sampled += 1
            for lit in (a, b, c):
                v = -lit if lit < 0 else lit
                votes[v] = votes.get(v, 0) + (-1 if lit < 0 else 1)

    changes = 0
    if votes:
        for v, score in votes.items():
            if score == 0:
                val = rng.next() & 1
            else:
                val = 1 if score > 0 else 0
            if assign[v] != val:
                assign[v] = val
                changes += 1
    return unsat, changes


def streaming_solver(n, pos_count, neg_count, tmp):
    rng = XorShift32()
    assign = build_initial_assignment(n, pos_count, neg_count, rng, 0)
    best_assign = assign[:]
    best_unsat = verify_stream(tmp, assign)
    if best_unsat == 0:
        return assign

    for restart in range(3):
        if restart:
            assign = build_initial_assignment(n, pos_count, neg_count, rng, restart)

        prev_unsat = None
        for _ in range(8):
            unsat, changes = greedy_stream_sweep(tmp, assign, pos_count, neg_count, rng)
            if unsat < best_unsat:
                best_unsat = unsat
                best_assign = assign[:]
            if unsat == 0:
                return assign
            if prev_unsat is not None and (unsat >= prev_unsat or changes == 0):
                unsat2, _ = vote_stream_sweep(tmp, assign, rng)
                if unsat2 < best_unsat:
                    best_unsat = unsat2
                    best_assign = assign[:]
                if unsat2 == 0:
                    return assign
                prev_unsat = unsat2
            else:
                prev_unsat = unsat

    return best_assign


def main():
    ints = fast_ints()
    try:
        n = next(ints)
        m = next(ints)
    except StopIteration:
        return

    moderate_mode = n <= 2000000 and m <= 5000000
    pos_count = array("I", [0]) * (n + 1)
    neg_count = array("I", [0]) * (n + 1)

    if moderate_mode:
        a_arr = array("i")
        b_arr = array("i")
        c_arr = array("i")

        for _ in range(m):
            a = next(ints)
            b = next(ints)
            c = next(ints)
            a_arr.append(a)
            b_arr.append(b)
            c_arr.append(c)
            if a > 0:
                pos_count[a] += 1
            else:
                neg_count[-a] += 1
            if b > 0:
                pos_count[b] += 1
            else:
                neg_count[-b] += 1
            if c > 0:
                pos_count[c] += 1
            else:
                neg_count[-c] += 1

        assign = walksat_solver(n, m, a_arr, b_arr, c_arr, pos_count, neg_count)
    else:
        tmp = tempfile.TemporaryFile()
        buf = array("i")
        flush_ints = 1 << 20

        for _ in range(m):
            a = next(ints)
            b = next(ints)
            c = next(ints)
            buf.append(a)
            buf.append(b)
            buf.append(c)
            if len(buf) >= flush_ints:
                buf.tofile(tmp)
                buf = array("i")
            if a > 0:
                pos_count[a] += 1
            else:
                neg_count[-a] += 1
            if b > 0:
                pos_count[b] += 1
            else:
                neg_count[-b] += 1
            if c > 0:
                pos_count[c] += 1
            else:
                neg_count[-c] += 1

        if buf:
            buf.tofile(tmp)
        assign = streaming_solver(n, pos_count, neg_count, tmp)

    out = sys.stdout.write
    out("SAT\n")
    if n == 0:
        out("\n")
        return
    chunk = []
    chunk_size = 16384
    for v in range(1, n + 1):
        chunk.append("1" if assign[v] else "0")
        if len(chunk) >= chunk_size:
            out(" ".join(chunk))
            if v != n:
                out(" ")
            chunk.clear()
    if chunk:
        out(" ".join(chunk))
    out("\n")


if __name__ == "__main__":
    main()
