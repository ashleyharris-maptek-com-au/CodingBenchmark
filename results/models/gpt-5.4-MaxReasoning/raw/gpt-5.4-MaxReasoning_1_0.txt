struct FastScanner {
    static constexpr std::size_t BUF_SIZE = 1u << 20;
    std::vector<char> buf;
    std::size_t idx = 0;
    std::size_t size = 0;

    FastScanner() : buf(BUF_SIZE) {}

    char nextChar() {
        if (idx >= size) {
            size = std::fread(buf.data(), 1, BUF_SIZE, stdin);
            idx = 0;
            if (size == 0) {
                return '\0';
            }
        }
        return buf[idx++];
    }

    void skipWhitespace() {
        char c;
        do {
            c = nextChar();
        } while (c != '\0' && static_cast<unsigned char>(c) <= ' ');
        if (c != '\0') {
            --idx;
        }
    }

    bool readUnsigned(std::size_t& out) {
        skipWhitespace();
        char c = nextChar();
        if (c == '\0') {
            return false;
        }
        std::size_t value = 0;
        while (c >= '0' && c <= '9') {
            value = value * 10 + static_cast<std::size_t>(c - '0');
            c = nextChar();
        }
        if (c != '\0') {
            --idx;
        }
        out = value;
        return true;
    }

    bool readFloat(float& out) {
        skipWhitespace();
        char c = nextChar();
        if (c == '\0') {
            return false;
        }

        bool neg = false;
        if (c == '-') {
            neg = true;
            c = nextChar();
        } else if (c == '+') {
            c = nextChar();
        }

        double value = 0.0;
        while (c >= '0' && c <= '9') {
            value = value * 10.0 + static_cast<double>(c - '0');
            c = nextChar();
        }

        if (c == '.') {
            double scale = 0.1;
            c = nextChar();
            while (c >= '0' && c <= '9') {
                value += static_cast<double>(c - '0') * scale;
                scale *= 0.1;
                c = nextChar();
            }
        }

        if (c == 'e' || c == 'E') {
            bool expNeg = false;
            c = nextChar();
            if (c == '-') {
                expNeg = true;
                c = nextChar();
            } else if (c == '+') {
                c = nextChar();
            }
            int exp = 0;
            while (c >= '0' && c <= '9') {
                exp = exp * 10 + (c - '0');
                c = nextChar();
            }
            double factor = 1.0;
            double base = 10.0;
            while (exp > 0) {
                if (exp & 1) {
                    factor *= base;
                }
                base *= base;
                exp >>= 1;
            }
            value = expNeg ? (value / factor) : (value * factor);
        }

        if (c != '\0') {
            --idx;
        }
        out = static_cast<float>(neg ? -value : value);
        return true;
    }
};

struct FastOutput {
    static constexpr std::size_t BUF_SIZE = 1u << 20;
    std::vector<char> buf;
    std::size_t idx = 0;

    FastOutput() : buf(BUF_SIZE) {}

    ~FastOutput() {
        flush();
    }

    void flush() {
        if (idx > 0) {
            std::fwrite(buf.data(), 1, idx, stdout);
            idx = 0;
        }
    }

    void pushChar(char c) {
        if (idx == BUF_SIZE) {
            flush();
        }
        buf[idx++] = c;
    }

    void writeUInt(std::uint32_t value) {
        char tmp[16];
        int len = 0;
        do {
            tmp[len++] = static_cast<char>('0' + (value % 10));
            value /= 10;
        } while (value != 0);
        while (len--) {
            pushChar(tmp[len]);
        }
    }
};

struct Point {
    float x = 0.0f;
    float y = 0.0f;
};

static inline double dist2(const Point& a, const Point& b) {
    const double dx = static_cast<double>(a.x) - static_cast<double>(b.x);
    const double dy = static_cast<double>(a.y) - static_cast<double>(b.y);
    return dx * dx + dy * dy;
}

static inline double dist(const Point& a, const Point& b) {
    return std::sqrt(dist2(a, b));
}

static inline std::uint64_t part1By1(std::uint32_t x) {
    std::uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8)) & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2)) & 0x3333333333333333ULL;
    v = (v | (v << 1)) & 0x5555555555555555ULL;
    return v;
}

static inline std::uint64_t morton2D(std::uint32_t x, std::uint32_t y) {
    return part1By1(x) | (part1By1(y) << 1);
}

static void rotateToZeroFront(std::vector<std::uint32_t>& tour) {
    auto it = std::find(tour.begin(), tour.end(), 0u);
    if (it != tour.end()) {
        std::rotate(tour.begin(), it, tour.end());
    }
}

static std::vector<std::uint32_t> solveExactHeldKarp(const std::vector<Point>& pts) {
    const int n = static_cast<int>(pts.size());
    if (n <= 1) {
        return n == 1 ? std::vector<std::uint32_t>{0u} : std::vector<std::uint32_t>{};
    }

    const int m = n - 1;
    const std::size_t states = std::size_t{1} << m;
    const double INF = std::numeric_limits<double>::infinity();

    std::vector<double> dm(static_cast<std::size_t>(n) * static_cast<std::size_t>(n), 0.0);
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            dm[static_cast<std::size_t>(i) * n + j] = dist(pts[i], pts[j]);
        }
    }

    std::vector<double> dp(states * static_cast<std::size_t>(m), INF);
    std::vector<std::int16_t> parent(states * static_cast<std::size_t>(m), static_cast<std::int16_t>(-1));

    for (int j = 0; j < m; ++j) {
        dp[(std::size_t{1} << j) * m + j] = dm[j + 1];
    }

    for (std::size_t mask = 1; mask < states; ++mask) {
        for (int j = 0; j < m; ++j) {
            if ((mask & (std::size_t{1} << j)) == 0) {
                continue;
            }
            const std::size_t prevMask = mask ^ (std::size_t{1} << j);
            if (prevMask == 0) {
                continue;
            }

            double best = INF;
            std::int16_t bestPrev = -1;
            for (int k = 0; k < m; ++k) {
                if ((prevMask & (std::size_t{1} << k)) == 0) {
                    continue;
                }
                const double cand = dp[prevMask * m + k] + dm[static_cast<std::size_t>(k + 1) * n + (j + 1)];
                if (cand < best) {
                    best = cand;
                    bestPrev = static_cast<std::int16_t>(k);
                }
            }
            dp[mask * m + j] = best;
            parent[mask * m + j] = bestPrev;
        }
    }

    const std::size_t fullMask = states - 1;
    double bestCost = INF;
    int last = -1;
    for (int j = 0; j < m; ++j) {
        const double cost = dp[fullMask * m + j] + dm[static_cast<std::size_t>(j + 1) * n];
        if (cost < bestCost) {
            bestCost = cost;
            last = j;
        }
    }

    std::vector<std::uint32_t> tail;
    tail.reserve(static_cast<std::size_t>(m));
    std::size_t mask = fullMask;
    while (last >= 0) {
        tail.push_back(static_cast<std::uint32_t>(last + 1));
        const std::int16_t prev = parent[mask * m + last];
        mask ^= (std::size_t{1} << last);
        last = prev;
    }
    std::reverse(tail.begin(), tail.end());

    std::vector<std::uint32_t> tour;
    tour.reserve(static_cast<std::size_t>(n));
    tour.push_back(0u);
    tour.insert(tour.end(), tail.begin(), tail.end());
    return tour;
}

static std::vector<std::uint32_t> solveGreedyNearest(const std::vector<Point>& pts) {
    const std::size_t n = pts.size();
    std::vector<std::uint32_t> tour;
    tour.reserve(n);
    if (n == 0) {
        return tour;
    }

    std::vector<std::uint8_t> used(n, 0);
    std::uint32_t current = 0;
    used[0] = 1;
    tour.push_back(0);

    for (std::size_t step = 1; step < n; ++step) {
        std::uint32_t best = 0;
        double bestD = std::numeric_limits<double>::infinity();
        for (std::uint32_t i = 1; i < static_cast<std::uint32_t>(n); ++i) {
            if (used[i]) {
                continue;
            }
            const double d = dist2(pts[current], pts[i]);
            if (d < bestD) {
                bestD = d;
                best = i;
            }
        }
        used[best] = 1;
        tour.push_back(best);
        current = best;
    }
    return tour;
}

static bool twoOptPass(std::vector<std::uint32_t>& tour, const std::vector<Point>& pts, std::size_t windowLimit) {
    const std::size_t n = tour.size();
    if (n < 4) {
        return false;
    }

    bool improved = false;
    const bool bounded = windowLimit != 0;
    for (std::size_t i = 0; i + 1 < n; ++i) {
        const std::uint32_t a = tour[i];
        const std::uint32_t b = tour[(i + 1) % n];
        const std::size_t maxJ = bounded ? std::min(n - 1, i + windowLimit) : (n - 1);
        for (std::size_t j = i + 2; j <= maxJ; ++j) {
            if (i == 0 && j == n - 1) {
                continue;
            }
            const std::uint32_t c = tour[j];
            const std::uint32_t d = tour[(j + 1) % n];
            const double oldLen = dist(pts[a], pts[b]) + dist(pts[c], pts[d]);
            const double newLen = dist(pts[a], pts[c]) + dist(pts[b], pts[d]);
            if (newLen + 1e-12 < oldLen) {
                std::reverse(tour.begin() + static_cast<std::ptrdiff_t>(i + 1),
                             tour.begin() + static_cast<std::ptrdiff_t>(j + 1));
                improved = true;
            }
        }
    }
    return improved;
}

static void improveTwoOpt(std::vector<std::uint32_t>& tour, const std::vector<Point>& pts, std::size_t windowLimit, int maxPasses) {
    for (int pass = 0; pass < maxPasses; ++pass) {
        if (!twoOptPass(tour, pts, windowLimit)) {
            break;
        }
    }
    rotateToZeroFront(tour);
}

static void radixSortByKey(std::vector<std::uint64_t>& keys, std::vector<std::uint32_t>& order) {
    const std::size_t n = keys.size();
    if (n <= 1) {
        return;
    }

    std::vector<std::uint64_t> tmpKeys(n);
    std::vector<std::uint32_t> tmpOrder(n);
    constexpr std::size_t RADIX = 1u << 16;
    std::vector<std::size_t> count(RADIX);

    for (unsigned shift = 0; shift < 64; shift += 16) {
        std::fill(count.begin(), count.end(), 0);
        for (std::size_t i = 0; i < n; ++i) {
            ++count[(keys[i] >> shift) & 0xFFFFu];
        }
        std::size_t sum = 0;
        for (std::size_t i = 0; i < RADIX; ++i) {
            const std::size_t c = count[i];
            count[i] = sum;
            sum += c;
        }
        for (std::size_t i = 0; i < n; ++i) {
            const std::size_t bucket = (keys[i] >> shift) & 0xFFFFu;
            const std::size_t pos = count[bucket]++;
            tmpKeys[pos] = keys[i];
            tmpOrder[pos] = order[i];
        }
        keys.swap(tmpKeys);
        order.swap(tmpOrder);
    }
}

static std::vector<std::uint32_t> solveMorton(const std::vector<Point>& pts, float minX, float maxX, float minY, float maxY) {
    const std::size_t n = pts.size();
    std::vector<std::uint64_t> keys(n);
    std::vector<std::uint32_t> order(n);

    const double rangeX = static_cast<double>(maxX) - static_cast<double>(minX);
    const double rangeY = static_cast<double>(maxY) - static_cast<double>(minY);
    const double scaleX = (rangeX > 0.0) ? (4294967295.0 / rangeX) : 0.0;
    const double scaleY = (rangeY > 0.0) ? (4294967295.0 / rangeY) : 0.0;

    for (std::uint32_t i = 0; i < static_cast<std::uint32_t>(n); ++i) {
        const auto& p = pts[i];
        const std::uint32_t qx = (rangeX > 0.0)
            ? static_cast<std::uint32_t>((static_cast<double>(p.x) - static_cast<double>(minX)) * scaleX + 0.5)
            : 0u;
        const std::uint32_t qy = (rangeY > 0.0)
            ? static_cast<std::uint32_t>((static_cast<double>(p.y) - static_cast<double>(minY)) * scaleY + 0.5)
            : 0u;
        keys[i] = morton2D(qx, qy);
        order[i] = i;
    }

    radixSortByKey(keys, order);
    rotateToZeroFront(order);
    return order;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    FastScanner scanner;
    std::size_t n = 0;
    if (!scanner.readUnsigned(n)) {
        return 0;
    }

    std::vector<Point> pts(n);
    float minX = std::numeric_limits<float>::infinity();
    float maxX = -std::numeric_limits<float>::infinity();
    float minY = std::numeric_limits<float>::infinity();
    float maxY = -std::numeric_limits<float>::infinity();

    for (std::size_t i = 0; i < n; ++i) {
        float x = 0.0f;
        float y = 0.0f;
        scanner.readFloat(x);
        scanner.readFloat(y);
        pts[i] = Point{x, y};
        if (x < minX) minX = x;
        if (x > maxX) maxX = x;
        if (y < minY) minY = y;
        if (y > maxY) maxY = y;
    }

    std::vector<std::uint32_t> tour;
    if (n == 0) {
        return 0;
    } else if (n <= 20) {
        tour = solveExactHeldKarp(pts);
    } else if (n <= 5000) {
        tour = solveGreedyNearest(pts);
        improveTwoOpt(tour, pts, 0, 6);
    } else {
        tour = solveMorton(pts, minX, maxX, minY, maxY);
        if (n <= 20000) {
            improveTwoOpt(tour, pts, 64, 4);
        } else if (n <= 100000) {
            improveTwoOpt(tour, pts, 32, 3);
        } else if (n <= 200000) {
            improveTwoOpt(tour, pts, 16, 2);
        } else if (n <= 2000000) {
            improveTwoOpt(tour, pts, 8, 1);
        }
    }

    rotateToZeroFront(tour);

    FastOutput out;
    for (std::size_t i = 0; i < tour.size(); ++i) {
        if (i) {
            out.pushChar(' ');
        }
        out.writeUInt(tour[i]);
    }
    out.pushChar('\n');
    out.flush();
    return 0;
}
