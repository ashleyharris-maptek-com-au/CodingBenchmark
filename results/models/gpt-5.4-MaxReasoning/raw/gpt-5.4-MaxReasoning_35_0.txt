use std::cmp::Reverse;
use std::collections::{BinaryHeap, HashMap, VecDeque};
use std::io::{self, Read};
use std::time::{Duration, Instant};

const INF: usize = usize::MAX / 4;
const EXACT_THRESHOLD: usize = 32;

fn main() {
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();
    let mut scanner = Scanner::new(&input);

    let n = match scanner.next_usize() {
        Some(v) => v,
        None => {
            println!("0");
            println!();
            return;
        }
    };
    let m = scanner.next_usize().unwrap_or(0);

    if n == 0 {
        println!("0");
        println!();
        return;
    }

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u = match scanner.next_usize() {
            Some(v) => v,
            None => break,
        };
        let v = match scanner.next_usize() {
            Some(v) => v,
            None => break,
        };
        if u >= n || v >= n || u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    for neighbors in &mut adj {
        neighbors.sort_unstable();
        neighbors.dedup();
    }

    let components = connected_components(&adj);
    let deadline = Instant::now() + Duration::from_secs(290);

    let mut local_of = vec![usize::MAX; n];
    let mut answer = Vec::new();

    for component in &components {
        let picked = solve_component(component, &adj, &mut local_of, deadline);
        answer.extend(picked);
    }

    answer.sort_unstable();
    answer.dedup();

    if !is_valid_solution(n, &adj, &answer) {
        answer = (0..n).collect();
    }

    println!("{}", answer.len());
    if answer.is_empty() {
        println!();
    } else {
        for (i, v) in answer.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{v}");
        }
        println!();
    }
}

fn connected_components(adj: &[Vec<usize>]) -> Vec<Vec<usize>> {
    let n = adj.len();
    let mut visited = vec![false; n];
    let mut components = Vec::new();

    for start in 0..n {
        if visited[start] {
            continue;
        }
        let mut stack = vec![start];
        visited[start] = true;
        let mut component = Vec::new();

        while let Some(u) = stack.pop() {
            component.push(u);
            for &v in &adj[u] {
                if !visited[v] {
                    visited[v] = true;
                    stack.push(v);
                }
            }
        }

        components.push(component);
    }

    components
}

fn solve_component(
    vertices: &[usize],
    global_adj: &[Vec<usize>],
    local_of: &mut [usize],
    deadline: Instant,
) -> Vec<usize> {
    let size = vertices.len();
    if size == 1 {
        return vec![vertices[0]];
    }

    for (i, &g) in vertices.iter().enumerate() {
        local_of[g] = i;
    }

    let mut neighbors = vec![Vec::<usize>::new(); size];
    let mut closed = vec![Vec::<usize>::new(); size];
    let mut edge_count = 0usize;

    for (i, &g) in vertices.iter().enumerate() {
        for &h in &global_adj[g] {
            let j = local_of[h];
            if j != usize::MAX {
                neighbors[i].push(j);
                if i < j {
                    edge_count += 1;
                }
            }
        }
        closed[i].push(i);
        closed[i].extend_from_slice(&neighbors[i]);
    }

    let local_solution = if edge_count + 1 == size {
        solve_tree_component(&neighbors)
    } else if size <= EXACT_THRESHOLD && Instant::now() < deadline {
        solve_exact_component(&closed)
    } else {
        solve_greedy_component(&neighbors, &closed, deadline)
    };

    let mut global_solution = Vec::with_capacity(local_solution.len());
    for local in local_solution {
        global_solution.push(vertices[local]);
    }

    for &g in vertices {
        local_of[g] = usize::MAX;
    }

    global_solution
}

fn solve_tree_component(neighbors: &[Vec<usize>]) -> Vec<usize> {
    let n = neighbors.len();
    if n == 0 {
        return Vec::new();
    }
    if n == 1 {
        return vec![0];
    }

    let root = 0usize;
    let mut parent = vec![usize::MAX; n];
    let mut order = Vec::with_capacity(n);
    let mut children = vec![Vec::<usize>::new(); n];
    let mut stack = vec![root];
    parent[root] = root;

    while let Some(u) = stack.pop() {
        order.push(u);
        for &v in &neighbors[u] {
            if v == parent[u] {
                continue;
            }
            parent[v] = u;
            children[u].push(v);
            stack.push(v);
        }
    }

    let mut dp0 = vec![INF; n];
    let mut dp1 = vec![INF; n];
    let mut dp2 = vec![INF; n];

    for &u in order.iter().rev() {
        if children[u].is_empty() {
            dp0[u] = 1;
            dp1[u] = INF;
            dp2[u] = 0;
            continue;
        }

        let mut sum0 = 1usize;
        let mut sum1 = 0usize;
        let mut sum2 = 0usize;
        let mut has_selected_child = false;
        let mut force_delta = INF;

        for &v in &children[u] {
            sum0 = add_cap(sum0, dp0[v].min(dp1[v]).min(dp2[v]), INF);
            sum2 = add_cap(sum2, dp1[v], INF);

            let best = dp0[v].min(dp1[v]);
            sum1 = add_cap(sum1, best, INF);
            if dp0[v] <= dp1[v] {
                has_selected_child = true;
            } else {
                force_delta = force_delta.min(dp0[v] - dp1[v]);
            }
        }

        dp0[u] = sum0;
        dp2[u] = sum2;
        dp1[u] = if has_selected_child {
            sum1
        } else if force_delta >= INF / 2 {
            INF
        } else {
            add_cap(sum1, force_delta, INF)
        };
    }

    let root_state = if dp0[root] <= dp1[root] { 0 } else { 1 };
    let mut chosen = Vec::new();
    let mut rebuild = vec![(root, root_state)];

    while let Some((u, state)) = rebuild.pop() {
        if state == 0 {
            chosen.push(u);
        }

        match state {
            0 => {
                for &v in children[u].iter().rev() {
                    let child_state = argmin3(dp0[v], dp1[v], dp2[v]);
                    rebuild.push((v, child_state));
                }
            }
            1 => {
                let mut child_states = Vec::with_capacity(children[u].len());
                let mut has_selected_child = false;
                let mut best_delta = (INF, 0usize);

                for (idx, &v) in children[u].iter().enumerate() {
                    if dp0[v] <= dp1[v] {
                        child_states.push(0usize);
                        has_selected_child = true;
                    } else {
                        child_states.push(1usize);
                        let delta = dp0[v] - dp1[v];
                        if delta < best_delta.0 {
                            best_delta = (delta, idx);
                        }
                    }
                }

                if !has_selected_child {
                    child_states[best_delta.1] = 0;
                }

                for (&v, &child_state) in children[u].iter().zip(child_states.iter()).rev() {
                    rebuild.push((v, child_state));
                }
            }
            2 => {
                for &v in children[u].iter().rev() {
                    rebuild.push((v, 1));
                }
            }
            _ => unreachable!(),
        }
    }

    chosen.sort_unstable();
    chosen
}

fn solve_exact_component(closed: &[Vec<usize>]) -> Vec<usize> {
    let n = closed.len();
    let mut closed_masks = vec![0u64; n];
    let mut order = (0..n).collect::<Vec<_>>();

    for v in 0..n {
        let mut mask = 0u64;
        for &u in &closed[v] {
            mask |= 1u64 << u;
        }
        closed_masks[v] = mask;
    }

    order.sort_unstable_by_key(|&v| closed_masks[v].count_ones());

    let all = if n == 64 { u64::MAX } else { (1u64 << n) - 1 };
    let mut best = greedy_upper_bound_masks(&closed_masks, all);
    let mut current = Vec::new();
    let mut memo = HashMap::<u64, usize>::new();

    exact_dfs(
        0,
        all,
        &closed_masks,
        &order,
        &mut current,
        &mut best,
        &mut memo,
    );

    best.sort_unstable();
    best
}

fn exact_dfs(
    dominated: u64,
    all: u64,
    closed_masks: &[u64],
    order: &[usize],
    current: &mut Vec<usize>,
    best: &mut Vec<usize>,
    memo: &mut HashMap<u64, usize>,
) {
    if current.len() >= best.len() {
        return;
    }
    if dominated == all {
        *best = current.clone();
        return;
    }

    if let Some(&seen) = memo.get(&dominated) {
        if seen <= current.len() {
            return;
        }
    }
    memo.insert(dominated, current.len());

    let remaining = all & !dominated;
    let lower = lower_bound(remaining, closed_masks, order);
    if current.len() + lower >= best.len() {
        return;
    }

    let pivot = choose_pivot(remaining, closed_masks);
    let mut candidates = bits_of(closed_masks[pivot]);
    candidates.sort_unstable_by_key(|&v| {
        Reverse((closed_masks[v] & remaining).count_ones() as usize)
    });

    for v in candidates {
        current.push(v);
        exact_dfs(
            dominated | closed_masks[v],
            all,
            closed_masks,
            order,
            current,
            best,
            memo,
        );
        current.pop();
    }
}

fn choose_pivot(remaining: u64, closed_masks: &[u64]) -> usize {
    let mut bits = remaining;
    let first = bits.trailing_zeros() as usize;
    let mut best = first;
    let mut best_branch = closed_masks[first].count_ones();
    let mut best_cover = (closed_masks[first] & remaining).count_ones();
    bits &= bits - 1;

    while bits != 0 {
        let v = bits.trailing_zeros() as usize;
        let branch = closed_masks[v].count_ones();
        let cover = (closed_masks[v] & remaining).count_ones();
        if branch < best_branch || (branch == best_branch && cover > best_cover) {
            best = v;
            best_branch = branch;
            best_cover = cover;
        }
        bits &= bits - 1;
    }

    best
}

fn lower_bound(remaining: u64, closed_masks: &[u64], order: &[usize]) -> usize {
    let mut max_cover = 1usize;
    for &mask in closed_masks {
        let cover = (mask & remaining).count_ones() as usize;
        if cover > max_cover {
            max_cover = cover;
        }
    }

    let uncovered = remaining.count_ones() as usize;
    let cover_bound = uncovered.div_ceil(max_cover);

    let mut used = 0u64;
    let mut packing = 0usize;
    for &v in order {
        if (remaining >> v) & 1 == 0 {
            continue;
        }
        let mask = closed_masks[v];
        if used & mask == 0 {
            used |= mask;
            packing += 1;
        }
    }

    cover_bound.max(packing)
}

fn greedy_upper_bound_masks(closed_masks: &[u64], all: u64) -> Vec<usize> {
    let n = closed_masks.len();
    let mut dominated = 0u64;
    let mut selected = vec![false; n];
    let mut solution = Vec::new();

    while dominated != all {
        let remaining = all & !dominated;
        let mut best_v = 0usize;
        let mut best_gain = 0usize;
        let mut best_size = 0usize;

        for v in 0..n {
            if selected[v] {
                continue;
            }
            let gain = (closed_masks[v] & remaining).count_ones() as usize;
            if gain > best_gain {
                best_gain = gain;
                best_size = closed_masks[v].count_ones() as usize;
                best_v = v;
            } else if gain == best_gain {
                let size = closed_masks[v].count_ones() as usize;
                if size > best_size {
                    best_size = size;
                    best_v = v;
                }
            }
        }

        selected[best_v] = true;
        dominated |= closed_masks[best_v];
        solution.push(best_v);
    }

    prune_mask_solution(closed_masks, &mut solution);
    solution
}

fn prune_mask_solution(closed_masks: &[u64], solution: &mut Vec<usize>) {
    let n = closed_masks.len();
    let mut in_solution = vec![false; n];
    for &v in solution.iter() {
        in_solution[v] = true;
    }

    let mut counts = vec![0usize; n];
    for &v in solution.iter() {
        let mut mask = closed_masks[v];
        while mask != 0 {
            let u = mask.trailing_zeros() as usize;
            counts[u] += 1;
            mask &= mask - 1;
        }
    }

    let mut changed = true;
    while changed {
        changed = false;
        for idx in 0..solution.len() {
            let v = solution[idx];
            if !in_solution[v] {
                continue;
            }
            let mut removable = true;
            let mut mask = closed_masks[v];
            while mask != 0 {
                let u = mask.trailing_zeros() as usize;
                if counts[u] <= 1 {
                    removable = false;
                    break;
                }
                mask &= mask - 1;
            }

            if removable {
                in_solution[v] = false;
                let mut mask = closed_masks[v];
                while mask != 0 {
                    let u = mask.trailing_zeros() as usize;
                    counts[u] -= 1;
                    mask &= mask - 1;
                }
                changed = true;
            }
        }
    }

    solution.retain(|&v| in_solution[v]);
}

fn solve_greedy_component(
    neighbors: &[Vec<usize>],
    closed: &[Vec<usize>],
    deadline: Instant,
) -> Vec<usize> {
    let n = closed.len();
    let rounds = if n <= 128 {
        8
    } else if n <= 1024 {
        6
    } else if n <= 8192 {
        4
    } else if n <= 65536 {
        2
    } else {
        1
    };

    let degrees = neighbors.iter().map(|v| v.len()).collect::<Vec<_>>();
    let mut best = construct_greedy(closed, &degrees, None);

    let mut seed = 0x9E37_79B9_7F4A_7C15u64
        ^ ((n as u64) << 32)
        ^ (closed.iter().map(|v| v.len() as u64).sum::<u64>() << 1);
    let mut rng = XorShift64::new(seed | 1);

    for _ in 1..rounds {
        if Instant::now() >= deadline {
            break;
        }
        seed ^= rng.next_u64().rotate_left(7);
        let candidate = construct_greedy(closed, &degrees, Some(seed | 1));
        if candidate.len() < best.len() {
            best = candidate;
        }
    }

    best
}

fn construct_greedy(
    closed: &[Vec<usize>],
    degrees: &[usize],
    random_seed: Option<u64>,
) -> Vec<usize> {
    let n = closed.len();
    let mut selected = vec![false; n];
    let mut dominated = vec![false; n];
    let mut coverage = closed.iter().map(|v| v.len()).collect::<Vec<_>>();
    let mut heap = BinaryHeap::<(usize, u64, usize)>::new();
    let mut tie = vec![0u64; n];

    match random_seed {
        Some(seed) => {
            let mut rng = XorShift64::new(seed);
            for t in &mut tie {
                *t = rng.next_u64();
            }
        }
        None => {
            for v in 0..n {
                tie[v] = ((degrees[v] as u64) << 32) ^ (u32::MAX as u64 - v as u64);
            }
        }
    }

    for v in 0..n {
        heap.push((coverage[v], tie[v], v));
    }

    let mut solution = Vec::new();
    let mut remaining = n;

    let forced = forced_leaf_supports(closed, degrees);
    for v in forced {
        if selected[v] {
            continue;
        }
        select_vertex(
            v,
            closed,
            &mut selected,
            &mut dominated,
            &mut coverage,
            &mut remaining,
            &mut heap,
            &tie,
        );
        solution.push(v);
    }

    while remaining > 0 {
        let mut chosen = None;
        while let Some((score, _, v)) = heap.pop() {
            if !selected[v] && coverage[v] == score && score > 0 {
                chosen = Some(v);
                break;
            }
        }

        let v = chosen.unwrap_or_else(|| {
            (0..n)
                .find(|&u| !dominated[u])
                .expect("undominated vertex should exist")
        });

        if !selected[v] {
            select_vertex(
                v,
                closed,
                &mut selected,
                &mut dominated,
                &mut coverage,
                &mut remaining,
                &mut heap,
                &tie,
            );
            solution.push(v);
        }
    }

    prune_selected(closed, &mut selected);
    let mut answer = Vec::new();
    for (v, &keep) in selected.iter().enumerate() {
        if keep {
            answer.push(v);
        }
    }
    answer
}

fn forced_leaf_supports(closed: &[Vec<usize>], degrees: &[usize]) -> Vec<usize> {
    let n = closed.len();
    let mut forced = vec![false; n];
    for u in 0..n {
        if degrees[u] != 1 {
            continue;
        }
        let mut support = u;
        for &v in &closed[u] {
            if v != u {
                support = v;
                break;
            }
        }
        if support != u && degrees[support] > 1 {
            forced[support] = true;
        }
    }

    let mut out = Vec::new();
    for v in 0..n {
        if forced[v] {
            out.push(v);
        }
    }
    out
}

fn select_vertex(
    v: usize,
    closed: &[Vec<usize>],
    selected: &mut [bool],
    dominated: &mut [bool],
    coverage: &mut [usize],
    remaining: &mut usize,
    heap: &mut BinaryHeap<(usize, u64, usize)>,
    tie: &[u64],
) {
    if selected[v] {
        return;
    }
    selected[v] = true;

    for &u in &closed[v] {
        if dominated[u] {
            continue;
        }
        dominated[u] = true;
        *remaining -= 1;
        for &w in &closed[u] {
            if !selected[w] && coverage[w] > 0 {
                coverage[w] -= 1;
                heap.push((coverage[w], tie[w], w));
            }
        }
    }
}

fn prune_selected(closed: &[Vec<usize>], selected: &mut [bool]) {
    let n = closed.len();
    let mut domination_count = vec![0usize; n];
    let mut queue = VecDeque::new();

    for v in 0..n {
        if selected[v] {
            queue.push_back(v);
            for &u in &closed[v] {
                domination_count[u] += 1;
            }
        }
    }

    while let Some(v) = queue.pop_front() {
        if !selected[v] {
            continue;
        }
        if closed[v].iter().all(|&u| domination_count[u] >= 2) {
            selected[v] = false;
            for &u in &closed[v] {
                domination_count[u] -= 1;
                for &w in &closed[u] {
                    if selected[w] {
                        queue.push_back(w);
                    }
                }
            }
        }
    }
}

fn is_valid_solution(n: usize, adj: &[Vec<usize>], solution: &[usize]) -> bool {
    let mut dominated = vec![false; n];
    for &v in solution {
        if v >= n {
            return false;
        }
        dominated[v] = true;
        for &u in &adj[v] {
            dominated[u] = true;
        }
    }
    dominated.into_iter().all(|x| x)
}

fn bits_of(mut mask: u64) -> Vec<usize> {
    let mut out = Vec::new();
    while mask != 0 {
        let bit = mask.trailing_zeros() as usize;
        out.push(bit);
        mask &= mask - 1;
    }
    out
}

fn argmin3(a: usize, b: usize, c: usize) -> usize {
    if a <= b && a <= c {
        0
    } else if b <= c {
        1
    } else {
        2
    }
}

fn add_cap(a: usize, b: usize, cap: usize) -> usize {
    let sum = a.saturating_add(b);
    if sum > cap { cap } else { sum }
}

struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        Self { state: seed.max(1) }
    }

    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 7;
        x ^= x >> 9;
        x ^= x << 8;
        self.state = x;
        x
    }
}

struct Scanner<'a> {
    data: &'a [u8],
    index: usize,
}

impl<'a> Scanner<'a> {
    fn new(data: &'a [u8]) -> Self {
        Self { data, index: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.index < self.data.len() && self.data[self.index].is_ascii_whitespace() {
            self.index += 1;
        }
        if self.index >= self.data.len() {
            return None;
        }
        let mut value = 0usize;
        while self.index < self.data.len() && !self.data[self.index].is_ascii_whitespace() {
            value = value * 10 + (self.data[self.index] - b'0') as usize;
            self.index += 1;
        }
        Some(value)
    }
}
