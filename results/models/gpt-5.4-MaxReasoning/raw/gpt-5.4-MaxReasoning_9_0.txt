using System;
using System.IO;
using System.Runtime.CompilerServices;

public static class Program
{
    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());

        int n = scanner.NextInt();
        int stockLength = scanner.NextInt();

        if (n <= 0)
        {
            using var emptyWriter = new FastWriter(Console.OpenStandardOutput());
            emptyWriter.WriteInt(0);
            emptyWriter.WriteLine();
            emptyWriter.Flush();
            return;
        }

        var cuts = new int[n];
        var next = new int[n];
        var order = new ulong[n];

        for (int i = 0; i < n; i++)
        {
            int length = scanner.NextInt();
            if (length < 0 || length > stockLength)
            {
                throw new InvalidOperationException("Cut length must be between 0 and stock_length.");
            }

            cuts[i] = length;
            order[i] = ((ulong)(uint.MaxValue - (uint)length) << 32) | (uint)i;
        }

        Array.Sort(order);

        var binHead = new int[n];
        var binTail = new int[n];

        var treap = new Treap(n);
        int binCount = 0;

        for (int i = 0; i < n; i++)
        {
            int cutIndex = (int)(order[i] & 0xFFFFFFFFu);
            int length = cuts[cutIndex];

            int node = treap.LowerBound(length);
            int binId;
            int remaining;

            if (node == 0)
            {
                binId = binCount++;
                binHead[binId] = cutIndex;
                binTail[binId] = cutIndex;
                next[cutIndex] = -1;

                remaining = stockLength - length;
                if (remaining > 0)
                {
                    treap.Insert(remaining, binId);
                }
            }
            else
            {
                int oldRemaining = treap.KeyRemaining(node);
                binId = treap.KeyBinId(node);
                treap.Erase(oldRemaining, binId);

                next[binTail[binId]] = cutIndex;
                binTail[binId] = cutIndex;
                next[cutIndex] = -1;

                remaining = oldRemaining - length;
                if (remaining > 0)
                {
                    treap.Insert(remaining, binId);
                }
            }
        }

        using var writer = new FastWriter(Console.OpenStandardOutput());
        writer.WriteInt(binCount);
        writer.WriteLine();

        for (int bin = 0; bin < binCount; bin++)
        {
            int current = binHead[bin];
            bool first = true;

            while (current != -1)
            {
                if (!first)
                {
                    writer.WriteSpace();
                }

                writer.WriteInt(current);
                first = false;
                current = next[current];
            }

            writer.WriteLine();
        }

        writer.Flush();
    }

    private sealed class Treap
    {
        private readonly int[] _left;
        private readonly int[] _right;
        private readonly int[] _priority;
        private readonly int[] _remaining;
        private readonly int[] _binId;
        private int _root;
        private int _count;
        private uint _rng;

        public Treap(int capacity)
        {
            _left = new int[capacity + 1];
            _right = new int[capacity + 1];
            _priority = new int[capacity + 1];
            _remaining = new int[capacity + 1];
            _binId = new int[capacity + 1];
            _rng = 2463534242u;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int KeyRemaining(int node) => _remaining[node];

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int KeyBinId(int node) => _binId[node];

        public int LowerBound(int targetRemaining)
        {
            int node = _root;
            int best = 0;

            while (node != 0)
            {
                if (_remaining[node] >= targetRemaining)
                {
                    best = node;
                    node = _left[node];
                }
                else
                {
                    node = _right[node];
                }
            }

            return best;
        }

        public void Insert(int remaining, int binId)
        {
            int node = ++_count;
            _left[node] = 0;
            _right[node] = 0;
            _remaining[node] = remaining;
            _binId[node] = binId;
            _priority[node] = (int)NextRandom();
            _root = InsertNode(_root, node);
        }

        public void Erase(int remaining, int binId)
        {
            _root = Erase(_root, remaining, binId);
        }

        private int InsertNode(int root, int node)
        {
            if (root == 0)
            {
                return node;
            }

            if (_priority[node] < _priority[root])
            {
                Split(root, _remaining[node], _binId[node], out int left, out int right);
                _left[node] = left;
                _right[node] = right;
                return node;
            }

            if (Compare(_remaining[node], _binId[node], _remaining[root], _binId[root]) < 0)
            {
                _left[root] = InsertNode(_left[root], node);
            }
            else
            {
                _right[root] = InsertNode(_right[root], node);
            }

            return root;
        }

        private int Erase(int root, int remaining, int binId)
        {
            if (root == 0)
            {
                return 0;
            }

            int cmp = Compare(remaining, binId, _remaining[root], _binId[root]);
            if (cmp == 0)
            {
                return Merge(_left[root], _right[root]);
            }

            if (cmp < 0)
            {
                _left[root] = Erase(_left[root], remaining, binId);
            }
            else
            {
                _right[root] = Erase(_right[root], remaining, binId);
            }

            return root;
        }

        private void Split(int root, int remaining, int binId, out int left, out int right)
        {
            if (root == 0)
            {
                left = 0;
                right = 0;
                return;
            }

            if (Compare(_remaining[root], _binId[root], remaining, binId) < 0)
            {
                Split(_right[root], remaining, binId, out int splitLeft, out right);
                _right[root] = splitLeft;
                left = root;
            }
            else
            {
                Split(_left[root], remaining, binId, out left, out int splitRight);
                _left[root] = splitRight;
                right = root;
            }
        }

        private int Merge(int left, int right)
        {
            if (left == 0)
            {
                return right;
            }

            if (right == 0)
            {
                return left;
            }

            if (_priority[left] < _priority[right])
            {
                _right[left] = Merge(_right[left], right);
                return left;
            }

            _left[right] = Merge(left, _left[right]);
            return right;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private static int Compare(int remainingA, int binIdA, int remainingB, int binIdB)
        {
            if (remainingA != remainingB)
            {
                return remainingA < remainingB ? -1 : 1;
            }

            if (binIdA == binIdB)
            {
                return 0;
            }

            return binIdA < binIdB ? -1 : 1;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private uint NextRandom()
        {
            uint x = _rng;
            x ^= x << 13;
            x ^= x >> 17;
            x ^= x << 5;
            _rng = x;
            return x;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _length;
        private int _position;

        public FastScanner(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            }
            while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int Read()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length == 0)
                {
                    return -1;
                }
            }

            return _buffer[_position++];
        }
    }

    private sealed class FastWriter : IDisposable
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _position;

        public FastWriter(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        public void WriteSpace() => WriteByte((byte)' ');

        public void WriteLine() => WriteByte((byte)'\n');

        public void WriteInt(int value)
        {
            if (value == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            if (value < 0)
            {
                WriteByte((byte)'-');
                value = -value;
            }

            Span<byte> digits = stackalloc byte[11];
            int count = 0;
            uint current = (uint)value;

            while (current > 0)
            {
                digits[count++] = (byte)('0' + (current % 10));
                current /= 10;
            }

            while (count > 0)
            {
                WriteByte(digits[--count]);
            }
        }

        public void Flush()
        {
            if (_position > 0)
            {
                _stream.Write(_buffer, 0, _position);
                _position = 0;
            }
        }

        public void Dispose()
        {
            Flush();
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private void WriteByte(byte value)
        {
            if (_position == _buffer.Length)
            {
                Flush();
            }

            _buffer[_position++] = value;
        }
    }
}
