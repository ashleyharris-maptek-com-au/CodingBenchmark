OpCapability Shader
OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main Invocations 1
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
%void = OpTypeVoid
%fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0 = OpConstant %float 0
%float_0_08 = OpConstant %float 0.08
%float_inv3 = OpConstant %float 0.333333343
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_3 = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_Input_arr_gl_PerVertex_in_3 = OpTypePointer Input %arr_gl_PerVertex_in_3
%ptr_Input_v4float = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_3 Input
%arr_v3float_3 = OpTypeArray %v3float %uint_3
%ptr_Input_arr_v3float_3 = OpTypePointer Input %arr_v3float_3
%ptr_Input_v3float = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_Input_arr_v3float_3 Input
%normal = OpVariable %ptr_Input_arr_v3float_3 Input
%color = OpVariable %ptr_Input_arr_v3float_3 Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %ptr_Output_v3float Output
%main = OpFunction %void None %fn
%label = OpLabel
%p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%p0_v4 = OpLoad %v4float %p0_ptr
%p1_v4 = OpLoad %v4float %p1_ptr
%p2_v4 = OpLoad %v4float %p2_ptr
%p0 = OpVectorShuffle %v3float %p0_v4 %p0_v4 0 1 2
%p1 = OpVectorShuffle %v3float %p1_v4 %p1_v4 0 1 2
%p2 = OpVectorShuffle %v3float %p2_v4 %p2_v4 0 1 2
%sum01 = OpFAdd %v3float %p0 %p1
%sum012 = OpFAdd %v3float %sum01 %p2
%center = OpVectorTimesScalar %v3float %sum012 %float_inv3
%e1 = OpFSub %v3float %p1 %p0
%e2 = OpFSub %v3float %p2 %p0
%cross = OpExtInst %v3float %1 Cross %e1 %e2
%n = OpExtInst %v3float %1 Normalize %cross
%offset3 = OpVectorTimesScalar %v3float %n %float_0_08
%ox = OpCompositeExtract %float %offset3 0
%oy = OpCompositeExtract %float %offset3 1
%oz = OpCompositeExtract %float %offset3 2
%offset4 = OpCompositeConstruct %v4float %ox %oy %oz %float_0
%pos0 = OpFAdd %v4float %p0_v4 %offset4
%pos1 = OpFAdd %v4float %p1_v4 %offset4
%pos2 = OpFAdd %v4float %p2_v4 %offset4
%c0_ptr = OpAccessChain %ptr_Input_v3float %color %int_0
%c1_ptr = OpAccessChain %ptr_Input_v3float %color %int_1
%c2_ptr = OpAccessChain %ptr_Input_v3float %color %int_2
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr
%c2 = OpLoad %v3float %c2_ptr
%out_pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr %pos0
OpStore %outColor %c0
OpEmitVertex
OpStore %out_pos_ptr %pos1
OpStore %outColor %c1
OpEmitVertex
OpStore %out_pos_ptr %pos2
OpStore %outColor %c2
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
