using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var solver = new Solver();
        solver.Run();
    }
}

internal enum Strategy
{
    SurfaceGreedy,
    LayeredGreedy,
    DeepGreedy
}

internal readonly struct MovePlan
{
    public MovePlan(int columnIndex, int x, int y, int z, int mass, int value, bool xFirst, int pathCost, long profit)
    {
        ColumnIndex = columnIndex;
        X = x;
        Y = y;
        Z = z;
        Mass = mass;
        Value = value;
        XFirst = xFirst;
        PathCost = pathCost;
        Profit = profit;
    }

    public int ColumnIndex { get; }
    public int X { get; }
    public int Y { get; }
    public int Z { get; }
    public int Mass { get; }
    public int Value { get; }
    public bool XFirst { get; }
    public int PathCost { get; }
    public long Profit { get; }
    public int Distance => X + Y;
}

internal readonly struct QueueEntry
{
    public QueueEntry(int columnIndex, int revision)
    {
        ColumnIndex = columnIndex;
        Revision = revision;
    }

    public int ColumnIndex { get; }
    public int Revision { get; }
}

internal sealed class Solver
{
    private const int InvalidCost = int.MaxValue / 4;
    private const byte LayeredUnknownSentinel = byte.MaxValue;

    private int _xCount;
    private int _yCount;
    private int _zCount;
    private int _columnCount;
    private Strategy _strategy;

    private int[] _surfaceZ = Array.Empty<int>();
    private byte[] _currentTopD = Array.Empty<byte>();

    private byte[] _deepWorld = Array.Empty<byte>();

    private int _layerKeepCount;
    private int[] _layeredZ = Array.Empty<int>();
    private byte[] _layeredD = Array.Empty<byte>();
    private byte[] _layeredTopPtr = Array.Empty<byte>();
    private bool[] _layeredTruncated = Array.Empty<bool>();

    private bool[] _pathBlocked = Array.Empty<bool>();
    private int[] _revisions = Array.Empty<int>();

    public void Run()
    {
        using var input = new FastInput(Console.OpenStandardInput(1 << 20));
        ReadHeader(input);
        ChooseStrategy();
        AllocateStorage();
        ReadWorld(input);

        var operations = PlanOperations();
        var output = new StringBuilder(Math.Max(16, operations.Count * 32));
        output.AppendLine(operations.Count.ToString(CultureInfo.InvariantCulture));
        foreach (string operation in operations)
        {
            output.AppendLine(operation);
        }

        Console.Write(output.ToString());
    }

    private void ReadHeader(FastInput input)
    {
        string header = input.ReadAsciiLine();
        if (string.IsNullOrWhiteSpace(header))
        {
            throw new InvalidDataException("Missing header.");
        }

        string[] parts = header.Split((char[])null!, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length != 6)
        {
            throw new InvalidDataException("Header must contain 6 values.");
        }

        _xCount = int.Parse(parts[0], CultureInfo.InvariantCulture);
        _yCount = int.Parse(parts[1], CultureInfo.InvariantCulture);
        _zCount = int.Parse(parts[2], CultureInfo.InvariantCulture);

        _ = double.Parse(parts[3], CultureInfo.InvariantCulture);
        _ = double.Parse(parts[4], CultureInfo.InvariantCulture);
        _ = double.Parse(parts[5], CultureInfo.InvariantCulture);

        if (_xCount <= 0 || _yCount <= 0 || _zCount <= 0)
        {
            throw new InvalidDataException("Voxel counts must be positive.");
        }

        checked
        {
            _columnCount = _xCount * _yCount;
        }
    }

    private void ChooseStrategy()
    {
        long totalVoxels = 1L * _xCount * _yCount * _zCount;

        if (totalVoxels <= 32_000_000L && _columnCount <= 1_500_000)
        {
            _strategy = Strategy.DeepGreedy;
            return;
        }

        if (totalVoxels <= 512_000_000L && _columnCount <= 4_000_000)
        {
            _strategy = Strategy.LayeredGreedy;
            return;
        }

        _strategy = Strategy.SurfaceGreedy;
    }

    private void AllocateStorage()
    {
        _surfaceZ = new int[_columnCount];
        Array.Fill(_surfaceZ, -1);

        _currentTopD = new byte[_columnCount];
        _revisions = new int[_columnCount];

        switch (_strategy)
        {
            case Strategy.DeepGreedy:
            {
                int total = checked(_columnCount * _zCount);
                _deepWorld = new byte[total];
                break;
            }
            case Strategy.LayeredGreedy:
            {
                _layerKeepCount = 6;
                int slots = checked(_columnCount * _layerKeepCount);
                _layeredZ = new int[slots];
                _layeredD = new byte[slots];
                _layeredTopPtr = new byte[_columnCount];
                Array.Fill(_layeredTopPtr, LayeredUnknownSentinel);
                _layeredTruncated = new bool[_columnCount];
                break;
            }
            case Strategy.SurfaceGreedy:
            {
                _pathBlocked = new bool[_columnCount];
                break;
            }
        }
    }

    private void ReadWorld(FastInput input)
    {
        switch (_strategy)
        {
            case Strategy.DeepGreedy:
                ReadDeepWorld(input);
                break;
            case Strategy.LayeredGreedy:
                ReadLayeredWorld(input);
                break;
            default:
                ReadSurfaceWorld(input);
                break;
        }
    }

    private void ReadDeepWorld(FastInput input)
    {
        for (int x = 0; x < _xCount; x++)
        {
            int columnStart = x * _yCount;
            for (int y = 0; y < _yCount; y++)
            {
                int columnIndex = columnStart + y;
                int baseOffset = columnIndex * _zCount;
                for (int z = 0; z < _zCount; z++)
                {
                    byte d = input.ReadRequiredByte();
                    _deepWorld[baseOffset + z] = d;
                    if (d != 0)
                    {
                        _surfaceZ[columnIndex] = z;
                        _currentTopD[columnIndex] = d;
                    }
                }
            }
        }
    }

    private void ReadSurfaceWorld(FastInput input)
    {
        for (int x = 0; x < _xCount; x++)
        {
            int columnStart = x * _yCount;
            for (int y = 0; y < _yCount; y++)
            {
                int columnIndex = columnStart + y;
                for (int z = 0; z < _zCount; z++)
                {
                    byte d = input.ReadRequiredByte();
                    if (d != 0)
                    {
                        _surfaceZ[columnIndex] = z;
                        _currentTopD[columnIndex] = d;
                    }
                }
            }
        }
    }

    private void ReadLayeredWorld(FastInput input)
    {
        for (int x = 0; x < _xCount; x++)
        {
            int columnStart = x * _yCount;
            for (int y = 0; y < _yCount; y++)
            {
                int columnIndex = columnStart + y;
                int slotBase = columnIndex * _layerKeepCount;
                int kept = 0;
                bool truncated = false;

                for (int z = 0; z < _zCount; z++)
                {
                    byte d = input.ReadRequiredByte();
                    if (d == 0)
                    {
                        continue;
                    }

                    _surfaceZ[columnIndex] = z;
                    _currentTopD[columnIndex] = d;

                    if (kept < _layerKeepCount)
                    {
                        _layeredZ[slotBase + kept] = z;
                        _layeredD[slotBase + kept] = d;
                        kept++;
                    }
                    else
                    {
                        truncated = true;
                        for (int i = 1; i < _layerKeepCount; i++)
                        {
                            _layeredZ[slotBase + i - 1] = _layeredZ[slotBase + i];
                            _layeredD[slotBase + i - 1] = _layeredD[slotBase + i];
                        }

                        _layeredZ[slotBase + _layerKeepCount - 1] = z;
                        _layeredD[slotBase + _layerKeepCount - 1] = d;
                    }
                }

                if (kept > 0)
                {
                    _layeredTopPtr[columnIndex] = (byte)(kept - 1);
                }

                _layeredTruncated[columnIndex] = truncated;
            }
        }
    }

    private List<string> PlanOperations()
    {
        int plantHeight = _surfaceZ[0];
        int candidateCap = GetCandidateCap();
        int operationCap = GetOperationCap();

        var selected = SelectCandidateColumns(candidateCap, plantHeight);
        var queue = new PriorityQueue<QueueEntry, long>();

        foreach (int columnIndex in selected)
        {
            if (!CanMineCurrentTop(columnIndex))
            {
                continue;
            }

            if (TryPlanCurrentMove(columnIndex, out MovePlan plan))
            {
                queue.Enqueue(new QueueEntry(columnIndex, _revisions[columnIndex]), -BuildQueuePriority(plan));
            }
        }

        var operations = new List<string>(Math.Min(operationCap, 4096));

        while (queue.Count > 0 && operations.Count < operationCap)
        {
            queue.TryDequeue(out QueueEntry entry, out _);

            if (entry.Revision != _revisions[entry.ColumnIndex])
            {
                continue;
            }

            if (!CanMineCurrentTop(entry.ColumnIndex))
            {
                continue;
            }

            if (!TryPlanCurrentMove(entry.ColumnIndex, out MovePlan plan))
            {
                continue;
            }

            operations.Add(BuildOperationLine(plan));
            ApplyMine(plan.ColumnIndex);

            if (TryPlanCurrentMove(plan.ColumnIndex, out MovePlan nextPlan))
            {
                queue.Enqueue(new QueueEntry(plan.ColumnIndex, _revisions[plan.ColumnIndex]), -BuildQueuePriority(nextPlan));
            }
        }

        return operations;
    }

    private int GetCandidateCap()
    {
        return _strategy switch
        {
            Strategy.DeepGreedy => _columnCount <= 100_000 ? _columnCount : 60_000,
            Strategy.LayeredGreedy => _columnCount <= 80_000 ? _columnCount : 40_000,
            _ => _columnCount <= 40_000 ? _columnCount : 15_000
        };
    }

    private int GetOperationCap()
    {
        return _strategy switch
        {
            Strategy.DeepGreedy => 50_000,
            Strategy.LayeredGreedy => 25_000,
            _ => 10_000
        };
    }

    private List<int> SelectCandidateColumns(int candidateCap, int plantHeight)
    {
        if (candidateCap >= _columnCount)
        {
            var all = new List<int>(_columnCount);
            for (int i = 0; i < _columnCount; i++)
            {
                if (CanMineCurrentTop(i))
                {
                    all.Add(i);
                }
            }

            return all;
        }

        var best = new PriorityQueue<int, long>();
        for (int i = 0; i < _columnCount; i++)
        {
            if (!CanMineCurrentTop(i))
            {
                continue;
            }

            long score = ApproximateScore(i, plantHeight);
            if (best.Count < candidateCap)
            {
                best.Enqueue(i, score);
                continue;
            }

            best.TryPeek(out _, out long smallestScore);
            if (score <= smallestScore)
            {
                continue;
            }

            best.Dequeue();
            best.Enqueue(i, score);
        }

        var result = new List<int>(best.Count);
        while (best.Count > 0)
        {
            result.Add(best.Dequeue());
        }

        return result;
    }

    private bool CanMineCurrentTop(int columnIndex)
    {
        if (_currentTopD[columnIndex] == 0)
        {
            return false;
        }

        int mass = _currentTopD[columnIndex] & 15;
        int value = _currentTopD[columnIndex] >> 4;
        if (mass <= 0 || value <= 0)
        {
            return false;
        }

        if (_strategy == Strategy.SurfaceGreedy)
        {
            if (columnIndex == 0 || _pathBlocked[columnIndex])
            {
                return false;
            }
        }

        if (_strategy == Strategy.LayeredGreedy)
        {
            byte ptr = _layeredTopPtr[columnIndex];
            if (ptr == LayeredUnknownSentinel)
            {
                return false;
            }

            if (_layeredTruncated[columnIndex] && ptr == 0)
            {
                return false;
            }
        }

        return true;
    }

    private long ApproximateScore(int columnIndex, int plantHeight)
    {
        byte d = _currentTopD[columnIndex];
        int mass = d & 15;
        int value = d >> 4;
        int x = columnIndex / _yCount;
        int y = columnIndex - (x * _yCount);
        int z = _surfaceZ[columnIndex];

        long revenue = 1000L * mass * value;
        long downhillBias = 5L * mass * Math.Max(0, z - plantHeight);
        long distancePenalty = 2L * mass * (x + y);
        return revenue + downhillBias - distancePenalty;
    }

    private static long BuildQueuePriority(in MovePlan plan)
    {
        return (plan.Profit * 100_000L) + plan.Distance;
    }

    private bool TryPlanCurrentMove(int columnIndex, out MovePlan plan)
    {
        plan = default;
        if (!CanMineCurrentTop(columnIndex))
        {
            return false;
        }

        byte d = _currentTopD[columnIndex];
        int mass = d & 15;
        int value = d >> 4;
        int z = _surfaceZ[columnIndex];
        int x = columnIndex / _yCount;
        int y = columnIndex - (x * _yCount);

        int xFirstCost = EvaluateRouteCost(x, y, z, xFirst: true);
        int yFirstCost = EvaluateRouteCost(x, y, z, xFirst: false);

        int bestCost;
        bool xFirst;
        if (xFirstCost <= yFirstCost)
        {
            bestCost = xFirstCost;
            xFirst = true;
        }
        else
        {
            bestCost = yFirstCost;
            xFirst = false;
        }

        if (bestCost >= InvalidCost)
        {
            return false;
        }

        long revenue = 1000L * mass * value;
        long cost = (long)mass * bestCost;
        long profit = revenue - cost;
        if (profit <= 0)
        {
            return false;
        }

        plan = new MovePlan(columnIndex, x, y, z, mass, value, xFirst, bestCost, profit);
        return true;
    }

    private int EvaluateRouteCost(int x, int y, int startZ, bool xFirst)
    {
        int currentZ = startZ;
        int cost = 0;
        int cx = x;
        int cy = y;

        if (xFirst)
        {
            while (cx > 0)
            {
                cx--;
                int nextIndex = (cx * _yCount) + cy;
                if (!TryAdvance(nextIndex, ref currentZ, ref cost))
                {
                    return InvalidCost;
                }
            }

            while (cy > 0)
            {
                cy--;
                int nextIndex = (cx * _yCount) + cy;
                if (!TryAdvance(nextIndex, ref currentZ, ref cost))
                {
                    return InvalidCost;
                }
            }
        }
        else
        {
            while (cy > 0)
            {
                cy--;
                int nextIndex = (cx * _yCount) + cy;
                if (!TryAdvance(nextIndex, ref currentZ, ref cost))
                {
                    return InvalidCost;
                }
            }

            while (cx > 0)
            {
                cx--;
                int nextIndex = (cx * _yCount) + cy;
                if (!TryAdvance(nextIndex, ref currentZ, ref cost))
                {
                    return InvalidCost;
                }
            }
        }

        return cost;
    }

    private bool TryAdvance(int nextIndex, ref int currentZ, ref int cost)
    {
        if (_strategy == Strategy.SurfaceGreedy && _pathBlocked[nextIndex])
        {
            return false;
        }

        int nextZ = _surfaceZ[nextIndex];
        if (nextZ > currentZ + 1)
        {
            return false;
        }

        if (nextZ >= currentZ)
        {
            cost += nextZ == currentZ ? 1 : 5;
        }

        currentZ = nextZ;
        return true;
    }

    private string BuildOperationLine(in MovePlan plan)
    {
        int pathLength = plan.Distance;
        if (pathLength == 0)
        {
            return string.Create(
                CultureInfo.InvariantCulture,
                $"0 {plan.X} {plan.Y} {plan.Z} 0 0");
        }

        int intermediateCount = pathLength - 1;
        var builder = new StringBuilder(32 + (pathLength * 10));
        builder
            .Append(intermediateCount.ToString(CultureInfo.InvariantCulture)).Append(' ')
            .Append(plan.X.ToString(CultureInfo.InvariantCulture)).Append(' ')
            .Append(plan.Y.ToString(CultureInfo.InvariantCulture)).Append(' ')
            .Append(plan.Z.ToString(CultureInfo.InvariantCulture));

        int x = plan.X;
        int y = plan.Y;
        int emitted = 0;

        if (plan.XFirst)
        {
            while (x > 0)
            {
                x--;
                emitted++;
                if (emitted < pathLength)
                {
                    builder.Append(' ')
                        .Append(x.ToString(CultureInfo.InvariantCulture))
                        .Append(' ')
                        .Append(y.ToString(CultureInfo.InvariantCulture));
                }
            }

            while (y > 0)
            {
                y--;
                emitted++;
                if (emitted < pathLength)
                {
                    builder.Append(' ')
                        .Append(x.ToString(CultureInfo.InvariantCulture))
                        .Append(' ')
                        .Append(y.ToString(CultureInfo.InvariantCulture));
                }
            }
        }
        else
        {
            while (y > 0)
            {
                y--;
                emitted++;
                if (emitted < pathLength)
                {
                    builder.Append(' ')
                        .Append(x.ToString(CultureInfo.InvariantCulture))
                        .Append(' ')
                        .Append(y.ToString(CultureInfo.InvariantCulture));
                }
            }

            while (x > 0)
            {
                x--;
                emitted++;
                if (emitted < pathLength)
                {
                    builder.Append(' ')
                        .Append(x.ToString(CultureInfo.InvariantCulture))
                        .Append(' ')
                        .Append(y.ToString(CultureInfo.InvariantCulture));
                }
            }
        }

        builder.Append(" 0 0");
        return builder.ToString();
    }

    private void ApplyMine(int columnIndex)
    {
        switch (_strategy)
        {
            case Strategy.DeepGreedy:
                ApplyDeepMine(columnIndex);
                break;
            case Strategy.LayeredGreedy:
                ApplyLayeredMine(columnIndex);
                break;
            default:
                ApplySurfaceMine(columnIndex);
                break;
        }

        _revisions[columnIndex]++;
    }

    private void ApplySurfaceMine(int columnIndex)
    {
        _currentTopD[columnIndex] = 0;
        _pathBlocked[columnIndex] = true;
    }

    private void ApplyLayeredMine(int columnIndex)
    {
        byte ptr = _layeredTopPtr[columnIndex];
        if (ptr == LayeredUnknownSentinel)
        {
            _currentTopD[columnIndex] = 0;
            _surfaceZ[columnIndex] = -1;
            return;
        }

        int slotBase = columnIndex * _layerKeepCount;
        _layeredD[slotBase + ptr] = 0;

        if (ptr == 0)
        {
            _layeredTopPtr[columnIndex] = LayeredUnknownSentinel;
            _currentTopD[columnIndex] = 0;
            _surfaceZ[columnIndex] = -1;
            return;
        }

        ptr--;
        _layeredTopPtr[columnIndex] = ptr;
        _currentTopD[columnIndex] = _layeredD[slotBase + ptr];
        _surfaceZ[columnIndex] = _layeredZ[slotBase + ptr];
    }

    private void ApplyDeepMine(int columnIndex)
    {
        int topZ = _surfaceZ[columnIndex];
        if (topZ < 0)
        {
            _currentTopD[columnIndex] = 0;
            return;
        }

        int baseOffset = columnIndex * _zCount;
        _deepWorld[baseOffset + topZ] = 0;

        int nextZ = topZ - 1;
        while (nextZ >= 0 && _deepWorld[baseOffset + nextZ] == 0)
        {
            nextZ--;
        }

        _surfaceZ[columnIndex] = nextZ;
        _currentTopD[columnIndex] = nextZ >= 0 ? _deepWorld[baseOffset + nextZ] : (byte)0;
    }
}

internal sealed class FastInput : IDisposable
{
    private readonly Stream _stream;
    private readonly byte[] _buffer;
    private int _length;
    private int _position;

    public FastInput(Stream stream)
    {
        _stream = stream;
        _buffer = new byte[1 << 20];
    }

    public string ReadAsciiLine()
    {
        var builder = new StringBuilder(128);
        while (true)
        {
            int value = ReadByteInternal();
            if (value < 0 || value == '\n')
            {
                break;
            }

            if (value != '\r')
            {
                builder.Append((char)value);
            }
        }

        return builder.ToString();
    }

    public byte ReadRequiredByte()
    {
        int value = ReadByteInternal();
        if (value < 0)
        {
            throw new EndOfStreamException("Unexpected end of voxel stream.");
        }

        return (byte)value;
    }

    private int ReadByteInternal()
    {
        if (_position >= _length)
        {
            _length = _stream.Read(_buffer, 0, _buffer.Length);
            _position = 0;
            if (_length == 0)
            {
                return -1;
            }
        }

        return _buffer[_position++];
    }

    public void Dispose()
    {
        _stream.Dispose();
    }
}
