declare const require: any;

const fs = require('fs');

type PackageInput = {
  id: unknown;
  x: number;
  y: number;
};

const EPS = 1e-9;

function main(): void {
  const input: string = fs.readFileSync(0, 'utf8');
  const data = JSON.parse(input);

  const depot = Array.isArray(data.depot) ? data.depot : [0, 0];
  const depotX = toNumber(depot[0]);
  const depotY = toNumber(depot[1]);
  const packagesRaw: PackageInput[] = Array.isArray(data.packages) ? data.packages : [];
  const n = packagesRaw.length;

  if (n === 0) {
    fs.writeFileSync(1, JSON.stringify({ routes: [] }));
    return;
  }

  const ids: unknown[] = new Array(n);
  const xs = new Float64Array(n);
  const ys = new Float64Array(n);
  const depDist = new Float64Array(n);
  const angle = new Float64Array(n);
  const radius2 = new Float64Array(n);
  const morton = new Uint32Array(n);

  let minX = Infinity;
  let maxX = -Infinity;
  let minY = Infinity;
  let maxY = -Infinity;

  for (let i = 0; i < n; i++) {
    const pkg = packagesRaw[i];
    const x = toNumber(pkg.x);
    const y = toNumber(pkg.y);
    ids[i] = pkg.id;
    xs[i] = x;
    ys[i] = y;

    if (x < minX) minX = x;
    if (x > maxX) maxX = x;
    if (y < minY) minY = y;
    if (y > maxY) maxY = y;
  }

  const rangeX = maxX > minX ? maxX - minX : 1;
  const rangeY = maxY > minY ? maxY - minY : 1;

  for (let i = 0; i < n; i++) {
    const dx = xs[i] - depotX;
    const dy = ys[i] - depotY;
    depDist[i] = Math.sqrt(dx * dx + dy * dy);
    angle[i] = Math.atan2(dy, dx);
    radius2[i] = dx * dx + dy * dy;

    const nx = clampUInt16(((xs[i] - minX) / rangeX) * 65535.0);
    const ny = clampUInt16(((ys[i] - minY) / rangeY) * 65535.0);
    morton[i] = mortonCode(nx, ny);
  }

  const baseIndices: number[] = [];
  for (let i = 0; i < n; i++) baseIndices.push(i);

  const angleOrder = baseIndices.slice();
  angleOrder.sort(function (a: number, b: number): number {
    const da = angle[a] - angle[b];
    if (da !== 0) return da;
    const dr = radius2[a] - radius2[b];
    if (dr !== 0) return dr;
    const dx = xs[a] - xs[b];
    if (dx !== 0) return dx;
    const dy = ys[a] - ys[b];
    if (dy !== 0) return dy;
    return a - b;
  });

  const xOrder = baseIndices.slice();
  xOrder.sort(function (a: number, b: number): number {
    const dx = xs[a] - xs[b];
    if (dx !== 0) return dx;
    const dy = ys[a] - ys[b];
    if (dy !== 0) return dy;
    return a - b;
  });

  const yOrder = baseIndices.slice();
  yOrder.sort(function (a: number, b: number): number {
    const dy = ys[a] - ys[b];
    if (dy !== 0) return dy;
    const dx = xs[a] - xs[b];
    if (dx !== 0) return dx;
    return a - b;
  });

  const mortonOrder = baseIndices.slice();
  mortonOrder.sort(function (a: number, b: number): number {
    const dm = morton[a] - morton[b];
    if (dm !== 0) return dm;
    const dx = xs[a] - xs[b];
    if (dx !== 0) return dx;
    const dy = ys[a] - ys[b];
    if (dy !== 0) return dy;
    return a - b;
  });

  const candidates: number[][] = [];
  candidates.push(bestRotationFromCycle(angleOrder, depDist, xs, ys));
  candidates.push(buildAngleSnake(angleOrder, radius2));
  candidates.push(bestRotationFromCycle(mortonOrder, depDist, xs, ys));
  candidates.push(buildStripSnake(xOrder, xs, ys, true));
  candidates.push(buildStripSnake(yOrder, xs, ys, false));

  if (n <= 450) {
    candidates.push(buildFarthestInsertion(depDist, xs, ys));
  }

  const neighborOrders: number[][] = [angleOrder, xOrder, yOrder, mortonOrder];
  const neighbors = buildCandidateNeighbors(n, neighborOrders, xs, ys);

  const scored: CandidateScore[] = [];
  for (let i = 0; i < candidates.length; i++) {
    scored.push({
      index: i,
      rawCost: routeCost(candidates[i], depDist, xs, ys),
    });
  }

  scored.sort(function (a: CandidateScore, b: CandidateScore): number {
    if (a.rawCost !== b.rawCost) return a.rawCost - b.rawCost;
    return a.index - b.index;
  });

  let improveCount = candidates.length;
  if (n > 2500) improveCount = Math.min(improveCount, 4);
  if (n > 12000) improveCount = Math.min(improveCount, 3);
  if (n > 40000) improveCount = Math.min(improveCount, 2);

  let bestOrder: number[] | null = null;
  let bestCost = Infinity;

  for (let rank = 0; rank < improveCount; rank++) {
    const candidate = candidates[scored[rank].index].slice();
    improveRoute(candidate, neighbors, depDist, xs, ys);
    const cost = routeCost(candidate, depDist, xs, ys);
    if (
      cost + EPS < bestCost ||
      (Math.abs(cost - bestCost) <= EPS && lexicographicallyBetter(candidate, bestOrder))
    ) {
      bestCost = cost;
      bestOrder = candidate;
    }
  }

  if (bestOrder === null) {
    bestOrder = candidates[0].slice();
  }

  const route: unknown[] = new Array(bestOrder.length);
  for (let i = 0; i < bestOrder.length; i++) {
    route[i] = ids[bestOrder[i]];
  }

  fs.writeFileSync(1, JSON.stringify({ routes: [route] }));
}

type CandidateScore = {
  index: number;
  rawCost: number;
};

function toNumber(value: unknown): number {
  const num = Number(value);
  return isFinite(num) ? num : 0;
}

function clampUInt16(value: number): number {
  if (value <= 0) return 0;
  if (value >= 65535) return 65535;
  return Math.floor(value + 0.5);
}

function part1By1(x: number): number {
  x &= 0xffff;
  x = (x | (x << 8)) & 0x00ff00ff;
  x = (x | (x << 4)) & 0x0f0f0f0f;
  x = (x | (x << 2)) & 0x33333333;
  x = (x | (x << 1)) & 0x55555555;
  return x >>> 0;
}

function mortonCode(x: number, y: number): number {
  return (part1By1(x) | (part1By1(y) << 1)) >>> 0;
}

function distIdx(a: number, b: number, xs: Float64Array, ys: Float64Array): number {
  const dx = xs[a] - xs[b];
  const dy = ys[a] - ys[b];
  return Math.sqrt(dx * dx + dy * dy);
}

function routeCost(
  order: number[],
  depDist: Float64Array,
  xs: Float64Array,
  ys: Float64Array
): number {
  const n = order.length;
  if (n === 0) return 0;

  let total = depDist[order[0]] + depDist[order[n - 1]];
  for (let i = 1; i < n; i++) {
    total += distIdx(order[i - 1], order[i], xs, ys);
  }
  return total;
}

function bestRotationFromCycle(
  cycle: number[],
  depDist: Float64Array,
  xs: Float64Array,
  ys: Float64Array
): number[] {
  const n = cycle.length;
  if (n <= 1) return cycle.slice();

  let bestStart = 0;
  let bestPenalty = Infinity;

  for (let start = 0; start < n; start++) {
    const prev = cycle[(start + n - 1) % n];
    const first = cycle[start];
    const penalty = depDist[prev] + depDist[first] - distIdx(prev, first, xs, ys);
    if (penalty + EPS < bestPenalty || (Math.abs(penalty - bestPenalty) <= EPS && start < bestStart)) {
      bestPenalty = penalty;
      bestStart = start;
    }
  }

  const result: number[] = new Array(n);
  for (let i = 0; i < n; i++) {
    result[i] = cycle[(bestStart + i) % n];
  }
  return result;
}

function buildAngleSnake(angleOrder: number[], radius2: Float64Array): number[] {
  const n = angleOrder.length;
  const block = Math.max(12, Math.floor(Math.sqrt(n)));
  const result: number[] = [];

  for (let start = 0, blockIndex = 0; start < n; start += block, blockIndex++) {
    const end = Math.min(n, start + block);
    const chunk = angleOrder.slice(start, end);
    chunk.sort(function (a: number, b: number): number {
      const dr = blockIndex % 2 === 0 ? radius2[a] - radius2[b] : radius2[b] - radius2[a];
      if (dr !== 0) return dr;
      return a - b;
    });
    for (let i = 0; i < chunk.length; i++) result.push(chunk[i]);
  }

  return result;
}

function buildStripSnake(
  primaryOrder: number[],
  xs: Float64Array,
  ys: Float64Array,
  primaryIsX: boolean
): number[] {
  const n = primaryOrder.length;
  const block = Math.max(12, Math.floor(Math.sqrt(n)));
  const result: number[] = [];

  for (let start = 0, blockIndex = 0; start < n; start += block, blockIndex++) {
    const end = Math.min(n, start + block);
    const chunk = primaryOrder.slice(start, end);
    chunk.sort(function (a: number, b: number): number {
      let d: number;
      if (primaryIsX) {
        d = blockIndex % 2 === 0 ? ys[a] - ys[b] : ys[b] - ys[a];
        if (d !== 0) return d;
        d = xs[a] - xs[b];
      } else {
        d = blockIndex % 2 === 0 ? xs[a] - xs[b] : xs[b] - xs[a];
        if (d !== 0) return d;
        d = ys[a] - ys[b];
      }
      if (d !== 0) return d;
      return a - b;
    });
    for (let i = 0; i < chunk.length; i++) result.push(chunk[i]);
  }

  return result;
}

function buildFarthestInsertion(
  depDist: Float64Array,
  xs: Float64Array,
  ys: Float64Array
): number[] {
  const n = depDist.length;
  if (n <= 1) {
    return n === 1 ? [0] : [];
  }

  let first = 0;
  for (let i = 1; i < n; i++) {
    if (depDist[i] > depDist[first] + EPS || (Math.abs(depDist[i] - depDist[first]) <= EPS && i < first)) {
      first = i;
    }
  }

  let second = -1;
  let bestSecondDist = -1;
  for (let i = 0; i < n; i++) {
    if (i === first) continue;
    const d = distIdx(first, i, xs, ys);
    if (d > bestSecondDist + EPS || (Math.abs(d - bestSecondDist) <= EPS && i < second)) {
      bestSecondDist = d;
      second = i;
    }
  }

  if (second < 0) return [first];

  const route: number[] = [first, second];
  const inRoute = new Uint8Array(n);
  const minToRoute = new Float64Array(n);
  inRoute[first] = 1;
  inRoute[second] = 1;

  for (let i = 0; i < n; i++) {
    if (inRoute[i]) {
      minToRoute[i] = 0;
    } else {
      const d1 = distIdx(i, first, xs, ys);
      const d2 = distIdx(i, second, xs, ys);
      minToRoute[i] = d1 < d2 ? d1 : d2;
    }
  }

  let used = 2;
  while (used < n) {
    let point = -1;
    let farthest = -1;
    for (let i = 0; i < n; i++) {
      if (inRoute[i]) continue;
      const score = minToRoute[i];
      if (score > farthest + EPS || (Math.abs(score - farthest) <= EPS && (point < 0 || i < point))) {
        farthest = score;
        point = i;
      }
    }

    let bestPos = 0;
    let bestIncrease = depDist[point] + distIdx(point, route[0], xs, ys) - depDist[route[0]];

    for (let i = 0; i < route.length - 1; i++) {
      const a = route[i];
      const b = route[i + 1];
      const increase = distIdx(a, point, xs, ys) + distIdx(point, b, xs, ys) - distIdx(a, b, xs, ys);
      if (increase + EPS < bestIncrease || (Math.abs(increase - bestIncrease) <= EPS && i + 1 < bestPos)) {
        bestIncrease = increase;
        bestPos = i + 1;
      }
    }

    const tailIncrease =
      distIdx(route[route.length - 1], point, xs, ys) + depDist[point] - depDist[route[route.length - 1]];
    if (tailIncrease + EPS < bestIncrease || (Math.abs(tailIncrease - bestIncrease) <= EPS && route.length < bestPos)) {
      bestPos = route.length;
    }

    route.splice(bestPos, 0, point);
    inRoute[point] = 1;
    used++;

    for (let i = 0; i < n; i++) {
      if (inRoute[i]) continue;
      const d = distIdx(i, point, xs, ys);
      if (d < minToRoute[i]) minToRoute[i] = d;
    }
  }

  return route;
}

function buildCandidateNeighbors(
  n: number,
  orders: number[][],
  xs: Float64Array,
  ys: Float64Array
): number[][] {
  const neighbors: number[][] = new Array(n);
  for (let i = 0; i < n; i++) neighbors[i] = [];

  let reach = 6;
  if (n > 2000) reach = 4;
  if (n > 20000) reach = 3;

  for (let orderIndex = 0; orderIndex < orders.length; orderIndex++) {
    const order = orders[orderIndex];
    for (let pos = 0; pos < n; pos++) {
      const node = order[pos];
      const list = neighbors[node];
      const from = pos - reach > 0 ? pos - reach : 0;
      const to = pos + reach < n - 1 ? pos + reach : n - 1;
      for (let p = from; p <= to; p++) {
        if (p !== pos) list.push(order[p]);
      }
    }
  }

  let maxNeighbors = 32;
  if (n > 2000) maxNeighbors = 24;
  if (n > 20000) maxNeighbors = 18;

  for (let i = 0; i < n; i++) {
    const raw = neighbors[i];
    raw.sort(function (a: number, b: number): number {
      return a - b;
    });

    const unique: number[] = [];
    let prev = -1;
    for (let j = 0; j < raw.length; j++) {
      const v = raw[j];
      if (v !== i && (j === 0 || v !== prev)) unique.push(v);
      prev = v;
    }

    unique.sort(function (a: number, b: number): number {
      const da = distIdx(i, a, xs, ys);
      const db = distIdx(i, b, xs, ys);
      if (da !== db) return da - db;
      return a - b;
    });

    if (unique.length > maxNeighbors) unique.length = maxNeighbors;
    neighbors[i] = unique;
  }

  return neighbors;
}

function improveRoute(
  order: number[],
  neighbors: number[][],
  depDist: Float64Array,
  xs: Float64Array,
  ys: Float64Array
): void {
  const n = order.length;
  if (n <= 2) return;

  if (n <= 700) {
    fullTwoOpt(order, depDist, xs, ys, 12);
    return;
  }

  if (n <= 3000) {
    candidateTwoOpt(order, neighbors, depDist, xs, ys, 4);
    windowTwoOpt(order, depDist, xs, ys, 96, 2);
    return;
  }

  if (n <= 15000) {
    candidateTwoOpt(order, neighbors, depDist, xs, ys, 3);
    windowTwoOpt(order, depDist, xs, ys, 48, 1);
    return;
  }

  candidateTwoOpt(order, neighbors, depDist, xs, ys, 2);
  windowTwoOpt(order, depDist, xs, ys, 24, 1);
}

function reverseRange(order: number[], left: number, right: number): void {
  while (left < right) {
    const temp = order[left];
    order[left] = order[right];
    order[right] = temp;
    left++;
    right--;
  }
}

function fullTwoOpt(
  order: number[],
  depDist: Float64Array,
  xs: Float64Array,
  ys: Float64Array,
  maxPasses: number
): void {
  const n = order.length;
  for (let pass = 0; pass < maxPasses; pass++) {
    let improved = false;
    for (let i = 0; i < n - 1; i++) {
      const a = i === 0 ? -1 : order[i - 1];
      const b = order[i];
      const oldAB = a < 0 ? depDist[b] : distIdx(a, b, xs, ys);

      for (let j = i + 1; j < n; j++) {
        const c = order[j];
        const d = j === n - 1 ? -1 : order[j + 1];
        const oldCD = d < 0 ? depDist[c] : distIdx(c, d, xs, ys);
        const newAC = a < 0 ? depDist[c] : distIdx(a, c, xs, ys);
        const newBD = d < 0 ? depDist[b] : distIdx(b, d, xs, ys);
        if (newAC + newBD + EPS < oldAB + oldCD) {
          reverseRange(order, i, j);
          improved = true;
          if (i > 1) i -= 2;
          break;
        }
      }
    }

    if (!improved) break;
  }
}

function candidateTwoOpt(
  order: number[],
  neighbors: number[][],
  depDist: Float64Array,
  xs: Float64Array,
  ys: Float64Array,
  maxPasses: number
): void {
  const routeLength = order.length;
  const position = new Int32Array(neighbors.length);
  for (let i = 0; i < routeLength; i++) position[order[i]] = i;

  for (let pass = 0; pass < maxPasses; pass++) {
    let improved = false;

    for (let i = 0; i < routeLength - 1; i++) {
      const a = i === 0 ? -1 : order[i - 1];
      const b = order[i];
      const bestHolder: BestMove = { j: -1, delta: -EPS };

      scanTwoOptCandidates(neighbors[b], i, a, b, order, position, depDist, xs, ys, bestHolder);
      if (a >= 0) {
        scanTwoOptCandidates(neighbors[a], i, a, b, order, position, depDist, xs, ys, bestHolder);
      }

      if (bestHolder.j >= 0 && bestHolder.delta < -EPS) {
        reverseRange(order, i, bestHolder.j);
        for (let p = i; p <= bestHolder.j; p++) position[order[p]] = p;
        improved = true;
        if (i > 1) i -= 2;
      }
    }

    if (!improved) break;
  }
}

type BestMove = {
  j: number;
  delta: number;
};

function scanTwoOptCandidates(
  candidates: number[],
  i: number,
  a: number,
  b: number,
  order: number[],
  position: Int32Array,
  depDist: Float64Array,
  xs: Float64Array,
  ys: Float64Array,
  best: BestMove
): void {
  const routeLength = order.length;
  const oldAB = a < 0 ? depDist[b] : distIdx(a, b, xs, ys);

  for (let idx = 0; idx < candidates.length; idx++) {
    const c = candidates[idx];
    const j = position[c];
    if (j <= i) continue;

    const d = j === routeLength - 1 ? -1 : order[j + 1];
    const oldCD = d < 0 ? depDist[c] : distIdx(c, d, xs, ys);
    const newAC = a < 0 ? depDist[c] : distIdx(a, c, xs, ys);
    const newBD = d < 0 ? depDist[b] : distIdx(b, d, xs, ys);
    const delta = newAC + newBD - oldAB - oldCD;

    if (delta < best.delta || (Math.abs(delta - best.delta) <= EPS && (best.j < 0 || j < best.j))) {
      best.delta = delta;
      best.j = j;
    }
  }
}

function windowTwoOpt(
  order: number[],
  depDist: Float64Array,
  xs: Float64Array,
  ys: Float64Array,
  maxWindow: number,
  maxPasses: number
): void {
  const n = order.length;
  if (maxWindow < 1) return;

  for (let pass = 0; pass < maxPasses; pass++) {
    let improved = false;

    for (let i = 0; i < n - 1; i++) {
      const a = i === 0 ? -1 : order[i - 1];
      const b = order[i];
      const oldAB = a < 0 ? depDist[b] : distIdx(a, b, xs, ys);

      let bestJ = -1;
      let bestDelta = -EPS;
      const limit = Math.min(n - 1, i + maxWindow);

      for (let j = i + 1; j <= limit; j++) {
        const c = order[j];
        const d = j === n - 1 ? -1 : order[j + 1];
        const oldCD = d < 0 ? depDist[c] : distIdx(c, d, xs, ys);
        const newAC = a < 0 ? depDist[c] : distIdx(a, c, xs, ys);
        const newBD = d < 0 ? depDist[b] : distIdx(b, d, xs, ys);
        const delta = newAC + newBD - oldAB - oldCD;
        if (delta < bestDelta || (Math.abs(delta - bestDelta) <= EPS && (bestJ < 0 || j < bestJ))) {
          bestDelta = delta;
          bestJ = j;
        }
      }

      if (bestJ >= 0) {
        reverseRange(order, i, bestJ);
        improved = true;
        if (i > 1) i -= 2;
      }
    }

    if (!improved) break;
  }
}

function lexicographicallyBetter(candidate: number[], current: number[] | null): boolean {
  if (current === null) return true;
  const n = candidate.length;
  for (let i = 0; i < n; i++) {
    if (candidate[i] !== current[i]) return candidate[i] < current[i];
  }
  return false;
}

main();
