struct Edge {
    uint32_t u = 0;
    uint32_t v = 0;
};

class FastInput {
public:
    FastInput() : in_(stdin), size_(0), index_(0) {}

    template <typename T>
    bool read_unsigned(T& out) {
        out = 0;
        int c = next_non_space();
        if (c == EOF) {
            return false;
        }
        while (c >= '0' && c <= '9') {
            out = static_cast<T>(out * 10 + (c - '0'));
            c = next_char();
        }
        return true;
    }

private:
    static constexpr size_t kBufferSize = 1u << 20;
    FILE* in_;
    std::array<char, kBufferSize> buffer_{};
    size_t size_;
    size_t index_;

    int next_char() {
        if (index_ >= size_) {
            size_ = std::fread(buffer_.data(), 1, buffer_.size(), in_);
            index_ = 0;
            if (size_ == 0) {
                return EOF;
            }
        }
        return static_cast<unsigned char>(buffer_[index_++]);
    }

    int next_non_space() {
        int c = next_char();
        while (c != EOF && c <= ' ') {
            c = next_char();
        }
        return c;
    }
};

class FastOutput {
public:
    ~FastOutput() {
        flush();
    }

    void write_uint64(uint64_t x) {
        char tmp[32];
        int len = 0;
        if (x == 0) {
            tmp[len++] = '0';
        } else {
            while (x > 0) {
                tmp[len++] = static_cast<char>('0' + (x % 10));
                x /= 10;
            }
        }
        while (len-- > 0) {
            push(tmp[len]);
        }
    }

    void write_uint32(uint32_t x) {
        write_uint64(static_cast<uint64_t>(x));
    }

    void write_space() {
        push(' ');
    }

    void write_newline() {
        push('\n');
    }

    void flush() {
        if (index_ != 0) {
            std::fwrite(buffer_.data(), 1, index_, stdout);
            index_ = 0;
        }
    }

private:
    static constexpr size_t kBufferSize = 1u << 20;
    std::array<char, kBufferSize> buffer_{};
    size_t index_ = 0;

    void push(char c) {
        if (index_ == buffer_.size()) {
            flush();
        }
        buffer_[index_++] = c;
    }
};

class DSU {
public:
    explicit DSU(uint32_t n = 0) : parent_or_size_(n, -1) {}

    uint32_t find(uint32_t x) {
        uint32_t root = x;
        while (parent_or_size_[root] >= 0) {
            root = static_cast<uint32_t>(parent_or_size_[root]);
        }
        while (x != root) {
            const uint32_t parent = static_cast<uint32_t>(parent_or_size_[x]);
            parent_or_size_[x] = static_cast<int>(root);
            x = parent;
        }
        return root;
    }

    bool unite(uint32_t a, uint32_t b) {
        a = find(a);
        b = find(b);
        if (a == b) {
            return false;
        }
        if (parent_or_size_[a] > parent_or_size_[b]) {
            std::swap(a, b);
        }
        parent_or_size_[a] += parent_or_size_[b];
        parent_or_size_[b] = static_cast<int>(a);
        return true;
    }

private:
    std::vector<int> parent_or_size_;
};

class EdgeSpool {
public:
    EdgeSpool() {
        path_ = make_temp_path();
        out_.open(path_, std::ios::binary);
        if (!out_) {
            std::cerr << "failed to create temporary edge spool\n";
            std::exit(1);
        }
        buffer_.reserve(kChunkEdges);
    }

    ~EdgeSpool() {
        close_writer();
        std::error_code ec;
        std::filesystem::remove(path_, ec);
    }

    void push(const Edge& e) {
        buffer_.push_back(e);
        if (buffer_.size() == buffer_.capacity()) {
            flush_buffer();
        }
    }

    void close_writer() {
        if (out_.is_open()) {
            flush_buffer();
            out_.close();
        }
    }

    template <class F>
    void scan(F&& fn) const {
        std::ifstream in(path_, std::ios::binary);
        if (!in) {
            std::cerr << "failed to reopen temporary edge spool\n";
            std::exit(1);
        }
        std::vector<Edge> chunk(kChunkEdges);
        while (in) {
            in.read(reinterpret_cast<char*>(chunk.data()),
                    static_cast<std::streamsize>(chunk.size() * sizeof(Edge)));
            const std::streamsize bytes = in.gcount();
            const size_t count = static_cast<size_t>(bytes / static_cast<std::streamsize>(sizeof(Edge)));
            for (size_t i = 0; i < count; ++i) {
                fn(chunk[i]);
            }
        }
    }

private:
    static constexpr size_t kChunkEdges = 1u << 15;
    std::filesystem::path path_;
    std::ofstream out_;
    std::vector<Edge> buffer_;

    static std::filesystem::path make_temp_path() {
        const auto base = std::filesystem::temp_directory_path();
        const uint64_t seed =
            static_cast<uint64_t>(std::random_device{}()) ^
            (static_cast<uint64_t>(std::chrono::high_resolution_clock::now().time_since_epoch().count()) << 1);
        for (uint32_t attempt = 0; attempt < 256; ++attempt) {
            const std::string name = "mincut_" + std::to_string(seed) + "_" + std::to_string(attempt) + ".bin";
            const auto candidate = base / name;
            if (!std::filesystem::exists(candidate)) {
                return candidate;
            }
        }
        return base / "mincut_fallback.bin";
    }

    void flush_buffer() {
        if (!buffer_.empty()) {
            out_.write(reinterpret_cast<const char*>(buffer_.data()),
                       static_cast<std::streamsize>(buffer_.size() * sizeof(Edge)));
            if (!out_) {
                std::cerr << "failed while writing temporary edge spool\n";
                std::exit(1);
            }
            buffer_.clear();
        }
    }
};

struct EdgeStorage {
    uint32_t n = 0;
    uint64_t m = 0;
    bool in_memory = false;
    std::vector<Edge> edges;
    std::unique_ptr<EdgeSpool> spool;
    std::vector<uint32_t> degree;
    uint32_t min_degree_vertex = 0;
    uint32_t min_degree = 0;

    template <class F>
    void scan_edges(F&& fn) const {
        if (in_memory) {
            for (const Edge& e : edges) {
                fn(e);
            }
        } else {
            spool->scan(fn);
        }
    }
};

struct CutResult {
    uint64_t value = std::numeric_limits<uint64_t>::max();
    std::vector<uint32_t> side;
};

struct MultiGraph {
    uint32_t n = 0;
    std::vector<Edge> edges;
    std::vector<std::vector<uint32_t>> members;
};

static bool exact_feasible(uint32_t n, uint64_t m) {
    const long double work = static_cast<long double>(n) * static_cast<long double>(m);
    return n <= 15000u && m <= 1500000ull && work <= 2.5e9L;
}

static bool certificate_feasible(uint32_t n, uint64_t k) {
    constexpr uint64_t kForestBudgetBytes = 2ull << 30;
    constexpr uint64_t kCertBudgetBytes = 2ull << 30;
    const unsigned long long forest_bytes = static_cast<unsigned long long>(n) * k * sizeof(int);
    const unsigned long long cert_bytes =
        static_cast<unsigned long long>(n > 0 ? (n - 1) : 0) * k * sizeof(Edge);
    return forest_bytes <= kForestBudgetBytes && cert_bytes <= kCertBudgetBytes;
}

static EdgeStorage read_graph() {
    FastInput input;
    uint64_t n64 = 0;
    uint64_t m64 = 0;
    if (!input.read_unsigned(n64)) {
        std::cerr << "missing N\n";
        std::exit(1);
    }
    if (!input.read_unsigned(m64)) {
        std::cerr << "missing M\n";
        std::exit(1);
    }
    if (n64 > static_cast<uint64_t>(std::numeric_limits<uint32_t>::max())) {
        std::cerr << "N is too large for this implementation\n";
        std::exit(1);
    }

    EdgeStorage storage;
    storage.n = static_cast<uint32_t>(n64);
    storage.degree.assign(storage.n, 0);
    constexpr unsigned long long kEdgeMemoryBudget = 1536ull << 20;
    storage.in_memory = m64 <= kEdgeMemoryBudget / sizeof(Edge);
    if (storage.in_memory) {
        storage.edges.reserve(static_cast<size_t>(std::min<uint64_t>(m64, static_cast<uint64_t>(std::numeric_limits<size_t>::max()))));
    } else {
        storage.spool = std::make_unique<EdgeSpool>();
    }

    uint64_t effective_edges = 0;
    for (uint64_t i = 0; i < m64; ++i) {
        uint64_t u64 = 0;
        uint64_t v64 = 0;
        if (!input.read_unsigned(u64) || !input.read_unsigned(v64)) {
            std::cerr << "truncated edge list\n";
            std::exit(1);
        }
        if (u64 >= n64 || v64 >= n64) {
            std::cerr << "vertex id out of range\n";
            std::exit(1);
        }
        const uint32_t u = static_cast<uint32_t>(u64);
        const uint32_t v = static_cast<uint32_t>(v64);
        if (u == v) {
            continue;
        }
        ++storage.degree[u];
        ++storage.degree[v];
        ++effective_edges;
        const Edge e{u, v};
        if (storage.in_memory) {
            storage.edges.push_back(e);
        } else {
            storage.spool->push(e);
        }
    }

    if (!storage.in_memory) {
        storage.spool->close_writer();
    }

    storage.m = effective_edges;
    storage.min_degree = storage.n == 0 ? 0 : std::numeric_limits<uint32_t>::max();
    for (uint32_t v = 0; v < storage.n; ++v) {
        if (storage.degree[v] < storage.min_degree) {
            storage.min_degree = storage.degree[v];
            storage.min_degree_vertex = v;
        }
    }
    return storage;
}

template <class Scanner>
static std::vector<Edge> build_certificate(uint32_t n, uint64_t m, uint32_t k, Scanner&& scan, bool& used_all_edges) {
    std::vector<DSU> forests;
    forests.reserve(k);
    for (uint32_t i = 0; i < k; ++i) {
        forests.emplace_back(n);
    }

    std::vector<Edge> cert;
    const uint64_t reserve_count = std::min<uint64_t>(m, static_cast<uint64_t>(k) * (n > 0 ? (n - 1ull) : 0ull));
    cert.reserve(static_cast<size_t>(std::min<uint64_t>(reserve_count, static_cast<uint64_t>(std::numeric_limits<size_t>::max()))));

    scan([&](const Edge& e) {
        for (uint32_t i = 0; i < k; ++i) {
            if (forests[i].unite(e.u, e.v)) {
                cert.push_back(e);
                return;
            }
        }
    });

    used_all_edges = (cert.size() == m);
    return cert;
}

static MultiGraph make_multigraph(uint32_t n, const std::vector<Edge>& edges) {
    MultiGraph g;
    g.n = n;
    g.edges = edges;
    g.members.resize(n);
    for (uint32_t v = 0; v < n; ++v) {
        g.members[v].push_back(v);
    }
    return g;
}

static CutResult stoer_wagner_sparse(MultiGraph graph) {
    CutResult best;
    if (graph.n <= 1) {
        best.value = 0;
        return best;
    }

    std::vector<uint32_t> deg(graph.n, 0);
    for (const Edge& e : graph.edges) {
        ++deg[e.u];
        ++deg[e.v];
    }

    std::vector<std::unordered_map<uint32_t, uint64_t>> adj(graph.n);
    for (uint32_t v = 0; v < graph.n; ++v) {
        adj[v].reserve(static_cast<size_t>(deg[v] * 2ull + 1ull));
    }
    for (const Edge& e : graph.edges) {
        ++adj[e.u][e.v];
        ++adj[e.v][e.u];
    }

    std::vector<uint8_t> active(graph.n, 1);
    uint32_t active_count = graph.n;

    while (active_count > 1) {
        std::vector<uint64_t> weight(graph.n, 0);
        std::vector<uint8_t> added(graph.n, 0);
        std::priority_queue<std::pair<uint64_t, uint32_t>> pq;
        for (uint32_t v = 0; v < graph.n; ++v) {
            if (active[v]) {
                pq.push({0, v});
            }
        }

        int32_t prev = -1;
        int32_t last = -1;
        uint64_t last_weight = 0;

        for (uint32_t step = 0; step < active_count; ++step) {
            uint32_t v = 0;
            bool found = false;
            while (!pq.empty()) {
                const auto [w, cand] = pq.top();
                pq.pop();
                if (active[cand] && !added[cand] && w == weight[cand]) {
                    v = cand;
                    last_weight = w;
                    found = true;
                    break;
                }
            }
            if (!found) {
                break;
            }

            added[v] = 1;
            prev = last;
            last = static_cast<int32_t>(v);

            for (const auto& [to, cap] : adj[v]) {
                if (active[to] && !added[to]) {
                    weight[to] += cap;
                    pq.push({weight[to], to});
                }
            }
        }

        if (last < 0) {
            break;
        }

        const uint32_t t = static_cast<uint32_t>(last);
        if (last_weight < best.value) {
            best.value = last_weight;
            best.side = graph.members[t];
        }

        if (active_count == 2 || prev < 0) {
            break;
        }

        uint32_t a = static_cast<uint32_t>(prev);
        uint32_t b = t;
        uint32_t keep = a;
        uint32_t drop = b;
        if (adj[keep].size() < adj[drop].size()) {
            keep = b;
            drop = a;
        }

        if (!graph.members[drop].empty()) {
            auto& dst = graph.members[keep];
            auto& src = graph.members[drop];
            dst.insert(dst.end(), src.begin(), src.end());
            src.clear();
        }

        std::vector<std::pair<uint32_t, uint64_t>> neighbors;
        neighbors.reserve(adj[drop].size());
        for (const auto& entry : adj[drop]) {
            neighbors.push_back(entry);
        }

        for (const auto& [to, cap] : neighbors) {
            if (!active[to] || to == keep) {
                continue;
            }
            adj[keep][to] += cap;
            adj[to][keep] += cap;
            adj[to].erase(drop);
        }

        adj[keep].erase(drop);
        adj[keep].erase(keep);
        adj[drop].clear();
        active[drop] = 0;
        --active_count;
    }

    if (best.side.empty() && graph.n >= 1) {
        best.side.push_back(0);
        best.value = std::numeric_limits<uint64_t>::max();
    }
    return best;
}

static MultiGraph compress_graph(const MultiGraph& graph, DSU& dsu) {
    std::vector<int32_t> root_to_new(graph.n, -1);
    uint32_t new_n = 0;
    for (uint32_t v = 0; v < graph.n; ++v) {
        const uint32_t r = dsu.find(v);
        if (root_to_new[r] < 0) {
            root_to_new[r] = static_cast<int32_t>(new_n++);
        }
    }

    MultiGraph out;
    out.n = new_n;
    out.members.resize(new_n);
    for (uint32_t v = 0; v < graph.n; ++v) {
        const uint32_t id = static_cast<uint32_t>(root_to_new[dsu.find(v)]);
        auto& dst = out.members[id];
        const auto& src = graph.members[v];
        dst.insert(dst.end(), src.begin(), src.end());
    }

    out.edges.reserve(graph.edges.size());
    for (const Edge& e : graph.edges) {
        const uint32_t a = static_cast<uint32_t>(root_to_new[dsu.find(e.u)]);
        const uint32_t b = static_cast<uint32_t>(root_to_new[dsu.find(e.v)]);
        if (a != b) {
            out.edges.push_back({a, b});
        }
    }
    return out;
}

static uint64_t mix_seed(uint64_t x) {
    x += 0x9e3779b97f4a7c15ull;
    x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ull;
    x = (x ^ (x >> 27)) * 0x94d049bb133111ebull;
    return x ^ (x >> 31);
}

static MultiGraph random_contract_to(MultiGraph graph, uint32_t target, std::mt19937_64& rng) {
    if (graph.n <= target) {
        return graph;
    }

    while (graph.n > target) {
        if (graph.edges.empty()) {
            break;
        }
        DSU dsu(graph.n);
        uint32_t active = graph.n;
        std::uniform_int_distribution<size_t> pick(0, graph.edges.size() - 1);
        const uint64_t max_trials = std::max<uint64_t>(graph.edges.size() * 4ull, 4096ull);
        uint64_t trials = 0;

        while (active > target && trials < max_trials) {
            const Edge& e = graph.edges[pick(rng)];
            ++trials;
            if (dsu.unite(e.u, e.v)) {
                --active;
            }
        }

        graph = compress_graph(graph, dsu);
    }

    return graph;
}

static CutResult karger_stein(MultiGraph graph, std::mt19937_64& rng) {
    if (graph.n <= 1) {
        return CutResult{0, {}};
    }
    if (exact_feasible(graph.n, graph.edges.size()) || graph.n <= 6) {
        return stoer_wagner_sparse(std::move(graph));
    }

    const uint32_t target = static_cast<uint32_t>(std::ceil(static_cast<double>(graph.n) / std::sqrt(2.0))) + 1u;
    MultiGraph g1 = random_contract_to(graph, target, rng);
    MultiGraph g2 = random_contract_to(std::move(graph), target, rng);

    CutResult c1 = karger_stein(std::move(g1), rng);
    CutResult c2 = karger_stein(std::move(g2), rng);
    return (c1.value <= c2.value) ? std::move(c1) : std::move(c2);
}

static CutResult solve_certificate(uint32_t n, const std::vector<Edge>& edges, uint64_t upper_bound) {
    MultiGraph graph = make_multigraph(n, edges);

    if (exact_feasible(n, edges.size())) {
        return stoer_wagner_sparse(std::move(graph));
    }

    CutResult best;
    best.value = upper_bound;
    best.side = {0};

    const uint32_t reps = std::max<uint32_t>(6u, static_cast<uint32_t>(2.0 * std::log2(static_cast<double>(std::max<uint32_t>(n, 2u))) + 2.0));
    const uint64_t seed_base =
        mix_seed(static_cast<uint64_t>(n) * 1469598103934665603ull + static_cast<uint64_t>(edges.size()));

    for (uint32_t rep = 0; rep < reps; ++rep) {
        std::mt19937_64 rng(seed_base ^ mix_seed(rep + 1u));
        CutResult cand = karger_stein(graph, rng);
        if (cand.value < best.value) {
            best = std::move(cand);
            if (best.value <= 1) {
                break;
            }
        }
    }
    return best;
}

static uint64_t count_cut_edges(const EdgeStorage& storage, const std::vector<uint32_t>& side) {
    if (side.empty() || side.size() >= storage.n) {
        return std::numeric_limits<uint64_t>::max();
    }

    std::vector<uint8_t> mark(storage.n, 0);
    for (uint32_t v : side) {
        if (v < storage.n) {
            mark[v] = 1;
        }
    }

    uint64_t count = 0;
    storage.scan_edges([&](const Edge& e) {
        if (mark[e.u] != mark[e.v]) {
            ++count;
        }
    });
    return count;
}

static void output_cut(const EdgeStorage& storage, const std::vector<uint32_t>& side, uint64_t cut_value) {
    std::vector<uint8_t> mark(storage.n, 0);
    for (uint32_t v : side) {
        if (v < storage.n) {
            mark[v] = 1;
        }
    }

    FastOutput out;
    out.write_uint64(cut_value);
    out.write_newline();
    storage.scan_edges([&](const Edge& e) {
        if (mark[e.u] != mark[e.v]) {
            out.write_uint32(e.u);
            out.write_space();
            out.write_uint32(e.v);
            out.write_newline();
        }
    });
    out.flush();
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    EdgeStorage storage = read_graph();

    if (storage.n <= 1 || storage.m == 0) {
        FastOutput out;
        out.write_uint64(0);
        out.write_newline();
        return 0;
    }

    CutResult best;
    best.value = storage.min_degree;
    best.side = {storage.min_degree_vertex};

    if (best.value <= 1) {
        output_cut(storage, best.side, best.value);
        return 0;
    }

    if (storage.in_memory && exact_feasible(storage.n, storage.m)) {
        CutResult exact = stoer_wagner_sparse(make_multigraph(storage.n, storage.edges));
        const uint64_t full_value = count_cut_edges(storage, exact.side);
        if (full_value < best.value) {
            best = std::move(exact);
            best.value = full_value;
        }
        output_cut(storage, best.side, best.value);
        return 0;
    }

    uint64_t k = 2;
    const uint64_t hard_k_cap =
        std::min<uint64_t>(best.value + 1ull, static_cast<uint64_t>(std::numeric_limits<uint32_t>::max()));
    while (k <= hard_k_cap && certificate_feasible(storage.n, k)) {
        bool used_all_edges = false;
        std::vector<Edge> cert = build_certificate(
            storage.n, storage.m, static_cast<uint32_t>(k),
            [&](auto&& fn) { storage.scan_edges(fn); },
            used_all_edges);

        CutResult cand = solve_certificate(storage.n, cert, best.value);
        const uint64_t full_value = count_cut_edges(storage, cand.side);
        if (full_value < best.value) {
            best = cand;
            best.value = full_value;
        }

        const bool deterministic = exact_feasible(storage.n, cert.size());
        if ((used_all_edges && deterministic) || (deterministic && cand.value < k && full_value == cand.value)) {
            best = std::move(cand);
            best.value = full_value;
            break;
        }

        if (best.value <= 1) {
            break;
        }

        const uint64_t next_k = std::min<uint64_t>(hard_k_cap, std::max<uint64_t>(k + 1ull, k * 2ull));
        if (next_k <= k) {
            break;
        }
        k = next_k;
    }

    output_cut(storage, best.side, best.value);
    return 0;
}
