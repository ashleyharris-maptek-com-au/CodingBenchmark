use std::cmp::Ordering;
use std::f64::consts::PI;
use std::io::{self, Read, Write};

#[derive(Clone, Copy, Default)]
struct Point {
    x: f64,
    y: f64,
}

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).unwrap();
        Self { buf: input, idx: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let mut value = 0usize;
        while self.idx < self.buf.len() && !self.buf[self.idx].is_ascii_whitespace() {
            value = value
                .saturating_mul(10)
                .saturating_add((self.buf[self.idx] - b'0') as usize);
            self.idx += 1;
        }
        Some(value)
    }
}

struct Graph {
    n: usize,
    offsets: Vec<usize>,
    adj: Vec<usize>,
}

impl Graph {
    fn from_edges(n: usize, edges: &[(usize, usize)], degrees: &[usize]) -> Self {
        let mut offsets = vec![0usize; n + 1];
        for i in 0..n {
            offsets[i + 1] = offsets[i] + degrees[i];
        }
        let mut fill = offsets[..n].to_vec();
        let mut adj = vec![0usize; offsets[n]];
        for &(u, v) in edges {
            adj[fill[u]] = v;
            fill[u] += 1;
            adj[fill[v]] = u;
            fill[v] += 1;
        }
        Self { n, offsets, adj }
    }

    fn degree(&self, node: usize) -> usize {
        self.offsets[node + 1] - self.offsets[node]
    }
}

struct LocalGraph {
    n: usize,
    offsets: Vec<usize>,
    adj: Vec<usize>,
    degrees: Vec<usize>,
    edges: Vec<(usize, usize)>,
}

impl LocalGraph {
    fn from_edges(n: usize, edges: Vec<(usize, usize)>, degrees: Vec<usize>) -> Self {
        let mut offsets = vec![0usize; n + 1];
        for i in 0..n {
            offsets[i + 1] = offsets[i] + degrees[i];
        }
        let mut fill = offsets[..n].to_vec();
        let mut adj = vec![0usize; offsets[n]];
        for &(u, v) in &edges {
            adj[fill[u]] = v;
            fill[u] += 1;
            adj[fill[v]] = u;
            fill[v] += 1;
        }
        Self {
            n,
            offsets,
            adj,
            degrees,
            edges,
        }
    }

    fn degree(&self, node: usize) -> usize {
        self.degrees[node]
    }
}

#[derive(Clone)]
struct ComponentLayout {
    nodes: Vec<usize>,
    coords: Vec<Point>,
    width: f64,
    height: f64,
    min_x: f64,
    min_y: f64,
}

fn main() {
    let mut sc = Scanner::new();
    let n = match sc.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m = sc.next_usize().unwrap_or(0);

    let mut edges = Vec::with_capacity(m);
    let mut degrees = vec![0usize; n];

    for _ in 0..m {
        let u = match sc.next_usize() {
            Some(v) => v,
            None => break,
        };
        let v = match sc.next_usize() {
            Some(v) => v,
            None => break,
        };
        if u >= n || v >= n || u == v {
            continue;
        }
        edges.push((u, v));
        degrees[u] += 1;
        degrees[v] += 1;
    }

    let graph = Graph::from_edges(n, &edges, &degrees);
    let coords = solve(&graph);

    let mut out = String::with_capacity(n * 32);
    for p in coords {
        let x = if p.x.is_finite() { p.x } else { 0.0 };
        let y = if p.y.is_finite() { p.y } else { 0.0 };
        out.push_str(&format!("{:.6} {:.6}\n", x, y));
    }
    io::stdout().write_all(out.as_bytes()).unwrap();
}

fn solve(graph: &Graph) -> Vec<Point> {
    let n = graph.n;
    if n == 0 {
        return Vec::new();
    }

    let mut visited = vec![false; n];
    let mut mapping = vec![usize::MAX; n];
    let mut queue = Vec::new();
    let mut components = Vec::new();
    let mut isolated = Vec::new();

    for start in 0..n {
        if visited[start] {
            continue;
        }
        visited[start] = true;
        queue.clear();
        queue.push(start);
        let mut head = 0usize;
        let mut nodes = Vec::new();
        while head < queue.len() {
            let u = queue[head];
            head += 1;
            nodes.push(u);
            for idx in graph.offsets[u]..graph.offsets[u + 1] {
                let v = graph.adj[idx];
                if !visited[v] {
                    visited[v] = true;
                    queue.push(v);
                }
            }
        }

        if nodes.len() == 1 && graph.degree(nodes[0]) == 0 {
            isolated.push(nodes[0]);
            continue;
        }

        for (i, &u) in nodes.iter().enumerate() {
            mapping[u] = i;
        }

        let mut local_edges = Vec::new();
        let mut local_degrees = vec![0usize; nodes.len()];
        for &u in &nodes {
            let lu = mapping[u];
            for idx in graph.offsets[u]..graph.offsets[u + 1] {
                let v = graph.adj[idx];
                if u < v {
                    let lv = mapping[v];
                    if lv != usize::MAX {
                        local_edges.push((lu, lv));
                        local_degrees[lu] += 1;
                        local_degrees[lv] += 1;
                    }
                }
            }
        }

        let local_graph = LocalGraph::from_edges(nodes.len(), local_edges, local_degrees);
        let mut coords = layout_component(&local_graph);
        normalize_layout(&mut coords, local_graph.n);
        let (min_x, min_y, width, height) = bbox(&coords);
        components.push(ComponentLayout {
            nodes,
            coords,
            width,
            height,
            min_x,
            min_y,
        });

        for &u in &components.last().unwrap().nodes {
            mapping[u] = usize::MAX;
        }
    }

    if !isolated.is_empty() {
        let coords = isolated_grid_layout(isolated.len());
        let (min_x, min_y, width, height) = bbox(&coords);
        components.push(ComponentLayout {
            nodes: isolated,
            coords,
            width,
            height,
            min_x,
            min_y,
        });
    }

    let mut result = vec![Point::default(); n];
    pack_components(&components, &mut result);
    result
}

fn layout_component(g: &LocalGraph) -> Vec<Point> {
    let n = g.n;
    let m = g.edges.len();

    if n == 1 {
        return vec![Point { x: 0.0, y: 0.0 }];
    }
    if n == 2 {
        return vec![Point { x: -1.0, y: 0.0 }, Point { x: 1.0, y: 0.0 }];
    }

    let max_degree = g.degrees.iter().copied().max().unwrap_or(0);
    let deg1_count = g.degrees.iter().filter(|&&d| d == 1).count();
    let cycle_rank = m.saturating_add(1).saturating_sub(n);

    if n <= 120 && m <= 600 {
        let layered = layout_layered(g);
        let mut candidates = Vec::with_capacity(4);
        candidates.push(layered.clone());
        candidates.push(layout_spectral(g));
        candidates.push(layout_circle_candidate(g));
        if max_degree <= 2 && deg1_count == 0 {
            if let Some(order) = cycle_order(g) {
                candidates.push(layout_circle_from_order(&order));
            }
        }
        return choose_best_candidate(g, candidates);
    }

    if m == n - 1 {
        return layout_layered(g);
    }

    if max_degree <= 2 && deg1_count == 0 {
        if let Some(order) = cycle_order(g) {
            return layout_circle_from_order(&order);
        }
    }

    if cycle_rank <= (n.max(16) / 16) && n <= 100_000 {
        return layout_layered(g);
    }

    let mut coords = layout_spectral(g);
    let smooth_iters = if n <= 256 {
        6
    } else if n <= 2_000 {
        4
    } else if n <= 20_000 {
        2
    } else {
        1
    };
    if smooth_iters > 0 {
        anchored_smooth(g, &mut coords, smooth_iters, 0.35);
    }
    coords
}

fn choose_best_candidate(g: &LocalGraph, candidates: Vec<Vec<Point>>) -> Vec<Point> {
    let mut best_idx = 0usize;
    let mut best_crossings = usize::MAX;
    let mut best_len = f64::INFINITY;

    for (i, coords) in candidates.iter().enumerate() {
        let crossings = exact_crossings(&g.edges, coords);
        let total_len = total_edge_length_sq(&g.edges, coords);
        if crossings < best_crossings
            || (crossings == best_crossings && total_len < best_len)
        {
            best_idx = i;
            best_crossings = crossings;
            best_len = total_len;
        }
    }

    candidates[best_idx].clone()
}

fn layout_layered(g: &LocalGraph) -> Vec<Point> {
    let root = pseudo_diameter_center(g);
    let (depth, parent) = bfs_tree(g, root);
    let max_depth = depth.iter().copied().max().unwrap_or(0);

    let mut children = vec![Vec::<usize>::new(); g.n];
    for v in 0..g.n {
        if parent[v] != usize::MAX {
            children[parent[v]].push(v);
        }
    }
    for kids in &mut children {
        kids.sort_by(|&a, &b| {
            g.degree(b)
                .cmp(&g.degree(a))
                .then_with(|| a.cmp(&b))
        });
    }

    let mut layers = vec![Vec::<usize>::new(); max_depth + 1];
    layers[0].push(root);
    for d in 0..max_depth {
        let mut next = Vec::new();
        for &u in &layers[d] {
            next.extend(children[u].iter().copied());
        }
        layers[d + 1] = next;
    }

    let mut pos = vec![0usize; g.n];
    for layer in &layers {
        for (i, &u) in layer.iter().enumerate() {
            pos[u] = i;
        }
    }

    let sweeps = if g.n <= 2_000 {
        4
    } else if g.n <= 50_000 {
        2
    } else {
        1
    };

    for _ in 0..sweeps {
        for d in 1..=max_depth {
            sort_layer_by_barycenter(g, &mut layers[d], &depth, d, &pos, true);
            for (i, &u) in layers[d].iter().enumerate() {
                pos[u] = i;
            }
        }
        for d in (0..max_depth).rev() {
            sort_layer_by_barycenter(g, &mut layers[d], &depth, d, &pos, false);
            for (i, &u) in layers[d].iter().enumerate() {
                pos[u] = i;
            }
        }
    }

    let mut coords = vec![Point::default(); g.n];
    let layer_gap = 1.6;
    for (d, layer) in layers.iter().enumerate() {
        let width = layer.len() as f64;
        let center = (width - 1.0) * 0.5;
        for (i, &u) in layer.iter().enumerate() {
            coords[u] = Point {
                x: i as f64 - center,
                y: d as f64 * layer_gap,
            };
        }
    }
    coords
}

fn sort_layer_by_barycenter(
    g: &LocalGraph,
    layer: &mut Vec<usize>,
    depth: &[usize],
    current_depth: usize,
    pos: &[usize],
    use_prev: bool,
) {
    layer.sort_by(|&a, &b| {
        let ka = barycenter_key(g, a, depth, current_depth, pos, use_prev);
        let kb = barycenter_key(g, b, depth, current_depth, pos, use_prev);
        ka.partial_cmp(&kb)
            .unwrap_or(Ordering::Equal)
            .then_with(|| g.degree(b).cmp(&g.degree(a)))
            .then_with(|| a.cmp(&b))
    });
}

fn barycenter_key(
    g: &LocalGraph,
    node: usize,
    depth: &[usize],
    current_depth: usize,
    pos: &[usize],
    use_prev: bool,
) -> f64 {
    let target_depth = if use_prev {
        current_depth.saturating_sub(1)
    } else {
        current_depth + 1
    };
    let mut sum = 0.0;
    let mut count = 0usize;
    for idx in g.offsets[node]..g.offsets[node + 1] {
        let v = g.adj[idx];
        if depth[v] == target_depth {
            sum += pos[v] as f64;
            count += 1;
        }
    }
    if count == 0 {
        pos[node] as f64
    } else {
        sum / count as f64
    }
}

fn bfs_tree(g: &LocalGraph, root: usize) -> (Vec<usize>, Vec<usize>) {
    let mut depth = vec![usize::MAX; g.n];
    let mut parent = vec![usize::MAX; g.n];
    let mut queue = Vec::with_capacity(g.n);
    queue.push(root);
    depth[root] = 0;
    let mut head = 0usize;
    while head < queue.len() {
        let u = queue[head];
        head += 1;
        for idx in g.offsets[u]..g.offsets[u + 1] {
            let v = g.adj[idx];
            if depth[v] == usize::MAX {
                depth[v] = depth[u] + 1;
                parent[v] = u;
                queue.push(v);
            }
        }
    }
    (depth, parent)
}

fn pseudo_diameter_center(g: &LocalGraph) -> usize {
    let mut seed = 0usize;
    for i in 1..g.n {
        if g.degree(i) > g.degree(seed) {
            seed = i;
        }
    }
    let (_, _, a) = bfs_farthest(g, seed);
    let (_, parent, b) = bfs_farthest(g, a);
    let mut path = Vec::new();
    let mut cur = b;
    path.push(cur);
    while cur != a && parent[cur] != usize::MAX {
        cur = parent[cur];
        path.push(cur);
    }
    path[path.len() / 2]
}

fn bfs_farthest(g: &LocalGraph, start: usize) -> (Vec<usize>, Vec<usize>, usize) {
    let mut dist = vec![usize::MAX; g.n];
    let mut parent = vec![usize::MAX; g.n];
    let mut queue = Vec::with_capacity(g.n);
    queue.push(start);
    dist[start] = 0;
    let mut head = 0usize;
    let mut farthest = start;
    while head < queue.len() {
        let u = queue[head];
        head += 1;
        if dist[u] > dist[farthest] || (dist[u] == dist[farthest] && g.degree(u) > g.degree(farthest))
        {
            farthest = u;
        }
        for idx in g.offsets[u]..g.offsets[u + 1] {
            let v = g.adj[idx];
            if dist[v] == usize::MAX {
                dist[v] = dist[u] + 1;
                parent[v] = u;
                queue.push(v);
            }
        }
    }
    (dist, parent, farthest)
}

fn layout_spectral(g: &LocalGraph) -> Vec<Point> {
    let n = g.n;
    let inv_sqrt_deg: Vec<f64> = g
        .degrees
        .iter()
        .map(|&d| if d == 0 { 0.0 } else { 1.0 / (d as f64).sqrt() })
        .collect();
    let mut principal: Vec<f64> = g
        .degrees
        .iter()
        .map(|&d| (d as f64).sqrt())
        .collect();
    normalize_vec(&mut principal);

    let mut q2 = seeded_vector(n, 0x9E3779B185EBCA87);
    project_out(&mut q2, &principal);
    if !normalize_with_fallback(&mut q2, 1) {
        return layout_circle_candidate(g);
    }

    let mut q3 = seeded_vector(n, 0xC2B2AE3D27D4EB4F);
    project_out(&mut q3, &principal);
    project_out(&mut q3, &q2);
    if !normalize_with_fallback_multi(&mut q3, &[&principal, &q2], 2) {
        return layout_circle_candidate(g);
    }

    let iters = if n <= 64 {
        48
    } else if n <= 500 {
        30
    } else if n <= 5_000 {
        18
    } else if n <= 50_000 {
        12
    } else {
        8
    };

    for _ in 0..iters {
        let (mut y2, mut y3) = matvec_normalized_pair(g, &q2, &q3, &inv_sqrt_deg);
        project_out(&mut y2, &principal);
        if !normalize_with_fallback(&mut y2, 3) {
            break;
        }
        project_out(&mut y3, &principal);
        project_out(&mut y3, &y2);
        if !normalize_with_fallback_multi(&mut y3, &[&principal, &y2], 4) {
            break;
        }
        q2 = y2;
        q3 = y3;
    }

    let mut coords = vec![Point::default(); n];
    for i in 0..n {
        coords[i] = Point { x: q2[i], y: q3[i] };
    }
    coords
}

fn matvec_normalized_pair(
    g: &LocalGraph,
    q2: &[f64],
    q3: &[f64],
    inv_sqrt_deg: &[f64],
) -> (Vec<f64>, Vec<f64>) {
    let n = g.n;
    let mut scaled2 = vec![0.0; n];
    let mut scaled3 = vec![0.0; n];
    for i in 0..n {
        scaled2[i] = q2[i] * inv_sqrt_deg[i];
        scaled3[i] = q3[i] * inv_sqrt_deg[i];
    }

    let mut y2 = vec![0.0; n];
    let mut y3 = vec![0.0; n];
    for u in 0..n {
        let mut sum2 = 0.0;
        let mut sum3 = 0.0;
        for idx in g.offsets[u]..g.offsets[u + 1] {
            let v = g.adj[idx];
            sum2 += scaled2[v];
            sum3 += scaled3[v];
        }
        y2[u] = sum2 * inv_sqrt_deg[u];
        y3[u] = sum3 * inv_sqrt_deg[u];
    }
    (y2, y3)
}

fn anchored_smooth(g: &LocalGraph, coords: &mut Vec<Point>, iters: usize, alpha: f64) {
    let base = coords.clone();
    let n = g.n;
    for t in 0..iters {
        let blend = alpha * (1.0 - (t as f64) / (iters as f64 + 1.0));
        let mut next = vec![Point::default(); n];
        for u in 0..n {
            if g.degree(u) == 0 {
                next[u] = base[u];
                continue;
            }
            let mut sx = 0.0;
            let mut sy = 0.0;
            for idx in g.offsets[u]..g.offsets[u + 1] {
                let v = g.adj[idx];
                sx += coords[v].x;
                sy += coords[v].y;
            }
            let inv_deg = 1.0 / g.degree(u) as f64;
            next[u] = Point {
                x: base[u].x * (1.0 - blend) + sx * inv_deg * blend,
                y: base[u].y * (1.0 - blend) + sy * inv_deg * blend,
            };
        }
        *coords = next;
    }
}

fn layout_circle_candidate(g: &LocalGraph) -> Vec<Point> {
    if let Some(order) = cycle_order(g) {
        return layout_circle_from_order(&order);
    }
    let order = bfs_order(g);
    layout_circle_from_order(&order)
}

fn layout_circle_from_order(order: &[usize]) -> Vec<Point> {
    let n = order.len();
    if n == 1 {
        return vec![Point { x: 0.0, y: 0.0 }];
    }
    let radius = (n as f64).max(2.0) / (2.0 * PI);
    let mut coords = vec![Point::default(); n];
    for (i, &node) in order.iter().enumerate() {
        let angle = 2.0 * PI * (i as f64) / (n as f64);
        coords[node] = Point {
            x: radius * angle.cos(),
            y: radius * angle.sin(),
        };
    }
    coords
}

fn bfs_order(g: &LocalGraph) -> Vec<usize> {
    let root = pseudo_diameter_center(g);
    let mut seen = vec![false; g.n];
    let mut order = Vec::with_capacity(g.n);
    let mut queue = Vec::with_capacity(g.n);
    queue.push(root);
    seen[root] = true;
    let mut head = 0usize;
    while head < queue.len() {
        let u = queue[head];
        head += 1;
        order.push(u);
        let mut nbrs = g.adj[g.offsets[u]..g.offsets[u + 1]].to_vec();
        nbrs.sort_by(|&a, &b| {
            g.degree(a)
                .cmp(&g.degree(b))
                .then_with(|| a.cmp(&b))
        });
        for v in nbrs {
            if !seen[v] {
                seen[v] = true;
                queue.push(v);
            }
        }
    }
    order
}

fn cycle_order(g: &LocalGraph) -> Option<Vec<usize>> {
    if g.n < 3 || g.degrees.iter().any(|&d| d != 2) {
        return None;
    }
    let mut visited = vec![false; g.n];
    let mut order = Vec::with_capacity(g.n);
    let mut prev = usize::MAX;
    let mut cur = 0usize;
    visited[cur] = true;
    order.push(cur);

    for _ in 1..g.n {
        let neighbors = unique_neighbors(g, cur);
        let mut next = usize::MAX;
        for &v in &neighbors {
            if v != prev && !visited[v] {
                next = v;
                break;
            }
        }
        if next == usize::MAX {
            return None;
        }
        visited[next] = true;
        order.push(next);
        prev = cur;
        cur = next;
    }

    let last_neighbors = unique_neighbors(g, cur);
    if last_neighbors.contains(&order[0]) {
        Some(order)
    } else {
        None
    }
}

fn unique_neighbors(g: &LocalGraph, node: usize) -> Vec<usize> {
    let mut res = Vec::new();
    for idx in g.offsets[node]..g.offsets[node + 1] {
        let v = g.adj[idx];
        if !res.contains(&v) {
            res.push(v);
        }
    }
    res
}

fn exact_crossings(edges: &[(usize, usize)], coords: &[Point]) -> usize {
    let mut count = 0usize;
    for i in 0..edges.len() {
        let (a, b) = edges[i];
        for j in i + 1..edges.len() {
            let (c, d) = edges[j];
            if a == c || a == d || b == c || b == d {
                continue;
            }
            if proper_intersection(coords[a], coords[b], coords[c], coords[d]) {
                count += 1;
            }
        }
    }
    count
}

fn proper_intersection(a: Point, b: Point, c: Point, d: Point) -> bool {
    let eps = 1e-12;
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);
    if o1 * o2 < -eps && o3 * o4 < -eps {
        return true;
    }
    if o1.abs() <= eps && on_segment(a, b, c) {
        return true;
    }
    if o2.abs() <= eps && on_segment(a, b, d) {
        return true;
    }
    if o3.abs() <= eps && on_segment(c, d, a) {
        return true;
    }
    if o4.abs() <= eps && on_segment(c, d, b) {
        return true;
    }
    false
}

fn orient(a: Point, b: Point, c: Point) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn on_segment(a: Point, b: Point, p: Point) -> bool {
    let min_x = a.x.min(b.x) - 1e-12;
    let max_x = a.x.max(b.x) + 1e-12;
    let min_y = a.y.min(b.y) - 1e-12;
    let max_y = a.y.max(b.y) + 1e-12;
    p.x >= min_x && p.x <= max_x && p.y >= min_y && p.y <= max_y
}

fn total_edge_length_sq(edges: &[(usize, usize)], coords: &[Point]) -> f64 {
    let mut sum = 0.0;
    for &(u, v) in edges {
        let dx = coords[u].x - coords[v].x;
        let dy = coords[u].y - coords[v].y;
        sum += dx * dx + dy * dy;
    }
    sum
}

fn normalize_layout(coords: &mut [Point], n: usize) {
    if coords.is_empty() {
        return;
    }
    let mut mean_x = 0.0;
    let mut mean_y = 0.0;
    for p in coords.iter() {
        mean_x += p.x;
        mean_y += p.y;
    }
    mean_x /= coords.len() as f64;
    mean_y /= coords.len() as f64;
    for (i, p) in coords.iter_mut().enumerate() {
        p.x -= mean_x;
        p.y -= mean_y;
        let jitter = deterministic_jitter(i as u64) * 1e-6;
        p.x += jitter;
        p.y -= jitter;
    }

    let (_, _, width, height) = bbox(coords);
    let extent = width.max(height);
    let target = (n as f64).sqrt().max(2.0) * 1.4;
    let scale = if extent > 1e-12 { target / extent } else { 1.0 };
    for p in coords.iter_mut() {
        p.x *= scale;
        p.y *= scale;
    }
}

fn deterministic_jitter(x: u64) -> f64 {
    let mut z = x.wrapping_add(0x9E3779B97F4A7C15);
    z = (z ^ (z >> 30)).wrapping_mul(0xBF58476D1CE4E5B9);
    z = (z ^ (z >> 27)).wrapping_mul(0x94D049BB133111EB);
    z ^= z >> 31;
    let value = ((z >> 11) as f64) / ((1u64 << 53) as f64);
    value - 0.5
}

fn bbox(coords: &[Point]) -> (f64, f64, f64, f64) {
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for p in coords {
        min_x = min_x.min(p.x);
        max_x = max_x.max(p.x);
        min_y = min_y.min(p.y);
        max_y = max_y.max(p.y);
    }
    let width = (max_x - min_x).max(0.0);
    let height = (max_y - min_y).max(0.0);
    (min_x, min_y, width, height)
}

fn isolated_grid_layout(count: usize) -> Vec<Point> {
    let cols = (count as f64).sqrt().ceil() as usize;
    let mut coords = vec![Point::default(); count];
    for i in 0..count {
        coords[i] = Point {
            x: (i % cols) as f64,
            y: (i / cols) as f64,
        };
    }
    coords
}

fn pack_components(components: &[ComponentLayout], out: &mut [Point]) {
    if components.is_empty() {
        return;
    }
    let gap = 4.0;
    let mut order: Vec<usize> = (0..components.len()).collect();
    order.sort_by(|&a, &b| {
        components[b]
            .height
            .partial_cmp(&components[a].height)
            .unwrap_or(Ordering::Equal)
    });

    let total_area: f64 = components
        .iter()
        .map(|c| (c.width + gap) * (c.height + gap))
        .sum();
    let target_row_width = total_area.sqrt().max(10.0);

    let mut cursor_x = 0.0;
    let mut cursor_y = 0.0;
    let mut row_height = 0.0;

    for idx in order {
        let comp = &components[idx];
        if cursor_x > 0.0 && cursor_x + comp.width > target_row_width {
            cursor_x = 0.0;
            cursor_y += row_height + gap;
            row_height = 0.0;
        }

        let offset_x = cursor_x - comp.min_x;
        let offset_y = cursor_y - comp.min_y;
        for (local_idx, &global_node) in comp.nodes.iter().enumerate() {
            out[global_node] = Point {
                x: comp.coords[local_idx].x + offset_x,
                y: comp.coords[local_idx].y + offset_y,
            };
        }

        cursor_x += comp.width + gap;
        row_height = row_height.max(comp.height);
    }
}

fn seeded_vector(n: usize, seed: u64) -> Vec<f64> {
    let mut v = vec![0.0; n];
    for i in 0..n {
        let a = deterministic_jitter(seed.wrapping_add(i as u64 * 2 + 1));
        let b = deterministic_jitter(seed.rotate_left(17).wrapping_add(i as u64 * 2 + 7));
        v[i] = a + 0.5 * b;
    }
    v
}

fn project_out(v: &mut [f64], basis: &[f64]) {
    let dot = dot(v, basis);
    for i in 0..v.len() {
        v[i] -= dot * basis[i];
    }
}

fn dot(a: &[f64], b: &[f64]) -> f64 {
    a.iter().zip(b.iter()).map(|(x, y)| x * y).sum()
}

fn normalize_vec(v: &mut [f64]) -> bool {
    let norm = dot(v, v).sqrt();
    if norm <= 1e-18 {
        return false;
    }
    for x in v.iter_mut() {
        *x /= norm;
    }
    true
}

fn normalize_with_fallback(v: &mut [f64], seed_offset: u64) -> bool {
    if normalize_vec(v) {
        return true;
    }
    for i in 0..v.len() {
        v[i] = deterministic_jitter(seed_offset.wrapping_add(i as u64 * 13));
    }
    normalize_vec(v)
}

fn normalize_with_fallback_multi(v: &mut [f64], bases: &[&[f64]], seed_offset: u64) -> bool {
    if normalize_vec(v) {
        return true;
    }
    for i in 0..v.len() {
        v[i] = deterministic_jitter(seed_offset.wrapping_add(i as u64 * 19))
            + ((i % 7) as f64) * 1e-3;
    }
    for basis in bases {
        project_out(v, basis);
    }
    normalize_vec(v)
}
