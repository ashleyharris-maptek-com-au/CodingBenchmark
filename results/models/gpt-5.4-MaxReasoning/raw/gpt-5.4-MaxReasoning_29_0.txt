using System;
using System.Buffers;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Runtime.CompilerServices;
using System.Text;
using System.Threading;
using System.Threading.Tasks;

public static class Program
{
    private const int SeedHardCap = 4096;
    private const int LocalExactSeedCap = 256;
    private const int LocalExactVertexLimit = 320;
    private const int GlobalExactVertexLimit = 1800;
    private static readonly TimeSpan TimeBudget = TimeSpan.FromSeconds(285);

    public static void Main()
    {
        var watch = Stopwatch.StartNew();
        long deadlineTicks = (long)(TimeBudget.TotalSeconds * Stopwatch.Frequency);

        var scanner = new FastScanner(Console.OpenStandardInput());
        int n = scanner.NextInt();
        if (n == int.MinValue)
        {
            return;
        }

        int m = scanner.NextInt();
        var graph = Graph.Read(scanner, n, m);

        if (graph.VertexCount == 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        int maxCore;
        int[] core = CoreDecomposition.Compute(graph, out maxCore);
        int upperBound = maxCore + 1;

        var best = new BestCliqueStore(graph.VertexCount);
        best.TryUpdate(new int[] { 0 }, 1);

        int[] ranking = BuildRanking(graph, core);
        int[] greedyScanClique = GreedyScan(graph, ranking, core);
        best.TryUpdate(greedyScanClique, greedyScanClique.Length);

        if (best.Size < upperBound && watch.ElapsedTicks < deadlineTicks)
        {
            RunParallelGreedy(graph, core, ranking, upperBound, best, watch, deadlineTicks);
        }

        if (best.Size < upperBound && watch.ElapsedTicks < deadlineTicks)
        {
            RunLocalExactNeighborhoods(graph, core, ranking, best, watch, deadlineTicks);
        }

        if (best.Size < upperBound && watch.ElapsedTicks < deadlineTicks)
        {
            RunGlobalExactIfSmall(graph, core, best, watch, deadlineTicks);
        }

        int[] answer = best.GetClique();
        Array.Sort(answer);

        var output = new StringBuilder();
        output.AppendLine(answer.Length.ToString());
        for (int i = 0; i < answer.Length; i++)
        {
            if (i > 0)
            {
                output.Append(' ');
            }

            output.Append(answer[i]);
        }

        Console.Write(output.ToString());
    }

    private static int[] BuildRanking(Graph graph, int[] core)
    {
        int n = graph.VertexCount;
        int[] order = new int[n];
        ulong[] keys = new ulong[n];
        for (int v = 0; v < n; v++)
        {
            order[v] = v;
            keys[v] = ((ulong)(uint)core[v] << 32) | (uint)graph.Degree[v];
        }

        Array.Sort(keys, order);
        return order;
    }

    private static int[] BuildSeedList(Graph graph, int[] ranking)
    {
        int n = graph.VertexCount;
        int topCount = n <= 20000 ? n : Math.Min(n, Math.Max(1024, SeedHardCap));
        var seen = new HashSet<int>();
        var seeds = new List<int>(topCount + (topCount / 2));

        for (int i = 0; i < topCount; i++)
        {
            int v = ranking[n - 1 - i];
            if (seen.Add(v))
            {
                seeds.Add(v);
            }
        }

        int spacedCount = Math.Min(n, topCount / 2);
        for (int i = 1; i <= spacedCount; i++)
        {
            int idx = n - 1 - (int)((long)i * n / (spacedCount + 1));
            if (idx < 0)
            {
                idx = 0;
            }

            int v = ranking[idx];
            if (seen.Add(v))
            {
                seeds.Add(v);
            }
        }

        return seeds.ToArray();
    }

    private static int[] GreedyScan(Graph graph, int[] ranking, int[] core)
    {
        var clique = new List<int>(16);
        for (int i = ranking.Length - 1; i >= 0; i--)
        {
            int v = ranking[i];
            if (core[v] + 1 <= clique.Count)
            {
                continue;
            }

            bool fits = true;
            for (int j = 0; j < clique.Count; j++)
            {
                if (!graph.HasEdge(v, clique[j]))
                {
                    fits = false;
                    break;
                }
            }

            if (fits)
            {
                clique.Add(v);
            }
        }

        return clique.ToArray();
    }

    private static void RunParallelGreedy(Graph graph, int[] core, int[] ranking, int upperBound, BestCliqueStore best, Stopwatch watch, long deadlineTicks)
    {
        int[] seeds = BuildSeedList(graph, ranking);
        var options = new ParallelOptions
        {
            MaxDegreeOfParallelism = Math.Max(1, Environment.ProcessorCount)
        };

        Parallel.ForEach(seeds, options, (seed, state) =>
        {
            if (watch.ElapsedTicks >= deadlineTicks || best.Size >= upperBound)
            {
                state.Stop();
                return;
            }

            if (core[seed] < best.Size)
            {
                return;
            }

            int[] clique = GrowCliqueGreedy(graph, core, seed, best, watch, deadlineTicks);
            best.TryUpdate(clique, clique.Length);

            if (best.Size >= upperBound || watch.ElapsedTicks >= deadlineTicks)
            {
                state.Stop();
            }
        });
    }

    private static int[] GrowCliqueGreedy(Graph graph, int[] core, int seed, BestCliqueStore best, Stopwatch watch, long deadlineTicks)
    {
        int degree = graph.Degree[seed];
        var clique = new List<int>(16) { seed };

        if (degree == 0)
        {
            return clique.ToArray();
        }

        int[] current = ArrayPool<int>.Shared.Rent(Math.Max(4, degree));
        int currentCount = 0;
        int minCore = best.Size;

        int start = graph.Offsets[seed];
        int end = graph.Offsets[seed + 1];
        for (int i = start; i < end; i++)
        {
            int u = graph.Adjacency[i];
            if (core[u] >= minCore)
            {
                current[currentCount++] = u;
            }
        }

        try
        {
            while (currentCount > 0)
            {
                if (watch.ElapsedTicks >= deadlineTicks)
                {
                    break;
                }

                int bestSize = best.Size;
                if (clique.Count + currentCount <= bestSize)
                {
                    break;
                }

                int chosen = -1;
                int chosenCore = -1;
                int chosenDegree = -1;
                for (int i = 0; i < currentCount; i++)
                {
                    int v = current[i];
                    int cv = core[v];
                    if (cv < bestSize)
                    {
                        continue;
                    }

                    int dv = graph.Degree[v];
                    if (cv > chosenCore || (cv == chosenCore && dv > chosenDegree) || (cv == chosenCore && dv == chosenDegree && v < chosen))
                    {
                        chosen = v;
                        chosenCore = cv;
                        chosenDegree = dv;
                    }
                }

                if (chosen < 0)
                {
                    break;
                }

                clique.Add(chosen);
                best.TryUpdate(clique.ToArray(), clique.Count);

                int[] next = ArrayPool<int>.Shared.Rent(Math.Max(4, currentCount));
                int nextCount = IntersectFiltered(current, currentCount, graph.Adjacency, graph.Offsets[chosen], graph.Offsets[chosen + 1], next, core, best.Size);
                ArrayPool<int>.Shared.Return(current);
                current = next;
                currentCount = nextCount;
            }

            return clique.ToArray();
        }
        finally
        {
            ArrayPool<int>.Shared.Return(current);
        }
    }

    private static int IntersectFiltered(int[] left, int leftCount, int[] rightArray, int rightStart, int rightEnd, int[] output, int[] core, int minCore)
    {
        int i = 0;
        int j = rightStart;
        int count = 0;
        while (i < leftCount && j < rightEnd)
        {
            int a = left[i];
            int b = rightArray[j];
            if (a < b)
            {
                i++;
            }
            else if (a > b)
            {
                j++;
            }
            else
            {
                if (core[a] >= minCore)
                {
                    output[count++] = a;
                }

                i++;
                j++;
            }
        }

        return count;
    }

    private static void RunLocalExactNeighborhoods(Graph graph, int[] core, int[] ranking, BestCliqueStore best, Stopwatch watch, long deadlineTicks)
    {
        int n = ranking.Length;
        int limit = Math.Min(LocalExactSeedCap, n);

        for (int i = 0; i < limit; i++)
        {
            if (watch.ElapsedTicks >= deadlineTicks)
            {
                return;
            }

            int seed = ranking[n - 1 - i];
            int globalBest = best.Size;
            if (core[seed] < globalBest)
            {
                continue;
            }

            int start = graph.Offsets[seed];
            int end = graph.Offsets[seed + 1];
            int degree = end - start;
            if (degree == 0 || degree + 1 <= globalBest || degree > 20000)
            {
                continue;
            }

            int[] candidates = ArrayPool<int>.Shared.Rent(degree);
            int count = 0;
            for (int p = start; p < end; p++)
            {
                int u = graph.Adjacency[p];
                if (core[u] >= globalBest)
                {
                    candidates[count++] = u;
                }
            }

            if (count + 1 <= globalBest || count > LocalExactVertexLimit)
            {
                ArrayPool<int>.Shared.Return(candidates);
                continue;
            }

            int[] original = new int[count];
            Array.Copy(candidates, 0, original, 0, count);
            ArrayPool<int>.Shared.Return(candidates);

            BitGraph localGraph = BitGraph.BuildFromVertices(graph, original);
            int[] initialOrder = localGraph.BuildDegreeOrder();
            var solver = new ExactCliqueSolver(localGraph, globalBest - 1, watch, deadlineTicks);
            solver.Solve(initialOrder);

            if (solver.BestSize + 1 > best.Size)
            {
                int[] localClique = solver.GetBestOriginalVertices();
                int[] fullClique = new int[localClique.Length + 1];
                fullClique[0] = seed;
                Array.Copy(localClique, 0, fullClique, 1, localClique.Length);
                best.TryUpdate(fullClique, fullClique.Length);
            }
        }
    }

    private static void RunGlobalExactIfSmall(Graph graph, int[] core, BestCliqueStore best, Stopwatch watch, long deadlineTicks)
    {
        int minCore = best.Size;
        int count = 0;
        for (int v = 0; v < graph.VertexCount; v++)
        {
            if (core[v] >= minCore)
            {
                count++;
            }
        }

        if (count <= best.Size || count > GlobalExactVertexLimit)
        {
            return;
        }

        int[] kernel = new int[count];
        int k = 0;
        for (int v = 0; v < graph.VertexCount; v++)
        {
            if (core[v] >= minCore)
            {
                kernel[k++] = v;
            }
        }

        BitGraph kernelGraph = BitGraph.BuildFromVertices(graph, kernel);
        int[] initialOrder = kernelGraph.BuildDegreeOrder();
        var solver = new ExactCliqueSolver(kernelGraph, best.Size, watch, deadlineTicks);
        solver.Solve(initialOrder);

        if (solver.BestSize > best.Size)
        {
            int[] clique = solver.GetBestOriginalVertices();
            best.TryUpdate(clique, clique.Length);
        }
    }

    private sealed class BestCliqueStore
    {
        private readonly object _gate = new object();
        private int[] _clique = Array.Empty<int>();
        private int _size;

        public BestCliqueStore(int vertexCount)
        {
            _size = 0;
        }

        public int Size
        {
            get { return Volatile.Read(ref _size); }
        }

        public bool TryUpdate(int[] clique, int count)
        {
            if (count <= 0)
            {
                return false;
            }

            if (count <= Volatile.Read(ref _size))
            {
                return false;
            }

            lock (_gate)
            {
                if (count <= _size)
                {
                    return false;
                }

                _size = count;
                _clique = new int[count];
                Array.Copy(clique, _clique, count);
                return true;
            }
        }

        public int[] GetClique()
        {
            lock (_gate)
            {
                int[] result = new int[_clique.Length];
                Array.Copy(_clique, result, _clique.Length);
                return result;
            }
        }
    }

    private sealed class Graph
    {
        public readonly int VertexCount;
        public readonly int[] Offsets;
        public readonly int[] Adjacency;
        public readonly int[] Degree;

        private Graph(int vertexCount, int[] offsets, int[] adjacency, int[] degree)
        {
            VertexCount = vertexCount;
            Offsets = offsets;
            Adjacency = adjacency;
            Degree = degree;
        }

        public static Graph Read(FastScanner scanner, int n, int m)
        {
            int[] degree = new int[n];
            int[] us = new int[m];
            int[] vs = new int[m];
            int kept = 0;

            for (int i = 0; i < m; i++)
            {
                int u = scanner.NextInt();
                int v = scanner.NextInt();
                if ((uint)u >= (uint)n || (uint)v >= (uint)n || u == v)
                {
                    continue;
                }

                us[kept] = u;
                vs[kept] = v;
                kept++;
                degree[u]++;
                degree[v]++;
            }

            int[] offsets = new int[n + 1];
            for (int v = 0; v < n; v++)
            {
                offsets[v + 1] = offsets[v] + degree[v];
            }

            int[] cursor = new int[n];
            Array.Copy(offsets, cursor, n);
            int[] adjacency = new int[offsets[n]];
            for (int i = 0; i < kept; i++)
            {
                int u = us[i];
                int v = vs[i];
                adjacency[cursor[u]++] = v;
                adjacency[cursor[v]++] = u;
            }

            for (int v = 0; v < n; v++)
            {
                int start = offsets[v];
                int len = offsets[v + 1] - start;
                if (len > 1)
                {
                    Array.Sort(adjacency, start, len);
                }
            }

            int[] uniqueOffsets = new int[n + 1];
            int write = 0;
            for (int v = 0; v < n; v++)
            {
                uniqueOffsets[v] = write;
                int start = offsets[v];
                int end = offsets[v + 1];
                int last = -1;
                for (int p = start; p < end; p++)
                {
                    int u = adjacency[p];
                    if (u != v && u != last)
                    {
                        adjacency[write++] = u;
                        last = u;
                    }
                }

                degree[v] = write - uniqueOffsets[v];
            }

            uniqueOffsets[n] = write;
            if (write != adjacency.Length)
            {
                Array.Resize(ref adjacency, write);
            }

            return new Graph(n, uniqueOffsets, adjacency, degree);
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public bool HasEdge(int a, int b)
        {
            if (a == b)
            {
                return false;
            }

            if (Degree[a] > Degree[b])
            {
                int tmp = a;
                a = b;
                b = tmp;
            }

            int lo = Offsets[a];
            int hi = Offsets[a + 1] - 1;
            while (lo <= hi)
            {
                int mid = lo + ((hi - lo) >> 1);
                int value = Adjacency[mid];
                if (value == b)
                {
                    return true;
                }

                if (value < b)
                {
                    lo = mid + 1;
                }
                else
                {
                    hi = mid - 1;
                }
            }

            return false;
        }
    }

    private static class CoreDecomposition
    {
        public static int[] Compute(Graph graph, out int maxCore)
        {
            int n = graph.VertexCount;
            int[] degree = new int[n];
            Array.Copy(graph.Degree, degree, n);

            int maxDegree = 0;
            for (int i = 0; i < n; i++)
            {
                if (degree[i] > maxDegree)
                {
                    maxDegree = degree[i];
                }
            }

            int[] bin = new int[maxDegree + 1];
            for (int i = 0; i < n; i++)
            {
                bin[degree[i]]++;
            }

            int start = 0;
            for (int d = 0; d <= maxDegree; d++)
            {
                int num = bin[d];
                bin[d] = start;
                start += num;
            }

            int[] pos = new int[n];
            int[] vert = new int[n];
            for (int v = 0; v < n; v++)
            {
                pos[v] = bin[degree[v]];
                vert[pos[v]] = v;
                bin[degree[v]]++;
            }

            for (int d = maxDegree; d >= 1; d--)
            {
                bin[d] = bin[d - 1];
            }

            bin[0] = 0;

            for (int i = 0; i < n; i++)
            {
                int v = vert[i];
                int dv = degree[v];
                for (int p = graph.Offsets[v]; p < graph.Offsets[v + 1]; p++)
                {
                    int u = graph.Adjacency[p];
                    if (degree[u] > dv)
                    {
                        int du = degree[u];
                        int pu = pos[u];
                        int pw = bin[du];
                        int w = vert[pw];
                        if (u != w)
                        {
                            vert[pu] = w;
                            vert[pw] = u;
                            pos[u] = pw;
                            pos[w] = pu;
                        }

                        bin[du]++;
                        degree[u]--;
                    }
                }
            }

            maxCore = 0;
            for (int i = 0; i < n; i++)
            {
                if (degree[i] > maxCore)
                {
                    maxCore = degree[i];
                }
            }

            return degree;
        }
    }

    private sealed class BitGraph
    {
        private readonly int _wordCount;
        private readonly ulong[][] _bits;

        public readonly int Count;
        public readonly int[] OriginalVertices;
        public readonly int[] Degree;

        private BitGraph(int count, int[] originalVertices, ulong[][] bits, int[] degree)
        {
            Count = count;
            OriginalVertices = originalVertices;
            Degree = degree;
            _bits = bits;
            _wordCount = (count + 63) >> 6;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public bool AreAdjacent(int a, int b)
        {
            return (_bits[a][b >> 6] & (1UL << (b & 63))) != 0UL;
        }

        public int[] BuildDegreeOrder()
        {
            int[] order = new int[Count];
            ulong[] keys = new ulong[Count];
            for (int i = 0; i < Count; i++)
            {
                order[i] = i;
                keys[i] = ((ulong)(uint)Degree[i] << 32) | (uint)(Count - 1 - i);
            }

            Array.Sort(keys, order);
            Array.Reverse(order);
            return order;
        }

        public static BitGraph BuildFromVertices(Graph graph, int[] vertices)
        {
            int count = vertices.Length;
            int words = (count + 63) >> 6;
            ulong[][] bits = new ulong[count][];
            int[] degree = new int[count];
            for (int i = 0; i < count; i++)
            {
                bits[i] = new ulong[words];
            }

            if (count == 0)
            {
                return new BitGraph(0, vertices, bits, degree);
            }

            if (count <= 4096)
            {
                var map = new Dictionary<int, int>(count * 2);
                for (int i = 0; i < count; i++)
                {
                    map[vertices[i]] = i;
                }

                for (int i = 0; i < count; i++)
                {
                    int v = vertices[i];
                    for (int p = graph.Offsets[v]; p < graph.Offsets[v + 1]; p++)
                    {
                        int u = graph.Adjacency[p];
                        int j;
                        if (map.TryGetValue(u, out j))
                        {
                            bits[i][j >> 6] |= 1UL << (j & 63);
                            degree[i]++;
                        }
                    }
                }
            }
            else
            {
                int[] map = new int[graph.VertexCount];
                Array.Fill(map, -1);
                for (int i = 0; i < count; i++)
                {
                    map[vertices[i]] = i;
                }

                for (int i = 0; i < count; i++)
                {
                    int v = vertices[i];
                    for (int p = graph.Offsets[v]; p < graph.Offsets[v + 1]; p++)
                    {
                        int u = graph.Adjacency[p];
                        int j = map[u];
                        if (j >= 0)
                        {
                            bits[i][j >> 6] |= 1UL << (j & 63);
                            degree[i]++;
                        }
                    }
                }
            }

            return new BitGraph(count, vertices, bits, degree);
        }
    }

    private sealed class ExactCliqueSolver
    {
        private readonly BitGraph _graph;
        private readonly Stopwatch _watch;
        private readonly long _deadlineTicks;
        private readonly int[] _current;
        private int[] _bestLocalIndices;
        private int _nodeCounter;

        public int BestSize { get; private set; }
        public bool TimedOut { get; private set; }

        public ExactCliqueSolver(BitGraph graph, int initialBestSize, Stopwatch watch, long deadlineTicks)
        {
            _graph = graph;
            _watch = watch;
            _deadlineTicks = deadlineTicks;
            _current = new int[graph.Count];
            _bestLocalIndices = Array.Empty<int>();
            BestSize = Math.Max(0, initialBestSize);
        }

        public void Solve(int[] initialOrder)
        {
            if (BestSize >= _graph.Count)
            {
                return;
            }

            Expand(initialOrder, initialOrder.Length, 0);
        }

        public int[] GetBestOriginalVertices()
        {
            int[] result = new int[_bestLocalIndices.Length];
            for (int i = 0; i < _bestLocalIndices.Length; i++)
            {
                result[i] = _graph.OriginalVertices[_bestLocalIndices[i]];
            }

            return result;
        }

        private void Expand(int[] candidates, int candidateCount, int depth)
        {
            if (TimedOut || depth + candidateCount <= BestSize)
            {
                return;
            }

            _nodeCounter++;
            if ((_nodeCounter & 2047) == 0 && _watch.ElapsedTicks >= _deadlineTicks)
            {
                TimedOut = true;
                return;
            }

            if (candidateCount == 0)
            {
                if (depth > BestSize)
                {
                    SaveCurrent(depth);
                }

                return;
            }

            int[] order = ArrayPool<int>.Shared.Rent(candidateCount);
            int[] colors = ArrayPool<int>.Shared.Rent(candidateCount);
            try
            {
                ColorSort(candidates, candidateCount, order, colors);
                for (int i = candidateCount - 1; i >= 0; i--)
                {
                    if (depth + colors[i] <= BestSize)
                    {
                        return;
                    }

                    int v = order[i];
                    _current[depth] = v;

                    if (i == 0)
                    {
                        if (depth + 1 > BestSize)
                        {
                            SaveCurrent(depth + 1);
                        }

                        continue;
                    }

                    int[] next = ArrayPool<int>.Shared.Rent(i);
                    int nextCount = 0;
                    for (int j = 0; j < i; j++)
                    {
                        int u = order[j];
                        if (_graph.AreAdjacent(v, u))
                        {
                            next[nextCount++] = u;
                        }
                    }

                    if (nextCount == 0)
                    {
                        if (depth + 1 > BestSize)
                        {
                            SaveCurrent(depth + 1);
                        }
                    }
                    else
                    {
                        Expand(next, nextCount, depth + 1);
                    }

                    ArrayPool<int>.Shared.Return(next);
                    if (TimedOut)
                    {
                        return;
                    }
                }
            }
            finally
            {
                ArrayPool<int>.Shared.Return(order);
                ArrayPool<int>.Shared.Return(colors);
            }
        }

        private void ColorSort(int[] candidates, int candidateCount, int[] order, int[] colors)
        {
            int[] work = ArrayPool<int>.Shared.Rent(candidateCount);
            int[] colorClass = ArrayPool<int>.Shared.Rent(candidateCount);
            try
            {
                Array.Copy(candidates, 0, work, 0, candidateCount);
                int workCount = candidateCount;
                int outCount = 0;
                int color = 0;

                while (workCount > 0)
                {
                    color++;
                    int classCount = 0;
                    int write = 0;
                    for (int i = 0; i < workCount; i++)
                    {
                        int v = work[i];
                        bool fits = true;
                        for (int j = 0; j < classCount; j++)
                        {
                            if (_graph.AreAdjacent(v, colorClass[j]))
                            {
                                fits = false;
                                break;
                            }
                        }

                        if (fits)
                        {
                            colorClass[classCount++] = v;
                            order[outCount] = v;
                            colors[outCount] = color;
                            outCount++;
                        }
                        else
                        {
                            work[write++] = v;
                        }
                    }

                    workCount = write;
                }
            }
            finally
            {
                ArrayPool<int>.Shared.Return(work);
                ArrayPool<int>.Shared.Return(colorClass);
            }
        }

        private void SaveCurrent(int depth)
        {
            BestSize = depth;
            _bestLocalIndices = new int[depth];
            Array.Copy(_current, 0, _bestLocalIndices, 0, depth);
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _length;
        private int _index;

        public FastScanner(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 16];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c < 0)
                {
                    return int.MinValue;
                }
            }
            while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (_index >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _index = 0;
                if (_length <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_index++];
        }
    }
