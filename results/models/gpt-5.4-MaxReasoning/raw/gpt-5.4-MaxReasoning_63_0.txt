declare const require: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Grid = string[];
type RobotInput = { id: number; x: number; y: number };
type JobInput = { id: unknown; x: number; y: number; priority: number };

interface JobInfo {
  id: unknown;
  x: number;
  y: number;
  priority: number;
  cell: number;
  component: number;
}

interface RobotState {
  id: number;
  startCell: number;
  cell: number;
  component: number;
  assigned: number[];
  candidateJob: number;
  candidateDist: number;
}

interface CandidateResult {
  job: number;
  dist: number;
}

const grid: Grid = Array.isArray(data.grid) ? data.grid : [];
const robotInputs: RobotInput[] = Array.isArray(data.robots) ? data.robots : [];
const jobInputs: JobInput[] = Array.isArray(data.jobs) ? data.jobs : [];

let width = 0;
for (const row of grid) {
  if (typeof row === 'string' && row.length > width) {
    width = row.length;
  }
}
const height = grid.length;
const cellCount = width * height;

const open = new Uint8Array(cellCount);
for (let y = 0; y < height; y++) {
  const row = typeof grid[y] === 'string' ? grid[y] : '';
  for (let x = 0; x < width; x++) {
    const cell = row[x];
    if (cell !== '#') {
      open[y * width + x] = 1;
    }
  }
}

function toInt(value: unknown): number {
  const n = Number(value);
  return Number.isFinite(n) ? Math.trunc(n) : -1;
}

function toCell(x: number, y: number): number {
  if (!Number.isFinite(x) || !Number.isFinite(y)) {
    return -1;
  }
  const ix = Math.trunc(x);
  const iy = Math.trunc(y);
  if (ix < 0 || iy < 0 || ix >= width || iy >= height) {
    return -1;
  }
  return iy * width + ix;
}

function compareGenericIds(a: unknown, b: unknown): number {
  if (typeof a === 'number' && typeof b === 'number') {
    return a - b;
  }
  const sa = String(JSON.stringify(a));
  const sb = String(JSON.stringify(b));
  if (sa < sb) {
    return -1;
  }
  if (sa > sb) {
    return 1;
  }
  return 0;
}

const componentOf = new Int32Array(cellCount);
componentOf.fill(-1);
const floodQueue = new Int32Array(Math.max(1, cellCount));

let componentCount = 0;
for (let start = 0; start < cellCount; start++) {
  if (!open[start] || componentOf[start] !== -1) {
    continue;
  }
  let head = 0;
  let tail = 0;
  floodQueue[tail++] = start;
  componentOf[start] = componentCount;

  while (head < tail) {
    const cell = floodQueue[head++];
    const x = cell % width;

    const left = cell - 1;
    if (x > 0 && open[left] && componentOf[left] === -1) {
      componentOf[left] = componentCount;
      floodQueue[tail++] = left;
    }

    const right = cell + 1;
    if (x + 1 < width && open[right] && componentOf[right] === -1) {
      componentOf[right] = componentCount;
      floodQueue[tail++] = right;
    }

    const up = cell - width;
    if (up >= 0 && open[up] && componentOf[up] === -1) {
      componentOf[up] = componentCount;
      floodQueue[tail++] = up;
    }

    const down = cell + width;
    if (down < cellCount && open[down] && componentOf[down] === -1) {
      componentOf[down] = componentCount;
      floodQueue[tail++] = down;
    }
  }

  componentCount++;
}

const jobs: JobInfo[] = jobInputs.map((job) => {
  const x = toInt(job.x);
  const y = toInt(job.y);
  const cell = toCell(x, y);
  const component = cell >= 0 && open[cell] ? componentOf[cell] : -1;
  return {
    id: job.id,
    x,
    y,
    priority: Number.isFinite(Number(job.priority)) ? Number(job.priority) : 0,
    cell,
    component,
  };
});

const jobsAtCell: Array<number[] | undefined> = new Array(cellCount);
const remainingJobsInComponent: number[] = Array(componentCount).fill(0);
for (let i = 0; i < jobs.length; i++) {
  const job = jobs[i];
  if (job.cell >= 0) {
    let list = jobsAtCell[job.cell];
    if (!list) {
      list = [];
      jobsAtCell[job.cell] = list;
    }
    list.push(i);
  }
  if (job.component >= 0) {
    remainingJobsInComponent[job.component]++;
  }
}

function compareJobPreference(aIndex: number, bIndex: number): number {
  const a = jobs[aIndex];
  const b = jobs[bIndex];
  if (a.priority !== b.priority) {
    return b.priority - a.priority;
  }
  const idCmp = compareGenericIds(a.id, b.id);
  if (idCmp !== 0) {
    return idCmp;
  }
  if (a.y !== b.y) {
    return a.y - b.y;
  }
  if (a.x !== b.x) {
    return a.x - b.x;
  }
  return aIndex - bIndex;
}

const robots: RobotState[] = [];
let maxRobotId = -1;
for (const robot of robotInputs) {
  const id = toInt(robot.id);
  if (id < 0) {
    continue;
  }
  const x = toInt(robot.x);
  const y = toInt(robot.y);
  const cell = toCell(x, y);
  const component = cell >= 0 && open[cell] ? componentOf[cell] : -1;
  robots.push({
    id,
    startCell: cell,
    cell,
    component,
    assigned: [],
    candidateJob: -1,
    candidateDist: Number.POSITIVE_INFINITY,
  });
  if (id > maxRobotId) {
    maxRobotId = id;
  }
}

robots.sort((a, b) => a.id - b.id);

const assignedGlobal = new Uint8Array(jobs.length);
const bfsQueue = new Int32Array(Math.max(1, cellCount));
const bfsSeen = new Int32Array(Math.max(1, cellCount));
const bfsDist = new Int32Array(Math.max(1, cellCount));
let bfsStamp = 1;

function nextBfsStamp(): number {
  bfsStamp++;
  if (bfsStamp === 0x7fffffff) {
    bfsSeen.fill(0);
    bfsStamp = 1;
  }
  return bfsStamp;
}

function bestRemainingGlobalJobAtCell(cell: number): number {
  const list = jobsAtCell[cell];
  if (!list) {
    return -1;
  }
  let best = -1;
  for (let i = 0; i < list.length; i++) {
    const jobIndex = list[i];
    if (assignedGlobal[jobIndex]) {
      continue;
    }
    if (best === -1 || compareJobPreference(jobIndex, best) < 0) {
      best = jobIndex;
    }
  }
  return best;
}

function findNearestGlobalJob(startCell: number, component: number): CandidateResult | null {
  if (startCell < 0 || component < 0 || remainingJobsInComponent[component] === 0) {
    return null;
  }

  const stamp = nextBfsStamp();
  let head = 0;
  let tail = 0;
  bfsQueue[tail++] = startCell;
  bfsSeen[startCell] = stamp;
  bfsDist[startCell] = 0;

  let bestJob = -1;
  let bestDist = -1;

  while (head < tail) {
    const cell = bfsQueue[head++];
    const dist = bfsDist[cell];

    if (bestDist !== -1 && dist > bestDist) {
      break;
    }

    const cellJob = bestRemainingGlobalJobAtCell(cell);
    if (cellJob !== -1) {
      bestDist = dist;
      if (bestJob === -1 || compareJobPreference(cellJob, bestJob) < 0) {
        bestJob = cellJob;
      }
      continue;
    }

    if (bestDist !== -1) {
      continue;
    }

    const x = cell % width;

    const left = cell - 1;
    if (x > 0 && open[left] && bfsSeen[left] !== stamp) {
      bfsSeen[left] = stamp;
      bfsDist[left] = dist + 1;
      bfsQueue[tail++] = left;
    }

    const right = cell + 1;
    if (x + 1 < width && open[right] && bfsSeen[right] !== stamp) {
      bfsSeen[right] = stamp;
      bfsDist[right] = dist + 1;
      bfsQueue[tail++] = right;
    }

    const up = cell - width;
    if (up >= 0 && open[up] && bfsSeen[up] !== stamp) {
      bfsSeen[up] = stamp;
      bfsDist[up] = dist + 1;
      bfsQueue[tail++] = up;
    }

    const down = cell + width;
    if (down < cellCount && open[down] && bfsSeen[down] !== stamp) {
      bfsSeen[down] = stamp;
      bfsDist[down] = dist + 1;
      bfsQueue[tail++] = down;
    }
  }

  return bestJob === -1 ? null : { job: bestJob, dist: bestDist };
}

function recomputeCandidate(robot: RobotState): void {
  const result = findNearestGlobalJob(robot.cell, robot.component);
  if (!result) {
    robot.candidateJob = -1;
    robot.candidateDist = Number.POSITIVE_INFINITY;
    return;
  }
  robot.candidateJob = result.job;
  robot.candidateDist = result.dist;
}

function compareRobotCandidates(a: RobotState, b: RobotState): number {
  if (a.candidateDist !== b.candidateDist) {
    return a.candidateDist - b.candidateDist;
  }
  const aJob = jobs[a.candidateJob];
  const bJob = jobs[b.candidateJob];
  if (aJob.priority !== bJob.priority) {
    return bJob.priority - aJob.priority;
  }
  if (a.id !== b.id) {
    return a.id - b.id;
  }
  const idCmp = compareGenericIds(aJob.id, bJob.id);
  if (idCmp !== 0) {
    return idCmp;
  }
  return a.candidateJob - b.candidateJob;
}

for (let i = 0; i < robots.length; i++) {
  recomputeCandidate(robots[i]);
}

let assignedCount = 0;
while (assignedCount < jobs.length) {
  let bestRobot = -1;
  for (let i = 0; i < robots.length; i++) {
    if (robots[i].candidateJob === -1) {
      continue;
    }
    if (bestRobot === -1 || compareRobotCandidates(robots[i], robots[bestRobot]) < 0) {
      bestRobot = i;
    }
  }

  if (bestRobot === -1) {
    break;
  }

  const robot = robots[bestRobot];
  const jobIndex = robot.candidateJob;

  if (jobIndex < 0 || assignedGlobal[jobIndex]) {
    recomputeCandidate(robot);
    continue;
  }

  assignedGlobal[jobIndex] = 1;
  assignedCount++;
  robot.assigned.push(jobIndex);

  const job = jobs[jobIndex];
  if (job.component >= 0) {
    remainingJobsInComponent[job.component]--;
  }
  robot.cell = job.cell;
  robot.component = job.component;

  recomputeCandidate(robot);
  for (let i = 0; i < robots.length; i++) {
    if (i !== bestRobot && robots[i].candidateJob === jobIndex) {
      recomputeCandidate(robots[i]);
    }
  }
}

if (assignedCount < jobs.length && robots.length > 0) {
  const fallbackRobot = robots[0];
  const leftovers: number[] = [];
  for (let i = 0; i < jobs.length; i++) {
    if (!assignedGlobal[i]) {
      leftovers.push(i);
    }
  }
  leftovers.sort(compareJobPreference);
  for (let i = 0; i < leftovers.length; i++) {
    fallbackRobot.assigned.push(leftovers[i]);
  }
}

function orderJobsForRobot(startCell: number, assignedJobs: number[]): number[] {
  if (assignedJobs.length <= 1) {
    return assignedJobs.slice();
  }

  const jobsByCell = new Map<number, number[]>();
  const localIndex = new Map<number, number>();
  for (let i = 0; i < assignedJobs.length; i++) {
    const jobIndex = assignedJobs[i];
    localIndex.set(jobIndex, i);
    const cell = jobs[jobIndex].cell;
    const existing = jobsByCell.get(cell);
    if (existing) {
      existing.push(jobIndex);
    } else {
      jobsByCell.set(cell, [jobIndex]);
    }
  }

  const used = new Uint8Array(assignedJobs.length);
  let remaining = assignedJobs.length;
  let currentCell = startCell;
  const ordered: number[] = [];

  const bestRemainingRouteJobAtCell = (cell: number): number => {
    const list = jobsByCell.get(cell);
    if (!list) {
      return -1;
    }
    let best = -1;
    for (let i = 0; i < list.length; i++) {
      const jobIndex = list[i];
      const local = localIndex.get(jobIndex);
      if (local === undefined || used[local]) {
        continue;
      }
      if (best === -1 || compareJobPreference(jobIndex, best) < 0) {
        best = jobIndex;
      }
    }
    return best;
  };

  const findNearestRouteJob = (start: number): CandidateResult | null => {
    if (start < 0) {
      return null;
    }

    const stamp = nextBfsStamp();
    let head = 0;
    let tail = 0;
    bfsQueue[tail++] = start;
    bfsSeen[start] = stamp;
    bfsDist[start] = 0;

    let bestJob = -1;
    let bestDist = -1;

    while (head < tail) {
      const cell = bfsQueue[head++];
      const dist = bfsDist[cell];

      if (bestDist !== -1 && dist > bestDist) {
        break;
      }

      const cellJob = bestRemainingRouteJobAtCell(cell);
      if (cellJob !== -1) {
        bestDist = dist;
        if (bestJob === -1 || compareJobPreference(cellJob, bestJob) < 0) {
          bestJob = cellJob;
        }
        continue;
      }

      if (bestDist !== -1) {
        continue;
      }

      const x = cell % width;

      const left = cell - 1;
      if (x > 0 && open[left] && bfsSeen[left] !== stamp) {
        bfsSeen[left] = stamp;
        bfsDist[left] = dist + 1;
        bfsQueue[tail++] = left;
      }

      const right = cell + 1;
      if (x + 1 < width && open[right] && bfsSeen[right] !== stamp) {
        bfsSeen[right] = stamp;
        bfsDist[right] = dist + 1;
        bfsQueue[tail++] = right;
      }

      const up = cell - width;
      if (up >= 0 && open[up] && bfsSeen[up] !== stamp) {
        bfsSeen[up] = stamp;
        bfsDist[up] = dist + 1;
        bfsQueue[tail++] = up;
      }

      const down = cell + width;
      if (down < cellCount && open[down] && bfsSeen[down] !== stamp) {
        bfsSeen[down] = stamp;
        bfsDist[down] = dist + 1;
        bfsQueue[tail++] = down;
      }
    }

    return bestJob === -1 ? null : { job: bestJob, dist: bestDist };
  };

  while (remaining > 0) {
    const next = findNearestRouteJob(currentCell);
    if (!next) {
      const leftovers: number[] = [];
      for (let i = 0; i < assignedJobs.length; i++) {
        if (!used[i]) {
          leftovers.push(assignedJobs[i]);
        }
      }
      leftovers.sort(compareJobPreference);
      for (let i = 0; i < leftovers.length; i++) {
        const jobIndex = leftovers[i];
        ordered.push(jobIndex);
        const local = localIndex.get(jobIndex);
        if (local !== undefined && !used[local]) {
          used[local] = 1;
          remaining--;
        }
      }
      break;
    }

    ordered.push(next.job);
    const local = localIndex.get(next.job);
    if (local !== undefined && !used[local]) {
      used[local] = 1;
      remaining--;
    }
    currentCell = jobs[next.job].cell;
  }

  return ordered;
}

const assignments: unknown[][] =
  maxRobotId >= 0 ? Array.from({ length: maxRobotId + 1 }, (): unknown[] => []) : [];

for (let i = 0; i < robots.length; i++) {
  const robot = robots[i];
  const ordered = orderJobsForRobot(robot.startCell, robot.assigned);
  const out: unknown[] = [];
  for (let j = 0; j < ordered.length; j++) {
    out.push(jobs[ordered[j]].id);
  }
  assignments[robot.id] = out;
}
