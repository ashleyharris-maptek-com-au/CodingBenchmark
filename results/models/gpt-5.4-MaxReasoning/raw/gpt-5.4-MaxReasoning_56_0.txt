const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = input.trim() ? JSON.parse(input) : {};

const depot = Array.isArray(data.depot) && data.depot.length >= 2 ? data.depot : [0, 0];
const depotX = Number(depot[0]) || 0;
const depotY = Number(depot[1]) || 0;
const rawPackages = Array.isArray(data.packages) ? data.packages : [];

if (rawPackages.length === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

let truckCount = Math.floor(Number(data.truckCount));
if (!Number.isFinite(truckCount) || truckCount < 1) truckCount = 1;
if (truckCount > rawPackages.length) truckCount = rawPackages.length;

function distXY(x1, y1, x2, y2) {
  const dx = x1 - x2;
  const dy = y1 - y2;
  return Math.sqrt(dx * dx + dy * dy);
}

function distPoints(a, b) {
  const dx = a.x - b.x;
  const dy = a.y - b.y;
  return Math.sqrt(dx * dx + dy * dy);
}

function reverseSegment(arr, start, end) {
  while (start < end) {
    const tmp = arr[start];
    arr[start] = arr[end];
    arr[end] = tmp;
    start++;
    end--;
  }
}

function tourCost(order) {
  if (order.length === 0) return 0;
  let total = distXY(depotX, depotY, order[0].x, order[0].y);
  for (let i = 0; i < order.length - 1; i++) total += distPoints(order[i], order[i + 1]);
  total += distXY(order[order.length - 1].x, order[order.length - 1].y, depotX, depotY);
  return total;
}

function twoOpt(order) {
  const n = order.length;
  if (n < 4) return order;

  let maxMoves;
  let window;
  if (n <= 80) {
    maxMoves = n * n;
    window = n - 1;
  } else if (n <= 300) {
    maxMoves = n * 6;
    window = 80;
  } else if (n <= 900) {
    maxMoves = n * 4;
    window = 40;
  } else {
    maxMoves = n * 2;
    window = 24;
  }

  let moves = 0;
  let improved = true;
  while (improved && moves < maxMoves) {
    improved = false;

    for (let i = 0; i < n - 1; i++) {
      const aX = i === 0 ? depotX : order[i - 1].x;
      const aY = i === 0 ? depotY : order[i - 1].y;
      const b = order[i];
      const bx = b.x;
      const by = b.y;
      const jLimit = Math.min(n - 1, i + window);

      for (let j = i + 1; j <= jLimit; j++) {
        if (i === 0 && j === n - 1) continue;

        const c = order[j];
        const dX = j === n - 1 ? depotX : order[j + 1].x;
        const dY = j === n - 1 ? depotY : order[j + 1].y;

        const oldCost =
          distXY(aX, aY, bx, by) +
          distXY(c.x, c.y, dX, dY);
        const newCost =
          distXY(aX, aY, c.x, c.y) +
          distXY(bx, by, dX, dY);

        if (newCost + 1e-9 < oldCost) {
          reverseSegment(order, i, j);
          improved = true;
          moves++;
          break;
        }
      }

      if (improved) break;
    }

    if (!improved) break;
  }

  return order;
}

function rotateCopy(arr, start) {
  if (start === 0) return arr.slice();
  return arr.slice(start).concat(arr.slice(0, start));
}

function part1By1(n) {
  n &= 0xffff;
  n = (n | (n << 8)) & 0x00ff00ff;
  n = (n | (n << 4)) & 0x0f0f0f0f;
  n = (n | (n << 2)) & 0x33333333;
  n = (n | (n << 1)) & 0x55555555;
  return n >>> 0;
}

function mortonCode(x, y) {
  return ((part1By1(y) << 1) | part1By1(x)) >>> 0;
}

function buildNearestNeighbor(seedPoint) {
  const n = points.length;
  const used = new Uint8Array(n);
  const order = new Array(n);
  let current = seedPoint;
  used[current.idx] = 1;
  order[0] = current;

  for (let pos = 1; pos < n; pos++) {
    let best = null;
    let bestDist = Infinity;

    for (let i = 0; i < n; i++) {
      const candidate = points[i];
      if (used[candidate.idx]) continue;
      const d = distPoints(current, candidate);
      if (
        best === null ||
        d < bestDist - 1e-9 ||
        (Math.abs(d - bestDist) <= 1e-9 && candidate.depotDist < best.depotDist) ||
        (Math.abs(d - bestDist) <= 1e-9 && candidate.depotDist === best.depotDist && candidate.idx < best.idx)
      ) {
        bestDist = d;
        best = candidate;
      }
    }

    used[best.idx] = 1;
    order[pos] = best;
    current = best;
  }

  return order;
}

const points = rawPackages.map((pkg, idx) => {
  const x = Number(pkg.x) || 0;
  const y = Number(pkg.y) || 0;
  const dx = x - depotX;
  const dy = y - depotY;
  return {
    idx,
    id: pkg.id,
    x,
    y,
    angle: Math.atan2(dy, dx),
    depotDist: Math.sqrt(dx * dx + dy * dy),
    radius2: dx * dx + dy * dy
  };
});

const candidates = [];
const seenNames = new Set();

function addCandidate(name, order) {
  if (seenNames.has(name)) return;
  seenNames.add(name);
  candidates.push({ name, order });
}

const angleSorted = points.slice().sort((a, b) =>
  a.angle - b.angle ||
  a.radius2 - b.radius2 ||
  a.idx - b.idx
);

const gapInfo = [];
for (let i = 0; i < angleSorted.length; i++) {
  const next = (i + 1) % angleSorted.length;
  let gap = angleSorted[next].angle - angleSorted[i].angle;
  if (gap < 0) gap += Math.PI * 2;
  gapInfo.push({ gap, start: next });
}
gapInfo.sort((a, b) => b.gap - a.gap || a.start - b.start);

const angleStarts = new Set([0]);
const angleVariants = Math.min(angleSorted.length, 8);
for (let i = 0; i < angleVariants; i++) angleStarts.add(gapInfo[i].start);
for (const start of angleStarts) addCandidate(`angle:${start}`, rotateCopy(angleSorted, start));

let minX = points[0].x;
let maxX = points[0].x;
let minY = points[0].y;
let maxY = points[0].y;
for (let i = 1; i < points.length; i++) {
  const p = points[i];
  if (p.x < minX) minX = p.x;
  if (p.x > maxX) maxX = p.x;
  if (p.y < minY) minY = p.y;
  if (p.y > maxY) maxY = p.y;
}

const scaleX = maxX === minX ? 0 : 65535 / (maxX - minX);
const scaleY = maxY === minY ? 0 : 65535 / (maxY - minY);
const mortonPoints = points.map((p) => {
  const xi = scaleX === 0 ? 0 : Math.max(0, Math.min(65535, Math.round((p.x - minX) * scaleX)));
  const yi = scaleY === 0 ? 0 : Math.max(0, Math.min(65535, Math.round((p.y - minY) * scaleY)));
  return { point: p, code: mortonCode(xi, yi) };
});
mortonPoints.sort((a, b) =>
  a.code - b.code ||
  a.point.x - b.point.x ||
  a.point.y - b.point.y ||
  a.point.idx - b.point.idx
);
addCandidate('morton', mortonPoints.map((entry) => entry.point));

addCandidate('x', points.slice().sort((a, b) => a.x - b.x || a.y - b.y || a.idx - b.idx));
addCandidate('y', points.slice().sort((a, b) => a.y - b.y || a.x - b.x || a.idx - b.idx));
addCandidate('diag+', points.slice().sort((a, b) => (a.x + a.y) - (b.x + b.y) || a.x - b.x || a.idx - b.idx));
addCandidate('diag-', points.slice().sort((a, b) => (a.x - a.y) - (b.x - b.y) || a.x - b.x || a.idx - b.idx));

if (points.length <= 1200) {
  let nearestToDepot = points[0];
  let farthestFromDepot = points[0];
  let minXP = points[0];
  let maxXP = points[0];

  for (let i = 1; i < points.length; i++) {
    const p = points[i];
    if (p.depotDist < nearestToDepot.depotDist) nearestToDepot = p;
    if (p.depotDist > farthestFromDepot.depotDist) farthestFromDepot = p;
    if (p.x < minXP.x || (p.x === minXP.x && p.y < minXP.y)) minXP = p;
    if (p.x > maxXP.x || (p.x === maxXP.x && p.y < maxXP.y)) maxXP = p;
  }

  addCandidate('nn-near', buildNearestNeighbor(nearestToDepot));
  addCandidate('nn-far', buildNearestNeighbor(farthestFromDepot));
  addCandidate('nn-left', buildNearestNeighbor(minXP));
  addCandidate('nn-right', buildNearestNeighbor(maxXP));
}

const quickSolutions = candidates.map((candidate) => ({
  name: candidate.name,
  estimatedCost: tourCost(candidate.order),
  order: candidate.order
}));

quickSolutions.sort((a, b) => a.estimatedCost - b.estimatedCost);

const refinementCount = Math.min(4, quickSolutions.length);
let best = null;

for (let i = 0; i < refinementCount; i++) {
  const order = quickSolutions[i].order.slice();
  twoOpt(order);
  const cost = tourCost(order);
  if (!best || cost < best.cost) best = { order, cost };
}

const output = {
  routes: [best.order.map((p) => p.id)]
};

process.stdout.write(JSON.stringify(output));
