struct Vec3 {
    double x = 0.0;
    double y = 0.0;
    double z = 0.0;

    Vec3() = default;
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}

    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
    Vec3& operator+=(const Vec3& o) { x += o.x; y += o.y; z += o.z; return *this; }
};

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

static inline double norm2(const Vec3& v) {
    return dot(v, v);
}

static inline double norm(const Vec3& v) {
    return std::sqrt(norm2(v));
}

static inline Vec3 normalized(const Vec3& v) {
    double n = norm(v);
    if (n == 0.0) return v;
    return v / n;
}

struct Point2 {
    double x = 0.0;
    double y = 0.0;

    Point2() = default;
    Point2(double x_, double y_) : x(x_), y(y_) {}

    Point2 operator+(const Point2& o) const { return {x + o.x, y + o.y}; }
    Point2 operator-(const Point2& o) const { return {x - o.x, y - o.y}; }
    Point2 operator*(double s) const { return {x * s, y * s}; }
};

static inline double cross2(const Point2& a, const Point2& b) {
    return a.x * b.y - a.y * b.x;
}

static inline double dot2(const Point2& a, const Point2& b) {
    return a.x * b.x + a.y * b.y;
}

static inline double norm22(const Point2& p) {
    return dot2(p, p);
}

struct AABB {
    Vec3 mn{
        std::numeric_limits<double>::infinity(),
        std::numeric_limits<double>::infinity(),
        std::numeric_limits<double>::infinity()
    };
    Vec3 mx{
        -std::numeric_limits<double>::infinity(),
        -std::numeric_limits<double>::infinity(),
        -std::numeric_limits<double>::infinity()
    };

    void expand(const Vec3& p) {
        mn.x = std::min(mn.x, p.x);
        mn.y = std::min(mn.y, p.y);
        mn.z = std::min(mn.z, p.z);
        mx.x = std::max(mx.x, p.x);
        mx.y = std::max(mx.y, p.y);
        mx.z = std::max(mx.z, p.z);
    }

    void expand(const AABB& b) {
        expand(b.mn);
        expand(b.mx);
    }

    Vec3 extent() const {
        return mx - mn;
    }

    bool overlap(const AABB& o, double eps = 0.0) const {
        return !(mx.x < o.mn.x - eps || mn.x > o.mx.x + eps ||
                 mx.y < o.mn.y - eps || mn.y > o.mx.y + eps ||
                 mx.z < o.mn.z - eps || mn.z > o.mx.z + eps);
    }

    bool contains(const Vec3& p, double pad = 0.0) const {
        return p.x >= mn.x - pad && p.x <= mx.x + pad &&
               p.y >= mn.y - pad && p.y <= mx.y + pad &&
               p.z >= mn.z - pad && p.z <= mx.z + pad;
    }
};

struct Projection2D {
    int ax0 = 0;
    int ax1 = 1;
    int dropped = 2;
    Vec3 planePoint;
    Vec3 planeNormal;

    Point2 project(const Vec3& p) const {
        if (dropped == 0) return {p.y, p.z};
        if (dropped == 1) return {p.x, p.z};
        return {p.x, p.y};
    }

    Vec3 unproject(const Point2& p) const {
        double d = dot(planeNormal, planePoint);
        if (dropped == 0) {
            double y = p.x;
            double z = p.y;
            double x = (d - planeNormal.y * y - planeNormal.z * z) / planeNormal.x;
            return {x, y, z};
        }
        if (dropped == 1) {
            double x = p.x;
            double z = p.y;
            double y = (d - planeNormal.x * x - planeNormal.z * z) / planeNormal.y;
            return {x, y, z};
        }
        double x = p.x;
        double y = p.y;
        double z = (d - planeNormal.x * x - planeNormal.y * y) / planeNormal.z;
        return {x, y, z};
    }
};

static Projection2D make_coplanar_projection(const Triangle& tri) {
    Vec3 n = tri.normal;
    double ax = std::fabs(n.x);
    double ay = std::fabs(n.y);
    double az = std::fabs(n.z);
    Projection2D pr;
    if (ax >= ay && ax >= az) {
        pr.dropped = 0;
        pr.ax0 = 1;
        pr.ax1 = 2;
    } else if (ay >= az) {
        pr.dropped = 1;
        pr.ax0 = 0;
        pr.ax1 = 2;
    } else {
        pr.dropped = 2;
        pr.ax0 = 0;
        pr.ax1 = 1;
    }
    pr.planePoint = tri.p[0];
    pr.planeNormal = tri.normal;
    return pr;
}

struct Basis2D {
    Vec3 origin;
    Vec3 u;
    Vec3 v;
    Vec3 n;

    explicit Basis2D(const Triangle& tri) {
        origin = tri.p[0];
        n = normalized(tri.normal);
        u = normalized(tri.p[1] - tri.p[0]);
        if (norm2(u) == 0.0) {
            u = normalized(tri.p[2] - tri.p[0]);
        }
        v = cross(n, u);
    }

    Point2 project(const Vec3& p) const {
        Vec3 d = p - origin;
        return {dot(d, u), dot(d, v)};
    }

    Vec3 unproject(const Point2& p) const {
        return origin + u * p.x + v * p.y;
    }
};

struct SceneEps {

static inline bool near_zero(double v, double eps) {
    return std::fabs(v) <= eps;
}

static inline bool same_point(const Vec3& a, const Vec3& b, double eps) {
    return norm2(a - b) <= eps * eps;
}

static inline bool same_point2(const Point2& a, const Point2& b, double eps) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy <= eps * eps;
}

static inline double orient2d(const Point2& a, const Point2& b, const Point2& c) {
    return cross2(b - a, c - a);
}

static inline double clamp01(double t) {
    if (t < 0.0) return 0.0;
    if (t > 1.0) return 1.0;
    return t;
}

static bool point_on_segment2(const Point2& p, const Point2& a, const Point2& b, double eps) {
    if (std::fabs(orient2d(a, b, p)) > eps) return false;
    double dp = dot2(p - a, b - a);
    if (dp < -eps) return false;
    if (dp - dot2(b - a, b - a) > eps) return false;
    return true;
}

static bool point_in_triangle2(const Point2& p, const std::array<Point2, 3>& tri, double eps) {
    double a = orient2d(tri[0], tri[1], p);
    double b = orient2d(tri[1], tri[2], p);
    double c = orient2d(tri[2], tri[0], p);
    bool hasNeg = (a < -eps) || (b < -eps) || (c < -eps);
    bool hasPos = (a > eps) || (b > eps) || (c > eps);
    return !(hasNeg && hasPos);
}

static double polygon_area2(const std::vector<Point2>& poly) {
    double area = 0.0;
    const int n = static_cast<int>(poly.size());
    for (int i = 0; i < n; ++i) {
        const Point2& a = poly[i];
        const Point2& b = poly[(i + 1) % n];
        area += cross2(a, b);
    }
    return 0.5 * area;
}

static Point2 polygon_centroid2(const std::vector<Point2>& poly) {
    double area2 = 0.0;
    double cx = 0.0;
    double cy = 0.0;
    const int n = static_cast<int>(poly.size());
    for (int i = 0; i < n; ++i) {
        const Point2& a = poly[i];
        const Point2& b = poly[(i + 1) % n];
        double cr = cross2(a, b);
        area2 += cr;
        cx += (a.x + b.x) * cr;
        cy += (a.y + b.y) * cr;
    }
    if (std::fabs(area2) < 1e-30) {
        Point2 avg{0.0, 0.0};
        for (const auto& p : poly) {
            avg.x += p.x;
            avg.y += p.y;
        }
        if (!poly.empty()) {
            avg.x /= static_cast<double>(poly.size());
            avg.y /= static_cast<double>(poly.size());
        }
        return avg;
    }
    return {cx / (3.0 * area2), cy / (3.0 * area2)};
}

static std::vector<Point2> remove_duplicate_ring_points(std::vector<Point2> poly, double eps) {
    std::vector<Point2> out;
    for (const auto& p : poly) {
        if (out.empty() || !same_point2(out.back(), p, eps)) {
            out.push_back(p);
        }
    }
    if (out.size() > 1 && same_point2(out.front(), out.back(), eps)) {
        out.pop_back();
    }
    return out;
}

struct Mesh;

struct BVHNode {
    AABB box;
    int left = -1;
    int right = -1;
    int start = 0;
    int count = 0;
    bool leaf = false;
};

struct BVH {
    std::vector<int> order;
    std::vector<BVHNode> nodes;
    int root = -1;
};

struct Mesh {
    std::vector<Vec3> vertices;
    std::vector<std::array<int, 3>> faces;
    std::vector<Triangle> tris;
    std::vector<std::array<int, 3>> neighbors;
    AABB box;
    BVH bvh;
};

static Triangle make_triangle(const std::vector<Vec3>& verts, const std::array<int, 3>& face) {
    Triangle tri;
    tri.idx = face;
    tri.p = {verts[face[0]], verts[face[1]], verts[face[2]]};
    tri.normal = cross(tri.p[1] - tri.p[0], tri.p[2] - tri.p[0]);
    tri.centroid = (tri.p[0] + tri.p[1] + tri.p[2]) / 3.0;
    tri.box.expand(tri.p[0]);
    tri.box.expand(tri.p[1]);
    tri.box.expand(tri.p[2]);
    return tri;
}

static void build_neighbors(Mesh& mesh) {
    mesh.neighbors.assign(mesh.faces.size(), {-1, -1, -1});
    std::unordered_map<std::uint64_t, std::pair<int, int>> edgeOwner;
    edgeOwner.reserve(mesh.faces.size() * 3);
    for (int fi = 0; fi < static_cast<int>(mesh.faces.size()); ++fi) {
        const auto& f = mesh.faces[fi];
        for (int e = 0; e < 3; ++e) {
            int a = f[e];
            int b = f[(e + 1) % 3];
            if (a > b) std::swap(a, b);
            std::uint64_t key = (static_cast<std::uint64_t>(static_cast<std::uint32_t>(a)) << 32) |
                                static_cast<std::uint32_t>(b);
            auto it = edgeOwner.find(key);
            if (it == edgeOwner.end()) {
                edgeOwner.emplace(key, std::make_pair(fi, e));
            } else {
                int of = it->second.first;
                int oe = it->second.second;
                mesh.neighbors[fi][e] = of;
                mesh.neighbors[of][oe] = fi;
            }
        }
    }
}

static int build_bvh_recursive(BVH& bvh, const std::vector<Triangle>& tris, int begin, int end) {
    int nodeIndex = static_cast<int>(bvh.nodes.size());
    bvh.nodes.push_back({});
    BVHNode& node = bvh.nodes[nodeIndex];
    node.start = begin;
    node.count = end - begin;
    for (int i = begin; i < end; ++i) node.box.expand(tris[bvh.order[i]].box);
    if (end - begin <= 8) {
        node.leaf = true;
        return nodeIndex;
    }

    AABB centroidBox;
    for (int i = begin; i < end; ++i) centroidBox.expand(tris[bvh.order[i]].centroid);
    Vec3 ext = centroidBox.extent();
    int axis = 0;
    if (ext.y > ext.x && ext.y >= ext.z) axis = 1;
    else if (ext.z > ext.x && ext.z >= ext.y) axis = 2;

    int mid = begin + (end - begin) / 2;
    auto coord = [&](const Triangle& t) {
        if (axis == 0) return t.centroid.x;
        if (axis == 1) return t.centroid.y;
        return t.centroid.z;
    };
    std::nth_element(
        bvh.order.begin() + begin,
        bvh.order.begin() + mid,
        bvh.order.begin() + end,
        [&](int a, int b) { return coord(tris[a]) < coord(tris[b]); });

    int left = build_bvh_recursive(bvh, tris, begin, mid);
    int right = build_bvh_recursive(bvh, tris, mid, end);
    bvh.nodes[nodeIndex].left = left;
    bvh.nodes[nodeIndex].right = right;
    return nodeIndex;
}

static void build_bvh(Mesh& mesh) {
    mesh.bvh.order.resize(mesh.tris.size());
    std::iota(mesh.bvh.order.begin(), mesh.bvh.order.end(), 0);
    mesh.bvh.nodes.clear();
    mesh.bvh.root = build_bvh_recursive(mesh.bvh, mesh.tris, 0, static_cast<int>(mesh.tris.size()));
}

static double ray_box_intersection_tmin(const AABB& box, const Vec3& o, const Vec3& invDir) {
    double t1 = (box.mn.x - o.x) * invDir.x;
    double t2 = (box.mx.x - o.x) * invDir.x;
    double tmin = std::min(t1, t2);
    double tmax = std::max(t1, t2);

    t1 = (box.mn.y - o.y) * invDir.y;
    t2 = (box.mx.y - o.y) * invDir.y;
    tmin = std::max(tmin, std::min(t1, t2));
    tmax = std::min(tmax, std::max(t1, t2));

    t1 = (box.mn.z - o.z) * invDir.z;
    t2 = (box.mx.z - o.z) * invDir.z;
    tmin = std::max(tmin, std::min(t1, t2));
    tmax = std::min(tmax, std::max(t1, t2));

    if (tmax >= std::max(0.0, tmin)) return tmin;
    return std::numeric_limits<double>::infinity();
}

static bool ray_intersects_triangle(const Vec3& origin, const Vec3& dir, const Triangle& tri, double eps, double& tOut) {
    const Vec3 e1 = tri.p[1] - tri.p[0];
    const Vec3 e2 = tri.p[2] - tri.p[0];
    const Vec3 pvec = cross(dir, e2);
    const double det = dot(e1, pvec);
    if (std::fabs(det) < eps) return false;
    const double invDet = 1.0 / det;
    const Vec3 tvec = origin - tri.p[0];
    const double u = dot(tvec, pvec) * invDet;
    if (u < -eps || u > 1.0 + eps) return false;
    const Vec3 qvec = cross(tvec, e1);
    const double v = dot(dir, qvec) * invDet;
    if (v < -eps || u + v > 1.0 + eps) return false;
    const double t = dot(e2, qvec) * invDet;
    if (t <= eps) return false;
    tOut = t;
    return true;
}

static int ray_cast_count(const Mesh& mesh, const Vec3& origin, const Vec3& dir, double eps) {
    if (mesh.tris.empty()) return 0;
    Vec3 invDir{
        std::fabs(dir.x) < eps ? std::copysign(1.0 / eps, dir.x == 0.0 ? 1.0 : dir.x) : 1.0 / dir.x,
        std::fabs(dir.y) < eps ? std::copysign(1.0 / eps, dir.y == 0.0 ? 1.0 : dir.y) : 1.0 / dir.y,
        std::fabs(dir.z) < eps ? std::copysign(1.0 / eps, dir.z == 0.0 ? 1.0 : dir.z) : 1.0 / dir.z
    };

    int hits = 0;
    std::vector<int> stack;
    stack.push_back(mesh.bvh.root);
    while (!stack.empty()) {
        int ni = stack.back();
        stack.pop_back();
        const BVHNode& node = mesh.bvh.nodes[ni];
        if (!std::isfinite(ray_box_intersection_tmin(node.box, origin, invDir))) continue;
        if (node.leaf) {
            for (int i = 0; i < node.count; ++i) {
                int tid = mesh.bvh.order[node.start + i];
                double tHit = 0.0;
                if (ray_intersects_triangle(origin, dir, mesh.tris[tid], eps, tHit)) {
                    ++hits;
                }
            }
        } else {
            stack.push_back(node.left);
            stack.push_back(node.right);
        }
    }
    return hits;
}

static bool point_on_triangle3(const Vec3& p, const Triangle& tri, double eps) {
    Vec3 n = tri.normal;
    double len = norm(n);
    if (len <= eps) return false;
    double dist = dot(n, p - tri.p[0]) / len;
    if (std::fabs(dist) > eps) return false;

    Vec3 v0 = tri.p[1] - tri.p[0];
    Vec3 v1 = tri.p[2] - tri.p[0];
    Vec3 v2 = p - tri.p[0];
    double d00 = dot(v0, v0);
    double d01 = dot(v0, v1);
    double d11 = dot(v1, v1);
    double d20 = dot(v2, v0);
    double d21 = dot(v2, v1);
    double denom = d00 * d11 - d01 * d01;
    if (std::fabs(denom) <= eps) return false;
    double inv = 1.0 / denom;
    double v = (d11 * d20 - d01 * d21) * inv;
    double w = (d00 * d21 - d01 * d20) * inv;
    double u = 1.0 - v - w;
    return u >= -eps && v >= -eps && w >= -eps;
}

static bool point_on_surface(const Mesh& mesh, const Vec3& p, double eps, Vec3* nOut = nullptr) {
    if (mesh.tris.empty()) return false;
    std::vector<int> stack;
    stack.push_back(mesh.bvh.root);
    while (!stack.empty()) {
        int ni = stack.back();
        stack.pop_back();
        const BVHNode& node = mesh.bvh.nodes[ni];
        if (!node.box.contains(p, eps)) continue;
        if (node.leaf) {
            for (int i = 0; i < node.count; ++i) {
                int tid = mesh.bvh.order[node.start + i];
                if (mesh.tris[tid].box.contains(p, eps) && point_on_triangle3(p, mesh.tris[tid], eps)) {
                    if (nOut) *nOut = normalized(mesh.tris[tid].normal);
                    return true;
                }
            }
        } else {
            stack.push_back(node.left);
            stack.push_back(node.right);
        }
    }
    return false;
}

static bool point_inside_mesh(const Mesh& mesh, const Vec3& p, const SceneEps& eps) {
    if (!mesh.box.contains(p, eps.point)) return false;
    if (point_on_surface(mesh, p, eps.delta, nullptr)) return true;

    static const std::array<Vec3, 3> dirs = {
        normalized(Vec3{1.0, 0.3125, 0.1171875}),
        normalized(Vec3{0.181, 1.0, 0.271}),
        normalized(Vec3{0.233, 0.419, 1.0})
    };

    int insideVotes = 0;
    for (const Vec3& dir : dirs) {
        int count = ray_cast_count(mesh, p, dir, eps.point);
        if (count & 1) ++insideVotes;
    }
    return insideVotes >= 2;
}

static bool collect_triangle_plane_section(
    const Triangle& tri,
    const std::array<double, 3>& sd,
    double eps,
    std::vector<Vec3>& pts) {

    pts.clear();
    auto add_pt = [&](const Vec3& p) {
        for (const auto& q : pts) {
            if (same_point(p, q, eps)) return;
        }
        pts.push_back(p);
    };

    for (int i = 0; i < 3; ++i) {
        if (std::fabs(sd[i]) <= eps) add_pt(tri.p[i]);
    }
    for (int i = 0; i < 3; ++i) {
        int j = (i + 1) % 3;
        double a = sd[i];
        double b = sd[j];
        if ((a < -eps && b > eps) || (a > eps && b < -eps)) {
            double t = a / (a - b);
            add_pt(tri.p[i] + (tri.p[j] - tri.p[i]) * t);
        }
    }

    if (pts.size() > 2) {
        std::vector<Vec3> unique;
        for (const auto& p : pts) {
            bool dup = false;
            for (const auto& q : unique) {
                if (same_point(p, q, eps)) {
                    dup = true;
                    break;
                }
            }
            if (!dup) unique.push_back(p);
        }
        pts.swap(unique);
    }
    return pts.size() >= 2;
}

static std::vector<Point2> clip_polygon_against_edge(
    const std::vector<Point2>& poly,
    const Point2& a,
    const Point2& b,
    double orient,
    double eps) {

    auto inside = [&](const Point2& p) {
        return orient * orient2d(a, b, p) >= -eps;
    };
    auto intersect_line = [&](const Point2& p0, const Point2& p1) {
        Point2 r = p1 - p0;
        Point2 s = b - a;
        double den = cross2(r, s);
        if (std::fabs(den) <= eps) return p0;
        double t = cross2(a - p0, s) / den;
        return p0 + r * t;
    };

    std::vector<Point2> out;
    if (poly.empty()) return out;
    for (int i = 0, n = static_cast<int>(poly.size()); i < n; ++i) {
        Point2 curr = poly[i];
        Point2 prev = poly[(i + n - 1) % n];
        bool inCurr = inside(curr);
        bool inPrev = inside(prev);
        if (inCurr) {
            if (!inPrev) out.push_back(intersect_line(prev, curr));
            out.push_back(curr);
        } else if (inPrev) {
            out.push_back(intersect_line(prev, curr));
        }
    }
    return remove_duplicate_ring_points(std::move(out), eps);
}

static std::vector<Vec3> coplanar_triangle_overlap(const Triangle& a, const Triangle& b, double eps) {
    Projection2D pr = make_coplanar_projection(a);
    std::vector<Point2> poly = {
        pr.project(a.p[0]),
        pr.project(a.p[1]),
        pr.project(a.p[2])
    };
    std::array<Point2, 3> clip = {
        pr.project(b.p[0]),
        pr.project(b.p[1]),
        pr.project(b.p[2])
    };
    double orient = polygon_area2(std::vector<Point2>{clip[0], clip[1], clip[2]}) >= 0.0 ? 1.0 : -1.0;
    for (int i = 0; i < 3 && !poly.empty(); ++i) {
        poly = clip_polygon_against_edge(poly, clip[i], clip[(i + 1) % 3], orient, eps);
    }
    poly = remove_duplicate_ring_points(std::move(poly), eps);
    if (poly.size() < 3 || std::fabs(polygon_area2(poly)) <= eps * eps) return {};

    std::vector<Vec3> out;
    out.reserve(poly.size());
    for (const auto& p : poly) out.push_back(pr.unproject(p));
    return out;
}

static std::vector<Segment3> triangle_triangle_cuts(const Triangle& a, const Triangle& b, const SceneEps& eps) {
    std::vector<Segment3> out;
    Vec3 n1 = normalized(a.normal);
    Vec3 n2 = normalized(b.normal);
    double d1 = -dot(n1, a.p[0]);
    double d2 = -dot(n2, b.p[0]);

    std::array<double, 3> sb{
        dot(n1, b.p[0]) + d1,
        dot(n1, b.p[1]) + d1,
        dot(n1, b.p[2]) + d1
    };
    std::array<double, 3> sa{
        dot(n2, a.p[0]) + d2,
        dot(n2, a.p[1]) + d2,
        dot(n2, a.p[2]) + d2
    };

    auto separated = [&](const std::array<double, 3>& s) {
        bool pos = true;
        bool neg = true;
        for (double v : s) {
            pos = pos && (v > eps.point);
            neg = neg && (v < -eps.point);
        }
        return pos || neg;
    };
    if (separated(sa) || separated(sb)) return out;

    Vec3 dir = cross(n1, n2);
    if (norm(dir) <= eps.point) {
        bool coplanar = true;
        for (double v : sa) coplanar = coplanar && std::fabs(v) <= eps.split;
        for (double v : sb) coplanar = coplanar && std::fabs(v) <= eps.split;
        if (!coplanar) return out;
        std::vector<Vec3> poly = coplanar_triangle_overlap(a, b, eps.split);
        if (poly.size() < 3) return out;
        for (int i = 0, n = static_cast<int>(poly.size()); i < n; ++i) {
            Vec3 p0 = poly[i];
            Vec3 p1 = poly[(i + 1) % n];
            if (!same_point(p0, p1, eps.split)) {
                out.push_back({p0, p1});
            }
        }
        return out;
    }

    dir = normalized(dir);
    std::vector<Vec3> secA;
    std::vector<Vec3> secB;
    if (!collect_triangle_plane_section(a, sa, eps.split, secA)) return out;
    if (!collect_triangle_plane_section(b, sb, eps.split, secB)) return out;
    if (secA.size() < 2 || secB.size() < 2) return out;

    auto order_segment = [&](std::vector<Vec3>& pts) {
        if (dot(pts[1] - pts[0], dir) < 0.0) std::swap(pts[0], pts[1]);
    };
    order_segment(secA);
    order_segment(secB);

    double a0 = dot(secA[0], dir);
    double a1 = dot(secA[1], dir);
    double b0 = dot(secB[0], dir);
    double b1 = dot(secB[1], dir);
    double lo = std::max(std::min(a0, a1), std::min(b0, b1));
    double hi = std::min(std::max(a0, a1), std::max(b0, b1));
    if (hi - lo <= eps.split) return out;

    Vec3 ref = secA[0];
    double refProj = dot(ref, dir);
    Vec3 p0 = ref + dir * (lo - refProj);
    Vec3 p1 = ref + dir * (hi - refProj);
    if (!same_point(p0, p1, eps.split)) out.push_back({p0, p1});
    return out;
}

struct PointKey2 {
    std::int64_t x;
    std::int64_t y;

    bool operator==(const PointKey2& o) const { return x == o.x && y == o.y; }
};

struct PointKey2Hash {
    std::size_t operator()(const PointKey2& k) const noexcept {
        std::size_t h1 = std::hash<std::int64_t>{}(k.x);
        std::size_t h2 = std::hash<std::int64_t>{}(k.y);
        return h1 * 1315423911u ^ (h2 + 0x9e3779b97f4a7c15ull + (h1 << 6) + (h1 >> 2));
    }
};

struct PointRegistry2 {
    double eps;
    std::vector<Point2> pts;
    std::unordered_map<PointKey2, std::vector<int>, PointKey2Hash> buckets;

    explicit PointRegistry2(double eps_) : eps(eps_) {}

    PointKey2 key_for(const Point2& p) const {
        return {
            static_cast<std::int64_t>(std::floor(p.x / eps)),
            static_cast<std::int64_t>(std::floor(p.y / eps))
        };
    }

    int add(const Point2& p) {
        PointKey2 k = key_for(p);
        for (int dx = -1; dx <= 1; ++dx) {
            for (int dy = -1; dy <= 1; ++dy) {
                PointKey2 nk{k.x + dx, k.y + dy};
                auto it = buckets.find(nk);
                if (it == buckets.end()) continue;
                for (int idx : it->second) {
                    if (same_point2(pts[idx], p, eps)) return idx;
                }
            }
        }
        int id = static_cast<int>(pts.size());
        pts.push_back(p);
        buckets[k].push_back(id);
        return id;
    }
};

static bool segment_intersection_params(
    const Segment2& s0,
    const Segment2& s1,
    double eps,
    std::vector<std::pair<double, Point2>>& out0,
    std::vector<std::pair<double, Point2>>& out1) {

    Point2 r = s0.b - s0.a;
    Point2 s = s1.b - s1.a;
    double den = cross2(r, s);
    Point2 qp = s1.a - s0.a;

    if (std::fabs(den) > eps) {
        double t = cross2(qp, s) / den;
        double u = cross2(qp, r) / den;
        if (t >= -eps && t <= 1.0 + eps && u >= -eps && u <= 1.0 + eps) {
            Point2 p = s0.a + r * t;
            out0.push_back({clamp01(t), p});
            out1.push_back({clamp01(u), p});
            return true;
        }
        return false;
    }

    if (std::fabs(cross2(qp, r)) > eps) return false;

    auto add_if_on = [&](const Point2& p, const Segment2& base, std::vector<std::pair<double, Point2>>& out) {
        Point2 d = base.b - base.a;
        double denom = dot2(d, d);
        if (denom <= eps * eps) return;
        double t = dot2(p - base.a, d) / denom;
        if (t >= -eps && t <= 1.0 + eps && point_on_segment2(p, base.a, base.b, eps)) {
            out.push_back({clamp01(t), p});
        }
    };

    add_if_on(s0.a, s1, out1);
    add_if_on(s0.b, s1, out1);
    add_if_on(s1.a, s0, out0);
    add_if_on(s1.b, s0, out0);
    return !out0.empty() || !out1.empty();
}

struct Fragment {
    std::vector<Point2> poly2;
    std::vector<Vec3> poly3;
    Vec3 centroid;
};

struct EdgeKey {
    int a = 0;
    int b = 0;

    bool operator==(const EdgeKey& o) const { return a == o.a && b == o.b; }
};

struct EdgeKeyHash {
    std::size_t operator()(const EdgeKey& e) const noexcept {
        return static_cast<std::size_t>(static_cast<std::uint64_t>(static_cast<std::uint32_t>(e.a)) << 32) ^
               static_cast<std::uint32_t>(e.b);
    }
};

struct DirEdge {
    int from = -1;
    int to = -1;
    int twin = -1;
    int next = -1;
    double ang = 0.0;
    bool used = false;
};

static std::vector<std::array<int, 3>> ear_clip_triangulate(std::vector<Point2> poly, double eps) {
    std::vector<std::array<int, 3>> tris;
    int n = static_cast<int>(poly.size());
    if (n < 3) return tris;

    if (polygon_area2(poly) < 0.0) std::reverse(poly.begin(), poly.end());

    std::vector<int> idx(n);
    std::iota(idx.begin(), idx.end(), 0);

    auto point_in_tri = [&](const Point2& p, const Point2& a, const Point2& b, const Point2& c) {
        std::array<Point2, 3> tri{a, b, c};
        return point_in_triangle2(p, tri, eps);
    };

    int guard = 0;
    while (idx.size() > 3 && guard < n * n) {
        bool clipped = false;
        int m = static_cast<int>(idx.size());
        for (int i = 0; i < m; ++i) {
            int ip = idx[(i + m - 1) % m];
            int ic = idx[i];
            int in = idx[(i + 1) % m];
            const Point2& a = poly[ip];
            const Point2& b = poly[ic];
            const Point2& c = poly[in];
            if (orient2d(a, b, c) <= eps) continue;

            bool ear = true;
            for (int j = 0; j < m; ++j) {
                int k = idx[j];
                if (k == ip || k == ic || k == in) continue;
                if (point_in_tri(poly[k], a, b, c)) {
                    ear = false;
                    break;
                }
            }
            if (ear) {
                tris.push_back({ip, ic, in});
                idx.erase(idx.begin() + i);
                clipped = true;
                break;
            }
        }
        if (!clipped) break;
        ++guard;
    }

    if (idx.size() == 3) tris.push_back({idx[0], idx[1], idx[2]});
    if (tris.empty() && n >= 3) {
        for (int i = 1; i + 1 < n; ++i) tris.push_back({0, i, i + 1});
    }
    return tris;
}

static std::vector<Fragment> fragment_triangle(const Triangle& tri, const std::vector<Segment3>& rawCuts, const SceneEps& eps) {
    Basis2D basis(tri);
    std::array<Point2, 3> tri2 = {basis.project(tri.p[0]), basis.project(tri.p[1]), basis.project(tri.p[2])};

    std::vector<Segment2> cuts;
    cuts.reserve(rawCuts.size());
    for (const auto& seg : rawCuts) {
        Point2 a = basis.project(seg.a);
        Point2 b = basis.project(seg.b);
        Point2 mid = (a + b) * 0.5;
        if (same_point2(a, b, eps.split)) continue;
        if (!point_in_triangle2(mid, tri2, eps.split)) continue;
        cuts.push_back({a, b});
    }

    if (cuts.empty()) {
        Fragment f;
        f.poly2 = {tri2[0], tri2[1], tri2[2]};
        f.poly3 = {tri.p[0], tri.p[1], tri.p[2]};
        f.centroid = tri.centroid;
        return {f};
    }

    PointRegistry2 registry(eps.split);
    int c0 = registry.add(tri2[0]);
    int c1 = registry.add(tri2[1]);
    int c2 = registry.add(tri2[2]);
    (void)c0;
    (void)c1;
    (void)c2;

    std::vector<std::vector<std::pair<double, int>>> segPoints(cuts.size());
    for (int i = 0; i < static_cast<int>(cuts.size()); ++i) {
        int a = registry.add(cuts[i].a);
        int b = registry.add(cuts[i].b);
        segPoints[i].push_back({0.0, a});
        segPoints[i].push_back({1.0, b});
    }

    for (int i = 0; i < static_cast<int>(cuts.size()); ++i) {
        for (int j = i + 1; j < static_cast<int>(cuts.size()); ++j) {
            std::vector<std::pair<double, Point2>> pa;
            std::vector<std::pair<double, Point2>> pb;
            if (!segment_intersection_params(cuts[i], cuts[j], eps.split, pa, pb)) continue;
            for (const auto& [t, p] : pa) {
                int id = registry.add(p);
                segPoints[i].push_back({t, id});
            }
            for (const auto& [t, p] : pb) {
                int id = registry.add(p);
                segPoints[j].push_back({t, id});
            }
        }
    }

    std::unordered_set<EdgeKey, EdgeKeyHash> edgeSet;
    auto add_edge = [&](int a, int b) {
        if (a == b) return;
        EdgeKey key{std::min(a, b), std::max(a, b)};
        edgeSet.insert(key);
    };

    for (int i = 0; i < static_cast<int>(cuts.size()); ++i) {
        auto& pts = segPoints[i];
        std::sort(pts.begin(), pts.end(), [](const auto& lhs, const auto& rhs) { return lhs.first < rhs.first; });
        std::vector<int> ordered;
        for (const auto& item : pts) {
            if (ordered.empty() || ordered.back() != item.second) ordered.push_back(item.second);
        }
        for (int k = 0; k + 1 < static_cast<int>(ordered.size()); ++k) {
            add_edge(ordered[k], ordered[k + 1]);
        }
    }

    std::array<std::pair<int, int>, 3> triEdgeIds{
        std::make_pair(registry.add(tri2[0]), registry.add(tri2[1])),
        std::make_pair(registry.add(tri2[1]), registry.add(tri2[2])),
        std::make_pair(registry.add(tri2[2]), registry.add(tri2[0]))
    };
    for (const auto& edge : triEdgeIds) {
        Point2 a = registry.pts[edge.first];
        Point2 b = registry.pts[edge.second];
        std::vector<std::pair<double, int>> onEdge;
        Point2 d = b - a;
        double denom = dot2(d, d);
        if (denom <= eps.area) continue;
        for (int id = 0; id < static_cast<int>(registry.pts.size()); ++id) {
            const Point2& p = registry.pts[id];
            if (!point_on_segment2(p, a, b, eps.split)) continue;
            double t = dot2(p - a, d) / denom;
            onEdge.push_back({clamp01(t), id});
        }
        std::sort(onEdge.begin(), onEdge.end(), [](const auto& lhs, const auto& rhs) { return lhs.first < rhs.first; });
        std::vector<int> ordered;
        for (const auto& item : onEdge) {
            if (ordered.empty() || ordered.back() != item.second) ordered.push_back(item.second);
        }
        for (int k = 0; k + 1 < static_cast<int>(ordered.size()); ++k) add_edge(ordered[k], ordered[k + 1]);
    }

    std::vector<EdgeKey> edges(edgeSet.begin(), edgeSet.end());
    std::vector<DirEdge> dirs;
    std::vector<std::vector<int>> outgoing(registry.pts.size());
    dirs.reserve(edges.size() * 2);
    for (const auto& e : edges) {
        const Point2& a = registry.pts[e.a];
        const Point2& b = registry.pts[e.b];
        if (same_point2(a, b, eps.split)) continue;
        int i0 = static_cast<int>(dirs.size());
        int i1 = i0 + 1;
        dirs.push_back({e.a, e.b, i1, -1, std::atan2(b.y - a.y, b.x - a.x), false});
        dirs.push_back({e.b, e.a, i0, -1, std::atan2(a.y - b.y, a.x - b.x), false});
        outgoing[e.a].push_back(i0);
        outgoing[e.b].push_back(i1);
    }

    std::vector<int> posInOut(dirs.size(), -1);
    for (int v = 0; v < static_cast<int>(outgoing.size()); ++v) {
        auto& out = outgoing[v];
        std::sort(out.begin(), out.end(), [&](int lhs, int rhs) {
            return dirs[lhs].ang < dirs[rhs].ang;
        });
        for (int i = 0; i < static_cast<int>(out.size()); ++i) posInOut[out[i]] = i;
    }

    for (int ei = 0; ei < static_cast<int>(dirs.size()); ++ei) {
        int v = dirs[ei].to;
        auto& out = outgoing[v];
        if (out.empty()) continue;
        int twinPos = posInOut[dirs[ei].twin];
        int nextPos = (twinPos - 1 + static_cast<int>(out.size())) % static_cast<int>(out.size());
        dirs[ei].next = out[nextPos];
    }

    std::vector<Fragment> fragments;
    for (int ei = 0; ei < static_cast<int>(dirs.size()); ++ei) {
        if (dirs[ei].used) continue;
        std::vector<int> face;
        int cur = ei;
        int guard = 0;
        while (!dirs[cur].used && guard < static_cast<int>(dirs.size()) + 5) {
            dirs[cur].used = true;
            face.push_back(dirs[cur].from);
            cur = dirs[cur].next;
            if (cur == ei) break;
            ++guard;
        }
        if (face.size() < 3 || cur != ei) continue;

        std::vector<Point2> poly2;
        poly2.reserve(face.size());
        for (int id : face) poly2.push_back(registry.pts[id]);
        poly2 = remove_duplicate_ring_points(std::move(poly2), eps.split);
        if (poly2.size() < 3) continue;
        double area = polygon_area2(poly2);
        if (area <= eps.area) continue;
        Point2 c2 = polygon_centroid2(poly2);
        if (!point_in_triangle2(c2, tri2, eps.split)) continue;

        Fragment frag;
        frag.poly2 = poly2;
        frag.poly3.reserve(poly2.size());
        for (const auto& p : poly2) frag.poly3.push_back(basis.unproject(p));
        frag.centroid = basis.unproject(c2);
        fragments.push_back(std::move(frag));
    }

    if (fragments.empty()) {
        Fragment f;
        f.poly2 = {tri2[0], tri2[1], tri2[2]};
        f.poly3 = {tri.p[0], tri.p[1], tri.p[2]};
        f.centroid = tri.centroid;
        fragments.push_back(std::move(f));
    }
    return fragments;
}

struct VertexKey3 {
    std::int64_t x;
    std::int64_t y;
    std::int64_t z;

    bool operator==(const VertexKey3& o) const {
        return x == o.x && y == o.y && z == o.z;
    }
};

struct VertexKey3Hash {
    std::size_t operator()(const VertexKey3& k) const noexcept {
        std::size_t h1 = std::hash<std::int64_t>{}(k.x);
        std::size_t h2 = std::hash<std::int64_t>{}(k.y);
        std::size_t h3 = std::hash<std::int64_t>{}(k.z);
        std::size_t h = h1;
        h ^= h2 + 0x9e3779b97f4a7c15ull + (h << 6) + (h >> 2);
        h ^= h3 + 0x9e3779b97f4a7c15ull + (h << 6) + (h >> 2);
        return h;
    }
};

struct TriKey {
    int a = 0;
    int b = 0;
    int c = 0;

    bool operator==(const TriKey& o) const {
        return a == o.a && b == o.b && c == o.c;
    }
};

struct TriKeyHash {
    std::size_t operator()(const TriKey& k) const noexcept {
        std::size_t h1 = std::hash<int>{}(k.a);
        std::size_t h2 = std::hash<int>{}(k.b);
        std::size_t h3 = std::hash<int>{}(k.c);
        std::size_t h = h1;
        h ^= h2 + 0x9e3779b97f4a7c15ull + (h << 6) + (h >> 2);
        h ^= h3 + 0x9e3779b97f4a7c15ull + (h << 6) + (h >> 2);
        return h;
    }
};

struct VertexWelder {
    double eps;
    std::vector<Vec3> verts;
    std::unordered_map<VertexKey3, std::vector<int>, VertexKey3Hash> buckets;

    explicit VertexWelder(double eps_) : eps(eps_) {}

    VertexKey3 key_for(const Vec3& p) const {
        return {
            static_cast<std::int64_t>(std::floor(p.x / eps)),
            static_cast<std::int64_t>(std::floor(p.y / eps)),
            static_cast<std::int64_t>(std::floor(p.z / eps))
        };
    }

    int add(const Vec3& p) {
        VertexKey3 k = key_for(p);
        for (int dx = -1; dx <= 1; ++dx) {
            for (int dy = -1; dy <= 1; ++dy) {
                for (int dz = -1; dz <= 1; ++dz) {
                    VertexKey3 nk{k.x + dx, k.y + dy, k.z + dz};
                    auto it = buckets.find(nk);
                    if (it == buckets.end()) continue;
                    for (int idx : it->second) {
                        if (same_point(verts[idx], p, eps)) return idx;
                    }
                }
            }
        }
        int id = static_cast<int>(verts.size());
        verts.push_back(p);
        buckets[k].push_back(id);
        return id;
    }
};

struct TriangleEmitter {
    VertexWelder welder;
    std::vector<std::array<int, 3>> faces;
    std::unordered_set<TriKey, TriKeyHash> seen;
    double areaEps;

    TriangleEmitter(double weldEps, double areaEps_) : welder(weldEps), areaEps(areaEps_) {}

    void emit_triangle(const Vec3& a, const Vec3& b, const Vec3& c) {
        if (norm(cross(b - a, c - a)) <= areaEps) return;
        int ia = welder.add(a);
        int ib = welder.add(b);
        int ic = welder.add(c);
        if (ia == ib || ib == ic || ic == ia) return;
        std::array<int, 3> sorted = {ia, ib, ic};
        std::sort(sorted.begin(), sorted.end());
        TriKey key{sorted[0], sorted[1], sorted[2]};
        if (!seen.insert(key).second) return;
        faces.push_back({ia, ib, ic});
    }
};

static bool should_keep_union_face(
    int meshId,
    const Vec3& centroid,
    const Vec3& normal,
    const Mesh& other,
    const SceneEps& eps) {

    Vec3 n = normalized(normal);
    Vec3 sample = centroid + n * eps.delta;
    if (point_inside_mesh(other, sample, eps)) return false;

    Vec3 otherNormal;
    if (point_on_surface(other, centroid, eps.delta, &otherNormal)) {
        if (dot(n, otherNormal) > 0.95) {
            return meshId == 0;
        }
    }
    return true;
}

static void emit_original_triangle(const Triangle& tri, TriangleEmitter& emitter) {
    emitter.emit_triangle(tri.p[0], tri.p[1], tri.p[2]);
}

static void emit_polygon_fragment(const Fragment& frag, TriangleEmitter& emitter, double eps) {
    std::vector<Point2> poly2 = frag.poly2;
    if (polygon_area2(poly2) < 0.0) {
        std::reverse(poly2.begin(), poly2.end());
    }

    auto localTris = ear_clip_triangulate(poly2, eps);
    std::vector<Vec3> poly3 = frag.poly3;
    if (polygon_area2(frag.poly2) < 0.0) std::reverse(poly3.begin(), poly3.end());
    for (const auto& t : localTris) {
        emitter.emit_triangle(poly3[t[0]], poly3[t[1]], poly3[t[2]]);
    }
}

static void emit_uncut_components(
    int meshId,
    const Mesh& mesh,
    const Mesh& other,
    const std::vector<std::vector<Segment3>>& cuts,
    const SceneEps& eps,
    TriangleEmitter& emitter) {

    std::vector<char> visited(mesh.tris.size(), 0);
    std::vector<int> stack;
    for (int i = 0; i < static_cast<int>(mesh.tris.size()); ++i) {
        if (visited[i] || !cuts[i].empty()) continue;
        visited[i] = 1;
        stack.clear();
        stack.push_back(i);
        std::vector<int> component;
        component.push_back(i);

        while (!stack.empty()) {
            int cur = stack.back();
            stack.pop_back();
            for (int nb : mesh.neighbors[cur]) {
                if (nb < 0 || visited[nb] || !cuts[nb].empty()) continue;
                visited[nb] = 1;
                stack.push_back(nb);
                component.push_back(nb);
            }
        }

        const Triangle& seed = mesh.tris[component.front()];
        if (!should_keep_union_face(meshId, seed.centroid, seed.normal, other, eps)) continue;
        for (int tid : component) emit_original_triangle(mesh.tris[tid], emitter);
    }
}

static void emit_cut_triangles(
    int meshId,
    const Mesh& mesh,
    const Mesh& other,
    const std::vector<std::vector<Segment3>>& cuts,
    const SceneEps& eps,
    TriangleEmitter& emitter) {

    for (int i = 0; i < static_cast<int>(mesh.tris.size()); ++i) {
        if (cuts[i].empty()) continue;
        const Triangle& tri = mesh.tris[i];
        std::vector<Fragment> fragments = fragment_triangle(tri, cuts[i], eps);
        for (const auto& frag : fragments) {
            if (!should_keep_union_face(meshId, frag.centroid, tri.normal, other, eps)) continue;
            emit_polygon_fragment(frag, emitter, eps.split);
        }
    }
}

static void collect_intersections(
    const Mesh& a,
    const Mesh& b,
    const SceneEps& eps,
    std::vector<std::vector<Segment3>>& cutsA,
    std::vector<std::vector<Segment3>>& cutsB) {

    if (a.tris.empty() || b.tris.empty()) return;
    std::vector<std::pair<int, int>> stack;
    stack.push_back({a.bvh.root, b.bvh.root});

    while (!stack.empty()) {
        auto [ia, ib] = stack.back();
        stack.pop_back();
        const BVHNode& na = a.bvh.nodes[ia];
        const BVHNode& nb = b.bvh.nodes[ib];
        if (!na.box.overlap(nb.box, eps.split)) continue;

        if (na.leaf && nb.leaf) {
            for (int ai = 0; ai < na.count; ++ai) {
                int ta = a.bvh.order[na.start + ai];
                const Triangle& triA = a.tris[ta];
                for (int bi = 0; bi < nb.count; ++bi) {
                    int tb = b.bvh.order[nb.start + bi];
                    const Triangle& triB = b.tris[tb];
                    if (!triA.box.overlap(triB.box, eps.split)) continue;
                    std::vector<Segment3> cuts = triangle_triangle_cuts(triA, triB, eps);
                    if (cuts.empty()) continue;
                    for (const auto& seg : cuts) {
                        cutsA[ta].push_back(seg);
                        cutsB[tb].push_back(seg);
                    }
                }
            }
            continue;
        }

        int sizeA = na.count;
        int sizeB = nb.count;
        if (!na.leaf && (nb.leaf || sizeA >= sizeB)) {
            stack.push_back({na.left, ib});
            stack.push_back({na.right, ib});
        } else {
            stack.push_back({ia, nb.left});
            stack.push_back({ia, nb.right});
        }
    }
}

static SceneEps make_scene_eps(const Mesh& a, const Mesh& b) {
    AABB box = a.box;
    box.expand(b.box);
    Vec3 e = box.extent();
    double scale = std::max({std::fabs(e.x), std::fabs(e.y), std::fabs(e.z), 1.0});
    SceneEps eps;
    eps.point = 1e-9 * scale;
    eps.split = 5e-8 * scale;
    eps.weld = 1e-8 * scale;
    eps.delta = 1e-6 * scale;
    eps.area = 1e-18 * scale * scale;
    return eps;
}

static Mesh read_mesh_from_stream(std::istream& in, int vCount, int fCount) {
    Mesh mesh;
    mesh.vertices.resize(vCount);
    mesh.faces.resize(fCount);
    for (int i = 0; i < vCount; ++i) {
        in >> mesh.vertices[i].x >> mesh.vertices[i].y >> mesh.vertices[i].z;
        mesh.box.expand(mesh.vertices[i]);
    }
    for (int i = 0; i < fCount; ++i) {
        in >> mesh.faces[i][0] >> mesh.faces[i][1] >> mesh.faces[i][2];
    }
    mesh.tris.reserve(fCount);
    for (const auto& f : mesh.faces) mesh.tris.push_back(make_triangle(mesh.vertices, f));
    build_neighbors(mesh);
    build_bvh(mesh);
    return mesh;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int VA = 0, FA = 0, VB = 0, FB = 0;
    if (!(std::cin >> VA >> FA >> VB >> FB)) {
        return 0;
    }

    Mesh A = read_mesh_from_stream(std::cin, VA, FA);
    Mesh B = read_mesh_from_stream(std::cin, VB, FB);
    SceneEps eps = make_scene_eps(A, B);

    TriangleEmitter emitter(eps.weld, eps.area);

    if (!A.box.overlap(B.box, eps.split)) {
        for (const auto& tri : A.tris) emit_original_triangle(tri, emitter);
        for (const auto& tri : B.tris) emit_original_triangle(tri, emitter);
    } else {
        std::vector<std::vector<Segment3>> cutsA(A.tris.size());
        std::vector<std::vector<Segment3>> cutsB(B.tris.size());
        collect_intersections(A, B, eps, cutsA, cutsB);

        emit_uncut_components(0, A, B, cutsA, eps, emitter);
        emit_uncut_components(1, B, A, cutsB, eps, emitter);
        emit_cut_triangles(0, A, B, cutsA, eps, emitter);
        emit_cut_triangles(1, B, A, cutsB, eps, emitter);
    }

    std::cout << emitter.welder.verts.size() << ' ' << emitter.faces.size() << '\n';
    std::cout << std::setprecision(17);
    for (const Vec3& v : emitter.welder.verts) {
        std::cout << v.x << ' ' << v.y << ' ' << v.z << '\n';
    }
    for (const auto& f : emitter.faces) {
        std::cout << f[0] << ' ' << f[1] << ' ' << f[2] << '\n';
    }
    return 0;
}
