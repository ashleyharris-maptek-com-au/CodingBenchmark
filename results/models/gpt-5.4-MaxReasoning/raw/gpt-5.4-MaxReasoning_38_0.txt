import sys
from functools import lru_cache


def parse_input():
    raw = sys.stdin.buffer.read().decode()
    raw = raw.replace("\r\n", "\n").replace("\r", "\n")
    if not raw.strip():
        return 0, []

    if "\n" in raw:
        header, rest = raw.split("\n", 1)
    else:
        header, rest = raw, ""

    u_str, s_str = header.split()
    universe_size = int(u_str)
    set_count = int(s_str)

    lines = rest.split("\n")
    if len(lines) < set_count:
        lines.extend([""] * (set_count - len(lines)))

    masks = []
    for line in lines[:set_count]:
        mask = 0
        if line.strip():
            for token in line.split():
                elem = int(token)
                if 0 <= elem < universe_size:
                    mask |= 1 << elem
        masks.append(mask)

    return universe_size, masks


def merge_sorted(index, items):
    merged = list(items)
    lo = 0
    hi = len(merged)
    while lo < hi:
        mid = (lo + hi) // 2
        if merged[mid] < index:
            lo = mid + 1
        else:
            hi = mid
    merged.insert(lo, index)
    return tuple(merged)


def solve_exact_cover(universe_size, masks):
    full_mask = (1 << universe_size) - 1
    usable = [(idx, mask) for idx, mask in enumerate(masks) if mask != 0]

    sets_by_element = [[] for _ in range(universe_size)]
    for idx, mask in usable:
        bits = mask
        while bits:
            bit = bits & -bits
            elem = bit.bit_length() - 1
            sets_by_element[elem].append((idx, mask))
            bits ^= bit

    @lru_cache(maxsize=None)
    def search(remaining_mask):
        if remaining_mask == 0:
            return ()

        pivot = -1
        candidates = None

        bits = remaining_mask
        while bits:
            bit = bits & -bits
            elem = bit.bit_length() - 1
            current = []
            for idx, mask in sets_by_element[elem]:
                if mask & remaining_mask == mask:
                    current.append((idx, mask))

            if not current:
                return None

            current.sort()
            if candidates is None or len(current) < len(candidates):
                pivot = elem
                candidates = current
                if len(candidates) == 1:
                    break

            bits ^= bit

        best = None
        for idx, mask in candidates:
            result = search(remaining_mask ^ mask)
            if result is None:
                continue

            merged = merge_sorted(idx, result)
            if best is None or merged < best:
                best = merged

        return best

    return search(full_mask)


def main():
    universe_size, masks = parse_input()
    solution = solve_exact_cover(universe_size, masks)

    out = sys.stdout
    if solution is None:
        out.write("NO SOLUTION\n")
        return

    out.write("SOLUTION\n")
    out.write(" ".join(map(str, solution)))
    out.write("\n")


if __name__ == "__main__":
    main()
