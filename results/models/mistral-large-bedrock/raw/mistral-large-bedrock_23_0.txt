use std::process;
use futures::stream::{self, StreamExt, TryStreamExt};

    let stdout = std::io::stdout();

    let mut reader = BufReader::new(stdin.lock());
    let mut writer = BufWriter::new(stdout.lock());

    let mut lines = reader.lines().map(Result::unwrap);

    let _ = lines.next().await;

    let mut state = lines.next().await.unwrap()

        .collect::<Vec<_>>();

    loop {
        // Calculate control outputs.
        let thrust = if state[3] < 0.0 { 1.0 } else { 0.0 }; // Thrust up if falling.
        let turn = if state[4] > 0.0 { -1.0 } else if state[4] < 0.0 { 1.0 } else { 0.0 }; // Turn to face target.

        // Write outputs to stdout.
        writeln!(writer, "{} {}", thrust, turn).await.unwrap();

            }
        };

}
