#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
#include <cstdint>

using namespace std;

static int n;
static vector<vector<uint64_t>> origAdj; // original graph adjacency (bitsets)
static int L; // number of 64-bit blocks per row

static vector<int> bestSet;
static vector<int> curSet;
static int bestSize = 0;

// Check if there is an edge in the original graph between a and b
inline bool origEdge(int a, int b) {
    if (a == b) return false;
    return (origAdj[a][b >> 6] >> (b & 63)) & 1ull;
}

// Edge in the complement graph (i.e., not connected in original and not the same vertex)
inline bool compEdge(int a, int b) {
    if (a == b) return false;
    return !origEdge(a, b);
}

static void mc(vector<int>& P) {
    if (P.empty()) {
        if ((int)curSet.size() > bestSize) {
            bestSize = (int)curSet.size();
            bestSet = curSet;
        }
        return;
    }

    // Greedy coloring on the subgraph induced by P (in the complement graph)
    vector<int> order = P;
    int m = (int)order.size();
    vector<int> color(m, 0);
    int maxColor = 0;

    for (int i = 0; i < m; ++i) {
        int v = order[i];
        // mark colors used by neighbors among previous vertices
        vector<char> used(maxColor + 1, false);
        for (int j = 0; j < i; ++j) {
            int u = order[j];
            if (compEdge(v, u)) {
                used[color[j]] = true;
            }
        }
        int c;
        for (c = 1; c <= maxColor; ++c) {
            if (!used[c]) break;
        }
        color[i] = c;
        if (c > maxColor) maxColor = c;
    }

    // Process in reverse order
    for (int i = m - 1; i >= 0; --i) {
        int v = order[i];
        if ((int)curSet.size() + color[i] <= bestSize) return;

        curSet.push_back(v);

        // Build P2 = neighbors of v inside order after i (in complement graph)
        vector<int> P2;
        P2.reserve(m - i - 1);
        for (int j = i + 1; j < m; ++j) {
            int u = order[j];
            if (compEdge(v, u)) P2.push_back(u);
        }

        if (P2.empty()) {
            if ((int)curSet.size() > bestSize) {
                bestSize = (int)curSet.size();
                bestSet = curSet;
            }
        } else {
            mc(P2);
        }

        curSet.pop_back();
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int nn;
    long long m;
    if (!(cin >> nn >> m)) return 0;
    n = nn;

    L = (n + 63) / 64;
    origAdj.assign(n, vector<uint64_t>(L, 0ull));

    for (long long i = 0; i < m; ++i) {
        int u, v;
        if (!(cin >> u >> v)) break;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        origAdj[u][v >> 6] |= (1ull << (v & 63));
        origAdj[v][u >> 6] |= (1ull << (u & 63));
    }

    // Initial simple greedy to get a decent starting bound (optional but helpful)
    vector<int> deg(n, 0);
    for (int i = 0; i < n; ++i) {
        int cnt = 0;
        for (int j = 0; j < n; ++j) {
            if (i == j) continue;
            if (origEdge(i, j)) continue;
            // In complement, (i,j) is an edge
            ++cnt;
        }
        deg[i] = (n - 1) - 0; // not used for exact bound here
    }
    // Simple deterministic ordering by original degree (ascending) to build a clique in complement
    vector<pair<int,int>> orderv;
    orderv.reserve(n);
    for (int i = 0; i < n; ++i) {
        int dw = 0;
        // count original degree
        for (int j = 0; j < n; ++j) {
            if (origEdge(i, j)) ++dw;
        }
        orderv.emplace_back(dw, i);
    }
    sort(orderv.begin(), orderv.end());
    vector<int> cliqueCand;
    for (auto &p : orderv) {
        int v = p.second;
        bool ok = true;
        for (int u : cliqueCand) {
            if (origEdge(v, u)) { ok = false; break; }
        }
        if (ok) cliqueCand.push_back(v);
    }
    if ((int)cliqueCand.size() > bestSize) {
        bestSize = (int)cliqueCand.size();
        bestSet = cliqueCand;
    }

    // Start exact search on complement graph
    vector<int> P(n);
    iota(P.begin(), P.end(), 0);
    bestSize = max(bestSize, 0);
    bestSet = bestSet; // keep previous if better
    curSet.clear();
    mc(P);

    // Output result: bestSet is a maximum independent set in the original graph
    cout << bestSet.size() << "\n";
    for (size_t i = 0; i < bestSet.size(); ++i) {
        if (i) cout << " ";
        cout << bestSet[i];
    }
    cout << "\n";

    return 0;
}
