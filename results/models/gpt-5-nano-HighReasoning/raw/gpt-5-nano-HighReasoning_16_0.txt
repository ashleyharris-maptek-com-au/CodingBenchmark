```cpp
#include <bits/stdc++.h>
using namespace std;

// Job-Shop Scheduling Solver with adaptive strategies.
// Input format:
// J M
// For each job j (0..J-1):
//   T_j
//   T_j lines: machine duration   (machine is 0-based, duration is int)
// Output format (one line per job):
//   For each task t in the job: "start_time machine"  (in order)
struct Task {
    int machine;
    int duration;
};

class JobShopSolver {
public:
    int J, M;
    vector<vector<Task>> jobs;      // jobs[j] = list of tasks for job j
    int totalTasks;

    // Best solution found (start times per task)
    vector<vector<long long>> bestStarts;
    long long bestMakespan = (1LL<<62);

    // For exact solver (small instances)
    vector<int> nextIndex;                 // next task index to schedule for each job
    vector<long long> machineFree;           // next free time for each machine
    vector<long long> lastFinish;            // finish time of the last scheduled task for each job
    vector<vector<long long>> startTimes;    // start time results for each task
    vector<long long> remJob;                 // remaining total duration per job
    vector<long long> remMachine;             // remaining total duration per machine
    long long currentMakespan = 0;

    // Constructor
    JobShopSolver(const vector<vector<Task>>& _jobs, int _M) {
        jobs = _jobs;
        J = (int)jobs.size();
        M = _M;
        totalTasks = 0;
        for (auto &v : jobs) totalTasks += (int)v.size();

        bestStarts.clear();
        bestMakespan = (1LL<<62);
    }

    // Public solve interface
    vector<vector<long long>> solve() {
        // Adaptive strategy:
        // - For very small problems (total tasks <= SMALL_N) try exact search (branch-and-bound).
        // - For larger problems, run greedy heuristics (two variants) and take the best.
        const int SMALL_N = 22; // safe limit for exhaustive-like search without heavy time cost

        if (totalTasks <= SMALL_N) {
            // Get a good initial bound with greedy (two variants could help)
            auto greedy0 = runGreedyVariant(0);
            long long g0 = computeMakespan(greedy0);
            auto greedy1 = runGreedyVariant(1);
            long long g1 = computeMakespan(greedy1);

            bestStarts = (g0 <= g1) ? greedy0 : greedy1;
            best MakespanHolder = (g0 <= g1) ? g0 : g1; // placeholder to avoid compile error
            // Note: We can't declare a variable named "best MakespanHolder".
            // We'll simply set bestMakespan using the chosen greedy result:
        }

        // Always prepare a strong greedy bound even for larger instances
        // and use that as initial bound for any potential exact search (if triggered)
        vector<vector<long long>> greedyBest;
        long long greedyBestMakespan = (1LL<<62);
        {
            auto s0 = runGreedyVariant(0);
            long long m0 = computeMakespan(s0);
            auto s1 = runGreedyVariant(1);
            long long m1 = computeMakespan(s1);
            if (m0 <= m1) {
                greedyBest = s0;
                greedyBestMakespan = m0;
            } else {
                greedyBest = s1;
                greedyBestMakespan = m1;
            }
        }
        // If small we may use exact; for simplicity in this implementation,
        // we implement an exact solver only for very small totals and otherwise fall back to the best greedy.
        // We will attempt exact if N <= 22; else we rely on greedyBest.

        bestStarts = greedyBest;
        bestMakespan = greedyBestMakespan;

        // If totalTasks is very small, try an exact search to possibly improve upon greedy bound
        if (totalTasks <= 22) {
            // Initialize exact solver state
            initExactState( /* no bound yet, best bound from greedy */ greedyBestMakespan );
            dfsExact(0);
            // bestStarts and bestMakespan updated inside dfsExact when a better solution is found
        }

        // If we did not compute a best (shouldn't happen), fall back to greedyBest
        if (bestStarts.empty()) {
            bestStarts = greedyBest;
        }

        return bestStarts;
    }

private:
    // Helpers

    // Compute makespan from a complete schedule (start times for every task)
    long long computeMakespan(const vector<vector<long long>>& starts) const {
        long long maxFinish = 0;
        for (int j = 0; j < J; ++j) {
            for (size_t t = 0; t < jobs[j].size(); ++t) {
                long long s = starts[j][t];
                long long f = s + jobs[j][t].duration;
                if (f > maxFinish) maxFinish = f;
            }
        }
        return maxFinish;
    }

    // Greedy variant solver (0 or 1)
    vector<vector<long long>> runGreedyVariant(int variant) {
        // variant 0: tie-breaker minimal finish time, then smaller duration
        // variant 1: tie-breaker minimal finish time, then larger duration (to diversify)
        int N = totalTasks;
        vector<int> nextIndexLocal(J, 0);
        vector<long long> lastFinishLocal(J, 0);
        vector<long long> machineFreeLocal(M, 0);
        vector<vector<long long>> startsLocal(J);
        for (int j = 0; j < J; ++j) startsLocal[j].assign(jobs[j].size(), -1);

        int scheduled = 0;
        // Precompute total durations per job (for bookkeeping)
        vector<long long> remJobLocal(J, 0);
        vector<long long> remMachineLocal(M, 0);
        for (int j = 0; j < J; ++j) {
            long long sum = 0;
            for (auto &tk : jobs[j]) sum += tk.duration;
            remJobLocal[j] = sum;
        }
        for (int m = 0; m < M; ++m) {
            long long sum = 0;
            for (int j = 0; j < J; ++j) {
                for (auto &tk : jobs[j]) if (tk.machine == m) sum += tk.duration;
            }
            remMachineLocal[m] = sum;
        }

        while (scheduled < N) {
            struct Cand { int job; int idx; int m; int d; long long s; long long f; };
            vector<Cand> cands;
            cands.reserve(J);

            for (int j = 0; j < J; ++j) {
                if (nextIndexLocal[j] < (int)jobs[j].size()) {
                    auto &tk = jobs[j][nextIndexLocal[j]];
                    int m = tk.machine;
                    int d = tk.duration;
                    long long s = max(machineFreeLocal[m], lastFinishLocal[j]);
                    long long f = s + d;
                    cands.push_back({j, nextIndexLocal[j], m, d, s, f});
                }
            }

            if (cands.empty()) break; // should not happen

            long long bestF = LLONG_MAX;
            for (auto &c : cands) if (c.f < bestF) bestF = c.f;
            vector<Cand> minFCands;
            for (auto &c : cands) if (c.f == bestF) minFCands.push_back(c);

            Cand chosen = minFCands[0];
            if (variant == 0) {
                // among minF, pick smallest duration, then smallest start
                for (auto &c : minFCands) {
                    if (c.d < chosen.d || (c.d == chosen.d && c.s < chosen.s)) chosen = c;
                }
            } else {
                // variant 1: pick largest duration among those with minF (to diversify)
                for (auto &c : minFCands) {
                    if (c.d > chosen.d || (c.d == chosen.d && c.s < chosen.s)) chosen = c;
                }
            }

            int j = chosen.job;
            int idx = chosen.idx;
            int m = chosen.m;
            int d = chosen.d;
            long long s = chosen.s;
            long long f = chosen.f;

            // record
            startsLocal[j][idx] = s;

            // update state
            long long oldMachineFree = machineFreeLocal[m];
            long long oldLastFinish = lastFinishLocal[j];
            long long oldRemJob = remJobLocal[j];
            long long oldRemMachine = remMachineLocal[m];
            int oldNextIndex = nextIndexLocal[j];

            machineFreeLocal[m] = f;
            lastFinishLocal[j] = f;
            remJobLocal[j] -= d;
            remMachineLocal[m] -= d;
            nextIndexLocal[j] = oldNextIndex + 1;

            scheduled++;
        }

        return startsLocal;
    }

    // Initialize exact solver state using an initial bound (optional)
    void initExactState(long long initialBound) {
        // Prepare state arrays for exact search
        nextIndex.assign(J, 0);
        machineFree.assign(M, 0);
        lastFinish.assign(J, 0);
        startTimes.assign(J, {});
        for (int j = 0; j < J; ++j) {
            startTimes[j].assign((int)jobs[j].size(), -1);
        }
        remJob.assign(J, 0);
        remMachine.assign(M, 0);
        currentMakespan = 0;

        for (int j = 0; j < J; ++j) {
            long long sum = 0;
            for (auto &tk : jobs[j]) sum += tk.duration;
            remJob[j] = sum;
        }
        for (int m = 0; m < M; ++m) remMachine[m] = 0;
        for (int j = 0; j < J; ++j) {
            for (size_t t = 0; t < jobs[j].size(); ++t) {
                remMachine[jobs[j][t].machine] += jobs[j][t].duration;
            }
        }

        bestMakespan = initialBound;
        bestStarts = bestStarts; // will be updated only if we find better
        // Note: bestStarts will be updated in dfsExact when a better solution is found.
    }

    // Exact solver (branch-and-bound) for small instances
    void dfsExact(int scheduledCount) {
        if (scheduledCount == totalTasks) {
            long long makespan = currentMakespan;
            if (makespan < bestMakespan) {
                bestMakespan = makespan;
                bestStarts = startTimes;
            }
            return;
        }

        // Lower bound pruning
        long long maxJobLB = 0;
        for (int j = 0; j < J; ++j) {
            long long lb = lastFinish[j] + remJob[j];
            if (lb > maxJobLB) maxJobLB = lb;
        }
        long long maxMachLB = 0;
        for (int m = 0; m < M; ++m) {
            long long lb = machineFree[m] + remMachine[m];
            if (lb > maxMachLB) maxMachLB = lb;
        }
        long long LB = max(maxJobLB, maxMachLB);
        if (LB >= bestMakespan) return;

        // Generate candidates
        struct Cand { int job; int idx; int m; int d; long long s; long long f; };
        vector<Cand> cands;
        cands.reserve(J);

        for (int j = 0; j < J; ++j) {
            if (nextIndex[j] < (int)jobs[j].size()) {
                auto &tk = jobs[j][nextIndex[j]];
                int m = tk.machine;
                int d = tk.duration;
                long long s = max(machineFree[m], lastFinish[j]);
                long long f = s + d;
                cands.push_back({j, nextIndex[j], m, d, s, f});
            }
        }
        if (cands.empty()) return; // should not happen

        // Sort candidates by a simple heuristic: earliest finish first
        sort(cands.begin(), cands.end(), [](const Cand& a, const Cand& b){
            if (a.f != b.f) return a.f < b.f;
            if (a.s != b.s) return a.s < b.s;
            if (a.d != b.d) return a.d < b.d;
            return a.job < b.job;
        });

        // Explore in that order (could be augmented with more heuristics or randomness)
        for (size_t i = 0; i < cands.size(); ++i) {
            const Cand &c = cands[i];
            int j = c.job;
            int idx = c.idx;
            int m = c.m;
            int d = c.d;
            long long s = c.s;
            long long f = c.f;

            // Save state
            long long oldMachineFree = machineFree[m];
            long long oldLastFinish = lastFinish[j];
            long long oldRemJob = remJob[j];
            long long oldRemMachine = remMachine[m];
            int oldNextIndex = nextIndex[j];
            long long oldCurrentMakespan = currentMakespan;

            // Apply
            startTimes[j][idx] = s;
            machineFree[m] = f;
            lastFinish[j] = f;
            remJob[j] -= d;
            remMachine[m] -= d;
            nextIndex[j] = oldNextIndex + 1;
            if (f > currentMakespan) currentMakespan = f;

            dfsExact(scheduledCount + 1);

            // Revert
            startTimes[j][idx] = -1;
            machineFree[m] = oldMachineFree;
            lastFinish[j] = oldLastFinish;
            remJob[j] = oldRemJob;
            remMachine[m] = oldRemMachine;
            nextIndex[j] = oldNextIndex;
            currentMakespan = oldCurrentMakespan;
        }
    }
};

            taskswitch:
            // Ensure machine index is valid (0-based). If input uses 1-based, adjust here.

        if (j + 1 < J) cout << "\n";
    }

    return 0;
```
