```cpp
#include <bits/stdc++.h>
using namespace std;

// We'll implement Dreyfus-Wagner exact DP for Steiner Tree problem
// with path reconstruction. Input format:
// n m K
// u v w   (undirected edges, 0-based indexing)
// t1 t2 ... tK  (terminals)
//
// Output:
// total_weight edge_count
// u1 v1
// ...

static inline uint64_t edge_key(int a, int b) {
    int u = a, v = b;
    if (u > v) swap(u, v);
    return (static_cast<uint64_t>(u) << 32) | static_cast<uint32_t>(v);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, K;
    if (!(cin >> n >> m >> K)) {
        return 0;
    }

    vector<vector<pair<int,int>>> adj(n);
    unordered_map<uint64_t, long long> edgeW; // to store min weight for an edge pair

    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        adj[u].push_back({v, (int)w});
        adj[v].push_back({u, (int)w});
        uint64_t key = edge_key(u, v);
        auto it = edgeW.find(key);
        if (it == edgeW.end()) edgeW[key] = w;
        else if (w < it->second) it->second = w;
    }

    vector<int> terminals(K);
    for (int i = 0; i < K; ++i) cin >> terminals[i];

    if (K == 0) {
        cout << 0 << " " << 0 << "\n";
        return 0;
    }

    int maxMask = 1 << K;
    const long long INF = (1LL << 62);

    // DP tables
    // dp[mask][v] = minimum cost to connect terminals in mask and end at vertex v
    vector<vector<long long>> dp(maxMask, vector<long long>(n, INF));
    // Tracing information for reconstruction
    // trace_type: BASE (-1), MERGE (0), RELAX (1), undefined (-2)
    vector<vector<signed char>> trace_type(maxMask, vector<signed char>(n, -2));
    vector<vector<int>> trace_split(maxMask, vector<int>(n, -1)); // valid for MERGE
    vector<vector<int>> trace_prev(maxMask, vector<int>(n, -1));  // valid for RELAX

    // Helper constants
    const signed char BASE = -1;
    const signed char MERGE = 0;
    const signed char RELAX = 1;

    // Dijkstra for a fixed mask using current dp[mask][*] as initial distances
    auto runDijkstra = [&](int mask) {
        using P = pair<long long,int>;
        priority_queue<P, vector<P>, greater<P>> pq;
        for (int v = 0; v < n; ++v) {
            if (dp[mask][v] < INF) pq.emplace(dp[mask][v], v);
        }
        while (!pq.empty()) {
            auto [dist, u] = pq.top(); pq.pop();
            if (dist != dp[mask][u]) continue;
            for (auto &e : adj[u]) {
                int to = e.first;
                long long nd = dist + e.second;
                if (nd < dp[mask][to]) {
                    dp[mask][to] = nd;
                    trace_type[mask][to] = RELAX;
                    trace_prev[mask][to] = u;
                    pq.emplace(nd, to);
                }
            }
        }
    };

    // Helper to fetch the bit position for a single-bit mask
    auto bitpos = [&](int mask)->int {
        int idx = 0;
        while (((mask >> idx) & 1) == 0) ++idx;
        return idx;
    };

    // Build edges by reconstructing from dp states
    unordered_set<uint64_t> edgeSet;
    vector<vector<char>> built(maxMask, vector<char>(n, 0)); // memoization for reconstruction

    function<void(int,int)> collect_edges;
    collect_edges = [&](int mask, int v) {
        if (built[mask][v]) return;
        built[mask][v] = 1;

        signed char t = trace_type[mask][v];
        if (t == BASE) {
            // base terminal node, no edges added here
            return;
        } else if (t == MERGE) {
            int s = trace_split[mask][v];
            int other = mask ^ s;
            collect_edges(s, v);
            collect_edges(other, v);
            // merging at v adds no new edge
            return;
        } else if (t == RELAX) {
            int u = trace_prev[mask][v];
            collect_edges(mask, u);
            uint64_t key = edge_key(u, v);
            edgeSet.insert(key);
            return;
        } else {
            // undefined path; ignore
            return;
        }
    };

    // Initialize DP
    for (int mask = 1; mask < maxMask; ++mask) {
        // Reset current row
        for (int v = 0; v < n; ++v) {
            dp[mask][v] = INF;
            trace_type[mask][v] = -2;
            trace_prev[mask][v] = -1;
            trace_split[mask][v] = -1;
        }

        if ((mask & (mask - 1)) == 0) {
            // Single terminal: start at its vertex with 0 cost
            int idx = bitpos(mask);
            int t = terminals[idx];
            dp[mask][t] = 0;
            trace_type[mask][t] = BASE;
            trace_prev[mask][t] = -1;
            runDijkstra(mask);
        } else {
            // Merge step: try all submasks
            for (int v = 0; v < n; ++v) {
                long long best = dp[mask][v];
                int best_s = -1;
                // Enumerate non-empty proper submasks
                for (int sub = (mask - 1) & mask; sub; sub = (sub - 1) & mask) {
                    int other = mask ^ sub;
                    long long val = dp[sub][v] + dp[other][v];
                    if (val < best) {
                        best = val;
                        best_s = sub;
                    }
                }
                if (best_s != -1) {
                    dp[mask][v] = best;
                    trace_type[mask][v] = MERGE;
                    trace_split[mask][v] = best_s;
                }
            }
            runDijkstra(mask);
        }
    }

    int fullMask = maxMask - 1;
    long long bestTotal = INF;
    int bestV = -1;
    for (int v = 0; v < n; ++v) {
        if (dp[fullMask][v] < bestTotal) {
            bestTotal = dp[fullMask][v];
            bestV = v;
        }
    }

    if (bestV == -1 || bestTotal >= INF/2) {
        // No solution (disconnected terminals)
        cout << 0 << " " << 0 << "\n";
        return 0;
    }

    // Reconstruct edges
    edgeSet.clear();
    for (int i = 0; i < maxMask; ++i) {
        fill(built[i].begin(), built[i].end(), 0);
    }
    collect_edges(fullMask, bestV);

    // If we used all edges from edgeSet, we can compute total weight from edge weights
    long long totalWeightFromEdges = 0;
    vector<pair<int,int>> edgeList;
    edgeList.reserve(edgeSet.size());
    for (auto key : edgeSet) {
        int u = int((key >> 32) & 0xFFFFFFFF);
        int v = int(key & 0xFFFFFFFF);
        edgeList.push_back({u, v});
        long long w = 0;
        auto it = edgeW.find(key);
        if (it != edgeW.end()) w = it->second;
        else {
            // Should not happen; fall back to scanning adj
            // Pick min weight among edges between u-v in adjacency
            long long mn = LLONG_MAX;
            for (auto &e : adj[u]) {
                if (e.first == v) mn = min(mn, (long long)e.second);
            }
            if (mn != LLONG_MAX) w = mn;
        }
        totalWeightFromEdges += w;
    }

    // The Steiner tree total weight should equal bestTotal; for safety, use bestTotal
    // Output as required
    cout << bestTotal << " " << edgeList.size() << "\n";
    for (auto &e : edgeList) {
        cout << e.first << " " << e.second << "\n";
    }

    return 0;
}
```
