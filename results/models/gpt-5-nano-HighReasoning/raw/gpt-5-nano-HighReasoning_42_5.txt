; SPIR-V geometry shader (Vulkan 1.0 compatible)
; Draw 8 radial line segments from center outward as shafts with arrowheads
; Input: input primitive = InputLines (2 vertices per invocation)
; Output: OutputTriangleStrip, max 7 vertices (shaft 4 + arrow 3)

OpCapability Geometry
OpMemoryModel Logical GLSL450

; GLSL std450 extension needed for sqrt and related math
%GLSL450 = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0

; Mat4 (for a minimal UBO block)
%mat4 = OpTypeMatrix %v4float 4
; Uniform Block (UBO)
%UBOBlock = OpTypeStruct %mat4
OpDecorate %UBOBlock Block

; Uniform block pointer
%UBOPtr = OpTypePointer Uniform %UBOBlock
%UBO = OpVariable %UBOPtr Uniform
OpDecorate %UBO DescriptorSet 0
OpDecorate %UBO Binding 0

; Inputs: worldPos[] and color[] (arrays sized to input primitive, here 2)
%const_2 = OpConstant %uint 2
[arr_index_none] = OpLine;  ; keep line numbers clean for readability (no-op)

%arr_v3_2 = OpTypeArray %v3float %const_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_2
 worldPos = OpVariable %ptr_in_arr_v3 Input
 color = OpVariable %ptr_in_arr_v3 Input

; Pointers to vec3 (per-vertex attributes)
%ptr_in_v3 = OpTypePointer Input %v3float

; Output: gl_PerVertex_out struct and outColor
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color per-vertex
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output
OpDecorate %outColor Location 0

; Entry point function type
%void = OpTypeVoid
%func_void = OpTypeFunction %void

; Forward declare main so OpEntryPoint can reference it after function is defined
%main = OpFunction %void None %func_void
%main_label = OpLabel

; Execution modes for the entry point (per requirements)
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

; The actual shader implementation
; Local constants
%one = OpConstant %float 1.0
%thick = OpConstant %float 0.015
%arrow = OpConstant %float 0.05
%three_quarter = OpConstant %float 0.75

; Shared temporaries (p0, p1, c0, c1, etc.)
; Load endpoints: worldPos[0], worldPos[1]
%p0_vec3_ptr = OpAccessChain %ptr_in_v3 worldPos %uint_0
%p0_vec3 = OpLoad %v3float %p0_vec3_ptr

%p1_vec3_ptr = OpAccessChain %ptr_in_v3 worldPos %uint_1
%p1_vec3 = OpLoad %v3float %p1_vec3_ptr

%c0_ptr = OpAccessChain %ptr_in_v3 color %uint_0
%c0_vec3 = OpLoad %v3float %c0_ptr

% c1
%c1_ptr = OpAccessChain %ptr_in_v3 color %uint_1
%c1_vec3 = OpLoad %v3float %c1_ptr

; Convert to vec4 with w = 1.0
%p0x = OpCompositeExtract %float %p0_vec3 0
%p0y = OpCompositeExtract %float %p0_vec3 1
%p0z = OpCompositeExtract %float %p0_vec3 2
%p0_vec4 = OpCompositeConstruct %v4float %p0x %p0y %p0z %one

%p1x = OpCompositeExtract %float %p1_vec3 0
%p1y = OpCompositeExtract %float %p1_vec3 1
%p1z = OpCompositeExtract %float %p1_vec3 2
%p1_vec4 = OpCompositeConstruct %v4float %p1x %p1y %p1z %one

; Midpoint: p0 + 0.75*(p1 - p0)
%dx = OpFSub %float %p1x %p0x
%dy = OpFSub %float %p1y %p0y
%dz = OpFSub %float %p1z %p0z

%dx_scaled = OpFMul %float %dx %three_quarter
%dy_scaled = OpFMul %float %dy %three_quarter
%dz_scaled = OpFMul %float %dz %three_quarter

%mid_x = OpFAdd %float %p0x % %dx_scaled
%mid_y = OpFAdd %float %p0y % %dy_scaled
%mid_z = OpFAdd %float %p0z % %dz_scaled

; Perpendicular in XY: perp = normalize(-dir.y, dir.x)
%neg_dy = OpFNegate %float %dy
; length = sqrt(dx*dx + dy*dy)
%dx2 = OpFMul %float %dx %dx
%dy2 = OpFMul %float %dy %dy
%len_sum = OpFAdd %float %dx2 %dy2
%len = OpExtInst %float %GLSL450 Sqrt %len_sum
%inv_len = OpFDiv %float %one %len
%perp_x = OpFMul %float %neg_dy % %inv_len
%perp_y = OpFMul %float %dx % %inv_len

; Shaft offset (0.015)
%ox = OpFMul %float %perp_x % thick
%oy = OpFMul %float %perp_y % thick

; Shaft vertices:
; v0 = p0 + (ox, oy, 0)
%v0x = OpFAdd %float %p0x % %ox
%v0y = OpFAdd %float %p0y % %oy
%v0z = %p0z
%v0 = OpCompositeConstruct %v4float %v0x %v0y %v0z %one

; v1 = p0 - (ox, oy, 0)
%v1x = OpFSub %float %p0x % %ox
%v1y = OpFSub %float %p0y % %oy
%v1z = %p0z
%v1 = OpCompositeConstruct %v4float %v1x %v1y %v1z %one

; v2 = mid + (ox, oy, 0)
%v2x = OpFAdd %float %mid_x % %ox
%v2y = OpFAdd %float %mid_y % %oy
%v2z = %mid_z
%v2 = OpCompositeConstruct %v4float %v2x %v2y %v2z %one

; v3 = mid - (ox, oy, 0)
%v3x = OpFSub %float %mid_x % %ox
%v3y = OpFSub %float %mid_y % %oy
%v3z = %mid_z
%v3 = OpCompositeConstruct %v4float %v3x %v3y %v3z %one

; Arrowhead offset (0.05 along perp):
%ox2 = OpFMul %float %perp_x % arrow
%oy2 = OpFMul %float %perp_y % arrow

; v4 = mid + (ox2, oy2, 0)
%v4x = OpFAdd %float %mid_x % %ox2
%v4y = OpFAdd %float %mid_y % %oy2
%v4z = %mid_z
%v4 = OpCompositeConstruct %v4float %v4x %v4y %v4z %one

; v5 = mid - (ox2, oy2, 0)
%v5x = OpFSub %float %mid_x % %ox2
%v5y = OpFSub %float %mid_y % %oy2
%v5z = %mid_z
%v5 = OpCompositeConstruct %v4float %v5x %v5y %v5z %one

; v6 = p1 (tip)
%v6 = %p1_vec4

; Emit shaft (4 verts) with color c0
; v0
; gl_Position = v0
%gl_pos_ptr0 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_pos_ptr0 %v0
; outColor = c0
OpStore %outColor %c0_vec3
OpEmitVertex

; v1
%gl_pos_ptr1 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_pos_ptr1 %v1
OpStore %outColor %c0_vec3
OpEmitVertex

; v2
%gl_pos_ptr2 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_pos_ptr2 %v2
OpStore %outColor %c0_vec3
OpEmitVertex

; v3
%gl_pos_ptr3 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_pos_ptr3 %v3
OpStore %outColor %c0_vec3
OpEmitVertex

; End shaft primitive
OpEndPrimitive

; Emit arrowhead (v4, v5, v6) with color c1
; v4
%gl_pos_ptr4 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_pos_ptr4 %v4
OpStore %outColor %c1_vec3
OpEmitVertex

; v5
%gl_pos_ptr5 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_pos_ptr5 %v5
OpStore %outColor %c1_vec3
OpEmitVertex

; v6
%gl_pos_ptr6 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_pos_ptr6 %v6
OpStore %outColor %c1_vec3
OpEmitVertex

; End arrow primitive
OpEndPrimitive

; Done emitting
OpReturn
OpFunctionEnd

; Entry point binding
OpEntryPoint Geometry %main "main" %worldPos %color %outColor
