```rust
use std::io::{self, Read};

fn main() {
    // Read entire stdin
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    // Read number of vertices
    let n_opt = it.next();
    if n_opt.is_none() {
        println!("0");
        println!();
        return;
    }
    let n: usize = n_opt.unwrap().parse().unwrap();

    // Read number of edges
    let m: usize = it.next().unwrap_or("0").parse().unwrap();

    // Build adjacency list (undirected)
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u < n && v < n {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    // Greedy dominating set heuristic
    let mut dominated = vec![false; n];
    let mut chosen: Vec<usize> = Vec::new();

    if n > 0 {
        loop {
            // If all vertices are dominated, stop
            if dominated.iter().all(|&d| d) {
                break;
            }

            // Pick vertex that maximizes the number of undominated vertices in its closed neighborhood
            let mut best_i = 0usize;
            let mut best_g = 0usize;

            for i in 0..n {
                let mut g = 0usize;
                if !dominated[i] { g += 1; } // i itself
                for &nb in &adj[i] {
                    if !dominated[nb] { g += 1; }
                }
                if g > best_g {
                    best_g = g;
                    best_i = i;
                }
            }

            // Add best_i to the dominating set
            chosen.push(best_i);
            // Mark its closed neighborhood as dominated
            if !dominated[best_i] {
                dominated[best_i] = true;
            }
            for &nb in &adj[best_i] {
                dominated[nb] = true;
            }
        }
    }

    // Output
    println!("{}", chosen.len());
    let mut line = String::new();
    for (idx, v) in chosen.iter().enumerate() {
        if idx > 0 {
            line.push(' ');
        }
        line.push_str(&v.to_string());
    }
    println!("{}", line);
}
```
