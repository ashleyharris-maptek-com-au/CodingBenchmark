Below is a complete, compilable C++20 solution implementing a scalable heuristic for the 2D Cutting Stock / Rectangle Bin Packing problem.

Key points:
- Uses a shelf-based greedy packing heuristic (guillotine-like). It is fast and scales well to tens of thousands of rectangles.
- Rectangles are allowed to rotate by 90 degrees.
- Processes rectangles in decreasing order of their largest side to improve packing quality.
- Tries to place in the current (last) board first, creating new shelves as needed; if the current board cannot accommodate, a new board is started.
- Outputs: number of boards used, followed by per-rectangle placements (board_index, x, y, rotated).

Notes:
- Board indices are 0-based in the output.
- Positions (x, y) are the top-left coordinates of the placed rectangle (origin (0,0) is board lower-left; y grows upwards in the shelf coordinate system as implemented).
- The algorithm is heuristic and aims to produce valid placements with minimal boards in a scalable manner.

Compile with MSVC or any standard C++20 capable compiler.

Code:

```cpp
#include <bits/stdc++.h>
using namespace std;

// Lightweight 2D shelf-based packing (greedy heuristic)

struct Rect {
    int w, h;
    int idx; // original index
};

struct Shelf {
    int height; // shelf height (max rectangle height in this shelf)
    int nextX;  // x position for next rectangle on this shelf
    int startY; // y position of this shelf (its vertical start)
};

struct Board {
    int width;
    int height;
    int usedHeight;           // total height used by shelves so far
    vector<Shelf> shelves;      // shelves on this board
};

struct Placement {
    int board;
    int x;
    int y;
    int rot; // 0 = not rotated, 1 = rotated
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    int boardW, boardH;
    if (!(cin >> N >> boardW >> boardH)) {
        return 0;
    }

    vector<Rect> rects;
    rects.reserve(N);
    for (int i = 0; i < N; ++i) {
        int w, h;
        cin >> w >> h;
        rects.push_back({w, h, i});
    }

    // Sort rectangles by decreasing largest side (and then by decreasing smallest side as tie-break)
    auto keyCmp = [](const Rect &a, const Rect &b) {
        int ma = max(a.w, a.h);
        int mb = max(b.w, b.h);
        if (ma != mb) return ma > mb;
        int sa = min(a.w, a.h);
        int sb = min(b.w, b.h);
        if (sa != sb) return sa > sb;
        return a.idx < b.idx;
    };
    sort(rects.begin(), rects.end(), keyCmp);

    vector<Board> boards;
    vector<Placement> placements(N);

    // Helper to place rectangle on current (last) board using given orientation
    // If placement is possible, fill placements[idx] and update board; returns true if placed
    auto placeOnLastBoard = [&](int rw, int rh, int orient, int rectIdx, int origIdx)->bool {
        if (boards.empty()) return false;
        Board &b = boards.back();
        // If there is at least one shelf, try to place on the last shelf
        if (!b.shelves.empty()) {
            Shelf &s = b.shelves.back();
            if (rh <= s.height && s.nextX + rw <= b.width) {
                int x = s.nextX;
                int y = s.startY;
                s.nextX += rw;
                placements[origIdx] = { (int)boards.size()-1, x, y, orient };
                return true;
            }
        }
        // Try to start a new shelf on this board
        if (b.usedHeight + rh <= b.height && rw <= b.width) {
            Shelf ns{rh, rw, b.usedHeight}; // new shelf starts at current usedHeight
            b.shelves.push_back(ns);
            int x = 0;
            int y = ns.startY;
            b.usedHeight += rh;
            placements[origIdx] = { (int)boards.size()-1, x, y, orient };
            return true;
        }
        return false;
    };

    // Main packing loop
    for (const auto &r : rects) {
        bool placed = false;

        // Try two orientations for current rectangle
        for (int orient = 0; orient < 2 && !placed; ++orient) {
            int rw = (orient == 0 ? r.w : r.h);
            int rh = (orient == 0 ? r.h : r.w);

            if (rw > boardW || rh > boardH) continue; // cannot fit in any board dimension

            // Try to place on the current last board (fast path)
            if (!boards.empty()) {
                if (placeOnLastBoard(rw, rh, orient, r.idx, r.idx)) {
                    placed = true;
                    break;
                }
            }

            // If not placed on current board, and we have at least one board, we could start a new shelf on it
            if (!placed && !boards.empty()) {
                Board &b = boards.back();
                if (b.usedHeight + rh <= b.height && rw <= b.width) {
                    Shelf ns{rh, rw, b.usedHeight};
                    b.shelves.push_back(ns);
                    int x = 0;
                    int y = ns.startY;
                    b.usedHeight += rh;
                    placements[r.idx] = { (int)boards.size()-1, x, y, orient };
                    placed = true;
                    break;
                }
            }

            // If still not placed, try to create/use a new board
            if (!placed) {
                // Create a new board and place rectangle as the first shelf
                if (rw <= boardW && rh <= boardH) {
                    Board nb;
                    nb.width = boardW;
                    nb.height = boardH;
                    nb.usedHeight = 0;
                    nb.shelves.clear();
                    Shelf ns{rh, rw, 0}; // first shelf at y=0, height rh, used width rw
                    nb.shelves.push_back(ns);
                    nb.usedHeight += rh;
                    boards.push_back(std::move(nb));
                    int bi = (int)boards.size() - 1;
                    placements[r.idx] = { bi, 0, 0, orient };
                    placed = true;
                    break;
                }
            }
        }

        // If not placed after both orientations and creating new boards (edge case), we skip (shouldn't happen if inputs fit)
        if (!placed) {
            // As a safe fallback (should not normally happen with valid inputs), place on a fresh board with orientation 0 if possible
            int rw = r.w, rh = r.h;
            if (rw <= boardW && rh <= boardH) {
                Board nb;
                nb.width = boardW;
                nb.height = boardH;
                nb.usedHeight = 0;
                nb.shelves.clear();
                Shelf ns{rh, rw, 0};
                nb.shelves.push_back(ns);
                nb.usedHeight += rh;
                boards.push_back(std::move(nb));
                int bi = (int)boards.size() - 1;
                placements[r.idx] = { bi, 0, 0, 0 };
            }
        }
    }

    // Output
    cout << boards.size() << "\n";
    for (int i = 0; i < N; ++i) {
        const Placement &p = placements[i];
        cout << p.board << " " << p.x << " " << p.y << " " << p.rot << "\n";
    }

    return 0;
}
```

How it works at a glance:
- Rectangles are sorted by the largest side to prioritize packing large pieces first.
- A board consists of shelves; each shelf has a fixed height (the maximum height of rectangles assigned to that shelf so far) and a nextX pointer indicating how far we’ve used the shelf horizontally.
- For each rectangle (with possible rotation), we try to place it on the current (last) board:
  - If it fits on the last shelf, place it there.
  - Otherwise, if it fits, start a new shelf on the current board and place it.
  - If the current board cannot accommodate it, we create a new board and place it as the first shelf of that board.
- This approach minimizes the number of boards by filling the current board as much as possible before opening a new board, while remaining fast enough for large problem sizes.
