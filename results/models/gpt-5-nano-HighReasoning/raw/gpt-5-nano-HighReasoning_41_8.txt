; Constants
%float_0_0 = OpConstant %float 0.0
%float_1_0 = OpConstant %float 1.0
%float_0_25 = OpConstant %float 0.25
%float_0_5 = OpConstant %float 0.5

%float_1_0_ = OpConstant %float 1.0

; Function and entry
%main = OpFunction %void None %void_func
%void_func = OpTypeFunction %void
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %vec3 %in_worldPos
%normal = OpLoad %vec3 %in_normal
%color = OpLoad %vec3 %in_color

; Load light position from UBO
%lightPos_ptr = OpAccessChain %vec4Ptr %ubo %idx3
%lightPos_vec4 = OpLoad %vec4 %lightPos_ptr
%lp_x = OpCompositeExtract %float %lightPos_vec4 0
%lp_y = OpCompositeExtract %float %lightPos_vec4 1
%lp_z = OpCompositeExtract %float %lightPos_vec4 2
%lightPos_xyz = OpCompositeConstruct %vec3 %lp_x %lp_y %lp_z

; Compute L = normalize(lightPos - worldPos)
%dir = OpFSub %vec3 %lightPos_xyz %worldPos
%dx = OpCompositeExtract %float %dir 0
%dy = OpCompositeExtract %float %dir 1
%dz = OpCompositeExtract %float %dir 2
%dx2 = OpFMul %float %dx %dx
%dy2 = OpFMul %float %dy %dy
%dz2 = OpFMul %float %dz %dz
%dirLen2 = OpFAdd %float %dx2 %dy2
%dirLen2 = OpFAdd %float %dirLen2 %dz2
%dirLen = OpExtInst %float %GLSLstd450 Sqrt %dirLen2
%invDirLen = OpFDiv %float %float_1_0 %dirLen
%Lx = OpFMul %float %dx %invDirLen
%Ly = OpFMul %float %dy %invDirLen
%Lz = OpFMul %float %dz %invDirLen
%L = OpCompositeConstruct %vec3 %Lx %Ly %Lz

; Compute N = normalize(normal)
$nx = OpCompositeExtract %float %normal 0
$ny = OpCompositeExtract %float %normal 1
$nz = OpCompositeExtract %float %normal 2
$nx2 = OpFMul %float $nx $nx
$ny2 = OpFMul %float $ny $ny
$nz2 = OpFMul %float $nz $nz
$lenN2 = OpFAdd %float $nx2 %ny2
$lenN2 = OpFAdd %float $lenN2 $nz2
$lenN = OpExtInst %float %GLSLstd450 Sqrt %lenN2
$invNLen = OpFDiv %float %float_1_0 %lenN
$nX = OpFMul %float $nx $invNLen
$nY = OpFMul %float $ny $invNLen
$nZ = OpFMul %float $nz $invNLen
$N = OpCompositeConstruct %vec3 $nX $nY $nZ

; Dot product d = max(dot(N,L),0)
$Lx2 = OpCompositeExtract %float %L 0
$Ly2 = OpCompositeExtract %float %L 1
$Lz2 = OpCompositeExtract %float %L 2
$dotNL = OpFMul %float $nX $Lx2
$temp = OpFAdd %float $dotNL % (OpFMul %float $nY $Ly2)
$dotNL = OpFAdd %float $temp (OpFMul %float $nZ $Lz2)

$zero = %float_0_0
$isPos = OpFOrdGreaterThan %bool %dotNL %zero
$d = OpSelect %float %isPos %dotNL %zero

; Toon band quantization using nested selects: 0.75, 0.5, 0.25 thresholds
%cond75 = OpFOrdGreaterThan %bool %d %float_0_75
%cond50 = OpFOrdGreaterThan %bool %d %float_0_5
%cond25 = OpFOrdGreaterThan %bool %d %float_0_25
%bandInner = OpSelect %float %cond25 %float_0_4 %float_0_2
%bandMid = OpSelect %float %cond50 %float_0_7 %bandInner
%band = OpSelect %float %cond75 %float_1_0 %bandMid

; Multiply color by band
%R = OpFMul %float (OpCompositeExtract %float %color 0) %band
%G = OpFMul %float (OpCompositeExtract %float %color 1) %band
%B = OpFMul %float (OpCompositeExtract %float %color 2) %band
%out = OpCompositeConstruct %vec4 %R %G %B %float_1_0
OpStore %outColor %out

OpReturn
