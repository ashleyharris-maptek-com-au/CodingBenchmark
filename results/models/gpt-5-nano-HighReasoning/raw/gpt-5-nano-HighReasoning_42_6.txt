OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%func_type = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; gl_PerVertex_in: struct with gl_Position
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; gl_PerVertex_out: struct with gl_Position
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Basic integer and array sizes
%uint = OpTypeInt 32 0
%uint_3 = OpConstant %uint 3

%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

; Input: gl_in array of gl_PerVertex_in
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Input: position/worldPos and other per-vertex inputs as arrays
%arr_vec3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_vec3 = OpTypePointer Input %arr_vec3
%worldPos = OpVariable %ptr_in_arr_vec3 Input
%normal = OpVariable %ptr_in_arr_vec3 Input
%color = OpVariable %ptr_in_arr_vec3 Input

; Output: per-vertex position (gl_Position)
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
; Output: per-vertex color for fragment shader
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output

; Convenience pointers for later use
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; Decorations for inputs/outputs locations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; Entry point interface (ALL interface variables)
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor

; Execution modes
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Function
%main = OpFunction %void None %func_type
%entry = OpLabel

; Vertex 0
%color0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%color0 = OpLoad %v3float %color0_ptr
OpStore %outColor %color0
%pos_in_ptr0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos_in_ptr0
%pos_out_ptr0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_out_ptr0 %pos0
OpEmitVertex

; Vertex 1
%color1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%color1 = OpLoad %v3float %color1_ptr
OpStore %outColor %color1
%pos_in_ptr1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %pos_in_ptr1
%pos_out_ptr1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_out_ptr1 %pos1
OpEmitVertex

; Vertex 2
%color2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%color2 = OpLoad %v3float %color2_ptr
OpStore %outColor %color2
%pos_in_ptr2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos_in_ptr2
%pos_out_ptr2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_out_ptr2 %pos2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
