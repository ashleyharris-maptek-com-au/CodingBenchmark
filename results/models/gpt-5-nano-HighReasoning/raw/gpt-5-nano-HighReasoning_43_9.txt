struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize the interpolated world-space normal
    float3 N = normalize(input.normal);

    // Compute the light direction (from world position to light position)
    float3 L = normalize(UBO.lightPos.xyz - input.worldPos);

    // Lambertian diffuse term
    float diff = max(dot(N, L), 0.0);

    // Lambertian shading factor as (0.5 + 0.5 * diffuse)
    float lambert = 0.5 + 0.5 * diff;

    // 8x8 checkerboard based on UVs
    int ix = int(input.uv.x * 8.0);
    int iy = int(input.uv.y * 8.0);
    int parity = (ix + iy) & 1;

    float3 colorA = float3(0.9, 0.9, 0.9);
    float3 colorB = float3(0.2, 0.2, 0.2);
    float3 baseColor = (parity == 0) ? colorA : colorB;

    // Apply shading
    float3 finalColor = baseColor * lambert;

    return float4(finalColor, 1.0);
}
