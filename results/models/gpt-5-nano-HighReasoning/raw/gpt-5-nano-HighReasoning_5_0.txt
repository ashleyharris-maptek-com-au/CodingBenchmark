use std::collections::VecDeque;
use std::io::{self, Read};

// Convert (x,y) to a linear index for a grid of given width
fn idx(x: usize, y: usize, width: usize) -> usize {
    y * width + x
}

// Depth-first search for a Hamiltonian path on the grid graph
fn dfs(
    cur: usize,
    depth: usize,
    n: usize,
    adj: &Vec<Vec<usize>>,
    visited: &mut Vec<bool>,
    path: &mut Vec<usize>,
    start_id: usize,
    require_cycle: bool,
) -> bool {
    // If we have visited all non-obstacle cells
    if depth == n {
        if require_cycle {
            // The path must end adjacent to the start
            for &nb in &adj[cur] {
                if nb == start_id {
                    return true;
                }
            }
            return false;
        } else {
            return true;
        }
    }

    // Gather unvisited neighbors as candidates
    let mut cand: Vec<usize> = Vec::new();
    for &nb in &adj[cur] {
        if !visited[nb] {
            cand.push(nb);
        }
    }
    if cand.is_empty() {
        return false;
    }

    // Warnsdorff-like ordering: prefer next steps with fewest onward moves
    cand.sort_by(|a, b| {
        let da = adj[*a].iter().filter(|&&u| !visited[u]).count();
        let db = adj[*b].iter().filter(|&&u| !visited[u]).count();
        da.cmp(&db)
    });

    for &next in &cand {
        visited[next] = true;
        path.push(next);

        // Prune using reachability of remaining unvisited nodes from the new endpoint
        if connectivity_ok(next, visited, adj) {
            if dfs(next, depth + 1, n, adj, visited, path, start_id, require_cycle) {
                return true;
            }
        }

        path.pop();
        visited[next] = false;
    }

    false
}

// Connectivity pruning:
// From the current endpoint (which is "next" in the caller),
// ensure every unvisited node is reachable through a path of unvisited nodes
// starting from the endpoint. If any unvisited node is unreachable, prune.
fn connectivity_ok(endpoint: usize, visited: &Vec<bool>, adj: &Vec<Vec<usize>>) -> bool {
    let n = adj.len();
    let mut seen = vec![false; n];
    let mut q: VecDeque<usize> = VecDeque::new();
    q.push_back(endpoint);
    seen[endpoint] = true;

    while let Some(u) = q.pop_front() {
        for &v in &adj[u] {
            if visited[v] {
                continue; // can't pass through already-visited nodes
            }
            if !seen[v] {
                seen[v] = true;
                q.push_back(v);
            }
        }
    }

    let mut reached = 0usize;
    let mut remaining = 0usize;
    for i in 0..n {
        if visited[i] {
            continue;
        }
        remaining += 1;
        if seen[i] {
            reached += 1;
        }
    }

    reached == remaining
}

fn main() {
    // Read entire input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    // Parse header
    let width: usize = it.next().unwrap().parse().unwrap();
    let height: usize = it.next().unwrap().parse().unwrap();
    let obstacle_count: usize = it.next().unwrap().parse().unwrap();
    let require_cycle_flag: i32 = it.next().unwrap().parse().unwrap();
    let require_cycle = require_cycle_flag != 0;

    // Prepare obstacle grid
    let grid_size = width * height;
    let mut blocked = vec![false; grid_size];
    for _ in 0..obstacle_count {
        let x: usize = it.next().unwrap().parse().unwrap();
        let y: usize = it.next().unwrap().parse().unwrap();
        if x < width && y < height {
            blocked[idx(x, y, width)] = true;
        }
    }

    // Start cell must exist and not be blocked
    if blocked[idx(0, 0, width)] {
        println!("0");
        return;
    }

    // Map non-blocked cells to compact IDs and store coordinates
    let mut id_of = vec![usize::MAX; grid_size];
    let mut coords: Vec<(i32, i32)> = Vec::new();
    let mut id: usize = 0;

    for y in 0..height {
        for x in 0..width {
            let gidx = idx(x, y, width);
            if !blocked[gidx] {
                id_of[gidx] = id;
                coords.push((x as i32, y as i32));
                id += 1;
            }
        }
    }

    let n = id;
    if n == 0 {
        println!("0");
        return;
    }

    // Build adjacency for the non-obstacle grid graph
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    let dirs = [(1i32, 0i32), (-1, 0), (0, 1), (0, -1)];
    for y in 0..height {
        for x in 0..width {
            let gidx = idx(x, y, width);
            let cur_id = id_of[gidx];
            if cur_id == usize::MAX {
                continue;
            }
            for (dx, dy) in &dirs {
                let nx = x as i32 + dx;
                let ny = y as i32 + dy;
                if nx >= 0 && ny >= 0 && (nx as usize) < width && (ny as usize) < height {
                    let nidx = idx(nx as usize, ny as usize, width);
                    let nb_id = id_of[nidx];
                    if nb_id != usize::MAX {
                        adj[cur_id].push(nb_id);
                    }
                }
            }
        }
    }

    // Start node
    let start_id = id_of[idx(0, 0, width)];
    if start_id == usize::MAX {
        println!("0");
        return;
    }

    // Prepare and run DFS with pruning
    let mut visited = vec![false; n];
    let mut path: Vec<usize> = Vec::with_capacity(n);
    visited[start_id] = true;
    path.push(start_id);

    let found = dfs(start_id, 1, n, &adj, &mut visited, &mut path, start_id, require_cycle);

    if found {
        // Output path
        println!("{}", n);
        for &vid in &path {
            let (x, y) = coords[vid];
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}
