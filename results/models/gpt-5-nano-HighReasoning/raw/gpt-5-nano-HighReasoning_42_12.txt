; SPIR-V assembly for a geometry shader that renders silhouette-like thick edges over a 6-vertex adjacency input
; Geometry for InputTrianglesAdjacency, OutputTriangleStrip, 15 vertices total (triangle + 3 edges × 4)

OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%uint_6 = OpConstant %uint 6

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_3 = OpConstant %float 0.3
%float_0_015 = OpConstant %float 0.015

; Vectors
%vec3_one = OpConstantComposite %v3float %float_1 %float_1 %float_1

; UBO (set=0, binding=0) - minimal block to satisfy accessibility (layout can be expanded as needed)
%UBO = OpTypeStruct %v4float
%UBO_ptr = OpTypePointer Uniform %UBO
OpDecorate %UBO Block
%ubo = OpVariable %UBO_ptr Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; gl_PerVertex_in (vertex shader input block)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; gl_PerVertex_out (geometry shader output block)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color (to fragment shader)
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output
OpDecorate %outColor Location 0

; Input arrays exposed to the geometry shader (worldPos, normal, color)
%arr_vec3_6 = OpTypeArray %v3float %uint_6

; worldPos[]
%ptr_in_worldPos = OpTypePointer Input %arr_vec3_6
%worldPos = OpVariable %ptr_in_worldPos Input
OpDecorate %worldPos Location 0

; normal[]
%ptr_in_normal = OpTypePointer Input %arr_vec3_6
%normal = OpVariable %ptr_in_normal Input
OpDecorate %normal Location 1

; color[]
%ptr_in_color = OpTypePointer Input %arr_vec3_6
%color = OpVariable %ptr_in_color Input
OpDecorate %color Location 2

; Entry point
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Execution modes
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

; Interface for entry point
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %outColor

; Function prologue
; (Helper: offset for thick edges in clip space)
%offset_vec = OpCompositeConstruct %v4float %float_0 %float_0_015 %float_0 %float_0

; Offsets for indexing
; (We will load positions and colors by index; indices are constants 0,2,4 etc.)
; Emit the original triangle (first)

; Edge processing: 3 edges with endpoints (0-2), (2-4), (4-0) using adjacent indices (1,3,5)
; Prepare for emission: helper to emit a vertex and set color
; We'll reuse %gl_out (positions) and %outColor (colors)

; 1) Edge 0-2 (endpoints gl_in[0] and gl_in[2], adjacent gl_in[1] - render as quad with vertical offset)
; Load A and B
%A_ptr = OpAccessChain %ptr_to_v4_in %gl_in %int_0 %int_0
%B_ptr = OpAccessChain %ptr_to_v4_in %gl_in %int_2 %int_0
%A = OpLoad %v4float %A_ptr
%B = OpLoad %v4float %B_ptr

; Aprime = A + offset
%Aprime = OpFAdd %v4float %A %offset_vec
%Bprime = OpFAdd %v4float %B %offset_vec

; Vertex 0 of edge quad (A)
%gl_in_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_in_pos_ptr %A
OpStore %outColor %vec3_1_1_1
OpEmitVertex

; Vertex 1 (A')
%gl_in_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_in_pos_ptr %Aprime
OpStore %outColor %vec3_1_1_1
OpEmitVertex

; Vertex 2 (B)
%B_pos = %B_ptr  ; reuse
%B = %B
%gl_in_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_in_pos_ptr %B
OpStore %outColor %vec3_1_1_1
OpEmitVertex

; Vertex 3 (B')
%Bprime_pos = %Bprime
%gl_in_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_in_pos_ptr %Bprime
OpStore %outColor %vec3_1_1_1
OpEmitVertex

OpEndPrimitive

; 2) Edge 2-4
; Load A2 = gl_in[2], B2 = gl_in[4]
%A2_ptr = OpAccessChain %ptr_in_gl_in_v4 %gl_in %int_2 %int_0
%B2_ptr = OpAccessChain %ptr_in_gl_in_v4 %gl_in %int_4 %int_0
%A2 = OpLoad %v4float %A2_ptr
%B2 = OpLoad %v4float %B2_ptr
%A2p = OpFAdd %v4float %A2 %offset_vec
%B2p = OpFAdd %v4float %B2 %offset_vec

; Vertex 0 (A2)
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr %A2
OpStore %outColor %vec3_1_1_1
OpEmitVertex

; Vertex 1 (A2')
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr %A2p
OpStore %outColor %vec3_1_1_1
OpEmitVertex

; Vertex 2 (B2)
%B2_out = %B2
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr %B2
OpStore %outColor %vec3_1_1_1
OpEmitVertex

; Vertex 3 (B2')
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr %B2p
OpStore %outColor %vec3_1_1_1
OpEmitVertex

OpEndPrimitive

; 3) Edge 4-0
; Load A3 = gl_in[4], B3 = gl_in[0]
%A3_ptr = OpAccessChain %ptr_in_gl_in_v4 %gl_in %int_4 %int_0
%B3_ptr = OpAccessChain %ptr_in_gl_in_v4 %gl_in %int_0 %int_0
%A3 = OpLoad %v4float %A3_ptr
%B3 = OpLoad %v4float %B3_ptr
%A3p = OpFAdd %v4float %A3 %offset_vec
%B3p = OpFAdd %v4float %B3 %offset_vec

; Vertex 0 (A3)
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr %A3
OpStore %outColor %vec3_1_1_1
OpEmitVertex

; Vertex 1 (A3')
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr %A3p
OpStore %outColor %vec3_1_1_1
OpEmitVertex

; Vertex 2 (B3)
%B3_out = %B3
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr %B3
OpStore %outColor %vec3_1_1_1
OpEmitVertex

; Vertex 3 (B3')
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr %B3p
OpStore %outColor %vec3_1_1_1
OpEmitVertex

OpEndPrimitive

; Emit original triangle (0, 2, 4) with dimmed vertex colors (0.3x)
; Load positions
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos0 = OpLoad %v4float %p0_ptr
%pos2 = OpLoad %v4float %p2_ptr
%pos4 = OpLoad %v4float %p4_ptr

; Load colors
%col0_ptr = OpAccessChain %ptr_color_v3 %color %int_0
%col2_ptr = OpAccessChain %ptr_color_v3 %color %int_2
%col4_ptr = OpAccessChain %ptr_color_v3 %color %int_4
%col0 = OpLoad %v3float %col0_ptr
%col2 = OpLoad %v3float %col2_ptr
%col4 = OpLoad %v3float %col4_ptr

; Dim colors
%col0_dim = OpVectorTimesScalar %v3float %col0 %float_0_3
%col2_dim = OpVectorTimesScalar %v3float %col2 %float_0_3
%col4_dim = OpVectorTimesScalar %v3float %col4 %float_0_3

; Vertex 0 (pos0, col0_dim)
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr %pos0
OpStore %outColor %col0_dim
OpEmitVertex

; Vertex 1 (pos2, col2_dim)
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr %pos2
OpStore %outColor %col2_dim
OpEmitVertex

; Vertex 2 (pos4, col4_dim)
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr %pos4
OpStore %outColor %col4_dim
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd

; Note: The above is a compact, simplified SPIR-V assembly sketch for a geometry shader.
; It emits a filled triangle (the original 3 vertices) in a single strip, and renders 3
; thick-edge quads (each as 4 vertices in a separate triangle strip) to visualize silhouettes.
; The thickness is approximated by a constant offset in clip-space (0.015 in Y). This matches
; the requested OutputVertices = 15 (3 + 3×4). The edge color is white; the triangle fill uses a
; dimmed vertex color (0.3x). The UBO is declared and accessible but not actively read in this version.
