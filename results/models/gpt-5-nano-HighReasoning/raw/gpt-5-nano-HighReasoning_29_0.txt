using System;

    // Fast integer input
    public sealed class FastScanner
    {
        private readonly System.IO.Stream _inStream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _bufLen = 0;
        private int _ptr = 0;

        public FastScanner()
        {
            _inStream = Console.OpenStandardInput();
        }

        private int ReadByte()
        {
            if (_ptr >= _bufLen)
            {
                _bufLen = _inStream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_bufLen <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
            } while (c <= 32); // skip spaces

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return val * sign;
        }
    }

    // BitSet with ulong words
    public class BitSet
    {
        public readonly ulong[] Bits;
        public int Words => Bits.Length;
        public BitSet(int words)
        {
            Bits = new ulong[words];
        }
        public BitSet(ulong[] bits)
        {
            Bits = (ulong[])bits.Clone();
        }

        public BitSet Clone()
        {
            return new BitSet(Bits);
        }

        public void Set(int idx)
        {
            Bits[idx >> 6] |= 1UL << (idx & 63);
        }

        public void Clear(int idx)
        {
            Bits[idx >> 6] &= ~(1UL << (idx & 63));
        }

        public bool IsSet(int idx)
        {
            return (Bits[idx >> 6] & (1UL << (idx & 63))) != 0;
        }

        public void AndWith(BitSet other)
        {
            for (int i = 0; i < Bits.Length; i++) Bits[i] &= other.Bits[i];
        }

        public void AndNotWith(BitSet other)
        {
            for (int i = 0; i < Bits.Length; i++) Bits[i] &= ~other.Bits[i];
        }

        public static BitSet And(BitSet a, BitSet b)
        {
            BitSet r = new BitSet(a.Bits.Length);
            for (int i = 0; i < a.Bits.Length; i++) r.Bits[i] = a.Bits[i] & b.Bits[i];
            return r;
        }

        public static BitSet AndNot(BitSet a, BitSet b)
        {
            BitSet r = new BitSet(a.Bits.Length);
            for (int i = 0; i < a.Bits.Length; i++) r.Bits[i] = a.Bits[i] & ~b.Bits[i];
            return r;
        }

        // Iterate over all set bits; calls action for each index
        public void ForEachSetBit(Action<int> action)
        {
            for (int w = 0; w < Bits.Length; w++)
            {
                ulong val = Bits[w];
                while (val != 0)
                {
                    int t = BitOperations.TrailingZeroCount(val);
                    int idx = (w << 6) + t;
                    action(idx);
                    val &= val - 1; // remove lowest set bit
                }
            }
        }

        // Return first set bit index or -1 if none
        public int FirstSetBit()
        {
            for (int w = 0; w < Bits.Length; w++)
            {
                ulong val = Bits[w];
                if (val != 0)
                {
                    int t = BitOperations.TrailingZeroCount(val);
                    int idx = (w << 6) + t;
                    return idx;
                }
            }
            return -1;
        }

        public int Count()
        {
            int c = 0;
            for (int i = 0; i < Bits.Length; i++) c += BitOperations.PopCount(Bits[i]);
            return c;
        }

        public override string ToString()
        {
            return $"BitSet(Words={Words})";
        }
    }

    // Exact maximum clique solver for moderate N using Tomita Bron–Kerbosch with bitsets
    public class ExactMaxClique
    {
        private readonly int _n;
        private readonly ulong[][] _adj; // adjacency as bitsets per vertex
        private readonly int _words;

        private List<int> _bestClique = new List<int>();
        private int _bestSize = 0;

        public ExactMaxClique(int n, ulong[][] adj)
        {
            _n = n;
            _adj = adj;
            _words = (n + 63) >> 6;
        }

        public List<int> Find()
        {
            // Seed with a quick greedy to improve pruning
            SeedGreedy();

            // Bron-Kerbosch with pivot and pruning
            var P = new BitSet(_words);
            for (int i = 0; i < _n; i++) P.Set(i);
            var X = new BitSet(_words);
            var R = new List<int>();

            BronKerbosch(P, X, R);

            return _bestClique;
        }

        private void SeedGreedy()
        {
            // Simple deterministic greedy expanding from each node (limited to speed)
            // Keep the best found as initial bound
            List<int> best = new List<int>();
            // Simple pass: try a few seeds (up to 100 random-ish)
            int limit = Math.Min(_n, 200);
            for (int s = 0; s < limit; s++)
            {
                int start = s;
                List<int> clique = new List<int> { start };
                BitSet cand = new BitSet(_words);
                // candidates are neighbors of start
                cand = new BitSet(_adj[start]);
                while (true)
                {
                    int v = cand.FirstSetBit();
                    if (v == -1) break;
                    clique.Add(v);
                    // cand = cand ∩ N(v)
                    cand = BitSet.And(cand, new BitSet(_adj[v]));
                }
                if (clique.Count > best.Count)
                {
                    best = new List<int>(clique);
                }
            }

            if (best.Count > _bestSize)
            {
                _bestClique = best;
                _bestSize = best.Count;
            }
        }

        private void BronKerbosch(BitSet P, BitSet X, List<int> R)
        {
            int pCount = P.Count();
            int rSize = R.Count;
            if (pCount == 0 && X.Count() == 0)
            {
                // Found a maximal clique
                if (rSize > _bestSize)
                {
                    _bestSize = rSize;
                    _bestClique = new List<int>(R);
                }
                return;
            }

            // Prune if cannot beat current best
            if (rSize + pCount <= _bestSize) return;

            // Choose pivot u from P ∪ X to reduce branching
            BitSet unionPX = P.Clone();
            unionPX.AndWith(X);
            int u = -1;
            int bestInter = -1;

            // Evaluate each candidate in unionPX for maximum intersection with P
            unionPX.ForEachSetBit(v =>
            {
                BitSet inter = BitSet.And(P, new BitSet(_adj[v]));
                int c = inter.Count();
                if (c > bestInter)
                {
                    bestInter = c;
                    u = v;
                }
            });

            if (u == -1)
            {
                // Fallback: pick any from P
                u = P.FirstSetBit();
                if (u == -1) return;
            }

            // Candidates to try: P \ N(u)
            BitSet Nu = new BitSet(_adj[u]);
            BitSet candidates = BitSet.AndNot(P, Nu);

            while (true)
            {
                int v = candidates.FirstSetBit();
                if (v == -1) break;

                // R' = R ∪ {v}
                R.Add(v);
                // P' = P ∩ N(v)
                BitSet P2 = BitSet.And(P, new BitSet(_adj[v]));
                // X' = X ∩ N(v)
                BitSet X2 = BitSet.And(X, new BitSet(_adj[v]));
                BronKerbosch(P2, X2, R);
                // backtrack
                R.RemoveAt(R.Count - 1);

                // P := P \ {v}; X := X ∪ {v}
                P.Clear(v);
                X.Set(v);

                // Update candidate set for next iterations
                Nu = new BitSet(_adj[u]);
                candidates = BitSet.AndNot(P, Nu);
            }
        }
    }

    // Heuristic maximum clique for very large graphs
    public class HeuristicMaxClique
    {
        private readonly int _n;
        private readonly List<int>[] _adj;
        private readonly int _degMax;
        private readonly int _words; // not used here, but kept for consistency
        public HeuristicMaxClique(int n, List<int>[] adj)
        {
            _n = n;
            _adj = adj;
            _degMax = 0;
            foreach (var a in adj)
                if (a.Count > _degMax) _degMax = a.Count;
        }

        public List<int> FindBest(int iterations = 200)
        {
            var best = new List<int>();
            // Precompute vertex order by degree (descending)
            var order = new int[_n];
            for (int i = 0; i < _n; i++) order[i] = i;
            Array.Sort(order, (a, b) => _adj[b].Count.CompareTo(_adj[a].Count));

            var rand = new Random(123456);
            // Seed with a deterministic first attempt using degree order
            var firstClique = GreedyFromSeed(order);
            if (firstClique.Count > best.Count) best = firstClique;

            // Additional randomized seeds
            int seeds = Math.Min(iterations, _n);
            for (int s = 0; s < seeds; s++)
            {
                // Create a shuffled subset order around degree order
                int start = order[s % _n];
                var seedOrder = new List<int> { start };
                // Build candidate set as neighbors of start
                var cand = new List<int>(_adj[start]);
                // Randomly shuffle around
                ShuffleList(cand, rand);
                foreach (var v in cand)
                {
                    if (!seedOrder.Contains(v))
                        seedOrder.Add(v);
                }
                // Build a clique greedily along this seed order
                var clique = new List<int> { seedOrder[0] };
                for (int i = 1; i < seedOrder.Count; i++)
                {
                    int v = seedOrder[i];
                    bool ok = true;
                    foreach (var u in clique)
                    {
                        if (!IsNeighbor(u, v)) { ok = false; break; }
                    }
                    if (ok) clique.Add(v);
                }
                if (clique.Count > best.Count) best = new List<int>(clique);
            }

            // If nothing found, guarantee at least one vertex
            if (best.Count == 0 && _n > 0) best.Add(0);
            return best;
        }

        private List<int> GreedyFromSeed(int[] order)
        {
            List<int> best = new List<int>();
            int N = order.Length;
            for (int idx = 0; idx < Math.Min(N, 1000); idx++)
            {
                int s = order[idx];
                var clique = new List<int> { s };
                // Candidate set: neighbors of s
                var cand = new List<int>(_adj[s]);
                ShuffleList(cand, new Random(123456 + s));

                foreach (var v in cand)
                {
                    bool ok = true;
                    foreach (var u in clique)
                    {
                        if (!IsNeighbor(u, v))
                        {
                            ok = false; break;
                        }
                    }
                    if (ok) clique.Add(v);
                }
                if (clique.Count > best.Count) best = new List<int>(clique);
            }
            return best;
        }

        private void ShuffleList<T>(List<T> list, Random rnd)
        {
            for (int i = list.Count - 1; i > 0; i--)
            {
                int j = rnd.Next(i + 1);
                T tmp = list[i]; list[i] = list[j]; list[j] = tmp;
            }
        }

        private bool IsNeighbor(int a, int b)
        {
            // Binary search in a's adjacency list for b
            var arr = _adj[a];
            int l = 0, r = arr.Count - 1;
            while (l <= r)
            {
                int m = (l + r) >> 1;
                int val = arr[m];
                if (val == b) return true;
                if (val < b) l = m + 1;
                else r = m - 1;
            }
            return false;
        }
        // Convenience
        private bool IsNeighborListContains(List<int> list, int v)
        {
            foreach (var x in list) if (x == v) return true;
            return false;
        }
    }
