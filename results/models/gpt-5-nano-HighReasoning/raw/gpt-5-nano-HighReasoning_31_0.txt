use std::io::{self, Read};

fn mitm(nums: &[i128], target: i128) -> Option<Vec<usize>> {
    let n = nums.len();
    if n == 0 {
        return if target == 0 { Some(Vec::new()) } else { None };
    }

    let n1 = n / 2;
    let n2 = n - n1;

    // Left half sums
    let left_size = 1usize << n1;
    let mut left: Vec<(i128, u32)> = Vec::with_capacity(left_size);
    for mask in 0..left_size {
        let mut s: i128 = 0;
        for i in 0..n1 {
            if ((mask >> i) & 1) != 0 {
                s += nums[i];
            }
        }
        left.push((s, mask as u32));
    }
    left.sort_by(|a, b| a.0.cmp(&b.0));

    // Right half sums
    let right_size = 1usize << n2;
    for mask2 in 0..right_size {
        let mut s2: i128 = 0;
        for j in 0..n2 {
            if ((mask2 >> j) & 1) != 0 {
                s2 += nums[n1 + j];
            }
        }
        let need = target - s2;

        // binary search in left for need
        let mut lo = 0usize;
        let mut hi = left.len();
        while lo < hi {
            let mid = (lo + hi) / 2;
            if left[mid].0 < need {
                lo = mid + 1;
            } else {
                hi = mid;
            }
        }
        if lo < left.len() && left[lo].0 == need {
            let left_mask = left[lo].1;
            let mut res: Vec<usize> = Vec::with_capacity(n);
            for i in 0..n1 {
                if ((left_mask >> i) & 1) != 0 {
                    res.push(i);
                }
            }
            for j in 0..n2 {
                if ((mask2 >> j) & 1) != 0 {
                    res.push(n1 + j);
                }
            }
            return Some(res);
        }
    }
    None
}

fn dp_nonneg(nums: &[i128], target: i128) -> Option<Vec<usize>> {
    if target < 0 {
        return None;
    }
    let t = target as usize;
    // Ensure all numbers are non-negative
    if nums.iter().any(|&x| x < 0) {
        return None;
    }
    if t == 0 {
        return Some(Vec::new());
    }

    let n = nums.len();
    let mut reachable = vec![false; t + 1];
    let mut prev = vec![-1isize; t + 1]; // which index used to reach sum
    reachable[0] = true;

    for i in 0..n {
        let ai = nums[i];
        let ai_u = ai as usize;
        if ai_u == 0 {
            // skip zeros to avoid infinite loops in reconstruction
            continue;
        }
        // update in decreasing order to avoid reusing same element
        for s in (ai_u..=t).rev() {
            if reachable[s - ai_u] && !reachable[s] {
                reachable[s] = true;
                prev[s] = i as isize;
            }
        }
    }

    if reachable[t] {
        let mut sum = t;
        let mut indices: Vec<usize> = Vec::new();
        while sum > 0 {
            let i = prev[sum] as usize;
            indices.push(i);
            sum -= nums[i] as usize;
        }
        indices.sort_unstable();
        return Some(indices);
    }
    None
}

fn dp_range(nums: &[i128], target: i128) -> Option<Vec<usize>> {
    // Handle general case with negative numbers by range DP
    // Compute min and max possible sums
    let mut min_sum: i128 = 0;
    let mut max_sum: i128 = 0;
    for &v in nums {
        if v < 0 {
            min_sum += v;
        } else {
            max_sum += v;
        }
    }

    if target < min_sum || target > max_sum {
        return None;
    }

    let offset: i128 = -min_sum;
    let m: usize = (max_sum - min_sum + 1) as usize;

    // State arrays
    let mut reach = vec![0u8; m];
    let mut from = vec![-1i32; m];
    let mut prev_sum = vec![-1i32; m];

    // Start with sum 0
    let start = (0i128 + offset) as usize;
    reach[start] = 1;

    for i in 0..nums.len() {
        let val = nums[i];
        // clone current reach to avoid reusing same item
        let old = reach.clone();
        for s in 0..m {
            if old[s] != 0 {
                let cur_sum = (s as i128) - offset;
                let new_sum = cur_sum + val;
                if new_sum < min_sum || new_sum > max_sum {
                    continue;
                }
                let idx2 = (new_sum + offset) as usize;
                if reach[idx2] == 0 {
                    reach[idx2] = 1;
                    from[idx2] = i as i32;
                    prev_sum[idx2] = s as i32;
                }
            }
        }
    }

    let target_idx = (target + offset) as usize;
    if target_idx >= m || reach[target_idx] == 0 {
        return None;
    }

    // Reconstruct path
    let mut res = Vec::new();
    let mut cur = target_idx;
    let base = start;
    while cur != base {
        let i = from[cur] as usize;
        res.push(i);
        cur = prev_sum[cur] as usize;
    }
    res.sort_unstable();
    Some(res)
}

fn main() {
    // Read entire input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    // Parse N and target
    let n_opt = it.next();
    if n_opt.is_none() {
        // No input
        println!("NO");
        return;
    }
    let n: usize = n_opt.unwrap().parse().unwrap();
    let target_str = it.next().unwrap_or("0");
    let target: i128 = target_str.parse().unwrap();

    // Parse numbers
    let mut nums: Vec<i128> = Vec::with_capacity(n);
    for _ in 0..n {
        if let Some(tok) = it.next() {
            let v: i128 = tok.parse().unwrap();
            nums.push(v);
        } else {
            // If fewer numbers provided, pad with 0
            nums.push(0);
        }
    }

    // Strategy thresholds
    const MITM_MAX_N: usize = 46; // avoid insane memory for large N
    const RANGE_M_LIMIT: usize = 4_000_000; // limit for DP over range

    // Try to find a subset
    let result: Option<Vec<usize>> = if n <= MITM_MAX_N {
        mitm(&nums, target)
    } else {
        // If all non-negative and target non-negative, use standard subset DP
        let all_nonneg = nums.iter().all(|&x| x >= 0);
        if all_nonneg && target >= 0 {
            dp_nonneg(&nums, target)
        } else {
            // Try range DP if feasible
            let mut min_sum: i128 = 0;
            let mut max_sum: i128 = 0;
            for &v in &nums {
                if v < 0 {
                    min_sum += v;
                } else {
                    max_sum += v;
                }
            }
            if target >= min_sum && target <= max_sum {
                let m = (max_sum - min_sum + 1) as usize;
                if m <= RANGE_M_LIMIT {
                    dp_range(&nums, target)
                } else {
                    // Fallback: attempt MITM if still feasible (n is large, but we try a conservative split)
                    // If n is still not too large, try MITM; otherwise give up
                    if n <= 60 {
                        mitm(&nums, target)
                    } else {
                        None
                    }
                }
            } else {
                None
            }
        }
    };

    if let Some(mut idxs) = result {
        idxs.sort_unstable();
        println!("YES");
        print!("[");
        for (i, v) in idxs.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", v);
        }
        println!("]");
    } else {
        println!("NO");
    }
}
