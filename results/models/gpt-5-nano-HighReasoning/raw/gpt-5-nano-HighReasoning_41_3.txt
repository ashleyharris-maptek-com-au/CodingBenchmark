; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 60
; Schema: 0

OpCapability Shader
OpMemoryModel Logical GLSL450

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %float 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform = OpTypePointer Uniform %UBO
%_ptr_Input_vec3 = OpTypePointer Input %v3float
%_ptr_Output_vec4 = OpTypePointer Output %v4float

%funcVoid = OpTypeFunction %void

; Variables
%colorIn = OpVariable %_ptr_Input_vec3 Input
%outColor = OpVariable %_ptr_Output_vec4 Output
%uboVar = OpVariable %_ptr_Uniform Uniform

; Decorations
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0
OpDecorate %colorIn Location 4
OpDecorate %outColor Location 0

; Entry point declaration will reference main after its definition
OpEntryPoint Fragment %main "main" %colorIn %outColor
OpExecutionMode %main OriginUpperLeft

; Function
%main = OpFunction %void None %funcVoid
%entryLabel = OpLabel

; Read input color (vec3), construct vec4 with alpha = 1.0
%col = OpLoad %v3float %colorIn
%cx = OpCompositeExtract %float %col 0
%cy = OpCompositeExtract %float %col 1
%cz = OpCompositeExtract %float %col 2
%one = OpConstant %float 1.0
%color4 = OpCompositeConstruct %v4float %cx %cy %cz %one

OpStore %outColor %color4
OpReturn
OpFunctionEnd
