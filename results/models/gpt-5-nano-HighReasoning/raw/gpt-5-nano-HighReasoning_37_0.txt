```csharp
using System;
using System.Collections.Generic;
using System.IO;

class Program
{
    // Graph data
    static int n;
    static List<int>[] Adj;
    static bool[] Removed;

    static void Main()
    {
        var sc = new FastScanner();
        if (!sc.TryReadInt(out n)) return;
        int m;
        sc.TryReadInt(out m);

        Adj = new List<int>[n];
        for (int i = 0; i < n; i++) Adj[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            int u, v;
            if (sc.TryReadInt(out u) && sc.TryReadInt(out v))
            {
                if (u >= 0 && u < n && v >= 0 && v < n)
                    Adj[u].Add(v);
            }
        }

        Removed = new bool[n];

        // Incrementally try K = 0 .. n to find minimum size
        for (int K = 0; K <= n; K++)
        {
            Array.Clear(Removed, 0, n);
            var chosen = new List<int>();
            if (SolveK(K, chosen))
            {
                Console.WriteLine(K);
                if (K > 0)
                    Console.WriteLine(string.Join(" ", chosen));
                else
                    Console.WriteLine();
                return;
            }
        }

        // Fallback (should not be reached because removing all vertices always yields DAG)
        Console.WriteLine(n);
        Console.WriteLine(string.Join(" ", Range(0, n)));
    }

    // Solve for a given K: can we delete at most K vertices to make the graph acyclic?
    static bool SolveK(int K, List<int> chosen)
    {
        var cycle = FindCycle();
        if (cycle == null)
            return true; // already a DAG

        if (K == 0)
            return false;

        // Order cycle vertices by descending outgoing degree to prune faster
        var cand = new List<int>(cycle);
        cand.Sort((a, b) => CountOut(b).CompareTo(CountOut(a)));

        foreach (int v in cand)
        {
            if (Removed[v]) continue;
            Removed[v] = true;
            chosen.Add(v);
            if (SolveK(K - 1, chosen))
                return true;
            chosen.RemoveAt(chosen.Count - 1);
            Removed[v] = false;
        }
        return false;
    }

    // Find a directed cycle in the current graph (ignoring removed vertices).
    // Returns a list of vertex ids on the cycle (distinct). If no cycle, returns null.
    static List<int> FindCycle()
    {
        int N = n;

        // Kahn's algorithm to identify vertices that are not in any cycle
        int[] indeg = new int[N];
        for (int u = 0; u < N; u++)
        {
            if (Removed[u]) continue;
            foreach (int v in Adj[u])
            {
                if (Removed[v]) continue;
                indeg[v]++;
            }
        }

        var q = new Queue<int>();
        bool[] removedInKahn = new bool[N];
        for (int i = 0; i < N; i++)
            if (!Removed[i] && indeg[i] == 0)
                q.Enqueue(i);

        while (q.Count > 0)
        {
            int u = q.Dequeue();
            removedInKahn[u] = true;
            foreach (int v in Adj[u])
            {
                if (Removed[v]) continue;
                indeg[v]--;
                if (indeg[v] == 0) q.Enqueue(v);
            }
        }

        int start = -1;
        for (int i = 0; i < N; i++)
            if (!Removed[i] && !removedInKahn[i])
            {
                start = i;
                break;
            }

        if (start == -1)
            return null; // acyclic

        // Iterative DFS to find an actual cycle
        int[] color = new int[N]; // 0 = unvisited, 1 = in stack, 2 = done
        int[] parent = new int[N];
        for (int i = 0; i < N; i++) parent[i] = -1;

        for (int s = 0; s < N; s++)
        {
            if (Removed[s] || color[s] != 0) continue;

            var st = new Stack<(int V, int NextIndex)>();
            st.Push((s, 0));
            color[s] = 1;
            while (st.Count > 0)
            {
                var cur = st.Pop();
                int u = cur.V;
                int idx = cur.NextIndex;

                if (idx < Adj[u].Count)
                {
                    // push current node back with next index
                    st.Push((u, idx + 1));
                    int w = Adj[u][idx];
                    if (Removed[w]) continue;

                    if (color[w] == 0)
                    {
                        color[w] = 1;
                        parent[w] = u;
                        st.Push((w, 0));
                    }
                    else if (color[w] == 1)
                    {
                        // cycle detected: reconstruct cycle from u back to w
                        List<int> cyc = new List<int>();
                        cyc.Add(w);
                        int curV = u;
                        cyc.Add(curV);
                        while (curV != w)
                        {
                            curV = parent[curV];
                            cyc.Add(curV);
                        }
                        // Remove potential duplicate of w at end
                        if (cyc.Count > 1 && cyc[cyc.Count - 1] == cyc[0])
                            cyc.RemoveAt(cyc.Count - 1);

                        // Deduplicate while preserving order
                        var unique = new List<int>();
                        var seen = new HashSet<int>();
                        foreach (int t in cyc)
                            if (seen.Add(t)) unique.Add(t);

                        return unique;
                    }
                }
                else
                {
                    color[u] = 2;
                }
            }
        }

        return null;
    }

    // Count number of outgoing edges from v to non-removed vertices
    static int CountOut(int v)
    {
        int c = 0;
        foreach (int w in Adj[v]) if (!Removed[w]) c++;
        return c;
    }

    // Utility: small range generator for fallback
    static IEnumerable<int> Range(int start, int count)
    {
        for (int i = start; i < start + count; i++) yield return i;
    }

    // Fast input reader
    class FastScanner
    {
        private readonly Stream _in = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len = 0, _ptr = 0;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _in.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public bool TryReadInt(out int value)
        {
            value = 0;
            int c;
            do
            {
                c = Read();
                if (c == -1) return false;
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            value = val * sign;
            return true;
        }
    }
}
```
