; SPIR-V geometry shader for expanding points into colored quads (triangle strip)
OpCapability Geometry
OpMemoryModel Logical GLSL450

; Uniform Block (UBO) - available at set=0 binding=0
OpDecorate %UBO_type Block
OpDecorate %UBO_var DescriptorSet 0
OpDecorate %UBO_var Binding 0
; Entry point and interfaces
OpEntryPoint Geometry %main "main" %worldPos_in %normal_in %color_in %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%mat4 = OpTypeMatrix %v4float 4

; UBO (dummy layout; not used directly in this shader)
%UBO_type = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v3float
%UBO_ptr = OpTypePointer Uniform %UBO_type
%UBO_var = OpVariable %UBO_ptr Uniform
; (Objects stored for decoration)
OpDecorate %UBO_var DescriptorSet 0
OpDecorate %UBO_var Binding 0

; gl_PerVertex input block (from VS): gl_Position
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
; gl_PerVertex output block (to FS): gl_Position
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

; Per-vertex inputs from Vertex Shader (arrays of same size as grid)
%arr_worldPos = OpTypeArray %v3float %uint_1
%ptr_in_arr_worldPos = OpTypePointer Input %arr_worldPos
%worldPos_in = OpVariable %ptr_in_arr_worldPos Input
OpDecorate %worldPos_in Location 0
%arr_normal = OpTypeArray %v3float %uint_1
%ptr_in_arr_normal = OpTypePointer Input %arr_normal
%normal_in = OpVariable %ptr_in_arr_normal Input
OpDecorate %normal_in Location 1
%arr_color = OpTypeArray %v3float %uint_1
%ptr_in_arr_color = OpTypePointer Input %arr_color
%color_in = OpVariable %ptr_in_arr_color Input
OpDecorate %color_in Location 2

; Output per-vertex color to fragment shader
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output
OpDecorate %outColor Location 0

; Convenience constants and temporary pointers
%int_0 = OpConstant %int 0
%h = OpConstant %float 0.04
%z0 = OpConstant %float 0.0
%one = OpConstant %float 1.0

; Function declaration
%void_ = %void
%fn_type = OpTypeFunction %void
%main = OpFunction %void None %fn_type
%entry = OpLabel

; Load the single input vertex position: gl_in[0].gl_Position
%gl_in_0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%gl_in_0_pos = OpLoad %v4float %gl_in_0_ptr

; Extract x, y
%x = OpCompositeExtract %float %gl_in_0_pos 0
%y = OpCompositeExtract %float %gl_in_0_pos 1

; Compute quad corners: (x-h, y-h), (x+h, y-h), (x-h, y+h), (x+h, y+h)
%x_minus_h = OpFSub %float %x %h
%x_plus_h  = OpFAdd %float %x %h
%y_minus_h = OpFSub %float %y %h
%y_plus_h  = OpFAdd %float %y %h

; z = 0, w = 1
; Build four corner positions
%pos1 = OpCompositeConstruct %v4float %x_minus_h %y_minus_h %z0 %one
%pos2 = OpCompositeConstruct %v4float %x_plus_h  %y_minus_h %z0 %one
%pos3 = OpCompositeConstruct %v4float %x_minus_h %y_plus_h  %z0 %one
%pos4 = OpCompositeConstruct %v4float %x_plus_h  %y_plus_h  %z0 %one

; Read color[0] (same color for all four verts)
%color_0_ptr = OpAccessChain %ptr_color_vec3 %color_in %int_0
%color_0 = OpLoad %v3float %color_0_ptr
; Store per-vertex color and position, emitting four vertices as a triangle strip
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Vertex 1
OpStore %gl_out_pos_ptr %pos1
OpStore %outColor %color_0
OpEmitVertex

; Vertex 2
OpStore %gl_out_pos_ptr %pos2
OpStore %outColor %color_0
OpEmitVertex

; Vertex 3
OpStore %gl_out_pos_ptr %pos3
OpStore %outColor %color_0
OpEmitVertex

; Vertex 4
OpStore %gl_out_pos_ptr %pos4
OpStore %outColor %color_0
OpEmitVertex

; End primitive
OpEndPrimitive

; Finish function
OpReturn
OpFunctionEnd

; Helper declarations (for pointer types used above)
; Pointers to uniform/inputs used in the code
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_color_vec3 = OpTypePointer Input %v3float

; End of shader
