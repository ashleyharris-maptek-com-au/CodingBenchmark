; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; Uniform block (std140-like layout)
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block

%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%_ptr_Uniform_vec3 = OpTypePointer Uniform %vec3
%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Input_vec2 = OpTypePointer Input %vec2
%_ptr_Output_vec4 = OpTypePointer Output %vec4

; GLSL std450
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Inputs
%worldPos = OpVariable %_ptr_Input_vec3 Input
OpDecorate %worldPos Location 0

%normal = OpVariable %_ptr_Input_vec3 Input
OpDecorate %normal Location 1

%uv = OpVariable %_ptr_Input_vec2 Input
OpDecorate %uv Location 2

%tangent = OpVariable %_ptr_Input_vec3 Input
OpDecorate %tangent Location 3

%color = OpVariable %_ptr_Input_vec3 Input
OpDecorate %color Location 4

; Outputs
$outColor = OpVariable %_ptr_Output_vec4 Output
OpDecorate %outColor Location 0

; Uniform instance
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Constants
%uint32 = OpTypeInt 32 0
%_3 = OpConstant %uint32 3
%_4 = OpConstant %uint32 4
%_2 = OpConstant %float 2.0
%_32 = OpConstant %float 32.0
%_0 = OpConstant %float 0.0
%_1 = OpConstant %float 1.0
%_0_2 = OpConstant %float 0.2
%_0_5 = OpConstant %float 0.5
%_0_8 = OpConstant %float 0.8
%_0_2_ = OpConstant %float 0.2
%_0_2b = OpConstant %float 0.2
%_4_0 = OpConstant %float 4.0
%_2_0 = OpConstant %float 2.0
%_3_0 = OpConstant %float 3.0
%_32_0 = OpConstant %float 32.0

; Type for function
%_func_void = OpTypeFunction %void

; Main
%main = OpFunction %void None %_func_void
%entry = OpLabel

; Load world position and normal
%wp = OpLoad %vec3 %worldPos
%n = OpLoad %vec3 %normal

; N = normalize(n)
%N = OpExtInst %vec3 %GLSLstd450 Normalize %n

; Load lightPos from UBO (member 3)
%lightPtr = OpAccessChain %_ptr_Uniform_vec4 %ubo %_3
%lightPV4 = OpLoad %vec4 %lightPtr
%lx = OpCompositeExtract %float %lightPV4 0
%ly = OpCompositeExtract %float %lightPV4 1
%lz = OpCompositeExtract %float %lightPV4 2
%lightP = OpCompositeConstruct %vec3 %lx %ly %lz

; L = normalize(lightPos - worldPos)
%Ldir = OpFSub %vec3 %lightP %wp
%L = OpExtInst %vec3 %GLSLstd450 Normalize %Ldir

; diffuse component
%dNL = OpDot %float %N %L
%diffuse = OpExtInst %float %GLSLstd450 FMax %dNL %_0

; -L and R = (-L) + 2*(N·L)*N
%negL = OpFNegate %vec3 %L
%twoNL = OpFMul %float %dNL %_2
%scaledN = OpVectorTimesScalar %vec3 %N %twoNL
%R = OpFAdd %vec3 %negL %scaledN

; V = normalize(camPos - worldPos)
%camPtr = OpAccessChain %_ptr_Uniform_vec4 %ubo %_4
%camPV4 = OpLoad %vec4 %camPtr
%cx = OpCompositeExtract %float %camPV4 0
%cy = OpCompositeExtract %float %camPV4 1
%cz = OpCompositeExtract %float %camPV4 2
%camP = OpCompositeConstruct %vec3 %cx %cy %cz
%Vdir = OpFSub %vec3 %camP %wp
%V = OpExtInst %vec3 %GLSLstd450 Normalize %Vdir

; dot(R,V) and specular
%dotRV = OpDot %float %R %V
%specBase = OpExtInst %float %GLSLstd450 FMax %dotRV %_0
%specular = OpExtInst %float %GLSLstd450 Pow %specBase %_32

; intensity = 0.2 + 0.5*diffuse + 0.8*specular
%halfDiffuse = OpFMul %float %diffuse %_0_5
%scaledSpec = OpFMul %float %specular %_0_8
%sum = OpFAdd %float %halfDiffuse %scaledSpec
%intensity = OpFAdd %float %sum %_0_2

; Output grayscale color
%col = OpCompositeConstruct %vec4 %intensity %intensity %intensity %_1
OpStore %outColor %col

OpReturn
OpFunctionEnd

; Execution mode and entry point
OpExecutionMode %main OriginUpperLeft
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
