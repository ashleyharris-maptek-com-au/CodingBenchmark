; SPIR-V
; Version: 1.0
; Generator: None
; Bound: 9999
; Schema: 0

OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; Uniform block type
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

; Pointers for inputs/outputs
%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Input_vec2 = OpTypePointer Input %vec2
%_ptr_Output_vec4 = OpTypePointer Output %vec4

; References to uniforms/IO
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%inWorldPos = OpVariable %_ptr_Input_vec3 Input
%inNormal = OpVariable %_ptr_Input_vec3 Input
%outColor = OpVariable %_ptr_Output_vec4 Output

; Decorations
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Uniform block member offsets (std140)
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; GLSL std 450 import
%GLSL = OpExtInstImport "GLSL.std.450"

; Entry point interface
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %outColor
OpExecutionMode %main OriginUpperLeft

; Types for main function
%main_type = OpTypeFunction %void

; Constants
%zero = OpConstant %float 0.0
%one = OpConstant %float 1.0
%three = OpConstant %float 3.0
%half = OpConstant %float 0.5
%f0_2 = OpConstant %float 0.2
%f0_3 = OpConstant %float 0.3
%f0_8 = OpConstant %float 0.8

; Base colors
%baseColor = OpConstantComposite %vec3 %f0_2 %f0_3 %f0_8
%rimTintVec3 = OpConstantComposite %vec3 %one %f0_8 %half

; Main function
%main = OpFunction %void None %main_type
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %vec3 %inWorldPos
%normal = OpLoad %vec3 %inNormal
; Normalize normal
%N = OpExtInst %vec3 %GLSL Normalize %normal

; Light position from UBO
%lightPos4_ptr = OpAccessChain %_ptr_Uniform_vec4 %ubo 3
%lightPos4 = OpLoad %vec4 %lightPos4_ptr
; Extract xyz components of lightPos4
%lx = OpCompositeExtract %float %lightPos4 0
%ly = OpCompositeExtract %float %lightPos4 1
%lz = OpCompositeExtract %float %lightPos4 2
%lightPos3 = OpCompositeConstruct %vec3 %lx %ly %lz

; Light direction
%lightDirTmp = OpVectorSub %vec3 %lightPos3 %worldPos
%L = OpExtInst %vec3 %GLSL Normalize %lightDirTmp

; View direction
%camPos4_ptr = OpAccessChain %_ptr_Uniform_vec4 %ubo 4
%camPos4 = OpLoad %vec4 %camPos4_ptr
; Extract xyz components of camPos4
%cx = OpCompositeExtract %float %camPos4 0
%cy = OpCompositeExtract %float %camPos4 1
%cz = OpCompositeExtract %float %camPos4 2
%camPos3 = OpCompositeConstruct %vec3 %cx %cy %cz
%viewDir = OpVectorSub %vec3 %camPos3 %worldPos
%V = OpExtInst %vec3 %GLSL Normalize %viewDir

; Rim factor
%NdotV = OpExtInst %float %GLSL Dot %N %V
%NdotVmax0 = OpExtInst %float %GLSL FMax %NdotV %zero
%oneMinus = OpFSub %float %one %NdotVmax0
%rim = OpExtInst %float %GLSL Pow %oneMinus %three

; Diffuse
%dotNL = OpExtInst %float %GLSL Dot %N %L
%diffuse = OpExtInst %float %GLSL FMax %dotNL %zero
%halfTimesDiffuse = OpExtInst %float %GLSL FMul %half %diffuse
%diffuseFactor = OpExtInst %float %GLSL FAdd %f0_3 %halfTimesDiffuse

; Colors
%diffuseColorVec3 = OpVectorTimesScalar %vec3 %baseColor %diffuseFactor
%rimColorVec3 = OpVectorTimesScalar %vec3 %rimTintVec3 %rim
%finalColorVec3 = OpFAdd %vec3 %diffuseColorVec3 %rimColorVec3

; Output color components
%R = OpCompositeExtract %float %finalColorVec3 0
%G = OpCompositeExtract %float %finalColorVec3 1
%B = OpCompositeExtract %float %finalColorVec3 2
%outColorVec4 = OpCompositeConstruct %vec4 %R %G %B %one
OpStore %outColor %outColorVec4

OpReturn
OpFunctionEnd
