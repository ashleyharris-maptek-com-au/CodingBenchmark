OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; Constants
%float_0_05 = OpConstant %float 0.05
%float_1 = OpConstant %float 1.0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_1 = OpConstant %uint 1

; Input per-vertex block (gl_Position)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Output per-vertex block (gl_Position)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Input arrays from vertex shader (worldPos[] used as Centers, color[])
%vec3 = OpTypeVector %float 3
%arr_vec3 = OpTypeArray %vec3 %uint_1
%ptr_in_arr_vec3 = OpTypePointer Input %arr_vec3
%worldPos_in = OpVariable %ptr_in_arr_vec3 Input
%color_in = OpVariable %ptr_in_arr_vec3 Input
OpDecorate %worldPos_in Location 0
OpDecorate %color_in Location 2

; Output color
%ptr_out_vec3 = OpTypePointer Output %vec3
%outColor = OpVariable %ptr_out_vec3 Output
OpDecorate %outColor Location 0

; Pointers to input/output vectors
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float

; Entry point
OpEntryPoint Geometry %main "main" %worldPos_in %color_in %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Function declaration (main)
%main = OpFunction %void None %func
%func = OpFunctionParameter %void
%entry = OpLabel

; Read input vertex data (for points, i = 0)
; Get gl_Position of input vertex 0
%gl_in_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%gl_in_pos = OpLoad %v4float %gl_in_pos_ptr

; Decompose gl_Position
%gx = OpCompositeExtract %float %gl_in_pos 0
%gy = OpCompositeExtract %float %gl_in_pos 1
%gz = OpCompositeExtract %float %gl_in_pos 2
%gw = OpCompositeExtract %float %gl_in_pos 3

; Diamond half-size
%gx_plus_h = OpFAdd %float %gx %%float_0_05
%gx_minus_h = OpFSub %float %gx %%float_0_05
%gy_plus_h = OpFAdd %float %gy %%float_0_05
%gy_minus_h = OpFSub %float %gy %%float_0_05

; Build diamond vertices (Top, Right, Left, Bottom)
%top = OpCompositeConstruct %v4float %gx %gy_plus_h %gz %gw
%right = OpCompositeConstruct %v4float %gx_plus_h %gy %gz %gw
%left = OpCompositeConstruct %v4float %gx_minus_h %gy %gz %gw
%bottom = OpCompositeConstruct %v4float %gx %gy_minus_h %gz %gw

; Read per-vertex color (color[0])
%color_ptr = OpAccessChain %ptr_in_arr_vec3 %color_in %int_0
%col = OpLoad %vec3 %color_ptr

; Emit Top vertex
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr %top
OpStore %outColor %col
OpEmitVertex

; Emit Right vertex
%gl_out_pos_ptr_r = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr_r %right
OpStore %outColor %col
OpEmitVertex

; Emit Left vertex
%gl_out_pos_ptr_l = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr_l %left
OpStore %outColor %col
OpEmitVertex

; Emit Bottom vertex
%gl_out_pos_ptr_b = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr_b %bottom
OpStore %outColor %col
OpEmitVertex

; End primitive
OpEndPrimitive

; Return
OpReturn
OpFunctionEnd

; Note: The GLSL GLSL450 layout maps for inputs are assumed as described:
; - worldPos_in[0] provides the center position in clip space for the diamond.
; - color_in[0] provides the per-vertex color used for all four vertices emitted.
