use std::io::{self, Read, Write, BufRead, BufReader, BufWriter};

/// Compute squared Euclidean distance between two 3D points
#[inline]
fn dist2(a: &[f32; 3], b: &[f32; 3]) -> f32 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

/// Simple deterministic RNG (LCG) for reproducible reinitialization in empty clusters
struct SimpleRng {
    state: u64,
}
impl SimpleRng {
    fn new(seed: u64) -> Self {
        SimpleRng { state: seed }
    }
    #[inline]
    fn next_u64(&mut self) -> u64 {
        // LCG parameters (32/64-bit friendly)
        self.state = self
            .state
            .wrapping_mul(2862933555777941757)
            .wrapping_add(1);
        self.state
    }
    #[inline]
    fn choice(&mut self, bound: usize) -> usize {
        // avoid modulo bias for simplicity
        if bound == 0 {
            return 0;
        }
        (self.next_u64() as usize) % bound
    }
}

/// Read a single binary point (12 bytes: 3 x f32 little-endian)
fn read_point<R: Read>(r: &mut R) -> io::Result<[f32; 3]> {
    let mut b = [0u8; 12];
    r.read_exact(&mut b)?;
    let x = f32::from_bits(u32::from_le_bytes([b[0], b[1], b[2], b[3]]));
    let y = f32::from_bits(u32::from_le_bytes([b[4], b[5], b[6], b[7]]));
    let z = f32::from_bits(u32::from_le_bytes([b[8], b[9], b[10], b[11]]));
    Ok([x, y, z])
}

/// Read all points into memory (for small datasets)
fn read_all_points<R: Read>(r: &mut R, n: usize) -> io::Result<Vec<[f32; 3]>> {
    let mut pts = Vec::with_capacity(n);
    for _ in 0..n {
        pts.push(read_point(r)?);
    }
    Ok(pts)
}

/// In-memory standard k-means ( Lloyd's algorithm )
/// Returns per-point cluster assignments (0..k-1)
fn kmeans_in_memory(points: &[[f32; 3]], k: usize) -> Vec<usize> {
    let n = points.len();
    let mut k_eff = k;
    if k_eff == 0 { return vec![]; }
    if k_eff > n {
        k_eff = n;
    }

    // Initialize centroids: use first k_eff points
    let mut centroids: Vec<[f32; 3]> = Vec::with_capacity(k_eff);
    for i in 0..k_eff {
        centroids.push(points[i]);
    }

    let mut labels = vec![0usize; n];
    let max_iter = 25;
    let mut iter = 0;
    let mut converged = false;

    // Simple RNG for reinitialization when a cluster becomes empty
    let mut rng = SimpleRng::new(123456789);

    while iter < max_iter && !converged {
        // Assign points to nearest centroid and accumulate sums
        let mut sums = vec![[0f32; 3]; k_eff];
        let mut counts = vec![0usize; k_eff];

        for i in 0..n {
            let p = points[i];
            let mut best_c = 0usize;
            let mut best_d = f32::MAX;
            for c in 0..k_eff {
                let d = dist2(&p, &centroids[c]);
                if d < best_d {
                    best_d = d;
                    best_c = c;
                }
            }
            labels[i] = best_c;
            sums[best_c][0] += p[0];
            sums[best_c][1] += p[1];
            sums[best_c][2] += p[2];
            counts[best_c] += 1;
        }

        // Update centroids
        converged = true;
        let old_centroids = centroids.clone();
        for c in 0..k_eff {
            if counts[c] > 0 {
                centroids[c][0] = sums[c][0] / counts[c] as f32;
                centroids[c][1] = sums[c][1] / counts[c] as f32;
                centroids[c][2] = sums[c][2] / counts[c] as f32;
            } else {
                // Reinitialize empty centroid to a random point
                let idx = rng.choice(n);
                centroids[c] = points[idx];
            }
        }

        // Check convergence via centroid movement
        let mut max_move2 = 0f32;
        for c in 0..k_eff {
            let dx = centroids[c][0] - old_centroids[c][0];
            let dy = centroids[c][1] - old_centroids[c][1];
            let dz = centroids[c][2] - old_centroids[c][2];
            let mv = dx * dx + dy * dy + dz * dz;
            if mv > max_move2 {
                max_move2 = mv;
            }
        }
        if max_move2 > 1e-6 {
            converged = false;
        }

        iter += 1;
    }

    // If there are more clusters than data points, adjust labels to existing centroids
    if k > n {
        for i in 0..n {
            labels[i] = i;
        }
    }

    labels
}

/// Streaming k-means (one-pass, suitable for large datasets)
/// - Initialize centroids with the first k points
/// - For each subsequent point, assign to nearest centroid and update that centroid with a simple online rule
fn streaming_kmeans<R: Read, W: Write>(
    r: &mut R,
    w: &mut W,
    npoints: usize,
    k: usize,
) -> io::Result<()> {
    if k == 0 { return Ok(()); }
    let mut k_eff = k;
    if k_eff > npoints {
        k_eff = npoints;
    }
    // Read first k_eff points as initial centroids
    let mut centroids: Vec<[f32; 3]> = vec![[0.0; 3]; k_eff];
    let mut counts: Vec<usize> = vec![0; k_eff];

    for i in 0..k_eff {
        let p = read_point(r)?;
        centroids[i] = p;
        counts[i] = 1;
        // Output the label for the i-th point: its own cluster
        let lab = i as u32;
        w.write_all(&lab.to_le_bytes())?;
    }

    // Process remaining points
    for i in k_eff..npoints {
        let p = match read_point(r) {
            Ok(pt) => pt,
            Err(e) => {
                // If we cannot read more data, stop gracefully
                return Err(e);
            }
        };
        // Nearest centroid
        let mut best = 0usize;
        let mut bestd = f32::MAX;
        for c in 0..k_eff {
            let d = dist2(&p, &centroids[c]);
            if d < bestd {
                bestd = d;
                best = c;
            }
        }
        // Output label
        let lab = best as u32;
        w.write_all(&lab.to_le_bytes())?;

        // Online update: centroid <- centroid + eta * (p - centroid)
        counts[best] += 1;
        let ncount = counts[best] as f32;
        let eta = 1.0f32 / ncount;
        centroids[best][0] = centroids[best][0] + eta * (p[0] - centroids[best][0]);
        centroids[best][1] = centroids[best][1] + eta * (p[1] - centroids[best][1]);
        centroids[best][2] = centroids[best][2] + eta * (p[2] - centroids[best][2]);
    }

    Ok(())
}

/// Entry point: adaptive strategy based on dataset size
fn main() {
    // Read header from stdin: "num_points num_clusters" (text)
    // Then the binary data: 12 bytes per point (float32 x 3), little-endian
    let stdin = io::stdin();
    let mut br = BufReader::new(stdin.lock());

    // Read first line (header)
    let mut header_line = String::new();
    if br.read_line(&mut header_line).is_err() {
        eprintln!("Failed to read header line.");
        return;
    }
    let header_line = header_line.trim();
    if header_line.is_empty() {
        eprintln!("Empty header line.");
        return;
    }
    let mut it = header_line.split_whitespace();
    let num_points: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            eprintln!("Invalid num_points in header.");
            return;
        }
    };
    let num_clusters: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            eprintln!("Invalid num_clusters in header.");
            return;
        }
    };

    // Output handle (binary)
    let stdout = io::stdout();
    let mut bw = BufWriter::new(stdout.lock());

    // Adaptive strategy
    // Thresholds chosen empirically to balance memory and time.
    const SMALL_DATASET: usize = 10_000_000; // <= 10M points -> in-memory k-means
    if num_points <= SMALL_DATASET {
        // In-memory k-means
        // We need to read all points into memory
        let mut points: Vec<[f32; 3]> = match read_all_points(&mut br, num_points) {
            Ok(v) => v,
            Err(e) => {
                eprintln!("Error reading points into memory: {}", e);
                return;
            }
        };

        let k = if num_clusters == 0 { 1 } else { num_clusters };
        let labels = kmeans_in_memory(&points, k);

        // Write labels to stdout
        for lab in labels {
            let bytes = (lab as u32).to_le_bytes();
            if bw.write_all(&bytes).is_err() {
                eprintln!("I/O error while writing output.");
                return;
            }
        }
        // Ensure all buffered output is flushed
        let _ = bw.flush();
    } else {
        // Large-scale streaming (one-pass)
        // We'll implement streaming_kmeans: initialize with first k points, then update
        // Note: We need to handle case where k > npoints gracefully
        // We run one pass; this is an approximation suitable for very large datasets.

        if num_clusters == 0 {
            // Nothing to do
            return;
        }

        // We pass br (which implements Read) and bw (Write)
        if let Err(e) = streaming_kmeans(&mut br, &mut bw, num_points, num_clusters) {
            eprintln!("Streaming clustering error: {}", e);
            return;
        }
        let _ = bw.flush();
    }
}
