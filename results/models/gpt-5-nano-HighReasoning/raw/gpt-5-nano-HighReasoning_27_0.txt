// Simple popcount for 64-bit values (no compiler-specific intrinsics)
static inline int popcount64(unsigned long long x) {
    int c = 0;
    while (x) {
        c += int(x & 1ULL);
        x >>= 1;
    }
    return c;
}

// Exact DSATUR-based coloring (backtracking) with time limit.
// Attempts to color with at most k colors. Returns true if a coloring is found.
bool dsatur_exact(int n, const std::vector<std::vector<int>>& adj, int k,
                  double time_limit_seconds, std::vector<int>& color_out) {
    if (n == 0) {
        color_out.clear();
        return true;
    }
    if (k > 64) return false; // we support up to 64 colors with bitmasks

    color_out.assign(n, -1);
    std::vector<unsigned long long> neigh_mask(n, 0ULL);
    std::vector<int> deg(n);
    for (int i = 0; i < n; ++i) deg[i] = (int)adj[i].size();

    auto tstart = std::chrono::steady_clock::now();
    auto time_ok = [&](double max_sec) {
        auto tnow = std::chrono::steady_clock::now();
        std::chrono::duration<double> dt = tnow - tstart;
        return dt.count() <= max_sec;
    };

    std::function<bool()> dfs = [&]() -> bool {
        if (!time_ok(time_limit_seconds)) return false;

        // Select uncolored vertex with max saturation (tie by degree)
        int v = -1;
        int best_sat = -1;
        int best_deg = -1;
        for (int i = 0; i < n; ++i) {
            if (color_out[i] != -1) continue;
            int sat = popcount64(neigh_mask[i]);
            if (sat > best_sat || (sat == best_sat && deg[i] > best_deg)) {
                best_sat = sat;
                best_deg = deg[i];
                v = i;
            }
        }
        if (v == -1) {
            // All colored
            return true;
        }

        unsigned long long forbidden = neigh_mask[v];
        for (int c = 0; c < k; ++c) {
            if ( ((forbidden >> c) & 1ULL) != 0ULL ) continue; // color used by neighbor
            color_out[v] = c;
            // Update neighbor saturation
            std::vector<int> changed;
            for (int nb : adj[v]) {
                if (color_out[nb] == -1) {
                    unsigned long long m = neigh_mask[nb];
                    if (((m >> c) & 1ULL) == 0ULL) {
                        neigh_mask[nb] = m | (1ULL << c);
                        changed.push_back(nb);
                    }
                }
            }
            if (dfs()) return true;
            // revert
            for (int u : changed) {
                neigh_mask[u] &= ~(1ULL << c);
            }
            color_out[v] = -1;
        }
        return false;
    };

    bool success = dfs();
    return success;
}

// Greedy coloring with a fixed vertex order.
// Returns true if coloring succeeded using colors in [0, k-1].
bool color_by_order(const std::vector<std::vector<int>>& adj, int k,
                    const std::vector<int>& order, std::vector<int>& color) {
    int n = (int)adj.size();
    color.assign(n, -1);
    int maxcol = -1;
    for (int v : order) {
        std::vector<char> used(k, 0);
        for (int nb : adj[v]) {
            int c = color[nb];
            if (c != -1 && c < k) used[c] = 1;
        }
        int c;
        for (c = 0; c < k; ++c) {
            if (!used[c]) break;
        }
        if (c == k) return false; // need more than k colors
        color[v] = c;
        if (c > maxcol) maxcol = c;
    }
    return true;
}

int main() {
    using namespace std;

    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    long long m;
    int k;
    if (!(cin >> n >> m >> k)) {
        return 0;
    }

    vector<vector<int>> adj(n);
    for (long long i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) continue;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    // Optional: remove duplicate edges to speed up (not strictly necessary)
    for (int i = 0; i < n; ++i) {
        sort(adj[i].begin(), adj[i].end());
        adj[i].erase(unique(adj[i].begin(), adj[i].end()), adj[i].end());
    }

    vector<int> color;
    bool solved = false;

    // Adaptive strategy selection

    // 1) If small enough, try exact DSATUR search first
    if (n <= 60 && k <= 64) {
        vector<int> cd;
        // Allow some time for the search
        if (dsatur_exact(n, adj, k, 2.0, cd)) {
            color = cd;
            solved = true;
        }
    }

    // 2) Greedy coloring attempts with multiple orders
    if (!solved) {
        // Degree ordering (large degrees first)
        vector<int> deg(n);
        for (int i = 0; i < n; ++i) deg[i] = (int)adj[i].size();
        vector<int> ord1(n);
        iota(ord1.begin(), ord1.end(), 0);
        sort(ord1.begin(), ord1.end(), [&](int a, int b){
            if (deg[a] != deg[b]) return deg[a] > deg[b];
            return a < b;
        });

        vector<int> ctemp;
        if (color_by_order(adj, k, ord1, ctemp)) {
            int maxc = *max_element(ctemp.begin(), ctemp.end());
            if (maxc < k) {
                color = ctemp;
                solved = true;
            }
        }

        // Randomized trials
        if (!solved) {
            std::mt19937_64 rng((unsigned)chrono::high_resolution_clock::now().time_since_epoch().count());
            vector<int> base(n);
            iota(base.begin(), base.end(), 0);
            int m_edges = (int)m;
            int max_iters = (n <= 20000 && m_edges <= 500000) ? 200 : 60;
            for (int it = 1; it <= max_iters && !solved; ++it) {
                vector<int> ord = base;
                if (it > 1) {
                    shuffle(ord.begin(), ord.end(), rng);
                }
                if (color_by_order(adj, k, ord, ctemp)) {
                    int maxc = *max_element(ctemp.begin(), ctemp.end());
                    if (maxc < k) {
                        color = ctemp;
                        solved = true;
                        break;
                    }
                }
            }
        }

        // Degeneracy ordering (reverse of removal order)
        if (!solved) {
            // Min-degree removal using a min-heap
            vector<int> cur_deg(n);
            vector<char> alive(n, 1);
            for (int i = 0; i < n; ++i) cur_deg[i] = (int)adj[i].size();
            using P = pair<int,int>;
            priority_queue<P, vector<P>, greater<P>> heap;
            for (int i = 0; i < n; ++i) heap.emplace(cur_deg[i], i);

            vector<int> order_deg_rem;
            order_deg_rem.reserve(n);
            while (!heap.empty()) {
                auto [d, v] = heap.top(); heap.pop();
                if (!alive[v]) continue;
                if (d != cur_deg[v]) { heap.emplace(cur_deg[v], v); continue; }
                alive[v] = 0;
                order_deg_rem.push_back(v);
                for (int nb : adj[v]) if (alive[nb]) {
                    cur_deg[nb]--;
                    heap.emplace(cur_deg[nb], nb);
                }
            }
            reverse(order_deg_rem.begin(), order_deg_rem.end()); // color in this order
            vector<int> cd2;
            if (color_by_order(adj, k, order_deg_rem, cd2)) {
                int maxc = *max_element(cd2.begin(), cd2.end());
                if (maxc < k) {
                    color = cd2;
                    solved = true;
                }
            }
        }
    }

    // 3) Final randomized/backtracking fallback (lightweight)
    if (!solved) {
        std::mt19937 rng((unsigned)chrono::high_resolution_clock::now().time_since_epoch().count());
        vector<int> base(n);
        iota(base.begin(), base.end(), 0);
        int max_iters = (n <= 20000 && m <= 1000000) ? 300 : 100;
        for (int it = 0; it < max_iters && !solved; ++it) {
            vector<int> ord = base;
            if (it > 0) shuffle(ord.begin(), ord.end(), rng);
            vector<int> ctemp;
            if (color_by_order(adj, k, ord, ctemp)) {
                int maxc = *max_element(ctemp.begin(), ctemp.end());
                if (maxc < k) {
                    color = ctemp;
                    solved = true;
                    break;
                }
            }
        }
    }

    // If some solution found, print
    if (solved) {
        // Ensure all colors are in [0, k-1]
        for (int i = 0; i < n; ++i) {
            int c = color[i];
            if (c < 0 || c >= k) {
                if (c < 0) c = 0;
                else c = c % k;
                color[i] = c;
            }
        }
        for (int i = 0; i < n; ++i) {
            if (i) cout << ' ';
            cout << color[i];
        }
        cout << '\n';
        return 0;
    }

    // As a last resort, attempt a small backtracking pass with time limit.
    // This is a shallow DFS designed to color with at most k colors for moderately sized graphs.
    // It may be slow on very large graphs, but provides a final chance for correctness.
    {
        vector<int> order_bt(n);
        iota(order_bt.begin(), order_bt.end(), 0);
        // Order by degree descending (heuristic)
        sort(order_bt.begin(), order_bt.end(), [&](int a, int b){
            if (adj[a].size() != adj[b].size()) return adj[a].size() > adj[b].size();
            return a < b;
        });

        vector<int> color_bt(n, -1);
        vector<char> seen(n, 0);
        bool found = false;

        auto t_start =chrono::steady_clock::now();
        function<void(int)> dfs = [&](int idx) {
            if (found) return;
            auto now = chrono::steady_clock::now();
            if (chrono::duration_cast<chrono::seconds>(now - t_start).count() > 4) return; // small cap
            if (idx == n) {
                // succeed
                color = color_bt;
                found = true;
                return;
            }
            int v = order_bt[idx];
            unsigned int forbidden = 0;
            for (int nb : adj[v]) {
                int c = color_bt[nb];
                if (c >= 0 && c < 32) forbidden |= (1u << c);
            }
            for (int c = 0; c < k; ++c) {
                bool conflict = false;
                if (c < 32) {
                    if ((forbidden >> c) & 1u) conflict = true;
                } else {
                    // c >= 32
                    for (int nb : adj[v]) if (color_bt[nb] == c) { conflict = true; break; }
                }
                if (conflict) continue;
                color_bt[v] = c;
                dfs(idx + 1);
                color_bt[v] = -1;
                if (found) return;
            }
        };

        dfs(0);
        if (found) {
            for (int i = 0; i < n; ++i) {
                if (i) cout << ' ';
                cout << color[i];
            }
            cout << '\n';
            return 0;
        }
    }

    // If nothing worked, output a safe default (best effort). This part should rarely be reached.
    // We output a valid coloring with 0 for all vertices only if there are no edges; otherwise it's invalid.
    // Since we must output something, still print something deterministic.
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << 0;
    }
    cout << '\n';
    return 0;
}
