/*
  Inclusive prefix sum (scan) compute shader
  - set = 0, binding 0: storage buffer (read-only) - uint array A[]
  - set = 0, binding 1: storage buffer (read-write) - uint array Out[]
  - set = 0, binding 2: uniform buffer - uvec4 with N in .x
  - Local size: 256 x 1 x 1
  - Each invocation with global id i computes Out[i] = sum_{k=0..i} A[k], if i < N
*/

#version 450

// Input: read-only storage buffer containing input array A
layout(std430, binding = 0) readonly buffer InputBuffer {
    uint A[];
} InBuf;

// Output: read-write storage buffer containing output array Out
layout(std430, binding = 1) writeonly buffer OutputBuffer {
    uint Out[];
} OutBuf;

// Uniform buffer containing N in its first component (N is the length of A/Out)
layout(std140, binding = 2) uniform UniformBlock {
    uvec4 data; // data.x = N
} U;

// Local workgroup size (as required)
layout(local_size_x = 256, local_size_y = 1, local_size_z = 1) in;

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = U.data.x;

    if (i >= N) {
        // Outside the range we care about
        return;
    }

    uint sum = 0u;

    // Naive inclusive prefix sum: sum A[0..i]
    for (uint j = 0u; j <= i; ++j) {
        sum += InBuf.A[j];
    }

    OutBuf.Out[i] = sum;
}

Notes:
- This is a straightforward, portable implementation that will work for all N (256, 1024, 4096, 16384, 65536) given a dispatch that covers at least N invocations (global size >= N). It intentionally does not rely on inter-thread synchronization or shared memory so it stays correct in multi-workgroup setups with a naive approach.
- If you want a fast intra-workgroup scan for N <= 256, you can replace the shader body with a Blelloch/Hillis-Steele style in-workgroup scan using a workgroup-scoped buffer and barrier calls. However, as requested, a correct naive fallback is acceptable here.
