OpCapability Shader
OpMemoryModel Logical GLSL450
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Pointers
%ptr_Uniform = OpTypePointer Uniform %UBOType
%ptr_Vec4 = OpTypePointer Uniform %vec4
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_UBO = %ptr_Uniform

; Inputs
worldPos = OpVariable %ptr_Input_vec3 Input
normal = OpVariable %ptr_Input_vec3 Input
uv = OpVariable %ptr_Input_vec2 Input
tangent = OpVariable %ptr_Input_vec3 Input
color = OpVariable %ptr_Input_vec3 Input

; Output
outColor = OpVariable %ptr_Output_vec4 Output

; Uniform
ubo = OpVariable %ptr_UBO Uniform

OpDecorate worldPos Location 0

%idx0 = OpConstant %int32 0
%idx3 = OpConstant %int32 3

; Function types
%voidfn = OpTypeFunction %void

; Functions
%main = OpFunction %void None %voidfn
%entry = OpLabel

%worldPosVal = OpLoad %vec3 %worldPos
%normalVal = OpLoad %vec3 %normal
%N = OpExtInst %vec3 %GLSLstd450 Normalize %normalVal

%lightPosPtr = OpAccessChain %ptr_Vec4 %ubo %idx3
%lightPos4 = OpLoad %vec4 %lightPosPtr
%lx = OpCompositeExtract %float %lightPos4 0
%ly = OpCompositeExtract %float %lightPos4 1
%lz = OpCompositeExtract %float %lightPos4 2
%lightPosXYZ = OpCompositeConstruct %vec3 %lx %ly %lz

%L = OpVectorSub %vec3 %lightPosXYZ %worldPosVal
%Lnorm = OpExtInst %vec3 %GLSLstd450 Normalize %L

%dotNL = OpExtInst %float %GLSLstd450 Dot %N %Lnorm
%zero = OpConstant %float 0.0
%one = OpConstant %float 1.0
%d = OpExtInst %float %GLSLstd450 Clamp %dotNL %zero %one

%col3 = OpCompositeConstruct %vec3 %d %d %d
%RGBA = OpCompositeConstruct %vec4 %col3 %one
OpStore %outColor %RGBA

OpReturn
OpFunctionEnd

; Entry point declaration
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor %ubo
