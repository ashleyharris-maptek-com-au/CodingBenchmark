```rust
use std::io::{self, Read, Write};

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = Vec::new();
        // Read entire stdin
        io::stdin().read_to_end(&mut input).unwrap();
        Self { buf: input, idx: 0 }
    }

    fn next_usize(&mut self) -> usize {
        // Skip non-digit characters
        while self.idx < self.buf.len() && (self.buf[self.idx] < b'0' || self.buf[self.idx] > b'9') {
            self.idx += 1;
        }
        let mut num: usize = 0;
        while self.idx < self.buf.len() {
            let b = self.buf[self.idx];
            if b < b'0' || b > b'9' {
                break;
            }
            num = num * 10 + (b - b'0') as usize;
            self.idx += 1;
        }
        num
    }
}

fn main() {
    // Fast input
    let mut sc = Scanner::new();
    let n = sc.next_usize();
    let m = sc.next_usize();

    // matched[v] == 1 means vertex v is part of the current maximal matching
    let mut matched: Vec<u8> = vec![0u8; n];

    // Process edges as a stream and build a maximal matching greedily
    for _ in 0..m {
        let u = sc.next_usize();
        let v = sc.next_usize();
        if matched[u] == 0 && matched[v] == 0 {
            matched[u] = 1;
            matched[v] = 1;
        }
    }

    // The vertex cover is exactly the set of matched vertices (endpoints of the maximal matching)
    let mut cover: Vec<usize> = Vec::new();
    cover.reserve(n / 2 + 1);
    for i in 0..n {
        if matched[i] != 0 {
            cover.push(i);
        }
    }
    let count = cover.len();

    // Output using a buffered writer
    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());

    // First line: number of vertices in the cover
    writeln!(out, "{}", count).unwrap();

    // Second line: the vertex indices space-separated (ascending order by construction)
    for (i, v) in cover.iter().enumerate() {
        if i > 0 {
            write!(out, " ").unwrap();
        }
        write!(out, "{}", v).unwrap();
    }
    writeln!(out).unwrap();
}
```
