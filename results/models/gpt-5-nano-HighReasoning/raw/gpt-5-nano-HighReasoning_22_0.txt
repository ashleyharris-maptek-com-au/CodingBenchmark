```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;
using System.Threading;

namespace TetrisAutoPlayer
{
    // A compact, self-contained Tetris AI for a streaming (interactive-like) judge.
    // - Maintains its own board state
    // - Chooses rotation/column to maximize a heuristic
    // - Uses a simple 4x4 rotation scheme generated from canonical Rot0 shapes
    // - Adapts strategy based on board complexity
    // - Keeps within a per-piece time budget (roughly 5 seconds)

    class Program
    {
        // Grid: grid[row][col], 0 <= row < height, 0 <= col < width
        static int BoardWidth;
        static int BoardHeight;
        static bool[][] Grid;

        // Piece rotations map: pieceChar -> list of 4 RotationStates
        static Dictionary<char, List<RotationState>> PieceRotations = new Dictionary<char, List<RotationState>>();

        // Time budget per piece (seconds)
        const double TimeBudgetPerPiece = 5.0;

        // Random seed for tie-breaks in case of equal scores (not strictly necessary)
        static Random Rand = new Random(12345);

        // Entry point
        static void Main(string[] args)
        {
            // Initialize a simple fast scanner that reads tokens across lines
            var scanner = new FastScanner();

            // Read initial board info
            string wStr = scanner.NextToken();
            string hStr = scanner.NextToken();
            string nPiecesStr = scanner.NextToken();

            if (string.IsNullOrEmpty(wStr) || string.IsNullOrEmpty(hStr) || string.IsNullOrEmpty(nPiecesStr))
            {
                // No input – exit gracefully
                return;
            }

            BoardWidth = int.Parse(wStr);
            BoardHeight = int.Parse(hStr);
            int NumPieces = int.Parse(nPiecesStr);

            // Initialize board grid as empty
            Grid = new bool[BoardHeight][];
            for (int r = 0; r < BoardHeight; r++)
                Grid[r] = new bool[BoardWidth];

            // Build rotation matrices for all piece types
            BuildAllRotations();

            // Start processing pieces
            for (int pieceIndex = 0; pieceIndex < NumPieces; pieceIndex++)
            {
                string pieceTypeStr = scanner.NextToken();
                if (string.IsNullOrEmpty(pieceTypeStr)) break;

                char pieceType = pieceTypeStr[0];
                if (!PieceRotations.ContainsKey(pieceType))
                {
                    // Unknown piece; skip
                    break;
                }

                // Decide best move for this piece based on current Grid
                var bestMove = SelectBestMove(pieceType);

                // Output: rotation (0-3) and column (leftmost column of piece)
                Console.WriteLine($"{bestMove.Rotation} {bestMove.Column}");
                Console.Out.Flush();

                // Now read the result from judge/runner
                string result = scanner.NextToken();
                if (result == null) break;

                // If game over, stop
                if (result.Equals("gameover", StringComparison.OrdinalIgnoreCase))
                {
                    break;
                }

                // Apply the move to internal Grid by simulating the drop again
                // (This mirrors the judge's behavior and keeps internal state in sync.)
                var afterMove = SimulateDrop(Grid, PieceRotations[pieceType][bestMove.Rotation], bestMove.Column);
                if (afterMove.GameOver)
                {
                    // If our simulation says gameover, stop
                    break;
                }

                // Update internal grid
                Grid = afterMove.NewGrid;

                // If the judge reports lines cleared, we could verify against afterMove.LinesCleared
                // but it's not required to print anything here.
            }
        }

        // Represents a rotation move
        struct Move
        {
            public int Rotation; // 0-3
            public int Column;   // 0-based leftmost column of the piece's footprint
        }

        // Rotation state: a particular rotation, with blocks and footprint size
        class RotationState
        {
            public List<(int x, int y)> Blocks; // coordinates within footprint, minX and minY are assumed to be 0
            public int Width;                    // footprint width (maxX+1)
            public int Height;                   // footprint height (maxY+1)
        }

        // Build rotation states for all pieces using a canonical Rot0 set
        // For each piece type we define a Rot0 in a 4x4 bounding box, then repeatedly rotate CW
        static void BuildAllRotations()
        {
            PieceRotations.Clear();

            // Canonical Rot0 coordinates for each piece
            var canonical = new Dictionary<char, List<(int x, int y)>>()
            {
                // I: 4x1 horizontal line
                // Rot0 coords: (0,1),(1,1),(2,1),(3,1)
                { 'I', new List<(int,int)>{ (0,1), (1,1), (2,1), (3,1) } },

                // O: 2x2 square
                // Rot0 coords: (1,1),(2,1),(1,2),(2,2)
                { 'O', new List<(int,int)>{ (1,1), (2,1), (1,2), (2,2) } },

                // T:
                // Rot0: (1,0),(0,1),(1,1),(2,1)
                { 'T', new List<(int,int)>{ (1,0), (0,1), (1,1), (2,1) } },

                // S:
                // Rot0: (1,0),(2,0),(0,1),(1,1)
                { 'S', new List<(int,int)>{ (1,0), (2,0), (0,1), (1,1) } },

                // Z:
                // Rot0: (0,0),(1,0),(1,1),(2,1)
                { 'Z', new List<(int,int)>{ (0,0), (1,0), (1,1), (2,1) } },

                // J:
                // Rot0: (0,0),(0,1),(1,1),(2,1)
                { 'J', new List<(int,int)>{ (0,0), (0,1), (1,1), (2,1) } },

                // L:
                // Rot0: (2,0),(0,1),(1,1),(2,1)
                { 'L', new List<(int,int)>{ (2,0), (0,1), (1,1), (2,1) } },
            };

            foreach (var kv in canonical)
            {
                char piece = kv.Key;
                var rot0 = kv.Value;

                // Build 4 rotations
                var rotations = new List<RotationState>();
                var current = new List<(int x, int y)>(rot0);

                for (int r = 0; r < 4; r++)
                {
                    // Normalize: translate so minX,minY are 0
                    int minX = current.Min(p => p.x);
                    int minY = current.Min(p => p.y);
                    var normalized = current.Select(p => (x: p.x - minX, y: p.y - minY)).ToList();

                    int maxX = normalized.Max(p => p.x);
                    int maxY = normalized.Max(p => p.y);

                    rotations.Add(new RotationState
                    {
                        Blocks = normalized,
                        Width = maxX + 1,
                        Height = maxY + 1
                    });

                    // Rotate CW for next rotation
                    current = RotateCW(current);
                }

                PieceRotations[piece] = rotations;
            }
        }

        // Rotate a set of blocks 90 degrees clockwise around origin
        static List<(int x, int y)> RotateCW(List<(int x, int y)> blocks)
        {
            var rotated = new List<(int x, int y)>();
            foreach (var b in blocks)
            {
                // (x,y) -> (y, -x)
                rotated.Add((b.y, -b.x));
            }
            // Normalize to non-negative coordinates
            int minX = rotated.Min(p => p.x);
            int minY = rotated.Min(p => p.y);
            var normalized = rotated.Select(p => (p.x - minX, p.y - minY)).ToList();
            return normalized;
        }

        // Fast scanner for tokens (across lines)
        class FastScanner
        {
            private readonly Queue<string> _tokens = new Queue<string>();

            public string NextToken()
            {
                while (_tokens.Count == 0)
                {
                    string line = Console.In.ReadLine();
                    if (line == null) return null;
                    foreach (var t in line.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries))
                        _tokens.Enqueue(t);
                }
                return _tokens.Dequeue();
            }
        }

        // Result of simulating a drop
        class DropResult
        {
            public bool GameOver;
            public int LinesCleared;
            public int LandingY;
            public bool[][] NewGrid;
        }

        // Simulate dropping a piece (rotation and column) onto a given grid
        static DropResult SimulateDrop(bool[][] grid, RotationState rotation, int column)
        {
            int width = BoardWidth;
            int height = BoardHeight;

            // If column invalid, treat as game over (to be safe)
            if (column < 0 || column + rotation.Width > width)
            {
                return new DropResult { GameOver = true };
            }

            // Starting y position high above board
            int y = -4;
            while (true)
            {
                // Check collision for next step
                if (Collision(grid, rotation.Blocks, column, y + 1, height, width))
                    break;
                y++;
                // If y becomes too large, just stop; but collision will handle bottom bound
                if (y > height + 4) break;
            }

            // If landing entirely above board (top of piece above board), may be game over
            if (y < 0)
            {
                // Some pieces may partially be inside when y<0; but generally this means top is blocked
                // Treat as game over to be safe
                return new DropResult { GameOver = true };
            }

            // Clone grid
            bool[][] newGrid = CloneGrid(grid);

            // Place blocks
            foreach (var b in rotation.Blocks)
            {
                int bx = column + b.x;
                int by = y + b.y;
                if (by >= 0 && by < height && bx >= 0 && bx < width)
                    newGrid[by][bx] = true;
            }

            // Clear full lines
            int linesCleared = 0;
            List<bool[]> remaining = new List<bool[]>();
            for (int r = 0; r < height; r++)
            {
                bool full = true;
                for (int c = 0; c < width; c++)
                {
                    if (!newGrid[r][c])
                    {
                        full = false;
                        break;
                    }
                }
                if (full) linesCleared++;
                else remaining.Add(newGrid[r]);
            }

            int emptyTop = height - remaining.Count;
            // Build final grid after line clears
            for (int i = 0; i < emptyTop; i++)
                newGrid[i] = new bool[width];
            for (int i = 0; i < remaining.Count; i++)
                newGrid[emptyTop + i] = remaining[i];

            return new DropResult
            {
                GameOver = false,
                LinesCleared = linesCleared,
                LandingY = y,
                NewGrid = newGrid
            };
        }

        // Collision check for a proposed placement at (column, y)
        static bool Collision(bool[][] grid, List<(int x, int y)> blocks, int column, int y, int height, int width)
        {
            foreach (var b in blocks)
            {
                int gx = column + b.x;
                int gy = y + b.y;
                if (gx < 0 || gx >= width) return true; // out of bounds horizontally
                if (gy >= height) return true; // hits floor
                if (gy >= 0)
                {
                    if (grid[gy][gx]) return true;
                }
                // gy < 0 is above visible board; ignore
            }
            return false;
        }

        // Clone a grid
        static bool[][] CloneGrid(bool[][] src)
        {
            int h = src.Length;
            var copy = new bool[h][];
            for (int r = 0; r < h; r++)
            {
                copy[r] = new bool[src[r].Length];
                Array.Copy(src[r], copy[r], src[r].Length);
            }
            return copy;
        }

        // Compute a simple heuristic score for a candidate board
        static int ScoreBoard(bool[][] grid)
        {
            int width = BoardWidth;
            int height = BoardHeight;
            // Heuristic components
            int totalHeight = 0;
            int[] colHeights = new int[width];
            bool[] hasBlockInCol = new bool[width];
            int filled = 0;

            for (int c = 0; c < width; c++)
            {
                int top = height;
                for (int r = 0; r < height; r++)
                {
                    if (grid[r][c])
                    {
                        top = r;
                        hasBlockInCol[c] = true;
                        break;
                    }
                }
                int hCol = hasBlockInCol[c] ? (height - top) : 0;
                colHeights[c] = hCol;
                totalHeight += hCol;
            }

            // Holes: empty cells with a block somewhere above in the same column
            int holes = 0;
            for (int c = 0; c < width; c++)
            {
                bool foundBlock = false;
                for (int r = height - 1; r >= 0; r--)
                {
                    if (grid[r][c])
                    {
                        foundBlock = true;
                    }
                    else if (foundBlock)
                    {
                        holes++;
                    }
                }
            }

            // Bumpiness: sum of abs diffs between adjacent column heights
            int bumpiness = 0;
            for (int c = 0; c < width - 1; c++)
                bumpiness += Math.Abs(colHeights[c] - colHeights[c + 1]);

            // Simple linear score (higher is better)
            int maxH = colHeights.Length > 0 ? colHeights.Max() : 0;

            // Density if needed
            int blocks = 0;
            for (int r = 0; r < height; r++)
                for (int c = 0; c < width; c++)
                    if (grid[r][c]) blocks++;
            double density = (double)blocks / (width * height);

            // Adaptive weights (simple two-mode heuristic)
            // We won't re-run the entire score based on density exactly; instead we'll bias based on global metrics
            int score;
            if (density < 0.4 && maxH < height * 0.6)
            {
                // "Line aggressive" mode
                score = (holes * -60) + (bumpiness * -3) + (totalHeight * -2) + (0); // base lines already accounted in grid; here a baseline
            }
            else
            {
                // "Hole minimization" mode
                score = (holes * -70) + (bumpiness * -5) + (totalHeight * -2);
            }

            // Slight incentive for more lines: if there were any complete lines cleared, this should be reflected outside
            // Here we don't know linesCleared from this board; we keep a stable baseline
            return score;
        }

        // Compute best move for the current piece type
        static Move SelectBestMove(char pieceType)
        {
            var rotations = PieceRotations[pieceType];
            int width = BoardWidth;
            // Time budget guard
            var sw = Stopwatch.StartNew();

            int bestRotation = 0;
            int bestColumn = 0;
            int bestScore = int.MinValue;
            bool bestFoundAny = false;

            // Complexity-based adaptive looping: try all rotations and all valid columns
            // Use a simple scoring function; keep best within time budget
            foreach (var (rotIdx, rot) in rotations.Select((r, i) => (i, r)))
            {
                int footprintW = rot.Width;
                for (int col = 0; col <= width - footprintW; col++)
                {
                    if (sw.Elapsed.TotalSeconds > TimeBudgetPerPiece)
                    {
                        // Time budget exhausted; return best so far
                        if (bestFoundAny)
                            return new Move { Rotation = bestRotation, Column = bestColumn };
                        else
                        {
                            // Fallback: try a safe default
                            return new Move { Rotation = 0, Column = 0 };
                        }
                    }

                    // Simulate drop on a temporary copy for evaluation
                    var res = SimulateDrop(Grid, rot, col);
                    if (res.GameOver)
                    {
                        continue;
                    }

                    // Heuristic score based on the resulting grid
                    int score = ScoreBoard(res.NewGrid);

                    // We can also bias by number of lines cleared if available:
                    // If we want to incorporate lines cleared, we would need to fetch it from res.
                    // For simplicity here, ScoreBoard accounts for board state.

                    // Keep the best
                    if (score > bestScore || !bestFoundAny)
                    {
                        bestScore = score;
                        bestRotation = rotIdx;
                        bestColumn = col;
                        bestFoundAny = true;
                    }
                }
            }

            if (!bestFoundAny)
            {
                // Fallback to something safe
                return new Move { Rotation = 0, Column = 0 };
            }

            return new Move { Rotation = bestRotation, Column = bestColumn };
        }

        // Helper: struct-like return for best move
        struct Move { public int Rotation; public int Column; }

        // Utility: compute a more informed score of a board during evaluation
        // (Not used directly in the Move struct; kept for clarity if extended)
        // (ScoreBoard currently returns a heuristic value based on holes/bumpiness/height)
    }
}
```
