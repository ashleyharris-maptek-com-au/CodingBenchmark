```csharp
using System;
using System.Collections.Generic;
using System.Text;

namespace CPP_Doubling_Euler
{
    // A compact solution for the undirected Chinese Postman problem
    // Strategy: Duplicate every edge (produce a Eulerian multigraph) and
    // compute a single Eulerian circuit starting from node 0 using Hierholzer's algorithm.
    // This guarantees a valid route that traverses every edge at least once
    // (in this approach, every original edge is traversed twice).

    class Program
    {
        // adjacency entry: neighbor and the undirected edge id
        struct Adj
        {
            public int To;
            public int Id;
            public Adj(int to, int id) { To = to; Id = id; }
        }

        // Fast integer scanner for stdin
        sealed class FastScanner
        {
            private readonly System.IO.Stream _stream;
            private readonly byte[] _buffer = new byte[1 << 20];
            private int _len = 0;
            private int _ptr = 0;

            public FastScanner()
            {
                _stream = Console.OpenStandardInput();
            }

            private int Read()
            {
                if (_ptr >= _len)
                {
                    _len = _stream.Read(_buffer, 0, _buffer.Length);
                    _ptr = 0;
                    if (_len <= 0) return -1;
                }
                return _buffer[_ptr++];
            }

            public int NextInt()
            {
                int c;
                do
                {
                    c = Read();
                } while (c <= 32 && c != -1);

                int sign = 1;
                if (c == '-')
                {
                    sign = -1;
                    c = Read();
                }

                int val = 0;
                while (c > 32)
                {
                    val = val * 10 + (c - '0');
                    c = Read();
                }
                return val * sign;
            }
        }

        static void Main(string[] args)
        {
            FastScanner fs = new FastScanner();

            int N = fs.NextInt();
            int M = fs.NextInt();

            // Read all edges; weights are not needed for the doubling Euler approach
            int[] U = new int[M];
            int[] V = new int[M];
            for (int i = 0; i < M; i++)
            {
                int u = fs.NextInt();
                int v = fs.NextInt();
                int w = fs.NextInt(); // weight ignored in this approach
                U[i] = u;
                V[i] = v;
            }

            // Build doubled multigraph: for each input edge, add two undirected copies
            // We will perform Hierholzer's algorithm on 2*M undirected edges.
            List<Adj>[] graph = new List<Adj>[N];
            for (int i = 0; i < N; i++) graph[i] = new List<Adj>();

            int edgeId = 0;
            for (int i = 0; i < M; i++)
            {
                int u = U[i], v = V[i];
                // original copy
                graph[u].Add(new Adj(v, edgeId));
                graph[v].Add(new Adj(u, edgeId));
                edgeId++;

                // duplicate copy
                graph[u].Add(new Adj(v, edgeId));
                graph[v].Add(new Adj(u, edgeId));
                edgeId++;
            }

            int totalEdges = edgeId;
            bool[] used = new bool[totalEdges];
            int[] idx = new int[N]; // next edge index to try per vertex

            // Hierholzer's algorithm to obtain Eulerian circuit starting at 0
            Stack<int> st = new Stack<int>();
            List<int> circuit = new List<int>();

            st.Push(0);

            while (st.Count > 0)
            {
                int v = st.Peek();
                List<Adj> neigh = graph[v];

                // advance to the next unused edge
                while (idx[v] < neigh.Count && used[neigh[idx[v]].Id])
                    idx[v]++;

                if (idx[v] == neigh.Count)
                {
                    // no more unused edges from v
                    st.Pop();
                    circuit.Add(v);
                }
                else
                {
                    Adj e = neigh[idx[v]++];
                    if (used[e.Id])
                        continue;

                    used[e.Id] = true;
                    st.Push(e.To);
                }
            }

            // circuit is produced in reverse; reverse to get proper order
            circuit.Reverse();

            // If there are no edges, circuit will be [0].
            // Output the route as whitespace-separated node indices.
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < circuit.Count; i++)
            {
                if (i > 0) sb.Append(' ');
                sb.Append(circuit[i]);
            }

            Console.WriteLine(sb.ToString());
        }
    }
}
```
