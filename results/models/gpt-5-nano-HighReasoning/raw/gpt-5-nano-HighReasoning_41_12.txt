; SPIR-V
; Version: 1.0
; Generator: Custom
OpCapability Shader
OpMemoryModel Logical GLSL450

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%i32 = OpTypeInt 32 1

; Constants
%f1_0 = OpConstant %float 1.0
%f0_5 = OpConstant %float 0.5
%f0_4 = OpConstant %float 0.4
%f0_6 = OpConstant %float 0.6
%f0_3 = OpConstant %float 0.3
%f0_15 = OpConstant %float 0.15
%f0_05 = OpConstant %float 0.05

; Vector constants
%skyColorVec = OpConstantComposite %v3float %f0_4 %f0_6 %f1_0
%groundColorVec = OpConstantComposite %v3float %f0_3 %f0_15 %f0_05
%whiteVec = OpConstantComposite %v3float %f1_0 %f1_0 %f1_0

; Types for uniforms
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%UBOPtr = OpTypePointer Uniform %UBO
%_ptrUniform_v4f = OpTypePointer Uniform %v4float
%_ptrInput_v3f = OpTypePointer Input %v3float
%_ptrInput_v2f = OpTypePointer Input %v2float
%_ptrOutput_v4f = OpTypePointer Output %v4float

; Input variables
%WorldPos = OpVariable %_ptrInput_v3f Input
%Normal = OpVariable %_ptrInput_v3f Input
%UV = OpVariable %_ptrInput_v2f Input
%Tangent = OpVariable %_ptrInput_v3f Input
%Color = OpVariable %_ptrInput_v3f Input
; Output variable
%outColor = OpVariable %_ptrOutput_v4f Output
; Uniform block
%ubo = OpVariable %UBOPtr Uniform

; Decorations
OpDecorate %WorldPos Location 0
OpDecorate %Normal Location 1
OpDecorate %UV Location 2
OpDecorate %Tangent Location 3
OpDecorate %Color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%i32_0 = OpConstant %i32 0
%i32_3 = OpConstant %i32 3

%void = OpTypeVoid
%funcType = OpTypeFunction %void
%main = OpFunction %void None %funcType
%mainLabel = OpLabel

; Load inputs
%worldPos = OpLoad %v3float %WorldPos
%normal = OpLoad %v3float %Normal
%uv = OpLoad %v2float %UV
%tangent = OpLoad %v3float %Tangent
%color = OpLoad %v3float %Color

; Normalize N
%dotNN = OpDot %float %normal %normal
%lenN = OpSqrt %float %dotNN
%invLenN = OpFDiv %float %f1_0 %lenN
%N = OpVectorTimesScalar %v3float %normal %invLenN

; Light position
%lightPosPtr = OpAccessChain %_ptrUniform_v4f %ubo %i32_3
%lightPos = OpLoad %v4float %lightPosPtr
%lp0 = OpCompositeExtract %float %lightPos 0
%lp1 = OpCompositeExtract %float %lightPos 1
%lp2 = OpCompositeExtract %float %lightPos 2
%lightPos_xyz = OpCompositeConstruct %v3float %lp0 %lp1 %lp2
%L = OpVectorSub %v3float %lightPos_xyz %worldPos

%d_ll = OpDot %float %L %L
%L_len = OpSqrt %float %d_ll
%L_invLen = OpFDiv %float %f1_0 %L_len
%L_norm = OpVectorTimesScalar %v3float %L %L_invLen

; t = N.y * 0.5 + 0.5
%Ny = OpCompositeExtract %float %N 1
%NyTimesHalf = OpFMul %float %Ny %f0_5
%tVal = OpFAdd %float %NyTimesHalf % 0.5
; ambient
%oneMinusT = OpFSub %float %f1_0 %tVal
%groundTerm = OpVectorTimesScalar %v3float %groundColorVec % oneMinusT
%skyTerm = OpVectorTimesScalar %v3float %skyColorVec % tVal
%ambient = OpVectorAdd %v3float %groundTerm %skyTerm

; diffuse
%d = OpDot %float %N %L_norm
%diffuseScalar = OpFMul %float %f0_3 %d
%diffuseVec = OpVectorTimesScalar %v3float %whiteVec % diffuseScalar

%finalColorVec = OpVectorAdd %v3float %ambient % diffuseVec
%alpha = %f1_0
%finalCol4 = OpCompositeConstruct %v4float %finalColorVec %alpha
OpStore %outColor %finalCol4

OpReturn
OpFunctionEnd

; Entry point and mode
OpEntryPoint Fragment %main "main" %WorldPos %Normal %UV %Tangent %Color %outColor
OpExecutionMode %main OriginUpperLeft
