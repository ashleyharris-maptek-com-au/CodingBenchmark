```rust
// A scalable, heuristic graph layout solver based on a Barnes-Hut accelerated
// force-directed approach (Fruchterman-Reingold style).
//
// - Handles arbitrary graph sizes by adapting iterations and using a quadtree
//   (Barnes-Hut) to approximate repulsive forces.
// - Ignores self-loops and parallel edges (duplicates) gracefully.
// - Produces 2D coordinates for each node (undirected graph).
//
// Usage:
//   Read N M and M edges from stdin (0-indexed). Output N lines: "x y".
//
// This is a self-contained Rust program and requires no external crates.

use std::cmp::{max, min};
use std::collections::HashSet;
use std::f64::INFINITY;
use std::io::{self, Read};

// QuadTree node for Barnes-Hut acceleration
struct QuadNode {
    // bounds of this node (square-ish region)
    xmin: f64,
    ymin: f64,
    xmax: f64,
    ymax: f64,

    // mass distribution
    mass: f64,
    cmx: f64,
    cmy: f64,

    // leaf data
    // idx >= 0 indicates this node is a leaf containing a point with index idx
    // idx == -2 means empty leaf
    // idx == -1 means internal node (has children)
    idx: i64,
    px: f64, // point coordinates if leaf
    py: f64,

    // children (4 quadrants). Lazily created as needed.
    children: Vec<Option<Box<QuadNode>>>, // length 4
}

impl QuadNode {
    fn new(xmin: f64, ymin: f64, xmax: f64, ymax: f64) -> Self {
        QuadNode {
            xmin,
            ymin,
            xmax,
            ymax,
            mass: 0.0,
            cmx: 0.0,
            cmy: 0.0,
            idx: -2, // empty leaf
            px: 0.0,
            py: 0.0,
            children: vec![None, None, None, None],
        }
    }

    // Determine which child quadrant a point belongs to
    fn child_of_point(&self, x: f64, y: f64) -> usize {
        let mx = (self.xmin + self.xmax) * 0.5;
        let my = (self.ymin + self.ymax) * 0.5;
        if x <= mx {
            if y <= my { 0 } else { 2 }
        } else {
            if y <= my { 1 } else { 3 }
        }
    }

    // Bounds for a given child index
    fn child_bounds(&self, idx: usize) -> (f64, f64, f64, f64) {
        let mx = (self.xmin + self.xmax) * 0.5;
        let my = (self.ymin + self.ymax) * 0.5;
        match idx {
            0 => (self.xmin, self.ymin, mx, my),       // NW
            1 => (mx, self.ymin, self.xmax, my),       // NE
            2 => (self.xmin, my, mx, self.ymax),       // SW
            3 => (mx, my, self.xmax, self.ymax),       // SE
            _ => (self.xmin, self.ymin, self.xmax, self.ymax),
        }
    }

    // Create a child quadrant if needed
    fn spawn_child(&mut self, idx: usize) {
        let (xmin, ymin, xmax, ymax) = self.child_bounds(idx);
        self.children[idx] = Some(Box::new(QuadNode::new(xmin, ymin, xmax, ymax)));
    }

    // Insert a point (index, x, y) into the quadtree
    fn insert(&mut self, idx_p: usize, x: f64, y: f64) {
        // Empty leaf -> store here
        if self.idx == -2 {
            self.idx = idx_p as i64;
            self.px = x;
            self.py = y;
            return;
        }

        // If this node currently stores a single point -> subdivide and reinsert both
        if self.idx >= 0 {
            let old_idx = self.idx as usize;
            let old_x = self.px;
            let old_y = self.py;

            self.idx = -1; // internal now

            let old_child = self.child_of_point(old_x, old_y);
            if self.children[old_child].is_none() {
                self.spawn_child(old_child);
            }
            self
                .children
                [old_child]
                .as_mut()
                .unwrap()
                .insert(old_idx, old_x, old_y);

            // Insert the new point
            let new_child = self.child_of_point(x, y);
            if self.children[new_child].is_none() {
                self.spawn_child(new_child);
            }
            self
                .children
                [new_child]
                .as_mut()
                .unwrap()
                .insert(idx_p, x, y);
            return;
        }

        // Internal node: forward to appropriate child
        let ci = self.child_of_point(x, y);
        if self.children[ci].is_none() {
            self.spawn_child(ci);
        }
        self
            .children[ci]
            .as_mut()
            .unwrap()
            .insert(idx_p, x, y);
    }

    // Recursively compute mass and center of mass
    fn compute_mass(&mut self, _points: &Vec<(f64, f64)>) {
        if self.idx >= 0 {
            // Leaf with a point
            self.mass = 1.0;
            self.cmx = self.px;
            self.cmy = self.py;
        } else {
            // Internal
            self.mass = 0.0;
            self.cmx = 0.0;
            self.cmy = 0.0;
            for ch in &mut self.children {
                if let Some(ref mut child) = ch {
                    child.compute_mass(_points);
                    self.mass += child.mass;
                    self.cmx += child.mass * child.cmx;
                    self.cmy += child.mass * child.cmy;
                }
            }
            if self.mass > 0.0 {
                self.cmx /= self.mass;
                self.cmy /= self.mass;
            }
        }
    }

    // Accumulate repulsive force on node i located at (xi, yi)
    // The force is approximated by Barnes-Hut using theta.
    fn accumulate_repulsion_for_point(
        &self,
        i: usize,
        xi: f64,
        yi: f64,
        theta: f64,
        acc: &mut (f64, f64),
        k: f64,
    ) {
        if self.mass <= 0.0 {
            return;
        }
        // If this is a leaf
        if self.idx >= 0 {
            if self.idx as usize == i {
                return;
            }
            // Use exact leaf's center of mass (its own point)
            let dx = xi - self.cmx;
            let dy = yi - self.cmy;
            let dist = (dx * dx + dy * dy).sqrt() + 1e-6;
            let F = k * k * self.mass / (dist * dist);
            *acc = (
                acc.0 + dx / dist * F,
                acc.1 + dy / dist * F,
            );
            return;
        }

        // Internal: decide to treat as a single mass or descend
        let dx = xi - self.cmx;
        let dy = yi - self.cmy;
        let dist = (dx * dx + dy * dy).sqrt() + 1e-6;
        let s_node = (self.xmax - self.xmin).max(self.ymax - self.ymin);
        if dist > 0.0 && s_node / dist < theta {
            let F = k * k * self.mass / (dist * dist);
            *acc = (acc.0 + dx / dist * F, acc.1 + dy / dist * F);
        } else {
            for ch in &self.children {
                if let Some(ref child) = ch {
                    child.accumulate_repulsion_for_point(i, xi, yi, theta, acc, k);
                }
            }
        }
    }
}

// Helper: clamp a value into [a, b]
fn clamp(v: f64, a: f64, b: f64) -> f64 {
    if v < a {
        a
    } else if v > b {
        b
    } else {
        v
    }
}

fn main() {
    // Read input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut edges_set: HashSet<(usize, usize)> = HashSet::new();
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u >= n || v >= n { continue; }
        if u == v {
            // ignore self-loops
            continue;
        }
        if u < v {
            edges_set.insert((u, v));
        } else {
            edges_set.insert((v, u));
        }
    }

    // Build edge list
    let edges: Vec<(usize, usize)> = edges_set.into_iter().collect();

    // Edge case: no edges
    // still layout nodes roughly on a circle
    // Initialize node positions (simple circle layout to have a reasonable starting point)
    let mut posx = vec![0.0f64; n];
    let mut posy = vec![0.0f64; n];
    if n == 0 {
        return;
    }
    let center_x = 0.0f64;
    let center_y = 0.0f64;
    let radius = (n as f64).sqrt() * 1.0 + 5.0;
    for i in 0..n {
        let a = 2.0 * std::f64::consts::PI * (i as f64) / (n as f64);
        posx[i] = center_x + a.cos() * radius;
        posy[i] = center_y + a.sin() * radius;
    }

    // Compute bounds for quadtree root
    let mut minx = INFINITY;
    let mut maxx = -INFINITY;
    let mut miny = INFINITY;
    let mut maxy = -INFINITY;
    for i in 0..n {
        minx = min(minx, posx[i]);
        maxx = max(maxx, posx[i]);
        miny = min(miny, posy[i]);
        maxy = max(maxy, posy[i]);
    }
    // Add a margin
    let margin = 2.0;
    minx -= margin;
    miny -= margin;
    maxx += margin;
    maxy += margin;

    // Build quadtree root
    let mut root = QuadNode::new(minx, miny, maxx, maxy);

    // Insert all points into quadtree
    for i in 0..n {
        root.insert(i, posx[i], posy[i]);
    }

    // Compute mass distribution for Barnes-Hut
    root.compute_mass(&Vec::new());

    // Determine forcing parameters
    // area and effective "k" constant
    let area = (maxx - minx) * (maxy - miny);
    let mut k = 1.0_f64;
    if area > 0.0 && n > 0 {
        k = (area / (n as f64)).sqrt();
        if k <= 0.0 { k = 1.0; }
    }

    // Layout parameters depending on graph size
    let max_iters: usize = if n <= 2000 {
        200
    } else if n <= 10000 {
        80
    } else {
        40
    };

    // Temperature (maximum displacement per iteration)
    let mut t = (maxx - minx).max(maxy - miny);
    t = t.max(1.0) * 0.5;

    // Precompute edge list for attractive forces
    // We'll also use a simple per-iteration vector of accelerations
    let eps = 1e-6;

    for iter in 0..max_iters {
        // 1) Compute repulsive forces via Barnes-Hut
        let mut accx = vec![0.0f64; n];
        let mut accy = vec![0.0f64; n];

        // Rebuild or reuse quadtree structure each iteration
        // For simplicity, rebuild each iteration using current positions
        let mut root_iter = QuadNode::new(minx, miny, maxx, maxy);
        for i in 0..n {
            // Possibly adjust bounds gradually to ensure all pointsfit
            // We'll keep initial bounds; positions may have moved outside slightly.
            // To be robust, we clamp to existing bounds by expanding if necessary
            let xi = posx[i];
            let yi = posy[i];
            // If out of bounds, extend bounds to include
            // Here we simply clamp coordinates to current root bounds
            // but to avoid frequent re-expansion, use current coords
            root_iter.insert(i, xi, yi);
        }
        root_iter.compute_mass(&Vec::new());

        // Theta parameter controls approximation accuracy
        let theta = 0.5; // moderate

        for i in 0..n {
            let (ax, ay) = {
                let mut acc = (0.0f64, 0.0f64);
                root_iter.accumulate_repulsion_for_point(i, posx[i], posy[i], theta, &mut acc, k);
                acc
            };
            accx[i] = ax;
            accy[i] = ay;
        }

        // 2) Attractive forces along edges
        for &(u, v) in &edges {
            let dx = posx[u] - posx[v];
            let dy = posy[u] - posy[v];
            let dist = (dx * dx + dy * dy).sqrt() + eps;
            let dirx = dx / dist;
            let diry = dy / dist;
            let Fatt = (dist * dist) / k; // attractive force magnitude
            // Apply to u (toward v)
            accx[u] -= dirx * Fatt;
            accy[u] -= diry * Fatt;
            // Apply to v (toward u)
            accx[v] += dirx * Fatt;
            accy[v] += diry * Fatt;
        }

        // 3) Update positions with computed accelerations and simple cooling
        // Normalize the step by temperature t
        let mut max_step = 0.0;
        for i in 0..n {
            let dx = accx[i];
            let dy = accy[i];
            let step = (dx * dx + dy * dy).sqrt();
            if step > max_step {
                max_step = step;
            }
        }
        // Avoid division by zero
        if max_step == 0.0 {
            // If nothing moved, small perturbation to escape stagnation
            for i in 0..n {
                posx[i] += 0.0;
                posy[i] += 0.0;
            }
        } else {
            // Scale by temperature
            let scale = if max_step > t { t / max_step } else { 1.0 };
            for i in 0..n {
                posx[i] += accx[i] * scale * 0.6; // 0.6 to dampen
                posy[i] += accy[i] * scale * 0.6;
            }
        }

        // Optional: clamp coordinates to keep within a reasonable area
        // Expand bounds slightly to accommodate movement
        let pad = 2.0;
        minx -= pad;
        miny -= pad;
        maxx += pad;
        maxy += pad;
        for i in 0..n {
            posx[i] = clamp(posx[i], minx, maxx);
            posy[i] = clamp(posy[i], miny, maxy);
        }

        // Cooling
        t *= 0.98;
        if t < 1e-3 {
            t = 1e-3;
        }

        // Early exit if layout stabilized (very rough)
        // We skip a formal convergence test to keep runtime predictable.
    }

    // Output final coordinates (N lines)
    // Print with reasonable precision
    let mut out = String::new();
    for i in 0..n {
        // Normalize or shift coordinates for nicer absolute values (optional)
        // Here we just print as-is
        out.push_str(&format!("{:.6} {:.6}\n", posx[i], posy[i]));
    }
    print!("{}", out);
}
```
