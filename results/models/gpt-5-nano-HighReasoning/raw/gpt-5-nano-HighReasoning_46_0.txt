Here is a complete GLSL 450 compute shader that performs one timestep of an N-body simulation using explicit Euler integration (velocity updated by acc*dt, position updated by vel*dt). It handles one body per invocation, guards against out-of-bounds indices, uses softening, and skips self-interaction. It writes both the updated position+mass and velocity to the output buffer.

```glsl
#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) writeonly buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint N = counts.x;
    uint i = gl_GlobalInvocationID.x;
    if (i >= N) {
        // Guard against out-of-bounds work-items
        return;
    }

    // Read current state for body i
    vec4 posMass_i = inData[i * 2];
    vec4 vel_i     = inData[i * 2 + 1];

    vec3 pos  = posMass_i.xyz;
    float mass = posMass_i.w;
    vec3 vel  = vel_i.xyz;

    // Read simulation parameters
    float dt = physics.x;
    float soft = physics.y;
    float G    = physics.z;

    // Compute acceleration due to all other bodies
    vec3 acc = vec3(0.0);

    for (uint j = 0u; j < N; ++j) {
        if (j == i) continue; // skip self-interaction

        vec4 posMass_j = inData[j * 2];
        vec3 posJ = posMass_j.xyz;
        float massJ = posMass_j.w;

        vec3 diff = posJ - pos;
        float distSqr = dot(diff, diff) + soft * soft;
        float invDist = inversesqrt(distSqr);
        float invDist3 = invDist * invDist * invDist;

        acc += G * massJ * diff * invDist3;
    }

    // Integration (explicit Euler)
    vec3 newVel = vel + acc * dt;
    vec3 newPos = pos + vel * dt;

    // Write updated state to output buffers
    outData[i * 2]     = vec4(newPos, mass); // keep mass unchanged
    outData[i * 2 + 1] = vec4(newVel, 0.0);
}
```

Notes:
- The shader assumes host code alternates between two buffers (ping-pong) for successive timesteps, launching ceil(N/256) workgroups each time.
- Self-interaction is skipped with the if (j == i) continue guard.
- Softening is applied as soft^2 in the distance computation to avoid singularities.
- Mass is carried over from the input to the output unchanged.
