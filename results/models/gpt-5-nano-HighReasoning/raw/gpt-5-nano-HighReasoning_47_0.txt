```wgsl
// WGSL compute shader: Proof-of-work hash mining (custom ChaCha-like hash)

@group(0) @binding(0) var<storage, read>      baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>             params   : vec4<u32>;
// params.x = difficulty, params.y = range_start, params.z = range_size

fn rotl32(x: u32, n: u32) -> u32 {
  return (x << n) | (x >> (32u - n));
}

fn ctz32(x: u32) -> u32 {
  var v = x;
  var c: u32 = 0u;
  // Count trailing zeros; if x == 0, return 32
  while ((v & 1u) == 0u && c < 32u) {
    v = v >> 1u;
    c = c + 1u;
  }
  return c;
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
  let range_start = params.y;
  let range_size  = params.z;
  let difficulty  = params.x;

  let tid = gid.x;

  // Guard: thread outside of search range
  if (tid >= range_size) {
    return;
  }

  // 1) Initialize state with baseData
  var state : array<u32, 16>;
  for (var i: u32 = 0u; i < 16u; i = i + 1u) {
    state[i] = baseData[i];
  }

  // 2) Apply nonce and XOR with data
  let nonce = range_start + tid;
  for (var i: u32 = 0u; i < 16u; i = i + 1u) {
    state[i] = state[i] ^ (nonce + i);
  }

  // 3) 8 rounds of ChaCha-like mixing (unrolled per group)
  for (var round: u32 = 0u; round < 8u; round = round + 1u) {
    // Group 0: indices 0..3
    {
      var a = state[0];
      var b = state[1];
      var c = state[2];
      var d = state[3];

      a = a + b; d = d ^ a; d = rotl32(d, 16u);
      c = c + d; b = b ^ c; b = rotl32(b, 12u);
      a = a + b; d = d ^ a; d = rotl32(d, 8u);
      c = c + d; b = b ^ c; b = rotl32(b, 7u);

      state[0] = a; state[1] = b; state[2] = c; state[3] = d;
    }

    // Group 1: indices 4..7
    {
      var a = state[4];
      var b = state[5];
      var c = state[6];
      var d = state[7];

      a = a + b; d = d ^ a; d = rotl32(d, 16u);
      c = c + d; b = b ^ c; b = rotl32(b, 12u);
      a = a + b; d = d ^ a; d = rotl32(d, 8u);
      c = c + d; b = b ^ c; b = rotl32(b, 7u);

      state[4] = a; state[5] = b; state[6] = c; state[7] = d;
    }

    // Group 2: indices 8..11
    {
      var a = state[8];
      var b = state[9];
      var c = state[10];
      var d = state[11];

      a = a + b; d = d ^ a; d = rotl32(d, 16u);
      c = c + d; b = b ^ c; b = rotl32(b, 12u);
      a = a + b; d = d ^ a; d = rotl32(d, 8u);
      c = c + d; b = b ^ c; b = rotl32(b, 7u);

      state[8]  = a; state[9]  = b; state[10] = c; state[11] = d;
    }

    // Group 3: indices 12..15
    {
      var a = state[12];
      var b = state[13];
      var c = state[14];
      var d = state[15];

      a = a + b; d = d ^ a; d = rotl32(d, 16u);
      c = c + d; b = b ^ c; b = rotl32(b, 12u);
      a = a + b; d = d ^ a; d = rotl32(d, 8u);
      c = c + d; b = b ^ c; b = rotl32(b, 7u);

      state[12] = a; state[13] = b; state[14] = c; state[15] = d;
    }
  }

  // 4) Fold 16 -> 2 words
  let h0 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
  let h1 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];

  // 5) Trailing zero count of 64-bit value (h0 | (h1<<32))
  var tz: u32;
  if (h0 != 0u) {
    tz = ctz32(h0);
  } else {
    tz = 32u + ctz32(h1);
  }

  // 6) If meets difficulty, write result
  if (tz >= difficulty) {
    result[0] = 1u;
    result[1] = nonce;
    result[2] = h0;
    result[3] = h1;
  }
}
```
