```cpp
#include <bits/stdc++.h>
using namespace std;

// Solver for Maximum Weight Independent Set on a general graph
// Internally uses a branch-and-bound search with bitset representation.
// Works component-wise to improve performance on larger graphs.

class MWIS {
public:
    using Mask = vector<uint64_t>;

    int n; // number of vertices in this component
    int L; // number of 64-bit blocks in the mask
    vector<long long> w;            // weights
    vector<Mask> adj;                 // adjacency masks, adj[i] has bits of neighbors of i

    vector<int> best_choice;            // best selection (0/1) for this component
    long long best_weight;

    vector<int> curr_choice;            // current selection

    MWIS(int n_, const vector<long long>& w_, const vector<Mask>& adj_)
        : n(n_), w(w_), adj(adj_) {
        L = (n + 63) / 64;
        best_choice.assign(n, 0);
        curr_choice.assign(n, 0);
        best_weight = 0;
    }

    // Solve and return pair<best_weight, best_choice>
    pair<long long, vector<int>> solve() {
        Mask R(L, 0);
        // Initial mask: all vertices are available
        for (int i = 0; i < n; ++i) {
            R[i / 64] |= (1ULL << (i % 64));
        }
        dfs(R, 0);
        return {best_weight, best_choice};
    }

private:
    static bool isZero(const Mask& R) {
        for (uint64_t x : R) if (x) return false;
        return true;
    }

    long long sumWeights(const Mask& R) const {
        long long s = 0;
        for (int word = 0; word < L; ++word) {
            uint64_t mask = R[word];
            while (mask) {
                int t = __builtin_ctzll(mask);
                int idx = word * 64 + t;
                if (idx < n) s += w[idx];
                mask &= mask - 1;
            }
        }
        return s;
    }

    void dfs(Mask& R, long long curW) {
        if (isZero(R)) {
            if (curW > best_weight) {
                best_weight = curW;
                best_choice = curr_choice;
            }
            return;
        }

        long long ub = curW + sumWeights(R);
        if (ub <= best_weight) return; // prune

        // pick a vertex v from R (first set bit)
        int v = -1;
        for (int word = 0; word < L; ++word) {
            if (R[word]) {
                int t = __builtin_ctzll(R[word]);
                v = word * 64 + t;
                break;
            }
        }
        if (v < 0 || v >= n) {
            // Shouldn't happen, but guard
            if (curW > best_weight) {
                best_weight = curW;
                best_choice = curr_choice;
            }
            return;
        }

        // Branch 1: include v
        Mask Rincl = R;
        // Remove neighbors of v
        for (int word = 0; word < L; ++word) Rincl[word] &= ~adj[v][word];
        // Also remove v itself
        Rincl[v / 64] &= ~(1ULL << (v % 64));

        curr_choice[v] = 1;
        dfs(Rincl, curW + w[v]);
        curr_choice[v] = 0;

        // Branch 2: exclude v
        Mask Rexcl = R;
        Rexcl[v / 64] &= ~(1ULL << (v % 64));

        curr_choice[v] = 0;
        dfs(Rexcl, curW);
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<long long> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    vector<pair<int,int>> edges;
    edges.reserve(m);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        // Assume 0-based indices as per problem statement (c_0 ... c_{n-1})
        // If input unexpectedly uses 1-based, you could uncomment the following two lines:
        // if (u > 0) --u;
        // if (v > 0) --v;
        edges.emplace_back(u, v);
    }

    // Drop non-positive weights (they can never help in MWIS)
    vector<int> mapToReduced(n, -1);
    vector<int> orig; orig.reserve(n);
    vector<long long> w;
    for (int i = 0; i < n; ++i) {
        if (c[i] > 0) {
            mapToReduced[i] = (int)orig.size();
            orig.push_back(i);
            w.push_back(c[i]);
        }
    }

    int n2 = (int)orig.size();
    if (n2 == 0) {
        // No positive weight vertices -> optimum is 0 with all zeros
        cout << 0 << "\n";
        for (int i = 0; i < n; ++i) {
            cout << 0 << (i + 1 == n ? '\n' : ' ');
        }
        return 0;
    }

    // Build adjacency for reduced graph
    int L = (n2 + 63) / 64;
    vector<vector<uint64_t>> adjGlobal(n2, vector<uint64_t>(L, 0));

    for (auto &e : edges) {
        int u = e.first, v = e.second;
        if (u < 0 || v < 0) continue;
        if (u >= n || v >= n) continue;
        int a = mapToReduced[u];
        int b = mapToReduced[v];
        if (a == -1 || b == -1) continue;
        // set edge a-b
        adjGlobal[a][b / 64] |= (1ULL << (b % 64));
        adjGlobal[b][a / 64] |= (1ULL << (a % 64));
    }

    // Compute connected components of the reduced graph
    vector<int> comp_id(n2, -1);
    vector<vector<int>> comps;

    for (int i = 0; i < n2; ++i) {
        if (comp_id[i] != -1) continue;
        // BFS/DFS to collect component
        vector<int> compNodes;
        queue<int> q;
        comp_id[i] = (int)comps.size();
        q.push(i);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            compNodes.push_back(v);
            // iterate neighbors of v in adjGlobal
            for (int word = 0; word < L; ++word) {
                uint64_t mask = adjGlobal[v][word];
                while (mask) {
                    int t = __builtin_ctzll(mask);
                    int nb = word * 64 + t;
                    if (nb < n2 && comp_id[nb] == -1) {
                        comp_id[nb] = (int)comps.size();
                        q.push(nb);
                    }
                    mask &= mask - 1;
                }
            }
        }
        comps.push_back(move(compNodes));
    }

    // Solve MWIS per component and accumulate
    vector<int> ansReduced(n2, 0);
    long long totalBest = 0;

    for (const auto &compNodes : comps) {
        int k = (int)compNodes.size();
        if (k == 0) continue;

        // Build local weights
        vector<long long> wLocal(k);
        for (int i = 0; i < k; ++i) wLocal[i] = w[ compNodes[i] ];

        // Build local adjacency
        int L2 = (k + 63) / 64;
        vector<vector<uint64_t>> adjLocal(k, vector<uint64_t>(L2, 0));
        for (int iLocal = 0; iLocal < k; ++iLocal) {
            int gi = compNodes[iLocal];
            for (int jLocal = 0; jLocal < k; ++jLocal) {
                if (iLocal == jLocal) continue;
                int gj = compNodes[jLocal];
                // if edge exists between gi and gj in the global reduced graph
                if ( (adjGlobal[gi][gj/64] >> (gj%64)) & 1ULL ) {
                    adjLocal[iLocal][ jLocal / 64 ] |= (1ULL << (jLocal % 64));
                }
            }
        }

        // Solve MWIS for this component
        MWIS solver(k, wLocal, adjLocal);
        auto res = solver.solve();
        long long bestW = res.first;
        const vector<int>& bestLocal = res.second;

        totalBest += bestW;

        for (int iLocal = 0; iLocal < k; ++iLocal) {
            if (bestLocal[iLocal] == 1) {
                int globalIdx = compNodes[iLocal];
                ansReduced[globalIdx] = 1;
            }
        }
    }

    // Build final answer over original n variables
    vector<int> ans(n, 0);
    for (int i = 0; i < n2; ++i) {
        if (ansReduced[i]) {
            int origIndex = orig[i];
            ans[origIndex] = 1;
        }
    }

    // Output
    cout << totalBest << "\n";
    for (int i = 0; i < n; ++i) {
        cout << ans[i] << (i + 1 == n ? '\n' : ' ');
    }

    return 0;
}
```
