using System;
using System.Globalization;

namespace OrbitalTravelingSalesman
{
    // A lightweight heuristic solver for the Orbital Travelling Salesman Problem.
    // - Uses a simple distance-like cost between orbital states to guide a greedy NN tour.
    // - Applies a 2-opt local optimization to improve the tour for larger inputs.
    // - Scales well from 2 up to ~100 stations with O(N^2) cost computations and a
    //   small 2-opt refinement.
    //
    // Input format (stdin):
    // Line 1: N mu
    // Line 2: start orbit: x y z vx vy vz
    // Lines 3..N+2: station orbits: x y z vx vy vz
    //
    // Output format (stdout):
    // One line containing N integers: the visitation order of station indices (0..N-1).
    //
    // Note: This is a practical heuristic. It provides reasonable delta-V estimates while
    // remaining efficient for large N and is deterministic (ties broken by lower index).

    class Program
    {
        // Simple 3D vector with basic operations
        struct Vec3
        {
            public double x, y, z;
            public Vec3(double x = 0, double y = 0, double z = 0)
            {
                this.x = x; this.y = y; this.z = z;
            }

            public double Length()
            {
                return Math.Sqrt(x * x + y * y + z * z);
            }

            public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.x - b.x, a.y - b.y, a.z - b.z);
            public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.x + b.x, a.y + b.y, a.z + b.z);
            public static Vec3 operator *(double k, Vec3 a) => new Vec3(a.x * k, a.y * k, a.z * k);
            public static Vec3 operator *(Vec3 a, double k) => new Vec3(a.x * k, a.y * k, a.z * k);

            public double Dot(Vec3 b) => x * b.x + y * b.y + z * b.z;
        }

        // Per-station state: position r and velocity v
        class OrbitState
        {
            public Vec3 r;
            public Vec3 v;
            public OrbitState()
            {
                r = new Vec3();
                v = new Vec3();
            }
        }

        // Fast scanner to parse input efficiently
        class FastScanner
        {
            private string[] data;
            private int idx;

            public FastScanner()
            {
                // Read all input and split by whitespace
                var s = Console.In.ReadToEnd();
                data = s.Split(new char[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                idx = 0;
            }

            public int NextInt()
            {
                return int.Parse(data[idx++], CultureInfo.InvariantCulture);
            }

            public double NextDouble()
            {
                return double.Parse(data[idx++], CultureInfo.InvariantCulture);
            }
        }

        // Cost between two orbital states (A -> B) using a simple impulsive-maneuver-inspired metric.
        // This is an approximation intended for scalable routing, not a full Lambert transfer solution.
        static double Cost(int aIndex, int bIndex, OrbitState[] stations, OrbitState start)
        {
            Vec3 rA, vA;
            if (aIndex == -1)
            {
                rA = start.r;
                vA = start.v;
            }
            else
            {
                rA = stations[aIndex].r;
                vA = stations[aIndex].v;
            }

            Vec3 rB = stations[bIndex].r;
            Vec3 vB = stations[bIndex].v;

            // Approximate delta-v needed as the norm of velocity difference
            double dv = (vB - vA).Length();

            // Small penalties to reflect radial distance differences and path length
            double radialDiff = Math.Abs(rB.Length() - rA.Length());
            double dR = (rB - rA).Length();

            // Weights are chosen to keep costs reasonable and scalable
            double cost = dv + 0.01 * radialDiff + 0.001 * dR;
            return cost;
        }

        // Total cost of a tour given an ordered route
        static double RouteCost(int[] route, OrbitState[] stations, OrbitState start)
        {
            if (route.Length == 0) return 0;
            double total = 0;
            total += Cost(-1, route[0], stations, start);
            for (int i = 0; i < route.Length - 1; i++)
            {
                total += Cost(route[i], route[i + 1], stations, start);
            }
            return total;
        }

        // 2-opt local optimization to improve a tour
        static int[] TwoOptImprove(int[] route, OrbitState[] stations, OrbitState start)
        {
            if (route.Length <= 3) return route;

            bool improved = true;
            while (improved)
            {
                improved = false;
                double currentCost = RouteCost(route, stations, start);

                for (int i = 0; i < route.Length - 1 && !improved; i++)
                {
                    for (int j = i + 2; j < route.Length; j++)
                    {
                        // Avoid reversing the entire route in a closed-loop sense
                        if (i == 0 && j == route.Length - 1) continue;

                        int[] newRoute = (int[])route.Clone();
                        Array.Reverse(newRoute, i + 1, j - (i + 1) + 1);

                        double newCost = RouteCost(newRoute, stations, start);
                        if (newCost < currentCost - 1e-12)
                        {
                            route = newRoute;
                            improved = true;
                            break;
                        }
                    }
                }
            }
            return route;
        }

        static void Main(string[] args)
        {
            var sc = new FastScanner();

            int N = sc.NextInt();
            double mu = sc.NextDouble(); // Not used directly here, but read for compatibility

            // Start state
            var start = new OrbitState();
            start.r = new Vec3(sc.NextDouble(), sc.NextDouble(), sc.NextDouble());
            start.v = new Vec3(sc.NextDouble(), sc.NextDouble(), sc.NextDouble());

            // Stations
            OrbitState[] stations = new OrbitState[N];
            for (int i = 0; i < N; i++)
            {
                stations[i] = new OrbitState();
                stations[i].r = new Vec3(sc.NextDouble(), sc.NextDouble(), sc.NextDouble());
                stations[i].v = new Vec3(sc.NextDouble(), sc.NextDouble(), sc.NextDouble());
            }

            // Greedy Nearest-Neighbor heuristic: start from the given initial state
            int[] route = new int[N];
            bool[] used = new bool[N];
            int current = -1;

            for (int step = 0; step < N; step++)
            {
                double best = double.PositiveInfinity;
                int bestIdx = -1;
                for (int j = 0; j < N; j++)
                {
                    if (used[j]) continue;
                    double c = Cost(current, j, stations, start);
                    if (c < best - 1e-12)
                    {
                        best = c;
                        bestIdx = j;
                    }
                }
                if (bestIdx == -1) break;
                route[step] = bestIdx;
                used[bestIdx] = true;
                current = bestIdx;
            }

            // Optional improvement with 2-opt
            route = TwoOptImprove(route, stations, start);

            // Output the visitation order
            Console.WriteLine(string.Join(" ", route));
        }
    }
}
