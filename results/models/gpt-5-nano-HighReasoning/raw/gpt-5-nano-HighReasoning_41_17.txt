; SPIR-V
; Version: 1.0
; Generator: SPIR-V text (brick shader)
OpCapability Shader
OpMemoryModel Logical GLSL450

%void = type_void
%func = type_function %void

%float = type_float 32
%v2float = type_vector %float 2
%v3float = type_vector %float 3
%v4float = type_vector %float 4
%mat4 = type_matrix %v4float 4
; UBO with std140-like offsets (model, view, projection, lightPos, cameraPos, params)
%_struct_UBO = type_struct %mat4 %mat4 %mat4 %v4float %v4float %v4float

; Pointer types
%_ptr_Uniform_UBO = type_pointer Uniform %_struct_UBO
%_ptr_Uniform_vec4 = type_pointer Uniform %v4float

%uint = type_int 32 0
; Indices/constants
%_const_uint3 = constant %uint 3
%_const_uint2 = constant %uint 2
%_const_uint0 = constant %uint 0
%_one = constant %float 1.0
%_zero = constant %float 0.0
%_four = constant %float 4.0
%_eight = constant %float 8.0
%_eight_int = constant %uint 8
%_tileU = constant %float 8.0
%_tileV = constant %float 16.0
%_mortarX = constant %float 0.05
%_mortarY = constant %float 0.10
%_half = constant %float 0.5
%_shade4 = constant %float 0.4
%_shade6 = constant %float 0.6

; Input/output and uniforms
%_ptr_Input_vec3 = type_pointer Input %v3float
%_ptr_Input_vec2 = type_pointer Input %v2float

%inWorldPos = variable %_ptr_Input_vec3
%inNormal   = variable %_ptr_Input_vec3
%inUV       = variable %_ptr_Input_vec2
%inTangent  = variable %_ptr_Input_vec3
%inColor    = variable %_ptr_Input_vec3

%_ptr_Output_vec4 = type_pointer Output %v4float
%outColor   = variable %_ptr_Output_vec4

; Uniform block instance
ubo = variable %_ptr_Uniform_UBO

; Decorations (locations, bindings, block, offsets)
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4

OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %_struct_UBO 0 Offset 0
OpMemberDecorate %_struct_UBO 1 Offset 64
OpMemberDecorate %_struct_UBO 2 Offset 128
OpMemberDecorate %_struct_UBO 3 Offset 192
OpMemberDecorate %_struct_UBO 4 Offset 208
OpMemberDecorate %_struct_UBO 5 Offset 224
OpDecorate %_struct_UBO Block

; Import GLSL std450 for fract/normalize/max etc.
%GLSLstd450 = import "GLSL.std.450"

; Entry point declaration
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor
OpExecutionMode %main OriginUpperLeft

; Function types and entry point function
%_ptr_Function_void = type_pointer Function %void
%main = function %void None
; Entry block
%Main = label

; Load inputs
%worldPos = OpLoad %v3float %inWorldPos
%normalIn = OpLoad %v3float %inNormal
%uv = OpLoad %v2float %inUV
%colorInput = OpLoad %v3float %inColor

; Normalize normal
%N = OpExtInst %v3float %GLSLstd450 Normalize %normalIn

; Light position from UBO
%lightPtr = OpAccessChain %_ptr_Uniform_vec4 %ubo %_const_uint3
%lightPosW = OpLoad %v4float %lightPtr
%lx = OpCompositeExtract %float %lightPosW 0
%ly = OpCompositeExtract %float %lightPosW 1
%lz = OpCompositeExtract %float %lightPosW 2
%lightPos = OpCompositeConstruct %v3float %lx %ly %lz

; Compute L = normalize(lightPos - worldPos)
%toLight = OpFSub %v3float %lightPos %worldPos
%L = OpExtInst %v3float %GLSLstd450 Normalize %toLight

; Diffuse term: max(dot(N, L), 0.0)
%doto = OpDot %float %N %L
%diffused = OpExtInst %float %GLSLstd450 Max %doto %_zero

; Lambert shading factor
%shadeMul = OpFMul %float %diffused %%_6
%shade = OpFAdd %float %_shade4 %%shadeMul
; (Note: above uses constants; fix names to actual IDs)
; We'll rewrite the last few lines properly:

; Recompute shading with defined constants (to ensure correctness in SPIR-V)
%_tmpShadeMul = OpFMul %float %diffused %%_shade6
%shadeFinal = OpFAdd %float %_shade4 %%_tmpShadeMul

; Pattern coordinates
; UV scale to 8x16 tiles
%u = OpCompositeExtract %float %uv 0
%v = OpCompositeExtract %float %uv 1
%U = OpFMul %float %u %_tileU
%V = OpFMul %float %v %_tileV

; Row parity for stagger (every other row offset by 0.5 bricks)
%rowInt = OpConvertFToU %uint %V
%rowMod2 = OpUMod %uint %rowInt %_8  ; 8 as uint tile width in bricks
%rowOdd = OpIEqual %bool %rowMod2 %_const_uint2
; Offset = 0.5 bricks if odd, else 0
%offsetBricks = OpSelect %float %rowOdd %_half %_zero
%Ushift = OpFSub %float %U %offsetBricks

; Wrap Ushift into [0,8)
%wrapFlag = OpFOrdLessThan %bool %%Ushift %_zero
%UshiftWrapped = OpFAdd %float %%Ushift %_eight
%UshiftFinal = OpSelect %float %wrapFlag %UshiftWrapped %Ushift

; Local coordinates within brick
%localX = OpExtInst %float %GLSLstd450 Fract %UshiftFinal
%localY = OpExtInst %float %GLSLstd450 Fract %V

; Mortar regions
%mX = %_mortarX
%mY = %_mortarY
%oneMinusMX = OpFSub %float %_one %mX
%condXLeft = OpFOrdLessThan %bool %localX %mX
%condXRight = OpFOrdGreaterThan %bool %localX %oneMinusMX
%orX = OpLogicalOr %bool %condXLeft %condXRight

%oneMinusMY = OpFSub %float %_one %mY
%condYLow = OpFOrdLessThan %bool %localY %mY
%condYHigh = OpFOrdGreaterThan %bool %localY %oneMinusMY
%orY = OpLogicalOr %bool %condYLow %condYHigh

%mortarBool = OpLogicalOr %bool %orX %orY

; Brick and mortar base colors
%brickR = OpConstant %float 0.7
%brickG = OpConstant %float 0.2
%brickB = OpConstant %float 0.1
%brickCol = OpCompositeConstruct %v3float %brickR %brickG %brickB

%mortarR = OpConstant %float 0.7
%mortarG = OpConstant %float 0.7
%mortarB = OpConstant %float 0.7
%mortarCol = OpCompositeConstruct %v3float %mortarR %mortarG %mortarB

; Shade brick and mortar bases
%shBrick = OpFMul %v3float %brickCol % %shadeFinal
%shMortar = OpFMul %v3float %mortarCol % %shadeFinal

; Select base color by mortar/brick region
%baseColor = OpSelect %v3float %mortarBool %shMortar %shBrick

; Multiply by vertex color
%finalColor3 = OpFMul %v3float %baseColor % %colorInput

; Output color with alpha = 1.0
%outR = OpCompositeExtract %float %finalColor3 0
%outG = OpCompositeExtract %float %finalColor3 1
%outB = OpCompositeExtract %float %finalColor3 2
%outA = OpConstant %float 1.0
%outColorVec4 = OpCompositeConstruct %v4float %outR %outG %outB %outA
OpStore %outColor %outColorVec4

OpReturn
OpFunctionEnd

; End of module
