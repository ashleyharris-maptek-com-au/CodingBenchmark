struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

// helper to compute fractional part (to avoid relying on fract/frac ambiguity)
float fract(float x) { return x - floor(x); }

float4 main(PSInput input) : SV_Target0 {
    // Simple Lambertian lighting
    float3 worldPos = input.worldPos;
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float lambert = 0.4 + 0.6 * diffuse;

    // Procedural bricks: 8x16 UV tiling with half-brick offset on alternating rows
    const float brickW = 1.0 / 8.0; // 8 bricks across
    int row = int(floor(input.uv.y * 16.0));
    float offset = (((row & 1) != 0) ? (0.5 * brickW) : 0.0);
    float s = input.uv.x + offset;
    float u = fract(s);

    // Mortar thickness in X (two-sided around vertical brick boundaries)
    const float mortarX = 0.05;
    float mortarXFrac = mortarX / brickW; // relative thickness within a brick cell
    bool isMortarX = (u < mortarXFrac) || (u > (1.0 - mortarXFrac));

    // Mortar thickness in Y: use stripe bands along horizontal grid lines
    float yBand = fract(input.uv.y * 16.0);
    bool isMortarY = (yBand < 0.1) || (yBand > 0.9); // 0.1 in sY units => ~0.1/16 uv units per band

    bool isMortar = isMortarX || isMortarY;

    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 baseColor = isMortar ? mortarColor : brickColor;

    float3 color = baseColor * lambert;

    return float4(color, 1.0);
}
