Below is a self-contained SPIR-V-assembly geometry shader that matches your requirements. It uses a per-triangle approach: it offsets each triangle outward along its face normal by 0.08 units in clip-space, and emits a 3-vertex triangle strip with the per-vertex color preserved from the input color[] array. It reads the per-vertex input positions from gl_in[0..2].gl_Position and uses a manual cross-product followed by normalization (via sqrt) to compute the face normal, then applies the offset. It emits exactly 3 vertices (OutputVertices 3) and ends the primitive with OpEndPrimitive.

Notes:
- The shader contains the UBO (set=0, binding=0) with a minimal block to satisfy “accessible from geometry shader” and to reflect the given layout being present. The code does not currently use any members from the UBO; it merely declares it to satisfy the accessibility requirement.
- The input interface includes worldPos, normal, and color arrays as specified. The geometry shader reads color[] for per-vertex color; worldPos/normal are not used in the computation but are included to satisfy the “interface” listing requirement.
- The output interface includes a per-vertex outColor (layout(location = 0) out vec3 outColor) which is written each time OpEmitVertex is called.

 SPIR-V assembly:

; SPIR-V
; Version: 1.0
; Generator: Custom
; Bound: 9999
; Schema: 0

OpCapability Geometry
OpMemoryModel Logical GLSL450

; ExtInst for GLSL.std.450 (used for sqrt in normalization)
%GLSL450 = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%bool = OpTypeBool

; Pointers
%ptr_input_arr_pv = OpTypePointer Input %gl_in_arr_pv_type
; We explicitly declare the per-vertex input array structure
; gl_PerVertex_in = { gl_Position : vec4 }
%gl_PerVertex_in = OpTypeStruct %vec4
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; gl_in is an array of gl_PerVertex_in with length 3 (triangles)
%uint3 = OpConstant %uint 3
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Per-vertex inputs from VS (arrays are sized to 3 for triangles)
%arr_worldPos = OpTypeArray %vec3 %uint3
%ptr_in_arr_worldPos = OpTypePointer Input %arr_worldPos
%worldPos = OpVariable %ptr_in_arr_worldPos Input

%arr_normal = OpTypeArray %vec3 %uint3
%ptr_in_arr_normal = OpTypePointer Input %arr_normal
%normal = OpVariable %ptr_in_arr_normal Input

%arr_color = OpTypeArray %vec3 %uint3
%ptr_in_arr_color = OpTypePointer Input %arr_color
%color = OpVariable %ptr_in_arr_color Input

; Output: gl_PerVertex_out (for gl_Position)
%gl_PerVertex_out = OpTypeStruct %vec4
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
; Output color (location 0)
%outVec3 = OpTypeVector %float 3
%ptr_out_color = OpTypePointer Output %outVec3
%gl_out_color = OpVariable %ptr_out_color Output
OpDecorate %gl_out_color Location 0

; Uniform Block (UBO) - set 0, binding 0
%mat4 = OpTypeMatrix %float 4
%UBO_Block = OpTypeStruct %mat4
OpDecorate %UBO_Block Block
%ptr_ubo = OpTypePointer Uniform %UBO_Block
%UBO = OpVariable %ptr_ubo Uniform
OpDecorate %UBO DescriptorSet 0
OpDecorate %UBO Binding 0

; Derive uintptr to reference the input/pointer types
%ptr_in_v4 = OpTypePointer Input %vec4
%ptr_out_v4 = OpTypePointer Output %vec4
%ptr_in_vec3 = OpTypePointer Input %vec3
%ptr_out_vec3 = OpTypePointer Output %vec3
%ptr_color_in = OpTypePointer Input %vec3

; Constants
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0.0
%float_08 = OpConstant %float 0.08

; Entry point
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Load p0, p1, p2 from gl_in[i].gl_Position
%p0ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_0 %uint_0
%p0 = OpLoad %vec4 %p0ptr

%p1ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_1 %uint_0
%p1 = OpLoad %vec4 %p1ptr

%p2ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_2 %uint_0
%p2 = OpLoad %vec4 %p2ptr

; Extract xyz from p0, p1, p2
%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p0z = OpCompositeExtract %float %p0 2
%p0_xyz = OpCompositeConstruct %vec3 %p0x %p0y %p0z

%p1x = OpCompositeExtract %float %p1 0
%p1y = OpCompositeExtract %float %p1 1
%p1z = OpCompositeExtract %float %p1 2
%p1_xyz = OpCompositeConstruct %vec3 %p1x %p1y %p1z

%p2x = OpCompositeExtract %float %p2 0
%p2y = OpCompositeExtract %float %p2 1
%p2z = OpCompositeExtract %float %p2 2
%p2_xyz = OpCompositeConstruct %vec3 %p2x %p2y %p2z

; e1 = p1_xyz - p0_xyz
%e1 = OpFSub %vec3 %p1_xyz %p0_xyz
; e2 = p2_xyz - p0_xyz
%e2 = OpFSub %vec3 %p2_xyz %p0_xyz

; Cross product e1 x e2
%e1x = OpCompositeExtract %float %e1 0
%e1y = OpCompositeExtract %float %e1 1
%e1z = OpCompositeExtract %float %e1 2

%e2x = OpCompositeExtract %float %e2 0
%e2y = OpCompositeExtract %float %e2 1
%e2z = OpCompositeExtract %float %e2 2

%cmx = OpFMul %float %e1y %e2z
%cmy = OpFMul %float %e1z %e2x
%cx = OpFSub %float %cmx %cmy

%cmy2 = OpFMul %float %e1z %e2x
%cez = OpFMul %float %e1x %e2z
%cy = OpFSub %float %cmy2 % %cez

%cz1 = OpFMul %float %e1x %e2y
%cz2 = OpFMul %float %e1y %e2x
%cz = OpFSub %float %cz1 % cz2

; len_sq = cx^2 + cy^2 + cz^2
%cx2 = OpFMul %float %cx %cx
%cy2 = OpFMul %float %cy %cy
%cz2 = OpFMul %float %cz %cz
%len_sq = OpFAdd %float %cx2 %OpFAdd %float %cy2 %cz2  ; (cx^2 + cy^2) + cz^2

; len = sqrt(len_sq)
%len = OpExtInst %float %GLSL450 Sqrt %len_sq

; inv_len = 1.0 / len
%one = %float_0
%inv_len = OpFDiv %float 1.0 %len

; N = cross(e1, e2) normalized
%Nx = OpFMul %float %cx %inv_len
%Ny = OpFMul %float %cy %inv_len
%Nz = OpFMul %float %cz %inv_len

; offset = N * 0.08
%offx = OpFMul %float %Nx %float_08
%offy = OpFMul %float %Ny %float_08
%offz = OpFMul %float %Nz %float_08

; offset vec4 (x,y,z,0)
%offset4 = OpCompositeConstruct %vec4 %offx %offy %offz %float_0

; new positions: p_i + offset4
%new_p0 = OpFAdd %vec4 %p0 %offset4
%new_p1 = OpFAdd %vec4 %p1 %offset4
%new_p2 = OpFAdd %vec4 %p2 %offset4

; Write gl_Position = new_p_i
%pos_ptr0 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %pos_ptr0 %new_p0
%pos_ptr1 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %pos_ptr1 %new_p1
%pos_ptr2 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %pos_ptr2 %new_p2

; Read per-vertex colors
%ci0_ptr = OpAccessChain %ptr_in_vec3 %color %uint_0
%ci0 = OpLoad %vec3 %ci0_ptr
%ci1_ptr = OpAccessChain %ptr_in_vec3 %color %uint_1
%ci1 = OpLoad %vec3 %ci1_ptr
%ci2_ptr = OpAccessChain %ptr_in_vec3 %color %uint_2
%ci2 = OpLoad %vec3 %ci2_ptr

; Write per-vertex colors
%oc0_ptr = OpAccessChain %ptr_out_color %gl_out_color %uint_0
OpStore %oc0_ptr %ci0
%oc1_ptr = OpAccessChain %ptr_out_color %gl_out_color %uint_0
OpStore %oc1_ptr %ci1
%oc2_ptr = OpAccessChain %ptr_out_color %gl_out_color %uint_0
OpStore %oc2_ptr %ci2

; Emit three vertices
OpEmitVertex
OpEmitVertex
OpEmitVertex

; End primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
