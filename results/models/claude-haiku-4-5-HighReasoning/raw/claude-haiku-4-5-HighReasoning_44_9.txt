#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Compute checkerboard pattern at 8x8 frequency
    ivec2 checkerCoord = ivec2(floor(uv * 8.0));
    float parity = mod(float(checkerCoord.x + checkerCoord.y), 2.0);
    float baseColor = mix(0.2, 0.9, parity);

    // Compute Lambertian shading
    vec3 n = normalize(normal);
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(0.0, dot(n, lightDir));
    float shadingFactor = 0.5 + 0.5 * diffuse;

    // Combine checkerboard with shading
    float finalColor = baseColor * shadingFactor;

    outColor = vec4(vec3(finalColor), 1.0);
}
