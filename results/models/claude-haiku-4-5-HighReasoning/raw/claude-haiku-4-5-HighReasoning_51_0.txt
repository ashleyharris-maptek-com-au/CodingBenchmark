import math

# Module-level state for PID controllers and history
_state_history = {
    'prev_time': 0,
    'altitude_integral': 0,
    'heading_integral': 0,
    'speed_integral': 0,
    'vs_integral': 0,
    'prev_alt': 0,
    'prev_hdg': 0,
    'prev_ias': 0,
    'prev_roll': 0,
    'trim_state': 0,
    'landing_phase': 'cruise',  # cruise, descent, approach, landing
    'gear_deployed': False,
    'flap_stage': 0,
}

def _get_valid_reading(readings, valid_flags):
    """Sensor fusion: returns median of valid sensors, or best guess if all fail."""
    valid_values = []
    for i, (reading, valid) in enumerate(zip(readings, valid_flags)):
        if valid and reading is not None:
            valid_values.append(reading)

    if valid_values:
        return sorted(valid_values)[len(valid_values) // 2]  # median

    # Fallback: use any reading that looks reasonable
    return next((r for r in readings if r is not None and r is not False), readings[0])

def _clamp(value, min_val, max_val):
    """Clamp value to range."""
    return max(min_val, min(max_val, value))

def autopilot_step(state: dict, dt: float) -> dict:
    global _state_history

    # Extract fused sensor readings
    alt_ft = _get_valid_reading(
        [state.get('baro_alt_1_ft'), state.get('baro_alt_2_ft'), state.get('baro_alt_3_ft')],
        [state.get('baro_alt_1_valid'), state.get('baro_alt_2_valid'), state.get('baro_alt_3_valid')]
    )
    radar_alt = _get_valid_reading(
        [state.get('radar_alt_1_ft'), state.get('radar_alt_2_ft'), state.get('radar_alt_3_ft')],
        [state.get('radar_alt_1_valid'), state.get('radar_alt_2_valid'), state.get('radar_alt_3_valid')]
    )

    ias_kt = _get_valid_reading(
        [state.get('ias_1_kt'), state.get('ias_2_kt'), state.get('ias_3_kt')],
        [state.get('ias_1_valid'), state.get('ias_2_valid'), state.get('ias_3_valid')]
    )

    mach = _get_valid_reading(
        [state.get('mach_1'), state.get('mach_2'), state.get('mach_3')],
        [True, True, True]  # mach doesn't have valid flags
    )

    vs_fpm = state.get('vertical_speed_fpm', 0)

    heading_deg = _get_valid_reading(
        [state.get('irs_1_heading_deg'), state.get('irs_2_heading_deg'), state.get('irs_3_heading_deg')],
        [state.get('irs_1_valid'), state.get('irs_2_valid'), state.get('irs_3_valid')]
    )

    pitch_deg = _get_valid_reading(
        [state.get('irs_1_pitch_deg'), state.get('irs_2_pitch_deg'), state.get('irs_3_pitch_deg')],
        [state.get('irs_1_valid'), state.get('irs_2_valid'), state.get('irs_3_valid')]
    )

    roll_deg = _get_valid_reading(
        [state.get('irs_1_roll_deg'), state.get('irs_2_roll_deg'), state.get('irs_3_roll_deg')],
        [state.get('irs_1_valid'), state.get('irs_2_valid'), state.get('irs_3_valid')]
    )

    gyro_pitch_rate = _get_valid_reading(
        [state.get('gyro_1_pitch_rate_dps'), state.get('gyro_2_pitch_rate_dps'), state.get('gyro_3_pitch_rate_dps')],
        [state.get('gyro_1_valid'), state.get('gyro_2_valid'), state.get('gyro_3_valid')]
    ) or 0

    gyro_roll_rate = _get_valid_reading(
        [state.get('gyro_1_roll_rate_dps'), state.get('gyro_2_roll_rate_dps'), state.get('gyro_3_roll_rate_dps')],
        [state.get('gyro_1_valid'), state.get('gyro_2_valid'), state.get('gyro_3_valid')]
    ) or 0

    gyro_yaw_rate = _get_valid_reading(
        [state.get('gyro_1_yaw_rate_dps'), state.get('gyro_2_yaw_rate_dps'), state.get('gyro_3_yaw_rate_dps')],
        [state.get('gyro_1_valid'), state.get('gyro_2_valid'), state.get('gyro_3_valid')]
    ) or 0

    # Target states
    target_alt = state.get('target_alt_ft', alt_ft)
    target_hdg = state.get('target_hdg_deg', heading_deg)
    target_ias = state.get('target_ias_kt', ias_kt)
    target_vs = state.get('target_vs_fpm', 0)

    # Engage status
    ap_engaged = state.get('ap_engaged', True)
    at_engaged = state.get('at_engaged', True)

    # Flight phase detection
    on_ground = any([state.get('weight_on_wheels_left'),
                     state.get('weight_on_wheels_nose'),
                     state.get('weight_on_wheels_right')])

    altitude_delta = target_alt - alt_ft

    # Determine flight phase
    if on_ground:
        _state_history['landing_phase'] = 'landing'
    elif altitude_delta > 500 and ias_kt < 200:
        _state_history['landing_phase'] = 'climb'
    elif altitude_delta < -500 and ias_kt < 220:
        _state_history['landing_phase'] = 'descent'
    elif radar_alt and radar_alt < 2000 and altitude_delta < 0:
        _state_history['landing_phase'] = 'approach'
    else:
        _state_history['landing_phase'] = 'cruise'

    # ==================== ALTITUDE CONTROL ====================
    alt_error = target_alt - alt_ft

    # Altitude PID
    kp_alt, ki_alt, kd_alt = 0.0005, 0.00005, 0.0002

    _state_history['altitude_integral'] += alt_error * dt
    _state_history['altitude_integral'] = _clamp(_state_history['altitude_integral'], -5000, 5000)

    alt_deriv = (alt_error - _state_history.get('prev_alt_error', 0)) / dt if dt > 0 else 0
    _state_history['prev_alt_error'] = alt_error

    pitch_command = (kp_alt * alt_error +
                    ki_alt * _state_history['altitude_integral'] +
                    kd_alt * alt_deriv)

    pitch_command = _clamp(pitch_command, -10, 15)  # degrees of pitch command

    # ==================== SPEED CONTROL ====================
    speed_error = target_ias - ias_kt

    # Speed PID for throttle
    kp_spd, ki_spd, kd_spd = 0.01, 0.001, 0.005

    _state_history['speed_integral'] += speed_error * dt
    _state_history['speed_integral'] = _clamp(_state_history['speed_integral'], -500, 500)

    spd_deriv = (speed_error - _state_history.get('prev_spd_error', 0)) / dt if dt > 0 else 0
    _state_history['prev_spd_error'] = speed_error

    throttle_command = (kp_spd * speed_error +
                       ki_spd * _state_history['speed_integral'] +
                       kd_spd * spd_deriv)

    throttle_1 = _clamp(0.5 + throttle_command, 0.0, 1.0)
    throttle_2 = _clamp(0.5 + throttle_command, 0.0, 1.0)

    # Handle overspeed (Vne ~ 365 KIAS / Mach 0.82)
    if ias_kt > 360 or (mach and mach > 0.80):
        throttle_1 = 0.0
        throttle_2 = 0.0
        pitch_command = _clamp(pitch_command + 5, -10, 15)  # nose up to slow down

    # Handle stall protection (typical stall ~ 110-130 KIAS clean)
    stall_speed = 130 if not state.get('flap_pos_left_deg', 0) else 110
    if ias_kt < stall_speed + 10:
        pitch_command = _clamp(pitch_command - 5, -10, 15)  # nose down
        throttle_1 = 0.8
        throttle_2 = 0.8

    # ==================== HEADING CONTROL ====================
    # Normalize heading difference
    hdg_error = target_hdg - heading_deg
    while hdg_error > 180:
        hdg_error -= 360
    while hdg_error < -180:
        hdg_error += 360

    # Heading PID
    kp_hdg, ki_hdg, kd_hdg = 0.05, 0.005, 0.02

    _state_history['heading_integral'] += hdg_error * dt
    _state_history['heading_integral'] = _clamp(_state_history['heading_integral'], -30, 30)

    hdg_deriv = (hdg_error - _state_history.get('prev_hdg_error', 0)) / dt if dt > 0 else 0
    _state_history['prev_hdg_error'] = hdg_error

    roll_command = (kp_hdg * hdg_error +
                   ki_hdg * _state_history['heading_integral'] +
                   kd_hdg * hdg_deriv)

    roll_command = _clamp(roll_command, -25, 25)  # max 25 degrees bank

    # Roll control (aileron)
    roll_error = roll_command - roll_deg
    aileron = _clamp(0.05 * roll_error - 0.02 * gyro_roll_rate, -1.0, 1.0)

    # Rudder damping and coordination
    rudder = _clamp(-0.01 * gyro_yaw_rate - 0.005 * roll_deg, -1.0, 1.0)

    # ==================== PITCH/ELEVATOR CONTROL ====================
    pitch_error = pitch_command - pitch_deg
    elevator = _clamp(0.05 * pitch_error - 0.02 * gyro_pitch_rate, -1.0, 1.0)

    # Elevator trim adjustment for load alleviation
    trim_rate = 0.05 * pitch_error * dt
    _state_history['trim_state'] += trim_rate
    _state_history['trim_state'] = _clamp(_state_history['trim_state'], -1.0, 1.0)
    elevator_trim = _state_history['trim_state']

    # ==================== FLAPS & GEAR MANAGEMENT ====================
    flaps = state.get('flap_pos_left_deg', 0)
    gear = 1 if state.get('gear_handle_down', False) else 0

    phase = _state_history['landing_phase']

    if phase == 'climb' and ias_kt > 200:
        flaps = 0  # retract flaps
        gear = 0   # retract gear
    elif phase == 'descent':
        if ias_kt > 220:
            flaps = 5
        else:
            flaps = 15
        gear = 0  # gear up during descent
    elif phase == 'approach':
        flaps = 25  # approach flaps
        if radar_alt and radar_alt < 1500:
            gear = 1  # deploy gear
    elif phase == 'landing':
        flaps = 40  # full flaps
        gear = 1

    # ==================== SPEED BRAKE ====================
    speedbrake = 0.0
    if ias_kt > target_ias + 30 and phase in ['descent', 'approach']:
        speedbrake = _clamp((ias_kt - target_ias) / 100, 0.0, 1.0)

    # ==================== OUTPUT ====================
    control = {
        'elevator': elevator,
        'aileron': aileron,
        'rudder': rudder,
        'throttle_1': throttle_1,
        'throttle_2': throttle_2,
        'elevator_trim': elevator_trim,
        'flaps': flaps,
        'speedbrake': speedbrake,
        'gear': gear,
    }

    return control
