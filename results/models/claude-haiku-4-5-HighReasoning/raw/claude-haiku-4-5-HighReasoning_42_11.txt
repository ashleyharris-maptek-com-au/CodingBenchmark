OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0.0
%float_0p5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_0p02 = OpConstant %float 0.02
%float_0p333 = OpConstant %float 0.333333
%float_0p666 = OpConstant %float 0.666667

%gl_PerVertex_in = OpTypeStruct %v4float
%arr4_pv = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr4_pv
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr4_v3 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr4_v3
%ptr_in_v3 = OpTypePointer Input %v3float
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
%normal_in = OpVariable %ptr_in_arr_v3 Input
%color_in = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%func_type = OpTypeFunction %void
%main = OpFunction %void None %func_type
%bb = OpLabel

; Load the 4 adjacent vertices (InputLinesAdjacency)
%p0_addr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_addr
%p1_addr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_addr
%p2_addr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_addr
%p3_addr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_addr

; Load colors
%c1_addr = OpAccessChain %ptr_in_v3 %color_in %int_1
%c1 = OpLoad %v3float %c1_addr
%c2_addr = OpAccessChain %ptr_in_v3 %color_in %int_2
%c2 = OpLoad %v3float %c2_addr

; Extract xyz components
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2
%p3_xyz = OpVectorShuffle %v3float %p3 %p3 0 1 2

; Segment vector: seg = p2 - p1
%seg = OpFSub %v3float %p2_xyz %p1_xyz

; Tangent for smooth perpendiculars
; t_start = normalize(p2 - p0)
%t_s_raw = OpFSub %v3float %p2_xyz %p0_xyz
%t_s = OpExtInst %v3float %glsl Normalize %t_s_raw

; t_end = normalize(p3 - p1)
%t_e_raw = OpFSub %v3float %p3_xyz %p1_xyz
%t_e = OpExtInst %v3float %glsl Normalize %t_e_raw

; Up vector
%up = OpConstantComposite %v3float %float_0 %float_1 %float_0

; Perpendicular at start
%perp_s_raw = OpExtInst %v3float %glsl Cross %t_s %up
%len_s = OpExtInst %float %glsl Length %perp_s_raw
%cond_s = OpFOrdEqual %bool %len_s %float_0
%alt_perp = OpConstantComposite %v3float %float_0 %float_0 %float_1
%perp_s_sel = OpSelect %v3float %cond_s %alt_perp %perp_s_raw
%perp_s_n = OpExtInst %v3float %glsl Normalize %perp_s_sel
%perp_s = OpVectorTimesScalar %v3float %perp_s_n %float_0p02

; Perpendicular at end
%perp_e_raw = OpExtInst %v3float %glsl Cross %t_e %up
%len_e = OpExtInst %float %glsl Length %perp_e_raw
%cond_e = OpFOrdEqual %bool %len_e %float_0
%perp_e_sel = OpSelect %v3float %cond_e %alt_perp %perp_e_raw
%perp_e_n = OpExtInst %v3float %glsl Normalize %perp_e_sel
%perp_e = OpVectorTimesScalar %v3float %perp_e_n %float_0p02

; === PART 1: t in [0, 1/3] ===

; Center points for part 1
%p1a = %p1_xyz
%seg_1_3 = OpVectorTimesScalar %v3float %seg %float_0p333
%p1b = OpFAdd %v3float %p1a %seg_1_3

; Perpendicular at t=1/3
%perp_mid_sum = OpFAdd %v3float %perp_s %perp_e
%perp_1_3 = OpVectorTimesScalar %v3float %perp_mid_sum %float_0p5

; Color interpolation
%c_mid_sum = OpFAdd %v3float %c1 %c2
%c_1_3 = OpVectorTimesScalar %v3float %c_mid_sum %float_0p5

; Helper function to construct v4 from v3
; v0: p1a - perp_s
%v0_p = OpFSub %v3float %p1a %perp_s
%v0_x = OpCompositeExtract %float %v0_p 0
%v0_y = OpCompositeExtract %float %v0_p 1
%v0_z = OpCompositeExtract %float %v0_p 2
%v0_4 = OpCompositeConstruct %v4float %v0_x %v0_y %v0_z %float_1
%gl_pos_addr_0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_pos_addr_0 %v0_4
OpStore %outColor %c1
OpEmitVertex

; v1: p1a + perp_s
%v1_p = OpFAdd %v3float %p1a %perp_s
%v1_x = OpCompositeExtract %float %v1_p 0
%v1_y = OpCompositeExtract %float %v1_p 1
%v1_z = OpCompositeExtract %float %v1_p 2
%v1_4 = OpCompositeConstruct %v4float %v1_x %v1_y %v1_z %float_1
OpStore %gl_pos_addr_0 %v1_4
OpStore %outColor %c1
OpEmitVertex

; v2: p1b - perp_1_3
%v2_p = OpFSub %v3float %p1b %perp_1_3
%v2_x = OpCompositeExtract %float %v2_p 0
%v2_y = OpCompositeExtract %float %v2_p 1
%v2_z = OpCompositeExtract %float %v2_p 2
%v2_4 = OpCompositeConstruct %v4float %v2_x %v2_y %v2_z %float_1
OpStore %gl_pos_addr_0 %v2_4
OpStore %outColor %c_1_3
OpEmitVertex

; v3: p1b + perp_1_3
%v3_p = OpFAdd %v3float %p1b %perp_1_3
%v3_x = OpCompositeExtract %float %v3_p 0
%v3_y = OpCompositeExtract %float %v3_p 1
%v3_z = OpCompositeExtract %float %v3_p 2
%v3_4 = OpCompositeConstruct %v4float %v3_x %v3_y %v3_z %float_1
OpStore %gl_pos_addr_0 %v3_4
OpStore %outColor %c_1_3
OpEmitVertex

OpEndPrimitive

; === PART 3: t in [2/3, 1] ===

; Center points for part 3
%seg_2_3 = OpVectorTimesScalar %v3float %seg %float_0p666
%p3a = OpFAdd %v3float %p1a %seg_2_3
%p3b = OpFAdd %v3float %p1a %seg

; Perpendicular at t=2/3
%perp_2_3 = %perp_1_3

; Color for part 3
%c_2_3 = %c_1_3

; v4: p3a - perp_2_3
%v4_p = OpFSub %v3float %p3a %perp_2_3
%v4_x = OpCompositeExtract %float %v4_p 0
%v4_y = OpCompositeExtract %float %v4_p 1
%v4_z = OpCompositeExtract %float %v4_p 2
%v4_4 = OpCompositeConstruct %v4float %v4_x %v4_y %v4_z %float_1
OpStore %gl_pos_addr_0 %v4_4
OpStore %outColor %c_2_3
OpEmitVertex

; v5: p3a + perp_2_3
%v5_p = OpFAdd %v3float %p3a %perp_2_3
%v5_x = OpCompositeExtract %float %v5_p 0
%v5_y = OpCompositeExtract %float %v5_p 1
%v5_z = OpCompositeExtract %float %v5_p 2
%v5_4 = OpCompositeConstruct %v4float %v5_x %v5_y %v5_z %float_1
OpStore %gl_pos_addr_0 %v5_4
OpStore %outColor %c_2_3
OpEmitVertex

; v6: p3b - perp_e
%v6_p = OpFSub %v3float %p3b %perp_e
%v6_x = OpCompositeExtract %float %v6_p 0
%v6_y = OpCompositeExtract %float %v6_p 1
%v6_z = OpCompositeExtract %float %v6_p 2
%v6_4 = OpCompositeConstruct %v4float %v6_x %v6_y %v6_z %float_1
OpStore %gl_pos_addr_0 %v6_4
OpStore %outColor %c2
OpEmitVertex

; v7: p3b + perp_e
%v7_p = OpFAdd %v3float %p3b %perp_e
%v7_x = OpCompositeExtract %float %v7_p 0
%v7_y = OpCompositeExtract %float %v7_p 1
%v7_z = OpCompositeExtract %float %v7_p 2
%v7_4 = OpCompositeConstruct %v4float %v7_x %v7_y %v7_z %float_1
OpStore %gl_pos_addr_0 %v7_4
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
