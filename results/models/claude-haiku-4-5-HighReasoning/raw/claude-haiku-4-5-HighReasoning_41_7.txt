OpCapability Shader
OpExtInstImport %glsl "GLSL.std.450"
OpMemoryModel Logical GLSL450

OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_color
OpExecutionMode %main OriginUpperLeft

; Input decorations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3

OpDecorate %out_color Location 0

OpDecorate %UBO Block

OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Type definitions
%void = OpTypeVoid
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4

; Variable declarations
%ubo_block = OpVariable %ptr_Uniform_UBO Uniform

%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2 Input
%in_tangent = OpVariable %ptr_Input_vec3 Input
%in_color = OpVariable %ptr_Input_vec3 Input
%out_color = OpVariable %ptr_Output_vec4 Output

; Constants
%c_0 = OpConstant %uint 0
%c_3 = OpConstant %uint 3
%c_4 = OpConstant %uint 4

%f_0_0 = OpConstant %float 0.0
%f_0_2 = OpConstant %float 0.2
%f_0_3 = OpConstant %float 0.3
%f_0_5 = OpConstant %float 0.5
%f_0_8 = OpConstant %float 0.8
%f_1_0 = OpConstant %float 1.0
%f_3_0 = OpConstant %float 3.0

; Function type
%func_type = OpTypeFunction %void

%main = OpFunction %void None %func_type
%label = OpLabel

; Load input values
%worldPos = OpLoad %vec3 %in_worldPos
%normal_raw = OpLoad %vec3 %in_normal

; Normalize normal
%normal = OpExtInst %vec3 %glsl Normalize %normal_raw

; Load light position from UBO (index 3)
%lightPos_ptr = OpAccessChain %ptr_Uniform_vec4 %ubo_block %c_3
%lightPos_v4 = OpLoad %vec4 %lightPos_ptr
%lightPos = OpVectorShuffle %vec3 %lightPos_v4 %lightPos_v4 0 1 2

; Load camera position from UBO (index 4)
%cameraPos_ptr = OpAccessChain %ptr_Uniform_vec4 %ubo_block %c_4
%cameraPos_v4 = OpLoad %vec4 %cameraPos_ptr
%cameraPos = OpVectorShuffle %vec3 %cameraPos_v4 %cameraPos_v4 0 1 2

; Calculate view direction: normalize(cameraPos - worldPos)
%to_camera = OpFSub %vec3 %cameraPos %worldPos
%viewDir = OpExtInst %vec3 %glsl Normalize %to_camera

; Calculate light direction: normalize(lightPos - worldPos)
%to_light = OpFSub %vec3 %lightPos %worldPos
%lightDir = OpExtInst %vec3 %glsl Normalize %to_light

; Calculate diffuse: max(dot(normal, lightDir), 0.0)
%dotNL = OpDot %float %normal %lightDir
%diffuse = OpExtInst %float %glsl FMax %dotNL %f_0_0

; Calculate Fresnel/Rim: pow(1.0 - max(dot(normal, viewDir), 0.0), 3.0)
%dotNV = OpDot %float %normal %viewDir
%dotNV_clamped = OpExtInst %float %glsl FMax %dotNV %f_0_0
%one_minus_dotNV = OpFSub %float %f_1_0 %dotNV_clamped
%rim = OpExtInst %float %glsl Pow %one_minus_dotNV %f_3_0

; Calculate modulation factor: 0.3 + 0.5 * diffuse
%diffuse_scaled = OpFMul %float %diffuse %f_0_5
%modulation = OpFAdd %float %f_0_3 %diffuse_scaled

; Base color: (0.2, 0.3, 0.8)
%base_color = OpCompositeConstruct %vec3 %f_0_2 %f_0_3 %f_0_8

; Base color modulated: base_color * modulation
%base_modulated = OpVectorTimesScalar %vec3 %base_color %modulation

; Rim tint: (1.0, 0.8, 0.5)
%rim_tint = OpCompositeConstruct %vec3 %f_1_0 %f_0_8 %f_0_5

; Rim contribution: rim_tint * rim
%rim_color = OpVectorTimesScalar %vec3 %rim_tint %rim

; Final color: base_modulated + rim_color
%final_rgb = OpFAdd %vec3 %base_modulated %rim_color

; Output color with alpha 1.0
%final_color = OpCompositeConstruct %vec4 %final_rgb %f_1_0

; Store result
OpStore %out_color %final_color

OpReturn
OpFunctionEnd
