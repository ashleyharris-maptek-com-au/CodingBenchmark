use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read input
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let n: usize = parts[0].parse().unwrap();
    let m: usize = parts[1].parse().unwrap();

    // Build adjacency list
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let u: usize = parts[0].parse().unwrap();
        let v: usize = parts[1].parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
    }

    // Find bisection
    let (cut, partition) = find_bisection(&adj, n);

    // Output
    println!("{}", cut);
    println!("{}", partition.iter().map(|x| x.to_string()).collect::<Vec<_>>().join(" "));
}

fn find_bisection(adj: &[Vec<usize>], n: usize) -> (usize, Vec<usize>) {
    let mut best_cut = usize::MAX;
    let mut best_partition = Vec::new();

    let num_trials = std::cmp::min(100, n * 2);

    for seed in 0..num_trials {
        let mut partition = random_partition(n, seed);
        improve(&mut partition, adj, n);

        let cut = count_cut(&partition, adj);
        if cut < best_cut {
            best_cut = cut;
            best_partition = partition;
        }
    }

    best_partition.sort();
    (best_cut, best_partition)
}

fn random_partition(n: usize, seed: usize) -> Vec<usize> {
    let mut partition = Vec::new();
    for i in 0..n {
        let hash = ((i as u64).wrapping_mul(2654435761).wrapping_add(seed as u64)) as usize;
        if hash.wrapping_mul(2246822519) % 2 == 0 && partition.len() < n / 2 {
            partition.push(i);
        }
    }
    while partition.len() < n / 2 {
        for i in 0..n {
            if !partition.contains(&i) {
                partition.push(i);
                if partition.len() == n / 2 {
                    break;
                }
            }
        }
    }
    partition
}

fn count_cut(partition: &[usize], adj: &[Vec<usize>]) -> usize {
    let set: HashSet<usize> = partition.iter().cloned().collect();
    let mut cut = 0;
    for &u in partition {
        for &v in &adj[u] {
            if !set.contains(&v) {
                cut += 1;
            }
        }
    }
    cut
}

fn improve(partition: &mut Vec<usize>, adj: &[Vec<usize>], n: usize) {
    loop {
        let set: HashSet<usize> = partition.iter().cloned().collect();
        let other: Vec<usize> = (0..n).filter(|x| !set.contains(x)).collect();

        let mut found_improvement = false;

        for i in 0..partition.len() {
            for &b in &other {
                let a = partition[i];

                // Count neighbors in each partition
                let mut a_to_a = 0;
                let mut a_to_b = 0;
                for &neighbor in &adj[a] {
                    if set.contains(&neighbor) {
                        a_to_a += 1;
                    } else {
                        a_to_b += 1;
                    }
                }

                let mut b_to_b = 0;
                let mut b_to_a = 0;
                for &neighbor in &adj[b] {
                    if set.contains(&neighbor) {
                        b_to_a += 1;
                    } else {
                        b_to_b += 1;
                    }
                }

                // Calculate improvement from swapping a and b
                let improvement = (a_to_b + b_to_a) as i32 - (a_to_a + b_to_b) as i32;

                if improvement > 0 {
                    partition[i] = b;
                    found_improvement = true;
                    break;
                }
            }
            if found_improvement {
                break;
            }
        }

        if !found_improvement {
            break;
        }
    }
}
