[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal : TEXCOORD1;
    [[vk::location(2)]] float2 uv : TEXCOORD2;
    [[vk::location(3)]] float3 tangent : TEXCOORD3;
    [[vk::location(4)]] float3 color : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    // Calculate stripe pattern based on uv.y
    float stripe = sin(input.uv.y * 3.14159 * 16.0);

    // Choose color: red for stripe > 0, blue for stripe <= 0
    float3 stripeColor = stripe > 0.0 ? float3(0.9, 0.1, 0.1) : float3(0.1, 0.1, 0.9);

    // Calculate lighting
    float3 normal = normalize(input.normal);
    float3 lightDir = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(0.0, dot(normal, lightDir));

    // Apply Lambertian shading: (0.4 + 0.6*diffuse)
    float3 finalColor = stripeColor * (0.4 + 0.6 * diffuse);

    return float4(finalColor, 1.0);
}
