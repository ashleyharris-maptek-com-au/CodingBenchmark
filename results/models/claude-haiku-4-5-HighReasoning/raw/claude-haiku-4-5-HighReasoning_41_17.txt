; Bound: 600

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_color
OpExecutionMode %main OriginUpperLeft

%glsl450 = OpExtInstImport "GLSL.std.450"

; Decorations
OpDecorate %UBO Block
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %out_color Location 0

OpMemberDecorate %UBO 0 Offset 0 ColMajor MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64 ColMajor MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128 ColMajor MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Type declarations
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Input_v2 = OpTypePointer Input %v2float
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Function_float = OpTypePointer Function %float
%ptr_Function_v2 = OpTypePointer Function %v2float
%ptr_Function_v3 = OpTypePointer Function %v3float
%ptr_Function_v4 = OpTypePointer Function %v4float
%ptr_Function_bool = OpTypePointer Function %bool

%void_func = OpTypeFunction %void

; Variables
%in_worldPos = OpVariable %ptr_Input_v3 Input
%in_normal = OpVariable %ptr_Input_v3 Input
%in_uv = OpVariable %ptr_Input_v2 Input
%in_tangent = OpVariable %ptr_Input_v3 Input
%in_color = OpVariable %ptr_Input_v3 Input
%out_color = OpVariable %ptr_Output_v4 Output
%uboVar = OpVariable %ptr_Uniform_UBO Uniform

; Constants
%f_0 = OpConstant %float 0.0
%f_0_05 = OpConstant %float 0.05
%f_0_1 = OpConstant %float 0.1
%f_0_2 = OpConstant %float 0.2
%f_0_4 = OpConstant %float 0.4
%f_0_5 = OpConstant %float 0.5
%f_0_6 = OpConstant %float 0.6
%f_0_7 = OpConstant %float 0.7
%f_1 = OpConstant %float 1.0
%f_2 = OpConstant %float 2.0
%f_8 = OpConstant %float 8.0
%f_16 = OpConstant %float 16.0
%i_0 = OpConstant %int 0
%i_2 = OpConstant %int 2

%main = OpFunction %void None %void_func
%bb_entry = OpLabel

; Load inputs
%ld_uv = OpLoad %v2float %in_uv
%ld_normal = OpLoad %v3float %in_normal

; Extract UV components
%ex_uvX = OpCompositeExtract %float %ld_uv 0
%ex_uvY = OpCompositeExtract %float %ld_uv 1

; Scale UV: x by 8, y by 16
%scaledX = OpFMul %float %ex_uvX %f_8
%scaledY = OpFMul %float %ex_uvY %f_16

; Get row index (integer part of scaledY)
%rowIdx_i = OpConvertFToS %int %scaledY
%rowIdx_f = OpConvertSToF %float %rowIdx_i
%rowFrac = OpFSub %float %scaledY %rowIdx_f

; Check if row is odd (determine stagger)
%rowMod2 = OpSMod %int %rowIdx_i %i_2
%isOddRow = OpINotEqual %bool %rowMod2 %i_0
%stagger = OpSelect %float %isOddRow %f_0_5 %f_0

; Apply stagger to X coordinate
%staggedX = OpFAdd %float %scaledX %stagger

; Get position within brick (modulo 1.0)
%brickX_raw = OpFRem %float %staggedX %f_1
%brickY_val = %rowFrac

; Detect mortar: check X edges
%mortarX_left = OpFOrdLessThan %bool %brickX_raw %f_0_05
%mortarX_right_threshold = OpFSub %float %f_1 %f_0_05
%mortarX_right = OpFOrdGreaterThan %bool %brickX_raw %mortarX_right_threshold
%mortarX = OpLogicalOr %bool %mortarX_left %mortarX_right

; Detect mortar: check Y edges
%mortarY_top = OpFOrdLessThan %bool %brickY_val %f_0_1
%mortarY_bottom_threshold = OpFSub %float %f_1 %f_0_1
%mortarY_bottom = OpFOrdGreaterThan %bool %brickY_val %mortarY_bottom_threshold
%mortarY = OpLogicalOr %bool %mortarY_top %mortarY_bottom

; Combine mortar detection
%isMortar = OpLogicalOr %bool %mortarX %mortarY

; Set colors
%mortarCol = OpCompositeConstruct %v3float %f_0_7 %f_0_7 %f_0_7
%brickCol = OpCompositeConstruct %v3float %f_0_7 %f_0_2 %f_0_1

; Select base color
%baseCol = OpSelect %v3float %isMortar %mortarCol %brickCol

; Compute diffuse lighting
; Normalize normal vector
%norm_normal = OpExtInst %v3float %glsl450 Normalize %ld_normal

; Create light direction (pointing towards normalized direction)
%light_dir_unnorm = OpCompositeConstruct %v3float %f_0_5 %f_0_5 %f_0_5
%light_dir = OpExtInst %v3float %glsl450 Normalize %light_dir_unnorm

; Compute dot product for diffuse
%dot_prod = OpDot %float %norm_normal %light_dir

; Clamp to [0, 1]
%diff_clamped = OpExtInst %float %glsl450 FMax %dot_prod %f_0

; Apply shading: 0.4 + 0.6 * diffuse
%shading_term = OpFMul %float %f_0_6 %diff_clamped
%shading = OpFAdd %float %f_0_4 %shading_term

; Apply shading to color
%final_col = OpVectorTimesScalar %v3float %baseCol %shading

; Extract components and construct output
%ex_r = OpCompositeExtract %float %final_col 0
%ex_g = OpCompositeExtract %float %final_col 1
%ex_b = OpCompositeExtract %float %final_col 2
%out_val = OpCompositeConstruct %v4float %ex_r %ex_g %ex_b %f_1

; Store output
OpStore %out_color %out_val

OpReturn
OpFunctionEnd
