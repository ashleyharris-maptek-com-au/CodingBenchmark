#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <limits>

using namespace std;

const double EPS = 1e-10;
const double PI = 3.14159265358979323846;

struct Vec3 {
    double x, y, z;

    Vec3(double x = 0, double y = 0, double z = 0) : x(x), y(y), z(z) {}

    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double t) const { return Vec3(x * t, y * t, z * t); }
    Vec3 operator/(double t) const {
        if (t == 0) return Vec3(0, 0, 0);
        return Vec3(x / t, y / t, z / t);
    }

    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const {
        return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x);
    }
    double length() const { return sqrt(x * x + y * y + z * z); }

    Vec3 normalize() const {
        double len = length();
        return len > EPS ? *this / len : Vec3(0, 0, 0);
    }
};

struct Triangle {
    int a, b, c;
    Triangle(int a = 0, int b = 0, int c = 0) : a(a), b(b), c(c) {}
};

struct Mesh {
    vector<Vec3> vertices;
    vector<Triangle> faces;
};

// Compute winding number using solid angle calculation
int computeWindingNumber(const Vec3& point, const Mesh& mesh) {
    double angle_sum = 0.0;

    for (const auto& face : mesh.faces) {
        const Vec3& v0 = mesh.vertices[face.a];
        const Vec3& v1 = mesh.vertices[face.b];
        const Vec3& v2 = mesh.vertices[face.c];

        // Vectors from point to vertices
        Vec3 a = v0 - point;
        Vec3 b = v1 - point;
        Vec3 c = v2 - point;

        double alen = a.length();
        double blen = b.length();
        double clen = c.length();

        // Skip degenerate triangles or if point is too close to vertex
        if (alen < EPS || blen < EPS || clen < EPS) continue;

        // Normalize vectors
        a = a / alen;
        b = b / blen;
        c = c / clen;

        // Compute signed solid angle
        double numerator = a.dot(b.cross(c));
        double denominator = 1.0 + a.dot(b) + b.dot(c) + c.dot(a);

        if (abs(denominator) > EPS) {
            angle_sum += atan2(numerator, denominator);
        }
    }

    // Winding number is the total solid angle divided by 2*pi
    int winding = (int)round(angle_sum / PI);
    return winding;
}

bool isInsideMesh(const Vec3& point, const Mesh& mesh) {
    // Point is inside if winding number is non-zero
    return computeWindingNumber(point, mesh) != 0;
}

Mesh computeUnion(const Mesh& meshA, const Mesh& meshB) {
    Mesh result;

    // Process mesh A: keep vertices and faces that are outside mesh B
    vector<int> vertexMapA(meshA.vertices.size(), -1);
    for (size_t i = 0; i < meshA.vertices.size(); ++i) {
        if (!isInsideMesh(meshA.vertices[i], meshB)) {
            vertexMapA[i] = (int)result.vertices.size();
            result.vertices.push_back(meshA.vertices[i]);
        }
    }

    for (const auto& face : meshA.faces) {
        // Keep face only if all three vertices are outside mesh B
        if (vertexMapA[face.a] >= 0 &&
            vertexMapA[face.b] >= 0 &&
            vertexMapA[face.c] >= 0) {
            result.faces.push_back(Triangle(
                vertexMapA[face.a],
                vertexMapA[face.b],
                vertexMapA[face.c]
            ));
        }
    }

    // Process mesh B: keep vertices and faces that are outside mesh A
    vector<int> vertexMapB(meshB.vertices.size(), -1);
    for (size_t i = 0; i < meshB.vertices.size(); ++i) {
        if (!isInsideMesh(meshB.vertices[i], meshA)) {
            vertexMapB[i] = (int)result.vertices.size();
            result.vertices.push_back(meshB.vertices[i]);
        }
    }

    for (const auto& face : meshB.faces) {
        // Keep face only if all three vertices are outside mesh A
        if (vertexMapB[face.a] >= 0 &&
            vertexMapB[face.b] >= 0 &&
            vertexMapB[face.c] >= 0) {
            result.faces.push_back(Triangle(
                vertexMapB[face.a],
                vertexMapB[face.b],
                vertexMapB[face.c]
            ));
        }
    }

    return result;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA, FA, VB, FB;
    cin >> VA >> FA >> VB >> FB;

    Mesh meshA, meshB;

    // Read mesh A vertices
    for (int i = 0; i < VA; ++i) {
        double x, y, z;
        cin >> x >> y >> z;
        meshA.vertices.push_back(Vec3(x, y, z));
    }

    // Read mesh A faces
    for (int i = 0; i < FA; ++i) {
        int a, b, c;
        cin >> a >> b >> c;
        meshA.faces.push_back(Triangle(a, b, c));
    }

    // Read mesh B vertices
    for (int i = 0; i < VB; ++i) {
        double x, y, z;
        cin >> x >> y >> z;
        meshB.vertices.push_back(Vec3(x, y, z));
    }

    // Read mesh B faces
    for (int i = 0; i < FB; ++i) {
        int a, b, c;
        cin >> a >> b >> c;
        meshB.faces.push_back(Triangle(a, b, c));
    }

    // Compute union
    Mesh result = computeUnion(meshA, meshB);

    // Output result
    cout << result.vertices.size() << " " << result.faces.size() << "\n";

    // Output vertices
    for (const auto& v : result.vertices) {
        cout << v.x << " " << v.y << " " << v.z << "\n";
    }

    // Output faces
    for (const auto& f : result.faces) {
        cout << f.a << " " << f.b << " " << f.c << "\n";
    }

    return 0;
}
