import math

def distance(p1, p2):
    """Calculate Euclidean distance between two points."""
    return math.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

def calculate_route_distance(route, depot, customer_positions):
    """Calculate total distance of a route (including return to depot)."""
    total = 0
    prev = depot
    for customer_idx in route:
        total += distance(prev, customer_positions[customer_idx])
        prev = customer_positions[customer_idx]
    total += distance(prev, depot)
    return total

def solve_cvrp(n, k, capacity, depot, customer_positions, demands):
    """Solve CVRP using nearest neighbor heuristic."""
    visited = [False] * n
    routes = []

    for _ in range(k):
        route = []
        current_pos = depot
        current_load = 0

        while True:
            best_customer = -1
            best_distance = float('inf')

            # Find nearest unvisited customer that fits capacity
            for customer_idx in range(n):
                if not visited[customer_idx]:
                    if demands[customer_idx] + current_load <= capacity:
                        dist = distance(current_pos, customer_positions[customer_idx])
                        if dist < best_distance:
                            best_distance = dist
                            best_customer = customer_idx

            if best_customer == -1:
                break

            route.append(best_customer)
            visited[best_customer] = True
            current_load += demands[best_customer]
            current_pos = customer_positions[best_customer]

        if route:
            routes.append(route)

        if all(visited):
            break

    total_distance = sum(calculate_route_distance(route, depot, customer_positions)
                        for route in routes)

    return total_distance, routes

# Read input
n, k, capacity = map(int, input().split())
depot_x, depot_y = map(float, input().split())
depot = (depot_x, depot_y)

customer_positions = []
demands = []
for _ in range(n):
    x, y, demand = map(float, input().split())
    customer_positions.append((x, y))
    demands.append(int(demand))

# Solve
total_distance, routes = solve_cvrp(n, k, capacity, depot, customer_positions, demands)

# Output
print(total_distance)
print(len(routes))
for route in routes:
    print(' '.join(map(str, route)))
