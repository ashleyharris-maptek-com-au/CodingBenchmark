OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos_in %normal_in %color_in %outColor

OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

OpDecorate %UBO DescriptorSet 0
OpDecorate %UBO Binding 0
OpDecorate %ubo_block Block
OpMemberDecorate %ubo_block 0 Offset 0
OpMemberDecorate %ubo_block 1 Offset 64
OpMemberDecorate %ubo_block 2 Offset 128
OpMemberDecorate %ubo_block 3 Offset 192
OpMemberDecorate %ubo_block 4 Offset 208
OpMemberDecorate %ubo_block 5 Offset 224

%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%m4float = OpTypeMatrix %v4float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0.0
%float_0_05 = OpConstant %float 0.05

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4float = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%ubo_block = OpTypeStruct %m4float %m4float %m4float %v4float %v4float %v4float
%ptr_uniform_block = OpTypePointer Uniform %ubo_block

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
%normal_in = OpVariable %ptr_in_arr_v3 Input
%color_in = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%UBO = OpVariable %ptr_uniform_block Uniform

%func_void = OpTypeFunction %void

%main = OpFunction %void None %func_void
%main_entry = OpLabel

%in_pos_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%in_pos = OpLoad %v4float %in_pos_ptr

%color_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
%color = OpLoad %v3float %color_ptr

%x = OpCompositeExtract %float %in_pos 0
%y = OpCompositeExtract %float %in_pos 1
%z = OpCompositeExtract %float %in_pos 2
%w = OpCompositeExtract %float %in_pos 3

%y_plus_h = OpFAdd %float %y %float_0_05
%top = OpCompositeConstruct %v4float %x %y_plus_h %z %w

%x_plus_h = OpFAdd %float %x %float_0_05
%right = OpCompositeConstruct %v4float %x_plus_h %y %z %w

%y_minus_h = OpFSub %float %y %float_0_05
%bottom = OpCompositeConstruct %v4float %x %y_minus_h %z %w

%x_minus_h = OpFSub %float %x %float_0_05
%left = OpCompositeConstruct %v4float %x_minus_h %y %z %w

%out_pos_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0

OpStore %out_pos_ptr %top
OpStore %outColor %color
OpEmitVertex

OpStore %out_pos_ptr %right
OpStore %outColor %color
OpEmitVertex

OpStore %out_pos_ptr %left
OpStore %outColor %color
OpEmitVertex

OpStore %out_pos_ptr %bottom
OpStore %outColor %color
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
