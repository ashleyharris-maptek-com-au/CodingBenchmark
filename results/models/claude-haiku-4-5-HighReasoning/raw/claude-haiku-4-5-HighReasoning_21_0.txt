#include <iostream>
#include <vector>
#include <queue>
#include <set>
#include <algorithm>
#include <cstring>

using namespace std;

struct Move {
    int axis;
    int direction;
};

struct State {
    vector<int> pos;
    vector<vector<int>> body;
    int foodEaten;
};

int D, maxTurns;
vector<int> bounds;
vector<vector<int>> food;
set<vector<int>> obstacles;
vector<vector<int>> snake;

bool isValid(const vector<int>& pos) {
    for (int i = 0; i < D; i++) {
        if (pos[i] < 0 || pos[i] >= bounds[i]) return false;
    }
    return obstacles.find(pos) == obstacles.end();
}

bool isSelfCollision(const vector<int>& pos, const vector<vector<int>>& body) {
    for (const auto& segment : body) {
        if (segment == pos) return true;
    }
    return false;
}

vector<int> addVectors(const vector<int>& a, const vector<int>& b) {
    vector<int> result(D);
    for (int i = 0; i < D; i++) result[i] = a[i] + b[i];
    return result;
}

pair<bool, vector<Move>> findPathBFS(vector<int> start, vector<int> target,
                                      const vector<vector<int>>& body, int foodIdx) {
    queue<pair<vector<int>, vector<Move>>> q;
    set<vector<int>> visited;

    q.push({start, {}});
    visited.insert(start);

    int maxSteps = 5000;
    int steps = 0;

    while (!q.empty() && steps < maxSteps) {
        steps++;
        auto [pos, moves] = q.front();
        q.pop();

        if (pos == target) {
            return {true, moves};
        }

        for (int axis = 0; axis < D; axis++) {
            for (int dir : {-1, 1}) {
                vector<int> newPos = pos;
                newPos[axis] += dir;

                if (!isValid(newPos)) continue;
                if (visited.count(newPos)) continue;
                if (isSelfCollision(newPos, body)) continue;

                vector<Move> newMoves = moves;
                newMoves.push_back({axis, dir});

                visited.insert(newPos);
                q.push({newPos, newMoves});

                if (visited.size() > 100000) {
                    return {false, {}};
                }
            }
        }
    }

    return {false, {}};
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> D >> maxTurns;

    bounds.resize(D);
    for (int i = 0; i < D; i++) {
        cin >> bounds[i];
    }

    int snakeLen;
    cin >> snakeLen;
    vector<int> head(D);
    for (int i = 0; i < D; i++) {
        cin >> head[i];
    }
    snake.push_back(head);
    for (int i = 1; i < snakeLen; i++) {
        snake.push_back(head);
    }

    int F;
    cin >> F;
    food.resize(F);
    for (int i = 0; i < F; i++) {
        food[i].resize(D);
        for (int j = 0; j < D; j++) {
            cin >> food[i][j];
        }
    }

    int O;
    cin >> O;
    for (int i = 0; i < O; i++) {
        vector<int> obs(D);
        for (int j = 0; j < D; j++) {
            cin >> obs[j];
        }
        obstacles.insert(obs);
    }

    int foodEaten = 0;
    int turn = 0;

    while (turn < maxTurns && foodEaten < F) {
        int targetFood = -1;
        int minDist = INT_MAX;

        for (int i = 0; i < F; i++) {
            int dist = 0;
            for (int d = 0; d < D; d++) {
                dist += abs(snake[0][d] - food[i][d]);
            }
            if (dist < minDist) {
                minDist = dist;
                targetFood = i;
            }
        }

        if (targetFood == -1) break;

        auto [found, moves] = findPathBFS(snake[0], food[targetFood], snake, targetFood);

        if (!found || moves.empty()) {
            vector<int> bestMove = {0, 1};
            int bestAxis = 0;
            int bestDir = 1;

            bool moved = false;
            for (int axis = 0; axis < D && !moved; axis++) {
                for (int dir : {-1, 1}) {
                    vector<int> newPos = snake[0];
                    newPos[axis] += dir;

                    if (isValid(newPos) && !isSelfCollision(newPos, snake)) {
                        bestAxis = axis;
                        bestDir = dir;
                        moved = true;
                        break;
                    }
                }
            }

            cout << bestAxis << " " << bestDir << "\n";

            vector<int> newHead = snake[0];
            newHead[bestAxis] += bestDir;

            if (newHead == food[targetFood]) {
                snake.insert(snake.begin(), newHead);
                foodEaten++;
                vector<vector<int>> newFood;
                for (int i = 0; i < F; i++) {
                    if (i != targetFood) {
                        newFood.push_back(food[i]);
                    }
                }
                food = newFood;
                F--;
            } else {
                snake.insert(snake.begin(), newHead);
                snake.pop_back();
            }

            turn++;
            continue;
        }

        for (const auto& move : moves) {
            vector<int> newHead = snake[0];
            newHead[move.axis] += move.direction;

            if (!isValid(newHead) || isSelfCollision(newHead, snake)) {
                break;
            }

            cout << move.axis << " " << move.direction << "\n";

            snake.insert(snake.begin(), newHead);

            if (newHead == food[targetFood]) {
                foodEaten++;
                vector<vector<int>> newFood;
                for (int i = 0; i < F; i++) {
                    if (i != targetFood) {
                        newFood.push_back(food[i]);
                    }
                }
                food = newFood;
                F--;
                break;
            } else {
                snake.pop_back();
            }

            turn++;
            if (turn >= maxTurns) break;
        }
    }

    return 0;
}
