#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main()
{
    // Normalize the interpolated normal
    vec3 n = normalize(normal);

    // Calculate light direction
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);

    // Lambertian diffuse factor
    float diffuse = max(0.0, dot(n, lightDir));
    float shadingFactor = 0.3 + 0.7 * diffuse;

    // Scale UV to 6x6 grid
    vec2 gridUV = uv * 6.0;

    // Get local position within the current cell
    vec2 localUV = fract(gridUV);

    // Distance from cell center (0.5, 0.5)
    vec2 cellCenter = vec2(0.5, 0.5);
    float distFromCenter = length(localUV - cellCenter);

    // Determine if inside dot (radius 0.3) or background
    vec3 dotColor = vec3(1.0, 0.8, 0.0);    // yellow
    vec3 bgColor = vec3(0.1, 0.1, 0.3);     // dark blue

    vec3 baseColor = distFromCenter < 0.3 ? dotColor : bgColor;

    // Apply shading and output
    vec3 finalColor = baseColor * shadingFactor;
    outColor = vec4(finalColor, 1.0);
}
