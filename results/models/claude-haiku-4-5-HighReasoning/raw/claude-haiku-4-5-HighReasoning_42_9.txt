OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %outColor Location 0
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2

          %void = OpTypeVoid
         %float = OpTypeFloat 32
        %v2float = OpTypeVector %float 2
        %v3float = OpTypeVector %float 3
        %v4float = OpTypeVector %float 4
           %int = OpTypeInt 32 1
          %uint = OpTypeInt 32 0

        %gl_PerVertex_in = OpTypeStruct %v4float
        %gl_PerVertex_out = OpTypeStruct %v4float

        %uint_4 = OpConstant %uint 4
        %int_0 = OpConstant %int 0
        %int_1 = OpConstant %int 1
        %int_2 = OpConstant %int 2
        %int_3 = OpConstant %int 3
        %float_025 = OpConstant %float 0.025

        %arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_4
        %arr_v3float = OpTypeArray %v3float %uint_4

        %ptr_Input_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
        %ptr_Input_arr_v3float = OpTypePointer Input %arr_v3float
        %ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
        %ptr_Output_v3float = OpTypePointer Output %v3float
        %ptr_Input_v4float = OpTypePointer Input %v4float
        %ptr_Output_v4float = OpTypePointer Output %v4float
        %ptr_Input_v3float = OpTypePointer Input %v3float

        %func_void_void = OpTypeFunction %void

        %gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in Input
        %worldPos = OpVariable %ptr_Input_arr_v3float Input
        %normal = OpVariable %ptr_Input_arr_v3float Input
        %color = OpVariable %ptr_Input_arr_v3float Input
        %gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
        %outColor = OpVariable %ptr_Output_v3float Output

        %main = OpFunction %void None %func_void_void
        %entry = OpLabel

        ; Load input positions
        %p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
        %pos0 = OpLoad %v4float %p0_ptr

        %p1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
        %pos1 = OpLoad %v4float %p1_ptr

        %p2_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
        %pos2 = OpLoad %v4float %p2_ptr

        %p3_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_3 %int_0
        %pos3 = OpLoad %v4float %p3_ptr

        ; Load colors
        %col1_ptr = OpAccessChain %ptr_Input_v3float %color %int_1
        %col1 = OpLoad %v3float %col1_ptr

        %col2_ptr = OpAccessChain %ptr_Input_v3float %color %int_2
        %col2 = OpLoad %v3float %col2_ptr

        ; Compute tangent at start: normalize(pos2.xyz - pos0.xyz)
        %diff0 = OpFSub %v4float %pos2 %pos0
        %t0_v3 = OpVectorShuffle %v3float %diff0 %diff0 0 1 2
        %t0 = OpExtInst %v3float %1 Normalize %t0_v3

        ; Compute tangent at end: normalize(pos3.xyz - pos1.xyz)
        %diff1 = OpFSub %v4float %pos3 %pos1
        %t1_v3 = OpVectorShuffle %v3float %diff1 %diff1 0 1 2
        %t1 = OpExtInst %v3float %1 Normalize %t1_v3

        ; Get 2D components for perpendiculars
        %t0_y = OpCompositeExtract %float %t0 1
        %t0_x = OpCompositeExtract %float %t0 0
        %perp0_x = OpFNegate %float %t0_y
        %perp0_2d = OpCompositeConstruct %v2float %perp0_x %t0_x

        %t1_y = OpCompositeExtract %float %t1 1
        %t1_x = OpCompositeExtract %float %t1 0
        %perp1_x = OpFNegate %float %t1_y
        %perp1_2d = OpCompositeConstruct %v2float %perp1_x %t1_x

        ; Apply thickness
        %thick_perp0 = OpVectorTimesScalar %v2float %perp0_2d %float_025
        %thick_perp1 = OpVectorTimesScalar %v2float %perp1_2d %float_025

        ; Get 2D components of positions
        %pos1_2d = OpVectorShuffle %v2float %pos1 %pos1 0 1
        %pos2_2d = OpVectorShuffle %v2float %pos2 %pos2 0 1

        ; Z and W components
        %pos1_z = OpCompositeExtract %float %pos1 2
        %pos1_w = OpCompositeExtract %float %pos1 3
        %pos2_z = OpCompositeExtract %float %pos2 2
        %pos2_w = OpCompositeExtract %float %pos2 3

        ; Emit vertices
        ; Vertex 1: pos1 + perp0 * thickness
        %v1_2d = OpFAdd %v2float %pos1_2d %thick_perp0
        %v1_x = OpCompositeExtract %float %v1_2d 0
        %v1_y = OpCompositeExtract %float %v1_2d 1
        %v1 = OpCompositeConstruct %v4float %v1_x %v1_y %pos1_z %pos1_w

        %out_pos = OpAccessChain %ptr_Output_v4float %gl_out %int_0
        OpStore %out_pos %v1
        OpStore %outColor %col1
        OpEmitVertex

        ; Vertex 2: pos1 - perp0 * thickness
        %neg_thick_perp0 = OpFNegate %v2float %thick_perp0
        %v2_2d = OpFAdd %v2float %pos1_2d %neg_thick_perp0
        %v2_x = OpCompositeExtract %float %v2_2d 0
        %v2_y = OpCompositeExtract %float %v2_2d 1
        %v2 = OpCompositeConstruct %v4float %v2_x %v2_y %pos1_z %pos1_w

        OpStore %out_pos %v2
        OpStore %outColor %col1
        OpEmitVertex

        ; Vertex 3: pos2 + perp1 * thickness
        %v3_2d = OpFAdd %v2float %pos2_2d %thick_perp1
        %v3_x = OpCompositeExtract %float %v3_2d 0
        %v3_y = OpCompositeExtract %float %v3_2d 1
        %v3 = OpCompositeConstruct %v4float %v3_x %v3_y %pos2_z %pos2_w

        OpStore %out_pos %v3
        OpStore %outColor %col2
        OpEmitVertex

        ; Vertex 4: pos2 - perp1 * thickness
        %neg_thick_perp1 = OpFNegate %v2float %thick_perp1
        %v4_2d = OpFAdd %v2float %pos2_2d %neg_thick_perp1
        %v4_x = OpCompositeExtract %float %v4_2d 0
        %v4_y = OpCompositeExtract %float %v4_2d 1
        %v4 = OpCompositeConstruct %v4float %v4_x %v4_y %pos2_z %pos2_w

        OpStore %out_pos %v4
        OpStore %outColor %col2
        OpEmitVertex

        OpEndPrimitive

        OpReturn
        OpFunctionEnd
