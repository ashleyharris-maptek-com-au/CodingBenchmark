declare const require: any;
const fs = require('fs');

interface Point {
  x: number;
  y: number;
}

interface Package {
  id: number;
  x: number;
  y: number;
}

interface Input {
  depot: [number, number];
  truckCount: number;
  packages: Package[];
}

interface RouteOutput {
  routes: number[][];
}

function distance(p1: Point, p2: Point): number {
  const dx = p1.x - p2.x;
  const dy = p1.y - p2.y;
  return Math.sqrt(dx * dx + dy * dy);
}

function routeCost(route: Package[], depot: Point): number {
  if (route.length === 0) return 0;
  let cost = distance(depot, route[0]);
  for (let i = 0; i < route.length - 1; i++) {
    cost += distance(route[i], route[i + 1]);
  }
  cost += distance(route[route.length - 1], depot);
  return cost;
}

function greedyInsert(packages: Package[], depot: Point, maxTrucks: number): Package[][] {
  const routes: Package[][] = Array.from({ length: maxTrucks }, () => []);
  const assigned = new Set<number>();

  // Sort packages by distance from depot (closer ones first)
  const sorted = [...packages].sort((a, b) =>
    distance(depot, a) - distance(depot, b)
  );

  for (const pkg of sorted) {
    let bestRoute = 0;
    let bestCost = Infinity;
    let bestPos = 0;

    // Find best route and position to insert package
    for (let routeIdx = 0; routeIdx < routes.length; routeIdx++) {
      const route = routes[routeIdx];

      if (route.length === 0) {
        const cost = 2 * distance(depot, pkg);
        if (cost < bestCost) {
          bestCost = cost;
          bestRoute = routeIdx;
          bestPos = 0;
        }
      } else {
        const oldCost = routeCost(route, depot);

        // Try inserting at each position
        for (let pos = 0; pos <= route.length; pos++) {
          const newRoute = [...route.slice(0, pos), pkg, ...route.slice(pos)];
          const newCost = routeCost(newRoute, depot);
          const increase = newCost - oldCost;

          if (increase < bestCost) {
            bestCost = increase;
            bestRoute = routeIdx;
            bestPos = pos;
          }
        }
      }
    }

    routes[bestRoute].splice(bestPos, 0, pkg);
    assigned.add(pkg.id);
  }

  return routes;
}

function localSearch(routes: Package[][], depot: Point, iterations: number = 500): Package[][] {
  let improved = true;
  let iter = 0;

  while (improved && iter < iterations) {
    improved = false;
    iter++;

    // 2-opt improvement within routes
    for (let r = 0; r < routes.length; r++) {
      const route = routes[r];
      if (route.length < 3) continue;

      for (let i = 0; i < route.length - 1; i++) {
        for (let j = i + 2; j < route.length; j++) {
          const a = route[i];
          const b = route[i + 1];
          const c = route[j];
          const d = j + 1 < route.length ? route[j + 1] : null;

          const oldDist = distance(a, b) + (d ? distance(c, d) : 0);
          const newDist = distance(a, c) + (d ? distance(b, d) : 0);

          if (newDist < oldDist - 1e-9) {
            // Reverse segment [i+1, j]
            const newRoute = [
              ...route.slice(0, i + 1),
              ...route.slice(i + 1, j + 1).reverse(),
              ...route.slice(j + 1)
            ];
            routes[r] = newRoute;
            improved = true;
            break;
          }
        }
        if (improved) break;
      }
    }

    // Try moving packages between routes
    if (!improved) {
      for (let r1 = 0; r1 < routes.length && !improved; r1++) {
        for (let i = 0; i < routes[r1].length && !improved; i++) {
          const pkg = routes[r1][i];
          const oldRoute1Cost = routeCost(routes[r1], depot);
          const route1Without = routes[r1].filter((_, idx) => idx !== i);
          const cost1Without = routeCost(route1Without, depot);

          for (let r2 = 0; r2 < routes.length && !improved; r2++) {
            if (r1 === r2) continue;

            const oldRoute2Cost = routeCost(routes[r2], depot);

            // Try different positions in route r2
            for (let j = 0; j <= routes[r2].length; j++) {
              const route2With = [...routes[r2].slice(0, j), pkg, ...routes[r2].slice(j)];
              const cost2With = routeCost(route2With, depot);

              const oldTotalCost = oldRoute1Cost + oldRoute2Cost;
              const newTotalCost = cost1Without + cost2With;

              if (newTotalCost < oldTotalCost - 1e-9) {
                routes[r1] = route1Without;
                routes[r2] = route2With;
                improved = true;
                break;
              }
            }
          }
        }
      }
    }
  }

  return routes;
}

const input: string = fs.readFileSync(0, 'utf8');
const data: Input = JSON.parse(input);

const depot: Point = { x: data.depot[0], y: data.depot[1] };
const packages = data.packages;
const truckCount = data.truckCount;

// Generate initial routes using greedy insertion
let routes = greedyInsert(packages, depot, truckCount);

// Apply local search improvement
routes = localSearch(routes, depot, 300);

// Filter out empty routes
routes = routes.filter(r => r.length > 0);

// Convert to output format
const output: RouteOutput = {
  routes: routes.map(route => route.map(pkg => pkg.id))
};
