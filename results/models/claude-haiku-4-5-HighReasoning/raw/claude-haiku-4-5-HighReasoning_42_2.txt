OpCapability Geometry

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %normal Location 1
OpDecorate %color Location 2

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %outColor Location 0

%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%c_int_0 = OpConstant %int 0
%c_int_1 = OpConstant %int 1
%c_uint_0 = OpConstant %uint 0
%c_uint_1 = OpConstant %uint 1
%c_uint_2 = OpConstant %uint 2

%c_float_0 = OpConstant %float 0.0
%c_float_1 = OpConstant %float 1.0
%c_float_0p04 = OpConstant %float 0.04
%c_float_0p02 = OpConstant %float 0.02
%c_float_0p03464 = OpConstant %float 0.034641
%c_float_neg0p02 = OpConstant %float -0.02
%c_float_neg0p04 = OpConstant %float -0.04
%c_float_neg0p03464 = OpConstant %float -0.034641
%c_float_0p8 = OpConstant %float 0.8

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %c_uint_1
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in

%normal = OpVariable %ptr_Input_arr_v3 Input
%color = OpVariable %ptr_Input_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex Output

%ptr_Output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3 Output

%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Output_v4 = OpTypePointer Output %v4float

%fn_void = OpTypeFunction %void

%main = OpFunction %void None %fn_void
%entry = OpLabel

%gl_pos_ptr = OpAccessChain %ptr_Input_v4 %gl_in %c_int_0 %c_int_0
%gl_pos = OpLoad %v4float %gl_pos_ptr

%color_ptr = OpAccessChain %ptr_Input_v3 %color %c_int_0
%in_color = OpLoad %v3float %color_ptr

%edge_color = OpVectorTimesScalar %v3float %in_color %c_float_0p8

%gl_pos_z = OpCompositeExtract %float %gl_pos %c_uint_2
%gl_pos_w = OpCompositeExtract %float %gl_pos %c_uint_3

%gl_pos_xy = OpVectorShuffle %v3float %gl_pos %gl_pos 0 1 2

%offset_e0 = OpCompositeConstruct %v3float %c_float_0p04 %c_float_0 %c_float_0
%offset_e1 = OpCompositeConstruct %v3float %c_float_0p02 %c_float_0p03464 %c_float_0
%offset_e2 = OpCompositeConstruct %v3float %c_float_neg0p02 %c_float_0p03464 %c_float_0
%offset_e3 = OpCompositeConstruct %v3float %c_float_neg0p04 %c_float_0 %c_float_0
%offset_e4 = OpCompositeConstruct %v3float %c_float_neg0p02 %c_float_neg0p03464 %c_float_0
%offset_e5 = OpCompositeConstruct %v3float %c_float_0p02 %c_float_neg0p03464 %c_float_0

%e0_xyz = OpFAdd %v3float %gl_pos_xy %offset_e0
%e1_xyz = OpFAdd %v3float %gl_pos_xy %offset_e1
%e2_xyz = OpFAdd %v3float %gl_pos_xy %offset_e2
%e3_xyz = OpFAdd %v3float %gl_pos_xy %offset_e3
%e4_xyz = OpFAdd %v3float %gl_pos_xy %offset_e4
%e5_xyz = OpFAdd %v3float %gl_pos_xy %offset_e5

%center_v4 = OpCompositeConstruct %v4float %gl_pos_xy %gl_pos_z %gl_pos_w
%e0_v4 = OpCompositeConstruct %v4float %e0_xyz %gl_pos_z %gl_pos_w
%e1_v4 = OpCompositeConstruct %v4float %e1_xyz %gl_pos_z %gl_pos_w
%e2_v4 = OpCompositeConstruct %v4float %e2_xyz %gl_pos_z %gl_pos_w
%e3_v4 = OpCompositeConstruct %v4float %e3_xyz %gl_pos_z %gl_pos_w
%e4_v4 = OpCompositeConstruct %v4float %e4_xyz %gl_pos_z %gl_pos_w
%e5_v4 = OpCompositeConstruct %v4float %e5_xyz %gl_pos_z %gl_pos_w

%pos_out_ptr = OpAccessChain %ptr_Output_v4 %gl_out %c_int_0

OpStore %pos_out_ptr %center_v4
OpStore %outColor %in_color
OpEmitVertex
