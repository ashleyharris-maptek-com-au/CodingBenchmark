#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

const int INF = 1e9;

struct Edge {
    int to, cap, flow;
    size_t rev;
};

class MaxFlow {
public:
    vector<vector<Edge>> graph;
    vector<int> level, iter;
    vector<bool> visited;
    int n;

    MaxFlow(int n) : n(n), graph(n), level(n), iter(n), visited(n) {}

    void add_edge(int from, int to, int cap) {
        graph[from].push_back({to, cap, 0, graph[to].size()});
        graph[to].push_back({from, 0, 0, graph[from].size() - 1});
    }

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q;
        level[s] = 0;
        q.push(s);

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (auto& e : graph[v]) {
                if (level[e.to] < 0 && e.cap > e.flow) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }

        return level[t] >= 0;
    }

    int dfs(int v, int t, int pushed) {
        if (v == t || pushed == 0) return pushed;

        for (int& cid = iter[v]; cid < (int)graph[v].size(); cid++) {
            Edge& e = graph[v][cid];
            if (level[v] + 1 != level[e.to] || e.cap <= e.flow)
                continue;

            int tr = dfs(e.to, t, min(pushed, e.cap - e.flow));
            if (tr > 0) {
                e.flow += tr;
                graph[e.to][e.rev].flow -= tr;
                return tr;
            }
        }

        return 0;
    }

    int max_flow(int s, int t) {
        int flow = 0;
        while (bfs(s, t)) {
            fill(iter.begin(), iter.end(), 0);
            while (int pushed = dfs(s, t, INF)) {
                flow += pushed;
            }
        }
        return flow;
    }

    void find_reachable(int s) {
        fill(visited.begin(), visited.end(), false);
        queue<int> q;
        q.push(s);
        visited[s] = true;

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (auto& e : graph[v]) {
                if (!visited[e.to] && e.cap > e.flow) {
                    visited[e.to] = true;
                    q.push(e.to);
                }
            }
        }
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<pair<int, int>> edges(m);

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        edges[i] = {u, v};
    }

    if (n <= 1) {
        cout << 0 << "\n";
        return 0;
    }

    int s = 0;
    int t = (n > 1) ? 1 : 0;
    if (n > 2) {
        t = n - 1;
    }

    MaxFlow mf(n);
    for (auto& e : edges) {
        mf.add_edge(e.first, e.second, 1);
        mf.add_edge(e.second, e.first, 1);
    }

    int min_cut = mf.max_flow(s, t);

    cout << min_cut << "\n";

    if (min_cut == 0) {
        return 0;
    }

    mf.find_reachable(s);

    int count = 0;
    for (int i = 0; i < m; i++) {
        int u = edges[i].first;
        int v = edges[i].second;

        if ((mf.visited[u] && !mf.visited[v]) || (!mf.visited[u] && mf.visited[v])) {
            cout << u << " " << v << "\n";
            count++;
            if (count >= min_cut) break;
        }
    }

    return 0;
}
