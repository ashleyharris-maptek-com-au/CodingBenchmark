use std::io::{self, BufRead};
use std::f64::consts::PI;

#[derive(Clone, Debug)]
struct Point {
    x: f64,
    y: f64,
}

#[derive(Clone)]
struct Polygon {
    vertices: Vec<Point>,
}

#[derive(Clone)]
struct PlacedPiece {
    polygon: Polygon,
    rotation: f64,
    tx: f64,
    ty: f64,
}

impl Polygon {
    fn from_coords(coords: Vec<(f64, f64)>) -> Self {
        Polygon {
            vertices: coords.into_iter().map(|(x, y)| Point { x, y }).collect(),
        }
    }

    fn bbox(&self) -> (f64, f64, f64, f64) {
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for v in &self.vertices {
            min_x = min_x.min(v.x);
            max_x = max_x.max(v.x);
            min_y = min_y.min(v.y);
            max_y = max_y.max(v.y);
        }

        (min_x, min_y, max_x, max_y)
    }

    fn transformed_bbox(&self, rotation: f64, tx: f64, ty: f64) -> (f64, f64, f64, f64) {
        let cos_a = rotation.cos();
        let sin_a = rotation.sin();
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for v in &self.vertices {
            let rx = v.x * cos_a - v.y * sin_a + tx;
            let ry = v.x * sin_a + v.y * cos_a + ty;
            min_x = min_x.min(rx);
            max_x = max_x.max(rx);
            min_y = min_y.min(ry);
            max_y = max_y.max(ry);
        }

        (min_x, min_y, max_x, max_y)
    }

    fn get_transformed_vertices(&self, rotation: f64, tx: f64, ty: f64) -> Vec<Point> {
        let cos_a = rotation.cos();
        let sin_a = rotation.sin();
        self.vertices
            .iter()
            .map(|v| {
                let rx = v.x * cos_a - v.y * sin_a + tx;
                let ry = v.x * sin_a + v.y * cos_a + ty;
                Point { x: rx, y: ry }
            })
            .collect()
    }

    fn contains_point(&self, p: &Point) -> bool {
        point_in_polygon(&self.vertices, p)
    }

    fn contains_polygon(&self, other: &[Point]) -> bool {
        other.iter().all(|p| self.contains_point(p))
    }
}

fn point_in_polygon(polygon: &[Point], point: &Point) -> bool {
    let mut inside = false;
    let mut j = polygon.len() - 1;

    for i in 0..polygon.len() {
        let xi = polygon[i].x;
        let yi = polygon[i].y;
        let xj = polygon[j].x;
        let yj = polygon[j].y;

        if ((yi > point.y) != (yj > point.y))
            && (point.x < (xj - xi) * (point.y - yi) / (yj - yi) + xi)
        {
            inside = !inside;
        }
        j = i;
    }

    inside
}

fn polygons_overlap(poly1: &[Point], poly2: &[Point]) -> bool {
    for v in poly1 {
        if point_in_polygon(poly2, v) {
            return true;
        }
    }
    for v in poly2 {
        if point_in_polygon(poly1, v) {
            return true;
        }
    }
    false
}

fn find_placement(
    stock: &Polygon,
    piece: &Polygon,
    rotation: f64,
    placed_pieces: &[PlacedPiece],
) -> Option<(f64, f64)> {
    let (min_x, min_y, max_x, max_y) = stock.bbox();
    let step = 1.0;

    let mut x = min_x;
    while x <= max_x {
        let mut y = min_y;
        while y <= max_y {
            let (pb_min_x, pb_min_y, pb_max_x, pb_max_y) = piece.transformed_bbox(rotation, x, y);

            if pb_max_x <= max_x + 1e-6 && pb_min_x >= min_x - 1e-6
                && pb_max_y <= max_y + 1e-6 && pb_min_y >= min_y - 1e-6
            {
                let transformed_piece = piece.get_transformed_vertices(rotation, x, y);

                if stock.contains_polygon(&transformed_piece) {
                    let mut valid = true;
                    for placed in placed_pieces {
                        let placed_verts = placed.polygon.get_transformed_vertices(placed.rotation, placed.tx, placed.ty);
                        if polygons_overlap(&transformed_piece, &placed_verts) {
                            valid = false;
                            break;
                        }
                    }

                    if valid {
                        return Some((x, y));
                    }
                }
            }

            y += step;
        }
        x += step;
    }

    None
}

fn solve_cutting_stock(
    stock: &Polygon,
    pieces: &[Polygon],
) -> (usize, Vec<(usize, f64, f64, f64)>) {
    let mut stocks: Vec<Vec<PlacedPiece>> = vec![Vec::new()];
    let mut placements = Vec::new();

    let rotations = [0.0, PI / 6.0, PI / 4.0, PI / 3.0, PI / 2.0, 2.0 * PI / 3.0, 3.0 * PI / 4.0, 5.0 * PI / 6.0];

    for piece in pieces {
        let mut placed = false;

        for (stock_idx, placed_pieces) in stocks.iter_mut().enumerate() {
            for &rotation in &rotations {
                if let Some((x, y)) = find_placement(stock, piece, rotation, placed_pieces) {
                    placements.push((stock_idx, x, y, rotation));
                    placed_pieces.push(PlacedPiece {
                        polygon: piece.clone(),
                        rotation,
                        tx: x,
                        ty: y,
                    });
                    placed = true;
                    break;
                }
            }

            if placed {
                break;
            }
        }

        if !placed {
            placements.push((stocks.len(), 0.0, 0.0, 0.0));
            stocks.push(vec![PlacedPiece {
                polygon: piece.clone(),
                rotation: 0.0,
                tx: 0.0,
                ty: 0.0,
            }]);
        }
    }

    (stocks.len(), placements)
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let header_line = lines.next().unwrap().unwrap();
    let header: Vec<usize> = header_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let sv = header[0];
    let np = header[1];

    let mut stock_coords = Vec::new();
    for _ in 0..sv {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<f64> = line
            .split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        stock_coords.push((coords[0], coords[1]));
    }
    let stock = Polygon::from_coords(stock_coords);

    let mut pieces = Vec::new();
    for _ in 0..np {
        let pv_line = lines.next().unwrap().unwrap();
        let pv: usize = pv_line.trim().parse().unwrap();
        let mut piece_coords = Vec::new();
        for _ in 0..pv {
            let line = lines.next().unwrap().unwrap();
            let coords: Vec<f64> = line
                .split_whitespace()
                .map(|s| s.parse().unwrap())
                .collect();
            piece_coords.push((coords[0], coords[1]));
        }
        pieces.push(Polygon::from_coords(piece_coords));
    }

    let (num_stocks, placements) = solve_cutting_stock(&stock, &pieces);

    println!("{}", num_stocks);
    for (stock_idx, x, y, rotation) in placements {
        println!("{} {} {} {}", stock_idx, x, y, rotation);
    }
}
