#include <bits/stdc++.h>
using namespace std;

int n, m;
vector<int> adj[100005];
vector<int> best_set;
int best_size = 0;

void solve(vector<bool>& in_set, vector<bool>& used, int last_added) {
    bool added_any = false;

    // Try to add more vertices to the independent set
    for (int v = last_added + 1; v < n; v++) {
        if (!used[v]) {
            added_any = true;

            // Include vertex v
            in_set[v] = true;
            vector<bool> prev_used = used;
            used[v] = true;

            // Mark v and all its neighbors as used
            for (int u : adj[v]) {
                used[u] = true;
            }

            // Recursively try to add more vertices
            solve(in_set, used, v);

            // Backtrack
            in_set[v] = false;
            used = prev_used;
        }
    }

    // If no vertex was added, we've reached a maximal independent set
    if (!added_any) {
        int size = 0;
        for (int i = 0; i < n; i++) {
            if (in_set[i]) size++;
        }

        // Update best solution if this is better
        if (size > best_size) {
            best_size = size;
            best_set.clear();
            for (int i = 0; i < n; i++) {
                if (in_set[i]) best_set.push_back(i);
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<bool> in_set(n, false);
    vector<bool> used(n, false);

    solve(in_set, used, -1);

    cout << best_size << "\n";
    for (int i = 0; i < best_set.size(); i++) {
        if (i > 0) cout << " ";
        cout << best_set[i];
    }
    cout << "\n";

    return 0;
}
