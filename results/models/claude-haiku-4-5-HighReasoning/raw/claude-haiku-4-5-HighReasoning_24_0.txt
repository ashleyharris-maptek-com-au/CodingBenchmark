#include <iostream>
#include <fstream>
#include <thread>
#include <mutex>
#include <atomic>
#include <cmath>
#include <sstream>
#include <chrono>
#include <vector>
#include <algorithm>
#include <queue>

using namespace std;

struct TerrainCell {
    uint16_t ground_top;
    uint16_t float_bottom;
    uint16_t float_top;
};

struct State {
    float x, y, z;
    float vx, vy, vz;
    float pitch, yaw, roll;
    float pitch_rate, yaw_rate, roll_rate;
    float fuel;
};

struct Command {
    float thrust, pitch_cmd, yaw_cmd, roll_cmd;
};

class LunarLander {
public:
    int width, depth;
    float height, gravity, max_thrust;
    float target_x, target_y, target_z;
    vector<TerrainCell> terrain;

    atomic<bool> have_state{false};
    atomic<bool> running{true};
    mutex state_mutex;
    State current_state{};

    bool load_terrain(const string& path) {
        ifstream file(path, ios::binary);
        if (!file.is_open()) return false;

        terrain.resize(width * depth);
        for (int x = 0; x < width; x++) {
            for (int y = 0; y < depth; y++) {
                TerrainCell cell;
                file.read((char*)&cell.ground_top, 2);
                file.read((char*)&cell.float_bottom, 2);
                file.read((char*)&cell.float_top, 2);
                terrain[x * depth + y] = cell;
            }
        }
        return true;
    }

    bool is_solid(float x, float y, float z) {
        if (x < 0 || x >= width || y < 0 || y >= depth || z < 0 || z >= height)
            return true;

        int ix = (int)x;
        int iy = (int)y;
        if (ix >= width) ix = width - 1;
        if (iy >= depth) iy = depth - 1;

        const TerrainCell& cell = terrain[ix * depth + iy];
        float zf = z;

        if (zf < cell.ground_top) return true;
        if (zf >= cell.float_bottom && zf < cell.float_top) return true;
        return false;
    }

    Command compute_control(const State& state) {
        float dx = target_x - state.x;
        float dy = target_y - state.y;
        float dz = target_z - state.z;

        float dist_sq = dx*dx + dy*dy + dz*dz;
        float dist = sqrt(dist_sq);
        float hdist = sqrt(dx*dx + dy*dy);

        float speed_h = sqrt(state.vx*state.vx + state.vy*state.vy);
        float speed_v = state.vz;
        float speed = sqrt(state.vx*state.vx + state.vy*state.vy + state.vz*state.vz);

        Command cmd{0, 0, 0, 0};

        // Landing zone check: within ~50m and slow
        bool near_target = dist < max(50.0f, max(width, depth) * 0.02f * 2);

        if (near_target && speed < 3.5f) {
            // Landing mode: gentle hover and level out
            cmd.thrust = 0.5f;
            cmd.pitch_cmd = -state.pitch * 0.2f - state.pitch_rate * 0.05f;
            cmd.yaw_cmd = -state.yaw * 0.2f - state.yaw_rate * 0.05f;
            cmd.roll_cmd = -state.roll * 0.2f - state.roll_rate * 0.05f;
        } else if (dist > 1.0f) {
            // Approach mode: navigate toward target

            // Horizontal control
            if (hdist > 0.1f) {
                // Calculate desired pitch and yaw to move toward target
                float desired_pitch = clamp((dx / hdist) * 0.4f, -0.5f, 0.5f);
                float desired_yaw = clamp(-(dy / hdist) * 0.4f, -0.5f, 0.5f);

                // PD control
                cmd.pitch_cmd = (desired_pitch - state.pitch) * 0.4f - state.pitch_rate * 0.1f;
                cmd.yaw_cmd = (desired_yaw - state.yaw) * 0.4f - state.yaw_rate * 0.1f;

                cmd.pitch_cmd = clamp(cmd.pitch_cmd, -1.0f, 1.0f);
                cmd.yaw_cmd = clamp(cmd.yaw_cmd, -1.0f, 1.0f);
            } else {
                // Directly above target, level out
                cmd.pitch_cmd = -state.pitch * 0.3f - state.pitch_rate * 0.1f;
                cmd.yaw_cmd = -state.yaw * 0.3f - state.yaw_rate * 0.1f;
            }

            // Vertical control
            if (dz > 500.0f) {
                cmd.thrust = 0.8f;
            } else if (dz > 100.0f) {
                cmd.thrust = 0.6f;
            } else if (dz > 20.0f) {
                cmd.thrust = 0.5f;
            } else {
                // Final approach: control descent rate
                cmd.thrust = max(0.2f, 0.5f - speed_v * 0.15f);
            }
        } else {
            // At target
            cmd.thrust = 0.5f;
            cmd.pitch_cmd = -state.pitch * 0.3f;
            cmd.yaw_cmd = -state.yaw * 0.3f;
            cmd.roll_cmd = -state.roll * 0.3f;
        }

        // Dampen roll
        cmd.roll_cmd = clamp(-state.roll * 0.3f - state.roll_rate * 0.1f, -1.0f, 1.0f);

        // Clamp all outputs
        cmd.thrust = clamp(cmd.thrust, 0.0f, 1.0f);

        return cmd;
    }

private:
    template<typename T>
    T clamp(T val, T lo, T hi) {
        return val < lo ? lo : (val > hi ? hi : val);
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    LunarLander lander;

    // Read initial header
    cin >> lander.width >> lander.depth >> lander.height
        >> lander.gravity >> lander.max_thrust;

    float start_x, start_y, start_z;
    cin >> start_x >> start_y >> start_z
        >> lander.target_x >> lander.target_y >> lander.target_z;

    cin.ignore(10000, '\n');
    string terrain_path;
    getline(cin, terrain_path);

    // Skip STATE header line
    string state_header;
    getline(cin, state_header);

    // Load terrain file
    if (!lander.load_terrain(terrain_path)) {
        cerr << "Failed to load terrain file: " << terrain_path << endl;
        return 1;
    }

    // Start reader thread for continuous stdin
    thread reader_thread([&lander]() {
        string line;
        while (getline(cin, line)) {
            if (line.empty()) continue;

            State state;
            istringstream iss(line);
            if (!(iss >> state.x >> state.y >> state.z
                     >> state.vx >> state.vy >> state.vz
                     >> state.pitch >> state.yaw >> state.roll
                     >> state.pitch_rate >> state.yaw_rate >> state.roll_rate
                     >> state.fuel)) {
                continue;
            }

            {
                lock_guard<mutex> lock(lander.state_mutex);
                lander.current_state = state;
                lander.have_state = true;
            }
        }
        lander.running = false;
    });
    reader_thread.detach();

    // Main control loop: output at 10 Hz
    auto last_output = chrono::high_resolution_clock::now();
    int output_count = 0;

    while (lander.running) {
        auto now = chrono::high_resolution_clock::now();
        auto elapsed_ms = chrono::duration<double>(now - last_output).count() * 1000.0;

        if (elapsed_ms >= 100.0 && lander.have_state) {
            State current;
            {
                lock_guard<mutex> lock(lander.state_mutex);
                current = lander.current_state;
            }

            Command cmd = lander.compute_control(current);

            // Output with consistent formatting
            cout << cmd.thrust << " " << cmd.pitch_cmd << " "
                 << cmd.yaw_cmd << " " << cmd.roll_cmd << "\n";
            cout.flush();

            last_output = now;
            output_count++;
        }

        // Sleep briefly to avoid busy-waiting
        this_thread::sleep_for(chrono::milliseconds(1));
    }

    return 0;
}
