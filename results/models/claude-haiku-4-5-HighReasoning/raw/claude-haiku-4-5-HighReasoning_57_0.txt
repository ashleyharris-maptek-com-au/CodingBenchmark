const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { width, height, grid, ignitions, towns, cutBudget } = data;

const gridArray = grid.map(row => row.split(''));

function simulateFire(cuts) {
  const burned = Array(height).fill(null).map(() => Array(width).fill(false));
  const queue = [];

  for (const [x, y] of ignitions) {
    if (gridArray[y][x] === '.') {
      burned[y][x] = true;
      queue.push([x, y]);
    }
  }

  while (queue.length > 0) {
    const [x, y] = queue.shift();
    for (const [dx, dy] of [[0, 1], [1, 0], [0, -1], [-1, 0]]) {
      const nx = x + dx;
      const ny = y + dy;
      if (nx >= 0 && nx < width && ny >= 0 && ny < height) {
        if (!burned[ny][nx] && gridArray[ny][nx] === '.' && !cuts[ny][nx]) {
          burned[ny][nx] = true;
          queue.push([nx, ny]);
        }
      }
    }
  }

  return burned;
}

const cuts = Array(height).fill(null).map(() => Array(width).fill(false));
let burned = simulateFire(cuts);

const firebreaks = [];

for (let budget = 0; budget < cutBudget; budget++) {
  let bestCell = null;
  let bestValue = 0;

  // Find cells on the fire boundary (unburned but adjacent to burned cells)
  const boundary = [];
  for (let y = 0; y < height; y++) {
    for (let x = 0; x < width; x++) {
      if (!burned[y][x] && gridArray[y][x] === '.' && !cuts[y][x]) {
        for (const [dx, dy] of [[0, 1], [1, 0], [0, -1], [-1, 0]]) {
          const nx = x + dx;
          const ny = y + dy;
          if (nx >= 0 && nx < width && ny >= 0 && ny < height && burned[ny][nx]) {
            boundary.push([x, y]);
            break;
          }
        }
      }
    }
  }

  // Evaluate each boundary cell
  for (const [x, y] of boundary) {
    cuts[y][x] = true;
    const newBurned = simulateFire(cuts);

    let savedValue = 0;
    for (const town of towns) {
      if (burned[town.y][town.x] && !newBurned[town.y][town.x]) {
        savedValue += town.value;
      }
    }

    if (savedValue > bestValue) {
      bestValue = savedValue;
      bestCell = [x, y];
    }

    cuts[y][x] = false;
  }

  if (!bestCell || bestValue === 0) break;

  cuts[bestCell[1]][bestCell[0]] = true;
  firebreaks.push(bestCell);
  burned = simulateFire(cuts);
}
