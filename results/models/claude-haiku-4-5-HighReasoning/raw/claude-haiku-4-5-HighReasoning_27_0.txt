#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    cin >> n >> m >> k;

    vector<vector<int>> adj(n);

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<int> color(n, -1);
    vector<int> last_used(k, -1);

    // Adaptive strategy: use vertex ordering for smaller graphs
    bool use_ordering = (n <= 5000);

    if (use_ordering) {
        // Welsh-Powell: sort vertices by degree (descending)
        vector<pair<int, int>> degree_vertex;
        for (int i = 0; i < n; i++) {
            degree_vertex.push_back({adj[i].size(), i});
        }
        sort(degree_vertex.rbegin(), degree_vertex.rend());

        int iter = 0;
        for (auto [deg, u] : degree_vertex) {
            iter++;
            for (int v : adj[u]) {
                if (color[v] != -1) {
                    last_used[color[v]] = iter;
                }
            }
            for (int c = 0; c < k; c++) {
                if (last_used[c] != iter) {
                    color[u] = c;
                    break;
                }
            }
        }
    } else {
        // Simple greedy coloring for large graphs
        int iter = 0;
        for (int u = 0; u < n; u++) {
            iter++;
            for (int v : adj[u]) {
                if (color[v] != -1) {
                    last_used[color[v]] = iter;
                }
            }
            for (int c = 0; c < k; c++) {
                if (last_used[c] != iter) {
                    color[u] = c;
                    break;
                }
            }
        }
    }

    // Output
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << " ";
        cout << color[i];
    }
    cout << "\n";

    return 0;
}
