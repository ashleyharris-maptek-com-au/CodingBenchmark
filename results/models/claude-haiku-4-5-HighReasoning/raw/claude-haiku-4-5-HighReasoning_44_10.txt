#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Create horizontal stripes using sin(uv.y * 3.14159 * 16.0)
    float stripe = sin(uv.y * 3.14159 * 16.0);

    // Map sine to [0, 1] range for color mixing
    float stripeFactor = stripe * 0.5 + 0.5;

    // Alternate between red and blue
    vec3 stripeColor = mix(vec3(0.9, 0.1, 0.1), vec3(0.1, 0.1, 0.9), stripeFactor);

    // Normalize the interpolated normal
    vec3 norm = normalize(normal);

    // Calculate light direction from fragment to light
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);

    // Calculate diffuse component (Lambertian)
    float diffuse = max(0.0, dot(norm, lightDir));

    // Apply Lambertian shading factor: 0.4 + 0.6 * diffuse
    float shadingFactor = 0.4 + 0.6 * diffuse;

    // Apply shading to stripe color
    vec3 finalColor = stripeColor * shadingFactor;

    // Output with alpha 1.0
    outColor = vec4(finalColor, 1.0);
}
