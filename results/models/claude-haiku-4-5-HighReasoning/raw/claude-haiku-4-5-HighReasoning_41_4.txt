OpCapability Shader
OpExtension "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_color
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %out_color Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

%glsl_ext = OpExtInstImport "GLSL.std.450"

%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2 Input
%in_tangent = OpVariable %ptr_Input_vec3 Input
%in_color = OpVariable %ptr_Input_vec3 Input
%out_color = OpVariable %ptr_Output_vec4 Output

%uboVar = OpVariable %ptr_Uniform_UBO Uniform

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%int_3 = OpConstant %int 3

%func_type = OpTypeFunction %void

%main = OpFunction %void None %func_type
%entry = OpLabel

%worldPos_val = OpLoad %vec3 %in_worldPos
%normal_val = OpLoad %vec3 %in_normal
%color_val = OpLoad %vec3 %in_color

%lightPos_ptr = OpAccessChain %ptr_Uniform_vec4 %uboVar %int_3
%lightPos_val = OpLoad %vec4 %lightPos_ptr

%lightPos_xyz = OpVectorShuffle %vec3 %lightPos_val %lightPos_val 0 1 2

%lightDir = OpFSub %vec3 %lightPos_xyz %worldPos_val
%L = OpExtInst %vec3 %glsl_ext Normalize %lightDir

%N = OpExtInst %vec3 %glsl_ext Normalize %normal_val

%dot_result = OpDot %float %N %L
%d = OpExtInst %float %glsl_ext FMax %dot_result %float_0

%color_scaled = OpVectorTimesScalar %vec3 %color_val %d
%final_color = OpCompositeConstruct %vec4 %color_scaled %float_1

OpStore %out_color %final_color

OpReturn
OpFunctionEnd
