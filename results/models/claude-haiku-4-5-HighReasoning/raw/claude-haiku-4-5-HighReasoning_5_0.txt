use std::io;
use std::collections::HashSet;

fn main() {
    let mut input = String::new();
    io::stdin().read_line(&mut input).unwrap();

    let parts: Vec<&str> = input.trim().split_whitespace().collect();
    let width: usize = parts[0].parse().unwrap();
    let height: usize = parts[1].parse().unwrap();
    let obstacle_count: usize = parts[2].parse().unwrap();
    let require_cycle: bool = parts[3].parse::<usize>().unwrap() != 0;

    let mut obstacles = HashSet::new();
    for _ in 0..obstacle_count {
        let mut line = String::new();
        io::stdin().read_line(&mut line).unwrap();
        let coords: Vec<usize> = line.trim().split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        obstacles.insert((coords[0], coords[1]));
    }

    let total_cells = width * height - obstacle_count;

    if obstacles.contains(&(0, 0)) {
        println!("0");
        return;
    }

    // Try constructive snake pattern for empty grids
    if obstacle_count == 0 && total_cells >= 100 {
        if let Some(path) = construct_snake_path(width, height, require_cycle) {
            println!("{}", path.len());
            for (x, y) in path {
                println!("{} {}", x, y);
            }
            return;
        }
    }

    // Use backtracking for other cases
    if let Some(path) = find_hamiltonian_path(width, height, &obstacles, require_cycle, total_cells) {
        println!("{}", path.len());
        for (x, y) in path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}

fn construct_snake_path(width: usize, height: usize, require_cycle: bool) -> Option<Vec<(usize, usize)>> {
    let mut path = Vec::new();

    // Create snake pattern: zigzag through grid
    for y in 0..height {
        if y % 2 == 0 {
            // Left to right
            for x in 0..width {
                path.push((x, y));
            }
        } else {
            // Right to left
            for x in (0..width).rev() {
                path.push((x, y));
            }
        }
    }

    if require_cycle {
        // Check if last cell is adjacent to (0, 0)
        let (lx, ly) = path[path.len() - 1];
        let is_adjacent = (lx == 1 && ly == 0) || (lx == 0 && ly == 1);
        if !is_adjacent {
            return None;
        }
    }

    Some(path)
}

fn find_hamiltonian_path(
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    require_cycle: bool,
    total_cells: usize,
) -> Option<Vec<(usize, usize)>> {
    let mut path = vec![(0, 0)];
    let mut visited = HashSet::new();
    visited.insert((0, 0));

    if backtrack(&mut path, &mut visited, width, height, obstacles, require_cycle, total_cells) {
        Some(path)
    } else {
        None
    }
}

fn backtrack(
    path: &mut Vec<(usize, usize)>,
    visited: &mut HashSet<(usize, usize)>,
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    require_cycle: bool,
    total_cells: usize,
) -> bool {
    if visited.len() == total_cells {
        if require_cycle {
            let (lx, ly) = path[path.len() - 1];
            return (lx == 1 && ly == 0) || (lx == 0 && ly == 1);
        }
        return true;
    }

    let (x, y) = path[path.len() - 1];

    // Get and sort candidate neighbors
    let mut candidates = Vec::new();
    if x > 0 { candidates.push((x - 1, y)); }
    if x < width - 1 { candidates.push((x + 1, y)); }
    if y > 0 { candidates.push((x, y - 1)); }
    if y < height - 1 { candidates.push((x, y + 1)); }

    // Heuristic: prioritize moves to cells with more unvisited neighbors
    // This helps avoid dead-ends early
    candidates.sort_by_key(|&(nx, ny)| {
        if visited.contains(&(nx, ny)) || obstacles.contains(&(nx, ny)) {
            return std::usize::MAX;
        }

        let mut unvisited_count = 0;

        let neighbors = vec![
            (nx.saturating_sub(1), ny),
            (nx + 1, ny),
            (nx, ny.saturating_sub(1)),
            (nx, ny + 1),
        ];

        for (nnx, nny) in neighbors {
            if nnx < width && nny < height && !visited.contains(&(nnx, nny)) && !obstacles.contains(&(nnx, nny)) {
                unvisited_count += 1;
            }
        }

        // Return negative so higher counts are tried first
        std::cmp::Reverse(unvisited_count)
    });

    for (nx, ny) in candidates {
        if !visited.contains(&(nx, ny)) && !obstacles.contains(&(nx, ny)) {
            path.push((nx, ny));
            visited.insert((nx, ny));

            if backtrack(path, visited, width, height, obstacles, require_cycle, total_cells) {
                return true;
            }

            path.pop();
            visited.remove(&(nx, ny));
        }
    }

    false
}
