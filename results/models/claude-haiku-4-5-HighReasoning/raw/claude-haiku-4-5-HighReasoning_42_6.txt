OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %color %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_color = OpTypeArray %v3float %uint_3
%ptr_in_arr_color = OpTypePointer Input %arr_color
%color = OpVariable %ptr_in_arr_color Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %func_void
%entry = OpLabel

%in_pos_0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%in_pos_0 = OpLoad %v4float %in_pos_0_ptr
%out_pos_0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_0_ptr %in_pos_0
%color_0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%color_0 = OpLoad %v3float %color_0_ptr
OpStore %outColor %color_0
OpEmitVertex

%in_pos_1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%in_pos_1 = OpLoad %v4float %in_pos_1_ptr
%out_pos_1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_1_ptr %in_pos_1
%color_1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%color_1 = OpLoad %v3float %color_1_ptr
OpStore %outColor %color_1
OpEmitVertex

%in_pos_2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%in_pos_2 = OpLoad %v4float %in_pos_2_ptr
%out_pos_2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_2_ptr %in_pos_2
%color_2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%color_2 = OpLoad %v3float %color_2_ptr
OpStore %outColor %color_2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
