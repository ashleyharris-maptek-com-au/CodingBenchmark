using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;
using System.Threading.Tasks;

class MaxCliqueFinder
{
    private Dictionary<int, HashSet<int>> adj;
    private int n, m;
    private Stopwatch timer;
    private const int TIMEOUT_SEC = 285;
    private List<int> bestClique;
    private object lockObj = new object();

    static void Main()
    {
        new MaxCliqueFinder().Solve();
    }

    void Solve()
    {
        timer = Stopwatch.StartNew();
        ReadInput();
        bestClique = new List<int>();
        FindMaxClique();
        OutputResult();
    }

    void ReadInput()
    {
        var first = Console.ReadLine().Split();
        n = int.Parse(first[0]);
        m = int.Parse(first[1]);

        adj = new Dictionary<int, HashSet<int>>();
        for (int i = 0; i < n; i++)
            adj[i] = new HashSet<int>();

        for (int i = 0; i < m; i++)
        {
            if (timer.Elapsed.TotalSeconds > TIMEOUT_SEC)
                break;
            var parts = Console.ReadLine().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            adj[u].Add(v);
            adj[v].Add(u);
        }
    }

    void FindMaxClique()
    {
        if (n <= 1500)
        {
            BranchAndBound();
        }
        else if (n <= 50000)
        {
            GreedyMultiStartParallel(Math.Min(50, n / 100));
        }
        else
        {
            GreedyMultiStartParallel(Math.Min(10, n / 100000));
        }
    }

    void BranchAndBound()
    {
        var allVertices = Enumerable.Range(0, n).ToList();
        var current = new List<int>();
        BnBRecursive(current, allVertices);
    }

    void BnBRecursive(List<int> current, List<int> candidates)
    {
        if (timer.Elapsed.TotalSeconds > TIMEOUT_SEC)
            return;

        if (candidates.Count == 0)
        {
            lock (lockObj)
            {
                if (current.Count > bestClique.Count)
                    bestClique = new List<int>(current);
            }
            return;
        }

        lock (lockObj)
        {
            if (current.Count + candidates.Count <= bestClique.Count)
                return;
        }

        candidates = candidates.OrderByDescending(v =>
            candidates.Count(u => adj[v].Contains(u))).ToList();

        for (int i = 0; i < candidates.Count; i++)
        {
            if (timer.Elapsed.TotalSeconds > TIMEOUT_SEC)
                return;

            int v = candidates[i];
            current.Add(v);

            var next = new List<int>();
            for (int j = i + 1; j < candidates.Count; j++)
            {
                if (adj[v].Contains(candidates[j]))
                    next.Add(candidates[j]);
            }

            BnBRecursive(current, next);
            current.RemoveAt(current.Count - 1);
        }
    }

    void GreedyMultiStartParallel(int numWorkers)
    {
        var orderedByDegree = Enumerable.Range(0, n)
            .OrderByDescending(v => adj[v].Count)
            .ToList();

        numWorkers = Math.Max(1, numWorkers);
        int step = Math.Max(1, n / numWorkers);

        var tasks = new List<Task>();
        for (int start = 0; start < Math.Min(n, 100) && timer.Elapsed.TotalSeconds < TIMEOUT_SEC; start++)
        {
            int s = orderedByDegree[start];
            tasks.Add(Task.Run(() =>
            {
                if (timer.Elapsed.TotalSeconds >= TIMEOUT_SEC)
                    return;

                var clique = GreedyFrom(s);
                lock (lockObj)
                {
                    if (clique.Count > bestClique.Count)
                        bestClique = clique;
                }
            }));

            if (tasks.Count >= numWorkers)
            {
                Task.WaitAll(tasks.ToArray());
                tasks.Clear();
            }
        }

        if (tasks.Count > 0)
            Task.WaitAll(tasks.ToArray());
    }

    List<int> GreedyFrom(int start)
    {
        if (timer.Elapsed.TotalSeconds > TIMEOUT_SEC)
            return new List<int> { start };

        var clique = new List<int> { start };
        var candidates = new HashSet<int>(adj[start]);

        while (candidates.Count > 0 && timer.Elapsed.TotalSeconds < TIMEOUT_SEC)
        {
            int best = -1;
            int maxDegree = -1;

            foreach (var v in candidates)
            {
                int degree = 0;
                foreach (var u in candidates)
                {
                    if (v != u && adj[v].Contains(u))
                        degree++;
                }
                if (degree > maxDegree)
                {
                    maxDegree = degree;
                    best = v;
                }
            }

            if (best == -1)
                break;

            clique.Add(best);

            var newCandidates = new HashSet<int>();
            foreach (var v in candidates)
            {
                if (adj[best].Contains(v))
                    newCandidates.Add(v);
            }
            candidates = newCandidates;
        }

        return clique;
    }

    void OutputResult()
    {
        Console.WriteLine(bestClique.Count);
        if (bestClique.Count > 0)
        {
            var sorted = bestClique.OrderBy(x => x).ToList();
            Console.WriteLine(string.Join(" ", sorted));
        }
    }
}
