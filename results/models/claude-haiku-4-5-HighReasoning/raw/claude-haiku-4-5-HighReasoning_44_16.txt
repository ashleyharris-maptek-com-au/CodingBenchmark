#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Map UV to complex plane: c = (uv.x*3-2, uv.y*2-1)
    vec2 c = vec2(uv.x * 3.0 - 2.0, uv.y * 2.0 - 1.0);

    // Mandelbrot iteration
    vec2 z = vec2(0.0);
    int iterations = 0;

    for (int i = 0; i < 20; i++) {
        // z = z^2 + c
        float zx = z.x * z.x - z.y * z.y + c.x;
        float zy = 2.0 * z.x * z.y + c.y;
        z = vec2(zx, zy);

        iterations++;

        // Escape when dot(z,z) > 4
        if (dot(z, z) > 4.0) {
            break;
        }
    }

    // Color with t=iterations/20 as (t, t*0.5, 1.0-t), alpha 1.0
    float t = float(iterations) / 20.0;
    outColor = vec4(t, t * 0.5, 1.0 - t, 1.0);
}
