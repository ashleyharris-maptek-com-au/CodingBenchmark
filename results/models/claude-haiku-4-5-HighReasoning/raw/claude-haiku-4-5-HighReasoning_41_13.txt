OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %out_color
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %out_color Location 0
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 MatrixStride 16
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_UBO = OpTypePointer Uniform %UBO
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Output_vec4 = OpTypePointer Output %vec4
%fn_type = OpTypeFunction %void
%in_worldPos = OpVariable %ptr_Input_vec3 Input
%out_color = OpVariable %ptr_Output_vec4 Output
%ubo_var = OpVariable %ptr_UBO Uniform
%float_0 = OpConstant %float 0.0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%vec3_bottom = OpConstantComposite %vec3 %float_0_1 %float_0_4 %float_0_1
%vec3_top = OpConstantComposite %vec3 %float_1 %float_1 %float_1
%main = OpFunction %void None %fn_type
%entry = OpLabel
%worldPos = OpLoad %vec3 %in_worldPos
%y = OpCompositeExtract %float %worldPos 1
%y_times_half = OpFMul %float %y %float_0_5
%y_plus = OpFAdd %float %y_times_half %float_0_5
%max_0 = OpFMax %float %y_plus %float_0
%t = OpFMin %float %max_0 %float_1
%one_minus_t = OpFSub %float %float_1 %t
%bottom_scaled = OpVectorTimesScalar %vec3 %vec3_bottom %one_minus_t
%top_scaled = OpVectorTimesScalar %vec3 %vec3_top %t
%rgb = OpFAdd %vec3 %bottom_scaled %top_scaled
%color = OpCompositeConstruct %vec4 %rgb %float_1
OpStore %out_color %color
OpReturn
OpFunctionEnd
