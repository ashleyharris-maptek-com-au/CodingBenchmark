OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_color
OpExecutionMode %main OriginUpperLeft

; Decorations for inputs/outputs
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %out_color Location 0

; Decorations for UBO
OpDecorate %ubo_type Block
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; UBO member decorations
OpMemberDecorate %ubo_type 0 ColMajor
OpMemberDecorate %ubo_type 0 MatrixStride 16
OpMemberDecorate %ubo_type 0 Offset 0
OpMemberDecorate %ubo_type 1 ColMajor
OpMemberDecorate %ubo_type 1 MatrixStride 16
OpMemberDecorate %ubo_type 1 Offset 64
OpMemberDecorate %ubo_type 2 ColMajor
OpMemberDecorate %ubo_type 2 MatrixStride 16
OpMemberDecorate %ubo_type 2 Offset 128
OpMemberDecorate %ubo_type 3 Offset 192
OpMemberDecorate %ubo_type 4 Offset 208
OpMemberDecorate %ubo_type 5 Offset 224

; Type declarations
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4float = OpTypeMatrix %v4float 4

%ubo_type = OpTypeStruct %mat4float %mat4float %mat4float %v4float %v4float %v4float

%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_ubo_type = OpTypePointer Uniform %ubo_type
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%main_type = OpTypeFunction %void

; Constants
%c_0 = OpConstant %float 0.0
%c_1 = OpConstant %float 1.0
%c_8 = OpConstant %float 8.0
%c_0_2 = OpConstant %float 0.2
%c_0_5 = OpConstant %float 0.5
%c_0_9 = OpConstant %float 0.9
%c_int_0 = OpConstant %int 0
%c_int_2 = OpConstant %int 2
%c_int_3 = OpConstant %int 3

; Variables
%in_worldPos = OpVariable %ptr_Input_v3float Input
%in_normal = OpVariable %ptr_Input_v3float Input
%in_uv = OpVariable %ptr_Input_v2float Input
%in_tangent = OpVariable %ptr_Input_v3float Input
%in_color = OpVariable %ptr_Input_v3float Input
%out_color = OpVariable %ptr_Output_v4float Output
%ubo_var = OpVariable %ptr_Uniform_ubo_type Uniform

; Main function
%main = OpFunction %void None %main_type
%label = OpLabel

; Load inputs
%uv = OpLoad %v2float %in_uv
%normal = OpLoad %v3float %in_normal

; Calculate checkerboard: scale UV by 8
%scaled = OpVectorTimesScalar %v2float %uv %c_8

; Extract components
%sx = OpCompositeExtract %float %scaled 0
%sy = OpCompositeExtract %float %scaled 1

; Convert to integers (implicit floor)
%ix = OpConvertFToS %int %sx
%iy = OpConvertFToS %int %sy

; Check parity: (ix + iy) % 2
%sum = OpIAdd %int %ix %iy
%parity = OpUMod %int %sum %c_int_2
%is_even = OpIEqual %bool %parity %c_int_0

; Select color based on parity (0.9 for even, 0.2 for odd)
%color = OpSelect %float %is_even %c_0_9 %c_0_2

; Load light position from UBO
%light_addr = OpAccessChain %ptr_Uniform_v4float %ubo_var %c_int_3
%light_pos = OpLoad %v4float %light_addr

; Extract XYZ from light position
%light_dir = OpVectorShuffle %v3float %light_pos %light_pos 0 1 2

; Normalize light direction
%light_norm = OpExtInst %v3float %ext OpNormalize %light_dir

; Calculate diffuse: dot(normal, light_dir)
%diffuse = OpDot %float %normal %light_norm

; Clamp diffuse to [0, 1]
%diffuse_clamped = OpExtInst %float %ext OpClamp %diffuse %c_0 %c_1

; Shading factor: 0.5 + 0.5 * diffuse
%half_diffuse = OpFMul %float %c_0_5 %diffuse_clamped
%shading = OpFAdd %float %c_0_5 %half_diffuse

; Final color = checkerboard color * shading factor
%final = OpFMul %float %color %shading

; Create RGBA output with alpha = 1.0
%result = OpCompositeConstruct %v4float %final %final %final %c_1

; Store result
OpStore %out_color %result

OpReturn
OpFunctionEnd
