use std::collections::{VecDeque, HashSet, HashMap};
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read dimensions
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let h: usize = parts[0].parse().unwrap();
    let w: usize = parts[1].parse().unwrap();

    // Read the maze
    let mut maze = Vec::with_capacity(h);
    let mut start = (0, 0);
    let mut end = (0, 0);

    for i in 0..h {
        let line = lines.next().unwrap().unwrap();
        let row: Vec<u8> = line
            .chars()
            .enumerate()
            .map(|(j, c)| {
                match c {
                    'A' => { start = (i, j); 0 },
                    'B' => { end = (i, j); 1 },
                    '#' => 2,
                    _ => 3, // space or any other character
                }
            })
            .collect();

        maze.push(row);
    }

    // BFS to find path
    let path = bfs(&maze, start, end, h, w);

    // Output
    if let Some(path) = path {
        println!("{}", path.len());
        for (y, x) in path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}

fn bfs(
    maze: &[Vec<u8>],
    start: (usize, usize),
    end: (usize, usize),
    h: usize,
    w: usize,
) -> Option<Vec<(usize, usize)>> {
    let mut visited = HashSet::new();
    let mut parent: HashMap<(usize, usize), (usize, usize)> = HashMap::new();
    let mut queue = VecDeque::new();

    queue.push_back(start);
    visited.insert(start);

    while let Some((y, x)) = queue.pop_front() {
        if (y, x) == end {
            // Reconstruct path from end to start using parent map
            let mut path = Vec::new();
            let mut current = end;

            loop {
                path.push(current);
                if current == start {
                    break;
                }
                current = *parent.get(&current).unwrap();
            }

            path.reverse();
            return Some(path);
        }

        // Try all 4 directions: right, down, left, up
        if x + 1 < w && !visited.contains(&(y, x + 1)) && maze[y][x + 1] != 2 {
            visited.insert((y, x + 1));
            parent.insert((y, x + 1), (y, x));
            queue.push_back((y, x + 1));
        }

        if y + 1 < h && !visited.contains(&(y + 1, x)) && maze[y + 1][x] != 2 {
            visited.insert((y + 1, x));
            parent.insert((y + 1, x), (y, x));
            queue.push_back((y + 1, x));
        }

        if x > 0 && !visited.contains(&(y, x - 1)) && maze[y][x - 1] != 2 {
            visited.insert((y, x - 1));
            parent.insert((y, x - 1), (y, x));
            queue.push_back((y, x - 1));
        }

        if y > 0 && !visited.contains(&(y - 1, x)) && maze[y - 1][x] != 2 {
            visited.insert((y - 1, x));
            parent.insert((y - 1, x), (y, x));
            queue.push_back((y - 1, x));
        }
    }

    None
}
