use std::io::{self, BufRead};
use std::f64;

#[derive(Clone)]
struct Node {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let n: usize = parts[0].parse().unwrap();
    let m: usize = parts[1].parse().unwrap();

    let mut edges = Vec::new();
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let u: usize = parts[0].parse().unwrap();
        let v: usize = parts[1].parse().unwrap();
        if u != v {
            edges.push((u, v));
        }
    }

    let mut nodes = vec![
        Node {
            x: 0.0,
            y: 0.0,
            vx: 0.0,
            vy: 0.0,
        };
        n
    ];

    if n == 0 {
        return;
    }

    initialize_layout(&mut nodes, &edges, n);

    if n <= 100 {
        force_directed(&mut nodes, &edges, 500, 0.01);
    } else if n <= 1000 {
        force_directed(&mut nodes, &edges, 200, 0.005);
    } else if n <= 10000 {
        force_directed(&mut nodes, &edges, 50, 0.002);
    } else {
        circular_layout_with_components(&mut nodes, &edges, n);
    }

    for node in &nodes {
        println!("{} {}", node.x, node.y);
    }
}

fn initialize_layout(nodes: &mut [Node], edges: &[(usize, usize)], n: usize) {
    let mut rng_state = 12345u64;

    for i in 0..n {
        rng_state = rng_state.wrapping_mul(1103515245).wrapping_add(12345);
        let angle = (rng_state as f64 / u64::MAX as f64) * 2.0 * std::f64::consts::PI;
        let radius = if edges.is_empty() {
            ((i as f64) * 0.5).sqrt() * 10.0
        } else {
            ((n as f64).sqrt()) * 5.0
        };
        nodes[i].x = angle.cos() * radius;
        nodes[i].y = angle.sin() * radius;
    }
}

fn force_directed(nodes: &mut [Node], edges: &[(usize, usize)], iterations: usize, damping: f64) {
    let n = nodes.len();
    let ideal_length = (n as f64).sqrt() * 2.0;
    let repulsion = 100.0;

    for _ in 0..iterations {
        for i in 0..n {
            nodes[i].vx = 0.0;
            nodes[i].vy = 0.0;
        }

        for i in 0..n {
            for j in (i + 1)..n {
                let dx = nodes[j].x - nodes[i].x;
                let dy = nodes[j].y - nodes[i].y;
                let dist_sq = dx * dx + dy * dy + 0.01;
                let dist = dist_sq.sqrt();

                let force = repulsion / dist_sq;
                let fx = (dx / dist) * force;
                let fy = (dy / dist) * force;

                nodes[i].vx -= fx;
                nodes[i].vy -= fy;
                nodes[j].vx += fx;
                nodes[j].vy += fy;
            }
        }

        for &(u, v) in edges {
            let dx = nodes[v].x - nodes[u].x;
            let dy = nodes[v].y - nodes[u].y;
            let dist_sq = dx * dx + dy * dy + 0.01;
            let dist = dist_sq.sqrt();

            let force = (dist - ideal_length) * 0.1;
            let fx = (dx / dist) * force;
            let fy = (dy / dist) * force;

            nodes[u].vx += fx;
            nodes[u].vy += fy;
            nodes[v].vx -= fx;
            nodes[v].vy -= fy;
        }

        for node in nodes.iter_mut() {
            node.vx *= damping;
            node.vy *= damping;
            node.x += node.vx;
            node.y += node.vy;
        }
    }
}

fn circular_layout_with_components(nodes: &mut [Node], edges: &[(usize, usize)], n: usize) {
    let mut visited = vec![false; n];
    let mut component_id = vec![0; n];
    let mut comp_count = 0;

    for start in 0..n {
        if !visited[start] {
            let mut stack = vec![start];
            while let Some(u) = stack.pop() {
                if visited[u] {
                    continue;
                }
                visited[u] = true;
                component_id[u] = comp_count;

                for &(a, b) in edges {
                    let neighbor = if a == u { b } else if b == u { a } else { continue };
                    if !visited[neighbor] {
                        stack.push(neighbor);
                    }
                }
            }
            comp_count += 1;
        }
    }

    let mut comp_nodes = vec![Vec::new(); comp_count];
    for i in 0..n {
        comp_nodes[component_id[i]].push(i);
    }

    let comp_radius = 100.0;
    let angle_per_comp = 2.0 * std::f64::consts::PI / comp_count.max(1) as f64;

    for (comp_idx, node_list) in comp_nodes.iter().enumerate() {
        let comp_x = (comp_idx as f64 * angle_per_comp).cos() * comp_radius;
        let comp_y = (comp_idx as f64 * angle_per_comp).sin() * comp_radius;

        let node_count = node_list.len();
        let node_radius = (node_count as f64).sqrt() * 2.0;

        for (local_idx, &node_id) in node_list.iter().enumerate() {
            let angle = 2.0 * std::f64::consts::PI * local_idx as f64 / node_count.max(1) as f64;
            nodes[node_id].x = comp_x + angle.cos() * node_radius;
            nodes[node_id].y = comp_y + angle.sin() * node_radius;
        }
    }
}
