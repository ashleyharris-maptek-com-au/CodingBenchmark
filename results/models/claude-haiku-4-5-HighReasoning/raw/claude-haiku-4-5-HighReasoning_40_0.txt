#include <iostream>
#include <vector>

using namespace std;

int main() {
    int n, m;
    cin >> n >> m;

    vector<int> c(n);
    for (int i = 0; i < n; i++) {
        cin >> c[i];
    }

    vector<pair<int, int>> constraints(m);
    for (int i = 0; i < m; i++) {
        cin >> constraints[i].first >> constraints[i].second;
    }

    int best_value = 0;
    long long best_mask = 0;

    // Try all 2^n subsets
    for (long long mask = 0; mask < (1LL << n); mask++) {
        // Calculate value
        int value = 0;
        for (int i = 0; i < n; i++) {
            if (mask & (1LL << i)) {
                value += c[i];
            }
        }

        // Check constraints
        bool valid = true;
        for (const auto& [u, v] : constraints) {
            if ((mask & (1LL << u)) && (mask & (1LL << v))) {
                valid = false;
                break;
            }
        }

        if (valid && value > best_value) {
            best_value = value;
            best_mask = mask;
        }
    }

    cout << best_value << "\n";
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << " ";
        cout << ((best_mask >> i) & 1);
    }
    cout << "\n";

    return 0;
}
