%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float

%fn_void_type = OpTypeFunction %void

%in_worldPos = OpVariable %ptr_Input_v3float Input
%in_normal = OpVariable %ptr_Input_v3float Input
%in_uv = OpVariable %ptr_Input_v2float Input
%in_tangent = OpVariable %ptr_Input_v3float Input
%in_color = OpVariable %ptr_Input_v3float Input
%out_color = OpVariable %ptr_Output_v4float Output
%ubo_instance = OpVariable %ptr_Uniform_UBO Uniform

; Constants
%const_0 = OpConstant %int 0
%const_3 = OpConstant %int 3

%f_0_0 = OpConstant %float 0.0
%f_0_3 = OpConstant %float 0.3
%f_0_4 = OpConstant %float 0.4
%f_0_5 = OpConstant %float 0.5
%f_0_6 = OpConstant %float 0.6
%f_0_15 = OpConstant %float 0.15
%f_0_05 = OpConstant %float 0.05
%f_1_0 = OpConstant %float 1.0

%sky_color = OpConstantComposite %v3float %f_0_4 %f_0_6 %f_1_0
%ground_color = OpConstantComposite %v3float %f_0_3 %f_0_15 %f_0_05

; Main function
%main = OpFunction %void None %fn_void_type
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %v3float %in_worldPos
%normal_input = OpLoad %v3float %in_normal
%uv = OpLoad %v2float %in_uv
%tangent = OpLoad %v3float %in_tangent
%color = OpLoad %v3float %in_color

; Normalize the normal (GLSL instruction 69: Normalize)
%normal = OpExtInst %v3float %glsl 69 %normal_input

; Get light position from UBO (member 3)
%lightPos_ptr = OpAccessChain %ptr_Uniform_v4float %ubo_instance %const_3
%lightPos_v4 = OpLoad %v4float %lightPos_ptr

; Extract xyz from lightPos
%lightPos = OpVectorShuffle %v3float %lightPos_v4 %lightPos_v4 0 1 2

; Calculate light direction vector (from surface to light)
%light_diff = OpFSub %v3float %lightPos %worldPos
%lightDir = OpExtInst %v3float %glsl 69 %light_diff

; Diffuse calculation: 0.3 * max(dot(N, L), 0)
%dot_nl = OpDot %float %normal %lightDir
%diffuse_coeff = OpExtInst %float %glsl 40 %dot_nl %f_0_0

; Scale color by 0.3 and then by diffuse coefficient
%diffuse_scaled = OpVectorTimesScalar %v3float %color %f_0_3
%diffuse_final = OpVectorTimesScalar %v3float %diffuse_scaled %diffuse_coeff

; Hemisphere ambient lighting
; Get normal.y component (index 1)
%normal_y = OpCompositeExtract %float %normal 1

; Calculate blend factor: N.y * 0.5 + 0.5
%blend_1 = OpFMul %float %normal_y %f_0_5
%blend = OpFAdd %float %blend_1 %f_0_5

; Clamp blend to [0, 1] (GLSL instruction 46: FClamp)
%blend_clamped = OpExtInst %float %glsl 46 %blend %f_0_0 %f_1_0

; Mix colors: lerp(ground, sky, blend)
; = ground * (1 - blend) + sky * blend
%one_minus_blend = OpFSub %float %f_1_0 %blend_clamped
%ground_contrib = OpVectorTimesScalar %v3float %ground_color %one_minus_blend
%sky_contrib = OpVectorTimesScalar %v3float %sky_color %blend_clamped
%hemisphere = OpFAdd %v3float %ground_contrib %sky_contrib

; Final color = hemisphere + diffuse
%final_rgb = OpFAdd %v3float %hemisphere %diffuse_final

; Create output vec4 with alpha = 1.0
%final_color = OpCompositeConstruct %v4float %final_rgb %f_1_0

; Store to output
OpStore %out_color %final_color

OpReturn
OpFunctionEnd
