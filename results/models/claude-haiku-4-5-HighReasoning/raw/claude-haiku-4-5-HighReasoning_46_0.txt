#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint gid = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against out of bounds invocations
    if (gid >= N) return;

    // Extract physics parameters
    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    // Read input data for this body
    // Body i: inData[i*2] = vec4(position.xyz, mass)
    //         inData[i*2+1] = vec4(velocity.xyz, 0.0)
    vec3 pos_i = inData[gid * 2].xyz;
    float mass_i = inData[gid * 2].w;
    vec3 vel_i = inData[gid * 2 + 1].xyz;

    // Compute acceleration from all other bodies
    vec3 acc = vec3(0.0);

    for (uint j = 0; j < N; ++j) {
        // Skip self-interaction
        if (j == gid) continue;

        vec3 pos_j = inData[j * 2].xyz;
        float mass_j = inData[j * 2].w;

        // Vector from body i to body j
        vec3 delta = pos_j - pos_i;
        float dist_sq = dot(delta, delta);

        // Compute (|pos_j - pos_i|^2 + softening^2)^(3/2)
        float r2_plus_soft = dist_sq + softening * softening;
        float denominator = r2_plus_soft * sqrt(r2_plus_soft);

        // Acceleration contribution: G * mass_j * (pos_j - pos_i) / (r^2 + softening^2)^(3/2)
        acc += G * mass_j * delta / denominator;
    }

    // Advance velocity by acceleration * dt (Euler method)
    vec3 new_vel = vel_i + acc * dt;

    // Advance position by velocity * dt
    vec3 new_pos = pos_i + new_vel * dt;

    // Write output data for this body
    outData[gid * 2] = vec4(new_pos, mass_i);
    outData[gid * 2 + 1] = vec4(new_vel, 0.0);
}
