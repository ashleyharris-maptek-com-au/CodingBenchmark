[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal : TEXCOORD1;
    [[vk::location(2)]] float2 uv : TEXCOORD2;
    [[vk::location(3)]] float3 tangent : TEXCOORD3;
    [[vk::location(4)]] float3 color : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0 {
    // Procedural brick pattern parameters
    float scaleX = 8.0;
    float scaleY = 16.0;
    float mortarThicknessX = 0.05;
    float mortarThicknessY = 0.1;

    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor = float3(0.7, 0.2, 0.1);

    // Scale UV coordinates
    float2 scaledUV = input.uv * float2(scaleX, scaleY);

    // Staggered rows: offset every other row by half a brick width
    float row = floor(scaledUV.y);
    float isOddRow = fmod(row, 2.0);
    float offset = isOddRow * 0.5;

    float2 brickUV = float2(scaledUV.x + offset, scaledUV.y);

    // Get fractional part for determining position within brick
    float2 localUV = frac(brickUV);

    // Determine if current pixel is mortar or brick
    bool isMortarX = (localUV.x < mortarThicknessX) || (localUV.x > (1.0 - mortarThicknessX));
    bool isMortarY = (localUV.y < mortarThicknessY) || (localUV.y > (1.0 - mortarThicknessY));
    bool isMortar = isMortarX || isMortarY;

    // Choose color based on mortar or brick
    float3 pixelColor = isMortar ? mortarColor : brickColor;

    // Calculate Lambertian shading
    float3 normal = normalize(input.normal);
    float3 lightDir = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(0.0, dot(normal, lightDir));
    float shadingFactor = 0.4 + 0.6 * diffuse;

    // Apply shading
    float3 finalColor = pixelColor * shadingFactor;

    return float4(finalColor, 1.0);
}
