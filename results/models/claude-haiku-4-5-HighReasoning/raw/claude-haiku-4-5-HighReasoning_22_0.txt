using System;
using System.Collections.Generic;
using System.Linq;

class TetrisPlayer
{
    private int width;
    private int height;
    private bool[,] board;

    private struct Pos { public int x, y; }

    private static readonly Dictionary<char, Pos[][]> Pieces = new()
    {
        { 'I', new[] { new[] { new Pos { x=0, y=0 }, new Pos { x=1, y=0 }, new Pos { x=2, y=0 }, new Pos { x=3, y=0 } }, new[] { new Pos { x=0, y=0 }, new Pos { x=0, y=1 }, new Pos { x=0, y=2 }, new Pos { x=0, y=3 } } } },
        { 'O', new[] { new[] { new Pos { x=0, y=0 }, new Pos { x=1, y=0 }, new Pos { x=0, y=1 }, new Pos { x=1, y=1 } } } },
        { 'T', new[] {
            new[] { new Pos { x=1, y=0 }, new Pos { x=0, y=1 }, new Pos { x=1, y=1 }, new Pos { x=2, y=1 } },
            new[] { new Pos { x=0, y=0 }, new Pos { x=0, y=1 }, new Pos { x=1, y=1 }, new Pos { x=0, y=2 } },
            new[] { new Pos { x=0, y=1 }, new Pos { x=1, y=1 }, new Pos { x=2, y=1 }, new Pos { x=1, y=2 } },
            new[] { new Pos { x=1, y=0 }, new Pos { x=0, y=1 }, new Pos { x=1, y=1 }, new Pos { x=1, y=2 } }
        } },
        { 'S', new[] {
            new[] { new Pos { x=1, y=0 }, new Pos { x=2, y=0 }, new Pos { x=0, y=1 }, new Pos { x=1, y=1 } },
            new[] { new Pos { x=0, y=0 }, new Pos { x=0, y=1 }, new Pos { x=1, y=1 }, new Pos { x=1, y=2 } }
        } },
        { 'Z', new[] {
            new[] { new Pos { x=0, y=0 }, new Pos { x=1, y=0 }, new Pos { x=1, y=1 }, new Pos { x=2, y=1 } },
            new[] { new Pos { x=1, y=0 }, new Pos { x=0, y=1 }, new Pos { x=1, y=1 }, new Pos { x=0, y=2 } }
        } },
        { 'J', new[] {
            new[] { new Pos { x=0, y=0 }, new Pos { x=0, y=1 }, new Pos { x=1, y=1 }, new Pos { x=2, y=1 } },
            new[] { new Pos { x=0, y=0 }, new Pos { x=1, y=0 }, new Pos { x=0, y=1 }, new Pos { x=0, y=2 } },
            new[] { new Pos { x=0, y=1 }, new Pos { x=1, y=1 }, new Pos { x=2, y=1 }, new Pos { x=2, y=0 } },
            new[] { new Pos { x=1, y=0 }, new Pos { x=1, y=1 }, new Pos { x=0, y=2 }, new Pos { x=1, y=2 } }
        } },
        { 'L', new[] {
            new[] { new Pos { x=2, y=0 }, new Pos { x=0, y=1 }, new Pos { x=1, y=1 }, new Pos { x=2, y=1 } },
            new[] { new Pos { x=0, y=0 }, new Pos { x=0, y=1 }, new Pos { x=0, y=2 }, new Pos { x=1, y=2 } },
            new[] { new Pos { x=0, y=1 }, new Pos { x=1, y=1 }, new Pos { x=2, y=1 }, new Pos { x=0, y=0 } },
            new[] { new Pos { x=0, y=0 }, new Pos { x=1, y=0 }, new Pos { x=1, y=1 }, new Pos { x=1, y=2 } }
        } }
    };

    static void Main()
    {
        var line = Console.ReadLine().Split();
        int width = int.Parse(line[0]);
        int height = int.Parse(line[1]);
        int numPieces = int.Parse(line[2]);

        var player = new TetrisPlayer(width, height);

        for (int i = 0; i < numPieces; i++)
        {
            char pieceType = char.Parse(Console.ReadLine().Trim());
            var (rotation, column) = player.FindBestPlacement(pieceType);
            Console.WriteLine($"{rotation} {column}");
            Console.Out.Flush();

            string result = Console.ReadLine().Trim();
            player.ApplyPlacement(pieceType, rotation, column);

            if (result.StartsWith("cleared"))
            {
                int clearedLines = int.Parse(result.Split('_')[1]);
                player.ClearLines(clearedLines);
            }
            else if (result == "gameover")
            {
                break;
            }
        }
    }

    TetrisPlayer(int w, int h)
    {
        width = w;
        height = h;
        board = new bool[height, width];
    }

    (int rotation, int column) FindBestPlacement(char pieceType)
    {
        int bestRotation = 0;
        int bestColumn = 0;
        int bestScore = int.MinValue;

        var rotations = Pieces[pieceType];

        for (int rot = 0; rot < rotations.Length; rot++)
        {
            var shape = rotations[rot];
            int maxX = shape.Max(p => p.x);

            for (int col = 0; col <= width - maxX - 1; col++)
            {
                int dropHeight = FindDropHeight(shape, col);
                if (dropHeight >= 0)
                {
                    int score = EvaluatePlacement(shape, col, dropHeight);
                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestRotation = rot;
                        bestColumn = col;
                    }
                }
            }
        }

        return (bestRotation, bestColumn);
    }

    int FindDropHeight(Pos[] shape, int col)
    {
        int maxDrop = height - 1;

        foreach (var pos in shape)
        {
            int blockCol = col + pos.x;

            if (blockCol < 0 || blockCol >= width)
                return -1;

            int landRow = height - 1 - pos.y;
            for (int row = height - 1; row >= 0; row--)
            {
                if (board[row, blockCol])
                {
                    landRow = row - 1 - pos.y;
                    break;
                }
            }

            maxDrop = Math.Min(maxDrop, landRow);
        }

        return Math.Max(0, maxDrop);
    }

    int EvaluatePlacement(Pos[] shape, int col, int dropRow)
    {
        int score = 0;
        int maxHeight = 0;
        int holes = 0;

        for (int c = 0; c < width; c++)
        {
            int colHeight = 0;
            bool foundBlock = false;
            int gapCount = 0;
            bool inGap = false;

            for (int r = 0; r < height; r++)
            {
                bool hasBlock = board[r, c] || shape.Any(p => col + p.x == c && dropRow + p.y == r);

                if (hasBlock)
                {
                    foundBlock = true;
                    if (inGap) gapCount++;
                    colHeight = r + 1;
                    inGap = false;
                }
                else if (foundBlock)
                {
                    inGap = true;
                }
            }

            maxHeight = Math.Max(maxHeight, colHeight);
            holes += gapCount;
        }

        score += (height - maxHeight) * 10;
        score -= holes * 50;

        return score;
    }

    void ApplyPlacement(char pieceType, int rotation, int column)
    {
        var shape = Pieces[pieceType][rotation];
        int dropHeight = FindDropHeight(shape, column);

        foreach (var pos in shape)
        {
            int r = dropHeight + pos.y;
            int c = column + pos.x;
            if (r >= 0 && r < height && c >= 0 && c < width)
                board[r, c] = true;
        }
    }

    void ClearLines(int count)
    {
        for (int i = 0; i < count; i++)
        {
            for (int r = height - 1; r > 0; r--)
            {
                bool full = true;
                for (int c = 0; c < width; c++)
                {
                    if (!board[r, c])
                    {
                        full = false;
                        break;
                    }
                }

                if (full)
                {
                    for (int rr = r; rr > 0; rr--)
                    {
                        for (int c = 0; c < width; c++)
                        {
                            board[rr, c] = board[rr - 1, c];
                        }
                    }
                    for (int c = 0; c < width; c++)
                    {
                        board[0, c] = false;
                    }
                    break;
                }
            }
        }
    }
}
