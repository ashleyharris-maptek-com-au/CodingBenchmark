use std::io::{self, BufReader, BufWriter, Read, Write};
use std::collections::HashMap;

const MAX_ITERATIONS: usize = 100;
const CONVERGENCE_THRESHOLD: f32 = 1e-4;
const BATCH_SIZE: usize = 100_000;
const SMALL_THRESHOLD: usize = 10_000_000;
const MEDIUM_THRESHOLD: usize = 100_000_000;
const LARGE_THRESHOLD: usize = 1_000_000_000;

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

impl Point {
    fn distance_squared(&self, other: &Point) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }

    fn add(&self, other: &Point) -> Point {
        Point {
            x: self.x + other.x,
            y: self.y + other.y,
            z: self.z + other.z,
        }
    }

    fn scale(&self, factor: f32) -> Point {
        Point {
            x: self.x * factor,
            y: self.y * factor,
            z: self.z * factor,
        }
    }
}

fn read_f32(data: &[u8], idx: usize) -> f32 {
    let bytes: [u8; 4] = [data[idx], data[idx + 1], data[idx + 2], data[idx + 3]];
    f32::from_le_bytes(bytes)
}

fn read_u32(data: &[u8], idx: usize) -> u32 {
    let bytes: [u8; 4] = [data[idx], data[idx + 1], data[idx + 2], data[idx + 3]];
    u32::from_le_bytes(bytes)
}

fn write_u32(data: &mut Vec<u8>, value: u32) {
    data.extend_from_slice(&value.to_le_bytes());
}

fn find_nearest_centroid(point: &Point, centroids: &[Point]) -> usize {
    let mut min_dist = f32::INFINITY;
    let mut nearest = 0;
    for (i, centroid) in centroids.iter().enumerate() {
        let dist = point.distance_squared(centroid);
        if dist < min_dist {
            min_dist = dist;
            nearest = i;
        }
    }
    nearest
}

fn kmeans_standard(
    num_points: u32,
    num_clusters: u32,
    reader: &mut BufReader<io::Stdin>,
) -> io::Result<Vec<u32>> {
    let nc = num_clusters as usize;
    let mut points = Vec::new();
    let mut buffer = vec![0u8; 4096];

    loop {
        match reader.read(&mut buffer)? {
            0 => break,
            n => {
                for i in (0..n).step_by(12) {
                    if i + 12 <= n {
                        let x = read_f32(&buffer, i);
                        let y = read_f32(&buffer, i + 4);
                        let z = read_f32(&buffer, i + 8);
                        points.push(Point { x, y, z });
                    }
                }
            }
        }
    }

    let mut centroids = initialize_centroids(&points, nc);
    let mut assignments = vec![0usize; points.len()];

    for _iter in 0..MAX_ITERATIONS {
        let mut new_assignments = vec![0usize; points.len()];
        for (i, point) in points.iter().enumerate() {
            new_assignments[i] = find_nearest_centroid(point, &centroids);
        }

        let new_centroids = compute_centroids(&points, &new_assignments, nc);
        let converged = centroids_converged(&centroids, &new_centroids);
        centroids = new_centroids;
        assignments = new_assignments;

        if converged {
            break;
        }
    }

    Ok(assignments.iter().map(|&x| x as u32).collect())
}

fn kmeans_minibatch(
    num_points: u32,
    num_clusters: u32,
    reader: &mut BufReader<io::Stdin>,
) -> io::Result<Vec<u32>> {
    let nc = num_clusters as usize;
    let mut centroids = vec![
        Point {
            x: 0.0,
            y: 0.0,
            z: 0.0
        };
        nc
    ];
    let mut centroid_counts = vec![0u32; nc];
    let mut assignments = Vec::new();
    let mut batch = Vec::new();
    let mut buffer = vec![0u8; 65536];
    let mut point_idx = 0u32;

    loop {
        batch.clear();
        match reader.read(&mut buffer)? {
            0 => break,
            n => {
                for i in (0..n).step_by(12) {
                    if i + 12 <= n {
                        let x = read_f32(&buffer, i);
                        let y = read_f32(&buffer, i + 4);
                        let z = read_f32(&buffer, i + 8);
                        batch.push(Point { x, y, z });
                    }
                }
            }
        }

        if batch.is_empty() {
            break;
        }

        if centroids[0].x == 0.0 && centroids[0].y == 0.0 && centroids[0].z == 0.0 {
            for (i, point) in batch.iter().enumerate().take(nc) {
                centroids[i] = *point;
                centroid_counts[i] = 1;
            }
        }

        for point in &batch {
            let cluster = find_nearest_centroid(point, &centroids);
            assignments.push(cluster as u32);

            centroid_counts[cluster] += 1;
            let count = centroid_counts[cluster] as f32;
            let lr = 1.0 / count;

            centroids[cluster] = centroids[cluster]
                .scale(1.0 - lr)
                .add(&point.scale(lr));
            point_idx += 1;
        }
    }

    Ok(assignments)
}

fn kmeans_streaming(
    num_points: u32,
    num_clusters: u32,
    reader: &mut BufReader<io::Stdin>,
) -> io::Result<Vec<u32>> {
    let nc = num_clusters as usize;
    let mut centroids = vec![
        Point {
            x: 0.0,
            y: 0.0,
            z: 0.0
        };
        nc
    ];
    let mut centroid_counts = vec![0u32; nc];
    let mut assignments = Vec::new();
    let mut buffer = vec![0u8; 1_000_000];
    let mut centroid_idx = 0;
    let mut point_count = 0u32;

    loop {
        match reader.read(&mut buffer)? {
            0 => break,
            n => {
                for i in (0..n).step_by(12) {
                    if i + 12 <= n {
                        let x = read_f32(&buffer, i);
                        let y = read_f32(&buffer, i + 4);
                        let z = read_f32(&buffer, i + 8);
                        let point = Point { x, y, z };

                        if centroid_idx < nc {
                            centroids[centroid_idx] = point;
                            centroid_counts[centroid_idx] = 1;
                            assignments.push(centroid_idx as u32);
                            centroid_idx += 1;
                        } else {
                            let cluster = find_nearest_centroid(&point, &centroids);
                            assignments.push(cluster as u32);

                            centroid_counts[cluster] += 1;
                            let count = centroid_counts[cluster] as f32;
                            let lr = 0.1 / (1.0 + (count / 1000.0));

                            centroids[cluster] = centroids[cluster]
                                .scale(1.0 - lr)
                                .add(&point.scale(lr));
                        }

                        point_count += 1;
                    }
                }
            }
        }
    }

    Ok(assignments)
}

fn initialize_centroids(points: &[Point], k: usize) -> Vec<Point> {
    let mut centroids = Vec::with_capacity(k);
    if points.len() >= k {
        let step = points.len() / k;
        for i in 0..k {
            centroids.push(points[i * step]);
        }
    } else {
        for point in points.iter().take(k) {
            centroids.push(*point);
        }
    }
    centroids
}

fn compute_centroids(
    points: &[Point],
    assignments: &[usize],
    k: usize,
) -> Vec<Point> {
    let mut sums = vec![
        Point {
            x: 0.0,
            y: 0.0,
            z: 0.0
        };
        k
    ];
    let mut counts = vec![0u32; k];

    for (point, &cluster) in points.iter().zip(assignments.iter()) {
        sums[cluster] = sums[cluster].add(point);
        counts[cluster] += 1;
    }

    sums.iter_mut().zip(counts.iter()).for_each(|(sum, &count)| {
        if count > 0 {
            *sum = sum.scale(1.0 / count as f32);
        }
    });

    sums
}

fn centroids_converged(old: &[Point], new: &[Point]) -> bool {
    old.iter()
        .zip(new.iter())
        .all(|(o, n)| o.distance_squared(n) < CONVERGENCE_THRESHOLD)
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin);
    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());

    let mut header = [0u8; 8];
    reader.read_exact(&mut header)?;

    let num_points = read_u32(&header, 0);
    let num_clusters = read_u32(&header, 4);

    let assignments = if num_points <= SMALL_THRESHOLD as u32 {
        kmeans_standard(num_points, num_clusters, &mut reader)?
    } else if num_points <= MEDIUM_THRESHOLD as u32 {
        kmeans_minibatch(num_points, num_clusters, &mut reader)?
    } else {
        kmeans_streaming(num_points, num_clusters, &mut reader)?
    };

    for &cluster_idx in &assignments {
        writer.write_all(&cluster_idx.to_le_bytes())?;
    }

    writer.flush()?;
    Ok(())
}
