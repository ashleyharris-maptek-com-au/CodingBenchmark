OpCapability Shader
OpExtension "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_color
OpExecutionMode %main OriginUpperLeft

%glsl = OpExtInstImport "GLSL.std.450"

; Decorations
OpDecorate %out_color Location 0
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0 ColMajor MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64 ColMajor MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128 ColMajor MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Type definitions
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; UBO structure
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

; Pointer types
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4

; Variables
%ubo = OpVariable %ptr_Uniform_UBO Uniform
%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2 Input
%in_tangent = OpVariable %ptr_Input_vec3 Input
%in_color = OpVariable %ptr_Input_vec3 Input
%out_color = OpVariable %ptr_Output_vec4 Output

; Float constants
%f0 = OpConstant %float 0.0
%f1 = OpConstant %float 1.0
%f0_2 = OpConstant %float 0.2
%f0_3 = OpConstant %float 0.3
%f0_7 = OpConstant %float 0.7
%f0_8 = OpConstant %float 0.8
%f1_5 = OpConstant %float 1.5
%f3_0 = OpConstant %float 3.0

; Int constants
%i0 = OpConstant %int 0
%i3 = OpConstant %int 3
%i4 = OpConstant %int 4

; Constant color vectors
%base_color = OpConstantComposite %vec3 %f0_8 %f0_2 %f0_2
%fog_color = OpConstantComposite %vec3 %f0_7 %f0_7 %f0_8

; Function type
%fn_void = OpTypeFunction %void

; Main function
%main = OpFunction %void None %fn_void
%bb_entry = OpLabel

; Load input worldPos and normal
%worldPos = OpLoad %vec3 %in_worldPos
%normal = OpLoad %vec3 %in_normal

; Access camera position from UBO (member index 4)
%ptr_cam = OpAccessChain %ptr_Uniform_vec4 %ubo %i4
%cam_v4 = OpLoad %vec4 %ptr_cam
%cam_pos = OpVectorShuffle %vec3 %cam_v4 %cam_v4 0 1 2

; Access light position from UBO (member index 3)
%ptr_light = OpAccessChain %ptr_Uniform_vec4 %ubo %i3
%light_v4 = OpLoad %vec4 %ptr_light
%light_pos = OpVectorShuffle %vec3 %light_v4 %light_v4 0 1 2

; Calculate distance from camera to fragment
%vec_to_cam = OpFSub %vec3 %worldPos %cam_pos
%dist = OpExtInst %float %glsl 1 %vec_to_cam

; Calculate fog factor
%dist_sub = OpFSub %float %dist %f1_5
%dist_div = OpFDiv %float %dist_sub %f3_0
%fog_factor = OpExtInst %float %glsl 43 %dist_div %f0 %f1

; Normalize normal vector
%norm_norm = OpExtInst %vec3 %glsl 69 %normal

; Calculate light direction
%vec_to_light = OpFSub %vec3 %light_pos %worldPos
%light_dir = OpExtInst %vec3 %glsl 69 %vec_to_light

; Calculate diffuse (dot product)
%diffuse = OpDot %float %norm_norm %light_dir

; Clamp diffuse to [0, 1]
%diffuse_clamped = OpExtInst %float %glsl 40 %diffuse %f0

; Calculate Lambertian factor: 0.3 + 0.7 * diffuse
%diffuse_scaled = OpFMul %float %f0_7 %diffuse_clamped
%lambertian_factor = OpFAdd %float %f0_3 %diffuse_scaled

; Apply lambertian to base color
%shaded = OpVectorTimesScalar %vec3 %base_color %lambertian_factor

; Calculate blend factor (1 - fog_factor)
%blend_factor = OpFSub %float %f1 %fog_factor

; Blend shaded color and fog color
%shaded_scaled = OpVectorTimesScalar %vec3 %shaded %blend_factor
%fog_scaled = OpVectorTimesScalar %vec3 %fog_color %fog_factor
%final_rgb = OpFAdd %vec3 %shaded_scaled %fog_scaled

; Create final output color with alpha = 1.0
%final_color = OpCompositeConstruct %vec4 %final_rgb %f1

; Store to output
OpStore %out_color %final_color

OpReturn
OpFunctionEnd
