using System;
using System.Diagnostics;
using System.Linq;

class QAPSolver
{
    private int n;
    private int[][] flow;
    private int[][] distance;
    private int[] bestPerm;
    private long bestCost;
    private Stopwatch timer;
    private const long TIME_LIMIT_MS = 290000; // 290 seconds (5 min - 10 sec buffer)

    static void Main()
    {
        var solver = new QAPSolver();
        solver.Solve();
    }

    void Solve()
    {
        timer = Stopwatch.StartNew();

        // Read input
        n = int.Parse(Console.ReadLine());
        flow = ReadMatrix();
        distance = ReadMatrix();

        // Initialize with identity permutation
        bestPerm = Enumerable.Range(0, n).ToArray();
        bestCost = CalculateCost(bestPerm);

        // Apply 2-opt local search with time limit
        LocalSearch();

        // Output
        Console.WriteLine(bestCost);
        Console.WriteLine(string.Join(" ", bestPerm));
    }

    void LocalSearch()
    {
        bool improved = true;

        while (improved && timer.ElapsedMilliseconds < TIME_LIMIT_MS)
        {
            improved = false;

            for (int i = 0; i < n - 1 && timer.ElapsedMilliseconds < TIME_LIMIT_MS; i++)
            {
                for (int j = i + 1; j < n && timer.ElapsedMilliseconds < TIME_LIMIT_MS; j++)
                {
                    // Calculate delta cost for swapping positions i and j
                    long delta = CalculateSwapDelta(i, j);

                    if (delta < 0)
                    {
                        // Perform swap
                        int temp = bestPerm[i];
                        bestPerm[i] = bestPerm[j];
                        bestPerm[j] = temp;
                        bestCost += delta;
                        improved = true;
                    }
                }
            }
        }
    }

    int[][] ReadMatrix()
    {
        int[][] matrix = new int[n][];
        for (int i = 0; i < n; i++)
        {
            matrix[i] = Console.ReadLine().Split().Select(int.Parse).ToArray();
        }
        return matrix;
    }

    long CalculateCost(int[] perm)
    {
        long cost = 0;
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                cost += (long)flow[i][j] * distance[perm[i]][perm[j]];
            }
        }
        return cost;
    }

    long CalculateSwapDelta(int posI, int posJ)
    {
        // Calculate the cost delta for swapping positions posI and posJ
        // bestPerm[posI] is the location assigned to facility posI

        int locI = bestPerm[posI];
        int locJ = bestPerm[posJ];

        long delta = 0;

        // Consider interactions with all other facilities
        for (int k = 0; k < n; k++)
        {
            if (k == posI || k == posJ) continue;

            int locK = bestPerm[k];

            // Facility posI changes location from locI to locJ
            delta += (long)flow[posI][k] * (distance[locJ][locK] - distance[locI][locK]);
            delta += (long)flow[k][posI] * (distance[locK][locJ] - distance[locK][locI]);

            // Facility posJ changes location from locJ to locI
            delta += (long)flow[posJ][k] * (distance[locI][locK] - distance[locJ][locK]);
            delta += (long)flow[k][posJ] * (distance[locK][locI] - distance[locK][locJ]);
        }

        // Direct interaction between posI and posJ
        delta += (long)flow[posI][posJ] * (distance[locJ][locI] - distance[locI][locJ]);
        delta += (long)flow[posJ][posI] * (distance[locI][locJ] - distance[locJ][locI]);

        return delta;
    }
}
