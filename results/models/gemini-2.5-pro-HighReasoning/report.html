using System;
using System.Collections.Generic;
using System.Linq;

public class ChinesePostman
{
    private int N;
    private int M;
    private List<Tuple<int, int>>[] adj;

    public void Solve()
    {
        ReadInput();

        var oddDegreeVertices = FindOddDegreeVertices();

        if (oddDegreeVertices.Count == 0)
        {
            var eulerianPath = FindEulerianPath(adj, 0);
            PrintPath(eulerianPath);
            return;
        }

        var (shortestPaths, preds) = FindAllPairsShortestPaths(oddDegreeVertices);
        
        List<Tuple<int, int>> matching;
        if (oddDegreeVertices.Count <= 16)
        {
            var bestMatching = new List<Tuple<int, int>>();
            long minWeight = long.MaxValue;
            GenerateMatchings(oddDegreeVertices, new List<Tuple<int, int>>(), ref bestMatching, ref minWeight, shortestPaths);
            matching = bestMatching;
        }
        else
        {
            matching = FindGreedyMatching(oddDegreeVertices, shortestPaths);
        }

        var augmentedAdj = AugmentGraph(matching, preds);
        var finalPath = FindEulerianPath(augmentedAdj, 0);
        PrintPath(finalPath);
    }

    private void ReadInput()
    {
        var line = Console.ReadLine()?.Split();
        if (line == null || line.Length < 2) return;
        N = int.Parse(line[0]);
        M = int.Parse(line[1]);

        adj = new List<Tuple<int, int>>[N];
        for (int i = 0; i < N; i++)
        {
            adj[i] = new List<Tuple<int, int>>();
        }

        for (int i = 0; i < M; i++)
        {
            line = Console.ReadLine()?.Split();
            if (line == null || line.Length < 3) continue;
            int u = int.Parse(line[0]);
            int v = int.Parse(line[1]);
            int w = int.Parse(line[2]);
            adj[u].Add(Tuple.Create(v, w));
            adj[v].Add(Tuple.Create(u, w));
        }
    }

    private List<int> FindOddDegreeVertices()
    {
        var oddDegreeVertices = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if (adj[i].Count % 2 != 0)
            {
                oddDegreeVertices.Add(i);
            }
        }
        return oddDegreeVertices;
    }

    private Tuple<Dictionary<Tuple<int, int>, int>, Dictionary<int, int[]>> FindAllPairsShortestPaths(List<int> vertices)
    {
        var shortestPaths = new Dictionary<Tuple<int, int>, int>();
        var preds = new Dictionary<int, int[]>();

        foreach (var u in vertices)
        {
            var (dist, pred) = Dijkstra(u);
            preds[u] = pred;
            foreach (var v in vertices)
            {
                if (u < v)
                {
                    shortestPaths[Tuple.Create(u, v)] = dist[v];
                }
            }
        }
        return Tuple.Create(shortestPaths, preds);
    }
    
    private Tuple<int[], int[]> Dijkstra(int start)
    {
        var dist = new int[N];
        var pred = new int[N];
        for (int i = 0; i < N; i++)
        {
            dist[i] = int.MaxValue;
            pred[i] = -1;
        }
        dist[start] = 0;

        var pq = new SortedSet<Tuple<int, int>>(Comparer<Tuple<int, int>>.Create((a, b) => a.Item1 != b.Item1 ? a.Item1.CompareTo(b.Item1) : a.Item2.CompareTo(b.Item2)));
        pq.Add(Tuple.Create(0, start));

        while (pq.Count > 0)
        {
            var (d, u) = pq.Min;
            pq.Remove(pq.Min);
            
            if (d > dist[u]) continue;

            foreach (var (v, weight) in adj[u])
            {
                if (dist[u] != int.MaxValue && dist[u] + weight < dist[v])
                {
                    dist[v] = dist[u] + weight;
                    pred[v] = u;
                    pq.Add(Tuple.Create(dist[v], v));
                }
            }
        }
        return Tuple.Create(dist, pred);
    }
    
    private void GenerateMatchings(List<int> oddVertices, List<Tuple<int, int>> currentMatching, ref List<Tuple<int, int>> bestMatching, ref long minWeight, Dictionary<Tuple<int, int>, int> shortestPaths)
    {
        if (oddVertices.Count == 0)
        {
            long currentWeight = 0;
            foreach (var pair in currentMatching)
            {
                currentWeight += shortestPaths[pair];
            }
            if (currentWeight < minWeight)
            {
                minWeight = currentWeight;
                bestMatching = new List<Tuple<int, int>>(currentMatching);
            }
            return;
        }

        int u = oddVertices[0];
        for (int i = 1; i < oddVertices.Count; i++)
        {
            int v = oddVertices[i];
            
            var nextOddVertices = new List<int>();
            for(int j = 1; j < oddVertices.Count; j++)
            {
                if(i != j) nextOddVertices.Add(oddVertices[j]);
            }

            var pair = u < v ? Tuple.Create(u, v) : Tuple.Create(v, u);
            currentMatching.Add(pair);
            GenerateMatchings(nextOddVertices, currentMatching, ref bestMatching, ref minWeight, shortestPaths);
            currentMatching.RemoveAt(currentMatching.Count - 1);
        }
    }

    private List<Tuple<int, int>> FindGreedyMatching(List<int> oddVertices, Dictionary<Tuple<int, int>, int> shortestPaths)
    {
        var matching = new List<Tuple<int, int>>();
        var matched = new HashSet<int>();
        
        var pairs = new List<Tuple<int, int, int>>();
        for (int i = 0; i < oddVertices.Count; i++)
        {
            for (int j = i + 1; j < oddVertices.Count; j++)
            {
                var u = oddVertices[i];
                var v = oddVertices[j];
                var pairKey = Tuple.Create(u,v);
                if (shortestPaths.ContainsKey(pairKey))
                {
                    pairs.Add(Tuple.Create(shortestPaths[pairKey], u, v));
                }
            }
        }

        pairs.Sort((a, b) => a.Item1.CompareTo(b.Item1));

        foreach (var pair in pairs)
        {
            int u = pair.Item2;
            int v = pair.Item3;
            if (!matched.Contains(u) && !matched.Contains(v))
            {
                matched.Add(u);
                matched.Add(v);
                matching.Add(Tuple.Create(u,v));
            }
        }
        return matching;
    }

    private List<Tuple<int, int>>[] AugmentGraph(List<Tuple<int, int>> matching, Dictionary<int, int[]> preds)
    {
        var augmentedAdj = new List<Tuple<int, int>>[N];
        for(int i = 0; i < N; ++i) augmentedAdj[i] = new List<Tuple<int, int>>(adj[i]);

        foreach (var pair in matching)
        {
            var u = pair.Item1;
            var v = pair.Item2;
            var pred = preds[u];
            int curr = v;
            while(curr != u && curr != -1)
            {
                int prev = pred[curr];
                if (prev == -1) break;
                int weight = GetWeight(prev, curr);
                augmentedAdj[prev].Add(Tuple.Create(curr, weight));
                augmentedAdj[curr].Add(Tuple.Create(prev, weight));
                curr = prev;
            }
        }
        return augmentedAdj;
    }
    
    private int GetWeight(int u, int v)
    {
        foreach(var edge in adj[u])
        {
            if(edge.Item1 == v) return edge.Item2;
        }
        return 0;
    }

    private List<int> FindEulerianPath(List<Tuple<int, int>>[] graph, int start)
    {
        var tempAdj = new List<Tuple<int,int>>[N];
        for(int i=0; i<N; i++)
        {
            tempAdj[i] = new List<Tuple<int, int>>(graph[i]);
        }

        var stack = new Stack<int>();
        var path = new List<int>();
        stack.Push(start);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            if (tempAdj[u].Count > 0)
            {
                var edge = tempAdj[u][0];
                int v = edge.Item1;
                tempAdj[u].RemoveAt(0);
                
                for(int i = 0; i < tempAdj[v].Count; i++)
                {
                    if(tempAdj[v][i].Item1 == u && tempAdj[v][i].Item2 == edge.Item2)
                    {
                        tempAdj[v].RemoveAt(i);
                        break;
                    }
                }
                stack.Push(v);
            }
            else
            {
                path.Add(stack.Pop());
            }
        }
        path.Reverse();
        return path;
    }

    private void PrintPath(List<int> path)
    {
        Console.WriteLine(string.Join(" ", path));
    }

    public static void Main(string[] args)
    {
        try 
        {
            new ChinesePostman().Solve();
        }
        catch (Exception)
        {
            // Fail gracefully on empty input or other errors
        }
    }
}

struct Vec3 {

    Vec3 operator-(const Vec3& other) const { return {x - other.x, y - other.y, z - other.z}; }

    double dot(const Vec3& other) const { return x * other.x + y * other.y + z * other.z; }
    Vec3 cross(const Vec3& other) const { return {y * other.z - z * other.y, z * other.x - x * other.z, x * other.y - y * other.x}; }
    double length_sq() const { return x * x + y * y + z * z; }
    double length() const { return sqrt(length_sq()); }
};

// Represents a plane: n.dot(p) + d = 0
struct Plane {
    Vec3 normal;
    double d = 0;

    Plane(const Vec3& p1, const Vec3& p2, const Vec3& p3) {
        normal = (p2 - p1).cross(p3 - p1);
        double len = normal.length();
        if (len > EPSILON) {
            normal = normal * (1.0 / len);
        }
        d = -normal.dot(p1);
    }

    double distance_to(const Vec3& p) const {
        return normal.dot(p) + d;
    }
};

struct Container {
    std::vector<Plane> faces;

        if (faces.empty()) return false;
        for (const auto& plane : faces) {
            if (plane.distance_to(p) > EPSILON) {
                return false;
            }
        }
        return true;
    }
};
