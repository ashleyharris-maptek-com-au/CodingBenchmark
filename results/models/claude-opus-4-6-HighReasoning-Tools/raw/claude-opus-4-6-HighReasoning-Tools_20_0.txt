use std::io::{self, BufRead, Write, BufWriter};

struct Sample {
    hole_idx: usize,
    sample_idx: usize,
    x: f64,
    y: f64,
    z: f64,
    values: Vec<f64>,
}

struct Suspect {
    hole_id: String,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

struct KdNode {
    sample_idx: usize,
    split_dim: usize,
    left: Option<Box<KdNode>>,
    right: Option<Box<KdNode>>,
}

fn build_kdtree(indices: &mut [usize], samples: &[Sample], depth: usize) -> Option<Box<KdNode>> {
    if indices.is_empty() {
        return None;
    }
    let dim = depth % 3;
    indices.sort_unstable_by(|&a, &b| {
        let va = match dim { 0 => samples[a].x, 1 => samples[a].y, _ => samples[a].z };
        let vb = match dim { 0 => samples[b].x, 1 => samples[b].y, _ => samples[b].z };
        va.partial_cmp(&vb).unwrap_or(std::cmp::Ordering::Equal)
    });
    let mid = indices.len() / 2;
    let node_idx = indices[mid];
    let (left_slice, right_slice) = {
        let (left, rest) = indices.split_at_mut(mid);
        let right = if rest.len() > 1 { &mut rest[1..] } else { &mut [] as &mut [usize] };
        (left, right)
    };
    Some(Box::new(KdNode {
        sample_idx: node_idx,
        split_dim: dim,
        left: build_kdtree(left_slice, samples, depth + 1),
        right: build_kdtree(right_slice, samples, depth + 1),
    }))
}

fn knn_search(
    node: &Option<Box<KdNode>>,
    samples: &[Sample],
    target: &Sample,
    k: usize,
    neighbors: &mut Vec<(usize, f64)>,
    skip_idx: usize,
) {
    let node = match node {
        Some(n) => n,
        None => return,
    };
    let s = &samples[node.sample_idx];
    let dx = s.x - target.x;
    let dy = s.y - target.y;
    let dz = s.z - target.z;
    let dist_sq = dx * dx + dy * dy + dz * dz;

    if node.sample_idx != skip_idx {
        if neighbors.len() < k {
            neighbors.push((node.sample_idx, dist_sq));
            if neighbors.len() == k {
                neighbors.sort_unstable_by(|a, b| a.1.partial_cmp(&b.1).unwrap_or(std::cmp::Ordering::Equal));
            }
        } else if dist_sq < neighbors[k - 1].1 {
            neighbors[k - 1] = (node.sample_idx, dist_sq);
            neighbors.sort_unstable_by(|a, b| a.1.partial_cmp(&b.1).unwrap_or(std::cmp::Ordering::Equal));
        }
    }

    let target_val = match node.split_dim { 0 => target.x, 1 => target.y, _ => target.z };
    let node_val = match node.split_dim { 0 => s.x, 1 => s.y, _ => s.z };
    let diff = target_val - node_val;
    let diff_sq = diff * diff;

    let (first, second) = if diff < 0.0 {
        (&node.left, &node.right)
    } else {
        (&node.right, &node.left)
    };

    knn_search(first, samples, target, k, neighbors, skip_idx);
    if neighbors.len() < k || diff_sq < neighbors[k - 1].1 {
        knn_search(second, samples, target, k, neighbors, skip_idx);
    }
}

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();

    let first_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => { writeln!(out, "0").ok(); return; }
    };
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    if parts.len() < 2 {
        writeln!(out, "0").ok();
        return;
    }
    let num_holes: usize = parts[0].parse().unwrap_or(0);
    let num_properties: usize = parts[1].parse().unwrap_or(0);

    if num_holes == 0 || num_properties == 0 {
        writeln!(out, "0").ok();
        return;
    }

    let prop_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => { writeln!(out, "0").ok(); return; }
    };
    let property_names: Vec<String> = prop_line.split_whitespace().map(|s| s.to_string()).collect();

    let mut samples: Vec<Sample> = Vec::new();
    let mut hole_ids: Vec<String> = Vec::new();

    for _ in 0..num_holes {
        let hole_line = match lines.next() {
            Some(Ok(l)) => l,
            _ => break,
        };
        let hp: Vec<&str> = hole_line.split_whitespace().collect();
        if hp.len() < 9 { continue; }

        let hole_id = hp[0].to_string();
        let sx: f64 = hp[1].parse().unwrap_or(0.0);
        let sy: f64 = hp[2].parse().unwrap_or(0.0);
        let sz: f64 = hp[3].parse().unwrap_or(0.0);
        let dx: f64 = hp[4].parse().unwrap_or(0.0);
        let dy: f64 = hp[5].parse().unwrap_or(0.0);
        let dz: f64 = hp[6].parse().unwrap_or(0.0);
        let _length: f64 = hp[7].parse().unwrap_or(0.0);
        let n_samples: usize = hp[8].parse().unwrap_or(0);

        let hole_idx = hole_ids.len();
        hole_ids.push(hole_id);

        let dir_len = (dx * dx + dy * dy + dz * dz).sqrt();
        let (ndx, ndy, ndz) = if dir_len > 1e-12 {
            (dx / dir_len, dy / dir_len, dz / dir_len)
        } else {
            (0.0, 0.0, -1.0)
        };

        for si in 0..n_samples {
            let sample_line = match lines.next() {
                Some(Ok(l)) => l,
                _ => break,
            };
            let sp: Vec<&str> = sample_line.split_whitespace().collect();
            if sp.is_empty() { continue; }

            let depth: f64 = sp[0].parse().unwrap_or(0.0);
            let mut values = Vec::with_capacity(num_properties);
            for pi in 0..num_properties {
                let v = if pi + 1 < sp.len() {
                    sp[pi + 1].parse().unwrap_or(f64::NAN)
                } else {
                    f64::NAN
                };
                values.push(v);
            }

            samples.push(Sample {
                hole_idx,
                sample_idx: si,
                x: sx + ndx * depth,
                y: sy + ndy * depth,
                z: sz + ndz * depth,
                values,
            });
        }
    }

    let total_samples = samples.len();
    if total_samples < 2 {
        writeln!(out, "0").ok();
        return;
    }

    let mut prop_mean = vec![0.0f64; num_properties];
    let mut prop_var = vec![0.0f64; num_properties];
    let mut prop_count = vec![0usize; num_properties];

    for s in &samples {
        for pi in 0..num_properties {
            let v = s.values[pi];
            if v.is_finite() {
                prop_mean[pi] += v;
                prop_count[pi] += 1;
            }
        }
    }
    for pi in 0..num_properties {
        if prop_count[pi] > 0 {
            prop_mean[pi] /= prop_count[pi] as f64;
        }
    }
    for s in &samples {
        for pi in 0..num_properties {
            let v = s.values[pi];
            if v.is_finite() {
                let d = v - prop_mean[pi];
                prop_var[pi] += d * d;
            }
        }
    }
    for pi in 0..num_properties {
        if prop_count[pi] > 1 {
            prop_var[pi] /= prop_count[pi] as f64;
        }
    }
    let prop_std: Vec<f64> = prop_var.iter().map(|v| v.sqrt()).collect();

    let k = if total_samples < 50 {
        (total_samples - 1).min(5).max(2)
    } else if total_samples < 1000 {
        8
    } else if total_samples < 50000 {
        12
    } else {
        16
    };

    let mut indices: Vec<usize> = (0..total_samples).collect();
    let tree = build_kdtree(&mut indices, &samples, 0);

    let mut suspects: Vec<Suspect> = Vec::new();

    let z_threshold = if total_samples < 100 { 2.5 } else { 3.0 };

    for i in 0..total_samples {
        let mut neighbors = Vec::with_capacity(k + 1);
        knn_search(&tree, &samples, &samples[i], k, &mut neighbors, i);

        if neighbors.is_empty() { continue; }

        for pi in 0..num_properties {
            let v = samples[i].values[pi];
            if !v.is_finite() { continue; }
            if prop_std[pi] < 1e-12 { continue; }

            let mut n_mean = 0.0f64;
            let mut n_count = 0usize;
            let mut n_vals: Vec<f64> = Vec::new();

            for &(ni, _dist_sq) in &neighbors {
                let nv = samples[ni].values[pi];
                if nv.is_finite() {
                    n_vals.push(nv);
                    n_mean += nv;
                    n_count += 1;
                }
            }

            if n_count < 2 { continue; }

            n_mean /= n_count as f64;
            let mut n_var = 0.0f64;
            for &nv in &n_vals {
                let d = nv - n_mean;
                n_var += d * d;
            }
            n_var /= n_count as f64;
            let n_std = n_var.sqrt();

            let effective_std = n_std.max(prop_std[pi] * 0.1).max(1e-12);

            let local_z = (v - n_mean).abs() / effective_std;
            let global_z = (v - prop_mean[pi]).abs() / prop_std[pi].max(1e-12);

            let combined = local_z * 0.7 + global_z * 0.3;

            if combined > z_threshold {
                let confidence = ((combined - z_threshold) * 25.0).min(100.0).max(0.1);
                let prop_name = if pi < property_names.len() {
                    property_names[pi].clone()
                } else {
                    format!("prop_{}", pi)
                };
                suspects.push(Suspect {
                    hole_id: hole_ids[samples[i].hole_idx].clone(),
                    sample_idx: samples[i].sample_idx,
                    property_name: prop_name,
                    confidence,
                });
            }
        }
    }

    suspects.sort_unstable_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap_or(std::cmp::Ordering::Equal));

    let max_output = if total_samples > 100000 { 10000 } else if total_samples > 10000 { 5000 } else { suspects.len() };
    let output_count = suspects.len().min(max_output);

    writeln!(out, "{}", output_count).ok();
    for s in suspects.iter().take(output_count) {
        writeln!(out, "{} {} {} {:.2}", s.hole_id, s.sample_idx, s.property_name, s.confidence).ok();
    }
}
