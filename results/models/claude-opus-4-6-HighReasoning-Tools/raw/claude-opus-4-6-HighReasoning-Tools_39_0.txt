use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_ascii_whitespace();
    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();
    let half = n / 2;

    let mut adj_list: Vec<Vec<usize>> = vec![vec![]; n];
    let mut adj_mat = vec![vec![false; n]; n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u != v && !adj_mat[u][v] {
            adj_mat[u][v] = true;
            adj_mat[v][u] = true;
            adj_list[u].push(v);
            adj_list[v].push(u);
        }
    }

    if n == 0 {
        println!("0");
        println!();
        return;
    }

    // Reorder vertices by degree descending for better pruning
    let mut order: Vec<usize> = (0..n).collect();
    order.sort_by(|&a, &b| adj_list[b].len().cmp(&adj_list[a].len()));
    let mut pos = vec![0usize; n];
    for (i, &v) in order.iter().enumerate() {
        pos[v] = i;
    }

    // Build reordered adjacency list (only neighbors with higher index, for updates)
    let mut nadj: Vec<Vec<usize>> = vec![vec![]; n];
    for u in 0..n {
        for &v in &adj_list[u] {
            nadj[pos[u]].push(pos[v]);
        }
        nadj[pos[u]].sort();
        nadj[pos[u]].dedup();
    }

    // Reordered adjacency matrix
    let mut nmat = vec![vec![false; n]; n];
    for i in 0..n {
        for &j in &nadj[i] {
            nmat[i][j] = true;
        }
    }

    // --- Kernighan-Lin heuristic for initial upper bound ---
    let mut part = vec![0u8; n]; // 1=A, 0=B
    for i in 0..half {
        part[i] = 1;
    }

    loop {
        let mut best_gain = 0i64;
        let mut best_swap = (0usize, 0usize);
        for a in 0..n {
            if part[a] != 1 { continue; }
            let mut da = 0i64;
            for &u in &nadj[a] {
                if part[u] == part[a] { da -= 1; } else { da += 1; }
            }
            for b in 0..n {
                if part[b] != 0 { continue; }
                let mut db = 0i64;
                for &u in &nadj[b] {
                    if part[u] == part[b] { db -= 1; } else { db += 1; }
                }
                let cab = if nmat[a][b] { 2i64 } else { 0 };
                let g = da + db - cab;
                if g > best_gain {
                    best_gain = g;
                    best_swap = (a, b);
                }
            }
        }
        if best_gain <= 0 { break; }
        part[best_swap.0] = 0;
        part[best_swap.1] = 1;
    }

    let mut init_cut = 0usize;
    for u in 0..n {
        for &v in &nadj[u] {
            if v > u && part[u] != part[v] {
                init_cut += 1;
            }
        }
    }

    let mut best_cut = init_cut;
    let mut best_assign: Vec<bool> = part.iter().map(|&x| x == 1).collect();

    // --- Branch and bound ---
    let mut assign = vec![false; n];
    let mut e2a = vec![0usize; n]; // edges to A-assigned among decided
    let mut e2b = vec![0usize; n]; // edges to B-assigned among decided

    bb(
        0, n, half, 0, 0, 0,
        &mut assign, &mut e2a, &mut e2b,
        &nadj, &mut best_cut, &mut best_assign,
    );

    // Map back to original labels
    let mut result: Vec<usize> = (0..n).filter(|&i| best_assign[i]).map(|i| order[i]).collect();
    result.sort();

    println!("{}", best_cut);
    println!("{}", result.iter().map(|x| x.to_string()).collect::<Vec<_>>().join(" "));
}

fn bb(
    v: usize, n: usize, half: usize,
    in_a: usize, in_b: usize,
    cut: usize,
    assign: &mut Vec<bool>,
    e2a: &mut Vec<usize>,
    e2b: &mut Vec<usize>,
    nadj: &[Vec<usize>],
    best_cut: &mut usize,
    best_assign: &mut Vec<bool>,
) {
    if cut >= *best_cut { return; }

    if v == n {
        if in_a == half && cut < *best_cut {
            *best_cut = cut;
            *best_assign = assign.clone();
        }
        return;
    }

    let can_a = in_a < half;
    let can_b = in_b < n - half;
    let remaining = n - v;
    if half.wrapping_sub(in_a) > remaining { return; }
    if (n - half).wrapping_sub(in_b) > remaining { return; }

    // Lower bound: for each undecided vertex, at least min(e2a, e2b) edges cross
    let mut lb = cut;
    for u in v..n {
        lb += e2a[u].min(e2b[u]);
    }
    if lb >= *best_cut { return; }

    let cut_a = cut + e2b[v];
    let cut_b = cut + e2a[v];
    let a_first = cut_a <= cut_b;

    for pass in 0..2u8 {
        let try_a = if a_first { pass == 0 } else { pass == 1 };

        if try_a && can_a {
            assign[v] = true;
            let mut changed = Vec::new();
            for &u in &nadj[v] {
                if u > v { e2a[u] += 1; changed.push(u); }
            }
            bb(v + 1, n, half, in_a + 1, in_b, cut_a, assign, e2a, e2b, nadj, best_cut, best_assign);
            for &u in &changed { e2a[u] -= 1; }
        } else if !try_a && can_b {
            assign[v] = false;
            let mut changed = Vec::new();
            for &u in &nadj[v] {
                if u > v { e2b[u] += 1; changed.push(u); }
            }
            bb(v + 1, n, half, in_a, in_b + 1, cut_b, assign, e2a, e2b, nadj, best_cut, best_assign);
            for &u in &changed { e2b[u] -= 1; }
        }
    }
}
