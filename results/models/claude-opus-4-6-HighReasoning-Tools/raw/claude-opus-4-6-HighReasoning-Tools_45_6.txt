import struct

def w(val):
    return struct.pack('<I', val & 0xFFFFFFFF)

def f2w(f):
    return struct.unpack('<I', struct.pack('<f', f))[0]

def str_words(s):
    b = s.encode('utf-8') + b'\x00'
    while len(b) % 4 != 0:
        b += b'\x00'
    return [struct.unpack('<I', b[i:i+4])[0] for i in range(0, len(b), 4)]

def inst(opcode, *operands):
    wc = 1 + len(operands)
    return w((wc << 16) | opcode) + b''.join(w(o) for o in operands)

def inst_str(opcode, pre, string, post):
    sw = str_words(string)
    return inst(opcode, *pre, *sw, *post)

_n = [0]
def nid():
    _n[0] += 1
    return _n[0]

# Types
VOID = nid()       # 1
FLOAT = nid()      # 2
VEC3 = nid()       # 3
VEC4 = nid()       # 4
MAT4 = nid()       # 5
INT = nid()        # 6
FUNC_T = nid()     # 7

# Pointer types
P_IN_VEC3 = nid()   # 8
P_OUT_VEC4 = nid()  # 9
P_UNI_UBO = nid()   # 10
P_UNI_VEC4 = nid()  # 11

# UBO struct
UBO_T = nid()       # 12

# Variables
V_WPOS = nid()     # 13
V_NORM = nid()     # 14
V_OUT = nid()      # 15
V_UBO = nid()      # 16

# Function/label
MAIN = nid()       # 17
GLSL = nid()       # 18
LABEL = nid()      # 19

# Float constants
C0 = nid()         # 20 - 0.0
C01 = nid()        # 21 - 0.1
C06 = nid()        # 22 - 0.6
C08 = nid()        # 23 - 0.8
C1 = nid()         # 24 - 1.0
C64 = nid()        # 25 - 64.0

# Int constants
CI3 = nid()        # 26 - 3 (lightPos index)
CI4 = nid()        # 27 - 4 (cameraPos index)

# SSA temps
r_wpos = nid()     # 28
r_norm = nid()     # 29
r_lp_ptr = nid()   # 30
r_lp = nid()       # 31
r_lp3 = nid()      # 32
r_cp_ptr = nid()   # 33
r_cp = nid()       # 34
r_cp3 = nid()      # 35
r_Lun = nid()      # 36
r_Vun = nid()      # 37
r_N = nid()        # 38
r_L = nid()        # 39
r_V = nid()        # 40
r_LpV = nid()      # 41
r_H = nid()        # 42
r_NdL = nid()      # 43
r_diff = nid()     # 44
r_NdH = nid()      # 45
r_NdHc = nid()     # 46
r_spec = nid()     # 47
r_t1 = nid()       # 48
r_t2 = nid()       # 49
r_t3 = nid()       # 50
r_inten = nid()    # 51
r_out = nid()      # 52

BOUND = _n[0] + 1  # 53

code = b''

# Header
code += w(0x07230203)
code += w(0x00010000)
code += w(0)
code += w(BOUND)
code += w(0)

# OpCapability Shader
code += inst(17, 1)

# OpExtInstImport GLSL.std.450
code += inst_str(11, [GLSL], "GLSL.std.450", [])

# OpMemoryModel Logical GLSL450
code += inst(14, 0, 1)

# OpEntryPoint Fragment %main "main" %V_WPOS %V_NORM %V_OUT
code += inst_str(15, [4, MAIN], "main", [V_WPOS, V_NORM, V_OUT])

# OpExecutionMode %main OriginUpperLeft
code += inst(16, MAIN, 7)

# --- Decorations ---
code += inst(71, V_WPOS, 30, 0)   # Location 0
code += inst(71, V_NORM, 30, 1)   # Location 1
code += inst(71, V_OUT, 30, 0)    # Output Location 0

code += inst(71, V_UBO, 34, 0)    # DescriptorSet 0
code += inst(71, V_UBO, 33, 0)    # Binding 0
code += inst(71, UBO_T, 2)        # Block

# Member decorations for UBO struct
# Member 0: mat4 model, offset 0
code += inst(72, UBO_T, 0, 5)         # ColMajor
code += inst(72, UBO_T, 0, 7, 16)     # MatrixStride 16
code += inst(72, UBO_T, 0, 35, 0)     # Offset 0

# Member 1: mat4 view, offset 64
code += inst(72, UBO_T, 1, 5)
code += inst(72, UBO_T, 1, 7, 16)
code += inst(72, UBO_T, 1, 35, 64)

# Member 2: mat4 projection, offset 128
code += inst(72, UBO_T, 2, 5)
code += inst(72, UBO_T, 2, 7, 16)
code += inst(72, UBO_T, 2, 35, 128)

# Member 3: vec4 lightPos, offset 192
code += inst(72, UBO_T, 3, 35, 192)

# Member 4: vec4 cameraPos, offset 208
code += inst(72, UBO_T, 4, 35, 208)

# Member 5: vec4 params, offset 224
code += inst(72, UBO_T, 5, 35, 224)

# --- Type declarations ---
code += inst(19, VOID)              # OpTypeVoid
code += inst(22, FLOAT, 32)         # OpTypeFloat 32
code += inst(23, VEC3, FLOAT, 3)    # OpTypeVector float 3
code += inst(23, VEC4, FLOAT, 4)    # OpTypeVector float 4
code += inst(24, MAT4, VEC4, 4)     # OpTypeMatrix vec4 4
code += inst(21, INT, 32, 1)        # OpTypeInt 32 signed
code += inst(33, FUNC_T, VOID)      # OpTypeFunction void

# Pointer types (StorageClass: Input=1, Output=3, Uniform=2)
code += inst(32, P_IN_VEC3, 1, VEC3)
code += inst(32, P_OUT_VEC4, 3, VEC4)

# UBO struct type: OpTypeStruct
code += inst(30, UBO_T, MAT4, MAT4, MAT4, VEC4, VEC4, VEC4)

code += inst(32, P_UNI_UBO, 2, UBO_T)
code += inst(32, P_UNI_VEC4, 2, VEC4)

# Variables (OpVariable: result_type result_id storage_class)
code += inst(59, P_IN_VEC3, V_WPOS, 1)    # Input
code += inst(59, P_IN_VEC3, V_NORM, 1)    # Input
code += inst(59, P_OUT_VEC4, V_OUT, 3)    # Output
code += inst(59, P_UNI_UBO, V_UBO, 2)     # Uniform

# Constants (OpConstant: result_type result_id value)
code += inst(43, FLOAT, C0, f2w(0.0))
code += inst(43, FLOAT, C01, f2w(0.1))
code += inst(43, FLOAT, C06, f2w(0.6))
code += inst(43, FLOAT, C08, f2w(0.8))
code += inst(43, FLOAT, C1, f2w(1.0))
code += inst(43, FLOAT, C64, f2w(64.0))
code += inst(43, INT, CI3, 3)
code += inst(43, INT, CI4, 4)

# --- Function ---
code += inst(54, VOID, MAIN, 0, FUNC_T)   # OpFunction
code += inst(248, LABEL)                    # OpLabel

# Load worldPos and normal
code += inst(61, VEC3, r_wpos, V_WPOS)     # OpLoad
code += inst(61, VEC3, r_norm, V_NORM)

# Load lightPos from UBO (member 3)
code += inst(65, P_UNI_VEC4, r_lp_ptr, V_UBO, CI3)  # OpAccessChain
code += inst(61, VEC4, r_lp, r_lp_ptr)                # OpLoad
code += inst(79, VEC3, r_lp3, r_lp, r_lp, 0, 1, 2)   # VectorShuffle xyz

# Load cameraPos from UBO (member 4)
code += inst(65, P_UNI_VEC4, r_cp_ptr, V_UBO, CI4)
code += inst(61, VEC4, r_cp, r_cp_ptr)
code += inst(79, VEC3, r_cp3, r_cp, r_cp, 0, 1, 2)

# L = normalize(lightPos.xyz - worldPos)
code += inst(131, VEC3, r_Lun, r_lp3, r_wpos)        # OpFSub
code += inst(12, VEC3, r_L, GLSL, 69, r_Lun)          # ExtInst Normalize

# V = normalize(cameraPos.xyz - worldPos)
code += inst(131, VEC3, r_Vun, r_cp3, r_wpos)
code += inst(12, VEC3, r_V, GLSL, 69, r_Vun)

# N = normalize(normal)
code += inst(12, VEC3, r_N, GLSL, 69, r_norm)

# H = normalize(L + V)
code += inst(129, VEC3, r_LpV, r_L, r_V)              # OpFAdd
code += inst(12, VEC3, r_H, GLSL, 69, r_LpV)

# diffuse = max(dot(N, L), 0.0)
code += inst(148, FLOAT, r_NdL, r_N, r_L)             # OpDot
code += inst(12, FLOAT, r_diff, GLSL, 40, r_NdL, C0)  # ExtInst FMax

# specular = pow(max(dot(N, H), 0.0), 64.0)
code += inst(148, FLOAT, r_NdH, r_N, r_H)
code += inst(12, FLOAT, r_NdHc, GLSL, 40, r_NdH, C0)
code += inst(12, FLOAT, r_spec, GLSL, 26, r_NdHc, C64)  # ExtInst Pow

# intensity = 0.1 + 0.6*diffuse + 0.8*specular
code += inst(133, FLOAT, r_t1, C06, r_diff)           # OpFMul
code += inst(133, FLOAT, r_t2, C08, r_spec)
code += inst(129, FLOAT, r_t3, C01, r_t1)             # OpFAdd
code += inst(129, FLOAT, r_inten, r_t3, r_t2)

# outColor = vec4(intensity, intensity, intensity, 1.0)
code += inst(80, VEC4, r_out, r_inten, r_inten, r_inten, C1)  # CompositeConstruct

# Store and return
code += inst(62, V_OUT, r_out)    # OpStore
code += inst(253)                  # OpReturn
code += inst(56)                   # OpFunctionEnd

hex_str = code.hex()
print(hex_str)

# Also write binary for validation
with open('test_shader.spv', 'wb') as f:
    f.write(code)
