import struct

def float_bits(f):
    return struct.unpack('<I', struct.pack('<f', f))[0]

def encode_str(s):
    b = s.encode('utf-8') + b'\x00'
    while len(b) % 4 != 0:
        b += b'\x00'
    result = []
    for i in range(0, len(b), 4):
        result.append(struct.unpack('<I', b[i:i+4])[0])
    return result

class B:
    def __init__(self):
        self.words = []
    def inst(self, opcode, *operands):
        wc = 1 + len(operands)
        self.words.append((wc << 16) | opcode)
        self.words.extend(operands)
    def raw(self, *ws):
        self.words.extend(ws)
    def to_hex(self, bound):
        header = [0x07230203, 0x00010000, 0, bound, 0]
        all_w = header + self.words
        return b''.join(struct.pack('<I', w) for w in all_w).hex()

b = B()

# IDs
GLSL=1; VOID=2; FUNC_T=3; FLOAT=4; VEC3=5; VEC4=6; MAT4=7; INT=8
PTR_IN_V3=9; PTR_OUT_V4=10; PTR_UNI_S=11; PTR_UNI_V4=12
V_WP=13; V_N=14; V_OUT=15; V_UBO=16; UBO_S=17; MAIN=18; LABEL=19
C0=20; C1=21; C03=22; C07=23; C08=24; C02=25; C15=26; C30=27
CI3=28; CI4=29
n = 30  # next SSA id

# --- Preamble ---
b.inst(17, 1)  # OpCapability Shader

# OpExtInstImport %1 "GLSL.std.450"
sw = encode_str("GLSL.std.450")
b.raw(((2+len(sw))<<16)|11, GLSL, *sw)

b.inst(14, 0, 1)  # OpMemoryModel Logical GLSL450

# OpEntryPoint Fragment %18 "main" %13 %14 %15
nw = encode_str("main")
ops = [4, MAIN] + nw + [V_WP, V_N, V_OUT]
b.raw(((1+len(ops))<<16)|15, *ops)

b.inst(16, MAIN, 7)  # OpExecutionMode OriginUpperLeft

# --- Decorations ---
b.inst(71, V_WP, 30, 0)   # Location 0
b.inst(71, V_N, 30, 1)    # Location 1
b.inst(71, V_OUT, 30, 0)  # Location 0
b.inst(71, UBO_S, 2)      # Block (no extra operand)
for m, off in [(0,0),(1,64),(2,128),(3,192),(4,208),(5,224)]:
    b.inst(72, UBO_S, m, 35, off)  # MemberDecorate Offset
for m in [0,1,2]:
    b.inst(72, UBO_S, m, 5)       # ColMajor
for m in [0,1,2]:
    b.inst(72, UBO_S, m, 7, 16)   # MatrixStride 16
b.inst(71, V_UBO, 34, 0)  # DescriptorSet 0
b.inst(71, V_UBO, 33, 0)  # Binding 0

# --- Types ---
b.inst(19, VOID)
b.inst(33, FUNC_T, VOID)
b.inst(22, FLOAT, 32)
b.inst(23, VEC3, FLOAT, 3)
b.inst(23, VEC4, FLOAT, 4)
b.inst(24, MAT4, VEC4, 4)
b.inst(21, INT, 32, 1)
# OpTypeStruct UBO_S: mat4 mat4 mat4 vec4 vec4 vec4
b.raw((8<<16)|30, UBO_S, MAT4, MAT4, MAT4, VEC4, VEC4, VEC4)
b.inst(32, PTR_IN_V3, 1, VEC3)
b.inst(32, PTR_OUT_V4, 3, VEC4)
b.inst(32, PTR_UNI_S, 2, UBO_S)
b.inst(32, PTR_UNI_V4, 2, VEC4)

# --- Global Variables ---
b.inst(59, PTR_IN_V3, V_WP, 1)
b.inst(59, PTR_IN_V3, V_N, 1)
b.inst(59, PTR_OUT_V4, V_OUT, 3)
b.inst(59, PTR_UNI_S, V_UBO, 2)

# --- Constants ---
b.inst(43, FLOAT, C0, float_bits(0.0))
b.inst(43, FLOAT, C1, float_bits(1.0))
b.inst(43, FLOAT, C03, float_bits(0.3))
b.inst(43, FLOAT, C07, float_bits(0.7))
b.inst(43, FLOAT, C08, float_bits(0.8))
b.inst(43, FLOAT, C02, float_bits(0.2))
b.inst(43, FLOAT, C15, float_bits(1.5))
b.inst(43, FLOAT, C30, float_bits(3.0))
b.inst(43, INT, CI3, 3)
b.inst(43, INT, CI4, 4)

# --- Function ---
b.inst(54, VOID, MAIN, 0, FUNC_T)  # OpFunction
b.inst(248, LABEL)                   # OpLabel

# Load worldPos and normal
WP=n; n+=1
b.inst(61, VEC3, WP, V_WP)
NR=n; n+=1
b.inst(61, VEC3, NR, V_N)

# Normalize normal
NN=n; n+=1
b.inst(12, VEC3, NN, GLSL, 69, NR)  # GLSL Normalize=69

# Load lightPos (member 3)
LP_PTR=n; n+=1
b.inst(65, PTR_UNI_V4, LP_PTR, V_UBO, CI3)  # OpAccessChain
LP_V4=n; n+=1
b.inst(61, VEC4, LP_V4, LP_PTR)

# lightPos.xyz via VectorShuffle
LP_V3=n; n+=1
b.raw((7<<16)|79, VEC3, LP_V3, LP_V4, LP_V4, 0, 1, 2)

# L = normalize(lightPos.xyz - worldPos)
L_RAW=n; n+=1
b.inst(131, VEC3, L_RAW, LP_V3, WP)  # OpFSub
L=n; n+=1
b.inst(12, VEC3, L, GLSL, 69, L_RAW)

# diffuse = max(dot(N, L), 0.0)
NDOTL=n; n+=1
b.inst(148, FLOAT, NDOTL, NN, L)
DIFF=n; n+=1
b.inst(12, FLOAT, DIFF, GLSL, 40, NDOTL, C0)  # FMax=40

# factor = 0.3 + 0.7*diffuse
F1=n; n+=1
b.inst(133, FLOAT, F1, C07, DIFF)
FACTOR=n; n+=1
b.inst(129, FLOAT, FACTOR, C03, F1)

# litColor = baseColor * factor
BR=n; n+=1
b.inst(133, FLOAT, BR, C08, FACTOR)
BG=n; n+=1
b.inst(133, FLOAT, BG, C02, FACTOR)
BB=n; n+=1
b.inst(133, FLOAT, BB, C02, FACTOR)
LIT=n; n+=1
b.inst(80, VEC3, LIT, BR, BG, BB)

# Load cameraPos (member 4)
CP_PTR=n; n+=1
b.inst(65, PTR_UNI_V4, CP_PTR, V_UBO, CI4)
CP_V4=n; n+=1
b.inst(61, VEC4, CP_V4, CP_PTR)
CP_V3=n; n+=1
b.raw((7<<16)|79, VEC3, CP_V3, CP_V4, CP_V4, 0, 1, 2)

# dist = length(worldPos - cameraPos)
DV=n; n+=1
b.inst(131, VEC3, DV, WP, CP_V3)
DIST=n; n+=1
b.inst(12, FLOAT, DIST, GLSL, 66, DV)  # Length=66

# fogFactor = clamp((dist-1.5)/3.0, 0, 1)
D1=n; n+=1
b.inst(131, FLOAT, D1, DIST, C15)
D2=n; n+=1
b.inst(136, FLOAT, D2, D1, C30)  # OpFDiv=136
FF=n; n+=1
b.inst(12, FLOAT, FF, GLSL, 43, D2, C0, C1)  # FClamp=43

# fogColor = vec3(0.7, 0.7, 0.8)
FOG=n; n+=1
b.inst(80, VEC3, FOG, C07, C07, C08)

# Splat fogFactor to vec3 for FMix
FF3=n; n+=1
b.inst(80, VEC3, FF3, FF, FF, FF)

# mix(litColor, fogColor, fogFactor)
MIX=n; n+=1
b.inst(12, VEC3, MIX, GLSL, 46, LIT, FOG, FF3)  # FMix=46

# Build output vec4
MR=n; n+=1
b.inst(81, FLOAT, MR, MIX, 0)
MG_=n; n+=1
b.inst(81, FLOAT, MG_, MIX, 1)
MB=n; n+=1
b.inst(81, FLOAT, MB, MIX, 2)
OUT_V=n; n+=1
b.inst(80, VEC4, OUT_V, MR, MG_, MB, C1)

# Store and return
b.inst(62, V_OUT, OUT_V)  # OpStore
b.inst(253)                # OpReturn
b.inst(56)                 # OpFunctionEnd

BOUND = n
print(b.to_hex(BOUND))
