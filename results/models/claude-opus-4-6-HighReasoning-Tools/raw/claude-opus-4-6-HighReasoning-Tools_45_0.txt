import struct

words = []

def w(val):
    words.append(struct.pack('<I', val))

def op(word_count, opcode, *operands):
    w((word_count << 16) | opcode)
    for o in operands:
        if isinstance(o, bytes):
            words.append(o)
        else:
            w(o)

def string_words(s):
    encoded = s.encode('utf-8') + b'\x00'
    while len(encoded) % 4 != 0:
        encoded += b'\x00'
    return encoded

# Header
w(0x07230203)  # Magic
w(0x00010000)  # Version 1.0
w(0x00000000)  # Generator
w(0x0000000C)  # Bound = 12
w(0x00000000)  # Reserved

# IDs:
# 1 = void
# 2 = func_type (void -> void)
# 3 = float32
# 4 = vec4
# 5 = ptr_output_vec4
# 6 = outColor variable
# 7 = constant 1.0
# 8 = constant 0.0
# 9 = main function
# 10 = label
# 11 = composite construct result

# OpCapability Shader
op(2, 17, 1)

# OpMemoryModel Logical GLSL450
op(3, 14, 0, 1)

# OpEntryPoint Fragment %9 "main" %6
name = string_words("main")
name_word_count = len(name) // 4
op(4 + name_word_count, 15, 4, 9, name, 6)

# OpExecutionMode %9 OriginUpperLeft
op(3, 16, 9, 7)

# OpDecorate %6 Location 0
op(4, 71, 6, 30, 0)

# OpTypeVoid %1
op(2, 19, 1)

# OpTypeFunction %2 %1
op(3, 33, 2, 1)

# OpTypeFloat %3 32
op(3, 22, 3, 32)

# OpTypeVector %4 %3 4
op(4, 23, 4, 3, 4)

# OpTypePointer %5 Output(3) %4
op(4, 32, 5, 3, 4)

# OpVariable %5 %6 Output(3)
op(4, 59, 5, 6, 3)

# OpConstant %3 %7 1.0f
float_1 = struct.unpack('<I', struct.pack('<f', 1.0))[0]
op(4, 43, 3, 7, float_1)

# OpConstant %3 %8 0.0f
float_0 = struct.unpack('<I', struct.pack('<f', 0.0))[0]
op(4, 43, 3, 8, float_0)

# OpFunction %1 %9 None(0) %2
op(5, 54, 1, 9, 0, 2)

# OpLabel %10
op(2, 248, 10)

# OpCompositeConstruct %4 %11 %7 %8 %8 %7
op(7, 80, 4, 11, 7, 8, 8, 7)

# OpStore %6 %11
op(3, 62, 6, 11)

# OpReturn
op(1, 253)

# OpFunctionEnd
op(1, 56)

binary = b''.join(words)
hex_str = binary.hex()
print(hex_str)
