; SPIR-V
; Version: 1.0
; Generator: Manual
; Bound: 90
; Schema: 0

               OpCapability Shader
          %glsl = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %out_color
               OpExecutionMode %main OriginUpperLeft

               ; Input/Output decorations
               OpDecorate %in_worldPos Location 0
               OpDecorate %in_normal Location 1
               OpDecorate %in_uv Location 2
               OpDecorate %out_color Location 0

               ; UBO decorations
               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %ubo_var DescriptorSet 0
               OpDecorate %ubo_var Binding 0

               ; Types
       %void = OpTypeVoid
   %func_type = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %mat4 = OpTypeMatrix %v4float 4
       %bool = OpTypeBool
        %int = OpTypeInt 32 1

               ; Pointer types
 %ptr_in_v3 = OpTypePointer Input %v3float
 %ptr_in_v2 = OpTypePointer Input %v2float
%ptr_out_v4 = OpTypePointer Output %v4float
       %UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
  %ptr_ubo  = OpTypePointer Uniform %UBO
  %ptr_u_v4 = OpTypePointer Uniform %v4float

               ; Variables
%in_worldPos = OpVariable %ptr_in_v3 Input
  %in_normal = OpVariable %ptr_in_v3 Input
      %in_uv = OpVariable %ptr_in_v2 Input
   %out_color = OpVariable %ptr_out_v4 Output
    %ubo_var = OpVariable %ptr_ubo Uniform

               ; Constants
      %int_3 = OpConstant %int 3
    %float_0 = OpConstant %float 0.0
  %float_0_1 = OpConstant %float 0.1
  %float_0_3 = OpConstant %float 0.3
  %float_0_5 = OpConstant %float 0.5
  %float_0_7 = OpConstant %float 0.7
  %float_0_8 = OpConstant %float 0.8
    %float_1 = OpConstant %float 1.0
    %float_6 = OpConstant %float 6.0

               ; Main function
       %main = OpFunction %void None %func_type
      %entry = OpLabel

               ; Load inputs
   %worldPos = OpLoad %v3float %in_worldPos
 %normal_raw = OpLoad %v3float %in_normal
         %uv = OpLoad %v2float %in_uv

               ; Normalize normal
     %normal = OpExtInst %v3float %glsl 69 %normal_raw

               ; Load lightPos from UBO member 3
%lightPos_ptr = OpAccessChain %ptr_u_v4 %ubo_var %int_3
   %lightPos = OpLoad %v4float %lightPos_ptr
     %lightX = OpCompositeExtract %float %lightPos 0
     %lightY = OpCompositeExtract %float %lightPos 1
     %lightZ = OpCompositeExtract %float %lightPos 2
   %lightVec = OpCompositeConstruct %v3float %lightX %lightY %lightZ

               ; Light direction = normalize(lightPos - worldPos)
  %toLight   = OpFSub %v3float %lightVec %worldPos
  %lightDir  = OpExtInst %v3float %glsl 69 %toLight

               ; Diffuse = max(dot(N, L), 0)
     %dotNL  = OpDot %float %normal %lightDir
   %diffuse  = OpExtInst %float %glsl 40 %dotNL %float_0

               ; Factor = 0.3 + 0.7 * diffuse
    %scaled  = OpFMul %float %float_0_7 %diffuse
    %factor  = OpFAdd %float %float_0_3 %scaled

               ; Scale UV by 6
       %uvX  = OpCompositeExtract %float %uv 0
       %uvY  = OpCompositeExtract %float %uv 1
     %sclX   = OpFMul %float %uvX %float_6
     %sclY   = OpFMul %float %uvY %float_6

               ; Fract to get position within cell
  %sclUV     = OpCompositeConstruct %v2float %sclX %sclY
  %fractUV   = OpExtInst %v2float %glsl 10 %sclUV

               ; Distance from cell center (0.5, 0.5)
   %center   = OpCompositeConstruct %v2float %float_0_5 %float_0_5
     %diff   = OpFSub %v2float %fractUV %center
     %dist   = OpExtInst %float %glsl 66 %diff

               ; Is this pixel inside a dot?
   %is_dot   = OpFOrdLessThan %bool %dist %float_0_3

               ; Select color per component (scalar select to avoid vector bool)
    %baseR   = OpSelect %float %is_dot %float_1 %float_0_1
    %baseG   = OpSelect %float %is_dot %float_0_8 %float_0_1
    %baseB   = OpSelect %float %is_dot %float_0 %float_0_3

               ; Apply shading factor
  %shadedR   = OpFMul %float %baseR %factor
  %shadedG   = OpFMul %float %baseG %factor
  %shadedB   = OpFMul %float %baseB %factor

               ; Output
  %outVal    = OpCompositeConstruct %v4float %shadedR %shadedG %shadedB %float_1
               OpStore %out_color %outVal
               OpReturn
               OpFunctionEnd
