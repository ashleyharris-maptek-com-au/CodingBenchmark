#version 450

layout(local_size_x = 256, local_size_y = 1, local_size_z = 1) in;

layout(set=0, binding=0) readonly buffer InputBuf {
    uint data[];
} input_buf;

layout(set=0, binding=1) buffer OutputBuf {
    uint data[];
} output_buf;

layout(set=0, binding=2) uniform UniformBuf {
    uvec4 params;
} uniform_buf;

void main() {
    uint gid = gl_GlobalInvocationID.x;
    uint N = uniform_buf.params.x;
    if (gid >= N) return;

    uint sum = 0u;
    for (uint i = 0u; i <= gid; i++) {
        sum += input_buf.data[i];
    }
    output_buf.data[gid] = sum;
}
