import struct

def float_bits(f):
    return struct.unpack('<I', struct.pack('<f', f))[0]

def encode_string(s):
    b = s.encode('utf-8') + b'\x00'
    while len(b) % 4 != 0:
        b += b'\x00'
    return [struct.unpack('<I', b[i:i+4])[0] for i in range(0, len(b), 4)]

def op(wc, opcode):
    return (wc << 16) | opcode

words = []

# IDs:
# 1=GLSL.std.450, 2=void, 3=func_type, 4=float, 5=vec3, 6=vec4
# 7=ptr_in_vec3, 8=ptr_out_vec4, 9=mat4, 10=ubo_struct
# 11=ptr_uniform_struct, 12=ptr_uniform_vec4, 13=int
# 14=worldPos, 15=normal, 16=outColor, 17=ubo_var
# 18=const_int3, 19=const_0.5, 20=const_1.0, 21=const_0.2
# 22=const_0.75, 23=const_0.7, 24=const_0.4
# 25=main, 26=label, 27-44=SSA values
BOUND = 45

# Header
words.extend([0x07230203, 0x00010000, 0, BOUND, 0])

# OpCapability Shader(1)
words.extend([op(2, 17), 1])

# OpExtInstImport %1 "GLSL.std.450"
name = encode_string("GLSL.std.450")
words.append(op(2 + len(name), 11))
words.append(1)
words.extend(name)

# OpMemoryModel Logical(0) GLSL450(1)
words.extend([op(3, 14), 0, 1])

# OpEntryPoint Fragment(4) %25 "main" %14 %15 %16
main_name = encode_string("main")
iface = [14, 15, 16]
words.append(op(3 + len(main_name) + len(iface), 15))
words.append(4)   # Fragment
words.append(25)  # entry func
words.extend(main_name)
words.extend(iface)

# OpExecutionMode %25 OriginUpperLeft(7)
words.extend([op(3, 16), 25, 7])

# --- Decorations ---
def decorate(target, dec, *operands):
    words.append(op(3 + len(operands), 71))
    words.extend([target, dec])
    words.extend(operands)

def member_decorate(stype, member, dec, *operands):
    words.append(op(4 + len(operands), 72))
    words.extend([stype, member, dec])
    words.extend(operands)

decorate(14, 30, 0)   # worldPos Location 0
decorate(15, 30, 1)   # normal Location 1
decorate(16, 30, 0)   # outColor Location 0
decorate(10, 2)       # UBO struct Block
decorate(17, 34, 0)   # DescriptorSet 0
decorate(17, 33, 0)   # Binding 0

for i in range(3):
    member_decorate(10, i, 5)          # ColMajor
    member_decorate(10, i, 7, 16)      # MatrixStride 16
    member_decorate(10, i, 35, i*64)   # Offset

member_decorate(10, 3, 35, 192)  # lightPos
member_decorate(10, 4, 35, 208)  # cameraPos
member_decorate(10, 5, 35, 224)  # params

# --- Types ---
# OpTypeVoid %2
words.extend([op(2, 19), 2])
# OpTypeFunction %3 %2
words.extend([op(3, 33), 3, 2])
# OpTypeFloat %4 32
words.extend([op(3, 22), 4, 32])
# OpTypeVector %5 %4 3
words.extend([op(4, 23), 5, 4, 3])
# OpTypeVector %6 %4 4
words.extend([op(4, 23), 6, 4, 4])
# OpTypeMatrix %9 %6 4
words.extend([op(4, 24), 9, 6, 4])
# OpTypeStruct %10 {mat4, mat4, mat4, vec4, vec4, vec4}
words.extend([op(8, 30), 10, 9, 9, 9, 6, 6, 6])
# OpTypePointer %7 Input(1) %5
words.extend([op(4, 32), 7, 1, 5])
# OpTypePointer %8 Output(3) %6
words.extend([op(4, 32), 8, 3, 6])
# OpTypePointer %11 Uniform(2) %10
words.extend([op(4, 32), 11, 2, 10])
# OpTypePointer %12 Uniform(2) %6
words.extend([op(4, 32), 12, 2, 6])
# OpTypeInt %13 32 1(signed)
words.extend([op(4, 21), 13, 32, 1])

# --- Constants ---
words.extend([op(4, 43), 13, 18, 3])                # int 3
words.extend([op(4, 43), 4, 19, float_bits(0.5)])    # 0.5
words.extend([op(4, 43), 4, 20, float_bits(1.0)])    # 1.0
words.extend([op(4, 43), 4, 21, float_bits(0.2)])    # 0.2
words.extend([op(4, 43), 4, 22, float_bits(0.75)])   # 0.75
words.extend([op(4, 43), 4, 23, float_bits(0.7)])    # 0.7
words.extend([op(4, 43), 4, 24, float_bits(0.4)])    # 0.4

# --- Variables ---
words.extend([op(4, 59), 7, 14, 1])    # worldPos Input
words.extend([op(4, 59), 7, 15, 1])    # normal Input
words.extend([op(4, 59), 8, 16, 3])    # outColor Output
words.extend([op(4, 59), 11, 17, 2])   # ubo Uniform

# --- Function ---
words.extend([op(5, 54), 2, 25, 0, 3])   # OpFunction void %25 None func_type
words.extend([op(2, 248), 26])             # OpLabel %26

# Load inputs
words.extend([op(4, 61), 5, 27, 15])      # %27 = load normal (vec3)
words.extend([op(4, 61), 5, 28, 14])      # %28 = load worldPos (vec3)

# Access UBO lightPos (member 3)
words.extend([op(5, 65), 12, 29, 17, 18]) # %29 = AccessChain ubo[3]
words.extend([op(4, 61), 6, 30, 29])      # %30 = load lightPos (vec4)

# Extract xyz from lightPos
words.extend([op(8, 79), 5, 31, 30, 30, 0, 1, 2])  # %31 = VectorShuffle -> vec3

# L = normalize(lightPos.xyz - worldPos)
words.extend([op(5, 131), 5, 32, 31, 28])            # %32 = FSub
words.extend([op(6, 12), 5, 33, 1, 69, 32])          # %33 = normalize(%32)

# N = normalize(normal)
words.extend([op(6, 12), 5, 34, 1, 69, 27])          # %34 = normalize(%27)

# t = dot(N,L) * 0.5 + 0.5
words.extend([op(5, 148), 4, 35, 34, 33])             # %35 = dot(N, L)
words.extend([op(5, 133), 4, 36, 35, 19])             # %36 = dot * 0.5
words.extend([op(5, 129), 4, 37, 36, 19])             # %37 = t = result + 0.5

# mix(cool, warm, t) = cool*(1-t) + warm*t
words.extend([op(5, 131), 4, 38, 20, 37])             # %38 = 1.0 - t
words.extend([op(6, 80), 5, 39, 21, 21, 22])          # %39 = cool vec3(0.2, 0.2, 0.75)
words.extend([op(6, 80), 5, 40, 23, 23, 24])          # %40 = warm vec3(0.7, 0.7, 0.4)
words.extend([op(5, 142), 5, 41, 39, 38])             # %41 = cool * (1-t)
words.extend([op(5, 142), 5, 42, 40, 37])             # %42 = warm * t
words.extend([op(5, 129), 5, 43, 41, 42])             # %43 = result vec3

# Output vec4(result, 1.0)
words.extend([op(5, 80), 6, 44, 43, 20])              # %44 = vec4(result, 1.0)
words.extend([op(3, 62), 16, 44])                      # store outColor
words.append(op(1, 253))                               # OpReturn
words.append(op(1, 56))                                # OpFunctionEnd

# Output hex string
hex_str = ''.join(struct.pack('<I', w).hex() for w in words)
print(hex_str)

# Write binary for validation
with open('shader.spv', 'wb') as f:
    for w in words:
        f.write(struct.pack('<I', w))

print(f"\nTotal words: {len(words)}, Total bytes: {len(words)*4}")
