using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

class Program
{
    static void Main()
    {
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;

        var line1 = Console.ReadLine().Trim().Split(new[]{' ','\t'}, StringSplitOptions.RemoveEmptyEntries);
        int V = int.Parse(line1[0]);
        int F = int.Parse(line1[1]);

        double[][] vertices = new double[V][];
        for (int i = 0; i < V; i++)
        {
            var parts = Console.ReadLine().Trim().Split(new[]{' ','\t'}, StringSplitOptions.RemoveEmptyEntries);
            vertices[i] = new double[] {
                double.Parse(parts[0], CultureInfo.InvariantCulture),
                double.Parse(parts[1], CultureInfo.InvariantCulture),
                double.Parse(parts[2], CultureInfo.InvariantCulture)
            };
        }

        for (int i = 0; i < F; i++) Console.ReadLine();

        var lastLine = Console.ReadLine().Trim().Split(new[]{' ','\t'}, StringSplitOptions.RemoveEmptyEntries);
        double W = double.Parse(lastLine[0], CultureInfo.InvariantCulture);
        double H = double.Parse(lastLine[1], CultureInfo.InvariantCulture);
        double D = double.Parse(lastLine[2], CultureInfo.InvariantCulture);

        var rotations = Generate24Rotations();

        long bestCount = 0;
        double[] bestQuat = new double[]{1,0,0,0};
        double bestSx = 0, bestSy = 0, bestSz = 0;
        double bestMinX = 0, bestMinY = 0, bestMinZ = 0;
        int bestNx = 0, bestNy = 0, bestNz = 0;

        double eps = 1e-7;

        foreach (var q in rotations)
        {
            double minX = double.MaxValue, minY = double.MaxValue, minZ = double.MaxValue;
            double maxX = double.MinValue, maxY = double.MinValue, maxZ = double.MinValue;

            for (int i = 0; i < V; i++)
            {
                var r = RotatePoint(q, vertices[i]);
                if (r[0] < minX) minX = r[0];
                if (r[1] < minY) minY = r[1];
                if (r[2] < minZ) minZ = r[2];
                if (r[0] > maxX) maxX = r[0];
                if (r[1] > maxY) maxY = r[1];
                if (r[2] > maxZ) maxZ = r[2];
            }

            double sx = maxX - minX + eps;
            double sy = maxY - minY + eps;
            double sz = maxZ - minZ + eps;

            if (sx > W || sy > H || sz > D) continue;

            int nx = (int)(W / sx);
            int ny = (int)(H / sy);
            int nz = (int)(D / sz);

            if (nx < 1 || ny < 1 || nz < 1) continue;

            long count = (long)nx * ny * nz;
            if (count > bestCount)
            {
                bestCount = count;
                bestQuat = q;
                bestSx = sx; bestSy = sy; bestSz = sz;
                bestMinX = minX; bestMinY = minY; bestMinZ = minZ;
                bestNx = nx; bestNy = ny; bestNz = nz;
            }
        }

        using (var stdout = new BufferedStream(Console.OpenStandardOutput(), 1 << 20))
        using (var writer = new StreamWriter(stdout, System.Text.Encoding.ASCII, 1 << 16))
        {
            writer.NewLine = "\n";
            writer.WriteLine(bestCount);

            string qStr = string.Format(CultureInfo.InvariantCulture,
                "{0:F6} {1:F6} {2:F6} {3:F6}",
                bestQuat[0], bestQuat[1], bestQuat[2], bestQuat[3]);

            for (int ix = 0; ix < bestNx; ix++)
            {
                double tx = ix * bestSx - bestMinX;
                for (int iy = 0; iy < bestNy; iy++)
                {
                    double ty = iy * bestSy - bestMinY;
                    for (int iz = 0; iz < bestNz; iz++)
                    {
                        double tz = iz * bestSz - bestMinZ;
                        writer.Write(tx.ToString("F6", CultureInfo.InvariantCulture));
                        writer.Write(' ');
                        writer.Write(ty.ToString("F6", CultureInfo.InvariantCulture));
                        writer.Write(' ');
                        writer.Write(tz.ToString("F6", CultureInfo.InvariantCulture));
                        writer.Write(' ');
                        writer.WriteLine(qStr);
                    }
                }
            }
        }
    }

    static double[] RotatePoint(double[] q, double[] p)
    {
        double qw = q[0], qx = q[1], qy = q[2], qz = q[3];
        double px = p[0], py = p[1], pz = p[2];

        double tx = 2.0 * (qy * pz - qz * py);
        double ty = 2.0 * (qz * px - qx * pz);
        double tz = 2.0 * (qx * py - qy * px);

        return new double[] {
            px + qw * tx + (qy * tz - qz * ty),
            py + qw * ty + (qz * tx - qx * tz),
            pz + qw * tz + (qx * ty - qy * tx)
        };
    }

    static List<double[]> Generate24Rotations()
    {
        var result = new List<double[]>();
        var seen = new HashSet<string>();

        double s = Math.Sqrt(2.0) / 2.0;

        double[][] bases = new double[][] {
            new double[] {1, 0, 0, 0},
            new double[] {s, s, 0, 0},
            new double[] {0, 1, 0, 0},
            new double[] {s, -s, 0, 0},
            new double[] {s, 0, s, 0},
            new double[] {0, 0, 1, 0},
            new double[] {s, 0, -s, 0},
            new double[] {s, 0, 0, s},
            new double[] {0, 0, 0, 1},
            new double[] {s, 0, 0, -s},
        };

        foreach (var q1 in bases)
        {
            foreach (var q2 in bases)
            {
                var q = QuatMul(q1, q2);
                double len = Math.Sqrt(q[0]*q[0]+q[1]*q[1]+q[2]*q[2]+q[3]*q[3]);
                q[0]/=len; q[1]/=len; q[2]/=len; q[3]/=len;

                if (q[0] < -1e-9 || (Math.Abs(q[0]) < 1e-9 && q[1] < -1e-9) ||
                    (Math.Abs(q[0]) < 1e-9 && Math.Abs(q[1]) < 1e-9 && q[2] < -1e-9) ||
                    (Math.Abs(q[0]) < 1e-9 && Math.Abs(q[1]) < 1e-9 && Math.Abs(q[2]) < 1e-9 && q[3] < -1e-9))
                {
                    q[0] = -q[0]; q[1] = -q[1]; q[2] = -q[2]; q[3] = -q[3];
                }

                string key = $"{Math.Round(q[0],3)},{Math.Round(q[1],3)},{Math.Round(q[2],3)},{Math.Round(q[3],3)}";
                if (seen.Add(key))
                    result.Add(q);
            }
        }

        return result;
    }

    static double[] QuatMul(double[] a, double[] b)
    {
        return new double[] {
            a[0]*b[0] - a[1]*b[1] - a[2]*b[2] - a[3]*b[3],
            a[0]*b[1] + a[1]*b[0] + a[2]*b[3] - a[3]*b[2],
            a[0]*b[2] - a[1]*b[3] + a[2]*b[0] + a[3]*b[1],
            a[0]*b[3] + a[1]*b[2] - a[2]*b[1] + a[3]*b[0]
        };
    }
}
