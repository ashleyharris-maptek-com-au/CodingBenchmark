; SPIR-V Geometry Shader: Expand points into diamonds
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

               ; Types
      %void  = OpTypeVoid
   %voidfn   = OpTypeFunction %void
     %float  = OpTypeFloat 32
   %v3float  = OpTypeVector %float 3
   %v4float  = OpTypeVector %float 4
      %int   = OpTypeInt 32 1
     %uint   = OpTypeInt 32 0

    %int_0   = OpConstant %int 0
    %int_1   = OpConstant %int 1
   %uint_1   = OpConstant %uint 1

  %float_0   = OpConstant %float 0.0
  %float_1   = OpConstant %float 1.0
  %float_h   = OpConstant %float 0.05
 %float_nh   = OpConstant %float -0.05

               ; Input gl_PerVertex block (array size 1 for InputPoints)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in   = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr  = OpTypePointer Input %arr_pv_in
%ptr_in_v4   = OpTypePointer Input %v4float
    %gl_in   = OpVariable %ptr_in_arr Input

               ; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv  = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4  = OpTypePointer Output %v4float
   %gl_out   = OpVariable %ptr_out_pv Output

               ; Input color (location 2, array of 1 vec3)
%arr_v3_1    = OpTypeArray %v3float %uint_1
%ptr_in_arr3 = OpTypePointer Input %arr_v3_1
%ptr_in_v3   = OpTypePointer Input %v3float
  %inColor   = OpVariable %ptr_in_arr3 Input

               ; Output color (location 0, vec3)
%ptr_out_v3  = OpTypePointer Output %v3float
 %outColor   = OpVariable %ptr_out_v3 Output

               ; Main function
     %main   = OpFunction %void None %voidfn
    %entry   = OpLabel

               ; Load input position: gl_in[0].gl_Position
 %pos_ptr    = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
 %pos        = OpLoad %v4float %pos_ptr

               ; Load input color: inColor[0]
 %col_ptr    = OpAccessChain %ptr_in_v3 %inColor %int_0
 %col        = OpLoad %v3float %col_ptr

               ; Extract x, y, z, w from position
 %px         = OpCompositeExtract %float %pos 0
 %py         = OpCompositeExtract %float %pos 1
 %pz         = OpCompositeExtract %float %pos 2
 %pw         = OpCompositeExtract %float %pos 3

               ; Compute diamond vertices
               ; Top: (x, y+h, z, w)
 %top_y      = OpFAdd %float %py %float_h
 %top_pos    = OpCompositeConstruct %v4float %px %top_y %pz %pw

               ; Right: (x+h, y, z, w)
 %right_x    = OpFAdd %float %px %float_h
 %right_pos  = OpCompositeConstruct %v4float %right_x %py %pz %pw

               ; Left: (x-h, y, z, w)
 %left_x     = OpFAdd %float %px %float_nh
 %left_pos   = OpCompositeConstruct %v4float %left_x %py %pz %pw

               ; Bottom: (x, y-h, z, w)
 %bot_y      = OpFAdd %float %py %float_nh
 %bot_pos    = OpCompositeConstruct %v4float %px %bot_y %pz %pw

               ; Emit vertex 1: Top
 %out_ptr1   = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_ptr1 %top_pos
               OpStore %outColor %col
               OpEmitVertex

               ; Emit vertex 2: Right
 %out_ptr2   = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_ptr2 %right_pos
               OpStore %outColor %col
               OpEmitVertex

               ; Emit vertex 3: Left
 %out_ptr3   = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_ptr3 %left_pos
               OpStore %outColor %col
               OpEmitVertex

               ; Emit vertex 4: Bottom
 %out_ptr4   = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_ptr4 %bot_pos
               OpStore %outColor %col
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
