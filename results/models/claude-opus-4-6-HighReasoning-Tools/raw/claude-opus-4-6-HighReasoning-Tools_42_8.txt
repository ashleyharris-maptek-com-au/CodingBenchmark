; Geometry shader: shrink triangles toward centroid
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
               OpExecutionMode %main Triangles
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               OpDecorate %in_color Location 2
               OpDecorate %out_color Location 0

               ; Types
      %void = OpTypeVoid
  %void_fn = OpTypeFunction %void
     %float = OpTypeFloat 32
   %v3float = OpTypeVector %float 3
   %v4float = OpTypeVector %float 4
      %uint = OpTypeInt 32 0
       %int = OpTypeInt 32 1

               ; Constants
     %int_0 = OpConstant %int 0
     %int_1 = OpConstant %int 1
     %int_2 = OpConstant %int 2
    %uint_3 = OpConstant %uint 3
  %float_0  = OpConstant %float 0.0
  %float_1  = OpConstant %float 1.0
  %float_3  = OpConstant %float 3.0
  %float_07 = OpConstant %float 0.7

               ; Input gl_PerVertex block (array of 3 for Triangles)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in_3 = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr  = OpTypePointer Input %arr_pv_in_3
     %gl_in  = OpVariable %ptr_in_arr Input
%ptr_in_v4   = OpTypePointer Input %v4float

               ; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv  = OpTypePointer Output %gl_PerVertex_out
    %gl_out  = OpVariable %ptr_out_pv Output
%ptr_out_v4  = OpTypePointer Output %v4float

               ; Input color array (location 2, vec3, array of 3)
%arr_v3_3    = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_3
  %in_color  = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3   = OpTypePointer Input %v3float

               ; Output color (location 0, vec3)
%ptr_out_v3  = OpTypePointer Output %v3float
 %out_color  = OpVariable %ptr_out_v3 Output

               ; Main function
      %main  = OpFunction %void None %void_fn
     %entry  = OpLabel

               ; Load positions from gl_in[0..2].gl_Position
  %ptr_p0    = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
      %p0    = OpLoad %v4float %ptr_p0
  %ptr_p1    = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
      %p1    = OpLoad %v4float %ptr_p1
  %ptr_p2    = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
      %p2    = OpLoad %v4float %ptr_p2

               ; Load colors from in_color[0..2]
  %ptr_c0    = OpAccessChain %ptr_in_v3 %in_color %int_0
      %c0    = OpLoad %v3float %ptr_c0
  %ptr_c1    = OpAccessChain %ptr_in_v3 %in_color %int_1
      %c1    = OpLoad %v3float %ptr_c1
  %ptr_c2    = OpAccessChain %ptr_in_v3 %in_color %int_2
      %c2    = OpLoad %v3float %ptr_c2

               ; Compute centroid C = (p0 + p1 + p2) / 3
    %sum01   = OpFAdd %v4float %p0 %p1
   %sum012   = OpFAdd %v4float %sum01 %p2
   %inv3_vec = OpCompositeConstruct %v4float %float_3 %float_3 %float_3 %float_3
  %centroid  = OpFDiv %v4float %sum012 %inv3_vec

               ; Shrink factor s = 0.7
   %s_vec    = OpCompositeConstruct %v4float %float_07 %float_07 %float_07 %float_07

               ; Vertex 0: new_pos = C + s * (p0 - C)
   %diff0    = OpFSub %v4float %p0 %centroid
 %scaled0    = OpFMul %v4float %s_vec %diff0
   %new0     = OpFAdd %v4float %centroid %scaled0
  %out_ptr0  = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_ptr0 %new0
               OpStore %out_color %c0
               OpEmitVertex

               ; Vertex 1: new_pos = C + s * (p1 - C)
   %diff1    = OpFSub %v4float %p1 %centroid
 %scaled1    = OpFMul %v4float %s_vec %diff1
   %new1     = OpFAdd %v4float %centroid %scaled1
  %out_ptr1  = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_ptr1 %new1
               OpStore %out_color %c1
               OpEmitVertex

               ; Vertex 2: new_pos = C + s * (p2 - C)
   %diff2    = OpFSub %v4float %p2 %centroid
 %scaled2    = OpFMul %v4float %s_vec %diff2
   %new2     = OpFAdd %v4float %centroid %scaled2
  %out_ptr2  = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_ptr2 %new2
               OpStore %out_color %c2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
