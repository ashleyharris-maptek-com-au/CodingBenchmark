#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#include <chrono>
#include <random>
#include <unordered_map>
#include <cmath>

using namespace std;
using namespace chrono;

static int J, M, N;
static vector<int> gJ, gT, gM, gD, oO;
static vector<vector<pair<int,int>>> jobs;

int computeSchedule(vector<vector<int>>& mo, vector<int>& st) {
    st.assign(N, 0);
    vector<int> inD(N, 0), mp(N, -1), mn(N, -1);
    for (int m = 0; m < M; m++) {
        for (int i = 1; i < (int)mo[m].size(); i++) {
            mp[mo[m][i]] = mo[m][i-1];
            inD[mo[m][i]]++;
        }
        for (int i = 0; i+1 < (int)mo[m].size(); i++)
            mn[mo[m][i]] = mo[m][i+1];
    }
    for (int j = 0; j < J; j++)
        for (int t = 1; t < (int)jobs[j].size(); t++)
            inD[oO[j]+t]++;

    queue<int> q;
    for (int i = 0; i < N; i++)
        if (!inD[i]) q.push(i);

    int ms = 0, cnt = 0;
    while (!q.empty()) {
        int u = q.front(); q.pop(); cnt++;
        int s = 0;
        if (gT[u] > 0) {
            int p = oO[gJ[u]] + gT[u] - 1;
            s = st[p] + gD[p];
        }
        if (mp[u] >= 0)
            s = max(s, st[mp[u]] + gD[mp[u]]);
        st[u] = s;
        ms = max(ms, s + gD[u]);
        if (gT[u]+1 < (int)jobs[gJ[u]].size()) {
            int nx = oO[gJ[u]] + gT[u] + 1;
            if (--inD[nx] == 0) q.push(nx);
        }
        if (mn[u] >= 0 && --inD[mn[u]] == 0)
            q.push(mn[u]);
    }
    return cnt == N ? ms : -1;
}

vector<pair<int,int>> findSwaps(vector<vector<int>>& mo, vector<int>& st) {
    vector<int> mp(N, -1), mn(N, -1);
    for (int m = 0; m < M; m++) {
        for (int i = 1; i < (int)mo[m].size(); i++)
            mp[mo[m][i]] = mo[m][i-1];
        for (int i = 0; i+1 < (int)mo[m].size(); i++)
            mn[mo[m][i]] = mo[m][i+1];
    }

    int ms = 0, last = -1;
    for (int i = 0; i < N; i++)
        if (st[i] + gD[i] > ms) { ms = st[i] + gD[i]; last = i; }
    if (last < 0) return {};

    vector<int> path;
    int cur = last;
    while (cur >= 0) {
        path.push_back(cur);
        int nxt = -1;
        if (mp[cur] >= 0 && st[mp[cur]] + gD[mp[cur]] == st[cur])
            nxt = mp[cur];
        else if (gT[cur] > 0) {
            int p = oO[gJ[cur]] + gT[cur] - 1;
            if (st[p] + gD[p] == st[cur]) nxt = p;
        }
        cur = nxt;
    }
    reverse(path.begin(), path.end());

    vector<pair<int,int>> swaps;
    int bs = 0;
    for (int i = 1; i <= (int)path.size(); i++) {
        bool same = false;
        if (i < (int)path.size()) {
            int u = path[i-1], v = path[i];
            if (gM[u] == gM[v] && mn[u] == v) same = true;
        }
        if (!same) {
            int be = i - 1, blen = be - bs + 1;
            if (blen >= 2) {
                int u = path[bs], v = path[bs+1];
                if (gJ[u] != gJ[v]) swaps.push_back({u, v});
                if (blen >= 3) {
                    u = path[be-1]; v = path[be];
                    if (gJ[u] != gJ[v]) swaps.push_back({u, v});
                }
            }
            bs = i;
        }
    }
    return swaps;
}

vector<vector<int>> buildGreedy(int rule, mt19937* rng = nullptr) {
    vector<vector<int>> mo(M);
    vector<int> jnxt(J, 0), jrdy(J, 0), mrdy(M, 0), rw(J, 0);
    for (int j = 0; j < J; j++)
        for (auto& p : jobs[j]) rw[j] += p.second;

    for (int it = 0; it < N; it++) {
        struct C { int j, s; long long p; };
        vector<C> cs;
        int minS = INT_MAX;

        for (int j = 0; j < J; j++) {
            int t = jnxt[j];
            if (t >= (int)jobs[j].size()) continue;
            int m = jobs[j][t].first, d = jobs[j][t].second;
            int s = max(jrdy[j], mrdy[m]);
            minS = min(minS, s);
            long long pr;
            switch(rule) {
                case 0: pr = d; break;
                case 1: pr = -d; break;
                case 2: pr = -(long long)rw[j]; break;
                case 3: pr = rw[j]; break;
                case 4: pr = j; break;
                case 5: pr = -(long long)((int)jobs[j].size()-t); break;
                default: pr = 0; break;
            }
            cs.push_back({j, s, pr});
        }

        int bi = 0;
        if (rng && (int)cs.size() > 1) {
            vector<int> el;
            for (int i = 0; i < (int)cs.size(); i++)
                if (cs[i].s <= minS) el.push_back(i);
            sort(el.begin(), el.end(), [&](int a, int b){ return cs[a].p < cs[b].p; });
            int k = min((int)el.size(), 3);
            bi = el[(*rng)() % k];
        } else {
            for (int i = 1; i < (int)cs.size(); i++)
                if (cs[i].s < cs[bi].s || (cs[i].s == cs[bi].s && cs[i].p < cs[bi].p))
                    bi = i;
        }

        int j = cs[bi].j, t = jnxt[j];
        int m = jobs[j][t].first, d = jobs[j][t].second;
        int s = max(jrdy[j], mrdy[m]);
        mo[m].push_back(oO[j]+t);
        jnxt[j]++; jrdy[j] = s+d; mrdy[m] = s+d; rw[j] -= d;
    }
    return mo;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    auto t0 = steady_clock::now();
    auto ems = [&]() -> long long {
        return duration_cast<milliseconds>(steady_clock::now()-t0).count();
    };

    cin >> J >> M;
    jobs.resize(J);
    N = 0;
    for (int j = 0; j < J; j++) {
        int T; cin >> T;
        jobs[j].resize(T);
        for (int t = 0; t < T; t++)
            cin >> jobs[j][t].first >> jobs[j][t].second;
        N += T;
    }

    if (N == 0) {
        for (int j = 0; j < J; j++) cout << '\n';
        return 0;
    }

    oO.resize(J, 0);
    for (int j = 1; j < J; j++) oO[j] = oO[j-1]+(int)jobs[j-1].size();
    gJ.resize(N); gT.resize(N); gM.resize(N); gD.resize(N);
    for (int j = 0; j < J; j++)
        for (int t = 0; t < (int)jobs[j].size(); t++) {
            int id = oO[j]+t;
            gJ[id]=j; gT[id]=t; gM[id]=jobs[j][t].first; gD[id]=jobs[j][t].second;
        }

    vector<vector<int>> bestMO;
    vector<int> bestSt;
    int bestMs = INT_MAX;

    for (int r = 0; r <= 6; r++) {
        auto mo = buildGreedy(r);
        vector<int> st;
        int ms = computeSchedule(mo, st);
        if (ms >= 0 && ms < bestMs) { bestMs = ms; bestMO = mo; bestSt = st; }
    }

    long long tLim;
    if (N <= 50) tLim = 1000;
    else if (N <= 500) tLim = 3000;
    else if (N <= 5000) tLim = 8000;
    else tLim = 25000;

    {
        auto mo = bestMO;
        auto st = bestSt;
        int ms = bestMs;
        int tenure = max(7, min((int)sqrt((double)N), 50));

        unordered_map<long long, int> tabu;
        auto tk = [](int u, int v) -> long long {
            return (long long)min(u,v)*200001LL + max(u,v);
        };

        vector<int> posM(N);
        for (int m = 0; m < M; m++)
            for (int i = 0; i < (int)mo[m].size(); i++)
                posM[mo[m][i]] = i;

        long long tabuLim = ems() + (tLim - ems()) * 7 / 10;
        int noImpr = 0, maxNoImpr = min(3000, max(100, N*2));

        for (int iter = 0; ems() < tabuLim && noImpr < maxNoImpr; iter++) {
            auto swaps = findSwaps(mo, st);
            if (swaps.empty()) break;

            int bestI = -1, bestNewMs = INT_MAX;
            vector<int> bestNewSt;

            for (int i = 0; i < (int)swaps.size(); i++) {
                auto [u,v] = swaps[i];
                if (gJ[u] == gJ[v]) continue;
                int m2 = gM[u];
                int pu = posM[u], pv = posM[v];

                swap(mo[m2][pu], mo[m2][pv]);
                vector<int> tst;
                int tms = computeSchedule(mo, tst);
                swap(mo[m2][pu], mo[m2][pv]);

                if (tms < 0) continue;
                bool isTabu = tabu.count(tk(u,v)) && tabu[tk(u,v)] > iter;
                bool aspire = tms < bestMs;

                if (!isTabu || aspire) {
                    if (tms < bestNewMs) {
                        bestI = i; bestNewMs = tms; bestNewSt = tst;
                    }
                }
            }

            if (bestI < 0) break;

            auto [u,v] = swaps[bestI];
            int m2 = gM[u];
            int pu = posM[u], pv = posM[v];
            swap(mo[m2][pu], mo[m2][pv]);
            posM[u] = pv; posM[v] = pu;
            ms = bestNewMs;
            st = bestNewSt;
            tabu[tk(u,v)] = iter + tenure;

            if (ms < bestMs) {
                bestMs = ms; bestMO = mo; bestSt = st;
                noImpr = 0;
            } else {
                noImpr++;
            }
        }
    }

    mt19937 rng(42);
    while (ems() < tLim - 200) {
        int rule = (int)(rng() % 7);
        auto mo = buildGreedy(rule, &rng);
        vector<int> st;
        int ms = computeSchedule(mo, st);
        if (ms < 0) continue;

        long long hcLim = min(ems() + 500, tLim - 100);
        bool improved = true;
        while (improved && ems() < hcLim) {
            improved = false;
            auto swaps = findSwaps(mo, st);
            vector<int> pm(N);
            for (int m = 0; m < M; m++)
                for (int i = 0; i < (int)mo[m].size(); i++)
                    pm[mo[m][i]] = i;

            for (auto [u,v] : swaps) {
                if (gJ[u] == gJ[v]) continue;
                int m2 = gM[u];
                int pu = pm[u], pv = pm[v];
                swap(mo[m2][pu], mo[m2][pv]);
                vector<int> tst;
                int tms = computeSchedule(mo, tst);
                if (tms >= 0 && tms < ms) {
                    ms = tms; st = tst;
                    pm[u] = pv; pm[v] = pu;
                    improved = true;
                    break;
                }
                swap(mo[m2][pu], mo[m2][pv]);
            }
        }

        if (ms < bestMs) {
            bestMs = ms; bestMO = mo; bestSt = st;
        }
    }

    for (int j = 0; j < J; j++) {
        for (int t = 0; t < (int)jobs[j].size(); t++) {
            if (t) cout << ' ';
            cout << bestSt[oO[j]+t] << ' ' << jobs[j][t].first;
        }
        cout << '\n';
    }
    return 0;
}
