import struct

words = []

def w(val):
    words.append(val)

# Header
w(0x07230203)  # Magic
w(0x00010000)  # Version 1.0
w(0x00000000)  # Generator
w(18)          # Bound (IDs 1-17)
w(0x00000000)  # Reserved

# IDs:
# 1=void, 2=func_type, 3=float, 4=vec2, 5=vec4
# 6=ptr_input_vec2, 7=ptr_output_vec4
# 8=uv_var, 9=outColor_var, 10=main, 11=label
# 12=loaded_uv, 13=uv_x, 14=uv_y, 15=const_0, 16=const_1, 17=result_vec4

# OpCapability Shader
w((2 << 16) | 17); w(1)

# OpMemoryModel Logical GLSL450
w((3 << 16) | 14); w(0); w(1)

# OpEntryPoint Fragment %10 "main" %8 %9
# "main\0" = 5 bytes -> 2 words
w((7 << 16) | 15); w(4); w(10)
w(0x6E69616D)  # "main" as little-endian word
w(0x00000000)  # null terminator + padding
w(8); w(9)

# OpExecutionMode %10 OriginUpperLeft(7)
w((3 << 16) | 16); w(10); w(7)

# Decorations
# OpDecorate %8 Location 2
w((4 << 16) | 71); w(8); w(30); w(2)
# OpDecorate %9 Location 0
w((4 << 16) | 71); w(9); w(30); w(0)

# Type declarations
# OpTypeVoid %1
w((2 << 16) | 19); w(1)
# OpTypeFunction %2 %1
w((3 << 16) | 33); w(2); w(1)
# OpTypeFloat %3 32
w((3 << 16) | 22); w(3); w(32)
# OpTypeVector %4 %3 2
w((4 << 16) | 23); w(4); w(3); w(2)
# OpTypeVector %5 %3 4
w((4 << 16) | 23); w(5); w(3); w(4)
# OpTypePointer %6 Input(1) %4
w((4 << 16) | 32); w(6); w(1); w(4)
# OpTypePointer %7 Output(3) %5
w((4 << 16) | 32); w(7); w(3); w(5)

# Constants
# OpConstant %3 %15 0.0
w((4 << 16) | 43); w(3); w(15); w(0x00000000)
# OpConstant %3 %16 1.0
w((4 << 16) | 43); w(3); w(16); w(0x3F800000)

# Variables
# OpVariable %6 %8 Input(1)
w((4 << 16) | 59); w(6); w(8); w(1)
# OpVariable %7 %9 Output(3)
w((4 << 16) | 59); w(7); w(9); w(3)

# Function
# OpFunction %1 %10 None(0) %2
w((5 << 16) | 54); w(1); w(10); w(0); w(2)
# OpLabel %11
w((2 << 16) | 248); w(11)
# OpLoad %4 %12 %8
w((4 << 16) | 61); w(4); w(12); w(8)
# OpCompositeExtract %3 %13 %12 0
w((5 << 16) | 81); w(3); w(13); w(12); w(0)
# OpCompositeExtract %3 %14 %12 1
w((5 << 16) | 81); w(3); w(14); w(12); w(1)
# OpCompositeConstruct %5 %17 %13 %14 %15 %16
w((7 << 16) | 80); w(5); w(17); w(13); w(14); w(15); w(16)
# OpStore %9 %17
w((3 << 16) | 62); w(9); w(17)
# OpReturn
w((1 << 16) | 253)
# OpFunctionEnd
w((1 << 16) | 56)

# Convert to little-endian hex
result = b''
for word in words:
    result += struct.pack('<I', word)

hex_str = result.hex()
print(hex_str)

with open('answer.txt', 'w') as f:
    f.write(hex_str)
