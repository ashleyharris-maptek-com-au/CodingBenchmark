import struct

words = []

def w(val):
    words.append(val)

def op(wordcount, opcode, *operands):
    words.append((wordcount << 16) | opcode)
    for o in operands:
        words.append(o)

def string_words(s):
    """Encode a string as SPIR-V literal string words (null-terminated, padded to 4 bytes)."""
    b = s.encode('utf-8') + b'\x00'
    while len(b) % 4 != 0:
        b += b'\x00'
    result = []
    for i in range(0, len(b), 4):
        result.append(struct.unpack('<I', b[i:i+4])[0])
    return result

def float_to_uint(f):
    return struct.unpack('<I', struct.pack('<f', f))[0]

# IDs
EXT_GLSL = 1
TYPE_VOID = 2
TYPE_FUNC = 3
TYPE_FLOAT = 4
TYPE_VEC3 = 5
TYPE_VEC4 = 6
PTR_INPUT_VEC3 = 7
PTR_OUTPUT_VEC4 = 8
VAR_WORLDPOS = 9
VAR_OUTCOLOR = 10
FUNC_MAIN = 11
CONST_HALF = 12
CONST_ZERO = 13
CONST_ONE = 14
CONST_01 = 15
CONST_04 = 16
LABEL = 17
LOAD_WP = 18
EXTRACT_Y = 19
MUL_RESULT = 20
ADD_RESULT = 21
CLAMP_RESULT = 22
VEC_BOTTOM = 23
VEC_TOP = 24
VEC_T = 25
MIX_RESULT = 26
EX_R = 27
EX_G = 28
EX_B = 29
OUT_VEC4 = 30
BOUND = 31

# Header
w(0x07230203)  # magic
w(0x00010000)  # version 1.0
w(0)           # generator
w(BOUND)       # bound
w(0)           # reserved

# OpCapability Shader (1)
op(2, 17, 1)

# OpExtInstImport %1 "GLSL.std.450"
name_w = string_words("GLSL.std.450")
op(2 + len(name_w), 11, EXT_GLSL, *name_w)

# OpMemoryModel Logical GLSL450
op(3, 14, 0, 1)

# OpEntryPoint Fragment %11 "main" %9 %10
name_main = string_words("main")
op(3 + len(name_main) + 2, 15, 4, FUNC_MAIN, *name_main, VAR_WORLDPOS, VAR_OUTCOLOR)

# OpExecutionMode %11 OriginUpperLeft(7)
op(3, 16, FUNC_MAIN, 7)

# Decorations
# OpDecorate %9 Location 0
op(4, 71, VAR_WORLDPOS, 30, 0)
# OpDecorate %10 Location 0
op(4, 71, VAR_OUTCOLOR, 30, 0)

# Type declarations
# OpTypeVoid %2
op(2, 19, TYPE_VOID)
# OpTypeFunction %3 %2
op(3, 33, TYPE_FUNC, TYPE_VOID)
# OpTypeFloat %4 32
op(3, 22, TYPE_FLOAT, 32)
# OpTypeVector %5 %4 3
op(4, 23, TYPE_VEC3, TYPE_FLOAT, 3)
# OpTypeVector %6 %4 4
op(4, 23, TYPE_VEC4, TYPE_FLOAT, 4)
# OpTypePointer %7 Input(1) %5
op(4, 32, PTR_INPUT_VEC3, 1, TYPE_VEC3)
# OpTypePointer %8 Output(3) %6
op(4, 32, PTR_OUTPUT_VEC4, 3, TYPE_VEC4)

# Variables
# OpVariable %7 %9 Input(1)
op(4, 59, PTR_INPUT_VEC3, VAR_WORLDPOS, 1)
# OpVariable %8 %10 Output(3)
op(4, 59, PTR_OUTPUT_VEC4, VAR_OUTCOLOR, 3)

# Constants
# OpConstant %4 %12 0.5
op(4, 43, TYPE_FLOAT, CONST_HALF, float_to_uint(0.5))
# OpConstant %4 %13 0.0
op(4, 43, TYPE_FLOAT, CONST_ZERO, float_to_uint(0.0))
# OpConstant %4 %14 1.0
op(4, 43, TYPE_FLOAT, CONST_ONE, float_to_uint(1.0))
# OpConstant %4 %15 0.1
op(4, 43, TYPE_FLOAT, CONST_01, float_to_uint(0.1))
# OpConstant %4 %16 0.4
op(4, 43, TYPE_FLOAT, CONST_04, float_to_uint(0.4))

# Function
# OpFunction %2 %11 None(0) %3
op(5, 54, TYPE_VOID, FUNC_MAIN, 0, TYPE_FUNC)
# OpLabel %17
op(2, 248, LABEL)

# OpLoad %5 %18 %9
op(4, 61, TYPE_VEC3, LOAD_WP, VAR_WORLDPOS)
# OpCompositeExtract %4 %19 %18 1
op(5, 81, TYPE_FLOAT, EXTRACT_Y, LOAD_WP, 1)
# OpFMul %4 %20 %19 %12
op(5, 133, TYPE_FLOAT, MUL_RESULT, EXTRACT_Y, CONST_HALF)
# OpFAdd %4 %21 %20 %12
op(5, 129, TYPE_FLOAT, ADD_RESULT, MUL_RESULT, CONST_HALF)
# OpExtInst %4 %22 %1 FClamp(43) %21 %13 %14
op(8, 12, TYPE_FLOAT, CLAMP_RESULT, EXT_GLSL, 43, ADD_RESULT, CONST_ZERO, CONST_ONE)

# Construct bottom and top vec3
# OpCompositeConstruct %5 %23 %15 %16 %15
op(6, 80, TYPE_VEC3, VEC_BOTTOM, CONST_01, CONST_04, CONST_01)
# OpCompositeConstruct %5 %24 %14 %14 %14
op(6, 80, TYPE_VEC3, VEC_TOP, CONST_ONE, CONST_ONE, CONST_ONE)
# OpCompositeConstruct %5 %25 %22 %22 %22
op(6, 80, TYPE_VEC3, VEC_T, CLAMP_RESULT, CLAMP_RESULT, CLAMP_RESULT)

# OpExtInst %5 %26 %1 FMix(46) %23 %24 %25
op(8, 12, TYPE_VEC3, MIX_RESULT, EXT_GLSL, 46, VEC_BOTTOM, VEC_TOP, VEC_T)

# Extract components and build vec4
# OpCompositeExtract %4 %27 %26 0
op(5, 81, TYPE_FLOAT, EX_R, MIX_RESULT, 0)
# OpCompositeExtract %4 %28 %26 1
op(5, 81, TYPE_FLOAT, EX_G, MIX_RESULT, 1)
# OpCompositeExtract %4 %29 %26 2
op(5, 81, TYPE_FLOAT, EX_B, MIX_RESULT, 2)
# OpCompositeConstruct %6 %30 %27 %28 %29 %14
op(7, 80, TYPE_VEC4, OUT_VEC4, EX_R, EX_G, EX_B, CONST_ONE)

# OpStore %10 %30
op(3, 62, VAR_OUTCOLOR, OUT_VEC4)
# OpReturn
op(1, 253)
# OpFunctionEnd
op(1, 56)

# Encode to bytes
data = b''
for word in words:
    data += struct.pack('<I', word)

hex_str = data.hex()
print(hex_str)
