import sys
import math
import random

def solve():
    data = sys.stdin.read().split()
    idx = 0
    def next_int():
        nonlocal idx; v = int(data[idx]); idx += 1; return v
    def next_float():
        nonlocal idx; v = float(data[idx]); idx += 1; return v

    N = next_int(); K = next_int(); CAP = next_int()
    depot = (next_float(), next_float())
    customers = []
    demands = []
    for _ in range(N):
        x, y, d = next_float(), next_float(), next_int()
        customers.append((x, y))
        demands.append(d)

    def dist(a, b):
        return math.hypot(a[0]-b[0], a[1]-b[1])

    dist_depot = [dist(depot, c) for c in customers]
    dist_matrix = [[0.0]*N for _ in range(N)]
    for i in range(N):
        for j in range(i+1, N):
            d = dist(customers[i], customers[j])
            dist_matrix[i][j] = d
            dist_matrix[j][i] = d

    def route_dist(route):
        if not route:
            return 0.0
        d = dist_depot[route[0]] + dist_depot[route[-1]]
        for i in range(len(route)-1):
            d += dist_matrix[route[i]][route[i+1]]
        return d

    def route_demand(route):
        return sum(demands[c] for c in route)

    # Clarke-Wright Savings Algorithm
    savings = []
    for i in range(N):
        for j in range(i+1, N):
            s = dist_depot[i] + dist_depot[j] - dist_matrix[i][j]
            savings.append((s, i, j))
    savings.sort(reverse=True)

    route_of = list(range(N))
    routes = {i: [i] for i in range(N)}
    route_demands = {i: demands[i] for i in range(N)}

    for s, i, j in savings:
        ri, rj = route_of[i], route_of[j]
        if ri == rj:
            continue
        if route_demands[ri] + route_demands[rj] > CAP:
            continue
        r_i = routes[ri]
        r_j = routes[rj]
        if r_i[-1] == i and r_j[0] == j:
            new_route = r_i + r_j
        elif r_i[0] == i and r_j[-1] == j:
            new_route = r_j + r_i
        elif r_i[-1] == i and r_j[-1] == j:
            new_route = r_i + r_j[::-1]
        elif r_i[0] == i and r_j[0] == j:
            new_route = r_i[::-1] + r_j
        else:
            continue
        new_demand = route_demands[ri] + route_demands[rj]
        del routes[ri]
        del routes[rj]
        new_id = id(new_route)
        routes[new_id] = new_route
        route_demands[new_id] = new_demand
        for c in new_route:
            route_of[c] = new_id

    result_routes = list(routes.values())

    # Split routes that exceed K vehicles by merging smallest
    while len(result_routes) > K:
        result_routes.sort(key=route_demand)
        r1 = result_routes[0]
        best_merge = None
        best_cost = float('inf')
        for idx2 in range(1, len(result_routes)):
            r2 = result_routes[idx2]
            if route_demand(r1) + route_demand(r2) <= CAP:
                cost = route_dist(r1 + r2)
                if cost < best_cost:
                    best_cost = cost
                    best_merge = idx2
        if best_merge is not None:
            r2 = result_routes[best_merge]
            merged = r1 + r2
            result_routes.pop(best_merge)
            result_routes.pop(0)
            result_routes.append(merged)
        else:
            break

    # 2-opt improvement for each route
    def two_opt(route):
        if len(route) < 2:
            return route
        improved = True
        route = list(route)
        while improved:
            improved = False
            for i in range(len(route) - 1):
                for j in range(i + 1, len(route)):
                    if i == 0:
                        old = dist_depot[route[i]]
                    else:
                        old = dist_matrix[route[i-1]][route[i]]
                    if j == len(route) - 1:
                        old += dist_depot[route[j]]
                    else:
                        old += dist_matrix[route[j]][route[j+1]]

                    if i == 0:
                        new = dist_depot[route[j]]
                    else:
                        new = dist_matrix[route[i-1]][route[j]]
                    if j == len(route) - 1:
                        new += dist_depot[route[i]]
                    else:
                        new += dist_matrix[route[i]][route[j+1]]

                    if new < old - 1e-10:
                        route[i:j+1] = route[i:j+1][::-1]
                        improved = True
        return route

    result_routes = [two_opt(r) for r in result_routes]

    # Inter-route: try relocating each customer to a better route
    def try_relocate(routes_list):
        improved = True
        while improved:
            improved = False
            for ri in range(len(routes_list)):
                for pos in range(len(routes_list[ri])):
                    c = routes_list[ri][pos]
                    old_route = routes_list[ri]
                    old_cost = route_dist(old_route)
                    new_source = old_route[:pos] + old_route[pos+1:]
                    new_source_cost = route_dist(new_source)
                    saving = old_cost - new_source_cost

                    best_gain = 0
                    best_rj = -1
                    best_insert = -1
                    for rj in range(len(routes_list)):
                        if rj == ri:
                            continue
                        if route_demand(routes_list[rj]) + demands[c] > CAP:
                            continue
                        dest = routes_list[rj]
                        dest_cost = route_dist(dest)
                        for ins in range(len(dest)+1):
                            new_dest = dest[:ins] + [c] + dest[ins:]
                            new_dest_cost = route_dist(new_dest)
                            insertion_cost = new_dest_cost - dest_cost
                            gain = saving - insertion_cost
                            if gain > best_gain + 1e-10:
                                best_gain = gain
                                best_rj = rj
                                best_insert = ins
                    if best_rj >= 0:
                        dest = routes_list[best_rj]
                        routes_list[best_rj] = dest[:best_insert] + [c] + dest[best_insert:]
                        routes_list[ri] = new_source
                        if not routes_list[ri]:
                            routes_list.pop(ri)
                        improved = True
                        break
                if improved:
                    break

    if N <= 500:
        try_relocate(result_routes)
        result_routes = [two_opt(r) for r in result_routes]

    # Or-opt: move segments of 1-2-3 customers within a route
    def or_opt_route(route):
        if len(route) < 3:
            return route
        route = list(route)
        improved = True
        while improved:
            improved = False
            for seg_len in (1, 2, 3):
                for i in range(len(route) - seg_len + 1):
                    seg = route[i:i+seg_len]
                    rest = route[:i] + route[i+seg_len:]
                    if not rest:
                        continue
                    base_cost = route_dist(route)
                    best_cost = base_cost
                    best_pos = -1
                    for j in range(len(rest)+1):
                        candidate = rest[:j] + seg + rest[j:]
                        c = route_dist(candidate)
                        if c < best_cost - 1e-10:
                            best_cost = c
                            best_pos = j
                    if best_pos >= 0:
                        route = rest[:best_pos] + seg + rest[best_pos:]
                        improved = True
                        break
                if improved:
                    break
        return route

    if N <= 300:
        result_routes = [or_opt_route(r) for r in result_routes]

    # Remove empty routes
    result_routes = [r for r in result_routes if r]

    total = sum(route_dist(r) for r in result_routes)
    print(f"{total:.6f}")
    print(len(result_routes))
    for r in result_routes:
        print(" ".join(str(c) for c in r))

solve()
