import struct

def le32(val):
    return struct.pack('<I', val & 0xFFFFFFFF)

def encode_string(s):
    b = s.encode('utf-8') + b'\x00'
    while len(b) % 4 != 0:
        b += b'\x00'
    return b

def inst(opcode, *args):
    payload = b''
    for a in args:
        if isinstance(a, bytes):
            payload += a
        elif isinstance(a, float):
            payload += struct.pack('<f', a)
        else:
            payload += struct.pack('<I', a & 0xFFFFFFFF)
    wc = 1 + len(payload) // 4
    return struct.pack('<I', (wc << 16) | opcode) + payload

# IDs
GLSL=1; T_VOID=2; T_FLOAT=3; T_VEC3=4; T_VEC4=5; T_MAT4=6; T_FN=7
T_PI3=8; T_PO4=9; T_INT=10; T_STRUCT=11; T_PUS=12; T_PUV4=13
V_WP=14; V_N=15; V_OUT=16; V_UBO=17
C0=18; C05=19; C1=20; C04=21; C06=22; C03=23; C015=24; C005=25; IC3=26
FN=27; LBL=28
LN=29; LWP=30; NY=31; NY05=32; BL=33; BLV=34; SKY=35; GND=36
SG=37; SC=38; AMB=39; LP=40; LL=41; LV3=42; LDR=43; LDN=44
DT=45; DC=46; DS=47; DV=48; FIN=49; FX=50; FY=51; FZ=52; OUTVEC=53
BOUND=54

spv = b''
# Header
spv += le32(0x07230203)
spv += le32(0x00010000)
spv += le32(0)
spv += le32(BOUND)
spv += le32(0)

# Preamble
spv += inst(17, 1)  # OpCapability Shader
spv += inst(11, GLSL, encode_string("GLSL.std.450"))
spv += inst(14, 0, 1)  # OpMemoryModel Logical GLSL450
spv += inst(15, 4, FN, encode_string("main"), V_WP, V_N, V_OUT)
spv += inst(16, FN, 7)  # OpExecutionMode OriginUpperLeft

# Decorations
spv += inst(71, V_WP, 30, 0)   # Location 0
spv += inst(71, V_N, 30, 1)    # Location 1
spv += inst(71, V_OUT, 30, 0)  # Location 0 output
spv += inst(71, V_UBO, 34, 0)  # DescriptorSet 0
spv += inst(71, V_UBO, 33, 0)  # Binding 0
spv += inst(71, T_STRUCT, 2)   # Block

# MemberDecorate mat4 members
for i, off in [(0,0),(1,64),(2,128)]:
    spv += inst(72, T_STRUCT, i, 35, off)  # Offset
    spv += inst(72, T_STRUCT, i, 5)         # ColMajor
    spv += inst(72, T_STRUCT, i, 7, 16)     # MatrixStride 16
# MemberDecorate vec4 members
for i, off in [(3,192),(4,208),(5,224)]:
    spv += inst(72, T_STRUCT, i, 35, off)

# Types
spv += inst(19, T_VOID)
spv += inst(22, T_FLOAT, 32)
spv += inst(21, T_INT, 32, 1)
spv += inst(23, T_VEC3, T_FLOAT, 3)
spv += inst(23, T_VEC4, T_FLOAT, 4)
spv += inst(24, T_MAT4, T_VEC4, 4)
spv += inst(30, T_STRUCT, T_MAT4, T_MAT4, T_MAT4, T_VEC4, T_VEC4, T_VEC4)
spv += inst(33, T_FN, T_VOID)
spv += inst(32, T_PI3, 1, T_VEC3)
spv += inst(32, T_PO4, 3, T_VEC4)
spv += inst(32, T_PUS, 2, T_STRUCT)
spv += inst(32, T_PUV4, 2, T_VEC4)

# Variables
spv += inst(59, T_PI3, V_WP, 1)
spv += inst(59, T_PI3, V_N, 1)
spv += inst(59, T_PO4, V_OUT, 3)
spv += inst(59, T_PUS, V_UBO, 2)

# Constants
spv += inst(43, T_FLOAT, C0, 0.0)
spv += inst(43, T_FLOAT, C05, 0.5)
spv += inst(43, T_FLOAT, C1, 1.0)
spv += inst(43, T_FLOAT, C04, 0.4)
spv += inst(43, T_FLOAT, C06, 0.6)
spv += inst(43, T_FLOAT, C03, 0.3)
spv += inst(43, T_FLOAT, C015, 0.15)
spv += inst(43, T_FLOAT, C005, 0.05)
spv += inst(43, T_INT, IC3, 3)

# Function
spv += inst(54, T_VOID, FN, 0, T_FN)
spv += inst(248, LBL)

# Load normal and worldPos
spv += inst(61, T_VEC3, LN, V_N)
spv += inst(61, T_VEC3, LWP, V_WP)

# Extract normal.y
spv += inst(81, T_FLOAT, NY, LN, 1)

# blend = ny * 0.5 + 0.5
spv += inst(133, T_FLOAT, NY05, NY, C05)
spv += inst(129, T_FLOAT, BL, NY05, C05)

# blend vec3
spv += inst(80, T_VEC3, BLV, BL, BL, BL)

# sky and ground colors
spv += inst(80, T_VEC3, SKY, C04, C06, C1)
spv += inst(80, T_VEC3, GND, C03, C015, C005)

# ambient = gnd + (sky - gnd) * blend
spv += inst(131, T_VEC3, SG, SKY, GND)
spv += inst(133, T_VEC3, SC, SG, BLV)
spv += inst(129, T_VEC3, AMB, GND, SC)

# Get light position from UBO
spv += inst(65, T_PUV4, LP, V_UBO, IC3)
spv += inst(61, T_VEC4, LL, LP)

# Extract xyz from lightPos vec4
spv += inst(79, T_VEC3, LV3, LL, LL, 0, 1, 2)

# Light direction = normalize(lightPos.xyz - worldPos)
spv += inst(131, T_VEC3, LDR, LV3, LWP)
spv += inst(12, T_VEC3, LDN, GLSL, 69, LDR)  # GLSL Normalize=69

# d = max(dot(N, L), 0)
spv += inst(148, T_FLOAT, DT, LN, LDN)
spv += inst(12, T_FLOAT, DC, GLSL, 40, DT, C0)  # GLSL FMax=40

# 0.3 * d
spv += inst(133, T_FLOAT, DS, C03, DC)

# diffuse vec3
spv += inst(80, T_VEC3, DV, DS, DS, DS)

# final = ambient + diffuse
spv += inst(129, T_VEC3, FIN, AMB, DV)

# Extract and construct output vec4
spv += inst(81, T_FLOAT, FX, FIN, 0)
spv += inst(81, T_FLOAT, FY, FIN, 1)
spv += inst(81, T_FLOAT, FZ, FIN, 2)
spv += inst(80, T_VEC4, OUTVEC, FX, FY, FZ, C1)

# Store and return
spv += inst(62, V_OUT, OUTVEC)
spv += inst(253)  # OpReturn
spv += inst(56)   # OpFunctionEnd

print(spv.hex())
