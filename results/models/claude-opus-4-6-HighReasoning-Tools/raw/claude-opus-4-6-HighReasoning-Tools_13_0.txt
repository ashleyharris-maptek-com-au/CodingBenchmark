#include <iostream>
#include <string>
#include <vector>
#include <unordered_set>
#include <algorithm>

using namespace std;
using ll = long long;

static const ll MOD1 = 1000000007LL;
static const ll MOD2 = 998244353LL;
static const ll BASE1 = 131;
static const ll BASE2 = 137;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    cin.ignore();

    vector<string> strings(n);
    for (int i = 0; i < n; i++) {
        getline(cin, strings[i]);
    }

    if (n == 0) {
        cout << "\n";
        return 0;
    }
    if (n == 1) {
        cout << strings[0] << "\n";
        return 0;
    }

    sort(strings.begin(), strings.end(), [](const string& a, const string& b) {
        return a.size() < b.size();
    });

    int maxLen = (int)strings[0].size();
    if (maxLen == 0) {
        cout << "\n";
        return 0;
    }

    string result;
    int lo = 1, hi = maxLen;

    while (lo <= hi) {
        int len = (lo + hi) / 2;

        ll pow1 = 1, pow2 = 1;
        for (int i = 0; i < len; i++) {
            pow1 = pow1 * BASE1 % MOD1;
            pow2 = pow2 * BASE2 % MOD2;
        }

        const string& s0 = strings[0];
        unordered_set<ll> current;
        current.reserve(s0.size() - len + 2);

        {
            ll h1 = 0, h2 = 0;
            for (int i = 0; i < len; i++) {
                h1 = (h1 * BASE1 + (unsigned char)s0[i]) % MOD1;
                h2 = (h2 * BASE2 + (unsigned char)s0[i]) % MOD2;
            }
            current.insert(h1 * MOD2 + h2);
            for (int i = len; i < (int)s0.size(); i++) {
                h1 = (h1 * BASE1 + (unsigned char)s0[i] - (unsigned char)s0[i - len] * pow1 % MOD1 + MOD1 * 2) % MOD1;
                h2 = (h2 * BASE2 + (unsigned char)s0[i] - (unsigned char)s0[i - len] * pow2 % MOD2 + MOD2 * 2) % MOD2;
                current.insert(h1 * MOD2 + h2);
            }
        }

        bool found = !current.empty();
        for (int si = 1; si < n && found; si++) {
            const string& s = strings[si];
            if (len > (int)s.size()) { found = false; break; }

            unordered_set<ll> next;
            ll h1 = 0, h2 = 0;
            for (int i = 0; i < len; i++) {
                h1 = (h1 * BASE1 + (unsigned char)s[i]) % MOD1;
                h2 = (h2 * BASE2 + (unsigned char)s[i]) % MOD2;
            }
            ll c = h1 * MOD2 + h2;
            if (current.count(c)) next.insert(c);
            for (int i = len; i < (int)s.size(); i++) {
                h1 = (h1 * BASE1 + (unsigned char)s[i] - (unsigned char)s[i - len] * pow1 % MOD1 + MOD1 * 2) % MOD1;
                h2 = (h2 * BASE2 + (unsigned char)s[i] - (unsigned char)s[i - len] * pow2 % MOD2 + MOD2 * 2) % MOD2;
                c = h1 * MOD2 + h2;
                if (current.count(c)) next.insert(c);
            }
            current = move(next);
            if (current.empty()) found = false;
        }

        if (found) {
            ll h1 = 0, h2 = 0;
            for (int i = 0; i < len; i++) {
                h1 = (h1 * BASE1 + (unsigned char)s0[i]) % MOD1;
                h2 = (h2 * BASE2 + (unsigned char)s0[i]) % MOD2;
            }
            if (current.count(h1 * MOD2 + h2)) {
                result = s0.substr(0, len);
            } else {
                for (int i = len; i < (int)s0.size(); i++) {
                    h1 = (h1 * BASE1 + (unsigned char)s0[i] - (unsigned char)s0[i - len] * pow1 % MOD1 + MOD1 * 2) % MOD1;
                    h2 = (h2 * BASE2 + (unsigned char)s0[i] - (unsigned char)s0[i - len] * pow2 % MOD2 + MOD2 * 2) % MOD2;
                    if (current.count(h1 * MOD2 + h2)) {
                        result = s0.substr(i - len + 1, len);
                        break;
                    }
                }
            }
            lo = len + 1;
        } else {
            hi = len - 1;
        }
    }

    if (!result.empty()) {
        bool valid = true;
        for (const auto& s : strings) {
            if (s.find(result) == string::npos) { valid = false; break; }
        }
        if (!valid) result.clear();
    }

    cout << result << "\n";
    return 0;
}
