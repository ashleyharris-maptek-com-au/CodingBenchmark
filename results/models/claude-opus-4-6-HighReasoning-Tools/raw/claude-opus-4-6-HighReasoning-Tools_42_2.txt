; SPIR-V Geometry Shader: Expand points into hexagons
; Input: Points (1 vertex), Output: TriangleStrip, max 18 vertices

               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 18
               OpExecutionMode %main Invocations 1

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

               ; Types
      %void  = OpTypeVoid
   %voidfn   = OpTypeFunction %void
     %float  = OpTypeFloat 32
   %v3float  = OpTypeVector %float 3
   %v4float  = OpTypeVector %float 4
      %int   = OpTypeInt 32 1
     %uint   = OpTypeInt 32 0

               ; Constants
    %int_0   = OpConstant %int 0
    %int_1   = OpConstant %int 1
   %uint_1   = OpConstant %uint 1
  %float_0   = OpConstant %float 0
  %float_1   = OpConstant %float 1
%float_0_04  = OpConstant %float 0.04
%float_0_7   = OpConstant %float 0.7

               ; Hexagon edge angles: 0, 60, 120, 180, 240, 300, 360 degrees in radians
               ; 0 = 0.0, 60 = 1.0472, 120 = 2.0944, 180 = 3.1416, 240 = 4.1888, 300 = 5.2360, 360 = 6.2832
               ; cos/sin values for each angle:
               ; 0:   cos=1.0,       sin=0.0
               ; 60:  cos=0.5,       sin=0.86602540
               ; 120: cos=-0.5,      sin=0.86602540
               ; 180: cos=-1.0,      sin=0.0
               ; 240: cos=-0.5,      sin=-0.86602540
               ; 300: cos=0.5,       sin=-0.86602540

%float_cos0   = OpConstant %float 1.0
%float_sin0   = OpConstant %float 0.0
%float_cos60  = OpConstant %float 0.5
%float_sin60  = OpConstant %float 0.86602540
%float_cos120 = OpConstant %float -0.5
%float_sin120 = OpConstant %float 0.86602540
%float_cos180 = OpConstant %float -1.0
%float_sin180 = OpConstant %float 0.0
%float_cos240 = OpConstant %float -0.5
%float_sin240 = OpConstant %float -0.86602540
%float_cos300 = OpConstant %float 0.5
%float_sin300 = OpConstant %float -0.86602540

               ; Input gl_PerVertex block (array size 1 for InputPoints)
%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_1  = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr  = OpTypePointer Input %arr_pv_1
 %ptr_in_v4  = OpTypePointer Input %v4float
    %gl_in   = OpVariable %ptr_in_arr Input

               ; Input color (location 2, array size 1)
%arr_v3_1    = OpTypeArray %v3float %uint_1
%ptr_in_arr3 = OpTypePointer Input %arr_v3_1
%ptr_in_v3   = OpTypePointer Input %v3float
   %inColor  = OpVariable %ptr_in_arr3 Input

               ; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv  = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4  = OpTypePointer Output %v4float
   %gl_out   = OpVariable %ptr_out_pv Output

               ; Output color (location 0)
%ptr_out_v3  = OpTypePointer Output %v3float
  %outColor  = OpVariable %ptr_out_v3 Output

               ; Main function
     %main   = OpFunction %void None %voidfn
    %entry   = OpLabel

               ; Load input position: gl_in[0].gl_Position
  %pos_ptr   = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
  %pos       = OpLoad %v4float %pos_ptr

               ; Load input color: inColor[0]
  %col_ptr   = OpAccessChain %ptr_in_v3 %inColor %int_0
  %col       = OpLoad %v3float %col_ptr

               ; Compute dark color = col * 0.7
  %dark_vec  = OpCompositeConstruct %v3float %float_0_7 %float_0_7 %float_0_7
  %darkCol   = OpFMul %v3float %col %dark_vec

               ; Extract center x, y, z, w
  %cx        = OpCompositeExtract %float %pos 0
  %cy        = OpCompositeExtract %float %pos 1
  %cz        = OpCompositeExtract %float %pos 2
  %cw        = OpCompositeExtract %float %pos 3

               ; Precompute edge positions (center + r * cos/sin for each angle)
               ; Edge 0 (0 degrees)
  %e0x       = OpFAdd %float %cx %float_0_04
  %e0y       = OpFAdd %float %cy %float_0
  %e0pos     = OpCompositeConstruct %v4float %e0x %e0y %cz %cw

               ; Edge 1 (60 degrees)
  %dx1       = OpFMul %float %float_0_04 %float_cos60
  %dy1       = OpFMul %float %float_0_04 %float_sin60
  %e1x       = OpFAdd %float %cx %dx1
  %e1y       = OpFAdd %float %cy %dy1
  %e1pos     = OpCompositeConstruct %v4float %e1x %e1y %cz %cw

               ; Edge 2 (120 degrees)
  %dx2       = OpFMul %float %float_0_04 %float_cos120
  %dy2       = OpFMul %float %float_0_04 %float_sin120
  %e2x       = OpFAdd %float %cx %dx2
  %e2y       = OpFAdd %float %cy %dy2
  %e2pos     = OpCompositeConstruct %v4float %e2x %e2y %cz %cw

               ; Edge 3 (180 degrees)
  %dx3       = OpFMul %float %float_0_04 %float_cos180
  %e3x       = OpFAdd %float %cx %dx3
  %e3y       = OpFAdd %float %cy %float_0
  %e3pos     = OpCompositeConstruct %v4float %e3x %e3y %cz %cw

               ; Edge 4 (240 degrees)
  %dx4       = OpFMul %float %float_0_04 %float_cos240
  %dy4       = OpFMul %float %float_0_04 %float_sin240
  %e4x       = OpFAdd %float %cx %dx4
  %e4y       = OpFAdd %float %cy %dy4
  %e4pos     = OpCompositeConstruct %v4float %e4x %e4y %cz %cw

               ; Edge 5 (300 degrees)
  %dx5       = OpFMul %float %float_0_04 %float_cos300
  %dy5       = OpFMul %float %float_0_04 %float_sin300
  %e5x       = OpFAdd %float %cx %dx5
  %e5y       = OpFAdd %float %cy %dy5
  %e5pos     = OpCompositeConstruct %v4float %e5x %e5y %cz %cw

               ; Output pointer for gl_Position
  %out_pos   = OpAccessChain %ptr_out_v4 %gl_out %int_0

               ; === Triangle 0: center, e0, e1 ===
               OpStore %out_pos %pos
               OpStore %outColor %col
               OpEmitVertex
               OpStore %out_pos %e0pos
               OpStore %outColor %darkCol
               OpEmitVertex
               OpStore %out_pos %e1pos
               OpStore %outColor %darkCol
               OpEmitVertex
               OpEndPrimitive

               ; === Triangle 1: center, e1, e2 ===
               OpStore %out_pos %pos
               OpStore %outColor %col
               OpEmitVertex
               OpStore %out_pos %e1pos
               OpStore %outColor %darkCol
               OpEmitVertex
               OpStore %out_pos %e2pos
               OpStore %outColor %darkCol
               OpEmitVertex
               OpEndPrimitive

               ; === Triangle 2: center, e2, e3 ===
               OpStore %out_pos %pos
               OpStore %outColor %col
               OpEmitVertex
               OpStore %out_pos %e2pos
               OpStore %outColor %darkCol
               OpEmitVertex
               OpStore %out_pos %e3pos
               OpStore %outColor %darkCol
               OpEmitVertex
               OpEndPrimitive

               ; === Triangle 3: center, e3, e4 ===
               OpStore %out_pos %pos
               OpStore %outColor %col
               OpEmitVertex
               OpStore %out_pos %e3pos
               OpStore %outColor %darkCol
               OpEmitVertex
               OpStore %out_pos %e4pos
               OpStore %outColor %darkCol
               OpEmitVertex
               OpEndPrimitive

               ; === Triangle 4: center, e4, e5 ===
               OpStore %out_pos %pos
               OpStore %outColor %col
               OpEmitVertex
               OpStore %out_pos %e4pos
               OpStore %outColor %darkCol
               OpEmitVertex
               OpStore %out_pos %e5pos
               OpStore %outColor %darkCol
               OpEmitVertex
               OpEndPrimitive

               ; === Triangle 5: center, e5, e0 ===
               OpStore %out_pos %pos
               OpStore %outColor %col
               OpEmitVertex
               OpStore %out_pos %e5pos
               OpStore %outColor %darkCol
               OpEmitVertex
               OpStore %out_pos %e0pos
               OpStore %outColor %darkCol
               OpEmitVertex
               OpEndPrimitive

               OpReturn
               OpFunctionEnd
