; SPIR-V Geometry Shader: Arrow (shaft + arrowhead) from InputLines
; Emits 4 verts for shaft strip + 3 verts for arrowhead strip = 7 max

               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inWorldPos %inNormal %inColor %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 7
               OpExecutionMode %main Invocations 1

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               OpDecorate %inWorldPos Location 0
               OpDecorate %inNormal Location 1
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

               OpDecorate %ubo_type Block
               OpMemberDecorate %ubo_type 0 Offset 0
               OpMemberDecorate %ubo_type 0 MatrixStride 16
               OpMemberDecorate %ubo_type 0 ColMajor
               OpMemberDecorate %ubo_type 1 Offset 64
               OpMemberDecorate %ubo_type 1 MatrixStride 16
               OpMemberDecorate %ubo_type 1 ColMajor
               OpMemberDecorate %ubo_type 2 Offset 128
               OpMemberDecorate %ubo_type 2 MatrixStride 16
               OpMemberDecorate %ubo_type 2 ColMajor
               OpMemberDecorate %ubo_type 3 Offset 192
               OpMemberDecorate %ubo_type 4 Offset 208
               OpMemberDecorate %ubo_type 5 Offset 224
               OpDecorate %ubo_var DescriptorSet 0
               OpDecorate %ubo_var Binding 0

       ; Types
       %void = OpTypeVoid
   %void_fun = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
  %mat4float = OpTypeMatrix %v4float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0

     %int_0  = OpConstant %int 0
     %int_1  = OpConstant %int 1
    %uint_2  = OpConstant %uint 2

  %float_0   = OpConstant %float 0.0
  %float_1   = OpConstant %float 1.0
  %float_0p75 = OpConstant %float 0.75
  %float_0p015 = OpConstant %float 0.015
  %float_0p05 = OpConstant %float 0.05
  %float_neg1 = OpConstant %float -1.0

       ; Input gl_PerVertex block (array of 2 for InputLines)
%gl_PerVertex_in = OpTypeStruct %v4float
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr = OpTypePointer Input %arr_pv_in
  %ptr_in_v4 = OpTypePointer Input %v4float
      %gl_in = OpVariable %ptr_in_arr Input

       ; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
 %ptr_out_v4 = OpTypePointer Output %v4float
     %gl_out = OpVariable %ptr_out_pv Output

       ; User-defined inputs (arrays of 2)
%arr2_v3    = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr2_v3
%ptr_in_v3  = OpTypePointer Input %v3float
 %inWorldPos = OpVariable %ptr_in_arr_v3 Input
   %inNormal = OpVariable %ptr_in_arr_v3 Input
    %inColor = OpVariable %ptr_in_arr_v3 Input

       ; Output color
%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       ; UBO
   %ubo_type = OpTypeStruct %mat4float %mat4float %mat4float %v4float %v4float %v4float
%ptr_ubo     = OpTypePointer Uniform %ubo_type
   %ubo_var  = OpVariable %ptr_ubo Uniform

       ; Function pointer types
%ptr_fun_v4  = OpTypePointer Function %v4float
%ptr_fun_v3  = OpTypePointer Function %v3float

       ; Main function
       %main = OpFunction %void None %void_fun
      %entry = OpLabel

       ; Load p0 = gl_in[0].gl_Position (clip space)
    %p0_ptr  = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
        %p0  = OpLoad %v4float %p0_ptr

       ; Load p1 = gl_in[1].gl_Position (clip space)
    %p1_ptr  = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
        %p1  = OpLoad %v4float %p1_ptr

       ; Load colors
   %c0_ptr   = OpAccessChain %ptr_in_v3 %inColor %int_0
       %c0   = OpLoad %v3float %c0_ptr
   %c1_ptr   = OpAccessChain %ptr_in_v3 %inColor %int_1
       %c1   = OpLoad %v3float %c1_ptr

       ; Extract xy from p0, p1
    %p0x     = OpCompositeExtract %float %p0 0
    %p0y     = OpCompositeExtract %float %p0 1
    %p0z     = OpCompositeExtract %float %p0 2
    %p0w     = OpCompositeExtract %float %p0 3
    %p1x     = OpCompositeExtract %float %p1 0
    %p1y     = OpCompositeExtract %float %p1 1
    %p1z     = OpCompositeExtract %float %p1 2
    %p1w     = OpCompositeExtract %float %p1 3

       ; dir = p1 - p0 (xy components)
    %dx      = OpFSub %float %p1x %p0x
    %dy      = OpFSub %float %p1y %p0y

       ; perp = normalize(-dir.y, dir.x, 0)
    %neg_dy  = OpFNegate %float %dy
    %perp_un = OpCompositeConstruct %v2float %neg_dy %dx
    %perp_n  = OpExtInst %v2float %1 Normalize %perp_un
    %perpx   = OpCompositeExtract %float %perp_n 0
    %perpy   = OpCompositeExtract %float %perp_n 1

       ; mid = p0 + 0.75 * (p1 - p0)
    %midx    = OpExtInst %float %1 FMix %p0x %p1x %float_0p75
    %midy    = OpExtInst %float %1 FMix %p0y %p1y %float_0p75
    %midz    = OpExtInst %float %1 FMix %p0z %p1z %float_0p75
    %midw    = OpExtInst %float %1 FMix %p0w %p1w %float_0p75

       ; Shaft offsets: perp * 0.015
    %sx      = OpFMul %float %perpx %float_0p015
    %sy      = OpFMul %float %perpy %float_0p015

       ; === SHAFT (4 vertices as triangle strip) ===

       ; Vertex 0: p0 + perp*0.015
    %s0x     = OpFAdd %float %p0x %sx
    %s0y     = OpFAdd %float %p0y %sy
    %s0pos   = OpCompositeConstruct %v4float %s0x %s0y %p0z %p0w
    %out0    = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out0 %s0pos
               OpStore %outColor %c0
               OpEmitVertex

       ; Vertex 1: p0 - perp*0.015
    %s1x     = OpFSub %float %p0x %sx
    %s1y     = OpFSub %float %p0y %sy
    %s1pos   = OpCompositeConstruct %v4float %s1x %s1y %p0z %p0w
               OpStore %out0 %s1pos
               OpStore %outColor %c0
               OpEmitVertex

       ; Vertex 2: mid + perp*0.015
    %s2x     = OpFAdd %float %midx %sx
    %s2y     = OpFAdd %float %midy %sy
    %s2pos   = OpCompositeConstruct %v4float %s2x %s2y %midz %midw
               OpStore %out0 %s2pos
               OpStore %outColor %c0
               OpEmitVertex

       ; Vertex 3: mid - perp*0.015
    %s3x     = OpFSub %float %midx %sx
    %s3y     = OpFSub %float %midy %sy
    %s3pos   = OpCompositeConstruct %v4float %s3x %s3y %midz %midw
               OpStore %out0 %s3pos
               OpStore %outColor %c0
               OpEmitVertex

               OpEndPrimitive

       ; === ARROWHEAD (3 vertices as triangle strip) ===

       ; Arrow offsets: perp * 0.05
    %ax      = OpFMul %float %perpx %float_0p05
    %ay      = OpFMul %float %perpy %float_0p05

       ; Vertex 4: mid + perp*0.05
    %a0x     = OpFAdd %float %midx %ax
    %a0y     = OpFAdd %float %midy %ay
    %a0pos   = OpCompositeConstruct %v4float %a0x %a0y %midz %midw
               OpStore %out0 %a0pos
               OpStore %outColor %c1
               OpEmitVertex

       ; Vertex 5: mid - perp*0.05
    %a1x     = OpFSub %float %midx %ax
    %a1y     = OpFSub %float %midy %ay
    %a1pos   = OpCompositeConstruct %v4float %a1x %a1y %midz %midw
               OpStore %out0 %a1pos
               OpStore %outColor %c1
               OpEmitVertex

       ; Vertex 6: p1 (tip)
               OpStore %out0 %p1
               OpStore %outColor %c1
               OpEmitVertex

               OpEndPrimitive

               OpReturn
               OpFunctionEnd
