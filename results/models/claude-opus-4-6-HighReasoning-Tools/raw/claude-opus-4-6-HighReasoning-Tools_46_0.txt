#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 physics;
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    if (i >= N) return;

    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    vec3 pos_i = inData[i * 2].xyz;
    float mass_i = inData[i * 2].w;
    vec3 vel_i = inData[i * 2 + 1].xyz;

    vec3 acc = vec3(0.0);

    for (uint j = 0; j < N; j++) {
        if (j == i) continue;

        vec3 pos_j = inData[j * 2].xyz;
        float mass_j = inData[j * 2].w;

        vec3 r = pos_j - pos_i;
        float distSq = dot(r, r) + softening * softening;
        float invDist = inversesqrt(distSq);
        float invDist3 = invDist * invDist * invDist;

        acc += G * mass_j * r * invDist3;
    }

    vec3 newVel = vel_i + acc * dt;
    vec3 newPos = pos_i + newVel * dt;

    outData[i * 2] = vec4(newPos, mass_i);
    outData[i * 2 + 1] = vec4(newVel, 0.0);
}
