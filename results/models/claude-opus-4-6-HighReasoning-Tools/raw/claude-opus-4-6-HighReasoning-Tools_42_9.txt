; SPIR-V Geometry Shader: Thick smooth curve with miter joins
; Input: InputLinesAdjacency (4 vertices), Output: OutputTriangleStrip (4 vertices)

               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               ; Decorations - input gl_PerVertex
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

               ; Decorations - output gl_PerVertex
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               ; Decorations - user IO
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

               ; Type declarations
      %void  = OpTypeVoid
  %voidfunc  = OpTypeFunction %void
     %float  = OpTypeFloat 32
   %v2float  = OpTypeVector %float 2
   %v3float  = OpTypeVector %float 3
   %v4float  = OpTypeVector %float 4
      %uint  = OpTypeInt 32 0
       %int  = OpTypeInt 32 1

               ; Constants
     %int_0  = OpConstant %int 0
     %int_1  = OpConstant %int 1
     %int_2  = OpConstant %int 2
     %int_3  = OpConstant %int 3
    %uint_4  = OpConstant %uint 4
   %float_0  = OpConstant %float 0.0
 %float_neg1 = OpConstant %float -1.0
%thickness   = OpConstant %float 0.025

               ; Input gl_PerVertex struct and array
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in   = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr  = OpTypePointer Input %arr_pv_in
%ptr_in_v4   = OpTypePointer Input %v4float

               ; Output gl_PerVertex struct
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv  = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4  = OpTypePointer Output %v4float

               ; Input/output color types
%arr_v3_4    = OpTypeArray %v3float %uint_4
%ptr_in_arr3 = OpTypePointer Input %arr_v3_4
%ptr_in_v3   = OpTypePointer Input %v3float
%ptr_out_v3  = OpTypePointer Output %v3float

               ; Variables
     %gl_in  = OpVariable %ptr_in_arr Input
    %gl_out  = OpVariable %ptr_out_pv Output
   %inColor  = OpVariable %ptr_in_arr3 Input
  %outColor  = OpVariable %ptr_out_v3 Output

               ; Main function
      %main  = OpFunction %void None %voidfunc
     %entry  = OpLabel

               ; Load all 4 positions from gl_in
   %ac_pos0  = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
      %pos0  = OpLoad %v4float %ac_pos0
   %ac_pos1  = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
      %pos1  = OpLoad %v4float %ac_pos1
   %ac_pos2  = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
      %pos2  = OpLoad %v4float %ac_pos2
   %ac_pos3  = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
      %pos3  = OpLoad %v4float %ac_pos3

               ; Load colors at indices 1 and 2
  %ac_col1   = OpAccessChain %ptr_in_v3 %inColor %int_1
    %color1  = OpLoad %v3float %ac_col1
  %ac_col2   = OpAccessChain %ptr_in_v3 %inColor %int_2
    %color2  = OpLoad %v3float %ac_col2

               ; Extract xy from positions
   %p0_xy    = OpVectorShuffle %v2float %pos0 %pos0 0 1
   %p1_xy    = OpVectorShuffle %v2float %pos1 %pos1 0 1
   %p2_xy    = OpVectorShuffle %v2float %pos2 %pos2 0 1
   %p3_xy    = OpVectorShuffle %v2float %pos3 %pos3 0 1

               ; Tangent at start: t0 = normalize(pos2 - pos0)
   %d0       = OpFSub %v2float %p2_xy %p0_xy
   %t0       = OpExtInst %v2float %1 Normalize %d0

               ; Tangent at end: t1 = normalize(pos3 - pos1)
   %d1       = OpFSub %v2float %p3_xy %p1_xy
   %t1       = OpExtInst %v2float %1 Normalize %d1

               ; Perpendicular at start: perp0 = (-t0.y, t0.x)
   %t0x      = OpCompositeExtract %float %t0 0
   %t0y      = OpCompositeExtract %float %t0 1
   %neg_t0y  = OpFNegate %float %t0y
   %perp0    = OpCompositeConstruct %v2float %neg_t0y %t0x

               ; Perpendicular at end: perp1 = (-t1.y, t1.x)
   %t1x      = OpCompositeExtract %float %t1 0
   %t1y      = OpCompositeExtract %float %t1 1
   %neg_t1y  = OpFNegate %float %t1y
   %perp1    = OpCompositeConstruct %v2float %neg_t1y %t1x

               ; Scale perpendiculars by thickness
   %thick2   = OpCompositeConstruct %v2float %thickness %thickness
   %offset0  = OpFMul %v2float %perp0 %thick2
   %offset1  = OpFMul %v2float %perp1 %thick2

               ; Extract z and w from start and end positions
   %p1z      = OpCompositeExtract %float %pos1 2
   %p1w      = OpCompositeExtract %float %pos1 3
   %p2z      = OpCompositeExtract %float %pos2 2
   %p2w      = OpCompositeExtract %float %pos2 3

               ; Vertex 1: start + perp0 * thickness
   %v1_xy    = OpFAdd %v2float %p1_xy %offset0
   %v1x      = OpCompositeExtract %float %v1_xy 0
   %v1y      = OpCompositeExtract %float %v1_xy 1
   %v1_pos   = OpCompositeConstruct %v4float %v1x %v1y %p1z %p1w
   %out_ac1  = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_ac1 %v1_pos
               OpStore %outColor %color1
               OpEmitVertex

               ; Vertex 2: start - perp0 * thickness
   %v2_xy    = OpFSub %v2float %p1_xy %offset0
   %v2x      = OpCompositeExtract %float %v2_xy 0
   %v2y      = OpCompositeExtract %float %v2_xy 1
   %v2_pos   = OpCompositeConstruct %v4float %v2x %v2y %p1z %p1w
               OpStore %out_ac1 %v2_pos
               OpStore %outColor %color1
               OpEmitVertex

               ; Vertex 3: end + perp1 * thickness
   %v3_xy    = OpFAdd %v2float %p2_xy %offset1
   %v3x      = OpCompositeExtract %float %v3_xy 0
   %v3y      = OpCompositeExtract %float %v3_xy 1
   %v3_pos   = OpCompositeConstruct %v4float %v3x %v3y %p2z %p2w
               OpStore %out_ac1 %v3_pos
               OpStore %outColor %color2
               OpEmitVertex

               ; Vertex 4: end - perp1 * thickness
   %v4_xy    = OpFSub %v2float %p2_xy %offset1
   %v4x      = OpCompositeExtract %float %v4_xy 0
   %v4y      = OpCompositeExtract %float %v4_xy 1
   %v4_pos   = OpCompositeConstruct %v4float %v4x %v4y %p2z %p2w
               OpStore %out_ac1 %v4_pos
               OpStore %outColor %color2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
