; SPIR-V Geometry Shader: Expand lines into thick ribbon quads
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
               OpExecutionMode %main InputLines
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               OpDecorate %in_color Location 2
               OpDecorate %out_color Location 0

       ; Types
       %void = OpTypeVoid
   %void_fun = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1

      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
     %uint_2 = OpConstant %uint 2

  %float_0   = OpConstant %float 0
  %float_1   = OpConstant %float 1
  %thickness = OpConstant %float 0.03

; Input gl_PerVertex block (array of 2 for InputLines)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input

; Input color (location 2, array of 2)
%arr_v3_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_2
%ptr_in_v3 = OpTypePointer Input %v3float
%in_color = OpVariable %ptr_in_arr_v3 Input

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_pv Output

; Output color (location 0)
%ptr_out_v3 = OpTypePointer Output %v3float
%out_color = OpVariable %ptr_out_v3 Output

; Function type pointers
%ptr_fun_float = OpTypePointer Function %float
%ptr_fun_v4 = OpTypePointer Function %v4float

       %main = OpFunction %void None %void_fun
      %entry = OpLabel

               ; Load gl_in[0].gl_Position (p0)
      %ac_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %ac_p0

               ; Load gl_in[1].gl_Position (p1)
      %ac_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %ac_p1

               ; Load color[0] and color[1]
      %ac_c0 = OpAccessChain %ptr_in_v3 %in_color %int_0
         %c0 = OpLoad %v3float %ac_c0
      %ac_c1 = OpAccessChain %ptr_in_v3 %in_color %int_1
         %c1 = OpLoad %v3float %ac_c1

               ; dir = p1 - p0
        %dir = OpFSub %v4float %p1 %p0

               ; Extract dir.x and dir.y
      %dir_x = OpCompositeExtract %float %dir 0
      %dir_y = OpCompositeExtract %float %dir 1

               ; perp_raw = (-dir.y, dir.x, 0, 0)
  %neg_dir_y = OpFNegate %float %dir_y
   %perp_raw = OpCompositeConstruct %v4float %neg_dir_y %dir_x %float_0 %float_0

               ; length = sqrt(perp_raw.x^2 + perp_raw.y^2)
   %perp_x   = OpCompositeExtract %float %perp_raw 0
   %perp_y   = OpCompositeExtract %float %perp_raw 1
   %px2       = OpFMul %float %perp_x %perp_x
   %py2       = OpFMul %float %perp_y %perp_y
   %len_sq    = OpFAdd %float %px2 %py2
   %len       = OpExtInst %float %1 Sqrt %len_sq

               ; normalize and scale by thickness
   %inv_len   = OpFDiv %float %thickness %len
   %perp_nx   = OpFMul %float %perp_x %inv_len
   %perp_ny   = OpFMul %float %perp_y %inv_len
   %perp      = OpCompositeConstruct %v4float %perp_nx %perp_ny %float_0 %float_0

               ; Vertex 0: p0 + perp, color = c0
       %v0   = OpFAdd %v4float %p0 %perp
    %ac_out0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %ac_out0 %v0
               OpStore %out_color %c0
               OpEmitVertex

               ; Vertex 1: p0 - perp, color = c0
       %v1   = OpFSub %v4float %p0 %perp
    %ac_out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %ac_out1 %v1
               OpStore %out_color %c0
               OpEmitVertex

               ; Vertex 2: p1 + perp, color = c1
       %v2   = OpFAdd %v4float %p1 %perp
    %ac_out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %ac_out2 %v2
               OpStore %out_color %c1
               OpEmitVertex

               ; Vertex 3: p1 - perp, color = c1
       %v3   = OpFSub %v4float %p1 %perp
    %ac_out3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %ac_out3 %v3
               OpStore %out_color %c1
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
