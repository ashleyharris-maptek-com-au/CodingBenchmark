#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 radii;
    vec4 weights;
    vec4 bounds;
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    if (i >= N) return;

    float dt        = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed  = weights.w;

    float boundSize  = bounds.x;
    float boundForce = bounds.y;

    vec3 pos = inData[i * 2].xyz;
    vec3 vel = inData[i * 2 + 1].xyz;

    vec3 sepForce = vec3(0.0);
    vec3 alignForce = vec3(0.0);
    vec3 cohForce = vec3(0.0);

    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;

    for (uint j = 0; j < N; j++) {
        if (j == i) continue;

        vec3 otherPos = inData[j * 2].xyz;
        vec3 otherVel = inData[j * 2 + 1].xyz;

        vec3 diff = pos - otherPos;
        float dist = length(diff);

        if (dist < 0.0001) continue;

        if (dist < sepRadius) {
            sepForce += diff / dist;
            sepCount++;
        }

        if (dist < alignRadius) {
            alignForce += otherVel;
            alignCount++;
        }

        if (dist < cohRadius) {
            cohForce += otherPos;
            cohCount++;
        }
    }

    vec3 accel = vec3(0.0);

    if (sepCount > 0) {
        sepForce /= float(sepCount);
        accel += sepForce * sepWeight;
    }

    if (alignCount > 0) {
        alignForce /= float(alignCount);
        accel += (alignForce - vel) * alignWeight;
    }

    if (cohCount > 0) {
        cohForce /= float(cohCount);
        accel += (cohForce - pos) * cohWeight;
    }

    float threshold = boundSize * 0.4;
    float scale = boundSize * 0.5;

    if (pos.x > threshold)  accel.x -= (pos.x - threshold) / scale * boundForce;
    if (pos.x < -threshold) accel.x -= (pos.x + threshold) / scale * boundForce;
    if (pos.y > threshold)  accel.y -= (pos.y - threshold) / scale * boundForce;
    if (pos.y < -threshold) accel.y -= (pos.y + threshold) / scale * boundForce;
    if (pos.z > threshold)  accel.z -= (pos.z - threshold) / scale * boundForce;
    if (pos.z < -threshold) accel.z -= (pos.z + threshold) / scale * boundForce;

    vel += accel * dt;

    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = vel / speed * maxSpeed;
    }

    pos += vel * dt;

    outData[i * 2]     = vec4(pos, 0.0);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
