; SPIR-V Geometry Shader: Flat-shade triangles from adjacency input
; InputTrianglesAdjacency -> OutputTriangleStrip, 3 vertices

               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

               ; Types
      %void  = OpTypeVoid
  %voidfunc  = OpTypeFunction %void
     %float  = OpTypeFloat 32
   %v3float  = OpTypeVector %float 3
   %v4float  = OpTypeVector %float 4
      %uint  = OpTypeInt 32 0
       %int  = OpTypeInt 32 1

               ; Constants
   %float_0  = OpConstant %float 0.0
   %float_1  = OpConstant %float 1.0
 %float_0_2  = OpConstant %float 0.2
 %float_0_8  = OpConstant %float 0.8
 %float_1o3  = OpConstant %float 0.333333343
     %int_0  = OpConstant %int 0
     %int_1  = OpConstant %int 1
     %int_2  = OpConstant %int 2
     %int_4  = OpConstant %int 4
    %uint_6  = OpConstant %uint 6

               ; Struct types
%gl_PerVertex_in  = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

               ; Array type for input (6 vertices for InputTrianglesAdjacency)
 %arr_pv_in  = OpTypeArray %gl_PerVertex_in %uint_6

               ; Input array types for color (array of 6 vec3)
%arr_v3_in_6 = OpTypeArray %v3float %uint_6

               ; Pointer types
%ptr_in_arr_pv  = OpTypePointer Input %arr_pv_in
  %ptr_in_v4    = OpTypePointer Input %v4float
%ptr_out_pv     = OpTypePointer Output %gl_PerVertex_out
  %ptr_out_v4   = OpTypePointer Output %v4float
%ptr_in_arr_v3  = OpTypePointer Input %arr_v3_in_6
  %ptr_in_v3    = OpTypePointer Input %v3float
  %ptr_out_v3   = OpTypePointer Output %v3float

               ; Variables
     %gl_in  = OpVariable %ptr_in_arr_pv Input
    %gl_out  = OpVariable %ptr_out_pv Output
   %inColor  = OpVariable %ptr_in_arr_v3 Input
  %outColor  = OpVariable %ptr_out_v3 Output

               ; Main function
      %main  = OpFunction %void None %voidfunc
     %entry  = OpLabel

               ; Load gl_in[0].gl_Position (index 0)
   %ac_p0    = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
   %pos0     = OpLoad %v4float %ac_p0

               ; Load gl_in[2].gl_Position (index 2)
   %ac_p2    = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
   %pos2     = OpLoad %v4float %ac_p2

               ; Load gl_in[4].gl_Position (index 4)
   %ac_p4    = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
   %pos4     = OpLoad %v4float %ac_p4

               ; Extract xyz from positions for normal computation
  %p0_xyz    = OpVectorShuffle %v3float %pos0 %pos0 0 1 2
  %p2_xyz    = OpVectorShuffle %v3float %pos2 %pos2 0 1 2
  %p4_xyz    = OpVectorShuffle %v3float %pos4 %pos4 0 1 2

               ; edge1 = p2 - p0
   %edge1    = OpFSub %v3float %p2_xyz %p0_xyz

               ; edge2 = p4 - p0
   %edge2    = OpFSub %v3float %p4_xyz %p0_xyz

               ; Manual cross product: edge1 x edge2
               ; Extract components of edge1
  %e1x       = OpCompositeExtract %float %edge1 0
  %e1y       = OpCompositeExtract %float %edge1 1
  %e1z       = OpCompositeExtract %float %edge1 2

               ; Extract components of edge2
  %e2x       = OpCompositeExtract %float %edge2 0
  %e2y       = OpCompositeExtract %float %edge2 1
  %e2z       = OpCompositeExtract %float %edge2 2

               ; cross.x = e1y*e2z - e1z*e2y
  %cx_a      = OpFMul %float %e1y %e2z
  %cx_b      = OpFMul %float %e1z %e2y
  %cx        = OpFSub %float %cx_a %cx_b

               ; cross.y = e1z*e2x - e1x*e2z
  %cy_a      = OpFMul %float %e1z %e2x
  %cy_b      = OpFMul %float %e1x %e2z
  %cy        = OpFSub %float %cy_a %cy_b

               ; cross.z = e1x*e2y - e1y*e2x
  %cz_a      = OpFMul %float %e1x %e2y
  %cz_b      = OpFMul %float %e1y %e2x
  %cz        = OpFSub %float %cz_a %cz_b

               ; Compose cross product vector
  %cross     = OpCompositeConstruct %v3float %cx %cy %cz

               ; Normalize the face normal
  %normal    = OpExtInst %v3float %1 Normalize %cross

               ; Light direction = normalize(vec3(1,1,1))
  %L_raw     = OpCompositeConstruct %v3float %float_1 %float_1 %float_1
  %L         = OpExtInst %v3float %1 Normalize %L_raw

               ; Diffuse = max(dot(N, L), 0.0)
  %NdotL     = OpDot %float %normal %L
  %diffuse   = OpExtInst %float %1 FMax %NdotL %float_0

               ; Load colors at indices 0, 2, 4
  %ac_c0     = OpAccessChain %ptr_in_v3 %inColor %int_0
  %col0      = OpLoad %v3float %ac_c0
  %ac_c2     = OpAccessChain %ptr_in_v3 %inColor %int_2
  %col2      = OpLoad %v3float %ac_c2
  %ac_c4     = OpAccessChain %ptr_in_v3 %inColor %int_4
  %col4      = OpLoad %v3float %ac_c4

               ; Base color = average of color[0], color[2], color[4]
  %sum_01    = OpFAdd %v3float %col0 %col2
  %sum_012   = OpFAdd %v3float %sum_01 %col4
  %scale3    = OpCompositeConstruct %v3float %float_1o3 %float_1o3 %float_1o3
  %base      = OpFMul %v3float %sum_012 %scale3

               ; Final color = base * (0.2 + 0.8 * diffuse)
  %scaled_d  = OpFMul %float %float_0_8 %diffuse
  %lighting  = OpFAdd %float %float_0_2 %scaled_d
  %light_v   = OpCompositeConstruct %v3float %lighting %lighting %lighting
  %finalCol  = OpFMul %v3float %base %light_v

               ; --- Emit vertex 0 ---
  %out_pos0  = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos0 %pos0
               OpStore %outColor %finalCol
               OpEmitVertex

               ; --- Emit vertex 2 ---
  %out_pos2  = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos2 %pos2
               OpStore %outColor %finalCol
               OpEmitVertex

               ; --- Emit vertex 4 ---
  %out_pos4  = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos4 %pos4
               OpStore %outColor %finalCol
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
