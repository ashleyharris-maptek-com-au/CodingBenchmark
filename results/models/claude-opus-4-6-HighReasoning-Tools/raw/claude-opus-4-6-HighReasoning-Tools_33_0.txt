using System;
using System.Diagnostics;

class QAP
{
    static int N;
    static int[][] flow, dist;
    static Random rng = new Random(42);

    static long ComputeCost(int[] p)
    {
        long cost = 0;
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                cost += (long)flow[i][j] * dist[p[i]][p[j]];
        return cost;
    }

    static long SwapDelta(int[] p, int r, int s)
    {
        long d = 0;
        int pr = p[r], ps = p[s];
        for (int k = 0; k < N; k++)
        {
            if (k == r || k == s) continue;
            int pk = p[k];
            d += (long)(flow[k][r] - flow[k][s]) * (dist[pk][ps] - dist[pk][pr])
               + (long)(flow[r][k] - flow[s][k]) * (dist[ps][pk] - dist[pr][pk]);
        }
        d += (long)(flow[r][r] - flow[s][s]) * (dist[ps][ps] - dist[pr][pr])
           + (long)(flow[r][s] - flow[s][r]) * (dist[ps][pr] - dist[pr][ps]);
        return d;
    }

    static int[] GreedyConstruction()
    {
        long[] flowSum = new long[N];
        long[] distSum = new long[N];
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
            {
                flowSum[i] += flow[i][j] + flow[j][i];
                distSum[i] += dist[i][j] + dist[j][i];
            }

        int[] flowOrder = new int[N];
        int[] distOrder = new int[N];
        for (int i = 0; i < N; i++) { flowOrder[i] = i; distOrder[i] = i; }
        Array.Sort(flowOrder, (a, b) => distSum[b].CompareTo(distSum[a])); // wrong, fix below
        // facilities with high flow -> locations with low distance
        Array.Sort(flowOrder, (a, b) => flowSum[b].CompareTo(flowSum[a]));
        Array.Sort(distOrder, (a, b) => distSum[a].CompareTo(distSum[b]));

        int[] p = new int[N];
        for (int i = 0; i < N; i++)
            p[flowOrder[i]] = distOrder[i];
        return p;
    }

    static int[] RandomPermutation()
    {
        int[] p = new int[N];
        for (int i = 0; i < N; i++) p[i] = i;
        for (int i = N - 1; i > 0; i--)
        {
            int j = rng.Next(i + 1);
            int tmp = p[i]; p[i] = p[j]; p[j] = tmp;
        }
        return p;
    }

    static void SimulatedAnnealing(int[] p, ref long bestCost, int[] bestPerm, Stopwatch sw, long timeLimitMs)
    {
        long cost = ComputeCost(p);
        if (cost < bestCost)
        {
            bestCost = cost;
            Array.Copy(p, bestPerm, N);
        }

        double T0, Tend;
        if (N <= 30) { T0 = bestCost * 0.05 / (N * N); Tend = T0 * 1e-5; }
        else if (N <= 200) { T0 = bestCost * 0.02 / (N * N); Tend = T0 * 1e-4; }
        else { T0 = bestCost * 0.01 / (N * N); Tend = T0 * 1e-3; }

        if (T0 < 1) T0 = 1;
        if (Tend < 0.0001) Tend = 0.0001;

        long remaining = timeLimitMs - sw.ElapsedMilliseconds;
        if (remaining <= 100) return;

        long totalIter = (long)((double)remaining / 1000.0) * (N <= 100 ? 2000000L : (N <= 500 ? 500000L : 100000L));
        if (totalIter < 10000) totalIter = 10000;

        double logRatio = Math.Log(Tend / T0);
        long checkInterval = Math.Max(1000, totalIter / 1000);

        for (long iter = 0; iter < totalIter; iter++)
        {
            if (iter % checkInterval == 0 && sw.ElapsedMilliseconds >= timeLimitMs - 200)
                break;

            int r = rng.Next(N);
            int s = rng.Next(N - 1);
            if (s >= r) s++;

            long delta = SwapDelta(p, r, s);

            if (delta < 0)
            {
                int tmp = p[r]; p[r] = p[s]; p[s] = tmp;
                cost += delta;
                if (cost < bestCost)
                {
                    bestCost = cost;
                    Array.Copy(p, bestPerm, N);
                }
            }
            else
            {
                double progress = (double)iter / totalIter;
                double T = T0 * Math.Exp(logRatio * progress);
                if (rng.NextDouble() < Math.Exp(-delta / T))
                {
                    int tmp = p[r]; p[r] = p[s]; p[s] = tmp;
                    cost += delta;
                }
            }
        }
    }

    static void Main()
    {
        var sw = Stopwatch.StartNew();
        long timeLimitMs = 280000;

        var input = Console.In.ReadToEnd().Split(new[] { ' ', '\t', '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        N = int.Parse(input[idx++]);

        flow = new int[N][];
        dist = new int[N][];
        for (int i = 0; i < N; i++)
        {
            flow[i] = new int[N];
            for (int j = 0; j < N; j++)
                flow[i][j] = int.Parse(input[idx++]);
        }
        for (int i = 0; i < N; i++)
        {
            dist[i] = new int[N];
            for (int j = 0; j < N; j++)
                dist[i][j] = int.Parse(input[idx++]);
        }

        int[] bestPerm = new int[N];
        long bestCost = long.MaxValue;

        if (N <= 1)
        {
            bestPerm[0] = 0;
            bestCost = (N == 1) ? (long)flow[0][0] * dist[0][0] : 0;
            Console.WriteLine(bestCost);
            Console.WriteLine(string.Join(" ", bestPerm));
            return;
        }

        // Greedy start
        int[] greedy = GreedyConstruction();
        long greedyCost = ComputeCost(greedy);
        bestCost = greedyCost;
        Array.Copy(greedy, bestPerm, N);

        // SA from greedy
        int[] working = (int[])greedy.Clone();
        SimulatedAnnealing(working, ref bestCost, bestPerm, sw, timeLimitMs);

        // Multiple restarts with remaining time
        int restarts = 0;
        while (sw.ElapsedMilliseconds < timeLimitMs - 2000)
        {
            restarts++;
            long perRestart = Math.Max(5000, (timeLimitMs - sw.ElapsedMilliseconds - 1000) / 3);
            long deadline = Math.Min(sw.ElapsedMilliseconds + perRestart, timeLimitMs - 500);

            int[] rp;
            if (restarts % 3 == 0)
                rp = (int[])bestPerm.Clone(); // restart from best
            else
                rp = RandomPermutation();

            SimulatedAnnealing(rp, ref bestCost, bestPerm, sw, deadline);
        }

        Console.WriteLine(bestCost);
        Console.WriteLine(string.Join(" ", bestPerm));
    }
}
