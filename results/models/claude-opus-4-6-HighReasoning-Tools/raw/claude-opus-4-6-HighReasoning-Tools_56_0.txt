'use strict';
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depot = data.depot;
const truckCount = data.truckCount;
const packages = data.packages;
const n = packages.length;

if (n === 0) {
  console.log(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const startTime = Date.now();
const dx = depot[0], dy = depot[1];
const px = new Float64Array(n);
const py = new Float64Array(n);
for (let i = 0; i < n; i++) { px[i] = packages[i].x; py[i] = packages[i].y; }

function dist(ax, ay, bx, by) {
  const ex = ax - bx, ey = ay - by;
  return Math.sqrt(ex * ex + ey * ey);
}

function timeLeft() { return 55000 - (Date.now() - startTime); }

const depotDist = new Float64Array(n);
for (let i = 0; i < n; i++) depotDist[i] = dist(dx, dy, px[i], py[i]);

// Spatial grid for nearest neighbor lookups
let minX = Infinity, maxX = -Infinity, minY = Infinity, maxY = -Infinity;
for (let i = 0; i < n; i++) {
  if (px[i] < minX) minX = px[i]; if (px[i] > maxX) maxX = px[i];
  if (py[i] < minY) minY = py[i]; if (py[i] > maxY) maxY = py[i];
}
const gridSize = Math.max(Math.ceil(Math.sqrt(n / 4)), 1);
const cellW = (maxX - minX + 1e-9) / gridSize;
const cellH = (maxY - minY + 1e-9) / gridSize;
const grid = new Array(gridSize * gridSize);
for (let i = 0; i < grid.length; i++) grid[i] = [];
for (let i = 0; i < n; i++) {
  const gx = Math.min(Math.floor((px[i] - minX) / cellW), gridSize - 1);
  const gy = Math.min(Math.floor((py[i] - minY) / cellH), gridSize - 1);
  grid[gy * gridSize + gx].push(i);
}

function kNearest(idx, k) {
  const gx0 = Math.min(Math.floor((px[idx] - minX) / cellW), gridSize - 1);
  const gy0 = Math.min(Math.floor((py[idx] - minY) / cellH), gridSize - 1);
  const cands = [];
  for (let r = 0; cands.length < k * 2 && r < gridSize; r++) {
    const xlo = Math.max(0, gx0 - r), xhi = Math.min(gridSize - 1, gx0 + r);
    const ylo = Math.max(0, gy0 - r), yhi = Math.min(gridSize - 1, gy0 + r);
    if (r === 0) {
      for (const j of grid[gy0 * gridSize + gx0]) if (j !== idx) cands.push(j);
    } else {
      for (let cx = xlo; cx <= xhi; cx++) {
        if (gy0 - r >= 0) for (const j of grid[(gy0 - r) * gridSize + cx]) if (j !== idx) cands.push(j);
        if (gy0 + r < gridSize && gy0 + r !== gy0 - r) for (const j of grid[(gy0 + r) * gridSize + cx]) if (j !== idx) cands.push(j);
      }
      for (let cy = ylo + 1; cy <= yhi - 1; cy++) {
        if (gx0 - r >= 0) for (const j of grid[cy * gridSize + gx0 - r]) if (j !== idx) cands.push(j);
        if (gx0 + r < gridSize && gx0 + r !== gx0 - r) for (const j of grid[cy * gridSize + gx0 + r]) if (j !== idx) cands.push(j);
      }
    }
    if (cands.length >= k) break;
  }
  const seen = new Set();
  const unique = [];
  for (const j of cands) { if (!seen.has(j)) { seen.add(j); unique.push(j); } }
  unique.sort((a, b) => ((px[idx]-px[a])**2 + (py[idx]-py[a])**2) - ((px[idx]-px[b])**2 + (py[idx]-py[b])**2));
  return unique.slice(0, k);
}

// Clark-Wright Savings - use KNN for large n
const K = n <= 500 ? n - 1 : Math.min(n - 1, Math.max(30, Math.ceil(300000 / n)));
const savings = [];
const seenPairs = new Set();

for (let i = 0; i < n; i++) {
  const nbrs = n <= 500 ? null : kNearest(i, K);
  const jList = nbrs || Array.from({length: n}, (_, j) => j);
  for (const j of jList) {
    if (j <= i) continue;
    if (n > 500) {
      const key = i * n + j;
      if (seenPairs.has(key)) continue;
      seenPairs.add(key);
    }
    const s = depotDist[i] + depotDist[j] - dist(px[i], py[i], px[j], py[j]);
    if (s > 0) savings.push([s, i, j]);
  }
}

savings.sort((a, b) => b[0] - a[0]);

const routeIdx = new Int32Array(n);
const routeList = new Array(n);
for (let i = 0; i < n; i++) { routeList[i] = [i]; routeIdx[i] = i; }
const rStart = new Int32Array(n);
const rEnd = new Int32Array(n);
for (let i = 0; i < n; i++) { rStart[i] = i; rEnd[i] = i; }
let numRoutes = n;

for (const [, i, j] of savings) {
  const ri = routeIdx[i], rj = routeIdx[j];
  if (ri === rj) continue;
  const isEndI = rEnd[ri] === i, isStartI = rStart[ri] === i;
  const isEndJ = rEnd[rj] === j, isStartJ = rStart[rj] === j;
  if (!isEndI && !isStartI) continue;
  if (!isEndJ && !isStartJ) continue;

  let mergeA = ri, mergeB = rj, flipA = false, flipB = false;
  if (isEndI && isStartJ) {}
  else if (isStartI && isEndJ) { mergeA = rj; mergeB = ri; }
  else if (isEndI && isEndJ) { flipB = true; }
  else if (isStartI && isStartJ) { flipA = true; }
  else continue;

  const rA = routeList[mergeA], rB = routeList[mergeB];
  if (!rA || !rB) continue;
  if (flipA) rA.reverse();
  if (flipB) rB.reverse();
  for (let k = 0; k < rB.length; k++) { rA.push(rB[k]); routeIdx[rB[k]] = mergeA; }
  routeList[mergeB] = null;
  rStart[mergeA] = rA[0]; rEnd[mergeA] = rA[rA.length - 1];
  numRoutes--;
}

let activeRoutes = routeList.filter(r => r && r.length > 0);

// Merge excess routes
while (activeRoutes.length > truckCount) {
  let minLen = Infinity, minIdx = -1;
  for (let i = 0; i < activeRoutes.length; i++) {
    if (activeRoutes[i].length < minLen) { minLen = activeRoutes[i].length; minIdx = i; }
  }
  const small = activeRoutes.splice(minIdx, 1)[0];
  for (const pidx of small) {
    let bestCost = Infinity, bestR = 0, bestPos = 0;
    for (let ri = 0; ri < activeRoutes.length; ri++) {
      const r = activeRoutes[ri];
      for (let pos = 0; pos <= r.length; pos++) {
        const pX = pos === 0 ? dx : px[r[pos - 1]];
        const pY = pos === 0 ? dy : py[r[pos - 1]];
        const nX = pos === r.length ? dx : px[r[pos]];
        const nY = pos === r.length ? dy : py[r[pos]];
        const cost = dist(pX, pY, px[pidx], py[pidx]) + dist(px[pidx], py[pidx], nX, nY) - dist(pX, pY, nX, nY);
        if (cost < bestCost) { bestCost = cost; bestR = ri; bestPos = pos; }
      }
    }
    activeRoutes[bestR].splice(bestPos, 0, pidx);
  }
}

// 2-opt with time limit, using neighbor lists for large routes
function twoOptNN(route, deadline) {
  const len = route.length;
  if (len < 4) return;

  const NNK = Math.min(len - 1, n <= 1000 ? len : 15);

  const xs = new Float64Array(len + 2);
  const ys = new Float64Array(len + 2);
  xs[0] = dx; ys[0] = dy;
  for (let i = 0; i < len; i++) { xs[i + 1] = px[route[i]]; ys[i + 1] = py[route[i]]; }
  xs[len + 1] = dx; ys[len + 1] = dy;

  // Position lookup: route index for each package
  const pos = new Int32Array(n).fill(-1);
  for (let i = 0; i < len; i++) pos[route[i]] = i;

  let improved = true;
  let checks = 0;
  while (improved) {
    if (Date.now() > deadline) break;
    improved = false;
    for (let i = 0; i < len - 1; i++) {
      if (NNK < len - 1) {
        // Use spatial neighbors
        const nbrs = kNearest(route[i], NNK);
        for (const nb of nbrs) {
          const j = pos[nb];
          if (j < 0 || j <= i + 1 || j >= len) continue;
          const d1 = dist(xs[i], ys[i], xs[i + 1], ys[i + 1]) + dist(xs[j], ys[j], xs[j + 1], ys[j + 1]);
          const d2 = dist(xs[i], ys[i], xs[j], ys[j]) + dist(xs[i + 1], ys[i + 1], xs[j + 1], ys[j + 1]);
          if (d2 < d1 - 1e-10) {
            let lo = i, hi = j - 1;
            while (lo < hi) { const t = route[lo]; route[lo] = route[hi]; route[hi] = t; lo++; hi--; }
            for (let k = i; k <= j - 1; k++) { xs[k + 1] = px[route[k]]; ys[k + 1] = py[route[k]]; pos[route[k]] = k; }
            improved = true;
          }
        }
      } else {
        for (let j = i + 2; j < len; j++) {
          const d1 = dist(xs[i], ys[i], xs[i + 1], ys[i + 1]) + dist(xs[j], ys[j], xs[j + 1], ys[j + 1]);
          const d2 = dist(xs[i], ys[i], xs[j], ys[j]) + dist(xs[i + 1], ys[i + 1], xs[j + 1], ys[j + 1]);
          if (d2 < d1 - 1e-10) {
            let lo = i, hi = j - 1;
            while (lo < hi) { const t = route[lo]; route[lo] = route[hi]; route[hi] = t; lo++; hi--; }
            for (let k = i; k <= j - 1; k++) { xs[k + 1] = px[route[k]]; ys[k + 1] = py[route[k]]; }
            improved = true;
          }
        }
      }
      if (++checks % 500 === 0 && Date.now() > deadline) break;
    }
  }
}

function orOpt(route, deadline) {
  const len = route.length;
  if (len < 3) return;
  for (let segLen = 1; segLen <= Math.min(3, len - 1); segLen++) {
    let improved = true;
    while (improved) {
      if (Date.now() > deadline) return;
      improved = false;
      for (let i = 0; i <= route.length - segLen && !improved; i++) {
        const rLen = route.length;
        const prevX = i === 0 ? dx : px[route[i - 1]];
        const prevY = i === 0 ? dy : py[route[i - 1]];
        const segSX = px[route[i]], segSY = py[route[i]];
        const segEX = px[route[i + segLen - 1]], segEY = py[route[i + segLen - 1]];
        const aftX = i + segLen >= rLen ? dx : px[route[i + segLen]];
        const aftY = i + segLen >= rLen ? dy : py[route[i + segLen]];
        const removeSave = dist(prevX, prevY, segSX, segSY) + dist(segEX, segEY, aftX, aftY) - dist(prevX, prevY, aftX, aftY);

        for (let j = 0; j <= rLen - segLen; j++) {
          if (j >= i - 1 && j <= i + segLen) continue;
          const jPX = j === 0 ? dx : px[route[j - 1]];
          const jPY = j === 0 ? dy : py[route[j - 1]];
          const jNX = j >= rLen ? dx : px[route[j]];
          const jNY = j >= rLen ? dy : py[route[j]];
          const insertCost = dist(jPX, jPY, segSX, segSY) + dist(segEX, segEY, jNX, jNY) - dist(jPX, jPY, jNX, jNY);
          if (insertCost < removeSave - 1e-10) {
            const seg = route.splice(i, segLen);
            route.splice(j > i ? j - segLen : j, 0, ...seg);
            improved = true;
            break;
          }
        }
      }
    }
  }
}

function crossRouteMove(allRoutes, deadline) {
  let improved = true;
  while (improved) {
    if (Date.now() > deadline) return;
    improved = false;
    for (let ri = 0; ri < allRoutes.length && !improved; ri++) {
      const rA = allRoutes[ri];
      for (let pi = 0; pi < rA.length && !improved; pi++) {
        const pidx = rA[pi];
        const pAx = pi === 0 ? dx : px[rA[pi - 1]];
        const pAy = pi === 0 ? dy : py[rA[pi - 1]];
        const nAx = pi + 1 >= rA.length ? dx : px[rA[pi + 1]];
        const nAy = pi + 1 >= rA.length ? dy : py[rA[pi + 1]];
        const removeGain = dist(pAx, pAy, px[pidx], py[pidx]) + dist(px[pidx], py[pidx], nAx, nAy) - dist(pAx, pAy, nAx, nAy);

        for (let rj = 0; rj < allRoutes.length && !improved; rj++) {
          if (ri === rj) continue;
          const rB = allRoutes[rj];
          for (let pos = 0; pos <= rB.length; pos++) {
            const pBx = pos === 0 ? dx : px[rB[pos - 1]];
            const pBy = pos === 0 ? dy : py[rB[pos - 1]];
            const nBx = pos >= rB.length ? dx : px[rB[pos]];
            const nBy = pos >= rB.length ? dy : py[rB[pos]];
            const insertCost = dist(pBx, pBy, px[pidx], py[pidx]) + dist(px[pidx], py[pidx], nBx, nBy) - dist(pBx, pBy, nBx, nBy);
            if (insertCost < removeGain - 1e-10) {
              rA.splice(pi, 1);
              rB.splice(pos, 0, pidx);
              improved = true;
              break;
            }
          }
        }
      }
    }
  }
}

// Run optimizations with time budgets
const totalBudget = 55000;
const phase1End = startTime + totalBudget * 0.5;
const phase2End = startTime + totalBudget * 0.8;
const finalEnd = startTime + totalBudget;

for (let i = 0; i < activeRoutes.length; i++) {
  twoOptNN(activeRoutes[i], phase1End);
  if (Date.now() > phase1End) break;
  orOpt(activeRoutes[i], phase1End);
  if (Date.now() > phase1End) break;
}

if (activeRoutes.length > 1) crossRouteMove(activeRoutes, phase2End);

for (let i = 0; i < activeRoutes.length; i++) {
  twoOptNN(activeRoutes[i], finalEnd);
  if (Date.now() > finalEnd) break;
}

activeRoutes = activeRoutes.filter(r => r.length > 0);
console.log(JSON.stringify({ routes: activeRoutes.map(r => r.map(i => packages[i].id)) }));
