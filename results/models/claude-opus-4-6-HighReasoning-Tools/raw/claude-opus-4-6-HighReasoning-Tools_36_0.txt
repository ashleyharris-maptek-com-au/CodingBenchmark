struct HK {
    int nl, nr;
    vector<vector<int>> g;
    vector<int> ml, mr, d;
    HK(int l, int r): nl(l), nr(r), g(l), ml(l,-1), mr(r,-1), d(l){}
    void ae(int u, int v){g[u].push_back(v);}
    bool bfs(){
        queue<int>q;
        for(int i=0;i<nl;i++){if(ml[i]<0){d[i]=0;q.push(i);}else d[i]=-1;}
        bool f=false;
        while(!q.empty()){int u=q.front();q.pop();
            for(int v:g[u]){int w=mr[v];if(w<0)f=true;
                else if(d[w]<0){d[w]=d[u]+1;q.push(w);}}}
        return f;
    }
    bool dfs(int u){
        for(int v:g[u]){int w=mr[v];
            if(w<0||(d[w]==d[u]+1&&dfs(w))){ml[u]=v;mr[v]=u;return true;}}
        d[u]=-1;return false;
    }
    int run(){int r=0;while(bfs())for(int i=0;i<nl;i++)if(ml[i]<0&&dfs(i))r++;return r;}
};

const int MW=8;
struct BS {
    uint64_t w[MW]={};
    void set(int i){w[i>>6]|=1ULL<<(i&63);}
    void clr(int i){w[i>>6]&=~(1ULL<<(i&63));}
    bool get(int i)const{return(w[i>>6]>>(i&63))&1;}
    bool empty(int nw)const{for(int i=0;i<nw;i++)if(w[i])return false;return true;}
    int cnt(int nw)const{int c=0;for(int i=0;i<nw;i++)c+=(int)std::popcount(w[i]);return c;}
    int first(int nw)const{for(int i=0;i<nw;i++)if(w[i])return i*64+(int)std::countr_zero(w[i]);return-1;}
};

static int NW;
static BS badj[512];
static int bbest;
static BS bset;

void bnb(BS cd, BS ch, int cc){
    bool changed=true;
    while(changed){changed=false;
        for(int wi=0;wi<NW;wi++){uint64_t ww=cd.w[wi];
            while(ww){int b=(int)std::countr_zero(ww);int v=wi*64+b;ww&=ww-1;
                if(!cd.get(v))continue;
                BS nb={};for(int i=0;i<NW;i++)nb.w[i]=badj[v].w[i]&cd.w[i];
                int d=nb.cnt(NW);
                if(d==0){ch.set(v);cc++;cd.clr(v);changed=true;}
                else if(d==1){ch.set(v);cc++;cd.clr(v);
                    int f=nb.first(NW);cd.clr(f);changed=true;}}}}
    if(cd.empty(NW)){if(cc>bbest){bbest=cc;bset=ch;}return;}
    int sz=cd.cnt(NW);
    if(cc+sz<=bbest)return;
    {BS mt={};int ms=0;
     for(int wi=0;wi<NW;wi++){uint64_t ww=cd.w[wi];
        while(ww){int b=(int)std::countr_zero(ww);int u=wi*64+b;ww&=ww-1;
            if(mt.get(u))continue;
            for(int i=0;i<NW;i++){uint64_t av=badj[u].w[i]&cd.w[i]&~mt.w[i];
                if(av){mt.set(u);mt.set(i*64+(int)std::countr_zero(av));ms++;break;}}}}
     if(cc+sz-ms<=bbest)return;}
    int bv=-1,bd=-1;
    for(int wi=0;wi<NW;wi++){uint64_t ww=cd.w[wi];
        while(ww){int b=(int)std::countr_zero(ww);int v=wi*64+b;ww&=ww-1;
            int dd=0;for(int i=0;i<NW;i++)dd+=(int)std::popcount(badj[v].w[i]&cd.w[i]);
            if(dd>bd){bd=dd;bv=v;}}}
    {BS nc=cd;nc.clr(bv);for(int i=0;i<NW;i++)nc.w[i]&=~badj[bv].w[i];
     BS nch=ch;nch.set(bv);bnb(nc,nch,cc+1);}
    {BS nc=cd;nc.clr(bv);bnb(nc,ch,cc);}
}

void solveSC(const vector<int>&sv,const vector<vector<int>>&la,
             const vector<int>&V,vector<int>&result,vector<int>&sid){
    int sn=(int)sv.size();
    for(int i=0;i<sn;i++)sid[sv[i]]=i;
    vector<int>col(sn,-1);bool bip=true;
    for(int i=0;i<sn&&bip;i++){
        if(col[i]>=0)continue;
        queue<int>q;q.push(i);col[i]=0;
        while(!q.empty()&&bip){int u=q.front();q.pop();
            for(int v:la[sv[u]]){int rv=sid[v];if(rv<0)continue;
                if(col[rv]<0){col[rv]=1-col[u];q.push(rv);}
                else if(col[rv]==col[u])bip=false;}}}
    if(bip){
        vector<int>L,R;vector<int>li2(sn,-1),ri2(sn,-1);
        for(int i=0;i<sn;i++){
            if(col[i]==0){li2[i]=(int)L.size();L.push_back(i);}
            else{ri2[i]=(int)R.size();R.push_back(i);}}
        HK hk((int)L.size(),(int)R.size());
        for(int i=0;i<(int)L.size();i++)
            for(int v:la[sv[L[i]]]){int rv=sid[v];if(rv>=0&&ri2[rv]>=0)hk.ae(i,ri2[rv]);}
        hk.run();
        vector<bool>inZ(sn,false);queue<int>q;
        for(int i=0;i<(int)L.size();i++)if(hk.ml[i]<0){inZ[L[i]]=true;q.push(L[i]);}
        while(!q.empty()){int u=q.front();q.pop();
            if(col[u]==0){int lu=li2[u];
                for(int v:la[sv[u]]){int rv=sid[v];if(rv<0||inZ[rv])continue;
                    int rr=ri2[rv];if(rr<0)continue;
                    if(hk.ml[lu]!=rr){inZ[rv]=true;q.push(rv);}}}
            else{int rr=ri2[u];if(hk.mr[rr]>=0){int li=hk.mr[rr];
                if(!inZ[L[li]]){inZ[L[li]]=true;q.push(L[li]);}}}}
        for(int i:L)if(inZ[i])result.push_back(V[sv[i]]);
        for(int i:R)if(!inZ[i])result.push_back(V[sv[i]]);
    } else if(sn<=512){
        NW=(sn+63)/64;
        for(int i=0;i<sn;i++){badj[i]={};
            for(int v:la[sv[i]]){int rv=sid[v];if(rv>=0)badj[i].set(rv);}}
        BS cd={},ch={};for(int i=0;i<sn;i++)cd.set(i);
        bbest=0;bset={};
        {BS gc=cd,gs={};int gcnt=0;
         while(!gc.empty(NW)){int bv=-1,bd=1000000;
            for(int wi=0;wi<NW;wi++){uint64_t ww=gc.w[wi];
                while(ww){int b=(int)std::countr_zero(ww);int v=wi*64+b;ww&=ww-1;
                    int dd=0;for(int i=0;i<NW;i++)dd+=(int)std::popcount(badj[v].w[i]&gc.w[i]);
                    if(dd<bd){bd=dd;bv=v;}}}
            gs.set(bv);gcnt++;gc.clr(bv);
            for(int i=0;i<NW;i++)gc.w[i]&=~badj[bv].w[i];}
         bbest=gcnt;bset=gs;}
        bnb(cd,ch,0);
        for(int i=0;i<sn;i++)if(bset.get(i))result.push_back(V[sv[i]]);
    } else {
        vector<bool>ga(sn,true);
        while(true){int bv=-1,bd=1000000;
            for(int i=0;i<sn;i++){if(!ga[i])continue;
                int dd=0;for(int j:la[sv[i]]){int rv=sid[j];if(rv>=0&&ga[rv])dd++;}
                if(dd<bd){bd=dd;bv=i;}}
            if(bv<0)break;
            result.push_back(V[sv[bv]]);ga[bv]=false;
            for(int j:la[sv[bv]]){int rv=sid[j];if(rv>=0)ga[rv]=false;}}
    }
    for(int i=0;i<sn;i++)sid[sv[i]]=-1;
}

int main(){
    int n,m;
    if(scanf("%d%d",&n,&m)!=2||n==0){printf("0\n\n");return 0;}
    vector<vector<int>>adj(n);
    for(int i=0;i<m;i++){int u,v;scanf("%d%d",&u,&v);
        if(u!=v){adj[u].push_back(v);adj[v].push_back(u);}}
    for(int i=0;i<n;i++){sort(adj[i].begin(),adj[i].end());
        adj[i].erase(unique(adj[i].begin(),adj[i].end()),adj[i].end());}
    vector<int>comp(n,-1);int nc=0;
    for(int i=0;i<n;i++){if(comp[i]>=0)continue;
        queue<int>q;q.push(i);comp[i]=nc;
        while(!q.empty()){int u=q.front();q.pop();
            for(int v:adj[u])if(comp[v]<0){comp[v]=nc;q.push(v);}}
        nc++;}
    vector<vector<int>>cv(nc);
    for(int i=0;i<n;i++)cv[comp[i]].push_back(i);
    vector<int>result;
    vector<int>lid(n,-1);
    for(int c=0;c<nc;c++){
        auto&V=cv[c];int cn=(int)V.size();
        if(cn==1){result.push_back(V[0]);continue;}
        for(int i=0;i<cn;i++)lid[V[i]]=i;
        vector<vector<int>>la(cn);
        for(int i=0;i<cn;i++)for(int v:adj[V[i]]){int lv=lid[v];if(lv>=0)la[i].push_back(lv);}
        vector<bool>act(cn,true);vector<int>forced;
        queue<int>rq;
        for(int i=0;i<cn;i++){int d=(int)la[i].size();if(d<=1)rq.push(i);}
        while(!rq.empty()){int v=rq.front();rq.pop();if(!act[v])continue;
            int d=0,nb=-1;for(int j:la[v])if(act[j]){d++;nb=j;}
            if(d==0){forced.push_back(v);act[v]=false;}
            else if(d==1){forced.push_back(v);act[v]=false;act[nb]=false;
                for(int k:la[nb])if(act[k])rq.push(k);}}
        vector<int>rem;for(int i=0;i<cn;i++)if(act[i])rem.push_back(i);
        if(!rem.empty()){
            vector<int>scc(cn,-1);int nscc=0;
            for(int i:rem){if(scc[i]>=0)continue;
                queue<int>q;q.push(i);scc[i]=nscc;
                while(!q.empty()){int u=q.front();q.pop();
                    for(int v:la[u])if(act[v]&&scc[v]<0){scc[v]=nscc;q.push(v);}}
                nscc++;}
            vector<vector<int>>scv(nscc);
            for(int i:rem)scv[scc[i]].push_back(i);
            vector<int>sid(cn,-1);
            for(int sc=0;sc<nscc;sc++)solveSC(scv[sc],la,V,result,sid);
        }
        for(int i:forced)result.push_back(V[i]);
        for(int i=0;i<cn;i++)lid[V[i]]=-1;
    }
    printf("%d\n",(int)result.size());
    for(int i=0;i<(int)result.size();i++){if(i)putchar(' ');printf("%d",result[i]);}
    putchar('\n');
}
