; Passthrough geometry shader - re-emits input triangle unchanged
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
               OpExecutionMode %main Triangles
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               OpDecorate %in_color Location 2
               OpDecorate %out_color Location 0

               ; Types
      %void  = OpTypeVoid
   %voidfn  = OpTypeFunction %void
     %float  = OpTypeFloat 32
   %v3float  = OpTypeVector %float 3
   %v4float  = OpTypeVector %float 4
       %int  = OpTypeInt 32 1
      %uint  = OpTypeInt 32 0

     %int_0  = OpConstant %int 0
     %int_1  = OpConstant %int 1
     %int_2  = OpConstant %int 2
    %uint_3  = OpConstant %uint 3

               ; Input gl_PerVertex block (array of 3 for Triangles)
%gl_PerVertex_in = OpTypeStruct %v4float
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr = OpTypePointer Input %arr_pv_in
 %ptr_in_v4 = OpTypePointer Input %v4float

     %gl_in  = OpVariable %ptr_in_arr Input

               ; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float

    %gl_out  = OpVariable %ptr_out_pv Output

               ; Input color (location 2, array of 3)
%arr_v3_3   = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_3
 %ptr_in_v3 = OpTypePointer Input %v3float

  %in_color  = OpVariable %ptr_in_arr_v3 Input

               ; Output color (location 0)
%ptr_out_v3 = OpTypePointer Output %v3float
 %out_color  = OpVariable %ptr_out_v3 Output

               ; Main function
      %main  = OpFunction %void None %voidfn
     %entry  = OpLabel

               ; Vertex 0
      %pos0  = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
  %pos0_val  = OpLoad %v4float %pos0
  %out_pos0  = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos0 %pos0_val
      %col0  = OpAccessChain %ptr_in_v3 %in_color %int_0
  %col0_val  = OpLoad %v3float %col0
               OpStore %out_color %col0_val
               OpEmitVertex

               ; Vertex 1
      %pos1  = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
  %pos1_val  = OpLoad %v4float %pos1
  %out_pos1  = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos1 %pos1_val
      %col1  = OpAccessChain %ptr_in_v3 %in_color %int_1
  %col1_val  = OpLoad %v3float %col1
               OpStore %out_color %col1_val
               OpEmitVertex

               ; Vertex 2
      %pos2  = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
  %pos2_val  = OpLoad %v4float %pos2
  %out_pos2  = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos2 %pos2_val
      %col2  = OpAccessChain %ptr_in_v3 %in_color %int_2
  %col2_val  = OpLoad %v3float %col2
               OpStore %out_color %col2_val
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
