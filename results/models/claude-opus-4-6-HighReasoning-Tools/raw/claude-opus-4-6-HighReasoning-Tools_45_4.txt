import struct

def w(val):
    return struct.pack('<I', val & 0xFFFFFFFF)

def string_words(s):
    b = s.encode('utf-8') + b'\x00'
    while len(b) % 4 != 0:
        b += b'\x00'
    words = []
    for i in range(0, len(b), 4):
        words.append(struct.unpack('<I', b[i:i+4])[0])
    return words

def inst(opcode, operands):
    wc = 1 + len(operands)
    result = w((wc << 16) | opcode)
    for op in operands:
        result += w(op)
    return result
