; SPIR-V Geometry Shader: Dashed thick curve from lines with adjacency
; Input: InputLinesAdjacency (4 vertices), Output: OutputTriangleStrip, 8 vertices max

               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 8
               OpExecutionMode %main Invocations 1

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

               ; Types
      %void  = OpTypeVoid
  %voidfunc  = OpTypeFunction %void
     %float  = OpTypeFloat 32
   %v2float  = OpTypeVector %float 2
   %v3float  = OpTypeVector %float 3
   %v4float  = OpTypeVector %float 4
      %uint  = OpTypeInt 32 0
       %int  = OpTypeInt 32 1

               ; Constants
   %int_0    = OpConstant %int 0
   %int_1    = OpConstant %int 1
   %int_2    = OpConstant %int 2
   %int_3    = OpConstant %int 3
  %uint_4    = OpConstant %uint 4
  %float_0   = OpConstant %float 0.0
  %float_1   = OpConstant %float 1.0
  %float_half = OpConstant %float 0.5
  %float_n1  = OpConstant %float -1.0
  %float_thickness = OpConstant %float 0.02
  %float_one_third = OpConstant %float 0.333333343
  %float_two_thirds = OpConstant %float 0.666666687

               ; Input gl_PerVertex block (array of 4)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in   = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr  = OpTypePointer Input %arr_pv_in
%ptr_in_v4   = OpTypePointer Input %v4float
    %gl_in   = OpVariable %ptr_in_arr Input

               ; Input color array
%arr_v3_4    = OpTypeArray %v3float %uint_4
%ptr_in_arr3 = OpTypePointer Input %arr_v3_4
%ptr_in_v3   = OpTypePointer Input %v3float
   %inColor  = OpVariable %ptr_in_arr3 Input

               ; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv  = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4  = OpTypePointer Output %v4float
   %gl_out   = OpVariable %ptr_out_pv Output

               ; Output color
%ptr_out_v3  = OpTypePointer Output %v3float
  %outColor  = OpVariable %ptr_out_v3 Output

               ; Function
     %main   = OpFunction %void None %voidfunc
    %entry   = OpLabel

               ; Load gl_in[0..3].gl_Position
  %p0_ptr    = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
  %p0        = OpLoad %v4float %p0_ptr
  %p1_ptr    = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
  %p1        = OpLoad %v4float %p1_ptr
  %p2_ptr    = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
  %p2        = OpLoad %v4float %p2_ptr
  %p3_ptr    = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
  %p3        = OpLoad %v4float %p3_ptr

               ; Load colors at vertices 1 and 2
  %c1_ptr    = OpAccessChain %ptr_in_v3 %inColor %int_1
  %c1        = OpLoad %v3float %c1_ptr
  %c2_ptr    = OpAccessChain %ptr_in_v3 %inColor %int_2
  %c2        = OpLoad %v3float %c2_ptr

               ; Extract xy from positions (clip space)
  %p0x       = OpCompositeExtract %float %p0 0
  %p0y       = OpCompositeExtract %float %p0 1
  %p1x       = OpCompositeExtract %float %p1 0
  %p1y       = OpCompositeExtract %float %p1 1
  %p2x       = OpCompositeExtract %float %p2 0
  %p2y       = OpCompositeExtract %float %p2 1
  %p3x       = OpCompositeExtract %float %p3 0
  %p3y       = OpCompositeExtract %float %p3 1

               ; Tangent at vertex 1: direction from p0 to p2
  %t1x_raw   = OpFSub %float %p2x %p0x
  %t1y_raw   = OpFSub %float %p2y %p0y
  %t1_len2   = OpFMul %float %t1x_raw %t1x_raw
  %t1_tmp    = OpFMul %float %t1y_raw %t1y_raw
  %t1_len2s  = OpFAdd %float %t1_len2 %t1_tmp
  %t1_len    = OpExtInst %float %1 Sqrt %t1_len2s
  %t1_lenr   = OpFDiv %float %float_1 %t1_len
  %t1x       = OpFMul %float %t1x_raw %t1_lenr
  %t1y       = OpFMul %float %t1y_raw %t1_lenr

               ; Perpendicular at vertex 1: (-t1y, t1x)
  %n1x       = OpFNegate %float %t1y
  %n1y       = OpCopyObject %float %t1x

               ; Tangent at vertex 2: direction from p1 to p3
  %t2x_raw   = OpFSub %float %p3x %p1x
  %t2y_raw   = OpFSub %float %p3y %p1y
  %t2_len2   = OpFMul %float %t2x_raw %t2x_raw
  %t2_tmp    = OpFMul %float %t2y_raw %t2y_raw
  %t2_len2s  = OpFAdd %float %t2_len2 %t2_tmp
  %t2_len    = OpExtInst %float %1 Sqrt %t2_len2s
  %t2_lenr   = OpFDiv %float %float_1 %t2_len
  %t2x       = OpFMul %float %t2x_raw %t2_lenr
  %t2y       = OpFMul %float %t2y_raw %t2_lenr

               ; Perpendicular at vertex 2: (-t2y, t2x)
  %n2x       = OpFNegate %float %t2y
  %n2y       = OpCopyObject %float %t2x

               ; Offset vectors: normal * thickness
  %off1x     = OpFMul %float %n1x %float_thickness
  %off1y     = OpFMul %float %n1y %float_thickness
  %off2x     = OpFMul %float %n2x %float_thickness
  %off2y     = OpFMul %float %n2y %float_thickness

               ; Segment direction from p1 to p2
  %dx        = OpFSub %float %p2x %p1x
  %dy        = OpFSub %float %p2y %p1y

               ; --- Compute points at t=0, 1/3, 2/3, 1 along the segment ---
               ; Point at t=0 (p1)
               ; Already have p1x, p1y

               ; Point at t=1/3
  %dx13      = OpFMul %float %dx %float_one_third
  %dy13      = OpFMul %float %dy %float_one_third
  %pt13x     = OpFAdd %float %p1x %dx13
  %pt13y     = OpFAdd %float %p1y %dy13

               ; Point at t=2/3
  %dx23      = OpFMul %float %dx %float_two_thirds
  %dy23      = OpFMul %float %dy %float_two_thirds
  %pt23x     = OpFAdd %float %p1x %dx23
  %pt23y     = OpFAdd %float %p1y %dy23

               ; Point at t=1 (p2)
               ; Already have p2x, p2y

               ; Interpolate normals at t=1/3
  %n13x_s1   = OpFMul %float %n1x %float_two_thirds
  %n13x_s2   = OpFMul %float %n2x %float_one_third
  %n13x_raw  = OpFAdd %float %n13x_s1 %n13x_s2
  %n13y_s1   = OpFMul %float %n1y %float_two_thirds
  %n13y_s2   = OpFMul %float %n2y %float_one_third
  %n13y_raw  = OpFAdd %float %n13y_s1 %n13y_s2
               ; Normalize
  %n13_l2a   = OpFMul %float %n13x_raw %n13x_raw
  %n13_l2b   = OpFMul %float %n13y_raw %n13y_raw
  %n13_l2    = OpFAdd %float %n13_l2a %n13_l2b
  %n13_l     = OpExtInst %float %1 Sqrt %n13_l2
  %n13_lr    = OpFDiv %float %float_1 %n13_l
  %n13x      = OpFMul %float %n13x_raw %n13_lr
  %n13y      = OpFMul %float %n13y_raw %n13_lr
  %off13x    = OpFMul %float %n13x %float_thickness
  %off13y    = OpFMul %float %n13y %float_thickness

               ; Interpolate normals at t=2/3
  %n23x_s1   = OpFMul %float %n1x %float_one_third
  %n23x_s2   = OpFMul %float %n2x %float_two_thirds
  %n23x_raw  = OpFAdd %float %n23x_s1 %n23x_s2
  %n23y_s1   = OpFMul %float %n1y %float_one_third
  %n23y_s2   = OpFMul %float %n2y %float_two_thirds
  %n23y_raw  = OpFAdd %float %n23y_s1 %n23y_s2
               ; Normalize
  %n23_l2a   = OpFMul %float %n23x_raw %n23x_raw
  %n23_l2b   = OpFMul %float %n23y_raw %n23y_raw
  %n23_l2    = OpFAdd %float %n23_l2a %n23_l2b
  %n23_l     = OpExtInst %float %1 Sqrt %n23_l2
  %n23_lr    = OpFDiv %float %float_1 %n23_l
  %n23x      = OpFMul %float %n23x_raw %n23_lr
  %n23y      = OpFMul %float %n23y_raw %n23_lr
  %off23x    = OpFMul %float %n23x %float_thickness
  %off23y    = OpFMul %float %n23y %float_thickness

               ; Interpolate colors at t=0, 1/3, 2/3, 1
               ; c_at_0 = c1
               ; c_at_13 = mix(c1, c2, 1/3)
  %c13_s1    = OpVectorTimesScalar %v3float %c1 %float_two_thirds
  %c13_s2    = OpVectorTimesScalar %v3float %c2 %float_one_third
  %c13       = OpFAdd %v3float %c13_s1 %c13_s2
               ; c_at_23 = mix(c1, c2, 2/3)
  %c23_s1    = OpVectorTimesScalar %v3float %c1 %float_one_third
  %c23_s2    = OpVectorTimesScalar %v3float %c2 %float_two_thirds
  %c23       = OpFAdd %v3float %c23_s1 %c23_s2
               ; c_at_1 = c2

               ; Extract p1 zw and p2 zw for interpolation
  %p1z       = OpCompositeExtract %float %p1 2
  %p1w       = OpCompositeExtract %float %p1 3
  %p2z       = OpCompositeExtract %float %p2 2
  %p2w       = OpCompositeExtract %float %p2 3

               ; z/w at t=1/3
  %z13_a     = OpFMul %float %p1z %float_two_thirds
  %z13_b     = OpFMul %float %p2z %float_one_third
  %z13       = OpFAdd %float %z13_a %z13_b
  %w13_a     = OpFMul %float %p1w %float_two_thirds
  %w13_b     = OpFMul %float %p2w %float_one_third
  %w13       = OpFAdd %float %w13_a %w13_b

               ; z/w at t=2/3
  %z23_a     = OpFMul %float %p1z %float_one_third
  %z23_b     = OpFMul %float %p2z %float_two_thirds
  %z23       = OpFAdd %float %z23_a %z23_b
  %w23_a     = OpFMul %float %p1w %float_one_third
  %w23_b     = OpFMul %float %p2w %float_two_thirds
  %w23       = OpFAdd %float %w23_a %w23_b

               ; ============ QUAD 1: t=0 to t=1/3 (1st dash) ============
               ; Vertex 0: p1 + offset1 (top-left)
  %q1v0x     = OpFAdd %float %p1x %off1x
  %q1v0y     = OpFAdd %float %p1y %off1y
  %q1v0      = OpCompositeConstruct %v4float %q1v0x %q1v0y %p1z %p1w
  %out_ptr0  = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_ptr0 %q1v0
               OpStore %outColor %c1
               OpEmitVertex

               ; Vertex 1: p1 - offset1 (bottom-left)
  %q1v1x     = OpFSub %float %p1x %off1x
  %q1v1y     = OpFSub %float %p1y %off1y
  %q1v1      = OpCompositeConstruct %v4float %q1v1x %q1v1y %p1z %p1w
               OpStore %out_ptr0 %q1v1
               OpStore %outColor %c1
               OpEmitVertex

               ; Vertex 2: pt13 + offset13 (top-right)
  %q1v2x     = OpFAdd %float %pt13x %off13x
  %q1v2y     = OpFAdd %float %pt13y %off13y
  %q1v2      = OpCompositeConstruct %v4float %q1v2x %q1v2y %z13 %w13
               OpStore %out_ptr0 %q1v2
               OpStore %outColor %c13
               OpEmitVertex

               ; Vertex 3: pt13 - offset13 (bottom-right)
  %q1v3x     = OpFSub %float %pt13x %off13x
  %q1v3y     = OpFSub %float %pt13y %off13y
  %q1v3      = OpCompositeConstruct %v4float %q1v3x %q1v3y %z13 %w13
               OpStore %out_ptr0 %q1v3
               OpStore %outColor %c13
               OpEmitVertex

               OpEndPrimitive

               ; ============ QUAD 2: t=2/3 to t=1 (3rd dash) ============
               ; Vertex 0: pt23 + offset23 (top-left)
  %q2v0x     = OpFAdd %float %pt23x %off23x
  %q2v0y     = OpFAdd %float %pt23y %off23y
  %q2v0      = OpCompositeConstruct %v4float %q2v0x %q2v0y %z23 %w23
               OpStore %out_ptr0 %q2v0
               OpStore %outColor %c23
               OpEmitVertex

               ; Vertex 1: pt23 - offset23 (bottom-left)
  %q2v1x     = OpFSub %float %pt23x %off23x
  %q2v1y     = OpFSub %float %pt23y %off23y
  %q2v1      = OpCompositeConstruct %v4float %q2v1x %q2v1y %z23 %w23
               OpStore %out_ptr0 %q2v1
               OpStore %outColor %c23
               OpEmitVertex

               ; Vertex 2: p2 + offset2 (top-right)
  %q2v2x     = OpFAdd %float %p2x %off2x
  %q2v2y     = OpFAdd %float %p2y %off2y
  %q2v2      = OpCompositeConstruct %v4float %q2v2x %q2v2y %p2z %p2w
               OpStore %out_ptr0 %q2v2
               OpStore %outColor %c2
               OpEmitVertex

               ; Vertex 3: p2 - offset2 (bottom-right)
  %q2v3x     = OpFSub %float %p2x %off2x
  %q2v3y     = OpFSub %float %p2y %off2y
  %q2v3      = OpCompositeConstruct %v4float %q2v3x %q2v3y %p2z %p2w
               OpStore %out_ptr0 %q2v3
               OpStore %outColor %c2
               OpEmitVertex

               OpEndPrimitive

               OpReturn
               OpFunctionEnd
