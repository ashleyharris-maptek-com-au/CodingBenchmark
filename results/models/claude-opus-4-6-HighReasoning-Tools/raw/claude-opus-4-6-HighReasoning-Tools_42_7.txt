; SPIR-V Geometry Shader: Explode triangles along face normal
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main Triangles
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

               ; Types
      %void  = OpTypeVoid
   %voidfun  = OpTypeFunction %void
     %float  = OpTypeFloat 32
   %v3float  = OpTypeVector %float 3
   %v4float  = OpTypeVector %float 4
       %int  = OpTypeInt 32 1
      %uint  = OpTypeInt 32 0

               ; Constants
     %int_0  = OpConstant %int 0
     %int_1  = OpConstant %int 1
     %int_2  = OpConstant %int 2
    %uint_3  = OpConstant %uint 3
   %float_3  = OpConstant %float 3
%float_0_08  = OpConstant %float 0.08
  %float_0   = OpConstant %float 0

               ; Input gl_PerVertex block (array of 3 for Triangles)
%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr  = OpTypePointer Input %arr_pv_in
 %ptr_in_v4  = OpTypePointer Input %v4float
     %gl_in  = OpVariable %ptr_in_arr Input

               ; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv  = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4  = OpTypePointer Output %v4float
    %gl_out  = OpVariable %ptr_out_pv Output

               ; Input color (location=2, array of 3)
%arr3_v3     = OpTypeArray %v3float %uint_3
%ptr_in_arr3 = OpTypePointer Input %arr3_v3
%ptr_in_v3   = OpTypePointer Input %v3float
   %inColor  = OpVariable %ptr_in_arr3 Input

               ; Output color (location=0)
%ptr_out_v3  = OpTypePointer Output %v3float
  %outColor  = OpVariable %ptr_out_v3 Output

               ; Function pointer types
%ptr_fun_v4  = OpTypePointer Function %v4float
%ptr_fun_v3  = OpTypePointer Function %v3float

       %main = OpFunction %void None %voidfun
      %entry = OpLabel

               ; Load gl_in[0].gl_Position
    %ac_p0   = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
       %p0   = OpLoad %v4float %ac_p0
               ; Load gl_in[1].gl_Position
    %ac_p1   = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
       %p1   = OpLoad %v4float %ac_p1
               ; Load gl_in[2].gl_Position
    %ac_p2   = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
       %p2   = OpLoad %v4float %ac_p2

               ; Extract xyz from positions
   %p0_xyz   = OpVectorShuffle %v3float %p0 %p0 0 1 2
   %p1_xyz   = OpVectorShuffle %v3float %p1 %p1 0 1 2
   %p2_xyz   = OpVectorShuffle %v3float %p2 %p2 0 1 2

               ; e1 = p1 - p0, e2 = p2 - p0
       %e1   = OpFSub %v3float %p1_xyz %p0_xyz
       %e2   = OpFSub %v3float %p2_xyz %p0_xyz

               ; N = normalize(cross(e1, e2))
    %cross   = OpExtInst %v3float %1 Cross %e1 %e2
     %norm   = OpExtInst %v3float %1 Normalize %cross

               ; offset = N * 0.08
  %off_vec   = OpVectorTimesScalar %v3float %norm %float_0_08

               ; offset4 = vec4(offset, 0.0)
  %offset4   = OpCompositeConstruct %v4float %off_vec %float_0

               ; --- Vertex 0 ---
 %new_p0     = OpFAdd %v4float %p0 %offset4
 %ac_out0    = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %ac_out0 %new_p0
 %ac_c0      = OpAccessChain %ptr_in_v3 %inColor %int_0
 %c0         = OpLoad %v3float %ac_c0
               OpStore %outColor %c0
               OpEmitVertex

               ; --- Vertex 1 ---
 %new_p1     = OpFAdd %v4float %p1 %offset4
               OpStore %ac_out0 %new_p1
 %ac_c1      = OpAccessChain %ptr_in_v3 %inColor %int_1
 %c1         = OpLoad %v3float %ac_c1
               OpStore %outColor %c1
               OpEmitVertex

               ; --- Vertex 2 ---
 %new_p2     = OpFAdd %v4float %p2 %offset4
               OpStore %ac_out0 %new_p2
 %ac_c2      = OpAccessChain %ptr_in_v3 %inColor %int_2
 %c2         = OpLoad %v3float %ac_c2
               OpStore %outColor %c2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
