import sys

def solve():
    input_data = sys.stdin.read().split('\n')
    idx = 0
    U, S = map(int, input_data[idx].split())
    idx += 1
    sets = []
    for i in range(S):
        line = input_data[idx].strip()
        idx += 1
        if line:
            sets.append(list(map(int, line.split())))
        else:
            sets.append([])

    # Algorithm X with Dancing Links (DLX)
    # Build sparse matrix: rows = sets, columns = elements 0..U-1

    # Node structure: each node has L, R, U, D, C (column header), row_id
    # We use arrays for speed

    L = []
    R = []
    Up = []
    D = []
    C = []  # column header index for this node
    row_id = []  # which set this node belongs to
    col_size = []  # size of each column

    num_nodes = 0

    def new_node():
        nonlocal num_nodes
        n = num_nodes
        L.append(0)
        R.append(0)
        Up.append(0)
        D.append(0)
        C.append(0)
        row_id.append(-1)
        num_nodes += 1
        return n

    # Root node
    root = new_node()  # node 0
    L[root] = R[root] = Up[root] = D[root] = root

    # Column headers: nodes 1..U
    col_headers = []
    col_size_list = []
    for j in range(U):
        h = new_node()
        col_headers.append(h)
        col_size_list.append(0)
        C[h] = h
        Up[h] = D[h] = h
        # Link into header row
        L[h] = L[root]
        R[L[root]] = h
        R[h] = root
        L[root] = h

    col_size = col_size_list

    # Add rows
    for i, s in enumerate(sets):
        if not s:
            continue
        first = -1
        for elem in s:
            n = new_node()
            row_id[n] = i
            c = col_headers[elem]
            C[n] = c
            col_size[elem] += 1
            # Insert into column (above header = bottom of column)
            Up[n] = Up[c]
            D[Up[c]] = n
            D[n] = c
            Up[c] = n
            # Link horizontally
            if first == -1:
                first = n
                L[n] = R[n] = n
            else:
                L[n] = L[first]
                R[L[first]] = n
                R[n] = first
                L[first] = n

    def cover(c):
        R[L[c]] = R[c]
        L[R[c]] = L[c]
        i = D[c]
        while i != c:
            j = R[i]
            while j != i:
                D[Up[j]] = D[j]
                Up[D[j]] = Up[j]
                col_size[col_headers.index(C[j])] -= 1 if False else 0
                j = R[j]
            i = D[i]

    def uncover(c):
        i = Up[c]
        while i != c:
            j = L[i]
            while j != i:
                D[Up[j]] = j
                Up[D[j]] = j
                j = L[j]
            i = Up[i]
        R[L[c]] = c
        L[R[c]] = c

    # Faster: track col sizes by column index
    # Redo with simpler approach using dict for col index
    col_idx = {}
    for j in range(U):
        col_idx[col_headers[j]] = j

    col_sz = [0] * U
    # Recount
    for j in range(U):
        h = col_headers[j]
        cnt = 0
        n = D[h]
        while n != h:
            cnt += 1
            n = D[n]
        col_sz[j] = cnt

    def cover2(c):
        ci = col_idx[c]
        R[L[c]] = R[c]
        L[R[c]] = L[c]
        i = D[c]
        while i != c:
            j = R[i]
            while j != i:
                D[Up[j]] = D[j]
                Up[D[j]] = Up[j]
                col_sz[col_idx[C[j]]] -= 1
                j = R[j]
            i = D[i]

    def uncover2(c):
        i = Up[c]
        while i != c:
            j = L[i]
            while j != i:
                D[Up[j]] = j
                Up[D[j]] = j
                col_sz[col_idx[C[j]]] += 1
                j = L[j]
            i = Up[i]
        R[L[c]] = c
        L[R[c]] = c

    solution = []

    def search():
        if R[root] == root:
            return True
        # Choose column with minimum size (MRV heuristic)
        min_sz = float('inf')
        chosen = -1
        j = R[root]
        while j != root:
            ci = col_idx[j]
            if col_sz[ci] < min_sz:
                min_sz = col_sz[ci]
                chosen = j
                if min_sz <= 1:
                    break
            j = R[j]
        if min_sz == 0:
            return False
        cover2(chosen)
        r = D[chosen]
        while r != chosen:
            solution.append(row_id[r])
            j = R[r]
            while j != r:
                cover2(C[j])
                j = R[j]
            if search():
                return True
            solution.pop()
            j = L[r]
            while j != r:
                uncover2(C[j])
                j = L[j]
            r = D[r]
        uncover2(chosen)
        return False

    if search():
        print("SOLUTION")
        print(" ".join(map(str, sorted(solution))))
    else:
        print("NO SOLUTION")

solve()
