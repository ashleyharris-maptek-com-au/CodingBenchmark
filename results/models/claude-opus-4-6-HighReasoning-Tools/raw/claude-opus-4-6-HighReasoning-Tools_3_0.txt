use std::io::{self, Read, Write, BufWriter};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut iter = input.split_whitespace();
    let n: usize = iter.next().unwrap().parse().unwrap();
    let m: usize = iter.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![vec![]; n];
    for _ in 0..m {
        let u: usize = iter.next().unwrap().parse().unwrap();
        let v: usize = iter.next().unwrap().parse().unwrap();
        if u == v || u >= n || v >= n {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    for i in 0..n {
        adj[i].sort_unstable();
        adj[i].dedup();
    }

    let mut x = vec![0.0f64; n];
    let mut y = vec![0.0f64; n];

    if n <= 1 {
        for i in 0..n {
            writeln!(out, "{:.6} {:.6}", i as f64, 0.0).unwrap();
        }
        return;
    }

    let mut comp = vec![usize::MAX; n];
    let mut comp_id = 0usize;
    let mut comp_sizes: Vec<usize> = Vec::new();
    let mut queue = Vec::new();
    for start in 0..n {
        if comp[start] != usize::MAX {
            continue;
        }
        let mut size = 0usize;
        queue.push(start);
        comp[start] = comp_id;
        while let Some(node) = queue.pop() {
            size += 1;
            for &nb in &adj[node] {
                if comp[nb] == usize::MAX {
                    comp[nb] = comp_id;
                    queue.push(nb);
                }
            }
        }
        comp_sizes.push(size);
        comp_id += 1;
    }

    let num_comps = comp_sizes.len();
    let mut comp_nodes: Vec<Vec<usize>> = vec![vec![]; num_comps];
    for i in 0..n {
        comp_nodes[comp[i]].push(i);
    }

    let mut comp_positions: Vec<Vec<(f64, f64)>> = Vec::with_capacity(num_comps);

    for c in 0..num_comps {
        let nodes = &comp_nodes[c];
        let cn = nodes.len();

        if cn == 1 {
            comp_positions.push(vec![(0.0, 0.0)]);
            continue;
        }

        if cn == 2 {
            comp_positions.push(vec![(0.0, 0.0), (1.0, 0.0)]);
            continue;
        }

        let mut local_id = vec![0usize; n];
        for (i, &node) in nodes.iter().enumerate() {
            local_id[node] = i;
        }
        let mut local_adj: Vec<Vec<usize>> = vec![vec![]; cn];
        for &node in nodes {
            let li = local_id[node];
            for &nb in &adj[node] {
                if comp[nb] == c {
                    local_adj[li].push(local_id[nb]);
                }
            }
        }

        let positions = if cn <= 5000 {
            layout_force_directed(&local_adj, cn)
        } else {
            layout_multilevel(&local_adj, cn)
        };

        comp_positions.push(positions);
    }

    let mut offset_x = 0.0f64;
    for c in 0..num_comps {
        let positions = &comp_positions[c];
        let nodes = &comp_nodes[c];

        let mut min_x = f64::MAX;
        let mut max_x = f64::MIN;
        let mut min_y = f64::MAX;
        let mut max_y = f64::MIN;
        for &(px, py) in positions.iter() {
            if px < min_x { min_x = px; }
            if px > max_x { max_x = px; }
            if py < min_y { min_y = py; }
            if py > max_y { max_y = py; }
        }

        let width = max_x - min_x;
        for (i, &node) in nodes.iter().enumerate() {
            x[node] = positions[i].0 - min_x + offset_x;
            y[node] = positions[i].1 - min_y;
        }
        offset_x += width + 2.0;
    }

    writeln!(out, "{}", n).unwrap();
    for i in 0..n {
        writeln!(out, "{:.6} {:.6}", x[i], y[i]).unwrap();
    }
}

fn layout_force_directed(adj: &[Vec<usize>], n: usize) -> Vec<(f64, f64)> {
    let mut x = vec![0.0f64; n];
    let mut y = vec![0.0f64; n];

    let mut max_deg_node = 0;
    let mut max_deg = 0;
    for i in 0..n {
        if adj[i].len() > max_deg {
            max_deg = adj[i].len();
            max_deg_node = i;
        }
    }

    let mut dist = vec![usize::MAX; n];
    let mut bfs_queue = std::collections::VecDeque::new();
    dist[max_deg_node] = 0;
    bfs_queue.push_back(max_deg_node);
    while let Some(node) = bfs_queue.pop_front() {
        for &nb in &adj[node] {
            if dist[nb] == usize::MAX {
                dist[nb] = dist[node] + 1;
                bfs_queue.push_back(nb);
            }
        }
    }

    let mut rng_state: u64 = 12345;
    let mut next_rand = || -> f64 {
        rng_state = rng_state.wrapping_mul(6364136223846793005).wrapping_add(1442695040888963407);
        ((rng_state >> 33) as f64) / (2147483648.0)
    };

    let k = (n as f64).sqrt();
    for i in 0..n {
        if dist[i] != usize::MAX {
            x[i] = dist[i] as f64 * k * 0.5 + next_rand() * k * 0.1;
            y[i] = next_rand() * k * (n as f64).sqrt() * 0.5;
        } else {
            x[i] = next_rand() * k * (n as f64).sqrt();
            y[i] = next_rand() * k * (n as f64).sqrt();
        }
    }

    let iterations = if n <= 50 { 500 } else if n <= 200 { 300 } else if n <= 1000 { 200 } else { 100 };
    let mut temp = k * (n as f64).sqrt() * 0.1;
    let cooling = 0.92;
    let use_barnes_hut = n > 200;

    for _iter in 0..iterations {
        let mut dx = vec![0.0f64; n];
        let mut dy = vec![0.0f64; n];

        if use_barnes_hut {
            barnes_hut_repulsion(&x, &y, n, k, &mut dx, &mut dy);
        } else {
            for i in 0..n {
                for j in (i + 1)..n {
                    let ddx = x[i] - x[j];
                    let ddy = y[i] - y[j];
                    let dist_sq = ddx * ddx + ddy * ddy;
                    let d = dist_sq.sqrt().max(0.01);
                    let force = k * k / d;
                    let fx = ddx / d * force;
                    let fy = ddy / d * force;
                    dx[i] += fx;
                    dy[i] += fy;
                    dx[j] -= fx;
                    dy[j] -= fy;
                }
            }
        }

        for i in 0..n {
            for &j in &adj[i] {
                if j > i { continue; }
                let ddx = x[i] - x[j];
                let ddy = y[i] - y[j];
                let d = (ddx * ddx + ddy * ddy).sqrt().max(0.01);
                let force = d / k;
                let fx = ddx / d * force;
                let fy = ddy / d * force;
                dx[i] -= fx;
                dy[i] -= fy;
                dx[j] += fx;
                dy[j] += fy;
            }
        }

        for i in 0..n {
            let disp = (dx[i] * dx[i] + dy[i] * dy[i]).sqrt().max(0.0001);
            let scale = temp.min(disp) / disp;
            x[i] += dx[i] * scale;
            y[i] += dy[i] * scale;
        }

        temp *= cooling;
    }

    (0..n).map(|i| (x[i], y[i])).collect()
}

struct QuadTree {
    cx: f64,
    cy: f64,
    mass: f64,
    size: f64,
    min_x: f64,
    min_y: f64,
    children: [Option<Box<QuadTree>>; 4],
    is_leaf: bool,
    node_x: f64,
    node_y: f64,
}

impl QuadTree {
    fn new(min_x: f64, min_y: f64, size: f64) -> Self {
        QuadTree {
            cx: 0.0, cy: 0.0, mass: 0.0, size,
            min_x, min_y,
            children: [None, None, None, None],
            is_leaf: true,
            node_x: 0.0, node_y: 0.0,
        }
    }

    fn insert(&mut self, px: f64, py: f64) {
        if self.mass == 0.0 {
            self.cx = px;
            self.cy = py;
            self.mass = 1.0;
            self.node_x = px;
            self.node_y = py;
            self.is_leaf = true;
            return;
        }

        if self.is_leaf && self.mass == 1.0 {
            if self.size < 1e-6 {
                self.mass += 1.0;
                self.cx = (self.cx * (self.mass - 1.0) + px) / self.mass;
                self.cy = (self.cy * (self.mass - 1.0) + py) / self.mass;
                return;
            }
            let old_x = self.node_x;
            let old_y = self.node_y;
            self.is_leaf = false;
            self.insert_into_child(old_x, old_y);
        }

        self.mass += 1.0;
        self.cx = (self.cx * (self.mass - 1.0) + px) / self.mass;
        self.cy = (self.cy * (self.mass - 1.0) + py) / self.mass;
        self.insert_into_child(px, py);
    }

    fn insert_into_child(&mut self, px: f64, py: f64) {
        let half = self.size / 2.0;
        let mid_x = self.min_x + half;
        let mid_y = self.min_y + half;
        let idx = if px < mid_x { 0 } else { 1 } + if py < mid_y { 0 } else { 2 };
        let child_min_x = if px < mid_x { self.min_x } else { mid_x };
        let child_min_y = if py < mid_y { self.min_y } else { mid_y };

        if self.children[idx].is_none() {
            self.children[idx] = Some(Box::new(QuadTree::new(child_min_x, child_min_y, half)));
        }
        self.children[idx].as_mut().unwrap().insert(px, py);
    }

    fn compute_force(&self, px: f64, py: f64, k: f64, theta: f64, fx: &mut f64, fy: &mut f64) {
        if self.mass == 0.0 { return; }

        let ddx = px - self.cx;
        let ddy = py - self.cy;
        let dist_sq = ddx * ddx + ddy * ddy;

        if self.is_leaf || self.size / dist_sq.sqrt().max(0.001) < theta {
            if dist_sq < 1e-8 { return; }
            let d = dist_sq.sqrt();
            let force = k * k * self.mass / d;
            *fx += ddx / d * force;
            *fy += ddy / d * force;
            return;
        }

        for child in &self.children {
            if let Some(c) = child {
                c.compute_force(px, py, k, theta, fx, fy);
            }
        }
    }
}

fn barnes_hut_repulsion(x: &[f64], y: &[f64], n: usize, k: f64, dx: &mut [f64], dy: &mut [f64]) {
    let mut min_x = f64::MAX;
    let mut max_x = f64::MIN;
    let mut min_y = f64::MAX;
    let mut max_y = f64::MIN;
    for i in 0..n {
        if x[i] < min_x { min_x = x[i]; }
        if x[i] > max_x { max_x = x[i]; }
        if y[i] < min_y { min_y = y[i]; }
        if y[i] > max_y { max_y = y[i]; }
    }
    let size = (max_x - min_x).max(max_y - min_y).max(1.0) * 1.01;

    let mut tree = QuadTree::new(min_x - 0.005 * size, min_y - 0.005 * size, size);
    for i in 0..n {
        tree.insert(x[i], y[i]);
    }

    let theta = 1.2;
    for i in 0..n {
        let mut fx = 0.0;
        let mut fy = 0.0;
        tree.compute_force(x[i], y[i], k, theta, &mut fx, &mut fy);
        dx[i] += fx;
        dy[i] += fy;
    }
}

fn layout_multilevel(adj: &[Vec<usize>], n: usize) -> Vec<(f64, f64)> {
    if n <= 5000 {
        return layout_force_directed(adj, n);
    }

    let mut matched = vec![false; n];
    let mut coarse_id = vec![0usize; n];
    let mut coarse_n = 0usize;

    for i in 0..n {
        if matched[i] { continue; }
        let mut best = usize::MAX;
        let mut best_weight = 0;
        for &nb in &adj[i] {
            if !matched[nb] && adj[nb].len() > best_weight {
                best = nb;
                best_weight = adj[nb].len();
            }
        }
        if best != usize::MAX {
            coarse_id[i] = coarse_n;
            coarse_id[best] = coarse_n;
            matched[i] = true;
            matched[best] = true;
        } else {
            coarse_id[i] = coarse_n;
            matched[i] = true;
        }
        coarse_n += 1;
    }

    let mut coarse_adj: Vec<Vec<usize>> = vec![vec![]; coarse_n];
    for i in 0..n {
        for &nb in &adj[i] {
            let ci = coarse_id[i];
            let cj = coarse_id[nb];
            if ci != cj {
                coarse_adj[ci].push(cj);
            }
        }
    }
    for i in 0..coarse_n {
        coarse_adj[i].sort_unstable();
        coarse_adj[i].dedup();
    }

    let coarse_positions = layout_multilevel(&coarse_adj, coarse_n);

    let k = (n as f64).sqrt();
    let mut rng_state: u64 = 67890;
    let mut next_rand = || -> f64 {
        rng_state = rng_state.wrapping_mul(6364136223846793005).wrapping_add(1442695040888963407);
        ((rng_state >> 33) as f64) / (2147483648.0) - 0.5
    };

    let mut x = vec![0.0f64; n];
    let mut y = vec![0.0f64; n];
    for i in 0..n {
        let (cx, cy) = coarse_positions[coarse_id[i]];
        x[i] = cx + next_rand() * k * 0.05;
        y[i] = cy + next_rand() * k * 0.05;
    }

    let iterations = 50;
    let mut temp = k * 0.5;
    let cooling = 0.9;

    for _iter in 0..iterations {
        let mut dx = vec![0.0f64; n];
        let mut dy = vec![0.0f64; n];

        barnes_hut_repulsion(&x, &y, n, k, &mut dx, &mut dy);

        for i in 0..n {
            for &j in &adj[i] {
                if j > i { continue; }
                let ddx = x[i] - x[j];
                let ddy = y[i] - y[j];
                let d = (ddx * ddx + ddy * ddy).sqrt().max(0.01);
                let force = d / k;
                let fx = ddx / d * force;
                let fy = ddy / d * force;
                dx[i] -= fx;
                dy[i] -= fy;
                dx[j] += fx;
                dy[j] += fy;
            }
        }

        for i in 0..n {
            let disp = (dx[i] * dx[i] + dy[i] * dy[i]).sqrt().max(0.0001);
            let scale = temp.min(disp) / disp;
            x[i] += dx[i] * scale;
            y[i] += dy[i] * scale;
        }

        temp *= cooling;
    }

    (0..n).map(|i| (x[i], y[i])).collect()
}
