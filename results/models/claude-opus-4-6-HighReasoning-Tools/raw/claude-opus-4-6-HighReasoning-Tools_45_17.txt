import struct

words = []

def w(val):
    words.append(val & 0xFFFFFFFF)

def op(opcode, *operands):
    wc = 1 + len(operands)
    w((wc << 16) | opcode)
    for o in operands:
        w(o)

def op_str(opcode, *operands_before_str, string="", operands_after=None):
    if operands_after is None:
        operands_after = []
    encoded = string.encode('utf-8') + b'\x00'
    while len(encoded) % 4 != 0:
        encoded += b'\x00'
    str_words = len(encoded) // 4
    wc = 1 + len(operands_before_str) + str_words + len(operands_after)
    w((wc << 16) | opcode)
    for o in operands_before_str:
        w(o)
    for i in range(str_words):
        chunk = encoded[i*4:(i+1)*4]
        w(struct.unpack('<I', chunk)[0])
    for o in operands_after:
        w(o)

def float_to_word(f):
    return struct.unpack('<I', struct.pack('<f', f))[0]
