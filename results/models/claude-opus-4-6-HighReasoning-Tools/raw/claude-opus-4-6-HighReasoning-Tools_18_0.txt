// Global min-cut via Stoer-Wagner algorithm
// For undirected graphs, finds the minimum edge connectivity (lambda).
// Time: O(V * E * log V) with priority queue, or O(V^3) with adjacency matrix.

struct Edge {
    int to, rev;
    int cap, flow;
};

// Dinic's max-flow for s-t min-cut
struct Dinic {
    int n;
    std::vector<std::vector<Edge>> graph;
    std::vector<int> level, iter;

    Dinic(int n) : n(n), graph(n), level(n), iter(n) {}

    void addEdge(int from, int to, int cap) {
        graph[from].push_back({to, (int)graph[to].size(), cap, 0});
        graph[to].push_back({from, (int)graph[from].size() - 1, cap, 0});
    }

    bool bfs(int s, int t) {
        std::fill(level.begin(), level.end(), -1);
        std::queue<int> q;
        level[s] = 0;
        q.push(s);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto& e : graph[v]) {
                if (e.cap - e.flow > 0 && level[e.to] < 0) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }
        return level[t] >= 0;
    }

    int dfs(int v, int t, int f) {
        if (v == t) return f;
        for (int& i = iter[v]; i < (int)graph[v].size(); i++) {
            Edge& e = graph[v][i];
            if (e.cap - e.flow > 0 && level[v] < level[e.to]) {
                int d = dfs(e.to, t, std::min(f, e.cap - e.flow));
                if (d > 0) {
                    e.flow += d;
                    graph[e.to][e.rev].flow -= d;
                    return d;
                }
            }
        }
        return 0;
    }

    int maxflow(int s, int t) {
        int flow = 0;
        while (bfs(s, t)) {
            std::fill(iter.begin(), iter.end(), 0);
            int d;
            while ((d = dfs(s, t, std::numeric_limits<int>::max())) > 0)
                flow += d;
        }
        return flow;
    }

    // After maxflow, find which nodes are reachable from s in residual graph
    std::vector<bool> minCutSide(int s) {
        std::vector<bool> visited(n, false);
        std::queue<int> q;
        visited[s] = 0;
        q.push(s);
        visited[s] = true;
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto& e : graph[v]) {
                if (e.cap - e.flow > 0 && !visited[e.to]) {
                    visited[e.to] = true;
                    q.push(e.to);
                }
            }
        }
        return visited;
    }
};

// Stoer-Wagner for dense/medium graphs using adjacency matrix representation
// Returns the minimum cut value and the partition
struct StoerWagner {
    int n;
    std::vector<std::vector<int>> w; // adjacency weight matrix (edge counts for multigraph)

    StoerWagner(int n) : n(n), w(n, std::vector<int>(n, 0)) {}

    void addEdge(int u, int v) {
        w[u][v]++;
        w[v][u]++;
    }

    // Returns {min_cut_value, partition (nodes in one side)}
    std::pair<int, std::vector<int>> solve() {
        int bestCut = std::numeric_limits<int>::max();
        std::vector<int> bestPartition;

        // merged[i] = list of original nodes merged into supernode i
        std::vector<std::vector<int>> merged(n);
        for (int i = 0; i < n; i++) merged[i] = {i};

        std::vector<bool> active(n, true);
        std::vector<int> activeNodes;
        for (int i = 0; i < n; i++) activeNodes.push_back(i);

        for (int phase = 0; phase < n - 1; phase++) {
            // Maximum adjacency ordering
            int numActive = (int)activeNodes.size();
            std::vector<int> key(n, 0);
            std::vector<bool> inA(n, false);

            int prev = -1, last = -1;

            for (int i = 0; i < numActive; i++) {
                // Find node not in A with max key
                int best = -1;
                for (int node : activeNodes) {
                    if (!inA[node] && (best == -1 || key[node] > key[best]))
                        best = node;
                }
                inA[best] = true;
                prev = last;
                last = best;

                // Update keys
                for (int node : activeNodes) {
                    if (!inA[node]) {
                        key[node] += w[best][node];
                    }
                }
            }

            // The cut-of-the-phase separates 'last' from the rest
            int cutValue = key[last];
            if (cutValue < bestCut) {
                bestCut = cutValue;
                bestPartition = merged[last];
            }

            // Merge last into prev
            if (prev >= 0) {
                for (int node : activeNodes) {
                    if (node != last && node != prev) {
                        w[prev][node] += w[last][node];
                        w[node][prev] += w[node][last];
                    }
                }
                for (int x : merged[last]) merged[prev].push_back(x);
                active[last] = false;
                activeNodes.erase(std::find(activeNodes.begin(), activeNodes.end(), last));
            }
        }

        return {bestCut, bestPartition};
    }
};

// For sparse graphs: use Dinic's with edge connectivity approach
// Edge connectivity = min over all t of maxflow(s=0, t)
// With the Gabow/NI tree approach or just sampling, we can speed this up.
// But a practical approach: fix s=0, try a few random t values, take min.
// Actually for correctness: edge connectivity = min_{t != s} maxflow(s, t).
// We use the NI (Nagamochi-Ibaraki) / Stoer-Wagner approach for the global min cut.

// For very large sparse graphs, we use a sparse Stoer-Wagner with priority queue
struct StoerWagnerSparse {
    int n;
    // adjacency list with weights
    std::vector<std::vector<std::pair<int,int>>> adj;

    StoerWagnerSparse(int n) : n(n), adj(n) {}

    void addEdge(int u, int v) {
        // We'll build the adjacency after all edges are added
        adj[u].push_back({v, 1});
        adj[v].push_back({u, 1});
    }

    std::pair<int, std::vector<int>> solve() {
        // Use sparse weight representation
        // merged[i] = original nodes in supernode i
        std::vector<std::vector<int>> merged(n);
        for (int i = 0; i < n; i++) merged[i] = {i};

        // Sparse weight map for each node
        std::vector<std::vector<std::pair<int,int>>> wadj(n);
        // Build from adj
        {
            // Consolidate adjacency
            for (int u = 0; u < n; u++) {
                std::vector<std::pair<int,int>>& a = adj[u];
                std::sort(a.begin(), a.end());
                wadj[u].clear();
                for (int i = 0; i < (int)a.size(); ) {
                    int j = i;
                    int v = a[i].first;
                    int wt = 0;
                    while (j < (int)a.size() && a[j].first == v) { wt += a[j].second; j++; }
                    wadj[u].push_back({v, wt});
                    i = j;
                }
            }
            adj.clear();
            adj.shrink_to_fit();
        }

        std::vector<bool> active(n, true);

        int bestCut = std::numeric_limits<int>::max();
        std::vector<int> bestPartition;

        for (int phase = 0; phase < n - 1; phase++) {
            // Maximum adjacency ordering using priority queue
            std::vector<int> key(n, 0);
            std::vector<bool> inA(n, false);
            // max-heap: {key, node}
            std::priority_queue<std::pair<int,int>> pq;

            // Find first active node
            int startNode = -1;
            for (int i = 0; i < n; i++) {
                if (active[i]) {
                    if (startNode == -1) startNode = i;
                    pq.push({0, i});
                }
            }

            int prev = -1, last = -1;
            int numActive = 0;
            for (int i = 0; i < n; i++) if (active[i]) numActive++;

            for (int i = 0; i < numActive; i++) {
                int best = -1;
                while (!pq.empty()) {
                    auto [k, node] = pq.top(); pq.pop();
                    if (!inA[node] && active[node] && k == key[node]) {
                        best = node;
                        break;
                    }
                }
                if (best == -1) break;

                inA[best] = true;
                prev = last;
                last = best;

                for (auto& [v, wt] : wadj[best]) {
                    if (active[v] && !inA[v]) {
                        key[v] += wt;
                        pq.push({key[v], v});
                    }
                }
            }

            int cutValue = key[last];
            if (cutValue < bestCut) {
                bestCut = cutValue;
                bestPartition = merged[last];
            }

            // Merge last into prev
            if (prev >= 0 && last >= 0) {
                // Merge adjacency lists
                // First, collect all neighbors of last (except prev)
                // and add them to prev's adjacency
                std::vector<std::pair<int,int>> newAdj;

                // Combine wadj[prev] and wadj[last], merging weights for shared neighbors
                // Remove self-loops (prev-last edges)
                std::vector<std::pair<int,int>>& aP = wadj[prev];
                std::vector<std::pair<int,int>>& aL = wadj[last];

                // Sort both
                std::sort(aP.begin(), aP.end());
                std::sort(aL.begin(), aL.end());

                int ip = 0, il = 0;
                while (ip < (int)aP.size() && il < (int)aL.size()) {
                    if (aP[ip].first == last) { ip++; continue; }
                    if (aL[il].first == prev) { il++; continue; }
                    if (!active[aP[ip].first]) { ip++; continue; }
                    if (!active[aL[il].first]) { il++; continue; }

                    if (aP[ip].first < aL[il].first) {
                        newAdj.push_back(aP[ip]); ip++;
                    } else if (aP[ip].first > aL[il].first) {
                        newAdj.push_back(aL[il]); il++;
                    } else {
                        newAdj.push_back({aP[ip].first, aP[ip].second + aL[il].second});
                        ip++; il++;
                    }
                }
                while (ip < (int)aP.size()) {
                    if (aP[ip].first != last && active[aP[ip].first])
                        newAdj.push_back(aP[ip]);
                    ip++;
                }
                while (il < (int)aL.size()) {
                    if (aL[il].first != prev && active[aL[il].first])
                        newAdj.push_back(aL[il]);
                    il++;
                }

                wadj[prev] = std::move(newAdj);

                // Update neighbors of last to point to prev instead
                for (auto& [v, wt] : wadj[last]) {
                    if (v == prev || !active[v]) continue;
                    for (auto& [nv, nw] : wadj[v]) {
                        if (nv == last) {
                            nv = prev;
                        }
                    }
                    // Consolidate duplicates in wadj[v]
                    std::sort(wadj[v].begin(), wadj[v].end());
                    std::vector<std::pair<int,int>> consolidated;
                    for (int i2 = 0; i2 < (int)wadj[v].size(); ) {
                        int j2 = i2;
                        int tw = 0;
                        int tv = wadj[v][i2].first;
                        while (j2 < (int)wadj[v].size() && wadj[v][j2].first == tv) {
                            tw += wadj[v][j2].second; j2++;
                        }
                        if (active[tv] && tv != v)
                            consolidated.push_back({tv, tw});
                        i2 = j2;
                    }
                    wadj[v] = std::move(consolidated);
                }

                for (int x : merged[last]) merged[prev].push_back(x);
                active[last] = false;
                wadj[last].clear();
                wadj[last].shrink_to_fit();
            }
        }

        return {bestCut, bestPartition};
    }
};

int main() {
    // Fast I/O
    std::vector<std::pair<int,int>> edges;

    int N, M;
    if (scanf("%d %d", &N, &M) != 2) return 1;

    if (N <= 1) {
        // Can't disconnect a single node or empty graph
        printf("0\n");
        return 0;
    }

    edges.reserve(M);
    for (int i = 0; i < M; i++) {
        int u, v;
        scanf("%d %d", &u, &v);
        edges.push_back({u, v});
    }

    // Strategy selection based on graph size
    // For N <= ~3000, use dense Stoer-Wagner O(N^3)
    // For larger sparse graphs, use sparse Stoer-Wagner
    // For very large graphs where min-degree is small, try degree-based heuristic first

    // Quick check: if any node has degree 1, that's a bridge candidate (cut = 1)
    std::vector<int> degree(N, 0);
    for (auto& [u, v] : edges) {
        degree[u]++;
        degree[v]++;
    }

    int minDeg = *std::min_element(degree.begin(), degree.end());

    // The edge connectivity <= minimum degree
    // If minDeg == 1, just find that node and output its single edge
    if (minDeg == 1) {
        for (int u = 0; u < N; u++) {
            if (degree[u] == 1) {
                for (auto& [a, b] : edges) {
                    if (a == u || b == u) {
                        printf("1\n%d %d\n", a, b);
                        return 0;
                    }
                }
            }
        }
    }

    // If minDeg == 0, graph already has isolated node (but problem says connected)
    // Check for bridges using DFS (cut value = 1)
    // This is O(N + M) and handles the most common fast case
    {
        std::vector<std::vector<std::pair<int,int>>> adj(N);
        for (int i = 0; i < M; i++) {
            auto [u, v] = edges[i];
            adj[u].push_back({v, i});
            adj[v].push_back({u, i});
        }

        // Find bridges using Tarjan's algorithm (iterative, multigraph-safe)
        std::vector<int> disc(N, -1), low(N, -1);
        int timer = 0;
        int bridgeIdx = -1;

        struct Frame {
            int node, parentEdgeIdx;
            int childIter;
        };
        std::vector<Frame> stack;

        for (int start = 0; start < N && bridgeIdx == -1; start++) {
            if (disc[start] != -1) continue;
            stack.push_back({start, -1, 0});
            disc[start] = low[start] = timer++;

            while (!stack.empty() && bridgeIdx == -1) {
                Frame& f = stack.back();
                bool pushed = false;

                while (f.childIter < (int)adj[f.node].size()) {
                    auto [v, eidx] = adj[f.node][f.childIter];
                    f.childIter++;

                    if (eidx == f.parentEdgeIdx) continue;

                    if (disc[v] == -1) {
                        disc[v] = low[v] = timer++;
                        stack.push_back({v, eidx, 0});
                        pushed = true;
                        break;
                    } else {
                        low[f.node] = std::min(low[f.node], disc[v]);
                    }
                }

                if (!pushed) {
                    int node = f.node;
                    int pedge = f.parentEdgeIdx;
                    stack.pop_back();

                    if (!stack.empty()) {
                        Frame& pf = stack.back();
                        low[pf.node] = std::min(low[pf.node], low[node]);

                        if (low[node] > disc[pf.node]) {
                            bridgeIdx = pedge;
                        }
                    }
                }
            }
        }

        if (bridgeIdx != -1) {
            printf("1\n%d %d\n", edges[bridgeIdx].first, edges[bridgeIdx].second);
            return 0;
        }
    }

    // For the general case, use Stoer-Wagner
    // The min cut <= minDeg. We can use this as an upper bound.

    if (N <= 2500) {
        // Dense Stoer-Wagner with adjacency matrix
        StoerWagner sw(N);
        for (auto& [u, v] : edges) {
            sw.addEdge(u, v);
        }

        auto [cutVal, partition] = sw.solve();

        if (cutVal == 0) {
            printf("0\n");
            return 0;
        }

        // Find the actual cut edges
        std::vector<bool> side(N, false);
        for (int x : partition) side[x] = true;

        std::vector<std::pair<int,int>> cutEdges;
        for (auto& [u, v] : edges) {
            if (side[u] != side[v]) {
                cutEdges.push_back({u, v});
            }
        }

        printf("%d\n", (int)cutEdges.size());
        for (auto& [u, v] : cutEdges) {
            printf("%d %d\n", u, v);
        }
    } else {
        // For larger graphs, use flow-based approach with the min-degree bound
        // Fix source s = node with minimum degree
        // Try multiple sinks, take minimum
        // This is often faster than full Stoer-Wagner for sparse graphs

        // Find node with minimum degree
        int s = 0;
        for (int i = 1; i < N; i++) {
            if (degree[i] < degree[s]) s = i;
        }

        int bestCut = minDeg; // upper bound
        std::vector<bool> bestSide;

        // We need to try enough sinks. For correctness, we'd need to try all,
        // but the min cut must separate some neighbor of some min-degree node.
        // So we try all neighbors of the min-degree node + a few others.

        // Actually, for global min cut using max-flow:
        // Fix s. The global min cut either separates s from some t, or doesn't involve s.
        // If it doesn't involve s, then both sides have s... wait, every cut separates into two parts.
        // One side contains s. So global min cut = min over all t of maxflow(s,t).
        // This is correct but expensive: O(N) max-flow computations.

        // Optimization: Gomory-Hu tree. But building it is also O(N) max-flows.
        // Better: use the fact that we have an upper bound (minDeg).
        // We can stop early if we find a cut of value minDeg.

        // Hao-Orlin approach or use Stoer-Wagner sparse

        // Let's use the sparse Stoer-Wagner
        StoerWagnerSparse sw(N);
        for (auto& [u, v] : edges) {
            sw.addEdge(u, v);
        }

        auto [cutVal, partition] = sw.solve();

        if (cutVal == 0) {
            printf("0\n");
            return 0;
        }

        std::vector<bool> side(N, false);
        for (int x : partition) side[x] = true;

        std::vector<std::pair<int,int>> cutEdges;
        for (auto& [u, v] : edges) {
            if (side[u] != side[v]) {
                cutEdges.push_back({u, v});
            }
        }

        printf("%d\n", (int)cutEdges.size());
        for (auto& [u, v] : cutEdges) {
            printf("%d %d\n", u, v);
        }
    }

    return 0;
}
