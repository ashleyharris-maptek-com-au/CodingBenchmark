; Geometry shader: expand lines into tapered ribbons
; InputLines -> OutputTriangleStrip, 4 vertices

               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
               OpExecutionMode %main InputLines
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               OpDecorate %in_color Location 2
               OpDecorate %out_color Location 0

       ; Types
       %void = OpTypeVoid
   %void_fun = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
       %bool = OpTypeBool

     %int_0  = OpConstant %int 0
     %int_1  = OpConstant %int 1
    %uint_2  = OpConstant %uint 2

  %float_0   = OpConstant %float 0.0
  %float_1   = OpConstant %float 1.0
  %thick_s   = OpConstant %float 0.04
  %thick_e   = OpConstant %float 0.005

       ; Input gl_PerVertex block (array of 2 for InputLines)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in  = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr = OpTypePointer Input %arr_pv_in
%ptr_in_v4  = OpTypePointer Input %v4float

      %gl_in = OpVariable %ptr_in_arr Input

       ; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float

     %gl_out = OpVariable %ptr_out_pv Output

       ; Input color (location 2, array of 2)
%arr_v3_2   = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_2
%ptr_in_v3  = OpTypePointer Input %v3float
   %in_color = OpVariable %ptr_in_arr_v3 Input

       ; Output color (location 0)
%ptr_out_v3 = OpTypePointer Output %v3float
  %out_color = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %void_fun
      %entry = OpLabel

               ; Load p0 = gl_in[0].gl_Position
      %p0ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0ptr

               ; Load p1 = gl_in[1].gl_Position
      %p1ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %p1ptr

               ; Load c0 = in_color[0]
      %c0ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
         %c0 = OpLoad %v3float %c0ptr

               ; Load c1 = in_color[1]
      %c1ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
         %c1 = OpLoad %v3float %c1ptr

               ; Compute direction: dir = p1 - p0 (in clip space xy)
       %p0x  = OpCompositeExtract %float %p0 0
       %p0y  = OpCompositeExtract %float %p0 1
       %p1x  = OpCompositeExtract %float %p1 0
       %p1y  = OpCompositeExtract %float %p1 1

        %dx  = OpFSub %float %p1x %p0x
        %dy  = OpFSub %float %p1y %p0y

               ; Perpendicular = (-dy, dx), then normalize
      %negdy = OpFNegate %float %dy
               ; len = sqrt(dx*dx + dy*dy)
     %dx2    = OpFMul %float %dx %dx
     %dy2    = OpFMul %float %dy %dy
     %sum2   = OpFAdd %float %dx2 %dy2
      %len   = OpExtInst %float %1 Sqrt %sum2

               ; Avoid division by zero: if len == 0, use (0,1)
     %is_zero = OpFOrdEqual %bool %len %float_0
               OpSelectionMerge %merge None
               OpBranchConditional %is_zero %zero_blk %nonzero_blk

  %zero_blk  = OpLabel
               OpBranch %merge

%nonzero_blk = OpLabel
    %inv_len  = OpFDiv %float %float_1 %len
    %pnx_nz   = OpFMul %float %negdy %inv_len
    %pny_nz   = OpFMul %float %dx %inv_len
               OpBranch %merge

     %merge  = OpLabel
       %pnx  = OpPhi %float %float_0 %zero_blk %pnx_nz %nonzero_blk
       %pny  = OpPhi %float %float_1 %zero_blk %pny_nz %nonzero_blk

               ; Offset at start: perp * thick_start
    %ox0_pos = OpFMul %float %pnx %thick_s
    %oy0_pos = OpFMul %float %pny %thick_s
    %ox0_neg = OpFNegate %float %ox0_pos
    %oy0_neg = OpFNegate %float %oy0_pos

               ; Offset at end: perp * thick_end
    %ox1_pos = OpFMul %float %pnx %thick_e
    %oy1_pos = OpFMul %float %pny %thick_e
    %ox1_neg = OpFNegate %float %ox1_pos
    %oy1_neg = OpFNegate %float %oy1_pos

               ; Vertex 1: p0 + perp*thick_start
     %v1x    = OpFAdd %float %p0x %ox0_pos
     %v1y    = OpFAdd %float %p0y %oy0_pos
     %p0z    = OpCompositeExtract %float %p0 2
     %p0w    = OpCompositeExtract %float %p0 3
     %v1pos  = OpCompositeConstruct %v4float %v1x %v1y %p0z %p0w
     %optr1  = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %optr1 %v1pos
               OpStore %out_color %c0
               OpEmitVertex

               ; Vertex 2: p0 - perp*thick_start
     %v2x    = OpFAdd %float %p0x %ox0_neg
     %v2y    = OpFAdd %float %p0y %oy0_neg
     %v2pos  = OpCompositeConstruct %v4float %v2x %v2y %p0z %p0w
               OpStore %optr1 %v2pos
               OpStore %out_color %c0
               OpEmitVertex

               ; Vertex 3: p1 + perp*thick_end
     %v3x    = OpFAdd %float %p1x %ox1_pos
     %v3y    = OpFAdd %float %p1y %oy1_pos
     %p1z    = OpCompositeExtract %float %p1 2
     %p1w    = OpCompositeExtract %float %p1 3
     %v3pos  = OpCompositeConstruct %v4float %v3x %v3y %p1z %p1w
               OpStore %optr1 %v3pos
               OpStore %out_color %c1
               OpEmitVertex

               ; Vertex 4: p1 - perp*thick_end
     %v4x    = OpFAdd %float %p1x %ox1_neg
     %v4y    = OpFAdd %float %p1y %oy1_neg
     %v4pos  = OpCompositeConstruct %v4float %v4x %v4y %p1z %p1w
               OpStore %optr1 %v4pos
               OpStore %out_color %c1
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
