#include <cstdio>
#include <cmath>
#include <vector>
#include <array>
#include <algorithm>
#include <unordered_map>
#include <numeric>

using namespace std;

const double EPS = 1e-10;
const double MEPS = 1e-8;

struct V3 {
    double x,y,z;
    V3():x(0),y(0),z(0){}
    V3(double a,double b,double c):x(a),y(b),z(c){}
    V3 operator+(const V3&o)const{return{x+o.x,y+o.y,z+o.z};}
    V3 operator-(const V3&o)const{return{x-o.x,y-o.y,z-o.z};}
    V3 operator*(double s)const{return{x*s,y*s,z*s};}
    V3 operator/(double s)const{return{x/s,y/s,z/s};}
    double dot(const V3&o)const{return x*o.x+y*o.y+z*o.z;}
    V3 cross(const V3&o)const{return{y*o.z-z*o.y,z*o.x-x*o.z,x*o.y-y*o.x};}
    double len2()const{return x*x+y*y+z*z;}
    double len()const{return sqrt(len2());}
    V3 norm()const{double l=len();return l>0?*this/l:V3();}
};

struct AABB {
    V3 mn{1e30,1e30,1e30}, mx{-1e30,-1e30,-1e30};
    void add(const V3&p){
        mn.x=min(mn.x,p.x);mn.y=min(mn.y,p.y);mn.z=min(mn.z,p.z);
        mx.x=max(mx.x,p.x);mx.y=max(mx.y,p.y);mx.z=max(mx.z,p.z);
    }
    void add(const AABB&o){add(o.mn);add(o.mx);}
    bool hit(const AABB&o)const{
        return mn.x<=o.mx.x&&mx.x>=o.mn.x&&mn.y<=o.mx.y&&mx.y>=o.mn.y&&mn.z<=o.mx.z&&mx.z>=o.mn.z;
    }
    V3 center()const{return(mn+mx)*0.5;}
};

struct BVHNode {
    AABB box;
    int left=-1,right=-1;
    int start,count;
};

struct Mesh {
    vector<V3> v;
    vector<array<int,3>> f;
};

struct BVH {
    vector<BVHNode> nodes;
    vector<int> idx;
    const Mesh* mesh;

    AABB tbox(int fi)const{
        AABB b;auto&f=mesh->f[fi];
        b.add(mesh->v[f[0]]);b.add(mesh->v[f[1]]);b.add(mesh->v[f[2]]);
        return b;
    }

    int build(vector<int>&ids,int s,int e,int dep){
        int ni=(int)nodes.size();
        nodes.push_back({});
        nodes[ni].start=s;nodes[ni].count=e-s;
        AABB box;
        for(int i=s;i<e;i++)box.add(tbox(ids[i]));
        nodes[ni].box=box;
        if(e-s<=4)return ni;
        V3 ext=box.mx-box.mn;
        int ax=0;
        if(ext.y>ext.x)ax=1;
        if(ext.z>(ax==0?ext.x:ext.y))ax=2;
        int mid=(s+e)/2;
        nth_element(ids.begin()+s,ids.begin()+mid,ids.begin()+e,[&](int a,int b){
            auto ca=tbox(a).center(),cb=tbox(b).center();
            return(ax==0?ca.x:ax==1?ca.y:ca.z)<(ax==0?cb.x:ax==1?cb.y:cb.z);
        });
        int left=build(ids,s,mid,dep+1);
        int right=build(ids,mid,e,dep+1);
        nodes[ni].left=left;
        nodes[ni].right=right;
        nodes[ni].count=0;
        return ni;
    }

    void init(const Mesh&m){
        mesh=&m;
        idx.resize(m.f.size());
        iota(idx.begin(),idx.end(),0);
        if(!m.f.empty()){
            nodes.reserve(m.f.size()*2);
            build(idx,0,(int)idx.size(),0);
        }
    }

    void query(int ni,const AABB&q,vector<int>&res)const{
        if(ni<0||!nodes[ni].box.hit(q))return;
        auto&nd=nodes[ni];
        if(nd.count>0){
            for(int i=nd.start;i<nd.start+nd.count;i++)
                if(tbox(idx[i]).hit(q))res.push_back(idx[i]);
            return;
        }
        query(nd.left,q,res);
        query(nd.right,q,res);
    }

    bool rayHitAABB(const V3&o,const V3&id,const AABB&b)const{
        double tn=-1e30,tf=1e30;
        for(int a=0;a<3;a++){
            double oa=a==0?o.x:a==1?o.y:o.z;
            double ia=a==0?id.x:a==1?id.y:id.z;
            double bmin=a==0?b.mn.x:a==1?b.mn.y:b.mn.z;
            double bmax=a==0?b.mx.x:a==1?b.mx.y:b.mx.z;
            if(fabs(ia)<1e-30){
                if(oa<bmin||oa>bmax)return false;
            } else {
                double t1=(bmin-oa)*ia,t2=(bmax-oa)*ia;
                if(t1>t2){double t=t1;t1=t2;t2=t;}
                tn=max(tn,t1);tf=min(tf,t2);
                if(tn>tf)return false;
            }
        }
        return tf>=0;
    }

    bool rayTri(const V3&o,const V3&d,int fi,double&t)const{
        auto&f=mesh->f[fi];
        V3 v0=mesh->v[f[0]],v1=mesh->v[f[1]],v2=mesh->v[f[2]];
        V3 e1=v1-v0,e2=v2-v0;
        V3 h=d.cross(e2);
        double a=e1.dot(h);
        if(fabs(a)<EPS)return false;
        double ff=1.0/a;
        V3 s=o-v0;
        double u=ff*s.dot(h);
        if(u<-EPS||u>1+EPS)return false;
        V3 q=s.cross(e1);
        double v=ff*d.dot(q);
        if(v<-EPS||u+v>1+EPS)return false;
        t=ff*e2.dot(q);
        return true;
    }

    int countRay(int ni,const V3&o,const V3&d,const V3&id)const{
        if(ni<0)return 0;
        auto&nd=nodes[ni];
        if(!rayHitAABB(o,id,nd.box))return 0;
        int c=0;
        if(nd.count>0){
            for(int i=nd.start;i<nd.start+nd.count;i++){
                double t;
                if(rayTri(o,d,idx[i],t)&&t>EPS)c++;
            }
            return c;
        }
        return countRay(nd.left,o,d,id)+countRay(nd.right,o,d,id);
    }
};

bool isInside(const V3&p,const BVH&bvh){
    if(bvh.nodes.empty())return false;
    V3 dirs[3]={V3(1,.3171,.1413).norm(),V3(.1413,1,.3171).norm(),V3(.3171,.1413,1).norm()};
    int votes=0;
    for(int i=0;i<3;i++){
        V3 id(1.0/dirs[i].x,1.0/dirs[i].y,1.0/dirs[i].z);
        int c=bvh.countRay(0,p,dirs[i],id);
        if(c&1)votes++;
    }
    return votes>=2;
}

struct VMerge {
    vector<V3> verts;
    unordered_map<long long,vector<int>> grid;
    double cs;
    VMerge(double c=MEPS*100):cs(c){}
    long long key(double x,double y,double z)const{
        long long a=(long long)floor(x/cs),b=(long long)floor(y/cs),c=(long long)floor(z/cs);
        return a*1000000007LL+b*1000003LL+c;
    }
    int add(const V3&p){
        for(int dx=-1;dx<=1;dx++)for(int dy=-1;dy<=1;dy++)for(int dz=-1;dz<=1;dz++){
            long long k=key(p.x+dx*cs,p.y+dy*cs,p.z+dz*cs);
            auto it=grid.find(k);
            if(it!=grid.end())for(int vi:it->second)
                if((verts[vi]-p).len2()<MEPS*MEPS)return vi;
        }
        int i=(int)verts.size();
        verts.push_back(p);
        grid[key(p.x,p.y,p.z)].push_back(i);
        return i;
    }
};

bool segTri(const V3&p0,const V3&p1,const V3&v0,const V3&v1,const V3&v2,double&t,double&u,double&v){
    V3 d=p1-p0,e1=v1-v0,e2=v2-v0;
    V3 h=d.cross(e2);
    double a=e1.dot(h);
    if(fabs(a)<EPS)return false;
    double f=1.0/a;
    V3 s=p0-v0;
    u=f*s.dot(h);
    if(u<-EPS||u>1+EPS)return false;
    V3 q=s.cross(e1);
    v=f*d.dot(q);
    if(v<-EPS||u+v>1+EPS)return false;
    t=f*e2.dot(q);
    return t>-EPS&&t<1+EPS;
}

int triTriIsect(const V3&a0,const V3&a1,const V3&a2,
                const V3&b0,const V3&b1,const V3&b2,V3&ip0,V3&ip1){
    vector<V3> pts;
    V3 ae[3][2]={{a0,a1},{a1,a2},{a2,a0}};
    V3 be[3][2]={{b0,b1},{b1,b2},{b2,b0}};
    for(int i=0;i<3;i++){
        double t,u,v;
        if(segTri(ae[i][0],ae[i][1],b0,b1,b2,t,u,v)){
            t=max(0.0,min(1.0,t));
            pts.push_back(ae[i][0]+(ae[i][1]-ae[i][0])*t);
        }
    }
    for(int i=0;i<3;i++){
        double t,u,v;
        if(segTri(be[i][0],be[i][1],a0,a1,a2,t,u,v)){
            t=max(0.0,min(1.0,t));
            pts.push_back(be[i][0]+(be[i][1]-be[i][0])*t);
        }
    }
    vector<V3> uniq;
    for(auto&p:pts){
        bool dup=false;
        for(auto&q:uniq)if((p-q).len2()<MEPS*MEPS){dup=true;break;}
        if(!dup)uniq.push_back(p);
    }
    if(uniq.size()<2)return (int)uniq.size()?(ip0=uniq[0],1):0;
    ip0=uniq[0];ip1=uniq[1];
    return 2;
}

int domAxis(const V3&n){
    double a=fabs(n.x),b=fabs(n.y),c=fabs(n.z);
    if(a>=b&&a>=c)return 0;
    if(b>=a&&b>=c)return 1;
    return 2;
}
void proj2(const V3&p,int ax,double&u,double&v){
    if(ax==0){u=p.y;v=p.z;}
    else if(ax==1){u=p.x;v=p.z;}
    else{u=p.x;v=p.y;}
}
double cross2(double ax,double ay,double bx,double by){return ax*by-ay*bx;}

bool computeChord(const V3&P0,const V3&P1,const V3&V0,const V3&V1,const V3&V2,V3&C0,V3&C1){
    V3 D=P1-P0;
    if(D.len2()<EPS*EPS)return false;
    V3 nrm=(V1-V0).cross(V2-V0);
    int ax=domAxis(nrm);
    double du,dv;proj2(D,ax,du,dv);
    V3 edges[3][2]={{V0,V1},{V1,V2},{V2,V0}};
    struct Hit{double t;V3 pt;};
    vector<Hit> hits;
    double p0u,p0v;proj2(P0,ax,p0u,p0v);
    for(int i=0;i<3;i++){
        double e0u,e0v,e1u,e1v;
        proj2(edges[i][0],ax,e0u,e0v);
        proj2(edges[i][1],ax,e1u,e1v);
        double deu=e1u-e0u,dev=e1v-e0v;
        double det=du*dev-dv*deu;
        if(fabs(det)<EPS)continue;
        double s=(du*(e0v-p0v)-dv*(e0u-p0u))/det;
        if(s<-MEPS||s>1+MEPS)continue;
        s=max(0.0,min(1.0,s));
        double t_val=(deu*(e0v-p0v)-dev*(e0u-p0u))/det;
        V3 hp=edges[i][0]+(edges[i][1]-edges[i][0])*s;
        hits.push_back({t_val,hp});
    }
    vector<Hit> uh;
    for(auto&h:hits){
        bool d=false;
        for(auto&u:uh)if((h.pt-u.pt).len2()<MEPS*MEPS){d=true;break;}
        if(!d)uh.push_back(h);
    }
    if(uh.size()<2)return false;
    sort(uh.begin(),uh.end(),[](const Hit&a,const Hit&b){return a.t<b.t;});
    C0=uh.front().pt;C1=uh.back().pt;
    return(C0-C1).len2()>MEPS*MEPS;
}

struct SubTri{int v[3];};

bool clipChordToTri(const V3&C0,const V3&C1,const V3&A,const V3&B,const V3&C,int ax,
                    double&tMin,double&tMax){
    tMin=0;tMax=1;
    double c0u,c0v,c1u,c1v;
    proj2(C0,ax,c0u,c0v);proj2(C1,ax,c1u,c1v);
    double du=c1u-c0u,dv=c1v-c0v;

    double au,av,bu,bv,cu,cv;
    proj2(A,ax,au,av);proj2(B,ax,bu,bv);proj2(C,ax,cu,cv);
    double sa=cross2(bu-au,bv-av,cu-au,cv-av);
    double flip=sa>0?1.0:-1.0;

    V3 triV[3]={A,B,C};
    for(int i=0;i<3;i++){
        int j=(i+1)%3;
        double e0u,e0v,e1u,e1v;
        proj2(triV[i],ax,e0u,e0v);proj2(triV[j],ax,e1u,e1v);
        double eu=e1u-e0u,ev=e1v-e0v;
        double nx=-ev*flip,ny=eu*flip;
        double num=nx*(e0u-c0u)+ny*(e0v-c0v);
        double den=nx*du+ny*dv;
        if(fabs(den)<EPS){
            if(num<-MEPS)return false;
        } else {
            double t=num/den;
            if(den>0)tMin=max(tMin,t);
            else tMax=min(tMax,t);
        }
    }
    return tMax-tMin>-MEPS;
}

void splitTriBySeg(const SubTri&tri,const V3*verts,int ax,
                   const V3&C0,const V3&C1,VMerge&vm,vector<SubTri>&out){
    V3 A=verts[tri.v[0]],B=verts[tri.v[1]],CC=verts[tri.v[2]];
    double tMin,tMax;
    if(!clipChordToTri(C0,C1,A,B,CC,ax,tMin,tMax)||tMax-tMin<EPS){
        out.push_back(tri);return;
    }
    V3 cp0=C0+(C1-C0)*max(0.0,tMin);
    V3 cp1=C0+(C1-C0)*min(1.0,tMax);
    if((cp0-cp1).len2()<MEPS*MEPS){out.push_back(tri);return;}

    int pi0=vm.add(cp0),pi1=vm.add(cp1);
    if(pi0==pi1){out.push_back(tri);return;}

    int tv[3]={tri.v[0],tri.v[1],tri.v[2]};

    // Check if pi0/pi1 coincide with existing triangle vertices
    int loc0=-1,loc1=-1;
    for(int i=0;i<3;i++){
        if(pi0==tv[i])loc0=i;
        if(pi1==tv[i])loc1=i;
    }

    // Find which edge each point is on (if not at vertex)
    auto findEdge=[&](const V3&p)->int{
        V3 triVerts[3]={A,B,CC};
        for(int i=0;i<3;i++){
            V3 e0=triVerts[i],e1=triVerts[(i+1)%3];
            V3 d=e1-e0;
            double l2=d.len2();
            if(l2<EPS*EPS)continue;
            double t=(p-e0).dot(d)/l2;
            if(t<-MEPS||t>1+MEPS)continue;
            t=max(0.0,min(1.0,t));
            V3 proj=e0+d*t;
            if((p-proj).len2()<MEPS*MEPS)return i;
        }
        return -1;
    };

    int edge0=(loc0>=0)?-1:findEdge(cp0);
    int edge1=(loc1>=0)?-1:findEdge(cp1);

    // Compute original orientation for consistent sub-triangle winding
    double au,av,bu,bv,cu,cv;
    proj2(A,ax,au,av);proj2(B,ax,bu,bv);proj2(CC,ax,cu,cv);
    double origOri=cross2(bu-au,bv-av,cu-au,cv-av);

    auto addTri=[&](int a,int b,int c){
        if(a==b||b==c||a==c)return;
        double a2u,a2v,b2u,b2v,c2u,c2v;
        proj2(vm.verts[a],ax,a2u,a2v);proj2(vm.verts[b],ax,b2u,b2v);proj2(vm.verts[c],ax,c2u,c2v);
        double o=cross2(b2u-a2u,b2v-a2v,c2u-a2u,c2v-a2v);
        if(fabs(o)<EPS*EPS)return;
        if(o*origOri<0)out.push_back({{a,c,b}});
        else out.push_back({{a,b,c}});
    };

    // Case VV: both at vertices -> no split
    if(loc0>=0&&loc1>=0){
        out.push_back(tri);return;
    }

    // Case VE: one at vertex, one on edge -> 2 sub-triangles
    if(loc0>=0&&edge1>=0){
        int ea=edge1,eb=(edge1+1)%3;
        int third=-1;
        for(int i=0;i<3;i++)if(i!=ea&&i!=eb)third=i;
        if(third<0){out.push_back(tri);return;}
        addTri(tv[third],tv[ea],pi1);
        addTri(tv[third],pi1,tv[eb]);
        return;
    }
    if(loc1>=0&&edge0>=0){
        int ea=edge0,eb=(edge0+1)%3;
        int third=-1;
        for(int i=0;i<3;i++)if(i!=ea&&i!=eb)third=i;
        if(third<0){out.push_back(tri);return;}
        addTri(tv[third],tv[ea],pi0);
        addTri(tv[third],pi0,tv[eb]);
        return;
    }

    // Case EE: both on different edges -> 3 sub-triangles
    if(edge0>=0&&edge1>=0){
        if(edge0==edge1){out.push_back(tri);return;}

        int e0a=edge0,e0b=(edge0+1)%3;
        int e1a=edge1,e1b=(edge1+1)%3;
        int shared=-1;
        if(e0a==e1a||e0a==e1b)shared=e0a;
        else if(e0b==e1a||e0b==e1b)shared=e0b;
        if(shared<0){out.push_back(tri);return;}

        int va_=-1,vb_=-1;
        for(int i=0;i<3;i++){
            if(i==shared)continue;
            if(i==e0a||i==e0b)va_=i;
            else vb_=i;
        }
        if(va_<0||vb_<0){out.push_back(tri);return;}

        addTri(tv[shared],pi0,pi1);
        addTri(pi0,tv[va_],tv[vb_]);
        addTri(pi0,tv[vb_],pi1);
        return;
    }

    // Fallback: couldn't classify endpoints properly, keep original
    out.push_back(tri);
}

int main(){
    int nva,nfa,nvb,nfb;
    scanf("%d%d%d%d",&nva,&nfa,&nvb,&nfb);

    Mesh A,B;
    A.v.resize(nva);A.f.resize(nfa);
    B.v.resize(nvb);B.f.resize(nfb);
    for(int i=0;i<nva;i++)scanf("%lf%lf%lf",&A.v[i].x,&A.v[i].y,&A.v[i].z);
    for(int i=0;i<nfa;i++)scanf("%d%d%d",&A.f[i][0],&A.f[i][1],&A.f[i][2]);
    for(int i=0;i<nvb;i++)scanf("%lf%lf%lf",&B.v[i].x,&B.v[i].y,&B.v[i].z);
    for(int i=0;i<nfb;i++)scanf("%d%d%d",&B.f[i][0],&B.f[i][1],&B.f[i][2]);

    BVH bvhA,bvhB;
    bvhA.init(A);bvhB.init(B);

    AABB bA,bB;
    for(auto&v:A.v)bA.add(v);
    for(auto&v:B.v)bB.add(v);

    if(!bA.hit(bB)){
        printf("%d %d\n",nva+nvb,nfa+nfb);
        for(auto&v:A.v)printf("%.17g %.17g %.17g\n",v.x,v.y,v.z);
        for(auto&v:B.v)printf("%.17g %.17g %.17g\n",v.x,v.y,v.z);
        for(auto&f:A.f)printf("%d %d %d\n",f[0],f[1],f[2]);
        for(auto&f:B.f)printf("%d %d %d\n",f[0]+nva,f[1]+nva,f[2]+nva);
        return 0;
    }

    VMerge vm;
    vector<int> mapAv(nva),mapBv(nvb);
    for(int i=0;i<nva;i++)mapAv[i]=vm.add(A.v[i]);
    for(int i=0;i<nvb;i++)mapBv[i]=vm.add(B.v[i]);

    vector<vector<pair<V3,V3>>> chordsA(nfa),chordsB(nfb);

    for(int fi=0;fi<nfa;fi++){
        AABB ab;
        ab.add(A.v[A.f[fi][0]]);ab.add(A.v[A.f[fi][1]]);ab.add(A.v[A.f[fi][2]]);
        vector<int> cands;
        if(!bvhB.nodes.empty())bvhB.query(0,ab,cands);
        for(int fj:cands){
            V3 ip0,ip1;
            int n=triTriIsect(A.v[A.f[fi][0]],A.v[A.f[fi][1]],A.v[A.f[fi][2]],
                              B.v[B.f[fj][0]],B.v[B.f[fj][1]],B.v[B.f[fj][2]],ip0,ip1);
            if(n==2){
                V3 ca0,ca1;
                if(computeChord(ip0,ip1,A.v[A.f[fi][0]],A.v[A.f[fi][1]],A.v[A.f[fi][2]],ca0,ca1))
                    chordsA[fi].push_back({ca0,ca1});
                V3 cb0,cb1;
                if(computeChord(ip0,ip1,B.v[B.f[fj][0]],B.v[B.f[fj][1]],B.v[B.f[fj][2]],cb0,cb1))
                    chordsB[fj].push_back({cb0,cb1});
            }
        }
    }

    bool anyIsect=false;
    for(int i=0;i<nfa&&!anyIsect;i++)if(!chordsA[i].empty())anyIsect=true;
    for(int i=0;i<nfb&&!anyIsect;i++)if(!chordsB[i].empty())anyIsect=true;

    if(!anyIsect){
        bool aInB=isInside(A.v[0],bvhB);
        bool bInA=isInside(B.v[0],bvhA);
        if(aInB){
            printf("%d %d\n",nvb,nfb);
            for(auto&v:B.v)printf("%.17g %.17g %.17g\n",v.x,v.y,v.z);
            for(auto&f:B.f)printf("%d %d %d\n",f[0],f[1],f[2]);
        } else if(bInA){
            printf("%d %d\n",nva,nfa);
            for(auto&v:A.v)printf("%.17g %.17g %.17g\n",v.x,v.y,v.z);
            for(auto&f:A.f)printf("%d %d %d\n",f[0],f[1],f[2]);
        } else {
            printf("%d %d\n",nva+nvb,nfa+nfb);
            for(auto&v:A.v)printf("%.17g %.17g %.17g\n",v.x,v.y,v.z);
            for(auto&v:B.v)printf("%.17g %.17g %.17g\n",v.x,v.y,v.z);
            for(auto&f:A.f)printf("%d %d %d\n",f[0],f[1],f[2]);
            for(auto&f:B.f)printf("%d %d %d\n",f[0]+nva,f[1]+nva,f[2]+nva);
        }
        return 0;
    }

    auto processMesh=[&](Mesh&M,vector<int>&mapV,vector<vector<pair<V3,V3>>>&chords,
                         int nf,const BVH&otherBvh,vector<array<int,3>>&result){
        for(int fi=0;fi<nf;fi++){
            auto&f=M.f[fi];
            int v0=mapV[f[0]],v1=mapV[f[1]],v2=mapV[f[2]];

            if(chords[fi].empty()){
                V3 cent=(M.v[f[0]]+M.v[f[1]]+M.v[f[2]])/3.0;
                if(!isInside(cent,otherBvh))
                    result.push_back({v0,v1,v2});
                continue;
            }

            V3 nrm=(M.v[f[1]]-M.v[f[0]]).cross(M.v[f[2]]-M.v[f[0]]);
            int ax=domAxis(nrm);

            vector<SubTri> subs={{{v0,v1,v2}}};

            // Deduplicate chords
            vector<pair<V3,V3>> uc;
            for(auto&c:chords[fi]){
                bool dup=false;
                for(auto&u:uc){
                    if(((c.first-u.first).len2()<MEPS*MEPS&&(c.second-u.second).len2()<MEPS*MEPS)||
                       ((c.first-u.second).len2()<MEPS*MEPS&&(c.second-u.first).len2()<MEPS*MEPS)){
                        dup=true;break;
                    }
                }
                if(!dup)uc.push_back(c);
            }

            for(auto&[c0,c1]:uc){
                vector<SubTri> next;
                for(auto&st:subs)
                    splitTriBySeg(st,vm.verts.data(),ax,c0,c1,vm,next);
                subs=next;
            }

            for(auto&st:subs){
                V3 cent=(vm.verts[st.v[0]]+vm.verts[st.v[1]]+vm.verts[st.v[2]])/3.0;
                V3 e1=vm.verts[st.v[1]]-vm.verts[st.v[0]];
                V3 e2=vm.verts[st.v[2]]-vm.verts[st.v[0]];
                if(e1.cross(e2).len2()<EPS*EPS)continue;
                if(!isInside(cent,otherBvh))
                    result.push_back({st.v[0],st.v[1],st.v[2]});
            }
        }
    };

    vector<array<int,3>> result;
    processMesh(A,mapAv,chordsA,nfa,bvhB,result);
    processMesh(B,mapBv,chordsB,nfb,bvhA,result);

    // Compact vertices
    vector<int> remap(vm.verts.size(),-1);
    int nxt=0;
    for(auto&f:result)for(int i=0;i<3;i++){
        if(remap[f[i]]<0)remap[f[i]]=nxt++;
        f[i]=remap[f[i]];
    }
    vector<V3> ov(nxt);
    for(int i=0;i<(int)vm.verts.size();i++)
        if(remap[i]>=0)ov[remap[i]]=vm.verts[i];

    printf("%d %d\n",nxt,(int)result.size());
    for(auto&v:ov)printf("%.17g %.17g %.17g\n",v.x,v.y,v.z);
    for(auto&f:result)printf("%d %d %d\n",f[0],f[1],f[2]);
    return 0;
}
