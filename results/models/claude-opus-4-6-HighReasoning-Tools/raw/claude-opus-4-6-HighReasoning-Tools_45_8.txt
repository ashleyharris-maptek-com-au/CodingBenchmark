import struct

words = []

def w(val):
    """Append a 32-bit word."""
    words.append(val & 0xFFFFFFFF)

def op(wordcount, opcode):
    """Encode instruction first word."""
    w((wordcount << 16) | opcode)

def str_words(s):
    """Encode a null-terminated string as 32-bit words."""
    b = s.encode('utf-8') + b'\x00'
    while len(b) % 4 != 0:
        b += b'\x00'
    result = []
    for i in range(0, len(b), 4):
        result.append(struct.unpack('<I', b[i:i+4])[0])
    return result

def float_bits(f):
    return struct.unpack('<I', struct.pack('<f', f))[0]

# --- Header ---
w(0x07230203)  # Magic
w(0x00010000)  # Version 1.0
w(0)           # Generator
w(BOUND)       # Bound
w(0)           # Reserved

# --- Capabilities ---
op(2, 17); w(1)  # OpCapability Shader

# --- Ext import ---
name_w = str_words("GLSL.std.450")
op(2 + len(name_w), 11); w(EXT)
for nw in name_w:
    w(nw)

# --- Memory model ---
op(3, 14); w(0); w(1)  # Logical, GLSL450

# --- Entry point ---
name_main = str_words("main")
interface_vars = [WORLD_POS, NORMAL, COLOR, OUT_COLOR]
wc = 3 + len(name_main) + len(interface_vars)
op(wc, 15); w(4); w(MAIN)  # Fragment
for nw in name_main:
    w(nw)
for iv in interface_vars:
    w(iv)

# --- Execution mode ---
op(3, 16); w(MAIN); w(7)  # OriginUpperLeft

# --- Decorations ---
# OpDecorate = opcode 71
# Location decorations
op(4, 71); w(WORLD_POS); w(30); w(0)   # Location 0
op(4, 71); w(NORMAL); w(30); w(1)      # Location 1
op(4, 71); w(COLOR); w(30); w(4)       # Location 4
op(4, 71); w(OUT_COLOR); w(30); w(0)   # Location 0

# UBO decorations
op(3, 71); w(UBO_TYPE); w(2)           # Block
op(4, 71); w(UBO); w(34); w(0)         # DescriptorSet 0
op(4, 71); w(UBO); w(33); w(0)         # Binding 0

# OpMemberDecorate = opcode 72
# Member offsets
op(5, 72); w(UBO_TYPE); w(0); w(35); w(0)    # Offset 0
op(5, 72); w(UBO_TYPE); w(1); w(35); w(64)   # Offset 64
op(5, 72); w(UBO_TYPE); w(2); w(35); w(128)  # Offset 128
op(5, 72); w(UBO_TYPE); w(3); w(35); w(192)  # Offset 192
op(5, 72); w(UBO_TYPE); w(4); w(35); w(208)  # Offset 208
op(5, 72); w(UBO_TYPE); w(5); w(35); w(224)  # Offset 224

# ColMajor (decoration=5) and MatrixStride (decoration=7) for mat4 members
op(4, 72); w(UBO_TYPE); w(0); w(5)           # ColMajor
op(5, 72); w(UBO_TYPE); w(0); w(7); w(16)    # MatrixStride 16
op(4, 72); w(UBO_TYPE); w(1); w(5)           # ColMajor
op(5, 72); w(UBO_TYPE); w(1); w(7); w(16)    # MatrixStride 16
op(4, 72); w(UBO_TYPE); w(2); w(5)           # ColMajor
op(5, 72); w(UBO_TYPE); w(2); w(7); w(16)    # MatrixStride 16

# --- Types ---
op(2, 19); w(VOID)                     # OpTypeVoid
op(3, 22); w(FLOAT); w(32)             # OpTypeFloat 32
op(4, 23); w(VEC3); w(FLOAT); w(3)     # OpTypeVector float 3
op(4, 23); w(VEC4); w(FLOAT); w(4)     # OpTypeVector float 4
op(4, 24); w(MAT4); w(VEC4); w(4)      # OpTypeMatrix vec4 4  (opcode 24)
op(2, 20); w(BOOL)                     # OpTypeBool
op(4, 21); w(INT); w(32); w(1)         # OpTypeInt 32 signed
op(3, 33); w(FUNC_VOID); w(VOID)       # OpTypeFunction void

# Pointer types: OpTypePointer = 32, storage classes: Input=1, Output=3, Uniform=2
op(4, 32); w(PTR_IN_VEC3); w(1); w(VEC3)
op(4, 32); w(PTR_OUT_VEC4); w(3); w(VEC4)
op(4, 32); w(PTR_UNI_VEC4); w(2); w(VEC4)

# UBO struct type
op(8, 30); w(UBO_TYPE); w(MAT4); w(MAT4); w(MAT4); w(VEC4); w(VEC4); w(VEC4)  # OpTypeStruct
op(4, 32); w(PTR_UNI_UBO); w(2); w(UBO_TYPE)

# --- Variables ---
# OpVariable = 59, storage class as operand
op(4, 59); w(PTR_IN_VEC3); w(WORLD_POS); w(1)   # Input
op(4, 59); w(PTR_IN_VEC3); w(NORMAL); w(1)       # Input
op(4, 59); w(PTR_IN_VEC3); w(COLOR); w(1)        # Input
op(4, 59); w(PTR_OUT_VEC4); w(OUT_COLOR); w(3)   # Output
op(4, 59); w(PTR_UNI_UBO); w(UBO); w(2)          # Uniform

# --- Constants ---
op(4, 43); w(FLOAT); w(F_0); w(float_bits(0.0))
op(4, 43); w(FLOAT); w(F_075); w(float_bits(0.75))
op(4, 43); w(FLOAT); w(F_05); w(float_bits(0.5))
op(4, 43); w(FLOAT); w(F_025); w(float_bits(0.25))
op(4, 43); w(FLOAT); w(F_1); w(float_bits(1.0))
op(4, 43); w(FLOAT); w(F_07); w(float_bits(0.7))
op(4, 43); w(FLOAT); w(F_04); w(float_bits(0.4))
op(4, 43); w(FLOAT); w(F_02); w(float_bits(0.2))
op(4, 43); w(INT); w(INT_3); w(3)

# --- Function ---
op(5, 54); w(VOID); w(MAIN); w(0); w(FUNC_VOID)  # OpFunction None
op(2, 248); w(LABEL)                               # OpLabel

# Load inputs
op(4, 61); w(VEC3); w(N_RAW); w(NORMAL)    # OpLoad
op(4, 61); w(VEC3); w(WP); w(WORLD_POS)    # OpLoad
op(4, 61); w(VEC3); w(COL); w(COLOR)       # OpLoad

# Normalize normal: ExtInst 69 = Normalize
op(6, 12); w(VEC3); w(N_NORM); w(EXT); w(69); w(N_RAW)

# Access lightPos from UBO: OpAccessChain
op(5, 65); w(PTR_UNI_VEC4); w(LP_PTR); w(UBO); w(INT_3)  # OpAccessChain
op(4, 61); w(VEC4); w(LP); w(LP_PTR)                       # OpLoad

# Extract xyz from lightPos
op(5, 81); w(FLOAT); w(LP_X); w(LP); w(0)  # CompositeExtract
op(5, 81); w(FLOAT); w(LP_Y); w(LP); w(1)
op(5, 81); w(FLOAT); w(LP_Z); w(LP); w(2)
op(6, 80); w(VEC3); w(LP_VEC3); w(LP_X); w(LP_Y); w(LP_Z)  # CompositeConstruct

# L = normalize(lightPos.xyz - worldPos)
op(5, 131); w(VEC3); w(DIFF); w(LP_VEC3); w(WP)  # OpFSub
op(6, 12); w(VEC3); w(L); w(EXT); w(69); w(DIFF)  # Normalize

# dot(N, L)
op(5, 148); w(FLOAT); w(DOT_NL); w(N_NORM); w(L)  # OpDot

# d = max(dot, 0.0) - ExtInst 40 = FMax
op(7, 12); w(FLOAT); w(D); w(EXT); w(40); w(DOT_NL); w(F_0)

# Comparisons: OpFOrdGreaterThanEqual = 190
op(5, 190); w(BOOL); w(CMP1); w(D); w(F_075)
op(5, 190); w(BOOL); w(CMP2); w(D); w(F_05)
op(5, 190); w(BOOL); w(CMP3); w(D); w(F_025)

# Selects: OpSelect = 169
op(6, 169); w(FLOAT); w(SEL1); w(CMP3); w(F_04); w(F_02)
op(6, 169); w(FLOAT); w(SEL2); w(CMP2); w(F_07); w(SEL1)
op(6, 169); w(FLOAT); w(BAND); w(CMP1); w(F_1); w(SEL2)

# band_vec = vec3(band, band, band)
op(6, 80); w(VEC3); w(BAND_VEC); w(BAND); w(BAND); w(BAND)

# result = color * band_vec
op(5, 133); w(VEC3); w(RESULT); w(COL); w(BAND_VEC)  # OpFMul

# Extract and construct vec4
op(5, 81); w(FLOAT); w(R); w(RESULT); w(0)
op(5, 81); w(FLOAT); w(G); w(RESULT); w(1)
op(5, 81); w(FLOAT); w(B); w(RESULT); w(2)
op(7, 80); w(VEC4); w(OUT); w(R); w(G); w(B); w(F_1)

# Store and return
op(3, 62); w(OUT_COLOR); w(OUT)  # OpStore
op(1, 253)                        # OpReturn
op(1, 56)                         # OpFunctionEnd

# Convert to bytes (little-endian)
data = b''
for word in words:
    data += struct.pack('<I', word)

hex_str = data.hex()
print(hex_str)

# Also write to file for validation
with open('shader.spv', 'wb') as f:
    f.write(data)
