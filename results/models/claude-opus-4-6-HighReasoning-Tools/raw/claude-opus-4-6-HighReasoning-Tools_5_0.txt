use std::io::{self, Read, Write, BufWriter};
use std::collections::VecDeque;
use std::time::{Instant, Duration};

const DX: [i32; 4] = [1, -1, 0, 0];
const DY: [i32; 4] = [0, 0, 1, -1];

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let out = io::stdout();
    let mut out = BufWriter::new(out.lock());

    let mut it = input.split_whitespace();
    let w: usize = p(&mut it);
    let h: usize = p(&mut it);
    let nobs: usize = p(&mut it);
    let cyc: bool = p::<u8>(&mut it) != 0;

    if w == 0 || h == 0 {
        writeln!(out, "0").unwrap();
        return;
    }

    let n = w * h;
    let mut blocked = vec![false; n];
    for _ in 0..nobs {
        let x: usize = p(&mut it);
        let y: usize = p(&mut it);
        blocked[y * w + x] = true;
    }

    if blocked[0] {
        writeln!(out, "0").unwrap();
        return;
    }

    let total = n - nobs;

    if total <= 1 {
        if total == 1 && !cyc {
            writeln!(out, "1").unwrap();
            writeln!(out, "0 0").unwrap();
        } else {
            writeln!(out, "0").unwrap();
        }
        return;
    }

    if !connected(&blocked, w, h, total) {
        writeln!(out, "0").unwrap();
        return;
    }

    let (ec, oc) = parity(&blocked, w, h);
    if cyc {
        if total % 2 != 0 || ec != oc {
            writeln!(out, "0").unwrap();
            return;
        }
    } else if ec != (total + 1) / 2 || oc != total / 2 {
        writeln!(out, "0").unwrap();
        return;
    }

    if nobs == 0 {
        if !cyc {
            emit(&mut out, &snake(w, h));
        } else if let Some(c) = make_cycle(w, h) {
            emit(&mut out, &c);
        } else {
            writeln!(out, "0").unwrap();
        }
        return;
    }

    let t0 = Instant::now();
    let lim = Duration::from_secs(270);

    if total <= 100 {
        if let Some(path) = bt_solve(w, h, &blocked, total, cyc, t0, lim) {
            emit(&mut out, &path);
            return;
        }
        if t0.elapsed() < lim {
            writeln!(out, "0").unwrap();
            return;
        }
    }

    if let Some(path) = warnsdorff_multi(w, h, &blocked, total, cyc, t0, lim) {
        emit(&mut out, &path);
        return;
    }

    if total > 100 && total <= 300 && t0.elapsed() < Duration::from_secs(250) {
        if let Some(path) = bt_solve(w, h, &blocked, total, cyc, t0, Duration::from_secs(285)) {
            emit(&mut out, &path);
            return;
        }
    }

    writeln!(out, "0").unwrap();
}

fn p<T: std::str::FromStr>(it: &mut std::str::SplitWhitespace) -> T
where T::Err: std::fmt::Debug {
    it.next().unwrap().parse().unwrap()
}

fn connected(blocked: &[bool], w: usize, h: usize, total: usize) -> bool {
    let mut seen = vec![false; w * h];
    seen[0] = true;
    let mut q = VecDeque::new();
    q.push_back(0usize);
    let mut cnt = 1usize;
    while let Some(pos) = q.pop_front() {
        let (x, y) = (pos % w, pos / w);
        for d in 0..4 {
            let nx = x as i32 + DX[d];
            let ny = y as i32 + DY[d];
            if nx >= 0 && nx < w as i32 && ny >= 0 && ny < h as i32 {
                let np = ny as usize * w + nx as usize;
                if !blocked[np] && !seen[np] {
                    seen[np] = true;
                    cnt += 1;
                    q.push_back(np);
                }
            }
        }
    }
    cnt == total
}

fn parity(blocked: &[bool], w: usize, h: usize) -> (usize, usize) {
    let (mut e, mut o) = (0, 0);
    for y in 0..h {
        for x in 0..w {
            if !blocked[y * w + x] {
                if (x + y) % 2 == 0 { e += 1; } else { o += 1; }
            }
        }
    }
    (e, o)
}

fn snake(w: usize, h: usize) -> Vec<(usize, usize)> {
    let mut v = Vec::with_capacity(w * h);
    for y in 0..h {
        if y % 2 == 0 {
            for x in 0..w { v.push((x, y)); }
        } else {
            for x in (0..w).rev() { v.push((x, y)); }
        }
    }
    v
}

fn make_cycle(w: usize, h: usize) -> Option<Vec<(usize, usize)>> {
    let n = w * h;
    if n % 2 != 0 { return None; }
    if n < 2 { return None; }
    if n == 2 { return Some(snake(w, h)); }
    if w < 2 || h < 2 { return None; }

    let mut v = Vec::with_capacity(n);
    if w % 2 == 0 {
        for y in 0..h { v.push((0, y)); }
        for x in 1..w { v.push((x, h - 1)); }
        let mut up = true;
        for c in (1..w).rev() {
            if up {
                for y in (0..h - 1).rev() { v.push((c, y)); }
            } else {
                for y in 0..h - 1 { v.push((c, y)); }
            }
            up = !up;
        }
    } else {
        for x in 0..w { v.push((x, 0)); }
        for y in 1..h { v.push((w - 1, y)); }
        let mut left = true;
        for r in (1..h).rev() {
            if left {
                for x in (0..w - 1).rev() { v.push((x, r)); }
            } else {
                for x in 0..w - 1 { v.push((x, r)); }
            }
            left = !left;
        }
    }
    Some(v)
}

fn emit(out: &mut impl Write, path: &[(usize, usize)]) {
    writeln!(out, "{}", path.len()).unwrap();
    for &(x, y) in path {
        writeln!(out, "{} {}", x, y).unwrap();
    }
}

fn bt_solve(w: usize, h: usize, blocked: &[bool], total: usize, cyc: bool,
            t0: Instant, lim: Duration) -> Option<Vec<(usize, usize)>> {
    let mut vis = blocked.to_vec();
    vis[0] = true;
    let mut path = vec![(0usize, 0usize)];
    let mut ck = 0u64;
    let mut scratch = vec![0u32; w * h];
    let mut gen = 0u32;
    let mut stk_buf = Vec::with_capacity(w * h);
    if bt_rec(w, h, &mut vis, &mut path, total, cyc, t0, lim, &mut ck,
              &mut scratch, &mut gen, &mut stk_buf) {
        Some(path)
    } else {
        None
    }
}

fn bt_rec(w: usize, h: usize, vis: &mut Vec<bool>, path: &mut Vec<(usize, usize)>,
          total: usize, cyc: bool, t0: Instant, lim: Duration, ck: &mut u64,
          scratch: &mut Vec<u32>, gen: &mut u32, stk_buf: &mut Vec<usize>) -> bool {
    *ck += 1;
    if *ck & 0xFFFF == 0 && t0.elapsed() > lim { return false; }

    let (cx, cy) = *path.last().unwrap();
    let rem = total - path.len();

    if rem == 0 {
        return !cyc || cx + cy == 1;
    }

    let mut nbrs = [(0usize, 0usize, 0usize); 4];
    let mut nc = 0;
    for d in 0..4 {
        let nx = cx as i32 + DX[d];
        let ny = cy as i32 + DY[d];
        if nx >= 0 && nx < w as i32 && ny >= 0 && ny < h as i32 {
            let (ux, uy) = (nx as usize, ny as usize);
            if !vis[uy * w + ux] {
                let deg = udeg(vis, w, h, ux, uy);
                nbrs[nc] = (ux, uy, deg);
                nc += 1;
            }
        }
    }

    if nc == 0 { return false; }

    nbrs[..nc].sort_unstable_by_key(|t| t.2);

    for i in 0..nc {
        let (nx, ny, _) = nbrs[i];
        let np = ny * w + nx;
        vis[np] = true;
        path.push((nx, ny));

        let ok = prune_ok(vis, w, h, nx, ny, rem - 1, cyc, scratch, gen, stk_buf);

        if ok && bt_rec(w, h, vis, path, total, cyc, t0, lim, ck, scratch, gen, stk_buf) {
            return true;
        }

        path.pop();
        vis[np] = false;
    }

    false
}

fn udeg(vis: &[bool], w: usize, h: usize, x: usize, y: usize) -> usize {
    let mut d = 0;
    for i in 0..4 {
        let nx = x as i32 + DX[i];
        let ny = y as i32 + DY[i];
        if nx >= 0 && nx < w as i32 && ny >= 0 && ny < h as i32 {
            if !vis[ny as usize * w + nx as usize] { d += 1; }
        }
    }
    d
}

fn prune_ok(vis: &[bool], w: usize, h: usize, cx: usize, cy: usize, rem: usize, cyc: bool,
            scratch: &mut Vec<u32>, gen: &mut u32, stk_buf: &mut Vec<usize>) -> bool {
    if rem == 0 { return true; }

    for d in 0..4 {
        let nx = cx as i32 + DX[d];
        let ny = cy as i32 + DY[d];
        if nx >= 0 && nx < w as i32 && ny >= 0 && ny < h as i32 {
            let np = ny as usize * w + nx as usize;
            if !vis[np] && udeg(vis, w, h, nx as usize, ny as usize) == 0 && rem > 1 {
                return false;
            }
        }
    }

    if cyc {
        let mut adj = 0;
        if w > 1 && !vis[1] { adj += 1; }
        if h > 1 && !vis[w] { adj += 1; }
        if adj == 0 { return false; }
    }

    if rem <= 80 {
        return conn_from(vis, w, h, cx, cy, rem, scratch, gen, stk_buf);
    }

    true
}

fn conn_from(vis: &[bool], w: usize, h: usize, sx: usize, sy: usize, expected: usize,
             scratch: &mut Vec<u32>, gen: &mut u32, stk: &mut Vec<usize>) -> bool {
    if expected == 0 { return true; }
    *gen = gen.wrapping_add(1);
    if *gen == 0 {
        scratch.iter_mut().for_each(|v| *v = 0);
        *gen = 1;
    }
    let g = *gen;

    let mut start = usize::MAX;
    for d in 0..4 {
        let nx = sx as i32 + DX[d];
        let ny = sy as i32 + DY[d];
        if nx >= 0 && nx < w as i32 && ny >= 0 && ny < h as i32 {
            let np = ny as usize * w + nx as usize;
            if !vis[np] { start = np; break; }
        }
    }
    if start == usize::MAX { return false; }

    scratch[start] = g;
    stk.clear();
    stk.push(start);
    let mut cnt = 1usize;
    while let Some(pos) = stk.pop() {
        let (px, py) = (pos % w, pos / w);
        for d in 0..4 {
            let nx = px as i32 + DX[d];
            let ny = py as i32 + DY[d];
            if nx >= 0 && nx < w as i32 && ny >= 0 && ny < h as i32 {
                let np = ny as usize * w + nx as usize;
                if !vis[np] && scratch[np] != g {
                    scratch[np] = g;
                    cnt += 1;
                    if cnt == expected { return true; }
                    stk.push(np);
                }
            }
        }
    }
    cnt == expected
}

struct Rng(u64);
impl Rng {
    fn new(s: u64) -> Self { Rng(s | 1) }
    fn next(&mut self) -> u64 {
        self.0 ^= self.0 << 13;
        self.0 ^= self.0 >> 7;
        self.0 ^= self.0 << 17;
        self.0
    }
    fn upto(&mut self, n: usize) -> usize { (self.next() % n as u64) as usize }
}

fn warnsdorff_multi(w: usize, h: usize, blocked: &[bool], total: usize, cyc: bool,
                     t0: Instant, lim: Duration) -> Option<Vec<(usize, usize)>> {
    let mut rng = Rng::new(0xCAFE_BABE_DEAD_BEEF);
    loop {
        if t0.elapsed() > lim { return None; }
        if let Some(p) = w_once(w, h, blocked, total, cyc, &mut rng) { return Some(p); }
    }
}

fn w_once(w: usize, h: usize, blocked: &[bool], total: usize, cyc: bool,
          rng: &mut Rng) -> Option<Vec<(usize, usize)>> {
    let mut vis = blocked.to_vec();
    vis[0] = true;
    let mut path = Vec::with_capacity(total);
    path.push((0, 0));
    let (mut x, mut y) = (0usize, 0usize);

    for step in 1..total {
        let mut best = usize::MAX;
        let mut cands = [(0usize, 0usize); 4];
        let mut cc = 0usize;

        for d in 0..4 {
            let nx = x as i32 + DX[d];
            let ny = y as i32 + DY[d];
            if nx >= 0 && nx < w as i32 && ny >= 0 && ny < h as i32 {
                let (ux, uy) = (nx as usize, ny as usize);
                if !vis[uy * w + ux] {
                    let mut deg = udeg(&vis, w, h, ux, uy);
                    if cyc && step < total - 1 && ux + uy == 1 { deg += 100; }

                    if deg < best {
                        best = deg;
                        cc = 1;
                        cands[0] = (ux, uy);
                    } else if deg == best && cc < 4 {
                        cands[cc] = (ux, uy);
                        cc += 1;
                    }
                }
            }
        }

        if cc == 0 { return None; }
        let (nx, ny) = cands[if cc == 1 { 0 } else { rng.upto(cc) }];
        vis[ny * w + nx] = true;
        path.push((nx, ny));
        x = nx;
        y = ny;
    }

    if cyc && x + y != 1 { return None; }
    Some(path)
}
