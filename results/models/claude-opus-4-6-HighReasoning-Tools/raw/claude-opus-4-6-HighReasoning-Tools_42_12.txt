; SPIR-V Geometry Shader - Silhouette/boundary edge detection
; Input: TrianglesAdjacency (6 verts), Output: TriangleStrip, max 15 verts
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inWorldPos %inNormal %inColor %outColor
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 15
               OpExecutionMode %main Invocations 1

               ; Decorations - input gl_PerVertex block
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

               ; Decorations - output gl_PerVertex block
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               ; Input locations
               OpDecorate %inWorldPos Location 0
               OpDecorate %inNormal Location 1
               OpDecorate %inColor Location 2

               ; Output location
               OpDecorate %outColor Location 0

       ; Types
       %void = OpTypeVoid
   %voidfunc = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeUint 32
        %int = OpTypeInt 32 1
       %bool = OpTypeBool

       ; Constants
     %int_0 = OpConstant %int 0
     %int_1 = OpConstant %int 1
     %int_2 = OpConstant %int 2
     %int_3 = OpConstant %int 3
     %int_4 = OpConstant %int 4
     %int_5 = OpConstant %int 5
    %uint_6 = OpConstant %uint 6
   %float_0 = OpConstant %float 0
   %float_1 = OpConstant %float 1
 %float_0p3 = OpConstant %float 0.3
%float_0p015 = OpConstant %float 0.015
%float_n0p015 = OpConstant %float -0.015

  %v3_white = OpConstantComposite %v3float %float_1 %float_1 %float_1

       ; Input struct and arrays
%gl_PerVertex_in = OpTypeStruct %v4float
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr = OpTypePointer Input %arr_pv_in
 %ptr_in_v4 = OpTypePointer Input %v4float
 %ptr_in_v3 = OpTypePointer Input %v3float
  %ptr_in_f = OpTypePointer Input %float

%arr6_v3 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr6_v3

       ; Output struct
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

       ; Variables
      %gl_in = OpVariable %ptr_in_arr Input
     %gl_out = OpVariable %ptr_out_pv Output
  %inWorldPos = OpVariable %ptr_in_arr_v3 Input
    %inNormal = OpVariable %ptr_in_arr_v3 Input
     %inColor = OpVariable %ptr_in_arr_v3 Input
    %outColor = OpVariable %ptr_out_v3 Output

       ; Function pointer types
  %ptr_func_bool = OpTypePointer Function %bool

       ; Main function
       %main = OpFunction %void None %voidfunc
      %entry = OpLabel

       ; Load main triangle positions (indices 0, 2, 4)
    %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
        %p0 = OpLoad %v4float %p0_ptr
    %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
        %p2 = OpLoad %v4float %p2_ptr
    %p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
        %p4 = OpLoad %v4float %p4_ptr

       ; Load adjacent vertex positions (indices 1, 3, 5)
    %a1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
        %a1 = OpLoad %v4float %a1_ptr
    %a3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
        %a3 = OpLoad %v4float %a3_ptr
    %a5_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_5 %int_0
        %a5 = OpLoad %v4float %a5_ptr

       ; Load colors for the main triangle vertices
   %c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
       %c0 = OpLoad %v3float %c0_ptr
   %c2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
       %c2 = OpLoad %v3float %c2_ptr
   %c4_ptr = OpAccessChain %ptr_in_v3 %inColor %int_4
       %c4 = OpLoad %v3float %c4_ptr

       ; ============================================================
       ; Emit the filled triangle (dimmed colors) - 3 vertices
       ; ============================================================
  %dim_c0 = OpVectorTimesScalar %v3float %c0 %float_0p3
  %dim_c2 = OpVectorTimesScalar %v3float %c2 %float_0p3
  %dim_c4 = OpVectorTimesScalar %v3float %c4 %float_0p3

       ; Vertex 0
  %out_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos0 %p0
               OpStore %outColor %dim_c0
               OpEmitVertex

       ; Vertex 2
               OpStore %out_pos0 %p2
               OpStore %outColor %dim_c2
               OpEmitVertex

       ; Vertex 4
               OpStore %out_pos0 %p4
               OpStore %outColor %dim_c4
               OpEmitVertex
               OpEndPrimitive

       ; ============================================================
       ; Edge 0->2 (adjacent vertex 1): check boundary
       ; Compare a1 with p0 and p2 - if same position, it's boundary
       ; ============================================================
       ; Compute difference a1 - p0
  %d10 = OpFSub %v4float %a1 %p0
  %d10x = OpCompositeExtract %float %d10 0
  %d10y = OpCompositeExtract %float %d10 1
  %d10z = OpCompositeExtract %float %d10 2
  %d10_sq = OpDot %float %d10 %d10

       ; Compute difference a1 - p2
  %d12 = OpFSub %v4float %a1 %p2
  %d12_sq = OpDot %float %d12 %d12

       ; Compute difference a1 - p4
  %d14 = OpFSub %v4float %a1 %p4
  %d14_sq = OpDot %float %d14 %d14

  %eps = OpConstant %float 0.0001
  %is_eq_p0 = OpFOrdLessThan %bool %d10_sq %eps
  %is_eq_p2 = OpFOrdLessThan %bool %d12_sq %eps
  %is_eq_p4 = OpFOrdLessThan %bool %d14_sq %eps
  %is_boundary_e0 = OpLogicalOr %bool %is_eq_p0 %is_eq_p2
  %is_boundary_e0b = OpLogicalOr %bool %is_boundary_e0 %is_eq_p4

               OpSelectionMerge %edge0_end None
               OpBranchConditional %is_boundary_e0b %edge0_emit %edge0_end

  %edge0_emit = OpLabel
       ; Emit thick quad for edge 0->2
       ; Direction along edge
  %e0_dir4 = OpFSub %v4float %p2 %p0
  %e0_dx = OpCompositeExtract %float %e0_dir4 0
  %e0_dy = OpCompositeExtract %float %e0_dir4 1
       ; Perpendicular in screen space: (-dy, dx)
       ; Normalize then scale by thickness
  %e0_neg_dy = OpFNegate %float %e0_dy
  %e0_len_sq = OpFMul %float %e0_dx %e0_dx
  %e0_dy_sq = OpFMul %float %e0_dy %e0_dy
  %e0_len_sq2 = OpFAdd %float %e0_len_sq %e0_dy_sq
  %e0_len = OpExtInst %float %1 Sqrt %e0_len_sq2
  %e0_inv_len = OpFDiv %float %float_1 %e0_len
  %e0_nx = OpFMul %float %e0_neg_dy %e0_inv_len
  %e0_ny = OpFMul %float %e0_dx %e0_inv_len
  %e0_offx_pos = OpFMul %float %e0_nx %float_0p015
  %e0_offy_pos = OpFMul %float %e0_ny %float_0p015
  %e0_offx_neg = OpFMul %float %e0_nx %float_n0p015
  %e0_offy_neg = OpFMul %float %e0_ny %float_n0p015

       ; p0 + offset
  %e0_p0x = OpCompositeExtract %float %p0 0
  %e0_p0y = OpCompositeExtract %float %p0 1
  %e0_p0z = OpCompositeExtract %float %p0 2
  %e0_p0w = OpCompositeExtract %float %p0 3
  %e0_v0_x = OpFAdd %float %e0_p0x %e0_offx_pos
  %e0_v0_y = OpFAdd %float %e0_p0y %e0_offy_pos
  %e0_v0 = OpCompositeConstruct %v4float %e0_v0_x %e0_v0_y %e0_p0z %e0_p0w

       ; p0 - offset
  %e0_v1_x = OpFAdd %float %e0_p0x %e0_offx_neg
  %e0_v1_y = OpFAdd %float %e0_p0y %e0_offy_neg
  %e0_v1 = OpCompositeConstruct %v4float %e0_v1_x %e0_v1_y %e0_p0z %e0_p0w

       ; p2 + offset
  %e0_p2x = OpCompositeExtract %float %p2 0
  %e0_p2y = OpCompositeExtract %float %p2 1
  %e0_p2z = OpCompositeExtract %float %p2 2
  %e0_p2w = OpCompositeExtract %float %p2 3
  %e0_v2_x = OpFAdd %float %e0_p2x %e0_offx_pos
  %e0_v2_y = OpFAdd %float %e0_p2y %e0_offy_pos
  %e0_v2 = OpCompositeConstruct %v4float %e0_v2_x %e0_v2_y %e0_p2z %e0_p2w

       ; p2 - offset
  %e0_v3_x = OpFAdd %float %e0_p2x %e0_offx_neg
  %e0_v3_y = OpFAdd %float %e0_p2y %e0_offy_neg
  %e0_v3 = OpCompositeConstruct %v4float %e0_v3_x %e0_v3_y %e0_p2z %e0_p2w

       ; Emit quad as triangle strip: v0, v1, v2, v3
  %out_pos_e0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos_e0 %e0_v0
               OpStore %outColor %v3_white
               OpEmitVertex
               OpStore %out_pos_e0 %e0_v1
               OpStore %outColor %v3_white
               OpEmitVertex
               OpStore %out_pos_e0 %e0_v2
               OpStore %outColor %v3_white
               OpEmitVertex
               OpStore %out_pos_e0 %e0_v3
               OpStore %outColor %v3_white
               OpEmitVertex
               OpEndPrimitive

               OpBranch %edge0_end
  %edge0_end = OpLabel

       ; ============================================================
       ; Edge 2->4 (adjacent vertex 3): check boundary
       ; ============================================================
  %d30 = OpFSub %v4float %a3 %p0
  %d30_sq = OpDot %float %d30 %d30
  %d32 = OpFSub %v4float %a3 %p2
  %d32_sq = OpDot %float %d32 %d32
  %d34 = OpFSub %v4float %a3 %p4
  %d34_sq = OpDot %float %d34 %d34

  %is_eq2_p0 = OpFOrdLessThan %bool %d30_sq %eps
  %is_eq2_p2 = OpFOrdLessThan %bool %d32_sq %eps
  %is_eq2_p4 = OpFOrdLessThan %bool %d34_sq %eps
  %is_boundary_e1 = OpLogicalOr %bool %is_eq2_p0 %is_eq2_p2
  %is_boundary_e1b = OpLogicalOr %bool %is_boundary_e1 %is_eq2_p4

               OpSelectionMerge %edge1_end None
               OpBranchConditional %is_boundary_e1b %edge1_emit %edge1_end

  %edge1_emit = OpLabel
       ; Emit thick quad for edge 2->4
  %e1_dir4 = OpFSub %v4float %p4 %p2
  %e1_dx = OpCompositeExtract %float %e1_dir4 0
  %e1_dy = OpCompositeExtract %float %e1_dir4 1
  %e1_neg_dy = OpFNegate %float %e1_dy
  %e1_len_sq = OpFMul %float %e1_dx %e1_dx
  %e1_dy_sq = OpFMul %float %e1_dy %e1_dy
  %e1_len_sq2 = OpFAdd %float %e1_len_sq %e1_dy_sq
  %e1_len = OpExtInst %float %1 Sqrt %e1_len_sq2
  %e1_inv_len = OpFDiv %float %float_1 %e1_len
  %e1_nx = OpFMul %float %e1_neg_dy %e1_inv_len
  %e1_ny = OpFMul %float %e1_dx %e1_inv_len
  %e1_offx_pos = OpFMul %float %e1_nx %float_0p015
  %e1_offy_pos = OpFMul %float %e1_ny %float_0p015
  %e1_offx_neg = OpFMul %float %e1_nx %float_n0p015
  %e1_offy_neg = OpFMul %float %e1_ny %float_n0p015

  %e1_p2x = OpCompositeExtract %float %p2 0
  %e1_p2y = OpCompositeExtract %float %p2 1
  %e1_p2z = OpCompositeExtract %float %p2 2
  %e1_p2w = OpCompositeExtract %float %p2 3
  %e1_v0_x = OpFAdd %float %e1_p2x %e1_offx_pos
  %e1_v0_y = OpFAdd %float %e1_p2y %e1_offy_pos
  %e1_v0 = OpCompositeConstruct %v4float %e1_v0_x %e1_v0_y %e1_p2z %e1_p2w

  %e1_v1_x = OpFAdd %float %e1_p2x %e1_offx_neg
  %e1_v1_y = OpFAdd %float %e1_p2y %e1_offy_neg
  %e1_v1 = OpCompositeConstruct %v4float %e1_v1_x %e1_v1_y %e1_p2z %e1_p2w

  %e1_p4x = OpCompositeExtract %float %p4 0
  %e1_p4y = OpCompositeExtract %float %p4 1
  %e1_p4z = OpCompositeExtract %float %p4 2
  %e1_p4w = OpCompositeExtract %float %p4 3
  %e1_v2_x = OpFAdd %float %e1_p4x %e1_offx_pos
  %e1_v2_y = OpFAdd %float %e1_p4y %e1_offy_pos
  %e1_v2 = OpCompositeConstruct %v4float %e1_v2_x %e1_v2_y %e1_p4z %e1_p4w

  %e1_v3_x = OpFAdd %float %e1_p4x %e1_offx_neg
  %e1_v3_y = OpFAdd %float %e1_p4y %e1_offy_neg
  %e1_v3 = OpCompositeConstruct %v4float %e1_v3_x %e1_v3_y %e1_p4z %e1_p4w

  %out_pos_e1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos_e1 %e1_v0
               OpStore %outColor %v3_white
               OpEmitVertex
               OpStore %out_pos_e1 %e1_v1
               OpStore %outColor %v3_white
               OpEmitVertex
               OpStore %out_pos_e1 %e1_v2
               OpStore %outColor %v3_white
               OpEmitVertex
               OpStore %out_pos_e1 %e1_v3
               OpStore %outColor %v3_white
               OpEmitVertex
               OpEndPrimitive

               OpBranch %edge1_end
  %edge1_end = OpLabel

       ; ============================================================
       ; Edge 4->0 (adjacent vertex 5): check boundary
       ; ============================================================
  %d50 = OpFSub %v4float %a5 %p0
  %d50_sq = OpDot %float %d50 %d50
  %d52 = OpFSub %v4float %a5 %p2
  %d52_sq = OpDot %float %d52 %d52
  %d54 = OpFSub %v4float %a5 %p4
  %d54_sq = OpDot %float %d54 %d54

  %is_eq3_p0 = OpFOrdLessThan %bool %d50_sq %eps
  %is_eq3_p2 = OpFOrdLessThan %bool %d52_sq %eps
  %is_eq3_p4 = OpFOrdLessThan %bool %d54_sq %eps
  %is_boundary_e2 = OpLogicalOr %bool %is_eq3_p0 %is_eq3_p2
  %is_boundary_e2b = OpLogicalOr %bool %is_boundary_e2 %is_eq3_p4

               OpSelectionMerge %edge2_end None
               OpBranchConditional %is_boundary_e2b %edge2_emit %edge2_end

  %edge2_emit = OpLabel
       ; Emit thick quad for edge 4->0
  %e2_dir4 = OpFSub %v4float %p0 %p4
  %e2_dx = OpCompositeExtract %float %e2_dir4 0
  %e2_dy = OpCompositeExtract %float %e2_dir4 1
  %e2_neg_dy = OpFNegate %float %e2_dy
  %e2_len_sq = OpFMul %float %e2_dx %e2_dx
  %e2_dy_sq = OpFMul %float %e2_dy %e2_dy
  %e2_len_sq2 = OpFAdd %float %e2_len_sq %e2_dy_sq
  %e2_len = OpExtInst %float %1 Sqrt %e2_len_sq2
  %e2_inv_len = OpFDiv %float %float_1 %e2_len
  %e2_nx = OpFMul %float %e2_neg_dy %e2_inv_len
  %e2_ny = OpFMul %float %e2_dx %e2_inv_len
  %e2_offx_pos = OpFMul %float %e2_nx %float_0p015
  %e2_offy_pos = OpFMul %float %e2_ny %float_0p015
  %e2_offx_neg = OpFMul %float %e2_nx %float_n0p015
  %e2_offy_neg = OpFMul %float %e2_ny %float_n0p015

  %e2_p4x = OpCompositeExtract %float %p4 0
  %e2_p4y = OpCompositeExtract %float %p4 1
  %e2_p4z = OpCompositeExtract %float %p4 2
  %e2_p4w = OpCompositeExtract %float %p4 3
  %e2_v0_x = OpFAdd %float %e2_p4x %e2_offx_pos
  %e2_v0_y = OpFAdd %float %e2_p4y %e2_offy_pos
  %e2_v0 = OpCompositeConstruct %v4float %e2_v0_x %e2_v0_y %e2_p4z %e2_p4w

  %e2_v1_x = OpFAdd %float %e2_p4x %e2_offx_neg
  %e2_v1_y = OpFAdd %float %e2_p4y %e2_offy_neg
  %e2_v1 = OpCompositeConstruct %v4float %e2_v1_x %e2_v1_y %e2_p4z %e2_p4w

  %e2_p0x = OpCompositeExtract %float %p0 0
  %e2_p0y = OpCompositeExtract %float %p0 1
  %e2_p0z = OpCompositeExtract %float %p0 2
  %e2_p0w = OpCompositeExtract %float %p0 3
  %e2_v2_x = OpFAdd %float %e2_p0x %e2_offx_pos
  %e2_v2_y = OpFAdd %float %e2_p0y %e2_offy_pos
  %e2_v2 = OpCompositeConstruct %v4float %e2_v2_x %e2_v2_y %e2_p0z %e2_p0w

  %e2_v3_x = OpFAdd %float %e2_p0x %e2_offx_neg
  %e2_v3_y = OpFAdd %float %e2_p0y %e2_offy_neg
  %e2_v3 = OpCompositeConstruct %v4float %e2_v3_x %e2_v3_y %e2_p0z %e2_p0w

  %out_pos_e2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos_e2 %e2_v0
               OpStore %outColor %v3_white
               OpEmitVertex
               OpStore %out_pos_e2 %e2_v1
               OpStore %outColor %v3_white
               OpEmitVertex
               OpStore %out_pos_e2 %e2_v2
               OpStore %outColor %v3_white
               OpEmitVertex
               OpStore %out_pos_e2 %e2_v3
               OpStore %outColor %v3_white
               OpEmitVertex
               OpEndPrimitive

               OpBranch %edge2_end
  %edge2_end = OpLabel

               OpReturn
               OpFunctionEnd
