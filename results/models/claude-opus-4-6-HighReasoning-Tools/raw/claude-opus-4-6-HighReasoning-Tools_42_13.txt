; SPIR-V Geometry Shader: Wireframe via thin quads from InputTrianglesAdjacency
; Each triangle edge (indices 0,2,4) is extruded into a thin quad strip.

               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450

               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor

               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 12
               OpExecutionMode %main Invocations 1

               ; Decorations - Input PerVertex block
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

               ; Decorations - Output PerVertex block
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               ; Input/output locations
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

       ; ---- Types ----
       %void = OpTypeVoid
   %voidfunc = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1

; Input PerVertex struct and array (size 6 for InputTrianglesAdjacency)
%gl_PerVertex_in = OpTypeStruct %v4float
     %uint_6 = OpConstant %uint 6
%arr_pv_in_6 = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in_6
      %gl_in = OpVariable %ptr_in_arr_pv Input

; Output PerVertex struct
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv  = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output

; Input color array (location 2, vec3, array size 6)
%arr_v3_6    = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_6
    %inColor = OpVariable %ptr_in_arr_v3 Input

; Output color (location 0, vec3)
%ptr_out_v3  = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

; Pointer types for access chains
%ptr_in_v4   = OpTypePointer Input %v4float
%ptr_out_v4  = OpTypePointer Output %v4float
%ptr_in_v3   = OpTypePointer Input %v3float

       ; ---- Constants ----
      %int_0 = OpConstant %int 0
      %int_2 = OpConstant %int 2
      %int_4 = OpConstant %int 4

    %float_0 = OpConstant %float 0.0
  %float_half = OpConstant %float 0.5
 %float_thickness = OpConstant %float 0.008

       ; ---- Main Function ----
       %main = OpFunction %void None %voidfunc
 %main_label = OpLabel

               ; Load positions of main triangle vertices (indices 0, 2, 4)
    %ptr_p0  = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
    %pos0    = OpLoad %v4float %ptr_p0
    %ptr_p2  = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
    %pos2    = OpLoad %v4float %ptr_p2
    %ptr_p4  = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
    %pos4    = OpLoad %v4float %ptr_p4

               ; Load colors of main triangle vertices (indices 0, 2, 4)
    %ptr_c0  = OpAccessChain %ptr_in_v3 %inColor %int_0
    %col0    = OpLoad %v3float %ptr_c0
    %ptr_c2  = OpAccessChain %ptr_in_v3 %inColor %int_2
    %col2    = OpLoad %v3float %ptr_c2
    %ptr_c4  = OpAccessChain %ptr_in_v3 %inColor %int_4
    %col4    = OpLoad %v3float %ptr_c4

               ; ===== EDGE 0→2 =====
               ; Direction vector (in clip space xy)
    %d02_4   = OpFSub %v4float %pos2 %pos0
    %d02_x   = OpCompositeExtract %float %d02_4 0
    %d02_y   = OpCompositeExtract %float %d02_4 1

               ; Perpendicular: (-dy, dx) normalized by just using thickness directly
               ; For thin quads in clip space, we use the perpendicular scaled to thickness
    %neg_d02y = OpFNegate %float %d02_y
               ; length of direction
    %len02_sq = OpExtInst %float %1 Fma %d02_x %d02_x %float_0
    %len02_sq2 = OpExtInst %float %1 Fma %d02_y %d02_y %len02_sq
    %len02   = OpExtInst %float %1 Sqrt %len02_sq2
               ; Normalized perpendicular scaled by thickness
    %inv_len02 = OpFDiv %float %float_thickness %len02
    %perp02_x = OpFMul %float %neg_d02y %inv_len02
    %perp02_y = OpFMul %float %d02_x %inv_len02

               ; Build offset vec4 (perp_x, perp_y, 0, 0)
    %off02   = OpCompositeConstruct %v4float %perp02_x %perp02_y %float_0 %float_0

               ; Average color for edge 0→2
    %avgcol02 = OpExtInst %v3float %1 FMix %col0 %col2 %float_half

               ; Emit 4 vertices: pos0-off, pos0+off, pos2-off, pos2+off
    %e02_v0  = OpFSub %v4float %pos0 %off02
    %store02_0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %store02_0 %e02_v0
               OpStore %outColor %avgcol02
               OpEmitVertex

    %e02_v1  = OpFAdd %v4float %pos0 %off02
               OpStore %store02_0 %e02_v1
               OpStore %outColor %avgcol02
               OpEmitVertex

    %e02_v2  = OpFSub %v4float %pos2 %off02
               OpStore %store02_0 %e02_v2
               OpStore %outColor %avgcol02
               OpEmitVertex

    %e02_v3  = OpFAdd %v4float %pos2 %off02
               OpStore %store02_0 %e02_v3
               OpStore %outColor %avgcol02
               OpEmitVertex

               OpEndPrimitive

               ; ===== EDGE 2→4 =====
    %d24_4   = OpFSub %v4float %pos4 %pos2
    %d24_x   = OpCompositeExtract %float %d24_4 0
    %d24_y   = OpCompositeExtract %float %d24_4 1
    %neg_d24y = OpFNegate %float %d24_y
    %len24_sq = OpExtInst %float %1 Fma %d24_x %d24_x %float_0
    %len24_sq2 = OpExtInst %float %1 Fma %d24_y %d24_y %len24_sq
    %len24   = OpExtInst %float %1 Sqrt %len24_sq2
    %inv_len24 = OpFDiv %float %float_thickness %len24
    %perp24_x = OpFMul %float %neg_d24y %inv_len24
    %perp24_y = OpFMul %float %d24_x %inv_len24
    %off24   = OpCompositeConstruct %v4float %perp24_x %perp24_y %float_0 %float_0

    %avgcol24 = OpExtInst %v3float %1 FMix %col2 %col4 %float_half

    %e24_v0  = OpFSub %v4float %pos2 %off24
               OpStore %store02_0 %e24_v0
               OpStore %outColor %avgcol24
               OpEmitVertex

    %e24_v1  = OpFAdd %v4float %pos2 %off24
               OpStore %store02_0 %e24_v1
               OpStore %outColor %avgcol24
               OpEmitVertex

    %e24_v2  = OpFSub %v4float %pos4 %off24
               OpStore %store02_0 %e24_v2
               OpStore %outColor %avgcol24
               OpEmitVertex

    %e24_v3  = OpFAdd %v4float %pos4 %off24
               OpStore %store02_0 %e24_v3
               OpStore %outColor %avgcol24
               OpEmitVertex

               OpEndPrimitive

               ; ===== EDGE 4→0 =====
    %d40_4   = OpFSub %v4float %pos0 %pos4
    %d40_x   = OpCompositeExtract %float %d40_4 0
    %d40_y   = OpCompositeExtract %float %d40_4 1
    %neg_d40y = OpFNegate %float %d40_y
    %len40_sq = OpExtInst %float %1 Fma %d40_x %d40_x %float_0
    %len40_sq2 = OpExtInst %float %1 Fma %d40_y %d40_y %len40_sq
    %len40   = OpExtInst %float %1 Sqrt %len40_sq2
    %inv_len40 = OpFDiv %float %float_thickness %len40
    %perp40_x = OpFMul %float %neg_d40y %inv_len40
    %perp40_y = OpFMul %float %d40_x %inv_len40
    %off40   = OpCompositeConstruct %v4float %perp40_x %perp40_y %float_0 %float_0

    %avgcol40 = OpExtInst %v3float %1 FMix %col4 %col0 %float_half

    %e40_v0  = OpFSub %v4float %pos4 %off40
               OpStore %store02_0 %e40_v0
               OpStore %outColor %avgcol40
               OpEmitVertex

    %e40_v1  = OpFAdd %v4float %pos4 %off40
               OpStore %store02_0 %e40_v1
               OpStore %outColor %avgcol40
               OpEmitVertex

    %e40_v2  = OpFSub %v4float %pos0 %off40
               OpStore %store02_0 %e40_v2
               OpStore %outColor %avgcol40
               OpEmitVertex

    %e40_v3  = OpFAdd %v4float %pos0 %off40
               OpStore %store02_0 %e40_v3
               OpStore %outColor %avgcol40
               OpEmitVertex

               OpEndPrimitive

               OpReturn
               OpFunctionEnd
