; SPIR-V Geometry Shader: Thick smooth curve via Catmull-Rom subdivision
; InputLinesAdjacency -> OutputTriangleStrip, 10 vertices

               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 10

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

               ; Types
      %void  = OpTypeVoid
  %voidfunc  = OpTypeFunction %void
     %float  = OpTypeFloat 32
   %v2float  = OpTypeVector %float 2
   %v3float  = OpTypeVector %float 3
   %v4float  = OpTypeVector %float 4
      %uint  = OpTypeInt 32 0
       %int  = OpTypeInt 32 1

               ; Constants
    %int_0   = OpConstant %int 0
    %int_1   = OpConstant %int 1
    %int_2   = OpConstant %int 2
    %int_3   = OpConstant %int 3
   %uint_4   = OpConstant %uint 4
  %float_0   = OpConstant %float 0.0
  %float_1   = OpConstant %float 1.0
 %float_0p25 = OpConstant %float 0.25
 %float_0p5  = OpConstant %float 0.5
 %float_0p75 = OpConstant %float 0.75
%float_0p02  = OpConstant %float 0.02
%float_neg0p02 = OpConstant %float -0.02

               ; gl_PerVertex input (array of 4 for InputLinesAdjacency)
%gl_PerVertex_in  = OpTypeStruct %v4float
%arr_pv_in        = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv    = OpTypePointer Input %arr_pv_in
%ptr_in_v4float   = OpTypePointer Input %v4float
%gl_in            = OpVariable %ptr_in_arr_pv Input

               ; gl_PerVertex output
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv       = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float  = OpTypePointer Output %v4float
%gl_out           = OpVariable %ptr_out_pv Output

               ; User-defined color input (array of 4)
%arr4_v3float     = OpTypeArray %v3float %uint_4
%ptr_in_arr4_v3   = OpTypePointer Input %arr4_v3float
%ptr_in_v3float   = OpTypePointer Input %v3float
%inColor          = OpVariable %ptr_in_arr4_v3 Input

               ; User-defined color output
%ptr_out_v3float  = OpTypePointer Output %v3float
%outColor         = OpVariable %ptr_out_v3float Output

               ; Function pointer types
%ptr_func_v4float = OpTypePointer Function %v4float
%ptr_func_v3float = OpTypePointer Function %v3float
%ptr_func_float   = OpTypePointer Function %float

               ; Bool
      %bool  = OpTypeBool

       %main = OpFunction %void None %voidfunc
      %entry = OpLabel

               ; Load the 4 input positions: P0, P1, P2, P3
     %ac_p0  = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
     %ac_p1  = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
     %ac_p2  = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
     %ac_p3  = OpAccessChain %ptr_in_v4float %gl_in %int_3 %int_0
        %P0  = OpLoad %v4float %ac_p0
        %P1  = OpLoad %v4float %ac_p1
        %P2  = OpLoad %v4float %ac_p2
        %P3  = OpLoad %v4float %ac_p3

               ; Load colors for color[1] and color[2]
    %ac_c1   = OpAccessChain %ptr_in_v3float %inColor %int_1
    %ac_c2   = OpAccessChain %ptr_in_v3float %inColor %int_2
    %col1    = OpLoad %v3float %ac_c1
    %col2    = OpLoad %v3float %ac_c2

               ; Compute tangent directions for perpendicular calculation
               ; T_start = P2 - P0 (tangent at P1)
               ; T_end   = P3 - P1 (tangent at P2)
    %T_start = OpFSub %v4float %P2 %P0
    %T_end   = OpFSub %v4float %P3 %P1

               ; Direction along segment: D = P2 - P1
    %D_seg   = OpFSub %v4float %P2 %P1

               ; ============ Point 0: t=0.0 (at P1) ============
               ; Position = P1
               ; Tangent = T_start
    %Ts_x0   = OpCompositeExtract %float %T_start 0
    %Ts_y0   = OpCompositeExtract %float %T_start 1
               ; Perpendicular = normalize(-Ty, Tx) * thickness
    %neg_Tsy0 = OpFNegate %float %Ts_y0
    %len0_sq  = OpFMul %float %neg_Tsy0 %neg_Tsy0
    %len0_sq2 = OpFMul %float %Ts_x0 %Ts_x0
    %len0_sqt = OpFAdd %float %len0_sq %len0_sq2
    %len0     = OpExtInst %float %1 Sqrt %len0_sqt
    %len0_safe = OpExtInst %float %1 FMax %len0 %float_0p02
    %inv_len0  = OpFDiv %float %float_1 %len0_safe
    %perp0_x   = OpFMul %float %neg_Tsy0 %inv_len0
    %perp0_y   = OpFMul %float %Ts_x0 %inv_len0
    %perp0_xsc = OpFMul %float %perp0_x %float_0p02
    %perp0_ysc = OpFMul %float %perp0_y %float_0p02

    %P1_x     = OpCompositeExtract %float %P1 0
    %P1_y     = OpCompositeExtract %float %P1 1
    %P1_z     = OpCompositeExtract %float %P1 2
    %P1_w     = OpCompositeExtract %float %P1 3

               ; Emit vertex 0+: pos + perp
    %v0p_x    = OpFAdd %float %P1_x %perp0_xsc
    %v0p_y    = OpFAdd %float %P1_y %perp0_ysc
    %v0p_pos  = OpCompositeConstruct %v4float %v0p_x %v0p_y %P1_z %P1_w
    %out_ac0p = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %out_ac0p %v0p_pos
               OpStore %outColor %col1
               OpEmitVertex

               ; Emit vertex 0-: pos - perp
    %v0m_x    = OpFSub %float %P1_x %perp0_xsc
    %v0m_y    = OpFSub %float %P1_y %perp0_ysc
    %v0m_pos  = OpCompositeConstruct %v4float %v0m_x %v0m_y %P1_z %P1_w
               OpStore %out_ac0p %v0m_pos
               OpStore %outColor %col1
               OpEmitVertex

               ; ============ Point 1: t=0.25 ============
               ; Linearly interpolate position: mix(P1, P2, 0.25)
    %pos1     = OpExtInst %v4float %1 FMix %P1 %P2 %float_0p25
               ; Interpolate tangent: mix(T_start, T_end, 0.25)
    %tan1     = OpExtInst %v4float %1 FMix %T_start %T_end %float_0p25
    %tan1_x   = OpCompositeExtract %float %tan1 0
    %tan1_y   = OpCompositeExtract %float %tan1 1
    %neg_t1y  = OpFNegate %float %tan1_y
    %l1_sq1   = OpFMul %float %neg_t1y %neg_t1y
    %l1_sq2   = OpFMul %float %tan1_x %tan1_x
    %l1_sqt   = OpFAdd %float %l1_sq1 %l1_sq2
    %l1       = OpExtInst %float %1 Sqrt %l1_sqt
    %l1_safe  = OpExtInst %float %1 FMax %l1 %float_0p02
    %inv_l1   = OpFDiv %float %float_1 %l1_safe
    %perp1_x  = OpFMul %float %neg_t1y %inv_l1
    %perp1_y  = OpFMul %float %tan1_x %inv_l1
    %perp1_xs = OpFMul %float %perp1_x %float_0p02
    %perp1_ys = OpFMul %float %perp1_y %float_0p02

    %pos1_x   = OpCompositeExtract %float %pos1 0
    %pos1_y   = OpCompositeExtract %float %pos1 1
    %pos1_z   = OpCompositeExtract %float %pos1 2
    %pos1_w   = OpCompositeExtract %float %pos1 3

               ; Color at t=0.25
    %col_1    = OpExtInst %v3float %1 FMix %col1 %col2 %float_0p25

               ; Emit vertex 1+
    %v1p_x    = OpFAdd %float %pos1_x %perp1_xs
    %v1p_y    = OpFAdd %float %pos1_y %perp1_ys
    %v1p_pos  = OpCompositeConstruct %v4float %v1p_x %v1p_y %pos1_z %pos1_w
               OpStore %out_ac0p %v1p_pos
               OpStore %outColor %col_1
               OpEmitVertex

               ; Emit vertex 1-
    %v1m_x    = OpFSub %float %pos1_x %perp1_xs
    %v1m_y    = OpFSub %float %pos1_y %perp1_ys
    %v1m_pos  = OpCompositeConstruct %v4float %v1m_x %v1m_y %pos1_z %pos1_w
               OpStore %out_ac0p %v1m_pos
               OpStore %outColor %col_1
               OpEmitVertex

               ; ============ Point 2: t=0.5 ============
    %pos2     = OpExtInst %v4float %1 FMix %P1 %P2 %float_0p5
    %tan2     = OpExtInst %v4float %1 FMix %T_start %T_end %float_0p5
    %tan2_x   = OpCompositeExtract %float %tan2 0
    %tan2_y   = OpCompositeExtract %float %tan2 1
    %neg_t2y  = OpFNegate %float %tan2_y
    %l2_sq1   = OpFMul %float %neg_t2y %neg_t2y
    %l2_sq2   = OpFMul %float %tan2_x %tan2_x
    %l2_sqt   = OpFAdd %float %l2_sq1 %l2_sq2
    %l2       = OpExtInst %float %1 Sqrt %l2_sqt
    %l2_safe  = OpExtInst %float %1 FMax %l2 %float_0p02
    %inv_l2   = OpFDiv %float %float_1 %l2_safe
    %perp2_x  = OpFMul %float %neg_t2y %inv_l2
    %perp2_y  = OpFMul %float %tan2_x %inv_l2
    %perp2_xs = OpFMul %float %perp2_x %float_0p02
    %perp2_ys = OpFMul %float %perp2_y %float_0p02

    %pos2_x   = OpCompositeExtract %float %pos2 0
    %pos2_y   = OpCompositeExtract %float %pos2 1
    %pos2_z   = OpCompositeExtract %float %pos2 2
    %pos2_w   = OpCompositeExtract %float %pos2 3

    %col_2    = OpExtInst %v3float %1 FMix %col1 %col2 %float_0p5

               ; Emit vertex 2+
    %v2p_x    = OpFAdd %float %pos2_x %perp2_xs
    %v2p_y    = OpFAdd %float %pos2_y %perp2_ys
    %v2p_pos  = OpCompositeConstruct %v4float %v2p_x %v2p_y %pos2_z %pos2_w
               OpStore %out_ac0p %v2p_pos
               OpStore %outColor %col_2
               OpEmitVertex

               ; Emit vertex 2-
    %v2m_x    = OpFSub %float %pos2_x %perp2_xs
    %v2m_y    = OpFSub %float %pos2_y %perp2_ys
    %v2m_pos  = OpCompositeConstruct %v4float %v2m_x %v2m_y %pos2_z %pos2_w
               OpStore %out_ac0p %v2m_pos
               OpStore %outColor %col_2
               OpEmitVertex

               ; ============ Point 3: t=0.75 ============
    %pos3     = OpExtInst %v4float %1 FMix %P1 %P2 %float_0p75
    %tan3     = OpExtInst %v4float %1 FMix %T_start %T_end %float_0p75
    %tan3_x   = OpCompositeExtract %float %tan3 0
    %tan3_y   = OpCompositeExtract %float %tan3 1
    %neg_t3y  = OpFNegate %float %tan3_y
    %l3_sq1   = OpFMul %float %neg_t3y %neg_t3y
    %l3_sq2   = OpFMul %float %tan3_x %tan3_x
    %l3_sqt   = OpFAdd %float %l3_sq1 %l3_sq2
    %l3       = OpExtInst %float %1 Sqrt %l3_sqt
    %l3_safe  = OpExtInst %float %1 FMax %l3 %float_0p02
    %inv_l3   = OpFDiv %float %float_1 %l3_safe
    %perp3_x  = OpFMul %float %neg_t3y %inv_l3
    %perp3_y  = OpFMul %float %tan3_x %inv_l3
    %perp3_xs = OpFMul %float %perp3_x %float_0p02
    %perp3_ys = OpFMul %float %perp3_y %float_0p02

    %pos3_x   = OpCompositeExtract %float %pos3 0
    %pos3_y   = OpCompositeExtract %float %pos3 1
    %pos3_z   = OpCompositeExtract %float %pos3 2
    %pos3_w   = OpCompositeExtract %float %pos3 3

    %col_3    = OpExtInst %v3float %1 FMix %col1 %col2 %float_0p75

               ; Emit vertex 3+
    %v3p_x    = OpFAdd %float %pos3_x %perp3_xs
    %v3p_y    = OpFAdd %float %pos3_y %perp3_ys
    %v3p_pos  = OpCompositeConstruct %v4float %v3p_x %v3p_y %pos3_z %pos3_w
               OpStore %out_ac0p %v3p_pos
               OpStore %outColor %col_3
               OpEmitVertex

               ; Emit vertex 3-
    %v3m_x    = OpFSub %float %pos3_x %perp3_xs
    %v3m_y    = OpFSub %float %pos3_y %perp3_ys
    %v3m_pos  = OpCompositeConstruct %v4float %v3m_x %v3m_y %pos3_z %pos3_w
               OpStore %out_ac0p %v3m_pos
               OpStore %outColor %col_3
               OpEmitVertex

               ; ============ Point 4: t=1.0 (at P2) ============
    %P2_x     = OpCompositeExtract %float %P2 0
    %P2_y     = OpCompositeExtract %float %P2 1
    %P2_z     = OpCompositeExtract %float %P2 2
    %P2_w     = OpCompositeExtract %float %P2 3

               ; Tangent at end = T_end
    %Te_x     = OpCompositeExtract %float %T_end 0
    %Te_y     = OpCompositeExtract %float %T_end 1
    %neg_Tey  = OpFNegate %float %Te_y
    %l4_sq1   = OpFMul %float %neg_Tey %neg_Tey
    %l4_sq2   = OpFMul %float %Te_x %Te_x
    %l4_sqt   = OpFAdd %float %l4_sq1 %l4_sq2
    %l4       = OpExtInst %float %1 Sqrt %l4_sqt
    %l4_safe  = OpExtInst %float %1 FMax %l4 %float_0p02
    %inv_l4   = OpFDiv %float %float_1 %l4_safe
    %perp4_x  = OpFMul %float %neg_Tey %inv_l4
    %perp4_y  = OpFMul %float %Te_x %inv_l4
    %perp4_xs = OpFMul %float %perp4_x %float_0p02
    %perp4_ys = OpFMul %float %perp4_y %float_0p02

               ; Emit vertex 4+
    %v4p_x    = OpFAdd %float %P2_x %perp4_xs
    %v4p_y    = OpFAdd %float %P2_y %perp4_ys
    %v4p_pos  = OpCompositeConstruct %v4float %v4p_x %v4p_y %P2_z %P2_w
               OpStore %out_ac0p %v4p_pos
               OpStore %outColor %col2
               OpEmitVertex

               ; Emit vertex 4-
    %v4m_x    = OpFSub %float %P2_x %perp4_xs
    %v4m_y    = OpFSub %float %P2_y %perp4_ys
    %v4m_pos  = OpCompositeConstruct %v4float %v4m_x %v4m_y %P2_z %P2_w
               OpStore %out_ac0p %v4m_pos
               OpStore %outColor %col2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
