import struct

words = []

def w(val):
    words.append(val & 0xFFFFFFFF)

def op(wordcount, opcode):
    w((wordcount << 16) | opcode)

def encode_string(s):
    b = s.encode('utf-8') + b'\x00'
    while len(b) % 4 != 0:
        b += b'\x00'
    result = []
    for i in range(0, len(b), 4):
        result.append(struct.unpack('<I', b[i:i+4])[0])
    return result

def float_to_word(f):
    return struct.unpack('<I', struct.pack('<f', f))[0]
