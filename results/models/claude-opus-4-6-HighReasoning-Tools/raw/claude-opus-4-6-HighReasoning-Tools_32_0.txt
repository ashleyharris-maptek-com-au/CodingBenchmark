#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <numeric>
#include <functional>
#include <intrin.h>

using namespace std;
typedef long long ll;
typedef pair<ll, int> pli;
const ll INF = 1e18;

struct Edge { int to; ll w; };

int N, M, K;
vector<vector<Edge>> adj;
vector<int> terminals;
vector<bool> isTerminal;

void dijkstra(int src, vector<ll>& dist, vector<int>& par) {
    dist.assign(N, INF);
    par.assign(N, -1);
    dist[src] = 0;
    priority_queue<pli, vector<pli>, greater<pli>> pq;
    pq.push({0, src});
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;
        for (auto& e : adj[u]) {
            if (dist[u] + e.w < dist[e.to]) {
                dist[e.to] = dist[u] + e.w;
                par[e.to] = u;
                pq.push({dist[e.to], e.to});
            }
        }
    }
}

ll edgeWeight(int u, int v) {
    ll best = INF;
    for (auto& e : adj[u])
        if (e.to == v) best = min(best, e.w);
    return best;
}

void solveDreyfusWagner() {
    int fullMask = (1 << K) - 1;
    int MASKS = fullMask + 1;

    vector<vector<ll>> dp(MASKS, vector<ll>(N, INF));
    vector<vector<int>> splitMask(MASKS, vector<int>(N, -1));
    vector<vector<int>> parVertex(MASKS, vector<int>(N, -1));

    for (int i = 0; i < K; i++)
        dp[1 << i][terminals[i]] = 0;

    for (int S = 1; S <= fullMask; S++) {
        for (int T = (S - 1) & S; T > 0; T = (T - 1) & S) {
            int U = S ^ T;
            if (T < U) continue;
            for (int v = 0; v < N; v++) {
                if (dp[T][v] == INF || dp[U][v] == INF) continue;
                ll cost = dp[T][v] + dp[U][v];
                if (cost < dp[S][v]) {
                    dp[S][v] = cost;
                    splitMask[S][v] = T;
                    parVertex[S][v] = -1;
                }
            }
        }

        priority_queue<pli, vector<pli>, greater<pli>> pq;
        for (int v = 0; v < N; v++)
            if (dp[S][v] < INF)
                pq.push({dp[S][v], v});

        while (!pq.empty()) {
            auto [d, u] = pq.top(); pq.pop();
            if (d > dp[S][u]) continue;
            for (auto& e : adj[u]) {
                ll nd = d + e.w;
                if (nd < dp[S][e.to]) {
                    dp[S][e.to] = nd;
                    splitMask[S][e.to] = -1;
                    parVertex[S][e.to] = u;
                    pq.push({nd, e.to});
                }
            }
        }
    }

    ll bestCost = INF;
    int bestRoot = 0;
    for (int v = 0; v < N; v++)
        if (dp[fullMask][v] < bestCost) {
            bestCost = dp[fullMask][v];
            bestRoot = v;
        }

    // Reconstruct
    vector<pair<int,int>> edges;
    struct State { int mask, v; };
    vector<State> stk;
    stk.push_back({fullMask, bestRoot});

    while (!stk.empty()) {
        auto [mask, v] = stk.back();
        stk.pop_back();
        if (mask == 0) continue;

        if (__popcnt((unsigned)mask) == 1) {
            int bit = 0;
            for (int b = 0; b < K; b++)
                if (mask & (1 << b)) { bit = b; break; }
            if (v == terminals[bit]) continue;
            if (parVertex[mask][v] != -1) {
                int u = parVertex[mask][v];
                edges.push_back({u, v});
                stk.push_back({mask, u});
            }
            continue;
        }

        if (splitMask[mask][v] != -1) {
            int T = splitMask[mask][v];
            int U = mask ^ T;
            stk.push_back({T, v});
            stk.push_back({U, v});
        } else if (parVertex[mask][v] != -1) {
            int u = parVertex[mask][v];
            edges.push_back({u, v});
            stk.push_back({mask, u});
        }
    }

    // Deduplicate edges (normalize direction)
    for (auto& [u, v] : edges)
        if (u > v) swap(u, v);
    sort(edges.begin(), edges.end());
    edges.erase(unique(edges.begin(), edges.end()), edges.end());

    // Build subgraph, BFS spanning tree, prune
    vector<vector<pair<int,ll>>> subAdj(N);
    for (auto& [u, v] : edges) {
        ll w = edgeWeight(u, v);
        subAdj[u].push_back({v, w});
        subAdj[v].push_back({u, w});
    }

    vector<pair<int,int>> treeEdges;
    vector<bool> visited(N, false);
    queue<int> bfsQ;
    visited[bestRoot] = true;
    bfsQ.push(bestRoot);
    while (!bfsQ.empty()) {
        int u = bfsQ.front(); bfsQ.pop();
        for (auto& [v, w] : subAdj[u]) {
            if (!visited[v]) {
                visited[v] = true;
                treeEdges.push_back({u, v});
                bfsQ.push(v);
            }
        }
    }

    // Prune non-terminal leaves
    vector<int> deg(N, 0);
    vector<vector<pair<int,int>>> tAdj(N);
    for (int i = 0; i < (int)treeEdges.size(); i++) {
        auto [u, v] = treeEdges[i];
        deg[u]++; deg[v]++;
        tAdj[u].push_back({v, i});
        tAdj[v].push_back({u, i});
    }

    vector<bool> alive(treeEdges.size(), true);
    queue<int> leafQ;
    for (int v = 0; v < N; v++)
        if (deg[v] == 1 && !isTerminal[v]) leafQ.push(v);
    while (!leafQ.empty()) {
        int u = leafQ.front(); leafQ.pop();
        if (deg[u] != 1 || isTerminal[u]) continue;
        for (auto& [v, eidx] : tAdj[u]) {
            if (alive[eidx]) {
                alive[eidx] = false;
                deg[u]--; deg[v]--;
                if (deg[v] == 1 && !isTerminal[v]) leafQ.push(v);
                break;
            }
        }
    }

    vector<pair<int,int>> finalEdges;
    ll finalWeight = 0;
    for (int i = 0; i < (int)treeEdges.size(); i++) {
        if (!alive[i]) continue;
        auto [u, v] = treeEdges[i];
        finalEdges.push_back({u, v});
        finalWeight += edgeWeight(u, v);
    }

    cout << finalWeight << " " << finalEdges.size() << "\n";
    for (auto& [u, v] : finalEdges)
        cout << u << " " << v << "\n";
}

void solveHeuristic() {
    vector<vector<ll>> dist(K);
    vector<vector<int>> par(K);
    for (int i = 0; i < K; i++)
        dijkstra(terminals[i], dist[i], par[i]);

    // Prim's on complete terminal graph
    vector<bool> inMST(K, false);
    vector<ll> key(K, INF);
    vector<int> mstPar(K, -1);
    key[0] = 0;

    for (int iter = 0; iter < K; iter++) {
        int u = -1;
        for (int i = 0; i < K; i++)
            if (!inMST[i] && (u == -1 || key[i] < key[u])) u = i;
        inMST[u] = true;
        for (int v = 0; v < K; v++)
            if (!inMST[v] && dist[u][terminals[v]] < key[v]) {
                key[v] = dist[u][terminals[v]];
                mstPar[v] = u;
            }
    }

    // Collect shortest-path edges
    vector<pair<int,int>> collected;
    for (int i = 1; i < K; i++) {
        int pi = mstPar[i];
        int cur = terminals[i];
        while (cur != terminals[pi]) {
            int prev = par[pi][cur];
            collected.push_back({min(prev, cur), max(prev, cur)});
            cur = prev;
        }
    }

    sort(collected.begin(), collected.end());
    collected.erase(unique(collected.begin(), collected.end()), collected.end());

    // Kruskal MST on subgraph
    struct WEdge { int u, v; ll w; };
    vector<WEdge> wedges;
    for (auto& [u, v] : collected)
        wedges.push_back({u, v, edgeWeight(u, v)});
    sort(wedges.begin(), wedges.end(), [](auto& a, auto& b) { return a.w < b.w; });

    vector<int> uf(N);
    iota(uf.begin(), uf.end(), 0);
    function<int(int)> find = [&](int x) {
        while (uf[x] != x) x = uf[x] = uf[uf[x]];
        return x;
    };

    vector<pair<int,int>> treeEdges;
    ll totalWeight = 0;
    for (auto& we : wedges) {
        int fu = find(we.u), fv = find(we.v);
        if (fu != fv) {
            uf[fu] = fv;
            treeEdges.push_back({we.u, we.v});
            totalWeight += we.w;
        }
    }

    // Prune non-terminal leaves
    vector<int> deg(N, 0);
    vector<vector<pair<int,int>>> tAdj(N);
    for (int i = 0; i < (int)treeEdges.size(); i++) {
        auto [u, v] = treeEdges[i];
        deg[u]++; deg[v]++;
        tAdj[u].push_back({v, i});
        tAdj[v].push_back({u, i});
    }

    vector<bool> alive(treeEdges.size(), true);
    queue<int> leafQ;
    for (int v = 0; v < N; v++)
        if (deg[v] == 1 && !isTerminal[v]) leafQ.push(v);
    while (!leafQ.empty()) {
        int u = leafQ.front(); leafQ.pop();
        if (deg[u] != 1 || isTerminal[u]) continue;
        for (auto& [v, eidx] : tAdj[u]) {
            if (alive[eidx]) {
                alive[eidx] = false;
                deg[u]--; deg[v]--;
                totalWeight -= edgeWeight(u, v);
                if (deg[v] == 1 && !isTerminal[v]) leafQ.push(v);
                break;
            }
        }
    }

    vector<pair<int,int>> finalEdges;
    for (int i = 0; i < (int)treeEdges.size(); i++)
        if (alive[i]) finalEdges.push_back(treeEdges[i]);

    cout << totalWeight << " " << finalEdges.size() << "\n";
    for (auto& [u, v] : finalEdges)
        cout << u << " " << v << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> N >> M >> K;
    adj.resize(N);
    isTerminal.assign(N, false);
    terminals.resize(K);

    for (int i = 0; i < M; i++) {
        int u, v; ll w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }
    for (int i = 0; i < K; i++) {
        cin >> terminals[i];
        isTerminal[terminals[i]] = true;
    }

    if (K <= 1) {
        cout << "0 0\n";
        return 0;
    }

    if (K == 2) {
        vector<ll> dist; vector<int> par;
        dijkstra(terminals[0], dist, par);
        vector<pair<int,int>> path;
        int cur = terminals[1];
        while (cur != terminals[0]) {
            int prev = par[cur];
            path.push_back({prev, cur});
            cur = prev;
        }
        cout << dist[terminals[1]] << " " << path.size() << "\n";
        for (auto& [u, v] : path) cout << u << " " << v << "\n";
        return 0;
    }

    if (K <= 13 && N <= 5000)
        solveDreyfusWagner();
    else
        solveHeuristic();

    return 0;
}
