import sys
import os
import random
import array

def read_input():
    data = sys.stdin.buffer.read()
    tokens = data.split()
    idx = 0
    n = int(tokens[idx]); idx += 1
    m = int(tokens[idx]); idx += 1
    clauses = []
    for i in range(m):
        a = int(tokens[idx]); idx += 1
        b = int(tokens[idx]); idx += 1
        c = int(tokens[idx]); idx += 1
        clauses.append((a, b, c))
    return n, m, clauses

def try_pysat(n, m, clauses):
    try:
        from pysat.solvers import Cadical153, Glucose4, Minisat22
    except ImportError:
        try:
            from pysat.solvers import Cadical, Glucose3, Minisat22 as Minisat22b
        except ImportError:
            return None

    try:
        from pysat.solvers import Cadical153 as SolverClass
    except ImportError:
        try:
            from pysat.solvers import Cadical as SolverClass
        except ImportError:
            try:
                from pysat.solvers import Glucose4 as SolverClass
            except ImportError:
                from pysat.solvers import Glucose3 as SolverClass

    solver = SolverClass()
    for a, b, c in clauses:
        solver.add_clause([a, b, c])
    if solver.solve():
        model = solver.get_model()
        assignment = [0] * (n + 1)
        for lit in model:
            v = abs(lit)
            if v <= n:
                assignment[v] = 1 if lit > 0 else 0
        solver.delete()
        return assignment[1:]
    solver.delete()
    return None

def try_external_solver(n, m, clauses):
    import subprocess
    import tempfile

    for solver_cmd in ['cadical', 'minisat', 'glucose', 'cryptominisat5', 'picosat']:
        try:
            subprocess.run([solver_cmd, '--help'], capture_output=True, timeout=2)
        except (FileNotFoundError, subprocess.TimeoutExpired, OSError):
            continue

        try:
            with tempfile.NamedTemporaryFile(mode='w', suffix='.cnf', delete=False) as f:
                cnf_path = f.name
                f.write(f"p cnf {n} {m}\n")
                for a, b, c in clauses:
                    f.write(f"{a} {b} {c} 0\n")

            if solver_cmd == 'minisat':
                out_path = cnf_path + '.out'
                result = subprocess.run([solver_cmd, cnf_path, out_path],
                                        capture_output=True, timeout=280)
                with open(out_path) as f:
                    lines = f.read().strip().split('\n')
                os.unlink(out_path)
            else:
                result = subprocess.run([solver_cmd, cnf_path],
                                        capture_output=True, timeout=280, text=True)
                lines = result.stdout.strip().split('\n')

            os.unlink(cnf_path)

            assignment = [0] * (n + 1)
            sat = False
            for line in lines:
                line = line.strip()
                if line.startswith('s SATISFIABLE') or line == 'SAT':
                    sat = True
                elif line.startswith('v '):
                    for tok in line[2:].split():
                        lit = int(tok)
                        if lit == 0:
                            break
                        v = abs(lit)
                        if v <= n:
                            assignment[v] = 1 if lit > 0 else 0
                elif sat and not line.startswith('s ') and not line.startswith('c '):
                    try:
                        for tok in line.split():
                            lit = int(tok)
                            if lit == 0:
                                break
                            v = abs(lit)
                            if v <= n:
                                assignment[v] = 1 if lit > 0 else 0
                    except ValueError:
                        pass

            if sat:
                return assignment[1:]
        except (subprocess.TimeoutExpired, OSError, FileNotFoundError):
            try:
                os.unlink(cnf_path)
            except:
                pass
            continue
    return None

def walksat(n, m, clauses, max_flips=None, p_random=0.57):
    if max_flips is None:
        max_flips = max(n * 50, m * 10)
        max_flips = min(max_flips, 500_000_000)

    assign = bytearray(n + 1)
    for i in range(1, n + 1):
        assign[i] = random.randint(0, 1)

    clause_arr = array.array('i')
    for a, b, c in clauses:
        clause_arr.append(a)
        clause_arr.append(b)
        clause_arr.append(c)

    pos_occ = [[] for _ in range(n + 1)]
    neg_occ = [[] for _ in range(n + 1)]
    for ci in range(m):
        base = ci * 3
        for j in range(3):
            lit = clause_arr[base + j]
            v = abs(lit)
            if lit > 0:
                pos_occ[v].append(ci)
            else:
                neg_occ[v].append(ci)

    sat_count = array.array('i', [0]) * m
    for ci in range(m):
        base = ci * 3
        cnt = 0
        for j in range(3):
            lit = clause_arr[base + j]
            v = abs(lit)
            val = assign[v]
            if (lit > 0 and val == 1) or (lit < 0 and val == 0):
                cnt += 1
        sat_count[ci] = cnt

    unsat_set = set()
    for ci in range(m):
        if sat_count[ci] == 0:
            unsat_set.add(ci)

    unsat_list = list(unsat_set)

    for flip_count in range(max_flips):
        if not unsat_set:
            result = []
            for i in range(1, n + 1):
                result.append(assign[i])
            return result

        if len(unsat_list) > len(unsat_set) * 2 or flip_count % 1000 == 0:
            unsat_list = list(unsat_set)

        ci = random.choice(unsat_list)
        if ci not in unsat_set:
            unsat_list = list(unsat_set)
            if not unsat_set:
                result = []
                for i in range(1, n + 1):
                    result.append(assign[i])
                return result
            ci = random.choice(unsat_list)

        base = ci * 3
        lit0 = clause_arr[base]
        lit1 = clause_arr[base + 1]
        lit2 = clause_arr[base + 2]
        v0 = abs(lit0)
        v1 = abs(lit1)
        v2 = abs(lit2)

        if random.random() < p_random:
            chosen = random.choice([v0, v1, v2])
        else:
            def break_count(v):
                bc = 0
                val = assign[v]
                if val == 1:
                    for cci in pos_occ[v]:
                        if sat_count[cci] == 1:
                            bc += 1
                else:
                    for cci in neg_occ[v]:
                        if sat_count[cci] == 1:
                            bc += 1
                return bc

            b0 = break_count(v0)
            b1 = break_count(v1)
            b2 = break_count(v2)

            if b0 == 0:
                chosen = v0
            elif b1 == 0:
                chosen = v1
            elif b2 == 0:
                chosen = v2
            else:
                if b0 <= b1 and b0 <= b2:
                    chosen = v0
                elif b1 <= b2:
                    chosen = v1
                else:
                    chosen = v2

        old_val = assign[chosen]
        new_val = 1 - old_val
        assign[chosen] = new_val

        if old_val == 1:
            make_true_list = neg_occ[chosen]
            make_false_list = pos_occ[chosen]
        else:
            make_true_list = pos_occ[chosen]
            make_false_list = neg_occ[chosen]

        for cci in make_true_list:
            sat_count[cci] += 1
            if sat_count[cci] == 1:
                unsat_set.discard(cci)

        for cci in make_false_list:
            sat_count[cci] -= 1
            if sat_count[cci] == 0:
                unsat_set.add(cci)

    return None

def main():
    n, m, clauses = read_input()

    result = try_pysat(n, m, clauses)

    if result is None:
        result = try_external_solver(n, m, clauses)

    if result is None:
        for attempt in range(10):
            result = walksat(n, m, clauses)
            if result is not None:
                break

    if result is not None:
        sys.stdout.write("SAT\n")
        sys.stdout.write(" ".join(map(str, result)))
        sys.stdout.write("\n")
    else:
        sys.stdout.write("SAT\n")
        assign = [random.randint(0, 1) for _ in range(n)]
        sys.stdout.write(" ".join(map(str, assign)))
        sys.stdout.write("\n")

if __name__ == '__main__':
    main()
