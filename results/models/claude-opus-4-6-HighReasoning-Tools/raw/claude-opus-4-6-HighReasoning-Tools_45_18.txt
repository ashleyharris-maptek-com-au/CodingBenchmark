import struct

def float_bits(f):
    return struct.unpack('<I', struct.pack('<f', f))[0]

def string_words(s):
    b = s.encode('utf-8') + b'\x00'
    while len(b) % 4:
        b += b'\x00'
    return [struct.unpack('<I', b[i:i+4])[0] for i in range(0, len(b), 4)]

class SPIRVBuilder:
    def __init__(self):
        self.instructions = []

    def inst(self, opcode, *operands):
        wc = 1 + len(operands)
        self.instructions.append([(wc << 16) | opcode] + list(operands))

    def inst_str(self, opcode, pre_ops, s, post_ops=None):
        if post_ops is None:
            post_ops = []
        sw = string_words(s)
        ops = list(pre_ops) + sw + list(post_ops)
        wc = 1 + len(ops)
        self.instructions.append([(wc << 16) | opcode] + ops)

    def build(self, bound):
        header = [0x07230203, 0x00010000, 0, bound, 0]
        all_words = header[:]
        for inst in self.instructions:
            all_words.extend(inst)
        return b''.join(struct.pack('<I', w) for w in all_words)

b.inst(61, T_VEC3, LOAD_WPOS, V_WPOS)

b.inst(61, T_VEC3, LOAD_NORM, V_NORM)

CAMPOS_XYZ = n; n += 1
b.inst(79, T_VEC3, CAMPOS_XYZ, LOAD_CAMPOS, LOAD_CAMPOS, 0, 1, 2)

# V_dir = cameraPos.xyz - worldPos

b.inst(131, T_VEC3, V_DIR, CAMPOS_XYZ, LOAD_WPOS)

# V = normalize(V_dir)  -- GLSL Normalize = 69

b.inst(12, T_VEC3, V_NORM_VEC, GLSL, 69, V_DIR)

# N = normalize(normal)

# dot(N, V)
DOT_NV = n; n += 1
b.inst(148, T_FLOAT, DOT_NV, N_NORM_VEC, V_NORM_VEC)

# NdotV = max(dot, 0.0)  -- GLSL FMax = 40

b.inst(12, T_FLOAT, NDOTV, GLSL, 40, DOT_NV, C_F0)

# cond = NdotV < 0.5  -- OpFOrdLessThan = 184

b.inst(184, T_BOOL, COND, NDOTV, C_F05)

# Branch A (NdotV < 0.5): r=1, g=2*NdotV, b=0
G_A = n; n += 1
b.inst(133, T_FLOAT, G_A, C_F2, NDOTV)

# Branch B (NdotV >= 0.5): t=(NdotV-0.5)*2, r=g=1-t, b=t
T1 = n; n += 1
b.inst(131, T_FLOAT, T1, NDOTV, C_F05)

T_VAL = n; n += 1
b.inst(133, T_FLOAT, T_VAL, T1, C_F2)

b.inst(131, T_FLOAT, ONE_MINUS_T, C_F1, T_VAL)

# Select components -- OpSelect = 169
R_SEL = n; n += 1

G_SEL = n; n += 1
b.inst(169, T_FLOAT, G_SEL, COND, G_A, ONE_MINUS_T)

B_SEL = n; n += 1
b.inst(169, T_FLOAT, B_SEL, COND, C_F0, T_VAL)

# Construct output vec4(r, g, b, 1.0)
OUT_VEC = n; n += 1
b.inst(80, T_VEC4, OUT_VEC, R_SEL, G_SEL, B_SEL, C_F1)

# Store output
b.inst(62, V_OUT, OUT_VEC)

# Return and end
