#include <cstdio>
#include <cmath>
#include <cstring>
#include <vector>
#include <algorithm>
#include <chrono>

using namespace std;
using namespace chrono;

static int N;
static float *px, *py;

inline float dist2(int a, int b) {
    float dx = px[a] - px[b], dy = py[a] - py[b];
    return dx * dx + dy * dy;
}
inline float calcDist(int a, int b) {
    return sqrtf(dist2(a, b));
}

int main() {
    scanf("%d", &N);
    px = new float[N];
    py = new float[N];
    for (int i = 0; i < N; i++) scanf("%f %f", &px[i], &py[i]);

    if (N <= 2) {
        printf("0");
        for (int i = 1; i < N; i++) printf(" %d", i);
        printf("\n");
        return 0;
    }

    auto t0 = steady_clock::now();
    auto secs = [&]() { return duration<double>(steady_clock::now() - t0).count(); };

    int* tour = new int[N];

    // Bounding box
    float xmin = px[0], xmax = px[0], ymin = py[0], ymax = py[0];
    for (int i = 1; i < N; i++) {
        if (px[i] < xmin) xmin = px[i]; if (px[i] > xmax) xmax = px[i];
        if (py[i] < ymin) ymin = py[i]; if (py[i] > ymax) ymax = py[i];
    }
    float rx = xmax - xmin, ry = ymax - ymin;
    if (rx < 1e-9f) rx = 1.0f;
    if (ry < 1e-9f) ry = 1.0f;

    // Grid
    int G = max(1, (int)sqrt((double)N / 4.0));
    if (G > 8000) G = 8000;
    int G2 = G * G;
    float invRx = (float)G / rx, invRy = (float)G / ry;

    auto cellOf = [&](int i) -> int {
        int cx = (int)((px[i] - xmin) * invRx); if (cx >= G) cx = G - 1;
        int cy = (int)((py[i] - ymin) * invRy); if (cy >= G) cy = G - 1;
        return cy * G + cx;
    };
    auto cellXY = [&](float fx, float fy, int &outX, int &outY) {
        outX = (int)((fx - xmin) * invRx); if (outX >= G) outX = G - 1; if (outX < 0) outX = 0;
        outY = (int)((fy - ymin) * invRy); if (outY >= G) outY = G - 1; if (outY < 0) outY = 0;
    };

    // Flat grid with O(1) removal
    vector<int> gcnt(G2, 0);
    for (int i = 0; i < N; i++) gcnt[cellOf(i)]++;
    vector<int> goff(G2 + 1, 0);
    for (int c = 0; c < G2; c++) goff[c + 1] = goff[c] + gcnt[c];
    vector<int> gdat(N), gpos(N);
    {
        vector<int> t(G2, 0);
        for (int i = 0; i < N; i++) {
            int c = cellOf(i), p = goff[c] + t[c]++;
            gdat[p] = i; gpos[i] = p;
        }
    }
    auto gridRm = [&](int city) {
        int c = cellOf(city); gcnt[c]--;
        int p = gpos[city], lp = goff[c] + gcnt[c], lc = gdat[lp];
        gdat[p] = lc; gpos[lc] = p;
    };

    // Nearest Neighbor construction
    float cw = rx / G, ch = ry / G;
    float minCD = cw < ch ? cw : ch;

    tour[0] = 0;
    gridRm(0);

    for (int step = 1; step < N; step++) {
        int cur = tour[step - 1];
        float cx = px[cur], cy = py[cur];
        int cX, cY; cellXY(cx, cy, cX, cY);

        int best = -1;
        float bestD2 = 1e30f;

        for (int r = 0; r <= G; r++) {
            if (best >= 0 && r >= 2) {
                float rd = (float)(r - 1) * minCD;
                if (rd * rd > bestD2) break;
            }
            int x0 = cX - r, x1 = cX + r, y0 = cY - r, y1 = cY + r;
            if (x0 < 0) x0 = 0; if (x1 >= G) x1 = G - 1;
            if (y0 < 0) y0 = 0; if (y1 >= G) y1 = G - 1;
            for (int gy = y0; gy <= y1; gy++) {
                for (int gx = x0; gx <= x1; gx++) {
                    if (r > 0 && gx > x0 && gx < x1 && gy > y0 && gy < y1) continue;
                    int ci = gy * G + gx, end = goff[ci] + gcnt[ci];
                    for (int p = goff[ci]; p < end; p++) {
                        int j = gdat[p];
                        float dx = px[j] - cx, dy = py[j] - cy;
                        float d = dx * dx + dy * dy;
                        if (d < bestD2) { bestD2 = d; best = j; }
                    }
                }
            }
            if (best >= 0 && r > 0) break;
        }
        tour[step] = best;
        gridRm(best);
    }

    // 2-opt with neighbor lists for moderate N
    if (N >= 4 && N <= 300000 && secs() < 200.0) {
        int* pos = new int[N];
        for (int i = 0; i < N; i++) pos[tour[i]] = i;

        // Rebuild grid for neighbor computation
        memset(gcnt.data(), 0, G2 * sizeof(int));
        for (int i = 0; i < N; i++) gcnt[cellOf(i)]++;
        goff[0] = 0;
        for (int c = 0; c < G2; c++) goff[c + 1] = goff[c] + gcnt[c];
        {
            vector<int> t(G2, 0);
            for (int i = 0; i < N; i++) {
                int c = cellOf(i);
                gdat[goff[c] + t[c]++] = i;
            }
        }

        int K = N <= 5000 ? 15 : (N <= 50000 ? 10 : 7);
        if (K >= N) K = N - 1;

        // Build K-nearest neighbor lists
        int* nn = new int[(size_t)N * K];
        for (int i = 0; i < N; i++) {
            int cX, cY; cellXY(px[i], py[i], cX, cY);
            vector<pair<float, int>> cands;
            cands.reserve(K * 8);
            for (int r = 0; r <= G; r++) {
                int x0 = cX - r, x1 = cX + r, y0 = cY - r, y1 = cY + r;
                if (x0 < 0) x0 = 0; if (x1 >= G) x1 = G - 1;
                if (y0 < 0) y0 = 0; if (y1 >= G) y1 = G - 1;
                for (int gy = y0; gy <= y1; gy++)
                    for (int gx = x0; gx <= x1; gx++) {
                        if (r > 0 && gx > x0 && gx < x1 && gy > y0 && gy < y1) continue;
                        int ci = gy * G + gx, end = goff[ci] + gcnt[ci];
                        for (int p = goff[ci]; p < end; p++) {
                            int j = gdat[p];
                            if (j != i) {
                                float dx = px[j] - px[i], dy = py[j] - py[i];
                                cands.push_back({dx * dx + dy * dy, j});
                            }
                        }
                    }
                if ((int)cands.size() >= K * 3 && r > 0) break;
            }
            int nk = min(K, (int)cands.size());
            partial_sort(cands.begin(), cands.begin() + nk, cands.end());
            for (int k = 0; k < nk; k++) nn[(size_t)i * K + k] = cands[k].second;
            for (int k = nk; k < K; k++) nn[(size_t)i * K + k] = -1;
        }

        double tLimit = N <= 5000 ? 200.0 : (N <= 50000 ? 120.0 : (N <= 100000 ? 60.0 : 30.0));

        bool improved = true;
        while (improved && secs() < tLimit) {
            improved = false;
            for (int ii = 0; ii < N; ii++) {
                int c1 = tour[ii];
                int inext = (ii + 1) % N;
                int c2 = tour[inext];
                float dc2val = dist2(c1, c2);

                for (int k = 0; k < K; k++) {
                    int nb = nn[(size_t)c1 * K + k];
                    if (nb < 0) break;
                    if (dist2(c1, nb) >= dc2val) break;

                    int j = pos[nb];
                    int a, b;
                    if (ii < j) { a = ii; b = j; }
                    else { a = j; b = ii; }
                    if (b - a <= 1 || (a == 0 && b == N - 1)) continue;

                    int bnext = (b + 1) % N;
                    float imp = calcDist(tour[a], tour[a + 1]) + calcDist(tour[b], tour[bnext])
                              - calcDist(tour[a], tour[b]) - calcDist(tour[a + 1], tour[bnext]);

                    if (imp > 1e-6f) {
                        int lo = a + 1, hi = b;
                        while (lo < hi) {
                            int tl = tour[lo], th = tour[hi];
                            tour[lo] = th; tour[hi] = tl;
                            pos[th] = lo; pos[tl] = hi;
                            lo++; hi--;
                        }
                        improved = true;
                        break;
                    }
                }
                if ((ii & 0x3FFF) == 0 && secs() > tLimit) break;
            }
        }

        // Or-opt: relocate single cities
        if (N <= 100000 && secs() < tLimit) {
            improved = true;
            while (improved && secs() < tLimit) {
                improved = false;
                for (int ii = 1; ii < N; ii++) {
                    int c = tour[ii];
                    int prevI = ii - 1;
                    int nextI = (ii + 1) % N;
                    int cp = tour[prevI], cn = tour[nextI];
                    float removeSave = calcDist(cp, c) + calcDist(c, cn) - calcDist(cp, cn);

                    for (int k = 0; k < K; k++) {
                        int nb = nn[(size_t)c * K + k];
                        if (nb < 0) break;
                        int j = pos[nb];
                        if (j == prevI || j == ii || j == nextI) continue;
                        int jnext = (j + 1) % N;
                        int jc = tour[jnext];
                        float insertCost = calcDist(nb, c) + calcDist(c, jc) - calcDist(nb, jc);
                        if (removeSave - insertCost > 1e-6f) {
                            if (ii < j) {
                                for (int s = ii; s < j; s++) {
                                    tour[s] = tour[s + 1];
                                    pos[tour[s]] = s;
                                }
                                tour[j] = c;
                                pos[c] = j;
                            } else {
                                for (int s = ii; s > j + 1; s--) {
                                    tour[s] = tour[s - 1];
                                    pos[tour[s]] = s;
                                }
                                tour[j + 1] = c;
                                pos[c] = j + 1;
                            }
                            improved = true;
                            break;
                        }
                    }
                    if ((ii & 0x3FFF) == 0 && secs() > tLimit) break;
                }
            }
        }

        delete[] pos;
        delete[] nn;
    }

    // Buffered output
    char* buf = new char[1 << 20];
    int bp = 0;
    auto flushBuf = [&]() { fwrite(buf, 1, bp, stdout); bp = 0; };
    auto writeInt = [&](int v) {
        if (bp > (1 << 20) - 20) flushBuf();
        if (v == 0) { buf[bp++] = '0'; return; }
        char tmp[12]; int len = 0;
        while (v > 0) { tmp[len++] = '0' + v % 10; v /= 10; }
        for (int i = len - 1; i >= 0; i--) buf[bp++] = tmp[i];
    };

    writeInt(tour[0]);
    for (int i = 1; i < N; i++) { buf[bp++] = ' '; writeInt(tour[i]); }
    buf[bp++] = '\n';
    flushBuf();

    delete[] tour;
    delete[] px;
    delete[] py;
    delete[] buf;
    return 0;
}
