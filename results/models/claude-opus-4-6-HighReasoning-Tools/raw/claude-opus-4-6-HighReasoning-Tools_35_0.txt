use std::io::{self, Read};
use std::time::Instant;

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut iter = input.split_whitespace();
    let n: usize = iter.next().unwrap().parse().unwrap();
    let m: usize = iter.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![vec![]; n];
    for _ in 0..m {
        let u: usize = iter.next().unwrap().parse().unwrap();
        let v: usize = iter.next().unwrap().parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
    }

    // Build adjacency bitsets for fast operations on smaller graphs
    let use_bitset = n <= 20_000;
    let words = (n + 63) / 64;
    let mut adj_bits: Vec<Vec<u64>> = Vec::new();
    let mut closed_bits: Vec<Vec<u64>> = Vec::new(); // N[v] union {v}

    if use_bitset {
        adj_bits = vec![vec![0u64; words]; n];
        closed_bits = vec![vec![0u64; words]; n];
        for v in 0..n {
            closed_bits[v][v / 64] |= 1u64 << (v % 64);
            for &u in &adj[v] {
                adj_bits[v][u / 64] |= 1u64 << (u % 64);
                closed_bits[v][u / 64] |= 1u64 << (u % 64);
            }
        }
    }

    let start = Instant::now();

    // --- Greedy phase: pick vertex covering most uncovered vertices ---
    let mut in_set = vec![false; n];
    let mut covered = vec![false; n];
    let mut cover_count: Vec<usize> = vec![0; n]; // how many times vertex is covered by set
    let mut num_covered: usize = 0;

    // degree array for greedy scoring
    let mut score: Vec<usize> = vec![0; n];
    for v in 0..n {
        // score = number of uncovered vertices in closed neighborhood
        score[v] = 1 + adj[v].len(); // initially all uncovered
    }

    while num_covered < n {
        // pick vertex with highest score
        let mut best = 0;
        let mut best_score = 0;
        for v in 0..n {
            if !in_set[v] && score[v] > best_score {
                best_score = score[v];
                best = v;
            }
        }
        // Add best to set
        in_set[best] = true;
        // Cover best and its neighbors
        if !covered[best] {
            covered[best] = true;
            num_covered += 1;
            cover_count[best] += 1;
            // update scores: best is now covered, so neighbors lose 1 from score
            for &u in &adj[best] {
                if score[u] > 0 { score[u] -= 1; }
            }
        } else {
            cover_count[best] += 1;
        }
        for &u in &adj[best] {
            if !covered[u] {
                covered[u] = true;
                num_covered += 1;
                cover_count[u] += 1;
                // update scores for neighbors of u
                for &w in &adj[u] {
                    if score[w] > 0 { score[w] -= 1; }
                }
            } else {
                cover_count[u] += 1;
            }
        }
        score[best] = 0;
    }

    let mut ds: Vec<usize> = (0..n).filter(|&v| in_set[v]).collect();

    // --- Local search improvement ---
    // Try to remove vertices from the set (redundancy removal)
    fn is_redundant(v: usize, in_set: &[bool], cover_count: &[usize], adj: &[Vec<usize>]) -> bool {
        // v is redundant if removing it still leaves all of closed_N(v) covered
        if cover_count[v] < 2 {
            // v itself would become uncovered unless another neighbor is in set
            // cover_count[v] counts how many set members dominate v (including v itself if in set)
            return false;
        }
        for &u in &adj[v] {
            if cover_count[u] < 2 {
                return false;
            }
        }
        true
    }

    // Remove redundant vertices (greedily, smallest degree first to maximize removals)
    let mut candidates: Vec<usize> = ds.clone();
    candidates.sort_by_key(|&v| adj[v].len());

    for &v in &candidates {
        if !in_set[v] { continue; }
        if is_redundant(v, &in_set, &cover_count, &adj) {
            in_set[v] = false;
            // update cover counts
            cover_count[v] -= 1;
            for &u in &adj[v] {
                cover_count[u] -= 1;
            }
        }
    }

    ds = (0..n).filter(|&v| in_set[v]).collect();

    // --- Swap-based local search (time-limited) ---
    // Try: for each vertex in DS, try removing it and adding a different vertex
    let time_limit_secs = 240.0; // 4 minutes max for local search
    let mut improved = true;

    while improved && start.elapsed().as_secs_f64() < time_limit_secs {
        improved = false;

        // Try removing each vertex in DS
        let current_ds: Vec<usize> = ds.clone();
        for &remove_v in &current_ds {
            if start.elapsed().as_secs_f64() > time_limit_secs { break; }

            // Temporarily remove remove_v
            in_set[remove_v] = false;
            cover_count[remove_v] -= 1;
            for &u in &adj[remove_v] {
                cover_count[u] -= 1;
            }

            // Find uncovered vertices
            let mut uncovered: Vec<usize> = Vec::new();
            if cover_count[remove_v] == 0 { uncovered.push(remove_v); }
            for &u in &adj[remove_v] {
                if cover_count[u] == 0 { uncovered.push(u); }
            }

            if uncovered.is_empty() {
                // remove_v was redundant (shouldn't happen after redundancy removal, but check)
                ds = (0..n).filter(|&v| in_set[v]).collect();
                improved = true;
                continue;
            }

            // Try to find a single vertex that covers all uncovered
            let mut found_replacement = false;

            if use_bitset && uncovered.len() <= 64 {
                // Use bitset intersection to find a vertex covering all uncovered
                // A vertex w covers uncovered vertex u if w == u or w is neighbor of u
                // So w must be in closed_N(u) for all u in uncovered
                // i.e., w is in intersection of closed_N(u) for all uncovered u
                let mut candidates_bits = vec![!0u64; words];
                // mask out bits beyond n
                if n % 64 != 0 {
                    candidates_bits[words - 1] = (1u64 << (n % 64)) - 1;
                }
                for &u in &uncovered {
                    for i in 0..words {
                        candidates_bits[i] &= closed_bits[u][i];
                    }
                }
                // Remove vertices already in set (we want a swap, not a no-op)
                // Actually we want vertices NOT in set (or remove_v itself)
                // Check each candidate
                for i in 0..words {
                    let mut bits = candidates_bits[i];
                    while bits != 0 {
                        let bit = bits.trailing_zeros() as usize;
                        let w = i * 64 + bit;
                        bits &= bits - 1;
                        if w >= n { break; }
                        if w != remove_v && !in_set[w] {
                            // Add w to set
                            in_set[w] = true;
                            cover_count[w] += 1;
                            for &u in &adj[w] {
                                cover_count[u] += 1;
                            }
                            // Now check if we can also remove another vertex (2-for-1 swap)
                            // For now just accept the 1-for-1 swap (net zero) only if
                            // it enables further redundancy removal
                            // Actually, just accepting remove without replacement if uncovered is empty
                            // was already handled. Here we have a 1-for-1 swap. Only useful if it
                            // enables further optimization. Skip for now - revert.
                            // Actually let's check: does this create new redundancies?
                            // For simplicity, revert and continue looking.
                            in_set[w] = false;
                            cover_count[w] -= 1;
                            for &u in &adj[w] {
                                cover_count[u] -= 1;
                            }
                            found_replacement = false; // we only want net reduction
                        }
                    }
                }
            }

            // If we can't find a good swap, restore remove_v
            if !found_replacement {
                in_set[remove_v] = true;
                cover_count[remove_v] += 1;
                for &u in &adj[remove_v] {
                    cover_count[u] += 1;
                }
            }
        }

        // Second pass: try 2-for-1 swaps (remove 2 vertices, add 1)
        if !improved && start.elapsed().as_secs_f64() < time_limit_secs && use_bitset {
            let current_ds: Vec<usize> = ds.clone();
            let ds_len = current_ds.len();
            'outer: for i in 0..ds_len {
                if start.elapsed().as_secs_f64() > time_limit_secs { break; }
                let v1 = current_ds[i];
                if !in_set[v1] { continue; }

                for j in (i+1)..ds_len {
                    if start.elapsed().as_secs_f64() > time_limit_secs { break 'outer; }
                    let v2 = current_ds[j];
                    if !in_set[v2] { continue; }

                    // Skip if v1 and v2 are far apart (heuristic: they should share neighborhoods)
                    // to avoid O(n^3) blowup
                    if ds_len > 500 { break 'outer; } // too many pairs

                    // Temporarily remove both
                    in_set[v1] = false;
                    in_set[v2] = false;
                    cover_count[v1] -= 1;
                    cover_count[v2] -= 1;
                    for &u in &adj[v1] { cover_count[u] -= 1; }
                    for &u in &adj[v2] { cover_count[u] -= 1; }

                    // Find uncovered vertices
                    let mut uncov_bits = vec![0u64; words];
                    let mut has_uncov = false;
                    // Check closed neighborhoods of v1 and v2
                    for &check in &[v1, v2] {
                        if cover_count[check] == 0 {
                            uncov_bits[check / 64] |= 1u64 << (check % 64);
                            has_uncov = true;
                        }
                        for &u in &adj[check] {
                            if cover_count[u] == 0 {
                                uncov_bits[u / 64] |= 1u64 << (u % 64);
                                has_uncov = true;
                            }
                        }
                    }

                    if has_uncov {
                        // Find intersection of closed neighborhoods of all uncovered
                        let mut cand_bits = vec![!0u64; words];
                        if n % 64 != 0 {
                            cand_bits[words - 1] = (1u64 << (n % 64)) - 1;
                        }
                        for wi in 0..words {
                            let mut bits = uncov_bits[wi];
                            while bits != 0 {
                                let bit = bits.trailing_zeros() as usize;
                                let u = wi * 64 + bit;
                                bits &= bits - 1;
                                if u >= n { break; }
                                for k in 0..words {
                                    cand_bits[k] &= closed_bits[u][k];
                                }
                            }
                        }

                        // Find a candidate not in set (and not v1 or v2)
                        let mut found = false;
                        'search: for wi in 0..words {
                            let mut bits = cand_bits[wi];
                            while bits != 0 {
                                let bit = bits.trailing_zeros() as usize;
                                let w = wi * 64 + bit;
                                bits &= bits - 1;
                                if w >= n { break; }
                                if !in_set[w] && w != v1 && w != v2 {
                                    // Add w, verify all covered
                                    in_set[w] = true;
                                    cover_count[w] += 1;
                                    for &u in &adj[w] { cover_count[u] += 1; }

                                    // Verify
                                    let mut valid = true;
                                    if cover_count[v1] == 0 { valid = false; }
                                    if valid && cover_count[v2] == 0 { valid = false; }
                                    if valid {
                                        for &u in &adj[v1] {
                                            if cover_count[u] == 0 { valid = false; break; }
                                        }
                                    }
                                    if valid {
                                        for &u in &adj[v2] {
                                            if cover_count[u] == 0 { valid = false; break; }
                                        }
                                    }

                                    if valid {
                                        found = true;
                                        ds = (0..n).filter(|&v| in_set[v]).collect();
                                        improved = true;
                                        break 'search;
                                    } else {
                                        in_set[w] = false;
                                        cover_count[w] -= 1;
                                        for &u in &adj[w] { cover_count[u] -= 1; }
                                    }
                                }
                            }
                        }

                        if found { break 'outer; }
                    }

                    // Restore
                    in_set[v1] = true;
                    in_set[v2] = true;
                    cover_count[v1] += 1;
                    cover_count[v2] += 1;
                    for &u in &adj[v1] { cover_count[u] += 1; }
                    for &u in &adj[v2] { cover_count[u] += 1; }
                }
            }
        }
    }

    // Final redundancy pass
    let mut candidates: Vec<usize> = ds.clone();
    candidates.sort_by_key(|&v| adj[v].len());
    for &v in &candidates {
        if !in_set[v] { continue; }
        if is_redundant(v, &in_set, &cover_count, &adj) {
            in_set[v] = false;
            cover_count[v] -= 1;
            for &u in &adj[v] {
                cover_count[u] -= 1;
            }
        }
    }

    ds = (0..n).filter(|&v| in_set[v]).collect();

    // Output
    println!("{}", ds.len());
    let parts: Vec<String> = ds.iter().map(|v| v.to_string()).collect();
    println!("{}", parts.join(" "));
}
