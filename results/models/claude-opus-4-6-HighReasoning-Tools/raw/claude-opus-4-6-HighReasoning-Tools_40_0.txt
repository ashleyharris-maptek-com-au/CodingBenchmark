#include <bits/stdc++.h>
using namespace std;

int n, m;
vector<long long> c;
vector<vector<int>> adj;

long long best_obj;
vector<int> best_sol;

// state: 0=free, 1=included, 2=excluded
// free_pos_sum: sum of positive weights among free variables
void solve(vector<int>& state, long long cur_obj, long long free_pos_sum) {
    if (cur_obj + free_pos_sum <= best_obj) return;

    // Pick free variable with highest positive weight
    int next = -1;
    for (int i = 0; i < n; i++) {
        if (state[i] == 0 && c[i] > 0) {
            if (next == -1 || c[i] > c[next]) next = i;
        }
    }

    if (next == -1) {
        if (cur_obj > best_obj) {
            best_obj = cur_obj;
            for (int i = 0; i < n; i++)
                best_sol[i] = (state[i] == 1) ? 1 : 0;
        }
        return;
    }

    // Include next: exclude its neighbors
    {
        state[next] = 1;
        long long removed = c[next] > 0 ? c[next] : 0;
        vector<int> exc;
        for (int v : adj[next]) {
            if (state[v] == 0) {
                state[v] = 2;
                exc.push_back(v);
                if (c[v] > 0) removed += c[v];
            }
        }
        solve(state, cur_obj + c[next], free_pos_sum - removed);
        for (int v : exc) state[v] = 0;
        state[next] = 0;
    }

    // Exclude next
    {
        state[next] = 2;
        long long removed = c[next] > 0 ? c[next] : 0;
        solve(state, cur_obj, free_pos_sum - removed);
        state[next] = 0;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;
    c.resize(n);
    adj.resize(n);
    for (int i = 0; i < n; i++) cin >> c[i];
    for (int i = 0; i < m; i++) {
        int u, v; cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    best_sol.assign(n, 0);
    best_obj = 0;

    // Greedy initial solution for a good lower bound
    {
        vector<int> order(n);
        iota(order.begin(), order.end(), 0);
        sort(order.begin(), order.end(), [](int a, int b){ return a < b; });
        sort(order.begin(), order.end(), [&](int a, int b){ return c[a] > c[b]; });
        vector<bool> blocked(n, false);
        long long obj = 0;
        for (int i : order) {
            if (c[i] <= 0) break;
            if (!blocked[i]) {
                best_sol[i] = 1;
                obj += c[i];
                for (int v : adj[i]) blocked[v] = true;
            }
        }
        best_obj = obj;
    }

    // Pre-exclude non-positive weight variables
    vector<int> state(n, 0);
    long long free_pos_sum = 0;
    for (int i = 0; i < n; i++) {
        if (c[i] <= 0) state[i] = 2;
        else free_pos_sum += c[i];
    }

    solve(state, 0, free_pos_sum);

    cout << best_obj << "\n";
    for (int i = 0; i < n; i++) {
        if (i) cout << " ";
        cout << best_sol[i];
    }
    cout << "\n";
}
