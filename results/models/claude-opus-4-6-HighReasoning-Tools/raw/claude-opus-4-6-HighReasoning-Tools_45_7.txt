import struct

def float_bits(f):
    return struct.unpack('<I', struct.pack('<f', f))[0]

def encode_str(s):
    b = s.encode('utf-8') + b'\x00'
    while len(b) % 4 != 0:
        b += b'\x00'
    return [struct.unpack('<I', b[i:i+4])[0] for i in range(0, len(b), 4)]

class SPIRVBuilder:
    def __init__(self):
        self.instructions = []
        self.next_id = 1

    def id(self):
        r = self.next_id
        self.next_id += 1
        return r

    def emit(self, opcode, *operands):
        wc = 1 + len(operands)
        self.instructions.append(((wc << 16) | opcode, list(operands)))

    def emit_str(self, opcode, pre, s, post=None):
        if post is None:
            post = []
        str_words = encode_str(s)
        ops = list(pre) + str_words + list(post)
        wc = 1 + len(ops)
        self.instructions.append(((wc << 16) | opcode, ops))

    def build(self):
        words = []
        words.append(0x07230203)  # magic
        words.append(0x00010000)  # version 1.0
        words.append(0)           # generator
        words.append(self.next_id)  # bound
        words.append(0)           # reserved
        for header, ops in self.instructions:
            words.append(header)
            words.extend(ops)
        return b''.join(struct.pack('<I', w) for w in words)
