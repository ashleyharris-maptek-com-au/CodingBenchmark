use std::f64::consts::PI;
use std::io::{self, BufRead, BufReader, BufWriter, Write};
use std::sync::mpsc;
use std::thread;
use std::time::Duration;

struct Config {
    gravity: f64,
    max_thrust: f64,
    target_x: f64,
    target_y: f64,
}

struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

fn normalize_angle(a: f64) -> f64 {
    (a + PI).rem_euclid(2.0 * PI) - PI
}

fn compute_control(s: &State, c: &Config) -> (f64, f64) {
    if s.fuel <= 0.001 {
        return (0.0, 0.0);
    }

    let dx = c.target_x - s.x;
    let dy = c.target_y - s.y;
    let g = c.gravity;
    let mt = c.max_thrust;
    let abs_dx = dx.abs();

    let vert_decel = (mt - g).max(mt * 0.05);
    let horiz_accel_cap = mt * 0.25;

    // Horizontal velocity target using sqrt-law deceleration profile
    let vx_target = if abs_dx > 3.0 {
        let max_vx = (2.0 * horiz_accel_cap * 0.4 * abs_dx).sqrt().min(12.0);
        max_vx * dx.signum()
    } else if abs_dx > 0.5 {
        let max_vx = (2.0 * horiz_accel_cap * 0.3 * abs_dx).sqrt().min(2.0);
        max_vx * dx.signum()
    } else {
        dx * 0.4
    };

    // Vertical velocity target: safe descent speed based on altitude above target
    let mut vy_target = if dy > 30.0 {
        (2.0 * vert_decel * 0.35 * dy).sqrt().min(12.0)
    } else if dy > 10.0 {
        (2.0 * vert_decel * 0.3 * dy).sqrt().min(5.0)
    } else if dy > 3.0 {
        (2.0 * vert_decel * 0.2 * dy).sqrt().min(2.0)
    } else if dy > 0.5 {
        (2.0 * vert_decel * 0.15 * dy).sqrt().min(1.2)
    } else if dy > 0.0 {
        0.3
    } else {
        (dy * 0.5).max(-4.0)
    };

    // Hold altitude until horizontally aligned
    if abs_dx > 8.0 && vy_target > 0.5 {
        vy_target = 0.5;
    } else if abs_dx > 3.0 && vy_target > 1.5 {
        vy_target = 1.5;
    }

    // PD velocity controller
    let kp_vel = 2.5;
    let ax_desired = kp_vel * (vx_target - s.vx);
    let ay_desired = kp_vel * (vy_target - s.vy);

    // Convert desired acceleration to engine angle and thrust
    // Engine produces: ax = mt*thrust*sin(angle), ay = -mt*thrust*cos(angle) + gravity
    let fx = ax_desired;
    let fy = ay_desired - g;

    let desired_angle = fx.atan2(-fy);
    let thrust_mag = (fx * fx + fy * fy).sqrt() / mt;
    let mut thrust = thrust_mag.clamp(0.0, 1.0);

    // Angle PD controller
    let angle_err = normalize_angle(desired_angle - s.angle);
    let kp_a = 5.0;
    let kd_a = 2.5;
    let turn = (kp_a * angle_err - kd_a * s.angular_vel).clamp(-1.0, 1.0);

    // Attenuate thrust when misaligned to avoid flying in the wrong direction
    if angle_err.abs() > 1.0 {
        thrust *= 0.1;
    } else if angle_err.abs() > 0.5 {
        thrust *= 0.4;
    } else if angle_err.abs() > 0.3 {
        thrust *= 0.7;
    }

    (thrust, turn)
}

fn main() {
    let (config_tx, config_rx) = mpsc::channel::<Config>();
    let (state_tx, state_rx) = mpsc::channel::<State>();

    // Reader thread: parses header, consumes map, streams state updates
    thread::spawn(move || {
        let stdin = io::stdin();
        let mut reader = BufReader::with_capacity(1 << 20, stdin.lock());
        let mut line = String::new();

        reader.read_line(&mut line).unwrap();
        let h: Vec<f64> = line
            .trim()
            .split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        let height = h[1] as usize;

        line.clear();
        reader.read_line(&mut line).unwrap();
        let p: Vec<f64> = line
            .trim()
            .split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();

        let _ = config_tx.send(Config {
            gravity: h[2],
            max_thrust: h[3],
            target_x: p[2],
            target_y: p[3],
        });

        // Consume map rows efficiently
        for _ in 0..height {
            line.clear();
            let _ = reader.read_line(&mut line);
        }

        // Stream state updates to control thread
        loop {
            line.clear();
            match reader.read_line(&mut line) {
                Ok(0) | Err(_) => break,
                _ => {}
            }
            let trimmed = line.trim();
            if trimmed.is_empty() {
                continue;
            }
            let vals: Vec<f64> = trimmed
                .split_whitespace()
                .filter_map(|s| s.parse().ok())
                .collect();
            if vals.len() >= 7 {
                let _ = state_tx.send(State {
                    x: vals[0],
                    y: vals[1],
                    vx: vals[2],
                    vy: vals[3],
                    angle: vals[4],
                    angular_vel: vals[5],
                    fuel: vals[6],
                });
            }
        }
    });

    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());

    // Output idle commands while waiting for config (during map parsing)
    let config = loop {
        match config_rx.try_recv() {
            Ok(c) => break c,
            Err(mpsc::TryRecvError::Empty) => {
                let _ = writeln!(writer, "0.0 0.0");
                let _ = writer.flush();
                thread::sleep(Duration::from_millis(20));
            }
            Err(_) => return,
        }
    };

    let mut current_thrust = 0.0_f64;
    let mut current_turn = 0.0_f64;

    // Main control loop at ~65Hz (well above the 20Hz minimum)
    loop {
        // Drain all pending state updates, keeping the latest
        loop {
            match state_rx.try_recv() {
                Ok(s) => {
                    let (t, tu) = compute_control(&s, &config);
                    current_thrust = t;
                    current_turn = tu;
                }
                Err(mpsc::TryRecvError::Empty) => break,
                Err(_) => return,
            }
        }

        let _ = writeln!(writer, "{:.4} {:.4}", current_thrust, current_turn);
        let _ = writer.flush();
        thread::sleep(Duration::from_millis(15));
    }
}
