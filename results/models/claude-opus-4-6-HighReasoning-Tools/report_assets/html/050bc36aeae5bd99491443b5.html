using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Numerics;
using System.Text;
using System.Threading;
using System.Threading.Tasks;

class Program
{
    static int N, M;
    static int[][] Adj;
    static Stopwatch SW;
    static volatile int GBest;
    static int[] GBestClique;
    static readonly object Lk = new object();
    static volatile bool Stop;
    const long TL = 270_000;
    static byte[] buf;
    static int bp;

    static void Main()
    {
        SW = Stopwatch.StartNew();
        ReadInput();
        if (N == 0) { Console.Write("0\n\n"); return; }
        if (M == 0) { Console.Write("1\n0\n"); return; }

        int[] order = new int[N], pos = new int[N];
        Degeneracy(order, pos);

        GBestClique = Greedy(order);
        GBest = GBestClique.Length;

        int cores = Math.Max(1, Environment.ProcessorCount);
        Parallel.For(0, N, new ParallelOptions { MaxDegreeOfParallelism = cores }, (rawIdx) =>
        {
            if (Stop) return;
            int i = N - 1 - rawIdx;
            if (i < 0) return;
            int v = order[i];

            var av = Adj[v];
            int rc = 0;
            for (int k = 0; k < av.Length; k++)
                if (pos[av[k]] > i) rc++;
            if (rc + 1 <= GBest) return;

            int[] rn = new int[rc];
            int ri = 0;
            for (int k = 0; k < av.Length; k++)
                if (pos[av[k]] > i) rn[ri++] = av[k];

            int sn = rn.Length;
            var vm = new Dictionary<int, int>(sn * 2);
            for (int j = 0; j < sn; j++) vm[rn[j]] = j;

            int words = (sn + 63) >> 6;
            var la = new ulong[sn][];
            for (int j = 0; j < sn; j++) la[j] = new ulong[words];

            for (int j = 0; j < sn; j++)
            {
                int u = rn[j];
                var au = Adj[u];
                for (int k = 0; k < au.Length; k++)
                {
                    if (vm.TryGetValue(au[k], out int wi) && wi > j)
                    {
                        la[j][wi >> 6] |= 1UL << (wi & 63);
                        la[wi][j >> 6] |= 1UL << (j & 63);
                    }
                }
            }

            ulong[] P = new ulong[words];
            for (int j = 0; j < sn; j++) P[j >> 6] |= 1UL << (j & 63);

            int[] cl = new int[sn];
            int[] lb = null;
            int lbl = GBest - 1;
            int nc = 0;

            Search(la, P, sn, cl, 0, ref lb, ref lbl, words, ref nc);

            if (lb != null)
            {
                var full = new int[lb.Length + 1];
                full[0] = v;
                for (int j = 0; j < lb.Length; j++) full[j + 1] = rn[lb[j]];
                lock (Lk)
                {
                    if (full.Length > GBest)
                    {
                        GBestClique = full;
                        GBest = full.Length;
                    }
                }
            }
        });

        var sb = new StringBuilder();
        sb.Append(GBest).Append('\n');
        for (int i = 0; i < GBestClique.Length; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(GBestClique[i]);
        }
        sb.Append('\n');
        Console.Write(sb.ToString());
    }

    static void Search(ulong[][] la, ulong[] P, int pl, int[] cl, int cLen,
        ref int[] best, ref int bl, int w, ref int nc)
    {
        if (Stop) return;
        if (++nc % 8192 == 0)
        {
            if (SW.ElapsedMilliseconds > TL) { Stop = true; return; }
            int gb = GBest;
            if (gb - 1 > bl) bl = gb - 1;
        }

        if (cLen + pl <= bl) return;

        if (pl == 0)
        {
            if (cLen > bl)
            {
                bl = cLen;
                best = new int[cLen];
                Array.Copy(cl, best, cLen);
            }
            return;
        }

        int[] vs = new int[pl];
        int vi = 0;
        for (int i = 0; i < w && vi < pl; i++)
        {
            ulong b = P[i];
            while (b != 0)
            {
                int bit = BitOperations.TrailingZeroCount(b);
                vs[vi++] = (i << 6) | bit;
                b &= b - 1;
            }
        }

        int[] col = new int[pl];
        int mc = 0;
        int[] used = new int[pl + 2];
        int marker = 0;

        for (int i = 0; i < pl; i++)
        {
            marker++;
            int v = vs[i];
            for (int j = 0; j < i; j++)
            {
                int u = vs[j];
                if ((la[v][u >> 6] & (1UL << (u & 63))) != 0)
                    used[col[j]] = marker;
            }
            int c = 1;
            while (used[c] == marker) c++;
            col[i] = c;
            if (c > mc) mc = c;
        }

        if (cLen + mc <= bl) return;

        Array.Sort(col, vs, 0, pl);

        ulong[] cp = new ulong[w];
        Array.Copy(P, cp, w);

        for (int ii = pl - 1; ii >= 0; ii--)
        {
            if (Stop) return;
            int v = vs[ii];
            int c = col[ii];
            if (cLen + c <= bl) return;

            ulong[] np = new ulong[w];
            int npl = 0;
            for (int i = 0; i < w; i++)
            {
                np[i] = cp[i] & la[v][i];
                npl += BitOperations.PopCount(np[i]);
            }

            cl[cLen] = v;
            Search(la, np, npl, cl, cLen + 1, ref best, ref bl, w, ref nc);

            cp[v >> 6] &= ~(1UL << (v & 63));
        }
    }

    static void ReadInput()
    {
        using var ms = new MemoryStream();
        using var si = Console.OpenStandardInput();
        si.CopyTo(ms);
        buf = ms.ToArray();
        bp = 0;
        N = RI(); M = RI();
        var al = new List<int>[N];
        for (int i = 0; i < N; i++) al[i] = new List<int>();
        for (int i = 0; i < M; i++)
        {
            int u = RI(), v = RI();
            al[u].Add(v); al[v].Add(u);
        }
        Adj = new int[N][];
        for (int i = 0; i < N; i++)
        {
            al[i].Sort();
            Adj[i] = al[i].ToArray();
        }
    }

    static int RI()
    {
        while (bp < buf.Length && buf[bp] <= (byte)' ') bp++;
        int r = 0;
        while (bp < buf.Length && buf[bp] >= (byte)'0' && buf[bp] <= (byte)'9')
            r = r * 10 + (buf[bp++] - '0');
        return r;
    }

    static void Degeneracy(int[] order, int[] pos)
    {
        int[] deg = new int[N];
        for (int i = 0; i < N; i++) deg[i] = Adj[i].Length;
        int md = 0;
        for (int i = 0; i < N; i++) if (deg[i] > md) md = deg[i];

        int[] hd = new int[md + 2];
        int[] nx = new int[N], pv = new int[N];
        Array.Fill(hd, -1);
        for (int i = N - 1; i >= 0; i--)
        {
            nx[i] = hd[deg[i]];
            pv[i] = -1;
            if (hd[deg[i]] != -1) pv[hd[deg[i]]] = i;
            hd[deg[i]] = i;
        }

        bool[] rm = new bool[N];
        int cur = 0;
        for (int it = 0; it < N; it++)
        {
            while (cur <= md && hd[cur] == -1) cur++;
            int v = hd[cur];
            hd[cur] = nx[v];
            if (nx[v] != -1) pv[nx[v]] = -1;
            rm[v] = true;
            order[it] = v;
            pos[v] = it;

            var av = Adj[v];
            for (int k = 0; k < av.Length; k++)
            {
                int u = av[k];
                if (rm[u]) continue;
                int od = deg[u];
                if (pv[u] == -1)
                {
                    hd[od] = nx[u];
                    if (nx[u] != -1) pv[nx[u]] = -1;
                }
                else
                {
                    nx[pv[u]] = nx[u];
                    if (nx[u] != -1) pv[nx[u]] = pv[u];
                }
                deg[u]--;
                nx[u] = hd[deg[u]];
                pv[u] = -1;
                if (hd[deg[u]] != -1) pv[hd[deg[u]]] = u;
                hd[deg[u]] = u;
                if (deg[u] < cur) cur = deg[u];
            }
        }
    }

    static int[] Greedy(int[] dOrder)
    {
        int[] deg = new int[N];
        for (int i = 0; i < N; i++) deg[i] = Adj[i].Length;
        int[] byDeg = new int[N];
        for (int i = 0; i < N; i++) byDeg[i] = i;
        Array.Sort(byDeg, (a, b) => deg[b].CompareTo(deg[a]));

        int[] best = { byDeg[0] };
        var starts = new HashSet<int>();
        int tries = Math.Min(N, 100);
        for (int t = 0; t < tries; t++) starts.Add(byDeg[t]);
        for (int t = 0; t < tries && N - 1 - t >= 0; t++) starts.Add(dOrder[N - 1 - t]);

        foreach (int start in starts)
        {
            if (SW.ElapsedMilliseconds > 5000) break;
            var clique = new List<int> { start };
            var cands = new List<int>(Adj[start]);
            cands.Sort((a, b) => deg[b].CompareTo(deg[a]));
            foreach (int v in cands)
            {
                bool ok = true;
                foreach (int u in clique)
                    if (Array.BinarySearch(Adj[v], u) < 0) { ok = false; break; }
                if (ok) clique.Add(v);
            }
            if (clique.Count > best.Length) best = clique.ToArray();
        }
        return best;
    }
