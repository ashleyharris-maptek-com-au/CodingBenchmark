    const double G = 6.67430e-11;

    static double EstimateEarthImpactDay()
    {
        double maxDays = WarningDays * 1.5 + 400;
        double dt = DAY * 0.5;
        int steps = (int)(maxDays * DAY / dt) + 1;

        double[][] pos = new double[NumBodies][];
        double[][] vel = new double[NumBodies][];
        for (int i = 0; i < NumBodies; i++)
        {
            pos[i] = (double[])BodyPos0[i].Clone();
            vel[i] = (double[])BodyVel0[i].Clone();
        }

        double minDist = double.MaxValue;
        double minDistDay = maxDays;

        double[][] acc = ComputeAccelerations(pos);

        for (int s = 0; s < steps; s++)
        {
            for (int i = 0; i < NumBodies; i++)
            {
                vel[i][0] += acc[i][0] * dt * 0.5;
                vel[i][1] += acc[i][1] * dt * 0.5;
                vel[i][2] += acc[i][2] * dt * 0.5;
            }
            for (int i = 0; i < NumBodies; i++)
            {
                pos[i][0] += vel[i][0] * dt;
                pos[i][1] += vel[i][1] * dt;
                pos[i][2] += vel[i][2] * dt;
            }
            acc = ComputeAccelerations(pos);
            for (int i = 0; i < NumBodies; i++)
            {
                vel[i][0] += acc[i][0] * dt * 0.5;
                vel[i][1] += acc[i][1] * dt * 0.5;
                vel[i][2] += acc[i][2] * dt * 0.5;
            }

            double dx = pos[AsteroidIdx][0] - pos[EarthIdx][0];
            double dy = pos[AsteroidIdx][1] - pos[EarthIdx][1];
            double dz = pos[AsteroidIdx][2] - pos[EarthIdx][2];
            double dist = Math.Sqrt(dx * dx + dy * dy + dz * dz);

            double dayNow = (s + 1) * dt / DAY;
            if (dist < minDist)
            {
                minDist = dist;
                minDistDay = dayNow;
            }
        }

        return minDistDay;
    }

    static double[][] ComputeAccelerations(double[][] pos)
    {
        double[][] acc = new double[NumBodies][];
        for (int i = 0; i < NumBodies; i++)
            acc[i] = new double[3];

        for (int i = 0; i < NumBodies; i++)
        {
            for (int j = i + 1; j < NumBodies; j++)
            {
                double dx = pos[j][0] - pos[i][0];
                double dy = pos[j][1] - pos[i][1];
                double dz = pos[j][2] - pos[i][2];
                double r2 = dx * dx + dy * dy + dz * dz;
                double r = Math.Sqrt(r2);
                double r3 = r * r2;

                double fx = dx / r3;
                double fy = dy / r3;
                double fz = dz / r3;

                acc[i][0] += GM[j] * fx;
                acc[i][1] += GM[j] * fy;
                acc[i][2] += GM[j] * fz;
                acc[j][0] -= GM[i] * fx;
                acc[j][1] -= GM[i] * fy;
                acc[j][2] -= GM[i] * fz;
            }
        }
        return acc;
    }

    struct BodyState
    {
        public double[][] Pos;
        public double[][] Vel;
    }

    static List<BodyState> PrecomputeEphemeris(double maxDays, double dtDays, out double actualDt)
    {
        actualDt = dtDays;
        double dt = dtDays * DAY;
        int steps = (int)(maxDays / dtDays) + 2;

        var states = new List<BodyState>(steps);

        double[][] pos = new double[NumBodies][];
        double[][] vel = new double[NumBodies][];
        for (int i = 0; i < NumBodies; i++)
        {
            pos[i] = (double[])BodyPos0[i].Clone();
            vel[i] = (double[])BodyVel0[i].Clone();
        }

        states.Add(CloneState(pos, vel));

        double[][] acc = ComputeAccelerations(pos);

        for (int s = 0; s < steps - 1; s++)
        {
            for (int i = 0; i < NumBodies; i++)
            {
                vel[i][0] += acc[i][0] * dt * 0.5;
                vel[i][1] += acc[i][1] * dt * 0.5;
                vel[i][2] += acc[i][2] * dt * 0.5;
            }
            for (int i = 0; i < NumBodies; i++)
            {
                pos[i][0] += vel[i][0] * dt;
                pos[i][1] += vel[i][1] * dt;
                pos[i][2] += vel[i][2] * dt;
            }
            acc = ComputeAccelerations(pos);
            for (int i = 0; i < NumBodies; i++)
            {
                vel[i][0] += acc[i][0] * dt * 0.5;
                vel[i][1] += acc[i][1] * dt * 0.5;
                vel[i][2] += acc[i][2] * dt * 0.5;
            }
            states.Add(CloneState(pos, vel));
        }

        return states;
    }

    static BodyState CloneState(double[][] pos, double[][] vel)
    {
        var s = new BodyState();
        s.Pos = new double[NumBodies][];
        s.Vel = new double[NumBodies][];
        for (int i = 0; i < NumBodies; i++)
        {
            s.Pos[i] = (double[])pos[i].Clone();
            s.Vel[i] = (double[])vel[i].Clone();
        }
        return s;
    }

    static void GetBodyStateAtDay(List<BodyState> ephemeris, double dtDays, double day, int bodyIdx,
        out double px, out double py, out double pz, out double vx, out double vy, out double vz)
    {
        double idx = day / dtDays;
        int i0 = (int)idx;
        double frac = idx - i0;

        if (i0 < 0) i0 = 0;
        if (i0 >= ephemeris.Count - 1)
        {
            i0 = ephemeris.Count - 1;
            frac = 0;
        }
        int i1 = Math.Min(i0 + 1, ephemeris.Count - 1);

        px = ephemeris[i0].Pos[bodyIdx][0] * (1 - frac) + ephemeris[i1].Pos[bodyIdx][0] * frac;
        py = ephemeris[i0].Pos[bodyIdx][1] * (1 - frac) + ephemeris[i1].Pos[bodyIdx][1] * frac;
        pz = ephemeris[i0].Pos[bodyIdx][2] * (1 - frac) + ephemeris[i1].Pos[bodyIdx][2] * frac;
        vx = ephemeris[i0].Vel[bodyIdx][0] * (1 - frac) + ephemeris[i1].Vel[bodyIdx][0] * frac;
        vy = ephemeris[i0].Vel[bodyIdx][1] * (1 - frac) + ephemeris[i1].Vel[bodyIdx][1] * frac;
        vz = ephemeris[i0].Vel[bodyIdx][2] * (1 - frac) + ephemeris[i1].Vel[bodyIdx][2] * frac;
    }

    static (double launchDay, double arrivalDay, double[] launchDv, double transferDays) FindBestTransfer(double impactDay)
    {
        double ephemDt = 0.5;
        double maxEphemDays = impactDay + 100;
        double actualDt;
        var ephemeris = PrecomputeEphemeris(maxEphemDays, ephemDt, out actualDt);

        double arrivalDeadline = impactDay - 1.0;
        if (arrivalDeadline < 10) arrivalDeadline = impactDay - 0.1;

        double bestCost = double.MaxValue;
        double bestLaunchDay = -1, bestArrivalDay = -1;
        double[] bestLaunchDv = null;

        int launchSteps, arrivalSteps;

        if (WarningDays < 800)
        {
            launchSteps = 60;
            arrivalSteps = 60;
        }
        else if (WarningDays < 5000)
        {
            launchSteps = 80;
            arrivalSteps = 80;
        }
        else
        {
            launchSteps = 100;
            arrivalSteps = 100;
        }

        double minLaunch = 1.0;
        double maxLaunch = WarningDays * 0.85;
        if (maxLaunch < 30) maxLaunch = WarningDays * 0.95;

        double minArrival = minLaunch + 30;
        double maxArrival = arrivalDeadline;

        double minTransferDays = 20;
        if (WarningDays > 3000) minTransferDays = 60;

        double dLaunch = (maxLaunch - minLaunch) / launchSteps;
        double dArrival = (maxArrival - minArrival) / arrivalSteps;

        double muSun = SunIdx >= 0 ? GM[SunIdx] : GM[0];

        for (int il = 0; il <= launchSteps; il++)
        {
            double launchDay = minLaunch + il * dLaunch;
            if (launchDay < 0 || launchDay > WarningDays) continue;

            GetBodyStateAtDay(ephemeris, actualDt, launchDay, EarthIdx,
                out double ex, out double ey, out double ez,
                out double evx, out double evy, out double evz);

            for (int ia = 0; ia <= arrivalSteps; ia++)
            {
                double arrivalDay = minArrival + ia * dArrival;
                double tof = arrivalDay - launchDay;
                if (tof < minTransferDays) continue;
                if (arrivalDay > arrivalDeadline) continue;

                GetBodyStateAtDay(ephemeris, actualDt, arrivalDay, AsteroidIdx,
                    out double ax, out double ay, out double az,
                    out double avx, out double avy, out double avz);

                double tofSec = tof * DAY;

                double[] v1, v2;
                bool ok = LambertSolve(muSun, ex, ey, ez, ax, ay, az, tofSec, out v1, out v2);
                if (!ok) continue;

                double dvx = v1[0] - evx;
                double dvy = v1[1] - evy;
                double dvz = v1[2] - evz;
                double dvLaunch = Vec3Mag(dvx, dvy, dvz);

                double impVx = v2[0] - avx;
                double impVy = v2[1] - avy;
                double impVz = v2[2] - avz;
                double dvArr = Vec3Mag(impVx, impVy, impVz);

                double timeToImpact = impactDay - arrivalDay;
                double deflectionScore = dvArr * timeToImpact;

                double cost = dvLaunch - deflectionScore * 0.0001;

                if (dvLaunch < DvBudget * 0.95 && cost < bestCost)
                {
                    bestCost = cost;
                    bestLaunchDay = launchDay;
                    bestArrivalDay = arrivalDay;
                    bestLaunchDv = new double[] { dvx, dvy, dvz };
                }
            }
        }

        if (bestLaunchDv == null)
        {
            bestCost = double.MaxValue;
            for (int il = 0; il <= launchSteps * 2; il++)
            {
                double launchDay = minLaunch + il * dLaunch * 0.5;
                if (launchDay < 0 || launchDay > WarningDays) continue;

                GetBodyStateAtDay(ephemeris, actualDt, launchDay, EarthIdx,
                    out double ex, out double ey, out double ez,
                    out double evx, out double evy, out double evz);

                for (int ia = 0; ia <= arrivalSteps * 2; ia++)
                {
                    double arrivalDay = minArrival + ia * dArrival * 0.5;
                    double tof = arrivalDay - launchDay;
                    if (tof < 10) continue;
                    if (arrivalDay > impactDay) continue;

                    GetBodyStateAtDay(ephemeris, actualDt, arrivalDay, AsteroidIdx,
                        out double ax, out double ay, out double az,
                        out double avx, out double avy, out double avz);

                    double tofSec = tof * DAY;
                    double[] v1, v2;
                    bool ok = LambertSolve(muSun, ex, ey, ez, ax, ay, az, tofSec, out v1, out v2);
                    if (!ok) continue;

                    double dvx = v1[0] - evx;
                    double dvy = v1[1] - evy;
                    double dvz = v1[2] - evz;
                    double dvLaunch = Vec3Mag(dvx, dvy, dvz);

                    if (dvLaunch < DvBudget * 1.0 && dvLaunch < bestCost)
                    {
                        bestCost = dvLaunch;
                        bestLaunchDay = launchDay;
                        bestArrivalDay = arrivalDay;
                        bestLaunchDv = new double[] { dvx, dvy, dvz };
                    }
                }
            }
        }

        if (bestLaunchDv == null)
        {
            bestLaunchDay = Math.Min(30, WarningDays * 0.1);
            bestArrivalDay = impactDay - 10;
            double tof = bestArrivalDay - bestLaunchDay;

            GetBodyStateAtDay(ephemeris, actualDt, bestLaunchDay, EarthIdx,
                out double ex, out double ey, out double ez,
                out double evx, out double evy, out double evz);
            GetBodyStateAtDay(ephemeris, actualDt, bestArrivalDay, AsteroidIdx,
                out double ax, out double ay, out double az,
                out double avx, out double avy, out double avz);

            double dx = ax - ex, dy = ay - ey, dz = az - ez;
            double dist = Vec3Mag(dx, dy, dz);
            double speed = dist / (tof * DAY);
            bestLaunchDv = new double[] { dx / dist * speed - evx + evx, dy / dist * speed, dz / dist * speed };

            double mag = Vec3Mag(bestLaunchDv[0], bestLaunchDv[1], bestLaunchDv[2]);
            if (mag > DvBudget * 0.8)
            {
                double sc = DvBudget * 0.8 / mag;
                bestLaunchDv[0] *= sc;
                bestLaunchDv[1] *= sc;
                bestLaunchDv[2] *= sc;
            }
        }

        bestLaunchDay = RefineTransfer(ephemeris, actualDt, bestLaunchDay, bestArrivalDay, impactDay, ref bestLaunchDv);

        return (bestLaunchDay, bestArrivalDay, bestLaunchDv, bestArrivalDay - bestLaunchDay);
    }

    static double RefineTransfer(List<BodyState> ephemeris, double ephemDt,
        double launchDay, double arrivalDay, double impactDay, ref double[] launchDv)
    {
        double muSun = SunIdx >= 0 ? GM[SunIdx] : GM[0];
        double bestDv = Vec3Mag(launchDv[0], launchDv[1], launchDv[2]);
        double bestLaunch = launchDay;
        double bestArr = arrivalDay;
        double[] bestDvVec = (double[])launchDv.Clone();

        for (int pass = 0; pass < 3; pass++)
        {
            double range = (pass == 0) ? 20 : (pass == 1) ? 5 : 1;
            int steps = 20;
            double dlStep = range * 2.0 / steps;
            double daStep = range * 2.0 / steps;

            for (int il = 0; il <= steps; il++)
            {
                double ld = bestLaunch - range + il * dlStep;
                if (ld < 0.5 || ld > WarningDays) continue;

                GetBodyStateAtDay(ephemeris, ephemDt, ld, EarthIdx,
                    out double ex, out double ey, out double ez,
                    out double evx, out double evy, out double evz);

                for (int ia = 0; ia <= steps; ia++)
                {
                    double ad = bestArr - range + ia * daStep;
                    double tof = ad - ld;
                    if (tof < 10) continue;
                    if (ad > impactDay - 0.5) continue;

                    GetBodyStateAtDay(ephemeris, ephemDt, ad, AsteroidIdx,
                        out double ax, out double ay, out double az,
                        out double avx, out double avy, out double avz);

                    double[] v1, v2;
                    bool ok = LambertSolve(muSun, ex, ey, ez, ax, ay, az, tof * DAY, out v1, out v2);
                    if (!ok) continue;

                    double dvx = v1[0] - evx;
                    double dvy = v1[1] - evy;
                    double dvz = v1[2] - evz;
                    double dvMag = Vec3Mag(dvx, dvy, dvz);

                    double impVx = v2[0] - avx;
                    double impVy = v2[1] - avy;
                    double impVz = v2[2] - avz;
                    double impVMag = Vec3Mag(impVx, impVy, impVz);

                    double timeToImpact = impactDay - ad;
                    double score = dvMag - impVMag * timeToImpact * 0.00005;

                    double bestScore = bestDv - Vec3Mag(bestDvVec[0], bestDvVec[1], bestDvVec[2]) * 0.0;

                    if (dvMag <= DvBudget * 0.92 && score < bestDv - Vec3Mag(bestDvVec[0], bestDvVec[1], bestDvVec[2]) * 0.0 + 0.0001
                        || (dvMag < bestDv && dvMag <= DvBudget * 0.92))
                    {
                        bestDv = dvMag;
                        bestLaunch = ld;
                        bestArr = ad;
                        bestDvVec = new double[] { dvx, dvy, dvz };
                    }
                }
            }
        }

        launchDv = bestDvVec;
        return bestLaunch;
    }

    static List<(double, double, double, double)> PlanMission(double launchDay, double arrivalDay, double[] launchDv, double impactDay)
    {
        var burns = new List<(double, double, double, double)>();

        double launchDvMag = Vec3Mag(launchDv[0], launchDv[1], launchDv[2]);
        double dvRemaining = DvBudget;

        double launchUseDv = Math.Min(launchDvMag, dvRemaining * 0.85);
        double launchScale = launchUseDv / launchDvMag;
        double ldvx = launchDv[0] * launchScale;
        double ldvy = launchDv[1] * launchScale;
        double ldvz = launchDv[2] * launchScale;

        burns.Add((launchDay, ldvx, ldvy, ldvz));
        dvRemaining -= Vec3Mag(ldvx, ldvy, ldvz);

        double ephemDt = 0.5;
        double maxEphemDays = impactDay + 100;
        double actualDt;
        var ephemeris = PrecomputeEphemeris(maxEphemDays, ephemDt, out actualDt);

        GetBodyStateAtDay(ephemeris, actualDt, launchDay, EarthIdx,
            out double scPx, out double scPy, out double scPz,
            out double scVx, out double scVy, out double scVz);

        scVx += ldvx;
        scVy += ldvy;
        scVz += ldvz;

        double midDay = launchDay + (arrivalDay - launchDay) * 0.5;
        if (dvRemaining > 10.0)
        {
            double dtProp = 3600.0;
            int stepsToMid = (int)((midDay - launchDay) * DAY / dtProp);
            double muSun = SunIdx >= 0 ? GM[SunIdx] : GM[0];

            double px = scPx, py = scPy, pz = scPz;
            double vx = scVx, vy = scVy, vz = scVz;

            for (int s = 0; s < stepsToMid; s++)
            {
                double dayNow = launchDay + s * dtProp / DAY;
                double ax = 0, ay = 0, az = 0;
                for (int b = 0; b < NumBodies; b++)
                {
                    GetBodyStateAtDay(ephemeris, actualDt, dayNow, b,
                        out double bpx, out double bpy, out double bpz,
                        out double bvx, out double bvy, out double bvz);
                    double dx = bpx - px, dy = bpy - py, dz = bpz - pz;
                    double r2 = dx * dx + dy * dy + dz * dz;
                    double r = Math.Sqrt(r2);
                    double f = GM[b] / (r * r2);
                    ax += f * dx; ay += f * dy; az += f * dz;
                }
                vx += ax * dtProp * 0.5; vy += ay * dtProp * 0.5; vz += az * dtProp * 0.5;
                px += vx * dtProp; py += vy * dtProp; pz += vz * dtProp;
                ax = 0; ay = 0; az = 0;
                double dayNow2 = launchDay + (s + 1) * dtProp / DAY;
                for (int b = 0; b < NumBodies; b++)
                {
                    GetBodyStateAtDay(ephemeris, actualDt, dayNow2, b,
                        out double bpx, out double bpy, out double bpz,
                        out double bvx, out double bvy, out double bvz);
                    double dx = bpx - px, dy = bpy - py, dz = bpz - pz;
                    double r2 = dx * dx + dy * dy + dz * dz;
                    double r = Math.Sqrt(r2);
                    double f = GM[b] / (r * r2);
                    ax += f * dx; ay += f * dy; az += f * dz;
                }
                vx += ax * dtProp * 0.5; vy += ay * dtProp * 0.5; vz += az * dtProp * 0.5;
            }

            GetBodyStateAtDay(ephemeris, actualDt, arrivalDay, AsteroidIdx,
                out double tgtPx, out double tgtPy, out double tgtPz,
                out double tgtVx, out double tgtVy, out double tgtVz);

            double timeLeft = (arrivalDay - midDay) * DAY;
            double desVx = (tgtPx - px) / timeLeft;
            double desVy = (tgtPy - py) / timeLeft;
            double desVz = (tgtPz - pz) / timeLeft;

            double corrDvx = desVx - vx;
            double corrDvy = desVy - vy;
            double corrDvz = desVz - vz;
            double corrMag = Vec3Mag(corrDvx, corrDvy, corrDvz);

            if (corrMag > 0.1 && corrMag < dvRemaining * 0.6)
            {
                burns.Add((midDay, corrDvx, corrDvy, corrDvz));
                dvRemaining -= corrMag;
                scVx = vx + corrDvx;
                scVy = vy + corrDvy;
                scVz = vz + corrDvz;
                scPx = px; scPy = py; scPz = pz;
            }
        }

        GetBodyStateAtDay(ephemeris, actualDt, arrivalDay, AsteroidIdx,
            out double astPx, out double astPy, out double astPz,
            out double astVx, out double astVy, out double astVz);

        GetBodyStateAtDay(ephemeris, actualDt, impactDay, EarthIdx,
            out double earthImpPx, out double earthImpPy, out double earthImpPz,
            out double earthImpVx, out double earthImpVy, out double earthImpVz);

        double toEarthX = earthImpPx - astPx;
        double toEarthY = earthImpPy - astPy;
        double toEarthZ = earthImpPz - astPz;
        double toEarthDist = Vec3Mag(toEarthX, toEarthY, toEarthZ);

        double impactDirX, impactDirY, impactDirZ;
        if (dvRemaining > 1.0)
        {
            double astVelMag = Vec3Mag(astVx, astVy, astVz);

            double perpX = toEarthX - (toEarthX * astVx + toEarthY * astVy + toEarthZ * astVz) / (astVelMag * astVelMag) * astVx;
            double perpY = toEarthY - (toEarthX * astVx + toEarthY * astVy + toEarthZ * astVz) / (astVelMag * astVelMag) * astVy;
            double perpZ = toEarthZ - (toEarthX * astVx + toEarthY * astVy + toEarthZ * astVz) / (astVelMag * astVelMag) * astVz;
            double perpMag = Vec3Mag(perpX, perpY, perpZ);

            if (perpMag > 1e-10)
            {
                impactDirX = -perpX / perpMag;
                impactDirY = -perpY / perpMag;
                impactDirZ = -perpZ / perpMag;
            }
            else
            {
                impactDirX = -astVx / astVelMag;
                impactDirY = -astVy / astVelMag;
                impactDirZ = -astVz / astVelMag;
            }

            double impactDvMag = Math.Min(dvRemaining * 0.99, dvRemaining);
            burns.Add((arrivalDay, impactDirX * impactDvMag, impactDirY * impactDvMag, impactDirZ * impactDvMag));
        }
        else
        {
            burns.Add((arrivalDay, 0, 0, 0.01));
        }

        return burns;
    }

    static bool LambertSolve(double mu, double x1, double y1, double z1,
        double x2, double y2, double z2, double tof, out double[] v1, out double[] v2)
    {
        v1 = null;
        v2 = null;

        double r1mag = Vec3Mag(x1, y1, z1);
        double r2mag = Vec3Mag(x2, y2, z2);

        double cx = y1 * z2 - z1 * y2;
        double cy = z1 * x2 - x1 * z2;
        double cz = x1 * y2 - y1 * x2;

        double dtheta = Math.Acos(Clamp((x1 * x2 + y1 * y2 + z1 * z2) / (r1mag * r2mag), -1, 1));
        if (cz < 0) dtheta = TWO_PI - dtheta;

        double A = Math.Sin(dtheta) * Math.Sqrt(r1mag * r2mag / (1.0 - Math.Cos(dtheta)));

        if (double.IsNaN(A) || double.IsInfinity(A) || Math.Abs(A) < 1e-30)
            return false;

        double zLow = -4 * PI * PI;
        double zHigh = 4 * PI * PI;

        double z = 0;
        double C, S, y, chi, F, dF;

        for (int iter = 0; iter < 200; iter++)
        {
            StumpffCS(z, out C, out S);

            y = r1mag + r2mag + A * (z * S - 1.0) / Math.Sqrt(C);

            if (y < 0)
            {
                zLow = z;
                z = (z + zHigh) * 0.5;
                continue;
            }

            chi = Math.Sqrt(y / C);
            F = Math.Pow(y / C, 1.5) * S + A * Math.Sqrt(y) - Math.Sqrt(mu) * tof;

            if (Math.Abs(F) < 1e-8)
                break;

            double Cy, Sy;
            StumpffCS(z, out Cy, out Sy);

            if (Math.Abs(z) > 1e-6)
            {
                dF = Math.Pow(y / C, 1.5) * (1.0 / (2.0 * z) * (C - 3.0 * S / (2.0 * C)) + 3.0 * S * S / (4.0 * C))
                     + (A / 8.0) * (3.0 * S * Math.Sqrt(y) / C + A * Math.Sqrt(C / y));
            }
            else
            {
                dF = Math.Sqrt(2.0) / 40.0 * Math.Pow(y, 1.5) + (A / 8.0) * (Math.Sqrt(y) + A * Math.Sqrt(1.0 / (2.0 * y)));
            }

            if (Math.Abs(dF) < 1e-30) break;

            double zNew = z - F / dF;
            if (F < 0) zLow = z;
            else zHigh = z;

            z = zNew;
            if (z < zLow || z > zHigh)
                z = (zLow + zHigh) * 0.5;
        }

        StumpffCS(z, out C, out S);
        y = r1mag + r2mag + A * (z * S - 1.0) / Math.Sqrt(C);

        if (y < 0) return false;

        double f = 1.0 - y / r1mag;
        double gdot = 1.0 - y / r2mag;
        double g = A * Math.Sqrt(y / mu);

        if (Math.Abs(g) < 1e-30) return false;

        v1 = new double[3];
        v2 = new double[3];

        v1[0] = (x2 - f * x1) / g;
        v1[1] = (y2 - f * y1) / g;
        v1[2] = (z2 - f * z1) / g;

        v2[0] = (gdot * x2 - x1) / g;
        v2[1] = (gdot * y2 - y1) / g;
        v2[2] = (gdot * z2 - z1) / g;

        if (double.IsNaN(v1[0]) || double.IsInfinity(v1[0])) return false;

        return true;
    }

    static void StumpffCS(double z, out double C, out double S)
    {
        if (z > 1e-6)
        {
            double sz = Math.Sqrt(z);
            C = (1.0 - Math.Cos(sz)) / z;
            S = (sz - Math.Sin(sz)) / (z * sz);
        }
        else if (z < -1e-6)
        {
            double sz = Math.Sqrt(-z);
            C = (1.0 - Math.Cosh(sz)) / z;
            S = (Math.Sinh(sz) - sz) / (-z * sz);
        }
        else
        {
            C = 1.0 / 2.0;
            S = 1.0 / 6.0;
        }
    }

    static double Vec3Mag(double x, double y, double z)
    {
        return Math.Sqrt(x * x + y * y + z * z);
    }

    static double Clamp(double v, double min, double max)
    {
        if (v < min) return min;
        if (v > max) return max;
        return v;
    }
