use std::io::{self, BufRead, Write};
use std::f32::consts::PI;

#[derive(Clone, Copy)]
struct Point {
    x: f32,
    y: f32,
}

#[derive(Clone, Copy)]
struct Edge {
    u: usize,
    v: usize,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read N and M
    let first_line = match lines.next() {
        Some(Ok(line)) => line,
        _ => return,
    };
    let parts: Vec<usize> = first_line
        .split_whitespace()
        .filter_map(|s| s.parse().ok())
        .collect();
    if parts.len() < 2 {
        return;
    }
    let n = parts[0];
    let m = parts[1];

    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let edge_parts: Vec<usize> = line
                .split_whitespace()
                .filter_map(|s| s.parse().ok())
                .collect();
            if edge_parts.len() >= 2 {
                edges.push(Edge {
                    u: edge_parts[0],
                    v: edge_parts[1],
                });
            }
        }
    }

    if n == 0 {
        return;
    }

    // Initial layout: Circular
    let mut pos = Vec::with_capacity(n);
    let radius = (n as f32).sqrt() * 10.0;
    for i in 0..n {
        let angle = 2.0 * PI * (i as f32) / (n as f32);
        pos.push(Point {
            x: radius * angle.cos(),
            y: radius * angle.sin(),
        });
    }

    // Fruchterman-Reingold parameters
    let area = (n as f32) * 100.0;
    let k = (area / n as f32).sqrt();
    let mut t = radius; // Initial temperature
    let iterations = if n > 10000 { 30 } else if n > 1000 { 50 } else { 100 };
    let dt = t / (iterations as f32);

    let mut disp = vec![Point { x: 0.0, y: 0.0 }; n];

    for _ in 0..iterations {
        // Reset displacement
        for d in disp.iter_mut() {
            d.x = 0.0;
            d.y = 0.0;
        }

        // Repulsive forces (Grid-based optimization for large N)
        if n > 500 {
            let mut min_x = pos[0].x;
            let mut max_x = pos[0].x;
            let mut min_y = pos[0].y;
            let mut max_y = pos[0].y;
            for p in &pos {
                if p.x < min_x { min_x = p.x; }
                if p.x > max_x { max_x = p.x; }
                if p.y < min_y { min_y = p.y; }
                if p.y > max_y { max_y = p.y; }
            }

            let grid_size = (n as f32).sqrt().ceil() as usize;
            let cell_w = (max_x - min_x + 1.0) / grid_size as f32;
            let cell_h = (max_y - min_y + 1.0) / grid_size as f32;
            
            let mut grid = vec![Vec::new(); grid_size * grid_size];
            for i in 0..n {
                let gx = (((pos[i].x - min_x) / cell_w) as usize).min(grid_size - 1);
                let gy = (((pos[i].y - min_y) / cell_h) as usize).min(grid_size - 1);
                grid[gy * grid_size + gx].push(i);
            }

            for i in 0..n {
                let gx = (((pos[i].x - min_x) / cell_w) as usize).min(grid_size - 1);
                let gy = (((pos[i].y - min_y) / cell_h) as usize).min(grid_size - 1);

                let start_x = if gx > 0 { gx - 1 } else { 0 };
                let end_x = if gx < grid_size - 1 { gx + 1 } else { grid_size - 1 };
                let start_y = if gy > 0 { gy - 1 } else { 0 };
                let end_y = if gy < grid_size - 1 { gy + 1 } else { grid_size - 1 };

                for ny in start_y..=end_y {
                    for nx in start_x..=end_x {
                        for &j in &grid[ny * grid_size + nx] {
                            if i == j { continue; }
                            let dx = pos[i].x - pos[j].x;
                            let dy = pos[i].y - pos[j].y;
                            let dist_sq = dx * dx + dy * dy + 0.01;
                            let dist = dist_sq.sqrt();
                            let fr = (k * k) / dist;
                            disp[i].x += (dx / dist) * fr;
                            disp[i].y += (dy / dist) * fr;
                        }
                    }
                }
            }
        } else {
            // O(N^2) for small N
            for i in 0..n {
                for j in 0..n {
                    if i == j { continue; }
                    let dx = pos[i].x - pos[j].x;
                    let dy = pos[i].y - pos[j].y;
                    let dist_sq = dx * dx + dy * dy + 0.01;
                    let dist = dist_sq.sqrt();
                    let fr = (k * k) / dist;
                    disp[i].x += (dx / dist) * fr;
                    disp[i].y += (dy / dist) * fr;
                }
            }
        }

        // Attractive forces
        for edge in &edges {
            let u = edge.u;
            let v = edge.v;
            if u >= n || v >= n { continue; }
            let dx = pos[u].x - pos[v].x;
            let dy = pos[u].y - pos[v].y;
            let dist_sq = dx * dx + dy * dy + 0.01;
            let dist = dist_sq.sqrt();
            let fa = (dist * dist) / k;
            let off_x = (dx / dist) * fa;
            let off_y = (dy / dist) * fa;
            disp[u].x -= off_x;
            disp[u].y -= off_y;
            disp[v].x += off_x;
            disp[v].y += off_y;
        }

        // Update positions
        for i in 0..n {
            let dx = disp[i].x;
            let dy = disp[i].y;
            let dist_sq = dx * dx + dy * dy + 0.01;
            let dist = dist_sq.sqrt();
            let limited_dist = dist.min(t);
            pos[i].x += (dx / dist) * limited_dist;
            pos[i].y += (dy / dist) * limited_dist;
        }

        // Cool
        t -= dt;
        if t < 0.0 { t = 0.0; }
    }

    // Output
    let stdout = io::stdout();
    let mut handle = io::BufWriter::new(stdout.lock());
    writeln!(handle, "{}", n).ok();
    for p in pos {
        writeln!(handle, "{:.4} {:.4}", p.x, p.y).ok();
    }
}
