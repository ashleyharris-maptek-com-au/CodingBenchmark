I have implemented a highly scalable C++ TSP solver designed to handle from 10 to 10,000,000+ cities efficiently within the 300-second time limit.

### Implementation Strategy:
1.  **Adaptive Logic**: The solver switches strategies based on the number of cities ($N$).
2.  **Small Scale ($N \le 3000$)**:
    *   Uses a **Nearest Neighbor** heuristic to build an initial tour.
    *   Refines the tour using a **time-limited 2-opt** local search (up to 10 seconds) to improve quality while ensuring responsiveness.
3.  **Large to Epic Scale ($N > 3000$)**:
    *   Uses a **Morton (Z-order) Curve** sorting approach. This maps 2D coordinates to a 1D space-filling curve, which preserves spatial locality.
    *   Sorting by Morton codes provides a very fast $O(N \log N)$ approximation that is significantly better than random or simple coordinate sorting.
    *   The implementation uses bit-interleaving techniques to generate 64-bit Morton codes from normalized 32-bit coordinate components.
4.  **Performance & Memory**:
    *   Memory usage is kept low (approximately 16-24 bytes per city) to stay well within the 32GB limit even for 10 million cities.
    *   I/O is optimized using `ios_base::sync_with_stdio(false)` and `cin.tie(NULL)`.

### Complete C++ Code:

```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <numeric>
#include <chrono>
#include <stdint.h>

using namespace std;

struct Point {
    float x, y;
    int id;
};

// Morton encoding (Z-order curve) for fast spatial sorting
// Expands a 32-bit integer by inserting 0s between bits
uint32_t expand_bits(uint32_t v) {
    v = (v | (v << 16)) & 0x0000FFFF;
    v = (v | (v << 8))  & 0x00FF00FF;
    v = (v | (v << 4))  & 0x0F0F0F0F;
    v = (v | (v << 2))  & 0x33333333;
    v = (v | (v << 1))  & 0x55555555;
    return v;
}

uint64_t morton_encode(float x, float y, float min_x, float min_y, float scale_x, float scale_y) {
    uint32_t ux = (uint32_t)((x - min_x) * scale_x);
    uint32_t uy = (uint32_t)((y - min_y) * scale_y);

    // Interleave bits of ux and uy
    return ((uint64_t)expand_bits(ux)) | (((uint64_t)expand_bits(uy)) << 1);
}

float dist_sq(const Point& a, const Point& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return dx * dx + dy * dy;
}

void solve_small(int N, vector<Point>& points) {
    // Nearest Neighbor Heuristic
    vector<int> tour;
    vector<bool> visited(N, false);
    tour.reserve(N);
    tour.push_back(0);
    visited[0] = true;

    for (int i = 1; i < N; ++i) {
        int last = tour.back();
        int best_next = -1;
        float min_d = 1e30f;
        for (int j = 0; j < N; ++j) {
            if (!visited[j]) {
                float d = dist_sq(points[last], points[j]);
                if (d < min_d) {
                    min_d = d;
                    best_next = j;
                }
            }
        }
        tour.push_back(best_next);
        visited[best_next] = true;
    }

    // 2-opt refinement (time-limited)
    bool improved = true;
    auto start_time = chrono::steady_clock::now();
    while (improved) {
        improved = false;
        auto now = chrono::steady_clock::now();
        if (chrono::duration_cast<chrono::seconds>(now - start_time).count() > 10) break;

        for (int i = 1; i < N - 1; ++i) {
            for (int j = i + 1; j < N; ++j) {
                int a = tour[i - 1];
                int b = tour[i];
                int c = tour[j];
                int d = (j + 1 == N) ? tour[0] : tour[j + 1];

                float d_old = dist_sq(points[a], points[b]) + dist_sq(points[c], points[d]);
                float d_new = dist_sq(points[a], points[c]) + dist_sq(points[b], points[d]);

                if (d_new < d_old) {
                    reverse(tour.begin() + i, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }

    for (int i = 0; i < N; ++i) {
        cout << points[tour[i]].id << (i == N - 1 ? "" : " ");
    }
    cout << endl;
}

void solve_large(int N, vector<Point>& points) {
    // Find bounding box for Morton normalization
    float min_x = points[0].x, max_x = points[0].x;
    float min_y = points[0].y, max_y = points[0].y;
    for (const auto& p : points) {
        if (p.x < min_x) min_x = p.x;
        if (p.x > max_x) max_x = p.x;
        if (p.y < min_y) min_y = p.y;
        if (p.y > max_y) max_y = p.y;
    }

    float range_x = max_x - min_x;
    float range_y = max_y - min_y;
    // Scale to fit into 31 bits (to avoid overflow during expansion)
    float scale_x = (range_x > 0) ? (float)((1ULL << 31) - 1) / range_x : 0;
    float scale_y = (range_y > 0) ? (float)((1ULL << 31) - 1) / range_y : 0;

    struct MortonPoint {
        uint64_t code;
        int index;
    };
    vector<MortonPoint> m_points(N);
    for (int i = 0; i < N; ++i) {
        m_points[i] = {morton_encode(points[i].x, points[i].y, min_x, min_y, scale_x, scale_y), i};
    }

    // Sort by Morton code
    sort(m_points.begin(), m_points.end(), [](const MortonPoint& a, const MortonPoint& b) {
        return a.code < b.code;
    });

    // Rotate the tour so it starts with city index 0
    int start_pos = 0;
    for (int i = 0; i < N; ++i) {
        if (points[m_points[i].index].id == 0) {
            start_pos = i;
            break;
        }
    }

    for (int i = 0; i < N; ++i) {
        int idx = (start_pos + i) % N;
        cout << points[m_points[idx].index].id << (i == N - 1 ? "" : " ");
    }
    cout << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    if (!(cin >> N)) return 0;

    vector<Point> points(N);
    for (int i = 0; i < N; ++i) {
        cin >> points[i].x >> points[i].y;
        points[i].id = i;
    }

    if (N <= 3000) {
        solve_small(N, points);
    } else {
        solve_large(N, points);
    }

    return 0;
}
```
