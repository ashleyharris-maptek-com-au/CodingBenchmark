use std::io::{self, BufRead};

struct Solver {
    width: usize,
    require_cycle: bool,
    grid: Vec<bool>, // true if blocked or visited
    total_cells: usize,
    path: Vec<usize>,
    neighbors: Vec<Vec<usize>>,
}

impl Solver {
    fn new(width: usize, height: usize, obstacles: Vec<(usize, usize)>, require_cycle: bool) -> Self {
        let mut grid = vec![false; width * height];
        let mut obstacle_count = 0;
        for (x, y) in obstacles {
            if x < width && y < height {
                grid[y * width + x] = true;
                obstacle_count += 1;
            }
        }

        let total_cells = width * height - obstacle_count;
        let mut neighbors = vec![Vec::with_capacity(4); width * height];

        for y in 0..height {
            for x in 0..width {
                let idx = y * width + x;
                if grid[idx] {
                    continue;
                }
                if x > 0 && !grid[idx - 1] {
                    neighbors[idx].push(idx - 1);
                }
                if x < width - 1 && !grid[idx + 1] {
                    neighbors[idx].push(idx + 1);
                }
                if y > 0 && !grid[idx - width] {
                    neighbors[idx].push(idx - width);
                }
                if y < height - 1 && !grid[idx + width] {
                    neighbors[idx].push(idx + width);
                }
            }
        }

        Solver {
            width,
            require_cycle,
            grid,
            total_cells,
            path: Vec::with_capacity(total_cells),
            neighbors,
        }
    }

    fn solve(&mut self) -> bool {
        if self.total_cells == 0 {
            return false;
        }
        if self.grid[0] {
            return false;
        }

        self.path.push(0);
        self.grid[0] = true;
        if self.backtrack(0, 1) {
            return true;
        }
        self.grid[0] = false;
        self.path.pop();
        false
    }

    fn backtrack(&mut self, curr: usize, count: usize) -> bool {
        if count == self.total_cells {
            if self.require_cycle {
                let x = curr % self.width;
                let y = curr / self.width;
                return (x == 1 && y == 0) || (x == 0 && y == 1);
            }
            return true;
        }

        let mut next_moves = Vec::with_capacity(4);

        for &next in &self.neighbors[curr] {
            if !self.grid[next] {
                let mut degree = 0;
                for &nn in &self.neighbors[next] {
                    if !self.grid[nn] {
                        degree += 1;
                    }
                }
                
                if degree == 0 {
                    if count + 1 == self.total_cells {
                        if !self.require_cycle {
                            self.path.push(next);
                            return true;
                        } else {
                            let x = next % self.width;
                            let y = next / self.width;
                            if (x == 1 && y == 0) || (x == 0 && y == 1) {
                                self.path.push(next);
                                return true;
                            }
                        }
                    }
                }

                next_moves.push((next, degree));
            }
        }

        next_moves.sort_by_key(|&(_, d)| d);

        for (next, _) in next_moves {
            self.grid[next] = true;
            if self.is_connected() {
                self.path.push(next);
                if self.backtrack(next, count + 1) {
                    return true;
                }
                self.path.pop();
            }
            self.grid[next] = false;
        }

        false
    }

    fn is_connected(&self) -> bool {
        let mut first_unvisited = None;
        let mut unvisited_count = 0;

        for idx in 0..self.grid.len() {
            if !self.grid[idx] {
                unvisited_count += 1;
                if first_unvisited.is_none() {
                    first_unvisited = Some(idx);
                }
            }
        }

        if unvisited_count == 0 { return true; }

        if let Some(start) = first_unvisited {
            let mut visited = vec![false; self.grid.len()];
            let mut queue = std::collections::VecDeque::new();
            queue.push_back(start);
            visited[start] = true;
            let mut reachable = 1;
            
            while let Some(curr) = queue.pop_front() {
                for &n in &self.neighbors[curr] {
                    if !self.grid[n] && !visited[n] {
                        visited[n] = true;
                        reachable += 1;
                        queue.push_back(n);
                    }
                }
            }
            
            return reachable == unvisited_count;
        }
        true
    }

    fn print_path(&self) {
        println!("{}", self.path.len());
        for &idx in &self.path {
            println!("{} {}", idx % self.width, idx / self.width);
        }
    }
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = match lines.next() {
        Some(Ok(line)) => line,
        _ => return,
    };

    let parts: Vec<usize> = first_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    if parts.len() < 4 {
        return;
    }

    let width = parts[0];
    let height = parts[1];
    let obstacle_count = parts[2];
    let require_cycle = parts[3] == 1;

    let mut obstacles = Vec::with_capacity(obstacle_count);
    for _ in 0..obstacle_count {
        if let Some(Ok(line)) = lines.next() {
            let coords: Vec<usize> = line
                .split_whitespace()
                .map(|s| s.parse().unwrap())
                .collect();
            if coords.len() == 2 {
                obstacles.push((coords[0], coords[1]));
            }
        }
    }

    let mut solver = Solver::new(width, height, obstacles, require_cycle);
    if solver.solve() {
        solver.print_path();
    } else {
        println!("0");
    }
}
