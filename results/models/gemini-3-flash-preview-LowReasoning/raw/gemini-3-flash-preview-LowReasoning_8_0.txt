use std::collections::VecDeque;
use std::io::{self, BufRead, BufWriter, Write};

fn main() {
    let stdin = io::stdin();
    let mut handle = stdin.lock();

    // Read H and W
    let mut header = String::new();
    if handle.read_line(&mut header).is_err() { return; }
    let mut parts = header.split_whitespace();
    let h: usize = parts.next().unwrap().parse().unwrap();
    let w: usize = parts.next().unwrap().parse().unwrap();

    let size = h * w;
    let mut maze = Vec::with_capacity(size);
    let mut start_pos = None;
    let mut end_pos = None;

    // Read maze
    let mut line = String::new();
    for r in 0..h {
        line.clear();
        let _ = handle.read_line(&mut line);
        let bytes = line.as_bytes();
        for c in 0..w {
            let b = if c < bytes.len() { bytes[c] } else { b' ' };
            if b == b'A' {
                start_pos = Some(r * w + c);
            } else if b == b'B' {
                end_pos = Some(r * w + c);
            }
            maze.push(b);
        }
    }

    let start_pos = match start_pos {
        Some(p) => p,
        None => return,
    };
    let end_pos = match end_pos {
        Some(p) => p,
        None => return,
    };

    // BFS setup
    let mut visited = vec![0u64; (size + 63) / 64];
    let mut parent_dir = vec![0u8; size]; // 0: none, 1: Up, 2: Down, 3: Left, 4: Right
    let mut queue = VecDeque::with_capacity(size.min(10_000_000));

    queue.push_back(start_pos);
    visited[start_pos / 64] |= 1 << (start_pos % 64);

    let mut found = false;
    while let Some(curr) = queue.pop_front() {
        if curr == end_pos {
            found = true;
            break;
        }

        let r = curr / w;
        let c = curr % w;

        // Neighbors: Up, Down, Left, Right
        // 1: moved Up, 2: moved Down, 3: moved Left, 4: moved Right
        if r > 0 {
            let next_pos = (r - 1) * w + c;
            if maze[next_pos] != b'#' {
                let idx = next_pos / 64;
                let bit = 1 << (next_pos % 64);
                if (visited[idx] & bit) == 0 {
                    visited[idx] |= bit;
                    parent_dir[next_pos] = 1;
                    queue.push_back(next_pos);
                }
            }
        }
        if r + 1 < h {
            let next_pos = (r + 1) * w + c;
            if maze[next_pos] != b'#' {
                let idx = next_pos / 64;
                let bit = 1 << (next_pos % 64);
                if (visited[idx] & bit) == 0 {
                    visited[idx] |= bit;
                    parent_dir[next_pos] = 2;
                    queue.push_back(next_pos);
                }
            }
        }
        if c > 0 {
            let next_pos = r * w + (c - 1);
            if maze[next_pos] != b'#' {
                let idx = next_pos / 64;
                let bit = 1 << (next_pos % 64);
                if (visited[idx] & bit) == 0 {
                    visited[idx] |= bit;
                    parent_dir[next_pos] = 3;
                    queue.push_back(next_pos);
                }
            }
        }
        if c + 1 < w {
            let next_pos = r * w + (c + 1);
            if maze[next_pos] != b'#' {
                let idx = next_pos / 64;
                let bit = 1 << (next_pos % 64);
                if (visited[idx] & bit) == 0 {
                    visited[idx] |= bit;
                    parent_dir[next_pos] = 4;
                    queue.push_back(next_pos);
                }
            }
        }
    }

    if found {
        let mut path = Vec::new();
        let mut curr = end_pos;
        while curr != start_pos {
            path.push(curr);
            let r = curr / w;
            let c = curr % w;
            match parent_dir[curr] {
                1 => curr = (r + 1) * w + c, // Came from Down (moved Up)
                2 => curr = (r - 1) * w + c, // Came from Up (moved Down)
                3 => curr = r * w + (c + 1), // Came from Right (moved Left)
                4 => curr = r * w + (c - 1), // Came from Left (moved Right)
                _ => break,
            }
        }
        path.push(start_pos);
        path.reverse();

        let stdout = io::stdout();
        let mut writer = BufWriter::new(stdout.lock());
        writeln!(writer, "{}", path.len()).unwrap();
        for pos in path {
            writeln!(writer, "{} {}", pos % w, pos / w).unwrap();
        }
    } else {
        // No path found
        println!("0");
    }
}
