using System;
using System.Collections.Generic;
using System.Linq;
using System.IO;
using System.Text;

public class ChinesePostmanSolver
{
    public class Edge
    {
        public int U, V, W;
    }

    public static void Main(string[] args)
    {
        FastReader reader = new FastReader(Console.OpenStandardInput());
        int N = reader.NextInt();
        int M = reader.NextInt();

        if (M == 0)
        {
            Console.WriteLine("0");
            return;
        }

        List<Edge> originalEdges = new List<Edge>(M);
        List<int>[] adj = new List<int>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<int>();

        for (int i = 0; i < M; i++)
        {
            int u = reader.NextInt();
            int v = reader.NextInt();
            int w = reader.NextInt();
            var edge = new Edge { U = u, V = v, W = w };
            originalEdges.Add(edge);
            adj[u].Add(i);
            adj[v].Add(i);
        }

        List<int> oddNodes = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if (adj[i].Count % 2 != 0)
                oddNodes.Add(i);
        }

        List<Edge> currentEdges = new List<Edge>(originalEdges);
        if (oddNodes.Count > 0)
        {
            GreedyMatch(N, adj, originalEdges, oddNodes, currentEdges);
            
            // Rebuild adjacency for the Eulerian circuit
            for (int i = 0; i < N; i++) adj[i].Clear();
            for (int i = 0; i < currentEdges.Count; i++)
            {
                adj[currentEdges[i].U].Add(i);
                adj[currentEdges[i].V].Add(i);
            }
        }

        List<int> circuit = FindEulerianCircuit(N, adj, currentEdges);
        
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < circuit.Count; i++)
        {
            sb.Append(circuit[i]);
            if (i < circuit.Count - 1) sb.Append(" ");
        }
        Console.WriteLine(sb.ToString());
    }

    static void GreedyMatch(int N, List<int>[] adj, List<Edge> originalEdges, List<int> oddNodes, List<Edge> currentEdges)
    {
        bool[] isUnmatched = new bool[N];
        foreach (int node in oddNodes) isUnmatched[node] = true;

        long[] dist = new long[N];
        for (int i = 0; i < N; i++) dist[i] = long.MaxValue;
        int[] parentEdge = new int[N];
        int[] parentNode = new int[N];
        PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
        List<int> visitedNodes = new List<int>();

        foreach (int startNode in oddNodes)
        {
            if (!isUnmatched[startNode]) continue;
            isUnmatched[startNode] = false;

            // Efficiently reset dist array using visitedNodes from previous run
            foreach (int node in visitedNodes) dist[node] = long.MaxValue;
            visitedNodes.Clear();

            dist[startNode] = 0;
            visitedNodes.Add(startNode);
            pq.Clear();
            pq.Enqueue(startNode, 0);

            int targetNode = -1;
            while (pq.Count > 0)
            {
                pq.TryDequeue(out int u, out long d);
                if (d > dist[u]) continue;

                if (isUnmatched[u])
                {
                    targetNode = u;
                    break;
                }

                foreach (int edgeIdx in adj[u])
                {
                    var e = originalEdges[edgeIdx];
                    int v = (e.U == u) ? e.V : e.U;
                    long nextDist = d + e.W;
                    if (nextDist < dist[v])
                    {
                        if (dist[v] == long.MaxValue) visitedNodes.Add(v);
                        dist[v] = nextDist;
                        parentEdge[v] = edgeIdx;
                        parentNode[v] = u;
                        pq.Enqueue(v, nextDist);
                    }
                }
            }

            if (targetNode != -1)
            {
                isUnmatched[targetNode] = false;
                int curr = targetNode;
                while (curr != startNode)
                {
                    int edgeIdx = parentEdge[curr];
                    var e = originalEdges[edgeIdx];
                    currentEdges.Add(new Edge { U = e.U, V = e.V, W = e.W });
                    curr = parentNode[curr];
                }
            }
        }
    }

    static List<int> FindEulerianCircuit(int N, List<int>[] adj, List<Edge> edges)
    {
        int[] edgePtr = new int[N];
        Stack<int> stack = new Stack<int>();
        List<int> circuit = new List<int>();
        bool[] usedEdge = new bool[edges.Count];

        stack.Push(0);
        while (stack.Count > 0)
        {
            int u = stack.Peek();
            bool found = false;
            while (edgePtr[u] < adj[u].Count)
            {
                int edgeIdx = adj[u][edgePtr[u]++];
                if (usedEdge[edgeIdx]) continue;

                usedEdge[edgeIdx] = true;
                var e = edges[edgeIdx];
                int v = (e.U == u) ? e.V : e.U;
                stack.Push(v);
                found = true;
                break;
            }

            if (!found)
            {
                circuit.Add(stack.Pop());
            }
        }

        return circuit;
    }

    public class FastReader
    {
        private Stream _stream;
        private byte[] _buffer = new byte[8192];
        private int _pos = 0;
        private int _len = 0;

        public FastReader(Stream stream)
        {
            _stream = stream;
        }

        private int ReadByte()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _pos = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_pos++];
        }

        public int NextInt()
        {
            int c = ReadByte();
            while (c != -1 && (c < '0' || c > '9')) c = ReadByte();
            if (c == -1) return 0;
            int res = 0;
            while (c >= '0' && c <= '9')
            {
                res = res * 10 + (c - '0');
                c = ReadByte();
            }
            return res;
        }
    }
}
