#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <random>
#include <chrono>

using namespace std;

struct Vector3 {
    double x, y, z;
    Vector3 operator+(const Vector3& v) const { return {x + v.x, y + v.y, z + v.z}; }
    Vector3 operator-(const Vector3& v) const { return {x - v.x, y - v.y, z - v.z}; }
    Vector3 operator*(double s) const { return {x * s, y * s, z * s}; }
    double dot(const Vector3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vector3 cross(const Vector3& v) const {
        return {y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x};
    }
    double lengthSq() const { return x * x + y * y + z * z; }
    double length() const { return sqrt(lengthSq()); }
    Vector3 normalize() const {
        double l = length();
        return l > 1e-9 ? (*this) * (1.0 / l) : Vector3{0, 0, 0};
    }
};

struct Matrix3x3 {
    double m[3][3];
    Vector3 operator*(const Vector3& v) const {
        return {
            m[0][0] * v.x + m[0][1] * v.y + m[0][2] * v.z,
            m[1][0] * v.x + m[1][1] * v.y + m[1][2] * v.z,
            m[2][0] * v.x + m[2][1] * v.y + m[2][2] * v.z
        };
    }
};

Matrix3x3 getRotationMatrix(double rx, double ry, double rz) {
    double cx = cos(rx), sx = sin(rx);
    double cy = cos(ry), sy = sin(ry);
    double cz = cos(rz), sz = sin(rz);
    return {{
        {cy * cz, -cy * sz, sy},
        {cx * sz + sx * sy * cz, cx * cz - sx * sy * sz, -sx * cy},
        {sx * sz - cx * sy * cz, sx * cz + cx * sy * sz, cx * cy}
    }};
}

struct Tetrahedron {
    Vector3 center;
    double rx, ry, rz;
    Vector3 vertices[4];

    void computeVertices(double e) {
        double h = e * sqrt(2.0 / 3.0);
        double R = e * sqrt(3.0 / 8.0);
        double r = e / sqrt(24.0);
        
        Vector3 local[4] = {
            {0, R, 0},
            {e / sqrt(3.0), -r, 0},
            {-e / (2.0 * sqrt(3.0)), -r, e / 2.0},
            {-e / (2.0 * sqrt(3.0)), -r, -e / 2.0}
        };

        Matrix3x3 rot = getRotationMatrix(rx, ry, rz);
        for (int i = 0; i < 4; ++i) {
            vertices[i] = center + rot * local[i];
        }
    }
};

struct Plane {
    Vector3 normal;
    double d;
    bool isInside(const Vector3& p) const {
        return normal.dot(p) + d <= 1e-7;
    }
};

bool isTetraInside(const Tetrahedron& t, const vector<Plane>& planes) {
    for (const auto& v : t.vertices) {
        for (const auto& p : planes) {
            if (!p.isInside(v)) return false;
        }
    }
    return true;
}

bool sat(const Tetrahedron& t1, const Tetrahedron& t2) {
    auto getAxes = [](const Tetrahedron& t, vector<Vector3>& axes) {
        for (int i = 0; i < 4; ++i) {
            for (int j = i + 1; j < 4; ++j) {
                axes.push_back((t.vertices[i] - t.vertices[j]).normalize());
            }
        }
        for (int i = 0; i < 4; ++i) {
            Vector3 v0 = t.vertices[i];
            Vector3 v1 = t.vertices[(i + 1) % 4];
            Vector3 v2 = t.vertices[(i + 2) % 4];
            axes.push_back((v1 - v0).cross(v2 - v0).normalize());
        }
    };

    vector<Vector3> axes;
    getAxes(t1, axes);
    getAxes(t2, axes);

    for (const auto& axis : axes) {
        if (axis.lengthSq() < 1e-9) continue;
        double min1 = 1e18, max1 = -1e18;
        double min2 = 1e18, max2 = -1e18;
        for (int i = 0; i < 4; ++i) {
            double p1 = axis.dot(t1.vertices[i]);
            min1 = min(min1, p1); max1 = max(max1, p1);
            double p2 = axis.dot(t2.vertices[i]);
            min2 = min(min2, p2); max2 = max(max2, p2);
        }
        if (max1 < min2 - 1e-7 || max2 < min1 - 1e-7) return false;
    }
    return true;
}

int main() {
    int V_count;
    double edge_length;
    if (!(cin >> V_count >> edge_length)) return 0;

    vector<Vector3> container_vertices(V_count);
    Vector3 min_bound = {1e18, 1e18, 1e18}, max_bound = {-1e18, -1e18, -1e18};
    for (int i = 0; i < V_count; ++i) {
        cin >> container_vertices[i].x >> container_vertices[i].y >> container_vertices[i].z;
        min_bound.x = min(min_bound.x, container_vertices[i].x);
        min_bound.y = min(min_bound.y, container_vertices[i].y);
        min_bound.z = min(min_bound.z, container_vertices[i].z);
        max_bound.x = max(max_bound.x, container_vertices[i].x);
        max_bound.y = max(max_bound.y, container_vertices[i].y);
        max_bound.z = max(max_bound.z, container_vertices[i].z);
    }

    // Extract planes for convex polyhedron
    vector<Plane> planes;
    for (int i = 0; i < V_count; ++i) {
        for (int j = i + 1; j < V_count; ++j) {
            for (int k = j + 1; k < V_count; ++k) {
                Vector3 v0 = container_vertices[i];
                Vector3 v1 = container_vertices[j];
                Vector3 v2 = container_vertices[k];
                Vector3 normal = (v1 - v0).cross(v2 - v0).normalize();
                if (normal.lengthSq() < 1e-9) continue;
                double d = -normal.dot(v0);
                
                int side_pos = 0, side_neg = 0;
                for (int l = 0; l < V_count; ++l) {
                    double dist = normal.dot(container_vertices[l]) + d;
                    if (dist > 1e-7) side_pos++;
                    else if (dist < -1e-7) side_neg++;
                }
                
                if (side_pos == 0 || side_neg == 0) {
                    if (side_pos > 0) { normal = normal * -1.0; d = -d; }
                    bool duplicate = false;
                    for (const auto& p : planes) {
                        if ((p.normal - normal).lengthSq() < 1e-7 && abs(p.d - d) < 1e-7) {
                            duplicate = true; break;
                        }
                    }
                    if (!duplicate) planes.push_back({normal, d});
                }
            }
        }
    }

    auto start_time = chrono::steady_clock::now();
    vector<Tetrahedron> placed;
    
    double step = edge_length * 0.2; // Start with a smaller step
    // Adjust step for large containers
    double vol = (max_bound.x - min_bound.x) * (max_bound.y - min_bound.y) * (max_bound.z - min_bound.z);
    if (vol > 1000) step = edge_length * 0.8;
    if (vol > 10000) step = edge_length * 1.2;

    for (double x = min_bound.x; x <= max_bound.x; x += step) {
        for (double y = min_bound.y; y <= max_bound.y; y += step) {
            for (double z = min_bound.z; z <= max_bound.z; z += step) {
                auto now = chrono::steady_clock::now();
                if (chrono::duration_cast<chrono::seconds>(now - start_time).count() > 28) goto end_packing;

                for (double rx : {0.0, 0.78, 1.57, 2.35}) {
                    for (double ry : {0.0, 0.78, 1.57, 2.35}) {
                        for (double rz : {0.0, 0.78}) {
                            Tetrahedron t;
                            t.center = {x, y, z};
                            t.rx = rx; t.ry = ry; t.rz = rz;
                            t.computeVertices(edge_length);

                            if (isTetraInside(t, planes)) {
                                bool collision = false;
                                for (int i = (int)placed.size() - 1; i >= 0; --i) {
                                    double dx = placed[i].center.x - t.center.x;
                                    double dy = placed[i].center.y - t.center.y;
                                    double dz = placed[i].center.z - t.center.z;
                                    if (dx*dx + dy*dy + dz*dz < edge_length * edge_length * 1.5) {
                                        if (sat(t, placed[i])) {
                                            collision = true;
                                            break;
                                        }
                                    }
                                }
                                if (!collision) {
                                    placed.push_back(t);
                                    goto next_pos;
                                }
                            }
                        }
                    }
                }
                next_pos:;
            }
        }
    }

end_packing:
    cout << placed.size() << endl;
    for (const auto& t : placed) {
        cout << t.center.x << " " << t.center.y << " " << t.center.z << " "
             << t.rx << " " << t.ry << " " << t.rz << endl;
    }

    return 0;
}
