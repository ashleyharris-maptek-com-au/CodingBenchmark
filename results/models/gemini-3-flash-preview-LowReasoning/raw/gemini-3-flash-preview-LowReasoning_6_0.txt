using System;
using System.Collections.Generic;
using System.Linq;

public class OrbitalTSP
{
    public struct Vector3
    {
        public double X, Y, Z;
        public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vector3 operator *(Vector3 a, double b) => new Vector3(a.X * b, a.Y * b, a.Z * b);
        public static Vector3 operator /(Vector3 a, double b) => new Vector3(a.X / b, a.Y / b, a.Z / b);
        public double Magnitude() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public double MagnitudeSq() => X * X + Y * Y + Z * Z;
        public static double Dot(Vector3 a, Vector3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vector3 Cross(Vector3 a, Vector3 b) => new Vector3(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
    }

    public struct State
    {
        public Vector3 P, V;
        public State(Vector3 p, Vector3 v) { P = p; V = v; }
    }

    static double mu;

    public static void Main(string[] args)
    {
        string line1 = Console.ReadLine();
        if (string.IsNullOrEmpty(line1)) return;
        string[] parts1 = line1.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(parts1[0]);
        mu = double.Parse(parts1[1]);

        string line2 = Console.ReadLine();
        if (string.IsNullOrEmpty(line2)) return;
        string[] parts2 = line2.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        State start = new State(
            new Vector3(double.Parse(parts2[0]), double.Parse(parts2[1]), double.Parse(parts2[2])),
            new Vector3(double.Parse(parts2[3]), double.Parse(parts2[4]), double.Parse(parts2[5]))
        );

        List<State> stations = new List<State>();
        for (int i = 0; i < N; i++)
        {
            string line = Console.ReadLine();
            if (string.IsNullOrEmpty(line)) break;
            string[] parts = line.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
            stations.Add(new State(
                new Vector3(double.Parse(parts[0]), double.Parse(parts[1]), double.Parse(parts[2])),
                new Vector3(double.Parse(parts[3]), double.Parse(parts[4]), double.Parse(parts[5]))
            ));
        }

        if (stations.Count == 0) return;

        double[,] costMatrix = BuildCostMatrix(start, stations);
        int[] path = SolveTSP(costMatrix);
        Console.WriteLine(string.Join(" ", path));
    }

    static double[,] BuildCostMatrix(State start, List<State> stations)
    {
        int N = stations.Count;
        double[,] matrix = new double[N + 1, N];

        for (int j = 0; j < N; j++)
        {
            matrix[0, j] = EstimateDeltaV(start, stations[j]);
        }

        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                if (i == j) matrix[i + 1, j] = double.PositiveInfinity;
                else matrix[i + 1, j] = EstimateDeltaV(stations[i], stations[j]);
            }
        }
        return matrix;
    }

    static double EstimateDeltaV(State s1, State s2)
    {
        double r1 = s1.P.Magnitude();
        double r2 = s2.P.Magnitude();
        double v1 = s1.V.Magnitude();
        double v2 = s2.V.Magnitude();

        Vector3 h1 = Vector3.Cross(s1.P, s1.V);
        Vector3 h2 = Vector3.Cross(s2.P, s2.V);
        
        double mag1 = h1.Magnitude();
        double mag2 = h2.Magnitude();
        
        double planeChangeDV = 0;
        if (mag1 > 1e-9 && mag2 > 1e-9)
        {
            double cosAngle = Vector3.Dot(h1, h2) / (mag1 * mag2);
            cosAngle = Math.Clamp(cosAngle, -1.0, 1.0);
            double angle = Math.Acos(cosAngle);
            planeChangeDV = 2 * v1 * Math.Sin(angle / 2);
        }
        
        double speedChangeDV = Math.Abs(v1 - v2);
        double posDiff = (s1.P - s2.P).Magnitude();

        return planeChangeDV + speedChangeDV + posDiff * 0.0001;
    }

    static int[] SolveTSP(double[,] costMatrix)
    {
        int N = costMatrix.GetLength(1);
        if (N <= 8) return SolveSmallTSP(costMatrix);
        return SolveLargeTSP(costMatrix);
    }

    static int[] SolveSmallTSP(double[,] costMatrix)
    {
        int N = costMatrix.GetLength(1);
        int[] bestPath = null;
        double minCost = double.PositiveInfinity;

        int[] currentPath = Enumerable.Range(0, N).ToArray();

        void Permute(int k)
        {
            if (k == N)
            {
                double currentCost = costMatrix[0, currentPath[0]];
                for (int i = 0; i < N - 1; i++)
                    currentCost += costMatrix[currentPath[i] + 1, currentPath[i + 1]];
                
                if (currentCost < minCost)
                {
                    minCost = currentCost;
                    bestPath = (int[])currentPath.Clone();
                }
                return;
            }

            for (int i = k; i < N; i++)
            {
                Swap(ref currentPath[k], ref currentPath[i]);
                Permute(k + 1);
                Swap(ref currentPath[k], ref currentPath[i]);
            }
        }

        Permute(0);
        return bestPath;
    }

    static void Swap(ref int a, ref int b) { int temp = a; a = b; b = temp; }

    static int[] SolveLargeTSP(double[,] costMatrix)
    {
        int N = costMatrix.GetLength(1);
        int[] path = new int[N];
        bool[] visited = new bool[N];
        
        int current = -1;
        double minDist = double.PositiveInfinity;
        for(int i=0; i<N; i++) {
            if(costMatrix[0, i] < minDist) {
                minDist = costMatrix[0, i];
                current = i;
            }
        }
        path[0] = current;
        visited[current] = true;

        for (int i = 1; i < N; i++)
        {
            int next = -1;
            minDist = double.PositiveInfinity;
            for(int j=0; j<N; j++) {
                if(!visited[j] && costMatrix[current + 1, j] < minDist) {
                    minDist = costMatrix[current + 1, j];
                    next = j;
                }
            }
            path[i] = next;
            visited[next] = true;
            current = next;
        }

        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 0; i < N - 1; i++)
            {
                for (int j = i + 1; j < N; j++)
                {
                    double currentCost = GetEdgeCost(i - 1, i, path, costMatrix) + GetEdgeCost(j, j + 1, path, costMatrix);
                    double newCost = GetEdgeCost(i - 1, j, path, costMatrix) + GetEdgeCost(i, j + 1, path, costMatrix);

                    if (newCost < currentCost - 1e-9)
                    {
                        Reverse(path, i, j);
                        improved = true;
                    }
                }
            }
        }

        return path;
    }

    static double GetEdgeCost(int idx1, int idx2, int[] path, double[,] costMatrix)
    {
        int N = path.Length;
        if (idx2 >= N || idx1 < -1) return 0;
        if (idx1 == -1) return costMatrix[0, path[idx2]];
        return costMatrix[path[idx1] + 1, path[idx2]];
    }

    static void Reverse(int[] path, int i, int j)
    {
        while (i < j)
        {
            Swap(ref path[i], ref path[j]);
            i++;
            j--;
        }
    }
}
