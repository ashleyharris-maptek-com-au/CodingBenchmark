struct Vector3 {
    double x, y, z;
    Vector3 operator+(const Vector3& v) const { return {x + v.x, y + v.y, z + v.z}; }
    Vector3 operator-(const Vector3& v) const { return {x - v.x, y - v.y, z - v.z}; }
    Vector3 operator*(double s) const { return {x * s, y * s, z * s}; }
    double dot(const Vector3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vector3 cross(const Vector3& v) const {
        return {y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x};
    }
    double lengthSq() const { return x * x + y * y + z * z; }
    double length() const { return std::sqrt(lengthSq()); }
    Vector3 normalized() const {
        double l = length();
        return l > 1e-12 ? *this * (1.0 / l) : Vector3{0, 0, 0};
    }
    bool operator<(const Vector3& v) const {
        if (std::abs(x - v.x) > 1e-9) return x < v.x;
        if (std::abs(y - v.y) > 1e-9) return y < v.y;
        return z < v.z - 1e-9;
    }
    bool operator==(const Vector3& v) const {
        return std::abs(x - v.x) < 1e-9 && std::abs(y - v.y) < 1e-9 && std::abs(z - v.z) < 1e-9;
    }
};

struct Triangle {
    int v[3];
};

struct AABB {
    Vector3 min, max;
    AABB() {
        min = {std::numeric_limits<double>::max(), std::numeric_limits<double>::max(), std::numeric_limits<double>::max()};
        max = {std::numeric_limits<double>::lowest(), std::numeric_limits<double>::lowest(), std::numeric_limits<double>::lowest()};
    }
    void expand(const Vector3& v) {
        min.x = std::min(min.x, v.x); min.y = std::min(min.y, v.y); min.z = std::min(min.z, v.z);
        max.x = std::max(max.x, v.x); max.y = std::max(max.y, v.y); max.z = std::max(max.z, v.z);
    }
    void expand(const AABB& b) {
        expand(b.min); expand(b.max);
    }
    bool contains(const Vector3& v) const {
        return v.x >= min.x - 1e-9 && v.x <= max.x + 1e-9 &&
               v.y >= min.y - 1e-9 && v.y <= max.y + 1e-9 &&
               v.z >= min.z - 1e-9 && v.z <= max.z + 1e-9;
    }
};

struct BVHNode {
    AABB bounds;
    std::unique_ptr<BVHNode> left, right;
    std::vector<int> triangleIndices;
    bool isLeaf() const { return left == nullptr && right == nullptr; }
};

class BVH {
public:
    BVH(const std::vector<Vector3>& vertices, const std::vector<Triangle>& triangles)
        : vertices(vertices), triangles(triangles) {
        std::vector<int> indices(triangles.size());
        for (int i = 0; i < (int)triangles.size(); ++i) indices[i] = i;
        root = build(indices);
    }

    bool isInside(const Vector3& p) const {
        if (!root->bounds.contains(p)) return false;
        
        static thread_local std::mt19937 gen(1337);
        std::uniform_real_distribution<double> dist(-1.0, 1.0);
        
        for (int attempt = 0; attempt < 5; ++attempt) {
            Vector3 rayDir = {dist(gen), dist(gen), dist(gen)};
            double len = rayDir.length();
            if (len < 1e-6) continue;
            rayDir = rayDir * (1.0 / len);
            
            int count = 0;
            bool onEdge = false;
            rayIntersect(root.get(), p, rayDir, count, onEdge);
            if (!onEdge) return (count % 2) != 0;
        }
        return false;
    }

private:
    const std::vector<Vector3>& vertices;
    const std::vector<Triangle>& triangles;
    std::unique_ptr<BVHNode> root;

    std::unique_ptr<BVHNode> build(std::vector<int>& indices) {
        auto node = std::make_unique<BVHNode>();
        for (int idx : indices) {
            node->bounds.expand(vertices[triangles[idx].v[0]]);
            node->bounds.expand(vertices[triangles[idx].v[1]]);
            node->bounds.expand(vertices[triangles[idx].v[2]]);
        }

        if (indices.size() <= 16) {
            node->triangleIndices = std::move(indices);
            return node;
        }

        Vector3 size = node->bounds.max - node->bounds.min;
        int axis = 0;
        if (size.y > size.x && size.y > size.z) axis = 1;
        else if (size.z > size.x && size.z > size.y) axis = 2;

        double mid = (axis == 0 ? node->bounds.min.x + node->bounds.max.x :
                      axis == 1 ? node->bounds.min.y + node->bounds.max.y :
                                  node->bounds.min.z + node->bounds.max.z) * 0.5;

        std::vector<int> leftIndices, rightIndices;
        for (int idx : indices) {
            double center = (axis == 0 ? (vertices[triangles[idx].v[0]].x + vertices[triangles[idx].v[1]].x + vertices[triangles[idx].v[2]].x) / 3.0 :
                             axis == 1 ? (vertices[triangles[idx].v[0]].y + vertices[triangles[idx].v[1]].y + vertices[triangles[idx].v[2]].y) / 3.0 :
                                         (vertices[triangles[idx].v[0]].z + vertices[triangles[idx].v[1]].z + vertices[triangles[idx].v[2]].z) / 3.0);
            if (center < mid) leftIndices.push_back(idx);
            else rightIndices.push_back(idx);
        }

        if (leftIndices.empty() || rightIndices.empty()) {
            node->triangleIndices = std::move(indices);
            return node;
        }

        node->left = build(leftIndices);
        node->right = build(rightIndices);
        return node;
    }

    void rayIntersect(BVHNode* node, const Vector3& p, const Vector3& d, int& count, bool& onEdge) const {
        if (node->isLeaf()) {
            for (int idx : node->triangleIndices) {
                int res = rayTriangleIntersect(p, d, vertices[triangles[idx].v[0]], vertices[triangles[idx].v[1]], vertices[triangles[idx].v[2]]);
                if (res == 1) count++;
                else if (res == -1) { onEdge = true; return; }
            }
        } else {
            if (node->left) rayIntersect(node->left.get(), p, d, count, onEdge);
            if (onEdge) return;
            if (node->right) rayIntersect(node->right.get(), p, d, count, onEdge);
        }
    }

    int rayTriangleIntersect(const Vector3& orig, const Vector3& dir, const Vector3& v0, const Vector3& v1, const Vector3& v2) const {
        Vector3 edge1 = v1 - v0;
        Vector3 edge2 = v2 - v0;
        Vector3 h = dir.cross(edge2);
        double a = edge1.dot(h);
        if (std::abs(a) < 1e-12) return 0;
        double f = 1.0 / a;
        Vector3 s = orig - v0;
        double u = f * s.dot(h);
        if (u < -1e-9 || u > 1.000000001) return 0;
        Vector3 q = s.cross(edge1);
        double v = f * dir.dot(q);
        if (v < -1e-9 || u + v > 1.000000001) return 0;
        double t = f * edge2.dot(q);
        if (std::abs(t) < 1e-9) return -1;
        if (t > 1e-9) {
            if (u < 1e-9 || u > 0.999999999 || v < 1e-9 || u + v > 0.999999999) return -1;
            return 1;
        }
        return 0;
    }
};

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) return 0;

    std::vector<Vector3> verticesA(VA);
    for (int i = 0; i < VA; ++i) std::cin >> verticesA[i].x >> verticesA[i].y >> verticesA[i].z;
    std::vector<Triangle> trianglesA(FA);
    for (int i = 0; i < FA; ++i) std::cin >> trianglesA[i].v[0] >> trianglesA[i].v[1] >> trianglesA[i].v[2];

    std::vector<Vector3> verticesB(VB);
    for (int i = 0; i < VB; ++i) std::cin >> verticesB[i].x >> verticesB[i].y >> verticesB[i].z;
    std::vector<Triangle> trianglesB(FB);
    for (int i = 0; i < FB; ++i) std::cin >> trianglesB[i].v[0] >> trianglesB[i].v[1] >> trianglesB[i].v[2];

    BVH bvhA(verticesA, trianglesA);
    BVH bvhB(verticesB, trianglesB);

    std::vector<Vector3> outVertices;
    std::vector<Triangle> outTriangles;
    std::map<Vector3, int> vertexMap;

    auto getVertexIdx = [&](const Vector3& v) {
        auto it = vertexMap.lower_bound(v);
        if (it != vertexMap.end() && it->first == v) return it->second;
        int idx = (int)outVertices.size();
        outVertices.push_back(v);
        vertexMap[v] = idx;
        return idx;
    };

    auto addMesh = [&](const std::vector<Vector3>& vs, const std::vector<Triangle>& ts, const BVH& other) {
        for (const auto& t : ts) {
            Vector3 v0 = vs[t.v[0]], v1 = vs[t.v[1]], v2 = vs[t.v[2]];
            Vector3 center = (v0 + v1 + v2) * (1.0 / 3.0);
            if (!other.isInside(center)) {
                int i0 = getVertexIdx(v0);
                int i1 = getVertexIdx(v1);
                int i2 = getVertexIdx(v2);
                outTriangles.push_back({i0, i1, i2});
            }
        }
    };

    addMesh(verticesA, trianglesA, bvhB);
    addMesh(verticesB, trianglesB, bvhA);

    printf("%zu %zu\n", outVertices.size(), outTriangles.size());
    for (const auto& v : outVertices) printf("%.10f %.10f %.10f\n", v.x, v.y, v.z);
    for (const auto& t : outTriangles) printf("%d %d %d\n", t.v[0], t.v[1], t.v[2]);

    return 0;
}
