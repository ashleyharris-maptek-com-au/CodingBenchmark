#include <cstdio>
#include <vector>
#include <algorithm>
#include <random>
#include <cstdint>
#include <queue>

using namespace std;

// Fast I/O for handling massive input files
class FastReader {
    static const int BUF_SIZE = 1 << 20;
    char buf[BUF_SIZE];
    int pos = 0, len = 0;
    char read_char() {
        if (pos == len) {
            len = (int)fread(buf, 1, BUF_SIZE, stdin);
            pos = 0;
            if (len == 0) return EOF;
        }
        return buf[pos++];
    }
public:
    bool read_uint32(uint32_t& n) {
        n = 0;
        char c = read_char();
        while (c != EOF && (c < '0' || c > '9')) c = read_char();
        if (c == EOF) return false;
        while (c >= '0' && c <= '9') {
            n = n * 10 + (c - '0');
            c = read_char();
        }
        return true;
    }
    bool read_uint64(uint64_t& n) {
        n = 0;
        char c = read_char();
        while (c != EOF && (c < '0' || c > '9')) c = read_char();
        if (c == EOF) return false;
        while (c >= '0' && c <= '9') {
            n = n * 10 + (c - '0');
            c = read_char();
        }
        return true;
    }
};

// Memory-efficient edge structure
// to: 4 bytes, next: 4 bytes. Total 8 bytes per directed edge.
struct Edge {
    uint32_t to;
    uint32_t next;
};

// Use global containers to manage memory tightly
vector<Edge> edges_pool;
vector<uint32_t> head;
vector<uint8_t> capacities;
vector<uint32_t> dist;
vector<uint32_t> q_buffer;
vector<bool> dead;
uint32_t num_nodes;
uint32_t edge_ptr = 0;

void add_undirected_edge(uint32_t u, uint32_t v) {
    // Forward edge
    edges_pool[edge_ptr] = {v, head[u]};
    capacities[edge_ptr] = 1;
    head[u] = edge_ptr++;
    
    // Backward edge
    edges_pool[edge_ptr] = {u, head[v]};
    capacities[edge_ptr] = 1;
    head[v] = edge_ptr++;
}

// Dinic's BFS for level graph
bool bfs_dinic(uint32_t s, uint32_t t) {
    fill(dist.begin(), dist.end(), UINT32_MAX);
    dist[s] = 0;
    uint32_t q_h = 0, q_t = 0;
    q_buffer[q_t++] = s;
    while (q_h < q_t) {
        uint32_t u = q_buffer[q_h++];
        for (uint32_t i = head[u]; i != UINT32_MAX; i = edges_pool[i].next) {
            if (capacities[i] > 0 && dist[edges_pool[i].to] == UINT32_MAX) {
                dist[edges_pool[i].to] = dist[u] + 1;
                q_buffer[q_t++] = edges_pool[i].to;
                if (edges_pool[i].to == t) return true;
            }
        }
    }
    return dist[t] != UINT32_MAX;
}

// Dinic's DFS for augmenting paths (unit capacity optimization)
uint32_t dfs_dinic(uint32_t u, uint32_t t) {
    if (u == t) return 1;
    for (uint32_t i = head[u]; i != UINT32_MAX; i = edges_pool[i].next) {
        uint32_t v = edges_pool[i].to;
        if (capacities[i] > 0 && !dead[v] && dist[v] == dist[u] + 1) {
            if (dfs_dinic(v, t)) {
                capacities[i] -= 1;
                capacities[i ^ 1] += 1;
                return 1;
            }
        }
    }
    dead[u] = true; // Mark node as dead for current level graph
    return 0;
}

uint32_t max_flow(uint32_t s, uint32_t t) {
    // Reset capacities for each trial
    for (uint32_t i = 0; i < edge_ptr; i++) capacities[i] = 1;
    uint32_t total_flow = 0;
    while (bfs_dinic(s, t)) {
        dead.assign(num_nodes, false);
        while (dfs_dinic(s, t)) {
            total_flow++;
        }
    }
    return total_flow;
}

// Perform BFS to find all nodes reachable from s in the residual graph
void find_reachable(uint32_t s) {
    fill(dist.begin(), dist.end(), UINT32_MAX);
    dist[s] = 0;
    uint32_t q_h = 0, q_t = 0;
    q_buffer[q_t++] = s;
    while (q_h < q_t) {
        uint32_t u = q_buffer[q_h++];
        for (uint32_t i = head[u]; i != UINT32_MAX; i = edges_pool[i].next) {
            if (capacities[i] > 0 && dist[edges_pool[i].to] == UINT32_MAX) {
                dist[edges_pool[i].to] = 0; // Using dist as a visited flag
                q_buffer[q_t++] = edges_pool[i].to;
            }
        }
    }
}

int main() {
    FastReader reader;
    uint32_t n;
    uint64_t m;
    if (!reader.read_uint32(n) || !reader.read_uint64(m)) return 0;
    num_nodes = n;
    
    if (num_nodes < 2) {
        printf("0\n");
        return 0;
    }

    head.assign(num_nodes, UINT32_MAX);
    // Limit pool size by 2*m or uint32 range
    uint64_t pool_size = 2 * m;
    if (pool_size > 4294967290ULL) pool_size = 4294967290ULL;
    edges_pool.resize(pool_size);
    capacities.resize(pool_size);
    
    for (uint64_t i = 0; i < m; ++i) {
        uint32_t u, v;
        if (reader.read_uint32(u) && reader.read_uint32(v)) {
            if (u == v || u >= num_nodes || v >= num_nodes) continue;
            if (edge_ptr + 2 > pool_size) break;
            add_undirected_edge(u, v);
        }
    }

    // Calculate min degree and identify a candidate source node
    uint32_t min_deg = UINT32_MAX;
    uint32_t s_node = 0;
    for (uint32_t i = 0; i < num_nodes; i++) {
        uint32_t deg = 0;
        for (uint32_t e = head[i]; e != UINT32_MAX; e = edges_pool[e].next) deg++;
        if (deg < min_deg) {
            min_deg = deg;
            s_node = i;
        }
    }

    uint32_t best_cut_size = min_deg;
    
    dist.resize(num_nodes);
    q_buffer.resize(num_nodes);
    dead.resize(num_nodes);
    
    // Strategy: Run max-flow between s_node (min-deg node) and various target nodes.
    // min-degree cut covers |S|=1. Balanced cuts are easily found by random pairs.
    vector<uint32_t> targets;
    
    // Perform BFS from s_node to find furthest nodes and check connectivity
    {
        fill(dist.begin(), dist.end(), UINT32_MAX);
        dist[s_node] = 0;
        uint32_t q_h = 0, q_t = 0;
        q_buffer[q_t++] = s_node;
        while (q_h < q_t) {
            uint32_t u = q_buffer[q_h++];
            for (uint32_t i = head[u]; i != UINT32_MAX; i = edges_pool[i].next) {
                if (dist[edges_pool[i].to] == UINT32_MAX) {
                    dist[edges_pool[i].to] = dist[u] + 1;
                    q_buffer[q_t++] = edges_pool[i].to;
                }
            }
        }
        
        // Connectivity check (though problem says it's connected)
        if (q_t < num_nodes) {
            // Pick any node not reachable as target for size 0 cut
            best_cut_size = 0;
            printf("0\n");
            return 0;
        }

        // Add some of the furthest nodes as targets
        for (int i = 0; i < 15 && q_t > 0; i++) {
            uint32_t t = q_buffer[--q_t];
            if (t != s_node) targets.push_back(t);
        }
    }
    
    // Add random targets
    mt19937 rng(1337);
    uniform_int_distribution<uint32_t> rnd_node(0, num_nodes - 1);
    int num_random_trials = (num_nodes < 50) ? (int)num_nodes : 30;
    for (int i = 0; i < num_random_trials; i++) {
        uint32_t t = rnd_node(rng);
        if (t != s_node) targets.push_back(t);
    }
    
    sort(targets.begin(), targets.end());
    targets.erase(unique(targets.begin(), targets.end()), targets.end());

    uint32_t best_t = UINT32_MAX;
    bool found_smaller = false;

    // Run max-flows
    for (uint32_t t : targets) {
        if (best_cut_size <= 1) break; 
        uint32_t flow = max_flow(s_node, t);
        if (flow < best_cut_size) {
            best_cut_size = flow;
            best_t = t;
            found_smaller = true;
        }
    }

    // Output result
    printf("%u\n", best_cut_size);
    if (!found_smaller) {
        // Output edges incident to s_node
        for (uint32_t i = head[s_node]; i != UINT32_MAX; i = edges_pool[i].next) {
            printf("%u %u\n", s_node, edges_pool[i].to);
        }
    } else {
        // Re-run the best max-flow to reconstruct the cut
        max_flow(s_node, best_t);
        find_reachable(s_node);
        for (uint32_t u = 0; u < num_nodes; u++) {
            if (dist[u] == UINT32_MAX) continue;
            for (uint32_t i = head[u]; i != UINT32_MAX; i = edges_pool[i].next) {
                if (dist[edges_pool[i].to] == UINT32_MAX) {
                    // Cut edge from S to T
                    printf("%u %u\n", u, edges_pool[i].to);
                }
            }
        }
    }

    return 0;
}
