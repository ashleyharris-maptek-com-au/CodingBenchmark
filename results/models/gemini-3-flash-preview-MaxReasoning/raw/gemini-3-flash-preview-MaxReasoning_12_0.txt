using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

namespace BinPacking3D
{
    public struct Vector3
    {
        public double X, Y, Z;
        public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public Vector3 Normalize()
        {
            double l = Length();
            return l < 1e-12 ? new Vector3(0, 0, 0) : new Vector3(X / l, Y / l, Z / l);
        }
        public static Vector3 Cross(Vector3 a, Vector3 b) => new Vector3(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
        public static double Dot(Vector3 a, Vector3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
    }

    public struct Quaternion
    {
        public double W, X, Y, Z;
        public Quaternion(double w, double x, double y, double z) { W = w; X = x; Y = y; Z = z; }

        public static Quaternion FromAxisAngle(Vector3 axis, double angle)
        {
            double s = Math.Sin(angle / 2);
            Vector3 a = axis.Normalize();
            return new Quaternion(Math.Cos(angle / 2), a.X * s, a.Y * s, a.Z * s);
        }

        public static Quaternion operator *(Quaternion a, Quaternion b)
        {
            return new Quaternion(
                a.W * b.W - a.X * b.X - a.Y * b.Y - a.Z * b.Z,
                a.W * b.X + a.X * b.W + a.Y * b.Z - a.Z * b.Y,
                a.W * b.Y - a.X * b.Z + a.Y * b.W + a.Z * b.X,
                a.W * b.Z + a.X * b.Y - a.Y * b.X + a.Z * b.W
            );
        }

        public Vector3 Rotate(Vector3 p)
        {
            double qx2 = X * X, qy2 = Y * Y, qz2 = Z * Z, qw2 = W * W;
            return new Vector3(
                p.X * (qw2 + qx2 - qy2 - qz2) + p.Y * 2 * (X * Y - W * Z) + p.Z * 2 * (X * Z + W * Y),
                p.X * 2 * (X * Y + W * Z) + p.Y * (qw2 - qx2 + qy2 - qz2) + p.Z * 2 * (Y * Z - W * X),
                p.X * 2 * (X * Z - W * Y) + p.Y * 2 * (Y * Z + W * X) + p.Z * (qw2 - qx2 - qy2 + qz2)
            );
        }

        public static List<Quaternion> GetOrthogonalRotations()
        {
            var res = new List<Quaternion>();
            Vector3[] axes = { new Vector3(1, 0, 0), new Vector3(-1, 0, 0), new Vector3(0, 1, 0), new Vector3(0, -1, 0), new Vector3(0, 0, 1), new Vector3(0, 0, -1) };
            foreach (var f in axes)
            {
                foreach (var u in axes)
                {
                    if (Math.Abs(Vector3.Dot(f, u)) < 0.01)
                    {
                        Vector3 r = Vector3.Cross(f, u);
                        res.Add(FromMatrix(f, u, r));
                    }
                }
            }
            return res;
        }

        public static Quaternion FromMatrix(Vector3 f, Vector3 u, Vector3 r)
        {
            double m00 = f.X, m10 = f.Y, m20 = f.Z;
            double m01 = u.X, m11 = u.Y, m21 = u.Z;
            double m02 = r.X, m12 = r.Y, m22 = r.Z;

            double tr = m00 + m11 + m22;
            if (tr > 0)
            {
                double s = Math.Sqrt(tr + 1.0) * 2;
                return new Quaternion(0.25 * s, (m21 - m12) / s, (m02 - m20) / s, (m10 - m01) / s);
            }
            else if ((m00 > m11) && (m00 > m22))
            {
                double s = Math.Sqrt(1.0 + m00 - m11 - m22) * 2;
                return new Quaternion((m21 - m12) / s, 0.25 * s, (m01 + m10) / s, (m02 + m20) / s);
            }
            else if (m11 > m22)
            {
                double s = Math.Sqrt(1.0 + m11 - m00 - m22) * 2;
                return new Quaternion((m02 - m20) / s, (m01 + m10) / s, 0.25 * s, (m12 + m21) / s);
            }
            else
            {
                double s = Math.Sqrt(1.0 + m22 - m00 - m11) * 2;
                return new Quaternion((m10 - m01) / s, (m02 + m20) / s, (m12 + m21) / s, 0.25 * s);
            }
        }

        public static Quaternion RotationBetween(Vector3 a, Vector3 b)
        {
            a = a.Normalize();
            b = b.Normalize();
            double dot = Vector3.Dot(a, b);
            if (dot > 0.99999999) return new Quaternion(1, 0, 0, 0);
            if (dot < -0.99999999)
            {
                Vector3 axis = Vector3.Cross(new Vector3(1, 0, 0), a);
                if (axis.Length() < 0.01) axis = Vector3.Cross(new Vector3(0, 1, 0), a);
                return Quaternion.FromAxisAngle(axis, Math.PI);
            }
            Vector3 c = Vector3.Cross(a, b);
            double s = Math.Sqrt((1 + dot) * 2);
            return new Quaternion(s / 2, c.X / s, c.Y / s, c.Z / s);
        }
    }

    class Program
    {
        static void Main(string[] args)
        {
            TextReader tr = Console.In;
            string firstLine = tr.ReadLine();
            if (string.IsNullOrWhiteSpace(firstLine)) return;
            string[] parts = firstLine.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int V = int.Parse(parts[0]);
            int F = int.Parse(parts[1]);

            Vector3[] vertices = new Vector3[V];
            for (int i = 0; i < V; i++)
            {
                string line = tr.ReadLine();
                string[] vParts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                vertices[i] = new Vector3(double.Parse(vParts[0], CultureInfo.InvariantCulture), double.Parse(vParts[1], CultureInfo.InvariantCulture), double.Parse(vParts[2], CultureInfo.InvariantCulture));
            }

            int[][] faces = new int[F][];
            for (int i = 0; i < F; i++)
            {
                string line = tr.ReadLine();
                string[] fParts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                faces[i] = new int[] { int.Parse(fParts[0]), int.Parse(fParts[1]), int.Parse(fParts[2]) };
            }

            string containerLine = tr.ReadLine();
            string[] cParts = containerLine.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            double containerW = double.Parse(cParts[0], CultureInfo.InvariantCulture);
            double containerH = double.Parse(cParts[1], CultureInfo.InvariantCulture);
            double containerD = double.Parse(cParts[2], CultureInfo.InvariantCulture);

            List<Quaternion> candidateRotations = GetCandidateRotations(vertices, faces);
            long bestCount = -1;
            Quaternion bestQ = new Quaternion(1, 0, 0, 0);
            double bestMinX = 0, bestMinY = 0, bestMinZ = 0;
            double bestW = 0, bestH = 0, bestD = 0;
            long bestNX = 0, bestNY = 0, bestNZ = 0;

            foreach (var q in candidateRotations)
            {
                double minX = double.MaxValue, maxX = double.MinValue;
                double minY = double.MaxValue, maxY = double.MinValue;
                double minZ = double.MaxValue, maxZ = double.MinValue;

                foreach (var v in vertices)
                {
                    Vector3 rv = q.Rotate(v);
                    if (rv.X < minX) minX = rv.X; if (rv.X > maxX) maxX = rv.X;
                    if (rv.Y < minY) minY = rv.Y; if (rv.Y > maxY) maxY = rv.Y;
                    if (rv.Z < minZ) minZ = rv.Z; if (rv.Z > maxZ) maxZ = rv.Z;
                }

                double w = maxX - minX;
                double h = maxY - minY;
                double d = maxZ - minZ;

                if (w < 1e-11 || h < 1e-11 || d < 1e-11) continue;

                long nx = (long)((containerW + 1e-9) / w);
                long ny = (long)((containerH + 1e-9) / h);
                long nz = (long)((containerD + 1e-9) / d);

                if (nx < 0) nx = 0;
                if (ny < 0) ny = 0;
                if (nz < 0) nz = 0;

                long count = 0;
                try { count = checked(nx * ny * nz); } catch { count = long.MaxValue; }

                if (count > bestCount)
                {
                    bestCount = count;
                    bestQ = q;
                    bestMinX = minX; bestMinY = minY; bestMinZ = minZ;
                    bestW = w; bestH = h; bestD = d;
                    bestNX = nx; bestNY = ny; bestNZ = nz;
                }
            }

            using (var sw = new StreamWriter(Console.OpenStandardOutput(), System.Text.Encoding.ASCII, 1024 * 1024))
            {
                sw.WriteLine(bestCount);
                if (bestCount <= 0) return;

                string rotPart = string.Format(CultureInfo.InvariantCulture, "{0:G7} {1:G7} {2:G7} {3:G7}", bestQ.W, bestQ.X, bestQ.Y, bestQ.Z);

                string[] tzStrings = new string[Math.Min(bestNZ, 1000000)];
                long nzToPrecalculate = Math.Min(bestNZ, 1000000);
                for (long k = 0; k < nzToPrecalculate; k++)
                {
                    tzStrings[k] = (-bestMinZ + k * bestD).ToString("G7", CultureInfo.InvariantCulture) + " " + rotPart;
                }

                for (long i = 0; i < bestNX; i++)
                {
                    string txStr = (-bestMinX + i * bestW).ToString("G7", CultureInfo.InvariantCulture) + " ";
                    for (long j = 0; j < bestNY; j++)
                    {
                        string txy = txStr + (-bestMinY + j * bestH).ToString("G7", CultureInfo.InvariantCulture) + " ";
                        for (long k = 0; k < bestNZ; k++)
                        {
                            if (k < nzToPrecalculate)
                            {
                                sw.WriteLine(txy + tzStrings[k]);
                            }
                            else
                            {
                                sw.Write(txy);
                                sw.Write((-bestMinZ + k * bestD).ToString("G7", CultureInfo.InvariantCulture));
                                sw.Write(' ');
                                sw.WriteLine(rotPart);
                            }
                        }
                    }
                }
            }
        }

        static List<Quaternion> GetCandidateRotations(Vector3[] vertices, int[][] faces)
        {
            var res = new List<Quaternion>();
            var seen = new HashSet<string>();

            Action<Quaternion> addQ = (q) =>
            {
                double mag = Math.Sqrt(q.W * q.W + q.X * q.X + q.Y * q.Y + q.Z * q.Z);
                if (mag < 1e-12) return;
                double w = q.W / mag, x = q.X / mag, y = q.Y / mag, z = q.Z / mag;
                if (w < 0) { w = -w; x = -x; y = -y; z = -z; }
                string s = string.Format("{0:F4}_{1:F4}_{2:F4}_{3:F4}", w, x, y, z);
                if (seen.Add(s)) res.Add(new Quaternion(w, x, y, z));
            };

            foreach (var q in Quaternion.GetOrthogonalRotations()) addQ(q);

            int faceLimit = Math.Min(faces.Length, 128);
            for (int i = 0; i < faceLimit; i++)
            {
                Vector3 v0 = vertices[faces[i][0]];
                Vector3 v1 = vertices[faces[i][1]];
                Vector3 v2 = vertices[faces[i][2]];
                Vector3 normal = Vector3.Cross(v1 - v0, v2 - v0).Normalize();
                if (normal.Length() > 0.5) AddOrientations(addQ, normal);
            }

            var edges = new HashSet<string>();
            for (int i = 0; i < faceLimit; i++)
            {
                for (int j = 0; j < 3; j++)
                {
                    int a = faces[i][j], b = faces[i][(j + 1) % 3];
                    if (a > b) { int t = a; a = b; b = t; }
                    string key = a + "_" + b;
                    if (edges.Add(key))
                    {
                        Vector3 dir = (vertices[b] - vertices[a]).Normalize();
                        if (dir.Length() > 0.5) AddOrientations(addQ, dir);
                        if (edges.Count >= 128) break;
                    }
                }
                if (edges.Count >= 128) break;
            }

            return res;
        }

        static void AddOrientations(Action<Quaternion> addQ, Vector3 dir)
        {
            Vector3[] targets = { new Vector3(1, 0, 0), new Vector3(0, 1, 0), new Vector3(0, 0, 1) };
            foreach (var target in targets)
            {
                Quaternion baseQ = Quaternion.RotationBetween(dir, target);
                for (int a = 0; a < 24; a++)
                {
                    Quaternion rot = Quaternion.FromAxisAngle(target, a * 2 * Math.PI / 24);
                    addQ(rot * baseQ);
                }
            }
        }
    }
