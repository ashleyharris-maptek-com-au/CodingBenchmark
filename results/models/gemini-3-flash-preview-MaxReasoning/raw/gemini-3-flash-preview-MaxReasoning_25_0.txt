using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;
using System.Numerics;
using System.Threading;
using System.Threading.Tasks;

namespace AsteroidInterception
{
    public struct Vector3d
    {
        public double X, Y, Z;
        public Vector3d(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vector3d operator +(Vector3d a, Vector3d b) => new Vector3d(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vector3d operator -(Vector3d a, Vector3d b) => new Vector3d(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vector3d operator *(Vector3d a, double b) => new Vector3d(a.X * b, a.Y * b, a.Z * b);
        public static Vector3d operator *(double a, Vector3d b) => new Vector3d(a * b.X, a * b.Y, a * b.Z);
        public static Vector3d operator /(Vector3d a, double b) => new Vector3d(a.X / b, a.Y / b, a.Z / b);
        public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public double LengthSquared() => X * X + Y * Y + Z * Z;
        public static double Dot(Vector3d a, Vector3d b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vector3d Cross(Vector3d a, Vector3d b) => new Vector3d(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
    }

    public class Body
    {
        public string Name;
        public double Mass;
        public Vector3d Position;
        public Vector3d Velocity;
        public Vector3d Acceleration;
        public Body Clone() => new Body { Name = Name, Mass = Mass, Position = Position, Velocity = Velocity };
    }

    public class Burn { public double Time; public Vector3d DV; }
    public class Result { public List<Burn> Burns = new List<Burn>(); public double Deflection; public double TotalDV; }

    class Program
    {
        static double WarningDays;
        static double ImpactorMass;
        static double DeltaVBudget;
        static double AsteroidMass;
        static double T0_Unix;
        static List<Body> InitialBodies = new List<Body>();
        static string DE421Path;
        static double SunMu;

        const double G = 6.67430e-11;

        static void Main(string[] args)
        {
            Thread.CurrentThread.CurrentCulture = CultureInfo.InvariantCulture;
            if (!ParseInput()) return;

            var sun = InitialBodies.OrderByDescending(b => b.Mass).First();
            SunMu = G * sun.Mass;

            // 1. Pre-propagate all bodies to find collision and build lookup tables
            int totalDays = (int)(Math.Max(30 * 365, WarningDays + 365 * 10));
            var positions = PrePropagate(totalDays, out var velocities, out double tCollision);

            // 2. Search for best trajectory
            var result = Solve(positions, velocities, tCollision);

            // 3. Refine best result
            if (result != null)
            {
                Refine(result, positions, velocities);
            }

            // 4. Output results
            if (result != null)
            {
                Console.WriteLine(result.Burns.Count);
                foreach (var burn in result.Burns)
                {
                    Console.WriteLine($"{burn.Time / 86400.0:F8} {burn.DV.X:F8} {burn.DV.Y:F8} {burn.DV.Z:F8}");
                }
            }
        }

        static void Refine(Result result, Vector3d[][] posTable, Vector3d[][] velTable)
        {
            double tLaunch = result.Burns[0].Time;
            double tImpact = result.Burns[1].Time;
            int asteroidIdx = InitialBodies.FindIndex(b => b.Name == "Asteroid");
            int earthIdx = InitialBodies.FindIndex(b => b.Name == "Earth");
            
            Vector3d vLaunchRel = result.Burns[0].DV;
            
            for (int iter = 0; iter < 10; iter++)
            {
                var bodies = InitialBodies.Select(b => b.Clone()).ToList();
                
                // Initial state of SC
                var sc = new Body { 
                    Name = "SC", 
                    Mass = ImpactorMass, 
                    Position = GetState(posTable, earthIdx, tLaunch), 
                    Velocity = GetState(velTable, earthIdx, tLaunch) + vLaunchRel 
                };
                
                // Advance all bodies to tLaunch
                double t = 0;
                double dt = 3600;
                while (t + dt <= tLaunch) { StepKDK(bodies, dt); t += dt; }
                if (t < tLaunch) { StepKDK(bodies, tLaunch - t); t = tLaunch; }
                
                // Add SC to simulation
                bodies.Add(sc);
                UpdateAccelerations(bodies);
                
                // Advance to tImpact
                while (t + dt <= tImpact) { StepKDK(bodies, dt); t += dt; }
                if (t < tImpact) { StepKDK(bodies, tImpact - t); t = tImpact; }
                
                Vector3d rSC = bodies.Last().Position;
                Vector3d rAst = bodies[asteroidIdx].Position;
                Vector3d error = rSC - rAst;
                
                if (error.Length() < 100) break; // Within 100m
                
                // Adjust vLaunchRel using approximate Jacobian (time*I)
                vLaunchRel = vLaunchRel - error / (tImpact - tLaunch);
                
                result.Burns[0].DV = vLaunchRel;
                double dvL = vLaunchRel.Length();
                if (dvL < DeltaVBudget)
                {
                    double remaining = DeltaVBudget - dvL;
                    Vector3d vRel = bodies.Last().Velocity - bodies[asteroidIdx].Velocity;
                    result.Burns[1].DV = vRel.Length() > 1e-3 ? vRel * (remaining / vRel.Length()) : new Vector3d(0, 0, 0);
                }
                else
                {
                    result.Burns[1].DV = new Vector3d(0, 0, 0);
                }
            }
        }

        static bool ParseInput()
        {
            string line = Console.ReadLine();
            while (line != null && string.IsNullOrWhiteSpace(line)) line = Console.ReadLine();
            if (line == null) return false;
            var parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int numBodies = int.Parse(parts[0]);
            WarningDays = double.Parse(parts[1]);
            ImpactorMass = double.Parse(parts[2]);
            DeltaVBudget = double.Parse(parts[3]);
            AsteroidMass = double.Parse(parts[4]);
            T0_Unix = double.Parse(parts[5]);

            for (int i = 0; i < numBodies; i++)
            {
                line = Console.ReadLine();
                while (string.IsNullOrWhiteSpace(line)) line = Console.ReadLine();
                parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                InitialBodies.Add(new Body
                {
                    Name = parts[0],
                    Mass = double.Parse(parts[1]),
                    Position = new Vector3d(double.Parse(parts[2]), double.Parse(parts[3]), double.Parse(parts[4])),
                    Velocity = new Vector3d(double.Parse(parts[5]), double.Parse(parts[6]), double.Parse(parts[7]))
                });
            }
            DE421Path = Console.ReadLine();
            return true;
        }

        static Vector3d[][] PrePropagate(int days, out Vector3d[][] velocities, out double tCollision)
        {
            var bodies = InitialBodies.Select(b => b.Clone()).ToList();
            int asteroidIdx = bodies.FindIndex(b => b.Name == "Asteroid");
            int earthIdx = bodies.FindIndex(b => b.Name == "Earth");

            Vector3d[][] posTable = new Vector3d[bodies.Count][];
            Vector3d[][] velTable = new Vector3d[bodies.Count][];
            for (int i = 0; i < bodies.Count; i++)
            {
                posTable[i] = new Vector3d[days + 1];
                velTable[i] = new Vector3d[days + 1];
            }

            double dt = 3600; 
            double t = 0;
            tCollision = days * 86400.0;
            double minDist = double.MaxValue;

            UpdateAccelerations(bodies);
            for (int d = 0; d <= days; d++)
            {
                for (int i = 0; i < bodies.Count; i++)
                {
                    posTable[i][d] = bodies[i].Position;
                    velTable[i][d] = bodies[i].Velocity;
                }

                if (d < days)
                {
                    for (int step = 0; step < 24; step++)
                    {
                        StepKDK(bodies, dt);
                        t += dt;
                        double dist = (bodies[asteroidIdx].Position - bodies[earthIdx].Position).Length();
                        if (dist < minDist)
                        {
                            minDist = dist;
                            if (dist < 1.3e7) tCollision = Math.Min(tCollision, t);
                        }
                    }
                }
            }

            velocities = velTable;
            return posTable;
        }

        static void UpdateAccelerations(List<Body> bodies)
        {
            for (int i = 0; i < bodies.Count; i++) bodies[i].Acceleration = new Vector3d(0, 0, 0);
            for (int i = 0; i < bodies.Count; i++)
            {
                for (int j = i + 1; j < bodies.Count; j++)
                {
                    Vector3d r = bodies[j].Position - bodies[i].Position;
                    double distSq = r.LengthSquared();
                    if (distSq < 1e-6) continue;
                    double dist = Math.Sqrt(distSq);
                    double f = G / (distSq * dist);
                    bodies[i].Acceleration += r * (bodies[j].Mass * f);
                    bodies[j].Acceleration -= r * (bodies[i].Mass * f);
                }
            }
        }

        static void StepKDK(List<Body> bodies, double dt)
        {
            for (int i = 0; i < bodies.Count; i++) bodies[i].Velocity += bodies[i].Acceleration * (dt * 0.5);
            for (int i = 0; i < bodies.Count; i++) bodies[i].Position += bodies[i].Velocity * dt;
            UpdateAccelerations(bodies);
            for (int i = 0; i < bodies.Count; i++) bodies[i].Velocity += bodies[i].Acceleration * (dt * 0.5);
        }

        static Result Solve(Vector3d[][] posTable, Vector3d[][] velTable, double tCollision)
        {
            int earthIdx = InitialBodies.FindIndex(b => b.Name == "Earth");
            int asteroidIdx = InitialBodies.FindIndex(b => b.Name == "Asteroid");

            Result best = null;
            double maxDeflection = -1;
            object lockObj = new object();

            int numThreads = Environment.ProcessorCount;
            Parallel.For(0, numThreads, threadId =>
            {
                Random rnd = new Random(threadId);
                for (int j = 0; j < 50000 / numThreads; j++)
                {
                    double tLaunch = rnd.NextDouble() * Math.Min(WarningDays, tCollision / 86400.0 - 60) * 86400.0;
                    double tImpact = tLaunch + (60 + rnd.NextDouble() * 2000) * 86400.0;
                    if (tImpact >= tCollision - 10 * 86400.0) continue;

                    Vector3d r1 = GetState(posTable, earthIdx, tLaunch);
                    Vector3d vEarth = GetState(velTable, earthIdx, tLaunch);
                    Vector3d r2 = GetState(posTable, asteroidIdx, tImpact);
                    Vector3d vAst = GetState(velTable, asteroidIdx, tImpact);

                    if (SolveLambert(r1, r2, tImpact - tLaunch, out Vector3d v1, out Vector3d v2, SunMu))
                    {
                        Vector3d dvLaunch = v1 - vEarth;
                        double dvL = dvLaunch.Length();
                        if (dvL <= DeltaVBudget)
                        {
                            double remaining = DeltaVBudget - dvL;
                            Vector3d vRel = v2 - vAst;
                            Vector3d dvImpact = vRel.Length() > 1e-3 ? vRel * (remaining / vRel.Length()) : new Vector3d(0, 0, 0);
                            
                            Vector3d vImpulse = (vRel + dvImpact) * (ImpactorMass / AsteroidMass);
                            double deflection = vImpulse.Length() * (tCollision - tImpact);

                            lock (lockObj)
                            {
                                if (deflection > maxDeflection)
                                {
                                    maxDeflection = deflection;
                                    best = new Result();
                                    best.Burns.Add(new Burn { Time = tLaunch, DV = dvLaunch });
                                    best.Burns.Add(new Burn { Time = tImpact, DV = dvImpact });
                                    best.Deflection = deflection;
                                    best.TotalDV = dvL + remaining;
                                }
                            }
                        }
                    }
                }
            });

            return best;
        }

        static Vector3d GetState(Vector3d[][] table, int bodyIdx, double t)
        {
            double day = t / 86400.0;
            int d = (int)day;
            if (d < 0) return table[bodyIdx][0];
            if (d >= table[bodyIdx].Length - 1) return table[bodyIdx].Last();
            double frac = day - d;
            return table[bodyIdx][d] * (1 - frac) + table[bodyIdx][d + 1] * frac;
        }

        static bool SolveLambert(Vector3d r1, Vector3d r2, double tof, out Vector3d v1, out Vector3d v2, double mu)
        {
            v1 = v2 = new Vector3d();
            double r1m = r1.Length();
            double r2m = r2.Length();
            double cosTheta = Vector3d.Dot(r1, r2) / (r1m * r2m);
            if (cosTheta > 1.0) cosTheta = 1.0; else if (cosTheta < -1.0) cosTheta = -1.0;
            
            Vector3d cross = Vector3d.Cross(r1, r2);
            double theta = Math.Acos(cosTheta);
            if (cross.Z < 0) theta = 2 * Math.PI - theta;

            double c = Math.Sqrt(r1m * r1m + r2m * r2m - 2 * r1m * r2m * cosTheta);
            double s = (r1m + r2m + c) / 2.0;
            
            double low = s / 2.0, high = 1e16;
            for (int i = 0; i < 40; i++)
            {
                double a = (low + high) / 2.0;
                double sinAlpha2 = Math.Sqrt(s / (2 * a));
                double sinBeta2 = Math.Sqrt((s - c) / (2 * a));
                double alpha = 2 * Math.Asin(sinAlpha2);
                double beta = 2 * Math.Asin(sinBeta2);
                
                double t;
                if (theta <= Math.PI)
                    t = Math.Sqrt(a * a * a / mu) * ((alpha - Math.Sin(alpha)) - (beta - Math.Sin(beta)));
                else
                    t = Math.Sqrt(a * a * a / mu) * ((2 * Math.PI - (alpha - Math.Sin(alpha))) - (beta - Math.Sin(beta)));
                
                if (t > tof) high = a; else low = a;
            }
            
            double aFinal = (low + high) / 2.0;
            double alphaF = 2 * Math.Asin(Math.Sqrt(s / (2 * aFinal)));
            double betaF = 2 * Math.Asin(Math.Sqrt((s - c) / (2 * aFinal)));
            if (theta > Math.PI) alphaF = 2 * Math.PI - alphaF;

            double de = alphaF - betaF;
            double f = 1 - (aFinal / r1m) * (1 - Math.Cos(de));
            double g = Math.Sqrt(aFinal * aFinal * aFinal / mu) * (de - Math.Sin(de));
            double gdot = 1 - (aFinal / r2m) * (1 - Math.Cos(de));

            if (Math.Abs(g) < 1e-3) return false;

            v1 = (r2 - r1 * f) / g;
            v2 = (r2 * gdot - r1) / g;
            
            return !double.IsNaN(v1.X);
        }
    }
}
