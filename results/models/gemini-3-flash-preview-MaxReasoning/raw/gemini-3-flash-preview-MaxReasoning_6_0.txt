using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Threading.Tasks;

/*
 * Orbital TSP Solver
 * Algorithm:
 * 1. Physics: Universal Variables Lambert Solver (Bate, Mueller, White) for Delta-V estimation.
 * 2. Performance: Parallelized cost matrix calculation (12 cores).
 * 3. Optimization: 2-opt heuristic with multiple random restarts (200-500) to find near-optimal visitation order.
 * 4. Scalability: Efficiently handles 2 to 100 stations.
 */
namespace OTSP
{
    class Program
    {
        public struct Vector3
        {
            public double X, Y, Z;
            public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
            public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
            public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
            public static Vector3 operator *(Vector3 a, double b) => new Vector3(a.X * b, a.Y * b, a.Z * b);
            public static Vector3 operator /(Vector3 a, double b) => new Vector3(a.X / b, a.Y / b, a.Z / b);
            public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);
            public static double Dot(Vector3 a, Vector3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
            public static Vector3 Cross(Vector3 a, Vector3 b) => new Vector3(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
        }

        public struct State
        {
            public Vector3 R, V;
            public State(Vector3 r, Vector3 v) { R = r; V = v; }
        }

        static double MU;

        static void Main(string[] args)
        {
            string line1 = Console.ReadLine();
            if (string.IsNullOrEmpty(line1)) return;
            string[] parts1 = line1.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts1.Length < 2) return;
            int N = int.Parse(parts1[0]);
            MU = double.Parse(parts1[1]);

            State start = ReadState();
            State[] stations = new State[N];
            for (int i = 0; i < N; i++)
            {
                stations[i] = ReadState();
            }

            if (N == 0) return;

            // Precompute orbital periods and cost matrix
            double[] periods = new double[N + 1];
            periods[0] = GetPeriod(start);
            for (int i = 0; i < N; i++) periods[i + 1] = GetPeriod(stations[i]);

            double[,] costs = new double[N + 1, N + 1];
            Parallel.For(0, N + 1, i =>
            {
                State s1 = (i == 0) ? start : stations[i - 1];
                for (int j = i + 1; j <= N; j++)
                {
                    State s2 = (j == 0) ? start : stations[j - 1];
                    
                    // Try a few transfer times to find a reasonable delta-V
                    double minT = Math.Min(periods[i], periods[j]);
                    double[] dts = { 0.1 * minT, 0.25 * minT, 0.4 * minT };
                    double minDV = double.MaxValue;
                    
                    foreach (double dt in dts)
                    {
                        double dv = EstimateDeltaV(s1, s2, dt);
                        if (dv < minDV) minDV = dv;
                    }
                    costs[i, j] = costs[j, i] = minDV;
                }
            });

            // Solve TSP
            int[] order = SolveTSP(N, costs);

            // Output order
            Console.WriteLine(string.Join(" ", order));
        }

        static State ReadState()
        {
            string line = Console.ReadLine();
            if (string.IsNullOrEmpty(line)) return new State(new Vector3(), new Vector3());
            string[] p = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            return new State(
                new Vector3(double.Parse(p[0]), double.Parse(p[1]), double.Parse(p[2])),
                new Vector3(double.Parse(p[3]), double.Parse(p[4]), double.Parse(p[5]))
            );
        }

        static double GetPeriod(State s)
        {
            double r = s.R.Length();
            double v2 = Vector3.Dot(s.V, s.V);
            double a_inv = 2.0 / r - v2 / MU;
            if (a_inv <= 0) return 3600 * 24; // Parabolic/Hyperbolic fallback
            double a = 1.0 / a_inv;
            return 2.0 * Math.PI * Math.Sqrt(a * a * a / MU);
        }

        static double EstimateDeltaV(State s1, State s2, double dt)
        {
            Vector3 v1t, v2t;
            if (SolveLambert(s1.R, s2.R, dt, out v1t, out v2t))
            {
                return (v1t - s1.V).Length() + (v2t - s2.V).Length();
            }
            // Fallback: simplified metric if Lambert fails
            double dist = (s1.R - s2.R).Length();
            double dv_approx = (s1.V - s2.V).Length() + dist * 0.001;
            return dv_approx;
        }

        static bool SolveLambert(Vector3 r1, Vector3 r2, double dt, out Vector3 v1, out Vector3 v2)
        {
            v1 = v2 = new Vector3();
            double r1m = r1.Length();
            double r2m = r2.Length();
            double dot = Vector3.Dot(r1, r2) / (r1m * r2m);
            dot = Math.Max(-1.0, Math.Min(1.0, dot));
            double theta = Math.Acos(dot);

            // Short-way assumption
            double A = Math.Sin(theta) * Math.Sqrt(r1m * r2m / (1.0 - dot));
            if (Math.Abs(A) < 1e-7) return false;

            double zLow = -4.0 * Math.PI * Math.PI;
            double zHigh = Math.PI * Math.PI - 0.1;
            double z = 0;

            for (int i = 0; i < 40; i++)
            {
                z = (zLow + zHigh) / 2.0;
                double c2, c3;
                GetStumpff(z, out c2, out c3);
                
                double y = r1m + r2m + A * (z * c3 - 1.0) / Math.Sqrt(c2);
                if (A > 0 && y < 0)
                {
                    // If y is negative, we need to adjust z to stay in the valid region
                    // For short-way A > 0, y increases as z increases (generally)
                    zLow = z;
                    continue;
                }

                double x = Math.Sqrt(y / c2);
                double t = (x * x * x * c3 + A * Math.Sqrt(y)) / Math.Sqrt(MU);

                if (t < dt) zLow = z;
                else zHigh = z;

                if (Math.Abs(t - dt) < 1e-6 * dt) break;
            }

            double c2_f, c3_f;
            GetStumpff(z, out c2_f, out c3_f);
            double y_f = r1m + r2m + A * (z * c3_f - 1.0) / Math.Sqrt(c2_f);
            double f = 1.0 - y_f / r1m;
            double g = A * Math.Sqrt(y_f / MU);
            double gDot = 1.0 - y_f / r2m;

            v1 = (r2 - r1 * f) / g;
            v2 = (r2 * gDot - r1) / g;
            return true;
        }

        static void GetStumpff(double z, out double c2, out double c3)
        {
            if (z > 1e-6)
            {
                double sz = Math.Sqrt(z);
                c2 = (1.0 - Math.Cos(sz)) / z;
                c3 = (sz - Math.Sin(sz)) / (sz * z);
            }
            else if (z < -1e-6)
            {
                double sz = Math.Sqrt(-z);
                c2 = (Math.Cosh(sz) - 1.0) / (-z);
                c3 = (Math.Sinh(sz) - sz) / (sz * (-z));
            }
            else
            {
                c2 = 0.5 - z / 24.0 + z * z / 720.0;
                c3 = 1.0 / 6.0 - z / 120.0 + z * z / 5040.0;
            }
        }

        static int[] SolveTSP(int N, double[,] costs)
        {
            // Initial path: Start(0) -> 1 -> 2 -> ... -> N
            int[] bestPath = new int[N + 1];
            for (int i = 0; i <= N; i++) bestPath[i] = i;
            
            // Initial Nearest Neighbor to get a better start
            bool[] visited = new bool[N + 1];
            visited[0] = true;
            int curr = 0;
            for (int i = 1; i <= N; i++)
            {
                int next = -1;
                double minDist = double.MaxValue;
                for (int j = 1; j <= N; j++)
                {
                    if (!visited[j] && costs[curr, j] < minDist)
                    {
                        minDist = costs[curr, j];
                        next = j;
                    }
                }
                bestPath[i] = next;
                visited[next] = true;
                curr = next;
            }

            double bestCost = CalculatePathCost(bestPath, costs);

            // 2-opt Refinement
            bestPath = Refine2Opt(bestPath, costs);
            bestCost = CalculatePathCost(bestPath, costs);

            // Try some random restarts to avoid local minima
            Random rng = new Random(42);
            int restarts = (N > 50) ? 200 : 500;
            if (N < 10) restarts = 10;

            for (int r = 0; r < restarts; r++)
            {
                int[] path = new int[N + 1];
                path[0] = 0;
                List<int> stations = Enumerable.Range(1, N).OrderBy(x => rng.Next()).ToList();
                for (int i = 0; i < N; i++) path[i + 1] = stations[i];

                path = Refine2Opt(path, costs);
                double cost = CalculatePathCost(path, costs);
                if (cost < bestCost)
                {
                    bestCost = cost;
                    bestPath = path;
                }
            }

            // Return indices 0..N-1
            int[] result = new int[N];
            for (int i = 0; i < N; i++) result[i] = bestPath[i + 1] - 1;
            return result;
        }

        static int[] Refine2Opt(int[] path, double[,] costs)
        {
            int n = path.Length;
            bool improved = true;
            while (improved)
            {
                improved = false;
                for (int i = 1; i < n - 1; i++)
                {
                    for (int j = i + 1; j < n; j++)
                    {
                        // Current edges: (i-1, i) and (j, j+1 if exists)
                        // New edges: (i-1, j) and (i, j+1 if exists)
                        
                        double d_old = costs[path[i - 1], path[i]];
                        double d_new = costs[path[i - 1], path[j]];
                        
                        if (j < n - 1)
                        {
                            d_old += costs[path[j], path[j + 1]];
                            d_new += costs[path[i], path[j + 1]];
                        }

                        if (d_new < d_old - 1e-7)
                        {
                            Reverse(path, i, j);
                            improved = true;
                        }
                    }
                }
            }
            return path;
        }

        static void Reverse(int[] path, int i, int j)
        {
            while (i < j)
            {
                int temp = path[i];
                path[i] = path[j];
                path[j] = temp;
                i++;
                j--;
            }
        }

        static double CalculatePathCost(int[] path, double[,] costs)
        {
            double cost = 0;
            for (int i = 0; i < path.Length - 1; i++)
            {
                cost += costs[path[i], path[i + 1]];
            }
            return cost;
        }
    }
}
