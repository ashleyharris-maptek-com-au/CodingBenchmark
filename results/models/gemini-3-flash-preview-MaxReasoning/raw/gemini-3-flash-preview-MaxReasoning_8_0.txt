use std::collections::VecDeque;
use std::io::{self, BufRead, Write};

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut stdin_lock = stdin.lock();
    let mut line = String::new();
    
    // Read H W
    if stdin_lock.read_line(&mut line)? == 0 {
        return Ok(());
    }
    let mut parts = line.split_whitespace();
    let h: usize = match parts.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => return Ok(()),
    };
    let w: usize = match parts.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => return Ok(()),
    };

    if h == 0 || w == 0 {
        return Ok(());
    }

    // parent array stores:
    // 0: Unvisited path (' ', 'A', 'B')
    // 1: Backtrack Up (Parent is at r-1)
    // 2: Backtrack Down (Parent is at r+1)
    // 3: Backtrack Left (Parent is at c-1)
    // 4: Backtrack Right (Parent is at c+1)
    // 5: Start marker
    // 6: Wall ('#')
    let mut parent = vec![6u8; h * w];
    let mut start = (0u16, 0u16);
    let mut end = (0u16, 0u16);
    let mut start_found = false;
    let mut end_found = false;

    // Wrap in a larger BufReader for performance on huge inputs
    let mut reader = io::BufReader::with_capacity(1024 * 1024, stdin_lock);
    let mut line_buf = Vec::with_capacity(w + 128);
    for r in 0..h {
        line_buf.clear();
        if reader.read_until(b'\n', &mut line_buf)? == 0 {
            break;
        }
        let mut col = 0;
        for &b in &line_buf {
            if col >= w { break; }
            if b == b'\r' || b == b'\n' { continue; }
            
            let idx = r * w + col;
            match b {
                b'A' => {
                    start = (r as u16, col as u16);
                    parent[idx] = 0;
                    start_found = true;
                }
                b'B' => {
                    end = (r as u16, col as u16);
                    parent[idx] = 0;
                    end_found = true;
                }
                b' ' => {
                    parent[idx] = 0;
                }
                _ => {
                    // Default is 6 (Wall)
                }
            }
            col += 1;
        }
    }

    if !start_found || !end_found {
        println!("0");
        return Ok(());
    }

    // BFS
    let mut queue = VecDeque::with_capacity(1024 * 1024);
    queue.push_back(start);
    
    let start_idx = start.0 as usize * w + start.1 as usize;
    parent[start_idx] = 5; // Mark start as visited

    let dr = [-1, 1, 0, 0];
    let dc = [0, 0, -1, 1];
    // Movement Up (-1, 0) -> To backtrack to parent, go Down (2)
    // Movement Down (1, 0) -> To backtrack to parent, go Up (1)
    // Movement Left (0, -1) -> To backtrack to parent, go Right (4)
    // Movement Right (0, 1) -> To backtrack to parent, go Left (3)
    let backtrack_dir = [2, 1, 4, 3];

    let mut found = false;
    while let Some((r, c)) = queue.pop_front() {
        if (r, c) == end {
            found = true;
            break;
        }

        for i in 0..4 {
            let nr = r as i32 + dr[i];
            let nc = c as i32 + dc[i];

            if nr >= 0 && nr < h as i32 && nc >= 0 && nc < w as i32 {
                let nr = nr as usize;
                let nc = nc as usize;
                let idx = nr * w + nc;
                if parent[idx] == 0 {
                    parent[idx] = backtrack_dir[i];
                    queue.push_back((nr as u16, nc as u16));
                }
            }
        }
    }

    if found {
        let mut path = Vec::new();
        let mut curr = end;
        while curr != start {
            path.push(curr);
            let idx = curr.0 as usize * w + curr.1 as usize;
            let dir = parent[idx];
            match dir {
                1 => curr.0 -= 1,
                2 => curr.0 += 1,
                3 => curr.1 -= 1,
                4 => curr.1 += 1,
                _ => break, // Should not happen if found
            }
        }
        path.push(start);

        let stdout = io::stdout();
        let mut out = io::BufWriter::new(stdout.lock());
        writeln!(out, "{}", path.len())?;
        for pos in path.iter().rev() {
            writeln!(out, "{} {}", pos.1, pos.0)?;
        }
    } else {
        println!("0");
    }

    Ok(())
}
