using System;
using System.Collections.Generic;
using System.Linq;

namespace TetrisAI
{
    public class Program
    {
        public static void Main(string[] args)
        {
            string line = Console.ReadLine();
            if (string.IsNullOrEmpty(line)) return;
            string[] parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length < 3) return;

            if (!int.TryParse(parts[0], out int width)) return;
            if (!int.TryParse(parts[1], out int height)) return;
            if (!int.TryParse(parts[2], out int numPieces)) return;

            Game game = new Game(width, height, numPieces);
            game.Run();
        }
    }

    public struct Point
    {
        public int X, Y;
        public Point(int x, int y) { X = x; Y = y; }
    }

    public class Move
    {
        public int Rotation;
        public int Column;
        public Move(int r, int c) { Rotation = r; Column = c; }
    }

    public class Game
    {
        private int width;
        private int height;
        private int totalPieces;
        private Board board;
        private Dictionary<string, Point[][]> pieceRotations;

        public Game(int w, int h, int total)
        {
            width = w;
            height = h;
            totalPieces = total;
            board = new Board(w, h);
            InitializePieces();
        }

        private void InitializePieces()
        {
            pieceRotations = new Dictionary<string, Point[][]>();
            AddPiece("I", new[] { new Point(0, 0), new Point(1, 0), new Point(2, 0), new Point(3, 0) });
            AddPiece("O", new[] { new Point(0, 0), new Point(1, 0), new Point(0, 1), new Point(1, 1) });
            AddPiece("T", new[] { new Point(1, 0), new Point(0, 1), new Point(1, 1), new Point(2, 1) });
            AddPiece("S", new[] { new Point(1, 0), new Point(2, 0), new Point(0, 1), new Point(1, 1) });
            AddPiece("Z", new[] { new Point(0, 0), new Point(1, 0), new Point(1, 1), new Point(2, 1) });
            AddPiece("J", new[] { new Point(0, 0), new Point(0, 1), new Point(1, 1), new Point(2, 1) });
            AddPiece("L", new[] { new Point(2, 0), new Point(0, 1), new Point(1, 1), new Point(2, 1) });
        }

        private void AddPiece(string name, Point[] r0)
        {
            Point[][] rotations = new Point[4][];
            rotations[0] = Normalize(r0);
            rotations[1] = Normalize(RotateCW(rotations[0]));
            rotations[2] = Normalize(RotateCW(rotations[1]));
            rotations[3] = Normalize(RotateCW(rotations[2]));
            pieceRotations[name] = rotations;
        }

        private Point[] RotateCW(Point[] points)
        {
            return points.Select(p => new Point(-p.Y, p.X)).ToArray();
        }

        private Point[] Normalize(Point[] points)
        {
            int minX = points.Min(p => p.X);
            int minY = points.Min(p => p.Y);
            return points.Select(p => new Point(p.X - minX, p.Y - minY)).ToArray();
        }

        public void Run()
        {
            for (int i = 0; i < totalPieces; i++)
            {
                string pieceType = Console.ReadLine()?.Trim();
                if (string.IsNullOrEmpty(pieceType)) break;

                Move bestMove = FindBestMove(pieceType);
                Console.WriteLine($"{bestMove.Rotation} {bestMove.Column}");

                string result = Console.ReadLine();
                if (string.IsNullOrEmpty(result) || result == "gameover") break;

                board.ApplyMove(pieceRotations[pieceType][bestMove.Rotation], bestMove.Column);
            }
        }

        private Move FindBestMove(string pieceType)
        {
            if (!pieceRotations.ContainsKey(pieceType)) return new Move(0, 0);

            Point[][] rotations = pieceRotations[pieceType];
            double bestScore = double.NegativeInfinity;
            Move bestMove = new Move(0, 0);

            for (int r = 0; r < 4; r++)
            {
                Point[] piece = rotations[r];
                int pieceWidth = piece.Max(p => p.X) + 1;

                for (int c = 0; c <= width - pieceWidth; c++)
                {
                    int landingY = board.FindLandingY(piece, c);
                    if (landingY < 0) continue;

                    Board nextBoard = board.Clone();
                    int linesCleared = nextBoard.ApplyMove(piece, c);
                    
                    double score = Evaluate(nextBoard, linesCleared, landingY);
                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestMove = new Move(r, c);
                    }
                }
            }

            if (bestScore == double.NegativeInfinity) return new Move(0, 0);
            return bestMove;
        }

        private double Evaluate(Board b, int linesCleared, int landingY)
        {
            int holes = b.CountHoles();
            int aggregateHeight = b.GetAggregateHeight();
            int bumpiness = b.GetBumpiness();
            int rowTrans = b.GetRowTransitions();
            int colTrans = b.GetColTransitions();
            int wells = b.GetWellSum();
            int maxHeight = b.GetMaxHeight();

            // Dellacherie-inspired weights
            double wHoles = -10.0;
            double wHeight = -0.5;
            double wBump = -0.5;
            double wRowTrans = -1.0;
            double wColTrans = -1.0;
            double wWells = -1.0;
            double wLines = 2.0;
            double wLanding = -1.0;

            // Adaptive Logic based on game complexity and state
            if (totalPieces < 50) 
            {
                wLines = 10.0; // In short games, prioritize immediate line clears
            }
            else if (totalPieces > 2000)
            {
                wHoles = -25.0; // In very long games, holes are extremely dangerous
                wWells = -2.0;
                wBump = -1.0;
            }

            if (maxHeight > height * 0.7)
            {
                wLines = 20.0; // Danger: clear lines to reduce height
                wHeight = -2.5;
                wHoles = -40.0;
            }

            if (width < 8)
            {
                wLines = 8.0; // Narrow boards require more aggressive clearing
                wWells = -3.0;
                wColTrans = -2.0;
            }

            double score = 0;
            score += holes * wHoles;
            score += aggregateHeight * wHeight;
            score += bumpiness * wBump;
            score += rowTrans * wRowTrans;
            score += colTrans * wColTrans;
            score += wells * wWells;
            score += linesCleared * wLines;
            score += (height - landingY) * wLanding;

            return score;
        }
    }

    public class Board
    {
        public int Width;
        public int Height;
        public bool[,] Cells;

        public Board(int w, int h)
        {
            Width = w;
            Height = h;
            Cells = new bool[w, h];
        }

        public Board Clone()
        {
            Board b = new Board(Width, Height);
            for (int x = 0; x < Width; x++)
            {
                for (int y = 0; y < Height; y++)
                {
                    b.Cells[x, y] = Cells[x, y];
                }
            }
            return b;
        }

        public int FindLandingY(Point[] piece, int col)
        {
            if (IsCollision(piece, col, 0)) return -1;
            for (int y = 0; y < Height - GetPieceHeight(piece); y++)
            {
                if (IsCollision(piece, col, y + 1)) return y;
            }
            return Height - GetPieceHeight(piece);
        }

        private bool IsCollision(Point[] piece, int col, int row)
        {
            foreach (var p in piece)
            {
                int x = col + p.X;
                int y = row + p.Y;
                if (x < 0 || x >= Width || y >= Height) return true;
                if (y >= 0 && Cells[x, y]) return true;
            }
            return false;
        }

        private int GetPieceHeight(Point[] piece)
        {
            int max = 0;
            foreach (var p in piece) if (p.Y > max) max = p.Y;
            return max + 1;
        }

        public int ApplyMove(Point[] piece, int col)
        {
            int y = FindLandingY(piece, col);
            if (y < 0) return 0;
            foreach (var p in piece)
            {
                int bx = col + p.X;
                int by = y + p.Y;
                if (bx >= 0 && bx < Width && by >= 0 && by < Height)
                    Cells[bx, by] = true;
            }
            return ClearFullLines();
        }

        private int ClearFullLines()
        {
            int cleared = 0;
            for (int y = Height - 1; y >= 0; y--)
            {
                bool full = true;
                for (int x = 0; x < Width; x++)
                {
                    if (!Cells[x, y]) { full = false; break; }
                }
                if (full)
                {
                    cleared++;
                    for (int ty = y; ty > 0; ty--)
                        for (int tx = 0; tx < Width; tx++)
                            Cells[tx, ty] = Cells[tx, ty - 1];
                    for (int tx = 0; tx < Width; tx++) Cells[tx, 0] = false;
                    y++;
                }
            }
            return cleared;
        }

        public int GetMaxHeight()
        {
            for (int y = 0; y < Height; y++)
                for (int x = 0; x < Width; x++)
                    if (Cells[x, y]) return Height - y;
            return 0;
        }

        public int GetAggregateHeight()
        {
            int sum = 0;
            for (int x = 0; x < Width; x++) sum += GetColHeight(x);
            return sum;
        }

        private int GetColHeight(int x)
        {
            for (int y = 0; y < Height; y++)
                if (Cells[x, y]) return Height - y;
            return 0;
        }

        public int CountHoles()
        {
            int holes = 0;
            for (int x = 0; x < Width; x++)
            {
                bool block = false;
                for (int y = 0; y < Height; y++)
                {
                    if (Cells[x, y]) block = true;
                    else if (block) holes++;
                }
            }
            return holes;
        }

        public int GetBumpiness()
        {
            int bump = 0;
            for (int x = 0; x < Width - 1; x++)
                bump += Math.Abs(GetColHeight(x) - GetColHeight(x + 1));
            return bump;
        }

        public int GetRowTransitions()
        {
            int trans = 0;
            for (int y = 0; y < Height; y++)
            {
                bool last = true; // Left boundary is filled
                for (int x = 0; x < Width; x++)
                {
                    if (Cells[x, y] != last) trans++;
                    last = Cells[x, y];
                }
                if (!last) trans++; // Right boundary is filled
            }
            return trans;
        }

        public int GetColTransitions()
        {
            int trans = 0;
            for (int x = 0; x < Width; x++)
            {
                bool last = true; // Bottom boundary is filled
                for (int y = Height - 1; y >= 0; y--)
                {
                    if (Cells[x, y] != last) trans++;
                    last = Cells[x, y];
                }
                if (last) trans++; // Top boundary is empty
            }
            return trans;
        }

        public int GetWellSum()
        {
            int sum = 0;
            for (int x = 0; x < Width; x++)
            {
                for (int y = 0; y < Height; y++)
                {
                    if (!Cells[x, y])
                    {
                        bool left = (x == 0) || Cells[x - 1, y];
                        bool right = (x == Width - 1) || Cells[x + 1, y];
                        if (left && right)
                        {
                            int depth = 0;
                            for (int ty = y; ty < Height && !Cells[x, ty]; ty++)
                            {
                                bool tLeft = (x == 0) || Cells[x - 1, ty];
                                bool tRight = (x == Width - 1) || Cells[x + 1, ty];
                                if (tLeft && tRight) depth++;
                                else break;
                            }
                            sum += depth * (depth + 1) / 2;
                            y += depth - 1;
                        }
                    }
                }
            }
            return sum;
        }
    }
}
