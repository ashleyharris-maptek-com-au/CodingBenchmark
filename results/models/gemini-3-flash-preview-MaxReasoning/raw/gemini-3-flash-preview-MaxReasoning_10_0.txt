#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct Rectangle {
    int w, h, id;
};

struct Placement {
    int board_idx, x, y;
    bool rotated;
};

struct SkylineNode {
    int x, y, w;
};

struct WasteRect {
    int x, y, w, h;
};

struct Board {
    int width, height;
    long long remaining_area;
    vector<SkylineNode> skyline;
    vector<WasteRect> waste;

    Board(int w, int h) : width(w), height(h), remaining_area((long long)w * h) {
        skyline.push_back({0, 0, w});
    }

    bool search_waste(int rw, int rh, int& px, int& py) const {
        int best_y = 2000000000;
        int best_x = 2000000000;
        bool found = false;
        for (const auto& r : waste) {
            if (r.w >= rw && r.h >= rh) {
                if (r.y < best_y || (r.y == best_y && r.x < best_x)) {
                    best_y = r.y;
                    best_x = r.x;
                    found = true;
                }
            }
        }
        if (found) {
            px = best_x;
            py = best_y;
        }
        return found;
    }

    void use_waste(int px, int py, int rw, int rh) {
        for (int i = 0; i < (int)waste.size(); ++i) {
            if (waste[i].x == px && waste[i].y == py && waste[i].w >= rw && waste[i].h >= rh) {
                WasteRect r = waste[i];
                waste.erase(waste.begin() + i);
                remaining_area -= (long long)rw * rh;
                
                if (r.w - rw > r.h - rh) { 
                    if (r.w > rw) waste.push_back({r.x + rw, r.y, r.w - rw, r.h});
                    if (r.h > rh) waste.push_back({r.x, r.y + rh, rw, r.h - rh});
                } else {
                    if (r.h > rh) waste.push_back({r.x, r.y + rh, r.w, r.h - rh});
                    if (r.w > rw) waste.push_back({r.x + rw, r.y, r.w - rw, rh});
                }
                return;
            }
        }
    }

    bool find_skyline(int rw, int rh, int& px, int& py, int& best_idx) const {
        int best_y = 2000000000;
        int best_x = 2000000000;
        best_idx = -1;

        for (int i = 0; i < (int)skyline.size(); ++i) {
            int cy = 0;
            int cw = 0;
            for (int j = i; j < (int)skyline.size(); ++j) {
                if (skyline[j].y > cy) cy = skyline[j].y;
                cw += skyline[j].w;
                if (cw >= rw) break;
            }
            if (cw >= rw && cy + rh <= height) {
                if (cy < best_y || (cy == best_y && skyline[i].x < best_x)) {
                    best_y = cy;
                    best_x = skyline[i].x;
                    best_idx = i;
                }
            }
        }

        if (best_idx != -1) {
            px = best_x;
            py = best_y;
            return true;
        }
        return false;
    }

    void apply_skyline(int best_idx, int rw, int rh, int px, int py) {
        remaining_area -= (long long)rw * rh;

        int cw = 0;
        for (int i = best_idx; i < (int)skyline.size(); ++i) {
            int node_x = skyline[i].x;
            int node_y = skyline[i].y;
            int node_w = skyline[i].w;
            
            int overlap_w = min(node_x + node_w, px + rw) - max(node_x, px);
            if (overlap_w > 0 && py > node_y) {
                waste.push_back({max(node_x, px), node_y, overlap_w, py - node_y});
            }
            
            cw += node_w;
            if (cw >= rw) break;
        }

        SkylineNode new_node = {px, py + rh, rw};
        int nodes_to_remove = 0;
        cw = 0;
        for (int i = best_idx; i < (int)skyline.size(); ++i) {
            cw += skyline[i].w;
            nodes_to_remove++;
            if (cw >= rw) break;
        }

        if (cw > rw) {
            skyline[best_idx + nodes_to_remove - 1].x += (rw - (cw - skyline[best_idx + nodes_to_remove - 1].w));
            skyline[best_idx + nodes_to_remove - 1].w = cw - rw;
            nodes_to_remove--;
        }

        if (nodes_to_remove > 0) {
            skyline.erase(skyline.begin() + best_idx, skyline.begin() + best_idx + nodes_to_remove);
        }
        skyline.insert(skyline.begin() + best_idx, new_node);

        for (int i = 0; i < (int)skyline.size() - 1; ++i) {
            if (skyline[i].y == skyline[i+1].y) {
                skyline[i].w += skyline[i+1].w;
                skyline.erase(skyline.begin() + i + 1);
                --i;
            }
        }
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    int N, board_w, board_h;
    if (!(cin >> N >> board_w >> board_h)) return 0;

    vector<Rectangle> rects(N);
    for (int i = 0; i < N; ++i) {
        cin >> rects[i].w >> rects[i].h;
        rects[i].id = i;
    }

    sort(rects.begin(), rects.end(), [](const Rectangle& a, const Rectangle& b) {
        int d1 = max(a.w, a.h);
        int d2 = max(b.w, b.h);
        if (d1 != d2) return d1 > d2;
        long long area1 = (long long)a.w * a.h;
        long long area2 = (long long)b.w * b.h;
        return area1 > area2;
    });

    vector<Placement> results(N);
    vector<Board> boards;
    boards.reserve(100);

    for (int i = 0; i < N; ++i) {
        int rw = rects[i].w;
        int rh = rects[i].h;
        long long r_area = (long long)rw * rh;

        int found_board = -1;
        int found_x = 0, found_y = 0;
        bool found_rotated = false;
        int found_skyline_idx = -1;
        bool found_from_waste = false;

        for (int b = 0; b < (int)boards.size(); ++b) {
            if (r_area > boards[b].remaining_area) continue;

            int wx1, wy1;
            bool wf1 = boards[b].search_waste(rw, rh, wx1, wy1);
            int wx2, wy2;
            bool wf2 = (rw != rh) && boards[b].search_waste(rh, rw, wx2, wy2);

            if (wf1 && wf2) {
                found_board = b; found_from_waste = true;
                if (wy1 < wy2 || (wy1 == wy2 && wx1 < wx2)) {
                    found_x = wx1; found_y = wy1; found_rotated = false;
                } else {
                    found_x = wx2; found_y = wy2; found_rotated = true;
                }
                break;
            } else if (wf1) {
                found_board = b; found_from_waste = true; found_x = wx1; found_y = wy1; found_rotated = false;
                break;
            } else if (wf2) {
                found_board = b; found_from_waste = true; found_x = wx2; found_y = wy2; found_rotated = true;
                break;
            }

            int sx1, sy1, si1;
            bool sf1 = boards[b].find_skyline(rw, rh, sx1, sy1, si1);
            int sx2, sy2, si2;
            bool sf2 = (rw != rh) && boards[b].find_skyline(rh, rw, sx2, sy2, si2);

            if (sf1 && sf2) {
                found_board = b;
                if (sy1 < sy2 || (sy1 == sy2 && sx1 < sx2)) {
                    found_x = sx1; found_y = sy1; found_rotated = false; found_skyline_idx = si1;
                } else {
                    found_x = sx2; found_y = sy2; found_rotated = true; found_skyline_idx = si2;
                }
                break;
            } else if (sf1) {
                found_board = b; found_x = sx1; found_y = sy1; found_rotated = false; found_skyline_idx = si1;
                break;
            } else if (sf2) {
                found_board = b; found_x = sx2; found_y = sy2; found_rotated = true; found_skyline_idx = si2;
                break;
            }
        }

        if (found_board == -1) {
            boards.emplace_back(board_w, board_h);
            found_board = (int)boards.size() - 1;
            
            int sx1, sy1, si1;
            bool sf1 = boards[found_board].find_skyline(rw, rh, sx1, sy1, si1);
            int sx2, sy2, si2;
            bool sf2 = (rw != rh) && boards[found_board].find_skyline(rh, rw, sx2, sy2, si2);
            
            if (sf1 && sf2) {
                if (rh < rw) {
                    found_x = sx2; found_y = sy2; found_rotated = true; found_skyline_idx = si2;
                } else {
                    found_x = sx1; found_y = sy1; found_rotated = false; found_skyline_idx = si1;
                }
            } else if (sf1) {
                found_x = sx1; found_y = sy1; found_rotated = false; found_skyline_idx = si1;
            } else if (sf2) {
                found_x = sx2; found_y = sy2; found_rotated = true; found_skyline_idx = si2;
            } else {
                found_x = 0; found_y = 0; found_rotated = false; found_skyline_idx = -1;
            }
        }

        if (found_from_waste) {
            boards[found_board].use_waste(found_x, found_y, found_rotated ? rh : rw, found_rotated ? rw : rh);
        } else if (found_skyline_idx != -1) {
            boards[found_board].apply_skyline(found_skyline_idx, found_rotated ? rh : rw, found_rotated ? rw : rh, found_x, found_y);
        }

        results[rects[i].id] = {found_board, found_x, found_y, found_rotated};
    }

    cout << boards.size() << "\n";
    for (int i = 0; i < N; ++i) {
        cout << results[i].board_idx << " " << results[i].x << " " << results[i].y << " " << (results[i].rotated ? 1 : 0) << "\n";
    }

    return 0;
}
