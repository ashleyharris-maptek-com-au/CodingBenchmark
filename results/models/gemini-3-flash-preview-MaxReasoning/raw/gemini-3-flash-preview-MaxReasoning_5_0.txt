use std::io::{self, BufRead};
use std::collections::VecDeque;

struct Xorshift {
    state: u64,
}

impl Xorshift {
    fn new(seed: u64) -> Self {
        Xorshift { state: seed }
    }
    fn next(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }
    fn next_usize(&mut self, n: usize) -> usize {
        if n == 0 { return 0; }
        (self.next() as usize) % n
    }
}

fn shuffle<T>(slice: &mut [T], rng: &mut Xorshift) {
    for i in (1..slice.len()).rev() {
        let j = rng.next_usize(i + 1);
        slice.swap(i, j);
    }
}

struct Solver {
    width: i32,
    height: i32,
    grid: Vec<bool>,
    degrees: Vec<u8>,
    n0: i32,
    n1: i32,
    require_cycle: bool,
    path: Vec<(i32, i32)>,
    unvisited_cells: Vec<i32>,
    cell_to_unvisited_idx: Vec<usize>,
    unv_x: Vec<i32>,
    unv_y: Vec<i32>,
    is_adj_start: Vec<bool>,
    rng: Xorshift,
    bfs_visited: Vec<bool>,
    bfs_queue: VecDeque<i32>,
}

impl Solver {
    fn new(width: i32, height: i32, obstacles: Vec<(i32, i32)>, require_cycle: bool) -> Self {
        let mut grid = vec![false; (width * height) as usize];
        for (x, y) in obstacles {
            if x >= 0 && x < width && y >= 0 && y < height {
                grid[(x * height + y) as usize] = true;
            }
        }

        let mut unvisited_cells = Vec::new();
        let mut cell_to_unvisited_idx = vec![0; (width * height) as usize];
        let mut unv_x = vec![0; (width * height) as usize];
        let mut unv_y = vec![0; (width * height) as usize];
        let mut is_adj_start = vec![false; (width * height) as usize];
        let mut n0 = 0;
        let mut n1 = 0;

        for x in 0..width {
            for y in 0..height {
                let idx = (x * height + y) as usize;
                if !grid[idx] {
                    cell_to_unvisited_idx[idx] = unvisited_cells.len();
                    unvisited_cells.push(idx as i32);
                    unv_x[idx] = x;
                    unv_y[idx] = y;
                    if (x - 0).abs() + (y - 0).abs() == 1 {
                        is_adj_start[idx] = true;
                    }
                    if (x + y) % 2 == 0 { n0 += 1; } else { n1 += 1; }
                }
            }
        }

        let mut degrees = vec![0; (width * height) as usize];
        let dx = [1, -1, 0, 0];
        let dy = [0, 0, 1, -1];
        for &idx in &unvisited_cells {
            let x = idx / height;
            let y = idx % height;
            let mut d = 0;
            for i in 0..4 {
                let nx = x + dx[i];
                let ny = y + dy[i];
                if nx >= 0 && nx < width && ny >= 0 && ny < height {
                    if !grid[(nx * height + ny) as usize] {
                        d += 1;
                    }
                }
            }
            degrees[idx as usize] = d;
        }

        let total_unvisited = unvisited_cells.len();
        Solver {
            width,
            height,
            grid,
            degrees,
            n0,
            n1,
            require_cycle,
            path: Vec::with_capacity(total_unvisited),
            unvisited_cells,
            cell_to_unvisited_idx,
            unv_x,
            unv_y,
            is_adj_start,
            rng: Xorshift::new(42),
            bfs_visited: vec![false; (width * height) as usize],
            bfs_queue: VecDeque::with_capacity(total_unvisited),
        }
    }

    fn solve(&mut self) -> bool {
        if self.unvisited_cells.is_empty() { return false; }
        if self.grid[0] { return false; }

        if self.require_cycle {
            if (self.n0 + self.n1) % 2 != 0 || self.n0 != self.n1 { return false; }
        } else {
            if self.n0 != self.n1 && self.n0 != self.n1 + 1 { return false; }
        }

        self.make_move(0, 0);
        if self.backtrack(0, 0) {
            return true;
        }
        false
    }

    fn check_connectivity(&mut self) -> bool {
        if self.unvisited_cells.is_empty() { return true; }
        let start_idx = self.unvisited_cells[0] as usize;
        self.bfs_visited.fill(false);
        self.bfs_visited[start_idx] = true;
        self.bfs_queue.clear();
        self.bfs_queue.push_back(start_idx as i32);
        
        let mut count = 1;
        let dx = [1, -1, 0, 0];
        let dy = [0, 0, 1, -1];
        
        while let Some(idx) = self.bfs_queue.pop_front() {
            let uidx_u = idx as usize;
            let ux = self.unv_x[uidx_u];
            let uy = self.unv_y[uidx_u];
            for i in 0..4 {
                let nx = ux + dx[i];
                let ny = uy + dy[i];
                if nx >= 0 && nx < self.width && ny >= 0 && ny < self.height {
                    let nidx = (nx * self.height + ny) as usize;
                    if !self.grid[nidx] && !self.bfs_visited[nidx] {
                        self.bfs_visited[nidx] = true;
                        self.bfs_queue.push_back(nidx as i32);
                        count += 1;
                    }
                }
            }
        }
        
        if count != self.unvisited_cells.len() { return false; }
        
        if self.require_cycle {
            let mut adj_start = false;
            for &uidx in &self.unvisited_cells {
                if self.is_adj_start[uidx as usize] {
                    adj_start = true;
                    break;
                }
            }
            if !adj_start { return false; }
        }
        true
    }

    fn backtrack(&mut self, x: i32, y: i32) -> bool {
        if self.unvisited_cells.is_empty() {
            if self.require_cycle {
                return (x - 0).abs() + (y - 0).abs() == 1;
            } else {
                return true;
            }
        }

        let mut count_deg1 = 0;
        let mut deg1_idx = -1;
        let mut forced_next = None;

        for &uidx in &self.unvisited_cells {
            let uidx_u = uidx as usize;
            let d = self.degrees[uidx_u];
            let ux = self.unv_x[uidx_u];
            let uy = self.unv_y[uidx_u];
            let is_adj_curr = (ux - x).abs() + (uy - y).abs() == 1;
            
            let mut real_d = d;
            if is_adj_curr { real_d += 1; }
            if self.require_cycle {
                if self.is_adj_start[uidx_u] { real_d += 1; }
            }
            
            if real_d == 0 { return false; }
            if real_d == 1 {
                count_deg1 += 1;
                deg1_idx = uidx;
                if is_adj_curr {
                    if forced_next.is_some() && forced_next != Some((ux, uy)) { return false; }
                    forced_next = Some((ux, uy));
                }
            }
        }

        if self.require_cycle {
            if count_deg1 > 0 { return false; }
        } else {
            if count_deg1 > 1 { return false; }
        }

        if let Some((nx, ny)) = forced_next {
            if self.try_move(nx, ny) { return true; }
            return false;
        }

        if !self.check_connectivity() { return false; }

        let mut candidates = Vec::with_capacity(4);
        let dx = [1, -1, 0, 0];
        let dy = [0, 0, 1, -1];
        for i in 0..4 {
            let nx = x + dx[i];
            let ny = y + dy[i];
            if nx >= 0 && nx < self.width && ny >= 0 && ny < self.height {
                let nidx = (nx * self.height + ny) as usize;
                if !self.grid[nidx] {
                    if !self.require_cycle && count_deg1 == 1 {
                        if nidx as i32 == deg1_idx {
                            if self.unvisited_cells.len() > 1 { continue; }
                        }
                    }
                    candidates.push((nx, ny, self.degrees[nidx]));
                }
            }
        }

        if candidates.is_empty() { return false; }

        if candidates.len() > 1 {
            let mid_x = self.width / 2;
            let mid_y = self.height / 2;
            shuffle(&mut candidates, &mut self.rng);
            candidates.sort_by(|a, b| {
                let res = a.2.cmp(&b.2);
                if res == std::cmp::Ordering::Equal {
                    let da = (a.0 - mid_x).pow(2) + (a.1 - mid_y).pow(2);
                    let db = (b.0 - mid_x).pow(2) + (b.1 - mid_y).pow(2);
                    db.cmp(&da)
                } else {
                    res
                }
            });
        }

        for (nx, ny, _) in candidates {
            if self.try_move(nx, ny) { return true; }
        }

        false
    }

    fn try_move(&mut self, nx: i32, ny: i32) -> bool {
        let nidx = (nx * self.height + ny) as usize;
        let pos = self.cell_to_unvisited_idx[nidx];
        
        self.grid[nidx] = true;
        self.path.push((nx, ny));
        let last_idx = self.unvisited_cells.len() - 1;
        let last_cell_idx = self.unvisited_cells[last_idx];
        self.unvisited_cells[pos] = last_cell_idx;
        self.cell_to_unvisited_idx[last_cell_idx as usize] = pos;
        self.unvisited_cells.pop();
        
        if (nx + ny) % 2 == 0 { self.n0 -= 1; } else { self.n1 -= 1; }
        
        let dx = [1, -1, 0, 0];
        let dy = [0, 0, 1, -1];
        for i in 0..4 {
            let nnx = nx + dx[i];
            let nny = ny + dy[i];
            if nnx >= 0 && nnx < self.width && nny >= 0 && nny < self.height {
                let nnidx = (nnx * self.height + nny) as usize;
                if !self.grid[nnidx] { self.degrees[nnidx] -= 1; }
            }
        }

        let mut ok = true;
        if self.require_cycle {
            let curr_color = (nx + ny) % 2;
            if curr_color == 0 {
                if self.n1 != self.n0 + 1 { ok = false; }
            } else {
                if self.n1 != self.n0 { ok = false; }
            }
        } else {
            if (self.n0 - self.n1).abs() > 1 { ok = false; }
        }

        if ok && self.backtrack(nx, ny) { return true; }

        for i in 0..4 {
            let nnx = nx + dx[i];
            let nny = ny + dy[i];
            if nnx >= 0 && nnx < self.width && nny >= 0 && nny < self.height {
                let nnidx = (nnx * self.height + nny) as usize;
                if !self.grid[nnidx] { self.degrees[nnidx] += 1; }
            }
        }
        if (nx + ny) % 2 == 0 { self.n0 += 1; } else { self.n1 += 1; }
        self.unvisited_cells.push(nidx as i32);
        let current_last = self.unvisited_cells.len() - 1;
        let cell_at_pos = self.unvisited_cells[pos];
        self.unvisited_cells[current_last] = cell_at_pos;
        self.cell_to_unvisited_idx[cell_at_pos as usize] = current_last;
        self.unvisited_cells[pos] = nidx as i32;
        self.cell_to_unvisited_idx[nidx] = pos;
        self.path.pop();
        self.grid[nidx] = false;
        
        false
    }

    fn make_move(&mut self, x: i32, y: i32) {
        let idx = (x * self.height + y) as usize;
        let pos = self.cell_to_unvisited_idx[idx];
        self.grid[idx] = true;
        self.path.push((x, y));
        let last_idx = self.unvisited_cells.len() - 1;
        let last_cell_idx = self.unvisited_cells[last_idx];
        self.unvisited_cells[pos] = last_cell_idx;
        self.cell_to_unvisited_idx[last_cell_idx as usize] = pos;
        self.unvisited_cells.pop();
        if (x + y) % 2 == 0 { self.n0 -= 1; } else { self.n1 -= 1; }
        let dx = [1, -1, 0, 0];
        let dy = [0, 0, 1, -1];
        for i in 0..4 {
            let nx = x + dx[i];
            let ny = y + dy[i];
            if nx >= 0 && nx < self.width && ny >= 0 && ny < self.height {
                let nidx = (nx * self.height + ny) as usize;
                if !self.grid[nidx] { self.degrees[nidx] -= 1; }
            }
        }
    }
}

fn run() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    let first_line = match lines.next() { Some(Ok(l)) => l, _ => return };
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    if parts.len() < 4 { return; }
    let width: i32 = parts[0].parse().unwrap_or(0);
    let height: i32 = parts[1].parse().unwrap_or(0);
    let obstacle_count: usize = parts[2].parse().unwrap_or(0);
    let require_cycle: bool = parts[3] == "1";
    let mut obstacles = Vec::with_capacity(obstacle_count);
    for _ in 0..obstacle_count {
        if let Some(Ok(line)) = lines.next() {
            let p: Vec<i32> = line.split_whitespace().filter_map(|s| s.parse().ok()).collect();
            if p.len() >= 2 { obstacles.push((p[0], p[1])); }
        }
    }
    let mut solver = Solver::new(width, height, obstacles, require_cycle);
    if solver.solve() {
        println!("{}", solver.path.len());
        for (x, y) in &solver.path { println!("{} {}", x, y); }
    } else {
        println!("0");
    }
}

fn main() {
    let child = std::thread::Builder::new().stack_size(128 * 1024 * 1024).spawn(run).unwrap();
    child.join().unwrap();
}
