using System;
using System.Collections.Generic;
using System.Linq;
using System.Diagnostics;
using System.Threading;
using System.Threading.Tasks;
using System.Numerics;
using System.IO;
using System.Runtime.CompilerServices;

namespace MaxClique
{
    class Program
    {
        static int numVertices;
        static int[][] adj;
        static ulong[][] bitAdj;
        static bool useBitAdj = false;
        static int bitRowSize;
        static int[] bestClique = new int[0];
        static object lockObj = new object();
        static CancellationTokenSource cts = new CancellationTokenSource();
        static int[] originalLabels;

        static ThreadLocal<ulong[][]> threadBuffers;
        static ThreadLocal<int[]> threadCurrent;

        static void Main(string[] args)
        {
            // Set up timeout at 4 min 50 sec
            var timer = new System.Timers.Timer(290000);
            timer.Elapsed += (s, e) => cts.Cancel();
            timer.AutoReset = false;
            timer.Start();

            if (!ReadAndPrepare()) return;

            // Pre-allocate thread-local buffers
            threadBuffers = new ThreadLocal<ulong[][]>(() =>
            {
                var buffers = new ulong[2048][];
                for (int i = 0; i < buffers.Length; i++) buffers[i] = new ulong[bitRowSize];
                return buffers;
            });
            threadCurrent = new ThreadLocal<int[]>(() => new int[numVertices]);

            // Initial greedy solution to set a lower bound
            bestClique = GetGreedyClique();

            // Start solving
            Solve();

            // Output the result
            Console.WriteLine(bestClique.Length);
            int[] result = bestClique.Select(v => originalLabels[v]).ToArray();
            Array.Sort(result);
            Console.WriteLine(string.Join(" ", result));
        }

        static bool ReadAndPrepare()
        {
            var reader = new FastReader(Console.OpenStandardInput());
            int n = reader.NextInt();
            long m = reader.NextLong();
            if (n <= 0) return false;
            numVertices = n;

            var tempAdj = new List<int>[n];
            for (int i = 0; i < n; i++) tempAdj[i] = new List<int>();

            for (long i = 0; i < m; i++)
            {
                int u = reader.NextInt();
                int v = reader.NextInt();
                if (u < 0 || v < 0 || u >= n || v >= n || u == v) continue;
                tempAdj[u].Add(v);
                tempAdj[v].Add(u);
            }

            // Simple degeneracy-like ordering: sort by degree
            int[] vertices = Enumerable.Range(0, n).ToArray();
            Array.Sort(vertices, (a, b) => tempAdj[a].Count.CompareTo(tempAdj[b].Count));

            originalLabels = vertices;
            int[] newLabels = new int[n];
            for (int i = 0; i < n; i++) newLabels[vertices[i]] = i;

            adj = new int[n][];
            for (int i = 0; i < n; i++)
            {
                int u = vertices[i];
                adj[i] = tempAdj[u].Select(v => newLabels[v]).Distinct().ToArray();
                Array.Sort(adj[i]);
                tempAdj[u] = null;
            }

            bitRowSize = (n + 63) / 64;
            // Use bit matrix if memory allows (approx 1GB limit here, ~100k vertices)
            if (n <= 90000)
            {
                try
                {
                    bitAdj = new ulong[n][];
                    for (int i = 0; i < n; i++)
                    {
                        bitAdj[i] = new ulong[bitRowSize];
                        foreach (int neighbor in adj[i])
                        {
                            bitAdj[i][neighbor >> 6] |= (1UL << (neighbor & 63));
                        }
                    }
                    useBitAdj = true;
                }
                catch (OutOfMemoryException)
                {
                    useBitAdj = false;
                    bitAdj = null;
                }
            }

            return true;
        }

        static int[] GetGreedyClique()
        {
            List<int> clique = new List<int>();
            int n = numVertices;
            // Process from highest degree to lowest
            for (int i = n - 1; i >= 0; i--)
            {
                bool canAdd = true;
                foreach (int c in clique)
                {
                    if (Array.BinarySearch(adj[i], c) < 0)
                    {
                        canAdd = false;
                        break;
                    }
                }
                if (canAdd)
                {
                    clique.Add(i);
                    if (clique.Count > 1000) break;
                }
            }
            return clique.ToArray();
        }

        static void Solve()
        {
            var options = new ParallelOptions
            {
                MaxDegreeOfParallelism = Environment.ProcessorCount,
                CancellationToken = cts.Token
            };

            try
            {
                // Iterate in reverse (highest degrees first) for top level branches
                Parallel.ForEach(Enumerable.Range(0, numVertices).Reverse(), options, i =>
                {
                    if (cts.Token.IsCancellationRequested) return;

                    // Only consider neighbors with higher index to follow degeneracy ordering pruning
                    List<int> candList = new List<int>();
                    foreach (int neighbor in adj[i])
                    {
                        if (neighbor > i) candList.Add(neighbor);
                    }

                    if (candList.Count + 1 > bestClique.Length)
                    {
                        ulong[] candidates = new ulong[bitRowSize];
                        foreach (int c in candList) candidates[c >> 6] |= (1UL << (c & 63));

                        int[] current = threadCurrent.Value;
                        current[0] = i;
                        RecursiveSearch(candidates, current, 1, candList.Count);
                    }
                });
            }
            catch (OperationCanceledException) { }
        }

        static void RecursiveSearch(ulong[] candidates, int[] current, int depth, int candCount)
        {
            if (cts.Token.IsCancellationRequested) return;

            if (candCount == 0)
            {
                if (depth > bestClique.Length)
                {
                    lock (lockObj)
                    {
                        if (depth > bestClique.Length)
                        {
                            int[] res = new int[depth];
                            Array.Copy(current, res, depth);
                            bestClique = res;
                        }
                    }
                }
                return;
            }

            if (depth + candCount <= bestClique.Length) return;

            var buffers = threadBuffers.Value;
            if (depth >= buffers.Length) return; // Should not happen with 2048 limit

            while (candCount > 0)
            {
                if (depth + candCount <= bestClique.Length) return;

                int v = GetLastSetBit(candidates);
                if (v == -1) break;

                // Remove v from candidates
                candidates[v >> 6] &= ~(1UL << (v & 63));
                candCount--;

                ulong[] nextCandidates = buffers[depth];
                int nextCount = 0;

                if (useBitAdj)
                {
                    ulong[] vAdj = bitAdj[v];
                    for (int i = 0; i < bitRowSize; i++)
                    {
                        ulong intersect = candidates[i] & vAdj[i];
                        nextCandidates[i] = intersect;
                        nextCount += BitOperations.PopCount(intersect);
                    }
                }
                else
                {
                    // Clear nextCandidates
                    for (int i = 0; i < bitRowSize; i++) nextCandidates[i] = 0;

                    int vDeg = adj[v].Length;
                    if (vDeg < candCount * 4) // Heuristic to choose between iterating adj or bits
                    {
                        foreach (int neighbor in adj[v])
                        {
                            if ((candidates[neighbor >> 6] & (1UL << (neighbor & 63))) != 0)
                            {
                                nextCandidates[neighbor >> 6] |= (1UL << (neighbor & 63));
                                nextCount++;
                            }
                        }
                    }
                    else
                    {
                        for (int i = 0; i < bitRowSize; i++)
                        {
                            ulong bits = candidates[i];
                            while (bits != 0)
                            {
                                int bit = BitOperations.TrailingZeroCount(bits);
                                int u = (i << 6) | bit;
                                if (Array.BinarySearch(adj[v], u) >= 0)
                                {
                                    nextCandidates[i] |= (1UL << bit);
                                    nextCount++;
                                }
                                bits &= ~(1UL << bit);
                            }
                        }
                    }
                }

                current[depth] = v;
                RecursiveSearch(nextCandidates, current, depth + 1, nextCount);
            }
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        static int GetLastSetBit(ulong[] bits)
        {
            for (int i = bitRowSize - 1; i >= 0; i--)
            {
                if (bits[i] != 0)
                    return (i << 6) | (63 - BitOperations.LeadingZeroCount(bits[i]));
            }
            return -1;
        }
    }

    class FastReader
    {
        private Stream stream;
        private byte[] buffer = new byte[1024 * 1024];
        private int pos, limit;

        public FastReader(Stream s) { stream = s; }

        public int NextInt()
        {
            int res = 0;
            byte b = Read();
            while (b < '0' || b > '9')
            {
                if (b == 0) return -1;
                b = Read();
            }
            while (b >= '0' && b <= '9')
            {
                res = res * 10 + (b - '0');
                b = Read();
            }
            return res;
        }

        public long NextLong()
        {
            long res = 0;
            byte b = Read();
            while (b < '0' || b > '9')
            {
                if (b == 0) return -1;
                b = Read();
            }
            while (b >= '0' && b <= '9')
            {
                res = res * 10 + (b - '0');
                b = Read();
            }
            return res;
        }

        private byte Read()
        {
            if (pos >= limit)
            {
                limit = stream.Read(buffer, 0, buffer.Length);
                pos = 0;
                if (limit == 0) return 0;
            }
            return buffer[pos++];
        }
    }
}
