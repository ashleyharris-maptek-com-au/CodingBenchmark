use std::io::{self, Read, Write, BufWriter};

struct FastReader<R: Read> {
    reader: R,
    buffer: [u8; 1 << 16],
    pos: usize,
    len: usize,
}

impl<R: Read> FastReader<R> {
    fn new(reader: R) -> Self {
        Self {
            reader,
            buffer: [0; 1 << 16],
            pos: 0,
            len: 0,
        }
    }

    fn next_u32(&mut self) -> Option<u32> {
        let mut n = 0;
        let mut found = false;
        loop {
            if self.pos >= self.len {
                let res = self.reader.read(&mut self.buffer);
                match res {
                    Ok(l) if l > 0 => {
                        self.len = l;
                        self.pos = 0;
                    }
                    _ => {
                        return if found { Some(n) } else { None };
                    }
                }
            }
            let b = self.buffer[self.pos];
            self.pos += 1;
            if b >= b'0' && b <= b'9' {
                n = n * 10 + (b - b'0') as u32;
                found = true;
            } else if found {
                return Some(n);
            }
        }
    }
}

fn add_to_bucket(v: usize, d: usize, heads: &mut [u32], next: &mut [u32], prev: &mut [u32]) {
    let v_u32 = v as u32;
    let head = heads[d];
    next[v] = head;
    prev[v] = u32::MAX;
    if head != u32::MAX {
        prev[head as usize] = v_u32;
    }
    heads[d] = v_u32;
}

fn remove_from_bucket(v: usize, d: usize, heads: &mut [u32], next: &mut [u32], prev: &mut [u32]) {
    let p = prev[v];
    let nxt = next[v];
    if p != u32::MAX {
        next[p as usize] = nxt;
    } else {
        heads[d] = nxt;
    }
    if nxt != u32::MAX {
        prev[nxt as usize] = p;
    }
    next[v] = u32::MAX;
    prev[v] = u32::MAX;
}

fn main() -> io::Result<()> {
    let mut fr = FastReader::new(io::stdin());
    let n = match fr.next_u32() {
        Some(v) => v as usize,
        None => return Ok(()),
    };
    let m = match fr.next_u32() {
        Some(v) => v as usize,
        None => 0,
    };

    if n == 0 {
        return Ok(());
    }

    let mut raw_edges = Vec::with_capacity(m);
    let mut degree = vec![0u32; n];
    for _ in 0..m {
        if let (Some(u_raw), Some(v_raw)) = (fr.next_u32(), fr.next_u32()) {
            let u = u_raw as usize;
            let v = v_raw as usize;
            if u < n && v < n {
                raw_edges.push((u_raw, v_raw));
                degree[u] += 1;
                degree[v] += 1;
            }
        }
    }

    let mut offsets = vec![0usize; n + 1];
    for i in 0..n {
        offsets[i + 1] = offsets[i] + degree[i] as usize;
    }
    let mut adj = vec![0u32; offsets[n]];
    let mut current_offset = offsets.clone();
    for &(u_raw, v_raw) in &raw_edges {
        let u = u_raw as usize;
        let v = v_raw as usize;
        adj[current_offset[u]] = v_raw;
        current_offset[u] += 1;
        adj[current_offset[v]] = u_raw;
        current_offset[v] += 1;
    }
    drop(raw_edges);
    drop(current_offset);

    let mut in_cover = vec![false; n];
    let mut is_removed = vec![false; n];
    let mut current_degree = degree;

    let mut heads = vec![u32::MAX; n + 1];
    let mut next = vec![u32::MAX; n];
    let mut prev = vec![u32::MAX; n];

    for i in 0..n {
        let d = current_degree[i] as usize;
        if d > 0 {
            add_to_bucket(i, d, &mut heads, &mut next, &mut prev);
        } else {
            is_removed[i] = true;
        }
    }

    let mut max_d = n;
    while max_d > 0 {
        if heads[max_d] == u32::MAX {
            max_d -= 1;
            continue;
        }

        let v = heads[max_d] as usize;
        remove_from_bucket(v, max_d, &mut heads, &mut next, &mut prev);
        
        in_cover[v] = true;
        is_removed[v] = true;
        current_degree[v] = 0;

        for i in offsets[v]..offsets[v+1] {
            let u = adj[i] as usize;
            if !is_removed[u] {
                let d = current_degree[u] as usize;
                remove_from_bucket(u, d, &mut heads, &mut next, &mut prev);
                current_degree[u] -= 1;
                let new_d = current_degree[u] as usize;
                if new_d > 0 {
                    add_to_bucket(u, new_d, &mut heads, &mut next, &mut prev);
                } else {
                    is_removed[u] = true;
                }
            }
        }
    }

    // Pruning: remove redundant vertices from cover
    for i in 0..n {
        if in_cover[i] {
            let mut redundant = true;
            for j in offsets[i]..offsets[i+1] {
                let u = adj[j] as usize;
                if !in_cover[u] {
                    redundant = false;
                    break;
                }
            }
            if redundant {
                in_cover[i] = false;
            }
        }
    }

    let mut count = 0;
    for &val in &in_cover {
        if val { count += 1; }
    }

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "{}", count)?;
    let mut first = true;
    for (i, &val) in in_cover.iter().enumerate() {
        if val {
            if !first {
                write!(out, " ")?;
            }
            write!(out, "{}", i)?;
            first = false;
        }
    }
    writeln!(out)?;

    Ok(())
}
