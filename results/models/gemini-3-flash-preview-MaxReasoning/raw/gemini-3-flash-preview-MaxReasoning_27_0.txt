#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#include <cstdint>
#include <cstdio>
#include <ctime>

#ifdef _MSC_VER
#include <intrin.h>
#define popcount64 __popcnt64
static inline int countr_zero64(uint64_t x) {
    unsigned long index;
    if (_BitScanForward64(&index, x)) return (int)index;
    return 64;
}
#else
#define popcount64 __builtin_popcountll
static inline int countr_zero64(uint64_t x) {
    if (x == 0) return 64;
    return __builtin_ctzll(x);
}
#endif

using namespace std;

// Fast I/O to handle massive graphs efficiently
struct FastReader {
    static const int BUF_SIZE = 8 * 1024 * 1024;
    char* buf;
    int pos = 0, len = 0;

    FastReader() {
        buf = new char[BUF_SIZE];
    }
    
    ~FastReader() {
        delete[] buf;
    }

    char readChar() {
        if (pos == len) {
            pos = 0;
            len = (int)fread(buf, 1, BUF_SIZE, stdin);
            if (len <= 0) return EOF;
        }
        return buf[pos++];
    }

    int readInt() {
        int x = 0;
        char ch = readChar();
        while (ch != EOF && (ch < '0' || ch > '9')) ch = readChar();
        if (ch == EOF) return -1;
        while (ch >= '0' && ch <= '9') {
            x = x * 10 + (ch - '0');
            ch = readChar();
        }
        return x;
    }
} reader;

// Node structure for DSATUR priority queue
struct Node {
    int id;
    int sat;
    int deg;
    bool operator<(const Node& other) const {
        // DSATUR priority: highest saturation degree first
        if (sat != other.sat) return sat < other.sat;
        // Tie-breaker: highest degree in uncolored subgraph (approximated by initial degree)
        return deg < other.deg;
    }
};

void solve() {
    int n = reader.readInt();
    int m = reader.readInt();
    int k = reader.readInt();
    if (n <= 0) return;

    // Use a temporary vector to store edges initially to build the adjacency list efficiently
    struct TempEdge {
        int u, v;
    };
    vector<TempEdge> temp_edges;
    if (m > 0) temp_edges.reserve(m);
    vector<int> deg(n, 0);
    for (int i = 0; i < m; ++i) {
        int u = reader.readInt();
        int v = reader.readInt();
        if (u >= 0 && u < n && v >= 0 && v < n && u != v) {
            temp_edges.push_back({u, v});
            deg[u]++;
            deg[v]++;
        }
    }

    // Build compact adjacency list
    vector<int> head(n + 1, 0);
    for (int i = 0; i < n; ++i) head[i + 1] = head[i] + deg[i];
    vector<int> adj(head[n]);
    vector<int> curr_head = head;
    for (const auto& e : temp_edges) {
        adj[curr_head[e.u]++] = e.v;
        adj[curr_head[e.v]++] = e.u;
    }
    temp_edges.clear();
    // Use swap trick for older C++ versions if shrink_to_fit is not enough
    vector<TempEdge>().swap(temp_edges);

    vector<int> color(n, -1);
    // used_mask tracks which of the first 64 colors are used by neighbors
    vector<uint64_t> used_mask(n, 0);
    vector<int> saturation(n, 0);
    priority_queue<Node> pq;

    // Initial population of the priority queue
    for (int i = 0; i < n; ++i) {
        pq.push({i, 0, deg[i]});
    }

    vector<int> last_seen(k, -1);
    int colored_count = 0;
    while (!pq.empty() && colored_count < n) {
        Node top = pq.top();
        pq.pop();

        int u = top.id;
        // Skip if already colored or if this is an outdated entry in the PQ
        if (color[u] != -1 || top.sat < saturation[u]) continue;

        int c = -1;
        if (k <= 64) {
            // Fast path for k <= 64 using bitmasks
            uint64_t mask = used_mask[u];
            uint64_t inverted = ~mask;
            c = countr_zero64(inverted);
        } else {
            // Slower path for k > 64
            for (int i = head[u]; i < head[u+1]; ++i) {
                int v = adj[i];
                if (color[v] != -1 && color[v] < k) last_seen[color[v]] = u;
            }
            c = 0;
            while (c < k && last_seen[c] == u) c++;
        }

        if (c >= k) {
            // If greedy DSATUR fails to find a color within [0, k-1], 
            // pick the color that minimizes conflicts among neighbors.
            vector<int> conflicts(k, 0);
            for (int i = head[u]; i < head[u+1]; ++i) {
                int v = adj[i];
                if (color[v] != -1 && color[v] < k) conflicts[color[v]]++;
            }
            int best_c = 0;
            for (int i = 1; i < k; ++i) {
                if (conflicts[i] < conflicts[best_c]) best_c = i;
            }
            c = best_c;
        }

        color[u] = c;
        colored_count++;

        // Update neighbors' saturation and masks
        for (int i = head[u]; i < head[u + 1]; ++i) {
            int v = adj[i];
            if (color[v] == -1) {
                uint64_t old_mask = used_mask[v];
                if (c < 64) used_mask[v] |= (1ULL << c);
                if (used_mask[v] != old_mask) {
                    saturation[v] = (int)popcount64(used_mask[v]);
                    pq.push({v, saturation[v], deg[v]});
                }
            }
        }
    }

    // Local search phase (Min-Conflicts) to resolve any conflicts introduced
    auto start_time = clock();
    vector<int> conflicted_nodes;
    vector<bool> is_conflicted(n, false);
    for (int u = 0; u < n; ++u) {
        for (int i = head[u]; i < head[u + 1]; ++i) {
            if (color[u] == color[adj[i]]) {
                if (!is_conflicted[u]) {
                    conflicted_nodes.push_back(u);
                    is_conflicted[u] = true;
                }
                break;
            }
        }
    }

    if (!conflicted_nodes.empty()) {
        // Cap iterations to stay within time limit
        vector<int> counts(k, 0);
        for (int iter = 0; iter < 100 && !conflicted_nodes.empty(); ++iter) {
            double elapsed = (double)(clock() - start_time) / CLOCKS_PER_SEC;
            if (elapsed > 270.0) break; // Leave time for output

            bool changed = false;
            for (int u : conflicted_nodes) {
                // Find best color in [0, k-1] for this node
                fill(counts.begin(), counts.end(), 0);
                for (int i = head[u]; i < head[u+1]; ++i) {
                    int cv = color[adj[i]];
                    if (cv >= 0 && cv < k) counts[cv]++;
                }
                int best_c = 0;
                for (int i = 1; i < k; ++i) {
                    if (counts[i] < counts[best_c]) best_c = i;
                }
                if (counts[best_c] < counts[color[u]]) {
                    color[u] = best_c;
                    changed = true;
                }
            }
            if (!changed && iter > 5) break;

            // Update list of conflicted nodes
            vector<int> next_conflicted;
            is_conflicted.assign(n, false);
            for (int u = 0; u < n; ++u) {
                for (int i = head[u]; i < head[u+1]; ++i) {
                    if (color[u] == color[adj[i]]) {
                        if (!is_conflicted[u]) {
                            next_conflicted.push_back(u);
                            is_conflicted[u] = true;
                        }
                        break;
                    }
                }
            }
            conflicted_nodes = next_conflicted;
        }
    }

    // Output final color assignment
    for (int i = 0; i < n; ++i) {
        printf("%d%c", color[i], (i == n - 1 ? '\n' : ' '));
    }
}

int main() {
    solve();
    return 0;
}
