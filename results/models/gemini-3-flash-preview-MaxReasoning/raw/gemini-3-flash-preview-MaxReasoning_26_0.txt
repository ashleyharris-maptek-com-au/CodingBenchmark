use std::io::{self, Read, Write, BufReader, BufWriter, BufRead};

// Simple PRNG for K-means++
struct Rng(u64);
impl Rng {
    fn new(seed: u64) -> Self { Rng(seed) }
    fn next_u32(&mut self) -> u32 {
        self.0 = self.0.wrapping_mul(6364136223846793005).wrapping_add(1);
        (self.0 >> 32) as u32
    }
    fn next_f32(&mut self) -> f32 {
        (self.next_u32() & 0xFFFFFF) as f32 / 16777216.0
    }
}

struct Centroids {
    x: Vec<f32>,
    y: Vec<f32>,
    z: Vec<f32>,
    c2: Vec<f32>,
    counts: Vec<u64>,
}

impl Centroids {
    fn new(k: usize) -> Self {
        Centroids {
            x: vec![0.0; k],
            y: vec![0.0; k],
            z: vec![0.0; k],
            c2: vec![0.0; k],
            counts: vec![0; k],
        }
    }

    #[inline(always)]
    fn find_nearest(&self, px: f32, py: f32, pz: f32) -> usize {
        let mut min_val = f32::MAX;
        let mut best_idx = 0;
        let n = self.x.len();
        
        let cx = &self.x;
        let cy = &self.y;
        let cz = &self.z;
        let c2 = &self.c2;

        for i in 0..n {
            // Minimize d^2 - p^2 = c^2 - 2(p.c)
            let val = c2[i] - 2.0 * (px * cx[i] + py * cy[i] + pz * cz[i]);
            if val < min_val {
                min_val = val;
                best_idx = i;
            }
        }
        best_idx
    }

    fn update(&mut self, idx: usize, px: f32, py: f32, pz: f32) {
        self.counts[idx] += 1;
        let count = self.counts[idx] as f64;
        self.x[idx] = ((self.x[idx] as f64 * (count - 1.0) + px as f64) / count) as f32;
        self.y[idx] = ((self.y[idx] as f64 * (count - 1.0) + py as f64) / count) as f32;
        self.z[idx] = ((self.z[idx] as f64 * (count - 1.0) + pz as f64) / count) as f32;
        self.c2[idx] = self.x[idx] * self.x[idx] + self.y[idx] * self.y[idx] + self.z[idx] * self.z[idx];
    }
}

struct CoarseIndex {
    coarse_x: Vec<f32>,
    coarse_y: Vec<f32>,
    coarse_z: Vec<f32>,
    coarse_c2: Vec<f32>,
    groups: Vec<Vec<usize>>,
}

impl CoarseIndex {
    fn new(centroids: &Centroids) -> Self {
        let k = centroids.x.len();
        let num_coarse = (k as f32).sqrt().ceil() as usize;
        let mut rng = Rng::new(99);
        
        let mut coarse_x = Vec::with_capacity(num_coarse);
        let mut coarse_y = Vec::with_capacity(num_coarse);
        let mut coarse_z = Vec::with_capacity(num_coarse);
        let mut coarse_c2 = Vec::with_capacity(num_coarse);
        
        for _ in 0..num_coarse {
            let idx = (rng.next_u32() as usize) % k;
            coarse_x.push(centroids.x[idx]);
            coarse_y.push(centroids.y[idx]);
            coarse_z.push(centroids.z[idx]);
            coarse_c2.push(centroids.c2[idx]);
        }
        
        let mut groups = vec![Vec::new(); num_coarse];
        for i in 0..k {
            let mut min_val = f32::MAX;
            let mut best_coarse = 0;
            for c in 0..num_coarse {
                let val = coarse_c2[c] - 2.0 * (centroids.x[i] * coarse_x[c] + centroids.y[i] * coarse_y[c] + centroids.z[i] * coarse_z[c]);
                if val < min_val {
                    min_val = val;
                    best_coarse = c;
                }
            }
            groups[best_coarse].push(i);
        }
        
        CoarseIndex { coarse_x, coarse_y, coarse_z, coarse_c2, groups }
    }
    
    #[inline(always)]
    fn find_nearest(&self, px: f32, py: f32, pz: f32, centroids: &Centroids) -> usize {
        let mut min_val_coarse = f32::MAX;
        let mut best_coarse = 0;
        for c in 0..self.coarse_x.len() {
            let val = self.coarse_c2[c] - 2.0 * (px * self.coarse_x[c] + py * self.coarse_y[c] + pz * self.coarse_z[c]);
            if val < min_val_coarse {
                min_val_coarse = val;
                best_coarse = c;
            }
        }
        
        let mut min_val = f32::MAX;
        let mut best_idx = 0;
        let group = &self.groups[best_coarse];
        if group.is_empty() {
             // Fallback to linear search if group is empty (should not happen)
             return centroids.find_nearest(px, py, pz);
        }
        best_idx = group[0];
        for &i in group {
            let val = centroids.c2[i] - 2.0 * (px * centroids.x[i] + py * centroids.y[i] + pz * centroids.z[i]);
            if val < min_val {
                min_val = val;
                best_idx = i;
            }
        }
        best_idx
    }
}

fn k_means_plus_plus(points: &[f32], k: usize, rng: &mut Rng) -> Centroids {
    let n = points.len() / 3;
    let mut centroids = Centroids::new(k);
    if n == 0 { return centroids; }

    let actual_k = if k > n { n } else { k };
    
    let idx = (rng.next_u32() as usize) % n;
    centroids.x[0] = points[idx * 3];
    centroids.y[0] = points[idx * 3 + 1];
    centroids.z[0] = points[idx * 3 + 2];
    centroids.c2[0] = centroids.x[0]*centroids.x[0] + centroids.y[0]*centroids.y[0] + centroids.z[0]*centroids.z[0];
    
    if actual_k == 1 { 
        for i in 1..k {
            centroids.x[i] = centroids.x[0];
            centroids.y[i] = centroids.y[0];
            centroids.z[i] = centroids.z[0];
            centroids.c2[i] = centroids.c2[0];
        }
        return centroids; 
    }

    let mut min_dists_sq = vec![f32::MAX; n];
    let complexity = actual_k as u64 * n as u64;
    
    if complexity > 500_000_000 {
        // Faster initialization for large sets
        for i in 1..actual_k {
            let idx = (rng.next_u32() as usize) % n;
            centroids.x[i] = points[idx * 3];
            centroids.y[i] = points[idx * 3 + 1];
            centroids.z[i] = points[idx * 3 + 2];
            centroids.c2[i] = centroids.x[i]*centroids.x[i] + centroids.y[i]*centroids.y[i] + centroids.z[i]*centroids.z[i];
        }
    } else {
        for i in 1..actual_k {
            let last_x = centroids.x[i-1];
            let last_y = centroids.y[i-1];
            let last_z = centroids.z[i-1];
            let mut sum = 0.0;
            for j in 0..n {
                let dx = points[j*3] - last_x;
                let dy = points[j*3+1] - last_y;
                let dz = points[j*3+2] - last_z;
                let d2 = dx*dx + dy*dy + dz*dz;
                if d2 < min_dists_sq[j] { min_dists_sq[j] = d2; }
                sum += min_dists_sq[j];
            }
            let target = rng.next_f32() * sum;
            let mut current_sum = 0.0;
            let mut next_idx = n - 1;
            for j in 0..n {
                current_sum += min_dists_sq[j];
                if current_sum >= target { next_idx = j; break; }
            }
            centroids.x[i] = points[next_idx * 3];
            centroids.y[i] = points[next_idx * 3 + 1];
            centroids.z[i] = points[next_idx * 3 + 2];
            centroids.c2[i] = centroids.x[i]*centroids.x[i] + centroids.y[i]*centroids.y[i] + centroids.z[i]*centroids.z[i];
        }
    }
    
    for i in actual_k..k {
        centroids.x[i] = centroids.x[i % actual_k];
        centroids.y[i] = centroids.y[i % actual_k];
        centroids.z[i] = centroids.z[i % actual_k];
        centroids.c2[i] = centroids.c2[i % actual_k];
    }
    centroids
}

fn run_in_memory<R: Read, W: Write>(
    num_points: u64,
    num_clusters: u32,
    mut input: R,
    mut output: W,
) -> io::Result<()> {
    let n = num_points as usize;
    let k = num_clusters as usize;
    let mut points = vec![0.0f32; n * 3];
    let mut buf = [0u8; 12];
    for i in 0..n {
        input.read_exact(&mut buf)?;
        points[i * 3] = f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]);
        points[i * 3 + 1] = f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]);
        points[i * 3 + 2] = f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]);
    }
    
    let mut rng = Rng::new(42);
    let sample_size = if n > 1_000_000 { 1_000_000 } else { n };
    let mut centroids = k_means_plus_plus(&points[0..sample_size*3], k, &mut rng);
    
    let complexity = n as u64 * k as u64;
    let iters = if complexity > 1_000_000_000 { 2 } else if complexity > 100_000_000 { 5 } else { 15 };
    
    for _ in 0..iters {
        let mut new_x = vec![0.0f64; k];
        let mut new_y = vec![0.0f64; k];
        let mut new_z = vec![0.0f64; k];
        let mut new_counts = vec![0u64; k];
        for i in 0..n {
            let px = points[i*3];
            let py = points[i*3+1];
            let pz = points[i*3+2];
            let best_idx = centroids.find_nearest(px, py, pz);
            new_x[best_idx] += px as f64;
            new_y[best_idx] += py as f64;
            new_z[best_idx] += pz as f64;
            new_counts[best_idx] += 1;
        }
        let mut moved = false;
        for i in 0..k {
            if new_counts[i] > 0 {
                let nx = (new_x[i] / new_counts[i] as f64) as f32;
                let ny = (new_y[i] / new_counts[i] as f64) as f32;
                let nz = (new_z[i] / new_counts[i] as f64) as f32;
                if (nx - centroids.x[i]).abs() > 1e-6 { moved = true; }
                centroids.x[i] = nx;
                centroids.y[i] = ny;
                centroids.z[i] = nz;
                centroids.c2[i] = nx*nx + ny*ny + nz*nz;
            }
        }
        if !moved { break; }
    }
    
    for i in 0..n {
        let best_idx = centroids.find_nearest(points[i*3], points[i*3+1], points[i*3+2]);
        output.write_all(&(best_idx as u32).to_le_bytes())?;
    }
    Ok(())
}

fn run_streaming<R: Read, W: Write>(
    num_points: u64,
    num_clusters: u32,
    mut input: R,
    mut output: W,
) -> io::Result<()> {
    let k = num_clusters as usize;
    let mut rng = Rng::new(42);
    let warm_up_n = if num_points > 1_000_000 { 1_000_000 } else { num_points as usize };
    let mut warm_up_points = vec![0.0f32; warm_up_n * 3];
    let mut buf = [0u8; 12];
    for i in 0..warm_up_n {
        input.read_exact(&mut buf)?;
        warm_up_points[i * 3] = f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]);
        warm_up_points[i * 3 + 1] = f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]);
        warm_up_points[i * 3 + 2] = f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]);
    }
    
    let mut centroids = k_means_plus_plus(&warm_up_points, k, &mut rng);
    let use_coarse = (num_points as f64 * k as f64) > 2e11;
    let coarse_index = if use_coarse { Some(CoarseIndex::new(&centroids)) } else { None };

    for i in 0..warm_up_n {
        let px = warm_up_points[i*3];
        let py = warm_up_points[i*3+1];
        let pz = warm_up_points[i*3+2];
        let idx = if let Some(ref idx) = coarse_index {
            idx.find_nearest(px, py, pz, &centroids)
        } else {
            centroids.find_nearest(px, py, pz)
        };
        output.write_all(&(idx as u32).to_le_bytes())?;
        centroids.update(idx, px, py, pz);
    }
    
    let mut remaining = num_points - warm_up_n as u64;
    let mut batch_buf = vec![0u8; 12 * 1024];
    while remaining > 0 {
        let n_to_read = if remaining > 1024 { 1024 } else { remaining as usize };
        input.read_exact(&mut batch_buf[..n_to_read * 12])?;
        for i in 0..n_to_read {
            let offset = i * 12;
            let px = f32::from_le_bytes([batch_buf[offset], batch_buf[offset+1], batch_buf[offset+2], batch_buf[offset+3]]);
            let py = f32::from_le_bytes([batch_buf[offset+4], batch_buf[offset+5], batch_buf[offset+6], batch_buf[offset+7]]);
            let pz = f32::from_le_bytes([batch_buf[offset+8], batch_buf[offset+9], batch_buf[offset+10], batch_buf[offset+11]]);
            let idx = if let Some(ref idx) = coarse_index {
                idx.find_nearest(px, py, pz, &centroids)
            } else {
                centroids.find_nearest(px, py, pz)
            };
            output.write_all(&(idx as u32).to_le_bytes())?;
            centroids.update(idx, px, py, pz);
        }
        remaining -= n_to_read as u64;
    }
    Ok(())
}

fn main() -> io::Result<()> {
    let mut stdin = BufReader::with_capacity(1 << 20, io::stdin());
    let mut stdout = BufWriter::with_capacity(1 << 20, io::stdout());
    let mut line = String::new();
    if stdin.read_line(&mut line)? == 0 { return Ok(()); }
    let parts: Vec<&str> = line.split_whitespace().collect();
    if parts.len() < 2 { return Ok(()); }
    let num_points: u64 = parts[0].parse().unwrap();
    let num_clusters: u32 = parts[1].parse().unwrap();
    if num_points == 0 || num_clusters == 0 { return Ok(()); }
    
    // Memory threshold: 50M points is safe for 32-bit systems
    if num_points <= 50_000_000 {
        run_in_memory(num_points, num_clusters, &mut stdin, &mut stdout)?;
    } else {
        run_streaming(num_points, num_clusters, &mut stdin, &mut stdout)?;
    }
    Ok(())
}
