use std::io::{self, Read, BufRead, Write, BufWriter};

struct FastReader<R: BufRead> {
    reader: R,
    buffer: [u8; 1 << 20],
    pos: usize,
    cap: usize,
}

impl<R: BufRead> FastReader<R> {
    fn new(reader: R) -> Self {
        Self {
            reader,
            buffer: [0; 1 << 20],
            pos: 0,
            cap: 0,
        }
    }

    fn next_u64(&mut self) -> Option<u64> {
        let mut res = 0;
        let mut started = false;
        loop {
            if self.pos >= self.cap {
                match self.reader.read(&mut self.buffer) {
                    Ok(n) if n > 0 => {
                        self.cap = n;
                        self.pos = 0;
                    }
                    _ => return if started { Some(res) } else { None },
                }
            }
            let byte = self.buffer[self.pos];
            self.pos += 1;
            if byte >= b'0' && byte <= b'9' {
                res = res * 10 + (byte - b'0') as u64;
                started = true;
            } else if started {
                return Some(res);
            }
        }
    }
}

struct SimpleRng {
    state: u64,
}

impl SimpleRng {
    fn new(seed: u64) -> Self {
        Self { state: if seed == 0 { 1 } else { seed } }
    }
    fn next_u64(&mut self) -> u64 {
        self.state ^= self.state << 13;
        self.state ^= self.state >> 7;
        self.state ^= self.state << 17;
        self.state
    }
}

struct GroupedItem {
    value: u64,
    indices: Vec<u32>,
}

fn print_success(indices: &[u32]) {
    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());
    let _ = writeln!(writer, "YES");
    let _ = write!(writer, "[");
    for (i, &idx) in indices.iter().enumerate() {
        if i > 0 {
            let _ = write!(writer, " ");
        }
        let _ = write!(writer, "{}", idx);
    }
    let _ = writeln!(writer, "]");
    let _ = writer.flush();
}

fn solve_bitset(target: u64, nums: &[(u64, u32)]) -> Option<Vec<u32>> {
    let target_usize = target as usize;
    let mut sorted = nums.to_vec();
    sorted.sort_unstable_by_key(|&(v, _)| v);
    
    let mut grouped = Vec::new();
    let mut i = 0;
    while i < sorted.len() {
        let val = sorted[i].0;
        let mut group_indices = Vec::new();
        while i < sorted.len() && sorted[i].0 == val {
            group_indices.push(sorted[i].1);
            i += 1;
        }
        
        let mut count = group_indices.len();
        let mut p = 1;
        let mut start = 0;
        while count >= p {
            let mut items = Vec::with_capacity(p);
            for k in 0..p {
                items.push(group_indices[start + k]);
            }
            grouped.push(GroupedItem { value: val * p as u64, indices: items });
            start += p;
            count -= p;
            p *= 2;
        }
        if count > 0 {
            let mut items = Vec::with_capacity(count);
            for k in 0..count {
                items.push(group_indices[start + k]);
            }
            grouped.push(GroupedItem { value: val * count as u64, indices: items });
        }
    }

    let num_chunks = (target_usize / 64) + 1;
    let mut bits = vec![0u64; num_chunks];
    bits[0] = 1;
    let mut last_idx = vec![u32::MAX; target_usize + 1];

    for (idx, item) in grouped.iter().enumerate() {
        let val = item.value as usize;
        if val > target_usize { continue; }
        
        let val_chunks = val / 64;
        let val_offset = val % 64;
        
        if val_offset == 0 {
            for j in (val_chunks..num_chunks).rev() {
                let shifted = bits[j - val_chunks];
                let new_bits = shifted & !bits[j];
                if new_bits != 0 {
                    let mut temp = new_bits;
                    while temp != 0 {
                        let bit = temp.trailing_zeros();
                        let s = j * 64 + bit as usize;
                        if s <= target_usize && last_idx[s] == u32::MAX {
                            last_idx[s] = idx as u32;
                        }
                        temp &= temp.wrapping_sub(1);
                    }
                    bits[j] |= shifted;
                }
            }
        } else {
            let inv_offset = 64 - val_offset;
            for j in (val_chunks..num_chunks).rev() {
                let mut shifted = bits[j - val_chunks] << val_offset;
                if j > val_chunks {
                    shifted |= bits[j - val_chunks - 1] >> inv_offset;
                }
                let new_bits = shifted & !bits[j];
                if new_bits != 0 {
                    let mut temp = new_bits;
                    while temp != 0 {
                        let bit = temp.trailing_zeros();
                        let s = j * 64 + bit as usize;
                        if s <= target_usize && last_idx[s] == u32::MAX {
                            last_idx[s] = idx as u32;
                        }
                        temp &= temp.wrapping_sub(1);
                    }
                    bits[j] |= shifted;
                }
            }
        }
        
        if (bits[target_usize / 64] & (1 << (target_usize % 64))) != 0 {
            let mut res = Vec::new();
            let mut curr = target_usize;
            while curr > 0 {
                let g_idx = last_idx[curr] as usize;
                for &orig_idx in &grouped[g_idx].indices {
                    res.push(orig_idx);
                }
                curr -= grouped[g_idx].value as usize;
            }
            return Some(res);
        }
    }
    None
}

fn solve_meet_in_middle(target: u64, nums: &[(u64, u32)]) -> Option<Vec<u32>> {
    let n = nums.len();
    if n == 0 { return None; }
    let n1 = n / 2;
    let n2 = n - n1;
    
    let mut left = Vec::with_capacity(1 << n1);
    for i in 0..(1 << n1) {
        let mut sum = 0;
        for j in 0..n1 {
            if (i >> j) & 1 == 1 {
                sum += nums[j].0;
            }
        }
        if sum <= target {
            left.push((sum, i as u64));
        }
    }
    left.sort_unstable_by_key(|&(s, _)| s);
    
    for i in 0..(1 << n2) {
        let mut sum = 0;
        for j in 0..n2 {
            if (i >> j) & 1 == 1 {
                sum += nums[n1 + j].0;
            }
        }
        if sum <= target {
            let needed = target - sum;
            if let Ok(idx) = left.binary_search_by_key(&needed, |&(s, _)| s) {
                let mut res = Vec::new();
                let left_bits = left[idx].1;
                for j in 0..n1 {
                    if (left_bits >> j) & 1 == 1 {
                        res.push(nums[j].1);
                    }
                }
                for j in 0..n2 {
                    if (i >> j) & 1 == 1 {
                        res.push(nums[n1 + j].1);
                    }
                }
                return Some(res);
            }
        }
    }
    None
}

fn solve_local_search(target: u64, nums: &[(u64, u32)]) -> Option<Vec<u32>> {
    let n = nums.len();
    let mut rng = SimpleRng::new(42);
    let mut in_subset = vec![false; n];
    let mut current_sum: u128 = 0;
    
    let mut sorted_indices: Vec<usize> = (0..n).collect();
    sorted_indices.sort_unstable_by(|&a, &b| nums[b].0.cmp(&nums[a].0));
    
    for &i in &sorted_indices {
        if current_sum + nums[i].0 as u128 <= target as u128 {
            current_sum += nums[i].0 as u128;
            in_subset[i] = true;
        }
    }
    
    if current_sum == target as u128 {
        let mut res = Vec::new();
        for i in 0..n {
            if in_subset[i] { res.push(nums[i].1); }
        }
        return Some(res);
    }
    
    for _ in 0..2_000_000 {
        let i = (rng.next_u64() as usize) % n;
        if in_subset[i] {
            let j = (rng.next_u64() as usize) % n;
            if !in_subset[j] {
                let next_sum = current_sum - nums[i].0 as u128 + nums[j].0 as u128;
                if next_sum == target as u128 {
                    in_subset[i] = false;
                    in_subset[j] = true;
                    let mut res = Vec::new();
                    for k in 0..n { if in_subset[k] { res.push(nums[k].1); } }
                    return Some(res);
                }
                let diff_curr = (target as i128 - current_sum as i128).abs();
                let diff_next = (target as i128 - next_sum as i128).abs();
                if diff_next < diff_curr || (rng.next_u64() % 100 == 0) {
                    in_subset[i] = false;
                    in_subset[j] = true;
                    current_sum = next_sum;
                }
            }
        } else {
            let next_sum = current_sum + nums[i].0 as u128;
            if next_sum == target as u128 {
                in_subset[i] = true;
                let mut res = Vec::new();
                for k in 0..n { if in_subset[k] { res.push(nums[k].1); } }
                return Some(res);
            }
            if next_sum < target as u128 {
                in_subset[i] = true;
                current_sum = next_sum;
            }
        }
    }
    None
}

fn main() {
    let stdin = io::stdin();
    let mut fr = FastReader::new(stdin.lock());
    
    let n = match fr.next_u64() {
        Some(val) => val as usize,
        None => return,
    };
    let target = match fr.next_u64() {
        Some(val) => val,
        None => return,
    };
    
    if target == 0 {
        println!("YES");
        println!("[]");
        return;
    }

    let mut nums = Vec::with_capacity(n);
    let mut total_sum: u128 = 0;
    for i in 0..n {
        if let Some(val) = fr.next_u64() {
            if val <= target {
                nums.push((val, i as u32));
                total_sum += val as u128;
            }
        }
    }

    if total_sum < target as u128 {
        println!("NO");
        return;
    }
    
    for (val, idx) in &nums {
        if *val == target {
            print_success(&[*idx]);
            return;
        }
    }
    
    nums.sort_unstable_by(|a, b| b.0.cmp(&a.0));
    
    let mut current_sum: u128 = 0;
    let mut chosen = Vec::new();
    for (val, idx) in &nums {
        if current_sum + *val as u128 <= target as u128 {
            current_sum += *val as u128;
            chosen.push(*idx);
            if current_sum == target as u128 {
                print_success(&chosen);
                return;
            }
        }
    }

    let seed = n as u64 ^ target;
    let mut rng = SimpleRng::new(seed);
    for _ in 0..50 {
        let len = nums.len();
        for i in (1..len).rev() {
            let j = (rng.next_u64() as usize) % (i + 1);
            nums.swap(i, j);
        }
        let mut c_sum: u128 = 0;
        let mut c_chosen = Vec::new();
        for (val, idx) in &nums {
            if c_sum + *val as u128 <= target as u128 {
                c_sum += *val as u128;
                c_chosen.push(*idx);
                if c_sum == target as u128 {
                    print_success(&c_chosen);
                    return;
                }
            }
        }
    }

    if target <= 5_000_000_000 {
        if let Some(indices) = solve_bitset(target, &nums) {
            print_success(&indices);
            return;
        }
    } else if nums.len() <= 45 {
        if let Some(indices) = solve_meet_in_middle(target, &nums) {
            print_success(&indices);
            return;
        }
    } else {
        if let Some(indices) = solve_local_search(target, &nums) {
            print_success(&indices);
            return;
        }
    }

    println!("NO");
}
