#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <cstdio>
#include <ctime>

using namespace std;

struct Point3D {
    double x, y, z;
};

Point3D operator-(Point3D a, Point3D b) { return {a.x - b.x, a.y - b.y, a.z - b.z}; }
Point3D operator+(Point3D a, Point3D b) { return {a.x + b.x, a.y + b.y, a.z + b.z}; }
Point3D operator*(Point3D a, double b) { return {a.x * b, a.y * b, a.z * b}; }
double dot(Point3D a, Point3D b) { return a.x * b.x + a.y * b.y + a.z * b.z; }
Point3D cross(Point3D a, Point3D b) {
    return {a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}

struct Face {
    Point3D normal;
    double d;
};

int main() {
    int V_in;
    double e;
    if (scanf("%d %lf", &V_in, &e) != 2) return 0;

    vector<Point3D> all_pts(V_in);
    double min_x = 1e18, max_x = -1e18;
    double min_y = 1e18, max_y = -1e18;
    double min_z = 1e18, max_z = -1e18;
    int idxs[6] = {0,0,0,0,0,0};

    for (int i = 0; i < V_in; ++i) {
        if (scanf("%lf %lf %lf", &all_pts[i].x, &all_pts[i].y, &all_pts[i].z) != 3) break;
        if (all_pts[i].x < min_x) { min_x = all_pts[i].x; idxs[0] = i; }
        if (all_pts[i].x > max_x) { max_x = all_pts[i].x; idxs[1] = i; }
        if (all_pts[i].y < min_y) { min_y = all_pts[i].y; idxs[2] = i; }
        if (all_pts[i].y > max_y) { max_y = all_pts[i].y; idxs[3] = i; }
        if (all_pts[i].z < min_z) { min_z = all_pts[i].z; idxs[4] = i; }
        if (all_pts[i].z > max_z) { max_z = all_pts[i].z; idxs[5] = i; }
    }

    if (V_in < 4) {
        printf("0\n");
        return 0;
    }

    // Select subset of points for hull to ensure speed (S^4 complexity)
    vector<Point3D> hull_pts;
    vector<bool> used(V_in, false);
    for (int i = 0; i < 6; ++i) {
        if (!used[idxs[i]]) {
            hull_pts.push_back(all_pts[idxs[i]]);
            used[idxs[i]] = true;
        }
    }
    
    // Add random points to improve hull approximation
    srand(42);
    int target_s = min(V_in, 100);
    while (hull_pts.size() < (size_t)target_s) {
        int r = rand() % V_in;
        if (!used[r]) {
            hull_pts.push_back(all_pts[r]);
            used[r] = true;
        }
    }

    // Compute Hull Faces O(S^4) - Simple but robust for small S
    vector<Face> faces;
    int S = hull_pts.size();
    Point3D center_pt = {0,0,0};
    for (const auto& p : hull_pts) { center_pt.x += p.x; center_pt.y += p.y; center_pt.z += p.z; }
    center_pt.x /= S; center_pt.y /= S; center_pt.z /= S;

    for (int i = 0; i < S; ++i) {
        for (int j = i + 1; j < S; ++j) {
            for (int k = j + 1; k < S; ++k) {
                Point3D n = cross(hull_pts[j] - hull_pts[i], hull_pts[k] - hull_pts[i]);
                double nl2 = dot(n, n);
                if (nl2 < 1e-12) continue;
                n = n * (1.0 / sqrt(nl2));
                
                bool all_one_side = true;
                bool all_other_side = true;
                for (int l = 0; l < S; ++l) {
                    if (l == i || l == j || l == k) continue;
                    double dist = dot(n, hull_pts[l] - hull_pts[i]);
                    if (dist > 1e-7) all_one_side = false;
                    else if (dist < -1e-7) all_other_side = false;
                    if (!all_one_side && !all_other_side) break;
                }
                
                if (all_one_side || all_other_side) {
                    // Orientation: normal should point OUT. center_pt is INSIDE.
                    if (dot(n, center_pt - hull_pts[i]) > 0) n = n * -1.0;
                    
                    double d_val = -dot(n, hull_pts[i]);
                    bool exists = false;
                    for (const auto& f : faces) {
                        if (dot(f.normal, n) > 0.99999 && abs(f.d - d_val) < 1e-7) {
                            exists = true; break;
                        }
                    }
                    if (!exists) faces.push_back({n, d_val});
                }
            }
        }
    }

    double a = e / sqrt(2.0);
    double offsets[4][3] = {{0,0,0}, {a,a,0}, {a,0,a}, {0,a,a}};

    // Pass 1: Count
    long long K = 0;
    for (double x = min_x; x + a <= max_x + 1e-7; x += a) {
        for (double y = min_y; y + a <= max_y + 1e-7; y += a) {
            double z_low = min_z, z_high = max_z;
            bool possible = true;
            for (const auto& f : faces) {
                for (int v_idx = 0; v_idx < 4; ++v_idx) {
                    double dx = offsets[v_idx][0], dy = offsets[v_idx][1], dz = offsets[v_idx][2];
                    double lhs_const = f.normal.x * (x + dx) + f.normal.y * (y + dy) + f.d;
                    double nz_coeff = f.normal.z;
                    double rhs = -lhs_const - nz_coeff * dz;
                    if (nz_coeff > 1e-9) {
                        z_high = min(z_high, rhs / nz_coeff);
                    } else if (nz_coeff < -1e-9) {
                        z_low = max(z_low, rhs / nz_coeff);
                    } else {
                        if (lhs_const > 1e-8) { possible = false; break; }
                    }
                }
                if (!possible) break;
            }
            if (possible && z_low <= z_high + 1e-9) {
                double start_z = min_z + ceil(max(0.0, z_low - min_z - 1e-9) / a) * a;
                for (double z = start_z; z + a <= z_high + 1e-7; z += a) {
                    K++;
                    if (K > 5000000) break; // Safety limit
                }
            }
            if (K > 5000000) break;
        }
        if (K > 5000000) break;
    }

    // Pass 2: Output
    printf("%lld\n", K);
    long long count = 0;
    for (double x = min_x; x + a <= max_x + 1e-7; x += a) {
        for (double y = min_y; y + a <= max_y + 1e-7; y += a) {
            double z_low = min_z, z_high = max_z;
            bool possible = true;
            for (const auto& f : faces) {
                for (int v_idx = 0; v_idx < 4; ++v_idx) {
                    double dx = offsets[v_idx][0], dy = offsets[v_idx][1], dz = offsets[v_idx][2];
                    double lhs_const = f.normal.x * (x + dx) + f.normal.y * (y + dy) + f.d;
                    double nz_coeff = f.normal.z;
                    double rhs = -lhs_const - nz_coeff * dz;
                    if (nz_coeff > 1e-9) {
                        z_high = min(z_high, rhs / nz_coeff);
                    } else if (nz_coeff < -1e-9) {
                        z_low = max(z_low, rhs / nz_coeff);
                    } else {
                        if (lhs_const > 1e-8) { possible = false; break; }
                    }
                }
                if (!possible) break;
            }
            if (possible && z_low <= z_high + 1e-9) {
                double start_z = min_z + ceil(max(0.0, z_low - min_z - 1e-9) / a) * a;
                for (double z = start_z; z + a <= z_high + 1e-7; z += a) {
                    if (count < K) {
                        printf("%.6f %.6f %.6f 0.000000 0.000000 0.000000\n", x + a/2.0, y + a/2.0, z + a/2.0);
                        count++;
                    } else break;
                }
            }
            if (count >= K) break;
        }
        if (count >= K) break;
    }

    return 0;
}
