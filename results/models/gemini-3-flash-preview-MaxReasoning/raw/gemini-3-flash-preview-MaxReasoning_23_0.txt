use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel_remaining: f64,
}

fn is_obstacle(bitset: &[u64], width: usize, height: usize, x: usize, y: usize) -> bool {
    if x >= width || y >= height {
        return false;
    }
    let idx = y * width + x;
    (bitset[idx / 64] & (1 << (idx % 64))) != 0
}

fn find_ground(bitset: &[u64], width: usize, height: usize, x: f64, y: f64) -> f64 {
    let ix = x.round() as i32;
    if ix < 0 || ix >= width as i32 {
        return height as f64;
    }
    let ux = ix as usize;
    let uy_start = (y.round() as i32).max(0) as usize;
    for uy in uy_start..height {
        if is_obstacle(bitset, width, height, ux, uy) {
            return uy as f64;
        }
    }
    height as f64
}

fn main() {
    let stdin = io::stdin();
    // Use a large buffer to read potentially massive maps efficiently.
    let mut reader = io::BufReader::with_capacity(1024 * 1024, stdin.lock());

    let mut line = String::new();

    // 1st line: width height gravity max_thrust fuel
    if reader.read_line(&mut line).is_err() || line.trim().is_empty() {
        return;
    }
    let config: Vec<f64> = line
        .split_whitespace()
        .filter_map(|s| s.parse().ok())
        .collect();
    if config.len() < 5 {
        return;
    }
    let width = config[0] as usize;
    let height = config[1] as usize;
    let gravity = config[2];
    let max_thrust = config[3];
    let fuel = config[4];
    line.clear();

    // 2nd line: start_x start_y target_x target_y
    if reader.read_line(&mut line).is_err() || line.trim().is_empty() {
        return;
    }
    let st_parts: Vec<f64> = line
        .split_whitespace()
        .filter_map(|s| s.parse().ok())
        .collect();
    if st_parts.len() < 4 {
        return;
    }
    let start_x = st_parts[0];
    let start_y = st_parts[1];
    let target_x = st_parts[2];
    let target_y = st_parts[3];
    line.clear();

    // Read the map.
    // For large maps (multi-GB), we store only '#' obstacles in a bitset.
    // 1 bit per voxel results in ~1.25 GB for a 100k x 100k map.
    let mut bitset = vec![0u64; (width * height + 63) / 64];
    for y in 0..height {
        let mut x = 0;
        loop {
            let buf = match reader.fill_buf() {
                Ok(b) if !b.is_empty() => b,
                _ => break,
            };
            let mut consumed = 0;
            for &b in buf {
                consumed += 1;
                if b == b'\n' {
                    break;
                }
                if x < width {
                    if b == b'#' {
                        let idx = y * width + x;
                        bitset[idx / 64] |= 1 << (idx % 64);
                    }
                    x += 1;
                }
            }
            reader.consume(consumed);
            if consumed > 0 && buf[consumed - 1] == b'\n' {
                break;
            }
        }
    }

    let state = Arc::new(Mutex::new(State {
        x: start_x,
        y: start_y,
        vx: 0.0,
        vy: 0.0,
        angle: 0.0,
        angular_vel: 0.0,
        fuel_remaining: fuel,
    }));

    let state_clone = Arc::clone(&state);
    // Spawn a background thread to read real-time state updates from stdin.
    thread::spawn(move || {
        let mut line = String::new();
        loop {
            line.clear();
            match reader.read_line(&mut line) {
                Ok(0) => break, // EOF
                Ok(_) => {
                    let parts: Vec<f64> = line
                        .split_whitespace()
                        .filter_map(|s| s.parse().ok())
                        .collect();
                    if parts.len() == 7 {
                        if let Ok(mut s) = state_clone.lock() {
                            s.x = parts[0];
                            s.y = parts[1];
                            s.vx = parts[2];
                            s.vy = parts[3];
                            s.angle = parts[4];
                            s.angular_vel = parts[5];
                            s.fuel_remaining = parts[6];
                        }
                    }
                }
                Err(_) => break,
            }
        }
    });

    let mut stdout = io::stdout().lock();
    let tick_dur = Duration::from_millis(40); // 25 Hz to satisfy the >20 Hz requirement.
    let mut next_tick = Instant::now();

    loop {
        let s = match state.lock() {
            Ok(s) => *s,
            Err(_) => break,
        };

        // Horizontal control (PD).
        let dx = target_x - s.x;
        // Target velocity scales with distance but is capped for safety.
        let mut target_vx = (dx * 0.5).clamp(-15.0, 15.0);
        // Fine-tune horizontal position when close to the target altitude.
        if (target_y - s.y).abs() < 5.0 {
            target_vx = (dx * 1.2).clamp(-1.0, 1.0);
        }
        let ax_des = (target_vx - s.vx) * 2.0;

        // Vertical control with terrain awareness.
        let ground_y = find_ground(&bitset, width, height, s.x, s.y);
        let dist_x = (target_x - s.x).abs();

        // Determine target altitude based on distance to landing site.
        let mut target_alt = target_y - 0.5; // Final landing height.
        if dist_x > 20.0 {
            // Stay high while traveling horizontally to avoid terrain.
            target_alt = (target_y - 30.0).min(ground_y - 20.0);
        } else if dist_x > 5.0 {
            // Gradual descent.
            target_alt = (target_y - 10.0).min(ground_y - 10.0);
        }

        let dy = target_alt - s.y;
        let mut target_vy = (dy * 0.5).clamp(-10.0, 5.0);
        // Soft landing constraint: limit downward velocity near ground.
        if (target_y - s.y).abs() < 3.0 {
            target_vy = target_vy.clamp(-1.2, 1.2);
        }

        // Vertical PD control.
        let mut ay_des = (target_vy - s.vy) * 2.5;

        // Emergency vertical brake to avoid terrain collision if not near target.
        let stop_dist_y = if s.vy > 0.0 {
            s.vy * s.vy / (2.0 * (max_thrust - gravity).max(0.1))
        } else {
            0.0
        };
        if (ground_y - s.y) < (stop_dist_y + 3.0) && s.vy > 0.5 && dist_x > 2.0 {
            ay_des = gravity - max_thrust; // Max thrust UP.
        }

        // Convert desired accelerations (ax_des, ay_des) into thrust and angle.
        // a_x = thrust * max_thrust * sin(angle)
        // a_y = gravity - thrust * max_thrust * cos(angle)
        let fx = ax_des;
        let fy = gravity - ay_des;
        let thrust_f = (fx * fx + fy * fy).sqrt();
        let thrust = (thrust_f / max_thrust).clamp(0.0, 1.0);
        let t_angle = fx.atan2(fy);

        // Angle control (PD) to achieve the target orientation.
        let mut angle_diff = t_angle - s.angle;
        while angle_diff > std::f64::consts::PI {
            angle_diff -= 2.0 * std::f64::consts::PI;
        }
        while angle_diff < -std::f64::consts::PI {
            angle_diff += 2.0 * std::f64::consts::PI;
        }
        let turn = (angle_diff * 8.0 - s.angular_vel * 3.0).clamp(-1.0, 1.0);

        // Output command at frequency.
        if writeln!(stdout, "{:.4} {:.4}", thrust, turn).is_err() {
            break;
        }
        let _ = stdout.flush();

        // Maintain constant frequency.
        let now = Instant::now();
        next_tick += tick_dur;
        if next_tick > now {
            thread::sleep(next_tick - now);
        } else {
            next_tick = now;
        }
    }
}
