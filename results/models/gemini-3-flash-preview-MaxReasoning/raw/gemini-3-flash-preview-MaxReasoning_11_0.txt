use std::io::{self, BufRead};
use std::f64::consts::PI;

struct Scanner<R: BufRead> {
    reader: R,
    buffer: Vec<String>,
}

impl<R: BufRead> Scanner<R> {
    fn new(reader: R) -> Self {
        Scanner { reader, buffer: Vec::new() }
    }
    fn next<T: std::str::FromStr>(&mut self) -> T {
        loop {
            if let Some(token) = self.buffer.pop() {
                return token.parse().ok().expect("Failed parse");
            }
            let mut line = String::new();
            if self.reader.read_line(&mut line).expect("Failed read") == 0 {
                return "0".parse().ok().unwrap();
            }
            self.buffer = line.split_whitespace().rev().map(String::from).collect();
        }
    }
}

#[derive(Clone, Copy, Debug)]
struct Point { x: f64, y: f64 }

struct Rect {
    min_x: f64, min_y: f64, max_x: f64, max_y: f64,
}

impl Rect {
    fn from_points(points: &[Point]) -> Self {
        let mut min_x = f64::INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut max_y = f64::NEG_INFINITY;
        for p in points {
            if p.x < min_x { min_x = p.x; }
            if p.y < min_y { min_y = p.y; }
            if p.x > max_x { max_x = p.x; }
            if p.y > max_y { max_y = p.y; }
        }
        Rect { min_x, min_y, max_x, max_y }
    }
    fn width(&self) -> f64 { self.max_x - self.min_x }
    fn height(&self) -> f64 { self.max_y - self.min_y }
}

#[derive(Clone)]
struct BitRow {
    data: Vec<u64>,
}

impl BitRow {
    fn new(n_bits: usize) -> Self {
        BitRow { data: vec![0; (n_bits + 63) / 64] }
    }
    fn copy_from(&mut self, other: &BitRow) {
        for i in 0..self.data.len().min(other.data.len()) {
            self.data[i] = other.data[i];
        }
    }
    fn or(&mut self, other: &BitRow) {
        for i in 0..self.data.len().min(other.data.len()) {
            self.data[i] |= other.data[i];
        }
    }
    fn or_shifted_right(&mut self, shift: usize) {
        if shift == 0 { return; }
        let word_shift = shift / 64;
        let bit_shift = shift % 64;
        let n = self.data.len();
        if bit_shift == 0 {
            for i in 0..n - word_shift {
                self.data[i] |= self.data[i + word_shift];
            }
        } else {
            let bit_shift_inv = 64 - bit_shift;
            for i in 0..n - word_shift {
                let mut val = self.data[i + word_shift] >> bit_shift;
                if i + word_shift + 1 < n {
                    val |= self.data[i + word_shift + 1] << bit_shift_inv;
                }
                self.data[i] |= val;
            }
        }
    }
    fn first_unset_bit(&self, max_bit: usize) -> Option<usize> {
        for i in 0..self.data.len() {
            let val = !self.data[i];
            if val != 0 {
                let bit = val.trailing_zeros() as usize;
                let res = i * 64 + bit;
                if res <= max_bit { return Some(res); }
            }
        }
        None
    }
    fn set_range(&mut self, start: usize, len: usize) {
        if len == 0 { return; }
        let end = start + len;
        let mut i = start;
        while i < end && i < self.data.len() * 64 {
            let word = i / 64;
            let bit = i % 64;
            let can_fill = 64 - bit;
            let to_fill = (end - i).min(can_fill);
            let mask = if to_fill == 64 { !0u64 } else { ((1u64 << to_fill) - 1) << bit };
            self.data[word] |= mask;
            i += to_fill;
        }
    }
}

fn apply_range_or(res: &mut BitRow, len: usize) {
    if len <= 1 { return; }
    let s = 1 << (usize::BITS - 1 - (len.leading_zeros()));
    let mut current_s = 1;
    while current_s < s {
        res.or_shifted_right(current_s);
        current_s *= 2;
    }
    if s < len {
        res.or_shifted_right(len - s);
    }
}

fn rasterize(poly: &[Point], grid_min_x: f64, grid_min_y: f64, res: f64, rows: usize, cols: usize) -> Vec<Vec<(usize, usize)>> {
    let mut intervals = vec![Vec::new(); rows];
    for r in 0..rows {
        let y = grid_min_y + (r as f64 + 0.5) * res;
        let mut intersects = Vec::new();
        for i in 0..poly.len() {
            let p1 = poly[i];
            let p2 = poly[(i + 1) % poly.len()];
            if (p1.y > y) != (p2.y > y) {
                let x = p1.x + (p2.x - p1.x) * (y - p1.y) / (p2.y - p1.y);
                let c = ((x - grid_min_x) / res).floor() as i32;
                intersects.push(c);
            }
        }
        intersects.sort();
        for i in (0..intersects.len()).step_by(2) {
            if i + 1 < intersects.len() {
                let start = intersects[i].max(0) as usize;
                let end = intersects[i+1].max(0) as i32;
                if start < cols {
                    let end_clamped = end.min(cols as i32 - 1) as usize;
                    if end_clamped >= start {
                        intervals[r].push((start, end_clamped - start + 1));
                    }
                }
            }
        }
    }
    intervals
}

struct RotationData {
    angle: f64,
    intervals: Vec<Vec<(usize, usize)>>,
    w: usize,
    h: usize,
    min_x: f64,
    min_y: f64,
}

struct PieceData {
    id: usize,
    rotations: Vec<RotationData>,
}

struct Stock {
    forbidden: Vec<BitRow>,
    st: Vec<Vec<BitRow>>,
    st_valid: bool,
}

impl Stock {
    fn new(forbidden: Vec<BitRow>) -> Self {
        Stock { forbidden, st: Vec::new(), st_valid: false }
    }
    fn rebuild_st(&mut self, rows: usize, cols: usize) {
        let n = rows;
        let max_k = (n as f32).log2() as usize + 1;
        self.st = vec![vec![BitRow::new(cols); n]; max_k];
        self.st[0] = self.forbidden.clone();
        for k in 1..max_k {
            let step = 1 << k;
            let half = 1 << (k - 1);
            for i in 0..=n - step {
                let mut row = self.st[k-1][i].clone();
                row.or(&self.st[k-1][i + half]);
                self.st[k][i] = row;
            }
        }
        self.st_valid = true;
    }
}

fn main() {
    let stdin = io::stdin();
    let mut scanner = Scanner::new(stdin.lock());
    
    let sv: usize = scanner.next();
    let np: usize = scanner.next();
    if sv == 0 { return; }
    
    let mut stock_poly = Vec::new();
    for _ in 0..sv {
        stock_poly.push(Point { x: scanner.next(), y: scanner.next() });
    }
    
    let mut pieces = Vec::new();
    for i in 0..np {
        let pv: usize = scanner.next();
        let mut poly = Vec::new();
        for _ in 0..pv {
            poly.push(Point { x: scanner.next(), y: scanner.next() });
        }
        let mut area = 0.0;
        for j in 0..poly.len() {
            let p1 = poly[j];
            let p2 = poly[(j + 1) % poly.len()];
            area += p1.x * p2.y - p2.x * p1.y;
        }
        pieces.push((i, poly, area.abs() * 0.5));
    }
    
    pieces.sort_by(|a, b| b.2.partial_cmp(&a.2).unwrap());
    
    let stock_bbox = Rect::from_points(&stock_poly);
    let res = (stock_bbox.width().max(stock_bbox.height()) / 1024.0).max(1e-6);
    let cols = (stock_bbox.width() / res).ceil() as usize + 1;
    let rows = (stock_bbox.height() / res).ceil() as usize + 1;
    
    let stock_intervals = rasterize(&stock_poly, stock_bbox.min_x, stock_bbox.min_y, res, rows, cols);
    let mut inside_mask = vec![BitRow::new(cols); rows];
    for r in 0..rows {
        for &(start, len) in &stock_intervals[r] {
            inside_mask[r].set_range(start, len);
        }
    }
    let mut forbidden_base = vec![BitRow::new(cols); rows];
    let last_word_mask = if cols % 64 == 0 { !0u64 } else { (1u64 << (cols % 64)) - 1 };
    for r in 0..rows {
        let n_words = forbidden_base[r].data.len();
        for c_word in 0..n_words {
            forbidden_base[r].data[c_word] = !inside_mask[r].data[c_word];
        }
        forbidden_base[r].data[n_words - 1] &= last_word_mask;
    }

    let mut piece_data = Vec::new();
    for (id, poly, _) in pieces {
        let mut rotations = Vec::new();
        let angles = if np > 1000 { vec![0.0, PI/2.0] } else { vec![0.0, PI/4.0, PI/2.0, 3.0*PI/4.0, PI, 5.0*PI/4.0, 3.0*PI/2.0, 7.0*PI/4.0] };
        for &angle in &angles {
            let mut rotated = Vec::new();
            let cos = angle.cos();
            let sin = angle.sin();
            for p in &poly {
                rotated.push(Point { x: p.x * cos - p.y * sin, y: p.x * sin + p.y * cos });
            }
            let bbox = Rect::from_points(&rotated);
            let w = (bbox.width() / res).ceil() as usize + 1;
            let h = (bbox.height() / res).ceil() as usize + 1;
            let intervals = rasterize(&rotated, bbox.min_x, bbox.min_y, res, h, w);
            rotations.push(RotationData { angle, intervals, w, h, min_x: bbox.min_x, min_y: bbox.min_y });
        }
        piece_data.push(PieceData { id, rotations });
    }
    
    let mut stocks = vec![Stock::new(forbidden_base.clone())];
    let mut results = vec![(0, 0.0, 0.0, 0.0); np];
    
    let mut combined_buf = BitRow::new(cols);
    for piece in piece_data {
        let mut placed = false;
        for s_idx in 0..stocks.len() {
            if !stocks[s_idx].st_valid {
                stocks[s_idx].rebuild_st(rows, cols);
            }
            for rot in &piece.rotations {
                if rot.h > rows || rot.w > cols { continue; }
                let k = (rot.h as f32).log2() as usize;
                let h_pow = 1 << k;
                
                for r in 0..=rows - rot.h {
                    combined_buf.copy_from(&stocks[s_idx].st[k][r]);
                    combined_buf.or(&stocks[s_idx].st[k][r + rot.h - h_pow]);
                    
                    apply_range_or(&mut combined_buf, rot.w);
                    if let Some(c) = combined_buf.first_unset_bit(cols - rot.w) {
                        for pr in 0..rot.h {
                            for &(start, len) in &rot.intervals[pr] {
                                stocks[s_idx].forbidden[r + pr].set_range(c + start, len);
                            }
                        }
                        let tx = stock_bbox.min_x + c as f64 * res - rot.min_x;
                        let ty = stock_bbox.min_y + r as f64 * res - rot.min_y;
                        results[piece.id] = (s_idx, tx, ty, rot.angle);
                        stocks[s_idx].st_valid = false;
                        placed = true;
                        break;
                    }
                }
                if placed { break; }
            }
            if placed { break; }
        }
        
        if !placed {
            let s_idx = stocks.len();
            stocks.push(Stock::new(forbidden_base.clone()));
            let rot = &piece.rotations[0];
            let r = 0;
            combined_buf.copy_from(&stocks[s_idx].forbidden[r]); // Simplified for new stock
            for pr in 1..rot.h {
                combined_buf.or(&stocks[s_idx].forbidden[r + pr]);
            }
            apply_range_or(&mut combined_buf, rot.w);
            let c = combined_buf.first_unset_bit(cols - rot.w).unwrap_or(0);
            
            for pr in 0..rot.h {
                for &(start, len) in &rot.intervals[pr] {
                    stocks[s_idx].forbidden[r + pr].set_range(c + start, len);
                }
            }
            let tx = stock_bbox.min_x + c as f64 * res - rot.min_x;
            let ty = stock_bbox.min_y + r as f64 * res - rot.min_y;
            results[piece.id] = (s_idx, tx, ty, rot.angle);
        }
    }
    
    println!("{}", stocks.len());
    for i in 0..np {
        let (s, x, y, r) = results[i];
        println!("{} {} {} {}", s, x, y, r);
    }
}
