#include <iostream>
#include <string>
#include <vector>
#include <thread>
#include <mutex>
#include <atomic>
#include <cmath>
#include <sstream>
#include <iomanip>
#include <fstream>
#include <algorithm>
#include <chrono>

#ifndef M_PI
#define M_PI 3.14159265358979323846
#endif

using namespace std;

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
    bool valid = false;
};

struct TerrainCell {
    uint16_t a, b, c;
};

class Terrain {
    string path;
    int width, depth;
    ifstream file;
    mutex file_mutex;
public:
    Terrain() : width(0), depth(0) {}
    void init(string p, int w, int d) {
        path = p;
        width = w;
        depth = d;
        file.open(path, ios::binary);
    }
    TerrainCell get(int x, int y) {
        if (x < 0 || x >= width || y < 0 || y >= depth) return {0, 0, 0};
        lock_guard<mutex> lock(file_mutex);
        if (!file.is_open()) return {0, 0, 0};
        file.seekg(((long long)x * depth + y) * 6, ios::beg);
        uint16_t vals[3] = {0, 0, 0};
        file.read((char*)vals, 6);
        return {vals[0], vals[1], vals[2]};
    }
};

State g_state;
mutex g_state_mutex;
atomic<bool> g_running(true);

void input_thread_func() {
    string line;
    while (g_running) {
        if (!getline(cin, line)) break;
        if (line.empty() || line == "STATE") continue;

        stringstream ss(line);
        State s;
        if (ss >> s.x >> s.y >> s.z >> s.vx >> s.vy >> s.vz 
               >> s.pitch >> s.yaw >> s.roll 
               >> s.pitch_rate >> s.yaw_rate >> s.roll_rate 
               >> s.fuel) {
            lock_guard<mutex> lock(g_state_mutex);
            g_state = s;
            g_state.valid = true;
        }
    }
    g_running = false;
}

atomic<double> out_thrust(0), out_p(0), out_y(0), out_r(0);

void output_thread_func() {
    while (g_running) {
        auto start = chrono::steady_clock::now();
        
        cout << fixed << setprecision(6) 
             << out_thrust.load() << " " << out_p.load() << " " 
             << out_y.load() << " " << out_r.load() << endl;
        
        auto end = chrono::steady_clock::now();
        auto diff = chrono::duration_cast<chrono::milliseconds>(end - start).count();
        if (diff < 100) {
            this_thread::sleep_for(chrono::milliseconds(100 - diff));
        }
    }
}

double normalize_angle(double a) {
    while (a > M_PI) a -= 2 * M_PI;
    while (a < -M_PI) a += 2 * M_PI;
    return a;
}

atomic<uint16_t> g_max_a(0), g_max_c(0);

void terrain_scan_thread_func(string path) {
    ifstream f(path, ios::binary);
    if (!f) return;
    const size_t buf_size = 64 * 1024;
    vector<uint16_t> buffer(buf_size);
    uint16_t local_max_a = 0, local_max_c = 0;
    while (g_running && f.read((char*)buffer.data(), buf_size * 2)) {
        size_t count = f.gcount() / 2;
        for (size_t i = 0; i < count; i += 3) {
            if (buffer[i] > local_max_a) local_max_a = buffer[i];
            if (i + 2 < count) {
                if (buffer[i+2] > local_max_c) local_max_c = buffer[i+2];
            }
        }
        g_max_a = local_max_a;
        g_max_c = local_max_c;
    }
    f.close();
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    double width, depth, height, gravity, max_thrust, fuel_initial;
    if (!(cin >> width >> depth >> height >> gravity >> max_thrust >> fuel_initial)) return 0;

    double start_x, start_y, start_z, target_x, target_y, target_z;
    if (!(cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z)) return 0;

    string terrain_file_path;
    getline(cin, terrain_file_path); // consume newline
    if (!getline(cin, terrain_file_path)) return 0;

    Terrain terrain;
    terrain.init(terrain_file_path, (int)width, (int)depth);

    string line;
    while (getline(cin, line)) {
        if (line == "STATE") break;
    }

    thread in_thread(input_thread_func);
    thread out_thread(output_thread_func);
    thread scan_thread(terrain_scan_thread_func, terrain_file_path);

    // Initial safe altitude estimate
    double safe_z = max(start_z, target_z) + 15.0;
    
    // Controller gains
    double Kp_pos = 0.5;
    double Kd_pos = 1.5;
    double Kp_rot = 10.0;
    double Kd_rot = 3.0;

    while (g_running) {
        State s;
        {
            lock_guard<mutex> lock(g_state_mutex);
            if (!g_state.valid) {
                lock.unlock();
                this_thread::sleep_for(chrono::milliseconds(10));
                continue;
            }
            s = g_state;
        }

        TerrainCell cur_cell = terrain.get((int)s.x, (int)s.y);
        
        double dx = target_x - s.x;
        double dy = target_y - s.y;
        double dist_xy = sqrt(dx*dx + dy*dy);
        
        double target_vz = 0;
        double cur_target_z = safe_z;

        // Dynamic safe altitude based on scanned terrain
        double current_max_c = (double)g_max_c.load();
        double current_max_a = (double)g_max_a.load();
        double global_safe_z = max(safe_z, current_max_c + 15.0);
        global_safe_z = min(global_safe_z, height - 5.0);

        // Landing logic
        double lz_radius = max(5.0, min(width, depth) * 0.02);
        if (dist_xy < lz_radius * 0.9) {
            cur_target_z = target_z;
            double dz = target_z - s.z;
            if (abs(dz) < 0.5) target_vz = 0;
            else target_vz = clamp(dz * 0.8, -2.0, 2.0);
        } else {
            // Check if we are under a rock
            if (s.z < (double)cur_cell.b && cur_cell.b != cur_cell.c) {
                // Stay in the gap
                cur_target_z = (cur_cell.a + cur_cell.b) / 2.0;
                target_vz = clamp((cur_target_z - s.z) * 0.5, -2.0, 2.0);
            } else {
                cur_target_z = global_safe_z;
                target_vz = clamp((cur_target_z - s.z) * 0.5, -5.0, 5.0);
            }
        }

        // Horizontal speed limit
        double max_v = 100.0; // Fast transit
        if (dist_xy < 200.0) max_v = 3.0 + (dist_xy / 200.0) * 97.0;
        if (dist_xy < 20.0) max_v = 1.0; // Very slow near target
        
        double des_vx = clamp(dx * 0.5, -max_v, max_v);
        double des_vy = clamp(dy * 0.5, -max_v, max_v);
        
        double ax = 1.5 * (des_vx - s.vx);
        double ay = 1.5 * (des_vy - s.vy);
        double az = 1.5 * (target_vz - s.vz) + 0.2 * (cur_target_z - s.z);

        // Limit horizontal tilt to avoid losing too much lift
        double total_a_horiz = sqrt(ax*ax + ay*ay);
        double max_a_horiz = max_thrust * 0.7;
        if (total_a_horiz > max_a_horiz) {
            ax *= max_a_horiz / total_a_horiz;
            ay *= max_a_horiz / total_a_horiz;
        }

        double total_ax = ax;
        double total_ay = ay;
        double total_az = az + gravity;

        double thrust_mag = sqrt(total_ax*total_ax + total_ay*total_ay + total_az*total_az);
        double thrust_cmd = clamp(thrust_mag / max_thrust, 0.0, 1.0);

        // Desired orientation
        double nx = total_ax / (thrust_mag + 1e-9);
        double ny = total_ay / (thrust_mag + 1e-9);
        double nz = total_az / (thrust_mag + 1e-9);

        double des_pitch = acos(clamp(nz, -1.0, 1.0));
        double des_yaw = atan2(-ny, -nx);

        if (thrust_mag < 0.1) {
            des_pitch = 0;
            des_yaw = s.yaw;
        }

        double pitch_err = des_pitch - s.pitch;
        double yaw_err = normalize_angle(des_yaw - s.yaw);

        double p_cmd = Kp_rot * pitch_err - Kd_rot * s.pitch_rate;
        double y_cmd = Kp_rot * yaw_err - Kd_rot * s.yaw_rate;
        double r_cmd = -2.0 * s.roll - 1.0 * s.roll_rate;

        out_thrust = thrust_cmd;
        out_p = clamp(p_cmd / 2.0, -1.0, 1.0);
        out_y = clamp(y_cmd / 2.0, -1.0, 1.0);
        out_r = clamp(r_cmd / 2.0, -1.0, 1.0);

        this_thread::sleep_for(chrono::milliseconds(20));
    }

    if (in_thread.joinable()) in_thread.join();
    if (out_thread.joinable()) out_thread.join();
    if (scan_thread.joinable()) scan_thread.join();

    return 0;
}
