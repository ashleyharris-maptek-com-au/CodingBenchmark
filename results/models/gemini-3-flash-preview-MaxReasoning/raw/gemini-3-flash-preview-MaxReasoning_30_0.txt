import sys
import array

def solve():
    # Use fast I/O
    # Reading num_vars and num_clauses
    line = sys.stdin.readline()
    while line and not line.strip():
        line = sys.stdin.readline()
    if not line:
        return
    
    try:
        parts = line.split()
        num_vars = int(parts[0])
        num_clauses = int(parts[1])
        first_lits = parts[2:]
    except (ValueError, IndexError):
        return

    # Literal index: 2*v for positive v, 2*v+1 for negative v
    # Store clauses in a flat array
    clauses = array.array('I', [0] * (3 * num_clauses))
    c_ptr = 0
    
    def get_words(initial_parts):
        for s in initial_parts:
            yield s
        for line in sys.stdin:
            for s in line.split():
                yield s

    # Read clauses
    words = get_words(first_lits)
    for s in words:
        v = int(s)
        if v > 0:
            clauses[c_ptr] = v << 1
        else:
            clauses[c_ptr] = ((-v) << 1) | 1
        c_ptr += 1
        if c_ptr >= 3 * num_clauses:
            break

    # Two-Watched Literals (TWL)
    # heads[lit] stores the index of the first watch in next_watch
    # next_watch entries are 2*cid for slot 0 and 2*cid+1 for slot 1
    heads = array.array('i', [-1] * (2 * num_vars + 2))
    next_watch = array.array('i', [-1] * (2 * num_clauses))
    
    for i in range(num_clauses):
        # Initial watches: first two literals
        l0, l1 = clauses[3*i], clauses[3*i+1]
        
        # Entry for slot 0 (2*i)
        next_watch[2*i] = heads[l0]
        heads[l0] = 2*i
        # Entry for slot 1 (2*i+1)
        next_watch[2*i+1] = heads[l1]
        heads[l1] = 2*i+1

    # Solver state
    assigned_lit = array.array('I', [0] * (num_vars + 1))
    trail = array.array('I', [0] * num_vars)
    trail_ptr = 0
    prop_ptr = 0
    
    # level_info: (trail_ptr, decision_lit)
    level_info = []
    
    curr_var = 1
    
    while True:
        # 1. Propagation
        conflict = False
        while prop_ptr < trail_ptr:
            lit = trail[prop_ptr]
            prop_ptr += 1
            neg_lit = lit ^ 1
            
            curr = heads[neg_lit]
            prev = -1
            while curr != -1:
                cid = curr >> 1
                slot = curr & 1
                c_base = 3 * cid
                
                # l0 is the literal being replaced, l1 is the other watch, l2 is third
                if slot == 0:
                    # slot 0 is clauses[c_base], slot 1 is clauses[c_base+1]
                    l1 = clauses[c_base+1]
                else:
                    l1 = clauses[c_base]
                
                # If the other watched literal is already True, clause is satisfied
                if assigned_lit[l1 >> 1] == l1:
                    prev = curr
                    curr = next_watch[curr]
                    continue
                
                # Try to find a new literal to watch in the third position
                l2 = clauses[c_base+2]
                if assigned_lit[l2 >> 1] != (l2 ^ 1):
                    # Found new literal to watch
                    if slot == 0:
                        clauses[c_base], clauses[c_base+2] = l2, clauses[c_base]
                    else:
                        clauses[c_base+1], clauses[c_base+2] = l2, clauses[c_base+1]
                    
                    # Update linked lists: remove curr from neg_lit list
                    nxt = next_watch[curr]
                    if prev == -1:
                        heads[neg_lit] = nxt
                    else:
                        next_watch[prev] = nxt
                    
                    # Add curr to the list of the new literal l2
                    next_watch[curr] = heads[l2]
                    heads[l2] = curr
                    
                    curr = nxt
                    continue
                
                # No new literal. Check if l1 is unassigned (unit clause) or False (conflict)
                if assigned_lit[l1 >> 1] == 0:
                    # Unit clause!
                    assigned_lit[l1 >> 1] = l1
                    trail[trail_ptr] = l1
                    trail_ptr += 1
                    prev = curr
                    curr = next_watch[curr]
                else:
                    # Conflict!
                    conflict = True
                    break
            if conflict:
                break
        
        if conflict:
            # 2. Backtrack
            while level_info:
                old_trail_ptr, dec_lit = level_info.pop()
                # Unassign literals
                for i in range(old_trail_ptr, trail_ptr):
                    assigned_lit[trail[i] >> 1] = 0
                trail_ptr = old_trail_ptr
                prop_ptr = old_trail_ptr
                
                if dec_lit != 0:
                    # Try the other branch for this variable
                    new_lit = dec_lit ^ 1
                    assigned_lit[new_lit >> 1] = new_lit
                    trail[trail_ptr] = new_lit
                    trail_ptr += 1
                    # Push back level_info with 0 to mark it as the second branch
                    level_info.append((old_trail_ptr, 0))
                    conflict = False
                    break
            if conflict:
                # Guaranteed satisfiable, so this shouldn't be reached
                return
        else:
            # 3. Decision
            while curr_var <= num_vars and assigned_lit[curr_var] != 0:
                curr_var += 1
            
            if curr_var > num_vars:
                # All variables assigned, SAT!
                break
            
            # Decision: try setting variable to True (2*curr_var)
            level_info.append((trail_ptr, 2 * curr_var))
            assigned_lit[curr_var] = 2 * curr_var
            trail[trail_ptr] = 2 * curr_var
            trail_ptr += 1
            
    # Output result
    print("SAT")
    res = ["1"] * num_vars
    for i in range(1, num_vars + 1):
        if (assigned_lit[i] & 1) == 1: # Lit index is odd -> False
            res[i-1] = "0"
    sys.stdout.write(" ".join(res) + "\n")

if __name__ == "__main__":
    solve()
