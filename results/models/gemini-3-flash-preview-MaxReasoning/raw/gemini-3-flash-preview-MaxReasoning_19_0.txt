using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

namespace MiningSolution
{
    class Program
    {
        static int XV, YV, ZV;
        static double XB, YB, ZB;
        static short[,] SurfaceHeight;
        static bool[,] HasOreBelow;
        static byte[] Grid; // Limited grid storage
        static int GridXV, GridYV; // Actual dimensions stored in Grid

        static void Main(string[] args)
        {
            try
            {
                var headerLine = Console.ReadLine();
                if (string.IsNullOrWhiteSpace(headerLine)) return;
                var parts = headerLine.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
                if (parts.Length < 6) return;

                XV = int.Parse(parts[0]);
                YV = int.Parse(parts[1]);
                ZV = int.Parse(parts[2]);
                XB = double.Parse(parts[3]);
                YB = double.Parse(parts[4]);
                ZB = double.Parse(parts[5]);

                SurfaceHeight = new short[XV, YV];
                HasOreBelow = new bool[XV, YV];

                // Determine how much of the grid we can store in memory
                // 32GB is plenty, but let's be conservative and use up to 4GB for the grid.
                // 4 * 10^9 voxels. 
                // If ZV is 100, we can store 40,000,000 columns. (6300x6300)
                // If ZV is 5000, we can store 800,000 columns. (894x894)
                GridXV = Math.Min(XV, 1000);
                GridYV = Math.Min(YV, 1000);
                while ((long)GridXV * GridYV * ZV > 1_000_000_000)
                {
                    if (GridXV > GridYV) GridXV--; else GridYV--;
                }

                Grid = new byte[(int)((long)GridXV * GridYV * ZV)];

                using (var stdin = Console.OpenStandardInput())
                {
                    byte[] buffer = new byte[65536];
                    for (int x = 0; x < XV; x++)
                    {
                        for (int y = 0; y < YV; y++)
                        {
                            SurfaceHeight[x, y] = -1;
                            for (int z = 0; z < ZV; z++)
                            {
                                int dByte = stdin.ReadByte();
                                if (dByte == -1) break;
                                byte d = (byte)dByte;
                                int mass = d % 16;
                                int val = d / 16;

                                if (mass > 0)
                                {
                                    SurfaceHeight[x, y] = (short)z;
                                    if (val > 0) HasOreBelow[x, y] = true;
                                }

                                if (x < GridXV && y < GridYV)
                                {
                                    int idx = (int)((long)x * GridYV * ZV + (long)y * ZV + z);
                                    Grid[idx] = d;
                                }
                            }
                        }
                    }
                }

                List<string> operations = new List<string>();

                int dumpX = XV - 1;
                int dumpY = YV - 1;

                // Layer-by-layer mining
                for (int z = ZV - 1; z >= 0; z--)
                {
                    for (int x = 0; x < GridXV; x++)
                    {
                        for (int y = 0; y < GridYV; y++)
                        {
                            if (SurfaceHeight[x, y] == z)
                            {
                                int idx = (int)((long)x * GridYV * ZV + (long)y * ZV + z);
                                byte d = Grid[idx];
                                int mass = d % 16;
                                int val = d / 16;

                                if (val > 0)
                                {
                                    // Valuable ore -> Processing plant (0,0)
                                    operations.Add(GenerateMove(x, y, z, 0, 0));
                                    SurfaceHeight[x, y]--;
                                }
                                else if (HasOreBelow[x, y])
                                {
                                    // Junk blocking ore -> Dump in a corner
                                    // Spread junk slightly to avoid huge piles if possible
                                    int curDumpX = dumpX;
                                    int curDumpY = dumpY;
                                    
                                    // If we are mining near the dump site, move the dump site
                                    if (Math.Abs(x - dumpX) < 2 && Math.Abs(y - dumpY) < 2)
                                    {
                                        curDumpX = 0;
                                        curDumpY = YV - 1;
                                    }

                                    operations.Add(GenerateMove(x, y, z, curDumpX, curDumpY));
                                    SurfaceHeight[x, y]--;
                                }
                                // Else skip junk to save fuel
                            }
                        }
                    }
                }

                Console.WriteLine(operations.Count);
                foreach (var op in operations)
                {
                    Console.WriteLine(op);
                }
            }
            catch (Exception)
            {
                // Silently exit or handle error
                Console.WriteLine(0);
            }
        }

        static string GenerateMove(int x, int y, int z, int tx, int ty)
        {
            List<int> steps = new List<int>();
            int cx = x, cy = y;
            while (cx != tx || cy != ty)
            {
                if (cx < tx) cx++;
                else if (cx > tx) cx--;
                else if (cy < ty) cy++;
                else if (cy > ty) cy--;
                steps.Add(cx);
                steps.Add(cy);
            }

            if (steps.Count == 0)
            {
                // Already at destination? (e.g. ore at 0,0)
                return $"0 {x} {y} {z} {tx} {ty}";
            }

            StringBuilder sb = new StringBuilder();
            sb.Append(steps.Count / 2);
            sb.Append(" ");
            sb.Append(x);
            sb.Append(" ");
            sb.Append(y);
            sb.Append(" ");
            sb.Append(z);
            sb.Append(" ");
            for (int i = 0; i < steps.Count; i++)
            {
                sb.Append(steps[i]);
                sb.Append(" ");
            }
            sb.Append(tx);
            sb.Append(" ");
            sb.Append(ty);
            return sb.ToString();
        }
    }
}
