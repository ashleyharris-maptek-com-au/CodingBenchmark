using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

namespace CPPSolver
{
    class Program
    {
        struct EdgeInput
        {
            public int U, V, W;
        }

        class FastReader
        {
            Stream stream;
            byte[] buffer = new byte[65536];
            int pos, len;

            public FastReader(Stream s)
            {
                stream = s;
            }

            private byte Read()
            {
                if (pos == len)
                {
                    pos = 0;
                    len = stream.Read(buffer, 0, buffer.Length);
                    if (len == 0) return 0;
                }
                return buffer[pos++];
            }

            public int NextInt()
            {
                int res = 0;
                byte b = Read();
                while (b != 0 && b <= 32) b = Read();
                if (b == 0) return -1;
                bool neg = false;
                if (b == '-') { neg = true; b = Read(); }
                while (b >= '0' && b <= '9')
                {
                    res = res * 10 + (b - '0');
                    b = Read();
                }
                return neg ? -res : res;
            }
        }

        static void Main(string[] args)
        {
            FastReader fr = new FastReader(Console.OpenStandardInput());
            int N = fr.NextInt();
            int M = fr.NextInt();
            if (N == -1) return;
            if (N == 0) return;

            EdgeInput[] inputEdges = new EdgeInput[M];
            int[] degree = new int[N];
            for (int i = 0; i < M; i++)
            {
                inputEdges[i].U = fr.NextInt();
                inputEdges[i].V = fr.NextInt();
                inputEdges[i].W = fr.NextInt();
                if (inputEdges[i].U >= 0 && inputEdges[i].U < N && 
                    inputEdges[i].V >= 0 && inputEdges[i].V < N)
                {
                    degree[inputEdges[i].U]++;
                    degree[inputEdges[i].V]++;
                }
            }

            // 1. MST using Kruskal's as a heuristic for matching
            long[] packedEdges = new long[M];
            for (int i = 0; i < M; i++)
            {
                packedEdges[i] = ((long)inputEdges[i].W << 32) | (uint)i;
            }
            Array.Sort(packedEdges);

            int[] parentDSU = new int[N];
            for (int i = 0; i < N; i++) parentDSU[i] = i;

            int Find(int i)
            {
                int root = i;
                while (parentDSU[root] != root) root = parentDSU[root];
                while (parentDSU[i] != root)
                {
                    int next = parentDSU[i];
                    parentDSU[i] = root;
                    i = next;
                }
                return root;
            }

            int[] mstHead = new int[N];
            for (int i = 0; i < N; i++) mstHead[i] = -1;
            int[] mstTo = new int[2 * N];
            int[] mstNext = new int[2 * N];
            int[] mstEdgeIdx = new int[2 * N];
            int mstEdgeCount = 0;

            for (int i = 0; i < M; i++)
            {
                int idx = (int)(packedEdges[i] & 0xFFFFFFFF);
                int rootU = Find(inputEdges[idx].U);
                int rootV = Find(inputEdges[idx].V);
                if (rootU != rootV)
                {
                    parentDSU[rootU] = rootV;
                    
                    mstTo[mstEdgeCount] = inputEdges[idx].V;
                    mstNext[mstEdgeCount] = mstHead[inputEdges[idx].U];
                    mstEdgeIdx[mstEdgeCount] = idx;
                    mstHead[inputEdges[idx].U] = mstEdgeCount++;

                    mstTo[mstEdgeCount] = inputEdges[idx].U;
                    mstNext[mstEdgeCount] = mstHead[inputEdges[idx].V];
                    mstEdgeIdx[mstEdgeCount] = idx;
                    mstHead[inputEdges[idx].V] = mstEdgeCount++;
                }
            }

            // 2. Parity fix using spanning forest
            List<int> extraEdges = new List<int>();
            int[] order = new int[N];
            int[] parentNode = new int[N];
            int[] parentEdgeIdx = new int[N];
            bool[] visited = new bool[N];
            int tailPtr = 0;

            for (int i = 0; i < N; i++)
            {
                if (!visited[i])
                {
                    int qHead = tailPtr;
                    order[tailPtr++] = i;
                    visited[i] = true;
                    parentNode[i] = -1;
                    while (qHead < tailPtr)
                    {
                        int u = order[qHead++];
                        for (int e = mstHead[u]; e != -1; e = mstNext[e])
                        {
                            int v = mstTo[e];
                            if (!visited[v])
                            {
                                visited[v] = true;
                                parentNode[v] = u;
                                parentEdgeIdx[v] = mstEdgeIdx[e];
                                order[tailPtr++] = v;
                            }
                        }
                    }
                }
            }

            for (int i = tailPtr - 1; i >= 0; i--)
            {
                int v = order[i];
                if (parentNode[v] != -1 && degree[v] % 2 != 0)
                {
                    extraEdges.Add(parentEdgeIdx[v]);
                    degree[v]++;
                    degree[parentNode[v]]++;
                }
            }

            // 3. Hierholzer's Algorithm for Eulerian Circuit
            int[] head = new int[N];
            for (int i = 0; i < N; i++) head[i] = -1;
            int totalEdgesCount = M + extraEdges.Count;
            int[] to = new int[2 * totalEdgesCount];
            int[] next = new int[2 * totalEdgesCount];
            bool[] used = new bool[totalEdgesCount];
            int edgeCount = 0;

            void AddEdge(int u, int v)
            {
                to[edgeCount] = v;
                next[edgeCount] = head[u];
                head[u] = edgeCount++;
                to[edgeCount] = u;
                next[edgeCount] = head[v];
                head[v] = edgeCount++;
            }

            for (int i = 0; i < M; i++)
            {
                AddEdge(inputEdges[i].U, inputEdges[i].V);
            }
            foreach (int idx in extraEdges)
            {
                AddEdge(inputEdges[idx].U, inputEdges[idx].V);
            }

            int[] route = new int[totalEdgesCount + 1];
            int routePtr = 0;
            Stack<int> stack = new Stack<int>();
            stack.Push(0);

            while (stack.Count > 0)
            {
                int u = stack.Peek();
                bool found = false;
                while (head[u] != -1)
                {
                    int e = head[u];
                    head[u] = next[e];
                    if (used[e / 2]) continue;
                    used[e / 2] = true;
                    stack.Push(to[e]);
                    found = true;
                    break;
                }
                if (!found)
                {
                    route[routePtr++] = stack.Pop();
                }
            }

            // 4. Output the result
            using (StreamWriter sw = new StreamWriter(Console.OpenStandardOutput(), Encoding.ASCII, 1024 * 1024))
            {
                sw.WriteLine(routePtr);
                char[] buf = new char[12];
                for (int i = 0; i < routePtr; i++)
                {
                    int val = route[i];
                    if (val == 0) sw.Write('0');
                    else
                    {
                        if (val < 0) { sw.Write('-'); val = -val; }
                        int p = 0;
                        while (val > 0)
                        {
                            buf[p++] = (char)('0' + (val % 10));
                            val /= 10;
                        }
                        while (p > 0) sw.Write(buf[--p]);
                    }
                    if (i < routePtr - 1) sw.Write(' ');
                }
                sw.WriteLine();
            }
        }
    }
}
