using namespace std;

const double EPS = 1e-9;

struct Vec3 {
    double x, y, z;
    Vec3 operator+(const Vec3& v) const { return {x + v.x, y + v.y, z + v.z}; }
    Vec3 operator-(const Vec3& v) const { return {x - v.x, y - v.y, z - v.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const { return {y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x}; }
    double lengthSq() const { return dot(*this); }
    double length() const { return sqrt(lengthSq()); }
};

struct BBox {
    Vec3 min, max;
    BBox() {
        min = {1e18, 1e18, 1e18};
        max = {-1e18, -1e18, -1e18};
    }
    void add(const Vec3& v) {
        min.x = std::min(min.x, v.x); min.y = std::min(min.y, v.y); min.z = std::min(min.z, v.z);
        max.x = std::max(max.x, v.x); max.y = std::max(max.y, v.y); max.z = std::max(max.z, v.z);
    }
    bool intersects(const BBox& other) const {
        return (min.x - EPS <= other.max.x && max.x + EPS >= other.min.x) &&
               (min.y - EPS <= other.max.y && max.y + EPS >= other.min.y) &&
               (min.z - EPS <= other.max.z && max.z + EPS >= other.min.z);
    }
};

struct Tri {
    int v[3];
};

struct Mesh {
    vector<Vec3> vertices;
    vector<Tri> faces;
};

struct BVHNode {
    BBox bbox;
    int left, right;
};

class BVH {
public:
    vector<BVHNode> nodes;
    vector<int> item_indices;
    const Mesh& mesh;

    BVH(const Mesh& m) : mesh(m) {
        int n = mesh.faces.size();
        item_indices.resize(n);
        for (int i = 0; i < n; ++i) item_indices[i] = i;
        if (n > 0) build(0, n);
    }

    int build(int start, int end) {
        int node_idx = nodes.size();
        nodes.emplace_back();
        BBox bounds;
        for (int i = start; i < end; ++i) {
            int tri_idx = item_indices[i];
            for (int k = 0; k < 3; ++k) bounds.add(mesh.vertices[mesh.faces[tri_idx].v[k]]);
        }
        nodes[node_idx].bbox = bounds;

        int count = end - start;
        if (count <= 8) {
            nodes[node_idx].left = -start - 1;
            nodes[node_idx].right = count;
        } else {
            Vec3 size = bounds.max - bounds.min;
            int axis = (size.x > size.y && size.x > size.z) ? 0 : (size.y > size.z ? 1 : 2);
            double mid = 0;
            for (int i = start; i < end; ++i) {
                const auto& f = mesh.faces[item_indices[i]];
                Vec3 center = (mesh.vertices[f.v[0]] + mesh.vertices[f.v[1]] + mesh.vertices[f.v[2]]) / 3.0;
                mid += (axis == 0 ? center.x : axis == 1 ? center.y : center.z);
            }
            mid /= count;
            auto it = partition(item_indices.begin() + start, item_indices.begin() + end, [&](int idx) {
                const auto& f = mesh.faces[idx];
                Vec3 c = (mesh.vertices[f.v[0]] + mesh.vertices[f.v[1]] + mesh.vertices[f.v[2]]) / 3.0;
                return (axis == 0 ? c.x : axis == 1 ? c.y : c.z) < mid;
            });
            int split = distance(item_indices.begin(), it);
            if (split == start || split == end) split = start + count / 2;
            int l = build(start, split);
            int r = build(split, end);
            nodes[node_idx].left = l;
            nodes[node_idx].right = r;
        }
        return node_idx;
    }

    void intersect(const BBox& box, vector<int>& candidates, int node = 0) const {
        if (nodes.empty()) return;
        if (!nodes[node].bbox.intersects(box)) return;
        if (nodes[node].left < 0) {
            int start = -nodes[node].left - 1;
            for (int i = 0; i < nodes[node].right; ++i) candidates.push_back(item_indices[start + i]);
        } else {
            intersect(box, candidates, nodes[node].left);
            intersect(box, candidates, nodes[node].right);
        }
    }
};

struct Plane {
    Vec3 n;
    double d;
    Plane(const Vec3& p0, const Vec3& p1, const Vec3& p2) {
        n = (p1 - p0).cross(p2 - p0);
        double len = n.length();
        if (len > 1e-18) n = n / len;
        d = n.dot(p0);
    }
    Plane(const Vec3& p0, const Vec3& normal) : n(normal), d(normal.dot(p0)) {}
};

bool isPointInTri(const Vec3& p, const Vec3& v0, const Vec3& v1, const Vec3& v2) {
    Vec3 n = (v1 - v0).cross(v2 - v0);
    if (n.dot((v1 - v0).cross(p - v0)) < -1e-10) return false;
    if (n.dot((v2 - v1).cross(p - v1)) < -1e-10) return false;
    if (n.dot((v0 - v2).cross(p - v2)) < -1e-10) return false;
    return true;
}

bool triTriIntersect(const Vec3* A, const Vec3* B) {
    BBox boxA, boxB;
    for (int i = 0; i < 3; ++i) { boxA.add(A[i]); boxB.add(B[i]); }
    if (!boxA.intersects(boxB)) return false;
    Plane pA(A[0], A[1], A[2]), pB(B[0], B[1], B[2]);
    auto side = [&](const Vec3* V, const Plane& p) {
        int pos = 0, neg = 0;
        for (int i = 0; i < 3; ++i) {
            double d = V[i].dot(p.n) - p.d;
            if (d > 1e-10) pos++; else if (d < -1e-10) neg++;
        }
        return (pos > 0 && neg > 0);
    };
    if (side(A, pB) || side(B, pA)) return true;
    if (abs(pA.n.dot(pB.n)) > 1.0 - 1e-7 && abs(pA.d - pB.d) < 1e-7) {
        // Coplanar overlap check
        for (int i = 0; i < 3; ++i) if (isPointInTri(A[i], B[0], B[1], B[2])) return true;
        for (int i = 0; i < 3; ++i) if (isPointInTri(B[i], A[0], A[1], A[2])) return true;
    }
    return false;
}

void splitPolygon(const vector<Vec3>& poly, const Plane& p, vector<vector<Vec3>>& out) {
    vector<Vec3> front, back;
    for (int i = 0; i < poly.size(); ++i) {
        Vec3 a = poly[i], b = poly[(i + 1) % poly.size()];
        double da = a.dot(p.n) - p.d;
        double db = b.dot(p.n) - p.d;
        if (abs(da) < 1e-10) da = 0;
        if (abs(db) < 1e-10) db = 0;
        if (da >= 0) front.push_back(a);
        if (da <= 0) back.push_back(a);
        if (da * db < 0) {
            Vec3 m = a + (b - a) * (da / (da - db));
            front.push_back(m); back.push_back(m);
        }
    }
    auto clean = [&](vector<Vec3>& v) {
        if (v.size() < 3) { v.clear(); return; }
        vector<Vec3> r;
        for (int i = 0; i < v.size(); ++i) if ((v[i] - v[(i + 1) % v.size()]).lengthSq() > 1e-18) r.push_back(v[i]);
        if (r.size() < 3) v.clear(); else v = r;
    };
    clean(front); clean(back);
    if (!front.empty()) out.push_back(front);
    if (!back.empty()) out.push_back(back);
}

bool intersectRayTri(const Vec3& orig, const Vec3& dir, const Vec3& v0, const Vec3& v1, const Vec3& v2, double& t) {
    Vec3 e1 = v1 - v0, e2 = v2 - v0;
    Vec3 pvec = dir.cross(e2);
    double det = e1.dot(pvec);
    if (abs(det) < 1e-12) return false;
    double invDet = 1.0 / det;
    Vec3 tvec = orig - v0;
    double u = tvec.dot(pvec) * invDet;
    if (u < 0 || u > 1) return false;
    Vec3 qvec = tvec.cross(e1);
    double v = dir.dot(qvec) * invDet;
    if (v < 0 || u + v > 1) return false;
    t = e2.dot(qvec) * invDet;
    return t > 1e-9;
}

int getWinding(const BVH& bvh, const Vec3& p) {
    int count = 0;
    Vec3 dir = {1.0, 0.0123, 0.0456};
    dir = dir / dir.length();
    vector<int> candidates;
    BBox rayBox; rayBox.add(p); rayBox.add(p + dir * 1e6);
    bvh.intersect(rayBox, candidates);
    for (int idx : candidates) {
        const auto& f = bvh.mesh.faces[idx];
        double t;
        if (intersectRayTri(p, dir, bvh.mesh.vertices[f.v[0]], bvh.mesh.vertices[f.v[1]], bvh.mesh.vertices[f.v[2]], t)) count++;
    }
    return count % 2;
}

struct Coord {
    long long x, y, z;
    bool operator==(const Coord& o) const { return x == o.x && y == o.y && z == o.z; }
};
struct CoordHasher {
    size_t operator()(const Coord& c) const {
        return hash<long long>{}(c.x) ^ (hash<long long>{}(c.y) << 1) ^ (hash<long long>{}(c.z) << 2);
    }
};

int main() {
    int VA, FA, VB, FB;
    if (scanf("%d %d %d %d", &VA, &FA, &VB, &FB) != 4) return 0;
    Mesh meshA, meshB;
    meshA.vertices.resize(VA); for (int i = 0; i < VA; ++i) scanf("%lf %lf %lf", &meshA.vertices[i].x, &meshA.vertices[i].y, &meshA.vertices[i].z);
    meshA.faces.resize(FA); for (int i = 0; i < FA; ++i) scanf("%d %d %d", &meshA.faces[i].v[0], &meshA.faces[i].v[1], &meshA.faces[i].v[2]);
    meshB.vertices.resize(VB); for (int i = 0; i < VB; ++i) scanf("%lf %lf %lf", &meshB.vertices[i].x, &meshB.vertices[i].y, &meshB.vertices[i].z);
    meshB.faces.resize(FB); for (int i = 0; i < FB; ++i) scanf("%d %d %d", &meshB.faces[i].v[0], &meshB.faces[i].v[1], &meshB.faces[i].v[2]);

    BVH bvhA(meshA), bvhB(meshB);
    unordered_map<int, vector<int>> interA, interB;
    for (int i = 0; i < FA; ++i) {
        BBox box; for (int k = 0; k < 3; ++k) box.add(meshA.vertices[meshA.faces[i].v[k]]);
        vector<int> cands; bvhB.intersect(box, cands);
        for (int j : cands) if (triTriIntersect(&meshA.vertices[meshA.faces[i].v[0]], &meshB.vertices[meshB.faces[j].v[0]])) {
            interA[i].push_back(j); interB[j].push_back(i);
        }
    }

    vector<Vec3> out_v; vector<Tri> out_f;
    auto process = [&](const Mesh& m, const Mesh& other, const BVH& other_bvh, const unordered_map<int, vector<int>>& inter, bool isA) {
        for (int i = 0; i < (int)m.faces.size(); ++i) {
            vector<vector<Vec3>> polys = {{m.vertices[m.faces[i].v[0]], m.vertices[m.faces[i].v[1]], m.vertices[m.faces[i].v[2]]}};
            if (inter.count(i)) {
                for (int j : inter.at(i)) {
                    Plane pB(other.vertices[other.faces[j].v[0]], other.vertices[other.faces[j].v[1]], other.vertices[other.faces[j].v[2]]);
                    Plane pA(m.vertices[m.faces[i].v[0]], m.vertices[m.faces[i].v[1]], m.vertices[m.faces[i].v[2]]);
                    if (abs(abs(pA.n.dot(pB.n)) - 1.0) < 1e-7) {
                        for (int k = 0; k < 3; ++k) {
                            Vec3 e0 = other.vertices[other.faces[j].v[k]], e1 = other.vertices[other.faces[j].v[(k + 1) % 3]];
                            Vec3 en = (e1 - e0).cross(pB.n); if (en.lengthSq() > 1e-18) {
                                Plane ep(e0, en / en.length()); vector<vector<Vec3>> next; for (auto& poly : polys) splitPolygon(poly, ep, next); polys = next;
                            }
                        }
                    } else {
                        vector<vector<Vec3>> next; for (auto& poly : polys) splitPolygon(poly, pB, next); polys = next;
                    }
                }
            }
            for (auto& poly : polys) {
                Vec3 center = {0,0,0}; for (auto& v : poly) center = center + v; center = center / (double)poly.size();
                bool discard = false;
                BBox cb; cb.add(center); vector<int> cands; other_bvh.intersect(cb, cands);
                for (int j : cands) if (isPointInTri(center, other.vertices[other.faces[j].v[0]], other.vertices[other.faces[j].v[1]], other.vertices[other.faces[j].v[2]])) {
                    Plane pA(m.vertices[m.faces[i].v[0]], m.vertices[m.faces[i].v[1]], m.vertices[m.faces[i].v[2]]);
                    Plane pB(other.vertices[other.faces[j].v[0]], other.vertices[other.faces[j].v[1]], other.vertices[other.faces[j].v[2]]);
                    if (pA.n.dot(pB.n) > 0.5) { if (!isA) discard = true; } else { discard = true; }
                    goto next_poly;
                }
                if (getWinding(other_bvh, center) != 0) discard = true;
                next_poly:
                if (!discard) for (int k = 1; k < (int)poly.size() - 1; ++k) {
                    int b = out_v.size(); out_v.push_back(poly[0]); out_v.push_back(poly[k]); out_v.push_back(poly[k+1]);
                    out_f.push_back({b, b + 1, b + 2});
                }
            }
        }
    };

    process(meshA, meshB, bvhB, interA, true);
    process(meshB, meshA, bvhA, interB, false);

    unordered_map<Coord, int, CoordHasher> vmap; vector<Vec3> fv; vector<Tri> ff;
    auto getv = [&](const Vec3& v) {
        Coord c = {(long long)floor(v.x * 1e8 + 0.5), (long long)floor(v.y * 1e8 + 0.5), (long long)floor(v.z * 1e8 + 0.5)};
        if (vmap.count(c)) return vmap[c];
        int id = (int)fv.size(); fv.push_back(v); vmap[c] = id; return id;
    };
    for (auto& f : out_f) ff.push_back({getv(out_v[f.v[0]]), getv(out_v[f.v[1]]), getv(out_v[f.v[2]])});
    printf("%zu %zu\n", fv.size(), ff.size());
    for (auto& v : fv) printf("%.10f %.10f %.10f\n", v.x, v.y, v.z);
    for (auto& f : ff) printf("%d %d %d\n", f.v[0], f.v[1], f.v[2]);

    return 0;
}
