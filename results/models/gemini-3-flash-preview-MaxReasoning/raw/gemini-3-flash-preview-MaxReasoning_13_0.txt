#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

struct Edge {
    unsigned char c;
    int to;
    int next;
};

vector<int> head;
vector<Edge> edges;
vector<int> len_arr, link_arr, first_pos;
vector<int> res, curr_match;
vector<int> sorted_nodes;
int nodes_count = 1, last = 0;

void add_edge(int u, unsigned char c, int v) {
    int new_idx = (int)edges.size();
    edges.push_back({c, v, head[u]});
    head[u] = new_idx;
}

int get_next(int u, unsigned char c) {
    for (int i = head[u]; i != -1; i = edges[i].next) {
        if (edges[i].c == c) return edges[i].to;
    }
    return -1;
}

void set_next(int u, unsigned char c, int v) {
    for (int i = head[u]; i != -1; i = edges[i].next) {
        if (edges[i].c == c) {
            edges[i].to = v;
            return;
        }
    }
    add_edge(u, c, v);
}

void copy_all_next(int from, int to) {
    for (int i = head[from]; i != -1; i = edges[i].next) {
        add_edge(to, edges[i].c, edges[i].to);
    }
}

void sam_init(int max_len) {
    int max_nodes = 2 * max_len + 5;
    head.assign(max_nodes, -1);
    len_arr.assign(max_nodes, 0);
    link_arr.assign(max_nodes, -1);
    first_pos.assign(max_nodes, 0);
    res.assign(max_nodes, 0);
    curr_match.assign(max_nodes, 0);
    sorted_nodes.assign(max_nodes, 0);
    edges.clear();
    edges.reserve(3 * max_len + 5);
    nodes_count = 1;
    last = 0;
}

void sam_extend(unsigned char c) {
    int cur = nodes_count++;
    len_arr[cur] = len_arr[last] + 1;
    first_pos[cur] = len_arr[cur] - 1;
    head[cur] = -1;
    
    int p = last;
    while (p != -1 && get_next(p, c) == -1) {
        add_edge(p, c, cur);
        p = link_arr[p];
    }
    if (p == -1) {
        link_arr[cur] = 0;
    } else {
        int q = get_next(p, c);
        if (len_arr[p] + 1 == len_arr[q]) {
            link_arr[cur] = q;
        } else {
            int clone = nodes_count++;
            len_arr[clone] = len_arr[p] + 1;
            head[clone] = -1;
            copy_all_next(q, clone);
            link_arr[clone] = link_arr[q];
            first_pos[clone] = first_pos[q];
            while (p != -1 && get_next(p, c) == q) {
                set_next(p, c, clone);
                p = link_arr[p];
            }
            link_arr[q] = link_arr[cur] = clone;
        }
    }
    last = cur;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    string dummy;
    getline(cin, dummy);

    if (N <= 0) {
        cout << endl;
        return 0;
    }

    vector<string> S;
    S.reserve(N);
    for (int i = 0; i < N; ++i) {
        string temp;
        if (!getline(cin, temp)) break;
        if (!temp.empty() && temp.back() == '\r') temp.pop_back();
        S.push_back(move(temp));
    }

    if (S.empty()) {
        cout << endl;
        return 0;
    }

    int shortest_idx = 0;
    for (int i = 1; i < (int)S.size(); ++i) {
        if (S[i].length() < S[shortest_idx].length()) {
            shortest_idx = i;
        }
    }
    swap(S[0], S[shortest_idx]);

    sam_init((int)S[0].length());
    for (char c : S[0]) {
        sam_extend((unsigned char)c);
    }

    // Counting sort for topological order (by length)
    int max_l = 0;
    for (int i = 0; i < nodes_count; ++i) if (len_arr[i] > max_l) max_l = len_arr[i];
    vector<int> cnt(max_l + 1, 0);
    for (int i = 0; i < nodes_count; ++i) cnt[len_arr[i]]++;
    for (int i = 1; i <= max_l; ++i) cnt[i] += cnt[i - 1];
    for (int i = 0; i < nodes_count; ++i) sorted_nodes[--cnt[len_arr[i]]] = i;

    // Initialize res with S[0] lengths
    for (int i = 0; i < nodes_count; ++i) res[i] = len_arr[i];

    for (int i = 1; i < (int)S.size(); ++i) {
        int v = 0, l = 0;
        for (char c : S[i]) {
            unsigned char uc = (unsigned char)c;
            while (v != 0 && get_next(v, uc) == -1) {
                v = link_arr[v];
                l = len_arr[v];
            }
            int next_v = get_next(v, uc);
            if (next_v != -1) {
                v = next_v;
                l++;
            }
            if (l > curr_match[v]) curr_match[v] = l;
        }

        for (int j = nodes_count - 1; j >= 1; --j) {
            int u = sorted_nodes[j];
            if (curr_match[u] > 0) {
                int p = link_arr[u];
                if (p != -1) {
                    int prop_len = min(len_arr[p], curr_match[u]);
                    if (curr_match[p] < prop_len) curr_match[p] = prop_len;
                }
            }
            if (curr_match[u] < res[u]) res[u] = curr_match[u];
            curr_match[u] = 0;
        }
        res[0] = 0;
        curr_match[0] = 0;
    }

    int max_res = 0, best_node = -1;
    for (int i = 0; i < nodes_count; ++i) {
        if (res[i] > max_res) {
            max_res = res[i];
            best_node = i;
        }
    }

    if (max_res == 0) {
        cout << endl;
    } else {
        cout << S[0].substr(first_pos[best_node] - max_res + 1, max_res) << endl;
    }

    return 0;
}
