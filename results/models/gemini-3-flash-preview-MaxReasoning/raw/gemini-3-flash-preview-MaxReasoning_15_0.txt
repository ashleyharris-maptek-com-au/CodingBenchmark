use std::io::{self, BufRead};

#[derive(Clone, Copy, Debug)]
struct Point2D {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy, Debug)]
struct Point3D {
    x: f64,
    y: f64,
    z: f64,
}

#[derive(Clone, Copy, Debug)]
struct Quaternion {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

impl Quaternion {
    fn new(w: f64, x: f64, y: f64, z: f64) -> Self {
        Self { w, x, y, z }
    }

    fn from_axis_angle(axis: Point3D, angle: f64) -> Self {
        let half_angle = angle * 0.5;
        let s = half_angle.sin();
        let c = half_angle.cos();
        let mag = (axis.x * axis.x + axis.y * axis.y + axis.z * axis.z).sqrt();
        if mag < 1e-10 {
            return Self::new(1.0, 0.0, 0.0, 0.0);
        }
        Self::new(c, axis.x / mag * s, axis.y / mag * s, axis.z / mag * s)
    }

    fn rotate(&self, p: Point3D) -> Point3D {
        let q_p = Quaternion::new(0.0, p.x, p.y, p.z);
        let q_inv = Quaternion::new(self.w, -self.x, -self.y, -self.z);
        let res = self.mul(q_p).mul(q_inv);
        Point3D {
            x: res.x,
            y: res.y,
            z: res.z,
        }
    }

    fn mul(&self, other: Quaternion) -> Quaternion {
        Quaternion {
            w: self.w * other.w - self.x * other.x - self.y * other.y - self.z * other.z,
            x: self.w * other.x + self.x * other.w + self.y * other.z - self.z * other.y,
            y: self.w * other.y - self.x * other.z + self.y * other.w + self.z * other.x,
            z: self.w * other.z + self.x * other.y - self.y * other.x + self.z * other.w,
        }
    }
}

fn cross(a: Point3D, b: Point3D) -> Point3D {
    Point3D {
        x: a.y * b.z - a.z * b.y,
        y: a.z * b.x - a.x * b.z,
        z: a.x * b.y - a.y * b.x,
    }
}

fn dot(a: Point3D, b: Point3D) -> f64 {
    a.x * b.x + a.y * b.y + a.z * b.z
}

struct Tetrahedron {
    p: Point3D,
    q: Quaternion,
    scale: f64,
}

fn get_standard_vertices() -> [Point3D; 4] {
    let c = 1.0 / 2.0f64.sqrt();
    [
        Point3D { x: 1.0, y: 0.0, z: -c },
        Point3D { x: -1.0, y: 0.0, z: -c },
        Point3D { x: 0.0, y: 1.0, z: c },
        Point3D { x: 0.0, y: -1.0, z: c },
    ]
}

fn project_shadow(v: Point3D, s: Point3D) -> Point2D {
    // v_shadow = v - (v.z / s.z) * s
    let t = -v.z / s.z;
    Point2D {
        x: v.x + t * s.x,
        y: v.y + t * s.y,
    }
}

fn get_shadow_vertices(tetra: &Tetrahedron, s: Point3D) -> Vec<Point2D> {
    let std_v = get_standard_vertices();
    std_v
        .iter()
        .map(|&v| {
            let v_transformed = Point3D {
                x: tetra.p.x + tetra.scale * tetra.q.rotate(v).x,
                y: tetra.p.y + tetra.scale * tetra.q.rotate(v).y,
                z: tetra.p.z + tetra.scale * tetra.q.rotate(v).z,
            };
            project_shadow(v_transformed, s)
        })
        .collect()
}

fn get_convex_hull(points: Vec<Point2D>) -> Vec<Point2D> {
    if points.len() <= 2 {
        return points;
    }
    let mut pts = points;
    pts.sort_by(|a, b| a.x.partial_cmp(&b.x).unwrap().then(a.y.partial_cmp(&b.y).unwrap()));
    
    fn cross_2d(o: Point2D, a: Point2D, b: Point2D) -> f64 {
        (a.x - o.x) * (b.y - o.y) - (a.y - o.y) * (b.x - o.x)
    }

    let mut lower = Vec::new();
    for p in &pts {
        while lower.len() >= 2 && cross_2d(lower[lower.len() - 2], lower[lower.len() - 1], *p) <= 0.0 {
            lower.pop();
        }
        lower.push(*p);
    }

    let mut upper = Vec::new();
    for p in pts.iter().rev() {
        while upper.len() >= 2 && cross_2d(upper[upper.len() - 2], upper[upper.len() - 1], *p) <= 0.0 {
            upper.pop();
        }
        upper.push(*p);
    }

    lower.pop();
    upper.pop();
    lower.extend(upper);
    lower
}

fn is_point_in_polygon(p: Point2D, poly: &[Point2D]) -> bool {
    if poly.len() < 3 {
        return false;
    }
    let mut positive = false;
    let mut negative = false;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let cp = (b.x - a.x) * (p.y - a.y) - (b.y - a.y) * (p.x - a.x);
        if cp > 1e-11 {
            positive = true;
        } else if cp < -1e-11 {
            negative = true;
        }
    }
    !(positive && negative)
}

fn triangulate(polygon: Vec<Point2D>) -> Vec<[Point2D; 3]> {
    let mut tris = Vec::new();
    if polygon.len() < 3 {
        return tris;
    }
    
    let mut verts = polygon;
    // Ensure counter-clockwise
    let mut area = 0.0;
    for i in 0..verts.len() {
        area += verts[i].x * verts[(i + 1) % verts.len()].y - verts[(i + 1) % verts.len()].x * verts[i].y;
    }
    if area < 0.0 {
        verts.reverse();
    }

    let mut indices: Vec<usize> = (0..verts.len()).collect();
    let mut iters = 0;
    while indices.len() > 3 && iters < 1000000 {
        iters += 1;
        let mut ear_found = false;
        for i in 0..indices.len() {
            let prev = indices[(i + indices.len() - 1) % indices.len()];
            let curr = indices[i];
            let next = indices[(i + 1) % indices.len()];

            let a = verts[prev];
            let b = verts[curr];
            let c = verts[next];

            let cp = (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x);
            if cp > 1e-11 {
                let mut contains_other = false;
                for j in 0..indices.len() {
                    let idx = indices[j];
                    if idx == prev || idx == curr || idx == next {
                        continue;
                    }
                    if is_point_in_polygon(verts[idx], &[a, b, c]) {
                        contains_other = true;
                        break;
                    }
                }
                if !contains_other {
                    tris.push([a, b, c]);
                    indices.remove(i);
                    ear_found = true;
                    break;
                }
            }
        }
        if !ear_found {
            break;
        }
    }
    if indices.len() == 3 {
        tris.push([verts[indices[0]], verts[indices[1]], verts[indices[2]]]);
    }
    tris
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let nv_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => return,
    };
    let nv: usize = match nv_line.trim().parse() {
        Ok(n) => n,
        Err(_) => return,
    };

    let mut polygon = Vec::new();
    for _ in 0..nv {
        let line = lines.next().unwrap().unwrap();
        let pts: Vec<f64> = line
            .split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        polygon.push(Point2D { x: pts[0], y: pts[1] });
    }

    let sun_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => return,
    };
    let sun_pts: Vec<f64> = sun_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let sun = Point3D {
        x: sun_pts[0],
        y: sun_pts[1],
        z: if sun_pts[2].abs() < 1e-10 { if sun_pts[2] < 0.0 { -1e-10 } else { 1e-10 } } else { sun_pts[2] },
    };

    let triangles = triangulate(polygon);
    let mut tetrahedrons = Vec::new();

    let sun_mag = (sun.x * sun.x + sun.y * sun.y + sun.z * sun.z).sqrt();
    let w_vec = Point3D {
        x: -sun.x / sun_mag,
        y: -sun.y / sun_mag,
        z: -sun.z / sun_mag,
    };
    let k_vec = Point3D { x: 0.0, y: 0.0, z: 1.0 };
    
    let rotation = if (w_vec.z - 1.0).abs() < 1e-10 {
        Quaternion::new(1.0, 0.0, 0.0, 0.0)
    } else if (w_vec.z + 1.0).abs() < 1e-10 {
        Quaternion::new(0.0, 1.0, 0.0, 0.0)
    } else {
        let axis = cross(k_vec, w_vec);
        let angle = dot(k_vec, w_vec).acos();
        Quaternion::from_axis_angle(axis, angle)
    };

    let p_z_sign = if sun.z < 0.0 { 1.0 } else { -1.0 };

    let mut stack = triangles;
    while let Some(tri) = stack.pop() {
        // Area check to avoid infinite recursion
        let tri_area = ((tri[1].x - tri[0].x) * (tri[2].y - tri[0].y) - (tri[1].y - tri[0].y) * (tri[2].x - tri[0].x)).abs() * 0.5;
        if tri_area < 1e-10 {
            continue;
        }

        let mut best_tetra = None;
        let scale = 10.0;
        
        let min_x = tri[0].x.min(tri[1].x.min(tri[2].x));
        let max_x = tri[0].x.max(tri[1].x.max(tri[2].x));
        let min_y = tri[0].y.min(tri[1].y.min(tri[2].y));
        let max_y = tri[0].y.max(tri[1].y.max(tri[2].y));
        let q_center = Point2D {
            x: (min_x + max_x) * 0.5,
            y: (min_y + max_y) * 0.5,
        };

        let p_z = p_z_sign * 20.0 * (tetrahedrons.len() as f64 + 1.0);
        let p_x = q_center.x + p_z * sun.x / sun.z;
        let p_y = q_center.y + p_z * sun.y / sun.z;
        
        let t = Tetrahedron {
            p: Point3D { x: p_x, y: p_y, z: p_z },
            q: rotation,
            scale,
        };
        
        let shadow_poly = get_convex_hull(get_shadow_vertices(&t, sun));
        if is_point_in_polygon(tri[0], &shadow_poly) &&
           is_point_in_polygon(tri[1], &shadow_poly) &&
           is_point_in_polygon(tri[2], &shadow_poly) {
            best_tetra = Some(t);
        }

        if let Some(t) = best_tetra {
            tetrahedrons.push(t);
        } else {
            let d01 = (tri[0].x - tri[1].x).powi(2) + (tri[0].y - tri[1].y).powi(2);
            let d12 = (tri[1].x - tri[2].x).powi(2) + (tri[1].y - tri[2].y).powi(2);
            let d20 = (tri[2].x - tri[0].x).powi(2) + (tri[2].y - tri[0].y).powi(2);
            
            if d01 >= d12 && d01 >= d20 {
                let mid = Point2D { x: (tri[0].x + tri[1].x) * 0.5, y: (tri[0].y + tri[1].y) * 0.5 };
                stack.push([tri[2], tri[0], mid]);
                stack.push([tri[1], tri[2], mid]);
            } else if d12 >= d01 && d12 >= d20 {
                let mid = Point2D { x: (tri[1].x + tri[2].x) * 0.5, y: (tri[1].y + tri[2].y) * 0.5 };
                stack.push([tri[0], tri[1], mid]);
                stack.push([tri[2], tri[0], mid]);
            } else {
                let mid = Point2D { x: (tri[2].x + tri[0].x) * 0.5, y: (tri[2].y + tri[0].y) * 0.5 };
                stack.push([tri[1], tri[2], mid]);
                stack.push([tri[0], tri[1], mid]);
            }
        }
    }

    println!("{}", tetrahedrons.len());
    for t in tetrahedrons {
        println!(
            "{:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6}",
            t.p.x, t.p.y, t.p.z, t.q.w, t.q.x, t.q.y, t.q.z, t.scale
        );
    }
}
