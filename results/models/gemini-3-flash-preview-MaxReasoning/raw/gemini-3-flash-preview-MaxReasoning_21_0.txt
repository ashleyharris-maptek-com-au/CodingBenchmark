#include <iostream>
#include <vector>
#include <deque>
#include <queue>
#include <unordered_set>
#include <unordered_map>
#include <algorithm>
#include <functional>
#include <cmath>

using namespace std;

typedef vector<int> Pos;

struct PosHash {
    size_t operator()(const Pos& v) const {
        size_t seed = v.size();
        for (int x : v) {
            seed ^= hash<int>{}(x) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
        }
        return seed;
    }
};

int D;
int max_turns_val;
Pos bounds;
int snake_length;
deque<Pos> snake_body;
unordered_map<Pos, int, PosHash> body_counts;
vector<Pos> foods;
unordered_set<Pos, PosHash> food_set;
unordered_set<Pos, PosHash> obstacles;

int manhattan(const Pos& a, const Pos& b) {
    int d = 0;
    for (int i = 0; i < D; ++i) d += abs(a[i] - b[i]);
    return d;
}

struct Step {
    int axis;
    int dir;
};

struct State {
    Pos p;
    int g, h;
    int axis, dir;
    int parent;
};

vector<Step> find_path_to_food() {
    if (foods.empty()) return {};

    vector<pair<int, int>> food_indices;
    food_indices.reserve(foods.size());
    for (int i = 0; i < (int)foods.size(); ++i) {
        food_indices.push_back({manhattan(snake_body[0], foods[i]), i});
    }
    
    int num_to_try = min((int)food_indices.size(), 3);
    if (num_to_try < (int)food_indices.size()) {
        nth_element(food_indices.begin(), food_indices.begin() + num_to_try - 1, food_indices.end());
    }
    sort(food_indices.begin(), food_indices.begin() + num_to_try);

    for (int fi = 0; fi < num_to_try; ++fi) {
        const Pos& target = foods[food_indices[fi].second];
        
        priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
        vector<State> nodes;
        nodes.reserve(100000);
        unordered_map<Pos, int, PosHash> closed;

        nodes.push_back({snake_body[0], 0, manhattan(snake_body[0], target), -1, 0, -1});
        pq.push({nodes[0].g + nodes[0].h, 0});

        while (!pq.empty()) {
            pair<int, int> top = pq.top();
            pq.pop();
            int idx = top.second;

            if (closed.count(nodes[idx].p) && closed[nodes[idx].p] <= nodes[idx].g) continue;
            closed[nodes[idx].p] = nodes[idx].g;

            if (nodes[idx].p == target) {
                vector<Step> path;
                int cur = idx;
                while (nodes[cur].parent != -1) {
                    path.push_back({nodes[cur].axis, nodes[cur].dir});
                    cur = nodes[cur].parent;
                }
                reverse(path.begin(), path.end());
                return path;
            }

            if (nodes.size() > 50000) break;

            for (int a = 0; a < D; ++a) {
                for (int d : {-1, 1}) {
                    Pos next_p = nodes[idx].p;
                    next_p[a] += d;

                    if (next_p[a] < 0 || next_p[a] >= bounds[a]) continue;
                    if (obstacles.count(next_p)) continue;
                    
                    if (body_counts.count(next_p)) {
                        // For planning, avoid the whole body to be safe.
                        continue;
                    }

                    int next_g = nodes[idx].g + 1;
                    if (!closed.count(next_p) || closed[next_p] > next_g) {
                        int next_h = manhattan(next_p, target);
                        pq.push({next_g + next_h, (int)nodes.size()});
                        nodes.push_back({next_p, next_g, next_h, a, d, idx});
                    }
                }
            }
        }
    }
    return {};
}

Step get_survival_move() {
    int best_score = -1;
    Step best_step = {-1, 0};
    
    for (int a = 0; a < D; ++a) {
        for (int d : {-1, 1}) {
            Pos next_p = snake_body[0];
            next_p[a] += d;
            
            if (next_p[a] < 0 || next_p[a] >= bounds[a]) continue;
            if (obstacles.count(next_p)) continue;
            
            bool is_body = false;
            if (body_counts.count(next_p)) {
                if (next_p == snake_body.back() && body_counts[next_p] == 1) {
                    if (food_set.count(next_p)) is_body = true;
                } else {
                    is_body = true;
                }
            }
            if (is_body) continue;
            
            int score = 0;
            for (int na = 0; na < D; ++na) {
                for (int nd : {-1, 1}) {
                    Pos nnext_p = next_p;
                    nnext_p[na] += nd;
                    if (nnext_p[na] >= 0 && nnext_p[na] < bounds[na] && 
                        !obstacles.count(nnext_p) && !body_counts.count(nnext_p)) {
                        score++;
                    }
                }
            }
            if (score > best_score) {
                best_score = score;
                best_step = {a, d};
            }
        }
    }
    return best_step;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    if (!(cin >> D >> max_turns_val)) return 0;
    bounds.resize(D);
    for (int i = 0; i < D; ++i) cin >> bounds[i];
    
    cin >> snake_length;
    Pos start_head(D);
    for (int i = 0; i < D; ++i) cin >> start_head[i];
    
    for (int i = 0; i < snake_length; ++i) {
        snake_body.push_back(start_head);
        body_counts[start_head]++;
    }
    
    int F;
    if (!(cin >> F)) return 0;
    for (int i = 0; i < F; ++i) {
        Pos f(D);
        for (int j = 0; j < D; ++j) cin >> f[j];
        foods.push_back(f);
        food_set.insert(f);
    }
    
    int O;
    if (!(cin >> O)) return 0;
    for (int i = 0; i < O; ++i) {
        Pos o(D);
        for (int j = 0; j < D; ++j) cin >> o[j];
        obstacles.insert(o);
    }
    
    int turn = 0;
    vector<Step> current_path;
    
    while (turn < max_turns_val && !foods.empty()) {
        if (current_path.empty()) {
            current_path = find_path_to_food();
        }
        
        Step s;
        if (current_path.empty()) {
            s = get_survival_move();
            if (s.axis == -1) break;
        } else {
            s = current_path[0];
            current_path.erase(current_path.begin());
            
            Pos next_p = snake_body[0];
            next_p[s.axis] += s.dir;
            bool safe = true;
            if (next_p[s.axis] < 0 || next_p[s.axis] >= bounds[s.axis]) safe = false;
            else if (obstacles.count(next_p)) safe = false;
            else if (body_counts.count(next_p)) {
                if (next_p == snake_body.back() && body_counts[next_p] == 1) {
                    if (food_set.count(next_p)) safe = false;
                } else {
                    safe = false;
                }
            }
            
            if (!safe) {
                current_path.clear();
                s = get_survival_move();
                if (s.axis == -1) break;
            }
        }
        
        cout << s.axis << " " << s.dir << "\n";
        
        Pos new_head = snake_body[0];
        new_head[s.axis] += s.dir;
        
        bool eaten = false;
        if (food_set.count(new_head)) {
            eaten = true;
            food_set.erase(new_head);
            for (int i = 0; i < (int)foods.size(); ++i) {
                if (foods[i] == new_head) {
                    foods[i] = foods.back();
                    foods.pop_back();
                    break;
                }
            }
            current_path.clear();
        }
        
        snake_body.push_front(new_head);
        body_counts[new_head]++;
        
        if (!eaten) {
            Pos tail = snake_body.back();
            snake_body.pop_back();
            body_counts[tail]--;
            if (body_counts[tail] == 0) body_counts.erase(tail);
        }
        
        turn++;
    }
    
    return 0;
}
