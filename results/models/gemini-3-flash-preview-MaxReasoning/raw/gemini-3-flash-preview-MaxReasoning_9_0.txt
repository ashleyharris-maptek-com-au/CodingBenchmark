using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

namespace CuttingStock
{
    public class Program
    {
        /// <summary>
        /// Represents a cut request with its length and original 0-indexed position.
        /// </summary>
        struct Cut : IComparable<Cut>
        {
            public long Length;
            public int OriginalIndex;

            public int CompareTo(Cut other)
            {
                // Sort by length descending for First Fit Decreasing heuristic.
                if (this.Length != other.Length)
                    return other.Length.CompareTo(this.Length);
                // Maintain stability with respect to original index.
                return this.OriginalIndex.CompareTo(other.OriginalIndex);
            }
        }

        public static void Main(string[] args)
        {
            // Use fast custom reader for massive input
            FastReader reader = new FastReader(Console.OpenStandardInput());
            
            int n = reader.ReadInt();
            if (n < 0) return;
            if (n == 0)
            {
                Console.WriteLine(0);
                return;
            }

            long stockLength = reader.ReadLong();
            Cut[] cuts = new Cut[n];
            for (int i = 0; i < n; i++)
            {
                cuts[i] = new Cut { Length = reader.ReadLong(), OriginalIndex = i };
            }

            // Step 1: Sort cuts in descending order of length (FFD heuristic)
            Array.Sort(cuts);

            // Step 2: Initialize Segment Tree to find the first bin with enough capacity in O(log N)
            int treeSize = 1;
            while (treeSize < n) treeSize *= 2;
            
            // tree[i] stores the maximum remaining capacity in the range of bins covered by node i.
            // Note: For N up to 10^7, this uses ~268MB. For N = 10^8, it might hit .NET array limits.
            long[] tree = new long[2 * treeSize];
            for (int i = 0; i < treeSize; i++) tree[treeSize + i] = stockLength;
            for (int i = treeSize - 1; i > 0; i--)
            {
                tree[i] = stockLength;
            }

            int[] cutToBin = new int[n];
            int numBins = 0;

            // Step 3: Assign cuts to bins using FFD with Segment Tree optimization
            for (int i = 0; i < n; )
            {
                int j = i;
                while (j < n && cuts[j].Length == cuts[i].Length) j++;

                long len = cuts[i].Length;
                for (int k = i; k < j; k++)
                {
                    int binIdx = FindFirst(tree, treeSize, len);
                    if (binIdx == -1) 
                    {
                        // According to constraints, all cuts fit in a stock piece.
                        k++;
                        continue;
                    }

                    // Optimization: If multiple items of the same length fit in the same bin,
                    // process them together to reduce tree updates.
                    long rem = tree[treeSize + binIdx];
                    int canFit = (int)(rem / len);
                    int toPut = Math.Min(canFit, j - k);
                    if (toPut < 1) toPut = 1;

                    for (int m = 0; m < toPut; m++)
                    {
                        cutToBin[k + m] = binIdx;
                    }
                    if (binIdx + 1 > numBins) numBins = binIdx + 1;

                    // Update tree: decrease capacity and propagate max up the tree
                    int node = treeSize + binIdx;
                    tree[node] -= (long)toPut * len;
                    while (node > 1)
                    {
                        node /= 2;
                        long left = tree[2 * node];
                        long right = tree[2 * node + 1];
                        long max = left > right ? left : right;
                        if (tree[node] == max) break;
                        tree[node] = max;
                    }

                    k += toPut - 1;
                }
                i = j;
            }

            // Step 4: Group indices by bin for the required output format
            int[] binCounts = new int[numBins];
            for (int i = 0; i < n; i++) binCounts[cutToBin[i]]++;

            int[] binStarts = new int[numBins + 1];
            for (int i = 0; i < numBins; i++) binStarts[i + 1] = binStarts[i] + binCounts[i];

            int[] sortedOriginalIndices = new int[n];
            int[] tempPtrs = new int[numBins];
            for (int i = 0; i < n; i++)
            {
                int b = cutToBin[i];
                sortedOriginalIndices[binStarts[b] + tempPtrs[b]++] = cuts[i].OriginalIndex;
            }

            // Step 5: Fast output
            using (var writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20))
            {
                writer.WriteLine(numBins);
                for (int b = 0; b < numBins; b++)
                {
                    int start = binStarts[b];
                    int end = binStarts[b + 1];
                    for (int idx = start; idx < end; idx++)
                    {
                        WriteInt(writer, sortedOriginalIndices[idx]);
                        if (idx < end - 1) writer.Write(' ');
                    }
                    writer.WriteLine();
                }
            }
        }

        /// <summary>
        /// Finds the first leaf index in the segment tree that has a value >= len.
        /// </summary>
        static int FindFirst(long[] tree, int treeSize, long len)
        {
            if (tree[1] < len) return -1;
            int node = 1;
            while (node < treeSize)
            {
                if (tree[2 * node] >= len) node = 2 * node;
                else node = 2 * node + 1;
            }
            return node - treeSize;
        }

        // Buffer for WriteInt to avoid repeated allocations and speed up conversion
        static char[] _intBuf = new char[16];
        static void WriteInt(StreamWriter sw, int n)
        {
            if (n == 0) { sw.Write('0'); return; }
            int pos = 15;
            while (n > 0)
            {
                _intBuf[--pos] = (char)('0' + (n % 10));
                n /= 10;
            }
            sw.Write(_intBuf, pos, 15 - pos);
        }
    }

    /// <summary>
    /// Fast custom reader to handle millions of integers from a stream efficiently.
    /// </summary>
    class FastReader
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _pos = 0;
        private int _len = 0;

        public FastReader(Stream stream) { _stream = stream; }

        private int ReadByte()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _pos = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_pos++];
        }

        public int ReadInt()
        {
            int c = ReadByte();
            while (c >= 0 && c <= 32) c = ReadByte();
            if (c == -1) return -1;
            int res = 0;
            while (c >= '0' && c <= '9')
            {
                res = res * 10 + (c - '0');
                c = ReadByte();
            }
            return res;
        }

        public long ReadLong()
        {
            int c = ReadByte();
            while (c >= 0 && c <= 32) c = ReadByte();
            if (c == -1) return -1;
            long res = 0;
            while (c >= '0' && c <= '9')
            {
                res = res * 10 + (c - '0');
                c = ReadByte();
            }
            return res;
        }
    }
}
