#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <stdint.h>
#include <execution>
#include <cstdio>

using namespace std;

/**
 * TSP Solver - Adaptive Strategy
 * ------------------------------
 * This solver uses different strategies based on the number of cities (N):
 * 1. Trivial/Small (N <= 2000): Nearest Neighbor followed by full 2-opt refinement.
 * 2. Medium/Large (N <= 1,000,000): Morton (Z-order) curve sorting followed by 
 *    windowed 2-opt refinement (window size = 200).
 * 3. Extreme/Epic (N > 1,000,000): Fast Morton curve sorting (O(N log N)).
 *
 * Parallel execution is used for sorting to maximize performance on multi-core systems.
 * Memory management is optimized for large-scale problems (up to 1,000,000,000 cities).
 */

struct Point {
    float x, y;
};

// Fast bit interleaving for Morton code (Z-order curve)
// Spreads 32 bits into 64 bits with gaps to interleave x and y
inline uint64_t split(uint32_t x) {
    uint64_t x64 = x;
    x64 = (x64 | (x64 << 16)) & 0x0000FFFF0000FFFFULL;
    x64 = (x64 | (x64 << 8))  & 0x00FF00FF00FF00FFULL;
    x64 = (x64 | (x64 << 4))  & 0x0F0F0F0F0F0F0F0FULL;
    x64 = (x64 | (x64 << 2))  & 0x3333333333333333ULL;
    x64 = (x64 | (x64 << 1))  & 0x5555555555555555ULL;
    return x64;
}

inline uint64_t morton(uint32_t x, uint32_t y) {
    return (split(x) << 1) | split(y);
}

inline float distSq(const Point& a, const Point& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return dx * dx + dy * dy;
}

int main() {
    int n;
    // Fast I/O using scanf/printf
    if (scanf("%d", &n) != 1) return 0;
    if (n <= 0) return 0;
    if (n == 1) {
        printf("0\n");
        return 0;
    }

    // Increase buffer sizes for large-scale data
    setvbuf(stdin, NULL, _IOFBF, 1 << 20);
    setvbuf(stdout, NULL, _IOFBF, 1 << 20);

    vector<Point> pts(n);
    float minX = 1e30f, maxX = -1e30f, minY = 1e30f, maxY = -1e30f;
    for (int i = 0; i < n; ++i) {
        if (scanf("%f %f", &pts[i].x, &pts[i].y) != 2) break;
        if (pts[i].x < minX) minX = pts[i].x;
        if (pts[i].x > maxX) maxX = pts[i].x;
        if (pts[i].y < minY) minY = pts[i].y;
        if (pts[i].y > maxY) maxY = pts[i].y;
    }

    vector<int> tour(n);
    for (int i = 0; i < n; ++i) tour[i] = i;

    if (n <= 2000) {
        // Nearest Neighbor heuristic for quality on small scales
        vector<bool> visited(n, false);
        vector<int> nnTour;
        nnTour.reserve(n);
        int curr = 0;
        nnTour.push_back(0);
        visited[0] = true;
        for (int i = 1; i < n; ++i) {
            int best = -1;
            float bestD = 1e38f;
            for (int j = 0; j < n; ++j) {
                if (!visited[j]) {
                    float d = distSq(pts[curr], pts[j]);
                    if (d < bestD) {
                        bestD = d;
                        best = j;
                    }
                }
            }
            curr = best;
            nnTour.push_back(curr);
            visited[curr] = true;
        }
        tour = nnTour;

        // Full 2-opt refinement
        bool improved = true;
        while (improved) {
            improved = false;
            for (int i = 0; i < n - 2; ++i) {
                for (int j = i + 2; j < n; ++j) {
                    int nextI = i + 1;
                    int nextJ = (j + 1) % n;
                    if (nextJ == i) continue;
                    float d1 = distSq(pts[tour[i]], pts[tour[nextI]]) + distSq(pts[tour[j]], pts[tour[nextJ]]);
                    float d2 = distSq(pts[tour[i]], pts[tour[j]]) + distSq(pts[tour[nextI]], pts[tour[nextJ]]);
                    if (d2 < d1) {
                        reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                        improved = true;
                    }
                }
            }
        }
    } else {
        // Morton Sorting for high scalability
        float rangeX = maxX - minX;
        float rangeY = maxY - minY;
        if (rangeX <= 0) rangeX = 1.0f;
        if (rangeY <= 0) rangeY = 1.0f;

        vector<uint64_t> codes(n);
        for (int i = 0; i < n; ++i) {
            double dx = (double)pts[i].x - minX;
            double dy = (double)pts[i].y - minY;
            uint32_t ux = (uint32_t)(dx / rangeX * 4294967295.0);
            uint32_t uy = (uint32_t)(dy / rangeY * 4294967295.0);
            codes[i] = morton(ux, uy);
        }

        // Parallel sort for fast linearization (C++17/20 standard)
        sort(execution::par_unseq, tour.begin(), tour.end(), [&](int a, int b) {
            return codes[a] < codes[b];
        });

        // Free memory for very large datasets
        if (n > 1000000) {
            codes.clear(); codes.shrink_to_fit();
        }

        // Ensure the tour starts at city 0
        int s = 0;
        for (int i = 0; i < n; ++i) if (tour[i] == 0) { s = i; break; }
        if (s != 0) rotate(tour.begin(), tour.begin() + s, tour.end());

        // Local 2-opt refinement with a sliding window for medium-large N
        if (n <= 1000000) {
            int window = (n <= 10000) ? n : 200;
            for (int iter = 0; iter < 3; ++iter) {
                bool improved = false;
                for (int i = 0; i < n - 3; ++i) {
                    int limit = min(n - 2, i + window);
                    for (int j = i + 2; j <= limit; ++j) {
                        int nextI = i + 1;
                        int nextJ = j + 1;
                        float d1 = distSq(pts[tour[i]], pts[tour[nextI]]) + distSq(pts[tour[j]], pts[tour[nextJ]]);
                        float d2 = distSq(pts[tour[i]], pts[tour[j]]) + distSq(pts[tour[nextI]], pts[tour[nextJ]]);
                        if (d2 < d1) {
                            reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                            improved = true;
                        }
                    }
                }
                if (!improved) break;
            }
        }
    }

    // Final memory cleanup before output
    pts.clear(); pts.shrink_to_fit();

    // Fast output of the resulting visit order
    for (int i = 0; i < n; ++i) {
        printf("%d%c", tour[i], (i == n - 1 ? '\n' : ' '));
    }

    return 0;
}
