#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstdint>
#include <cstdio>
#include <numeric>

using namespace std;

// Fast I/O for reading input efficiently
struct FastIO {
    static const int BUF_SIZE = 1 << 20;
    char buf[BUF_SIZE];
    int pos = 0, len = 0;

    int getChar() {
        if (pos == len) {
            pos = 0;
            len = (int)fread(buf, 1, BUF_SIZE, stdin);
            if (!len) return EOF;
        }
        return buf[pos++];
    }

    bool readInt(int &x) {
        x = 0;
        int c = getChar();
        while (c != EOF && (c < '0' || c > '9')) c = getChar();
        if (c == EOF) return false;
        while (c >= '0' && c <= '9') {
            x = x * 10 + (c - '0');
            c = getChar();
        }
        return true;
    }

    bool readFloat(float &x) {
        double val = 0;
        int c = getChar();
        while (c != EOF && (c < '0' || c > '9') && c != '-' && c != '.') c = getChar();
        if (c == EOF) return false;
        bool neg = false;
        if (c == '-') {
            neg = true;
            c = getChar();
        }
        while (c >= '0' && c <= '9') {
            val = val * 10 + (c - '0');
            c = getChar();
        }
        if (c == '.') {
            double div = 10;
            c = getChar();
            while (c >= '0' && c <= '9') {
                val += (c - '0') / div;
                div *= 10;
                c = getChar();
            }
        }
        if (neg) val = -val;
        x = (float)val;
        return true;
    }
} reader;

// Fast Output for writing results efficiently
struct FastOutput {
    static const int BUF_SIZE = 1 << 20;
    char buf[BUF_SIZE];
    int pos = 0;

    void writeInt(int x) {
        if (x == 0) {
            writeChar('0');
            return;
        }
        char s[12];
        int i = 0;
        while (x) {
            s[i++] = x % 10 + '0';
            x /= 10;
        }
        while (i--) writeChar(s[i]);
    }

    void writeChar(char c) {
        if (pos == BUF_SIZE) {
            fwrite(buf, 1, pos, stdout);
            pos = 0;
        }
        buf[pos++] = c;
    }

    void flush() {
        if (pos) {
            fwrite(buf, 1, pos, stdout);
            pos = 0;
        }
    }
    ~FastOutput() {
        flush();
    }
} writer;

// Hilbert curve mapping for 2D points
// Maps (x, y) coordinates to a 1D index on the Hilbert curve.
// This preserves spatial locality, making it an excellent heuristic for TSP.
inline uint64_t hilbert(uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (int s = 31; s >= 0; s--) {
        uint32_t rx = (x >> s) & 1;
        uint32_t ry = (y >> s) & 1;
        d += (uint64_t)((3 * rx) ^ ry) << (2ULL * s);
        if (ry == 0) {
            if (rx == 1) {
                x = ((1U << s) - 1) - (x ^ (1U << s));
                y = ((1U << s) - 1) - y;
            }
            uint32_t t = x;
            x = y;
            y = t;
        }
    }
    return d;
}

struct Point {
    float x, y;
};

inline double distSq(const Point& a, const Point& b) {
    return (double)(a.x - b.x) * (a.x - b.x) + (double)(a.y - b.y) * (a.y - b.y);
}

// Nearest Neighbor + 2-opt for small problem sizes
void solveSmall(int N, const vector<Point>& points) {
    vector<int> tour;
    tour.reserve(N);
    vector<bool> used(N, false);
    
    tour.push_back(0);
    used[0] = true;
    
    for (int i = 1; i < N; ++i) {
        int last = tour.back();
        int best = -1;
        double bestD = 1e30;
        for (int j = 0; j < N; ++j) {
            if (!used[j]) {
                double d = distSq(points[last], points[j]);
                if (d < bestD) {
                    bestD = d;
                    best = j;
                }
            }
        }
        tour.push_back(best);
        used[best] = true;
    }

    bool improved = true;
    int maxPasses = 1000;
    while (improved && maxPasses--) {
        improved = false;
        for (int i = 0; i < N; ++i) {
            for (int j = i + 2; j < N; ++j) {
                if (i == 0 && j == N - 1) continue;
                
                int next_i = (i + 1) % N;
                int next_j = (j + 1) % N;
                
                double d_curr = distSq(points[tour[i]], points[tour[next_i]]) + distSq(points[tour[j]], points[tour[next_j]]);
                double d_prop = distSq(points[tour[i]], points[tour[j]]) + distSq(points[tour[next_i]], points[tour[next_j]]);
                
                if (d_prop < d_curr - 1e-7) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }

    for (int i = 0; i < N; ++i) {
        writer.writeInt(tour[i]);
        writer.writeChar(i == N - 1 ? '\n' : ' ');
    }
}

struct HilbertPair {
    uint64_t h;
    int id;
    bool operator<(const HilbertPair& other) const {
        return h < other.h;
    }
};

int main() {
    int N;
    if (!reader.readInt(N)) return 0;
    if (N <= 0) return 0;
    if (N == 1) {
        writer.writeInt(0);
        writer.writeChar('\n');
        return 0;
    }

    vector<Point> points(N);
    double minX = 1e30, maxX = -1e30;
    double minY = 1e30, maxY = -1e30;

    for (int i = 0; i < N; ++i) {
        if (!reader.readFloat(points[i].x)) break;
        if (!reader.readFloat(points[i].y)) break;
        if (points[i].x < minX) minX = points[i].x;
        if (points[i].x > maxX) maxX = points[i].x;
        if (points[i].y < minY) minY = points[i].y;
        if (points[i].y > maxY) maxY = points[i].y;
    }

    if (N <= 2000) {
        solveSmall(N, points);
        return 0;
    }

    // Adaptive logic: for N > 2000, use Hilbert Curve heuristic
    uint64_t hRes = 1ULL << 31;
    double rangeX = maxX - minX;
    double rangeY = maxY - minY;
    if (rangeX == 0) rangeX = 1.0;
    if (rangeY == 0) rangeY = 1.0;

    vector<HilbertPair> h(N);
    for (int i = 0; i < N; ++i) {
        uint32_t x = (uint32_t)((double)(points[i].x - minX) / rangeX * (hRes - 1));
        uint32_t y = (uint32_t)((double)(points[i].y - minY) / rangeY * (hRes - 1));
        h[i] = {hilbert(x, y), i};
    }

    // Free memory if N is very large to stay within 32GB limit
    if (N > 1000000) {
        points.clear();
        points.shrink_to_fit();
    }

    sort(h.begin(), h.end());

    int startPos = -1;
    for (int i = 0; i < N; ++i) {
        if (h[i].second == 0) {
            startPos = i;
            break;
        }
    }
    if (startPos == -1) startPos = 0;

    for (int i = 0; i < N; ++i) {
        writer.writeInt(h[(startPos + i) % N].second);
        writer.writeChar(i == N - 1 ? '\n' : ' ');
    }

    return 0;
}
