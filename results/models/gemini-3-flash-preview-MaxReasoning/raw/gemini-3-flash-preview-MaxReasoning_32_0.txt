#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <set>
#include <bit>
#include <cstdint>

using namespace std;

/**
 * Problem: Steiner Tree (NP-Hard)
 * Algorithm: Dreyfus-Wagner DP
 * Complexity: O(3^k * n + 2^k * (n^2 + m))
 * Optimization: Bitset-based split step to skip INF states.
 */

typedef long long ll;
const ll INF = 1e16;

struct Edge {
    int to;
    int weight;
};

// Simple bitset for tracking active vertices per mask
struct CustomBitset {
    vector<uint64_t> bits;
    CustomBitset(int n = 0) {
        bits.resize((n + 63) / 64, 0);
    }
    void set(int i) {
        if (i >= 0 && i < (int)bits.size() * 64) {
            bits[i / 64] |= (1ULL << (i % 64));
        }
    }
};

void solve() {
    int n, m, k_in;
    if (!(cin >> n >> m >> k_in)) return;

    vector<vector<Edge>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        if (u >= 0 && u < n && v >= 0 && v < n) {
            adj[u].push_back({v, w});
            adj[v].push_back({u, w});
        }
    }

    vector<int> terminals_raw(k_in);
    for (int i = 0; i < k_in; ++i) {
        cin >> terminals_raw[i];
    }
    sort(terminals_raw.begin(), terminals_raw.end());
    terminals_raw.erase(unique(terminals_raw.begin(), terminals_raw.end()), terminals_raw.end());
    int k = (int)terminals_raw.size();

    // Base cases for small k
    if (k == 0) {
        cout << "0 0" << endl;
        return;
    }
    if (k == 1) {
        cout << "0 0" << endl;
        return;
    }

    // dp[mask][v] stores min weight to connect terminals in 'mask' with v in the tree.
    vector<vector<ll>> dp(1 << k, vector<ll>(n, INF));
    // split[mask][v] stores the submask T that gave the optimal split at v.
    vector<vector<int>> split(1 << k, vector<int>(n, 0));
    // pre[mask][v] stores the predecessor in the shortest path tree for the current mask.
    vector<vector<int>> pre(1 << k, vector<int>(n, -1));
    // active[mask] tracks which vertices v have dp[mask][v] < INF.
    vector<CustomBitset> active(1 << k, CustomBitset(n));

    for (int i = 0; i < k; ++i) {
        int t = terminals_raw[i];
        if (t >= 0 && t < n) {
            dp[1 << i][t] = 0;
            active[1 << i].set(t);
        }
    }

    for (int mask = 1; mask < (1 << k); ++mask) {
        // Step 1: Split Step
        // dp[mask][v] = min(dp[T][v] + dp[mask ^ T][v]) for all T subset of mask.
        if ((mask & (mask - 1)) != 0) { // Only split if more than 1 bit is set
            for (int T = (mask - 1) & mask; T > 0; T = (T - 1) & mask) {
                // Optimization: T and mask ^ T are symmetric, only check half.
                if (T < (mask ^ T)) continue;
                
                const auto& b1 = active[T].bits;
                const auto& b2 = active[mask ^ T].bits;
                for (int i = 0; i < (int)b1.size(); ++i) {
                    uint64_t common = b1[i] & b2[i];
                    while (common) {
                        int bit = std::countr_zero(common);
                        int v = i * 64 + bit;
                        ll val = dp[T][v] + dp[mask ^ T][v];
                        if (val < dp[mask][v]) {
                            dp[mask][v] = val;
                            split[mask][v] = T;
                            pre[mask][v] = -1;
                        }
                        common &= (common - 1);
                    }
                }
            }
        }

        // Step 2: Dijkstra Step
        // Relax edges: dp[mask][v] = min(dp[mask][v], dp[mask][u] + weight(u, v))
        priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
        for (int v = 0; v < n; ++v) {
            if (dp[mask][v] < INF) {
                pq.push({dp[mask][v], v});
            }
        }

        while (!pq.empty()) {
            ll d = pq.top().first;
            int u = pq.top().second;
            pq.pop();

            if (d > dp[mask][u]) continue;

            for (const auto& edge : adj[u]) {
                if (dp[mask][edge.to] > dp[mask][u] + edge.weight) {
                    dp[mask][edge.to] = dp[mask][u] + edge.weight;
                    pre[mask][edge.to] = u;
                    pq.push({dp[mask][edge.to], edge.to});
                }
            }
        }

        // Update active set for future split steps
        for (int v = 0; v < n; ++v) {
            if (dp[mask][v] < INF) active[mask].set(v);
        }
    }

    // Find the global optimum
    ll min_total_w = INF;
    int best_root = -1;
    for (int v = 0; v < n; ++v) {
        if (dp[(1 << k) - 1][v] < min_total_w) {
            min_total_w = dp[(1 << k) - 1][v];
            best_root = v;
        }
    }

    if (best_root == -1) return;

    // Reconstruct edges from DP choices
    set<pair<int, int>> initial_edges;
    vector<vector<bool>> vis(1 << k, vector<bool>(n, false));
    
    auto reconstruct_path = [&](auto self, int msk, int v) -> void {
        if (msk == 0 || vis[msk][v]) return;
        vis[msk][v] = true;
        if (pre[msk][v] != -1) {
            int u = pre[msk][v];
            int a = min(v, u), b = max(v, u);
            initial_edges.insert({a, b});
            self(self, msk, u);
        } else if (split[msk][v] != 0) {
            int T = split[msk][v];
            self(self, T, v);
            self(self, msk ^ T, v);
        }
    };
    reconstruct_path(reconstruct_path, (1 << k) - 1, best_root);

    // Final step: ensure the subgraph is a tree and handle zero-weight cycles
    // We build the MST of the selected edges.
    struct FinalEdge { int u, v; ll w; };
    vector<FinalEdge> candidate_edges;
    for (const auto& p : initial_edges) {
        int u = p.first, v = p.second;
        ll min_e_w = 2000000000000000LL; // Very large
        for (const auto& e : adj[u]) {
            if (e.to == v) min_e_w = min(min_e_w, (ll)e.weight);
        }
        candidate_edges.push_back({u, v, min_e_w});
    }
    sort(candidate_edges.begin(), candidate_edges.end(), [](const FinalEdge& a, const FinalEdge& b) {
        return a.w < b.w;
    });

    vector<int> dsu(n);
    for (int i = 0; i < n; ++i) dsu[i] = i;
    auto dsu_find = [&](auto self, int i) -> int {
        return (dsu[i] == i) ? i : (dsu[i] = self(self, dsu[i]));
    };

    vector<pair<int, int>> tree_edges;
    ll final_weight = 0;
    for (const auto& fe : candidate_edges) {
        int r1 = dsu_find(dsu_find, fe.u);
        int r2 = dsu_find(dsu_find, fe.v);
        if (r1 != r2) {
            dsu[r1] = r2;
            tree_edges.push_back({fe.u, fe.v});
            final_weight += fe.w;
        }
    }

    // Output result
    cout << final_weight << " " << tree_edges.size() << endl;
    for (const auto& e : tree_edges) {
        cout << e.first << " " << e.second << endl;
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
