using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading.Tasks;
using System.Collections.Concurrent;

public class MinesweeperSolver
{
    public static void Main(string[] args)
    {
        string line = Console.ReadLine();
        if (string.IsNullOrEmpty(line)) return;
        string[] parts = line.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        int H = int.Parse(parts[0]);
        int W = int.Parse(parts[1]);

        byte[] board = new byte[H * W];
        for (int r = 0; r < H; r++)
        {
            string row = Console.ReadLine();
            for (int c = 0; c < W; c++)
            {
                board[r * W + c] = (byte)row[c];
            }
        }

        var safeCells = Solve(H, W, board);

        Console.WriteLine(safeCells.Count);
        foreach (var p in safeCells)
        {
            Console.WriteLine($"{p % W} {p / W}");
        }
    }

    public static List<int> Solve(int H, int W, byte[] board)
    {
        List<int> globalSafe = new List<int>();
        HashSet<int> safeSet = new HashSet<int>();
        HashSet<int> mineSet = new HashSet<int>();

        bool changed = true;
        while (changed)
        {
            changed = false;
            for (int r = 0; r < H; r++)
            {
                for (int c = 0; c < W; c++)
                {
                    int idx = r * W + c;
                    byte val = board[idx];
                    if (val >= '0' && val <= '8')
                    {
                        int count = val - '0';
                        int unknownCount = 0;
                        int flags = 0;

                        for (int dr = -1; dr <= 1; dr++)
                        {
                            for (int dc = -1; dc <= 1; dc++)
                            {
                                if (dr == 0 && dc == 0) continue;
                                int nr = r + dr, nc = c + dc;
                                if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                                {
                                    int nidx = nr * W + nc;
                                    if (board[nidx] == ' ') unknownCount++;
                                    else if (board[nidx] == '*') flags++;
                                }
                            }
                        }

                        if (unknownCount > 0)
                        {
                            if (flags == count)
                            {
                                for (int dr = -1; dr <= 1; dr++)
                                {
                                    for (int dc = -1; dc <= 1; dc++)
                                    {
                                        if (dr == 0 && dc == 0) continue;
                                        int nr = r + dr, nc = c + dc;
                                        if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                                        {
                                            int nidx = nr * W + nc;
                                            if (board[nidx] == ' ' && safeSet.Add(nidx))
                                            {
                                                board[nidx] = (byte)'.'; // Temp marker
                                                changed = true;
                                            }
                                        }
                                    }
                                }
                            }
                            else if (flags + unknownCount == count)
                            {
                                for (int dr = -1; dr <= 1; dr++)
                                {
                                    for (int dc = -1; dc <= 1; dc++)
                                    {
                                        if (dr == 0 && dc == 0) continue;
                                        int nr = r + dr, nc = c + dc;
                                        if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                                        {
                                            int nidx = nr * W + nc;
                                            if (board[nidx] == ' ' && mineSet.Add(nidx))
                                            {
                                                board[nidx] = (byte)'*';
                                                changed = true;
                                            }
                                        }
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }

        foreach (var idx in safeSet) { globalSafe.Add(idx); board[idx] = (byte)'.'; }

        // Advanced Logic
        var frontierUnknown = new List<int>();
        int[] unknownToIndex = new int[H * W];
        Array.Fill(unknownToIndex, -1);

        for (int i = 0; i < H * W; i++)
        {
            if (board[i] == ' ')
            {
                int r = i / W, c = i % W;
                bool nearRevealed = false;
                for (int dr = -1; dr <= 1; dr++)
                {
                    for (int dc = -1; dc <= 1; dc++)
                    {
                        if (dr == 0 && dc == 0) continue;
                        int nr = r + dr, nc = c + dc;
                        if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                        {
                            byte val = board[nr * W + nc];
                            if (val >= '0' && val <= '8') { nearRevealed = true; break; }
                        }
                    }
                    if (nearRevealed) break;
                }
                if (nearRevealed)
                {
                    unknownToIndex[i] = frontierUnknown.Count;
                    frontierUnknown.Add(i);
                }
            }
        }

        if (frontierUnknown.Count > 0)
        {
            var constraints = new List<Constraint>();
            for (int i = 0; i < H * W; i++)
            {
                byte val = board[i];
                if (val >= '0' && val <= '8')
                {
                    int r = i / W, c = i % W;
                    var unknowns = new List<int>();
                    int flags = 0;
                    for (int dr = -1; dr <= 1; dr++)
                    {
                        for (int dc = -1; dc <= 1; dc++)
                        {
                            if (dr == 0 && dc == 0) continue;
                            int nr = r + dr, nc = c + dc;
                            if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                            {
                                int nidx = nr * W + nc;
                                if (board[nidx] == ' ') unknowns.Add(unknownToIndex[nidx]);
                                else if (board[nidx] == '*') flags++;
                            }
                        }
                    }
                    if (unknowns.Count > 0)
                        constraints.Add(new Constraint { Unknowns = unknowns.ToArray(), Target = (val - '0') - flags });
                }
            }

            // Group into islands using DSU
            int[] parent = new int[frontierUnknown.Count];
            for (int i = 0; i < parent.Length; i++) parent[i] = i;

            int Find(int i)
            {
                if (parent[i] == i) return i;
                return parent[i] = Find(parent[i]);
            }

            void Union(int i, int j)
            {
                int rootI = Find(i);
                int rootJ = Find(j);
                if (rootI != rootJ) parent[rootI] = rootJ;
            }

            foreach (var cons in constraints)
            {
                for (int i = 1; i < cons.Unknowns.Length; i++)
                {
                    Union(cons.Unknowns[0], cons.Unknowns[i]);
                }
            }

            var islandsMap = new Dictionary<int, List<int>>();
            for (int i = 0; i < frontierUnknown.Count; i++)
            {
                int root = Find(i);
                if (!islandsMap.TryGetValue(root, out var list))
                {
                    list = new List<int>();
                    islandsMap[root] = list;
                }
                list.Add(i);
            }

            var safeResult = new System.Collections.Concurrent.ConcurrentBag<int>();
            Parallel.ForEach(islandsMap.Values, islandUnknowns =>
            {
                var islandSet = new HashSet<int>(islandUnknowns);
                var islandConstraints = constraints.Where(c => c.Unknowns.Any(u => islandSet.Contains(u))).ToList();

                if (islandUnknowns.Count <= 60) // Increased threshold
                {
                    SolveIsland(islandUnknowns, islandConstraints, safeResult, frontierUnknown);
                }
            });

            foreach (var idx in safeResult) if (safeSet.Add(idx)) globalSafe.Add(idx);
        }

        return globalSafe;
    }

    public class Constraint
    {
        public int[] Unknowns;
        public int Target;
    }

    private static void SolveIsland(List<int> unknowns, List<Constraint> constraints, System.Collections.Concurrent.ConcurrentBag<int> globalSafe, List<int> unknownList)
    {
        int n = unknowns.Count;
        int[] unknownMap = unknowns.ToArray();
        int[] reverseMap = new int[unknownList.Count];
        for (int i = 0; i < n; i++) reverseMap[unknownMap[i]] = i;

        var localConstraints = constraints.Select(c => new Constraint
        {
            Unknowns = c.Unknowns.Select(u => reverseMap[u]).ToArray(),
            Target = c.Target
        }).ToList();

        // Variable ordering: Heuristic - pick variables that appear in most constraints first
        int[] varConstraintCount = new int[n];
        foreach (var c in localConstraints) foreach (var u in c.Unknowns) varConstraintCount[u]++;
        int[] order = Enumerable.Range(0, n).OrderByDescending(i => varConstraintCount[i]).ToArray();
        
        int[] invOrder = new int[n];
        for (int i = 0; i < n; i++) invOrder[order[i]] = i;

        var mappedConstraints = localConstraints.Select(c => new Constraint
        {
            Unknowns = c.Unknowns.Select(u => invOrder[u]).OrderBy(u => u).ToArray(),
            Target = c.Target
        }).ToList();

        // For each variable, which constraints it participates in
        var varToConstraints = new List<int>[n];
        for (int i = 0; i < n; i++) varToConstraints[i] = new List<int>();
        for (int i = 0; i < mappedConstraints.Count; i++)
        {
            foreach (var u in mappedConstraints[i].Unknowns) varToConstraints[u].Add(i);
        }

        long[] mineCounts = new long[n];
        long validSolutions = 0;
        bool[] assignment = new bool[n];
        int[] currentMines = new int[mappedConstraints.Count];
        int[] remainingUnknowns = mappedConstraints.Select(c => c.Unknowns.Length).ToArray();

        Backtrack(0, assignment, mappedConstraints, varToConstraints, currentMines, remainingUnknowns, mineCounts, ref validSolutions);

        if (validSolutions > 0)
        {
            for (int i = 0; i < n; i++)
            {
                if (mineCounts[i] == 0) globalSafe.Add(unknownList[unknownMap[order[i]]]);
            }
        }
    }

    private static void Backtrack(int idx, bool[] assignment, List<Constraint> constraints, List<int>[] varToConstraints, int[] currentMines, int[] remainingUnknowns, long[] mineCounts, ref long validSolutions)
    {
        if (idx == assignment.Length)
        {
            validSolutions++;
            for (int i = 0; i < assignment.Length; i++) if (assignment[i]) mineCounts[i]++;
            return;
        }

        // Try false
        assignment[idx] = false;
        if (CheckAndUpdate(idx, false, constraints, varToConstraints, currentMines, remainingUnknowns))
        {
            Backtrack(idx + 1, assignment, constraints, varToConstraints, currentMines, remainingUnknowns, mineCounts, ref validSolutions);
        }
        UndoUpdate(idx, false, varToConstraints, currentMines, remainingUnknowns);

        // Try true
        assignment[idx] = true;
        if (CheckAndUpdate(idx, true, constraints, varToConstraints, currentMines, remainingUnknowns))
        {
            Backtrack(idx + 1, assignment, constraints, varToConstraints, currentMines, remainingUnknowns, mineCounts, ref validSolutions);
        }
        UndoUpdate(idx, true, varToConstraints, currentMines, remainingUnknowns);
    }

    private static bool CheckAndUpdate(int idx, bool val, List<Constraint> constraints, List<int>[] varToConstraints, int[] currentMines, int[] remainingUnknowns)
    {
        bool possible = true;
        foreach (var cIdx in varToConstraints[idx])
        {
            if (val) currentMines[cIdx]++;
            remainingUnknowns[cIdx]--;
            
            if (currentMines[cIdx] > constraints[cIdx].Target || currentMines[cIdx] + remainingUnknowns[cIdx] < constraints[cIdx].Target)
            {
                possible = false;
            }
        }
        return possible;
    }

    private static void UndoUpdate(int idx, bool val, List<int>[] varToConstraints, int[] currentMines, int[] remainingUnknowns)
    {
        foreach (var cIdx in varToConstraints[idx])
        {
            if (val) currentMines[cIdx]--;
            remainingUnknowns[cIdx]++;
        }
    }
}
