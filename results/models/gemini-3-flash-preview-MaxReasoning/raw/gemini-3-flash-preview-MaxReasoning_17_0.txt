using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;

public class Program
{
    public class Box
    {
        public int Index;
        public int W, H, D;
        public long Volume;
        public int[][] Rotations;

        public Box(int index, int w, int h, int d)
        {
            Index = index;
            W = w;
            H = h;
            D = d;
            Volume = (long)w * h * d;
            
            // Pre-calculate unique rotations
            int[] dims = new int[] { w, h, d };
            Array.Sort(dims);
            int d0 = dims[0], d1 = dims[1], d2 = dims[2];

            if (d0 == d1 && d1 == d2)
            {
                Rotations = new int[][] { new int[] { d0, d0, d0 } };
            }
            else if (d0 == d1)
            {
                Rotations = new int[][] {
                    new int[] { d0, d0, d2 },
                    new int[] { d0, d2, d0 },
                    new int[] { d2, d0, d0 }
                };
            }
            else if (d1 == d2)
            {
                Rotations = new int[][] {
                    new int[] { d1, d1, d0 },
                    new int[] { d1, d0, d1 },
                    new int[] { d0, d1, d1 }
                };
            }
            else
            {
                Rotations = new int[][] {
                    new int[] { d0, d1, d2 },
                    new int[] { d0, d2, d1 },
                    new int[] { d1, d0, d2 },
                    new int[] { d1, d2, d0 },
                    new int[] { d2, d0, d1 },
                    new int[] { d2, d1, d0 }
                };
            }
        }
    }

    public struct FreeSpace
    {
        public int X, Y, Z;
        public int W, H, D;
        public long Volume;

        public FreeSpace(int x, int y, int z, int w, int h, int d)
        {
            X = x; Y = y; Z = z; W = w; H = h; D = d;
            Volume = (long)w * h * d;
        }
    }

    public struct PackedBox
    {
        public int Index;
        public int X, Y, Z;
        public int W, H, D;
    }

    public class GuillotinePacker
    {
        public int ContainerW, ContainerH, ContainerD;
        public List<FreeSpace> FreeSpaces = new List<FreeSpace>();
        public List<PackedBox> PackedResult = new List<PackedBox>();

        public GuillotinePacker(int w, int h, int d)
        {
            ContainerW = w;
            ContainerH = h;
            ContainerD = d;
            FreeSpaces.Add(new FreeSpace(0, 0, 0, w, h, d));
        }

        public void Pack(List<Box> boxes, Stopwatch sw)
        {
            foreach (var box in boxes)
            {
                // Safety check for time limit within one packing run
                // Though for 10k boxes it should be fast enough.
                if (TryPack(box, out var packed))
                {
                    PackedResult.Add(packed);
                }
            }
        }

        private bool TryPack(Box box, out PackedBox result)
        {
            result = default;
            int bestSpaceIndex = -1;
            int bw = 0, bh = 0, bd = 0;
            long minScore = long.MaxValue;

            for (int i = 0; i < FreeSpaces.Count; i++)
            {
                var fs = FreeSpaces[i];
                if (box.Volume > fs.Volume) continue;

                foreach (var r in box.Rotations)
                {
                    if (r[0] <= fs.W && r[1] <= fs.H && r[2] <= fs.D)
                    {
                        // Best Short Side Fit (BSSF)
                        int dw = fs.W - r[0];
                        int dh = fs.H - r[1];
                        int dd = fs.D - r[2];
                        
                        // Manual sort of diffs to avoid allocation/overhead
                        int s1 = dw, s2 = dh, s3 = dd;
                        if (s1 > s2) { int t = s1; s1 = s2; s2 = t; }
                        if (s2 > s3) { int t = s2; s2 = s3; s3 = t; }
                        if (s1 > s2) { int t = s1; s1 = s2; s2 = t; }
                        
                        // Score prioritizes smaller leftover dimensions
                        long score = ((long)s1 << 40) | ((long)s2 << 20) | (long)s3;

                        if (score < minScore)
                        {
                            minScore = score;
                            bestSpaceIndex = i;
                            bw = r[0]; bh = r[1]; bd = r[2];
                            if (score == 0) break;
                        }
                    }
                }
                if (minScore == 0) break;
            }

            if (bestSpaceIndex != -1)
            {
                var fs = FreeSpaces[bestSpaceIndex];
                result = new PackedBox { Index = box.Index, X = fs.X, Y = fs.Y, Z = fs.Z, W = bw, H = bh, D = bd };
                FreeSpaces.RemoveAt(bestSpaceIndex);
                Split(fs, bw, bh, bd);
                return true;
            }

            return false;
        }

        private void Split(FreeSpace fs, int bw, int bh, int bd)
        {
            int rw = fs.W - bw;
            int rh = fs.H - bh;
            int rd = fs.D - bd;

            // Split along axes in order of smallest leftover (Shorter Axis Split)
            int[] order = new int[] { 0, 1, 2 }; // 0:W, 1:H, 2:D
            int[] leftovers = new int[] { rw, rh, rd };
            
            if (leftovers[0] > leftovers[1]) { Swap(ref leftovers[0], ref leftovers[1]); Swap(ref order[0], ref order[1]); }
            if (leftovers[1] > leftovers[2]) { Swap(ref leftovers[1], ref leftovers[2]); Swap(ref order[1], ref order[2]); }
            if (leftovers[0] > leftovers[1]) { Swap(ref leftovers[0], ref leftovers[1]); Swap(ref order[0], ref order[1]); }
            
            int curW = fs.W, curH = fs.H, curD = fs.D;
            int curX = fs.X, curY = fs.Y, curZ = fs.Z;
            
            for (int i = 0; i < 3; i++)
            {
                int axis = order[i];
                if (axis == 0) // W
                {
                    if (rw > 0) FreeSpaces.Add(new FreeSpace(curX + bw, curY, curZ, rw, curH, curD));
                    curW = bw;
                }
                else if (axis == 1) // H
                {
                    if (rh > 0) FreeSpaces.Add(new FreeSpace(curX, curY + bh, curZ, curW, rh, curD));
                    curH = bh;
                }
                else // D
                {
                    if (rd > 0) FreeSpaces.Add(new FreeSpace(curX, curY, curZ + bd, curW, curH, rd));
                    curD = bd;
                }
            }
        }

        private static void Swap(ref int a, ref int b) { int t = a; a = b; b = t; }
    }

    static void Main(string[] args)
    {
        Stopwatch sw = Stopwatch.StartNew();
        
        string firstLine = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(firstLine)) return;
        string[] parts = firstLine.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 4) return;
        
        int N = int.Parse(parts[0]);
        int W = int.Parse(parts[1]);
        int H = int.Parse(parts[2]);
        int D = int.Parse(parts[3]);

        List<Box> originalBoxes = new List<Box>(N);
        for (int i = 0; i < N; i++)
        {
            string line = Console.ReadLine();
            if (line == null) break;
            string[] bp = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (bp.Length < 3) break;
            originalBoxes.Add(new Box(i, int.Parse(bp[0]), int.Parse(bp[1]), int.Parse(bp[2])));
        }

        List<PackedBox> bestPacked = null;
        int bestCount = -1;

        // Strategies for sorting boxes to maximize packed count
        // Since we want to maximize count, prioritizing smaller boxes is key.
        List<Action<List<Box>>> strategies = new List<Action<List<Box>>>();
        
        // 1. Volume Ascending
        strategies.Add(list => list.Sort((a, b) => a.Volume.CompareTo(b.Volume)));
        // 2. Shortest Dimension Ascending
        strategies.Add(list => list.Sort((a, b) => {
            int ma = Math.Min(a.W, Math.Min(a.H, a.D));
            int mb = Math.Min(b.W, Math.Min(b.H, b.D));
            int cmp = ma.CompareTo(mb);
            return cmp != 0 ? cmp : a.Volume.CompareTo(b.Volume);
        }));
        // 3. Max Dimension Ascending
        strategies.Add(list => list.Sort((a, b) => {
            int ma = Math.Max(a.W, Math.Max(a.H, a.D));
            int mb = Math.Max(b.W, Math.Max(b.H, b.D));
            int cmp = ma.CompareTo(mb);
            return cmp != 0 ? cmp : a.Volume.CompareTo(b.Volume);
        }));
        // 4. Surface Area Ascending
        strategies.Add(list => list.Sort((a, b) => {
            long sa = (long)a.W * a.H + (long)a.H * a.D + (long)a.D * a.W;
            long sb = (long)b.W * b.H + (long)b.H * b.D + (long)b.D * b.W;
            int cmp = sa.CompareTo(sb);
            return cmp != 0 ? cmp : a.Volume.CompareTo(b.Volume);
        }));
        // 5. Perimeter (W+H+D) Ascending
        strategies.Add(list => list.Sort((a, b) => {
            int pa = a.W + a.H + a.D;
            int pb = b.W + b.H + b.D;
            int cmp = pa.CompareTo(pb);
            return cmp != 0 ? cmp : a.Volume.CompareTo(b.Volume);
        }));

        foreach (var strategy in strategies)
        {
            if (sw.Elapsed.TotalSeconds > 26) break;

            List<Box> boxesToPack = new List<Box>(originalBoxes);
            strategy(boxesToPack);

            GuillotinePacker packer = new GuillotinePacker(W, H, D);
            packer.Pack(boxesToPack, sw);

            if (packer.PackedResult.Count > bestCount)
            {
                bestCount = packer.PackedResult.Count;
                bestPacked = packer.PackedResult;
            }

            if (bestCount == N) break;
        }
        
        // If we still have time, try descending versions or slightly different mixes
        if (bestCount < N && sw.Elapsed.TotalSeconds < 24)
        {
             List<Box> boxesToPack = new List<Box>(originalBoxes);
             // Volume Descending (standard for bin packing, might help here too)
             boxesToPack.Sort((a, b) => b.Volume.CompareTo(a.Volume));
             GuillotinePacker packer = new GuillotinePacker(W, H, D);
             packer.Pack(boxesToPack, sw);
             if (packer.PackedResult.Count > bestCount)
             {
                 bestCount = packer.PackedResult.Count;
                 bestPacked = packer.PackedResult;
             }
        }

        // Final Output
        if (bestPacked != null)
        {
            Console.WriteLine(bestCount);
            foreach (var pb in bestPacked)
            {
                Console.WriteLine($"{pb.Index} {pb.X} {pb.Y} {pb.Z}");
            }
        }
        else
        {
            Console.WriteLine(0);
        }
    }
}
