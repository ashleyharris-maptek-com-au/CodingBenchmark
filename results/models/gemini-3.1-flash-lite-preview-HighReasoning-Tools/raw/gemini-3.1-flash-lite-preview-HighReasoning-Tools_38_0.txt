import sys

def solve():
    # Read all input at once
    input_data = sys.stdin.read().splitlines()
    if not input_data:
        return

    # Parse U and S
    try:
        header_raw = input_data[0]
        header = "".join(c for c in header_raw if c.isdigit() or c.isspace()).split()
        if not header:
            return
        u, s = map(int, header)
    except ValueError:
        return

    # Parse sets
    sets = []
    for i in range(1, s + 1):
        if i >= len(input_data):
            break
        line = input_data[i]
        elements = list(map(int, line.split()))
        sets.append(set(elements))

    # element_to_sets[element] = [list of set indices containing element]
    element_to_sets = [[] for _ in range(u)]
    for i, st in enumerate(sets):
        for e in st:
            if 0 <= e < u:
                element_to_sets[e].append(i)

    solution = []
    used_sets = [False] * s
    covered_elements = [False] * u

    def backtrack(num_covered):
        if num_covered == u:
            return True
        
        # Pick the element with the fewest remaining options (heuristic)
        min_sets = float('inf')
        best_element = -1
        
        for e in range(u):
            if not covered_elements[e]:
                count = 0
                for s_idx in element_to_sets[e]:
                    if not used_sets[s_idx]:
                        count += 1
                if count < min_sets:
                    min_sets = count
                    best_element = e
        
        if best_element == -1 or min_sets == 0:
            return False
        
        for s_idx in element_to_sets[best_element]:
            if not used_sets[s_idx]:
                # Try to use set s_idx
                st = sets[s_idx]
                
                # Check compatibility
                compatible = True
                for e in st:
                    if covered_elements[e]:
                        compatible = False
                        break
                
                if compatible:
                    # Apply
                    used_sets[s_idx] = True
                    for e in st:
                        covered_elements[e] = True
                    solution.append(s_idx)
                    
                    if backtrack(num_covered + len(st)):
                        return True
                    
                    # Backtrack
                    solution.pop()
                    for e in st:
                        covered_elements[e] = False
                    used_sets[s_idx] = False
        return False

    if backtrack(0):
        print("SOLUTION")
        print(*(sorted(solution)))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    solve()
