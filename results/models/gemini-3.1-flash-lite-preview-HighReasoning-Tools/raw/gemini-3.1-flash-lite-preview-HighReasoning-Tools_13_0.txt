#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <deque>

// Memory-efficient Suffix Array construction (O(N log N))
struct SuffixArray {
    std::string s;
    int n;
    std::vector<int> sa, lcp, rank;

    SuffixArray(const std::string& str) : s(str), n(str.size()) {
        sa.resize(n);
        rank.resize(n);
        for (int i = 0; i < n; ++i) sa[i] = i, rank[i] = s[i];

        for (int k = 1; k < n; k <<= 1) {
            auto cmp = [&](int i, int j) {
                if (rank[i] != rank[j]) return rank[i] < rank[j];
                int ri = (i + k < n) ? rank[i + k] : -1;
                int rj = (j + k < n) ? rank[j + k] : -1;
                return ri < rj;
            };
            std::sort(sa.begin(), sa.end(), cmp);

            std::vector<int> next_rank(n);
            next_rank[sa[0]] = 0;
            for (int i = 1; i < n; ++i)
                next_rank[sa[i]] = next_rank[sa[i - 1]] + (cmp(sa[i - 1], sa[i]) ? 1 : 0);
            rank = next_rank;
            if (rank[sa[n - 1]] == n - 1) break;
        }
    }

    void build_lcp() {
        lcp.assign(n, 0);
        int h = 0;
        for (int i = 0; i < n; ++i) {
            if (rank[i] > 0) {
                int j = sa[rank[i] - 1];
                if (h > 0) h--;
                while (i + h < n && j + h < n && s[i + h] == s[j + h]) h++;
                lcp[rank[i]] = h;
            }
        }
    }
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n;
    if (!(std::cin >> n)) return 0;
    std::string combined;
    std::vector<int> string_id;
    
    char separator = 1; 
    for (int i = 0; i < n; ++i) {
        std::string s;
        std::cin >> s;
        for (char c : s) {
            combined += c;
            string_id.push_back(i);
        }
        combined += separator++;
        string_id.push_back(-1);
    }

    SuffixArray sa(combined);
    sa.build_lcp();

    // Sliding window logic
    int m = sa.n;
    std::vector<int> count(n, 0);
    int distinct_strings = 0;
    int left = 0;
    int max_len = 0;
    int best_start = -1;

    std::deque<int> dq; // For RMQ on LCP

    for (int right = 0; right < m; ++right) {
        int id = string_id[sa.sa[right]];
        if (id != -1) {
            if (count[id]++ == 0) distinct_strings++;
        }

        while (distinct_strings == n) {
            // Check LCP window
            if (right > left) {
                // Find minimum LCP in [left+1, right]
                // This is a naive approach, but sufficient for now.
                int min_lcp = 1e9;
                for (int i = left + 1; i <= right; ++i) min_lcp = std::min(min_lcp, sa.lcp[i]);
                
                if (min_lcp > max_len) {
                    max_len = min_lcp;
                    best_start = sa.sa[right];
                }
            }
            
            int left_id = string_id[sa.sa[left]];
            if (left_id != -1) {
                if (--count[left_id] == 0) distinct_strings--;
            }
            left++;
        }
    }

    if (max_len > 0) {
        std::cout << combined.substr(best_start, max_len) << std::endl;
    } else {
        std::cout << std::endl;
    }

    return 0;
}
