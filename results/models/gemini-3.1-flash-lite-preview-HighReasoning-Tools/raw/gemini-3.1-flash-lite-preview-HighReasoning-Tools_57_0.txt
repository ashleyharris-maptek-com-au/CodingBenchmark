const fs = require('fs');

function solve() {
  const input = fs.readFileSync(0, 'utf8');
  const data = JSON.parse(input);
  const { width, height, grid, ignitions, towns, cutBudget } = data;

  // 1. Map grid to a 2D array for easier access
  const map = grid.map(row => row.split(''));

  // 2. Find paths from ignitions to all cells
  // Use BFS to find the shortest distance from any ignition point to all other cells
  const dist = Array.from({ length: height }, () => Array(width).fill(Infinity));
  const queue = [];

  for (const [x, y] of ignitions) {
    if (map[y][x] === '.') {
      dist[y][x] = 0;
      queue.push([x, y]);
    }
  }

  let head = 0;
  while (head < queue.length) {
    const [x, y] = queue[head++];
    const d = dist[y][x];

    const dirs = [[0, 1], [0, -1], [1, 0], [-1, 0]];
    for (const [dx, dy] of dirs) {
      const nx = x + dx;
      const ny = y + dy;
      if (nx >= 0 && nx < width && ny >= 0 && ny < height && map[ny][nx] === '.' && dist[ny][nx] === Infinity) {
        dist[ny][nx] = d + 1;
        queue.push([nx, ny]);
      }
    }
  }

  // 3. Identify potential firebreak locations
  // Count how many high-value towns are affected by each burnable cell.
  // This is a simple heuristic: if a cell is on a path to a town, it's a candidate.
  const candidates = new Map();

  for (const town of towns) {
    const { x, y, value } = town;
    if (dist[y][x] === Infinity) continue; // Already blocked or unreachable

    // Backtrack from town to ignition to find the path
    let currX = x;
    let currY = y;
    while (dist[currY][currX] > 0) {
      let nextX = currX;
      let nextY = currY;
      let minD = dist[currY][currX];

      const dirs = [[0, 1], [0, -1], [1, 0], [-1, 0]];
      for (const [dx, dy] of dirs) {
        const nx = currX + dx;
        const ny = currY + dy;
        if (nx >= 0 && nx < width && ny >= 0 && ny < height && dist[ny][nx] < minD) {
          minD = dist[ny][nx];
          nextX = nx;
          nextY = ny;
        }
      }
      currX = nextX;
      currY = nextY;

      // Add as candidate if it's not the starting point
      if (dist[currY][currX] > 0) {
        const key = `${currX},${currY}`;
        candidates.set(key, (candidates.get(key) || 0) + value);
      }
    }
  }

  // 4. Select the best firebreaks
  const sortedCandidates = Array.from(candidates.entries())
    .sort((a, b) => b[1] - a[1]);

  const firebreaks = [];
  for (let i = 0; i < Math.min(cutBudget, sortedCandidates.length); i++) {
    const [coords] = sortedCandidates[i];
    firebreaks.push(coords.split(',').map(Number));
  }

  // 5. Output
  process.stdout.write(JSON.stringify({ firebreaks }));
}

solve();
