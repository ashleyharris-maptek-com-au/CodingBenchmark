struct InputData {
    data: array<u32, 16>, // 512 bits = 16 * 32
    target: array<u32, 8>, // 256 bits = 8 * 32
};

struct ResultData {
    found_nonce: u32,
    success: u32,
};

@group(0) @binding(0) var<storage, read> input: InputData;
@group(0) @binding(1) var<storage, read_write> result: ResultData;

fn rotr(x: u32, n: u32) -> u32 {
    return (x >> n) | (x << (32u - n));
}

fn sha256_compress(data: array<u32, 16>) -> array<u32, 8> {
    var h: array<u32, 8> = array<u32, 8>(
        0x6a09e667u, 0xbb67ae85u, 0x3c6ef372u, 0xa54ff53au, 0x510e527fu, 0x9b05688cu, 0x1f83d9abu, 0x5be0cd19u
    );
    
    var w: array<u32, 64>;
    
    // Message schedule
    for (var i: u32 = 0u; i < 16u; i++) {
        w[i] = data[i];
    }
    for (var i: u32 = 16u; i < 64u; i++) {
        let s0 = rotr(w[i - 15u], 7u) ^ rotr(w[i - 15u], 18u) ^ (w[i - 15u] >> 3u);
        let s1 = rotr(w[i - 2u], 17u) ^ rotr(w[i - 2u], 19u) ^ (w[i - 2u] >> 10u);
        w[i] = w[i - 16u] + s0 + w[i - 7u] + s1;
    }
    
    // Compression
    var a = h[0];
    var b = h[1];
    var c = h[2];
    var d = h[3];
    var e = h[4];
    var f = h[5];
    var g = h[6];
    var h_val = h[7];
    
    for (var i: u32 = 0u; i < 64u; i++) {
        let S1 = rotr(e, 6u) ^ rotr(e, 11u) ^ rotr(e, 25u);
        let ch = (e & f) ^ ((~e) & g);
        let temp1 = h_val + S1 + ch + K[i] + w[i];
        let S0 = rotr(a, 2u) ^ rotr(a, 13u) ^ rotr(a, 22u);
        let maj = (a & b) ^ (a & c) ^ (b & c);
        let temp2 = S0 + maj;
        
        h_val = g;
        g = f;
        f = e;
        e = d + temp1;
        d = c;
        c = b;
        b = a;
        a = temp1 + temp2;
    }
    
    h[0] += a;
    h[1] += b;
    h[2] += c;
    h[3] += d;
    h[4] += e;
    h[5] += f;
    h[6] += g;
    h[7] += h_val;
    
    return h;
}

@compute @workgroup_size(64)
fn main(@builtin(global_invocation_id) id: vec3<u32>) {
    if (result.success == 1u) {
        return;
    }
    
    var data = input.data;
    // Assume last 4 bytes of input.data are the nonce
    data[15] = id.x;
    
    let hash = sha256_compress(data);
    
    // Compare hash with target (big-endian check)
    var is_less = false;
    for (var i: u32 = 0u; i < 8u; i++) {
        if (hash[i] < input.target[i]) {
            is_less = true;
            break;
        } else if (hash[i] > input.target[i]) {
            is_less = false;
            break;
        }
    }
    
    if (is_less) {
        result.found_nonce = id.x;
        result.success = 1u;
    }
}
