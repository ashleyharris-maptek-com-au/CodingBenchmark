OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_uniform_float = OpTypePointer Uniform %float
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%float_0 = OpConstant %float 0.0
%float_0_5 = OpConstant %float 0.5
%float_0_3 = OpConstant %float 0.3
%float_1 = OpConstant %float 1.0
%vec3_sky = OpConstantComposite %vec3 %float_0_4 %float_0_6 %float_1_0
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_1_0 = OpConstant %float 1.0
%vec3_ground = OpConstantComposite %vec3 %float_0_3 %float_0_15 %float_0_05
%float_0_3_const = OpConstant %float 0.3
%float_0_15 = OpConstant %float 0.15
%float_0_05 = OpConstant %float 0.05
%glsl = OpExtInstImport "GLSL.std.450"

%ubo = OpVariable %ptr_uniform_UBO Uniform
%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output

%main = OpFunction %void None %func
%label = OpLabel
%n_raw = OpLoad %vec3 %normal
%n = OpExtInst %vec3 %glsl Normalize %n_raw
%pos_raw = OpLoad %vec3 %worldPos

%ptr_lightPos = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%lightPos_raw = OpLoad %vec4 %ptr_lightPos
%lightPos_vec3 = OpVectorShuffle %vec3 %lightPos_raw %lightPos_raw 0 1 2
%lightDir_raw = OpFSub %vec3 %lightPos_vec3 %pos_raw
%lightDir = OpExtInst %vec3 %glsl Normalize %lightDir_raw

%n_y = OpCompositeExtract %float %n 1
%blend = OpFAdd %float %float_0_5 %OpFMul %float %n_y %float_0_5
%ambient = OpExtInst %vec3 %glsl FMix %vec3_ground %vec3_sky %blend

%diffuse_dot = OpExtInst %float %glsl Dot %n %lightDir
%diffuse_clamped = OpExtInst %float %glsl FClamp %diffuse_dot %float_0 %float_1
%diffuse = OpFMul %float %float_0_3_const %diffuse_clamped
%diffuse_vec3 = OpCompositeConstruct %vec3 %diffuse %diffuse %diffuse
%finalColor_vec3 = OpFAdd %vec3 %ambient %diffuse_vec3
%finalColor_vec4 = OpVectorShuffle %vec4 %finalColor_vec3 %finalColor_vec3 0 1 2 4
%finalColor_vec4_fixed = OpCompositeInsert %vec4 %float_1 %finalColor_vec4 3

OpStore %outColor %finalColor_vec4_fixed
OpReturn
OpFunctionEnd
