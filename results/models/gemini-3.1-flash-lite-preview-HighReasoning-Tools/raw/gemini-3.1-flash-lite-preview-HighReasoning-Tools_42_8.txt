; SPIR-V Assembly
; Geometry Shader - Shrink Triangle towards Centroid

OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in Input
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_out Output
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_3 = OpConstant %float 3.0
%vec3_3 = OpConstantComposite %v3float %float_3 %float_3 %float_3
%float_0_7 = OpConstant %float 0.7

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_arr_pv_in = OpTypePointer Input %arr_pv_in
%ptr_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_v4 = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_pv_out = OpTypePointer Output %gl_PerVertex_out
%ptr_v4_out = OpTypePointer Output %v4float

%arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_arr_v3_in = OpTypePointer Input %arr_v3_in
%ptr_v3 = OpTypePointer Input %v3float
%ptr_v3_out = OpTypePointer Output %v3float

; Variables
%gl_in = OpVariable %ptr_arr_pv_in Input
%worldPos = OpVariable %ptr_arr_v3_in Input
%normal = OpVariable %ptr_arr_v3_in Input
%color = OpVariable %ptr_arr_v3_in Input
%gl_out = OpVariable %ptr_pv_out Output
%outColor = OpVariable %ptr_v3_out Output

%main_type = OpTypeFunction %void
%main = OpFunction %void None %main_type
%main_entry = OpLabel

; Read positions
%ptr_pv0 = OpAccessChain %ptr_pv %gl_in %int_0
%pv0 = OpLoad %gl_PerVertex_in %ptr_pv0
%pos0_v4 = OpCompositeExtract %v4float %pv0 0
%p0 = OpVectorShuffle %v3float %pos0_v4 %pos0_v4 0 1 2

%ptr_pv1 = OpAccessChain %ptr_pv %gl_in %int_1
%pv1 = OpLoad %gl_PerVertex_in %ptr_pv1
%pos1_v4 = OpCompositeExtract %v4float %pv1 0
%p1 = OpVectorShuffle %v3float %pos1_v4 %pos1_v4 0 1 2

%ptr_pv2 = OpAccessChain %ptr_pv %gl_in %int_2
%pv2 = OpLoad %gl_PerVertex_in %ptr_pv2
%pos2_v4 = OpCompositeExtract %v4float %pv2 0
%p2 = OpVectorShuffle %v3float %pos2_v4 %pos2_v4 0 1 2

; Compute centroid
%sum01 = OpFAdd %v3float %p0 %p1
%sum012 = OpFAdd %v3float %sum01 %p2
%centroid = OpFDiv %v3float %sum012 %vec3_3

; Vertex 0
%diff0 = OpFSub %v3float %p0 %centroid
%scaled0 = OpVectorTimesScalar %v3float %diff0 %float_0_7
%new_p0 = OpFAdd %v3float %centroid %scaled0
%w0 = OpCompositeExtract %float %pos0_v4 3
%new_p0_v4 = OpCompositeConstruct %v4float %new_p0 %w0
%ptr_out_pos = OpAccessChain %ptr_v4_out %gl_out %int_0
OpStore %ptr_out_pos %new_p0_v4
%ptr_color0 = OpAccessChain %ptr_v3 %color %int_0
%c0 = OpLoad %v3float %ptr_color0
OpStore %outColor %c0
OpEmitVertex

; Vertex 1
%diff1 = OpFSub %v3float %p1 %centroid
%scaled1 = OpVectorTimesScalar %v3float %diff1 %float_0_7
%new_p1 = OpFAdd %v3float %centroid %scaled1
%w1 = OpCompositeExtract %float %pos1_v4 3
%new_p1_v4 = OpCompositeConstruct %v4float %new_p1 %w1
OpStore %ptr_out_pos %new_p1_v4
%ptr_color1 = OpAccessChain %ptr_v3 %color %int_1
%c1 = OpLoad %v3float %ptr_color1
OpStore %outColor %c1
OpEmitVertex

; Vertex 2
%diff2 = OpFSub %v3float %p2 %centroid
%scaled2 = OpVectorTimesScalar %v3float %diff2 %float_0_7
%new_p2 = OpFAdd %v3float %centroid %scaled2
%w2 = OpCompositeExtract %float %pos2_v4 3
%new_p2_v4 = OpCompositeConstruct %v4float %new_p2 %w2
OpStore %ptr_out_pos %new_p2_v4
%ptr_color2 = OpAccessChain %ptr_v3 %color %int_2
%c2 = OpLoad %v3float %ptr_color2
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
