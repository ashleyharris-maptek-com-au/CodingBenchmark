; SPIR-V
; Version: 1.0
; Generator: 0
; Bound: 100
; Schema: 0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %out_outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

; Annotations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %out_outColor Location 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; Types
%void = OpTypeVoid
%void_func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_ubo = OpTypePointer Uniform %UBO
%ptr_vec3_input = OpTypePointer Input %vec3
%ptr_vec4_output = OpTypePointer Output %vec4
%ptr_vec4_uniform = OpTypePointer Uniform %vec4
%int = OpTypeInt 32 0
%int_3 = OpConstant %int 3

; Variables
%in_worldPos = OpVariable %ptr_vec3_input Input
%in_normal = OpVariable %ptr_vec3_input Input
%out_outColor = OpVariable %ptr_vec4_output Output
%ubo_var = OpVariable %ptr_ubo Uniform

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0

; Entry point
%main = OpFunction %void None %void_func
%label = OpLabel

; Load inputs
%worldPos = OpLoad %vec3 %in_worldPos
%normal = OpLoad %vec3 %in_normal

; Load uniform lightPos
%lightPos_ptr = OpAccessChain %ptr_vec4_uniform %ubo_var %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%lightPos_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2

; Light direction
%lightDir = OpFSub %vec3 %lightPos_xyz %worldPos
%L = OpExtInst %vec3 %1 Normalize %lightDir

; Normal direction
%N = OpExtInst %vec3 %1 Normalize %normal

; Lambertian diffuse
%dot_NL = OpDot %float %N %L
%intensity = OpExtInst %float %1 FMax %dot_NL %float_0

; Output color
%color = OpCompositeConstruct %vec4 %intensity %intensity %intensity %float_1
OpStore %out_outColor %color
OpReturn
OpFunctionEnd
