%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor

; Input variables
OpDecorate %worldPos Location 0

; Output variable

; UBO

OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 0
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4

; Constants
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_9 = OpConstant %float 0.9
%float_1_0 = OpConstant %float 1.0

; Variables
%worldPos = OpVariable %ptr_input_vec3 Input

%uv = OpVariable %ptr_input_vec2 Input
%tangent = OpVariable %ptr_input_vec3 Input
%color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_UBO Uniform

%fn_void = OpTypeFunction %void
%main = OpFunction %void None %fn_void
%label_entry = OpLabel
%pos = OpLoad %vec3 %worldPos
%norm = OpLoad %vec3 %normal
%uv_val = OpLoad %vec2 %uv
%ptr_light = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%lightPosVal = OpLoad %vec4 %ptr_light
%lightPosVec3 = OpVectorShuffle %vec3 %lightPosVal %lightPosVal 0 1 2

; Lambertian
%l_minus_w = OpFSub %vec3 %lightPosVec3 %pos
%n_norm = OpExtInst %vec3 %glsl Normalize %norm
%l_norm = OpExtInst %vec3 %glsl Normalize %l_minus_w
%dot = OpDot %float %n_norm %l_norm
%factor1 = OpFMul %float %float_0_5 %dot
%factor = OpFAdd %float %float_0_5 %factor1

; Checkerboard
%uv_x = OpCompositeExtract %float %uv_val 0
%uv_y = OpCompositeExtract %float %uv_val 1
%uvx8 = OpFMul %float %uv_x %float_8_0
%uvy8 = OpFMul %float %uv_y %float_8_0
%ix = OpExtInst %float %glsl Floor %uvx8
%iy = OpExtInst %float %glsl Floor %uvy8
%sum = OpFAdd %float %ix %iy
%mod = OpFMod %sum %float_2_0
%is_even = OpFOrdEqual %bool %mod %float_0
%checker = OpSelect %float %is_even %float_0_9 %float_0_2

; Final Color
%final_c = OpFMul %float %factor %checker
%final_v4 = OpCompositeConstruct %vec4 %final_c %final_c %final_c %float_1_0
OpStore %outColor %final_v4
OpReturn
OpFunctionEnd
