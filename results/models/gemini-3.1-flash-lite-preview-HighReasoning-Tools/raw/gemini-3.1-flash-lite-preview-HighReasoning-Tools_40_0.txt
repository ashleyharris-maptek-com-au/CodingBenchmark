#include <iostream>
#include <vector>
#include <numeric>
#include <algorithm>

using namespace std;

int n, m;
vector<long long> weights;
vector<vector<int>> adj;
long long max_weight = -1;
vector<int> best_assignment;

// Branch and Bound: try to take or not take a node
void backtrack(int node, long long current_weight, vector<int>& current_assignment, const vector<long long>& remaining_weights) {
    if (node == n) {
        if (current_weight > max_weight) {
            max_weight = current_weight;
            best_assignment = current_assignment;
        }
        return;
    }

    // Pruning: if current + remaining potential is not better than max_weight, stop.
    // Note: only add positive weights to potential.
    if (current_weight + remaining_weights[node] <= max_weight) {
        return;
    }

    // Option 1: Take node (if possible)
    bool can_take = true;
    for (int neighbor : adj[node]) {
        // Since we only care about already processed neighbors (u < node)
        // or future neighbors (which won't be in assignment yet),
        // we only need to check neighbors already in the set.
        if (neighbor < node && current_assignment[neighbor] == 1) {
            can_take = false;
            break;
        }
    }

    if (can_take) {
        current_assignment[node] = 1;
        backtrack(node + 1, current_weight + weights[node], current_assignment, remaining_weights);
        current_assignment[node] = 0; // backtrack
    }

    // Option 2: Don't take node
    backtrack(node + 1, current_weight, current_assignment, remaining_weights);
}

int main() {
    // Faster I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> n >> m)) return 0;
    
    weights.resize(n);
    for (int i = 0; i < n; ++i) cin >> weights[i];
    
    adj.resize(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u >= 0 && u < n && v >= 0 && v < n) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }

    // Precompute remaining weights for pruning
    vector<long long> remaining_weights(n + 1, 0);
    for (int i = n - 1; i >= 0; --i) {
        remaining_weights[i] = remaining_weights[i + 1] + (weights[i] > 0 ? weights[i] : 0);
    }

    vector<int> current_assignment(n, 0);
    best_assignment.resize(n, 0);
    
    // Initialize max_weight to 0, if all weights are negative, we pick nothing
    max_weight = 0;

    backtrack(0, 0, current_assignment, remaining_weights);

    cout << max_weight << endl;
    for (int i = 0; i < n; ++i) {
        cout << best_assignment[i] << (i == n - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
}
