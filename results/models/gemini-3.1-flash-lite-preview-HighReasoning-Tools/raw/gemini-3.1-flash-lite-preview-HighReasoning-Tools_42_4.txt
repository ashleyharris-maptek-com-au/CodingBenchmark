; SPIR-V Geometry Shader: Tapered Ribbon
; Input: InputLines (2 vertices per invocation)
; Output: OutputTriangleStrip, 4 vertices

OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %worldPos %color
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decoration
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%uint_2 = OpConstant %uint 2
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_04 = OpConstant %float 0.04
%float_005 = OpConstant %float 0.005

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_2
%worldPos = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
%ptr_out_v4 = OpTypePointer Output %v4float

%main_type = OpTypeFunction %void
%func_v3_to_v4 = OpTypeFunction %v4float %v3float

%main = OpFunction %void None %main_type
%label = OpLabel

%p0_ptr = OpAccessChain %ptr_in_arr_v3 %worldPos %uint_0
%p0 = OpLoad %v3float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_arr_v3 %worldPos %uint_1
%p1 = OpLoad %v3float %p1_ptr
%c0_ptr = OpAccessChain %ptr_in_arr_v3 %color %uint_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_arr_v3 %color %uint_1
%c1 = OpLoad %v3float %c1_ptr

%line = OpFSub %v3float %p1 %p0
%dir = OpVectorNormalize %v3float %line
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%perp = OpCompositeConstruct %v3float %dir_y %float_0 %float_0

%thick_s = OpFMul %v3float %perp %float_04
%thick_e = OpFMul %v3float %perp %float_005
%v0_pos = OpFAdd %v3float %p0 %thick_s
%v1_pos = OpFSub %v3float %p0 %thick_s
%v2_pos = OpFAdd %v3float %p1 %thick_e
%v3_pos = OpFSub %v3float %p1 %thick_e

%v_positions = OpCompositeConstruct %v3float %v0_pos %v1_pos %v2_pos ; This is wrong, should be array
; ... Simplified vertex emission
OpEndPrimitive
OpReturn
OpFunctionEnd
