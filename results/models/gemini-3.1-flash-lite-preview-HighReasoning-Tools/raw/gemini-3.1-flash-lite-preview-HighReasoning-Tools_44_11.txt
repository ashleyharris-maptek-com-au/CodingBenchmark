#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    vec2 uv_scaled = fract(uv * 6.0);
    vec2 cell_center = vec2(0.5, 0.5);
    float dist = length(uv_scaled - cell_center);
    
    // Smooth step for anti-aliasing
    float is_dot = smoothstep(0.3 + 0.01, 0.3 - 0.01, dist);
    
    vec3 dot_color = vec3(1.0, 0.8, 0.0);
    vec3 background_color = vec3(0.1, 0.1, 0.3);
    
    vec3 base_color = mix(background_color, dot_color, is_dot);
    
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    vec3 N = normalize(normal);
    
    float diffuse = max(dot(N, L), 0.0);
    float shading = 0.3 + 0.7 * diffuse;
    
    outColor = vec4(base_color * shading, 1.0);
}
