```spirv
OpCapability Shader
OpCapability Geometry
OpExtension "SPV_KHR_shader_draw_parameters"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorate
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_in = OpTypeArray %gl_PerVertex_in 6
%ptr_arr_gl_in = OpTypePointer Input %arr_gl_in
%arr_v3_6 = OpTypeArray %v3float 6
%ptr_arr_v3_6 = OpTypePointer Input %arr_v3_6
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_v3float_out = OpTypePointer Output %v3float

; Constants
%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1.0
%vec3_1 = OpConstantComposite %v3float %float_1 %float_1 %float_1

; Variables
%gl_in = OpVariable %ptr_arr_gl_in Input
%gl_out = OpVariable %ptr_gl_PerVertex_out Output
%worldPos = OpVariable %ptr_arr_v3_6 Input
%normal = OpVariable %ptr_arr_v3_6 Input
%color = OpVariable %ptr_arr_v3_6 Input
%outColor = OpVariable %ptr_v3float_out Output

; Ext inst
%std450 = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None None
%label = OpLabel

; Load positions (0, 2, 4)
%p0_ptr = OpAccessChain %ptr_v3float_in %worldPos %int_0
%p0 = OpLoad %v3float %p0_ptr
%p2_ptr = OpAccessChain %ptr_v3float_in %worldPos %int_2
%p2 = OpLoad %v3float %p2_ptr
%p4_ptr = OpAccessChain %ptr_v3float_in %worldPos %int_4
%p4 = OpLoad %v3float %p4_ptr

; Load colors (0, 2, 4)
%c0_ptr = OpAccessChain %ptr_v3float_in %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c2_ptr = OpAccessChain %ptr_v3float_in %color %int_2
%c2 = OpLoad %v3float %c2_ptr
%c4_ptr = OpAccessChain %ptr_v3float_in %color %int_4
%c4 = OpLoad %v3float %c4_ptr

; Face normal: N = normalize(cross(p2-p0, p4-p0))
%v1 = OpFSub %v3float %p2 %p0
%v2 = OpFSub %v3float %p4 %p0
%cross = OpExtInst %v3float %std450 Cross %v1 %v2
%N = OpExtInst %v3float %std450 Normalize %cross

; Diffuse: max(dot(N, L), 0.0)
%L = OpExtInst %v3float %std450 Normalize %vec3_1
%dot = OpDot %float %N %L
%diffuse = OpExtInst %float %std450 FMax %dot %float_0

; Base color: average(color[0], color[2], color[4])
%c_sum = OpFAdd %v3float %c0 %c2
%c_sum2 = OpFAdd %v3float %c_sum %c4
%base_color = OpFDiv %v3float %c_sum2 %constant_3 ; need to define constant_3

; Final color = base * (0.2 + 0.8 * diffuse)
%diff_factor = OpFMul %float %float_0_8 %diffuse
%factor = OpFAdd %float %float_0_2 %diff_factor
%final_color = OpVectorTimesScalar %v3float %base_color %factor

; Loop indices: 0, 2, 4
%indices = OpConstantComposite %type_arr_int_3 %int_0 %int_2 %int_4 ; need to define

; Emit Loop
OpStore %outColor %final_color
; Need to emit 3 vertices...

OpFunctionEnd
```
