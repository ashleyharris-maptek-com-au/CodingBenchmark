const fs = require('fs');

function solve() {
  const input = fs.readFileSync(0, 'utf8');
  if (!input) return;
  const data = JSON.parse(input);
  const { depot, truckCount, packages } = data;

  if (packages.length === 0) {
    console.log(JSON.stringify({ routes: [] }));
    return;
  }

  // Calculate angle for each package relative to the depot
  const packagesWithAngle = packages.map(pkg => ({
    ...pkg,
    angle: Math.atan2(pkg.y - depot[1], pkg.x - depot[0])
  }));

  // Sort by angle to enable clustering
  packagesWithAngle.sort((a, b) => a.angle - b.angle);

  // Divide packages into trucks
  const routes = [];
  const chunkSize = Math.ceil(packagesWithAngle.length / truckCount);

  for (let i = 0; i < truckCount; i++) {
    const chunk = packagesWithAngle.slice(i * chunkSize, (i + 1) * chunkSize);
    if (chunk.length === 0) continue;

    // Nearest Neighbor for each truck
    const route = [];
    let currentPos = { x: depot[0], y: depot[1] };
    const unvisited = [...chunk];

    while (unvisited.length > 0) {
      let nearestIndex = -1;
      let minDistance = Infinity;

      for (let j = 0; j < unvisited.length; j++) {
        const pkg = unvisited[j];
        const dist = Math.sqrt(
          Math.pow(pkg.x - currentPos.x, 2) + Math.pow(pkg.y - currentPos.y, 2)
        );
        if (dist < minDistance) {
          minDistance = dist;
          nearestIndex = j;
        }
      }

      const nextPkg = unvisited.splice(nearestIndex, 1)[0];
      route.push(nextPkg.id);
      currentPos = { x: nextPkg.x, y: nextPkg.y };
    }
    routes.push(route);
  }

  console.log(JSON.stringify({ routes }));
}

solve();
