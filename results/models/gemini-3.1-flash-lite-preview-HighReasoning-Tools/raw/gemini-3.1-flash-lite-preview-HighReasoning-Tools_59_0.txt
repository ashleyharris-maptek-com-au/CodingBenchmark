const fs = require('fs');
const input = fs.readFileSync(0, 'utf8').replace(/^\uFEFF/, '');
const data = JSON.parse(input);

const { probability, drones, maxStepsPerDrone } = data;
const height = probability.length;
const width = probability[0].length;
const visitedByAnyone = new Set();
const paths = [];

for (const drone of drones) {
  let { x, y } = drone;
  const path = [[x, y]];
  visitedByAnyone.add(`${x},${y}`);

  for (let step = 0; step < maxStepsPerDrone; step++) {
    let bestMove = { x, y };
    let maxScore = -1;

    const moves = [
      { dx: 0, dy: 0 },
      { dx: 0, dy: -1 },
      { dx: 0, dy: 1 },
      { dx: -1, dy: 0 },
      { dx: 1, dy: 0 }
    ];

    for (const move of moves) {
      const nx = x + move.dx;
      const ny = y + move.dy;

      if (nx >= 0 && nx < width && ny >= 0 && ny < height && probability[ny][nx] !== -1) {
        let score = 0;
        if (!visitedByAnyone.has(`${nx},${ny}`)) {
          score = probability[ny][nx];
        }
        if (score > maxScore) {
          maxScore = score;
          bestMove = { x: nx, y: ny };
        }
      }
    }

    x = bestMove.x;
    y = bestMove.y;
    path.push([x, y]);
    visitedByAnyone.add(`${x},${y}`);
  }
  paths.push(path);
}

process.stdout.write(JSON.stringify({ paths }));
