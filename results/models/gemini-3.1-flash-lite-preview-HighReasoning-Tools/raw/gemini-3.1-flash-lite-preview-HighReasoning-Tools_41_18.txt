; SPIR-V
; Version: 1.0
OpCapability Shader
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

%void = OpTypeVoid
%void_f = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_UBO = OpTypePointer Uniform %UBO
%ptr_v4float = OpTypePointer Uniform %v4float
%ptr_v3float = OpTypePointer Input %v3float
%ptr_v2float = OpTypePointer Input %v2float
%ptr_out_v4float = OpTypePointer Output %v4float
%int = OpTypeInt 32 1
%i_0 = OpConstant %int 0
%i_1 = OpConstant %int 1
%i_2 = OpConstant %int 2
%i_3 = OpConstant %int 3
%i_4 = OpConstant %int 4
%f0 = OpConstant %float 0.0
%f05 = OpConstant %float 0.5
%f1 = OpConstant %float 1.0
%f2 = OpConstant %float 2.0

%in_worldPos = OpVariable %ptr_v3float Input
%in_normal = OpVariable %ptr_v3float Input
%in_uv = OpVariable %ptr_v2float Input
%in_tangent = OpVariable %ptr_v3float Input
%in_color = OpVariable %ptr_v3float Input
%outColor = OpVariable %ptr_out_v4float Output
%ubo_var = OpVariable %ptr_UBO Uniform

%main = OpFunction %void None %void_f
%label = OpLabel

%n_raw = OpLoad %v3float %in_normal
%n = OpExtInst %v3float %glsl450 Normalize %n_raw

%wpos = OpLoad %v3float %in_worldPos
%cpos_ptr = OpAccessChain %ptr_v4float %ubo_var %i_4
%cpos_v4 = OpLoad %v4float %cpos_ptr
%cpos = OpVectorShuffle %v3float %cpos_v4 %cpos_v4 0 1 2
%v_raw = OpFSub %v3float %cpos %wpos
%v = OpExtInst %v3float %glsl450 Normalize %v_raw

%dot = OpDot %float %n %v
%n_dot_v = OpExtInst %float %glsl450 FMax %dot %f0

%less = OpFLessThan %bool %n_dot_v %f05
OpSelectionMerge %merge None
OpBranchConditional %less %if_less %if_greater

%if_less = OpLabel
%g_l = OpFMul %float %f2 %n_dot_v
%color_less = OpCompositeConstruct %v4float %f1 %g_l %f0 %f1
OpBranch %merge

%if_greater = OpLabel
%diff = OpFSub %float %n_dot_v %f05
%term = OpFMul %float %diff %f2
%val = OpFSub %float %f1 %term
%color_greater = OpCompositeConstruct %v4float %val %val %term %f1
OpBranch %merge

%merge = OpLabel
%final_color = OpPhi %v4float %color_less %if_less %color_greater %if_greater
OpStore %outColor %final_color
OpReturn
OpFunctionEnd
