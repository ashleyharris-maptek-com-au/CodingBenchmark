declare const require: any;
declare const process: any;
const fs = require('fs');

interface Point {
    x: number;
    y: number;
}

interface Robot extends Point {
    id: number;
}

interface Job extends Point {
    id: number;
    priority: number;
}

interface Input {
    grid: string[];
    robots: Robot[];
    jobs: Job[];
}

function getDistances(start: Point, grid: string[]): number[][] {
    const height = grid.length;
    const width = grid[0].length;
    const distances: number[][] = Array.from({ length: height }, () => Array(width).fill(Infinity));
    
    const queue: Point[] = [start];
    distances[start.y][start.x] = 0;
    
    const directions = [
        { x: 0, y: 1 },
        { x: 0, y: -1 },
        { x: 1, y: 0 },
        { x: -1, y: 0 }
    ];
    
    while (queue.length > 0) {
        const { x, y } = queue.shift()!;
        
        for (const dir of directions) {
            const nx = x + dir.x;
            const ny = y + dir.y;
            
            if (nx >= 0 && nx < width && ny >= 0 && ny < height && grid[ny][nx] === '.' && distances[ny][nx] === Infinity) {
                distances[ny][nx] = distances[y][x] + 1;
                queue.push({ x: nx, y: ny });
            }
        }
    }
    
    return distances;
}

function solve() {
    let input: string = fs.readFileSync(0, 'utf8');
    // Strip BOM if present
    if (input.length > 0 && input.charCodeAt(0) === 0xFEFF) {
        input = input.slice(1);
    }
    const data: Input = JSON.parse(input);
    
    const robotDistances = data.robots.map(robot => getDistances(robot, data.grid));
    
    const maxRobotId = Math.max(...data.robots.map(r => r.id));
    const assignments: number[][] = Array.from({ length: maxRobotId + 1 }, () => []);
    
    for (const job of data.jobs) {
        let minDistance = Infinity;
        let bestRobotId = -1;
        
        for (const robot of data.robots) {
            const dist = robotDistances[robot.id][job.y][job.x];
            if (dist < minDistance) {
                minDistance = dist;
                bestRobotId = robot.id;
            }
        }
        
        if (bestRobotId !== -1) {
            assignments[bestRobotId].push(job.id);
        }
    }
    
    process.stdout.write(JSON.stringify({ assignments }));
}

solve();
