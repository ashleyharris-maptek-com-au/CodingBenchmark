; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_1_0 = OpConstant %float 1.0
%float_3_0 = OpConstant %float 3.0
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

%uboVar = OpVariable %ptr_Uniform_UBO Uniform
%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2 Input
%in_tangent = OpVariable %ptr_Input_vec3 Input
%in_color = OpVariable %ptr_Input_vec3 Input
%outColor = OpVariable %ptr_Output_vec4 Output

%main = OpFunction %void None %void_func = OpTypeFunction %void
%label = OpLabel
%pos = OpLoad %vec3 %in_worldPos
%norm = OpLoad %vec3 %in_normal
%n = OpExtInst %vec3 %1 Normalize %norm
%ptr_lightPos = OpAccessChain %ptr_Uniform_vec4 %uboVar %int_3
%lightPos = OpLoad %vec4 %ptr_lightPos
%ptr_cameraPos = OpAccessChain %ptr_Uniform_vec4 %uboVar %int_4
%cameraPos = OpLoad %vec4 %ptr_cameraPos
%L_vec = OpFSub %vec3 %lightPos %pos
%L = OpExtInst %vec3 %1 Normalize %L_vec
%V_vec = OpFSub %vec3 %cameraPos %pos
%V = OpExtInst %vec3 %1 Normalize %V_vec
%dotNV = OpDot %float %n %V
%maxNV = OpExtInst %float %1 FMax %dotNV %float_0
%rim_base = OpFSub %float %float_1_0 %maxNV
%rim = OpExtInst %float %1 Pow %rim_base %float_3_0
%dotNL = OpDot %float %n %L
%diffuse = OpExtInst %float %1 FMax %dotNL %float_0
%baseColor = OpCompositeConstruct %vec3 %float_0_2 %float_0_3 %float_0_8
%rimTint = OpCompositeConstruct %vec3 %float_1_0 %float_0_8 %float_0_5
%diffuseTerm = OpFMul %float %float_0_5 %diffuse
%modDiffuse = OpFAdd %float %float_0_3 %diffuseTerm
%finalBase = OpVectorTimesScalar %vec3 %baseColor %modDiffuse
%rimColor = OpVectorTimesScalar %vec3 %rimTint %rim
%finalColor = OpFAdd %vec3 %finalBase %rimColor
%finalColor4 = OpVectorShuffle %vec4 %finalColor %finalColor 0 1 2 4
%outVec = OpCompositeInsert %vec4 %float_1_0 %finalColor4 3
OpStore %outColor %outVec
OpReturn
OpFunctionEnd
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int = OpTypeInt 32 1
%1 = OpExtInstImport "GLSL.std.450"
%void_func = OpTypeFunction %void
