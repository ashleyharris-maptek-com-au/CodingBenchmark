; SPIR-V assembly for Geometry Shader
; Input: InputTrianglesAdjacency (6 vertices)
; Output: OutputTriangleStrip, OutputVertices 12

               OpCapability Geometry
               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor %ubo

; --- Decorators ---
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 12
               OpExecutionMode %main Invocations 1

; Layouts
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               OpDecorate %worldPos_in Location 0
               OpDecorate %normal_in Location 1
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224

               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

; --- Types and Constants ---
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%float_0 = OpConstant %float 0.0
%float_0_008 = OpConstant %float 0.008
%float_0_5 = OpConstant %float 0.5

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec3 %vec3 %float
%ptr_UBO = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_UBO Uniform

%gl_PerVertex_in = OpTypeStruct %vec4
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_vec3_6 = OpTypeArray %vec3 %uint_6
%ptr_in_arr_vec3 = OpTypePointer Input %arr_vec3_6
%worldPos_in = OpVariable %ptr_in_arr_vec3 Input
%normal_in = OpVariable %ptr_in_arr_vec3 Input
%color_in = OpVariable %ptr_in_arr_vec3 Input

%gl_PerVertex_out = OpTypeStruct %vec4
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_vec4 = OpTypePointer Output %vec4
%ptr_out_vec3 = OpTypePointer Output %vec3
%outColor = OpVariable %ptr_out_vec3 Output

%ptr_in_vec4 = OpTypePointer Input %vec4
%ptr_in_vec3 = OpTypePointer Input %vec3
%ptr_int = OpTypePointer Function %int

%func = OpTypeFunction %void

; --- Main ---
%main = OpFunction %void None %func
%label = OpLabel

%thickness_ptr = OpAccessChain %ptr_in_float %ubo %int_5
%thickness = OpLoad %float %thickness_ptr

; Indices: [0, 2, 2, 4, 4, 0]
%arr_int_6 = OpTypeArray %int %uint_6
%indices = OpConstantComposite %arr_int_6 %int_0 %int_2 %int_2 %int_4 %int_4 %int_0
%idx_var = OpVariable %ptr_int_6 Function
OpStore %idx_var %indices
%ptr_int_6 = OpTypePointer Function %arr_int_6

%k = OpVariable %ptr_int_ Function
OpStore %k %int_0

OpBranch %loop_cond

%loop_cond = OpLabel
%k_val = OpLoad %int %k
%cond = OpSLessThan %bool %k_val %int_3
OpLoopMerge %loop_merge %loop_continue None
OpBranchConditional %cond %loop_body %loop_merge

%loop_body = OpLabel
; Get edge indices
%k2 = OpIMul %int %k_val %int_2
%idx0 = OpAccessChain %ptr_int %idx_var %k2
%k2_p1 = OpIAdd %int %k2 %int_1
%idx1 = OpAccessChain %ptr_int %idx_var %k2_p1
%i0 = OpLoad %int %idx0
%i1 = OpLoad %int %idx1

; Retrieve positions and colors
%p0_ptr = OpAccessChain %ptr_in_vec4 %gl_in %i0 %int_0
%p0 = OpLoad %vec4 %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_vec4 %gl_in %i1 %int_0
%p1 = OpLoad %vec4 %p1_ptr

%c0_ptr = OpAccessChain %ptr_in_vec3 %color_in %i0
%c0 = OpLoad %vec3 %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_vec3 %color_in %i1
%c1 = OpLoad %vec3 %c1_ptr
%c_avg = OpFAdd %vec3 %c0 %c1
%c_avg2 = OpVectorTimesScalar %vec3 %c_avg %float_0_5

OpStore %outColor %c_avg2

; Edge vector and perpendicular
%edge = OpFSub %vec4 %p1 %p0
%edge_x = OpCompositeExtract %float %edge 0
%edge_y = OpCompositeExtract %float %edge 1
%perp = OpCompositeConstruct %vec4 %edge_y %float_0_minus_x %float_0 %float_0 ; Simplified
%perp_norm = OpVectorNormalize %vec4 %perp
%offset = OpVectorTimesScalar %vec4 %perp_norm %float_0_008

; Emit 4 vertices for the quad
%v0 = OpFAdd %vec4 %p0 %offset
%v1 = OpFSub %vec4 %p0 %offset
%v2 = OpFAdd %vec4 %p1 %offset
%v3 = OpFSub %vec4 %p1 %offset

%pos_out = OpAccessChain %ptr_out_vec4 %gl_out %int_0
OpStore %pos_out %v0
OpEmitVertex
OpStore %pos_out %v1
OpEmitVertex
OpStore %pos_out %v2
OpEmitVertex
OpStore %pos_out %v3
OpEmitVertex

OpEndPrimitive

OpBranch %loop_continue

%loop_continue = OpLabel
%new_k = OpIAdd %int %k_val %int_1
OpStore %k %new_k
OpBranch %loop_cond

%loop_merge = OpLabel
OpReturn
OpFunctionEnd

%ptr_in_float = OpTypePointer Uniform %float
%float_0_minus_x = OpFNegate %float %edge_x
