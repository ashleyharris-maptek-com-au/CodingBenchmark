; SPIR-V Geometry Shader Pass-through
; Assembled with: spirv-as

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %in_pos %out_pos
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Types and Constants
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%ptr_v4float = OpTypePointer Output %v4float
%ptr_in_v4float = OpTypePointer Input %v4float

; Inputs: Array of 3 vertices (triangles)
%in_pos_type = OpTypeArray %v4float %int_3
%in_pos = OpVariable Input %in_pos_type

; Outputs: Single vertex position
%out_pos = OpVariable Output %v4float

; Main Function
%void_func = OpTypeFunction %void
%main = OpFunction %void None %void_func
%label = OpLabel

; Vertex 0
%ptr0 = OpAccessChain %ptr_in_v4float %in_pos %int_0
%pos0 = OpLoad %v4float %ptr0
OpStore %out_pos %pos0
OpEmitVertex

; Vertex 1
%ptr1 = OpAccessChain %ptr_in_v4float %in_pos %int_1
%pos1 = OpLoad %v4float %ptr1
OpStore %out_pos %pos1
OpEmitVertex

; Vertex 2
%ptr2 = OpAccessChain %ptr_in_v4float %in_pos %int_2
%pos2 = OpLoad %v4float %ptr2
OpStore %out_pos %pos2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
