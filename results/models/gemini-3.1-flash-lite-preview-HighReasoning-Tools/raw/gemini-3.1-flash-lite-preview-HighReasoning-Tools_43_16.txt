// Basic texture-mapped pixel shader for Vulkan
// Expects a texture bound to register t0 and a sampler bound to register s0

struct PSInput {
    float4 position : SV_POSITION;
    float2 uv : TEXCOORD;
};

Texture2D mainTexture : register(t0);
SamplerState mainSampler : register(s0);

float4 main(PSInput input) : SV_TARGET {
    // Sample the texture using the provided UV coordinates
    return mainTexture.Sample(mainSampler, input.uv);
}
