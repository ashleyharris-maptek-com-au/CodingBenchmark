```
; SPIR-V Assembly
; Geometry shader for exploding triangle mesh outward along face normals

OpCapability Geometry
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

; Entry point with all interface variables
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor

; Execution modes
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%void_f = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%f32_0_08 = OpConstant %float 0.08

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_3_pv_in = OpTypeArray %gl_PerVertex_in %int_3
%ptr_arr_3_pv_in = OpTypePointer Input %arr_3_pv_in
%ptr_pv_in = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_arr_3_pv_in Input

%arr_3_v3 = OpTypeArray %v3float %int_3
%ptr_arr_3_v3 = OpTypePointer Input %arr_3_v3
%ptr_in_v3 = OpTypePointer Input %v3float
%inColor = OpVariable %ptr_arr_3_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_gl_PerVertex_out Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Main Function
%main = OpFunction %void None %void_f
%label = OpLabel

; Load Positions
%p0_pv_ptr = OpAccessChain %ptr_pv_in %gl_in %int_0
%p0_pos_ptr = OpAccessChain %ptr_in_v4 %p0_pv_ptr %int_0
%p0 = OpLoad %v4float %p0_pos_ptr

%p1_pv_ptr = OpAccessChain %ptr_pv_in %gl_in %int_1
%p1_pos_ptr = OpAccessChain %ptr_in_v4 %p1_pv_ptr %int_0
%p1 = OpLoad %v4float %p1_pos_ptr

%p2_pv_ptr = OpAccessChain %ptr_pv_in %gl_in %int_2
%p2_pos_ptr = OpAccessChain %ptr_in_v4 %p2_pv_ptr %int_0
%p2 = OpLoad %v4float %p2_pos_ptr

; Extract xyz components for normal calculation
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

; Compute Face Normal: N = normalize(cross(p1-p0, p2-p0))
%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz
%cross = OpExtInst %v3float %glsl Cross %e1 %e2
%normal = OpExtInst %v3float %glsl Normalize %cross

; Offset = N * 0.08
%offset = OpVectorTimesScalar %v3float %normal %f32_0_08

; Helper to emit a shifted vertex
%emit_v = OpFunction %void None %void_f
%emit_label = OpLabel
%v_in = OpFunctionParameter %v4float
%c_in = OpFunctionParameter %v3float
%v_xyz = OpVectorShuffle %v3float %v_in %v_in 0 1 2
%v_w = OpCompositeExtract %float %v_in 3
%v_new_xyz = OpFAdd %v3float %v_xyz %offset
%v_new = OpCompositeConstruct %v4float %v_new_xyz %v_w
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr %v_new
OpStore %outColor %c_in
OpEmitVertex
OpReturn
OpFunctionEnd

; Emit 3 vertices
%c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0 = OpLoad %v3float %c0_ptr
%dummy0 = OpFunctionCall %void %emit_v %p0 %c0

%c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1 = OpLoad %v3float %c1_ptr
%dummy1 = OpFunctionCall %void %emit_v %p1 %c1

%c2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
%c2 = OpLoad %v3float %c2_ptr
%dummy2 = OpFunctionCall %void %emit_v %p2 %c2

OpEndPrimitive
OpReturn
OpFunctionEnd
```
