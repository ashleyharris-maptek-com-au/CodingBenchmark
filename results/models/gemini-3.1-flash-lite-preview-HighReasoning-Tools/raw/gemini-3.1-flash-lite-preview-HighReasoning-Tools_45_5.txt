import struct

def pack_instruction(opcode, operands):
    word_count = len(operands) + 1
    first_word = (word_count << 16) | opcode
    binary = struct.pack('<I', first_word)
    for op in operands:
        if isinstance(op, float):
            binary += struct.pack('<f', op)
        else:
            binary += struct.pack('<I', op)
    return binary

def pack_string(s):
    encoded = s.encode('utf-8') + b'\x00'
    padding = (4 - (len(encoded) % 4)) % 4
    encoded += b'\x00' * padding
    words = []
    for i in range(0, len(encoded), 4):
        words.append(struct.unpack('<I', encoded[i:i+4])[0])
    return words

magic = 0x07230203

instructions = [
