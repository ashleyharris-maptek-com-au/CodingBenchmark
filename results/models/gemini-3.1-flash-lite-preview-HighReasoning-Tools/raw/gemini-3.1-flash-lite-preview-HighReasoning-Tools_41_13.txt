; SPIR-V
; Version: 1.0
OpCapability Shader
%glsl_std_450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %worldPos Location 0
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4

; Variables
%worldPos = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output

; Constants
%const_0_5 = OpConstant %float 0.5
%const_1_0 = OpConstant %float 1.0
%const_0_0 = OpConstant %float 0.0
%const_0_1 = OpConstant %float 0.1
%const_0_4 = OpConstant %float 0.4
%bottom_c = OpConstantComposite %vec3 %const_0_1 %const_0_4 %const_0_1
%top_c = OpConstantComposite %vec3 %const_1_0 %const_1_0 %const_1_0

; Main function
%main = OpFunction %void None %func
%label = OpLabel
%load_pos = OpLoad %vec3 %worldPos
%y = OpCompositeExtract %float %load_pos 1
%t1 = OpFMul %float %y %const_0_5
%t2 = OpFAdd %float %t1 %const_0_5
%t = OpExtInst %float %glsl_std_450 Clamp %t2 %const_0_0 %const_1_0
%color = OpExtInst %vec3 %glsl_std_450 Mix %bottom_c %top_c %t
%out = OpCompositeConstruct %vec4 %color %const_1_0
OpStore %outColor %out
OpReturn
OpFunctionEnd
