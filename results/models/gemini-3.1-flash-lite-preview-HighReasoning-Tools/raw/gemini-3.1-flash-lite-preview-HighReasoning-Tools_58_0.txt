const fs = require('fs');

function solve() {
    const input = fs.readFileSync(0, 'utf8');
    const { grid, robots, jobs } = JSON.parse(input);

    const robotPositions = {};
    const assignments = {};
    const robotIds = robots.map(r => r.id);
    const maxRobotId = Math.max(...robotIds);
    
    // Initialize assignments and robot positions
    for (const robot of robots) {
        robotPositions[robot.id] = { x: robot.x, y: robot.y };
        assignments[robot.id] = [];
    }

    const distanceCache = new Map();

    function getDistance(start, end) {
        const key = `${start.x},${start.y}->${end.x},${end.y}`;
        if (distanceCache.has(key)) return distanceCache.get(key);
        
        const queue = [[start.x, start.y, 0]];
        const visited = new Set();
        visited.add(`${start.x},${start.y}`);
        
        while (queue.length > 0) {
            const [x, y, dist] = queue.shift();
            if (x === end.x && y === end.y) {
                distanceCache.set(key, dist);
                return dist;
            }
            const neighbors = [[x + 1, y], [x - 1, y], [x, y + 1], [x, y - 1]];
            for (const [nx, ny] of neighbors) {
                if (nx >= 0 && nx < grid[0].length && ny >= 0 && ny < grid.length &&
                    grid[ny][nx] !== '#' && !visited.has(`${nx},${ny}`)) {
                    visited.add(`${nx},${ny}`);
                    queue.push([nx, ny, dist + 1]);
                }
            }
        }
        
        distanceCache.set(key, Infinity);
        return Infinity;
    }

    const remainingJobs = [...jobs];

    while (remainingJobs.length > 0) {
        let bestJobIndex = -1;
        let bestRobotId = -1;
        let minTotalDist = Infinity;

        for (let i = 0; i < remainingJobs.length; i++) {
            const job = remainingJobs[i];
            
            for (const robotId of robotIds) {
                const dist = getDistance(robotPositions[robotId], job);
                if (dist < minTotalDist) {
                    minTotalDist = dist;
                    bestJobIndex = i;
                    bestRobotId = robotId;
                }
            }
        }

        if (bestJobIndex === -1) break; // Should not happen if jobs are reachable

        const job = remainingJobs[bestJobIndex];
        assignments[bestRobotId].push(job.id);
        robotPositions[bestRobotId] = { x: job.x, y: job.y };
        remainingJobs.splice(bestJobIndex, 1);
    }

    const result = {
        assignments: Array.from({ length: maxRobotId + 1 }, (_, i) => assignments[i] || [])
    };

    process.stdout.write(JSON.stringify(result));
}

solve();
