const fs = require('fs');

function solve() {
    let input = '';
    try {
        input = fs.readFileSync(0, 'utf8');
    } catch (e) {
        // Handle potential read error
        return;
    }
    
    if (!input) return;

    const data = JSON.parse(input);
    const { tasks } = data;

    const taskMap = new Map();
    for (const task of tasks) {
        taskMap.set(task.id, task);
    }

    const graph = new Map(); // dependencyId -> [taskId_that_depends_on_it]
    const inDegree = new Map(); // taskId -> count

    for (const task of tasks) {
        inDegree.set(task.id, task.deps.length);
        for (const depId of task.deps) {
            if (!graph.has(depId)) {
                graph.set(depId, []);
            }
            graph.get(depId).push(task.id);
        }
    }

    const readyQueue = [];
    for (const task of tasks) {
        if (inDegree.get(task.id) === 0) {
            readyQueue.push(task);
        }
    }

    const order = [];
    while (readyQueue.length > 0) {
        // Sort readyQueue by heuristic: lowest deadline, highest weight
        readyQueue.sort((a, b) => {
            const deadlineA = a.deadline ?? Infinity;
            const deadlineB = b.deadline ?? Infinity;
            if (deadlineA !== deadlineB) {
                return deadlineA - deadlineB;
            }
            return b.weight - a.weight;
        });

        const current = readyQueue.shift();
        order.push(current.id);

        const dependents = graph.get(current.id) || [];
        for (const depId of dependents) {
            inDegree.set(depId, inDegree.get(depId) - 1);
            if (inDegree.get(depId) === 0) {
                readyQueue.push(taskMap.get(depId));
            }
        }
    }

    process.stdout.write(JSON.stringify({ order }));
}

solve();
