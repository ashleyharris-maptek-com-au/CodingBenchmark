import struct

def pack_instruction(opcode, operands):
    word_count = len(operands) + 1
    instruction = (word_count << 16) | opcode
    words = [instruction] + operands
    return struct.pack('<' + 'I' * len(words), *words)

def pack_string(s):
    # SPIR-V strings are null-terminated and padded to 4-byte boundaries
    b = s.encode('utf-8') + b'\x00'
    while len(b) % 4 != 0:
        b += b'\x00'
    return [int.from_bytes(b[i:i+4], 'little') for i in range(0, len(b), 4)]
