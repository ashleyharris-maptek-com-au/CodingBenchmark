const fs = require('fs');

interface Task {
    id: number;
    duration: number;
    deps: number[];
    deadline: number;
    weight: number;
}

interface Input {
    machineCount: number;
    tasks: Task[];
}

function solve() {
    let input: string;
    try {
        input = fs.readFileSync(0, 'utf8');
    } catch (e) {
        return;
    }
    
    if (!input) return;
    
    const data: Input = JSON.parse(input);

    // Topological Sort
    const taskMap = new Map<number, Task>();
    data.tasks.forEach(task => taskMap.set(task.id, task));

    const adj = new Map<number, number[]>(); // taskId -> dependents
    const inDegree = new Map<number, number>();

    data.tasks.forEach(task => {
        inDegree.set(task.id, task.deps.length);
        task.deps.forEach(dep => {
            if (!adj.has(dep)) adj.set(dep, []);
            adj.get(dep)!.push(task.id);
        });
    });

    const queue: number[] = [];
    data.tasks.forEach(task => {
        if (inDegree.get(task.id) === 0) {
            queue.push(task.id);
        }
    });

    const order: number[] = [];
    while (queue.length > 0) {
        // Greedy heuristic: prefer earlier deadline
        queue.sort((a, b) => {
            const ta = taskMap.get(a)!;
            const tb = taskMap.get(b)!;
            return ta.deadline - tb.deadline;
        });

        const u = queue.shift()!;
        order.push(u);

        const dependents = adj.get(u) || [];
        dependents.forEach(v => {
            inDegree.set(v, inDegree.get(v)! - 1);
            if (inDegree.get(v) === 0) {
                queue.push(v);
            }
        });
    }

    process.stdout.write(JSON.stringify({ order }));
}

solve();
