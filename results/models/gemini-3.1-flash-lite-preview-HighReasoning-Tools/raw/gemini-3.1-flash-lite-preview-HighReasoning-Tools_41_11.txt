; SPIR-V
; Version: 1.0
OpCapability Shader
%ext = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block

%int = OpTypeInt 32 0
%int_3 = OpConstant %int 3
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%bool = OpTypeBool
%func = OpTypeFunction %void
%ptr_ubo = OpTypePointer Uniform %UBO
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_float = OpTypePointer Uniform %float
%ptr_vec4 = OpTypePointer Uniform %vec4
%ptr_bool = OpTypePointer Function %bool

%ubo = OpVariable %ptr_ubo Uniform
%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%in_uv = OpVariable %ptr_input_vec2 Input
%outColor = OpVariable %ptr_output_vec4 Output

%f0_0 = OpConstant %float 0.0
%f0_1 = OpConstant %float 0.1
%f0_3 = OpConstant %float 0.3
%f0_5 = OpConstant %float 0.5
%f0_7 = OpConstant %float 0.7
%f0_8 = OpConstant %float 0.8
%f1_0 = OpConstant %float 1.0
%f6_0 = OpConstant %float 6.0

%yellow = OpConstantComposite %vec3 %f1_0 %f0_8 %f0_0
%dark_blue = OpConstantComposite %vec3 %f0_1 %f0_1 %f0_3

%main = OpFunction %void None %func
%label = OpLabel
%uv = OpLoad %vec2 %in_uv
%uv_grid = OpVectorTimesScalar %vec2 %uv %f6_0
%fract_uv = OpExtInst %vec2 %ext Fract %uv_grid
%cell_uv = OpFSub %vec2 %fract_uv %f0_5
%dist = OpExtInst %float %ext Length %cell_uv
%is_dot = OpFLessThan %bool %dist %f0_3

%worldPos = OpLoad %vec3 %in_worldPos
%normal = OpLoad %vec3 %in_normal
%n_norm = OpExtInst %vec3 %ext Normalize %normal

%lpos_ptr = OpAccessChain %ptr_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %lpos_ptr
%lightPos3 = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%L_raw = OpFSub %vec3 %lightPos3 %worldPos
%L = OpExtInst %vec3 %ext Normalize %L_raw
%OpDot = OpDot %float %n_norm %L
%diff = OpExtInst %float %ext FMax %f0_0 %OpDot
%OpFMul = OpFMul %float %f0_7 %diff
%lambert = OpFAdd %float %f0_3 %OpFMul

%color_base = OpSelect %vec3 %is_dot %yellow %dark_blue
%color_shaded = OpVectorTimesScalar %vec3 %color_base %lambert
%out = OpCompositeConstruct %vec4 %color_shaded %f1_0
OpStore %outColor %out
OpReturn
OpFunctionEnd
