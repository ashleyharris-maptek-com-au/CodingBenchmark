```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    vec2 tiledUV = uv * vec2(8.0, 16.0);

    // Staggered rows: shift every other row by 0.5
    if (int(floor(tiledUV.y)) % 2 != 0) {
        tiledUV.x += 0.5;
    }

    vec2 f = fract(tiledUV);

    vec3 brickColor = vec3(0.7, 0.2, 0.1);
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);

    // Mortar check: thickness 0.05 in X and 0.1 in Y
    vec3 finalColor = brickColor;
    if (f.x < 0.05 || f.y < 0.1) {
        finalColor = mortarColor;
    }

    // Lambertian lighting: (0.4 + 0.6 * diffuse)
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float lighting = 0.4 + 0.6 * diffuse;

    outColor = vec4(finalColor * lighting, 1.0);
}
```
