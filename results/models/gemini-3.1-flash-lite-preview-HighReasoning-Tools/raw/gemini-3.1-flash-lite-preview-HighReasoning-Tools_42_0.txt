```assembly
; Geometry Shader SPIR-V Assembly
OpCapability Geometry
OpMemoryModel Logical GLSL450

; Entry point and Execution Modes
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %out_color
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %out_color Location 0

; Types and Constants
%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%h = OpConstant %float 0.04
%nh = OpConstant %float -0.04

; Structs
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

; Pointers
%ptr_in_pv_struct = OpTypePointer Input %gl_PerVertex_in
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%out_color = OpVariable %ptr_out_v3 Output

; Main function
%main = OpFunction %void None %fn_void
%label = OpLabel

; Read gl_in[0].gl_Position
%pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p = OpLoad %v4float %pos_ptr
%px = OpCompositeExtract %float %p 0
%py = OpCompositeExtract %float %p 1

; Read color[0]
%col_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%c = OpLoad %v3float %col_ptr
OpStore %out_color %c

; Vertex 1: (x-h, y-h)
%v1x = OpFSub %float %px %h
%v1y = OpFSub %float %py %h
%v1 = OpCompositeConstruct %v4float %v1x %v1y %float_0 %float_1
%pos_out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_out1 %v1
OpEmitVertex

; Vertex 2: (x+h, y-h)
%v2x = OpFAdd %float %px %h
%v2 = OpCompositeConstruct %v4float %v2x %v1y %float_0 %float_1
OpStore %pos_out1 %v2
OpEmitVertex

; Vertex 3: (x-h, y+h)
%v3y = OpFAdd %float %py %h
%v3 = OpCompositeConstruct %v4float %v1x %v3y %float_0 %float_1
OpStore %pos_out1 %v3
OpEmitVertex

; Vertex 4: (x+h, y+h)
%v4 = OpCompositeConstruct %v4float %v2x %v3y %float_0 %float_1
OpStore %pos_out1 %v4
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
