declare const process: any;

interface Town {
  id: number;
  x: number;
  y: number;
  value: number;
}

interface InputData {
  width: number;
  height: number;
  grid: string[];
  ignitions: [number, number][];
  towns: Town[];
  cutBudget: number;
}

function processData(input: string) {
  // Remove BOM if present
  const cleanedInput = input.replace(/^\uFEFF/, '');
  const data: InputData = JSON.parse(cleanedInput);

  const { width, height, grid, ignitions, towns, cutBudget } = data;

  // BFS to find shortest path from any ignition to each cell
  const dist = Array.from({ length: height }, () => Array(width).fill(Infinity));
  const parent = Array.from({ length: height }, () => Array(width).fill(null));
  const queue: [number, number][] = [];

  for (const [x, y] of ignitions) {
    dist[y][x] = 0;
    queue.push([x, y]);
  }

  let head = 0;
  while (head < queue.length) {
    const [x, y] = queue[head++];
    const neighbors: [number, number][] = [
      [x + 1, y], [x - 1, y], [x, y + 1], [x, y - 1]
    ];

    for (const [nx, ny] of neighbors) {
      if (nx >= 0 && nx < width && ny >= 0 && ny < height && grid[ny][nx] === '.' && dist[ny][nx] === Infinity) {
        dist[ny][nx] = dist[y][x] + 1;
        parent[ny][nx] = [x, y];
        queue.push([nx, ny]);
      }
    }
  }

  // Identify vulnerable towns
  const vulnerableTowns = towns.filter(t => dist[t.y][t.x] !== Infinity);
  
  // For each vulnerable town, find the path back to an ignition
  const cellScores = new Map<string, number>();

  for (const town of vulnerableTowns) {
    let curr: [number, number] | null = [town.x, town.y];
    while (curr) {
      const [x, y] = curr as [number, number];
      // Don't cut the town itself (could be problematic) or ignitions (they are already on fire)
      if (grid[y][x] === '.' && dist[y][x] > 0) {
        const key = `${x},${y}`;
        cellScores.set(key, (cellScores.get(key) || 0) + town.value);
      }
      curr = parent[y][x];
    }
  }

  // Pick top N cells
  const sortedCells = Array.from(cellScores.entries())
    .sort((a, b) => b[1] - a[1])
    .slice(0, cutBudget);

  const firebreaks = sortedCells.map(c => c[0].split(',').map(Number) as [number, number]);

  process.stdout.write(JSON.stringify({ firebreaks }));
}

let input = '';
process.stdin.on('data', (chunk: any) => input += chunk);
process.stdin.on('end', () => processData(input));
