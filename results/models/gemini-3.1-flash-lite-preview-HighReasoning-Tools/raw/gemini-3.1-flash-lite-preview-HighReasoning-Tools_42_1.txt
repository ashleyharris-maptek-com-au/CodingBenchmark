```spirv
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0

%h = OpConstant %float 0.05

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_input_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_input_v4 = OpTypePointer Input %v4float

%arr_v3 = OpTypeArray %v3float %uint_1
%ptr_input_arr_v3 = OpTypePointer Input %arr_v3
%ptr_input_v3 = OpTypePointer Input %v3float

%ptr_output_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_output_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_input_arr_pv Input
%color = OpVariable %ptr_input_arr_v3 Input
%gl_out = OpVariable %ptr_output_pv Output
%outColor = OpVariable %ptr_output_v3 Output

%func = OpTypeFunction %void

%main = OpFunction %void None %func
%label = OpLabel

; Load Position and Color
%ptr_gl_in_0 = OpAccessChain %ptr_input_pv %gl_in %int_0
%ptr_pos_in = OpAccessChain %ptr_input_v4 %ptr_gl_in_0 %int_0
%pos = OpLoad %v4float %ptr_pos_in

%ptr_color_0 = OpAccessChain %ptr_input_v3 %color %int_0
%c = OpLoad %v3float %ptr_color_0

%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1
%z = OpCompositeExtract %float %pos 2
%w = OpCompositeExtract %float %pos 3

%y_plus_h = OpFAdd %float %y %h
%y_minus_h = OpFSub %float %y %h
%x_plus_h = OpFAdd %float %x %h
%x_minus_h = OpFSub %float %x %h

%ptr_gl_out_pos = OpAccessChain %ptr_output_v4 %gl_out %int_0

; 1. Top
%top = OpCompositeConstruct %v4float %x %y_plus_h %z %w
OpStore %ptr_gl_out_pos %top
OpStore %outColor %c
OpEmitVertex

; 2. Right
%right = OpCompositeConstruct %v4float %x_plus_h %y %z %w
OpStore %ptr_gl_out_pos %right
OpStore %outColor %c
OpEmitVertex

; 3. Left
%left = OpCompositeConstruct %v4float %x_minus_h %y %z %w
OpStore %ptr_gl_out_pos %left
OpStore %outColor %c
OpEmitVertex

; 4. Bottom
%bottom = OpCompositeConstruct %v4float %x %y_minus_h %z %w
OpStore %ptr_gl_out_pos %bottom
OpStore %outColor %c
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
