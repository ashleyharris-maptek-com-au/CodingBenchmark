#version 430 core

layout(local_size_x = 256) in;

struct Particle {
    vec4 position; // xyz: position, w: mass
    vec4 velocity; // xyz: velocity, w: padding
};

layout(std430, binding = 0) buffer ParticleBuffer {
    Particle particles[];
};

uniform float G;
uniform float dt;
uniform float softening;
uniform int numParticles;

void main() {
    uint i = gl_GlobalInvocationID.x;
    if (i >= numParticles) return;

    vec3 pos = particles[i].position.xyz;
    vec3 vel = particles[i].velocity.xyz;
    float mass = particles[i].position.w;

    vec3 force = vec3(0.0);

    for (int j = 0; j < numParticles; j++) {
        if (i == j) continue;

        vec3 r = particles[j].position.xyz - pos;
        float distSq = dot(r, r) + softening * softening;
        float invDist = 1.0 / sqrt(distSq);
        float invDist3 = invDist * invDist * invDist;

        force += G * mass * particles[j].position.w * r * invDist3;
    }

    // Euler integration
    vel += force * dt / mass;
    pos += vel * dt;

    particles[i].position.xyz = pos;
    particles[i].velocity.xyz = vel;
}
