import struct

def to_word(opcode, operands):
    word_count = len(operands) + 1
    return [((word_count << 16) | opcode)] + operands

def to_string_words(s):
    s_bytes = s.encode('utf-8') + b'\x00'
    while len(s_bytes) % 4 != 0:
        s_bytes += b'\x00'
    words = []
    for i in range(0, len(s_bytes), 4):
        words.append(struct.unpack('<I', s_bytes[i:i+4])[0])
    return words
