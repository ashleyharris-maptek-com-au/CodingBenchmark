// (Note: This is a simplified implementation for demonstration. Full robust CSG is complex.)
#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>

struct Vec3 {
    double x, y, z;
    Vec3 operator+(const Vec3& other) const { return {x + other.x, y + other.y, z + other.z}; }
    Vec3 operator-(const Vec3& other) const { return {x - other.x, y - other.y, z - other.z}; }
    Vec3 operator*(double scalar) const { return {x * scalar, y * scalar, z * scalar}; }
    double dot(const Vec3& other) const { return x * other.x + y * other.y + z * other.z; }
    Vec3 cross(const Vec3& other) const { return {y * other.z - z * other.y, z * other.x - x * other.z, x * other.y - y * other.x}; }
};

struct Face {
    int a, b, c;
};

struct Mesh {
    std::vector<Vec3> vertices;
    std::vector<Face> faces;
};

Mesh read_mesh(int v_count, int f_count) {
    Mesh mesh;
    mesh.vertices.resize(v_count);
    for (int i = 0; i < v_count; ++i) {
        std::cin >> mesh.vertices[i].x >> mesh.vertices[i].y >> mesh.vertices[i].z;
    }
    mesh.faces.resize(f_count);
    for (int i = 0; i < f_count; ++i) {
        std::cin >> mesh.faces[i].a >> mesh.faces[i].b >> mesh.faces[i].c;
    }
    return mesh;
}

void write_mesh(const Mesh& mesh) {
    std::cout << mesh.vertices.size() << " " << mesh.faces.size() << "\n";
    for (const auto& v : mesh.vertices) {
        std::cout << v.x << " " << v.y << " " << v.z << "\n";
    }
    for (const auto& f : mesh.faces) {
        std::cout << f.a << " " << f.b << " " << f.c << "\n";
    }
}

// Simple point-in-mesh test using ray casting
bool is_point_inside(const Mesh& mesh, const Vec3& p) {
    int count = 0;
    // Ray in +X direction
    Vec3 ray_dir = {1.0, 0.0, 0.0};
    for (const auto& f : mesh.faces) {
        Vec3 v0 = mesh.vertices[f.a];
        Vec3 v1 = mesh.vertices[f.b];
        Vec3 v2 = mesh.vertices[f.c];
        
        // Ray-triangle intersection
        Vec3 edge1 = v1 - v0;
        Vec3 edge2 = v2 - v0;
        Vec3 h = ray_dir.cross(edge2);
        double a = edge1.dot(h);
        if (std::abs(a) < 1e-9) continue;
        double f_inv = 1.0 / a;
        Vec3 s = p - v0;
        double u = f_inv * s.dot(h);
        if (u < 0.0 || u > 1.0) continue;
        Vec3 q = s.cross(edge1);
        double v = f_inv * ray_dir.dot(q);
        if (v < 0.0 || u + v > 1.0) continue;
        double t = f_inv * edge2.dot(q);
        if (t > 1e-9) count++;
    }
    return (count % 2 == 1);
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int va, fa, vb, fb;
    if (!(std::cin >> va >> fa >> vb >> fb)) return 0;

    Mesh meshA = read_mesh(va, fa);
    Mesh meshB = read_mesh(vb, fb);
    
    // Union approach:
    // 1. Keep faces of A that are not inside B
    // 2. Keep faces of B that are not inside A
    Mesh unionMesh;
    
    for (const auto& f : meshA.faces) {
        Vec3 p = (meshA.vertices[f.a] + meshA.vertices[f.b] + meshA.vertices[f.c]) * (1.0/3.0);
        if (!is_point_inside(meshB, p)) {
            unionMesh.faces.push_back(f);
        }
    }
    
    for (const auto& f : meshB.faces) {
        Vec3 p = (meshB.vertices[f.a] + meshB.vertices[f.b] + meshB.vertices[f.c]) * (1.0/3.0);
        if (!is_point_inside(meshA, p)) {
            unionMesh.faces.push_back({f.a + va, f.b + va, f.c + va});
        }
    }
    
    unionMesh.vertices = meshA.vertices;
    unionMesh.vertices.insert(unionMesh.vertices.end(), meshB.vertices.begin(), meshB.vertices.end());
    
    write_mesh(unionMesh);
    
    return 0;
}
