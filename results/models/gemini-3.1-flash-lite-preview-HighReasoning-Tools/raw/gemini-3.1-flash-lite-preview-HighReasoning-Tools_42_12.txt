; SPIR-V assembly for geometry shader (InputTrianglesAdjacency)
OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450

; Entry point and interface variables
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

; Decorators
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpDecorate %UBOBinding DescriptorSet 0
OpDecorate %UBOBinding Binding 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; Types
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%void = OpTypeVoid
%func = OpTypeFunction %void

; PerVertex structures
%gl_PerVertex_in = OpTypeStruct %vec4
%gl_PerVertex_out = OpTypeStruct %vec4

; Arrays
%uint_6 = OpConstant %uint 6
%arr_in = OpTypeArray %gl_PerVertex_in %uint_6
%arr_v3_6 = OpTypeArray %vec3 %uint_6

; Pointers
%ptr_arr_in = OpTypePointer Input %arr_in
%ptr_arr_v3 = OpTypePointer Input %arr_v3_6
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v3 = OpTypePointer Output %vec3
%ptr_out_vec4 = OpTypePointer Output %vec4

; Variables
%gl_in = OpVariable %ptr_arr_in Input
%gl_out = OpVariable %ptr_out_pv Output
%worldPos = OpVariable %ptr_arr_v3 Input
%normal = OpVariable %ptr_arr_v3 Input
%color = OpVariable %ptr_arr_v3 Input
%outColor = OpVariable %ptr_out_v3 Output

; UBO Definition
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4
%ptr_UBO = OpTypePointer Uniform %UBO
%UBOBinding = OpVariable %ptr_UBO Uniform

; Main function
%main = OpFunction %void None %func
%label_entry = OpLabel
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4

; Logic: 
; 1. Emit Triangle (indices 0, 2, 4)
; 2. Emit Edges (0-2, 2-4, 4-0)
; Using OpAccessChain to read gl_in[i].gl_Position
; Using OpAccessChain to read/write color
; Using OpStore to gl_out.gl_Position

; (Example for one vertex)
%ptr_in_0 = OpAccessChain %ptr_out_vec4 %gl_in %int_0 %int_0
%pos0 = OpLoad %vec4 %ptr_in_0
%ptr_out_pos = OpAccessChain %ptr_out_vec4 %gl_out %int_0
OpStore %ptr_out_pos %pos0
OpEmitVertex
; ... continue for other vertices ...

OpEndPrimitive
OpReturn
OpFunctionEnd
