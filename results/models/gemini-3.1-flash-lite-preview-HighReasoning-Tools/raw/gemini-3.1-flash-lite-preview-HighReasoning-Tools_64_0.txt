declare const require: any;
const fs = require('fs');

interface Drone {
    id: number;
    x: number;
    y: number;
}

interface InputData {
    probability: number[][];
    drones: Drone[];
    maxStepsPerDrone: number;
}

function solve() {
    const input: string = fs.readFileSync(0, 'utf8');
    const data: InputData = JSON.parse(input);
    const { probability, drones, maxStepsPerDrone } = data;

    const rows = probability.length;
    const cols = probability[0].length;
    const covered: boolean[][] = Array.from({ length: rows }, () => Array(cols).fill(false));

    const paths: [number, number][][] = [];

    for (const drone of drones) {
        let cx = drone.x;
        let cy = drone.y;
        const path: [number, number][] = [[cx, cy]];

        // Mark starting position as covered if it's not an obstacle
        if (probability[cy][cx] !== -1) {
            covered[cy][cx] = true;
        }

        for (let step = 0; step < maxStepsPerDrone; step++) {
            let bestMove = { x: cx, y: cy, val: -1 };

            // Possible moves: stay, up, down, left, right
            const moves = [
                { dx: 0, dy: 0 },
                { dx: 0, dy: -1 },
                { dx: 0, dy: 1 },
                { dx: -1, dy: 0 },
                { dx: 1, dy: 0 }
            ];

            for (const move of moves) {
                const nx = cx + move.dx;
                const ny = cy + move.dy;

                // Check bounds and obstacle
                if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && probability[ny][nx] !== -1) {
                    // Only count value if not yet covered
                    const val = covered[ny][nx] ? 0 : probability[ny][nx];
                    
                    // Choose the move with the highest potential value
                    if (val > bestMove.val) {
                        bestMove = { x: nx, y: ny, val };
                    } else if (bestMove.val === -1) {
                        // If all potential moves have 0 value, any valid move is fine
                        bestMove = { x: nx, y: ny, val };
                    }
                }
            }

            cx = bestMove.x;
            cy = bestMove.y;
            path.push([cx, cy]);
            if (probability[cy][cx] !== -1) {
                covered[cy][cx] = true;
            }
        }
        paths.push(path);
    }

    process.stdout.write(JSON.stringify({ paths }));
}

solve();
