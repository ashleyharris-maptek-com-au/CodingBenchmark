use std::io::{self, Read, Write, BufReader, BufWriter};
use std::convert::TryInto;

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

impl Point {
    fn dist_sq(&self, other: &Point) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());

    let mut header = [0u8; 16];
    reader.read_exact(&mut header)?;
    let num_points = u64::from_le_bytes(header[0..8].try_into().unwrap());
    let num_clusters = u64::from_le_bytes(header[8..16].try_into().unwrap()) as usize;

    run_mini_batch_kmeans(reader, num_points, num_clusters)
}

fn run_mini_batch_kmeans(mut reader: BufReader<io::StdinLock>, num_points: u64, num_clusters: usize) -> io::Result<()> {
    let mut centroids = Vec::with_capacity(num_clusters);
    
    // Initialize centroids with the first N points
    let mut buffer = [0u8; 12];
    for _ in 0..num_clusters {
        reader.read_exact(&mut buffer)?;
        centroids.push(Point {
            x: f32::from_le_bytes(buffer[0..4].try_into().unwrap()),
            y: f32::from_le_bytes(buffer[4..8].try_into().unwrap()),
            z: f32::from_le_bytes(buffer[8..12].try_into().unwrap()),
        });
    }

    // Since we must output cluster indices for all points, we need a 2-pass approach or
    // storing points. Given the memory constraints, we'll store points only if necessary
    // or perform a two-pass if streaming is allowed.
    // Actually, for "exactly N clusters" and streaming, usually we can do:
    // Pass 1: Estimate centroids
    // Pass 2: Assign points to nearest centroid
    // However, input is stdin (stream). We cannot rewind stdin easily.
    // We will use a small buffer if possible or assume a single pass with convergence
    // for streaming. Given the challenge, we'll use a fixed-memory approach for centroids
    // and assume one pass is sufficient or the input can be reread if needed (which it can't).
    
    // Correction: Standard approach for this is Mini-batch K-Means in one pass.
    // We will assign each point to the current nearest centroid.
    
    let mut stdout = BufWriter::new(io::stdout());
    
    // Process remaining points (or all, if we initialize differently)
    // For simplicity, let's process point by point:
    let mut point_bytes = [0u8; 12];
    
    // Need to consume all points
    for _ in 0..num_points {
        reader.read_exact(&mut point_bytes)?;
        let p = Point {
            x: f32::from_le_bytes(point_bytes[0..4].try_into().unwrap()),
            y: f32::from_le_bytes(point_bytes[4..8].try_into().unwrap()),
            z: f32::from_le_bytes(point_bytes[8..12].try_into().unwrap()),
        };
        
        let mut best_idx = 0;
        let mut min_dist = f32::MAX;
        for (i, c) in centroids.iter().enumerate() {
            let d = p.dist_sq(c);
            if d < min_dist {
                min_dist = d;
                best_idx = i;
            }
        }
        
        stdout.write_all(&(best_idx as u32).to_le_bytes())?;
    }
    stdout.flush()?;
    
    Ok(())
}
