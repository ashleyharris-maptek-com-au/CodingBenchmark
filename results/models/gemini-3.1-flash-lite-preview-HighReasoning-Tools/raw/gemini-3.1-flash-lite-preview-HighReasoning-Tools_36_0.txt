#include <iostream>
#include <vector>
#include <algorithm>
#include <bit>
#include <cstdint>

struct Bitset {
    std::vector<uint64_t> bits;
    int n;
    Bitset(int n = 0) : n(n), bits((n + 63) / 64, 0) {}
    void set(int i) { bits[i / 64] |= (1ULL << (i % 64)); }
    void reset(int i) { bits[i / 64] &= ~(1ULL << (i % 64)); }
    bool test(int i) const { return (bits[i / 64] & (1ULL << (i % 64))) != 0; }
    void set_all() { std::fill(bits.begin(), bits.end(), ~0ULL); }
    void reset_all() { std::fill(bits.begin(), bits.end(), 0); }
    bool none() const { for (auto b : bits) if (b != 0) return false; return true; }
    int count() const {
        int c = 0;
        for (auto b : bits) c += std::popcount(b);
        return c;
    }
    Bitset operator&(const Bitset& other) const {
        Bitset res(n);
        for (size_t i = 0; i < bits.size(); ++i) res.bits[i] = bits[i] & other.bits[i];
        return res;
    }
    Bitset operator|(const Bitset& other) const {
        Bitset res(n);
        for (size_t i = 0; i < bits.size(); ++i) res.bits[i] = bits[i] | other.bits[i];
        return res;
    }
    Bitset operator~() const {
        Bitset res(n);
        for (size_t i = 0; i < bits.size(); ++i) res.bits[i] = ~bits[i];
        if (n % 64 != 0) res.bits.back() &= (1ULL << (n % 64)) - 1;
        return res;
    }
    int find_first() const {
        for (size_t i = 0; i < bits.size(); ++i) {
            if (bits[i] != 0) return (int)(i * 64 + std::countr_zero(bits[i]));
        }
        return -1;
    }
};

std::vector<Bitset> adj;
int N;
std::vector<int> max_clique;

void max_clique_recursive(Bitset R, Bitset P, Bitset X) {
    if (P.none() && X.none()) {
        if (R.count() > (int)max_clique.size()) {
            max_clique.clear();
            for (int i = 0; i < N; ++i) {
                if (R.test(i)) max_clique.push_back(i);
            }
        }
        return;
    }
    if (P.none()) return;

    if (R.count() + P.count() <= (int)max_clique.size()) return;

    Bitset P_or_X = P | X;
    int pivot = P_or_X.find_first();
    
    Bitset P_without_neighbors_of_pivot = P & ~adj[pivot];

    for (int v = 0; v < N; ++v) {
        if (P_without_neighbors_of_pivot.test(v)) {
            Bitset new_R = R; new_R.set(v);
            max_clique_recursive(new_R, P & adj[v], X & adj[v]);
            P.reset(v);
            X.set(v);
        }
    }
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int m;
    if (!(std::cin >> N >> m)) return 0;

    adj.resize(N, Bitset(N));
    for (int i = 0; i < N; ++i) {
        adj[i].set_all();
        adj[i].reset(i);
    }

    for (int i = 0; i < m; ++i) {
        int u, v;
        std::cin >> u >> v;
        adj[u].reset(v);
        adj[v].reset(u);
    }

    Bitset R(N), P(N), X(N);
    for (int i = 0; i < N; ++i) P.set(i);

    max_clique_recursive(R, P, X);

    std::cout << max_clique.size() << "\n";
    for (int i = 0; i < (int)max_clique.size(); ++i) {
        std::cout << max_clique[i] << (i == (int)max_clique.size() - 1 ? "" : " ");
    }
    std::cout << "\n";

    return 0;
}
