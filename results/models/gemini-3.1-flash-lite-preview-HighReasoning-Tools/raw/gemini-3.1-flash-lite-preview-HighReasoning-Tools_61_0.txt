declare const require: any;
const fs = require('fs');

interface Package { id: number; x: number; y: number; }
interface Data { depot: [number, number]; truckCount: number; packages: Package[]; }

function dist(p1: [number, number], p2: [number, number]): number {
  return Math.sqrt(Math.pow(p1[0] - p2[0], 2) + Math.pow(p1[1] - p2[1], 2));
}

function solve() {
  const input: string = fs.readFileSync(0, 'utf8');
  const data: Data = JSON.parse(input);
  const { depot, truckCount, packages } = data;

  const routes: number[][] = Array.from({ length: truckCount }, () => []);
  const packageLocations: Map<number, [number, number]> = new Map();
  packages.forEach(p => packageLocations.set(p.id, [p.x, p.y]));

  for (const pkg of packages) {
    let bestInsertion = {
      routeIndex: -1,
      position: -1,
      cost: Infinity
    };

    for (let r = 0; r < truckCount; r++) {
      const route = routes[r];
      
      // Try inserting at all possible positions
      for (let pos = 0; pos <= route.length; pos++) {
        const prev = pos === 0 ? depot : packageLocations.get(route[pos - 1])!;
        const next = pos === route.length ? depot : packageLocations.get(route[pos])!;
        
        const oldDist = dist(prev, next);
        const newDist = dist(prev, [pkg.x, pkg.y]) + dist([pkg.x, pkg.y], next);
        const cost = newDist - oldDist;

        if (cost < bestInsertion.cost) {
          bestInsertion = { routeIndex: r, position: pos, cost };
        }
      }
    }

    routes[bestInsertion.routeIndex].splice(bestInsertion.position, 0, pkg.id);
  }

  process.stdout.write(JSON.stringify({ routes }));
}

solve();
