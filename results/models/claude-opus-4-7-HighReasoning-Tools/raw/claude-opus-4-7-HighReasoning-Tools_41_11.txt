; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %color Location 4
               OpDecorate %outColor Location 0

               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
      %mat4  = OpTypeMatrix %v4float 4
        %UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
        %ubo = OpVariable %ptr_uniform_UBO Uniform

  %ptr_in_v3 = OpTypePointer Input %v3float
  %ptr_in_v2 = OpTypePointer Input %v2float
 %ptr_out_v4 = OpTypePointer Output %v4float
%ptr_uniform_v4 = OpTypePointer Uniform %v4float

   %worldPos = OpVariable %ptr_in_v3 Input
     %normal = OpVariable %ptr_in_v3 Input
         %uv = OpVariable %ptr_in_v2 Input
    %tangent = OpVariable %ptr_in_v3 Input
      %color = OpVariable %ptr_in_v3 Input
   %outColor = OpVariable %ptr_out_v4 Output

        %int = OpTypeInt 32 1
      %int_3 = OpConstant %int 3
       %bool = OpTypeBool

       %f_0  = OpConstant %float 0.0
       %f_1  = OpConstant %float 1.0
       %f_6  = OpConstant %float 6.0
     %f_0_5  = OpConstant %float 0.5
     %f_0_3  = OpConstant %float 0.3
     %f_0_7  = OpConstant %float 0.7
    %f_0_09  = OpConstant %float 0.09

      %yel_r = OpConstant %float 1.0
      %yel_g = OpConstant %float 0.8
      %yel_b = OpConstant %float 0.0
       %bg_r = OpConstant %float 0.1
       %bg_g = OpConstant %float 0.1
       %bg_b = OpConstant %float 0.3

       %main = OpFunction %void None %fn_void
      %entry = OpLabel

     %uv_val = OpLoad %v2float %uv
  %uv_scaled = OpVectorTimesScalar %v2float %uv_val %f_6
   %uv_floor = OpExtInst %v2float %1 Floor %uv_scaled
    %uv_cell = OpFSub %v2float %uv_scaled %uv_floor
      %half2 = OpCompositeConstruct %v2float %f_0_5 %f_0_5
          %d = OpFSub %v2float %uv_cell %half2
         %d2 = OpDot %float %d %d

     %is_dot = OpFOrdLessThan %bool %d2 %f_0_09

     %yellow = OpCompositeConstruct %v3float %yel_r %yel_g %yel_b
         %bg = OpCompositeConstruct %v3float %bg_r %bg_g %bg_b
       %base = OpSelect %v3float %is_dot %yellow %bg

      %n_raw = OpLoad %v3float %normal
          %n = OpExtInst %v3float %1 Normalize %n_raw

     %lp_ptr = OpAccessChain %ptr_uniform_v4 %ubo %int_3
        %lp4 = OpLoad %v4float %lp_ptr
        %lp3 = OpVectorShuffle %v3float %lp4 %lp4 0 1 2

         %wp = OpLoad %v3float %worldPos
     %ld_raw = OpFSub %v3float %lp3 %wp
         %ld = OpExtInst %v3float %1 Normalize %ld_raw

      %ndotl = OpDot %float %n %ld
       %diff = OpExtInst %float %1 FMax %ndotl %f_0
    %dscaled = OpFMul %float %diff %f_0_7
     %factor = OpFAdd %float %f_0_3 %dscaled

     %shaded = OpVectorTimesScalar %v3float %base %factor
          %r = OpCompositeExtract %float %shaded 0
          %g = OpCompositeExtract %float %shaded 1
          %b2 = OpCompositeExtract %float %shaded 2
       %out4 = OpCompositeConstruct %v4float %r %g %b2 %f_1
               OpStore %outColor %out4

               OpReturn
               OpFunctionEnd
