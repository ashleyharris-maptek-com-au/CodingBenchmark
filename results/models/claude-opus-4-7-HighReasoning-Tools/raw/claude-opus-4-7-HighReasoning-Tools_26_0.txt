use std::io::{self, Read, Write, BufReader, BufWriter};

#[inline(always)]
fn dist2(a: &[f32; 3], b: &[f32; 3]) -> f32 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

#[inline(always)]
fn assign(p: &[f32; 3], centroids: &[[f32; 3]]) -> u32 {
    let mut best = 0u32;
    let mut best_d = f32::MAX;
    for (i, c) in centroids.iter().enumerate() {
        let d = dist2(p, c);
        if d < best_d {
            best_d = d;
            best = i as u32;
        }
    }
    best
}

struct Rng(u64);
impl Rng {
    fn new(seed: u64) -> Self {
        Rng(seed.wrapping_mul(2862933555777941757).wrapping_add(3037000493))
    }
    fn next_u64(&mut self) -> u64 {
        self.0 = self.0.wrapping_mul(6364136223846793005).wrapping_add(1442695040888963407);
        self.0
    }
    fn next_usize(&mut self, n: usize) -> usize {
        (self.next_u64() as usize) % n.max(1)
    }
    fn next_f32(&mut self) -> f32 {
        ((self.next_u64() >> 40) as f32) / ((1u64 << 24) as f32)
    }
}

fn read_header<R: Read>(r: &mut R) -> io::Result<(u64, u32)> {
    let mut header = String::new();
    let mut byte = [0u8; 1];
    loop {
        let n = r.read(&mut byte)?;
        if n == 0 { break; }
        if byte[0] == b'\n' { break; }
        header.push(byte[0] as char);
    }
    let parts: Vec<&str> = header.trim().split_whitespace().collect();
    let num_points: u64 = parts[0].parse().unwrap();
    let num_clusters: u32 = parts[1].parse().unwrap();
    Ok((num_points, num_clusters))
}

fn read_point<R: Read>(r: &mut R, buf: &mut [u8; 12]) -> io::Result<Option<[f32; 3]>> {
    match r.read_exact(buf) {
        Ok(()) => {
            let x = f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]);
            let y = f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]);
            let z = f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]);
            Ok(Some([x, y, z]))
        }
        Err(e) if e.kind() == io::ErrorKind::UnexpectedEof => Ok(None),
        Err(e) => Err(e),
    }
}

#[inline(always)]
fn write_idx<W: Write>(w: &mut W, idx: u32) -> io::Result<()> {
    w.write_all(&idx.to_le_bytes())
}

fn kmeans_pp_init(points: &[[f32; 3]], k: usize, rng: &mut Rng) -> Vec<[f32; 3]> {
    let n = points.len();
    let mut centroids: Vec<[f32; 3]> = Vec::with_capacity(k);
    if n == 0 {
        for _ in 0..k { centroids.push([0.0, 0.0, 0.0]); }
        return centroids;
    }
    let sample_size = n.min(50_000);
    let stride = (n / sample_size).max(1);
    let sample_idx: Vec<usize> = (0..sample_size).map(|i| (i * stride).min(n - 1)).collect();

    centroids.push(points[rng.next_usize(n)]);
    let mut d2: Vec<f32> = vec![f32::MAX; sample_size];

    for _ in 1..k {
        let last = *centroids.last().unwrap();
        let mut sum = 0.0f64;
        for (i, &si) in sample_idx.iter().enumerate() {
            let d = dist2(&points[si], &last);
            if d < d2[i] { d2[i] = d; }
            sum += d2[i] as f64;
        }
        if sum <= 0.0 {
            centroids.push(points[rng.next_usize(n)]);
            continue;
        }
        let target = rng.next_f32() as f64 * sum;
        let mut acc = 0.0f64;
        let mut chosen = sample_idx[sample_size - 1];
        for (i, &si) in sample_idx.iter().enumerate() {
            acc += d2[i] as f64;
            if acc >= target { chosen = si; break; }
        }
        centroids.push(points[chosen]);
    }
    centroids
}

fn lloyd(points: &[[f32; 3]], centroids: &mut Vec<[f32; 3]>, max_iter: usize, rng: &mut Rng) {
    let k = centroids.len();
    let n = points.len();
    if n == 0 { return; }
    let mut sums = vec![[0f64; 3]; k];
    let mut counts = vec![0u64; k];
    for _iter in 0..max_iter {
        for s in sums.iter_mut() { *s = [0.0; 3]; }
        for c in counts.iter_mut() { *c = 0; }
        for p in points.iter() {
            let a = assign(p, centroids) as usize;
            sums[a][0] += p[0] as f64;
            sums[a][1] += p[1] as f64;
            sums[a][2] += p[2] as f64;
            counts[a] += 1;
        }
        let mut moved = 0.0f32;
        for i in 0..k {
            if counts[i] > 0 {
                let nc = [
                    (sums[i][0] / counts[i] as f64) as f32,
                    (sums[i][1] / counts[i] as f64) as f32,
                    (sums[i][2] / counts[i] as f64) as f32,
                ];
                moved += dist2(&centroids[i], &nc);
                centroids[i] = nc;
            } else {
                centroids[i] = points[rng.next_usize(n)];
                moved += 1.0;
            }
        }
        if moved < 1e-6 { break; }
    }
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut reader = BufReader::with_capacity(1 << 20, stdin.lock());
    let mut writer = BufWriter::with_capacity(1 << 20, stdout.lock());

    let (num_points, num_clusters) = read_header(&mut reader)?;
    let n = num_points as usize;
    let k = num_clusters as usize;
    let mut rng = Rng::new(0xC0FFEE_u64 ^ (n as u64).wrapping_mul(k as u64 + 1));

    if k == 0 {
        return Ok(());
    }

    // Strategy by scale
    let train_size: usize = if n <= 2_000_000 {
        n
    } else if n <= 20_000_000 {
        2_000_000
    } else if n <= 200_000_000 {
        1_500_000
    } else {
        1_000_000
    };
    let train_size = train_size.min(n).max(k);

    let max_iter: usize = if n <= 1_000_000 {
        20
    } else if n <= 10_000_000 {
        15
    } else if n <= 100_000_000 {
        10
    } else {
        6
    };

    if n <= 5_000_000 {
        // Load everything, k-means in memory
        let mut points: Vec<[f32; 3]> = Vec::with_capacity(n);
        let mut buf = [0u8; 12];
        for _ in 0..n {
            match read_point(&mut reader, &mut buf)? {
                Some(p) => points.push(p),
                None => break,
            }
        }
        let mut centroids = kmeans_pp_init(&points, k, &mut rng);
        lloyd(&points, &mut centroids, max_iter, &mut rng);
        for p in points.iter() {
            let idx = assign(p, &centroids);
            write_idx(&mut writer, idx)?;
        }
    } else {
        // Streaming: train on prefix, then mini-batch update on rest
        let mut train_points: Vec<[f32; 3]> = Vec::with_capacity(train_size);
        let mut buf = [0u8; 12];
        let mut read_count: usize = 0;
        while read_count < train_size {
            match read_point(&mut reader, &mut buf)? {
                Some(p) => { train_points.push(p); read_count += 1; }
                None => break,
            }
        }
        let mut centroids = kmeans_pp_init(&train_points, k, &mut rng);
        lloyd(&train_points, &mut centroids, max_iter, &mut rng);

        // Write assignments for buffered training points
        for p in train_points.iter() {
            let idx = assign(p, &centroids);
            write_idx(&mut writer, idx)?;
        }
        drop(train_points);

        // Mini-batch streaming for remainder
        let mut counts = vec![1u64; k];
        let batch_size: usize = if n > 500_000_000 { 65_536 } else { 16_384 };
        let mut batch: Vec<[f32; 3]> = Vec::with_capacity(batch_size);
        let mut batch_assign: Vec<u32> = Vec::with_capacity(batch_size);

        let mut remaining = n.saturating_sub(read_count);
        while remaining > 0 {
            batch.clear();
            batch_assign.clear();
            let take = batch_size.min(remaining);
            let mut got = 0usize;
            for _ in 0..take {
                match read_point(&mut reader, &mut buf)? {
                    Some(p) => { batch.push(p); got += 1; }
                    None => break,
                }
            }
            if got == 0 { break; }

            for p in batch.iter() {
                let a = assign(p, &centroids);
                batch_assign.push(a);
            }
            for &a in batch_assign.iter() {
                write_idx(&mut writer, a)?;
            }
            for (p, &a) in batch.iter().zip(batch_assign.iter()) {
                let ai = a as usize;
                counts[ai] += 1;
                let lr = 1.0f32 / (counts[ai] as f32);
                centroids[ai][0] += lr * (p[0] - centroids[ai][0]);
                centroids[ai][1] += lr * (p[1] - centroids[ai][1]);
                centroids[ai][2] += lr * (p[2] - centroids[ai][2]);
            }
            remaining -= got;
        }
    }

    writer.flush()?;
    Ok(())
}
