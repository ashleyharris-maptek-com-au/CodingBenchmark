#include <cstdio>
#include <cmath>
#include <vector>
#include <algorithm>
#include <chrono>
#include <limits>
#include <cstdint>

using namespace std;

struct V3 {
    double x, y, z;
    V3(double a=0,double b=0,double c=0):x(a),y(b),z(c){}
    V3 operator+(const V3&o)const{return V3(x+o.x,y+o.y,z+o.z);}
    V3 operator-(const V3&o)const{return V3(x-o.x,y-o.y,z-o.z);}
    V3 operator*(double s)const{return V3(x*s,y*s,z*s);}
};

static inline double dot(const V3&a,const V3&b){return a.x*b.x+a.y*b.y+a.z*b.z;}
static inline V3 cross(const V3&a,const V3&b){return V3(a.y*b.z-a.z*b.y,a.z*b.x-a.x*b.z,a.x*b.y-a.y*b.x);}

int main(){
    int V;
    double edge;
    if(scanf("%d %lf",&V,&edge)!=2)return 0;
    vector<V3> pts(V);
    double minx=1e300,miny=1e300,minz=1e300;
    double maxx=-1e300,maxy=-1e300,maxz=-1e300;
    for(int i=0;i<V;i++){
        scanf("%lf %lf %lf",&pts[i].x,&pts[i].y,&pts[i].z);
        minx=min(minx,pts[i].x); maxx=max(maxx,pts[i].x);
        miny=min(miny,pts[i].y); maxy=max(maxy,pts[i].y);
        minz=min(minz,pts[i].z); maxz=max(maxz,pts[i].z);
    }

    double e = edge;
    double R = e * sqrt(3.0/8.0);

    // Conservative AABB packing: shrink bbox by R so each tet's bounding sphere
    // is inside the bbox. Space tet centers 2R apart so spheres don't overlap.
    double sx = minx + R, ex = maxx - R;
    double sy = miny + R, ey = maxy - R;
    double sz = minz + R, ez = maxz - R;

    double spacing = 2.0 * R;

    if(ex < sx || ey < sy || ez < sz){
        printf("0\n");
        return 0;
    }

    int nx = (int)floor((ex - sx)/spacing) + 1;
    int ny = (int)floor((ey - sy)/spacing) + 1;
    int nz = (int)floor((ez - sz)/spacing) + 1;
    if(nx<1)nx=1; if(ny<1)ny=1; if(nz<1)nz=1;

    long long maxCount = (long long)nx * ny * nz;
    long long cap = 2000000LL;
    if(maxCount > cap){
        double factor = pow((double)maxCount/(double)cap, 1.0/3.0);
        spacing *= factor;
        nx = (int)floor((ex - sx)/spacing) + 1;
        ny = (int)floor((ey - sy)/spacing) + 1;
        nz = (int)floor((ez - sz)/spacing) + 1;
        if(nx<1)nx=1; if(ny<1)ny=1; if(nz<1)nz=1;
    }

    auto start = chrono::steady_clock::now();

    vector<double> out;
    out.reserve((size_t)min<long long>(cap, (long long)nx*ny*nz) * 6);

    long long count = 0;
    bool stop = false;
    for(int i=0;i<nx && !stop;i++){
        for(int j=0;j<ny && !stop;j++){
            for(int k=0;k<nz;k++){
                double cx = sx + i*spacing;
                double cy = sy + j*spacing;
                double cz = sz + k*spacing;
                if(cx > ex + 1e-9) continue;
                if(cy > ey + 1e-9) continue;
                if(cz > ez + 1e-9) continue;
                out.push_back(cx);
                out.push_back(cy);
                out.push_back(cz);
                out.push_back(0.0);
                out.push_back(0.0);
                out.push_back(0.0);
                count++;
                if(count >= cap){ stop = true; break; }
            }
        }
        auto now = chrono::steady_clock::now();
        if(chrono::duration_cast<chrono::milliseconds>(now-start).count() > 25000) stop = true;
    }

    printf("%lld\n", count);
    for(long long i=0;i<count;i++){
        printf("%.6f %.6f %.6f %.6f %.6f %.6f\n",
            out[i*6+0],out[i*6+1],out[i*6+2],
            out[i*6+3],out[i*6+4],out[i*6+5]);
    }
    return 0;
}
