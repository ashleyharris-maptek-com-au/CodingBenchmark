using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    static int N;
    static int M;
    static List<int>[] adj;
    static int[] edgeU;
    static int[] edgeV;
    static long[] edgeW;
    static int edgeCount;
    static int edgeCapacity;

    static void EnsureEdgeCapacity(int needed)
    {
        if (needed <= edgeCapacity) return;
        int newCap = Math.Max(edgeCapacity * 2, needed);
        Array.Resize(ref edgeU, newCap);
        Array.Resize(ref edgeV, newCap);
        Array.Resize(ref edgeW, newCap);
        edgeCapacity = newCap;
    }

    static void AddEdge(int u, int v, long w)
    {
        EnsureEdgeCapacity(edgeCount + 1);
        edgeU[edgeCount] = u;
        edgeV[edgeCount] = v;
        edgeW[edgeCount] = w;
        adj[u].Add(edgeCount);
        adj[v].Add(edgeCount);
        edgeCount++;
    }

    static void Main()
    {
        var input = Console.In.ReadToEnd();
        var tokens = input.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int pos = 0;
        N = int.Parse(tokens[pos++]);
        M = int.Parse(tokens[pos++]);
        adj = new List<int>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<int>();
        edgeCapacity = Math.Max(16, M * 2);
        edgeU = new int[edgeCapacity];
        edgeV = new int[edgeCapacity];
        edgeW = new long[edgeCapacity];
        edgeCount = 0;
        for (int i = 0; i < M; i++)
        {
            int u = int.Parse(tokens[pos++]);
            int v = int.Parse(tokens[pos++]);
            long w = long.Parse(tokens[pos++]);
            AddEdge(u, v, w);
        }

        int[] deg = new int[N];
        for (int e = 0; e < edgeCount; e++)
        {
            deg[edgeU[e]]++;
            deg[edgeV[e]]++;
        }
        List<int> oddVerts = new List<int>();
        for (int i = 0; i < N; i++) if ((deg[i] & 1) == 1) oddVerts.Add(i);

        int K = oddVerts.Count;
        if (K > 0)
        {
            if (K <= 18 && N <= 2000)
            {
                ExactMatching(oddVerts);
            }
            else
            {
                GreedyMatching(oddVerts);
            }
        }

        var route = Hierholzer(0);

        var sb = new StringBuilder();
        for (int i = 0; i < route.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(route[i]);
        }
        Console.Out.Write(sb.ToString());
    }

    static void Dijkstra(int src, long[] dist, int[] parentEdge)
    {
        for (int i = 0; i < N; i++) { dist[i] = long.MaxValue; parentEdge[i] = -1; }
        dist[src] = 0;
        var pq = new BinaryHeap();
        pq.Push(0L, src);
        while (pq.Count > 0)
        {
            pq.Pop(out long d, out int u);
            if (d > dist[u]) continue;
            var au = adj[u];
            for (int i = 0; i < au.Count; i++)
            {
                int e = au[i];
                int v = (edgeU[e] == u) ? edgeV[e] : edgeU[e];
                long nd = d + edgeW[e];
                if (nd < dist[v])
                {
                    dist[v] = nd;
                    parentEdge[v] = e;
                    pq.Push(nd, v);
                }
            }
        }
    }

    static void MultiSourceDijkstra(List<int> sources, long[] dist, int[] sourceOf, int[] parentEdge, bool[] isSource)
    {
        for (int i = 0; i < N; i++) { dist[i] = long.MaxValue; sourceOf[i] = -1; parentEdge[i] = -1; }
        var pq = new BinaryHeap();
        foreach (var s in sources)
        {
            if (!isSource[s]) continue;
            dist[s] = 0;
            sourceOf[s] = s;
            pq.Push(0L, s);
        }
        while (pq.Count > 0)
        {
            pq.Pop(out long d, out int u);
            if (d > dist[u]) continue;
            var au = adj[u];
            for (int i = 0; i < au.Count; i++)
            {
                int e = au[i];
                int v = (edgeU[e] == u) ? edgeV[e] : edgeU[e];
                long nd = d + edgeW[e];
                if (nd < dist[v])
                {
                    dist[v] = nd;
                    sourceOf[v] = sourceOf[u];
                    parentEdge[v] = e;
                    pq.Push(nd, v);
                }
            }
        }
    }

    static void ExactMatching(List<int> oddVerts)
    {
        int K = oddVerts.Count;
        long[][] D = new long[K][];
        int[][] parentEdges = new int[K][];
        long[] dist = new long[N];
        int[] parent = new int[N];
        for (int i = 0; i < K; i++)
        {
            Dijkstra(oddVerts[i], dist, parent);
            D[i] = new long[K];
            parentEdges[i] = new int[N];
            for (int j = 0; j < K; j++) D[i][j] = dist[oddVerts[j]];
            Array.Copy(parent, parentEdges[i], N);
        }

        int FULL = 1 << K;
        long[] dp = new long[FULL];
        int[] pairing = new int[FULL];
        for (int i = 0; i < FULL; i++) { dp[i] = long.MaxValue; pairing[i] = -1; }
        dp[0] = 0;
        for (int mask = 0; mask < FULL; mask++)
        {
            if (dp[mask] == long.MaxValue) continue;
            if (mask == FULL - 1) continue;
            int i = 0;
            while ((mask & (1 << i)) != 0) i++;
            for (int j = i + 1; j < K; j++)
            {
                if ((mask & (1 << j)) != 0) continue;
                int nm = mask | (1 << i) | (1 << j);
                long cost = D[i][j];
                if (cost == long.MaxValue) continue;
                long nc = dp[mask] + cost;
                if (nc < dp[nm])
                {
                    dp[nm] = nc;
                    pairing[nm] = j;
                }
            }
        }

        int curMask = FULL - 1;
        var pairs = new List<(int, int)>();
        while (curMask != 0)
        {
            int i = 0;
            while ((curMask & (1 << i)) == 0) i++;
            int j = pairing[curMask];
            if (j < 0) break;
            pairs.Add((i, j));
            curMask &= ~((1 << i) | (1 << j));
        }

        foreach (var (i, j) in pairs)
        {
            int src = oddVerts[i];
            int dst = oddVerts[j];
            int cur = dst;
            var path = new List<int>();
            while (cur != src)
            {
                int e = parentEdges[i][cur];
                if (e < 0) break;
                path.Add(e);
                int other = (edgeU[e] == cur) ? edgeV[e] : edgeU[e];
                cur = other;
            }
            foreach (var e in path)
            {
                AddEdge(edgeU[e], edgeV[e], edgeW[e]);
            }
        }
    }

    static void GreedyMatching(List<int> oddVerts)
    {
        var sourceOf = new int[N];
        var dist = new long[N];
        var parentEdge = new int[N];
        var isSource = new bool[N];
        foreach (var v in oddVerts) isSource[v] = true;

        MultiSourceDijkstra(oddVerts, dist, sourceOf, parentEdge, isSource);

        var candidates = new List<(long cost, int a, int b, int u, int v, int eIdx)>();
        int origEdgeCount = edgeCount;
        for (int e = 0; e < origEdgeCount; e++)
        {
            int u = edgeU[e], v = edgeV[e];
            int su = sourceOf[u], sv = sourceOf[v];
            if (su < 0 || sv < 0) continue;
            if (su == sv) continue;
            long c = dist[u] + edgeW[e] + dist[v];
            candidates.Add((c, su, sv, u, v, e));
        }

        candidates.Sort((x, y) => x.cost.CompareTo(y.cost));

        var matched = new bool[N];
        var pairsToConnect = new List<(int srcOdd, int dstOdd, int u, int v, int eIdx)>();
        foreach (var c in candidates)
        {
            if (matched[c.a] || matched[c.b]) continue;
            matched[c.a] = true;
            matched[c.b] = true;
            pairsToConnect.Add((c.a, c.b, c.u, c.v, c.eIdx));
        }

        var leftovers = new List<int>();
        foreach (var v in oddVerts) if (!matched[v]) leftovers.Add(v);

        foreach (var p in pairsToConnect)
        {
            AddPathFromSourceTo(p.srcOdd, p.u, parentEdge);
            AddEdge(edgeU[p.eIdx], edgeV[p.eIdx], edgeW[p.eIdx]);
            AddPathFromSourceTo(p.dstOdd, p.v, parentEdge);
        }

        while (leftovers.Count >= 2)
        {
            int src = leftovers[0];
            leftovers.RemoveAt(0);
            var d2 = new long[N];
            var par2 = new int[N];
            Dijkstra(src, d2, par2);
            int best = -1; long bestD = long.MaxValue;
            for (int i = 0; i < leftovers.Count; i++)
            {
                int v = leftovers[i];
                if (d2[v] < bestD) { bestD = d2[v]; best = i; }
            }
            if (best < 0) break;
            int target = leftovers[best];
            leftovers.RemoveAt(best);
            int cur = target;
            while (cur != src)
            {
                int e = par2[cur];
                if (e < 0) break;
                int prev = (edgeU[e] == cur) ? edgeV[e] : edgeU[e];
                AddEdge(edgeU[e], edgeV[e], edgeW[e]);
                cur = prev;
            }
        }
    }

    static void AddPathFromSourceTo(int source, int node, int[] parentEdge)
    {
        int cur = node;
        while (cur != source)
        {
            int e = parentEdge[cur];
            if (e < 0) break;
            int other = (edgeU[e] == cur) ? edgeV[e] : edgeU[e];
            AddEdge(edgeU[e], edgeV[e], edgeW[e]);
            cur = other;
        }
    }

    static List<int> Hierholzer(int start)
    {
        bool[] used = new bool[edgeCount];
        int[] idxPtr = new int[N];

        var stack = new Stack<int>();
        var circuit = new List<int>();
        stack.Push(start);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            var au = adj[u];
            while (idxPtr[u] < au.Count && used[au[idxPtr[u]]]) idxPtr[u]++;
            if (idxPtr[u] == au.Count)
            {
                circuit.Add(u);
                stack.Pop();
            }
            else
            {
                int e = au[idxPtr[u]];
                used[e] = true;
                int v = (edgeU[e] == u) ? edgeV[e] : edgeU[e];
                stack.Push(v);
            }
        }

        circuit.Reverse();
        return circuit;
    }

    class BinaryHeap
    {
        long[] keys = new long[16];
        int[] vals = new int[16];
        int n = 0;
        public int Count => n;

        public void Push(long k, int v)
        {
            if (n == keys.Length)
            {
                Array.Resize(ref keys, keys.Length * 2);
                Array.Resize(ref vals, vals.Length * 2);
            }
            keys[n] = k; vals[n] = v;
            int i = n; n++;
            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (keys[p] > keys[i])
                {
                    (keys[p], keys[i]) = (keys[i], keys[p]);
                    (vals[p], vals[i]) = (vals[i], vals[p]);
                    i = p;
                }
                else break;
            }
        }

        public void Pop(out long k, out int v)
        {
            k = keys[0]; v = vals[0];
            n--;
            if (n > 0)
            {
                keys[0] = keys[n]; vals[0] = vals[n];
                int i = 0;
                while (true)
                {
                    int l = 2 * i + 1, r = 2 * i + 2, best = i;
                    if (l < n && keys[l] < keys[best]) best = l;
                    if (r < n && keys[r] < keys[best]) best = r;
                    if (best != i)
                    {
                        (keys[best], keys[i]) = (keys[i], keys[best]);
                        (vals[best], vals[i]) = (vals[i], vals[best]);
                        i = best;
                    }
                    else break;
                }
            }
        }
    }
}
