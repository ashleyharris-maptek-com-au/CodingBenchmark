; SPIR-V
; Version: 1.0
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 12

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4

     %uint_6 = OpConstant %uint 6
      %int_0 = OpConstant %int 0
      %int_2 = OpConstant %int 2
      %int_4 = OpConstant %int 4
    %float_1 = OpConstant %float 1
    %float_0_5 = OpConstant %float 0.5
%float_half_thick = OpConstant %float 0.004
   %float_eps = OpConstant %float 0.00001

%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
  %ptr_in_v4 = OpTypePointer Input %v4float

  %arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
    %inColor = OpVariable %ptr_in_arr_v3 Input
  %ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
 %ptr_out_v4 = OpTypePointer Output %v4float
 %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fn_void
      %entry = OpLabel

      %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
      %p0 = OpLoad %v4float %p0_ptr
      %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
      %p2 = OpLoad %v4float %p2_ptr
      %p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
      %p4 = OpLoad %v4float %p4_ptr

      %c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
      %c0 = OpLoad %v3float %c0_ptr
      %c2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
      %c2 = OpLoad %v3float %c2_ptr
      %c4_ptr = OpAccessChain %ptr_in_v3 %inColor %int_4
      %c4 = OpLoad %v3float %c4_ptr

      %c02_sum = OpFAdd %v3float %c0 %c2
      %c02 = OpVectorTimesScalar %v3float %c02_sum %float_0_5
      %c24_sum = OpFAdd %v3float %c2 %c4
      %c24 = OpVectorTimesScalar %v3float %c24_sum %float_0_5
      %c40_sum = OpFAdd %v3float %c4 %c0
      %c40 = OpVectorTimesScalar %v3float %c40_sum %float_0_5

      %p0_x = OpCompositeExtract %float %p0 0
      %p0_y = OpCompositeExtract %float %p0 1
      %p0_z = OpCompositeExtract %float %p0 2
      %p0_w = OpCompositeExtract %float %p0 3
      %p2_x = OpCompositeExtract %float %p2 0
      %p2_y = OpCompositeExtract %float %p2 1
      %p2_z = OpCompositeExtract %float %p2 2
      %p2_w = OpCompositeExtract %float %p2 3
      %p4_x = OpCompositeExtract %float %p4 0
      %p4_y = OpCompositeExtract %float %p4 1
      %p4_z = OpCompositeExtract %float %p4 2
      %p4_w = OpCompositeExtract %float %p4 3

      ; ===== Edge 0 -> 2 =====
      %e02_dx = OpFSub %float %p2_x %p0_x
      %e02_dy = OpFSub %float %p2_y %p0_y
      %e02_a = OpFMul %float %e02_dx %e02_dx
      %e02_b = OpFMul %float %e02_dy %e02_dy
      %e02_lsq = OpFAdd %float %e02_a %e02_b
      %e02_len = OpExtInst %float %1 Sqrt %e02_lsq
      %e02_len_safe = OpExtInst %float %1 FMax %e02_len %float_eps
      %e02_inv = OpFDiv %float %float_1 %e02_len_safe
      %e02_neg_dy = OpFNegate %float %e02_dy
      %e02_perp_x = OpFMul %float %e02_neg_dy %e02_inv
      %e02_perp_y = OpFMul %float %e02_dx %e02_inv
      %e02_ox = OpFMul %float %e02_perp_x %float_half_thick
      %e02_oy = OpFMul %float %e02_perp_y %float_half_thick

      %v0a_x = OpFAdd %float %p0_x %e02_ox
      %v0a_y = OpFAdd %float %p0_y %e02_oy
      %v0a = OpCompositeConstruct %v4float %v0a_x %v0a_y %p0_z %p0_w
      %op_e02_a = OpAccessChain %ptr_out_v4 %gl_out %int_0
      OpStore %op_e02_a %v0a
      OpStore %outColor %c02
      OpEmitVertex

      %v1a_x = OpFSub %float %p0_x %e02_ox
      %v1a_y = OpFSub %float %p0_y %e02_oy
      %v1a = OpCompositeConstruct %v4float %v1a_x %v1a_y %p0_z %p0_w
      %op_e02_b = OpAccessChain %ptr_out_v4 %gl_out %int_0
      OpStore %op_e02_b %v1a
      OpStore %outColor %c02
      OpEmitVertex

      %v2a_x = OpFAdd %float %p2_x %e02_ox
      %v2a_y = OpFAdd %float %p2_y %e02_oy
      %v2a = OpCompositeConstruct %v4float %v2a_x %v2a_y %p2_z %p2_w
      %op_e02_c = OpAccessChain %ptr_out_v4 %gl_out %int_0
      OpStore %op_e02_c %v2a
      OpStore %outColor %c02
      OpEmitVertex

      %v3a_x = OpFSub %float %p2_x %e02_ox
      %v3a_y = OpFSub %float %p2_y %e02_oy
      %v3a = OpCompositeConstruct %v4float %v3a_x %v3a_y %p2_z %p2_w
      %op_e02_d = OpAccessChain %ptr_out_v4 %gl_out %int_0
      OpStore %op_e02_d %v3a
      OpStore %outColor %c02
      OpEmitVertex
      OpEndPrimitive

      ; ===== Edge 2 -> 4 =====
      %e24_dx = OpFSub %float %p4_x %p2_x
      %e24_dy = OpFSub %float %p4_y %p2_y
      %e24_a = OpFMul %float %e24_dx %e24_dx
      %e24_b = OpFMul %float %e24_dy %e24_dy
      %e24_lsq = OpFAdd %float %e24_a %e24_b
      %e24_len = OpExtInst %float %1 Sqrt %e24_lsq
      %e24_len_safe = OpExtInst %float %1 FMax %e24_len %float_eps
      %e24_inv = OpFDiv %float %float_1 %e24_len_safe
      %e24_neg_dy = OpFNegate %float %e24_dy
      %e24_perp_x = OpFMul %float %e24_neg_dy %e24_inv
      %e24_perp_y = OpFMul %float %e24_dx %e24_inv
      %e24_ox = OpFMul %float %e24_perp_x %float_half_thick
      %e24_oy = OpFMul %float %e24_perp_y %float_half_thick

      %v0b_x = OpFAdd %float %p2_x %e24_ox
      %v0b_y = OpFAdd %float %p2_y %e24_oy
      %v0b = OpCompositeConstruct %v4float %v0b_x %v0b_y %p2_z %p2_w
      %op_e24_a = OpAccessChain %ptr_out_v4 %gl_out %int_0
      OpStore %op_e24_a %v0b
      OpStore %outColor %c24
      OpEmitVertex

      %v1b_x = OpFSub %float %p2_x %e24_ox
      %v1b_y = OpFSub %float %p2_y %e24_oy
      %v1b = OpCompositeConstruct %v4float %v1b_x %v1b_y %p2_z %p2_w
      %op_e24_b = OpAccessChain %ptr_out_v4 %gl_out %int_0
      OpStore %op_e24_b %v1b
      OpStore %outColor %c24
      OpEmitVertex

      %v2b_x = OpFAdd %float %p4_x %e24_ox
      %v2b_y = OpFAdd %float %p4_y %e24_oy
      %v2b = OpCompositeConstruct %v4float %v2b_x %v2b_y %p4_z %p4_w
      %op_e24_c = OpAccessChain %ptr_out_v4 %gl_out %int_0
      OpStore %op_e24_c %v2b
      OpStore %outColor %c24
      OpEmitVertex

      %v3b_x = OpFSub %float %p4_x %e24_ox
      %v3b_y = OpFSub %float %p4_y %e24_oy
      %v3b = OpCompositeConstruct %v4float %v3b_x %v3b_y %p4_z %p4_w
      %op_e24_d = OpAccessChain %ptr_out_v4 %gl_out %int_0
      OpStore %op_e24_d %v3b
      OpStore %outColor %c24
      OpEmitVertex
      OpEndPrimitive

      ; ===== Edge 4 -> 0 =====
      %e40_dx = OpFSub %float %p0_x %p4_x
      %e40_dy = OpFSub %float %p0_y %p4_y
      %e40_a = OpFMul %float %e40_dx %e40_dx
      %e40_b = OpFMul %float %e40_dy %e40_dy
      %e40_lsq = OpFAdd %float %e40_a %e40_b
      %e40_len = OpExtInst %float %1 Sqrt %e40_lsq
      %e40_len_safe = OpExtInst %float %1 FMax %e40_len %float_eps
      %e40_inv = OpFDiv %float %float_1 %e40_len_safe
      %e40_neg_dy = OpFNegate %float %e40_dy
      %e40_perp_x = OpFMul %float %e40_neg_dy %e40_inv
      %e40_perp_y = OpFMul %float %e40_dx %e40_inv
      %e40_ox = OpFMul %float %e40_perp_x %float_half_thick
      %e40_oy = OpFMul %float %e40_perp_y %float_half_thick

      %v0c_x = OpFAdd %float %p4_x %e40_ox
      %v0c_y = OpFAdd %float %p4_y %e40_oy
      %v0c = OpCompositeConstruct %v4float %v0c_x %v0c_y %p4_z %p4_w
      %op_e40_a = OpAccessChain %ptr_out_v4 %gl_out %int_0
      OpStore %op_e40_a %v0c
      OpStore %outColor %c40
      OpEmitVertex

      %v1c_x = OpFSub %float %p4_x %e40_ox
      %v1c_y = OpFSub %float %p4_y %e40_oy
      %v1c = OpCompositeConstruct %v4float %v1c_x %v1c_y %p4_z %p4_w
      %op_e40_b = OpAccessChain %ptr_out_v4 %gl_out %int_0
      OpStore %op_e40_b %v1c
      OpStore %outColor %c40
      OpEmitVertex

      %v2c_x = OpFAdd %float %p0_x %e40_ox
      %v2c_y = OpFAdd %float %p0_y %e40_oy
      %v2c = OpCompositeConstruct %v4float %v2c_x %v2c_y %p0_z %p0_w
      %op_e40_c = OpAccessChain %ptr_out_v4 %gl_out %int_0
      OpStore %op_e40_c %v2c
      OpStore %outColor %c40
      OpEmitVertex

      %v3c_x = OpFSub %float %p0_x %e40_ox
      %v3c_y = OpFSub %float %p0_y %e40_oy
      %v3c = OpCompositeConstruct %v4float %v3c_x %v3c_y %p0_z %p0_w
      %op_e40_d = OpAccessChain %ptr_out_v4 %gl_out %int_0
      OpStore %op_e40_d %v3c
      OpStore %outColor %c40
      OpEmitVertex
      OpEndPrimitive

      OpReturn
      OpFunctionEnd
