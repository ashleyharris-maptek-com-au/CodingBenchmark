; SPIR-V
; Version: 1.0
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
     %fnvoid = OpTypeFunction %void
      %float = OpTypeFloat 32
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4

     %uint_4 = OpConstant %uint 4
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
%float_thick = OpConstant %float 0.025

%gl_PerVertex_in  = OpTypeStruct %v4float
   %arr_pv_in     = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv    = OpTypePointer Input %arr_pv_in
       %gl_in     = OpVariable %ptr_in_arr_pv Input
   %ptr_in_v4     = OpTypePointer Input %v4float

   %arr_v3_in     = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3    = OpTypePointer Input %arr_v3_in
   %ptr_in_v3     = OpTypePointer Input %v3float
   %worldPos      = OpVariable %ptr_in_arr_v3 Input
     %normal      = OpVariable %ptr_in_arr_v3 Input
      %color      = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
   %ptr_out_pv    = OpTypePointer Output %gl_PerVertex_out
      %gl_out     = OpVariable %ptr_out_pv Output
   %ptr_out_v4    = OpTypePointer Output %v4float

   %ptr_out_v3    = OpTypePointer Output %v3float
   %outColor      = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fnvoid
      %entry = OpLabel

      %p0ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0ptr
      %p1ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %p1ptr
      %p2ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
         %p2 = OpLoad %v4float %p2ptr
      %p3ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
         %p3 = OpLoad %v4float %p3ptr

       %p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
       %p1xy = OpVectorShuffle %v2float %p1 %p1 0 1
       %p2xy = OpVectorShuffle %v2float %p2 %p2 0 1
       %p3xy = OpVectorShuffle %v2float %p3 %p3 0 1

         %d0 = OpFSub %v2float %p2xy %p0xy
         %t0 = OpExtInst %v2float %1 Normalize %d0
         %d1 = OpFSub %v2float %p3xy %p1xy
         %t1 = OpExtInst %v2float %1 Normalize %d1

        %t0x = OpCompositeExtract %float %t0 0
        %t0y = OpCompositeExtract %float %t0 1
       %nt0y = OpFNegate %float %t0y
      %perp0 = OpCompositeConstruct %v2float %nt0y %t0x

        %t1x = OpCompositeExtract %float %t1 0
        %t1y = OpCompositeExtract %float %t1 1
       %nt1y = OpFNegate %float %t1y
      %perp1 = OpCompositeConstruct %v2float %nt1y %t1x

       %off0 = OpVectorTimesScalar %v2float %perp0 %float_thick
       %off1 = OpVectorTimesScalar %v2float %perp1 %float_thick

      %c1ptr = OpAccessChain %ptr_in_v3 %color %int_1
         %c1 = OpLoad %v3float %c1ptr
      %c2ptr = OpAccessChain %ptr_in_v3 %color %int_2
         %c2 = OpLoad %v3float %c2ptr

        %p1z = OpCompositeExtract %float %p1 2
        %p1w = OpCompositeExtract %float %p1 3
        %p2z = OpCompositeExtract %float %p2 2
        %p2w = OpCompositeExtract %float %p2 3

       %v0xy = OpFAdd %v2float %p1xy %off0
        %v0x = OpCompositeExtract %float %v0xy 0
        %v0y = OpCompositeExtract %float %v0xy 1
         %v0 = OpCompositeConstruct %v4float %v0x %v0y %p1z %p1w
     %glout0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %glout0 %v0
               OpStore %outColor %c1
               OpEmitVertex

       %v1xy = OpFSub %v2float %p1xy %off0
        %v1x = OpCompositeExtract %float %v1xy 0
        %v1y = OpCompositeExtract %float %v1xy 1
         %v1 = OpCompositeConstruct %v4float %v1x %v1y %p1z %p1w
     %glout1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %glout1 %v1
               OpStore %outColor %c1
               OpEmitVertex

       %v2xy = OpFAdd %v2float %p2xy %off1
        %v2x = OpCompositeExtract %float %v2xy 0
        %v2y = OpCompositeExtract %float %v2xy 1
        %v2v = OpCompositeConstruct %v4float %v2x %v2y %p2z %p2w
     %glout2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %glout2 %v2v
               OpStore %outColor %c2
               OpEmitVertex

       %v3xy = OpFSub %v2float %p2xy %off1
        %v3x = OpCompositeExtract %float %v3xy 0
        %v3y = OpCompositeExtract %float %v3xy 1
        %v3v = OpCompositeConstruct %v4float %v3x %v3y %p2z %p2w
     %glout3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %glout3 %v3v
               OpStore %outColor %c2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
