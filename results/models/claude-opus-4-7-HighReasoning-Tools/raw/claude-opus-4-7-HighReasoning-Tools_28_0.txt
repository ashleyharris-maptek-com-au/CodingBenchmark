use std::io::{self, Read, Write, BufWriter};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut it = input.split_ascii_whitespace();
    let n: usize = match it.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => { writeln!(out, "0").unwrap(); writeln!(out).unwrap(); return; }
    };
    let m: usize = it.next().map(|s| s.parse().unwrap_or(0)).unwrap_or(0);

    let mut edges: Vec<(u32, u32)> = Vec::with_capacity(m);
    for _ in 0..m {
        let u: u32 = match it.next() { Some(s) => s.parse().unwrap_or(0), None => 0 };
        let v: u32 = match it.next() { Some(s) => s.parse().unwrap_or(0), None => 0 };
        if u == v { continue; }
        edges.push((u, v));
    }

    let mut deg: Vec<u32> = vec![0u32; n];
    for &(u, v) in &edges {
        deg[u as usize] += 1;
        deg[v as usize] += 1;
    }

    let mut head: Vec<u32> = vec![0u32; n + 1];
    for i in 0..n {
        head[i + 1] = head[i] + deg[i];
    }
    let total: usize = head[n] as usize;
    let mut adj: Vec<u32> = vec![0u32; total];
    let mut cursor: Vec<u32> = head[..n].to_vec();
    for &(u, v) in &edges {
        let cu = cursor[u as usize] as usize;
        adj[cu] = v;
        cursor[u as usize] += 1;
        let cv = cursor[v as usize] as usize;
        adj[cv] = u;
        cursor[v as usize] += 1;
    }
    drop(edges);
    drop(cursor);

    let mut cur_deg: Vec<u32> = deg.clone();
    drop(deg);
    let mut removed: Vec<bool> = vec![false; n];
    let mut in_cover: Vec<bool> = vec![false; n];

    let mut max_d: u32 = 0;
    for i in 0..n {
        if cur_deg[i] > max_d { max_d = cur_deg[i]; }
    }

    let mut buckets: Vec<Vec<u32>> = vec![Vec::new(); (max_d as usize) + 1];
    let mut pos_in_bucket: Vec<u32> = vec![u32::MAX; n];
    let mut bucket_of: Vec<i32> = vec![-1i32; n];
    for i in 0..n {
        if cur_deg[i] >= 2 {
            let d = cur_deg[i] as usize;
            pos_in_bucket[i] = buckets[d].len() as u32;
            bucket_of[i] = d as i32;
            buckets[d].push(i as u32);
        }
    }

    let mut deg1_q: Vec<u32> = Vec::new();
    for i in 0..n {
        if cur_deg[i] == 1 {
            deg1_q.push(i as u32);
        }
    }

    let mut top: usize = max_d as usize;

    fn remove_from_bucket(
        i: usize,
        buckets: &mut Vec<Vec<u32>>,
        pos_in_bucket: &mut Vec<u32>,
        bucket_of: &mut Vec<i32>,
    ) {
        let d = bucket_of[i];
        if d < 0 { return; }
        let d = d as usize;
        let p = pos_in_bucket[i] as usize;
        let len = buckets[d].len();
        let last = buckets[d][len - 1] as usize;
        if p != len - 1 {
            buckets[d][p] = last as u32;
            pos_in_bucket[last] = p as u32;
        }
        buckets[d].pop();
        pos_in_bucket[i] = u32::MAX;
        bucket_of[i] = -1;
    }

    fn move_bucket(
        i: usize,
        new_d: usize,
        buckets: &mut Vec<Vec<u32>>,
        pos_in_bucket: &mut Vec<u32>,
        bucket_of: &mut Vec<i32>,
    ) {
        let d = bucket_of[i];
        if d >= 0 {
            let d = d as usize;
            let p = pos_in_bucket[i] as usize;
            let len = buckets[d].len();
            let last = buckets[d][len - 1] as usize;
            if p != len - 1 {
                buckets[d][p] = last as u32;
                pos_in_bucket[last] = p as u32;
            }
            buckets[d].pop();
        }
        if new_d >= buckets.len() {
            buckets.resize(new_d + 1, Vec::new());
        }
        pos_in_bucket[i] = buckets[new_d].len() as u32;
        bucket_of[i] = new_d as i32;
        buckets[new_d].push(i as u32);
    }

    loop {
        while let Some(v_u32) = deg1_q.pop() {
            let v = v_u32 as usize;
            if removed[v] || cur_deg[v] != 1 { continue; }
            let start = head[v] as usize;
            let end = head[v + 1] as usize;
            let mut w_opt: Option<usize> = None;
            for k in start..end {
                let x = adj[k] as usize;
                if !removed[x] { w_opt = Some(x); break; }
            }
            let w = match w_opt { Some(x) => x, None => { removed[v] = true; cur_deg[v] = 0; continue; } };
            if !in_cover[w] { in_cover[w] = true; }
            remove_from_bucket(w, &mut buckets, &mut pos_in_bucket, &mut bucket_of);
            removed[w] = true;
            let ws = head[w] as usize;
            let we = head[w + 1] as usize;
            for k in ws..we {
                let y = adj[k] as usize;
                if !removed[y] {
                    if cur_deg[y] > 0 { cur_deg[y] -= 1; }
                    let nd = cur_deg[y] as usize;
                    if nd >= 2 {
                        move_bucket(y, nd, &mut buckets, &mut pos_in_bucket, &mut bucket_of);
                    } else {
                        remove_from_bucket(y, &mut buckets, &mut pos_in_bucket, &mut bucket_of);
                        if nd == 1 {
                            deg1_q.push(y as u32);
                        } else {
                            removed[y] = true;
                        }
                    }
                }
            }
            cur_deg[w] = 0;
            removed[v] = true;
            cur_deg[v] = 0;
        }

        while top > 0 && buckets[top].is_empty() {
            top -= 1;
        }
        if top < 2 { break; }

        let w = *buckets[top].last().unwrap() as usize;
        if removed[w] || (cur_deg[w] as usize) != top {
            remove_from_bucket(w, &mut buckets, &mut pos_in_bucket, &mut bucket_of);
            if !removed[w] {
                let d = cur_deg[w] as usize;
                if d >= 2 {
                    move_bucket(w, d, &mut buckets, &mut pos_in_bucket, &mut bucket_of);
                }
            }
            continue;
        }
        if !in_cover[w] { in_cover[w] = true; }
        remove_from_bucket(w, &mut buckets, &mut pos_in_bucket, &mut bucket_of);
        removed[w] = true;
        let ws = head[w] as usize;
        let we = head[w + 1] as usize;
        for k in ws..we {
            let y = adj[k] as usize;
            if !removed[y] {
                if cur_deg[y] > 0 { cur_deg[y] -= 1; }
                let nd = cur_deg[y] as usize;
                if nd >= 2 {
                    move_bucket(y, nd, &mut buckets, &mut pos_in_bucket, &mut bucket_of);
                } else {
                    remove_from_bucket(y, &mut buckets, &mut pos_in_bucket, &mut bucket_of);
                    if nd == 1 {
                        deg1_q.push(y as u32);
                    } else {
                        removed[y] = true;
                    }
                }
            }
        }
        cur_deg[w] = 0;
    }

    // Final safety: ensure all edges covered.
    for u in 0..n {
        let s = head[u] as usize;
        let e = head[u + 1] as usize;
        for k in s..e {
            let v = adj[k] as usize;
            if u < v && !in_cover[u] && !in_cover[v] {
                in_cover[u] = true;
                break;
            }
        }
    }

    let cover: Vec<usize> = (0..n).filter(|&i| in_cover[i]).collect();
    writeln!(out, "{}", cover.len()).unwrap();
    let mut s = String::with_capacity(cover.len() * 8);
    for (i, &v) in cover.iter().enumerate() {
        if i > 0 { s.push(' '); }
        s.push_str(&v.to_string());
    }
    writeln!(out, "{}", s).unwrap();
}
