using System;
using System.Diagnostics;
using System.IO;
using System.Text;

public class QAP
{
    static int N;
    static int[] flow;
    static int[] dist;
    static Stopwatch sw;
    static long timeLimitMs;
    static Random rng = new Random(1234567);

    static long ComputeCost(int[] p)
    {
        long c = 0;
        for (int i = 0; i < N; i++)
        {
            int pi = p[i];
            int fi = i * N;
            int di = pi * N;
            for (int j = 0; j < N; j++)
            {
                c += (long)flow[fi + j] * dist[di + p[j]];
            }
        }
        return c;
    }

    static long SwapDelta(int[] p, int i, int j)
    {
        int pi = p[i], pj = p[j];
        long d = 0;
        int fi = i * N, fj = j * N;
        int dpi = pi * N, dpj = pj * N;
        for (int k = 0; k < N; k++)
        {
            if (k == i || k == j) continue;
            int pk = p[k];
            int dpk = pk * N;
            d += (long)(flow[fi + k] - flow[fj + k]) * (dist[dpj + pk] - dist[dpi + pk]);
            d += (long)(flow[k * N + i] - flow[k * N + j]) * (dist[dpk + pj] - dist[dpk + pi]);
        }
        d += (long)flow[fi + j] * (dist[dpj + pi] - dist[dpi + pj]);
        d += (long)flow[fj + i] * (dist[dpi + pj] - dist[dpj + pi]);
        d += (long)flow[fi + i] * (dist[dpj + pj] - dist[dpi + pi]);
        d += (long)flow[fj + j] * (dist[dpi + pi] - dist[dpj + pj]);
        return d;
    }

    static int[] GreedyConstruct()
    {
        long[] fSum = new long[N];
        long[] dSum = new long[N];
        for (int i = 0; i < N; i++)
        {
            long fs = 0, ds = 0;
            int fi = i * N, di = i * N;
            for (int j = 0; j < N; j++)
            {
                fs += flow[fi + j] + flow[j * N + i];
                ds += dist[di + j] + dist[j * N + i];
            }
            fSum[i] = fs;
            dSum[i] = ds;
        }
        int[] facs = new int[N];
        int[] locs = new int[N];
        for (int i = 0; i < N; i++) { facs[i] = i; locs[i] = i; }
        Array.Sort(facs, (a, b) => fSum[b].CompareTo(fSum[a]));
        Array.Sort(locs, (a, b) => dSum[a].CompareTo(dSum[b]));
        int[] p = new int[N];
        for (int i = 0; i < N; i++) p[facs[i]] = locs[i];
        return p;
    }

    static long ElapsedMs() => sw.ElapsedMilliseconds;

    static long LocalSearch(int[] p, long cost, long deadlineMs)
    {
        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 0; i < N - 1; i++)
            {
                if (ElapsedMs() > deadlineMs) return cost;
                for (int j = i + 1; j < N; j++)
                {
                    long d = SwapDelta(p, i, j);
                    if (d < 0)
                    {
                        (p[i], p[j]) = (p[j], p[i]);
                        cost += d;
                        improved = true;
                    }
                }
            }
        }
        return cost;
    }

    static void Perturb(int[] p, int strength)
    {
        for (int s = 0; s < strength; s++)
        {
            int i = rng.Next(N);
            int j = rng.Next(N);
            if (i != j) (p[i], p[j]) = (p[j], p[i]);
        }
    }

    public static void Main()
    {
        sw = Stopwatch.StartNew();
        var reader = new StreamReader(Console.OpenStandardInput());
        string all = reader.ReadToEnd();
        var tokens = all.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        N = int.Parse(tokens[idx++]);
        flow = new int[N * N];
        dist = new int[N * N];
        for (int i = 0; i < N * N; i++) flow[i] = int.Parse(tokens[idx++]);
        for (int i = 0; i < N * N; i++) dist[i] = int.Parse(tokens[idx++]);

        timeLimitMs = 5 * 60 * 1000 - 15000;

        int[] best = GreedyConstruct();
        long bestCost = ComputeCost(best);

        bestCost = LocalSearch(best, bestCost, timeLimitMs - 5000);

        int strength = Math.Max(2, N / 20);
        while (ElapsedMs() < timeLimitMs - 5000)
        {
            int[] candidate = (int[])best.Clone();
            Perturb(candidate, strength);
            long cc = ComputeCost(candidate);
            cc = LocalSearch(candidate, cc, timeLimitMs - 5000);
            if (cc < bestCost)
            {
                bestCost = cc;
                best = candidate;
                strength = Math.Max(2, N / 20);
            }
            else
            {
                strength = Math.Min(Math.Max(3, N / 2), strength + 1);
            }
        }

        var outSb = new StringBuilder();
        outSb.Append(bestCost).Append('\n');
        for (int i = 0; i < N; i++)
        {
            if (i > 0) outSb.Append(' ');
            outSb.Append(best[i]);
        }
        outSb.Append('\n');
        Console.Out.Write(outSb.ToString());
    }
}
