using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

class OrbitalTSP
{
    static double MU = 398600.4418;

    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
    }

    struct State
    {
        public Vec3 R, V;
        public State(Vec3 r, Vec3 v) { R = r; V = v; }
    }

    static State Propagate(State s0, double dt)
    {
        if (Math.Abs(dt) < 1e-9) return s0;
        double r0 = s0.R.Norm();
        double v0 = s0.V.Norm();
        double vr0 = s0.R.Dot(s0.V) / r0;
        double alpha = 2.0 / r0 - v0 * v0 / MU;
        double sqrtMu = Math.Sqrt(MU);

        double chi;
        if (alpha > 1e-6)
        {
            chi = sqrtMu * dt * alpha;
        }
        else if (alpha < -1e-6)
        {
            double a = 1.0 / alpha;
            double denom = s0.R.Dot(s0.V) + Math.Sign(dt) * Math.Sqrt(-MU * a) * (1.0 - r0 * alpha);
            if (Math.Abs(denom) < 1e-12) denom = 1e-12;
            chi = Math.Sign(dt) * Math.Sqrt(-a) * Math.Log(Math.Max(1e-12, -2.0 * MU * alpha * dt / denom));
        }
        else
        {
            chi = sqrtMu * dt / r0;
        }

        for (int iter = 0; iter < 200; iter++)
        {
            double psi = chi * chi * alpha;
            double c2, c3;
            StumpC(psi, out c2, out c3);
            double r = chi * chi * c2 + vr0 / sqrtMu * chi * (1.0 - psi * c3) + r0 * (1.0 - psi * c2);
            double F = vr0 / sqrtMu * chi * chi * c2 + (1.0 - r0 * alpha) * chi * chi * chi * c3 + r0 * chi - sqrtMu * dt;
            if (Math.Abs(r) < 1e-12) break;
            double dchi = F / r;
            chi -= dchi;
            if (Math.Abs(dchi) < 1e-10) break;
        }

        double psi2 = chi * chi * alpha;
        double c2f, c3f;
        StumpC(psi2, out c2f, out c3f);
        double f = 1.0 - chi * chi / r0 * c2f;
        double g = dt - chi * chi * chi / sqrtMu * c3f;
        Vec3 R1 = s0.R * f + s0.V * g;
        double r1n = R1.Norm();
        double fdot = sqrtMu / (r1n * r0) * chi * (psi2 * c3f - 1.0);
        double gdot = 1.0 - chi * chi / r1n * c2f;
        Vec3 V1 = s0.R * fdot + s0.V * gdot;
        return new State(R1, V1);
    }

    static void StumpC(double psi, out double c2, out double c3)
    {
        if (psi > 1e-6)
        {
            double sp = Math.Sqrt(psi);
            c2 = (1.0 - Math.Cos(sp)) / psi;
            c3 = (sp - Math.Sin(sp)) / Math.Sqrt(psi * psi * psi);
        }
        else if (psi < -1e-6)
        {
            double sp = Math.Sqrt(-psi);
            c2 = (1.0 - Math.Cosh(sp)) / psi;
            c3 = (Math.Sinh(sp) - sp) / Math.Sqrt(-psi * psi * psi);
        }
        else
        {
            c2 = 0.5; c3 = 1.0 / 6.0;
        }
    }

    static bool LambertUV(Vec3 R1, Vec3 R2, double tof, bool shortWay, out Vec3 V1, out Vec3 V2)
    {
        V1 = new Vec3(); V2 = new Vec3();
        double r1 = R1.Norm();
        double r2 = R2.Norm();
        double cosdnu = R1.Dot(R2) / (r1 * r2);
        cosdnu = Math.Max(-1.0, Math.Min(1.0, cosdnu));
        double A = (shortWay ? 1.0 : -1.0) * Math.Sqrt(r1 * r2 * (1.0 + cosdnu));
        if (Math.Abs(A) < 1e-9) return false;

        double psiLow = -4.0 * Math.PI * Math.PI;
        double psiUp = 4.0 * Math.PI * Math.PI;
        double psi = 0.0;
        double c2 = 0.5, c3 = 1.0 / 6.0;
        double sqrtMu = Math.Sqrt(MU);
        double y = 0, chi = 0;

        for (int iter = 0; iter < 200; iter++)
        {
            StumpC(psi, out c2, out c3);
            if (c2 <= 0) { psi += 0.1; continue; }
            y = r1 + r2 + A * (psi * c3 - 1.0) / Math.Sqrt(c2);
            if (A > 0 && y < 0)
            {
                int k = 0;
                while (y < 0 && k < 100)
                {
                    psi += 0.1;
                    StumpC(psi, out c2, out c3);
                    if (c2 <= 0) { k++; continue; }
                    y = r1 + r2 + A * (psi * c3 - 1.0) / Math.Sqrt(c2);
                    k++;
                }
            }
            if (y < 0) return false;
            chi = Math.Sqrt(y / c2);
            double t = (chi * chi * chi * c3 + A * Math.Sqrt(y)) / sqrtMu;
            if (Math.Abs(t - tof) < 1e-5) break;
            if (t <= tof) psiLow = psi;
            else psiUp = psi;
            psi = 0.5 * (psiUp + psiLow);
        }

        double f = 1.0 - y / r1;
        double gdot = 1.0 - y / r2;
        double g = A * Math.Sqrt(y / MU);
        if (Math.Abs(g) < 1e-12) return false;
        V1 = (R2 - R1 * f) * (1.0 / g);
        V2 = (R2 * gdot - R1) * (1.0 / g);
        return double.IsFinite(V1.X) && double.IsFinite(V2.X);
    }

    static double TransferCostDV(State s, State target0, double tStart, double dt)
    {
        State targetAtArrive = Propagate(target0, tStart + dt);
        if (!double.IsFinite(targetAtArrive.R.X)) return double.PositiveInfinity;
        Vec3 V1s, V2s, V1l, V2l;
        bool okS = LambertUV(s.R, targetAtArrive.R, dt, true, out V1s, out V2s);
        bool okL = LambertUV(s.R, targetAtArrive.R, dt, false, out V1l, out V2l);
        double best = double.PositiveInfinity;
        if (okS)
        {
            double dv = (V1s - s.V).Norm() + (targetAtArrive.V - V2s).Norm();
            if (dv < best) best = dv;
        }
        if (okL)
        {
            double dv = (V1l - s.V).Norm() + (targetAtArrive.V - V2l).Norm();
            if (dv < best) best = dv;
        }
        return best;
    }

    static double CostWithFallback(State sA, double tA, State sB, double dt)
    {
        double c = TransferCostDV(sA, sB, tA, dt);
        if (double.IsInfinity(c) || double.IsNaN(c))
        {
            double[] alts = { 800.0, 2400.0, 600.0, 3600.0, 1200.0, 5000.0 };
            foreach (double altDt in alts)
            {
                c = TransferCostDV(sA, sB, tA, altDt);
                if (!double.IsInfinity(c) && !double.IsNaN(c)) return c;
            }
            return 1e6;
        }
        return c;
    }

    static void Main()
    {
        var ci = CultureInfo.InvariantCulture;
        var input = Console.In.ReadToEnd();
        var lines = input.Split(new[] { '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);
        var first = lines[0].Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(first[0], ci);
        if (first.Length > 1) MU = double.Parse(first[1], ci);

        var startToks = lines[1].Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        // Strip leading non-numeric label tokens if present (e.g., "start orbit:")
        int si = 0;
        while (si < startToks.Length && !IsNumeric(startToks[si])) si++;
        State start = new State(
            new Vec3(double.Parse(startToks[si], ci), double.Parse(startToks[si + 1], ci), double.Parse(startToks[si + 2], ci)),
            new Vec3(double.Parse(startToks[si + 3], ci), double.Parse(startToks[si + 4], ci), double.Parse(startToks[si + 5], ci))
        );

        State[] stations = new State[N];
        for (int i = 0; i < N; i++)
        {
            var t = lines[2 + i].Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int sj = 0;
            while (sj < t.Length && !IsNumeric(t[sj])) sj++;
            stations[i] = new State(
                new Vec3(double.Parse(t[sj], ci), double.Parse(t[sj + 1], ci), double.Parse(t[sj + 2], ci)),
                new Vec3(double.Parse(t[sj + 3], ci), double.Parse(t[sj + 4], ci), double.Parse(t[sj + 5], ci))
            );
        }

        double defaultDt = 1500.0;

        int[] best = GreedyNN(start, stations, defaultDt);
        double bestCost = EvalTour(start, stations, best, defaultDt);

        int trials = N <= 10 ? Math.Min(N, 8) : (N <= 30 ? 8 : (N <= 60 ? 4 : 2));
        var rng = new Random(42);
        var sw = System.Diagnostics.Stopwatch.StartNew();
        double totalBudgetMs = N <= 10 ? 10000 : (N <= 20 ? 60000 : (N <= 50 ? 150000 : 240000));

        for (int tt = 0; tt < trials && sw.ElapsedMilliseconds < totalBudgetMs * 0.3; tt++)
        {
            int firstPick = tt < N ? tt : rng.Next(N);
            int[] cand = GreedyFromFirst(start, stations, firstPick, defaultDt);
            double cc = EvalTour(start, stations, cand, defaultDt);
            if (cc < bestCost) { bestCost = cc; best = cand; }
        }

        // 2-opt
        bool improved = true;
        while (improved && sw.ElapsedMilliseconds < totalBudgetMs)
        {
            improved = false;
            for (int i = 0; i < N - 1 && sw.ElapsedMilliseconds < totalBudgetMs; i++)
            {
                for (int j = i + 1; j < N && sw.ElapsedMilliseconds < totalBudgetMs; j++)
                {
                    int[] cand = (int[])best.Clone();
                    Array.Reverse(cand, i, j - i + 1);
                    double cc = EvalTour(start, stations, cand, defaultDt);
                    if (cc < bestCost - 1e-6)
                    {
                        bestCost = cc;
                        best = cand;
                        improved = true;
                    }
                }
            }
            if (N > 30) break;
        }

        Console.WriteLine(string.Join(" ", best));
    }

    static bool IsNumeric(string s)
    {
        if (string.IsNullOrEmpty(s)) return false;
        return double.TryParse(s, NumberStyles.Float, CultureInfo.InvariantCulture, out _);
    }

    static double EvalTour(State start, State[] stations, int[] order, double dt)
    {
        double total = 0;
        State cur = start;
        double t = 0;
        for (int k = 0; k < order.Length; k++)
        {
            int s = order[k];
            double c = CostWithFallback(cur, t, stations[s], dt);
            total += c;
            t += dt;
            cur = Propagate(stations[s], t);
        }
        return total;
    }

    static int[] GreedyNN(State start, State[] stations, double dt)
    {
        int N = stations.Length;
        var visited = new bool[N];
        var order = new int[N];
        State cur = start;
        double t = 0;
        for (int k = 0; k < N; k++)
        {
            int bestIdx = -1; double bestC = double.PositiveInfinity;
            for (int i = 0; i < N; i++)
            {
                if (visited[i]) continue;
                double c = CostWithFallback(cur, t, stations[i], dt);
                if (c < bestC) { bestC = c; bestIdx = i; }
            }
            if (bestIdx < 0)
            {
                for (int i = 0; i < N; i++) if (!visited[i]) { bestIdx = i; break; }
            }
            order[k] = bestIdx;
            visited[bestIdx] = true;
            t += dt;
            cur = Propagate(stations[bestIdx], t);
        }
        return order;
    }

    static int[] GreedyFromFirst(State start, State[] stations, int firstPick, double dt)
    {
        int N = stations.Length;
        var visited = new bool[N];
        var order = new int[N];
        order[0] = firstPick;
        visited[firstPick] = true;
        double t = dt;
        State cur = Propagate(stations[firstPick], t);
        for (int k = 1; k < N; k++)
        {
            int bestIdx = -1; double bestC = double.PositiveInfinity;
            for (int i = 0; i < N; i++)
            {
                if (visited[i]) continue;
                double c = CostWithFallback(cur, t, stations[i], dt);
                if (c < bestC) { bestC = c; bestIdx = i; }
            }
            if (bestIdx < 0)
            {
                for (int i = 0; i < N; i++) if (!visited[i]) { bestIdx = i; break; }
            }
            order[k] = bestIdx;
            visited[bestIdx] = true;
            t += dt;
            cur = Propagate(stations[bestIdx], t);
        }
        return order;
    }
}
