use std::io::{self, Read, Write, BufWriter};

#[derive(Clone, Copy, Debug)]
struct Pt { x: f64, y: f64 }

fn read_input() -> (Vec<Pt>, Vec<Vec<Pt>>) {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_ascii_whitespace();
    let mut next_f = || -> f64 { it.next().unwrap().parse().unwrap() };
    let sv = next_f() as usize;
    let np = next_f() as usize;
    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x = next_f(); let y = next_f();
        stock.push(Pt { x, y });
    }
    let mut pieces = Vec::with_capacity(np);
    for _ in 0..np {
        let pv = next_f() as usize;
        let mut p = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x = next_f(); let y = next_f();
            p.push(Pt { x, y });
        }
        pieces.push(p);
    }
    (stock, pieces)
}

fn bbox(poly: &[Pt]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for p in poly {
        if p.x < minx { minx = p.x; }
        if p.y < miny { miny = p.y; }
        if p.x > maxx { maxx = p.x; }
        if p.y > maxy { maxy = p.y; }
    }
    (minx, miny, maxx, maxy)
}

fn point_in_poly(px: f64, py: f64, poly: &[Pt]) -> bool {
    let n = poly.len();
    let mut inside = false;
    let mut j = n - 1;
    for i in 0..n {
        let xi = poly[i].x; let yi = poly[i].y;
        let xj = poly[j].x; let yj = poly[j].y;
        if ((yi > py) != (yj > py))
            && (px < (xj - xi) * (py - yi) / (yj - yi + 1e-300) + xi)
        {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn rects_overlap(a: (f64, f64, f64, f64), b: (f64, f64, f64, f64)) -> bool {
    !(a.2 <= b.0 || b.2 <= a.0 || a.3 <= b.1 || b.3 <= a.1)
}

struct Stock {
    shelf_y: f64,
    shelf_h: f64,
    shelf_x: f64,
    placed: Vec<(f64, f64, f64, f64)>,
}

fn try_place(st: &Stock, x: f64, y: f64, w: f64, h: f64,
             sw: f64, sh: f64, sminx: f64, sminy: f64,
             stock: &[Pt], margin: f64) -> bool {
    if x + w > sw + 1e-9 { return false; }
    if y + h > sh + 1e-9 { return false; }
    let rect = (x, y, x + w, y + h);
    for &r in &st.placed {
        if rects_overlap(rect, r) { return false; }
    }
    let ax = sminx + x;
    let ay = sminy + y;
    let bx = sminx + x + w;
    let by = sminy + y + h;
    let test_pts = [
        (ax + margin, ay + margin),
        (bx - margin, ay + margin),
        (ax + margin, by - margin),
        (bx - margin, by - margin),
        ((ax + bx) * 0.5, (ay + by) * 0.5),
    ];
    for (px, py) in test_pts {
        if !point_in_poly(px, py, stock) { return false; }
    }
    true
}

fn main() {
    let (stock, pieces) = read_input();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let (sminx, sminy, smaxx, smaxy) = bbox(&stock);
    let sw = smaxx - sminx;
    let sh = smaxy - sminy;

    let np = pieces.len();
    let mut piece_bb: Vec<(f64, f64, f64, f64)> = Vec::with_capacity(np);
    for p in &pieces {
        piece_bb.push(bbox(p));
    }

    let mut order: Vec<usize> = (0..np).collect();
    order.sort_by(|&a, &b| {
        let ha = piece_bb[a].3 - piece_bb[a].1;
        let hb = piece_bb[b].3 - piece_bb[b].1;
        hb.partial_cmp(&ha).unwrap()
    });

    let mut placements: Vec<(usize, f64, f64, f64)> = vec![(0, 0.0, 0.0, 0.0); np];

    let mut stocks: Vec<Stock> = Vec::new();
    stocks.push(Stock { shelf_y: 0.0, shelf_h: 0.0, shelf_x: 0.0, placed: Vec::new() });

    let margin = (sw.max(sh)) * 1e-9 + 1e-12;

    for &idx in &order {
        let (pminx, pminy, pmaxx, pmaxy) = piece_bb[idx];
        let pw = pmaxx - pminx;
        let ph = pmaxy - pminy;

        if pw > sw + 1e-9 || ph > sh + 1e-9 {
            stocks.push(Stock { shelf_y: 0.0, shelf_h: 0.0, shelf_x: 0.0, placed: Vec::new() });
            let si = stocks.len() - 1;
            let tx = sminx - pminx;
            let ty = sminy - pminy;
            placements[idx] = (si, tx, ty, 0.0);
            continue;
        }

        let mut placed = false;
        for si in 0..stocks.len() {
            let (sx, sy_, sh_cur) = {
                let st = &stocks[si];
                (st.shelf_x, st.shelf_y, st.shelf_h)
            };

            if try_place(&stocks[si], sx, sy_, pw, ph, sw, sh, sminx, sminy, &stock, margin) {
                let st = &mut stocks[si];
                let x = sx; let y = sy_;
                st.placed.push((x, y, x + pw, y + ph));
                st.shelf_x = x + pw;
                if ph > st.shelf_h { st.shelf_h = ph; }
                let tx = sminx + x - pminx;
                let ty = sminy + y - pminy;
                placements[idx] = (si, tx, ty, 0.0);
                placed = true;
                break;
            }
            let new_y = sy_ + sh_cur;
            if try_place(&stocks[si], 0.0, new_y, pw, ph, sw, sh, sminx, sminy, &stock, margin) {
                let st = &mut stocks[si];
                st.shelf_y = new_y;
                st.shelf_h = ph;
                st.shelf_x = pw;
                st.placed.push((0.0, new_y, pw, new_y + ph));
                let tx = sminx - pminx;
                let ty = sminy + new_y - pminy;
                placements[idx] = (si, tx, ty, 0.0);
                placed = true;
                break;
            }
        }

        if !placed {
            let mut ns = Stock { shelf_y: 0.0, shelf_h: ph, shelf_x: pw, placed: Vec::new() };
            ns.placed.push((0.0, 0.0, pw, ph));
            stocks.push(ns);
            let si = stocks.len() - 1;
            let tx = sminx - pminx;
            let ty = sminy - pminy;
            placements[idx] = (si, tx, ty, 0.0);
        }
    }

    writeln!(out, "{}", stocks.len()).unwrap();
    for &(si, tx, ty, rot) in &placements {
        writeln!(out, "{} {} {} {}", si, tx, ty, rot).unwrap();
    }
}
