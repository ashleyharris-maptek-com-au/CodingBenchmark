#include <iostream>
#include <vector>
#include <queue>
#include <unordered_set>
#include <unordered_map>
#include <deque>
#include <algorithm>
#include <climits>
#include <cstdint>
#include <cstdlib>

using namespace std;

static int D;
static vector<int> Bounds;

struct VecHash {
    size_t operator()(const vector<int>& v) const noexcept {
        size_t h = 1469598103934665603ull;
        for (int x : v) {
            h ^= (size_t)(uint32_t)x;
            h *= 1099511628211ull;
        }
        return h;
    }
};

static inline int manhattan(const vector<int>& a, const vector<int>& b) {
    int s = 0;
    for (int i = 0; i < D; i++) s += std::abs(a[i] - b[i]);
    return s;
}

// A* from start to goal avoiding `avoid` cells. Returns sequence of (axis,dir).
static vector<pair<int,int>> astar(const vector<int>& start,
                                   const vector<int>& goal,
                                   const unordered_set<vector<int>, VecHash>& avoid,
                                   int maxNodes) {
    if (start == goal) return {};

    struct Node { int f; int g; vector<int> pos; };
    struct Cmp { bool operator()(const Node& a, const Node& b) const { return a.f > b.f; } };

    priority_queue<Node, vector<Node>, Cmp> pq;
    unordered_map<vector<int>, int, VecHash> gscore;
    unordered_map<vector<int>, pair<vector<int>, pair<int,int>>, VecHash> parent;

    gscore[start] = 0;
    pq.push({manhattan(start, goal), 0, start});

    int expanded = 0;
    while (!pq.empty() && expanded < maxNodes) {
        Node cur = pq.top(); pq.pop();
        auto git = gscore.find(cur.pos);
        if (git != gscore.end() && cur.g > git->second) continue;
        if (cur.pos == goal) {
            vector<pair<int,int>> moves;
            vector<int> p = cur.pos;
            while (p != start) {
                auto& pr = parent[p];
                moves.push_back(pr.second);
                p = pr.first;
            }
            reverse(moves.begin(), moves.end());
            return moves;
        }
        expanded++;

        for (int ax = 0; ax < D; ax++) {
            for (int d = -1; d <= 1; d += 2) {
                vector<int> np = cur.pos;
                np[ax] += d;
                if (np[ax] < 0 || np[ax] >= Bounds[ax]) continue;
                if (np != goal && avoid.count(np)) continue;
                int ng = cur.g + 1;
                auto it = gscore.find(np);
                if (it == gscore.end() || ng < it->second) {
                    gscore[np] = ng;
                    parent[np] = {cur.pos, {ax, d}};
                    pq.push({ng + manhattan(np, goal), ng, np});
                }
            }
        }
    }
    return {};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int maxTurns;
    if (!(cin >> D >> maxTurns)) return 0;
    Bounds.assign(D, 0);
    for (int i = 0; i < D; i++) cin >> Bounds[i];

    int snakeLen;
    cin >> snakeLen;
    vector<int> head(D);
    for (int i = 0; i < D; i++) cin >> head[i];

    int F; cin >> F;
    vector<vector<int>> foods(F, vector<int>(D));
    for (int i = 0; i < F; i++)
        for (int j = 0; j < D; j++) cin >> foods[i][j];

    int O; cin >> O;
    unordered_set<vector<int>, VecHash> obstacles;
    obstacles.reserve((size_t)O * 2 + 16);
    for (int i = 0; i < O; i++) {
        vector<int> o(D);
        for (int j = 0; j < D; j++) cin >> o[j];
        obstacles.insert(std::move(o));
    }

    // We only know the head's initial position. Track body as a deque and grow
    // it toward `targetLen` as the snake eats food.
    deque<vector<int>> body;
    body.push_front(head);
    int targetLen = snakeLen;

    unordered_set<vector<int>, VecHash> foodSet;
    foodSet.reserve((size_t)F * 2 + 16);
    for (auto& f : foods) foodSet.insert(f);

    int turn = 0;

    auto nodeBudget = [&]() {
        if (D >= 15) return 40000;
        if (D >= 10) return 80000;
        return 200000;
    };

    while (turn < maxTurns && !foodSet.empty()) {
        vector<int> curHead = body.front();

        // Avoid set = obstacles + body cells (exclude tail, which moves).
        unordered_set<vector<int>, VecHash> avoid = obstacles;
        avoid.reserve(obstacles.size() + body.size() + 8);
        if ((int)body.size() >= 2) {
            auto it = body.begin();
            auto last = body.end(); --last;
            for (; it != last; ++it) avoid.insert(*it);
        }

        // Try foods nearest by Manhattan distance first.
        vector<vector<int>> sortedFoods(foodSet.begin(), foodSet.end());
        sort(sortedFoods.begin(), sortedFoods.end(),
             [&](const vector<int>& a, const vector<int>& b) {
                 return manhattan(curHead, a) < manhattan(curHead, b);
             });

        vector<pair<int,int>> path;
        for (auto& f : sortedFoods) {
            path = astar(curHead, f, avoid, nodeBudget());
            if (!path.empty()) break;
        }

        auto applyMove = [&](int ax, int d) -> bool {
            vector<int> np = body.front();
            np[ax] += d;
            if (np[ax] < 0 || np[ax] >= Bounds[ax]) return false;
            cout << ax << ' ' << d << '\n';
            cout.flush();
            bool ate = foodSet.erase(np) > 0;
            body.push_front(np);
            if (ate) targetLen++;
            while ((int)body.size() > targetLen) body.pop_back();
            turn++;
            return true;
        };

        if (path.empty()) {
            // Fallback: take the safe move that most reduces distance to a food.
            vector<int> nearest;
            int bestD = INT_MAX;
            for (auto& f : foodSet) {
                int dd = manhattan(curHead, f);
                if (dd < bestD) { bestD = dd; nearest = f; }
            }
            vector<tuple<int,int,int>> cand;
            for (int ax = 0; ax < D; ax++) {
                for (int d = -1; d <= 1; d += 2) {
                    vector<int> np = curHead; np[ax] += d;
                    if (np[ax] < 0 || np[ax] >= Bounds[ax]) continue;
                    if (avoid.count(np)) continue;
                    int score = nearest.empty() ? 0 : manhattan(np, nearest);
                    cand.emplace_back(score, ax, d);
                }
            }
            sort(cand.begin(), cand.end());
            bool moved = false;
            for (auto& c : cand) {
                if (applyMove(get<1>(c), get<2>(c))) { moved = true; break; }
            }
            if (!moved) break;
            continue;
        }

        // Walk path; re-plan after each meal (length changed).
        for (auto& mv : path) {
            if (turn >= maxTurns) break;
            vector<int> next = body.front();
            next[mv.first] += mv.second;
            bool selfHit = false;
            if ((int)body.size() >= 2) {
                auto it = body.begin();
                auto last = body.end(); --last;
                for (; it != last; ++it) {
                    if (*it == next) { selfHit = true; break; }
                }
            }
            if (selfHit || obstacles.count(next)) break;
            bool wasFood = foodSet.count(next) > 0;
            if (!applyMove(mv.first, mv.second)) break;
            if (wasFood) break;
        }
    }

    return 0;
}
