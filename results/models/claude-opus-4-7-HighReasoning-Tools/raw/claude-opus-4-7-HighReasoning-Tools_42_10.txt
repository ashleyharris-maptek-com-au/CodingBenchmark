OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 10

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
     %fnvoid = OpTypeFunction %void
      %float = OpTypeFloat 32
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4

      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
     %uint_4 = OpConstant %uint 4

       %f_0    = OpConstant %float 0.0
       %f_025  = OpConstant %float 0.25
       %f_05   = OpConstant %float 0.5
       %f_075  = OpConstant %float 0.75
       %f_1    = OpConstant %float 1.0
       %f_thick= OpConstant %float 0.02
       %f_eps  = OpConstant %float 0.000001

%gl_PerVertex_in  = OpTypeStruct %v4float
%arr_pv_in        = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv    = OpTypePointer Input %arr_pv_in
      %gl_in      = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4        = OpTypePointer Input %v4float

%arr_v3_in        = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3    = OpTypePointer Input %arr_v3_in
     %inColor     = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3        = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv       = OpTypePointer Output %gl_PerVertex_out
     %gl_out      = OpVariable %ptr_out_pv Output
%ptr_out_v4       = OpTypePointer Output %v4float

%ptr_out_v3       = OpTypePointer Output %v3float
    %outColor     = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fnvoid
      %entry = OpLabel

   %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
   %P0     = OpLoad %v4float %p0_ptr
   %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
   %P1     = OpLoad %v4float %p1_ptr
   %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
   %P2     = OpLoad %v4float %p2_ptr
   %p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
   %P3     = OpLoad %v4float %p3_ptr

   %c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
   %C1     = OpLoad %v3float %c1_ptr
   %c2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
   %C2     = OpLoad %v3float %c2_ptr

   %T1 = OpFSub %v4float %P2 %P0
   %T2 = OpFSub %v4float %P3 %P1
   %DPOS = OpFSub %v4float %P2 %P1
   %DCOL = OpFSub %v3float %C2 %C1
   %DT  = OpFSub %v4float %T2 %T1

   ; ===== Point 0: t = 0.0 =====
   %t0x = OpCompositeExtract %float %T1 0
   %t0y = OpCompositeExtract %float %T1 1
   %t0nx_raw = OpFNegate %float %t0y
   %t0ny_raw = OpCopyObject %float %t0x
   %t0aa = OpFMul %float %t0nx_raw %t0nx_raw
   %t0bb = OpFMul %float %t0ny_raw %t0ny_raw
   %t0ll = OpFAdd %float %t0aa %t0bb
   %t0len = OpExtInst %float %1 Sqrt %t0ll
   %t0lensafe = OpFAdd %float %t0len %f_eps
   %t0nx = OpFDiv %float %t0nx_raw %t0lensafe
   %t0ny = OpFDiv %float %t0ny_raw %t0lensafe
   %t0ox = OpFMul %float %t0nx %f_thick
   %t0oy = OpFMul %float %t0ny %f_thick
   %p0x = OpCompositeExtract %float %P1 0
   %p0y = OpCompositeExtract %float %P1 1
   %p0z = OpCompositeExtract %float %P1 2
   %p0w = OpCompositeExtract %float %P1 3
   %v0ax = OpFAdd %float %p0x %t0ox
   %v0ay = OpFAdd %float %p0y %t0oy
   %v0a = OpCompositeConstruct %v4float %v0ax %v0ay %p0z %p0w
   %v0bx = OpFSub %float %p0x %t0ox
   %v0by = OpFSub %float %p0y %t0oy
   %v0b = OpCompositeConstruct %v4float %v0bx %v0by %p0z %p0w

   %out0a = OpAccessChain %ptr_out_v4 %gl_out %int_0
   OpStore %out0a %v0a
   OpStore %outColor %C1
   OpEmitVertex
   %out0b = OpAccessChain %ptr_out_v4 %gl_out %int_0
   OpStore %out0b %v0b
   OpStore %outColor %C1
   OpEmitVertex

   ; ===== Point 1: t = 0.25 =====
   %d1s = OpVectorTimesScalar %v4float %DPOS %f_025
   %pos1 = OpFAdd %v4float %P1 %d1s
   %dt1s = OpVectorTimesScalar %v4float %DT %f_025
   %tan1 = OpFAdd %v4float %T1 %dt1s
   %t1x = OpCompositeExtract %float %tan1 0
   %t1y = OpCompositeExtract %float %tan1 1
   %t1nx_raw = OpFNegate %float %t1y
   %t1ny_raw = OpCopyObject %float %t1x
   %t1aa = OpFMul %float %t1nx_raw %t1nx_raw
   %t1bb = OpFMul %float %t1ny_raw %t1ny_raw
   %t1ll = OpFAdd %float %t1aa %t1bb
   %t1len = OpExtInst %float %1 Sqrt %t1ll
   %t1lensafe = OpFAdd %float %t1len %f_eps
   %t1nx = OpFDiv %float %t1nx_raw %t1lensafe
   %t1ny = OpFDiv %float %t1ny_raw %t1lensafe
   %t1ox = OpFMul %float %t1nx %f_thick
   %t1oy = OpFMul %float %t1ny %f_thick
   %p1x = OpCompositeExtract %float %pos1 0
   %p1y = OpCompositeExtract %float %pos1 1
   %p1z = OpCompositeExtract %float %pos1 2
   %p1w = OpCompositeExtract %float %pos1 3
   %v1ax = OpFAdd %float %p1x %t1ox
   %v1ay = OpFAdd %float %p1y %t1oy
   %v1a = OpCompositeConstruct %v4float %v1ax %v1ay %p1z %p1w
   %v1bx = OpFSub %float %p1x %t1ox
   %v1by = OpFSub %float %p1y %t1oy
   %v1b = OpCompositeConstruct %v4float %v1bx %v1by %p1z %p1w
   %dc1s = OpVectorTimesScalar %v3float %DCOL %f_025
   %col1 = OpFAdd %v3float %C1 %dc1s

   %out1a = OpAccessChain %ptr_out_v4 %gl_out %int_0
   OpStore %out1a %v1a
   OpStore %outColor %col1
   OpEmitVertex
   %out1b = OpAccessChain %ptr_out_v4 %gl_out %int_0
   OpStore %out1b %v1b
   OpStore %outColor %col1
   OpEmitVertex

   ; ===== Point 2: t = 0.5 =====
   %d2s = OpVectorTimesScalar %v4float %DPOS %f_05
   %pos2 = OpFAdd %v4float %P1 %d2s
   %dt2s = OpVectorTimesScalar %v4float %DT %f_05
   %tan2 = OpFAdd %v4float %T1 %dt2s
   %t2x = OpCompositeExtract %float %tan2 0
   %t2y = OpCompositeExtract %float %tan2 1
   %t2nx_raw = OpFNegate %float %t2y
   %t2ny_raw = OpCopyObject %float %t2x
   %t2aa = OpFMul %float %t2nx_raw %t2nx_raw
   %t2bb = OpFMul %float %t2ny_raw %t2ny_raw
   %t2ll = OpFAdd %float %t2aa %t2bb
   %t2len = OpExtInst %float %1 Sqrt %t2ll
   %t2lensafe = OpFAdd %float %t2len %f_eps

   %t2ny = OpFDiv %float %t2ny_raw %t2lensafe
   %t2ox = OpFMul %float %t2nx %f_thick
   %t2oy = OpFMul %float %t2ny %f_thick
   %p2x = OpCompositeExtract %float %pos2 0
   %p2y = OpCompositeExtract %float %pos2 1
   %p2z = OpCompositeExtract %float %pos2 2
   %p2w = OpCompositeExtract %float %pos2 3

   %v2bx = OpFSub %float %p2x %t2ox
   %v2by = OpFSub %float %p2y %t2oy
   %v2b = OpCompositeConstruct %v4float %v2bx %v2by %p2z %p2w
   %dc2s = OpVectorTimesScalar %v3float %DCOL %f_05
   %col2 = OpFAdd %v3float %C1 %dc2s

   %out2a = OpAccessChain %ptr_out_v4 %gl_out %int_0
   OpStore %out2a %v2a
   OpStore %outColor %col2
   OpEmitVertex
   %out2b = OpAccessChain %ptr_out_v4 %gl_out %int_0
   OpStore %out2b %v2b
   OpStore %outColor %col2
   OpEmitVertex

   ; ===== Point 3: t = 0.75 =====
   %d3s = OpVectorTimesScalar %v4float %DPOS %f_075
   %pos3 = OpFAdd %v4float %P1 %d3s
   %dt3s = OpVectorTimesScalar %v4float %DT %f_075

   %t3ox = OpFMul %float %t3nx %f_thick
   %t3oy = OpFMul %float %t3ny %f_thick
   %p3x = OpCompositeExtract %float %pos3 0
   %p3y = OpCompositeExtract %float %pos3 1
   %p3z = OpCompositeExtract %float %pos3 2
   %p3w = OpCompositeExtract %float %pos3 3
   %v3ax = OpFAdd %float %p3x %t3ox
   %v3ay = OpFAdd %float %p3y %t3oy
   %v3a = OpCompositeConstruct %v4float %v3ax %v3ay %p3z %p3w
   %v3bx = OpFSub %float %p3x %t3ox
   %v3by = OpFSub %float %p3y %t3oy
   %v3b = OpCompositeConstruct %v4float %v3bx %v3by %p3z %p3w
   %dc3s = OpVectorTimesScalar %v3float %DCOL %f_075

   %out3a = OpAccessChain %ptr_out_v4 %gl_out %int_0

   %t4bb = OpFMul %float %t4ny_raw %t4ny_raw

   %t4nx = OpFDiv %float %t4nx_raw %t4lensafe
   %t4ny = OpFDiv %float %t4ny_raw %t4lensafe
   %t4ox = OpFMul %float %t4nx %f_thick
   %t4oy = OpFMul %float %t4ny %f_thick
   %p4x = OpCompositeExtract %float %P2 0

   %v4by = OpFSub %float %p4y %t4oy
   %v4b = OpCompositeConstruct %v4float %v4bx %v4by %p4z %p4w

   %out4a = OpAccessChain %ptr_out_v4 %gl_out %int_0
   OpStore %out4a %v4a
   OpStore %outColor %C2

   %out4b = OpAccessChain %ptr_out_v4 %gl_out %int_0
   OpStore %out4b %v4b
   OpStore %outColor %C2
   OpEmitVertex

   OpEndPrimitive
   OpReturn
   OpFunctionEnd
