using namespace std;
using Clock = chrono::steady_clock;

static double elapsed(Clock::time_point t0) {
    return chrono::duration<double>(Clock::now() - t0).count();
}

struct Pt { float x, y; };

static inline double dist(const Pt& a, const Pt& b) {
    double dx = a.x - b.x, dy = a.y - b.y;
    return sqrt(dx*dx + dy*dy);
}
static inline float distf(const Pt& a, const Pt& b) {
    float dx = a.x - b.x, dy = a.y - b.y;
    return sqrtf(dx*dx + dy*dy);
}

static void output_tour(const vector<int>& tour) {
    int N = (int)tour.size();
    int idx0 = 0;
    for (int i = 0; i < N; ++i) if (tour[i] == 0) { idx0 = i; break; }
    string out;
    out.reserve((size_t)N * 8);
    char buf[16];
    for (int i = 0; i < N; ++i) {
        int v = tour[(idx0 + i) % N];
        int len = snprintf(buf, sizeof(buf), "%d", v);
        out.append(buf, len);
        if (i + 1 < N) out.push_back(' ');
    }
    out.push_back('\n');
    fwrite(out.data(), 1, out.size(), stdout);
    fflush(stdout);
}

// Hilbert curve order for very large N
static inline uint64_t hilbert_d(uint32_t order, uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (uint32_t s = order / 2; s > 0; s >>= 1) {
        uint32_t rx = (x & s) ? 1u : 0u;
        uint32_t ry = (y & s) ? 1u : 0u;
        d += (uint64_t)s * (uint64_t)s * ((3u * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) { x = s - 1 - x; y = s - 1 - y; }
            uint32_t t = x; x = y; y = t;
        }
    }
    return d;
}

static vector<int> hilbert_tour(const vector<Pt>& pts) {
    int N = (int)pts.size();
    float minx = pts[0].x, maxx = pts[0].x, miny = pts[0].y, maxy = pts[0].y;
    for (int i = 1; i < N; ++i) {
        if (pts[i].x < minx) minx = pts[i].x;
        if (pts[i].x > maxx) maxx = pts[i].x;
        if (pts[i].y < miny) miny = pts[i].y;
        if (pts[i].y > maxy) maxy = pts[i].y;
    }
    float rx = maxx - minx; if (rx < 1e-9f) rx = 1.0f;
    float ry = maxy - miny; if (ry < 1e-9f) ry = 1.0f;
    const uint32_t ORDER = 1u << 16;
    const float scale = (float)(ORDER - 1);

    vector<pair<uint64_t,int>> kv(N);
    for (int i = 0; i < N; ++i) {
        uint32_t xi = (uint32_t)((pts[i].x - minx) / rx * scale);
        uint32_t yi = (uint32_t)((pts[i].y - miny) / ry * scale);
        if (xi >= ORDER) xi = ORDER - 1;
        if (yi >= ORDER) yi = ORDER - 1;
        kv[i] = { hilbert_d(ORDER, xi, yi), i };
    }
    sort(kv.begin(), kv.end(), [](const pair<uint64_t,int>& a, const pair<uint64_t,int>& b){
        return a.first < b.first;
    });
    vector<int> tour(N);
    for (int i = 0; i < N; ++i) tour[i] = kv[i].second;
    return tour;
}

struct Grid {
    float minx, miny, cell;
    int gw, gh;
    vector<vector<int>> cells;
};

static Grid build_grid(const vector<Pt>& pts, int target_per_cell = 4) {
    Grid g;
    int N = (int)pts.size();
    float minx = pts[0].x, maxx = pts[0].x, miny = pts[0].y, maxy = pts[0].y;
    for (int i = 1; i < N; ++i) {
        if (pts[i].x < minx) minx = pts[i].x;
        if (pts[i].x > maxx) maxx = pts[i].x;
        if (pts[i].y < miny) miny = pts[i].y;
        if (pts[i].y > maxy) maxy = pts[i].y;
    }
    float w = maxx - minx, h = maxy - miny;
    if (w < 1e-6f) w = 1e-6f;
    if (h < 1e-6f) h = 1e-6f;
    double area = (double)w * (double)h;
    double cell_area = area * (double)target_per_cell / (double)N;
    float cell = (float)sqrt(cell_area);
    if (cell <= 0) cell = max(w, h);
    int gw = max(1, (int)ceil(w / cell) + 1);
    int gh = max(1, (int)ceil(h / cell) + 1);
    long long total = (long long)gw * gh;
    long long max_cells = 8LL * N + 1024;
    while (total > max_cells && cell < max(w, h)) {
        cell *= 1.5f;
        gw = max(1, (int)ceil(w / cell) + 1);
        gh = max(1, (int)ceil(h / cell) + 1);
        total = (long long)gw * gh;
    }
    g.minx = minx; g.miny = miny; g.cell = cell; g.gw = gw; g.gh = gh;
    g.cells.assign((size_t)gw * gh, {});
    for (int i = 0; i < N; ++i) {
        int cx = (int)((pts[i].x - minx) / cell);
        int cy = (int)((pts[i].y - miny) / cell);
        if (cx < 0) cx = 0; if (cx >= gw) cx = gw - 1;
        if (cy < 0) cy = 0; if (cy >= gh) cy = gh - 1;
        g.cells[(size_t)cy * gw + cx].push_back(i);
    }
    return g;
}

static int nearest_in_grid(const Grid& g, const vector<Pt>& pts, const vector<uint8_t>& visited, int from) {
    int cx = (int)((pts[from].x - g.minx) / g.cell);
    int cy = (int)((pts[from].y - g.miny) / g.cell);
    if (cx < 0) cx = 0; if (cx >= g.gw) cx = g.gw - 1;
    if (cy < 0) cy = 0; if (cy >= g.gh) cy = g.gh - 1;

    int best = -1;
    float best_d2 = 1e38f;
    int maxR = max(g.gw, g.gh) + 1;
    for (int r = 0; r <= maxR; ++r) {
        int x0 = max(0, cx - r), x1 = min(g.gw - 1, cx + r);
        int y0 = max(0, cy - r), y1 = min(g.gh - 1, cy + r);
        for (int yy = y0; yy <= y1; ++yy) {
            for (int xx = x0; xx <= x1; ++xx) {
                if (r > 0 && xx != cx - r && xx != cx + r && yy != cy - r && yy != cy + r) continue;
                const auto& cell = g.cells[(size_t)yy * g.gw + xx];
                for (int idx : cell) {
                    if (visited[idx]) continue;
                    float dx = pts[idx].x - pts[from].x;
                    float dy = pts[idx].y - pts[from].y;
                    float d2 = dx*dx + dy*dy;
                    if (d2 < best_d2) { best_d2 = d2; best = idx; }
                }
            }
        }
        if (best >= 0) {
            float reach = (float)r * g.cell;
            if (best_d2 <= reach * reach || r >= maxR) return best;
        }
    }
    return best;
}

static vector<int> greedy_nn_grid(const vector<Pt>& pts, int start = 0) {
    int N = (int)pts.size();
    Grid g = build_grid(pts, 4);
    vector<uint8_t> visited(N, 0);
    vector<int> tour;
    tour.reserve(N);
    int cur = start;
    tour.push_back(cur);
    visited[cur] = 1;
    for (int step = 1; step < N; ++step) {
        int nxt = nearest_in_grid(g, pts, visited, cur);
        if (nxt < 0) {
            float best_d2 = 1e38f; int best = -1;
            for (int i = 0; i < N; ++i) if (!visited[i]) {
                float dx = pts[i].x - pts[cur].x;
                float dy = pts[i].y - pts[cur].y;
                float d2 = dx*dx + dy*dy;
                if (d2 < best_d2) { best_d2 = d2; best = i; }
            }
            nxt = best;
        }
        tour.push_back(nxt);
        visited[nxt] = 1;
        cur = nxt;
    }
    return tour;
}

// 2-opt with neighbor lists
static vector<int> two_opt(vector<int> tour, const vector<Pt>& pts, double deadline, Clock::time_point t0) {
    int N = (int)tour.size();
    if (N < 4) return tour;
    vector<int> pos(N);
    for (int i = 0; i < N; ++i) pos[tour[i]] = i;

    int K = min(15, N - 1);
    vector<vector<int>> nn(N);

    Grid g = build_grid(pts, 4);
    for (int i = 0; i < N; ++i) {
        int cx = (int)((pts[i].x - g.minx) / g.cell);
        int cy = (int)((pts[i].y - g.miny) / g.cell);
        if (cx < 0) cx = 0; if (cx >= g.gw) cx = g.gw - 1;
        if (cy < 0) cy = 0; if (cy >= g.gh) cy = g.gh - 1;
        vector<pair<float,int>> cands;
        int r = 0;
        int maxR = max(g.gw, g.gh);
        while ((int)cands.size() < K + 1 && r <= maxR) {
            int x0 = max(0, cx - r), x1 = min(g.gw - 1, cx + r);
            int y0 = max(0, cy - r), y1 = min(g.gh - 1, cy + r);
            for (int yy = y0; yy <= y1; ++yy) {
                for (int xx = x0; xx <= x1; ++xx) {
                    if (r > 0 && xx != cx - r && xx != cx + r && yy != cy - r && yy != cy + r) continue;
                    for (int idx : g.cells[(size_t)yy * g.gw + xx]) {
                        if (idx == i) continue;
                        float dx = pts[idx].x - pts[i].x, dy = pts[idx].y - pts[i].y;
                        cands.push_back({dx*dx+dy*dy, idx});
                    }
                }
            }
            ++r;
        }
        if ((int)cands.size() > K) {
            nth_element(cands.begin(), cands.begin()+K, cands.end(), [](auto&a,auto&b){return a.first<b.first;});
            cands.resize(K);
        }
        sort(cands.begin(), cands.end(), [](auto&a,auto&b){return a.first<b.first;});
        nn[i].resize(cands.size());
        for (size_t k = 0; k < cands.size(); ++k) nn[i][k] = cands[k].second;
        if ((i & 4095) == 0 && elapsed(t0) > deadline) return tour;
    }

    bool improved = true;
    int passes = 0;
    while (improved) {
        improved = false;
        ++passes;
        for (int i = 0; i < N; ++i) {
            if ((i & 1023) == 0 && elapsed(t0) > deadline) return tour;
            int a = tour[i];
            int b = tour[(i + 1) % N];
            float d_ab = distf(pts[a], pts[b]);
            for (int c : nn[a]) {
                if (c == a || c == b) continue;
                int pc = pos[c];
                int d_node = tour[(pc + 1) % N];
                if (d_node == a || c == b) continue;
                float d_ac = distf(pts[a], pts[c]);
                if (d_ac >= d_ab) break;
                float d_cd = distf(pts[c], pts[d_node]);
                float d_bd = distf(pts[b], pts[d_node]);
                if (d_ab + d_cd > d_ac + d_bd + 1e-9f) {
                    int li = i + 1, ri = pc;
                    if (ri < li) ri += N;
                    int len = ri - li + 1;
                    if (len <= 0 || len >= N) continue;
                    for (int s = 0; s < len / 2; ++s) {
                        int p1 = (li + s) % N;
                        int p2 = (ri - s + N) % N;
                        swap(tour[p1], tour[p2]);
                        pos[tour[p1]] = p1;
                        pos[tour[p2]] = p2;
                    }
                    improved = true;
                    break;
                }
            }
        }
        if (passes > 30) break;
    }
    return tour;
}

static vector<int> brute_force(const vector<Pt>& pts) {
    int N = (int)pts.size();
    vector<int> best(N);
    for (int i = 0; i < N; ++i) best[i] = i;
    double best_d = 0;
    for (int i = 0; i < N; ++i) best_d += dist(pts[best[i]], pts[best[(i+1)%N]]);

    vector<int> sub(N - 1);
    for (int i = 0; i < N - 1; ++i) sub[i] = i + 1;
    sort(sub.begin(), sub.end());
    do {
        double d = dist(pts[0], pts[sub[0]]);
        for (int i = 0; i + 1 < (int)sub.size(); ++i) d += dist(pts[sub[i]], pts[sub[i+1]]);
        d += dist(pts[sub.back()], pts[0]);
        if (d < best_d) {
            best_d = d;
            best[0] = 0;
            for (int i = 0; i < (int)sub.size(); ++i) best[i+1] = sub[i];
        }
    } while (next_permutation(sub.begin(), sub.end()));
    return best;
}

int main() {
    auto t0 = Clock::now();

    int N;
    if (scanf("%d", &N) != 1) return 0;
    vector<Pt> pts(N);
    for (int i = 0; i < N; ++i) {
        scanf("%f %f", &pts[i].x, &pts[i].y);
    }

    const double TIME_LIMIT = 280.0;
    vector<int> tour;

    if (N <= 0) return 0;
    if (N == 1) { printf("0\n"); return 0; }
    if (N == 2) { printf("0 1\n"); return 0; }

    if (N <= 11) {
        tour = brute_force(pts);
    } else if (N <= 200000) {
        tour = greedy_nn_grid(pts, 0);
        double remaining = TIME_LIMIT - elapsed(t0) - 5.0;
        if (remaining > 1.0 && N <= 100000) {
            double deadline = elapsed(t0) + remaining;
            tour = two_opt(tour, pts, deadline, t0);
        }
    } else if (N <= 2000000) {
        tour = greedy_nn_grid(pts, 0);
    } else {
        tour = hilbert_tour(pts);
    }

    output_tour(tour);
    return 0;
}
