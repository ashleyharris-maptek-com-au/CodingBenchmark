using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        var stdin = Console.OpenStandardInput();
        var ms = new MemoryStream();
        byte[] buf = new byte[1 << 16];
        int r;
        while ((r = stdin.Read(buf, 0, buf.Length)) > 0) ms.Write(buf, 0, r);
        byte[] data = ms.GetBuffer();
        int dataLen = (int)ms.Length;

        int pos = 0;
        long ReadLong()
        {
            while (pos < dataLen && (data[pos] < (byte)'0' || data[pos] > (byte)'9') && data[pos] != (byte)'-') pos++;
            bool neg = false;
            if (pos < dataLen && data[pos] == (byte)'-') { neg = true; pos++; }
            long v = 0;
            while (pos < dataLen && data[pos] >= (byte)'0' && data[pos] <= (byte)'9')
            {
                v = v * 10 + (data[pos] - (byte)'0');
                pos++;
            }
            return neg ? -v : v;
        }

        int N = (int)ReadLong();
        long stockLength = ReadLong();
        int[] cuts = new int[N];
        for (int i = 0; i < N; i++) cuts[i] = (int)ReadLong();

        // Best-Fit Decreasing using SortedSet keyed by (remaining, stockId).
        int[] idx = new int[N];
        for (int i = 0; i < N; i++) idx[i] = i;
        Array.Sort(idx, (a, b) => cuts[b].CompareTo(cuts[a]));

        var stockRemaining = new List<int>(Math.Min(N, 1024));
        var stockAssignments = new List<List<int>>(Math.Min(N, 1024));

        long MUL = (long)N + 2;
        var sset = new SortedSet<long>();

        for (int k = 0; k < N; k++)
        {
            int ci = idx[k];
            int len = cuts[ci];
            if (len > stockLength)
            {
                int sid0 = stockRemaining.Count;
                stockRemaining.Add(0);
                stockAssignments.Add(new List<int>(1) { ci });
                continue;
            }

            long lowerKey = (long)len * MUL;
            var view = sset.GetViewBetween(lowerKey, long.MaxValue);
            if (view.Count > 0)
            {
                long key = view.Min;
                sset.Remove(key);
                int sid = (int)(key % MUL);
                int newRem = stockRemaining[sid] - len;
                stockRemaining[sid] = newRem;
                stockAssignments[sid].Add(ci);
                long newKey = (long)newRem * MUL + sid;
                sset.Add(newKey);
            }
            else
            {
                int sid = stockRemaining.Count;
                int newRem = (int)stockLength - len;
                stockRemaining.Add(newRem);
                stockAssignments.Add(new List<int>(4) { ci });
                long newKey = (long)newRem * MUL + sid;
                sset.Add(newKey);
            }
        }

        var sb = new StringBuilder();
        sb.Append(stockAssignments.Count).Append('\n');
        for (int i = 0; i < stockAssignments.Count; i++)
        {
            var lst = stockAssignments[i];
            for (int j = 0; j < lst.Count; j++)
            {
                if (j > 0) sb.Append(' ');
                sb.Append(lst[j]);
            }
            sb.Append('\n');
        }

        var stdout = Console.OpenStandardOutput();
        var bytes = Encoding.ASCII.GetBytes(sb.ToString());
        stdout.Write(bytes, 0, bytes.Length);
        stdout.Flush();
    }
}
