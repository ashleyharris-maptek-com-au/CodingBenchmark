; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %color Location 4
               OpDecorate %outColor Location 0

               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

       %void = OpTypeVoid
   %voidFunc = OpTypeFunction %void
      %float = OpTypeFloat 32
       %vec2 = OpTypeVector %float 2
       %vec3 = OpTypeVector %float 3
       %vec4 = OpTypeVector %float 4
       %mat4 = OpTypeMatrix %vec4 4

        %UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
   %_ptr_UBO = OpTypePointer Uniform %UBO
        %ubo = OpVariable %_ptr_UBO Uniform

  %_ptr_in_vec3 = OpTypePointer Input %vec3
  %_ptr_in_vec2 = OpTypePointer Input %vec2
 %_ptr_out_vec4 = OpTypePointer Output %vec4
   %_ptr_u_vec4 = OpTypePointer Uniform %vec4

   %worldPos = OpVariable %_ptr_in_vec3 Input
     %normal = OpVariable %_ptr_in_vec3 Input
         %uv = OpVariable %_ptr_in_vec2 Input
    %tangent = OpVariable %_ptr_in_vec3 Input
      %color = OpVariable %_ptr_in_vec3 Input
   %outColor = OpVariable %_ptr_out_vec4 Output

        %int = OpTypeInt 32 1
      %int_3 = OpConstant %int 3
        %f_0 = OpConstant %float 0.0
        %f_1 = OpConstant %float 1.0
      %f_0_5 = OpConstant %float 0.5
      %f_0_3 = OpConstant %float 0.3
      %sky_r = OpConstant %float 0.4
      %sky_g = OpConstant %float 0.6
      %sky_b = OpConstant %float 1.0
      %gnd_r = OpConstant %float 0.3
      %gnd_g = OpConstant %float 0.15
      %gnd_b = OpConstant %float 0.05
   %skyColor = OpConstantComposite %vec3 %sky_r %sky_g %sky_b
   %gndColor = OpConstantComposite %vec3 %gnd_r %gnd_g %gnd_b

       %main = OpFunction %void None %voidFunc
      %entry = OpLabel

         %n0 = OpLoad %vec3 %normal
          %N = OpExtInst %vec3 %1 Normalize %n0
        %nyP = OpCompositeExtract %float %N 1
        %ny5 = OpFMul %float %nyP %f_0_5
          %t = OpFAdd %float %ny5 %f_0_5

      %tvec0 = OpCompositeConstruct %vec3 %t %t %t
      %sky_t = OpFMul %vec3 %skyColor %tvec0
      %oneMt = OpFSub %float %f_1 %t
      %omvec = OpCompositeConstruct %vec3 %oneMt %oneMt %oneMt
      %gnd_t = OpFMul %vec3 %gndColor %omvec
       %hemi = OpFAdd %vec3 %sky_t %gnd_t

    %lightPP = OpAccessChain %_ptr_u_vec4 %ubo %int_3
     %lightV = OpLoad %vec4 %lightPP
     %lightX = OpCompositeExtract %float %lightV 0
     %lightY = OpCompositeExtract %float %lightV 1
     %lightZ = OpCompositeExtract %float %lightV 2
     %lightP = OpCompositeConstruct %vec3 %lightX %lightY %lightZ
         %wp = OpLoad %vec3 %worldPos
        %ld0 = OpFSub %vec3 %lightP %wp
         %ld = OpExtInst %vec3 %1 Normalize %ld0
         %dd = OpDot %float %N %ld
         %dm = OpExtInst %float %1 FMax %dd %f_0
        %dmv = OpFMul %float %dm %f_0_3
      %diffV = OpCompositeConstruct %vec3 %dmv %dmv %dmv
      %final = OpFAdd %vec3 %hemi %diffV

         %fr = OpCompositeExtract %float %final 0
         %fg = OpCompositeExtract %float %final 1
         %fb = OpCompositeExtract %float %final 2
        %out = OpCompositeConstruct %vec4 %fr %fg %fb %f_1
               OpStore %outColor %out
               OpReturn
               OpFunctionEnd
