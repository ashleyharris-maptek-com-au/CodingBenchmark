use std::io::{self, Read, Write, BufWriter};

#[derive(Clone)]
struct Sample {
    hole_idx: u32,
    sample_idx: u32,
    x: f32,
    y: f32,
    z: f32,
    values: Vec<f32>,
}

struct Hole {
    id: String,
}

fn parse_f32(s: &str) -> f32 {
    s.parse::<f32>().unwrap_or(f32::NAN)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).ok();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut it = input.split_ascii_whitespace();
    let num_holes: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => { writeln!(out, "0").ok(); return; }
    };
    let num_props: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => { writeln!(out, "0").ok(); return; }
    };

    let mut prop_names: Vec<String> = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        if let Some(n) = it.next() {
            prop_names.push(n.to_string());
        } else {
            prop_names.push(String::from("?"));
        }
    }

    let mut holes: Vec<Hole> = Vec::with_capacity(num_holes);
    let mut all_samples: Vec<Sample> = Vec::new();

    for _ in 0..num_holes {
        let hole_id = match it.next() { Some(s) => s.to_string(), None => break };
        let sx = parse_f32(it.next().unwrap_or("0"));
        let sy = parse_f32(it.next().unwrap_or("0"));
        let sz = parse_f32(it.next().unwrap_or("0"));
        let mut dx = parse_f32(it.next().unwrap_or("0"));
        let mut dy = parse_f32(it.next().unwrap_or("0"));
        let mut dz = parse_f32(it.next().unwrap_or("1"));
        let _length = parse_f32(it.next().unwrap_or("0"));
        let nsamp: usize = it.next().and_then(|s| s.parse().ok()).unwrap_or(0);

        let mag = (dx*dx + dy*dy + dz*dz).sqrt();
        if mag > 1e-12 {
            dx /= mag; dy /= mag; dz /= mag;
        } else {
            dz = 1.0;
        }

        let hole_idx = holes.len() as u32;
        holes.push(Hole { id: hole_id });

        for si in 0..nsamp {
            let depth = parse_f32(it.next().unwrap_or("0"));
            let mut vals: Vec<f32> = Vec::with_capacity(num_props);
            for _ in 0..num_props {
                let tok = it.next().unwrap_or("");
                let v = if tok.is_empty() || tok.eq_ignore_ascii_case("nan") || tok == "-" || tok == "?" {
                    f32::NAN
                } else {
                    tok.parse::<f32>().unwrap_or(f32::NAN)
                };
                vals.push(v);
            }
            let x = sx + dx * depth;
            let y = sy + dy * depth;
            let z = sz + dz * depth;
            all_samples.push(Sample {
                hole_idx,
                sample_idx: si as u32,
                x, y, z,
                values: vals,
            });
        }
    }

    let n = all_samples.len();
    if n == 0 {
        writeln!(out, "0").ok();
        return;
    }

    let k_neighbors = if n < 1000 { 8 } else if n < 50_000 { 12 } else { 16 };
    let use_grid = n > 500;

    let mut prop_med = vec![0.0f32; num_props];
    let mut prop_mad = vec![1.0f32; num_props];
    let mut prop_log = vec![false; num_props];

    for p in 0..num_props {
        let mut vals: Vec<f32> = all_samples.iter()
            .map(|s| s.values[p])
            .filter(|v| v.is_finite())
            .collect();
        if vals.is_empty() { continue; }

        let mut all_pos = true;
        let mut min_v = f32::INFINITY;
        let mut max_v = f32::NEG_INFINITY;
        for &v in &vals {
            if v <= 0.0 { all_pos = false; }
            if v < min_v { min_v = v; }
            if v > max_v { max_v = v; }
        }
        if all_pos && min_v > 0.0 && max_v > min_v * 100.0 {
            prop_log[p] = true;
            for v in vals.iter_mut() { *v = v.ln(); }
        }

        vals.sort_by(|a,b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
        let med = vals[vals.len()/2];
        let mut devs: Vec<f32> = vals.iter().map(|v| (v - med).abs()).collect();
        devs.sort_by(|a,b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
        let mad = devs[devs.len()/2].max(1e-6);
        prop_med[p] = med;
        prop_mad[p] = (mad * 1.4826).max(1e-6);
    }

    let mut min_x = f32::INFINITY; let mut min_y = f32::INFINITY; let mut min_z = f32::INFINITY;
    let mut max_x = f32::NEG_INFINITY; let mut max_y = f32::NEG_INFINITY; let mut max_z = f32::NEG_INFINITY;
    for s in &all_samples {
        if s.x.is_finite() {
            if s.x < min_x { min_x = s.x; } if s.x > max_x { max_x = s.x; }
            if s.y < min_y { min_y = s.y; } if s.y > max_y { max_y = s.y; }
            if s.z < min_z { min_z = s.z; } if s.z > max_z { max_z = s.z; }
        }
    }
    if !min_x.is_finite() { min_x = 0.0; max_x = 1.0; min_y = 0.0; max_y = 1.0; min_z = 0.0; max_z = 1.0; }

    let extent_x = (max_x - min_x).max(1e-3);
    let extent_y = (max_y - min_y).max(1e-3);
    let extent_z = (max_z - min_z).max(1e-3);
    let vol = extent_x * extent_y * extent_z;
    let target_cells = (n as f32 / 8.0).max(1.0);
    let cell_size = (vol / target_cells).powf(1.0/3.0).max(0.5);

    let gx = ((extent_x / cell_size) as usize + 1).max(1);
    let gy = ((extent_y / cell_size) as usize + 1).max(1);
    let gz = ((extent_z / cell_size) as usize + 1).max(1);

    let num_cells = gx.saturating_mul(gy).saturating_mul(gz);
    let mut grid: Vec<Vec<u32>> = Vec::new();
    let use_grid = use_grid && num_cells < 50_000_000;
    if use_grid {
        grid.resize(num_cells, Vec::new());
        for (i, s) in all_samples.iter().enumerate() {
            if s.x.is_finite() {
                let cx = (((s.x - min_x) / cell_size) as isize).max(0).min(gx as isize - 1) as usize;
                let cy = (((s.y - min_y) / cell_size) as isize).max(0).min(gy as isize - 1) as usize;
                let cz = (((s.z - min_z) / cell_size) as isize).max(0).min(gz as isize - 1) as usize;
                let ci = cx + cy*gx + cz*gx*gy;
                grid[ci].push(i as u32);
            }
        }
    }

    let mut suspects: Vec<(u32, u32, usize, f32)> = Vec::new();

    for i in 0..n {
        let s = &all_samples[i];
        if !s.x.is_finite() { continue; }

        let mut candidates: Vec<u32> = Vec::new();
        if use_grid {
            let cx0 = (((s.x - min_x) / cell_size) as isize).max(0).min(gx as isize - 1);
            let cy0 = (((s.y - min_y) / cell_size) as isize).max(0).min(gy as isize - 1);
            let cz0 = (((s.z - min_z) / cell_size) as isize).max(0).min(gz as isize - 1);

            let mut radius: isize = 1;
            while candidates.len() < (k_neighbors + 1) && radius <= 5 {
                candidates.clear();
                for dz in -radius..=radius {
                    let cz = cz0 + dz;
                    if cz < 0 || cz >= gz as isize { continue; }
                    for dy in -radius..=radius {
                        let cy = cy0 + dy;
                        if cy < 0 || cy >= gy as isize { continue; }
                        for dx in -radius..=radius {
                            let cx = cx0 + dx;
                            if cx < 0 || cx >= gx as isize { continue; }
                            let ci = cx as usize + cy as usize*gx + cz as usize*gx*gy;
                            for &idx in &grid[ci] {
                                candidates.push(idx);
                            }
                        }
                    }
                }
                radius += 1;
                if candidates.len() > 300 { break; }
            }
        } else {
            candidates = (0..n as u32).collect();
        }

        let mut dist_pairs: Vec<(f32, u32)> = Vec::with_capacity(candidates.len());
        for &j in &candidates {
            if j as usize == i { continue; }
            let o = &all_samples[j as usize];
            if !o.x.is_finite() { continue; }
            let dx = s.x - o.x;
            let dy = s.y - o.y;
            let dz = s.z - o.z;
            let d2 = dx*dx + dy*dy + dz*dz;
            dist_pairs.push((d2, j));
        }
        if dist_pairs.is_empty() { continue; }
        dist_pairs.sort_by(|a,b| a.0.partial_cmp(&b.0).unwrap_or(std::cmp::Ordering::Equal));
        let k = dist_pairs.len().min(k_neighbors);

        for p in 0..num_props {
            let v = s.values[p];
            if !v.is_finite() { continue; }
            let v_t = if prop_log[p] {
                if v > 0.0 { v.ln() } else { continue; }
            } else { v };

            let mut nvals: Vec<f32> = Vec::with_capacity(k);
            for q in 0..k {
                let nj = dist_pairs[q].1 as usize;
                let nv = all_samples[nj].values[p];
                if nv.is_finite() {
                    let nt = if prop_log[p] {
                        if nv > 0.0 { nv.ln() } else { continue; }
                    } else { nv };
                    nvals.push(nt);
                }
            }
            if nvals.len() < 3 { continue; }

            nvals.sort_by(|a,b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
            let nmed = nvals[nvals.len()/2];
            let mut ndev: Vec<f32> = nvals.iter().map(|x| (x - nmed).abs()).collect();
            ndev.sort_by(|a,b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
            let local_mad = (ndev[ndev.len()/2] * 1.4826)
                .max(prop_mad[p] * 0.1)
                .max(1e-6);

            let local_dev = (v_t - nmed).abs() / local_mad;
            let global_dev = (v_t - prop_med[p]).abs() / prop_mad[p];

            if local_dev > 3.5 && global_dev > 2.0 {
                suspects.push((s.hole_idx, s.sample_idx, p, local_dev));
            }
        }
    }

    suspects.sort_by(|a, b| b.3.partial_cmp(&a.3).unwrap_or(std::cmp::Ordering::Equal));

    let cap = (n.max(100)).min(suspects.len());
    let suspects = &suspects[..cap];

    writeln!(out, "{}", suspects.len()).ok();
    for &(hi, si, p, conf) in suspects {
        writeln!(out, "{} {} {} {:.4}", holes[hi as usize].id, si, prop_names[p], conf).ok();
    }
}
