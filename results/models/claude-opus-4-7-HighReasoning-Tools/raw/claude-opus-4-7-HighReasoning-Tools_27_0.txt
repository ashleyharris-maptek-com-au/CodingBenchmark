#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <vector>
#include <algorithm>
#include <random>
#include <chrono>
#include <cstdint>
#include <iostream>
#include <climits>

using namespace std;
using namespace std::chrono;

static int N, M, K;
static vector<vector<int>> adj;
static steady_clock::time_point t_start;
static double TIME_LIMIT_SEC = 270.0;

static inline double elapsed() {
    return duration_cast<duration<double>>(steady_clock::now() - t_start).count();
}

static bool greedyColor(const vector<int>& order, int kLimit, vector<int>& out) {
    out.assign(N, -1);
    vector<int> forbidden(kLimit + 2, -1);
    for (int v : order) {
        for (int u : adj[v]) {
            if (out[u] >= 0 && out[u] < kLimit) forbidden[out[u]] = v;
        }
        int c = -1;
        for (int i = 0; i < kLimit; ++i) {
            if (forbidden[i] != v) { c = i; break; }
        }
        if (c < 0) return false;
        out[v] = c;
    }
    return true;
}

static bool dsaturColor(int kLimit, vector<int>& out) {
    out.assign(N, -1);
    vector<int> sat(N, 0);
    vector<vector<uint8_t>> usedColor(N);
    for (int i = 0; i < N; ++i) usedColor[i].assign(kLimit, 0);
    vector<int> deg(N);
    for (int i = 0; i < N; ++i) deg[i] = (int)adj[i].size();

    int start = 0;
    for (int i = 1; i < N; ++i) if (deg[i] > deg[start]) start = i;

    for (int step = 0; step < N; ++step) {
        int best = -1;
        int bestSat = -1, bestDeg = -1;
        if (step == 0) {
            best = start;
        } else {
            for (int v = 0; v < N; ++v) {
                if (out[v] != -1) continue;
                int s = sat[v];
                if (s > bestSat || (s == bestSat && deg[v] > bestDeg)) {
                    bestSat = s; bestDeg = deg[v]; best = v;
                }
            }
        }
        if (best == -1) return false;
        int c = -1;
        for (int i = 0; i < kLimit; ++i) if (!usedColor[best][i]) { c = i; break; }
        if (c < 0) return false;
        out[best] = c;
        for (int u : adj[best]) {
            if (out[u] == -1 && !usedColor[u][c]) {
                usedColor[u][c] = 1;
                sat[u]++;
            }
        }
        if (elapsed() > TIME_LIMIT_SEC * 0.4) return false;
    }
    return true;
}

static long long countConflicts(const vector<int>& col) {
    long long c = 0;
    for (int u = 0; u < N; ++u) {
        for (int v : adj[u]) if (v > u && col[u] == col[v]) ++c;
    }
    return c;
}

static bool localSearchFix(vector<int>& col, int kLimit, double timeBudget) {
    double deadline = elapsed() + timeBudget;
    if (deadline > TIME_LIMIT_SEC) deadline = TIME_LIMIT_SEC;

    vector<int> confCount(N, 0);
    long long totalConf = 0;
    for (int u = 0; u < N; ++u) {
        int cu = col[u];
        for (int v : adj[u]) if (col[v] == cu) confCount[u]++;
        totalConf += confCount[u];
    }
    totalConf /= 2;
    if (totalConf == 0) return true;

    vector<vector<int>> tabu;
    bool useTabu = ((long long)N * kLimit <= 50000000LL);
    if (useTabu) {
        tabu.assign(N, vector<int>(kLimit, 0));
    }
    mt19937 rng(12345);
    long long iter = 0;
    long long bestConf = totalConf;

    vector<int> ncount(kLimit, 0);
    vector<int> conflictVerts;
    conflictVerts.reserve(N);

    while (totalConf > 0 && elapsed() < deadline) {
        ++iter;
        conflictVerts.clear();
        for (int v = 0; v < N; ++v) if (confCount[v] > 0) conflictVerts.push_back(v);
        if (conflictVerts.empty()) break;

        int bestV = -1, bestC = -1;
        int bestDelta = INT_MAX;
        int candidates = 0;

        int sampleCount = (int)conflictVerts.size();
        if (sampleCount > 200) sampleCount = 200;

        for (int idx = 0; idx < sampleCount; ++idx) {
            int v;
            if ((int)conflictVerts.size() <= 200) v = conflictVerts[idx];
            else v = conflictVerts[rng() % conflictVerts.size()];

            int cv = col[v];
            for (int c = 0; c < kLimit; ++c) ncount[c] = 0;
            for (int u : adj[v]) {
                int cu = col[u];
                if (cu >= 0 && cu < kLimit) ncount[cu]++;
            }
            int currConf = ncount[cv];
            for (int c = 0; c < kLimit; ++c) {
                if (c == cv) continue;
                int delta = ncount[c] - currConf;
                bool aspiration = (totalConf + delta < bestConf);
                if (useTabu && tabu[v][c] > iter && !aspiration) continue;
                if (delta < bestDelta) {
                    bestDelta = delta; bestV = v; bestC = c; candidates = 1;
                } else if (delta == bestDelta) {
                    candidates++;
                    if ((rng() % candidates) == 0) { bestV = v; bestC = c; }
                }
            }
        }

        if (bestV == -1) {
            int v = conflictVerts[rng() % conflictVerts.size()];
            bestV = v;
            bestC = (col[v] + 1 + (int)(rng() % (kLimit - 1))) % kLimit;
            for (int c = 0; c < kLimit; ++c) ncount[c] = 0;
            for (int u : adj[v]) { int cu = col[u]; if (cu>=0 && cu<kLimit) ncount[cu]++; }
            bestDelta = ncount[bestC] - ncount[col[v]];
        }

        int oldC = col[bestV];
        col[bestV] = bestC;

        int dec = 0, inc = 0;
        for (int u : adj[bestV]) {
            if (col[u] == oldC) { confCount[u]--; dec++; }
            else if (col[u] == bestC) { confCount[u]++; inc++; }
        }
        confCount[bestV] = inc;
        totalConf += (inc - dec);
        if (totalConf < bestConf) bestConf = totalConf;

        if (useTabu) tabu[bestV][oldC] = (int)(iter + 10 + (rng() % 10) + (totalConf / 4));

        if ((iter & 1023) == 0 && elapsed() > deadline) break;
    }

    return totalConf == 0;
}

int main() {
    t_start = steady_clock::now();
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> N >> M >> K)) return 0;
    adj.assign(N, {});
    for (int i = 0; i < M; ++i) {
        int u, v;
        if (!(cin >> u >> v)) break;
        if (u == v) continue;
        if (u < 0 || v < 0 || u >= N || v >= N) continue;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    for (int i = 0; i < N; ++i) {
        auto& a = adj[i];
        sort(a.begin(), a.end());
        a.erase(unique(a.begin(), a.end()), a.end());
        a.erase(remove(a.begin(), a.end(), i), a.end());
    }

    if (K <= 0) {
        for (int i = 0; i < N; ++i) cout << 0 << (i+1==N?'\n':' ');
        return 0;
    }

    vector<int> order(N);
    for (int i = 0; i < N; ++i) order[i] = i;
    sort(order.begin(), order.end(), [&](int a, int b){
        return adj[a].size() > adj[b].size();
    });

    vector<int> best;
    vector<int> tmp;
    if (greedyColor(order, K, tmp)) {
        best = tmp;
    }

    if ((best.empty() || true) && N <= 20000 && elapsed() < TIME_LIMIT_SEC * 0.4) {
        vector<int> dscol;
        if (dsaturColor(K, dscol)) {
            if (best.empty()) best = dscol;
            else {
                int mxa = 0, mxb = 0;
                for (int c : best) mxa = max(mxa, c);
                for (int c : dscol) mxb = max(mxb, c);
                if (mxb < mxa) best = dscol;
            }
        }
    }

    mt19937 rng(2026);
    while (best.empty() && elapsed() < TIME_LIMIT_SEC * 0.5) {
        shuffle(order.begin(), order.end(), rng);
        if (greedyColor(order, K, tmp)) {
            best = tmp;
            break;
        }
    }

    if (best.empty()) {
        int hiK = max(K, 1024);
        vector<int> ord2(N);
        for (int i = 0; i < N; ++i) ord2[i] = i;
        sort(ord2.begin(), ord2.end(), [&](int a, int b){ return adj[a].size() > adj[b].size(); });
        greedyColor(ord2, hiK, tmp);
        for (int& c : tmp) if (c >= K || c < 0) c = ((c % K) + K) % K;
        best = tmp;
        localSearchFix(best, K, TIME_LIMIT_SEC - elapsed() - 5.0);
    }

    long long conf = countConflicts(best);
    if (conf > 0 && elapsed() < TIME_LIMIT_SEC - 2.0) {
        localSearchFix(best, K, TIME_LIMIT_SEC - elapsed() - 2.0);
    }

    for (int i = 0; i < N; ++i) {
        if (best[i] < 0 || best[i] >= K) best[i] = 0;
    }

    for (int i = 0; i < N; ++i) {
        cout << best[i];
        cout << (i + 1 == N ? '\n' : ' ');
    }
    return 0;
}
