; SPIR-V
; Version: 1.0
               OpCapability Shader
       %glsl = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %color Location 4
               OpDecorate %outColor Location 0

               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

       %void = OpTypeVoid
     %fnvoid = OpTypeFunction %void
      %float = OpTypeFloat 32
       %bool = OpTypeBool
        %v2f = OpTypeVector %float 2
        %v3f = OpTypeVector %float 3
        %v4f = OpTypeVector %float 4
        %m4f = OpTypeMatrix %v4f 4

        %UBO = OpTypeStruct %m4f %m4f %m4f %v4f %v4f %v4f
   %ptr_UBO  = OpTypePointer Uniform %UBO
        %ubo = OpVariable %ptr_UBO Uniform

  %ptr_in_v3 = OpTypePointer Input %v3f
  %ptr_in_v2 = OpTypePointer Input %v2f
 %ptr_out_v4 = OpTypePointer Output %v4f
%ptr_unif_v4 = OpTypePointer Uniform %v4f

   %worldPos = OpVariable %ptr_in_v3 Input
     %normal = OpVariable %ptr_in_v3 Input
         %uv = OpVariable %ptr_in_v2 Input
    %tangent = OpVariable %ptr_in_v3 Input
      %color = OpVariable %ptr_in_v3 Input
   %outColor = OpVariable %ptr_out_v4 Output

        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_3 = OpConstant %int 3

     %f_zero = OpConstant %float 0.0
      %f_one = OpConstant %float 1.0
   %f_pifreq = OpConstant %float 50.26544
       %f_p4 = OpConstant %float 0.4
       %f_p6 = OpConstant %float 0.6
     %f_r_hi = OpConstant %float 0.9
     %f_r_lo = OpConstant %float 0.1

        %red = OpConstantComposite %v3f %f_r_hi %f_r_lo %f_r_lo
       %blue = OpConstantComposite %v3f %f_r_lo %f_r_lo %f_r_hi

       %main = OpFunction %void None %fnvoid
      %label = OpLabel

        %uvv = OpLoad %v2f %uv
        %uvy = OpCompositeExtract %float %uvv 1
        %arg = OpFMul %float %uvy %f_pifreq
         %sv = OpExtInst %float %glsl Sin %arg

        %nrm = OpLoad %v3f %normal
         %nn = OpExtInst %v3f %glsl Normalize %nrm

     %lp_ptr = OpAccessChain %ptr_unif_v4 %ubo %int_3
        %lp4 = OpLoad %v4f %lp_ptr
        %lp3 = OpVectorShuffle %v3f %lp4 %lp4 0 1 2

         %wp = OpLoad %v3f %worldPos
       %ldir = OpFSub %v3f %lp3 %wp
         %ln = OpExtInst %v3f %glsl Normalize %ldir

       %ddot = OpDot %float %nn %ln
     %dclamp = OpExtInst %float %glsl FMax %ddot %f_zero

    %dscaled = OpFMul %float %dclamp %f_p6
    %lambert = OpFAdd %float %dscaled %f_p4

      %isRed = OpFOrdGreaterThanEqual %bool %sv %f_zero
     %stripe = OpSelect %v3f %isRed %red %blue

   %shaded_x = OpCompositeExtract %float %stripe 0
   %shaded_y = OpCompositeExtract %float %stripe 1
   %shaded_z = OpCompositeExtract %float %stripe 2
         %sx = OpFMul %float %shaded_x %lambert
         %sy = OpFMul %float %shaded_y %lambert
         %sz = OpFMul %float %shaded_z %lambert

      %final = OpCompositeConstruct %v4f %sx %sy %sz %f_one
               OpStore %outColor %final
               OpReturn
               OpFunctionEnd
