use std::io::{self, Read, Write, BufWriter};

struct Solver {
    w: i32,
    h: i32,
    n: usize,
    blocked: Vec<bool>,
    visited: Vec<bool>,
    path: Vec<usize>,
    require_cycle: bool,
    start: usize,
    deadline: std::time::Instant,
    time_limit: std::time::Duration,
    nodes: u64,
    timed_out: bool,
}

#[inline]
fn idx(x: i32, y: i32, w: i32) -> usize { (y as usize) * (w as usize) + (x as usize) }

impl Solver {
    fn in_bounds(&self, x: i32, y: i32) -> bool {
        x >= 0 && y >= 0 && x < self.w && y < self.h
    }

    fn neighbors(&self, p: usize) -> [Option<usize>; 4] {
        let x = (p % (self.w as usize)) as i32;
        let y = (p / (self.w as usize)) as i32;
        let mut out = [None; 4];
        let dirs = [(1i32,0i32),(-1,0),(0,1),(0,-1)];
        for (i,(dx,dy)) in dirs.iter().enumerate() {
            let nx = x + dx; let ny = y + dy;
            if self.in_bounds(nx, ny) {
                let q = idx(nx, ny, self.w);
                if !self.blocked[q] {
                    out[i] = Some(q);
                }
            }
        }
        out
    }

    fn free_deg(&self, p: usize) -> usize {
        let mut c = 0;
        for n in self.neighbors(p).iter().flatten() {
            if !self.visited[*n] { c += 1; }
        }
        c
    }

    // Connectivity + dead-end pruning on the unvisited subgraph reachable from `cur`'s unvisited neighbors.
    fn prune_ok(&self, cur: usize, remaining: usize) -> bool {
        if remaining == 0 { return true; }
        let mut stack: Vec<usize> = Vec::with_capacity(remaining);
        let mut seen = vec![false; self.n];
        // start BFS from unvisited neighbors of cur
        let mut started = false;
        for n in self.neighbors(cur).iter().flatten() {
            let nn = *n;
            if !self.visited[nn] && !seen[nn] {
                seen[nn] = true;
                stack.push(nn);
                started = true;
            }
        }
        if !started { return false; }
        let mut count = 0usize;
        let mut dead_ends = 0usize;
        // We will track dead-end cells (deg==1 in unvisited subgraph, excluding cur).
        // The path has exactly one remaining endpoint, so >=2 dead ends means infeasible
        // (unless cycle mode, then 0 allowed except possibly adjacent to start).
        let mut nodes_in_comp: Vec<usize> = Vec::with_capacity(remaining);
        while let Some(p) = stack.pop() {
            count += 1;
            nodes_in_comp.push(p);
            for n in self.neighbors(p).iter().flatten() {
                let nn = *n;
                if !self.visited[nn] && !seen[nn] {
                    seen[nn] = true;
                    stack.push(nn);
                }
            }
        }
        if count != remaining { return false; }
        // dead-end check
        for &p in &nodes_in_comp {
            let mut deg = 0usize;
            for n in self.neighbors(p).iter().flatten() {
                if !self.visited[*n] { deg += 1; }
            }
            if deg <= 1 {
                if deg == 0 { return false; }
                dead_ends += 1;
                if dead_ends > 1 { return false; }
                if self.require_cycle {
                    let x = (p % (self.w as usize)) as i32;
                    let y = (p / (self.w as usize)) as i32;
                    let sx = (self.start % (self.w as usize)) as i32;
                    let sy = (self.start / (self.w as usize)) as i32;
                    if (x-sx).abs() + (y-sy).abs() != 1 { return false; }
                }
            }
        }
        true
    }

    fn dfs(&mut self, cur: usize, remaining: usize) -> bool {
        if self.nodes & 0xFFFF == 0 {
            if self.deadline.elapsed() >= self.time_limit {
                self.timed_out = true;
                return false;
            }
        }
        self.nodes += 1;

        if remaining == 0 {
            if self.require_cycle {
                let x = (cur % (self.w as usize)) as i32;
                let y = (cur / (self.w as usize)) as i32;
                let sx = (self.start % (self.w as usize)) as i32;
                let sy = (self.start / (self.w as usize)) as i32;
                return (x - sx).abs() + (y - sy).abs() == 1;
            } else {
                return true;
            }
        }

        if !self.prune_ok(cur, remaining) { return false; }

        let nbrs = self.neighbors(cur);
        let mut cand: [(usize, usize); 4] = [(usize::MAX, 0); 4];
        let mut k = 0;
        for n in nbrs.iter().flatten() {
            let nn = *n;
            if !self.visited[nn] {
                let d = self.free_deg(nn);
                cand[k] = (nn, d);
                k += 1;
            }
        }
        if k == 0 { return false; }
        let slice = &mut cand[..k];
        slice.sort_by_key(|x| x.1);

        for i in 0..k {
            let (nn, _d) = cand[i];
            self.visited[nn] = true;
            self.path.push(nn);
            if self.dfs(nn, remaining - 1) { return true; }
            self.path.pop();
            self.visited[nn] = false;
            if self.timed_out { return false; }
        }
        false
    }
}

fn parity_feasible(w: i32, h: i32, blocked: &[bool], require_cycle: bool, start: usize) -> bool {
    let mut a = 0i64; let mut b = 0i64;
    for y in 0..h {
        for x in 0..w {
            let p = idx(x,y,w);
            if !blocked[p] {
                if (x+y) & 1 == 0 { a += 1; } else { b += 1; }
            }
        }
    }
    let f = a + b;
    let sx = (start as i32) % w;
    let sy = (start as i32) / w;
    let sp = (sx + sy) & 1;
    let (need_s, need_o) = if f % 2 == 1 {
        ((f+1)/2, f/2)
    } else {
        (f/2, f/2)
    };
    let (cs, co) = if sp == 0 { (a, b) } else { (b, a) };
    if cs != need_s || co != need_o { return false; }
    if require_cycle {
        if f % 2 != 0 { return false; }
        if a != b { return false; }
    }
    true
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut it = input.split_ascii_whitespace();
    let w: i32 = it.next().unwrap().parse().unwrap();
    let h: i32 = it.next().unwrap().parse().unwrap();
    let obstacle_count: usize = it.next().unwrap().parse().unwrap();
    let require_cycle_raw: i32 = it.next().unwrap().parse().unwrap();
    let require_cycle = require_cycle_raw != 0;

    let n = (w as usize) * (h as usize);
    let mut blocked = vec![false; n];
    for _ in 0..obstacle_count {
        let x: i32 = it.next().unwrap().parse().unwrap();
        let y: i32 = it.next().unwrap().parse().unwrap();
        if x >= 0 && y >= 0 && x < w && y < h {
            blocked[idx(x,y,w)] = true;
        }
    }

    let start = idx(0, 0, w);
    if blocked[start] {
        writeln!(out, "0").unwrap();
        return;
    }

    let free_count: usize = blocked.iter().filter(|b| !**b).count();

    if free_count == 0 {
        writeln!(out, "0").unwrap();
        return;
    }
    if free_count == 1 {
        if require_cycle {
            writeln!(out, "0").unwrap();
        } else {
            writeln!(out, "1").unwrap();
            writeln!(out, "0 0").unwrap();
        }
        return;
    }

    if !parity_feasible(w, h, &blocked, require_cycle, start) {
        writeln!(out, "0").unwrap();
        return;
    }

    {
        let mut seen = vec![false; n];
        let mut stack = vec![start];
        seen[start] = true;
        let mut c = 1usize;
        while let Some(p) = stack.pop() {
            let x = (p % (w as usize)) as i32;
            let y = (p / (w as usize)) as i32;
            for (dx,dy) in [(1i32,0i32),(-1,0),(0,1),(0,-1)] {
                let nx = x+dx; let ny = y+dy;
                if nx>=0 && ny>=0 && nx<w && ny<h {
                    let q = idx(nx,ny,w);
                    if !blocked[q] && !seen[q] {
                        seen[q] = true; c += 1; stack.push(q);
                    }
                }
            }
        }
        if c != free_count {
            writeln!(out, "0").unwrap();
            return;
        }
    }

    let time_limit = std::time::Duration::from_secs(270);
    let deadline = std::time::Instant::now();

    let mut solver = Solver {
        w, h, n,
        blocked,
        visited: vec![false; n],
        path: Vec::with_capacity(free_count),
        require_cycle,
        start,
        deadline,
        time_limit,
        nodes: 0,
        timed_out: false,
    };

    solver.visited[start] = true;
    solver.path.push(start);

    let found = solver.dfs(start, free_count - 1);

    if !found {
        writeln!(out, "0").unwrap();
    } else {
        writeln!(out, "{}", solver.path.len()).unwrap();
        for &p in &solver.path {
            let x = (p % (w as usize)) as i32;
            let y = (p / (w as usize)) as i32;
            writeln!(out, "{} {}", x, y).unwrap();
        }
    }
}
