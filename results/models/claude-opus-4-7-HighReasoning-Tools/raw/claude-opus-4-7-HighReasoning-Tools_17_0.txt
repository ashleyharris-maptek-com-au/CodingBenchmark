using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Text;

class Program
{
    struct Box
    {
        public int Index;
        public int W, H, D;
        public long Volume;
    }

    struct Placement
    {
        public int Index;
        public int X, Y, Z;
        public int W, H, D;
    }

    struct FreeSpace
    {
        public int X, Y, Z;
        public int W, H, D;
        public FreeSpace(int x, int y, int z, int w, int h, int d)
        {
            X = x; Y = y; Z = z; W = w; H = h; D = d;
        }
    }

    static int CW, CH, CD;
    static Stopwatch SW;
    static long TimeBudgetMs;

    static void Main()
    {
        SW = Stopwatch.StartNew();
        TimeBudgetMs = 27000;

        var input = Console.In.ReadToEnd();
        var tokens = input.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int p = 0;
        int N = int.Parse(tokens[p++]);
        CW = int.Parse(tokens[p++]);
        CH = int.Parse(tokens[p++]);
        CD = int.Parse(tokens[p++]);

        var boxes = new Box[N];
        for (int i = 0; i < N; i++)
        {
            boxes[i].Index = i;
            boxes[i].W = int.Parse(tokens[p++]);
            boxes[i].H = int.Parse(tokens[p++]);
            boxes[i].D = int.Parse(tokens[p++]);
            boxes[i].Volume = (long)boxes[i].W * boxes[i].H * boxes[i].D;
        }

        List<Placement> best = PackGuillotine(boxes, sortMode: 0);

        int[] sortModes = { 1, 2 };
        foreach (var mode in sortModes)
        {
            if (SW.ElapsedMilliseconds > TimeBudgetMs * 2 / 3) break;
            var result = PackGuillotine(boxes, sortMode: mode);
            if (result.Count > best.Count) best = result;
        }

        var sb = new StringBuilder();
        sb.Append(best.Count).Append('\n');
        foreach (var pl in best)
        {
            sb.Append(pl.Index).Append(' ').Append(pl.X).Append(' ').Append(pl.Y).Append(' ').Append(pl.Z).Append('\n');
        }
        Console.Write(sb.ToString());
    }

    static List<Placement> PackGuillotine(Box[] boxesIn, int sortMode)
    {
        var boxes = (Box[])boxesIn.Clone();
        switch (sortMode)
        {
            case 0:
                Array.Sort(boxes, (a, b) => b.Volume.CompareTo(a.Volume));
                break;
            case 1:
                Array.Sort(boxes, (a, b) =>
                {
                    int ma = Math.Max(a.W, Math.Max(a.H, a.D));
                    int mb = Math.Max(b.W, Math.Max(b.H, b.D));
                    return mb.CompareTo(ma);
                });
                break;
            case 2:
                Array.Sort(boxes, (a, b) =>
                {
                    int ma = a.W + a.H + a.D;
                    int mb = b.W + b.H + b.D;
                    return mb.CompareTo(ma);
                });
                break;
        }

        var placements = new List<Placement>();
        var free = new List<FreeSpace>();
        free.Add(new FreeSpace(0, 0, 0, CW, CH, CD));

        int N = boxes.Length;
        bool large = (long)CW * CH * CD > 1_000_000L || N > 2000;

        for (int i = 0; i < N; i++)
        {
            if ((i & 255) == 0 && SW.ElapsedMilliseconds > TimeBudgetMs) break;

            var box = boxes[i];
            int bestFs = -1;
            int bestOrient = -1;
            long bestScore = long.MaxValue;

            int[,] orients = new int[6, 3]
            {
                { box.W, box.H, box.D },
                { box.W, box.D, box.H },
                { box.H, box.W, box.D },
                { box.H, box.D, box.W },
                { box.D, box.W, box.H },
                { box.D, box.H, box.W }
            };

            int fc = free.Count;
            for (int j = 0; j < fc; j++)
            {
                var fs = free[j];
                for (int o = 0; o < 6; o++)
                {
                    int bw = orients[o, 0];
                    int bh = orients[o, 1];
                    int bd = orients[o, 2];
                    if (bw <= fs.W && bh <= fs.H && bd <= fs.D)
                    {
                        long score = (long)fs.Z * 10000000L + (long)fs.Y * 1000L + fs.X;
                        long leftover = (long)(fs.W - bw) + (fs.H - bh) + (fs.D - bd);
                        score = score * 100 + leftover;
                        if (score < bestScore)
                        {
                            bestScore = score;
                            bestFs = j;
                            bestOrient = o;
                        }
                    }
                }
            }

            if (bestFs < 0) continue;

            var chosen = free[bestFs];
            int cw = orients[bestOrient, 0];
            int ch = orients[bestOrient, 1];
            int cd = orients[bestOrient, 2];

            placements.Add(new Placement
            {
                Index = box.Index,
                X = chosen.X,
                Y = chosen.Y,
                Z = chosen.Z,
                W = cw,
                H = ch,
                D = cd
            });

            free.RemoveAt(bestFs);

            int rx = chosen.W - cw;
            int ry = chosen.H - ch;
            int rz = chosen.D - cd;

            FreeSpace s1, s2, s3;
            if (rx >= ry && rx >= rz)
            {
                s1 = new FreeSpace(chosen.X + cw, chosen.Y, chosen.Z, rx, chosen.H, chosen.D);
                s2 = new FreeSpace(chosen.X, chosen.Y + ch, chosen.Z, cw, ry, chosen.D);
                s3 = new FreeSpace(chosen.X, chosen.Y, chosen.Z + cd, cw, ch, rz);
            }
            else if (ry >= rx && ry >= rz)
            {
                s1 = new FreeSpace(chosen.X, chosen.Y + ch, chosen.Z, chosen.W, ry, chosen.D);
                s2 = new FreeSpace(chosen.X + cw, chosen.Y, chosen.Z, rx, ch, chosen.D);
                s3 = new FreeSpace(chosen.X, chosen.Y, chosen.Z + cd, cw, ch, rz);
            }
            else
            {
                s1 = new FreeSpace(chosen.X, chosen.Y, chosen.Z + cd, chosen.W, chosen.H, rz);
                s2 = new FreeSpace(chosen.X + cw, chosen.Y, chosen.Z, rx, chosen.H, cd);
                s3 = new FreeSpace(chosen.X, chosen.Y + ch, chosen.Z, cw, ry, cd);
            }

            if (s1.W > 0 && s1.H > 0 && s1.D > 0) free.Add(s1);
            if (s2.W > 0 && s2.H > 0 && s2.D > 0) free.Add(s2);
            if (s3.W > 0 && s3.H > 0 && s3.D > 0) free.Add(s3);

            if (large && free.Count > 4000)
            {
                free.Sort((a, b) =>
                {
                    long va = (long)a.W * a.H * a.D;
                    long vb = (long)b.W * b.H * b.D;
                    return vb.CompareTo(va);
                });
                free.RemoveRange(2000, free.Count - 2000);
            }
        }

        return placements;
    }
}
