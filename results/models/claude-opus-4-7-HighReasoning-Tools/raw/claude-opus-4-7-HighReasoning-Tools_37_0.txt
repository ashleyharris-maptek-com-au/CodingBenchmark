using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Text;
using System.Diagnostics;

class Program
{
    static int N;
    static HashSet<int>[] outAdj;
    static HashSet<int>[] inAdj;
    static bool[] alive;
    static Stopwatch sw = new Stopwatch();
    static long TIME_LIMIT_MS = 25000;

    static void Main()
    {
        sw.Start();
        var data = Console.In.ReadToEnd();
        var toks = data.Split(new[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        int n = int.Parse(toks[idx++]);
        int m = int.Parse(toks[idx++]);
        N = n;
        outAdj = new HashSet<int>[n];
        inAdj = new HashSet<int>[n];
        for (int i = 0; i < n; i++) { outAdj[i] = new HashSet<int>(); inAdj[i] = new HashSet<int>(); }
        alive = new bool[n];
        for (int i = 0; i < n; i++) alive[i] = true;

        var fvs = new HashSet<int>();
        for (int i = 0; i < m; i++)
        {
            int u = int.Parse(toks[idx++]);
            int v = int.Parse(toks[idx++]);
            if (u == v) { fvs.Add(u); }
            else { outAdj[u].Add(v); inAdj[v].Add(u); }
        }

        foreach (var v in fvs) RemoveVertex(v);

        ApplyReductions();

        var sccs = ComputeSCCs();
        foreach (var scc in sccs)
        {
            if (scc.Count <= 1) continue;
            var sccSet = new HashSet<int>(scc);
            var part = SolveSCC(sccSet);
            foreach (var v in part) fvs.Add(v);
        }

        var sb = new StringBuilder();
        sb.Append(fvs.Count).Append('\n');
        sb.Append(string.Join(" ", fvs));
        Console.Write(sb.ToString());
    }

    static void RemoveVertex(int v)
    {
        alive[v] = false;
        foreach (var u in inAdj[v]) outAdj[u].Remove(v);
        foreach (var w in outAdj[v]) inAdj[w].Remove(v);
        inAdj[v].Clear();
        outAdj[v].Clear();
    }

    static void ApplyReductions()
    {
        bool changed = true;
        while (changed)
        {
            changed = false;
            for (int v = 0; v < N; v++)
            {
                if (!alive[v]) continue;
                if (inAdj[v].Count == 0 || outAdj[v].Count == 0)
                {
                    RemoveVertex(v);
                    changed = true;
                }
            }
        }
    }

    static List<List<int>> ComputeSCCs()
    {
        var index = new int[N];
        var low = new int[N];
        var onStk = new bool[N];
        var vis = new bool[N];
        int ctr = 0;
        var stk = new Stack<int>();
        var result = new List<List<int>>();

        for (int s = 0; s < N; s++)
        {
            if (!alive[s] || vis[s]) continue;
            var work = new Stack<(int u, IEnumerator<int> it)>();
            vis[s] = true; index[s] = low[s] = ctr++; stk.Push(s); onStk[s] = true;
            work.Push((s, outAdj[s].GetEnumerator()));
            while (work.Count > 0)
            {
                var top = work.Peek();
                int u = top.u;
                bool advanced = false;
                while (top.it.MoveNext())
                {
                    int w = top.it.Current;
                    if (!alive[w]) continue;
                    if (!vis[w])
                    {
                        vis[w] = true; index[w] = low[w] = ctr++; stk.Push(w); onStk[w] = true;
                        work.Push((w, outAdj[w].GetEnumerator()));
                        advanced = true;
                        break;
                    }
                    else if (onStk[w])
                    {
                        if (index[w] < low[u]) low[u] = index[w];
                    }
                }
                if (!advanced)
                {
                    work.Pop();
                    if (work.Count > 0)
                    {
                        int p = work.Peek().u;
                        if (low[u] < low[p]) low[p] = low[u];
                    }
                    if (low[u] == index[u])
                    {
                        var comp = new List<int>();
                        int w;
                        do { w = stk.Pop(); onStk[w] = false; comp.Add(w); } while (w != u);
                        result.Add(comp);
                    }
                }
            }
        }
        return result;
    }

    static List<int> SolveSCC(HashSet<int> scc)
    {
        var greedy = GreedyFVS(scc);

        if (scc.Count <= 200 && sw.ElapsedMilliseconds < TIME_LIMIT_MS)
        {
            var bnb = new BnB(N, outAdj, inAdj, scc, greedy);
            var exact = bnb.Run(TIME_LIMIT_MS - sw.ElapsedMilliseconds);
            if (exact != null && exact.Count <= greedy.Count) return exact;
        }
        return greedy;
    }

    static List<int> GreedyFVS(HashSet<int> sccIn)
    {
        var result = new List<int>();
        var localAlive = new HashSet<int>(sccIn);

        Func<int, int> indeg = v => { int c = 0; foreach (var u in inAdj[v]) if (localAlive.Contains(u)) c++; return c; };
        Func<int, int> outdeg = v => { int c = 0; foreach (var u in outAdj[v]) if (localAlive.Contains(u)) c++; return c; };

        bool progress = true;
        while (progress && localAlive.Count > 0)
        {
            progress = false;
            var toRemove = new List<int>();
            foreach (var v in localAlive)
            {
                if (indeg(v) == 0 || outdeg(v) == 0) toRemove.Add(v);
            }
            foreach (var v in toRemove) { localAlive.Remove(v); progress = true; }
            if (localAlive.Count == 0) break;
            int pick = -1; long bestScore = -1;
            foreach (var v in localAlive)
            {
                long s = (long)indeg(v) * outdeg(v);
                if (s > bestScore) { bestScore = s; pick = v; }
            }
            if (pick == -1) break;
            result.Add(pick);
            localAlive.Remove(pick);
            progress = true;
        }
        return result;
    }
}

class BnB
{
    int N;
    HashSet<int>[] outAdj, inAdj;
    bool[] active;
    int bestSize;
    List<int> best;
    List<int> current;
    Stopwatch sw;
    long limitMs;
    bool timeout;

    public BnB(int n, HashSet<int>[] o, HashSet<int>[] i, HashSet<int> scc, List<int> initial)
    {
        N = n; outAdj = o; inAdj = i;
        active = new bool[n];
        foreach (var v in scc) active[v] = true;
        best = new List<int>(initial);
        bestSize = best.Count;
        current = new List<int>();
        sw = new Stopwatch();
    }

    public List<int> Run(long ms)
    {
        limitMs = ms;
        sw.Start();
        Search();
        if (timeout) return best;
        return best;
    }

    void Search()
    {
        if (timeout) return;
        if (sw.ElapsedMilliseconds > limitMs) { timeout = true; return; }
        if (current.Count >= bestSize) return;

        var removedHere = new List<int>();
        bool changed = true;
        while (changed)
        {
            changed = false;
            for (int v = 0; v < N; v++)
            {
                if (!active[v]) continue;
                bool hasIn = false;
                foreach (var u in inAdj[v]) { if (active[u]) { hasIn = true; break; } }
                if (!hasIn) { active[v] = false; removedHere.Add(v); changed = true; continue; }
                bool hasOut = false;
                foreach (var u in outAdj[v]) { if (active[u]) { hasOut = true; break; } }
                if (!hasOut) { active[v] = false; removedHere.Add(v); changed = true; }
            }
        }

        var cycle = FindShortCycle();
        if (cycle == null)
        {
            if (current.Count < bestSize) { best = new List<int>(current); bestSize = current.Count; }
        }
        else
        {
            foreach (var v in cycle)
            {
                if (timeout) break;
                if (current.Count + 1 >= bestSize) continue;
                active[v] = false;
                current.Add(v);
                Search();
                current.RemoveAt(current.Count - 1);
                active[v] = true;
            }
        }

        foreach (var v in removedHere) active[v] = true;
    }

    List<int> FindShortCycle()
    {
        List<int> shortest = null;
        int shortestLen = int.MaxValue;
        var dist = new int[N];
        var par = new int[N];
        for (int s = 0; s < N; s++)
        {
            if (!active[s]) continue;
            for (int i = 0; i < N; i++) { dist[i] = -1; par[i] = -1; }
            dist[s] = 0;
            var q = new Queue<int>();
            q.Enqueue(s);
            bool found = false;
            while (q.Count > 0 && !found)
            {
                int u = q.Dequeue();
                if (dist[u] + 1 >= shortestLen) continue;
                foreach (var w in outAdj[u])
                {
                    if (!active[w]) continue;
                    if (w == s)
                    {
                        int len = dist[u] + 1;
                        if (len < shortestLen)
                        {
                            shortestLen = len;
                            var cyc = new List<int>();
                            int cur = u;
                            while (cur != -1) { cyc.Add(cur); cur = par[cur]; }
                            shortest = cyc;
                        }
                        found = true;
                        break;
                    }
                    if (dist[w] == -1)
                    {
                        dist[w] = dist[u] + 1;
                        par[w] = u;
                        q.Enqueue(w);
                    }
                }
            }
            if (shortestLen <= 2) return shortest;
        }
        return shortest;
    }
}
