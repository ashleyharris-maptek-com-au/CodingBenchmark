use std::io::{self, Read, Write, BufWriter};

fn parse_input(buf: &[u8]) -> (i64, i64, Vec<i64>) {
    let mut nums: Vec<i64> = Vec::new();
    let mut i = 0usize;
    let n_bytes = buf.len();

    fn read_int(buf: &[u8], i: &mut usize) -> Option<i64> {
        let n_bytes = buf.len();
        while *i < n_bytes {
            let c = buf[*i];
            if c == b'-' || (c >= b'0' && c <= b'9') {
                break;
            }
            *i += 1;
        }
        if *i >= n_bytes {
            return None;
        }
        let mut neg = false;
        if buf[*i] == b'-' {
            neg = true;
            *i += 1;
        }
        let mut v: i64 = 0;
        let mut any = false;
        while *i < n_bytes {
            let c = buf[*i];
            if c >= b'0' && c <= b'9' {
                v = v.wrapping_mul(10).wrapping_add((c - b'0') as i64);
                *i += 1;
                any = true;
            } else {
                break;
            }
        }
        if !any {
            return None;
        }
        Some(if neg { -v } else { v })
    }

    let n = read_int(buf, &mut i).unwrap_or(0);
    let t = read_int(buf, &mut i).unwrap_or(0);
    nums.reserve(n.max(0) as usize);
    for _ in 0..n {
        match read_int(buf, &mut i) {
            Some(v) => nums.push(v),
            None => break,
        }
    }
    (n, t, nums)
}

fn meet_in_the_middle(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let half = n / 2;
    let left = &nums[..half];
    let right = &nums[half..];

    let l_count = 1usize << left.len();
    let mut l_sums: Vec<(i64, u32)> = Vec::with_capacity(l_count);
    for m in 0..l_count {
        let mut s: i64 = 0;
        for b in 0..left.len() {
            if (m >> b) & 1 == 1 {
                s = s.wrapping_add(left[b]);
            }
        }
        l_sums.push((s, m as u32));
    }
    l_sums.sort_by_key(|x| x.0);

    let r_count = 1usize << right.len();
    for m in 0..r_count {
        let mut s: i64 = 0;
        for b in 0..right.len() {
            if (m >> b) & 1 == 1 {
                s = s.wrapping_add(right[b]);
            }
        }
        let need = target.wrapping_sub(s);
        let mut lo = 0usize;
        let mut hi = l_sums.len();
        while lo < hi {
            let mid = (lo + hi) / 2;
            if l_sums[mid].0 < need {
                lo = mid + 1;
            } else {
                hi = mid;
            }
        }
        if lo < l_sums.len() && l_sums[lo].0 == need {
            let lmask = l_sums[lo].1 as usize;
            let mut indices: Vec<usize> = Vec::new();
            for b in 0..left.len() {
                if (lmask >> b) & 1 == 1 {
                    indices.push(b);
                }
            }
            for b in 0..right.len() {
                if (m >> b) & 1 == 1 {
                    indices.push(half + b);
                }
            }
            return Some(indices);
        }
    }
    None
}

fn bitset_dp(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    if target < 0 {
        return None;
    }
    let t = target as usize;
    let words = (t / 64) + 1;
    let mut dp: Vec<u64> = vec![0u64; words];
    dp[0] = 1u64;

    let n = nums.len();
    let mut took_at: Vec<Vec<u64>> = Vec::with_capacity(n);

    for i in 0..n {
        let v = nums[i];
        if v < 0 || (v as usize) > t {
            took_at.push(Vec::new());
            continue;
        }
        let shift = v as usize;
        let word_shift = shift / 64;
        let bit_shift = shift % 64;

        let mut shifted: Vec<u64> = vec![0u64; words];
        if bit_shift == 0 {
            for w in (word_shift..words).rev() {
                shifted[w] = dp[w - word_shift];
            }
        } else {
            for w in (word_shift..words).rev() {
                let lo = dp[w - word_shift] << bit_shift;
                let hi = if w > word_shift {
                    dp[w - word_shift - 1] >> (64 - bit_shift)
                } else {
                    0
                };
                shifted[w] = lo | hi;
            }
        }
        let last_bits = (t % 64) + 1;
        if last_bits < 64 {
            let mask = (1u64 << last_bits) - 1;
            shifted[words - 1] &= mask;
        }

        let mut newly = vec![0u64; words];
        for w in 0..words {
            newly[w] = shifted[w] & !dp[w];
            dp[w] |= shifted[w];
        }
        took_at.push(newly);

        let tw = t / 64;
        let tb = t % 64;
        if (dp[tw] >> tb) & 1 == 1 {
            let mut indices: Vec<usize> = Vec::new();
            let mut cur = t;
            let mut j = i as isize;
            while j >= 0 {
                let ju = j as usize;
                let tw2 = cur / 64;
                let tb2 = cur % 64;
                if !took_at[ju].is_empty() && (took_at[ju][tw2] >> tb2) & 1 == 1 {
                    indices.push(ju);
                    let vv = nums[ju] as usize;
                    if vv > cur {
                        break;
                    }
                    cur -= vv;
                    if cur == 0 {
                        break;
                    }
                }
                j -= 1;
            }
            indices.reverse();
            return Some(indices);
        }
    }
    None
}

fn quick_checks(nums: &[i64], target: i64) -> Option<Option<Vec<usize>>> {
    if target == 0 {
        return Some(Some(Vec::new()));
    }
    if nums.is_empty() {
        return Some(None);
    }
    for (i, &v) in nums.iter().enumerate() {
        if v == target {
            return Some(Some(vec![i]));
        }
    }
    let mut total: i128 = 0;
    for &v in nums {
        total += v as i128;
    }
    if total == target as i128 {
        return Some(Some((0..nums.len()).collect()));
    }
    None
}

fn emit(out: &mut impl Write, res: Option<Vec<usize>>) {
    match res {
        Some(indices) => {
            writeln!(out, "YES").ok();
            let mut s = String::new();
            s.push('[');
            for (k, idx) in indices.iter().enumerate() {
                if k > 0 {
                    s.push(' ');
                }
                s.push_str(&idx.to_string());
            }
            s.push(']');
            writeln!(out, "{}", s).ok();
        }
        None => {
            writeln!(out, "NO").ok();
        }
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).ok();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let (_n_decl, target, nums) = parse_input(input.as_bytes());

    if let Some(res) = quick_checks(&nums, target) {
        emit(&mut out, res);
        return;
    }

    let n = nums.len();

    let all_nonneg = nums.iter().all(|&v| v >= 0) && target >= 0;
    let bitset_words_estimate: u128 = if target >= 0 {
        ((target as u128) / 64) + 1
    } else {
        u128::MAX
    };
    let mem_budget_bytes: u128 = 400u128 * 1024 * 1024;
    let per_index_bytes: u128 = bitset_words_estimate.saturating_mul(8);
    let total_bytes: u128 = per_index_bytes.saturating_mul(n as u128 + 1);

    if all_nonneg && total_bytes <= mem_budget_bytes {
        let r = bitset_dp(&nums, target);
        emit(&mut out, r);
        return;
    }

    if n <= 40 {
        let r = meet_in_the_middle(&nums, target);
        emit(&mut out, r);
        return;
    }

    if target >= 0 && nums.iter().all(|&v| v >= 0) {
        let mut sorted_idx: Vec<usize> = (0..n).collect();
        sorted_idx.sort_by(|&a, &b| nums[b].cmp(&nums[a]));
        let mut remaining = target;
        let mut chosen: Vec<usize> = Vec::new();
        for &i in &sorted_idx {
            let v = nums[i];
            if v >= 0 && v <= remaining {
                chosen.push(i);
                remaining -= v;
                if remaining == 0 {
                    break;
                }
            }
        }
        if remaining == 0 {
            chosen.sort();
            emit(&mut out, Some(chosen));
            return;
        }
    }

    writeln!(out, "NO").ok();
}
