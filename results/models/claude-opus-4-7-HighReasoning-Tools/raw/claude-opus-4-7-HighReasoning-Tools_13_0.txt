#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cstdint>
#include <iostream>

#include <algorithm>
#include <unordered_set>
#include <unordered_map>
#include <limits>
#include <intrin.h>
using namespace std;

struct DH {
    uint64_t h1, h2;
    bool operator==(const DH& o) const { return h1 == o.h1 && h2 == o.h2; }
};
struct DHHash {
    size_t operator()(const DH& d) const noexcept {
        return std::hash<uint64_t>()(d.h1) ^ (std::hash<uint64_t>()(d.h2) << 1);
    }
};

static const uint64_t MOD1 = (1ULL << 61) - 1;
static const uint64_t BASE1 = 131;
static const uint64_t MOD2 = 1000000007ULL;
static const uint64_t BASE2 = 137;

static inline uint64_t mulmod61(uint64_t a, uint64_t b) {
    unsigned long long hi;
    unsigned long long lo = _umul128(a, b, &hi);
    uint64_t r = ((hi << 3) | (lo >> 61)) + (lo & MOD1);
    if (r >= MOD1) r -= MOD1;
    return r;
}

struct RH {
    vector<uint64_t> h1, h2;
    void build(const string& s) {
        int n = (int)s.size();
        h1.assign(n + 1, 0);
        h2.assign(n + 1, 0);
        for (int i = 0; i < n; i++) {
            unsigned char c = (unsigned char)s[i] + 1;
            uint64_t v = mulmod61(h1[i], BASE1) + c;
            if (v >= MOD1) v -= MOD1;
            h1[i + 1] = v;
            h2[i + 1] = (h2[i] * BASE2 + c) % MOD2;
        }
    }
};

static vector<uint64_t> POW1, POW2;
static void ensurePow(int n) {
    if ((int)POW1.size() >= n + 1) return;
    int old = (int)POW1.size();
    POW1.resize(n + 1);
    POW2.resize(n + 1);
    if (old == 0) { POW1[0] = 1; POW2[0] = 1; old = 1; }
    for (int i = old; i <= n; i++) {
        POW1[i] = mulmod61(POW1[i - 1], BASE1);
        POW2[i] = (POW2[i - 1] * BASE2) % MOD2;
    }
}

static inline DH getHash(const RH& r, int l, int len) {
    uint64_t a = r.h1[l + len] + MOD1 - mulmod61(r.h1[l], POW1[len]);
    if (a >= MOD1) a -= MOD1;
    uint64_t b = (r.h2[l + len] + MOD2 - (r.h2[l] * POW2[len]) % MOD2) % MOD2;
    return {a, b};
}

static int hasCommonOfLen(const vector<string>& strs, const vector<RH>& rhs, int L) {
    int N = (int)strs.size();
    if (L == 0) return 0;
    if ((int)strs[0].size() < L) return -1;
    unordered_map<DH, int, DHHash> firstIdx;
    firstIdx.reserve((size_t)(strs[0].size() - L + 1) * 2);
    for (int i = 0; i + L <= (int)strs[0].size(); i++) {
        DH h = getHash(rhs[0], i, L);
        firstIdx.emplace(h, i);
    }
    unordered_set<DH, DHHash> cur;
    cur.reserve(firstIdx.size());
    for (auto& kv : firstIdx) cur.insert(kv.first);

    for (int k = 1; k < N; k++) {
        if ((int)strs[k].size() < L) return -1;
        unordered_set<DH, DHHash> nxt;
        nxt.reserve(cur.size());
        for (int i = 0; i + L <= (int)strs[k].size(); i++) {
            DH h = getHash(rhs[k], i, L);
            if (cur.count(h)) nxt.insert(h);
        }
        cur.swap(nxt);
        if (cur.empty()) return -1;
    }
    for (auto& h : cur) {
        auto it = firstIdx.find(h);
        if (it != firstIdx.end()) return it->second;
    }
    return -1;
}

static string lcs_dp_two(const string& a, const string& b) {
    int n = (int)a.size(), m = (int)b.size();
    if (n == 0 || m == 0) return "";
    vector<int> prev(m + 1, 0), cur(m + 1, 0);
    int best = 0, bestEnd = 0;
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            if (a[i - 1] == b[j - 1]) {
                cur[j] = prev[j - 1] + 1;
                if (cur[j] > best) { best = cur[j]; bestEnd = i; }
            } else cur[j] = 0;
        }
        swap(prev, cur);
        fill(cur.begin(), cur.end(), 0);
    }
    return a.substr(bestEnd - best, best);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) { cout << "\n"; return 0; }
    cin.ignore(numeric_limits<streamsize>::max(), '\n');

    vector<string> strs(N);
    for (int i = 0; i < N; i++) {
        if (!std::getline(cin, strs[i])) strs[i] = "";
        while (!strs[i].empty() && (strs[i].back() == '\r' || strs[i].back() == '\n'))
            strs[i].pop_back();
    }

    if (N == 0) { cout << "\n"; return 0; }
    if (N == 1) { cout << strs[0] << "\n"; return 0; }

    size_t minLen = strs[0].size();
    int minIdx = 0;
    for (int i = 1; i < N; i++) {
        if (strs[i].size() < minLen) { minLen = strs[i].size(); minIdx = i; }
    }
    if (minLen == 0) { cout << "\n"; return 0; }

    if (minIdx != 0) swap(strs[0], strs[minIdx]);

    if (N == 2 && strs[0].size() <= 3000 && strs[1].size() <= 3000) {
        string r = lcs_dp_two(strs[0], strs[1]);
        cout << r << "\n";
        return 0;
    }

    ensurePow((int)minLen + 2);
    vector<RH> rhs(N);
    for (int i = 0; i < N; i++) rhs[i].build(strs[i]);

    int lo = 1, hi = (int)minLen;
    int bestL = 0, bestStart = -1;

    while (lo <= hi) {
        int mid = lo + (hi - lo) / 2;
        int idx = hasCommonOfLen(strs, rhs, mid);
        if (idx >= 0) {
            bestL = mid;
            bestStart = idx;
            lo = mid + 1;
        } else {
            hi = mid - 1;
        }
    }

    if (bestL <= 0 || bestStart < 0) {
        cout << "\n";
        return 0;
    }

    string cand = strs[0].substr(bestStart, bestL);
    bool ok = true;
    for (int k = 1; k < N && ok; k++) {
        if (strs[k].find(cand) == string::npos) ok = false;
    }
    if (ok) {
        cout << cand << "\n";
    } else {
        bool found = false;
        for (int i = 0; i + bestL <= (int)strs[0].size() && !found; i++) {
            string c = strs[0].substr(i, bestL);
            bool all = true;
            for (int k = 1; k < N; k++) {
                if (strs[k].find(c) == string::npos) { all = false; break; }
            }
            if (all) { cout << c << "\n"; found = true; }
        }
        if (!found) cout << "\n";
    }
    return 0;
}
