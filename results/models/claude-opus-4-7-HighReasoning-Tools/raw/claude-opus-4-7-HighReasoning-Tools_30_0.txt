import sys
from array import array
import random


def solve():
    data = sys.stdin.buffer.read().split()
    pos = 0
    n = int(data[pos]); pos += 1
    m = int(data[pos]); pos += 1

    total = m * 3
    # Build flat literal array
    lits = array('i', map(int, data[pos:pos + total]))

    # Start with all variables True
    assign = bytearray(n + 2)
    for i in range(1, n + 1):
        assign[i] = 1

    def clause_sat(ci):
        b = ci * 3
        l = lits[b]
        if l > 0:
            if assign[l]: return True
        else:
            if not assign[-l]: return True
        l = lits[b + 1]
        if l > 0:
            if assign[l]: return True
        else:
            if not assign[-l]: return True
        l = lits[b + 2]
        if l > 0:
            if assign[l]: return True
        else:
            if not assign[-l]: return True
        return False

    # Find initially unsatisfied clauses
    unsat_list = []
    for ci in range(m):
        if not clause_sat(ci):
            unsat_list.append(ci)

    if unsat_list:
        # Build CSR occurrence lists: for each variable, list of clauses containing it
        deg = array('i', [0]) * (n + 2)
        for v in lits:
            av = v if v > 0 else -v
            deg[av + 1] += 1
        for i in range(1, n + 2):
            deg[i] += deg[i - 1]
        occ = array('i', [0]) * total
        cursor = array('i', deg)
        for ci in range(m):
            b = ci * 3
            v1 = lits[b]; av = v1 if v1 > 0 else -v1
            occ[cursor[av]] = ci; cursor[av] += 1
            v1 = lits[b + 1]; av = v1 if v1 > 0 else -v1
            occ[cursor[av]] = ci; cursor[av] += 1
            v1 = lits[b + 2]; av = v1 if v1 > 0 else -v1
            occ[cursor[av]] = ci; cursor[av] += 1

        unsat_set = set(unsat_list)
        unsat_arr = unsat_list  # we'll keep mutating
        rng = random.Random(1)
        max_flips = 50 * m + 1000000
        flips = 0

        while unsat_set and flips < max_flips:
            # Pick a random unsat clause
            ci = rng.choice(unsat_arr) if unsat_arr else next(iter(unsat_set))
            if ci not in unsat_set:
                # stale; rebuild list occasionally
                unsat_arr = list(unsat_set)
                if not unsat_arr:
                    break
                ci = rng.choice(unsat_arr)

            b = ci * 3
            l1 = lits[b]; l2 = lits[b + 1]; l3 = lits[b + 2]
            v1 = l1 if l1 > 0 else -l1
            v2 = l2 if l2 > 0 else -l2
            v3 = l3 if l3 > 0 else -l3

            if rng.random() < 0.4:
                chosen = (v1, v2, v3)[rng.randint(0, 2)]
            else:
                # min-break heuristic
                best = None
                best_br = 10**9
                for v in (v1, v2, v3):
                    assign[v] ^= 1
                    br = 0
                    s = deg[v]; e = deg[v + 1]
                    for j in range(s, e):
                        cj = occ[j]
                        if cj in unsat_set:
                            continue
                        if not clause_sat(cj):
                            br += 1
                            if br >= best_br:
                                break
                    assign[v] ^= 1
                    if br < best_br:
                        best_br = br
                        best = v
                        if br == 0:
                            break
                chosen = best

            assign[chosen] ^= 1
            flips += 1

            s = deg[chosen]; e = deg[chosen + 1]
            for j in range(s, e):
                cj = occ[j]
                if clause_sat(cj):
                    if cj in unsat_set:
                        unsat_set.discard(cj)
                else:
                    if cj not in unsat_set:
                        unsat_set.add(cj)

            if flips % 100000 == 0:
                unsat_arr = list(unsat_set)
                if not unsat_arr:
                    break

    sys.stdout.write("SAT\n")
    sys.stdout.write(" ".join("1" if assign[i] else "0" for i in range(1, n + 1)))
    sys.stdout.write("\n")


solve()
