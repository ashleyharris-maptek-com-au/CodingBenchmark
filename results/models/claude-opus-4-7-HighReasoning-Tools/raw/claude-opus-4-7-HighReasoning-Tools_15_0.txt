use std::io::{self, Read, Write, BufWriter};

#[derive(Clone, Copy, Debug)]
struct V2 { x: f64, y: f64 }

#[derive(Clone, Copy, Debug)]
struct V3 { x: f64, y: f64, z: f64 }

fn v3(x: f64, y: f64, z: f64) -> V3 { V3 { x, y, z } }

fn scl3(a: V3, s: f64) -> V3 { v3(a.x*s, a.y*s, a.z*s) }
fn dot3(a: V3, b: V3) -> f64 { a.x*b.x + a.y*b.y + a.z*b.z }
fn norm3(a: V3) -> V3 {
    let l = dot3(a, a).sqrt();
    if l < 1e-12 { v3(0.0, 0.0, -1.0) } else { scl3(a, 1.0/l) }
}

fn quat_rot(q: (f64, f64, f64, f64), v: V3) -> V3 {
    let (w, x, y, z) = q;
    let t2 = w*x; let t3 = w*y; let t4 = w*z;
    let t5 = -x*x; let t6 = x*y; let t7 = x*z;
    let t8 = -y*y; let t9 = y*z; let t10 = -z*z;
    v3(
        2.0*((t8 + t10)*v.x + (t6 - t4)*v.y + (t3 + t7)*v.z) + v.x,
        2.0*((t4 + t6)*v.x + (t5 + t10)*v.y + (t9 - t2)*v.z) + v.y,
        2.0*((t7 - t3)*v.x + (t2 + t9)*v.y + (t5 + t8)*v.z) + v.z,
    )
}

fn read_input() -> (Vec<V2>, V3) {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_ascii_whitespace();
    let nv: usize = it.next().unwrap().parse().unwrap();
    let mut poly = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        poly.push(V2 { x, y });
    }
    let sx: f64 = it.next().unwrap().parse().unwrap();
    let sy: f64 = it.next().unwrap().parse().unwrap();
    let sz: f64 = it.next().unwrap().parse().unwrap();
    (poly, v3(sx, sy, sz))
}

fn polygon_bbox(poly: &[V2]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY; let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY; let mut maxy = f64::NEG_INFINITY;
    for p in poly {
        if p.x < minx { minx = p.x; }
        if p.y < miny { miny = p.y; }
        if p.x > maxx { maxx = p.x; }
        if p.y > maxy { maxy = p.y; }
    }
    (minx, miny, maxx, maxy)
}

fn point_in_poly(poly: &[V2], px: f64, py: f64) -> bool {
    let mut inside = false;
    let n = poly.len();
    if n < 3 { return false; }
    let mut j = n - 1;
    for i in 0..n {
        let (xi, yi) = (poly[i].x, poly[i].y);
        let (xj, yj) = (poly[j].x, poly[j].y);
        if ((yi > py) != (yj > py)) &&
            (px < (xj - xi) * (py - yi) / (yj - yi + 1e-30) + xi) {
            inside = !inside;
        }
        j = i;
    }
    inside
}

// Compute shadow vertices (polygon, possibly not convex-ordered) of a tetra
// with given center offset (cx,cy), height lift hz, scale s, identity rotation,
// under sun direction (must have sun.z < 0).
fn tet_shadow_points(cx: f64, cy: f64, hz: f64, scale: f64, sun: V3,
                     q: (f64,f64,f64,f64)) -> [V2; 4] {
    let base = [
        v3(1.0, 0.0, -0.707),
        v3(-1.0, 0.0, -0.707),
        v3(0.0, 1.0, 0.707),
        v3(0.0, -1.0, 0.707),
    ];
    let mut out = [V2{x:0.0,y:0.0}; 4];
    for i in 0..4 {
        let r = scl3(quat_rot(q, base[i]), scale);
        let wp = v3(r.x + cx, r.y + cy, r.z + hz);
        let t = -wp.z / sun.z;
        out[i] = V2 { x: wp.x + t*sun.x, y: wp.y + t*sun.y };
    }
    out
}

fn main() {
    let (poly, sun_in) = read_input();
    let mut sun = norm3(sun_in);
    if sun.z >= 0.0 {
        // Force pointing downward by flipping z (defensive)
        sun = v3(sun.x, sun.y, -sun.z.abs().max(1e-6));
        let l = dot3(sun, sun).sqrt();
        sun = scl3(sun, 1.0/l);
    }

    let (minx, miny, maxx, maxy) = polygon_bbox(&poly);
    let w = maxx - minx;
    let h = maxy - miny;
    let diag = (w*w + h*h).sqrt();

    let id_q = (1.0_f64, 0.0, 0.0, 0.0);

    // For identity rotation, lifted base (min vertex on z=0):
    // base z: -0.707, -0.707, 0.707, 0.707 -> after lift +0.707:
    // vertices: (1,0,0), (-1,0,0), (0,1,1.414), (0,-1,1.414). scaled by s.
    // Bottom edge is along x-axis. xy bbox: [-s,s] x [-s,s].
    let base_lift_unit = 0.707_f64; // lift per unit scale

    let mut tets: Vec<(f64,f64,f64,(f64,f64,f64,f64),f64)> = Vec::new();

    // Choose scale based on polygon size.
    let s_max = 10.0_f64;
    let target_scale = if diag < 2.0 {
        (diag * 0.6 + 0.5).clamp(0.2, s_max)
    } else {
        s_max
    };

    // Compute reference shadow at (0,0,lift) to find shadow's xy offset/extents
    let lift = base_lift_unit * target_scale;
    let ref_shadow = tet_shadow_points(0.0, 0.0, lift, target_scale, sun, id_q);
    let mut scx = 0.0; let mut scy = 0.0;
    for p in &ref_shadow { scx += p.x; scy += p.y; }
    scx /= 4.0; scy /= 4.0;
    // Inscribed axis-aligned square inside the (sheared) diamond shadow:
    // compute conservative half-extent as min distance from centroid to any vertex,
    // divided by sqrt(2).
    let mut min_r = f64::INFINITY;
    for p in &ref_shadow {
        let dx = p.x - scx; let dy = p.y - scy;
        let r = (dx*dx + dy*dy).sqrt();
        if r < min_r { min_r = r; }
    }
    let inscribed_half = (min_r / std::f64::consts::SQRT_2).max(0.05);
    let step = inscribed_half * 1.8; // step between centers (slight overlap)

    // 3D non-intersection: keep all tetras in single layer at lift, with xy
    // spacing >= 2*s (so bounding boxes don't overlap). If step < 2*s we add
    // additional layers stacked in z.
    let xy_min_spacing = 2.01 * target_scale;
    let use_xy_step = step.max(xy_min_spacing);

    let margin = use_xy_step;
    let gx0 = minx - margin;
    let gy0 = miny - margin;
    let gx1 = maxx + margin;
    let gy1 = maxy + margin;

    // Single-tetra special case
    if diag <= inscribed_half * 1.6 {
        // place one tetra so shadow centroid matches polygon center
        let pcx = (minx + maxx) * 0.5;
        let pcy = (miny + maxy) * 0.5;
        tets.push((pcx - scx, pcy - scy, lift, id_q, target_scale));
    } else {
        let mut y = gy0;
        while y <= gy1 + 1e-9 {
            let mut x = gx0;
            while x <= gx1 + 1e-9 {
                // shadow center if tetra at (x,y,lift)
                let sx_c = x + scx;
                let sy_c = y + scy;
                // Test small sample grid inside intended inscribed square
                let mut need = false;
                let samples = 4;
                'outer: for ix in 0..=samples {
                    for iy in 0..=samples {
                        let fx = (ix as f64 / samples as f64 - 0.5) * 2.0 * inscribed_half;
                        let fy = (iy as f64 / samples as f64 - 0.5) * 2.0 * inscribed_half;
                        if point_in_poly(&poly, sx_c + fx, sy_c + fy) {
                            need = true; break 'outer;
                        }
                    }
                }
                if need {
                    tets.push((x, y, lift, id_q, target_scale));
                }
                x += use_xy_step;
            }
            y += use_xy_step;
        }
        if tets.is_empty() {
            let pcx = (minx + maxx) * 0.5;
            let pcy = (miny + maxy) * 0.5;
            tets.push((pcx - scx, pcy - scy, lift, id_q, target_scale));
        }
    }

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "{}", tets.len()).unwrap();
    for &(px, py, pz, q, sc) in &tets {
        writeln!(out, "{} {} {} {} {} {} {} {}",
                 px, py, pz, q.0, q.1, q.2, q.3, sc).unwrap();
    }
}
