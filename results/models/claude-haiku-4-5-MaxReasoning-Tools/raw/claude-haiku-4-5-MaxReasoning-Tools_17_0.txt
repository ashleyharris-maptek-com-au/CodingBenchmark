using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    struct Box
    {
        public int Index;
        public int W, H, D;
    }

    struct Placement
    {
        public int Index;
        public int X, Y, Z;
        public int W, H, D;
    }

    static void Main()
    {
        var line = Console.ReadLine().Split();
        int n = int.Parse(line[0]);
        int W = int.Parse(line[1]);
        int H = int.Parse(line[2]);
        int D = int.Parse(line[3]);

        var boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            var dims = Console.ReadLine().Split();
            boxes[i] = new Box
            {
                Index = i,
                W = int.Parse(dims[0]),
                H = int.Parse(dims[1]),
                D = int.Parse(dims[2])
            };
        }

        // Sort by volume descending (largest first)
        var sortedBoxes = boxes.OrderByDescending(b => (long)b.W * b.H * b.D).ToArray();

        var placements = new List<Placement>();
        var candidates = new HashSet<(int, int, int)> { (0, 0, 0) };

        foreach (var box in sortedBoxes)
        {
            var orientations = GetOrientations(box);
            bool placed = false;

            // Try candidates sorted by position (leftmost, bottom-most, front-most)
            var sortedCandidates = candidates.OrderBy(c => c.Item1).ThenBy(c => c.Item2).ThenBy(c => c.Item3).ToList();

            foreach (var oriented in orientations)
            {
                foreach (var (x, y, z) in sortedCandidates)
                {
                    // Check if box fits within container bounds
                    if (x + oriented.W > W || y + oriented.H > H || z + oriented.D > D)
                        continue;

                    // Check for overlaps with already placed boxes
                    bool overlaps = false;
                    foreach (var p in placements)
                    {
                        if (BoxesOverlap(x, y, z, oriented.W, oriented.H, oriented.D,
                                       p.X, p.Y, p.Z, p.W, p.H, p.D))
                        {
                            overlaps = true;
                            break;
                        }
                    }

                    // Place the box if no overlap
                    if (!overlaps)
                    {
                        var placement = new Placement { Index = oriented.Index, X = x, Y = y, Z = z, W = oriented.W, H = oriented.H, D = oriented.D };
                        placements.Add(placement);

                        // Add new candidate positions from the corners of this placed box
                        candidates.Add((x + oriented.W, y, z));
                        candidates.Add((x, y + oriented.H, z));
                        candidates.Add((x, y, z + oriented.D));

                        placed = true;
                        break;
                    }
                }

                if (placed) break;
            }
        }

        // Output results
        Console.WriteLine(placements.Count);
        var result = placements.OrderBy(p => p.Index).ToList();
        foreach (var p in result)
        {
            Console.WriteLine($"{p.Index} {p.X} {p.Y} {p.Z}");
        }
    }

    // Generate all unique orientations (permutations of dimensions)
    static List<Box> GetOrientations(Box box)
    {
        var dims = new[] { box.W, box.H, box.D };
        var orientations = new HashSet<(int, int, int)>();

        // Try all permutations of dimensions
        foreach (var w in dims)
            foreach (var h in dims)
                foreach (var d in dims)
                    orientations.Add((w, h, d));

        return orientations.Select(o => new Box
        {
            Index = box.Index,
            W = o.Item1,
            H = o.Item2,
            D = o.Item3
        }).ToList();
    }

    // Check if two axis-aligned boxes overlap
    static bool BoxesOverlap(int x1, int y1, int z1, int w1, int h1, int d1,
                            int x2, int y2, int z2, int w2, int h2, int d2)
    {
        return x1 < x2 + w2 && x1 + w1 > x2 &&
               y1 < y2 + h2 && y1 + h1 > y2 &&
               z1 < z2 + d2 && z1 + d1 > z2;
    }
}
