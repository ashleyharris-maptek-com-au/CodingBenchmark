use std::io::{self, BufRead};

#[derive(Clone, Copy, Debug)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

#[derive(Clone, Copy, Debug)]
struct Quaternion {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

#[derive(Clone, Copy, Debug)]
struct Point2D {
    x: f64,
    y: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let nv: usize = lines
        .next()
        .unwrap()
        .unwrap()
        .trim()
        .parse()
        .unwrap();

    let mut polygon = Vec::new();
    for _ in 0..nv {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.trim().split_whitespace().collect();
        let x: f64 = parts[0].parse().unwrap();
        let y: f64 = parts[1].parse().unwrap();
        polygon.push(Point2D { x, y });
    }

    let sun_line = lines.next().unwrap().unwrap();
    let sun_parts: Vec<&str> = sun_line.trim().split_whitespace().collect();
    let mut sun_dir = Vec3 {
        x: sun_parts[0].parse().unwrap(),
        y: sun_parts[1].parse().unwrap(),
        z: sun_parts[2].parse().unwrap(),
    };

    let sun_len = (sun_dir.x * sun_dir.x + sun_dir.y * sun_dir.y + sun_dir.z * sun_dir.z).sqrt();
    if sun_len > 1e-10 {
        sun_dir.x /= sun_len;
        sun_dir.y /= sun_len;
        sun_dir.z /= sun_len;
    }

    let tetrahedrons = solve_covering(&polygon, sun_dir);

    println!("{}", tetrahedrons.len());
    for (pos, quat, scale) in tetrahedrons {
        println!(
            "{} {} {} {} {} {} {} {}",
            pos.x, pos.y, pos.z, quat.w, quat.x, quat.y, quat.z, scale
        );
    }
}

fn ref_tetrahedron_vertices() -> Vec<Vec3> {
    vec![
        Vec3 {
            x: 1.0,
            y: 0.0,
            z: -0.707,
        },
        Vec3 {
            x: -1.0,
            y: 0.0,
            z: -0.707,
        },
        Vec3 {
            x: 0.0,
            y: 1.0,
            z: 0.707,
        },
        Vec3 {
            x: 0.0,
            y: -1.0,
            z: 0.707,
        },
    ]
}

fn solve_covering(
    polygon: &[Point2D],
    sun_dir: Vec3,
) -> Vec<(Vec3, Quaternion, f64)> {
    let mut result = Vec::new();

    if polygon.is_empty() {
        return result;
    }

    let bbox = compute_bbox(polygon);
    let width = bbox.max_x - bbox.min_x;
    let height = bbox.max_y - bbox.min_y;

    let target_scale = 3.0;
    let cell_size = target_scale * 1.5;
    let grid_cols = ((width / cell_size).ceil() as usize).max(1);
    let grid_rows = ((height / cell_size).ceil() as usize).max(1);

    let col_step = if grid_cols > 0 {
        width / (grid_cols as f64)
    } else {
        cell_size
    };
    let row_step = if grid_rows > 0 {
        height / (grid_rows as f64)
    } else {
        cell_size
    };

    let ref_verts = ref_tetrahedron_vertices();

    for row in 0..grid_rows {
        for col in 0..grid_cols {
            let cx = bbox.min_x + (col as f64 + 0.5) * col_step;
            let cy = bbox.min_y + (row as f64 + 0.5) * row_step;
            let test_point = Point2D { x: cx, y: cy };

            if point_in_polygon(&test_point, polygon) {
                let scale = (target_scale).min(10.0);
                let shadow = compute_shadow(&ref_verts, cx, cy, 0.5, scale, sun_dir);

                if shadow_area(&shadow) > 1e-6 {
                    let quat = Quaternion {
                        w: 1.0,
                        x: 0.0,
                        y: 0.0,
                        z: 0.0,
                    };
                    result.push((Vec3 { x: cx, y: cy, z: 0.5 }, quat, scale));
                }
            }
        }
    }

    if result.is_empty() && !polygon.is_empty() {
        let centroid = polygon_centroid(polygon);
        let scale = 3.0_f64.min(10.0);
        let quat = Quaternion {
            w: 1.0,
            x: 0.0,
            y: 0.0,
            z: 0.0,
        };
        result.push((
            Vec3 {
                x: centroid.x,
                y: centroid.y,
                z: 0.5,
            },
            quat,
            scale,
        ));
    }

    result
}

fn compute_bbox(polygon: &[Point2D]) -> BBox {
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;

    for p in polygon {
        min_x = min_x.min(p.x);
        max_x = max_x.max(p.x);
        min_y = min_y.min(p.y);
        max_y = max_y.max(p.y);
    }

    BBox {
        min_x,
        max_x,
        min_y,
        max_y,
    }
}

struct BBox {
    min_x: f64,
    max_x: f64,
    min_y: f64,
    max_y: f64,
}

fn point_in_polygon(p: &Point2D, polygon: &[Point2D]) -> bool {
    let mut inside = false;
    let n = polygon.len();
    let mut j = n - 1;

    for i in 0..n {
        let pi = polygon[i];
        let pj = polygon[j];

        if ((pi.y > p.y) != (pj.y > p.y))
            && (p.x < (pj.x - pi.x) * (p.y - pi.y) / (pj.y - pi.y) + pi.x)
        {
            inside = !inside;
        }
        j = i;
    }

    inside
}

fn polygon_centroid(polygon: &[Point2D]) -> Point2D {
    let mut cx = 0.0;
    let mut cy = 0.0;
    for p in polygon {
        cx += p.x;
        cy += p.y;
    }
    let n = polygon.len() as f64;
    Point2D {
        x: cx / n,
        y: cy / n,
    }
}

fn compute_shadow(
    ref_verts: &[Vec3],
    px: f64,
    py: f64,
    pz: f64,
    scale: f64,
    sun_dir: Vec3,
) -> Vec<Point2D> {
    let mut shadow = Vec::new();

    for v in ref_verts {
        let scaled_x = v.x * scale;
        let scaled_y = v.y * scale;
        let scaled_z = v.z * scale;

        let world_x = px + scaled_x;
        let world_y = py + scaled_y;
        let world_z = pz + scaled_z;

        if sun_dir.z.abs() > 1e-10 {
            let t = -world_z / sun_dir.z;
            let proj_x = world_x + t * sun_dir.x;
            let proj_y = world_y + t * sun_dir.y;
            shadow.push(Point2D {
                x: proj_x,
                y: proj_y,
            });
        }
    }

    shadow
}

fn shadow_area(shadow: &[Point2D]) -> f64 {
    if shadow.len() < 3 {
        return 0.0;
    }

    let mut area = 0.0;
    for i in 0..shadow.len() {
        let j = (i + 1) % shadow.len();
        area += shadow[i].x * shadow[j].y;
        area -= shadow[j].x * shadow[i].y;
    }
    (area / 2.0).abs()
}
