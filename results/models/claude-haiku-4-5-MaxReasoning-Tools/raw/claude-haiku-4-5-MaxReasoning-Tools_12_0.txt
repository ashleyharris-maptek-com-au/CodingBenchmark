using System;
using System.Collections.Generic;
using System.Linq;

class Vector3
{
    public double X, Y, Z;
    public Vector3(double x = 0, double y = 0, double z = 0) { X = x; Y = y; Z = z; }
    public Vector3 Add(Vector3 v) => new Vector3(X + v.X, Y + v.Y, Z + v.Z);
    public Vector3 Sub(Vector3 v) => new Vector3(X - v.X, Y - v.Y, Z - v.Z);
    public Vector3 Mul(double s) => new Vector3(X * s, Y * s, Z * s);
    public double Dot(Vector3 v) => X * v.X + Y * v.Y + Z * v.Z;
    public Vector3 Cross(Vector3 v) => new Vector3(Y * v.Z - Z * v.Y, Z * v.X - X * v.Z, X * v.Y - Y * v.X);
    public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);
    public Vector3 Normalize() { double len = Length(); return len > 1e-10 ? Mul(1 / len) : new Vector3(1, 0, 0); }
}

class Quaternion
{
    public double W, X, Y, Z;
    public Quaternion(double w = 1, double x = 0, double y = 0, double z = 0) { W = w; X = x; Y = y; Z = z; }

    public static Quaternion FromAxisAngle(Vector3 axis, double angle)
    {
        axis = axis.Normalize();
        double half = angle / 2;
        double s = Math.Sin(half);
        return new Quaternion(Math.Cos(half), axis.X * s, axis.Y * s, axis.Z * s);
    }

    public Vector3 RotatePoint(Vector3 p)
    {
        double x = W * p.X + Y * p.Z - Z * p.Y;
        double y = W * p.Y + Z * p.X - X * p.Z;
        double z = W * p.Z + X * p.Y - Y * p.X;
        double w = -X * p.X - Y * p.Y - Z * p.Z;

        return new Vector3(
            w * -X + x * W - y * Z + z * Y,
            w * -Y + y * W - z * X + x * Z,
            w * -Z + z * W - x * Y + y * X
        );
    }
}

class Polyhedron
{
    public Vector3[] Vertices;
    public int[][] Faces;
    public Vector3 MinBound, MaxBound;
    public Vector3 Center;

    public Polyhedron(Vector3[] vertices, int[][] faces)
    {
        Vertices = vertices;
        Faces = faces;
        ComputeBounds();
    }

    void ComputeBounds()
    {
        MinBound = new Vector3(double.MaxValue, double.MaxValue, double.MaxValue);
        MaxBound = new Vector3(double.MinValue, double.MinValue, double.MinValue);
        Vector3 sum = new Vector3();

        foreach (var v in Vertices)
        {
            MinBound = new Vector3(Math.Min(MinBound.X, v.X), Math.Min(MinBound.Y, v.Y), Math.Min(MinBound.Z, v.Z));
            MaxBound = new Vector3(Math.Max(MaxBound.X, v.X), Math.Max(MaxBound.Y, v.Y), Math.Max(MaxBound.Z, v.Z));
            sum = sum.Add(v);
        }
        Center = sum.Mul(1.0 / Vertices.Length);
    }

    public Polyhedron Transform(Vector3 translation, Quaternion rotation)
    {
        var newVerts = Vertices.Select(v => rotation.RotatePoint(v).Add(translation)).ToArray();
        return new Polyhedron(newVerts, Faces);
    }

    public bool FitsInContainer(double w, double h, double d)
    {
        var bounds = GetBounds();
        return bounds[0] >= -1e-6 && bounds[1] >= -1e-6 && bounds[2] >= -1e-6 &&
               bounds[3] <= w + 1e-6 && bounds[4] <= h + 1e-6 && bounds[5] <= d + 1e-6;
    }

    public double[] GetBounds()
    {
        double minX = double.MaxValue, minY = double.MaxValue, minZ = double.MaxValue;
        double maxX = double.MinValue, maxY = double.MinValue, maxZ = double.MinValue;

        foreach (var v in Vertices)
        {
            minX = Math.Min(minX, v.X); maxX = Math.Max(maxX, v.X);
            minY = Math.Min(minY, v.Y); maxY = Math.Max(maxY, v.Y);
            minZ = Math.Min(minZ, v.Z); maxZ = Math.Max(maxZ, v.Z);
        }
        return new[] { minX, minY, minZ, maxX, maxY, maxZ };
    }
}

class Placement
{
    public Vector3 Translation;
    public Quaternion Rotation;
    public Polyhedron Transformed;
}

class BinPacker
{
    Polyhedron basePolyhedron;
    double containerW, containerH, containerD;
    List<Placement> placements = new List<Placement>();

    public BinPacker(Polyhedron poly, double w, double h, double d)
    {
        basePolyhedron = poly;
        containerW = w;
        containerH = h;
        containerD = d;
    }

    public void Pack()
    {
        int scale = DetermineScale();

        switch (scale)
        {
            case 1: PackTrivial(); break;
            case 2: PackMedium(); break;
            case 3: PackLarge(); break;
            case 4: PackExtreme(); break;
        }
    }

    int DetermineScale()
    {
        double maxDim = Math.Max(containerW, Math.Max(containerH, containerD));
        if (maxDim <= 10) return 1;
        if (maxDim <= 30) return 2;
        if (maxDim <= 150) return 3;
        return 4;
    }

    void PackTrivial()
    {
        PackWithStrategy(new[] { new Quaternion(1, 0, 0, 0) }, 2);
    }

    void PackMedium()
    {
        var rotations = GenerateRotations(4);
        PackWithStrategy(rotations, 1.5);
    }

    void PackLarge()
    {
        var rotations = GenerateRotations(2);
        PackWithStrategy(rotations, 1);
    }

    void PackExtreme()
    {
        var rotations = new[] { new Quaternion(1, 0, 0, 0) };
        PackWithStrategy(rotations, 0.5);
    }

    Quaternion[] GenerateRotations(int count)
    {
        var rots = new List<Quaternion> { new Quaternion(1, 0, 0, 0) };

        for (int i = 0; i < count; i++)
        {
            var axis = new Vector3(1, 1, 1).Normalize();
            rots.Add(Quaternion.FromAxisAngle(new Vector3(1, 0, 0), Math.PI / 2));
            rots.Add(Quaternion.FromAxisAngle(new Vector3(0, 1, 0), Math.PI / 2));
            rots.Add(Quaternion.FromAxisAngle(new Vector3(0, 0, 1), Math.PI / 2));
        }

        return rots.Distinct(new QuatComparer()).ToArray();
    }

    void PackWithStrategy(Quaternion[] rotations, double gridSpacing)
    {
        double step = Math.Max(gridSpacing, 0.5);
        bool improved = true;

        while (improved && placements.Count < 1000)
        {
            improved = false;
            for (double x = 0; x < containerW; x += step)
            {
                for (double y = 0; y < containerH; y += step)
                {
                    for (double z = 0; z < containerD; z += step)
                    {
                        var pos = new Vector3(x, y, z);
                        foreach (var rot in rotations)
                        {
                            if (TryPlace(pos, rot))
                            {
                                improved = true;
                                break;
                            }
                        }
                        if (improved) break;
                    }
                    if (improved) break;
                }
                if (improved) break;
            }
        }
    }

    bool TryPlace(Vector3 pos, Quaternion rot)
    {
        var transformed = basePolyhedron.Transform(pos, rot);

        if (!transformed.FitsInContainer(containerW, containerH, containerD))
            return false;

        foreach (var p in placements)
        {
            if (CollidesWith(transformed, p.Transformed))
                return false;
        }

        placements.Add(new Placement { Translation = pos, Rotation = rot, Transformed = transformed });
        return true;
    }

    bool CollidesWith(Polyhedron a, Polyhedron b)
    {
        var boundsA = a.GetBounds();
        var boundsB = b.GetBounds();

        double minDist = 1e-6;
        return !(boundsA[3] + minDist < boundsB[0] || boundsB[3] + minDist < boundsA[0] ||
                 boundsA[4] + minDist < boundsB[1] || boundsB[4] + minDist < boundsA[1] ||
                 boundsA[5] + minDist < boundsB[2] || boundsB[5] + minDist < boundsA[2]);
    }

    public void WriteSolution()
    {
        Console.WriteLine(placements.Count);
        foreach (var p in placements)
        {
            Console.WriteLine($"{p.Translation.X:F6} {p.Translation.Y:F6} {p.Translation.Z:F6} {p.Rotation.W:F6} {p.Rotation.X:F6} {p.Rotation.Y:F6} {p.Rotation.Z:F6}");
        }
    }
}

class QuatComparer : IEqualityComparer<Quaternion>
{
    public bool Equals(Quaternion a, Quaternion b) =>
        Math.Abs(a.W - b.W) < 1e-6 && Math.Abs(a.X - b.X) < 1e-6 &&
        Math.Abs(a.Y - b.Y) < 1e-6 && Math.Abs(a.Z - b.Z) < 1e-6;
    public int GetHashCode(Quaternion q) => q.W.GetHashCode();
}

class Program
{
    static void Main()
    {
        string[] lines = System.IO.File.ReadAllLines("input.txt");
        var parts = lines[0].Split();
        int V = int.Parse(parts[0]), F = int.Parse(parts[1]);

        var vertices = new Vector3[V];
        for (int i = 0; i < V; i++)
        {
            var p = lines[i + 1].Split();
            vertices[i] = new Vector3(double.Parse(p[0]), double.Parse(p[1]), double.Parse(p[2]));
        }

        var faces = new int[F][];
        for (int i = 0; i < F; i++)
        {
            var p = lines[V + 1 + i].Split();
            faces[i] = new[] { int.Parse(p[0]), int.Parse(p[1]), int.Parse(p[2]) };
        }

        var containerParts = lines[V + F + 1].Split();
        double W = double.Parse(containerParts[0]);
        double H = double.Parse(containerParts[1]);
        double D = double.Parse(containerParts[2]);

        var poly = new Polyhedron(vertices, faces);
        var packer = new BinPacker(poly, W, H, D);
        packer.Pack();
        packer.WriteSolution();
    }
}
