#include <iostream>
#include <vector>
#include <queue>
#include <set>
#include <unordered_map>
#include <algorithm>
#include <cmath>

using namespace std;

typedef vector<int> Position;

struct State {
    Position pos;
    vector<Position> path;
};

int D;
vector<int> bounds;
set<Position> obstacles;
set<Position> foods;
vector<Position> snake;
int max_turns;

bool isValid(const Position& p) {
    if (p.size() != D) return false;
    for (int i = 0; i < D; i++) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    return true;
}

bool isOccupied(const Position& p) {
    return obstacles.count(p) ||
           (find(snake.begin(), snake.end(), p) != snake.end());
}

vector<int> findPathToFood(const Position& start, const Position& goal) {
    queue<pair<Position, vector<int>>> q;
    set<Position> visited;

    q.push({start, vector<int>()});
    visited.insert(start);

    int maxStates = 100000;
    int stateCount = 0;

    while (!q.empty() && stateCount < maxStates) {
        stateCount++;
        auto [pos, moves] = q.front();
        q.pop();

        if (pos == goal) {
            return moves;
        }

        for (int axis = 0; axis < D; axis++) {
            for (int dir : {-1, 1}) {
                Position next = pos;
                next[axis] += dir;

                if (isValid(next) && !visited.count(next) && !isOccupied(next)) {
                    visited.insert(next);
                    vector<int> newMoves = moves;
                    newMoves.push_back(axis);
                    newMoves.push_back(dir);
                    q.push({next, newMoves});
                }
            }
        }
    }

    return vector<int>();
}

Position findNearestFood(const Position& head) {
    Position nearest = *foods.begin();
    int minDist = 0;
    for (int i = 0; i < D; i++) {
        minDist += abs(head[i] - nearest[i]);
    }

    for (const auto& food : foods) {
        int dist = 0;
        for (int i = 0; i < D; i++) {
            dist += abs(head[i] - food[i]);
        }
        if (dist < minDist) {
            minDist = dist;
            nearest = food;
        }
    }
    return nearest;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int snake_length;
    cin >> D >> max_turns;

    bounds.resize(D);
    for (int i = 0; i < D; i++) {
        cin >> bounds[i];
    }

    cin >> snake_length;
    Position head(D);
    for (int i = 0; i < D; i++) {
        cin >> head[i];
    }
    snake.push_back(head);

    int F;
    cin >> F;
    for (int i = 0; i < F; i++) {
        Position food(D);
        for (int j = 0; j < D; j++) {
            cin >> food[j];
        }
        foods.insert(food);
    }

    int O;
    cin >> O;
    for (int i = 0; i < O; i++) {
        Position obs(D);
        for (int j = 0; j < D; j++) {
            cin >> obs[j];
        }
        obstacles.insert(obs);
    }

    for (int turn = 0; turn < max_turns && !foods.empty(); turn++) {
        Position currentHead = snake.front();
        Position target = findNearestFood(currentHead);

        vector<int> moves = findPathToFood(currentHead, target);

        if (moves.empty()) {
            int axis = turn % D;
            int dir = (turn / D) % 2 == 0 ? 1 : -1;

            Position next = currentHead;
            next[axis] += dir;
            if (!isValid(next) || isOccupied(next)) {
                for (int a = 0; a < D; a++) {
                    for (int d : {1, -1}) {
                        Position candidate = currentHead;
                        candidate[a] += d;
                        if (isValid(candidate) && !isOccupied(candidate)) {
                            axis = a;
                            dir = d;
                            next = candidate;
                            break;
                        }
                    }
                }
            }

            cout << axis << " " << dir << "\n";
            snake.insert(snake.begin(), next);
        } else {
            int axis = moves[0];
            int dir = moves[1];
            cout << axis << " " << dir << "\n";

            Position next = currentHead;
            next[axis] += dir;
            snake.insert(snake.begin(), next);
        }

        if (foods.count(snake.front())) {
            foods.erase(snake.front());
        } else {
            snake.pop_back();
        }
    }

    return 0;
}
