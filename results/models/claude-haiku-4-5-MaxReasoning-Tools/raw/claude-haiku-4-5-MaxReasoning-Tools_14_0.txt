using System;
using System.Collections.Generic;

class MineSweeper
{
    private int height, width;
    private Dictionary<(int, int), char> board;
    private HashSet<(int, int)> knownMines;
    private HashSet<(int, int)> unknownCells;

    static void Main()
    {
        new MineSweeper().Solve();
    }

    void Solve()
    {
        ReadInput();
        PropagateSafeCells();
        OutputResult();
    }

    void ReadInput()
    {
        var line = Console.ReadLine().Split();
        height = int.Parse(line[0]);
        width = int.Parse(line[1]);

        board = new Dictionary<(int, int), char>();
        knownMines = new HashSet<(int, int)>();
        unknownCells = new HashSet<(int, int)>();

        for (int row = 0; row < height; row++)
        {
            string input = Console.ReadLine();
            for (int col = 0; col < width; col++)
            {
                char cell = input[col];
                if (cell == ' ')
                {
                    unknownCells.Add((col, row));
                }
                else
                {
                    board[(col, row)] = cell;
                    if (cell == '*')
                    {
                        knownMines.Add((col, row));
                    }
                }
            }
        }
    }

    void PropagateSafeCells()
    {
        bool changed = true;
        while (changed)
        {
            changed = false;
            var cellsToProcess = new List<(int, int)>(board.Keys);

            foreach (var (col, row) in cellsToProcess)
            {
                char cellChar = board[(col, row)];
                if (cellChar < '0' || cellChar > '8') continue;

                int mineCount = cellChar - '0';
                var neighbors = GetNeighbors(col, row);

                int knownCount = 0;
                var unknownNeighbors = new List<(int, int)>();

                foreach (var neighbor in neighbors)
                {
                    if (knownMines.Contains(neighbor))
                    {
                        knownCount++;
                    }
                    else if (unknownCells.Contains(neighbor))
                    {
                        unknownNeighbors.Add(neighbor);
                    }
                }

                // All mines accounted for - remaining unknowns are safe
                if (knownCount == mineCount && unknownNeighbors.Count > 0)
                {
                    foreach (var cell in unknownNeighbors)
                    {
                        unknownCells.Remove(cell);
                    }
                    changed = true;
                }

                // Mines equal unknowns - all unknowns must be mines
                else if (knownCount + unknownNeighbors.Count == mineCount && unknownNeighbors.Count > 0)
                {
                    foreach (var cell in unknownNeighbors)
                    {
                        knownMines.Add(cell);
                        unknownCells.Remove(cell);
                    }
                    changed = true;
                }
            }
        }
    }

    List<(int, int)> GetNeighbors(int col, int row)
    {
        var result = new List<(int, int)>(8);
        for (int dr = -1; dr <= 1; dr++)
        {
            for (int dc = -1; dc <= 1; dc++)
            {
                if (dr == 0 && dc == 0) continue;
                int nc = col + dc;
                int nr = row + dr;
                if (nr >= 0 && nr < height && nc >= 0 && nc < width)
                {
                    result.Add((nc, nr));
                }
            }
        }
        return result;
    }

    void OutputResult()
    {
        var cells = new List<(int, int)>(unknownCells);
        cells.Sort((a, b) => a.Item1 == b.Item1 ? a.Item2.CompareTo(b.Item2) : a.Item1.CompareTo(b.Item1));

        Console.WriteLine(cells.Count);
        foreach (var (col, row) in cells)
        {
            Console.WriteLine($"{col} {row}");
        }
    }
}
