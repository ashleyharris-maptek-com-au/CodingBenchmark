using System;
using System.Collections.Generic;
using System.Linq;

public class TetrisAI
{
    private int boardWidth;
    private int boardHeight;
    private int[] boardHeights;
    private bool[,] board;
    private int linesCleared;
    private int piecesPlaced;

    private static readonly Dictionary<char, int[][]> PieceShapes = new()
    {
        { 'I', new[] { new[] { 0, 0, 0, 0, 1, 1, 1, 1 }, new[] { 0, 1, 0, 0, 0, 1, 0, 0, 0, 1, 0, 0, 0, 1, 0, 0 } } },
        { 'O', new[] { new[] { 0, 1, 1, 0, 0, 1, 1, 0 } } },
        { 'T', new[] {
            new[] { 0, 1, 0, 1, 1, 1 },
            new[] { 0, 1, 0, 0, 1, 1, 0, 1, 0 },
            new[] { 1, 1, 1, 0, 1, 0 },
            new[] { 0, 1, 0, 1, 1, 0, 0, 1, 0 }
        } },
        { 'S', new[] {
            new[] { 0, 1, 1, 1, 1, 0 },
            new[] { 0, 1, 0, 0, 1, 1, 0, 0, 1 }
        } },
        { 'Z', new[] {
            new[] { 1, 1, 0, 0, 1, 1 },
            new[] { 0, 0, 1, 0, 1, 1, 1, 0, 0 }
        } },
        { 'J', new[] {
            new[] { 0, 1, 0, 0, 1, 0, 1, 1, 0 },
            new[] { 1, 0, 0, 1, 1, 1, 0, 0, 1 },
            new[] { 1, 1, 0, 1, 0, 0, 1, 0, 0 },
            new[] { 1, 0, 0, 0, 0, 1, 1, 1, 0 }
        } },
        { 'L', new[] {
            new[] { 1, 0, 0, 0, 1, 0, 1, 1, 0 },
            new[] { 0, 0, 1, 1, 1, 1, 1, 0, 0 },
            new[] { 1, 1, 0, 1, 0, 0, 1, 0, 0 },
            new[] { 1, 0, 0, 1, 1, 1, 0, 0, 1 }
        } }
    };

    public TetrisAI()
    {
        linesCleared = 0;
        piecesPlaced = 0;
    }

    public void Initialize(int width, int height)
    {
        boardWidth = width;
        boardHeight = height;
        boardHeights = new int[width];
        board = new bool[height, width];
        Array.Clear(boardHeights, 0, width);
    }

    public (int rotation, int column) FindBestMove(char pieceType)
    {
        int[] rotations = GetRotations(pieceType);
        (int, int) bestMove = (0, 0);
        double bestScore = double.MinValue;

        foreach (int rotation in rotations)
        {
            var shape = GetPieceShape(pieceType, rotation);
            var (width, height) = GetShapeDimensions(shape);

            for (int col = 0; col <= boardWidth - width; col++)
            {
                if (CanPlace(shape, col))
                {
                    var testBoard = (bool[,])board.Clone();
                    var testHeights = (int[])boardHeights.Clone();

                    Place(shape, col, testBoard, testHeights);
                    int clearedLines = ClearLines(testBoard);

                    double score = EvaluateBoard(testBoard, testHeights, clearedLines);

                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestMove = (rotation, col);
                    }
                }
            }
        }

        return bestMove;
    }

    private int[] GetRotations(char pieceType)
    {
        int maxRotations = PieceShapes[pieceType].Length;
        if (maxRotations == 1) return new[] { 0 };
        if (maxRotations == 2) return new[] { 0, 1 };
        return new[] { 0, 1, 2, 3 };
    }

    private int[] GetPieceShape(char pieceType, int rotation)
    {
        rotation = rotation % PieceShapes[pieceType].Length;
        return PieceShapes[pieceType][rotation];
    }

    private (int, int) GetShapeDimensions(int[] shape)
    {
        int maxWidth = 4, maxHeight = 4;
        for (int h = 1; h <= 4; h++)
        {
            for (int w = 1; w <= 4; w++)
            {
                if (h * w == shape.Length)
                {
                    bool[] row = new bool[w];
                    for (int i = 0; i < shape.Length; i++)
                    {
                        if (shape[i] == 1) row[i % w] = true;
                    }
                    return (w, h);
                }
            }
        }
        return (4, 4);
    }

    private bool CanPlace(int[] shape, int col)
    {
        var (width, height) = GetShapeDimensions(shape);

        for (int row = 0; row < height; row++)
        {
            for (int c = 0; c < width; c++)
            {
                int idx = row * width + c;
                if (idx < shape.Length && shape[idx] == 1)
                {
                    if (col + c >= boardWidth) return false;
                }
            }
        }
        return true;
    }

    private void Place(int[] shape, int col, bool[,] testBoard, int[] testHeights)
    {
        var (width, height) = GetShapeDimensions(shape);

        for (int row = 0; row < height; row++)
        {
            for (int c = 0; c < width; c++)
            {
                int idx = row * width + c;
                if (idx < shape.Length && shape[idx] == 1)
                {
                    int boardCol = col + c;
                    int dropRow = testHeights[boardCol];
                    if (dropRow < boardHeight)
                    {
                        testBoard[dropRow, boardCol] = true;
                        testHeights[boardCol]++;
                    }
                }
            }
        }
    }

    private int ClearLines(bool[,] testBoard)
    {
        int cleared = 0;
        for (int row = boardHeight - 1; row >= 0; row--)
        {
            bool full = true;
            for (int col = 0; col < boardWidth; col++)
            {
                if (!testBoard[row, col])
                {
                    full = false;
                    break;
                }
            }
            if (full) cleared++;
        }
        return cleared;
    }

    private double EvaluateBoard(bool[,] testBoard, int[] testHeights, int clearedLines)
    {
        double score = clearedLines * 100;

        double maxHeight = testHeights.Max();
        double avgHeight = testHeights.Average();
        double bumpiness = 0;

        for (int i = 0; i < testHeights.Length - 1; i++)
        {
            bumpiness += Math.Abs(testHeights[i] - testHeights[i + 1]);
        }

        int holes = 0;
        for (int col = 0; col < boardWidth; col++)
        {
            bool blockFound = false;
            for (int row = boardHeight - 1; row >= 0; row--)
            {
                if (testBoard[row, col])
                {
                    blockFound = true;
                }
                else if (blockFound && !testBoard[row, col])
                {
                    holes++;
                }
            }
        }

        score += avgHeight * 2;
        score -= maxHeight * 5;
        score -= bumpiness * 1.5;
        score -= holes * 10;

        return score;
    }

    public void ProcessResult(string result)
    {
        if (result.StartsWith("cleared_"))
        {
            int count = int.Parse(result.Split('_')[1]);
            linesCleared += count;
        }
        piecesPlaced++;
    }

    public static void Main()
    {
        var ai = new TetrisAI();
        string line = Console.ReadLine();

        if (line == null) return;

        int[] dims = Array.ConvertAll(line.Split(), int.Parse);
        ai.Initialize(dims[0], dims[1]);

        string pieceInput;
        while ((pieceInput = Console.ReadLine()) != null)
        {
            char piece = pieceInput.Trim()[0];
            var (rotation, column) = ai.FindBestMove(piece);

            Console.WriteLine($"{rotation} {column}");
            Console.Out.Flush();

            string result = Console.ReadLine();
            if (result != null && result.Contains("gameover"))
            {
                break;
            }
            ai.ProcessResult(result ?? "ok");
        }
    }
}
