use std::io::{Read, Write, BufReader, BufWriter};

fn main() {
    let stdin = std::io::stdin();
    let stdout = std::io::stdout();

    let mut reader = BufReader::new(stdin.lock());
    let mut writer = BufWriter::new(stdout.lock());

    // Read header (num_points and num_clusters as binary u32)
    let mut header = [0u8; 8];
    if reader.read_exact(&mut header).is_err() {
        return;
    }

    let num_points = u32::from_le_bytes([header[0], header[1], header[2], header[3]]) as usize;
    let num_clusters = u32::from_le_bytes([header[4], header[5], header[6], header[7]]) as usize;

    if num_points == 0 || num_clusters == 0 {
        return;
    }

    // Read all points
    let mut points = Vec::with_capacity(num_points);
    let mut point_bytes = [0u8; 12];

    for _ in 0..num_points {
        if reader.read_exact(&mut point_bytes).is_err() {
            break;
        }
        let x = f32::from_le_bytes([point_bytes[0], point_bytes[1], point_bytes[2], point_bytes[3]]);
        let y = f32::from_le_bytes([point_bytes[4], point_bytes[5], point_bytes[6], point_bytes[7]]);
        let z = f32::from_le_bytes([point_bytes[8], point_bytes[9], point_bytes[10], point_bytes[11]]);
        points.push([x, y, z]);
    }

    // Choose clustering strategy based on dataset size
    let assignments = if num_points <= 10_000_000 {
        kmeans_full(&points, num_clusters)
    } else {
        kmeans_streaming(&points, num_clusters)
    };

    // Write cluster assignments as binary u32
    for &cluster_id in &assignments {
        let bytes = (cluster_id as u32).to_le_bytes();
        let _ = writer.write_all(&bytes);
    }
    let _ = writer.flush();
}

fn kmeans_full(points: &[[f32; 3]], k: usize) -> Vec<usize> {
    if points.is_empty() || k == 0 {
        return vec![];
    }

    let k = k.min(points.len());

    // Initialize centroids using k-means++ selection
    let mut centroids = Vec::new();
    let mut used_indices = std::collections::HashSet::new();

    centroids.push(points[0]);
    used_indices.insert(0);

    for _ in 1..k {
        let mut max_dist = 0.0;
        let mut best_idx = 0;

        for (i, point) in points.iter().enumerate() {
            if used_indices.contains(&i) {
                continue;
            }

            let mut min_dist = f32::INFINITY;
            for centroid in &centroids {
                let d = distance(point, centroid);
                if d < min_dist {
                    min_dist = d;
                }
            }

            if min_dist > max_dist {
                max_dist = min_dist;
                best_idx = i;
            }
        }

        centroids.push(points[best_idx]);
        used_indices.insert(best_idx);
    }

    // K-means iterations with early convergence detection
    for iteration in 0..50 {
        let mut new_centroids = vec![[0.0, 0.0, 0.0]; k];
        let mut counts = vec![0usize; k];

        // Assign points to nearest centroid
        for point in points.iter() {
            let cluster_id = nearest_centroid(point, &centroids);
            new_centroids[cluster_id][0] += point[0];
            new_centroids[cluster_id][1] += point[1];
            new_centroids[cluster_id][2] += point[2];
            counts[cluster_id] += 1;
        }

        // Update centroids and check convergence
        let mut converged = true;
        for i in 0..k {
            if counts[i] > 0 {
                new_centroids[i][0] /= counts[i] as f32;
                new_centroids[i][1] /= counts[i] as f32;
                new_centroids[i][2] /= counts[i] as f32;

                if distance(&new_centroids[i], &centroids[i]) > 1e-5 {
                    converged = false;
                }
            } else {
                new_centroids[i] = centroids[i];
            }
        }

        centroids = new_centroids;

        if converged {
            break;
        }
    }

    // Final assignment
    let mut assignments = vec![0usize; points.len()];
    for (i, point) in points.iter().enumerate() {
        assignments[i] = nearest_centroid(point, &centroids);
    }

    assignments
}

fn kmeans_streaming(points: &[[f32; 3]], k: usize) -> Vec<usize> {
    if points.is_empty() || k == 0 {
        return vec![];
    }

    let k = k.min(points.len());
    let mut assignments = vec![0usize; points.len()];

    // Initialize centroids from sample of points
    let mut centroids = Vec::new();
    let sample_stride = (points.len() / k).max(1);
    for i in 0..k {
        let idx = (i * sample_stride).min(points.len() - 1);
        centroids.push(points[idx]);
    }

    // Mini-batch k-means with streaming updates
    let batch_size = 10000.max(points.len() / 100);

    for iteration in 0..10 {
        let mut batch_sums = vec![[0.0, 0.0, 0.0]; k];
        let mut batch_counts = vec![0usize; k];
        let mut any_changed = false;

        // Process in batches for memory efficiency
        for batch_start in (0..points.len()).step_by(batch_size) {
            let batch_end = (batch_start + batch_size).min(points.len());

            for i in batch_start..batch_end {
                let cluster_id = nearest_centroid(&points[i], &centroids);
                if assignments[i] != cluster_id {
                    any_changed = true;
                }
                assignments[i] = cluster_id;

                batch_sums[cluster_id][0] += points[i][0];
                batch_sums[cluster_id][1] += points[i][1];
                batch_sums[cluster_id][2] += points[i][2];
                batch_counts[cluster_id] += 1;
            }
        }

        // Update centroids with exponential moving average
        let alpha = 0.3 / ((iteration + 1) as f32).sqrt();
        for i in 0..k {
            if batch_counts[i] > 0 {
                let new_x = batch_sums[i][0] / batch_counts[i] as f32;
                let new_y = batch_sums[i][1] / batch_counts[i] as f32;
                let new_z = batch_sums[i][2] / batch_counts[i] as f32;

                centroids[i][0] = (1.0 - alpha) * centroids[i][0] + alpha * new_x;
                centroids[i][1] = (1.0 - alpha) * centroids[i][1] + alpha * new_y;
                centroids[i][2] = (1.0 - alpha) * centroids[i][2] + alpha * new_z;
            }
        }

        if !any_changed {
            break;
        }
    }

    assignments
}

fn nearest_centroid(point: &[f32; 3], centroids: &[[f32; 3]]) -> usize {
    let mut best_idx = 0;
    let mut best_dist = f32::INFINITY;

    for (i, centroid) in centroids.iter().enumerate() {
        let d = distance(point, centroid);
        if d < best_dist {
            best_dist = d;
            best_idx = i;
        }
    }

    best_idx
}

#[inline]
fn distance(p1: &[f32; 3], p2: &[f32; 3]) -> f32 {
    let dx = p1[0] - p2[0];
    let dy = p1[1] - p2[1];
    let dz = p1[2] - p2[2];
    dx * dx + dy * dy + dz * dz
}
