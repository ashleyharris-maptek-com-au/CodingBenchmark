use std::io::{self, BufRead};
use std::f64::consts::PI;

#[derive(Clone, Debug)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn new(x: f64, y: f64) -> Self {
        Point { x, y }
    }

    fn add(&self, other: &Point) -> Point {
        Point {
            x: self.x + other.x,
            y: self.y + other.y,
        }
    }

    fn rotate(&self, angle: f64) -> Point {
        let cos = angle.cos();
        let sin = angle.sin();
        Point {
            x: self.x * cos - self.y * sin,
            y: self.x * sin + self.y * cos,
        }
    }

    fn distance_to(&self, other: &Point) -> f64 {
        ((self.x - other.x).powi(2) + (self.y - other.y).powi(2)).sqrt()
    }
}

#[derive(Clone, Debug)]
struct Polygon {
    vertices: Vec<Point>,
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        Polygon { vertices }
    }

    fn translate(&self, offset: &Point) -> Polygon {
        let translated: Vec<Point> = self.vertices.iter().map(|v| v.add(offset)).collect();
        Polygon::new(translated)
    }

    fn rotate(&self, angle: f64) -> Polygon {
        let rotated: Vec<Point> = self.vertices.iter().map(|v| v.rotate(angle)).collect();
        Polygon::new(rotated)
    }

    fn transform(&self, angle: f64, offset: &Point) -> Polygon {
        self.rotate(angle).translate(offset)
    }

    fn bounding_box(&self) -> (f64, f64, f64, f64) {
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for v in &self.vertices {
            min_x = min_x.min(v.x);
            max_x = max_x.max(v.x);
            min_y = min_y.min(v.y);
            max_y = max_y.max(v.y);
        }

        (min_x, max_x, min_y, max_y)
    }

    fn centroid(&self) -> Point {
        let sum_x: f64 = self.vertices.iter().map(|v| v.x).sum();
        let sum_y: f64 = self.vertices.iter().map(|v| v.y).sum();
        Point::new(sum_x / self.vertices.len() as f64, sum_y / self.vertices.len() as f64)
    }
}

fn point_in_polygon(point: &Point, polygon: &Polygon) -> bool {
    let mut inside = false;
    let verts = &polygon.vertices;
    let n = verts.len();

    for i in 0..n {
        let v1 = &verts[i];
        let v2 = &verts[(i + 1) % n];

        if (v1.y > point.y) != (v2.y > point.y) {
            let xinters = (v2.x - v1.x) * (point.y - v1.y) / (v2.y - v1.y) + v1.x;
            if point.x < xinters {
                inside = !inside;
            }
        }
    }

    inside
}

fn polygon_inside_polygon(inner: &Polygon, outer: &Polygon) -> bool {
    inner.vertices.iter().all(|v| point_in_polygon(v, outer))
}

fn segment_intersection(p1: &Point, p2: &Point, p3: &Point, p4: &Point) -> bool {
    let ccw = |a: &Point, b: &Point, c: &Point| -> bool {
        (c.y - a.y) * (b.x - a.x) > (b.y - a.y) * (c.x - a.x)
    };

    ccw(p1, p3, p4) != ccw(p2, p3, p4) && ccw(p1, p2, p3) != ccw(p1, p2, p4)
}

fn polygons_overlap(poly1: &Polygon, poly2: &Polygon) -> bool {
    let verts1 = &poly1.vertices;
    let verts2 = &poly2.vertices;

    for i in 0..verts1.len() {
        for j in 0..verts2.len() {
            if segment_intersection(
                &verts1[i],
                &verts1[(i + 1) % verts1.len()],
                &verts2[j],
                &verts2[(j + 1) % verts2.len()],
            ) {
                return true;
            }
        }
    }

    if let Some(first) = verts1.first() {
        if point_in_polygon(first, poly2) {
            return true;
        }
    }

    if let Some(first) = verts2.first() {
        if point_in_polygon(first, poly1) {
            return true;
        }
    }

    false
}

struct Placement {
    stock_idx: usize,
    offset: Point,
    rotation: f64,
}

fn find_placement(
    piece: &Polygon,
    stock: &Polygon,
    placed_pieces: &[Polygon],
    num_rotations: usize,
) -> Option<Placement> {
    let (min_x, max_x, min_y, max_y) = stock.bounding_box();
    let stock_width = max_x - min_x;
    let stock_height = max_y - min_y;

    let grid_size = (stock_width * stock_height / 100.0).sqrt().max(5.0).min(50.0) as usize;
    let step_x = stock_width / grid_size as f64;
    let step_y = stock_height / grid_size as f64;

    for rotation_idx in 0..num_rotations {
        let angle = (rotation_idx as f64) * PI * 2.0 / num_rotations as f64;
        let rotated = piece.rotate(angle);

        for i in 0..grid_size {
            for j in 0..grid_size {
                let test_x = min_x + step_x * (i as f64 + 0.5);
                let test_y = min_y + step_y * (j as f64 + 0.5);

                let candidate_centroid = rotated.centroid();
                let offset = Point::new(test_x - candidate_centroid.x, test_y - candidate_centroid.y);
                let transformed = rotated.translate(&offset);

                if !polygon_inside_polygon(&transformed, stock) {
                    continue;
                }

                let mut overlaps = false;
                for placed in placed_pieces {
                    if polygons_overlap(&transformed, placed) {
                        overlaps = true;
                        break;
                    }
                }

                if !overlaps {
                    return Some(Placement {
                        stock_idx: 0,
                        offset,
                        rotation: angle,
                    });
                }
            }
        }
    }

    None
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let sv: usize = parts[0].parse().unwrap();
    let np: usize = parts[1].parse().unwrap();

    let mut stock_vertices = Vec::new();
    for _ in 0..sv {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
        stock_vertices.push(Point::new(coords[0], coords[1]));
    }

    let stock_polygon = Polygon::new(stock_vertices);

    let mut pieces = Vec::new();
    for _ in 0..np {
        let line = lines.next().unwrap().unwrap();
        let pv: usize = line.parse().unwrap();
        let mut piece_vertices = Vec::new();
        for _ in 0..pv {
            let line = lines.next().unwrap().unwrap();
            let coords: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
            piece_vertices.push(Point::new(coords[0], coords[1]));
        }
        pieces.push(Polygon::new(piece_vertices));
    }

    let mut stocks = vec![Vec::new()];
    let mut placements = Vec::new();

    let num_rotations = if np > 1000 { 4 } else if np > 100 { 8 } else { 16 };

    for piece in &pieces {
        let mut placed = false;

        for (stock_idx, placed_pieces) in stocks.iter().enumerate() {
            if let Some(mut placement) = find_placement(piece, &stock_polygon, placed_pieces, num_rotations) {
                placement.stock_idx = stock_idx;
                placements.push(placement);
                stocks[stock_idx].push(piece.transform(placement.rotation, &placement.offset));
                placed = true;
                break;
            }
        }

        if !placed {
            stocks.push(Vec::new());
            let stock_idx = stocks.len() - 1;
            if let Some(mut placement) = find_placement(piece, &stock_polygon, &[], num_rotations) {
                placement.stock_idx = stock_idx;
                placements.push(placement);
                stocks[stock_idx].push(piece.transform(placement.rotation, &placement.offset));
            }
        }
    }

    println!("{}", stocks.len());
    for placement in placements {
        println!("{} {} {} {}", placement.stock_idx, placement.offset.x, placement.offset.y, placement.rotation);
    }
}
