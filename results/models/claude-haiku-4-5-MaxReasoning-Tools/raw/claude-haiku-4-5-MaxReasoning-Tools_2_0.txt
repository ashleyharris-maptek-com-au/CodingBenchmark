using System;
using System.Collections.Generic;
using System.Linq;

class ChinesePostmanSolver
{
    private class Edge
    {
        public int U, V, W;
    }

    static void Main()
    {
        var line = Console.ReadLine().Split();
        int n = int.Parse(line[0]);
        int m = int.Parse(line[1]);

        var edges = new List<Edge>();
        var adj = new List<(int v, int w, int idx)>[n];
        for (int i = 0; i < n; i++) adj[i] = new List<(int, int, int)>();

        for (int i = 0; i < m; i++)
        {
            var parts = Console.ReadLine().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            int w = int.Parse(parts[2]);
            edges.Add(new Edge { U = u, V = v, W = w });
            adj[u].Add((v, w, i));
            adj[v].Add((u, w, i));
        }

        var solver = new ChinesePostmanSolver();
        var route = solver.Solve(n, edges, adj);
        Console.WriteLine(string.Join(" ", route));
    }

    private List<int> Solve(int n, List<Edge> edges, List<(int, int, int)>[] adj)
    {
        var degree = new int[n];
        foreach (var e in edges)
        {
            degree[e.U]++;
            degree[e.V]++;
        }

        var oddVertices = new List<int>();
        for (int i = 0; i < n; i++)
            if (degree[i] % 2 == 1)
                oddVertices.Add(i);

        if (oddVertices.Count == 0)
            return FindEulerianCircuit(n, edges, adj);

        var dist = ComputeOddDistances(oddVertices, n, adj);
        var matching = MinWeightMatching(oddVertices.Count, dist);

        var newAdj = new List<(int, int, int)>[n];
        for (int i = 0; i < n; i++) newAdj[i] = new List<(int, int, int)>();

        for (int i = 0; i < edges.Count; i++)
        {
            var e = edges[i];
            newAdj[e.U].Add((e.V, e.W, i));
            newAdj[e.V].Add((e.U, e.W, i));
        }

        foreach (var (u, v) in matching)
        {
            var path = ShortestPath(u, v, n, adj);
            for (int i = 0; i < path.Count - 1; i++)
            {
                int from = path[i];
                int to = path[i + 1];
                var edge = adj[from].FirstOrDefault(e => e.v == to);
                edges.Add(new Edge { U = from, V = to, W = edge.w });
                newAdj[from].Add((to, edge.w, edges.Count - 1));
                newAdj[to].Add((from, edge.w, edges.Count - 1));
            }
        }

        return FindEulerianCircuit(n, edges, newAdj);
    }

    private int[][] ComputeOddDistances(List<int> oddVertices, int n, List<(int, int, int)>[] adj)
    {
        int k = oddVertices.Count;
        var dist = new int[k][];
        for (int i = 0; i < k; i++)
        {
            dist[i] = Dijkstra(oddVertices[i], n, adj);
            for (int j = 0; j < k; j++)
                dist[i][j] = dist[i][oddVertices[j]];
        }
        return dist;
    }

    private int[] Dijkstra(int start, int n, List<(int, int, int)>[] adj)
    {
        var dist = new int[n];
        for (int i = 0; i < n; i++) dist[i] = int.MaxValue;
        dist[start] = 0;

        var pq = new PriorityQueue<(int, int), int>();
        pq.Enqueue((0, start), 0);

        while (pq.Count > 0)
        {
            var (d, u) = pq.Dequeue();
            if (d > dist[u]) continue;

            foreach (var (v, w, _) in adj[u])
            {
                if (dist[u] + w < dist[v])
                {
                    dist[v] = dist[u] + w;
                    pq.Enqueue((dist[v], v), dist[v]);
                }
            }
        }
        return dist;
    }

    private List<int> ShortestPath(int start, int end, int n, List<(int, int, int)>[] adj)
    {
        var parent = new int[n];
        for (int i = 0; i < n; i++) parent[i] = -1;

        var q = new Queue<int>();
        q.Enqueue(start);
        parent[start] = start;

        while (q.Count > 0)
        {
            int u = q.Dequeue();
            if (u == end) break;

            foreach (var (v, _, _) in adj[u])
            {
                if (parent[v] == -1)
                {
                    parent[v] = u;
                    q.Enqueue(v);
                }
            }
        }

        var path = new List<int>();
        int cur = end;
        while (cur != start)
        {
            path.Add(cur);
            cur = parent[cur];
        }
        path.Add(start);
        path.Reverse();
        return path;
    }

    private List<(int, int)> MinWeightMatching(int k, int[][] dist)
    {
        var matching = new List<(int, int)>();
        var used = new bool[k];

        for (int i = 0; i < k; i++)
        {
            if (used[i]) continue;
            int bestJ = -1, bestDist = int.MaxValue;
            for (int j = i + 1; j < k; j++)
            {
                if (!used[j] && dist[i][j] < bestDist)
                {
                    bestDist = dist[i][j];
                    bestJ = j;
                }
            }
            if (bestJ != -1)
            {
                used[i] = true;
                used[bestJ] = true;
                matching.Add((i, bestJ));
            }
        }
        return matching;
    }

    private List<int> FindEulerianCircuit(int n, List<Edge> edges, List<(int, int, int)>[] adj)
    {
        var stack = new Stack<int>();
        var circuit = new List<int>();
        var used = new bool[edges.Count];

        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            bool found = false;

            foreach (var (v, _, idx) in adj[u])
            {
                if (!used[idx])
                {
                    used[idx] = true;
                    stack.Push(v);
                    found = true;
                    break;
                }
            }

            if (!found)
                circuit.Add(stack.Pop());
        }

        circuit.Reverse();
        return circuit;
    }
}
