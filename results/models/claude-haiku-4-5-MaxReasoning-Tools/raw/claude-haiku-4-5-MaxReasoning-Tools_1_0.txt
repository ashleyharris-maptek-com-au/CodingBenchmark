#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <limits>

using namespace std;

struct City {
    double x, y;
    int id;
};

inline double distance(const City& a, const City& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return sqrt(dx * dx + dy * dy);
}

vector<int> nearestNeighbor(const vector<City>& cities) {
    int n = cities.size();
    vector<int> tour;
    tour.reserve(n);
    vector<bool> visited(n, false);

    tour.push_back(0);
    visited[0] = true;

    for (int i = 1; i < n; i++) {
        int last = tour.back();
        double minDist = numeric_limits<double>::max();
        int nearest = -1;

        for (int j = 0; j < n; j++) {
            if (!visited[j]) {
                double d = distance(cities[last], cities[j]);
                if (d < minDist) {
                    minDist = d;
                    nearest = j;
                }
            }
        }

        tour.push_back(nearest);
        visited[nearest] = true;
    }

    return tour;
}

vector<int> angleSortTour(vector<City> cities) {
    int n = cities.size();

    // Find centroid
    double cx = 0, cy = 0;
    for (const auto& c : cities) {
        cx += c.x;
        cy += c.y;
    }
    cx /= n;
    cy /= n;

    // Sort by angle from centroid - creates a spiral tour
    sort(cities.begin(), cities.end(), [cx, cy](const City& a, const City& b) {
        double aa = atan2(a.y - cy, a.x - cx);
        double ab = atan2(b.y - cy, b.x - cx);
        return aa < ab;
    });

    // Find where city 0 ended up and rotate to put it at start
    int start = 0;
    for (int i = 0; i < n; i++) {
        if (cities[i].id == 0) {
            start = i;
            break;
        }
    }

    vector<int> tour;
    tour.reserve(n);
    for (int i = 0; i < n; i++) {
        tour.push_back(cities[(start + i) % n].id);
    }

    return tour;
}

vector<int> twoOpt(const vector<City>& cities, vector<int> tour, int maxIterations) {
    int n = tour.size();

    for (int iter = 0; iter < maxIterations; iter++) {
        bool improved = false;

        for (int i = 0; i < n - 2; i++) {
            for (int j = i + 2; j < n; j++) {
                int a = tour[i], b = tour[i + 1];
                int c = tour[j], d = tour[(j + 1) % n];

                double oldDist = distance(cities[a], cities[b]) + distance(cities[c], cities[d]);
                double newDist = distance(cities[a], cities[c]) + distance(cities[b], cities[d]);

                if (newDist < oldDist - 1e-10) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                    break;
                }
            }
            if (improved) break;
        }

        if (!improved) break;
    }

    return tour;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<City> cities(n);
    cities.reserve(n);
    for (int i = 0; i < n; i++) {
        cin >> cities[i].x >> cities[i].y;
        cities[i].id = i;
    }

    vector<int> tour;

    if (n <= 10000) {
        // For small instances, use nearest neighbor for good quality
        tour = nearestNeighbor(cities);

        // Apply 2-opt improvement for very small instances
        if (n <= 500) {
            tour = twoOpt(cities, tour, 5);
        }
    } else {
        // For large instances (>10000), use angle sort: O(N log N) instead of O(N^2)
        // This sorts cities by angle from centroid, creating a non-crossing spiral tour
        tour = angleSortTour(cities);
    }

    // Output the tour
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << " ";
        cout << tour[i];
    }
    cout << "\n";

    return 0;
}
