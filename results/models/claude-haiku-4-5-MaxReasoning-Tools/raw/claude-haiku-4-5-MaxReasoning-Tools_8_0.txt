use std::collections::{VecDeque, HashMap};
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read H and W
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let h: usize = parts[0].parse().unwrap();
    let w: usize = parts[1].parse().unwrap();

    // Read maze and find start/goal
    let mut maze = Vec::new();
    let mut start = None;
    let mut goal = None;

    for row in 0..h {
        let line = lines.next().unwrap().unwrap();
        let mut row_chars: Vec<char> = line.chars().collect();

        // Ensure exactly w characters
        while row_chars.len() < w {
            row_chars.push(' ');
        }
        row_chars.truncate(w);

        for (col, &ch) in row_chars.iter().enumerate() {
            if ch == 'A' {
                start = Some((col, row));
            } else if ch == 'B' {
                goal = Some((col, row));
            }
        }

        maze.push(row_chars);
    }

    match (start, goal) {
        (Some(start), Some(goal)) => {
            if let Some(path) = bfs(&maze, start, goal, h, w) {
                println!("{}", path.len());
                for (x, y) in path {
                    println!("{} {}", x, y);
                }
            } else {
                println!("0");
            }
        }
        _ => println!("0"),
    }
}

fn bfs(maze: &[Vec<char>], start: (usize, usize), goal: (usize, usize), h: usize, w: usize) -> Option<Vec<(usize, usize)>> {
    let mut visited = vec![vec![false; w]; h];
    let mut queue = VecDeque::new();
    let mut parent = HashMap::new();

    queue.push_back(start);
    visited[start.1][start.0] = true;

    while let Some((x, y)) = queue.pop_front() {
        if (x, y) == goal {
            // Reconstruct path
            let mut path = vec![(x, y)];
            let mut current = (x, y);
            while current != start {
                current = parent[&current];
                path.push(current);
            }
            path.reverse();
            return Some(path);
        }

        // Check all 4 neighbors: right, left, down, up
        let neighbors = [
            if x + 1 < w { Some((x + 1, y)) } else { None },
            if x > 0 { Some((x - 1, y)) } else { None },
            if y + 1 < h { Some((x, y + 1)) } else { None },
            if y > 0 { Some((x, y - 1)) } else { None },
        ];

        for (nx, ny) in neighbors.into_iter().flatten() {
            if visited[ny][nx] {
                continue;
            }

            if maze[ny][nx] == '#' {
                continue;
            }

            visited[ny][nx] = true;
            parent.insert((nx, ny), (x, y));
            queue.push_back((nx, ny));
        }
    }

    None
}
