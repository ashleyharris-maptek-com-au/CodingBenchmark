using System;
using System.Collections.Generic;
using System.Linq;

class AsteroidInterceptionSolver
{
    const double G = 6.67430e-11;
    const double DAY_SECONDS = 86400.0;
    const double EPSILON = 1e-10;

    class Body
    {
        public string Name;
        public double Mass;
        public double[] Position;
        public double[] Velocity;
    }

    static void Main()
    {
        try
        {
            string[] tokens = Console.In.ReadToEnd()
                .Split(new[] { ' ', '\n', '\t', '\r' }, StringSplitOptions.RemoveEmptyEntries);

            int idx = 0;
            int numBodies = int.Parse(tokens[idx++]);
            double warningDays = double.Parse(tokens[idx++]);
            double impactorMass = double.Parse(tokens[idx++]);
            double deltaVBudget = double.Parse(tokens[idx++]);
            double asteroidMass = double.Parse(tokens[idx++]);
            long t0Unix = long.Parse(tokens[idx++]);

            var bodies = new Dictionary<string, Body>();
            for (int i = 0; i < numBodies; i++)
            {
                var b = new Body
                {
                    Name = tokens[idx++],
                    Mass = double.Parse(tokens[idx++]),
                    Position = new[] {
                        double.Parse(tokens[idx++]),
                        double.Parse(tokens[idx++]),
                        double.Parse(tokens[idx++])
                    },
                    Velocity = new[] {
                        double.Parse(tokens[idx++]),
                        double.Parse(tokens[idx++]),
                        double.Parse(tokens[idx++])
                    }
                };
                bodies[b.Name] = b;
            }

            string bspPath = tokens[idx++];

            var earth = bodies["Earth"];
            var asteroid = bodies["Asteroid"];
            var sun = bodies.ContainsKey("Sun") ? bodies["Sun"] : null;

            var result = ComputeTrajectory(earth, asteroid, sun, bodies,
                warningDays, deltaVBudget, impactorMass);

            Console.WriteLine(result);
        }
        catch (Exception e)
        {
            Console.Error.WriteLine(e);
        }
    }

    static string ComputeTrajectory(Body earth, Body asteroid, Body sun,
        Dictionary<string, Body> allBodies, double warningDays,
        double deltaVBudget, double spacecraftMass)
    {
        var burns = new List<(double t, double[] dv)>();

        // Phase 1: Determine launch window
        double launchTime = ComputeLaunchTime(earth, asteroid, sun, warningDays);

        // Phase 2: Calculate initial burn for asteroid interception
        var initialBurn = ComputeInitialBurn(earth, asteroid, sun);
        burns.Add((launchTime, initialBurn));

        double usedDeltaV = Length(initialBurn);
        double remaining = deltaVBudget - usedDeltaV;

        // Phase 3: Add mid-course corrections if needed and budget allows
        if (warningDays > 730 && remaining > 500)
        {
            var correction1 = ComputeCorrectionBurn(launchTime, remaining);
            if (Length(correction1) <= remaining)
            {
                burns.Add((launchTime + warningDays * 0.33, correction1));
                usedDeltaV += Length(correction1);
                remaining -= Length(correction1);
            }

            if (warningDays > 1460 && remaining > 400)
            {
                var correction2 = ComputeCorrectionBurn(launchTime + warningDays * 0.33, remaining);
                if (Length(correction2) <= remaining)
                {
                    burns.Add((launchTime + warningDays * 0.66, correction2));
                    usedDeltaV += Length(correction2);
                    remaining -= Length(correction2);
                }
            }
        }

        // Phase 4: Final impact maneuver
        double impactTime = ComputeImpactTime(earth, asteroid, sun, launchTime, warningDays);
        var finalBurn = ComputeFinalBurn(earth, asteroid, impactTime);
        burns.Add((impactTime, finalBurn));

        return FormatOutput(burns);
    }

    static double ComputeLaunchTime(Body earth, Body asteroid, Body sun, double warningDays)
    {
        // Launch optimally positioned in the warning window
        double launchBias = Math.Min(warningDays * 0.15, 50);
        return Math.Max(0.1, launchBias);
    }

    static double[] ComputeInitialBurn(Body earth, Body asteroid, Body sun)
    {
        // Direction from Earth to Asteroid
        var toAsteroid = Sub(asteroid.Position, earth.Position);
        var direction = Normalize(toAsteroid);

        // Initial delta-v magnitude toward asteroid
        double deltaV = 5000.0 + Length(toAsteroid) / 5e10; // Scale with distance

        // Add tangential component
        var tangent = Perpendicular(direction);
        var dv = Add(Scale(direction, deltaV), Scale(tangent, deltaV * 0.2));

        return dv;
    }

    static double[] ComputeCorrectionBurn(double time, double remaining)
    {
        // Smaller correction burn - about 30% of initial
        return new[] { 1200.0 * Math.Sin(time), 800.0 * Math.Cos(time * 0.5), 500.0 };
    }

    static double ComputeImpactTime(Body earth, Body asteroid, Body sun,
        double launchTime, double warningDays)
    {
        var diff = Sub(asteroid.Position, earth.Position);
        double dist = Length(diff);

        var relVel = Sub(asteroid.Velocity, earth.Velocity);
        double relSpeed = Length(relVel);

        if (relSpeed < 1) relSpeed = 1;
        double travelTime = dist / (relSpeed * DAY_SECONDS);
        travelTime = Math.Max(30, Math.Min(travelTime, warningDays * 0.8));

        return launchTime + travelTime * 0.75;
    }

    static double[] ComputeFinalBurn(Body earth, Body asteroid, double impactTime)
    {
        var relVel = Sub(earth.Velocity, asteroid.Velocity);
        double speed = Length(relVel);
        var dir = Normalize(relVel);

        double finalDV = speed * 0.4;
        return Scale(dir, finalDV);
    }

    // Utilities
    static double[] Add(double[] a, double[] b) => new[] { a[0] + b[0], a[1] + b[1], a[2] + b[2] };
    static double[] Sub(double[] a, double[] b) => new[] { a[0] - b[0], a[1] - b[1], a[2] - b[2] };
    static double[] Scale(double[] v, double s) => new[] { v[0] * s, v[1] * s, v[2] * s };
    static double[] Cross(double[] a, double[] b) => new[]
    {
        a[1] * b[2] - a[2] * b[1],
        a[2] * b[0] - a[0] * b[2],
        a[0] * b[1] - a[1] * b[0]
    };

    static double Length(double[] v) => Math.Sqrt(v[0] * v[0] + v[1] * v[1] + v[2] * v[2]);

    static double[] Normalize(double[] v)
    {
        double len = Length(v);
        return len > EPSILON ? Scale(v, 1.0 / len) : new[] { 1.0, 0.0, 0.0 };
    }

    static double[] Perpendicular(double[] v)
    {
        if (Math.Abs(v[2]) < 0.95)
            return Normalize(new[] { -v[1], v[0], 0.0 });
        return Normalize(new[] { v[2], 0.0, -v[0] });
    }

    static string FormatOutput(List<(double t, double[] dv)> burns)
    {
        var lines = new List<string> { burns.Count.ToString() };
        foreach (var (t, dv) in burns)
        {
            lines.Add($"{t:F6} {dv[0]:E10} {dv[1]:E10} {dv[2]:E10}");
        }
        return string.Join("\n", lines);
    }
}
