using System;
using System.Collections.Generic;
using System.Linq;

struct Orbit
{
    public double X, Y, Z, VX, VY, VZ;
}

class Program
{
    static void Main()
    {
        var line = Console.ReadLine();
        if (line.StartsWith("﻿"))
            line = line.Substring(1);
        var input = line.Trim().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int n = int.Parse(input[0]);
        double mu = double.Parse(input[1]);

        var start = ParseOrbit(Console.ReadLine());
        var stations = new Orbit[n];
        for (int i = 0; i < n; i++)
            stations[i] = ParseOrbit(Console.ReadLine());

        var solver = new TSPSolver(start, stations, mu);
        var result = solver.Solve();

        Console.WriteLine(string.Join(" ", result));
    }

    static Orbit ParseOrbit(string line)
    {
        var parts = line.Trim().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        return new Orbit
        {
            X = double.Parse(parts[0]),
            Y = double.Parse(parts[1]),
            Z = double.Parse(parts[2]),
            VX = double.Parse(parts[3]),
            VY = double.Parse(parts[4]),
            VZ = double.Parse(parts[5])
        };
    }
}

class TSPSolver
{
    private Orbit start;
    private Orbit[] stations;
    private double mu;
    private double[][] costMatrix;
    private int n;

    public TSPSolver(Orbit start, Orbit[] stations, double mu)
    {
        this.start = start;
        this.stations = stations;
        this.mu = mu;
        this.n = stations.Length;
        this.costMatrix = new double[n][];

        ComputeCostMatrix();
    }

    private void ComputeCostMatrix()
    {
        for (int i = 0; i < n; i++)
        {
            costMatrix[i] = new double[n];
            for (int j = 0; j < n; j++)
            {
                if (i == j)
                    costMatrix[i][j] = 0;
                else
                    costMatrix[i][j] = EstimateTransferCost(stations[i], stations[j]);
            }
        }
    }

    private double EstimateTransferCost(Orbit from, Orbit to)
    {
        double dx = to.X - from.X;
        double dy = to.Y - from.Y;
        double dz = to.Z - from.Z;
        double distance = Math.Sqrt(dx * dx + dy * dy + dz * dz);

        double fromVel = Math.Sqrt(from.VX * from.VX + from.VY * from.VY + from.VZ * from.VZ);
        double toVel = Math.Sqrt(to.VX * to.VX + to.VY * to.VY + to.VZ * to.VZ);

        double fromR = Math.Sqrt(from.X * from.X + from.Y * from.Y + from.Z * from.Z);
        double toR = Math.Sqrt(to.X * to.X + to.Y * to.Y + to.Z * to.Z);

        double deltaV = Math.Abs(fromVel - toVel);
        deltaV += Math.Abs(toR - fromR) / 1000.0;
        deltaV += Math.Sqrt(dx * dx + dy * dy + dz * dz) / 10000.0;

        return deltaV * 1e6;
    }

    public int[] Solve()
    {
        if (n <= 1)
            return new int[] { 0 };

        if (n <= 10)
            return SolveSmall();
        else if (n <= 20)
            return SolveMedium();
        else
            return SolveLarge();
    }

    private int[] SolveSmall()
    {
        var best = Enumerable.Range(0, n).ToArray();
        double bestCost = CalculatePathCost(best);

        var perm = new int[n];
        Array.Copy(best, perm, n);

        if (n <= 9)
        {
            GeneratePermutations(perm, 0, ref best, ref bestCost);
        }
        else
        {
            var improved = LocalSearch(best, 10000);
            return improved;
        }

        return best;
    }

    private int[] SolveMedium()
    {
        var best = NearestNeighbor();
        best = TwoOpt(best, 30000);
        return best;
    }

    private int[] SolveLarge()
    {
        var best = NearestNeighbor();
        best = TwoOptFast(best, 5000);
        return best;
    }

    private void GeneratePermutations(int[] arr, int k, ref int[] best, ref double bestCost)
    {
        if (k == arr.Length - 1)
        {
            double cost = CalculatePathCost(arr);
            if (cost < bestCost)
            {
                bestCost = cost;
                Array.Copy(arr, best, arr.Length);
            }
            return;
        }

        for (int i = k; i < arr.Length; i++)
        {
            Swap(ref arr[k], ref arr[i]);
            GeneratePermutations(arr, k + 1, ref best, ref bestCost);
            Swap(ref arr[k], ref arr[i]);
        }
    }

    private void Swap(ref int a, ref int b)
    {
        int temp = a;
        a = b;
        b = temp;
    }

    private int[] NearestNeighbor()
    {
        var visited = new bool[n];
        var path = new int[n];

        int current = 0;
        path[0] = 0;
        visited[0] = true;

        for (int i = 1; i < n; i++)
        {
            int nearest = -1;
            double minCost = double.MaxValue;

            for (int j = 0; j < n; j++)
            {
                if (!visited[j] && costMatrix[current][j] < minCost)
                {
                    minCost = costMatrix[current][j];
                    nearest = j;
                }
            }

            current = nearest;
            path[i] = current;
            visited[current] = true;
        }

        return path;
    }

    private int[] TwoOpt(int[] path, int maxIterations)
    {
        var best = (int[])path.Clone();
        double bestCost = CalculatePathCost(best);
        bool improved = true;
        int iterations = 0;

        while (improved && iterations < maxIterations)
        {
            improved = false;
            iterations++;

            for (int i = 0; i < n - 2; i++)
            {
                for (int j = i + 2; j < n; j++)
                {
                    double delta = costMatrix[best[i]][best[j]] + costMatrix[best[i + 1]][best[(j + 1) % n]]
                                  - costMatrix[best[i]][best[i + 1]] - costMatrix[best[j]][best[(j + 1) % n]];

                    if (delta < -1e-9)
                    {
                        Array.Reverse(best, i + 1, j - i);
                        bestCost += delta;
                        improved = true;
                    }
                }
            }
        }

        return best;
    }

    private int[] TwoOptFast(int[] path, int maxIterations)
    {
        var best = (int[])path.Clone();
        bool improved = true;
        int iterations = 0;

        while (improved && iterations < maxIterations)
        {
            improved = false;
            iterations++;

            for (int i = 0; i < Math.Min(n - 2, 15); i++)
            {
                for (int j = i + 2; j < Math.Min(n, i + 12); j++)
                {
                    double delta = costMatrix[best[i]][best[j]] + costMatrix[best[i + 1]][best[(j + 1) % n]]
                                  - costMatrix[best[i]][best[i + 1]] - costMatrix[best[j]][best[(j + 1) % n]];

                    if (delta < -1e-9)
                    {
                        Array.Reverse(best, i + 1, j - i);
                        improved = true;
                    }
                }
            }
        }

        return best;
    }

    private int[] LocalSearch(int[] path, int maxIterations)
    {
        var best = (int[])path.Clone();
        double bestCost = CalculatePathCost(best);
        int iterations = 0;

        while (iterations < maxIterations)
        {
            bool found = false;
            iterations++;

            for (int i = 0; i < n && !found; i++)
            {
                for (int j = i + 1; j < n && !found; j++)
                {
                    Swap(ref best[i], ref best[j]);
                    double newCost = CalculatePathCost(best);

                    if (newCost < bestCost)
                    {
                        bestCost = newCost;
                        found = true;
                    }
                    else
                    {
                        Swap(ref best[i], ref best[j]);
                    }
                }
            }

            if (!found)
                break;
        }

        return best;
    }

    private double CalculatePathCost(int[] path)
    {
        double cost = 0;
        for (int i = 0; i < n; i++)
        {
            int next = (i + 1) % n;
            cost += costMatrix[path[i]][path[next]];
        }
        return cost;
    }
}
