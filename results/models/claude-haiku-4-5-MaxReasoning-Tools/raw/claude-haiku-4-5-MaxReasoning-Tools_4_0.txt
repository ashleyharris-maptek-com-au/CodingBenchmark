#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <array>

using namespace std;

struct Vec3 {
    double x, y, z;

    Vec3(double x = 0, double y = 0, double z = 0) : x(x), y(y), z(z) {}

    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double t) const { return Vec3(x * t, y * t, z * t); }

    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const { return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x); }

    double length() const { return sqrt(x * x + y * y + z * z); }
    double lengthSq() const { return x * x + y * y + z * z; }
};

bool canPlaceTetrahedron(const Vec3& center, double circumradius,
                         const vector<Vec3>& placedCenters) {
    double minDist = circumradius * 2.0 * 0.98;
    for (const auto& other : placedCenters) {
        if ((center - other).lengthSq() < minDist * minDist) {
            return false;
        }
    }
    return true;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double edge_length;
    cin >> V >> edge_length;

    vector<Vec3> vertices(V);
    Vec3 centroid;

    for (int i = 0; i < V; i++) {
        cin >> vertices[i].x >> vertices[i].y >> vertices[i].z;
        centroid = centroid + vertices[i];
    }
    centroid = centroid * (1.0 / V);

    double height = edge_length * sqrt(2.0 / 3.0);
    double circumradius = edge_length * sqrt(3.0 / 8.0);
    double inradius = edge_length / sqrt(24.0);

    Vec3 minBound = vertices[0];
    Vec3 maxBound = vertices[0];
    for (const auto& v : vertices) {
        minBound.x = min(minBound.x, v.x);
        minBound.y = min(minBound.y, v.y);
        minBound.z = min(minBound.z, v.z);
        maxBound.x = max(maxBound.x, v.x);
        maxBound.y = max(maxBound.y, v.y);
        maxBound.z = max(maxBound.z, v.z);
    }

    double gridSpacing = circumradius * 2.0;
    double margin = circumradius * 0.5;

    vector<Vec3> placedCenters;

    // Grid-based packing
    for (double x = minBound.x + margin; x < maxBound.x - margin; x += gridSpacing) {
        for (double y = minBound.y + margin; y < maxBound.y - margin; y += gridSpacing) {
            for (double z = minBound.z + margin; z < maxBound.z - margin; z += gridSpacing) {
                Vec3 candidate(x, y, z);
                if (canPlaceTetrahedron(candidate, circumradius, placedCenters)) {
                    placedCenters.push_back(candidate);
                }
            }
        }
    }

    // Staggered grids
    double offset = gridSpacing * 0.5;
    for (double x = minBound.x + margin + offset; x < maxBound.x - margin; x += gridSpacing) {
        for (double y = minBound.y + margin + offset; y < maxBound.y - margin; y += gridSpacing) {
            for (double z = minBound.z + margin; z < maxBound.z - margin; z += gridSpacing) {
                Vec3 candidate(x, y, z);
                if (canPlaceTetrahedron(candidate, circumradius, placedCenters)) {
                    placedCenters.push_back(candidate);
                }
            }
        }
    }

    for (double x = minBound.x + margin; x < maxBound.x - margin; x += gridSpacing) {
        for (double y = minBound.y + margin + offset; y < maxBound.y - margin; y += gridSpacing) {
            for (double z = minBound.z + margin + offset; z < maxBound.z - margin; z += gridSpacing) {
                Vec3 candidate(x, y, z);
                if (canPlaceTetrahedron(candidate, circumradius, placedCenters)) {
                    placedCenters.push_back(candidate);
                }
            }
        }
    }

    cout << placedCenters.size() << "\n";
    for (const auto& center : placedCenters) {
        cout << center.x << " " << center.y << " " << center.z << " 0 0 0\n";
    }

    return 0;
}
