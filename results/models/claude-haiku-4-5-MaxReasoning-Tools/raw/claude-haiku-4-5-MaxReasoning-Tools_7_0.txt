#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>

using namespace std;

const double EPS = 1e-10;

struct Vec3 {
    double x, y, z;

    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x, double y, double z) : x(x), y(y), z(z) {}

    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const {
        return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x);
    }
    double length() const { return sqrt(x * x + y * y + z * z); }
    Vec3 normalize() const {
        double len = length();
        return len > EPS ? Vec3(x / len, y / len, z / len) : Vec3(1, 0, 0);
    }
};

struct Triangle {
    int a, b, c;
};

struct Mesh {
    vector<Vec3> vertices;
    vector<Triangle> triangles;
};

bool rayTriangleIntersect(const Vec3& origin, const Vec3& dir, const Vec3& v0,
                          const Vec3& v1, const Vec3& v2) {
    Vec3 e1 = v1 - v0;
    Vec3 e2 = v2 - v0;
    Vec3 h = dir.cross(e2);
    double a = e1.dot(h);

    if (fabs(a) < EPS) return false;

    double f = 1.0 / a;
    Vec3 s = origin - v0;
    double u = f * s.dot(h);

    if (u < -EPS || u > 1.0 + EPS) return false;

    Vec3 q = s.cross(e1);
    double v = f * dir.dot(q);

    if (v < -EPS || u + v > 1.0 + EPS) return false;

    double t = f * e2.dot(q);
    return t > EPS;
}

bool isInsideMesh(const Vec3& point, const Mesh& mesh) {
    // Try multiple ray directions for robustness
    double dirs[][3] = {{1, 0, 0}, {0, 1, 0}, {0, 0, 1}, {0.577, 0.577, 0.577}};

    for (int d = 0; d < 4; d++) {
        Vec3 dir(dirs[d][0], dirs[d][1], dirs[d][2]);
        dir = dir.normalize();

        int count = 0;
        for (const auto& tri : mesh.triangles) {
            if (rayTriangleIntersect(point, dir, mesh.vertices[tri.a],
                                     mesh.vertices[tri.b], mesh.vertices[tri.c])) {
                count++;
            }
        }

        if (count % 2 == 1) return true;
    }
    return false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA, FA, VB, FB;
    cin >> VA >> FA >> VB >> FB;

    Mesh A, B;

    for (int i = 0; i < VA; i++) {
        double x, y, z;
        cin >> x >> y >> z;
        A.vertices.push_back(Vec3(x, y, z));
    }

    for (int i = 0; i < FA; i++) {
        int a, b, c;
        cin >> a >> b >> c;
        A.triangles.push_back({a, b, c});
    }

    for (int i = 0; i < VB; i++) {
        double x, y, z;
        cin >> x >> y >> z;
        B.vertices.push_back(Vec3(x, y, z));
    }

    for (int i = 0; i < FB; i++) {
        int a, b, c;
        cin >> a >> b >> c;
        B.triangles.push_back({a, b, c});
    }

    vector<Vec3> resultVerts;
    vector<Triangle> resultTris;

    // Add vertices from A
    for (const auto& v : A.vertices) {
        resultVerts.push_back(v);
    }

    // Add triangles from A that are outside B
    for (const auto& tri : A.triangles) {
        Vec3 center = (A.vertices[tri.a] + A.vertices[tri.b] + A.vertices[tri.c]) * (1.0 / 3.0);
        if (!isInsideMesh(center, B)) {
            resultTris.push_back(tri);
        }
    }

    // Add vertices from B and update indices
    int offset = A.vertices.size();
    for (const auto& v : B.vertices) {
        resultVerts.push_back(v);
    }

    // Add triangles from B that are outside A
    for (const auto& tri : B.triangles) {
        Vec3 center = (B.vertices[tri.a] + B.vertices[tri.b] + B.vertices[tri.c]) * (1.0 / 3.0);
        if (!isInsideMesh(center, A)) {
            resultTris.push_back({tri.a + offset, tri.b + offset, tri.c + offset});
        }
    }

    // Output result
    cout << resultVerts.size() << " " << resultTris.size() << "\n";
    cout.precision(15);
    cout << fixed;

    for (const auto& v : resultVerts) {
        cout << v.x << " " << v.y << " " << v.z << "\n";
    }

    for (const auto& tri : resultTris) {
        cout << tri.a << " " << tri.b << " " << tri.c << "\n";
    }

    return 0;
}
