use std::io::{self, BufRead};

const PI: f64 = std::f64::consts::PI;

#[derive(Clone, Copy, Debug)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    fn new(x: f64, y: f64) -> Self {
        Vec2 { x, y }
    }

    fn zero() -> Self {
        Vec2 { x: 0.0, y: 0.0 }
    }

    fn add(&self, other: Vec2) -> Vec2 {
        Vec2 {
            x: self.x + other.x,
            y: self.y + other.y,
        }
    }

    fn sub(&self, other: Vec2) -> Vec2 {
        Vec2 {
            x: self.x - other.x,
            y: self.y - other.y,
        }
    }

    fn mul(&self, scalar: f64) -> Vec2 {
        Vec2 {
            x: self.x * scalar,
            y: self.y * scalar,
        }
    }

    fn length(&self) -> f64 {
        (self.x * self.x + self.y * self.y).sqrt()
    }

    fn normalize(&self) -> Vec2 {
        let len = self.length();
        if len < 1e-10 {
            Vec2::zero()
        } else {
            self.mul(1.0 / len)
        }
    }
}

struct Graph {
    n: usize,
    edges: Vec<(usize, usize)>,
    adj: Vec<Vec<usize>>,
}

impl Graph {
    fn new(n: usize) -> Self {
        Graph {
            n,
            edges: Vec::new(),
            adj: vec![Vec::new(); n],
        }
    }

    fn add_edge(&mut self, u: usize, v: usize) {
        if u != v && u < self.n && v < self.n {
            self.edges.push((u, v));
            self.adj[u].push(v);
            if u != v {
                self.adj[v].push(u);
            }
        }
    }
}

fn layout_force_directed(graph: &Graph, positions: &mut [Vec2]) {
    let n = graph.n;
    if n == 0 {
        return;
    }

    let mut forces = vec![Vec2::zero(); n];

    let area = (n as f64).sqrt() * 100.0;
    let k = (area / (n as f64).sqrt()).max(1.0);
    let k_repel = k * k * 0.5;

    let iterations = match n {
        0..=50 => 200,
        51..=200 => 100,
        201..=1000 => 50,
        1001..=5000 => 20,
        _ => 10,
    };

    for iter in 0..iterations {
        for i in 0..n {
            forces[i] = Vec2::zero();
        }

        for i in 0..n {
            for j in (i + 1)..n {
                let delta = positions[j].sub(positions[i]);
                let dist = delta.length().max(0.01);
                let repel_force = delta.normalize().mul(k_repel / (dist * dist));
                forces[i] = forces[i].sub(repel_force);
                forces[j] = forces[j].add(repel_force);
            }
        }

        for &(u, v) in &graph.edges {
            let delta = positions[v].sub(positions[u]);
            let dist = delta.length().max(0.01);
            let attract = delta.normalize().mul((dist - k) / k);
            let force_mag = attract.length().min(5.0);
            let force = attract.normalize().mul(force_mag);
            forces[u] = forces[u].add(force);
            forces[v] = forces[v].sub(force);
        }

        let max_disp = iterations as f64;
        let cooling = 1.0 - (iter as f64) / (iterations as f64);
        let max_step = cooling * max_disp;

        for i in 0..n {
            let disp = forces[i];
            let disp_len = disp.length();
            if disp_len > 0.0 {
                let step = (disp_len.min(max_step)) / disp_len;
                positions[i] = positions[i].add(disp.mul(step));
            }
        }
    }
}

fn layout_circular_ordered(graph: &Graph, positions: &mut [Vec2]) {
    let n = graph.n;
    if n == 0 {
        return;
    }

    let radius = (n as f64).sqrt() * 30.0;
    let mut degree = vec![0usize; n];
    for (u, v) in &graph.edges {
        degree[*u] += 1;
        degree[*v] += 1;
    }

    let mut order: Vec<usize> = (0..n).collect();
    order.sort_by_key(|&i| -(degree[i] as i32));

    for (idx, &node) in order.iter().enumerate() {
        let angle = 2.0 * PI * (idx as f64) / (n as f64);
        positions[node] = Vec2::new(radius * angle.cos(), radius * angle.sin());
    }

    for _ in 0..10 {
        let mut forces = vec![Vec2::zero(); n];

        for i in 0..n {
            for j in (i + 1)..n {
                let delta = positions[j].sub(positions[i]);
                let dist = delta.length().max(1.0);
                let repel = delta.normalize().mul(100.0 / dist);
                forces[i] = forces[i].sub(repel);
                forces[j] = forces[j].add(repel);
            }
        }

        for &(u, v) in &graph.edges {
            let delta = positions[v].sub(positions[u]);
            let dist = delta.length().max(0.1);
            let target_dist = 50.0;
            if dist < target_dist {
                let repel = delta.normalize().mul(5.0);
                forces[u] = forces[u].sub(repel);
                forces[v] = forces[v].add(repel);
            }
        }

        for i in 0..n {
            let disp = forces[i].mul(0.1);
            positions[i] = positions[i].add(disp);
        }
    }
}

fn layout_grid_based(graph: &Graph, positions: &mut [Vec2]) {
    let n = graph.n;
    if n == 0 {
        return;
    }

    let side = (n as f64).sqrt().ceil() as usize;
    let spacing = 100.0;

    for i in 0..n {
        let row = (i / side) as f64;
        let col = (i % side) as f64;
        positions[i] = Vec2::new(col * spacing, row * spacing);
    }

    for _ in 0..5 {
        let mut forces = vec![Vec2::zero(); n];

        for i in 0..n {
            for j in (i + 1)..n {
                let delta = positions[j].sub(positions[i]);
                let dist = delta.length().max(1.0);
                if dist < 200.0 {
                    let repel = delta.normalize().mul(1000.0 / (dist * dist));
                    forces[i] = forces[i].sub(repel);
                    forces[j] = forces[j].add(repel);
                }
            }
        }

        for &(u, v) in &graph.edges {
            let delta = positions[v].sub(positions[u]);
            let dist = delta.length().max(0.1);
            let target_dist = 80.0;
            let error = dist - target_dist;
            let force = delta.normalize().mul(error * 0.1);
            forces[u] = forces[u].add(force);
            forces[v] = forces[v].sub(force);
        }

        for i in 0..n {
            let disp = forces[i].mul(0.05);
            positions[i] = positions[i].add(disp);
        }
    }
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let n: usize = parts[0].parse().unwrap();
    let m: usize = parts[1].parse().unwrap();

    let mut graph = Graph::new(n);

    for _ in 0..m {
        if let Ok(line) = lines.next().unwrap() {
            let parts: Vec<&str> = line.split_whitespace().collect();
            if parts.len() >= 2 {
                let u: usize = parts[0].parse().unwrap();
                let v: usize = parts[1].parse().unwrap();
                graph.add_edge(u, v);
            }
        }
    }

    let mut positions = vec![Vec2::zero(); n];

    if n == 0 {
        println!("0");
        return;
    }

    if n == 1 {
        println!("0 0");
        return;
    }

    let radius = (n as f64).sqrt() * 50.0;
    for i in 0..n {
        let angle = 2.0 * PI * (i as f64) / (n as f64);
        positions[i] = Vec2::new(radius * angle.cos(), radius * angle.sin());
    }

    match n {
        0..=500 => {
            layout_force_directed(&graph, &mut positions);
        }
        501..=5000 => {
            layout_circular_ordered(&graph, &mut positions);
            layout_force_directed(&graph, &mut positions);
        }
        _ => {
            layout_grid_based(&graph, &mut positions);
            layout_circular_ordered(&graph, &mut positions);
        }
    }

    for i in 0..n {
        println!("{} {}", positions[i].x, positions[i].y);
    }
}
