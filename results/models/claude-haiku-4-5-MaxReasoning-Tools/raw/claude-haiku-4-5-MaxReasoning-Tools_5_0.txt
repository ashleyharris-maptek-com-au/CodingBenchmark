use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let width: usize = parts[0].parse().unwrap();
    let height: usize = parts[1].parse().unwrap();
    let obstacle_count: usize = parts[2].parse().unwrap();
    let require_cycle: bool = parts[3].parse::<i32>().unwrap() != 0;

    let mut obstacles = HashSet::new();
    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<&str> = line.split_whitespace().collect();
        let x: usize = coords[0].parse().unwrap();
        let y: usize = coords[1].parse().unwrap();
        obstacles.insert((x, y));
    }

    let total_cells = width * height - obstacle_count;

    if total_cells == 0 {
        println!("0");
        return;
    }

    // For larger grids with low obstacle density, try constructive approach first
    if width * height > 100 && (obstacle_count as f64 / (width * height) as f64) < 0.3 {
        if let Some(path) = try_snake_pattern(width, height, &obstacles, require_cycle) {
            println!("{}", path.len());
            for (x, y) in path {
                println!("{} {}", x, y);
            }
            return;
        }
    }

    // Use backtracking with heuristics
    let mut path = vec![(0, 0)];
    let mut visited = HashSet::new();
    visited.insert((0, 0));

    // Quick check: (0,0) must not be blocked
    if obstacles.contains(&(0, 0)) {
        println!("0");
        return;
    }

    if find_path(&mut path, &mut visited, width, height, total_cells, &obstacles, require_cycle) {
        println!("{}", path.len());
        for (x, y) in path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}

fn try_snake_pattern(width: usize, height: usize, obstacles: &HashSet<(usize, usize)>, require_cycle: bool) -> Option<Vec<(usize, usize)>> {
    let mut path = Vec::new();

    for y in 0..height {
        if y % 2 == 0 {
            for x in 0..width {
                if !obstacles.contains(&(x, y)) {
                    path.push((x, y));
                }
            }
        } else {
            for x in (0..width).rev() {
                if !obstacles.contains(&(x, y)) {
                    path.push((x, y));
                }
            }
        }
    }

    if path.is_empty() || path[0] != (0, 0) {
        return None;
    }

    // Check connectivity
    for i in 1..path.len() {
        let (x1, y1) = path[i - 1];
        let (x2, y2) = path[i];
        if (x1.abs_diff(x2) + y1.abs_diff(y2)) != 1 {
            return None;
        }
    }

    if require_cycle {
        let (last_x, last_y) = path[path.len() - 1];
        if (last_x.abs_diff(0) == 1 && last_y == 0) || (last_x == 0 && last_y.abs_diff(0) == 1) {
            return Some(path);
        }
        return None;
    }

    Some(path)
}

fn is_valid_cell(x: i32, y: i32, width: usize, height: usize) -> bool {
    x >= 0 && x < width as i32 && y >= 0 && y < height as i32
}

fn count_unvisited_neighbors(
    x: usize,
    y: usize,
    width: usize,
    height: usize,
    visited: &HashSet<(usize, usize)>,
    obstacles: &HashSet<(usize, usize)>,
) -> usize {
    let directions = [(0, 1), (1, 0), (0, -1), (-1, 0)];
    let mut count = 0;

    for (dx, dy) in directions {
        let nx = x as i32 + dx;
        let ny = y as i32 + dy;

        if is_valid_cell(nx, ny, width, height) {
            let nx = nx as usize;
            let ny = ny as usize;
            if !visited.contains(&(nx, ny)) && !obstacles.contains(&(nx, ny)) {
                count += 1;
            }
        }
    }

    count
}

fn find_path(
    path: &mut Vec<(usize, usize)>,
    visited: &mut HashSet<(usize, usize)>,
    width: usize,
    height: usize,
    total_cells: usize,
    obstacles: &HashSet<(usize, usize)>,
    require_cycle: bool,
) -> bool {
    if visited.len() == total_cells {
        if require_cycle {
            let (last_x, last_y) = path[path.len() - 1];
            let (start_x, start_y) = (0, 0);

            if (last_x.abs_diff(start_x) == 1 && last_y == start_y) ||
               (last_x == start_x && last_y.abs_diff(start_y) == 1) {
                return true;
            }
            return false;
        }
        return true;
    }

    let (x, y) = path[path.len() - 1];

    let mut next_cells = vec![];
    let directions = [(0, 1), (1, 0), (0, -1), (-1, 0)];

    for (dx, dy) in directions {
        let nx = x as i32 + dx;
        let ny = y as i32 + dy;

        if is_valid_cell(nx, ny, width, height) {
            let nx = nx as usize;
            let ny = ny as usize;

            if !visited.contains(&(nx, ny)) && !obstacles.contains(&(nx, ny)) {
                let neighbor_count = count_unvisited_neighbors(nx, ny, width, height, visited, obstacles);
                next_cells.push((neighbor_count, nx, ny));
            }
        }
    }

    // Prefer cells with more unvisited neighbors (to avoid dead ends)
    next_cells.sort_by(|a, b| b.0.cmp(&a.0));

    for (_, nx, ny) in next_cells {
        path.push((nx, ny));
        visited.insert((nx, ny));

        if find_path(path, visited, width, height, total_cells, obstacles, require_cycle) {
            return true;
        }

        path.pop();
        visited.remove(&(nx, ny));
    }

    false
}
