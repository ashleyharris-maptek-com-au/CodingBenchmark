#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    cin >> J >> M;

    vector<vector<pair<int, int>>> jobs(J);

    // Read input: for each job, store (machine, duration) pairs for each task
    for (int j = 0; j < J; j++) {
        int T;
        cin >> T;
        for (int t = 0; t < T; t++) {
            int machine, duration;
            cin >> machine >> duration;
            jobs[j].push_back({machine, duration});
        }
    }

    // Track when each machine becomes free
    vector<long long> machine_free_time(M, 0);

    // Schedule each job and output results
    for (int j = 0; j < J; j++) {
        long long job_completion_time = 0;

        for (int t = 0; t < jobs[j].size(); t++) {
            int machine = jobs[j][t].first;
            int duration = jobs[j][t].second;

            // Task must start after:
            // 1. Previous task in this job completes
            // 2. Machine becomes available
            long long start_time = max(job_completion_time, machine_free_time[machine]);
            long long end_time = start_time + duration;

            // Output: space-separated pairs of (start_time machine) for each task
            if (t > 0) cout << " ";
            cout << start_time << " " << machine;

            // Update when this machine next becomes free
            machine_free_time[machine] = end_time;

            // Update when this job's tasks are done
            job_completion_time = end_time;
        }
        cout << "\n";
    }

    return 0;
}
