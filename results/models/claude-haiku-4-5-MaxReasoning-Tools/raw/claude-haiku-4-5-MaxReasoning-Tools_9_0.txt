using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        // Read input
        var firstLine = Console.ReadLine().Split();
        int n = int.Parse(firstLine[0]);
        int stockLength = int.Parse(firstLine[1]);

        var cuts = Console.ReadLine().Split().Select(int.Parse).ToArray();

        // Create array with original indices
        var indexedCuts = cuts.Select((length, idx) => (length, idx)).ToArray();

        // Sort by length descending (First Fit Decreasing strategy)
        Array.Sort(indexedCuts, (a, b) => b.length.CompareTo(a.length));

        // Pack cuts into stocks
        var stocks = new List<List<int>>();
        var stockRemaining = new List<int>();

        foreach (var (length, originalIdx) in indexedCuts)
        {
            // Try to fit in existing stock
            int bestStock = -1;
            int minWaste = int.MaxValue;

            for (int i = 0; i < stocks.Count; i++)
            {
                if (stockRemaining[i] >= length)
                {
                    int waste = stockRemaining[i] - length;
                    if (waste < minWaste)
                    {
                        minWaste = waste;
                        bestStock = i;
                    }
                }
            }

            if (bestStock >= 0)
            {
                // Add to existing stock
                stocks[bestStock].Add(originalIdx);
                stockRemaining[bestStock] -= length;
            }
            else
            {
                // Create new stock
                stocks.Add(new List<int> { originalIdx });
                stockRemaining.Add(stockLength - length);
            }
        }

        // Output
        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            Console.WriteLine(string.Join(" ", stock));
        }
    }
}
