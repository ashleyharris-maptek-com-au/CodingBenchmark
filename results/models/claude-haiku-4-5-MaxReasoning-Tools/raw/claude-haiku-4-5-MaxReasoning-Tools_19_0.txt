using System;
using System.Collections.Generic;
using System.Linq;

class Mining
{
    struct Voxel
    {
        public int mass;
        public int value;
    }

    struct Operation
    {
        public int x, y, z;
        public List<(int, int)> path;
        public int destX, destY;
    }

    static Voxel[,,] voxels;
    static bool[,,] removed;
    static int xCount, yCount, zCount;
    static double xBlockSize, yBlockSize, zBlockSize;

    static void Main()
    {
        string header = Console.ReadLine();
        var parts = header.Split(' ');
        xCount = int.Parse(parts[0]);
        yCount = int.Parse(parts[1]);
        zCount = int.Parse(parts[2]);
        xBlockSize = double.Parse(parts[3]);
        yBlockSize = double.Parse(parts[4]);
        zBlockSize = double.Parse(parts[5]);

        voxels = new Voxel[xCount, yCount, zCount];
        removed = new bool[xCount, yCount, zCount];

        // Read voxel data
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    int b = Console.In.Read();
                    voxels[x, y, z].mass = b % 16;
                    voxels[x, y, z].value = b / 16;
                }
            }
        }

        var operations = new List<Operation>();

        // Greedy strategy: prioritize high-value ore that's accessible
        // Build priority list of voxels to extract
        var candidates = new List<(int x, int y, int z, double profit)>();

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    if (!removed[x, y, z] && voxels[x, y, z].mass > 0)
                    {
                        // Check if exposed
                        if (IsExposed(x, y, z))
                        {
                            int mass = voxels[x, y, z].mass;
                            int value = voxels[x, y, z].value;

                            // Calculate profit: revenue - fuel cost
                            double revenue = mass * value * 1000.0;
                            double fuelCost = EstimateFuelCost(x, y, z);
                            double profit = revenue - fuelCost;

                            if (profit > 0 || value > 0) // Extract valuable ore or profitable ore
                            {
                                candidates.Add((x, y, z, profit));
                            }
                        }
                    }
                }
            }
        }

        candidates.Sort((a, b) => b.profit.CompareTo(a.profit));

        // Process candidates
        foreach (var (x, y, z, profit) in candidates.Take(Math.Min(candidates.Count, 10000)))
        {
            if (!removed[x, y, z] && IsExposed(x, y, z))
            {
                var path = FindPath(x, y, z, 0, 0);
                if (path != null && path.Count > 0)
                {
                    var op = new Operation
                    {
                        x = x,
                        y = y,
                        z = z,
                        path = path,
                        destX = 0,
                        destY = 0
                    };
                    operations.Add(op);
                    removed[x, y, z] = true;
                }
            }
        }

        // Output
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.Write($"{op.path.Count} {op.x} {op.y} {op.z}");
            foreach (var step in op.path)
            {
                Console.Write($" {step.Item1} {step.Item2}");
            }
            Console.WriteLine($" {op.destX} {op.destY}");
        }
    }

    static bool IsExposed(int x, int y, int z)
    {
        // Check if voxel is exposed from above or sides
        if (z == zCount - 1) return true; // Top is exposed

        // Check if exposed from above
        if (z + 1 < zCount && (removed[x, y, z + 1] || voxels[x, y, z + 1].mass == 0))
            return true;

        // Check sides
        int[] dx = { 1, -1, 0, 0 };
        int[] dy = { 0, 0, 1, -1 };

        for (int i = 0; i < 4; i++)
        {
            int nx = x + dx[i];
            int ny = y + dy[i];

            if (nx >= 0 && nx < xCount && ny >= 0 && ny < yCount)
            {
                if (removed[nx, ny, z] || voxels[nx, ny, z].mass == 0)
                    return true;
            }
            else if (nx < 0 || nx >= xCount || ny < 0 || ny >= yCount)
            {
                return true; // Edge is exposed
            }
        }

        return false;
    }

    static double EstimateFuelCost(int x, int y, int z)
    {
        int mass = voxels[x, y, z].mass;
        int dist = x + y; // Manhattan distance to (0,0)

        // Simple estimate: mostly flat movement + some slope
        double cost = dist * mass * 1.0;

        return cost;
    }

    static List<(int, int)> FindPath(int fromX, int fromY, int fromZ, int toX, int toY)
    {
        var path = new List<(int, int)>();

        // BFS with slope constraint
        var queue = new Queue<(int x, int y, int z, List<(int, int)> p)>();
        var visited = new HashSet<(int, int, int)>();

        queue.Enqueue((fromX, fromY, fromZ, path));
        visited.Add((fromX, fromY, fromZ));

        while (queue.Count > 0)
        {
            var (x, y, z, curPath) = queue.Dequeue();

            if (x == toX && y == toY)
            {
                return curPath;
            }

            // Try 4 directions
            int[] dx = { 1, -1, 0, 0 };
            int[] dy = { 0, 0, 1, -1 };

            for (int i = 0; i < 4; i++)
            {
                int nx = x + dx[i];
                int ny = y + dy[i];

                if (nx < 0 || nx >= xCount || ny < 0 || ny >= yCount)
                    continue;

                // Find ground level at new position
                int nz = z;
                for (int checkZ = z; checkZ >= 0; checkZ--)
                {
                    if (!removed[nx, ny, checkZ] && voxels[nx, ny, checkZ].mass > 0)
                    {
                        nz = checkZ;
                        break;
                    }
                    if (checkZ == 0) nz = -1;
                }

                if (nz < 0) nz = 0;

                // Check slope constraint
                if (Math.Abs(nz - z) > 1)
                    continue;

                if (!visited.Contains((nx, ny, nz)))
                {
                    visited.Add((nx, ny, nz));
                    var newPath = new List<(int, int)>(curPath);
                    newPath.Add((nx, ny));
                    queue.Enqueue((nx, ny, nz, newPath));
                }
            }
        }

        return path.Count > 0 ? path : null;
    }
}
