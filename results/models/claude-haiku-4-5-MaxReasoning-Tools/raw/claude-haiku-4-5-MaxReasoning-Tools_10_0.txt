#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct Rect {
    int w, h;
    int id;
    int board = -1;
    int x = 0, y = 0;
    int rotated = 0;
};

struct Shelf {
    int y, x_pos, height, width;
};

struct Board {
    int width, height;
    vector<Shelf> shelves;

    Board(int w, int h) : width(w), height(h) {}

    bool canFit(int w, int h, int& out_x, int& out_y) {
        for (const auto& shelf : shelves) {
            int avail_x = shelf.width - shelf.x_pos;
            if (avail_x >= w && h <= shelf.height) {
                out_x = shelf.x_pos;
                out_y = shelf.y;
                return true;
            }
        }

        if (shelves.empty()) {
            if (w <= width && h <= height) {
                out_x = 0;
                out_y = 0;
                return true;
            }
        } else {
            int next_y = shelves.back().y + shelves.back().height;
            if (next_y + h <= height && w <= width) {
                out_x = 0;
                out_y = next_y;
                return true;
            }
        }
        return false;
    }

    void place(int x, int y, int w, int h) {
        // Find or create shelf for this y position
        bool found = false;
        for (auto& shelf : shelves) {
            if (shelf.y == y) {
                shelf.x_pos += w;
                found = true;
                break;
            }
        }

        if (!found) {
            shelves.push_back({y, w, h, width});
        }
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, bw, bh;
    cin >> n >> bw >> bh;

    vector<Rect> rects(n);
    for (int i = 0; i < n; i++) {
        cin >> rects[i].w >> rects[i].h;
        rects[i].id = i;
    }

    // Sort by area descending
    sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        return (long long)a.w * a.h > (long long)b.w * b.h;
    });

    vector<Board> boards;

    for (auto& rect : rects) {
        bool placed = false;

        // Try existing boards
        for (int bi = 0; bi < (int)boards.size() && !placed; bi++) {
            int px, py;

            // Try without rotation
            if (boards[bi].canFit(rect.w, rect.h, px, py)) {
                boards[bi].place(px, py, rect.w, rect.h);
                rect.board = bi;
                rect.x = px;
                rect.y = py;
                rect.rotated = 0;
                placed = true;
            }
            // Try with rotation
            else if (rect.w != rect.h && boards[bi].canFit(rect.h, rect.w, px, py)) {
                boards[bi].place(px, py, rect.h, rect.w);
                rect.board = bi;
                rect.x = px;
                rect.y = py;
                rect.rotated = 1;
                placed = true;
            }
        }

        // Create new board if needed
        if (!placed) {
            boards.emplace_back(bw, bh);
            int px, py;

            if (boards.back().canFit(rect.w, rect.h, px, py)) {
                boards.back().place(px, py, rect.w, rect.h);
                rect.board = boards.size() - 1;
                rect.x = px;
                rect.y = py;
                rect.rotated = 0;
            } else if (rect.w != rect.h && boards.back().canFit(rect.h, rect.w, px, py)) {
                boards.back().place(px, py, rect.h, rect.w);
                rect.board = boards.size() - 1;
                rect.x = px;
                rect.y = py;
                rect.rotated = 1;
            }
        }
    }

    cout << boards.size() << "\n";

    vector<Rect> output = rects;
    sort(output.begin(), output.end(), [](const Rect& a, const Rect& b) {
        return a.id < b.id;
    });

    for (const auto& rect : output) {
        cout << rect.board << " " << rect.x << " " << rect.y << " " << rect.rotated << "\n";
    }

    return 0;
}
