use std::io::{self, BufRead};
use std::collections::HashMap;

#[derive(Debug, Clone)]
struct Sample {
    hole_id: usize,
    sample_idx: usize,
    position: [f64; 3],
    properties: Vec<f64>,
}

#[derive(Debug)]
struct Anomaly {
    hole_id: usize,
    sample_idx: usize,
    property_idx: usize,
    confidence: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Parse header
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let num_holes: usize = parts[0].parse().unwrap();
    let num_properties: usize = parts[1].parse().unwrap();

    // Parse property names
    let prop_names_line = lines.next().unwrap().unwrap();
    let property_names: Vec<String> = prop_names_line
        .split_whitespace()
        .map(|s| s.to_string())
        .collect();

    let mut all_samples = Vec::new();

    // Parse hole data
    for _ in 0..num_holes {
        let hole_line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = hole_line.split_whitespace().collect();

        let hole_id: usize = parts[0].parse().unwrap();
        let start_x: f64 = parts[1].parse().unwrap();
        let start_y: f64 = parts[2].parse().unwrap();
        let start_z: f64 = parts[3].parse().unwrap();
        let dir_x: f64 = parts[4].parse().unwrap();
        let dir_y: f64 = parts[5].parse().unwrap();
        let dir_z: f64 = parts[6].parse().unwrap();
        let num_samples: usize = parts[8].parse().unwrap();

        // Normalize direction vector
        let dir_len = (dir_x * dir_x + dir_y * dir_y + dir_z * dir_z).sqrt();
        let (norm_x, norm_y, norm_z) = if dir_len > 1e-10 {
            (dir_x / dir_len, dir_y / dir_len, dir_z / dir_len)
        } else {
            (0.0, 0.0, 0.0)
        };

        // Parse samples for this hole
        for sample_idx in 0..num_samples {
            let sample_line = lines.next().unwrap().unwrap();
            let sample_parts: Vec<&str> = sample_line.split_whitespace().collect();

            let depth: f64 = sample_parts[0].parse().unwrap();
            let properties: Vec<f64> = sample_parts[1..]
                .iter()
                .filter_map(|s| s.parse::<f64>().ok())
                .collect();

            // Calculate 3D position: start + depth * direction
            let x = start_x + depth * norm_x;
            let y = start_y + depth * norm_y;
            let z = start_z + depth * norm_z;

            all_samples.push(Sample {
                hole_id,
                sample_idx,
                position: [x, y, z],
                properties,
            });
        }
    }

    let mut anomalies = Vec::new();

    // Use adaptive spatial indexing based on dataset size
    if all_samples.len() < 5000 {
        find_anomalies_brute_force(&all_samples, &mut anomalies, 100.0);
    } else {
        find_anomalies_grid_based(&all_samples, &mut anomalies, 50.0);
    }

    // Sort by confidence descending
    anomalies.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap());

    // Output results
    println!("{}", anomalies.len());
    for anomaly in anomalies {
        println!(
            "{} {} {} {:.6}",
            anomaly.hole_id,
            anomaly.sample_idx,
            property_names[anomaly.property_idx],
            anomaly.confidence
        );
    }
}

fn find_anomalies_brute_force(
    samples: &[Sample],
    anomalies: &mut Vec<Anomaly>,
    search_radius: f64,
) {
    let radius_sq = search_radius * search_radius;

    for (idx, sample) in samples.iter().enumerate() {
        // Find neighbors within radius
        let mut neighbors = Vec::new();
        for (other_idx, other) in samples.iter().enumerate() {
            if idx != other_idx {
                let dx = sample.position[0] - other.position[0];
                let dy = sample.position[1] - other.position[1];
                let dz = sample.position[2] - other.position[2];
                let dist_sq = dx * dx + dy * dy + dz * dz;

                if dist_sq <= radius_sq {
                    neighbors.push(other_idx);
                }
            }
        }

        // Check each property for anomalies
        check_anomalies(samples, idx, &neighbors, anomalies);
    }
}

fn find_anomalies_grid_based(
    samples: &[Sample],
    anomalies: &mut Vec<Anomaly>,
    cell_size: f64,
) {
    let mut grid: HashMap<(i32, i32, i32), Vec<usize>> = HashMap::new();

    // Build spatial grid
    for (idx, sample) in samples.iter().enumerate() {
        let cell_x = (sample.position[0] / cell_size).floor() as i32;
        let cell_y = (sample.position[1] / cell_size).floor() as i32;
        let cell_z = (sample.position[2] / cell_size).floor() as i32;

        grid.entry((cell_x, cell_y, cell_z))
            .or_insert_with(Vec::new)
            .push(idx);
    }

    // Find anomalies
    for (idx, sample) in samples.iter().enumerate() {
        let cell_x = (sample.position[0] / cell_size).floor() as i32;
        let cell_y = (sample.position[1] / cell_size).floor() as i32;
        let cell_z = (sample.position[2] / cell_size).floor() as i32;

        // Collect neighbors from nearby cells
        let mut neighbors = Vec::new();
        for dx in -1..=1 {
            for dy in -1..=1 {
                for dz in -1..=1 {
                    let key = (cell_x + dx, cell_y + dy, cell_z + dz);
                    if let Some(cell_samples) = grid.get(&key) {
                        for &other_idx in cell_samples {
                            if other_idx != idx {
                                neighbors.push(other_idx);
                            }
                        }
                    }
                }
            }
        }

        check_anomalies(samples, idx, &neighbors, anomalies);
    }
}

fn check_anomalies(
    samples: &[Sample],
    idx: usize,
    neighbors: &[usize],
    anomalies: &mut Vec<Anomaly>,
) {
    let sample = &samples[idx];

    // Need at least 3 neighbors for meaningful statistics
    if neighbors.len() < 3 {
        return;
    }

    // Check each property
    for prop_idx in 0..sample.properties.len() {
        let value = sample.properties[prop_idx];

        // Collect neighbor values for this property
        let neighbor_values: Vec<f64> = neighbors
            .iter()
            .map(|&i| samples[i].properties[prop_idx])
            .collect();

        // Calculate statistics
        let mean = neighbor_values.iter().sum::<f64>() / neighbor_values.len() as f64;
        let variance = neighbor_values
            .iter()
            .map(|&v| (v - mean) * (v - mean))
            .sum::<f64>()
            / neighbor_values.len() as f64;
        let std_dev = variance.sqrt();

        // Compute z-score as confidence measure
        if std_dev > 1e-10 {
            let z_score = (value - mean).abs() / std_dev;

            // Threshold of 2.0 standard deviations catches most anomalies
            if z_score > 2.0 {
                anomalies.push(Anomaly {
                    hole_id: sample.hole_id,
                    sample_idx: sample.sample_idx,
                    property_idx,
                    confidence: z_score,
                });
            }
        }
    }
}
