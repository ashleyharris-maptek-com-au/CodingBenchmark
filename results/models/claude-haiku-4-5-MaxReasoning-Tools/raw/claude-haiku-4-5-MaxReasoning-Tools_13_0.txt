#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <chrono>
#include <climits>

using namespace std;

auto g_startTime = chrono::high_resolution_clock::now();

bool isTimeout(int seconds = 280) {
    auto now = chrono::high_resolution_clock::now();
    return chrono::duration_cast<chrono::seconds>(now - g_startTime).count() > seconds;
}

bool hasCommonSubstringOfLength(const vector<string>& strings, int len, string& found) {
    if (strings.empty() || isTimeout()) return false;

    // Use shortest string as source for efficiency
    int srcIdx = 0;
    for (int i = 1; i < (int)strings.size(); i++) {
        if (strings[i].length() < strings[srcIdx].length()) {
            srcIdx = i;
        }
    }

    const string& src = strings[srcIdx];
    if ((int)src.length() < len) return false;

    for (int i = 0; i <= (int)src.length() - len; i++) {
        string sub = src.substr(i, len);
        bool inAll = true;

        for (int j = 0; j < (int)strings.size(); j++) {
            if (j == srcIdx) continue;
            if (strings[j].find(sub) == string::npos) {
                inAll = false;
                break;
            }
        }

        if (inAll) {
            found = sub;
            return true;
        }
    }

    return false;
}

string solve(const vector<string>& strings) {
    if (strings.empty()) return "";

    // Find maximum possible length
    int maxLen = INT_MAX;
    for (const auto& s : strings) {
        maxLen = min(maxLen, (int)s.length());
    }

    if (maxLen == INT_MAX) maxLen = 0;

    int left = 1, right = maxLen;
    string result = "";

    while (left <= right) {
        int mid = (left + right) / 2;
        string found;

        if (hasCommonSubstringOfLength(strings, mid, found)) {
            result = found;
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }

    return result;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    g_startTime = chrono::high_resolution_clock::now();

    int n;
    cin >> n;
    cin.ignore();

    vector<string> strings(n);
    for (int i = 0; i < n; i++) {
        getline(cin, strings[i]);
    }

    string result = solve(strings);
    cout << result << endl;

    return 0;
}
