```rust
use std::io::{self, BufReader, BufWriter, Read, Write};

// Maze buffer cell values:
//   '#' (0x23) = wall (unchanged throughout)
//   0xFF       = unvisited passable cell
//   0xFE       = start cell A (no parent)
//   0..3       = visited; direction to reach parent:
//                 0 = UP    (parent at row-1, pos - w)
//                 1 = DOWN  (parent at row+1, pos + w)
//                 2 = LEFT  (parent at col-1, pos - 1)
//                 3 = RIGHT (parent at col+1, pos + 1)

const UNVISITED: u8 = 0xFF;
const START_CELL: u8 = 0xFE;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut reader = BufReader::with_capacity(8 << 20, stdin.lock());
    let writer = BufWriter::with_capacity(64 << 20, stdout.lock());
    solve(&mut reader, writer);
}

fn solve<R: Read, W: Write>(reader: &mut R, mut writer: W) {
    // Read entire input at once for maximum throughput
    let mut raw = Vec::new();
    reader.read_to_end(&mut raw).unwrap();

    let mut p = 0;
    let h = parse_usize(&raw, &mut p);
    let w = parse_usize(&raw, &mut p);

    if h == 0 || w == 0 {
        writer.write_all(b"0\n").unwrap();
        return;
    }

    let total = h * w;
    let w32 = w as u32;

    // Parse maze grid and locate A / B
    let mut maze: Vec<u8> = Vec::with_capacity(total);
    let mut start: u32 = 0;
    let mut goal: u32 = 0;
    let mut has_a = false;
    let mut has_b = false;

    let mut ip = p;
    for r in 0..h {
        // Skip line endings between rows
        while ip < raw.len() && (raw[ip] == b'\n' || raw[ip] == b'\r') {
            ip += 1;
        }
        for c in 0..w {
            let ch = if ip < raw.len() { raw[ip] } else { b'#' };
            ip += 1;
            if ch == b'A' {
                start = (r * w + c) as u32;
                has_a = true;
            } else if ch == b'B' {
                goal = (r * w + c) as u32;
                has_b = true;
            }
            maze.push(ch);
        }
    }

    drop(raw); // free input buffer before BFS

    if !has_a || !has_b {
        writer.write_all(b"0\n").unwrap();
        return;
    }

    // Mark every passable cell as UNVISITED; walls stay '#'
    for cell in maze.iter_mut() {
        if *cell != b'#' {
            *cell = UNVISITED;
        }
    }

    // ── Breadth-First Search ──────────────────────────────────
    maze[start as usize] = START_CELL;
    let mut q: Vec<u32> = Vec::with_capacity(total.min(1 << 20));
    let mut head: usize = 0;
    q.push(start);

    let total_less_w = total - w; // first index of the last row

    while head < q.len() {
        let pos = q[head];
        head += 1;

        if pos == goal {
            break;
        }

        let col = pos % w32;

        // Up neighbour (row - 1)
        if pos >= w32 {
            let np = pos - w32;
            unsafe {
                if *maze.get_unchecked(np as usize) == UNVISITED {
                    *maze.get_unchecked_mut(np as usize) = 1; // go DOWN to reach parent
                    q.push(np);
                }
            }
        }
        // Down neighbour (row + 1)
        if (pos as usize) < total_less_w {
            let np = pos + w32;
            unsafe {
                if *maze.get_unchecked(np as usize) == UNVISITED {
                    *maze.get_unchecked_mut(np as usize) = 0; // go UP to reach parent
                    q.push(np);
                }
            }
        }
        // Left neighbour (col - 1)
        if col > 0 {
            let np = pos - 1;
            unsafe {
                if *maze.get_unchecked(np as usize) == UNVISITED {
                    *maze.get_unchecked_mut(np as usize) = 3; // go RIGHT to reach parent
                    q.push(np);
                }
            }
        }
        // Right neighbour (col + 1)
        if col + 1 < w32 {
            let np = pos + 1;
            unsafe {
                if *maze.get_unchecked(np as usize) == UNVISITED {
                    *maze.get_unchecked_mut(np as usize) = 2; // go LEFT to reach parent
                    q.push(np);
                }
            }
        }
    }

    // No path exists
    if maze[goal as usize] == UNVISITED {
        writer.write_all(b"0\n").unwrap();
        return;
    }

    // ── Reconstruct path from goal back to start ─────────────
    let mut path: Vec<u32> = Vec::new();
    let mut cur = goal;
    loop {
        path.push(cur);
        if cur == start {
            break;
        }
        cur = match maze[cur as usize] {
            0 => cur - w32, // UP:    parent at row-1
            1 => cur + w32, // DOWN:  parent at row+1
            2 => cur - 1,   // LEFT:  parent at col-1
            3 => cur + 1,   // RIGHT: parent at col+1
            _ => break,     // START_CELL / invalid — safety
        };
    }

    // Free BFS structures before output
    drop(maze);
    drop(q);

    // ── Write output ─────────────────────────────────────────
    let n = path.len();
    let mut obuf: Vec<u8> = Vec::with_capacity(64 << 20);

    write_u32(&mut obuf, n as u32);
    obuf.push(b'\n');

    // Iterate path in reverse → A-to-B order
    for i in (0..path.len()).rev() {
        let v = path[i];
        write_u32(&mut obuf, v % w32); // x = column
        obuf.push(b' ');
        write_u32(&mut obuf, v / w32); // y = row
        obuf.push(b'\n');

        if obuf.len() > 63 << 20 {
            writer.write_all(&obuf).unwrap();
            obuf.clear();
        }
    }

    if !obuf.is_empty() {
        writer.write_all(&obuf).unwrap();
    }
    writer.flush().unwrap();
}

// ── Helpers ──────────────────────────────────────────────────

#[inline(always)]
fn parse_usize(data: &[u8], pos: &mut usize) -> usize {
    while *pos < data.len() && matches!(data[*pos], b' ' | b'\n' | b'\r') {
        *pos += 1;
    }
    let mut r: usize = 0;
    while *pos < data.len() && data[*pos].is_ascii_digit() {
        r = r * 10 + (data[*pos] - b'0') as usize;
        *pos += 1;
    }
    r
}

#[inline(always)]
fn write_u32(buf: &mut Vec<u8>, mut n: u32) {
    if n == 0 {
        buf.push(b'0');
        return;
    }
    let mut tmp = [0u8; 10];
    let mut len = 0usize;
    while n > 0 {
        tmp[len] = b'0' + (n % 10) as u8;
        n /= 10;
        len += 1;
    }
    for i in (0..len).rev() {
        buf.push(tmp[i]);
    }
}
```

**Key design decisions and why:**

| Decision | Rationale |
|---|---|
| **Single buffer for maze + parent** | Directions 0-3 replace the cell value after visiting — no separate visited/parent arrays needed. Saves ~5 GB at 50000² scale. |
| **`u32` position encoding** | `row * W + col` fits in u32 since 50000² = 2.5B < 4.3B. Queue entries are 4 bytes instead of 8. |
| **Vec + head index queue** | Avoids `VecDeque` ring-buffer overhead; simple `push`/index pattern. |
| **`unsafe` get_unchecked in BFS** | Bounds already verified by boundary checks — eliminates per-access bounds check in the hottest loop. |
| **read_to_end + raw byte parsing** | Single bulk read, no per-line allocation or UTF-8 validation. |
| **Custom write_u32** | Avoids `fmt::Display` trait dispatch — ~3-4× faster for billions of integer formats. |
| **Manual 64 MB output buffer** | Amortizes write syscalls; flushes at ~63 MB boundaries. |
| **Early `drop(raw)`, `drop(maze+q)`** | Frees large buffers as soon as they're no longer needed, reducing peak memory. |
| **Reverse-iterate path** | Path is built B→A; iterating the Vec in reverse gives A→B order with no extra copy. |
