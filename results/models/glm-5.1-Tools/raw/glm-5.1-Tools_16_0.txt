Here is the complete, compilable C++ solver:

```cpp
#include <cstdio>
#include <cstdlib>
#include <vector>
#include <algorithm>
#include <random>
#include <chrono>
#include <climits>
#include <cstdint>

using namespace std;

struct Task { int machine, duration, start; };
struct IVal { int s, e; };

// Find earliest time >= ready to fit [ready, ready+dur) without overlapping
// any interval in the sorted non-overlapping list iv
static int find_earliest(const vector<IVal>& iv, int ready, int dur) {
    int t = ready, n = (int)iv.size();
    if (!n) return t;
    // Binary search: first interval whose end > ready
    int lo = 0, hi = n;
    while (lo < hi) {
        int mid = (lo + hi) >> 1;
        if (iv[mid].e > ready) hi = mid; else lo = mid + 1;
    }
    // Linear scan: advance t past any blocking intervals
    for (int i = lo; i < n; i++) {
        if (t + dur <= iv[i].s) return t;   // fits before interval i
        if (t < iv[i].e) t = iv[i].e;       // skip past interval i
    }
    return t;
}

// Insert interval [s,e) maintaining sort by start time
static void insert_iv(vector<IVal>& iv, int s, int e) {
    auto it = lower_bound(iv.begin(), iv.end(), IVal{s, e},
                          [](const IVal& a, const IVal& b) { return a.s < b.s; });
    iv.insert(it, {s, e});
}

enum Rule { MWR, MOR, LPT, SPT, PERM };

// Greedy dispatch: at each step schedule the highest-priority job's next task
static void dispatch(vector<vector<Task>>& jobs, vector<vector<IVal>>& mi,
                     int J, int M, Rule rule, const vector<int>* so = nullptr) {
    vector<int> nt(J, 0), jr(J, 0), rw(J, 0), ro(J, 0);
    for (int j = 0; j < J; j++)
        for (auto& t : jobs[j]) { rw[j] += t.duration; ro[j]++; }
    vector<bool> active(J);
    for (int j = 0; j < J; j++) active[j] = !jobs[j].empty();
    int total = 0;
    for (int j = 0; j < J; j++) total += (int)jobs[j].size();
    int done = 0;
    while (done < total) {
        int best = -1;
        int64_t bpri = INT64_MIN;
        for (int j = 0; j < J; j++) {
            if (!active[j]) continue;
            int tb = so ? (*so)[j] : j;
            int64_t pri;
            switch (rule) {
                case MWR: pri = (int64_t)rw[j] * J + tb; break;
                case MOR: pri = (int64_t)ro[j] * J + tb; break;
                case LPT: pri = (int64_t)jobs[j][nt[j]].duration * J + tb; break;
                case SPT: pri = -(int64_t)jobs[j][nt[j]].duration * J + tb; break;
                case PERM: pri = tb; break;
            }
            if (best == -1 || pri > bpri) { best = j; bpri = pri; }
        }
        if (best == -1) break;
        int j = best, ti = nt[j];
        int m = jobs[j][ti].machine, d = jobs[j][ti].duration;
        int s = find_earliest(mi[m], jr[j], d);
        jobs[j][ti].start = s;
        insert_iv(mi[m], s, s + d);
        jr[j] = s + d; rw[j] -= d; ro[j]--; nt[j]++;
        if (nt[j] >= (int)jobs[j].size()) active[j] = false;
        done++;
    }
}

static int makespan(const vector<vector<Task>>& jobs) {
    int ms = 0;
    for (auto& job : jobs) for (auto& t : job) ms = max(ms, t.start + t.duration);
    return ms;
}

// Build machine intervals from scheduled tasks, optionally excluding jobs
static void build_mi(const vector<vector<Task>>& jobs, int M,
                     vector<vector<IVal>>& mi, const vector<bool>* exc = nullptr) {
    mi.assign(M, {});
    for (int j = 0; j < (int)jobs.size(); j++) {
        if (exc && (*exc)[j]) continue;
        for (auto& t : jobs[j])
            if (t.start >= 0) mi[t.machine].push_back({t.start, t.start + t.duration});
    }
    for (int m = 0; m < M; m++)
        sort(mi[m].begin(), mi[m].end(), [](const IVal& a, const IVal& b) { return a.s < b.s; });
}

// Re-insert a set of removed jobs into existing machine intervals
static void resched(vector<vector<Task>>& jobs, vector<vector<IVal>>& mi,
                    const vector<int>& jids, Rule rule) {
    int J = (int)jobs.size();
    for (int j : jids) for (auto& t : jobs[j]) t.start = -1;
    vector<int> nt(J, 0), jr(J, 0), rw(J, 0), ro(J, 0);
    for (int j : jids) for (auto& t : jobs[j]) { rw[j] += t.duration; ro[j]++; }
    vector<bool> active(J, false);
    for (int j : jids) active[j] = true;
    int total = 0;
    for (int j : jids) total += (int)jobs[j].size();
    int done = 0;
    while (done < total) {
        int best = -1, bpri = INT_MIN;
        for (int j : jids) {
            if (!active[j]) continue;
            int pri;
            switch (rule) {
                case MWR: pri = rw[j]; break;
                case MOR: pri = ro[j]; break;
                case LPT: pri = jobs[j][nt[j]].duration; break;
                case SPT: pri = -jobs[j][nt[j]].duration; break;
                default: pri = rw[j]; break;
            }
            if (best == -1 || pri > bpri || (pri == bpri && j < best)) { best = j; bpri = pri; }
        }
        if (best == -1) break;
        int j = best, ti = nt[j];
        int m = jobs[j][ti].machine, d = jobs[j][ti].duration;
        int s = find_earliest(mi[m], jr[j], d);
        jobs[j][ti].start = s;
        insert_iv(mi[m], s, s + d);
        jr[j] = s + d; rw[j] -= d; ro[j]--; nt[j]++;
        if (nt[j] >= (int)jobs[j].size()) active[j] = false;
        done++;
    }
}

int main() {
    int J, M;
    scanf("%d%d", &J, &M);
    vector<vector<Task>> jobs(J);
    for (int j = 0; j < J; j++) {
        int T; scanf("%d", &T);
        jobs[j].resize(T);
        for (int t = 0; t < T; t++) {
            scanf("%d%d", &jobs[j][t].machine, &jobs[j][t].duration);
            jobs[j][t].start = -1;
        }
    }
    int total = 0;
    for (int j = 0; j < J; j++) total += (int)jobs[j].size();
    if (!total) return 0;

    mt19937 rng(54321);
    auto t0 = chrono::steady_clock::now();
    int best_ms = INT_MAX;
    auto best_jobs = jobs;

    // === Phase 1: Deterministic priority rules ===
    Rule rules[] = {MWR, MOR, LPT, SPT};
    int nrules = total <= 5000 ? 4 : 2;
    for (int r = 0; r < nrules; r++) {
        for (int j = 0; j < J; j++) for (auto& t : jobs[j]) t.start = -1;
        vector<vector<IVal>> mi(M);
        dispatch(jobs, mi, J, M, rules[r]);
        int ms = makespan(jobs);
        if (ms < best_ms) { best_ms = ms; best_jobs = jobs; }
    }

    // === Phase 2: Randomized dispatching for diversification ===
    int nperm = total <= 200 ? 500 : total <= 5000 ? 100 : 10;
    Rule prules[] = {MWR, MOR, PERM};
    for (int p = 0; p < nperm; p++) {
        auto now = chrono::steady_clock::now();
        if (chrono::duration<double>(now - t0).count() > 3.0) break;
        for (int j = 0; j < J; j++) for (auto& t : jobs[j]) t.start = -1;
        vector<int> perm(J);
        for (int j = 0; j < J; j++) perm[j] = j;
        shuffle(perm.begin(), perm.end(), rng);
        vector<int> so(J);
        for (int i = 0; i < J; i++) so[perm[i]] = J - i;
        vector<vector<IVal>> mi(M);
        dispatch(jobs, mi, J, M, prules[p % 3], &so);
        int ms = makespan(jobs);
        if (ms < best_ms) { best_ms = ms; best_jobs = jobs; }
    }

    jobs = best_jobs;

    // === Phase 3: Large Neighborhood Search ===
    if (total <= 15000 && J >= 2) {
        double tl = total <= 200 ? 4.5 : 8.5;
        int mxit = total <= 200 ? 30000 : total <= 5000 ? 8000 : 2000;
        for (int it = 0; it < mxit; it++) {
            auto now = chrono::steady_clock::now();
            if (chrono::duration<double>(now - t0).count() > tl) break;

            int ds;
            switch (rng() % 10) {
                case 0: case 1: case 2: ds = max(1, J / 8); break;
                case 3: case 4: case 5: ds = max(1, J / 4); break;
                case 6: case 7:        ds = max(1, J / 2); break;
                default:               ds = max(1, (int)(J * 0.75)); break;
            }
            ds = max(1, min(ds, J));

            // Critical-path biased removal: prefer jobs finishing late
            vector<pair<int, int>> cf;
            for (int j = 0; j < J; j++)
                if (!best_jobs[j].empty())
                    cf.push_back({best_jobs[j].back().start + best_jobs[j].back().duration, j});
            sort(cf.rbegin(), cf.rend());

            vector<int> perm(J);
            for (int j = 0; j < J; j++) perm[j] = j;
            shuffle(perm.begin(), perm.end(), rng);

            vector<bool> in_rm(J, false);
            vector<int> removed;
            for (int i = 0; i < min(ds / 2, (int)cf.size()); i++) {
                removed.push_back(cf[i].second);
                in_rm[cf[i].second] = true;
            }
            for (int j : perm) {
                if ((int)removed.size() >= ds) break;
                if (!in_rm[j]) { removed.push_back(j); in_rm[j] = true; }
            }

            jobs = best_jobs;
            for (int j : removed) for (auto& t : jobs[j]) t.start = -1;

            vector<bool> exc(J, false);
            for (int j : removed) exc[j] = true;
            vector<vector<IVal>> mi(M);
            build_mi(jobs, M, mi, &exc);

            Rule rr = (rng() % 3 == 0) ? MOR : MWR;
            resched(jobs, mi, removed, rr);

            int ms = makespan(jobs);
            if (ms <= best_ms) { best_ms = ms; best_jobs = jobs; }
        }
    }

    // === Output ===
    vector<char> out;
    out.reserve(total * 24);
    for (int j = 0; j < J; j++) {
        for (int t = 0; t < (int)best_jobs[j].size(); t++) {
            if (t) out.push_back(' ');
            char tmp[32];
            int len = sprintf(tmp, "%d %d", best_jobs[j][t].start, best_jobs[j][t].machine);
            for (int i = 0; i < len; i++) out.push_back(tmp[i]);
        }
        out.push_back('\n');
    }
    fwrite(out.data(), 1, out.size(), stdout);
    return 0;
}
```

**Algorithm overview:**

The solver uses an adaptive 3-phase approach that scales with problem size:

**Phase 1 — Deterministic Priority Dispatching:** Tries up to 4 priority rules (MWR, MOR, LPT, SPT) to build initial feasible schedules. Each rule determines which job's next task gets priority when multiple are ready. Tasks are always placed at the earliest feasible time using efficient binary-search-based gap finding on per-machine interval lists.

**Phase 2 — Randomized Diversification:** Generates random job orderings via shuffle, using them as tiebreakers (for MWR/MOR) or as primary ordering (PERM rule). This explores a wide variety of dispatch sequences cheaply, controlled by time budget (≤3s).

**Phase 3 — Large Neighborhood Search (small/medium only):** Iteratively destroys part of the best solution (removing 12.5%–75% of jobs, biased toward critical-path jobs) and greedily re-inserts them. Accepts equal-or-better solutions. Runs up to 30K iterations for tiny instances, limited by time budget.

**Scale adaptation:**
- **Trivial** (≤200 tasks): All 4 rules + 500 random trials + 30K LNS iterations (4.5s limit)
- **Medium** (≤5K tasks): 4 rules + 100 random trials + 8K LNS iterations (8.5s limit)  
- **Large** (≤15K tasks): 4 rules + 100 random trials + 2K LNS iterations (8.5s limit)
- **Extreme** (>15K tasks): 2 rules + 10 random trials, no LNS (fast greedy only)
