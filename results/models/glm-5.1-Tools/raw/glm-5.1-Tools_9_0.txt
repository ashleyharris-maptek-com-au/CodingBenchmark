Here's my complete, optimized C# solution for the 1D Cutting Stock Problem:

```csharp
using System;
using System.Collections.Generic;
using System.IO;

public class Program
{
    static void Main()
    {
        var sr = new StreamReader(Console.OpenStandardInput(), System.Text.Encoding.ASCII, false, 1 << 16);
        var line1 = sr.ReadLine().Split(' ');
        int n = int.Parse(line1[0]);
        int stockLen = int.Parse(line1[1]);

        int[] cuts = new int[n];
        int idx = 0;
        string? line;
        while (idx < n && (line = sr.ReadLine()) != null)
        {
            int start = 0;
            int len = line.Length;
            while (start < len && idx < n)
            {
                while (start < len && line[start] == ' ') start++;
                if (start >= len) break;
                int numStart = start;
                while (start < len && line[start] != ' ') start++;
                cuts[idx++] = int.Parse(line.AsSpan(numStart, start - numStart));
            }
        }

        var bins = Solve(cuts, stockLen, n);

        using var sw = new StreamWriter(Console.OpenStandardOutput(), System.Text.Encoding.ASCII, 1 << 16);
        sw.WriteLine(bins.Count);
        for (int i = 0; i < bins.Count; i++)
        {
            var b = bins[i];
            for (int j = 0; j < b.Count; j++)
            {
                if (j > 0) sw.Write(' ');
                sw.Write(b[j]);
            }
            sw.WriteLine();
        }
    }

    static List<List<int>> Solve(int[] cuts, int stockLen, int n)
    {
        if (n == 0) return new List<List<int>>();

        // Sort indices by cut length descending
        int[] order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        Array.Sort(order, (a, b) => cuts[b].CompareTo(cuts[a]));

        // Best Fit Decreasing with SortedSet for O(log n) lookups
        int binCount = 0;
        int[] cutBin = new int[n];
        var remSet = new SortedSet<int>();
        var remDict = new Dictionary<int, List<int>>();

        for (int oi = 0; oi < n; oi++)
        {
            int ci = order[oi];
            int cl = cuts[ci];
            int bestBin = -1;
            int bestRemKey = -1;

            // Find bin with smallest remaining >= cl (best fit)
            var view = remSet.GetViewBetween(cl, stockLen);
            if (view.Count > 0)
            {
                bestRemKey = view.Min;
                var lst = remDict[bestRemKey];
                bestBin = lst[^1];
                lst.RemoveAt(lst.Count - 1);
                if (lst.Count == 0)
                {
                    remDict.Remove(bestRemKey);
                    remSet.Remove(bestRemKey);
                }
            }

            int newRem;
            if (bestBin >= 0)
            {
                newRem = bestRemKey - cl;
            }
            else
            {
                bestBin = binCount++;
                newRem = stockLen - cl;
            }
            cutBin[ci] = bestBin;

            if (!remDict.TryGetValue(newRem, out var nl))
            {
                nl = new List<int>(2);
                remDict[newRem] = nl;
                remSet.Add(newRem);
            }
            nl.Add(bestBin);
        }

        // Build bins
        var bins = new List<int>[binCount];
        for (int i = 0; i < binCount; i++) bins[i] = new List<int>();
        for (int i = 0; i < n; i++) bins[cutBin[i]].Add(i);

        // Collect non-empty bins
        var result = new List<List<int>>();
        for (int i = 0; i < binCount; i++)
            if (bins[i].Count > 0)
                result.Add(bins[i]);

        // Improvement phase
        result = Improve(result, cuts, stockLen);
        return result;
    }

    static List<List<int>> Improve(List<List<int>> bins, int[] cuts, int stockLen)
    {
        int numBins = bins.Count;

        // Compute bin used amounts
        int[] binUsed = new int[numBins];
        for (int i = 0; i < numBins; i++)
        {
            int s = 0;
            var b = bins[i];
            for (int j = 0; j < b.Count; j++) s += cuts[b[j]];
            binUsed[i] = s;
        }

        // Reinsertion improvement: try to empty least-used bins
        for (int round = 0; round < 5; round++)
        {
            bool anyImproved = false;

            // Sort bins by used (ascending) - least full first
            int[] binOrder = new int[numBins];
            for (int i = 0; i < numBins; i++) binOrder[i] = i;
            Array.Sort(binOrder, (a, b) => binUsed[a].CompareTo(binUsed[b]));

            // Build SortedSet structure for efficient best-fit
            var remSet = new SortedSet<int>();
            var remDict = new Dictionary<int, List<int>>();
            int[] curRem = new int[numBins];
            bool[] removed = new bool[numBins];

            for (int i = 0; i < numBins; i++)
            {
                curRem[i] = stockLen - binUsed[i];
                AddStruct(remSet, remDict, curRem[i], i);
            }

            int maxTry = Math.Min(numBins, 10000);
            for (int idx = 0; idx < maxTry; idx++)
            {
                int bi = binOrder[idx];
                if (removed[bi]) continue;

                // Remove bi from structure
                RemStruct(remSet, remDict, curRem[bi], bi);

                // Sort cuts from this bin by length descending
                var sortedCuts = new List<int>(bins[bi]);
                sortedCuts.Sort((a, b) => cuts[b].CompareTo(cuts[a]));

                var changes = new List<int>();
                bool allFit = true;

                for (int k = 0; k < sortedCuts.Count; k++)
                {
                    int ci = sortedCuts[k];
                    int cl = cuts[ci];
                    var view = remSet.GetViewBetween(cl, stockLen);
                    if (view.Count > 0)
                    {
                        int brk = view.Min;
                        var lst = remDict[brk];
                        int tb = lst[^1];
                        lst.RemoveAt(lst.Count - 1);
                        if (lst.Count == 0) { remDict.Remove(brk); remSet.Remove(brk); }

                        int oldRem = curRem[tb];
                        curRem[tb] = oldRem - cl;
                        AddStruct(remSet, remDict, oldRem - cl, tb);
                        changes.Add(tb);
                    }
                    else
                    {
                        allFit = false;
                        break;
                    }
                }

                if (allFit)
                {
                    // Apply: move cuts to target bins
                    for (int k = 0; k < sortedCuts.Count; k++)
                    {
                        bins[changes[k]].Add(sortedCuts[k]);
                        binUsed[changes[k]] += cuts[sortedCuts[k]];
                    }
                    bins[bi] = new List<int>();
                    binUsed[bi] = 0;
                    removed[bi] = true;
                    anyImproved = true;
                }
                else
                {
                    // Undo changes
                    for (int k = changes.Count - 1; k >= 0; k--)
                    {
                        int tb = changes[k];
                        int cl = cuts[sortedCuts[k]];
                        RemStruct(remSet, remDict, curRem[tb], tb);
                        curRem[tb] += cl;
                        AddStruct(remSet, remDict, curRem[tb], tb);
                    }
                    // Put bi back
                    AddStruct(remSet, remDict, curRem[bi], bi);
                }
            }

            if (!anyImproved) break;

            // Rebuild bins
            var newBins = new List<List<int>>();
            var newUsed = new List<int>();
            for (int i = 0; i < numBins; i++)
            {
                if (!removed[i] && bins[i].Count > 0)
                {
                    newBins.Add(bins[i]);
                    newUsed.Add(binUsed[i]);
                }
            }
            bins = newBins;
            binUsed = newUsed.ToArray();
            numBins = bins.Count;
        }

        // Merge improvement: try to merge pairs of bins using two pointers
        for (int mergeRound = 0; mergeRound < 100; mergeRound++)
        {
            // Sort bins by used ascending
            int[] order = new int[numBins];
            for (int i = 0; i < numBins; i++) order[i] = i;
            Array.Sort(order, (a, b) => binUsed[a].CompareTo(binUsed[b]));

            bool[] merged = new bool[numBins];
            bool didMerge = false;
            int lo = 0, hi = numBins - 1;

            while (lo < hi)
            {
                while (lo < hi && merged[order[lo]]) lo++;
                while (lo < hi && merged[order[hi]]) hi--;
                if (lo >= hi) break;

                int bi = order[lo], bj = order[hi];
                if (binUsed[bi] + binUsed[bj] <= stockLen)
                {
                    bins[bi].AddRange(bins[bj]);
                    binUsed[bi] += binUsed[bj];
                    bins[bj] = new List<int>();
                    binUsed[bj] = 0;
                    merged[bj] = true;
                    didMerge = true;
                    lo++;
                }
                else
                {
                    hi--;
                }
            }

            if (!didMerge) break;

            var newBins = new List<List<int>>();
            var newUsed = new List<int>();
            for (int i = 0; i < numBins; i++)
            {
                if (bins[i].Count > 0)
                {
                    newBins.Add(bins[i]);
                    newUsed.Add(binUsed[i]);
                }
            }
            bins = newBins;
            binUsed = newUsed.ToArray();
            numBins = bins.Count;
        }

        return bins;
    }

    static void AddStruct(SortedSet<int> rs, Dictionary<int, List<int>> rd, int rem, int bin)
    {
        if (!rd.TryGetValue(rem, out var lst))
        {
            lst = new List<int>(2);
            rd[rem] = lst;
            rs.Add(rem);
        }
        lst.Add(bin);
    }

    static void RemStruct(SortedSet<int> rs, Dictionary<int, List<int>> rd, int rem, int bin)
    {
        var lst = rd[rem];
        for (int i = lst.Count - 1; i >= 0; i--)
        {
            if (lst[i] == bin)
            {
                lst.RemoveAt(i);
                break;
            }
        }
        if (lst.Count == 0)
        {
            rd.Remove(rem);
            rs.Remove(rem);
        }
    }
}
```

**Algorithm summary:**

1. **Best Fit Decreasing (BFD)** — The core heuristic sorts cuts by length descending, then for each cut finds the bin with the smallest remaining capacity ≥ cut length (best fit). Uses `SortedSet<int>` + `Dictionary<int, List<int>>` for O(log n) ceiling queries, giving O(n log n) total packing time.

2. **Reinsertion improvement** — Iteratively tries to empty the least-used bins by redistributing their cuts into other bins using the same SortedSet-based best-fit. Up to 5 rounds, examining at most 10,000 candidate bins per round. Failed attempts are fully undone.

3. **Merge improvement** — Uses a two-pointer technique on bins sorted by used capacity: smallest + largest that fit together get merged. Repeats until no more merges are found.

This achieves the known 11/9 × OPT + 6/9 approximation guarantee for BFD, with iterative improvement typically closing the gap further for practical instances.
