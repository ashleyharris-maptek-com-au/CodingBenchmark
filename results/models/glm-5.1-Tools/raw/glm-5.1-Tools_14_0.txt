```csharp
    static int[] dy = { -1, 0, 1, -1, 1, -1, 0, 1 };

    public static void Main()

        bool progress = true;

            SimpleInference(allSafe);
            int before = allSafe.Count;

            // Mark newly found safe cells on board (as 'S' = known safe, unknown number)

        }

        // Deduplicate and output

        var unique = new List<(int, int)>();

            if (seen.Add(s)) unique.Add(s);

    static void ReadInput()
    {
        var sr = new StreamReader(Console.OpenStandardInput());
        var dims = sr.ReadLine().Split();
        H = int.Parse(dims[0]);
        W = int.Parse(dims[1]);
        board = new char[H][];
        for (int y = 0; y < H; y++)
        {
            string line = sr.ReadLine();
            board[y] = new char[W];
            int len = line != null ? line.Length : 0;
            for (int x = 0; x < W; x++)
                board[y][x] = x < len ? line[x] : ' ';
        }
    }

    // Simple rule-based inference: if remaining mines == 0, all unknowns safe;
    // if remaining mines == unknown count, all unknowns are mines.
    static void SimpleInference(List<(int, int)> allSafe)
    {
        bool changed = true;
        while (changed)
        {
            changed = false;
            for (int y = 0; y < H; y++)
            for (int x = 0; x < W; x++)
            {
                char c = board[y][x];
                if (c < '0' || c > '8') continue;
                int num = c - '0', mines = 0;
                var unk = new List<(int, int)>();
                for (int d = 0; d < 8; d++)
                {
                    int nx = x + dx[d], ny = y + dy[d];
                    if (nx >= 0 && nx < W && ny >= 0 && ny < H)
                    {
                        if (board[ny][nx] == '*') mines++;
                        else if (board[ny][nx] == ' ') unk.Add((nx, ny));
                    }
                }
                int rem = num - mines;
                if (rem < 0 || rem > unk.Count) continue;
                if (rem == 0 && unk.Count > 0)
                {
                    foreach (var (ux, uy) in unk)
                    { allSafe.Add((ux, uy)); board[uy][ux] = 'S'; }
                    changed = true;
                }
                else if (rem == unk.Count && unk.Count > 0)
                {
                    foreach (var (ux, uy) in unk) board[uy][ux] = '*';
                    changed = true;
                }
            }
        }
    }

    // Constraint-based solving with adaptive strategy per component
    static bool ConstraintSolve(List<(int, int)> allSafe)
    {
        // Collect frontier cells (unknown cells adjacent to numbers) and constraints
        var frontier = new List<(int, int)>();
        var fi = new Dictionary<(int, int), int>();
        var cl = new List<(int[], int)>();

        for (int y = 0; y < H; y++)
        for (int x = 0; x < W; x++)
        {
            char c = board[y][x];
            if (c < '0' || c > '8') continue;
            int num = c - '0', mines = 0;
            var vars = new List<int>();
            for (int d = 0; d < 8; d++)
            {
                int nx = x + dx[d], ny = y + dy[d];
                if (nx >= 0 && nx < W && ny >= 0 && ny < H)
                {
                    if (board[ny][nx] == '*') mines++;
                    else if (board[ny][nx] == ' ')
                    {
                        if (!fi.ContainsKey((nx, ny)))
                        { fi[(nx, ny)] = frontier.Count; frontier.Add((nx, ny)); }
                        vars.Add(fi[(nx, ny)]);
                    }
                }
            }
            int rem = num - mines;
            if (vars.Count > 0 && rem >= 0 && rem <= vars.Count)
            { vars.Sort(); cl.Add((vars.ToArray(), rem)); }
        }

        int n = frontier.Count;
        if (n == 0) return false;

        // Deduplicate constraints
        var dd = new HashSet<string>();
        var cons = new List<(int[], int)>();
        foreach (var (v, t) in cl)
        { string k = string.Join(",", v) + ":" + t; if (dd.Add(k)) cons.Add((v, t)); }

        // Pairwise subset: derive new constraints from subset relationships
        if (cons.Count <= 5000)
        {
            var extra = PairwiseSubset(cons);
            foreach (var (v, t) in extra)
            { string k = string.Join(",", v) + ":" + t; if (dd.Add(k)) cons.Add((v, t)); }
        }

        // Connected components via BFS
        var adj = new HashSet<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new HashSet<int>();
        foreach (var (vars, _) in cons)
            for (int i = 0; i < vars.Length; i++)
            for (int j = i + 1; j < vars.Length; j++)
            { adj[vars[i]].Add(vars[j]); adj[vars[j]].Add(vars[i]); }

        var cid = new int[n];
        for (int i = 0; i < n; i++) cid[i] = -1;
        int nc = 0;
        for (int s = 0; s < n; s++)
        {
            if (cid[s] >= 0) continue;
            int id = nc++;
            var q = new Queue<int>(); q.Enqueue(s); cid[s] = id;
            while (q.Count > 0)
            { int u = q.Dequeue(); foreach (int v in adj[u]) if (cid[v] < 0) { cid[v] = id; q.Enqueue(v); } }
        }

        var cv = new List<int>[nc];
        var cc = new List<(int[], int)>[nc];
        for (int c = 0; c < nc; c++) { cv[c] = new List<int>(); cc[c] = new List<(int[], int)>(); }
        for (int i = 0; i < n; i++) cv[cid[i]].Add(i);
        foreach (var (vars, target) in cons)
            if (vars.Length > 0) cc[cid[vars[0]]].Add((vars, target));

        // Solve each component with adaptive strategy
        bool foundMine = false;
        for (int c = 0; c < nc; c++)
        {
            var li = new Dictionary<int, int>();
            for (int i = 0; i < cv[c].Count; i++) li[cv[c][i]] = i;
            var lc = new List<(int[], int)>();
            foreach (var (vars, target) in cc[c])
            {
                var rv = new int[vars.Length];
                for (int i = 0; i < vars.Length; i++) rv[i] = li[vars[i]];
                lc.Add((rv, target));
            }
            int cn = cv[c].Count;
            bool mf;
            if (cn <= 25)
                mf = EnumerateAndDeduce(cn, lc, cv[c], frontier, allSafe);
            else if (cn <= 1500)
                mf = GaussAndBound(cn, lc, cv[c], frontier, allSafe);
            else
                mf = SimpleBoundOnly(cn, lc, cv[c], frontier, allSafe);
            if (mf) foundMine = true;
        }
        return foundMine;
    }

    // Pairwise subset constraint generation
    static List<(int[], int)> PairwiseSubset(List<(int[], int)> cons)
    {
        var extra = new List<(int[], int)>();
        int m = cons.Count;
        var sorted = new int[m];
        for (int i = 0; i < m; i++) sorted[i] = i;
        Array.Sort(sorted, (a, b) => cons[a].Item1.Length.CompareTo(cons[b].Item1.Length));

        for (int ii = 0; ii < m; ii++)
        {
            int i = sorted[ii];
            var (va, ta) = cons[i];
            if (va.Length == 0) continue;
            for (int jj = ii + 1; jj < m; jj++)
            {
                int j = sorted[jj];
                var (vb, tb) = cons[j];
                if (va.Length == vb.Length) continue;
                var (small, large, ts, tl) = va.Length < vb.Length
                    ? (va, vb, ta, tb) : (vb, va, tb, ta);
                if (IsSortedSubset(small, large))
                {
                    var diff = SetDiff(large, small);
                    int nt = tl - ts;
                    if (diff.Count > 0 && nt >= 0 && nt <= diff.Count)
                        extra.Add((diff.ToArray(), nt));
                }
                if (extra.Count > 10000) return extra;
            }
        }
        return extra;
    }

    static bool IsSortedSubset(int[] small, int[] large)
    {
        int si = 0;
        for (int li = 0; li < large.Length && si < small.Length; li++)
            if (large[li] == small[si]) si++;
        return si == small.Length;
    }

    static List<int> SetDiff(int[] large, int[] small)
    {
        var diff = new List<int>();
        int si = 0;
        for (int li = 0; li < large.Length; li++)
        {
            if (si < small.Length && large[li] == small[si]) si++;
            else diff.Add(large[li]);
        }
        return diff;
    }

    // Simple bound analysis for very large components (>1500 vars)
    static bool SimpleBoundOnly(int n, List<(int[], int)> constraints,
        List<int> gv, List<(int, int)> frontier, List<(int, int)> allSafe)
    {
        bool fm = false;
        foreach (var (vars, target) in constraints)
        {
            if (target == 0)
                foreach (int v in vars) allSafe.Add(frontier[gv[v]]);
            else if (target == vars.Length)
                foreach (int v in vars)
                { var (mx, my) = frontier[gv[v]]; if (board[my][mx] == ' ') { board[my][mx] = '*'; fm = true; } }
        }
        return fm;
    }

    // Full enumeration with pruning for small components (<=25 variables)
    static bool EnumerateAndDeduce(int n, List<(int[], int)> constraints,
        List<int> gv, List<(int, int)> frontier, List<(int, int)> allSafe)
    {
        int mc = constraints.Count;
        var cV = new int[mc][];
        var cT = new int[mc];
        for (int i = 0; i < mc; i++) { cV[i] = constraints[i].Item1; cT[i] = constraints[i].Item2; }

        // Reorder: most-constrained variable first for better pruning
        var vcc = new int[n];
        for (int ci = 0; ci < mc; ci++) foreach (int v in cV[ci]) vcc[v]++;
        var ord = new int[n]; for (int i = 0; i < n; i++) ord[i] = i;
        Array.Sort(ord, (a, b) => vcc[b].CompareTo(vcc[a]));
        var rev = new int[n]; for (int i = 0; i < n; i++) rev[ord[i]] = i;
        var rgv = new int[n]; for (int i = 0; i < n; i++) rgv[rev[i]] = gv[i];

        for (int ci = 0; ci < mc; ci++)
        {
            for (int i = 0; i < cV[ci].Length; i++) cV[ci][i] = rev[cV[ci][i]];
            Array.Sort(cV[ci]);
        }

        var vCon = new List<int>[n]; for (int i = 0; i < n; i++) vCon[i] = new List<int>();
        for (int ci = 0; ci < mc; ci++) foreach (int v in cV[ci]) vCon[v].Add(ci);

        // Precompute remaining variable counts for pruning
        var rem = new int[mc, n + 2];
        for (int c = 0; c < mc; c++)
        for (int b = 0; b <= n + 1; b++)
        { int cnt = 0; foreach (int v in cV[c]) if (v >= b) cnt++; rem[c, b] = cnt; }

        long alwaysZero = ~0L, alwaysOne = ~0L, solCount = 0;
        var ps = new int[mc];
        EnumRec(0, 0L, ps, n, mc, cV, cT, vCon, rem, ref alwaysZero, ref alwaysOne, ref solCount);

        bool fm = false;
        if (solCount > 0)
        {
            for (int i = 0; i < n; i++)
            {
                if ((alwaysZero & (1L << i)) != 0) allSafe.Add(frontier[rgv[i]]);
                if ((alwaysOne & (1L << i)) != 0)
                { var (mx, my) = frontier[rgv[i]]; if (board[my][mx] == ' ') { board[my][mx] = '*'; fm = true; } }
            }
        }
        return fm;
    }

    static void EnumRec(int bit, long asgn, int[] ps, int n, int mc,
        int[][] cV, int[] cT, List<int>[] vCon, int[,] rem,
        ref long alwaysZero, ref long alwaysOne, ref long solCount)
    {
        if (bit == n)
        {
            for (int c = 0; c < mc; c++) if (ps[c] != cT[c]) return;
            solCount++; alwaysZero &= ~asgn; alwaysOne &= asgn;
            return;
        }
        // Try x_bit = 0
        bool ok0 = true;
        foreach (int c in vCon[bit])
            if (ps[c] > cT[c] || ps[c] + rem[c, bit + 1] < cT[c]) { ok0 = false; break; }
        if (ok0) EnumRec(bit + 1, asgn, ps, n, mc, cV, cT, vCon, rem, ref alwaysZero, ref alwaysOne, ref solCount);
        // Try x_bit = 1
        bool ok1 = true;
        foreach (int c in vCon[bit])
        { int ns = ps[c] + 1; if (ns > cT[c] || ns + rem[c, bit + 1] < cT[c]) { ok1 = false; break; } }
        if (ok1)
        {
            foreach (int c in vCon[bit]) ps[c]++;
            EnumRec(bit + 1, asgn | (1L << bit), ps, n, mc, cV, cT, vCon, rem, ref alwaysZero, ref alwaysOne, ref solCount);
            foreach (int c in vCon[bit]) ps[c]--;
        }
    }

    // Gaussian elimination + bound analysis for medium components (26-1500 vars)
    static bool GaussAndBound(int n, List<(int[], int)> constraints,
        List<int> gv, List<(int, int)> frontier, List<(int, int)> allSafe)
    {
        int m = constraints.Count;
        // Rational arithmetic matrix: (numerator, denominator) per entry
        long[,] nm = new long[m, n + 1], dn = new long[m, n + 1];
        for (int r = 0; r < m; r++)
        {
            for (int c = 0; c <= n; c++) { nm[r, c] = 0; dn[r, c] = 1; }
            foreach (int v in constraints[r].Item1) nm[r, v] = 1;
            nm[r, n] = constraints[r].Item2;
        }

        // Gauss-Jordan elimination to Reduced Row Echelon Form
        int[] pc = new int[m]; for (int r = 0; r < m; r++) pc[r] = -1;
        int cr = 0;
        for (int col = 0; col < n && cr < m; col++)
        {
            int pr = -1;
            for (int r = cr; r < m; r++) if (nm[r, col] != 0) { pr = r; break; }
            if (pr < 0) continue;
            if (pr != cr) for (int c = 0; c <= n; c++)
            { (nm[cr, c], nm[pr, c]) = (nm[pr, c], nm[cr, c]); (dn[cr, c], dn[pr, c]) = (dn[pr, c], dn[cr, c]); }
            pc[cr] = col;
            for (int r = 0; r < m; r++)
            {
                if (r == cr || nm[r, col] == 0) continue;
                long pN = nm[cr, col], pD = dn[cr, col], rN = nm[r, col], rD = dn[r, col];
                for (int c2 = 0; c2 <= n; c2++)
                {
                    long nn = nm[r, c2] * pN * rD * dn[cr, c2] - nm[cr, c2] * rN * pD * dn[r, c2];
                    long nd = dn[r, c2] * pD * dn[cr, c2] * rD;
                    if (nn == 0) { nd = 1; }
                    else
                    {
                        long g = GCD(Math.Abs(nn), Math.Abs(nd));
                        if (g > 1) { nn /= g; nd /= g; }
                        if (nd < 0) { nn = -nn; nd = -nd; }
                    }
                    nm[r, c2] = nn; dn[r, c2] = nd;
                }
            }
            cr++;
        }
        int rank = cr;

        // Convert RREF rows to integer coefficients for bound analysis
        var rowC = new long[rank, n]; var rowR = new long[rank]; var rowV = new bool[rank];
        for (int r = 0; r < rank; r++)
        {
            rowV[r] = false;
            long rd = 1; bool ok = true;
            for (int c = 0; c <= n; c++)
            { long d = dn[r, c]; if (d <= 0) { ok = false; break; } long nl = LCM(rd, d); if (nl > 1L << 40) { ok = false; break; } rd = nl; }
            if (!ok || rd == 0) continue;
            try
            {
                rowR[r] = checked(nm[r, n] * (rd / dn[r, n]));
                for (int c = 0; c < n; c++) rowC[r, c] = checked(nm[r, c] * (rd / dn[r, c]));
                rowV[r] = true;
            }
            catch (OverflowException) { }
        }

        bool fm = false;

        // Bound analysis on each RREF row
        for (int r = 0; r < rank; r++)
        {
            if (!rowV[r]) continue;
            long ir = rowR[r], sp = 0, sn = 0;
            for (int c = 0; c < n; c++) { if (rowC[r, c] > 0) sp += rowC[r, c]; else if (rowC[r, c] < 0) sn += rowC[r, c]; }

            // rhs == sum of negatives: positive-coeff vars = 0 (safe), negative-coeff vars = 1 (mine)
            if (ir == sn)
            { for (int c = 0; c < n; c++) if (rowC[r, c] > 0) allSafe.Add(frontier[gv[c]]); else if (rowC[r, c] < 0) { var (mx, my) = frontier[gv[c]]; if (board[my][mx] == ' ') { board[my][mx] = '*'; fm = true; } } }
            // rhs == sum of positives: negative-coeff vars = 0 (safe), positive-coeff vars = 1 (mine)
            if (ir == sp)
            { for (int c = 0; c < n; c++) if (rowC[r, c] < 0) allSafe.Add(frontier[gv[c]]); else if (rowC[r, c] > 0) { var (mx, my) = frontier[gv[c]]; if (board[my][mx] == ' ') { board[my][mx] = '*'; fm = true; } } }

            // Per-variable bound analysis
            for (int j = 0; j < n; j++)
            {
                if (rowC[r, j] == 0) continue;
                long cf = rowC[r, j];
                long op = cf > 0 ? sp - cf : sp, on = cf < 0 ? sn - cf : sn;
                bool cantBe1 = (ir - cf < on || ir - cf > op);
                bool cantBe0 = (ir < on || ir > op);
                if (cantBe1 && !cantBe0) allSafe.Add(frontier[gv[j]]);
                if (cantBe0 && !cantBe1) { var (mx, my) = frontier[gv[j]]; if (board[my][mx] == ' ') { board[my][mx] = '*'; fm = true; } }
            }
        }

        // Per-variable feasibility check using RREF bounds (for components <= 500)
        if (n <= 500)
        {
            for (int j = 0; j < n; j++)
            {
                bool f0 = true, f1 = true;
                for (int r = 0; r < rank && (f0 || f1); r++)
                {
                    if (!rowV[r]) continue;
                    long cf = rowC[r, j], op = 0, on = 0;
                    for (int k = 0; k < n; k++) { if (k == j) continue; if (rowC[r, k] > 0) op += rowC[r, k]; else if (rowC[r, k] < 0) on += rowC[r, k]; }
                    if (f0 && (rowR[r] < on || rowR[r] > op)) f0 = false;
                    if (f1 && (rowR[r] - cf < on || rowR[r] - cf > op)) f1 = false;
                }
                if (f1 && !f0) allSafe.Add(frontier[gv[j]]);
                if (f0 && !f1) { var (mx, my) = frontier[gv[j]]; if (board[my][mx] == ' ') { board[my][mx] = '*'; fm = true; } }
            }
        }
        return fm;
    }

    static long GCD(long a, long b) { while (b != 0) { long t = b; b = a % b; a = t; } return a; }
    static long LCM(long a, long b) { if (a == 0) return b; if (b == 0) return a; return a / GCD(a, b) * b; }
}
```

**Architecture summary:**

| Component Size | Strategy | Time Complexity |
|---|---|---|
| ≤25 vars | Full enumeration with pruning + variable reordering | O(2^n × m) worst, pruning cuts dramatically |
| 26–1500 vars | Gaussian elimination (exact rationals) → RREF bound analysis → per-variable feasibility | O(m·n²) for elimination, O(n·rank·n) for feasibility |
| >1500 vars | Simple per-constraint bounds only | O(m) |

**Key design decisions:**
- **Iteration loop**: Safe cells and mines found by constraint solving are marked on the board (`'S'`/`'*'`), triggering re-evaluation of simple rules — effectively iterative constraint propagation
- **Pairwise subset**: Before component splitting, derives new constraints (e.g., if A⊆B, then B\A has target difference), which is powerful for Minesweeper's overlapping number neighborhoods
- **Exact rationals**: Gaussian elimination uses (numerator, denominator) pairs with GCD reduction and overflow detection (LCM capped at 2^40), avoiding floating-point errors
- **Always-safe tracking**: Enumeration computes `alwaysZero &= ~assignment` — a bit is set iff that variable is 0 in every valid solution, guaranteeing safety
- **Variable reordering**: Enumeration sorts by constraint count (descending) so the most-constrained variables are decided first, maximizing pruning
