#include <bits/stdc++.h>
using namespace std;

struct Solver {
    int n, words;
    vector<vector<uint64_t>> adj;
    vector<int> best;
    vector<uint64_t> allmask;

    static inline int popcount64(uint64_t x) {
        return (int)__builtin_popcountll(x);
    }

    int count_set_bits(const vector<uint64_t>& bs) const {
        int s = 0;
        for (uint64_t x : bs) s += popcount64(x);
        return s;
    }

    bool empty_set(const vector<uint64_t>& bs) const {
        for (uint64_t x : bs) if (x) return false;
        return true;
    }

    void set_bit(vector<uint64_t>& bs, int v) const {
        bs[v >> 6] |= 1ULL << (v & 63);
    }

    void clear_bit(vector<uint64_t>& bs, int v) const {
        bs[v >> 6] &= ~(1ULL << (v & 63));
    }

    bool has_bit(const vector<uint64_t>& bs, int v) const {
        return (bs[v >> 6] >> (v & 63)) & 1ULL;
    }

    void remove_vertices(vector<uint64_t>& bs, const vector<uint64_t>& rem) const {
        for (int i = 0; i < words; ++i) bs[i] &= ~rem[i];
    }

    int max_degree_vertex(const vector<uint64_t>& cand) const {
        int bestv = -1, bestd = -1;
        for (int w = 0; w < words; ++w) {
            uint64_t x = cand[w];
            while (x) {
                int b = __builtin_ctzll(x);
                int v = w * 64 + b;
                int d = 0;
                for (int i = 0; i < words; ++i) d += popcount64(adj[v][i] & cand[i]);
                if (d > bestd) {
                    bestd = d;
                    bestv = v;
                }
                x &= x - 1;
            }
        }
        return bestv;
    }

    vector<int> greedy_coloring_bound(const vector<uint64_t>& cand) const {
        vector<uint64_t> rem = cand;
        vector<int> colors;
        while (!empty_set(rem)) {
            vector<uint64_t> layer(words, 0);
            for (int w = 0; w < words; ++w) {
                uint64_t x = rem[w];
                while (x) {
                    int b = __builtin_ctzll(x);
                    int v = w * 64 + b;
                    x &= x - 1;
                    bool ok = true;
                    for (int i = 0; i < words; ++i) {
                        if (layer[i] & adj[v][i]) { ok = false; break; }
                    }
                    if (ok) set_bit(layer, v);
                }
            }
            remove_vertices(rem, layer);
            colors.push_back(count_set_bits(layer));
        }
        return colors;
    }

    void expand(vector<uint64_t> cand, vector<int>& cur) {
        if (cur.size() + count_set_bits(cand) <= best.size()) return;
        if (empty_set(cand)) {
            if (cur.size() > best.size()) best = cur;
            return;
        }

        vector<int> colors = greedy_coloring_bound(cand);
        int v = max_degree_vertex(cand);
        if (v == -1) return;

        vector<uint64_t> cand1 = cand;
        clear_bit(cand1, v);
        remove_vertices(cand1, adj[v]);
        cur.push_back(v);
        expand(cand1, cur);
        cur.pop_back();

        clear_bit(cand, v);
        if (cur.size() + count_set_bits(cand) > best.size()) {
            if (colors.size() == 1) return;
            expand(cand, cur);
        }
    }

    vector<int> solve() {
        vector<uint64_t> cand = allmask;
        vector<int> cur;
        expand(cand, cur);
        sort(best.begin(), best.end());
        return best;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    Solver s;
    s.n = n;
    s.words = (n + 63) / 64;
    s.adj.assign(n, vector<uint64_t>(s.words, 0));
    s.allmask.assign(s.words, ~0ULL);
    if (n % 64) s.allmask.back() = (1ULL << (n % 64)) - 1ULL;

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue;
        s.adj[u][v >> 6] |= 1ULL << (v & 63);
        s.adj[v][u >> 6] |= 1ULL << (u & 63);
    }

    vector<int> ans = s.solve();
    cout << ans.size() << '\n';
    for (size_t i = 0; i < ans.size(); ++i) {
        if (i) cout << ' ';
        cout << ans[i];
    }
    cout << '\n';
    return 0;
}
