OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %ubo
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %outColor "outColor"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %ubo "ubo"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1

%void = OpTypeVoid
%funct = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%ptr_uniform_v4 = OpTypePointer Uniform %v4float
%UBO = OpTypeStruct %v4float %v4float %v4float %v4float %v4float %v4float
%uint = OpTypeInt 32 0
%GLSLstd450 = OpExtInstImport "GLSL.std.450"
%const0 = OpConstant %float 0
%const1 = OpConstant %float 1
%const02 = OpConstant %float 0.2
%const03 = OpConstant %float 0.3
%const05 = OpConstant %float 0.5
%const08 = OpConstant %float 0.8
%const3 = OpConstant %float 3
%uint4 = OpConstant %uint 4
%base = OpConstantComposite %v3float %const02 %const03 %const08
%rimTint = OpConstantComposite %v3float %const1 %const08 %const05

%worldPos = OpVariable %ptr_input_v3 Input
%normal = OpVariable %ptr_input_v3 Input
%outColor = OpVariable %ptr_output_v4 Output
%ubo = OpVariable %ptr_uniform_ubo Uniform

%main = OpFunction %void None %funct
%label = OpLabel
%wp = OpLoad %v3float %worldPos
%n0 = OpLoad %v3float %normal
%n = OpExtInst %v3float %GLSLstd450 Normalize %n0
%camv4ptr = OpAccessChain %ptr_uniform_v4 %ubo %uint4
%camv4 = OpLoad %v4float %camv4ptr
%cam = OpVectorShuffle %v3float %camv4 %camv4 0 1 2
%vdir0 = OpFSub %v3float %cam %wp
%vdir = OpExtInst %v3float %GLSLstd450 Normalize %vdir0
%ndotv = OpDot %float %n %vdir
%ndotv_clamped = OpExtInst %float %GLSLstd450 FMax %ndotv %const0
%one_minus = OpFSub %float %const1 %ndotv_clamped
%rim = OpExtInst %float %GLSLstd450 Pow %one_minus %const3
%diffuse = OpExtInst %float %GLSLstd450 FMax %ndotv %const0
%diffuse_scale = OpFMul %float %const05 %diffuse
%lit_scale = OpFAdd %float %const03 %diffuse_scale
%base_scaled = OpVectorTimesScalar %v3float %base %lit_scale
%rim_scaled = OpVectorTimesScalar %v3float %rimTint %rim
%rgb = OpFAdd %v3float %base_scaled %rim_scaled
%color = OpCompositeConstruct %v4float %rgb %const1
OpStore %outColor %color
OpReturn
OpFunctionEnd
