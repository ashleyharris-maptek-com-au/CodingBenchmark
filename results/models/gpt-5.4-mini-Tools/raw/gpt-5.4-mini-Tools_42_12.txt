%fn = OpTypeFunction %void

%v3_0 = OpConstantComposite %v3float %f_0 %f_0 %f_0

%ptr_out_v3_arr = OpTypePointer Output %arr_v3_out

%ptr_out_v4 = OpTypePointer Output %v4float

OpEndPrimitive

%m12 = OpFAdd %v3float %w1 %w2

%h01 = OpFMul %v3float %m01 %v3_0
%h12 = OpFMul %v3float %m12 %v3_0
%h20 = OpFMul %v3float %m20 %v3_0

%edge0a = OpFAdd %v3float %w0 %v3_0
%edge0b = OpFAdd %v3float %w1 %v3_0
%edge1a = OpFAdd %v3float %w1 %v3_0
%edge1b = OpFAdd %v3float %w2 %v3_0
%edge2a = OpFAdd %v3float %w2 %v3_0
%edge2b = OpFAdd %v3float %w0 %v3_0

%emit0 = OpCompositeConstruct %v4float %f_0 %f_0 %f_0 %f_1
%emit1 = OpCompositeConstruct %v4float %f_0 %f_0 %f_0 %f_1
%emit2 = OpCompositeConstruct %v4float %f_0 %f_0 %f_0 %f_1
%emit3 = OpCompositeConstruct %v4float %f_0 %f_0 %f_0 %f_1

%e0p0 = OpCompositeExtract %float %w0 0
%e0p1 = OpCompositeExtract %float %w0 1
%e0p2 = OpCompositeExtract %float %w0 2

%edgep0 = OpCompositeConstruct %v4float %e0p0 %e0p1 %e0p2 %f_1
%edgep1 = OpCompositeConstruct %v4float %e0p0 %e0p1 %e0p2 %f_1
%edgep2 = OpCompositeConstruct %v4float %e0p0 %e0p1 %e0p2 %f_1
%edgep3 = OpCompositeConstruct %v4float %e0p0 %e0p1 %e0p2 %f_1

OpStore %outp0 %edgep0
OpStore %outColor %v3_white
