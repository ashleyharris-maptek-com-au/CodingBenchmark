; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
OpExecutionMode %main OriginUpperLeft
%1 = OpExtInstImport "GLSL.std.450"

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%ptr_input_v3float = OpTypePointer Input %v3float
%ptr_output_v4float = OpTypePointer Output %v4float
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%ptr_uniform_v4float = OpTypePointer Uniform %v4float
%ptr_uniform_mat4 = OpTypePointer Uniform %mat4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%float_0 = OpConstant %float 0
%float_02 = OpConstant %float 0.200000003
%float_04 = OpConstant %float 0.400000006
%float_05 = OpConstant %float 0.5
%float_07 = OpConstant %float 0.699999988
%float_075 = OpConstant %float 0.75
%float_1 = OpConstant %float 1
%int_3 = OpConstant %int 3
%v3_02 = OpConstantComposite %v3float %float_02 %float_02 %float_075
%v3_07 = OpConstantComposite %v3float %float_07 %float_07 %float_04
%v3_1 = OpConstantComposite %v3float %float_1 %float_1 %float_1
%worldPos = OpVariable %ptr_input_v3float Input
%normal = OpVariable %ptr_input_v3float Input
%outColor = OpVariable %ptr_output_v4float Output
%ubo = OpVariable %ptr_uniform_ubo Uniform

%main = OpFunction %void None %func
%label = OpLabel
%n = OpLoad %v3float %normal
%nlen = OpExtInst %v3float %1 Normalize %n
%light = OpAccessChain %ptr_uniform_v4float %ubo %int_3
%lightv4 = OpLoad %v4float %light
%lightv3 = OpVectorShuffle %v3float %lightv4 %lightv4 0 1 2
%lvec = OpFSub %v3float %lightv3 %worldPos
%llen = OpExtInst %v3float %1 Normalize %lvec
%ndotl = OpDot %float %nlen %llen
%scaled = OpFAdd %float %ndotl %float_1
%t = OpFMul %float %scaled %float_05
%t3 = OpCompositeConstruct %v3float %t %t %t
%one_minus_t = OpFSub %v3float %v3_1 %t3
%cool_term = OpFMul %v3float %v3_02 %one_minus_t
%warm_term = OpFMul %v3float %v3_07 %t3
%rgb = OpFAdd %v3float %cool_term %warm_term
%outv4 = OpCompositeConstruct %v4float %rgb %float_1
OpStore %outColor %outv4
OpReturn
OpFunctionEnd
