declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Point = { x: number; y: number; id?: number };

const depot: Point = { x: data.depot[0], y: data.depot[1] };
const truckCount: number = Math.max(1, data.truckCount | 0);
const packages: Point[] = (data.packages || []).map((p: any) => ({ x: +p.x, y: +p.y, id: p.id }));

const n = packages.length;
if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

function dist(a: Point, b: Point): number {
  return Math.hypot(a.x - b.x, a.y - b.y);
}

const pkgCount = packages.length;

// Precompute distances from/to depot and between packages for fast scoring.
const depotDist = new Float64Array(pkgCount);
for (let i = 0; i < pkgCount; i++) depotDist[i] = dist(depot, packages[i]);

const matrix: number[][] = new Array(pkgCount);
for (let i = 0; i < pkgCount; i++) {
  const row = new Array<number>(pkgCount);
  for (let j = 0; j < pkgCount; j++) row[j] = i === j ? 0 : dist(packages[i], packages[j]);
  matrix[i] = row;
}

// Build up to truckCount seed routes using farthest-first selection.
const assigned = new Uint8Array(pkgCount);
const routes: number[][] = [];
const routeLoadTarget = Math.max(1, Math.ceil(pkgCount / truckCount));

function nearestUnassignedToDepot(): number {
  let best = -1;
  let bestScore = Infinity;
  for (let i = 0; i < pkgCount; i++) {
    if (assigned[i]) continue;
    const s = depotDist[i];
    if (s < bestScore) {
      bestScore = s;
      best = i;
    }
  }
  return best;
}

function farthestUnassignedFromSet(seedSet: number[]): number {
  let best = -1;
  let bestScore = -Infinity;
  for (let i = 0; i < pkgCount; i++) {
    if (assigned[i]) continue;
    let minD = Infinity;
    for (let k = 0; k < seedSet.length; k++) {
      const s = matrix[i][seedSet[k]];
      if (s < minD) minD = s;
    }
    const score = minD + depotDist[i] * 0.25;
    if (score > bestScore) {
      bestScore = score;
      best = i;
    }
  }
  return best;
}

function insertionDelta(route: number[], pos: number, node: number): number {
  const prev = pos === 0 ? -1 : route[pos - 1];
  const next = pos === route.length ? -1 : route[pos];
  const before = (prev === -1 ? depotDist[node] : matrix[prev][node]) + (next === -1 ? depotDist[node] : matrix[node][next]);
  const after = prev === -1 && next === -1 ? 0 : (prev === -1 ? depotDist[next] : next === -1 ? depotDist[prev] : matrix[prev][next]);
  return before - after;
}

function bestInsertion(route: number[], node: number): { pos: number; delta: number } {
  let bestPos = 0;
  let bestDelta = Infinity;
  for (let pos = 0; pos <= route.length; pos++) {
    const delta = insertionDelta(route, pos, node);
    if (delta < bestDelta) {
      bestDelta = delta;
      bestPos = pos;
    }
  }
  return { pos: bestPos, delta: bestDelta };
}

// Seed routes with spread-out packages, then greedily insert remaining ones.
for (let r = 0; r < Math.min(truckCount, pkgCount); r++) {
  let start: number;
  if (r === 0) {
    start = nearestUnassignedToDepot();
  } else {
    const seedSet: number[] = [];
    for (const rt of routes) {
      if (rt.length > 0) seedSet.push(rt[0], rt[rt.length - 1]);
    }
    start = farthestUnassignedFromSet(seedSet.length ? seedSet : [0]);
  }
  if (start < 0) break;
  assigned[start] = 1;
  routes.push([start]);
}

if (routes.length === 0) {
  routes.push([0]);
  assigned[0] = 1;
}

while (true) {
  let remaining = -1;
  let bestScore = -Infinity;
  for (let i = 0; i < pkgCount; i++) {
    if (assigned[i]) continue;
    const score = depotDist[i];
    if (score > bestScore) {
      bestScore = score;
      remaining = i;
    }
  }
  if (remaining < 0) break;

  let bestRoute = 0;
  let bestPos = 0;
  let bestDelta = Infinity;

  for (let r = 0; r < routes.length; r++) {
    const route = routes[r];
    const ins = bestInsertion(route, remaining);
    // Prefer keeping routes reasonably balanced while still minimizing distance.
    const balancePenalty = route.length >= routeLoadTarget ? 0.15 * depotDist[remaining] : 0;
    const score = ins.delta + balancePenalty;
    if (score < bestDelta) {
      bestDelta = score;
      bestRoute = r;
      bestPos = ins.pos;
    }
  }

  routes[bestRoute].splice(bestPos, 0, remaining);
  assigned[remaining] = 1;
}

// Optional local improvement: relocate nodes from long routes into better positions elsewhere.
let improved = true;
let iter = 0;
while (improved && iter++ < 2) {
  improved = false;
  for (let fromR = 0; fromR < routes.length; fromR++) {
    const from = routes[fromR];
    for (let idx = 0; idx < from.length; idx++) {
      const node = from[idx];
      const prev = idx === 0 ? -1 : from[idx - 1];
      const next = idx + 1 === from.length ? -1 : from[idx + 1];
      const removalGain = (prev === -1 ? depotDist[node] : matrix[prev][node]) + (next === -1 ? depotDist[node] : matrix[node][next]) - (prev === -1 && next === -1 ? 0 : (prev === -1 ? depotDist[next] : next === -1 ? depotDist[prev] : matrix[prev][next]));
      let bestToR = -1;
      let bestToPos = -1;
      let bestNet = 0;
      for (let toR = 0; toR < routes.length; toR++) {
        if (toR === fromR) continue;
        const to = routes[toR];
        for (let pos = 0; pos <= to.length; pos++) {
          const delta = insertionDelta(to, pos, node);
          const net = removalGain - delta;
          if (net > bestNet + 1e-9) {
            bestNet = net;
            bestToR = toR;
            bestToPos = pos;
          }
        }
      }
      if (bestToR >= 0) {
        from.splice(idx, 1);
        routes[bestToR].splice(bestToPos, 0, node);
        improved = true;
        break;
      }
    }
    if (improved) break;
  }
}

// Convert internal indices back to package ids.
const outRoutes = routes
  .filter(r => r.length > 0)
  .map(r => r.map(idx => packages[idx].id));

process.stdout.write(JSON.stringify({ routes: outRoutes }));
