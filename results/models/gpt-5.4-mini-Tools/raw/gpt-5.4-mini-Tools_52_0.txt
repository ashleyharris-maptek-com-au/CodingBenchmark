import math
from statistics import median


_state = {
    "speed_kmh": None,
    "x_m": None,
    "y_m": None,
    "heading_deg": None,
    "yaw_rate_dps": 0.0,
    "lane_offset_m": 0.0,
    "steer_i": 0.0,
    "speed_i": 0.0,
    "last_lane": 1,
    "last_target_lane": 1,
    "last_time": None,
}


def _clamp(v, lo, hi):
    return lo if v < lo else hi if v > hi else v


def _finite(v):
    return isinstance(v, (int, float)) and math.isfinite(v)


def _get(state, key, default=None):
    v = state.get(key, default)
    return v if _finite(v) else default


def _valid(state, key):
    return bool(state.get(key, False))


def _median_vals(vals):
    vals = [v for v in vals if _finite(v)]
    return median(vals) if vals else None


def _angle_wrap_deg(a):
    while a > 180:
        a -= 360
    while a < -180:
        a += 360
    return a


def _avg_angle_deg(angles):
    angles = [a for a in angles if _finite(a)]
    if not angles:
        return None
    s = sum(math.sin(math.radians(a)) for a in angles)
    c = sum(math.cos(math.radians(a)) for a in angles)
    return math.degrees(math.atan2(s, c))


def _safe_mean(vals):
    vals = [v for v in vals if _finite(v)]
    return sum(vals) / len(vals) if vals else None


def _sensor_vote(vals, valid_flags=None):
    items = []
    for i, v in enumerate(vals):
        if _finite(v) and (valid_flags is None or valid_flags[i]):
            items.append(v)
    if not items:
        return None
    return median(items)


def autopilot_step(state: dict, dt: float) -> dict:
    global _state

    dt = dt if _finite(dt) and dt > 1e-3 else _get(state, "dt", 0.1) or 0.1
    dt = _clamp(dt, 0.02, 0.2)

    wheel_vals = [
        _get(state, "wheel_speed_fl_kmh") if _valid(state, "wheel_speed_fl_valid") else None,
        _get(state, "wheel_speed_fr_kmh") if _valid(state, "wheel_speed_fr_valid") else None,
        _get(state, "wheel_speed_rl_kmh") if _valid(state, "wheel_speed_rl_valid") else None,
        _get(state, "wheel_speed_rr_kmh") if _valid(state, "wheel_speed_rr_valid") else None,
    ]
    wheel_speed = _median_vals(wheel_vals)

    gps_speeds = [
        _get(state, "gps_1_speed_kmh") if _valid(state, "gps_1_valid") else None,
        _get(state, "gps_2_speed_kmh") if _valid(state, "gps_2_valid") else None,
    ]
    gps_speed = _median_vals(gps_speeds)

    speed_candidates = [v for v in [wheel_speed, gps_speed, _get(state, "vehicle_speed_1_kmh"), _get(state, "vehicle_speed_2_kmh")] if _finite(v)]
    speed_kmh = median(speed_candidates) if speed_candidates else (_state["speed_kmh"] if _state["speed_kmh"] is not None else 0.0)
    if _state["speed_kmh"] is None:
        _state["speed_kmh"] = speed_kmh
    else:
        alpha = 0.25 if speed_kmh < 40 else 0.15
        _state["speed_kmh"] = (1 - alpha) * _state["speed_kmh"] + alpha * speed_kmh

    gyro_vals = [
        _get(state, "gyro_1_yaw_rate_dps") if _valid(state, "gyro_1_valid") else None,
        _get(state, "gyro_2_yaw_rate_dps") if _valid(state, "gyro_2_valid") else None,
        _get(state, "gyro_3_yaw_rate_dps") if _valid(state, "gyro_3_valid") else None,
    ]
    yaw_rate = _median_vals(gyro_vals) or 0.0
    _state["yaw_rate_dps"] = 0.7 * _state["yaw_rate_dps"] + 0.3 * yaw_rate

    gps_headings = [
        _get(state, "gps_1_heading_deg") if _valid(state, "gps_1_valid") else None,
        _get(state, "gps_2_heading_deg") if _valid(state, "gps_2_valid") else None,
    ]
    heading = _avg_angle_deg(gps_headings)
    if heading is None:
        heading = _state["heading_deg"] if _state["heading_deg"] is not None else 0.0
    if _state["heading_deg"] is None:
        _state["heading_deg"] = heading
    else:
        err = _angle_wrap_deg(heading - _state["heading_deg"])
        _state["heading_deg"] = _angle_wrap_deg(_state["heading_deg"] + _clamp(err, -10.0, 10.0) * 0.2)
        _state["heading_deg"] = _angle_wrap_deg(_state["heading_deg"] + _state["yaw_rate_dps"] * dt)

    gps_x = _median_vals([
        _get(state, "gps_1_x_m") if _valid(state, "gps_1_valid") else None,
        _get(state, "gps_2_x_m") if _valid(state, "gps_2_valid") else None,
    ])
    gps_y = _median_vals([
        _get(state, "gps_1_y_m") if _valid(state, "gps_1_valid") else None,
        _get(state, "gps_2_y_m") if _valid(state, "gps_2_valid") else None,
    ])
    if gps_x is not None:
        _state["x_m"] = gps_x if _state["x_m"] is None else 0.8 * _state["x_m"] + 0.2 * gps_x
    if gps_y is not None:
        _state["y_m"] = gps_y if _state["y_m"] is None else 0.8 * _state["y_m"] + 0.2 * gps_y

    lane_offsets = []
    for i in (1, 2, 3):
        if _valid(state, f"vision_{i}_lane_offset_m"):
            v = _get(state, f"vision_{i}_lane_offset_m")
            if _finite(v):
                lane_offsets.append(v)
    if lane_offsets:
        lane_offset = median(lane_offsets)
        _state["lane_offset_m"] = 0.7 * _state["lane_offset_m"] + 0.3 * lane_offset
    else:
        _state["lane_offset_m"] *= 0.98

    target_lane = state.get("target_lane", _state["last_target_lane"])
    if not isinstance(target_lane, int):
        target_lane = _state["last_target_lane"]
    target_lane = int(_clamp(target_lane, 0, 2))
    current_lane = state.get("current_lane", _state["last_lane"])
    if not isinstance(current_lane, int):
        current_lane = _state["last_lane"]
    current_lane = int(_clamp(current_lane, 0, 2))
    _state["last_lane"] = current_lane
    _state["last_target_lane"] = target_lane

    speed_limit_vals = [
        _get(state, "vision_1_speed_limit_kmh") if _valid(state, "vision_1_speed_limit_valid") else None,
        _get(state, "vision_2_speed_limit_kmh") if _valid(state, "vision_2_speed_limit_valid") else None,
        _get(state, "vision_3_speed_limit_kmh") if _valid(state, "vision_3_speed_limit_valid") else None,
    ]
    speed_limit = _median_vals(speed_limit_vals) or 130.0

    mu = _get(state, "estimated_mu", 1.0)
    if not _finite(mu):
        mu = 1.0
    traction_limit = 18.0 + 45.0 * _clamp(mu, 0.15, 1.0)
    desired_speed = min(_get(state, "target_speed_kmh", speed_limit), speed_limit, traction_limit)
    if mu < 0.35:
        desired_speed = min(desired_speed, 45.0)
    if mu < 0.2:
        desired_speed = min(desired_speed, 25.0)

    obstacle_ranges = []
    obstacle_rel_speeds = []
    for i in (1, 2, 3):
        if _valid(state, f"vision_{i}_obstacle_valid") and state.get(f"vision_{i}_obstacle_detected"):
            d = _get(state, f"vision_{i}_obstacle_dist_m")
            rs = _get(state, f"vision_{i}_obstacle_rel_speed_kmh")
            if _finite(d):
                obstacle_ranges.append(d)
            if _finite(rs):
                obstacle_rel_speeds.append(rs)
    lidar_front = []
    for ang in (+5, +15, +30, -5, -15, -30, +45, -45):
        k = f"lidar_{ang:+04d}_deg_range_m"
        vk = f"lidar_{ang:+04d}_deg_valid"
        if _valid(state, vk):
            d = _get(state, k)
            if _finite(d):
                lidar_front.append(d)
    front_range = _median_vals(obstacle_ranges + lidar_front)

    brake_pressure = _median_vals([
        _get(state, "brake_pressure_1_pct") if _valid(state, "brake_pressure_1_valid") else None,
        _get(state, "brake_pressure_2_pct") if _valid(state, "brake_pressure_2_valid") else None,
    ]) or 0.0

    accel_long = _safe_mean([
        _get(state, "accel_1_longitudinal_ms2") if _valid(state, "accel_1_valid") else None,
        _get(state, "accel_2_longitudinal_ms2") if _valid(state, "accel_2_valid") else None,
        _get(state, "accel_3_longitudinal_ms2") if _valid(state, "accel_3_valid") else None,
    ]) or 0.0
    accel_lat = _safe_mean([
        _get(state, "accel_1_lateral_ms2") if _valid(state, "accel_1_valid") else None,
        _get(state, "accel_2_lateral_ms2") if _valid(state, "accel_2_valid") else None,
        _get(state, "accel_3_lateral_ms2") if _valid(state, "accel_3_valid") else None,
    ]) or 0.0

    lane_change_needed = target_lane != current_lane
    if front_range is not None and front_range < 18.0:
        desired_speed = min(desired_speed, max(0.0, _state["speed_kmh"] - 15.0))
    if front_range is not None and front_range < 10.0:
        desired_speed = 0.0

    speed_error = desired_speed - _state["speed_kmh"]
    _state["speed_i"] = _clamp(_state["speed_i"] + speed_error * dt * 0.02, -8.0, 8.0)
    throttle = _clamp(0.18 + 0.018 * speed_error + _state["speed_i"], 0.0, 1.0)
    brake = 0.0

    if speed_error < -2.0:
        brake = _clamp((-speed_error) / 25.0, 0.0, 1.0)
        throttle = min(throttle, 0.15)
    if front_range is not None:
        if front_range < 25.0:
            brake = max(brake, _clamp((25.0 - front_range) / 20.0, 0.0, 1.0))
            throttle = min(throttle, 0.2)
        if front_range < 8.0:
            brake = 1.0
            throttle = 0.0

    if mu < 0.3:
        throttle = min(throttle, 0.35)
        brake = min(1.0, brake * 1.2)

    if brake_pressure > 50.0:
        brake = min(1.0, brake + 0.3)

    if _get(state, "throttle_stuck", 0):
        throttle = min(throttle, 0.1)
        brake = max(brake, 0.25)

    left_line = _median_vals([
        _get(state, "vision_1_left_line_dist_m") if _valid(state, "vision_1_left_line_valid") else None,
        _get(state, "vision_2_left_line_dist_m") if _valid(state, "vision_2_left_line_valid") else None,
        _get(state, "vision_3_left_line_dist_m") if _valid(state, "vision_3_left_line_valid") else None,
    ])
    right_line = _median_vals([
        _get(state, "vision_1_right_line_dist_m") if _valid(state, "vision_1_right_line_valid") else None,
        _get(state, "vision_2_right_line_dist_m") if _valid(state, "vision_2_right_line_valid") else None,
        _get(state, "vision_3_right_line_dist_m") if _valid(state, "vision_3_right_line_valid") else None,
    ])
    curvature = _median_vals([
        _get(state, "vision_1_road_curvature_1pm"),
        _get(state, "vision_2_road_curvature_1pm"),
        _get(state, "vision_3_road_curvature_1pm"),
    ]) or 0.0

    lane_center_error = -_state["lane_offset_m"]
    if left_line is not None and right_line is not None:
        lane_width = max(2.8, min(4.5, left_line + right_line))
        lane_center_error = (right_line - left_line) * 0.5
        lane_center_error = _clamp(lane_center_error, -lane_width * 0.4, lane_width * 0.4)
    elif left_line is not None:
        lane_center_error = 1.7 - left_line
    elif right_line is not None:
        lane_center_error = right_line - 1.7

    if lane_change_needed:
        lane_center_error += 0.8 * (target_lane - current_lane)

    heading_error = _angle_wrap_deg((-curvature * 57.3) - (_state["yaw_rate_dps"] * 0.15))
    steer = 0.65 * lane_center_error + 0.05 * heading_error - 0.015 * _state["yaw_rate_dps"] - 0.02 * accel_lat
    if front_range is not None and front_range < 20.0:
        steer += 0.2 * (1 if target_lane > current_lane else -1 if target_lane < current_lane else 0)
    if mu < 0.3:
        steer *= 0.65
    steer = _clamp(steer, -1.0, 1.0)

    if lane_change_needed and front_range is not None and front_range > 20.0:
        steer += _clamp(0.25 * (target_lane - current_lane), -0.35, 0.35)
        steer = _clamp(steer, -1.0, 1.0)

    if abs(_state["yaw_rate_dps"]) > 12.0 and _state["speed_kmh"] > 60.0:
        steer *= 0.85
        brake = max(brake, 0.1)

    if abs(accel_lat) > 3.5:
        throttle = min(throttle, 0.25)
        brake = max(brake, 0.05)

    _state["steer_i"] = _clamp(_state["steer_i"] + lane_center_error * dt * 0.02, -0.5, 0.5)
    steer = _clamp(steer + _state["steer_i"], -1.0, 1.0)

    if front_range is not None and front_range < 6.0 and _state["speed_kmh"] > 5.0:
        brake = 1.0
        throttle = 0.0
        steer = _clamp(steer * 0.5, -0.5, 0.5)

    if _state["speed_kmh"] < 1.0 and desired_speed <= 1.0:
        throttle = 0.0
        brake = max(brake, 0.2)

    _state["speed_kmh"] = max(0.0, _state["speed_kmh"] + (throttle * 2.5 - brake * 4.5) * dt * 3.6)
    _state["speed_kmh"] = 0.7 * _state["speed_kmh"] + 0.3 * speed_kmh

    out = {}
    out["steering"] = float(_clamp(steer, -1.0, 1.0))
    out["throttle"] = float(_clamp(throttle, 0.0, 1.0))
    out["brake"] = float(_clamp(brake, 0.0, 1.0))
    return out
