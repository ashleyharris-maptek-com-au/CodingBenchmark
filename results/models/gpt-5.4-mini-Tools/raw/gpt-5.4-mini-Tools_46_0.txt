#version 450

layout(local_size_x = 256) in;

// Each body occupies two vec4s in the buffers:
//   inData[bodyIndex * 2 + 0] = vec4(position.xyz, mass)
//   inData[bodyIndex * 2 + 1] = vec4(velocity.xyz, 0.0)
// The output buffer uses the same layout.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;   // counts.x = N
    vec4 physics;   // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    if (i >= N) {
        return;
    }

    uint base = i * 2u;
    vec4 selfPosMass = inData[base + 0u];
    vec4 selfVelPad  = inData[base + 1u];

    vec3 pos = selfPosMass.xyz;
    float mass = selfPosMass.w;
    vec3 vel = selfVelPad.xyz;

    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;
    float softening2 = softening * softening;

    vec3 acc = vec3(0.0);

    for (uint j = 0u; j < N; ++j) {
        if (j == i) {
            continue;
        }

        uint jBase = j * 2u;
        vec3 posJ = inData[jBase + 0u].xyz;
        float massJ = inData[jBase + 0u].w;

        vec3 r = posJ - pos;
        float dist2 = dot(r, r) + softening2;
        float invDist = inversesqrt(dist2);
        float invDist3 = invDist * invDist * invDist;

        acc += G * massJ * r * invDist3;
    }

    vel += acc * dt;
    pos += vel * dt;

    outData[base + 0u] = vec4(pos, mass);
    outData[base + 1u] = vec4(vel, 0.0);
}
