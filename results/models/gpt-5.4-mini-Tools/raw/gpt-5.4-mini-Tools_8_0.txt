use std::collections::VecDeque;
use std::io::{self, Read, Write};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let h: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let w: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };

    let mut grid: Vec<Vec<u8>> = Vec::with_capacity(h);
    let mut start = None;
    let mut goal = None;

    for y in 0..h {
        let row = it.next().unwrap_or("");
        let mut bytes = row.as_bytes().to_vec();
        if bytes.len() < w {
            bytes.resize(w, b' ');
        } else if bytes.len() > w {
            bytes.truncate(w);
        }
        for x in 0..w {
            match bytes[x] {
                b'A' => start = Some((x, y)),
                b'B' => goal = Some((x, y)),
                _ => {}
            }
        }
        grid.push(bytes);
    }

    let (sx, sy) = match start {
        Some(p) => p,
        None => {
            return;
        }
    };
    let (gx, gy) = match goal {
        Some(p) => p,
        None => {
            return;
        }
    };

    let n = h.saturating_mul(w);
    let mut prev = vec![usize::MAX; n];
    let mut q = VecDeque::new();
    let sidx = sy * w + sx;
    let gidx = gy * w + gx;
    prev[sidx] = sidx;
    q.push_back((sx, sy));

    while let Some((x, y)) = q.pop_front() {
        if x == gx && y == gy {
            break;
        }

        if x > 0 {
            let nx = x - 1;
            let ny = y;
            let idx = ny * w + nx;
            if prev[idx] == usize::MAX && grid[ny][nx] != b'#' {
                prev[idx] = y * w + x;
                q.push_back((nx, ny));
            }
        }
        if x + 1 < w {
            let nx = x + 1;
            let ny = y;
            let idx = ny * w + nx;
            if prev[idx] == usize::MAX && grid[ny][nx] != b'#' {
                prev[idx] = y * w + x;
                q.push_back((nx, ny));
            }
        }
        if y > 0 {
            let nx = x;
            let ny = y - 1;
            let idx = ny * w + nx;
            if prev[idx] == usize::MAX && grid[ny][nx] != b'#' {
                prev[idx] = y * w + x;
                q.push_back((nx, ny));
            }
        }
        if y + 1 < h {
            let nx = x;
            let ny = y + 1;
            let idx = ny * w + nx;
            if prev[idx] == usize::MAX && grid[ny][nx] != b'#' {
                prev[idx] = y * w + x;
                q.push_back((nx, ny));
            }
        }
    }

    let mut out = String::new();
    if prev[gidx] == usize::MAX {
        out.push_str("0\n");
        let mut stdout = io::BufWriter::new(io::stdout().lock());
        stdout.write_all(out.as_bytes()).unwrap();
        return;
    }

    let mut path = Vec::new();
    let mut cur = gidx;
    while cur != sidx {
        path.push(cur);
        cur = prev[cur];
    }
    path.push(sidx);
    path.reverse();

    out.push_str(&format!("{}\n", path.len()));
    for idx in path {
        let y = idx / w;
        let x = idx % w;
        out.push_str(&format!("{} {}\n", x, y));
    }

    let mut stdout = io::BufWriter::new(io::stdout().lock());
    stdout.write_all(out.as_bytes()).unwrap();
}
