#include <bits/stdc++.h>
using namespace std;

struct Solver {
    int n;
    vector<long long> w;
    vector<vector<int>> adj;
    vector<int> bestAssign;
    vector<int> curAssign;
    vector<char> alive;
    long long bestValue = LLONG_MIN;
    long long curValue = 0;

    long long upperBound() const {
        long long ub = curValue;
        for (int i = 0; i < n; ++i) {
            if (alive[i] && w[i] > 0) ub += w[i];
        }
        return ub;
    }

    int pickVertex() const {
        int best = -1;
        int bestDeg = -1;
        for (int i = 0; i < n; ++i) {
            if (!alive[i]) continue;
            int deg = 0;
            for (int v : adj[i]) if (alive[v]) ++deg;
            if (deg > bestDeg || (deg == bestDeg && (best == -1 || w[i] > w[best]))) {
                bestDeg = deg;
                best = i;
            }
        }
        return best;
    }

    void removeVertex(int v, vector<int>& removed) {
        if (!alive[v]) return;
        alive[v] = 0;
        removed.push_back(v);
    }

    void restoreVertices(const vector<int>& removed) {
        for (int v : removed) alive[v] = 1;
    }

    void dfs() {
        if (curValue > bestValue) {
            bestValue = curValue;
            bestAssign = curAssign;
        }
        if (upperBound() <= bestValue) return;

        int v = pickVertex();
        if (v == -1) return;

        {
            vector<int> removedV;
            vector<int> removedNbrs;
            removeVertex(v, removedV);
            for (int u : adj[v]) if (alive[u]) removeVertex(u, removedNbrs);
            curAssign[v] = 1;
            curValue += w[v];
            dfs();
            curValue -= w[v];
            curAssign[v] = -1;
            restoreVertices(removedNbrs);
            restoreVertices(removedV);
        }

        {
            vector<int> removedV;
            removeVertex(v, removedV);
            curAssign[v] = 0;
            dfs();
            curAssign[v] = -1;
            restoreVertices(removedV);
        }
    }

    void solve() {
        bestAssign.assign(n, 0);
        curAssign.assign(n, -1);
        alive.assign(n, 1);
        bestValue = 0;
        dfs();
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;
    Solver s;
    s.n = n;
    s.w.resize(n);
    s.adj.assign(n, {});
    for (int i = 0; i < n; ++i) cin >> s.w[i];
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        --u; --v;
        if (u == v) continue;
        s.adj[u].push_back(v);
        s.adj[v].push_back(u);
    }

    s.solve();

    long long obj = 0;
    for (int i = 0; i < n; ++i) if (s.bestAssign[i]) obj += s.w[i];
    cout << obj << '\n';
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << (s.bestAssign[i] ? 1 : 0);
    }
    cout << '\n';
    return 0;
}
