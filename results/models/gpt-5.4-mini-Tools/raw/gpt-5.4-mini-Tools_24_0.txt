#include <vector>

using namespace std;

struct Vec3 {
    double x = 0, y = 0, z = 0;
};

static Vec3 operator+(const Vec3& a, const Vec3& b) { return {a.x + b.x, a.y + b.y, a.z + b.z}; }
static Vec3 operator-(const Vec3& a, const Vec3& b) { return {a.x - b.x, a.y - b.y, a.z - b.z}; }
static Vec3 operator*(const Vec3& a, double s) { return {a.x * s, a.y * s, a.z * s}; }
static Vec3 operator/(const Vec3& a, double s) { return {a.x / s, a.y / s, a.z / s}; }

static double dot(const Vec3& a, const Vec3& b) { return a.x * b.x + a.y * b.y + a.z * b.z; }
static double norm(const Vec3& a) { return sqrt(dot(a, a)); }

static double clampd(double v, double lo, double hi) { return max(lo, min(hi, v)); }

struct Terrain {
    int width = 0, depth = 0, height = 0;
    string path;
    atomic<bool> ready{false};
    vector<uint16_t> a, b, c;

    size_t idx(int x, int y) const { return static_cast<size_t>(x) * depth + y; }

    void load() {
        a.resize(static_cast<size_t>(width) * depth);
        b.resize(static_cast<size_t>(width) * depth);
        c.resize(static_cast<size_t>(width) * depth);
        ifstream in(path, ios::binary);
        if (!in) {
            ready = true;
            return;
        }
        for (int x = 0; x < width; ++x) {
            for (int y = 0; y < depth; ++y) {
                unsigned char buf[6];
                in.read(reinterpret_cast<char*>(buf), 6);
                if (in.gcount() != 6) {
                    ready = true;
                    return;
                }
                uint16_t va = static_cast<uint16_t>(buf[0] | (buf[1] << 8));
                uint16_t vb = static_cast<uint16_t>(buf[2] | (buf[3] << 8));
                uint16_t vc = static_cast<uint16_t>(buf[4] | (buf[5] << 8));
                size_t i = idx(x, y);
                a[i] = va;
                b[i] = vb;
                c[i] = vc;
            }
        }
        ready = true;
    }

    double groundHeight(double x, double y) const {
        int ix = (int)floor(x);
        int iy = (int)floor(y);
        if (ix < 0 || iy < 0 || ix >= width || iy >= depth) return 1e9;
        return a[idx(ix, iy)];
    }
};

struct State {
    Vec3 p, v;
    Vec3 ang;
    Vec3 rates;
    double fuel = 0;
};

static bool parseState(const string& line, State& s) {
    istringstream iss(line);
    iss >> s.p.x >> s.p.y >> s.p.z >> s.v.x >> s.v.y >> s.v.z;
    iss >> s.ang.x >> s.ang.y >> s.ang.z;
    iss >> s.rates.x >> s.rates.y >> s.rates.z >> s.fuel;
    return !iss.fail();
}

static Vec3 basisFromAngles(double pitch, double yaw) {
    double cp = cos(pitch), sp = sin(pitch);
    double cy = cos(yaw), sy = sin(yaw);
    return {sp * cy, sp * sy, cp};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Terrain terrain;
    double gravity = 1.62, max_thrust = 1.0;
    Vec3 start{}, target{};
    string terrainPath;

    if (!(cin >> terrain.width >> terrain.depth >> terrain.height >> gravity >> max_thrust >> terrainPath)) {
        return 0;
    }
    if (!(cin >> start.x >> start.y >> start.z >> target.x >> target.y >> target.z)) {
        return 0;
    }
    string dummy;
    getline(cin, dummy);
    getline(cin, terrain.path);
    if (terrain.path.empty()) terrain.path = terrainPath;

    thread loader([&] { terrain.load(); });
    loader.detach();

    const double dt = 0.1;
    const double maxTilt = 0.75;
    const double kpAng = 1.8, kdAng = 0.4;
    const double kpVel = 0.35, kdVel = 0.18;
    const double kpAlt = 0.08, kdAlt = 0.6;

    cout.setf(ios::fixed);
    cout.precision(6);

    string line;
    while (true) {
        if (!getline(cin, line)) break;
        if (line.empty()) continue;
        State s;
        if (!parseState(line, s)) continue;

        Vec3 toTarget = target - s.p;
        double distXY = sqrt(toTarget.x * toTarget.x + toTarget.y * toTarget.y);
        double desiredAlt = target.z + max(5.0, min((double)min(terrain.width, terrain.depth) * 0.02, 50.0));
        if (distXY < 30.0) desiredAlt = target.z + 8.0;
        if (distXY < 10.0) desiredAlt = target.z + 3.5;

        double altErr = s.p.z - desiredAlt;
        double vzDes = clampd(-kpAlt * altErr, -12.0, 8.0);
        if (distXY < 20.0) vzDes = clampd(-kpAlt * altErr, -6.0, 4.0);
        double thrustZ = (gravity + kdAlt * (vzDes - s.v.z)) / max_thrust;

        Vec3 lateralWanted{0, 0, 0};
        if (distXY > 1e-6) {
            double kxy = distXY < 40.0 ? 0.12 : 0.06;
            lateralWanted.x = clampd(-kxy * toTarget.x - 0.22 * s.v.x, -0.9, 0.9);
            lateralWanted.y = clampd(-kxy * toTarget.y - 0.22 * s.v.y, -0.9, 0.9);
        }

        Vec3 desiredAcc{lateralWanted.x, lateralWanted.y, thrustZ * max_thrust - gravity};
        Vec3 desiredUp = desiredAcc;
        double n = norm(desiredUp);
        if (n < 1e-6) desiredUp = {0, 0, 1}, n = 1.0;
        desiredUp = desiredUp / n;

        double desiredPitch = asin(clampd(desiredUp.x, -1.0, 1.0));
        double desiredYaw = atan2(desiredUp.y, max(1e-9, desiredUp.z));
        if (fabs(desiredYaw) > 1.2) desiredYaw = clampd(desiredYaw, -1.2, 1.2);
        desiredPitch = clampd(desiredPitch, -maxTilt, maxTilt);

        double pitchCmd = clampd(kpAng * (desiredPitch - s.ang.x) - kdAng * s.rates.x, -1.0, 1.0);
        double yawCmd = clampd(kpAng * (desiredYaw - s.ang.y) - kdAng * s.rates.y, -1.0, 1.0);
        double rollCmd = clampd(-0.8 * s.ang.z - 0.3 * s.rates.z, -1.0, 1.0);

        thrustZ = clampd(thrustZ, 0.0, 1.0);
        if (s.fuel <= 0.01) thrustZ = 0.0;

        cout << thrustZ << ' ' << pitchCmd << ' ' << yawCmd << ' ' << rollCmd << '\n';
        cout.flush();
    }
    return 0;
}
