using namespace std;

static constexpr double PI = 3.141592653589793238462643383279502884;

struct TerrainColumn {
    uint16_t a = 0, b = 0, c = 0;
};

struct World {
    int width = 0, depth = 0, height = 0;
    double gravity = 1.62;
    double max_thrust = 1.0;
    double fuel_capacity = 0.0;
    double target_x = 0, target_y = 0, target_z = 0;
    vector<TerrainColumn> cols;
};

struct State {
    double x = 0, y = 0, z = 0;
    double vx = 0, vy = 0, vz = 0;
    double pitch = 0, yaw = 0, roll = 0;
    double pitch_rate = 0, yaw_rate = 0, roll_rate = 0;
    double fuel = 0;
};

static inline double clampd(double v, double lo, double hi) {
    return max(lo, min(hi, v));
}

static inline double wrap_pi(double a) {
    while (a > PI) a -= 2.0 * PI;
    while (a < -PI) a += 2.0 * PI;
    return a;
}

static inline double safe_sqrt(double v) {
    return sqrt(max(0.0, v));
}

static bool read_exact_u16_le(ifstream& f, uint16_t& out) {
    unsigned char b[2];
    if (!f.read(reinterpret_cast<char*>(b), 2)) return false;
    out = static_cast<uint16_t>(b[0] | (static_cast<uint16_t>(b[1]) << 8));
    return true;
}

static void parse_terrain_async(World* world, string path, atomic<bool>* ready) {
    ifstream f(path, ios::binary);
    if (!f) {
        ready->store(true, memory_order_release);
        return;
    }
    world->cols.resize(static_cast<size_t>(world->width) * static_cast<size_t>(world->depth));
    for (int x = 0; x < world->width; ++x) {
        for (int y = 0; y < world->depth; ++y) {
            TerrainColumn col;
            if (!read_exact_u16_le(f, col.a) || !read_exact_u16_le(f, col.b) || !read_exact_u16_le(f, col.c)) {
                col.a = col.b = col.c = 0;
            }
            world->cols[static_cast<size_t>(x) * world->depth + y] = col;
        }
    }
    ready->store(true, memory_order_release);
}

static double ground_height_at(const World& w, double x, double y) {
    if (w.cols.empty()) return 0.0;
    int ix = static_cast<int>(floor(x));
    int iy = static_cast<int>(floor(y));
    ix = clamp(ix, 0, w.width - 1);
    iy = clamp(iy, 0, w.depth - 1);
    const auto& c = w.cols[static_cast<size_t>(ix) * w.depth + iy];
    return static_cast<double>(c.a);
}

static double local_clearance(const World& w, double x, double y, double z) {
    if (w.cols.empty()) return z;
    int ix = static_cast<int>(floor(x));
    int iy = static_cast<int>(floor(y));
    ix = clamp(ix, 0, w.width - 1);
    iy = clamp(iy, 0, w.depth - 1);
    const auto& c = w.cols[static_cast<size_t>(ix) * w.depth + iy];
    if (z < c.a) return z - static_cast<double>(c.a);
    if (z >= c.b && z < c.c) return static_cast<double>(c.b) - z;
    return min(z - static_cast<double>(c.a), static_cast<double>(c.c) - z);
}

static void make_command(const World& w, const State& s, double& thrust, double& pitch_cmd, double& yaw_cmd, double& roll_cmd) {
    const double dx = w.target_x - s.x;
    const double dy = w.target_y - s.y;
    const double dz = w.target_z - s.z;
    const double dist_xy = hypot(dx, dy);
    const double dist = sqrt(dx * dx + dy * dy + dz * dz);

    double safe_ground = ground_height_at(w, s.x, s.y) + 2.0;
    if (!w.cols.empty()) {
        safe_ground = max(safe_ground, w.target_z);
        safe_ground = max(safe_ground, ground_height_at(w, w.target_x, w.target_y) + 1.0);
    }
    double clearance = local_clearance(w, s.x, s.y, s.z);

    double desired_vz = 0.0;
    if (s.z > safe_ground + 25.0) desired_vz = -min(25.0, (s.z - safe_ground) * 0.35);
    if (s.z < safe_ground + 12.0) desired_vz = -min(4.0, max(0.5, (s.z - safe_ground) * 0.15));
    if (dist < 30.0) desired_vz = clampd((w.target_z - s.z) * 0.8, -2.5, 2.5);

    double desired_ax = clampd(dx * 0.04 + s.vx * 0.7, -6.0, 6.0);
    double desired_ay = clampd(dy * 0.04 + s.vy * 0.7, -6.0, 6.0);

    if (dist_xy < 40.0) {
        desired_ax = clampd(dx * 0.1 + s.vx * 0.8, -4.0, 4.0);
        desired_ay = clampd(dy * 0.1 + s.vy * 0.8, -4.0, 4.0);
    }
    if (dist_xy < 12.0) {
        desired_ax = clampd(dx * 0.18 + s.vx * 0.9, -2.0, 2.0);
        desired_ay = clampd(dy * 0.18 + s.vy * 0.9, -2.0, 2.0);
    }

    double ax = desired_ax;
    double ay = desired_ay;
    double az = w.gravity + (desired_vz - s.vz) * 0.9;
    az = clampd(az, 0.0, w.max_thrust * 0.98);

    double lateral = sqrt(ax * ax + ay * ay);
    if (clearance < 10.0) {
        ax *= 0.35;
        ay *= 0.35;
        az = max(az, w.gravity + 0.8);
    }

    double total = sqrt(ax * ax + ay * ay + az * az);
    if (total > w.max_thrust && total > 1e-9) {
        double scale = w.max_thrust / total;
        ax *= scale;
        ay *= scale;
        az *= scale;
        total = w.max_thrust;
    }

    thrust = clampd(total / max(1e-9, w.max_thrust), 0.0, 1.0);

    double desired_pitch = 0.0;
    double desired_yaw = 0.0;
    if (total > 1e-6) {
        desired_pitch = -atan2(ax, max(1e-9, az));
        desired_yaw = atan2(ay, max(1e-9, az));
    }
    desired_pitch = clampd(desired_pitch, -PI / 3.0, PI / 3.0);
    desired_yaw = clampd(desired_yaw, -PI / 3.0, PI / 3.0);

    double pitch_err = wrap_pi(desired_pitch - s.pitch);
    double yaw_err = wrap_pi(desired_yaw - s.yaw);
    double roll_err = wrap_pi(0.0 - s.roll);

    pitch_cmd = clampd(pitch_err * 2.5 - s.pitch_rate * 1.1, -1.0, 1.0);
    yaw_cmd = clampd(yaw_err * 2.5 - s.yaw_rate * 1.1, -1.0, 1.0);
    roll_cmd = clampd(roll_err * 1.5 - s.roll_rate * 0.8, -1.0, 1.0);

    if (dist_xy < 20.0 && fabs(s.vx) + fabs(s.vy) < 4.0) {
        roll_cmd = clampd(-s.roll * 1.0 - s.roll_rate * 0.8, -1.0, 1.0);
    }
}

static bool read_state_line(istream& in, State& s) {
    if (!(in >> s.x >> s.y >> s.z >> s.vx >> s.vy >> s.vz >> s.pitch >> s.yaw >> s.roll
          >> s.pitch_rate >> s.yaw_rate >> s.roll_rate >> s.fuel)) {
        return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    World world;
    if (!(cin >> world.width >> world.depth >> world.height >> world.gravity >> world.max_thrust >> world.fuel_capacity)) {
        return 0;
    }
    State initial;
    if (!(cin >> initial.x >> initial.y >> initial.z >> world.target_x >> world.target_y >> world.target_z)) {
        return 0;
    }
    string terrain_path;
    getline(cin, terrain_path);
    if (terrain_path.empty()) getline(cin, terrain_path);

    atomic<bool> terrain_ready{false};
    thread terrain_thread(parse_terrain_async, &world, terrain_path, &terrain_ready);
    terrain_thread.detach();

    cout.setf(ios::fixed);
    cout << setprecision(6);

    State s = initial;
    while (true) {
        if (!read_state_line(cin, s)) break;

        double thrust = 0.0, pitch_cmd = 0.0, yaw_cmd = 0.0, roll_cmd = 0.0;
        make_command(world, s, thrust, pitch_cmd, yaw_cmd, roll_cmd);

        if (s.fuel <= 0.0) thrust = 0.0;
        if (!terrain_ready.load(memory_order_acquire) && world.cols.empty()) {
            // During early startup or very large maps, keep the controller responsive.
            thrust = min(thrust, 1.0);
        }

        cout << thrust << ' ' << pitch_cmd << ' ' << yaw_cmd << ' ' << roll_cmd << '\n';
        cout.flush();
    }
    return 0;
}
