#include <bits/stdc++.h>
using namespace std;

struct Task {
    int machine;
    long long dur;
};

struct ScheduledTask {
    int job;
    int idx;
    int machine;
    long long start;
    long long dur;
};

static inline long long maxll(long long a, long long b) { return a > b ? a : b; }

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    if (!(cin >> J >> M)) return 0;

    vector<vector<Task>> jobs(J);
    long long totalTasks = 0;
    long long maxDur = 0;
    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        totalTasks += T;
        for (int i = 0; i < T; ++i) {
            int machine;
            long long dur;
            cin >> machine >> dur;
            jobs[j][i] = {machine, dur};
            maxDur = maxll(maxDur, dur);
        }
    }

    vector<vector<long long>> starts(J);
    for (int j = 0; j < J; ++j) starts[j].assign(jobs[j].size(), 0);

    if (J == 0) return 0;

    // Small instances: use a more exact greedy with tie-breaking by earliest completion.
    if (totalTasks <= 2000) {
        vector<long long> jobReady(J, 0), machReady(M, 0);
        vector<size_t> pos(J, 0);
        long long done = 0;
        while (done < totalTasks) {
            long long bestStart = LLONG_MAX, bestEnd = LLONG_MAX;
            int bestJob = -1;
            for (int j = 0; j < J; ++j) {
                if (pos[j] >= jobs[j].size()) continue;
                const auto &t = jobs[j][pos[j]];
                long long s = maxll(jobReady[j], machReady[t.machine]);
                long long e = s + t.dur;
                if (e < bestEnd || (e == bestEnd && s < bestStart) || (e == bestEnd && s == bestStart && j < bestJob)) {
                    bestEnd = e;
                    bestStart = s;
                    bestJob = j;
                }
            }
            if (bestJob == -1) break;
            auto &t = jobs[bestJob][pos[bestJob]];
            long long s = maxll(jobReady[bestJob], machReady[t.machine]);
            starts[bestJob][pos[bestJob]] = s;
            long long e = s + t.dur;
            jobReady[bestJob] = e;
            machReady[t.machine] = e;
            ++pos[bestJob];
            ++done;
        }
    } else {
        // Large instances: dispatch jobs in a round-robin fashion while respecting precedence and machine availability.
        vector<long long> jobReady(J, 0), machReady(M, 0);
        vector<size_t> pos(J, 0);
        vector<int> active;
        active.reserve(J);
        for (int j = 0; j < J; ++j) if (!jobs[j].empty()) active.push_back(j);

        size_t idx = 0;
        long long done = 0;
        while (!active.empty()) {
            if (idx >= active.size()) idx = 0;
            int j = active[idx];
            if (pos[j] >= jobs[j].size()) {
                active[idx] = active.back();
                active.pop_back();
                if (idx >= active.size()) idx = 0;
                continue;
            }
            const auto &t = jobs[j][pos[j]];
            long long s = maxll(jobReady[j], machReady[t.machine]);
            starts[j][pos[j]] = s;
            long long e = s + t.dur;
            jobReady[j] = e;
            machReady[t.machine] = e;
            ++pos[j];
            ++done;
            ++idx;
        }
    }

    ostringstream out;
    for (int j = 0; j < J; ++j) {
        for (size_t i = 0; i < jobs[j].size(); ++i) {
            if (i) out << ' ';
            out << starts[j][i] << ' ' << jobs[j][i].machine;
        }
        out << '\n';
    }
    cout << out.str();
    return 0;
}
