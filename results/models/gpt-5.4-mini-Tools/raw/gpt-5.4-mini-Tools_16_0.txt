#include <bits/stdc++.h>
using namespace std;

struct Task {
    int machine;
    int duration;
};

struct ReadyItem {
    long long priority;
    int job;
    int idx;
    long long release;
    int duration;
};

struct ReadyCmp {
    bool operator()(const ReadyItem& a, const ReadyItem& b) const {
        if (a.priority != b.priority) return a.priority > b.priority;
        if (a.release != b.release) return a.release > b.release;
        if (a.duration != b.duration) return a.duration > b.duration;
        if (a.job != b.job) return a.job > b.job;
        return a.idx > b.idx;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    if (!(cin >> J >> M)) return 0;

    vector<vector<Task>> jobs(J);
    long long totalTasks = 0;
    long long totalDuration = 0;
    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        totalTasks += T;
        for (int i = 0; i < T; ++i) {
            cin >> jobs[j][i].machine >> jobs[j][i].duration;
            totalDuration += jobs[j][i].duration;
        }
    }

    vector<vector<long long>> start(J);
    for (int j = 0; j < J; ++j) start[j].assign(jobs[j].size(), 0);

    vector<long long> jobReady(J, 0);
    vector<long long> machineReady(M, 0);
    vector<long long> remainingWork(J, 0);
    for (int j = 0; j < J; ++j) {
        for (const auto& t : jobs[j]) remainingWork[j] += t.duration;
    }

    const bool smallInstance = (totalTasks <= 2000 && J <= 80 && M <= 80);

    if (smallInstance) {
        // Small instances: stronger priority signal using remaining work, machine load, and shortest duration.
        vector<long long> machineLoad(M, 0);
        for (int j = 0; j < J; ++j) {
            for (auto &t : jobs[j]) machineLoad[t.machine] += t.duration;
        }

        priority_queue<ReadyItem, vector<ReadyItem>, ReadyCmp> pq;
        for (int j = 0; j < J; ++j) {
            if (!jobs[j].empty()) {
                const auto& t = jobs[j][0];
                long long pr = -(remainingWork[j] + machineLoad[t.machine]);
                pq.push({pr, j, 0, 0, t.duration});
            }
        }

        while (!pq.empty()) {
            auto cur = pq.top();
            pq.pop();
            int j = cur.job, i = cur.idx;
            const auto& t = jobs[j][i];
            long long s = max(jobReady[j], machineReady[t.machine]);
            start[j][i] = s;
            long long finish = s + t.duration;
            jobReady[j] = finish;
            machineReady[t.machine] = finish;
            remainingWork[j] -= t.duration;
            if (i + 1 < (int)jobs[j].size()) {
                const auto& nxt = jobs[j][i + 1];
                long long pr = -(remainingWork[j] + max(0LL, machineReady[nxt.machine] - jobReady[j]) + nxt.duration);
                pq.push({pr, j, i + 1, jobReady[j], nxt.duration});
            }
        }
    } else {
        // Large instances: scalable greedy list scheduling with a light priority on critical jobs.
        priority_queue<ReadyItem, vector<ReadyItem>, ReadyCmp> pq;
        for (int j = 0; j < J; ++j) {
            if (!jobs[j].empty()) {
                const auto& t = jobs[j][0];
                long long pr = -(remainingWork[j]);
                pq.push({pr, j, 0, 0, t.duration});
            }
        }

        while (!pq.empty()) {
            auto cur = pq.top();
            pq.pop();
            int j = cur.job, i = cur.idx;
            const auto& t = jobs[j][i];
            long long s = max(jobReady[j], machineReady[t.machine]);
            start[j][i] = s;
            long long finish = s + t.duration;
            jobReady[j] = finish;
            machineReady[t.machine] = finish;
            remainingWork[j] -= t.duration;
            if (i + 1 < (int)jobs[j].size()) {
                const auto& nxt = jobs[j][i + 1];
                long long pr = -(remainingWork[j] + nxt.duration);
                pq.push({pr, j, i + 1, jobReady[j], nxt.duration});
            }
        }
    }

    ostringstream out;
    for (int j = 0; j < J; ++j) {
        for (int i = 0; i < (int)jobs[j].size(); ++i) {
            if (i) out << ' ';
            out << start[j][i] << ' ' << jobs[j][i].machine;
        }
        out << '\n';
    }
    cout << out.str();
    return 0;
}
