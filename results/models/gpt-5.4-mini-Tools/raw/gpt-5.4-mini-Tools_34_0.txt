import sys
import math
import random


def dist(a, b):
    return math.hypot(a[0] - b[0], a[1] - b[1])


def route_cost(route, pts, depot):
    if not route:
        return 0.0
    total = dist(depot, pts[route[0]])
    for i in range(len(route) - 1):
        total += dist(pts[route[i]], pts[route[i + 1]])
    total += dist(pts[route[-1]], depot)
    return total


def parse_input():
    data = sys.stdin.read().strip().split()
    if not data:
        return 0, 0, 0, (0.0, 0.0), [], []
    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    cap = float(next(it))
    depot = (float(next(it)), float(next(it)))
    pts = []
    demands = []
    for _ in range(n):
        x = float(next(it))
        y = float(next(it))
        d = float(next(it))
        pts.append((x, y))
        demands.append(d)
    return n, k, cap, depot, pts, demands


def build_initial_routes(n, k, cap, depot, pts, demands):
    order = sorted(range(n), key=lambda i: (math.atan2(pts[i][1] - depot[1], pts[i][0] - depot[0]), dist(depot, pts[i])))
    routes = []
    cur = []
    load = 0.0
    for i in order:
        d = demands[i]
        if cur and load + d > cap:
            routes.append(cur)
            cur = []
            load = 0.0
        cur.append(i)
        load += d
    if cur:
        routes.append(cur)

    while len(routes) > k:
        routes.sort(key=lambda r: route_cost(r, pts, depot), reverse=True)
        a = routes.pop(0)
        best_idx = None
        best_cost = None
        for j, r in enumerate(routes):
            if sum(demands[x] for x in r) + sum(demands[x] for x in a) <= cap:
                c = route_cost(r + a, pts, depot)
                if best_cost is None or c < best_cost:
                    best_cost = c
                    best_idx = j
        if best_idx is None:
            routes.append(a)
            break
        routes[best_idx].extend(a)
    return routes


def improve_routes(routes, cap, depot, pts, demands):
    improved = True
    while improved:
        improved = False
        route_loads = [sum(demands[i] for i in r) for r in routes]
        for i in range(len(routes)):
            r = routes[i]
            if len(r) < 2:
                continue
            best_r = r[:]
            best_cost = route_cost(r, pts, depot)
            for a in range(len(r)):
                for b in range(a + 1, len(r) + 1):
                    cand = r[:a] + r[a:b][::-1] + r[b:]
                    c = route_cost(cand, pts, depot)
                    if c + 1e-12 < best_cost:
                        best_cost = c
                        best_r = cand
                        improved = True
            routes[i] = best_r

        for i in range(len(routes)):
            for j in range(i + 1, len(routes)):
                ri, rj = routes[i], routes[j]
                li, lj = route_loads[i], route_loads[j]
                best_pair = None
                best_delta = 0.0
                for a in range(len(ri)):
                    ci = ri[a]
                    di = demands[ci]
                    for b in range(len(rj)):
                        cj = rj[b]
                        dj = demands[cj]
                        if li - di + dj > cap or lj - dj + di > cap:
                            continue
                        nri = ri[:a] + [cj] + ri[a + 1:]
                        nrj = rj[:b] + [ci] + rj[b + 1:]
                        old = route_cost(ri, pts, depot) + route_cost(rj, pts, depot)
                        new = route_cost(nri, pts, depot) + route_cost(nrj, pts, depot)
                        delta = old - new
                        if delta > best_delta + 1e-12:
                            best_delta = delta
                            best_pair = (nri, nrj)
                if best_pair is not None:
                    routes[i], routes[j] = best_pair
                    improved = True
                    route_loads[i] = sum(demands[x] for x in routes[i])
                    route_loads[j] = sum(demands[x] for x in routes[j])
    return routes


def total_cost(routes, depot, pts):
    return sum(route_cost(r, pts, depot) for r in routes)


def main():
    n, k, cap, depot, pts, demands = parse_input()
    if n == 0:
        return

    random.seed(0)
    routes = build_initial_routes(n, k, cap, depot, pts, demands)
    routes = improve_routes(routes, cap, depot, pts, demands)

    seen = set()
    final_routes = []
    for r in routes:
        nr = [x for x in r if x not in seen]
        for x in nr:
            seen.add(x)
        if nr:
            final_routes.append(nr)

    missing = [i for i in range(n) if i not in seen]
    if missing:
        cur = []
        load = 0.0
        for i in missing:
            d = demands[i]
            if cur and load + d > cap:
                final_routes.append(cur)
                cur = []
                load = 0.0
            cur.append(i)
            load += d
        if cur:
            final_routes.append(cur)

    if len(final_routes) > k:
        all_nodes = [x for r in final_routes for x in r]
        final_routes = []
        cur = []
        load = 0.0
        for i in all_nodes:
            d = demands[i]
            if cur and load + d > cap:
                final_routes.append(cur)
                cur = []
                load = 0.0
            cur.append(i)
            load += d
        if cur:
            final_routes.append(cur)

    while len(final_routes) < k and any(final_routes):
        final_routes.append([])

    print(f"{total_cost(final_routes, depot, pts):.6f}")
    print(len([r for r in final_routes if r]))
    for r in final_routes:
        if r:
            print(" ".join(map(str, r)))


if __name__ == "__main__":
    main()
