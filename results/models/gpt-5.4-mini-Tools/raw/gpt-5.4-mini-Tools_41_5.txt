; SPIR-V
; Version: 1.0
; Generator: Codex
; Bound: 50
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpName %_ ""
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %_ DescriptorSet 0
OpDecorate %_ Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%ptr_uniform_v4 = OpTypePointer Uniform %v4float
%ptr_function_v3 = OpTypePointer Function %v3float
%ptr_function_float = OpTypePointer Function %float
%UBO = OpTypeStruct %v4float %v4float %v4float %v4float %v4float %v4float
%worldPos = OpVariable %ptr_input_v3 Input
%normal = OpVariable %ptr_input_v3 Input
%outColor = OpVariable %ptr_output_v4 Output
%ubo = OpVariable %ptr_uniform_ubo Uniform
%float_0 = OpConstant %float 0
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%uint_32 = OpConstant %uint 32

%main = OpFunction %void None %func
%label = OpLabel
%N = OpVariable %ptr_function_v3 Function
%L = OpVariable %ptr_function_v3 Function
%V = OpVariable %ptr_function_v3 Function
%R = OpVariable %ptr_function_v3 Function
%diffuse = OpVariable %ptr_function_float Function
%specular = OpVariable %ptr_function_float Function
%intensity = OpVariable %ptr_function_float Function
%lightPtr = OpAccessChain %ptr_uniform_v4 %ubo %int_3
%camPtr = OpAccessChain %ptr_uniform_v4 %ubo %int_4
%light4 = OpLoad %v4float %lightPtr
%cam4 = OpLoad %v4float %camPtr
%wp = OpLoad %v3float %worldPos
%nrm = OpLoad %v3float %normal
%normN = OpExtInst %v3float %1 Normalize %nrm
%light3 = OpVectorShuffle %v3float %light4 %light4 0 1 2
%cam3 = OpVectorShuffle %v3float %cam4 %cam4 0 1 2
%toL = OpFSub %v3float %light3 %wp
%toV = OpFSub %v3float %cam3 %wp
%Ln = OpExtInst %v3float %1 Normalize %toL
%Vn = OpExtInst %v3float %1 Normalize %toV
OpStore %N %normN
OpStore %L %Ln
OpStore %V %Vn
%ndotl = OpDot %float %normN %Ln
%diff = OpExtInst %float %1 FMax %ndotl %float_0
OpStore %diffuse %diff
%negL = OpFNegate %v3float %Ln
%twoDot = OpFMul %float (OpDot %float %negL %normN) %float_0_2
%twoDotVec = OpCompositeConstruct %v3float %twoDot %twoDot %twoDot
%Rtmp = OpFSub %v3float %negL %twoDotVec
OpStore %R %Rtmp
%rdotv = OpDot %float %Rtmp %Vn
%rdotv0 = OpExtInst %float %1 FMax %rdotv %float_0
%spec = OpExtInst %float %1 Pow %rdotv0 %uint_32
OpStore %specular %spec
%dterm = OpFMul %float %float_0_5 %diff
%sterm = OpFMul %float %float_0_8 %spec
%sum = OpFAdd %float %float_0_2 %dterm
%sum2 = OpFAdd %float %sum %sterm
OpStore %intensity %sum2
%gray = OpLoad %float %intensity
%color = OpCompositeConstruct %v4float %gray %gray %gray %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
