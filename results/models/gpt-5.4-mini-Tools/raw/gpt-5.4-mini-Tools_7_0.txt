static constexpr double EPS = 1e-10;

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
    Vec3 operator+(const Vec3& o) const { return Vec3(x + o.x, y + o.y, z + o.z); }
    Vec3 operator-(const Vec3& o) const { return Vec3(x - o.x, y - o.y, z - o.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
    Vec3 operator/(double s) const { return Vec3(x / s, y / s, z / s); }
};

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return Vec3(
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    );
}

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline double norm2(const Vec3& a) {
    return dot(a, a);
}

static inline Vec3 normalize(const Vec3& a) {
    double n = std::sqrt(norm2(a));
    if (n < EPS) return Vec3(0, 0, 0);
    return a / n;
}

struct Tri {
    array<int, 3> v;
};

struct AABB {
    Vec3 mn, mx;
    void expand(const Vec3& p) {
        mn.x = std::min(mn.x, p.x); mn.y = std::min(mn.y, p.y); mn.z = std::min(mn.z, p.z);
        mx.x = std::max(mx.x, p.x); mx.y = std::max(mx.y, p.y); mx.z = std::max(mx.z, p.z);
    }
};

static inline AABB tri_aabb(const vector<Vec3>& pts, const Tri& t) {
    AABB b;
    b.mn = b.mx = pts[t.v[0]];
    b.expand(pts[t.v[1]]);
    b.expand(pts[t.v[2]]);
    return b;
}

static inline bool aabb_overlap(const AABB& a, const AABB& b) {
    return !(a.mx.x < b.mn.x - EPS || b.mx.x < a.mn.x - EPS ||
             a.mx.y < b.mn.y - EPS || b.mx.y < a.mn.y - EPS ||
             a.mx.z < b.mn.z - EPS || b.mx.z < a.mn.z - EPS);
}

struct Plane {
    Vec3 n;
    double d;
};

static inline Plane plane_from_triangle(const Vec3& a, const Vec3& b, const Vec3& c) {
    Plane p;
    p.n = cross(b - a, c - a);
    p.d = -dot(p.n, a);
    return p;
}

static inline double signed_distance(const Plane& p, const Vec3& q) {
    return dot(p.n, q) + p.d;
}

static bool point_in_triangle(const Vec3& p, const Vec3& a, const Vec3& b, const Vec3& c) {
    Vec3 n = cross(b - a, c - a);
    double area2 = norm2(n);
    if (area2 < EPS) return false;
    Vec3 c0 = cross(b - a, p - a);
    Vec3 c1 = cross(c - b, p - b);
    Vec3 c2 = cross(a - c, p - c);
    if (dot(n, c0) < -EPS) return false;
    if (dot(n, c1) < -EPS) return false;
    if (dot(n, c2) < -EPS) return false;
    return true;
}

static bool segment_triangle_intersection(const Vec3& p0, const Vec3& p1,
                                          const Vec3& a, const Vec3& b, const Vec3& c,
                                          double& t_out) {
    Plane pl = plane_from_triangle(a, b, c);
    Vec3 dir = p1 - p0;
    double denom = dot(pl.n, dir);
    double s0 = signed_distance(pl, p0);
    double s1 = signed_distance(pl, p1);
    if (std::abs(denom) < EPS) {
        if (std::abs(s0) < EPS && std::abs(s1) < EPS) return false;
        return false;
    }
    double t = -s0 / denom;
    if (t < -EPS || t > 1.0 + EPS) return false;
    Vec3 p = p0 + dir * t;
    if (!point_in_triangle(p, a, b, c)) return false;
    t_out = t;
    return true;
}

static bool ray_intersects_mesh(const Vec3& origin, const Vec3& dir,
                                const vector<Vec3>& pts, const vector<Tri>& tris) {
    int hits = 0;
    for (const auto& t : tris) {
        const Vec3& a = pts[t.v[0]];
        const Vec3& b = pts[t.v[1]];
        const Vec3& c = pts[t.v[2]];
        Plane pl = plane_from_triangle(a, b, c);
        double denom = dot(pl.n, dir);
        if (std::abs(denom) < EPS) continue;
        double tt = -(dot(pl.n, origin) + pl.d) / denom;
        if (tt < EPS) continue;
        Vec3 p = origin + dir * tt;
        if (point_in_triangle(p, a, b, c)) {
            if (denom < 0) ++hits;
        }
    }
    return (hits & 1) != 0;
}

struct BBoxGrid {
    Vec3 mn, mx, scale;
    int nx, ny, nz;
    vector<vector<int>> cells;
    vector<AABB> triBoxes;

    BBoxGrid(const vector<Vec3>& pts, const vector<Tri>& tris) {
        mn = Vec3(std::numeric_limits<double>::infinity(),
                  std::numeric_limits<double>::infinity(),
                  std::numeric_limits<double>::infinity());
        mx = Vec3(-std::numeric_limits<double>::infinity(),
                  -std::numeric_limits<double>::infinity(),
                  -std::numeric_limits<double>::infinity());
        triBoxes.reserve(tris.size());
        for (const auto& t : tris) {
            AABB b = tri_aabb(pts, t);
            triBoxes.push_back(b);
            mn.x = std::min(mn.x, b.mn.x); mn.y = std::min(mn.y, b.mn.y); mn.z = std::min(mn.z, b.mn.z);
            mx.x = std::max(mx.x, b.mx.x); mx.y = std::max(mx.y, b.mx.y); mx.z = std::max(mx.z, b.mx.z);
        }
        double diag = std::max({mx.x - mn.x, mx.y - mn.y, mx.z - mn.z, 1.0});
        int target = (int)std::max(1.0, std::sqrt((double)tris.size()));
        nx = ny = nz = std::max(1, target);
        scale = Vec3((nx - 1) / diag, (ny - 1) / diag, (nz - 1) / diag);
        cells.resize((size_t)nx * ny * nz);
        for (int i = 0; i < (int)tris.size(); ++i) {
            const AABB& b = triBoxes[i];
            int x0 = clampi((int)((b.mn.x - mn.x) * scale.x), 0, nx - 1);
            int y0 = clampi((int)((b.mn.y - mn.y) * scale.y), 0, ny - 1);
            int z0 = clampi((int)((b.mn.z - mn.z) * scale.z), 0, nz - 1);
            int x1 = clampi((int)((b.mx.x - mn.x) * scale.x), 0, nx - 1);
            int y1 = clampi((int)((b.mx.y - mn.y) * scale.y), 0, ny - 1);
            int z1 = clampi((int)((b.mx.z - mn.z) * scale.z), 0, nz - 1);
            for (int x = x0; x <= x1; ++x)
                for (int y = y0; y <= y1; ++y)
                    for (int z = z0; z <= z1; ++z)
                        cells[(size_t)((x * ny + y) * nz + z)].push_back(i);
        }
    }

    static int clampi(int v, int lo, int hi) {
        return std::max(lo, std::min(hi, v));
    }
};

static bool any_triangle_overlap(const vector<Vec3>& A, const vector<Tri>& TA,
                                 const vector<Vec3>& B, const vector<Tri>& TB) {
    BBoxGrid gA(A, TA);
    BBoxGrid gB(B, TB);
    for (size_t i = 0; i < TA.size(); ++i) {
        const AABB& ba = gA.triBoxes[i];
        int x0 = BBoxGrid::clampi((int)((ba.mn.x - gB.mn.x) * gB.scale.x), 0, gB.nx - 1);
        int y0 = BBoxGrid::clampi((int)((ba.mn.y - gB.mn.y) * gB.scale.y), 0, gB.ny - 1);
        int z0 = BBoxGrid::clampi((int)((ba.mn.z - gB.mn.z) * gB.scale.z), 0, gB.nz - 1);
        int x1 = BBoxGrid::clampi((int)((ba.mx.x - gB.mn.x) * gB.scale.x), 0, gB.nx - 1);
        int y1 = BBoxGrid::clampi((int)((ba.mx.y - gB.mn.y) * gB.scale.y), 0, gB.ny - 1);
        int z1 = BBoxGrid::clampi((int)((ba.mx.z - gB.mn.z) * gB.scale.z), 0, gB.nz - 1);
        for (int x = x0; x <= x1; ++x)
            for (int y = y0; y <= y1; ++y)
                for (int z = z0; z <= z1; ++z) {
                    const auto& cell = gB.cells[(size_t)((x * gB.ny + y) * gB.nz + z)];
                    for (int j : cell) if (aabb_overlap(ba, gB.triBoxes[j])) return true;
                }
    }
    return false;
}

static Vec3 tri_centroid(const Vec3& a, const Vec3& b, const Vec3& c) {
    return (a + b + c) / 3.0;
}

static bool inside_mesh(const Vec3& p, const vector<Vec3>& pts, const vector<Tri>& tris) {
    Vec3 dir(1.0, 0.123456789, 0.246813579);
    return ray_intersects_mesh(p, dir, pts, tris);
}

static int64_t quant_key(double v) {
    return (int64_t)std::llround(v * 1e9);
}

struct Key3 {
    int64_t x, y, z;
    bool operator==(const Key3& o) const { return x == o.x && y == o.y && z == o.z; }
};

struct Key3Hash {
    size_t operator()(const Key3& k) const noexcept {
        size_t h1 = std::hash<int64_t>{}(k.x);
        size_t h2 = std::hash<int64_t>{}(k.y);
        size_t h3 = std::hash<int64_t>{}(k.z);
        return h1 ^ (h2 << 1) ^ (h3 << 2);
    }
};

static int add_vertex(vector<Vec3>& verts, std::unordered_map<Key3, int, Key3Hash>& mp, const Vec3& p) {
    Key3 k{quant_key(p.x), quant_key(p.y), quant_key(p.z)};
    auto it = mp.find(k);
    if (it != mp.end()) return it->second;
    int id = (int)verts.size();
    verts.push_back(p);
    mp.emplace(k, id);
    return id;
}

static void append_tri(vector<array<int,3>>& faces, int a, int b, int c) {
    if (a == b || b == c || a == c) return;
    faces.push_back({a, b, c});
}

static vector<array<int,3>> copy_faces_with_offset(const vector<Tri>& tris, int offset) {
    vector<array<int,3>> out;
    out.reserve(tris.size());
    for (const auto& t : tris) out.push_back({t.v[0] + offset, t.v[1] + offset, t.v[2] + offset});
    return out;
}

int main() {
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(cin >> VA >> FA >> VB >> FB)) return 0;

    vector<Vec3> A(VA), B(VB);
    vector<Tri> TA(FA), TB(FB);
    for (int i = 0; i < VA; ++i) cin >> A[i].x >> A[i].y >> A[i].z;
    for (int i = 0; i < FA; ++i) cin >> TA[i].v[0] >> TA[i].v[1] >> TA[i].v[2];
    for (int i = 0; i < VB; ++i) cin >> B[i].x >> B[i].y >> B[i].z;
    for (int i = 0; i < FB; ++i) cin >> TB[i].v[0] >> TB[i].v[1] >> TB[i].v[2];

    bool overlap = any_triangle_overlap(A, TA, B, TB);

    vector<Vec3> outVerts;
    vector<array<int,3>> outFaces;
    std::unordered_map<Key3, int, Key3Hash> vmap;
    vmap.reserve((size_t)(VA + VB) * 2 + 16);

    if (!overlap) {
        for (const auto& p : A) add_vertex(outVerts, vmap, p);
        for (const auto& p : B) add_vertex(outVerts, vmap, p);
        outFaces = copy_faces_with_offset(TA, 0);
        int offset = (int)A.size();
        auto fb = copy_faces_with_offset(TB, offset);
        outFaces.insert(outFaces.end(), fb.begin(), fb.end());
        cout << outVerts.size() << ' ' << outFaces.size() << '\n';
        for (const auto& p : outVerts) cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
        for (const auto& f : outFaces) cout << f[0] << ' ' << f[1] << ' ' << f[2] << '\n';
        return 0;
    }

    vector<char> keepA(FA, 1), keepB(FB, 1);
    for (int i = 0; i < FA; ++i) {
        Vec3 ca = tri_centroid(A[TA[i].v[0]], A[TA[i].v[1]], A[TA[i].v[2]]);
        if (inside_mesh(ca, B, TB)) keepA[i] = 0;
    }
    for (int i = 0; i < FB; ++i) {
        Vec3 cb = tri_centroid(B[TB[i].v[0]], B[TB[i].v[1]], B[TB[i].v[2]]);
        if (inside_mesh(cb, A, TA)) keepB[i] = 0;
    }

    for (int i = 0; i < VA; ++i) add_vertex(outVerts, vmap, A[i]);
    int offB = (int)A.size();
    for (int i = 0; i < VB; ++i) add_vertex(outVerts, vmap, B[i]);

    for (int i = 0; i < FA; ++i) if (keepA[i]) {
        append_tri(outFaces, TA[i].v[0], TA[i].v[1], TA[i].v[2]);
    }
    for (int i = 0; i < FB; ++i) if (keepB[i]) {
        append_tri(outFaces, TB[i].v[0] + offB, TB[i].v[1] + offB, TB[i].v[2] + offB);
    }

    cout << outVerts.size() << ' ' << outFaces.size() << '\n';
    for (const auto& p : outVerts) cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    for (const auto& f : outFaces) cout << f[0] << ' ' << f[1] << ' ' << f[2] << '\n';
    return 0;
}
