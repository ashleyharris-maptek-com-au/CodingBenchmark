const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);
const tasks = Array.isArray(data.tasks) ? data.tasks : [];
const n = tasks.length;

if (n === 0) {
  process.stdout.write(JSON.stringify({ order: [] }));
  process.exit(0);
}

const idToIndex = new Map();
for (let i = 0; i < n; i++) idToIndex.set(tasks[i].id, i);

const indeg = new Int32Array(n);
const outgoing = Array.from({ length: n }, () => []);
const duration = new Float64Array(n);
const deadline = new Float64Array(n);
const weight = new Float64Array(n);
const depCount = new Int32Array(n);

for (let i = 0; i < n; i++) {
  const t = tasks[i];
  duration[i] = Number.isFinite(t.duration) ? t.duration : 0;
  deadline[i] = Number.isFinite(t.deadline) ? t.deadline : Number.POSITIVE_INFINITY;
  weight[i] = Number.isFinite(t.weight) ? t.weight : 1;
  const deps = Array.isArray(t.deps) ? t.deps : [];
  depCount[i] = deps.length;
  indeg[i] = deps.length;
  for (let k = 0; k < deps.length; k++) {
    const j = idToIndex.get(deps[k]);
    if (j !== undefined) outgoing[j].push(i);
  }
}

// Compute a downstream criticality score: task duration plus longest path of successors.
const topo = [];
const q = [];
for (let i = 0; i < n; i++) if (indeg[i] === 0) q.push(i);
for (let head = 0; head < q.length; head++) {
  const u = q[head];
  topo.push(u);
  for (const v of outgoing[u]) {
    indeg[v]--;
    if (indeg[v] === 0) q.push(v);
  }
}

// If malformed cyclic input appears, fall back to original order.
if (topo.length !== n) {
  process.stdout.write(JSON.stringify({ order: tasks.map(t => t.id) }));
  process.exit(0);
}

const crit = new Float64Array(n);
for (let idx = n - 1; idx >= 0; idx--) {
  const u = topo[idx];
  let bestSucc = 0;
  for (const v of outgoing[u]) {
    if (crit[v] > bestSucc) bestSucc = crit[v];
  }
  crit[u] = duration[u] + bestSucc;
}

// Rank tasks by a blend of urgency, critical path, and weight.
const scored = new Array(n);
for (let i = 0; i < n; i++) {
  const d = duration[i];
  const dd = deadline[i];
  const w = weight[i];
  const slack = Number.isFinite(dd) ? dd - crit[i] : 1e18;
  const urgency = Number.isFinite(dd) ? (d / Math.max(1, dd)) : 0;
  const score = (w * 1000) + (crit[i] * 10) + (urgency * 100) - Math.max(0, slack) * 0.001;
  scored[i] = { idx: i, score, deadline: dd, crit: crit[i], weight: w, duration: d };
}

scored.sort((a, b) => {
  if (b.score !== a.score) return b.score - a.score;
  if (a.deadline !== b.deadline) return a.deadline - b.deadline;
  if (b.crit !== a.crit) return b.crit - a.crit;
  if (b.weight !== a.weight) return b.weight - a.weight;
  if (a.duration !== b.duration) return a.duration - b.duration;
  return a.idx - b.idx;
});

const order = scored.map(x => tasks[x.idx].id);
process.stdout.write(JSON.stringify({ order }));
