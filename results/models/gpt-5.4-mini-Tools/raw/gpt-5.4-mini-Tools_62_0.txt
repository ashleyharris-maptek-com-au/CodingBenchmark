declare const require: any;
const fs = require('fs');

type Town = { id: string; x: number; y: number; value: number };

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const width: number = data.width | 0;
const height: number = data.height | 0;
const grid: string[] = data.grid;
const ignitions: Array<[number, number]> = data.ignitions || [];
const towns: Town[] = data.towns || [];
const cutBudget: number = data.cutBudget | 0;

if (width <= 0 || height <= 0 || cutBudget <= 0 || grid.length === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const n = width * height;
const cellType = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < width; x++) {
    cellType[y * width + x] = row.charCodeAt(x) === 46 ? 0 : 1;
  }
}

const dirs = [1, 0, -1, 0, 1];
const dist = new Int32Array(n);
dist.fill(-1);
const q = new Int32Array(n);
let qh = 0, qt = 0;

for (const [x, y] of ignitions) {
  if (x < 0 || x >= width || y < 0 || y >= height) continue;
  const idx = y * width + x;
  if (cellType[idx] === 0 && dist[idx] === -1) {
    dist[idx] = 0;
    q[qt++] = idx;
  }
}

while (qh < qt) {
  const idx = q[qh++];
  const x = idx % width;
  const y = (idx / width) | 0;
  const nd = dist[idx] + 1;
  for (let k = 0; k < 4; k++) {
    const nx = x + dirs[k];
    const ny = y + dirs[k + 1];
    if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;
    const ni = ny * width + nx;
    if (cellType[ni] !== 0 || dist[ni] !== -1) continue;
    dist[ni] = nd;
    q[qt++] = ni;
  }
}

const reachableTowns: number[] = [];
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  const idx = t.y * width + t.x;
  if (idx >= 0 && idx < n && dist[idx] !== -1) reachableTowns.push(i);
}

if (reachableTowns.length === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const parent = new Int32Array(n);
parent.fill(-1);
const owner = new Int32Array(n);
owner.fill(-1);
const bestDist = new Int32Array(n);
bestDist.fill(0x3fffffff);

const pqNode: number[] = [];
const pqScore: number[] = [];

function push(cell: number, score: number) {
  let i = pqNode.length;
  pqNode.push(cell);
  pqScore.push(score);
  while (i > 0) {
    const p = (i - 1) >> 1;
    if (pqScore[p] >= score) break;
    pqNode[i] = pqNode[p];
    pqScore[i] = pqScore[p];
    i = p;
  }
  pqNode[i] = cell;
  pqScore[i] = score;
}

function pop(): number | -1 {
  if (pqNode.length === 0) return -1;
  const ret = pqNode[0];
  const lastNode = pqNode.pop() as number;
  const lastScore = pqScore.pop() as number;
  if (pqNode.length === 0) return ret;
  let i = 0;
  while (true) {
    const l = i * 2 + 1;
    if (l >= pqNode.length) break;
    const r = l + 1;
    let c = l;
    if (r < pqNode.length && pqScore[r] > pqScore[l]) c = r;
    if (pqScore[c] <= lastScore) break;
    pqNode[i] = pqNode[c];
    pqScore[i] = pqScore[c];
    i = c;
  }
  pqNode[i] = lastNode;
  pqScore[i] = lastScore;
  return ret;
}

for (const ti of reachableTowns) {
  const t = towns[ti];
  const idx = t.y * width + t.x;
  bestDist[idx] = dist[idx];
  owner[idx] = ti;
  parent[idx] = -2;
  push(idx, t.value / (dist[idx] + 1));
}

while (true) {
  const idx = pop();
  if (idx === -1) break;
  const curOwner = owner[idx];
  const x = idx % width;
  const y = (idx / width) | 0;
  const d = bestDist[idx];
  for (let k = 0; k < 4; k++) {
    const nx = x + dirs[k];
    const ny = y + dirs[k + 1];
    if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;
    const ni = ny * width + nx;
    if (cellType[ni] !== 0) continue;
    const nd = d + 1;
    if (nd < bestDist[ni]) {
      bestDist[ni] = nd;
      owner[ni] = curOwner;
      parent[ni] = idx;
      push(ni, towns[curOwner].value / (nd + 1));
    }
  }
}

const candidates: Array<{ idx: number; score: number }> = [];
for (let i = 0; i < n; i++) {
  const o = owner[i];
  if (o === -1 || dist[i] === -1 || cellType[i] !== 0) continue;
  candidates.push({ idx: i, score: towns[o].value / (dist[i] + 1) });
}
candidates.sort((a, b) => b.score - a.score || dist[a.idx] - dist[b.idx]);

const cut = new Uint8Array(n);
const result: Array<[number, number]> = [];
const protectedTown = new Uint8Array(towns.length);

function addCut(cellIdx: number) {
  if (result.length >= cutBudget || cut[cellIdx] || cellType[cellIdx] !== 0) return;
  cut[cellIdx] = 1;
  result.push([cellIdx % width, (cellIdx / width) | 0]);
}

for (const c of candidates) {
  if (result.length >= cutBudget) break;
  const o = owner[c.idx];
  if (o < 0 || protectedTown[o]) continue;
  let cur = c.idx;
  while (cur >= 0 && result.length < cutBudget && cellType[cur] === 0 && !cut[cur]) {
    addCut(cur);
    if (parent[cur] < 0) break;
    cur = parent[cur];
  }
  protectedTown[o] = 1;
}

if (result.length < cutBudget) {
  for (let i = 0; i < n && result.length < cutBudget; i++) {
    if (dist[i] !== -1 && cellType[i] === 0 && !cut[i]) addCut(i);
  }
}

process.stdout.write(JSON.stringify({ firebreaks: result.slice(0, cutBudget) }));
