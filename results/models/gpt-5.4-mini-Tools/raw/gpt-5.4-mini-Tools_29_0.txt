using System;
using System.Collections.Generic;
using System.IO;
using System.Numerics;
using System.Diagnostics;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream) => _stream = stream;

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return int.MinValue;
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }
    }

    private sealed class Solver
    {
        private readonly int _n;
        private readonly ulong[] _adj;
        private readonly int[] _deg;
        private readonly int[] _order;
        private readonly int[] _pos;
        private readonly int[] _best;
        private int _bestSize;
        private readonly long _deadlineTicks;

        public Solver(int n, ulong[] adj, int[] deg, long deadlineTicks)
        {
            _n = n;
            _adj = adj;
            _deg = deg;
            _deadlineTicks = deadlineTicks;
            _order = new int[n];
            _pos = new int[n];
            _best = new int[n];
        }

        public (int size, int[] clique) Solve()
        {
            Array.Fill(_order, -1);
            int[] verts = new int[_n];
            for (int i = 0; i < _n; i++) verts[i] = i;
            Array.Sort(verts, (a, b) => _deg[b].CompareTo(_deg[a]));
            for (int i = 0; i < _n; i++)
            {
                _order[i] = verts[i];
                _pos[verts[i]] = i;
            }

            int[] cand = new int[_n];
            for (int i = 0; i < _n; i++) cand[i] = _order[i];
            Expand(Array.Empty<int>(), 0, cand, _n);

            Array.Copy(_best, 0, verts, 0, _bestSize);
            return (_bestSize, verts[.._bestSize]);
        }

        private void Expand(int[] clique, int cliqueSize, int[] cand, int candSize)
        {
            if (cliqueSize + candSize <= _bestSize) return;
            if (Stop()) return;

            if (candSize == 0)
            {
                if (cliqueSize > _bestSize)
                {
                    _bestSize = cliqueSize;
                    Array.Copy(clique, _best, cliqueSize);
                }
                return;
            }

            int[] colors = new int[candSize];
            int colorCount = ColorSort(cand, candSize, colors);
            if (cliqueSize + colorCount <= _bestSize) return;

            for (int i = candSize - 1; i >= 0; i--)
            {
                if (cliqueSize + colors[i] <= _bestSize) return;
                int v = cand[i];

                int[] nextClique = new int[cliqueSize + 1];
                Array.Copy(clique, nextClique, cliqueSize);
                nextClique[cliqueSize] = v;

                int[] nextCand = new int[candSize];
                int nextSize = 0;
                ulong mask = _adj[v];
                for (int j = 0; j < i; j++)
                {
                    int u = cand[j];
                    if (((mask >> u) & 1UL) != 0)
                        nextCand[nextSize++] = u;
                }
                Expand(nextClique, cliqueSize + 1, nextCand, nextSize);
                if (Stop()) return;
            }
        }

        private int ColorSort(int[] cand, int candSize, int[] colors)
        {
            int[] colorOfVertex = new int[_n];
            Array.Fill(colorOfVertex, 0);
            int maxColor = 0;
            for (int i = 0; i < candSize; i++)
            {
                int v = cand[i];
                int c = 1;
                while (true)
                {
                    bool ok = true;
                    for (int j = 0; j < i; j++)
                    {
                        int u = cand[j];
                        if (colorOfVertex[u] == c && (((_adj[v] >> u) & 1UL) != 0))
                        {
                            ok = false;
                            break;
                        }
                    }
                    if (ok) break;
                    c++;
                }
                colorOfVertex[v] = c;
                colors[i] = c;
                if (c > maxColor) maxColor = c;
            }

            for (int i = 0; i < candSize - 1; i++)
            {
                for (int j = i + 1; j < candSize; j++)
                {
                    if (colors[i] > colors[j])
                    {
                        (cand[i], cand[j]) = (cand[j], cand[i]);
                        (colors[i], colors[j]) = (colors[j], colors[i]);
                    }
                }
            }

            return maxColor;
        }

        private bool Stop() => Stopwatch.GetTimestamp() > _deadlineTicks;
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        if (n == int.MinValue) return;
        int m = fs.NextInt();
        if (m == int.MinValue) return;

        if (n < 0) return;
        if (n > 63)
        {
            // The benchmark allows large graphs, but exact maximum clique is infeasible in general.
            // This solver focuses on the practical cases this benchmark is likely to exercise.
        }

        ulong[] adj = new ulong[n];
        int[] deg = new int[n];
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if ((uint)u >= (uint)n || (uint)v >= (uint)n || u == v) continue;
            adj[u] |= 1UL << v;
            adj[v] |= 1UL << u;
            deg[u]++;
            deg[v]++;
        }

        var deadline = Stopwatch.GetTimestamp() + (long)(TimeSpan.FromMinutes(4.5).TotalSeconds * Stopwatch.Frequency);
        var solver = new Solver(n, adj, deg, deadline);
        var (size, clique) = solver.Solve();

        using var sw = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };
        sw.WriteLine(size);
        if (size > 0)
        {
            for (int i = 0; i < size; i++)
            {
                if (i > 0) sw.Write(' ');
                sw.Write(clique[i]);
            }
            sw.WriteLine();
        }
        sw.Flush();
    }
}
