using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public class Program
{
    struct Box
    {
        public int W, H, D, Id;
        public int Volume;
    }

    struct Placement
    {
        public int Id, X, Y, Z;
        public Placement(int id, int x, int y, int z)
        {
            Id = id; X = x; Y = y; Z = z;
        }
    }

    sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        private byte ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do c = ReadByte(); while (c <= 32 && c != 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return val * sign;
        }
    }

    static int[] Orientations(Box b)
    {
        int[] dims = { b.W, b.H, b.D };
        Array.Sort(dims);
        return dims;
    }

    static void Permute(int[] a, int l, List<int[]> res)
    {
        if (l == a.Length)
        {
            res.Add((int[])a.Clone());
            return;
        }
        for (int i = l; i < a.Length; i++)
        {
            (a[l], a[i]) = (a[i], a[l]);
            Permute(a, l + 1, res);
            (a[l], a[i]) = (a[i], a[l]);
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int W = fs.NextInt();
        int H = fs.NextInt();
        int D = fs.NextInt();

        var boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            int w = fs.NextInt();
            int h = fs.NextInt();
            int d = fs.NextInt();
            boxes[i] = new Box { W = w, H = h, D = d, Id = i, Volume = w * h * d };
        }

        Array.Sort(boxes, (a, b) =>
        {
            if (a.Volume != b.Volume) return b.Volume.CompareTo(a.Volume);
            if (a.W != b.W) return a.W.CompareTo(b.W);
            if (a.H != b.H) return a.H.CompareTo(b.H);
            return a.D.CompareTo(b.D);
        });

        // Heuristic: choose a repeated cell size from a seed box orientation,
        // then fill the container on a regular grid with any boxes that fit.
        int bestCount = 0;
        List<Placement> best = new List<Placement>();

        int seedLimit = Math.Min(n, 50);
        for (int seed = 0; seed < seedLimit; seed++)
        {
            var b0 = boxes[seed];
            var perms = new List<int[]>();
            Permute(new[] { b0.W, b0.H, b0.D }, 0, perms);
            foreach (var p in perms)
            {
                int sx = p[0], sy = p[1], sz = p[2];
                if (sx <= 0 || sy <= 0 || sz <= 0) continue;
                int nx = W / sx;
                int ny = H / sy;
                int nz = D / sz;
                if (nx == 0 || ny == 0 || nz == 0) continue;

                var used = new bool[n];
                var cur = new List<Placement>();

                for (int ix = 0; ix < nx; ix++)
                {
                    int x = ix * sx;
                    for (int iy = 0; iy < ny; iy++)
                    {
                        int y = iy * sy;
                        for (int iz = 0; iz < nz; iz++)
                        {
                            int z = iz * sz;
                            for (int i = 0; i < n; i++)
                            {
                                if (used[i]) continue;
                                var b = boxes[i];
                                bool ok =
                                    (b.W <= sx && b.H <= sy && b.D <= sz) ||
                                    (b.W <= sx && b.D <= sy && b.H <= sz) ||
                                    (b.H <= sx && b.W <= sy && b.D <= sz) ||
                                    (b.H <= sx && b.D <= sy && b.W <= sz) ||
                                    (b.D <= sx && b.W <= sy && b.H <= sz) ||
                                    (b.D <= sx && b.H <= sy && b.W <= sz);
                                if (!ok) continue;
                                used[i] = true;
                                cur.Add(new Placement(b.Id, x, y, z));
                                break;
                            }
                        }
                    }
                }

                if (cur.Count > bestCount)
                {
                    bestCount = cur.Count;
                    best = cur;
                }
            }
        }

        if (bestCount == 0)
        {
            var outList = new List<Placement>();
            int x = 0, y = 0, z = 0;
            int rowH = 0, layerD = 0;
            for (int i = 0; i < n; i++)
            {
                var b = boxes[i];
                int[] o = Orientations(b);
                int pw = o[0], ph = o[1], pd = o[2];
                if (x + pw > W)
                {
                    x = 0;
                    y += rowH;
                    rowH = 0;
                }
                if (y + ph > H)
                {
                    y = 0;
                    x = 0;
                    z += layerD;
                    layerD = 0;
                }
                if (z + pd > D) continue;
                outList.Add(new Placement(b.Id, x, y, z));
                x += pw;
                rowH = Math.Max(rowH, ph);
                layerD = Math.Max(layerD, pd);
            }
            best = outList;
        }

        var sb = new StringBuilder();
        sb.AppendLine(best.Count.ToString());
        for (int i = 0; i < best.Count; i++)
        {
            var p = best[i];
            sb.Append(p.Id).Append(' ').Append(p.X).Append(' ').Append(p.Y).Append(' ').Append(p.Z).AppendLine();
        }
        Console.Write(sb.ToString());
    }
}
