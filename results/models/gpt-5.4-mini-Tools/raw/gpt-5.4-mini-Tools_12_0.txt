using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;
using System.Text;

public static class Program
{
    private struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    }

    private static readonly CultureInfo CI = CultureInfo.InvariantCulture;

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int V = fs.NextInt();
        int F = fs.NextInt();
        var verts = new Vec3[V];
        for (int i = 0; i < V; i++)
            verts[i] = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
        for (int i = 0; i < F; i++)
        {
            fs.NextInt(); fs.NextInt(); fs.NextInt();
        }
        double W = fs.NextDouble();
        double H = fs.NextDouble();
        double D = fs.NextDouble();

        double minX = double.PositiveInfinity, minY = double.PositiveInfinity, minZ = double.PositiveInfinity;
        double maxX = double.NegativeInfinity, maxY = double.NegativeInfinity, maxZ = double.NegativeInfinity;
        for (int i = 0; i < V; i++)
        {
            var p = verts[i];
            if (p.X < minX) minX = p.X; if (p.X > maxX) maxX = p.X;
            if (p.Y < minY) minY = p.Y; if (p.Y > maxY) maxY = p.Y;
            if (p.Z < minZ) minZ = p.Z; if (p.Z > maxZ) maxZ = p.Z;
        }

        double sx = maxX - minX, sy = maxY - minY, sz = maxZ - minZ;
        if (V == 0 || sx <= 0 || sy <= 0 || sz <= 0 || W <= 0 || H <= 0 || D <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        bool large = (W * H * D >= 200000 || Math.Max(W, Math.Max(H, D)) >= 50 || V > 50 || F > 100);
        bool extreme = (W * H * D >= 2000000 || Math.Max(W, Math.Max(H, D)) >= 200 || V > 200 || F > 400);

        var orientations = GenerateOrientations(verts);
        orientations = DeduplicateOrientations(orientations);
        if (orientations.Count == 0) orientations.Add(new BoxOrient(sx, sy, sz, 1, 0, 0, 0, minX, minY, minZ));

        if (extreme)
            orientations = SelectBestOrientations(orientations, 2);
        else if (large)
            orientations = SelectBestOrientations(orientations, 4);
        else
            orientations = SelectBestOrientations(orientations, Math.Min(orientations.Count, 12));

        BoxOrient best = orientations[0];
        double bestCount = -1;
        foreach (var o in orientations)
        {
            long cx = (long)Math.Floor(W / o.W);
            long cy = (long)Math.Floor(H / o.H);
            long cz = (long)Math.Floor(D / o.D);
            double score = (double)(cx * cy * cz);
            if (score > bestCount)
            {
                bestCount = score;
                best = o;
            }
        }

        long nx = (long)Math.Floor(W / best.W);
        long ny = (long)Math.Floor(H / best.H);
        long nz = (long)Math.Floor(D / best.D);
        long count = nx * ny * nz;
        if (count < 0) count = 0;

        var sb = new StringBuilder();
        sb.AppendLine(count.ToString(CI));
        for (long ix = 0; ix < nx; ix++)
        for (long iy = 0; iy < ny; iy++)
        for (long iz = 0; iz < nz; iz++)
        {
            double tx = ix * best.W - best.MinX;
            double ty = iy * best.H - best.MinY;
            double tz = iz * best.D - best.MinZ;
            sb.Append(tx.ToString("0.######", CI)).Append(' ')
              .Append(ty.ToString("0.######", CI)).Append(' ')
              .Append(tz.ToString("0.######", CI)).Append(' ')
              .Append(best.Qw.ToString("0.######", CI)).Append(' ')
              .Append(best.Qx.ToString("0.######", CI)).Append(' ')
              .Append(best.Qy.ToString("0.######", CI)).Append(' ')
              .Append(best.Qz.ToString("0.######", CI)).AppendLine();
        }

        Console.Write(sb.ToString());
    }

    private static List<BoxOrient> GenerateOrientations(Vec3[] verts)
    {
        var result = new List<BoxOrient>();
        var quats = new (double w, double x, double y, double z)[]
        {
            (1,0,0,0),
            (0.7071067811865476,0,0,0.7071067811865476),
            (0.7071067811865476,0.7071067811865476,0,0),
            (0.7071067811865476,0,0.7071067811865476,0),
            (0.5,0.5,0.5,0.5),
            (0.5,-0.5,0.5,0.5),
            (0.5,0.5,-0.5,0.5),
            (0.5,0.5,0.5,-0.5),
            (0.0,0.0,0.7071067811865476,0.7071067811865476),
            (0.0,0.7071067811865476,0.0,0.7071067811865476),
            (0.0,0.7071067811865476,0.7071067811865476,0.0),
            (0.9238795325,0.3826834324,0,0),
            (0.9238795325,0,0.3826834324,0),
            (0.9238795325,0,0,0.3826834324)
        };
        foreach (var q in quats)
        {
            var bounds = BoundsAfterRotation(verts, q.w, q.x, q.y, q.z);
            result.Add(new BoxOrient(bounds.maxX - bounds.minX, bounds.maxY - bounds.minY, bounds.maxZ - bounds.minZ, q.w, q.x, q.y, q.z, bounds.minX, bounds.minY, bounds.minZ));
        }
        return result;
    }

    private static List<BoxOrient> DeduplicateOrientations(List<BoxOrient> input)
    {
        var dict = new Dictionary<string, BoxOrient>();
        foreach (var o in input)
        {
            string key = Math.Round(o.W, 6).ToString(CI) + "," + Math.Round(o.H, 6).ToString(CI) + "," + Math.Round(o.D, 6).ToString(CI);
            if (!dict.ContainsKey(key) || o.W * o.H * o.D < dict[key].W * dict[key].H * dict[key].D)
                dict[key] = o;
        }
        return dict.Values.OrderBy(o => o.W * o.H * o.D).ToList();
    }

    private static List<BoxOrient> SelectBestOrientations(List<BoxOrient> input, int k)
    {
        return input.OrderByDescending(o => 1.0 / (o.W * o.H * o.D)).Take(k).ToList();
    }

    private static (double minX, double minY, double minZ, double maxX, double maxY, double maxZ) BoundsAfterRotation(Vec3[] verts, double qw, double qx, double qy, double qz)
    {
        double minX = double.PositiveInfinity, minY = double.PositiveInfinity, minZ = double.PositiveInfinity;
        double maxX = double.NegativeInfinity, maxY = double.NegativeInfinity, maxZ = double.NegativeInfinity;
        foreach (var p in verts)
        {
            var r = Rotate(p, qw, qx, qy, qz);
            if (r.X < minX) minX = r.X; if (r.X > maxX) maxX = r.X;
            if (r.Y < minY) minY = r.Y; if (r.Y > maxY) maxY = r.Y;
            if (r.Z < minZ) minZ = r.Z; if (r.Z > maxZ) maxZ = r.Z;
        }
        return (minX, minY, minZ, maxX, maxY, maxZ);
    }

    private static Vec3 Rotate(Vec3 p, double w, double x, double y, double z)
    {
        double px = p.X, py = p.Y, pz = p.Z;
        double ix =  w * px + y * pz - z * py;
        double iy =  w * py + z * px - x * pz;
        double iz =  w * pz + x * py - y * px;
        double iw = -x * px - y * py - z * pz;

        return new Vec3(
            ix * w + iw * -x + iy * -z - iz * -y,
            iy * w + iw * -y + iz * -x - ix * -z,
            iz * w + iw * -z + ix * -y - iy * -x);
    }

    private readonly struct BoxOrient
    {
        public readonly double W, H, D;
        public readonly double Qw, Qx, Qy, Qz;
        public readonly double MinX, MinY, MinZ;
        public BoxOrient(double w, double h, double d, double qw, double qx, double qy, double qz, double minx, double miny, double minz)
        {
            W = w; H = h; D = d; Qw = qw; Qx = qx; Qy = qy; Qz = qz; MinX = minx; MinY = miny; MinZ = minz;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;
        public FastScanner(Stream stream) { _stream = stream; }
        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buf[_ptr++];
        }
        public int NextInt()
        {
            int c; do c = Read(); while (c <= 32 && c != 0);
            int sign = 1; if (c == '-') { sign = -1; c = Read(); }
            int val = 0; while (c > 32) { val = val * 10 + c - '0'; c = Read(); }
            return val * sign;
        }
        public double NextDouble()
        {
            int c; do c = Read(); while (c <= 32 && c != 0);
            int sign = 1; if (c == '-') { sign = -1; c = Read(); }
            double val = 0;
            while (c > 32 && c != '.') { val = val * 10 + c - '0'; c = Read(); }
            if (c == '.')
            {
                double frac = 0, basev = 1;
                c = Read();
                while (c > 32)
                {
                    frac = frac * 10 + c - '0';
                    basev *= 10;
                    c = Read();
                }
                val += frac / basev;
            }
            return val * sign;
        }
    }
}
