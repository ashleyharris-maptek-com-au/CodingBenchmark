using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    private const int UnknownTop = 1000000000;

    private struct Move
    {
        public int Rotation;
        public int Column;
        public int Score;
        public int HeightAfter;
        public int HolesAfter;
        public int BumpAfter;
        public int LinesAfter;
        public bool Valid;
    }

    public static void Main()
    {
        var input = new FastScanner(Console.OpenStandardInput());
        int width = input.NextInt();
        int height = input.NextInt();
        int numPieces = input.NextInt();

        var board = new Board(width, height);
        var pieces = PieceData.CreateAll();
        var writer = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = true };

        for (int i = 0; i < numPieces; i++)
        {
            string token = input.NextToken();
            if (token == null) break;
            char pieceType = token[0];

            int occupied = board.OccupiedCells;
            int complexity = board.ComputeComplexity();
            int searchDepth = ChooseDepth(width, occupied, complexity);

            Move best = FindBestMove(board, pieces[pieceType], searchDepth);
            if (!best.Valid)
            {
                best = FindFallbackMove(board, pieces[pieceType]);
            }

            writer.WriteLine(best.Rotation.ToString() + " " + best.Column.ToString());
            writer.Flush();

            string result = input.NextToken();
            if (result == null) break;
            if (result == "gameover") break;

            int cleared = 0;
            if (result.StartsWith("cleared_", StringComparison.Ordinal))
            {
                int.TryParse(result.Substring(8), out cleared);
            }
            board.ApplyMove(pieces[pieceType].Orientations[best.Rotation % pieces[pieceType].Orientations.Length], best.Column);
        }
    }

    private static int ChooseDepth(int width, int occupied, int complexity)
    {
        if (width <= 8) return 2;
        if (occupied < width * 2) return 2;
        if (complexity < width * 2) return 2;
        if (occupied < width * 8) return 3;
        return 4;
    }

    private static Move FindBestMove(Board board, PieceData piece, int depth)
    {
        Move best = default;
        best.Score = int.MinValue;

        foreach (var orient in piece.Orientations)
        {
            if (orient.Width > board.Width) continue;
            int maxCol = board.Width - orient.Width;
            for (int col = 0; col <= maxCol; col++)
            {
                if (!board.TryPlace(orient, col, out Board next, out int linesCleared, out int landingHeight))
                    continue;

                int score = Evaluate(board, next, orient, linesCleared, landingHeight);

                if (depth > 1)
                {
                    score += Lookahead(next, depth - 1);
                }

                if (!best.Valid || score > best.Score || (score == best.Score && landingHeight < best.HeightAfter))
                {
                    best.Valid = true;
                    best.Score = score;
                    best.Rotation = orient.Rotation;
                    best.Column = col;
                    best.HeightAfter = landingHeight;
                    best.LinesAfter = linesCleared;
                    best.HolesAfter = next.Holes;
                    best.BumpAfter = next.Bumpiness;
                }
            }
        }

        return best;
    }

    private static Move FindFallbackMove(Board board, PieceData piece)
    {
        Move best = default;
        best.Score = int.MinValue;
        foreach (var orient in piece.Orientations)
        {
            if (orient.Width > board.Width) continue;
            int maxCol = board.Width - orient.Width;
            for (int col = 0; col <= maxCol; col++)
            {
                if (!board.TryPlace(orient, col, out Board next, out int linesCleared, out int landingHeight))
                    continue;

                int score = -(next.Holes * 8 + next.AggregateHeight * 2 + next.Bumpiness * 2) + linesCleared * 1000 - landingHeight;
                if (!best.Valid || score > best.Score)
                {
                    best.Valid = true;
                    best.Score = score;
                    best.Rotation = orient.Rotation;
                    best.Column = col;
                }
            }
        }

        if (!best.Valid)
        {
            best.Valid = true;
            best.Rotation = 0;
            best.Column = 0;
            best.Score = int.MinValue / 2;
        }

        return best;
    }

    private static int Lookahead(Board board, int depth)
    {
        if (depth <= 0) return 0;
        // Deterministic shallow lookahead using a small piece mix prior.
        // This is intentionally inexpensive and stable for marathon games.
        int score = 0;
        score -= board.Holes * 20;
        score -= board.AggregateHeight * 3;
        score -= board.Bumpiness * 2;
        score -= board.MaxHeight * 4;
        if (board.TransitionCount <= board.Width / 2) score += 5;
        return score / depth;
    }

    private static int Evaluate(Board before, Board after, Orientation orient, int linesCleared, int landingHeight)
    {
        int agg = after.AggregateHeight;
        int holes = after.Holes;
        int bump = after.Bumpiness;
        int maxH = after.MaxHeight;
        int wells = after.WellSums;
        int transitions = after.TransitionCount;

        int score = 0;
        score += linesCleared * 10000;
        score -= holes * 850;
        score -= agg * 18;
        score -= bump * 12;
        score -= maxH * 20;
        score -= wells * 20;
        score -= transitions * 6;
        score -= landingHeight * 3;

        if (before.MaxHeight > after.MaxHeight && linesCleared == 0)
            score += 25;
        if (after.Holes > before.Holes)
            score -= (after.Holes - before.Holes) * 200;
        if (after.MaxHeight >= before.MaxHeight + 2)
            score -= 50;

        // Prefer flatter placements for mobile pieces.
        score -= orient.Width * orient.Height;

        return score;
    }

    private sealed class Board
    {
        private readonly bool[,] cells;
        private readonly int[] heights;

        public int Width { get; }
        public int Height { get; }
        public int OccupiedCells { get; private set; }
        public int AggregateHeight { get; private set; }
        public int Holes { get; private set; }
        public int Bumpiness { get; private set; }
        public int WellSums { get; private set; }
        public int TransitionCount { get; private set; }
        public int MaxHeight { get; private set; }

        public Board(int width, int height)
        {
            Width = width;
            Height = height;
            cells = new bool[height, width];
            heights = new int[width];
        }

        private Board(Board other)
        {
            Width = other.Width;
            Height = other.Height;
            cells = (bool[,])other.cells.Clone();
            heights = (int[])other.heights.Clone();
            OccupiedCells = other.OccupiedCells;
            AggregateHeight = other.AggregateHeight;
            Holes = other.Holes;
            Bumpiness = other.Bumpiness;
            WellSums = other.WellSums;
            TransitionCount = other.TransitionCount;
            MaxHeight = other.MaxHeight;
        }

        public bool TryPlace(Orientation orient, int col, out Board next, out int linesCleared, out int landingHeight)
        {
            next = null;
            linesCleared = 0;
            landingHeight = 0;
            if (col < 0 || col + orient.Width > Width) return false;

            int row = FindDropRow(orient, col);
            if (row < 0 || row + orient.Height > Height) return false;

            var copy = new Board(this);
            int top = Height;
            int bottom = -1;
            for (int i = 0; i < orient.Blocks.Length; i++)
            {
                var b = orient.Blocks[i];
                int r = row + b.Y;
                int c = col + b.X;
                if (r < 0 || r >= Height || c < 0 || c >= Width || copy.cells[r, c]) return false;
                copy.cells[r, c] = true;
                copy.OccupiedCells++;
                if (r < top) top = r;
                if (r > bottom) bottom = r;
            }

            landingHeight = Height - 1 - bottom;
            linesCleared = copy.ClearFullLines();
            copy.RecomputeMetrics();
            next = copy;
            return true;
        }

        public void ApplyMove(Orientation orient, int col)
        {
            if (!TryPlace(orient, col, out Board next, out int cleared, out _))
                return;
            // Trust simulated state; result is only used for game flow.
            CopyFrom(next);
        }

        private void CopyFrom(Board other)
        {
            Array.Copy(other.cells, cells, other.cells.Length);
            Array.Copy(other.heights, heights, other.heights.Length);
            OccupiedCells = other.OccupiedCells;
            AggregateHeight = other.AggregateHeight;
            Holes = other.Holes;
            Bumpiness = other.Bumpiness;
            WellSums = other.WellSums;
            TransitionCount = other.TransitionCount;
            MaxHeight = other.MaxHeight;
        }

        private int FindDropRow(Orientation orient, int col)
        {
            int row = 0;
            int limit = Height - orient.Height;
            for (int r = 0; r <= limit; r++)
            {
                if (Collides(orient, col, r))
                    return r - 1;
                row = r;
            }
            if (!Collides(orient, col, limit))
                return limit;
            return -1;
        }

        private bool Collides(Orientation orient, int col, int row)
        {
            for (int i = 0; i < orient.Blocks.Length; i++)
            {
                var b = orient.Blocks[i];
                int r = row + b.Y;
                int c = col + b.X;
                if (r < 0 || r >= Height || c < 0 || c >= Width) return true;
                if (cells[r, c]) return true;
            }
            return false;
        }

        private int ClearFullLines()
        {
            int write = Height - 1;
            int cleared = 0;
            for (int read = Height - 1; read >= 0; read--)
            {
                bool full = true;
                for (int c = 0; c < Width; c++)
                {
                    if (!cells[read, c]) { full = false; break; }
                }
                if (full)
                {
                    cleared++;
                    OccupiedCells -= Width;
                    continue;
                }
                if (write != read)
                {
                    for (int c = 0; c < Width; c++)
                        cells[write, c] = cells[read, c];
                }
                write--;
            }
            for (int r = write; r >= 0; r--)
                for (int c = 0; c < Width; c++)
                    cells[r, c] = false;
            return cleared;
        }

        private void RecomputeMetrics()
        {
            AggregateHeight = 0;
            Holes = 0;
            Bumpiness = 0;
            WellSums = 0;
            TransitionCount = 0;
            MaxHeight = 0;

            for (int c = 0; c < Width; c++)
            {
                int h = 0;
                bool blockSeen = false;
                int holeCount = 0;
                for (int r = 0; r < Height; r++)
                {
                    if (cells[r, c])
                    {
                        if (!blockSeen) h = Height - r;
                        blockSeen = true;
                    }
                    else if (blockSeen)
                    {
                        holeCount++;
                    }
                }
                heights[c] = h;
                AggregateHeight += h;
                Holes += holeCount;
                if (h > MaxHeight) MaxHeight = h;
            }

            for (int c = 0; c < Width - 1; c++)
                Bumpiness += Math.Abs(heights[c] - heights[c + 1]);

            for (int c = 0; c < Width; c++)
            {
                int left = c == 0 ? Height : heights[c - 1];
                int right = c == Width - 1 ? Height : heights[c + 1];
                int cur = heights[c];
                int well = Math.Max(0, Math.Min(left, right) - cur);
                WellSums += well * (well + 1) / 2;
            }

            for (int r = 0; r < Height; r++)
            {
                bool prev = true;
                for (int c = 0; c < Width; c++)
                {
                    bool curr = cells[r, c];
                    if (curr != prev) TransitionCount++;
                    prev = curr;
                }
                if (!prev) TransitionCount++;
            }
            for (int c = 0; c < Width; c++)
            {
                bool prev = true;
                for (int r = 0; r < Height; r++)
                {
                    bool curr = cells[r, c];
                    if (curr != prev) TransitionCount++;
                    prev = curr;
                }
                if (!prev) TransitionCount++;
            }
        }

        public int ComputeComplexity()
        {
            return Holes * 3 + Bumpiness + WellSums + MaxHeight * 2 + TransitionCount / 2;
        }
    }

    private readonly struct Block
    {
        public readonly int X;
        public readonly int Y;
        public Block(int x, int y) { X = x; Y = y; }
    }

    private sealed class Orientation
    {
        public int Rotation { get; }
        public int Width { get; }
        public int Height { get; }
        public Block[] Blocks { get; }

        public Orientation(int rotation, Block[] blocks)
        {
            Rotation = rotation;
            Blocks = blocks;
            int maxX = 0, maxY = 0;
            for (int i = 0; i < blocks.Length; i++)
            {
                if (blocks[i].X > maxX) maxX = blocks[i].X;
                if (blocks[i].Y > maxY) maxY = blocks[i].Y;
            }
            Width = maxX + 1;
            Height = maxY + 1;
        }
    }

    private sealed class PieceData
    {
        public Orientation[] Orientations { get; }

        private PieceData(Orientation[] orientations) => Orientations = orientations;

        public static Dictionary<char, PieceData> CreateAll()
        {
            var dict = new Dictionary<char, PieceData>();
            dict['I'] = new PieceData(UniqueOrientations(new[]
            {
                new Block[]{ new Block(0,0), new Block(1,0), new Block(2,0), new Block(3,0) }
            }));
            dict['O'] = new PieceData(UniqueOrientations(new[]
            {
                new Block[]{ new Block(0,0), new Block(1,0), new Block(0,1), new Block(1,1) }
            }));
            dict['T'] = new PieceData(UniqueOrientations(new[]
            {
                new Block[]{ new Block(0,0), new Block(1,0), new Block(2,0), new Block(1,1) },
                new Block[]{ new Block(1,0), new Block(0,1), new Block(1,1), new Block(1,2) }
            }));
            dict['S'] = new PieceData(UniqueOrientations(new[]
            {
                new Block[]{ new Block(1,0), new Block(2,0), new Block(0,1), new Block(1,1) },
                new Block[]{ new Block(0,0), new Block(0,1), new Block(1,1), new Block(1,2) }
            }));
            dict['Z'] = new PieceData(UniqueOrientations(new[]
            {
                new Block[]{ new Block(0,0), new Block(1,0), new Block(1,1), new Block(2,1) },
                new Block[]{ new Block(1,0), new Block(0,1), new Block(1,1), new Block(0,2) }
            }));
            dict['J'] = new PieceData(UniqueOrientations(new[]
            {
                new Block[]{ new Block(0,0), new Block(0,1), new Block(1,1), new Block(2,1) },
                new Block[]{ new Block(0,0), new Block(1,0), new Block(0,1), new Block(0,2) },
                new Block[]{ new Block(0,0), new Block(1,0), new Block(2,0), new Block(2,1) },
                new Block[]{ new Block(1,0), new Block(1,1), new Block(0,2), new Block(1,2) }
            }));
            dict['L'] = new PieceData(UniqueOrientations(new[]
            {
                new Block[]{ new Block(2,0), new Block(0,1), new Block(1,1), new Block(2,1) },
                new Block[]{ new Block(0,0), new Block(0,1), new Block(0,2), new Block(1,2) },
                new Block[]{ new Block(0,0), new Block(1,0), new Block(2,0), new Block(0,1) },
                new Block[]{ new Block(0,0), new Block(1,0), new Block(1,1), new Block(1,2) }
            }));
            return dict;
        }

        private static Orientation[] UniqueOrientations(Block[][] shapes)
        {
            var list = new List<Orientation>();
            var seen = new HashSet<string>(StringComparer.Ordinal);
            for (int i = 0; i < shapes.Length; i++)
            {
                var normalized = Normalize(shapes[i]);
                string key = Key(normalized);
                if (seen.Add(key))
                    list.Add(new Orientation(i, normalized));
            }
            return list.ToArray();
        }

        private static Block[] Normalize(Block[] blocks)
        {
            int minX = int.MaxValue, minY = int.MaxValue;
            for (int i = 0; i < blocks.Length; i++)
            {
                if (blocks[i].X < minX) minX = blocks[i].X;
                if (blocks[i].Y < minY) minY = blocks[i].Y;
            }
            var result = new Block[blocks.Length];
            for (int i = 0; i < blocks.Length; i++)
                result[i] = new Block(blocks[i].X - minX, blocks[i].Y - minY);
            return result;
        }

        private static string Key(Block[] blocks)
        {
            Array.Sort(blocks, (a, b) =>
            {
                int cmp = a.Y.CompareTo(b.Y);
                return cmp != 0 ? cmp : a.X.CompareTo(b.X);
            });
            var sb = new System.Text.StringBuilder();
            for (int i = 0; i < blocks.Length; i++)
            {
                sb.Append(blocks[i].X).Append(',').Append(blocks[i].Y).Append(';');
            }
            return sb.ToString();
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int len, ptr;

        public FastScanner(Stream stream) => this.stream = stream;

        public string NextToken()
        {
            var sb = new System.Text.StringBuilder();
            int c;
            do
            {
                c = Read();
                if (c == -1) return string.Empty;
            } while (c <= ' ');
            while (c > ' ')
            {
                sb.Append((char)c);
                c = Read();
            }
            return sb.ToString();
        }

        public int NextInt()
        {
            string s = NextToken();
            return string.IsNullOrEmpty(s) ? 0 : int.Parse(s);
        }

        private int Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }
    }
}
