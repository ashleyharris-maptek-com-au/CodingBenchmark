using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Text;

public class Program
{
    private const int MaxDepthComplex = 1;
    private const int MaxDepthSimple = 2;
    private const int BeamWidth = 6;

    private static readonly Dictionary<char, PieceDef[]> PieceDefs = new()
    {
        ['I'] = new[]
        {
            new PieceDef(new[] { (0,0), (1,0), (2,0), (3,0) }, 4, 1),
            new PieceDef(new[] { (0,0), (0,1), (0,2), (0,3) }, 1, 4),
            new PieceDef(new[] { (0,0), (1,0), (2,0), (3,0) }, 4, 1),
            new PieceDef(new[] { (0,0), (0,1), (0,2), (0,3) }, 1, 4),
        },
        ['O'] = new[]
        {
            new PieceDef(new[] { (0,0), (1,0), (0,1), (1,1) }, 2, 2),
            new PieceDef(new[] { (0,0), (1,0), (0,1), (1,1) }, 2, 2),
            new PieceDef(new[] { (0,0), (1,0), (0,1), (1,1) }, 2, 2),
            new PieceDef(new[] { (0,0), (1,0), (0,1), (1,1) }, 2, 2),
        },
        ['T'] = new[]
        {
            new PieceDef(new[] { (0,0), (1,0), (2,0), (1,1) }, 3, 2),
            new PieceDef(new[] { (0,0), (0,1), (0,2), (1,1) }, 2, 3),
            new PieceDef(new[] { (1,0), (0,1), (1,1), (2,1) }, 3, 2),
            new PieceDef(new[] { (0,1), (1,0), (1,1), (1,2) }, 2, 3),
        },
        ['S'] = new[]
        {
            new PieceDef(new[] { (1,0), (2,0), (0,1), (1,1) }, 3, 2),
            new PieceDef(new[] { (0,0), (0,1), (1,1), (1,2) }, 2, 3),
            new PieceDef(new[] { (1,0), (2,0), (0,1), (1,1) }, 3, 2),
            new PieceDef(new[] { (0,0), (0,1), (1,1), (1,2) }, 2, 3),
        },
        ['Z'] = new[]
        {
            new PieceDef(new[] { (0,0), (1,0), (1,1), (2,1) }, 3, 2),
            new PieceDef(new[] { (1,0), (0,1), (1,1), (0,2) }, 2, 3),
            new PieceDef(new[] { (0,0), (1,0), (1,1), (2,1) }, 3, 2),
            new PieceDef(new[] { (1,0), (0,1), (1,1), (0,2) }, 2, 3),
        },
        ['J'] = new[]
        {
            new PieceDef(new[] { (0,0), (0,1), (1,1), (2,1) }, 3, 2),
            new PieceDef(new[] { (0,0), (1,0), (0,1), (0,2) }, 2, 3),
            new PieceDef(new[] { (0,0), (1,0), (2,0), (2,1) }, 3, 2),
            new PieceDef(new[] { (1,0), (1,1), (1,2), (0,2) }, 2, 3),
        },
        ['L'] = new[]
        {
            new PieceDef(new[] { (2,0), (0,1), (1,1), (2,1) }, 3, 2),
            new PieceDef(new[] { (0,0), (0,1), (0,2), (1,0) }, 2, 3),
            new PieceDef(new[] { (0,0), (1,0), (2,0), (0,1) }, 3, 2),
            new PieceDef(new[] { (0,2), (1,0), (1,1), (1,2) }, 2, 3),
        },
    };

    public static void Main()
    {
        var input = new FastScanner(Console.OpenStandardInput());
        int width = input.NextInt();
        int height = input.NextInt();
        int numPieces = input.NextInt();

        var board = new Board(width, height);
        var rng = new Random(1);
        int pieceIndex = 0;

        while (pieceIndex < numPieces)
        {
            string token = input.NextToken();
            if (token == null) break;
            char pieceType = token[0];

            bool simpleMode = board.MaxHeight() <= Math.Max(4, height / 3) && board.Holes() <= width / 2;
            int depth = simpleMode ? MaxDepthSimple : MaxDepthComplex;

            Move best = ChooseMove(board, pieceType, depth, pieceIndex, rng);
            Console.WriteLine($"{best.Rotation} {best.Column}");
            Console.Out.Flush();

            string result = input.NextToken();
            if (result == null) break;
            if (result == "gameover") break;
            if (result.StartsWith("cleared_", StringComparison.Ordinal))
            {
                int cleared = int.Parse(result.AsSpan(8));
                if (cleared > 0) board.ClearLines(cleared);
            }

            board.Place(pieceType, best.Rotation, best.Column, out bool ok);
            if (!ok) break;
            pieceIndex++;
        }
    }

    private static Move ChooseMove(Board board, char pieceType, int depth, int pieceIndex, Random rng)
    {
        var candidates = GenerateMoves(board, pieceType);
        if (candidates.Count == 0) return new Move(0, 0, double.NegativeInfinity);

        var scored = new List<Move>();
        foreach (var mv in candidates)
        {
            var sim = board.Clone();
            if (!sim.Place(pieceType, mv.Rotation, mv.Column, out bool ok) || !ok)
            {
                mv.Score = double.NegativeInfinity;
                scored.Add(mv);
                continue;
            }
            mv.Score = Evaluate(sim, 0);
            if (depth > 1)
            {
                mv.Score += LookaheadScore(sim, depth - 1, pieceIndex + 1, rng) * 0.7;
            }
            scored.Add(mv);
        }

        return scored.OrderByDescending(m => m.Score)
                     .ThenBy(m => Math.Abs(m.Column - board.Width / 2))
                     .ThenBy(m => m.Rotation)
                     .First();
    }

    private static double LookaheadScore(Board board, int depth, int seed, Random rng)
    {
        if (depth <= 0) return Evaluate(board, 0);
        double best = double.NegativeInfinity;
        char[] bag = { 'I', 'O', 'T', 'S', 'Z', 'J', 'L' };
        int branches = Math.Min(4, bag.Length);
        for (int i = 0; i < branches; i++)
        {
            char piece = bag[(seed + i * 3 + rng.Next(7)) % bag.Length];
            foreach (var mv in GenerateMoves(board, piece).Take(BeamWidth))
            {
                var sim = board.Clone();
                if (!sim.Place(piece, mv.Rotation, mv.Column, out bool ok) || !ok) continue;
                double score = Evaluate(sim, 0);
                if (depth > 1) score += 0.6 * LookaheadScore(sim, depth - 1, seed + 1, rng);
                if (score > best) best = score;
            }
        }
        return best == double.NegativeInfinity ? Evaluate(board, 0) : best;
    }

    private static List<Move> GenerateMoves(Board board, char pieceType)
    {
        var defs = PieceDefs[pieceType];
        var moves = new List<Move>();
        var seen = new HashSet<(int,int)>();
        for (int r = 0; r < 4; r++)
        {
            var def = defs[r];
            for (int c = 0; c <= board.Width - def.Width; c++)
            {
                if (seen.Add((r, c))) moves.Add(new Move(r, c, 0));
            }
        }
        return moves;
    }

    private static double Evaluate(Board b, int cleared)
    {
        int aggHeight = b.AggregateHeight();
        int holes = b.Holes();
        int bump = b.Bumpiness();
        int maxH = b.MaxHeight();
        int lines = cleared;
        return lines * 120.0 - aggHeight * 4.5 - holes * 9.0 - bump * 3.0 - maxH * 2.0;
    }

    private readonly record struct PieceDef((int X, int Y)[] Cells, int Width, int Height);

    private struct Move
    {
        public int Rotation;
        public int Column;
        public double Score;
        public Move(int rotation, int column, double score)
        {
            Rotation = rotation;
            Column = column;
            Score = score;
        }
    }

    private sealed class Board
    {
        private readonly bool[,] cells;
        public int Width { get; }
        public int Height { get; }

        public Board(int width, int height)
        {
            Width = width;
            Height = height;
            cells = new bool[width, height];
        }

        private Board(int width, int height, bool[,] source)
        {
            Width = width;
            Height = height;
            cells = (bool[,])source.Clone();
        }

        public Board Clone() => new Board(Width, Height, cells);

        public bool Place(char pieceType, int rotation, int column, out bool ok)
        {
            var def = PieceDefs[pieceType][rotation & 3];
            int y = FindDropY(def, column);
            if (y < 0)
            {
                ok = false;
                return false;
            }
            foreach (var (dx, dy) in def.Cells)
            {
                int x = column + dx;
                int yy = y + dy;
                if (x < 0 || x >= Width || yy < 0 || yy >= Height || cells[x, yy])
                {
                    ok = false;
                    return false;
                }
            }
            foreach (var (dx, dy) in def.Cells)
            {
                cells[column + dx, y + dy] = true;
            }
            ClearFullLines();
            ok = true;
            return true;
        }

        public void ClearLines(int count)
        {
            if (count <= 0) return;
            ClearFullLines();
        }

        private int FindDropY(PieceDef def, int column)
        {
            int maxY = Height - def.Height;
            for (int y = maxY; y >= 0; y--)
            {
                bool collides = false;
                foreach (var (dx, dy) in def.Cells)
                {
                    int x = column + dx;
                    int yy = y + dy;
                    if (x < 0 || x >= Width || yy < 0 || yy >= Height || cells[x, yy])
                    {
                        collides = true;
                        break;
                    }
                }
                if (!collides) return y;
            }
            return -1;
        }

        private void ClearFullLines()
        {
            int writeY = 0;
            for (int y = 0; y < Height; y++)
            {
                bool full = true;
                for (int x = 0; x < Width; x++)
                {
                    if (!cells[x, y]) { full = false; break; }
                }
                if (!full)
                {
                    if (writeY != y)
                    {
                        for (int x = 0; x < Width; x++) cells[x, writeY] = cells[x, y];
                    }
                    writeY++;
                }
            }
            for (int y = writeY; y < Height; y++)
                for (int x = 0; x < Width; x++)
                    cells[x, y] = false;
        }

        public int MaxHeight()
        {
            int max = 0;
            for (int x = 0; x < Width; x++)
            {
                int h = ColumnHeight(x);
                if (h > max) max = h;
            }
            return max;
        }

        public int AggregateHeight()
        {
            int sum = 0;
            for (int x = 0; x < Width; x++) sum += ColumnHeight(x);
            return sum;
        }

        public int Holes()
        {
            int holes = 0;
            for (int x = 0; x < Width; x++)
            {
                bool blockSeen = false;
                for (int y = Height - 1; y >= 0; y--)
                {
                    if (cells[x, y]) blockSeen = true;
                    else if (blockSeen) holes++;
                }
            }
            return holes;
        }

        public int Bumpiness()
        {
            int b = 0;
            int prev = ColumnHeight(0);
            for (int x = 1; x < Width; x++)
            {
                int cur = ColumnHeight(x);
                b += Math.Abs(cur - prev);
                prev = cur;
            }
            return b;
        }

        private int ColumnHeight(int x)
        {
            for (int y = Height - 1; y >= 0; y--)
                if (cells[x, y]) return y + 1;
            return 0;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int len, ptr;

        public FastScanner(Stream stream) { this.stream = stream; }

        public string NextToken()
        {
            var sb = new StringBuilder();
            int c;
            do
            {
                c = Read();
                if (c == -1) return null;
            } while (char.IsWhiteSpace((char)c));

            while (c != -1 && !char.IsWhiteSpace((char)c))
            {
                sb.Append((char)c);
                c = Read();
            }
            return sb.ToString();
        }

        public int NextInt() => int.Parse(NextToken());

        private int Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }
    }
}
