import math

_state = {
    "last_trim": 0.0,
    "alt_i": 0.0,
    "spd_i": 0.0,
    "hdg_i": 0.0,
    "vs_i": 0.0,
    "last_hdg": None,
    "last_track": None,
}


def _clamp(x, lo, hi):
    return max(lo, min(hi, x))


def _wrap_deg(a):
    a = math.fmod(a, 360.0)
    if a < 0:
        a += 360.0
    return a


def _ang_diff(a, b):
    d = (_wrap_deg(a) - _wrap_deg(b) + 180.0) % 360.0 - 180.0
    return d


def _median(vals):
    vals = [v for v in vals if v is not None and not isinstance(v, bool)]
    if not vals:
        return None
    vals = sorted(vals)
    n = len(vals)
    mid = n // 2
    if n % 2:
        return vals[mid]
    return 0.5 * (vals[mid - 1] + vals[mid])


def _robust_pick(values, valid_flags=None, max_spread=None):
    xs = []
    for i, v in enumerate(values):
        if v is None:
            continue
        if valid_flags is not None and i < len(valid_flags) and valid_flags[i] is False:
            continue
        xs.append(float(v))
    if not xs:
        return None
    med = _median(xs)
    if max_spread is None or len(xs) < 3:
        return med
    kept = [x for x in xs if abs(x - med) <= max_spread]
    if kept:
        return _median(kept)
    return med


def _sensor_triplet(state, base, unit=""):
    vals = [state.get(f"{base}_{i}{unit}") for i in (1, 2, 3)]
    vflags = [state.get(f"{base}_{i}_valid") for i in (1, 2, 3)] if f"{base}_1_valid" in state else None
    return _robust_pick(vals, vflags)


def _choose_heading(state):
    hdgs = [
        state.get("magnetic_heading_deg"),
        state.get("true_heading_deg"),
        state.get("track_deg"),
        state.get("irs_1_heading_deg"),
        state.get("irs_2_heading_deg"),
        state.get("irs_3_heading_deg"),
    ]
    h = _robust_pick(hdgs, max_spread=45.0)
    if h is None:
        h = _state["last_hdg"] if _state["last_hdg"] is not None else 0.0
    return _wrap_deg(h)


def _choose_track(state):
    trk = _robust_pick([
        state.get("track_deg"),
        state.get("gps_1_track_deg"),
        state.get("gps_2_track_deg"),
        state.get("gps_3_track_deg"),
        state.get("irs_1_track_deg"),
        state.get("irs_2_track_deg"),
        state.get("irs_3_track_deg"),
    ], max_spread=30.0)
    if trk is None:
        trk = _state["last_track"] if _state["last_track"] is not None else 0.0
    return _wrap_deg(trk)


def _altitude_estimate(state):
    baro = _robust_pick([
        state.get("baro_alt_1_ft"),
        state.get("baro_alt_2_ft"),
        state.get("baro_alt_3_ft"),
    ], [state.get("baro_alt_1_valid"), state.get("baro_alt_2_valid"), state.get("baro_alt_3_valid")], max_spread=500.0)
    radar = _robust_pick([
        state.get("radar_alt_1_ft"),
        state.get("radar_alt_2_ft"),
        state.get("radar_alt_3_ft"),
    ], [state.get("radar_alt_1_valid"), state.get("radar_alt_2_valid"), state.get("radar_alt_3_valid")], max_spread=100.0)
    gps = _robust_pick([
        state.get("gps_1_alt_ft"),
        state.get("gps_2_alt_ft"),
        state.get("gps_3_alt_ft"),
    ], [state.get("gps_1_valid"), state.get("gps_2_valid"), state.get("gps_3_valid")], max_spread=500.0)
    irs = _robust_pick([
        state.get("irs_1_lat") and state.get("irs_1_gs_kt") and state.get("irs_1_vs_fpm"),
        state.get("irs_2_lat") and state.get("irs_2_gs_kt") and state.get("irs_2_vs_fpm"),
        state.get("irs_3_lat") and state.get("irs_3_gs_kt") and state.get("irs_3_vs_fpm"),
    ])
    choices = []
    if radar is not None and radar < 2500:
        choices.append(radar)
    if baro is not None:
        choices.append(baro)
    if gps is not None:
        choices.append(gps)
    if not choices:
        return 0.0
    return _median(choices)


def autopilot_step(state: dict, dt: float) -> dict:
    dt = float(state.get("dt", dt) or dt or 0.05)
    dt = _clamp(dt, 0.01, 0.2)

    alt = _altitude_estimate(state)
    target_alt = float(state.get("target_alt_ft", alt))
    target_hdg = float(state.get("target_hdg_deg", _choose_heading(state)))
    target_ias = float(state.get("target_ias_kt", 250.0))
    target_vs = float(state.get("target_vs_fpm", 0.0))

    ias = _robust_pick([
        state.get("ias_1_kt"),
        state.get("ias_2_kt"),
        state.get("ias_3_kt"),
    ], [state.get("ias_1_valid"), state.get("ias_2_valid"), state.get("ias_3_valid")], max_spread=40.0)
    gs = _robust_pick([state.get("ground_speed_kt"), state.get("gps_1_gs_kt"), state.get("gps_2_gs_kt"), state.get("gps_3_gs_kt"), state.get("irs_1_gs_kt"), state.get("irs_2_gs_kt"), state.get("irs_3_gs_kt")], max_spread=50.0)
    if ias is None:
        ias = gs if gs is not None else 180.0

    vs = _robust_pick([state.get("vertical_speed_fpm"), state.get("irs_1_vs_fpm"), state.get("irs_2_vs_fpm"), state.get("irs_3_vs_fpm")], max_spread=3000.0)
    if vs is None:
        vs = 0.0

    aoa = _robust_pick([state.get("aoa_1_deg"), state.get("aoa_2_deg"), state.get("aoa_3_deg")], [state.get("aoa_1_valid"), state.get("aoa_2_valid"), state.get("aoa_3_valid")], max_spread=8.0)
    if aoa is None:
        aoa = 5.0

    pitch = _robust_pick([state.get("irs_1_pitch_deg"), state.get("irs_2_pitch_deg"), state.get("irs_3_pitch_deg")], max_spread=15.0)
    roll = _robust_pick([state.get("irs_1_roll_deg"), state.get("irs_2_roll_deg"), state.get("irs_3_roll_deg")], max_spread=20.0)
    if pitch is None:
        pitch = 0.0
    if roll is None:
        roll = 0.0

    rudder = 0.0
    aileron = 0.0
    elevator = 0.0
    trim = _state["last_trim"]
    throttle_1 = state.get("throttle_1_pos", 0.75)
    throttle_2 = state.get("throttle_2_pos", 0.75)
    flaps = 0
    speedbrake = 0.0
    gear = 0

    # Speed schedule and stall protection.
    stall_margin = max(0.0, 9.0 - aoa)
    min_safe_ias = max(120.0, 1.3 * (65.0 + 3.0 * max(0.0, aoa - 5.0)))
    if alt < 2000.0:
        min_safe_ias = max(min_safe_ias, 115.0)
    if ias < min_safe_ias + 8.0 or aoa > 10.5:
        throttle_1 = 0.98
        throttle_2 = 0.98
        flaps = 20 if alt < 15000.0 else 0
        speedbrake = 0.0
        elevator = _clamp(0.15 + 0.015 * (min_safe_ias - ias) - 0.03 * vs / 1000.0, -0.25, 0.35)
    else:
        spd_err = target_ias - ias
        thrust_cmd = 0.55 + 0.0045 * spd_err + 0.00015 * (target_alt - alt) - 0.00008 * vs
        thrust_cmd = _clamp(thrust_cmd, 0.2, 1.0)
        if alt > 30000.0 and target_ias > 270.0:
            thrust_cmd = max(thrust_cmd, 0.78)
        throttle_1 = thrust_cmd
        throttle_2 = thrust_cmd
        if abs(spd_err) > 25.0 and alt < 10000.0:
            flaps = 5 if spd_err < 0 else 0
        if ias > 0.92 * 365.0:
            speedbrake = min(1.0, (ias - 330.0) / 40.0)
        elevator = _clamp(0.00065 * (target_alt - alt) - 0.00008 * vs + 0.0008 * (target_vs - vs), -0.35, 0.35)

    # Altitude hold / climb / descent blending.
    alt_err = target_alt - alt
    _state["alt_i"] = _clamp(_state["alt_i"] + alt_err * dt, -20000.0, 20000.0)
    elevator += _clamp(0.00003 * _state["alt_i"], -0.08, 0.08)

    # Trim slowly drives out sustained elevator demand.
    trim -= _clamp(elevator * 0.01, -0.01, 0.01)
    trim = _clamp(trim, -1.0, 1.0)
    _state["last_trim"] = trim
    elevator = _clamp(elevator - 0.3 * trim, -1.0, 1.0)

    # Heading / bank control.
    hdg = _choose_heading(state)
    trk = _choose_track(state)
    _state["last_hdg"] = hdg
    _state["last_track"] = trk
    hdg_err = _ang_diff(target_hdg, hdg)
    cross_track = _ang_diff(target_hdg, trk)
    bank_cmd = _clamp(0.02 * hdg_err + 0.01 * cross_track - 0.35 * roll, -25.0, 25.0)
    aileron = _clamp(bank_cmd / 25.0 - 0.08 * roll / 30.0, -1.0, 1.0)

    # Rudder coordinates yaw and engine-out asymmetry.
    n1 = state.get("engine_1_n1_pct", 0.0) or 0.0
    n2 = state.get("engine_2_n1_pct", 0.0) or 0.0
    thr1 = float(state.get("throttle_1_pos", throttle_1) or throttle_1)
    thr2 = float(state.get("throttle_2_pos", throttle_2) or throttle_2)
    asym = (thr2 - thr1) + 0.003 * (n2 - n1)
    rudder = _clamp(0.015 * hdg_err + 0.02 * asym - 0.02 * roll - 0.0008 * (state.get("yaw_rate_dps", 0.0) or 0.0), -1.0, 1.0)

    # Gear and flap management.
    if alt < 1200.0 and ias < 160.0 and target_alt < 3000.0:
        gear = 1
    if alt > 1800.0 or ias > 170.0:
        gear = 0
    if alt < 800.0 and target_alt < 2000.0:
        flaps = max(flaps, 10)
    if alt < 500.0 and ias < 140.0:
        flaps = max(flaps, 15)

    # Engine protection and symmetric thrust bias.
    if state.get("engine_1_fire") or state.get("engine_1_running") is False:
        throttle_1 = 0.0
    if state.get("engine_2_fire") or state.get("engine_2_running") is False:
        throttle_2 = 0.0
    if state.get("engine_1_vibration", 0.0) > 0.85 or state.get("engine_1_egt_c", 0.0) > 930.0:
        throttle_1 = min(throttle_1, 0.85)
    if state.get("engine_2_vibration", 0.0) > 0.85 or state.get("engine_2_egt_c", 0.0) > 930.0:
        throttle_2 = min(throttle_2, 0.85)

    # Final envelope and smoothing.
    throttle_1 = _clamp(throttle_1, 0.0, 1.0)
    throttle_2 = _clamp(throttle_2, 0.0, 1.0)
    flaps = int(_clamp(round(flaps / 5.0) * 5, 0, 40))
    speedbrake = _clamp(speedbrake, 0.0, 1.0)
    gear = 1 if gear else 0

    # If the aircraft is unstable, prioritize recovery.
    if pitch > 18.0 or aoa > 13.0:
        elevator = min(elevator, -0.2)
        speedbrake = 0.0
        throttle_1 = max(throttle_1, 0.9)
        throttle_2 = max(throttle_2, 0.9)
    if roll > 45.0:
        aileron = _clamp(-0.03 * roll, -1.0, 1.0)
        rudder = _clamp(rudder - 0.02 * roll, -1.0, 1.0)
    if abs(vs) > 3500.0:
        elevator = _clamp(elevator - 0.00005 * vs, -0.6, 0.6)

    return {
        "elevator": _clamp(elevator, -1.0, 1.0),
        "aileron": _clamp(aileron, -1.0, 1.0),
        "rudder": _clamp(rudder, -1.0, 1.0),
        "throttle_1": throttle_1,
        "throttle_2": throttle_2,
        "elevator_trim": trim,
        "flaps": flaps,
        "speedbrake": speedbrake,
        "gear": gear,
    }
