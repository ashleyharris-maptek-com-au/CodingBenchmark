#include <algorithm>
#include <array>
#include <cstdint>

#include <unordered_set>
#include <utility>
#include <vector>

using namespace std;

struct VecHash {
    size_t operator()(const vector<int>& v) const noexcept {
        size_t h = 1469598103934665603ull;
        for (int x : v) {
            uint64_t y = static_cast<uint64_t>(static_cast<int64_t>(x)) ^ 0x9e3779b97f4a7c15ull;
            h ^= static_cast<size_t>(y + 0x9e3779b97f4a7c15ull + (h << 6) + (h >> 2));
        }
        return h;
    }
};

struct State {
    int D = 0;
    vector<int> bounds;
    vector<int> head;
    unordered_set<vector<int>, VecHash> blocked;
    vector<vector<int>> foods;
};

static bool in_bounds(const vector<int>& p, const vector<int>& bounds) {
    for (size_t i = 0; i < p.size(); ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    return true;
}

static vector<int> apply_move(const vector<int>& p, int axis, int dir) {
    vector<int> q = p;
    q[axis] += dir;
    return q;
}

static int manhattan_nd(const vector<int>& a, const vector<int>& b) {
    int s = 0;
    for (size_t i = 0; i < a.size(); ++i) s += abs(a[i] - b[i]);
    return s;
}

static optional<pair<int,int>> choose_greedy_move(const State& st) {
    if (st.foods.empty()) return nullopt;
    const vector<int>& target = st.foods.front();
    int best_dist = numeric_limits<int>::max();
    optional<pair<int,int>> best;

    for (int axis = 0; axis < st.D; ++axis) {
        for (int dir : {-1, 1}) {
            vector<int> nxt = apply_move(st.head, axis, dir);
            if (!in_bounds(nxt, st.bounds)) continue;
            if (st.blocked.find(nxt) != st.blocked.end()) continue;
            int d = manhattan_nd(nxt, target);
            if (d < best_dist) {
                best_dist = d;
                best = make_pair(axis, dir);
            }
        }
    }
    return best;
}

static optional<pair<int,int>> choose_safe_move(const State& st) {
    int best_score = -1;
    optional<pair<int,int>> best;

    for (int axis = 0; axis < st.D; ++axis) {
        for (int dir : {-1, 1}) {
            vector<int> nxt = apply_move(st.head, axis, dir);
            if (!in_bounds(nxt, st.bounds)) continue;
            if (st.blocked.find(nxt) != st.blocked.end()) continue;

            int score = 0;
            for (int a = 0; a < st.D; ++a) {
                for (int d : {-1, 1}) {
                    vector<int> nn = apply_move(nxt, a, d);
                    if (in_bounds(nn, st.bounds) && st.blocked.find(nn) == st.blocked.end()) {
                        ++score;
                    }
                }
            }
            if (score > best_score) {
                best_score = score;
                best = make_pair(axis, dir);
            }
        }
    }
    return best;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    State st;
    int max_turns = 0;
    int snake_length = 0;

    if (!(cin >> st.D >> max_turns)) return 0;
    st.bounds.resize(st.D);
    for (int i = 0; i < st.D; ++i) cin >> st.bounds[i];

    cin >> snake_length;
    st.head.resize(st.D);
    for (int i = 0; i < st.D; ++i) cin >> st.head[i];

    int F = 0;
    cin >> F;
    st.foods.resize(F);
    for (int i = 0; i < F; ++i) {
        st.foods[i].resize(st.D);
        for (int j = 0; j < st.D; ++j) cin >> st.foods[i][j];
    }

    int O = 0;
    cin >> O;
    for (int i = 0; i < O; ++i) {
        vector<int> p(st.D);
        for (int j = 0; j < st.D; ++j) cin >> p[j];
        st.blocked.insert(std::move(p));
    }

    for (int turn = 0; turn < max_turns; ++turn) {
        optional<pair<int,int>> mv = choose_greedy_move(st);
        if (!mv) mv = choose_safe_move(st);
        if (!mv) break;

        cout << mv->first << ' ' << mv->second << '\n';

        st.head = apply_move(st.head, mv->first, mv->second);
        st.blocked.insert(st.head);

        if (!st.foods.empty() && st.head == st.foods.front()) {
            st.foods.erase(st.foods.begin());
        }
    }

    return 0;
}
