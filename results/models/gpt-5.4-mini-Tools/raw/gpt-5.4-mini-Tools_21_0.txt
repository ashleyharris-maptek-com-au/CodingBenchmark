#include <algorithm>
#include <array>
#include <iostream>
#include <limits>
#include <queue>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using namespace std;

struct VecHash {
    size_t operator()(const vector<int>& v) const noexcept {
        size_t h = 1469598103934665603ull;
        for (int x : v) {
            size_t k = static_cast<size_t>(x) + 0x9e3779b97f4a7c15ull;
            h ^= k + (h << 6) + (h >> 2);
        }
        return h;
    }
};

static bool inBounds(const vector<int>& p, const vector<int>& b) {
    for (size_t i = 0; i < p.size(); ++i) {
        if (p[i] < 0 || p[i] >= b[i]) return false;
    }
    return true;
}

static int manhattan(const vector<int>& a, const vector<int>& b) {
    int d = 0;
    for (size_t i = 0; i < a.size(); ++i) d += abs(a[i] - b[i]);
    return d;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int D, max_turns;
    if (!(cin >> D >> max_turns)) return 0;

    vector<int> bounds(D);
    for (int i = 0; i < D; ++i) cin >> bounds[i];

    int snake_length;
    vector<int> head(D);
    cin >> snake_length;
    for (int i = 0; i < D; ++i) cin >> head[i];

    int F;
    cin >> F;
    vector<vector<int>> food(F, vector<int>(D));
    for (int i = 0; i < F; ++i) {
        for (int j = 0; j < D; ++j) cin >> food[i][j];
    }

    int O;
    cin >> O;
    unordered_set<vector<int>, VecHash> blocked;
    blocked.reserve(static_cast<size_t>(O) * 2 + static_cast<size_t>(snake_length) * 2 + 32);
    for (int i = 0; i < O; ++i) {
        vector<int> p(D);
        for (int j = 0; j < D; ++j) cin >> p[j];
        blocked.insert(move(p));
    }

    vector<vector<int>> body;
    body.reserve(max(1, snake_length));
    for (int i = 0; i < snake_length; ++i) body.push_back(head);
    blocked.insert(head);

    auto isFood = [&](const vector<int>& p) {
        for (const auto& f : food) if (f == p) return true;
        return false;
    };

    auto removeFood = [&](const vector<int>& p) {
        auto it = remove(food.begin(), food.end(), p);
        food.erase(it, food.end());
    };

    auto chooseMove = [&]() -> pair<int, int> {
        vector<pair<int,int>> candidates;
        candidates.reserve(D * 2);
        for (int ax = 0; ax < D; ++ax) {
            for (int delta : {-1, 1}) {
                vector<int> np = head;
                np[ax] += delta;
                if (!inBounds(np, bounds)) continue;
                if (blocked.find(np) != blocked.end()) continue;
                candidates.push_back({ax, delta});
            }
        }

        if (candidates.empty()) return {-1, 0};

        if (!food.empty()) {
            int bestScore = numeric_limits<int>::max();
            pair<int,int> best = candidates[0];
            for (auto [ax, delta] : candidates) {
                vector<int> np = head;
                np[ax] += delta;
                int nearest = numeric_limits<int>::max();
                for (const auto& f : food) nearest = min(nearest, manhattan(np, f));
                int score = nearest * 10;
                if (isFood(np)) score -= 1000;
                if (score < bestScore) {
                    bestScore = score;
                    best = {ax, delta};
                }
            }
            return best;
        }

        auto centerScore = [&](const vector<int>& p) {
            int s = 0;
            for (int i = 0; i < D; ++i) {
                int mid = (bounds[i] - 1) / 2;
                s += abs(p[i] - mid);
            }
            return s;
        };

        int bestScore = numeric_limits<int>::max();
        pair<int,int> best = candidates[0];
        for (auto [ax, delta] : candidates) {
            vector<int> np = head;
            np[ax] += delta;
            int score = centerScore(np);
            if (score < bestScore) {
                bestScore = score;
                best = {ax, delta};
            }
        }
        return best;
    };

    for (int turn = 0; turn < max_turns; ++turn) {
        auto [ax, delta] = chooseMove();
        if (ax < 0) break;
        cout << ax << ' ' << delta << '\n';
        cout.flush();

        head[ax] += delta;
        bool ate = false;
        if (isFood(head)) {
            ate = true;
            removeFood(head);
        }

        body.insert(body.begin(), head);
        blocked.insert(head);
        if (!ate && !body.empty()) {
            vector<int> tail = body.back();
            body.pop_back();
            blocked.erase(tail);
        }

        if (food.empty()) break;
    }

    return 0;
}
