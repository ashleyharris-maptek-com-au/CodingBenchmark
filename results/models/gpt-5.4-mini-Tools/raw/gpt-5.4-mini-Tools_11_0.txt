use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }
}

fn cross(a: Pt, b: Pt) -> f64 {
    a.x * b.y - a.y * b.x
}

fn sub(a: Pt, b: Pt) -> Pt {
    Pt::new(a.x - b.x, a.y - b.y)
}

fn add(a: Pt, b: Pt) -> Pt {
    Pt::new(a.x + b.x, a.y + b.y)
}

fn rotate(p: Pt, ang: f64) -> Pt {
    let (s, c) = ang.sin_cos();
    Pt::new(p.x * c - p.y * s, p.x * s + p.y * c)
}

fn poly_area(poly: &[Pt]) -> f64 {
    let n = poly.len();
    let mut s = 0.0;
    for i in 0..n {
        s += cross(poly[i], poly[(i + 1) % n]);
    }
    s * 0.5
}

fn point_in_poly(pt: Pt, poly: &[Pt]) -> bool {
    let mut inside = false;
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        let cond = ((a.y > pt.y) != (b.y > pt.y))
            && (pt.x < (b.x - a.x) * (pt.y - a.y) / (b.y - a.y + 1e-18) + a.x);
        if cond {
            inside = !inside;
        }
    }
    inside
}

fn bbox(poly: &[Pt]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for p in poly {
        minx = minx.min(p.x);
        miny = miny.min(p.y);
        maxx = maxx.max(p.x);
        maxy = maxy.max(p.y);
    }
    (minx, miny, maxx, maxy)
}

fn normalize(poly: &[Pt]) -> Vec<Pt> {
    let a = poly_area(poly);
    if a < 0.0 {
        let mut rev = poly.to_vec();
        rev.reverse();
        rev
    } else {
        poly.to_vec()
    }
}

fn edge_angles(poly: &[Pt]) -> Vec<f64> {
    let mut angs = Vec::new();
    let n = poly.len();
    for i in 0..n {
        let e = sub(poly[(i + 1) % n], poly[i]);
        let ang = e.y.atan2(e.x);
        angs.push(ang.rem_euclid(PI));
    }
    angs.sort_by(|a, b| a.partial_cmp(b).unwrap());
    angs.dedup_by(|a, b| (*a - *b).abs() < 1e-10);
    angs
}

fn best_rotation_for_piece(piece: &[Pt]) -> f64 {
    let mut candidates = vec![0.0, PI / 2.0];
    for a in edge_angles(piece) {
        candidates.push(a);
        candidates.push((a + PI / 2.0).rem_euclid(PI));
    }
    candidates.sort_by(|a, b| a.partial_cmp(b).unwrap());
    candidates.dedup_by(|a, b| (*a - *b).abs() < 1e-10);

    let mut best = 0.0;
    let mut best_score = f64::INFINITY;
    for &ang in &candidates {
        let rotated: Vec<Pt> = piece.iter().copied().map(|p| rotate(p, ang)).collect();
        let (minx, miny, maxx, maxy) = bbox(&rotated);
        let score = (maxx - minx) * (maxy - miny);
        if score < best_score {
            best_score = score;
            best = ang;
        }
    }
    best
}

fn translate_to_place(piece: &[Pt], stock: &[Pt], rot: f64, dx: f64, dy: f64) -> (f64, f64) {
    let rotated: Vec<Pt> = piece.iter().copied().map(|p| rotate(p, rot)).collect();
    let (pminx, pminy, _, _) = bbox(&rotated);
    let mut tx = dx - pminx;
    let mut ty = dy - pminy;
    let (sminx, sminy, smaxx, smaxy) = bbox(stock);
    let (pminx2, pminy2, pmaxx2, pmaxy2) = bbox(&rotated);
    let w = pmaxx2 - pminx2;
    let h = pmaxy2 - pminy2;
    if tx < sminx - pminx2 {
        tx = sminx - pminx2;
    }
    if ty < sminy - pminy2 {
        ty = sminy - pminy2;
    }
    if tx + w > smaxx {
        tx = smaxx - w - pminx2;
    }
    if ty + h > smaxy {
        ty = smaxy - h - pminy2;
    }
    (tx, ty)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let sv: usize = it.next().unwrap().parse().unwrap();
    let np: usize = it.next().unwrap().parse().unwrap();

    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock.push(Pt::new(x, y));
    }
    stock = normalize(&stock);

    let mut pieces: Vec<Vec<Pt>> = Vec::with_capacity(np);
    for _ in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut poly = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            poly.push(Pt::new(x, y));
        }
        pieces.push(normalize(&poly));
    }

    let (sminx, sminy, smaxx, smaxy) = bbox(&stock);
    let sw = smaxx - sminx;
    let sh = smaxy - sminy;
    let mut placements = Vec::with_capacity(np);
    let mut cur_x = sminx;
    let mut cur_y = sminy;
    let mut row_h = 0.0;
    let mut stock_idx = 0usize;

    for piece in &pieces {
        let rot = best_rotation_for_piece(piece);
        let rotated: Vec<Pt> = piece.iter().copied().map(|p| rotate(p, rot)).collect();
        let (pminx, pminy, pmaxx, pmaxy) = bbox(&rotated);
        let pw = pmaxx - pminx;
        let ph = pmaxy - pminy;

        if cur_x + pw > sminx + sw {
            cur_x = sminx;
            cur_y += row_h;
            row_h = 0.0;
        }
        if cur_y + ph > sminy + sh {
            stock_idx += 1;
            cur_x = sminx;
            cur_y = sminy;
            row_h = 0.0;
        }

        let tx = cur_x - pminx;
        let ty = cur_y - pminy;
        placements.push((stock_idx, tx, ty, rot));
        cur_x += pw;
        row_h = row_h.max(ph);
    }

    println!("{}", stock_idx + 1);
    for (sid, x, y, r) in placements {
        println!("{} {:.9} {:.9} {:.9}", sid, x, y, r);
    }
}
