declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number; priority: number };

const grid: string[] = data.grid || [];
const robots: Robot[] = data.robots || [];
const jobs: Job[] = data.jobs || [];

const h = grid.length;
const w = h > 0 ? grid[0].length : 0;

function cellIndex(x: number, y: number): number {
  return y * w + x;
}

function inBounds(x: number, y: number): boolean {
  return x >= 0 && y >= 0 && x < w && y < h;
}

function isOpen(x: number, y: number): boolean {
  return inBounds(x, y) && grid[y][x] === '.';
}

const dirs = [1, 0, -1, 0, 1];

function bfsFrom(sx: number, sy: number): Int32Array {
  const dist = new Int32Array(w * h);
  dist.fill(-1);
  if (!isOpen(sx, sy)) return dist;
  const qx = new Int32Array(w * h);
  const qy = new Int32Array(w * h);
  let head = 0;
  let tail = 0;
  qx[tail] = sx;
  qy[tail] = sy;
  tail++;
  dist[cellIndex(sx, sy)] = 0;
  while (head < tail) {
    const x = qx[head];
    const y = qy[head];
    head++;
    const base = dist[cellIndex(x, y)] + 1;
    for (let k = 0; k < 4; k++) {
      const nx = x + dirs[k];
      const ny = y + dirs[k + 1];
      if (!isOpen(nx, ny)) continue;
      const idx = cellIndex(nx, ny);
      if (dist[idx] !== -1) continue;
      dist[idx] = base;
      qx[tail] = nx;
      qy[tail] = ny;
      tail++;
    }
  }
  return dist;
}

const allTargets: { x: number; y: number }[] = [];
for (const r of robots) allTargets.push({ x: r.x, y: r.y });
for (const j of jobs) allTargets.push({ x: j.x, y: j.y });

const distCache = new Map<string, Int32Array>();
function getDist(x: number, y: number): Int32Array {
  const key = `${x},${y}`;
  let d = distCache.get(key);
  if (!d) {
    d = bfsFrom(x, y);
    distCache.set(key, d);
  }
  return d;
}

const jobOrder = jobs
  .map((j, idx) => ({ ...j, idx }))
  .sort((a, b) => b.priority - a.priority || a.id - b.id || a.idx - b.idx);

const assignments: number[][] = Array.from({ length: robots.length }, () => []);
const robotPos = robots.map(r => ({ x: r.x, y: r.y }));
const robotLastDist = robots.map(r => getDist(r.x, r.y));

function distBetween(i: number, x: number, y: number): number {
  const d = robotLastDist[i];
  if (d) {
    const idx = cellIndex(x, y);
    if (idx >= 0 && idx < d.length) return d[idx];
  }
  return -1;
}

for (const job of jobOrder) {
  let bestRobot = 0;
  let bestScore = Number.POSITIVE_INFINITY;
  for (let i = 0; i < robots.length; i++) {
    const d = distBetween(i, job.x, job.y);
    const travel = d >= 0 ? d : 1e9;
    const tie = assignments[i].length;
    const score = travel * 1000000 + tie * 1000 + i;
    if (score < bestScore) {
      bestScore = score;
      bestRobot = i;
    }
  }
  assignments[bestRobot].push(job.id);
  robotPos[bestRobot] = { x: job.x, y: job.y };
  robotLastDist[bestRobot] = getDist(job.x, job.y);
}

process.stdout.write(JSON.stringify({ assignments }));
