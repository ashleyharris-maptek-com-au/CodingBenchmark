const fs = require('fs');

function dist(a, b) {
  const dx = a.x - b.x;
  const dy = a.y - b.y;
  return Math.hypot(dx, dy);
}

function buildRoutes(data) {
  const depot = { x: data.depot[0], y: data.depot[1] };
  const packages = (data.packages || []).map(p => ({ id: p.id, x: p.x, y: p.y }));
  const n = packages.length;
  const k = Math.max(1, Math.min(data.truckCount || 1, n || 1));
  if (n === 0) return [];

  // Start with angular sweep around the depot to create spatially coherent buckets.
  const items = packages.map((p, idx) => ({
    ...p,
    idx,
    angle: Math.atan2(p.y - depot.y, p.x - depot.x),
    radius: dist(p, depot),
  }));
  items.sort((a, b) => a.angle - b.angle || a.radius - b.radius);

  // Balance route sizes while keeping contiguous angular neighborhoods together.
  const routeCount = Math.min(k, n);
  const base = Math.floor(n / routeCount);
  let rem = n % routeCount;
  const buckets = [];
  let pos = 0;
  for (let r = 0; r < routeCount; r++) {
    const size = base + (rem > 0 ? 1 : 0);
    if (rem > 0) rem--;
    buckets.push(items.slice(pos, pos + size));
    pos += size;
  }

  // Improve each bucket with a simple nearest-neighbor order, then 2-opt cleanup.
  function routeLength(route) {
    let total = 0;
    let prev = depot;
    for (const p of route) {
      total += dist(prev, p);
      prev = p;
    }
    total += dist(prev, depot);
    return total;
  }

  function nnOrder(points) {
    const remaining = points.slice();
    const route = [];
    let cur = depot;
    while (remaining.length) {
      let best = 0;
      let bestD = Infinity;
      for (let i = 0; i < remaining.length; i++) {
        const d = dist(cur, remaining[i]);
        if (d < bestD) {
          bestD = d;
          best = i;
        }
      }
      const next = remaining.splice(best, 1)[0];
      route.push(next);
      cur = next;
    }
    return route;
  }

  function twoOpt(route) {
    if (route.length < 4) return route;
    let improved = true;
    while (improved) {
      improved = false;
      for (let i = 0; i < route.length - 2; i++) {
        const a = i === 0 ? depot : route[i - 1];
        const b = route[i];
        for (let j = i + 1; j < route.length - 1; j++) {
          const c = route[j];
          const d = j === route.length - 1 ? depot : route[j + 1];
          const current = dist(a, b) + dist(c, d);
          const swapped = dist(a, c) + dist(b, d);
          if (swapped + 1e-12 < current) {
            for (let l = i, r = j; l < r; l++, r--) {
              const tmp = route[l];
              route[l] = route[r];
              route[r] = tmp;
            }
            improved = true;
          }
        }
      }
    }
    return route;
  }

  const routes = buckets.map(bucket => {
    const ordered = twoOpt(nnOrder(bucket));
    return ordered.map(p => p.id);
  });

  // If truckCount exceeds package count, the spec allows up to max routes; we keep only non-empty routes.
  return routes.filter(r => r.length > 0);
}

function main() {
  const input = fs.readFileSync(0, 'utf8');
  const data = JSON.parse(input);
  const routes = buildRoutes(data);
  process.stdout.write(JSON.stringify({ routes }));
}

main();
