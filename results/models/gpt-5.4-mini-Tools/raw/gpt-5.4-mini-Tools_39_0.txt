use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();
    let mut adj = vec![0u64; n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        adj[u] |= 1u64 << v;
        adj[v] |= 1u64 << u;
    }

    let half = n / 2;
    let all_mask: u64 = if n == 64 { u64::MAX } else { (1u64 << n) - 1 };
    let mut best_cut = usize::MAX;
    let mut best_mask = 0u64;

    if n <= 24 {
        let limit = 1u64 << n;
        for mask in 0u64..limit {
            if mask.count_ones() as usize != half { continue; }
            let mut cut = 0usize;
            for u in 0..n {
                let in_a = (mask >> u) & 1 == 1;
                let neighbors = adj[u] & all_mask;
                let cross = (neighbors & mask).count_ones() as usize;
                if in_a {
                    cut += ((neighbors & (!mask & all_mask)).count_ones()) as usize;
                } else {
                    cut += cross;
                }
            }
            cut /= 2;
            if cut < best_cut {
                best_cut = cut;
                best_mask = mask;
            }
        }
    } else {
        // For larger instances, use a simple heuristic local search and report a valid bisection.
        // This fallback keeps the program complete, but exact optimality is only guaranteed
        // for small n where exhaustive search is feasible.
        let mut mask = (1u64 << half) - 1;
        best_mask = mask;
        let mut improved = true;
        while improved {
            improved = false;
            let mut current_cut = 0usize;
            for u in 0..n {
                let in_a = ((mask >> u) & 1) == 1;
                let diff = if in_a { (!mask) & all_mask } else { mask };
                current_cut += (adj[u] & diff).count_ones() as usize;
            }
            current_cut /= 2;
            best_cut = current_cut;
            for i in 0..n {
                let bi = ((mask >> i) & 1) == 1;
                for j in 0..n {
                    let bj = ((mask >> j) & 1) == 1;
                    if bi && !bj {
                        let new_mask = mask ^ (1u64 << i) ^ (1u64 << j);
                        let mut cut = 0usize;
                        for u in 0..n {
                            let in_a = ((new_mask >> u) & 1) == 1;
                            let diff = if in_a { (!new_mask) & all_mask } else { new_mask };
                            cut += (adj[u] & diff).count_ones() as usize;
                        }
                        cut /= 2;
                        if cut < best_cut {
                            best_cut = cut;
                            best_mask = new_mask;
                            mask = new_mask;
                            improved = true;
                            break;
                        }
                    }
                }
                if improved { break; }
            }
        }
    }

    let mut part = Vec::new();
    for v in 0..n {
        if (best_mask >> v) & 1 == 1 {
            part.push(v);
        }
    }
    if part.len() != half {
        part.clear();
        for v in 0..half { part.push(v); }
        best_cut = 0;
        for &u in &part {
            for v in 0..n {
                if !part.contains(&v) && ((adj[u] >> v) & 1 == 1) {
                    best_cut += 1;
                }
            }
        }
    }

    println!("{}", best_cut);
    for (i, v) in part.iter().enumerate() {
        if i > 0 { print!(" "); }
        print!("{}", v);
    }
    println!();
}
