float4 main(PSInput input) : SV_Target0
{
    float2 uv = input.uv;
    float2 brickUV = uv * float2(8.0, 16.0);

    float row = floor(brickUV.y);
    float2 cell = frac(brickUV);

    if (fmod(row, 2.0) >= 1.0)
    {
        cell.x = frac(cell.x + 0.5);
    }

    float mortarX = step(cell.x, 0.05) + step(1.0 - 0.05, cell.x);
    float mortarY = step(cell.y, 0.1) + step(1.0 - 0.1, cell.y);
    float mortar = saturate(mortarX + mortarY);

    float3 albedo = lerp(float3(0.7, 0.2, 0.1), float3(0.7, 0.7, 0.7), mortar);

    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shade = 0.4 + 0.6 * diffuse;

    return float4(albedo * shade, 1.0);
}
