#include <algorithm>
#include <array>
#include <iostream>
#include <string>
#include <vector>

using namespace std;

struct SuffixAutomaton {
    struct State {
        int link = -1;
        int len = 0;
        int first_pos = -1;
        array<int, 256> next;
        State() { next.fill(-1); }
    };

    vector<State> st;
    int last = 0;

    explicit SuffixAutomaton(size_t reserve_states = 0) {
        st.reserve(max<size_t>(2, reserve_states));
        st.push_back(State());
    }

    void extend(unsigned char c, int pos) {
        int cur = (int)st.size();
        st.push_back(State());
        st[cur].len = st[last].len + 1;
        st[cur].first_pos = pos;

        int p = last;
        while (p != -1 && st[p].next[c] == -1) {
            st[p].next[c] = cur;
            p = st[p].link;
        }
        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = st[p].next[c];
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = (int)st.size();
                st.push_back(st[q]);
                st[clone].len = st[p].len + 1;
                while (p != -1 && st[p].next[c] == q) {
                    st[p].next[c] = clone;
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }
};

static string longest_common_substring(const vector<string>& s) {
    if (s.empty()) return "";

    int anchor = 0;
    for (int i = 1; i < (int)s.size(); ++i) {
        if (s[i].size() < s[anchor].size()) anchor = i;
    }

    const string& base = s[anchor];
    SuffixAutomaton sa(base.size() * 2 + 1);
    for (int i = 0; i < (int)base.size(); ++i) sa.extend((unsigned char)base[i], i);

    vector<int> best(sa.st.size(), 0), cur(sa.st.size(), 0), order(sa.st.size());
    for (int i = 0; i < (int)sa.st.size(); ++i) order[i] = i;
    sort(order.begin(), order.end(), [&](int a, int b) { return sa.st[a].len > sa.st[b].len; });

    for (int idx = 0; idx < (int)s.size(); ++idx) {
        if (idx == anchor) continue;
        fill(cur.begin(), cur.end(), 0);

        int v = 0, l = 0;
        for (unsigned char c : s[idx]) {
            while (v && sa.st[v].next[c] == -1) {
                v = sa.st[v].link;
                l = sa.st[v].len;
            }
            if (sa.st[v].next[c] != -1) {
                v = sa.st[v].next[c];
                ++l;
            } else {
                v = 0;
                l = 0;
            }
            if (cur[v] < l) cur[v] = l;
        }

        for (int vtx : order) {
            int p = sa.st[vtx].link;
            if (p != -1) {
                int propagated = min(cur[vtx], sa.st[p].len);
                if (cur[p] < propagated) cur[p] = propagated;
            }
        }

        for (size_t i = 0; i < best.size(); ++i) {
            best[i] = min(best[i] == 0 ? cur[i] : best[i], cur[i]);
        }
    }

    int ans_len = 0;
    int ans_state = 0;
    for (int i = 0; i < (int)sa.st.size(); ++i) {
        if (best[i] > ans_len) {
            ans_len = best[i];
            ans_state = i;
        }
    }
    if (ans_len == 0) return "";

    int end_pos = sa.st[ans_state].first_pos;
    return base.substr(end_pos - ans_len + 1, ans_len);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    string dummy;
    getline(cin, dummy);

    vector<string> strings(N);
    for (int i = 0; i < N; ++i) getline(cin, strings[i]);

    cout << longest_common_substring(strings) << '\n';
    return 0;
}
