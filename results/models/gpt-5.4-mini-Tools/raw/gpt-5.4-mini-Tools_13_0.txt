#include <algorithm>
#include <iostream>
#include <limits>
#include <string>
#include <unordered_map>
#include <vector>

using namespace std;

struct SuffixAutomaton {
    struct State {
        int link = -1;
        int len = 0;
        unordered_map<char, int> next;
    };

    vector<State> st;
    int last = 0;

    explicit SuffixAutomaton(size_t reserve_states = 0) {
        st.reserve(max<size_t>(2, reserve_states));
        st.push_back(State{});
    }

    void extend(char c) {
        int cur = (int)st.size();
        st.push_back(State{});
        st[cur].len = st[last].len + 1;

        int p = last;
        while (p != -1 && !st[p].next.count(c)) {
            st[p].next[c] = cur;
            p = st[p].link;
        }

        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = st[p].next[c];
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = (int)st.size();
                st.push_back(st[q]);
                st[clone].len = st[p].len + 1;
                while (p != -1 && st[p].next[c] == q) {
                    st[p].next[c] = clone;
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }
};

static bool containsSubstring(const string& s, const string& t) {
    if (t.empty()) return true;
    return s.find(t) != string::npos;
}

static string lcsTwo(const string& a, const string& b) {
    if (a.empty() || b.empty()) return "";
    vector<int> prev(b.size() + 1, 0), cur(b.size() + 1, 0);
    size_t bestLen = 0;
    size_t bestEnd = 0;
    for (size_t i = 1; i <= a.size(); ++i) {
        for (size_t j = 1; j <= b.size(); ++j) {
            if (a[i - 1] == b[j - 1]) {
                cur[j] = prev[j - 1] + 1;
                if ((size_t)cur[j] > bestLen) {
                    bestLen = (size_t)cur[j];
                    bestEnd = i;
                }
            } else {
                cur[j] = 0;
            }
        }
        swap(prev, cur);
    }
    return a.substr(bestEnd - bestLen, bestLen);
}

static string bruteForceSmall(const vector<string>& s) {
    string base = s[0];
    string best;
    for (size_t start = 0; start < base.size(); ++start) {
        for (size_t len = base.size() - start; len > 0; --len) {
            if (len <= best.size()) break;
            string cand = base.substr(start, len);
            bool ok = true;
            for (size_t i = 1; i < s.size(); ++i) {
                if (!containsSubstring(s[i], cand)) {
                    ok = false;
                    break;
                }
            }
            if (ok) {
                best = move(cand);
                break;
            }
        }
    }
    return best;
}

static string lcsManyViaSAM(const vector<string>& s) {
    size_t minIndex = 0;
    for (size_t i = 1; i < s.size(); ++i) {
        if (s[i].size() < s[minIndex].size()) minIndex = i;
    }

    string base = s[minIndex];
    if (base.empty()) return "";

    SuffixAutomaton sam(base.size() * 2 + 1);
    for (char c : base) sam.extend(c);

    vector<int> best(sam.st.size(), numeric_limits<int>::max());
    for (size_t idx = 0; idx < s.size(); ++idx) {
        if (idx == minIndex) continue;
        const string& t = s[idx];
        vector<int> cur(sam.st.size(), 0);
        int v = 0;
        int l = 0;
        for (char c : t) {
            while (v && !sam.st[v].next.count(c)) {
                v = sam.st[v].link;
                l = sam.st[v].len;
            }
            auto it = sam.st[v].next.find(c);
            if (it != sam.st[v].next.end()) {
                v = it->second;
                ++l;
            } else {
                v = 0;
                l = 0;
            }
            cur[v] = max(cur[v], l);
        }

        vector<int> order(sam.st.size());
        for (size_t i = 0; i < order.size(); ++i) order[i] = (int)i;
        sort(order.begin(), order.end(), [&](int x, int y) { return sam.st[x].len > sam.st[y].len; });
        for (int vtx : order) {
            int p = sam.st[vtx].link;
            if (p >= 0) cur[p] = max(cur[p], min(cur[vtx], sam.st[p].len));
        }
        for (size_t i = 0; i < sam.st.size(); ++i) best[i] = min(best[i], cur[i]);
    }

    int state = 0;
    int length = 0;
    for (size_t i = 0; i < sam.st.size(); ++i) {
        if (best[i] > length) {
            length = best[i];
            state = (int)i;
        }
    }
    if (length <= 0) return "";

    int need = length;
    int v = state;
    while (v != 0 && sam.st[sam.st[v].link].len >= need) v = sam.st[v].link;

    string candidate = base.substr(0, 0);
    for (size_t start = 0; start + (size_t)length <= base.size(); ++start) {
        string cand = base.substr(start, length);
        bool ok = true;
        for (const string& t : s) {
            if (!containsSubstring(t, cand)) {
                ok = false;
                break;
            }
        }
        if (ok) return cand;
    }
    return "";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    string dummy;
    getline(cin, dummy);

    vector<string> s(N);
    size_t totalLen = 0;
    for (int i = 0; i < N; ++i) {
        getline(cin, s[i]);
        totalLen += s[i].size();
    }

    if (N == 0) {
        cout << '\n';
        return 0;
    }
    for (const auto& x : s) {
        if (x.empty()) {
            cout << '\n';
            return 0;
        }
    }

    string ans;
    if (N <= 3 && totalLen <= 1000) {
        ans = bruteForceSmall(s);
    } else if (N == 2) {
        ans = lcsTwo(s[0], s[1]);
    } else {
        ans = lcsManyViaSAM(s);
    }

    cout << ans << '\n';
    return 0;
}
