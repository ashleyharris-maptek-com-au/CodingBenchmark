%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%bool = OpTypeBool

%float_0 = OpConstant %float 0
%float_025 = OpConstant %float 0.025
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v3 = OpTypePointer Output %v3float
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%outColor = OpVariable %ptr_out_v3 Output

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_ubo Uniform

%main = OpFunction %void None %func
%label = OpLabel
%prev_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%start_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%end_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%next_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0

%prev = OpLoad %v4float %prev_ptr
%start = OpLoad %v4float %start_ptr
%end = OpLoad %v4float %end_ptr
%next = OpLoad %v4float %next_ptr

%d0 = OpFSub %v4float %end %prev
%d1 = OpFSub %v4float %next %start
%t0 = OpExtInst %v4float %1 Normalize %d0
%t1 = OpExtInst %v4float %1 Normalize %d1
%t0xy = OpVectorShuffle %v2float %t0 %t0 0 1
%t1xy = OpVectorShuffle %v2float %t1 %t1 0 1
%t0x = OpCompositeExtract %float %t0xy 0
%t0y = OpCompositeExtract %float %t0xy 1
%t1x = OpCompositeExtract %float %t1xy 0
%t1y = OpCompositeExtract %float %t1xy 1
%perp0 = OpCompositeConstruct %v2float %t0y %t0x
%perp1 = OpCompositeConstruct %v2float %t1y %t1x
%perp0n = OpCompositeConstruct %v2float %float_0 %float_0
%perp1n = OpCompositeConstruct %v2float %float_0 %float_0
%perp0n = OpFSub %v2float %perp0n %perp0
%perp1n = OpFSub %v2float %perp1n %perp1
%thick0 = OpVectorTimesScalar %v2float %perp0n %float_025
%thick1 = OpVectorTimesScalar %v2float %perp1n %float_025

%start_xy = OpVectorShuffle %v2float %start %start 0 1
%end_xy = OpVectorShuffle %v2float %end %end 0 1
%start_pos = OpFAdd %v2float %start_xy %thick0
%start_neg = OpFSub %v2float %start_xy %thick0
%end_pos = OpFAdd %v2float %end_xy %thick1
%end_neg = OpFSub %v2float %end_xy %thick1

%start_z = OpCompositeExtract %float %start 2
%start_w = OpCompositeExtract %float %start 3
%end_z = OpCompositeExtract %float %end 2
%end_w = OpCompositeExtract %float %end 3

%color1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%color2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%color1 = OpLoad %v3float %color1_ptr
%color2 = OpLoad %v3float %color2_ptr

%v0 = OpCompositeExtract %float %start_pos 0
%v1 = OpCompositeExtract %float %start_pos 1
%p0 = OpCompositeConstruct %v4float %v0 %v1 %start_z %start_w
OpStore %outColor %color1
%out_ptr0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_ptr0 %p0
OpEmitVertex

%v2 = OpCompositeExtract %float %start_neg 0
%v3 = OpCompositeExtract %float %start_neg 1
%p1 = OpCompositeConstruct %v4float %v2 %v3 %start_z %start_w
OpStore %outColor %color1
%out_ptr1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_ptr1 %p1
OpEmitVertex

%v4 = OpCompositeExtract %float %end_pos 0
%v5 = OpCompositeExtract %float %end_pos 1
%p2 = OpCompositeConstruct %v4float %v4 %v5 %end_z %end_w
OpStore %outColor %color2
%out_ptr2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_ptr2 %p2
OpEmitVertex

%v6 = OpCompositeExtract %float %end_neg 0
%v7 = OpCompositeExtract %float %end_neg 1
%p3 = OpCompositeConstruct %v4float %v6 %v7 %end_z %end_w
OpStore %outColor %color2
%out_ptr3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_ptr3 %p3
OpEmitVertex

OpEndPrimitive
