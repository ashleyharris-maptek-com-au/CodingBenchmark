using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public class Program
{
    private struct Voxel
    {
        public int X, Y, Z;
        public byte Data;
        public int Mass;
        public int Value;
        public int Score;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buf = new byte[1 << 20];
        private int _len, _ptr;

        private byte ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do c = ReadByte(); while (c <= 32 && c != 0);
            int sign = 1;
            if (c == '-') { sign = -1; c = ReadByte(); }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return val * sign;
        }

        public double NextDouble()
        {
            return double.Parse(NextToken(), System.Globalization.CultureInfo.InvariantCulture);
        }

        private string NextToken()
        {
            byte c;
            do c = ReadByte(); while (c <= 32 && c != 0);
            var sb = new StringBuilder();
            while (c > 32)
            {
                sb.Append((char)c);
                c = ReadByte();
            }
            return sb.ToString();
        }

        public int ReadByteValue() => ReadByte();
    }

    static void Main()
    {
        var fs = new FastScanner();
        int xCount = fs.NextInt();
        int yCount = fs.NextInt();
        int zCount = fs.NextInt();
        double xBlock = fs.NextDouble();
        double yBlock = fs.NextDouble();
        double zBlock = fs.NextDouble();

        long total = (long)xCount * yCount * zCount;
        var ore = new List<Voxel>(Math.Min((int)Math.Min(total, 2000000L), 2000000));
        for (int x = 0; x < xCount; x++)
        for (int y = 0; y < yCount; y++)
        for (int z = 0; z < zCount; z++)
        {
            int d = fs.ReadByteValue();
            int mass = d & 15;
            int val = d >> 4;
            if (mass == 0 || val == 0) continue;
            int score = mass * val;
            ore.Add(new Voxel { X = x, Y = y, Z = z, Data = (byte)d, Mass = mass, Value = val, Score = score });
        }

        ore.Sort((a, b) => b.Score.CompareTo(a.Score));

        bool large = total > 250000;
        int limit = large ? Math.Min(ore.Count, 5000) : ore.Count;

        var ops = new List<string>(limit + 8);

        for (int i = 0; i < limit; i++)
        {
            var v = ore[i];
            if (v.Value <= 0 || v.Mass <= 0) continue;

            int dx = v.X;
            int dy = v.Y;
            int dz = v.Z;
            int steps = dx + dy;

            var sb = new StringBuilder();
            sb.Append(1);
            sb.Append(' ').Append(dx).Append(' ').Append(dy).Append(' ').Append(dz);
            int cx = 0, cy = 0;
            for (int x = 1; x <= dx; x++)
            {
                sb.Append(' ').Append(x).Append(' ').Append(cy);
                cx = x;
            }
            for (int y = 1; y <= dy; y++)
            {
                sb.Append(' ').Append(cx).Append(' ').Append(y);
                cy = y;
            }
            sb.Append(' ').Append(0).Append(' ').Append(0);
            ops.Add(sb.ToString());
        }

        var output = new StringBuilder();
        output.AppendLine(ops.Count.ToString());
        for (int i = 0; i < ops.Count; i++) output.AppendLine(ops[i]);
        Console.Write(output.ToString());
    }
}
