OpCapability Shader
OpExtInstImport %1 "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %ubo
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %outColor "outColor"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %uv "uv"
OpName %tangent "tangent"
OpName %color "color"
OpName %ubo "ubo"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4

%void = OpTypeVoid
%functype = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%bool = OpTypeBool
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%const_0 = OpConstant %float 0.0
%const_05 = OpConstant %float 0.5
%const_1 = OpConstant %float 1.0
%const_2 = OpConstant %float 2.0
%const_int_4 = OpConstant %int 4

%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%uv = OpVariable %_ptr_Input_v2float Input
%tangent = OpVariable %_ptr_Input_v3float Input
%color = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%main = OpFunction %void None %functype
%entry = OpLabel
%pos = OpLoad %v3float %worldPos
%nrm = OpLoad %v3float %normal
%cam_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %const_int_4
%cam = OpLoad %v4float %cam_ptr
%cam_xyz = OpVectorShuffle %v3float %cam %cam 0 1 2
%view_dir_raw = OpFSub %v3float %cam_xyz %pos
%view_dir = OpExtInst %v3float %1 Normalize %view_dir_raw
%norm = OpExtInst %v3float %1 Normalize %nrm
%ndotv_raw = OpDot %float %norm %view_dir
%ndotv = OpExtInst %float %1 FMax %ndotv_raw %const_0
%cmp = OpFOrdLessThan %bool %ndotv %const_05
%g_lo = OpFMul %float %ndotv %const_2
%t = OpFSub %float %ndotv %const_05
%t2 = OpFMul %float %t %const_2
%rg_hi = OpFSub %float %const_1 %t2
%select_r = OpSelect %float %cmp %const_1 %rg_hi
%select_g = OpSelect %float %cmp %g_lo %rg_hi
%select_b = OpSelect %float %cmp %const_0 %t2
%col = OpCompositeConstruct %v4float %select_r %select_g %select_b %const_1
OpStore %outColor %col
OpReturn
OpFunctionEnd
