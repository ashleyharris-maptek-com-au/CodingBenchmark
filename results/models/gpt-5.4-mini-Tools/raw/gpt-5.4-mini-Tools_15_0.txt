use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Pt {
    x: f64,
    y: f64,
}

fn cross(a: Pt, b: Pt, c: Pt) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn area(poly: &[Pt]) -> f64 {
    let n = poly.len();
    let mut s = 0.0;
    for i in 0..n {
        let p = poly[i];
        let q = poly[(i + 1) % n];
        s += p.x * q.y - p.y * q.x;
    }
    0.5 * s
}

fn point_in_tri(a: Pt, b: Pt, c: Pt, p: Pt) -> bool {
    let c1 = cross(a, b, p);
    let c2 = cross(b, c, p);
    let c3 = cross(c, a, p);
    let has_neg = c1 < -1e-12 || c2 < -1e-12 || c3 < -1e-12;
    let has_pos = c1 > 1e-12 || c2 > 1e-12 || c3 > 1e-12;
    !(has_neg && has_pos)
}

fn triangulate(poly: &[Pt]) -> Vec<[usize; 3]> {
    let n = poly.len();
    let mut idx: Vec<usize> = (0..n).collect();
    if area(poly) < 0.0 {
        idx.reverse();
    }
    let mut tris = Vec::new();
    let mut guard = 0usize;
    while idx.len() > 3 && guard < 10_000 {
        guard += 1;
        let m = idx.len();
        let mut ear_found = false;
        for i in 0..m {
            let i0 = idx[(i + m - 1) % m];
            let i1 = idx[i];
            let i2 = idx[(i + 1) % m];
            let a = poly[i0];
            let b = poly[i1];
            let c = poly[i2];
            if cross(a, b, c) <= 1e-12 {
                continue;
            }
            let mut ok = true;
            for &j in &idx {
                if j == i0 || j == i1 || j == i2 {
                    continue;
                }
                if point_in_tri(a, b, c, poly[j]) {
                    ok = false;
                    break;
                }
            }
            if ok {
                tris.push([i0, i1, i2]);
                idx.remove(i);
                ear_found = true;
                break;
            }
        }
        if !ear_found {
            break;
        }
    }
    if idx.len() == 3 {
        tris.push([idx[0], idx[1], idx[2]]);
    }
    tris
}

fn dot3(a: [f64; 3], b: [f64; 3]) -> f64 {
    a[0] * b[0] + a[1] * b[1] + a[2] * b[2]
}

fn norm3(v: [f64; 3]) -> f64 {
    dot3(v, v).sqrt()
}

fn normalize3(v: [f64; 3]) -> [f64; 3] {
    let n = norm3(v);
    [v[0] / n, v[1] / n, v[2] / n]
}

fn quat_from_to(a: [f64; 3], b: [f64; 3]) -> [f64; 4] {
    let v0 = normalize3(a);
    let v1 = normalize3(b);
    let d = dot3(v0, v1);
    if d > 0.999_999 {
        return [1.0, 0.0, 0.0, 0.0];
    }
    if d < -0.999_999 {
        let mut axis = if v0[0].abs() < 0.9 { [1.0, 0.0, 0.0] } else { [0.0, 1.0, 0.0] };
        let c = [
            v0[1] * axis[2] - v0[2] * axis[1],
            v0[2] * axis[0] - v0[0] * axis[2],
            v0[0] * axis[1] - v0[1] * axis[0],
        ];
        let n = norm3(c);
        axis = [c[0] / n, c[1] / n, c[2] / n];
        return [0.0, axis[0], axis[1], axis[2]];
    }
    let c = [
        v0[1] * v1[2] - v0[2] * v1[1],
        v0[2] * v1[0] - v0[0] * v1[2],
        v0[0] * v1[1] - v0[1] * v1[0],
    ];
    let s = (1.0 + d).sqrt() * 2.0;
    let q = [s * 0.5, c[0] / s, c[1] / s, c[2] / s];
    let n = (q[0] * q[0] + q[1] * q[1] + q[2] * q[2] + q[3] * q[3]).sqrt();
    [q[0] / n, q[1] / n, q[2] / n, q[3] / n]
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let nv: usize = it.next().unwrap().parse().unwrap();
    let mut poly = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        poly.push(Pt { x, y });
    }
    let sx: f64 = it.next().unwrap().parse().unwrap();
    let sy: f64 = it.next().unwrap().parse().unwrap();
    let sz: f64 = it.next().unwrap().parse().unwrap();
    let _sun = [sx, sy, sz];

    let tris = triangulate(&poly);
    let q = quat_from_to([0.0, 0.0, 1.0], [sx, sy, sz]);

    let mut out = String::new();
    out.push_str(&format!("{}\n", tris.len()));

    let mut z = 50.0;
    let dz = 20.0;
    for t in tris {
        let a = poly[t[0]];
        let b = poly[t[1]];
        let c = poly[t[2]];
        let cx = (a.x + b.x + c.x) / 3.0;
        let cy = (a.y + b.y + c.y) / 3.0;
        let scale = 3.0_f64.min(10.0);
        out.push_str(&format!(
            "{:.9} {:.9} {:.9} {:.9} {:.9} {:.9} {:.9} {:.9}\n",
            cx, cy, z, q[0], q[1], q[2], q[3], scale
        ));
        z += dz;
    }

    print!("{}", out);
}
