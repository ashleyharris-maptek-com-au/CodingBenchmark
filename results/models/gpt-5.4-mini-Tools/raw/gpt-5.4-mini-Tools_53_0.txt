from math import atan2, copysign, sqrt, sin, cos, pi

_state = {
    'initialized': False,
    'prev_rel': None,
    'prev_time': None,
    'bias_y': 0.0,
    'bias_z': 0.0,
    'bias_x': 0.0,
}

THRUST_LIMIT = 1.0
ATT_LIMIT = 0.035
G0 = 9.80665
MAX_THRUST_ACCEL = (4 * 220.0) / 5000.0  # conservative normalized accel scaling


def _clamp(v, lo, hi):
    return lo if v < lo else hi if v > hi else v


def _valid(state, key):
    return bool(state.get(key, False))


def _v(state, key, default=0.0):
    try:
        return float(state.get(key, default))
    except Exception:
        return default


def _mean_valid(state, value_keys, valid_keys, default=0.0):
    vals = []
    for vk, ok in zip(value_keys, valid_keys):
        if _valid(state, ok):
            vals.append(_v(state, vk, default))
    if vals:
        return sum(vals) / len(vals)
    return default


def _estimate_rel(state):
    # Prefer radar, then lidar, then camera/range fallbacks.
    candidates = []
    if _valid(state, 'radar_rel_x_valid') and _valid(state, 'radar_rel_y_valid') and _valid(state, 'radar_rel_z_valid'):
        candidates.append((
            _v(state, 'radar_rel_pos_x_m'),
            _v(state, 'radar_rel_pos_y_m'),
            _v(state, 'radar_rel_pos_z_m'),
            _v(state, 'radar_rel_vel_x_ms'),
            _v(state, 'radar_rel_vel_y_ms'),
            _v(state, 'radar_rel_vel_z_ms'),
            3.0,
        ))
    if _valid(state, 'lidar_rel_x_valid') and _valid(state, 'lidar_rel_y_valid') and _valid(state, 'lidar_rel_z_valid'):
        candidates.append((
            _v(state, 'lidar_rel_pos_x_m'),
            _v(state, 'lidar_rel_pos_y_m'),
            _v(state, 'lidar_rel_pos_z_m'),
            _v(state, 'lidar_rel_vel_x_ms'),
            _v(state, 'lidar_rel_vel_y_ms'),
            _v(state, 'lidar_rel_vel_z_ms'),
            2.0,
        ))

    if candidates:
        wx = wy = wz = vx = vy = vz = ws = 0.0
        for x, y, z, dx, dy, dz, w in candidates:
            wx += x * w; wy += y * w; wz += z * w
            vx += dx * w; vy += dy * w; vz += dz * w; ws += w
        if ws > 0:
            return wx / ws, wy / ws, wz / ws, vx / ws, vy / ws, vz / ws, 'rf'

    # Fallback from range and camera bearing if available.
    rng = None
    for k in ('range_1_m', 'range_2_m', 'range_3_m'):
        if _valid(state, k.replace('_m', '_valid')):
            val = _v(state, k)
            if val > 0:
                rng = val if rng is None else min(rng, val)
    bx = _v(state, 'dock_camera_bearing_x', 0.0) if _valid(state, 'dock_camera_bearing_x_valid') else 0.0
    by = _v(state, 'dock_camera_bearing_y', 0.0) if _valid(state, 'dock_camera_bearing_y_valid') else 0.0
    bz = _v(state, 'dock_camera_bearing_z', 0.0) if _valid(state, 'dock_camera_bearing_z_valid') else 0.0
    norm = sqrt(bx * bx + by * by + bz * bz)
    if rng is not None and norm > 1e-6:
        scale = rng / norm
        return bx * scale, by * scale, bz * scale, 0.0, 0.0, 0.0, 'cam'

    # Last resort: use GPS/ground data only for rough approach state, no relative vector.
    return 0.0, -100.0, 0.0, 0.0, 0.0, 0.0, 'none'


def _estimate_velocity_from_history(rel, dt):
    if _state['prev_rel'] is None or dt <= 0:
        return rel[3], rel[4], rel[5]
    px, py, pz, _, _, _ = _state['prev_rel']
    x, y, z, _, _, _ = rel
    return (x - px) / dt, (y - py) / dt, (z - pz) / dt


def _approach_profile(r, vy):
    dist = sqrt(r[0] * r[0] + r[1] * r[1] + r[2] * r[2])
    if dist > 200.0:
        return 0.22
    if dist > 50.0:
        return 0.14
    if dist > 10.0:
        return 0.07
    return 0.03


def autopilot_step(state: dict, dt: float) -> dict:
    global _state

    rel = _estimate_rel(state)
    x, y, z, vx, vy, vz, source = rel
    vx_h, vy_h, vz_h = _estimate_velocity_from_history(rel, dt)
    if abs(vx) < 1e-6 and abs(vy) < 1e-6 and abs(vz) < 1e-6 and source != 'none':
        vx, vy, vz = vx_h, vy_h, vz_h

    dist = sqrt(x * x + y * y + z * z)
    speed = sqrt(vx * vx + vy * vy + vz * vz)

    # Bias cancellation for small persistent uncommanded thrusts.
    if _state['initialized']:
        _state['bias_y'] = 0.995 * _state['bias_y'] + 0.005 * _clamp(-vy * 0.02, -0.04, 0.04)
        _state['bias_z'] = 0.995 * _state['bias_z'] + 0.005 * _clamp(-vz * 0.02, -0.04, 0.04)
        _state['bias_x'] = 0.995 * _state['bias_x'] + 0.005 * _clamp(-vx * 0.02, -0.04, 0.04)
    else:
        _state['initialized'] = True

    # Desired closing speed shrinks as we get close.
    if dist > 200.0:
        target_speed = 0.8
    elif dist > 50.0:
        target_speed = 0.35
    elif dist > 10.0:
        target_speed = 0.12
    else:
        target_speed = 0.05

    # Command along-track primarily; use radial/cross-track damped PD.
    # Negative y means behind station, so prograde y thrust is usually wrong for closing;
    # we bias toward retrograde when behind and toward braking near rendezvous.
    y_err = y
    x_err = x
    z_err = z

    # In orbital rendezvous, to reduce along-track separation from behind we typically need retrograde.
    if y_err < -20.0:
        y_cmd = -0.22
    elif y_err < -5.0:
        y_cmd = -0.14
    elif y_err > 20.0:
        y_cmd = 0.14
    elif y_err > 5.0:
        y_cmd = 0.08
    else:
        y_cmd = -_clamp(vy * 0.35 + y_err * 0.01, -0.08, 0.08)

    # Radial: hold near station radius; avoid direct aggressive radial motion.
    x_cmd = _clamp(-x_err * 0.015 - vx * 0.25, -0.12, 0.12)
    # Cross-track: null z error and rate.
    z_cmd = _clamp(-z_err * 0.02 - vz * 0.30, -0.12, 0.12)

    # Speed gating near target to avoid overrun.
    if dist < 50.0:
        brake = _clamp((speed - target_speed) * 1.8, 0.0, 0.16)
        if vy > 0:
            y_cmd -= brake
        else:
            y_cmd += brake * 0.25
        x_cmd -= _clamp(vx * 0.15, -0.05, 0.05)
        z_cmd -= _clamp(vz * 0.15, -0.05, 0.05)

    if dist < 10.0:
        x_cmd *= 0.4
        y_cmd *= 0.4
        z_cmd *= 0.4
        if speed > 0.1:
            y_cmd = _clamp(y_cmd - copysign(0.05, vy), -0.08, 0.08)

    # Normalize if needed.
    max_abs = max(abs(x_cmd), abs(y_cmd), abs(z_cmd), 1e-6)
    if max_abs > 1.0:
        x_cmd /= max_abs
        y_cmd /= max_abs
        z_cmd /= max_abs

    # Gentle attitude tracking; if camera sees port, align yaw/pitch to bearing.
    pitch = yaw = roll = 0.0
    if _valid(state, 'dock_camera_bearing_x_valid') and _valid(state, 'dock_camera_bearing_y_valid') and _valid(state, 'dock_camera_bearing_z_valid'):
        bx = _v(state, 'dock_camera_bearing_x')
        by2 = _v(state, 'dock_camera_bearing_y')
        bz = _v(state, 'dock_camera_bearing_z')
        bnorm = sqrt(bx * bx + by2 * by2 + bz * bz)
        if bnorm > 1e-6:
            pitch = _clamp(-bz / bnorm * 0.02, -ATT_LIMIT, ATT_LIMIT)
            yaw = _clamp(by2 / bnorm * 0.02, -ATT_LIMIT, ATT_LIMIT)
            roll = _clamp(-bx / bnorm * 0.01, -ATT_LIMIT, ATT_LIMIT)

    _state['prev_rel'] = (x, y, z, vx, vy, vz)
    _state['prev_time'] = _state.get('prev_time', 0.0) + dt

    # Fuel-aware reduction when low fuel: prioritize only essential braking.
    fuel_pct = _v(state, 'fuel_remaining_pct', 100.0)
    if fuel_pct < 15.0:
        x_cmd *= 0.7
        z_cmd *= 0.7
        y_cmd = _clamp(y_cmd, -0.18, 0.18)
    if fuel_pct < 5.0:
        x_cmd *= 0.5
        z_cmd *= 0.5
        y_cmd = _clamp(y_cmd, -0.12, 0.12)

    # If a range sensor is the only clue and it is close, slow down more aggressively.
    if source == 'cam' and dist < 30.0:
        y_cmd = _clamp(y_cmd, -0.08, 0.08)

    return {
        'thrust_x': _clamp(x_cmd + _state['bias_x'], -THRUST_LIMIT, THRUST_LIMIT),
        'thrust_y': _clamp(y_cmd + _state['bias_y'], -THRUST_LIMIT, THRUST_LIMIT),
        'thrust_z': _clamp(z_cmd + _state['bias_z'], -THRUST_LIMIT, THRUST_LIMIT),
        'att_rate_roll': _clamp(roll, -ATT_LIMIT, ATT_LIMIT),
        'att_rate_pitch': _clamp(pitch, -ATT_LIMIT, ATT_LIMIT),
        'att_rate_yaw': _clamp(yaw, -ATT_LIMIT, ATT_LIMIT),
    }
