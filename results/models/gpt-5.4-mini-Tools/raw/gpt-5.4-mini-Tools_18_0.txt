#include <algorithm>
#include <iostream>
#include <limits>
#include <queue>
#include <utility>
#include <vector>

using namespace std;

struct Edge {
    int u, v;
};

static vector<Edge> read_edges(int m) {
    vector<Edge> edges;
    edges.reserve(m);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        edges.push_back({u, v});
    }
    return edges;
}

static vector<Edge> find_bridges_cut(int n, const vector<Edge>& edges) {
    vector<vector<pair<int,int>>> g(n);
    for (int i = 0; i < (int)edges.size(); ++i) {
        auto [u, v] = edges[i];
        if (u == v) continue;
        g[u].push_back({v, i});
        g[v].push_back({u, i});
    }
    vector<int> tin(n, -1), low(n, -1), it(n, 0), parent_edge(n, -1), parent(n, -1);
    int timer = 0;
    vector<int> st;
    st.reserve(n);
    for (int s = 0; s < n; ++s) {
        if (tin[s] != -1) continue;
        st.push_back(s);
        parent[s] = -1;
        while (!st.empty()) {
            int v = st.back();
            if (tin[v] == -1) tin[v] = low[v] = timer++;
            if (it[v] < (int)g[v].size()) {
                auto [to, id] = g[v][it[v]++];
                if (id == parent_edge[v]) continue;
                if (tin[to] == -1) {
                    parent[to] = v;
                    parent_edge[to] = id;
                    st.push_back(to);
                } else {
                    low[v] = min(low[v], tin[to]);
                }
            } else {
                st.pop_back();
                if (parent[v] != -1) low[parent[v]] = min(low[parent[v]], low[v]);
            }
        }
    }
    vector<Edge> cut;
    for (int v = 0; v < n; ++v) {
        if (parent[v] != -1 && low[v] > tin[parent[v]]) cut.push_back(edges[parent_edge[v]]);
    }
    return cut;
}

static vector<Edge> stoer_wagner_cut(int n, const vector<Edge>& edges) {
    vector<vector<long long>> w(n, vector<long long>(n, 0));
    for (auto &e : edges) {
        if (e.u == e.v) continue;
        w[e.u][e.v] += 1;
        w[e.v][e.u] += 1;
    }
    vector<int> verts(n);
    for (int i = 0; i < n; ++i) verts[i] = i;
    vector<char> in_set(n);
    vector<long long> ws(n);
    vector<int> best_side;
    long long best = numeric_limits<long long>::max();

    while ((int)verts.size() > 1) {
        fill(in_set.begin(), in_set.end(), false);
        fill(ws.begin(), ws.end(), 0);
        int prev = -1, sel = -1;
        for (int phase = 0; phase < (int)verts.size(); ++phase) {
            sel = -1;
            for (int v : verts) {
                if (in_set[v]) continue;
                if (sel == -1 || ws[v] > ws[sel]) sel = v;
            }
            if (phase + 1 == (int)verts.size()) {
                if (ws[sel] < best) {
                    best = ws[sel];
                    best_side.clear();
                    for (int v : verts) if (in_set[v]) best_side.push_back(v);
                }
                if (prev != -1) {
                    for (int v : verts) {
                        if (v == prev || v == sel) continue;
                        w[prev][v] += w[sel][v];
                        w[v][prev] = w[prev][v];
                    }
                    verts.erase(remove(verts.begin(), verts.end(), sel), verts.end());
                }
                break;
            }
            in_set[sel] = true;
            prev = sel;
            for (int v : verts) if (!in_set[v]) ws[v] += w[sel][v];
        }
    }

    if (best_side.empty()) return {};
    vector<char> side(n, false);
    for (int v : best_side) side[v] = true;
    vector<Edge> cut;
    for (auto &e : edges) if (side[e.u] != side[e.v]) cut.push_back(e);
    return cut;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;
    auto edges = read_edges(m);

    if (n <= 1) {
        cout << 0 << '\n';
        return 0;
    }

    if (n <= 250) {
        auto cut = stoer_wagner_cut(n, edges);
        cout << cut.size() << '\n';
        for (auto &e : cut) cout << e.u << ' ' << e.v << '\n';
        return 0;
    }

    auto cut = find_bridges_cut(n, edges);
    if (!cut.empty()) {
        cout << cut.size() << '\n';
        for (auto &e : cut) cout << e.u << ' ' << e.v << '\n';
        return 0;
    }

    if (n <= 3000 && 1LL * n * n <= 20000000LL) {
        cut = stoer_wagner_cut(n, edges);
        if (!cut.empty()) {
            cout << cut.size() << '\n';
            for (auto &e : cut) cout << e.u << ' ' << e.v << '\n';
            return 0;
        }
    }

    int best_v = 0;
    for (int i = 1; i < n; ++i) if (edges.empty() ? false : false) {}
    vector<char> side(n, false);
    side[0] = true;
    for (auto &e : edges) {
        if (e.u == 0) side[e.v] = true;
        else if (e.v == 0) side[e.u] = true;
    }
    cut.clear();
    for (auto &e : edges) if (side[e.u] != side[e.v]) cut.push_back(e);
    if (cut.empty()) cut.push_back(edges[0]);
    cout << cut.size() << '\n';
    for (auto &e : cut) cout << e.u << ' ' << e.v << '\n';
    return 0;
}
