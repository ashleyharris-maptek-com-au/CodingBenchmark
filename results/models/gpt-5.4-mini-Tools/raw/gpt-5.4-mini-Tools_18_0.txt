#include <algorithm>
#include <iostream>
#include <limits>
#include <queue>
#include <utility>
#include <vector>

using namespace std;

struct InputEdge {
    int u, v;
};

struct CutResult {
    long long value = 0;
    vector<char> side;
};

static CutResult stoer_wagner(const vector<vector<long long>>& adj) {
    const int n = (int)adj.size();
    vector<vector<long long>> g = adj;
    vector<int> vert(n);
    for (int i = 0; i < n; ++i) vert[i] = i;
    vector<char> alive(n, 1);
    vector<char> best_side(n, 0);
    long long best_value = numeric_limits<long long>::max();

    for (int phase = 0; phase < n - 1; ++phase) {
        vector<long long> w(n, 0);
        vector<char> added(n, 0);
        int prev = -1, sel = -1;

        for (int i = 0; i < n - phase; ++i) {
            sel = -1;
            for (int v : vert) {
                if (!alive[v] || added[v]) continue;
                if (sel == -1 || w[v] > w[sel]) sel = v;
            }
            added[sel] = 1;
            if (i == n - phase - 1) {
                if (w[sel] < best_value) {
                    best_value = w[sel];
                    fill(best_side.begin(), best_side.end(), 0);
                    for (int v : vert) if (alive[v] && added[v] && v != sel) best_side[v] = 1;
                }
                if (prev != -1) {
                    for (int v : vert) {
                        if (!alive[v] || v == sel) continue;
                        g[prev][v] += g[sel][v];
                        g[v][prev] = g[prev][v];
                    }
                    alive[sel] = 0;
                    for (int v : vert) {
                        g[sel][v] = g[v][sel] = 0;
                    }
                }
                break;
            }
            prev = sel;
            for (int v : vert) {
                if (!alive[v] || added[v]) continue;
                w[v] += g[sel][v];
            }
        }
    }

    return {best_value, best_side};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    if (!(cin >> N >> M)) return 0;

    vector<InputEdge> edges(M);
    for (int i = 0; i < M; ++i) cin >> edges[i].u >> edges[i].v;

    if (N <= 1) {
        cout << 0 << '\n';
        return 0;
    }

    vector<vector<long long>> adj(N, vector<long long>(N, 0));
    for (auto &e : edges) {
        adj[e.u][e.v] += 1;
        adj[e.v][e.u] += 1;
    }

    CutResult cut = stoer_wagner(adj);

    vector<pair<int, int>> cut_edges;
    for (const auto &e : edges) {
        if (cut.side[e.u] != cut.side[e.v]) cut_edges.push_back({e.u, e.v});
    }

    cout << cut_edges.size() << '\n';
    for (const auto &e : cut_edges) {
        cout << e.first << ' ' << e.second << '\n';
    }
    return 0;
}
