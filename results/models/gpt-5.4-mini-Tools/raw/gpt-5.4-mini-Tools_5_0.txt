use std::collections::{HashMap, HashSet, VecDeque};
use std::io::{self, Read};

#[derive(Clone, Copy, PartialEq, Eq, Hash, Debug)]
struct P { x: i32, y: i32 }

fn main() {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_whitespace();
    let w: i32 = match it.next().and_then(|v| v.parse().ok()) { Some(v) => v, None => return, };
    let h: i32 = it.next().and_then(|v| v.parse().ok()).unwrap();
    let m: usize = it.next().and_then(|v| v.parse().ok()).unwrap();
    let require_cycle: i32 = it.next().and_then(|v| v.parse().ok()).unwrap();
    let mut blocked = HashSet::new();
    for _ in 0..m {
        let x: i32 = it.next().and_then(|v| v.parse().ok()).unwrap();
        let y: i32 = it.next().and_then(|v| v.parse().ok()).unwrap();
        blocked.insert(P { x, y });
    }

    if w <= 0 || h <= 0 || blocked.contains(&P { x: 0, y: 0 }) {
        println!("0");
        return;
    }

    let mut cells = Vec::new();
    for y in 0..h {
        for x in 0..w {
            let p = P { x, y };
            if !blocked.contains(&p) { cells.push(p); }
        }
    }
    let n = cells.len();
    if n == 0 { println!("0"); return; }

    if require_cycle == 1 && n < 4 { println!("0"); return; }

    let mut idx = HashMap::new();
    for (i, &p) in cells.iter().enumerate() { idx.insert(p, i); }
    let start = *idx.get(&P { x: 0, y: 0 }).unwrap();

    let neigh = |p: P| -> [P; 4] { [P{x:p.x+1,y:p.y}, P{x:p.x-1,y:p.y}, P{x:p.x,y:p.y+1}, P{x:p.x,y:p.y-1}] };
    let mut adj = vec![Vec::new(); n];
    for (i, &p) in cells.iter().enumerate() {
        for q in neigh(p) {
            if let Some(&j) = idx.get(&q) { adj[i].push(j); }
        }
    }

    let degree: Vec<usize> = adj.iter().map(|v| v.len()).collect();
    if degree.iter().any(|&d| d == 0) { println!("0"); return; }
    if require_cycle == 1 && degree.iter().any(|&d| d < 2) { println!("0"); return; }

    // Fast exact cases: full rectangles with no obstacles.
    if blocked.is_empty() {
        if let Some(path) = full_rectangle_path(w as usize, h as usize, require_cycle == 1) {
            print_path(&path);
            return;
        }
    }

    // General fallback: heuristic DFS with pruning, designed for moderate obstacle grids.
    if let Some(path) = solve_backtracking(&cells, &adj, start, require_cycle == 1) {
        print_path(&path);
    } else {
        println!("0");
    }
}

fn print_path(path: &[P]) {
    println!("{}", path.len());
    for p in path {
        println!("{} {}", p.x, p.y);
    }
}

fn full_rectangle_path(w: usize, h: usize, cycle: bool) -> Option<Vec<P>> {
    if w == 1 && h == 1 {
        return if cycle { None } else { Some(vec![P { x: 0, y: 0 }]) };
    }
    let mut path = Vec::with_capacity(w * h);
    if w == 1 {
        if cycle && h % 2 == 1 { return None; }
        for y in 0..h { path.push(P { x: 0, y: y as i32 }); }
        if cycle && (h < 2 || (path.last().unwrap().y - path[0].y).abs() != 1) { return None; }
        return Some(path);
    }
    if h == 1 {
        if cycle && w % 2 == 1 { return None; }
        for x in 0..w { path.push(P { x: x as i32, y: 0 }); }
        return Some(path);
    }
    if cycle {
        if (w * h) % 2 == 1 { return None; }
        for y in 0..h {
            if y % 2 == 0 {
                for x in 0..w { path.push(P { x: x as i32, y: y as i32 }); }
            } else {
                for x in (0..w).rev() { path.push(P { x: x as i32, y: y as i32 }); }
            }
        }
        if !adjacent(path[0], *path.last().unwrap()) { return None; }
        Some(path)
    } else {
        for y in 0..h {
            if y % 2 == 0 {
                for x in 0..w { path.push(P { x: x as i32, y: y as i32 }); }
            } else {
                for x in (0..w).rev() { path.push(P { x: x as i32, y: y as i32 }); }
            }
        }
        Some(path)
    }
}

fn adjacent(a: P, b: P) -> bool { (a.x - b.x).abs() + (a.y - b.y).abs() == 1 }

fn solve_backtracking(cells: &[P], adj: &[Vec<usize>], start: usize, cycle: bool) -> Option<Vec<P>> {
    let n = cells.len();
    let mut used = vec![false; n];
    let mut path = Vec::with_capacity(n);
    let mut stack = Vec::with_capacity(n);
    used[start] = true;
    path.push(start);
    stack.push((start, 0usize));

    while let Some((v, mut i)) = stack.pop() {
        if path.len() == n {
            if !cycle || adj[v].iter().any(|&u| u == start) { return Some(path.iter().map(|&id| cells[id]).collect()); }
        }
        let mut nexts: Vec<usize> = adj[v].iter().copied().filter(|&u| !used[u]).collect();
        nexts.sort_by_key(|&u| adj[u].iter().filter(|&&x| !used[x] || x == start).count());
        while i < nexts.len() {
            let u = nexts[i];
            i += 1;
            if used[u] { continue; }
            used[u] = true;
            path.push(u);
            stack.push((v, i));
            stack.push((u, 0));
            break;
        }
        if i >= nexts.len() {
            used[v] = false;
            path.pop();
        }
    }
    None
}
