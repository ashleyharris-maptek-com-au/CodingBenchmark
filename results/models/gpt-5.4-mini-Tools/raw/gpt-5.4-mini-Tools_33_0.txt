using System;
using System.IO;
using System.Text;
using System.Collections.Generic;
using System.Runtime.InteropServices;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        long[][] flow = new long[n][];
        long[][] dist = new long[n][];
        for (int i = 0; i < n; i++)
        {
            flow[i] = new long[n];
            for (int j = 0; j < n; j++) flow[i][j] = fs.NextLong();
        }
        for (int i = 0; i < n; i++)
        {
            dist[i] = new long[n];
            for (int j = 0; j < n; j++) dist[i][j] = fs.NextLong();
        }

        int[] perm = BuildInitialPermutation(flow, dist, n);
        long cost = ComputeCost(flow, dist, perm);

        if (n <= 250)
        {
            ImproveLocalSearch(flow, dist, perm, ref cost);
        }
        else
        {
            ImproveLarge(flow, dist, perm, ref cost);
        }

        var sb = new StringBuilder();
        sb.AppendLine(cost.ToString());
        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(perm[i]);
        }
        sb.AppendLine();
        Console.Write(sb.ToString());
    }

    static int[] BuildInitialPermutation(long[][] flow, long[][] dist, int n)
    {
        long[] fScore = new long[n];
        long[] dScore = new long[n];
        for (int i = 0; i < n; i++)
        {
            long sf = 0, sd = 0;
            var fi = flow[i];
            var di = dist[i];
            for (int j = 0; j < n; j++)
            {
                sf += fi[j];
                sd += di[j];
            }
            fScore[i] = sf;
            dScore[i] = sd;
        }

        int[] facilities = new int[n];
        int[] locations = new int[n];
        for (int i = 0; i < n; i++) { facilities[i] = i; locations[i] = i; }
        Array.Sort(facilities, (a, b) => fScore[b].CompareTo(fScore[a]));
        Array.Sort(locations, (a, b) => dScore[a].CompareTo(dScore[b]));

        int[] perm = new int[n];
        for (int k = 0; k < n; k++) perm[facilities[k]] = locations[k];
        return perm;
    }

    static long ComputeCost(long[][] flow, long[][] dist, int[] perm)
    {
        int n = perm.Length;
        long total = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = perm[i];
            var fi = flow[i];
            for (int j = 0; j < n; j++)
            {
                total += fi[j] * dist[pi][perm[j]];
            }
        }
        return total;
    }

    static void ImproveLocalSearch(long[][] flow, long[][] dist, int[] perm, ref long cost)
    {
        int n = perm.Length;
        int[] posOfLoc = new int[n];
        for (int i = 0; i < n; i++) posOfLoc[perm[i]] = i;

        bool improved = true;
        int passes = 0;
        while (improved && passes < 3)
        {
            improved = false;
            passes++;
            for (int a = 0; a < n; a++)
            {
                for (int b = a + 1; b < n; b++)
                {
                    long delta = SwapDelta(flow, dist, perm, a, b);
                    if (delta < 0)
                    {
                        int pa = perm[a], pb = perm[b];
                        perm[a] = pb; perm[b] = pa;
                        posOfLoc[pa] = b; posOfLoc[pb] = a;
                        cost += delta;
                        improved = true;
                    }
                }
            }
        }
    }

    static void ImproveLarge(long[][] flow, long[][] dist, int[] perm, ref long cost)
    {
        int n = perm.Length;
        var rng = new XorShift64((ulong)(n * 1469598103934665603UL));
        int trials = Math.Min(120000, n * 40);
        for (int t = 0; t < trials; t++)
        {
            int a = (int)(rng.Next() % (uint)n);
            int b = (int)(rng.Next() % (uint)n);
            if (a == b) continue;
            if (a > b) { int tmp = a; a = b; b = tmp; }
            long delta = SwapDelta(flow, dist, perm, a, b);
            if (delta < 0)
            {
                int pa = perm[a], pb = perm[b];
                perm[a] = pb; perm[b] = pa;
                cost += delta;
            }
        }
    }

    static long SwapDelta(long[][] flow, long[][] dist, int[] perm, int a, int b)
    {
        if (a == b) return 0;
        int n = perm.Length;
        int pa = perm[a], pb = perm[b];
        long delta = 0;
        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;
            int pk = perm[k];
            delta += flow[a][k] * (dist[pb][pk] - dist[pa][pk]);
            delta += flow[b][k] * (dist[pa][pk] - dist[pb][pk]);
            delta += flow[k][a] * (dist[pk][pb] - dist[pk][pa]);
            delta += flow[k][b] * (dist[pk][pa] - dist[pk][pb]);
        }
        delta += flow[a][a] * (dist[pb][pb] - dist[pa][pa]);
        delta += flow[b][b] * (dist[pa][pa] - dist[pb][pb]);
        delta += flow[a][b] * (dist[pb][pa] - dist[pa][pb]);
        delta += flow[b][a] * (dist[pa][pb] - dist[pb][pa]);
        return delta;
    }

    sealed class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int len, ptr;

        public FastScanner(Stream stream) { this.stream = stream; }

        private int ReadByte()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = ReadByte(); } while (c <= ' ' && c >= 0);
            int sign = 1;
            if (c == '-') { sign = -1; c = ReadByte(); }
            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return val * sign;
        }

        public long NextLong()
        {
            int c;
            do { c = ReadByte(); } while (c <= ' ' && c >= 0);
            long sign = 1;
            if (c == '-') { sign = -1; c = ReadByte(); }
            long val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return val * sign;
        }
    }

    struct XorShift64
    {
        private ulong x;
        public XorShift64(ulong seed) { x = seed == 0 ? 88172645463393265UL : seed; }
        public ulong Next()
        {
            ulong z = x;
            z ^= z << 13;
            z ^= z >> 7;
            z ^= z << 17;
            x = z;
            return z;
        }
    }
}
