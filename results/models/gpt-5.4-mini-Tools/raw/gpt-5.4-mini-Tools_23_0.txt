use std::f64::consts::PI;
use std::io::{self, Read, Write};

#[derive(Clone, Debug)]
struct TokenReader {
    data: Vec<u8>,
    pos: usize,
}

impl TokenReader {
    fn new() -> io::Result<Self> {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data)?;
        Ok(Self { data, pos: 0 })
    }

    fn next_str(&mut self) -> Option<&str> {
        while self.pos < self.data.len() && self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        if self.pos >= self.data.len() {
            return None;
        }
        let start = self.pos;
        while self.pos < self.data.len() && !self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        std::str::from_utf8(&self.data[start..self.pos]).ok()
    }

    fn next_f64(&mut self) -> Option<f64> {
        self.next_str()?.parse().ok()
    }

    fn next_usize(&mut self) -> Option<usize> {
        self.next_str()?.parse().ok()
    }
}

#[derive(Clone, Copy)]
struct Pad {
    x1: f64,
    x2: f64,
    y: f64,
}

fn clamp(v: f64, lo: f64, hi: f64) -> f64 {
    v.max(lo).min(hi)
}

fn angle_wrap(mut a: f64) -> f64 {
    while a > PI {
        a -= 2.0 * PI;
    }
    while a < -PI {
        a += 2.0 * PI;
    }
    a
}

fn parse_pad(map: &[Vec<u8>], target_x: usize, target_y: usize) -> Pad {
    let h = map.len();
    let w = map.first().map(|r| r.len()).unwrap_or(0);

    let y = target_y.min(h.saturating_sub(1));
    let row = map.get(y);

    let mut x1 = target_x;
    let mut x2 = target_x;

    if let Some(r) = row {
        let mut i = target_x;
        while i > 0 && r[i - 1] == b'L' {
            i -= 1;
        }
        x1 = i;
        let mut j = target_x;
        while j + 1 < w && r[j + 1] == b'L' {
            j += 1;
        }
        x2 = j;
    }

    Pad {
        x1: x1 as f64 + 0.5,
        x2: x2 as f64 + 0.5,
        y: y as f64 + 0.5,
    }
}

fn main() {
    let mut input = match TokenReader::new() {
        Ok(r) => r,
        Err(_) => return,
    };

    let _width = match input.next_usize() {
        Some(v) => v,
        None => return,
    };
    let height = match input.next_usize() {
        Some(v) => v,
        None => return,
    };
    let gravity = match input.next_f64() {
        Some(v) => v,
        None => return,
    };
    let max_thrust = match input.next_f64() {
        Some(v) => v,
        None => return,
    };
    let _fuel_total = input.next_f64().unwrap_or(0.0);

    let start_x = input.next_usize().unwrap_or(0);
    let start_y = input.next_usize().unwrap_or(0);
    let target_x = input.next_usize().unwrap_or(0);
    let target_y = input.next_usize().unwrap_or(0);

    let mut map = Vec::with_capacity(height);
    for _ in 0..height {
        let row = input.next_str().unwrap_or("");
        map.push(row.as_bytes().to_vec());
    }

    let pad = parse_pad(&map, target_x, target_y);
    let mut stdout = io::BufWriter::new(io::stdout().lock());

    let mut last_angle = 0.0;
    let mut have_last = false;
    let mut _spawn_hint = (start_x as f64 + 0.5, start_y as f64 + 0.5);

    loop {
        let x = match input.next_f64() {
            Some(v) => v,
            None => break,
        };
        let y = match input.next_f64() {
            Some(v) => v,
            None => break,
        };
        let vx = match input.next_f64() {
            Some(v) => v,
            None => break,
        };
        let vy = match input.next_f64() {
            Some(v) => v,
            None => break,
        };
        let angle = match input.next_f64() {
            Some(v) => v,
            None => break,
        };
        let angular_vel = match input.next_f64() {
            Some(v) => v,
            None => break,
        };
        let fuel_remaining = match input.next_f64() {
            Some(v) => v,
            None => break,
        };

        let pad_center_x = (pad.x1 + pad.x2) * 0.5;
        let pad_half_w = (pad.x2 - pad.x1) * 0.5;
        let dx = x - pad_center_x;
        let dy = y - pad.y;

        let altitude = dy.max(0.0);
        let vx_abs = vx.abs();

        let mut target_angle;
        let mut thrust;

        let horizontal_goal = clamp(-dx * 0.08 - vx * 0.18, -0.65, 0.65);
        let near_pad = altitude < 35.0;
        let on_pad_axis = dx.abs() < pad_half_w + 1.0;

        if altitude > 55.0 {
            target_angle = horizontal_goal;
            if vy > 18.0 {
                thrust = 1.0;
            } else if vy > 8.0 {
                thrust = 0.9;
            } else {
                thrust = 0.75;
            }
            if vx_abs > 12.0 {
                thrust = 1.0;
            }
        } else if altitude > 18.0 {
            target_angle = clamp(horizontal_goal, -0.45, 0.45);
            let desired_vy = if on_pad_axis { 2.0 } else { 6.0 };
            let vy_err = vy - desired_vy;
            thrust = clamp((gravity + vy_err * 0.12) / max_thrust, 0.25, 1.0);
            if vx_abs > 5.0 {
                thrust = thrust.max(0.85);
            }
        } else {
            target_angle = clamp(horizontal_goal * 0.7, -0.25, 0.25);
            let desired_vy = if on_pad_axis { 1.2 } else { 3.0 };
            let vy_err = vy - desired_vy;
            thrust = clamp((gravity + vy_err * 0.18) / max_thrust, 0.15, 1.0);
            if dx.abs() < pad_half_w + 0.8 && vx_abs < 1.5 && vy < 3.0 {
                target_angle = 0.0;
                thrust = clamp((gravity + vy * 0.18) / max_thrust, 0.0, 1.0);
            }
        }

        if fuel_remaining < 2.5 {
            thrust = thrust.min(0.65);
        }

        if near_pad && dx.abs() < pad_half_w + 1.5 && vy > 5.0 {
            thrust = thrust.max(0.9);
        }

        let angle_err = angle_wrap(target_angle - angle);
        let turn = clamp(angle_err * 2.5 - angular_vel * 0.8, -1.0, 1.0);

        if have_last && (angle - last_angle).abs() > PI * 1.5 {
            thrust = thrust.max(0.8);
        }
        last_angle = angle;
        have_last = true;

        let thrust = clamp(thrust, 0.0, 1.0);
        let turn = clamp(turn, -1.0, 1.0);

        let _ = writeln!(stdout, "{:.3} {:.3}", thrust, turn);
        let _ = stdout.flush();
    }
}
