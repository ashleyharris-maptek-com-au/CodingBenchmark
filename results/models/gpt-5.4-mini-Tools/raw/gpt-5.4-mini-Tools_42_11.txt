OpCapability Geometry

OpExecutionMode %main OutputTriangleStrip

OpExecutionMode %main Invocations 1
OpSource GLSL 450

OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%uint_4 = OpConstant %uint 4
%_arr_gl_PerVertex_in_uint_4 = OpTypeArray %gl_PerVertex_in %uint_4
%_arr_v3float_uint_4 = OpTypeArray %v3float %uint_4
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_in_arr_pv = OpTypePointer Input %_arr_gl_PerVertex_in_uint_4
%ptr_in_arr_v3 = OpTypePointer Input %_arr_v3float_uint_4
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%float_0 = OpConstant %float 0

%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v4 Output

%main = OpFunction %void None %func
%entry = OpLabel
%s_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%e_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%a_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%b_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%s4 = OpLoad %v4float %s_ptr
%e4 = OpLoad %v4float %e_ptr
%a4 = OpLoad %v4float %a_ptr
%b4 = OpLoad %v4float %b_ptr
%s = OpVectorShuffle %v3float %s4 %s4 0 1 2
%e = OpVectorShuffle %v3float %e4 %e4 0 1 2
%a = OpVectorShuffle %v3float %a4 %a4 0 1 2
%b = OpVectorShuffle %v3float %b4 %b4 0 1 2
%t0 = OpFSub %v3float %s %a
%t1 = OpFSub %v3float %b %e
%tangent = OpFAdd %v3float %t0 %t1
%tangent_n = OpExtInst %v3float %1 Normalize %tangent
%up = OpCompositeConstruct %v3float %float_0 %float_0 %float_1
%side = OpExtInst %v3float %1 Cross %up %tangent_n
%side_n = OpExtInst %v3float %1 Normalize %side
%half_w = OpFMul %float %float_0_02 %float_0_5
%offset = OpVectorTimesScalar %v3float %side_n %half_w
%dir = OpFSub %v3float %e %s
%one_third = OpFDiv %float %float_1 %float_3
%two_third = OpFMul %float %one_third %float_2
%seg1 = OpVectorTimesScalar %v3float %dir %one_third
%seg2 = OpVectorTimesScalar %v3float %dir %two_third
%p1 = OpFAdd %v3float %s %seg1
%p2 = OpFAdd %v3float %s %seg2
%c1 = OpLoad %v3float %color %int_1
%c2 = OpLoad %v3float %color %int_2

%v0p = OpFAdd %v3float %s %offset
%v1p = OpFSub %v3float %s %offset
%v2p = OpFAdd %v3float %p1 %offset
%v3p = OpFSub %v3float %p1 %offset
%v4p = OpFAdd %v3float %p2 %offset
%v5p = OpFSub %v3float %p2 %offset
%v6p = OpFAdd %v3float %e %offset
%v7p = OpFSub %v3float %e %offset

%v0 = OpCompositeConstruct %v4float %float_0 %float_0 %float_0 %float_1
%v1 = OpCompositeConstruct %v4float %float_0 %float_0 %float_0 %float_1
%v2 = OpCompositeConstruct %v4float %float_0 %float_0 %float_0 %float_1
%v3 = OpCompositeConstruct %v4float %float_0 %float_0 %float_0 %float_1
%v4 = OpCompositeConstruct %v4float %float_0 %float_0 %float_0 %float_1
%v5 = OpCompositeConstruct %v4float %float_0 %float_0 %float_0 %float_1
%v6 = OpCompositeConstruct %v4float %float_0 %float_0 %float_0 %float_1
%v7 = OpCompositeConstruct %v4float %float_0 %float_0 %float_0 %float_1

OpStore %gl_out %v0
