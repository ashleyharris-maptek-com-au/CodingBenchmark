use std::cmp::Reverse;
use std::collections::{BinaryHeap, HashSet};
use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }
    for v in 0..n {
        adj[v].sort_unstable();
        adj[v].dedup();
    }

    if n == 0 {
        println!("0");
        println!();
        return;
    }

    let mut dominated = vec![false; n];
    let mut in_set = vec![false; n];
    let mut dom_count = vec![1usize; n];
    for v in 0..n {
        dom_count[v] += adj[v].len();
    }

    // Greedy construction: repeatedly pick the vertex that newly dominates the most undominated vertices.
    let mut heap = BinaryHeap::new();
    for v in 0..n {
        heap.push((dom_count[v], Reverse(v)));
    }

    let mut set = Vec::new();
    let mut undominated = n;

    while undominated > 0 {
        let mut chosen = None;
        while let Some((score, Reverse(v))) = heap.pop() {
            if in_set[v] {
                continue;
            }
            let mut actual = if !dominated[v] { 1 } else { 0 };
            for &to in &adj[v] {
                if !dominated[to] {
                    actual += 1;
                }
            }
            if actual == score {
                chosen = Some(v);
                break;
            }
            heap.push((actual, Reverse(v)));
        }

        let v = chosen.unwrap_or_else(|| {
            let mut best = 0usize;
            let mut best_score = 0usize;
            for u in 0..n {
                if in_set[u] { continue; }
                let mut score = if !dominated[u] { 1 } else { 0 };
                for &to in &adj[u] {
                    if !dominated[to] { score += 1; }
                }
                if score > best_score {
                    best_score = score;
                    best = u;
                }
            }
            best
        });

        if in_set[v] {
            continue;
        }
        in_set[v] = true;
        set.push(v);
        if !dominated[v] {
            dominated[v] = true;
            undominated -= 1;
        }
        for &to in &adj[v] {
            if !dominated[to] {
                dominated[to] = true;
                undominated -= 1;
            }
        }
    }

    // Local improvement: remove redundant vertices in descending usefulness order.
    let mut covered_by = vec![0usize; n];
    for &v in &set {
        covered_by[v] += 1;
        for &to in &adj[v] {
            covered_by[to] += 1;
        }
    }

    let mut changed = true;
    while changed {
        changed = false;
        for i in 0..set.len() {
            let v = set[i];
            let mut ok = true;
            if covered_by[v] <= 1 { ok = false; }
            if ok {
                for &to in &adj[v] {
                    if covered_by[to] <= 1 {
                        ok = false;
                        break;
                    }
                }
            }
            if ok {
                changed = true;
                covered_by[v] -= 1;
                for &to in &adj[v] {
                    covered_by[to] -= 1;
                }
                in_set[v] = false;
                set[i] = usize::MAX;
            }
        }
        if changed {
            set.retain(|&v| v != usize::MAX);
        }
    }

    set.sort_unstable();
    println!("{}", set.len());
    if !set.is_empty() {
        for (i, v) in set.iter().enumerate() {
            if i > 0 { print!(" "); }
            print!("{}", v);
        }
    }
    println!();
}
