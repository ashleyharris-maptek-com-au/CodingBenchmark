; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %uv "uv"
OpName %tangent "tangent"
OpName %color "color"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_input_v2 = OpTypePointer Input %v2float
%ptr_output_v4 = OpTypePointer Output %v4float
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%ptr_uniform_v4 = OpTypePointer Uniform %v4float
%extfloat = OpExtInstImport "GLSL.std.450"
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0.0
%float_0_05 = OpConstant %float 0.05
%float_0_1 = OpConstant %float 0.1
%float_0_2 = OpConstant %float 0.2
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_0_6 = OpConstant %float 0.6
%float_0_7 = OpConstant %float 0.7
%float_0_9 = OpConstant %float 0.9
%float_0_95 = OpConstant %float 0.95
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0
%float_8 = OpConstant %float 8.0
%float_16 = OpConstant %float 16.0
%v2_scale = OpConstantComposite %v2float %float_8 %float_16
%brick_rgb = OpConstantComposite %v3float %float_0_7 %float_0_2 %float_0_1
%mortar_rgb = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_0_7
%alpha_one = OpConstantComposite %v4float %float_1 %float_1 %float_1 %float_1

%worldPos = OpVariable %ptr_input_v3 Input
%normal = OpVariable %ptr_input_v3 Input
%uv = OpVariable %ptr_input_v2 Input
%tangent = OpVariable %ptr_input_v3 Input
%color = OpVariable %ptr_input_v3 Input
%outColor = OpVariable %ptr_output_v4 Output
%ubo = OpVariable %ptr_uniform_ubo Uniform

%main = OpFunction %void None %func
%entry = OpLabel
%uv_in = OpLoad %v2float %uv
%uv_scaled = OpFMul %v2float %uv_in %v2_scale
%x = OpCompositeExtract %float %uv_scaled 0
%y = OpCompositeExtract %float %uv_scaled 1
%y_floor = OpExtInst %float %extfloat Floor %y
%y_frac = OpFSub %float %y %y_floor
%y_mod2 = OpFMod %float %y_floor %float_2
%row_odd = OpFOrdEqual %bool %y_mod2 %float_1
%offset = OpSelect %float %row_odd %float_0_5 %float_0
%x_off = OpFAdd %float %x %offset
%x_floor = OpExtInst %float %extfloat Floor %x_off
%x_frac = OpFSub %float %x_off %x_floor
%mortar_x_lo = OpFOrdLessThan %bool %x_frac %float_0_05
%mortar_x_hi = OpFOrdGreaterThan %bool %x_frac %float_0_95
%mortar_y_lo = OpFOrdLessThan %bool %y_frac %float_0_1
%mortar_y_hi = OpFOrdGreaterThan %bool %y_frac %float_0_9
%mortar_x_a = OpLogicalOr %bool %mortar_x_lo %mortar_x_hi
%mortar_y_a = OpLogicalOr %bool %mortar_y_lo %mortar_y_hi
%is_mortar = OpLogicalOr %bool %mortar_x_a %mortar_y_a

%n_in = OpLoad %v3float %normal
%n_len = OpExtInst %float %extfloat Length %n_in
%n_norm = OpFDiv %v3float %n_in %n_len
%world_in = OpLoad %v3float %worldPos
%light_ptr = OpAccessChain %ptr_uniform_v4 %ubo %int_3
%light_v4 = OpLoad %v4float %light_ptr
%light_v3 = OpVectorShuffle %v3float %light_v4 %light_v4 0 1 2
%light_dir_raw = OpFSub %v3float %light_v3 %world_in
%light_len = OpExtInst %float %extfloat Length %light_dir_raw
%light_dir = OpFDiv %v3float %light_dir_raw %light_len
%ndl = OpDot %float %n_norm %light_dir
%diffuse = OpExtInst %float %extfloat FMax %ndl %float_0
%shade_base = OpFMul %float %float_0_6 %diffuse
%shade = OpFAdd %float %float_0_4 %shade_base

%brick_shaded = OpVectorTimesScalar %v3float %brick_rgb %shade
%mortar_shaded = OpVectorTimesScalar %v3float %mortar_rgb %shade
%rgb = OpSelect %v3float %is_mortar %mortar_shaded %brick_shaded
%out = OpCompositeInsert %v4float %float_1 %rgb 3
OpStore %outColor %out
OpReturn
OpFunctionEnd
