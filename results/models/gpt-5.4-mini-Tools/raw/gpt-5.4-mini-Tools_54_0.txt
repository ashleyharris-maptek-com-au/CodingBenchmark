"""Autopilot logic for ILS autoland benchmark.

The function is intentionally conservative: it prioritises stabilised approach,
then flare, touchdown, and rollout, while triggering go-around when the aircraft
is not in a safe landing envelope.
"""

from __future__ import annotations

from math import isfinite

_phase = "intercept"
_go_around_armed = False
_last_ra = None
_last_vs = 0.0
_last_loc = 0.0
_last_gs = 0.0
_last_time = 0.0
_flare_started = False
_touchdown_confirmed = False
_rollout_timer = 0.0


def _valid_number(x):
    return isinstance(x, (int, float)) and isfinite(float(x))


def _median(values):
    values = sorted(v for v in values if _valid_number(v))
    if not values:
        return None
    n = len(values)
    mid = n // 2
    return values[mid] if n % 2 else 0.5 * (values[mid - 1] + values[mid])


def _clamp(x, lo, hi):
    return max(lo, min(hi, x))


def _saturate(x, limit=1.0):
    return _clamp(x, -limit, limit)


def _best_of_valid(state, keys):
    vals = [state.get(k) for k in keys if _valid_number(state.get(k))]
    return _median(vals)


def _sense_alt_ft(state):
    ra = state.get("radio_alt_ft")
    if _valid_number(ra):
        return float(ra)
    baro = _best_of_valid(state, ["baro_alt_1_ft", "baro_alt_2_ft", "baro_alt_3_ft"])
    return float(baro) if baro is not None else None


def _sense_ias(state):
    ias = _best_of_valid(state, ["ias_1_kt", "ias_2_kt", "ias_3_kt"])
    if ias is not None:
        return float(ias)
    tas = _best_of_valid(state, ["tas_1_kt", "tas_2_kt", "tas_3_kt"])
    return float(tas) if tas is not None else None


def _sense_track(state):
    trk = _best_of_valid(state, ["gps_1_track_deg", "gps_2_track_deg", "gps_3_track_deg", "irs_1_track_deg", "irs_2_track_deg", "irs_3_track_deg"])
    return float(trk) if trk is not None else None


def _sense_heading(state):
    hdg = _best_of_valid(state, ["irs_1_heading_deg", "irs_2_heading_deg", "irs_3_heading_deg"])
    return float(hdg) if hdg is not None else None


def _sense_vs(state):
    vs = _best_of_valid(state, ["irs_1_vs_fpm", "irs_2_vs_fpm", "irs_3_vs_fpm"])
    return float(vs) if vs is not None else None


def _sense_loc(state):
    loc = _best_of_valid(state, ["localizer_1_dots", "localizer_2_dots", "localizer_3_dots"])
    return float(loc) if loc is not None else None


def _sense_gs(state):
    gs = _best_of_valid(state, ["glideslope_1_dots", "glideslope_2_dots", "glideslope_3_dots"])
    return float(gs) if gs is not None else None


def _sensor_disagree(values, limit):
    vals = [v for v in values if _valid_number(v)]
    if len(vals) < 2:
        return False
    return max(vals) - min(vals) > limit


def autopilot_step(state: dict, dt: float) -> dict:
    global _phase, _go_around_armed, _last_ra, _last_alt, _last_vs, _last_loc, _last_gs, _last_time, _flare_started, _touchdown_confirmed, _rollout_timer

    dt = float(dt) if _valid_number(dt) and dt > 0 else 0.02
    ra = _sense_alt_ft(state)
    ias = _sense_ias(state)
    loc = _sense_loc(state)
    gs = _sense_gs(state)
    track = _sense_track(state)
    hdg = _sense_heading(state)
    vs = _sense_vs(state)
    target_ias = float(state.get("target_ias_kt") or 0.0)
    target_vref = float(state.get("target_vref_kt") or (target_ias - 10.0 if target_ias else 137.0))
    target_loc = float(state.get("target_loc_dots") or 0.0)
    target_gs = float(state.get("target_gs_dots") or 0.0)
    target_alt = float(state.get("target_alt_ft") or 0.0)
    target_vs = float(state.get("target_vs_fpm") or -700.0)
    on_ground = bool(state.get("on_ground")) or bool(state.get("weight_on_wheels"))
    wow = bool(state.get("weight_on_wheels"))
    below_dh = bool(state.get("below_decision_height"))
    groundspeed = state.get("groundspeed_kt")
    dme = _best_of_valid(state, ["dme_1_nm", "dme_2_nm"])

    if _sensor_disagree([loc, state.get("localizer_1_dots"), state.get("localizer_2_dots"), state.get("localizer_3_dots")], 1.5) and ra is not None and ra > 200:
        _go_around_armed = True
    if _sensor_disagree([gs, state.get("glideslope_1_dots"), state.get("glideslope_2_dots"), state.get("glideslope_3_dots")], 1.5) and ra is not None and ra > 200:
        _go_around_armed = True
    if ias is not None and target_vref > 0 and ias > max(target_vref + 45.0, target_ias + 35.0):
        _go_around_armed = True
    if ra is not None and ra > 300 and vs is not None and vs < -2500:
        _go_around_armed = True

    if on_ground or wow:
        _phase = "rollout"
    elif ra is not None:
        if ra < 50:
            _phase = "flare"
        elif ra < 200:
            _phase = "approach"
        else:
            _phase = "intercept"
    elif dme is not None and dme < 1.0:
        _phase = "flare" if below_dh else "approach"

    if _go_around_armed and not on_ground:
        return {
            "throttle_1": 1.0,
            "throttle_2": 1.0,
            "elevator": _saturate(0.45 if ra is None or ra > 500 else 0.35),
            "aileron": _saturate(-0.4 * (loc or 0.0)),
            "rudder": _saturate(-0.2 * (loc or 0.0) + 0.1 * (0.0 if hdg is None or track is None else ((hdg - track + 540.0) % 360.0 - 180.0) / 30.0)),
            "elevator_trim": 0.1,
            "flaps": 20.0,
            "gear": 1.0,
            "speedbrake": 0.0,
            "spoiler": 0.0,
            "autobrake": 0.0,
            "reverse_1": 0.0,
            "reverse_2": 0.0,
        }

    out = {"gear": 1.0, "speedbrake": 0.0, "spoiler": 0.0, "autobrake": 0.0, "reverse_1": 0.0, "reverse_2": 0.0}

    if _phase == "intercept":
        loc_err = (loc if loc is not None else 0.0) - target_loc
        gs_err = (gs if gs is not None else 0.0) - target_gs
        speed_err = (ias if ias is not None else target_vref) - max(target_vref + 15.0, target_ias - 5.0)
        hdg_err = 0.0
        if hdg is not None and track is not None:
            hdg_err = ((hdg - track + 540.0) % 360.0) - 180.0
        out.update({
            "flaps": 15.0 if (ias or 0.0) < target_vref + 40.0 else 10.0,
            "throttle_1": _clamp(0.55 + 0.03 * speed_err + 0.05 * max(0.0, -gs_err), 0.25, 0.95),
            "throttle_2": _clamp(0.55 + 0.03 * speed_err + 0.05 * max(0.0, -gs_err), 0.25, 0.95),
            "elevator": _saturate(-0.10 * (gs_err) + 0.04 * (target_vs / 1000.0 if target_vs else -0.7)),
            "aileron": _saturate(-0.35 * loc_err - 0.03 * hdg_err / 10.0),
            "rudder": _saturate(-0.15 * loc_err + 0.04 * hdg_err / 10.0),
            "elevator_trim": _saturate(0.05 * (-target_vs / 1000.0)),
        })
    elif _phase == "approach":
        loc_err = (loc if loc is not None else 0.0) - target_loc
        gs_err = (gs if gs is not None else 0.0) - target_gs
        speed_ref = max(target_vref, target_ias - 10.0)
        speed_err = (ias if ias is not None else speed_ref) - speed_ref
        out.update({
            "flaps": 30.0 if (ias or 0.0) < target_vref + 20.0 else 20.0,
            "throttle_1": _clamp(0.45 - 0.03 * speed_err - 0.05 * gs_err, 0.18, 0.82),
            "throttle_2": _clamp(0.45 - 0.03 * speed_err - 0.05 * gs_err, 0.18, 0.82),
            "elevator": _saturate(-0.12 * gs_err + 0.015 * ((target_vref - (ias or target_vref)) if ias is not None else 0.0)),
            "aileron": _saturate(-0.45 * loc_err),
            "rudder": _saturate(-0.10 * loc_err),
            "elevator_trim": _saturate(0.05 * ((target_vref - (ias or target_vref)) / 20.0)),
        })
    elif _phase == "flare":
        _flare_started = True
        speed_err = (ias if ias is not None else target_vref) - target_vref
        sink_cmd = -150.0 if ra is not None and ra < 10.0 else -350.0
        if ra is not None and ra < 8.0:
            sink_cmd = -80.0
        out.update({
            "flaps": 40.0,
            "throttle_1": 0.0 if ra is not None and ra < 20.0 else _clamp(0.15 - 0.02 * speed_err, 0.0, 0.25),
            "throttle_2": 0.0 if ra is not None and ra < 20.0 else _clamp(0.15 - 0.02 * speed_err, 0.0, 0.25),
            "elevator": _saturate(0.22 + (-sink_cmd / 4000.0)),
            "aileron": _saturate(-0.35 * ((loc if loc is not None else 0.0) - target_loc)),
            "rudder": _saturate(-0.12 * ((loc if loc is not None else 0.0) - target_loc)),
            "elevator_trim": _saturate(0.08),
        })
    else:
        _touchdown_confirmed = True
        _rollout_timer += dt
        brake = 1.0 if (groundspeed or 0.0) > 80.0 else 0.7
        out.update({
            "throttle_1": 0.0,
            "throttle_2": 0.0,
            "elevator": -0.05,
            "aileron": _saturate(-0.10 * (loc if loc is not None else 0.0)),
            "rudder": _saturate(-0.20 * (loc if loc is not None else 0.0)),
            "elevator_trim": 0.0,
            "flaps": 40.0,
            "speedbrake": 1.0,
            "spoiler": 1.0,
            "autobrake": brake,
        })
        if (groundspeed or 999.0) < 50.0:
            out["reverse_1"] = 0.0
            out["reverse_2"] = 0.0
        else:
            out["reverse_1"] = 1.0 if state.get("engine_1_running", True) else 0.0
            out["reverse_2"] = 1.0 if state.get("engine_2_running", True) else 0.0

    if ra is not None:
        _last_ra = ra
    if _valid_number(vs):
        _last_vs = vs
    if _valid_number(loc):
        _last_loc = loc
    if _valid_number(gs):
        _last_gs = gs
    _last_time += dt
    if _valid_number(target_alt) and ra is not None and target_alt < 50 and ra < 5 and not wow:
        out["elevator"] = max(out.get("elevator", 0.0), 0.15)
    if ias is not None and ra is not None and ra < 30:
        if ias > target_vref + 15:
            out["elevator"] = _saturate(out.get("elevator", 0.0) + 0.05)
        elif ias < target_vref - 10:
            out["elevator"] = _saturate(out.get("elevator", 0.0) - 0.03)
    return out
