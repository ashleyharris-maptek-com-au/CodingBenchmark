#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <utility>
#include <vector>

using namespace std;

struct Rect {
    int w, h, idx;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct Shelf {
    int y = 0;
    int height = 0;
    int used = 0;
};

struct Board {
    vector<Shelf> shelves;
    int next_y = 0;
    int used_area = 0;
};

static inline int area_ll(int w, int h) {
    return w * h;
}

static bool can_place_dims(const Rect& r, int bw, int bh, int& rot) {
    if (r.w <= bw && r.h <= bh) {
        rot = 0;
        return true;
    }
    if (r.h <= bw && r.w <= bh) {
        rot = 1;
        return true;
    }
    return false;
}

static bool shelf_fit(const Shelf& s, const Rect& r, int bw, int bh, int& rot, int& pw, int& ph) {
    if (s.used + r.w <= bw && s.height >= r.h && s.y + s.height <= bh) {
        rot = 0;
        pw = r.w;
        ph = r.h;
        return true;
    }
    if (s.used + r.h <= bw && s.height >= r.w && s.y + s.height <= bh) {
        rot = 1;
        pw = r.h;
        ph = r.w;
        return true;
    }
    return false;
}

static bool open_shelf_fit(const Rect& r, int bw, int bh, int y, int& rot, int& pw, int& ph) {
    if (y + r.h <= bh && r.w <= bw) {
        rot = 0;
        pw = r.w;
        ph = r.h;
        return true;
    }
    if (y + r.w <= bh && r.h <= bw) {
        rot = 1;
        pw = r.h;
        ph = r.w;
        return true;
    }
    return false;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, BW, BH;
    if (!(cin >> N >> BW >> BH)) return 0;

    vector<Rect> rects;
    rects.reserve(N);
    for (int i = 0; i < N; ++i) {
        int w, h;
        cin >> w >> h;
        rects.push_back({w, h, i});
    }

    vector<Placement> ans(N);
    vector<Board> boards;
    boards.reserve(max(1, N / 4));

    vector<int> order(N);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        long long aa = 1LL * max(rects[a].w, rects[a].h) * max(rects[a].w, rects[a].h);
        long long ab = 1LL * max(rects[b].w, rects[b].h) * max(rects[b].w, rects[b].h);
        if (aa != ab) return aa > ab;
        long long area_a = 1LL * rects[a].w * rects[a].h;
        long long area_b = 1LL * rects[b].w * rects[b].h;
        if (area_a != area_b) return area_a > area_b;
        return rects[a].w + rects[a].h > rects[b].w + rects[b].h;
    });

    for (int id : order) {
        const Rect& r = rects[id];
        int best_board = -1, best_shelf = -1, best_rot = 0, best_x = 0, best_y = 0;
        int best_score = numeric_limits<int>::max();

        for (int b = 0; b < (int)boards.size(); ++b) {
            Board& bd = boards[b];
            for (int s = 0; s < (int)bd.shelves.size(); ++s) {
                const Shelf& sh = bd.shelves[s];
                int rot, pw, ph;
                if (!shelf_fit(sh, r, BW, BH, rot, pw, ph)) continue;
                int rem_width = BW - (sh.used + pw);
                int rem_height = BH - (sh.y + sh.height);
                int score = rem_height * BW + rem_width;
                if (score < best_score) {
                    best_score = score;
                    best_board = b;
                    best_shelf = s;
                    best_rot = rot;
                    best_x = sh.used;
                    best_y = sh.y;
                }
            }
            if (bd.next_y < BH) {
                int rot, pw, ph;
                if (open_shelf_fit(r, BW, BH, bd.next_y, rot, pw, ph)) {
                    int shelf_height = ph;
                    int remaining_height = BH - (bd.next_y + shelf_height);
                    int score = remaining_height * BW;
                    if (score < best_score) {
                        best_score = score;
                        best_board = b;
                        best_shelf = -1;
                        best_rot = rot;
                        best_x = 0;
                        best_y = bd.next_y;
                    }
                }
            }
        }

        if (best_board == -1) {
            boards.push_back(Board{});
            best_board = (int)boards.size() - 1;
            best_shelf = -1;
            best_x = 0;
            best_y = 0;
            int rot, pw, ph;
            if (!open_shelf_fit(r, BW, BH, 0, rot, pw, ph)) {
                cout << -1 << '\n';
                return 0;
            }
            best_rot = rot;
        }

        Board& bd = boards[best_board];
        int pw = best_rot ? r.h : r.w;
        int ph = best_rot ? r.w : r.h;
        if (best_shelf >= 0) {
            Shelf& sh = bd.shelves[best_shelf];
            sh.used += pw;
            ans[id] = {best_board, best_x, best_y, best_rot};
        } else {
            Shelf sh;
            sh.y = bd.next_y;
            sh.height = ph;
            sh.used = pw;
            bd.shelves.push_back(sh);
            bd.next_y += ph;
            ans[id] = {best_board, best_x, best_y, best_rot};
        }
        bd.used_area += pw * ph;
    }

    cout << boards.size() << '\n';
    for (int i = 0; i < N; ++i) {
        cout << ans[i].board << ' ' << ans[i].x << ' ' << ans[i].y << ' ' << ans[i].rotated << '\n';
    }
    return 0;
}
