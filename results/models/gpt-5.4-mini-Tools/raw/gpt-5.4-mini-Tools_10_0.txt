#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <vector>

using namespace std;

struct Rect {
    int w, h;
    int id;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct Shelf {
    int y = 0;
    int h = 0;
    int x = 0;
    int remaining = 0;
};

struct Board {
    int used_h = 0;
    vector<Shelf> shelves;
};

static inline bool fits(int w, int h, int bw, int bh) {
    return w <= bw && h <= bh;
}

static inline int scoreShelf(const Shelf& s, int w, int h, int bw, int bh) {
    if (w > s.remaining || h > s.h) return numeric_limits<int>::max();
    int waste_width = s.remaining - w;
    int waste_height = s.h - h;
    int board_headroom = bh - (s.y + s.h);
    int fit_penalty = 0;
    if (waste_height > 0) fit_penalty += 2;
    if (waste_width > 0) fit_penalty += 1;
    return waste_height * 1000000 + waste_width * 1000 + board_headroom * 10 + fit_penalty;
}

static inline int scoreBoardForNewShelf(const Board& b, int h, int bh) {
    if (b.used_h + h > bh) return numeric_limits<int>::max();
    int leftover = bh - (b.used_h + h);
    return leftover;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, BW, BH;
    if (!(cin >> N >> BW >> BH)) return 0;

    vector<Rect> rects(N);
    for (int i = 0; i < N; ++i) {
        cin >> rects[i].w >> rects[i].h;
        rects[i].id = i;
    }

    vector<int> order(N);
    iota(order.begin(), order.end(), 0);

    sort(order.begin(), order.end(), [&](int a, int b) {
        long long areaA = 1LL * rects[a].w * rects[a].h;
        long long areaB = 1LL * rects[b].w * rects[b].h;
        int maxA = max(rects[a].w, rects[a].h);
        int maxB = max(rects[b].w, rects[b].h);
        if (maxA != maxB) return maxA > maxB;
        if (areaA != areaB) return areaA > areaB;
        int minA = min(rects[a].w, rects[a].h);
        int minB = min(rects[b].w, rects[b].h);
        if (minA != minB) return minA > minB;
        return a < b;
    });

    vector<Board> boards;
    vector<Placement> ans(N);
    boards.reserve(max(1, N / 8));

    for (int idx : order) {
        int w0 = rects[idx].w, h0 = rects[idx].h;
        int w1 = rects[idx].h, h1 = rects[idx].w;

        struct Choice {
            int type = 0; // 0 = shelf, 1 = new shelf
            int board = -1;
            int shelf = -1;
            int rotated = 0;
            int score = numeric_limits<int>::max();
            int x = 0;
            int y = 0;
            int w = 0;
            int h = 0;
        } best;

        auto consider = [&](int boardIndex, int shelfIndex, int w, int h, int rotated) {
            const Board& b = boards[boardIndex];
            const Shelf& s = b.shelves[shelfIndex];
            int sc = scoreShelf(s, w, h, BW, BH);
            if (sc < best.score) {
                best.type = 0;
                best.board = boardIndex;
                best.shelf = shelfIndex;
                best.rotated = rotated;
                best.score = sc;
                best.x = s.x;
                best.y = s.y;
                best.w = w;
                best.h = h;
            }
        };

        auto considerNewShelf = [&](int boardIndex, int w, int h, int rotated) {
            const Board& b = boards[boardIndex];
            int sc = scoreBoardForNewShelf(b, h, BH);
            if (w > BW || h > BH) return;
            if (b.used_h + h > BH) return;
            sc = sc * 1000 + (BW - w);
            if (sc < best.score) {
                best.type = 1;
                best.board = boardIndex;
                best.shelf = -1;
                best.rotated = rotated;
                best.score = sc;
                best.x = 0;
                best.y = b.used_h;
                best.w = w;
                best.h = h;
            }
        };

        for (int b = 0; b < (int)boards.size(); ++b) {
            const Board& bd = boards[b];
            for (int s = 0; s < (int)bd.shelves.size(); ++s) {
                consider(b, s, w0, h0, 0);
                if (w1 != w0 || h1 != h0) consider(b, s, w1, h1, 1);
            }
            considerNewShelf(b, w0, h0, 0);
            if (w1 != w0 || h1 != h0) considerNewShelf(b, w1, h1, 1);
        }

        if (best.board == -1) {
            Board b;
            boards.push_back(std::move(b));
            int bi = (int)boards.size() - 1;
            if (fits(w0, h0, BW, BH) && (!fits(w1, h1, BW, BH) ||
                h0 < h1 || (h0 == h1 && w0 <= w1))) {
                boards[bi].used_h = h0;
                boards[bi].shelves.push_back(Shelf{0, h0, w0, BW - w0});
                ans[idx] = {bi, 0, 0, 0};
            } else {
                boards[bi].used_h = h1;
                boards[bi].shelves.push_back(Shelf{0, h1, w1, BW - w1});
                ans[idx] = {bi, 0, 0, 1};
            }
            continue;
        }

        Board& b = boards[best.board];
        if (best.type == 0) {
            Shelf& s = b.shelves[best.shelf];
            ans[idx] = {best.board, s.x, s.y, best.rotated};
            s.x += best.w;
            s.remaining -= best.w;
        } else {
            int y = b.used_h;
            b.used_h += best.h;
            b.shelves.push_back(Shelf{y, best.h, best.w, BW - best.w});
            ans[idx] = {best.board, 0, y, best.rotated};
        }

    }

    cout << boards.size() << '\n';
    for (int i = 0; i < N; ++i) {
        cout << ans[i].board << ' ' << ans[i].x << ' ' << ans[i].y << ' ' << ans[i].rotated << '\n';
    }
    return 0;
}
