using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    private struct Point
    {
        public int X;
        public int Y;
        public Point(int x, int y) { X = x; Y = y; }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = ReadByte(); } while (c <= ' ' && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }
            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return val * sign;
        }

        public string NextRow(int width)
        {
            char[] row = new char[width];
            int filled = 0;
            while (filled < width)
            {
                int c = ReadByte();
                if (c < 0) break;
                if (c == '\r') continue;
                if (c == '\n')
                {
                    if (filled == 0) continue;
                    break;
                }
                row[filled++] = (char)c;
            }
            while (filled < width) row[filled++] = ' ';
            return new string(row);
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int h = fs.NextInt();
        int w = fs.NextInt();

        var board = new char[h][];
        for (int y = 0; y < h; y++) board[y] = fs.NextRow(w).ToCharArray();

        var safe = Solve(board, h, w);

        var sb = new System.Text.StringBuilder();
        sb.AppendLine(safe.Count.ToString());
        foreach (var p in safe)
        {
            sb.Append(p.X).Append(' ').Append(p.Y).AppendLine();
        }
        Console.Write(sb.ToString());
    }

    private static List<Point> Solve(char[][] board, int h, int w)
    {
        var result = new HashSet<long>();
        bool largeBoard = (long)h * w >= 2_000_000L;
        LocalDeductions(board, h, w, result);
        if (!largeBoard) ComponentDeductions(board, h, w, result);

        var list = new List<Point>(result.Count);
        foreach (long key in result)
        {
            int x = (int)(key >> 32);
            int y = (int)key;
            list.Add(new Point(x, y));
        }
        list.Sort((a, b) => a.Y != b.Y ? a.Y.CompareTo(b.Y) : a.X.CompareTo(b.X));
        return list;
    }

    private static void LocalDeductions(char[][] board, int h, int w, HashSet<long> safe)
    {
        bool changed;
        do
        {
            changed = false;
            for (int y = 0; y < h; y++)
            {
                for (int x = 0; x < w; x++)
                {
                    char c = board[y][x];
                    if (c < '0' || c > '8') continue;
                    int clue = c - '0';
                    int flagged = 0;
                    var unknowns = new List<Point>(8);
                    for (int dy = -1; dy <= 1; dy++)
                    {
                        int ny = y + dy;
                        if (ny < 0 || ny >= h) continue;
                        for (int dx = -1; dx <= 1; dx++)
                        {
                            int nx = x + dx;
                            if (dx == 0 && dy == 0) continue;
                            if (nx < 0 || nx >= w) continue;
                            char nc = board[ny][nx];
                            if (nc == '*') flagged++;
                            else if (nc == ' ') unknowns.Add(new Point(nx, ny));
                        }
                    }
                    if (unknowns.Count == 0) continue;
                    if (flagged == clue)
                    {
                        foreach (var p in unknowns)
                        {
                            long key = ((long)p.X << 32) | (uint)p.Y;
                            if (safe.Add(key)) changed = true;
                        }
                    }
                }
            }
        } while (changed);
    }

    private static void ComponentDeductions(char[][] board, int h, int w, HashSet<long> safe)
    {
        var frontierIndex = new Dictionary<long, int>();
        var frontierCells = new List<Point>();
        var clues = new List<Clue>();

        for (int y = 0; y < h; y++)
        {
            for (int x = 0; x < w; x++)
            {
                char c = board[y][x];
                if (c < '0' || c > '8') continue;
                int clue = c - '0';
                int flagged = 0;
                var vars = new List<int>(8);
                for (int dy = -1; dy <= 1; dy++)
                {
                    int ny = y + dy;
                    if (ny < 0 || ny >= h) continue;
                    for (int dx = -1; dx <= 1; dx++)
                    {
                        int nx = x + dx;
                        if (dx == 0 && dy == 0) continue;
                        if (nx < 0 || nx >= w) continue;
                        char nc = board[ny][nx];
                        if (nc == '*') flagged++;
                        else if (nc == ' ')
                        {
                            long key = ((long)nx << 32) | (uint)ny;
                            if (!frontierIndex.TryGetValue(key, out int idx))
                            {
                                idx = frontierCells.Count;
                                frontierIndex[key] = idx;
                                frontierCells.Add(new Point(nx, ny));
                            }
                            vars.Add(idx);
                        }
                    }
                }
                if (vars.Count == 0) continue;
                int rem = clue - flagged;
                if (rem < 0 || rem > vars.Count) continue;
                clues.Add(new Clue { Cells = vars.ToArray(), Remaining = rem });
            }
        }

        if (frontierCells.Count == 0 || clues.Count == 0) return;

        var cellToClues = new List<int>[frontierCells.Count];
        for (int i = 0; i < cellToClues.Length; i++) cellToClues[i] = new List<int>();
        for (int i = 0; i < clues.Count; i++)
            foreach (int v in clues[i].Cells)
                cellToClues[v].Add(i);

        var components = BuildComponents(frontierCells.Count, clues, cellToClues);

        foreach (var comp in components)
        {
            if (comp.Count == 0) continue;
            if (comp.Count > 22) continue;
            DeduceComponent(comp, clues, cellToClues, frontierCells, safe);
        }
    }

    private static List<List<int>> BuildComponents(int n, List<Clue> clues, List<int>[] cellToClues)
    {
        var visited = new bool[n];
        var comps = new List<List<int>>();
        var q = new Queue<int>();
        var clueVisited = new bool[clues.Count];
        for (int i = 0; i < n; i++)
        {
            if (visited[i]) continue;
            visited[i] = true;
            q.Enqueue(i);
            var comp = new List<int>();
            while (q.Count > 0)
            {
                int u = q.Dequeue();
                comp.Add(u);
                foreach (int clueIdx in cellToClues[u])
                {
                    if (clueVisited[clueIdx]) continue;
                    clueVisited[clueIdx] = true;
                    foreach (int v in clues[clueIdx].Cells)
                    {
                        if (visited[v]) continue;
                        visited[v] = true;
                        q.Enqueue(v);
                    }
                }
            }
            comps.Add(comp);
        }
        return comps;
    }

    private static void DeduceComponent(List<int> comp, List<Clue> clues, List<int>[] cellToClues, List<Point> frontierCells, HashSet<long> safe)
    {
        var compSet = new HashSet<int>(comp);
        var relevantClues = new HashSet<int>();
        foreach (int cell in comp)
            foreach (int clue in cellToClues[cell])
                relevantClues.Add(clue);

        var clueList = new List<ClueState>();
        foreach (int idx in relevantClues)
        {
            var c = clues[idx];
            var vars = new List<int>();
            foreach (int v in c.Cells)
                if (compSet.Contains(v))
                    vars.Add(v);
            if (vars.Count == 0) continue;
            clueList.Add(new ClueState { Cells = vars.ToArray(), Remaining = c.Remaining });
        }

        var cells = comp.ToArray();
        int n = cells.Length;
        if (n == 0 || n > 22) return;

        var indexMap = new Dictionary<int, int>();
        for (int i = 0; i < n; i++) indexMap[cells[i]] = i;

        var remClues = new List<(int[] cells, int remaining)>();
        foreach (var cs in clueList)
        {
            var mapped = new int[cs.Cells.Length];
            for (int i = 0; i < cs.Cells.Length; i++) mapped[i] = indexMap[cs.Cells[i]];
            remClues.Add((mapped, cs.Remaining));
        }

        ulong total = 1UL << n;
        var possibleMine = new bool[n];
        bool any = false;

        for (ulong mask = 0; mask < total; mask++)
        {
            bool ok = true;
            foreach (var clue in remClues)
            {
                int count = 0;
                foreach (int v in clue.cells)
                    if (((mask >> v) & 1UL) != 0) count++;
                if (count != clue.remaining) { ok = false; break; }
            }
            if (!ok) continue;
            any = true;
            for (int i = 0; i < n; i++)
                if (((mask >> i) & 1UL) != 0) possibleMine[i] = true;
        }

        if (!any) return;
        for (int i = 0; i < n; i++)
        {
            if (!possibleMine[i])
            {
                var p = frontierCells[cells[i]];
                safe.Add(((long)p.X << 32) | (uint)p.Y);
            }
        }
    }

    private sealed class Clue
    {
        public int[] Cells = Array.Empty<int>();
        public int Remaining;
    }

    private sealed class ClueState
    {
        public int[] Cells = Array.Empty<int>();
        public int Remaining;
    }
}
