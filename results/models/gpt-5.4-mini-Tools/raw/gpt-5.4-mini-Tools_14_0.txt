using System;
using System.Collections.Generic;
using System.IO;

public class Program
{
    private const int ExactComponentLimit = 22;

    private readonly record struct Pt(int X, int Y);

    private sealed class Constraint
    {
        public int[] Cells = Array.Empty<int>();
        public int Required;
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int h = fs.NextInt();
        int w = fs.NextInt();
        var board = new char[h][];
        for (int y = 0; y < h; y++) board[y] = fs.NextRow(w);

        var safe = Solve(board, h, w);
        using var sw = new StreamWriter(Console.OpenStandardOutput());
        sw.WriteLine(safe.Count);
        foreach (var p in safe) sw.WriteLine($"{p.X} {p.Y}");
    }

    private static List<Pt> Solve(char[][] board, int h, int w)
    {
        int[] dx = { -1, 0, 1, -1, 1, -1, 0, 1 };
        int[] dy = { -1, -1, -1, 0, 0, 1, 1, 1 };
        int n = h * w;
        int Idx(int x, int y) => y * w + x;

        var safe = new bool[h, w];
        var clueIds = new List<int>();

        int[] clueMineAdj = new int[n];
        int[] clueUnknownAdj = new int[n];
        var clueUnknownCells = new List<int>[n];
        for (int i = 0; i < n; i++) clueUnknownCells[i] = new List<int>(8);

        for (int y = 0; y < h; y++)
        {
            for (int x = 0; x < w; x++)
            {
                char c = board[y][x];
                if (c < '0' || c > '8') continue;
                int id = Idx(x, y);
                clueIds.Add(id);
                int mines = 0, unknown = 0;
                for (int dir = 0; dir < 8; dir++)
                {
                    int nx = x + dx[dir], ny = y + dy[dir];
                    if ((uint)nx >= (uint)w || (uint)ny >= (uint)h) continue;
                    char nc = board[ny][nx];
                    if (nc == '*') mines++;
                    else if (nc == ' ')
                    {
                        unknown++;
                        clueUnknownCells[id].Add(Idx(nx, ny));
                    }
                }
                clueMineAdj[id] = mines;
                clueUnknownAdj[id] = unknown;
                if (c - '0' - mines == 0)
                {
                    foreach (int cell in clueUnknownCells[id])
                        safe[cell / w, cell % w] = true;
                }
            }
        }

        bool changed;
        do
        {
            changed = false;
            foreach (int id in clueIds)
            {
                int y = id / w, x = id % w;
                char c = board[y][x];
                int need = c - '0' - clueMineAdj[id];
                if (need < 0) continue;

                int unknown = 0;
                clueUnknownCells[id].Clear();
                for (int dir = 0; dir < 8; dir++)
                {
                    int nx = x + dx[dir], ny = y + dy[dir];
                    if ((uint)nx >= (uint)w || (uint)ny >= (uint)h) continue;
                    if (board[ny][nx] == ' ' && !safe[ny, nx])
                    {
                        unknown++;
                        clueUnknownCells[id].Add(Idx(nx, ny));
                    }
                }
                clueUnknownAdj[id] = unknown;
                if (need == 0)
                {
                    foreach (int cell in clueUnknownCells[id])
                    {
                        int cy = cell / w, cx = cell % w;
                        if (!safe[cy, cx])
                        {
                            safe[cy, cx] = true;
                            changed = true;
                        }
                    }
                }
            }

            if (changed) continue;

            var vars = new List<int>();
            var varIndex = new Dictionary<int, int>();
            for (int y = 0; y < h; y++)
            {
                for (int x = 0; x < w; x++)
                {
                    if (board[y][x] != ' ' || safe[y, x]) continue;
                    bool nearClue = false;
                    for (int dir = 0; dir < 8 && !nearClue; dir++)
                    {
                        int nx = x + dx[dir], ny = y + dy[dir];
                        if ((uint)nx >= (uint)w || (uint)ny >= (uint)h) continue;
                        if (board[ny][nx] >= '0' && board[ny][nx] <= '8') nearClue = true;
                    }
                    if (nearClue)
                    {
                        int id = Idx(x, y);
                        varIndex[id] = vars.Count;
                        vars.Add(id);
                    }
                }
            }

            if (vars.Count == 0) break;

            var constraints = new List<Constraint>();
            var varToCons = new List<int>[vars.Count];
            for (int i = 0; i < vars.Count; i++) varToCons[i] = new List<int>();
            var constraintMap = new Dictionary<string, int>();

            foreach (int clueId in clueIds)
            {
                int y = clueId / w, x = clueId % w;
                int required = board[y][x] - '0';
                var cells = new List<int>();
                for (int dir = 0; dir < 8; dir++)
                {
                    int nx = x + dx[dir], ny = y + dy[dir];
                    if ((uint)nx >= (uint)w || (uint)ny >= (uint)h) continue;
                    char nc = board[ny][nx];
                    if (nc == '*') required--;
                    else if (nc == ' ' && !safe[ny, nx])
                    {
                        int cid = Idx(nx, ny);
                        if (varIndex.TryGetValue(cid, out int vi)) cells.Add(vi);
                    }
                }
                if (cells.Count == 0) continue;
                cells.Sort();
                int u = 0;
                for (int i = 0; i < cells.Count; i++)
                    if (i == 0 || cells[i] != cells[i - 1]) cells[u++] = cells[i];
                cells.RemoveRange(u, cells.Count - u);
                if (cells.Count == 0) continue;
                string key = string.Join(",", cells);
                if (!constraintMap.ContainsKey(key))
                {
                    int cid = constraints.Count;
                    constraintMap[key] = cid;
                    constraints.Add(new Constraint { Cells = cells.ToArray(), Required = required });
                    foreach (int v in cells) varToCons[v].Add(cid);
                }
            }

            var comps = BuildComponents(varToCons);
            foreach (var comp in comps)
            {
                if (comp.Count == 0 || comp.Count > ExactComponentLimit) continue;
                if (TryExact(comp, constraints, vars, varToCons, safe, out var newlySafe))
                {
                    foreach (var p in newlySafe)
                    {
                        if (!safe[p.Y, p.X])
                        {
                            safe[p.Y, p.X] = true;
                            changed = true;
                        }
                    }
                }
            }
        } while (changed);

        var ans = new List<Pt>();
        for (int y = 0; y < h; y++)
            for (int x = 0; x < w; x++)
                if (safe[y, x] && board[y][x] == ' ')
                    ans.Add(new Pt(x, y));
        return ans;
    }

    private static List<List<int>> BuildComponents(List<int>[] varToCons)
    {
        int n = varToCons.Length;
        var visited = new bool[n];
        var comps = new List<List<int>>();
        var consToVars = new Dictionary<int, List<int>>();
        for (int v = 0; v < n; v++)
            foreach (int c in varToCons[v])
            {
                if (!consToVars.TryGetValue(c, out var lst))
                    consToVars[c] = lst = new List<int>();
                lst.Add(v);
            }

        for (int i = 0; i < n; i++)
        {
            if (visited[i]) continue;
            var q = new Queue<int>();
            var comp = new List<int>();
            q.Enqueue(i);
            visited[i] = true;
            while (q.Count > 0)
            {
                int v = q.Dequeue();
                comp.Add(v);
                foreach (int c in varToCons[v])
                {
                    if (!consToVars.TryGetValue(c, out var neigh)) continue;
                    foreach (int nv in neigh)
                    {
                        if (!visited[nv])
                        {
                            visited[nv] = true;
                            q.Enqueue(nv);
                        }
                    }
                }
            }
            comps.Add(comp);
        }
        return comps;
    }

    private static bool TryExact(List<int> comp, List<Constraint> constraints, List<int>[] vars, List<int>[] varToCons, bool[,] safe, out List<Pt> newlySafe)
    {
        newlySafe = new List<Pt>();
        if (comp.Count == 0 || comp.Count > ExactComponentLimit) return false;

        var pos = new Dictionary<int, int>();
        for (int i = 0; i < comp.Count; i++) pos[comp[i]] = i;

        var relevant = new List<int>();
        var inComp = new HashSet<int>(comp);
        foreach (int ci in EnumerableRange(constraints.Count))
        {
            foreach (int v in constraints[ci].Cells)
                if (inComp.Contains(v)) { relevant.Add(ci); break; }
        }

        int n = comp.Count;
        int[] assign = new int[n];
        Array.Fill(assign, -1);
        bool[] allZero = new bool[n];
        bool[] allOne = new bool[n];
        Array.Fill(allZero, true);
        Array.Fill(allOne, true);
        int solutions = 0;

        bool CheckPartial()
        {
            foreach (int ci in relevant)
            {
                int assigned = 0, unknown = 0;
                foreach (int v in constraints[ci].Cells)
                {
                    if (!pos.TryGetValue(v, out int pi)) continue;
                    int a = assign[pi];
                    if (a == 1) assigned++;
                    else if (a == -1) unknown++;
                }
                int req = constraints[ci].Required;
                if (assigned > req) return false;
                if (assigned + unknown < req) return false;
            }
            return true;
        }

        void Dfs(int idx)
        {
            if (solutions > 4096) return;
            if (!CheckPartial()) return;
            if (idx == n)
            {
                foreach (int ci in relevant)
                {
                    int sum = 0;
                    foreach (int v in constraints[ci].Cells)
                        if (pos.TryGetValue(v, out int pi) && assign[pi] == 1) sum++;
                    if (sum != constraints[ci].Required) return;
                }
                solutions++;
                for (int i = 0; i < n; i++)
                {
                    if (assign[i] == 0) allOne[i] = false;
                    else allZero[i] = false;
                }
                return;
            }

            assign[idx] = 0;
            Dfs(idx + 1);
            assign[idx] = 1;
            Dfs(idx + 1);
            assign[idx] = -1;
        }

        Dfs(0);
        if (solutions == 0) return false;

        for (int i = 0; i < n; i++)
        {
            if (allZero[i])
            {
                int id = comp[i];
                newlySafe.Add(new Pt(id % (safe.GetLength(1)), id / safe.GetLength(1)));
            }
        }
        return newlySafe.Count > 0;
    }

    private static IEnumerable<int> EnumerableRange(int n)
    {
        for (int i = 0; i < n; i++) yield return i;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream) => _stream = stream;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }

        public char[] NextRow(int w)
        {
            var row = new char[w];
            int i = 0;
            while (i < w)
            {
                int c = Read();
                if (c == '\r') continue;
                if (c == '\n')
                {
                    if (i == 0) continue;
                    while (i < w) row[i++] = ' ';
                    break;
                }
                row[i++] = (char)c;
            }
            return row;
        }
    }
}
