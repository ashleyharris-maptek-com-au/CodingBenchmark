OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1
               OpSource GLSL 450

               OpName %main "main"
               OpName %gl_PerVertex_in "gl_PerVertex"
               OpMemberName %gl_PerVertex_in 0 "gl_Position"
               OpName %gl_in "gl_in"
               OpName %worldPos "worldPos"
               OpName %normal "normal"
               OpName %color "color"
               OpName %gl_PerVertex_out "gl_PerVertex"
               OpMemberName %gl_PerVertex_out 0 "gl_Position"
               OpName %gl_out "gl_out"
               OpName %outColor "outColor"
               OpName %ubo "ubo"
               OpMemberName %ubo 0 "model"
               OpMemberName %ubo 1 "view"
               OpMemberName %ubo 2 "proj"
               OpMemberName %ubo 3 "lightPos"
               OpMemberName %ubo 4 "cameraPos"
               OpMemberName %ubo 5 "params"
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_in Block
               OpDecorate %gl_in BuiltIn Position
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpDecorate %gl_out BuiltIn Position
               OpDecorate %outColor Location 0
               OpMemberDecorate %ubo 0 ColMajor
               OpMemberDecorate %ubo 0 Offset 0
               OpMemberDecorate %ubo 1 ColMajor
               OpMemberDecorate %ubo 1 Offset 64
               OpMemberDecorate %ubo 2 ColMajor
               OpMemberDecorate %ubo 2 Offset 128
               OpMemberDecorate %ubo 3 Offset 192
               OpMemberDecorate %ubo 4 Offset 204
               OpMemberDecorate %ubo 5 Offset 216
               OpDecorate %ubo Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

%void          = OpTypeVoid
%func          = OpTypeFunction %void
%float         = OpTypeFloat 32
%v2float       = OpTypeVector %float 2
%v3float       = OpTypeVector %float 3
%v4float       = OpTypeVector %float 4
%uint          = OpTypeInt 32 0
%int           = OpTypeInt 32 1
%int_0         = OpConstant %int 0
%int_1         = OpConstant %int 1
%int_2         = OpConstant %int 2
%int_3         = OpConstant %int 3
%uint_1        = OpConstant %uint 1
%float_0       = OpConstant %float 0.0
%float_05      = OpConstant %float 0.05
%v2_05_0       = OpConstantComposite %v2float %float_05 %float_0
%v2_0_05       = OpConstantComposite %v2float %float_0 %float_05
%v2_n05_0      = OpConstantComposite %v2float %float_05 %float_0
%v2_0_n05      = OpConstantComposite %v2float %float_0 %float_05
%v4_0         = OpConstantComposite %v4float %float_0 %float_0 %float_0 %float_0
%v4_1         = OpConstantComposite %v4float %float_0 %float_0 %float_0 %float_0
%v4_005       = OpConstantComposite %v4float %float_0 %float_0 %float_0 %float_0
%pv_in         = OpTypeStruct %v4float
%arr_pv_in     = OpTypeArray %pv_in %uint_1
%ptr_in_arr_pv  = OpTypePointer Input %arr_pv_in
%ptr_in_pv     = OpTypePointer Input %pv_in
%ptr_in_v4     = OpTypePointer Input %v4float
%arr_v3_in     = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3     = OpTypePointer Input %v3float
%pv_out        = OpTypeStruct %v4float
%ptr_out_pv    = OpTypePointer Output %pv_out
%ptr_out_v4    = OpTypePointer Output %v4float
%ptr_out_v3    = OpTypePointer Output %v3float
%ubo_struct    = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v3float %v3float %v4float
%ptr_ubo       = OpTypePointer Uniform %ubo_struct
%ptr_ubo_v4    = OpTypePointer Uniform %v4float
%mat4v4float   = OpTypeMatrix %v4float 4

%gl_in         = OpVariable %ptr_in_arr_pv Input
%worldPos      = OpVariable %ptr_in_arr_v3 Input
%normal       = OpVariable %ptr_in_arr_v3 Input
%color        = OpVariable %ptr_in_arr_v3 Input
%gl_out        = OpVariable %ptr_out_pv Output
%outColor      = OpVariable %ptr_out_v3 Output
%ubo           = OpVariable %ptr_ubo Uniform

%main = OpFunction %void None %func
%label = OpLabel
%pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr
%col_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%col = OpLoad %v3float %col_ptr
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1
%z = OpCompositeExtract %float %pos 2
%w = OpCompositeExtract %float %pos 3
%top_y = OpFAdd %float %y %float_05
%bot_y = OpFSub %float %y %float_05
%right_x = OpFAdd %float %x %float_05
%left_x = OpFSub %float %x %float_05
%top = OpCompositeConstruct %v4float %x %top_y %z %w
%right = OpCompositeConstruct %v4float %right_x %y %z %w
%left = OpCompositeConstruct %v4float %left_x %y %z %w
%bottom = OpCompositeConstruct %v4float %x %bot_y %z %w

%out_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos0 %top
OpStore %outColor %col
OpEmitVertex

%out_pos1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos1 %right
OpStore %outColor %col
OpEmitVertex

%out_pos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos2 %left
OpStore %outColor %col
OpEmitVertex

%out_pos3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos3 %bottom
OpStore %outColor %col
OpEmitVertex

OpEndPrimitive
OpReturn
