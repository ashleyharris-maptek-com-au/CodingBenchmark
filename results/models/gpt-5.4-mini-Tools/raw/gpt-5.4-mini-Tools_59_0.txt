const fs = require('fs');

function solve(data) {
  const grid = data.probability || [];
  const drones = data.drones || [];
  const maxSteps = Math.max(0, data.maxStepsPerDrone | 0);
  const h = grid.length;
  const w = h ? grid[0].length : 0;
  const n = drones.length;

  const paths = drones.map(d => [[d.x, d.y]]);
  if (!h || !w || n === 0 || maxSteps === 0) return { paths };

  const dirs = [[0,0],[1,0],[-1,0],[0,1],[0,-1]];
  const inBounds = (x, y) => x >= 0 && y >= 0 && x < w && y < h;
  const idx = (x, y) => y * w + x;

  const visited = new Set();
  const startKeys = new Set();
  for (const d of drones) {
    if (inBounds(d.x, d.y) && grid[d.y][d.x] !== -1) {
      visited.add(idx(d.x, d.y));
      startKeys.add(idx(d.x, d.y));
    }
  }

  function cellScore(x, y) {
    const base = grid[y][x];
    if (base <= 0) return 0;
    let bonus = 0;
    for (let dy = -1; dy <= 1; dy++) {
      for (let dx = -1; dx <= 1; dx++) {
        if (!dx && !dy) continue;
        const nx = x + dx, ny = y + dy;
        if (inBounds(nx, ny) && grid[ny][nx] !== -1) bonus += Math.max(0, grid[ny][nx]);
      }
    }
    return base + 0.15 * bonus;
  }

  const goalCells = [];
  for (let y = 0; y < h; y++) {
    for (let x = 0; x < w; x++) {
      if (grid[y][x] > 0) goalCells.push([cellScore(x, y), x, y]);
    }
  }
  goalCells.sort((a, b) => b[0] - a[0]);

  const targets = goalCells.slice(0, Math.min(goalCells.length, Math.max(20, n * 8)));

  const occupiedAt = Array.from({ length: maxSteps + 1 }, () => new Set());
  for (let i = 0; i < n; i++) occupiedAt[0].add(idx(drones[i].x, drones[i].y));

  function bfsPath(sx, sy, tx, ty, forbiddenTimes) {
    const qx = new Int32Array(w * h * 8 + 8);
    const qy = new Int32Array(qx.length);
    const qt = new Int32Array(qx.length);
    const prev = new Map();
    let head = 0, tail = 0;
    qx[tail] = sx; qy[tail] = sy; qt[tail] = 0; tail++;
    prev.set(`${sx},${sy},0`, null);
    while (head < tail) {
      const x = qx[head], y = qy[head], t = qt[head]; head++;
      if (x === tx && y === ty) {
        const path = [];
        let key = `${x},${y},${t}`;
        while (key) {
          const [kx, ky, kt] = key.split(',').map(Number);
          path.push([kx, ky]);
          key = prev.get(key);
        }
        return path.reverse();
      }
      if (t >= maxSteps) continue;
      for (const [dx, dy] of dirs) {
        const nx = x + dx, ny = y + dy, nt = t + 1;
        if (!inBounds(nx, ny) || grid[ny][nx] === -1) continue;
        const nk = `${nx},${ny},${nt}`;
        if (prev.has(nk)) continue;
        if (forbiddenTimes && forbiddenTimes.has(`${nx},${ny},${nt}`)) continue;
        prev.set(nk, `${x},${y},${t}`);
        qx[tail] = nx; qy[tail] = ny; qt[tail] = nt; tail++;
        if (tail >= qx.length) break;
      }
    }
    return null;
  }

  const claimed = new Set(startKeys);
  const plans = drones.map((d, i) => {
    const path = [[d.x, d.y]];
    let x = d.x, y = d.y;
    let t = 0;
    while (t < maxSteps) {
      let best = null;
      let bestGain = -1;
      for (const [score, tx, ty] of targets) {
        const md = Math.abs(tx - x) + Math.abs(ty - y);
        if (md > maxSteps - t) continue;
        const gain = score - md * 0.01 - (claimed.has(idx(tx, ty)) ? 0.5 : 0);
        if (gain > bestGain) {
          bestGain = gain;
          best = [tx, ty];
        }
      }
      let next = null;
      if (best) {
        const fb = new Set();
        for (let tt = 0; tt <= maxSteps; tt++) {
          for (const v of occupiedAt[tt]) fb.add(`${v % w},${Math.floor(v / w)},${tt}`);
        }
        const p = bfsPath(x, y, best[0], best[1], fb);
        if (p && p.length > 1) next = p[1];
      }
      if (!next) {
        let bestCell = [x, y];
        let bestVal = -1;
        for (const [dx, dy] of dirs) {
          const nx = x + dx, ny = y + dy;
          if (!inBounds(nx, ny) || grid[ny][nx] === -1) continue;
          const val = grid[ny][nx] > 0 && !visited.has(idx(nx, ny)) ? grid[ny][nx] : 0;
          if (val > bestVal) { bestVal = val; bestCell = [nx, ny]; }
        }
        next = bestCell;
      }
      x = next[0]; y = next[1]; t++;
      path.push([x, y]);
      occupiedAt[t].add(idx(x, y));
      visited.add(idx(x, y));
      claimed.add(idx(x, y));
    }
    return path;
  });

  return { paths: plans };
}

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);
process.stdout.write(JSON.stringify(solve(data)));
