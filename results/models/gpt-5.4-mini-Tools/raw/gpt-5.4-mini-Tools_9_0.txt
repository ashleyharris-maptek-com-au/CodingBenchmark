using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream) => _stream = stream;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = Read(); } while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }
            return value * sign;
        }
    }

    private struct Cut
    {
        public int Length;
        public int Index;
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int stockLength = fs.NextInt();

        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = new Cut { Length = fs.NextInt(), Index = i };
        }

        Array.Sort(cuts, (a, b) =>
        {
            int cmp = b.Length.CompareTo(a.Length);
            if (cmp != 0) return cmp;
            return a.Index.CompareTo(b.Index);
        });

        var binRem = new List<int>(n);
        var cutToBin = new int[n];

        // Maintain bins in descending remaining-capacity order:
        // each entry is (remaining, binId). We scan only a small prefix for best fit.
        var ordered = new List<(int remaining, int binId)>();

        for (int i = 0; i < n; i++)
        {
            int len = cuts[i].Length;
            int cutIndex = cuts[i].Index;

            int bestPos = -1;
            int bestRemainingAfter = int.MaxValue;

            // Search a bounded prefix for a good fit; this keeps packing quality decent
            // while avoiding expensive full scans on huge instances.
            int scanLimit = ordered.Count < 64 ? ordered.Count : 64;
            for (int j = 0; j < scanLimit; j++)
            {
                int rem = ordered[j].remaining;
                if (rem >= len)
                {
                    int after = rem - len;
                    if (after < bestRemainingAfter)
                    {
                        bestRemainingAfter = after;
                        bestPos = j;
                        if (after == 0) break;
                    }
                }
            }

            int binId;
            if (bestPos >= 0)
            {
                binId = ordered[bestPos].binId;
                int newRemaining = ordered[bestPos].remaining - len;
                ordered.RemoveAt(bestPos);

                int insertAt = ordered.Count;
                while (insertAt > 0 && ordered[insertAt - 1].remaining < newRemaining)
                    insertAt--;
                ordered.Insert(insertAt, (newRemaining, binId));
            }
            else
            {
                binId = binRem.Count;
                binRem.Add(stockLength);
                int newRemaining = stockLength - len;

                int insertAt = ordered.Count;
                while (insertAt > 0 && ordered[insertAt - 1].remaining < newRemaining)
                    insertAt--;
                ordered.Insert(insertAt, (newRemaining, binId));
            }

            cutToBin[cutIndex] = binId;
        }

        // Reconstruct per-stock cut lists.
        var stockCuts = new List<int>[binRem.Count];
        for (int i = 0; i < stockCuts.Length; i++) stockCuts[i] = new List<int>();
        for (int i = 0; i < n; i++)
        {
            int binId = cutToBin[i];
            stockCuts[binId].Add(i);
        }

        var sb = new StringBuilder();
        sb.AppendLine(binRem.Count.ToString());
        for (int i = 0; i < stockCuts.Length; i++)
        {
            var list = stockCuts[i];
            for (int j = 0; j < list.Count; j++)
            {
                if (j > 0) sb.Append(' ');
                sb.Append(list[j]);
            }
            sb.AppendLine();
        }

        Console.Write(sb.ToString());
    }
}
