; SPIR-V
; Geometry shader: tapered ribbon from line input
OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpSource GLSL 450

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_005 = OpConstant %float 0.005
%float_04 = OpConstant %float 0.04

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%ptr_in_v4 = OpTypePointer Input %v4float

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

OpDecorate %outColor Location 0
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func
%label = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr

%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p0z = OpCompositeExtract %float %p0 2
%p0w = OpCompositeExtract %float %p0 3
%p1x = OpCompositeExtract %float %p1 0
%p1y = OpCompositeExtract %float %p1 1
%p1z = OpCompositeExtract %float %p1 2
%p1w = OpCompositeExtract %float %p1 3

%dx = OpFSub %float %p1x %p0x
%dy = OpFSub %float %p1y %p0y
%len2 = OpExtInstImport "GLSL.std.450"
%len = OpExtInst %float %len2 Sqrt %dx
%len_y = OpExtInst %float %len2 Sqrt %dy
; Use 2D perpendicular from x/y components.
%inv_len = OpFDiv %float %float_1 %len
%nx = OpFMul %float %dx %inv_len
%ny = OpFMul %float %dy %inv_len
%perp_x = OpFNegate %float %ny
%perp_y = OpCopyObject %float %nx

%off0x = OpFMul %float %perp_x %float_04
%off0y = OpFMul %float %perp_y %float_04
%off1x = OpFMul %float %perp_x %float_005
%off1y = OpFMul %float %perp_y %float_005

%p0plusx = OpFAdd %float %p0x %off0x
%p0plusy = OpFAdd %float %p0y %off0y
%p0minusx = OpFSub %float %p0x %off0x
%p0minusy = OpFSub %float %p0y %off0y
%p1plusx = OpFAdd %float %p1x %off1x
%p1plusy = OpFAdd %float %p1y %off1y
%p1minusx = OpFSub %float %p1x %off1x
%p1minusy = OpFSub %float %p1y %off1y

%v0 = OpCompositeConstruct %v4float %p0plusx %p0plusy %p0z %p0w
%v1 = OpCompositeConstruct %v4float %p0minusx %p0minusy %p0z %p0w
%v2 = OpCompositeConstruct %v4float %p1plusx %p1plusy %p1z %p1w
%v3 = OpCompositeConstruct %v4float %p1minusx %p1minusy %p1z %p1w

%out0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out0 %v0
OpStore %outColor %c0
OpEmitVertex

%out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out1 %v1
OpStore %outColor %c0
OpEmitVertex

%out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out2 %v2
OpStore %outColor %c1
OpEmitVertex

%out3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out3 %v3
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpFunctionEnd
