#include <algorithm>
#include <chrono>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <random>
#include <vector>

using namespace std;

struct Solver {
    int n, m, k;
    vector<pair<int, int>> edges;
    vector<vector<int>> adj;
    vector<int> deg;
    vector<int> color;

    bool greedy_order(const vector<int>& order) {
        fill(color.begin(), color.end(), -1);
        vector<int> seen(k, -1);
        int token = 0;
        for (int v : order) {
            ++token;
            for (int to : adj[v]) {
                int c = color[to];
                if (c >= 0) seen[c] = token;
            }
            int chosen = -1;
            for (int c = 0; c < k; ++c) {
                if (seen[c] != token) {
                    chosen = c;
                    break;
                }
            }
            if (chosen < 0) return false;
            color[v] = chosen;
        }
        return true;
    }

    bool verify() const {
        for (auto [u, v] : edges) {
            if (u == v) return false;
            if (color[u] < 0 || color[u] >= k) return false;
            if (color[v] < 0 || color[v] >= k) return false;
            if (color[u] == color[v]) return false;
        }
        return true;
    }

    bool exact_small(int limit_n = 70) {
        if (n > limit_n || k > 12) return false;

        vector<int> order(n);
        iota(order.begin(), order.end(), 0);
        sort(order.begin(), order.end(), [&](int a, int b) {
            if (deg[a] != deg[b]) return deg[a] > deg[b];
            return a < b;
        });

        vector<int> pos(n);
        for (int i = 0; i < n; ++i) pos[order[i]] = i;

        vector<vector<int>> ladj(n);
        for (int v = 0; v < n; ++v) {
            for (int to : adj[v]) {
                if (pos[to] < pos[v]) ladj[v].push_back(to);
            }
        }

        vector<int> assign(n, -1);
        vector<int> sat(n, 0);
        vector<int> forbidden(k, 0);

        auto pick_vertex = [&]() {
            int best = -1;
            int bestSat = -1;
            int bestDeg = -1;
            for (int v : order) {
                if (assign[v] != -1) continue;
                if (sat[v] > bestSat || (sat[v] == bestSat && deg[v] > bestDeg)) {
                    bestSat = sat[v];
                    bestDeg = deg[v];
                    best = v;
                }
            }
            return best;
        };

        function<bool()> dfs = [&]() -> bool {
            int v = pick_vertex();
            if (v == -1) return true;

            fill(forbidden.begin(), forbidden.end(), 0);
            for (int to : adj[v]) {
                if (assign[to] != -1) forbidden[assign[to]] = 1;
            }

            vector<int> choices;
            for (int c = 0; c < k; ++c) if (!forbidden[c]) choices.push_back(c);
            sort(choices.begin(), choices.end(), [&](int a, int b) {
                int scoreA = 0, scoreB = 0;
                for (int to : adj[v]) {
                    if (assign[to] == -1) {
                        bool freeA = true, freeB = true;
                        for (int t2 : adj[to]) {
                            if (assign[t2] == a) freeA = false;
                            if (assign[t2] == b) freeB = false;
                        }
                        scoreA += freeA;
                        scoreB += freeB;
                    }
                }
                return scoreA > scoreB;
            });

            for (int c : choices) {
                assign[v] = c;
                vector<int> changed;
                for (int to : adj[v]) {
                    if (assign[to] == -1) {
                        ++sat[to];
                        changed.push_back(to);
                    }
                }
                if (dfs()) {
                    color = assign;
                    return true;
                }
                for (int to : changed) --sat[to];
                assign[v] = -1;
            }
            return false;
        };

        if (dfs()) {
            color = assign;
            return true;
        }
        return false;
    }

    void solve() {
        vector<int> byDeg(n), byDegAsc(n), byId(n);
        iota(byDeg.begin(), byDeg.end(), 0);
        byDegAsc = byDeg;
        byId = byDeg;

        sort(byDeg.begin(), byDeg.end(), [&](int a, int b) {
            if (deg[a] != deg[b]) return deg[a] > deg[b];
            return a < b;
        });
        sort(byDegAsc.begin(), byDegAsc.end(), [&](int a, int b) {
            if (deg[a] != deg[b]) return deg[a] < deg[b];
            return a < b;
        });

        if (exact_small()) return;

        if (k == 1) {
            color.assign(n, 0);
            return;
        }

        if (greedy_order(byDeg)) return;
        if (greedy_order(byId)) return;
        if (greedy_order(byDegAsc)) return;

        vector<int> perm(n);
        iota(perm.begin(), perm.end(), 0);
        mt19937_64 rng(0x123456789abcdef0ull ^ (uint64_t)n ^ ((uint64_t)m << 1) ^ ((uint64_t)k << 32));
        for (int attempt = 0; attempt < 32; ++attempt) {
            shuffle(perm.begin(), perm.end(), rng);
            if (greedy_order(perm)) return;
        }

        color.assign(n, 0);
        for (int v = 0; v < n; ++v) {
            vector<char> bad(k, 0);
            for (int to : adj[v]) {
                if (color[to] >= 0 && color[to] < k) bad[color[to]] = 1;
            }
            int chosen = 0;
            while (chosen < k && bad[chosen]) ++chosen;
            if (chosen == k) chosen = v % k;
            color[v] = chosen;
        }
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Solver s;
    if (!(cin >> s.n >> s.m >> s.k)) return 0;

    s.edges.reserve(s.m);
    s.adj.assign(s.n, {});
    s.deg.assign(s.n, 0);
    s.color.assign(s.n, -1);

    for (int i = 0; i < s.m; ++i) {
        int u, v;
        cin >> u >> v;
        s.edges.push_back({u, v});
        if (u == v) continue;
        s.adj[u].push_back(v);
        s.adj[v].push_back(u);
        ++s.deg[u];
        ++s.deg[v];
    }

    if (s.n == 0) return 0;
    if (s.k <= 0) s.k = 1;

    s.solve();

    if (!s.verify()) {
        s.color.assign(s.n, 0);
        for (int v = 0; v < s.n; ++v) {
            vector<char> bad(s.k, 0);
            for (int to : s.adj[v]) {
                if (s.color[to] >= 0 && s.color[to] < s.k) bad[s.color[to]] = 1;
            }
            int chosen = 0;
            while (chosen < s.k && bad[chosen]) ++chosen;
            if (chosen == s.k) chosen = 0;
            s.color[v] = chosen;
        }
    }

    for (int i = 0; i < s.n; ++i) {
        if (i) cout << ' ';
        cout << (s.color[i] < 0 ? 0 : s.color[i]);
    }
    cout << '\n';
    return 0;
}
