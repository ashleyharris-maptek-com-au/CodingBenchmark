use std::cmp::Ordering;
use std::io::{self, Read};

#[derive(Clone)]
struct Sample {
    depth: f64,
    vals: Vec<Option<f64>>,
}

#[derive(Clone)]
struct Hole {
    id: String,
    samples: Vec<Sample>,
}

#[derive(Clone)]
struct Suspect {
    hole_id: String,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

fn parse_f64(tok: Option<&str>) -> Option<f64> {
    tok.and_then(|t| t.parse::<f64>().ok()).filter(|v| v.is_finite())
}

fn median(mut v: Vec<f64>) -> f64 {
    if v.is_empty() {
        return 0.0;
    }
    v.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let n = v.len();
    if n % 2 == 1 { v[n / 2] } else { (v[n / 2 - 1] + v[n / 2]) * 0.5 }
}

fn robust_scale(values: &[f64]) -> (f64, f64) {
    if values.is_empty() {
        return (0.0, 1.0);
    }
    let med = median(values.to_vec());
    let mut devs = Vec::with_capacity(values.len());
    for &x in values {
        devs.push((x - med).abs());
    }
    let mut mad = median(devs);
    if mad <= 1e-12 {
        let mut sorted = values.to_vec();
        sorted.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
        let q1 = sorted[sorted.len() / 4];
        let q3 = sorted[sorted.len() * 3 / 4];
        mad = (q3 - q1).abs() * 0.7413;
    }
    if mad <= 1e-12 { mad = 1.0; }
    (med, mad)
}

fn normalized_depth(_start: [f64; 3], dir: [f64; 3], depth: f64) -> f64 {
    let norm = (dir[0] * dir[0] + dir[1] * dir[1] + dir[2] * dir[2]).sqrt();
    if norm <= 1e-12 { depth } else { depth / norm.max(1e-12) }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let num_holes = it.next().and_then(|s| s.parse::<usize>().ok()).unwrap_or(0);
    let num_props = it.next().and_then(|s| s.parse::<usize>().ok()).unwrap_or(0);

    let mut prop_names = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        prop_names.push(it.next().unwrap_or("").to_string());
    }

    let mut holes = Vec::with_capacity(num_holes);
    let mut global_values: Vec<Vec<f64>> = vec![Vec::new(); num_props];

    for _ in 0..num_holes {
        let hole_id = match it.next() { Some(s) => s.to_string(), None => break };
        let sx = parse_f64(it.next()).unwrap_or(0.0);
        let sy = parse_f64(it.next()).unwrap_or(0.0);
        let sz = parse_f64(it.next()).unwrap_or(0.0);
        let dx = parse_f64(it.next()).unwrap_or(0.0);
        let dy = parse_f64(it.next()).unwrap_or(0.0);
        let dz = parse_f64(it.next()).unwrap_or(0.0);
        let len = parse_f64(it.next()).unwrap_or(0.0);
        let num_samples = it.next().and_then(|s| s.parse::<usize>().ok()).unwrap_or(0);
        let mut samples = Vec::with_capacity(num_samples);

        for _ in 0..num_samples {
            let depth = parse_f64(it.next()).unwrap_or(0.0);
            let mut vals = Vec::with_capacity(num_props);
            for p in 0..num_props {
                let val = parse_f64(it.next());
                if let Some(v) = val {
                    global_values[p].push(v);
                }
                vals.push(val);
            }
            samples.push(Sample { depth: normalized_depth([sx, sy, sz], [dx, dy, dz], depth.min(len).max(0.0)), vals });
        }
        samples.sort_by(|a, b| a.depth.partial_cmp(&b.depth).unwrap_or(Ordering::Equal));
        holes.push(Hole { id: hole_id, samples });
    }

    let total_samples: usize = holes.iter().map(|h| h.samples.len()).sum();
    let small_dataset = total_samples <= 4000 || num_holes <= 25;

    let mut global_stats = Vec::with_capacity(num_props);
    for p in 0..num_props {
        global_stats.push(robust_scale(&global_values[p]));
    }

    let mut suspects: Vec<Suspect> = Vec::new();

    for hole in &holes {
        let n = hole.samples.len();
        if n == 0 { continue; }
        for i in 0..n {
            for p in 0..num_props {
                let v = match hole.samples[i].vals[p] { Some(x) => x, None => continue };
                let mut neigh = Vec::new();
                if i > 0 {
                    if let Some(x) = hole.samples[i - 1].vals[p] { neigh.push(x); }
                }
                if i + 1 < n {
                    if let Some(x) = hole.samples[i + 1].vals[p] { neigh.push(x); }
                }
                if !small_dataset && i > 1 {
                    if let Some(x) = hole.samples[i - 2].vals[p] { neigh.push(x); }
                }
                if !small_dataset && i + 2 < n {
                    if let Some(x) = hole.samples[i + 2].vals[p] { neigh.push(x); }
                }

                let local_conf = if neigh.len() >= 2 {
                    let (m, s) = robust_scale(&neigh);
                    ((v - m).abs() / s).max(0.0)
                } else {
                    let (m, s) = global_stats[p];
                    ((v - m).abs() / s).max(0.0)
                };

                let mut confidence = local_conf;
                if neigh.len() >= 2 {
                    let (gm, gs) = global_stats[p];
                    let global_conf = ((v - gm).abs() / gs).max(0.0);
                    confidence = confidence.max(global_conf * 0.7);
                }

                if confidence >= if small_dataset { 3.0 } else { 4.0 } {
                    suspects.push(Suspect {
                        hole_id: hole.id.clone(),
                        sample_idx: i,
                        property_name: prop_names.get(p).cloned().unwrap_or_else(|| format!("prop{}", p + 1)),
                        confidence,
                    });
                }
            }
        }
    }

    suspects.sort_by(|a, b| {
        b.confidence.partial_cmp(&a.confidence).unwrap_or(Ordering::Equal)
            .then_with(|| a.hole_id.cmp(&b.hole_id))
            .then_with(|| a.sample_idx.cmp(&b.sample_idx))
            .then_with(|| a.property_name.cmp(&b.property_name))
    });

    if suspects.len() > 50000 {
        suspects.truncate(50000);
    }

    println!("{}", suspects.len());
    for s in suspects {
        println!("{} {} {} {:.6}", s.hole_id, s.sample_idx, s.property_name, s.confidence);
    }
}
