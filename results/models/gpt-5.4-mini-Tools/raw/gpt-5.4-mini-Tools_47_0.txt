@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write>  result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>              params   : vec4<u32>;

fn rotl(x : u32, s : u32) -> u32 {
    return (x << s) | (x >> (32u - s));
}

fn quarter_round(a0 : u32, b0 : u32, c0 : u32, d0 : u32) -> vec4<u32> {
    var a : u32 = a0;
    var b : u32 = b0;
    var c : u32 = c0;
    var d : u32 = d0;

    a = a + b;
    d = d ^ a;
    d = rotl(d, 16u);

    c = c + d;
    b = b ^ c;
    b = rotl(b, 12u);

    a = a + b;
    d = d ^ a;
    d = rotl(d, 8u);

    c = c + d;
    b = b ^ c;
    b = rotl(b, 7u);

    return vec4<u32>(a, b, c, d);
}

fn custom_hash(data : array<u32, 16>, nonce : u32) -> vec2<u32> {
    var state : array<u32, 16>;

    for (var i : u32 = 0u; i < 16u; i = i + 1u) {
        state[i] = data[i] ^ (nonce + i);
    }

    for (var round : u32 = 0u; round < 8u; round = round + 1u) {
        let r0 = quarter_round(state[0u], state[1u], state[2u], state[3u]);
        state[0u] = r0.x;
        state[1u] = r0.y;
        state[2u] = r0.z;
        state[3u] = r0.w;

        let r1 = quarter_round(state[4u], state[5u], state[6u], state[7u]);
        state[4u] = r1.x;
        state[5u] = r1.y;
        state[6u] = r1.z;
        state[7u] = r1.w;

        let r2 = quarter_round(state[8u], state[9u], state[10u], state[11u]);
        state[8u] = r2.x;
        state[9u] = r2.y;
        state[10u] = r2.z;
        state[11u] = r2.w;

        let r3 = quarter_round(state[12u], state[13u], state[14u], state[15u]);
        state[12u] = r3.x;
        state[13u] = r3.y;
        state[14u] = r3.z;
        state[15u] = r3.w;
    }

    var h0 : u32 = state[0u] ^ state[2u] ^ state[4u] ^ state[6u] ^ state[8u] ^ state[10u] ^ state[12u] ^ state[14u];
    var h1 : u32 = state[1u] ^ state[3u] ^ state[5u] ^ state[7u] ^ state[9u] ^ state[11u] ^ state[13u] ^ state[15u];
    return vec2<u32>(h0, h1);
}

fn trailing_zeros64(lo : u32, hi : u32) -> u32 {
    if (lo != 0u) {
        return countTrailingZeros(lo);
    }
    return 32u + countTrailingZeros(hi);
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    if (gid.x >= params.z) {
        return;
    }

    let nonce : u32 = params.y + gid.x;
    let hash : vec2<u32> = custom_hash(baseData, nonce);
    let tz : u32 = trailing_zeros64(hash.x, hash.y);

    if (tz >= params.x) {
        result[0] = 1u;
        result[1] = nonce;
        result[2] = hash.x;
        result[3] = hash.y;
    }
}
