use std::collections::VecDeque;
use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Edge {
    to: usize,
    next: i32,
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut head = vec![-1i32; n];
    let mut edges: Vec<Edge> = Vec::with_capacity(m * 2);
    let mut degrees = vec![0usize; n];
    let mut self_loop = vec![0usize; n];

    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            if u < n {
                self_loop[u] += 1;
            }
            continue;
        }
        if u >= n || v >= n {
            continue;
        }
        add_edge(u, v, &mut head, &mut edges);
        add_edge(v, u, &mut head, &mut edges);
        degrees[u] += 1;
        degrees[v] += 1;
    }

    let mut comp_id = vec![usize::MAX; n];
    let mut comps: Vec<Vec<usize>> = Vec::new();
    for s in 0..n {
        if comp_id[s] != usize::MAX {
            continue;
        }
        let id = comps.len();
        let mut q = VecDeque::new();
        q.push_back(s);
        comp_id[s] = id;
        let mut comp = Vec::new();
        while let Some(u) = q.pop_front() {
            comp.push(u);
            let mut e = head[u];
            while e != -1 {
                let v = edges[e as usize].to;
                if comp_id[v] == usize::MAX {
                    comp_id[v] = id;
                    q.push_back(v);
                }
                e = edges[e as usize].next;
            }
        }
        comps.push(comp);
    }

    let c = comps.len().max(1);
    let cols = (c as f64).sqrt().ceil() as usize;
    let rows = (c + cols - 1) / cols;
    let cell_w = 200.0;
    let cell_h = 200.0;
    let mut pos = vec![(0.0f64, 0.0f64); n];

    for (idx, comp) in comps.iter().enumerate() {
        let r = idx / cols;
        let col = idx % cols;
        let base_x = col as f64 * cell_w;
        let base_y = r as f64 * cell_h;
        let k = comp.len();
        if k == 1 {
            let u = comp[0];
            pos[u] = (base_x, base_y);
            continue;
        }

        let mut is_tree = true;
        let mut edge_count = 0usize;
        for &u in comp {
            edge_count += degrees[u];
        }
        edge_count /= 2;
        if edge_count >= k {
            is_tree = false;
        }

        if is_tree && k <= 2000 {
            let root = choose_root(comp, &degrees, &head, &edges);
            layout_tree(root, &mut pos, base_x, base_y, &head, &edges, &comp_id);
        } else if k <= 80 {
            layout_circular(comp, &mut pos, base_x, base_y, &degrees);
        } else {
            layout_large(comp, &mut pos, base_x, base_y, &degrees);
        }

        let scale = if k <= 12 {
            30.0
        } else if k <= 80 {
            45.0
        } else if k <= 1000 {
            60.0
        } else {
            80.0
        };
        normalize_component(comp, &mut pos, base_x, base_y, scale);
    }

    let mut out = String::new();
    for i in 0..n {
        let (x, y) = pos[i];
        out.push_str(&format!("{:.6} {:.6}\n", x, y));
    }
    print!("{out}");

    let _ = self_loop;
    let _ = rows;
}

fn add_edge(u: usize, v: usize, head: &mut [i32], edges: &mut Vec<Edge>) {
    let next = head[u];
    edges.push(Edge { to: v, next });
    head[u] = (edges.len() - 1) as i32;
}

fn choose_root(
    comp: &[usize],
    degrees: &[usize],
    head: &[i32],
    edges: &[Edge],
) -> usize {
    let mut best = comp[0];
    let mut best_score = usize::MAX;
    for &u in comp {
        let mut sum = 0usize;
        let mut cnt = 0usize;
        let mut e = head[u];
        while e != -1 {
            let v = edges[e as usize].to;
            sum += degrees[v];
            cnt += 1;
            e = edges[e as usize].next;
        }
        let score = sum + cnt * degrees[u];
        if score < best_score {
            best_score = score;
            best = u;
        }
    }
    best
}

fn layout_tree(
    root: usize,
    pos: &mut [(f64, f64)],
    base_x: f64,
    base_y: f64,
    head: &[i32],
    edges: &[Edge],
    comp_id: &[usize],
) {
    let comp = comp_id[root];
    let mut parent = vec![usize::MAX; pos.len()];
    let mut order = Vec::new();
    let mut stack = vec![root];
    parent[root] = root;
    while let Some(u) = stack.pop() {
        order.push(u);
        let mut e = head[u];
        while e != -1 {
            let v = edges[e as usize].to;
            if comp_id[v] == comp && parent[v] == usize::MAX {
                parent[v] = u;
                stack.push(v);
            }
            e = edges[e as usize].next;
        }
    }

    let mut children: Vec<Vec<usize>> = vec![Vec::new(); pos.len()];
    for &u in &order {
        if u != root && parent[u] != usize::MAX {
            children[parent[u]].push(u);
        }
    }

    let mut widths = vec![0.0f64; pos.len()];
    fn dfs(u: usize, children: &Vec<Vec<usize>>, widths: &mut Vec<f64>) -> f64 {
        if children[u].is_empty() {
            widths[u] = 1.0;
            return 1.0;
        }
        let mut sum = 0.0;
        for &v in &children[u] {
            sum += dfs(v, children, widths) + 0.4;
        }
        if sum < 1.0 {
            sum = 1.0;
        }
        widths[u] = sum;
        sum
    }
    dfs(root, &children, &mut widths);

    fn place(
        u: usize,
        children: &Vec<Vec<usize>>,
        widths: &Vec<f64>,
        pos: &mut [(f64, f64)],
        x0: f64,
        y0: f64,
        depth: usize,
    ) {
        let y = y0 + depth as f64 * 18.0;
        if children[u].is_empty() {
            pos[u] = (x0 + widths[u] * 10.0 * 0.5, y);
            return;
        }
        let mut cur = x0;
        let mut centers = Vec::with_capacity(children[u].len());
        for &v in &children[u] {
            let w = widths[v] * 10.0;
            place(v, children, widths, pos, cur, y0, depth + 1);
            centers.push(cur + w * 0.5);
            cur += w + 4.0;
        }
        let center = if centers.is_empty() {
            x0
        } else {
            centers.iter().sum::<f64>() / centers.len() as f64
        };
        pos[u] = (center, y);
    }
    place(root, &children, &widths, pos, base_x, base_y, 0);
}

fn layout_circular(comp: &[usize], pos: &mut [(f64, f64)], base_x: f64, base_y: f64, degrees: &[usize]) {
    let mut nodes = comp.to_vec();
    nodes.sort_by_key(|&u| std::cmp::Reverse(degrees[u]));
    let n = nodes.len();
    let r = 12.0 + (n as f64).sqrt() * 6.0;
    for (i, &u) in nodes.iter().enumerate() {
        let t = 2.0 * PI * (i as f64) / (n as f64);
        pos[u] = (base_x + r * t.cos(), base_y + r * t.sin());
    }
}

fn layout_large(comp: &[usize], pos: &mut [(f64, f64)], base_x: f64, base_y: f64, degrees: &[usize]) {
    let mut nodes = comp.to_vec();
    nodes.sort_by_key(|&u| std::cmp::Reverse(degrees[u]));
    let n = nodes.len();
    let cols = ((n as f64).sqrt().ceil() as usize).max(1);
    let step = 10.0;
    for (i, &u) in nodes.iter().enumerate() {
        let r = i / cols;
        let c = i % cols;
        pos[u] = (
            base_x + c as f64 * step + (r as f64 * 0.35).sin() * 2.0,
            base_y + r as f64 * step + (c as f64 * 0.35).cos() * 2.0,
        );
    }
}

fn normalize_component(comp: &[usize], pos: &mut [(f64, f64)], base_x: f64, base_y: f64, scale: f64) {
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for &u in comp {
        let (x, y) = pos[u];
        if x < min_x { min_x = x; }
        if x > max_x { max_x = x; }
        if y < min_y { min_y = y; }
        if y > max_y { max_y = y; }
    }
    let cx = (min_x + max_x) * 0.5;
    let cy = (min_y + max_y) * 0.5;
    let span = (max_x - min_x).max(max_y - min_y).max(1.0);
    let s = scale / span;
    for &u in comp {
        let (x, y) = pos[u];
        pos[u] = (base_x + (x - cx) * s, base_y + (y - cy) * s);
    }
}
