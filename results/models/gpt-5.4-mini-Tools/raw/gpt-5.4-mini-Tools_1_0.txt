using namespace std;

struct Point {
    double x = 0.0;
    double y = 0.0;
};

static inline double dist2(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static vector<int> exact_small_tsp(const vector<Point>& pts) {
    int n = (int)pts.size();
    vector<vector<double>> d(n, vector<double>(n));
    for (int i = 0; i < n; ++i) for (int j = 0; j < n; ++j) d[i][j] = sqrt(dist2(pts[i], pts[j]));

    int m = n - 1;
    int full = 1 << m;
    const double INF = numeric_limits<double>::infinity();
    vector<vector<double>> dp(full, vector<double>(m, INF));
    vector<vector<int>> parent(full, vector<int>(m, -1));

    for (int i = 0; i < m; ++i) dp[1 << i][i] = d[0][i + 1];

    for (int mask = 1; mask < full; ++mask) {
        for (int last = 0; last < m; ++last) {
            double cur = dp[mask][last];
            if (!isfinite(cur)) continue;
            int rem = (full - 1) ^ mask;
            while (rem) {
                int bit = rem & -rem;
                int nxt = __builtin_ctz((unsigned)bit);
                int nmask = mask | bit;
                double cand = cur + d[last + 1][nxt + 1];
                if (cand < dp[nmask][nxt]) {
                    dp[nmask][nxt] = cand;
                    parent[nmask][nxt] = last;
                }
                rem ^= bit;
            }
        }
    }

    int best = 0;
    double best_cost = INF;
    int all = full - 1;
    for (int last = 0; last < m; ++last) {
        double cand = dp[all][last] + d[last + 1][0];
        if (cand < best_cost) {
            best_cost = cand;
            best = last;
        }
    }

    vector<int> order;
    order.reserve(n);
    order.push_back(0);
    int mask = all, last = best;
    vector<int> rev;
    while (last != -1) {
        rev.push_back(last + 1);
        int pl = parent[mask][last];
        mask ^= 1 << last;
        last = pl;
    }
    reverse(rev.begin(), rev.end());
    for (int v : rev) order.push_back(v);
    return order;
}

static vector<int> nearest_neighbor(const vector<Point>& pts) {
    int n = (int)pts.size();
    vector<int> order;
    order.reserve(n);
    vector<char> used(n, 0);
    order.push_back(0);
    used[0] = 1;
    int cur = 0;
    for (int step = 1; step < n; ++step) {
        int best = -1;
        double bestd = numeric_limits<double>::infinity();
        for (int i = 0; i < n; ++i) if (!used[i]) {
            double d = dist2(pts[cur], pts[i]);
            if (d < bestd) {
                bestd = d;
                best = i;
            }
        }
        used[best] = 1;
        order.push_back(best);
        cur = best;
    }
    return order;
}

static uint32_t part1by1(uint32_t x) {
    x &= 0x0000ffffu;
    x = (x | (x << 8)) & 0x00FF00FFu;
    x = (x | (x << 4)) & 0x0F0F0F0Fu;
    x = (x | (x << 2)) & 0x33333333u;
    x = (x | (x << 1)) & 0x55555555u;
    return x;
}

static uint64_t morton_code(double x, double y, double minx, double miny, double scale) {
    uint32_t ix = (uint32_t)max(0.0, min(65535.0, (x - minx) * scale));
    uint32_t iy = (uint32_t)max(0.0, min(65535.0, (y - miny) * scale));
    return (uint64_t(part1by1(iy)) << 1) | uint64_t(part1by1(ix));
}

static vector<int> spatial_order(const vector<Point>& pts) {
    int n = (int)pts.size();
    double minx = pts[0].x, maxx = pts[0].x, miny = pts[0].y, maxy = pts[0].y;
    for (const auto& p : pts) {
        minx = min(minx, p.x); maxx = max(maxx, p.x);
        miny = min(miny, p.y); maxy = max(maxy, p.y);
    }
    double dx = maxx - minx, dy = maxy - miny;
    double scale = 0.0;
    if (dx > 0.0 || dy > 0.0) scale = 65535.0 / max(dx, dy);
    vector<pair<uint64_t, int>> keys;
    keys.reserve(n);
    for (int i = 0; i < n; ++i) keys.push_back({morton_code(pts[i].x, pts[i].y, minx, miny, scale), i});
    sort(keys.begin(), keys.end(), [](const auto& a, const auto& b) {
        if (a.first != b.first) return a.first < b.first;
        return a.second < b.second;
    });
    vector<int> order;
    order.reserve(n);
    for (auto& kv : keys) order.push_back(kv.second);
    auto pos0 = find(order.begin(), order.end(), 0);
    if (pos0 != order.begin()) {
        rotate(order.begin(), pos0, pos0 + 1);
    }
    return order;
}

static void two_opt_pass(vector<int>& order, const vector<Point>& pts, int max_swaps) {
    int n = (int)order.size();
    if (n < 4) return;
    int swaps = 0;
    bool improved = true;
    while (improved && swaps < max_swaps) {
        improved = false;
        for (int i = 1; i + 2 < n && swaps < max_swaps; ++i) {
            for (int k = i + 1; k + 1 < n && swaps < max_swaps; ++k) {
                int a = order[i - 1], b = order[i];
                int c = order[k], d = order[k + 1];
                double before = sqrt(dist2(pts[a], pts[b])) + sqrt(dist2(pts[c], pts[d]));
                double after = sqrt(dist2(pts[a], pts[c])) + sqrt(dist2(pts[b], pts[d]));
                if (after + 1e-12 < before) {
                    reverse(order.begin() + i, order.begin() + k + 1);
                    improved = true;
                    ++swaps;
                }
            }
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    vector<Point> pts(N);
    for (int i = 0; i < N; ++i) cin >> pts[i].x >> pts[i].y;

    vector<int> order;
    if (N <= 12) {
        order = exact_small_tsp(pts);
    } else if (N <= 1500) {
        order = nearest_neighbor(pts);
        two_opt_pass(order, pts, 2000);
    } else {
        order = spatial_order(pts);
        if (N <= 50000) two_opt_pass(order, pts, 1000);
    }

    if (!order.empty() && order[0] != 0) {
        auto it = find(order.begin(), order.end(), 0);
        if (it != order.end()) rotate(order.begin(), it, it + 1);
    }

    for (int i = 0; i < N; ++i) {
        if (i) cout << ' ';
        cout << order[i];
    }
    cout << '\n';
    return 0;
}
