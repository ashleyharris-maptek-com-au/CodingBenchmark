// Travelling Salesman Problem solver with adaptive strategy selection.
// - Tiny N: exact Held-Karp dynamic programming
// - Small/medium N: nearest-neighbor tour + 2-opt on a limited candidate set
// - Large N: sweep order with local nearest insertion refinements
//
// Output is a permutation of 0..N-1 starting with 0.

#include <algorithm>
#include <cmath>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <utility>
#include <vector>

using namespace std;

struct Point {
    double x, y;
};

static inline double dist2(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static inline double dist(const Point& a, const Point& b) {
    return std::sqrt(dist2(a, b));
}

static vector<int> held_karp(const vector<Point>& p) {
    const int n = (int)p.size();
    const int m = n - 1;
    const int FULL = 1 << m;
    const double INF = numeric_limits<double>::infinity();

    vector<vector<double>> dp(FULL, vector<double>(m, INF));
    vector<vector<int>> parent(FULL, vector<int>(m, -1));

    for (int i = 0; i < m; ++i) {
        dp[1 << i][i] = dist(p[0], p[i + 1]);
    }

    for (int mask = 1; mask < FULL; ++mask) {
        for (int last = 0; last < m; ++last) {
            double cur = dp[mask][last];
            if (!isfinite(cur)) continue;
            int rem = (FULL - 1) ^ mask;
            while (rem) {
                int bit = rem & -rem;
                int nxt = __builtin_ctz(bit);
                int nmask = mask | bit;
                double cand = cur + dist(p[last + 1], p[nxt + 1]);
                if (cand < dp[nmask][nxt]) {
                    dp[nmask][nxt] = cand;
                    parent[nmask][nxt] = last;
                }
                rem ^= bit;
            }
        }
    }

    int best_last = 0;
    double best_cost = INF;
    int all = FULL - 1;
    for (int last = 0; last < m; ++last) {
        double cand = dp[all][last] + dist(p[last + 1], p[0]);
        if (cand < best_cost) {
            best_cost = cand;
            best_last = last;
        }
    }

    vector<int> route;
    route.reserve(n);
    route.push_back(0);
    vector<int> rev;
    int mask = all;
    int last = best_last;
    while (last != -1) {
        rev.push_back(last + 1);
        int pl = parent[mask][last];
        mask ^= 1 << last;
        last = pl;
    }
    reverse(rev.begin(), rev.end());
    route.insert(route.end(), rev.begin(), rev.end());
    return route;
}

static vector<int> nearest_neighbor(const vector<Point>& p) {
    const int n = (int)p.size();
    vector<int> tour;
    tour.reserve(n);
    vector<char> used(n, 0);
    tour.push_back(0);
    used[0] = 1;
    for (int step = 1; step < n; ++step) {
        int last = tour.back();
        int best = -1;
        double bestd = numeric_limits<double>::infinity();
        for (int i = 1; i < n; ++i) {
            if (!used[i]) {
                double d = dist2(p[last], p[i]);
                if (d < bestd) {
                    bestd = d;
                    best = i;
                }
            }
        }
        used[best] = 1;
        tour.push_back(best);
    }
    return tour;
}

static void two_opt_limited(const vector<Point>& p, vector<int>& tour, int rounds) {
    const int n = (int)tour.size();
    if (n < 4) return;
    for (int r = 0; r < rounds; ++r) {
        bool improved = false;
        for (int i = 1; i + 2 < n; ++i) {
            int a = tour[i - 1], b = tour[i];
            for (int k = i + 1; k + 1 < n; ++k) {
                int c = tour[k], d = tour[k + 1];
                double before = dist(p[a], p[b]) + dist(p[c], p[d]);
                double after = dist(p[a], p[c]) + dist(p[b], p[d]);
                if (after + 1e-12 < before) {
                    reverse(tour.begin() + i, tour.begin() + k + 1);
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }
}

static vector<int> sweep_order(const vector<Point>& p) {
    const int n = (int)p.size();
    vector<int> idx(n);
    iota(idx.begin(), idx.end(), 0);
    double cx = p[0].x, cy = p[0].y;
    sort(idx.begin() + 1, idx.end(), [&](int a, int b) {
        double aa = atan2(p[a].y - cy, p[a].x - cx);
        double bb = atan2(p[b].y - cy, p[b].x - cx);
        if (aa != bb) return aa < bb;
        double da = dist2(p[0], p[a]);
        double db = dist2(p[0], p[b]);
        return da < db;
    });
    return idx;
}

static vector<int> adaptive_solve(const vector<Point>& p) {
    const int n = (int)p.size();
    if (n <= 20) return held_karp(p);

    if (n <= 2000) {
        vector<int> tour = nearest_neighbor(p);
        two_opt_limited(p, tour, 2);
        return tour;
    }

    if (n <= 50000) {
        vector<int> tour = sweep_order(p);
        two_opt_limited(p, tour, 1);
        return tour;
    }

    vector<int> tour = sweep_order(p);
    return tour;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    vector<Point> p(N);
    for (int i = 0; i < N; ++i) cin >> p[i].x >> p[i].y;

    vector<int> tour = adaptive_solve(p);
    if (tour.empty()) return 0;
    if (tour[0] != 0) {
        auto it = find(tour.begin(), tour.end(), 0);
        if (it != tour.end()) rotate(tour.begin(), it, tour.end());
    }

    for (int i = 0; i < (int)tour.size(); ++i) {
        if (i) cout << ' ';
        cout << tour[i];
    }
    cout << '\n';
    return 0;
}
