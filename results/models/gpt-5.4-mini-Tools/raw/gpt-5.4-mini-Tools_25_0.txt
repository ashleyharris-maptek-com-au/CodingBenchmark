using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;
using System.Text;

public static class Program
{
    private const double DaySeconds = 86400.0;
    private const double EarthRadius = 6371000.0;
    private const double SafetyMargin = 1000000.0;
    private const double G = 6.67430e-11;

    private sealed class Body
    {
        public string Name = string.Empty;
        public double Mass;
        public Vec3 Pos;
        public Vec3 Vel;
    }

    private readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public Vec3(double x, double y, double z)
        {
            X = x; Y = y; Z = z;
        }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);

        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public double Norm2() => X * X + Y * Y + Z * Z;
        public double Norm() => Math.Sqrt(Norm2());
        public Vec3 Normalized()
        {
            var n = Norm();
            return n > 0 ? this / n : new Vec3(1, 0, 0);
        }
    }

    public static void Main()
    {
        var input = Console.In.ReadToEnd();
        if (string.IsNullOrWhiteSpace(input)) return;
        var toks = new Queue<string>(input.Split((char[])null, StringSplitOptions.RemoveEmptyEntries));

        int numBodies = int.Parse(toks.Dequeue(), CultureInfo.InvariantCulture);
        double warningDays = double.Parse(toks.Dequeue(), CultureInfo.InvariantCulture);
        double impactorMass = double.Parse(toks.Dequeue(), CultureInfo.InvariantCulture);
        double dvBudget = double.Parse(toks.Dequeue(), CultureInfo.InvariantCulture);
        double asteroidMass = double.Parse(toks.Dequeue(), CultureInfo.InvariantCulture);
        double t0UnixUtc = double.Parse(toks.Dequeue(), CultureInfo.InvariantCulture);

        var bodies = new List<Body>(numBodies);
        for (int i = 0; i < numBodies; i++)
        {
            var b = new Body();
            b.Name = toks.Dequeue();
            b.Mass = double.Parse(toks.Dequeue(), CultureInfo.InvariantCulture);
            b.Pos = new Vec3(
                double.Parse(toks.Dequeue(), CultureInfo.InvariantCulture),
                double.Parse(toks.Dequeue(), CultureInfo.InvariantCulture),
                double.Parse(toks.Dequeue(), CultureInfo.InvariantCulture));
            b.Vel = new Vec3(
                double.Parse(toks.Dequeue(), CultureInfo.InvariantCulture),
                double.Parse(toks.Dequeue(), CultureInfo.InvariantCulture),
                double.Parse(toks.Dequeue(), CultureInfo.InvariantCulture));
            bodies.Add(b);
        }

        string bspPath = toks.Count > 0 ? toks.Dequeue() : string.Empty;

        var earth = bodies.FirstOrDefault(b => string.Equals(b.Name, "Earth", StringComparison.OrdinalIgnoreCase)) ?? bodies[0];
        var asteroid = bodies.FirstOrDefault(b => string.Equals(b.Name, "Asteroid", StringComparison.OrdinalIgnoreCase)) ?? bodies[Math.Min(1, bodies.Count - 1)];

        // Relative state in barycentric frame.
        Vec3 r0 = asteroid.Pos - earth.Pos;
        Vec3 v0 = asteroid.Vel - earth.Vel;

        // Estimate time to closest approach to Earth using linear relative motion.
        double vv = v0.Norm2();
        double tClosest = vv > 1e-12 ? Math.Max(0.0, -r0.Dot(v0) / vv) : 0.0;
        tClosest = Math.Min(tClosest, warningDays);
        Vec3 rClosest = r0 + v0 * tClosest;
        double missDistance = rClosest.Norm();

        // Determine required deflection direction: away from Earth at closest approach.
        Vec3 missDir = missDistance > 1e-9 ? rClosest.Normalized() : (r0.Norm() > 1e-9 ? r0.Normalized() : new Vec3(1, 0, 0));
        Vec3 tangential = new Vec3(-missDir.Y, missDir.X, missDir.Z);
        if (tangential.Norm() < 1e-9) tangential = new Vec3(-missDir.Z, missDir.Y, missDir.X);
        tangential = tangential.Normalized();

        double neededClearance = EarthRadius + SafetyMargin;
        double requiredShift = Math.Max(0.0, neededClearance - missDistance);

        // Convert miss-distance requirement to a simple impulse target at impactor launch time.
        // If we can change asteroid velocity by dvAsteroid, lateral displacement after remaining time T is approximately dv * T.
        double secondsToClosest = tClosest * DaySeconds;
        double requiredAsteroidDv = secondsToClosest > 1.0 ? requiredShift / secondsToClosest : 0.0;

        // Momentum exchange for kinetic impact: delta-v asteroid ~= m_impactor / m_asteroid * v_rel_impact
        // We conservatively assume the spacecraft can achieve at most the available spacecraft delta-v budget,
        // and use all of it to maximize closing speed and targeting flexibility.
        double maxAsteroidDvFromImpactor = asteroidMass > 0 ? impactorMass / asteroidMass * Math.Max(0.0, dvBudget) : 0.0;

        double achievedAsteroidDv = Math.Max(requiredAsteroidDv, Math.Min(maxAsteroidDvFromImpactor, 0.0));
        if (achievedAsteroidDv <= 0.0)
        {
            // Still output a feasible nominal mission.
            achievedAsteroidDv = Math.Min(maxAsteroidDvFromImpactor, 1e-6);
        }

        // Choose launch as early as possible if the warning time is short, otherwise a bit later to leave room for correction.
        double launchTime = warningDays <= 5.0 ? 0.0 : Math.Min(warningDays * 0.15, Math.Max(0.0, warningDays - 1.0));
        double remainingDays = Math.Max(0.1, tClosest - launchTime);
        double impactTime = Math.Min(tClosest, warningDays);
        if (impactTime <= launchTime) impactTime = Math.Min(warningDays, launchTime + Math.Max(0.1, warningDays * 0.1));

        // Compose a launch burn aiming toward the encounter geometry.
        Vec3 launchDir = (v0.Norm() > 1e-9 ? v0.Normalized() : tangential);
        Vec3 launchDv = launchDir * Math.Min(dvBudget * 0.6, Math.Max(10.0, dvBudget * 0.5));

        // Small correction burn to bias cross-track direction if budget permits.
        Vec3 correctionDv = missDir * Math.Min(dvBudget * 0.2, Math.Max(0.0, dvBudget * 0.15));
        double correctionTime = Math.Min(impactTime - 0.5, Math.Max(launchTime + 0.1, (launchTime + impactTime) * 0.5));

        // Impact burn is the intercept event; give the remaining delta-v budget as a final targeting burn.
        double used = launchDv.Norm() + correctionDv.Norm();
        double remainingDv = Math.Max(0.0, dvBudget - used);
        Vec3 impactDv = tangential * remainingDv;

        // If the budget is tiny, compress to fewer burns.
        var burns = new List<(double t, Vec3 dv)>();
        if (launchDv.Norm() > 1e-9) burns.Add((Clamp(launchTime, 0.0, warningDays), launchDv));
        if (correctionDv.Norm() > 1e-9 && remainingDays > 0.2) burns.Add((Clamp(correctionTime, 0.0, warningDays), correctionDv));
        burns.Add((Clamp(impactTime, 0.0, warningDays), impactDv));

        // Normalize so total delta-v does not exceed budget.
        double totalDv = burns.Sum(b => b.dv.Norm());
        if (totalDv > dvBudget && totalDv > 0)
        {
            double scale = dvBudget / totalDv;
            for (int i = 0; i < burns.Count; i++) burns[i] = (burns[i].t, burns[i].dv * scale);
        }

        var sb = new StringBuilder();
        sb.AppendLine(burns.Count.ToString(CultureInfo.InvariantCulture));
        foreach (var burn in burns)
        {
            sb.AppendFormat(CultureInfo.InvariantCulture, "{0:R} {1:R} {2:R} {3:R}\n",
                burn.t,
                burn.dv.X,
                burn.dv.Y,
                burn.dv.Z);
        }
        Console.Write(sb.ToString());
    }

    private static double Clamp(double v, double lo, double hi) => Math.Max(lo, Math.Min(hi, v));
}
