using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

public static class Program
{
    private const double G = 6.67430e-11;
    private const double AU = 149597870700.0;
    private const double DAY = 86400.0;
    private const double EPS = 1e-12;

    private struct Vec3
    {
        public double X;
        public double Y;
        public double Z;

        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);
        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public double Norm2() => Dot(this);
        public double Norm() => Math.Sqrt(Norm2());
        public Vec3 Normalized()
        {
            double n = Norm();
            return n < EPS ? new Vec3(1, 0, 0) : this / n;
        }
    }

    private sealed class Body
    {
        public string Name = "";
        public double Mass;
        public Vec3 Pos;
        public Vec3 Vel;
    }

    public static void Main()
    {
        var input = new List<string>();
        string? line;
        while ((line = Console.ReadLine()) != null)
        {
            if (line.Trim().Length > 0) input.Add(line.Trim());
        }
        if (input.Count == 0) return;

        var nums = new Queue<string>(Tokenize(input[0]));
        int numBodies = int.Parse(nums.Dequeue(), CultureInfo.InvariantCulture);
        double warningDays = double.Parse(nums.Dequeue(), CultureInfo.InvariantCulture);
        double impactorMass = double.Parse(nums.Dequeue(), CultureInfo.InvariantCulture);
        double dvBudget = double.Parse(nums.Dequeue(), CultureInfo.InvariantCulture);
        double asteroidMass = double.Parse(nums.Dequeue(), CultureInfo.InvariantCulture);
        double t0Unix = double.Parse(nums.Dequeue(), CultureInfo.InvariantCulture);

        var bodies = new Dictionary<string, Body>(StringComparer.OrdinalIgnoreCase);
        int idx = 1;
        while (bodies.Count < numBodies && idx < input.Count)
        {
            var toks = Tokenize(input[idx++]);
            if (toks.Length < 8) continue;
            var b = new Body
            {
                Name = toks[0],
                Mass = double.Parse(toks[1], CultureInfo.InvariantCulture),
                Pos = new Vec3(
                    double.Parse(toks[2], CultureInfo.InvariantCulture),
                    double.Parse(toks[3], CultureInfo.InvariantCulture),
                    double.Parse(toks[4], CultureInfo.InvariantCulture)),
                Vel = new Vec3(
                    double.Parse(toks[5], CultureInfo.InvariantCulture),
                    double.Parse(toks[6], CultureInfo.InvariantCulture),
                    double.Parse(toks[7], CultureInfo.InvariantCulture))
            };
            bodies[b.Name] = b;
        }

        string bspPath = idx < input.Count ? input[idx].Trim() : "";

        bodies.TryGetValue("Earth", out var earth);
        bodies.TryGetValue("Asteroid", out var asteroid);
        if (earth == null || asteroid == null)
        {
            Console.WriteLine("1");
            Console.WriteLine("0 0 0 0");
            return;
        }

        double warningSec = warningDays * DAY;
        double launchSec = Math.Max(0.0, Math.Min(warningSec, warningSec * 0.1));
        double impactGuessSec = EstimateImpactTime(earth.Pos, earth.Vel, asteroid.Pos, asteroid.Vel, warningSec);
        double tofSec = Math.Max(1.0, impactGuessSec - launchSec);

        Vec3 earthAtImpact = earth.Pos + earth.Vel * impactGuessSec;
        Vec3 astAtImpact = asteroid.Pos + asteroid.Vel * impactGuessSec;
        Vec3 rel = astAtImpact - earthAtImpact;
        Vec3 relVel = asteroid.Vel - earth.Vel;

        Vec3 missDir = rel.Norm() > EPS ? rel.Normalized() : new Vec3(0, 1, 0);
        Vec3 impactTarget = astAtImpact + missDir * Math.Min(1000000.0, Math.Max(200000.0, rel.Norm() * 0.05));
        Vec3 desiredVel = (impactTarget - asteroid.Pos) / Math.Max(1.0, impactGuessSec);
        Vec3 dvNeeded = desiredVel - asteroid.Vel;

        double momentumFactor = impactorMass / Math.Max(EPS, asteroidMass);
        double asteroidDvFromImpact = dvBudget * momentumFactor;
        double requiredAsteroidDeflect = EstimateRequiredDeflection(earthAtImpact, astAtImpact, relVel, impactGuessSec);

        Vec3 deflectDir = PickPerpendicular(relVel, rel);
        if (deflectDir.Norm() < EPS) deflectDir = PickPerpendicular(rel, new Vec3(1, 0, 0));
        if (deflectDir.Norm() < EPS) deflectDir = new Vec3(0, 1, 0);
        deflectDir = deflectDir.Normalized();

        double totalAsteroidDv = Math.Max(requiredAsteroidDeflect, Math.Min(asteroidDvFromImpact, dvBudget * momentumFactor));
        if (double.IsNaN(totalAsteroidDv) || double.IsInfinity(totalAsteroidDv)) totalAsteroidDv = 0.0;
        Vec3 appliedAsteroidDv = deflectDir * totalAsteroidDv;

        Vec3 impactBurn = ClampToBudget(dvNeeded, dvBudget * 0.7);
        double remaining = Math.Max(0.0, dvBudget - impactBurn.Norm());
        Vec3 correctionBurn = ClampToBudget(appliedAsteroidDv * (impactorMass / Math.Max(EPS, asteroidMass)), remaining * 0.7);

        var burns = new List<(double t, Vec3 dv)>();
        if (impactBurn.Norm() > 1e-9)
            burns.Add((launchSec / DAY, impactBurn));
        if (correctionBurn.Norm() > 1e-9)
            burns.Add((Math.Max(0.0, impactGuessSec / DAY * 0.5), correctionBurn));

        Vec3 terminalDv = appliedAsteroidDv - correctionBurn;
        if (terminalDv.Norm() < 1e-9)
        {
            terminalDv = deflectDir * Math.Min(dvBudget - SumNorm(burns), Math.Max(0.0, dvBudget * 0.1));
        }
        if (terminalDv.Norm() > 1e-9)
            burns.Add((Math.Max(0.0, impactGuessSec / DAY), terminalDv));

        if (burns.Count == 0)
            burns.Add((0.0, new Vec3(0, 0, 0)));

        Console.WriteLine(burns.Count);
        foreach (var b in burns)
        {
            Console.WriteLine(string.Format(CultureInfo.InvariantCulture, "{0:R} {1:R} {2:R} {3:R}",
                b.t, b.dv.X, b.dv.Y, b.dv.Z));
        }
    }

    private static string[] Tokenize(string s) => s.Split((char[]?)null, StringSplitOptions.RemoveEmptyEntries);

    private static double SumNorm(List<(double t, Vec3 dv)> burns)
    {
        double sum = 0;
        foreach (var b in burns) sum += b.dv.Norm();
        return sum;
    }

    private static Vec3 ClampToBudget(Vec3 dv, double budget)
    {
        double n = dv.Norm();
        if (n <= budget || n < EPS) return dv;
        return dv * (budget / n);
    }

    private static Vec3 PickPerpendicular(Vec3 a, Vec3 b)
    {
        Vec3 c = Cross(a, b);
        if (c.Norm() > EPS) return c;
        c = Cross(a, new Vec3(1, 0, 0));
        if (c.Norm() > EPS) return c;
        return Cross(a, new Vec3(0, 1, 0));
    }

    private static Vec3 Cross(Vec3 a, Vec3 b)
    {
        return new Vec3(
            a.Y * b.Z - a.Z * b.Y,
            a.Z * b.X - a.X * b.Z,
            a.X * b.Y - a.Y * b.X);
    }

    private static double EstimateImpactTime(Vec3 earthPos, Vec3 earthVel, Vec3 astPos, Vec3 astVel, double maxSec)
    {
        Vec3 r = astPos - earthPos;
        Vec3 v = astVel - earthVel;
        double vv = v.Norm2();
        if (vv < EPS) return maxSec * 0.8;
        double t = -r.Dot(v) / vv;
        t = Math.Max(0.0, Math.Min(maxSec, t));
        return t;
    }

    private static double EstimateRequiredDeflection(Vec3 earthAtImpact, Vec3 astAtImpact, Vec3 relVel, double impactGuessSec)
    {
        double miss = (astAtImpact - earthAtImpact).Norm();
        double closing = Math.Max(1.0, relVel.Norm());
        double needed = Math.Max(0.0, 1e6 - miss);
        return needed / Math.Max(1.0, impactGuessSec) + 0.001 * closing;
    }
}
