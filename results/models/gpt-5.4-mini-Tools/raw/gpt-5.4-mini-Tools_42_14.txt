OpCapability Geometry
OpMemoryModel Logical GLSL450
%GLSLstd450 = OpExtInstImport "GLSL.std.450"
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpSource GLSL 450
OpName %main "main"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpDecorate %gl_PerVertex_in Block
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn = OpTypeFunction %void
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%uint_6 = OpConstant %uint 6
%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1.0
%float_1_over_3 = OpConstant %float 0.333333343
%v3float_111 = OpConstantComposite %v3float %float_1 %float_1 %float_1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%arr_v3_6 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_6
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %fn
%label = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p0_v4 = OpLoad %v4float %p0_ptr
%p2_v4 = OpLoad %v4float %p2_ptr
%p4_v4 = OpLoad %v4float %p4_ptr
%p0 = OpVectorShuffle %v3float %p0_v4 %p0_v4 0 1 2
%p2 = OpVectorShuffle %v3float %p2_v4 %p2_v4 0 1 2
%p4 = OpVectorShuffle %v3float %p4_v4 %p4_v4 0 1 2

%a = OpFSub %v3float %p2 %p0
%b = OpFSub %v3float %p4 %p0
%ax = OpCompositeExtract %float %a 0
%ay = OpCompositeExtract %float %a 1
%az = OpCompositeExtract %float %a 2
%bx = OpCompositeExtract %float %b 0
%by = OpCompositeExtract %float %b 1
%bz = OpCompositeExtract %float %b 2
%aybz = OpFMul %float %ay %bz
%azby = OpFMul %float %az %by
%azbx = OpFMul %float %az %bx
%axbz = OpFMul %float %ax %bz
%axby = OpFMul %float %ax %by
%aybx = OpFMul %float %ay %bx
%cx = OpFSub %float %aybz %azby
%cy = OpFSub %float %azbx %axbz
%cz = OpFSub %float %axby %aybx
%cross = OpCompositeConstruct %v3float %cx %cy %cz
%N = OpExtInst %v3float %GLSLstd450 Normalize %cross
%L = OpExtInst %v3float %GLSLstd450 Normalize %v3float_111
%diff0 = OpDot %float %N %L
%diffuse = OpExtInst %float %GLSLstd450 FMax %diff0 %float_0

%c0 = OpLoad %v3float %color %int_0
%c2 = OpLoad %v3float %color %int_2
%c4 = OpLoad %v3float %color %int_4
%sum1 = OpFAdd %v3float %c0 %c2
%sum2 = OpFAdd %v3float %sum1 %c4
%base = OpVectorTimesScalar %v3float %sum2 %float_1_over_3
%diffuseScale = OpFMul %float %float_0_8 %diffuse
%shade = OpFAdd %float %float_0_2 %diffuseScale
%finalColor = OpVectorTimesScalar %v3float %base %shade

%outPos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outPos0 %p0_v4
OpStore %outColor %finalColor
OpEmitVertex

%outPos1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outPos1 %p2_v4
OpStore %outColor %finalColor
OpEmitVertex

%outPos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outPos2 %p4_v4
OpStore %outColor %finalColor
OpEmitVertex
OpEndPrimitive
OpFunctionEnd
