import sys


def main():
    data = sys.stdin.read().strip().splitlines()
    if not data:
        return

    first = data[0].split()
    if len(first) != 2:
        print("NO SOLUTION")
        return

    U = int(first[0])
    S = int(first[1])
    sets = []
    for i in range(S):
        if i + 1 < len(data):
            line = data[i + 1].strip()
            if line:
                s = list(map(int, line.split()))
            else:
                s = []
        else:
            s = []
        sets.append(s)

    elem_to_sets = [[] for _ in range(U)]
    for idx, s in enumerate(sets):
        seen = set()
        ok = True
        for x in s:
            if x < 0 or x >= U or x in seen:
                ok = False
                break
            seen.add(x)
        if not ok:
            continue
        for x in seen:
            elem_to_sets[x].append(idx)

    cover_count = [0] * U
    chosen = []

    def select(idx):
        chosen.append(idx)
        for x in sets[idx]:
            cover_count[x] += 1

    def deselect(idx):
        for x in sets[idx]:
            cover_count[x] -= 1
        chosen.pop()

    def search():
        # Find an uncovered element with the fewest compatible candidate sets.
        target = -1
        options = None
        best_len = 10**9
        for e in range(U):
            if cover_count[e] != 0:
                continue
            cand = []
            for idx in elem_to_sets[e]:
                good = True
                for x in sets[idx]:
                    if cover_count[x] != 0:
                        good = False
                        break
                if good:
                    cand.append(idx)
            if not cand:
                return False
            if len(cand) < best_len:
                best_len = len(cand)
                target = e
                options = cand
                if best_len == 1:
                    break

        if target == -1:
            print("SOLUTION")
            print(" ".join(map(str, chosen)))
            return True

        for idx in options:
            select(idx)
            if search():
                return True
            deselect(idx)
        return False

    if not search():
        print("NO SOLUTION")


if __name__ == "__main__":
    main()
