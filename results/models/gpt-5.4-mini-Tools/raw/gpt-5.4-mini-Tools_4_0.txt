#include <cmath>
#include <cstdint>
#include <iostream>
#include <limits>
#include <array>
#include <vector>
#include <algorithm>

using namespace std;

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
    Vec3 operator+(const Vec3& o) const { return Vec3(x + o.x, y + o.y, z + o.z); }
    Vec3 operator-(const Vec3& o) const { return Vec3(x - o.x, y - o.y, z - o.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
};

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return Vec3(
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    );
}

static inline double norm2(const Vec3& a) { return dot(a, a); }

struct Face {
    Vec3 n;
    double d;
};

static vector<Face> buildHalfspaces(const vector<Vec3>& p) {
    Vec3 c(0, 0, 0);
    for (auto& v : p) c = c + v;
    c = c * (1.0 / p.size());

    vector<Face> faces;
    int V = (int)p.size();
    for (int i = 0; i < V; ++i) {
        for (int j = i + 1; j < V; ++j) {
            for (int k = j + 1; k < V; ++k) {
                Vec3 a = p[j] - p[i];
                Vec3 b = p[k] - p[i];
                Vec3 n = cross(a, b);
                double nn = norm2(n);
                if (nn < 1e-18) continue;
                double sign = 0.0;
                bool valid = true;
                for (int t = 0; t < V; ++t) {
                    if (t == i || t == j || t == k) continue;
                    double s = dot(n, p[t] - p[i]);
                    if (fabs(s) < 1e-12) continue;
                    if (sign == 0.0) sign = (s > 0.0 ? 1.0 : -1.0);
                    else if ((s > 0.0 ? 1.0 : -1.0) != sign) { valid = false; break; }
                }
                if (!valid) continue;
                if (dot(n, c - p[i]) > 0.0) n = n * -1.0;
                Face f;
                f.n = n;
                f.d = dot(n, p[i]);
                faces.push_back(f);
            }
        }
    }
    return faces;
}

static bool insidePoly(const vector<Face>& faces, const Vec3& q) {
    for (const auto& f : faces) {
        if (dot(f.n, q) - f.d > 1e-9) return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double e;
    if (!(cin >> V >> e)) return 0;
    vector<Vec3> p(V);
    for (int i = 0; i < V; ++i) cin >> p[i].x >> p[i].y >> p[i].z;

    auto faces = buildHalfspaces(p);
    if (faces.empty() || V < 4) {
        cout << 0 << '\n';
        return 0;
    }

    Vec3 mn(numeric_limits<double>::infinity(), numeric_limits<double>::infinity(), numeric_limits<double>::infinity());
    Vec3 mx(-numeric_limits<double>::infinity(), -numeric_limits<double>::infinity(), -numeric_limits<double>::infinity());
    for (auto& v : p) {
        mn.x = min(mn.x, v.x); mn.y = min(mn.y, v.y); mn.z = min(mn.z, v.z);
        mx.x = max(mx.x, v.x); mx.y = max(mx.y, v.y); mx.z = max(mx.z, v.z);
    }

    const double s = e;
    const double h = s * sqrt(2.0 / 3.0);
    const double stepX = s;
    const double stepY = sqrt(3.0) * 0.5 * s;
    const double stepZ = h;

    vector<array<double, 6>> out;
    out.reserve(100000);

    int xi0 = (int)floor(mn.x / stepX) - 2;
    int yi0 = (int)floor(mn.y / stepY) - 2;
    int zi0 = (int)floor(mn.z / stepZ) - 2;
    int xi1 = (int)ceil(mx.x / stepX) + 2;
    int yi1 = (int)ceil(mx.y / stepY) + 2;
    int zi1 = (int)ceil(mx.z / stepZ) + 2;

    Vec3 localVerts[4] = {
        Vec3(0, 0, 0),
        Vec3(s, 0, 0),
        Vec3(0.5 * s, sqrt(3.0) * 0.5 * s, 0),
        Vec3(0.5 * s, sqrt(3.0) / 6.0 * s, h)
    };

    for (int z = zi0; z <= zi1; ++z) {
        for (int y = yi0; y <= yi1; ++y) {
            for (int x = xi0; x <= xi1; ++x) {
                for (int parity = 0; parity < 2; ++parity) {
                    double ox = x * stepX + (parity ? 0.5 * s : 0.0);
                    double oy = y * stepY + (parity ? sqrt(3.0) / 6.0 * s : 0.0);
                    double oz = z * stepZ + (parity ? h * 0.5 : 0.0);
                    Vec3 base(ox, oy, oz);

                    Vec3 center = base + Vec3(0.5 * s, sqrt(3.0) / 6.0 * s, h * 0.25);
                    if (!insidePoly(faces, center)) continue;

                    bool allInside = true;
                    for (int i = 0; i < 4; ++i) {
                        if (!insidePoly(faces, base + localVerts[i])) { allInside = false; break; }
                    }
                    if (!allInside) continue;

                    out.push_back({center.x, center.y, center.z, 0.0, 0.0, 0.0});
                }
            }
        }
    }

    cout.setf(std::ios::fixed);
    cout.precision(10);
    cout << out.size() << '\n';
    for (auto& t : out) {
        cout << t[0] << ' ' << t[1] << ' ' << t[2] << ' ' << t[3] << ' ' << t[4] << ' ' << t[5] << '\n';
    }
    return 0;
}
