#include <algorithm>
#include <array>
#include <cmath>
#include <iomanip>
#include <iostream>
#include <limits>
#include <vector>

using namespace std;

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
    Vec3 operator+(const Vec3& o) const { return Vec3(x + o.x, y + o.y, z + o.z); }
    Vec3 operator-(const Vec3& o) const { return Vec3(x - o.x, y - o.y, z - o.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
};

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return Vec3(
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    );
}

static inline double norm(const Vec3& a) {
    return sqrt(dot(a, a));
}

struct Plane {
    Vec3 n;
    double d;
};

struct Face {
    int a, b, c;
};

static vector<Face> convexHullFaces(const vector<Vec3>& p) {
    const int n = (int)p.size();
    vector<Face> faces;
    if (n < 4) return faces;
    const double eps = 1e-10;
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            for (int k = j + 1; k < n; ++k) {
                Vec3 nrm = cross(p[j] - p[i], p[k] - p[i]);
                double ln = norm(nrm);
                if (ln < eps) continue;
                int sign = 0;
                bool ok = true;
                for (int m = 0; m < n; ++m) {
                    if (m == i || m == j || m == k) continue;
                    double s = dot(nrm, p[m] - p[i]);
                    if (fabs(s) <= 1e-9) continue;
                    int cur = s > 0 ? 1 : -1;
                    if (sign == 0) sign = cur;
                    else if (sign != cur) { ok = false; break; }
                }
                if (!ok || sign == 0) continue;
                if (sign > 0) swap(j, k);
                faces.push_back({i, j, k});
            }
        }
    }
    return faces;
}

static vector<Plane> hullPlanes(const vector<Vec3>& pts) {
    vector<Plane> planes;
    auto faces = convexHullFaces(pts);
    for (auto f : faces) {
        Vec3 a = pts[f.a], b = pts[f.b], c = pts[f.c];
        Vec3 n = cross(b - a, c - a);
        double ln = norm(n);
        if (ln < 1e-12) continue;
        n = n * (1.0 / ln);
        double d = dot(n, a);
        bool flipped = false;
        for (const auto& q : pts) {
            if (dot(n, q) > d + 1e-8) { flipped = true; break; }
        }
        if (flipped) {
            n = n * -1.0;
            d = -d;
        }
        planes.push_back({n, d});
    }
    return planes;
}

static vector<Vec3> readPoints(int v) {
    vector<Vec3> pts(v);
    for (int i = 0; i < v; ++i) cin >> pts[i].x >> pts[i].y >> pts[i].z;
    return pts;
}

static pair<Vec3, Vec3> bbox(const vector<Vec3>& pts) {
    Vec3 mn(numeric_limits<double>::infinity(), numeric_limits<double>::infinity(), numeric_limits<double>::infinity());
    Vec3 mx(-numeric_limits<double>::infinity(), -numeric_limits<double>::infinity(), -numeric_limits<double>::infinity());
    for (auto& p : pts) {
        mn.x = min(mn.x, p.x); mn.y = min(mn.y, p.y); mn.z = min(mn.z, p.z);
        mx.x = max(mx.x, p.x); mx.y = max(mx.y, p.y); mx.z = max(mx.z, p.z);
    }
    return {mn, mx};
}

struct Tet {
    Vec3 c;
    double rx, ry, rz;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double edge;
    if (!(cin >> V >> edge)) return 0;
    auto pts = readPoints(V);
    auto [mn, mx] = bbox(pts);
    double dx = mx.x - mn.x, dy = mx.y - mn.y, dz = mx.z - mn.z;
    double volume = max(0.0, dx * dy * dz);
    double cell = max(1e-9, edge * 0.8660254037844386);
    int nx = max(1, (int)floor(dx / cell));
    int ny = max(1, (int)floor(dy / cell));
    int nz = max(1, (int)floor(dz / cell));
    long long cap = 1LL * nx * ny * nz * 2;
    if (volume < 1e-12) cap = 0;
    vector<Tet> ans;
    ans.reserve((size_t)min<long long>(cap, 200000));

    if (cap > 0) {
        Vec3 base = mn;
        double step = cell;
        for (int ix = 0; ix < nx; ++ix) {
            for (int iy = 0; iy < ny; ++iy) {
                for (int iz = 0; iz < nz; ++iz) {
                    double x = base.x + (ix + 0.5) * step;
                    double y = base.y + (iy + 0.5) * step;
                    double z = base.z + (iz + 0.5) * step;
                    if (x > mx.x || y > mx.y || z > mx.z) continue;
                    ans.push_back({Vec3(x, y, z), 0.0, 0.0, 0.0});
                    if ((int)ans.size() >= 200000) break;
                    if (x + 0.25 * step <= mx.x && y + 0.25 * step <= mx.y && z + 0.25 * step <= mx.z) {
                        ans.push_back({Vec3(x + 0.25 * step, y + 0.25 * step, z + 0.25 * step), 0.0, 1.5707963267948966, 0.0});
                    }
                    if ((int)ans.size() >= 200000) break;
                }
                if ((int)ans.size() >= 200000) break;
            }
            if ((int)ans.size() >= 200000) break;
        }
    }

    cout.setf(ios::fixed);
    cout << setprecision(10);
    cout << ans.size() << "\n";
    for (auto& t : ans) {
        cout << t.c.x << ' ' << t.c.y << ' ' << t.c.z << ' ' << t.rx << ' ' << t.ry << ' ' << t.rz << "\n";
    }
    return 0;
}
