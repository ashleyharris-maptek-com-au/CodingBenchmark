const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.grid || [];
const robots = data.robots || [];
const jobs = data.jobs || [];
const H = grid.length;
const W = H ? grid[0].length : 0;

function idx(x, y) { return y * W + x; }
function inBounds(x, y) { return x >= 0 && y >= 0 && x < W && y < H; }
function isOpen(x, y) { return inBounds(x, y) && grid[y][x] !== '#'; }

const robotCount = robots.length;
const jobCount = jobs.length;
const assignments = Array.from({ length: robotCount }, () => []);

if (jobCount === 0 || robotCount === 0) {
  process.stdout.write(JSON.stringify({ assignments }));
  process.exit(0);
}

const jobOrder = jobs.slice().sort((a, b) => {
  if (b.priority !== a.priority) return b.priority - a.priority;
  if (a.y !== b.y) return a.y - b.y;
  if (a.x !== b.x) return a.x - b.x;
  return a.id - b.id;
});

// Precompute job positions for quick access.
const jobPos = jobs.map(j => ({ x: j.x, y: j.y, priority: j.priority, id: j.id }));
const robotPos = robots.map(r => ({ x: r.x, y: r.y, id: r.id }));

// Multi-source BFS from all robot start positions to estimate a good first assignment.
const INF = 1e9;
const distFromRobots = new Int32Array(H * W);
distFromRobots.fill(-1);
const owner = new Int32Array(H * W);
owner.fill(-1);
const qx = new Int32Array(H * W);
const qy = new Int32Array(H * W);
let qh = 0, qt = 0;
for (let i = 0; i < robotCount; i++) {
  const { x, y } = robotPos[i];
  if (!isOpen(x, y)) continue;
  const k = idx(x, y);
  if (distFromRobots[k] === -1) {
    distFromRobots[k] = 0;
    owner[k] = i;
    qx[qt] = x;
    qy[qt] = y;
    qt++;
  }
}
while (qh < qt) {
  const x = qx[qh], y = qy[qh++];
  const k = idx(x, y);
  const d = distFromRobots[k];
  const o = owner[k];
  const neighbors = [[x + 1, y], [x - 1, y], [x, y + 1], [x, y - 1]];
  for (let t = 0; t < 4; t++) {
    const nx = neighbors[t][0], ny = neighbors[t][1];
    if (!isOpen(nx, ny)) continue;
    const nk = idx(nx, ny);
    if (distFromRobots[nk] !== -1) continue;
    distFromRobots[nk] = d + 1;
    owner[nk] = o;
    qx[qt] = nx;
    qy[qt] = ny;
    qt++;
  }
}

// Use a two-stage heuristic:
// 1) seed each job to the robot with the smallest grid distance from the robot start.
// 2) rebalance by always giving the next job to the robot whose route increases least.
const robotRoutes = Array.from({ length: robotCount }, (_, i) => ({
  x: robotPos[i].x,
  y: robotPos[i].y,
  cost: 0,
  load: 0,
}));

function manhattan(a, b) {
  return Math.abs(a.x - b.x) + Math.abs(a.y - b.y);
}

// First pass: closest-start assignment by BFS distance where available, else Manhattan fallback.
for (const job of jobOrder) {
  let best = 0;
  let bestScore = INF;
  for (let r = 0; r < robotCount; r++) {
    let score;
    const startIdx = idx(jobPos[job.id].x, jobPos[job.id].y);
    if (distFromRobots[startIdx] >= 0 && owner[startIdx] === r) {
      score = distFromRobots[startIdx];
    } else {
      score = manhattan(robotRoutes[r], jobPos[job.id]);
    }
    score += robotRoutes[r].load * 0.001;
    if (score < bestScore) {
      bestScore = score;
      best = r;
    }
  }
  assignments[best].push(job.id);
  robotRoutes[best].x = jobPos[job.id].x;
  robotRoutes[best].y = jobPos[job.id].y;
  robotRoutes[best].load++;
}

// Local refinement: if a robot has a much longer route than another, move a low-priority tail job.
function routeLength(start, seq) {
  let x = start.x, y = start.y, total = 0;
  for (const jid of seq) {
    const j = jobPos[jid];
    total += Math.abs(x - j.x) + Math.abs(y - j.y);
    x = j.x; y = j.y;
  }
  return total;
}

for (let iter = 0; iter < 2; iter++) {
  const lengths = assignments.map((seq, i) => routeLength(robotPos[i], seq));
  let maxI = 0, minI = 0;
  for (let i = 1; i < robotCount; i++) {
    if (lengths[i] > lengths[maxI]) maxI = i;
    if (lengths[i] < lengths[minI]) minI = i;
  }
  if (lengths[maxI] <= lengths[minI] + 1) break;
  if (assignments[maxI].length <= 1) break;
  const moved = assignments[maxI].pop();
  assignments[minI].push(moved);
}

// Output assignments sorted by descending priority within each robot, then stable by id.
for (let i = 0; i < robotCount; i++) {
  assignments[i].sort((a, b) => {
    const ja = jobs[a], jb = jobs[b];
    if (jb.priority !== ja.priority) return jb.priority - ja.priority;
    return ja.id - jb.id;
  });
}

process.stdout.write(JSON.stringify({ assignments }));
