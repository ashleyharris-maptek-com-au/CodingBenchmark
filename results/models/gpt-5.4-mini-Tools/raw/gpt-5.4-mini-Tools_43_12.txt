float4 main(PSInput input) : SV_Target0
{
    float3 N = normalize(input.normal);
    float t = saturate(N.y * 0.5f + 0.5f);
    float3 skyColor = float3(0.4f, 0.6f, 1.0f);
    float3 groundColor = float3(0.3f, 0.15f, 0.05f);
    float3 ambient = lerp(groundColor, skyColor, t);

    float3 L = normalize(lightPos.xyz - input.worldPos);
    float d = max(dot(N, L), 0.0f);
    float3 color = ambient + 0.3f * d;

    return float4(color, 1.0f);
}
