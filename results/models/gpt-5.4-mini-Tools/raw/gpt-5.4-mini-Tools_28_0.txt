use std::collections::VecDeque;
use std::io::{self, Read, Write};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut u = Vec::with_capacity(m);
    let mut v = Vec::with_capacity(m);
    let mut deg = vec![0usize; n];

    for _ in 0..m {
        let a: usize = it.next().unwrap().parse().unwrap();
        let b: usize = it.next().unwrap().parse().unwrap();
        u.push(a);
        v.push(b);
        if a != b {
            deg[a] += 1;
            deg[b] += 1;
        } else {
            deg[a] += 2;
        }
    }

    let mut alive = vec![true; n];
    let mut cover = Vec::new();
    let mut q = VecDeque::new();

    for i in 0..n {
        if deg[i] == 0 {
            alive[i] = false;
        } else if deg[i] == 1 {
            q.push_back(i);
        }
    }

    while let Some(x) = q.pop_front() {
        if !alive[x] || deg[x] != 1 {
            continue;
        }

        let mut neighbor = None;
        for e in 0..m {
            let a = u[e];
            let b = v[e];
            if !alive[a] || !alive[b] {
                continue;
            }
            if a == x && b != x {
                neighbor = Some(b);
                break;
            }
            if b == x && a != x {
                neighbor = Some(a);
                break;
            }
            if a == x && b == x {
                neighbor = Some(x);
                break;
            }
        }

        let Some(y) = neighbor else {
            alive[x] = false;
            continue;
        };

        if alive[y] {
            alive[y] = false;
            cover.push(y);
        }
        alive[x] = false;
    }

    let mut uncovered = vec![false; m];
    for i in 0..m {
        let a = u[i];
        let b = v[i];
        if alive[a] && alive[b] {
            uncovered[i] = true;
        }
    }

    let mut changed = true;
    while changed {
        changed = false;
        let mut best_vertex = None;
        let mut best_score = 0usize;
        let mut local_deg = vec![0usize; n];

        for i in 0..m {
            if !uncovered[i] {
                continue;
            }
            let a = u[i];
            let b = v[i];
            if alive[a] {
                local_deg[a] += 1;
            }
            if alive[b] {
                local_deg[b] += 1;
            }
        }

        for i in 0..n {
            if alive[i] && local_deg[i] > best_score {
                best_score = local_deg[i];
                best_vertex = Some(i);
            }
        }

        if let Some(x) = best_vertex {
            if best_score == 0 {
                break;
            }
            alive[x] = false;
            cover.push(x);
            changed = true;
        }
    }

    let mut selected = vec![false; n];
    for &x in &cover {
        if x < n {
            selected[x] = true;
        }
    }

    let mut out = Vec::new();
    for i in 0..n {
        if selected[i] {
            out.push(i);
        }
    }

    let mut covered = vec![false; m];
    for i in 0..m {
        let a = u[i];
        let b = v[i];
        if a == b {
            if selected[a] {
                covered[i] = true;
            }
        } else if selected[a] || selected[b] {
            covered[i] = true;
        }
    }

    let mut remaining_edges = Vec::new();
    for i in 0..m {
        if !covered[i] {
            remaining_edges.push(i);
        }
    }

    if !remaining_edges.is_empty() {
        let mut remaining_vertices = vec![0usize; n];
        for &e in &remaining_edges {
            let a = u[e];
            let b = v[e];
            remaining_vertices[a] += 1;
            remaining_vertices[b] += 1;
        }

        loop {
            let mut best = None;
            let mut best_deg = 0usize;
            for i in 0..n {
                if !selected[i] && remaining_vertices[i] > best_deg {
                    best_deg = remaining_vertices[i];
                    best = Some(i);
                }
            }
            let Some(x) = best else { break };
            if best_deg == 0 {
                break;
            }
            selected[x] = true;
            out.push(x);
            for &e in &remaining_edges {
                let a = u[e];
                let b = v[e];
                if !covered[e] && (a == x || b == x) {
                    covered[e] = true;
                    if a != b {
                        if remaining_vertices[a] > 0 {
                            remaining_vertices[a] -= 1;
                        }
                        if remaining_vertices[b] > 0 {
                            remaining_vertices[b] -= 1;
                        }
                    } else if remaining_vertices[a] > 0 {
                        remaining_vertices[a] = remaining_vertices[a].saturating_sub(2);
                    }
                }
            }
        }
    }

    let mut final_cover = Vec::new();
    for i in 0..n {
        if selected[i] {
            final_cover.push(i);
        }
    }

    let mut stdout = io::BufWriter::new(io::stdout());
    writeln!(stdout, "{}", final_cover.len()).unwrap();
    if final_cover.is_empty() {
        writeln!(stdout).unwrap();
    } else {
        for (i, x) in final_cover.iter().enumerate() {
            if i > 0 {
                write!(stdout, " ").unwrap();
            }
            write!(stdout, "{}", x).unwrap();
        }
        writeln!(stdout).unwrap();
    }
}
