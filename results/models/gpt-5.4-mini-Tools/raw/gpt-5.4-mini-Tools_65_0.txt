declare const require: any;
const fs = require('fs');

type Task = {
  id: string;
  duration: number;
  deps: string[];
  deadline: number;
  weight: number;
};

type InputData = {
  machineCount: number;
  tasks: Task[];
};

const input: string = fs.readFileSync(0, 'utf8');
const data: InputData = JSON.parse(input);

const tasks = data.tasks;
const n = tasks.length;

const idToIndex = new Map<string, number>();
for (let i = 0; i < n; i++) idToIndex.set(tasks[i].id, i);

const indeg = new Int32Array(n);
const outs: number[][] = Array.from({ length: n }, () => []);
const critical = new Float64Array(n);

for (let i = 0; i < n; i++) {
  const t = tasks[i];
  let maxDepCritical = 0;
  for (const depId of t.deps) {
    const j = idToIndex.get(depId);
    if (j === undefined) continue;
    indeg[i]++;
    outs[j].push(i);
    if (critical[j] > maxDepCritical) maxDepCritical = critical[j];
  }
  critical[i] = maxDepCritical + t.duration;
}

class MaxHeap {
  private a: number[] = [];

  private better(i: number, j: number): boolean {
    const ti = tasks[i];
    const tj = tasks[j];
    if (critical[i] !== critical[j]) return critical[i] > critical[j];
    const si = ti.deadline - critical[i];
    const sj = tj.deadline - critical[j];
    if (si !== sj) return si < sj;
    if (ti.weight !== tj.weight) return ti.weight > tj.weight;
    if (ti.duration !== tj.duration) return ti.duration > tj.duration;
    return ti.id < tj.id;
  }

  push(v: number): void {
    const a = this.a;
    a.push(v);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.better(a[p], a[i])) break;
      [a[p], a[i]] = [a[i], a[p]];
      i = p;
    }
  }

  pop(): number | undefined {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop()!;
    if (a.length > 0) {
      a[0] = last;
      let i = 0;
      for (;;) {
        let l = i * 2 + 1;
        let r = l + 1;
        let best = i;
        if (l < a.length && !this.better(a[best], a[l])) best = l;
        if (r < a.length && !this.better(a[best], a[r])) best = r;
        if (best === i) break;
        [a[i], a[best]] = [a[best], a[i]];
        i = best;
      }
    }
    return top;
  }

  get size(): number {
    return this.a.length;
  }
}

const heap = new MaxHeap();
for (let i = 0; i < n; i++) if (indeg[i] === 0) heap.push(i);

const order: string[] = [];
while (heap.size > 0) {
  const i = heap.pop()!;
  order.push(tasks[i].id);
  for (const j of outs[i]) {
    indeg[j]--;
    if (indeg[j] === 0) heap.push(j);
  }
}

if (order.length !== n) {
  const seen = new Set(order);
  for (const t of tasks) if (!seen.has(t.id)) order.push(t.id);
}

process.stdout.write(JSON.stringify({ order }));
