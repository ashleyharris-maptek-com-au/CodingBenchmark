using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

public class Program
{
    struct Edge
    {
        public int U, V;
        public long W;
    }

    sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        private byte ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do c = ReadByte(); while (c <= 32 && c != 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }
            int val = 0;
            while (c > 32 && c != 0)
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return val * sign;
        }

        public long NextLong()
        {
            byte c;
            do c = ReadByte(); while (c <= 32 && c != 0);
            long sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }
            long val = 0;
            while (c > 32 && c != 0)
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return val * sign;
        }
    }

    sealed class MinHeap
    {
        private int[] node = new int[16];
        private long[] key = new long[16];
        public int Count { get; private set; }

        public void Push(int v, long k)
        {
            if (Count == node.Length)
            {
                Array.Resize(ref node, Count << 1);
                Array.Resize(ref key, Count << 1);
            }
            int i = Count++;
            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (key[p] <= k) break;
                node[i] = node[p];
                key[i] = key[p];
                i = p;
            }
            node[i] = v;
            key[i] = k;
        }

        public bool TryPop(out int v, out long k)
        {
            if (Count == 0)
            {
                v = 0; k = 0; return false;
            }
            v = node[0];
            k = key[0];
            int lastV = node[--Count];
            long lastK = key[Count];
            int i = 0;
            while (true)
            {
                int l = i * 2 + 1;
                if (l >= Count) break;
                int r = l + 1;
                int c = (r < Count && key[r] < key[l]) ? r : l;
                if (key[c] >= lastK) break;
                node[i] = node[c];
                key[i] = key[c];
                i = c;
            }
            node[i] = lastV;
            key[i] = lastK;
            return true;
        }
    }

    static int[] head, to, next, edgeId;
    static int edgePtr;
    static bool[] used;
    static int[] degree;
    static int[] odd;
    static long[] dist;
    static int[] prevV, prevE;

    static void AddDirected(int u, int v, int id)
    {
        to[edgePtr] = v;
        edgeId[edgePtr] = id;
        next[edgePtr] = head[u];
        head[u] = edgePtr++;
    }

    static void Dijkstra(int src, long[] w, int n)
    {
        Array.Fill(dist, long.MaxValue);
        Array.Fill(prevV, -1);
        Array.Fill(prevE, -1);
        var pq = new MinHeap();
        dist[src] = 0;
        pq.Push(src, 0);
        while (pq.TryPop(out int v, out long d))
        {
            if (d != dist[v]) continue;
            for (int e = head[v]; e != -1; e = next[e])
            {
                int nv = to[e];
                long nd = d + w[edgeId[e]];
                if (nd < dist[nv])
                {
                    dist[nv] = nd;
                    prevV[nv] = v;
                    prevE[nv] = edgeId[e];
                    pq.Push(nv, nd);
                }
            }
        }
    }

    static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int m = fs.NextInt();

        var edges = new Edge[m];
        var w = new long[m];
        degree = new int[n];
        head = new int[n];
        Array.Fill(head, -1);
        to = new int[m * 2];
        next = new int[m * 2];
        edgeId = new int[m * 2];
        used = new bool[m];
        edgePtr = 0;

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            long ww = fs.NextLong();
            edges[i] = new Edge { U = u, V = v, W = ww };
            w[i] = ww;
            degree[u]++;
            degree[v]++;
            AddDirected(u, v, i);
            AddDirected(v, u, i);
        }

        odd = new int[n];
        int oddCnt = 0;
        for (int i = 0; i < n; i++) if ((degree[i] & 1) != 0) odd[oddCnt++] = i;

        if (oddCnt == 0)
        {
            BuildEulerTour(n, m);
            return;
        }

        dist = new long[n];
        prevV = new int[n];
        prevE = new int[n];

        int pairCnt = oddCnt / 2;
        var dup = new List<int>(pairCnt * 4);

        if (oddCnt <= 14)
        {
            int full = 1 << oddCnt;
            var memoCost = new long[full];
            var memoPick = new int[full];
            Array.Fill(memoCost, -1);
            memoCost[0] = 0;

            long SolveMask(int mask)
            {
                if (memoCost[mask] >= 0) return memoCost[mask];
                int first = 0;
                while (((mask >> first) & 1) == 0) first++;
                long best = long.MaxValue;
                int bestj = -1;
                int m1 = mask ^ (1 << first);
                for (int j = first + 1; j < oddCnt; j++)
                {
                    if (((m1 >> j) & 1) == 0) continue;
                    int nmask = m1 ^ (1 << j);
                    long cand = SolveMask(nmask) + DistBetween(odd[first], odd[j], n);
                    if (cand < best)
                    {
                        best = cand;
                        bestj = j;
                    }
                }
                memoPick[mask] = bestj;
                return memoCost[mask] = best;
            }

            long DistBetween(int s, int t, int nLocal)
            {
                Dijkstra(s, w, nLocal);
                return dist[t];
            }

            SolveMask(full - 1);
            int mask2 = full - 1;
            while (mask2 != 0)
            {
                int first = 0;
                while (((mask2 >> first) & 1) == 0) first++;
                int j = memoPick[mask2];
                int a = odd[first], b = odd[j];
                Dijkstra(a, w, n);
                int cur = b;
                while (cur != a)
                {
                    int pe = prevE[cur];
                    if (pe < 0) break;
                    dup.Add(pe);
                    cur = prevV[cur];
                }
                mask2 ^= (1 << first) | (1 << j);
            }
        }
        else
        {
            var usedOdd = new bool[oddCnt];
            for (int i = 0; i < oddCnt; i++)
            {
                if (usedOdd[i]) continue;
                Dijkstra(odd[i], w, n);
                int bestJ = -1;
                long bestD = long.MaxValue;
                for (int j = i + 1; j < oddCnt; j++)
                {
                    if (usedOdd[j]) continue;
                    if (dist[odd[j]] < bestD)
                    {
                        bestD = dist[odd[j]];
                        bestJ = j;
                    }
                }
                if (bestJ < 0) break;
                usedOdd[i] = usedOdd[bestJ] = true;
                int a = odd[i], b = odd[bestJ];
                Dijkstra(a, w, n);
                int cur = b;
                while (cur != a)
                {
                    int pe = prevE[cur];
                    dup.Add(pe);
                    cur = prevV[cur];
                }
            }
        }

        int totalM = m + dup.Count;
        int[] eTo = new int[totalM * 2];
        int[] eNext = new int[totalM * 2];
        int[] eId = new int[totalM * 2];
        int[] eHead = new int[n];
        Array.Fill(eHead, -1);
        int p = 0;
        void AddAug(int u, int v, int id)
        {
            eTo[p] = v;
            eId[p] = id;
            eNext[p] = eHead[u];
            eHead[u] = p++;
        }

        for (int i = 0; i < m; i++)
        {
            AddAug(edges[i].U, edges[i].V, i);
            AddAug(edges[i].V, edges[i].U, i);
        }
        for (int i = 0; i < dup.Count; i++)
        {
            int id = dup[i];
            AddAug(edges[id].U, edges[id].V, m + i);
            AddAug(edges[id].V, edges[id].U, m + i);
        }

        var used2 = new bool[totalM];
        var st = new Stack<int>();
        var circuit = new List<int>(totalM + 1);
        st.Push(0);
        while (st.Count > 0)
        {
            int v = st.Peek();
            while (eHead[v] != -1 && used2[eId[eHead[v]]]) eHead[v] = eNext[eHead[v]];
            if (eHead[v] == -1)
            {
                circuit.Add(v);
                st.Pop();
            }
            else
            {
                int e = eHead[v];
                eHead[v] = eNext[e];
                int id = eId[e];
                if (used2[id]) continue;
                used2[id] = true;
                st.Push(eTo[e]);
            }
        }

        circuit.Reverse();
        var sb = new StringBuilder(circuit.Count * 3);
        for (int i = 0; i < circuit.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(circuit[i]);
        }
        Console.Write(sb.ToString());
    }

    static void BuildEulerTour(int n, int m)
    {
        var st = new Stack<int>();
        var circuit = new List<int>(m + 1);
        st.Push(0);
        while (st.Count > 0)
        {
            int v = st.Peek();
            while (head[v] != -1 && used[edgeId[head[v]]]) head[v] = next[head[v]];
            if (head[v] == -1)
            {
                circuit.Add(v);
                st.Pop();
            }
            else
            {
                int e = head[v];
                head[v] = next[e];
                int id = edgeId[e];
                if (used[id]) continue;
                used[id] = true;
                st.Push(to[e]);
            }
        }
        circuit.Reverse();
        var sb = new StringBuilder(circuit.Count * 3);
        for (int i = 0; i < circuit.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(circuit[i]);
        }
        Console.Write(sb.ToString());
    }
}
