using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;
using System.Text;

public static class Program
{
    const double MuDefault = 398600.4418;
    const double EarthRadius = 6371.0;
    const double TwoPi = Math.PI * 2.0;

    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double k) => new Vec3(a.X * k, a.Y * k, a.Z * k);
        public static Vec3 operator /(Vec3 a, double k) => new Vec3(a.X / k, a.Y / k, a.Z / k);
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public Vec3 Cross(Vec3 b) => new Vec3(Y * b.Z - Z * b.Y, Z * b.X - X * b.Z, X * b.Y - Y * b.X);
    }

    struct Orbit
    {
        public Vec3 R, V;
    }

    struct Elements
    {
        public double A, E, I, Omega, ArgP, Nu, M, Period;
        public bool Circular, Equatorial;
        public Vec3 H;
    }

    static double Mu;
    static Orbit[] Stations;
    static Elements[] Els;
    static double[] Radius;
    static double[] Weight;
    static double[,] Cost;

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        Mu = fs.NextDouble();
        Orbit start = new Orbit { R = ReadVec(fs), V = ReadVec(fs) };
        Stations = new Orbit[n];
        for (int i = 0; i < n; i++)
            Stations[i] = new Orbit { R = ReadVec(fs), V = ReadVec(fs) };

        Els = new Elements[n];
        Radius = new double[n];
        Weight = new double[n];
        for (int i = 0; i < n; i++)
        {
            Els[i] = ComputeElements(Stations[i]);
            Radius[i] = Stations[i].R.Norm();
            Weight[i] = Radius[i] / EarthRadius + Stations[i].V.Norm() / 7.5;
        }

        Cost = new double[n + 1, n + 1];
        for (int i = 0; i < n; i++)
        {
            Cost[n, i] = TransferCost(start, Stations[i]);
            Cost[i, n] = Cost[n, i];
        }
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                Cost[i, j] = i == j ? 0.0 : TransferCost(Stations[i], Stations[j]);

        int[] order = BuildOrder(n, start);
        Console.WriteLine(string.Join(" ", order));
    }

    static Vec3 ReadVec(FastScanner fs) => new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());

    static int[] BuildOrder(int n, Orbit start)
    {
        if (n <= 8)
        {
            return ExactOrNearExact(n, start);
        }

        int[] order = GreedyConstruct(n);
        if (n <= 25)
            order = ImproveOrder(order, 2500);
        else if (n <= 60)
            order = ImproveOrder(order, 1200);
        else
            order = ImproveOrder(order, 500);

        return order;
    }

    static int[] ExactOrNearExact(int n, Orbit start)
    {
        int[] best = null;
        double bestCost = double.PositiveInfinity;
        int[] perm = Enumerable.Range(0, n).ToArray();
        Permute(0);
        return best;

        void Permute(int idx)
        {
            if (idx == n)
            {
                double c = Cost[n, perm[0]];
                for (int i = 0; i < n - 1; i++) c += Cost[perm[i], perm[i + 1]];
                if (c < bestCost)
                {
                    bestCost = c;
                    best = (int[])perm.Clone();
                }
                return;
            }
            for (int i = idx; i < n; i++)
            {
                (perm[idx], perm[i]) = (perm[i], perm[idx]);
                Permute(idx + 1);
                (perm[idx], perm[i]) = (perm[i], perm[idx]);
            }
        }
    }

    static int[] GreedyConstruct(int n)
    {
        int startNode = -1;
        double best0 = double.PositiveInfinity;
        for (int i = 0; i < n; i++)
        {
            double c = Cost[n, i] + 0.25 * Weight[i];
            if (c < best0) { best0 = c; startNode = i; }
        }

        var order = new List<int>(n) { startNode };
        var used = new bool[n];
        used[startNode] = true;

        for (int step = 1; step < n; step++)
        {
            int last = order[order.Count - 1];
            int best = -1;
            double bestScore = double.PositiveInfinity;
            for (int i = 0; i < n; i++)
            {
                if (used[i]) continue;
                double score = Cost[last, i] + 0.12 * Weight[i];
                if (step + 1 < n) score += 0.05 * MinOutgoing(i, used);
                if (score < bestScore)
                {
                    bestScore = score;
                    best = i;
                }
            }
            order.Add(best);
            used[best] = true;
        }

        return order.ToArray();
    }

    static double MinOutgoing(int i, bool[] used)
    {
        double best = double.PositiveInfinity;
        for (int j = 0; j < used.Length; j++)
            if (!used[j] && i != j)
                best = Math.Min(best, Cost[i, j]);
        return double.IsInfinity(best) ? 0.0 : best;
    }

    static int[] ImproveOrder(int[] order, int iterations)
    {
        int n = order.Length;
        double current = RouteCost(order);
        var rng = new XorShift64(123456789);
        for (int it = 0; it < iterations; it++)
        {
            bool improved = false;

            for (int i = 0; i < n; i++)
            {
                int a = order[(i - 1 + n) % n];
                int b = order[i];
                int c = order[(i + 1) % n];
                for (int j = i + 2; j < n; j++)
                {
                    if (i == 0 && j == n - 1) continue;
                    int d = order[(j + 1) % n];
                    int e = order[j];
                    double delta = -Cost[a, b] - Cost[e, d] + Cost[a, e] + Cost[b, d];
                    if (delta < -1e-9)
                    {
                        Reverse(order, i, j);
                        current += delta;
                        improved = true;
                    }
                }
            }

            if (!improved)
            {
                int i = rng.Next(n);
                int j = rng.Next(n);
                if (i > j) (i, j) = (j, i);
                if (j - i > 1)
                {
                    Reverse(order, i, j);
                    double nc = RouteCost(order);
                    if (nc + 1e-12 < current)
                        current = nc;
                    else
                        Reverse(order, i, j);
                }
            }
        }
        return order;
    }

    static double RouteCost(int[] order)
    {
        double c = Cost[order[order.Length - 1], order[0]];
        for (int i = 0; i < order.Length - 1; i++) c += Cost[order[i], order[i + 1]];
        return c;
    }

    static void Reverse(int[] arr, int i, int j)
    {
        while (i < j) { (arr[i], arr[j]) = (arr[j], arr[i]); i++; j--; }
    }

    static double TransferCost(Orbit from, Orbit to)
    {
        // Heuristic: combine Lambert-like geometric difficulty, orbit mismatch, and angular momentum change.
        Vec3 dr = to.R - from.R;
        Vec3 dv = to.V - from.V;
        double dist = dr.Norm();
        double dvMag = dv.Norm();
        double r1 = from.R.Norm();
        double r2 = to.R.Norm();

        Vec3 h1 = from.R.Cross(from.V);
        Vec3 h2 = to.R.Cross(to.V);
        double hDiff = (h1 - h2).Norm() / 10000.0;

        double energy1 = 0.5 * from.V.Dot(from.V) - Mu / r1;
        double energy2 = 0.5 * to.V.Dot(to.V) - Mu / r2;
        double eDiff = Math.Abs(energy2 - energy1) / 50.0;

        double plane = 0.0;
        double h1n = h1.Norm();
        double h2n = h2.Norm();
        if (h1n > 1e-9 && h2n > 1e-9)
        {
            double cos = h1.Dot(h2) / (h1n * h2n);
            cos = Math.Max(-1.0, Math.Min(1.0, cos));
            plane = (1.0 - cos) * 5.0;
        }

        double radiusPenalty = Math.Abs(r1 - r2) / 1000.0;
        return dist / 1200.0 + dvMag + hDiff + eDiff + plane + radiusPenalty;
    }

    static Elements ComputeElements(Orbit o)
    {
        Vec3 r = o.R;
        Vec3 v = o.V;
        double rmag = r.Norm();
        double vmag2 = v.Dot(v);
        Vec3 h = r.Cross(v);
        double hmag = h.Norm();
        Vec3 n = new Vec3(-h.Y, h.X, 0.0);
        double nmag = n.Norm();
        Vec3 evec = (v.Cross(h) / Mu) - (r / rmag);
        double e = evec.Norm();
        double energy = vmag2 / 2.0 - Mu / rmag;
        double a = Math.Abs(energy) < 1e-12 ? 1e12 : -Mu / (2.0 * energy);
        double inc = hmag < 1e-12 ? 0.0 : Math.Acos(Math.Max(-1.0, Math.Min(1.0, h.Z / hmag)));

        double Omega = 0.0;
        if (nmag > 1e-12)
        {
            Omega = Math.Acos(Math.Max(-1.0, Math.Min(1.0, n.X / nmag)));
            if (n.Y < 0) Omega = TwoPi - Omega;
        }

        double argP = 0.0;
        if (nmag > 1e-12 && e > 1e-12)
        {
            argP = Math.Acos(Math.Max(-1.0, Math.Min(1.0, n.Dot(evec) / (nmag * e))));
            if (evec.Z < 0) argP = TwoPi - argP;
        }

        double nu = 0.0;
        if (e > 1e-12)
        {
            nu = Math.Acos(Math.Max(-1.0, Math.Min(1.0, evec.Dot(r) / (e * rmag))));
            if (r.Dot(v) < 0) nu = TwoPi - nu;
        }
        else if (nmag > 1e-12)
        {
            nu = Math.Acos(Math.Max(-1.0, Math.Min(1.0, n.Dot(r) / (nmag * rmag))));
            if (r.Z < 0) nu = TwoPi - nu;
        }

        double E = 0.0;
        if (e < 1.0)
        {
            double cosE = (1.0 - rmag / a) / e;
            cosE = Math.Max(-1.0, Math.Min(1.0, cosE));
            E = Math.Acos(cosE);
            if (r.Dot(v) < 0) E = TwoPi - E;
        }
        double M = e < 1.0 ? E - e * Math.Sin(E) : 0.0;
        double period = a > 0 && e < 1.0 ? TwoPi * Math.Sqrt(a * a * a / Mu) : double.PositiveInfinity;

        return new Elements
        {
            A = a, E = e, I = inc, Omega = Omega, ArgP = argP, Nu = nu, M = M,
            Period = period, Circular = e < 1e-6, Equatorial = inc < 1e-6, H = h
        };
    }

    sealed class FastScanner
    {
        private readonly System.IO.Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            int c; do c = Read(); while (c <= 32);
            int sign = 1; if (c == '-') { sign = -1; c = Read(); }
            int v = 0;
            while (c > 32) { v = v * 10 + c - '0'; c = Read(); }
            return v * sign;
        }

        public double NextDouble()
        {
            int c; do c = Read(); while (c <= 32);
            int sign = 1; if (c == '-') { sign = -1; c = Read(); }
            double val = 0;
            while (c > 32 && c != '.') { val = val * 10 + (c - '0'); c = Read(); }
            if (c == '.')
            {
                double mul = 0.1;
                c = Read();
                while (c > 32)
                {
                    val += (c - '0') * mul;
                    mul *= 0.1;
                    c = Read();
                }
            }
            return val * sign;
        }
    }

    sealed class XorShift64
    {
        private ulong x;
        public XorShift64(ulong seed) { x = seed == 0 ? 88172645463393265UL : seed; }
        public int Next(int n)
        {
            x ^= x << 7;
            x ^= x >> 9;
            return (int)(x % (uint)n);
        }
    }
}
