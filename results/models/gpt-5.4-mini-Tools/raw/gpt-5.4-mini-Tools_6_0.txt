using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    const double EarthRadius = 6371.0;
    const double Eps = 1e-12;

    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double k) => new Vec3(a.X * k, a.Y * k, a.Z * k);
        public static Vec3 operator /(Vec3 a, double k) => new Vec3(a.X / k, a.Y / k, a.Z / k);
        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public double Norm2() => X * X + Y * Y + Z * Z;
        public double Norm() => Math.Sqrt(Norm2());
    }

    struct State
    {
        public Vec3 R, V;
        public State(Vec3 r, Vec3 v) { R = r; V = v; }
    }

    struct Station
    {
        public int Id;
        public State S;
        public double Radius;
        public double Speed;
        public double Alt;
        public double IncHint;
        public Station(int id, State s)
        {
            Id = id; S = s;
            Radius = s.R.Norm();
            Speed = s.V.Norm();
            Alt = Radius - EarthRadius;
            var h = Cross(s.R, s.V);
            IncHint = Math.Acos(Clamp(h.Z / (h.Norm() + Eps), -1.0, 1.0));
        }
    }

    static void Main()
    {
        var ci = CultureInfo.InvariantCulture;
        var tokens = Console.In.ReadToEnd().Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        if (tokens.Length < 2) return;
        int p = 0;
        int n = int.Parse(tokens[p++], ci);
        double mu = double.Parse(tokens[p++], ci);

        var start = new State(
            new Vec3(double.Parse(tokens[p++], ci), double.Parse(tokens[p++], ci), double.Parse(tokens[p++], ci)),
            new Vec3(double.Parse(tokens[p++], ci), double.Parse(tokens[p++], ci), double.Parse(tokens[p++], ci)));

        var stations = new Station[n];
        for (int i = 0; i < n; i++)
        {
            var s = new State(
                new Vec3(double.Parse(tokens[p++], ci), double.Parse(tokens[p++], ci), double.Parse(tokens[p++], ci)),
                new Vec3(double.Parse(tokens[p++], ci), double.Parse(tokens[p++], ci), double.Parse(tokens[p++], ci)));
            stations[i] = new Station(i, s);
        }

        var cost = BuildCostMatrix(start, stations, mu);
        int[] order = Solve(stations, cost);
        Console.WriteLine(string.Join(" ", order));
    }

    static int[] Solve(Station[] stations, double[,] cost)
    {
        int n = stations.Length;
        if (n <= 1) return Enumerable.Range(0, n).ToArray();
        if (n <= 10) return ExactSearch(cost);

        var order = GreedyConstruction(stations, cost);
        Improve2Opt(order, cost, 4000);
        if (n <= 30) ImproveOrOpt(order, cost, 2000);
        if (n > 30) ImproveLight(order, cost, 1500);
        return order;
    }

    static int[] ExactSearch(double[,] cost)
    {
        int n = cost.GetLength(0);
        if (n <= 10)
        {
            int[] best = null;
            double bestCost = double.PositiveInfinity;
            int[] perm = Enumerable.Range(0, n).ToArray();
            Permute(0);
            return best ?? perm;

            void Permute(int idx)
            {
                if (idx == n)
                {
                    double c = PathCost(perm, cost);
                    if (c < bestCost)
                    {
                        bestCost = c;
                        best = (int[])perm.Clone();
                    }
                    return;
                }
                for (int i = idx; i < n; i++)
                {
                    Swap(ref perm[idx], ref perm[i]);
                    Permute(idx + 1);
                    Swap(ref perm[idx], ref perm[i]);
                }
            }
        }
        return Enumerable.Range(0, n).ToArray();
    }

    static int[] GreedyConstruction(Station[] stations, double[,] cost)
    {
        int n = stations.Length;
        var used = new bool[n];
        var order = new List<int>(n);
        int cur = ArgMinFromStart(cost, n);
        order.Add(cur); used[cur] = true;
        for (int step = 1; step < n; step++)
        {
            int best = -1;
            double bestC = double.PositiveInfinity;
            for (int j = 0; j < n; j++) if (!used[j])
            {
                double c = cost[cur, j] + 0.05 * stations[j].Alt + 0.02 * stations[j].IncHint;
                if (c < bestC) { bestC = c; best = j; }
            }
            order.Add(best);
            used[best] = true;
            cur = best;
        }

        // Diversify with insertion on a few candidate seeds and keep the best.
        var bestOrder = order.ToArray();
        double bestCost = PathCost(bestOrder, cost);
        for (int seed = 0; seed < Math.Min(8, n); seed++)
        {
            var trial = ConstructSeeded(cost, n, seed);
            double c = PathCost(trial, cost);
            if (c < bestCost)
            {
                bestCost = c;
                bestOrder = trial;
            }
        }
        return bestOrder;
    }

    static int[] ConstructSeeded(double[,] cost, int n, int seed)
    {
        var used = new bool[n];
        var order = new List<int>(n);
        int cur = (seed * 9973 + 17) % n;
        order.Add(cur); used[cur] = true;
        while (order.Count < n)
        {
            int best = -1;
            double bestC = double.PositiveInfinity;
            for (int j = 0; j < n; j++) if (!used[j])
            {
                double c = cost[cur, j];
                if (c < bestC) { bestC = c; best = j; }
            }
            order.Add(best); used[best] = true; cur = best;
        }
        return order.ToArray();
    }

    static void Improve2Opt(int[] order, double[,] cost, int iterations)
    {
        int n = order.Length;
        bool improved = true;
        while (improved && iterations-- > 0)
        {
            improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                for (int k = i + 1; k < n; k++)
                {
                    double delta = Delta2Opt(order, cost, i, k);
                    if (delta < -1e-9)
                    {
                        Array.Reverse(order, i, k - i + 1);
                        improved = true;
                    }
                }
            }
        }
    }

    static void ImproveOrOpt(int[] order, double[,] cost, int passes)
    {
        int n = order.Length;
        while (passes-- > 0)
        {
            bool any = false;
            for (int len = 1; len <= Math.Min(3, n); len++)
            {
                for (int i = 0; i + len <= n; i++)
                {
                    for (int j = 0; j <= n - len; j++)
                    {
                        if (j >= i && j <= i + len) continue;
                        double delta = DeltaRelocate(order, cost, i, len, j);
                        if (delta < -1e-9)
                        {
                            MoveSegment(order, i, len, j);
                            any = true;
                        }
                    }
                }
            }
            if (!any) break;
        }
    }

    static void ImproveLight(int[] order, double[,] cost, int attempts)
    {
        int n = order.Length;
        var rand = new Random(1);
        for (int t = 0; t < attempts; t++)
        {
            int i = rand.Next(n - 1);
            int k = rand.Next(i + 1, n);
            if (Delta2Opt(order, cost, i, k) < 0)
                Array.Reverse(order, i, k - i + 1);
        }
    }

    static double[,] BuildCostMatrix(State start, Station[] stations, double mu)
    {
        int n = stations.Length;
        var cost = new double[n + 1, n];
        for (int j = 0; j < n; j++)
            cost[n, j] = TransferCost(start, stations[j].S, mu);
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                cost[i, j] = i == j ? 0.0 : TransferCost(stations[i].S, stations[j].S, mu);
        return cost;
    }

    static double TransferCost(State a, State b, double mu)
    {
        var dr = b.R - a.R;
        var dv = b.V - a.V;
        double direct = dv.Norm();
        double r1 = a.R.Norm();
        double r2 = b.R.Norm();
        double radial = Math.Abs(r2 - r1) * Math.Sqrt(mu / Math.Max(r1 * r2, 1.0));
        double ang = AngleBetween(a.R, b.R);
        double plane = Math.Sin(ang * 0.5) * (a.V.Norm() + b.V.Norm()) * 0.5;
        double align = Math.Abs(Unit(a.R).Dot(Unit(b.R)));
        double geom = dr.Norm() * 0.01 + plane * 0.02 + (1.0 - align) * 25.0;
        return direct + 0.15 * radial + geom;
    }

    static double PathCost(int[] order, double[,] cost)
    {
        int n = order.Length;
        if (n == 0) return 0;
        int startIdx = cost.GetLength(0) - 1;
        double sum = cost[startIdx, order[0]];
        for (int i = 0; i + 1 < n; i++) sum += cost[order[i], order[i + 1]];
        return sum;
    }

    static double Delta2Opt(int[] order, double[,] cost, int i, int k)
    {
        int n = order.Length;
        int a = i == 0 ? -1 : order[i - 1];
        int b = order[i];
        int c = order[k];
        int d = k + 1 < n ? order[k + 1] : -1;
        double before = (a == -1 ? cost[n, b] : cost[a, b]) + (d == -1 ? 0 : cost[c, d]);
        double after = (a == -1 ? cost[n, c] : cost[a, c]) + (d == -1 ? 0 : cost[b, d]);
        return after - before;
    }

    static double DeltaRelocate(int[] order, double[,] cost, int i, int len, int j)
    {
        int n = order.Length;
        var trial = (int[])order.Clone();
        MoveSegment(trial, i, len, j);
        return LocalCost(trial, cost) - LocalCost(order, cost);
    }

    static void MoveSegment(int[] order, int i, int len, int j)
    {
        if (j > i) j -= len;
        var seg = new int[len];
        Array.Copy(order, i, seg, 0, len);
        if (j < i) Array.Copy(order, j, order, j + len, i - j);
        else Array.Copy(order, i + len, order, i, j - i);
        Array.Copy(seg, 0, order, j, len);
    }

    static double LocalCost(int[] order, double[,] cost)
    {
        int n = order.Length;
        if (n == 0) return 0;
        double sum = cost[cost.GetLength(0) - 1, order[0]];
        for (int i = 0; i + 1 < n; i++) sum += cost[order[i], order[i + 1]];
        return sum;
    }

    static double EdgeCost(double[,] cost, int a, int b, int n)
    {
        if (a == -1 && b == -1) return 0;
        if (a == -1) return cost[n, b];
        if (b == -1) return 0;
        return cost[a, b];
    }

    static int ArgMinFromStart(double[,] cost, int n)
    {
        int best = 0; double bestC = double.PositiveInfinity;
        for (int j = 0; j < n; j++)
        {
            double c = cost[n, j];
            if (c < bestC) { bestC = c; best = j; }
        }
        return best;
    }

    static double AngleBetween(Vec3 a, Vec3 b)
    {
        double denom = a.Norm() * b.Norm() + Eps;
        return Math.Acos(Clamp(a.Dot(b) / denom, -1.0, 1.0));
    }

    static Vec3 Cross(Vec3 a, Vec3 b) => new Vec3(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
    static Vec3 Unit(Vec3 a) { double n = a.Norm(); return n < Eps ? new Vec3(0, 0, 0) : a / n; }
    static double Clamp(double x, double lo, double hi) => x < lo ? lo : (x > hi ? hi : x);
    static void Swap(ref int a, ref int b) { int t = a; a = b; b = t; }
}
