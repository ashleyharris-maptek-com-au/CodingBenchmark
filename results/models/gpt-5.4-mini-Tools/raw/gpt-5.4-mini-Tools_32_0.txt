#include <algorithm>
#include <iostream>
#include <limits>
#include <queue>
#include <set>
#include <utility>
#include <vector>

using namespace std;

struct Edge {
    int to;
    long long w;
    int id;
};

struct State {
    long long dist;
    int v;
    int mask;
    bool operator>(const State& other) const { return dist > other.dist; }
};

static const long long INF = (1LL << 60);

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<vector<Edge>> g(n);
    vector<int> eu(m), ev(m);
    vector<long long> ew(m);
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        eu[i] = u;
        ev[i] = v;
        ew[i] = w;
        g[u].push_back({v, w, i});
        g[v].push_back({u, w, i});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) cin >> terminals[i];

    if (k == 0) {
        cout << 0 << ' ' << 0 << '\n';
        return 0;
    }
    if (k == 1) {
        cout << 0 << ' ' << 0 << '\n';
        return 0;
    }

    auto dijkstra = [&](int src) {
        vector<long long> dist(n, INF);
        vector<int> parentV(n, -1), parentE(n, -1);
        priority_queue<pair<long long,int>, vector<pair<long long,int>>, greater<pair<long long,int>>> pq;
        dist[src] = 0;
        pq.push({0, src});
        while (!pq.empty()) {
            auto [d, v] = pq.top(); pq.pop();
            if (d != dist[v]) continue;
            for (const auto& e : g[v]) {
                long long nd = d + e.w;
                if (nd < dist[e.to]) {
                    dist[e.to] = nd;
                    parentV[e.to] = v;
                    parentE[e.to] = e.id;
                    pq.push({nd, e.to});
                }
            }
        }
        return tuple<vector<long long>, vector<int>, vector<int>>(move(dist), move(parentV), move(parentE));
    };

    vector<int> chosenEdges;
    auto add_path = [&](int src, int dst) {
        auto [dist, pv, pe] = dijkstra(src);
        if (dist[dst] >= INF / 2) return false;
        int cur = dst;
        while (cur != src) {
            int id = pe[cur];
            if (id < 0) return false;
            chosenEdges.push_back(id);
            cur = pv[cur];
        }
        return true;
    };

    // Exact DP for small terminal sets, with path reconstruction via heuristic backbone.
    // For larger cases, connect terminals through a shortest-path tree rooted at the first terminal.
    if (k <= 12) {
        int full = 1 << k;
        vector<vector<long long>> dp(full, vector<long long>(n, INF));
        vector<vector<int>> parentMask(full, vector<int>(n, -1));
        vector<vector<int>> parentVertex(full, vector<int>(n, -1));
        vector<vector<int>> parentEdge(full, vector<int>(n, -1));

        for (int i = 0; i < k; ++i) dp[1 << i][terminals[i]] = 0;

        for (int mask = 1; mask < full; ++mask) {
            if ((mask & (mask - 1)) != 0) {
                for (int sub = (mask - 1) & mask; sub; sub = (sub - 1) & mask) {
                    int other = mask ^ sub;
                    if (!other) continue;
                    for (int v = 0; v < n; ++v) {
                        long long cand = dp[sub][v] + dp[other][v];
                        if (cand < dp[mask][v]) {
                            dp[mask][v] = cand;
                            parentMask[mask][v] = sub;
                            parentVertex[mask][v] = v;
                        }
                    }
                }
            }

            priority_queue<pair<long long,int>, vector<pair<long long,int>>, greater<pair<long long,int>>> pq;
            vector<char> vis(n, 0);
            for (int v = 0; v < n; ++v) {
                if (dp[mask][v] < INF) pq.push({dp[mask][v], v});
            }
            while (!pq.empty()) {
                auto [d, v] = pq.top(); pq.pop();
                if (vis[v] || d != dp[mask][v]) continue;
                vis[v] = 1;
                for (const auto& e : g[v]) {
                    long long nd = d + e.w;
                    if (nd < dp[mask][e.to]) {
                        dp[mask][e.to] = nd;
                        parentMask[mask][e.to] = -2;
                        parentVertex[mask][e.to] = v;
                        parentEdge[mask][e.to] = e.id;
                        pq.push({nd, e.to});
                    }
                }
            }
        }

        int bestV = 0;
        for (int v = 1; v < n; ++v) if (dp[full - 1][v] < dp[full - 1][bestV]) bestV = v;

        // Reconstruct approximately by building shortest paths from terminals to bestV and unioning them.
        set<int> edgeSet;
        for (int t : terminals) {
            auto [dist, pv, pe] = dijkstra(t);
            if (dist[bestV] >= INF / 2) continue;
            int cur = bestV;
            while (cur != t) {
                int id = pe[cur];
                if (id < 0) break;
                edgeSet.insert(id);
                cur = pv[cur];
            }
        }
        long long total = 0;
        for (int id : edgeSet) total += ew[id];
        cout << total << ' ' << edgeSet.size() << '\n';
        for (int id : edgeSet) cout << eu[id] << ' ' << ev[id] << '\n';
        return 0;
    }

    // Heuristic fallback: connect terminals to a shared root by shortest paths, then prune cycles with DSU.
    int root = terminals[0];
    set<int> edgeSet;
    for (int i = 1; i < k; ++i) {
        auto [dist, pv, pe] = dijkstra(terminals[i]);
        if (dist[root] >= INF / 2) continue;
        int cur = root;
        while (cur != terminals[i]) {
            int id = pe[cur];
            if (id < 0) break;
            edgeSet.insert(id);
            cur = pv[cur];
        }
    }

    vector<int> ids(edgeSet.begin(), edgeSet.end());
    sort(ids.begin(), ids.end(), [&](int a, int b) { return ew[a] < ew[b]; });
    vector<int> p(n), r(n, 0);
    for (int i = 0; i < n; ++i) p[i] = i;
    auto findp = [&](auto&& self, int x) -> int {
        if (p[x] == x) return x;
        return p[x] = self(self, p[x]);
    };
    vector<int> finalEdges;
    for (int id : ids) {
        int a = findp(findp, eu[id]);
        int b = findp(findp, ev[id]);
        if (a == b) continue;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) ++r[a];
        finalEdges.push_back(id);
    }

    long long total = 0;
    for (int id : finalEdges) total += ew[id];
    cout << total << ' ' << finalEdges.size() << '\n';
    for (int id : finalEdges) cout << eu[id] << ' ' << ev[id] << '\n';
    return 0;
}
