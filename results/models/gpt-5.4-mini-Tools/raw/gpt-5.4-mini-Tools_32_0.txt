#include <algorithm>
#include <iostream>
#include <limits>
#include <queue>
#include <set>
#include <utility>
#include <vector>

using namespace std;

struct Edge {
    int to;
    long long w;
    int id;
};

struct ParentInfo {
    int prev = -1;
    int edge_id = -1;
};

struct DijkstraResult {
    vector<long long> dist;
    vector<ParentInfo> parent;
};

static const long long INF = numeric_limits<long long>::max() / 4;

DijkstraResult dijkstra_from_source(int n, const vector<vector<Edge>>& adj, int src) {
    vector<long long> dist(n, INF);
    vector<ParentInfo> parent(n);
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    dist[src] = 0;
    pq.push({0, src});
    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (d != dist[u]) continue;
        for (const auto& e : adj[u]) {
            long long nd = d + e.w;
            if (nd < dist[e.to]) {
                dist[e.to] = nd;
                parent[e.to] = {u, e.id};
                pq.push({nd, e.to});
            }
        }
    }
    return {move(dist), move(parent)};
}

vector<int> restore_path_edges(int src, int dst, const vector<ParentInfo>& parent) {
    vector<int> edges;
    int cur = dst;
    while (cur != src) {
        if (cur < 0 || parent[cur].prev < 0) break;
        edges.push_back(parent[cur].edge_id);
        cur = parent[cur].prev;
    }
    reverse(edges.begin(), edges.end());
    return edges;
}

struct DSU {
    vector<int> p, r;
    DSU(int n = 0) { init(n); }
    void init(int n) {
        p.resize(n);
        r.assign(n, 0);
        for (int i = 0; i < n; ++i) p[i] = i;
    }
    int find(int x) { return p[x] == x ? x : p[x] = find(p[x]); }
    bool unite(int a, int b) {
        a = find(a); b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) ++r[a];
        return true;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<int> U(m), V(m);
    vector<long long> W(m);
    vector<vector<Edge>> adj(n);
    for (int i = 0; i < m; ++i) {
        cin >> U[i] >> V[i] >> W[i];
        adj[U[i]].push_back({V[i], W[i], i});
        adj[V[i]].push_back({U[i], W[i], i});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) cin >> terminals[i];
    if (k == 0) {
        cout << 0 << " 0\n";
        return 0;
    }
    if (k == 1) {
        cout << 0 << " 0\n";
        return 0;
    }

    // Build a candidate Steiner tree by using metric-closure MST over terminals
    // and expanding shortest paths in the original graph.
    vector<DijkstraResult> all;
    all.reserve(k);
    for (int t : terminals) all.push_back(dijkstra_from_source(n, adj, t));

    struct TEdge { long long w; int a, b; };
    vector<TEdge> complete;
    for (int i = 0; i < k; ++i) {
        for (int j = i + 1; j < k; ++j) {
            complete.push_back({all[i].dist[terminals[j]], i, j});
        }
    }
    sort(complete.begin(), complete.end(), [](const TEdge& x, const TEdge& y) {
        if (x.w != y.w) return x.w < y.w;
        if (x.a != y.a) return x.a < y.a;
        return x.b < y.b;
    });

    DSU dsu(k);
    vector<pair<int, int>> chosen_pairs;
    for (const auto& e : complete) {
        if (dsu.unite(e.a, e.b)) chosen_pairs.push_back({e.a, e.b});
    }

    set<int> used_edges;
    for (auto [a, b] : chosen_pairs) {
        int src = terminals[a];
        int dst = terminals[b];
        const auto& parent = all[a].parent;
        vector<int> path = restore_path_edges(src, dst, parent);
        for (int id : path) used_edges.insert(id);
    }

    // Remove cycles by taking an MST of the selected edges' induced subgraph.
    vector<int> edge_list(used_edges.begin(), used_edges.end());
    sort(edge_list.begin(), edge_list.end(), [&](int e1, int e2) {
        if (W[e1] != W[e2]) return W[e1] < W[e2];
        return e1 < e2;
    });

    DSU tree_dsu(n);
    vector<int> answer_edges;
    long long total = 0;
    for (int id : edge_list) {
        if (tree_dsu.unite(U[id], V[id])) {
            answer_edges.push_back(id);
            total += W[id];
        }
    }

    // As a fallback, if the above produced disconnected output due to unreachable pairs,
    // connect terminals via a simple union of shortest paths from the first terminal.
    bool ok = true;
    int root = terminals[0];
    for (int t : terminals) {
        if (all[0].dist[t] >= INF / 2) {
            ok = false;
            break;
        }
    }
    if (!ok) {
        cout << 0 << " 0\n";
        return 0;
    }

    // Verify connectivity of terminals in output; if some terminal is missing, build a star.
    vector<int> comp(n, -1);
    DSU check(n);
    for (int id : answer_edges) check.unite(U[id], V[id]);
    int rep = check.find(root);
    for (int t : terminals) {
        if (check.find(t) != rep) {
            answer_edges.clear();
            total = 0;
            set<int> star_edges;
            for (int t2 : terminals) {
                auto path = restore_path_edges(root, t2, all[0].parent);
                for (int id : path) star_edges.insert(id);
            }
            tree_dsu.init(n);
            for (int id : star_edges) {
                if (tree_dsu.unite(U[id], V[id])) {
                    answer_edges.push_back(id);
                    total += W[id];
                }
            }
            break;
        }
    }

    cout << total << ' ' << answer_edges.size() << '\n';
    for (int id : answer_edges) {
        cout << U[id] << ' ' << V[id] << '\n';
    }
    return 0;
}
