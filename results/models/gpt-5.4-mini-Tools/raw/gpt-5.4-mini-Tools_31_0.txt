use std::collections::HashMap;
use std::io::{self, Read, Write};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(v) => v.parse().unwrap_or(0),
        None => return,
    };
    let target: i64 = match it.next() {
        Some(v) => v.parse().unwrap_or(0),
        None => return,
    };

    let mut values = Vec::with_capacity(n);
    for _ in 0..n {
        if let Some(v) = it.next() {
            values.push(v.parse::<i64>().unwrap_or(0));
        }
    }

    let mut out = io::BufWriter::new(io::stdout());

    if target == 0 {
        writeln!(out, "YES").unwrap();
        writeln!(out).unwrap();
        return;
    }
    if values.is_empty() {
        writeln!(out, "NO").unwrap();
        return;
    }

    if values.len() <= 44 {
        if let Some(ans) = meet_in_middle(&values, target) {
            writeln!(out, "YES").unwrap();
            write_indices(&mut out, &ans);
            return;
        }
    }

    if values.iter().all(|&x| x >= 0) && target >= 0 {
        if let Some(ans) = subset_sum_nonnegative(&values, target as usize) {
            writeln!(out, "YES").unwrap();
            write_indices(&mut out, &ans);
            return;
        }
    }

    if let Some(ans) = sparse_dp(&values, target) {
        writeln!(out, "YES").unwrap();
        write_indices(&mut out, &ans);
        return;
    }

    writeln!(out, "NO").unwrap();
}

fn write_indices(out: &mut impl Write, indices: &[usize]) {
    if indices.is_empty() {
        writeln!(out).unwrap();
        return;
    }
    for (i, idx) in indices.iter().enumerate() {
        if i > 0 {
            write!(out, " ").unwrap();
        }
        write!(out, "{}", idx).unwrap();
    }
    writeln!(out).unwrap();
}

fn meet_in_middle(values: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = values.len();
    let mid = n / 2;
    let (left, right) = values.split_at(mid);
    if left.len() > 24 || right.len() > 24 {
        return None;
    }

    let mut left_sums = Vec::with_capacity(1usize << left.len());
    for mask in 0u64..(1u64 << left.len()) {
        let mut sum = 0i64;
        for i in 0..left.len() {
            if ((mask >> i) & 1) != 0 {
                sum = sum.wrapping_add(left[i]);
            }
        }
        left_sums.push((sum, mask));
    }
    left_sums.sort_unstable_by_key(|x| x.0);

    for mask in 0u64..(1u64 << right.len()) {
        let mut sum = 0i64;
        for i in 0..right.len() {
            if ((mask >> i) & 1) != 0 {
                sum = sum.wrapping_add(right[i]);
            }
        }
        let need = target.wrapping_sub(sum);
        if let Ok(pos) = left_sums.binary_search_by_key(&need, |x| x.0) {
            let lm = left_sums[pos].1;
            let mut ans = Vec::new();
            for i in 0..left.len() {
                if ((lm >> i) & 1) != 0 {
                    ans.push(i);
                }
            }
            for i in 0..right.len() {
                if ((mask >> i) & 1) != 0 {
                    ans.push(mid + i);
                }
            }
            return Some(ans);
        }
    }
    None
}

fn subset_sum_nonnegative(values: &[i64], target: usize) -> Option<Vec<usize>> {
    if target > 2_000_000 {
        return None;
    }
    let mut prev_sum = vec![usize::MAX; target + 1];
    let mut prev_idx = vec![usize::MAX; target + 1];
    let mut reachable = vec![false; target + 1];
    reachable[0] = true;

    for (idx, &v) in values.iter().enumerate() {
        let x = v as usize;
        if x > target {
            continue;
        }
        for s in (x..=target).rev() {
            if !reachable[s] && reachable[s - x] {
                reachable[s] = true;
                prev_sum[s] = s - x;
                prev_idx[s] = idx;
            }
        }
        if reachable[target] {
            break;
        }
    }

    if !reachable[target] {
        return None;
    }

    let mut res = Vec::new();
    let mut cur = target;
    while cur != 0 {
        let idx = prev_idx[cur];
        if idx == usize::MAX {
            return None;
        }
        res.push(idx);
        cur = prev_sum[cur];
    }
    res.reverse();
    Some(res)
}

fn sparse_dp(values: &[i64], target: i64) -> Option<Vec<usize>> {
    let mut parent: HashMap<i64, (i64, usize)> = HashMap::new();
    parent.insert(0, (0, usize::MAX));
    let mut frontier = vec![0i64];
    let max_states = 200_000usize;

    for (idx, &v) in values.iter().enumerate() {
        let mut next = Vec::new();
        for &sum in &frontier {
            let ns = sum.wrapping_add(v);
            if !parent.contains_key(&ns) {
                parent.insert(ns, (sum, idx));
                next.push(ns);
            }
        }
        if parent.contains_key(&target) {
            break;
        }
        frontier.extend(next);
        if parent.len() > max_states {
            break;
        }
    }

    if !parent.contains_key(&target) {
        return None;
    }

    let mut res = Vec::new();
    let mut cur = target;
    while cur != 0 {
        let (prev, idx) = *parent.get(&cur)?;
        if idx == usize::MAX {
            break;
        }
        res.push(idx);
        cur = prev;
    }
    res.reverse();
    Some(res)
}
