const fs = require('fs');
const input = fs.readFileSync(0, 'utf8').trim();
if (!input) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const data = JSON.parse(input);
const width = data.width | 0;
const height = data.height | 0;
const grid = data.grid;
const cutBudget = data.cutBudget | 0;
const ignitions = data.ignitions || [];
const towns = data.towns || [];

const n = width * height;
const blocked = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row = grid[y];
  for (let x = 0; x < width; x++) {
    if (row.charCodeAt(x) !== 46) blocked[y * width + x] = 1;
  }
}

const burn = new Int8Array(n);
const q = new Int32Array(n);
let qh = 0, qt = 0;
for (let i = 0; i < ignitions.length; i++) {
  const x = ignitions[i][0] | 0;
  const y = ignitions[i][1] | 0;
  if (x < 0 || x >= width || y < 0 || y >= height) continue;
  const idx = y * width + x;
  if (blocked[idx] || burn[idx]) continue;
  burn[idx] = 1;
  q[qt++] = idx;
}

const neigh = [1, -1, width, -width];
while (qh < qt) {
  const idx = q[qh++];
  const x = idx % width;
  const y = (idx / width) | 0;
  if (x + 1 < width) {
    const ni = idx + 1;
    if (!blocked[ni] && !burn[ni]) { burn[ni] = 1; q[qt++] = ni; }
  }
  if (x > 0) {
    const ni = idx - 1;
    if (!blocked[ni] && !burn[ni]) { burn[ni] = 1; q[qt++] = ni; }
  }
  if (y + 1 < height) {
    const ni = idx + width;
    if (!blocked[ni] && !burn[ni]) { burn[ni] = 1; q[qt++] = ni; }
  }
  if (y > 0) {
    const ni = idx - width;
    if (!blocked[ni] && !burn[ni]) { burn[ni] = 1; q[qt++] = ni; }
  }
}

const townValue = new Float64Array(n);
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  const x = t.x | 0, y = t.y | 0;
  if (x < 0 || x >= width || y < 0 || y >= height) continue;
  const idx = y * width + x;
  townValue[idx] += +t.value || 0;
}

const best = new Map();
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  const tx = t.x | 0, ty = t.y | 0;
  if (tx < 0 || tx >= width || ty < 0 || ty >= height) continue;
  const start = ty * width + tx;
  if (!burn[start] || blocked[start]) continue;

  const seen = new Int32Array(n);
  let mark = i + 1;
  const qq = new Int32Array(n);
  let sh = 0, st = 0;
  seen[start] = mark;
  qq[st++] = start;
  let found = -1;
  while (sh < st) {
    const idx = qq[sh++];
    const x = idx % width;
    const y = (idx / width) | 0;
    if (x + 1 < width) {
      const ni = idx + 1;
      if (burn[ni]) { found = ni; break; }
      if (!blocked[ni] && seen[ni] !== mark) { seen[ni] = mark; qq[st++] = ni; }
    }
    if (x > 0) {
      const ni = idx - 1;
      if (burn[ni]) { found = ni; break; }
      if (!blocked[ni] && seen[ni] !== mark) { seen[ni] = mark; qq[st++] = ni; }
    }
    if (y + 1 < height) {
      const ni = idx + width;
      if (burn[ni]) { found = ni; break; }
      if (!blocked[ni] && seen[ni] !== mark) { seen[ni] = mark; qq[st++] = ni; }
    }
    if (y > 0) {
      const ni = idx - width;
      if (burn[ni]) { found = ni; break; }
      if (!blocked[ni] && seen[ni] !== mark) { seen[ni] = mark; qq[st++] = ni; }
    }
  }
  if (found >= 0) {
    const x = found % width;
    const y = (found / width) | 0;
    const key = y * width + x;
    best.set(key, (best.get(key) || 0) + (+t.value || 0));
  }
}

const candidates = [];
for (const [key, score] of best.entries()) {
  candidates.push([score, key]);
}
candidates.sort((a, b) => b[0] - a[0]);

const firebreaks = [];
const used = new Uint8Array(n);
for (let i = 0; i < candidates.length && firebreaks.length < cutBudget; i++) {
  const key = candidates[i][1];
  if (used[key] || blocked[key] || !burn[key]) continue;
  used[key] = 1;
  firebreaks.push([key % width, (key / width) | 0]);
}

process.stdout.write(JSON.stringify({ firebreaks }));
