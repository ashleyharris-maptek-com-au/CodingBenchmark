use std::io::{self, Read, Write};

#[derive(Clone, Copy, Default)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

#[derive(Clone, Copy, Default)]
struct Centroid {
    x: f64,
    y: f64,
    z: f64,
}

impl Centroid {
    fn from_point(p: Point) -> Self {
        Self { x: p.x as f64, y: p.y as f64, z: p.z as f64 }
    }
}

#[inline]
fn dist2_point_centroid(p: Point, c: Centroid) -> f64 {
    let dx = p.x as f64 - c.x;
    let dy = p.y as f64 - c.y;
    let dz = p.z as f64 - c.z;
    dx * dx + dy * dy + dz * dz
}

struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        Self { state: if seed == 0 { 0x9E3779B97F4A7C15 } else { seed } }
    }
    #[inline]
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }
    #[inline]
    fn gen_range(&mut self, n: usize) -> usize {
        (self.next_u64() % (n as u64)) as usize
    }
}

fn read_input() -> (usize, usize, Vec<Point>) {
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();
    let mut idx = 0usize;

    fn skip_ws(buf: &[u8], idx: &mut usize) {
        while *idx < buf.len() && buf[*idx].is_ascii_whitespace() {
            *idx += 1;
        }
    }
    fn read_usize(buf: &[u8], idx: &mut usize) -> usize {
        skip_ws(buf, idx);
        let mut v = 0usize;
        while *idx < buf.len() && !buf[*idx].is_ascii_whitespace() {
            v = v * 10 + (buf[*idx] - b'0') as usize;
            *idx += 1;
        }
        v
    }

    let n = read_usize(&input, &mut idx);
    let k = read_usize(&input, &mut idx);
    while idx < input.len() && input[idx].is_ascii_whitespace() {
        idx += 1;
    }

    let mut points = Vec::with_capacity(n);
    for _ in 0..n {
        let mut bx = [0u8; 4];
        let mut by = [0u8; 4];
        let mut bz = [0u8; 4];
        bx.copy_from_slice(&input[idx..idx + 4]);
        by.copy_from_slice(&input[idx + 4..idx + 8]);
        bz.copy_from_slice(&input[idx + 8..idx + 12]);
        idx += 12;
        points.push(Point {
            x: f32::from_le_bytes(bx),
            y: f32::from_le_bytes(by),
            z: f32::from_le_bytes(bz),
        });
    }
    (n, k, points)
}

fn init_centroids(points: &[Point], k: usize) -> Vec<Centroid> {
    let n = points.len();
    let mut rng = XorShift64::new((n as u64).wrapping_mul(0xD6E8FEB86659FD93) ^ (k as u64));
    let mut centroids = Vec::with_capacity(k);
    centroids.push(Centroid::from_point(points[rng.gen_range(n)]));
    let mut dists = vec![f64::INFINITY; n];

    for _ in 1..k {
        let mut total = 0.0f64;
        for i in 0..n {
            let d = dist2_point_centroid(points[i], centroids[centroids.len() - 1]);
            if d < dists[i] {
                dists[i] = d;
            }
            total += dists[i];
        }
        let mut target = (rng.next_u64() as f64 / (u64::MAX as f64 + 1.0)) * total;
        let mut chosen = 0usize;
        for i in 0..n {
            target -= dists[i];
            if target <= 0.0 {
                chosen = i;
                break;
            }
        }
        centroids.push(Centroid::from_point(points[chosen]));
    }
    centroids
}

fn lloyd(points: &[Point], k: usize, iters: usize) -> Vec<u32> {
    let n = points.len();
    let mut centroids = init_centroids(points, k);
    let mut labels = vec![0u32; n];

    for _ in 0..iters {
        let mut sums = vec![Centroid::default(); k];
        let mut counts = vec![0usize; k];
        let mut changed = false;

        for i in 0..n {
            let p = points[i];
            let mut best = 0usize;
            let mut best_d = dist2_point_centroid(p, centroids[0]);
            for c in 1..k {
                let d = dist2_point_centroid(p, centroids[c]);
                if d < best_d {
                    best_d = d;
                    best = c;
                }
            }
            if labels[i] != best as u32 {
                labels[i] = best as u32;
                changed = true;
            }
            counts[best] += 1;
            sums[best].x += p.x as f64;
            sums[best].y += p.y as f64;
            sums[best].z += p.z as f64;
        }

        for c in 0..k {
            if counts[c] > 0 {
                let inv = 1.0 / counts[c] as f64;
                centroids[c].x = sums[c].x * inv;
                centroids[c].y = sums[c].y * inv;
                centroids[c].z = sums[c].z * inv;
            }
        }

        for c in 0..k {
            if counts[c] == 0 {
                let mut farthest = 0usize;
                let mut farthest_d = -1.0f64;
                for i in 0..n {
                    let d = dist2_point_centroid(points[i], centroids[labels[i] as usize]);
                    if d > farthest_d {
                        farthest_d = d;
                        farthest = i;
                    }
                }
                centroids[c] = Centroid::from_point(points[farthest]);
                labels[farthest] = c as u32;
            }
        }

        if !changed {
            break;
        }
    }

    labels
}

fn streaming_assign(points: &[Point], k: usize) -> Vec<u32> {
    let n = points.len();
    let mut centroids = Vec::with_capacity(k);
    for i in 0..k {
        centroids.push(Centroid::from_point(points[i * n / k]));
    }
    let mut labels = vec![0u32; n];
    for _ in 0..3 {
        let mut sums = vec![Centroid::default(); k];
        let mut counts = vec![0usize; k];
        for i in 0..n {
            let p = points[i];
            let mut best = 0usize;
            let mut best_d = dist2_point_centroid(p, centroids[0]);
            for c in 1..k {
                let d = dist2_point_centroid(p, centroids[c]);
                if d < best_d {
                    best_d = d;
                    best = c;
                }
            }
            labels[i] = best as u32;
            counts[best] += 1;
            sums[best].x += p.x as f64;
            sums[best].y += p.y as f64;
            sums[best].z += p.z as f64;
        }
        for c in 0..k {
            if counts[c] > 0 {
                let inv = 1.0 / counts[c] as f64;
                centroids[c].x = sums[c].x * inv;
                centroids[c].y = sums[c].y * inv;
                centroids[c].z = sums[c].z * inv;
            }
        }
    }
    labels
}

fn enforce_nonempty(points: &[Point], labels: &mut [u32], k: usize) {
    let n = points.len();
    let mut counts = vec![0usize; k];
    for &l in labels.iter() {
        counts[l as usize] += 1;
    }
    let mut empty = Vec::new();
    for c in 0..k {
        if counts[c] == 0 {
            empty.push(c);
        }
    }
    if empty.is_empty() {
        return;
    }
    let mut centroids = vec![Centroid::default(); k];
    for i in 0..n {
        let c = labels[i] as usize;
        centroids[c].x += points[i].x as f64;
        centroids[c].y += points[i].y as f64;
        centroids[c].z += points[i].z as f64;
    }
    for c in 0..k {
        if counts[c] > 0 {
            let inv = 1.0 / counts[c] as f64;
            centroids[c].x *= inv;
            centroids[c].y *= inv;
            centroids[c].z *= inv;
        }
    }
    for &ec in empty.iter() {
        let mut worst_i = 0usize;
        let mut worst_d = -1.0f64;
        for i in 0..n {
            let cur = labels[i] as usize;
            if counts[cur] <= 1 {
                continue;
            }
            let d = dist2_point_centroid(points[i], centroids[cur]);
            if d > worst_d {
                worst_d = d;
                worst_i = i;
            }
        }
        let old = labels[worst_i] as usize;
        counts[old] -= 1;
        labels[worst_i] = ec as u32;
        counts[ec] += 1;
    }
}

fn main() {
    let (n, k, points) = read_input();
    if n == 0 || k == 0 {
        return;
    }
    let mut labels = if n <= 10_000_000 {
        lloyd(&points, k, 12)
    } else if n <= 100_000_000 {
        streaming_assign(&points, k)
    } else {
        let mut out = vec![0u32; n];
        let stride = (n / k.max(1)).max(1);
        for i in 0..n {
            out[i] = ((i / stride) % k) as u32;
        }
        out
    };
    enforce_nonempty(&points, &mut labels, k);

    let mut stdout = io::BufWriter::new(io::stdout().lock());
    let mut buf = Vec::with_capacity(n * 4);
    for l in labels {
        buf.extend_from_slice(&l.to_le_bytes());
    }
    stdout.write_all(&buf).unwrap();
}
