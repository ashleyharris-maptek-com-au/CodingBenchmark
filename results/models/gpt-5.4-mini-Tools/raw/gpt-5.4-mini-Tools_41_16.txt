; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %uv %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %uv "uv"
OpName %outColor "outColor"
OpDecorate %uv Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v4float = OpTypeVector %float 4
%ptr_input_v2float = OpTypePointer Input %v2float
%ptr_output_v4float = OpTypePointer Output %v4float

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_20 = OpConstant %int 20
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0
%float_3 = OpConstant %float 3.0
%float_4 = OpConstant %float 4.0
%float_05 = OpConstant %float 0.5
%float_m2 = OpConstant %float -2.0
%float_m1 = OpConstant %float -1.0

%uv = OpVariable %ptr_input_v2float Input
%outColor = OpVariable %ptr_output_v4float Output

%main = OpFunction %void None %func
%entry = OpLabel
%uv_val = OpLoad %v2float %uv
%uv_x = OpCompositeExtract %float %uv_val 0
%uv_y = OpCompositeExtract %float %uv_val 1
%cx_scale = OpFMul %float %uv_x %float_3
%cx = OpFAdd %float %cx_scale %float_m2
%cy_scale = OpFMul %float %uv_y %float_2
%cy = OpFAdd %float %cy_scale %float_m1
%c = OpCompositeConstruct %v2float %cx %cy
OpBranch %loop

%loop = OpLabel
%iter = OpPhi %int %int_0 %entry %iter_next %body
%z = OpPhi %v2float %c %entry %z_next %body
%zx = OpCompositeExtract %float %z 0
%zy = OpCompositeExtract %float %z 1
%zx2 = OpFMul %float %zx %zx
%zy2 = OpFMul %float %zy %zy
%mag2 = OpFAdd %float %zx2 %zy2
%escaped = OpFOrdGreaterThan %bool %mag2 %float_4
%done = OpIEqual %bool %iter %int_20
%stop = OpLogicalOr %bool %escaped %done
OpSelectionMerge %after_loop None
OpBranchConditional %stop %after_loop %body

%body = OpLabel
%zxzy = OpFMul %float %zx %zy
%two_zxzy = OpFMul %float %zxzy %float_2
%zx_sq_minus_zy_sq = OpFSub %float %zx2 %zy2
%new_x = OpFAdd %float %zx_sq_minus_zy_sq %cx
%new_y = OpFAdd %float %two_zxzy %cy
%z_next = OpCompositeConstruct %v2float %new_x %new_y
%iter_next = OpIAdd %int %iter %int_1
OpBranch %loop

%after_loop = OpLabel
%iter_f = OpConvertSToF %float %iter
%t = OpFDiv %float %iter_f %float_20
%g = OpFMul %float %t %float_05
%one_minus_t = OpFSub %float %float_1 %t
%color = OpCompositeConstruct %v4float %t %g %one_minus_t %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
