using System;
using System.Collections.Generic;
using System.IO;
using System.Runtime.CompilerServices;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32 && c != -1)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }
    }

    private static int _n;
    private static ulong[][] _adjBits = Array.Empty<ulong[]>();
    private static int[] _deg = Array.Empty<int>();
    private static int[] _order = Array.Empty<int>();
    private static int[] _best = Array.Empty<int>();
    private static int _bestSize;
    private static long _deadlineTicks;

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        if (n == int.MinValue) return;
        int m = fs.NextInt();

        _n = n;
        _deg = new int[n];
        var edges = new List<(int u, int v)>(Math.Max(0, m));
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if ((uint)u >= (uint)n || (uint)v >= (uint)n || u == v) continue;
            if (u > v) (u, v) = (v, u);
            edges.Add((u, v));
            _deg[u]++;
            _deg[v]++;
        }

        int words = (n + 63) >> 6;
        _adjBits = new ulong[n][];
        for (int i = 0; i < n; i++) _adjBits[i] = new ulong[words];
        foreach (var e in edges)
        {
            _adjBits[e.u][e.v >> 6] |= 1UL << (e.v & 63);
            _adjBits[e.v][e.u >> 6] |= 1UL << (e.u & 63);
        }

        _order = new int[n];
        for (int i = 0; i < n; i++) _order[i] = i;
        Array.Sort(_order, (a, b) =>
        {
            int c = _deg[b].CompareTo(_deg[a]);
            return c != 0 ? c : a.CompareTo(b);
        });

        _best = new int[Math.Max(1, n)];
        _bestSize = 0;
        _deadlineTicks = DateTime.UtcNow.AddMinutes(4.6).Ticks;

        GreedySeed();

        var candidates = new int[n];
        Array.Copy(_order, candidates, n);
        Expand(Array.Empty<int>(), candidates, n);

        var sw = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };
        sw.WriteLine(_bestSize);
        for (int i = 0; i < _bestSize; i++)
        {
            if (i > 0) sw.Write(' ');
            sw.Write(_best[i]);
        }
        sw.WriteLine();
        sw.Flush();
    }

    private static void GreedySeed()
    {
        int tries = Math.Min(_n, 64);
        for (int start = 0; start < tries; start++)
        {
            int seed = _order[start];
            Span<int> clique = stackalloc int[Math.Min(_n, 256)];
            int size = 0;
            clique[size++] = seed;
            for (int i = 0; i < _n && size < clique.Length; i++)
            {
                int u = _order[i];
                if (u == seed) continue;
                bool ok = true;
                for (int j = 0; j < size; j++)
                {
                    if (!IsEdge(u, clique[j])) { ok = false; break; }
                }
                if (ok) clique[size++] = u;
            }
            if (size > _bestSize)
            {
                _bestSize = size;
                for (int i = 0; i < size; i++) _best[i] = clique[i];
            }
        }
    }

    private static void Expand(int[] clique, int[] candidates, int candidateCount)
    {
        if (DateTime.UtcNow.Ticks >= _deadlineTicks) return;
        if (candidateCount == 0)
        {
            if (clique.Length > _bestSize)
            {
                _bestSize = clique.Length;
                Array.Copy(clique, _best, clique.Length);
            }
            return;
        }

        int[] colors = new int[candidateCount];
        ColorSort(candidates, candidateCount, colors);

        for (int idx = candidateCount - 1; idx >= 0; idx--)
        {
            if (clique.Length + colors[idx] <= _bestSize) return;

            int v = candidates[idx];
            int[] nextCand = new int[idx];
            int nextCount = 0;
            for (int i = 0; i < idx; i++)
            {
                int u = candidates[i];
                if (IsEdge(v, u)) nextCand[nextCount++] = u;
            }

            int[] nextClique = new int[clique.Length + 1];
            Array.Copy(clique, nextClique, clique.Length);
            nextClique[clique.Length] = v;
            Expand(nextClique, nextCand, nextCount);

            if (DateTime.UtcNow.Ticks >= _deadlineTicks) return;
        }
    }

    private static void ColorSort(int[] candidates, int count, int[] colors)
    {
        Array.Copy(candidates, 0, candidates, 0, count);
        for (int i = 0; i < count; i++)
        {
            int v = candidates[i];
            int c = 1;
            while (true)
            {
                bool ok = true;
                for (int j = 0; j < i; j++)
                {
                    if (colors[j] == c && IsEdge(v, candidates[j]))
                    {
                        ok = false;
                        break;
                    }
                }
                if (ok) break;
                c++;
            }
            colors[i] = c;
        }

        Array.Sort(colors, candidates, 0, count, Comparer<int>.Create((a, b) => b.CompareTo(a)));
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static bool IsEdge(int u, int v)
    {
        return ((_adjBits[u][v >> 6] >> (v & 63)) & 1UL) != 0;
    }
