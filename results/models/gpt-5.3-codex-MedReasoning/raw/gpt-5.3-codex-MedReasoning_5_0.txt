use std::cmp::Ordering;
use std::collections::VecDeque;
use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Debug)]
struct Cell {
    x: i32,
    y: i32,
}

#[derive(Clone)]
struct Graph {
    w: i32,
    h: i32,
    n: usize,
    cells: Vec<Cell>,
    id_of: Vec<i32>,
    adj: Vec<Vec<usize>>,
    start: usize,
    require_cycle: bool,
}

#[derive(Clone)]
struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        Self { state: if seed == 0 { 0x9E3779B97F4A7C15 } else { seed } }
    }
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }
}

fn idx(w: i32, x: i32, y: i32) -> usize {
    (y * w + x) as usize
}

fn parse_input() -> Option<(i32, i32, usize, bool, Vec<(i32, i32)>)> {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).ok()?;
    let mut it = s.split_whitespace();

    let w: i32 = it.next()?.parse().ok()?;
    let h: i32 = it.next()?.parse().ok()?;
    let k: usize = it.next()?.parse().ok()?;
    let require_cycle_num: i32 = it.next()?.parse().ok()?;
    let require_cycle = require_cycle_num != 0;

    let mut obs = Vec::with_capacity(k);
    for _ in 0..k {
        let x: i32 = it.next()?.parse().ok()?;
        let y: i32 = it.next()?.parse().ok()?;
        obs.push((x, y));
    }
    Some((w, h, k, require_cycle, obs))
}

fn build_graph(w: i32, h: i32, obstacles: &[(i32, i32)], require_cycle: bool) -> Option<Graph> {
    if w <= 0 || h <= 0 {
        return None;
    }
    let total = (w as i64) * (h as i64);
    if total <= 0 || total > (usize::MAX as i64) {
        return None;
    }

    let mut blocked = vec![false; total as usize];
    for &(x, y) in obstacles {
        if x >= 0 && x < w && y >= 0 && y < h {
            blocked[idx(w, x, y)] = true;
        }
    }

    if blocked[idx(w, 0, 0)] {
        return None;
    }

    let mut cells = Vec::new();
    let mut id_of = vec![-1; total as usize];

    for y in 0..h {
        for x in 0..w {
            let gi = idx(w, x, y);
            if !blocked[gi] {
                let id = cells.len() as i32;
                id_of[gi] = id;
                cells.push(Cell { x, y });
            }
        }
    }

    let n = cells.len();
    if n == 0 {
        return None;
    }

    let mut adj = vec![Vec::new(); n];
    let dirs = [(1, 0), (-1, 0), (0, 1), (0, -1)];
    for (u, c) in cells.iter().enumerate() {
        for (dx, dy) in dirs {
            let nx = c.x + dx;
            let ny = c.y + dy;
            if nx >= 0 && nx < w && ny >= 0 && ny < h {
                let v = id_of[idx(w, nx, ny)];
                if v >= 0 {
                    adj[u].push(v as usize);
                }
            }
        }
    }

    let start = id_of[idx(w, 0, 0)] as usize;

    Some(Graph {
        w,
        h,
        n,
        cells,
        id_of,
        adj,
        start,
        require_cycle,
    })
}

fn obstacle_free(obstacles: &[(i32, i32)]) -> bool {
    obstacles.is_empty()
}

fn snake_path(w: i32, h: i32) -> Vec<Cell> {
    let mut p = Vec::with_capacity((w * h) as usize);
    for y in 0..h {
        if y % 2 == 0 {
            for x in 0..w {
                p.push(Cell { x, y });
            }
        } else {
            for x in (0..w).rev() {
                p.push(Cell { x, y });
            }
        }
    }
    p
}

fn cycle_rect_even_h(w: i32, h: i32) -> Option<Vec<Cell>> {
    if w <= 1 || h <= 1 || h % 2 != 0 {
        return None;
    }
    let mut p = Vec::with_capacity((w * h) as usize);

    p.push(Cell { x: 0, y: 0 });
    for x in 1..w {
        p.push(Cell { x, y: 0 });
    }
    for y in 1..h {
        if y % 2 == 1 {
            p.push(Cell { x: w - 1, y });
            for x in (1..(w - 1)).rev() {
                p.push(Cell { x, y });
            }
        } else {
            p.push(Cell { x: 1, y });
            for x in 2..w {
                p.push(Cell { x, y });
            }
        }
    }
    for y in (1..h).rev() {
        p.push(Cell { x: 0, y });
    }

    Some(p)
}

fn transpose_cells(cells: &[Cell]) -> Vec<Cell> {
    cells.iter().map(|c| Cell { x: c.y, y: c.x }).collect()
}

fn construct_full_grid(w: i32, h: i32, require_cycle: bool) -> Option<Vec<Cell>> {
    if !require_cycle {
        return Some(snake_path(w, h));
    }

    if w <= 1 || h <= 1 {
        return None;
    }
    if ((w * h) % 2) != 0 {
        return None;
    }

    if h % 2 == 0 {
        return cycle_rect_even_h(w, h);
    }
    if w % 2 == 0 {
        let transposed = cycle_rect_even_h(h, w)?;
        return Some(transpose_cells(&transposed));
    }
    None
}

fn check_global_necessary(g: &Graph) -> bool {
    if g.n == 0 {
        return false;
    }

    let mut q = VecDeque::new();
    let mut seen = vec![false; g.n];
    seen[g.start] = true;
    q.push_back(g.start);
    while let Some(u) = q.pop_front() {
        for &v in &g.adj[u] {
            if !seen[v] {
                seen[v] = true;
                q.push_back(v);
            }
        }
    }
    if seen.iter().any(|&b| !b) {
        return false;
    }

    let mut black = 0usize;
    let mut white = 0usize;
    for c in &g.cells {
        if ((c.x + c.y) & 1) == 0 {
            black += 1;
        } else {
            white += 1;
        }
    }

    if g.require_cycle {
        if black != white {
            return false;
        }
        if g.adj[g.start].len() < 2 {
            return false;
        }
        for u in 0..g.n {
            if g.adj[u].len() < 2 {
                return false;
            }
        }
    } else {
        let start_black = ((g.cells[g.start].x + g.cells[g.start].y) & 1) == 0;
        if g.n % 2 == 0 {
            if black != white {
                return false;
            }
        } else {
            if start_black {
                if black != white + 1 {
                    return false;
                }
            } else if white != black + 1 {
                return false;
            }
        }
    }

    true
}

fn is_adj(a: Cell, b: Cell) -> bool {
    (a.x - b.x).abs() + (a.y - b.y).abs() == 1
}

fn validate_path(g: &Graph, path: &[usize]) -> bool {
    if path.len() != g.n || path.first().copied() != Some(g.start) {
        return false;
    }
    let mut seen = vec![false; g.n];
    for (i, &u) in path.iter().enumerate() {
        if seen[u] {
            return false;
        }
        seen[u] = true;
        if i > 0 {
            let a = g.cells[path[i - 1]];
            let b = g.cells[u];
            if !is_adj(a, b) {
                return false;
            }
        }
    }
    if g.require_cycle {
        let a = g.cells[*path.last().unwrap()];
        let b = g.cells[g.start];
        if !is_adj(a, b) {
            return false;
        }
    }
    true
}

fn available_degree(g: &Graph, u: usize, used: &[bool], current: usize, left: usize) -> usize {
    let mut d = 0usize;
    for &v in &g.adj[u] {
        if !used[v] || v == current {
            d += 1;
        }
    }
    if left == 1 { d } else { d }
}

fn remaining_connected(g: &Graph, used: &[bool], current: usize) -> bool {
    let mut start = None;
    for u in 0..g.n {
        if !used[u] || u == current {
            start = Some(u);
            break;
        }
    }
    let Some(s) = start else { return true; };

    let mut seen = vec![false; g.n];
    let mut q = VecDeque::new();
    seen[s] = true;
    q.push_back(s);

    while let Some(u) = q.pop_front() {
        for &v in &g.adj[u] {
            if (!used[v] || v == current) && !seen[v] {
                seen[v] = true;
                q.push_back(v);
            }
        }
    }

    for u in 0..g.n {
        if (!used[u] || u == current) && !seen[u] {
            return false;
        }
    }
    true
}

fn prune(g: &Graph, used: &[bool], current: usize, depth: usize) -> bool {
    let left = g.n - depth;

    let mut leaves = 0usize;
    for u in 0..g.n {
        if !used[u] {
            let d = available_degree(g, u, used, current, left);
            if d == 0 {
                return false;
            }
            if d == 1 {
                leaves += 1;
            }
        }
    }

    if g.require_cycle {
        if leaves > 0 {
            return false;
        }
    } else if leaves > 2 {
        return false;
    }

    true
}

fn solve_with_dfs(g: &Graph, time_limit: Duration) -> Option<Vec<usize>> {
    let start_time = Instant::now();
    let mut rng = XorShift64::new((g.n as u64).wrapping_mul(0xD1B54A32D192ED03));

    let mut order: Vec<usize> = (0..g.n).collect();
    order.sort_by_key(|&u| g.adj[u].len());
    let _ = order;

    let mut attempts = 0usize;

    while start_time.elapsed() < time_limit {
        attempts += 1;

        let mut used = vec![false; g.n];
        let mut path = Vec::with_capacity(g.n);
        used[g.start] = true;
        path.push(g.start);

        let mut next_idx = vec![0usize; g.n];
        let mut cand_cache: Vec<Vec<usize>> = vec![Vec::new(); g.n];

        let mut depth = 1usize;
        let mut fail_steps = 0usize;

        while depth > 0 {
            if start_time.elapsed() >= time_limit {
                break;
            }

            if depth == g.n {
                let end = *path.last().unwrap();
                if !g.require_cycle || g.adj[end].contains(&g.start) {
                    return Some(path);
                }
            }

            let current = *path.last().unwrap();

            if cand_cache[depth - 1].is_empty() {
                let mut cands: Vec<usize> = g.adj[current]
                    .iter()
                    .copied()
                    .filter(|&v| !used[v])
                    .collect();

                cands.sort_by(|&a, &b| {
                    let da = g.adj[a].iter().filter(|&&z| !used[z]).count();
                    let db = g.adj[b].iter().filter(|&&z| !used[z]).count();
                    match da.cmp(&db) {
                        Ordering::Equal => {
                            let ra = (rng.next_u64() & 0xFFFF) as u16;
                            let rb = (rng.next_u64() & 0xFFFF) as u16;
                            ra.cmp(&rb)
                        }
                        x => x,
                    }
                });

                cand_cache[depth - 1] = cands;
                next_idx[depth - 1] = 0;
            }

            let mut advanced = false;
            while next_idx[depth - 1] < cand_cache[depth - 1].len() {
                let v = cand_cache[depth - 1][next_idx[depth - 1]];
                next_idx[depth - 1] += 1;

                used[v] = true;
                path.push(v);
                depth += 1;

                let do_connectivity = depth % 12 == 0 || depth + 16 >= g.n;
                let ok = prune(g, &used, v, depth)
                    && (!do_connectivity || remaining_connected(g, &used, v));

                if ok {
                    if depth - 1 < cand_cache.len() {
                        cand_cache[depth - 1].clear();
                        next_idx[depth - 1] = 0;
                    }
                    advanced = true;
                    break;
                }

                depth -= 1;
                path.pop();
                used[v] = false;
            }

            if !advanced {
                cand_cache[depth - 1].clear();
                next_idx[depth - 1] = 0;

                if depth == 1 {
                    break;
                }

                let last = path.pop().unwrap();
                used[last] = false;
                depth -= 1;
                fail_steps += 1;

                if g.n > 300 && fail_steps > 150_000 {
                    break;
                }
            }
        }

        if g.n <= 80 && attempts > 24 {
            break;
        }
        if g.n > 80 && attempts > 8 {
            break;
        }
    }

    None
}

fn main() {
    let Some((w, h, _k, require_cycle, obstacles)) = parse_input() else {
        println!("0");
        return;
    };

    if obstacle_free(&obstacles) {
        if let Some(path) = construct_full_grid(w, h, require_cycle) {
            println!("{}", path.len());
            for c in path {
                println!("{} {}", c.x, c.y);
            }
            return;
        }
    }

    let Some(g) = build_graph(w, h, &obstacles, require_cycle) else {
        println!("0");
        return;
    };

    if !check_global_necessary(&g) {
        println!("0");
        return;
    }

    if g.n == 1 {
        if g.require_cycle {
            println!("0");
        } else {
            let c = g.cells[g.start];
            println!("1");
            println!("{} {}", c.x, c.y);
        }
        return;
    }

    let time_limit = Duration::from_secs(290);
    if let Some(path_ids) = solve_with_dfs(&g, time_limit) {
        if validate_path(&g, &path_ids) {
            println!("{}", path_ids.len());
            for u in path_ids {
                let c = g.cells[u];
                println!("{} {}", c.x, c.y);
            }
            return;
        }
    }

    println!("0");
}
