struct Point {
    float x;
    float y;
};

static inline double dist2(const Point& a, const Point& b) {
    const double dx = static_cast<double>(a.x) - static_cast<double>(b.x);
    const double dy = static_cast<double>(a.y) - static_cast<double>(b.y);
    return dx * dx + dy * dy;
}

static inline uint64_t part1by1(uint32_t v) {
    uint64_t x = v;
    x = (x | (x << 16)) & 0x0000FFFF0000FFFFULL;
    x = (x | (x << 8)) & 0x00FF00FF00FF00FFULL;
    x = (x | (x << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    x = (x | (x << 2)) & 0x3333333333333333ULL;
    x = (x | (x << 1)) & 0x5555555555555555ULL;
    return x;
}

static inline uint64_t morton2D(uint32_t x, uint32_t y) {
    return (part1by1(y) << 1) | part1by1(x);
}

std::vector<int> solveExactDP(const std::vector<Point>& pts) {
    const int n = static_cast<int>(pts.size());
    const int full = 1 << n;
    const double INF = std::numeric_limits<double>::infinity();

    std::vector<double> dp(static_cast<size_t>(full) * n, INF);
    std::vector<int> parent(static_cast<size_t>(full) * n, -1);

    auto at = [n](int mask, int last) -> size_t {
        return static_cast<size_t>(mask) * n + last;
    };

    dp[at(1, 0)] = 0.0;

    for (int mask = 1; mask < full; ++mask) {
        if ((mask & 1) == 0) continue;
        for (int last = 0; last < n; ++last) {
            const double cur = dp[at(mask, last)];
            if (!std::isfinite(cur)) continue;
            for (int nxt = 0; nxt < n; ++nxt) {
                if (mask & (1 << nxt)) continue;
                const int nmask = mask | (1 << nxt);
                const double cand = cur + dist2(pts[last], pts[nxt]);
                size_t idx = at(nmask, nxt);
                if (cand < dp[idx]) {
                    dp[idx] = cand;
                    parent[idx] = last;
                }
            }
        }
    }

    const int endMask = full - 1;
    int bestLast = 0;
    double best = INF;
    for (int last = 0; last < n; ++last) {
        const double cand = dp[at(endMask, last)] + dist2(pts[last], pts[0]);
        if (cand < best) {
            best = cand;
            bestLast = last;
        }
    }

    std::vector<int> route(n, 0);
    int mask = endMask;
    int cur = bestLast;
    for (int pos = n - 1; pos >= 1; --pos) {
        route[pos] = cur;
        const int p = parent[at(mask, cur)];
        mask ^= (1 << cur);
        cur = p;
    }
    route[0] = 0;
    return route;
}

std::vector<int> solveNearestNeighbor(const std::vector<Point>& pts) {
    const int n = static_cast<int>(pts.size());
    std::vector<int> route;
    route.reserve(n);
    std::vector<unsigned char> used(n, 0);

    int cur = 0;
    route.push_back(cur);
    used[cur] = 1;

    for (int step = 1; step < n; ++step) {
        int best = -1;
        double bestD = std::numeric_limits<double>::infinity();
        for (int i = 0; i < n; ++i) {
            if (used[i]) continue;
            const double d = dist2(pts[cur], pts[i]);
            if (d < bestD) {
                bestD = d;
                best = i;
            }
        }
        cur = best;
        used[cur] = 1;
        route.push_back(cur);
    }
    return route;
}

void twoOptWindow(std::vector<int>& route, const std::vector<Point>& pts, int window, int passes) {
    const int n = static_cast<int>(route.size());
    if (n < 4) return;

    for (int pass = 0; pass < passes; ++pass) {
        bool improved = false;

        for (int i = 0; i + 2 < n; ++i) {
            int a = route[i];
            int b = route[i + 1];

            const int jMax = std::min(n - 2, i + window);
            for (int j = i + 2; j <= jMax; ++j) {
                int c = route[j];
                int d = route[j + 1];

                const double oldCost = dist2(pts[a], pts[b]) + dist2(pts[c], pts[d]);
                const double newCost = dist2(pts[a], pts[c]) + dist2(pts[b], pts[d]);

                if (newCost + 1e-12 < oldCost) {
                    std::reverse(route.begin() + (i + 1), route.begin() + (j + 1));
                    improved = true;
                    b = route[i + 1];
                }
            }
        }

        if (n >= 4) {
            int a = route[n - 1];
            int b = route[0];
            for (int j = 1; j + 1 < std::min(n, window + 1); ++j) {
                int c = route[j];

                const double oldCost = dist2(pts[a], pts[b]) + dist2(pts[c], pts[d]);

                    improved = true;
                    b = route[0];
                }
            }
        }

        if (!improved) break;
    }
}

std::vector<int> solveMorton(const std::vector<Point>& pts) {
    const int n = static_cast<int>(pts.size());
    std::vector<int> idx(n);
    std::iota(idx.begin(), idx.end(), 0);

    float minX = pts[0].x, maxX = pts[0].x;
    float minY = pts[0].y, maxY = pts[0].y;
    for (const auto& p : pts) {
        if (p.x < minX) minX = p.x;
        if (p.x > maxX) maxX = p.x;
        if (p.y < minY) minY = p.y;
        if (p.y > maxY) maxY = p.y;
    }

    const double rangeX = static_cast<double>(maxX) - static_cast<double>(minX);
    const double rangeY = static_cast<double>(maxY) - static_cast<double>(minY);
    constexpr uint32_t SCALE = (1u << 21) - 1u;

    std::vector<uint64_t> keys(n);
    for (int i = 0; i < n; ++i) {
        uint32_t qx = 0, qy = 0;
        if (rangeX > 0.0) {
            const double nx = (static_cast<double>(pts[i].x) - minX) / rangeX;
            qx = static_cast<uint32_t>(nx * SCALE);
        }
        if (rangeY > 0.0) {
            const double ny = (static_cast<double>(pts[i].y) - minY) / rangeY;
            qy = static_cast<uint32_t>(ny * SCALE);
        }
        keys[i] = morton2D(qx, qy);
    }

    std::sort(idx.begin(), idx.end(), [&](int a, int b) {
        if (keys[a] != keys[b]) return keys[a] < keys[b];
        if (pts[a].x != pts[b].x) return pts[a].x < pts[b].x;
        return pts[a].y < pts[b].y;
    });

    return idx;
}

void rotateToStartZero(std::vector<int>& route) {
    const int n = static_cast<int>(route.size());
    int pos0 = -1;
    for (int i = 0; i < n; ++i) {
        if (route[i] == 0) {
            pos0 = i;
            break;
        }
    }
    if (pos0 <= 0) return;
    std::rotate(route.begin(), route.begin() + pos0, route.end());
}

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }

    {

        int n = fs.NextInt();
        if (n == int.MinValue)
        {
            Console.WriteLine("0");
            return;

        if (n <= 0)
        {
            Console.WriteLine("0");
            return;

        int[] v = new int[m];

        for (int i = 0; i < m; i++)

            int a = fs.NextInt();
            int b = fs.NextInt();

            u[i] = a;
            v[i] = b;

            deg[a] += 2;
            deg[b] += 2;

        int[] start = new int[n + 1];

        int halfEdgeCount = start[n]; // equals 4*m (each of 2 copies contributes 2 half-edges)
        int[] to = new int[halfEdgeCount];

            eid[p] = e0;
            p = cur[b]++;
            to[p] = a;
            eid[p] = e0;

            p = cur[a]++;
            to[p] = b;
            eid[p] = e1;
            p = cur[b]++;
            to[p] = a;
            eid[p] = e1;

        Array.Copy(start, ptr, n);

            return;

        {

            sb.Append(circuit[i]);
        }

struct XorShift64 {

impl XorShift64 {
    fn new(seed: u64) -> Self {
        let s = if seed == 0 { 0x9e3779b97f4a7c15 } else { seed };
        Self { state: s }
    }

    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }

    fn next_f64(&mut self) -> f64 {
        let v = self.next_u64() >> 11;
        (v as f64) * (1.0 / ((1u64 << 53) as f64))
    }

        if upper <= 1 {

    }
}

fn bfs_dist(

fn layout_component_radial(

    comp_id: usize,
    comp_of: &[usize],
    offsets: &[usize],
    adj: &[usize],
    deg: &[usize],

    y: &mut [f64],
    rng: &mut XorShift64,

fn force_refine(
    nodes: &[usize],
    edges: &[(usize, usize)],
    x: &mut [f64],
    y: &mut [f64],

) {

            dy[u] = 0.0;

                    let mut ry = y[u] - y[v];
                    let d2 = rx * rx + ry * ry + 1e-9;
                    let inv_d = 1.0 / d2.sqrt();
                    let force = (k * k) / d2;
                    rx *= inv_d * force;
                    ry *= inv_d * force;
                    dx[u] += rx;
                    dy[u] += ry;
                    dx[v] -= rx;
                    dy[v] -= ry;
                }
            }
        } else {
            for &u in nodes {
                for _ in 0..sample_rep {
                    let v = nodes[rng.gen_range(n)];
                    if v == u {
                        continue;

                    let mut rx = x[u] - x[v];
                    let mut ry = y[u] - y[v];
                    let d2 = rx * rx + ry * ry + 1e-9;

                    let force = (k * k) / d2;
                    rx *= inv_d * force;
                    ry *= inv_d * force;
                    dx[u] += rx;
                    dy[u] += ry;
                }
            }

        for &(u, v) in edges {
            let mut ax = x[u] - x[v];
            let mut ay = y[u] - y[v];
            let dist = (ax * ax + ay * ay + 1e-9).sqrt();
            let force = (dist * dist) / k;
            ax = ax / dist * force;
            ay = ay / dist * force;
            dx[u] -= ax;
            dy[u] -= ay;
            dx[v] += ax;
            dy[v] += ay;

        let mut cx = 0.0;
        let mut cy = 0.0;

            cy += y[u];

        cx /= n as f64;
        cy /= n as f64;

        for &u in nodes {
            dx[u] += (cx - x[u]) * 0.004;
            dy[u] += (cy - y[u]) * 0.004;

        // Apply with clipping by temperature.
        for &u in nodes {
            let len = (dx[u] * dx[u] + dy[u] * dy[u]).sqrt();
            if len > 0.0 {
                let step = temp.min(len);
                x[u] += dx[u] / len * step;
                y[u] += dy[u] / len * step;
            }
        }
    }

struct Plane {
    Vec3 n; // outward normal (unit length)
    double d{}; // n.p + d <= eps is inside
};

struct FaceKey {
    int a, b, c;
    bool operator==(const FaceKey& o) const { return a == o.a && b == o.b && c == o.c; }
};

struct FaceKeyHash {
    std::size_t operator()(const FaceKey& k) const {
        std::size_t h1 = std::hash<int>{}(k.a);
        std::size_t h2 = std::hash<int>{}(k.b);
        std::size_t h3 = std::hash<int>{}(k.c);
        return h1 ^ (h2 * 1315423911u) ^ (h3 * 2654435761u);
    }
};

static std::vector<Plane> buildConvexHullHalfspaces(const std::vector<Vec3>& p) {
    const int n = static_cast<int>(p.size());
    const double eps = 1e-10;
    std::vector<Plane> planes;

    // For very large vertex counts, avoid O(V^4): use AABB halfspaces.
    if (n > 120) {
        Vec3 mn = p[0], mx = p[0];
        for (const auto& v : p) {
            mn.x = std::min(mn.x, v.x); mn.y = std::min(mn.y, v.y); mn.z = std::min(mn.z, v.z);
            mx.x = std::max(mx.x, v.x); mx.y = std::max(mx.y, v.y); mx.z = std::max(mx.z, v.z);
        }
        planes.push_back({{ 1, 0, 0}, -mx.x});
        planes.push_back({{-1, 0, 0},  mn.x});
        planes.push_back({{ 0, 1, 0}, -mx.y});
        planes.push_back({{ 0,-1, 0},  mn.y});
        planes.push_back({{ 0, 0, 1}, -mx.z});
        planes.push_back({{ 0, 0,-1},  mn.z});
        return planes;
    }

    std::unordered_map<FaceKey, bool, FaceKeyHash> seen;
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            for (int k = j + 1; k < n; ++k) {
                Vec3 nrm = cross(p[j] - p[i], p[k] - p[i]);
                double len = norm(nrm);
                if (len < eps) continue;

                int pos = 0, neg = 0;
                for (int m = 0; m < n; ++m) {
                    double s = dot(nrm, p[m] - p[i]);
                    if (s > 1e-8) ++pos;
                    else if (s < -1e-8) ++neg;
                    if (pos && neg) break;
                }
                if (pos && neg) continue;

                // Orient such that all points are inside: n.p + d <= eps
                if (pos > 0) nrm = nrm * -1.0;
                double inv = 1.0 / norm(nrm);
                nrm = nrm * inv;
                double d = -dot(nrm, p[i]);

                std::array<int, 3> ids{i, j, k};
                std::sort(ids.begin(), ids.end());
                FaceKey key{ids[0], ids[1], ids[2]};
                if (seen.find(key) == seen.end()) {
                    seen[key] = true;
                    planes.push_back({nrm, d});
                }
            }
        }
    }

    if (planes.empty()) {
        // Degenerate fallback: AABB.
        Vec3 mn = p[0], mx = p[0];
        for (const auto& v : p) {
            mn.x = std::min(mn.x, v.x); mn.y = std::min(mn.y, v.y); mn.z = std::min(mn.z, v.z);
            mx.x = std::max(mx.x, v.x); mx.y = std::max(mx.y, v.y); mx.z = std::max(mx.z, v.z);
        }
        planes.push_back({{ 1, 0, 0}, -mx.x});
        planes.push_back({{-1, 0, 0},  mn.x});
        planes.push_back({{ 0, 1, 0}, -mx.y});
        planes.push_back({{ 0,-1, 0},  mn.y});
        planes.push_back({{ 0, 0, 1}, -mx.z});
        planes.push_back({{ 0, 0,-1},  mn.z});
    }
    return planes;
}

static bool insideHalfspaces(const std::vector<Plane>& planes, const Vec3& p, double eps = 1e-9) {
    for (const auto& pl : planes) {
        if (dot(pl.n, p) + pl.d > eps) return false;
    }
    return true;
}

static Vec3 rotateEuler(const Vec3& v, const Vec3& r) {
    const double cx = std::cos(r.x), sx = std::sin(r.x);
    const double cy = std::cos(r.y), sy = std::sin(r.y);
    const double cz = std::cos(r.z), sz = std::sin(r.z);

    // Rz * Ry * Rx
    Vec3 t = {v.x, cx * v.y - sx * v.z, sx * v.y + cx * v.z};
    t = {cy * t.x + sy * t.z, t.y, -sy * t.x + cy * t.z};
    t = {cz * t.x - sz * t.y, sz * t.x + cz * t.y, t.z};
    return t;
}

fn available_degree(g: &Graph, u: usize, used: &[bool], current: usize, left: usize) -> usize {
    let mut d = 0usize;
    for &v in &g.adj[u] {
        if !used[v] || v == current {
            d += 1;
        }
    }
    if left == 1 { d } else { d }
}

fn remaining_connected(g: &Graph, used: &[bool], current: usize) -> bool {
    let mut start = None;
    for u in 0..g.n {
        if !used[u] || u == current {
            start = Some(u);
            break;
        }
    }
    let Some(s) = start else { return true; };

    let mut seen = vec![false; g.n];
    let mut q = VecDeque::new();
    seen[s] = true;
    q.push_back(s);

    while let Some(u) = q.pop_front() {
        for &v in &g.adj[u] {
            if (!used[v] || v == current) && !seen[v] {
                seen[v] = true;
                q.push_back(v);
            }
        }
    }

    for u in 0..g.n {
        if (!used[u] || u == current) && !seen[u] {
            return false;
        }
    }
    true
}

fn prune(g: &Graph, used: &[bool], current: usize, depth: usize) -> bool {
    let left = g.n - depth;

    let mut leaves = 0usize;
    for u in 0..g.n {
        if !used[u] {
            let d = available_degree(g, u, used, current, left);
            if d == 0 {
                return false;
            }
            if d == 1 {
                leaves += 1;
            }
        }
    }

    if g.require_cycle {
        if leaves > 0 {
            return false;
        }
    } else if leaves > 2 {
        return false;
    }

    true
}

fn solve_with_dfs(g: &Graph, time_limit: Duration) -> Option<Vec<usize>> {

    let mut order: Vec<usize> = (0..g.n).collect();

        let mut depth = 1usize;
        let mut fail_steps = 0usize;

        while depth > 0 {
            if start_time.elapsed() >= time_limit {
                break;
            }

            if depth == g.n {
                let end = *path.last().unwrap();
                if !g.require_cycle || g.adj[end].contains(&g.start) {
                    return Some(path);
                }

            let current = *path.last().unwrap();

                let mut cands: Vec<usize> = g.adj[current]
                    .iter()
                    .copied()
                    .filter(|&v| !used[v])
                    .collect();

                cands.sort_by(|&a, &b| {
                    let da = g.adj[a].iter().filter(|&&z| !used[z]).count();
                    let db = g.adj[b].iter().filter(|&&z| !used[z]).count();
                    match da.cmp(&db) {
                        Ordering::Equal => {
                            let ra = (rng.next_u64() & 0xFFFF) as u16;
                            let rb = (rng.next_u64() & 0xFFFF) as u16;
                            ra.cmp(&rb)
                        }
                        x => x,
                    }
                });

                cand_cache[depth - 1] = cands;
                next_idx[depth - 1] = 0;
            }

            let mut advanced = false;
            while next_idx[depth - 1] < cand_cache[depth - 1].len() {
                let v = cand_cache[depth - 1][next_idx[depth - 1]];
                next_idx[depth - 1] += 1;

                used[v] = true;
                path.push(v);
                depth += 1;

                let do_connectivity = depth % 12 == 0 || depth + 16 >= g.n;
                let ok = prune(g, &used, v, depth)
                    && (!do_connectivity || remaining_connected(g, &used, v));

                if ok {
                    if depth - 1 < cand_cache.len() {
                        cand_cache[depth - 1].clear();
                        next_idx[depth - 1] = 0;
                    }
                    advanced = true;
                    break;
                }

                depth -= 1;
                path.pop();
                used[v] = false;
            }

            if !advanced {
                cand_cache[depth - 1].clear();
                next_idx[depth - 1] = 0;

                if depth == 1 {
                    break;
                }

                let last = path.pop().unwrap();
                used[last] = false;
                depth -= 1;
                fail_steps += 1;

                if g.n > 300 && fail_steps > 150_000 {
                    break;
                }
            }
        }

            break;
        }
        if g.n > 80 && attempts > 8 {
            break;
        }
    }

    None
}

    {
        int prev = pos == 0 ? -1 : path[pos - 1];
        int next = pos == path.Count ? -1 : path[pos];

        double add = EdgeFrom(prev, node, cStart, c);
        if (next >= 0) add += c[node][next];

        double rem = 0;
        if (next >= 0) rem += EdgeFrom(prev, next, cStart, c);

        return add - rem;
    }

    private static double PathCost(List<int> path, double[] cStart, double[][] c)
    {
        if (path.Count == 0) return 0;
        double s = cStart[path[0]];
        for (int i = 1; i < path.Count; i++) s += c[path[i - 1]][path[i]];
        return s;
    }

    private static void ImprovePath2Opt(List<int> path, double[] cStart, double[][] c, int maxPasses, int candidateWindow)
    {
        int n = path.Count;
        if (n < 4) return;

        candidateWindow = Math.Max(2, Math.Min(candidateWindow, n - 1));

        for (int pass = 0; pass < maxPasses; pass++)
        {
            bool improved = false;

            for (int i = 0; i < n - 2; i++)
            {
                int maxK = Math.Min(n - 1, i + candidateWindow);
                for (int k = i + 1; k <= maxK; k++)
                {
                    int prev = i == 0 ? -1 : path[i - 1];
                    int a = path[i];
                    int b = path[k];
                    int next = (k == n - 1) ? -1 : path[k + 1];

                    double oldCost = EdgeFrom(prev, a, cStart, c) + (next >= 0 ? c[b][next] : 0.0);
                    double newCost = EdgeFrom(prev, b, cStart, c) + (next >= 0 ? c[a][next] : 0.0);

                    if (newCost + 1e-12 < oldCost)
                    {
                        path.Reverse(i, k - i + 1);
                        improved = true;
                    }
                }
            }

            if (!improved) break;
        }
    }

    private static void ImprovePathRelocate(List<int> path, double[] cStart, double[][] c, int maxPasses)
    {
        int n = path.Count;
        if (n < 4) return;

        for (int pass = 0; pass < maxPasses; pass++)
        {
            bool improved = false;

            for (int i = 0; i < n; i++)
            {
                int node = path[i];
                int prevI = (i == 0) ? -1 : path[i - 1];
                int nextI = (i == n - 1) ? -1 : path[i + 1];

                double removeDelta = -(EdgeFrom(prevI, node, cStart, c) + (nextI >= 0 ? c[node][nextI] : 0.0))
                                     + (nextI >= 0 ? EdgeFrom(prevI, nextI, cStart, c) : 0.0);

                for (int j = 0; j <= n; j++)
                {
                    if (j == i || j == i + 1) continue;

                    int prevJ, nextJ;
                    if (j <= i)
                    {
                        prevJ = (j == 0) ? -1 : path[j - 1];
                        nextJ = path[j];
                    }
                    else
                    {
                        prevJ = path[j - 1];
                        nextJ = (j == n) ? -1 : path[j];
                    }

                    double insertDelta = EdgeFrom(prevJ, node, cStart, c) + (nextJ >= 0 ? c[node][nextJ] : 0.0)
                                       - (nextJ >= 0 ? EdgeFrom(prevJ, nextJ, cStart, c) : 0.0);

                    double totalDelta = removeDelta + insertDelta;
                    if (totalDelta < -1e-12)
                    {
                        path.RemoveAt(i);
                        int ins = j;
                        if (j > i) ins = j - 1;
                        path.Insert(ins, node);
                        improved = true;
                        goto NextPass;
                    }
                }
            }

            NextPass:
            if (!improved) break;
            n = path.Count;
        }
    }
