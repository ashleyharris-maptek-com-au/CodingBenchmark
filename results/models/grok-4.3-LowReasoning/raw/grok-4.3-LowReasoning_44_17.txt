```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    vec2 scale = vec2(8.0, 16.0);
    vec2 uvScaled = uv * scale;
    float row = floor(uvScaled.y);
    float offset = mod(row, 2.0) * 0.5;
    vec2 brickUV = vec2(uvScaled.x + offset, uvScaled.y);
    vec2 f = fract(brickUV);

    float mx = 0.05;
    float my = 0.1;
    bool isMortar = f.x < mx || f.x > (1.0 - mx) || f.y < my || f.y > (1.0 - my);

    vec3 brickCol = vec3(0.7, 0.2, 0.1);
    vec3 mortarCol = vec3(0.7, 0.7, 0.7);
    vec3 baseColor = isMortar ? mortarCol : brickCol;

    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);

    vec3 shaded = baseColor * (0.4 + 0.6 * diffuse);
    outColor = vec4(shaded, 1.0);
}
```
