OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"

OpEntryPoint Geometry %main "main" %gl_in %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%uint = OpTypeInt 32 0
%uint_2 = OpConstant %uint 2
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%arr_v3_2 = OpTypeArray %v3float %uint_2
%ptr_in_v3arr = OpTypePointer Input %arr_v3_2
%color = OpVariable %ptr_in_v3arr Input
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%float_0_03 = OpConstant %float 0.03
%main = OpFunction %void None %func
%entry = OpLabel
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%dir4 = OpFSub %v4float %p1 %p0
%dir = OpVectorShuffle %v2float %dir4 %dir4 0 1
%neg_dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y_n = OpFNegate %float %neg_dir_y
%dir_x = OpCompositeExtract %float %dir 0
%perp2 = OpCompositeConstruct %v2float %neg_dir_y_n %dir_x
%len = OpExtInst %float %1 Length %perp2
%inv_len = OpFDiv %float %float_0_03 %len
%perp_scaled = OpVectorTimesScalar %v2float %perp2 %inv_len
%perp4 = OpCompositeConstruct %v4float %perp_scaled 0 0
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %c1_ptr
; vertex 0: p0 + perp
%pos0 = OpFAdd %v4float %p0 %perp4
%out0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out0_ptr %pos0
%out_c0_ptr = OpAccessChain %ptr_out_v3 %outColor
OpStore %out_c0_ptr %c0
OpEmitVertex
; vertex 1: p0 - perp
%neg_perp4 = OpFNegate %v4float %perp4
%pos1 = OpFAdd %v4float %p0 %neg_perp4
%out1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out1_ptr %pos1
%out_c1_ptr = OpAccessChain %ptr_out_v3 %outColor
OpStore %out_c1_ptr %c0
OpEmitVertex
; vertex 2: p1 + perp
%pos2 = OpFAdd %v4float %p1 %perp4
%out2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out2_ptr %pos2
%out_c2_ptr = OpAccessChain %ptr_out_v3 %outColor
OpStore %out_c2_ptr %c1
OpEmitVertex
; vertex 3: p1 - perp
%pos3 = OpFAdd %v4float %p1 %neg_perp4
%out3_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out3_ptr %pos3
%out_c3_ptr = OpAccessChain %ptr_out_v3 %outColor
OpStore %out_c3_ptr %c1
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
