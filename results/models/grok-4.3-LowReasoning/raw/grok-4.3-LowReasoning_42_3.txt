; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
; Schema: 0

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_color %gl_out %out_color
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_color Location 2
OpDecorate %out_color Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_thick = OpConstant %float 0.03

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_gl = OpTypePointer Input %arr_gl_in
%gl_in = OpVariable %ptr_in_gl Input

%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_v3arr = OpTypePointer Input %arr_v3_in
%in_color = OpVariable %ptr_in_v3arr Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_gl = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl Output

%ptr_out_v3 = OpTypePointer Output %v3float
%out_color = OpVariable %ptr_out_v3 Output

%fn_main = OpTypeFunction %void
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %fn_main
%entry = OpLabel

; Load endpoints
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

; Load colors
%c0_ptr = OpAccessChain %ptr_in_v3arr %in_color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3arr %in_color %int_1
%c1 = OpLoad %v3float %c1_ptr

; dir.xy = p1.xy - p0.xy
%p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir = OpFSub %v2float %p1xy %p0xy

; perp = normalize((-dir.y, dir.x))
%dir_y = OpCompositeExtract %float %dir 1
%dir_x = OpCompositeExtract %float %dir 0
%neg_dir_y = OpFNegate %float %dir_y
%perp_raw = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%len = OpDot %float %perp_raw %perp_raw
%sqrt_len = OpSqrt %float %len
%inv_len = OpFDiv %float %float_1 %sqrt_len
%perp_unit = OpVectorTimesScalar %v2float %perp_raw %inv_len
%perp = OpVectorTimesScalar %v2float %perp_unit %float_thick

; Build offset vectors (z=0,w=0)
%zero2 = OpCompositeConstruct %v2float %float_0 %float_0
%perp4 = OpVectorShuffle %v4float %perp %zero2 0 1 2 3
%neg_perp = OpFNegate %v2float %perp
%neg_perp4 = OpVectorShuffle %v4float %neg_perp %zero2 0 1 2 3

; v0 = p0 + perp
%v0 = OpFAdd %v4float %p0 %perp4
; v1 = p0 - perp
%v1 = OpFAdd %v4float %p0 %neg_perp4
; v2 = p1 + perp
%v2 = OpFAdd %v4float %p1 %perp4
; v3 = p1 - perp
%v3 = OpFAdd %v4float %p1 %neg_perp4

; Emit v0 (c0)
%out0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out0 %v0
OpStore %out_color %c0
OpEmitVertex

; Emit v1 (c0)
%out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out1 %v1
OpStore %out_color %c0
OpEmitVertex

; Emit v2 (c1)
%out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out2 %v2
OpStore %out_color %c1
OpEmitVertex

; Emit v3 (c1)
%out3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out3 %v3
OpStore %out_color %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
