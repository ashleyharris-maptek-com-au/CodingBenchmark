```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 100
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %gl_in DescriptorSet 0
OpDecorate %gl_in Binding 0

OpDecorate %color Location 2
OpDecorate %outColor Location 0

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0_03 = OpConstant %float 0.03
%float_0 = OpConstant %float 0

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%in_color = OpTypeArray %v3float %uint_2
%ptr_in_color = OpTypePointer Input %in_color
%color = OpVariable %ptr_in_color Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%main = OpFunction %void None %func
%entry = OpLabel

; Load p0, p1
%chain0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %chain0
%chain1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %chain1

; Load c0, c1
%c0_chain = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %c0_chain
%c1_chain = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %c1_chain

; dir = p1 - p0 (xy only)
%dir4 = OpFSub %v4float %p1 %p0
%dir = OpVectorShuffle %v3float %dir4 %dir4 0 1 2
%dir_xy = OpVectorShuffle %v2float %dir4 %dir4 0 1

; perp = normalize(-dir.y, dir.x)
%neg_y = OpCompositeExtract %float %dir_xy 1
%neg_y = OpFNegate %float %neg_y
%px = OpCompositeExtract %float %dir_xy 0
%perp_vec2 = OpCompositeConstruct %v2float %neg_y %px
%len = OpDot %float %perp_vec2 %perp_vec2
%len_sqrt = OpSqrt %float %len
%inv_len = OpFDiv %float %float_0_03 %len_sqrt
%perp_scaled = OpVectorTimesScalar %v2float %perp_vec2 %inv_len

; Build four positions
%zero = OpCompositeConstruct %v2float %float_0 %float_0
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1

%v0_xy = OpFAdd %v2float %p0_xy %perp_scaled
%v1_xy = OpFSub %v2float %p0_xy %perp_scaled
%v2_xy = OpFAdd %v2float %p1_xy %perp_scaled
%v3_xy = OpFSub %v2float %p1_xy %perp_scaled

%v0 = OpCompositeConstruct %v4float %v0_xy %float_0 %float_1
%v1 = OpCompositeConstruct %v4float %v1_xy %float_0 %float_1
%v2 = OpCompositeConstruct %v4float %v2_xy %float_0 %float_1
%v3 = OpCompositeConstruct %v4float %v3_xy %float_0 %float_1

; Emit vertex 0 (c0)
%store0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %store0 %v0
%out0 = OpAccessChain %ptr_out_v3 %outColor
OpStore %out0 %c0
OpEmitVertex

; Emit vertex 1 (c0)
%store1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %store1 %v1
%out1 = OpAccessChain %ptr_out_v3 %outColor
OpStore %out1 %c0
OpEmitVertex

; Emit vertex 2 (c1)
%store2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %store2 %v2
%out2 = OpAccessChain %ptr_out_v3 %outColor
OpStore %out2 %c1
OpEmitVertex

; Emit vertex 3 (c1)
%store3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %store3 %v3
%out3 = OpAccessChain %ptr_out_v3 %outColor
OpStore %out3 %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
