```typescript
declare const require: any;
const fs = require('fs');

interface Package {
  id: number;
  x: number;
  y: number;
}

interface Input {
  depot: [number, number];
  truckCount: number;
  packages: Package[];
}

interface Output {
  routes: number[][];
}

function distance(a: [number, number], b: [number, number]): number {
  const dx = a[0] - b[0];
  const dy = a[1] - b[1];
  return Math.sqrt(dx * dx + dy * dy);
}

function angleFromDepot(depot: [number, number], p: Package): number {
  return Math.atan2(p.y - depot[1], p.x - depot[0]);
}

function nearestNeighborTSP(depot: [number, number], pkgs: Package[]): number[] {
  if (pkgs.length === 0) return [];
  const remaining = new Set(pkgs.map(p => p.id));
  const pkgMap = new Map(pkgs.map(p => [p.id, p]));
  const route: number[] = [];
  let current: [number, number] = depot;

  while (remaining.size > 0) {
    let bestId = -1;
    let bestDist = Infinity;
    for (const id of remaining) {
      const p = pkgMap.get(id)!;
      const d = distance(current, [p.x, p.y]);
      if (d < bestDist) {
        bestDist = d;
        bestId = id;
      }
    }
    route.push(bestId);
    const nextPkg = pkgMap.get(bestId)!;
    current = [nextPkg.x, nextPkg.y];
    remaining.delete(bestId);
  }
  return route;
}

const input: string = fs.readFileSync(0, 'utf8');
const data: Input = JSON.parse(input);

const depot: [number, number] = data.depot;
const truckCount: number = Math.max(1, data.truckCount);
const packages: Package[] = data.packages;

if (packages.length === 0) {
  const output: Output = { routes: [] };
  console.log(JSON.stringify(output));
  process.exit(0);
}

// Sort by angle
const sorted = [...packages].sort((a, b) => {
  const aa = angleFromDepot(depot, a);
  const ab = angleFromDepot(depot, b);
  return aa - ab;
});

// Divide into sectors
const routes: number[][] = [];
const n = sorted.length;
const baseSize = Math.floor(n / truckCount);
let extra = n % truckCount;
let idx = 0;

for (let i = 0; i < truckCount && idx < n; i++) {
  const size = baseSize + (extra > 0 ? 1 : 0);
  if (extra > 0) extra--;
  const group = sorted.slice(idx, idx + size);
  idx += size;
  const route = nearestNeighborTSP(depot, group);
  if (route.length > 0) {
    routes.push(route);
  }
}

const output: Output = { routes };
console.log(JSON.stringify(output));
```
