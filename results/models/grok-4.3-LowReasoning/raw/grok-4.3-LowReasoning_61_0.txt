```typescript
declare const require: any;
const fs = require('fs');

interface Package {
  id: number;
  x: number;
  y: number;
}

interface Input {
  depot: [number, number];
  truckCount: number;
  packages: Package[];
}

interface Output {
  routes: number[][];
}

const input: string = fs.readFileSync(0, 'utf8');
const data: Input = JSON.parse(input);

const [dx, dy] = data.depot;
const truckCount = Math.max(1, data.truckCount);
const pkgs = data.packages;

if (pkgs.length === 0) {
  console.log(JSON.stringify({ routes: [] }));
} else {
  const withAngle = pkgs.map(p => {
    const angle = Math.atan2(p.y - dy, p.x - dx);
    return { ...p, angle };
  });

  withAngle.sort((a, b) => a.angle - b.angle);

  const routes: number[][] = Array.from({ length: truckCount }, () => []);
  for (let i = 0; i < withAngle.length; i++) {
    const idx = i % truckCount;
    routes[idx].push(withAngle[i].id);
  }

  const nonEmptyRoutes = routes.filter(r => r.length > 0);
  const output: Output = { routes: nonEmptyRoutes };
  console.log(JSON.stringify(output));
}
```
