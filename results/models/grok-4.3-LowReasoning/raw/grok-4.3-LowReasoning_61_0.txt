```typescript
declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Pkg {
  id: number;
  x: number;
  y: number;
}

const depot: [number, number] = data.depot;
const truckCount: number = data.truckCount;
const packages: Pkg[] = data.packages;

function getAngle(pkg: Pkg): number {
  const dx = pkg.x - depot[0];
  const dy = pkg.y - depot[1];
  return Math.atan2(dy, dx);
}

if (packages.length === 0) {
  console.log(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const sorted = [...packages].sort((a, b) => getAngle(a) - getAngle(b));

const numRoutes = Math.min(truckCount, sorted.length);
const routes: number[][] = [];
const chunkSize = Math.ceil(sorted.length / numRoutes);

for (let i = 0; i < numRoutes; i++) {
  const start = i * chunkSize;
  const end = Math.min(start + chunkSize, sorted.length);
  if (start < end) {
    const chunk = sorted.slice(start, end);
    routes.push(chunk.map(p => p.id));
  }
}

console.log(JSON.stringify({ routes }));
```
