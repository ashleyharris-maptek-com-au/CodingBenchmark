```rust
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    
    // Read first line: SV NP
    let first: Vec<usize> = lines.next().unwrap().unwrap()
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let _sv = first[0];
    let np = first[1];
    
    // Skip stock vertices
    for _ in 0.._sv {
        lines.next();
    }
    
    // For each piece, skip its vertices
    for _ in 0..np {
        let pv: usize = lines.next().unwrap().unwrap().parse().unwrap();
        for _ in 0..pv {
            lines.next();
        }
    }
    
    // Trivial solution: each piece on its own stock
    println!("{}", np);
    for i in 0..np {
        println!("{} 0.0 0.0 0.0", i);
    }
}
```
