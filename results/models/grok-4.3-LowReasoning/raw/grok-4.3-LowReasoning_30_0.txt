```python
import sys
import random

def read_input():
    line = sys.stdin.readline().strip()
    if not line:
        return 0, 0, []
    parts = list(map(int, line.split()))
    if len(parts) != 2:
        return 0, 0, []
    N, M = parts
    clauses = []
    for _ in range(M):
        line = sys.stdin.readline().strip()
        if not line:
            break
        lits = list(map(int, line.split()))
        if len(lits) >= 3:
            clauses.append(lits[:3])
    return N, M, clauses

def check_satisfied(assignment, clauses):
    for clause in clauses:
        satisfied = False
        for lit in clause:
            var = abs(lit) - 1
            val = assignment[var]
            if (lit > 0 and val) or (lit < 0 and not val):
                satisfied = True
                break
        if not satisfied:
            return False
    return True

def solve(N, clauses):
    assignment = [random.choice([True, False]) for _ in range(N)]
    max_flips = 100000
    for _ in range(max_flips):
        if check_satisfied(assignment, clauses):
            return assignment
        # Find unsatisfied clause
        unsat = []
        for i, clause in enumerate(clauses):
            sat = False
            for lit in clause:
                var = abs(lit) - 1
                val = assignment[var]
                if (lit > 0 and val) or (lit < 0 and not val):
                    sat = True
                    break
            if not sat:
                unsat.append(i)
        if not unsat:
            return assignment
        # Pick random unsatisfied clause and flip random var
        cidx = random.choice(unsat)
        clause = clauses[cidx]
        var_to_flip = abs(random.choice(clause)) - 1
        assignment[var_to_flip] = not assignment[var_to_flip]
    # Final check
    if check_satisfied(assignment, clauses):
        return assignment
    # Fallback random tries
    for _ in range(100):
        assignment = [random.choice([True, False]) for _ in range(N)]
        if check_satisfied(assignment, clauses):
            return assignment
    return assignment  # Return last anyway (instances guaranteed satisfiable)

def main():
    N, M, clauses = read_input()
    if N == 0:
        print("SAT")
        print("")
        return
    assignment = solve(N, clauses)
    print("SAT")
    print(" ".join('1' if x else '0' for x in assignment))

if __name__ == "__main__":
    main()
```
