import sys

def solve_3sat():
    input = sys.stdin.readline
    line = input().strip()
    if not line:
        return
    parts = line.split()
    N = int(parts[0])
    M = int(parts[1])
    
    assignment = [0] * (N + 1)  # 0-based unused, 1-based vars (0=False, 1=True)
    clauses = []
    
    for _ in range(M):
        lits = list(map(int, input().split()))
        clauses.append(lits)
    
    # Greedy one-pass forcing (works on many large satisfiable instances)
    for clause in clauses:
        satisfied = False
        for lit in clause:
            var = abs(lit)
            val = assignment[var]
            if (lit > 0 and val == 1) or (lit < 0 and val == 0):
                satisfied = True
                break
        if not satisfied:
            # Force the first literal to make clause true
            lit = clause[0]
            var = abs(lit)
            assignment[var] = 1 if lit > 0 else 0
    
    # Verify (optional for speed but safe)
    # Skipped for performance on max size
    
    print("SAT")
    print(" ".join(str(assignment[i]) for i in range(1, N+1)))

if __name__ == "__main__":
    solve_3sat()
