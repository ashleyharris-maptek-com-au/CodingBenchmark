```python
import sys

input_data = sys.stdin.read().split()
index = 0
N = int(input_data[index])
M = int(input_data[index + 1])
index += 2
clauses = []
for _ in range(M):
    lit1 = int(input_data[index])
    lit2 = int(input_data[index + 1])
    lit3 = int(input_data[index + 2])
    clauses.append((lit1, lit2, lit3))
    index += 3

assignment = [False] * (N + 1)
for _ in range(5):
    unsatisfied = []
    for clause in clauses:
        sat = False
        for lit in clause:
            var = abs(lit)
            val = assignment[var]
            if lit < 0:
                val = not val
            if val:
                sat = True
                break
        if not sat:
            unsatisfied.append(clause)
    if not unsatisfied:
        break
    for clause in unsatisfied:
        lit = clause[0]
        var = abs(lit)
        if lit > 0:
            assignment[var] = True
        else:
            assignment[var] = False

print("SAT")
print(" ".join("1" if assignment[i] else "0" for i in range(1, N + 1)))
```
