```javascript
const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const machineCount = data.machineCount;
const tasks = data.tasks;

const taskMap = new Map();
for (const t of tasks) {
  taskMap.set(t.id, t);
}

const indegree = new Map();
const successors = new Map();
for (const t of tasks) {
  indegree.set(t.id, t.deps.length);
  successors.set(t.id, []);
}
for (const t of tasks) {
  for (const dep of t.deps) {
    if (successors.has(dep)) {
      successors.get(dep).push(t.id);
    }
  }
}

const ready = [];
for (const t of tasks) {
  if (indegree.get(t.id) === 0) {
    ready.push(t.id);
  }
}

function priority(id) {
  const t = taskMap.get(id);
  return t.deadline / Math.max(1, t.weight);
}

const order = [];
while (ready.length > 0) {
  ready.sort((a, b) => priority(a) - priority(b));
  const current = ready.shift();
  order.push(current);
  for (const succ of successors.get(current)) {
    indegree.set(succ, indegree.get(succ) - 1);
    if (indegree.get(succ) === 0) {
      ready.push(succ);
    }
  }
}

if (order.length !== tasks.length) {
  console.error("Cycle detected");
  process.exit(1);
}

const output = { order: order };
console.log(JSON.stringify(output));
```
