```rust
use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::Duration;

#[derive(Clone, Copy, Default)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    ang_vel: f64,
    fuel: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Parse initial header (width height gravity max_thrust fuel)
    let header: Vec<f64> = lines
        .next()
        .unwrap()
        .unwrap()
        .split_whitespace()
        .filter_map(|s| s.parse().ok())
        .collect();
    let _gravity = header.get(2).copied().unwrap_or(1.0);

    // Parse start/target
    let pos_line: Vec<f64> = lines
        .next()
        .unwrap()
        .unwrap()
        .split_whitespace()
        .filter_map(|s| s.parse().ok())
        .collect();
    let target_x = pos_line.get(2).copied().unwrap_or(0.0);
    let target_y = pos_line.get(3).copied().unwrap_or(0.0);

    // Skip map rows (read until we see numeric state lines)
    let current_state = Arc::new(Mutex::new(State::default()));
    let has_state = Arc::new(Mutex::new(false));
    let state_clone = current_state.clone();
    let has_clone = has_state.clone();

    thread::spawn(move || {
        let stdin = io::stdin();
        let handle = stdin.lock();
        for line in handle.lines() {
            if let Ok(l) = line {
                let nums: Vec<f64> = l
                    .split_whitespace()
                    .filter_map(|s| s.parse::<f64>().ok())
                    .collect();
                if nums.len() == 7 {
                    let mut st = state_clone.lock().unwrap();
                    *st = State {
                        x: nums[0],
                        y: nums[1],
                        vx: nums[2],
                        vy: nums[3],
                        angle: nums[4],
                        ang_vel: nums[5],
                        fuel: nums[6],
                    };
                    *has_clone.lock().unwrap() = true;
                }
            }
        }
    });

    let mut stdout = io::stdout();
    let mut last_thrust = 0.0f64;
    let mut last_turn = 0.0f64;

    loop {
        let ready = *has_state.lock().unwrap();
        if ready {
            let s = *current_state.lock().unwrap();

            // Simple PD controller for soft landing on target
            let ex = target_x - s.x;
            let ey = target_y - s.y;

            // Desired vertical velocity: slow descent near target
            let desired_vy = if ey < 30.0 { 0.8 } else { 2.5 };
            let vy_err = desired_vy - s.vy;

            // Horizontal correction
            let desired_vx = (ex * 0.15).clamp(-4.0, 4.0);
            let vx_err = desired_vx - s.vx;

            // Compute desired angle (0 = up). Positive angle turns clockwise.
            let desired_angle = (vx_err * 0.4).clamp(-0.8, 0.8);

            // Turn command
            let angle_err = desired_angle - s.angle;
            let turn = (angle_err * 2.5 - s.ang_vel * 0.8).clamp(-1.0, 1.0);

            // Thrust: counteract gravity + correct vy
            let thrust = (0.65 + vy_err * 0.25).clamp(0.05, 0.95);

            last_thrust = thrust;
            last_turn = turn;
        }

        // Output at ~25 Hz
        let _ = writeln!(stdout, "{:.4} {:.4}", last_thrust, last_turn);
        let _ = stdout.flush();
        thread::sleep(Duration::from_millis(40));
    }
}
```
