```rust
use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::Duration;

#[derive(Clone, Copy, Default)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Parse initial header
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let width: usize = parts[0].parse().unwrap();
    let height: usize = parts[1].parse().unwrap();
    let gravity: f64 = parts[2].parse().unwrap();
    let max_thrust: f64 = parts[3].parse().unwrap();
    let _fuel_capacity: f64 = parts[4].parse().unwrap();

    let second_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = second_line.split_whitespace().collect();
    let _start_x: f64 = parts[0].parse().unwrap();
    let _start_y: f64 = parts[1].parse().unwrap();
    let target_x: f64 = parts[2].parse().unwrap();
    let target_y: f64 = parts[3].parse().unwrap();

    // Read map (skip storage for memory efficiency on large maps)
    for _ in 0..height {
        if let Some(Ok(_)) = lines.next() {}
    }

    let state = Arc::new(Mutex::new(State::default()));
    let state_clone = Arc::clone(&state);

    // Reader thread
    thread::spawn(move || {
        for line in lines {
            if let Ok(line) = line {
                let parts: Vec<&str> = line.split_whitespace().collect();
                if parts.len() >= 7 {
                    let mut s = state_clone.lock().unwrap();
                    s.x = parts[0].parse().unwrap_or(s.x);
                    s.y = parts[1].parse().unwrap_or(s.y);
                    s.vx = parts[2].parse().unwrap_or(s.vx);
                    s.vy = parts[3].parse().unwrap_or(s.vy);
                    s.angle = parts[4].parse().unwrap_or(s.angle);
                    s.angular_vel = parts[5].parse().unwrap_or(s.angular_vel);
                    s.fuel = parts[6].parse().unwrap_or(s.fuel);
                }
            }
        }
    });

    // Control loop at ~50 Hz
    let mut stdout = io::stdout();
    loop {
        thread::sleep(Duration::from_millis(20));

        let s = {
            let lock = state.lock().unwrap();
            *lock
        };

        if s.fuel <= 0.0 {
            continue;
        }

        // Simple PID-like controller
        let dx = target_x - s.x;
        let dy = target_y - s.y;

        // Target descent rate: gentle when close
        let target_vy = if dy < 30.0 { -1.5 } else { -3.0 };

        // Vertical error
        let vy_err = s.vy - target_vy;
        let mut thrust = 0.6 + vy_err * 0.15;
        if dy < 10.0 {
            thrust = (thrust * 0.6).max(0.3);
        }
        thrust = thrust.clamp(0.0, 1.0);

        // Horizontal correction via angle
        let target_vx = (dx * 0.08).clamp(-4.0, 4.0);
        let vx_err = s.vx - target_vx;

        // Desired angle (0 = up)
        let mut desired_angle = -vx_err * 0.25;
        desired_angle = desired_angle.clamp(-0.8, 0.8);

        // Simple turn rate towards desired angle
        let angle_err = desired_angle - s.angle;
        let turn = (angle_err * 2.5).clamp(-1.0, 1.0);

        // Final safety clamp near ground
        if dy < 5.0 && s.vy > -2.0 {
            thrust = thrust.min(0.4);
        }

        writeln!(stdout, "{:.3} {:.3}", thrust, turn).ok();
        stdout.flush().ok();
    }
}
```
