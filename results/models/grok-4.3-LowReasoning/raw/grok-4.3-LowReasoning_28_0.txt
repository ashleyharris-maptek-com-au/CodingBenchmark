```rust
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let edge_line = lines.next().unwrap().unwrap();
        let mut edge_parts = edge_line.split_whitespace();
        let u: usize = edge_parts.next().unwrap().parse().unwrap();
        let v: usize = edge_parts.next().unwrap().parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
    }
    let mut used = vec![false; n];
    let mut cover: Vec<usize> = Vec::new();
    for u in 0..n {
        if used[u] {
            continue;
        }
        for &v in &adj[u] {
            if !used[v] {
                used[u] = true;
                used[v] = true;
                cover.push(u);
                cover.push(v);
                break;
            }
        }
    }
    println!("{}", cover.len());
    for (i, &v) in cover.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
```
