```rust
use std::io::{self, BufRead, Write};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();

    let mut in_cover = vec![false; n];

    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut ep = line.split_whitespace();
        let u: usize = ep.next().unwrap().parse().unwrap();
        let v: usize = ep.next().unwrap().parse().unwrap();
        if u < n && v < n && !in_cover[u] && !in_cover[v] {
            in_cover[u] = true;
            in_cover[v] = true;
        }
    }

    let mut cover: Vec<usize> = Vec::new();
    for (i, &covered) in in_cover.iter().enumerate() {
        if covered {
            cover.push(i);
        }
    }

    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());
    writeln!(out, "{}", cover.len()).unwrap();
    for (i, v) in cover.iter().enumerate() {
        if i > 0 {
            write!(out, " ").unwrap();
        }
        write!(out, "{}", v).unwrap();
    }
    writeln!(out).unwrap();
}
```
