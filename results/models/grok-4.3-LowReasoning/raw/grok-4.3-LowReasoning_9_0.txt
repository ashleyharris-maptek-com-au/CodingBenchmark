using System;
using System.Collections.Generic;
using System.Linq;
using System.IO;

public class Program
{
    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split();
        int N = int.Parse(firstLine[0]);
        long L = long.Parse(firstLine[1]);

        long[] lengths = new long[N];
        string[] secondLine = Console.ReadLine().Split();
        for (int i = 0; i < N; i++)
        {
            lengths[i] = long.Parse(secondLine[i]);
        }

        var items = new List<(long len, int idx)>();
        for (int i = 0; i < N; i++)
        {
            if (lengths[i] > L)
            {
                // Invalid but per constraints shouldn't happen, skip or handle
                continue;
            }
            items.Add((lengths[i], i));
        }

        items.Sort((a, b) => b.len.CompareTo(a.len));

        var stocks = new List<(long remain, List<int> cuts)>();
        foreach (var item in items)
        {
            bool placed = false;
            for (int s = 0; s < stocks.Count; s++)
            {
                if (stocks[s].remain >= item.len)
                {
                    stocks[s].cuts.Add(item.idx);
                    stocks[s] = (stocks[s].remain - item.len, stocks[s].cuts);
                    placed = true;
                    break;
                }
            }
            if (!placed)
            {
                stocks.Add((L - item.len, new List<int> { item.idx }));
            }
        }

        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            Console.WriteLine(string.Join(" ", stock.cuts));
        }
    }
}
