```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.IO;
using System.Text;

class Program
{
    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split();
        int N = int.Parse(firstLine[0]);
        long stock = long.Parse(firstLine[1]);

        string[] cutsStr = Console.ReadLine().Split();
        var items = new List<(long length, int index)>(N);
        for (int i = 0; i < N; i++)
        {
            long len = long.Parse(cutsStr[i]);
            if (len > stock) { /* invalid but assume not */ }
            items.Add((len, i));
        }

        items.Sort((a, b) => b.length.CompareTo(a.length));

        var stocks = new List<List<int>>();
        long currentRem = 0;
        List<int> current = null;

        foreach (var item in items)
        {
            if (current == null || item.length > currentRem)
            {
                if (current != null) stocks.Add(current);
                current = new List<int>();
                currentRem = stock;
            }
            current.Add(item.index);
            currentRem -= item.length;
        }
        if (current != null) stocks.Add(current);

        Console.WriteLine(stocks.Count);
        foreach (var stockCuts in stocks)
        {
            Console.WriteLine(string.Join(" ", stockCuts));
        }
    }
}
```
