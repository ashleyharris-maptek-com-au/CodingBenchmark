```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

class Program
{
    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split();
        int N = int.Parse(firstLine[0]);
        int stockLength = int.Parse(firstLine[1]);

        string[] cutStr = Console.ReadLine().Split();
        var items = new List<(int length, int index)>(N);
        for (int i = 0; i < N; i++)
        {
            int len = int.Parse(cutStr[i]);
            if (len > stockLength)
            {
                Console.WriteLine("Invalid cut length exceeds stock");
                return;
            }
            items.Add((len, i));
        }

        items.Sort((a, b) => b.length.CompareTo(a.length));

        var bins = new List<(int remaining, List<int> cuts)>();
        foreach (var item in items)
        {
            bool placed = false;
            for (int b = 0; b < bins.Count; b++)
            {
                if (bins[b].remaining >= item.length)
                {
                    bins[b].cuts.Add(item.index);
                    bins[b] = (bins[b].remaining - item.length, bins[b].cuts);
                    placed = true;
                    break;
                }
            }
            if (!placed)
            {
                var newCuts = new List<int> { item.index };
                bins.Add((stockLength - item.length, newCuts));
            }
        }

        var sb = new StringBuilder();
        sb.AppendLine(bins.Count.ToString());
        foreach (var bin in bins)
        {
            sb.AppendLine(string.Join(" ", bin.cuts));
        }
        Console.Write(sb.ToString());
    }
}
```
