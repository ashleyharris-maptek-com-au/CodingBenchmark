```python
import sys
import math

def read_input():
    lines = sys.stdin.read().strip().split('\n')
    first = lines[0].split()
    N = int(first[0])
    K = int(first[1])
    CAPACITY = float(first[2])
    depot = tuple(map(float, lines[1].split()))
    customers = []
    demands = []
    for i in range(2, 2 + N):
        parts = lines[i].split()
        x, y, d = float(parts[0]), float(parts[1]), float(parts[2])
        customers.append((x, y))
        demands.append(d)
    return N, K, CAPACITY, depot, customers, demands

def distance(p1, p2):
    return math.hypot(p1[0] - p2[0], p1[1] - p2[1])

def solve_cvrp(N, K, CAPACITY, depot, customers, demands):
    visited = [False] * N
    routes = []
    total_distance = 0.0
    for _ in range(K):
        if all(visited):
            break
        route = []
        current_load = 0.0
        current_pos = depot
        route_distance = 0.0
        while True:
            best_i = -1
            best_d = float('inf')
            for i in range(N):
                if not visited[i] and current_load + demands[i] <= CAPACITY:
                    d = distance(current_pos, customers[i])
                    if d < best_d:
                        best_d = d
                        best_i = i
            if best_i == -1:
                break
            visited[best_i] = True
            route.append(best_i)
            route_distance += best_d
            current_load += demands[best_i]
            current_pos = customers[best_i]
        if route:
            route_distance += distance(current_pos, depot)
            routes.append(route)
            total_distance += route_distance
    return total_distance, routes

def main():
    N, K, CAPACITY, depot, customers, demands = read_input()
    total_distance, routes = solve_cvrp(N, K, CAPACITY, depot, customers, demands)
    print(total_distance)
    print(len(routes))
    for route in routes:
        print(' '.join(map(str, route)))

if __name__ == "__main__":
    main()
```
