import sys
import math
from itertools import combinations

def read_input():
    lines = sys.stdin.read().strip().split('\n')
    first = lines[0].split()
    N = int(first[0])
    K = int(first[1])
    CAP = float(first[2])
    depot = list(map(float, lines[1].split()))
    customers = []
    demands = []
    for i in range(N):
        parts = lines[2 + i].split()
        customers.append((float(parts[0]), float(parts[1])))
        demands.append(float(parts[2]))
    return N, K, CAP, depot, customers, demands

def euclid(a, b):
    return math.hypot(a[0] - b[0], a[1] - b[1])

def solve_cvrp(N, K, CAP, depot, customers, demands):
    points = [depot] + customers
    dist = [[0.0] * (N + 1) for _ in range(N + 1)]
    for i in range(N + 1):
        for j in range(N + 1):
            dist[i][j] = euclid(points[i], points[j])
    
    # Clarke-Wright savings
    savings = []
    for i, j in combinations(range(1, N + 1), 2):
        sav = dist[0][i] + dist[0][j] - dist[i][j]
        savings.append((sav, i, j))
    savings.sort(reverse=True)
    
    # Initial singletons
    routes = [[i] for i in range(1, N + 1)]
    route_demand = [demands[i-1] for i in range(N)]
    route_load = route_demand[:]
    
    parent = list(range(N))
    
    def find(x):
        while parent[x] != x:
            parent[x] = parent[parent[x]]
            x = parent[x]
        return x
    
    for sav, i, j in savings:
        ri = find(i - 1)
        rj = find(j - 1)
        if ri == rj:
            continue
        # Check merge feasibility
        new_load = route_load[ri] + route_load[rj]
        if new_load > CAP:
            continue
        # Merge: attach rj to ri
        parent[rj] = ri
        route_load[ri] = new_load
        # Rebuild actual routes later
    
    # Collect final routes
    route_map = {}
    for idx in range(N):
        r = find(idx)
        if r not in route_map:
            route_map[r] = []
        route_map[r].append(idx + 1)
    
    final_routes = []
    for r, custs in route_map.items():
        # Simple ordering: sort by angle from depot for better shape
        custs.sort(key=lambda c: math.atan2(customers[c-1][1] - depot[1], customers[c-1][0] - depot[0]))
        final_routes.append(custs)
    
    # Limit to K routes if more
    while len(final_routes) > K:
        # Merge two smallest load routes
        final_routes.sort(key=lambda r: sum(demands[c-1] for c in r))
        r1 = final_routes.pop(0)
        r2 = final_routes.pop(0)
        merged = r1 + r2
        merged.sort(key=lambda c: math.atan2(customers[c-1][1] - depot[1], customers[c-1][0] - depot[0]))
        final_routes.append(merged)
    
    # Compute total distance
    total = 0.0
    for route in final_routes:
        if not route:
            continue
        total += dist[0][route[0]]
        for a, b in zip(route, route[1:]):
            total += dist[a][b]
        total += dist[route[-1]][0]
    
    return total, final_routes

def main():
    N, K, CAP, depot, customers, demands = read_input()
    total, routes = solve_cvrp(N, K, CAP, depot, customers, demands)
    
    print(f"{total:.2f}")
    print(len(routes))
    for route in routes:
        print(" ".join(map(str, [c-1 for c in route])))

if __name__ == "__main__":
    main()
