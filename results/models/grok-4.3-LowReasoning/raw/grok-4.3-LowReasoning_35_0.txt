```rust
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let _m: usize = parts.next().unwrap().parse().unwrap();
    
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    
    for line in lines {
        let line = line.unwrap();
        if line.trim().is_empty() {
            continue;
        }
        let mut p = line.split_whitespace();
        let u: usize = p.next().unwrap().parse().unwrap();
        let v: usize = p.next().unwrap().parse().unwrap();
        if u < n && v < n && u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }
    
    // Dedup adjacency lists
    for i in 0..n {
        adj[i].sort_unstable();
        adj[i].dedup();
    }
    
    let mut dominated: Vec<bool> = vec![false; n];
    let mut dominating_set: Vec<usize> = Vec::new();
    
    while dominated.iter().any(|&d| !d) {
        let mut best_v = 0;
        let mut best_count = 0usize;
        
        for v in 0..n {
            let mut count = 0usize;
            if !dominated[v] {
                count += 1;
            }
            for &nb in &adj[v] {
                if !dominated[nb] {
                    count += 1;
                }
            }
            if count > best_count {
                best_count = count;
                best_v = v;
            }
        }
        
        if best_count == 0 {
            // Fallback: pick any undominated vertex
            for v in 0..n {
                if !dominated[v] {
                    dominating_set.push(v);
                    dominated[v] = true;
                    for &nb in &adj[v] {
                        dominated[nb] = true;
                    }
                    break;
                }
            }
            continue;
        }
        
        dominating_set.push(best_v);
        dominated[best_v] = true;
        for &nb in &adj[best_v] {
            dominated[nb] = true;
        }
    }
    
    println!("{}", dominating_set.len());
    if !dominating_set.is_empty() {
        println!(
            "{}",
            dominating_set
                .iter()
                .map(|x| x.to_string())
                .collect::<Vec<_>>()
                .join(" ")
        );
    } else {
        println!();
    }
}
```
