```python
import numpy as np
from collections import deque

# Module-level persistent state for filtering, PID, and history
_prev_steering = 0.0
_prev_throttle = 0.0
_prev_brake = 0.0
_lane_error_hist = deque(maxlen=8)
_speed_error_hist = deque(maxlen=8)
_last_valid_lane_offset = 0.0
_last_valid_speed = 100.0
_obstacle_persist = 0
_low_mu_persist = 0
_lane_change_timer = 0
_target_lane_persist = 0

def _median(vals):
    vals = [v for v in vals if v is not None and not np.isnan(v)]
    return float(np.median(vals)) if vals else 0.0

def _safe_get(state, key, default=0.0):
    return state.get(key, default) if state.get(key) is not None else default

def autopilot_step(state: dict, dt: float) -> dict:
    global _prev_steering, _prev_throttle, _prev_brake
    global _lane_error_hist, _speed_error_hist
    global _last_valid_lane_offset, _last_valid_speed
    global _obstacle_persist, _low_mu_persist, _lane_change_timer, _target_lane_persist

    # === Sensor Fusion ===
    # Speed fusion (wheel speeds + vehicle speed + GPS)
    wheel_speeds = [
        _safe_get(state, 'wheel_speed_fl_kmh'),
        _safe_get(state, 'wheel_speed_fr_kmh'),
        _safe_get(state, 'wheel_speed_rl_kmh'),
        _safe_get(state, 'wheel_speed_rr_kmh'),
        _safe_get(state, 'vehicle_speed_1_kmh'),
        _safe_get(state, 'vehicle_speed_2_kmh'),
        _safe_get(state, 'gps_1_speed_kmh'),
        _safe_get(state, 'gps_2_speed_kmh'),
    ]
    speed_kmh = max(0.0, _median(wheel_speeds))
    if speed_kmh < 1.0:
        speed_kmh = _last_valid_speed
    else:
        _last_valid_speed = speed_kmh

    # Yaw rate fusion (gyroscopes)
    yaw_rates = [
        _safe_get(state, 'gyro_1_yaw_rate_dps'),
        _safe_get(state, 'gyro_2_yaw_rate_dps'),
        _safe_get(state, 'gyro_3_yaw_rate_dps'),
    ]
    yaw_rate = _median(yaw_rates)

    # Lane / vision fusion (majority vote + median)
    lane_offsets = []
    left_dists = []
    right_dists = []
    curvatures = []
    speed_limits = []
    obstacles = []
    obstacle_dists = []
    obstacle_rel_speeds = []

    for i in [1, 2, 3]:
        if _safe_get(state, f'vision_{i}_left_line_valid', 0) and _safe_get(state, f'vision_{i}_right_line_valid', 0):
            lane_offsets.append(_safe_get(state, f'vision_{i}_lane_offset_m'))
            left_dists.append(_safe_get(state, f'vision_{i}_left_line_dist_m'))
            right_dists.append(_safe_get(state, f'vision_{i}_right_line_dist_m'))
            curvatures.append(_safe_get(state, f'vision_{i}_road_curvature_1pm'))
            if _safe_get(state, f'vision_{i}_speed_limit_valid', 0):
                speed_limits.append(_safe_get(state, f'vision_{i}_speed_limit_kmh'))

        if _safe_get(state, f'vision_{i}_obstacle_valid', 0) and _safe_get(state, f'vision_{i}_obstacle_detected', 0):
            obstacles.append(True)
            obstacle_dists.append(_safe_get(state, f'vision_{i}_obstacle_dist_m'))
            obstacle_rel_speeds.append(_safe_get(state, f'vision_{i}_obstacle_rel_speed_kmh'))

    # Lidar front fusion
    front_ranges = [
        _safe_get(state, 'lidar_+005_deg_range_m'),
        _safe_get(state, 'lidar_+015_deg_range_m'),
        _safe_get(state, 'lidar_-005_deg_range_m'),
        _safe_get(state, 'lidar_-015_deg_range_m'),
    ]
    min_front_dist = min([r for r in front_ranges if r > 0.1] or [200.0])

    # === State Estimation ===
    lane_offset = _median(lane_offsets) if lane_offsets else _last_valid_lane_offset
    if abs(lane_offset) > 4.0:
        lane_offset = _last_valid_lane_offset
    else:
        _last_valid_lane_offset = lane_offset

    curvature = _median(curvatures) if curvatures else 0.0
    speed_limit = min(speed_limits) if speed_limits else 130.0

    target_speed = min(_safe_get(state, 'target_speed_kmh', 120.0), speed_limit)
    target_lane = int(_safe_get(state, 'target_lane', 1))

    mu = _safe_get(state, 'estimated_mu', 1.0)
    if mu < 0.5:
        _low_mu_persist = 20
    if _low_mu_persist > 0:
        _low_mu_persist -= 1
        target_speed = min(target_speed, 70.0)

    # === Obstacle Detection ===
    obstacle_ahead = False
    if obstacles and min(obstacle_dists) < 80.0:
        rel_speed = _median(obstacle_rel_speeds)
        if rel_speed < -5.0 or min(obstacle_dists) < 40.0:
            obstacle_ahead = True
            _obstacle_persist = 30

    if min_front_dist < 50.0:
        obstacle_ahead = True
        _obstacle_persist = 25

    if _obstacle_persist > 0:
        _obstacle_persist -= 1
        obstacle_ahead = True

    # === Control ===
    # Lane keeping PID
    lane_error = lane_offset + 0.3 * curvature * (speed_kmh / 3.6)
    _lane_error_hist.append(lane_error)
    lane_deriv = (lane_error - _lane_error_hist[0]) / max(len(_lane_error_hist) * dt, dt)
    lane_integral = sum(_lane_error_hist) * dt
    steering = -0.8 * lane_error - 0.15 * lane_deriv - 0.02 * lane_integral
    steering = np.clip(steering, -1.0, 1.0)

    # Speed control
    speed_error = (target_speed - speed_kmh) / 3.6
    _speed_error_hist.append(speed_error)
    speed_deriv = (speed_error - _speed_error_hist[0]) / max(len(_speed_error_hist) * dt, dt)

    if obstacle_ahead:
        throttle = 0.0
        brake = 0.6 if min_front_dist < 35.0 or (obstacles and min(obstacle_dists) < 25.0) else 0.3
    else:
        throttle = np.clip(0.4 + 0.6 * speed_error + 0.1 * speed_deriv, 0.0, 1.0)
        brake = 0.0
        if speed_kmh > target_speed + 5.0:
            throttle = 0.0
            brake = 0.2

    # Low mu / ice handling
    if _low_mu_persist > 0:
        throttle = min(throttle, 0.4)
        steering *= 0.6

    # Gentle smoothing
    steering = 0.7 * _prev_steering + 0.3 * steering
    throttle = 0.8 * _prev_throttle + 0.2 * throttle
    brake = 0.7 * _prev_brake + 0.3 * brake

    _prev_steering = float(np.clip(steering, -1.0, 1.0))
    _prev_throttle = float(np.clip(throttle, 0.0, 1.0))
    _prev_brake = float(np.clip(brake, 0.0, 1.0))

    return {
        "steering": _prev_steering,
        "throttle": _prev_throttle,
        "brake": _prev_brake
    }
```
