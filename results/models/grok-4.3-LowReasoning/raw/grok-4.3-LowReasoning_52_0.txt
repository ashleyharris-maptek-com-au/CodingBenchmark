import numpy as np
from collections import deque

# Module-level persistent state
_prev_steering = 0.0
_prev_throttle = 0.0
_prev_brake = 0.0
_speed_pid = {'integral': 0.0, 'prev_error': 0.0}
_lane_pid = {'integral': 0.0, 'prev_error': 0.0}
_last_valid_lane_offset = 0.0
_last_valid_speed = 0.0
_obstacle_memory = deque(maxlen=5)
_low_mu_counter = 0

def _median(values):
    return float(np.median([v for v in values if v is not None]))

def _safe_get(state, key, default=0.0):
    val = state.get(key, default)
    return val if val is not None else default

def autopilot_step(state: dict, dt: float) -> dict:
    global _prev_steering, _prev_throttle, _prev_brake
    global _speed_pid, _lane_pid, _last_valid_lane_offset
    global _last_valid_speed, _obstacle_memory, _low_mu_counter

    if dt <= 0:
        dt = 0.05

    # === SENSOR FUSION ===
    # Vehicle speed (median of available sources)
    speeds = []
    for k in ['vehicle_speed_1_kmh', 'vehicle_speed_2_kmh', 'gps_1_speed_kmh', 'gps_2_speed_kmh']:
        if state.get(k + '_valid', True) and k in state:
            speeds.append(state[k])
    speed = _median(speeds) if speeds else _last_valid_speed
    if speed > 0:
        _last_valid_speed = speed

    # Yaw rate (median of valid gyros)
    yaws = []
    for i in [1, 2, 3]:
        if state.get(f'gyro_{i}_valid', True):
            yaws.append(state.get(f'gyro_{i}_yaw_rate_dps', 0.0))
    yaw_rate = _median(yaws) if yaws else 0.0

    # Lane offset fusion (vision systems)
    offsets = []
    for i in [1, 2, 3]:
        if state.get(f'vision_{i}_left_line_valid', True) and state.get(f'vision_{i}_right_line_valid', True):
            off = state.get(f'vision_{i}_lane_offset_m', 0.0)
            offsets.append(off)
    if offsets:
        lane_offset = _median(offsets)
        _last_valid_lane_offset = lane_offset
    else:
        lane_offset = _last_valid_lane_offset

    # Obstacle detection (any vision system + lidar confirmation)
    obstacle_dist = 100.0
    obstacle_rel_speed = 0.0
    for i in [1, 2, 3]:
        if state.get(f'vision_{i}_obstacle_valid', False) and state.get(f'vision_{i}_obstacle_detected', False):
            d = state.get(f'vision_{i}_obstacle_dist_m', 100.0)
            if d < obstacle_dist:
                obstacle_dist = d
                obstacle_rel_speed = state.get(f'vision_{i}_obstacle_rel_speed_kmh', 0.0)
    _obstacle_memory.append(obstacle_dist)
    min_obs_dist = min(_obstacle_memory) if _obstacle_memory else 100.0

    # Friction
    mu = _safe_get(state, 'estimated_mu', 1.0)
    if mu < 0.4:
        _low_mu_counter = min(_low_mu_counter + 1, 20)
    else:
        _low_mu_counter = max(_low_mu_counter - 2, 0)

    # === CONTROL TARGETS ===
    target_speed = state.get('target_speed_kmh', 100.0)
    target_lane = state.get('target_lane', 1)

    # Speed limit respect
    for i in [1, 2, 3]:
        if state.get(f'vision_{i}_speed_limit_valid', False):
            limit = state.get(f'vision_{i}_speed_limit_kmh', 120)
            target_speed = min(target_speed, limit - 5)

    # === LANE KEEPING (simple P + I) ===
    lane_error = lane_offset
    _lane_pid['integral'] = np.clip(_lane_pid['integral'] + lane_error * dt, -2.0, 2.0)
    steering = 0.6 * lane_error + 0.1 * _lane_pid['integral'] + 0.2 * yaw_rate / 30.0

    # === OBSTACLE RESPONSE ===
    brake_intensity = 0.0
    if min_obs_dist < 80.0:
        ttc = min_obs_dist / max((speed - obstacle_rel_speed) / 3.6, 0.1)
        if ttc < 2.5 or min_obs_dist < 25.0:
            brake_intensity = np.clip(1.2 - ttc / 3.0, 0.0, 1.0)
            steering *= 0.3  # reduce steering authority under hard braking

    # === SPEED CONTROL ===
    speed_error = target_speed - speed
    _speed_pid['integral'] = np.clip(_speed_pid['integral'] + speed_error * dt, -50, 50)
    throttle = 0.015 * speed_error + 0.002 * _speed_pid['integral']

    # === TRACTION MANAGEMENT ===
    if _low_mu_counter > 5 or mu < 0.35:
        throttle *= 0.4
        brake_intensity = min(brake_intensity, 0.6)
        steering *= 0.5

    # === FINAL OUTPUT CLAMPING ===
    steering = np.clip(steering, -1.0, 1.0)
    throttle = np.clip(throttle, 0.0, 1.0)
    brake = np.clip(brake_intensity, 0.0, 1.0)

    # Gentle rate limiting
    steering = 0.7 * steering + 0.3 * _prev_steering
    throttle = 0.6 * throttle + 0.4 * _prev_throttle
    brake = 0.5 * brake + 0.5 * _prev_brake

    _prev_steering = steering
    _prev_throttle = throttle
    _prev_brake = brake

    return {
        'steering': float(steering),
        'throttle': float(throttle),
        'brake': float(brake)
    }
