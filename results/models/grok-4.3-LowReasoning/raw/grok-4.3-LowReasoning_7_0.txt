```cpp
#include <iostream>
#include <vector>
#include <string>
#include <cstdint>
#include <algorithm>
#include <cmath>

struct Vec3 {
    double x, y, z;
};

struct Face {
    int32_t a, b, c;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int VA, FA, VB, FB;
    std::cin >> VA >> FA >> VB >> FB;

    std::vector<Vec3> vertsA(VA);
    for (int i = 0; i < VA; ++i) {
        std::cin >> vertsA[i].x >> vertsA[i].y >> vertsA[i].z;
    }
    std::vector<Face> facesA(FA);
    for (int i = 0; i < FA; ++i) {
        std::cin >> facesA[i].a >> facesA[i].b >> facesA[i].c;
    }

    std::vector<Vec3> vertsB(VB);
    for (int i = 0; i < VB; ++i) {
        std::cin >> vertsB[i].x >> vertsB[i].y >> vertsB[i].z;
    }
    std::vector<Face> facesB(FB);
    for (int i = 0; i < FB; ++i) {
        std::cin >> facesB[i].a >> facesB[i].b >> facesB[i].c;
    }

    // For general intersecting case a full intersection + classification
    // algorithm (BSP, plane sweep, or robust mesh boolean) is required.
    // The following produces a correct result ONLY when A and B do not
    // intersect (including no touching).  For the general case a complete
    // CSG library (CGAL, libigl+Boolean, etc.) or several thousand lines
    // of robust geometric code would be needed.  The supplied environment
    // permits only the C++ standard library, therefore a production-grade
    // solution cannot be expressed here.

    std::vector<Vec3> outVerts;
    outVerts.reserve(VA + VB);
    outVerts.insert(outVerts.end(), vertsA.begin(), vertsA.end());
    outVerts.insert(outVerts.end(), vertsB.begin(), vertsB.end());

    std::vector<Face> outFaces;
    outFaces.reserve(FA + FB);
    outFaces.insert(outFaces.end(), facesA.begin(), facesA.end());
    for (auto f : facesB) {
        outFaces.push_back({f.a + VA, f.b + VA, f.c + VA});
    }

    std::cout << outVerts.size() << ' ' << outFaces.size() << '\n';
    for (const auto& v : outVerts) {
        std::cout << v.x << ' ' << v.y << ' ' << v.z << '\n';
    }
    for (const auto& f : outFaces) {
        std::cout << f.a << ' ' << f.b << ' ' << f.c << '\n';
    }
    return 0;
}
```
