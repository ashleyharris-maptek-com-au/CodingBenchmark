; SPIR-V
; Version: 1.0
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %worldPos Location 0
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%in_ptr = OpTypePointer Input %v3float
%out_ptr = OpTypePointer Output %v4float
%worldPos = OpVariable %in_ptr Input
%outColor = OpVariable %out_ptr Output
%float_0 = OpConstant %float 0
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1
%bottom = OpConstantComposite %v3float %float_0_1 %float_0_4 %float_0_1
%top = OpConstantComposite %v3float %float_1 %float_1 %float_1
%main = OpFunction %void None %func
%entry = OpLabel
%pos = OpLoad %v3float %worldPos
%y = OpCompositeExtract %float %pos 1
%scaled = OpFMul %float %y %float_0_5
%biased = OpFAdd %float %scaled %float_0_5
%t = OpExtInst %float %glsl 43 %biased %float_0 %float_1
%color = OpExtInst %v3float %glsl 46 %bottom %top %t
%out = OpCompositeConstruct %v4float %color %float_1
OpStore %outColor %out
OpReturn
OpFunctionEnd
