#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <numeric>
#include <limits>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N;
    std::cin >> N;
    std::vector<std::pair<double, double>> pts(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> pts[i].first >> pts[i].second;
    }

    std::vector<int> tour;
    if (N <= 1) {
        tour.push_back(0);
    } else if (N <= 15000) {
        // Nearest Neighbor heuristic (O(N^2) feasible up to ~15k)
        std::vector<bool> visited(N, false);
        tour.reserve(N);
        tour.push_back(0);
        visited[0] = true;
        for (int i = 1; i < N; ++i) {
            int last = tour.back();
            double min_d = std::numeric_limits<double>::max();
            int next = -1;
            for (int j = 0; j < N; ++j) {
                if (!visited[j]) {
                    double dx = pts[j].first - pts[last].first;
                    double dy = pts[j].second - pts[last].second;
                    double d = dx * dx + dy * dy;
                    if (d < min_d) {
                        min_d = d;
                        next = j;
                    }
                }
            }
            tour.push_back(next);
            visited[next] = true;
        }
    } else {
        // Sort by x-coordinate then y (O(N log N))
        tour.resize(N);
        std::iota(tour.begin(), tour.end(), 0);
        std::sort(tour.begin(), tour.end(), [&](int a, int b) {
            if (pts[a].first != pts[b].first) return pts[a].first < pts[b].first;
            return pts[a].second < pts[b].second;
        });
        auto it = std::find(tour.begin(), tour.end(), 0);
        std::rotate(tour.begin(), it, tour.end());
    }

    for (int i = 0; i < N; ++i) {
        if (i > 0) std::cout << ' ';
        std::cout << tour[i];
    }
    std::cout << '\n';
    return 0;
}
