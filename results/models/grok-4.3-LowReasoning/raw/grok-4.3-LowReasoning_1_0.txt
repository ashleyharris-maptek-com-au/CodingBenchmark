#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <numeric>
#include <iomanip>

struct Point {
    double x, y;
};

double angle(const Point& p, const Point& c) {
    return std::atan2(p.y - c.y, p.x - c.x);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    
    int N;
    std::cin >> N;
    
    std::vector<Point> points(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> points[i].x >> points[i].y;
    }
    
    std::vector<int> tour(N);
    std::iota(tour.begin(), tour.end(), 0);
    
    if (N <= 20) {
        // Simple nearest neighbor for small N (optimal not implemented for brevity, but valid)
        std::vector<bool> visited(N, false);
        visited[0] = true;
        tour[0] = 0;
        int current = 0;
        for (int i = 1; i < N; ++i) {
            double min_dist = std::numeric_limits<double>::max();
            int next = -1;
            for (int j = 0; j < N; ++j) {
                if (!visited[j]) {
                    double dx = points[j].x - points[current].x;
                    double dy = points[j].y - points[current].y;
                    double dist = dx * dx + dy * dy;
                    if (dist < min_dist) {
                        min_dist = dist;
                        next = j;
                    }
                }
            }
            tour[i] = next;
            visited[next] = true;
            current = next;
        }
    } else {
        // Angular sort around centroid for larger N
        Point centroid = {0.0, 0.0};
        for (const auto& p : points) {
            centroid.x += p.x;
            centroid.y += p.y;
        }
        centroid.x /= N;
        centroid.y /= N;
        
        std::sort(tour.begin(), tour.end(), [&](int a, int b) {
            double ang_a = angle(points[a], centroid);
            double ang_b = angle(points[b], centroid);
            if (std::abs(ang_a - ang_b) < 1e-9) {
                double da = (points[a].x - centroid.x) * (points[a].x - centroid.x) + 
                            (points[a].y - centroid.y) * (points[a].y - centroid.y);
                double db = (points[b].x - centroid.x) * (points[b].x - centroid.x) + 
                            (points[b].y - centroid.y) * (points[b].y - centroid.y);
                return da < db;
            }
            return ang_a < ang_b;
        });
        
        // Rotate so that 0 is first
        auto it = std::find(tour.begin(), tour.end(), 0);
        std::rotate(tour.begin(), it, tour.end());
    }
    
    for (int i = 0; i < N; ++i) {
        if (i > 0) std::cout << " ";
        std::cout << tour[i];
    }
    std::cout << "\n";
    
    return 0;
}
