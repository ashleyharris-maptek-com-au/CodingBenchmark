```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <numeric>
#include <iomanip>

struct Point {
    double x, y;
    int id;
};

double dist(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return std::sqrt(dx*dx + dy*dy);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    
    int N;
    std::cin >> N;
    
    std::vector<Point> cities(N);
    int start_idx = -1;
    for (int i = 0; i < N; ++i) {
        std::cin >> cities[i].x >> cities[i].y;
        cities[i].id = i;
        if (i == 0) start_idx = i;
    }
    
    std::vector<int> order;
    
    if (N <= 12) {
        // Exact DP for tiny N
        const int MAX = 1 << N;
        std::vector<std::vector<double>> dp(MAX, std::vector<double>(N, 1e18));
        std::vector<std::vector<int>> prev(MAX, std::vector<int>(N, -1));
        dp[1][0] = 0;
        
        for (int mask = 0; mask < MAX; ++mask) {
            for (int u = 0; u < N; ++u) {
                if (dp[mask][u] > 1e17) continue;
                for (int v = 0; v < N; ++v) {
                    if (mask & (1 << v)) continue;
                    double nd = dp[mask][u] + dist(cities[u], cities[v]);
                    int nmask = mask | (1 << v);
                    if (nd < dp[nmask][v]) {
                        dp[nmask][v] = nd;
                        prev[nmask][v] = u;
                    }
                }
            }
        }
        
        // Reconstruct
        double best = 1e18;
        int end = -1;
        for (int u = 1; u < N; ++u) {
            if (dp[MAX-1][u] < best) {
                best = dp[MAX-1][u];
                end = u;
            }
        }
        order.resize(N);
        int mask = MAX-1;
        int cur = end;
        for (int i = N-1; i >= 0; --i) {
            order[i] = cur;
            int p = prev[mask][cur];
            mask ^= (1 << cur);
            cur = p;
        }
        // Rotate to start with 0
        auto it = std::find(order.begin(), order.end(), 0);
        std::rotate(order.begin(), it, order.end());
    } else if (N <= 2000) {
        // Nearest neighbor + simple improvement
        std::vector<bool> visited(N, false);
        order.reserve(N);
        order.push_back(0);
        visited[0] = true;
        int cur = 0;
        for (int i = 1; i < N; ++i) {
            double best_d = 1e18;
            int next = -1;
            for (int j = 0; j < N; ++j) {
                if (!visited[j]) {
                    double d = dist(cities[cur], cities[j]);
                    if (d < best_d) {
                        best_d = d;
                        next = j;
                    }
                }
            }
            order.push_back(next);
            visited[next] = true;
            cur = next;
        }
    } else {
        // Large scale: sort by x then y, rotate to start at 0
        std::sort(cities.begin(), cities.end(), [](const Point& a, const Point& b) {
            if (std::abs(a.x - b.x) > 1e-9) return a.x < b.x;
            return a.y < b.y;
        });
        
        order.resize(N);
        for (int i = 0; i < N; ++i) order[i] = cities[i].id;
        
        // Rotate so starts with 0
        auto it = std::find(order.begin(), order.end(), 0);
        std::rotate(order.begin(), it, order.end());
    }
    
    for (size_t i = 0; i < order.size(); ++i) {
        if (i > 0) std::cout << " ";
        std::cout << order[i];
    }
    std::cout << "\n";
    
    return 0;
}
```
