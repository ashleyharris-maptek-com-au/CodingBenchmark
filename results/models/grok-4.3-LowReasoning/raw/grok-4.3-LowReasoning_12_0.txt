using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

class Program
{
    static void Main(string[] args)
    {
        var culture = CultureInfo.InvariantCulture;
        var input = Console.In.ReadToEnd().Split(new[] { '\r', '\n', ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;

        int V = int.Parse(input[idx++]);
        int F = int.Parse(input[idx++]);

        double minX = double.MaxValue, minY = double.MaxValue, minZ = double.MaxValue;
        double maxX = double.MinValue, maxY = double.MinValue, maxZ = double.MinValue;

        for (int i = 0; i < V; i++)
        {
            double x = double.Parse(input[idx++], culture);
            double y = double.Parse(input[idx++], culture);
            double z = double.Parse(input[idx++], culture);
            if (x < minX) minX = x;
            if (y < minY) minY = y;
            if (z < minZ) minZ = z;
            if (x > maxX) maxX = x;
            if (y > maxY) maxY = y;
            if (z > maxZ) maxZ = z;
        }

        // Skip faces
        idx += F * 3;

        double W = double.Parse(input[idx++], culture);
        double H = double.Parse(input[idx++], culture);
        double D = double.Parse(input[idx++], culture);

        double dx = maxX - minX;
        double dy = maxY - minY;
        double dz = maxZ - minZ;

        if (dx <= 0 || dy <= 0 || dz <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        // Adaptive strategy based on scale
        int scale = (int)Math.Max(Math.Max(W, H), D);
        bool useRotation = scale < 50; // Try basic rotations only for small/medium

        List<string> placements = new List<string>();

        if (useRotation)
        {
            // For smaller scales, try identity and one 90 deg Z rotation
            TryPack(W, H, D, dx, dy, dz, minX, minY, minZ, 1, 0, 0, 0, placements);
            TryPack(W, H, D, dy, dx, dz, minY, minX, minZ, 0.70710678118, 0, 0, 0.70710678118, placements);
        }
        else
        {
            // Grid packing for large/extreme scales - fast and scalable
            TryPack(W, H, D, dx, dy, dz, minX, minY, minZ, 1, 0, 0, 0, placements);
        }

        Console.WriteLine(placements.Count);
        foreach (var p in placements)
            Console.WriteLine(p);
    }

    static void TryPack(double W, double H, double D, double dx, double dy, double dz,
        double minX, double minY, double minZ, double qw, double qx, double qy, double qz,
        List<string> placements)
    {
        if (dx > W || dy > H || dz > D) return;

        int nx = (int)Math.Floor(W / dx);
        int ny = (int)Math.Floor(H / dy);
        int nz = (int)Math.Floor(D / dz);

        if (nx <= 0 || ny <= 0 || nz <= 0) return;

        for (int ix = 0; ix < nx; ix++)
        {
            for (int iy = 0; iy < ny; iy++)
            {
                for (int iz = 0; iz < nz; iz++)
                {
                    double tx = ix * dx - minX;
                    double ty = iy * dy - minY;
                    double tz = iz * dz - minZ;

                    // Ensure fit
                    if (tx + minX >= 0 && tx + dx <= W &&
                        ty + minY >= 0 && ty + dy <= H &&
                        tz + minZ >= 0 && tz + dz <= D)
                    {
                        placements.Add(string.Format(CultureInfo.InvariantCulture,
                            "{0:F6} {1:F6} {2:F6} {3:F6} {4:F6} {5:F6} {6:F6}",
                            tx, ty, tz, qw, qx, qy, qz));
                    }
                }
            }
        }
    }
}
