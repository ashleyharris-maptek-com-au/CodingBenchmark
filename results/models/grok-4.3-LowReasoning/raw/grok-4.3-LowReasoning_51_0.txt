import math
from collections import deque

# Module-level persistence for PIDs, filters, and state
_prev = {}
_integrals = {'alt': 0.0, 'vs': 0.0, 'hdg': 0.0, 'ias': 0.0}
_filters = {}
_last_valid = {}
_flight_phase = 'cruise'
_cmd_history = deque(maxlen=20)

def _median(vals):
    s = sorted([v for v in vals if v is not None])
    n = len(s)
    if n == 0:
        return None
    return s[n//2] if n % 2 else (s[n//2-1] + s[n//2]) / 2

def _safe_get(state, key, default=0.0):
    return state.get(key, default)

def _fuse_triple(state, base):
    v1 = _safe_get(state, f"{base}_1")
    v2 = _safe_get(state, f"{base}_2")
    v3 = _safe_get(state, f"{base}_3")
    valid1 = _safe_get(state, f"{base}_1_valid", True)
    valid2 = _safe_get(state, f"{base}_2_valid", True)
    valid3 = _safe_get(state, f"{base}_3_valid", True)
    
    candidates = []
    if valid1: candidates.append(v1)
    if valid2: candidates.append(v2)
    if valid3: candidates.append(v3)
    
    if len(candidates) >= 2:
        return _median(candidates)
    elif candidates:
        return candidates[0]
    else:
        # All invalid - use last known good or raw median
        return _median([v1, v2, v3]) or _last_valid.get(base, 0.0)

def autopilot_step(state: dict, dt: float) -> dict:
    global _prev, _integrals, _filters, _last_valid, _flight_phase
    
    # === SENSOR FUSION ===
    alt = _fuse_triple(state, 'baro_alt') or _fuse_triple(state, 'gps_alt') or 30000.0
    ias = _fuse_triple(state, 'ias') or 250.0
    mach = _fuse_triple(state, 'mach') or 0.7
    vs = _fuse_triple(state, 'irs_vs') or 0.0
    hdg = _fuse_triple(state, 'irs_heading') or _safe_get(state, 'magnetic_heading_deg', 0.0)
    pitch = _fuse_triple(state, 'irs_pitch') or 0.0
    roll = _fuse_triple(state, 'irs_roll') or 0.0
    aoa = _fuse_triple(state, 'aoa') or 4.0
    
    target_alt = _safe_get(state, 'target_alt_ft', alt)
    target_hdg = _safe_get(state, 'target_hdg_deg', hdg)
    target_ias = _safe_get(state, 'target_ias_kt', ias)
    
    # Update last valid
    _last_valid['alt'] = alt
    _last_valid['ias'] = ias
    
    # === BASIC PROTECTIONS ===
    cmds = {}
    
    # Speed protection (Vne ~365 KIAS / M0.82)
    if ias > 355 or mach > 0.80:
        cmds['speedbrake'] = min(1.0, (ias - 355) / 30)
        cmds['throttle_1'] = 0.3
        cmds['throttle_2'] = 0.3
    else:
        cmds['speedbrake'] = 0.0
    
    # Stall protection
    if aoa > 12.0:
        cmds['elevator'] = -0.6
        cmds['throttle_1'] = 0.95
        cmds['throttle_2'] = 0.95
    
    # === ALTITUDE / PITCH CONTROL ===
    alt_err = target_alt - alt
    vs_target = max(-1500, min(1500, alt_err * 0.8))
    
    # Simple VS hold with pitch
    vs_err = vs_target - vs
    _integrals['vs'] = max(-5, min(5, _integrals['vs'] + vs_err * dt * 0.001))
    pitch_cmd = 0.015 * vs_err + 0.002 * _integrals['vs']
    
    # Pitch limiter
    pitch_cmd = max(-8, min(12, pitch_cmd))
    
    # Convert to elevator (relative to trim)
    elevator = (pitch_cmd - pitch) * 0.12
    elevator = max(-1.0, min(1.0, elevator))
    cmds['elevator'] = elevator
    
    # === HEADING / ROLL CONTROL ===
    hdg_err = ((target_hdg - hdg + 180) % 360) - 180
    roll_cmd = max(-25, min(25, hdg_err * 1.8))
    aileron = (roll_cmd - roll) * 0.08
    aileron = max(-1.0, min(1.0, aileron))
    cmds['aileron'] = aileron
    
    # Rudder - basic yaw damping
    yaw_rate = _fuse_triple(state, 'gyro_yaw_rate') or 0.0
    cmds['rudder'] = -yaw_rate * 0.15
    
    # === AUTOTHRUST ===
    ias_err = target_ias - ias
    _integrals['ias'] += ias_err * dt * 0.0008
    _integrals['ias'] = max(-20, min(20, _integrals['ias']))
    
    thr = 0.55 + ias_err * 0.004 + _integrals['ias'] * 0.0005
    thr = max(0.15, min(0.95, thr))
    
    if 'speedbrake' not in cmds or cmds['speedbrake'] < 0.3:
        cmds['throttle_1'] = thr
        cmds['throttle_2'] = thr
    
    # === CONFIG ===
    # Simple flap schedule
    if alt < 2000:
        cmds['flaps'] = 20
    elif alt < 5000:
        cmds['flaps'] = 10
    else:
        cmds['flaps'] = 0
    
    # Gear
    if alt < 1500 and _safe_get(state, 'weight_on_wheels_nose', 0) == 0:
        cmds['gear'] = 1
    elif alt > 2000:
        cmds['gear'] = 0
    
    # Trim slowly toward current attitude
    cmds['elevator_trim'] = pitch * 0.015
    
    return cmds
