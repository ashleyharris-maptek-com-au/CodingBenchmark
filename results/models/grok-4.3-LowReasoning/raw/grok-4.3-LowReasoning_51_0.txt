```python
import math
from typing import Dict, Any

# Module-level persistence for PID states, filters, and mode logic
_prev = {
    'alt_err_int': 0.0,
    'hdg_err_int': 0.0,
    'spd_err_int': 0.0,
    'vs_err_int': 0.0,
    'pitch_cmd': 0.0,
    'roll_cmd': 0.0,
    'last_valid_alt': 0.0,
    'last_valid_ias': 250.0,
    'last_valid_hdg': 0.0,
    'trim': 0.0,
    'flaps_cmd': 0.0,
    'gear_cmd': 0,
    'throttle_cmd': 0.5,
}

def _safe_get(d: Dict[str, Any], key: str, default: Any) -> Any:
    return d.get(key, default)

def _fuse_triple(d: Dict[str, Any], base: str, default: float = 0.0) -> float:
    """Cross-check triple-redundant sensors. Reject obvious outliers."""
    vals, valids = [], []
    for i in (1, 2, 3):
        v = _safe_get(d, f"{base}_{i}", default)
        valid = _safe_get(d, f"{base}_{i}_valid", True)
        vals.append(v)
        valids.append(valid)
    
    # Prefer values that are close to each other
    candidates = []
    for i in range(3):
        if valids[i]:
            candidates.append(vals[i])
    
    if not candidates:
        return default
    
    if len(candidates) == 1:
        return candidates[0]
    
    # Median filter for robustness
    candidates.sort()
    if len(candidates) == 3:
        return candidates[1]
    return (candidates[0] + candidates[1]) / 2

def autopilot_step(state: dict, dt: float) -> dict:
    global _prev
    
    # --- Sensor Fusion ---
    alt = _fuse_triple(state, "baro_alt", 5000.0)
    if alt < 2000:
        rad_alt = _fuse_triple(state, "radar_alt", alt)
        alt = min(alt, rad_alt) if rad_alt > 0 else alt
    
    ias = _fuse_triple(state, "ias", 250.0)
    mach = _fuse_triple(state, "mach", 0.4)
    vs = _fuse_triple(state, "irs", "vs_fpm", 0.0)
    hdg = _fuse_triple(state, "irs", "heading", 0.0)
    pitch = _fuse_triple(state, "irs", "pitch", 0.0)
    roll = _fuse_triple(state, "irs", "roll", 0.0)
    n1 = [_safe_get(state, f"engine_{i}_n1_pct", 80.0) for i in (1, 2)]
    
    target_alt = _safe_get(state, "target_alt_ft", alt)
    target_hdg = _safe_get(state, "target_hdg_deg", hdg)
    target_ias = _safe_get(state, "target_ias_kt", ias)
    ap_on = _safe_get(state, "ap_engaged", False)
    at_on = _safe_get(state, "at_engaged", False)
    
    # --- Limits and Protections ---
    if mach > 0.82 or ias > 365:
        target_ias = min(target_ias, 350)
    
    # Stall / low energy protection
    if ias < 140 and alt > 100:
        target_ias = 160
    
    # --- Basic Control Laws (simple PIDs) ---
    controls = {}
    
    if ap_on:
        # Altitude / VS hold
        alt_err = target_alt - alt
        _prev['alt_err_int'] = max(-500, min(500, _prev['alt_err_int'] + alt_err * dt * 0.001))
        vs_cmd = 0.8 * alt_err + 0.02 * _prev['alt_err_int']
        vs_cmd = max(-2000, min(2000, vs_cmd))
        
        # Pitch command from VS
        pitch_cmd = vs_cmd / 500.0 + (pitch * -0.3)  # light damping
        pitch_cmd = max(-15, min(15, pitch_cmd))
        _prev['pitch_cmd'] = 0.7 * _prev['pitch_cmd'] + 0.3 * pitch_cmd
        
        controls['elevator'] = _prev['pitch_cmd'] * 0.08
        controls['elevator_trim'] = max(-0.5, min(0.5, _prev['trim'] + _prev['pitch_cmd'] * 0.001))
        _prev['trim'] = controls['elevator_trim']
        
        # Heading hold
        hdg_err = (target_hdg - hdg + 180) % 360 - 180
        _prev['hdg_err_int'] = max(-30, min(30, _prev['hdg_err_int'] + hdg_err * dt * 0.01))
        roll_cmd = 0.8 * hdg_err + 0.05 * _prev['hdg_err_int']
        roll_cmd = max(-25, min(25, roll_cmd))
        _prev['roll_cmd'] = 0.6 * _prev['roll_cmd'] + 0.4 * roll_cmd
        controls['aileron'] = _prev['roll_cmd'] * 0.04
        controls['rudder'] = -roll * 0.02  # basic yaw damping
        
        # Autothrust
        if at_on:
            spd_err = target_ias - ias
            _prev['spd_err_int'] = max(-20, min(20, _prev['spd_err_int'] + spd_err * dt * 0.05))
            thr = 0.6 + 0.02 * spd_err + 0.005 * _prev['spd_err_int']
            
            # Asymmetric thrust handling
            if abs(n1[0] - n1[1]) > 15:
                thr = min(thr, 0.85)
            
            thr = max(0.0, min(1.0, thr))
            controls['throttle_1'] = thr
            controls['throttle_2'] = thr
            _prev['throttle_cmd'] = thr
    else:
        # AP off - gentle hands-off
        controls['elevator'] = -pitch * 0.03
        controls['aileron'] = -roll * 0.02
    
    # --- Configuration ---
    if alt < 2000:
        controls['gear'] = 1
        controls['flaps'] = 20 if alt < 1500 else 10
    elif alt > 3000:
        controls['gear'] = 0
        controls['flaps'] = 0
    
    controls['speedbrake'] = 0.0
    
    # Update persistence
    _prev['last_valid_alt'] = alt
    _prev['last_valid_ias'] = ias
    _prev['last_valid_hdg'] = hdg
    
    return controls
```
