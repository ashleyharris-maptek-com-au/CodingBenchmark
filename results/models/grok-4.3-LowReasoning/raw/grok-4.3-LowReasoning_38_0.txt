```python
import sys

lines = sys.stdin.read().strip().split('\n')
first = lines[0].split()
U = int(first[0])
S = int(first[1])
sets = []
for i in range(1, len(lines)):
    line = lines[i].strip()
    if line:
        sets.append(list(map(int, line.split())))
    else:
        sets.append([])

while len(sets) < S:
    sets.append([])

def solve(start, covered, selected):
    if len(covered) == U:
        return selected[:]
    for i in range(start, S):
        s = sets[i]
        if all(e not in covered for e in s):
            new_covered = covered | set(s)
            selected.append(i)
            res = solve(i + 1, new_covered, selected)
            if res is not None:
                return res
            selected.pop()
    return None

result = solve(0, set(), [])
if result is not None:
    print("SOLUTION")
    print(' '.join(map(str, result)))
else:
    print("NO SOLUTION")
```
