```python
import sys

lines = sys.stdin.read().splitlines()
if not lines:
    print("NO SOLUTION")
    sys.exit(0)

first = lines[0].split()
U = int(first[0])
S = int(first[1])

sets_masks = []
for i in range(S):
    line_idx = i + 1
    if line_idx < len(lines):
        nums = list(map(int, lines[line_idx].split()))
        mask = 0
        for e in nums:
            if 0 <= e < U:
                mask |= (1 << e)
        sets_masks.append(mask)
    else:
        sets_masks.append(0)

full = (1 << U) - 1

def solve(pos, current_mask, solution):
    if current_mask == full:
        return solution[:]
    if pos == S:
        return None
    # skip
    res = solve(pos + 1, current_mask, solution)
    if res is not None:
        return res
    # take
    m = sets_masks[pos]
    if (current_mask & m) == 0:
        solution.append(pos)
        res = solve(pos + 1, current_mask | m, solution)
        if res is not None:
            return res
        solution.pop()
    return None

result = solve(0, 0, [])
if result is not None:
    result.sort()
    print("SOLUTION")
    print(' '.join(map(str, result)))
else:
    print("NO SOLUTION")
```
