; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_ubo = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_ubo Uniform
%ptr_in_vec3 = OpTypePointer Input %vec3
%ptr_in_vec2 = OpTypePointer Input %vec2
%ptr_out_vec4 = OpTypePointer Output %vec4
%in_worldPos = OpVariable %ptr_in_vec3 Input
%in_normal = OpVariable %ptr_in_vec3 Input
%in_uv = OpVariable %ptr_in_vec2 Input
%in_tangent = OpVariable %ptr_in_vec3 Input
%in_color = OpVariable %ptr_in_vec3 Input
%outColor = OpVariable %ptr_out_vec4 Output
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_pi16 = OpConstant %float 50.26544
%float_0_9 = OpConstant %float 0.9
%float_0_1 = OpConstant %float 0.1
%red = OpConstantComposite %vec3 %float_0_9 %float_0_1 %float_0_1
%blue = OpConstantComposite %vec3 %float_0_1 %float_0_1 %float_0_9
%main = OpFunction %void None %func
%entry = OpLabel
%uv = OpLoad %vec2 %in_uv
%ny = OpCompositeExtract %float %uv 1
%arg = OpFMul %float %ny %float_pi16
%s = OpSin %float %arg
%sel = OpFOrdGreaterThan %bool %s %float_0
%base = OpSelect %vec3 %sel %red %blue
%nrm = OpLoad %vec3 %in_normal
%norm = OpNormalize %vec3 %nrm
%wp = OpLoad %vec3 %in_worldPos
%lp_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%lp4 = OpLoad %vec4 %lp_ptr
%lp = OpVectorShuffle %vec3 %lp4 %lp4 0 1 2
%ldir = OpSub %vec3 %lp %wp
%ldn = OpNormalize %vec3 %ldir
%diff = OpDot %float %norm %ldn
%diffc = OpExtInst %float %glsl_max %diff %float_0
%fac = OpFMul %float %float_0_6 %diffc
%shad = OpFAdd %float %float_0_4 %fac
%col = OpVectorTimesScalar %vec3 %base %shad
%out = OpCompositeConstruct %vec4 %col %float_1
OpStore %outColor %out
OpReturn
OpFunctionEnd
%int_3 = OpConstant %int 3
%bool = OpTypeBool
%glsl_max = OpExtInstImport "GLSL.std.450"
