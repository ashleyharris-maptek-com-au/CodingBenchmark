import math

# Module-level persistence (reset between scenarios/go-arounds by test harness)
_prev_pitch = 0.0
_flare_start_alt = None
_touchdown_detected = False
_gear_deployed = False
_spoilers_deployed = False
_reverse_deployed = False
_autobrake_set = False
_phase = "intercept"
_goaround_triggered = False
_integrator_loc = 0.0
_integrator_gs = 0.0
_last_valid_loc = 0.0
_last_valid_gs = 0.0

def _median_valid(values, valid_flags):
    valid = [v for v, ok in zip(values, valid_flags) if ok]
    if not valid:
        return None
    valid.sort()
    n = len(valid)
    return valid[n // 2] if n % 2 else (valid[n // 2 - 1] + valid[n // 2]) / 2

def autopilot_step(state: dict, dt: float) -> dict:
    global _prev_pitch, _flare_start_alt, _touchdown_detected
    global _gear_deployed, _spoilers_deployed, _reverse_deployed
    global _autobrake_set, _phase, _goaround_triggered
    global _integrator_loc, _integrator_gs, _last_valid_loc, _last_valid_gs

    dt = max(dt, 0.01)

    # --- Sensor fusion ---
    gs_vals = [state.get(f"glideslope_{i}_dots", 0.0) for i in range(1, 4)]
    gs_valid = [state.get(f"glideslope_{i}_valid", False) for i in range(1, 4)]
    loc_vals = [state.get(f"localizer_{i}_dots", 0.0) for i in range(1, 4)]
    loc_valid = [state.get(f"localizer_{i}_valid", False) for i in range(1, 4)]

    gs_dots = _median_valid(gs_vals, gs_valid)
    loc_dots = _median_valid(loc_vals, loc_valid)

    if gs_dots is None:
        gs_dots = _last_valid_gs
    else:
        _last_valid_gs = gs_dots
    if loc_dots is None:
        loc_dots = _last_valid_loc
    else:
        _last_valid_loc = loc_dots

    radio_alt = state.get("radio_alt_ft", 10000.0)
    on_ground = state.get("on_ground", False) or state.get("weight_on_wheels", False)
    ias = state.get("ias_1_kt", 140.0)
    target_ias = state.get("target_ias_kt", 137.0)
    vs = state.get("irs_1_vs_fpm", 0.0)

    # --- Phase management ---
    if on_ground and not _touchdown_detected:
        _touchdown_detected = True
        _phase = "rollout"

    if radio_alt > 1200:
        _phase = "intercept"
    elif radio_alt > 200:
        _phase = "approach"
    elif radio_alt < 50 and not on_ground:
        _phase = "flare"

    # --- Go-around logic (conservative) ---
    if (not _goaround_triggered and
            ((abs(gs_dots) > 1.8 and radio_alt < 800) or
             (abs(loc_dots) > 2.5 and radio_alt < 600) or
             (ias < 110 and radio_alt < 300) or
             (vs < -1200 and radio_alt < 500))):
        _goaround_triggered = True
        return {
            "throttle_1": 1.0,
            "throttle_2": 1.0,
            "elevator": 0.6,
            "elevator_trim": 0.3,
            "flaps": 20,
            "gear": 0.0,
            "speedbrake": 0.0,
            "spoiler": 0.0,
            "reverse_1": 0.0,
            "reverse_2": 0.0,
        }

    # --- Lateral control (localizer) ---
    if loc_dots is not None:
        _integrator_loc += loc_dots * dt
        _integrator_loc = max(min(_integrator_loc, 5.0), -5.0)
        aileron = -0.8 * loc_dots - 0.03 * _integrator_loc
        rudder = -0.4 * loc_dots
    else:
        aileron = 0.0
        rudder = 0.0

    aileron = max(min(aileron, 0.8), -0.8)
    rudder = max(min(rudder, 0.6), -0.6)

    # --- Vertical control (glideslope + flare) ---
    if _phase == "flare":
        if _flare_start_alt is None:
            _flare_start_alt = radio_alt
        flare_progress = max(0.0, min(1.0, (50 - radio_alt) / 50))
        target_pitch = 2.5 + 4.0 * flare_progress
        pitch_error = target_pitch - state.get("irs_1_pitch_deg", 0.0)
        elevator = 0.15 * pitch_error
        throttle_cmd = 0.05
    else:
        if gs_dots is not None:
            _integrator_gs += gs_dots * dt
            _integrator_gs = max(min(_integrator_gs, 8.0), -8.0)
            elevator = -0.55 * gs_dots - 0.015 * _integrator_gs
        else:
            elevator = _prev_pitch * 0.7
        throttle_cmd = 0.55 if ias < target_ias - 3 else 0.42 if ias > target_ias + 5 else 0.48

    elevator = max(min(elevator, 0.9), -0.6)
    _prev_pitch = elevator

    # --- Configuration ---
    flaps = 30 if radio_alt < 1500 else 25 if radio_alt < 3000 else 15
    gear = 1.0 if radio_alt < 2000 or _gear_deployed else 0.0
    _gear_deployed = gear > 0.5

    # --- Post-touchdown ---
    if _phase == "rollout":
        elevator = -0.1
        aileron = 0.0
        rudder = 0.0
        throttle_cmd = 0.0
        speedbrake = 1.0
        spoiler = 1.0
        autobrake = 0.85
        reverse = 0.9 if ias > 60 else 0.3
        _spoilers_deployed = True
        _reverse_deployed = True
        _autobrake_set = True
        return {
            "elevator": elevator,
            "aileron": aileron,
            "rudder": rudder,
            "throttle_1": 0.0,
            "throttle_2": 0.0,
            "flaps": flaps,
            "gear": 1.0,
            "speedbrake": speedbrake,
            "spoiler": spoiler,
            "autobrake": autobrake,
            "reverse_1": reverse,
            "reverse_2": reverse,
        }

    return {
        "elevator": float(elevator),
        "aileron": float(aileron),
        "rudder": float(rudder),
        "throttle_1": float(throttle_cmd),
        "throttle_2": float(throttle_cmd),
        "flaps": float(flaps),
        "gear": float(gear),
        "speedbrake": 0.0,
        "spoiler": 0.0,
    }
