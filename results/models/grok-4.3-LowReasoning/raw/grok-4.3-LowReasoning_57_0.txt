```javascript
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { width, height, grid, ignitions, towns, cutBudget } = data;
const dirs = [[0,1],[1,0],[0,-1],[-1,0]];

function inBounds(x, y) {
  return x >= 0 && x < width && y >= 0 && y < height;
}

function isBurnable(x, y) {
  return grid[y][x] === '.';
}

// Find threatened towns by simulating spread
const visited = Array.from({ length: height }, () => Array(width).fill(false));
const queue = [];
for (const [x, y] of ignitions) {
  if (inBounds(x, y) && isBurnable(x, y)) {
    queue.push([x, y]);
    visited[y][x] = true;
  }
}
let qi = 0;
while (qi < queue.length) {
  const [x, y] = queue[qi++];
  for (const [dx, dy] of dirs) {
    const nx = x + dx, ny = y + dy;
    if (inBounds(nx, ny) && !visited[ny][nx] && isBurnable(nx, ny)) {
      visited[ny][nx] = true;
      queue.push([nx, ny]);
    }
  }
}

const threatened = towns.filter(t => visited[t.y][t.x]).sort((a, b) => b.value - a.value);

const cuts = new Set();
let remaining = cutBudget;

for (const town of threatened) {
  if (remaining <= 0) break;
  const candidates = [];
  for (const [dx, dy] of dirs) {
    const nx = town.x + dx, ny = town.y + dy;
    if (inBounds(nx, ny) && isBurnable(nx, ny) && !visited[ny][nx] === false) {
      const key = `${nx},${ny}`;
      if (!cuts.has(key)) candidates.push([nx, ny]);
    }
  }
  for (const c of candidates) {
    if (remaining <= 0) break;
    const key = `${c[0]},${c[1]}`;
    if (!cuts.has(key)) {
      cuts.add(key);
      remaining--;
    }
  }
}

const firebreaks = Array.from(cuts).map(s => s.split(',').map(Number));
console.log(JSON.stringify({ firebreaks }));
```
