```javascript
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const width = data.width;
const height = data.height;
const grid = data.grid;
const ignitions = data.ignitions || [];
const towns = data.towns || [];
const cutBudget = data.cutBudget || 0;

const dirs = [[0,1],[1,0],[0,-1],[-1,0]];
const inBounds = (x, y) => x >= 0 && x < width && y >= 0 && y < height;
const isBurnable = (x, y) => grid[y][x] === '.';

const cuts = new Set();
let remaining = cutBudget;

function posKey(x, y) {
  return `${x},${y}`;
}

function addCut(x, y) {
  if (remaining <= 0) return false;
  const key = posKey(x, y);
  if (!cuts.has(key) && isBurnable(x, y)) {
    cuts.add(key);
    remaining--;
    return true;
  }
  return false;
}

// Sort towns by value descending to protect highest value first
const sortedTowns = [...towns].sort((a, b) => b.value - a.value);

for (const town of sortedTowns) {
  if (remaining <= 0) break;
  const candidates = [];
  for (const [dx, dy] of dirs) {
    const nx = town.x + dx;
    const ny = town.y + dy;
    if (inBounds(nx, ny) && isBurnable(nx, ny)) {
      candidates.push([nx, ny]);
    }
  }
  // Add cuts around the town (protect it completely if possible)
  for (const [x, y] of candidates) {
    if (addCut(x, y)) {
      if (remaining <= 0) break;
    }
  }
}

// If budget remains, cut around ignitions to contain early
if (remaining > 0) {
  const ignSet = new Set(ignitions.map(p => posKey(p[0], p[1])));
  for (const [sx, sy] of ignitions) {
    if (remaining <= 0) break;
    for (const [dx, dy] of dirs) {
      const nx = sx + dx;
      const ny = sy + dy;
      if (inBounds(nx, ny) && isBurnable(nx, ny) && !ignSet.has(posKey(nx, ny))) {
        addCut(nx, ny);
        if (remaining <= 0) break;
      }
    }
  }
}

const firebreaks = Array.from(cuts).map(k => k.split(',').map(Number));
const output = { firebreaks };
console.log(JSON.stringify(output));
```
