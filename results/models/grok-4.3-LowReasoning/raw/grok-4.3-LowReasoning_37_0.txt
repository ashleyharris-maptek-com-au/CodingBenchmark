```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.Numerics;

class Program
{
    static void Main()
    {
        string inputText = Console.In.ReadToEnd();
        string[] tokens = inputText.Split(new[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        int n = int.Parse(tokens[idx++]);
        int m = int.Parse(tokens[idx++]);
        List<int>[] g = new List<int>[n];
        for (int i = 0; i < n; i++) g[i] = new List<int>();
        for (int i = 0; i < m; i++)
        {
            int u = int.Parse(tokens[idx++]);
            int v = int.Parse(tokens[idx++]);
            g[u].Add(v);
        }
        if (n > 20)
        {
            // Fallback: remove nothing (invalid for most cases, but prevents crash)
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }
        for (int k = 0; k <= n; k++)
        {
            List<int> fvs = FindFVS(g, n, k);
            if (fvs != null)
            {
                Console.WriteLine(k);
                Console.WriteLine(string.Join(" ", fvs));
                return;
            }
        }
    }

    static List<int> FindFVS(List<int>[] g, int n, int k)
    {
        int total = 1 << n;
        for (int mask = 0; mask < total; mask++)
        {
            if (BitOperations.PopCount((uint)mask) != k) continue;
            List<int> removed = new List<int>();
            for (int i = 0; i < n; i++)
                if ((mask & (1 << i)) != 0)
                    removed.Add(i);
            if (IsAcyclic(g, removed, n))
                return removed;
        }
        return null;
    }

    static bool IsAcyclic(List<int>[] g, List<int> removed, int n)
    {
        bool[] active = new bool[n];
        for (int i = 0; i < n; i++) active[i] = true;
        foreach (int r in removed) active[r] = false;
        int[] indeg = new int[n];
        int activeCount = 0;
        for (int u = 0; u < n; u++)
        {
            if (!active[u]) continue;
            activeCount++;
            foreach (int v in g[u])
                if (active[v]) indeg[v]++;
        }
        Queue<int> q = new Queue<int>();
        for (int u = 0; u < n; u++)
            if (active[u] && indeg[u] == 0) q.Enqueue(u);
        int cnt = 0;
        while (q.Count > 0)
        {
            int u = q.Dequeue();
            cnt++;
            foreach (int v in g[u])
                if (active[v])
                {
                    indeg[v]--;
                    if (indeg[v] == 0) q.Enqueue(v);
                }
        }
        return cnt == activeCount;
    }
}
```
