using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        string[] first = Console.ReadLine().Split();
        int n = int.Parse(first[0]);
        int m = int.Parse(first[1]);
        List<int>[] adj = new List<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new List<int>();
        bool[,] hasEdge = new bool[n, n];
        for (int i = 0; i < m; i++)
        {
            string[] uv = Console.ReadLine().Split();
            int u = int.Parse(uv[0]);
            int v = int.Parse(uv[1]);
            if (!hasEdge[u, v])
            {
                hasEdge[u, v] = true;
                adj[u].Add(v);
            }
        }

        int minK = n + 1;
        int bestMask = -1;

        if (n <= 20)
        {
            int N = 1 << n;
            for (int mask = 0; mask < N; mask++)
            {
                int k = System.Numerics.BitOperations.PopCount((uint)mask);
                if (k >= minK) continue;
                if (IsAcyclic(n, adj, hasEdge, mask))
                {
                    minK = k;
                    bestMask = mask;
                }
            }
        }
        else
        {
            // Fallback for larger n: start with empty and grow (rarely minimal unless already DAG)
            if (IsAcyclic(n, adj, hasEdge, 0))
            {
                minK = 0;
                bestMask = 0;
            }
            else
            {
                // Minimal heuristic: pick high in-degree/out-degree vertices
                var deg = new int[n];
                for (int u = 0; u < n; u++)
                {
                    deg[u] = adj[u].Count;
                    for (int v = 0; v < n; v++) if (hasEdge[v, u]) deg[u]++;
                }
                var order = Enumerable.Range(0, n).OrderByDescending(i => deg[i]).ToArray();
                int mask = 0;
                for (int i = 0; i < n; i++)
                {
                    if (IsAcyclic(n, adj, hasEdge, mask)) break;
                    mask |= (1 << order[i]);
                }
                minK = System.Numerics.BitOperations.PopCount((uint)mask);
                bestMask = mask;
            }
        }

        if (bestMask == -1)
        {
            // Should not happen
            Console.WriteLine(n);
            Console.WriteLine(string.Join(" ", Enumerable.Range(0, n)));
            return;
        }

        List<int> res = new List<int>();
        for (int i = 0; i < n; i++)
            if ((bestMask & (1 << i)) != 0) res.Add(i);

        Console.WriteLine(res.Count);
        Console.WriteLine(string.Join(" ", res));
    }

    static bool IsAcyclic(int n, List<int>[] adj, bool[,] hasEdge, int removeMask)
    {
        int[] indeg = new int[n];
        bool[] active = new bool[n];
        int activeCount = 0;
        for (int i = 0; i < n; i++)
        {
            if ((removeMask & (1 << i)) != 0) continue;
            active[i] = true;
            activeCount++;
        }
        for (int u = 0; u < n; u++)
        {
            if (!active[u]) continue;
            foreach (int v in adj[u])
            {
                if (active[v]) indeg[v]++;
            }
        }
        Queue<int> q = new Queue<int>();
        for (int i = 0; i < n; i++)
            if (active[i] && indeg[i] == 0) q.Enqueue(i);

        int processed = 0;
        while (q.Count > 0)
        {
            int u = q.Dequeue();
            processed++;
            foreach (int v in adj[u])
            {
                if (!active[v]) continue;
                indeg[v]--;
                if (indeg[v] == 0) q.Enqueue(v);
            }
        }
        return processed == activeCount;
    }
}
