using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        var input = Console.In.ReadToEnd().Split(new[] { ' ', '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        int n = int.Parse(input[idx++]);
        int m = int.Parse(input[idx++]);
        var graph = new List<int>[n];
        for (int i = 0; i < n; i++) graph[i] = new List<int>();
        for (int i = 0; i < m; i++)
        {
            int u = int.Parse(input[idx++]);
            int v = int.Parse(input[idx++]);
            graph[u].Add(v);
        }

        Func<HashSet<int>, bool> IsAcyclic = (removed) =>
        {
            var indeg = new int[n];
            var adj = new List<int>[n];
            for (int i = 0; i < n; i++)
            {
                if (removed.Contains(i)) continue;
                adj[i] = graph[i].Where(v => !removed.Contains(v)).ToList();
                foreach (var v in adj[i]) indeg[v]++;
            }
            var queue = new Queue<int>();
            for (int i = 0; i < n; i++)
                if (!removed.Contains(i) && indeg[i] == 0) queue.Enqueue(i);
            int count = 0;
            while (queue.Count > 0)
            {
                int u = queue.Dequeue();
                count++;
                if (adj[u] == null) continue;
                foreach (var v in adj[u])
                {
                    indeg[v]--;
                    if (indeg[v] == 0) queue.Enqueue(v);
                }
            }
            return count == n - removed.Count;
        };

        for (int k = 0; k <= n; k++)
        {
            if (FindCombo(0, k, new HashSet<int>(), n, IsAcyclic, out var set))
            {
                Console.WriteLine(k);
                Console.WriteLine(string.Join(" ", set.OrderBy(x => x)));
                return;
            }
        }
        // fallback empty (should not happen for valid input)
        Console.WriteLine(0);
        Console.WriteLine();
    }

    static bool FindCombo(int start, int remain, HashSet<int> current, int n, Func<HashSet<int>, bool> check, out List<int> result)
    {
        if (remain == 0)
        {
            if (check(current))
            {
                result = current.ToList();
                return true;
            }
            result = null;
            return false;
        }
        for (int i = start; i < n; i++)
        {
            current.Add(i);
            if (FindCombo(i + 1, remain - 1, current, n, check, out result)) return true;
            current.Remove(i);
        }
        result = null;
        return false;
    }
}
