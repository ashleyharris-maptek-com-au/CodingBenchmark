```rust
use std::io::{self, BufRead, BufReader};
use std::collections::HashMap;

#[derive(Clone)]
struct Sample {
    depth: f64,
    props: Vec<f64>,
    pos: [f64; 3],
}

#[derive(Clone)]
struct Hole {
    id: String,
    samples: Vec<Sample>,
}

fn main() {
    let stdin = io::stdin();
    let reader = BufReader::new(stdin.lock());
    let mut lines = reader.lines();

    let first = lines.next().unwrap().unwrap();
    let mut parts = first.split_whitespace();
    let num_holes: usize = parts.next().unwrap().parse().unwrap();
    let num_props: usize = parts.next().unwrap().parse().unwrap();

    let prop_names: Vec<String> = lines.next().unwrap().unwrap()
        .split_whitespace()
        .map(|s| s.to_string())
        .collect();

    let mut holes: Vec<Hole> = Vec::with_capacity(num_holes);

    for _ in 0..num_holes {
        let line = match lines.next() {
            Some(Ok(l)) => l,
            _ => break,
        };
        let mut p = line.split_whitespace();
        let hole_id = p.next().unwrap().to_string();
        let sx: f64 = p.next().unwrap().parse().unwrap();
        let sy: f64 = p.next().unwrap().parse().unwrap();
        let sz: f64 = p.next().unwrap().parse().unwrap();
        let dx: f64 = p.next().unwrap().parse().unwrap();
        let dy: f64 = p.next().unwrap().parse().unwrap();
        let dz: f64 = p.next().unwrap().parse().unwrap();
        let length: f64 = p.next().unwrap().parse().unwrap();
        let num_samples: usize = p.next().unwrap().parse().unwrap();

        let dir_len = (dx*dx + dy*dy + dz*dz).sqrt();
        let (ux, uy, uz) = if dir_len > 1e-9 {
            (dx / dir_len, dy / dir_len, dz / dir_len)
        } else {
            (0.0, 0.0, 1.0)
        };

        let mut samples = Vec::with_capacity(num_samples);
        for _ in 0..num_samples {
            let sline = match lines.next() {
                Some(Ok(l)) => l,
                _ => break,
            };
            let mut sp = sline.split_whitespace();
            let depth: f64 = sp.next().unwrap().parse().unwrap();
            let mut props = Vec::with_capacity(num_props);
            for _ in 0..num_props {
                props.push(sp.next().unwrap().parse().unwrap());
            }
            let t = if length > 1e-9 { depth / length } else { 0.0 };
            let pos = [
                sx + t * ux * length,
                sy + t * uy * length,
                sz + t * uz * length,
            ];
            samples.push(Sample { depth, props, pos });
        }
        holes.push(Hole { id: hole_id, samples });
    }

    let total_samples: usize = holes.iter().map(|h| h.samples.len()).sum();

    let mut suspects: Vec<(f64, String, usize, String)> = Vec::new();

    if total_samples < 50_000 {
        // Medium scale: neighbor comparison along hole
        for hole in &holes {
            for i in 0..hole.samples.len() {
                let s = &hole.samples[i];
                for p in 0..num_props {
                    let val = s.props[p];
                    let mut neighbors = Vec::new();
                    if i > 0 { neighbors.push(hole.samples[i-1].props[p]); }
                    if i + 1 < hole.samples.len() { neighbors.push(hole.samples[i+1].props[p]); }
                    if neighbors.is_empty() { continue; }
                    let mean: f64 = neighbors.iter().sum::<f64>() / neighbors.len() as f64;
                    let diff = (val - mean).abs();
                    if diff > 1e-9 {
                        let conf = (diff / (mean.abs() + 1.0)).min(10.0);
                        if conf > 0.8 {
                            suspects.push((conf, hole.id.clone(), i, prop_names[p].clone()));
                        }
                    }
                }
            }
        }
    } else {
        // Large scale: simple z-score per hole per property
        for hole in &holes {
            if hole.samples.is_empty() { continue; }
            for p in 0..num_props {
                let vals: Vec<f64> = hole.samples.iter().map(|s| s.props[p]).collect();
                let n = vals.len() as f64;
                let mean: f64 = vals.iter().sum::<f64>() / n;
                let var: f64 = vals.iter().map(|v| (v - mean).powi(2)).sum::<f64>() / n;
                let std = var.sqrt();
                if std < 1e-9 { continue; }
                for (i, &val) in vals.iter().enumerate() {
                    let z = (val - mean) / std;
                    if z.abs() > 3.5 {
                        let conf = (z.abs() - 3.0).min(8.0);
                        suspects.push((conf, hole.id.clone(), i, prop_names[p].clone()));
                    }
                }
            }
        }
    }

    suspects.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap());
    let mut seen = HashMap::new();
    let mut unique: Vec<_> = Vec::new();
    for s in suspects {
        let key = (s.1.clone(), s.2, s.3.clone());
        if !seen.contains_key(&key) {
            seen.insert(key, true);
            unique.push(s);
        }
    }

    println!("{}", unique.len());
    for (conf, hid, idx, pname) in unique {
        println!("{} {} {} {:.3}", hid, idx, pname, conf);
    }
}
```
