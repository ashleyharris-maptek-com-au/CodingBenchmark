```rust
use std::collections::HashMap;
use std::io::{self, BufRead, BufReader};

#[derive(Clone)]
struct Sample {
    depth: f64,
    values: Vec<f64>,
}

#[derive(Clone)]
struct Hole {
    id: String,
    samples: Vec<Sample>,
}

fn main() {
    let stdin = io::stdin();
    let reader = BufReader::new(stdin.lock());
    let mut lines = reader.lines();

    // Parse header
    let header: Vec<String> = lines.next().unwrap().unwrap()
        .split_whitespace().map(|s| s.to_string()).collect();
    let num_holes: usize = header[0].parse().unwrap();
    let num_props: usize = header[1].parse().unwrap();

    let prop_names: Vec<String> = lines.next().unwrap().unwrap()
        .split_whitespace().map(|s| s.to_string()).collect();

    let mut holes: Vec<Hole> = Vec::with_capacity(num_holes);
    let mut total_samples: usize = 0;

    for _ in 0..num_holes {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let hole_id = parts[0].to_string();
        let num_samples: usize = parts[7].parse().unwrap();

        let mut samples: Vec<Sample> = Vec::with_capacity(num_samples);
        for _ in 0..num_samples {
            let sline = lines.next().unwrap().unwrap();
            let s parts: Vec<&str> = sline.split_whitespace().collect();
            if s parts.len() < 1 + num_props { continue; }
            let depth: f64 = s parts[0].parse().unwrap_or(0.0);
            let mut values = Vec::with_capacity(num_props);
            for i in 0..num_props {
                values.push(s parts[1 + i].parse().unwrap_or(0.0));
            }
            samples.push(Sample { depth, values });
        }
        total_samples += samples.len();
        holes.push(Hole { id: hole_id, samples });
    }

    // Adaptive strategy selection
    let large_scale = total_samples > 50_000;

    let mut suspects: Vec<(String, usize, String, f64)> = Vec::new();

    for hole in &holes {
        for p in 0..num_props {
            let prop_name = &prop_names[p];
            let vals: Vec<f64> = hole.samples.iter()
                .map(|s| s.values[p]).collect();

            if vals.len() < 3 { continue; }

            let mean = vals.iter().sum::<f64>() / vals.len() as f64;
            let variance = vals.iter().map(|v| (v - mean).powi(2)).sum::<f64>() / vals.len() as f64;
            let std = variance.sqrt();

            if std < 1e-9 { continue; }

            for (i, &v) in vals.iter().enumerate() {
                let z = (v - mean) / std;
                let conf = z.abs();

                let is_outlier = if large_scale {
                    conf > 3.5
                } else {
                    // More sensitive for small data + local check
                    if i > 0 && i + 1 < vals.len() {
                        let local_avg = (vals[i-1] + vals[i+1]) / 2.0;
                        let local_dev = ((v - local_avg).abs() / (std + 1e-9)).min(10.0);
                        conf > 2.8 || local_dev > 3.5
                    } else {
                        conf > 2.8
                    }
                };

                if is_outlier {
                    suspects.push((
                        hole.id.clone(),
                        i,
                        prop_name.clone(),
                        conf,
                    ));
                }
            }
        }
    }

    // Sort by confidence descending
    suspects.sort_by(|a, b| b.3.partial_cmp(&a.3).unwrap());

    // Dedup (keep highest confidence per hole/sample/prop)
    let mut seen = HashMap::new();
    let mut final_suspects = Vec::new();
    for s in suspects {
        let key = (s.0.clone(), s.1, s.2.clone());
        if !seen.contains_key(&key) {
            seen.insert(key, true);
            final_suspects.push(s);
        }
    }

    // Output
    println!("{}", final_suspects.len());
    for (hole_id, idx, prop, conf) in final_suspects {
        println!("{} {} {} {:.3}", hole_id, idx, prop, conf);
    }
}
```
