```typescript
declare const require: any;
const fs = require('fs');

interface Task {
  id: number;
  duration: number;
  deps: number[];
  deadline: number;
  weight: number;
}

interface Input {
  machineCount: number;
  tasks: Task[];
}

interface Output {
  order?: number[];
  schedule?: Array<{ id: number; start: number; machine: number }>;
}

const input: string = fs.readFileSync(0, 'utf8');
const data: Input = JSON.parse(input);

const tasks = data.tasks;
const n = tasks.length;
const taskMap = new Map<number, Task>();
for (const t of tasks) {
  taskMap.set(t.id, t);
}

const adj = new Map<number, number[]>();
const indegree = new Map<number, number>();
for (const t of tasks) {
  adj.set(t.id, []);
  indegree.set(t.id, t.deps.length);
}
for (const t of tasks) {
  for (const d of t.deps) {
    if (adj.has(d)) {
      adj.get(d)!.push(t.id);
    }
  }
}

const ready: number[] = [];
for (const t of tasks) {
  if (t.deps.length === 0) {
    ready.push(t.id);
  }
}

function priority(id: number): number {
  const t = taskMap.get(id)!;
  return t.deadline * 100000 - t.weight;
}

const order: number[] = [];
while (ready.length > 0) {
  ready.sort((a, b) => priority(a) - priority(b));
  const current = ready.shift()!;
  order.push(current);
  for (const nei of adj.get(current) || []) {
    const deg = indegree.get(nei)! - 1;
    indegree.set(nei, deg);
    if (deg === 0) {
      ready.push(nei);
    }
  }
}

if (order.length !== n) {
  // fallback to any valid topo if cycle (should not happen)
  console.log(JSON.stringify({ order: tasks.map(t => t.id) }));
} else {
  const out: Output = { order };
  console.log(JSON.stringify(out));
}
```
