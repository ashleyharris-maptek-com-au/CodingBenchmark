```typescript
declare const require: any;
const fs = require('fs');

interface Task {
  id: string;
  duration: number;
  deps: string[];
  deadline: number;
  weight: number;
}

interface Input {
  machineCount: number;
  tasks: Task[];
}

interface ScheduleEntry {
  id: string;
  start: number;
  machine: number;
}

interface Output {
  schedule?: ScheduleEntry[];
  order?: string[];
}

const input: string = fs.readFileSync(0, 'utf8');
const data: Input = JSON.parse(input);

const tasks = data.tasks;
const machineCount = data.machineCount;

const taskMap = new Map<string, Task>();
const indegree = new Map<string, number>();
const adj = new Map<string, string[]>();

for (const task of tasks) {
  taskMap.set(task.id, task);
  indegree.set(task.id, task.deps.length);
  adj.set(task.id, []);
}

for (const task of tasks) {
  for (const dep of task.deps) {
    if (adj.has(dep)) {
      adj.get(dep)!.push(task.id);
    }
  }
}

const ready: string[] = [];
for (const [id, deg] of indegree.entries()) {
  if (deg === 0) {
    ready.push(id);
  }
}

const order: string[] = [];
const startTime = new Map<string, number>();
const machineAssign = new Map<string, number>();
const finishTime = new Map<string, number>();
const machineFree = new Array(machineCount).fill(0);

while (ready.length > 0) {
  // Select task with highest weight, then earliest deadline
  ready.sort((a, b) => {
    const ta = taskMap.get(a)!;
    const tb = taskMap.get(b)!;
    if (tb.weight !== ta.weight) return tb.weight - ta.weight;
    return ta.deadline - tb.deadline;
  });
  const current = ready.shift()!;
  order.push(current);

  // Assign to machine with earliest free time (greedy)
  let bestMachine = 0;
  let bestFree = machineFree[0];
  for (let m = 1; m < machineCount; m++) {
    if (machineFree[m] < bestFree) {
      bestFree = machineFree[m];
      bestMachine = m;
    }
  }

  const st = Math.max(bestFree, 0);
  const task = taskMap.get(current)!;
  const ft = st + task.duration;

  startTime.set(current, st);
  machineAssign.set(current, bestMachine);
  finishTime.set(current, ft);
  machineFree[bestMachine] = ft;

  // Update dependents
  for (const succ of adj.get(current) || []) {
    const newDeg = indegree.get(succ)! - 1;
    indegree.set(succ, newDeg);
    if (newDeg === 0) {
      ready.push(succ);
    }
  }
}

const schedule: ScheduleEntry[] = order.map(id => ({
  id,
  start: startTime.get(id)!,
  machine: machineAssign.get(id)!
}));

const output: Output = { schedule };
console.log(JSON.stringify(output));
```
