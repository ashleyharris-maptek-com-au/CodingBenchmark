```csharp
using System;
using System.Collections.Generic;
using System.IO;

class TetrisAI
{
    static int width, height, numPieces;
    static bool[,] board;
    static Random rng = new Random();

    static void Main()
    {
        string[] first = Console.ReadLine().Trim().Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        width = int.Parse(first[0]);
        height = int.Parse(first[1]);
        numPieces = int.Parse(first[2]);
        board = new bool[height, width];

        for (int p = 0; p < numPieces; p++)
        {
            string type = Console.ReadLine().Trim();
            int bestRot = 0, bestCol = 0;
            double bestScore = double.MinValue;

            int maxRot = (type == "O") ? 1 : (type == "I" || type == "S" || type == "Z") ? 2 : 4;

            for (int rot = 0; rot < maxRot; rot++)
            {
                var shape = GetShape(type, rot);
                int maxC = width - shape[0].Length;

                for (int c = 0; c <= maxC; c++)
                {
                    int landingRow = FindLandingRow(shape, c);
                    if (landingRow < 0) continue;

                    double score = Evaluate(shape, landingRow, c, type, rot);
                    
                    // Adaptive strategy based on stack height
                    int stackHeight = GetStackHeight();
                    if (stackHeight > height * 0.7)
                        score = AggressiveScore(shape, landingRow, c);
                    else if (stackHeight < height * 0.3)
                        score = CarefulScore(shape, landingRow, c);

                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestRot = rot;
                        bestCol = c;
                    }
                }
            }

            Console.WriteLine($"{bestRot} {bestCol}");
            string result = Console.ReadLine().Trim();
            
            if (result == "gameover") break;
            if (result.StartsWith("cleared_"))
            {
                int cleared = int.Parse(result.Substring(7));
                UpdateBoardAfterClear(cleared);
            }
            
            // Apply placement to our board representation
            ApplyPlacement(GetShape(type, bestRot), bestCol);
        }
    }

    static int[][] GetShape(string type, int rot)
    {
        if (type == "I")
        {
            if (rot == 0) return new[] { new[] { 1, 1, 1, 1 } };
            return new[] { new[] { 1 }, new[] { 1 }, new[] { 1 }, new[] { 1 } };
        }
        if (type == "O") return new[] { new[] { 1, 1 }, new[] { 1, 1 } };
        if (type == "T")
        {
            if (rot == 0) return new[] { new[] { 0, 1, 0 }, new[] { 1, 1, 1 } };
            if (rot == 1) return new[] { new[] { 1, 0 }, new[] { 1, 1 }, new[] { 1, 0 } };
            if (rot == 2) return new[] { new[] { 1, 1, 1 }, new[] { 0, 1, 0 } };
            return new[] { new[] { 0, 1 }, new[] { 1, 1 }, new[] { 0, 1 } };
        }
        if (type == "S")
        {
            if (rot % 2 == 0) return new[] { new[] { 0, 1, 1 }, new[] { 1, 1, 0 } };
            return new[] { new[] { 1, 0 }, new[] { 1, 1 }, new[] { 0, 1 } };
        }
        if (type == "Z")
        {
            if (rot % 2 == 0) return new[] { new[] { 1, 1, 0 }, new[] { 0, 1, 1 } };
            return new[] { new[] { 0, 1 }, new[] { 1, 1 }, new[] { 1, 0 } };
        }
        if (type == "J")
        {
            if (rot == 0) return new[] { new[] { 1, 0, 0 }, new[] { 1, 1, 1 } };
            if (rot == 1) return new[] { new[] { 1, 1 }, new[] { 1, 0 }, new[] { 1, 0 } };
            if (rot == 2) return new[] { new[] { 1, 1, 1 }, new[] { 0, 0, 1 } };
            return new[] { new[] { 0, 1 }, new[] { 0, 1 }, new[] { 1, 1 } };
        }
        // L
        if (rot == 0) return new[] { new[] { 0, 0, 1 }, new[] { 1, 1, 1 } };
        if (rot == 1) return new[] { new[] { 1, 0 }, new[] { 1, 0 }, new[] { 1, 1 } };
        if (rot == 2) return new[] { new[] { 1, 1, 1 }, new[] { 1, 0, 0 } };
        return new[] { new[] { 1, 1 }, new[] { 0, 1 }, new[] { 0, 1 } };
    }

    static int FindLandingRow(int[][] shape, int col)
    {
        int h = shape.Length, w = shape[0].Length;
        for (int r = height - h; r >= -1; r--)
        {
            if (!CanPlace(shape, r, col)) return r + 1;
        }
        return -1;
    }

    static bool CanPlace(int[][] shape, int row, int col)
    {
        int h = shape.Length, w = shape[0].Length;
        for (int i = 0; i < h; i++)
        {
            for (int j = 0; j < w; j++)
            {
                if (shape[i][j] == 0) continue;
                int br = row + i, bc = col + j;
                if (br < 0 || br >= height || bc < 0 || bc >= width || board[br, bc]) return false;
            }
        }
        return true;
    }

    static void ApplyPlacement(int[][] shape, int col)
    {
        int row = FindLandingRow(shape, col);
        if (row < 0) return;
        int h = shape.Length;
        for (int i = 0; i < h; i++)
            for (int j = 0; j < shape[i].Length; j++)
                if (shape[i][j] == 1) board[row + i, col + j] = true;
    }

    static double Evaluate(int[][] shape, int row, int col, string type, int rot)
    {
        int holes = CountHolesAfter(shape, row, col);
        int lines = CountLinesCleared(shape, row, col);
        int heightPenalty = row + shape.Length;
        return lines * 100 - holes * 10 - heightPenalty;
    }

    static double AggressiveScore(int[][] shape, int row, int col)
    {
        return Evaluate(shape, row, col, "", 0) + CountLinesCleared(shape, row, col) * 50;
    }

    static double CarefulScore(int[][] shape, int row, int col)
    {
        return Evaluate(shape, row, col, "", 0) - CountHolesAfter(shape, row, col) * 20;
    }

    static int CountHolesAfter(int[][] shape, int row, int col)
    {
        int holes = 0;
        int h = shape.Length, w = shape[0].Length;
        for (int j = 0; j < w; j++)
        {
            bool block = false;
            for (int i = h - 1; i >= 0; i--)
            {
                if (shape[i][j] == 1) block = true;
                else if (block && row + i < height && !board[row + i, col + j]) holes++;
            }
        }
        return holes;
    }

    static int CountLinesCleared(int[][] shape, int row, int col)
    {
        int cleared = 0;
        int h = shape.Length;
        for (int i = 0; i < h; i++)
        {
            int r = row + i;
            if (r >= height) continue;
            bool full = true;
            for (int c = 0; c < width; c++)
            {
                bool occupied = board[r, c];
                for (int j = 0; j < shape[i].Length; j++)
                    if (col + j == c && shape[i][j] == 1) occupied = true;
                if (!occupied) { full = false; break; }
            }
            if (full) cleared++;
        }
        return cleared;
    }

    static int GetStackHeight()
    {
        for (int r = 0; r < height; r++)
            for (int c = 0; c < width; c++)
                if (board[r, c]) return height - r;
        return 0;
    }

    static void UpdateBoardAfterClear(int cleared)
    {
        // Simple: shift rows down (simplified for demo)
        for (int r = height - 1; r >= cleared; r--)
            for (int c = 0; c < width; c++)
                board[r, c] = board[r - cleared, c];
        for (int r = 0; r < cleared; r++)
            for (int c = 0; c < width; c++)
                board[r, c] = false;
    }
}
```
