#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#include <cstdint>
#include <limits>

using namespace std;

using u32 = uint32_t;
using u64 = uint64_t;

bool is_safe(u32 v, u32 c, const vector<vector<u32>>& adj, const vector<u32>& colors) {
    for (u32 nei : adj[v]) {
        if (colors[nei] == c) return false;
    }
    return true;
}

bool backtrack(u32 idx, const vector<u32>& order, const vector<vector<u32>>& adj, 
               vector<u32>& colors, u32 k) {
    if (idx == order.size()) return true;
    
    u32 v = order[idx];
    
    // Try colors in order
    for (u32 c = 0; c < k; ++c) {
        if (is_safe(v, c, adj, colors)) {
            colors[v] = c;
            if (backtrack(idx + 1, order, adj, colors, k)) return true;
            colors[v] = numeric_limits<u32>::max();
        }
    }
    return false;
}

vector<u32> solve_small(const vector<vector<u32>>& adj, u32 k) {
    u32 n = adj.size();
    vector<u32> colors(n, numeric_limits<u32>::max());
    vector<u32> order(n);
    for (u32 i = 0; i < n; ++i) order[i] = i;
    
    // Order by degree descending
    sort(order.begin(), order.end(), [&](u32 a, u32 b) {
        return adj[a].size() > adj[b].size();
    });
    
    backtrack(0, order, adj, colors, k);
    return colors;
}

vector<u32> solve_dsatur(const vector<vector<u32>>& adj, u32 k) {
    u32 n = adj.size();
    vector<u32> colors(n, numeric_limits<u32>::max());
    vector<u32> sat(n, 0);
    vector<u32> deg(n);
    for (u32 i = 0; i < n; ++i) deg[i] = adj[i].size();
    
    auto cmp = [&](u32 a, u32 b) {
        if (sat[a] != sat[b]) return sat[a] < sat[b];
        return deg[a] < deg[b];
    };
    
    priority_queue<u32, vector<u32>, decltype(cmp)> pq(cmp);
    for (u32 i = 0; i < n; ++i) pq.push(i);
    
    vector<bool> done(n, false);
    
    while (!pq.empty()) {
        u32 v = pq.top(); pq.pop();
        if (done[v]) continue;
        
        // Find smallest available color
        vector<bool> used(k, false);
        for (u32 nei : adj[v]) {
            if (colors[nei] < k) used[colors[nei]] = true;
        }
        
        u32 col = 0;
        for (; col < k; ++col) {
            if (!used[col]) break;
        }
        
        if (col >= k) col = 0; // fallback
        colors[v] = col;
        done[v] = true;
        
        // Update saturation of neighbors
        for (u32 nei : adj[v]) {
            if (!done[nei]) {
                // Recalculate saturation (simplified: just mark as changed)
                pq.push(nei);
            }
        }
    }
    
    return colors;
}

vector<u32> solve_greedy(const vector<vector<u32>>& adj, u32 k) {
    u32 n = adj.size();
    vector<u32> colors(n, numeric_limits<u32>::max());
    vector<pair<u32, u32>> vertices(n);
    
    for (u32 i = 0; i < n; ++i) {
        vertices[i] = {adj[i].size(), i};
    }
    sort(vertices.rbegin(), vertices.rend()); // descending degree
    
    for (auto& p : vertices) {
        u32 v = p.second;
        vector<bool> used(k, false);
        for (u32 nei : adj[v]) {
            if (colors[nei] < k) used[colors[nei]] = true;
        }
        for (u32 c = 0; c < k; ++c) {
            if (!used[c]) {
                colors[v] = c;
                break;
            }
        }
        if (colors[v] == numeric_limits<u32>::max()) colors[v] = 0;
    }
    return colors;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    u32 n, m, k;
    cin >> n >> m >> k;
    
    vector<vector<u32>> adj(n);
    for (u32 i = 0; i < m; ++i) {
        u32 u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    
    // Deduplicate edges
    for (auto& lst : adj) {
        sort(lst.begin(), lst.end());
        auto it = unique(lst.begin(), lst.end());
        lst.resize(it - lst.begin());
    }
    
    vector<u32> colors;
    
    if (n <= 300) {
        colors = solve_small(adj, k);
    } else if (n <= 50000) {
        colors = solve_dsatur(adj, k);
    } else {
        colors = solve_greedy(adj, k);
    }
    
    // Ensure all colored
    for (u32 i = 0; i < n; ++i) {
        if (colors[i] == numeric_limits<u32>::max()) colors[i] = 0;
    }
    
    for (u32 i = 0; i < n; ++i) {
        if (i > 0) cout << " ";
        cout << colors[i];
    }
    cout << "\n";
    
    return 0;
}
