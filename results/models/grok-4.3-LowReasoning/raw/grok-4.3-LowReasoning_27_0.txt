```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdint>
#include <limits>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    uint32_t n, m, k;
    std::cin >> n >> m >> k;

    std::vector<std::vector<uint32_t>> adj(n);
    for (uint32_t i = 0; i < m; ++i) {
        uint32_t u, v;
        std::cin >> u >> v;
        if (u != v) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }

    // Remove duplicate neighbors (simple graphs may have multis)
    for (auto& neighbors : adj) {
        std::sort(neighbors.begin(), neighbors.end());
        auto last = std::unique(neighbors.begin(), neighbors.end());
        neighbors.erase(last, neighbors.end());
    }

    std::vector<int> color(n, -1);

    if (n <= 256) {
        // Backtracking for small graphs
        std::vector<uint32_t> order(n);
        for (uint32_t i = 0; i < n; ++i) order[i] = i;

        auto is_safe = [&](uint32_t v, int c) -> bool {
            for (uint32_t nei : adj[v]) {
                if (color[nei] == c) return false;
            }
            return true;
        };

        std::function<bool(uint32_t)> bt = [&](uint32_t idx) -> bool {
            if (idx == n) return true;
            uint32_t v = order[idx];
            for (int c = 0; c < static_cast<int>(k); ++c) {
                if (is_safe(v, c)) {
                    color[v] = c;
                    if (bt(idx + 1)) return true;
                    color[v] = -1;
                }
            }
            return false;
        };

        bt(0);
    } else {
        // Greedy by decreasing degree for larger graphs
        std::vector<std::pair<uint32_t, uint32_t>> verts(n);
        for (uint32_t i = 0; i < n; ++i) {
            verts[i] = {static_cast<uint32_t>(adj[i].size()), i};
        }
        std::sort(verts.rbegin(), verts.rend());

        std::vector<bool> forbidden(k);
        for (auto& p : verts) {
            uint32_t v = p.second;
            std::fill(forbidden.begin(), forbidden.end(), false);
            for (uint32_t nei : adj[v]) {
                if (color[nei] != -1) {
                    forbidden[color[nei]] = true;
                }
            }
            int chosen = -1;
            for (uint32_t c = 0; c < k; ++c) {
                if (!forbidden[c]) {
                    chosen = c;
                    break;
                }
            }
            color[v] = chosen;
        }
    }

    // Output colors
    for (uint32_t i = 0; i < n; ++i) {
        if (i > 0) std::cout << ' ';
        std::cout << (color[i] == -1 ? 0 : color[i]);
    }
    std::cout << '\n';

    return 0;
}
```
