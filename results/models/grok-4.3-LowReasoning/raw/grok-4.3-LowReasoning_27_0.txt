#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdint>
#include <limits>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    uint32_t n = 0, m = 0, k = 0;
    std::cin >> n >> m >> k;

    std::vector<std::vector<uint32_t>> adj(n);
    for (uint32_t i = 0; i < m; ++i) {
        uint32_t u, v;
        std::cin >> u >> v;
        if (u != v) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }

    for (auto& lst : adj) {
        std::sort(lst.begin(), lst.end());
        lst.erase(std::unique(lst.begin(), lst.end()), lst.end());
    }

    std::vector<uint32_t> color(n, std::numeric_limits<uint32_t>::max());

    // Adaptive strategy selection
    auto greedy_color = [&](const std::vector<uint32_t>& order) -> bool {
        for (uint32_t v : order) {
            std::vector<bool> used(k, false);
            for (uint32_t nei : adj[v]) {
                if (color[nei] < k) used[color[nei]] = true;
            }
            uint32_t c = 0;
            for (; c < k; ++c) if (!used[c]) break;
            if (c == k) return false;
            color[v] = c;
        }
        return true;
    };

    bool success = false;

    if (n <= 200) {
        // Simple backtracking for tiny graphs
        std::vector<uint32_t> order(n);
        for (uint32_t i = 0; i < n; ++i) order[i] = i;
        std::sort(order.begin(), order.end(), [&](uint32_t a, uint32_t b) {
            return adj[a].size() > adj[b].size();
        });

        std::function<bool(uint32_t)> bt = [&](uint32_t idx) -> bool {
            if (idx == n) return true;
            uint32_t v = order[idx];
            std::vector<bool> used(k, false);
            for (uint32_t nei : adj[v]) {
                if (color[nei] < k) used[color[nei]] = true;
            }
            for (uint32_t c = 0; c < k; ++c) {
                if (!used[c]) {
                    color[v] = c;
                    if (bt(idx + 1)) return true;
                    color[v] = std::numeric_limits<uint32_t>::max();
                }
            }
            return false;
        };

        success = bt(0);
    } else {
        // Degree-ordered greedy for larger graphs
        std::vector<uint32_t> order(n);
        for (uint32_t i = 0; i < n; ++i) order[i] = i;
        std::sort(order.begin(), order.end(), [&](uint32_t a, uint32_t b) {
            return adj[a].size() > adj[b].size();
        });
        success = greedy_color(order);
    }

    if (!success) {
        // Fallback: natural order greedy
        std::vector<uint32_t> order(n);
        for (uint32_t i = 0; i < n; ++i) order[i] = i;
        success = greedy_color(order);
    }

    // Output
    for (uint32_t i = 0; i < n; ++i) {
        if (i > 0) std::cout << ' ';
        std::cout << (color[i] < k ? color[i] : 0);
    }
    std::cout << '\n';

    return 0;
}
