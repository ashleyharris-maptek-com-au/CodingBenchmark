; SPIR-V
; Version: 1.0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_normal %in_worldPos %ubo %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_normal Location 1
OpDecorate %in_worldPos Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_ubo Uniform
%ptr_in_v3 = OpTypePointer Input %v3float
%in_normal = OpVariable %ptr_in_v3 Input
%in_worldPos = OpVariable %ptr_in_v3 Input
%ptr_out_v4 = OpTypePointer Output %v4float
%outColor = OpVariable %ptr_out_v4 Output
%ptr_uniform_v4 = OpTypePointer Uniform %v4float
%main = OpFunction %void None %func
%entry = OpLabel
%norm_ptr = OpAccessChain %ptr_in_v3 %in_normal
%Nraw = OpLoad %v3float %norm_ptr
%N = OpExtInst %v3float %1 Normalize %Nraw
%wpos_ptr = OpAccessChain %ptr_in_v3 %in_worldPos
%wpos = OpLoad %v3float %wpos_ptr
%light_mem = OpAccessChain %ptr_uniform_v4 %ubo %int_3
%lightPos4 = OpLoad %v4float %light_mem
%lightPos = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2
%Ldir = OpSub %v3float %lightPos %wpos
%L = OpExtInst %v3float %1 Normalize %Ldir
%dotNL = OpDot %float %N %L
%half = OpConstant %float 0.5
%dot05 = OpFMul %float %dotNL %half
%t = OpFAdd %float %dot05 %half
%cool = OpConstantComposite %v3float %float_0_2 %float_0_2 %float_0_75
%warm = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_0_4
%col = OpExtInst %v3float %1 FMix %cool %warm %t
%alpha = OpConstant %float 1.0
%final = OpCompositeConstruct %v4float %col %alpha
OpStore %outColor %final
OpReturn
OpFunctionEnd
%int_3 = OpConstant %int 3
%float_0_2 = OpConstant %float 0.2
%float_0_75 = OpConstant %float 0.75
%float_0_7 = OpConstant %float 0.7
%float_0_4 = OpConstant %float 0.4
