```javascript
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);
const depot = data.depot;
const K = data.truckCount;
const packages = data.packages || [];
if (packages.length === 0 || K <= 0) {
  console.log(JSON.stringify({routes: []}));
  process.exit(0);
}
function angle(p) {
  return Math.atan2(p.y - depot[1], p.x - depot[0]);
}
let pkgs = packages.map(p => ({id: p.id, x: p.x, y: p.y, ang: angle(p)}));
pkgs.sort((a, b) => a.ang - b.ang);
const routes = [];
const n = pkgs.length;
const base = Math.floor(n / K);
let rem = n % K;
let idx = 0;
for (let i = 0; i < K && idx < n; i++) {
  let size = base + (rem > 0 ? 1 : 0);
  if (rem > 0) rem--;
  if (size <= 0) size = 1;
  if (idx + size > n) size = n - idx;
  const group = pkgs.slice(idx, idx + size);
  const route = group.map(g => g.id);
  routes.push(route);
  idx += size;
}
while (routes.length < K) routes.push([]);
console.log(JSON.stringify({routes}));
```
