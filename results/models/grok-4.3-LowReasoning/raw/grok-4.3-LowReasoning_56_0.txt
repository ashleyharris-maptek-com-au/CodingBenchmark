```javascript
const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depot = data.depot;
const truckCount = data.truckCount;
const packages = data.packages;

if (packages.length === 0) {
  console.log(JSON.stringify({ routes: [] }));
  process.exit(0);
}

function angleFromDepot(pkg) {
  const dx = pkg.x - depot[0];
  const dy = pkg.y - depot[1];
  return Math.atan2(dy, dx);
}

const sorted = packages.slice().sort((a, b) => {
  return angleFromDepot(a) - angleFromDepot(b);
});

const routes = [];
const n = sorted.length;
const baseSize = Math.floor(n / truckCount);
let remainder = n % truckCount;
let idx = 0;

for (let i = 0; i < truckCount && idx < n; i++) {
  const size = baseSize + (remainder > 0 ? 1 : 0);
  if (remainder > 0) remainder--;
  const group = sorted.slice(idx, idx + size);
  idx += size;

  if (group.length > 0) {
    const route = group.map(p => p.id);
    routes.push(route);
  }
}

while (routes.length < truckCount && routes.length > 0) {
  // pad with empty if needed, but not necessary
  break;
}

console.log(JSON.stringify({ routes }));
```
