```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 256
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %outColor %ubo
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_gl_PerVertex_in ArrayStride 16
OpDecorate %gl_in DescriptorSet 0
OpDecorate %gl_in Binding 1

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 80
OpMemberDecorate %UBO 3 Offset 96
OpMemberDecorate %UBO 4 Offset 112

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%uint_5 = OpConstant %uint 5
%uint_6 = OpConstant %uint 6

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_3 = OpConstant %float 0.3
%float_thick = OpConstant %float 0.015

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr Input

%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_v3arr = OpTypePointer Input %arr_v3_in
%in_worldPos = OpVariable %ptr_in_v3arr Input
%in_normal = OpVariable %ptr_in_v3arr Input
%in_color = OpVariable %ptr_in_v3arr Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%UBO = OpTypeStruct %v4float %v4float %v4float %v4float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_ubo Uniform

%main = OpFunction %void None %func
%entry = OpLabel

; Load main triangle vertices (0,2,4)
%pos0_ptr = OpAccessChain %ptr_in_v3arr %in_worldPos %int_0
%pos0 = OpLoad %v3float %pos0_ptr
%pos2_ptr = OpAccessChain %ptr_in_v3arr %in_worldPos %int_2
%pos2 = OpLoad %v3float %pos2_ptr
%pos4_ptr = OpAccessChain %ptr_in_v3arr %in_worldPos %int_4
%pos4 = OpLoad %v3float %pos4_ptr

; Adjacent verts (1,3,5)
%adj1_ptr = OpAccessChain %ptr_in_v3arr %in_worldPos %int_1
%adj1 = OpLoad %v3float %adj1_ptr
%adj3_ptr = OpAccessChain %ptr_in_v3arr %in_worldPos %int_3
%adj3 = OpLoad %v3float %adj3_ptr
%adj5_ptr = OpAccessChain %ptr_in_v3arr %in_worldPos %int_5
%adj5 = OpLoad %v3float %adj5_ptr

; Colors
%col0_ptr = OpAccessChain %ptr_in_v3arr %in_color %int_0
%col0 = OpLoad %v3float %col0_ptr
%col2_ptr = OpAccessChain %ptr_in_v3arr %in_color %int_2
%col2 = OpLoad %v3float %col2_ptr
%col4_ptr = OpAccessChain %ptr_in_v3arr %in_color %int_4
%col4 = OpLoad %v3float %col4_ptr

; Dimmed triangle color
%dim = OpConstantComposite %v3float %float_0_3 %float_0_3 %float_0_3
%dim_col0 = OpFMul %v3float %col0 %dim
%dim_col2 = OpFMul %v3float %col2 %dim
%dim_col4 = OpFMul %v3float %col4 %dim

; Emit original triangle (dimmed)
OpStore %outColor %dim_col0
%out0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%pos0_4 = OpCompositeConstruct %v4float %pos0 %float_1
OpStore %out0 %pos0_4
OpEmitVertex

OpStore %outColor %dim_col2
%out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%pos2_4 = OpCompositeConstruct %v4float %pos2 %float_1
OpStore %out2 %pos2_4
OpEmitVertex

OpStore %outColor %dim_col4
%out4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%pos4_4 = OpCompositeConstruct %v4float %pos4 %float_1
OpStore %out4 %pos4_4
OpEmitVertex

OpEndPrimitive

; Edge 0-2 (adjacent 1) - boundary check simplified: always emit for demo
%white = OpConstantComposite %v3float %float_1 %float_1 %float_1
; Simple thick quad using small perpendicular offset in clip XY
%dx = OpFSub %float %float_0 %float_thick
%dy = OpFMul %float %float_thick %float_1

; Emit 4 verts per edge (triangle strip quad)
OpStore %outColor %white
%q0 = OpCompositeConstruct %v4float %pos0 %float_1
OpStore %out0 %q0
OpEmitVertex

OpStore %outColor %white
%q1 = OpCompositeConstruct %v4float %pos2 %float_1
OpStore %out2 %q1
OpEmitVertex

; offset verts (simplified)
%off0 = OpCompositeConstruct %v3float %float_thick %float_0 %float_0
%pos0_off = OpFAdd %v3float %pos0 %off0
%pos0o4 = OpCompositeConstruct %v4float %pos0_off %float_1
OpStore %out0 %pos0o4
OpEmitVertex

%pos2_off = OpFAdd %v3float %pos2 %off0
%pos2o4 = OpCompositeConstruct %v4float %pos2_off %float_1
OpStore %out2 %pos2o4
OpEmitVertex

OpEndPrimitive

; Edge 2-4 (adjacent 3)
OpStore %outColor %white
OpStore %out2 %pos2_4
OpEmitVertex
OpStore %outColor %white
OpStore %out4 %pos4_4
OpEmitVertex
%pos4_off = OpFAdd %v3float %pos4 %off0
%pos4o4 = OpCompositeConstruct %v4float %pos4_off %float_1
OpStore %out4 %pos4o4
OpEmitVertex
OpStore %out2 %pos2o4
OpEmitVertex
OpEndPrimitive

; Edge 4-0 (adjacent 5)
OpStore %outColor %white
OpStore %out4 %pos4_4
OpEmitVertex
OpStore %outColor %white
OpStore %out0 %pos0_4
OpEmitVertex
OpStore %out4 %pos4o4
OpEmitVertex
OpStore %out0 %pos0o4
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
```
