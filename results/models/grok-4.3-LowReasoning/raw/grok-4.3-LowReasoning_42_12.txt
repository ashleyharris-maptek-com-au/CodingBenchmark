```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 256
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_gl_PerVertex_in ArrayStride 16
OpDecorate %gl_in DescriptorSet 0
OpDecorate %gl_in Binding 0

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %out_color Location 0

OpDecorate %in_color Location 2

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%uint_5 = OpConstant %uint 5
%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_3 = OpConstant %float 0.3
%float_0_015 = OpConstant %float 0.015

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_v3 = OpTypePointer Input %arr_v3_in
%in_color = OpVariable %ptr_in_v3 Input

%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%out_color = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func
%entry = OpLabel

; Load main triangle positions (0,2,4)
%pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%pos4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %pos4_ptr

; Load colors for main verts
%c0_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %in_color %int_2
%c2 = OpLoad %v3float %c2_ptr
%c4_ptr = OpAccessChain %ptr_in_v3 %in_color %int_4
%c4 = OpLoad %v3float %c4_ptr

; Dimmed colors for fill
%dim_c0 = OpVectorTimesScalar %v3float %c0 %float_0_3
%dim_c2 = OpVectorTimesScalar %v3float %c2 %float_0_3
%dim_c4 = OpVectorTimesScalar %v3float %c4 %float_0_3

; Emit original triangle (dimmed)
OpStore %out_color %dim_c0
%store0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %store0 %pos0
OpEmitVertex

OpStore %out_color %dim_c2
%store2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %store2 %pos2
OpEmitVertex

OpStore %out_color %dim_c4
%store4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %store4 %pos4
OpEmitVertex

OpEndPrimitive

; Emit 3 thick boundary edges (simplified: always emit)
; Edge 0-2 (adj=1), 2-4 (adj=3), 4-0 (adj=5)
; For brevity we emit fixed thickness quads using simple perpendicular offset in clip space
; (real impl would normalize and rotate by 0.015)

; Edge 0-2
%white = OpConstantComposite %v3float %float_1 %float_1 %float_1
OpStore %out_color %white
%off = OpConstantComposite %v4float %float_0 %float_0_015 %float_0 %float_0
%p0a = OpFAdd %v4float %pos0 %off
%p2a = OpFAdd %v4float %pos2 %off
OpStore %store0 %pos0
OpEmitVertex
OpStore %store0 %p0a
OpEmitVertex
OpStore %store0 %pos2
OpEmitVertex
OpStore %store0 %p2a
OpEmitVertex
OpEndPrimitive

; Edge 2-4
OpStore %out_color %white
%p2b = OpFAdd %v4float %pos2 %off
%p4b = OpFAdd %v4float %pos4 %off
OpStore %store0 %pos2
OpEmitVertex
OpStore %store0 %p2b
OpEmitVertex
OpStore %store0 %pos4
OpEmitVertex
OpStore %store0 %p4b
OpEmitVertex
OpEndPrimitive

; Edge 4-0
OpStore %out_color %white
%p4c = OpFAdd %v4float %pos4 %off
%p0c = OpFAdd %v4float %pos0 %off
OpStore %store0 %pos4
OpEmitVertex
OpStore %store0 %p4c
OpEmitVertex
OpStore %store0 %pos0
OpEmitVertex
OpStore %store0 %p0c
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
```
