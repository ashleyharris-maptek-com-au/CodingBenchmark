; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_ubo Uniform
%ptr_in_v3 = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_in_v3 Input
%normal = OpVariable %ptr_in_v3 Input
%ptr_out_v4 = OpTypePointer Output %v4float
%outColor = OpVariable %ptr_out_v4 Output
%ptr_ubo_v4 = OpTypePointer Uniform %v4float
%float_0 = OpConstant %float 0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%main = OpFunction %void None
%entry = OpLabel
%camPosPtr = OpAccessChain %ptr_ubo_v4 %ubo %int_4
%camPos = OpLoad %v4float %camPosPtr
%cam3 = OpVectorShuffle %v3float %camPos %camPos 0 1 2
%wp = OpLoad %v3float %worldPos
%V = OpSub %v3float %cam3 %wp
%Vlen = OpExtInst %float %glsl_normalize_length %V
%Vnorm = OpExtInst %v3float %glsl_normalize %V
%Nin = OpLoad %v3float %normal
%Nnorm = OpExtInst %v3float %glsl_normalize %Nin
%dotNV = OpDot %float %Nnorm %Vnorm
%NdotV = OpExtInst %float %glsl_fmax %dotNV %float_0
%cmp = OpFOrdLessThan %bool %NdotV %float_0_5
%halfN = OpFMul %float %NdotV %float_2
%elseVal = OpFSub %float %NdotV %float_0_5
%elseScale = OpFMul %float %elseVal %float_2
%elseR = OpFSub %float %float_1 %elseScale
%elseB = OpFMul %float %elseVal %float_2
%R = OpSelect %float %cmp %float_1 %elseR
%G = OpSelect %float %cmp %halfN %elseR
%B = OpSelect %float %cmp %float_0 %elseB
%outVec = OpCompositeConstruct %v4float %R %G %B %float_1
OpStore %outColor %outVec
OpReturn
OpFunctionEnd
%glsl_normalize = OpExtInstImport "GLSL.std.450"
%glsl_normalize_length = OpExtInstImport "GLSL.std.450"
%glsl_fmax = OpExtInstImport "GLSL.std.450"
%bool = OpTypeBool
%int = OpTypeInt 32 1
%int_4 = OpConstant %int 4
