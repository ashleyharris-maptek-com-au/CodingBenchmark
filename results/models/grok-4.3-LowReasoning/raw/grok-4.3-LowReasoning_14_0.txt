```csharp
using System;
using System.Collections.Generic;
using System.Linq;

public class Program
{
    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split();
        int H = int.Parse(firstLine[0]);
        int W = int.Parse(firstLine[1]);
        string[] board = new string[H];
        for (int i = 0; i < H; i++)
            board[i] = Console.ReadLine();

        bool[,] isSafe = new bool[H, W];
        bool[,] isMine = new bool[H, W];
        int[] dx = { -1, -1, -1, 0, 0, 1, 1, 1 };
        int[] dy = { -1, 0, 1, -1, 1, -1, 0, 1 };

        bool changed = true;
        while (changed)
        {
            changed = false;
            for (int y = 0; y < H; y++)
            {
                for (int x = 0; x < W; x++)
                {
                    char c = board[y][x];
                    if (c < '0' || c > '8') continue;
                    int val = c - '0';

                    int flagged = 0, unknown = 0;
                    List<(int, int)> unkList = new List<(int, int)>();
                    for (int d = 0; d < 8; d++)
                    {
                        int ny = y + dy[d], nx = x + dx[d];
                        if (ny < 0 || ny >= H || nx < 0 || nx >= W) continue;
                        if (isMine[ny, nx] || board[ny][nx] == '*') flagged++;
                        else if (!isSafe[ny, nx] && board[ny][nx] == ' ')
                        {
                            unknown++;
                            unkList.Add((nx, ny));
                        }
                    }

                    if (val == flagged && unknown > 0)
                    {
                        foreach (var (nx, ny) in unkList)
                        {
                            if (!isSafe[ny, nx])
                            {
                                isSafe[ny, nx] = true;
                                changed = true;
                            }
                        }
                    }
                    else if (val == flagged + unknown && unknown > 0)
                    {
                        foreach (var (nx, ny) in unkList)
                        {
                            if (!isMine[ny, nx])
                            {
                                isMine[ny, nx] = true;
                                changed = true;
                            }
                        }
                    }
                }
            }
        }

        List<(int, int)> result = new List<(int, int)>();
        for (int y = 0; y < H; y++)
            for (int x = 0; x < W; x++)
                if (board[y][x] == ' ' && isSafe[y, x])
                    result.Add((x, y));

        Console.WriteLine(result.Count);
        foreach (var (x, y) in result)
            Console.WriteLine($"{x} {y}");
    }
}
```
