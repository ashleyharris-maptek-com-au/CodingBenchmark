```csharp
using System;
using System.Collections.Generic;
using System.Text;

public class Program
{
    static void Main()
    {
        string[] first = Console.ReadLine().Split();
        int H = int.Parse(first[0]);
        int W = int.Parse(first[1]);
        string[] board = new string[H];
        for (int i = 0; i < H; i++)
            board[i] = Console.ReadLine();

        bool large = (long)H * W > 200000;
        HashSet<(int x, int y)> safes = new HashSet<(int, int)>();

        int[] dr = { -1, -1, -1, 0, 0, 1, 1, 1 };
        int[] dc = { -1, 0, 1, -1, 1, -1, 0, 1 };

        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                char ch = board[r][c];
                if (ch < '0' || ch > '8') continue;
                int num = ch - '0';

                int knownMines = 0;
                List<(int x, int y)> unknowns = new List<(int, int)>();

                for (int d = 0; d < 8; d++)
                {
                    int nr = r + dr[d];
                    int nc = c + dc[d];
                    if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;
                    char nch = board[nr][nc];
                    if (nch == '*') knownMines++;
                    else if (nch == ' ') unknowns.Add((nc, nr));
                }

                if (knownMines == num)
                {
                    foreach (var p in unknowns)
                        safes.Add(p);
                }
            }
        }

        // For larger boards skip iterative passes to keep linear time
        if (!large && safes.Count > 0)
        {
            // one additional pass for very small boards only
            for (int r = 0; r < H; r++)
            {
                for (int c = 0; c < W; c++)
                {
                    char ch = board[r][c];
                    if (ch < '0' || ch > '8') continue;
                    int num = ch - '0';

                    int knownMines = 0;
                    List<(int x, int y)> unknowns = new List<(int, int)>();

                    for (int d = 0; d < 8; d++)
                    {
                        int nr = r + dr[d];
                        int nc = c + dc[d];
                        if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;
                        char nch = board[nr][nc];
                        if (nch == '*') knownMines++;
                        else if (nch == ' ' && !safes.Contains((nc, nr)))
                            unknowns.Add((nc, nr));
                    }

                    if (knownMines == num)
                    {
                        foreach (var p in unknowns)
                            safes.Add(p);
                    }
                }
            }
        }

        Console.WriteLine(safes.Count);
        foreach (var (x, y) in safes)
            Console.WriteLine($"{x} {y}");
    }
}
```
