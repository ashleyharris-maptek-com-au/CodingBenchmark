```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading;

class MaxClique
{
    static int n;
    static List<int>[] adj;
    static bool[][] conn;
    static int maxSize = 0;
    static List<int> bestClique = new List<int>();
    static object lockObj = new object();
    static DateTime startTime;
    static TimeSpan timeLimit = TimeSpan.FromMinutes(4.5);

    static void Main()
    {
        startTime = DateTime.Now;
        var input = Console.In.ReadToEnd().Split(new[] { ' ', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        n = int.Parse(input[idx++]);
        int m = int.Parse(input[idx++]);

        adj = new List<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new List<int>();

        var edgeSet = new HashSet<long>();
        for (int i = 0; i < m; i++)
        {
            int u = int.Parse(input[idx++]);
            int v = int.Parse(input[idx++]);
            if (u == v) continue;
            long key = ((long)Math.Min(u, v) << 32) | Math.Max(u, v);
            if (edgeSet.Add(key))
            {
                adj[u].Add(v);
                adj[v].Add(u);
            }
        }

        for (int i = 0; i < n; i++)
        {
            adj[i] = adj[i].Distinct().OrderBy(x => x).ToList();
        }

        if (n <= 64)
        {
            conn = new bool[n][];
            for (int i = 0; i < n; i++)
            {
                conn[i] = new bool[n];
                foreach (var j in adj[i]) conn[i][j] = true;
            }
            ExactSearchBit(0, 0, new List<int>());
        }
        else
        {
            // Heuristic for large n: greedy degree-based
            var vertices = Enumerable.Range(0, n).OrderByDescending(v => adj[v].Count).ToList();
            var clique = new List<int>();
            foreach (var v in vertices)
            {
                bool canAdd = true;
                foreach (var u in clique)
                {
                    if (!adj[v].Contains(u)) { canAdd = false; break; }
                }
                if (canAdd) clique.Add(v);
                if (DateTime.Now - startTime > timeLimit) break;
            }
            lock (lockObj)
            {
                if (clique.Count > maxSize)
                {
                    maxSize = clique.Count;
                    bestClique = new List<int>(clique);
                }
            }
        }

        // Output
        Console.WriteLine(maxSize);
        if (maxSize > 0)
            Console.WriteLine(string.Join(" ", bestClique.OrderBy(x => x)));
        else
            Console.WriteLine();
    }

    static void ExactSearchBit(int pos, int size, List<int> current)
    {
        if (DateTime.Now - startTime > timeLimit) return;
        if (size > maxSize)
        {
            lock (lockObj)
            {
                if (size > maxSize)
                {
                    maxSize = size;
                    bestClique = new List<int>(current);
                }
            }
        }
        if (pos >= n) return;

        // Upper bound pruning
        int remaining = n - pos;
        if (size + remaining <= maxSize) return;

        // Try including pos
        bool canInclude = true;
        foreach (var u in current)
        {
            if (!conn[u][pos]) { canInclude = false; break; }
        }
        if (canInclude)
        {
            current.Add(pos);
            ExactSearchBit(pos + 1, size + 1, current);
            current.RemoveAt(current.Count - 1);
        }

        // Skip pos
        ExactSearchBit(pos + 1, size, current);
    }
}
```
