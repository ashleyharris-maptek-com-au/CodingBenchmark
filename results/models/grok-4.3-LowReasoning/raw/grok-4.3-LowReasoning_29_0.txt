```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

class MaxClique
{
    static int n;
    static bool[,] adj;
    static List<int> bestClique = new List<int>();
    static object lockObj = new object();
    static CancellationTokenSource cts = new CancellationTokenSource();
    static DateTime startTime;

    static void Main()
    {
        startTime = DateTime.Now;
        var input = Console.In.ReadToEnd().Split(new[] { ' ', '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        n = int.Parse(input[idx++]);
        int m = int.Parse(input[idx++]);

        if (n > 10000)
        {
            // Heuristic for large graphs: greedy by degree
            var graph = new List<int>[n];
            for (int i = 0; i < n; i++) graph[i] = new List<int>();
            for (int i = 0; i < m; i++)
            {
                int u = int.Parse(input[idx++]);
                int v = int.Parse(input[idx++]);
                graph[u].Add(v);
                graph[v].Add(u);
            }
            for (int i = 0; i < n; i++) graph[i] = graph[i].Distinct().OrderByDescending(x => graph[x].Count).ToList();

            var clique = GreedyClique(graph);
            Console.WriteLine(clique.Count);
            Console.WriteLine(string.Join(" ", clique));
            return;
        }

        adj = new bool[n, n];
        for (int i = 0; i < m; i++)
        {
            int u = int.Parse(input[idx++]);
            int v = int.Parse(input[idx++]);
            adj[u, v] = true;
            adj[v, u] = true;
        }
        for (int i = 0; i < n; i++) adj[i, i] = true;

        // Parallel branch and bound from high degree vertices
        var degrees = Enumerable.Range(0, n).OrderByDescending(i => Enumerable.Range(0, n).Count(j => adj[i, j])).ToList();
        int maxThreads = Math.Min(Environment.ProcessorCount, 16);
        Parallel.For(0, Math.Min(maxThreads, degrees.Count), new ParallelOptions { CancellationToken = cts.Token }, i =>
        {
            if ((DateTime.Now - startTime).TotalMinutes > 4.5) { cts.Cancel(); return; }
            var start = new List<int> { degrees[i] };
            var candidates = Enumerable.Range(0, n).Where(v => adj[degrees[i], v] && v != degrees[i]).ToList();
            Search(start, candidates);
        });

        Console.WriteLine(bestClique.Count);
        Console.WriteLine(string.Join(" ", bestClique.OrderBy(x => x)));
    }

    static void Search(List<int> clique, List<int> candidates)
    {
        if (cts.IsCancellationRequested) return;
        if (clique.Count > bestClique.Count)
        {
            lock (lockObj)
            {
                if (clique.Count > bestClique.Count)
                    bestClique = new List<int>(clique);
            }
        }
        if ((DateTime.Now - startTime).TotalMinutes > 4.5) { cts.Cancel(); return; }
        if (clique.Count + candidates.Count <= bestClique.Count) return;

        for (int i = 0; i < candidates.Count; i++)
        {
            int v = candidates[i];
            var newClique = new List<int>(clique) { v };
            var newCands = candidates.Skip(i + 1).Where(u => adj[v, u]).ToList();
            Search(newClique, newCands);
        }
    }

    static List<int> GreedyClique(List<int>[] graph)
    {
        var clique = new List<int>();
        var used = new bool[n];
        var verts = Enumerable.Range(0, n).OrderByDescending(v => graph[v].Count).ToList();
        foreach (var v in verts)
        {
            if (used[v]) continue;
            bool canAdd = true;
            foreach (var u in clique)
                if (!graph[v].Contains(u)) { canAdd = false; break; }
            if (canAdd)
            {
                clique.Add(v);
                foreach (var nei in graph[v]) used[nei] = false; // reset not needed
                used[v] = true;
            }
        }
        return clique;
    }
}
```
