```csharp
using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        string line = Console.ReadLine();
        if (string.IsNullOrEmpty(line)) return;
        var parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int n = int.Parse(parts[0]);
        int m = int.Parse(parts[1]);

        var adj = new HashSet<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new HashSet<int>();

        for (int i = 0; i < m; i++)
        {
            line = Console.ReadLine();
            if (string.IsNullOrEmpty(line)) continue;
            parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length < 2) continue;
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            if (u != v)
            {
                adj[u].Add(v);
                adj[v].Add(u);
            }
        }

        if (n == 0)
        {
            Console.WriteLine(0);
            return;
        }

        var vertices = Enumerable.Range(0, n)
            .OrderByDescending(v => adj[v].Count)
            .ToArray();

        var clique = new List<int>();
        foreach (int v in vertices)
        {
            bool canAdd = true;
            foreach (int c in clique)
            {
                if (!adj[v].Contains(c))
                {
                    canAdd = false;
                    break;
                }
            }
            if (canAdd) clique.Add(v);
        }

        Console.WriteLine(clique.Count);
        Console.WriteLine(string.Join(" ", clique));
    }
}
```
