```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%uint_2 = OpConstant %uint 2
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_gl_PerVertex = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr_gl_PerVertex Input
%arr_v3float_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3float_2
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_PerVertex Output
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%main = OpFunction %void None %func
%entry = OpLabel
; Load p0, p1 (clip-space positions)
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
; Load colors
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %c1_ptr
; dir = p1 - p0 (use xy only for 2D perp)
%dir4 = OpVectorSubtract %v4float %p1 %p0
%dir = OpVectorShuffle %v3float %dir4 %dir4 0 1 2
%neg_dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %neg_dir_y
%dir_x = OpCompositeExtract %float %dir 0
%perp_vec = OpCompositeConstruct %v3float %neg_dir_y %dir_x %float_0
%perp = OpExtInst %v3float %1 Normalize %perp_vec
; thickness constants
%thick = OpFMul %v3float %perp %float_0_015
%arrow_thick = OpFMul %v3float %perp %float_0_05
; mid = p0 + 0.75*(p1-p0)
%delta = OpVectorSubtract %v4float %p1 %p0
%scale = OpVectorTimesScalar %v4float %delta %float_0_75
%mid4 = OpVectorAdd %v4float %p0 %scale
; ===== SHAFT (4 vertices) =====
; v0: p0 + thick
%v0 = OpVectorAdd %v4float %p0 %thick
%out0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out0 %v0
OpStore %outColor %c0
OpEmitVertex
; v1: p0 - thick
%v1 = OpVectorSubtract %v4float %p0 %thick
%out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out1 %v1
OpStore %outColor %c0
OpEmitVertex
; v2: mid + thick
%v2 = OpVectorAdd %v4float %mid4 %thick
%out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out2 %v2
OpStore %outColor %c0
OpEmitVertex
; v3: mid - thick
%v3 = OpVectorSubtract %v4float %mid4 %thick
%out3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out3 %v3
OpStore %outColor %c0
OpEmitVertex
OpEndPrimitive
; ===== ARROWHEAD (3 vertices) =====
; a0: mid + arrow_thick
%a0 = OpVectorAdd %v4float %mid4 %arrow_thick
%outa0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outa0 %a0
OpStore %outColor %c1
OpEmitVertex
; a1: mid - arrow_thick
%a1 = OpVectorSubtract %v4float %mid4 %arrow_thick
%outa1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outa1 %a1
OpStore %outColor %c1
OpEmitVertex
; a2: p1 (tip)
%outa2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outa2 %p1
OpStore %outColor %c1
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
```
