%mid = OpFAdd %v3float %p0 %scaled_dir

%shaft_off = OpVectorTimesScalar %v3float %perp %float_0_015
%arrow_off = OpVectorTimesScalar %v3float %perp %float_0_05

%v1_pos3 = OpFAdd %v3float %p0 %shaft_off
%v1_pos4 = OpCompositeConstruct %v4float %v1_pos3 %float_1

; v3: mid + shaft_off
%v3_pos3 = OpFAdd %v3float %mid %shaft_off
%v3_pos4 = OpCompositeConstruct %v4float %v3_pos3 %float_1
%v3_chain = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v3_chain %v3_pos4
%oc3_chain = OpAccessChain %ptr_out_v3 %outColor

OpEndPrimitive

; === Arrowhead (3 vertices) ===
; a0: mid + arrow_off
%a0_pos3 = OpFAdd %v3float %mid %arrow_off

%a0_chain = OpAccessChain %ptr_out_v4 %gl_out %int_0
