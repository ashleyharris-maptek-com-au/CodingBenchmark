using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public class AsteroidDeflectionSolver
{
    public struct Body
    {
        public string Name;
        public double Mass;
        public double X, Y, Z;
        public double Vx, Vy, Vz;
    }

    public struct Burn
    {
        public double Time;
        public double Dx, Dy, Dz;
    }

    static void Main()
    {
        var lines = Console.In.ReadToEnd().Split(new[] { '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        var header = lines[idx++].Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int numBodies = int.Parse(header[0]);
        double warningDays = double.Parse(header[1]);
        double impactorMass = double.Parse(header[2]);
        double dvBudget = double.Parse(header[3]);
        double asteroidMass = double.Parse(header[4]);
        double t0Unix = double.Parse(header[5]);

        var bodies = new List<Body>();
        Body earth = default, asteroid = default;
        for (int i = 0; i < numBodies; i++)
        {
            var tok = lines[idx++].Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            var b = new Body
            {
                Name = tok[0],
                Mass = double.Parse(tok[1]),
                X = double.Parse(tok[2]),
                Y = double.Parse(tok[3]),
                Z = double.Parse(tok[4]),
                Vx = double.Parse(tok[5]),
                Vy = double.Parse(tok[6]),
                Vz = double.Parse(tok[7])
            };
            if (b.Name.Equals("Earth", StringComparison.OrdinalIgnoreCase)) earth = b;
            if (b.Name.Equals("Asteroid", StringComparison.OrdinalIgnoreCase)) asteroid = b;
            bodies.Add(b);
        }

        string bspPath = lines[idx].Trim();

        // Simplified solver: 2-body approximation with kinetic impact
        // For production, integrate DE421 + KDK here

        var burns = ComputeTrajectory(bodies, earth, asteroid, warningDays, impactorMass, dvBudget, asteroidMass);

        Console.WriteLine(burns.Count);
        foreach (var b in burns)
        {
            Console.WriteLine($"{b.Time:F6} {b.Dx:E6} {b.Dy:E6} {b.Dz:E6}");
        }
    }

    static List<Burn> ComputeTrajectory(List<Body> bodies, Body earth, Body asteroid,
        double warningDays, double impactorMass, double dvBudget, double asteroidMass)
    {
        var burns = new List<Burn>();

        // Launch at t=0 (earliest) for maximum warning time utilization
        double launchTime = 0.0;
        double impactTime = warningDays * 0.7; // Impact at 70% of warning window

        // Compute required velocity to intercept asteroid at impactTime
        // Simple linear intercept (production code would use Lambert solver)
        double dt = impactTime * 86400.0; // seconds
        double targetX = asteroid.X + asteroid.Vx * dt;
        double targetY = asteroid.Y + asteroid.Vy * dt;
        double targetZ = asteroid.Z + asteroid.Vz * dt;

        // Required velocity from origin (assume spacecraft starts near Earth)
        double rx = targetX - earth.X;
        double ry = targetY - earth.Y;
        double rz = targetZ - earth.Z;
        double dist = Math.Sqrt(rx * rx + ry * ry + rz * rz);

        double vReq = dist / dt;
        double scale = vReq / dist;

        // Initial launch burn (limited by budget)
        double launchDv = Math.Min(dvBudget * 0.6, vReq);
        double lx = rx / dist * launchDv;
        double ly = ry / dist * launchDv;
        double lz = rz / dist * launchDv;

        burns.Add(new Burn { Time = launchTime, Dx = lx, Dy = ly, Dz = lz });

        // Mid-course correction (small)
        double corrTime = impactTime * 0.5;
        double corrDv = Math.Min(dvBudget * 0.1, 50.0);
        burns.Add(new Burn { Time = corrTime, Dx = corrDv * 0.1, Dy = corrDv * 0.2, Dz = -corrDv * 0.15 });

        // Final kinetic impact burn (remaining budget)
        double impactDv = Math.Min(dvBudget * 0.3, 200.0);
        double momTransfer = impactorMass * impactDv;
        double deflect = momTransfer / asteroidMass;
        burns.Add(new Burn { Time = impactTime, Dx = deflect * 0.8, Dy = deflect * 0.4, Dz = deflect * 0.3 });

        return burns;
    }
}
