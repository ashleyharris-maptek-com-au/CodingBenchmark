using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public class AsteroidDeflectionSolver
{
    public struct Body
    {
        public string Name;
        public double Mass;
        public double X, Y, Z;
        public double Vx, Vy, Vz;
    }

    public struct Burn
    {
        public double Time; // days since t=0
        public double Dx, Dy, Dz; // m/s
    }

    static void Main()
    {
        var input = Console.In.ReadToEnd().Split(new[] { ' ', '\t', '\r', '\n' }, 
            StringSplitOptions.RemoveEmptyEntries);
        
        int idx = 0;
        int numBodies = int.Parse(input[idx++]);
        double warningDays = double.Parse(input[idx++], CultureInfo.InvariantCulture);
        double impactorMass = double.Parse(input[idx++], CultureInfo.InvariantCulture);
        double deltaVBudget = double.Parse(input[idx++], CultureInfo.InvariantCulture);
        double asteroidMass = double.Parse(input[idx++], CultureInfo.InvariantCulture);
        double t0Unix = double.Parse(input[idx++], CultureInfo.InvariantCulture);
        string de421Path = "";

        var bodies = new List<Body>();
        for (int i = 0; i < numBodies; i++)
        {
            var b = new Body
            {
                Name = input[idx++],
                Mass = double.Parse(input[idx++], CultureInfo.InvariantCulture),
                X = double.Parse(input[idx++], CultureInfo.InvariantCulture),
                Y = double.Parse(input[idx++], CultureInfo.InvariantCulture),
                Z = double.Parse(input[idx++], CultureInfo.InvariantCulture),
                Vx = double.Parse(input[idx++], CultureInfo.InvariantCulture),
                Vy = double.Parse(input[idx++], CultureInfo.InvariantCulture),
                Vz = double.Parse(input[idx++], CultureInfo.InvariantCulture)
            };
            bodies.Add(b);
        }
        de421Path = input[idx];

        // Locate asteroid and Earth
        var asteroid = bodies.First(b => b.Name.Equals("Asteroid", StringComparison.OrdinalIgnoreCase));
        var earth = bodies.First(b => b.Name.Equals("Earth", StringComparison.OrdinalIgnoreCase));
        
        // Simple intercept strategy: direct transfer at optimal time within window
        // For complex multi-flyby cases, replace this with optimization routine
        var solution = ComputeDirectIntercept(asteroid, earth, warningDays, deltaVBudget, asteroidMass);
        
        // Output
        Console.WriteLine(solution.Count);
        foreach (var burn in solution)
        {
            Console.WriteLine(string.Format(CultureInfo.InvariantCulture, 
                "{0:F6} {1:E6} {2:E6} {3:E6}", burn.Time, burn.Dx, burn.Dy, burn.Dz));
        }
    }

    static List<Burn> ComputeDirectIntercept(Body asteroid, Body earth, 
        double warningDays, double dvBudget, double asteroidMass)
    {
        var burns = new List<Burn>();
        
        // Launch at t=0 for maximum warning time utilization (conservative)
        double launchTime = 0.0;
        
        // Compute required velocity to intercept asteroid at mid-warning time
        double interceptTime = warningDays * 0.6; // 60% of warning for approach margin
        
        // Relative position at intercept
        double dt = interceptTime * 86400.0; // convert to seconds
        double ax = asteroid.X + asteroid.Vx * dt;
        double ay = asteroid.Y + asteroid.Vy * dt;
        double az = asteroid.Z + asteroid.Vz * dt;
        
        // Required velocity from origin (simplified - assumes spacecraft starts near Earth)
        double rx = ax - earth.X;
        double ry = ay - earth.Y;
        double rz = az - earth.Z;
        double dist = Math.Sqrt(rx*rx + ry*ry + rz*rz);
        
        // Average velocity needed (m/s)
        double vNeeded = dist / dt;
        
        // Direction unit vector
        double ux = rx / dist;
        double uy = ry / dist;
        double uz = rz / dist;
        
        // Apply delta-v (assume we start with Earth's velocity + burn)
        double burnMag = Math.Min(vNeeded * 0.3, dvBudget); // fraction for realistic transfer
        
        burns.Add(new Burn 
        { 
            Time = launchTime, 
            Dx = ux * burnMag * 0.6, 
            Dy = uy * burnMag * 0.6, 
            Dz = uz * burnMag * 0.6 
        });
        
        // Mid-course correction at 50% time
        burns.Add(new Burn 
        { 
            Time = interceptTime * 0.5, 
            Dx = ux * burnMag * 0.2, 
            Dy = uy * burnMag * 0.2, 
            Dz = uz * burnMag * 0.2 
        });
        
        // Final approach / impact burn (kinetic impactor)
        double impactDv = Math.Min(500.0, dvBudget - burnMag * 0.8); // up to 500 m/s impact
        burns.Add(new Burn 
        { 
            Time = interceptTime, 
            Dx = ux * impactDv, 
            Dy = uy * impactDv, 
            Dz = uz * impactDv 
        });
        
        return burns;
    }
}
