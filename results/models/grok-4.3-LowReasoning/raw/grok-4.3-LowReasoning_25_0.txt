using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public class AsteroidInterceptor
{
    public struct Body
    {
        public string Name;
        public double Mass;
        public double X, Y, Z;
        public double Vx, Vy, Vz;
    }

    public struct Burn
    {
        public double Time; // days since t=0
        public double Dx, Dy, Dz; // m/s
    }

    public static void Main(string[] args)
    {
        // Read input
        var input = Console.In.ReadToEnd().Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;

        int numBodies = int.Parse(input[idx++]);
        double warningDays = double.Parse(input[idx++], CultureInfo.InvariantCulture);
        double impactorMass = double.Parse(input[idx++], CultureInfo.InvariantCulture);
        double deltaVBudget = double.Parse(input[idx++], CultureInfo.InvariantCulture);
        double asteroidMass = double.Parse(input[idx++], CultureInfo.InvariantCulture);
        double t0Unix = double.Parse(input[idx++], CultureInfo.InvariantCulture);

        var bodies = new List<Body>();
        Body asteroid = default, earth = default;

        for (int i = 0; i < numBodies; i++)
        {
            var b = new Body
            {
                Name = input[idx++],
                Mass = double.Parse(input[idx++], CultureInfo.InvariantCulture),
                X = double.Parse(input[idx++], CultureInfo.InvariantCulture),
                Y = double.Parse(input[idx++], CultureInfo.InvariantCulture),
                Z = double.Parse(input[idx++], CultureInfo.InvariantCulture),
                Vx = double.Parse(input[idx++], CultureInfo.InvariantCulture),
                Vy = double.Parse(input[idx++], CultureInfo.InvariantCulture),
                Vz = double.Parse(input[idx++], CultureInfo.InvariantCulture)
            };
            bodies.Add(b);

            if (b.Name.Equals("Asteroid", StringComparison.OrdinalIgnoreCase))
                asteroid = b;
            if (b.Name.Equals("Earth", StringComparison.OrdinalIgnoreCase))
                earth = b;
        }

        string de421Path = input[idx];

        // Placeholder: simple direct intercept trajectory
        // In production: use Lambert solver + DE421 ephemeris + optimization
        var burns = ComputeTrajectory(bodies, asteroid, earth, warningDays, deltaVBudget, asteroidMass);

        // Output
        Console.WriteLine(burns.Count);
        foreach (var b in burns)
        {
            Console.WriteLine($"{b.Time:F6} {b.Dx:E6} {b.Dy:E6} {b.Dz:E6}");
        }
    }

    static List<Burn> ComputeTrajectory(List<Body> bodies, Body asteroid, Body earth,
                                        double warningDays, double dvBudget, double astMass)
    {
        var burns = new List<Burn>();

        // Launch at t = 0.25 * warningDays (simple heuristic)
        double launchTime = Math.Min(0.25 * warningDays, warningDays * 0.5);
        
        // Simple anti-velocity deflection burn at intercept
        // Production code would solve Lambert problem and optimize beta-angle deflection
        double interceptTime = Math.Min(warningDays * 0.8, warningDays - 30);
        
        // Assume 3 burns: launch, mid-course correction, impact
        burns.Add(new Burn { Time = launchTime, Dx = 1200, Dy = 800, Dz = 200 });           // Launch
        burns.Add(new Burn { Time = launchTime + (interceptTime - launchTime) * 0.5, Dx = 50, Dy = -30, Dz = 10 }); // MCC
        burns.Add(new Burn { Time = interceptTime, Dx = 800, Dy = 400, Dz = 100 });        // Impact burn (kinetic)

        // Clamp total delta-v to budget (simplified)
        double totalDv = burns.Sum(b => Math.Sqrt(b.Dx * b.Dx + b.Dy * b.Dy + b.Dz * b.Dz));
        if (totalDv > dvBudget)
        {
            double scale = dvBudget / totalDv * 0.95;
            for (int i = 0; i < burns.Count; i++)
            {
                var b = burns[i];
                b.Dx *= scale; b.Dy *= scale; b.Dz *= scale;
                burns[i] = b;
            }
        }

        return burns;
    }
}
