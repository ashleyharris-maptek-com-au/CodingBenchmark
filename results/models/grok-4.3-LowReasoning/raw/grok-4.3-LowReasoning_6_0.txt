using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        var input = Console.In.ReadToEnd().Split(new[] { '\r', '\n', ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        int N = int.Parse(input[idx++]);
        double mu = double.Parse(input[idx++]);

        // Read start (ignored for heuristic)
        idx += 6;

        var stations = new List<double[]>();
        for (int i = 0; i < N; i++)
        {
            var pos = new double[3];
            for (int j = 0; j < 3; j++) pos[j] = double.Parse(input[idx++]);
            idx += 3; // skip velocity
            stations.Add(pos);
        }

        int[] order;
        if (N <= 8)
        {
            order = SolveExact(stations);
        }
        else
        {
            order = SolveNearestNeighbor(stations);
        }

        Console.WriteLine(string.Join(" ", order));
    }

    static int[] SolveExact(List<double[]> stations)
    {
        int N = stations.Count;
        var indices = Enumerable.Range(0, N).ToArray();
        int[] bestOrder = null;
        double bestCost = double.MaxValue;

        foreach (var perm in GetPermutations(indices, N))
        {
            double cost = 0;
            var prev = stations[perm[0]];
            for (int i = 1; i < N; i++)
            {
                var curr = stations[perm[i]];
                cost += Distance(prev, curr);
                prev = curr;
            }
            if (cost < bestCost)
            {
                bestCost = cost;
                bestOrder = perm.ToArray();
            }
        }
        return bestOrder;
    }

    static IEnumerable<int[]> GetPermutations(int[] arr, int n)
    {
        if (n == 1)
        {
            yield return (int[])arr.Clone();
            yield break;
        }
        for (int i = 0; i < n; i++)
        {
            foreach (var perm in GetPermutations(arr, n - 1))
            {
                yield return perm;
            }
            // Rotate
            int temp = arr[0];
            for (int j = 0; j < n - 1; j++) arr[j] = arr[j + 1];
            arr[n - 1] = temp;
        }
    }

    static int[] SolveNearestNeighbor(List<double[]> stations)
    {
        int N = stations.Count;
        var visited = new bool[N];
        var order = new List<int>();
        int current = 0;
        order.Add(current);
        visited[current] = true;

        for (int i = 1; i < N; i++)
        {
            double minDist = double.MaxValue;
            int next = -1;
            for (int j = 0; j < N; j++)
            {
                if (!visited[j])
                {
                    double d = Distance(stations[current], stations[j]);
                    if (d < minDist)
                    {
                        minDist = d;
                        next = j;
                    }
                }
            }
            order.Add(next);
            visited[next] = true;
            current = next;
        }
        return order.ToArray();
    }

    static double Distance(double[] a, double[] b)
    {
        double dx = a[0] - b[0];
        double dy = a[1] - b[1];
        double dz = a[2] - b[2];
        return Math.Sqrt(dx * dx + dy * dy + dz * dz);
    }
}
