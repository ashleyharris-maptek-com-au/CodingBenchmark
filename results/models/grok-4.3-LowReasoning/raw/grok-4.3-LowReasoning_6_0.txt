using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split();
        int N = int.Parse(firstLine[0]);
        double mu = double.Parse(firstLine[1]);

        string[] startLine = Console.ReadLine().Split();
        double[] startPos = { double.Parse(startLine[0]), double.Parse(startLine[1]), double.Parse(startLine[2]) };

        double[][] stations = new double[N][];
        for (int i = 0; i < N; i++)
        {
            string[] line = Console.ReadLine().Split();
            stations[i] = new double[] { double.Parse(line[0]), double.Parse(line[1]), double.Parse(line[2]) };
        }

        int[] order = SolveTSP(startPos, stations, N);
        Console.WriteLine(string.Join(" ", order));
    }

    static int[] SolveTSP(double[] start, double[][] stations, int N)
    {
        if (N <= 10)
        {
            return BruteForce(start, stations, N);
        }
        else
        {
            return SimulatedAnnealing(start, stations, N);
        }
    }

    static int[] BruteForce(double[] start, double[][] stations, int N)
    {
        int[] bestOrder = null;
        double bestCost = double.MaxValue;
        int[] perm = Enumerable.Range(0, N).ToArray();

        do
        {
            double cost = ComputeCost(start, stations, perm);
            if (cost < bestCost)
            {
                bestCost = cost;
                bestOrder = (int[])perm.Clone();
            }
        } while (NextPermutation(perm));

        return bestOrder;
    }

    static bool NextPermutation(int[] a)
    {
        int n = a.Length;
        int i = n - 2;
        while (i >= 0 && a[i] >= a[i + 1]) i--;
        if (i < 0) return false;
        int j = n - 1;
        while (a[j] <= a[i]) j--;
        (a[i], a[j]) = (a[j], a[i]);
        Array.Reverse(a, i + 1, n - i - 1);
        return true;
    }

    static int[] SimulatedAnnealing(double[] start, double[][] stations, int N)
    {
        Random rnd = new Random(42);
        int[] order = Enumerable.Range(0, N).ToArray();
        order = order.OrderBy(x => rnd.Next()).ToArray();

        double currentCost = ComputeCost(start, stations, order);
        double bestCost = currentCost;
        int[] bestOrder = (int[])order.Clone();

        double temp = 10000.0;
        double cooling = 0.995;

        for (int iter = 0; iter < 500000; iter++)
        {
            int i = rnd.Next(N);
            int j = rnd.Next(N);
            if (i == j) continue;

            (order[i], order[j]) = (order[j], order[i]);
            double newCost = ComputeCost(start, stations, order);

            if (newCost < currentCost || rnd.NextDouble() < Math.Exp((currentCost - newCost) / temp))
            {
                currentCost = newCost;
                if (currentCost < bestCost)
                {
                    bestCost = currentCost;
                    bestOrder = (int[])order.Clone();
                }
            }
            else
            {
                (order[i], order[j]) = (order[j], order[i]);
            }

            temp *= cooling;
        }

        return bestOrder;
    }

    static double ComputeCost(double[] start, double[][] stations, int[] order)
    {
        double cost = Distance(start, stations[order[0]]);
        for (int i = 0; i < order.Length - 1; i++)
        {
            cost += Distance(stations[order[i]], stations[order[i + 1]]);
        }
        return cost;
    }

    static double Distance(double[] a, double[] b)
    {
        double dx = a[0] - b[0];
        double dy = a[1] - b[1];
        double dz = a[2] - b[2];
        return Math.Sqrt(dx * dx + dy * dy + dz * dz);
    }
}
