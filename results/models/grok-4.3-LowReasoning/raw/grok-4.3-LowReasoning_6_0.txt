using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

class Program
{
    static void Main()
    {
        var culture = CultureInfo.InvariantCulture;
        var input = Console.In.ReadToEnd().Split(new[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        int N = int.Parse(input[idx++], culture);
        double mu = double.Parse(input[idx++], culture);

        double[] startPos = new double[3];
        for (int i = 0; i < 3; i++) startPos[i] = double.Parse(input[idx++], culture);
        for (int i = 0; i < 3; i++) idx++; // skip start vel

        double[][] positions = new double[N][];
        for (int i = 0; i < N; i++)
        {
            positions[i] = new double[3];
            for (int j = 0; j < 3; j++) positions[i][j] = double.Parse(input[idx++], culture);
            for (int j = 0; j < 3; j++) idx++; // skip vel
        }

        int[] order;
        if (N <= 9)
        {
            order = SolveBruteForce(positions, startPos);
        }
        else if (N <= 20)
        {
            order = SolveNearestNeighborWith2Opt(positions, startPos);
        }
        else
        {
            order = SolveNearestNeighbor(positions, startPos);
        }

        Console.WriteLine(string.Join(" ", order));
    }

    static double Dist(double[] a, double[] b)
    {
        double dx = a[0] - b[0];
        double dy = a[1] - b[1];
        double dz = a[2] - b[2];
        return Math.Sqrt(dx * dx + dy * dy + dz * dz);
    }

    static int[] SolveBruteForce(double[][] pos, double[] start)
    {
        int N = pos.Length;
        int[] bestOrder = null;
        double bestCost = double.MaxValue;
        var indices = Enumerable.Range(0, N).ToArray();
        do
        {
            double cost = Dist(start, pos[indices[0]]);
            for (int i = 0; i < N - 1; i++)
                cost += Dist(pos[indices[i]], pos[indices[i + 1]]);
            if (cost < bestCost)
            {
                bestCost = cost;
                bestOrder = (int[])indices.Clone();
            }
        } while (NextPermutation(indices));
        return bestOrder;
    }

    static bool NextPermutation(int[] a)
    {
        int n = a.Length;
        int i = n - 2;
        while (i >= 0 && a[i] >= a[i + 1]) i--;
        if (i < 0) return false;
        int j = n - 1;
        while (a[j] <= a[i]) j--;
        (a[i], a[j]) = (a[j], a[i]);
        Array.Reverse(a, i + 1, n - i - 1);
        return true;
    }

    static int[] SolveNearestNeighbor(double[][] pos, double[] start)
    {
        int N = pos.Length;
        var unvisited = new HashSet<int>(Enumerable.Range(0, N));
        var order = new List<int>(N);
        double[] current = start;
        while (unvisited.Count > 0)
        {
            int next = -1;
            double minD = double.MaxValue;
            foreach (int i in unvisited)
            {
                double d = Dist(current, pos[i]);
                if (d < minD)
                {
                    minD = d;
                    next = i;
                }
            }
            order.Add(next);
            unvisited.Remove(next);
            current = pos[next];
        }
        return order.ToArray();
    }

    static int[] SolveNearestNeighborWith2Opt(double[][] pos, double[] start)
    {
        int[] order = SolveNearestNeighbor(pos, start);
        // simple 2-opt improvement
        int N = order.Length;
        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 0; i < N - 1; i++)
            {
                for (int j = i + 2; j < N; j++)
                {
                    double currentCost = Dist(i == 0 ? start : pos[order[i - 1]], pos[order[i]]) +
                                         Dist(pos[order[j - 1]], pos[order[j]]);
                    double newCost = Dist(i == 0 ? start : pos[order[i - 1]], pos[order[j - 1]]) +
                                     Dist(pos[order[i]], pos[order[j]]);
                    if (newCost < currentCost - 1e-6)
                    {
                        Array.Reverse(order, i, j - i);
                        improved = true;
                    }
                }
            }
        }
        return order;
    }
}
