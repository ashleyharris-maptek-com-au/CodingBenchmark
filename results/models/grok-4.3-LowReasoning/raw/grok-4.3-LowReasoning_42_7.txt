; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_color %gl_out %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_gl_PerVertex_in ArrayStride 16
OpDecorate %gl_in DescriptorSet 0
OpDecorate %gl_in Binding 0
OpDecorate %in_color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0 = OpConstant %float 0
%float_3 = OpConstant %float 3
%float_0_08 = OpConstant %float 0.08
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr Input
%ptr_in_v4 = OpTypePointer Input %v4float
%arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%in_color = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
%main = OpFunction %void None %func
%entry = OpLabel
; Load 3 positions
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
; Extract xyz
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2
; e1 = p1 - p0, e2 = p2 - p0
%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz
; cross(e1, e2)
%e1x = OpCompositeExtract %float %e1 0
%e1y = OpCompositeExtract %float %e1 1
%e1z = OpCompositeExtract %float %e1 2
%e2x = OpCompositeExtract %float %e2 0
%e2y = OpCompositeExtract %float %e2 1
%e2z = OpCompositeExtract %float %e2 2
%cx = OpFMul %float %e1y %e2z
%cy = OpFMul %float %e1z %e2y
%cross_x = OpFSub %float %cx %cy
%cx2 = OpFMul %float %e1z %e2x
%cy2 = OpFMul %float %e1x %e2z
%cross_y = OpFSub %float %cx2 %cy2
%cx3 = OpFMul %float %e1x %e2y
%cy3 = OpFMul %float %e1y %e2x
%cross_z = OpFSub %float %cx3 %cy3
%N = OpCompositeConstruct %v3float %cross_x %cross_y %cross_z
; normalize N
%N_len = OpExtInst %float %1 Length %N
%N_norm = OpExtInst %v3float %1 Normalize %N
; offset = N * 0.08
%scale = OpFMul %float %float_0_08 %N_len
%offset = OpVectorTimesScalar %v3float %N_norm %float_0_08
%offset4 = OpCompositeConstruct %v4float %offset %float_0
; Emit vertex 0
%pos0 = OpFAdd %v4float %p0 %offset4
%out0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out0 %pos0
%c0_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%c0 = OpLoad %v3float %c0_ptr
OpStore %outColor %c0
OpEmitVertex
; Emit vertex 1
%pos1 = OpFAdd %v4float %p1 %offset4
%out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out1 %pos1
%c1_ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
%c1 = OpLoad %v3float %c1_ptr
OpStore %outColor %c1
OpEmitVertex
; Emit vertex 2
%pos2 = OpFAdd %v4float %p2 %offset4
%out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out2 %pos2
%c2_ptr = OpAccessChain %ptr_in_v3 %in_color %int_2
%c2 = OpLoad %v3float %c2_ptr
OpStore %outColor %c2
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
