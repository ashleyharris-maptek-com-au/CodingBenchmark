#include <iostream>
#include <vector>
#include <string>
#include <unordered_set>
#include <algorithm>
#include <cstdint>

using ull = unsigned long long;
using u128 = __uint128_t;

struct HashPair {
    ull h1, h2;
    bool operator==(const HashPair& other) const {
        return h1 == other.h1 && h2 == other.h2;
    }
};

struct HashPairHash {
    size_t operator()(const HashPair& p) const {
        return p.h1 ^ (p.h2 << 1);
    }
};

class RollingHash {
private:
    static constexpr ull B1 = 131;
    static constexpr ull B2 = 137;
    std::vector<ull> p1, p2, h1, h2;
    int n;
public:
    RollingHash(const std::string& s) {
        n = s.size();
        p1.resize(n+1, 1);
        p2.resize(n+1, 1);
        h1.resize(n+1, 0);
        h2.resize(n+1, 0);
        for (int i = 0; i < n; ++i) {
            p1[i+1] = p1[i] * B1;
            p2[i+1] = p2[i] * B2;
            h1[i+1] = h1[i] * B1 + (unsigned char)s[i];
            h2[i+1] = h2[i] * B2 + (unsigned char)s[i];
        }
    }
    HashPair get(int l, int r) const {  // [l, r)
        ull res1 = h1[r] - h1[l] * p1[r-l];
        ull res2 = h2[r] - h2[l] * p2[r-l];
        return {res1, res2};
    }
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    int N;
    std::cin >> N;
    std::string dummy;
    std::getline(std::cin, dummy);  // consume newline
    std::vector<std::string> strs(N);
    int min_len = INT_MAX;
    for (int i = 0; i < N; ++i) {
        std::getline(std::cin, strs[i]);
        if ((int)strs[i].size() < min_len) min_len = strs[i].size();
    }
    if (min_len == 0) {
        std::cout << "\n";
        return 0;
    }
    // Binary search on length
    int low = 0, high = min_len;
    std::string result = "";
    while (low <= high) {
        int mid = (low + high) / 2;
        if (mid == 0) {
            low = mid + 1;
            continue;
        }
        bool found = false;
        std::string candidate = "";
        // Use smallest string for candidates
        int min_idx = 0;
        for (int i = 1; i < N; ++i) {
            if (strs[i].size() < strs[min_idx].size()) min_idx = i;
        }
        RollingHash rh(strs[min_idx]);
        std::unordered_set<HashPair, HashPairHash> cand_hashes;
        for (int i = 0; i <= (int)strs[min_idx].size() - mid; ++i) {
            cand_hashes.insert(rh.get(i, i + mid));
        }
        for (int j = 0; j < N; ++j) {
            if (j == min_idx) continue;
            RollingHash rh2(strs[j]);
            bool has_common = false;
            for (int i = 0; i <= (int)strs[j].size() - mid; ++i) {
                HashPair hp = rh2.get(i, i + mid);
                if (cand_hashes.count(hp)) {
                    has_common = true;
                    if (candidate.empty()) {
                        candidate = strs[j].substr(i, mid);
                    }
                    break;
                }
            }
            if (!has_common) {
                found = false;
                break;
            }
            found = true;
        }
        if (found && N == 1) {
            candidate = strs[0].substr(0, mid);
            found = true;
        }
        if (found) {
            result = candidate;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    std::cout << result << "\n";
    return 0;
}
