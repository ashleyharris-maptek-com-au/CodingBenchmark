#include <bits/stdc++.h>
using namespace std;

using ll = long long;
using ull = unsigned long long;

const ull BASE1 = 131;
const ull BASE2 = 137;
const ull MOD1 = 1000000007ULL;
const ull MOD2 = 1000000009ULL;

struct HashPair {
    ull h1, h2;
    bool operator==(const HashPair& other) const {
        return h1 == other.h1 && h2 == other.h2;
    }
};

struct HashPairHash {
    size_t operator()(const HashPair& p) const {
        return p.h1 ^ (p.h2 << 1);
    }
};

vector<pair<ull, ull>> precompute_hashes(const string& s) {
    int n = s.size();
    vector<pair<ull, ull>> prefix(n + 1, {0, 0});
    vector<pair<ull, ull>> pow(n + 1, {1, 1});
    for (int i = 0; i < n; ++i) {
        prefix[i + 1].first = (prefix[i].first * BASE1 + (unsigned char)s[i]) % MOD1;
        prefix[i + 1].second = (prefix[i].second * BASE2 + (unsigned char)s[i]) % MOD2;
        pow[i + 1].first = pow[i].first * BASE1 % MOD1;
        pow[i + 1].second = pow[i].second * BASE2 % MOD2;
    }
    return prefix;
}

struct StringHashes {
    vector<ull> p1, p2, h1, h2;
    StringHashes(const string& s) {
        int n = s.size();
        p1.resize(n+1, 1); p2.resize(n+1, 1);
        h1.resize(n+1, 0); h2.resize(n+1, 0);
        for (int i = 0; i < n; ++i) {
            h1[i+1] = (h1[i] * BASE1 + (unsigned char)s[i]) % MOD1;
            h2[i+1] = (h2[i] * BASE2 + (unsigned char)s[i]) % MOD2;
            p1[i+1] = p1[i] * BASE1 % MOD1;
            p2[i+1] = p2[i] * BASE2 % MOD2;
        }
    }
    pair<ull, ull> get(int l, int r) const {
        ull res1 = (h1[r] + MOD1 - h1[l] * p1[r-l] % MOD1) % MOD1;
        ull res2 = (h2[r] + MOD2 - h2[l] * p2[r-l] % MOD2) % MOD2;
        return {res1, res2};
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N;
    cin >> N;
    vector<string> strs(N);
    int min_len = INT_MAX;
    for (int i = 0; i < N; ++i) {
        cin >> strs[i];
        min_len = min(min_len, (int)strs[i].size());
    }
    
    if (N == 0 || min_len == 0) {
        cout << "" << endl;
        return 0;
    }
    
    // Binary search on length
    int low = 0, high = min_len;
    int best_len = 0;
    pair<ull, ull> best_hash = {0, 0};
    string best_str = "";
    
    vector<vector<StringHashes>> hashes(N);
    for (int i = 0; i < N; ++i) {
        hashes[i].emplace_back(strs[i]); // one per string
    }
    
    while (low <= high) {
        int mid = (low + high) / 2;
        if (mid == 0) {
            low = mid + 1;
            continue;
        }
        
        unordered_set<HashPair, HashPairHash> common;
        auto& sh0 = hashes[0][0];
        for (int j = 0; j <= (int)strs[0].size() - mid; ++j) {
            auto hp = sh0.get(j, j + mid);
            common.insert({hp.first, hp.second});
        }
        
        bool possible = !common.empty();
        for (int i = 1; i < N && possible; ++i) {
            unordered_set<HashPair, HashPairHash> next_common;
            auto& shi = hashes[i][0];
            for (int j = 0; j <= (int)strs[i].size() - mid; ++j) {
                auto hp = shi.get(j, j + mid);
                if (common.count({hp.first, hp.second})) {
                    next_common.insert({hp.first, hp.second});
                }
            }
            common = std::move(next_common);
            possible = !common.empty();
        }
        
        if (possible && !common.empty()) {
            best_len = mid;
            best_hash = *common.begin();
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    
    if (best_len == 0) {
        cout << "" << endl;
        return 0;
    }
    
    // Recover the string from first occurrence in strs[0]
    auto& sh0 = hashes[0][0];
    for (int j = 0; j <= (int)strs[0].size() - best_len; ++j) {
        auto hp = sh0.get(j, j + best_len);
        if (hp.first == best_hash.first && hp.second == best_hash.second) {
            cout << strs[0].substr(j, best_len) << endl;
            return 0;
        }
    }
    
    cout << "" << endl;
    return 0;
}
