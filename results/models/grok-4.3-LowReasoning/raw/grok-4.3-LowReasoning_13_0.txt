#include <bits/stdc++.h>
using namespace std;

using ull = unsigned long long;
using pll = pair<ull, ull>;

struct Hash {
    vector<ull> h1, h2, p1, p2;
    ull b1 = 131, b2 = 137;
    Hash(const string& s) {
        int n = s.size();
        h1.resize(n+1); h2.resize(n+1);
        p1.resize(n+1, 1); p2.resize(n+1, 1);
        for (int i = 0; i < n; ++i) {
            h1[i+1] = h1[i] * b1 + (s[i] + 1);
            h2[i+1] = h2[i] * b2 + (s[i] + 1);
            p1[i+1] = p1[i] * b1;
            p2[i+1] = p2[i] * b2;
        }
    }
    pll get(int l, int r) { // [l, r)
        ull v1 = h1[r] - h1[l] * p1[r-l];
        ull v2 = h2[r] - h2[l] * p2[r-l];
        return {v1, v2};
    }
};

string longestCommonSubstring(const vector<string>& strs) {
    int N = strs.size();
    if (N == 0) return "";
    if (N == 1) return strs[0];
    size_t minL = strs[0].size();
    for (auto& s : strs) minL = min(minL, s.size());
    if (minL == 0) return "";
    
    // Binary search on length
    int low = 0, high = minL;
    string result = "";
    while (low <= high) {
        int mid = (low + high) / 2;
        if (mid == 0) break;
        // Collect candidates from first string
        Hash hs(strs[0]);
        unordered_set<ull> cands1;
        unordered_map<ull, string> hashToStr; // to recover
        for (size_t i = 0; i + mid <= strs[0].size(); ++i) {
            pll ph = hs.get(i, i+mid);
            ull key = ph.first ^ (ph.second << 1); // simple combine
            cands1.insert(key);
            if (hashToStr.find(key) == hashToStr.end())
                hashToStr[key] = strs[0].substr(i, mid);
        }
        bool found = true;
        for (int j = 1; j < N && found; ++j) {
            Hash hj(strs[j]);
            unordered_set<ull> nextc;
            for (size_t i = 0; i + mid <= strs[j].size(); ++i) {
                pll ph = hj.get(i, i+mid);
                ull key = ph.first ^ (ph.second << 1);
                if (cands1.count(key)) nextc.insert(key);
            }
            cands1 = move(nextc);
            if (cands1.empty()) found = false;
        }
        if (!cands1.empty()) {
            // recover one
            ull any = *cands1.begin();
            result = hashToStr[any];
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    return result;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int N;
    cin >> N;
    vector<string> strs(N);
    string line;
    getline(cin, line); // consume rest
    for (int i = 0; i < N; ++i) {
        getline(cin, strs[i]);
    }
    string ans = longestCommonSubstring(strs);
    cout << ans << endl;
    return 0;
}
