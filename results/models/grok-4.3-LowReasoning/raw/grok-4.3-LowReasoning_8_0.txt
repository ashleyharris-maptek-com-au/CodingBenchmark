```rust
use std::collections::VecDeque;
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut reader = io::BufReader::new(stdin.lock());
    let mut line = String::new();
    reader.read_line(&mut line).unwrap();
    let mut dims = line.split_whitespace();
    let h: usize = dims.next().unwrap().parse().unwrap();
    let w: usize = dims.next().unwrap().parse().unwrap();

    let mut maze: Vec<Vec<char>> = Vec::with_capacity(h);
    let mut start = (0usize, 0usize);
    let mut goal = (0usize, 0usize);

    for i in 0..h {
        line.clear();
        reader.read_line(&mut line).unwrap();
        let row: Vec<char> = line.trim_end_matches('\n').chars().collect();
        for j in 0..w {
            if row[j] == 'A' {
                start = (i, j);
            } else if row[j] == 'B' {
                goal = (i, j);
            }
        }
        maze.push(row);
    }

    if start == goal {
        println!("1");
        println!("{} {}", start.1, start.0);
        return;
    }

    let dirs: [(isize, isize); 4] = [(0, 1), (1, 0), (0, -1), (-1, 0)];
    let mut parent: Vec<Vec<Option<(usize, usize)>>> = vec![vec![None; w]; h];
    let mut queue: VecDeque<(usize, usize)> = VecDeque::new();

    queue.push_back(start);
    parent[start.0][start.1] = Some(start);

    let mut found = false;
    while let Some((x, y)) = queue.pop_front() {
        if (x, y) == goal {
            found = true;
            break;
        }
        for &(dx, dy) in &dirs {
            let nx = x as isize + dx;
            let ny = y as isize + dy;
            if nx >= 0 && nx < h as isize && ny >= 0 && ny < w as isize {
                let nx = nx as usize;
                let ny = ny as usize;
                if maze[nx][ny] != '#' && parent[nx][ny].is_none() {
                    parent[nx][ny] = Some((x, y));
                    queue.push_back((nx, ny));
                }
            }
        }
    }

    if !found {
        println!("0");
        return;
    }

    let mut path: Vec<(usize, usize)> = Vec::new();
    let mut current = goal;
    while current != start {
        path.push(current);
        current = parent[current.0][current.1].unwrap();
    }
    path.push(start);
    path.reverse();

    println!("{}", path.len());
    for &(r, c) in &path {
        println!("{} {}", c, r);
    }
}
```
