```rust
use std::io::{self, BufRead, BufWriter, Write};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    
    let first_line = lines.next().unwrap().unwrap();
    let mut hw = first_line.split_whitespace();
    let h: usize = hw.next().unwrap().parse().unwrap();
    let w: usize = hw.next().unwrap().parse().unwrap();
    
    let mut maze: Vec<Vec<char>> = Vec::with_capacity(h);
    let mut start: (usize, usize) = (0, 0);
    let mut goal: (usize, usize) = (0, 0);
    
    for y in 0..h {
        let line = lines.next().unwrap().unwrap();
        let row: Vec<char> = line.chars().collect();
        for (x, &c) in row.iter().enumerate() {
            if c == 'A' {
                start = (x, y);
            } else if c == 'B' {
                goal = (x, y);
            }
        }
        maze.push(row);
    }
    
    let total = h * w;
    let mut visited = vec![false; total];
    let mut parent = vec![usize::MAX; total];
    let mut queue: VecDeque<usize> = VecDeque::new();
    
    let dirs: [(isize, isize); 4] = [(0, 1), (1, 0), (0, -1), (-1, 0)];
    
    let start_idx = start.1 * w + start.0;
    queue.push_back(start_idx);
    visited[start_idx] = true;
    
    let mut found = false;
    let mut goal_idx = 0;
    
    while let Some(idx) = queue.pop_front() {
        if idx == goal.1 * w + goal.0 {
            found = true;
            goal_idx = idx;
            break;
        }
        
        let x = idx % w;
        let y = idx / w;
        
        for (dx, dy) in dirs.iter() {
            let nx = x as isize + dx;
            let ny = y as isize + dy;
            if nx >= 0 && nx < w as isize && ny >= 0 && ny < h as isize {
                let nidx = ny as usize * w + nx as usize;
                if !visited[nidx] {
                    let c = maze[ny as usize][nx as usize];
                    if c != '#' {
                        visited[nidx] = true;
                        parent[nidx] = idx;
                        queue.push_back(nidx);
                    }
                }
            }
        }
    }
    
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    
    if !found {
        writeln!(out, "0").unwrap();
        return;
    }
    
    let mut path: Vec<(usize, usize)> = Vec::new();
    let mut current = goal_idx;
    while current != usize::MAX {
        let x = current % w;
        let y = current / w;
        path.push((x, y));
        if current == start_idx {
            break;
        }
        current = parent[current];
    }
    
    if path.last() != Some(&start) {
        writeln!(out, "0").unwrap();
        return;
    }
    
    path.reverse();
    
    writeln!(out, "{}", path.len()).unwrap();
    for (x, y) in path {
        writeln!(out, "{} {}", x, y).unwrap();
    }
}
```
