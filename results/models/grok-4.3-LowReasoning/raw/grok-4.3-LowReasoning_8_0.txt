```rust
use std::io::{self, BufRead, Write};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut dims = first_line.split_whitespace();
    let h: usize = dims.next().unwrap().parse().unwrap();
    let w: usize = dims.next().unwrap().parse().unwrap();

    let mut maze: Vec<Vec<char>> = Vec::with_capacity(h);
    let mut start = (0usize, 0usize);
    let mut goal = (0usize, 0usize);

    for y in 0..h {
        let line = lines.next().unwrap().unwrap();
        let row: Vec<char> = line.chars().collect();
        for x in 0..w {
            if row[x] == 'A' {
                start = (y, x);
            } else if row[x] == 'B' {
                goal = (y, x);
            }
        }
        maze.push(row);
    }

    let total = h * w;
    let mut parent: Vec<usize> = vec![usize::MAX; total];
    let mut queue: VecDeque<usize> = VecDeque::new();
    let start_idx = start.0 * w + start.1;
    queue.push_back(start_idx);
    parent[start_idx] = start_idx;

    let dirs: [(isize, isize); 4] = [(-1, 0), (1, 0), (0, -1), (0, 1)];
    let mut found = false;
    let mut goal_idx = 0usize;

    while let Some(idx) = queue.pop_front() {
        let y = idx / w;
        let x = idx % w;
        if (y, x) == goal {
            found = true;
            goal_idx = idx;
            break;
        }
        for &(dy, dx) in &dirs {
            let ny = y as isize + dy;
            let nx = x as isize + dx;
            if ny >= 0 && ny < h as isize && nx >= 0 && nx < w as isize {
                let nyu = ny as usize;
                let nxu = nx as usize;
                if maze[nyu][nxu] != '#' {
                    let nidx = nyu * w + nxu;
                    if parent[nidx] == usize::MAX {
                        parent[nidx] = idx;
                        queue.push_back(nidx);
                    }
                }
            }
        }
    }

    if !found {
        println!("0");
        return;
    }

    let mut path: Vec<(usize, usize)> = Vec::new();
    let mut current = goal_idx;
    while current != start_idx {
        let y = current / w;
        let x = current % w;
        path.push((x, y));
        current = parent[current];
    }
    path.push((start.1, start.0));
    path.reverse();

    let stdout = io::stdout();
    let mut handle = stdout.lock();
    writeln!(handle, "{}", path.len()).unwrap();
    for &(x, y) in &path {
        writeln!(handle, "{} {}", x, y).unwrap();
    }
}
```
