const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.grid;
const robots = data.robots;
const jobs = data.jobs;

const rows = grid.length;
const cols = grid[0].length;

const dirs = [[-1, 0], [1, 0], [0, -1], [0, 1]];

function isValid(x, y) {
  return x >= 0 && x < cols && y >= 0 && y < rows && grid[y][x] !== '#';
}

function bfsDist(sx, sy, tx, ty) {
  if (sx === tx && sy === ty) return 0;
  const queue = [[sx, sy, 0]];
  const visited = Array.from({ length: rows }, () => Array(cols).fill(false));
  visited[sy][sx] = true;
  while (queue.length > 0) {
    const [x, y, d] = queue.shift();
    for (const [dx, dy] of dirs) {
      const nx = x + dx;
      const ny = y + dy;
      if (nx === tx && ny === ty) return d + 1;
      if (isValid(nx, ny) && !visited[ny][nx]) {
        visited[ny][nx] = true;
        queue.push([nx, ny, d + 1]);
      }
    }
  }
  return Infinity;
}

const R = robots.length;
const J = jobs.length;

const jobList = [...jobs].sort((a, b) => {
  if (b.priority !== a.priority) return b.priority - a.priority;
  return a.id - b.id;
});

const currentPos = robots.map(r => ({ x: r.x, y: r.y }));
const assignments = Array.from({ length: R }, () => []);

const remaining = new Set(jobList.map(j => j.id));
const jobMap = new Map(jobs.map(j => [j.id, j]));

for (const job of jobList) {
  let bestRobot = -1;
  let bestDist = Infinity;
  for (let r = 0; r < R; r++) {
    const cp = currentPos[r];
    const dist = bfsDist(cp.x, cp.y, job.x, job.y);
    if (dist < bestDist) {
      bestDist = dist;
      bestRobot = r;
    }
  }
  if (bestRobot !== -1) {
    assignments[bestRobot].push(job.id);
    currentPos[bestRobot] = { x: job.x, y: job.y };
  }
}

const output = { assignments };
console.log(JSON.stringify(output));
