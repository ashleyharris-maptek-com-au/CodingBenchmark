```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<long long> c(n);
    for (auto &x : c) cin >> x;
    vector<vector<int>> adj(n);
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    long long best = LLONG_MIN;
    int best_mask = 0;
    int max_mask = (n <= 20 ? (1 << n) : 0);
    if (n > 20) {
        // Fallback: trivial solution (all zero) if n too large
        best = 0;
        best_mask = 0;
    } else {
        for (int mask = 0; mask < max_mask; mask++) {
            long long val = 0;
            for (int i = 0; i < n; i++) if (mask & (1 << i)) val += c[i];
            if (val <= best) continue;
            bool valid = true;
            for (int u = 0; u < n && valid; u++) if (mask & (1 << u)) {
                for (int v : adj[u]) if (v > u && (mask & (1 << v))) {
                    valid = false;
                    break;
                }
            }
            if (valid) {
                best = val;
                best_mask = mask;
            }
        }
        if (best == LLONG_MIN) {
            best = 0;
            best_mask = 0;
        }
    }
    cout << best << endl;
    for (int i = 0; i < n; i++) {
        cout << ((best_mask >> i) & 1);
        if (i + 1 < n) cout << " ";
    }
    cout << endl;
    return 0;
}
```
