```cpp
#include <bits/stdc++.h>
using namespace std;

using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, m;
    cin >> n >> m;
    vector<ll> c(n);
    for (auto &x : c) cin >> x;
    vector<vector<int>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u > v) swap(u, v);
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    for (int i = 0; i < n; ++i) {
        sort(adj[i].begin(), adj[i].end());
        auto it = unique(adj[i].begin(), adj[i].end());
        adj[i].resize(it - adj[i].begin());
    }
    ll best_val = LLONG_MIN / 2;
    vector<int> best_assign(n, 0);
    int max_mask = 1 << n;
    for (int mask = 0; mask < max_mask; ++mask) {
        bool valid = true;
        ll val = 0;
        for (int i = 0; i < n && valid; ++i) {
            if (mask & (1 << i)) {
                val += c[i];
                for (int j : adj[i]) {
                    if (j > i && (mask & (1 << j))) {
                        valid = false;
                        break;
                    }
                }
            }
        }
        if (valid && val > best_val) {
            best_val = val;
            for (int i = 0; i < n; ++i) {
                best_assign[i] = (mask & (1 << i)) ? 1 : 0;
            }
        }
    }
    cout << best_val << '\n';
    for (int i = 0; i < n; ++i) {
        cout << best_assign[i];
        if (i + 1 < n) cout << ' ';
    }
    cout << '\n';
    return 0;
}
```
