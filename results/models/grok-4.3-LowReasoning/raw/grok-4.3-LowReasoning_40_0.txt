#include <bits/stdc++.h>
using namespace std;

int main() {
  int n, m;
  cin >> n >> m;
  vector<long long> c(n);
  for (auto& x : c) cin >> x;
  vector<vector<int>> adj(n);
  for (int i = 0; i < m; i++) {
    int u, v;
    cin >> u >> v;
    adj[u].push_back(v);
    adj[v].push_back(u);
  }
  long long best = LLONG_MIN / 2;
  vector<int> bestx(n, 0);
  int N = 1 << n;
  for (int mask = 0; mask < N; mask++) {
    vector<int> x(n, 0);
    long long val = 0;
    for (int i = 0; i < n; i++) {
      if (mask & (1 << i)) {
        x[i] = 1;
        val += c[i];
      }
    }
    bool valid = true;
    for (int i = 0; i < n && valid; i++) {
      if (x[i]) {
        for (int j : adj[i]) {
          if (j > i && x[j]) {
            valid = false;
            break;
          }
        }
      }
    }
    if (valid && val > best) {
      best = val;
      bestx = x;
    }
  }
  cout << best << endl;
  for (int i = 0; i < n; i++) {
    cout << bestx[i];
    if (i + 1 < n) cout << " ";
  }
  cout << endl;
  return 0;
}
