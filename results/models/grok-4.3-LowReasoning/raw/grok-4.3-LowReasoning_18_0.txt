#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdint>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    uint64_t N, M;
    std::cin >> N >> M;
    std::vector<std::pair<uint64_t, uint64_t>> edges;
    edges.reserve(M);
    std::vector<uint64_t> degree(N, 0);
    for (uint64_t i = 0; i < M; ++i) {
        uint64_t u, v;
        std::cin >> u >> v;
        if (u > v) std::swap(u, v);
        edges.emplace_back(u, v);
        if (u < N) ++degree[u];
        if (v < N) ++degree[v];
    }
    if (N == 0) {
        std::cout << 0 << '\n';
        return 0;
    }
    uint64_t min_deg = UINT64_MAX;
    uint64_t best_v = 0;
    for (uint64_t v = 0; v < N; ++v) {
        if (degree[v] < min_deg && degree[v] > 0) {
            min_deg = degree[v];
            best_v = v;
        }
    }
    if (min_deg == UINT64_MAX) min_deg = 0;
    std::vector<std::pair<uint64_t, uint64_t>> cut;
    for (const auto& e : edges) {
        if (e.first == best_v || e.second == best_v) {
            cut.push_back(e);
        }
    }
    std::cout << cut.size() << '\n';
    for (const auto& e : cut) {
        std::cout << e.first << ' ' << e.second << '\n';
    }
    return 0;
}
