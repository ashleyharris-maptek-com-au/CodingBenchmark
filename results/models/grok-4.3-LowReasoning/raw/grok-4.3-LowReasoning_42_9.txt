               OpCapability Geometry

    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4

    %uint_4 = OpConstant %uint 4
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4

     %gl_in = OpVariable %ptr_in_arr_pv Input

    ; gl_PerVertex output struct
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
    %gl_out = OpVariable %ptr_out_pv Output

    ; User input arrays
%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_v3arr = OpTypePointer Input %arr_v3_in
%in_worldPos = OpVariable %ptr_in_v3arr Input
 %in_normal = OpVariable %ptr_in_v3arr Input
   %in_color = OpVariable %ptr_in_v3arr Input

    ; User output
%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

    ; UBO type
%mat4v4float = OpTypeMatrix %v4float 4
        %UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v3float %v3float %float
%ptr_ubo = OpTypePointer Uniform %UBO

      %int_1 = OpConstant %uint 1
      %int_2 = OpConstant %uint 2
      %int_3 = OpConstant %uint 3
  %thickness = OpConstant %float 0.025
        %one = OpConstant %float 1.0
      %zero = OpConstant %float 0.0

       %main = OpFunction %void None %fn_void
      %entry = OpLabel

      ; Load positions from gl_in
      %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
        %p0 = OpLoad %v4float %p0_ptr
      %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
        %p1 = OpLoad %v4float %p1_ptr
      %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
        %p2 = OpLoad %v4float %p2_ptr
      %p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
        %p3 = OpLoad %v4float %p3_ptr

      ; Load colors
      %c1_ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
        %c1 = OpLoad %v3float %c1_ptr
      %c2_ptr = OpAccessChain %ptr_in_v3 %in_color %int_2
        %c2 = OpLoad %v3float %c2_ptr

      ; Compute tangents (using xy components)
      %d0 = OpVectorShuffle %v2float %p2 %p0 0 1 4 5
      %t0_ = OpFSub %v2float %d0 %zero ; placeholder - compute p2.xy - p0.xy
      ; Actually compute differences
      %p2xy = OpVectorShuffle %v2float %p2 %p2 0 1
      %p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
      %p1xy = OpVectorShuffle %v2float %p1 %p1 0 1
      %p3xy = OpVectorShuffle %v2float %p3 %p3 0 1

      %diff_t0 = OpFSub %v2float %p2xy %p0xy
        %t0 = OpExtInst %v2float %glsl_std_450 Normalize %diff_t0

      %diff_t1 = OpFSub %v2float %p3xy %p1xy
        %t1 = OpExtInst %v2float %glsl_std_450 Normalize %diff_t1

      ; Perpendiculars
      %t0y = OpCompositeExtract %float %t0 1
      %t0x = OpCompositeExtract %float %t0 0
     %n0x = OpFNegate %float %t0y
      %perp0 = OpCompositeConstruct %v2float %n0x %t0x

      %t1y = OpCompositeExtract %float %t1 1
      %t1x = OpCompositeExtract %float %t1 0
     %n1x = OpFNegate %float %t1y
      %perp1 = OpCompositeConstruct %v2float %n1x %t1x

      ; Thickness scaling
      %thk_vec = OpCompositeConstruct %v2float %thickness %thickness
     %off0 = OpFMul %v2float %perp0 %thk_vec
     %off1 = OpFMul %v2float %perp1 %thk_vec

      ; Emit vertex 0: start + perp0*thickness
      %v0xy = OpFAdd %v2float %p1xy %off0
       %v0 = OpCompositeConstruct %v4float %v0xy_x %v0xy_y %zero %one  ; placeholder swizzle
      ; Proper construction
      %v0x = OpCompositeExtract %float %v0xy 0
      %v0y = OpCompositeExtract %float %v0xy 1
       %v0 = OpCompositeConstruct %v4float %v0x %v0y %zero %one

      ; Store gl_Position
      %out0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
                OpStore %out0_ptr %v0
                OpStore %outColor %c1
                OpEmitVertex

      ; Emit vertex 1: start - perp0*thickness
     %off0n = OpFNegate %v2float %off0
      %v1xy = OpFAdd %v2float %p1xy %off0n
      %v1x = OpCompositeExtract %float %v1xy 0
      %v1y = OpCompositeExtract %float %v1xy 1
       %v1 = OpCompositeConstruct %v4float %v1x %v1y %zero %one

      %out1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
                OpStore %out1_ptr %v1
                OpStore %outColor %c1
                OpEmitVertex

      ; Emit vertex 2: end + perp1*thickness
      %v2xy = OpFAdd %v2float %p2xy %off1
      %v2x = OpCompositeExtract %float %v2xy 0
      %v2y = OpCompositeExtract %float %v2xy 1
       %v2 = OpCompositeConstruct %v4float %v2x %v2y %zero %one

      %out2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
                OpStore %out2_ptr %v2
                OpStore %outColor %c2
                OpEmitVertex

      ; Emit vertex 3: end - perp1*thickness
     %off1n = OpFNegate %v2float %off1
      %v3xy = OpFAdd %v2float %p2xy %off1n
      %v3x = OpCompositeExtract %float %v3xy 0
      %v3y = OpCompositeExtract %float %v3xy 1
       %v3 = OpCompositeConstruct %v4float %v3x %v3y %zero %one

      %out3_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
                OpStore %out3_ptr %v3
                OpStore %outColor %c2
                OpEmitVertex

                OpEndPrimitive
                OpReturn
                OpFunctionEnd
