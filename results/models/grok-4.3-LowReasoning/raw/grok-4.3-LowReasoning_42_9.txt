OpCapability Geometry

OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4

OpDecorate %outColor Location 0

%ptr_out_v3 = OpTypePointer Output %v3float

%main = OpFunction %void None %func
%entry = OpLabel

; Load positions (clip space from gl_Position)
%p0_chain = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %p0_chain

%p1_chain = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %p1_chain
%xy1 = OpVectorShuffle %v2float %pos1 %pos1 0 1

%p2_chain = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0

; tangents
%diff0 = OpFSub %v2float %xy2 %xy0
%t0 = OpNormalize %v2float %diff0
%diff1 = OpFSub %v2float %xy3 %xy1
%t1 = OpNormalize %v2float %diff1

; perpendiculars
%t0y = OpCompositeExtract %float %t0 1
%neg_t0y = OpFNegate %float %t0y
%t0x = OpCompositeExtract %float %t0 0

%t1y = OpCompositeExtract %float %t1 1
%neg_t1y = OpFNegate %float %t1y
%t1x = OpCompositeExtract %float %t1 0
%perp1 = OpCompositeConstruct %v2float %neg_t1y %t1x

; thickness
%thick = OpConstant %float 0.025

; offsets
%off0p = OpVectorTimesScalar %v2float %perp0 %thick
%off0m = OpVectorTimesScalar %v2float %perp0 %thick
%off0m = OpFNegate %v2float %off0m

%off1m = OpVectorTimesScalar %v2float %perp1 %thick
%off1m = OpFNegate %v2float %off1m

; vertices
%v0 = OpFAdd %v2float %xy1 %off0p
%v1 = OpFAdd %v2float %xy1 %off0m
%v2 = OpFAdd %v2float %xy2 %off1p
%v3 = OpFAdd %v2float %xy2 %off1m

; Load colors
%c1_chain = OpAccessChain %ptr_in_v3 %color_in %int_1

%c2_chain = OpAccessChain %ptr_in_v3 %color_in %int_2

%out0_chain = OpAccessChain %ptr_out_v4 %gl_out %int_0
