```typescript
declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const width: number = data.width;
const height: number = data.height;
const grid: string[] = data.grid;
const ignitions: number[][] = data.ignitions;
const towns: any[] = data.towns;
const cutBudget: number = data.cutBudget;

const dirs: number[][] = [[0,1],[1,0],[0,-1],[-1,0]];

function isValid(x: number, y: number): boolean {
  return x >= 0 && x < width && y >= 0 && y < height;
}

function getCell(x: number, y: number): string {
  return grid[y][x];
}

const firebreaks: number[][] = [];
const cutSet: Set<string> = new Set();

for (const [ix, iy] of ignitions) {
  for (const [dx, dy] of dirs) {
    const nx = ix + dx;
    const ny = iy + dy;
    if (isValid(nx, ny) && getCell(nx, ny) === '.' && firebreaks.length < cutBudget) {
      const key = `${nx},${ny}`;
      if (!cutSet.has(key)) {
        cutSet.add(key);
        firebreaks.push([nx, ny]);
      }
    }
  }
}

while (firebreaks.length < cutBudget) {
  let added = false;
  for (let y = 0; y < height && firebreaks.length < cutBudget; y++) {
    for (let x = 0; x < width && firebreaks.length < cutBudget; x++) {
      if (getCell(x, y) === '.') {
        const key = `${x},${y}`;
        if (!cutSet.has(key)) {
          cutSet.add(key);
          firebreaks.push([x, y]);
          added = true;
        }
      }
    }
  }
  if (!added) break;
}

console.log(JSON.stringify({ firebreaks }));
```
