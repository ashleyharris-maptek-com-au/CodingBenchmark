#include <iostream>
#include <vector>
#include <queue>
#include <unordered_set>
#include <unordered_map>
#include <string>
#include <algorithm>
#include <sstream>

using Position = std::vector<int>;

std::string pos_to_key(const Position& p) {
    std::string key;
    for (size_t i = 0; i < p.size(); ++i) {
        key += std::to_string(p[i]);
        if (i + 1 < p.size()) key += ",";
    }
    return key;
}

Position add_delta(const Position& p, int axis, int dir, const std::vector<int>& bounds) {
    Position np = p;
    np[axis] += dir;
    if (np[axis] < 0 || np[axis] >= bounds[axis]) return {};
    return np;
}

int main() {
    int D, max_turns;
    std::cin >> D >> max_turns;

    std::vector<int> bounds(D);
    for (int i = 0; i < D; ++i) {
        std::cin >> bounds[i];
    }

    int snake_len;
    std::cin >> snake_len;
    Position head(D);
    for (int i = 0; i < D; ++i) {
        std::cin >> head[i];
    }

    std::deque<Position> snake;
    snake.push_back(head);
    std::unordered_set<std::string> occupied;
    occupied.insert(pos_to_key(head));

    // Assume initial snake of length 1 at head for generality
    // (full body not provided in input, so start minimal)

    int F;
    std::cin >> F;
    std::unordered_set<std::string> foods;
    for (int i = 0; i < F; ++i) {
        Position food(D);
        for (int j = 0; j < D; ++j) std::cin >> food[j];
        foods.insert(pos_to_key(food));
    }

    int O;
    std::cin >> O;
    std::unordered_set<std::string> obstacles;
    for (int i = 0; i < O; ++i) {
        Position obs(D);
        for (int j = 0; j < D; ++j) std::cin >> obs[j];
        obstacles.insert(pos_to_key(obs));
    }

    // Simple solver: BFS to nearest food each time
    auto find_path = [&](const Position& start) -> std::vector<std::pair<int, int>> {
        std::queue<Position> q;
        std::unordered_map<std::string, std::pair<Position, std::pair<int, int>>> parent;
        std::unordered_set<std::string> vis;

        q.push(start);
        vis.insert(pos_to_key(start));
        parent[pos_to_key(start)] = {{}, {-1, 0}};

        while (!q.empty()) {
            Position cur = q.front(); q.pop();
            std::string ck = pos_to_key(cur);

            if (foods.count(ck)) {
                // reconstruct path
                std::vector<std::pair<int, int>> path;
                Position at = cur;
                while (pos_to_key(at) != pos_to_key(start)) {
                    auto [prev, move] = parent[pos_to_key(at)];
                    path.push_back(move);
                    at = prev;
                }
                std::reverse(path.begin(), path.end());
                return path;
            }

            for (int axis = 0; axis < D; ++axis) {
                for (int dir : {-1, 1}) {
                    Position np = add_delta(cur, axis, dir, bounds);
                    if (np.empty()) continue;
                    std::string nk = pos_to_key(np);
                    if (vis.count(nk) || obstacles.count(nk) || occupied.count(nk)) continue;
                    vis.insert(nk);
                    q.push(np);
                    parent[nk] = {cur, {axis, dir}};
                }
            }
        }
        return {};
    };

    int turns = 0;
    while (turns < max_turns && !foods.empty()) {
        auto path = find_path(snake.back());
        if (path.empty()) {
            // fallback random valid move
            bool moved = false;
            for (int axis = 0; axis < D && !moved; ++axis) {
                for (int dir : {-1, 1}) {
                    Position np = add_delta(snake.back(), axis, dir, bounds);
                    if (!np.empty() && !obstacles.count(pos_to_key(np)) && !occupied.count(pos_to_key(np))) {
                        std::cout << axis << " " << dir << std::endl;
                        // simulate
                        snake.push_back(np);
                        occupied.insert(pos_to_key(np));
                        std::string k = pos_to_key(np);
                        if (foods.count(k)) foods.erase(k);
                        else {
                            occupied.erase(pos_to_key(snake.front()));
                            snake.pop_front();
                        }
                        moved = true;
                        break;
                    }
                }
            }
            if (!moved) break;
        } else {
            auto [axis, dir] = path[0];
            std::cout << axis << " " << dir << std::endl;
            Position np = add_delta(snake.back(), axis, dir, bounds);
            snake.push_back(np);
            occupied.insert(pos_to_key(np));
            std::string k = pos_to_key(np);
            if (foods.count(k)) foods.erase(k);
            else {
                occupied.erase(pos_to_key(snake.front()));
                snake.pop_front();
            }
        }
        ++turns;
    }

    return 0;
}
