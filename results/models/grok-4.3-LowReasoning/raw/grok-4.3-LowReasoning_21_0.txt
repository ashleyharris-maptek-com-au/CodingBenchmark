#include <iostream>
#include <vector>
#include <deque>
#include <set>
#include <map>
#include <algorithm>
#include <string>
#include <sstream>

using namespace std;

using Pos = vector<int>;

bool operator<(const Pos& a, const Pos& b) {
    return a < b;
}

int D;
vector<int> bounds;
deque<Pos> snake;
set<Pos> foods;
set<Pos> obstacles;
long long max_turns;
int turn = 0;

bool in_bounds(const Pos& p) {
    for (int i = 0; i < D; ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    return true;
}

bool is_occupied(const Pos& p, bool ignore_tail = false) {
    if (obstacles.count(p)) return true;
    auto it = snake.begin();
    auto end = snake.end();
    if (ignore_tail && !snake.empty()) --end;
    for (; it != end; ++it) {
        if (*it == p) return true;
    }
    return false;
}

long long manhattan(const Pos& a, const Pos& b) {
    long long dist = 0;
    for (int i = 0; i < D; ++i) {
        dist += abs(a[i] - b[i]);
    }
    return dist;
}

Pos find_closest_food(const Pos& head) {
    if (foods.empty()) return {};
    Pos closest = *foods.begin();
    long long min_dist = manhattan(head, closest);
    for (const auto& f : foods) {
        long long d = manhattan(head, f);
        if (d < min_dist) {
            min_dist = d;
            closest = f;
        }
    }
    return closest;
}

pair<int, int> choose_move(const Pos& head) {
    if (foods.empty()) return {-1, 0};
    Pos target = find_closest_food(head);
    vector<pair<long long, pair<int, int>>> candidates;
    for (int axis = 0; axis < D; ++axis) {
        for (int dir : {-1, 1}) {
            Pos next = head;
            next[axis] += dir;
            if (!in_bounds(next)) continue;
            if (is_occupied(next, true)) continue;
            long long dist = manhattan(next, target);
            candidates.emplace_back(dist, make_pair(axis, dir));
        }
    }
    if (candidates.empty()) {
        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                Pos next = head;
                next[axis] += dir;
                if (!in_bounds(next)) continue;
                if (is_occupied(next, true)) continue;
                return {axis, dir};
            }
        }
        return {-1, 0};
    }
    sort(candidates.begin(), candidates.end());
    return candidates[0].second;
}

void apply_move(int axis, int dir) {
    Pos head = snake.front();
    Pos next = head;
    next[axis] += dir;
    snake.push_front(next);
    bool ate = foods.erase(next) > 0;
    if (!ate) {
        snake.pop_back();
    }
}

int main() {
    cin >> D >> max_turns;
    bounds.resize(D);
    for (int i = 0; i < D; ++i) cin >> bounds[i];
    int snake_len;
    cin >> snake_len;
    Pos start(D);
    for (int i = 0; i < D; ++i) cin >> start[i];
    snake.push_back(start);
    // Read remaining snake body? Assume head only given, body follows? But input says "start_pos" for head
    // For simplicity assume initial snake is just head, length=1 or body implied straight? But to handle general:
    // Actually input: snake_length start_pos (head)
    // Assume initial body not given, so for generality, start with length 1, but to fix:
    for (int i = 1; i < snake_len; ++i) {
        Pos p = start; // placeholder, assume no initial body or degenerate
        snake.push_back(p);
    }
    int F;
    cin >> F;
    for (int i = 0; i < F; ++i) {
        Pos f(D);
        for (int j = 0; j < D; ++j) cin >> f[j];
        foods.insert(f);
    }
    int O;
    cin >> O;
    for (int i = 0; i < O; ++i) {
        Pos o(D);
        for (int j = 0; j < D; ++j) cin >> o[j];
        obstacles.insert(o);
    }
    while (!foods.empty() && turn < max_turns) {
        Pos head = snake.front();
        auto [axis, dir] = choose_move(head);
        if (axis == -1) break;
        cout << axis << " " << dir << endl;
        cout.flush();
        apply_move(axis, dir);
        ++turn;
    }
    return 0;
}
