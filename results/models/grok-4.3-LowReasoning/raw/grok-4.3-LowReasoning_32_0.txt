```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <tuple>

using namespace std;

using ll = long long;
const ll INF = numeric_limits<ll>::max() / 2;

struct Edge {
    int u, v;
    ll w;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V, E, K;
    cin >> V >> E >> K;

    vector<vector<pair<int, ll>>> g(V);
    vector<Edge> all_edges;
    for (int i = 0; i < E; ++i) {
        int u, v;
        ll w;
        cin >> u >> v >> w;
        g[u].emplace_back(v, w);
        g[v].emplace_back(u, w);
        all_edges.push_back({u, v, w});
    }

    vector<int> terminals(K);
    for (int i = 0; i < K; ++i) {
        cin >> terminals[i];
    }

    if (K == 0) {
        cout << "0 0\n";
        return 0;
    }
    if (K == 1) {
        cout << "0 0\n";
        return 0;
    }

    // All-pairs shortest paths among terminals using Dijkstra
    vector<vector<ll>> dist(K, vector<ll>(K, INF));
    vector<vector<int>> prev(K, vector<int>(V, -1));
    vector<vector<int>> prev_edge_from(K, vector<int>(V, -1));

    auto dijkstra = [&](int src_idx) {
        int src = terminals[src_idx];
        vector<ll> d(V, INF);
        vector<int> p(V, -1);
        vector<int> pe(V, -1);
        d[src] = 0;
        priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
        pq.emplace(0, src);
        while (!pq.empty()) {
            auto [cost, u] = pq.top(); pq.pop();
            if (cost > d[u]) continue;
            for (auto [v, w] : g[u]) {
                if (d[v] > d[u] + w) {
                    d[v] = d[u] + w;
                    p[v] = u;
                    pe[v] = 1; // marker
                    pq.emplace(d[v], v);
                }
            }
        }
        for (int i = 0; i < K; ++i) {
            dist[src_idx][i] = d[terminals[i]];
        }
        prev[src_idx] = p;
    };

    for (int i = 0; i < K; ++i) {
        dijkstra(i);
    }

    // Build MST on terminals (Prim)
    vector<vector<int>> mst(K);
    vector<bool> used(K, false);
    vector<ll> min_e(K, INF);
    vector<int> sel_e(K, -1);
    min_e[0] = 0;
    for (int i = 0; i < K; ++i) {
        int v = -1;
        for (int j = 0; j < K; ++j) {
            if (!used[j] && (v == -1 || min_e[j] < min_e[v])) v = j;
        }
        if (min_e[v] == INF) {
            // disconnected
            cout << "0 0\n";
            return 0;
        }
        used[v] = true;
        if (sel_e[v] != -1) {
            mst[sel_e[v]].push_back(v);
            mst[v].push_back(sel_e[v]);
        }
        for (int j = 0; j < K; ++j) {
            if (dist[v][j] < min_e[j]) {
                min_e[j] = dist[v][j];
                sel_e[j] = v;
            }
        }
    }

    // Recover edges from shortest paths in MST
    vector<pair<int, int>> tree_edges;
    vector<bool> vis(V, false);

    function<void(int, int)> recover = [&](int a, int b) {
        int src_idx = a;
        int tgt = terminals[b];
        int cur = tgt;
        vector<pair<int, int>> path;
        auto& p = prev[src_idx];
        while (cur != terminals[src_idx]) {
            int pr = p[cur];
            if (pr == -1) break;
            path.emplace_back(min(pr, cur), max(pr, cur));
            cur = pr;
        }
        for (auto& e : path) {
            tree_edges.push_back(e);
        }
    };

    function<void(int, int)> dfs_mst = [&](int u, int par) {
        for (int v : mst[u]) {
            if (v == par) continue;
            recover(u, v);
            dfs_mst(v, u);
        }
    };
    dfs_mst(0, -1);

    // Dedup edges and compute weight
    sort(tree_edges.begin(), tree_edges.end());
    auto it = unique(tree_edges.begin(), tree_edges.end());
    tree_edges.resize(it - tree_edges.begin());

    ll total = 0;
    for (auto [u, v] : tree_edges) {
        // find weight
        for (auto& ed : all_edges) {
            if ((ed.u == u && ed.v == v) || (ed.u == v && ed.v == u)) {
                total += ed.w;
                break;
            }
        }
    }

    cout << total << " " << tree_edges.size() << "\n";
    for (auto [u, v] : tree_edges) {
        cout << u << " " << v << "\n";
    }

    return 0;
}
```
