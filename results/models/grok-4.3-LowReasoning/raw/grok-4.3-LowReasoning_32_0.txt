```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#include <limits>
#include <tuple>

using namespace std;

struct Edge {
    int u, v, w;
    bool operator<(const Edge& other) const {
        return w < other.w;
    }
};

struct UF {
    vector<int> parent;
    UF(int n) : parent(n) {
        for (int i = 0; i < n; ++i) parent[i] = i;
    }
    int find(int x) {
        if (parent[x] != x) parent[x] = find(parent[x]);
        return parent[x];
    }
    bool unite(int x, int y) {
        int px = find(x), py = find(y);
        if (px == py) return false;
        parent[px] = py;
        return true;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M, K;
    cin >> N >> M >> K;

    vector<Edge> edges;
    edges.reserve(M);
    for (int i = 0; i < M; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        edges.push_back({u, v, w});
    }

    vector<bool> is_terminal(N, false);
    vector<int> terminals(K);
    for (int i = 0; i < K; ++i) {
        cin >> terminals[i];
        is_terminal[terminals[i]] = true;
    }

    // Kruskal MST
    sort(edges.begin(), edges.end());
    UF uf(N);
    vector<Edge> mst_edges;
    long long mst_weight = 0;
    int used = 0;
    for (const auto& e : edges) {
        if (uf.unite(e.u, e.v)) {
            mst_edges.push_back(e);
            mst_weight += e.w;
            ++used;
            if (used == N - 1) break;
        }
    }

    // Build adj list for tree (using original edges)
    vector<vector<pair<int, int>>> adj(N); // {neighbor, weight}
    vector<tuple<int, int, int>> tree_edge_list; // for output: u, v, w
    for (const auto& e : mst_edges) {
        adj[e.u].emplace_back(e.v, e.w);
        adj[e.v].emplace_back(e.u, e.w);
        tree_edge_list.emplace_back(min(e.u, e.v), max(e.u, e.v), e.w);
    }

    // Prune non-terminal leaves iteratively
    vector<bool> active(N, true);
    queue<int> leaves;
    for (int i = 0; i < N; ++i) {
        if (active[i] && adj[i].size() == 1 && !is_terminal[i]) {
            leaves.push(i);
        }
    }

    long long final_weight = mst_weight;
    int remaining_edges = static_cast<int>(mst_edges.size());

    while (!leaves.empty()) {
        int leaf = leaves.front();
        leaves.pop();
        if (!active[leaf] || is_terminal[leaf]) continue;
        active[leaf] = false;
        for (auto& p : adj[leaf]) {
            int nei = p.first;
            int w = p.second;
            if (active[nei]) {
                // remove edge
                final_weight -= w;
                --remaining_edges;
                // remove from nei's adj
                for (auto it = adj[nei].begin(); it != adj[nei].end(); ++it) {
                    if (it->first == leaf) {
                        adj[nei].erase(it);
                        break;
                    }
                }
                if (adj[nei].size() == 1 && !is_terminal[nei]) {
                    leaves.push(nei);
                }
            }
        }
        adj[leaf].clear();
    }

    // Collect remaining edges
    vector<pair<int, int>> result_edges;
    for (int i = 0; i < N; ++i) {
        if (!active[i]) continue;
        for (auto& p : adj[i]) {
            if (p.first > i && active[p.first]) {
                result_edges.emplace_back(i, p.first);
            }
        }
    }

    cout << final_weight << " " << result_edges.size() << "\n";
    for (auto& e : result_edges) {
        cout << e.first << " " << e.second << "\n";
    }

    return 0;
}
```
