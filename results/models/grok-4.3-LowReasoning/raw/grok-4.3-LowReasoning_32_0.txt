```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <tuple>

using namespace std;

using ll = long long;
const ll INF = numeric_limits<ll>::max() / 2;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    cin >> n >> m >> k;

    vector<vector<pair<int, ll>>> g(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        ll w;
        cin >> u >> v >> w;
        g[u].emplace_back(v, w);
        g[v].emplace_back(u, w);
    }

    vector<int> terms(k);
    for (int i = 0; i < k; ++i) {
        cin >> terms[i];
    }

    if (k == 0) {
        cout << "0 0\n";
        return 0;
    }
    if (k == 1) {
        cout << "0 0\n";
        return 0;
    }

    // All-pairs shortest paths from terminals using Dijkstra
    vector<vector<ll>> dist(k, vector<ll>(n, INF));
    vector<vector<int>> parent(k, vector<int>(n, -1));
    vector<int> term_idx(n, -1);
    for (int i = 0; i < k; ++i) {
        term_idx[terms[i]] = i;
    }

    auto dijkstra = [&](int src, int kidx) {
        vector<ll>& d = dist[kidx];
        vector<int>& p = parent[kidx];
        fill(d.begin(), d.end(), INF);
        fill(p.begin(), p.end(), -1);
        d[src] = 0;
        priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
        pq.emplace(0, src);
        while (!pq.empty()) {
            auto [cost, u] = pq.top(); pq.pop();
            if (cost > d[u]) continue;
            for (auto [v, w] : g[u]) {
                if (d[v] > d[u] + w) {
                    d[v] = d[u] + w;
                    p[v] = u;
                    pq.emplace(d[v], v);
                }
            }
        }
    };

    for (int i = 0; i < k; ++i) {
        dijkstra(terms[i], i);
    }

    // Build complete graph on terminals
    vector<tuple<ll, int, int>> edges;
    for (int i = 0; i < k; ++i) {
        for (int j = i + 1; j < k; ++j) {
            if (dist[i][terms[j]] < INF) {
                edges.emplace_back(dist[i][terms[j]], i, j);
            }
        }
    }

    sort(edges.begin(), edges.end());

    // Kruskal MST on terminals
    vector<int> par(k);
    for (int i = 0; i < k; ++i) par[i] = i;

    function<int(int)> find = [&](int x) -> int {
        return par[x] == x ? x : par[x] = find(par[x]);
    };

    vector<pair<int, int>> mst_term;
    ll mst_cost = 0;
    for (auto [w, i, j] : edges) {
        int pi = find(i), pj = find(j);
        if (pi != pj) {
            par[pi] = pj;
            mst_term.emplace_back(i, j);
            mst_cost += w;
        }
    }

    // Reconstruct actual edges
    vector<vector<pair<int, ll>>> tree(n);
    vector<pair<int, int>> result_edges;
    auto add_path = [&](int kidx, int target) {
        int cur = target;
        while (parent[kidx][cur] != -1) {
            int nxt = parent[kidx][cur];
            int a = min(cur, nxt), b = max(cur, nxt);
            result_edges.emplace_back(a, b);
            cur = nxt;
        }
    };

    for (auto [i, j] : mst_term) {
        add_path(i, terms[j]);
    }

    // Dedup edges and compute total weight
    sort(result_edges.begin(), result_edges.end());
    auto it = unique(result_edges.begin(), result_edges.end());
    result_edges.resize(it - result_edges.begin());

    ll total_w = 0;
    for (auto [u, v] : result_edges) {
        // find original weight
        for (auto [to, w] : g[u]) {
            if (to == v) {
                total_w += w;
                break;
            }
        }
    }

    // Output
    cout << total_w << ' ' << result_edges.size() << '\n';
    for (auto [u, v] : result_edges) {
        cout << u << ' ' << v << '\n';
    }

    return 0;
}
```
