#include <bits/stdc++.h>
using namespace std;

using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    cin >> J >> M;

    vector<vector<pair<int, ll>>> jobs(J);
    vector<int> task_counts(J);

    ll total_tasks = 0;
    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        task_counts[j] = T;
        jobs[j].resize(T);
        for (int t = 0; t < T; ++t) {
            int m;
            ll d;
            cin >> m >> d;
            jobs[j][t] = {m, d};
        }
        total_tasks += T;
    }

    // Adaptive strategy selection
    bool use_job_order = (total_tasks <= 100000); // always use simple for speed on large

    vector<ll> machine_free(M, 0);
    vector<vector<ll>> start_times(J);

    if (use_job_order) {
        // Simple sequential per job - always feasible and fast
        for (int j = 0; j < J; ++j) {
            ll current = 0;
            start_times[j].resize(task_counts[j]);
            for (int t = 0; t < task_counts[j]; ++t) {
                int m = jobs[j][t].first;
                ll d = jobs[j][t].second;
                ll st = max(current, machine_free[m]);
                start_times[j][t] = st;
                current = st + d;
                machine_free[m] = current;
            }
        }
    } else {
        // Placeholder for future more advanced heuristics (e.g., priority queue)
        // Falls back to same
        for (int j = 0; j < J; ++j) {
            ll current = 0;
            start_times[j].resize(task_counts[j]);
            for (int t = 0; t < task_counts[j]; ++t) {
                int m = jobs[j][t].first;
                ll d = jobs[j][t].second;
                ll st = max(current, machine_free[m]);
                start_times[j][t] = st;
                current = st + d;
                machine_free[m] = current;
            }
        }
    }

    // Output
    for (int j = 0; j < J; ++j) {
        for (int t = 0; t < task_counts[j]; ++t) {
            if (t > 0) cout << ' ';
            cout << start_times[j][t] << ' ' << jobs[j][t].first;
        }
        cout << '\n';
    }

    return 0;
}
