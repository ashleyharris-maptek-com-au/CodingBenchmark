#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int J, M;
    cin >> J >> M;
    vector<vector<pair<int, int>>> jobs(J);
    size_t total_tasks = 0;
    for (int j = 0; j < J; j++) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        for (int t = 0; t < T; t++) {
            int m, d;
            cin >> m >> d;
            jobs[j][t] = {m, d};
        }
        total_tasks += T;
    }
    vector<ll> mach_free(M, 0);
    vector<ll> job_time(J, 0);
    vector<vector<pair<ll, int>>> schedule(J);
    bool large_scale = (total_tasks > 10000 || J > 200);
    if (!large_scale) {
        // Simple sequential for small/medium
        for (int j = 0; j < J; j++) {
            for (auto& task : jobs[j]) {
                int m = task.first;
                int d = task.second;
                ll st = max(job_time[j], mach_free[m]);
                schedule[j].emplace_back(st, m);
                job_time[j] = st + d;
                mach_free[m] = st + d;
            }
        }
    } else {
        // For large: same greedy but with fast path (identical logic, optimized I/O already done)
        for (int j = 0; j < J; j++) {
            for (auto& task : jobs[j]) {
                int m = task.first;
                int d = task.second;
                ll st = max(job_time[j], mach_free[m]);
                schedule[j].emplace_back(st, m);
                job_time[j] = st + d;
                mach_free[m] = st + d;
            }
        }
    }
    for (int j = 0; j < J; j++) {
        for (size_t t = 0; t < schedule[j].size(); t++) {
            if (t > 0) cout << ' ';
            cout << schedule[j][t].first << ' ' << schedule[j][t].second;
        }
        cout << '\n';
    }
}
