```cpp
#include <bits/stdc++.h>
using namespace std;

using ll = long long;

struct Task {
    int machine;
    ll duration;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int J, M;
    cin >> J >> M;
    
    vector<vector<Task>> jobs(J);
    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        for (int t = 0; t < T; ++t) {
            cin >> jobs[j][t].machine >> jobs[j][t].duration;
        }
    }
    
    // Adaptive strategy selection based on scale
    bool use_simple = (J <= 100 && M <= 20); // trivial to medium: simple sequential
    // For extreme scale: still use same O(N) greedy (fast enough, 50k tasks)
    
    vector<ll> machine_free(M, 0LL);
    vector<vector<ll>> start_times(J);
    
    for (int j = 0; j < J; ++j) {
        int T = jobs[j].size();
        start_times[j].resize(T);
        ll prev_end = 0;
        for (int t = 0; t < T; ++t) {
            int m = jobs[j][t].machine;
            ll d = jobs[j][t].duration;
            ll st = max(prev_end, machine_free[m]);
            start_times[j][t] = st;
            machine_free[m] = st + d;
            prev_end = st + d;
        }
    }
    
    // Output
    for (int j = 0; j < J; ++j) {
        for (size_t t = 0; t < jobs[j].size(); ++t) {
            if (t > 0) cout << " ";
            cout << start_times[j][t] << " " << jobs[j][t].machine;
        }
        cout << "\n";
    }
    
    return 0;
}
```
