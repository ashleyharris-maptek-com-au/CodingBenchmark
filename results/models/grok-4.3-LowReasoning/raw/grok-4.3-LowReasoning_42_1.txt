```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
; Schema: 0

OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_color %gl_out %out_color
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_color Location 2
OpDecorate %out_color Location 0

%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%float_h = OpConstant %float 0.05
%float_nh = OpConstant %float -0.05
%float_0 = OpConstant %float 0.0
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_1
%_ptr_Input_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %_ptr_Input_arr_gl_PerVertex_in Input

%arr_v3float_1 = OpTypeArray %v3float %uint_1
%_ptr_Input_arr_v3float = OpTypePointer Input %arr_v3float_1
%in_color = OpVariable %_ptr_Input_arr_v3float Input

%gl_PerVertex_out = OpTypeStruct %v4float
%_ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %_ptr_Output_gl_PerVertex_out Output
%_ptr_Output_v3float = OpTypePointer Output %v3float
%out_color = OpVariable %_ptr_Output_v3float Output

%_ptr_Input_v4float = OpTypePointer Input %v4float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%void = OpTypeVoid
%func = OpTypeFunction %void

%main = OpFunction %void None %func
%entry = OpLabel

; load input position and color
%in0 = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%pos = OpLoad %v4float %in0
%col_arr = OpAccessChain %_ptr_Input_v3float %in_color %int_0
%col = OpLoad %v3float %col_arr

; extract x,y
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1
%z = OpCompositeExtract %float %pos 2
%w = OpCompositeExtract %float %pos 3

; top (x, y+h, z, w)
%top_y = OpFAdd %float %y %float_h
%top = OpCompositeConstruct %v4float %x %top_y %z %w

; right (x+h, y, z, w)
%right_x = OpFAdd %float %x %float_h
%right = OpCompositeConstruct %v4float %right_x %y %z %w

; left (x-h, y, z, w)
%left_x = OpFAdd %float %x %float_nh
%left = OpCompositeConstruct %v4float %left_x %y %z %w

; bottom (x, y-h, z, w)
%bot_y = OpFAdd %float %y %float_nh
%bottom = OpCompositeConstruct %v4float %x %bot_y %z %w

; emit top
%out0 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %out0 %top
OpStore %out_color %col
OpEmitVertex

; emit right
%out1 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %out1 %right
OpStore %out_color %col
OpEmitVertex

; emit left
%out2 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %out2 %left
OpStore %out_color %col
OpEmitVertex

; emit bottom
%out3 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %out3 %bottom
OpStore %out_color %col
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
