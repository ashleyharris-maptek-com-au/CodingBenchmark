; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
; Schema: 0

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%voidf = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%float_h = OpConstant %float 0.05

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_Input_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output

%arr_v3float = OpTypeArray %v3float %uint_1
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3float
%in_worldPos = OpVariable %ptr_Input_arr_v3 Input
%in_normal = OpVariable %ptr_Input_arr_v3 Input
%in_color = OpVariable %ptr_Input_arr_v3 Input

%ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3float Output

%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Output_v4float = OpTypePointer Output %v4float

%main = OpFunction %void None %voidf
%entry = OpLabel

; Load input position (gl_in[0].gl_Position)
%in_pos_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%in_pos = OpLoad %v4float %in_pos_ptr

; Load input color
%in_col_ptr = OpAccessChain %ptr_Input_v3float %in_color %int_0
%in_col = OpLoad %v3float %in_col_ptr

; Extract components
%px = OpCompositeExtract %float %in_pos 0
%py = OpCompositeExtract %float %in_pos 1
%pz = OpCompositeExtract %float %in_pos 2
%pw = OpCompositeExtract %float %in_pos 3

; Offsets
%py_top = OpFAdd %float %py %float_h
%px_right = OpFAdd %float %px %float_h
%py_bot = OpFSub %float %py %float_h
%px_left = OpFSub %float %px %float_h

; Build 4 diamond vertices
%pos_top = OpCompositeConstruct %v4float %px %py_top %pz %pw
%pos_right = OpCompositeConstruct %v4float %px_right %py %pz %pw
%pos_left = OpCompositeConstruct %v4float %px_left %py %pz %pw
%pos_bot = OpCompositeConstruct %v4float %px %py_bot %pz %pw

; Emit top
%top_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %top_ptr %pos_top
OpStore %outColor %in_col
OpEmitVertex

; Emit right
%right_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %right_ptr %pos_right
OpStore %outColor %in_col
OpEmitVertex

; Emit left
%left_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %left_ptr %pos_left
OpStore %outColor %in_col
OpEmitVertex

; Emit bottom
%bot_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %bot_ptr %pos_bot
OpStore %outColor %in_col
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
