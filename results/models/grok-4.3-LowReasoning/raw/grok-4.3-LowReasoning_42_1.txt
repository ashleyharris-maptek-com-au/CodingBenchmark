```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 100
OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %outColor %ubo
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%float_h = OpConstant %float 0.05

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_v3 = OpTypePointer Input %arr_v3_in
%in_worldPos = OpVariable %ptr_in_v3 Input
%in_normal = OpVariable %ptr_in_v3 Input
%in_color = OpVariable %ptr_in_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; UBO (same layout as test 41)
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_ubo Uniform

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %func
%entry = OpLabel

; Load input position (clip space)
%pin = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pin

; Load input color
%cin = OpAccessChain %ptr_in_v3 %in_color %int_0
%col = OpLoad %v3float %cin

; Half size offsets
%h = OpCompositeConstruct %v4float %float_h %float_h %float_h %float_h
%hx = OpCompositeConstruct %v4float %float_h %float_0 %float_0 %float_0
%hy = OpCompositeConstruct %v4float %float_0 %float_h %float_0 %float_0

; Diamond vertices (top, right, left, bottom)
%top = OpFAdd %v4float %pos %hy
%right = OpFAdd %v4float %pos %hx
%left = OpFSub %v4float %pos %hx
%bot = OpFSub %v4float %pos %hy

; Emit top
%o0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %o0 %top
%oc0 = OpAccessChain %ptr_out_v3 %outColor
OpStore %oc0 %col
OpEmitVertex

; Emit right
%o1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %o1 %right
%oc1 = OpAccessChain %ptr_out_v3 %outColor
OpStore %oc1 %col
OpEmitVertex

; Emit left
%o2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %o2 %left
%oc2 = OpAccessChain %ptr_out_v3 %outColor
OpStore %oc2 %col
OpEmitVertex

; Emit bottom
%o3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %o3 %bot
%oc3 = OpAccessChain %ptr_out_v3 %outColor
OpStore %oc3 %col
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
