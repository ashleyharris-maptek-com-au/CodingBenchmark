```glsl
#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 radii;
    vec4 weights;
    vec4 bounds;
};

void main() {
    uint id = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (id >= N) return;

    vec3 position = inData[id * 2].xyz;
    vec3 velocity = inData[id * 2 + 1].xyz;

    float dt = radii.x;
    float sepR = radii.y;
    float aliR = radii.z;
    float cohR = radii.w;
    float sepW = weights.x;
    float aliW = weights.y;
    float cohW = weights.z;
    float maxSpeed = weights.w;
    float boundSize = bounds.x;
    float boundForce = bounds.y;

    vec3 sep = vec3(0.0);
    vec3 ali = vec3(0.0);
    vec3 coh = vec3(0.0);
    int sepCount = 0;
    int aliCount = 0;
    int cohCount = 0;

    for (uint j = 0; j < N; ++j) {
        if (j == id) continue;
        vec3 oPos = inData[j * 2].xyz;
        vec3 oVel = inData[j * 2 + 1].xyz;
        vec3 diff = position - oPos;
        float dist = length(diff);
        if (dist < 0.001) continue;

        if (dist < sepR) {
            sep += diff / dist;
            sepCount++;
        }
        if (dist < aliR) {
            ali += oVel;
            aliCount++;
        }
        if (dist < cohR) {
            coh += oPos;
            cohCount++;
        }
    }

    vec3 accel = vec3(0.0);

    if (sepCount > 0) {
        accel += sepW * (sep / float(sepCount));
    }
    if (aliCount > 0) {
        accel += aliW * (ali / float(aliCount));
    }
    if (cohCount > 0) {
        vec3 avgPos = coh / float(cohCount);
        accel += cohW * (avgPos - position);
    }

    // Boundary force
    float limit = boundSize * 0.4;
    float center = boundSize * 0.5;
    if (position.x > limit) accel.x -= boundForce * center;
    if (position.x < -limit) accel.x += boundForce * center;
    if (position.y > limit) accel.y -= boundForce * center;
    if (position.y < -limit) accel.y += boundForce * center;
    if (position.z > limit) accel.z -= boundForce * center;
    if (position.z < -limit) accel.z += boundForce * center;

    velocity += accel * dt;
    float speed = length(velocity);
    if (speed > maxSpeed && speed > 0.0) {
        velocity = (velocity / speed) * maxSpeed;
    }
    position += velocity * dt;

    outData[id * 2] = vec4(position, 0.0);
    outData[id * 2 + 1] = vec4(velocity, 0.0);
}
```
