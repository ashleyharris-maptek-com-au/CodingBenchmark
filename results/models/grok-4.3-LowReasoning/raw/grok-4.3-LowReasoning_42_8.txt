```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

; gl_PerVertex input block (size 3 for triangles)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr3_pv = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr3_pv
%gl_in = OpVariable %ptr_in_arr_pv Input

; gl_PerVertex output block
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; User inputs (arrays of size 3)
%arr3_v3 = OpTypeArray %v3float %uint_3
%ptr_in_v3arr = OpTypePointer Input %arr3_v3

%in_worldPos = OpVariable %ptr_in_v3arr Input
OpDecorate %in_worldPos Location 0

%in_normal = OpVariable %ptr_in_v3arr Input
OpDecorate %in_normal Location 1

%in_color = OpVariable %ptr_in_v3arr Input
OpDecorate %in_color Location 2

; User output
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; Constants
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_3 = OpConstant %float 3.0
%float_s = OpConstant %float 0.7

; Main function
%main = OpFunction %void None %func
%entry = OpLabel

; Load 3 positions from gl_in
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%v0 = OpCompositeExtract %float %p0 0
%v1 = OpCompositeExtract %float %p0 1
%v2 = OpCompositeExtract %float %p0 2
%p0v3 = OpCompositeConstruct %v3float %v0 %v1 %v2

%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%w0 = OpCompositeExtract %float %p1 0
%w1 = OpCompositeExtract %float %p1 1
%w2 = OpCompositeExtract %float %p1 2
%p1v3 = OpCompositeConstruct %v3float %w0 %w1 %w2

%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%x0 = OpCompositeExtract %float %p2 0
%x1 = OpCompositeExtract %float %p2 1
%x2 = OpCompositeExtract %float %p2 2
%p2v3 = OpCompositeConstruct %v3float %x0 %x1 %x2

; Centroid C = (p0 + p1 + p2) / 3
%sum01 = OpFAdd %v3float %p0v3 %p1v3
%sum   = OpFAdd %v3float %sum01 %p2v3
%centroid = OpVectorTimesScalar %v3float %sum %float_3

; Shrink factor s = 0.7
; For i=0
%diff0 = OpFSub %v3float %p0v3 %centroid
%scaled0 = OpVectorTimesScalar %v3float %diff0 %float_s
%new0 = OpFAdd %v3float %centroid %scaled0
%new0_4 = OpCompositeConstruct %v4float %new0 %float_1

; For i=1
%diff1 = OpFSub %v3float %p1v3 %centroid
%scaled1 = OpVectorTimesScalar %v3float %diff1 %float_s
%new1 = OpFAdd %v3float %centroid %scaled1
%new1_4 = OpCompositeConstruct %v4float %new1 %float_1

; For i=2
%diff2 = OpFSub %v3float %p2v3 %centroid
%scaled2 = OpVectorTimesScalar %v3float %diff2 %float_s
%new2 = OpFAdd %v3float %centroid %scaled2
%new2_4 = OpCompositeConstruct %v4float %new2 %float_1

; Load colors
%c0_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
%c1 = OpLoad %v3float %c1_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %in_color %int_2
%c2 = OpLoad %v3float %c2_ptr

; Emit vertex 0
%out_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos0 %new0_4
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 1
%out_pos1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos1 %new1_4
OpStore %outColor %c1
OpEmitVertex

; Emit vertex 2
%out_pos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos2 %new2_4
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
