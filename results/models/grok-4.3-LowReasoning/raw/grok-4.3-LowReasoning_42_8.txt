```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 100
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %out_color
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %out_color Location 0

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_3 = OpConstant %float 3
%float_0_7 = OpConstant %float 0.7

; gl_PerVertex input block
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_gl_PerVertex = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr_gl_PerVertex Input

; User inputs
%arr_v3float_3 = OpTypeArray %v3float %uint_3
%ptr_in_v3arr = OpTypePointer Input %arr_v3float_3
%in_worldPos = OpVariable %ptr_in_v3arr Input
%in_normal = OpVariable %ptr_in_v3arr Input
%in_color = OpVariable %ptr_in_v3arr Input

; Output block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_PerVertex Output

; Output color
%ptr_out_v3 = OpTypePointer Output %v3float
%out_color = OpVariable %ptr_out_v3 Output

; Main
%main = OpFunction %void None %func
%entry = OpLabel

; Load input positions
%ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_p0
%ptr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr_p1
%ptr_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %ptr_p2

; Compute centroid (xyz only)
%xyz0 = OpVectorShuffle %v3float %p0 %p0 0 1 2
%xyz1 = OpVectorShuffle %v3float %p1 %p1 0 1 2
%xyz2 = OpVectorShuffle %v3float %p2 %p2 0 1 2
%sum01 = OpFAdd %v3float %xyz0 %xyz1
%sum = OpFAdd %v3float %sum01 %xyz2
%centroid = OpFDiv %v3float %sum %float_3

; Shrink factor
%s = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_0_7

; Process vertex 0
%diff0 = OpFSub %v3float %xyz0 %centroid
%scaled0 = OpFMul %v3float %diff0 %s
%new0 = OpFAdd %v3float %centroid %scaled0
%new_p0 = OpCompositeConstruct %v4float %new0 %float_1
%ptr_out0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out0 %new_p0
%ptr_c0 = OpAccessChain %ptr_in_v3 %in_color %int_0
%c0 = OpLoad %v3float %ptr_c0
OpStore %out_color %c0
OpEmitVertex

; Process vertex 1
%diff1 = OpFSub %v3float %xyz1 %centroid
%scaled1 = OpFMul %v3float %diff1 %s
%new1 = OpFAdd %v3float %centroid %scaled1
%new_p1 = OpCompositeConstruct %v4float %new1 %float_1
%ptr_out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out1 %new_p1
%ptr_c1 = OpAccessChain %ptr_in_v3 %in_color %int_1
%c1 = OpLoad %v3float %ptr_c1
OpStore %out_color %c1
OpEmitVertex

; Process vertex 2
%diff2 = OpFSub %v3float %xyz2 %centroid
%scaled2 = OpFMul %v3float %diff2 %s
%new2 = OpFAdd %v3float %centroid %scaled2
%new_p2 = OpCompositeConstruct %v4float %new2 %float_1
%ptr_out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out2 %new_p2
%ptr_c2 = OpAccessChain %ptr_in_v3 %in_color %int_2
%c2 = OpLoad %v3float %ptr_c2
OpStore %out_color %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd

; Helper pointer types (declared for clarity)
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%float_1 = OpConstant %float 1
```
