import math

# Module-level persistence
_prev_y_err = 0.0
_prev_x_err = 0.0
_prev_z_err = 0.0
_integral_y = 0.0
_phase = "acquire"
_last_range = 9999.0
_fuel_used = 0.0

def _safe(val, valid_key, state, default=0.0):
    if state.get(valid_key, False):
        return state.get(val, default)
    return default

def autopilot_step(state, dt):
    global _prev_y_err, _prev_x_err, _prev_z_err, _integral_y, _phase, _last_range

    # Prefer lidar > radar > docking camera for relative state
    rx = _safe("lidar_rel_pos_x_m", "lidar_rel_x_valid", state,
               _safe("radar_rel_pos_x_m", "radar_rel_x_valid", state, 0.0))
    ry = _safe("lidar_rel_pos_y_m", "lidar_rel_y_valid", state,
               _safe("radar_rel_pos_y_m", "radar_rel_y_valid", state, 0.0))
    rz = _safe("lidar_rel_pos_z_m", "lidar_rel_z_valid", state,
               _safe("radar_rel_pos_z_m", "radar_rel_z_valid", state, 0.0))

    vx = _safe("lidar_rel_vel_x_ms", "lidar_relv_x_valid", state,
               _safe("radar_rel_vel_x_ms", "radar_relv_x_valid", state, 0.0))
    vy = _safe("lidar_rel_vel_y_ms", "lidar_relv_y_valid", state,
               _safe("radar_rel_vel_y_ms", "radar_relv_y_valid", state, 0.0))
    vz = _safe("lidar_rel_vel_z_ms", "lidar_relv_z_valid", state,
               _safe("radar_rel_vel_z_ms", "radar_relv_z_valid", state, 0.0))

    # Fallback to docking camera range/bearing when close
    cam_range = state.get("dock_camera_range_m", 999.0)
    cam_active = state.get("dock_camera_active", False)
    if cam_active and cam_range < 80:
        rx = rx or 0.0
        ry = -cam_range  # assume camera points along -y for V-bar
        vx, vy, vz = 0.0, -0.3, 0.0  # conservative when camera dominant

    range_est = math.sqrt(rx**2 + ry**2 + rz**2)
    if range_est < 1.0:
        range_est = state.get("target_dock_range_m", range_est)

    # Simple phase machine
    if range_est > 180:
        _phase = "far"
    elif range_est > 35:
        _phase = "medium"
    else:
        _phase = "close"

    # Target velocity profile (very gentle final approach)
    if _phase == "far":
        target_vy = -0.8 if ry < -50 else -0.4
    elif _phase == "medium":
        target_vy = -0.25
    else:
        target_vy = -0.08 if range_est > 8 else -0.03

    # PD control with orbital awareness
    # y (along-track): use retrograde to close when behind
    y_err = ry
    y_derr = (y_err - _prev_y_err) / max(dt, 0.01)
    _integral_y = max(min(_integral_y + y_err * dt, 30), -30)
    thrust_y = -0.6 * (y_err / 200.0) - 1.8 * (vy - target_vy) - 0.01 * _integral_y
    thrust_y = max(min(thrust_y, 0.9), -0.9)

    # x (radial) - keep near zero with small corrections
    x_err = rx
    x_derr = (x_err - _prev_x_err) / max(dt, 0.01)
    thrust_x = -0.35 * (x_err / 100.0) - 0.9 * vx
    thrust_x = max(min(thrust_x, 0.6), -0.6)

    # z (cross-track)
    z_err = rz
    z_derr = (z_err - _prev_z_err) / max(dt, 0.01)
    thrust_z = -0.4 * (z_err / 80.0) - 1.1 * vz
    thrust_z = max(min(thrust_z, 0.7), -0.7)

    # Attitude rates (point nose roughly along velocity vector in LVLH)
    att_rate_pitch = -0.008 * rx - 0.015 * vx
    att_rate_yaw = 0.008 * rz + 0.015 * vz
    att_rate_roll = 0.0

    # Very conservative near docking port
    if range_est < 12:
        thrust_x *= 0.4
        thrust_y *= 0.35
        thrust_z *= 0.4
        att_rate_pitch *= 0.5
        att_rate_yaw *= 0.5

    # Clamp all outputs
    thrust_x = max(min(thrust_x, 1.0), -1.0)
    thrust_y = max(min(thrust_y, 1.0), -1.0)
    thrust_z = max(min(thrust_z, 1.0), -1.0)
    att_rate_pitch = max(min(att_rate_pitch, 0.035), -0.035)
    att_rate_yaw = max(min(att_rate_yaw, 0.035), -0.035)
    att_rate_roll = max(min(att_rate_roll, 0.035), -0.035)

    _prev_y_err = y_err
    _prev_x_err = x_err
    _prev_z_err = z_err
    _last_range = range_est

    return {
        "thrust_x": thrust_x,
        "thrust_y": thrust_y,
        "thrust_z": thrust_z,
        "att_rate_roll": att_rate_roll,
        "att_rate_pitch": att_rate_pitch,
        "att_rate_yaw": att_rate_yaw,
    }
