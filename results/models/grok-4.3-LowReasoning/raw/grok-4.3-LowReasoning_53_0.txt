import math

# Module-level persistence
_prev_err_y = 0.0
_prev_err_x = 0.0
_prev_err_z = 0.0
_integral_y = 0.0
_integral_x = 0.0
_integral_z = 0.0
_phase = 0
_last_range = 1e9
_fuel_used = 0.0
_thrust_history = []

def _safe_get(state, key, default=0.0):
    val = state.get(key, default)
    if val is None:
        return default
    return val

def _get_best_rel_state(state):
    # Prefer lidar > radar > range finders
    x = y = z = vx = vy = vz = 0.0
    valid = False

    # Lidar first
    if _safe_get(state, 'lidar_rel_x_valid', False):
        x = _safe_get(state, 'lidar_rel_pos_x_m')
        y = _safe_get(state, 'lidar_rel_pos_y_m')
        z = _safe_get(state, 'lidar_rel_pos_z_m')
        vx = _safe_get(state, 'lidar_rel_vel_x_ms')
        vy = _safe_get(state, 'lidar_rel_vel_y_ms')
        vz = _safe_get(state, 'lidar_rel_vel_z_ms')
        valid = True
    elif _safe_get(state, 'radar_rel_x_valid', False):
        x = _safe_get(state, 'radar_rel_pos_x_m')
        y = _safe_get(state, 'radar_rel_pos_y_m')
        z = _safe_get(state, 'radar_rel_pos_z_m')
        vx = _safe_get(state, 'radar_rel_vel_x_ms')
        vy = _safe_get(state, 'radar_rel_vel_y_ms')
        vz = _safe_get(state, 'radar_rel_vel_z_ms')
        valid = True
    else:
        # Fall back to range finders (scalar distance)
        ranges = []
        for i in [1, 2, 3]:
            if _safe_get(state, f'range_{i}_valid', False):
                r = _safe_get(state, f'range_{i}_m', 1e9)
                if r < 10000:
                    ranges.append(r)
        if ranges:
            r = min(ranges)
            # Assume mostly along-track when far (common case)
            y = -r if _safe_get(state, 'target_dock_range_m', r) > 0 else r
            valid = True

    return x, y, z, vx, vy, vz, valid

def autopilot_step(state: dict, dt: float) -> dict:
    global _prev_err_y, _prev_err_x, _prev_err_z
    global _integral_y, _integral_x, _integral_z
    global _phase, _last_range, _fuel_used

    x, y, z, vx, vy, vz, rel_valid = _get_best_rel_state(state)
    range_m = _safe_get(state, 'target_dock_range_m', abs(y) if y != 0 else 1000.0)
    tgt_speed = _safe_get(state, 'target_dock_speed_ms', 0.0)

    # Simple fuel tracking
    fuel_pct = _safe_get(state, 'fuel_remaining_pct', 100.0)
    if fuel_pct < 5.0:
        return {'thrust_x': 0.0, 'thrust_y': 0.0, 'thrust_z': 0.0,
                'att_rate_roll': 0.0, 'att_rate_pitch': 0.0, 'att_rate_yaw': 0.0}

    # Phase management
    if range_m > 500:
        _phase = 0
    elif range_m > 50:
        _phase = 1
    else:
        _phase = 2

    # Attitude: keep rates near zero (stable docking attitude)
    att_roll = 0.0
    att_pitch = 0.0
    att_yaw = 0.0

    thrust_x = thrust_y = thrust_z = 0.0

    if not rel_valid or range_m > 2000:
        # Very far or no good data - coast or small corrective burn
        if y < -800:  # behind
            thrust_y = -0.15  # gentle retrograde to lower orbit
        return {'thrust_x': thrust_x, 'thrust_y': thrust_y, 'thrust_z': thrust_z,
                'att_rate_roll': att_roll, 'att_rate_pitch': att_pitch, 'att_rate_yaw': att_yaw}

    # === Far phase: Hohmann-style (V-bar approach) ===
    if _phase == 0:
        if y < -100:  # behind target
            # Retrograde burn to lower orbit and catch up
            thrust_y = -0.25 if abs(vy) < 0.8 else 0.0
        elif y > 100:  # ahead
            thrust_y = 0.2  # prograde to raise orbit
        else:
            # Small radial correction if needed
            if abs(x) > 30:
                thrust_x = -0.1 * math.copysign(1, x)

    # === Medium phase: gentle V-bar approach ===
    elif _phase == 1:
        # Close y gap while keeping relative velocity low
        err_y = y
        desired_vy = -0.4 if y < 0 else 0.2  # approach speed
        err_vy = vy - desired_vy

        _integral_y += err_y * dt * 0.05
        _integral_y = max(min(_integral_y, 5.0), -5.0)

        thrust_y = -0.18 * err_y - 0.6 * err_vy + 0.02 * _integral_y
        thrust_y = max(min(thrust_y, 0.6), -0.6)

        # Radial correction
        if abs(x) > 8:
            thrust_x = -0.12 * math.copysign(1, x) - 0.4 * vx

    # === Close phase: linear regime (Clohessy-Wiltshire approximation) ===
    else:
        # Very gentle final approach
        desired_v = -0.08 if range_m > 8 else -0.03
        err_y = y
        err_vy = vy - desired_v

        _integral_y += err_y * dt * 0.1
        _integral_y = max(min(_integral_y, 3.0), -3.0)

        thrust_y = -0.25 * err_y - 0.9 * err_vy + 0.03 * _integral_y
        thrust_y = max(min(thrust_y, 0.4), -0.4)

        # Null out radial and cross-track motion
        thrust_x = -0.5 * x - 0.8 * vx
        thrust_z = -0.5 * z - 0.8 * vz

        thrust_x = max(min(thrust_x, 0.35), -0.35)
        thrust_z = max(min(thrust_z, 0.35), -0.35)

    # Clamp all thrusts
    thrust_x = max(min(thrust_x, 0.9), -0.9)
    thrust_y = max(min(thrust_y, 0.9), -0.9)
    thrust_z = max(min(thrust_z, 0.9), -0.9)

    # Update persistence
    _prev_err_y = y
    _prev_err_x = x
    _prev_err_z = z
    _last_range = range_m

    return {
        'thrust_x': thrust_x,
        'thrust_y': thrust_y,
        'thrust_z': thrust_z,
        'att_rate_roll': att_roll,
        'att_rate_pitch': att_pitch,
        'att_rate_yaw': att_yaw
    }
