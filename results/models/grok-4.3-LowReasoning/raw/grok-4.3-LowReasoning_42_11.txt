```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_gl_PerVertex_in ArrayStride 16
OpDecorate %gl_in DescriptorSet 0
OpDecorate %gl_in Binding 0

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %out_color Location 0

OpDecorate %in_color Location 2

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3float_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3float_4
%in_color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%out_color = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%main = OpFunction %void None %func
%entry = OpLabel

; Load p0, p1=start, p2=end, p3
%chain_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%start = OpLoad %v4float %chain_p1
%chain_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%end = OpLoad %v4float %chain_p2

; Tangent from adjacency for smooth perpendicular
%chain_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %chain_p0
%chain_p3 = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %chain_p3

%dir = OpFSub %v4float %end %start
%tan_in = OpFSub %v4float %start %p0
%tan_out = OpFSub %v4float %p3 %end
%tan_avg = OpFAdd %v4float %tan_in %tan_out
%tan_n = OpCompositeConstruct %v4float %float_0 %float_0 %float_0 %float_0 ; placeholder normalize
; simple perpendicular in xy (assume clip-space 2D projection)
%dx = OpCompositeExtract %float %dir 0
%dy = OpCompositeExtract %float %dir 1
%nx = OpFNegate %float %dy
%ny = OpFAdd %float %dx %float_0
%len = OpFMul %float %dx %dx
%len2 = OpFMul %float %dy %dy
%len_s = OpFAdd %float %len %len2
%len_r = OpExtInst %float %sqrt %len_s
%inv_len = OpFDiv %float %float_1 %len_r
%px = OpFMul %float %nx %inv_len
%py = OpFMul %float %ny %inv_len
%half_thick = OpConstant %float 0.01
%offx = OpFMul %float %px %half_thick
%offy = OpFMul %float %py %half_thick
%offset = OpCompositeConstruct %v4float %offx %offy %float_0 %float_0

; Load colors
%col_chain1 = OpAccessChain %ptr_in_v3 %in_color %int_1
%col1 = OpLoad %v3float %col_chain1
%col_chain2 = OpAccessChain %ptr_in_v3 %in_color %int_2
%col2 = OpLoad %v3float %col_chain2

; Segment split: 3 equal parts, draw 1st and 3rd
; Part 1: start -> 1/3
%third = OpConstant %float 0.333333
%delta = OpFSub %v4float %end %start
%delta13 = OpVectorTimesScalar %v4float %delta %third
%pt1 = OpFAdd %v4float %start %delta13

; Quad 1 (part 1)
%q1a = OpFAdd %v4float %start %offset
%q1b = OpFSub %v4float %start %offset
%q1c = OpFAdd %v4float %pt1 %offset
%q1d = OpFSub %v4float %pt1 %offset

; Emit quad 1
OpStore %out_color %col1
%store1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %store1 %q1a
OpEmitVertex

OpStore %out_color %col1
%store2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %store2 %q1b
OpEmitVertex

OpStore %out_color %col2
%store3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %store3 %q1c
OpEmitVertex

OpStore %out_color %col2
%store4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %store4 %q1d
OpEmitVertex
OpEndPrimitive

; Part 3: 2/3 -> end
%pt2 = OpFSub %v4float %end %delta13

; Quad 2 (part 3)
%q2a = OpFAdd %v4float %pt2 %offset
%q2b = OpFSub %v4float %pt2 %offset
%q2c = OpFAdd %v4float %end %offset
%q2d = OpFSub %v4float %end %offset

; Emit quad 2
OpStore %out_color %col1
%store5 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %store5 %q2a
OpEmitVertex

OpStore %out_color %col1
%store6 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %store6 %q2b
OpEmitVertex

OpStore %out_color %col2
%store7 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %store7 %q2c
OpEmitVertex

OpStore %out_color %col2
%store8 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %store8 %q2d
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
```
