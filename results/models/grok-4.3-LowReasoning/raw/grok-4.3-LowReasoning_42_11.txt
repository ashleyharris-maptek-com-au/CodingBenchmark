```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 256
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4
%uint_8 = OpConstant %uint 8
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_02 = OpConstant %float 0.02
%float_3 = OpConstant %float 3
%float_1_3 = OpConstant %float 0.333333
%float_2_3 = OpConstant %float 0.666666
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_v3 = OpTypePointer Input %v3float
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%main = OpFunction %void None %func
%entry = OpLabel
; load positions
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
% p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_ptr
; colors
%c1_ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
%c1 = OpLoad %v3float %c1_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %in_color %int_2
%c2 = OpLoad %v3float %c2_ptr
; compute dir and perps using adjacency
%start_to_end = OpFSub %v4float %p2 %p1
%dir3 = OpVectorShuffle %v3float %start_to_end %start_to_end 0 1 2
%len = OpDot %float %dir3 %dir3
%sqrt_len = OpExtInst %float %sqrt %len
%dir = OpFDiv %v3float %dir3 %sqrt_len
%perp = OpCompositeConstruct %v3float %float_0 %float_0 %float_1 ; fallback z-up
%cross1 = OpExtInst %v3float %cross %dir %perp
%cross_len = OpDot %float %cross1 %cross1
%cross_s = OpExtInst %float %sqrt %cross_len
%norm_perp = OpFDiv %v3float %cross1 %cross_s
%offset = OpVectorTimesScalar %v3float %norm_perp %float_0_02
; split points (1/3 and 2/3)
%seg = OpFSub %v3float %dir3 %float_0
%third = OpVectorTimesScalar %v3float %seg %float_1_3
%pA = OpFAdd %v3float %p1 %third
%pB = OpFAdd %v3float %p1 %third
%two_third = OpVectorTimesScalar %v3float %seg %float_2_3
%pC = OpFAdd %v3float %p1 %two_third
%pD = OpFAdd %v3float %p1 %two_third
; Quad 1 (1st part)
%q1_v0 = OpFAdd %v3float %p1 %offset
%q1_v1 = OpFSub %v3float %p1 %offset
%q1_v2 = OpFAdd %v3float %pA %offset
%q1_v3 = OpFSub %v3float %pA %offset
; emit quad1
OpStore %outColor %c1
%out0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%v0_4 = OpCompositeConstruct %v4float %q1_v0 %float_1
OpStore %out0 %v0_4
OpEmitVertex
OpStore %outColor %c1
%out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%v1_4 = OpCompositeConstruct %v4float %q1_v1 %float_1
OpStore %out1 %v1_4
OpEmitVertex
OpStore %outColor %c1
%out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%v2_4 = OpCompositeConstruct %v4float %q1_v2 %float_1
OpStore %out2 %v2_4
OpEmitVertex
OpStore %outColor %c1
%out3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%v3_4 = OpCompositeConstruct %v4float %q1_v3 %float_1
OpStore %out3 %v3_4
OpEmitVertex
OpEndPrimitive
; Quad 2 (3rd part)
%q2_v0 = OpFAdd %v3float %pC %offset
%q2_v1 = OpFSub %v3float %pC %offset
%q2_v2 = OpFAdd %v3float %p2 %offset
%q2_v3 = OpFSub %v3float %p2 %offset
OpStore %outColor %c2
%o0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%qv0 = OpCompositeConstruct %v4float %q2_v0 %float_1
OpStore %o0 %qv0
OpEmitVertex
OpStore %outColor %c2
%o1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%qv1 = OpCompositeConstruct %v4float %q2_v1 %float_1
OpStore %o1 %qv1
OpEmitVertex
OpStore %outColor %c2
%o2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%qv2 = OpCompositeConstruct %v4float %q2_v2 %float_1
OpStore %o2 %qv2
OpEmitVertex
OpStore %outColor %c2
%o3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%qv3 = OpCompositeConstruct %v4float %q2_v3 %float_1
OpStore %o3 %qv3
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
```
