```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_color Location 2
OpDecorate %out_color Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%v2float = OpTypeVector %float 2
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_333 = OpConstant %float 0.333333
%float_0_666 = OpConstant %float 0.666667
%float_half_thick = OpConstant %float 0.01
%float_0_5 = OpConstant %float 0.5

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%in_color = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

%out_color = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func
%entry = OpLabel

; Load positions
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%P1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%P2 = OpLoad %v4float %p2_ptr

; Compute direction and perpendicular (clip-space xy)
%dir4 = OpFSub %v4float %P2 %P1
%dir = OpVectorShuffle %v2float %dir4 %dir4 0 1
%dir_len = OpCompositeExtract %float %dir 0
%dir_len2 = OpCompositeExtract %float %dir 1
%dir_len_sq = OpFMul %float %dir_len %dir_len
%tmp = OpFMul %float %dir_len2 %dir_len2
%len2 = OpFAdd %float %dir_len_sq %tmp
%len = OpSqrt %float %len2
%dir_n = OpFDiv %v2float %dir %len
%neg_y = OpCompositeExtract %float %dir_n 1
%neg_y = OpFNegate %float %neg_y
%px = OpCompositeExtract %float %dir_n 0
%perp = OpCompositeConstruct %v2float %neg_y %px
%perp_scaled = OpVectorTimesScalar %v2float %perp %float_half_thick

; Build offset vectors (preserve zw from original)
%offP = OpCompositeConstruct %v4float %perp_scaled %float_0 %float_0
%offM = OpCompositeConstruct %v4float %float_0 %float_0 %float_0 %float_0
%offM = OpFSub %v4float %offP %offP   ; zero then subtract to get negative
%offM = OpFNegate %v4float %offP

; === First dash segment (0 -> 1/3) ===
%B = OpFMix %v4float %P1 %P2 %float_0_333

; Quad 1: A=P1, E=B
%V0 = OpFAdd %v4float %P1 %offP
%V1 = OpFAdd %v4float %P1 %offM
%V2 = OpFAdd %v4float %B %offP
%V3 = OpFAdd %v4float %B %offM

; Emit quad 1
OpStore %out_color %float_0_5   ; dummy (will overwrite with real color)
%col1_ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
%Col = OpLoad %v3float %col1_ptr
OpStore %out_color %Col

%pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos0 %V0
OpEmitVertex

%pos1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos1 %V1
OpEmitVertex

%pos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos2 %V2
OpEmitVertex

%pos3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos3 %V3
OpEmitVertex
OpEndPrimitive

; === Second dash segment (2/3 -> 1) ===
%C = OpFMix %v4float %P1 %P2 %float_0_666

; Quad 2: C to P2
%V4 = OpFAdd %v4float %C %offP
%V5 = OpFAdd %v4float %C %offM
%V6 = OpFAdd %v4float %P2 %offP
%V7 = OpFAdd %v4float %P2 %offM

; Emit quad 2
OpStore %out_color %Col
%pos4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos4 %V4
OpEmitVertex

%pos5 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos5 %V5
OpEmitVertex

%pos6 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos6 %V6
OpEmitVertex

%pos7 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos7 %V7
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
```
