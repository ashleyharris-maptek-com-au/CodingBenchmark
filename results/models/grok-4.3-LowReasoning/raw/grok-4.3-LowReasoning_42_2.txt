; Generator: manual

               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %in_color %gl_out %out_color
               OpExecutionMode %main InputPoints
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 18

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %arr_pv_in ArrayStride 16
               OpDecorate %gl_in DescriptorSet 0
               OpDecorate %gl_in Binding 0

               OpDecorate %in_color Location 2
               OpDecorate %arr_color ArrayStride 12

               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_out DescriptorSet 0
               OpDecorate %gl_out Binding 1

               OpDecorate %out_color Location 0

       %void = OpTypeVoid
       %func = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
    %v3float = OpTypeVector %float 3
       %int = OpTypeInt 32 1
      %uint = OpTypeInt 32 0

%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
     %gl_in = OpVariable %ptr_in_arr_pv Input

 %arr_color = OpTypeArray %v3float %uint_1
%ptr_in_color = OpTypePointer Input %arr_color
  %in_color = OpVariable %ptr_in_color Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
    %gl_out = OpVariable %ptr_out_pv Output

%ptr_out_color = OpTypePointer Output %v3float
  %out_color = OpVariable %ptr_out_color Output

    %uint_1 = OpConstant %uint 1
    %int_0 = OpConstant %int 0
    %int_1 = OpConstant %int 1
    %float_0 = OpConstant %float 0
    %float_1 = OpConstant %float 1
  %float_r = OpConstant %float 0.04
 %float_0_7 = OpConstant %float 0.7
%float_pi = OpConstant %float 3.1415926535
%float_2pi = OpConstant %float 6.283185307
%float_60deg = OpConstant %float 1.0471975512

 %glsl450 = OpExtInstImport "GLSL.std.450"

       %main = OpFunction %void None %func
      %entry = OpLabel

; Load input data
   %in_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
     %center4 = OpLoad %v4float %in_pos_ptr
    %center3 = OpVectorShuffle %v3float %center4 %center4 0 1 2

   %in_col_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
     %inColor = OpLoad %v3float %in_col_ptr
    %edgeColor = OpVectorTimesScalar %v3float %inColor %float_0_7

; Emit 6 triangles (center, edge_i, edge_{i+1})
       %i = OpVariable %ptr_int Function
            OpStore %i %int_0

  %loop_cond = OpLabel
        %cur = OpLoad %int %i
      %cmp = OpSLessThan %bool %cur %int_6
            OpBranchConditional %cmp %loop_body %loop_end

  %loop_body = OpLabel
        %idx = OpLoad %int %i
      %angle0 = OpConvertSToF %float %idx
      %angle0 = OpFMul %float %angle0 %float_60deg
        %c0 = OpExtInst %float %glsl450 Cos %angle0
        %s0 = OpExtInst %float %glsl450 Sin %angle0
      %off0 = OpCompositeConstruct %v3float %c0 %s0 %float_0
      %off0 = OpVectorTimesScalar %v3float %off0 %float_r
      %edge0 = OpFAdd %v3float %center3 %off0
      %edge0_4 = OpCompositeConstruct %v4float %edge0 %float_1

      %next = OpIAdd %int %idx %int_1
     %angle1 = OpConvertSToF %float %next
     %angle1 = OpFMul %float %angle1 %float_60deg
        %c1 = OpExtInst %float %glsl450 Cos %angle1
        %s1 = OpExtInst %float %glsl450 Sin %angle1
      %off1 = OpCompositeConstruct %v3float %c1 %s1 %float_0
      %off1 = OpVectorTimesScalar %v3float %off1 %float_r
      %edge1 = OpFAdd %v3float %center3 %off1
      %edge1_4 = OpCompositeConstruct %v4float %edge1 %float_1

; Emit triangle: center, edge0, edge1
   %pos0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
            OpStore %pos0_ptr %center4
   %col0_ptr = OpAccessChain %ptr_out_v3 %out_color
            OpStore %col0_ptr %inColor
            OpEmitVertex

   %pos1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
            OpStore %pos1_ptr %edge0_4
   %col1_ptr = OpAccessChain %ptr_out_v3 %out_color
            OpStore %col1_ptr %edgeColor
            OpEmitVertex

   %pos2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
            OpStore %pos2_ptr %edge1_4
   %col2_ptr = OpAccessChain %ptr_out_v3 %out_color
            OpStore %col2_ptr %edgeColor
            OpEmitVertex

            OpEndPrimitive

      %nexti = OpIAdd %int %idx %int_1
            OpStore %i %nexti
            OpBranch %loop_cond

   %loop_end = OpLabel
            OpReturn
            OpFunctionEnd
