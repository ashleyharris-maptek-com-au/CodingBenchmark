```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 256
OpCapability Geometry
OpExtension "SPV_KHR_vulkan_memory_model"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_gl_PerVertex_in ArrayStride 16

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_1 = OpConstant %uint 1
%uint_18 = OpConstant %uint 18

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr Input

%arr_v3 = OpTypeArray %v3float %uint_1
%ptr_in_v3arr = OpTypePointer Input %arr_v3
%worldPos = OpVariable %ptr_in_v3arr Input
%normal = OpVariable %ptr_in_v3arr Input
%color = OpVariable %ptr_in_v3arr Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_ubo Uniform

%glsl = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %func
%entry = OpLabel

; Load input point
%in_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%in_pos = OpLoad %v4float %in_pos_ptr
%in_col_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%in_col = OpLoad %v3float %in_col_ptr

; radius
%r = OpConstant %float 0.04
%two_pi = OpConstant %float 6.2831853
%six = OpConstant %float 6.0
%pi_3 = OpConstant %float 1.04719755   ; 60 deg

; darker color for edges
%dark = OpConstant %float 0.6
%edge_col = OpVectorTimesScalar %v3float %in_col %dark

; Emit 6 triangles (center + two edge verts each)
%zero = OpConstant %float 0.0
%i = OpVariable %ptr_out_pv Function   ; not really needed, just label

; helper macro idea - unrolled for clarity
OpStore %outColor %in_col
%center_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %center_pos %in_pos
OpEmitVertex

; Triangle 0: center, e0, e1
OpStore %outColor %edge_col
%ang0 = OpConstant %float 0.0
%dx0 = OpExtInst %float %glsl Cos %ang0
%dy0 = OpExtInst %float %glsl Sin %ang0
%off0 = OpCompositeConstruct %v4float %dx0 %dy0 %zero %zero
%e0 = OpFAdd %v4float %in_pos %off0
OpStore %center_pos %e0
OpEmitVertex

%ang1 = %pi_3
%dx1 = OpExtInst %float %glsl Cos %ang1
%dy1 = OpExtInst %float %glsl Sin %ang1
%off1 = OpCompositeConstruct %v4float %dx1 %dy1 %zero %zero
%e1 = OpFAdd %v4float %in_pos %off1
OpStore %center_pos %e1
OpEmitVertex
OpEndPrimitive

; Triangle 1: center, e1, e2
OpStore %outColor %in_col
OpStore %center_pos %in_pos
OpEmitVertex
OpStore %outColor %edge_col
OpStore %center_pos %e1
OpEmitVertex

%ang2 = OpFMul %float %pi_3 %float_2
%dx2 = OpExtInst %float %glsl Cos %ang2
%dy2 = OpExtInst %float %glsl Sin %ang2
%off2 = OpCompositeConstruct %v4float %dx2 %dy2 %zero %zero
%e2 = OpFAdd %v4float %in_pos %off2
OpStore %center_pos %e2
OpEmitVertex
OpEndPrimitive

; Triangle 2: center, e2, e3
OpStore %outColor %in_col
OpStore %center_pos %in_pos
OpEmitVertex
OpStore %outColor %edge_col
OpStore %center_pos %e2
OpEmitVertex

%ang3 = OpFMul %float %pi_3 %float_3
%dx3 = OpExtInst %float %glsl Cos %ang3
%dy3 = OpExtInst %float %glsl Sin %ang3
%off3 = OpCompositeConstruct %v4float %dx3 %dy3 %zero %zero
%e3 = OpFAdd %v4float %in_pos %off3
OpStore %center_pos %e3
OpEmitVertex
OpEndPrimitive

; Triangle 3: center, e3, e4
OpStore %outColor %in_col
OpStore %center_pos %in_pos
OpEmitVertex
OpStore %outColor %edge_col
OpStore %center_pos %e3
OpEmitVertex

%ang4 = OpFMul %float %pi_3 %float_4
%dx4 = OpExtInst %float %glsl Cos %ang4
%dy4 = OpExtInst %float %glsl Sin %ang4
%off4 = OpCompositeConstruct %v4float %dx4 %dy4 %zero %zero
%e4 = OpFAdd %v4float %in_pos %off4
OpStore %center_pos %e4
OpEmitVertex
OpEndPrimitive

; Triangle 4: center, e4, e5
OpStore %outColor %in_col
OpStore %center_pos %in_pos
OpEmitVertex
OpStore %outColor %edge_col
OpStore %center_pos %e4
OpEmitVertex

%ang5 = OpFMul %float %pi_3 %float_5
%dx5 = OpExtInst %float %glsl Cos %ang5
%dy5 = OpExtInst %float %glsl Sin %ang5
%off5 = OpCompositeConstruct %v4float %dx5 %dy5 %zero %zero
%e5 = OpFAdd %v4float %in_pos %off5
OpStore %center_pos %e5
OpEmitVertex
OpEndPrimitive

; Triangle 5: center, e5, e0 (close)
OpStore %outColor %in_col
OpStore %center_pos %in_pos
OpEmitVertex
OpStore %outColor %edge_col
OpStore %center_pos %e5
OpEmitVertex
OpStore %center_pos %e0
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
```
