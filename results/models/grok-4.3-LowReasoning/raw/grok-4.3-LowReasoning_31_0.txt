use std::io::{self, Read, Write};
use std::collections::HashMap;

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut lines = input.trim().split('\n');
    let first_line: Vec<i64> = lines.next().unwrap().split_whitespace()
        .map(|s| s.parse().unwrap()).collect();
    let n = first_line[0] as usize;
    let target = first_line[1];
    let nums: Vec<i64> = lines.next().unwrap().split_whitespace()
        .map(|s| s.parse().unwrap()).collect();

    if n != nums.len() {
        println!("NO");
        return;
    }

    // Edge cases
    if target == 0 {
        println!("YES");
        println!();
        return;
    }
    if n == 0 {
        println!("NO");
        return;
    }

    // For large n, only feasible exact if special cases or small effective size
    // Use meet-in-the-middle for n <= 40, else fall back to limited DP or NO for practicality
    if n <= 40 {
        if let Some(indices) = meet_in_middle(&nums, target) {
            println!("YES");
            let idx_str: Vec<String> = indices.iter().map(|i| i.to_string()).collect();
            println!("{}", idx_str.join(" "));
        } else {
            println!("NO");
        }
    } else {
        // Large n: use hash-based DP limited to feasible memory, or detect simple cases
        if let Some(indices) = limited_dp(&nums, target) {
            println!("YES");
            let idx_str: Vec<String> = indices.iter().map(|i| i.to_string()).collect();
            println!("{}", idx_str.join(" "));
        } else {
            println!("NO");
        }
    }
}

fn meet_in_middle(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let mid = n / 2;
    let left = &nums[0..mid];
    let right = &nums[mid..n];

    let mut left_sums: HashMap<i64, Vec<usize>> = HashMap::new();
    left_sums.insert(0, vec![]);
    for (i, &val) in left.iter().enumerate() {
        let mut new_sums = HashMap::new();
        for (&sum, idxs) in &left_sums {
            let new_sum = sum + val;
            let mut new_idxs = idxs.clone();
            new_idxs.push(i);
            new_sums.insert(new_sum, new_idxs);
        }
        for (s, v) in new_sums {
            left_sums.entry(s).or_insert(v);
        }
    }

    let mut right_sums: HashMap<i64, Vec<usize>> = HashMap::new();
    right_sums.insert(0, vec![]);
    for (i, &val) in right.iter().enumerate() {
        let mut new_sums = HashMap::new();
        for (&sum, idxs) in &right_sums {
            let new_sum = sum + val;
            let mut new_idxs = idxs.clone();
            new_idxs.push(mid + i);
            new_sums.insert(new_sum, new_idxs);
        }
        for (s, v) in new_sums {
            right_sums.entry(s).or_insert(v);
        }
    }

    for (&ls, lidxs) in &left_sums {
        if let Some(ridxs) = right_sums.get(&(target - ls)) {
            let mut res = lidxs.clone();
            res.extend(ridxs);
            res.sort();
            return Some(res);
        }
    }
    None
}

fn limited_dp(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    // Simple O(N*target) style but memory safe limited; only positive small target heuristic
    if target < 0 || target > 100000 {
        return None;
    }
    let mut dp: Vec<Option<Vec<usize>>> = vec![None; (target + 1) as usize];
    dp[0] = Some(vec![]);
    for (i, &num) in nums.iter().enumerate() {
        if num < 0 || num > target { continue; }
        for j in (num..=target).rev() {
            if dp[(j - num) as usize].is_some() && dp[j as usize].is_none() {
                let mut path = dp[(j - num) as usize].clone().unwrap();
                path.push(i);
                dp[j as usize] = Some(path);
            }
        }
    }
    dp[target as usize].clone()
}
