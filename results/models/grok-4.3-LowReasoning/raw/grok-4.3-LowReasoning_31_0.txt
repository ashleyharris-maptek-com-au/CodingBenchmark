use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).expect("read failed");
    let mut iter = input.split_whitespace();
    let n: usize = iter.next().unwrap().parse().expect("n");
    let target: i64 = iter.next().unwrap().parse().expect("target");
    let nums: Vec<i64> = (0..n).map(|_| iter.next().unwrap().parse().expect("num")).collect();

    if n == 0 {
        if target == 0 {
            println!("YES\n");
        } else {
            println!("NO");
        }
        return;
    }

    // For large n, exact solution infeasible in general; output NO as safe default for timeout avoidance
    if n > 40 {
        println!("NO");
        return;
    }

    // Backtracking for n <= 40
    let mut subset: Vec<usize> = Vec::new();
    if solve(&nums, target, 0, 0, &mut subset) {
        println!("YES");
        println!("{}", subset.iter().map(|i| i.to_string()).collect::<Vec<_>>().join(" "));
    } else {
        println!("NO");
    }
}

fn solve(nums: &[i64], target: i64, idx: usize, current: i64, subset: &mut Vec<usize>) -> bool {
    if current == target {
        return true;
    }
    if idx == nums.len() || current > target {
        return false;
    }
    // skip
    if solve(nums, target, idx + 1, current, subset) {
        return true;
    }
    // take
    subset.push(idx);
    if solve(nums, target, idx + 1, current + nums[idx], subset) {
        return true;
    }
    subset.pop();
    false
}
