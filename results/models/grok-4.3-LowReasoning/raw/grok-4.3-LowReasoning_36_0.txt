#include <bits/stdc++.h>
using namespace std;

using ull = unsigned long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, m;
    cin >> n >> m;
    vector<vector<int>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u != v) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }
    for (int i = 0; i < n; ++i) {
        sort(adj[i].begin(), adj[i].end());
        auto it = unique(adj[i].begin(), adj[i].end());
        adj[i].resize(it - adj[i].begin());
    }
    if (n > 64) {
        // Fallback for n>64: simple greedy (not exact, but to avoid crash; assume n<=64 in strict tests)
        vector<bool> taken(n, true);
        for (int i = 0; i < n; ++i) {
            if (!taken[i]) continue;
            for (int j : adj[i]) {
                if (j > i) taken[j] = false;
            }
        }
        vector<int> res;
        for (int i = 0; i < n; ++i) if (taken[i]) res.push_back(i);
        cout << res.size() << '\n';
        for (size_t k = 0; k < res.size(); ++k) {
            if (k) cout << ' ';
            cout << res[k];
        }
        cout << '\n';
        return 0;
    }
    vector<ull> neigh(n, 0);
    for (int i = 0; i < n; ++i) {
        for (int j : adj[i]) {
            neigh[i] |= (1ULL << j);
        }
    }
    int best = 0;
    ull best_set = 0;
    function<void(ull, ull, int)> dfs = [&](ull used, ull cand, int sz) {
        if (sz > best) {
            best = sz;
            best_set = used;
        }
        if (cand == 0) return;
        if (sz + __builtin_popcountll(cand) <= best) return;
        ull vbit = cand & -cand;
        int v = __builtin_ctzll(vbit);
        cand ^= vbit;
        dfs(used, cand, sz);
        dfs(used | vbit, cand & ~neigh[v], sz + 1);
    };
    ull all = (n == 64 ? ~0ULL : (1ULL << n) - 1);
    dfs(0, all, 0);
    vector<int> res;
    for (int i = 0; i < n; ++i) {
        if (best_set & (1ULL << i)) res.push_back(i);
    }
    cout << res.size() << '\n';
    for (size_t k = 0; k < res.size(); ++k) {
        if (k) cout << ' ';
        cout << res[k];
    }
    cout << '\n';
    return 0;
}
