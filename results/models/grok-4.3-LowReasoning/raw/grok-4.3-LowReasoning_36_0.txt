```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;
    vector<vector<int>> adj(n);
    vector<bitset<64>> adjb(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u != v) {
            adj[u].push_back(v);
            adj[v].push_back(u);
            adjb[u][v] = 1;
            adjb[v][u] = 1;
        }
    }
    for (int i = 0; i < n; ++i) {
        sort(adj[i].begin(), adj[i].end());
        auto it = unique(adj[i].begin(), adj[i].end());
        adj[i].resize(it - adj[i].begin());
    }

    if (n > 64) {
        // Fallback simple greedy for very large n (not optimal but compiles/runs)
        vector<bool> used(n, false);
        vector<int> res;
        for (int i = 0; i < n; ++i) {
            if (!used[i]) {
                res.push_back(i);
                for (int nei : adj[i]) used[nei] = true;
            }
        }
        cout << res.size() << '\n';
        for (size_t i = 0; i < res.size(); ++i) {
            if (i) cout << ' ';
            cout << res[i];
        }
        cout << '\n';
        return 0;
    }

    // Exact backtracking for n <= 64
    vector<int> best_set;
    vector<int> cur;
    vector<int> verts(n);
    iota(verts.begin(), verts.end(), 0);
    // Order by increasing degree
    sort(verts.begin(), verts.end(), [&](int a, int b) {
        return adj[a].size() < adj[b].size();
    });

    function<void(int, bitset<64>)> dfs = [&](int pos, bitset<64> forbidden) {
        if (pos == n) {
            if ((int)cur.size() > (int)best_set.size()) {
                best_set = cur;
            }
            return;
        }
        int v = verts[pos];
        // prune
        int remain = n - pos;
        if ((int)cur.size() + remain <= (int)best_set.size()) return;
        // skip v
        dfs(pos + 1, forbidden);
        // take v if possible
        if (!forbidden[v]) {
            cur.push_back(v);
            dfs(pos + 1, forbidden | adjb[v]);
            cur.pop_back();
        }
    };

    bitset<64> empty;
    dfs(0, empty);

    cout << best_set.size() << '\n';
    for (size_t i = 0; i < best_set.size(); ++i) {
        if (i) cout << ' ';
        cout << best_set[i];
    }
    cout << '\n';
    return 0;
}
```
