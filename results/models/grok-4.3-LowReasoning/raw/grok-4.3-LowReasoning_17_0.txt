```csharp
using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        var first = Console.ReadLine().Split();
        int N = int.Parse(first[0]);
        long W = long.Parse(first[1]);
        long H = long.Parse(first[2]);
        long D = long.Parse(first[3]);

        var boxes = new List<(int idx, long[] dims)>();
        for (int i = 0; i < N; i++)
        {
            var p = Console.ReadLine().Split().Select(long.Parse).ToArray();
            boxes.Add((i, p));
        }

        boxes = boxes.OrderByDescending(b => b.dims[0] * b.dims[1] * b.dims[2]).ToList();

        var placed = new List<(int idx, long x, long y, long z)>();
        var occupied = new List<(long x, long y, long z, long w, long h, long d)>();

        foreach (var box in boxes)
        {
            var dims = box.dims;
            bool found = false;
            long bestX = -1, bestY = -1, bestZ = -1, bestW = 0, bestH = 0, bestD = 0;

            for (int r = 0; r < 6; r++)
            {
                long w = dims[r % 3];
                long h = dims[(r + 1) % 3];
                long d = dims[(r + 2) % 3];
                if (w > W || h > H || d > D) continue;

                for (long x = 0; x <= W - w; x += Math.Max(1, (W - w) / 50 + 1))
                {
                    for (long y = 0; y <= H - h; y += Math.Max(1, (H - h) / 50 + 1))
                    {
                        for (long z = 0; z <= D - d; z += Math.Max(1, (D - d) / 50 + 1))
                        {
                            if (Fits(x, y, z, w, h, d, occupied))
                            {
                                bestX = x; bestY = y; bestZ = z;
                                bestW = w; bestH = h; bestD = d;
                                found = true;
                                goto nextBox;
                            }
                        }
                    }
                }
            }
        nextBox:
            if (found)
            {
                placed.Add((box.idx, bestX, bestY, bestZ));
                occupied.Add((bestX, bestY, bestZ, bestW, bestH, bestD));
            }
        }

        Console.WriteLine(placed.Count);
        foreach (var p in placed)
        {
            Console.WriteLine($"{p.idx} {p.x} {p.y} {p.z}");
        }
    }

    static bool Fits(long x, long y, long z, long w, long h, long d, List<(long x, long y, long z, long w, long h, long d)> occ)
    {
        foreach (var o in occ)
        {
            if (!(x + w <= o.x || x >= o.x + o.w ||
                  y + h <= o.y || y >= o.y + o.h ||
                  z + d <= o.z || z >= o.z + o.d))
                return false;
        }
        return true;
    }
}
```
