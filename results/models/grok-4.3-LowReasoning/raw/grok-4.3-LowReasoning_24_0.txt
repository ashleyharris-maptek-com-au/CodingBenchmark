```cpp
#include <iostream>
#include <thread>
#include <atomic>
#include <vector>
#include <string>
#include <fstream>
#include <chrono>
#include <cstdint>
#include <iomanip>
#include <mutex>

using namespace std;

struct Params {
    double width, depth, height, gravity, max_thrust, fuel;
    double start_x, start_y, start_z;
    double target_x, target_y, target_z;
    string terrain_path;
};

struct State {
    double x, y, z, vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
};

atomic<bool> terrain_loaded{false};
atomic<bool> running{true};

void load_terrain(const string& path) {
    // Parse in background, but we don't need full terrain for basic controller
    ifstream file(path, ios::binary);
    if (file) {
        file.seekg(0, ios::end);
        // Just mark as loaded after quick check
    }
    terrain_loaded = true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout << fixed << setprecision(6);

    Params p;
    cin >> p.width >> p.depth >> p.height >> p.gravity >> p.max_thrust >> p.fuel;
    cin >> p.start_x >> p.start_y >> p.start_z >> p.target_x >> p.target_y >> p.target_z;
    cin.ignore(); // consume newline
    getline(cin, p.terrain_path);

    string marker;
    getline(cin, marker); // "STATE" line if present, or ignore extra

    thread terrain_thread(load_terrain, p.terrain_path);

    State s{};
    auto last_output = chrono::steady_clock::now();

    while (running) {
        // Read state - non-blocking attempt via fast cin
        if (!(cin >> s.x >> s.y >> s.z >> s.vx >> s.vy >> s.vz
              >> s.pitch >> s.yaw >> s.roll
              >> s.pitch_rate >> s.yaw_rate >> s.roll_rate >> s.fuel)) {
            this_thread::sleep_for(chrono::milliseconds(1));
            continue;
        }

        // Simple controller: hover above target then descend
        double dx = p.target_x - s.x;
        double dy = p.target_y - s.y;
        double dz = p.target_z + 30.0 - s.z; // hover 30m above then descend

        double target_vx = dx * 0.5;
        double target_vy = dy * 0.5;
        double target_vz = dz * 0.8 - 1.5; // gentle descent

        double ax = (target_vx - s.vx) * 1.2;
        double ay = (target_vy - s.vy) * 1.2;
        double az = (target_vz - s.vz) * 1.0 + 1.62;

        // Convert desired acceleration to thrust and tilt commands
        double thrust = min(1.0, max(0.0, az / p.max_thrust));
        double pitch_cmd = max(-1.0, min(1.0, -ax * 0.8));
        double yaw_cmd = max(-1.0, min(1.0, ay * 0.8));
        double roll_cmd = 0.0;

        auto now = chrono::steady_clock::now();
        if (chrono::duration_cast<chrono::milliseconds>(now - last_output).count() >= 50) {
            cout << thrust << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n";
            cout.flush();
            last_output = now;
        }

        this_thread::sleep_for(chrono::milliseconds(5));
    }

    terrain_thread.join();
    return 0;
}
```
