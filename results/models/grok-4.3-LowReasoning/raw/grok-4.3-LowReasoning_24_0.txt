#include <iostream>
#include <fstream>
#include <string>
#include <thread>
#include <atomic>
#include <chrono>
#include <vector>
#include <cstdint>
#include <mutex>
#include <algorithm>
#include <cmath>

using namespace std;

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
    bool valid = false;
};

atomic<bool> running{true};
mutex state_mutex;
State current_state;

void io_thread_func() {
    string line;
    while (running && getline(cin, line)) {
        if (line.empty()) continue;
        State s;
        int n = sscanf(line.c_str(), "%lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf",
                       &s.x, &s.y, &s.z, &s.vx, &s.vy, &s.vz,
                       &s.pitch, &s.yaw, &s.roll,
                       &s.pitch_rate, &s.yaw_rate, &s.roll_rate, &s.fuel);
        if (n == 13) {
            lock_guard<mutex> lock(state_mutex);
            current_state = s;
            current_state.valid = true;
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    cout.setf(ios::fixed);
    cout.precision(6);

    double width, depth, height, gravity, max_thrust, fuel;
    double start_x, start_y, start_z, target_x, target_y, target_z;
    string terrain_path;

    cin >> width >> depth >> height >> gravity >> max_thrust >> fuel;
    cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');
    getline(cin, terrain_path);

    // Skip "STATE" line if present
    string marker;
    getline(cin, marker);

    // Start IO thread immediately
    thread io_thread(io_thread_func);

    // Optional: terrain parsing thread (background)
    thread terrain_thread([terrain_path, width, depth]() {
        ifstream terrain_file(terrain_path, ios::binary);
        if (!terrain_file) return;
        size_t total_cells = static_cast<size_t>(width) * static_cast<size_t>(depth);
        vector<uint16_t> terrain_data(total_cells * 3);
        terrain_file.read(reinterpret_cast<char*>(terrain_data.data()), total_cells * 6);
    });
    terrain_thread.detach();

    // Simple controller loop
    const auto tick = chrono::milliseconds(50); // 20 Hz output
    while (running) {
        State s;
        {
            lock_guard<mutex> lock(state_mutex);
            s = current_state;
        }

        if (!s.valid) {
            this_thread::sleep_for(tick);
            continue;
        }

        double dx = target_x - s.x;
        double dy = target_y - s.y;
        double dz = target_z - s.z;

        double dist_xy = sqrt(dx*dx + dy*dy);
        double desired_vz = -1.5; // gentle descent
        if (dist_xy > 20) desired_vz = -0.8;

        double vz_error = desired_vz - s.vz;
        double thrust = 0.55 + vz_error * 0.25;
        thrust = clamp(thrust, 0.0, 1.0);

        // Simple attitude commands to point roughly toward target
        double desired_pitch = clamp(-dx * 0.008, -0.8, 0.8);
        double desired_yaw = clamp(dy * 0.008, -0.8, 0.8);

        double pitch_cmd = clamp((desired_pitch - s.pitch) * 3.0, -1.0, 1.0);
        double yaw_cmd = clamp((desired_yaw - s.yaw) * 3.0, -1.0, 1.0);
        double roll_cmd = 0.0;

        cout << thrust << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n";
        cout.flush();

        this_thread::sleep_for(tick);
    }

    running = false;
    io_thread.join();
    return 0;
}
