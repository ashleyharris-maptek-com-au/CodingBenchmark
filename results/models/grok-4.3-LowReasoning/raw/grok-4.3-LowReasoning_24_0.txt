#include <iostream>
#include <thread>
#include <atomic>
#include <string>
#include <vector>
#include <chrono>
#include <cstdint>
#include <fstream>
#include <mutex>
#include <condition_variable>
#include <cmath>
#include <iomanip>

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
    bool valid = false;
};

std::atomic<bool> running{true};
std::mutex state_mutex;
State current_state;
std::condition_variable state_cv;

void read_loop() {
    std::string line;
    while (running) {
        if (!std::getline(std::cin, line)) {
            running = false;
            break;
        }
        if (line.empty()) continue;
        
        State new_state;
        int count = sscanf(line.c_str(), "%lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf",
                          &new_state.x, &new_state.y, &new_state.z,
                          &new_state.vx, &new_state.vy, &new_state.vz,
                          &new_state.pitch, &new_state.yaw, &new_state.roll,
                          &new_state.pitch_rate, &new_state.yaw_rate, &new_state.roll_rate,
                          &new_state.fuel);
        if (count == 13) {
            std::lock_guard<std::mutex> lock(state_mutex);
            current_state = new_state;
            current_state.valid = true;
            state_cv.notify_one();
        }
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    std::cout << std::fixed << std::setprecision(6);
    
    double width, depth, height, gravity, max_thrust, fuel;
    double start_x, start_y, start_z, target_x, target_y, target_z;
    
    std::cin >> width >> depth >> height >> gravity >> max_thrust >> fuel;
    std::cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z;
    
    std::string dummy;
    std::getline(std::cin, dummy); // consume rest of line
    std::string terrain_path;
    std::getline(std::cin, terrain_path);
    
    // Background terrain parsing thread (placeholder, minimal parse)
    std::thread terrain_thread([terrain_path, width, depth]() {
        if (terrain_path.empty()) return;
        std::ifstream file(terrain_path, std::ios::binary);
        if (!file) return;
        // Skip full load for huge maps, just seek to end
        file.seekg(0, std::ios::end);
    });
    terrain_thread.detach();
    
    std::thread reader(read_loop);
    
    // Simple landing controller
    auto last_time = std::chrono::steady_clock::now();
    while (running) {
        State s;
        {
            std::unique_lock<std::mutex> lock(state_mutex);
            state_cv.wait_for(lock, std::chrono::milliseconds(50), []{ return current_state.valid; });
            if (!current_state.valid) continue;
            s = current_state;
        }
        
        double dx = target_x - s.x;
        double dy = target_y - s.y;
        double dz = target_z - s.z;
        
        double dist_xy = std::sqrt(dx*dx + dy*dy);
        double desired_vz = -1.5; // gentle descent
        
        if (dist_xy > 10.0) {
            desired_vz = -0.5;
        }
        
        double vz_error = desired_vz - s.vz;
        double thrust = 0.3 + 0.4 * vz_error;
        thrust = std::max(0.0, std::min(1.0, thrust));
        
        // Simple attitude control to point thrust toward target
        double pitch_cmd = 0.0;
        double yaw_cmd = 0.0;
        
        if (dist_xy > 2.0) {
            double desired_pitch = std::atan2(dx, std::max(1.0, std::abs(dz))) * 0.5;
            pitch_cmd = (desired_pitch - s.pitch) * 2.0;
            yaw_cmd = (std::atan2(dy, std::max(1.0, std::abs(dz))) - s.yaw) * 1.5;
        }
        
        pitch_cmd = std::max(-1.0, std::min(1.0, pitch_cmd));
        yaw_cmd = std::max(-1.0, std::min(1.0, yaw_cmd));
        double roll_cmd = 0.0;
        
        std::cout << thrust << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n";
        std::cout.flush();
        
        // Throttle output rate
        auto now = std::chrono::steady_clock::now();
        auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(now - last_time);
        if (elapsed.count() < 80) {
            std::this_thread::sleep_for(std::chrono::milliseconds(80 - elapsed.count()));
        }
        last_time = std::chrono::steady_clock::now();
        
        if (s.z < target_z + 2.0 && std::abs(s.vz) < 2.0 && dist_xy < 20.0) {
            // Final landing phase
            thrust = 0.6;
            std::cout << thrust << " 0 0 0\n";
            std::cout.flush();
        }
    }
    
    running = false;
    reader.join();
    return 0;
}
