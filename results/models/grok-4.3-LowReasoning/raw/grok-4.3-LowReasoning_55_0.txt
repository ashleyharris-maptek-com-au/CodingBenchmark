#include <iostream>
#include <string>
#include <chrono>
#include <thread>

int main() {
    std::string timeout_line;
    std::string fen;
    std::getline(std::cin, timeout_line);
    std::getline(std::cin, fen);
    
    double timeout = 5.0;
    try {
        timeout = std::stod(timeout_line);
    } catch (...) {}
    
    auto start = std::chrono::steady_clock::now();
    auto end_time = start + std::chrono::duration<double>(timeout - 0.2);
    
    // Simple fixed opening move for starting position or similar
    // In real engine this would be replaced by search
    std::string best_move = "pe2e4";
    
    while (std::chrono::steady_clock::now() < end_time) {
        std::this_thread::sleep_for(std::chrono::milliseconds(10));
    }
    
    std::cout << best_move << std::endl;
    return 0;
}
