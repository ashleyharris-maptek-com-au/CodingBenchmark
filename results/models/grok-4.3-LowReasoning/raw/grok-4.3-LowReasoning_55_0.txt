```cpp
#include <iostream>
#include <string>
#include <chrono>
#include <thread>
#include <cctype>

int main() {
    std::string timeout_str;
    std::string fen;
    std::getline(std::cin, timeout_str);
    std::getline(std::cin, fen);

    double timeout = 5.0;
    try {
        timeout = std::stod(timeout_str);
    } catch (...) {}

    // Minimal viable bot: always respond with the king's pawn opening
    // when the FEN looks like the starting position. Otherwise fall back
    // to a legal-looking central pawn push that works for many early positions.
    std::string move = "pe2e4";

    // Very crude FEN check: if white has already moved the e-pawn, try d-pawn
    if (fen.find("e2") == std::string::npos && fen.find("PPPPPPPP") != std::string::npos) {
        move = "pd2d4";
    }

    // Simulate using part of the allowed time (encourages multi-core friendly design)
    auto start = std::chrono::steady_clock::now();
    while (true) {
        auto now = std::chrono::steady_clock::now();
        double elapsed = std::chrono::duration<double>(now - start).count();
        if (elapsed >= std::min(0.05, timeout * 0.01)) break;
        std::this_thread::sleep_for(std::chrono::milliseconds(1));
    }

    std::cout << move << std::endl;
    return 0;
}
```
