; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %normal %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
%void = OpTypeVoid
%3 = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%_ptr_Input_v3float = OpTypePointer Input %v3float
%normal = OpVariable %_ptr_Input_v3float Input
%_ptr_Output_v4float = OpTypePointer Output %v4float
%outColor = OpVariable %_ptr_Output_v4float Output
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1
%19 = OpConstantComposite %v3float %float_1 %float_1 %float_1
%20 = OpConstantComposite %v3float %float_0_5 %float_0_5 %float_0_5
%main = OpFunction %void None %3
%5 = OpLabel
%12 = OpLoad %v3float %normal
%13 = OpNormalize %v3float %12
%21 = OpFAdd %v3float %13 %19
%22 = OpFMul %v3float %21 %20
%nx = OpCompositeExtract %float %22 0
%ny = OpCompositeExtract %float %22 1
%nz = OpCompositeExtract %float %22 2
%24 = OpCompositeConstruct %v4float %nx %ny %nz %float_1
OpStore %outColor %24
OpReturn
OpFunctionEnd
