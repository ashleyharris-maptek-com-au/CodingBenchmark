; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%_ptr_Input_vec3 = OpTypePointer Input %vec3
%in_worldPos = OpVariable %_ptr_Input_vec3 Input
%in_normal = OpVariable %_ptr_Input_vec3 Input
%in_uv = OpVariable %_ptr_Input_vec3 Input
%in_tangent = OpVariable %_ptr_Input_vec3 Input
%in_color = OpVariable %_ptr_Input_vec3 Input
%_ptr_Output_vec4 = OpTypePointer Output %vec4
%outColor = OpVariable %_ptr_Output_vec4 Output
%_ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_32 = OpConstant %float 32
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%main = OpFunction %void None
%entry = OpLabel
%lightPos = OpAccessChain %_ptr_Uniform_vec4 %ubo %int_3
%lightPosVal = OpLoad %vec4 %lightPos
%cameraPos = OpAccessChain %_ptr_Uniform_vec4 %ubo %int_4
%cameraPosVal = OpLoad %vec4 %cameraPos
%N_in = OpLoad %vec3 %in_normal
%N = OpNormalize %vec3 %N_in
%W = OpLoad %vec3 %in_worldPos
%L_sub = OpVectorShuffle %vec3 %lightPosVal %lightPosVal 0 1 2
%L_vec = OpSub %vec3 %L_sub %W
%L = OpNormalize %vec3 %L_vec
%V_sub = OpVectorShuffle %vec3 %cameraPosVal %cameraPosVal 0 1 2
%V_vec = OpSub %vec3 %V_sub %W
%V = OpNormalize %vec3 %V_vec
%L_neg = OpFNegate %vec3 %L
%R = OpReflect %vec3 %L_neg %N
%dotNL = OpDot %float %N %L
%diff = OpExtInst %float %glsl_max %dotNL %float_0
%dotRV = OpDot %float %R %V
%spec_base = OpExtInst %float %glsl_max %dotRV %float_0
%spec = OpExtInst %float %glsl_pow %spec_base %float_32
%term1 = OpFMul %float %float_0_5 %diff
%term2 = OpFMul %float %float_0_8 %spec
%sum = OpFAdd %float %float_0_2 %term1
%intensity = OpFAdd %float %sum %term2
%outVec = OpCompositeConstruct %vec4 %intensity %intensity %intensity %float_1
OpStore %outColor %outVec
OpReturn
OpFunctionEnd
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%glsl_max = OpExtInstImport "GLSL.std.450"
%glsl_pow = OpExtInstImport "GLSL.std.450"
