```glsl
#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 physics;
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) return;

    vec4 p0 = inData[i * 2];
    vec3 pos = p0.xyz;
    float mass = p0.w;
    vec3 vel = inData[i * 2 + 1].xyz;

    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;
    float soft2 = softening * softening;

    vec3 acc = vec3(0.0);
    for (uint j = 0; j < N; ++j) {
        if (i == j) continue;
        vec4 pj = inData[j * 2];
        vec3 diff = pj.xyz - pos;
        float dist2 = dot(diff, diff) + soft2;
        float invDist3 = inversesqrt(dist2) * inversesqrt(dist2) * inversesqrt(dist2);
        acc += G * pj.w * diff * invDist3;
    }

    vel += acc * dt;
    pos += vel * dt;

    outData[i * 2] = vec4(pos, mass);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
```
