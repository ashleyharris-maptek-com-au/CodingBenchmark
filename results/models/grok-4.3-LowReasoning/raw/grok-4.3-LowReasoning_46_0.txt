```glsl
#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 physics;
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) return;

    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    vec4 posMass = inData[i * 2];
    vec3 pos = posMass.xyz;
    float mass = posMass.w;

    vec4 velData = inData[i * 2 + 1];
    vec3 vel = velData.xyz;

    vec3 acc = vec3(0.0);

    for (uint j = 0; j < N; ++j) {
        if (i == j) continue;
        vec4 jPosMass = inData[j * 2];
        vec3 jPos = jPosMass.xyz;
        float jMass = jPosMass.w;

        vec3 diff = jPos - pos;
        float dist2 = dot(diff, diff) + softening * softening;
        float invDist = inversesqrt(dist2);
        float invDist3 = invDist * invDist * invDist;

        acc += G * jMass * diff * invDist3;
    }

    vel += acc * dt;
    pos += vel * dt;

    outData[i * 2] = vec4(pos, mass);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
```
