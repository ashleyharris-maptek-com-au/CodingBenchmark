```typescript
declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const prob: number[][] = data.probability;
const drones: { id: number; x: number; y: number }[] = data.drones;
const maxSteps: number = data.maxStepsPerDrone;

const rows = prob.length;
const cols = prob[0].length;

const dirs = [[0, 0], [0, 1], [0, -1], [1, 0], [-1, 0]];

function isValid(x: number, y: number): boolean {
  return x >= 0 && x < cols && y >= 0 && y < rows && prob[y][x] !== -1;
}

const covered = Array.from({ length: rows }, () => Array(cols).fill(false));
const paths: number[][][] = [];

for (const drone of drones) {
  let cx = drone.x;
  let cy = drone.y;
  const path: number[][] = [[cx, cy]];
  if (isValid(cx, cy)) {
    covered[cy][cx] = true;
  }

  for (let step = 0; step < maxSteps; step++) {
    let bestScore = -1;
    let nx = cx;
    let ny = cy;

    for (const [dx, dy] of dirs) {
      const tx = cx + dx;
      const ty = cy + dy;
      if (isValid(tx, ty)) {
        const score = covered[ty][tx] ? 0 : prob[ty][tx];
        if (score > bestScore) {
          bestScore = score;
          nx = tx;
          ny = ty;
        }
      }
    }

    path.push([nx, ny]);
    if (isValid(nx, ny) && !covered[ny][nx]) {
      covered[ny][nx] = true;
    }
    cx = nx;
    cy = ny;
  }

  paths.push(path);
}

const output = { paths };
console.log(JSON.stringify(output));
```
