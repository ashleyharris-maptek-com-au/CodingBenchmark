```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.IO;
using System.Text;

class Program
{
    static void Main()
    {
        var input = Console.In;
        string[] first = input.ReadLine().Split();
        int N = int.Parse(first[0]);
        int M = int.Parse(first[1]);

        var adj = new List<int>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<int>();

        for (int i = 0; i < M; i++)
        {
            string[] parts = input.ReadLine().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            adj[u].Add(v);
            adj[v].Add(u);
        }

        // Duplicate all edges to make Eulerian
        var adjEuler = new List<int>[N];
        for (int i = 0; i < N; i++)
        {
            adjEuler[i] = new List<int>(adj[i].Count * 2);
            adjEuler[i].AddRange(adj[i]);
            adjEuler[i].AddRange(adj[i]);
        }

        // Hierholzer's algorithm
        var stack = new Stack<int>();
        var circuit = new List<int>();
        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            if (adjEuler[u].Count > 0)
            {
                int v = adjEuler[u][adjEuler[u].Count - 1];
                adjEuler[u].RemoveAt(adjEuler[u].Count - 1);
                // Remove reverse edge
                for (int j = 0; j < adjEuler[v].Count; j++)
                {
                    if (adjEuler[v][j] == u)
                    {
                        adjEuler[v].RemoveAt(j);
                        break;
                    }
                }
                stack.Push(v);
            }
            else
            {
                circuit.Add(stack.Pop());
            }
        }

        // Reverse to get correct order
        circuit.Reverse();

        // Output
        var sb = new StringBuilder();
        foreach (int node in circuit)
        {
            sb.Append(node).Append(' ');
        }
        Console.WriteLine(sb.ToString().Trim());
    }
}
```
