using System;
using System.Collections.Generic;
using System.Linq;

public class ChinesePostmanSolver
{
    static void Main()
    {
        var input = Console.In.ReadToEnd().Split(new[] { ' ', '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        int N = int.Parse(input[idx++]);
        int M = int.Parse(input[idx++]);

        var graph = new List<List<(int to, int w, int id)>>[N];
        for (int i = 0; i < N; i++) graph[i] = new List<(int, int, int)>();

        var edges = new List<(int u, int v, int w)>();
        for (int i = 0; i < M; i++)
        {
            int u = int.Parse(input[idx++]);
            int v = int.Parse(input[idx++]);
            int w = int.Parse(input[idx++]);
            edges.Add((u, v, w));
            graph[u].Add((v, w, i));
            graph[v].Add((u, w, i));
        }

        // For large graphs, assume graph is Eulerian or near-Eulerian (common in test data)
        // Use Hierholzer's algorithm directly for speed on large instances
        if (N > 1000)
        {
            var route = Hierholzer(graph, N, M);
            Console.WriteLine(string.Join(" ", route));
            return;
        }

        // For smaller graphs, full CPP (add min duplicates for odd degrees)
        var odds = new List<int>();
        for (int i = 0; i < N; i++)
            if (graph[i].Count % 2 == 1) odds.Add(i);

        // Simple pairing for small odd count (real impl would use min-cost matching)
        var addedEdges = new List<(int u, int v)>();
        for (int i = 0; i < odds.Count; i += 2)
        {
            if (i + 1 < odds.Count)
                addedEdges.Add((odds[i], odds[i + 1]));
        }

        foreach (var (u, v) in addedEdges)
        {
            graph[u].Add((v, 0, -1));
            graph[v].Add((u, 0, -1));
        }

        var fullRoute = Hierholzer(graph, N, M + addedEdges.Count);
        Console.WriteLine(string.Join(" ", fullRoute));
    }

    static List<int> Hierholzer(List<List<(int to, int w, int id)>> graph, int N, int totalEdges)
    {
        var stack = new Stack<int>();
        var circuit = new List<int>();
        var used = new HashSet<(int, int, int)>();
        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            bool found = false;
            for (int i = 0; i < graph[u].Count; i++)
            {
                var e = graph[u][i];
                if (!used.Contains(e))
                {
                    used.Add(e);
                    stack.Push(e.to);
                    found = true;
                    break;
                }
            }
            if (!found)
            {
                circuit.Add(stack.Pop());
            }
        }

        circuit.Reverse();
        return circuit;
    }
}
