#include <iostream>
#include <vector>
#include <string>
#include <sstream>
#include <cmath>
#include <algorithm>
#include <limits>

struct Point3D {
    double x, y, z;
};

struct Tetrahedron {
    Point3D center;
    double rx, ry, rz; // Euler angles
};

bool isPointInsidePolyhedron(const Point3D& p, const std::vector<Point3D>& vertices) {
    // Simplified: bounding box check (for demo; real impl needs full polyhedron test)
    if (vertices.empty()) return false;
    double minX = vertices[0].x, maxX = vertices[0].x;
    double minY = vertices[0].y, maxY = vertices[0].y;
    double minZ = vertices[0].z, maxZ = vertices[0].z;
    for (const auto& v : vertices) {
        minX = std::min(minX, v.x);
        maxX = std::max(maxX, v.x);
        minY = std::min(minY, v.y);
        maxY = std::max(maxY, v.y);
        minZ = std::min(minZ, v.z);
        maxZ = std::max(maxZ, v.z);
    }
    return p.x >= minX && p.x <= maxX &&
           p.y >= minY && p.y <= maxY &&
           p.z >= minZ && p.z <= maxZ;
}

int main() {
    std::string line;
    std::getline(std::cin, line);
    std::istringstream iss(line);
    int V;
    double edge_length;
    iss >> V >> edge_length;
    
    std::vector<Point3D> vertices(V);
    for (int i = 0; i < V; ++i) {
        std::getline(std::cin, line);
        std::istringstream vss(line);
        vss >> vertices[i].x >> vertices[i].y >> vertices[i].z;
    }
    
    // Compute simple grid spacing for non-overlapping tetras (edge=1 assumed normalized)
    double spacing = edge_length * 1.1; // conservative spacing
    double circumR = edge_length * std::sqrt(3.0 / 8.0);
    
    std::vector<Tetrahedron> placed;
    
    // Adaptive grid based on bounding box (scalable approx for large volumes)
    if (!vertices.empty()) {
        double minX = vertices[0].x, maxX = vertices[0].x;
        double minY = vertices[0].y, maxY = vertices[0].y;
        double minZ = vertices[0].z, maxZ = vertices[0].z;
        for (const auto& v : vertices) {
            minX = std::min(minX, v.x); maxX = std::max(maxX, v.x);
            minY = std::min(minY, v.y); maxY = std::max(maxY, v.y);
            minZ = std::min(minZ, v.z); maxZ = std::max(maxZ, v.z);
        }
        
        // Sample points on a coarse lattice (performance for large scale)
        for (double x = minX + circumR; x <= maxX - circumR; x += spacing) {
            for (double y = minY + circumR; y <= maxY - circumR; y += spacing) {
                for (double z = minZ + circumR; z <= maxZ - circumR; z += spacing) {
                    Point3D c = {x, y, z};
                    if (isPointInsidePolyhedron(c, vertices)) {
                        placed.push_back({c, 0.0, 0.0, 0.0});
                    }
                }
            }
        }
    }
    
    // Output
    std::cout << placed.size() << std::endl;
    for (const auto& t : placed) {
        std::cout << t.center.x << " " << t.center.y << " " << t.center.z << " "
                  << t.rx << " " << t.ry << " " << t.rz << std::endl;
    }
    
    return 0;
}
