; Version: 1.0
; Generator: manual

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor %ubo
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %gl_PerVertex_out Block

OpDecorate %worldPos Location 0

OpDecorate %color Location 2
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%func = OpTypeFunction %void
%float = OpTypeFloat 32

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_2

%normal = OpVariable %ptr_in_v3 Input
%color = OpVariable %ptr_in_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out

%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; UBO (optional but declared)
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v3float

%main = OpFunction %void None %func
%entry = OpLabel

; Load input line endpoints
%ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_p0
%ptr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr_p1

; Perpendicular in clip space (xy plane)
%dir = OpVectorShuffle %v4float %p1 %p1 0 1 2 3
%delta = OpFSub %v4float %dir %p0
%dx = OpCompositeExtract %float %delta 0
%dy = OpCompositeExtract %float %delta 1
%neg_dy = OpFNegate %float %dy
%perp = OpCompositeConstruct %v4float %dy %neg_dy %float_0 %float_0
%len = OpExtInst %float %glsl_sqrt (OpFMul %float %dx %dx) (OpFMul %float %dy %dy)
%inv_len = OpFDiv %float %float_1 %len
%perp_n = OpVectorTimesScalar %v4float %perp %inv_len

; Thick offsets
%off0 = OpVectorTimesScalar %v4float %perp_n %thick_start
%off1 = OpVectorTimesScalar %v4float %perp_n %thick_end

; Load colors
%ptr_c0 = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %ptr_c0
%ptr_c1 = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %ptr_c1

; Emit vertex 0: p0 + off0, color c0
%v0_pos = OpFAdd %v4float %p0 %off0
%ptr_out0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out0 %v0_pos
%ptr_col0 = OpAccessChain %ptr_out_v3 %outColor
OpStore %ptr_col0 %c0
OpEmitVertex

; Emit vertex 1: p0 - off0, color c0
%neg_off0 = OpFNegate %v4float %off0
%v1_pos = OpFAdd %v4float %p0 %neg_off0
%ptr_out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out1 %v1_pos
%ptr_col1 = OpAccessChain %ptr_out_v3 %outColor
OpStore %ptr_col1 %c0

; Emit vertex 2: p1 + off1, color c1
%v2_pos = OpFAdd %v4float %p1 %off1
%ptr_out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out2 %v2_pos
%ptr_col2 = OpAccessChain %ptr_out_v3 %outColor

; Emit vertex 3: p1 - off1, color c1
%neg_off1 = OpFNegate %v4float %off1

%ptr_out3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out3 %v3_pos
%ptr_col3 = OpAccessChain %ptr_out_v3 %outColor

OpEndPrimitive
OpReturn
