@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>             params   : vec4<u32>;

fn rotl(x: u32, shift: u32) -> u32 {
    return (x << shift) | (x >> (32u - shift));
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    if (gid.x >= params.z) {
        return;
    }

    let nonce = params.y + gid.x;

    var state : array<u32, 16>;
    for (var i: u32 = 0u; i < 16u; i = i + 1u) {
        state[i] = baseData[i] ^ (nonce + i);
    }

    for (var round: u32 = 0u; round < 8u; round = round + 1u) {
        // group 0-3
        {
            let k = 0u;
            var a = state[k + 0u];
            var b = state[k + 1u];
            var c = state[k + 2u];
            var d = state[k + 3u];
            a += b; d ^= a; d = rotl(d, 16u);
            c += d; b ^= c; b = rotl(b, 12u);
            a += b; d ^= a; d = rotl(d, 8u);
            c += d; b ^= c; b = rotl(b, 7u);
            state[k + 0u] = a;
            state[k + 1u] = b;
            state[k + 2u] = c;
            state[k + 3u] = d;
        }
        // group 4-7
        {
            let k = 4u;
            var a = state[k + 0u];
            var b = state[k + 1u];
            var c = state[k + 2u];
            var d = state[k + 3u];
            a += b; d ^= a; d = rotl(d, 16u);
            c += d; b ^= c; b = rotl(b, 12u);
            a += b; d ^= a; d = rotl(d, 8u);
            c += d; b ^= c; b = rotl(b, 7u);
            state[k + 0u] = a;
            state[k + 1u] = b;
            state[k + 2u] = c;
            state[k + 3u] = d;
        }
        // group 8-11
        {
            let k = 8u;
            var a = state[k + 0u];
            var b = state[k + 1u];
            var c = state[k + 2u];
            var d = state[k + 3u];
            a += b; d ^= a; d = rotl(d, 16u);
            c += d; b ^= c; b = rotl(b, 12u);
            a += b; d ^= a; d = rotl(d, 8u);

            state[k + 0u] = a;
            state[k + 1u] = b;
            state[k + 2u] = c;
            state[k + 3u] = d;
        }
        // group 12-15
        {
            let k = 12u;
            var a = state[k + 0u];
            var b = state[k + 1u];
            var c = state[k + 2u];
            var d = state[k + 3u];
            a += b; d ^= a; d = rotl(d, 16u);
            c += d; b ^= c; b = rotl(b, 12u);
            a += b; d ^= a; d = rotl(d, 8u);
            c += d; b ^= c; b = rotl(b, 7u);
            state[k + 0u] = a;
            state[k + 1u] = b;
            state[k + 2u] = c;
            state[k + 3u] = d;
        }
    }

    let h0 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
    let h1 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];

    let tz = select(countTrailingZeros(h0), 32u + countTrailingZeros(h1), h0 == 0u);

    if (tz >= params.x) {
        result[0] = 1u;
        result[1] = nonce;
        result[2] = h0;
        result[3] = h1;
    }
}
