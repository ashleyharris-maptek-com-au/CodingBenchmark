@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>             params   : vec4<u32>;

fn rotl(x: u32, shift: u32) -> u32 {
    return (x << shift) | (x >> (32u - shift));
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    if (gid.x >= params.z) {
        return;
    }

    let nonce = params.y + gid.x;
    var state: array<u32, 16>;

    for (var i = 0u; i < 16u; i = i + 1u) {
        state[i] = baseData[i] ^ (nonce + i);
    }

    for (var round = 0u; round < 8u; round = round + 1u) {
        // group 0-3
        {
            let i = 0u;
            var a = state[i + 0u];
            var b = state[i + 1u];
            var c = state[i + 2u];
            var d = state[i + 3u];
            a = a + b; d = d ^ a; d = rotl(d, 16u);
            c = c + d; b = b ^ c; b = rotl(b, 12u);
            a = a + b; d = d ^ a; d = rotl(d, 8u);
            c = c + d; b = b ^ c; b = rotl(b, 7u);
            state[i + 0u] = a;
            state[i + 1u] = b;
            state[i + 2u] = c;
            state[i + 3u] = d;
        }
        // group 4-7
        {
            let i = 4u;
            var a = state[i + 0u];
            var b = state[i + 1u];
            var c = state[i + 2u];
            var d = state[i + 3u];
            a = a + b; d = d ^ a; d = rotl(d, 16u);

            a = a + b; d = d ^ a; d = rotl(d, 8u);
            c = c + d; b = b ^ c; b = rotl(b, 7u);
            state[i + 0u] = a;
            state[i + 1u] = b;
            state[i + 2u] = c;
            state[i + 3u] = d;
        }
        // group 8-11
        {
            let i = 8u;
            var a = state[i + 0u];
            var b = state[i + 1u];
            var c = state[i + 2u];
            var d = state[i + 3u];
            a = a + b; d = d ^ a; d = rotl(d, 16u);
            c = c + d; b = b ^ c; b = rotl(b, 12u);
            a = a + b; d = d ^ a; d = rotl(d, 8u);

            state[i + 1u] = b;
            state[i + 2u] = c;
            state[i + 3u] = d;
        }
        // group 12-15
        {
            let i = 12u;
            var a = state[i + 0u];
            var b = state[i + 1u];
            var c = state[i + 2u];
            var d = state[i + 3u];
            a = a + b; d = d ^ a; d = rotl(d, 16u);
            c = c + d; b = b ^ c; b = rotl(b, 12u);
            a = a + b; d = d ^ a; d = rotl(d, 8u);
            c = c + d; b = b ^ c; b = rotl(b, 7u);
            state[i + 0u] = a;
            state[i + 1u] = b;
            state[i + 2u] = c;
            state[i + 3u] = d;
        }
    }

    let h0 = state[0u] ^ state[2u] ^ state[4u] ^ state[6u] ^ state[8u] ^ state[10u] ^ state[12u] ^ state[14u];
    let h1 = state[1u] ^ state[3u] ^ state[5u] ^ state[7u] ^ state[9u] ^ state[11u] ^ state[13u] ^ state[15u];

    let tz = select(countTrailingZeros(h0), 32u + countTrailingZeros(h1), h0 == 0u);

    if (tz >= params.x) {
        result[0] = 1u;
        result[1] = nonce;
        result[2] = h0;
        result[3] = h1;
    }
}
