```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdint>

struct Item {
    int w;
    int h;
    int id;
    int orig_w;
    int orig_h;
};

struct Placement {
    int bin;
    int x;
    int y;
    int rotated;
};

int main() {
    int N, boardW, boardH;
    std::cin >> N >> boardW >> boardH;
    std::vector<Item> items(N);
    for (int i = 0; i < N; ++i) {
        int w, h;
        std::cin >> w >> h;
        items[i].orig_w = w;
        items[i].orig_h = h;
        items[i].id = i;
        if (w > h) {
            items[i].w = h;
            items[i].h = w;
        } else {
            items[i].w = w;
            items[i].h = h;
        }
    }
    std::sort(items.begin(), items.end(), [](const Item& a, const Item& b) {
        return a.h > b.h;
    });
    std::vector<Placement> placements(N);
    int current_bin = -1;
    int current_y = 0;
    int current_level_h = 0;
    int current_x = 0;
    int remaining_w = 0;
    for (auto& item : items) {
        bool placed = false;
        if (current_bin >= 0) {
            if (item.w <= remaining_w && item.h <= current_level_h) {
                placements[item.id].bin = current_bin;
                placements[item.id].x = current_x;
                placements[item.id].y = current_y;
                placements[item.id].rotated = (item.w != item.orig_w) ? 1 : 0;
                current_x += item.w;
                remaining_w -= item.w;
                placed = true;
            }
        }
        if (!placed && current_bin >= 0) {
            if (current_y + current_level_h + item.h <= boardH && item.w <= boardW) {
                current_y += current_level_h;
                current_level_h = item.h;
                current_x = 0;
                remaining_w = boardW - item.w;
                placements[item.id].bin = current_bin;
                placements[item.id].x = current_x;
                placements[item.id].y = current_y;
                placements[item.id].rotated = (item.w != item.orig_w) ? 1 : 0;
                current_x += item.w;
                remaining_w -= item.w;
                placed = true;
            }
        }
        if (!placed) {
            ++current_bin;
            current_y = 0;
            current_level_h = item.h;
            current_x = 0;
            remaining_w = boardW - item.w;
            placements[item.id].bin = current_bin;
            placements[item.id].x = current_x;
            placements[item.id].y = current_y;
            placements[item.id].rotated = (item.w != item.orig_w) ? 1 : 0;
            current_x += item.w;
            remaining_w -= item.w;
        }
    }
    int num_boards = current_bin + 1;
    std::cout << num_boards << std::endl;
    for (int i = 0; i < N; ++i) {
        const auto& p = placements[i];
        std::cout << p.bin << " " << p.x << " " << p.y << " " << p.rotated << std::endl;
    }
    return 0;
}
```
