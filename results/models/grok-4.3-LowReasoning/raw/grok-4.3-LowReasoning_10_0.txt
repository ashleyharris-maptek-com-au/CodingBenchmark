```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdint>

struct Item {
    int w;
    int h;
    int id;
};

struct Placement {
    int bin;
    int x;
    int y;
    int rotated;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N, BW, BH;
    std::cin >> N >> BW >> BH;

    std::vector<Item> items(N);
    for (int i = 0; i < N; ++i) {
        int w, h;
        std::cin >> w >> h;
        items[i].w = w;
        items[i].h = h;
        items[i].id = i;
    }

    // Sort by decreasing max side (prefer tall)
    std::sort(items.begin(), items.end(), [](const Item& a, const Item& b) {
        int ma = std::max(a.w, a.h);
        int mb = std::max(b.w, b.h);
        return ma > mb;
    });

    std::vector<Placement> placements(N);
    int num_bins = 0;

    // Simple next-fit level algorithm with rotation
    int cur_bin = -1;
    int cur_y = 0;
    int cur_level_h = 0;
    int cur_x = 0;

    auto start_new_bin = [&]() {
        ++num_bins;
        cur_bin = num_bins - 1;
        cur_y = 0;
        cur_level_h = 0;
        cur_x = 0;
    };

    start_new_bin();

    for (auto& it : items) {
        // Try both orientations, prefer original if possible
        int best_w = it.w;
        int best_h = it.h;
        int rot = 0;

        if (it.w > BW || it.h > BH) {
            // must rotate
            best_w = it.h;
            best_h = it.w;
            rot = 1;
        } else if (it.h > it.w && it.h <= BH && it.w <= BW) {
            // keep tall
            best_w = it.w;
            best_h = it.h;
            rot = 0;
        } else if (it.w > it.h && it.w <= BH && it.h <= BW) {
            // rotate to make taller
            best_w = it.h;
            best_h = it.w;
            rot = 1;
        }

        // Ensure fits board
        if (best_w > BW || best_h > BH) {
            // swap if needed
            std::swap(best_w, best_h);
            rot = 1 - rot;
        }

        bool placed = false;

        // Try current level
        if (cur_x + best_w <= BW && best_h <= cur_level_h) {
            placements[it.id] = {cur_bin, cur_x, cur_y, rot};
            cur_x += best_w;
            placed = true;
        }

        if (!placed) {
            // Close current level, open new level
            cur_y += cur_level_h;
            cur_level_h = best_h;
            cur_x = best_w;

            if (cur_y + cur_level_h > BH) {
                start_new_bin();
                cur_level_h = best_h;
                cur_x = best_w;
                cur_y = 0;
            }

            placements[it.id] = {cur_bin, 0, cur_y, rot};
            cur_x = best_w;
            placed = true;
        }
    }

    // Output
    std::cout << num_bins << '\n';
    for (int i = 0; i < N; ++i) {
        const auto& p = placements[i];
        std::cout << p.bin << ' ' << p.x << ' ' << p.y << ' ' << p.rot << '\n';
    }

    return 0;
}
```
