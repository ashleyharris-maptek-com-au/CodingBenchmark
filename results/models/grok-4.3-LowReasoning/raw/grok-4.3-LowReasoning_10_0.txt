```cpp
#include <bits/stdc++.h>
using namespace std;

struct Rect {
    int w, h, id;
};

struct Placement {
    int bin, x, y, rotated;
};

struct Shelf {
    int y;
    int height;
    int fill_x;
};

struct Bin {
    vector<Shelf> shelves;
    int current_height = 0;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, W, H;
    cin >> N >> W >> H;
    
    vector<Rect> rects(N);
    for (int i = 0; i < N; i++) {
        cin >> rects[i].w >> rects[i].h;
        rects[i].id = i;
    }
    
    // Sort by decreasing height
    sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        return max(a.w, a.h) > max(b.w, b.h);
    });
    
    vector<Bin> bins;
    vector<Placement> placements(N);
    
    auto can_place = [&](int rw, int rh, Bin& bin, Placement& p) -> bool {
        // Try existing shelves
        for (size_t s = 0; s < bin.shelves.size(); ++s) {
            Shelf& sh = bin.shelves[s];
            if (sh.height >= rh && sh.fill_x + rw <= W) {
                p.x = sh.fill_x;
                p.y = sh.y;
                p.rotated = (rw != rects[0].w && rw != rects[0].h ? 1 : 0); // placeholder, set later
                sh.fill_x += rw;
                return true;
            }
        }
        // Try new shelf
        if (bin.current_height + rh <= H) {
            Shelf sh;
            sh.y = bin.current_height;
            sh.height = rh;
            sh.fill_x = rw;
            bin.shelves.push_back(sh);
            bin.current_height += rh;
            p.x = 0;
            p.y = sh.y;
            return true;
        }
        return false;
    };
    
    for (auto& r : rects) {
        int rw = r.w, rh = r.h;
        bool rotated = false;
        
        Placement best_p;
        int best_bin = -1;
        
        // Try all existing bins
        for (size_t b = 0; b < bins.size(); ++b) {
            // Try non-rotated
            if (rw <= W && rh <= H) {
                Placement p;
                Bin temp = bins[b];
                if (can_place(rw, rh, temp, p)) {
                    best_bin = b;
                    best_p = p;
                    best_p.rotated = 0;
                    bins[b] = temp;
                    goto placed;
                }
            }
            // Try rotated
            if (rh <= W && rw <= H) {
                Placement p;
                Bin temp = bins[b];
                if (can_place(rh, rw, temp, p)) {
                    best_bin = b;
                    best_p = p;
                    best_p.rotated = 1;
                    bins[b] = temp;
                    goto placed;
                }
            }
        }
        
        // New bin
        {
            Bin newbin;
            bins.push_back(newbin);
            int b = bins.size() - 1;
            if (rw <= W && rh <= H) {
                Placement p;
                if (can_place(rw, rh, bins[b], p)) {
                    best_bin = b;
                    best_p = p;
                    best_p.rotated = 0;
                    goto placed;
                }
            }
            if (rh <= W && rw <= H) {
                Placement p;
                if (can_place(rh, rw, bins[b], p)) {
                    best_bin = b;
                    best_p = p;
                    best_p.rotated = 1;
                    goto placed;
                }
            }
            // Fallback: force first shelf
            Shelf sh = {0, min(rh, H), min(rw, W)};
            bins[b].shelves.push_back(sh);
            bins[b].current_height = sh.height;
            best_bin = b;
            best_p.x = 0;
            best_p.y = 0;
            best_p.rotated = (rw > W ? 1 : 0);
        }
        
    placed:
        placements[r.id] = {best_bin, best_p.x, best_p.y, best_p.rotated};
    }
    
    int num_bins = bins.size();
    cout << num_bins << '\n';
    for (int i = 0; i < N; i++) {
        auto& pl = placements[i];
        cout << pl.bin << ' ' << pl.x << ' ' << pl.y << ' ' << pl.rotated << '\n';
    }
    
    return 0;
}
```
