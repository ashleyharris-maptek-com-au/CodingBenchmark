```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 100
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint = OpTypeInt 32 0

; Input gl_PerVertex block (array size 1 for InputPoints)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr Input

; Input color array
%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_color = OpTypePointer Input %arr_v3_in
%in_color = OpVariable %ptr_in_color Input

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Constants
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%float_h = OpConstant %float 0.04
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%v4_0001 = OpConstantComposite %v4float %float_0 %float_0 %float_0 %float_1

; Main
%main = OpFunction %void None %func
%entry = OpLabel

; Read input position and color
%in_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_0 %int_0
%in_pos = OpLoad %v4float %in_pos_ptr
%in_col_ptr = OpAccessChain %ptr_in_v3 %in_color %uint_0
%in_col = OpLoad %v3float %in_col_ptr

; Extract x,y
%px = OpCompositeExtract %float %in_pos 0
%py = OpCompositeExtract %float %in_pos 1

; Compute four corners
%mx = OpFSub %float %px %float_h
%pxh = OpFAdd %float %px %float_h
%my = OpFSub %float %py %float_h
%pyh = OpFAdd %float %py %float_h

; Vertex 0: (mx, my)
%v0 = OpCompositeConstruct %v4float %mx %my %float_0 %float_1
OpStore %outColor %in_col
%out0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out0 %v0
OpEmitVertex

; Vertex 1: (pxh, my)
%v1 = OpCompositeConstruct %v4float %pxh %my %float_0 %float_1
OpStore %outColor %in_col
%out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out1 %v1
OpEmitVertex

; Vertex 2: (mx, pyh)
%v2 = OpCompositeConstruct %v4float %mx %pyh %float_0 %float_1
OpStore %outColor %in_col
%out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out2 %v2
OpEmitVertex

; Vertex 3: (pxh, pyh)
%v3 = OpCompositeConstruct %v4float %pxh %pyh %float_0 %float_1
OpStore %outColor %in_col
%out3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out3 %v3
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
