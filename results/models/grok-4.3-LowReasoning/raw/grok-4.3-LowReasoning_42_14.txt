```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%uint_6 = OpConstant %uint 6
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
%arr_v3float_6 = OpTypeArray %v3float %uint_6
%ptr_in_v3arr = OpTypePointer Input %arr_v3float_6
%worldPos = OpVariable %ptr_in_v3arr Input
%normal = OpVariable %ptr_in_v3arr Input
%color = OpVariable %ptr_in_v3arr Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%main = OpFunction %void None %func
%entry = OpLabel
; Load p0, p2, p4 (xyz)
%chain0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0v4 = OpLoad %v4float %chain0
%p0 = OpVectorShuffle %v3float %p0v4 %p0v4 0 1 2
%chain2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2v4 = OpLoad %v4float %chain2
%p2 = OpVectorShuffle %v3float %p2v4 %p2v4 0 1 2
%chain4 = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p4v4 = OpLoad %v4float %chain4
%p4 = OpVectorShuffle %v3float %p4v4 %p4v4 0 1 2
; Edges
%e1 = OpFSub %v3float %p2 %p0
%e2 = OpFSub %v3float %p4 %p0
; Cross product
%cy = OpFMul %float %e1y %e2z wait, expand manually:
%e1x = OpCompositeExtract %float %e1 0
%e1y = OpCompositeExtract %float %e1 1
%e1z = OpCompositeExtract %float %e1 2
%e2x = OpCompositeExtract %float %e2 0
%e2y = OpCompositeExtract %float %e2 1
%e2z = OpCompositeExtract %float %e2 2
%cx = OpFMul %float %e1y %e2z
%cy_ = OpFMul %float %e1z %e2x
%cz = OpFMul %float %e1x %e2y
%mx = OpFMul %float %e1z %e2y
%my = OpFMul %float %e1x %e2z
%mz = OpFMul %float %e1y %e2x
%nx_ = OpFSub %float %cx %mx
%ny_ = OpFSub %float %cy_ %my
%nz_ = OpFSub %float %cz %mz
%Nraw = OpCompositeConstruct %v3float %nx_ %ny_ %nz_
%N = OpExtInst %v3float %1 Normalize %Nraw
; Lighting
%L = OpConstantComposite %v3float %float_0_577 %float_0_577 %float_0_577 ; approx 1/sqrt(3)
%diff = OpExtInst %float %1 FMax %float_0 %dotNL
%dotNL = OpDot %float %N %L
%diffuse = OpExtInst %float %1 FMax %float_0 %dotNL
; Average base color
%col0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%col0 = OpLoad %v3float %col0_ptr
%col2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%col2 = OpLoad %v3float %col2_ptr
%col4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
%col4 = OpLoad %v3float %col4_ptr
%sum1 = OpFAdd %v3float %col0 %col2
%base = OpFAdd %v3float %sum1 %col4
%inv3 = OpConstant %float 0.333333
%baseCol = OpVectorTimesScalar %v3float %base %inv3
; Final color
%k = OpFMul %float %float_0_8 %diffuse
%scale = OpFAdd %float %float_0_2 %k
%finalC = OpVectorTimesScalar %v3float %baseCol %scale
; Emit 3 vertices
; vert 0
%outp0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outp0 %p0v4
OpStore %outColor %finalC
OpEmitVertex
; vert 1
%outp2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outp2 %p2v4
OpStore %outColor %finalC
OpEmitVertex
; vert 2
%outp4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outp4 %p4v4
OpStore %outColor %finalC
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
```
