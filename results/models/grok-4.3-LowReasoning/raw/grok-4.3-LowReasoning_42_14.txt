```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%uint_6 = OpConstant %uint 6
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_v3 = OpTypePointer Input %arr_v3_in
%in_worldPos = OpVariable %ptr_in_v3 Input
%in_normal = OpVariable %ptr_in_v3 Input
%in_color = OpVariable %ptr_in_v3 Input
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3_elem = OpTypePointer Input %v3float
%float_0 = OpConstant %float 0
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1
%float_3 = OpConstant %float 3
%v3_one = OpConstantComposite %v3float %float_1 %float_1 %float_1
%main = OpFunction %void None %func
%entry = OpLabel
; Load positions p0, p2, p4
%p0_ptr = OpAccessChain %ptr_in_v3_elem %in_worldPos %int_0
%p0 = OpLoad %v3float %p0_ptr
%p2_ptr = OpAccessChain %ptr_in_v3_elem %in_worldPos %int_2
%p2 = OpLoad %v3float %p2_ptr
%p4_ptr = OpAccessChain %ptr_in_v3_elem %in_worldPos %int_4
%p4 = OpLoad %v3float %p4_ptr
; Compute edges
%e1 = OpFSub %v3float %p2 %p0
%e2 = OpFSub %v3float %p4 %p0
; Cross product e1 × e2
%cy = OpFMul %float %e1_y %e2_z
%cz = OpFMul %float %e1_z %e2_y
%cx = OpFMul %float %e1_z %e2_x
%cw = OpFMul %float %e1_x %e2_z
%cx2 = OpFMul %float %e1_x %e2_y
%cy2 = OpFMul %float %e1_y %e2_x
%cross_x = OpFSub %float %cy %cz
%cross_y = OpFSub %float %cw %cx
%cross_z = OpFSub %float %cx2 %cy2
%cross = OpCompositeConstruct %v3float %cross_x %cross_y %cross_z
; Normalize face normal
%N = OpExtInst %v3float %1 Normalize %cross
; Light dir
%L = OpExtInst %v3float %1 Normalize %v3_one
; dot(N, L)
%dotNL = OpDot %float %N %L
%diff = OpExtInst %float %1 FMax %dotNL %float_0
; Base color = average of colors 0,2,4
%c0_ptr = OpAccessChain %ptr_in_v3_elem %in_color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c2_ptr = OpAccessChain %ptr_in_v3_elem %in_color %int_2
%c2 = OpLoad %v3float %c2_ptr
%c4_ptr = OpAccessChain %ptr_in_v3_elem %in_color %int_4
%c4 = OpLoad %v3float %c4_ptr
%sum1 = OpFAdd %v3float %c0 %c2
%sum = OpFAdd %v3float %sum1 %c4
%base = OpVectorTimesScalar %v3float %sum %float_3
; Final color
%scale = OpFMul %float %float_0_8 %diff
%atten = OpFAdd %float %float_0_2 %scale
%finalC = OpVectorTimesScalar %v3float %base %atten
; Emit vertex 0
%pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%out_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos0 %pos0
OpStore %outColor %finalC
OpEmitVertex
; Emit vertex 2
%pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%out_pos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos2 %pos2
OpStore %outColor %finalC
OpEmitVertex
; Emit vertex 4
%pos4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %pos4_ptr
%out_pos4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos4 %pos4
OpStore %outColor %finalC
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
```
