use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    
    let first_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => {
            println!("0");
            return;
        }
    };
    
    let parts: Vec<usize> = first_line
        .split_whitespace()
        .filter_map(|s| s.parse().ok())
        .collect();
    
    if parts.len() < 2 {
        println!("0");
        return;
    }
    
    let n = parts[0];
    let m = parts[1];
    
    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);
    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let uv: Vec<usize> = line
                .split_whitespace()
                .filter_map(|s| s.parse().ok())
                .collect();
            if uv.len() >= 2 {
                let u = uv[0];
                let v = uv[1];
                if u != v && u < n && v < n {
                    edges.push((u.min(v), u.max(v)));
                }
            }
        }
    }
    
    // Remove duplicates for layout purposes
    edges.sort_unstable();
    edges.dedup();
    
    // Output N optionally
    println!("{}", n);
    
    if n == 0 {
        return;
    }
    
    // Adaptive layout based on scale
    if n <= 800 {
        force_directed_layout(n, &edges);
    } else {
        grid_layout(n);
    }
}

fn force_directed_layout(n: usize, edges: &[(usize, usize)]) {
    let area: f64 = 1000.0 * 1000.0;
    let k: f64 = (area / n as f64).sqrt();
    let mut positions: Vec<(f64, f64)> = Vec::with_capacity(n);
    
    // Initial random placement
    for i in 0..n {
        let x = (i as f64 * 12.34567).sin().abs() * 900.0 + 50.0;
        let y = (i as f64 * 67.89012).cos().abs() * 900.0 + 50.0;
        positions.push((x, y));
    }
    
    let iterations = if n <= 100 { 300 } else if n <= 300 { 150 } else { 80 };
    let temp_start = 900.0 / n as f64 * 20.0 + 50.0;
    
    let mut temp = temp_start;
    
    for iter in 0..iterations {
        let mut disp: Vec<(f64, f64)> = vec![(0.0, 0.0); n];
        
        // Repulsive forces
        for i in 0..n {
            for j in (i + 1)..n {
                let dx = positions[i].0 - positions[j].0;
                let dy = positions[i].1 - positions[j].1;
                let dist_sq = dx * dx + dy * dy + 0.01;
                let dist = dist_sq.sqrt();
                let force = k * k / dist;
                let fx = dx / dist * force;
                let fy = dy / dist * force;
                disp[i].0 += fx;
                disp[i].1 += fy;
                disp[j].0 -= fx;
                disp[j].1 -= fy;
            }
        }
        
        // Attractive forces
        for &(u, v) in edges {
            let dx = positions[u].0 - positions[v].0;
            let dy = positions[u].1 - positions[v].1;
            let dist_sq = dx * dx + dy * dy + 0.01;
            let dist = dist_sq.sqrt();
            let force = dist * dist / k;
            let fx = dx / dist * force;
            let fy = dy / dist * force;
            disp[u].0 -= fx;
            disp[u].1 -= fy;
            disp[v].0 += fx;
            disp[v].1 += fy;
        }
        
        // Apply forces with temperature
        for i in 0..n {
            let dx = disp[i].0;
            let dy = disp[i].1;
            let dist = (dx * dx + dy * dy).sqrt() + 0.001;
            let limited_dist = dist.min(temp);
            positions[i].0 += dx / dist * limited_dist;
            positions[i].1 += dy / dist * limited_dist;
            
            // Clamp to bounds
            positions[i].0 = positions[i].0.max(10.0).min(990.0);
            positions[i].1 = positions[i].1.max(10.0).min(990.0);
        }
        
        temp *= 0.92;
        if temp < 0.1 {
            break;
        }
    }
    
    // Output coordinates
    for (x, y) in positions {
        println!("{:.4} {:.4}", x, y);
    }
}

fn grid_layout(n: usize) {
    let cols = (n as f64).sqrt().ceil() as usize;
    let spacing = 1000.0 / (cols as f64 + 1.0);
    
    for i in 0..n {
        let row = i / cols;
        let col = i % cols;
        let x = (col as f64 + 1.0) * spacing;
        let y = (row as f64 + 1.0) * spacing;
        println!("{:.4} {:.4}", x, y);
    }
}
