```rust
use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first = lines.next().unwrap().unwrap();
    let mut parts = first.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![vec![]; n];
    let mut seen: HashSet<(usize, usize)> = HashSet::new();

    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut ep = line.split_whitespace();
        let u: usize = ep.next().unwrap().parse().unwrap();
        let v: usize = ep.next().unwrap().parse().unwrap();
        if u == v || seen.contains(&(u.min(v), u.max(v))) {
            continue;
        }
        seen.insert((u.min(v), u.max(v)));
        adj[u].push(v);
        adj[v].push(u);
    }

    let mut x: Vec<f64> = vec![0.0; n];
    let mut y: Vec<f64> = vec![0.0; n];

    if n <= 2000 {
        // Force-directed layout (Fruchterman-Reingold style)
        let area = (n as f64) * 100.0;
        let k = (area / n as f64).sqrt();
        let mut pos: Vec<(f64, f64)> = (0..n).map(|i| {
            let t = (i as f64) * 0.6180339887;
            (n as f64 * 0.5 * (t.fract() * 2.0 - 1.0), n as f64 * 0.5 * ((t * 1.3).fract() * 2.0 - 1.0))
        }).collect();

        let iterations = if n < 200 { 300 } else if n < 500 { 150 } else { 80 };
        let temp0 = (n as f64).sqrt() * 2.0;

        for iter in 0..iterations {
            let mut disp: Vec<(f64, f64)> = vec![(0.0, 0.0); n];
            let temp = temp0 * (1.0 - (iter as f64) / (iterations as f64));

            // Repulsion
            for i in 0..n {
                for j in (i + 1)..n {
                    let dx = pos[i].0 - pos[j].0;
                    let dy = pos[i].1 - pos[j].1;
                    let dist2 = dx * dx + dy * dy + 1e-9;
                    let dist = dist2.sqrt();
                    let force = k * k / dist;
                    let fx = force * dx / dist;
                    let fy = force * dy / dist;
                    disp[i].0 += fx;
                    disp[i].1 += fy;
                    disp[j].0 -= fx;
                    disp[j].1 -= fy;
                }
            }

            // Attraction
            for u in 0..n {
                for &v in &adj[u] {
                    if v > u {
                        let dx = pos[u].0 - pos[v].0;
                        let dy = pos[u].1 - pos[v].1;
                        let dist = (dx * dx + dy * dy).sqrt() + 1e-9;
                        let force = dist * dist / k;
                        let fx = force * dx / dist;
                        let fy = force * dy / dist;
                        disp[u].0 -= fx;
                        disp[u].1 -= fy;
                        disp[v].0 += fx;
                        disp[v].1 += fy;
                    }
                }
            }

            // Update
            for i in 0..n {
                let dx = disp[i].0;
                let dy = disp[i].1;
                let dlen = (dx * dx + dy * dy).sqrt();
                if dlen > 0.0 {
                    let d = dlen.min(temp);
                    pos[i].0 += dx / dlen * d;
                    pos[i].1 += dy / dlen * d;
                }
            }
        }

        for i in 0..n {
            x[i] = pos[i].0;
            y[i] = pos[i].1;
        }
    } else {
        // Fast scalable layout for large graphs: spiral + degree-based radius
        let mut deg: Vec<usize> = adj.iter().map(|v| v.len()).collect();
        let max_deg = *deg.iter().max().unwrap_or(&1) as f64;
        for i in 0..n {
            let r = (i as f64 + 1.0).sqrt() * (1.0 + 0.3 * (deg[i] as f64 / max_deg));
            let theta = (i as f64) * 2.4;
            x[i] = r * theta.cos();
            y[i] = r * theta.sin();
        }
    }

    // Output
    println!("{}", n);
    for i in 0..n {
        println!("{:.6} {:.6}", x[i], y[i]);
    }
}
```
