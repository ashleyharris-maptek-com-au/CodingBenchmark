; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%_ptr_Input_v3float = OpTypePointer Input %v3float
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%uv = OpVariable %_ptr_Input_v3float Input
%tangent = OpVariable %_ptr_Input_v3float Input
%color = OpVariable %_ptr_Input_v3float Input
%_ptr_Output_v4float = OpTypePointer Output %v4float
%outColor = OpVariable %_ptr_Output_v4float Output
%void = OpTypeVoid
%func = OpTypeFunction %void
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%float_0_5 = OpConstant %float 0.5
%skyColor = OpConstantComposite %v3float %float_0_5 %float_0_5 %float_0_5
%groundColor = OpConstantComposite %v3float %float_0_5 %float_0_5 %float_0_5
%one = OpConstant %float 1
%zero = OpConstant %float 0
%main = OpFunction %void None %func
%entry = OpLabel
%Nraw = OpLoad %v3float %normal
%N = OpNormalize %v3float %Nraw
%Ny = OpCompositeExtract %float %N 1
%blend = OpFMul %float %Ny %float_0_5
%blend2 = OpFAdd %float %blend %float_0_5
%amb = OpExtInst %v3float %1 FMix %groundColor %skyColor %blend2
%LposPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%Lpos = OpLoad %v4float %LposPtr
%Lxyz = OpVectorShuffle %v3float %Lpos %Lpos 0 1 2
%W = OpLoad %v3float %worldPos
%Ldir = OpFSub %v3float %Lxyz %W
%L = OpNormalize %v3float %Ldir
%d = OpDot %float %N %L
%d0 = OpExtInst %float %1 FMax %d %zero
%diff = OpFMul %float %d0 %float_0_5
%diff3 = OpCompositeConstruct %v3float %diff %diff %diff
%col = OpFAdd %v3float %amb %diff3
%out = OpCompositeConstruct %v4float %col %one
OpStore %outColor %out
OpReturn
OpFunctionEnd
%int_3 = OpConstant %int 3
%1 = OpExtInstImport "GLSL.std.450"
