```csharp
using System;
using System.IO;
using System.Linq;

class Program
{
    static void Main()
    {
        var input = Console.In.ReadToEnd().Split(new[] { ' ', '\t', '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);
        int index = 0;
        int N = int.Parse(input[index++]);
        long[,] flow = new long[N, N];
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                flow[i, j] = long.Parse(input[index++]);
        long[,] dist = new long[N, N];
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                dist[i, j] = long.Parse(input[index++]);

        int[] p = new int[N];
        for (int i = 0; i < N; i++) p[i] = i;

        // Simple local search with delta evaluation (O(N) per swap)
        bool improved = true;
        int maxIter = Math.Min(10000, N * 50);
        int iter = 0;
        while (improved && iter < maxIter)
        {
            improved = false;
            iter++;
            for (int i = 0; i < N && !improved; i++)
            {
                for (int j = i + 1; j < N; j++)
                {
                    long delta = ComputeDelta(flow, dist, p, i, j, N);
                    if (delta < 0)
                    {
                        Swap(p, i, j);
                        improved = true;
                        break;
                    }
                }
            }
        }

        long cost = ComputeCost(flow, dist, p, N);
        Console.WriteLine(cost);
        Console.WriteLine(string.Join(" ", p));
    }

    static long ComputeDelta(long[,] flow, long[,] dist, int[] p, int a, int b, int N)
    {
        long delta = 0;
        int pa = p[a], pb = p[b];
        for (int k = 0; k < N; k++)
        {
            if (k == a || k == b) continue;
            int pk = p[k];
            delta += (flow[a, k] - flow[b, k]) * (dist[pa, pk] - dist[pb, pk]);
            delta += (flow[k, a] - flow[k, b]) * (dist[pk, pa] - dist[pk, pb]);
        }
        delta += (flow[a, b] - flow[b, a]) * (dist[pa, pb] - dist[pb, pa]); // usually 0 if symmetric
        return 2 * delta;
    }

    static long ComputeCost(long[,] flow, long[,] dist, int[] p, int N)
    {
        long cost = 0;
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                cost += flow[i, j] * dist[p[i], p[j]];
        return cost;
    }

    static void Swap(int[] p, int i, int j)
    {
        int tmp = p[i]; p[i] = p[j]; p[j] = tmp;
    }
}
```
