using System;
using System.Collections.Generic;
using System.IO;

class Program
{
    static void Main()
    {
        // Read header
        string header = Console.ReadLine();
        if (string.IsNullOrEmpty(header)) return;
        string[] parts = header.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 6) return;

        int XV = int.Parse(parts[0]);
        int YV = int.Parse(parts[1]);
        int ZV = int.Parse(parts[2]);
        // Block sizes ignored for logic but parsed
        double XB = double.Parse(parts[3]);
        double YB = double.Parse(parts[4]);
        double ZB = double.Parse(parts[5]);

        long totalVoxels = (long)XV * YV * ZV;
        bool largeScale = totalVoxels > 50_000_000L;

        // Adaptive strategy selection
        if (largeScale)
        {
            ProcessLargeScale(XV, YV, ZV);
        }
        else
        {
            ProcessSmallScale(XV, YV, ZV);
        }
    }

    static void ProcessSmallScale(int XV, int YV, int ZV)
    {
        // Full voxel storage for small/medium
        byte[,,] voxels = new byte[XV, YV, ZV];
        int bytesRead = 0;
        using (var stdin = Console.OpenStandardInput())
        {
            for (int x = 0; x < XV; x++)
                for (int y = 0; y < YV; y++)
                    for (int z = 0; z < ZV; z++)
                    {
                        int b = stdin.ReadByte();
                        if (b == -1) b = 0;
                        voxels[x, y, z] = (byte)b;
                        bytesRead++;
                    }
        }

        // Simple greedy: find high-value surface voxels and move to (0,0)
        var ops = new List<string>();
        // For demo: pick top exposed valuable voxels near surface
        for (int x = 0; x < XV && ops.Count < 1000; x++)
        {
            for (int y = 0; y < YV && ops.Count < 1000; y++)
            {
                // Find highest valuable exposed from top
                for (int z = ZV - 1; z >= 0; z--)
                {
                    byte d = voxels[x, y, z];
                    int mass = d % 16;
                    int val = d / 16;
                    if (mass > 0 && val > 0)
                    {
                        // Simple flat path to (0,0)
                        ops.Add($"1 {x} {y} {z} 0 0");
                        break;
                    }
                }
            }
        }

        Console.WriteLine(ops.Count);
        foreach (var op in ops)
            Console.WriteLine(op);
    }

    static void ProcessLargeScale(int XV, int YV, int ZV)
    {
        // Memory-efficient streaming: only track top layer + high-value candidates
        // Process layer-by-layer from top, greedy surface mining only
        var ops = new List<string>();
        int maxOps = 50000; // Limit ops for time

        using (var stdin = Console.OpenStandardInput())
        {
            // Skip to top layers only for large scale
            int topLayers = Math.Min(5, ZV);
            long skipBytes = (long)XV * YV * (ZV - topLayers);
            // Simple skip (not perfect but adaptive)
            for (long i = 0; i < skipBytes && stdin.CanRead; i++)
                stdin.ReadByte();

            for (int x = 0; x < XV && ops.Count < maxOps; x++)
            {
                for (int y = 0; y < YV && ops.Count < maxOps; y++)
                {
                    for (int z = 0; z < topLayers; z++)
                    {
                        int b = stdin.ReadByte();
                        if (b == -1) b = 0;
                        int mass = b % 16;
                        int val = b / 16;
                        if (mass > 0 && val >= 8) // High value only
                        {
                            // Direct Manhattan to plant, assume flat enough near top
                            ops.Add($"1 {x} {y} {ZV - topLayers + z} 0 0");
                            break;
                        }
                    }
                }
            }
        }

        Console.WriteLine(ops.Count);
        foreach (var op in ops)
            Console.WriteLine(op);
    }
}
