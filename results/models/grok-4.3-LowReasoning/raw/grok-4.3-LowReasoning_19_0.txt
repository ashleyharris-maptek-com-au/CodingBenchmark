using System;
using System.Collections.Generic;
using System.IO;

class Program
{
    static void Main()
    {
        var input = Console.OpenStandardInput();
        var reader = new BinaryReader(input);

        // Read header line (text until newline)
        string headerLine = "";
        int b;
        while ((b = reader.ReadByte()) != '\n')
        {
            headerLine += (char)b;
        }
        var parts = headerLine.Trim().Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        int XVoxelCount = int.Parse(parts[0]);
        int YVoxelCount = int.Parse(parts[1]);
        int ZVoxelCount = int.Parse(parts[2]);
        // Block sizes ignored for pathing (use voxel units)

        long totalVoxels = (long)XVoxelCount * YVoxelCount * ZVoxelCount;

        List<string> operations = new List<string>();

        if (totalVoxels <= 500000) // small: attempt greedy surface valuable moves
        {
            byte[] data = reader.ReadBytes((int)totalVoxels);
            // Simple strategy: scan for high value near assumed top, move directly to (0,0)
            // Assume Z increases upward; expose from high Z
            for (int x = 0; x < XVoxelCount && operations.Count < 200; x++)
            {
                for (int y = 0; y < YVoxelCount && operations.Count < 200; y++)
                {
                    for (int z = ZVoxelCount - 1; z >= 0; z--)
                    {
                        int idx = x * (YVoxelCount * ZVoxelCount) + y * ZVoxelCount + z;
                        if (idx >= data.Length) continue;
                        byte d = data[idx];
                        int mass = d % 16;
                        int val = d / 16;
                        if (mass > 0 && val > 0)
                        {
                            // Direct manhattan path on surface-ish, slope compliant (flat for simplicity)
                            string op = $"1 {x} {y} {z} 0 0 {0} {0}";
                            operations.Add(op);
                            break;
                        }
                    }
                }
            }
        }
        else // large scale: limited greedy passes on first bytes only
        {
            // Stream limited number, process top layer only heuristically
            int maxScan = Math.Min(1000000, (int)totalVoxels);
            byte[] sample = reader.ReadBytes(maxScan);
            for (int i = 0; i < sample.Length && operations.Count < 50; i++)
            {
                byte d = sample[i];
                int val = d / 16;
                int mass = d % 16;
                if (val >= 8 && mass > 0)
                {
                    // Approximate position (linearize), dump to plant if valuable
                    int approxX = i / (YVoxelCount * ZVoxelCount);
                    int approxY = (i / ZVoxelCount) % YVoxelCount;
                    int approxZ = i % ZVoxelCount;
                    if (approxX < XVoxelCount && approxY < YVoxelCount)
                    {
                        string op = $"1 {approxX} {approxY} {approxZ} 0 0 0 0";
                        operations.Add(op);
                    }
                }
            }
        }

        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.WriteLine(op);
        }
    }
}
