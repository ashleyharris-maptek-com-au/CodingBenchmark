; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_color %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_ubo = OpTypePointer Uniform %UBO
%ptr_in_vec3 = OpTypePointer Input %vec3
%ptr_out_vec4 = OpTypePointer Output %vec4
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%in_worldPos = OpVariable %ptr_in_vec3 Input
%in_normal = OpVariable %ptr_in_vec3 Input
%in_color = OpVariable %ptr_in_vec3 Input
%outColor = OpVariable %ptr_out_vec4 Output
%ubo = OpVariable %ptr_ubo Uniform
%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%cameraPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_1 = OpConstant %float 0.1
%float_0_05 = OpConstant %float 0.05
%float_32 = OpConstant %float 32
%vec3_amb = OpConstantComposite %vec3 %float_0_1 %float_0_05 %float_0_05
%vec3_spec_col = OpConstantComposite %vec3 %float_1 %float_1 %float_1
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%main = OpFunction %void None %func
%entry = OpLabel
%wp = OpLoad %vec3 %in_worldPos
%N_in = OpLoad %vec3 %in_normal
%N = OpNormalize %vec3 %N_in
%col = OpLoad %vec3 %in_color
%LP4 = OpLoad %vec4 %lightPos_ptr
%LP = OpVectorShuffle %vec3 %LP4 %LP4 0 1 2
%Lvec = OpSub %vec3 %LP %wp
%L = OpNormalize %vec3 %Lvec
%CP4 = OpLoad %vec4 %cameraPos_ptr
%CP = OpVectorShuffle %vec3 %CP4 %CP4 0 1 2
%Vvec = OpSub %vec3 %CP %wp
%V = OpNormalize %vec3 %Vvec
%negL = OpFNegate %vec3 %L
%R = OpReflect %vec3 %negL %N
%diff_dot = OpDot %float %N %L
%diff_max = OpExtInst %float %glsl_max %diff_dot %float_0
%diff = OpFMul %vec3 %col %diff_max
%spec_dot = OpDot %float %R %V
%spec_max = OpExtInst %float %glsl_max %spec_dot %float_0
%spec_pow = OpExtInst %float %glsl_pow %spec_max %float_32
%spec = OpFMul %vec3 %vec3_spec_col %spec_pow
%sum1 = OpFAdd %vec3 %vec3_amb %diff
%sum = OpFAdd %vec3 %sum1 %spec
%clamped = OpExtInst %vec3 %glsl_fclamp %sum %float_0 %float_1
%out = OpCompositeConstruct %vec4 %clamped %float_1
OpStore %outColor %out
OpReturn
OpFunctionEnd
%glsl_max = OpExtInstImport "GLSL.std.450"
%glsl_pow = OpExtInstImport "GLSL.std.450"
%glsl_fclamp = OpExtInstImport "GLSL.std.450"
