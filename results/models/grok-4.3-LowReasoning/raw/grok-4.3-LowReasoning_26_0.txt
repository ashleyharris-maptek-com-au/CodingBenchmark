```rust
use std::io::{Read, Write, BufReader, BufWriter};
use std::time::Instant;

const BATCH_SIZE: usize = 65536;
const MAX_ITERATIONS: usize = 100;
const MINI_BATCH_SIZE: usize = 1024;
const CONVERGENCE_THRESHOLD: f32 = 1e-4;

#[derive(Clone, Copy)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

fn distance_sq(a: &Point, b: &Point) -> f32 {
    let dx = a.x - b.x;
    let dy = a.y - b.y;
    let dz = a.z - b.z;
    dx * dx + dy * dy + dz * dz
}

fn read_points_stdin(num_points: usize) -> Vec<Point> {
    let stdin = std::io::stdin();
    let mut reader = BufReader::new(stdin.lock());
    let mut points = Vec::with_capacity(num_points.min(10_000_000));
    
    let mut buffer = [0u8; 12];
    for _ in 0..num_points {
        if reader.read_exact(&mut buffer).is_ok() {
            let x = f32::from_le_bytes([buffer[0], buffer[1], buffer[2], buffer[3]]);
            let y = f32::from_le_bytes([buffer[4], buffer[5], buffer[6], buffer[7]]);
            let z = f32::from_le_bytes([buffer[8], buffer[9], buffer[10], buffer[11]]);
            points.push(Point { x, y, z });
        } else {
            break;
        }
    }
    points
}

fn kmeans_plus_plus(points: &[Point], k: usize) -> Vec<Point> {
    use rand::Rng;
    let mut rng = rand::thread_rng();
    let mut centroids = Vec::with_capacity(k);
    
    // First centroid random
    centroids.push(points[rng.gen_range(0..points.len())]);
    
    let mut min_dist_sq: Vec<f32> = vec![f32::MAX; points.len()];
    
    for _ in 1..k {
        let mut sum = 0.0f64;
        for (i, p) in points.iter().enumerate() {
            let d = distance_sq(p, &centroids[centroids.len() - 1]);
            if d < min_dist_sq[i] {
                min_dist_sq[i] = d;
            }
            sum += min_dist_sq[i] as f64;
        }
        
        let target = rng.gen::<f64>() * sum;
        let mut cumsum = 0.0f64;
        let mut chosen = 0;
        for (i, d) in min_dist_sq.iter().enumerate() {
            cumsum += *d as f64;
            if cumsum >= target {
                chosen = i;
                break;
            }
        }
        centroids.push(points[chosen]);
    }
    centroids
}

fn assign_clusters(points: &[Point], centroids: &[Point]) -> Vec<u32> {
    points.iter().map(|p| {
        let mut best = 0;
        let mut best_dist = f32::MAX;
        for (i, c) in centroids.iter().enumerate() {
            let d = distance_sq(p, c);
            if d < best_dist {
                best_dist = d;
                best = i;
            }
        }
        best as u32
    }).collect()
}

fn update_centroids(points: &[Point], assignments: &[u32], k: usize) -> Vec<Point> {
    let mut sums = vec![(0.0f64, 0.0f64, 0.0f64, 0usize); k];
    
    for (p, &a) in points.iter().zip(assignments.iter()) {
        let idx = a as usize;
        sums[idx].0 += p.x as f64;
        sums[idx].1 += p.y as f64;
        sums[idx].2 += p.z as f64;
        sums[idx].3 += 1;
    }
    
    sums.into_iter().map(|(sx, sy, sz, cnt)| {
        if cnt > 0 {
            Point {
                x: (sx / cnt as f64) as f32,
                y: (sy / cnt as f64) as f32,
                z: (sz / cnt as f64) as f32,
            }
        } else {
            Point { x: 0.0, y: 0.0, z: 0.0 }
        }
    }).collect()
}

fn mini_batch_kmeans(points: &[Point], k: usize, max_iter: usize) -> Vec<Point> {
    use rand::seq::SliceRandom;
    let mut rng = rand::thread_rng();
    
    let mut centroids = kmeans_plus_plus(points, k);
    let mut counts = vec![0usize; k];
    
    for _ in 0..max_iter {
        let batch: Vec<&Point> = points.choose_multiple(&mut rng, MINI_BATCH_SIZE.min(points.len())).collect();
        
        let mut new_centroids = centroids.clone();
        
        for p in batch {
            let mut best = 0;
            let mut best_dist = f32::MAX;
            for (i, c) in centroids.iter().enumerate() {
                let d = distance_sq(p, c);
                if d < best_dist {
                    best_dist = d;
                    best = i;
                }
            }
            
            let eta = 1.0 / (counts[best] as f32 + 1.0).sqrt();
            counts[best] += 1;
            
            new_centroids[best].x = (1.0 - eta) * new_centroids[best].x + eta * p.x;
            new_centroids[best].y = (1.0 - eta) * new_centroids[best].y + eta * p.y;
            new_centroids[best].z = (1.0 - eta) * new_centroids[best].z + eta * p.z;
        }
        
        centroids = new_centroids;
    }
    centroids
}

fn write_assignments(assignments: &[u32]) {
    let stdout = std::io::stdout();
    let mut writer = BufWriter::new(stdout.lock());
    for &a in assignments {
        let bytes = a.to_le_bytes();
        writer.write_all(&bytes).unwrap();
    }
    writer.flush().unwrap();
}

fn main() {
    let start = Instant::now();
    
    let args: Vec<String> = std::env::args().collect();
    if args.len() < 3 {
        eprintln!("Usage: program <num_points> <num_clusters>");
        std::process::exit(1);
    }
    
    let num_points: usize = args[1].parse().unwrap();
    let k: usize = args[2].parse().unwrap();
    
    // Adaptive strategy selection
    let points = if num_points <= 5_000_000 {
        read_points_stdin(num_points)
    } else {
        // For large data, sample for centroid initialization
        let sample_size = 1_000_000.min(num_points);
        read_points_stdin(sample_size)
    };
    
    if points.is_empty() {
        return;
    }
    
    let centroids = if num_points <= 1_000_000 && k <= 100 {
        // Standard k-means for small data
        let mut cents = kmeans_plus_plus(&points, k);
        for _ in 0..MAX_ITERATIONS {
            let assignments = assign_clusters(&points, &cents);
            let new_cents = update_centroids(&points, &assignments, k);
            let shift: f32 = cents.iter().zip(new_cents.iter())
                .map(|(a, b)| distance_sq(a, b).sqrt())
                .sum();
            cents = new_cents;
            if shift < CONVERGENCE_THRESHOLD * k as f32 {
                break;
            }
        }
        cents
    } else {
        // Mini-batch for larger data
        mini_batch_kmeans(&points, k, 50)
    };
    
    // For very large datasets, we would stream assignments without storing all points
    // Here we re-read for assignment (in practice, buffer and assign on the fly)
    let all_points = if num_points <= 10_000_000 {
        points
    } else {
        // Re-read full data for assignment phase
        read_points_stdin(num_points)
    };
    
    let assignments = assign_clusters(&all_points, &centroids);
    write_assignments(&assignments);
    
    eprintln!("Completed in {:?}", start.elapsed());
}
```
