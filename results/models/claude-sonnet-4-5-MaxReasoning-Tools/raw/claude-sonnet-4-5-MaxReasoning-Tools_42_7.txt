OpCapability Geometry
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
               OpExecutionMode %main Triangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1
               OpSource GLSL 450
               OpName %main "main"
               OpName %gl_PerVertex_in "gl_PerVertex"
               OpMemberName %gl_PerVertex_in 0 "gl_Position"
               OpName %gl_in "gl_in"
               OpName %color_in "color_in"
               OpName %gl_PerVertex_out "gl_PerVertex"
               OpMemberName %gl_PerVertex_out 0 "gl_Position"
               OpName %gl_out "gl_out"
               OpName %outColor "outColor"
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %color_in Location 2
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %outColor Location 0
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
       %uint = OpTypeInt 32 0
     %uint_3 = OpConstant %uint 3
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
%ptr_in_v4float = OpTypePointer Input %v4float
%arr_color_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_color = OpTypePointer Input %arr_color_in
   %color_in = OpVariable %ptr_in_arr_color Input
%ptr_in_v3float = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_out_v3float = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3float Output
  %float_1_0 = OpConstant %float 1.0
  %float_3_0 = OpConstant %float 3.0
%float_0_08 = OpConstant %float 0.08
  %float_0_0 = OpConstant %float 0.0
%ptr_fn_v4float = OpTypePointer Function %v4float
%ptr_fn_v3float = OpTypePointer Function %v3float
       %main = OpFunction %void None %3
          %5 = OpLabel

               ; Load the 3 vertex positions
         %p0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0_ptr
         %p1_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %p1_ptr
         %p2_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
         %p2 = OpLoad %v4float %p2_ptr

               ; Compute face center: center = (p0 + p1 + p2) / 3
         %sum01 = OpFAdd %v4float %p0 %p1
         %sum012 = OpFAdd %v4float %sum01 %p2
         %center = OpVectorTimesScalar %v4float %sum012 %float_1_0
         %center_div3 = OpVectorTimesScalar %v4float %center %float_1_0
         %float_inv3 = OpFDiv %float %float_1_0 %float_3_0
         %center_final = OpVectorTimesScalar %v4float %sum012 %float_inv3

               ; Extract xyz from positions for edge computation
         %p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
         %p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
         %p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

               ; Compute edges: e1 = p1 - p0, e2 = p2 - p0
         %e1 = OpFSub %v3float %p1_xyz %p0_xyz
         %e2 = OpFSub %v3float %p2_xyz %p0_xyz

               ; Compute face normal: N = normalize(cross(e1, e2))
         %cross_result = OpExtInst %v3float %1 Cross %e1 %e2
         %normal = OpExtInst %v3float %1 Normalize %cross_result

               ; Compute offset: offset = N * 0.08
         %offset = OpVectorTimesScalar %v3float %normal %float_0_08

               ; Create vec4(offset, 0) for adding to positions
         %offset_v4 = OpCompositeConstruct %v4float %offset %float_0_0

               ; Emit vertex 0
         %out_pos0 = OpFAdd %v4float %p0 %offset_v4
         %out_ptr0 = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %out_ptr0 %out_pos0
         %color0_ptr = OpAccessChain %ptr_in_v3float %color_in %int_0
         %color0 = OpLoad %v3float %color0_ptr
               OpStore %outColor %color0
               OpEmitVertex

               ; Emit vertex 1
         %out_pos1 = OpFAdd %v4float %p1 %offset_v4
         %out_ptr1 = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %out_ptr1 %out_pos1
         %color1_ptr = OpAccessChain %ptr_in_v3float %color_in %int_1
         %color1 = OpLoad %v3float %color1_ptr
               OpStore %outColor %color1
               OpEmitVertex

               ; Emit vertex 2
         %out_pos2 = OpFAdd %v4float %p2 %offset_v4
         %out_ptr2 = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %out_ptr2 %out_pos2
         %color2_ptr = OpAccessChain %ptr_in_v3float %color_in %int_2
         %color2 = OpLoad %v3float %color2_ptr
               OpStore %outColor %color2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
