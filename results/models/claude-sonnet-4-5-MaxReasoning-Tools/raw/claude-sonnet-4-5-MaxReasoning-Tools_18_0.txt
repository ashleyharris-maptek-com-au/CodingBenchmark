// Stoer-Wagner algorithm for minimum cut
class MinCutSolver {
private:
    int n;
    vector<vector<int>> adj;

    struct Edge {
        int u, v;
    };

    int best_cut;
    vector<Edge> best_cut_edges;

public:
    MinCutSolver(int nodes) : n(nodes), adj(nodes, vector<int>(nodes, 0)), best_cut(numeric_limits<int>::max()) {}

    void addEdge(int u, int v) {
        adj[u][v]++;
        adj[v][u]++;
    }

    pair<int, int> minimumCutPhase(vector<bool>& merged) {
        vector<int> weight(n, 0);
        vector<bool> added(n, false);

        int prev = -1, last = -1;

        // Find first unmerged node
        for (int i = 0; i < n; i++) {
            if (!merged[i]) {
                prev = i;
                break;
            }
        }

        for (int phase = 0; phase < n; phase++) {
            if (merged[phase]) continue;

            int maxWeight = -1;
            int maxNode = -1;

            // Find the node with maximum weight to the current set
            for (int i = 0; i < n; i++) {
                if (!merged[i] && !added[i] && weight[i] > maxWeight) {
                    maxWeight = weight[i];
                    maxNode = i;
                }
            }

            if (maxNode == -1) break;

            added[maxNode] = true;
            prev = last;
            last = maxNode;

            // Update weights
            for (int i = 0; i < n; i++) {
                if (!merged[i] && !added[i]) {
                    weight[i] += adj[maxNode][i];
                }
            }
        }

        return {prev, last};
    }

    void stoerWagner() {
        vector<bool> merged(n, false);

        for (int phase = 0; phase < n - 1; phase++) {
            auto [s, t] = minimumCutPhase(merged);

            if (s == -1 || t == -1) break;

            // Calculate cut weight
            int cutWeight = 0;
            for (int i = 0; i < n; i++) {
                if (!merged[i]) {
                    cutWeight += adj[t][i];
                }
            }

            if (cutWeight < best_cut) {
                best_cut = cutWeight;
                // Store which nodes are on t's side
                vector<bool> t_side(n, false);
                t_side[t] = true;

                // Track the cut edges
                best_cut_edges.clear();
                for (int i = 0; i < n; i++) {
                    if (!merged[i] && t_side[i]) {
                        for (int j = 0; j < n; j++) {
                            if (!merged[j] && !t_side[j] && adj[i][j] > 0) {
                                for (int k = 0; k < adj[i][j]; k++) {
                                    best_cut_edges.push_back({i, j});
                                }
                            }
                        }
                    }
                }
            }

            // Merge t into s
            merged[t] = true;
            for (int i = 0; i < n; i++) {
                adj[s][i] += adj[t][i];
                adj[i][s] += adj[i][t];
            }
            adj[s][t] = 0;
            adj[t][s] = 0;
        }
    }

    int getMinCut() {
        return best_cut;
    }

    vector<Edge> getCutEdges() {
        return best_cut_edges;
    }
};

// Fast BFS-based approach for large sparse graphs
class FastMinCut {
private:
    int n, m;
    vector<vector<pair<int, int>>> graph; // adjacency list with edge indices
    vector<pair<int, int>> edges;

    int bfsMinCut(int source, const vector<bool>& available_edges) {
        vector<bool> visited(n, false);
        queue<int> q;
        q.push(source);
        visited[source] = true;
        int reachable = 1;

        while (!q.empty()) {
            int u = q.front();
            q.pop();

            for (auto [v, edge_idx] : graph[u]) {
                if (!visited[v] && available_edges[edge_idx]) {
                    visited[v] = true;
                    q.push(v);
                    reachable++;
                }
            }
        }

        return reachable;
    }

public:
    FastMinCut(int nodes, int num_edges) : n(nodes), m(num_edges), graph(nodes) {}

    void addEdge(int u, int v, int idx) {
        edges.push_back({u, v});
        graph[u].push_back({v, idx});
        graph[v].push_back({u, idx});
    }

    vector<pair<int, int>> solve() {
        // Try removing small sets of edges and check connectivity
        vector<bool> available(m, true);
        vector<pair<int, int>> result;

        // For very large graphs, use a heuristic: find bridges and articulation edges
        // For now, use a simple greedy approach

        // Count degree for each node
        vector<int> degree(n, 0);
        for (int i = 0; i < m; i++) {
            degree[edges[i].first]++;
            degree[edges[i].second]++;
        }

        // Find minimum degree nodes
        int min_degree = n;
        int min_node = 0;
        for (int i = 0; i < n; i++) {
            if (degree[i] < min_degree) {
                min_degree = degree[i];
                min_node = i;
            }
        }

        // Remove all edges connected to the minimum degree node
        for (auto [v, edge_idx] : graph[min_node]) {
            result.push_back(edges[edge_idx]);
        }

        return result;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    // Adaptive strategy based on graph size
    long long graph_size = (long long)n * n * 4; // Rough memory estimate for adjacency matrix
    long long available_memory = 1LL << 30; // 1GB threshold

    if (graph_size < available_memory && n <= 1000) {
        // Small to medium graph - use Stoer-Wagner (optimal)
        MinCutSolver solver(n);

        for (int i = 0; i < m; i++) {
            int u, v;
            cin >> u >> v;
            solver.addEdge(u, v);
        }

        solver.stoerWagner();

        cout << solver.getMinCut() << "\n";
        for (auto edge : solver.getCutEdges()) {
            cout << edge.u << " " << edge.v << "\n";
        }
    } else {
        // Large graph - use faster heuristic approach
        FastMinCut solver(n, m);

        for (int i = 0; i < m; i++) {
            int u, v;
            cin >> u >> v;
            solver.addEdge(u, v, i);
        }

        auto result = solver.solve();

        cout << result.size() << "\n";
        for (auto [u, v] : result) {
            cout << u << " " << v << "\n";
        }
    }

    return 0;
}
