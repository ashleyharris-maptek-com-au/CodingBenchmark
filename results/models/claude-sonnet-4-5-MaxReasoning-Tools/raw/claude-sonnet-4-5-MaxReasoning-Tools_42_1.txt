OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

; Types
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
    %v3float = OpTypeVector %float 3
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %uint_1 = OpConstant %uint 1

; Constants
 %float_0_05 = OpConstant %float 0.05

; Input types
%gl_PerVertex_in = OpTypeStruct %v4float
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
 %ptr_in_v4 = OpTypePointer Input %v4float
 %arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
 %ptr_in_v3 = OpTypePointer Input %v3float

; Output types
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; Variables
     %gl_in = OpVariable %ptr_in_arr_pv Input
  %color_in = OpVariable %ptr_in_arr_v3 Input
    %gl_out = OpVariable %ptr_out_pv Output
  %outColor = OpVariable %ptr_out_v3 Output

; Main function
       %main = OpFunction %void None %3
          %5 = OpLabel

; Read input position from gl_in[0].gl_Position
    %pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
        %pos = OpLoad %v4float %pos_ptr

; Read input color from color_in[0]
  %color_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
      %color = OpLoad %v3float %color_ptr

; Extract components from position
          %x = OpCompositeExtract %float %pos 0
          %y = OpCompositeExtract %float %pos 1
          %z = OpCompositeExtract %float %pos 2
          %w = OpCompositeExtract %float %pos 3

; Calculate offset positions (h = 0.05)
      %y_top = OpFAdd %float %y %float_0_05
    %x_right = OpFAdd %float %x %float_0_05
   %y_bottom = OpFSub %float %y %float_0_05
     %x_left = OpFSub %float %x %float_0_05

; Construct diamond vertex positions
    %pos_top = OpCompositeConstruct %v4float %x %y_top %z %w
  %pos_right = OpCompositeConstruct %v4float %x_right %y %z %w
   %pos_left = OpCompositeConstruct %v4float %x_left %y %z %w
 %pos_bottom = OpCompositeConstruct %v4float %x %y_bottom %z %w

; Get output position pointer
%gl_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Emit top vertex
              OpStore %gl_pos_ptr %pos_top
              OpStore %outColor %color
              OpEmitVertex

; Emit right vertex
              OpStore %gl_pos_ptr %pos_right
              OpStore %outColor %color
              OpEmitVertex

; Emit left vertex
              OpStore %gl_pos_ptr %pos_left
              OpStore %outColor %color
              OpEmitVertex

; Emit bottom vertex
              OpStore %gl_pos_ptr %pos_bottom
              OpStore %outColor %color
              OpEmitVertex

; End primitive
              OpEndPrimitive

              OpReturn
              OpFunctionEnd
