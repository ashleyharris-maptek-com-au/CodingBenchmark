#include <iostream>
#include <vector>
#include <string>
#include <unordered_set>
#include <unordered_map>
#include <algorithm>
#include <climits>

using namespace std;

typedef unsigned long long ull;
const ull BASE = 257;
const ull MOD = 1000000007;

// Direct string comparison for smaller cases
bool hasCommonSubstringDirect(const vector<string>& strings, int len, string& result) {
    if (len == 0) {
        result = "";
        return true;
    }
    if (strings.empty() || strings[0].length() < len) return false;

    unordered_set<string> common;
    const string& first = strings[0];

    for (size_t i = 0; i + len <= first.length(); i++) {
        common.insert(first.substr(i, len));
    }

    for (size_t str_idx = 1; str_idx < strings.size(); str_idx++) {
        const string& s = strings[str_idx];
        if (s.length() < len) return false;

        unordered_set<string> current;
        for (size_t i = 0; i + len <= s.length(); i++) {
            string sub = s.substr(i, len);
            if (common.count(sub)) {
                current.insert(sub);
            }
        }

        common = move(current);
        if (common.empty()) return false;
    }

    if (!common.empty()) {
        result = *common.begin();
        return true;
    }

    return false;
}

// Rolling hash with verification for larger cases
bool hasCommonSubstringHash(const vector<string>& strings, int len, string& result) {
    if (len == 0) {
        result = "";
        return true;
    }
    if (strings.empty() || strings[0].length() < len) return false;

    ull power = 1;
    for (int i = 0; i < len; i++) {
        power = (power * BASE) % MOD;
    }

    const string& first = strings[0];
    unordered_map<ull, vector<int>> hash_to_pos;

    ull hash = 0;
    for (int i = 0; i < len; i++) {
        hash = (hash * BASE + (unsigned char)first[i]) % MOD;
    }
    hash_to_pos[hash].push_back(0);

    for (size_t i = len; i < first.length(); i++) {
        hash = (hash + MOD - ((unsigned char)first[i - len] * power) % MOD) % MOD;
        hash = (hash * BASE + (unsigned char)first[i]) % MOD;
        hash_to_pos[hash].push_back(i - len + 1);
    }

    for (size_t str_idx = 1; str_idx < strings.size(); str_idx++) {
        const string& s = strings[str_idx];
        if (s.length() < len) return false;

        unordered_map<ull, vector<int>> new_hash_to_pos;

        hash = 0;
        for (int i = 0; i < len; i++) {
            hash = (hash * BASE + (unsigned char)s[i]) % MOD;
        }

        if (hash_to_pos.count(hash)) {
            string current_sub = s.substr(0, len);
            for (int pos : hash_to_pos[hash]) {
                if (first.substr(pos, len) == current_sub) {
                    new_hash_to_pos[hash] = hash_to_pos[hash];
                    break;
                }
            }
        }

        for (size_t i = len; i < s.length(); i++) {
            hash = (hash + MOD - ((unsigned char)s[i - len] * power) % MOD) % MOD;
            hash = (hash * BASE + (unsigned char)s[i]) % MOD;

            if (hash_to_pos.count(hash) && !new_hash_to_pos.count(hash)) {
                string current_sub = s.substr(i - len + 1, len);
                for (int pos : hash_to_pos[hash]) {
                    if (first.substr(pos, len) == current_sub) {
                        new_hash_to_pos[hash] = hash_to_pos[hash];
                        break;
                    }
                }
            }
        }

        hash_to_pos = move(new_hash_to_pos);
        if (hash_to_pos.empty()) return false;
    }

    if (!hash_to_pos.empty()) {
        int pos = hash_to_pos.begin()->second[0];
        result = first.substr(pos, len);
        return true;
    }

    return false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    cin.ignore();

    vector<string> strings(n);
    size_t min_len = SIZE_MAX;
    size_t total_len = 0;

    for (int i = 0; i < n; i++) {
        getline(cin, strings[i]);
        if (i == 0 || strings[i].length() < min_len) {
            min_len = strings[i].length();
        }
        total_len += strings[i].length();
    }

    if (n == 0 || min_len == 0 || min_len == SIZE_MAX) {
        cout << endl;
        return 0;
    }

    size_t left = 0, right = min_len;
    string result = "";

    while (left <= right) {
        size_t mid = left + (right - left) / 2;
        string candidate;

        bool found;
        // Adaptive: use direct method for small substrings, hash method for large
        if (mid < 100 || total_len < 1000000) {
            found = hasCommonSubstringDirect(strings, mid, candidate);
        } else {
            found = hasCommonSubstringHash(strings, mid, candidate);
        }

        if (found) {
            result = candidate;
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }

    cout << result << endl;

    return 0;
}
