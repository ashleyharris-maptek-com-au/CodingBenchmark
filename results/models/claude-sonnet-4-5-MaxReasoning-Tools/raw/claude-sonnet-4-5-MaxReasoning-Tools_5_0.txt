use std::io::{self, BufRead};
use std::collections::VecDeque;

#[derive(Clone)]
struct Grid {
    width: usize,
    height: usize,
    obstacles: Vec<bool>,
    target_cells: usize,
    require_cycle: bool,
}

impl Grid {
    fn new(width: usize, height: usize, require_cycle: bool) -> Self {
        let size = width * height;
        Grid {
            width,
            height,
            obstacles: vec![false; size],
            target_cells: size,
            require_cycle,
        }
    }

    #[inline]
    fn idx(&self, x: usize, y: usize) -> usize {
        y * self.width + x
    }

    #[inline]
    fn coords(&self, idx: usize) -> (usize, usize) {
        (idx % self.width, idx / self.width)
    }

    fn add_obstacle(&mut self, x: usize, y: usize) {
        let idx = self.idx(x, y);
        if !self.obstacles[idx] {
            self.obstacles[idx] = true;
            self.target_cells -= 1;
        }
    }

    fn neighbors(&self, x: usize, y: usize) -> Vec<(usize, usize)> {
        let mut result = Vec::with_capacity(4);
        if x > 0 { result.push((x - 1, y)); }
        if x + 1 < self.width { result.push((x + 1, y)); }
        if y > 0 { result.push((x, y - 1)); }
        if y + 1 < self.height { result.push((x, y + 1)); }
        result
    }

    fn count_unvisited_neighbors(&self, x: usize, y: usize, visited: &[bool]) -> usize {
        self.neighbors(x, y)
            .iter()
            .filter(|&&(nx, ny)| {
                let idx = self.idx(nx, ny);
                !self.obstacles[idx] && !visited[idx]
            })
            .count()
    }

    fn is_connected(&self, visited: &[bool]) -> bool {
        let mut to_visit = VecDeque::new();
        let mut temp_visited = vec![false; visited.len()];

        // Find first unvisited non-obstacle cell
        let start = (0..visited.len())
            .find(|&i| !visited[i] && !self.obstacles[i]);

        let start = match start {
            Some(s) => s,
            None => return true, // All cells visited
        };

        to_visit.push_back(start);
        temp_visited[start] = true;
        let mut count = 1;

        while let Some(idx) = to_visit.pop_front() {
            let (x, y) = self.coords(idx);
            for (nx, ny) in self.neighbors(x, y) {
                let nidx = self.idx(nx, ny);
                if !visited[nidx] && !self.obstacles[nidx] && !temp_visited[nidx] {
                    temp_visited[nidx] = true;
                    to_visit.push_back(nidx);
                    count += 1;
                }
            }
        }

        // Check if all unvisited non-obstacle cells are connected
        let expected = (0..visited.len())
            .filter(|&i| !visited[i] && !self.obstacles[i])
            .count();

        count == expected
    }

    fn solve(&self) -> Option<Vec<(usize, usize)>> {
        if self.obstacles[0] {
            return None; // Start is blocked
        }

        let mut path = Vec::with_capacity(self.target_cells);
        let mut visited = vec![false; self.width * self.height];

        path.push((0, 0));
        visited[0] = true;

        if self.backtrack(&mut path, &mut visited) {
            Some(path)
        } else {
            None
        }
    }

    fn backtrack(&self, path: &mut Vec<(usize, usize)>, visited: &mut [bool]) -> bool {
        if path.len() == self.target_cells {
            if self.require_cycle {
                let (last_x, last_y) = path[path.len() - 1];
                return self.neighbors(last_x, last_y)
                    .iter()
                    .any(|&(nx, ny)| nx == 0 && ny == 0);
            }
            return true;
        }

        let (curr_x, curr_y) = path[path.len() - 1];

        // Get neighbors and sort by heuristic (Warnsdorff's rule)
        let mut neighbors: Vec<_> = self.neighbors(curr_x, curr_y)
            .into_iter()
            .filter(|&(nx, ny)| {
                let idx = self.idx(nx, ny);
                !self.obstacles[idx] && !visited[idx]
            })
            .collect();

        if neighbors.is_empty() {
            return false;
        }

        // Sort by number of unvisited neighbors (ascending)
        neighbors.sort_by_key(|&(nx, ny)| {
            self.count_unvisited_neighbors(nx, ny, visited)
        });

        for (nx, ny) in neighbors {
            let nidx = self.idx(nx, ny);

            // Pruning: check for dead ends and connectivity
            visited[nidx] = true;

            // Check if this move creates an unreachable cell
            let mut is_valid = true;

            // Dead-end check: if a neighbor of current has only one unvisited neighbor
            // and we're not at the last cell, it's a dead end
            if path.len() + 1 < self.target_cells {
                for (nnx, nny) in self.neighbors(curr_x, curr_y) {
                    if nnx == nx && nny == ny {
                        continue;
                    }
                    let nnidx = self.idx(nnx, nny);
                    if !self.obstacles[nnidx] && !visited[nnidx] {
                        let count = self.count_unvisited_neighbors(nnx, nny, visited);
                        if count == 0 {
                            is_valid = false;
                            break;
                        }
                    }
                }
            }

            // For larger grids or later in the search, check connectivity
            if is_valid && path.len() > self.target_cells / 2 {
                if !self.is_connected(visited) {
                    is_valid = false;
                }
            }

            if is_valid {
                path.push((nx, ny));
                if self.backtrack(path, visited) {
                    return true;
                }
                path.pop();
            }

            visited[nidx] = false;
        }

        false
    }
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read first line
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    let width = parts[0];
    let height = parts[1];
    let obstacle_count = parts[2];
    let require_cycle = parts[3] == 1;

    let mut grid = Grid::new(width, height, require_cycle);

    // Read obstacles
    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<usize> = line
            .split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        grid.add_obstacle(coords[0], coords[1]);
    }

    // Solve
    match grid.solve() {
        Some(path) => {
            println!("{}", path.len());
            for (x, y) in path {
                println!("{} {}", x, y);
            }
        }
        None => {
            println!("0");
        }
    }
}
