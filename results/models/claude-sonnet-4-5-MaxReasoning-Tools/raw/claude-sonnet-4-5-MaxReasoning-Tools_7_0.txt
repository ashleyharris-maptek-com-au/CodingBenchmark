using namespace std;

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x, double y, double z) : x(x), y(y), z(z) {}
    Vec3 operator+(const Vec3& v) const { return Vec3(x+v.x, y+v.y, z+v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x-v.x, y-v.y, z-v.z); }
    Vec3 operator*(double s) const { return Vec3(x*s, y*s, z*s); }
    Vec3 operator/(double s) const { return Vec3(x/s, y/s, z/s); }
    double dot(const Vec3& v) const { return x*v.x + y*v.y + z*v.z; }
    Vec3 cross(const Vec3& v) const { return Vec3(y*v.z-z*v.y, z*v.x-x*v.z, x*v.y-y*v.x); }
    double length() const { return sqrt(x*x + y*y + z*z); }
    Vec3 normalize() const { double len = length(); return len > 1e-12 ? (*this)/len : Vec3(0,0,0); }
};

struct Triangle {
    int a, b, c;
};

struct AABB {
    Vec3 min, max;
    AABB() : min(Vec3(1e100, 1e100, 1e100)), max(Vec3(-1e100, -1e100, -1e100)) {}
    void expand(const Vec3& p) {
        min.x = std::min(min.x, p.x); min.y = std::min(min.y, p.y); min.z = std::min(min.z, p.z);
        max.x = std::max(max.x, p.x); max.y = std::max(max.y, p.y); max.z = std::max(max.z, p.z);
    }
    void expand(const AABB& b) {
        min.x = std::min(min.x, b.min.x); min.y = std::min(min.y, b.min.y); min.z = std::min(min.z, b.min.z);
        max.x = std::max(max.x, b.max.x); max.y = std::max(max.y, b.max.y); max.z = std::max(max.z, b.max.z);
    }
    bool intersects(const AABB& other) const {
        return !(max.x < other.min.x || min.x > other.max.x ||
                 max.y < other.min.y || min.y > other.max.y ||
                 max.z < other.min.z || min.z > other.max.z);
    }
    bool containsPoint(const Vec3& p) const {
        return p.x >= min.x && p.x <= max.x &&
               p.y >= min.y && p.y <= max.y &&
               p.z >= min.z && p.z <= max.z;
    }
    Vec3 center() const { return (min + max) * 0.5; }
    double surfaceArea() const {
        Vec3 d = max - min;
        return 2.0 * (d.x * d.y + d.y * d.z + d.z * d.x);
    }
};

class Mesh {
public:
    vector<Vec3> vertices;
    vector<Triangle> faces;

    AABB computeAABB(int faceIdx) const {
        AABB box;
        const Triangle& f = faces[faceIdx];
        box.expand(vertices[f.a]);
        box.expand(vertices[f.b]);
        box.expand(vertices[f.c]);
        return box;
    }

    Vec3 faceNormal(int faceIdx) const {
        const Triangle& f = faces[faceIdx];
        Vec3 v0 = vertices[f.a];
        Vec3 v1 = vertices[f.b];
        Vec3 v2 = vertices[f.c];
        return (v1 - v0).cross(v2 - v0).normalize();
    }

    Vec3 faceCenter(int faceIdx) const {
        const Triangle& f = faces[faceIdx];
        return (vertices[f.a] + vertices[f.b] + vertices[f.c]) / 3.0;
    }
};

// BVH Node for acceleration
struct BVHNode {
    AABB bbox;
    unique_ptr<BVHNode> left, right;
    vector<int> faceIndices;
    bool isLeaf;

    BVHNode() : isLeaf(false) {}
};

class BVH {
public:
    unique_ptr<BVHNode> root;
    const Mesh* mesh;

    BVH(const Mesh* m) : mesh(m) {
        vector<int> allFaces(mesh->faces.size());
        for (size_t i = 0; i < mesh->faces.size(); i++) {
            allFaces[i] = i;
        }
        root = build(allFaces);
    }

private:
    unique_ptr<BVHNode> build(vector<int>& faceIndices) {
        auto node = make_unique<BVHNode>();

        // Compute bounding box
        for (int idx : faceIndices) {
            node->bbox.expand(mesh->computeAABB(idx));
        }

        // Leaf node threshold
        if (faceIndices.size() <= 8) {
            node->isLeaf = true;
            node->faceIndices = faceIndices;
            return node;
        }

        // Find best split axis (longest axis)
        Vec3 extent = node->bbox.max - node->bbox.min;
        int axis = 0;
        if (extent.y > extent.x) axis = 1;
        if (extent.z > extent.y && extent.z > extent.x) axis = 2;

        // Sort by centroid along axis
        sort(faceIndices.begin(), faceIndices.end(), [&](int a, int b) {
            Vec3 ca = mesh->faceCenter(a);
            Vec3 cb = mesh->faceCenter(b);
            if (axis == 0) return ca.x < cb.x;
            if (axis == 1) return ca.y < cb.y;
            return ca.z < cb.z;
        });

        // Split in middle
        size_t mid = faceIndices.size() / 2;
        vector<int> leftFaces(faceIndices.begin(), faceIndices.begin() + mid);
        vector<int> rightFaces(faceIndices.begin() + mid, faceIndices.end());

        node->left = build(leftFaces);
        node->right = build(rightFaces);
        return node;
    }
};

// Ray-triangle intersection using Möller-Trumbore
bool rayTriangleIntersect(const Vec3& origin, const Vec3& dir,
                          const Vec3& v0, const Vec3& v1, const Vec3& v2,
                          double& t) {
    const double eps = 1e-9;
    Vec3 e1 = v1 - v0;
    Vec3 e2 = v2 - v0;
    Vec3 pvec = dir.cross(e2);
    double det = e1.dot(pvec);

    if (fabs(det) < eps) return false;

    double invDet = 1.0 / det;
    Vec3 tvec = origin - v0;
    double u = tvec.dot(pvec) * invDet;
    if (u < -eps || u > 1.0 + eps) return false;

    Vec3 qvec = tvec.cross(e1);
    double v = dir.dot(qvec) * invDet;
    if (v < -eps || u + v > 1.0 + eps) return false;

    t = e2.dot(qvec) * invDet;
    return t > eps;
}

// Ray-AABB intersection
bool rayAABBIntersect(const Vec3& origin, const Vec3& dir, const AABB& bbox) {
    double tmin = -1e100, tmax = 1e100;

    for (int i = 0; i < 3; i++) {
        double o = (i == 0) ? origin.x : (i == 1 ? origin.y : origin.z);
        double d = (i == 0) ? dir.x : (i == 1 ? dir.y : dir.z);
        double bmin = (i == 0) ? bbox.min.x : (i == 1 ? bbox.min.y : bbox.min.z);
        double bmax = (i == 0) ? bbox.max.x : (i == 1 ? bbox.max.y : bbox.max.z);

        if (fabs(d) > 1e-9) {
            double t1 = (bmin - o) / d;
            double t2 = (bmax - o) / d;
            if (t1 > t2) swap(t1, t2);
            tmin = max(tmin, t1);
            tmax = min(tmax, t2);
            if (tmin > tmax) return false;
        } else {
            if (o < bmin || o > bmax) return false;
        }
    }
    return tmax > 0;
}

// Count ray intersections with BVH
int countRayIntersections(const Vec3& origin, const Vec3& dir,
                          const BVHNode* node, const Mesh* mesh) {
    if (!node) return 0;

    if (!rayAABBIntersect(origin, dir, node->bbox)) {
        return 0;
    }

    if (node->isLeaf) {
        int count = 0;
        for (int idx : node->faceIndices) {
            const Triangle& f = mesh->faces[idx];
            double t;
            if (rayTriangleIntersect(origin, dir,
                                     mesh->vertices[f.a],
                                     mesh->vertices[f.b],
                                     mesh->vertices[f.c], t)) {
                count++;
            }
        }
        return count;
    }

    return countRayIntersections(origin, dir, node->left.get(), mesh) +
           countRayIntersections(origin, dir, node->right.get(), mesh);
}

// Point-in-mesh test using BVH-accelerated ray casting
bool pointInMesh(const Vec3& p, const BVH& bvh) {
    Vec3 rayDir(1.0, 0.0, 0.0);
    int intersections = countRayIntersections(p, rayDir, bvh.root.get(), bvh.mesh);
    return (intersections % 2) == 1;
}

// CSG Union with BVH acceleration
void computeUnion(const Mesh& meshA, const Mesh& meshB, Mesh& result) {
    // Build BVH for both meshes
    BVH bvhA(&meshA);
    BVH bvhB(&meshB);

    // Collect faces from A that are outside B
    vector<int> facesFromA;
    for (size_t i = 0; i < meshA.faces.size(); i++) {
        Vec3 center = meshA.faceCenter(i);
        if (!pointInMesh(center, bvhB)) {
            facesFromA.push_back(i);
        }
    }

    // Collect faces from B that are outside A
    vector<int> facesFromB;
    for (size_t i = 0; i < meshB.faces.size(); i++) {
        Vec3 center = meshB.faceCenter(i);
        if (!pointInMesh(center, bvhA)) {
            facesFromB.push_back(i);
        }
    }

    // Build vertex map for deduplication
    unordered_map<int, int> vertexMapA;
    unordered_map<int, int> vertexMapB;

    // Add vertices and faces from A
    for (int faceIdx : facesFromA) {
        const Triangle& f = meshA.faces[faceIdx];
        int indices[3] = {f.a, f.b, f.c};
        int newIndices[3];

        for (int j = 0; j < 3; j++) {
            if (vertexMapA.find(indices[j]) == vertexMapA.end()) {
                vertexMapA[indices[j]] = result.vertices.size();
                result.vertices.push_back(meshA.vertices[indices[j]]);
            }
            newIndices[j] = vertexMapA[indices[j]];
        }

        result.faces.push_back({newIndices[0], newIndices[1], newIndices[2]});
    }

    // Add vertices and faces from B
    for (int faceIdx : facesFromB) {
        const Triangle& f = meshB.faces[faceIdx];
        int indices[3] = {f.a, f.b, f.c};
        int newIndices[3];

        for (int j = 0; j < 3; j++) {
            if (vertexMapB.find(indices[j]) == vertexMapB.end()) {
                vertexMapB[indices[j]] = result.vertices.size();
                result.vertices.push_back(meshB.vertices[indices[j]]);
            }
            newIndices[j] = vertexMapB[indices[j]];
        }

        result.faces.push_back({newIndices[0], newIndices[1], newIndices[2]});
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA, FA, VB, FB;
    cin >> VA >> FA >> VB >> FB;

    Mesh meshA, meshB;

    // Read mesh A
    meshA.vertices.resize(VA);
    for (int i = 0; i < VA; i++) {
        cin >> meshA.vertices[i].x >> meshA.vertices[i].y >> meshA.vertices[i].z;
    }
    meshA.faces.resize(FA);
    for (int i = 0; i < FA; i++) {
        cin >> meshA.faces[i].a >> meshA.faces[i].b >> meshA.faces[i].c;
    }

    // Read mesh B
    meshB.vertices.resize(VB);
    for (int i = 0; i < VB; i++) {
        cin >> meshB.vertices[i].x >> meshB.vertices[i].y >> meshB.vertices[i].z;
    }
    meshB.faces.resize(FB);
    for (int i = 0; i < FB; i++) {
        cin >> meshB.faces[i].a >> meshB.faces[i].b >> meshB.faces[i].c;
    }

    // Compute union
    Mesh result;
    computeUnion(meshA, meshB, result);

    // Output result
    cout << result.vertices.size() << " " << result.faces.size() << "\n";
    cout << fixed << setprecision(10);
    for (const Vec3& v : result.vertices) {
        cout << v.x << " " << v.y << " " << v.z << "\n";
    }
    for (const Triangle& f : result.faces) {
        cout << f.a << " " << f.b << " " << f.c << "\n";
    }

    return 0;
}
