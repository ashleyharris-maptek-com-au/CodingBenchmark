OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_uv %in_normal %out_color %ubo
OpExecutionMode %main OriginUpperLeft

; Decorations for inputs/outputs
OpDecorate %in_uv Location 2
OpDecorate %in_normal Location 1
OpDecorate %out_color Location 0

; Decorations for UBO
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4

%func_type = OpTypeFunction %void

; Constants
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_0_6 = OpConstant %float 0.6
%float_8 = OpConstant %float 8.0
%float_16 = OpConstant %float 16.0
%float_0_05 = OpConstant %float 0.05
%float_0_1 = OpConstant %float 0.1
%float_0_95 = OpConstant %float 0.95
%float_0_9 = OpConstant %float 0.9
%vec2_scale = OpConstantComposite %vec2 %float_8 %float_16
%vec3_brick = OpConstantComposite %vec3 0.7 0.2 0.1
%vec3_mortar = OpConstantComposite %vec3 0.7 0.7 0.7

; Variables
%in_uv = OpVariable %ptr_input_vec2 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%out_color = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_ubo Uniform

; Main function
%main = OpFunction %void None %func_type
%entry = OpLabel

; Load inputs
%uv = OpLoad %vec2 %in_uv
%normal_raw = OpLoad %vec3 %in_normal

; Normalize normal
%normal = OpExtInst %vec3 %1 Normalize %normal_raw

; Scale UV by (8, 16)
%scaled_uv = OpFMul %vec2 %uv %vec2_scale

; Extract components
%uv_x = OpCompositeExtract %float %scaled_uv 0
%uv_y = OpCompositeExtract %float %scaled_uv 1

; Get row index (floor of Y)
%row = OpExtInst %float %1 Floor %uv_y

; Check if row is odd: fract(row/2) >= 0.5
%row_half = OpFMul %float %row %float_0_5
%row_fract = OpExtInst %float %1 Fract %row_half
%is_odd = OpFOrdGreaterThanEqual %bool %row_fract %float_0_5

; Apply offset for odd rows
%offset = OpSelect %float %is_odd %float_0_5 %float_0

; Offset X coordinate
%uv_x_offset = OpFAdd %float %uv_x %offset

; Get fractional parts for brick detection
%brick_x = OpExtInst %float %1 Fract %uv_x_offset
%brick_y = OpExtInst %float %1 Fract %uv_y

; Check if in mortar region
; Mortar if brick_x < 0.05 OR brick_x > 0.95 OR brick_y < 0.1 OR brick_y > 0.9
%in_mortar_x_low = OpFOrdLessThan %bool %brick_x %float_0_05
%in_mortar_x_high = OpFOrdGreaterThan %bool %brick_x %float_0_95
%in_mortar_y_low = OpFOrdLessThan %bool %brick_y %float_0_1
%in_mortar_y_high = OpFOrdGreaterThan %bool %brick_y %float_0_9

%in_mortar_x = OpLogicalOr %bool %in_mortar_x_low %in_mortar_x_high
%in_mortar_y = OpLogicalOr %bool %in_mortar_y_low %in_mortar_y_high
%in_mortar = OpLogicalOr %bool %in_mortar_x %in_mortar_y

; Select base color (brick or mortar)
%base_color = OpSelect %vec3 %in_mortar %vec3_mortar %vec3_brick

; Load light position from UBO
%light_pos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%light_pos_4 = OpLoad %vec4 %light_pos_ptr
%light_pos = OpVectorShuffle %vec3 %light_pos_4 %light_pos_4 0 1 2

; Calculate light direction (we need worldPos, but it's not used in simple shading)
; For this shader, we'll use the light direction as just the normalized light position
%light_dir = OpExtInst %vec3 %1 Normalize %light_pos

; Calculate diffuse term: max(0, dot(normal, light_dir))
%ndotl = OpDot %float %normal %light_dir
%diffuse = OpExtInst %float %1 FMax %ndotl %float_0

; Apply shading factor: 0.4 + 0.6 * diffuse
%diffuse_scaled = OpFMul %float %diffuse %float_0_6
%shading = OpFAdd %float %float_0_4 %diffuse_scaled

; Apply shading to base color
%final_rgb = OpVectorTimesScalar %vec3 %base_color %shading

; Construct output color with alpha = 1.0
%final_color = OpCompositeConstruct %vec4 %final_rgb %float_1

; Store output
OpStore %out_color %final_color

OpReturn
OpFunctionEnd
