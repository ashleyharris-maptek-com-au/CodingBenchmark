#include <iostream>
#include <fstream>
#include <string>
#include <sstream>
#include <thread>
#include <atomic>
#include <mutex>
#include <cmath>
#include <algorithm>
#include <vector>
#include <chrono>

// State structure to hold lander state
struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel_remaining;
    bool valid = false;
};

// Global state variables with atomic/mutex protection
std::atomic<bool> running(true);
std::atomic<bool> state_ready(false);
std::mutex state_mutex;
State current_state;

// Mission parameters
struct Mission {
    double width, depth, height, gravity, max_thrust, initial_fuel;
    double start_x, start_y, start_z;
    double target_x, target_y, target_z;
    std::string terrain_file;
};

Mission mission;

// PID Controller class
class PIDController {
public:
    double kp, ki, kd;
    double integral, prev_error;
    double integral_limit;

    PIDController(double p, double i, double d, double limit = 100.0)
        : kp(p), ki(i), kd(d), integral(0), prev_error(0), integral_limit(limit) {}

    double update(double error, double dt) {
        integral += error * dt;
        integral = std::clamp(integral, -integral_limit, integral_limit);

        double derivative = (error - prev_error) / dt;
        prev_error = error;

        return kp * error + ki * integral + kd * derivative;
    }

    void reset() {
        integral = 0;
        prev_error = 0;
    }
};

// Input reader thread
void input_reader_thread() {
    while (running) {
        State new_state;
        if (!(std::cin >> new_state.x >> new_state.y >> new_state.z
                       >> new_state.vx >> new_state.vy >> new_state.vz
                       >> new_state.pitch >> new_state.yaw >> new_state.roll
                       >> new_state.pitch_rate >> new_state.yaw_rate >> new_state.roll_rate
                       >> new_state.fuel_remaining)) {
            running = false;
            break;
        }

        new_state.valid = true;

        {
            std::lock_guard<std::mutex> lock(state_mutex);
            current_state = new_state;
            state_ready = true;
        }
    }
}

// Main control loop
void control_loop() {
    // Wait for first state
    while (!state_ready && running) {
        std::this_thread::sleep_for(std::chrono::milliseconds(1));
    }

    // PID controllers for position and velocity
    PIDController pos_x_pid(0.3, 0.01, 0.8, 50.0);
    PIDController pos_y_pid(0.3, 0.01, 0.8, 50.0);
    PIDController pos_z_pid(0.5, 0.02, 1.2, 50.0);

    PIDController vel_x_pid(0.4, 0.05, 0.3, 20.0);
    PIDController vel_y_pid(0.4, 0.05, 0.3, 20.0);
    PIDController vel_z_pid(0.6, 0.05, 0.5, 20.0);

    auto last_time = std::chrono::steady_clock::now();

    while (running) {
        auto current_time = std::chrono::steady_clock::now();
        double dt = std::chrono::duration<double>(current_time - last_time).count();

        if (dt < 0.001) dt = 0.001; // Minimum dt
        if (dt > 0.2) dt = 0.1; // Cap dt for stability

        State state;
        {
            std::lock_guard<std::mutex> lock(state_mutex);
            state = current_state;
        }

        if (!state.valid) {
            std::this_thread::sleep_for(std::chrono::milliseconds(10));
            continue;
        }

        // Calculate position and velocity errors
        double pos_error_x = mission.target_x - state.x;
        double pos_error_y = mission.target_y - state.y;
        double pos_error_z = mission.target_z - state.z;

        double distance = std::sqrt(pos_error_x * pos_error_x +
                                   pos_error_y * pos_error_y +
                                   pos_error_z * pos_error_z);

        // Adaptive target velocity based on distance
        double speed_limit = std::min(10.0, distance * 0.5);
        speed_limit = std::max(2.0, speed_limit);

        double target_vx = 0.0, target_vy = 0.0, target_vz = 0.0;
        if (distance > 1.0) {
            double norm = std::sqrt(pos_error_x * pos_error_x + pos_error_y * pos_error_y + pos_error_z * pos_error_z);
            target_vx = (pos_error_x / norm) * speed_limit;
            target_vy = (pos_error_y / norm) * speed_limit;
            target_vz = (pos_error_z / norm) * speed_limit;
        }

        // In final approach, target zero velocity
        if (distance < 5.0) {
            target_vx = target_vy = target_vz = 0.0;
        }

        double vel_error_x = target_vx - state.vx;
        double vel_error_y = target_vy - state.vy;
        double vel_error_z = target_vz - state.vz;

        // Combined position and velocity control
        double control_x = pos_x_pid.update(pos_error_x, dt) + vel_x_pid.update(vel_error_x, dt);
        double control_y = pos_y_pid.update(pos_error_y, dt) + vel_y_pid.update(vel_error_y, dt);
        double control_z = pos_z_pid.update(pos_error_z, dt) + vel_z_pid.update(vel_error_z, dt);

        // Calculate desired thrust vector (in world frame)
        double desired_thrust_x = control_x;
        double desired_thrust_y = control_y;
        double desired_thrust_z = control_z + mission.gravity; // Compensate for gravity

        double total_desired = std::sqrt(desired_thrust_x * desired_thrust_x +
                                        desired_thrust_y * desired_thrust_y +
                                        desired_thrust_z * desired_thrust_z);

        // Calculate required pitch and yaw to point thrust in desired direction
        double desired_pitch = 0.0;
        double desired_yaw = 0.0;

        if (total_desired > 0.1) {
            // Normalize desired thrust
            double nx = desired_thrust_x / total_desired;
            double ny = desired_thrust_y / total_desired;
            double nz = desired_thrust_z / total_desired;

            // Calculate pitch: angle from vertical
            // pitch positive tilts forward (-X direction)
            desired_pitch = -std::asin(std::clamp(nx, -1.0, 1.0));

            // Calculate yaw from horizontal component
            desired_yaw = std::atan2(ny, std::sqrt(std::max(0.0, 1.0 - nx * nx - ny * ny)));

            // Clamp pitch to allowed range
            desired_pitch = std::clamp(desired_pitch, -M_PI/2.0, M_PI/2.0);
        }

        // Attitude control
        double pitch_error = desired_pitch - state.pitch;
        double yaw_error = desired_yaw - state.yaw;

        // Normalize yaw error to [-pi, pi]
        while (yaw_error > M_PI) yaw_error -= 2.0 * M_PI;
        while (yaw_error < -M_PI) yaw_error += 2.0 * M_PI;

        // Simple P controller for attitude
        double pitch_cmd = std::clamp(pitch_error * 5.0 - state.pitch_rate * 0.5, -1.0, 1.0);
        double yaw_cmd = std::clamp(yaw_error * 5.0 - state.yaw_rate * 0.5, -1.0, 1.0);
        double roll_cmd = std::clamp(-state.roll * 2.0 - state.roll_rate * 0.5, -1.0, 1.0);

        // Thrust magnitude
        double thrust = 0.0;
        if (state.fuel_remaining > 0.1) {
            thrust = std::clamp(total_desired / mission.max_thrust, 0.0, 1.0);

            // Reduce thrust when not aligned
            double cos_pitch = std::cos(state.pitch - desired_pitch);
            double cos_yaw = std::cos(state.yaw - desired_yaw);
            double alignment = std::min(cos_pitch, cos_yaw);
            if (alignment < 0.9) {
                thrust *= 0.5; // Reduce thrust when misaligned
            }
        }

        // Output command
        std::cout << thrust << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << std::endl;
        std::cout.flush();

        last_time = current_time;

        // Target 20Hz update rate
        std::this_thread::sleep_for(std::chrono::milliseconds(50));
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    // Read mission parameters
    std::cin >> mission.width >> mission.depth >> mission.height
             >> mission.gravity >> mission.max_thrust >> mission.initial_fuel;
    std::cin >> mission.start_x >> mission.start_y >> mission.start_z
             >> mission.target_x >> mission.target_y >> mission.target_z;

    std::cin.ignore(1000, '\n'); // Skip to next line
    std::getline(std::cin, mission.terrain_file);

    // Read "STATE" marker
    std::string state_marker;
    std::getline(std::cin, state_marker);

    // Start terrain loading in background (optional - we do minimal processing)
    std::thread terrain_thread([&]() {
        // We could load terrain here for collision detection, but for this
        // implementation we rely on the simulator to detect collisions.
        // For very large terrains, loading would be done in chunks or skipped.
    });

    // Start input reader thread
    std::thread reader(input_reader_thread);

    // Run control loop in main thread
    control_loop();

    // Cleanup
    running = false;
    if (reader.joinable()) reader.join();
    if (terrain_thread.joinable()) terrain_thread.join();

    return 0;
}
