import sys
import math

def distance(p1, p2):
    """Calculate Euclidean distance between two points."""
    return math.sqrt((p1[0] - p2[0]) ** 2 + (p1[1] - p2[1]) ** 2)

def nearest_neighbor_cvrp(depot, customers, capacity, max_vehicles):
    """
    Solve CVRP using a nearest neighbor heuristic.

    Args:
        depot: (x, y) coordinates of depot
        customers: list of (x, y, demand) for each customer
        capacity: vehicle capacity
        max_vehicles: maximum number of vehicles

    Returns:
        routes: list of routes, each route is a list of customer indices
        total_distance: total distance of all routes
    """
    n = len(customers)
    unvisited = set(range(n))
    routes = []

    while unvisited:
        route = []
        current_capacity = 0
        current_pos = depot

        while unvisited:
            # Find nearest unvisited customer that fits in current capacity
            best_customer = None
            best_distance = float('inf')

            for idx in unvisited:
                customer = customers[idx]
                demand = customer[2]

                # Check if customer fits in current route
                if current_capacity + demand <= capacity:
                    dist = distance(current_pos, (customer[0], customer[1]))
                    if dist < best_distance:
                        best_distance = dist
                        best_customer = idx

            # If no customer fits, end this route
            if best_customer is None:
                break

            # Add customer to route
            route.append(best_customer)
            unvisited.remove(best_customer)
            current_capacity += customers[best_customer][2]
            current_pos = (customers[best_customer][0], customers[best_customer][1])

        if route:
            routes.append(route)

    # Calculate total distance
    total_distance = 0
    for route in routes:
        # Distance from depot to first customer
        total_distance += distance(depot, (customers[route[0]][0], customers[route[0]][1]))

        # Distance between consecutive customers
        for i in range(len(route) - 1):
            c1 = customers[route[i]]
            c2 = customers[route[i + 1]]
            total_distance += distance((c1[0], c1[1]), (c2[0], c2[1]))

        # Distance from last customer back to depot
        last = customers[route[-1]]
        total_distance += distance((last[0], last[1]), depot)

    return routes, total_distance

def main():
    # Read input
    lines = sys.stdin.read().strip().split('\n')

    # Parse first line
    first_line = lines[0].split()
    n = int(first_line[0])
    k = int(first_line[1])
    capacity = int(first_line[2])

    # Parse depot
    depot_line = lines[1].split()
    depot = (float(depot_line[0]), float(depot_line[1]))

    # Parse customers
    customers = []
    for i in range(2, 2 + n):
        parts = lines[i].split()
        x = float(parts[0])
        y = float(parts[1])
        demand = int(parts[2])
        customers.append((x, y, demand))

    # Solve CVRP
    routes, total_distance = nearest_neighbor_cvrp(depot, customers, capacity, k)

    # Output results
    print(f"{total_distance:.6f}")
    print(len(routes))
    for route in routes:
        print(' '.join(map(str, route)))

if __name__ == "__main__":
    main()
