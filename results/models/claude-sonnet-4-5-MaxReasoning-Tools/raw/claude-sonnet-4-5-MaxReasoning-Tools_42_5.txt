OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 7
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
     %uint_2 = OpConstant %uint 2
     %uint_3 = OpConstant %uint 3

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_color_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_col = OpTypePointer Input %arr_color_in
   %color_in = OpVariable %ptr_in_arr_col Input
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

 %float_0_75 = OpConstant %float 0.75
%float_0_015 = OpConstant %float 0.015
 %float_0_05 = OpConstant %float 0.05
  %float_n1 = OpConstant %float -1.0
   %float_0 = OpConstant %float 0.0
   %float_1 = OpConstant %float 1.0

       %main = OpFunction %void None %3
          %5 = OpLabel

; Load p0 = gl_in[0].gl_Position
       %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0_ptr

; Load p1 = gl_in[1].gl_Position
       %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %p1_ptr

; Load c0 = color_in[0]
       %c0_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
         %c0 = OpLoad %v3float %c0_ptr

; Load c1 = color_in[1]
       %c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
         %c1 = OpLoad %v3float %c1_ptr

; Compute dir = p1 - p0
        %dir = OpFSub %v4float %p1 %p0

; Extract dir.x and dir.y for perpendicular computation
      %dir_x = OpCompositeExtract %float %dir 0
      %dir_y = OpCompositeExtract %float %dir 1

; Compute perpendicular: perp = (-dir.y, dir.x)
   %perp_x_raw = OpFNegate %float %dir_y
   %perp_y_raw = OpFCopy %float %dir_x
    %perp_raw = OpCompositeConstruct %v2float %perp_x_raw %perp_y_raw

; Normalize perpendicular
       %perp = OpExtInst %v2float %1 Normalize %perp_raw

; Compute mid = p0 + 0.75 * dir
   %dir_scaled = OpVectorTimesScalar %v4float %dir %float_0_75
        %mid = OpFAdd %v4float %p0 %dir_scaled

; Compute perp * 0.015 (shaft thickness)
%perp_shaft = OpVectorTimesScalar %v2float %perp %float_0_015

; Compute perp * 0.05 (arrowhead width)
%perp_arrow = OpVectorTimesScalar %v2float %perp %float_0_05

; Extract p0.xy for offset computation
      %p0_x = OpCompositeExtract %float %p0 0
      %p0_y = OpCompositeExtract %float %p0 1
      %p0_z = OpCompositeExtract %float %p0 2
      %p0_w = OpCompositeExtract %float %p0 3

; Extract mid.xy
     %mid_x = OpCompositeExtract %float %mid 0
     %mid_y = OpCompositeExtract %float %mid 1
     %mid_z = OpCompositeExtract %float %mid 2
     %mid_w = OpCompositeExtract %float %mid 3

; Extract perp_shaft components
%perp_shaft_x = OpCompositeExtract %float %perp_shaft 0
%perp_shaft_y = OpCompositeExtract %float %perp_shaft 1

; Extract perp_arrow components
%perp_arrow_x = OpCompositeExtract %float %perp_arrow 0
%perp_arrow_y = OpCompositeExtract %float %perp_arrow 1

; ===== EMIT SHAFT (4 vertices) =====
; Vertex 0: p0 - perp_shaft
   %v0_x = OpFSub %float %p0_x %perp_shaft_x
   %v0_y = OpFSub %float %p0_y %perp_shaft_y
     %v0 = OpCompositeConstruct %v4float %v0_x %v0_y %p0_z %p0_w
%v0_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %v0_out_ptr %v0
               OpStore %outColor %c0
               OpEmitVertex

; Vertex 1: p0 + perp_shaft
   %v1_x = OpFAdd %float %p0_x %perp_shaft_x
   %v1_y = OpFAdd %float %p0_y %perp_shaft_y
     %v1 = OpCompositeConstruct %v4float %v1_x %v1_y %p0_z %p0_w
               OpStore %v0_out_ptr %v1
               OpStore %outColor %c0
               OpEmitVertex

; Vertex 2: mid - perp_shaft
   %v2_x = OpFSub %float %mid_x %perp_shaft_x
   %v2_y = OpFSub %float %mid_y %perp_shaft_y
     %v2 = OpCompositeConstruct %v4float %v2_x %v2_y %mid_z %mid_w
               OpStore %v0_out_ptr %v2
               OpStore %outColor %c0
               OpEmitVertex

; Vertex 3: mid + perp_shaft
   %v3_x = OpFAdd %float %mid_x %perp_shaft_x
   %v3_y = OpFAdd %float %mid_y %perp_shaft_y
     %v3 = OpCompositeConstruct %v4float %v3_x %v3_y %mid_z %mid_w
               OpStore %v0_out_ptr %v3
               OpStore %outColor %c0
               OpEmitVertex

               OpEndPrimitive

; ===== EMIT ARROWHEAD (3 vertices) =====
; Vertex 0: mid + perp_arrow
   %a0_x = OpFAdd %float %mid_x %perp_arrow_x
   %a0_y = OpFAdd %float %mid_y %perp_arrow_y
     %a0 = OpCompositeConstruct %v4float %a0_x %a0_y %mid_z %mid_w
               OpStore %v0_out_ptr %a0
               OpStore %outColor %c1
               OpEmitVertex

; Vertex 1: mid - perp_arrow
   %a1_x = OpFSub %float %mid_x %perp_arrow_x
   %a1_y = OpFSub %float %mid_y %perp_arrow_y
     %a1 = OpCompositeConstruct %v4float %a1_x %a1_y %mid_z %mid_w
               OpStore %v0_out_ptr %a1
               OpStore %outColor %c1
               OpEmitVertex

; Vertex 2: p1 (tip)
               OpStore %v0_out_ptr %p1
               OpStore %outColor %c1
               OpEmitVertex

               OpEndPrimitive

               OpReturn
               OpFunctionEnd

          %1 = OpExtInstImport "GLSL.std.450"
