using System;
using System.Collections.Generic;
using System.Linq;

class ChinesePostman
{
    static void Main()
    {
        var input = Console.ReadLine().Split();
        int n = int.Parse(input[0]);
        int m = int.Parse(input[1]);

        // Build adjacency list
        var originalGraph = new List<(int to, int weight)>[n];
        var multiGraph = new List<(int to, int edgeId)>[n];

        for (int i = 0; i < n; i++)
        {
            originalGraph[i] = new List<(int, int)>();
            multiGraph[i] = new List<(int, int)>();
        }

        for (int edgeId = 0; edgeId < m; edgeId++)
        {
            var edge = Console.ReadLine().Split();
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            int w = int.Parse(edge[2]);

            originalGraph[u].Add((v, w));
            originalGraph[v].Add((u, w));
            multiGraph[u].Add((v, edgeId));
            multiGraph[v].Add((u, edgeId));
        }

        // Find vertices with odd degree
        var oddVertices = new List<int>();
        for (int i = 0; i < n; i++)
        {
            if (multiGraph[i].Count % 2 == 1)
                oddVertices.Add(i);
        }

        int nextEdgeId = m;

        // If there are odd-degree vertices, we need to match them
        if (oddVertices.Count > 0)
        {
            // Compute shortest paths between all pairs of odd vertices
            var distances = new Dictionary<(int, int), (int dist, List<int> path)>();

            foreach (int source in oddVertices)
            {
                var (dist, prev) = Dijkstra(originalGraph, n, source);

                foreach (int target in oddVertices)
                {
                    if (source < target)
                    {
                        var path = ReconstructPath(prev, source, target);
                        distances[(source, target)] = (dist[target], path);
                    }
                }
            }

            // Find minimum weight perfect matching
            List<(int, int)> matching;
            if (oddVertices.Count <= 20)
            {
                matching = MinWeightMatchingDP(oddVertices, distances);
            }
            else
            {
                matching = MinWeightMatchingGreedy(oddVertices, distances);
            }

            // Add matched paths to multigraph
            foreach (var (u, v) in matching)
            {
                var key = (Math.Min(u, v), Math.Max(u, v));
                var path = distances[key].path;

                for (int i = 0; i < path.Count - 1; i++)
                {
                    int from = path[i];
                    int to = path[i + 1];
                    multiGraph[from].Add((to, nextEdgeId));
                    multiGraph[to].Add((from, nextEdgeId));
                }
                nextEdgeId++;
            }
        }

        // Find Eulerian circuit starting from node 0
        var circuit = FindEulerianCircuit(multiGraph, n, 0);

        Console.WriteLine(string.Join(" ", circuit));
    }

    static (int[] dist, int[] prev) Dijkstra(List<(int to, int weight)>[] graph, int n, int start)
    {
        var dist = new int[n];
        var prev = new int[n];
        Array.Fill(dist, int.MaxValue / 2);
        Array.Fill(prev, -1);
        dist[start] = 0;

        var pq = new SortedSet<(int dist, int node)>();
        pq.Add((0, start));

        while (pq.Count > 0)
        {
            var (d, u) = pq.Min;
            pq.Remove(pq.Min);

            if (d > dist[u]) continue;

            foreach (var (v, w) in graph[u])
            {
                int newDist = dist[u] + w;
                if (newDist < dist[v])
                {
                    pq.Remove((dist[v], v));
                    dist[v] = newDist;
                    prev[v] = u;
                    pq.Add((dist[v], v));
                }
            }
        }

        return (dist, prev);
    }

    static List<int> ReconstructPath(int[] prev, int start, int end)
    {
        var path = new List<int>();
        int current = end;

        while (current != -1)
        {
            path.Add(current);
            if (current == start) break;
            current = prev[current];
        }

        path.Reverse();
        return path;
    }

    static List<(int, int)> MinWeightMatchingDP(List<int> vertices, Dictionary<(int, int), (int dist, List<int> path)> distances)
    {
        int k = vertices.Count;
        int fullMask = (1 << k) - 1;

        var dp = new int[1 << k];
        var parent = new (int prevMask, int idx1, int idx2)[1 << k];
        Array.Fill(dp, int.MaxValue / 2);
        dp[0] = 0;

        for (int mask = 0; mask < (1 << k); mask++)
        {
            if (dp[mask] >= int.MaxValue / 2) continue;

            // Find first unmatched vertex
            int first = -1;
            for (int i = 0; i < k; i++)
            {
                if ((mask & (1 << i)) == 0)
                {
                    first = i;
                    break;
                }
            }

            if (first == -1) continue;

            // Try matching with all other unmatched vertices
            for (int second = first + 1; second < k; second++)
            {
                if ((mask & (1 << second)) != 0) continue;

                int newMask = mask | (1 << first) | (1 << second);
                int u = vertices[first];
                int v = vertices[second];
                var key = (Math.Min(u, v), Math.Max(u, v));
                int weight = distances[key].dist;

                if (dp[mask] + weight < dp[newMask])
                {
                    dp[newMask] = dp[mask] + weight;
                    parent[newMask] = (mask, first, second);
                }
            }
        }

        // Reconstruct matching
        var matching = new List<(int, int)>();
        int currMask = fullMask;

        while (currMask > 0)
        {
            var (prevMask, idx1, idx2) = parent[currMask];
            matching.Add((vertices[idx1], vertices[idx2]));
            currMask = prevMask;
        }

        return matching;
    }

    static List<(int, int)> MinWeightMatchingGreedy(List<int> vertices, Dictionary<(int, int), (int dist, List<int> path)> distances)
    {
        var matching = new List<(int, int)>();
        var used = new HashSet<int>();

        // Create list of all pairs with distances
        var pairs = new List<(int u, int v, int dist)>();
        for (int i = 0; i < vertices.Count; i++)
        {
            for (int j = i + 1; j < vertices.Count; j++)
            {
                int u = vertices[i];
                int v = vertices[j];
                var key = (Math.Min(u, v), Math.Max(u, v));
                pairs.Add((u, v, distances[key].dist));
            }
        }

        // Sort by distance
        pairs.Sort((a, b) => a.dist.CompareTo(b.dist));

        // Greedily match closest pairs
        foreach (var (u, v, _) in pairs)
        {
            if (!used.Contains(u) && !used.Contains(v))
            {
                used.Add(u);
                used.Add(v);
                matching.Add((u, v));

                if (used.Count == vertices.Count)
                    break;
            }
        }

        return matching;
    }

    static List<int> FindEulerianCircuit(List<(int to, int edgeId)>[] graph, int n, int start)
    {
        var usedEdges = new HashSet<int>();
        var stack = new Stack<int>();
        var circuit = new List<int>();

        stack.Push(start);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            bool foundEdge = false;

            foreach (var (v, edgeId) in graph[u])
            {
                if (!usedEdges.Contains(edgeId))
                {
                    usedEdges.Add(edgeId);
                    stack.Push(v);
                    foundEdge = true;
                    break;
                }
            }

            if (!foundEdge)
            {
                circuit.Add(stack.Pop());
            }
        }

        circuit.Reverse();
        return circuit;
    }
}
