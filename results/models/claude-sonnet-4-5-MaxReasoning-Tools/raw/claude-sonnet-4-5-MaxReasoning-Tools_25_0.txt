using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

class AsteroidInterceptor
{
    const double G = 6.67430e-11;
    const double AU = 1.495978707e11;
    const double EarthRadius = 6.371e6;
    const double InterceptThreshold = 1000e3;

    class Body
    {
        public string Name;
        public double Mass;
        public double[] Pos = new double[3];
        public double[] Vel = new double[3];
    }

    class State
    {
        public double[] Pos = new double[3];
        public double[] Vel = new double[3];
        public double Time;
    }

    class Burn
    {
        public double Time;
        public double[] DV = new double[3];
    }

    static void Main()
    {
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;

        var input = ReadInput();
        var solution = SolveTrajectory(input);
        WriteSolution(solution);
    }

    static Dictionary<string, object> ReadInput()
    {
        var parts = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int numBodies = int.Parse(parts[0]);
        double warningDays = double.Parse(parts[1]);
        double impactorMass = double.Parse(parts[2]);
        double deltaVBudget = double.Parse(parts[3]);
        double asteroidMass = double.Parse(parts[4]);
        double t0Unix = double.Parse(parts[5]);

        var bodies = new List<Body>();
        for (int i = 0; i < numBodies; i++)
        {
            var bodyParts = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            var body = new Body
            {
                Name = bodyParts[0],
                Mass = double.Parse(bodyParts[1])
            };
            for (int j = 0; j < 3; j++)
            {
                body.Pos[j] = double.Parse(bodyParts[2 + j]);
                body.Vel[j] = double.Parse(bodyParts[5 + j]);
            }
            bodies.Add(body);
        }

        string de421Path = Console.ReadLine()?.Trim();

        return new Dictionary<string, object>
        {
            ["numBodies"] = numBodies,
            ["warningDays"] = warningDays,
            ["impactorMass"] = impactorMass,
            ["deltaVBudget"] = deltaVBudget,
            ["asteroidMass"] = asteroidMass,
            ["t0Unix"] = t0Unix,
            ["bodies"] = bodies,
            ["de421Path"] = de421Path
        };
    }

    static List<Burn> SolveTrajectory(Dictionary<string, object> input)
    {
        var bodies = (List<Body>)input["bodies"];
        double warningDays = (double)input["warningDays"];
        double deltaVBudget = (double)input["deltaVBudget"];
        double impactorMass = (double)input["impactorMass"];
        double asteroidMass = (double)input["asteroidMass"];

        var earth = bodies.First(b => b.Name == "Earth");
        var asteroid = bodies.First(b => b.Name == "Asteroid");
        var sun = bodies.FirstOrDefault(b => b.Name == "Sun");

        double sunMass = sun?.Mass ?? 1.989e30;

        // Find optimal launch time and initial trajectory
        var (launchTime, burns) = FindOptimalTrajectory(
            earth, asteroid, bodies, warningDays, deltaVBudget,
            impactorMass, asteroidMass, sunMass);

        return burns;
    }

    static (double, List<Burn>) FindOptimalTrajectory(
        Body earth, Body asteroid, List<Body> bodies,
        double warningDays, double deltaVBudget,
        double impactorMass, double asteroidMass, double sunMass)
    {
        // Predict when asteroid will be closest to Earth
        double tClose = PredictClosestApproach(earth, asteroid, sunMass, warningDays);

        // Try different launch times
        double bestScore = double.MaxValue;
        List<Burn> bestBurns = null;

        int numSamples = Math.Min(50, (int)warningDays + 1);
        for (int i = 0; i < numSamples; i++)
        {
            double launchTime = i * warningDays / Math.Max(1, numSamples - 1);
            double tof = Math.Max(10, tClose - launchTime - 5);

            var burns = ComputeInterceptTrajectory(
                earth, asteroid, bodies, launchTime, tof,
                deltaVBudget, impactorMass, asteroidMass, sunMass);

            if (burns != null)
            {
                double totalDV = burns.Sum(b => Magnitude(b.DV));
                if (totalDV <= deltaVBudget && totalDV < bestScore)
                {
                    bestScore = totalDV;
                    bestBurns = burns;
                }
            }
        }

        // If no solution found, try simple direct approach
        if (bestBurns == null)
        {
            double launchTime = Math.Min(1.0, warningDays * 0.1);
            double tof = warningDays - launchTime - 10;
            bestBurns = ComputeInterceptTrajectory(
                earth, asteroid, bodies, launchTime, tof,
                deltaVBudget, impactorMass, asteroidMass, sunMass);
        }

        return (0, bestBurns ?? GenerateFallbackSolution(earth, asteroid, warningDays, deltaVBudget, sunMass));
    }

    static List<Burn> ComputeInterceptTrajectory(
        Body earth, Body asteroid, List<Body> bodies,
        double launchTime, double tof, double deltaVBudget,
        double impactorMass, double asteroidMass, double sunMass)
    {
        // Propagate Earth and asteroid to launch time
        var earthLaunch = Propagate(earth, launchTime, sunMass);
        var asteroidIntercept = Propagate(asteroid, launchTime + tof, sunMass);

        // Solve Lambert's problem for transfer orbit
        var (v1, v2) = SolveLambert(earthLaunch.Pos, asteroidIntercept.Pos, tof * 86400, sunMass);

        if (v1 == null) return null;

        // Calculate launch delta-v
        double[] launchDV = Subtract(v1, earthLaunch.Vel);
        double launchDVMag = Magnitude(launchDV);

        if (launchDVMag > deltaVBudget * 0.95) return null;

        var burns = new List<Burn>
        {
            new Burn { Time = launchTime, DV = launchDV }
        };

        // Mid-course correction at midpoint
        double midTime = launchTime + tof / 2;
        var scMid = PropagateSC(earthLaunch.Pos, v1, tof / 2 * 86400, sunMass);
        var asteroidMid = Propagate(asteroid, midTime, sunMass);

        double[] toTarget = Subtract(asteroidMid.Pos, scMid.Pos);
        double dist = Magnitude(toTarget);

        if (dist > 0.1 * AU)
        {
            double[] correctionDV = Scale(Normalize(toTarget), Math.Min(500, deltaVBudget - launchDVMag));
            if (Magnitude(correctionDV) > 10)
            {
                burns.Add(new Burn { Time = midTime, DV = correctionDV });
            }
        }

        // Final impact
        double impactTime = launchTime + tof;
        double impactSpeed = Math.Min(5000, deltaVBudget - burns.Sum(b => Magnitude(b.DV)));

        var asteroidFinal = Propagate(asteroid, impactTime, sunMass);
        double[] impactDir = Normalize(Subtract(asteroidFinal.Pos, earth.Pos));
        double[] impactDV = Scale(impactDir, -impactSpeed);

        burns.Add(new Burn { Time = impactTime, DV = impactDV });

        double totalDV = burns.Sum(b => Magnitude(b.DV));
        if (totalDV > deltaVBudget) return null;

        return burns;
    }

    static List<Burn> GenerateFallbackSolution(
        Body earth, Body asteroid, double warningDays,
        double deltaVBudget, double sunMass)
    {
        double launchTime = Math.Min(5.0, warningDays * 0.2);
        double interceptTime = warningDays * 0.9;

        double[] toAsteroid = Subtract(asteroid.Pos, earth.Pos);
        double dist = Magnitude(toAsteroid);

        double dvLaunch = Math.Min(deltaVBudget * 0.6, 30000);
        double dvImpact = deltaVBudget - dvLaunch;

        double[] launchDir = Normalize(Add(toAsteroid, asteroid.Vel));
        double[] launchDV = Scale(launchDir, dvLaunch);

        double[] impactDir = Normalize(Subtract(asteroid.Pos, earth.Pos));
        double[] impactDV = Scale(impactDir, -dvImpact);

        return new List<Burn>
        {
            new Burn { Time = launchTime, DV = launchDV },
            new Burn { Time = interceptTime, DV = impactDV }
        };
    }

    static double PredictClosestApproach(Body earth, Body asteroid, double sunMass, double maxDays)
    {
        double minDist = double.MaxValue;
        double closestTime = maxDays;

        for (double t = 0; t < maxDays; t += 1)
        {
            var e = Propagate(earth, t, sunMass);
            var a = Propagate(asteroid, t, sunMass);
            double dist = Distance(e.Pos, a.Pos);
            if (dist < minDist)
            {
                minDist = dist;
                closestTime = t;
            }
        }

        return closestTime;
    }

    static State Propagate(Body body, double days, double sunMass)
    {
        double dt = days * 86400;
        double[] pos = (double[])body.Pos.Clone();
        double[] vel = (double[])body.Vel.Clone();

        int steps = Math.Max(1, (int)(days * 10));
        double h = dt / steps;

        for (int i = 0; i < steps; i++)
        {
            double[] acc = Acceleration(pos, sunMass);
            for (int j = 0; j < 3; j++)
            {
                vel[j] += acc[j] * h;
                pos[j] += vel[j] * h;
            }
        }

        return new State { Pos = pos, Vel = vel, Time = days };
    }

    static State PropagateSC(double[] pos0, double[] vel0, double dt, double sunMass)
    {
        double[] pos = (double[])pos0.Clone();
        double[] vel = (double[])vel0.Clone();

        int steps = Math.Max(10, (int)(dt / 3600));
        double h = dt / steps;

        for (int i = 0; i < steps; i++)
        {
            double[] acc = Acceleration(pos, sunMass);
            for (int j = 0; j < 3; j++)
            {
                vel[j] += acc[j] * h;
                pos[j] += vel[j] * h;
            }
        }

        return new State { Pos = pos, Vel = vel };
    }

    static double[] Acceleration(double[] pos, double sunMass)
    {
        double r = Magnitude(pos);
        double a = -G * sunMass / (r * r * r);
        return new double[] { a * pos[0], a * pos[1], a * pos[2] };
    }

    static (double[], double[]) SolveLambert(double[] r1, double[] r2, double dt, double sunMass)
    {
        double rm1 = Magnitude(r1);
        double rm2 = Magnitude(r2);
        double[] c = Subtract(r2, r1);
        double cm = Magnitude(c);

        double[] h = Cross(r1, r2);
        double hm = Magnitude(h);

        if (hm < 1e6) return (null, null);

        double[] ih = Normalize(h);
        double alpha = 2 * Math.Asin(Math.Sqrt(0.5 * cm / (rm1 + rm2 + cm)));

        double beta = 2 * Math.Asin(Math.Sqrt(0.5 * cm / (rm1 + rm2 - cm)));
        if (double.IsNaN(beta)) beta = Math.PI;

        double mu = G * sunMass;
        double a = (rm1 + rm2 + cm) / 4;

        for (int iter = 0; iter < 20; iter++)
        {
            double alpha2 = alpha * alpha;
            double alpha3 = alpha2 * alpha;
            double tof = Math.Sqrt(a * a * a / mu) * (alpha3 / 6 + Math.PI - alpha);

            if (Math.Abs(tof - dt) < dt * 0.01) break;

            double dtda = 1.5 * Math.Sqrt(a / mu) * (alpha3 / 6 + Math.PI - alpha);
            a = a - (tof - dt) / dtda * a;
            a = Math.Max(a, (rm1 + rm2) / 4);
        }

        double p = 4 * a * (rm1 + rm2 - cm) / (2 * rm1 * rm2 * (1 - Math.Cos(alpha)));
        double ecosE = 1 - rm1 / a;
        double esinE = Math.Sqrt(Math.Max(0, 1 - p / a));

        double[] v1 = Scale(Normalize(Subtract(r2, Scale(r1, (1 - rm1/a)))),
            Math.Sqrt(mu / p));
        double[] v2 = Scale(Normalize(Subtract(Scale(r2, (1 - rm2/a)), r1)),
            Math.Sqrt(mu / p));

        double vm = Math.Sqrt(mu * (2/rm1 - 1/a));
        v1 = Scale(Normalize(Add(Subtract(r2, r1), Scale(ih, 0.1))), vm);

        return (v1, v2);
    }

    static void WriteSolution(List<Burn> burns)
    {
        Console.WriteLine(burns.Count);
        foreach (var burn in burns)
        {
            Console.WriteLine($"{burn.Time:F6} {burn.DV[0]:E} {burn.DV[1]:E} {burn.DV[2]:E}");
        }
    }

    static double[] Add(double[] a, double[] b) =>
        new double[] { a[0] + b[0], a[1] + b[1], a[2] + b[2] };

    static double[] Subtract(double[] a, double[] b) =>
        new double[] { a[0] - b[0], a[1] - b[1], a[2] - b[2] };

    static double[] Scale(double[] a, double s) =>
        new double[] { a[0] * s, a[1] * s, a[2] * s };

    static double Magnitude(double[] a) =>
        Math.Sqrt(a[0] * a[0] + a[1] * a[1] + a[2] * a[2]);

    static double[] Normalize(double[] a)
    {
        double m = Magnitude(a);
        return m > 0 ? Scale(a, 1.0 / m) : new double[3];
    }

    static double Distance(double[] a, double[] b) =>
        Magnitude(Subtract(a, b));

    static double[] Cross(double[] a, double[] b) => new double[]
    {
        a[1] * b[2] - a[2] * b[1],
        a[2] * b[0] - a[0] * b[2],
        a[0] * b[1] - a[1] * b[0]
    };
}
