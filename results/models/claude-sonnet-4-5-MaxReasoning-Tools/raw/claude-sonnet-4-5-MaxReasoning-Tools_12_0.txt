using System;
using System.Collections.Generic;
using System.Linq;
using System.Numerics;

class BinPacking3D
{
    // Polyhedron structure
    struct Polyhedron
    {
        public Vector3[] Vertices;
        public int[][] Faces;
        public Vector3 Min;
        public Vector3 Max;
        public Vector3 Size;
    }

    // Placement structure
    struct Placement
    {
        public Vector3 Translation;
        public Quaternion Rotation;
    }

    // Grid cell for spatial hashing
    class SpatialGrid
    {
        private Dictionary<(int, int, int), List<int>> grid;
        private float cellSize;

        public SpatialGrid(float cellSize)
        {
            this.cellSize = cellSize;
            this.grid = new Dictionary<(int, int, int), List<int>>();
        }

        private (int, int, int) GetCell(Vector3 pos)
        {
            return ((int)(pos.X / cellSize), (int)(pos.Y / cellSize), (int)(pos.Z / cellSize));
        }

        public void Add(int id, Vector3 min, Vector3 max)
        {
            var minCell = GetCell(min);
            var maxCell = GetCell(max);

            for (int x = minCell.Item1; x <= maxCell.Item1; x++)
                for (int y = minCell.Item2; y <= maxCell.Item2; y++)
                    for (int z = minCell.Item3; z <= maxCell.Item3; z++)
                    {
                        var cell = (x, y, z);
                        if (!grid.ContainsKey(cell))
                            grid[cell] = new List<int>();
                        grid[cell].Add(id);
                    }
        }

        public List<int> Query(Vector3 min, Vector3 max)
        {
            var result = new HashSet<int>();
            var minCell = GetCell(min);
            var maxCell = GetCell(max);

            for (int x = minCell.Item1; x <= maxCell.Item1; x++)
                for (int y = minCell.Item2; y <= maxCell.Item2; y++)
                    for (int z = minCell.Item3; z <= maxCell.Item3; z++)
                    {
                        var cell = (x, y, z);
                        if (grid.ContainsKey(cell))
                            foreach (var id in grid[cell])
                                result.Add(id);
                    }

            return result.ToList();
        }
    }

    static void Main()
    {
        // Parse input
        var line = Console.ReadLine().Split();
        int V = int.Parse(line[0]);
        int F = int.Parse(line[1]);

        var vertices = new Vector3[V];
        for (int i = 0; i < V; i++)
        {
            var parts = Console.ReadLine().Split();
            vertices[i] = new Vector3(float.Parse(parts[0]), float.Parse(parts[1]), float.Parse(parts[2]));
        }

        var faces = new int[F][];
        for (int i = 0; i < F; i++)
        {
            var parts = Console.ReadLine().Split();
            faces[i] = new int[] { int.Parse(parts[0]), int.Parse(parts[1]), int.Parse(parts[2]) };
        }

        line = Console.ReadLine().Split();
        float W = float.Parse(line[0]);
        float H = float.Parse(line[1]);
        float D = float.Parse(line[2]);

        // Calculate bounding box
        var poly = new Polyhedron
        {
            Vertices = vertices,
            Faces = faces
        };

        CalculateBounds(ref poly);

        // Adaptive strategy selection
        var containerVolume = W * H * D;
        var polyVolume = poly.Size.X * poly.Size.Y * poly.Size.Z;
        var maxPossible = (int)(containerVolume / polyVolume);

        List<Placement> placements;

        if (containerVolume < 2000)
        {
            // Trivial: exhaustive search with multiple rotations
            placements = PackTrivial(poly, W, H, D);
        }
        else if (containerVolume < 100000)
        {
            // Medium: grid-based with selective rotations
            placements = PackMedium(poly, W, H, D);
        }
        else if (containerVolume < 2000000)
        {
            // Large: coarse grid with limited rotations
            placements = PackLarge(poly, W, H, D);
        }
        else
        {
            // Extreme: ultra-fast grid with identity rotation only
            placements = PackExtreme(poly, W, H, D);
        }

        // Output
        Console.WriteLine(placements.Count);
        foreach (var p in placements)
        {
            Console.WriteLine($"{p.Translation.X} {p.Translation.Y} {p.Translation.Z} {p.Rotation.W} {p.Rotation.X} {p.Rotation.Y} {p.Rotation.Z}");
        }
    }

    static void CalculateBounds(ref Polyhedron poly)
    {
        poly.Min = new Vector3(float.MaxValue, float.MaxValue, float.MaxValue);
        poly.Max = new Vector3(float.MinValue, float.MinValue, float.MinValue);

        foreach (var v in poly.Vertices)
        {
            poly.Min = Vector3.Min(poly.Min, v);
            poly.Max = Vector3.Max(poly.Max, v);
        }

        poly.Size = poly.Max - poly.Min;
    }

    static List<Quaternion> GetRotations(int level)
    {
        var rotations = new List<Quaternion> { Quaternion.Identity };

        if (level >= 1)
        {
            // 90 degree rotations around each axis
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitX, MathF.PI / 2));
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitY, MathF.PI / 2));
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitZ, MathF.PI / 2));
        }

        if (level >= 2)
        {
            // 180 and 270 degree rotations
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitX, MathF.PI));
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitY, MathF.PI));
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitZ, MathF.PI));
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitX, 3 * MathF.PI / 2));
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitY, 3 * MathF.PI / 2));
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitZ, 3 * MathF.PI / 2));
        }

        return rotations;
    }

    static Vector3[] TransformVertices(Vector3[] vertices, Vector3 translation, Quaternion rotation)
    {
        var result = new Vector3[vertices.Length];
        for (int i = 0; i < vertices.Length; i++)
        {
            result[i] = Vector3.Transform(vertices[i], rotation) + translation;
        }
        return result;
    }

    static (Vector3, Vector3) GetTransformedBounds(Polyhedron poly, Vector3 translation, Quaternion rotation)
    {
        var transformed = TransformVertices(poly.Vertices, translation, rotation);
        var min = new Vector3(float.MaxValue, float.MaxValue, float.MaxValue);
        var max = new Vector3(float.MinValue, float.MinValue, float.MinValue);

        foreach (var v in transformed)
        {
            min = Vector3.Min(min, v);
            max = Vector3.Max(max, v);
        }

        return (min, max);
    }

    static bool IsInContainer(Vector3 min, Vector3 max, float W, float H, float D)
    {
        return min.X >= -0.001f && min.Y >= -0.001f && min.Z >= -0.001f &&
               max.X <= W + 0.001f && max.Y <= H + 0.001f && max.Z <= D + 0.001f;
    }

    static bool CheckCollision(Vector3 min1, Vector3 max1, Vector3 min2, Vector3 max2)
    {
        return !(max1.X < min2.X || min1.X > max2.X ||
                 max1.Y < min2.Y || min1.Y > max2.Y ||
                 max1.Z < min2.Z || min1.Z > max2.Z);
    }

    static List<Placement> PackTrivial(Polyhedron poly, float W, float H, float D)
    {
        var placements = new List<Placement>();
        var rotations = GetRotations(2);
        var step = Math.Max(0.5f, poly.Size.Length() / 4);

        var occupied = new List<(Vector3, Vector3)>();

        for (float x = 0; x < W; x += step)
        {
            for (float y = 0; y < H; y += step)
            {
                for (float z = 0; z < D; z += step)
                {
                    foreach (var rot in rotations)
                    {
                        var translation = new Vector3(x, y, z);
                        var (min, max) = GetTransformedBounds(poly, translation, rot);

                        if (!IsInContainer(min, max, W, H, D))
                            continue;

                        bool collides = false;
                        foreach (var (omin, omax) in occupied)
                        {
                            if (CheckCollision(min, max, omin, omax))
                            {
                                collides = true;
                                break;
                            }
                        }

                        if (!collides)
                        {
                            placements.Add(new Placement { Translation = translation, Rotation = rot });
                            occupied.Add((min, max));
                            goto NextPosition;
                        }
                    }
                NextPosition:;
                }
            }
        }

        return placements;
    }

    static List<Placement> PackMedium(Polyhedron poly, float W, float H, float D)
    {
        var placements = new List<Placement>();
        var rotations = GetRotations(1);
        var step = Math.Max(1.0f, poly.Size.Length() / 3);

        var grid = new SpatialGrid(step * 2);
        var occupied = new List<(Vector3, Vector3, int)>();

        for (float x = 0; x < W; x += step)
        {
            for (float y = 0; y < H; y += step)
            {
                for (float z = 0; z < D; z += step)
                {
                    foreach (var rot in rotations)
                    {
                        var translation = new Vector3(x, y, z);
                        var (min, max) = GetTransformedBounds(poly, translation, rot);

                        if (!IsInContainer(min, max, W, H, D))
                            continue;

                        var nearby = grid.Query(min, max);
                        bool collides = false;

                        foreach (var idx in nearby)
                        {
                            if (CheckCollision(min, max, occupied[idx].Item1, occupied[idx].Item2))
                            {
                                collides = true;
                                break;
                            }
                        }

                        if (!collides)
                        {
                            int id = placements.Count;
                            placements.Add(new Placement { Translation = translation, Rotation = rot });
                            occupied.Add((min, max, id));
                            grid.Add(id, min, max);
                            goto NextPosition;
                        }
                    }
                NextPosition:;
                }
            }
        }

        return placements;
    }

    static List<Placement> PackLarge(Polyhedron poly, float W, float H, float D)
    {
        var placements = new List<Placement>();
        var step = Math.Max(2.0f, poly.Size.Length() / 2);
        var grid = new SpatialGrid(step * 3);
        var occupied = new List<(Vector3, Vector3, int)>();

        for (float x = 0; x < W; x += step)
        {
            for (float y = 0; y < H; y += step)
            {
                for (float z = 0; z < D; z += step)
                {
                    var translation = new Vector3(x, y, z);
                    var (min, max) = GetTransformedBounds(poly, translation, Quaternion.Identity);

                    if (!IsInContainer(min, max, W, H, D))
                        continue;

                    var nearby = grid.Query(min, max);
                    bool collides = false;

                    foreach (var idx in nearby)
                    {
                        if (CheckCollision(min, max, occupied[idx].Item1, occupied[idx].Item2))
                        {
                            collides = true;
                            break;
                        }
                    }

                    if (!collides)
                    {
                        int id = placements.Count;
                        placements.Add(new Placement { Translation = translation, Rotation = Quaternion.Identity });
                        occupied.Add((min, max, id));
                        grid.Add(id, min, max);
                    }
                }
            }
        }

        return placements;
    }

    static List<Placement> PackExtreme(Polyhedron poly, float W, float H, float D)
    {
        var placements = new List<Placement>();
        var step = Math.Max(3.0f, poly.Size.Length());

        // Ultra-fast: simple grid without collision checks for extreme scale
        var nx = (int)(W / step);
        var ny = (int)(H / step);
        var nz = (int)(D / step);

        for (int ix = 0; ix < nx; ix++)
        {
            for (int iy = 0; iy < ny; iy++)
            {
                for (int iz = 0; iz < nz; iz++)
                {
                    var translation = new Vector3(ix * step, iy * step, iz * step);
                    var (min, max) = GetTransformedBounds(poly, translation, Quaternion.Identity);

                    if (IsInContainer(min, max, W, H, D))
                    {
                        placements.Add(new Placement { Translation = translation, Rotation = Quaternion.Identity });
                    }
                }
            }
        }

        return placements;
    }
}
