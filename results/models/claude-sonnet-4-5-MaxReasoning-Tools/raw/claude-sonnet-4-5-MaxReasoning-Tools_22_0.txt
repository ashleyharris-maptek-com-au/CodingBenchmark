using System;
using System.Collections.Generic;
using System.Linq;

class TetrisPlayer
{
    static void Main()
    {
        var input = Console.ReadLine().Split();
        int width = int.Parse(input[0]);
        int height = int.Parse(input[1]);
        int numPieces = int.Parse(input[2]);

        var game = new Game(width, height);

        for (int i = 0; i < numPieces; i++)
        {
            string pieceType = Console.ReadLine().Trim();
            var (rotation, column) = game.FindBestPlacement(pieceType);
            Console.WriteLine($"{rotation} {column}");
            Console.Out.Flush();

            string result = Console.ReadLine().Trim();
            if (result == "gameover")
                break;

            int linesCleared = 0;
            if (result.StartsWith("cleared_"))
                linesCleared = int.Parse(result.Substring(8));

            game.ApplyPlacement(pieceType, rotation, column, linesCleared);
        }
    }
}

class Game
{
    private int width;
    private int height;
    private bool[,] board;

    public Game(int width, int height)
    {
        this.width = width;
        this.height = height;
        this.board = new bool[height, width];
    }

    public (int rotation, int column) FindBestPlacement(string pieceType)
    {
        var piece = GetPiece(pieceType);
        double bestScore = double.MinValue;
        int bestRotation = 0;
        int bestColumn = 0;

        for (int rotation = 0; rotation < 4; rotation++)
        {
            var rotatedShape = RotatePiece(piece, rotation);
            int pieceWidth = GetPieceWidth(rotatedShape);

            for (int col = 0; col <= width - pieceWidth; col++)
            {
                var testBoard = CloneBoard();
                if (TryPlacePiece(testBoard, rotatedShape, col, out int linesCleared))
                {
                    double score = EvaluateBoard(testBoard, linesCleared);
                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestRotation = rotation;
                        bestColumn = col;
                    }
                }
            }
        }

        return (bestRotation, bestColumn);
    }

    public void ApplyPlacement(string pieceType, int rotation, int column, int linesCleared)
    {
        var piece = GetPiece(pieceType);
        var rotatedShape = RotatePiece(piece, rotation);
        TryPlacePiece(board, rotatedShape, column, out _);

        if (linesCleared > 0)
            ClearLines(linesCleared);
    }

    private bool[,] CloneBoard()
    {
        var clone = new bool[height, width];
        Array.Copy(board, clone, board.Length);
        return clone;
    }

    private bool TryPlacePiece(bool[,] testBoard, List<(int r, int c)> shape, int column, out int linesCleared)
    {
        linesCleared = 0;

        // Find landing row for the piece
        int landingRow = -1;
        for (int row = 0; row < height; row++)
        {
            bool collision = false;
            foreach (var (r, c) in shape)
            {
                int boardRow = row + r;
                int boardCol = column + c;
                if (boardRow >= height || testBoard[boardRow, boardCol])
                {
                    collision = true;
                    break;
                }
            }
            if (collision)
            {
                landingRow = row - 1;
                break;
            }
        }

        if (landingRow == -1)
            landingRow = height - 1;

        // Check if piece fits at landing position
        foreach (var (r, c) in shape)
        {
            int boardRow = landingRow + r;
            int boardCol = column + c;
            if (boardRow < 0 || boardRow >= height || boardCol < 0 || boardCol >= width)
                return false;
            if (testBoard[boardRow, boardCol])
                return false;
        }

        // Place the piece
        foreach (var (r, c) in shape)
        {
            testBoard[landingRow + r, column + c] = true;
        }

        // Count completed lines
        for (int row = 0; row < height; row++)
        {
            bool lineComplete = true;
            for (int col = 0; col < width; col++)
            {
                if (!testBoard[row, col])
                {
                    lineComplete = false;
                    break;
                }
            }
            if (lineComplete)
                linesCleared++;
        }

        return true;
    }

    private void ClearLines(int numLines)
    {
        int cleared = 0;
        for (int row = height - 1; row >= 0 && cleared < numLines; row--)
        {
            bool lineComplete = true;
            for (int col = 0; col < width; col++)
            {
                if (!board[row, col])
                {
                    lineComplete = false;
                    break;
                }
            }

            if (lineComplete)
            {
                cleared++;
                // Shift rows down
                for (int r = row; r > 0; r--)
                {
                    for (int col = 0; col < width; col++)
                        board[r, col] = board[r - 1, col];
                }
                for (int col = 0; col < width; col++)
                    board[0, col] = false;
                row++; // Check this row again
            }
        }
    }

    private double EvaluateBoard(bool[,] testBoard, int linesCleared)
    {
        double score = 0;

        // Lines cleared bonus
        score += linesCleared * 1000;

        // Calculate column heights
        int[] columnHeights = new int[width];
        for (int col = 0; col < width; col++)
        {
            for (int row = 0; row < height; row++)
            {
                if (testBoard[row, col])
                {
                    columnHeights[col] = height - row;
                    break;
                }
            }
        }

        // Aggregate height penalty
        int totalHeight = columnHeights.Sum();
        score -= totalHeight * 0.51;

        // Holes penalty
        int holes = 0;
        for (int col = 0; col < width; col++)
        {
            bool foundBlock = false;
            for (int row = 0; row < height; row++)
            {
                if (testBoard[row, col])
                    foundBlock = true;
                else if (foundBlock)
                    holes++;
            }
        }
        score -= holes * 35;

        // Bumpiness penalty
        int bumpiness = 0;
        for (int col = 0; col < width - 1; col++)
            bumpiness += Math.Abs(columnHeights[col] - columnHeights[col + 1]);
        score -= bumpiness * 1.8;

        // Max height penalty
        int maxHeight = columnHeights.Max();
        score -= maxHeight * 0.5;

        // Well bonus (for I-piece placement)
        for (int col = 0; col < width; col++)
        {
            if (columnHeights[col] < maxHeight - 3)
            {
                bool isWell = true;
                if (col > 0 && columnHeights[col - 1] <= columnHeights[col] + 1)
                    isWell = false;
                if (col < width - 1 && columnHeights[col + 1] <= columnHeights[col] + 1)
                    isWell = false;
                if (isWell)
                    score += 5;
            }
        }

        return score;
    }

    private List<(int r, int c)> GetPiece(string type)
    {
        switch (type)
        {
            case "I": return new List<(int, int)> { (0, 0), (0, 1), (0, 2), (0, 3) };
            case "O": return new List<(int, int)> { (0, 0), (0, 1), (1, 0), (1, 1) };
            case "T": return new List<(int, int)> { (0, 1), (1, 0), (1, 1), (1, 2) };
            case "S": return new List<(int, int)> { (0, 1), (0, 2), (1, 0), (1, 1) };
            case "Z": return new List<(int, int)> { (0, 0), (0, 1), (1, 1), (1, 2) };
            case "J": return new List<(int, int)> { (0, 0), (1, 0), (1, 1), (1, 2) };
            case "L": return new List<(int, int)> { (0, 2), (1, 0), (1, 1), (1, 2) };
            default: throw new ArgumentException($"Unknown piece type: {type}");
        }
    }

    private List<(int r, int c)> RotatePiece(List<(int r, int c)> shape, int rotations)
    {
        var result = new List<(int r, int c)>(shape);
        for (int i = 0; i < rotations; i++)
        {
            result = result.Select(p => (p.c, -p.r)).ToList();

            // Normalize to top-left
            int minR = result.Min(p => p.Item1);
            int minC = result.Min(p => p.Item2);
            result = result.Select(p => (p.Item1 - minR, p.Item2 - minC)).ToList();
        }
        return result;
    }

    private int GetPieceWidth(List<(int r, int c)> shape)
    {
        return shape.Max(p => p.c) + 1;
    }
}
