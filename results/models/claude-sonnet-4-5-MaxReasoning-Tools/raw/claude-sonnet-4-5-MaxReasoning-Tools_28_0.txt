use std::io::{self, BufRead, Write};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read number of vertices and edges
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let n = parts[0];
    let m = parts[1];

    // Read all edges
    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let edge_parts: Vec<usize> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        edges.push((edge_parts[0], edge_parts[1]));
    }

    // Solve using greedy approach with local optimization
    let cover = solve_vertex_cover(n, &edges);

    // Output the result
    let stdout = io::stdout();
    let mut out = stdout.lock();
    writeln!(out, "{}", cover.len()).unwrap();
    for (i, &v) in cover.iter().enumerate() {
        if i > 0 {
            write!(out, " ").unwrap();
        }
        write!(out, "{}", v).unwrap();
    }
    writeln!(out).unwrap();
}

fn solve_vertex_cover(n: usize, edges: &[(usize, usize)]) -> Vec<usize> {
    if edges.is_empty() {
        return Vec::new();
    }

    // Phase 1: Greedy selection based on degree
    let mut cover = HashSet::new();
    let mut covered = vec![false; edges.len()];

    loop {
        // Count uncovered edges incident to each vertex
        let mut degree = vec![0_usize; n];
        for (i, &(u, v)) in edges.iter().enumerate() {
            if !covered[i] {
                degree[u] += 1;
                degree[v] += 1;
            }
        }

        // Find vertex with maximum degree among uncovered edges
        let max_result = degree.iter()
            .enumerate()
            .max_by_key(|(_, &d)| d);

        match max_result {
            Some((v, &d)) if d > 0 => {
                // Add this vertex to the cover
                cover.insert(v);

                // Mark all edges incident to this vertex as covered
                for (i, &(u, w)) in edges.iter().enumerate() {
                    if u == v || w == v {
                        covered[i] = true;
                    }
                }
            }
            _ => break, // No more uncovered edges
        }
    }

    // Phase 2: Local optimization - try to remove redundant vertices
    let vertices: Vec<usize> = cover.iter().copied().collect();
    for &v in &vertices {
        // Temporarily remove vertex v
        cover.remove(&v);

        // Check if all edges are still covered
        let mut all_covered = true;
        for &(u, w) in edges {
            if !cover.contains(&u) && !cover.contains(&w) {
                all_covered = false;
                break;
            }
        }

        // If removing v leaves some edge uncovered, put it back
        if !all_covered {
            cover.insert(v);
        }
    }

    // Return sorted result
    let mut result: Vec<usize> = cover.into_iter().collect();
    result.sort_unstable();
    result
}
