OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 15
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos_in Location 0
               OpDecorate %normal_in Location 1
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
       %func = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0

      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
      %int_4 = OpConstant %int 4
      %int_5 = OpConstant %int 5
     %uint_6 = OpConstant %uint 6

    %float_0 = OpConstant %float 0.0
  %float_0_3 = OpConstant %float 0.3
    %float_1 = OpConstant %float 1.0
%float_0_015 = OpConstant %float 0.015

 %white_vec3 = OpConstantComposite %v3float %float_1 %float_1 %float_1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in_6 = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in_6
      %gl_in = OpVariable %ptr_in_arr_pv Input
  %ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_in_6 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in_6
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
 %normal_in  = OpVariable %ptr_in_arr_v3 Input
  %color_in  = OpVariable %ptr_in_arr_v3 Input
  %ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
 %ptr_out_v4 = OpTypePointer Output %v4float

%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %func
      %entry = OpLabel

; Load positions of main triangle vertices (0, 2, 4)
    %pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
        %pos0 = OpLoad %v4float %pos0_ptr
    %pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
        %pos2 = OpLoad %v4float %pos2_ptr
    %pos4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
        %pos4 = OpLoad %v4float %pos4_ptr

; Load colors for the main triangle
   %col0_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
       %col0 = OpLoad %v3float %col0_ptr
   %col2_ptr = OpAccessChain %ptr_in_v3 %color_in %int_2
       %col2 = OpLoad %v3float %col2_ptr
   %col4_ptr = OpAccessChain %ptr_in_v3 %color_in %int_4
       %col4 = OpLoad %v3float %col4_ptr

; Compute dimmed colors for triangle fill (color * 0.3)
 %dim_col0_s = OpVectorTimesScalar %v3float %col0 %float_0_3
 %dim_col2_s = OpVectorTimesScalar %v3float %col2 %float_0_3
 %dim_col4_s = OpVectorTimesScalar %v3float %col4 %float_0_3

; Helper: compute perpendicular offset for thick lines
; Extract xy from position, compute perpendicular direction, scale by thickness
; For edge from p1 to p2: dir = p2.xy - p1.xy, perp = (-dir.y, dir.x) normalized * thickness

; === Edge 0->2 ===
  %p0_xy = OpVectorShuffle %v3float %pos0 %pos0 0 1 2
  %p2_xy = OpVectorShuffle %v3float %pos2 %pos2 0 1 2
%edge02_dir = OpFSub %v3float %p2_xy %p0_xy
%edge02_dx = OpCompositeExtract %float %edge02_dir 0
%edge02_dy = OpCompositeExtract %float %edge02_dir 1
%edge02_neg_dy = OpFNegate %float %edge02_dy
%edge02_perp_pre = OpCompositeConstruct %v3float %edge02_neg_dy %edge02_dx %float_0
%edge02_len_sq = OpDot %float %edge02_dir %edge02_dir
%edge02_len = OpExtInst %float %1 Sqrt %edge02_len_sq
%edge02_scale = OpFDiv %float %float_0_015 %edge02_len
%edge02_offset = OpVectorTimesScalar %v3float %edge02_perp_pre %edge02_scale

; Emit quad for edge 0->2 (white color)
; Vertex 1: pos0 + offset
%off02_v4_p = OpCompositeConstruct %v4float %edge02_offset %float_0
%quad02_v1_pos = OpFAdd %v4float %pos0 %off02_v4_p
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos_ptr %quad02_v1_pos
               OpStore %outColor %white_vec3
               OpEmitVertex

; Vertex 2: pos0 - offset
%edge02_offset_neg = OpFNegate %v3float %edge02_offset
%off02_v4_n = OpCompositeConstruct %v4float %edge02_offset_neg %float_0
%quad02_v2_pos = OpFAdd %v4float %pos0 %off02_v4_n
               OpStore %out_pos_ptr %quad02_v2_pos
               OpStore %outColor %white_vec3
               OpEmitVertex

; Vertex 3: pos2 + offset
%quad02_v3_pos = OpFAdd %v4float %pos2 %off02_v4_p
               OpStore %out_pos_ptr %quad02_v3_pos
               OpStore %outColor %white_vec3
               OpEmitVertex

; Vertex 4: pos2 - offset
%quad02_v4_pos = OpFAdd %v4float %pos2 %off02_v4_n
               OpStore %out_pos_ptr %quad02_v4_pos
               OpStore %outColor %white_vec3
               OpEmitVertex
               OpEndPrimitive

; === Edge 2->4 ===
  %p4_xy = OpVectorShuffle %v3float %pos4 %pos4 0 1 2
%edge24_dir = OpFSub %v3float %p4_xy %p2_xy
%edge24_dx = OpCompositeExtract %float %edge24_dir 0
%edge24_dy = OpCompositeExtract %float %edge24_dir 1
%edge24_neg_dy = OpFNegate %float %edge24_dy
%edge24_perp_pre = OpCompositeConstruct %v3float %edge24_neg_dy %edge24_dx %float_0
%edge24_len_sq = OpDot %float %edge24_dir %edge24_dir
%edge24_len = OpExtInst %float %1 Sqrt %edge24_len_sq
%edge24_scale = OpFDiv %float %float_0_015 %edge24_len
%edge24_offset = OpVectorTimesScalar %v3float %edge24_perp_pre %edge24_scale

; Emit quad for edge 2->4
%off24_v4_p = OpCompositeConstruct %v4float %edge24_offset %float_0
%quad24_v1_pos = OpFAdd %v4float %pos2 %off24_v4_p
               OpStore %out_pos_ptr %quad24_v1_pos
               OpStore %outColor %white_vec3
               OpEmitVertex

%edge24_offset_neg = OpFNegate %v3float %edge24_offset
%off24_v4_n = OpCompositeConstruct %v4float %edge24_offset_neg %float_0
%quad24_v2_pos = OpFAdd %v4float %pos2 %off24_v4_n
               OpStore %out_pos_ptr %quad24_v2_pos
               OpStore %outColor %white_vec3
               OpEmitVertex

%quad24_v3_pos = OpFAdd %v4float %pos4 %off24_v4_p
               OpStore %out_pos_ptr %quad24_v3_pos
               OpStore %outColor %white_vec3
               OpEmitVertex

%quad24_v4_pos = OpFAdd %v4float %pos4 %off24_v4_n
               OpStore %out_pos_ptr %quad24_v4_pos
               OpStore %outColor %white_vec3
               OpEmitVertex
               OpEndPrimitive

; === Edge 4->0 ===
%edge40_dir = OpFSub %v3float %p0_xy %p4_xy
%edge40_dx = OpCompositeExtract %float %edge40_dir 0
%edge40_dy = OpCompositeExtract %float %edge40_dir 1
%edge40_neg_dy = OpFNegate %float %edge40_dy
%edge40_perp_pre = OpCompositeConstruct %v3float %edge40_neg_dy %edge40_dx %float_0
%edge40_len_sq = OpDot %float %edge40_dir %edge40_dir
%edge40_len = OpExtInst %float %1 Sqrt %edge40_len_sq
%edge40_scale = OpFDiv %float %float_0_015 %edge40_len
%edge40_offset = OpVectorTimesScalar %v3float %edge40_perp_pre %edge40_scale

; Emit quad for edge 4->0
%off40_v4_p = OpCompositeConstruct %v4float %edge40_offset %float_0
%quad40_v1_pos = OpFAdd %v4float %pos4 %off40_v4_p
               OpStore %out_pos_ptr %quad40_v1_pos
               OpStore %outColor %white_vec3
               OpEmitVertex

%edge40_offset_neg = OpFNegate %v3float %edge40_offset
%off40_v4_n = OpCompositeConstruct %v4float %edge40_offset_neg %float_0
%quad40_v2_pos = OpFAdd %v4float %pos4 %off40_v4_n
               OpStore %out_pos_ptr %quad40_v2_pos
               OpStore %outColor %white_vec3
               OpEmitVertex

%quad40_v3_pos = OpFAdd %v4float %pos0 %off40_v4_p
               OpStore %out_pos_ptr %quad40_v3_pos
               OpStore %outColor %white_vec3
               OpEmitVertex

%quad40_v4_pos = OpFAdd %v4float %pos0 %off40_v4_n
               OpStore %out_pos_ptr %quad40_v4_pos
               OpStore %outColor %white_vec3
               OpEmitVertex
               OpEndPrimitive

; === Emit dimmed triangle (0, 2, 4) ===
               OpStore %out_pos_ptr %pos0
               OpStore %outColor %dim_col0_s
               OpEmitVertex

               OpStore %out_pos_ptr %pos2
               OpStore %outColor %dim_col2_s
               OpEmitVertex

               OpStore %out_pos_ptr %pos4
               OpStore %outColor %dim_col4_s
               OpEmitVertex
               OpEndPrimitive

               OpReturn
               OpFunctionEnd

          %1 = OpExtInstImport "GLSL.std.450"
