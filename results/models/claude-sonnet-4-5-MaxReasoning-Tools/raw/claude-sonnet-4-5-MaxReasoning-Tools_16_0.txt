#include <iostream>
#include <vector>
#include <limits>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    cin >> J >> M;

    // Read jobs: each job has tasks (machine, duration)
    vector<vector<pair<int, int>>> jobs(J);
    for (int j = 0; j < J; j++) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        for (int t = 0; t < T; t++) {
            cin >> jobs[j][t].first >> jobs[j][t].second;
        }
    }

    // Count total tasks for adaptive strategy selection
    int total_tasks = 0;
    for (const auto& job : jobs) total_tasks += job.size();

    // Tracking structures
    vector<int> machine_free(M, 0);     // When each machine becomes free
    vector<int> job_free(J, 0);         // When each job can start next task
    vector<int> job_progress(J, 0);     // Next task index for each job
    vector<vector<pair<int, int>>> result(J);  // Schedule results

    for (int j = 0; j < J; j++) {
        result[j].resize(jobs[j].size());
    }

    // Greedy scheduling: always schedule the task that can start earliest
    // This O(J*N) approach scales well even for large problems (1000 jobs, 50k tasks)
    // For extreme cases, could optimize to O(N log J) with priority queue
    for (int scheduled = 0; scheduled < total_tasks; scheduled++) {
        int best_job = -1;
        int best_start = numeric_limits<int>::max();

        // Find the task that can start earliest
        for (int j = 0; j < J; j++) {
            if (job_progress[j] < (int)jobs[j].size()) {
                int t_idx = job_progress[j];
                int mach = jobs[j][t_idx].first;
                // Task can start when both job and machine are ready
                int start = max(job_free[j], machine_free[mach]);

                if (start < best_start) {
                    best_start = start;
                    best_job = j;
                }
            }
        }

        // Schedule the selected task
        int j = best_job;
        int t = job_progress[j];
        int m = jobs[j][t].first;
        int d = jobs[j][t].second;

        result[j][t] = {best_start, m};

        // Update availability
        machine_free[m] = best_start + d;
        job_free[j] = best_start + d;
        job_progress[j]++;
    }

    // Output schedule
    for (int j = 0; j < J; j++) {
        for (size_t t = 0; t < result[j].size(); t++) {
            if (t > 0) cout << " ";
            cout << result[j][t].first << " " << result[j][t].second;
        }
        cout << "\n";
    }

    return 0;
}
