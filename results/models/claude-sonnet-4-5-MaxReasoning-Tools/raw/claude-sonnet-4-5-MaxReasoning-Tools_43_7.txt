struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize the interpolated normal
    float3 N = normalize(input.normal);

    // Calculate view vector (from surface to camera)
    float3 V = normalize(cameraPos.xyz - input.worldPos);

    // Calculate light vector (from surface to light)
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Calculate diffuse lighting
    float diffuse = max(dot(N, L), 0.0);

    // Calculate rim lighting: rim = pow(1 - max(dot(N,V),0), 3)
    float rim = pow(1.0 - max(dot(N, V), 0.0), 3.0);

    // Base color modulated by lighting
    float3 baseColor = float3(0.2, 0.3, 0.8);
    float3 litColor = baseColor * (0.3 + 0.5 * diffuse);

    // Add rim tint
    float3 rimTint = float3(1.0, 0.8, 0.5);
    float3 finalColor = litColor + rimTint * rim;

    // Output with alpha 1.0
    return float4(finalColor, 1.0);
}
