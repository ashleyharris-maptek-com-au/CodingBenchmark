OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1
OpSource GLSL 450
%glsl = OpExtInstImport "GLSL.std.450"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%void_func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_25 = OpConstant %float 0.25
%float_0_5 = OpConstant %float 0.5
%float_0_75 = OpConstant %float 0.75
%float_0_02 = OpConstant %float 0.02

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in_4 = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in_4
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %void_func
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_ptr

%c1_ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
%color1 = OpLoad %v3float %c1_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %in_color %int_2
%color2 = OpLoad %v3float %c2_ptr

%tangent_start = OpFSub %v4float %p2 %p0
%tangent_end = OpFSub %v4float %p3 %p1

%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

%pos0 = OpCopy %v4float %p1
%col0 = OpCopy %v3float %color1
%tan0_xy = OpVectorShuffle %v2float %tangent_start %tangent_start 0 1
%tan0_x = OpCompositeExtract %float %tan0_xy 0
%tan0_y = OpCompositeExtract %float %tan0_xy 1
%tan0_y_neg = OpFNegate %float %tan0_y
%perp0_xy = OpCompositeConstruct %v2float %tan0_y_neg %tan0_x
%perp0_len = OpExtInst %float %glsl Length %perp0_xy
%perp0_inv = OpFDiv %float %float_1 %perp0_len
%perp0_norm = OpVectorTimesScalar %v2float %perp0_xy %perp0_inv
%perp0_scaled = OpVectorTimesScalar %v2float %perp0_norm %float_0_02
%perp0_x = OpCompositeExtract %float %perp0_scaled 0
%perp0_y = OpCompositeExtract %float %perp0_scaled 1
%perp0 = OpCompositeConstruct %v4float %perp0_x %perp0_y %float_0 %float_0
%pos0_plus = OpFAdd %v4float %pos0 %perp0
%pos0_minus = OpFSub %v4float %pos0 %perp0
OpStore %out_pos_ptr %pos0_plus
OpStore %outColor %col0
OpEmitVertex
OpStore %out_pos_ptr %pos0_minus
OpStore %outColor %col0
OpEmitVertex

%p1_w1 = OpVectorTimesScalar %v4float %p1 %float_0_75
%p2_w1 = OpVectorTimesScalar %v4float %p2 %float_0_25
%pos1 = OpFAdd %v4float %p1_w1 %p2_w1
%c1_w1 = OpVectorTimesScalar %v3float %color1 %float_0_75
%c2_w1 = OpVectorTimesScalar %v3float %color2 %float_0_25
%col1 = OpFAdd %v3float %c1_w1 %c2_w1
%seg1 = OpFSub %v4float %p2 %p1
%tan1_xy = OpVectorShuffle %v2float %seg1 %seg1 0 1
%tan1_x = OpCompositeExtract %float %tan1_xy 0
%tan1_y = OpCompositeExtract %float %tan1_xy 1
%tan1_y_neg = OpFNegate %float %tan1_y
%perp1_xy = OpCompositeConstruct %v2float %tan1_y_neg %tan1_x
%perp1_len = OpExtInst %float %glsl Length %perp1_xy
%perp1_inv = OpFDiv %float %float_1 %perp1_len
%perp1_norm = OpVectorTimesScalar %v2float %perp1_xy %perp1_inv
%perp1_scaled = OpVectorTimesScalar %v2float %perp1_norm %float_0_02
%perp1_x = OpCompositeExtract %float %perp1_scaled 0
%perp1_y = OpCompositeExtract %float %perp1_scaled 1
%perp1 = OpCompositeConstruct %v4float %perp1_x %perp1_y %float_0 %float_0
%pos1_plus = OpFAdd %v4float %pos1 %perp1
%pos1_minus = OpFSub %v4float %pos1 %perp1
OpStore %out_pos_ptr %pos1_plus
OpStore %outColor %col1
OpEmitVertex
OpStore %out_pos_ptr %pos1_minus
OpStore %outColor %col1
OpEmitVertex

%p1_w2 = OpVectorTimesScalar %v4float %p1 %float_0_5
%p2_w2 = OpVectorTimesScalar %v4float %p2 %float_0_5
%pos2 = OpFAdd %v4float %p1_w2 %p2_w2
%c1_w2 = OpVectorTimesScalar %v3float %color1 %float_0_5
%c2_w2 = OpVectorTimesScalar %v3float %color2 %float_0_5
%col2 = OpFAdd %v3float %c1_w2 %c2_w2
%seg2 = OpFSub %v4float %p2 %p1
%tan2_xy = OpVectorShuffle %v2float %seg2 %seg2 0 1
%tan2_x = OpCompositeExtract %float %tan2_xy 0
%tan2_y = OpCompositeExtract %float %tan2_xy 1
%tan2_y_neg = OpFNegate %float %tan2_y
%perp2_xy = OpCompositeConstruct %v2float %tan2_y_neg %tan2_x
%perp2_len = OpExtInst %float %glsl Length %perp2_xy
%perp2_inv = OpFDiv %float %float_1 %perp2_len
%perp2_norm = OpVectorTimesScalar %v2float %perp2_xy %perp2_inv
%perp2_scaled = OpVectorTimesScalar %v2float %perp2_norm %float_0_02
%perp2_x = OpCompositeExtract %float %perp2_scaled 0
%perp2_y = OpCompositeExtract %float %perp2_scaled 1
%perp2 = OpCompositeConstruct %v4float %perp2_x %perp2_y %float_0 %float_0
%pos2_plus = OpFAdd %v4float %pos2 %perp2
%pos2_minus = OpFSub %v4float %pos2 %perp2
OpStore %out_pos_ptr %pos2_plus
OpStore %outColor %col2
OpEmitVertex
OpStore %out_pos_ptr %pos2_minus
OpStore %outColor %col2
OpEmitVertex

%p1_w3 = OpVectorTimesScalar %v4float %p1 %float_0_25
%p2_w3 = OpVectorTimesScalar %v4float %p2 %float_0_75
%pos3 = OpFAdd %v4float %p1_w3 %p2_w3
%c1_w3 = OpVectorTimesScalar %v3float %color1 %float_0_25
%c2_w3 = OpVectorTimesScalar %v3float %color2 %float_0_75
%col3 = OpFAdd %v3float %c1_w3 %c2_w3
%seg3 = OpFSub %v4float %p2 %p1
%tan3_xy = OpVectorShuffle %v2float %seg3 %seg3 0 1
%tan3_x = OpCompositeExtract %float %tan3_xy 0
%tan3_y = OpCompositeExtract %float %tan3_xy 1
%tan3_y_neg = OpFNegate %float %tan3_y
%perp3_xy = OpCompositeConstruct %v2float %tan3_y_neg %tan3_x
%perp3_len = OpExtInst %float %glsl Length %perp3_xy
%perp3_inv = OpFDiv %float %float_1 %perp3_len

OpStore %out_pos_ptr %pos3_plus
OpStore %outColor %col3
OpEmitVertex
OpStore %out_pos_ptr %pos3_minus
OpStore %outColor %col3
OpEmitVertex

%pos4 = OpCopy %v4float %p2
%col4 = OpCopy %v3float %color2
%tan4_xy = OpVectorShuffle %v2float %tangent_end %tangent_end 0 1
%tan4_x = OpCompositeExtract %float %tan4_xy 0
%tan4_y = OpCompositeExtract %float %tan4_xy 1

OpEmitVertex

OpEndPrimitive
