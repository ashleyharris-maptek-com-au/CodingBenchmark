using System;
using System.Collections.Generic;
using System.Linq;

class MinesweeperSolver
{
    static void Main()
    {
        var input = Console.ReadLine().Split();
        int H = int.Parse(input[0]);
        int W = int.Parse(input[1]);

        char[,] board = new char[H, W];
        for (int i = 0; i < H; i++)
        {
            string line = Console.ReadLine();
            for (int j = 0; j < W; j++)
            {
                board[i, j] = j < line.Length ? line[j] : ' ';
            }
        }

        var solver = new Solver(board, H, W);
        var safeCells = solver.FindSafeCells();

        Console.WriteLine(safeCells.Count);
        foreach (var cell in safeCells)
        {
            Console.WriteLine($"{cell.x} {cell.y}");
        }
    }
}

class Solver
{
    private char[,] board;
    private int H, W;
    private HashSet<(int x, int y)> knownMines = new HashSet<(int, int)>();
    private HashSet<(int x, int y)> knownSafe = new HashSet<(int, int)>();

    public Solver(char[,] board, int H, int W)
    {
        this.board = board;
        this.H = H;
        this.W = W;

        // Initialize known mines from flagged cells
        for (int y = 0; y < H; y++)
        {
            for (int x = 0; x < W; x++)
            {
                if (board[y, x] == '*')
                {
                    knownMines.Add((x, y));
                }
            }
        }
    }

    public List<(int x, int y)> FindSafeCells()
    {
        bool changed = true;
        int iterations = 0;
        int maxIterations = Math.Min(1000, H * W);

        // Determine strategy based on board size
        bool useAdvancedSolver = H * W > 10000;

        while (changed && iterations < maxIterations)
        {
            changed = false;
            iterations++;

            // Basic constraint solving
            for (int y = 0; y < H; y++)
            {
                for (int x = 0; x < W; x++)
                {
                    if (char.IsDigit(board[y, x]))
                    {
                        if (ProcessCell(x, y))
                        {
                            changed = true;
                        }
                    }
                }
            }

            // Advanced solving for complex cases
            if (!changed && useAdvancedSolver && iterations < 50)
            {
                changed = TryAdvancedDeduction();
            }
        }

        // Return only unknown cells that we proved safe
        var result = new List<(int x, int y)>();
        foreach (var (x, y) in knownSafe)
        {
            if (board[y, x] == ' ')
            {
                result.Add((x, y));
            }
        }

        result.Sort();
        return result;
    }

    private bool ProcessCell(int x, int y)
    {
        int mineCount = board[y, x] - '0';
        var neighbors = GetNeighbors(x, y);

        int adjacentMines = 0;
        int unknownCount = 0;
        var unknownCells = new List<(int, int)>();

        foreach (var (nx, ny) in neighbors)
        {
            if (knownMines.Contains((nx, ny)))
            {
                adjacentMines++;
            }
            else if (board[ny, nx] == ' ' && !knownSafe.Contains((nx, ny)))
            {
                unknownCount++;
                unknownCells.Add((nx, ny));
            }
        }

        bool changed = false;

        // If we've found all mines, remaining unknowns are safe
        if (adjacentMines == mineCount && unknownCount > 0)
        {
            foreach (var cell in unknownCells)
            {
                if (knownSafe.Add(cell))
                {
                    changed = true;
                }
            }
        }

        // If remaining unknowns must all be mines
        if (adjacentMines + unknownCount == mineCount && unknownCount > 0)
        {
            foreach (var cell in unknownCells)
            {
                if (knownMines.Add(cell))
                {
                    changed = true;
                }
            }
        }

        return changed;
    }

    private bool TryAdvancedDeduction()
    {
        // Collect all constraints
        var constraints = new List<Constraint>();
        var allUnknowns = new HashSet<(int, int)>();

        for (int y = 0; y < H; y++)
        {
            for (int x = 0; x < W; x++)
            {
                if (char.IsDigit(board[y, x]))
                {
                    int mineCount = board[y, x] - '0';
                    var neighbors = GetNeighbors(x, y);

                    int adjacentMines = 0;
                    var unknownNeighbors = new List<(int, int)>();

                    foreach (var (nx, ny) in neighbors)
                    {
                        if (knownMines.Contains((nx, ny)))
                        {
                            adjacentMines++;
                        }
                        else if (board[ny, nx] == ' ' && !knownSafe.Contains((nx, ny)))
                        {
                            unknownNeighbors.Add((nx, ny));
                            allUnknowns.Add((nx, ny));
                        }
                    }

                    if (unknownNeighbors.Count > 0)
                    {
                        constraints.Add(new Constraint
                        {
                            Cells = unknownNeighbors,
                            RemainingMines = mineCount - adjacentMines
                        });
                    }
                }
            }
        }

        if (constraints.Count == 0 || allUnknowns.Count == 0)
            return false;

        // Try subset reasoning
        return TrySubsetDeduction(constraints);
    }

    private bool TrySubsetDeduction(List<Constraint> constraints)
    {
        bool changed = false;

        // Look for constraint subsets and supersets
        for (int i = 0; i < constraints.Count; i++)
        {
            for (int j = i + 1; j < constraints.Count; j++)
            {
                var c1 = constraints[i];
                var c2 = constraints[j];

                var set1 = new HashSet<(int, int)>(c1.Cells);
                var set2 = new HashSet<(int, int)>(c2.Cells);

                // If c1 is subset of c2
                if (set1.IsSubsetOf(set2) && set1.Count < set2.Count)
                {
                    // Cells in c2 but not c1 have (c2.mines - c1.mines) mines
                    var diff = set2.Except(set1).ToList();
                    int diffMines = c2.RemainingMines - c1.RemainingMines;

                    if (diffMines == 0)
                    {
                        // All cells in diff are safe
                        foreach (var cell in diff)
                        {
                            if (knownSafe.Add(cell))
                                changed = true;
                        }
                    }
                    else if (diffMines == diff.Count)
                    {
                        // All cells in diff are mines
                        foreach (var cell in diff)
                        {
                            if (knownMines.Add(cell))
                                changed = true;
                        }
                    }
                }

                // If c2 is subset of c1
                if (set2.IsSubsetOf(set1) && set2.Count < set1.Count)
                {
                    var diff = set1.Except(set2).ToList();
                    int diffMines = c1.RemainingMines - c2.RemainingMines;

                    if (diffMines == 0)
                    {
                        foreach (var cell in diff)
                        {
                            if (knownSafe.Add(cell))
                                changed = true;
                        }
                    }
                    else if (diffMines == diff.Count)
                    {
                        foreach (var cell in diff)
                        {
                            if (knownMines.Add(cell))
                                changed = true;
                        }
                    }
                }
            }
        }

        return changed;
    }

    private List<(int x, int y)> GetNeighbors(int x, int y)
    {
        var neighbors = new List<(int, int)>();
        for (int dy = -1; dy <= 1; dy++)
        {
            for (int dx = -1; dx <= 1; dx++)
            {
                if (dx == 0 && dy == 0) continue;

                int nx = x + dx;
                int ny = y + dy;

                if (nx >= 0 && nx < W && ny >= 0 && ny < H)
                {
                    neighbors.Add((nx, ny));
                }
            }
        }
        return neighbors;
    }
}

class Constraint
{
    public List<(int x, int y)> Cells { get; set; }
    public int RemainingMines { get; set; }
}
