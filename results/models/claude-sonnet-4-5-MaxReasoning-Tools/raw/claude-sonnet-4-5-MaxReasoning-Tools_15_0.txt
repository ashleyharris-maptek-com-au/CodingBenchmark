use std::io::{self, BufRead};

#[derive(Debug, Clone, Copy)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Vec3 { x, y, z }
    }

    fn dot(&self, other: &Vec3) -> f64 {
        self.x * other.x + self.y * other.y + self.z * other.z
    }

    fn cross(&self, other: &Vec3) -> Vec3 {
        Vec3::new(
            self.y * other.z - self.z * other.y,
            self.z * other.x - self.x * other.z,
            self.x * other.y - self.y * other.x,
        )
    }

    fn length(&self) -> f64 {
        (self.x * self.x + self.y * self.y + self.z * self.z).sqrt()
    }

    fn normalize(&self) -> Vec3 {
        let len = self.length();
        Vec3::new(self.x / len, self.y / len, self.z / len)
    }

    fn scale(&self, s: f64) -> Vec3 {
        Vec3::new(self.x * s, self.y * s, self.z * s)
    }

    fn add(&self, other: &Vec3) -> Vec3 {
        Vec3::new(self.x + other.x, self.y + other.y, self.z + other.z)
    }

    fn sub(&self, other: &Vec3) -> Vec3 {
        Vec3::new(self.x - other.x, self.y - other.y, self.z - other.z)
    }
}

#[derive(Debug, Clone, Copy)]
struct Quaternion {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

impl Quaternion {
    fn new(w: f64, x: f64, y: f64, z: f64) -> Self {
        Quaternion { w, x, y, z }
    }

    fn identity() -> Self {
        Quaternion::new(1.0, 0.0, 0.0, 0.0)
    }

    fn from_axis_angle(axis: &Vec3, angle: f64) -> Self {
        let half_angle = angle / 2.0;
        let s = half_angle.sin();
        let normalized = axis.normalize();
        Quaternion::new(
            half_angle.cos(),
            normalized.x * s,
            normalized.y * s,
            normalized.z * s,
        )
    }

    fn rotate_vector(&self, v: &Vec3) -> Vec3 {
        let qv = Vec3::new(self.x, self.y, self.z);
        let uv = qv.cross(v);
        let uuv = qv.cross(&uv);
        let uv = uv.scale(2.0 * self.w);
        let uuv = uuv.scale(2.0);
        v.add(&uv).add(&uuv)
    }
}

#[derive(Debug, Clone)]
struct Tetrahedron {
    position: Vec3,
    rotation: Quaternion,
    scale: f64,
}

impl Tetrahedron {
    fn get_vertices(&self) -> Vec<Vec3> {
        let base_vertices = vec![
            Vec3::new(1.0, 0.0, -0.707),
            Vec3::new(-1.0, 0.0, -0.707),
            Vec3::new(0.0, 1.0, 0.707),
            Vec3::new(0.0, -1.0, 0.707),
        ];

        base_vertices
            .iter()
            .map(|v| {
                let scaled = v.scale(self.scale);
                let rotated = self.rotation.rotate_vector(&scaled);
                rotated.add(&self.position)
            })
            .collect()
    }
}

fn compute_polygon_bounds(polygon: &[Vec3]) -> (f64, f64, f64, f64) {
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;

    for v in polygon {
        min_x = min_x.min(v.x);
        max_x = max_x.max(v.x);
        min_y = min_y.min(v.y);
        max_y = max_y.max(v.y);
    }

    (min_x, max_x, min_y, max_y)
}

fn compute_polygon_centroid(polygon: &[Vec3]) -> Vec3 {
    let sum = polygon.iter().fold(Vec3::new(0.0, 0.0, 0.0), |acc, v| {
        acc.add(v)
    });
    let n = polygon.len() as f64;
    Vec3::new(sum.x / n, sum.y / n, 0.0)
}

fn point_in_polygon(point: &Vec3, polygon: &[Vec3]) -> bool {
    let mut inside = false;
    let n = polygon.len();

    for i in 0..n {
        let j = (i + 1) % n;
        let vi = &polygon[i];
        let vj = &polygon[j];

        if ((vi.y > point.y) != (vj.y > point.y)) &&
           (point.x < (vj.x - vi.x) * (point.y - vi.y) / (vj.y - vi.y) + vi.x) {
            inside = !inside;
        }
    }

    inside
}

fn convex_hull(points: &[Vec3]) -> Vec<Vec3> {
    if points.is_empty() {
        return Vec::new();
    }

    let mut pts = points.to_vec();
    pts.sort_by(|a, b| {
        a.x.partial_cmp(&b.x)
            .unwrap()
            .then(a.y.partial_cmp(&b.y).unwrap())
    });

    let cross = |o: &Vec3, a: &Vec3, b: &Vec3| -> f64 {
        (a.x - o.x) * (b.y - o.y) - (a.y - o.y) * (b.x - o.x)
    };

    let mut lower = Vec::new();
    for p in &pts {
        while lower.len() >= 2 && cross(&lower[lower.len() - 2], &lower[lower.len() - 1], p) <= 0.0 {
            lower.pop();
        }
        lower.push(*p);
    }

    let mut upper = Vec::new();
    for p in pts.iter().rev() {
        while upper.len() >= 2 && cross(&upper[upper.len() - 2], &upper[upper.len() - 1], p) <= 0.0 {
            upper.pop();
        }
        upper.push(*p);
    }

    lower.pop();
    upper.pop();
    lower.extend(upper);
    lower
}

fn create_covering_tetrahedrons(polygon: &[Vec3], sun_dir: &Vec3) -> Vec<Tetrahedron> {
    let centroid = compute_polygon_centroid(polygon);
    let (min_x, max_x, min_y, max_y) = compute_polygon_bounds(polygon);

    let width = max_x - min_x;
    let height = max_y - min_y;
    let max_dim = width.max(height);

    let hull = convex_hull(polygon);
    let is_simple = hull.len() == polygon.len();

    let mut tetrahedrons = Vec::new();

    if is_simple && polygon.len() <= 6 {
        let scale = (max_dim * 1.5).min(10.0);
        let height_above = scale * 2.0;

        let sun_normalized = sun_dir.normalize();
        let up = Vec3::new(0.0, 0.0, 1.0);
        let axis = up.cross(&sun_normalized);
        let axis_len = axis.length();

        let rotation = if axis_len > 1e-6 {
            let angle = up.dot(&sun_normalized).acos();
            Quaternion::from_axis_angle(&axis.normalize(), angle)
        } else {
            Quaternion::identity()
        };

        let position = Vec3::new(centroid.x, centroid.y, height_above);

        tetrahedrons.push(Tetrahedron {
            position,
            rotation,
            scale,
        });
    } else {
        let num_subdivisions = if polygon.len() > 20 { 4 } else { 3 };
        let cell_size = max_dim / num_subdivisions as f64;
        let scale = (cell_size * 0.8).min(10.0);
        let height_above = scale * 1.5;

        let sun_normalized = sun_dir.normalize();
        let up = Vec3::new(0.0, 0.0, 1.0);
        let axis = up.cross(&sun_normalized);
        let axis_len = axis.length();

        let rotation = if axis_len > 1e-6 {
            let angle = up.dot(&sun_normalized).acos();
            Quaternion::from_axis_angle(&axis.normalize(), angle)
        } else {
            Quaternion::identity()
        };

        for i in 0..num_subdivisions {
            for j in 0..num_subdivisions {
                let x = min_x + (i as f64 + 0.5) * cell_size;
                let y = min_y + (j as f64 + 0.5) * cell_size;

                let test_point = Vec3::new(x, y, 0.0);
                if point_in_polygon(&test_point, polygon) {
                    let position = Vec3::new(x, y, height_above);
                    tetrahedrons.push(Tetrahedron {
                        position,
                        rotation,
                        scale,
                    });
                }
            }
        }

        if tetrahedrons.is_empty() {
            let scale = (max_dim * 1.5).min(10.0);
            let position = Vec3::new(centroid.x, centroid.y, scale * 2.0);
            tetrahedrons.push(Tetrahedron {
                position,
                rotation,
                scale,
            });
        }
    }

    tetrahedrons
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let nv: usize = lines.next().unwrap().unwrap().trim().parse().unwrap();

    let mut polygon = Vec::new();
    for _ in 0..nv {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<f64> = line
            .split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        polygon.push(Vec3::new(coords[0], coords[1], 0.0));
    }

    let sun_line = lines.next().unwrap().unwrap();
    let sun_coords: Vec<f64> = sun_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let sun_dir = Vec3::new(sun_coords[0], sun_coords[1], sun_coords[2]);

    let tetrahedrons = create_covering_tetrahedrons(&polygon, &sun_dir);

    println!("{}", tetrahedrons.len());
    for tet in &tetrahedrons {
        println!(
            "{} {} {} {} {} {} {} {}",
            tet.position.x,
            tet.position.y,
            tet.position.z,
            tet.rotation.w,
            tet.rotation.x,
            tet.rotation.y,
            tet.rotation.z,
            tet.scale
        );
    }
}
