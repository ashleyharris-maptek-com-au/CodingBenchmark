#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#include <bitset>
#include <cstring>

using namespace std;

class MaxIndependentSet {
private:
    int n;
    vector<vector<int>> adj;
    vector<int> best_set;
    int best_size;

    // Check if graph is bipartite and return max independent set if it is
    bool solveBipartite(vector<int>& result) {
        vector<int> color(n, -1);
        vector<int> side0, side1;

        for (int start = 0; start < n; start++) {
            if (color[start] != -1) continue;

            queue<int> q;
            q.push(start);
            color[start] = 0;

            while (!q.empty()) {
                int u = q.front();
                q.pop();

                for (int v : adj[u]) {
                    if (color[v] == -1) {
                        color[v] = 1 - color[u];
                        q.push(v);
                    } else if (color[v] == color[u]) {
                        return false; // Not bipartite
                    }
                }
            }
        }

        // Graph is bipartite - take the larger side
        for (int i = 0; i < n; i++) {
            if (color[i] == 0) side0.push_back(i);
            else side1.push_back(i);
        }

        result = (side0.size() >= side1.size()) ? side0 : side1;
        return true;
    }

    // Greedy coloring for upper bound
    int greedyColoring(const vector<bool>& available) {
        vector<int> color(n, -1);
        int num_colors = 0;

        for (int u = 0; u < n; u++) {
            if (!available[u]) continue;

            vector<bool> used_colors(n, false);
            for (int v : adj[u]) {
                if (available[v] && color[v] != -1) {
                    used_colors[color[v]] = true;
                }
            }

            for (int c = 0; c < n; c++) {
                if (!used_colors[c]) {
                    color[u] = c;
                    num_colors = max(num_colors, c + 1);
                    break;
                }
            }
        }

        return num_colors;
    }

    // Branch and bound solver
    void branchAndBound(vector<int>& current_set, vector<bool>& in_set,
                        vector<bool>& available, int start_vertex) {
        if (current_set.size() > best_size) {
            best_size = current_set.size();
            best_set = current_set;
        }

        // Upper bound: current size + greedy coloring of remaining
        int upper_bound = current_set.size() + greedyColoring(available);
        if (upper_bound <= best_size) return;

        // Find best vertex to branch on (highest degree in available vertices)
        int best_vertex = -1;
        int max_degree = -1;
        for (int v = start_vertex; v < n; v++) {
            if (!available[v]) continue;
            int degree = 0;
            for (int u : adj[v]) {
                if (available[u]) degree++;
            }
            if (degree > max_degree) {
                max_degree = degree;
                best_vertex = v;
            }
        }

        if (best_vertex == -1) return; // No more vertices

        // Branch 1: Include best_vertex
        current_set.push_back(best_vertex);
        in_set[best_vertex] = true;
        available[best_vertex] = false;

        vector<int> disabled;
        for (int u : adj[best_vertex]) {
            if (available[u]) {
                available[u] = false;
                disabled.push_back(u);
            }
        }

        branchAndBound(current_set, in_set, available, best_vertex + 1);

        // Restore
        for (int u : disabled) {
            available[u] = true;
        }
        current_set.pop_back();
        in_set[best_vertex] = false;
        available[best_vertex] = true;

        // Branch 2: Exclude best_vertex
        available[best_vertex] = false;
        branchAndBound(current_set, in_set, available, best_vertex + 1);
        available[best_vertex] = true;
    }

public:
    MaxIndependentSet(int n) : n(n), adj(n), best_size(0) {}

    void addEdge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<int> solve() {
        vector<int> result;

        // Special case: empty graph
        if (n == 0) return result;

        // Try bipartite algorithm first
        if (solveBipartite(result)) {
            return result;
        }

        // For small graphs, use branch and bound
        if (n <= 60) {
            vector<int> current_set;
            vector<bool> in_set(n, false);
            vector<bool> available(n, true);

            branchAndBound(current_set, in_set, available, 0);
            return best_set;
        }

        // For larger graphs, use greedy heuristic
        // (This shouldn't happen if test cases are reasonable)
        vector<bool> in_set(n, false);
        vector<bool> available(n, true);

        while (true) {
            int min_degree = n + 1;
            int best_v = -1;

            for (int v = 0; v < n; v++) {
                if (!available[v]) continue;
                int degree = 0;
                for (int u : adj[v]) {
                    if (available[u]) degree++;
                }
                if (degree < min_degree) {
                    min_degree = degree;
                    best_v = v;
                }
            }

            if (best_v == -1) break;

            result.push_back(best_v);
            in_set[best_v] = true;
            available[best_v] = false;

            for (int u : adj[best_v]) {
                available[u] = false;
            }
        }

        return result;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    MaxIndependentSet mis(n);

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        mis.addEdge(u, v);
    }

    vector<int> result = mis.solve();

    cout << result.size() << "\n";
    for (size_t i = 0; i < result.size(); i++) {
        if (i > 0) cout << " ";
        cout << result[i];
    }
    if (!result.empty()) cout << "\n";

    return 0;
}
