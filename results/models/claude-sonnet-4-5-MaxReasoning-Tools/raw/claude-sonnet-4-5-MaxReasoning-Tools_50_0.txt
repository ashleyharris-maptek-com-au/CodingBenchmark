#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against out of bounds
    if (i >= N) {
        return;
    }

    // Extract parameters
    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;
    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;
    float boundSize = bounds.x;
    float boundForce = bounds.y;

    // Read current boid state
    vec3 pos = inData[i * 2].xyz;
    vec3 vel = inData[i * 2 + 1].xyz;

    // Flocking forces
    vec3 separation = vec3(0.0);
    vec3 alignment = vec3(0.0);
    vec3 cohesion = vec3(0.0);

    uint sepCount = 0;
    uint alignCount = 0;
    uint cohCount = 0;

    // Loop through all other boids
    for (uint j = 0; j < N; j++) {
        if (i == j) continue;  // Skip self

        vec3 otherPos = inData[j * 2].xyz;
        vec3 otherVel = inData[j * 2 + 1].xyz;

        vec3 diff = pos - otherPos;
        float dist = length(diff);

        // Skip zero-distance neighbors
        if (dist < 0.0001) continue;

        // Separation: steer away from nearby boids
        if (dist < sepRadius) {
            separation += diff / dist;  // Normalized direction away
            sepCount++;
        }

        // Alignment: match velocity with nearby boids
        if (dist < alignRadius) {
            alignment += otherVel;
            alignCount++;
        }

        // Cohesion: steer toward average position
        if (dist < cohRadius) {
            cohesion += otherPos;
            cohCount++;
        }
    }

    // Average the forces
    vec3 accel = vec3(0.0);

    if (sepCount > 0) {
        separation /= float(sepCount);
        accel += separation * sepWeight;
    }

    if (alignCount > 0) {
        alignment /= float(alignCount);
        // Alignment is a steering force toward the average velocity
        accel += (alignment - vel) * alignWeight;
    }

    if (cohCount > 0) {
        cohesion /= float(cohCount);
        // Cohesion is a steering force toward the average position
        accel += (cohesion - pos) * cohWeight;
    }

    // Boundary force: push toward center if too far out
    float boundThreshold = boundSize * 0.4;
    float boundScale = boundSize * 0.5;

    if (abs(pos.x) > boundThreshold) {
        accel.x -= (pos.x / boundScale) * boundForce;
    }
    if (abs(pos.y) > boundThreshold) {
        accel.y -= (pos.y / boundScale) * boundForce;
    }
    if (abs(pos.z) > boundThreshold) {
        accel.z -= (pos.z / boundScale) * boundForce;
    }

    // Update velocity
    vel += accel * dt;

    // Clamp speed to maxSpeed
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = (vel / speed) * maxSpeed;
    }

    // Update position
    pos += vel * dt;

    // Write output
    outData[i * 2] = vec4(pos, 0.0);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
