; Generator: Manual
; Bound: 100

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_uv %in_normal %in_worldPos %out_color
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

; Decorations for inputs/outputs
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %out_color Location 0

; Decorations for UBO
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%void_fn = OpTypeFunction %void
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%int = OpTypeInt 32 1
%bool = OpTypeBool

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4

%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4

; Constants
%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_9 = OpConstant %float 0.9
%float_1 = OpConstant %float 1.0
%float_8 = OpConstant %float 8.0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_3 = OpConstant %int 3

; GLSL extended instruction set
%glsl_ext = OpExtInstImport "GLSL.std.450"

; Variables
%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%in_uv = OpVariable %ptr_input_vec2 Input
%out_color = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_UBO Uniform

; Main function
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Load inputs
%uv = OpLoad %vec2 %in_uv
%normal = OpLoad %vec3 %in_normal
%worldPos = OpLoad %vec3 %in_worldPos

; Normalize normal vector
%normal_normalized = OpExtInst %vec3 %glsl_ext Normalize %normal

; Get light position from UBO (member 3, offset 192)
%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%lightPos_vec4 = OpLoad %vec4 %lightPos_ptr
%lightPos = OpVectorShuffle %vec3 %lightPos_vec4 %lightPos_vec4 0 1 2

; Calculate light direction: normalize(lightPos - worldPos)
%lightDir_unnorm = OpFSub %vec3 %lightPos %worldPos
%lightDir = OpExtInst %vec3 %glsl_ext Normalize %lightDir_unnorm

; Calculate Lambertian diffuse: max(0, dot(normal, lightDir))
%diffuse_dot = OpDot %float %normal_normalized %lightDir
%diffuse = OpExtInst %float %glsl_ext FMax %diffuse_dot %float_0

; Calculate shading factor: 0.5 + 0.5 * diffuse
%diffuse_scaled = OpFMul %float %float_0_5 %diffuse
%shading = OpFAdd %float %float_0_5 %diffuse_scaled

; Calculate UV checkerboard at 8x8 frequency
%uv_scaled = OpVectorTimesScalar %vec2 %uv %float_8
%uv_floored = OpExtInst %vec2 %glsl_ext Floor %uv_scaled

; Extract u and v components
%u_floor = OpCompositeExtract %float %uv_floored 0
%v_floor = OpCompositeExtract %float %uv_floored 1

; Convert to int
%u_int = OpConvertFToS %int %u_floor
%v_int = OpConvertFToS %int %v_floor

; Compute parity using XOR of least significant bits
%u_bit = OpBitwiseAnd %int %u_int %int_1
%v_bit = OpBitwiseAnd %int %v_int %int_1
%parity_int = OpBitwiseXor %int %u_bit %v_bit

; Check if parity is 1
%is_odd = OpIEqual %bool %parity_int %int_1

; Select checker color: 0.9 if odd, 0.2 if even
%checker_color = OpSelect %float %is_odd %float_0_9 %float_0_2

; Apply shading to checker color
%final_intensity = OpFMul %float %checker_color %shading

; Build output color (grayscale with alpha 1.0)
%final_color = OpCompositeConstruct %vec4 %final_intensity %final_intensity %final_intensity %float_1

; Store output
OpStore %out_color %final_color

OpReturn
OpFunctionEnd
