OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpSource GLSL 450

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%void_func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_h = OpConstant %float 0.04

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_Input_arr_gl_PerVertex = OpTypePointer Input %arr_gl_PerVertex_in
%ptr_Input_v4float = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float

%arr_v3float_1 = OpTypeArray %v3float %uint_1
%ptr_Input_arr_v3float = OpTypePointer Input %arr_v3float_1
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex Input
%color_in = OpVariable %ptr_Input_arr_v3float Input
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %ptr_Output_v3float Output

%main = OpFunction %void None %void_func
%entry = OpLabel

%in_pos_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%in_pos = OpLoad %v4float %in_pos_ptr

%in_color_ptr = OpAccessChain %ptr_Input_v3float %color_in %int_0
%in_color = OpLoad %v3float %in_color_ptr

%x = OpCompositeExtract %float %in_pos 0
%y = OpCompositeExtract %float %in_pos 1

%x_minus_h = OpFSub %float %x %float_h
%x_plus_h = OpFAdd %float %x %float_h
%y_minus_h = OpFSub %float %y %float_h
%y_plus_h = OpFAdd %float %y %float_h

%out_pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0

%pos0 = OpCompositeConstruct %v4float %x_minus_h %y_minus_h %float_0 %float_1
OpStore %out_pos_ptr %pos0
OpStore %outColor %in_color
OpEmitVertex

%pos1 = OpCompositeConstruct %v4float %x_plus_h %y_minus_h %float_0 %float_1
OpStore %out_pos_ptr %pos1
OpStore %outColor %in_color
OpEmitVertex

%pos2 = OpCompositeConstruct %v4float %x_minus_h %y_plus_h %float_0 %float_1
OpStore %out_pos_ptr %pos2
OpStore %outColor %in_color
OpEmitVertex

%pos3 = OpCompositeConstruct %v4float %x_plus_h %y_plus_h %float_0 %float_1
OpStore %out_pos_ptr %pos3
OpStore %outColor %in_color
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
