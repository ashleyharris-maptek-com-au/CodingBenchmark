OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
     %uint_2 = OpConstant %uint 2
  %thickness = OpConstant %float 0.03
    %float_0 = OpConstant %float 0
    %float_1 = OpConstant %float 1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in_2 = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in_2
      %gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4float = OpTypePointer Input %v4float

%arr_color_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_color = OpTypePointer Input %arr_color_in
   %color_in = OpVariable %ptr_in_arr_color Input
%ptr_in_v3float = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4float = OpTypePointer Output %v4float

%ptr_out_v3float = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3float Output

       %main = OpFunction %void None %3
      %entry = OpLabel

         ; Load p0 = gl_in[0].gl_Position
         %p0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
            %p0 = OpLoad %v4float %p0_ptr

         ; Load p1 = gl_in[1].gl_Position
         %p1_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
            %p1 = OpLoad %v4float %p1_ptr

         ; Load c0 = color_in[0]
         %c0_ptr = OpAccessChain %ptr_in_v3float %color_in %int_0
            %c0 = OpLoad %v3float %c0_ptr

         ; Load c1 = color_in[1]
         %c1_ptr = OpAccessChain %ptr_in_v3float %color_in %int_1
            %c1 = OpLoad %v3float %c1_ptr

         ; Compute dir = p1 - p0 (only xy components)
         %dir = OpFSub %v4float %p1 %p0
         %dir_x = OpCompositeExtract %float %dir 0
         %dir_y = OpCompositeExtract %float %dir 1

         ; Compute perpendicular: perp = (-dir.y, dir.x)
         %perp_x_raw = OpFNegate %float %dir_y
         %perp_y_raw = OpCopy %float %dir_x

         ; Normalize perpendicular
         %perp_raw = OpCompositeConstruct %v2float %perp_x_raw %perp_y_raw
         %perp_normalized = OpExtInst %v2float %1 Normalize %perp_raw

         ; Scale by thickness
         %perp_scaled = OpVectorTimesScalar %v2float %perp_normalized %thickness
         %perp_x = OpCompositeExtract %float %perp_scaled 0
         %perp_y = OpCompositeExtract %float %perp_scaled 1

         ; Get p0 and p1 components
         %p0_x = OpCompositeExtract %float %p0 0
         %p0_y = OpCompositeExtract %float %p0 1
         %p0_z = OpCompositeExtract %float %p0 2
         %p0_w = OpCompositeExtract %float %p0 3
         %p1_x = OpCompositeExtract %float %p1 0
         %p1_y = OpCompositeExtract %float %p1 1
         %p1_z = OpCompositeExtract %float %p1 2
         %p1_w = OpCompositeExtract %float %p1 3

         ; Vertex 0: p0 + perp
         %v0_x = OpFAdd %float %p0_x %perp_x
         %v0_y = OpFAdd %float %p0_y %perp_y
         %v0 = OpCompositeConstruct %v4float %v0_x %v0_y %p0_z %p0_w
         %gl_pos_ptr_0 = OpAccessChain %ptr_out_v4float %gl_out %int_0
         OpStore %gl_pos_ptr_0 %v0
         OpStore %outColor %c0
         OpEmitVertex

         ; Vertex 1: p0 - perp
         %v1_x = OpFSub %float %p0_x %perp_x
         %v1_y = OpFSub %float %p0_y %perp_y
         %v1 = OpCompositeConstruct %v4float %v1_x %v1_y %p0_z %p0_w
         %gl_pos_ptr_1 = OpAccessChain %ptr_out_v4float %gl_out %int_0
         OpStore %gl_pos_ptr_1 %v1
         OpStore %outColor %c0
         OpEmitVertex

         ; Vertex 2: p1 + perp
         %v2_x = OpFAdd %float %p1_x %perp_x
         %v2_y = OpFAdd %float %p1_y %perp_y
         %v2 = OpCompositeConstruct %v4float %v2_x %v2_y %p1_z %p1_w
         %gl_pos_ptr_2 = OpAccessChain %ptr_out_v4float %gl_out %int_0
         OpStore %gl_pos_ptr_2 %v2
         OpStore %outColor %c1
         OpEmitVertex

         ; Vertex 3: p1 - perp
         %v3_x = OpFSub %float %p1_x %perp_x
         %v3_y = OpFSub %float %p1_y %perp_y
         %v3 = OpCompositeConstruct %v4float %v3_x %v3_y %p1_z %p1_w
         %gl_pos_ptr_3 = OpAccessChain %ptr_out_v4float %gl_out %int_0
         OpStore %gl_pos_ptr_3 %v3
         OpStore %outColor %c1
         OpEmitVertex

         OpEndPrimitive
         OpReturn
         OpFunctionEnd
