using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        var input = Console.ReadLine().Split();
        int n = int.Parse(input[0]);
        int m = int.Parse(input[1]);

        var graph = new List<int>[n];
        for (int i = 0; i < n; i++)
            graph[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            var edge = Console.ReadLine().Split();
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            graph[u].Add(v);
        }

        var fvs = FindMinimumFVS(graph, n);

        Console.WriteLine(fvs.Count);
        if (fvs.Count > 0)
            Console.WriteLine(string.Join(" ", fvs.OrderBy(x => x)));
    }

    static List<int> FindMinimumFVS(List<int>[] graph, int n)
    {
        // Find strongly connected components
        var sccs = FindSCCs(graph, n);
        var result = new List<int>();

        foreach (var scc in sccs)
        {
            if (scc.Count == 1)
            {
                int v = scc[0];
                // Check for self-loop
                if (graph[v].Contains(v))
                    result.Add(v);
            }
            else
            {
                // Find minimum FVS for this SCC
                var sccFvs = FindSCCFVS(graph, scc);
                result.AddRange(sccFvs);
            }
        }

        return result;
    }

    static List<int> FindSCCFVS(List<int>[] graph, List<int> scc)
    {
        int size = scc.Count;

        // For small SCCs, use exhaustive search
        if (size <= 18)
        {
            // Try all subsets from size 1 to size-1
            for (int k = 1; k < size; k++)
            {
                var result = new List<int>();
                if (TryFindFVS(graph, scc, k, 0, result))
                    return result;
            }
        }
        else
        {
            // For larger SCCs, use greedy with local search
            return FindFVSGreedy(graph, scc);
        }

        return scc; // Fallback: remove all vertices
    }

    static bool TryFindFVS(List<int>[] graph, List<int> scc, int k, int start, List<int> current)
    {
        if (current.Count == k)
        {
            return IsAcyclic(graph, scc, current.ToHashSet());
        }

        for (int i = start; i <= scc.Count - (k - current.Count); i++)
        {
            current.Add(scc[i]);
            if (TryFindFVS(graph, scc, k, i + 1, current))
                return true;
            current.RemoveAt(current.Count - 1);
        }

        return false;
    }

    static List<int> FindFVSGreedy(List<int>[] graph, List<int> scc)
    {
        var sccSet = scc.ToHashSet();
        var removed = new HashSet<int>();

        // Greedy: repeatedly remove vertex with highest cycle involvement
        while (true)
        {
            if (IsAcyclic(graph, scc, removed))
                break;

            // Find vertex in most cycles (approximate by in-degree + out-degree in SCC)
            int bestVertex = -1;
            int bestScore = -1;

            foreach (var v in scc)
            {
                if (removed.Contains(v))
                    continue;

                int score = 0;
                foreach (var u in graph[v])
                {
                    if (sccSet.Contains(u) && !removed.Contains(u))
                        score++;
                }

                // Also count incoming edges (approximate)
                foreach (var u in scc)
                {
                    if (!removed.Contains(u) && graph[u].Contains(v))
                        score++;
                }

                if (score > bestScore)
                {
                    bestScore = score;
                    bestVertex = v;
                }
            }

            if (bestVertex != -1)
                removed.Add(bestVertex);
            else
                break;
        }

        // Try to remove vertices one by one (local optimization)
        var result = removed.ToList();
        for (int i = 0; i < result.Count; i++)
        {
            var test = removed.ToHashSet();
            test.Remove(result[i]);
            if (IsAcyclic(graph, scc, test))
            {
                removed = test;
                result.RemoveAt(i);
                i--;
            }
        }

        return removed.ToList();
    }

    static bool IsAcyclic(List<int>[] graph, List<int> scc, HashSet<int> removed)
    {
        var active = scc.Where(v => !removed.Contains(v)).ToHashSet();
        if (active.Count == 0)
            return true;

        var visited = new HashSet<int>();
        var recStack = new HashSet<int>();

        foreach (var v in active)
        {
            if (!visited.Contains(v))
            {
                if (HasCycleDFS(graph, v, active, visited, recStack))
                    return false;
            }
        }

        return true;
    }

    static bool HasCycleDFS(List<int>[] graph, int v, HashSet<int> active,
                           HashSet<int> visited, HashSet<int> recStack)
    {
        visited.Add(v);
        recStack.Add(v);

        foreach (var u in graph[v])
        {
            if (!active.Contains(u))
                continue;

            if (!visited.Contains(u))
            {
                if (HasCycleDFS(graph, u, active, visited, recStack))
                    return true;
            }
            else if (recStack.Contains(u))
            {
                return true;
            }
        }

        recStack.Remove(v);
        return false;
    }

    static List<List<int>> FindSCCs(List<int>[] graph, int n)
    {
        var index = new int[n];
        var lowlink = new int[n];
        var onStack = new bool[n];
        var stack = new Stack<int>();
        var result = new List<List<int>>();
        int idx = 0;

        for (int i = 0; i < n; i++)
            index[i] = -1;

        for (int i = 0; i < n; i++)
        {
            if (index[i] == -1)
                TarjanDFS(graph, i, index, lowlink, onStack, stack, result, ref idx);
        }

        return result;
    }

    static void TarjanDFS(List<int>[] graph, int v, int[] index, int[] lowlink,
                         bool[] onStack, Stack<int> stack, List<List<int>> result, ref int idx)
    {
        index[v] = idx;
        lowlink[v] = idx;
        idx++;
        stack.Push(v);
        onStack[v] = true;

        foreach (var w in graph[v])
        {
            if (index[w] == -1)
            {
                TarjanDFS(graph, w, index, lowlink, onStack, stack, result, ref idx);
                lowlink[v] = Math.Min(lowlink[v], lowlink[w]);
            }
            else if (onStack[w])
            {
                lowlink[v] = Math.Min(lowlink[v], index[w]);
            }
        }

        if (lowlink[v] == index[v])
        {
            var scc = new List<int>();
            int w;
            do
            {
                w = stack.Pop();
                onStack[w] = false;
                scc.Add(w);
            } while (w != v);

            result.Add(scc);
        }
    }
}
