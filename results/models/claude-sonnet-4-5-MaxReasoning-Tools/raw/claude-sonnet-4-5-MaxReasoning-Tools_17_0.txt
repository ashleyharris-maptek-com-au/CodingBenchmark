using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        // Parse input
        var firstLine = Console.ReadLine().Split();
        int N = int.Parse(firstLine[0]);
        int W = int.Parse(firstLine[1]);
        int H = int.Parse(firstLine[2]);
        int D = int.Parse(firstLine[3]);

        var boxes = new List<Box>();
        for (int i = 0; i < N; i++)
        {
            var dims = Console.ReadLine().Split();
            boxes.Add(new Box(i, int.Parse(dims[0]), int.Parse(dims[1]), int.Parse(dims[2])));
        }

        // Pack boxes
        var solution = PackBoxes(boxes, W, H, D);

        // Output
        Console.WriteLine(solution.Count);
        foreach (var p in solution)
        {
            Console.WriteLine($"{p.BoxIndex} {p.X} {p.Y} {p.Z}");
        }
    }

    static List<Placement> PackBoxes(List<Box> boxes, int containerW, int containerH, int containerD)
    {
        // Sort by volume descending (larger first for better packing)
        boxes = boxes.OrderByDescending(b => (long)b.W * b.H * b.D).ToList();

        var placements = new List<Placement>();
        var occupied = new List<Box3D>();

        // Track extreme points where boxes can be placed
        var extremePoints = new SortedSet<Point3D>(Comparer<Point3D>.Create((a, b) =>
        {
            int cmp = a.X.CompareTo(b.X);
            if (cmp != 0) return cmp;
            cmp = a.Y.CompareTo(b.Y);
            if (cmp != 0) return cmp;
            return a.Z.CompareTo(b.Z);
        }));
        extremePoints.Add(new Point3D(0, 0, 0));

        foreach (var box in boxes)
        {
            var orientations = GetOrientations(box.W, box.H, box.D);
            bool placed = false;

            // Try each orientation
            foreach (var (w, h, d) in orientations)
            {
                if (placed) break;

                // Try placing at each extreme point
                var pointsList = extremePoints.ToList();
                foreach (var point in pointsList)
                {
                    // Check if box fits in container
                    if (point.X + w > containerW || point.Y + h > containerH || point.Z + d > containerD)
                        continue;

                    var candidate = new Box3D(point.X, point.Y, point.Z, w, h, d);

                    // Check for overlaps with existing boxes
                    if (!occupied.Any(existing => Overlaps(existing, candidate)))
                    {
                        // Place the box
                        placements.Add(new Placement(box.Index, point.X, point.Y, point.Z));
                        occupied.Add(candidate);
                        placed = true;

                        // Generate new extreme points from this placement
                        var newPoints = new List<Point3D>
                        {
                            new Point3D(point.X + w, point.Y, point.Z),
                            new Point3D(point.X, point.Y + h, point.Z),
                            new Point3D(point.X, point.Y, point.Z + d)
                        };

                        foreach (var np in newPoints)
                        {
                            if (np.X <= containerW && np.Y <= containerH && np.Z <= containerD)
                            {
                                extremePoints.Add(np);
                            }
                        }

                        break;
                    }
                }
            }
        }

        return placements;
    }

    static bool Overlaps(Box3D a, Box3D b)
    {
        return !(a.X + a.W <= b.X || b.X + b.W <= a.X ||
                 a.Y + a.H <= b.Y || b.Y + b.H <= a.Y ||
                 a.Z + a.D <= b.Z || b.Z + b.D <= a.Z);
    }

    static List<(int w, int h, int d)> GetOrientations(int w, int h, int d)
    {
        var orientations = new (int, int, int)[]
        {
            (w, h, d), (w, d, h), (h, w, d),
            (h, d, w), (d, w, h), (d, h, w)
        };
        return orientations.Distinct().ToList();
    }

    class Box
    {
        public int Index;
        public int W, H, D;

        public Box(int index, int w, int h, int d)
        {
            Index = index;
            W = w;
            H = h;
            D = d;
        }
    }

    class Box3D
    {
        public int X, Y, Z;
        public int W, H, D;

        public Box3D(int x, int y, int z, int w, int h, int d)
        {
            X = x;
            Y = y;
            Z = z;
            W = w;
            H = h;
            D = d;
        }
    }

    class Point3D
    {
        public int X, Y, Z;

        public Point3D(int x, int y, int z)
        {
            X = x;
            Y = y;
            Z = z;
        }
    }

    class Placement
    {
        public int BoxIndex;
        public int X, Y, Z;

        public Placement(int boxIndex, int x, int y, int z)
        {
            BoxIndex = boxIndex;
            X = x;
            Y = y;
            Z = z;
        }
    }
}
