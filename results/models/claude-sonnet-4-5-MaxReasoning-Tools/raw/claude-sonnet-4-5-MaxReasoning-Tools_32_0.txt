#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <set>
#include <map>
#include <functional>

using namespace std;

const long long INF = numeric_limits<long long>::max() / 2;

struct Edge {
    int to;
    long long weight;
};

// Dijkstra's shortest path algorithm
vector<long long> dijkstra(int src, const vector<vector<Edge>>& graph, vector<int>& parent) {
    int n = graph.size();
    vector<long long> dist(n, INF);
    parent.assign(n, -1);
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;

    dist[src] = 0;
    pq.push({0, src});

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();

        if (d > dist[u]) continue;

        for (const auto& e : graph[u]) {
            if (dist[u] + e.weight < dist[e.to]) {
                dist[e.to] = dist[u] + e.weight;
                parent[e.to] = u;
                pq.push({dist[e.to], e.to});
            }
        }
    }

    return dist;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    cin >> n >> m >> k;

    vector<vector<Edge>> graph(n);

    for (int i = 0; i < m; i++) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        graph[u].push_back({v, w});
        graph[v].push_back({u, w});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; i++) {
        cin >> terminals[i];
    }

    // Special case: single terminal
    if (k == 1) {
        cout << "0 0\n";
        return 0;
    }

    // Step 1: Compute shortest paths between all pairs of terminals
    vector<vector<long long>> term_dist(k, vector<long long>(k));
    vector<vector<vector<int>>> paths(k, vector<vector<int>>(k));

    for (int i = 0; i < k; i++) {
        vector<int> parent;
        auto dist = dijkstra(terminals[i], graph, parent);

        for (int j = 0; j < k; j++) {
            term_dist[i][j] = dist[terminals[j]];

            if (i != j) {
                int curr = terminals[j];
                vector<int> path;
                while (curr != terminals[i]) {
                    path.push_back(curr);
                    curr = parent[curr];
                }
                path.push_back(terminals[i]);
                reverse(path.begin(), path.end());
                paths[i][j] = path;
            }
        }
    }

    // Step 2: Build MST on complete graph of terminals using Prim's algorithm
    vector<bool> in_mst(k, false);
    vector<long long> min_cost(k, INF);
    vector<int> mst_parent(k, -1);

    min_cost[0] = 0;

    for (int iter = 0; iter < k; iter++) {
        int u = -1;
        for (int i = 0; i < k; i++) {
            if (!in_mst[i] && (u == -1 || min_cost[i] < min_cost[u])) {
                u = i;
            }
        }

        in_mst[u] = true;

        for (int v = 0; v < k; v++) {
            if (!in_mst[v] && term_dist[u][v] < min_cost[v]) {
                min_cost[v] = term_dist[u][v];
                mst_parent[v] = u;
            }
        }
    }

    // Step 3: Collect all nodes in shortest paths between MST terminals
    set<int> nodes_in_tree;

    for (int i = 1; i < k; i++) {
        const auto& path = paths[mst_parent[i]][i];
        for (int v : path) {
            nodes_in_tree.insert(v);
        }
    }

    // Step 4: Build subgraph on collected nodes
    map<int, int> node_to_idx;
    vector<int> idx_to_node;
    for (int v : nodes_in_tree) {
        node_to_idx[v] = idx_to_node.size();
        idx_to_node.push_back(v);
    }

    int sub_n = idx_to_node.size();
    vector<tuple<long long, int, int>> sub_edges;

    for (int u_orig : nodes_in_tree) {
        int u = node_to_idx[u_orig];
        for (const auto& e : graph[u_orig]) {
            if (nodes_in_tree.count(e.to)) {
                int v = node_to_idx[e.to];
                if (u < v) {
                    sub_edges.push_back({e.weight, u, v});
                }
            }
        }
    }

    sort(sub_edges.begin(), sub_edges.end());

    // Step 5: Build MST on subgraph using Kruskal's algorithm
    vector<int> uf_parent(sub_n);
    for (int i = 0; i < sub_n; i++) uf_parent[i] = i;

    function<int(int)> find = [&](int x) {
        return uf_parent[x] == x ? x : uf_parent[x] = find(uf_parent[x]);
    };

    vector<pair<int, int>> result_edges;
    long long total_weight = 0;

    for (const auto& [w, u, v] : sub_edges) {
        int pu = find(u), pv = find(v);
        if (pu != pv) {
            uf_parent[pu] = pv;
            result_edges.push_back({idx_to_node[u], idx_to_node[v]});
            total_weight += w;
        }
    }

    // Output result
    cout << total_weight << " " << result_edges.size() << "\n";
    for (const auto& [u, v] : result_edges) {
        cout << u << " " << v << "\n";
    }

    return 0;
}
