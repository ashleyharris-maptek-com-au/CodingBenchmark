using System;
using System.Linq;

class QAP
{
    static void Main()
    {
        int n = int.Parse(Console.ReadLine());

        int[][] flow = new int[n][];
        int[][] dist = new int[n][];

        for (int i = 0; i < n; i++)
        {
            flow[i] = Console.ReadLine().Split().Select(int.Parse).ToArray();
        }

        for (int i = 0; i < n; i++)
        {
            dist[i] = Console.ReadLine().Split().Select(int.Parse).ToArray();
        }

        // Solve QAP
        int[] solution = SolveQAP(n, flow, dist);
        long cost = CalculateCost(solution, flow, dist);

        Console.WriteLine(cost);
        Console.WriteLine(string.Join(" ", solution));
    }

    static int[] SolveQAP(int n, int[][] flow, int[][] dist)
    {
        Random rand = new Random(42);

        // Initial greedy solution
        int[] best = GreedyConstruction(n, flow, dist);
        long bestCost = CalculateCost(best, flow, dist);

        // Simulated annealing
        int[] current = (int[])best.Clone();
        long currentCost = bestCost;

        double temp = 1000.0;
        double coolingRate = 0.9995;
        int iterations = Math.Min(1000000, n * n * 100);

        for (int iter = 0; iter < iterations; iter++)
        {
            // Random swap
            int i = rand.Next(n);
            int j = rand.Next(n);
            if (i == j) continue;

            // Calculate delta cost
            long delta = CalculateDelta(current, flow, dist, i, j);

            if (delta < 0 || rand.NextDouble() < Math.Exp(-delta / temp))
            {
                // Accept move
                Swap(current, i, j);
                currentCost += delta;

                if (currentCost < bestCost)
                {
                    best = (int[])current.Clone();
                    bestCost = currentCost;
                }
            }

            temp *= coolingRate;
        }

        // Local search refinement
        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 0; i < n && !improved; i++)
            {
                for (int j = i + 1; j < n; j++)
                {
                    long delta = CalculateDelta(best, flow, dist, i, j);
                    if (delta < 0)
                    {
                        Swap(best, i, j);
                        bestCost += delta;
                        improved = true;
                        break;
                    }
                }
            }
        }

        return best;
    }

    static int[] GreedyConstruction(int n, int[][] flow, int[][] dist)
    {
        int[] perm = new int[n];
        bool[] usedLoc = new bool[n];
        bool[] assignedFac = new bool[n];

        // Calculate facility and location weights
        double[] facWeight = new double[n];
        double[] locWeight = new double[n];

        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                facWeight[i] += flow[i][j] + flow[j][i];
                locWeight[i] += dist[i][j] + dist[j][i];
            }
        }

        // Sort facilities by weight (descending)
        var facOrder = Enumerable.Range(0, n).OrderByDescending(i => facWeight[i]).ToArray();
        var locOrder = Enumerable.Range(0, n).OrderBy(i => locWeight[i]).ToArray();

        // Assign high-flow facilities to low-distance locations
        for (int k = 0; k < n; k++)
        {
            perm[facOrder[k]] = locOrder[k];
        }

        return perm;
    }

    static long CalculateCost(int[] perm, int[][] flow, int[][] dist)
    {
        int n = perm.Length;
        long cost = 0;

        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                cost += (long)flow[i][j] * dist[perm[i]][perm[j]];
            }
        }

        return cost;
    }

    static long CalculateDelta(int[] perm, int[][] flow, int[][] dist, int i, int j)
    {
        int n = perm.Length;
        if (i == j) return 0;

        long delta = 0;
        int pi = perm[i];
        int pj = perm[j];

        for (int k = 0; k < n; k++)
        {
            if (k == i || k == j) continue;
            int pk = perm[k];

            // Remove old costs
            delta -= (long)flow[i][k] * dist[pi][pk];
            delta -= (long)flow[k][i] * dist[pk][pi];
            delta -= (long)flow[j][k] * dist[pj][pk];
            delta -= (long)flow[k][j] * dist[pk][pj];

            // Add new costs
            delta += (long)flow[i][k] * dist[pj][pk];
            delta += (long)flow[k][i] * dist[pk][pj];
            delta += (long)flow[j][k] * dist[pi][pk];
            delta += (long)flow[k][j] * dist[pk][pi];
        }

        // Handle i-j interaction
        delta -= (long)flow[i][j] * dist[pi][pj];
        delta -= (long)flow[j][i] * dist[pj][pi];
        delta += (long)flow[i][j] * dist[pj][pi];
        delta += (long)flow[j][i] * dist[pi][pj];

        return delta;
    }

    static void Swap(int[] arr, int i, int j)
    {
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
}
