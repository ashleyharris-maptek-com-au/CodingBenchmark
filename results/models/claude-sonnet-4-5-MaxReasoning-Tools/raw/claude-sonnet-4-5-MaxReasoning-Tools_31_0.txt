use std::collections::HashMap;
use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();

    let mut lines = input.lines();
    let first_line = lines.next().unwrap();
    let mut parts = first_line.split_whitespace();
    let _n: usize = parts.next().unwrap().parse().unwrap();
    let target: i64 = parts.next().unwrap().parse().unwrap();

    let second_line = lines.next().unwrap();
    let numbers: Vec<i64> = second_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    if let Some(result) = solve_subset_sum(&numbers, target) {
        println!("YES");
        let indices: Vec<String> = result.iter().map(|&i| i.to_string()).collect();
        println!("[{}]", indices.join(" "));
    } else {
        println!("NO");
    }
}

fn solve_subset_sum(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = numbers.len();

    // Edge case: target is 0 (empty subset)
    if target == 0 {
        return Some(vec![]);
    }

    // Check if target is directly in the set
    for (i, &num) in numbers.iter().enumerate() {
        if num == target {
            return Some(vec![i]);
        }
    }

    // Choose algorithm based on problem size
    if n <= 30 {
        meet_in_the_middle(numbers, target)
    } else if n <= 5000 {
        dp_with_tracking(numbers, target)
    } else {
        // For very large sets, use heuristic approaches
        heuristic_search(numbers, target)
    }
}

fn meet_in_the_middle(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = numbers.len();
    let mid = n / 2;

    // Generate all subsets of left half
    let mut left_map: HashMap<i64, Vec<usize>> = HashMap::new();
    for mask in 0..(1_u64 << mid) {
        let mut sum = 0i64;
        let mut indices = Vec::new();
        for i in 0..mid {
            if mask & (1 << i) != 0 {
                sum = sum.wrapping_add(numbers[i]);
                indices.push(i);
            }
        }
        if !left_map.contains_key(&sum) {
            left_map.insert(sum, indices);
        }
    }

    // Generate all subsets of right half and check
    for mask in 0..(1_u64 << (n - mid)) {
        let mut sum = 0i64;
        let mut right_indices = Vec::new();
        for i in 0..(n - mid) {
            if mask & (1 << i) != 0 {
                sum = sum.wrapping_add(numbers[mid + i]);
                right_indices.push(mid + i);
            }
        }

        let need = target.wrapping_sub(sum);
        if let Some(left_indices) = left_map.get(&need) {
            let mut result = left_indices.clone();
            result.extend(right_indices);
            return Some(result);
        }
    }

    None
}

fn dp_with_tracking(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let mut dp: HashMap<i64, Vec<usize>> = HashMap::new();
    dp.insert(0, vec![]);

    for (idx, &num) in numbers.iter().enumerate() {
        let keys: Vec<i64> = dp.keys().copied().collect();

        for sum in keys {
            let new_sum = sum.wrapping_add(num);

            if new_sum == target {
                let mut result = dp[&sum].clone();
                result.push(idx);
                return Some(result);
            }

            if !dp.contains_key(&new_sum) {
                let mut new_indices = dp[&sum].clone();
                new_indices.push(idx);
                dp.insert(new_sum, new_indices);
            }
        }

        // Limit memory usage for very large DP tables
        if dp.len() > 10_000_000 {
            break;
        }
    }

    dp.get(&target).cloned()
}

fn heuristic_search(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    // Try limited DP first
    if let Some(result) = limited_dp(numbers, target, 20_000_000) {
        return Some(result);
    }

    // Try greedy approaches with different orderings
    greedy_subset_sum(numbers, target)
}

fn limited_dp(numbers: &[i64], target: i64, max_states: usize) -> Option<Vec<usize>> {
    let mut dp: HashMap<i64, Vec<usize>> = HashMap::new();
    dp.insert(0, vec![]);

    for (idx, &num) in numbers.iter().enumerate() {
        if dp.len() > max_states {
            break;
        }

        let keys: Vec<i64> = dp.keys().copied().collect();

        for sum in keys {
            let new_sum = sum.wrapping_add(num);

            if new_sum == target {
                let mut result = dp[&sum].clone();
                result.push(idx);
                return Some(result);
            }

            if !dp.contains_key(&new_sum) {
                let mut new_indices = dp[&sum].clone();
                new_indices.push(idx);
                dp.insert(new_sum, new_indices);
            }

            if dp.len() > max_states {
                break;
            }
        }
    }

    dp.get(&target).cloned()
}

fn greedy_subset_sum(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    // Create indexed pairs
    let mut indexed: Vec<(usize, i64)> = numbers.iter().enumerate().map(|(i, &v)| (i, v)).collect();

    // Try sorting by absolute value descending (prefer larger values)
    indexed.sort_by_key(|(_, v)| -v.abs());
    if let Some(result) = greedy_search_ordered(&indexed, target) {
        return Some(result);
    }

    // Try sorting by value ascending
    indexed.sort_by_key(|(_, v)| *v);
    if let Some(result) = greedy_search_ordered(&indexed, target) {
        return Some(result);
    }

    // Try sorting by proximity to target
    indexed.sort_by_key(|(_, v)| (target - v).abs());
    greedy_search_ordered(&indexed, target)
}

fn greedy_search_ordered(indexed: &[(usize, i64)], target: i64) -> Option<Vec<usize>> {
    let n = indexed.len();
    let search_limit = n.min(100); // Only search first 100 elements

    let mut dp: HashMap<i64, Vec<usize>> = HashMap::new();
    dp.insert(0, vec![]);

    for i in 0..search_limit {
        let (idx, num) = indexed[i];

        let keys: Vec<i64> = dp.keys().copied().collect();

        for sum in keys {
            let new_sum = sum.wrapping_add(num);

            if new_sum == target {
                let mut result = dp[&sum].clone();
                result.push(idx);
                return Some(result);
            }

            if !dp.contains_key(&new_sum) {
                let mut new_indices = dp[&sum].clone();
                new_indices.push(idx);
                dp.insert(new_sum, new_indices);
            }
        }

        // Limit DP table size
        if dp.len() > 1_000_000 {
            break;
        }
    }

    dp.get(&target).cloned()
}
