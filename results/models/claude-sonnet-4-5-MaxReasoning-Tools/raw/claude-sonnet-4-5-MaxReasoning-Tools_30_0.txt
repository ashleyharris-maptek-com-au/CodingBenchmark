import sys
from collections import defaultdict, deque

class SATSolver:
    def __init__(self, num_vars, clauses):
        self.num_vars = num_vars
        self.clauses = clauses
        self.assignment = [None] * (num_vars + 1)  # 1-indexed
        self.var_activity = [0.0] * (num_vars + 1)
        self.activity_decay = 0.95
        self.var_inc = 1.0

        # Watched literals: for each literal, which clauses watch it
        self.watches = defaultdict(list)

        # Initialize watched literals (watch first two literals of each clause)
        for i, clause in enumerate(clauses):
            if len(clause) >= 1:
                self.watches[clause[0]].append(i)
            if len(clause) >= 2:
                self.watches[clause[1]].append(i)

        # Decision level tracking
        self.level = 0
        self.trail = []
        self.trail_lim = []

    def lit_value(self, lit):
        """Returns True/False/None for a literal's value"""
        var = abs(lit)
        val = self.assignment[var]
        if val is None:
            return None
        return val if lit > 0 else not val

    def assign(self, lit, level):
        """Assign a literal to true"""
        var = abs(lit)
        val = lit > 0
        self.assignment[var] = val
        self.trail.append(lit)

    def unassign(self, var):
        """Unassign a variable"""
        self.assignment[var] = None

    def propagate(self):
        """Unit propagation with watched literals. Returns conflict clause or None"""
        propagated = 0
        while propagated < len(self.trail):
            lit = self.trail[propagated]
            propagated += 1

            # Check all clauses watching ~lit
            neg_lit = -lit
            watches_copy = self.watches[neg_lit][:]
            self.watches[neg_lit].clear()

            for clause_idx in watches_copy:
                clause = self.clauses[clause_idx]

                # Check if clause is already satisfied
                satisfied = False
                for l in clause:
                    if self.lit_value(l) is True:
                        satisfied = True
                        break

                if satisfied:
                    self.watches[neg_lit].append(clause_idx)
                    continue

                # Find a new watched literal
                found = False
                for l in clause:
                    if l != neg_lit and self.lit_value(l) is not False:
                        # Watch this literal instead
                        self.watches[l].append(clause_idx)
                        found = True
                        break

                if not found:
                    # Check if this is a unit clause or conflict
                    unassigned = None
                    all_false = True
                    for l in clause:
                        val = self.lit_value(l)
                        if val is None:
                            unassigned = l
                            all_false = False
                        elif val is True:
                            all_false = False
                            break

                    if all_false:
                        # Conflict
                        # Restore watches
                        for remaining in watches_copy[watches_copy.index(clause_idx)+1:]:
                            self.watches[neg_lit].append(remaining)
                        self.watches[neg_lit].append(clause_idx)
                        return clause_idx
                    elif unassigned is not None:
                        # Unit clause
                        self.assign(unassigned, self.level)
                        self.watches[neg_lit].append(clause_idx)
                    else:
                        self.watches[neg_lit].append(clause_idx)
                else:
                    pass  # New watch was set

        return None

    def pick_branching_variable(self):
        """Pick unassigned variable with highest activity"""
        best_var = None
        best_activity = -1
        for var in range(1, self.num_vars + 1):
            if self.assignment[var] is None:
                if self.var_activity[var] > best_activity:
                    best_activity = self.var_activity[var]
                    best_var = var
        return best_var

    def bump_activity(self, var):
        """Increase variable activity"""
        self.var_activity[var] += self.var_inc
        if self.var_activity[var] > 1e100:
            # Rescale
            for v in range(1, self.num_vars + 1):
                self.var_activity[v] *= 1e-100
            self.var_inc *= 1e-100

    def decay_activities(self):
        """Decay all variable activities"""
        self.var_inc /= self.activity_decay

    def solve(self):
        """Main DPLL search"""
        # Initial unit propagation
        conflict = self.propagate()
        if conflict is not None:
            return False

        decisions = 0
        max_decisions = 10000000

        while decisions < max_decisions:
            # Pick branching variable
            var = self.pick_branching_variable()
            if var is None:
                # All variables assigned - SAT
                return True

            # Make decision (try True first)
            decisions += 1
            self.level += 1
            self.trail_lim.append(len(self.trail))

            # Try positive literal first
            self.assign(var, self.level)
            self.bump_activity(var)

            conflict = self.propagate()

            if conflict is not None:
                # Backtrack
                if self.level == 0:
                    return False

                # Simple chronological backtracking for efficiency
                self.backtrack(self.level - 1)

                # Try negative instead
                self.assign(-var, self.level)
                conflict = self.propagate()

                if conflict is not None:
                    # Both failed, backtrack further
                    if self.level == 0:
                        return False
                    self.backtrack(0)
                    # Restart search
                    self.decay_activities()

            self.decay_activities()

        return False

    def backtrack(self, target_level):
        """Backtrack to target decision level"""
        while self.level > target_level:
            if len(self.trail_lim) == 0:
                break
            trail_pos = self.trail_lim.pop()
            while len(self.trail) > trail_pos:
                lit = self.trail.pop()
                var = abs(lit)
                self.unassign(var)
            self.level -= 1

def main():
    lines = sys.stdin.read().strip().split('\n')
    num_vars, num_clauses = map(int, lines[0].split())

    clauses = []
    for i in range(1, num_clauses + 1):
        clause = list(map(int, lines[i].split()))
        clauses.append(clause)

    solver = SATSolver(num_vars, clauses)

    if solver.solve():
        print("SAT")
        result = []
        for var in range(1, num_vars + 1):
            val = solver.assignment[var]
            if val is None:
                val = False  # Unassigned can be anything
            result.append(1 if val else 0)
        print(' '.join(map(str, result)))
    else:
        print("UNSAT")

if __name__ == "__main__":
    main()
