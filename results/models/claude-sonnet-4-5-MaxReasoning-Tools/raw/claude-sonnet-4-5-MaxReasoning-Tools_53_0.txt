import math

# Module-level state persistence across calls
_state_history = []
_initialized = False
_integral_error = {'x': 0.0, 'y': 0.0, 'z': 0.0}
_last_error = {'x': 0.0, 'y': 0.0, 'z': 0.0}

def autopilot_step(state: dict, dt: float) -> dict:
    """
    Spacecraft autopilot for orbital docking.
    Handles sensor fusion, orbital mechanics, and multi-phase approach.
    """
    global _state_history, _initialized, _integral_error, _last_error

    # Reset state on first call or if position jumps (new scenario)
    pos = estimate_position(state)
    if not _initialized or (_state_history and
        math.sqrt(sum((pos[i] - _state_history[-1]['pos'][i])**2 for i in range(3))) > 1000):
        _state_history = []
        _integral_error = {'x': 0.0, 'y': 0.0, 'z': 0.0}
        _last_error = {'x': 0.0, 'y': 0.0, 'z': 0.0}
        _initialized = True

    # Sensor fusion - get best estimates
    position = estimate_position(state)  # [x, y, z] in LVLH frame
    velocity = estimate_velocity(state)  # [vx, vy, vz]
    attitude = estimate_attitude(state)  # [roll, pitch, yaw]

    # Calculate range and approach speed
    range_m = math.sqrt(sum(p**2 for p in position))
    approach_speed = -sum(p*v for p, v in zip(position, velocity)) / (range_m + 1e-6)

    # Store history
    _state_history.append({
        'pos': position,
        'vel': velocity,
        'range': range_m,
        'approach_speed': approach_speed,
        'fuel': state['fuel_remaining_kg']
    })
    if len(_state_history) > 100:
        _state_history.pop(0)

    # Determine control phase based on range
    if range_m > 200:
        thrust = far_range_control(position, velocity, range_m)
    elif range_m > 50:
        thrust = medium_range_control(position, velocity, range_m)
    elif range_m > 10:
        thrust = close_range_control(position, velocity, range_m, dt)
    else:
        thrust = final_approach_control(position, velocity, range_m, dt)

    # Attitude control - point thrust vector correctly
    att_rates = attitude_control(attitude, thrust, state)

    # Fuel conservation - reduce thrust if low on fuel
    if state['fuel_remaining_kg'] < 100:
        scale = max(0.3, state['fuel_remaining_kg'] / 100)
        thrust = [t * scale for t in thrust]

    return {
        'thrust_x': clamp(thrust[0], -1.0, 1.0),
        'thrust_y': clamp(thrust[1], -1.0, 1.0),
        'thrust_z': clamp(thrust[2], -1.0, 1.0),
        'att_rate_roll': clamp(att_rates[0], -0.035, 0.035),
        'att_rate_pitch': clamp(att_rates[1], -0.035, 0.035),
        'att_rate_yaw': clamp(att_rates[2], -0.035, 0.035)
    }


def estimate_position(state: dict) -> list:
    """Fuse multiple sensors to estimate relative position."""
    positions = []
    weights = []

    # Radar position
    if (state.get('radar_rel_x_valid') and state.get('radar_rel_y_valid') and
        state.get('radar_rel_z_valid')):
        positions.append([
            state['radar_rel_pos_x_m'],
            state['radar_rel_pos_y_m'],
            state['radar_rel_pos_z_m']
        ])
        weights.append(2.0)

    # Lidar position
    if (state.get('lidar_rel_x_valid') and state.get('lidar_rel_y_valid') and
        state.get('lidar_rel_z_valid')):
        positions.append([
            state['lidar_rel_pos_x_m'],
            state['lidar_rel_pos_y_m'],
            state['lidar_rel_pos_z_m']
        ])
        weights.append(2.0)

    # Docking camera bearing + range
    if state.get('dock_camera_active') and state.get('dock_camera_range_m'):
        if (state.get('dock_camera_bearing_x_valid') and
            state.get('dock_camera_bearing_y_valid') and
            state.get('dock_camera_bearing_z_valid')):
            r = state['dock_camera_range_m']
            bx = state['dock_camera_bearing_x']
            by = state['dock_camera_bearing_y']
            bz = state['dock_camera_bearing_z']
            norm = math.sqrt(bx**2 + by**2 + bz**2) + 1e-6
            positions.append([r * bx / norm, r * by / norm, r * bz / norm])
            weights.append(1.5)

    # Use history as fallback with velocity integration
    if not positions and _state_history:
        last = _state_history[-1]
        # Dead reckoning from last known position
        return last['pos']

    if not positions:
        # Emergency fallback - assume we're behind
        return [0.0, -200.0, 0.0]

    # Weighted average
    total_weight = sum(weights)
    pos = [sum(p[i] * w for p, w in zip(positions, weights)) / total_weight
           for i in range(3)]

    return pos


def estimate_velocity(state: dict) -> list:
    """Fuse multiple sensors to estimate relative velocity."""
    velocities = []
    weights = []

    # Radar velocity
    if (state.get('radar_relv_x_valid') and state.get('radar_relv_y_valid') and
        state.get('radar_relv_z_valid')):
        velocities.append([
            state['radar_rel_vel_x_ms'],
            state['radar_rel_vel_y_ms'],
            state['radar_rel_vel_z_ms']
        ])
        weights.append(2.0)

    # Lidar velocity
    if (state.get('lidar_relv_x_valid') and state.get('lidar_relv_y_valid') and
        state.get('lidar_relv_z_valid')):
        velocities.append([
            state['lidar_rel_vel_x_ms'],
            state['lidar_rel_vel_y_ms'],
            state['lidar_rel_vel_z_ms']
        ])
        weights.append(2.0)

    # Numerical differentiation from position history
    if len(_state_history) >= 2:
        p1 = _state_history[-1]['pos']
        p2 = _state_history[-2]['pos']
        dt_hist = 0.1  # Assume ~0.1s between steps
        vel_diff = [(p1[i] - p2[i]) / dt_hist for i in range(3)]
        velocities.append(vel_diff)
        weights.append(0.5)

    if not velocities:
        return [0.0, 0.0, 0.0]

    # Weighted average
    total_weight = sum(weights)
    vel = [sum(v[i] * w for v, w in zip(velocities, weights)) / total_weight
           for i in range(3)]

    return vel


def estimate_attitude(state: dict) -> list:
    """Estimate spacecraft attitude from star trackers and gyros."""
    # Prefer star trackers (absolute) over gyros (relative)
    attitudes = []
    weights = []

    # Star tracker 1
    if (state.get('star_tracker_1_roll_valid') and
        state.get('star_tracker_1_pitch_valid') and
        state.get('star_tracker_1_yaw_valid')):
        attitudes.append([
            math.radians(state['star_tracker_1_roll_deg']),
            math.radians(state['star_tracker_1_pitch_deg']),
            math.radians(state['star_tracker_1_yaw_deg'])
        ])
        weights.append(2.0)

    # Star tracker 2
    if (state.get('star_tracker_2_roll_valid') and
        state.get('star_tracker_2_pitch_valid') and
        state.get('star_tracker_2_yaw_valid')):
        attitudes.append([
            math.radians(state['star_tracker_2_roll_deg']),
            math.radians(state['star_tracker_2_pitch_deg']),
            math.radians(state['star_tracker_2_yaw_deg'])
        ])
        weights.append(2.0)

    if not attitudes:
        return [0.0, 0.0, 0.0]

    total_weight = sum(weights)
    att = [sum(a[i] * w for a, w in zip(attitudes, weights)) / total_weight
           for i in range(3)]

    return att


def far_range_control(pos: list, vel: list, range_m: float) -> list:
    """
    Far range (>200m): Orbital mechanics dominant.
    Use Hohmann-like transfer approach.
    """
    x, y, z = pos
    vx, vy, vz = vel

    thrust = [0.0, 0.0, 0.0]

    # Along-track (y-axis): If behind (y < 0), thrust retrograde to catch up
    # If ahead (y > 0), thrust prograde to slow down
    if abs(y) > 10:
        target_vy = 0.3 * (-y / abs(y)) * min(1.0, abs(y) / 200)  # Target closing velocity
        vy_error = target_vy - vy
        thrust[1] = -0.3 * vy_error  # Retrograde to speed up if behind

    # Radial (x-axis): Small corrections only
    if abs(x) > 20:
        thrust[0] = -0.1 * x / range_m - 0.2 * vx

    # Cross-track (z-axis): Direct correction
    if abs(z) > 10:
        thrust[2] = -0.15 * z / range_m - 0.25 * vz

    return thrust


def medium_range_control(pos: list, vel: list, range_m: float) -> list:
    """
    Medium range (50-200m): Transition regime.
    V-bar approach with gentle corrections.
    """
    x, y, z = pos
    vx, vy, vz = vel

    thrust = [0.0, 0.0, 0.0]

    # Desired approach velocity scales with range
    desired_approach = min(0.5, range_m / 200.0)

    # Along-track: Close the gap
    if abs(y) > 5:
        target_vy = desired_approach * (-y / abs(y))
        thrust[1] = -0.4 * (vy - target_vy)

    # Radial: Keep small, correct slowly
    if abs(x) > 10:
        thrust[0] = -0.3 * x / range_m - 0.4 * vx
    else:
        thrust[0] = -0.5 * vx  # Damp velocity

    # Cross-track: Direct correction
    if abs(z) > 5:
        thrust[2] = -0.3 * z / range_m - 0.5 * vz
    else:
        thrust[2] = -0.6 * vz

    return thrust


def close_range_control(pos: list, vel: list, range_m: float, dt: float) -> list:
    """
    Close range (10-50m): Linear regime with PD control.
    """
    global _integral_error, _last_error

    x, y, z = pos
    vx, vy, vz = vel

    # PD gains
    kp = 0.02
    kd = 0.5
    ki = 0.001

    thrust = [0.0, 0.0, 0.0]

    for i, (p, v) in enumerate([(x, vx), (y, vy), (z, vz)]):
        axis = ['x', 'y', 'z'][i]

        # Error and derivative
        error = -p
        deriv = -v

        # Integral with anti-windup
        _integral_error[axis] += error * dt
        _integral_error[axis] = clamp(_integral_error[axis], -100, 100)

        # PID control
        thrust[i] = kp * error + kd * deriv + ki * _integral_error[axis]

        _last_error[axis] = error

    return thrust


def final_approach_control(pos: list, vel: list, range_m: float, dt: float) -> list:
    """
    Final approach (<10m): Very gentle, precision docking.
    """
    global _integral_error

    x, y, z = pos
    vx, vy, vz = vel

    # Very conservative gains for final approach
    kp = 0.03
    kd = 0.8
    ki = 0.0005

    thrust = [0.0, 0.0, 0.0]

    # Target approach speed: very slow
    max_approach_speed = 0.15 if range_m > 3 else 0.08

    for i, (p, v) in enumerate([(x, vx), (y, vy), (z, vz)]):
        axis = ['x', 'y', 'z'][i]

        error = -p
        deriv = -v

        # Limit approach speed
        if abs(error) > 0.1:
            desired_v = max_approach_speed * (error / abs(error))
            deriv = v - desired_v

        _integral_error[axis] += error * dt
        _integral_error[axis] = clamp(_integral_error[axis], -50, 50)

        thrust[i] = kp * error + kd * deriv + ki * _integral_error[axis]

    # Very gentle thrust for final meters
    if range_m < 2.5:
        thrust = [t * 0.5 for t in thrust]

    return thrust


def attitude_control(current_att: list, thrust_vector: list, state: dict) -> list:
    """
    Control spacecraft attitude to point thrust vector correctly.
    For simplicity, keep attitude near zero and use damping.
    """
    roll, pitch, yaw = current_att

    # Get gyro rates for damping
    gyro_rates = estimate_gyro_rates(state)

    # Simple damping control - keep spacecraft stable
    kp_att = 0.5
    kd_att = 2.0

    att_rates = [
        -kp_att * roll - kd_att * gyro_rates[0],
        -kp_att * pitch - kd_att * gyro_rates[1],
        -kp_att * yaw - kd_att * gyro_rates[2]
    ]

    return att_rates


def estimate_gyro_rates(state: dict) -> list:
    """Estimate angular rates from gyros with redundancy."""
    rates = [[], [], []]  # [roll_rates, pitch_rates, yaw_rates]

    # Gyro 1
    if (state.get('gyro_1_roll_valid') and state.get('gyro_1_pitch_valid') and
        state.get('gyro_1_yaw_valid')):
        rates[0].append(math.radians(state['gyro_1_roll_dps']))
        rates[1].append(math.radians(state['gyro_1_pitch_dps']))
        rates[2].append(math.radians(state['gyro_1_yaw_dps']))

    # Gyro 2
    if (state.get('gyro_2_roll_valid') and state.get('gyro_2_pitch_valid') and
        state.get('gyro_2_yaw_valid')):
        rates[0].append(math.radians(state['gyro_2_roll_dps']))
        rates[1].append(math.radians(state['gyro_2_pitch_dps']))
        rates[2].append(math.radians(state['gyro_2_yaw_dps']))

    # Gyro 3
    if (state.get('gyro_3_roll_valid') and state.get('gyro_3_pitch_valid') and
        state.get('gyro_3_yaw_valid')):
        rates[0].append(math.radians(state['gyro_3_roll_dps']))
        rates[1].append(math.radians(state['gyro_3_pitch_dps']))
        rates[2].append(math.radians(state['gyro_3_yaw_dps']))

    # Average available rates, median filter for outliers
    result = []
    for axis_rates in rates:
        if len(axis_rates) == 0:
            result.append(0.0)
        elif len(axis_rates) == 1:
            result.append(axis_rates[0])
        else:
            # Use median to reject outliers
            axis_rates.sort()
            result.append(axis_rates[len(axis_rates) // 2])

    return result


def clamp(value: float, min_val: float, max_val: float) -> float:
    """Clamp value between min and max."""
    return max(min_val, min(max_val, value))
