import sys

def read_input():
    lines = sys.stdin.read().strip().split('\n')
    U, S = map(int, lines[0].split())
    sets = []
    for i in range(1, S + 1):
        if i < len(lines) and lines[i].strip():
            elements = list(map(int, lines[i].split()))
            sets.append(set(elements))
        else:
            sets.append(set())
    return U, S, sets

def solve_exact_cover(U, sets):
    n = len(sets)

    def backtrack(covered, chosen):
        # Check if all elements are covered
        if len(covered) == U:
            return sorted(chosen)

        # Find the smallest uncovered element for determinism
        uncovered = None
        for i in range(U):
            if i not in covered:
                uncovered = i
                break

        # Try each set that contains this uncovered element
        for i in range(n):
            if i not in chosen and uncovered in sets[i]:
                # Check if this set has no conflict with already covered elements
                if sets[i].isdisjoint(covered):
                    # Choose this set and recurse
                    result = backtrack(covered | sets[i], chosen | {i})
                    if result is not None:
                        return result

        return None

    return backtrack(set(), set())

def main():
    U, S, sets = read_input()
    solution = solve_exact_cover(U, sets)

    if solution is not None:
        print("SOLUTION")
        print(' '.join(map(str, solution)))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    main()
