```python
import math

# Module-level persistent state for PID controllers
_pitch_integral = 0.0
_roll_integral = 0.0
_yaw_integral = 0.0
_speed_integral = 0.0
_alt_integral = 0.0
_prev_pitch_error = 0.0
_prev_roll_error = 0.0
_prev_yaw_error = 0.0
_prev_speed_error = 0.0
_prev_alt_error = 0.0
_trim_integral = 0.0

def autopilot_step(state: dict, dt: float) -> dict:
    """
    Autopilot for twin-engine airliner with full sensor fusion and envelope protection.
    """
    global _pitch_integral, _roll_integral, _yaw_integral, _speed_integral, _alt_integral
    global _prev_pitch_error, _prev_roll_error, _prev_yaw_error, _prev_speed_error, _prev_alt_error
    global _trim_integral

    # ============================================================================
    # SENSOR FUSION - Vote among redundant sensors, detect failures
    # ============================================================================

    def vote_sensors(v1, valid1, v2, valid2, v3, valid3, tolerance=10.0):
        """Vote among 3 sensors, returning best estimate and confidence."""
        vals = []
        if valid1: vals.append(v1)
        if valid2: vals.append(v2)
        if valid3: vals.append(v3)

        if len(vals) == 0:
            return None, 0.0
        if len(vals) == 1:
            return vals[0], 0.5
        if len(vals) == 2:
            if abs(vals[0] - vals[1]) < tolerance:
                return (vals[0] + vals[1]) / 2, 0.8
            return vals[0], 0.3  # Disagreement

        # Three sensors - find majority
        vals.sort()
        if abs(vals[0] - vals[1]) < tolerance:
            return (vals[0] + vals[1]) / 2, 0.95
        if abs(vals[1] - vals[2]) < tolerance:
            return (vals[1] + vals[2]) / 2, 0.95
        return vals[1], 0.7  # Middle value

    # Altitude fusion - cross-check baro, radar, GPS
    baro_alt, baro_conf = vote_sensors(
        state.get('baro_alt_1_ft', 0), state.get('baro_alt_1_valid', False),
        state.get('baro_alt_2_ft', 0), state.get('baro_alt_2_valid', False),
        state.get('baro_alt_3_ft', 0), state.get('baro_alt_3_valid', False),
        tolerance=50
    )

    gps_alt, gps_conf = vote_sensors(
        state.get('gps_1_alt_ft', 0), state.get('gps_1_valid', False),
        state.get('gps_2_alt_ft', 0), state.get('gps_2_valid', False),
        state.get('gps_3_alt_ft', 0), state.get('gps_3_valid', False),
        tolerance=100
    )

    radar_alt = None
    if state.get('radar_alt_1_valid') and state.get('radar_alt_1_ft', 0) < 2500:
        radar_alt = state.get('radar_alt_1_ft')
    elif state.get('radar_alt_2_valid') and state.get('radar_alt_2_ft', 0) < 2500:
        radar_alt = state.get('radar_alt_2_ft')
    elif state.get('radar_alt_3_valid') and state.get('radar_alt_3_ft', 0) < 2500:
        radar_alt = state.get('radar_alt_3_ft')

    # Choose best altitude source
    if radar_alt is not None and radar_alt < 2500:
        altitude = radar_alt
        on_approach = True
    elif baro_alt is not None and baro_conf > 0.5:
        altitude = baro_alt
        on_approach = False
    elif gps_alt is not None:
        altitude = gps_alt
        on_approach = False
    else:
        altitude = state.get('baro_alt_1_ft', 0)
        on_approach = False

    # Airspeed fusion
    ias, ias_conf = vote_sensors(
        state.get('ias_1_kt', 0), state.get('ias_1_valid', False),
        state.get('ias_2_kt', 0), state.get('ias_2_valid', False),
        state.get('ias_3_kt', 0), state.get('ias_3_valid', False),
        tolerance=5
    )
    if ias is None or ias < 50:
        ias = max(state.get('ias_1_kt', 150), state.get('ias_2_kt', 150), state.get('ias_3_kt', 150))

    # Mach number
    mach = (state.get('mach_1', 0) + state.get('mach_2', 0) + state.get('mach_3', 0)) / 3

    # Attitude fusion (IRS)
    pitch, pitch_conf = vote_sensors(
        state.get('irs_1_pitch_deg', 0), state.get('irs_1_valid', False),
        state.get('irs_2_pitch_deg', 0), state.get('irs_2_valid', False),
        state.get('irs_3_pitch_deg', 0), state.get('irs_3_valid', False),
        tolerance=2
    )
    if pitch is None:
        pitch = state.get('irs_1_pitch_deg', 0)

    roll, roll_conf = vote_sensors(
        state.get('irs_1_roll_deg', 0), state.get('irs_1_valid', False),
        state.get('irs_2_roll_deg', 0), state.get('irs_2_valid', False),
        state.get('irs_3_roll_deg', 0), state.get('irs_3_valid', False),
        tolerance=2
    )
    if roll is None:
        roll = state.get('irs_1_roll_deg', 0)

    heading, hdg_conf = vote_sensors(
        state.get('irs_1_heading_deg', 0), state.get('irs_1_valid', False),
        state.get('irs_2_heading_deg', 0), state.get('irs_2_valid', False),
        state.get('irs_3_heading_deg', 0), state.get('irs_3_valid', False),
        tolerance=3
    )
    if heading is None:
        heading = state.get('irs_1_heading_deg', 0)

    # Vertical speed fusion
    vs, vs_conf = vote_sensors(
        state.get('irs_1_vs_fpm', 0), state.get('irs_1_valid', False),
        state.get('irs_2_vs_fpm', 0), state.get('irs_2_valid', False),
        state.get('irs_3_vs_fpm', 0), state.get('irs_3_valid', False),
        tolerance=200
    )
    if vs is None:
        vs = state.get('vertical_speed_fpm', 0)

    # Angular rates
    pitch_rate = state.get('gyro_1_pitch_rate_dps', 0)
    roll_rate = state.get('gyro_1_roll_rate_dps', 0)
    yaw_rate = state.get('gyro_1_yaw_rate_dps', 0)
    if not state.get('gyro_1_valid', False):
        pitch_rate = state.get('gyro_2_pitch_rate_dps', 0)
        roll_rate = state.get('gyro_2_roll_rate_dps', 0)
        yaw_rate = state.get('gyro_2_yaw_rate_dps', 0)

    # Angle of attack
    aoa, aoa_conf = vote_sensors(
        state.get('aoa_1_deg', 0), state.get('aoa_1_valid', False),
        state.get('aoa_2_deg', 0), state.get('aoa_2_valid', False),
        state.get('aoa_3_deg', 0), state.get('aoa_3_valid', False),
        tolerance=1.5
    )
    if aoa is None:
        aoa = 5.0  # Nominal cruise

    # ============================================================================
    # TARGETS
    # ============================================================================
    target_alt = state.get('target_alt_ft', altitude)
    target_hdg = state.get('target_hdg_deg', heading)
    target_ias = state.get('target_ias_kt', 250)
    target_vs = state.get('target_vs_fpm', 0)

    # ============================================================================
    # FLIGHT PHASE DETECTION
    # ============================================================================
    on_ground = state.get('weight_on_wheels_left', False) or state.get('weight_on_wheels_nose', False)
    gear_down = state.get('gear_handle_down', True)

    # ============================================================================
    # ENVELOPE PROTECTION
    # ============================================================================

    # Speed limits
    Vne_ias = 365
    Vne_mach = 0.82
    stall_speed = 120  # With flaps
    stall_speed_clean = 140

    # Overspeed protection
    if ias > Vne_ias * 0.95 or mach > Vne_mach * 0.95:
        target_ias = min(target_ias, Vne_ias * 0.90)

    # Stall protection
    flap_pos = (state.get('flap_pos_left_deg', 0) + state.get('flap_pos_right_deg', 0)) / 2
    min_speed = stall_speed if flap_pos > 5 else stall_speed_clean

    if ias < min_speed * 1.1 or aoa > 12:
        # Stall recovery: push nose down, increase thrust
        target_pitch = -2.0
        target_throttle = 0.95
        stall_mode = True
    else:
        stall_mode = False

    # ============================================================================
    # PITCH CONTROL (Altitude + Vertical Speed)
    # ============================================================================

    if not stall_mode:
        # Altitude error
        alt_error = target_alt - altitude

        # Target VS based on altitude error
        desired_vs = max(min(alt_error * 0.5, 2000), -2000)
        if target_vs != 0:
            desired_vs = target_vs

        # VS error
        vs_error = desired_vs - vs

        # Target pitch based on VS
        target_pitch = vs_error * 0.01  # VS to pitch mapping
        target_pitch = max(min(target_pitch, 15), -10)  # Pitch limits

        # Add altitude integral term for precision
        _alt_integral += alt_error * dt
        _alt_integral = max(min(_alt_integral, 5000), -5000)
        target_pitch += _alt_integral * 0.0001

    # Pitch error
    pitch_error = target_pitch - pitch
    _pitch_integral += pitch_error * dt
    _pitch_integral = max(min(_pitch_integral, 50), -50)  # Anti-windup

    pitch_derivative = (pitch_error - _prev_pitch_error) / dt if dt > 0 else 0
    _prev_pitch_error = pitch_error

    # PID gains for pitch
    Kp_pitch = 0.15
    Ki_pitch = 0.02
    Kd_pitch = 0.3

    elevator = (Kp_pitch * pitch_error +
                Ki_pitch * _pitch_integral +
                Kd_pitch * pitch_derivative -
                pitch_rate * 0.02)  # Rate damping

    elevator = max(min(elevator, 1.0), -1.0)

    # Trim adjustment for sustained pitch changes
    if abs(elevator) > 0.3:
        _trim_integral += elevator * dt * 0.05
        _trim_integral = max(min(_trim_integral, 1.0), -1.0)

    # ============================================================================
    # ROLL CONTROL (Heading)
    # ============================================================================

    # Heading error (handle wraparound)
    hdg_error = target_hdg - heading
    if hdg_error > 180:
        hdg_error -= 360
    elif hdg_error < -180:
        hdg_error += 360

    # Target bank angle from heading error
    target_roll = hdg_error * 0.5
    target_roll = max(min(target_roll, 30), -30)  # Bank limit

    # Roll error
    roll_error = target_roll - roll
    _roll_integral += roll_error * dt
    _roll_integral = max(min(_roll_integral, 30), -30)

    roll_derivative = (roll_error - _prev_roll_error) / dt if dt > 0 else 0
    _prev_roll_error = roll_error

    # PID gains for roll
    Kp_roll = 0.08
    Ki_roll = 0.01
    Kd_roll = 0.2

    aileron = (Kp_roll * roll_error +
               Ki_roll * _roll_integral +
               Kd_roll * roll_derivative -
               roll_rate * 0.03)  # Rate damping

    aileron = max(min(aileron, 1.0), -1.0)

    # ============================================================================
    # YAW DAMPING (Coordinated turns + asymmetric thrust compensation)
    # ============================================================================

    # Check for asymmetric thrust
    thrust_1 = state.get('engine_1_thrust_pct', 0)
    thrust_2 = state.get('engine_2_thrust_pct', 0)
    engine_1_running = state.get('engine_1_running', True)
    engine_2_running = state.get('engine_2_running', True)

    if not engine_1_running:
        thrust_1 = 0
    if not engine_2_running:
        thrust_2 = 0

    thrust_asymmetry = thrust_1 - thrust_2

    # Rudder compensates for yaw rate + asymmetric thrust
    target_yaw_rate = roll * 0.3  # Coordinated turn
    yaw_error = target_yaw_rate - yaw_rate

    # Compensate for thrust asymmetry (right engine out needs left rudder)
    yaw_compensation = -thrust_asymmetry * 0.3

    _yaw_integral += yaw_error * dt
    _yaw_integral = max(min(_yaw_integral, 20), -20)

    yaw_derivative = (yaw_error - _prev_yaw_error) / dt if dt > 0 else 0
    _prev_yaw_error = yaw_error

    Kp_yaw = 0.05
    Ki_yaw = 0.01
    Kd_yaw = 0.1

    rudder = (Kp_yaw * yaw_error +
              Ki_yaw * _yaw_integral +
              Kd_yaw * yaw_derivative +
              yaw_compensation)

    rudder = max(min(rudder, 1.0), -1.0)

    # ============================================================================
    # THROTTLE CONTROL (Airspeed)
    # ============================================================================

    if stall_mode:
        throttle_1 = 0.95
        throttle_2 = 0.95
    else:
        speed_error = target_ias - ias
        _speed_integral += speed_error * dt
        _speed_integral = max(min(_speed_integral, 500), -500)

        speed_derivative = (speed_error - _prev_speed_error) / dt if dt > 0 else 0
        _prev_speed_error = speed_error

        Kp_speed = 0.015
        Ki_speed = 0.002
        Kd_speed = 0.01

        throttle_cmd = (Kp_speed * speed_error +
                       Ki_speed * _speed_integral +
                       Kd_speed * speed_derivative)

        # Base throttle depends on flight phase
        if altitude < 1000 and not on_ground:
            base_throttle = 0.7  # Approach
        elif abs(vs) > 500:
            base_throttle = 0.75  # Climbing/descending
        else:
            base_throttle = 0.5  # Cruise

        throttle_1 = base_throttle + throttle_cmd
        throttle_2 = base_throttle + throttle_cmd

        # Handle engine failures
        if not engine_1_running:
            throttle_1 = 0
            throttle_2 = min(throttle_2 * 1.3, 1.0)  # Compensate
        if not engine_2_running:
            throttle_2 = 0
            throttle_1 = min(throttle_1 * 1.3, 1.0)

        throttle_1 = max(min(throttle_1, 1.0), 0.0)
        throttle_2 = max(min(throttle_2, 1.0), 0.0)

    # ============================================================================
    # FLAPS, GEAR, SPEEDBRAKES
    # ============================================================================

    # Flaps based on speed and altitude
    if ias < 180 and altitude < 3000:
        flaps = 20
    elif ias < 160 and altitude < 1500:
        flaps = 30
    elif ias < 145 and altitude < 500:
        flaps = 40
    elif ias < 220:
        flaps = 10
    else:
        flaps = 0

    # Gear based on altitude and flaps
    if altitude < 1500 and ias < 250:
        gear = 1
    else:
        gear = 0 if not gear_down else 1

    # Speedbrakes for excess energy
    speedbrake = 0.0
    if ias > target_ias + 20 and altitude > target_alt - 500:
        speedbrake = min((ias - target_ias - 20) / 50, 0.8)

    # ============================================================================
    # RETURN CONTROLS
    # ============================================================================

    return {
        'elevator': elevator,
        'aileron': aileron,
        'rudder': rudder,
        'throttle_1': throttle_1,
        'throttle_2': throttle_2,
        'elevator_trim': _trim_integral,
        'flaps': flaps,
        'speedbrake': speedbrake,
        'gear': gear
    }
```
