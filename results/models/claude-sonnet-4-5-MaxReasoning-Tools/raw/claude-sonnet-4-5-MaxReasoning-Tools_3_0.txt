use std::io::{self, BufRead};
use std::f64::consts::PI;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read N and M
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let n = parts[0];
    let m = parts[1];

    // Read edges
    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<usize> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        if parts[0] != parts[1] { // Skip self-loops
            edges.push((parts[0], parts[1]));
        }
    }

    // Choose algorithm based on graph size
    let positions = if n <= 500 {
        layout_force_directed(n, &edges, 500)
    } else if n <= 5000 {
        layout_force_directed(n, &edges, 100)
    } else if n <= 20000 {
        layout_force_directed(n, &edges, 30)
    } else {
        layout_fast_grid(n, &edges)
    };

    // Output
    println!("{}", n);
    for (x, y) in positions {
        println!("{} {}", x, y);
    }
}

fn layout_force_directed(n: usize, edges: &[(usize, usize)], iterations: usize) -> Vec<(f64, f64)> {
    let mut positions = initialize_positions(n);

    // Build adjacency list
    let mut adj = vec![Vec::new(); n];
    for &(u, v) in edges {
        adj[u].push(v);
        adj[v].push(u);
    }

    // Parameters
    let k = (1000.0 / (n as f64).sqrt()).max(10.0); // Ideal spring length
    let area = 1000.0 * 1000.0;
    let mut temp = 100.0;
    let cooling = 0.95;

    for _ in 0..iterations {
        let mut forces = vec![(0.0, 0.0); n];

        // Repulsive forces between all pairs
        for i in 0..n {
            for j in (i+1)..n {
                let dx = positions[j].0 - positions[i].0;
                let dy = positions[j].1 - positions[i].1;
                let dist = (dx * dx + dy * dy).sqrt().max(0.01);
                let repulsion = k * k / dist;
                let fx = (dx / dist) * repulsion;
                let fy = (dy / dist) * repulsion;
                forces[i].0 -= fx;
                forces[i].1 -= fy;
                forces[j].0 += fx;
                forces[j].1 += fy;
            }
        }

        // Attractive forces for edges
        for &(u, v) in edges {
            let dx = positions[v].0 - positions[u].0;
            let dy = positions[v].1 - positions[u].1;
            let dist = (dx * dx + dy * dy).sqrt().max(0.01);
            let attraction = dist * dist / k;
            let fx = (dx / dist) * attraction;
            let fy = (dy / dist) * attraction;
            forces[u].0 += fx;
            forces[u].1 += fy;
            forces[v].0 -= fx;
            forces[v].1 -= fy;
        }

        // Apply forces
        for i in 0..n {
            let force_mag = (forces[i].0 * forces[i].0 + forces[i].1 * forces[i].1).sqrt();
            if force_mag > 0.01 {
                let displacement = force_mag.min(temp);
                positions[i].0 += (forces[i].0 / force_mag) * displacement;
                positions[i].1 += (forces[i].1 / force_mag) * displacement;

                // Keep within bounds
                positions[i].0 = positions[i].0.max(0.0).min(1000.0);
                positions[i].1 = positions[i].1.max(0.0).min(1000.0);
            }
        }

        temp *= cooling;
    }

    positions
}

fn layout_fast_grid(n: usize, edges: &[(usize, usize)]) -> Vec<(f64, f64)> {
    // For very large graphs, use a simple grid-based layout
    let cols = (n as f64).sqrt().ceil() as usize;
    let spacing = 1000.0 / cols as f64;

    let mut positions = Vec::with_capacity(n);
    for i in 0..n {
        let row = i / cols;
        let col = i % cols;
        let x = col as f64 * spacing + spacing / 2.0;
        let y = row as f64 * spacing + spacing / 2.0;
        positions.push((x, y));
    }

    positions
}

fn initialize_positions(n: usize) -> Vec<(f64, f64)> {
    // Circular initialization for better distribution
    let mut positions = Vec::with_capacity(n);
    let center = 500.0;
    let radius = 400.0;

    for i in 0..n {
        let angle = 2.0 * PI * (i as f64) / (n as f64);
        let x = center + radius * angle.cos();
        let y = center + radius * angle.sin();
        positions.push((x, y));
    }

    positions
}
