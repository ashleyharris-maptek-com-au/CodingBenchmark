#include <iostream>
#include <vector>

using namespace std;

int n, m;
vector<int> c;
vector<vector<int>> adj;

long long best_value = 0;
vector<int> best_solution;

bool is_valid(const vector<int>& solution, int v) {
    for (int u : adj[v]) {
        if (solution[u] == 1) return false;
    }
    return true;
}

void backtrack(vector<int>& solution, int idx, long long current_value) {
    if (idx == n) {
        if (current_value > best_value) {
            best_value = current_value;
            best_solution = solution;
        }
        return;
    }

    // Optimistic upper bound: current value + all remaining positive weights
    long long upper = current_value;
    for (int i = idx; i < n; i++) {
        if (c[i] > 0) upper += c[i];
    }

    if (upper <= best_value) return; // Prune this branch

    // Try not including variable idx
    solution[idx] = 0;
    backtrack(solution, idx + 1, current_value);

    // Try including variable idx if it has positive coefficient and is valid
    if (c[idx] > 0 && is_valid(solution, idx)) {
        solution[idx] = 1;
        backtrack(solution, idx + 1, current_value + c[idx]);
        solution[idx] = 0; // Restore
    }
}

int main() {
    cin >> n >> m;
    c.resize(n);
    adj.resize(n);

    for (int i = 0; i < n; i++) {
        cin >> c[i];
    }

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    best_solution.resize(n, 0);
    vector<int> solution(n, 0);

    backtrack(solution, 0, 0);

    cout << best_value << endl;
    for (int i = 0; i < n; i++) {
        cout << best_solution[i];
        if (i < n - 1) cout << " ";
    }
    cout << endl;

    return 0;
}
