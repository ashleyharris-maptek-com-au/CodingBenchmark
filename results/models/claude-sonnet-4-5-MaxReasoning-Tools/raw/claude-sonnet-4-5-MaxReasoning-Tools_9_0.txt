using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        // Read input
        var line1 = Console.ReadLine().Split();
        int n = int.Parse(line1[0]);
        int stockLength = int.Parse(line1[1]);

        var line2 = Console.ReadLine().Split();
        var cuts = new (int length, int index)[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = (int.Parse(line2[i]), i);
        }

        // Sort by length descending (First Fit Decreasing algorithm)
        Array.Sort(cuts, (a, b) => b.length.CompareTo(a.length));

        // Pack cuts into stocks
        var stocks = new List<List<int>>();
        var stockRemaining = new List<int>();

        foreach (var cut in cuts)
        {
            // Find first stock that fits
            int stockIndex = -1;
            for (int i = 0; i < stockRemaining.Count; i++)
            {
                if (stockRemaining[i] >= cut.length)
                {
                    stockIndex = i;
                    break;
                }
            }

            if (stockIndex == -1)
            {
                // Need a new stock
                stocks.Add(new List<int> { cut.index });
                stockRemaining.Add(stockLength - cut.length);
            }
            else
            {
                // Add to existing stock
                stocks[stockIndex].Add(cut.index);
                stockRemaining[stockIndex] -= cut.length;
            }
        }

        // Output
        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            Console.WriteLine(string.Join(" ", stock));
        }
    }
}
