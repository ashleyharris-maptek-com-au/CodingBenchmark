#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <chrono>
#include <unordered_set>
#include <string>

using namespace std;

const double PI = 3.14159265358979323846;
const double EPSILON = 1e-9;

struct Vec3 {
    double x, y, z;

    Vec3(double x = 0, double y = 0, double z = 0) : x(x), y(y), z(z) {}

    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
    Vec3 operator/(double s) const { return Vec3(x / s, y / s, z / s); }

    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }

    Vec3 cross(const Vec3& v) const {
        return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x);
    }

    double length() const { return sqrt(x * x + y * y + z * z); }
    double lengthSq() const { return x * x + y * y + z * z; }
    Vec3 normalized() const {
        double l = length();
        return l > EPSILON ? *this / l : *this;
    }
};

struct Plane {
    Vec3 normal;
    double d;

    Plane() : normal(Vec3(0, 0, 1)), d(0) {}

    Plane(const Vec3& p1, const Vec3& p2, const Vec3& p3) {
        Vec3 v1 = p2 - p1;
        Vec3 v2 = p3 - p1;
        normal = v1.cross(v2).normalized();
        d = -normal.dot(p1);
    }

    double signedDistance(const Vec3& p) const {
        return normal.dot(p) + d;
    }
};

// Rotate a point using Euler angles (ZYX convention)
Vec3 rotatePoint(const Vec3& p, double rx, double ry, double rz) {
    // Rotate around Z
    double cx = cos(rx), sx = sin(rx);
    double cy = cos(ry), sy = sin(ry);
    double cz = cos(rz), sz = sin(rz);

    // Combined rotation matrix (ZYX)
    Vec3 result;
    result.x = p.x * (cy * cz) + p.y * (sx * sy * cz - cx * sz) + p.z * (cx * sy * cz + sx * sz);
    result.y = p.x * (cy * sz) + p.y * (sx * sy * sz + cx * cz) + p.z * (cx * sy * sz - sx * cz);
    result.z = p.x * (-sy) + p.y * (sx * cy) + p.z * (cx * cy);

    return result;
}

struct Tetrahedron {
    Vec3 center;
    double rx, ry, rz;

    Tetrahedron(const Vec3& c, double rx = 0, double ry = 0, double rz = 0)
        : center(c), rx(rx), ry(ry), rz(rz) {}

    vector<Vec3> getVertices(double edgeLength) const {
        double h = edgeLength * sqrt(2.0 / 3.0);
        double R = edgeLength * sqrt(3.0 / 8.0);

        // Standard tetrahedron vertices (centered at origin)
        vector<Vec3> verts(4);
        verts[0] = Vec3(0, 0, h - R);
        for (int i = 1; i < 4; i++) {
            double angle = 2 * PI * (i - 1) / 3.0;
            verts[i] = Vec3(R * cos(angle), R * sin(angle), -R);
        }

        // Apply rotation and translation
        for (auto& v : verts) {
            v = rotatePoint(v, rx, ry, rz);
            v = v + center;
        }

        return verts;
    }
};

// Compute convex hull faces (simplified for polyhedra)
struct ConvexHull {
    vector<Plane> planes;

    ConvexHull(const vector<Vec3>& vertices) {
        int n = vertices.size();
        if (n < 4) return;

        // For each triplet of vertices, create a potential face
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                for (int k = j + 1; k < n; k++) {
                    Plane plane(vertices[i], vertices[j], vertices[k]);

                    // Check if all other points are on one side or on the plane
                    bool valid = true;
                    int positive = 0, negative = 0;

                    for (int m = 0; m < n; m++) {
                        if (m == i || m == j || m == k) continue;
                        double dist = plane.signedDistance(vertices[m]);
                        if (dist > EPSILON) positive++;
                        if (dist < -EPSILON) negative++;
                    }

                    // This is a valid face if all points are on one side
                    if (positive > 0 && negative == 0) {
                        planes.push_back(plane);
                    } else if (negative > 0 && positive == 0) {
                        // Flip the plane
                        plane.normal = plane.normal * -1;
                        plane.d = -plane.d;
                        planes.push_back(plane);
                    }
                }
            }
        }
    }

    bool contains(const Vec3& point, double margin = 0.0) const {
        for (const auto& plane : planes) {
            if (plane.signedDistance(point) < -margin) {
                return false;
            }
        }
        return true;
    }
};

// Spatial hash for efficient collision detection
struct SpatialHash {
    double cellSize;
    unordered_set<string> occupiedCells;

    SpatialHash(double cellSize) : cellSize(cellSize) {}

    string getKey(const Vec3& pos) const {
        int x = (int)floor(pos.x / cellSize);
        int y = (int)floor(pos.y / cellSize);
        int z = (int)floor(pos.z / cellSize);
        return to_string(x) + "," + to_string(y) + "," + to_string(z);
    }

    bool isOccupied(const Vec3& pos) const {
        return occupiedCells.count(getKey(pos)) > 0;
    }

    void occupy(const Vec3& pos) {
        occupiedCells.insert(getKey(pos));
    }
};

int main() {
    auto startTime = chrono::high_resolution_clock::now();

    int V;
    double edgeLength;
    cin >> V >> edgeLength;

    vector<Vec3> containerVertices(V);
    for (int i = 0; i < V; i++) {
        cin >> containerVertices[i].x >> containerVertices[i].y >> containerVertices[i].z;
    }

    // Compute bounding box
    Vec3 minP = containerVertices[0], maxP = containerVertices[0];
    for (const auto& v : containerVertices) {
        minP.x = min(minP.x, v.x);
        minP.y = min(minP.y, v.y);
        minP.z = min(minP.z, v.z);
        maxP.x = max(maxP.x, v.x);
        maxP.y = max(maxP.y, v.y);
        maxP.z = max(maxP.z, v.z);
    }

    Vec3 size = maxP - minP;
    double volume = size.x * size.y * size.z;

    // Build convex hull for containment testing
    ConvexHull hull(containerVertices);

    // Key geometry parameters
    double circumradius = edgeLength * sqrt(3.0 / 8.0);
    double inradius = edgeLength / sqrt(24.0);

    // Adaptive grid spacing based on volume
    double spacing = edgeLength * 0.45;
    if (volume > 1000) spacing = edgeLength * 0.5;
    if (volume > 10000) spacing = edgeLength * 0.55;
    if (volume > 100000) spacing = edgeLength * 0.6;

    // Spatial hash for collision detection
    SpatialHash spatialHash(edgeLength * 0.7);

    vector<Tetrahedron> placed;

    // Multiple orientations for better packing
    vector<array<double, 3>> orientations = {
        {0, 0, 0},
        {0, 0, PI/3},
        {0, 0, 2*PI/3},
        {PI, 0, 0},
    };

    // Reduce orientations for large volumes
    if (volume > 10000) {
        orientations = {{0, 0, 0}, {0, 0, PI/3}};
    }
    if (volume > 100000) {
        orientations = {{0, 0, 0}};
    }

    double margin = circumradius * 1.1;

    // Grid-based placement
    for (const auto& orient : orientations) {
        for (double x = minP.x + margin; x <= maxP.x - margin; x += spacing) {
            for (double y = minP.y + margin; y <= maxP.y - margin; y += spacing) {
                for (double z = minP.z + margin; z <= maxP.z - margin; z += spacing) {
                    // Time check
                    auto currentTime = chrono::high_resolution_clock::now();
                    auto elapsed = chrono::duration_cast<chrono::seconds>(currentTime - startTime).count();
                    if (elapsed > 28) goto output;

                    Vec3 center(x, y, z);

                    // Quick spatial hash check
                    if (spatialHash.isOccupied(center)) {
                        continue;
                    }

                    // Check if center is inside container
                    if (!hull.contains(center, inradius)) {
                        continue;
                    }

                    // Create candidate tetrahedron
                    Tetrahedron candidate(center, orient[0], orient[1], orient[2]);
                    auto verts = candidate.getVertices(edgeLength);

                    // Check if all vertices are inside container
                    bool allInside = true;
                    for (const auto& v : verts) {
                        if (!hull.contains(v, EPSILON)) {
                            allInside = false;
                            break;
                        }
                    }

                    if (!allInside) continue;

                    // Check collision with nearby tetrahedrons
                    bool collides = false;
                    double minDist = edgeLength * 0.85;

                    for (const auto& tet : placed) {
                        double dist = (center - tet.center).length();
                        if (dist < minDist) {
                            collides = true;
                            break;
                        }
                    }

                    if (!collides) {
                        placed.push_back(candidate);
                        spatialHash.occupy(center);
                    }
                }
            }
        }
    }

output:
    // Output results
    cout << placed.size() << endl;
    cout.precision(10);
    for (const auto& tet : placed) {
        cout << tet.center.x << " " << tet.center.y << " " << tet.center.z << " "
             << tet.rx << " " << tet.ry << " " << tet.rz << endl;
    }

    return 0;
}
