#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <limits>

using namespace std;

struct Point {
    float x, y;
};

inline float dist2(const Point& a, const Point& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return dx * dx + dy * dy;
}

// Nearest neighbor heuristic - O(N^2) for small instances
vector<int> nearestNeighbor(const vector<Point>& cities) {
    int n = cities.size();
    vector<int> tour;
    tour.reserve(n);
    vector<bool> visited(n, false);

    tour.push_back(0);
    visited[0] = true;

    for (int i = 1; i < n; i++) {
        int last = tour.back();
        int nearest = -1;
        float minDist = numeric_limits<float>::max();

        for (int j = 0; j < n; j++) {
            if (!visited[j]) {
                float d = dist2(cities[last], cities[j]);
                if (d < minDist) {
                    minDist = d;
                    nearest = j;
                }
            }
        }

        tour.push_back(nearest);
        visited[nearest] = true;
    }

    return tour;
}

// Strip-based sweep - O(N log N) for medium instances
vector<int> stripTour(const vector<Point>& cities) {
    int n = cities.size();

    vector<int> indices(n);
    for (int i = 0; i < n; i++) indices[i] = i;

    // Sort by x-coordinate
    sort(indices.begin(), indices.end(), [&](int a, int b) {
        return cities[a].x < cities[b].x;
    });

    // Divide into vertical strips
    int numStrips = max(1, (int)sqrt(n) / 10);
    numStrips = max(1, min(numStrips, n));
    int stripSize = (n + numStrips - 1) / numStrips;

    vector<int> tour;
    tour.reserve(n);

    for (int strip = 0; strip < numStrips; strip++) {
        int start = strip * stripSize;
        int end = min(start + stripSize, n);

        vector<pair<float, int>> stripCities;
        for (int i = start; i < end; i++) {
            stripCities.push_back({cities[indices[i]].y, indices[i]});
        }

        // Alternate sweep direction for serpentine path
        if (strip % 2 == 0) {
            sort(stripCities.begin(), stripCities.end());
        } else {
            sort(stripCities.begin(), stripCities.end(), greater<pair<float, int>>());
        }

        for (const auto& p : stripCities) {
            tour.push_back(p.second);
        }
    }

    // Rotate to start at city 0
    auto it = find(tour.begin(), tour.end(), 0);
    if (it != tour.end()) {
        rotate(tour.begin(), it, tour.end());
    }

    return tour;
}

// Space-filling curve (Morton/Z-order) - O(N log N) for large instances
vector<int> spaceFilling(const vector<Point>& cities) {
    int n = cities.size();

    // Find bounding box
    float minX = cities[0].x, maxX = cities[0].x;
    float minY = cities[0].y, maxY = cities[0].y;
    for (const auto& p : cities) {
        minX = min(minX, p.x);
        maxX = max(maxX, p.x);
        minY = min(minY, p.y);
        maxY = max(maxY, p.y);
    }

    float rangeX = maxX - minX;
    float rangeY = maxY - minY;
    if (rangeX < 1e-9f) rangeX = 1.0f;
    if (rangeY < 1e-9f) rangeY = 1.0f;

    vector<pair<unsigned long long, int>> sorted;
    sorted.reserve(n);

    for (int i = 0; i < n; i++) {
        // Normalize coordinates to [0, 1]
        float nx = (cities[i].x - minX) / rangeX;
        float ny = (cities[i].y - minY) / rangeY;

        // Convert to integer coordinates
        unsigned int ix = (unsigned int)(nx * 0xFFFFFFFF);
        unsigned int iy = (unsigned int)(ny * 0xFFFFFFFF);

        // Compute Morton code (interleave bits)
        unsigned long long morton = 0;
        for (int b = 0; b < 32; b++) {
            morton |= ((unsigned long long)((ix >> b) & 1) << (2 * b));
            morton |= ((unsigned long long)((iy >> b) & 1) << (2 * b + 1));
        }

        sorted.push_back({morton, i});
    }

    sort(sorted.begin(), sorted.end());

    vector<int> tour;
    tour.reserve(n);
    for (const auto& p : sorted) {
        tour.push_back(p.second);
    }

    // Rotate to start at city 0
    auto it = find(tour.begin(), tour.end(), 0);
    if (it != tour.end()) {
        rotate(tour.begin(), it, tour.end());
    }

    return tour;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<Point> cities(n);
    for (int i = 0; i < n; i++) {
        cin >> cities[i].x >> cities[i].y;
    }

    vector<int> tour;

    // Adaptive strategy based on problem size
    if (n <= 5000) {
        // Small instances: use nearest neighbor (O(N^2))
        tour = nearestNeighbor(cities);
    } else if (n <= 100000) {
        // Medium instances: use strip-based sweep (O(N log N))
        tour = stripTour(cities);
    } else {
        // Large instances: use space-filling curve (O(N log N))
        tour = spaceFilling(cities);
    }

    // Output tour
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << " ";
        cout << tour[i];
    }
    cout << "\n";

    return 0;
}
