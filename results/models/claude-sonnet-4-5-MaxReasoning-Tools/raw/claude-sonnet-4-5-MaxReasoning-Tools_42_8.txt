OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
               OpExecutionMode %main Triangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

; Types
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
    %v3float = OpTypeVector %float 3
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
     %uint_3 = OpConstant %uint 3
  %float_0_7 = OpConstant %float 0.7
%float_one_third = OpConstant %float 0.333333343

; Input types
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_3
%ptr_in_v3 = OpTypePointer Input %v3float

; Output types
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; Variables
     %gl_in = OpVariable %ptr_in_arr_pv Input
    %inColor = OpVariable %ptr_in_arr_v3 Input
    %gl_out = OpVariable %ptr_out_pv Output
   %outColor = OpVariable %ptr_out_v3 Output

; Main function
       %main = OpFunction %void None %3
          %5 = OpLabel

; Load the 3 input positions
         %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0_ptr
         %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %p1_ptr
         %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
         %p2 = OpLoad %v4float %p2_ptr

; Compute centroid: C = (p0 + p1 + p2) / 3
         %sum01 = OpFAdd %v4float %p0 %p1
         %sum012 = OpFAdd %v4float %sum01 %p2
         %centroid = OpVectorTimesScalar %v4float %sum012 %float_one_third

; Process vertex 0: new_pos = C + 0.7 * (p0 - C)
         %diff0 = OpFSub %v4float %p0 %centroid
         %scaled0 = OpVectorTimesScalar %v4float %diff0 %float_0_7
         %new_p0 = OpFAdd %v4float %centroid %scaled0

; Emit vertex 0
         %out_pos_ptr_0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
         OpStore %out_pos_ptr_0 %new_p0
         %color0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
         %color0 = OpLoad %v3float %color0_ptr
         OpStore %outColor %color0
         OpEmitVertex

; Process vertex 1: new_pos = C + 0.7 * (p1 - C)
         %diff1 = OpFSub %v4float %p1 %centroid
         %scaled1 = OpVectorTimesScalar %v4float %diff1 %float_0_7
         %new_p1 = OpFAdd %v4float %centroid %scaled1

; Emit vertex 1
         OpStore %out_pos_ptr_0 %new_p1
         %color1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
         %color1 = OpLoad %v3float %color1_ptr
         OpStore %outColor %color1
         OpEmitVertex

; Process vertex 2: new_pos = C + 0.7 * (p2 - C)
         %diff2 = OpFSub %v4float %p2 %centroid
         %scaled2 = OpVectorTimesScalar %v4float %diff2 %float_0_7
         %new_p2 = OpFAdd %v4float %centroid %scaled2

; Emit vertex 2
         OpStore %out_pos_ptr_0 %new_p2
         %color2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
         %color2 = OpLoad %v3float %color2_ptr
         OpStore %outColor %color2
         OpEmitVertex

; End the triangle strip
         OpEndPrimitive
         OpReturn
         OpFunctionEnd
