using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Text;

class Program
{
    static void Main()
    {
        using var stdin = Console.OpenStandardInput();
        using var reader = new StreamReader(stdin, Encoding.ASCII);

        // Read header
        var header = reader.ReadLine().Split(' ');
        int xCount = int.Parse(header[0]);
        int yCount = int.Parse(header[1]);
        int zCount = int.Parse(header[2]);

        // Read voxel data
        var voxels = new byte[xCount, yCount, zCount];
        var buffer = new byte[xCount * yCount * zCount];
        int totalRead = 0;
        while (totalRead < buffer.Length)
        {
            int read = stdin.Read(buffer, totalRead, buffer.Length - totalRead);
            if (read == 0) break;
            totalRead += read;
        }

        int idx = 0;
        for (int x = 0; x < xCount; x++)
            for (int y = 0; y < yCount; y++)
                for (int z = 0; z < zCount; z++)
                    voxels[x, y, z] = buffer[idx++];

        var solver = new MiningSolver(xCount, yCount, zCount, voxels);
        var operations = solver.Solve();

        // Output
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.Write($"{op.Steps.Count} {op.DigX} {op.DigY} {op.DigZ}");
            foreach (var step in op.Steps)
                Console.Write($" {step.X} {step.Y}");
            Console.WriteLine($" {op.DestX} {op.DestY}");
        }
    }
}

struct Voxel
{
    public int X, Y, Z;
    public byte Data;
    public int Mass => Data & 15;
    public int Value => Data >> 4;
    public long Profit => (long)Mass * Value * 1000;

    public Voxel(int x, int y, int z, byte data)
    {
        X = x; Y = y; Z = z; Data = data;
    }
}

struct Operation
{
    public int DigX, DigY, DigZ;
    public List<(int X, int Y)> Steps;
    public int DestX, DestY;
}

class MiningSolver
{
    int xCount, yCount, zCount;
    byte[,,] voxels;
    int[,] groundHeight;
    HashSet<(int, int, int)> removed;
    List<Operation> operations;

    public MiningSolver(int xCount, int yCount, int zCount, byte[,,] voxels)
    {
        this.xCount = xCount;
        this.yCount = yCount;
        this.zCount = zCount;
        this.voxels = voxels;
        this.groundHeight = new int[xCount, yCount];
        this.removed = new HashSet<(int, int, int)>();
        this.operations = new List<Operation>();

        // Initialize ground heights
        for (int x = 0; x < xCount; x++)
            for (int y = 0; y < yCount; y++)
            {
                int h = zCount;
                for (int z = zCount - 1; z >= 0; z--)
                {
                    if (voxels[x, y, z] != 0)
                    {
                        h = z + 1;
                        break;
                    }
                }
                groundHeight[x, y] = h;
            }
    }

    public List<Operation> Solve()
    {
        // Collect valuable voxels
        var valuable = new List<Voxel>();
        for (int x = 0; x < xCount; x++)
            for (int y = 0; y < yCount; y++)
                for (int z = 0; z < zCount; z++)
                {
                    byte data = voxels[x, y, z];
                    if (data != 0)
                    {
                        int value = data >> 4;
                        if (value > 0)
                            valuable.Add(new Voxel(x, y, z, data));
                    }
                }

        // Sort by profit descending
        valuable.Sort((a, b) => b.Profit.CompareTo(a.Profit));

        // Process most valuable first
        int maxToProcess = Math.Min(valuable.Count, xCount * yCount < 100 ? 1000 : 10000);
        for (int i = 0; i < maxToProcess; i++)
        {
            var v = valuable[i];
            if (TryMineVoxel(v.X, v.Y, v.Z, 0, 0))
            {
                // Success
            }
        }

        return operations;
    }

    bool IsExposed(int x, int y, int z)
    {
        if (removed.Contains((x, y, z))) return false;
        if (voxels[x, y, z] == 0) return false;

        // Check if top is exposed
        if (z == groundHeight[x, y] - 1) return true;
        if (z < zCount - 1 && (voxels[x, y, z + 1] == 0 || removed.Contains((x, y, z + 1))))
            return true;

        // Check sides
        if (x > 0 && (voxels[x - 1, y, z] == 0 || removed.Contains((x - 1, y, z)))) return true;
        if (x < xCount - 1 && (voxels[x + 1, y, z] == 0 || removed.Contains((x + 1, y, z)))) return true;
        if (y > 0 && (voxels[x, y - 1, z] == 0 || removed.Contains((x, y - 1, z)))) return true;
        if (y < yCount - 1 && (voxels[x, y + 1, z] == 0 || removed.Contains((x, y + 1, z)))) return true;

        return false;
    }

    bool TryMineVoxel(int x, int y, int z, int destX, int destY)
    {
        if (!IsExposed(x, y, z)) return false;

        var path = FindPath(x, y, z, destX, destY);
        if (path == null) return false;

        // Calculate profit
        byte data = voxels[x, y, z];
        int mass = data & 15;
        int value = data >> 4;
        long revenue = (long)mass * value * 1000;

        // Calculate cost
        long cost = 0;
        int currentZ = z;
        foreach (var step in path)
        {
            int nextZ = groundHeight[step.X, step.Y];
            if (nextZ > currentZ)
                cost += (long)mass * 5;
            else
                cost += (long)mass * 1;
            currentZ = nextZ;
        }

        // Only proceed if profitable
        if (revenue <= cost && destX == 0 && destY == 0) return false;

        // Create operation
        var op = new Operation
        {
            DigX = x,
            DigY = y,
            DigZ = z,
            Steps = path,
            DestX = destX,
            DestY = destY
        };

        operations.Add(op);
        removed.Add((x, y, z));

        // Update ground height at destination
        if (destX != 0 || destY != 0)
            groundHeight[destX, destY]++;

        // Update source ground height
        if (z == groundHeight[x, y] - 1)
        {
            int newH = 0;
            for (int zz = z - 1; zz >= 0; zz--)
            {
                if (voxels[x, y, zz] != 0 && !removed.Contains((x, y, zz)))
                {
                    newH = zz + 1;
                    break;
                }
            }
            groundHeight[x, y] = newH;
        }

        return true;
    }

    List<(int X, int Y)> FindPath(int fromX, int fromY, int fromZ, int toX, int toY)
    {
        // Simple A* pathfinding with slope constraints
        var open = new PriorityQueue<(int X, int Y, int Z), int>();
        var cameFrom = new Dictionary<(int, int), (int, int)>();
        var gScore = new Dictionary<(int, int), int>();

        int startZ = groundHeight[fromX, fromY];
        open.Enqueue((fromX, fromY, startZ), 0);
        gScore[(fromX, fromY)] = 0;

        while (open.Count > 0)
        {
            var current = open.Dequeue();

            if (current.X == toX && current.Y == toY)
            {
                // Reconstruct path
                var path = new List<(int, int)>();
                var pos = (current.X, current.Y);
                while (cameFrom.ContainsKey(pos))
                {
                    path.Add(pos);
                    pos = cameFrom[pos];
                }
                path.Reverse();
                return path;
            }

            // Try all four directions
            var neighbors = new[] {
                (current.X - 1, current.Y),
                (current.X + 1, current.Y),
                (current.X, current.Y - 1),
                (current.X, current.Y + 1)
            };

            foreach (var next in neighbors)
            {
                if (next.Item1 < 0 || next.Item1 >= xCount || next.Item2 < 0 || next.Item2 >= yCount)
                    continue;

                int nextZ = groundHeight[next.Item1, next.Item2];

                // Check slope constraint (1 Z per X/Y)
                if (Math.Abs(nextZ - current.Z) > 1)
                    continue;

                int tentativeG = gScore[(current.X, current.Y)] + 1;

                if (!gScore.ContainsKey(next) || tentativeG < gScore[next])
                {
                    cameFrom[next] = (current.X, current.Y);
                    gScore[next] = tentativeG;
                    int h = Math.Abs(next.Item1 - toX) + Math.Abs(next.Item2 - toY);
                    open.Enqueue((next.Item1, next.Item2, nextZ), tentativeG + h);
                }
            }
        }

        return null;
    }
}

class PriorityQueue<T, P> where P : IComparable<P>
{
    List<(T item, P priority)> heap = new List<(T, P)>();

    public int Count => heap.Count;

    public void Enqueue(T item, P priority)
    {
        heap.Add((item, priority));
        int i = heap.Count - 1;
        while (i > 0)
        {
            int parent = (i - 1) / 2;
            if (heap[i].priority.CompareTo(heap[parent].priority) >= 0) break;
            var tmp = heap[i];
            heap[i] = heap[parent];
            heap[parent] = tmp;
            i = parent;
        }
    }

    public T Dequeue()
    {
        var result = heap[0].item;
        heap[0] = heap[heap.Count - 1];
        heap.RemoveAt(heap.Count - 1);

        int i = 0;
        while (true)
        {
            int left = 2 * i + 1;
            int right = 2 * i + 2;
            int smallest = i;

            if (left < heap.Count && heap[left].priority.CompareTo(heap[smallest].priority) < 0)
                smallest = left;
            if (right < heap.Count && heap[right].priority.CompareTo(heap[smallest].priority) < 0)
                smallest = right;

            if (smallest == i) break;

            var tmp = heap[i];
            heap[i] = heap[smallest];
            heap[smallest] = tmp;
            i = smallest;
        }

        return result;
    }
}
