OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
    %v3float = OpTypeVector %float 3
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
     %uint_2 = OpConstant %uint 2
  %float_0_5 = OpConstant %float 0.5
 %float_0_04 = OpConstant %float 0.04
%float_0_005 = OpConstant %float 0.005

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_color_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_color = OpTypePointer Input %arr_color_in
   %color_in = OpVariable %ptr_in_arr_color Input
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %3
          %5 = OpLabel

        ; Load positions
       %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0_ptr
       %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %p1_ptr

        ; Load colors
       %c0_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
         %c0 = OpLoad %v3float %c0_ptr
       %c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
         %c1 = OpLoad %v3float %c1_ptr

        ; Compute center (for perpendicular calculation)
       %center = OpVectorTimesScalar %v4float %p0 %float_0_5
        %half_p1 = OpVectorTimesScalar %v4float %p1 %float_0_5
     %midpoint = OpFAdd %v4float %center %half_p1

        ; Compute line direction
        %dir = OpFSub %v4float %p1 %p0

        ; Compute perpendicular (swap xy, negate x)
      %dir_x = OpCompositeExtract %float %dir 0
      %dir_y = OpCompositeExtract %float %dir 1
      %dir_z = OpCompositeExtract %float %dir 2
      %dir_w = OpCompositeExtract %float %dir 3
   %neg_dir_y = OpFNegate %float %dir_y
       %perp = OpCompositeConstruct %v4float %neg_dir_y %dir_x %dir_z %dir_w

        ; Normalize perpendicular (approximate - length then divide)
    %perp_x2 = OpFMul %float %neg_dir_y %neg_dir_y
    %perp_y2 = OpFMul %float %dir_x %dir_x
  %len_sq = OpFAdd %float %perp_x2 %perp_y2
     %len = OpExtInst %float %1 Sqrt %len_sq
   %perp_norm_x = OpFDiv %float %neg_dir_y %len
   %perp_norm_y = OpFDiv %float %dir_x %len
  %perp_norm = OpCompositeConstruct %v4float %perp_norm_x %perp_norm_y %dir_z %dir_w

        ; Scale perpendicular for start and end widths
  %perp_start = OpVectorTimesScalar %v4float %perp_norm %float_0_04
    %perp_end = OpVectorTimesScalar %v4float %perp_norm %float_0_005

        ; Emit vertex 0: p0 + perp_start
        %v0 = OpFAdd %v4float %p0 %perp_start
     %gl_pos_0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %gl_pos_0 %v0
               OpStore %outColor %c0
               OpEmitVertex

        ; Emit vertex 1: p0 - perp_start
        %v1 = OpFSub %v4float %p0 %perp_start
     %gl_pos_1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %gl_pos_1 %v1
               OpStore %outColor %c0
               OpEmitVertex

        ; Emit vertex 2: p1 + perp_end
        %v2 = OpFAdd %v4float %p1 %perp_end
     %gl_pos_2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %gl_pos_2 %v2
               OpStore %outColor %c1
               OpEmitVertex

        ; Emit vertex 3: p1 - perp_end
        %v3 = OpFSub %v4float %p1 %perp_end
     %gl_pos_3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %gl_pos_3 %v3
               OpStore %outColor %c1
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
