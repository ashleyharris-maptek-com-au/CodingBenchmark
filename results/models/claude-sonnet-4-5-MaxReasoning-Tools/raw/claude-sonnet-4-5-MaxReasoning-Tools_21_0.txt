#include <iostream>
#include <vector>
#include <queue>
#include <set>
#include <deque>
#include <algorithm>
#include <cmath>

using namespace std;

struct Position {
    vector<int> coords;

    Position() {}
    Position(int d) : coords(d, 0) {}
    Position(const vector<int>& c) : coords(c) {}

    bool operator<(const Position& other) const {
        return coords < other.coords;
    }

    bool operator==(const Position& other) const {
        return coords == other.coords;
    }

    Position move(int axis, int dir) const {
        Position result = *this;
        result.coords[axis] += dir;
        return result;
    }

    int manhattan(const Position& other) const {
        int dist = 0;
        for (size_t i = 0; i < coords.size(); i++) {
            dist += abs(coords[i] - other.coords[i]);
        }
        return dist;
    }
};

struct Move {
    int axis;
    int direction;
};

struct State {
    Position head;
    deque<Position> body;
    set<Position> food;
    int turns_used;

    bool isValid(const Position& pos, const vector<int>& bounds, const set<Position>& obstacles) const {
        // Check bounds
        for (size_t i = 0; i < pos.coords.size(); i++) {
            if (pos.coords[i] < 0 || pos.coords[i] >= bounds[i]) {
                return false;
            }
        }

        // Check obstacles
        if (obstacles.count(pos)) {
            return false;
        }

        // Check self collision (body excluding tail which will move)
        for (size_t i = 0; i < body.size() - 1; i++) {
            if (body[i] == pos) {
                return false;
            }
        }

        return true;
    }

    void applyMove(const Move& m, bool grow) {
        Position newHead = head.move(m.axis, m.direction);
        body.push_front(head);
        head = newHead;

        if (!grow) {
            body.pop_back();
        }
    }
};

struct BFSNode {
    Position pos;
    vector<Move> path;
    set<Position> visited_body;

    int dist;

    bool operator>(const BFSNode& other) const {
        return dist > other.dist;
    }
};

vector<Move> findPath(const State& state, const vector<int>& bounds, const set<Position>& obstacles) {
    int D = bounds.size();

    // Find nearest food using BFS
    priority_queue<BFSNode, vector<BFSNode>, greater<BFSNode>> pq;
    set<Position> visited;

    BFSNode start;
    start.pos = state.head;
    start.dist = 0;
    start.visited_body.insert(state.body.begin(), state.body.end());

    pq.push(start);
    visited.insert(start.pos);

    while (!pq.empty()) {
        BFSNode current = pq.top();
        pq.pop();

        // Check if we reached food
        if (state.food.count(current.pos)) {
            return current.path;
        }

        // Try all possible moves
        for (int axis = 0; axis < D; axis++) {
            for (int dir = -1; dir <= 1; dir += 2) {
                Position next = current.pos.move(axis, dir);

                if (visited.count(next)) continue;

                // Check bounds
                bool valid = true;
                for (int i = 0; i < D; i++) {
                    if (next.coords[i] < 0 || next.coords[i] >= bounds[i]) {
                        valid = false;
                        break;
                    }
                }
                if (!valid) continue;

                // Check obstacles
                if (obstacles.count(next)) continue;

                // Check body collision (simulate snake movement)
                set<Position> future_body = current.visited_body;
                if (current.path.size() < state.body.size()) {
                    // Tail hasn't moved yet in simulation
                    if (future_body.count(next)) continue;
                } else {
                    // Remove tail position from check
                    if (future_body.count(next) && current.path.size() > 0) {
                        continue;
                    }
                }

                BFSNode next_node;
                next_node.pos = next;
                next_node.path = current.path;
                next_node.path.push_back({axis, dir});
                next_node.dist = current.dist + 1;

                // Update body simulation
                next_node.visited_body = future_body;
                next_node.visited_body.insert(current.pos);
                if (next_node.path.size() > state.body.size()) {
                    // Remove tail
                    size_t tail_idx = state.body.size() - (next_node.path.size() - state.body.size()) - 1;
                    if (tail_idx < state.body.size()) {
                        next_node.visited_body.erase(state.body[tail_idx]);
                    }
                }

                visited.insert(next);
                pq.push(next_node);
            }
        }
    }

    // No path found, try any valid move
    for (int axis = 0; axis < D; axis++) {
        for (int dir = -1; dir <= 1; dir += 2) {
            Position next = state.head.move(axis, dir);
            if (state.isValid(next, bounds, obstacles)) {
                return {{axis, dir}};
            }
        }
    }

    // No valid move (should not happen in valid game)
    return {{0, 1}};
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int D, max_turns;
    cin >> D >> max_turns;

    vector<int> bounds(D);
    for (int i = 0; i < D; i++) {
        cin >> bounds[i];
    }

    State state;
    int snake_length;
    cin >> snake_length;

    state.head.coords.resize(D);
    for (int i = 0; i < D; i++) {
        cin >> state.head.coords[i];
    }

    // Initialize body (all at start position)
    for (int i = 1; i < snake_length; i++) {
        state.body.push_back(state.head);
    }

    int F;
    cin >> F;
    for (int i = 0; i < F; i++) {
        Position food(D);
        for (int j = 0; j < D; j++) {
            cin >> food.coords[j];
        }
        state.food.insert(food);
    }

    int O;
    cin >> O;
    set<Position> obstacles;
    for (int i = 0; i < O; i++) {
        Position obs(D);
        for (int j = 0; j < D; j++) {
            cin >> obs.coords[j];
        }
        obstacles.insert(obs);
    }

    state.turns_used = 0;

    // Main game loop
    while (!state.food.empty() && state.turns_used < max_turns) {
        vector<Move> path = findPath(state, bounds, obstacles);

        if (path.empty()) {
            break;
        }

        Move m = path[0];
        Position nextHead = state.head.move(m.axis, m.direction);

        // Check if next position is valid
        if (!state.isValid(nextHead, bounds, obstacles)) {
            break;
        }

        // Check if we're eating food
        bool grow = state.food.count(nextHead) > 0;
        if (grow) {
            state.food.erase(nextHead);
        }

        // Apply move
        state.applyMove(m, grow);
        state.turns_used++;

        // Output move
        cout << m.axis << " " << m.direction << "\n";
    }

    return 0;
}
