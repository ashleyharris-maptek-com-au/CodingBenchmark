using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;
using System.Collections.Concurrent;

class MaximumClique
{
    static int n, m;
    static bool[,] adj;
    static int[] degree;
    static int[] bestClique;
    static int bestSize;
    static object lockObj = new object();
    static DateTime startTime;
    static TimeSpan timeLimit = TimeSpan.FromMinutes(4.5);
    static CancellationTokenSource cts = new CancellationTokenSource();

    static void Main()
    {
        startTime = DateTime.Now;
        ReadInput();

        // Use exact algorithm for smaller graphs, heuristic for larger
        if (n <= 5000)
        {
            SolveExact();
        }
        else
        {
            SolveHeuristic();
        }

        // Output result
        Console.WriteLine(bestSize);
        Console.WriteLine(string.Join(" ", bestClique.Take(bestSize)));
    }

    static void ReadInput()
    {
        var firstLine = Console.ReadLine().Split();
        n = int.Parse(firstLine[0]);
        m = int.Parse(firstLine[1]);

        adj = new bool[n, n];
        degree = new int[n];

        for (int i = 0; i < m; i++)
        {
            var edge = Console.ReadLine().Split();
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            adj[u, v] = adj[v, u] = true;
            degree[u]++;
            degree[v]++;
        }

        bestClique = new int[n];
        bestSize = 0;
    }

    static void SolveExact()
    {
        // Order vertices by degree (descending)
        var vertices = Enumerable.Range(0, n)
            .OrderByDescending(v => degree[v])
            .ToArray();

        // Try parallel search from multiple starting points
        int numThreads = Math.Min(12, Math.Max(1, n / 100));
        var tasks = new List<Task>();

        for (int t = 0; t < numThreads; t++)
        {
            int threadId = t;
            tasks.Add(Task.Run(() =>
            {
                var clique = new List<int>();
                var candidates = new List<int>();

                // Each thread starts from different vertices
                for (int i = threadId; i < Math.Min(100, n); i += numThreads)
                {
                    if (TimeElapsed() > timeLimit || cts.Token.IsCancellationRequested)
                        break;

                    candidates.Clear();
                    clique.Clear();
                    clique.Add(vertices[i]);

                    // Add neighbors as candidates
                    for (int j = 0; j < n; j++)
                    {
                        if (adj[vertices[i], j])
                            candidates.Add(j);
                    }

                    BranchAndBound(clique, candidates);
                }
            }, cts.Token));
        }

        Task.WaitAll(tasks.ToArray());

        // If no solution found, use greedy
        if (bestSize == 0)
        {
            GreedyClique();
        }
    }

    static void BranchAndBound(List<int> clique, List<int> candidates)
    {
        if (TimeElapsed() > timeLimit || cts.Token.IsCancellationRequested)
            return;

        // Update best if current is better
        if (clique.Count > bestSize)
        {
            lock (lockObj)
            {
                if (clique.Count > bestSize)
                {
                    bestSize = clique.Count;
                    for (int i = 0; i < clique.Count; i++)
                        bestClique[i] = clique[i];
                }
            }
        }

        // Prune if can't improve
        if (clique.Count + candidates.Count <= bestSize)
            return;

        // Order candidates by degree in subgraph
        var orderedCandidates = candidates
            .OrderByDescending(v => candidates.Count(u => adj[v, u]))
            .ToList();

        for (int i = 0; i < orderedCandidates.Count; i++)
        {
            if (TimeElapsed() > timeLimit || cts.Token.IsCancellationRequested)
                return;

            int v = orderedCandidates[i];

            // Prune
            if (clique.Count + orderedCandidates.Count - i <= bestSize)
                break;

            // Check if v is connected to all vertices in current clique
            bool isConnected = true;
            for (int j = 0; j < clique.Count; j++)
            {
                if (!adj[v, clique[j]])
                {
                    isConnected = false;
                    break;
                }
            }

            if (isConnected)
            {
                // Add v to clique
                clique.Add(v);

                // New candidates: vertices connected to all in clique
                var newCandidates = new List<int>();
                for (int j = i + 1; j < orderedCandidates.Count; j++)
                {
                    int u = orderedCandidates[j];
                    if (adj[v, u])
                        newCandidates.Add(u);
                }

                BranchAndBound(clique, newCandidates);

                // Remove v from clique
                clique.RemoveAt(clique.Count - 1);
            }
        }
    }

    static void SolveHeuristic()
    {
        // Run multiple greedy heuristics in parallel
        var tasks = new List<Task>();

        for (int t = 0; t < 12; t++)
        {
            int seed = t;
            tasks.Add(Task.Run(() => GreedyCliqueRandom(seed)));
        }

        // Also run deterministic greedy
        GreedyClique();

        Task.WaitAll(tasks.ToArray());

        // Local search improvement
        LocalSearchImprove();
    }

    static void GreedyClique()
    {
        var clique = new List<int>();
        var candidates = Enumerable.Range(0, n)
            .OrderByDescending(v => degree[v])
            .ToList();

        while (candidates.Count > 0)
        {
            int v = candidates[0];
            clique.Add(v);

            // Keep only candidates connected to v
            candidates = candidates
                .Where(u => u != v && adj[v, u])
                .OrderByDescending(u => candidates.Count(w => adj[u, w]))
                .ToList();
        }

        UpdateBest(clique);
    }

    static void GreedyCliqueRandom(int seed)
    {
        var rnd = new Random(seed);

        for (int iter = 0; iter < 100 && TimeElapsed() < timeLimit; iter++)
        {
            var clique = new List<int>();
            var candidates = Enumerable.Range(0, n).ToList();

            while (candidates.Count > 0)
            {
                // Pick vertex with high degree, with some randomness
                var sorted = candidates
                    .OrderByDescending(v => degree[v] + rnd.NextDouble() * 10)
                    .ToList();

                int v = sorted[0];
                clique.Add(v);

                // Keep only candidates connected to v
                candidates = candidates
                    .Where(u => u != v && adj[v, u])
                    .ToList();
            }

            UpdateBest(clique);
        }
    }

    static void LocalSearchImprove()
    {
        if (bestSize == 0) return;

        var clique = new HashSet<int>(bestClique.Take(bestSize));
        var rnd = new Random();

        for (int iter = 0; iter < 1000 && TimeElapsed() < timeLimit; iter++)
        {
            // Try to expand
            bool expanded = false;
            var candidates = Enumerable.Range(0, n)
                .Where(v => !clique.Contains(v))
                .Where(v => clique.All(u => adj[v, u]))
                .ToList();

            if (candidates.Count > 0)
            {
                clique.Add(candidates[0]);
                expanded = true;
            }

            if (expanded)
            {
                UpdateBest(clique.ToList());
            }
            else
            {
                // Try swap: remove one, add another
                if (clique.Count > 1 && rnd.Next(10) < 3)
                {
                    var toRemove = clique.ElementAt(rnd.Next(clique.Count));
                    clique.Remove(toRemove);
                }
            }
        }
    }

    static void UpdateBest(List<int> clique)
    {
        if (clique.Count > bestSize)
        {
            lock (lockObj)
            {
                if (clique.Count > bestSize)
                {
                    bestSize = clique.Count;
                    for (int i = 0; i < clique.Count; i++)
                        bestClique[i] = clique[i];
                }
            }
        }
    }

    static TimeSpan TimeElapsed()
    {
        return DateTime.Now - startTime;
    }
}
