use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read n and m
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let n = parts[0];
    let m = parts[1];

    // Build adjacency list
    let mut adj = vec![vec![]; n];
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let edge: Vec<usize> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        let u = edge[0];
        let v = edge[1];
        adj[u].push(v);
        adj[v].push(u);
    }

    let half = n / 2;
    let mut best_cut = usize::MAX;
    let mut best_partition = Vec::new();

    // Generate all possible bisections and find minimum cut
    let mut partition = vec![0; half];
    for i in 0..half {
        partition[i] = i;
    }

    loop {
        // Mark which vertices are in partition A
        let mut in_partition = vec![false; n];
        for &v in &partition {
            in_partition[v] = true;
        }

        // Count cut edges
        let mut cut = 0;
        for u in 0..n {
            if in_partition[u] {
                for &v in &adj[u] {
                    if !in_partition[v] {
                        cut += 1;
                    }
                }
            }
        }

        // Update best if this is better
        if cut < best_cut {
            best_cut = cut;
            best_partition = partition.clone();
        }

        // Generate next combination
        if !next_combination(&mut partition, n) {
            break;
        }
    }

    // Output result
    println!("{}", best_cut);
    for i in 0..best_partition.len() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", best_partition[i]);
    }
    println!();
}

fn next_combination(a: &mut Vec<usize>, n: usize) -> bool {
    let k = a.len();
    // Find rightmost element that can be incremented
    for i in (0..k).rev() {
        if a[i] < n - k + i {
            a[i] += 1;
            // Reset all elements to the right
            for j in i+1..k {
                a[j] = a[j-1] + 1;
            }
            return true;
        }
    }
    false
}
