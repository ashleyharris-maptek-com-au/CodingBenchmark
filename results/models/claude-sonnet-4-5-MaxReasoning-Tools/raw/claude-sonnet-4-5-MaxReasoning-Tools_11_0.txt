use std::io::{self, BufRead};
use std::f64::consts::PI;

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn new(x: f64, y: f64) -> Self {
        Point { x, y }
    }

    fn rotate(&self, angle: f64) -> Point {
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        Point {
            x: self.x * cos_a - self.y * sin_a,
            y: self.x * sin_a + self.y * cos_a,
        }
    }

    fn translate(&self, dx: f64, dy: f64) -> Point {
        Point {
            x: self.x + dx,
            y: self.y + dy,
        }
    }
}

#[derive(Clone, Debug)]
struct Polygon {
    vertices: Vec<Point>,
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        Polygon { vertices }
    }

    fn rotate(&self, angle: f64) -> Polygon {
        Polygon {
            vertices: self.vertices.iter().map(|p| p.rotate(angle)).collect(),
        }
    }

    fn translate(&self, dx: f64, dy: f64) -> Polygon {
        Polygon {
            vertices: self.vertices.iter().map(|p| p.translate(dx, dy)).collect(),
        }
    }

    fn bounds(&self) -> (f64, f64, f64, f64) {
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for p in &self.vertices {
            min_x = min_x.min(p.x);
            max_x = max_x.max(p.x);
            min_y = min_y.min(p.y);
            max_y = max_y.max(p.y);
        }

        (min_x, min_y, max_x, max_y)
    }

    fn contains_point(&self, point: &Point) -> bool {
        let mut inside = false;
        let n = self.vertices.len();

        for i in 0..n {
            let j = (i + 1) % n;
            let vi = &self.vertices[i];
            let vj = &self.vertices[j];

            if ((vi.y > point.y) != (vj.y > point.y)) &&
               (point.x < (vj.x - vi.x) * (point.y - vi.y) / (vj.y - vi.y) + vi.x) {
                inside = !inside;
            }
        }

        inside
    }

    fn contains_polygon(&self, other: &Polygon) -> bool {
        for v in &other.vertices {
            if !self.contains_point(v) {
                return false;
            }
        }
        true
    }

    fn intersects(&self, other: &Polygon) -> bool {
        for v in &other.vertices {
            if self.contains_point(v) {
                return true;
            }
        }
        for v in &self.vertices {
            if other.contains_point(v) {
                return true;
            }
        }

        for i in 0..self.vertices.len() {
            let j = (i + 1) % self.vertices.len();
            for k in 0..other.vertices.len() {
                let l = (k + 1) % other.vertices.len();
                if segments_intersect(
                    &self.vertices[i], &self.vertices[j],
                    &other.vertices[k], &other.vertices[l]
                ) {
                    return true;
                }
            }
        }

        false
    }
}

fn segments_intersect(p1: &Point, p2: &Point, p3: &Point, p4: &Point) -> bool {
    let d1 = direction(p3, p4, p1);
    let d2 = direction(p3, p4, p2);
    let d3 = direction(p1, p2, p3);
    let d4 = direction(p1, p2, p4);

    if ((d1 > 0.0 && d2 < 0.0) || (d1 < 0.0 && d2 > 0.0)) &&
       ((d3 > 0.0 && d4 < 0.0) || (d3 < 0.0 && d4 > 0.0)) {
        return true;
    }

    false
}

fn direction(p1: &Point, p2: &Point, p3: &Point) -> f64 {
    (p3.x - p1.x) * (p2.y - p1.y) - (p2.x - p1.x) * (p3.y - p1.y)
}

struct Placement {
    stock_index: usize,
    x: f64,
    y: f64,
    rotation: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines().map(|l| l.unwrap());

    let first_line = lines.next().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let sv = parts[0];
    let np = parts[1];

    let mut stock_vertices = Vec::new();
    for _ in 0..sv {
        let line = lines.next().unwrap();
        let coords: Vec<f64> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        stock_vertices.push(Point::new(coords[0], coords[1]));
    }
    let stock = Polygon::new(stock_vertices);

    let mut pieces = Vec::new();
    for _ in 0..np {
        let pv_line = lines.next().unwrap();
        let pv: usize = pv_line.trim().parse().unwrap();

        let mut piece_vertices = Vec::new();
        for _ in 0..pv {
            let line = lines.next().unwrap();
            let coords: Vec<f64> = line.split_whitespace()
                .map(|s| s.parse().unwrap())
                .collect();
            piece_vertices.push(Point::new(coords[0], coords[1]));
        }
        pieces.push(Polygon::new(piece_vertices));
    }

    let placements = solve(&stock, &pieces);

    let num_stocks = if placements.is_empty() {
        0
    } else {
        placements.iter().map(|p| p.stock_index).max().unwrap() + 1
    };

    println!("{}", num_stocks);
    for p in placements {
        println!("{} {} {} {}", p.stock_index, p.x, p.y, p.rotation);
    }
}

fn solve(stock: &Polygon, pieces: &[Polygon]) -> Vec<Placement> {
    let mut placements = Vec::new();
    let mut stocks = vec![Vec::new()];

    let rotation_steps = if pieces.len() < 50 {
        16
    } else if pieces.len() < 500 {
        8
    } else {
        4
    };

    let grid_step = if pieces.len() < 100 {
        2.0
    } else if pieces.len() < 1000 {
        5.0
    } else {
        10.0
    };

    for piece in pieces.iter() {
        let mut placed = false;

        for stock_idx in 0..stocks.len() {
            if let Some((x, y, rot)) = try_place_on_stock(
                stock, piece, &stocks[stock_idx], rotation_steps, grid_step
            ) {
                placements.push(Placement {
                    stock_index: stock_idx,
                    x,
                    y,
                    rotation: rot,
                });
                stocks[stock_idx].push(piece.rotate(rot).translate(x, y));
                placed = true;
                break;
            }
        }

        if !placed {
            if let Some((x, y, rot)) = try_place_on_stock(
                stock, piece, &Vec::new(), rotation_steps, grid_step
            ) {
                let stock_idx = stocks.len();
                placements.push(Placement {
                    stock_index: stock_idx,
                    x,
                    y,
                    rotation: rot,
                });
                stocks.push(vec![piece.rotate(rot).translate(x, y)]);
            } else {
                let stock_idx = stocks.len();
                placements.push(Placement {
                    stock_index: stock_idx,
                    x: 0.0,
                    y: 0.0,
                    rotation: 0.0,
                });
                stocks.push(vec![piece.clone()]);
            }
        }
    }

    placements
}

fn try_place_on_stock(
    stock: &Polygon,
    piece: &Polygon,
    placed_pieces: &[Polygon],
    rotation_steps: usize,
    grid_step: f64,
) -> Option<(f64, f64, f64)> {
    let stock_bounds = stock.bounds();

    for r in 0..rotation_steps {
        let angle = (r as f64) * 2.0 * PI / (rotation_steps as f64);
        let rotated_piece = piece.rotate(angle);
        let piece_bounds = rotated_piece.bounds();
        let piece_width = piece_bounds.2 - piece_bounds.0;
        let piece_height = piece_bounds.3 - piece_bounds.1;

        let mut y = stock_bounds.1;
        while y <= stock_bounds.3 - piece_height {
            let mut x = stock_bounds.0;
            while x <= stock_bounds.2 - piece_width {
                let offset_x = x - piece_bounds.0;
                let offset_y = y - piece_bounds.1;
                let translated = rotated_piece.translate(offset_x, offset_y);

                if stock.contains_polygon(&translated) {
                    let mut valid = true;
                    for placed in placed_pieces {
                        if translated.intersects(placed) {
                            valid = false;
                            break;
                        }
                    }

                    if valid {
                        return Some((offset_x, offset_y, angle));
                    }
                }

                x += grid_step;
            }
            y += grid_step;
        }
    }

    None
}
