OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %in_color %gl_out %out_color
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 12
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %in_color Location 2
               OpDecorate %out_color Location 0

       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
    %v3float = OpTypeVector %float 3
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
      %int_4 = OpConstant %int 4
     %uint_6 = OpConstant %uint 6
   %float_05 = OpConstant %float 0.5
 %float_008 = OpConstant %float 0.008
    %float_1 = OpConstant %float 1.0

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_color_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_color = OpTypePointer Input %arr_color_in
   %in_color = OpVariable %ptr_in_arr_color Input
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
  %out_color = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %3
          %5 = OpLabel

; Load positions of main triangle vertices (indices 0, 2, 4)
      %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0_ptr
      %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
         %p2 = OpLoad %v4float %p2_ptr
      %p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
         %p4 = OpLoad %v4float %p4_ptr

; Load colors of main triangle vertices
      %c0_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
         %c0 = OpLoad %v3float %c0_ptr
      %c2_ptr = OpAccessChain %ptr_in_v3 %in_color %int_2
         %c2 = OpLoad %v3float %c2_ptr
      %c4_ptr = OpAccessChain %ptr_in_v3 %in_color %int_4
         %c4 = OpLoad %v3float %c4_ptr

; Edge 0->2
       %e02 = OpFSub %v4float %p2 %p0
    %e02_xy = OpVectorShuffle %v3float %e02 %e02 0 1 3
   %perp02_x = OpCompositeExtract %float %e02_xy 1
   %perp02_y = OpCompositeExtract %float %e02_xy 0
 %perp02_y_n = OpFNegate %float %perp02_y
    %perp02a = OpCompositeConstruct %v3float %perp02_x %perp02_y_n %float_1
    %perp02b = OpDot %float %perp02a %perp02a
    %perp02c = OpExtInst %float %1 Sqrt %perp02b
    %perp02d = OpFDiv %float %float_008 %perp02c
    %perp02e = OpVectorTimesScalar %v3float %perp02a %perp02d
    %perp02f = OpCompositeExtract %float %perp02e 0
    %perp02g = OpCompositeExtract %float %perp02e 1
    %perp02h = OpCompositeConstruct %v4float %perp02f %perp02g %float_1 %float_1
     %off02 = OpVectorTimesScalar %v4float %perp02h %float_1
    %col02a = OpFAdd %v3float %c0 %c2
     %col02 = OpVectorTimesScalar %v3float %col02a %float_05

      %v02_0 = OpFSub %v4float %p0 %off02
      %v02_1 = OpFAdd %v4float %p0 %off02
      %v02_2 = OpFSub %v4float %p2 %off02
      %v02_3 = OpFAdd %v4float %p2 %off02

 %gl_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %gl_out_ptr %v02_0
               OpStore %out_color %col02
               OpEmitVertex
               OpStore %gl_out_ptr %v02_1
               OpStore %out_color %col02
               OpEmitVertex
               OpStore %gl_out_ptr %v02_2
               OpStore %out_color %col02
               OpEmitVertex
               OpStore %gl_out_ptr %v02_3
               OpStore %out_color %col02
               OpEmitVertex
               OpEndPrimitive

; Edge 2->4
       %e24 = OpFSub %v4float %p4 %p2
    %e24_xy = OpVectorShuffle %v3float %e24 %e24 0 1 3
   %perp24_x = OpCompositeExtract %float %e24_xy 1
   %perp24_y = OpCompositeExtract %float %e24_xy 0
 %perp24_y_n = OpFNegate %float %perp24_y
    %perp24a = OpCompositeConstruct %v3float %perp24_x %perp24_y_n %float_1
    %perp24b = OpDot %float %perp24a %perp24a
    %perp24c = OpExtInst %float %1 Sqrt %perp24b
    %perp24d = OpFDiv %float %float_008 %perp24c
    %perp24e = OpVectorTimesScalar %v3float %perp24a %perp24d
    %perp24f = OpCompositeExtract %float %perp24e 0
    %perp24g = OpCompositeExtract %float %perp24e 1
    %perp24h = OpCompositeConstruct %v4float %perp24f %perp24g %float_1 %float_1
     %off24 = OpVectorTimesScalar %v4float %perp24h %float_1
    %col24a = OpFAdd %v3float %c2 %c4
     %col24 = OpVectorTimesScalar %v3float %col24a %float_05

      %v24_0 = OpFSub %v4float %p2 %off24
      %v24_1 = OpFAdd %v4float %p2 %off24
      %v24_2 = OpFSub %v4float %p4 %off24
      %v24_3 = OpFAdd %v4float %p4 %off24

               OpStore %gl_out_ptr %v24_0
               OpStore %out_color %col24
               OpEmitVertex
               OpStore %gl_out_ptr %v24_1
               OpStore %out_color %col24
               OpEmitVertex
               OpStore %gl_out_ptr %v24_2
               OpStore %out_color %col24
               OpEmitVertex
               OpStore %gl_out_ptr %v24_3
               OpStore %out_color %col24
               OpEmitVertex
               OpEndPrimitive

; Edge 4->0
       %e40 = OpFSub %v4float %p0 %p4
    %e40_xy = OpVectorShuffle %v3float %e40 %e40 0 1 3
   %perp40_x = OpCompositeExtract %float %e40_xy 1
   %perp40_y = OpCompositeExtract %float %e40_xy 0
 %perp40_y_n = OpFNegate %float %perp40_y
    %perp40a = OpCompositeConstruct %v3float %perp40_x %perp40_y_n %float_1
    %perp40b = OpDot %float %perp40a %perp40a
    %perp40c = OpExtInst %float %1 Sqrt %perp40b
    %perp40d = OpFDiv %float %float_008 %perp40c
    %perp40e = OpVectorTimesScalar %v3float %perp40a %perp40d
    %perp40f = OpCompositeExtract %float %perp40e 0
    %perp40g = OpCompositeExtract %float %perp40e 1
    %perp40h = OpCompositeConstruct %v4float %perp40f %perp40g %float_1 %float_1
     %off40 = OpVectorTimesScalar %v4float %perp40h %float_1
    %col40a = OpFAdd %v3float %c4 %c0
     %col40 = OpVectorTimesScalar %v3float %col40a %float_05

      %v40_0 = OpFSub %v4float %p4 %off40
      %v40_1 = OpFAdd %v4float %p4 %off40
      %v40_2 = OpFSub %v4float %p0 %off40
      %v40_3 = OpFAdd %v4float %p0 %off40

               OpStore %gl_out_ptr %v40_0
               OpStore %out_color %col40
               OpEmitVertex
               OpStore %gl_out_ptr %v40_1
               OpStore %out_color %col40
               OpEmitVertex
               OpStore %gl_out_ptr %v40_2
               OpStore %out_color %col40
               OpEmitVertex
               OpStore %gl_out_ptr %v40_3
               OpStore %out_color %col40
               OpEmitVertex
               OpEndPrimitive

               OpReturn
               OpFunctionEnd
