#include <iostream>
#include <vector>
#include <algorithm>
#include <tuple>
#include <cstdint>

using namespace std;

struct Rect {
    int w, h, id;
};

struct FreeRect {
    int x, y, w, h;
};

struct Placement {
    int board, x, y, rotated;
};

class BinPacker {
private:
    int board_w, board_h;
    vector<Rect> rects;
    vector<Placement> placements;
    vector<vector<FreeRect>> boards;

    bool canFit(const FreeRect& fr, int w, int h) {
        return w <= fr.w && h <= fr.h;
    }

    void splitFreeRect(vector<FreeRect>& freeRects, int idx, int x, int y, int w, int h) {
        FreeRect fr = freeRects[idx];
        freeRects.erase(freeRects.begin() + idx);

        // Create new free rectangles from the split
        // Right side
        if (fr.x + fr.w > x + w) {
            freeRects.push_back({x + w, fr.y, fr.x + fr.w - (x + w), fr.h});
        }
        // Top side
        if (fr.y + fr.h > y + h) {
            freeRects.push_back({fr.x, y + h, fr.w, fr.y + fr.h - (y + h)});
        }
        // Left side (if placement doesn't start at free rect's left edge)
        if (x > fr.x) {
            freeRects.push_back({fr.x, fr.y, x - fr.x, fr.h});
        }
        // Bottom side (if placement doesn't start at free rect's bottom edge)
        if (y > fr.y) {
            freeRects.push_back({fr.x, fr.y, fr.w, y - fr.y});
        }

        // Remove redundant and contained rectangles
        pruneFreeRects(freeRects);
    }

    void pruneFreeRects(vector<FreeRect>& freeRects) {
        // Remove rectangles that are completely contained in others
        for (int i = 0; i < (int)freeRects.size(); i++) {
            for (int j = i + 1; j < (int)freeRects.size(); ) {
                if (contains(freeRects[i], freeRects[j])) {
                    freeRects.erase(freeRects.begin() + j);
                } else if (contains(freeRects[j], freeRects[i])) {
                    freeRects.erase(freeRects.begin() + i);
                    i--;
                    break;
                } else {
                    j++;
                }
            }
        }
    }

    bool contains(const FreeRect& a, const FreeRect& b) {
        return a.x <= b.x && a.y <= b.y &&
               a.x + a.w >= b.x + b.w && a.y + a.h >= b.y + b.h;
    }

    bool tryPlace(int boardIdx, const Rect& rect, bool rotated) {
        int w = rotated ? rect.h : rect.w;
        int h = rotated ? rect.w : rect.h;

        if (w > board_w || h > board_h) return false;

        auto& freeRects = boards[boardIdx];

        // Find best free rectangle using Bottom-Left heuristic
        int bestIdx = -1;
        int bestY = INT32_MAX;
        int bestX = INT32_MAX;

        for (int i = 0; i < (int)freeRects.size(); i++) {
            if (canFit(freeRects[i], w, h)) {
                if (freeRects[i].y < bestY ||
                    (freeRects[i].y == bestY && freeRects[i].x < bestX)) {
                    bestIdx = i;
                    bestY = freeRects[i].y;
                    bestX = freeRects[i].x;
                }
            }
        }

        if (bestIdx == -1) return false;

        int px = freeRects[bestIdx].x;
        int py = freeRects[bestIdx].y;

        placements[rect.id] = {boardIdx, px, py, rotated ? 1 : 0};
        splitFreeRect(freeRects, bestIdx, px, py, w, h);

        return true;
    }

    void addBoard() {
        boards.push_back({{0, 0, board_w, board_h}});
    }

public:
    BinPacker(int bw, int bh) : board_w(bw), board_h(bh) {}

    void solve(vector<Rect>& input_rects) {
        rects = input_rects;
        int n = rects.size();
        placements.resize(n);

        // Sort rectangles by area (largest first)
        sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
            int area_a = a.w * a.h;
            int area_b = b.w * b.h;
            if (area_a != area_b) return area_a > area_b;
            return max(a.w, a.h) > max(b.w, b.h);
        });

        // Pack rectangles
        for (const auto& rect : rects) {
            bool placed = false;

            // Try to place in existing boards
            for (int b = 0; b < (int)boards.size() && !placed; b++) {
                // Try without rotation
                if (tryPlace(b, rect, false)) {
                    placed = true;
                    break;
                }
                // Try with rotation
                if (rect.w != rect.h && tryPlace(b, rect, true)) {
                    placed = true;
                    break;
                }
            }

            // If not placed, create new board
            if (!placed) {
                addBoard();
                int b = boards.size() - 1;

                if (!tryPlace(b, rect, false)) {
                    // Try rotation if normal doesn't work
                    if (rect.w != rect.h && !tryPlace(b, rect, true)) {
                        // Should not happen if rectangles fit in board
                        cerr << "Error: Cannot place rectangle " << rect.id << endl;
                    }
                }
            }
        }
    }

    void output() {
        cout << boards.size() << "\n";
        for (const auto& p : placements) {
            cout << p.board << " " << p.x << " " << p.y << " " << p.rotated << "\n";
        }
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, board_w, board_h;
    cin >> n >> board_w >> board_h;

    vector<Rect> rects(n);
    for (int i = 0; i < n; i++) {
        cin >> rects[i].w >> rects[i].h;
        rects[i].id = i;
    }

    BinPacker packer(board_w, board_h);
    packer.solve(rects);
    packer.output();

    return 0;
}
