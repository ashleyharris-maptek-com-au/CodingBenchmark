OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos_in Location 0
               OpDecorate %normal_in Location 1
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
     %uint_4 = OpConstant %uint 4

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
     %gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
%normal_in = OpVariable %ptr_in_arr_v3 Input
%color_in = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

%float_0_025 = OpConstant %float 0.025
  %float_n1 = OpConstant %float -1.0

       %main = OpFunction %void None %3
      %label = OpLabel

; Load all 4 input positions
   %pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
       %pos0 = OpLoad %v4float %pos0_ptr
   %pos1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
       %pos1 = OpLoad %v4float %pos1_ptr
   %pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
       %pos2 = OpLoad %v4float %pos2_ptr
   %pos3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
       %pos3 = OpLoad %v4float %pos3_ptr

; Load colors for the actual line segment (indices 1 and 2)
  %color1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
      %color1 = OpLoad %v3float %color1_ptr
  %color2_ptr = OpAccessChain %ptr_in_v3 %color_in %int_2
      %color2 = OpLoad %v3float %color2_ptr

; Extract xy components for tangent computation
    %pos0_xy = OpVectorShuffle %v2float %pos0 %pos0 0 1
    %pos1_xy = OpVectorShuffle %v2float %pos1 %pos1 0 1
    %pos2_xy = OpVectorShuffle %v2float %pos2 %pos2 0 1
    %pos3_xy = OpVectorShuffle %v2float %pos3 %pos3 0 1

; Compute tangent at start: t0 = normalize(pos2 - pos0)
   %diff_t0 = OpFSub %v2float %pos2_xy %pos0_xy
        %t0 = OpExtInst %v2float %1 Normalize %diff_t0

; Compute tangent at end: t1 = normalize(pos3 - pos1)
   %diff_t1 = OpFSub %v2float %pos3_xy %pos1_xy
        %t1 = OpExtInst %v2float %1 Normalize %diff_t1

; Compute perpendicular at start: perp0 = vec2(-t0.y, t0.x)
      %t0_x = OpCompositeExtract %float %t0 0
      %t0_y = OpCompositeExtract %float %t0 1
  %neg_t0_y = OpFMul %float %t0_y %float_n1
     %perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x

; Compute perpendicular at end: perp1 = vec2(-t1.y, t1.x)
      %t1_x = OpCompositeExtract %float %t1 0
      %t1_y = OpCompositeExtract %float %t1 1
  %neg_t1_y = OpFMul %float %t1_y %float_n1
     %perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x

; Scale perpendiculars by thickness
 %perp0_thick = OpVectorTimesScalar %v2float %perp0 %float_0_025
 %perp1_thick = OpVectorTimesScalar %v2float %perp1 %float_0_025

; Extract z and w components from pos1 and pos2
     %pos1_z = OpCompositeExtract %float %pos1 2
     %pos1_w = OpCompositeExtract %float %pos1 3
     %pos2_z = OpCompositeExtract %float %pos2 2
     %pos2_w = OpCompositeExtract %float %pos2 3

; Vertex 1: start + perp0*thickness
     %v1_xy = OpFAdd %v2float %pos1_xy %perp0_thick
     %v1_x = OpCompositeExtract %float %v1_xy 0
     %v1_y = OpCompositeExtract %float %v1_xy 1
    %v1_pos = OpCompositeConstruct %v4float %v1_x %v1_y %pos1_z %pos1_w
%gl_pos_ptr1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %gl_pos_ptr1 %v1_pos
               OpStore %outColor %color1
               OpEmitVertex

; Vertex 2: start - perp0*thickness
     %v2_xy = OpFSub %v2float %pos1_xy %perp0_thick
     %v2_x = OpCompositeExtract %float %v2_xy 0
     %v2_y = OpCompositeExtract %float %v2_xy 1
    %v2_pos = OpCompositeConstruct %v4float %v2_x %v2_y %pos1_z %pos1_w
%gl_pos_ptr2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %gl_pos_ptr2 %v2_pos
               OpStore %outColor %color1
               OpEmitVertex

; Vertex 3: end + perp1*thickness
     %v3_xy = OpFAdd %v2float %pos2_xy %perp1_thick
     %v3_x = OpCompositeExtract %float %v3_xy 0
     %v3_y = OpCompositeExtract %float %v3_xy 1
    %v3_pos = OpCompositeConstruct %v4float %v3_x %v3_y %pos2_z %pos2_w
%gl_pos_ptr3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %gl_pos_ptr3 %v3_pos
               OpStore %outColor %color2
               OpEmitVertex

; Vertex 4: end - perp1*thickness
     %v4_xy = OpFSub %v2float %pos2_xy %perp1_thick
     %v4_x = OpCompositeExtract %float %v4_xy 0
     %v4_y = OpCompositeExtract %float %v4_xy 1
    %v4_pos = OpCompositeConstruct %v4float %v4_x %v4_y %pos2_z %pos2_w
%gl_pos_ptr4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %gl_pos_ptr4 %v4_pos
               OpStore %outColor %color2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
