OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 8
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos_in Location 0
               OpDecorate %normal_in Location 1
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
     %uint_4 = OpConstant %uint 4
     %uint_1 = OpConstant %uint 1
    %float_0 = OpConstant %float 0
  %float_0_5 = OpConstant %float 0.5
    %float_1 = OpConstant %float 1
    %float_2 = OpConstant %float 2
    %float_3 = OpConstant %float 3
 %float_0_02 = OpConstant %float 0.02
%float_third = OpConstant %float 0.333333343

          %1 = OpExtInstImport "GLSL.std.450"

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
 %normal_in = OpVariable %ptr_in_arr_v3 Input
  %color_in = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %3
      %label = OpLabel

    ; Load positions
    %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
        %p0 = OpLoad %v4float %p0_ptr
    %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
        %p1 = OpLoad %v4float %p1_ptr
    %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
        %p2 = OpLoad %v4float %p2_ptr
    %p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
        %p3 = OpLoad %v4float %p3_ptr

    ; Load colors
    %c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
        %c1 = OpLoad %v3float %c1_ptr
    %c2_ptr = OpAccessChain %ptr_in_v3 %color_in %int_2
        %c2 = OpLoad %v3float %c2_ptr

    ; Extract xy components for 2D calculations
      %p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
      %p1xy = OpVectorShuffle %v2float %p1 %p1 0 1
      %p2xy = OpVectorShuffle %v2float %p2 %p2 0 1
      %p3xy = OpVectorShuffle %v2float %p3 %p3 0 1

    ; Compute tangent at p1 (average of p0->p1 and p1->p2)
    %t1_a = OpFSub %v2float %p1xy %p0xy
    %t1_b = OpFSub %v2float %p2xy %p1xy
    %t1_sum = OpFAdd %v2float %t1_a %t1_b
       %t1 = OpVectorTimesScalar %v2float %t1_sum %float_0_5

    ; Compute tangent at p2 (average of p1->p2 and p2->p3)
    %t2_a = OpFSub %v2float %p2xy %p1xy
    %t2_b = OpFSub %v2float %p3xy %p2xy
    %t2_sum = OpFAdd %v2float %t2_a %t2_b
       %t2 = OpVectorTimesScalar %v2float %t2_sum %float_0_5

    ; Normalize tangents
     %t1_x = OpCompositeExtract %float %t1 0
     %t1_y = OpCompositeExtract %float %t1 1
 %t1_x_sq = OpFMul %float %t1_x %t1_x
 %t1_y_sq = OpFMul %float %t1_y %t1_y
 %t1_len_sq = OpFAdd %float %t1_x_sq %t1_y_sq
    %t1_len = OpExtInst %float %1 Sqrt %t1_len_sq
  %t1_len_r = OpFDiv %float %float_1 %t1_len
      %t1_n = OpVectorTimesScalar %v2float %t1 %t1_len_r

     %t2_x = OpCompositeExtract %float %t2 0
     %t2_y = OpCompositeExtract %float %t2 1
 %t2_x_sq = OpFMul %float %t2_x %t2_x
 %t2_y_sq = OpFMul %float %t2_y %t2_y
 %t2_len_sq = OpFAdd %float %t2_x_sq %t2_y_sq
    %t2_len = OpExtInst %float %1 Sqrt %t2_len_sq
  %t2_len_r = OpFDiv %float %float_1 %t2_len
      %t2_n = OpVectorTimesScalar %v2float %t2 %t2_len_r

    ; Compute perpendiculars (rotate tangent by 90 degrees)
   %t1n_x = OpCompositeExtract %float %t1_n 0
   %t1n_y = OpCompositeExtract %float %t1_n 1
  %perp1_x = OpFNegate %float %t1n_y
  %perp1_y = %t1n_x
     %perp1 = OpCompositeConstruct %v2float %perp1_x %perp1_y

   %t2n_x = OpCompositeExtract %float %t2_n 0
   %t2n_y = OpCompositeExtract %float %t2_n 1
  %perp2_x = OpFNegate %float %t2n_y
  %perp2_y = %t2n_x
     %perp2 = OpCompositeConstruct %v2float %perp2_x %perp2_y

    ; Scale perpendiculars by thickness
    %perp1_t = OpVectorTimesScalar %v2float %perp1 %float_0_02
    %perp2_t = OpVectorTimesScalar %v2float %perp2 %float_0_02

    ; Compute segment positions at 1/3 and 2/3
    %delta = OpFSub %v2float %p2xy %p1xy
  %delta_3 = OpVectorTimesScalar %v2float %delta %float_third
   %p_1_3 = OpFAdd %v2float %p1xy %delta_3
  %delta_2 = OpVectorTimesScalar %v2float %delta_3 %float_2
   %p_2_3 = OpFAdd %v2float %p1xy %delta_2

    ; Interpolate perpendiculars at 1/3 and 2/3
  %perp_diff = OpFSub %v2float %perp2_t %perp1_t
%perp_diff_3 = OpVectorTimesScalar %v2float %perp_diff %float_third
  %perp_1_3 = OpFAdd %v2float %perp1_t %perp_diff_3
%perp_diff_2 = OpVectorTimesScalar %v2float %perp_diff_3 %float_2
  %perp_2_3 = OpFAdd %v2float %perp1_t %perp_diff_2

    ; Interpolate colors at 1/3 and 2/3
%color_diff = OpFSub %v3float %c2 %c1
%color_diff_3 = OpVectorTimesScalar %v3float %color_diff %float_third
  %col_1_3 = OpFAdd %v3float %c1 %color_diff_3
%color_diff_2 = OpVectorTimesScalar %v3float %color_diff_3 %float_2
  %col_2_3 = OpFAdd %v3float %c1 %color_diff_2

    ; First quad (0 to 1/3)
    ; Vertex 0: p1 - perp1
   %q1_v0_xy = OpFSub %v2float %p1xy %perp1_t
 %q1_v0_xy_x = OpCompositeExtract %float %q1_v0_xy 0
 %q1_v0_xy_y = OpCompositeExtract %float %q1_v0_xy 1
     %q1_v0 = OpCompositeConstruct %v4float %q1_v0_xy_x %q1_v0_xy_y %float_0 %float_1
  %q1_v0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %q1_v0_ptr %q1_v0
               OpStore %outColor %c1
               OpEmitVertex

    ; Vertex 1: p1 + perp1
   %q1_v1_xy = OpFAdd %v2float %p1xy %perp1_t
 %q1_v1_xy_x = OpCompositeExtract %float %q1_v1_xy 0
 %q1_v1_xy_y = OpCompositeExtract %float %q1_v1_xy 1
     %q1_v1 = OpCompositeConstruct %v4float %q1_v1_xy_x %q1_v1_xy_y %float_0 %float_1
  %q1_v1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %q1_v1_ptr %q1_v1
               OpStore %outColor %c1
               OpEmitVertex

    ; Vertex 2: p_1_3 - perp_1_3
   %q1_v2_xy = OpFSub %v2float %p_1_3 %perp_1_3
 %q1_v2_xy_x = OpCompositeExtract %float %q1_v2_xy 0
 %q1_v2_xy_y = OpCompositeExtract %float %q1_v2_xy 1
     %q1_v2 = OpCompositeConstruct %v4float %q1_v2_xy_x %q1_v2_xy_y %float_0 %float_1
  %q1_v2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %q1_v2_ptr %q1_v2
               OpStore %outColor %col_1_3
               OpEmitVertex

    ; Vertex 3: p_1_3 + perp_1_3
   %q1_v3_xy = OpFAdd %v2float %p_1_3 %perp_1_3
 %q1_v3_xy_x = OpCompositeExtract %float %q1_v3_xy 0
 %q1_v3_xy_y = OpCompositeExtract %float %q1_v3_xy 1
     %q1_v3 = OpCompositeConstruct %v4float %q1_v3_xy_x %q1_v3_xy_y %float_0 %float_1
  %q1_v3_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %q1_v3_ptr %q1_v3
               OpStore %outColor %col_1_3
               OpEmitVertex

               OpEndPrimitive

    ; Second quad (2/3 to 1)
    ; Vertex 0: p_2_3 - perp_2_3
   %q2_v0_xy = OpFSub %v2float %p_2_3 %perp_2_3
 %q2_v0_xy_x = OpCompositeExtract %float %q2_v0_xy 0
 %q2_v0_xy_y = OpCompositeExtract %float %q2_v0_xy 1
     %q2_v0 = OpCompositeConstruct %v4float %q2_v0_xy_x %q2_v0_xy_y %float_0 %float_1
  %q2_v0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %q2_v0_ptr %q2_v0
               OpStore %outColor %col_2_3
               OpEmitVertex

    ; Vertex 1: p_2_3 + perp_2_3
   %q2_v1_xy = OpFAdd %v2float %p_2_3 %perp_2_3
 %q2_v1_xy_x = OpCompositeExtract %float %q2_v1_xy 0
 %q2_v1_xy_y = OpCompositeExtract %float %q2_v1_xy 1
     %q2_v1 = OpCompositeConstruct %v4float %q2_v1_xy_x %q2_v1_xy_y %float_0 %float_1
  %q2_v1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %q2_v1_ptr %q2_v1
               OpStore %outColor %col_2_3
               OpEmitVertex

    ; Vertex 2: p2 - perp2
   %q2_v2_xy = OpFSub %v2float %p2xy %perp2_t
 %q2_v2_xy_x = OpCompositeExtract %float %q2_v2_xy 0
 %q2_v2_xy_y = OpCompositeExtract %float %q2_v2_xy 1
     %q2_v2 = OpCompositeConstruct %v4float %q2_v2_xy_x %q2_v2_xy_y %float_0 %float_1
  %q2_v2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %q2_v2_ptr %q2_v2
               OpStore %outColor %c2
               OpEmitVertex

    ; Vertex 3: p2 + perp2
   %q2_v3_xy = OpFAdd %v2float %p2xy %perp2_t
 %q2_v3_xy_x = OpCompositeExtract %float %q2_v3_xy 0
 %q2_v3_xy_y = OpCompositeExtract %float %q2_v3_xy 1
     %q2_v3 = OpCompositeConstruct %v4float %q2_v3_xy_x %q2_v3_xy_y %float_0 %float_1
  %q2_v3_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %q2_v3_ptr %q2_v3
               OpStore %outColor %c2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
