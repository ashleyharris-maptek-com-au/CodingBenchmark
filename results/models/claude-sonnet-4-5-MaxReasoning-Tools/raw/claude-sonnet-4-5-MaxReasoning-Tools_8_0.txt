use std::collections::VecDeque;
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read dimensions
    let first_line = lines.next().unwrap().unwrap();
    let dims: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let (h, w) = (dims[0], dims[1]);

    // Read maze
    let mut maze = Vec::with_capacity(h);
    let mut start = (0, 0);
    let mut end = (0, 0);

    for y in 0..h {
        let line = lines.next().unwrap().unwrap();
        let row: Vec<char> = line.chars().collect();
        for x in 0..row.len() {
            if row[x] == 'A' {
                start = (y, x);
            } else if row[x] == 'B' {
                end = (y, x);
            }
        }
        maze.push(row);
    }

    // BFS to find shortest path
    let path = bfs(&maze, h, w, start, end);

    if let Some(p) = path {
        println!("{}", p.len());
        for (y, x) in p {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}

fn bfs(maze: &[Vec<char>], h: usize, w: usize, start: (usize, usize), end: (usize, usize)) -> Option<Vec<(usize, usize)>> {
    let mut visited = vec![vec![false; w]; h];
    let mut parent = vec![vec![None; w]; h];
    let mut queue = VecDeque::new();

    queue.push_back(start);
    visited[start.0][start.1] = true;

    let directions = [(0, 1), (1, 0), (0, -1), (-1, 0)];

    while let Some((y, x)) = queue.pop_front() {
        if (y, x) == end {
            // Reconstruct path
            let mut path = Vec::new();
            let mut current = end;

            while let Some(prev) = parent[current.0][current.1] {
                path.push(current);
                current = prev;
            }
            path.push(start);
            path.reverse();

            return Some(path);
        }

        for &(dy, dx) in &directions {
            let ny = y as isize + dy;
            let nx = x as isize + dx;

            if ny >= 0 && ny < h as isize && nx >= 0 && nx < w as isize {
                let ny = ny as usize;
                let nx = nx as usize;

                if !visited[ny][nx] && maze[ny][nx] != '#' {
                    visited[ny][nx] = true;
                    parent[ny][nx] = Some((y, x));
                    queue.push_back((ny, nx));
                }
            }
        }
    }

    None
}
