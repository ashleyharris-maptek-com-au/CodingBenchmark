#include <iostream>
#include <vector>
#include <algorithm>
#include <set>
#include <queue>
#include <chrono>
#include <random>

using namespace std;

class GraphColoring {
private:
    int n, m, k;
    vector<vector<int>> adj;
    vector<int> colors;
    chrono::steady_clock::time_point start_time;
    const int TIME_LIMIT_MS = 280000; // 4:40 safety margin

    bool timeExpired() {
        auto now = chrono::steady_clock::now();
        auto elapsed = chrono::duration_cast<chrono::milliseconds>(now - start_time).count();
        return elapsed > TIME_LIMIT_MS;
    }

    int getSaturation(int v, const vector<set<int>>& neighbor_colors) {
        return neighbor_colors[v].size();
    }

    int getDegree(int v) {
        return adj[v].size();
    }

    // DSATUR algorithm - greedy approach
    bool colorDSATUR() {
        colors.assign(n, -1);
        vector<bool> colored(n, false);
        vector<set<int>> neighbor_colors(n);

        // Find vertex with maximum degree to start
        int max_deg = -1, first_v = 0;
        for (int i = 0; i < n; i++) {
            if ((int)adj[i].size() > max_deg) {
                max_deg = adj[i].size();
                first_v = i;
            }
        }

        colors[first_v] = 0;
        colored[first_v] = true;
        for (int u : adj[first_v]) {
            neighbor_colors[u].insert(0);
        }

        for (int iter = 1; iter < n; iter++) {
            if (timeExpired()) return false;

            // Find uncolored vertex with max saturation (then max degree)
            int best_v = -1;
            int best_sat = -1;
            int best_deg = -1;

            for (int v = 0; v < n; v++) {
                if (!colored[v]) {
                    int sat = getSaturation(v, neighbor_colors);
                    int deg = getDegree(v);

                    if (best_v == -1 || sat > best_sat ||
                        (sat == best_sat && deg > best_deg)) {
                        best_v = v;
                        best_sat = sat;
                        best_deg = deg;
                    }
                }
            }

            if (best_v == -1) break;

            // Assign smallest available color
            int color = 0;
            while (color < k && neighbor_colors[best_v].count(color)) {
                color++;
            }

            if (color >= k) {
                return false; // Can't color with k colors
            }

            colors[best_v] = color;
            colored[best_v] = true;

            for (int u : adj[best_v]) {
                if (!colored[u]) {
                    neighbor_colors[u].insert(color);
                }
            }
        }

        return true;
    }

    // Backtracking with forward checking for smaller graphs
    bool backtrack(int v, vector<set<int>>& available) {
        if (timeExpired()) return false;

        if (v == n) return true;

        vector<int> colors_to_try;
        for (int c : available[v]) {
            colors_to_try.push_back(c);
        }

        for (int c : colors_to_try) {
            colors[v] = c;

            // Forward checking
            vector<set<int>> saved_available(n);
            bool valid = true;

            for (int u : adj[v]) {
                if (colors[u] == -1) {
                    saved_available[u] = available[u];
                    available[u].erase(c);
                    if (available[u].empty()) {
                        valid = false;
                        break;
                    }
                }
            }

            if (valid && backtrack(v + 1, available)) {
                return true;
            }

            // Restore
            for (int u : adj[v]) {
                if (colors[u] == -1 && !saved_available[u].empty()) {
                    available[u] = saved_available[u];
                }
            }

            colors[v] = -1;
        }

        return false;
    }

    bool colorBacktrack() {
        colors.assign(n, -1);
        vector<set<int>> available(n);

        for (int i = 0; i < n; i++) {
            for (int c = 0; c < k; c++) {
                available[i].insert(c);
            }
        }

        return backtrack(0, available);
    }

    // Greedy coloring with vertex ordering
    bool colorGreedy(vector<int>& order) {
        colors.assign(n, -1);

        for (int v : order) {
            if (timeExpired()) return false;

            vector<bool> used(k, false);
            for (int u : adj[v]) {
                if (colors[u] != -1) {
                    used[colors[u]] = true;
                }
            }

            int c = 0;
            while (c < k && used[c]) c++;

            if (c >= k) return false;
            colors[v] = c;
        }

        return true;
    }

    bool isValidColoring() {
        for (int v = 0; v < n; v++) {
            if (colors[v] < 0 || colors[v] >= k) return false;
            for (int u : adj[v]) {
                if (colors[v] == colors[u]) return false;
            }
        }
        return true;
    }

public:
    GraphColoring(int vertices, int edges, int num_colors)
        : n(vertices), m(edges), k(num_colors) {
        adj.resize(n);
        start_time = chrono::steady_clock::now();
    }

    void addEdge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    bool solve() {
        // Adaptive strategy selection
        bool success = false;

        if (n <= 100 && m <= 500) {
            // Small graph: try backtracking
            success = colorBacktrack();
            if (!success) success = colorDSATUR();
        } else if (n <= 10000) {
            // Medium graph: DSATUR
            success = colorDSATUR();

            // If failed, try multiple greedy orderings
            if (!success) {
                vector<pair<int, int>> degree_order;
                for (int i = 0; i < n; i++) {
                    degree_order.push_back({adj[i].size(), i});
                }
                sort(degree_order.rbegin(), degree_order.rend());

                vector<int> order;
                for (auto& p : degree_order) {
                    order.push_back(p.second);
                }

                success = colorGreedy(order);
            }
        } else {
            // Large graph: DSATUR only
            success = colorDSATUR();
        }

        if (success && !isValidColoring()) {
            return false;
        }

        return success;
    }

    void printSolution() {
        for (int i = 0; i < n; i++) {
            if (i > 0) cout << " ";
            cout << colors[i];
        }
        cout << "\n";
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    cin >> n >> m >> k;

    GraphColoring gc(n, m, k);

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        gc.addEdge(u, v);
    }

    if (gc.solve()) {
        gc.printSolution();
    } else {
        // Failed to find coloring - output something
        for (int i = 0; i < n; i++) {
            if (i > 0) cout << " ";
            cout << (i % k);
        }
        cout << "\n";
    }

    return 0;
}
