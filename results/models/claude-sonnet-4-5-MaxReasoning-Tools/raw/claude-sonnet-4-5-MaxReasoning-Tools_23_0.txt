use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

struct Config {
    gravity: f64,
    max_thrust: f64,
    target_x: f64,
    target_y: f64,
}

fn main() {
    // Read initial configuration from stdin
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // First line: width height gravity max_thrust fuel
    let first_line = lines.next().unwrap().unwrap();
    let params: Vec<f64> = first_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    let height = params[1] as usize;
    let gravity = params[2];
    let max_thrust = params[3];
    let initial_fuel = params[4];

    // Second line: start_x start_y target_x target_y
    let second_line = lines.next().unwrap().unwrap();
    let positions: Vec<f64> = second_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    let start_x = positions[0];
    let start_y = positions[1];
    let target_x = positions[2];
    let target_y = positions[3];

    let config = Config {
        gravity,
        max_thrust,
        target_x,
        target_y,
    };

    // Read and discard map (don't store multi-GB maps in memory)
    for _ in 0..height {
        lines.next();
    }

    // Shared state between threads
    let current_state = Arc::new(Mutex::new(State {
        x: start_x,
        y: start_y,
        vx: 0.0,
        vy: 0.0,
        angle: 0.0,
        angular_vel: 0.0,
        fuel: initial_fuel,
    }));

    // Clone for reader thread
    let state_reader = current_state.clone();

    // Spawn background thread to continuously read state updates
    thread::spawn(move || {
        for line in lines {
            if let Ok(line_str) = line {
                let values: Vec<f64> = line_str
                    .split_whitespace()
                    .filter_map(|s| s.parse().ok())
                    .collect();

                if values.len() >= 7 {
                    if let Ok(mut state) = state_reader.lock() {
                        state.x = values[0];
                        state.y = values[1];
                        state.vx = values[2];
                        state.vy = values[3];
                        state.angle = values[4];
                        state.angular_vel = values[5];
                        state.fuel = values[6];
                    }
                }
            }
        }
    });

    // Main control loop - output at 50Hz (20ms intervals, exceeds 20Hz requirement)
    let stdout = io::stdout();
    let mut output = stdout.lock();
    let output_interval = Duration::from_millis(20);

    loop {
        let loop_start = Instant::now();

        // Get current state (quick lock)
        let state = {
            let s = current_state.lock().unwrap();
            s.clone()
        };

        // Compute control commands
        let (thrust, turn) = compute_control(&state, &config);

        // Output to stdout
        writeln!(output, "{:.6} {:.6}", thrust, turn).unwrap();
        output.flush().unwrap();

        // Maintain 50Hz output rate
        let elapsed = loop_start.elapsed();
        if elapsed < output_interval {
            thread::sleep(output_interval - elapsed);
        }
    }
}

fn compute_control(state: &State, config: &Config) -> (f64, f64) {
    // Calculate position and velocity relative to target
    let dx = config.target_x - state.x;
    let dy = config.target_y - state.y;
    let distance = (dx * dx + dy * dy).sqrt();

    // ===== TURN CONTROL =====
    // Desired angle: mostly upward (0), but tilted toward target horizontally
    let desired_angle = if distance > 10.0 {
        // Far from target: tilt more to move horizontally
        (dx / 15.0).clamp(-0.4, 0.4)
    } else if distance > 3.0 {
        // Medium distance: reduce tilt
        (dx / 20.0).clamp(-0.2, 0.2)
    } else {
        // Close to target: point straight up for vertical landing
        (dx / 30.0).clamp(-0.1, 0.1)
    };

    // PD controller for angle
    let angle_error = desired_angle - state.angle;
    let turn = (angle_error * 4.0 - state.angular_vel * 1.0).clamp(-1.0, 1.0);

    // ===== THRUST CONTROL =====
    if state.fuel <= 0.05 {
        // Out of fuel
        return (0.0, turn);
    }

    // Hover thrust needed to counteract gravity (when pointing straight up)
    let hover_thrust = config.gravity / config.max_thrust;

    // Adjust for angle (when tilted, need more thrust for same vertical force)
    let angle_factor = 1.0 / state.angle.cos().max(0.5);
    let adjusted_hover = hover_thrust * angle_factor;

    // Desired vertical velocity based on altitude above target
    let desired_vy = if dy < 0.0 {
        // Below target: need to go up
        -2.0
    } else if dy > 30.0 {
        // Far above: descend faster
        4.0
    } else if dy > 15.0 {
        // Medium distance: moderate descent
        3.0
    } else if dy > 5.0 {
        // Getting close: slow down
        2.0
    } else {
        // Very close: gentle descent for soft landing
        1.0
    };

    // PD controller for vertical velocity
    let vy_error = state.vy - desired_vy;
    let thrust_correction = vy_error * 0.4;

    // Final thrust command
    let thrust = (adjusted_hover + thrust_correction).clamp(0.0, 1.0);

    // Emergency thrust if approaching too fast and close to ground
    let thrust = if dy < 3.0 && state.vy > 2.5 {
        1.0 // Full thrust to slow down
    } else {
        thrust
    };

    (thrust, turn)
}
