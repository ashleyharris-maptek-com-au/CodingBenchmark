import math

# Module-level state persistence across calls within a scenario
state_memory = {
    'phase': 'INTERCEPT',
    'prev_gs_error': 0.0,
    'prev_loc_error': 0.0,
    'prev_ias_error': 0.0,
    'prev_vs_error': 0.0,
    'integral_gs': 0.0,
    'integral_loc': 0.0,
    'integral_ias': 0.0,
    'integral_vs': 0.0,
    'flare_start_alt': None,
    'flare_start_vs': None,
    'touchdown_detected': False,
    'time_unstable': 0.0,
    'time_below_50': 0.0,
    'go_around_initiated': False,
    'single_engine': False,
    'dual_engine_failure': False,
}

def get_valid_sensor(values, validity_flags):
    """Return average of valid sensors, or first value if none valid."""
    valid_values = [v for v, valid in zip(values, validity_flags) if valid]
    if valid_values:
        return sum(valid_values) / len(valid_values)
    return values[0] if values else 0.0

def autopilot_step(state: dict, dt: float) -> dict:
    global state_memory

    # Sensor fusion - use redundant sensors with voting
    gs_dots = get_valid_sensor(
        [state.get('glideslope_1_dots', 0), state.get('glideslope_2_dots', 0), state.get('glideslope_3_dots', 0)],
        [state.get('glideslope_1_valid', False), state.get('glideslope_2_valid', False), state.get('glideslope_3_valid', False)]
    )

    loc_dots = get_valid_sensor(
        [state.get('localizer_1_dots', 0), state.get('localizer_2_dots', 0), state.get('localizer_3_dots', 0)],
        [state.get('localizer_1_valid', False), state.get('localizer_2_valid', False), state.get('localizer_3_valid', False)]
    )

    radio_alt = state.get('radio_alt_ft', 0)
    baro_alt = get_valid_sensor(
        [state.get('baro_alt_1_ft', 0), state.get('baro_alt_2_ft', 0), state.get('baro_alt_3_ft', 0)],
        [state.get('baro_alt_1_valid', False), state.get('baro_alt_2_valid', False), state.get('baro_alt_3_valid', False)]
    )

    ias = get_valid_sensor(
        [state.get('ias_1_kt', 0), state.get('ias_2_kt', 0), state.get('ias_3_kt', 0)],
        [state.get('ias_1_valid', False), state.get('ias_2_valid', False), state.get('ias_3_valid', False)]
    )

    pitch = get_valid_sensor(
        [state.get('irs_1_pitch_deg', 0), state.get('irs_2_pitch_deg', 0), state.get('irs_3_pitch_deg', 0)],
        [state.get('irs_1_valid', False), state.get('irs_2_valid', False), state.get('irs_3_valid', False)]
    )

    roll = get_valid_sensor(
        [state.get('irs_1_roll_deg', 0), state.get('irs_2_roll_deg', 0), state.get('irs_3_roll_deg', 0)],
        [state.get('irs_1_valid', False), state.get('irs_2_valid', False), state.get('irs_3_valid', False)]
    )

    vs = get_valid_sensor(
        [state.get('irs_1_vs_fpm', 0), state.get('irs_2_vs_fpm', 0), state.get('irs_3_vs_fpm', 0)],
        [state.get('irs_1_valid', False), state.get('irs_2_valid', False), state.get('irs_3_valid', False)]
    )

    groundspeed = state.get('groundspeed_kt', ias)
    weight_on_wheels = state.get('weight_on_wheels', False)
    on_ground = state.get('on_ground', False)
    target_vref = state.get('target_vref_kt', 137)

    # Engine status detection
    eng1_running = state.get('engine_1_running', True)
    eng2_running = state.get('engine_2_running', True)
    eng1_thrust = state.get('engine_1_thrust_pct', 0)
    eng2_thrust = state.get('engine_2_thrust_pct', 0)

    if not eng1_running or eng1_thrust < 5:
        state_memory['single_engine'] = True
    if not eng2_running or eng2_thrust < 5:
        state_memory['single_engine'] = True
    if (not eng1_running or eng1_thrust < 5) and (not eng2_running or eng2_thrust < 5):
        state_memory['dual_engine_failure'] = True

    # Use radio alt when available, otherwise use baro
    altitude = radio_alt if radio_alt > 0 else baro_alt

    # Initialize output
    cmd = {
        'gear': 1.0,  # Gear always down for landing
        'flaps': 30,  # Default landing flaps
    }

    # Phase transitions
    if altitude > 1000:
        state_memory['phase'] = 'INTERCEPT'
    elif altitude > 200:
        state_memory['phase'] = 'APPROACH'
    elif altitude > 50 and not weight_on_wheels:
        state_memory['phase'] = 'APPROACH'
    elif altitude <= 50 and not weight_on_wheels:
        state_memory['phase'] = 'FLARE'
        if state_memory['flare_start_alt'] is None:
            state_memory['flare_start_alt'] = altitude
            state_memory['flare_start_vs'] = vs
    elif weight_on_wheels or on_ground:
        state_memory['phase'] = 'ROLLOUT'
        state_memory['touchdown_detected'] = True

    # Stability monitoring for go-around
    unstable = False
    if state_memory['phase'] in ['APPROACH', 'FLARE']:
        if altitude > 200:
            # Check approach stability
            if abs(gs_dots) > 1.5 or abs(loc_dots) > 1.5:
                unstable = True
            if ias < target_vref - 15 or ias > target_vref + 30:
                unstable = True
            if abs(roll) > 8:
                unstable = True

        if unstable:
            state_memory['time_unstable'] += dt
        else:
            state_memory['time_unstable'] = 0.0

        # Go-around if unstable for >3 seconds and above decision height
        if state_memory['time_unstable'] > 3.0 and altitude > 200 and not state_memory['go_around_initiated']:
            state_memory['go_around_initiated'] = True
            state_memory['phase'] = 'GO_AROUND'

    # Handle go-around
    if state_memory['phase'] == 'GO_AROUND' or state_memory['go_around_initiated']:
        cmd.update({
            'throttle_1': 1.0 if eng1_running else 0.0,
            'throttle_2': 1.0 if eng2_running else 0.0,
            'elevator': 0.3,
            'speedbrake': 0.0,
            'flaps': 20,
        })
        if altitude > 1500:
            # Go-around complete, scenario will restart
            pass
        return cmd

    # ROLLOUT phase
    if state_memory['phase'] == 'ROLLOUT':
        cmd.update({
            'throttle_1': 0.0,
            'throttle_2': 0.0,
            'elevator': -0.2,  # Nose down for aerodynamic braking
            'spoiler': 1.0,
            'speedbrake': 1.0,
            'reverse_1': 1.0 if eng1_running else 0.0,
            'reverse_2': 1.0 if eng2_running else 0.0,
            'autobrake': 0.8,
            'flaps': 40,
        })

        # Rudder for directional control
        rudder_cmd = -loc_dots * 0.3
        cmd['rudder'] = max(-1.0, min(1.0, rudder_cmd))

        # Aileron to keep wings level
        aileron_cmd = -roll * 0.05
        cmd['aileron'] = max(-1.0, min(1.0, aileron_cmd))

        return cmd

    # FLARE phase
    if state_memory['phase'] == 'FLARE':
        # Exponential flare - reduce descent rate as altitude decreases
        target_vs = -100 * (altitude / 50.0) if altitude > 5 else -50

        # Pitch control for flare
        vs_error = target_vs - vs
        pitch_target = 2.0 + (50 - altitude) * 0.15  # Gradually pitch up
        pitch_error = pitch_target - pitch

        elevator_cmd = pitch_error * 0.08 + vs_error * 0.0003
        cmd['elevator'] = max(-1.0, min(1.0, elevator_cmd))

        # Reduce thrust during flare
        if state_memory['dual_engine_failure']:
            thrust = 0.0
        elif altitude > 30:
            thrust = 0.15
        elif altitude > 10:
            thrust = 0.05
        else:
            thrust = 0.0

        cmd.update({
            'throttle_1': thrust if eng1_running else 0.0,
            'throttle_2': thrust if eng2_running else 0.0,
            'flaps': 40,
        })

        # Maintain centerline during flare
        aileron_cmd = -loc_dots * 0.2 - roll * 0.08
        cmd['aileron'] = max(-1.0, min(1.0, aileron_cmd))

        rudder_cmd = -loc_dots * 0.15
        cmd['rudder'] = max(-1.0, min(1.0, rudder_cmd))

        return cmd

    # APPROACH and INTERCEPT phases
    # Target speed - adjust for conditions
    target_speed = target_vref + 5
    if state_memory['single_engine'] and not state_memory['dual_engine_failure']:
        target_speed = target_vref + 10
    if altitude > 500:
        target_speed = target_vref + 10

    # Glideslope tracking with PID
    Kp_gs = 0.15
    Ki_gs = 0.01
    Kd_gs = 0.3

    gs_error = -gs_dots  # Negative because positive dots means above glideslope
    state_memory['integral_gs'] += gs_error * dt
    state_memory['integral_gs'] = max(-5, min(5, state_memory['integral_gs']))  # Anti-windup

    gs_derivative = (gs_error - state_memory['prev_gs_error']) / dt if dt > 0 else 0
    state_memory['prev_gs_error'] = gs_error

    pitch_cmd_gs = Kp_gs * gs_error + Ki_gs * state_memory['integral_gs'] + Kd_gs * gs_derivative

    # Pitch attitude control
    target_pitch = pitch_cmd_gs
    if altitude > 1000:
        target_pitch = max(-2, min(3, target_pitch))
    else:
        target_pitch = max(-1, min(5, target_pitch))

    pitch_error = target_pitch - pitch
    elevator_cmd = pitch_error * 0.12 + gs_error * 0.08

    cmd['elevator'] = max(-1.0, min(1.0, elevator_cmd))

    # Localizer tracking with PID
    Kp_loc = 0.25
    Ki_loc = 0.005
    Kd_loc = 0.4

    loc_error = -loc_dots
    state_memory['integral_loc'] += loc_error * dt
    state_memory['integral_loc'] = max(-5, min(5, state_memory['integral_loc']))

    loc_derivative = (loc_error - state_memory['prev_loc_error']) / dt if dt > 0 else 0
    state_memory['prev_loc_error'] = loc_error

    roll_cmd_loc = Kp_loc * loc_error + Ki_loc * state_memory['integral_loc'] + Kd_loc * loc_derivative

    # Bank angle limit
    max_bank = 25 if altitude > 500 else 15
    target_roll = max(-max_bank, min(max_bank, roll_cmd_loc))

    roll_error = target_roll - roll
    aileron_cmd = roll_error * 0.1 + loc_error * 0.05

    cmd['aileron'] = max(-1.0, min(1.0, aileron_cmd))

    # Rudder for coordination
    rudder_cmd = -loc_dots * 0.1 + roll * 0.02
    cmd['rudder'] = max(-1.0, min(1.0, rudder_cmd))

    # Speed control with autothrottle
    ias_error = target_speed - ias
    state_memory['integral_ias'] += ias_error * dt
    state_memory['integral_ias'] = max(-100, min(100, state_memory['integral_ias']))

    ias_derivative = (ias_error - state_memory['prev_ias_error']) / dt if dt > 0 else 0
    state_memory['prev_ias_error'] = ias_error

    Kp_ias = 0.015
    Ki_ias = 0.001
    Kd_ias = 0.02

    throttle_cmd = Kp_ias * ias_error + Ki_ias * state_memory['integral_ias'] + Kd_ias * ias_derivative

    # Base thrust for different conditions
    base_thrust = 0.35
    if state_memory['dual_engine_failure']:
        base_thrust = 0.0
        throttle_cmd = 0.0
    elif state_memory['single_engine']:
        base_thrust = 0.55

    if altitude < 200:
        base_thrust *= 0.8

    final_throttle = base_thrust + throttle_cmd
    final_throttle = max(0.0, min(1.0, final_throttle))

    # Handle dual engine failure - glide approach
    if state_memory['dual_engine_failure']:
        cmd.update({
            'throttle_1': 0.0,
            'throttle_2': 0.0,
        })
        # Adjust pitch for best glide
        if altitude > 100:
            cmd['elevator'] = max(-1.0, min(1.0, elevator_cmd - 0.1))
    else:
        # Asymmetric thrust compensation for single engine
        if not eng1_running or eng1_thrust < 5:
            cmd.update({
                'throttle_1': 0.0,
                'throttle_2': final_throttle * 1.1,  # Increase working engine
            })
            cmd['rudder'] = max(-1.0, min(1.0, cmd.get('rudder', 0) + 0.15))  # Compensate for yaw
        elif not eng2_running or eng2_thrust < 5:
            cmd.update({
                'throttle_1': final_throttle * 1.1,
                'throttle_2': 0.0,
            })
            cmd['rudder'] = max(-1.0, min(1.0, cmd.get('rudder', 0) - 0.15))
        else:
            cmd.update({
                'throttle_1': final_throttle,
                'throttle_2': final_throttle,
            })

    # Configuration management
    if altitude > 1500:
        cmd['flaps'] = 5
    elif altitude > 1000:
        cmd['flaps'] = 15
    elif altitude > 500:
        cmd['flaps'] = 20
    elif altitude > 200:
        cmd['flaps'] = 30
    else:
        cmd['flaps'] = 40

    cmd['gear'] = 1.0
    cmd['speedbrake'] = 0.0

    return cmd
