use std::collections::HashMap;
use std::io::{self, BufRead};

#[derive(Clone, Debug)]
struct Sample {
    hole_id: usize,
    sample_idx: usize,
    depth: f64,
    position: [f64; 3],
    properties: Vec<f64>,
}

#[derive(Clone, Debug)]
struct Hole {
    id: usize,
    start: [f64; 3],
    direction: [f64; 3],
    length: f64,
    samples: Vec<(f64, Vec<f64>)>, // (depth, properties)
}

struct Outlier {
    hole_id: usize,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Parse header
    let header = lines.next().unwrap().unwrap();
    let mut parts = header.split_whitespace();
    let num_holes: usize = parts.next().unwrap().parse().unwrap();
    let num_properties: usize = parts.next().unwrap().parse().unwrap();

    // Parse property names
    let prop_line = lines.next().unwrap().unwrap();
    let property_names: Vec<String> = prop_line.split_whitespace()
        .map(|s| s.to_string())
        .collect();

    // Parse holes
    let mut holes = Vec::with_capacity(num_holes);

    while let Some(Ok(line)) = lines.next() {
        if line.trim().is_empty() {
            continue;
        }

        let parts: Vec<&str> = line.split_whitespace().collect();
        let hole_id: usize = parts[0].parse().unwrap();
        let start_x: f64 = parts[1].parse().unwrap();
        let start_y: f64 = parts[2].parse().unwrap();
        let start_z: f64 = parts[3].parse().unwrap();
        let dir_x: f64 = parts[4].parse().unwrap();
        let dir_y: f64 = parts[5].parse().unwrap();
        let dir_z: f64 = parts[6].parse().unwrap();
        let length: f64 = parts[7].parse().unwrap();
        let num_samples: usize = parts[8].parse().unwrap();

        let mut samples = Vec::with_capacity(num_samples);

        for _ in 0..num_samples {
            if let Some(Ok(sample_line)) = lines.next() {
                let sample_parts: Vec<&str> = sample_line.split_whitespace().collect();
                let depth: f64 = sample_parts[0].parse().unwrap();
                let properties: Vec<f64> = sample_parts[1..]
                    .iter()
                    .map(|s| s.parse().unwrap())
                    .collect();
                samples.push((depth, properties));
            }
        }

        holes.push(Hole {
            id: hole_id,
            start: [start_x, start_y, start_z],
            direction: [dir_x, dir_y, dir_z],
            length,
            samples,
        });
    }

    // Calculate all sample positions and flatten
    let mut all_samples = Vec::new();
    for hole in &holes {
        // Normalize direction vector
        let dir_len = (hole.direction[0].powi(2) +
                       hole.direction[1].powi(2) +
                       hole.direction[2].powi(2)).sqrt();
        let dir_norm = if dir_len > 0.0 {
            [hole.direction[0] / dir_len,
             hole.direction[1] / dir_len,
             hole.direction[2] / dir_len]
        } else {
            [0.0, 0.0, 1.0]
        };

        for (sample_idx, (depth, properties)) in hole.samples.iter().enumerate() {
            let position = [
                hole.start[0] + dir_norm[0] * depth,
                hole.start[1] + dir_norm[1] * depth,
                hole.start[2] + dir_norm[2] * depth,
            ];

            all_samples.push(Sample {
                hole_id: hole.id,
                sample_idx,
                depth: *depth,
                position,
                properties: properties.clone(),
            });
        }
    }

    let total_samples = all_samples.len();

    // Adaptive strategy based on dataset size
    let (search_radius, min_neighbors, grid_size) = if total_samples < 1000 {
        (100.0, 3, 10.0)
    } else if total_samples < 10000 {
        (75.0, 5, 20.0)
    } else if total_samples < 100000 {
        (50.0, 8, 30.0)
    } else {
        (40.0, 10, 40.0)
    };

    // Build spatial grid for efficient neighbor search
    let mut spatial_grid: HashMap<(i32, i32, i32), Vec<usize>> = HashMap::new();

    for (idx, sample) in all_samples.iter().enumerate() {
        let grid_x = (sample.position[0] / grid_size).floor() as i32;
        let grid_y = (sample.position[1] / grid_size).floor() as i32;
        let grid_z = (sample.position[2] / grid_size).floor() as i32;

        spatial_grid.entry((grid_x, grid_y, grid_z))
            .or_insert_with(Vec::new)
            .push(idx);
    }

    // Detect outliers
    let mut outliers = Vec::new();

    for (idx, sample) in all_samples.iter().enumerate() {
        let grid_x = (sample.position[0] / grid_size).floor() as i32;
        let grid_y = (sample.position[1] / grid_size).floor() as i32;
        let grid_z = (sample.position[2] / grid_size).floor() as i32;

        // Search neighboring grid cells
        let mut neighbors = Vec::new();

        for dx in -2..=2 {
            for dy in -2..=2 {
                for dz in -2..=2 {
                    if let Some(cell_samples) = spatial_grid.get(&(grid_x + dx, grid_y + dy, grid_z + dz)) {
                        for &neighbor_idx in cell_samples {
                            if neighbor_idx == idx {
                                continue;
                            }

                            let neighbor = &all_samples[neighbor_idx];
                            let dist = distance_3d(&sample.position, &neighbor.position);

                            if dist < search_radius && dist > 0.0 {
                                neighbors.push((neighbor_idx, dist));
                            }
                        }
                    }
                }
            }
        }

        if neighbors.len() < min_neighbors {
            continue;
        }

        // Sort neighbors by distance and take closest ones
        neighbors.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap());
        let max_neighbors = (50).min(neighbors.len());
        neighbors.truncate(max_neighbors);

        // Check each property for outliers
        for prop_idx in 0..num_properties {
            let value = sample.properties[prop_idx];

            // Collect neighbor values with distance weighting
            let mut neighbor_values = Vec::new();
            let mut weights = Vec::new();

            for &(neighbor_idx, dist) in &neighbors {
                let neighbor_value = all_samples[neighbor_idx].properties[prop_idx];
                if neighbor_value.is_finite() {
                    neighbor_values.push(neighbor_value);
                    // Inverse distance weighting
                    weights.push(1.0 / (dist + 1.0));
                }
            }

            if neighbor_values.is_empty() {
                continue;
            }

            // Calculate weighted statistics
            let total_weight: f64 = weights.iter().sum();
            let weighted_mean: f64 = neighbor_values.iter()
                .zip(weights.iter())
                .map(|(v, w)| v * w)
                .sum::<f64>() / total_weight;

            // Calculate weighted standard deviation
            let weighted_variance: f64 = neighbor_values.iter()
                .zip(weights.iter())
                .map(|(v, w)| (v - weighted_mean).powi(2) * w)
                .sum::<f64>() / total_weight;
            let weighted_std = weighted_variance.sqrt();

            if weighted_std < 1e-10 {
                continue;
            }

            // Calculate z-score
            let z_score = ((value - weighted_mean) / weighted_std).abs();

            // Additional robustness: calculate median absolute deviation
            let mut sorted_values = neighbor_values.clone();
            sorted_values.sort_by(|a, b| a.partial_cmp(b).unwrap());
            let median = if sorted_values.len() % 2 == 0 {
                (sorted_values[sorted_values.len() / 2 - 1] + sorted_values[sorted_values.len() / 2]) / 2.0
            } else {
                sorted_values[sorted_values.len() / 2]
            };

            let median_deviation = (value - median).abs();
            let mad_values: Vec<f64> = sorted_values.iter()
                .map(|v| (v - median).abs())
                .collect();
            let mut mad_sorted = mad_values.clone();
            mad_sorted.sort_by(|a, b| a.partial_cmp(b).unwrap());
            let mad = if mad_sorted.len() % 2 == 0 {
                (mad_sorted[mad_sorted.len() / 2 - 1] + mad_sorted[mad_sorted.len() / 2]) / 2.0
            } else {
                mad_sorted[mad_sorted.len() / 2]
            };

            let mad_score = if mad > 1e-10 {
                median_deviation / (mad * 1.4826)
            } else {
                0.0
            };

            // Combined confidence score
            let confidence = (z_score * 0.6 + mad_score * 0.4).max(0.0).min(100.0);

            // Threshold for reporting
            let threshold = if total_samples < 1000 {
                3.0
            } else if total_samples < 10000 {
                3.5
            } else {
                4.0
            };

            if confidence > threshold {
                outliers.push(Outlier {
                    hole_id: sample.hole_id,
                    sample_idx: sample.sample_idx,
                    property_name: property_names[prop_idx].clone(),
                    confidence,
                });
            }
        }
    }

    // Sort by confidence descending
    outliers.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap());

    // Limit output for very large datasets
    let max_output = if total_samples < 10000 {
        10000
    } else {
        5000
    };
    outliers.truncate(max_output);

    // Output results
    println!("{}", outliers.len());
    for outlier in outliers {
        println!("{} {} {} {:.2}",
                 outlier.hole_id,
                 outlier.sample_idx,
                 outlier.property_name,
                 outlier.confidence);
    }
}

fn distance_3d(a: &[f64; 3], b: &[f64; 3]) -> f64 {
    ((a[0] - b[0]).powi(2) + (a[1] - b[1]).powi(2) + (a[2] - b[2]).powi(2)).sqrt()
}
