using System;
using System.Collections.Generic;
using System.Linq;

class OrbitalTSP
{
    const double MU = 398600.4418; // km³/s²
    const double EARTH_RADIUS = 6371.0; // km

    struct Vector3
    {
        public double X, Y, Z;

        public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }

        public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vector3 operator *(Vector3 a, double s) => new Vector3(a.X * s, a.Y * s, a.Z * s);
        public static Vector3 operator /(Vector3 a, double s) => new Vector3(a.X / s, a.Y / s, a.Z / s);

        public double Magnitude() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public Vector3 Normalized() { double m = Magnitude(); return m > 0 ? this / m : this; }
        public static double Dot(Vector3 a, Vector3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vector3 Cross(Vector3 a, Vector3 b) =>
            new Vector3(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
    }

    struct OrbitalState
    {
        public Vector3 Position;
        public Vector3 Velocity;

        public OrbitalState(Vector3 pos, Vector3 vel) { Position = pos; Velocity = vel; }
    }

    static OrbitalState PropagateOrbit(OrbitalState initial, double dt)
    {
        // Simple Runge-Kutta 4th order propagation
        var (k1p, k1v) = Derivatives(initial.Position, initial.Velocity);
        var (k2p, k2v) = Derivatives(initial.Position + k1p * (dt / 2), initial.Velocity + k1v * (dt / 2));
        var (k3p, k3v) = Derivatives(initial.Position + k2p * (dt / 2), initial.Velocity + k2v * (dt / 2));
        var (k4p, k4v) = Derivatives(initial.Position + k3p * dt, initial.Velocity + k3v * dt);

        var newPos = initial.Position + (k1p + k2p * 2 + k3p * 2 + k4p) * (dt / 6);
        var newVel = initial.Velocity + (k1v + k2v * 2 + k3v * 2 + k4v) * (dt / 6);

        return new OrbitalState(newPos, newVel);
    }

    static (Vector3, Vector3) Derivatives(Vector3 pos, Vector3 vel)
    {
        double r = pos.Magnitude();
        Vector3 acc = pos * (-MU / (r * r * r));
        return (vel, acc);
    }

    static double EstimateDeltaV(OrbitalState from, OrbitalState to, double timeOfFlight = 1800.0)
    {
        // Simplified delta-V estimation using Hohmann-like transfer
        double r1 = from.Position.Magnitude();
        double r2 = to.Position.Magnitude();

        // Use vis-viva equation for circular orbit approximation
        double v1 = Math.Sqrt(MU / r1);
        double v2 = Math.Sqrt(MU / r2);

        // Hohmann transfer approximation
        double a_transfer = (r1 + r2) / 2;
        double v_p1 = Math.Sqrt(MU * (2.0 / r1 - 1.0 / a_transfer));
        double v_p2 = Math.Sqrt(MU * (2.0 / r2 - 1.0 / a_transfer));

        double dv1 = Math.Abs(v_p1 - from.Velocity.Magnitude());
        double dv2 = Math.Abs(to.Velocity.Magnitude() - v_p2);

        // Add penalty for plane changes
        Vector3 h1 = Vector3.Cross(from.Position, from.Velocity);
        Vector3 h2 = Vector3.Cross(to.Position, to.Velocity);
        double planeAngle = Math.Acos(Math.Clamp(Vector3.Dot(h1, h2) / (h1.Magnitude() * h2.Magnitude()), -1, 1));
        double planeChangeDv = v1 * Math.Abs(Math.Sin(planeAngle));

        return dv1 + dv2 + planeChangeDv * 0.5;
    }

    static double CalculateRouteCost(List<int> route, OrbitalState start, List<OrbitalState> stations, double[,] costMatrix)
    {
        double totalCost = costMatrix[stations.Count, route[0]]; // Start to first station

        for (int i = 0; i < route.Count - 1; i++)
        {
            totalCost += costMatrix[route[i], route[i + 1]];
        }

        return totalCost;
    }

    static List<int> NearestNeighbor(int n, OrbitalState start, List<OrbitalState> stations, double[,] costMatrix)
    {
        List<int> route = new List<int>();
        HashSet<int> visited = new HashSet<int>();
        int current = n; // Start from special index representing start position

        while (route.Count < n)
        {
            double minCost = double.MaxValue;
            int next = -1;

            for (int i = 0; i < n; i++)
            {
                if (!visited.Contains(i) && costMatrix[current, i] < minCost)
                {
                    minCost = costMatrix[current, i];
                    next = i;
                }
            }

            route.Add(next);
            visited.Add(next);
            current = next;
        }

        return route;
    }

    static List<int> TwoOpt(List<int> route, double[,] costMatrix, OrbitalState start, List<OrbitalState> stations, int maxIterations = 1000)
    {
        List<int> bestRoute = new List<int>(route);
        double bestCost = CalculateRouteCost(bestRoute, start, stations, costMatrix);
        bool improved = true;
        int iterations = 0;

        while (improved && iterations < maxIterations)
        {
            improved = false;
            iterations++;

            for (int i = 0; i < bestRoute.Count - 1; i++)
            {
                for (int j = i + 1; j < bestRoute.Count; j++)
                {
                    List<int> newRoute = new List<int>(bestRoute);

                    // Reverse segment [i+1, j]
                    for (int k = 0; k <= (j - i - 1) / 2; k++)
                    {
                        int temp = newRoute[i + 1 + k];
                        newRoute[i + 1 + k] = newRoute[j - k];
                        newRoute[j - k] = temp;
                    }

                    double newCost = CalculateRouteCost(newRoute, start, stations, costMatrix);

                    if (newCost < bestCost)
                    {
                        bestRoute = newRoute;
                        bestCost = newCost;
                        improved = true;
                    }
                }
            }
        }

        return bestRoute;
    }

    static List<int> SolveSmallTSP(int n, OrbitalState start, List<OrbitalState> stations, double[,] costMatrix)
    {
        // Brute force for very small n (≤ 8)
        if (n <= 8)
        {
            List<int> indices = Enumerable.Range(0, n).ToList();
            return Permutations(indices)
                .Select(perm => perm.ToList())
                .OrderBy(route => CalculateRouteCost(route, start, stations, costMatrix))
                .First();
        }
        else
        {
            // Nearest neighbor + 2-opt
            var route = NearestNeighbor(n, start, stations, costMatrix);
            return TwoOpt(route, costMatrix, start, stations, 2000);
        }
    }

    static List<int> SolveLargeTSP(int n, OrbitalState start, List<OrbitalState> stations, double[,] costMatrix)
    {
        // Multiple nearest neighbor starts + 2-opt
        List<int> bestRoute = null;
        double bestCost = double.MaxValue;

        // Try nearest neighbor from multiple starting points
        int samples = Math.Min(n, 10);
        for (int startIdx = 0; startIdx < samples; startIdx++)
        {
            var route = NearestNeighbor(n, start, stations, costMatrix);

            // Apply 2-opt improvement
            route = TwoOpt(route, costMatrix, start, stations, 500);

            double cost = CalculateRouteCost(route, start, stations, costMatrix);
            if (cost < bestCost)
            {
                bestCost = cost;
                bestRoute = route;
            }
        }

        return bestRoute;
    }

    static IEnumerable<IEnumerable<T>> Permutations<T>(IEnumerable<T> list)
    {
        if (!list.Any()) yield return Enumerable.Empty<T>();
        else
        {
            int count = 0;
            foreach (var element in list)
            {
                var remaining = list.Take(count).Concat(list.Skip(count + 1));
                foreach (var perm in Permutations(remaining))
                {
                    yield return new[] { element }.Concat(perm);
                }
                count++;
            }
        }
    }

    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split();
        int n = int.Parse(firstLine[0]);
        double mu = double.Parse(firstLine[1]);

        string[] startLine = Console.ReadLine().Split();
        OrbitalState startState = new OrbitalState(
            new Vector3(double.Parse(startLine[0]), double.Parse(startLine[1]), double.Parse(startLine[2])),
            new Vector3(double.Parse(startLine[3]), double.Parse(startLine[4]), double.Parse(startLine[5]))
        );

        List<OrbitalState> stations = new List<OrbitalState>();
        for (int i = 0; i < n; i++)
        {
            string[] stationLine = Console.ReadLine().Split();
            stations.Add(new OrbitalState(
                new Vector3(double.Parse(stationLine[0]), double.Parse(stationLine[1]), double.Parse(stationLine[2])),
                new Vector3(double.Parse(stationLine[3]), double.Parse(stationLine[4]), double.Parse(stationLine[5]))
            ));
        }

        // Build cost matrix
        double[,] costMatrix = new double[n + 1, n];

        // Start to each station
        for (int i = 0; i < n; i++)
        {
            costMatrix[n, i] = EstimateDeltaV(startState, stations[i]);
        }

        // Station to station
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (i != j)
                {
                    costMatrix[i, j] = EstimateDeltaV(stations[i], stations[j]);
                }
            }
        }

        // Solve TSP based on problem size
        List<int> route;
        if (n <= 10)
        {
            route = SolveSmallTSP(n, startState, stations, costMatrix);
        }
        else
        {
            route = SolveLargeTSP(n, startState, stations, costMatrix);
        }

        Console.WriteLine(string.Join(" ", route));
    }
}
