#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against out-of-bounds invocations
    if (i >= N) {
        return;
    }

    // Extract simulation parameters
    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;
    float softening2 = softening * softening;

    // Read current body's position, mass, and velocity
    vec4 positionMass = inData[i * 2];
    vec4 velocityPad = inData[i * 2 + 1];

    vec3 pos_i = positionMass.xyz;
    float mass_i = positionMass.w;
    vec3 vel_i = velocityPad.xyz;

    // Compute acceleration from all other bodies
    vec3 acc = vec3(0.0, 0.0, 0.0);

    for (uint j = 0; j < N; j++) {
        // Skip self-interaction
        if (i == j) {
            continue;
        }

        // Read body j's position and mass
        vec4 positionMass_j = inData[j * 2];
        vec3 pos_j = positionMass_j.xyz;
        float mass_j = positionMass_j.w;

        // Compute displacement vector
        vec3 r = pos_j - pos_i;

        // Compute distance squared with softening
        float dist2 = dot(r, r) + softening2;

        // Compute 1 / (dist^3) = (dist2)^(-3/2)
        float dist3_inv = pow(dist2, -1.5);

        // Accumulate acceleration: acc += G * mass_j * r / dist^3
        acc += G * mass_j * r * dist3_inv;
    }

    // Update velocity and position using symplectic Euler integration
    vec3 vel_new = vel_i + acc * dt;
    vec3 pos_new = pos_i + vel_new * dt;

    // Write updated position+mass and velocity to output
    outData[i * 2] = vec4(pos_new, mass_i);
    outData[i * 2 + 1] = vec4(vel_new, 0.0);
}
