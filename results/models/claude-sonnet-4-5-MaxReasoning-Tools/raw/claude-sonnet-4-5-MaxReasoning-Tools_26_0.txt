use std::io::{self, Read, Write, BufReader, BufWriter};
use std::f32;

const SMALL_THRESHOLD: usize = 10_000_000;
const MEDIUM_THRESHOLD: usize = 100_000_000;
const LARGE_THRESHOLD: usize = 1_000_000_000;

#[derive(Clone, Copy, Debug)]
struct Point3D {
    x: f32,
    y: f32,
    z: f32,
}

impl Point3D {
    fn distance_squared(&self, other: &Point3D) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }

    fn add(&self, other: &Point3D) -> Point3D {
        Point3D {
            x: self.x + other.x,
            y: self.y + other.y,
            z: self.z + other.z,
        }
    }

    fn scale(&self, factor: f32) -> Point3D {
        Point3D {
            x: self.x * factor,
            y: self.y * factor,
            z: self.z * factor,
        }
    }

    fn zero() -> Point3D {
        Point3D { x: 0.0, y: 0.0, z: 0.0 }
    }
}

fn read_header<R: Read>(reader: &mut R) -> io::Result<(usize, usize)> {
    let mut buf = [0u8; 16];
    reader.read_exact(&mut buf)?;

    let num_points = usize::from_le_bytes([buf[0], buf[1], buf[2], buf[3], buf[4], buf[5], buf[6], buf[7]]);
    let num_clusters = usize::from_le_bytes([buf[8], buf[9], buf[10], buf[11], buf[12], buf[13], buf[14], buf[15]]);

    Ok((num_points, num_clusters))
}

fn read_point<R: Read>(reader: &mut R) -> io::Result<Point3D> {
    let mut buf = [0u8; 12];
    reader.read_exact(&mut buf)?;

    let x = f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]);
    let y = f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]);
    let z = f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]);

    Ok(Point3D { x, y, z })
}

fn write_cluster<W: Write>(writer: &mut W, cluster: u32) -> io::Result<()> {
    writer.write_all(&cluster.to_le_bytes())
}

fn initialize_centroids(points: &[Point3D], k: usize) -> Vec<Point3D> {
    if points.is_empty() {
        return vec![Point3D::zero(); k];
    }

    let mut centroids = Vec::with_capacity(k);

    // Use k-means++ initialization for better initial centroids
    let first_idx = (points.len() / 2).min(points.len() - 1);
    centroids.push(points[first_idx]);

    let mut distances = vec![f32::INFINITY; points.len()];

    for _ in 1..k {
        let mut sum_distances = 0.0;

        for (i, point) in points.iter().enumerate() {
            let mut min_dist = f32::INFINITY;
            for centroid in &centroids {
                let dist = point.distance_squared(centroid);
                if dist < min_dist {
                    min_dist = dist;
                }
            }
            distances[i] = min_dist;
            sum_distances += min_dist;
        }

        if sum_distances <= 0.0 {
            // All points are at centroids, pick random points
            let idx = (centroids.len() * 137) % points.len();
            centroids.push(points[idx]);
        } else {
            let threshold = sum_distances * 0.5;
            let mut cumsum = 0.0;
            let mut selected_idx = 0;

            for (i, &dist) in distances.iter().enumerate() {
                cumsum += dist;
                if cumsum >= threshold {
                    selected_idx = i;
                    break;
                }
            }

            centroids.push(points[selected_idx]);
        }
    }

    centroids
}

fn assign_clusters(points: &[Point3D], centroids: &[Point3D]) -> Vec<u32> {
    points.iter().map(|point| {
        let mut min_dist = f32::INFINITY;
        let mut best_cluster = 0;

        for (i, centroid) in centroids.iter().enumerate() {
            let dist = point.distance_squared(centroid);
            if dist < min_dist {
                min_dist = dist;
                best_cluster = i;
            }
        }

        best_cluster as u32
    }).collect()
}

fn update_centroids(points: &[Point3D], assignments: &[u32], k: usize) -> Vec<Point3D> {
    let mut sums = vec![Point3D::zero(); k];
    let mut counts = vec![0usize; k];

    for (point, &cluster) in points.iter().zip(assignments.iter()) {
        sums[cluster as usize] = sums[cluster as usize].add(point);
        counts[cluster as usize] += 1;
    }

    sums.iter().zip(counts.iter()).enumerate().map(|(i, (sum, &count))| {
        if count > 0 {
            sum.scale(1.0 / count as f32)
        } else {
            // Handle empty cluster by using a point far from existing centroids
            if !points.is_empty() {
                points[i % points.len()]
            } else {
                Point3D::zero()
            }
        }
    }).collect()
}

fn kmeans_small(points: Vec<Point3D>, k: usize) -> Vec<u32> {
    let mut centroids = initialize_centroids(&points, k);
    let max_iterations = 100;

    for _ in 0..max_iterations {
        let assignments = assign_clusters(&points, &centroids);
        let new_centroids = update_centroids(&points, &assignments, k);

        // Check convergence
        let mut converged = true;
        for (old, new) in centroids.iter().zip(new_centroids.iter()) {
            if old.distance_squared(new) > 1e-6 {
                converged = false;
                break;
            }
        }

        centroids = new_centroids;

        if converged {
            break;
        }
    }

    assign_clusters(&points, &centroids)
}

fn kmeans_streaming<R: Read, W: Write>(
    reader: &mut R,
    writer: &mut W,
    num_points: usize,
    k: usize,
    batch_size: usize,
) -> io::Result<()> {
    // Initialize centroids from first batch
    let init_size = (batch_size * 2).min(num_points);
    let mut init_points = Vec::with_capacity(init_size);

    for _ in 0..init_size {
        init_points.push(read_point(reader)?);
    }

    let mut centroids = initialize_centroids(&init_points, k);
    let mut centroid_counts = vec![0usize; k];

    // Write assignments for init points
    for point in &init_points {
        let mut min_dist = f32::INFINITY;
        let mut best_cluster = 0;

        for (i, centroid) in centroids.iter().enumerate() {
            let dist = point.distance_squared(centroid);
            if dist < min_dist {
                min_dist = dist;
                best_cluster = i;
            }
        }

        write_cluster(writer, best_cluster as u32)?;
        centroid_counts[best_cluster] += 1;
    }

    // Process remaining points in streaming fashion
    let remaining = num_points - init_size;
    let mut batch = Vec::with_capacity(batch_size);
    let update_rate = if num_points > LARGE_THRESHOLD { 0.001 } else { 0.01 };

    for i in 0..remaining {
        let point = read_point(reader)?;
        batch.push(point);

        if batch.len() >= batch_size || i == remaining - 1 {
            // Assign and update centroids for batch
            for point in &batch {
                let mut min_dist = f32::INFINITY;
                let mut best_cluster = 0;

                for (j, centroid) in centroids.iter().enumerate() {
                    let dist = point.distance_squared(centroid);
                    if dist < min_dist {
                        min_dist = dist;
                        best_cluster = j;
                    }
                }

                write_cluster(writer, best_cluster as u32)?;

                // Incremental centroid update (mini-batch k-means)
                let old_centroid = centroids[best_cluster];
                let new_point = *point;
                let learning_rate = update_rate;

                centroids[best_cluster] = Point3D {
                    x: old_centroid.x + learning_rate * (new_point.x - old_centroid.x),
                    y: old_centroid.y + learning_rate * (new_point.y - old_centroid.y),
                    z: old_centroid.z + learning_rate * (new_point.z - old_centroid.z),
                };

                centroid_counts[best_cluster] += 1;
            }

            batch.clear();
        }
    }

    Ok(())
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::with_capacity(1024 * 1024, stdin.lock());

    let stdout = io::stdout();
    let mut writer = BufWriter::with_capacity(1024 * 1024, stdout.lock());

    // Read header
    let (num_points, num_clusters) = read_header(&mut reader)?;

    // Choose algorithm based on dataset size
    if num_points <= SMALL_THRESHOLD {
        // Small dataset: load all points into memory and use standard k-means
        let mut points = Vec::with_capacity(num_points);
        for _ in 0..num_points {
            points.push(read_point(&mut reader)?);
        }

        let assignments = kmeans_small(points, num_clusters);

        for assignment in assignments {
            write_cluster(&mut writer, assignment)?;
        }
    } else if num_points <= MEDIUM_THRESHOLD {
        // Medium dataset: streaming with moderate batch size
        let batch_size = 10000;
        kmeans_streaming(&mut reader, &mut writer, num_points, num_clusters, batch_size)?;
    } else if num_points <= LARGE_THRESHOLD {
        // Large dataset: streaming with smaller batch size
        let batch_size = 5000;
        kmeans_streaming(&mut reader, &mut writer, num_points, num_clusters, batch_size)?;
    } else {
        // Very large dataset: streaming with minimal batch size
        let batch_size = 1000;
        kmeans_streaming(&mut reader, &mut writer, num_points, num_clusters, batch_size)?;
    }

    writer.flush()?;
    Ok(())
}
