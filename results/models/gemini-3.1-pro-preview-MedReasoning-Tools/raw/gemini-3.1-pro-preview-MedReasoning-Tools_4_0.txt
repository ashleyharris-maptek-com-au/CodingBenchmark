#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <map>
#include <set>
#include <random>
#include <iomanip>

using namespace std;

struct Point3 {
    double x, y, z;
    Point3 operator-(const Point3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    double dot(const Point3& o) const { return x * o.x + y * o.y + z * o.z; }
    Point3 cross(const Point3& o) const { return {y * o.z - z * o.y, z * o.x - x * o.z, x * o.y - y * o.x}; }
    double length() const { return std::sqrt(x * x + y * y + z * z); }
};

struct Face {
    int v[3];
    Point3 normal;
    double d;
    bool dead = false;
    void compute(const vector<Point3>& pts) {
        normal = (pts[v[1]] - pts[v[0]]).cross(pts[v[2]] - pts[v[0]]);
        double len = normal.length();
        if (len > 1e-12) {
            normal.x /= len; normal.y /= len; normal.z /= len;
        }
        d = -normal.dot(pts[v[0]]);
    }
    double dist(const Point3& p) const {
        return normal.dot(p) + d;
    }
};

vector<Face> build_hull(vector<Point3>& pts) {
    int n = pts.size();
    if (n < 4) return {};
    
    mt19937 rng(42);
    uniform_real_distribution<double> dist_noise(-1e-8, 1e-8);
    for (auto& p : pts) {
        p.x += dist_noise(rng);
        p.y += dist_noise(rng);
        p.z += dist_noise(rng);
    }
    
    int p0 = 0, p1 = 1, p2 = 2, p3 = 3;
    double max_vol = -1;
    int search_limit = min(20, n);
    for (int i = 0; i < search_limit; ++i) {
        for (int j = i + 1; j < search_limit; ++j) {
            for (int k = j + 1; k < search_limit; ++k) {
                for (int l = k + 1; l < search_limit; ++l) {
                    double vol = std::abs((pts[j] - pts[i]).dot((pts[k] - pts[i]).cross(pts[l] - pts[i])));
                    if (vol > max_vol) {
                        max_vol = vol;
                        p0 = i; p1 = j; p2 = k; p3 = l;
                    }
                }
            }
        }
    }
    
    if (max_vol < 1e-9) return {};
    
    vector<Point3> init_pts = {pts[p0], pts[p1], pts[p2], pts[p3]};
    vector<Point3> new_pts = init_pts;
    for (int i = 0; i < n; ++i) {
        if (i != p0 && i != p1 && i != p2 && i != p3) {
            new_pts.push_back(pts[i]);
        }
    }
    pts = new_pts;
    
    vector<Face> faces;
    faces.push_back({ {0, 1, 2} });
    faces.push_back({ {0, 2, 3} });
    faces.push_back({ {0, 3, 1} });
    faces.push_back({ {1, 3, 2} });
    
    Point3 center = {(pts[0].x + pts[1].x + pts[2].x + pts[3].x) / 4.0,
                     (pts[0].y + pts[1].y + pts[2].y + pts[3].y) / 4.0,
                     (pts[0].z + pts[1].z + pts[2].z + pts[3].z) / 4.0};
                     
    for (auto& f : faces) {
        f.compute(pts);
        if (f.dist(center) > 0) {
            swap(f.v[0], f.v[1]);
            f.compute(pts);
        }
    }
    
    for (int i = 4; i < n; ++i) {
        vector<int> visible;
        for (int j = 0; j < (int)faces.size(); ++j) {
            if (!faces[j].dead && faces[j].dist(pts[i]) > 1e-10) {
                visible.push_back(j);
            }
        }
        if (visible.empty()) continue;
        
        map<pair<int, int>, int> edge_counts;
        for (int j : visible) {
            faces[j].dead = true;
            for (int k = 0; k < 3; ++k) {
                int a = faces[j].v[k];
                int b = faces[j].v[(k + 1) % 3];
                edge_counts[{a, b}]++;
            }
        }
        
        for (int j : visible) {
            for (int k = 0; k < 3; ++k) {
                int a = faces[j].v[k];
                int b = faces[j].v[(k + 1) % 3];
                if (edge_counts[{a, b}] == 1 && edge_counts[{b, a}] == 0) {
                    Face new_f = { {a, b, i} };
                    new_f.compute(pts);
                    faces.push_back(new_f);
                }
            }
        }
    }
    
    vector<Face> final_faces;
    for (const auto& f : faces) {
        if (!f.dead) final_faces.push_back(f);
    }
    return final_faces;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int V_count;
    double edge_length;
    if (!(cin >> V_count >> edge_length)) return 0;

    vector<Point3> pts(V_count);
    for (int i = 0; i < V_count; ++i) {
        cin >> pts[i].x >> pts[i].y >> pts[i].z;
    }

    vector<Face> hull = build_hull(pts);
    if (hull.empty()) {
        cout << 0 << "\n";
        return 0;
    }

    double e = edge_length;
    double R = e * sqrt(3.0 / 8.0);
    double r = e / sqrt(24.0);

    Point3 V_tet[4];
    V_tet[0] = {0, 0, R};
    V_tet[1] = {e / sqrt(3.0), 0, -r};
    V_tet[2] = {-e / (2.0 * sqrt(3.0)), e / 2.0, -r};
    V_tet[3] = {-e / (2.0 * sqrt(3.0)), -e / 2.0, -r};

    double M[3][3] = {
        {1.0 / sqrt(6.0), 1.0 / sqrt(6.0), -2.0 / sqrt(6.0)},
        {1.0 / sqrt(2.0), -1.0 / sqrt(2.0), 0.0},
        {-1.0 / sqrt(3.0), -1.0 / sqrt(3.0), -1.0 / sqrt(3.0)}
    };

    double a = e / sqrt(2.0);

    double min_u = 1e18, max_u = -1e18;
    double min_v = 1e18, max_v = -1e18;
    double min_w = 1e18, max_w = -1e18;

    for (const auto& p : pts) {
        double u = M[0][0] * p.x + M[1][0] * p.y + M[2][0] * p.z;
        double v = M[0][1] * p.x + M[1][1] * p.y + M[2][1] * p.z;
        double w = M[0][2] * p.x + M[1][2] * p.y + M[2][2] * p.z;
        min_u = min(min_u, u); max_u = max(max_u, u);
        min_v = min(min_v, v); max_v = max(max_v, v);
        min_w = min(min_w, w); max_w = max(max_w, w);
    }

    int start_i = floor(min_u / a) - 1;
    int end_i   = ceil(max_u / a) + 1;
    int start_j = floor(min_v / a) - 1;
    int end_j   = ceil(max_v / a) + 1;
    int start_k = floor(min_w / a) - 1;
    int end_k   = ceil(max_w / a) + 1;

    long long total_cells = (long long)(end_i - start_i + 1) * 
                            (end_j - start_j + 1) * 
                            (end_k - start_k + 1);

    int grid_steps = 1;
    if (total_cells * hull.size() < 20000000LL) {
        grid_steps = 4;
    } else if (total_cells * hull.size() < 50000000LL) {
        grid_steps = 3;
    } else if (total_cells * hull.size() < 200000000LL) {
        grid_steps = 2;
    }

    double step = a / grid_steps;
    int max_placed = -1;
    vector<Point3> best_centers;

    for (int ox_i = 0; ox_i < grid_steps; ++ox_i) {
        for (int oy_i = 0; oy_i < grid_steps; ++oy_i) {
            for (int oz_i = 0; oz_i < grid_steps; ++oz_i) {
                double ox = ox_i * step;
                double oy = oy_i * step;
                double oz = oz_i * step;
                
                vector<Point3> current_centers;
                for (int i = start_i; i <= end_i; ++i) {
                    for (int j = start_j; j <= end_j; ++j) {
                        for (int k = start_k; k <= end_k; ++k) {
                            Point3 C_unrot = {i * a + ox, j * a + oy, k * a + oz};
                            
                            Point3 C_real;
                            C_real.x = M[0][0] * C_unrot.x + M[0][1] * C_unrot.y + M[0][2] * C_unrot.z;
                            C_real.y = M[1][0] * C_unrot.x + M[1][1] * C_unrot.y + M[1][2] * C_unrot.z;
                            C_real.z = M[2][0] * C_unrot.x + M[2][1] * C_unrot.y + M[2][2] * C_unrot.z;
                            
                            bool ok = true;
                            for (int n = 0; n < 4; ++n) {
                                Point3 P = {V_tet[n].x + C_real.x, 
                                            V_tet[n].y + C_real.y, 
                                            V_tet[n].z + C_real.z};
                                for (const auto& f : hull) {
                                    if (f.dist(P) > 1e-7) {
                                        ok = false;
                                        break;
                                    }
                                }
                                if (!ok) break;
                            }
                            if (ok) {
                                current_centers.push_back(C_real);
                            }
                        }
                    }
                }
                if ((int)current_centers.size() > max_placed) {
                    max_placed = current_centers.size();
                    best_centers = current_centers;
                }
            }
        }
    }

    cout << best_centers.size() << "\n";
    cout << fixed << setprecision(6);
    for (const auto& c : best_centers) {
        cout << c.x << " " << c.y << " " << c.z << " 0.000000 0.000000 0.000000\n";
    }

    return 0;
}
