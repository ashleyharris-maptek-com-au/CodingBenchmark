using namespace std;

const int BUF_SIZE = 1 << 24;
char buf[BUF_SIZE];
int buf_pos = 0;
int buf_len = 0;

inline char next_char() {
    if (buf_pos >= buf_len) {
        buf_len = (int)fread(buf, 1, BUF_SIZE, stdin);
        buf_pos = 0;
        if (buf_len == 0) return EOF;
    }
    return buf[buf_pos++];
}

inline float read_float_fast() {
    char c = next_char();
    while (c <= ' ' && c != EOF) c = next_char();
    if (c == EOF) return 0;
    bool neg = false;
    if (c == '-') {
        neg = true;
        c = next_char();
    }
    double val = 0;
    while (c >= '0' && c <= '9') {
        val = val * 10.0 + (c - '0');
        c = next_char();
    }
    if (c == '.') {
        c = next_char();
        double frac = 1.0;
        while (c >= '0' && c <= '9') {
            frac *= 0.1;
            val += (c - '0') * frac;
            c = next_char();
        }
    }
    if (c == 'e' || c == 'E') {
        c = next_char();
        bool eneg = false;
        if (c == '-') { eneg = true; c = next_char(); }
        else if (c == '+') { c = next_char(); }
        int exp = 0;
        while (c >= '0' && c <= '9') {
            exp = exp * 10 + (c - '0');
            c = next_char();
        }
        if (eneg) {
            while (exp--) val *= 0.1;
        } else {
            while (exp--) val *= 10.0;
        }
    }
    return (float)(neg ? -val : val);
}

inline int read_int_fast() {
    char c = next_char();
    while (c <= ' ' && c != EOF) c = next_char();
    if (c == EOF) return 0;
    bool neg = false;
    if (c == '-') {
        neg = true;
        c = next_char();
    }
    int val = 0;
    while (c >= '0' && c <= '9') {
        val = val * 10 + (c - '0');
        c = next_char();
    }
    return neg ? -val : val;
}

char out_buf[BUF_SIZE];
int out_pos = 0;

inline void flush_out() {
    if (out_pos > 0) {
        fwrite(out_buf, 1, out_pos, stdout);
        out_pos = 0;
    }
}

inline void write_int_fast(int x) {
    if (x == 0) {
        if (out_pos + 2 >= BUF_SIZE) flush_out();
        out_buf[out_pos++] = '0';
        out_buf[out_pos++] = ' ';
        return;
    }
    if (x < 0) {
        if (out_pos + 1 >= BUF_SIZE) flush_out();
        out_buf[out_pos++] = '-';
        x = -x;
    }
    char temp[12];
    int t_pos = 0;
    while (x > 0) {
        temp[t_pos++] = '0' + (x % 10);
        x /= 10;
    }
    if (out_pos + t_pos + 1 >= BUF_SIZE) flush_out();
    while (t_pos > 0) {
        out_buf[out_pos++] = temp[--t_pos];
    }
    out_buf[out_pos++] = ' ';
}

struct Point {
    float x, y;
};
Point* pts = nullptr;

inline double get_dist(int a, int b) {
    double dx = pts[a].x - pts[b].x;
    double dy = pts[a].y - pts[b].y;
    return std::sqrt(dx * dx + dy * dy);
}

inline double get_dist_sq(int a, int b) {
    double dx = pts[a].x - pts[b].x;
    double dy = pts[a].y - pts[b].y;
    return dx * dx + dy * dy;
}

struct PointRef {
    uint64_t hcode;
    uint32_t id;
    bool operator<(const PointRef& other) const {
        return hcode < other.hcode;
    }
};

int main() {
#ifdef _WIN32
    _setmode(_fileno(stdin), _O_BINARY);
    _setmode(_fileno(stdout), _O_BINARY);
#endif

    auto start_time = std::chrono::steady_clock::now();

    int N = read_int_fast();
    if (N <= 0) return 0;

    pts = (Point*)malloc((size_t)N * sizeof(Point));
    for (int i = 0; i < N; i++) {
        pts[i].x = read_float_fast();
        pts[i].y = read_float_fast();
    }

    int* tour = (int*)malloc((size_t)N * sizeof(int));

    if (N <= 11) {
        std::vector<int> p(N);
        std::iota(p.begin(), p.end(), 0);
        double best_dist = 1e30;
        std::vector<int> best_p = p;
        do {
            double d = 0;
            for (int i = 0; i < N - 1; i++) d += get_dist(p[i], p[i+1]);
            d += get_dist(p[N-1], p[0]);
            if (d < best_dist) {
                best_dist = d;
                best_p = p;
            }
        } while (std::next_permutation(p.begin() + 1, p.end()));
        for(int i = 0; i < N; i++) tour[i] = best_p[i];
    } else if (N <= 2000) {
        double best_global_dist = 1e30;
        int* current_tour = (int*)malloc((size_t)N * sizeof(int));
        bool* visited = (bool*)malloc((size_t)N * sizeof(bool));
        
        int num_restarts = std::min(N, (N < 500 ? N : 200)); 
        
        for (int start_node = 0; start_node < num_restarts; start_node++) {
            auto curr_time = std::chrono::steady_clock::now();
            if (std::chrono::duration<double>(curr_time - start_time).count() > 285.0) break;
            
            memset(visited, 0, N * sizeof(bool));
            current_tour[0] = start_node;
            visited[start_node] = true;
            for (int step = 1; step < N; step++) {
                int last = current_tour[step - 1];
                int best_next = -1;
                double best_d_sq = 1e30;
                for (int i = 0; i < N; i++) {
                    if (!visited[i]) {
                        double d_sq = get_dist_sq(last, i);
                        if (d_sq < best_d_sq) {
                            best_d_sq = d_sq;
                            best_next = i;
                        }
                    }
                }
                current_tour[step] = best_next;
                visited[best_next] = true;
            }
            
            bool improved = true;
            while (improved) {
                improved = false;
                for (int i = 0; i < N - 2; i++) {
                    for (int j = i + 2; j < N - 1; j++) {
                        double d1 = get_dist(current_tour[i], current_tour[i+1]) + get_dist(current_tour[j], current_tour[j+1]);
                        double d2 = get_dist(current_tour[i], current_tour[j]) + get_dist(current_tour[i+1], current_tour[j+1]);
                        if (d2 < d1 - 1e-7) {
                            std::reverse(current_tour + i + 1, current_tour + j + 1);
                            improved = true;
                        }
                    }
                }
            }
            
            double current_dist = 0;
            for (int i = 0; i < N - 1; i++) current_dist += get_dist(current_tour[i], current_tour[i+1]);
            current_dist += get_dist(current_tour[N-1], current_tour[0]);
            
            if (current_dist < best_global_dist) {
                best_global_dist = current_dist;
                memcpy(tour, current_tour, (size_t)N * sizeof(int));
            }
        }
        free(current_tour);
        free(visited);
    } else {
        PointRef* refs = (PointRef*)malloc((size_t)N * sizeof(PointRef));
        float min_x = 1e30f, max_x = -1e30f;
        float min_y = 1e30f, max_y = -1e30f;
        for (int i = 0; i < N; i++) {
            if (pts[i].x < min_x) min_x = pts[i].x;
            if (pts[i].x > max_x) max_x = pts[i].x;
            if (pts[i].y < min_y) min_y = pts[i].y;
            if (pts[i].y > max_y) max_y = pts[i].y;
        }
        double range_x = max_x - min_x;
        double range_y = max_y - min_y;
        double range = std::max(range_x, range_y);
        if (range == 0.0) range = 1.0;
        double scale = 4294967295.0 / range;
        
        for (int i = 0; i < N; i++) {
            uint32_t ix = (uint32_t) std::min<double>(4294967295.0, std::max<double>(0.0, (pts[i].x - min_x) * scale));
            uint32_t iy = (uint32_t) std::min<double>(4294967295.0, std::max<double>(0.0, (pts[i].y - min_y) * scale));
            
            uint64_t d = 0;
            for (int bit = 31; bit >= 0; bit--) {
                uint32_t rx = (ix >> bit) & 1;
                uint32_t ry = (iy >> bit) & 1;
                uint64_t val = (3 * rx) ^ ry;
                d = (d << 2) | val;
                if (ry == 0) {
                    if (rx == 1) {
                        ix = ~ix;
                        iy = ~iy;
                    }
                    uint32_t t = ix;
                    ix = iy;
                    iy = t;
                }
            }
            refs[i].hcode = d;
            refs[i].id = i;
        }
        
        std::sort(refs, refs + N);
        for (int i = 0; i < N; i++) {
            tour[i] = refs[i].id;
        }
        free(refs);
        
        int window_size = 40;
        if (N <= 20000) window_size = 1000;
        else if (N <= 100000) window_size = 500;
        else if (N <= 1000000) window_size = 150;
        else if (N <= 5000000) window_size = 60;
        else window_size = 30;
        
        bool global_improved = true;
        int iter = 0;
        while (global_improved && iter < 50) {
            global_improved = false;
            bool local_improved = true;
            while (local_improved) {
                local_improved = false;
                for (int i = 0; i < N - 2; i++) {
                    if ((i & 65535) == 0) {
                        auto curr = std::chrono::steady_clock::now();
                        if (std::chrono::duration<double>(curr - start_time).count() > 285.0) goto timeout;
                    }
                    int limit = std::min(N - 1, i + window_size);
                    for (int j = i + 2; j < limit; j++) {
                        int t_i = tour[i];
                        int t_i1 = tour[i + 1];
                        int t_j = tour[j];
                        int t_j1 = tour[j + 1];

                        double d1 = get_dist(t_i, t_i1) + get_dist(t_j, t_j1);
                        double d2 = get_dist(t_i, t_j) + get_dist(t_i1, t_j1);

                        if (d2 < d1 - 1e-7) { 
                            std::reverse(tour + i + 1, tour + j + 1);
                            local_improved = true;
                            global_improved = true;
                        }
                    }
                }
            }
            if (global_improved) {
                int shift = N / 3;
                if (shift > 0) std::rotate(tour, tour + shift, tour + N);
            }
            iter++;
        }
    timeout:;
    }

    int zero_idx = -1;
    for (int i = 0; i < N; i++) {
        if (tour[i] == 0) {
            zero_idx = i;
            break;
        }
    }
    if (zero_idx != -1 && zero_idx != 0) {
        std::rotate(tour, tour + zero_idx, tour + N);
    }

    for (int i = 0; i < N; i++) {
        write_int_fast(tour[i]);
    }
    flush_out();
    printf("\n");

    free(pts);
    free(tour);

    return 0;
}
