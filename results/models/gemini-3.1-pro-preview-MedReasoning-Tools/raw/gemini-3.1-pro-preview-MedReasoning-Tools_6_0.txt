using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;
using System.Linq;

namespace OrbitalTSP
{
    class Program
    {
        struct State
        {
            public int id;
            public double x, y, z, vx, vy, vz;
        }

        static double CalculateCost(State s1, State s2, double mu)
        {
            double r1 = Math.Sqrt(s1.x * s1.x + s1.y * s1.y + s1.z * s1.z);
            double v1_mag = Math.Sqrt(s1.vx * s1.vx + s1.vy * s1.vy + s1.vz * s1.vz);

            double r2 = Math.Sqrt(s2.x * s2.x + s2.y * s2.y + s2.z * s2.z);
            double v2_mag = Math.Sqrt(s2.vx * s2.vx + s2.vy * s2.vy + s2.vz * s2.vz);

            double e1_n = v1_mag * v1_mag / 2.0 - mu / r1;
            double e2_n = v2_mag * v2_mag / 2.0 - mu / r2;

            double a1 = Math.Max(1.0, -mu / (2.0 * e1_n));
            double a2 = Math.Max(1.0, -mu / (2.0 * e2_n));

            double h1x = s1.y * s1.vz - s1.z * s1.vy;
            double h1y = s1.z * s1.vx - s1.x * s1.vz;
            double h1z = s1.x * s1.vy - s1.y * s1.vx;
            double h1 = Math.Sqrt(h1x * h1x + h1y * h1y + h1z * h1z);

            double h2x = s2.y * s2.vz - s2.z * s2.vy;
            double h2y = s2.z * s2.vx - s2.x * s2.vz;
            double h2z = s2.x * s2.vy - s2.y * s2.vx;
            double h2 = Math.Sqrt(h2x * h2x + h2y * h2y + h2z * h2z);

            double dot_h = h1x * h2x + h1y * h2y + h1z * h2z;
            double cos_theta = dot_h / (h1 * h2);
            cos_theta = Math.Max(-1.0, Math.Min(1.0, cos_theta));
            double theta = Math.Acos(cos_theta);

            double vc1 = Math.Sqrt(Math.Max(0, mu / a1));
            double vc2 = Math.Sqrt(Math.Max(0, mu / a2));

            double a_tx = (a1 + a2) / 2.0;
            double v_tx1 = Math.Sqrt(Math.Max(0, mu * (2.0 / a1 - 1.0 / a_tx)));
            double v_tx2 = Math.Sqrt(Math.Max(0, mu * (2.0 / a2 - 1.0 / a_tx)));

            double dv_hohmann = Math.Abs(v_tx1 - vc1) + Math.Abs(v_tx2 - vc2);
            double v_ap = Math.Min(v_tx1, v_tx2);
            double dv_plane = 2.0 * v_ap * Math.Sin(theta / 2.0);

            double dot_rv1 = s1.x * s1.vx + s1.y * s1.vy + s1.z * s1.vz;
            double term1_1 = v1_mag * v1_mag - mu / r1;
            double e1x = (term1_1 * s1.x - dot_rv1 * s1.vx) / mu;
            double e1y = (term1_1 * s1.y - dot_rv1 * s1.vy) / mu;
            double e1z = (term1_1 * s1.z - dot_rv1 * s1.vz) / mu;

            double dot_rv2 = s2.x * s2.vx + s2.y * s2.vy + s2.z * s2.vz;
            double term1_2 = v2_mag * v2_mag - mu / r2;
            double e2x = (term1_2 * s2.x - dot_rv2 * s2.vx) / mu;
            double e2y = (term1_2 * s2.y - dot_rv2 * s2.vy) / mu;
            double e2z = (term1_2 * s2.z - dot_rv2 * s2.vz) / mu;

            double decc_x = e1x - e2x;
            double decc_y = e1y - e2y;
            double decc_z = e1z - e2z;
            double decc = Math.Sqrt(decc_x * decc_x + decc_y * decc_y + decc_z * decc_z);
            double dv_ecc = decc * Math.Max(vc1, vc2);

            return dv_hohmann + dv_plane + dv_ecc;
        }

        static void Main(string[] args)
        {
            string line1 = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(line1)) return;
            string[] parts = line1.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int N = int.Parse(parts[0], CultureInfo.InvariantCulture);
            double mu = double.Parse(parts[1], CultureInfo.InvariantCulture);

            string line2 = Console.ReadLine();
            parts = line2.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            State startOrbit = new State
            {
                id = -1,
                x = double.Parse(parts[0], CultureInfo.InvariantCulture),
                y = double.Parse(parts[1], CultureInfo.InvariantCulture),
                z = double.Parse(parts[2], CultureInfo.InvariantCulture),
                vx = double.Parse(parts[3], CultureInfo.InvariantCulture),
                vy = double.Parse(parts[4], CultureInfo.InvariantCulture),
                vz = double.Parse(parts[5], CultureInfo.InvariantCulture)
            };

            List<State> stations = new List<State>();
            for (int i = 0; i < N; i++)
            {
                string line = Console.ReadLine();
                parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                stations.Add(new State
                {
                    id = i,
                    x = double.Parse(parts[0], CultureInfo.InvariantCulture),
                    y = double.Parse(parts[1], CultureInfo.InvariantCulture),
                    z = double.Parse(parts[2], CultureInfo.InvariantCulture),
                    vx = double.Parse(parts[3], CultureInfo.InvariantCulture),
                    vy = double.Parse(parts[4], CultureInfo.InvariantCulture),
                    vz = double.Parse(parts[5], CultureInfo.InvariantCulture)
                });
            }

            int totalNodes = N + 1;
            double[,] cost = new double[totalNodes, totalNodes];
            State[] allNodes = new State[totalNodes];
            allNodes[0] = startOrbit;
            for (int i = 0; i < N; i++) allNodes[i + 1] = stations[i];

            for (int i = 0; i < totalNodes; i++)
            {
                for (int j = 0; j < totalNodes; j++)
                {
                    if (i == j) cost[i, j] = 0;
                    else cost[i, j] = CalculateCost(allNodes[i], allNodes[j], mu);
                }
            }

            SolveTSP(N, cost, out int[] bestPath, out double bestCost);

            for (int i = 0; i < N; i++)
            {
                Console.Write((bestPath[i] - 1) + (i == N - 1 ? "" : " "));
            }
            Console.WriteLine();
        }

        static void SolveTSP(int N, double[,] cost, out int[] bestPath, out double bestCost)
        {
            bestCost = double.MaxValue;
            bestPath = null;

            if (N <= 9)
            {
                int[] path = new int[N];
                for (int i = 0; i < N; i++) path[i] = i + 1;
                
                do
                {
                    double currentCost = EvaluateCost(path, cost);
                    if (currentCost < bestCost)
                    {
                        bestCost = currentCost;
                        bestPath = (int[])path.Clone();
                    }
                } while (NextPermutation(path));
                return;
            }

            Random rnd = new Random(42);
            Stopwatch sw = Stopwatch.StartNew();
            long timeLimitMs = N <= 20 ? 1000 : (N <= 50 ? 5000 : 20000); 
            
            int[] nnPath = new int[N];
            bool[] visited = new bool[N + 1];
            visited[0] = true;
            int current = 0;
            for (int i = 0; i < N; i++)
            {
                double minCost = double.MaxValue;
                int next = -1;
                for (int j = 1; j <= N; j++)
                {
                    if (!visited[j] && cost[current, j] < minCost)
                    {
                        minCost = cost[current, j];
                        next = j;
                    }
                }
                nnPath[i] = next;
                visited[next] = true;
                current = next;
            }
            
            TwoOpt(nnPath, cost);
            bestCost = EvaluateCost(nnPath, cost);
            bestPath = (int[])nnPath.Clone();
            
            while (sw.ElapsedMilliseconds < timeLimitMs)
            {
                int[] currentPath = GenerateRandomPath(N, rnd);
                TwoOpt(currentPath, cost);
                double cCost = EvaluateCost(currentPath, cost);
                
                if (cCost < bestCost)
                {
                    bestCost = cCost;
                    bestPath = (int[])currentPath.Clone();
                }
                
                double temp = 10.0;
                int saIters = 5000;
                double factor = Math.Pow(0.001 / 10.0, 1.0 / saIters);
                
                for (int iter = 0; iter < saIters; iter++)
                {
                    if (sw.ElapsedMilliseconds >= timeLimitMs) break;
                    
                    int i = rnd.Next(N);
                    int j = rnd.Next(N);
                    if (i == j) continue;
                    
                    int tempVal = currentPath[i];
                    currentPath[i] = currentPath[j];
                    currentPath[j] = tempVal;
                    
                    double newCost = EvaluateCost(currentPath, cost);
                    if (newCost < cCost || Math.Exp((cCost - newCost) / temp) > rnd.NextDouble())
                    {
                        cCost = newCost;
                        if (cCost < bestCost)
                        {
                            bestCost = cCost;
                            bestPath = (int[])currentPath.Clone();
                        }
                    }
                    else
                    {
                        tempVal = currentPath[i];
                        currentPath[i] = currentPath[j];
                        currentPath[j] = tempVal;
                    }
                    
                    if (Math.Abs(i - j) > 1)
                    {
                        int start = Math.Min(i, j);
                        int end = Math.Max(i, j);
                        ReverseSegment(currentPath, start, end);
                        newCost = EvaluateCost(currentPath, cost);
                        if (newCost < cCost || Math.Exp((cCost - newCost) / temp) > rnd.NextDouble())
                        {
                            cCost = newCost;
                            if (cCost < bestCost)
                            {
                                bestCost = cCost;
                                bestPath = (int[])currentPath.Clone();
                            }
                        }
                        else
                        {
                            ReverseSegment(currentPath, start, end);
                        }
                    }
                    
                    temp *= factor;
                }
                
                TwoOpt(currentPath, cost);
                cCost = EvaluateCost(currentPath, cost);
                if (cCost < bestCost)
                {
                    bestCost = cCost;
                    bestPath = (int[])currentPath.Clone();
                }
            }
        }

        static bool NextPermutation(int[] array)
        {
            int i = array.Length - 2;
            while (i >= 0 && array[i] >= array[i + 1]) i--;
            if (i < 0) return false;
            
            int j = array.Length - 1;
            while (array[j] <= array[i]) j--;
            
            int temp = array[i];
            array[i] = array[j];
            array[j] = temp;
            
            Array.Reverse(array, i + 1, array.Length - (i + 1));
            return true;
        }

        static double EvaluateCost(int[] path, double[,] cost)
        {
            double c = cost[0, path[0]];
            for (int i = 0; i < path.Length - 1; i++)
            {
                c += cost[path[i], path[i + 1]];
            }
            return c;
        }

        static void TwoOpt(int[] path, double[,] cost)
        {
            bool improved = true;
            while (improved)
            {
                improved = false;
                for (int i = 0; i < path.Length - 1; i++)
                {
                    for (int k = i + 1; k < path.Length; k++)
                    {
                        double costBefore = 0;
                        if (i == 0) costBefore += cost[0, path[0]];
                        else costBefore += cost[path[i - 1], path[i]];
                        
                        if (k < path.Length - 1) costBefore += cost[path[k], path[k + 1]];
                        
                        double costAfter = 0;
                        if (i == 0) costAfter += cost[0, path[k]];
                        else costAfter += cost[path[i - 1], path[k]];
                        
                        if (k < path.Length - 1) costAfter += cost[path[i], path[k + 1]];
                        
                        if (costAfter < costBefore - 1e-9)
                        {
                            ReverseSegment(path, i, k);
                            improved = true;
                        }
                    }
                }
            }
        }

        static void ReverseSegment(int[] path, int start, int end)
        {
            while (start < end)
            {
                int temp = path[start];
                path[start] = path[end];
                path[end] = temp;
                start++;
                end--;
            }
        }

        static int[] GenerateRandomPath(int n, Random rnd)
        {
            int[] path = new int[n];
            for (int i = 0; i < n; i++) path[i] = i + 1;
            
            for (int i = 0; i < n; i++)
            {
                int j = rnd.Next(i, n);
                int temp = path[i];
                path[i] = path[j];
                path[j] = temp;
            }
            return path;
        }
    }
}
