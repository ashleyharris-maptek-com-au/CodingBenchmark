using System;
using System.IO;
using System.Collections.Generic;
using System.Diagnostics;

class FastIO {
    private Stream stream;
    private byte[] buffer = new byte[1024 * 1024];
    private int head;
    private int tail;

    public FastIO(Stream stream) {
        this.stream = stream;
    }

    public int ReadInt() {
        int c = Read();
        while (c <= 32) {
            if (c == -1) return -1;
            c = Read();
        }
        int res = 0;
        while (c > 32) {
            res = res * 10 + (c - '0');
            c = Read();
        }
        return res;
    }

    private int Read() {
        if (head >= tail) {
            head = 0;
            tail = stream.Read(buffer, 0, buffer.Length);
            if (tail == 0) return -1;
        }
        return buffer[head++];
    }
}

class FastOutput {
    private Stream stream;
    private byte[] buffer = new byte[1024 * 1024];
    private int pos;

    public FastOutput(Stream stream) {
        this.stream = stream;
    }

    public void WriteInt(int v) {
        if (v == 0) {
            WriteChar('0');
            return;
        }
        int len = 0;
        int temp = v;
        while (temp > 0) {
            len++;
            temp /= 10;
        }
        if (pos + len > buffer.Length) Flush();
        int end = pos + len - 1;
        while (v > 0) {
            buffer[end--] = (byte)((v % 10) + '0');
            v /= 10;
        }
        pos += len;
    }

    public void WriteChar(char c) {
        if (pos == buffer.Length) Flush();
        buffer[pos++] = (byte)c;
    }

    public void Flush() {
        if (pos > 0) {
            stream.Write(buffer, 0, pos);
            pos = 0;
        }
    }
}

class FastFF {
    int[] tree;
    int[] bin_cap;
    public int[] head;
    public int[] next;
    public int[] itemIdx;
    public int num_bins;
    int stock_length;
    int N;

    public FastFF(int n, int stock) {
        N = n;
        stock_length = stock;
        head = new int[N];
        next = new int[N];
        itemIdx = new int[N];
        tree = new int[4 * Math.Max(1, N)];
        bin_cap = new int[N];
    }

    public void ClearAndBuild() {
        num_bins = 0;
        Array.Fill(head, -1);
        Array.Fill(bin_cap, stock_length);
        if (N > 0) BuildTree(1, 0, N - 1);
    }

    void BuildTree(int node, int l, int r) {
        tree[node] = stock_length;
        if (l == r) return;
        int mid = (l + r) / 2;
        BuildTree(2 * node, l, mid);
        BuildTree(2 * node + 1, mid + 1, r);
    }

    public void AddCut(int i, int length, int origIdx) {
        int b = FindTree(1, 0, N - 1, length);
        bin_cap[b] -= length;
        UpdateTree(1, 0, N - 1, b, bin_cap[b]);
        
        itemIdx[i] = origIdx;
        next[i] = head[b];
        head[b] = i;

        if (b >= num_bins) num_bins = b + 1;
    }

    void UpdateTree(int node, int l, int r, int idx, int val) {
        if (l == r) {
            tree[node] = val;
            return;
        }
        int mid = (l + r) / 2;
        if (idx <= mid) UpdateTree(2 * node, l, mid, idx, val);
        else UpdateTree(2 * node + 1, mid + 1, r, idx, val);
        tree[node] = Math.Max(tree[2 * node], tree[2 * node + 1]);
    }

    int FindTree(int node, int l, int r, int S) {
        if (l == r) return l;
        int mid = (l + r) / 2;
        if (tree[2 * node] >= S) return FindTree(2 * node, l, mid, S);
        return FindTree(2 * node + 1, mid + 1, r, S);
    }
}

class FastBFD {
    public int[] head;
    public int[] next;
    public int[] itemIdx;
    public int num_bins;
    int stock_length;
    int N;
    
    struct Bin : IComparable<Bin> {
        public int space;
        public int id;
        public int CompareTo(Bin other) {
            int cmp = space.CompareTo(other.space);
            if (cmp == 0) cmp = id.CompareTo(other.id);
            return cmp;
        }
    }

    public FastBFD(int n, int stock) {
        N = n;
        stock_length = stock;
        head = new int[N];
        next = new int[N];
        itemIdx = new int[N];
    }

    public void Solve(Program.Cut[] cuts) {
        num_bins = 0;
        Array.Fill(head, -1);
        SortedSet<Bin> binsSet = new SortedSet<Bin>();

        for (int i = 0; i < N; i++) {
            int len = cuts[i].length;
            Bin dummy = new Bin { space = len, id = -1 };
            var view = binsSet.GetViewBetween(dummy, new Bin { space = int.MaxValue, id = int.MaxValue });
            
            bool found = false;
            Bin bestBin = default(Bin);
            using (var enumerator = view.GetEnumerator()) {
                if (enumerator.MoveNext()) {
                    bestBin = enumerator.Current;
                    found = true;
                }
            }

            if (found) {
                binsSet.Remove(bestBin);
                bestBin.space -= len;
                
                itemIdx[i] = cuts[i].id;
                next[i] = head[bestBin.id];
                head[bestBin.id] = i;
                
                binsSet.Add(bestBin);
            } else {
                int b = num_bins++;
                Bin newBin = new Bin { space = stock_length - len, id = b };
                itemIdx[i] = cuts[i].id;
                next[i] = head[b];
                head[b] = i;
                binsSet.Add(newBin);
            }
        }
    }
}

class Program {
    public struct Cut : IComparable<Cut> {
        public int length;
        public int id;
        public int CompareTo(Cut other) {
            int cmp = other.length.CompareTo(this.length);
            if (cmp == 0) cmp = this.id.CompareTo(other.id);
            return cmp;
        }
    }

    static void Main() {
        using (Stream stdin = Console.OpenStandardInput())
        using (Stream stdout = Console.OpenStandardOutput()) {
            FastIO io = new FastIO(stdin);
            int N = io.ReadInt();
            if (N <= 0) return;
            int stock_length = io.ReadInt();

            Cut[] cuts = new Cut[N];
            long sum = 0;
            for (int i = 0; i < N; i++) {
                cuts[i].length = io.ReadInt();
                cuts[i].id = i;
                sum += cuts[i].length;
            }

            long lower_bound = (sum + stock_length - 1) / stock_length;
            Array.Sort(cuts);

            int best_bins = int.MaxValue;
            int[] best_head = new int[N];
            int[] best_next = new int[N];
            int[] best_itemIdx = new int[N];

            FastFF ff = new FastFF(N, stock_length);
            ff.ClearAndBuild();
            for (int i = 0; i < N; i++) {
                ff.AddCut(i, cuts[i].length, cuts[i].id);
            }
            
            if (ff.num_bins < best_bins) {
                best_bins = ff.num_bins;
                Array.Copy(ff.head, best_head, N);
                Array.Copy(ff.next, best_next, N);
                Array.Copy(ff.itemIdx, best_itemIdx, N);
            }

            if (best_bins == lower_bound) {
                PrintResult(stdout, best_bins, best_head, best_next, best_itemIdx);
                return;
            }

            FastBFD bfd = new FastBFD(N, stock_length);
            bfd.Solve(cuts);
            if (bfd.num_bins < best_bins) {
                best_bins = bfd.num_bins;
                Array.Copy(bfd.head, best_head, N);
                Array.Copy(bfd.next, best_next, N);
                Array.Copy(bfd.itemIdx, best_itemIdx, N);
            }

            if (best_bins == lower_bound) {
                PrintResult(stdout, best_bins, best_head, best_next, best_itemIdx);
                return;
            }

            Stopwatch sw = Stopwatch.StartNew();
            Random rng = new Random(42);
            Cut[] perturbed = new Cut[N];
            Array.Copy(cuts, perturbed, N);

            while (sw.Elapsed.TotalSeconds < 2.0) {
                for (int k = 0; k < N / 10 + 1; k++) {
                    int idx1 = rng.Next(N);
                    int idx2 = idx1 + rng.Next(-5, 6);
                    if (idx2 >= 0 && idx2 < N) {
                        Cut temp = perturbed[idx1];
                        perturbed[idx1] = perturbed[idx2];
                        perturbed[idx2] = temp;
                    }
                }
                
                ff.ClearAndBuild();
                for (int i = 0; i < N; i++) {
                    ff.AddCut(i, perturbed[i].length, perturbed[i].id);
                }

                if (ff.num_bins < best_bins) {
                    best_bins = ff.num_bins;
                    Array.Copy(ff.head, best_head, N);
                    Array.Copy(ff.next, best_next, N);
                    Array.Copy(ff.itemIdx, best_itemIdx, N);
                    if (best_bins == lower_bound) break;
                }
                
                Array.Copy(cuts, perturbed, N);
            }

            PrintResult(stdout, best_bins, best_head, best_next, best_itemIdx);
        }
    }

    static void PrintResult(Stream stdout, int num_bins, int[] head, int[] next, int[] itemIdx) {
        FastOutput fo = new FastOutput(stdout);
        fo.WriteInt(num_bins);
        fo.WriteChar('\n');
        for (int b = 0; b < num_bins; b++) {
            int curr = head[b];
            bool first = true;
            while (curr != -1) {
                if (!first) fo.WriteChar(' ');
                fo.WriteInt(itemIdx[curr]);
                first = false;
                curr = next[curr];
            }
            fo.WriteChar('\n');
        }
        fo.Flush();
    }
}
