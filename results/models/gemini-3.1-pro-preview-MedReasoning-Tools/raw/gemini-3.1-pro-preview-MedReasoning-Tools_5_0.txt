use std::io::{self, BufRead};

fn main() {
    std::thread::Builder::new()
        .stack_size(32 * 1024 * 1024)
        .spawn(solve)
        .unwrap()
        .join()
        .unwrap();
}

fn solve() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => return,
    };
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    if parts.len() < 4 { return; }
    let w: usize = parts[0].parse().unwrap();
    let h: usize = parts[1].parse().unwrap();
    let obs_count: usize = parts[2].parse().unwrap();
    let req_cycle: u8 = parts[3].parse().unwrap();
    let require_cycle = req_cycle != 0;

    let mut unvisited = vec![true; w * h];
    for _ in 0..obs_count {
        if let Some(Ok(line)) = lines.next() {
            let p: Vec<usize> = line.split_whitespace().map(|x| x.parse().unwrap()).collect();
            if p.len() >= 2 {
                unvisited[p[1] * w + p[0]] = false;
            }
        }
    }

    let start = 0; // (0,0)
    if !unvisited[start] {
        println!("0");
        return;
    }

    unvisited[start] = false;
    let mut unvisited_count = 0;
    let mut even_count = 0;
    let mut odd_count = 0;
    for y in 0..h {
        for x in 0..w {
            if unvisited[y * w + x] {
                unvisited_count += 1;
                if (x + y) % 2 == 0 {
                    even_count += 1;
                } else {
                    odd_count += 1;
                }
            }
        }
    }

    if require_cycle {
        if unvisited_count % 2 == 0 || odd_count != even_count + 1 {
            println!("0");
            return;
        }
    } else {
        if unvisited_count % 2 != 0 {
            if odd_count != even_count + 1 {
                println!("0");
                return;
            }
        } else {
            if odd_count != even_count {
                println!("0");
                return;
            }
        }
    }

    let mut solver = Solver {
        w,
        h,
        require_cycle,
        unvisited,
        unvisited_count,
        path: vec![start],
        bfs_visited: vec![0; w * h],
        bfs_generation: 0,
        deg: vec![0; w * h],
        bfs_queue: Vec::with_capacity(w * h),
        ones: Vec::new(),
        start,
        first_unvisited: 0,
        nodes: w * h,
    };

    if solver.dfs(start) {
        println!("{}", solver.path.len());
        for &p in &solver.path {
            println!("{} {}", p % w, p / w);
        }
    } else {
        println!("0");
    }
}

struct Solver {
    w: usize,
    h: usize,
    require_cycle: bool,
    unvisited: Vec<bool>,
    unvisited_count: usize,
    path: Vec<usize>,
    bfs_visited: Vec<u32>,
    bfs_generation: u32,
    deg: Vec<u8>,
    bfs_queue: Vec<usize>,
    ones: Vec<usize>,
    start: usize,
    first_unvisited: usize,
    nodes: usize,
}

impl Solver {
    fn is_adj(&self, a: usize, b: usize) -> bool {
        if a > b {
            (a - b == 1 && a % self.w != 0) || a - b == self.w
        } else {
            (b - a == 1 && b % self.w != 0) || b - a == self.w
        }
    }

    fn neighbors(&self, idx: usize) -> Neighbors {
        Neighbors {
            idx,
            x: idx % self.w,
            y: idx / self.w,
            w: self.w,
            h: self.h,
            state: 0,
        }
    }

    fn dfs(&mut self, u: usize) -> bool {
        if self.unvisited_count == 0 {
            if self.require_cycle {
                return self.is_adj(u, self.start);
            }
            return true;
        }

        if self.unvisited_count == 1 {
            let mut found = None;
            for nxt in self.neighbors(u) {
                if self.unvisited[nxt] {
                    found = Some(nxt);
                    break;
                }
            }
            if let Some(nxt) = found {
                if self.require_cycle && !self.is_adj(nxt, self.start) {
                    return false;
                }
                self.unvisited[nxt] = false;
                self.unvisited_count -= 1;
                self.path.push(nxt);
                let res = self.dfs(nxt);
                if res { return true; }
                self.path.pop();
                self.unvisited_count += 1;
                self.unvisited[nxt] = true;
            }
            return false;
        }

        self.bfs_generation += 1;
        let gen = self.bfs_generation;

        let old_first = self.first_unvisited;
        while self.first_unvisited < self.nodes && !self.unvisited[self.first_unvisited] {
            self.first_unvisited += 1;
        }
        let first = self.first_unvisited;

        self.bfs_queue.clear();
        self.bfs_queue.push(first);
        self.bfs_visited[first] = gen;
        let mut bfs_count = 1;

        let mut head = 0;
        while head < self.bfs_queue.len() {
            let curr = self.bfs_queue[head];
            head += 1;

            let mut d = 0;
            for nxt in self.neighbors(curr) {
                if self.unvisited[nxt] {
                    d += 1;
                    if self.bfs_visited[nxt] != gen {
                        self.bfs_visited[nxt] = gen;
                        bfs_count += 1;
                        self.bfs_queue.push(nxt);
                    }
                }
            }
            self.deg[curr] = d;
        }

        if bfs_count != self.unvisited_count {
            self.first_unvisited = old_first;
            return false;
        }

        self.ones.clear();
        for i in 0..self.bfs_queue.len() {
            let node = self.bfs_queue[i];
            if self.deg[node] == 1 {
                self.ones.push(node);
            }
        }

        if self.ones.len() > 2 {
            self.first_unvisited = old_first;
            return false;
        }

        let mut not_in_n = 0;
        for &x in &self.ones {
            if !self.is_adj(x, u) {
                not_in_n += 1;
            }
        }

        if not_in_n > 1 {
            self.first_unvisited = old_first;
            return false;
        }

        let mut forced = None;
        if not_in_n == 1 && self.ones.len() == 2 {
            for &x in &self.ones {
                if self.is_adj(x, u) {
                    forced = Some(x);
                }
            }
        }

        if forced.is_none() {
            for &x in &self.ones {
                if self.is_adj(x, u) {
                    if self.require_cycle && !self.is_adj(x, self.start) {
                        forced = Some(x);
                        break;
                    }
                }
            }
        }

        let mut candidates = [0; 4];
        let mut cand_count = 0;

        if let Some(f) = forced {
            candidates[0] = f;
            cand_count = 1;
        } else {
            for nxt in self.neighbors(u) {
                if self.unvisited[nxt] {
                    candidates[cand_count] = nxt;
                    cand_count += 1;
                }
            }
            candidates[..cand_count].sort_by_key(|&c| self.deg[c]);
        }

        for i in 0..cand_count {
            let c = candidates[i];
            self.unvisited[c] = false;
            self.unvisited_count -= 1;
            self.path.push(c);
            if self.dfs(c) {
                return true;
            }
            self.path.pop();
            self.unvisited_count += 1;
            self.unvisited[c] = true;
        }

        self.first_unvisited = old_first;
        false
    }
}

struct Neighbors {
    idx: usize,
    x: usize,
    y: usize,
    w: usize,
    h: usize,
    state: u8,
}

impl Iterator for Neighbors {
    type Item = usize;
    fn next(&mut self) -> Option<usize> {
        while self.state < 4 {
            let s = self.state;
            self.state += 1;
            match s {
                0 => if self.x > 0 { return Some(self.idx - 1); },
                1 => if self.x + 1 < self.w { return Some(self.idx + 1); },
                2 => if self.y > 0 { return Some(self.idx - self.w); },
                3 => if self.y + 1 < self.h { return Some(self.idx + self.w); },
                _ => unreachable!(),
            }
        }
        None
    }
}
