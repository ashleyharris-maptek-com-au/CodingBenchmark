using System;
using System.IO;
using System.Collections.Generic;

class Edge {
    public int to;
    public Edge rev;
    public bool used;
}

struct OrigEdge {
    public int to;
    public long w;
}

class Program {
    static void AddUndirectedEdge(int u, int v, List<Edge>[] adj) {
        Edge e1 = new Edge { to = v };
        Edge e2 = new Edge { to = u };
        e1.rev = e2;
        e2.rev = e1;
        adj[u].Add(e1);
        adj[v].Add(e2);
    }

    static void AddShortestPath(int u, int target, List<OrigEdge>[] origAdj, int N, List<Edge>[] adj) {
        long INF = 1_000_000_000_000_000_000L;
        long[] dist = new long[N];
        int[] parent = new int[N];
        Array.Fill(dist, INF);
        Array.Fill(parent, -1);
        dist[u] = 0;
        
        PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
        pq.Enqueue(u, 0);

        while(pq.Count > 0) {
            pq.TryDequeue(out int curr, out long d);
            if (curr == target) break;
            if (d > dist[curr]) continue;
            
            foreach(var e in origAdj[curr]) {
                if (dist[curr] + e.w < dist[e.to]) {
                    dist[e.to] = dist[curr] + e.w;
                    parent[e.to] = curr;
                    pq.Enqueue(e.to, dist[e.to]);
                }
            }
        }

        int temp = target;
        while(temp != u) {
            int p = parent[temp];
            AddUndirectedEdge(p, temp, adj);
            temp = p;
        }
    }

    static void Main() {
        int N = Scanner.NextInt();
        if (N == -1) return; 
        int M = Scanner.NextInt();

        List<OrigEdge>[] origAdj = new List<OrigEdge>[N];
        List<Edge>[] adj = new List<Edge>[N];
        for (int i = 0; i < N; i++) {
            origAdj[i] = new List<OrigEdge>();
            adj[i] = new List<Edge>();
        }

        int[] degree = new int[N];
        var edgesList = new List<(int u, int v, long w)>();

        for (int i = 0; i < M; i++) {
            int u = Scanner.NextInt();
            int v = Scanner.NextInt();
            long w = Scanner.NextLong();
            origAdj[u].Add(new OrigEdge { to = v, w = w });
            origAdj[v].Add(new OrigEdge { to = u, w = w });
            AddUndirectedEdge(u, v, adj);
            degree[u]++;
            degree[v]++;
            if (u < v) edgesList.Add((u, v, w));
            else edgesList.Add((v, u, w));
        }

        List<int> oddNodes = new List<int>();
        for (int i = 0; i < N; i++) {
            if (degree[i] % 2 != 0) {
                oddNodes.Add(i);
            }
        }

        int O = oddNodes.Count;
        long INF = 1_000_000_000_000_000_000L;

        if (O > 0) {
            long ops = (long)O * M;
            if (O <= 20) {
                long[,] distODD = new long[O, O];
                for (int i = 0; i < O; i++) {
                    int src = oddNodes[i];
                    long[] dist = new long[N];
                    Array.Fill(dist, INF);
                    dist[src] = 0;
                    PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
                    pq.Enqueue(src, 0);
                    while (pq.Count > 0) {
                        pq.TryDequeue(out int curr, out long d);
                        if (d > dist[curr]) continue;
                        foreach (var e in origAdj[curr]) {
                            if (dist[curr] + e.w < dist[e.to]) {
                                dist[e.to] = dist[curr] + e.w;
                                pq.Enqueue(e.to, dist[e.to]);
                            }
                        }
                    }
                    for (int j = 0; j < O; j++) {
                        distODD[i, j] = dist[oddNodes[j]];
                    }
                }

                long[] dp = new long[1 << O];
                int[] pair_with = new int[1 << O];
                Array.Fill(dp, INF);
                dp[0] = 0;

                for (int mask = 0; mask < (1 << O); mask++) {
                    if (dp[mask] >= INF) continue;
                    int i = 0;
                    while ((mask & (1 << i)) != 0) i++;
                    if (i >= O) continue;

                    for (int j = i + 1; j < O; j++) {
                        if ((mask & (1 << j)) == 0) {
                            if (distODD[i, j] >= INF) continue;
                            int nextMask = mask | (1 << i) | (1 << j);
                            long newCost = dp[mask] + distODD[i, j];
                            if (newCost < dp[nextMask]) {
                                dp[nextMask] = newCost;
                                pair_with[nextMask] = (i << 16) | j;
                            }
                        }
                    }
                }

                int currMask = (1 << O) - 1;
                while (currMask > 0) {
                    int p = pair_with[currMask];
                    int i = p >> 16;
                    int j = p & 0xFFFF;
                    AddShortestPath(oddNodes[i], oddNodes[j], origAdj, N, adj);
                    currMask ^= (1 << i) | (1 << j);
                }

            } else if (ops <= 50_000_000) {
                long[,] distODD = new long[O, O];
                for (int i = 0; i < O; i++) {
                    int src = oddNodes[i];
                    long[] dist = new long[N];
                    Array.Fill(dist, INF);
                    dist[src] = 0;
                    PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
                    pq.Enqueue(src, 0);
                    while (pq.Count > 0) {
                        pq.TryDequeue(out int curr, out long d);
                        if (d > dist[curr]) continue;
                        foreach (var e in origAdj[curr]) {
                            if (dist[curr] + e.w < dist[e.to]) {
                                dist[e.to] = dist[curr] + e.w;
                                pq.Enqueue(e.to, dist[e.to]);
                            }
                        }
                    }
                    for (int j = 0; j < O; j++) {
                        distODD[i, j] = dist[oddNodes[j]];
                    }
                }

                bool[] matched = new bool[O];
                List<int> currentMatching = new List<int>();
                var pairs = new List<(int u, int v, long d)>();
                for (int i = 0; i < O; i++)
                    for (int j = i + 1; j < O; j++) {
                        if (distODD[i, j] < INF)
                            pairs.Add((i, j, distODD[i, j]));
                    }
                pairs.Sort((a, b) => a.d.CompareTo(b.d));

                foreach (var p in pairs) {
                    if (!matched[p.u] && !matched[p.v]) {
                        matched[p.u] = true;
                        matched[p.v] = true;
                        currentMatching.Add(p.u);
                        currentMatching.Add(p.v);
                    }
                }

                bool improved = true;
                while (improved) {
                    improved = false;
                    for (int i = 0; i < currentMatching.Count; i += 2) {
                        for (int j = i + 2; j < currentMatching.Count; j += 2) {
                            int u1 = currentMatching[i];
                            int v1 = currentMatching[i + 1];
                            int u2 = currentMatching[j];
                            int v2 = currentMatching[j + 1];

                            long d1 = (distODD[u1, v1] >= INF || distODD[u2, v2] >= INF) ? INF : distODD[u1, v1] + distODD[u2, v2];
                            long d2 = (distODD[u1, u2] >= INF || distODD[v1, v2] >= INF) ? INF : distODD[u1, u2] + distODD[v1, v2];
                            long d3 = (distODD[u1, v2] >= INF || distODD[v1, u2] >= INF) ? INF : distODD[u1, v2] + distODD[v1, u2];

                            if (d2 < d1 && d2 <= d3 && d2 < INF) {
                                currentMatching[i + 1] = u2;
                                currentMatching[j] = v1;
                                improved = true;
                            } else if (d3 < d1 && d3 < d2 && d3 < INF) {
                                currentMatching[i + 1] = v2;
                                currentMatching[j + 1] = v1;
                                improved = true;
                            }
                        }
                    }
                }

                for (int i = 0; i < currentMatching.Count; i += 2) {
                    AddShortestPath(oddNodes[currentMatching[i]], oddNodes[currentMatching[i + 1]], origAdj, N, adj);
                }

            } else {
                edgesList.Sort((a, b) => a.w.CompareTo(b.w));
                int[] parent = new int[N];
                for (int i = 0; i < N; i++) parent[i] = i;
                int Find(int i) {
                    if (parent[i] == i) return i;
                    return parent[i] = Find(parent[i]);
                }
                void Union(int i, int j) {
                    parent[Find(i)] = Find(j);
                }

                var mst = new List<int>[N];
                for (int i = 0; i < N; i++) mst[i] = new List<int>();

                foreach (var e in edgesList) {
                    if (Find(e.u) != Find(e.v)) {
                        Union(e.u, e.v);
                        mst[e.u].Add(e.v);
                        mst[e.v].Add(e.u);
                    }
                }

                int[] subOdd = new int[N];
                for (int i = 0; i < N; i++) {
                    if (degree[i] % 2 != 0) subOdd[i] = 1;
                }

                bool[] visitedMST = new bool[N];

                for (int i = 0; i < N; i++) {
                    if (!visitedMST[i]) {
                        var stack = new Stack<(int u, int p)>();
                        var postOrder = new List<(int u, int p)>();

                        stack.Push((i, -1));
                        while (stack.Count > 0) {
                            var curr = stack.Pop();
                            postOrder.Add(curr);
                            visitedMST[curr.u] = true;
                            foreach (var v in mst[curr.u]) {
                                if (v != curr.p) {
                                    stack.Push((v, curr.u));
                                }
                            }
                        }

                        postOrder.Reverse();

                        foreach (var curr in postOrder) {
                            if (curr.p != -1) {
                                if (subOdd[curr.u] == 1) {
                                    AddUndirectedEdge(curr.u, curr.p, adj);
                                    subOdd[curr.p] ^= 1;
                                }
                            }
                        }
                    }
                }
            }
        }

        List<int> circuit = new List<int>();
        Stack<int> circuitStack = new Stack<int>();
        int[] head = new int[N];

        circuitStack.Push(0);
        while (circuitStack.Count > 0) {
            int u = circuitStack.Peek();
            bool found = false;
            while (head[u] < adj[u].Count) {
                Edge e = adj[u][head[u]++];
                if (!e.used) {
                    e.used = true;
                    e.rev.used = true;
                    circuitStack.Push(e.to);
                    found = true;
                    break;
                }
            }
            if (!found) {
                circuit.Add(u);
                circuitStack.Pop();
            }
        }

        circuit.Reverse();

        using var writer = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 65536);
        foreach (var node in circuit) {
            writer.Write(node + " ");
        }
    }
}

static class Scanner {
    static Stream stream = Console.OpenStandardInput();
    static byte[] buffer = new byte[65536];
    static int head = 0;
    static int tail = 0;

    public static int NextInt() {
        int c = Read();
        while (c < '0' || c > '9') {
            if (c == '-') {
                int c2 = Read();
                int res2 = 0;
                while (c2 >= '0' && c2 <= '9') {
                    res2 = res2 * 10 + (c2 - '0');
                    c2 = Read();
                }
                return -res2;
            }
            if (c == -1) return -1;
            c = Read();
        }
        int res = 0;
        while (c >= '0' && c <= '9') {
            res = res * 10 + (c - '0');
            c = Read();
        }
        return res;
    }

    public static long NextLong() {
        int c = Read();
        while (c < '0' || c > '9') {
            if (c == '-') {
                int c2 = Read();
                long res2 = 0;
                while (c2 >= '0' && c2 <= '9') {
                    res2 = res2 * 10 + (c2 - '0');
                    c2 = Read();
                }
                return -res2;
            }
            if (c == -1) return -1;
            c = Read();
        }
        long res = 0;
        while (c >= '0' && c <= '9') {
            res = res * 10 + (c - '0');
            c = Read();
        }
        return res;
    }

    static int Read() {
        if (head >= tail) {
            head = 0;
            tail = stream.Read(buffer, 0, buffer.Length);
            if (tail == 0) return -1;
        }
        return buffer[head++];
    }
}
