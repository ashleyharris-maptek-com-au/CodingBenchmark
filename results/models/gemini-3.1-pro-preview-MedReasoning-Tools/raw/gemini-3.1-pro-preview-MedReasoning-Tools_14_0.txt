using System;
using System.IO;
using System.Collections.Generic;

class MinesweeperSolver
{
    static int H, W;
    static byte[] grid;
    static List<int> safeCells = new List<int>();
    static Queue<int> simpleQueue = new Queue<int>();
    static bool[] inQueue;
    static bool[] isActiveNumber;
    static List<int> activeNumbers = new List<int>();

    static int[] dx = { -1, 0, 1, -1, 1, -1, 0, 1 };
    static int[] dy = { -1, -1, -1, 0, 0, 1, 1, 1 };

    static void Main(string[] args)
    {
        using var stream = Console.OpenStandardInput();
        using var sr = new StreamReader(stream, System.Text.Encoding.ASCII, false, 65536);
        
        string firstLine = sr.ReadLine();
        while (string.IsNullOrWhiteSpace(firstLine))
        {
            firstLine = sr.ReadLine();
            if (firstLine == null) return;
        }

        string[] parts = firstLine.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        H = int.Parse(parts[0]);
        W = int.Parse(parts[1]);

        grid = new byte[H * W];
        inQueue = new bool[H * W];
        isActiveNumber = new bool[H * W];

        int idx = 0;
        for (int i = 0; i < H; i++)
        {
            string line = sr.ReadLine();
            while (line != null && line.Length == 0)
            {
                line = sr.ReadLine();
            }
            if (line == null) break;

            for (int j = 0; j < W; j++)
            {
                char c = j < line.Length ? line[j] : ' ';
                if (c == ' ') grid[idx++] = 9;
                else if (c == '*') grid[idx++] = 10;
                else if (c >= '0' && c <= '8') grid[idx++] = (byte)(c - '0');
                else grid[idx++] = 9; 
            }
        }

        for (int y = 0; y < H; y++)
        {
            for (int x = 0; x < W; x++)
            {
                int id = y * W + x;
                if (grid[id] <= 8)
                {
                    if (HasUnknown(x, y))
                    {
                        simpleQueue.Enqueue(id);
                        inQueue[id] = true;
                        isActiveNumber[id] = true;
                        activeNumbers.Add(id);
                    }
                }
            }
        }

        while (true)
        {
            if (ProcessSimpleQueue()) continue;
            if (ProcessSubsetRule()) continue;
            if (ProcessCSP()) continue;
            break;
        }

        using var stdout = Console.OpenStandardOutput();
        using var sw = new StreamWriter(stdout, System.Text.Encoding.ASCII, 65536);
        sw.WriteLine(safeCells.Count);
        foreach (int id in safeCells)
        {
            sw.WriteLine($"{id % W} {id / W}");
        }
    }

    static bool ProcessSimpleQueue()
    {
        bool progress = false;
        while (simpleQueue.Count > 0)
        {
            int id = simpleQueue.Dequeue();
            inQueue[id] = false;

            if (grid[id] > 8) continue;

            int x = id % W;
            int y = id / W;
            int req = grid[id];

            int unkCount = 0;
            int mineCount = 0;

            for (int i = 0; i < 8; i++)
            {
                int nx = x + dx[i];
                int ny = y + dy[i];
                if (nx >= 0 && nx < W && ny >= 0 && ny < H)
                {
                    byte val = grid[ny * W + nx];
                    if (val == 9) unkCount++;
                    else if (val == 10 || val == 12) mineCount++;
                }
            }

            if (unkCount == 0) continue;

            if (mineCount == req)
            {
                for (int i = 0; i < 8; i++)
                {
                    int nx = x + dx[i];
                    int ny = y + dy[i];
                    if (nx >= 0 && nx < W && ny >= 0 && ny < H)
                    {
                        int nid = ny * W + nx;
                        if (grid[nid] == 9)
                        {
                            grid[nid] = 11;
                            safeCells.Add(nid);
                            progress = true;
                            WakeUpNeighbors(nx, ny);
                        }
                    }
                }
            }
            else if (mineCount + unkCount == req)
            {
                for (int i = 0; i < 8; i++)
                {
                    int nx = x + dx[i];
                    int ny = y + dy[i];
                    if (nx >= 0 && nx < W && ny >= 0 && ny < H)
                    {
                        int nid = ny * W + nx;
                        if (grid[nid] == 9)
                        {
                            grid[nid] = 12;
                            progress = true;
                            WakeUpNeighbors(nx, ny);
                        }
                    }
                }
            }
        }
        return progress;
    }

    static void WakeUpNeighbors(int x, int y)
    {
        for (int i = 0; i < 8; i++)
        {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if (nx >= 0 && nx < W && ny >= 0 && ny < H)
            {
                int nid = ny * W + nx;
                if (grid[nid] <= 8)
                {
                    if (!inQueue[nid])
                    {
                        simpleQueue.Enqueue(nid);
                        inQueue[nid] = true;
                    }
                    if (!isActiveNumber[nid])
                    {
                        isActiveNumber[nid] = true;
                        activeNumbers.Add(nid);
                    }
                }
            }
        }
    }

    static bool HasUnknown(int x, int y)
    {
        for (int i = 0; i < 8; i++)
        {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if (nx >= 0 && nx < W && ny >= 0 && ny < H)
            {
                if (grid[ny * W + nx] == 9) return true;
            }
        }
        return false;
    }

    static List<int> GetUnknowns(int x, int y, out int req)
    {
        List<int> unk = new List<int>(8);
        int mines = 0;
        for (int i = 0; i < 8; i++)
        {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if (nx >= 0 && nx < W && ny >= 0 && ny < H)
            {
                int nid = ny * W + nx;
                if (grid[nid] == 9) unk.Add(nid);
                else if (grid[nid] == 10 || grid[nid] == 12) mines++;
            }
        }
        req = grid[y * W + x] - mines;
        return unk;
    }

    static bool ProcessSubsetRule()
    {
        for (int i = 0; i < activeNumbers.Count; i++)
        {
            int id = activeNumbers[i];
            if (!HasUnknown(id % W, id / W))
            {
                isActiveNumber[id] = false;
                activeNumbers[i] = activeNumbers[activeNumbers.Count - 1];
                activeNumbers.RemoveAt(activeNumbers.Count - 1);
                i--;
            }
        }

        foreach (int idA in activeNumbers)
        {
            int ax = idA % W;
            int ay = idA / W;

            List<int> unkA = GetUnknowns(ax, ay, out int reqA);
            if (unkA.Count == 0) continue;

            for (int dy = -2; dy <= 2; dy++)
            {
                for (int dx = -2; dx <= 2; dx++)
                {
                    if (dx == 0 && dy == 0) continue;
                    int bx = ax + dx;
                    int by = ay + dy;
                    if (bx >= 0 && bx < W && by >= 0 && by < H)
                    {
                        int idB = by * W + bx;
                        if (isActiveNumber[idB])
                        {
                            List<int> unkB = GetUnknowns(bx, by, out int reqB);
                            if (unkB.Count > unkA.Count)
                            {
                                bool isSubset = true;
                                foreach (int u in unkA)
                                {
                                    if (!unkB.Contains(u))
                                    {
                                        isSubset = false;
                                        break;
                                    }
                                }

                                if (isSubset)
                                {
                                    int diffReq = reqB - reqA;
                                    List<int> diffCells = new List<int>();
                                    foreach (int u in unkB)
                                    {
                                        if (!unkA.Contains(u)) diffCells.Add(u);
                                    }

                                    if (diffReq == 0)
                                    {
                                        foreach (int u in diffCells)
                                        {
                                            grid[u] = 11;
                                            safeCells.Add(u);
                                            WakeUpNeighbors(u % W, u / W);
                                        }
                                        return true;
                                    }
                                    else if (diffReq == diffCells.Count)
                                    {
                                        foreach (int u in diffCells)
                                        {
                                            grid[u] = 12;
                                            WakeUpNeighbors(u % W, u / W);
                                        }
                                        return true;
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
        return false;
    }

    class CspConstraint
    {
        public int RequiredMines;
        public int CurrentMines;
        public int UnassignedVars;
        public List<int> VarIndices = new List<int>();
    }

    class CspVar
    {
        public List<CspConstraint> Constraints = new List<CspConstraint>();
    }

    static bool ProcessCSP()
    {
        HashSet<int> visitedNumbers = new HashSet<int>();
        HashSet<int> visitedUnknowns = new HashSet<int>();

        bool progress = false;

        for (int i = 0; i < activeNumbers.Count; i++)
        {
            int id = activeNumbers[i];
            if (!HasUnknown(id % W, id / W))
            {
                isActiveNumber[id] = false;
                activeNumbers[i] = activeNumbers[activeNumbers.Count - 1];
                activeNumbers.RemoveAt(activeNumbers.Count - 1);
                i--;
            }
        }

        foreach (int startId in activeNumbers)
        {
            if (visitedNumbers.Contains(startId)) continue;

            List<int> compNumbers = new List<int>();
            List<int> compUnknowns = new List<int>();

            Queue<int> q = new Queue<int>();
            q.Enqueue(startId);
            visitedNumbers.Add(startId);

            while (q.Count > 0)
            {
                int curr = q.Dequeue();
                compNumbers.Add(curr);

                int cx = curr % W;
                int cy = curr / W;
                List<int> unks = GetUnknowns(cx, cy, out _);

                foreach (int u in unks)
                {
                    if (!visitedUnknowns.Contains(u))
                    {
                        visitedUnknowns.Add(u);
                        compUnknowns.Add(u);

                        int ux = u % W;
                        int uy = u / W;
                        for (int i = 0; i < 8; i++)
                        {
                            int nx = ux + dx[i];
                            int ny = uy + dy[i];
                            if (nx >= 0 && nx < W && ny >= 0 && ny < H)
                            {
                                int nid = ny * W + nx;
                                if (isActiveNumber[nid] && !visitedNumbers.Contains(nid))
                                {
                                    visitedNumbers.Add(nid);
                                    q.Enqueue(nid);
                                }
                            }
                        }
                    }
                }
            }

            if (compUnknowns.Count > 0 && compUnknowns.Count <= 150)
            {
                if (SolveComponent(compUnknowns, compNumbers))
                {
                    progress = true;
                }
            }
        }

        return progress;
    }

    static bool SolveComponent(List<int> vars, List<int> numbers)
    {
        int nVars = vars.Count;
        CspVar[] cvars = new CspVar[nVars];
        for (int i = 0; i < nVars; i++) cvars[i] = new CspVar();

        CspConstraint[] cconstrs = new CspConstraint[numbers.Count];

        Dictionary<int, int> varIndexMap = new Dictionary<int, int>();
        for (int i = 0; i < nVars; i++) varIndexMap[vars[i]] = i;

        for (int i = 0; i < numbers.Count; i++)
        {
            int nid = numbers[i];
            List<int> unks = GetUnknowns(nid % W, nid / W, out int req);

            CspConstraint c = new CspConstraint
            {
                RequiredMines = req,
                CurrentMines = 0,
                UnassignedVars = unks.Count
            };

            foreach (int u in unks)
            {
                if (varIndexMap.TryGetValue(u, out int vIdx))
                {
                    c.VarIndices.Add(vIdx);
                    cvars[vIdx].Constraints.Add(c);
                }
            }
            cconstrs[i] = c;
        }

        foreach (var c in cconstrs)
        {
            if (c.RequiredMines < 0 || c.RequiredMines > c.UnassignedVars) return false;
        }

        long totalValid = 0;
        long[] mineCount = new long[nVars];
        int recursiveCalls = 0;
        bool aborted = false;

        void Backtrack(int vIdx)
        {
            if (aborted) return;
            recursiveCalls++;
            if (recursiveCalls > 500000)
            {
                aborted = true;
                return;
            }

            if (vIdx == nVars)
            {
                totalValid++;
                return;
            }

            CspVar v = cvars[vIdx];

            bool canBeSafe = true;
            foreach (var c in v.Constraints)
            {
                if (c.CurrentMines + (c.UnassignedVars - 1) < c.RequiredMines)
                {
                    canBeSafe = false;
                    break;
                }
            }

            if (canBeSafe)
            {
                foreach (var c in v.Constraints) c.UnassignedVars--;
                Backtrack(vIdx + 1);
                foreach (var c in v.Constraints) c.UnassignedVars++;
            }

            bool canBeMine = true;
            foreach (var c in v.Constraints)
            {
                if (c.CurrentMines + 1 > c.RequiredMines)
                {
                    canBeMine = false;
                    break;
                }
            }

            if (canBeMine)
            {
                foreach (var c in v.Constraints)
                {
                    c.UnassignedVars--;
                    c.CurrentMines++;
                }

                long before = totalValid;
                Backtrack(vIdx + 1);
                long after = totalValid;
                mineCount[vIdx] += (after - before);

                foreach (var c in v.Constraints)
                {
                    c.UnassignedVars++;
                    c.CurrentMines--;
                }
            }
        }

        Backtrack(0);

        if (aborted || totalValid == 0) return false;

        bool progress = false;
        for (int i = 0; i < nVars; i++)
        {
            if (mineCount[i] == 0)
            {
                grid[vars[i]] = 11;
                safeCells.Add(vars[i]);
                WakeUpNeighbors(vars[i] % W, vars[i] / W);
                progress = true;
            }
            else if (mineCount[i] == totalValid)
            {
                grid[vars[i]] = 12;
                WakeUpNeighbors(vars[i] % W, vars[i] / W);
                progress = true;
            }
        }

        return progress;
    }
}
