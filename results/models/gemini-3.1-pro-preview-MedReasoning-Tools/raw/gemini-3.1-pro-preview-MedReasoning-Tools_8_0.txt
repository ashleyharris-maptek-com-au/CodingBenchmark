use std::io::{BufRead, Read, Write};
use std::collections::VecDeque;

fn main() {
    let stdin = std::io::stdin();
    let mut handle = stdin.lock();
    
    let mut first_line = String::new();
    if handle.read_line(&mut first_line).unwrap_or(0) == 0 {
        println!("0");
        return;
    }
    
    let first_line = first_line.trim();
    if first_line.is_empty() {
        println!("0");
        return;
    }
    let mut parts = first_line.split_whitespace();
    let h_str = parts.next();
    let w_str = parts.next();
    
    if h_str.is_none() || w_str.is_none() {
        println!("0");
        return;
    }
    
    let h: usize = h_str.unwrap().parse().unwrap_or(0);
    let w: usize = w_str.unwrap().parse().unwrap_or(0);
    
    if h == 0 || w == 0 {
        println!("0");
        return;
    }
    
    let expected_capacity = h * (w + 2) + 128;
    let mut buf = Vec::with_capacity(expected_capacity);
    handle.read_to_end(&mut buf).unwrap();
    
    if buf.is_empty() {
        println!("0");
        return;
    }

    let mut row_stride = w;
    if w < buf.len() {
        if buf[w] == b'\r' {
            row_stride = w + 2;
        } else if buf[w] == b'\n' {
            row_stride = w + 1;
        }
    }

    let expected_min_len = (h.saturating_sub(1)) * row_stride + w;
    if buf.len() < expected_min_len {
        println!("0");
        return;
    }

    let mut start_idx = usize::MAX;
    let mut end_idx = usize::MAX;

    for idx in 0..buf.len() {
        let c = buf[idx];
        if c == b'A' {
            start_idx = idx;
            if end_idx != usize::MAX { break; }
        } else if c == b'B' {
            end_idx = idx;
            if start_idx != usize::MAX { break; }
        }
    }

    if start_idx == usize::MAX || end_idx == usize::MAX {
        println!("0");
        return;
    }

    if start_idx == end_idx {
        let r = start_idx / row_stride;
        let c = start_idx % row_stride;
        println!("1\n{} {}", c, r);
        return;
    }

    let mut q: VecDeque<(u32, u32)> = VecDeque::with_capacity(65536);
    let start_r = (start_idx / row_stride) as u32;
    let start_c = (start_idx % row_stride) as u32;
    q.push_back((start_r, start_c));
    
    let mut found = false;
    let mut b_came_from = b'X';

    while let Some((r, c)) = q.pop_front() {
        let curr = (r as usize) * row_stride + (c as usize);
        
        // Up
        if r > 0 {
            let n = curr - row_stride;
            let val = buf[n];
            if val == b' ' {
                buf[n] = b'D';
                q.push_back((r - 1, c));
            } else if val == b'B' {
                b_came_from = b'D';
                found = true;
                break;
            }
        }
        // Down
        if (r as usize) + 1 < h {
            let n = curr + row_stride;
            let val = buf[n];
            if val == b' ' {
                buf[n] = b'U';
                q.push_back((r + 1, c));
            } else if val == b'B' {
                b_came_from = b'U';
                found = true;
                break;
            }
        }
        // Left
        if c > 0 {
            let n = curr - 1;
            let val = buf[n];
            if val == b' ' {
                buf[n] = b'R';
                q.push_back((r, c - 1));
            } else if val == b'B' {
                b_came_from = b'R';
                found = true;
                break;
            }
        }
        // Right
        if (c as usize) + 1 < w {
            let n = curr + 1;
            let val = buf[n];
            if val == b' ' {
                buf[n] = b'L';
                q.push_back((r, c + 1));
            } else if val == b'B' {
                b_came_from = b'L';
                found = true;
                break;
            }
        }
    }

    if !found {
        println!("0");
        return;
    }

    let mut curr = end_idx;
    let mut came_from = b_came_from;
    let mut count = 1;

    while curr != start_idx {
        let parent = match came_from {
            b'U' => curr - row_stride,
            b'D' => curr + row_stride,
            b'L' => curr - 1,
            b'R' => curr + 1,
            _ => unreachable!(),
        };
        
        let forward_dir = match came_from {
            b'U' => b'D',
            b'D' => b'U',
            b'L' => b'R',
            b'R' => b'L',
            _ => unreachable!(),
        };
        
        if parent != start_idx {
            let next_came_from = buf[parent];
            buf[parent] = forward_dir;
            came_from = next_came_from;
        } else {
            buf[parent] = forward_dir;
        }
        
        curr = parent;
        count += 1;
    }

    let stdout = std::io::stdout();
    let mut out = std::io::BufWriter::new(stdout.lock());

    writeln!(out, "{}", count).unwrap();

    let mut curr = start_idx;
    let mut r = curr / row_stride;
    let mut c = curr % row_stride;

    for _ in 0..count {
        writeln!(out, "{} {}", c, r).unwrap();
        
        if curr == end_idx {
            break;
        }
        
        match buf[curr] {
            b'U' => { curr -= row_stride; r -= 1; },
            b'D' => { curr += row_stride; r += 1; },
            b'L' => { curr -= 1; c -= 1; },
            b'R' => { curr += 1; c += 1; },
            _ => unreachable!(),
        }
    }
}
