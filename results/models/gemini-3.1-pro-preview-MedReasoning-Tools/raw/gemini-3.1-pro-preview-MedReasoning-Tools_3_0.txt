use std::io::{self, Read};
use std::time::Instant;
use std::collections::VecDeque;

struct Scanner {
    buffer: Vec<u8>,
    pos: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut buffer = Vec::new();
        let _ = io::stdin().read_to_end(&mut buffer);
        Scanner { buffer, pos: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.pos < self.buffer.len() && self.buffer[self.pos] <= b' ' {
            self.pos += 1;
        }
        if self.pos >= self.buffer.len() {
            return None;
        }
        let mut res = 0;
        while self.pos < self.buffer.len() && self.buffer[self.pos] > b' ' {
            res = res * 10 + (self.buffer[self.pos] - b'0') as usize;
            self.pos += 1;
        }
        Some(res)
    }
}

#[derive(Clone, Copy)]
struct Point {
    x: f32,
    y: f32,
    id: usize,
}

#[derive(Clone, Copy)]
struct BoundingBox {
    min_x: f32,
    min_y: f32,
    max_x: f32,
    max_y: f32,
}

#[derive(Default, Clone, Copy)]
struct TreeNode {
    mass: f32,
    cm_x: f32,
    cm_y: f32,
    min_x: f32,
    min_y: f32,
    max_x: f32,
    max_y: f32,
    children: [usize; 4],
    is_leaf: bool,
    single_node_id: usize,
}

const MAX_DEPTH: usize = 64;

fn build_tree(points: &mut [Point], bounds: BoundingBox, depth: usize, tree: &mut Vec<TreeNode>) -> usize {
    if points.is_empty() {
        return !0;
    }

    let node_idx = tree.len();
    tree.push(TreeNode {
        min_x: bounds.min_x,
        min_y: bounds.min_y,
        max_x: bounds.max_x,
        max_y: bounds.max_y,
        children: [!0; 4],
        ..Default::default()
    });

    if points.len() == 1 || depth >= MAX_DEPTH {
        let mut cm_x = 0.0;
        let mut cm_y = 0.0;
        for p in points.iter() {
            cm_x += p.x;
            cm_y += p.y;
        }
        let mass = points.len() as f32;
        tree[node_idx].mass = mass;
        tree[node_idx].cm_x = cm_x / mass;
        tree[node_idx].cm_y = cm_y / mass;
        tree[node_idx].is_leaf = true;
        tree[node_idx].single_node_id = if points.len() == 1 { points[0].id } else { !0 };
        return node_idx;
    }

    let mid_x = (bounds.min_x + bounds.max_x) / 2.0;
    let mid_y = (bounds.min_y + bounds.max_y) / 2.0;

    let mut i = 0;
    let mut j = points.len();
    while i < j {
        if points[i].y < mid_y {
            i += 1;
        } else {
            j -= 1;
            points.swap(i, j);
        }
    }
    let split_y = i;

    let mut i = 0;
    let mut j = split_y;
    while i < j {
        if points[i].x < mid_x {
            i += 1;
        } else {
            j -= 1;
            points.swap(i, j);
        }
    }
    let split_top_x = i;

    let mut i = split_y;
    let mut j = points.len();
    while i < j {
        if points[i].x < mid_x {
            i += 1;
        } else {
            j -= 1;
            points.swap(i, j);
        }
    }
    let split_bottom_x = i;

    let c0 = build_tree(
        &mut points[0..split_top_x],
        BoundingBox {
            min_x: bounds.min_x,
            min_y: bounds.min_y,
            max_x: mid_x,
            max_y: mid_y,
        },
        depth + 1,
        tree,
    );
    let c1 = build_tree(
        &mut points[split_top_x..split_y],
        BoundingBox {
            min_x: mid_x,
            min_y: bounds.min_y,
            max_x: bounds.max_x,
            max_y: mid_y,
        },
        depth + 1,
        tree,
    );
    let c2 = build_tree(
        &mut points[split_y..split_bottom_x],
        BoundingBox {
            min_x: bounds.min_x,
            min_y: mid_y,
            max_x: mid_x,
            max_y: bounds.max_y,
        },
        depth + 1,
        tree,
    );
    let c3 = build_tree(
        &mut points[split_bottom_x..points.len()],
        BoundingBox {
            min_x: mid_x,
            min_y: mid_y,
            max_x: bounds.max_x,
            max_y: bounds.max_y,
        },
        depth + 1,
        tree,
    );

    tree[node_idx].children = [c0, c1, c2, c3];

    let mut mass = 0.0;
    let mut cm_x = 0.0;
    let mut cm_y = 0.0;
    for &c in &tree[node_idx].children {
        if c != !0 {
            let child_mass = tree[c].mass;
            mass += child_mass;
            cm_x += tree[c].cm_x * child_mass;
            cm_y += tree[c].cm_y * child_mass;
        }
    }
    tree[node_idx].mass = mass;
    if mass > 0.0 {
        tree[node_idx].cm_x = cm_x / mass;
        tree[node_idx].cm_y = cm_y / mass;
    }
    tree[node_idx].is_leaf = false;

    node_idx
}

fn compute_repulsion(
    node_idx: usize,
    ux: f32,
    uy: f32,
    tree: &[TreeNode],
    tree_idx: usize,
    theta_sq: f32,
) -> (f32, f32) {
    let mut stack = [0; 256];
    let mut stack_ptr = 1;
    stack[0] = tree_idx;

    let mut fx = 0.0;
    let mut fy = 0.0;

    while stack_ptr > 0 {
        stack_ptr -= 1;
        let curr_idx = stack[stack_ptr];
        let node = &tree[curr_idx];

        let dx = ux - node.cm_x;
        let dy = uy - node.cm_y;
        let dist_sq = dx * dx + dy * dy;

        if node.is_leaf {
            if node.single_node_id != node_idx && dist_sq > 1e-6 {
                let f = node.mass / dist_sq;
                fx += f * dx;
                fy += f * dy;
            }
        } else {
            let width = node.max_x - node.min_x;
            let height = node.max_y - node.min_y;
            let size = if width > height { width } else { height };
            if size * size < theta_sq * dist_sq {
                if dist_sq > 1e-6 {
                    let f = node.mass / dist_sq;
                    fx += f * dx;
                    fy += f * dy;
                }
            } else {
                for &c in &node.children {
                    if c != !0 {
                        stack[stack_ptr] = c;
                        stack_ptr += 1;
                    }
                }
            }
        }
    }
    (fx, fy)
}

fn power_iteration(s: &[Vec<f32>], k: usize, num_iter: usize) -> Vec<f32> {
    let mut v = vec![1.0; k];
    let mut len = 0.0;
    for &x in &v {
        len += x * x;
    }
    len = len.sqrt();
    if len > 1e-9 {
        for x in &mut v {
            *x /= len;
        }
    }

    for _ in 0..num_iter {
        let mut next_v = vec![0.0; k];
        for i in 0..k {
            for j in 0..k {
                next_v[i] += s[i][j] * v[j];
            }
        }
        let mut len = 0.0;
        for &x in &next_v {
            len += x * x;
        }
        len = len.sqrt();
        if len > 1e-9 {
            for x in &mut next_v {
                *x /= len;
            }
            v = next_v;
        } else {
            break;
        }
    }
    v
}

fn power_iteration_2(s: &[Vec<f32>], k: usize, num_iter: usize, v1: &[f32]) -> Vec<f32> {
    let mut v = vec![1.0; k];
    let mut dot = 0.0;
    for i in 0..k {
        dot += v[i] * v1[i];
    }
    for i in 0..k {
        v[i] -= dot * v1[i];
    }

    let mut len = 0.0;
    for &x in &v {
        len += x * x;
    }
    len = len.sqrt();
    if len > 1e-9 {
        for x in &mut v {
            *x /= len;
        }
    }

    for _ in 0..num_iter {
        let mut next_v = vec![0.0; k];
        for i in 0..k {
            for j in 0..k {
                next_v[i] += s[i][j] * v[j];
            }
        }
        let mut dot = 0.0;
        for i in 0..k {
            dot += next_v[i] * v1[i];
        }
        for i in 0..k {
            next_v[i] -= dot * v1[i];
        }

        let mut len = 0.0;
        for &x in &next_v {
            len += x * x;
        }
        len = len.sqrt();
        if len > 1e-9 {
            for x in &mut next_v {
                *x /= len;
            }
            v = next_v;
        } else {
            break;
        }
    }
    v
}

fn pivot_mds(comp: &[usize], adj: &[Vec<usize>]) -> (Vec<f32>, Vec<f32>) {
    let n_c = adj.len();
    let num_pivots = n_c.min(50);
    let mut pivots = Vec::with_capacity(num_pivots);
    for i in 0..num_pivots {
        pivots.push(i * n_c / num_pivots);
    }

    let mut dist = vec![vec![0.0f32; num_pivots]; n_c];
    let mut queue = VecDeque::new();
    let mut dist_array = vec![-1.0; n_c];

    for (p_idx, &p) in pivots.iter().enumerate() {
        dist_array.fill(-1.0);
        dist_array[p] = 0.0;
        queue.push_back(p);
        
        while let Some(u) = queue.pop_front() {
            for &v in &adj[u] {
                if dist_array[v] < 0.0 {
                    dist_array[v] = dist_array[u] + 1.0;
                    queue.push_back(v);
                }
            }
        }

        let mut max_d = 0.0;
        for i in 0..n_c {
            if dist_array[i] > max_d {
                max_d = dist_array[i];
            }
        }
        for i in 0..n_c {
            if dist_array[i] < 0.0 {
                dist[i][p_idx] = max_d + 2.0;
            } else {
                dist[i][p_idx] = dist_array[i];
            }
        }
    }

    for j in 0..num_pivots {
        let mut sum = 0.0;
        for i in 0..n_c {
            sum += dist[i][j];
        }
        let mean = sum / n_c as f32;
        for i in 0..n_c {
            dist[i][j] -= mean;
        }
    }

    let mut s = vec![vec![0.0; num_pivots]; num_pivots];
    for i in 0..num_pivots {
        for j in 0..num_pivots {
            let mut dot = 0.0;
            for k in 0..n_c {
                dot += dist[k][i] * dist[k][j];
            }
            s[i][j] = dot;
        }
    }

    let v1 = power_iteration(&s, num_pivots, 50);
    let v2 = power_iteration_2(&s, num_pivots, 50, &v1);

    let mut xs = vec![0.0; n_c];
    let mut ys = vec![0.0; n_c];

    for i in 0..n_c {
        for j in 0..num_pivots {
            xs[i] += dist[i][j] * v1[j];
            ys[i] += dist[i][j] * v2[j];
        }
    }

    let target_stddev = (n_c as f32 / 12.0).sqrt();

    let mut scale_coord = |coords: &mut Vec<f32>| {
        let mut sum = 0.0;
        for &x in coords.iter() {
            sum += x;
        }
        let mean = sum / n_c as f32;
        let mut var = 0.0;
        for x in coords.iter_mut() {
            *x -= mean;
            var += (*x) * (*x);
        }
        var /= n_c as f32;
        let stddev = var.sqrt();
        if stddev > 1e-6 {
            let factor = target_stddev / stddev;
            for x in coords.iter_mut() {
                *x *= factor;
            }
        }
    };

    scale_coord(&mut xs);
    scale_coord(&mut ys);

    for i in 0..n_c {
        xs[i] += ((i * 137) % 1000) as f32 / 10000.0;
        ys[i] += ((i * 271) % 1000) as f32 / 10000.0;
    }

    (xs, ys)
}

fn layout_component(comp: &[usize], global_adj: &[Vec<usize>], start_time: Instant) -> (Vec<f32>, Vec<f32>) {
    let n_c = comp.len();
    if n_c == 1 {
        return (vec![0.0], vec![0.0]);
    }
    if n_c == 2 {
        return (vec![-0.5, 0.5], vec![0.0, 0.0]);
    }

    let mut local_idx = vec![!0; global_adj.len()];
    for (i, &u) in comp.iter().enumerate() {
        local_idx[u] = i;
    }

    let mut adj = vec![vec![]; n_c];
    for (i, &u) in comp.iter().enumerate() {
        for &v in &global_adj[u] {
            let j = local_idx[v];
            if j != !0 {
                adj[i].push(j);
            }
        }
    }

    let mut xs = vec![0.0; n_c];
    let mut ys = vec![0.0; n_c];

    if n_c >= 50 {
        let (mds_x, mds_y) = pivot_mds(comp, &adj);
        xs = mds_x;
        ys = mds_y;
    } else {
        let size = (n_c as f32).sqrt();
        for i in 0..n_c {
            xs[i] = ((i * 137) % 1000) as f32 / 1000.0 * size;
            ys[i] = ((i * 271) % 1000) as f32 / 1000.0 * size;
        }
    }

    let max_iter = 250;
    let mut temperature = (n_c as f32).sqrt().max(5.0);
    let final_temperature = 0.05;
    let cooling = (final_temperature / temperature).powf(1.0 / max_iter as f32);

    let mut points = vec![Point { x: 0.0, y: 0.0, id: 0 }; n_c];
    let mut tree = Vec::with_capacity(n_c * 4);
    let mut fx = vec![0.0; n_c];
    let mut fy = vec![0.0; n_c];

    let theta_sq = 1.44;

    for _ in 0..max_iter {
        if start_time.elapsed().as_secs_f32() > 28.5 {
            break;
        }

        let mut min_x = f32::MAX;
        let mut min_y = f32::MAX;
        let mut max_x = f32::MIN;
        let mut max_y = f32::MIN;
        for i in 0..n_c {
            if xs[i] < min_x { min_x = xs[i]; }
            if xs[i] > max_x { max_x = xs[i]; }
            if ys[i] < min_y { min_y = ys[i]; }
            if ys[i] > max_y { max_y = ys[i]; }
        }
        
        if max_x - min_x < 1e-4 {
            max_x += 1e-4;
            min_x -= 1e-4;
        }
        if max_y - min_y < 1e-4 {
            max_y += 1e-4;
            min_y -= 1e-4;
        }

        for i in 0..n_c {
            points[i] = Point {
                x: xs[i],
                y: ys[i],
                id: i,
            };
        }

        let bounds = BoundingBox {
            min_x,
            min_y,
            max_x,
            max_y,
        };
        
        tree.clear();
        build_tree(&mut points, bounds, 0, &mut tree);

        fx.fill(0.0);
        fy.fill(0.0);

        for i in 0..n_c {
            let (rx, ry) = compute_repulsion(i, xs[i], ys[i], &tree, 0, theta_sq);
            fx[i] += rx;
            fy[i] += ry;
            
            fx[i] -= 0.01 * xs[i];
            fy[i] -= 0.01 * ys[i];
        }

        for u in 0..n_c {
            for &v in &adj[u] {
                if u < v {
                    let dx = xs[v] - xs[u];
                    let dy = ys[v] - ys[u];

                    fx[u] += dx;
                    fy[u] += dy;
                    fx[v] -= dx;
                    fy[v] -= dy;
                }
            }
        }

        for i in 0..n_c {
            let len = (fx[i] * fx[i] + fy[i] * fy[i]).sqrt();
            if len > 1e-6 {
                let disp = len.min(temperature);
                xs[i] += (fx[i] / len) * disp;
                ys[i] += (fy[i] / len) * disp;
            }
        }

        temperature *= cooling;
    }

    (xs, ys)
}

struct CompBox {
    comp_idx: usize,
    w: f32,
    h: f32,
    min_x: f32,
    min_y: f32,
}

fn main() {
    let start_time = Instant::now();
    let mut scanner = Scanner::new();

    let n = match scanner.next_usize() {
        Some(x) => x,
        None => return,
    };
    let m = match scanner.next_usize() {
        Some(x) => x,
        None => return,
    };

    let mut adj = vec![vec![]; n];
    for _ in 0..m {
        if let (Some(u), Some(v)) = (scanner.next_usize(), scanner.next_usize()) {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    for i in 0..n {
        adj[i].sort_unstable();
        adj[i].dedup();
        adj[i].retain(|&x| x != i);
    }

    let mut visited = vec![false; n];
    let mut components = Vec::new();

    for i in 0..n {
        if !visited[i] {
            let mut comp = Vec::new();
            let mut q = vec![i];
            visited[i] = true;
            let mut head = 0;
            while head < q.len() {
                let u = q[head];
                head += 1;
                comp.push(u);
                for &v in &adj[u] {
                    if !visited[v] {
                        visited[v] = true;
                        q.push(v);
                    }
                }
            }
            components.push(comp);
        }
    }

    let mut global_x = vec![0.0f32; n];
    let mut global_y = vec![0.0f32; n];

    let mut comp_boxes = Vec::with_capacity(components.len());

    for comp in &components {
        if start_time.elapsed().as_secs_f32() > 29.0 {
            let size = (comp.len() as f32).sqrt();
            let mut xs = vec![0.0; comp.len()];
            let mut ys = vec![0.0; comp.len()];
            for i in 0..comp.len() {
                xs[i] = ((i * 137) % 1000) as f32 / 1000.0 * size;
                ys[i] = ((i * 271) % 1000) as f32 / 1000.0 * size;
            }
            
            let mut min_x = f32::MAX;
            let mut min_y = f32::MAX;
            let mut max_x = f32::MIN;
            let mut max_y = f32::MIN;

            for i in 0..comp.len() {
                let u = comp[i];
                global_x[u] = xs[i];
                global_y[u] = ys[i];

                if xs[i] < min_x { min_x = xs[i]; }
                if xs[i] > max_x { max_x = xs[i]; }
                if ys[i] < min_y { min_y = ys[i]; }
                if ys[i] > max_y { max_y = ys[i]; }
            }

            if comp.is_empty() {
                min_x = 0.0; min_y = 0.0; max_x = 0.0; max_y = 0.0;
            }

            comp_boxes.push(CompBox {
                comp_idx: comp_boxes.len(),
                w: (max_x - min_x).max(0.1),
                h: (max_y - min_y).max(0.1),
                min_x,
                min_y,
            });
            
            continue;
        }

        let (xs, ys) = layout_component(comp, &adj, start_time);

        let mut min_x = f32::MAX;
        let mut min_y = f32::MAX;
        let mut max_x = f32::MIN;
        let mut max_y = f32::MIN;

        for i in 0..comp.len() {
            let u = comp[i];
            global_x[u] = xs[i];
            global_y[u] = ys[i];

            if xs[i] < min_x { min_x = xs[i]; }
            if xs[i] > max_x { max_x = xs[i]; }
            if ys[i] < min_y { min_y = ys[i]; }
            if ys[i] > max_y { max_y = ys[i]; }
        }

        if comp.is_empty() {
            min_x = 0.0; min_y = 0.0; max_x = 0.0; max_y = 0.0;
        }

        comp_boxes.push(CompBox {
            comp_idx: comp_boxes.len(),
            w: (max_x - min_x).max(0.1),
            h: (max_y - min_y).max(0.1),
            min_x,
            min_y,
        });
    }

    comp_boxes.sort_unstable_by(|a, b| b.h.partial_cmp(&a.h).unwrap());

    let mut total_area = 0.0;
    for b in &comp_boxes {
        total_area += b.w * b.h;
    }
    let max_width = total_area.sqrt() * 1.5;

    let mut curr_x = 0.0;
    let mut curr_y = 0.0;
    let mut row_h = 0.0;
    let padding = 2.0;

    for b in &comp_boxes {
        if curr_x + b.w > max_width && curr_x > 0.0 {
            curr_x = 0.0;
            curr_y += row_h + padding;
            row_h = 0.0;
        }

        let dx = curr_x - b.min_x;
        let dy = curr_y - b.min_y;

        for &u in &components[b.comp_idx] {
            global_x[u] += dx;
            global_y[u] += dy;
        }

        curr_x += b.w + padding;
        if b.h > row_h {
            row_h = b.h;
        }
    }

    for i in 0..n {
        println!("{} {}", global_x[i], global_y[i]);
    }
}
