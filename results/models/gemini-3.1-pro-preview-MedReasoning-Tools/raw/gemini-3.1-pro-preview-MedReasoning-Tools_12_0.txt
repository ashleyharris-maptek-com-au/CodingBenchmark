    public struct Vector3
    {
        public double X, Y, Z;
        public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
    }

    public struct Quaternion
    {
        public double W, X, Y, Z;
        public Quaternion(double w, double x, double y, double z) { W = w; X = x; Y = y; Z = z; }
        
        public static readonly Quaternion Identity = new Quaternion(1, 0, 0, 0);

        public Vector3 Rotate(Vector3 v)
        {
            double num = X * 2.0;
            double num2 = Y * 2.0;
            double num3 = Z * 2.0;
            double num4 = X * num;
            double num5 = Y * num2;
            double num6 = Z * num3;
            double num7 = X * num2;
            double num8 = X * num3;
            double num9 = Y * num3;
            double num10 = W * num;
            double num11 = W * num2;
            double num12 = W * num3;

            return new Vector3(
                (1.0 - (num5 + num6)) * v.X + (num7 - num12) * v.Y + (num8 + num11) * v.Z,
                (num7 + num12) * v.X + (1.0 - (num4 + num6)) * v.Y + (num9 - num10) * v.Z,
                (num8 - num11) * v.X + (num9 + num10) * v.Y + (1.0 - (num4 + num5)) * v.Z
            );
        }

        public Quaternion Multiply(Quaternion q)
        {
            return new Quaternion(
                W * q.W - X * q.X - Y * q.Y - Z * q.Z,
                W * q.X + X * q.W + Y * q.Z - Z * q.Y,
                W * q.Y - X * q.Z + Y * q.W + Z * q.X,
                W * q.Z + X * q.Y - Y * q.X + Z * q.W
            );
        }

        public Quaternion Normalize()
        {
            double mag = Math.Sqrt(W * W + X * X + Y * Y + Z * Z);
            if (mag < 1e-9) return Identity;
            return new Quaternion(W / mag, X / mag, Y / mag, Z / mag);
        }
    }

    class EvalResult : IComparable<EvalResult>
    {
        public Quaternion Rot;
        public long Count;
        public double Volume;
        public double Sx, Sy, Sz;
        public double MinX, MinY, MinZ;

        public int CompareTo(EvalResult other)
        {
            int c = other.Count.CompareTo(this.Count);
            if (c != 0) return c;
            return this.Volume.CompareTo(other.Volume); 
        }
    }

        if (string.IsNullOrWhiteSpace(firstLine)) return;

        int F = int.Parse(parts[1]);

        for (int i = 0; i < V; i++)

        double H = double.Parse(parts[1], CultureInfo.InvariantCulture);
        double D = double.Parse(parts[2], CultureInfo.InvariantCulture);

        double containerVolume = W * H * D;
        
        int baseRotations = 1000000;
        int localSearchTopK = 50;
        int localSearchIters = 6;
        int localSearchSamples = 10000;

        if (containerVolume < 2000)
        {
            baseRotations = 3000000;
            localSearchTopK = 100;
            localSearchIters = 10;
        }
        else if (containerVolume > 1000000)
        {
            baseRotations = 300000;
            localSearchTopK = 20;
            localSearchIters = 4;
            localSearchSamples = 5000;
        }

        List<Quaternion> initialRotations = new List<Quaternion>(baseRotations);
        initialRotations.Add(Quaternion.Identity);
        initialRotations.AddRange(GenerateAxisAlignedRotations());

        Random rnd = new Random(42);
        while (initialRotations.Count < baseRotations)
        {
            initialRotations.Add(GetRandomRotation(rnd));
        }

        Func<Quaternion, EvalResult> evaluateSingle = (rot) =>
        {
            double minX = double.MaxValue, minY = double.MaxValue, minZ = double.MaxValue;
            double maxX = double.MinValue, maxY = double.MinValue, maxZ = double.MinValue;

            for (int i = 0; i < V; i++)
            {
                Vector3 rv = rot.Rotate(vertices[i]);
                if (rv.X < minX) minX = rv.X;
                if (rv.Y < minY) minY = rv.Y;
                if (rv.Z < minZ) minZ = rv.Z;
                if (rv.X > maxX) maxX = rv.X;
                if (rv.Y > maxY) maxY = rv.Y;
                if (rv.Z > maxZ) maxZ = rv.Z;
            }

            double sx = Math.Max(maxX - minX + 1e-9, 1e-9);
            double sy = Math.Max(maxY - minY + 1e-9, 1e-9);
            double sz = Math.Max(maxZ - minZ + 1e-9, 1e-9);

            if (sx > W || sy > H || sz > D) return null;

            long countX = (long)(W / sx);
            long countY = (long)(H / sy);
            long countZ = (long)(D / sz);
            
            long total = countX * countY * countZ;

            if (total > 0)
            {
                return new EvalResult 
                { 
                    Rot = rot, Count = total, Volume = sx * sy * sz, 
                    Sx = sx, Sy = sy, Sz = sz, 
                    MinX = minX, MinY = minY, MinZ = minZ 
                };
            }
            return null;
        };

        object globalLock = new object();
        List<EvalResult> globalTopResults = new List<EvalResult>();

        Parallel.ForEach(initialRotations, 
            () => new List<EvalResult>(),
            (rot, state, localList) => 
            {
                var res = evaluateSingle(rot);
                if (res != null)
                {
                    localList.Add(res);
                    if (localList.Count > 1000)
                    {
                        localList.Sort();
                        localList.RemoveRange(localSearchTopK, localList.Count - localSearchTopK);
                    }
                }
                return localList;
            },
            (localList) => 
            {
                lock (globalLock)
                {
                    globalTopResults.AddRange(localList);
                }
            }
        );

        globalTopResults.Sort();
        List<EvalResult> currentTop = globalTopResults.Take(localSearchTopK).ToList();

        double angleSpread = 0.2; 
        
        for (int iter = 0; iter < localSearchIters; iter++)
        {
            if (sw.ElapsedMilliseconds > 250000) break;

            List<EvalResult> iterTopResults = new List<EvalResult>();
            iterTopResults.AddRange(currentTop);

            Parallel.For(0, currentTop.Count, 
                () => new List<EvalResult>(),
                (i, state, localList) =>
                {
                    Random localRnd = new Random(42 + i + iter * 100);
                    var baseRot = currentTop[i].Rot;

                    for (int s = 0; s < localSearchSamples; s++)
                    {
                        Vector3 axis = new Vector3(
                            localRnd.NextDouble() * 2 - 1, 
                            localRnd.NextDouble() * 2 - 1, 
                            localRnd.NextDouble() * 2 - 1);
                        
                        double len = Math.Sqrt(axis.X*axis.X + axis.Y*axis.Y + axis.Z*axis.Z);
                        if (len < 0.001) continue;
                        axis.X /= len; axis.Y /= len; axis.Z /= len;

                        double angle = (localRnd.NextDouble() * 2 - 1) * angleSpread;
                        Quaternion perturb = CreateFromAxisAngle(axis, angle);
                        
                        Quaternion newRot = baseRot.Multiply(perturb).Normalize();
                        var res = evaluateSingle(newRot);
                        if (res != null)
                        {
                            localList.Add(res);
                            if (localList.Count > 500)
                            {
                                localList.Sort();
                                localList.RemoveRange(localSearchTopK, localList.Count - localSearchTopK);
                            }
                        }
                    }
                    return localList;
                },
                (localList) =>
                {
                    lock (globalLock)
                    {
                        iterTopResults.AddRange(localList);
                    }
                }
            );

            iterTopResults.Sort();
            currentTop = iterTopResults.Take(localSearchTopK).ToList();
            angleSpread *= 0.6;
        }

        if (currentTop.Count == 0 || currentTop[0].Count == 0)
        {
            Console.WriteLine("0");
            return;
        }

        var bestGlobal = currentTop[0];
        long maxCount = bestGlobal.Count;
        Console.WriteLine(maxCount);

        long cX = (long)(W / bestGlobal.Sx);
        long cY = (long)(H / bestGlobal.Sy);
        long cZ = (long)(D / bestGlobal.Sz);

        long packed = 0;
        
        using (StreamWriter writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 65536))
        {
            for (long i = 0; i < cX; i++)
            {
                for (long j = 0; j < cY; j++)
                {
                    for (long k = 0; k < cZ; k++)
                    {
                        if (packed >= maxCount) break;

                        double tx = i * bestGlobal.Sx - bestGlobal.MinX;
                        double ty = j * bestGlobal.Sy - bestGlobal.MinY;

                        packed++;

                if (packed >= maxCount) break;

        }
    }

    static List<Quaternion> GenerateAxisAlignedRotations()
    {
        List<Quaternion> list = new List<Quaternion>();
        double[] angles = { 0, Math.PI / 2, Math.PI, 3 * Math.PI / 2 };
        foreach (var yaw in angles)
            foreach (var pitch in angles)
                foreach (var roll in angles)
                {
                    list.Add(CreateFromYawPitchRoll(yaw, pitch, roll));
                }
        return list;
    }

    static Quaternion CreateFromYawPitchRoll(double yaw, double pitch, double roll)
    {
        double sr = Math.Sin(roll * 0.5);
        double cr = Math.Cos(roll * 0.5);
        double sp = Math.Sin(pitch * 0.5);
        double cp = Math.Cos(pitch * 0.5);
        double sy = Math.Sin(yaw * 0.5);
        double cy = Math.Cos(yaw * 0.5);

        return new Quaternion(
            cr * cp * cy + sr * sp * sy,
            sr * cp * cy - cr * sp * sy,
            cr * sp * cy + sr * cp * sy,
            cr * cp * sy - sr * sp * cy
        );
    }

    static Quaternion GetRandomRotation(Random rnd)
    {
        double u1 = rnd.NextDouble();
        double u2 = rnd.NextDouble();
        double u3 = rnd.NextDouble();
        double sqrt1u1 = Math.Sqrt(1 - u1);
        double sqrtu1 = Math.Sqrt(u1);
        double w = sqrt1u1 * Math.Sin(2 * Math.PI * u2);
        double x = sqrt1u1 * Math.Cos(2 * Math.PI * u2);
        double y = sqrtu1 * Math.Sin(2 * Math.PI * u3);
        double z = sqrtu1 * Math.Cos(2 * Math.PI * u3);
        return new Quaternion(w, x, y, z);
    }

    static Quaternion CreateFromAxisAngle(Vector3 axis, double angle)
    {
        double halfAngle = angle * 0.5;
        double s = Math.Sin(halfAngle);
        double c = Math.Cos(halfAngle);
        return new Quaternion(c, axis.X * s, axis.Y * s, axis.Z * s);
    }
