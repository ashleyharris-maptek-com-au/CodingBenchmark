use std::io::{self, BufRead};
use std::f64;

#[derive(Clone, Copy, Debug)]
struct Point { x: f64, y: f64 }

#[derive(Clone)]
struct AABB {
    min_x: f64, max_x: f64,
    min_y: f64, max_y: f64,
}

impl AABB {
    fn intersects(&self, other: &AABB) -> bool {
        !(self.max_x < other.min_x - 1e-5 ||
          self.min_x > other.max_x + 1e-5 ||
          self.max_y < other.min_y - 1e-5 ||
          self.min_y > other.max_y + 1e-5)
    }
    
    fn inside(&self, other: &AABB) -> bool {
        self.min_x >= other.min_x - 1e-5 &&
        self.max_x <= other.max_x + 1e-5 &&
        self.min_y >= other.min_y - 1e-5 &&
        self.max_y <= other.max_y + 1e-5
    }
}

#[derive(Clone)]
struct Piece {
    id: usize,
    vertices: Vec<Point>,
    aabb: AABB,
}

impl Piece {
    fn new(id: usize, vertices: Vec<Point>) -> Self {
        let mut min_x = f64::MAX;
        let mut max_x = f64::MIN;
        let mut min_y = f64::MAX;
        let mut max_y = f64::MIN;
        for p in &vertices {
            if p.x < min_x { min_x = p.x; }
            if p.x > max_x { max_x = p.x; }
            if p.y < min_y { min_y = p.y; }
            if p.y > max_y { max_y = p.y; }
        }
        Piece {
            id,
            vertices,
            aabb: AABB { min_x, max_x, min_y, max_y },
        }
    }
    
    fn area(&self) -> f64 {
        let mut a = 0.0;
        for i in 0..self.vertices.len() {
            let p1 = self.vertices[i];
            let p2 = self.vertices[(i+1)%self.vertices.len()];
            a += p1.x * p2.y - p2.x * p1.y;
        }
        a.abs() / 2.0
    }
    
    fn translate(&self, dx: f64, dy: f64) -> Self {
        let v: Vec<Point> = self.vertices.iter().map(|p| Point{x: p.x + dx, y: p.y + dy}).collect();
        Piece::new(self.id, v)
    }

    fn rotate(&self, angle: f64) -> Self {
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        let v: Vec<Point> = self.vertices.iter().map(|p| {
            Point {
                x: p.x * cos_a - p.y * sin_a,
                y: p.x * sin_a + p.y * cos_a,
            }
        }).collect();
        Piece::new(self.id, v)
    }
}

fn cross_product(a: Point, b: Point, c: Point) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn on_segment(a: Point, b: Point, c: Point) -> bool {
    if cross_product(a, b, c).abs() > 1e-5 {
        return false;
    }
    c.x >= a.x.min(b.x) - 1e-5 && c.x <= a.x.max(b.x) + 1e-5 &&
    c.y >= a.y.min(b.y) - 1e-5 && c.y <= a.y.max(b.y) + 1e-5
}

fn strict_segments_intersect(a: Point, b: Point, c: Point, d: Point) -> bool {
    let cp1 = cross_product(a, b, c);
    let cp2 = cross_product(a, b, d);
    let cp3 = cross_product(c, d, a);
    let cp4 = cross_product(c, d, b);

    if ((cp1 > 1e-5 && cp2 < -1e-5) || (cp1 < -1e-5 && cp2 > 1e-5)) &&
       ((cp3 > 1e-5 && cp4 < -1e-5) || (cp3 < -1e-5 && cp4 > 1e-5)) {
        return true;
    }
    false
}

fn point_in_polygon(pt: Point, poly: &[Point]) -> bool {
    let mut inside = false;
    let mut j = poly.len() - 1;
    for i in 0..poly.len() {
        let pi = poly[i];
        let pj = poly[j];
        
        if ((pi.y > pt.y) != (pj.y > pt.y)) &&
           (pt.x < (pj.x - pi.x) * (pt.y - pi.y) / (pj.y - pi.y) + pi.x) {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn point_strictly_inside_polygon(pt: Point, poly: &[Point]) -> bool {
    for i in 0..poly.len() {
        let j = (i + 1) % poly.len();
        if on_segment(poly[i], poly[j], pt) {
            return false;
        }
    }
    point_in_polygon(pt, poly)
}

fn get_interior_point(poly: &[Point]) -> Point {
    let mut cx = 0.0;
    let mut cy = 0.0;
    for p in poly {
        cx += p.x;
        cy += p.y;
    }
    let n = poly.len() as f64;
    let centroid = Point { x: cx/n, y: cy/n };
    if point_in_polygon(centroid, poly) {
        return centroid;
    }
    
    let mut min_x = poly[0].x;
    let mut max_x = poly[0].x;
    let mut min_y = poly[0].y;
    let mut max_y = poly[0].y;
    for p in poly {
        min_x = min_x.min(p.x);
        max_x = max_x.max(p.x);
        min_y = min_y.min(p.y);
        max_y = max_y.max(p.y);
    }
    
    let steps = 20;
    for i in 1..steps {
        for j in 1..steps {
            let pt = Point {
                x: min_x + (max_x - min_x) * (i as f64 / steps as f64),
                y: min_y + (max_y - min_y) * (j as f64 / steps as f64),
            };
            if point_in_polygon(pt, poly) {
                return pt;
            }
        }
    }
    centroid
}

fn translate_pt(p: Point, tx: f64, ty: f64) -> Point {
    Point { x: p.x + tx, y: p.y + ty }
}

fn point_strictly_inside_polygon_trans(pt: Point, poly: &[Point], tx: f64, ty: f64) -> bool {
    let local_pt = Point { x: pt.x - tx, y: pt.y - ty };
    point_strictly_inside_polygon(local_pt, poly)
}

fn poly_inside_stock(poly: &[Point], tx: f64, ty: f64, stock: &[Point]) -> bool {
    for i in 0..poly.len() {
        let a1 = translate_pt(poly[i], tx, ty);
        let b1 = translate_pt(poly[(i + 1) % poly.len()], tx, ty);
        for j in 0..stock.len() {
            let a2 = stock[j];
            let b2 = stock[(j + 1) % stock.len()];
            if strict_segments_intersect(a1, b1, a2, b2) {
                return false;
            }
        }
    }
    
    let mut int_p = get_interior_point(poly);
    int_p = translate_pt(int_p, tx, ty);
    
    if !point_in_polygon(int_p, stock) {
        return false;
    }
    
    for p in stock {
        if point_strictly_inside_polygon_trans(*p, poly, tx, ty) {
            return false;
        }
    }
    true
}

fn polys_overlap(p1: &[Point], tx1: f64, ty1: f64, p2: &[Point]) -> bool {
    for i in 0..p1.len() {
        let a1 = translate_pt(p1[i], tx1, ty1);
        let b1 = translate_pt(p1[(i + 1) % p1.len()], tx1, ty1);
        for j in 0..p2.len() {
            let a2 = p2[j];
            let b2 = p2[(j + 1) % p2.len()];
            if strict_segments_intersect(a1, b1, a2, b2) {
                return true;
            }
        }
    }
    
    let mut int1 = get_interior_point(p1);
    int1 = translate_pt(int1, tx1, ty1);
    if point_strictly_inside_polygon(int1, p2) { return true; }
    
    let int2 = get_interior_point(p2);
    if point_strictly_inside_polygon_trans(int2, p1, tx1, ty1) { return true; }
    
    false
}

fn try_place(piece: &Piece, stock: &Piece, placed: &[Piece], angles: &[f64], grid_w: usize, grid_h: usize) -> Option<(f64, f64, f64)> {
    let s_aabb = &stock.aabb;
    let mut dx = (s_aabb.max_x - s_aabb.min_x) / grid_w as f64;
    let mut dy = (s_aabb.max_y - s_aabb.min_y) / grid_h as f64;
    if dx < 1e-7 { dx = 1.0; }
    if dy < 1e-7 { dy = 1.0; }

    for &angle in angles {
        let rotated = piece.rotate(angle);
        let r_aabb = &rotated.aabb;
        
        let piece_w = r_aabb.max_x - r_aabb.min_x;
        let piece_h = r_aabb.max_y - r_aabb.min_y;

        if piece_w > s_aabb.max_x - s_aabb.min_x + 1e-5 || piece_h > s_aabb.max_y - s_aabb.min_y + 1e-5 {
            continue;
        }

        let max_cx = s_aabb.max_x - piece_w;
        let max_cy = s_aabb.max_y - piece_h;

        let mut candidates = Vec::with_capacity((grid_w + 1) * (grid_h + 1) + 1);
        candidates.push((s_aabb.min_x, s_aabb.min_y));

        for i in 0..=grid_w {
            let cx = s_aabb.min_x + i as f64 * dx;
            if cx > max_cx + 1e-5 { continue; }
            for j in 0..=grid_h {
                let cy = s_aabb.min_y + j as f64 * dy;
                if cy > max_cy + 1e-5 { continue; }
                candidates.push((cx, cy));
            }
        }

        candidates.sort_by(|a, b| {
            a.1.partial_cmp(&b.1).unwrap().then(a.0.partial_cmp(&b.0).unwrap())
        });
        
        candidates.dedup_by(|a, b| (a.0 - b.0).abs() < 1e-7 && (a.1 - b.1).abs() < 1e-7);

        for (cx, cy) in candidates {
            let tx = cx - r_aabb.min_x;
            let ty = cy - r_aabb.min_y;

            let t_aabb = AABB {
                min_x: r_aabb.min_x + tx,
                max_x: r_aabb.max_x + tx,
                min_y: r_aabb.min_y + ty,
                max_y: r_aabb.max_y + ty,
            };

            let mut overlapping_placed = Vec::new();
            for p in placed {
                if t_aabb.intersects(&p.aabb) {
                    overlapping_placed.push(p);
                }
            }

            if !poly_inside_stock(&rotated.vertices, tx, ty, &stock.vertices) {
                continue;
            }

            let mut valid = true;
            for p in &overlapping_placed {
                if polys_overlap(&rotated.vertices, tx, ty, &p.vertices) {
                    valid = false;
                    break;
                }
            }

            if valid {
                return Some((tx, ty, angle));
            }
        }
    }
    None
}

fn solve(stock_vertices: Vec<Point>, mut pieces: Vec<Piece>) -> Vec<(usize, usize, f64, f64, f64)> {
    pieces.sort_by(|a, b| b.area().partial_cmp(&a.area()).unwrap());
    
    let mut stocks: Vec<Vec<Piece>> = Vec::new();
    let mut stock_used_area: Vec<f64> = Vec::new();
    let stock_poly = Piece::new(0, stock_vertices);
    let stock_area = stock_poly.area();
    
    let mut piece_results = Vec::new();
    
    let mut angles = Vec::new();
    for i in 0..8 {
        angles.push(i as f64 * std::f64::consts::PI / 4.0);
    }

    for piece in pieces.iter() {
        let mut placed = false;
        let p_area = piece.area();
        
        let start_idx = if stocks.len() > 100 { stocks.len() - 100 } else { 0 };
        for stock_idx in start_idx..stocks.len() {
            if stock_used_area[stock_idx] + p_area > stock_area {
                continue;
            }
            if let Some((x, y, rot)) = try_place(piece, &stock_poly, &stocks[stock_idx], &angles, 30, 30) {
                let placed_piece = piece.rotate(rot).translate(x, y);
                stocks[stock_idx].push(placed_piece);
                stock_used_area[stock_idx] += p_area;
                piece_results.push((piece.id, stock_idx, x, y, rot));
                placed = true;
                break;
            }
        }
        
        if !placed {
            if let Some((x, y, rot)) = try_place(piece, &stock_poly, &[], &angles, 60, 60) {
                let placed_piece = piece.rotate(rot).translate(x, y);
                stocks.push(vec![placed_piece]);
                stock_used_area.push(p_area);
                piece_results.push((piece.id, stocks.len() - 1, x, y, rot));
            } else {
                stocks.push(vec![piece.clone()]);
                stock_used_area.push(p_area);
                piece_results.push((piece.id, stocks.len() - 1, 0.0, 0.0, 0.0));
            }
        }
    }
    
    piece_results.sort_by_key(|r| r.0);
    piece_results
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines().map(|l| l.unwrap());

    if let Some(first_line) = lines.next() {
        let parts: Vec<usize> = first_line.split_whitespace().map(|s| s.parse().unwrap()).collect();
        let sv = parts[0];
        let np = parts[1];

        let mut stock_vertices = Vec::with_capacity(sv);
        for _ in 0..sv {
            let line = lines.next().unwrap();
            let coords: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
            stock_vertices.push(Point { x: coords[0], y: coords[1] });
        }

        let mut pieces = Vec::with_capacity(np);
        for i in 0..np {
            let line = lines.next().unwrap();
            let pv: usize = line.trim().parse().unwrap();
            let mut vertices = Vec::with_capacity(pv);
            for _ in 0..pv {
                let line = lines.next().unwrap();
                let coords: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
                vertices.push(Point { x: coords[0], y: coords[1] });
            }
            pieces.push(Piece::new(i, vertices));
        }

        let results = solve(stock_vertices, pieces);

        let max_stock = results.iter().map(|r| r.1).max().unwrap_or(0);
        println!("{}", max_stock + 1);
        for r in results {
            println!("{} {} {} {}", r.1, r.2, r.3, r.4);
        }
    }
}
