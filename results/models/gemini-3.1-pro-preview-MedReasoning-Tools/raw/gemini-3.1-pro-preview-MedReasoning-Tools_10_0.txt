#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct Item {
    int id;
    int w, h;
    int board_idx;
    int x, y;
    bool rotated;
};

bool compare_items(const Item& a, const Item& b) {
    int max_a = max(a.w, a.h);
    int max_b = max(b.w, b.h);
    if (max_a != max_b) return max_a > max_b;

    int min_a = min(a.w, a.h);
    int min_b = min(b.w, b.h);
    if (min_a != min_b) return min_a > min_b;

    long long area_a = (long long)a.w * a.h;
    long long area_b = (long long)b.w * b.h;
    if (area_a != area_b) return area_a > area_b;

    return a.id < b.id;
}

struct FreeRect { int x, y, w, h; };

bool merge_rects(FreeRect& a, const FreeRect& b) {
    if (a.x == b.x && a.w == b.w) {
        if (a.y + a.h == b.y) { a.h += b.h; return true; }
        if (b.y + b.h == a.y) { a.y = b.y; a.h += b.h; return true; }
    }
    if (a.y == b.y && a.h == b.h) {
        if (a.x + a.w == b.x) { a.w += b.w; return true; }
        if (b.x + b.w == a.x) { a.x = b.x; a.w += b.w; return true; }
    }
    return false;
}

class Bin {
public:
    vector<FreeRect> free_rects;
    int max_w, max_h;

    Bin(int w, int h) {
        free_rects.push_back({0, 0, w, h});
        max_w = w;
        max_h = h;
    }

    void update_maxes() {
        max_w = 0;
        max_h = 0;
        for (const auto& fr : free_rects) {
            max_w = max(max_w, fr.w);
            max_h = max(max_h, fr.h);
        }
    }

    void merge_free_rects() {
        bool merged = true;
        while (merged) {
            merged = false;
            for (size_t i = 0; i < free_rects.size(); ++i) {
                for (size_t j = i + 1; j < free_rects.size(); ++j) {
                    if (merge_rects(free_rects[i], free_rects[j])) {
                        free_rects.erase(free_rects.begin() + j);
                        merged = true;
                        break;
                    }
                }
                if (merged) break;
            }
        }
    }

    bool can_fit_roughly(int iw, int ih) const {
        return (iw <= max_w && ih <= max_h) || (iw <= max_h && ih <= max_w);
    }

    bool insert(int iw, int ih, int& out_x, int& out_y, bool& out_rotated) {
        if (!can_fit_roughly(iw, ih)) return false;

        int best_idx = -1;
        int best_score1 = 2e9;
        int best_score2 = 2e9;
        bool rotated = false;

        for (int i = 0; i < (int)free_rects.size(); ++i) {
            const auto& fr = free_rects[i];
            
            if (iw <= fr.w && ih <= fr.h) {
                int score1 = min(fr.w - iw, fr.h - ih);
                int score2 = max(fr.w - iw, fr.h - ih);
                if (score1 < best_score1 || (score1 == best_score1 && score2 < best_score2)) {
                    best_score1 = score1;
                    best_score2 = score2;
                    best_idx = i;
                    rotated = false;
                }
            }
            
            if (ih <= fr.w && iw <= fr.h) {
                int score1 = min(fr.w - ih, fr.h - iw);
                int score2 = max(fr.w - ih, fr.h - iw);
                if (score1 < best_score1 || (score1 == best_score1 && score2 < best_score2)) {
                    best_score1 = score1;
                    best_score2 = score2;
                    best_idx = i;
                    rotated = true;
                }
            }
        }

        if (best_idx == -1) return false;

        FreeRect best = free_rects[best_idx];
        out_x = best.x;
        out_y = best.y;
        out_rotated = rotated;

        int rw = rotated ? ih : iw;
        int rh = rotated ? iw : ih;

        free_rects.erase(free_rects.begin() + best_idx);

        int w_left = best.w - rw;
        int h_left = best.h - rh;

        FreeRect r1, r2;
        int max_area_v = max(w_left * best.h, rw * h_left);
        int max_area_h = max(w_left * rh, best.w * h_left);

        if (max_area_v > max_area_h) {
            r1 = {best.x + rw, best.y, w_left, best.h};
            r2 = {best.x, best.y + rh, rw, h_left};
        } else {
            r1 = {best.x + rw, best.y, w_left, rh};
            r2 = {best.x, best.y + rh, best.w, h_left};
        }

        if (r1.w > 0 && r1.h > 0) free_rects.push_back(r1);
        if (r2.w > 0 && r2.h > 0) free_rects.push_back(r2);

        merge_free_rects();
        update_maxes();

        return true;
    }
};

struct SegNode {
    int max_w, max_h;
};

vector<SegNode> tree;
vector<Bin> bins;

void update_tree(int node, int L, int R, int idx, int mw, int mh) {
    if (L == R) {
        tree[node].max_w = mw;
        tree[node].max_h = mh;
        return;
    }
    int mid = L + (R - L) / 2;
    if (idx <= mid) update_tree(2 * node, L, mid, idx, mw, mh);
    else update_tree(2 * node + 1, mid + 1, R, idx, mw, mh);
    
    tree[node].max_w = max(tree[2 * node].max_w, tree[2 * node + 1].max_w);
    tree[node].max_h = max(tree[2 * node].max_h, tree[2 * node + 1].max_h);
}

bool search_and_insert(int node, int L, int R, int max_valid_idx, int iw, int ih, int& out_bin, int& out_x, int& out_y, bool& out_rotated) {
    if (L > max_valid_idx) return false;
    
    if ((iw > tree[node].max_w || ih > tree[node].max_h) && 
        (ih > tree[node].max_w || iw > tree[node].max_h)) {
        return false;
    }
    
    if (L == R) {
        if (bins[L].insert(iw, ih, out_x, out_y, out_rotated)) {
            out_bin = L;
            tree[node].max_w = bins[L].max_w;
            tree[node].max_h = bins[L].max_h;
            return true;
        }
        return false;
    }
    
    int mid = L + (R - L) / 2;
    if (search_and_insert(2 * node, L, mid, max_valid_idx, iw, ih, out_bin, out_x, out_y, out_rotated)) {
        tree[node].max_w = max(tree[2 * node].max_w, tree[2 * node + 1].max_w);
        tree[node].max_h = max(tree[2 * node].max_h, tree[2 * node + 1].max_h);
        return true;
    }
    if (mid + 1 <= max_valid_idx) {
        if (search_and_insert(2 * node + 1, mid + 1, R, max_valid_idx, iw, ih, out_bin, out_x, out_y, out_rotated)) {
            tree[node].max_w = max(tree[2 * node].max_w, tree[2 * node + 1].max_w);
            tree[node].max_h = max(tree[2 * node].max_h, tree[2 * node + 1].max_h);
            return true;
        }
    }
    return false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, board_width, board_height;
    if (!(cin >> N >> board_width >> board_height)) return 0;
    if (N == 0) {
        cout << "0\n";
        return 0;
    }

    vector<Item> items(N);
    for (int i = 0; i < N; ++i) {
        items[i].id = i;
        cin >> items[i].w >> items[i].h;
    }

    vector<Item> sorted_items = items;
    sort(sorted_items.begin(), sorted_items.end(), compare_items);

    tree.resize(4 * N, {0, 0});
    bins.reserve(N);

    int num_bins = 0;

    for (int i = 0; i < N; ++i) {
        int iw = sorted_items[i].w;
        int ih = sorted_items[i].h;
        int out_bin = -1, out_x = 0, out_y = 0;
        bool out_rotated = false;

        bool inserted = false;
        if (num_bins > 0) {
            inserted = search_and_insert(1, 0, N - 1, num_bins - 1, iw, ih, out_bin, out_x, out_y, out_rotated);
        }

        if (!inserted) {
            out_bin = num_bins++;
            bins.push_back(Bin(board_width, board_height));
            bins[out_bin].insert(iw, ih, out_x, out_y, out_rotated);
            update_tree(1, 0, N - 1, out_bin, bins[out_bin].max_w, bins[out_bin].max_h);
        }

        sorted_items[i].board_idx = out_bin;
        sorted_items[i].x = out_x;
        sorted_items[i].y = out_y;
        sorted_items[i].rotated = out_rotated;
    }

    sort(sorted_items.begin(), sorted_items.end(), [](const Item& a, const Item& b) {
        return a.id < b.id;
    });

    cout << num_bins << "\n";
    for (int i = 0; i < N; ++i) {
        cout << sorted_items[i].board_idx << " " 
             << sorted_items[i].x << " " 
             << sorted_items[i].y << " " 
             << (sorted_items[i].rotated ? 1 : 0) << "\n";
    }

    return 0;
}
