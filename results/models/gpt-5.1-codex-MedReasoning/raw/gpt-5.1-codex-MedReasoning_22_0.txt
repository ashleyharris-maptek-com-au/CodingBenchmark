using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

internal static class Program
{
    private static void Main()
    {
        var scanner = new FastScanner();
        string? widthToken = scanner.Next();
        string? heightToken = scanner.Next();
        string? pieceCountToken = scanner.Next();

        if (widthToken == null || heightToken == null || pieceCountToken == null)
        {
            return;
        }

        if (!int.TryParse(widthToken, out int width) ||
            !int.TryParse(heightToken, out int height))
        {
            return;
        }

        if (!int.TryParse(pieceCountToken, out int totalPieces))
        {
            totalPieces = -1;
        }

        var solver = new TetrisSolver(width, height);
        int processedPieces = 0;

        while (totalPieces < 0 || processedPieces < totalPieces)
        {
            string? pieceToken = scanner.Next();
            if (pieceToken == null)
            {
                break;
            }

            char pieceType = char.ToUpperInvariant(pieceToken[0]);
            MoveDecision decision = solver.DecideMove(pieceType);

            Console.WriteLine("{0} {1}", decision.Rotation, decision.Column);
            Console.Out.Flush();

            string? resultToken = scanner.Next();
            if (resultToken == null)
            {
                break;
            }

            solver.CommitMove(pieceType, decision);
            processedPieces++;

            if (resultToken.StartsWith("gameover", StringComparison.OrdinalIgnoreCase))
            {
                break;
            }
        }
    }
}

internal struct MoveDecision
{
    public int Rotation;
    public int Column;
    public int LandingY;
    public int ExpectedLines;
    public double Score;
    public bool IsValid => LandingY >= 0;
}

internal readonly struct Cell
{
    public int X { get; }
    public int Y { get; }
    public Cell(int x, int y)
    {
        X = x;
        Y = y;
    }
}

internal sealed class PieceRotation
{
    public Cell[] Cells { get; }
    public int Width { get; }
    public int Height { get; }

    public PieceRotation(Cell[] cells, int width, int height)
    {
        Cells = cells;
        Width = width;
        Height = height;
    }
}

internal sealed class PieceDefinition
{
    public PieceRotation[] Rotations { get; }
    public PieceDefinition(PieceRotation[] rotations)
    {
        Rotations = rotations;
    }
}

internal static class PieceFactory
{
    public static Dictionary<char, PieceDefinition> CreateDefinitions()
    {
        return new Dictionary<char, PieceDefinition>
        {
            ['I'] = BuildPiece(new[] { "####" }),
            ['O'] = BuildPiece(new[] { "##", "##" }),
            ['T'] = BuildPiece(new[] { ".#.", "###" }),
            ['S'] = BuildPiece(new[] { ".##", "##." }),
            ['Z'] = BuildPiece(new[] { "##.", ".##" }),
            ['J'] = BuildPiece(new[] { "#..", "###" }),
            ['L'] = BuildPiece(new[] { "..#", "###" })
        };
    }

    private static PieceDefinition BuildPiece(string[] baseLayout)
    {
        var rotations = new PieceRotation[4];
        var current = Trim(baseLayout);
        for (int i = 0; i < 4; i++)
        {
            rotations[i] = BuildRotation(current);
            current = Trim(Rotate(current));
        }
        return new PieceDefinition(rotations);
    }

    private static string[] Rotate(string[] layout)
    {
        int rows = layout.Length;
        int cols = layout[0].Length;
        var rotated = new string[cols];
        for (int col = 0; col < cols; col++)
        {
            char[] newRow = new char[rows];
            for (int row = 0; row < rows; row++)
            {
                newRow[row] = layout[rows - 1 - row][col];
            }
            rotated[col] = new string(newRow);
        }
        return rotated;
    }

    private static string[] Trim(string[] layout)
    {
        int top = 0;
        int bottom = layout.Length - 1;
        while (top <= bottom && IsRowEmpty(layout[top]))
        {
            top++;
        }

        while (bottom >= top && IsRowEmpty(layout[bottom]))
        {
            bottom--;
        }

        if (bottom < top)
        {
            return new[] { "" };
        }

        int rows = bottom - top + 1;
        int cols = layout[0].Length;
        int left = 0;
        int right = cols - 1;

        while (left <= right && IsColumnEmpty(layout, top, bottom, left))
        {
            left++;
        }

        while (right >= left && IsColumnEmpty(layout, top, bottom, right))
        {
            right--;
        }

        int newCols = Math.Max(1, right - left + 1);
        var trimmed = new string[rows];
        for (int r = 0; r < rows; r++)
        {
            trimmed[r] = layout[top + r].Substring(left, newCols);
        }

        return trimmed;
    }

    private static bool IsRowEmpty(string row)
    {
        for (int i = 0; i < row.Length; i++)
        {
            char c = row[i];
            if (c != '.' && c != ' ')
            {
                return false;
            }
        }
        return true;
    }

    private static bool IsColumnEmpty(string[] layout, int top, int bottom, int column)
    {
        for (int r = top; r <= bottom; r++)
        {
            var row = layout[r];
            if (column < row.Length)
            {
                char c = row[column];
                if (c != '.' && c != ' ')
                {
                    return false;
                }
            }
        }
        return true;
    }

    private static PieceRotation BuildRotation(string[] layout)
    {
        var cells = new List<Cell>();
        int rows = layout.Length;
        for (int r = 0; r < rows; r++)
        {
            string row = layout[r];
            for (int c = 0; c < row.Length; c++)
            {
                if (row[c] != '.' && row[c] != ' ')
                {
                    int cellX = c;
                    int cellY = rows - 1 - r;
                    cells.Add(new Cell(cellX, cellY));
                }
            }
        }

        if (cells.Count == 0)
        {
            return new PieceRotation(Array.Empty<Cell>(), 0, 0);
        }

        int minX = cells[0].X;
        int maxX = cells[0].X;
        int minY = cells[0].Y;
        int maxY = cells[0].Y;

        for (int i = 1; i < cells.Count; i++)
        {
            var cell = cells[i];
            if (cell.X < minX) minX = cell.X;
            if (cell.X > maxX) maxX = cell.X;
            if (cell.Y < minY) minY = cell.Y;
            if (cell.Y > maxY) maxY = cell.Y;
        }

        var normalized = new Cell[cells.Count];
        for (int i = 0; i < cells.Count; i++)
        {
            normalized[i] = new Cell(cells[i].X - minX, cells[i].Y - minY);
        }

        int width = maxX - minX + 1;
        int height = maxY - minY + 1;
        return new PieceRotation(normalized, width, height);
    }
}

internal sealed class TetrisSolver
{
    private readonly int width;
    private readonly int height;
    private readonly bool[,] board;
    private readonly bool[,] tempBoard;
    private readonly int[] columnBuffer;
    private readonly Dictionary<char, PieceDefinition> pieces;

    public TetrisSolver(int width, int height)
    {
        this.width = width;
        this.height = height;
        board = new bool[height, width];
        tempBoard = new bool[height, width];
        columnBuffer = new int[width];
        pieces = PieceFactory.CreateDefinitions();
    }

    public MoveDecision DecideMove(char pieceType)
    {
        if (!pieces.TryGetValue(pieceType, out var piece))
        {
            piece = pieces['I'];
        }

        MoveDecision best = new MoveDecision
        {
            Rotation = 0,
            Column = Clamp(width / 2, 0, Math.Max(0, width - 1)),
            LandingY = -1,
            Score = double.NegativeInfinity
        };

        for (int rotationIndex = 0; rotationIndex < piece.Rotations.Length; rotationIndex++)
        {
            var rotation = piece.Rotations[rotationIndex];
            if (rotation == null || rotation.Width == 0 || rotation.Height == 0)
            {
                continue;
            }

            int maxColumn = width - rotation.Width;
            if (maxColumn < 0)
            {
                continue;
            }

            for (int column = 0; column <= maxColumn; column++)
            {
                int landingY = ComputeLanding(board, rotation, column);
                if (landingY < 0)
                {
                    continue;
                }

                CopyBoard(board, tempBoard);
                PlaceOnBoard(tempBoard, rotation, column, landingY);
                int linesCleared = ClearLines(tempBoard);
                double score = EvaluateBoard(tempBoard, linesCleared, rotation, landingY, column);

                if (score > best.Score)
                {
                    best.Score = score;
                    best.Rotation = rotationIndex;
                    best.Column = column;
                    best.LandingY = landingY;
                    best.ExpectedLines = linesCleared;
                }
            }
        }

        if (best.LandingY < 0)
        {
            var rotation = piece.Rotations[0];
            int maxColumn = Math.Max(0, width - Math.Max(1, rotation.Width));
            int fallbackColumn = Clamp(width / 2, 0, maxColumn);
            int landingY = ComputeLanding(board, rotation, fallbackColumn);
            best.Rotation = 0;
            best.Column = fallbackColumn;
            best.LandingY = landingY;
        }

        return best;
    }

    public void CommitMove(char pieceType, MoveDecision decision)
    {
        if (!pieces.TryGetValue(pieceType, out var piece))
        {
            piece = pieces['I'];
        }

        int rotationIndex = ((decision.Rotation % piece.Rotations.Length) + piece.Rotations.Length) % piece.Rotations.Length;
        var rotation = piece.Rotations[rotationIndex];
        if (rotation == null || rotation.Width == 0 || rotation.Height == 0)
        {
            return;
        }

        int landingY = decision.LandingY;
        if (landingY < 0)
        {
            landingY = ComputeLanding(board, rotation, decision.Column);
            if (landingY < 0)
            {
                return;
            }
        }

        PlaceOnBoard(board, rotation, decision.Column, landingY);
        ClearLines(board);
    }

    private int ComputeLanding(bool[,] state, PieceRotation rotation, int column)
    {
        if (column < 0 || column + rotation.Width > width)
        {
            return -1;
        }

        int y = height + rotation.Height + 1;
        int lastSafe = int.MinValue;
        bool hasSafe = false;
        int steps = height + rotation.Height + 5;

        while (steps-- > 0)
        {
            y--;
            bool collision = false;
            foreach (var cell in rotation.Cells)
            {
                int x = column + cell.X;
                if (x < 0 || x >= width)
                {
                    collision = true;
                    break;
                }

                int targetY = y + cell.Y;
                if (targetY < 0)
                {
                    collision = true;
                    break;
                }

                if (targetY < height && state[targetY, x])
                {
                    collision = true;
                    break;
                }
            }

            if (!collision)
            {
                lastSafe = y;
                hasSafe = true;
                continue;
            }

            break;
        }

        if (!hasSafe)
        {
            return -1;
        }

        foreach (var cell in rotation.Cells)
        {
            int targetY = lastSafe + cell.Y;
            if (targetY < 0 || targetY >= height)
            {
                return -1;
            }
        }

        return lastSafe;
    }

    private void CopyBoard(bool[,] source, bool[,] destination)
    {
        for (int y = 0; y < height; y++)
        {
            for (int x = 0; x < width; x++)
            {
                destination[y, x] = source[y, x];
            }
        }
    }

    private void PlaceOnBoard(bool[,] state, PieceRotation rotation, int column, int landingY)
    {
        foreach (var cell in rotation.Cells)
        {
            int x = column + cell.X;
            int y = landingY + cell.Y;
            if (x >= 0 && x < width && y >= 0 && y < height)
            {
                state[y, x] = true;
            }
        }
    }

    private int ClearLines(bool[,] state)
    {
        int cleared = 0;
        for (int y = 0; y < height; y++)
        {
            bool full = true;
            for (int x = 0; x < width; x++)
            {
                if (!state[y, x])
                {
                    full = false;
                    break;
                }
            }

            if (full)
            {
                cleared++;
                continue;
            }

            if (cleared > 0)
            {
                for (int x = 0; x < width; x++)
                {
                    state[y - cleared, x] = state[y, x];
                }
            }
        }

        if (cleared > 0)
        {
            for (int y = height - cleared; y < height; y++)
            {
                for (int x = 0; x < width; x++)
                {
                    state[y, x] = false;
                }
            }
        }

        return cleared;
    }

    private double EvaluateBoard(bool[,] state, int linesCleared, PieceRotation rotation, int landingY, int column)
    {
        int aggregateHeight = 0;
        int maxHeight = 0;
        int holes = 0;
        int blockades = 0;

        for (int x = 0; x < width; x++)
        {
            int h = 0;
            for (int y = height - 1; y >= 0; y--)
            {
                if (state[y, x])
                {
                    h = y + 1;
                    break;
                }
            }

            columnBuffer[x] = h;
            aggregateHeight += h;
            if (h > maxHeight)
            {
                maxHeight = h;
            }

            int columnHeight = h;
            int holesInColumn = 0;
            for (int y = columnHeight - 1; y >= 0; y--)
            {
                if (!state[y, x])
                {
                    holesInColumn++;
                }
                else
                {
                    blockades += holesInColumn;
                }
            }

            holes += holesInColumn;
        }

        int bumpiness = 0;
        for (int x = 0; x < width - 1; x++)
        {
            bumpiness += Math.Abs(columnBuffer[x] - columnBuffer[x + 1]);
        }

        int wellDepth = 0;
        for (int x = 0; x < width; x++)
        {
            int left = x == 0 ? height : columnBuffer[x - 1];
            int right = x == width - 1 ? height : columnBuffer[x + 1];
            int deep = Math.Min(left, right) - columnBuffer[x];
            if (deep > 0)
            {
                wellDepth += deep;
            }
        }

        int rowTransitions = 0;
        for (int y = 0; y < height; y++)
        {
            bool prev = true;
            for (int x = 0; x < width; x++)
            {
                bool filled = state[y, x];
                if (filled != prev)
                {
                    rowTransitions++;
                }
                prev = filled;
            }

            if (!prev)
            {
                rowTransitions++;
            }
        }

        int colTransitions = 0;
        for (int x = 0; x < width; x++)
        {
            bool prev = true;
            for (int y = 0; y < height; y++)
            {
                bool filled = state[y, x];
                if (filled != prev)
                {
                    colTransitions++;
                }
                prev = filled;
            }

            if (!prev)
            {
                colTransitions++;
            }
        }

        double pressure = height == 0 ? 0.0 : (double)maxHeight / height;
        double holeNormalized = Math.Min(1.0, holes / (double)Math.Max(1, width * 3));
        double complexity = Clamp((pressure + holeNormalized) * 0.5, 0.0, 1.0);
        double survival = 0.6 + 0.4 * complexity;
        double aggression = 1.3 - 0.8 * complexity;

        double score = 0.0;
        double lineReward = 8.0 + aggression * 4.0;
        score += lineReward * linesCleared;
        if (linesCleared >= 4)
        {
            score += 5.0 * aggression;
        }

        score += (-0.25 - 0.45 * survival) * aggregateHeight;
        score += (-0.5 - 0.9 * survival) * maxHeight;
        score += (-3.5 - 4.5 * survival) * holes;
        score += (-0.2 - 0.25 * survival) * bumpiness;
        score += (-0.1 - 0.3 * survival) * wellDepth;
        score += (-0.18 - 0.32 * survival) * rowTransitions;
        score += (-0.16 - 0.3 * survival) * colTransitions;
        score += (-0.12 - 0.3 * survival) * blockades;

        double landingTop = landingY + rotation.Height;
        double landingNorm = landingTop / Math.Max(1.0, height);
        score += (-1.0 - 0.8 * survival) * landingNorm;

        double boardCenter = (width - 1) / 2.0;
        double pieceCenter = column + (rotation.Width - 1) / 2.0;
        double centerDenom = Math.Max(1.0, (width - 1) * 0.5);
        double centerBias = Clamp(1.0 - Math.Abs(pieceCenter - boardCenter) / centerDenom, -0.25, 1.0);
        score += centerBias * (0.4 + 0.3 * aggression);

        return score;
    }

    private static int Clamp(int value, int min, int max)
    {
        if (value < min) return min;
        if (value > max) return max;
        return value;
    }

    private static double Clamp(double value, double min, double max)
    {
        if (value < min) return min;
        if (value > max) return max;
        return value;
    }
}

internal sealed class FastScanner
{
    private readonly Stream stream = Console.OpenStandardInput();
    private readonly byte[] buffer = new byte[1 << 16];
    private int length;
    private int pointer;

    public string? Next()
    {
        var sb = new StringBuilder();
        int c;

        do
        {
            c = Read();
            if (c == -1)
            {
                return null;
            }
        } while (char.IsWhiteSpace((char)c));

        do
        {
            sb.Append((char)c);
            c = Read();
        } while (c != -1 && !char.IsWhiteSpace((char)c));

        return sb.ToString();
    }

    private int Read()
    {
        if (pointer >= length)
        {
            length = stream.Read(buffer, 0, buffer.Length);
            pointer = 0;
            if (length <= 0)
            {
                return -1;
            }
        }

        return buffer[pointer++];
    }
}
