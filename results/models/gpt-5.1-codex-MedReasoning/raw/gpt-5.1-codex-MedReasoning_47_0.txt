@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>             params   : vec4<u32>;

fn rotl(x : u32, s : u32) -> u32 {
    return (x << s) | (x >> (32u - s));
}

fn trailingZeros64(h0 : u32, h1 : u32) -> u32 {
    var count : u32 = 0u;
    var value : u32 = h0;

    if (value == 0u) {
        count = count + 32u;
        value = h1;
        if (value == 0u) {
            return 64u;
        }
    }

    loop {
        if ((value & 1u) != 0u) {
            break;
        }
        count = count + 1u;
        value = value >> 1u;
    }

    return count;
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    let tid : u32 = gid.x;
    if (tid >= params.z) {
        return;
    }

    let nonce : u32 = params.y + tid;

    var state : array<u32, 16>;
    for (var i : u32 = 0u; i < 16u; i = i + 1u) {
        state[i] = baseData[i] ^ (nonce + i);
    }

    for (var round : u32 = 0u; round < 8u; round = round + 1u) {
        // state[0..3]
        var a0 : u32 = state[0];
        var b0 : u32 = state[1];
        var c0 : u32 = state[2];
        var d0 : u32 = state[3];
        a0 = a0 + b0; d0 = rotl(d0 ^ a0, 16u);
        c0 = c0 + d0; b0 = rotl(b0 ^ c0, 12u);
        a0 = a0 + b0; d0 = rotl(d0 ^ a0, 8u);
        c0 = c0 + d0; b0 = rotl(b0 ^ c0, 7u);
        state[0] = a0; state[1] = b0; state[2] = c0; state[3] = d0;

        // state[4..7]
        var a1 : u32 = state[4];
        var b1 : u32 = state[5];
        var c1 : u32 = state[6];
        var d1 : u32 = state[7];
        a1 = a1 + b1; d1 = rotl(d1 ^ a1, 16u);
        c1 = c1 + d1; b1 = rotl(b1 ^ c1, 12u);
        a1 = a1 + b1; d1 = rotl(d1 ^ a1, 8u);
        c1 = c1 + d1; b1 = rotl(b1 ^ c1, 7u);
        state[4] = a1; state[5] = b1; state[6] = c1; state[7] = d1;

        // state[8..11]
        var a2 : u32 = state[8];
        var b2 : u32 = state[9];
        var c2 : u32 = state[10];
        var d2 : u32 = state[11];
        a2 = a2 + b2; d2 = rotl(d2 ^ a2, 16u);
        c2 = c2 + d2; b2 = rotl(b2 ^ c2, 12u);
        a2 = a2 + b2; d2 = rotl(d2 ^ a2, 8u);
        c2 = c2 + d2; b2 = rotl(b2 ^ c2, 7u);
        state[8] = a2; state[9] = b2; state[10] = c2; state[11] = d2;

        // state[12..15]
        var a3 : u32 = state[12];
        var b3 : u32 = state[13];
        var c3 : u32 = state[14];
        var d3 : u32 = state[15];
        a3 = a3 + b3; d3 = rotl(d3 ^ a3, 16u);
        c3 = c3 + d3; b3 = rotl(b3 ^ c3, 12u);
        a3 = a3 + b3; d3 = rotl(d3 ^ a3, 8u);
        c3 = c3 + d3; b3 = rotl(b3 ^ c3, 7u);
        state[12] = a3; state[13] = b3; state[14] = c3; state[15] = d3;
    }

    var h0 : u32 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
    var h1 : u32 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];

    let tz : u32 = trailingZeros64(h0, h1);

    if (tz >= params.x) {
        result[0] = 1u;
        result[1] = nonce;
        result[2] = h0;
        result[3] = h1;
    }
}
