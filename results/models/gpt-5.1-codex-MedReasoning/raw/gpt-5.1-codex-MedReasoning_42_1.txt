OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color_in %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %arr_v3float ArrayStride 16
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%uint = OpTypeInt 32 0
%uint_1 = OpConstant %uint 1
%float_0_05 = OpConstant %float 0.0500000007

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_Input_v4 = OpTypePointer Input %v4float

%arr_v3float = OpTypeArray %v3float %uint_1
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3float
%ptr_Input_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_glpv = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Output_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_Input_arr_pv Input
%worldPos = OpVariable %ptr_Input_arr_v3 Input
%normal = OpVariable %ptr_Input_arr_v3 Input
%color_in = OpVariable %ptr_Input_arr_v3 Input
%gl_out = OpVariable %ptr_Output_glpv Output
%outColor = OpVariable %ptr_Output_v3 Output

%main = OpFunction %void None %func
%label = OpLabel
%pos_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%basePos = OpLoad %v4float %pos_ptr
%color_ptr = OpAccessChain %ptr_Input_v3 %color_in %int_0
%color_val = OpLoad %v3float %color_ptr
%out_pos_ptr = OpAccessChain %ptr_Output_v4 %gl_out %int_0
%base_x = OpCompositeExtract %float %basePos 0
%base_y = OpCompositeExtract %float %basePos 1
%base_z = OpCompositeExtract %float %basePos 2
%base_w = OpCompositeExtract %float %basePos 3

%top_y = OpFAdd %float %base_y %float_0_05
%top_vec = OpCompositeConstruct %v4float %base_x %top_y %base_z %base_w
OpStore %out_pos_ptr %top_vec
OpStore %outColor %color_val
OpEmitVertex

%right_x = OpFAdd %float %base_x %float_0_05
%right_vec = OpCompositeConstruct %v4float %right_x %base_y %base_z %base_w
OpStore %out_pos_ptr %right_vec
OpStore %outColor %color_val
OpEmitVertex

%left_x = OpFSub %float %base_x %float_0_05
%left_vec = OpCompositeConstruct %v4float %left_x %base_y %base_z %base_w
OpStore %out_pos_ptr %left_vec
OpStore %outColor %color_val
OpEmitVertex

%bottom_y = OpFSub %float %base_y %float_0_05
%bottom_vec = OpCompositeConstruct %v4float %base_x %bottom_y %base_z %base_w
OpStore %out_pos_ptr %bottom_vec
OpStore %outColor %color_val
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
