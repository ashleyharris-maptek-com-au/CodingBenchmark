```glsl
#version 450

layout(local_size_x = 256) in;

// Input body data: [position.xyz, mass], [velocity.xyz, 0]
layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

// Output body data written in same layout
layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

// Simulation parameters: counts.x = N, physics = (dt, softening, G, unused)
layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4  physics;
};

void main()
{
    uint idx = gl_GlobalInvocationID.x;
    uint N   = counts.x;

    if (idx >= N)
        return;

    float dt        = physics.x;
    float softening = physics.y;
    float G         = physics.z;

    uint baseIndex = idx * 2u;
    vec4 posMass = inData[baseIndex];
    vec4 velData = inData[baseIndex + 1u];

    vec3 position = posMass.xyz;
    float mass    = posMass.w;
    vec3 velocity = velData.xyz;

    vec3 acceleration = vec3(0.0);

    for (uint j = 0u; j < N; ++j) {
        if (j == idx)
            continue;

        uint otherBase = j * 2u;
        vec4 otherPosMass = inData[otherBase];

        vec3 delta = otherPosMass.xyz - position;
        float distSqr = dot(delta, delta) + softening * softening;
        float invDist = inversesqrt(distSqr);
        float invDist3 = invDist * invDist * invDist;

        acceleration += G * otherPosMass.w * delta * invDist3;
    }

    velocity += acceleration * dt;
    position += velocity * dt;

    outData[baseIndex]       = vec4(position, mass);
    outData[baseIndex + 1u]  = vec4(velocity, 0.0);
}
```
