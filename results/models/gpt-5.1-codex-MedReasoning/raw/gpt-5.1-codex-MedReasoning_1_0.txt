#include <algorithm>
#include <cstdint>
#include <cstdio>
#include <cmath>
#include <limits>
#include <numeric>
#include <vector>

struct City {
    double x;
    double y;
};

class FastInput {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buffer[BUFSIZE];
    size_t idx;
    size_t len;

    inline int read() {
        if (idx >= len) {
            len = std::fread(buffer, 1, BUFSIZE, stdin);
            idx = 0;
            if (len == 0) {
                return EOF;
            }
        }
        return buffer[idx++];
    }

public:
    FastInput() : idx(0), len(0) {}

    bool readLong(long long& out) {
        int c = read();
        if (c == EOF) return false;
        while (c != '-' && (c < '0' || c > '9')) {
            c = read();
            if (c == EOF) return false;
        }
        int sign = 1;
        if (c == '-') {
            sign = -1;
            c = read();
        }
        long long val = 0;
        while (c >= '0' && c <= '9') {
            val = val * 10 + (c - '0');
            c = read();
        }
        out = val * sign;
        return true;
    }

    bool readDouble(double& out) {
        int c = read();
        if (c == EOF) return false;
        while (c <= ' ') {
            c = read();
            if (c == EOF) return false;
        }
        bool neg = false;
        if (c == '-') {
            neg = true;
            c = read();
        } else if (c == '+') {
            c = read();
        }
        double val = 0.0;
        while (c >= '0' && c <= '9') {
            val = val * 10.0 + (c - '0');
            c = read();
        }
        if (c == '.') {
            double base = 0.1;
            c = read();
            while (c >= '0' && c <= '9') {
                val += (c - '0') * base;
                base *= 0.1;
                c = read();
            }
        }
        if (c == 'e' || c == 'E') {
            int expSign = 1;
            c = read();
            if (c == '+') {
                c = read();
            } else if (c == '-') {
                expSign = -1;
                c = read();
            }
            int expVal = 0;
            while (c >= '0' && c <= '9') {
                expVal = expVal * 10 + (c - '0');
                c = read();
            }
            val *= std::pow(10.0, expSign * expVal);
        }
        out = neg ? -val : val;
        return true;
    }
};

class FastOutput {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buffer[BUFSIZE];
    size_t idx;

public:
    FastOutput() : idx(0) {}
    ~FastOutput() { flush(); }

    inline void pushChar(char c) {
        if (idx >= BUFSIZE) flush();
        buffer[idx++] = c;
    }

    void writeIndex(size_t value, char endChar) {
        if (value == 0) {
            pushChar('0');
        } else {
            char temp[32];
            int len = 0;
            while (value > 0) {
                temp[len++] = static_cast<char>('0' + (value % 10));
                value /= 10;
            }
            for (int i = len - 1; i >= 0; --i) {
                pushChar(temp[i]);
            }
        }
        pushChar(endChar);
    }

    void flush() {
        if (idx) {
            std::fwrite(buffer, 1, idx, stdout);
            idx = 0;
        }
    }
};

inline double squaredDistance(const City& a, const City& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

inline double euclideanDistance(const City& a, const City& b) {
    return std::sqrt(squaredDistance(a, b));
}

std::vector<int> bruteForceSmall(const std::vector<City>& cities) {
    size_t n = cities.size();
    std::vector<int> path(n);
    if (n == 0) return path;
    if (n == 1) {
        path[0] = 0;
        return path;
    }
    std::vector<int> perm(n - 1);
    std::iota(perm.begin(), perm.end(), 1);
    double best = std::numeric_limits<double>::infinity();
    do {
        double len = 0.0;
        int prev = 0;
        for (int idx : perm) {
            len += euclideanDistance(cities[prev], cities[idx]);
            prev = idx;
        }
        if (len < best) {
            best = len;
            path[0] = 0;
            for (size_t i = 0; i < perm.size(); ++i) {
                path[i + 1] = perm[i];
            }
        }
    } while (std::next_permutation(perm.begin(), perm.end()));
    return path;
}

std::vector<int> greedyNearestPath(const std::vector<City>& cities) {
    size_t n = cities.size();
    std::vector<int> path;
    path.reserve(n);
    std::vector<char> used(n, 0);
    path.push_back(0);
    used[0] = 1;
    for (size_t step = 1; step < n; ++step) {
        int current = path.back();
        const City& curCity = cities[current];
        double best = std::numeric_limits<double>::infinity();
        int bestIdx = -1;
        for (size_t j = 0; j < n; ++j) {
            if (!used[j]) {
                double d = squaredDistance(curCity, cities[j]);
                if (d < best) {
                    best = d;
                    bestIdx = static_cast<int>(j);
                }
            }
        }
        if (bestIdx == -1) {
            for (size_t j = 0; j < n; ++j) {
                if (!used[j]) {
                    bestIdx = static_cast<int>(j);
                    break;
                }
            }
        }
        used[bestIdx] = 1;
        path.push_back(bestIdx);
    }
    return path;
}

inline uint64_t part1by1(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8)) & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2)) & 0x3333333333333333ULL;
    v = (v | (v << 1)) & 0x5555555555555555ULL;
    return v;
}

inline uint64_t mortonCode(uint32_t x, uint32_t y) {
    return (part1by1(y) << 1) | part1by1(x);
}

void twoOpt(std::vector<int>& path, const std::vector<City>& cities, int maxIterations, int spanLimit) {
    int n = static_cast<int>(path.size());
    if (n < 4 || maxIterations <= 0) return;
    if (spanLimit <= 0 || spanLimit > n) spanLimit = n;
    spanLimit = std::max(1, std::min(spanLimit, n - 2));
    for (int iter = 0; iter < maxIterations; ++iter) {
        bool improved = false;
        for (int i = 1; i < n - 2; ++i) {
            int maxJ = std::min(n - 2, i + spanLimit);
            for (int j = i + 1; j <= maxJ; ++j) {
                int a = path[i - 1];
                int b = path[i];
                int c = path[j];
                int d = path[j + 1];
                const City& A = cities[a];
                const City& B = cities[b];
                const City& C = cities[c];
                const City& D = cities[d];
                double before = euclideanDistance(A, B) + euclideanDistance(C, D);
                double after = euclideanDistance(A, C) + euclideanDistance(B, D);
                if (after + 1e-9 < before) {
                    std::reverse(path.begin() + i, path.begin() + j + 1);
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }
}

std::vector<int> spaceFillingOrder(const std::vector<City>& cities, bool applyLocalOpt, int spanLimit, int passes) {
    size_t n = cities.size();
    std::vector<int> order(n);
    std::iota(order.begin(), order.end(), 0);
    if (n <= 1) return order;

    double minx = cities[0].x, maxx = cities[0].x;
    double miny = cities[0].y, maxy = cities[0].y;
    for (size_t i = 1; i < n; ++i) {
        minx = std::min(minx, cities[i].x);
        maxx = std::max(maxx, cities[i].x);
        miny = std::min(miny, cities[i].y);
        maxy = std::max(maxy, cities[i].y);
    }
    double rangex = maxx - minx;
    double rangey = maxy - miny;
    if (std::abs(rangex) < 1e-12) rangex = 1.0;
    if (std::abs(rangey) < 1e-12) rangey = 1.0;

    constexpr double SCALE = 4294967295.0;
    double scaleX = SCALE / rangex;
    double scaleY = SCALE / rangey;

    std::vector<uint64_t> keys(n);
    for (size_t i = 0; i < n; ++i) {
        double sx = (cities[i].x - minx) * scaleX;
        double sy = (cities[i].y - miny) * scaleY;
        if (sx < 0.0) sx = 0.0; else if (sx > SCALE) sx = SCALE;
        if (sy < 0.0) sy = 0.0; else if (sy > SCALE) sy = SCALE;
        uint32_t xi = static_cast<uint32_t>(sx + 0.5);
        uint32_t yi = static_cast<uint32_t>(sy + 0.5);
        keys[i] = mortonCode(xi, yi);
    }

    std::sort(order.begin(), order.end(), [&](int a, int b) {
        if (keys[a] != keys[b]) return keys[a] < keys[b];
        return a < b;
    });

    auto zeroPos = std::find(order.begin(), order.end(), 0);
    if (zeroPos != order.begin()) {
        std::rotate(order.begin(), zeroPos, order.end());
    }

    if (applyLocalOpt && passes > 0 && spanLimit > 0) {
        twoOpt(order, cities, passes, spanLimit);
    }

    return order;
}

int main() {
    FastInput input;
    FastOutput output;

    long long nLL;
    if (!input.readLong(nLL) || nLL <= 0) {
        return 0;
    }
    size_t N = static_cast<size_t>(nLL);
    std::vector<City> cities(N);
    for (size_t i = 0; i < N; ++i) {
        double x = 0.0, y = 0.0;
        input.readDouble(x);
        input.readDouble(y);
        cities[i] = City{x, y};
    }

    std::vector<int> path;
    if (N == 1) {
        path = {0};
    } else if (N <= 11) {
        path = bruteForceSmall(cities);
    } else if (N <= 2000) {
        path = greedyNearestPath(cities);
        int span = (N <= 400) ? static_cast<int>(N) : 400;
        int iterations = (N <= 400) ? 15 : 8;
        twoOpt(path, cities, iterations, span);
    } else {
        bool applyLocal = false;
        int span = 0;
        int passes = 0;
        if (N <= 50000) {
            applyLocal = true;
            span = (N <= 20000) ? 80 : 50;
            passes = (N <= 20000) ? 3 : 2;
        } else if (N <= 200000) {
            applyLocal = true;
            span = 20;
            passes = 1;
        }
        path = spaceFillingOrder(cities, applyLocal, span, passes);
    }

    if (!path.empty() && path[0] != 0) {
        auto zeroPos = std::find(path.begin(), path.end(), 0);
        if (zeroPos != path.end()) {
            std::rotate(path.begin(), zeroPos, path.end());
        }
    }

    for (size_t i = 0; i < path.size(); ++i) {
        char end = (i + 1 == path.size()) ? '\n' : ' ';
        output.writeIndex(static_cast<size_t>(path[i]), end);
    }
    return 0;
}
