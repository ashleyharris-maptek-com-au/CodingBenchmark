```csharp
using System;
using System.IO;
using System.Diagnostics;

class Program
{
    static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n;
        try
        {
            n = scanner.NextInt();
        }
        catch
        {
            return;
        }

        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        int[][] flow = new int[n][];
        int[][] dist = new int[n][];
        long[] flowRowSum = new long[n];
        long[] flowColSum = new long[n];
        long[] distRowSum = new long[n];
        long[] distColSum = new long[n];

        for (int i = 0; i < n; i++)
        {
            flow[i] = new int[n];
            long rowSum = 0;
            for (int j = 0; j < n; j++)
            {
                int val = scanner.NextInt();
                flow[i][j] = val;
                long absVal = Math.Abs((long)val);
                rowSum += absVal;
                flowColSum[j] += absVal;
            }
            flowRowSum[i] = rowSum;
        }

        for (int i = 0; i < n; i++)
        {
            dist[i] = new int[n];
            long rowSum = 0;
            for (int j = 0; j < n; j++)
            {
                int val = scanner.NextInt();
                dist[i][j] = val;
                long absVal = Math.Abs((long)val);
                rowSum += absVal;
                distColSum[j] += absVal;
            }
            distRowSum[i] = rowSum;
        }

        long[] facilityWeight = new long[n];
        long[] locationWeight = new long[n];
        for (int i = 0; i < n; i++)
        {
            facilityWeight[i] = flowRowSum[i] + flowColSum[i];
            locationWeight[i] = distRowSum[i] + distColSum[i];
        }

        int[] facilityOrder = BuildIndexArray(n);
        Array.Sort(facilityOrder, (a, b) =>
        {
            int cmp = facilityWeight[b].CompareTo(facilityWeight[a]);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });

        int[] locationOrderAsc = BuildIndexArray(n);
        Array.Sort(locationOrderAsc, (a, b) =>
        {
            int cmp = locationWeight[a].CompareTo(locationWeight[b]);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });
        int[] locationOrderDesc = (int[])locationOrderAsc.Clone();
        Array.Reverse(locationOrderDesc);

        var rng = new Random(1);
        int[] locationOrderShuffle = (int[])locationOrderAsc.Clone();
        Shuffle(locationOrderShuffle, rng);

        int topCount = Math.Min(n, Math.Max(5, n / 5));
        if (topCount > 80) topCount = 80;
        int[] topCandidates = new int[topCount];
        for (int i = 0; i < topCount; i++)
            topCandidates[i] = facilityOrder[i];

        long timeLimit = Math.Max(800, Math.Min(4800, 800 + n * 2));
        var stopwatch = Stopwatch.StartNew();

        long bestCost = long.MaxValue;
        int[] bestPerm = null;

        EvaluateCandidate(BuildPermutation(facilityOrder, locationOrderAsc), flow, dist, topCandidates, rng, stopwatch, timeLimit, ref bestCost, ref bestPerm);
        EvaluateCandidate(BuildPermutation(facilityOrder, locationOrderDesc), flow, dist, topCandidates, rng, stopwatch, timeLimit, ref bestCost, ref bestPerm);
        EvaluateCandidate(CreateIdentity(n), flow, dist, topCandidates, rng, stopwatch, timeLimit, ref bestCost, ref bestPerm);
        EvaluateCandidate(BuildPermutation(facilityOrder, locationOrderShuffle), flow, dist, topCandidates, rng, stopwatch, timeLimit, ref bestCost, ref bestPerm);

        if (bestPerm == null)
        {
            bestPerm = CreateIdentity(n);
            bestCost = CalculateCost(bestPerm, flow, dist);
        }

        Console.WriteLine(bestCost);
        Console.WriteLine(string.Join(" ", bestPerm));
    }

    static void EvaluateCandidate(int[] perm, int[][] flow, int[][] dist, int[] topCandidates, Random rng, Stopwatch sw, long timeLimit, ref long bestCost, ref int[] bestPerm)
    {
        long cost = CalculateCost(perm, flow, dist);
        if (bestPerm == null || cost < bestCost)
        {
            bestCost = cost;
            bestPerm = (int[])perm.Clone();
        }

        if (sw.ElapsedMilliseconds >= timeLimit)
            return;

        ImproveSolution(perm, flow, dist, topCandidates, rng, sw, timeLimit, cost, ref bestCost, ref bestPerm);
    }

    static long ImproveSolution(int[] perm, int[][] flow, int[][] dist, int[] topCandidates, Random rng, Stopwatch sw, long timeLimit, long currentCost, ref long bestCost, ref int[] bestPerm)
    {
        int n = perm.Length;
        long cost = currentCost;

        if (topCandidates != null && topCandidates.Length >= 2)
        {
            bool improved = true;
            while (improved && sw.ElapsedMilliseconds < timeLimit)
            {
                improved = false;
                for (int a = 0; a < topCandidates.Length && sw.ElapsedMilliseconds < timeLimit; a++)
                {
                    int i = topCandidates[a];
                    for (int b = a + 1; b < topCandidates.Length; b++)
                    {
                        int j = topCandidates[b];
                        long delta = SwapDelta(i, j, perm, flow, dist);
                        if (delta < 0)
                        {
                            Swap(perm, i, j);
                            cost += delta;
                            improved = true;
                            if (cost < bestCost)
                            {
                                bestCost = cost;
                                bestPerm = (int[])perm.Clone();
                            }
                        }
                    }
                }
            }
        }

        if (sw.ElapsedMilliseconds >= timeLimit)
            return cost;

        int iterationsPerRound = Math.Max(n * 3, 2000);
        while (sw.ElapsedMilliseconds < timeLimit)
        {
            bool improvedRound = false;
            int tries = iterationsPerRound;
            while (tries-- > 0 && sw.ElapsedMilliseconds < timeLimit)
            {
                int i = rng.Next(n);
                int j = rng.Next(n);
                if (i == j) continue;

                long delta = SwapDelta(i, j, perm, flow, dist);
                if (delta < 0)
                {
                    Swap(perm, i, j);
                    cost += delta;
                    improvedRound = true;
                    if (cost < bestCost)
                    {
                        bestCost = cost;
                        bestPerm = (int[])perm.Clone();
                    }
                }
            }
            if (!improvedRound)
                break;
        }

        return cost;
    }

    static long SwapDelta(int i, int j, int[] perm, int[][] flow, int[][] dist)
    {
        if (i == j) return 0;
        int n = perm.Length;
        int pi = perm[i];
        int pj = perm[j];
        long delta = 0;

        for (int k = 0; k < n; k++)
        {
            if (k == i || k == j) continue;
            int pk = perm[k];
            delta += (long)flow[i][k] * (dist[pj][pk] - dist[pi][pk]);
            delta += (long)flow[k][i] * (dist[pk][pj] - dist[pk][pi]);
            delta += (long)flow[j][k] * (dist[pi][pk] - dist[pj][pk]);
            delta += (long)flow[k][j] * (dist[pk][pi] - dist[pk][pj]);
        }

        delta += (long)flow[i][i] * (dist[pj][pj] - dist[pi][pi]);
        delta += (long)flow[j][j] * (dist[pi][pi] - dist[pj][pj]);
        delta += (long)flow[i][j] * (dist[pj][pi] - dist[pi][pj]);
        delta += (long)flow[j][i] * (dist[pi][pj] - dist[pj][pi]);

        return delta;
    }

    static long CalculateCost(int[] perm, int[][] flow, int[][] dist)
    {
        long total = 0;
        int n = perm.Length;
        for (int i = 0; i < n; i++)
        {
            int pi = perm[i];
            var flowRow = flow[i];
            var distRow = dist[pi];
            for (int j = 0; j < n; j++)
            {
                total += (long)flowRow[j] * distRow[perm[j]];
            }
        }
        return total;
    }

    static int[] BuildPermutation(int[] facilityOrder, int[] locationOrder)
    {
        int n = facilityOrder.Length;
        int[] perm = new int[n];
        for (int idx = 0; idx < n; idx++)
        {
            perm[facilityOrder[idx]] = locationOrder[idx];
        }
        return perm;
    }

    static int[] CreateIdentity(int n)
    {
        int[] perm = new int[n];
        for (int i = 0; i < n; i++) perm[i] = i;
        return perm;
    }

    static int[] BuildIndexArray(int n)
    {
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) arr[i] = i;
        return arr;
    }

    static void Shuffle(int[] array, Random rng)
    {
        for (int i = array.Length - 1; i > 0; i--)
        {
            int j = rng.Next(i + 1);
            int tmp = array[i];
            array[i] = array[j];
            array[j] = tmp;
        }
    }

    static void Swap(int[] arr, int i, int j)
    {
        int tmp = arr[i];
        arr[i] = arr[j];
        arr[j] = tmp;
    }

    sealed class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int ptr, len;

        public FastScanner(Stream stream) => this.stream = stream;

        private int Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) throw new InvalidOperationException("No more input");
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int result = 0;
            while (c > ' ')
            {
                result = result * 10 + (c - '0');
                c = Read();
            }
            return result * sign;
        }
    }
}
```
