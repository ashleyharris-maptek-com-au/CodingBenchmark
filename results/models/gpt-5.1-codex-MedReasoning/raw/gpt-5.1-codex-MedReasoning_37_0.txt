Here is complete C# code that reads a directed graph, decomposes it by strongly connected components, and for every cyclic component runs an exact branch-and-bound solver to find a minimum feedback vertex set.  
If the graph has no cycles, it outputs `0` and a blank second line as required.

```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    static void Main()
    {
        var scanner = new FastScanner();
        if (!scanner.TryReadInt(out int n))
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }
        int m = scanner.NextInt();

        var adj = new List<int>[n];
        var radj = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            adj[i] = new List<int>();
            radj[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            if (0 <= u && u < n && 0 <= v && v < n)
            {
                adj[u].Add(v);
                radj[v].Add(u);
            }
        }

        var components = GetStronglyConnectedComponents(adj, radj);
        var result = new List<int>();
        var globalToLocal = new int[n];
        for (int i = 0; i < n; i++)
        {
            globalToLocal[i] = -1;
        }

        foreach (var comp in components)
        {
            int size = comp.Count;
            if (size == 0)
            {
                continue;
            }

            bool hasCycle = false;
            if (size == 1)
            {
                int node = comp[0];
                var neighbors = adj[node];
                for (int j = 0; j < neighbors.Count; j++)
                {
                    if (neighbors[j] == node)
                    {
                        hasCycle = true;
                        break;
                    }
                }
            }
            else
            {
                hasCycle = true;
            }

            if (!hasCycle)
            {
                continue;
            }

            var localAdj = new List<int>[size];
            var localRadj = new List<int>[size];
            var localToGlobal = new int[size];

            for (int i = 0; i < size; i++)
            {
                int node = comp[i];
                localToGlobal[i] = node;
                globalToLocal[node] = i;
                localAdj[i] = new List<int>();
                localRadj[i] = new List<int>();
            }

            for (int i = 0; i < size; i++)
            {
                int uGlobal = localToGlobal[i];
                var neighbors = adj[uGlobal];
                for (int j = 0; j < neighbors.Count; j++)
                {
                    int vGlobal = neighbors[j];
                    int localIdx = globalToLocal[vGlobal];
                    if (localIdx != -1)
                    {
                        localAdj[i].Add(localIdx);
                        localRadj[localIdx].Add(i);
                    }
                }
            }

            var solver = new ComponentSolver(localAdj, localRadj);
            List<int> localSolution = solver.Solve();
            for (int i = 0; i < localSolution.Count; i++)
            {
                result.Add(localToGlobal[localSolution[i]]);
            }

            for (int i = 0; i < size; i++)
            {
                globalToLocal[localToGlobal[i]] = -1;
            }
        }

        result.Sort();
        var sb = new StringBuilder();
        sb.AppendLine(result.Count.ToString());
        if (result.Count > 0)
        {
            sb.AppendLine(string.Join(" ", result));
        }
        else
        {
            sb.AppendLine();
        }
        Console.Write(sb.ToString());
    }

    private static List<List<int>> GetStronglyConnectedComponents(List<int>[] adj, List<int>[] radj)
    {
        int n = adj.Length;
        var order = new List<int>(n);
        var visited = new bool[n];
        var stackNodes = new int[n];
        var stackEdgeIndex = new int[n];

        for (int start = 0; start < n; start++)
        {
            if (visited[start])
            {
                continue;
            }

            int stackSize = 0;
            stackNodes[stackSize] = start;
            stackEdgeIndex[stackSize] = 0;
            stackSize++;

            while (stackSize > 0)
            {
                int idx = stackSize - 1;
                int node = stackNodes[idx];
                if (!visited[node])
                {
                    visited[node] = true;
                }

                if (stackEdgeIndex[idx] < adj[node].Count)
                {
                    int to = adj[node][stackEdgeIndex[idx]++];
                    if (visited[to])
                    {
                        continue;
                    }

                    stackNodes[stackSize] = to;
                    stackEdgeIndex[stackSize] = 0;
                    stackSize++;
                }
                else
                {
                    order.Add(node);
                    stackSize--;
                }
            }
        }

        var visited2 = new bool[n];
        var components = new List<List<int>>();
        var stack = new int[n];

        for (int i = order.Count - 1; i >= 0; i--)
        {
            int node = order[i];
            if (visited2[node])
            {
                continue;
            }

            var comp = new List<int>();
            int stackSize = 0;
            stack[stackSize++] = node;
            visited2[node] = true;

            while (stackSize > 0)
            {
                int v = stack[--stackSize];
                comp.Add(v);
                var preds = radj[v];
                for (int j = 0; j < preds.Count; j++)
                {
                    int to = preds[j];
                    if (visited2[to])
                    {
                        continue;
                    }
                    visited2[to] = true;
                    stack[stackSize++] = to;
                }
            }

            components.Add(comp);
        }

        return components;
    }
}

class ComponentSolver
{
    private readonly int n;
    private readonly List<int>[] adj;
    private readonly List<int>[] radj;

    private readonly bool[] removed;
    private readonly List<int> currentSolution;
    private List<int> bestSolution;
    private int bestSize;

    private readonly int[] visitState;
    private readonly int[] stackPosition;
    private readonly int[] nextEdgeIndex;
    private readonly int[] dfsStack;
    private readonly List<int> cycleBuffer;

    private readonly bool[] lbVisited1;
    private readonly bool[] lbVisited2;
    private readonly int[] orderStackNodes;
    private readonly int[] orderStackEdge;
    private readonly int[] reverseStack;
    private readonly List<int> orderBuffer;
    private readonly List<int> componentBuffer;

    private bool[] tempRemovedForLb;
    private bool[] greedyRemoved;

    private readonly int[] branchScores;
    private const int CyclePackingIterationLimit = 80;

    public ComponentSolver(List<int>[] adj, List<int>[] radj)
    {
        this.adj = adj;
        this.radj = radj;
        n = adj.Length;

        removed = new bool[n];
        currentSolution = new List<int>(n);
        bestSolution = new List<int>(n);

        visitState = new int[n];
        stackPosition = new int[n];
        nextEdgeIndex = new int[n];
        dfsStack = new int[n];
        cycleBuffer = new List<int>(Math.Max(4, n));

        lbVisited1 = new bool[n];
        lbVisited2 = new bool[n];
        orderStackNodes = new int[n];
        orderStackEdge = new int[n];
        reverseStack = new int[n];
        orderBuffer = new List<int>(n);
        componentBuffer = new List<int>(n);

        branchScores = new int[n];
        if (n > 0)
        {
            Array.Fill(stackPosition, -1);
        }
    }

    public List<int> Solve()
    {
        if (n == 0)
        {
            return new List<int>();
        }

        List<int> greedy = BuildGreedy();
        bestSize = greedy.Count;
        bestSolution = new List<int>(greedy);
        if (bestSize == 0)
        {
            return new List<int>(bestSolution);
        }

        Array.Fill(removed, false);
        currentSolution.Clear();
        Search(0);
        return new List<int>(bestSolution);
    }

    private void Search(int removedCount)
    {
        if (removedCount >= bestSize)
        {
            return;
        }

        int lb = LowerBound(removedCount);
        if (removedCount + lb >= bestSize)
        {
            return;
        }

        List<int> cycle = FindCycle(removed);
        if (cycle == null)
        {
            bestSize = removedCount;
            bestSolution = new List<int>(currentSolution);
            return;
        }

        if (cycle.Count == 0)
        {
            return;
        }

        if (cycle.Count == 1)
        {
            int node = cycle[0];
            if (!removed[node])
            {
                removed[node] = true;
                currentSolution.Add(node);
                Search(removedCount + 1);
                currentSolution.RemoveAt(currentSolution.Count - 1);
                removed[node] = false;
            }
            return;
        }

        for (int i = 0; i < cycle.Count; i++)
        {
            int node = cycle[i];
            branchScores[node] = DegreeScore(node, removed);
        }

        cycle.Sort((a, b) => branchScores[b].CompareTo(branchScores[a]));

        foreach (int node in cycle)
        {
            if (removed[node])
            {
                continue;
            }

            removed[node] = true;
            currentSolution.Add(node);
            Search(removedCount + 1);
            currentSolution.RemoveAt(currentSolution.Count - 1);
            removed[node] = false;

            if (bestSize == removedCount + 1)
            {
                break;
            }
        }
    }

    private int LowerBound(int removedCount)
    {
        int lb1 = CyclicSccCount();
        int lb2 = DisjointCycleLowerBound(removedCount);
        return lb1 > lb2 ? lb1 : lb2;
    }

    private int CyclicSccCount()
    {
        Array.Fill(lbVisited1, false);
        orderBuffer.Clear();

        for (int start = 0; start < n; start++)
        {
            if (removed[start] || lbVisited1[start])
            {
                continue;
            }

            int stackSize = 0;
            orderStackNodes[stackSize] = start;
            orderStackEdge[stackSize] = 0;
            stackSize++;

            while (stackSize > 0)
            {
                int idx = stackSize - 1;
                int node = orderStackNodes[idx];
                if (!lbVisited1[node])
                {
                    lbVisited1[node] = true;
                }

                if (orderStackEdge[idx] < adj[node].Count)
                {
                    int to = adj[node][orderStackEdge[idx]++];
                    if (removed[to] || lbVisited1[to])
                    {
                        continue;
                    }

                    orderStackNodes[stackSize] = to;
                    orderStackEdge[stackSize] = 0;
                    stackSize++;
                }
                else
                {
                    orderBuffer.Add(node);
                    stackSize--;
                }
            }
        }

        Array.Fill(lbVisited2, false);
        int cycles = 0;

        for (int i = orderBuffer.Count - 1; i >= 0; i--)
        {
            int node = orderBuffer[i];
            if (removed[node] || lbVisited2[node])
            {
                continue;
            }

            componentBuffer.Clear();
            int stackSize = 0;
            reverseStack[stackSize++] = node;
            lbVisited2[node] = true;

            while (stackSize > 0)
            {
                int v = reverseStack[--stackSize];
                componentBuffer.Add(v);
                var preds = radj[v];
                for (int j = 0; j < preds.Count; j++)
                {
                    int to = preds[j];
                    if (removed[to] || lbVisited2[to])
                    {
                        continue;
                    }
                    lbVisited2[to] = true;
                    reverseStack[stackSize++] = to;
                }
            }

            bool hasCycle = false;
            if (componentBuffer.Count > 1)
            {
                hasCycle = true;
            }
            else
            {
                int only = componentBuffer[0];
                var outs = adj[only];
                for (int j = 0; j < outs.Count; j++)
                {
                    int to = outs[j];
                    if (removed[to])
                    {
                        continue;
                    }
                    if (to == only)
                    {
                        hasCycle = true;
                        break;
                    }
                }
            }

            if (hasCycle)
            {
                cycles++;
            }
        }

        return cycles;
    }

    private int DisjointCycleLowerBound(int removedCount)
    {
        int remainingGap = bestSize - removedCount;
        if (remainingGap <= 0)
        {
            return 0;
        }

        int limit = remainingGap;
        if (limit > CyclePackingIterationLimit)
        {
            limit = CyclePackingIterationLimit;
        }
        if (limit == 0)
        {
            return 0;
        }

        if (tempRemovedForLb == null || tempRemovedForLb.Length != n)
        {
            tempRemovedForLb = new bool[n];
        }
        Array.Copy(removed, tempRemovedForLb, n);

        int count = 0;
        while (count < limit)
        {
            List<int> cycle = FindCycle(tempRemovedForLb);
            if (cycle == null || cycle.Count == 0)
            do
            {
                break;
            }
            while (false);

            if (cycle == null || cycle.Count == 0)
            {
                break;
            }

            for (int i = 0; i < cycle.Count; i++)
            {
                tempRemovedForLb[cycle[i]] = true;
            }
            count++;
        }

        return count;
    }

    private List<int> BuildGreedy()
    {
        var result = new List<int>();
        if (n == 0)
        {
            return result;
        }

        if (greedyRemoved == null || greedyRemoved.Length != n)
        {
            greedyRemoved = new bool[n];
        }
        else
        {
            Array.Fill(greedyRemoved, false);
        }

        while (true)
        {
            List<int> cycle = FindCycle(greedyRemoved);
            if (cycle == null || cycle.Count == 0)
            {
                break;
            }

            int bestNode = cycle[0];
            int bestScore = DegreeScore(bestNode, greedyRemoved);
            for (int i = 1; i < cycle.Count; i++)
            {
                int node = cycle[i];
                int score = DegreeScore(node, greedyRemoved);
                if (score > bestScore)
                {
                    bestScore = score;
                    bestNode = node;
                }
            }

            greedyRemoved[bestNode] = true;
            result.Add(bestNode);
        }

        return result;
    }

    private int DegreeScore(int node, bool[] mask)
    {
        int score = 0;
        var outs = adj[node];
        for (int i = 0; i < outs.Count; i++)
        {
            if (!mask[outs[i]])
            {
                score++;
            }
        }

        var ins = radj[node];
        for (int i = 0; i < ins.Count; i++)
        {
            if (!mask[ins[i]])
            {
                score++;
            }
        }

        return score;
    }

    private List<int> FindCycle(bool[] mask)
    {
        if (n == 0)
        {
            return null;
        }

        Array.Fill(visitState, 0);
        int stackSize = 0;

        for (int start = 0; start < n; start++)
        {
            if (mask[start] || visitState[start] != 0)
            {
                continue;
            }

            stackSize = 0;
            dfsStack[stackSize++] = start;
            nextEdgeIndex[start] = 0;

            while (stackSize > 0)
            {
                int node = dfsStack[stackSize - 1];

                if (visitState[node] == 0)
                {
                    visitState[node] = 1;
                    stackPosition[node] = stackSize - 1;
                }

                if (nextEdgeIndex[node] < adj[node].Count)
                {
                    int to = adj[node][nextEdgeIndex[node]++];
                    if (mask[to])
                    {
                        continue;
                    }

                    if (visitState[to] == 0)
                    {
                        dfsStack[stackSize++] = to;
                        nextEdgeIndex[to] = 0;
                        continue;
                    }

                    if (visitState[to] == 1)
                    {
                        cycleBuffer.Clear();
                        for (int idx = stackPosition[to]; idx < stackSize; idx++)
                        {
                            cycleBuffer.Add(dfsStack[idx]);
                        }
                        return cycleBuffer;
                    }
                }
                else
                {
                    visitState[node] = 2;
                    stackPosition[node] = -1;
                    stackSize--;
                }
            }
        }

        return null;
    }
}

class FastScanner
{
    private readonly Stream stream = Console.OpenStandardInput();
    private readonly byte[] buffer = new byte[1 << 16];
    private int ptr;
    private int len;

    private int Read()
    {
        if (ptr >= len)
        {
            len = stream.Read(buffer, 0, buffer.Length);
            ptr = 0;
            if (len <= 0)
            {
                return -1;
            }
        }
        return buffer[ptr++];
    }

    public bool TryReadInt(out int value)
    {
        int c;
        while ((c = Read()) != -1 && c <= ' ')
        {
        }
        if (c == -1)
        {
            value = 0;
            return false;
        }

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        int result = 0;
        while (c > ' ')
        {
            result = result * 10 + (c - '0');
            c = Read();
        }

        value = result * sign;
        return true;
    }

    public int NextInt()
    {
        if (TryReadInt(out int value))
        {
            return value;
        }
        throw new InvalidOperationException("Unexpected end of input");
    }
}
```
