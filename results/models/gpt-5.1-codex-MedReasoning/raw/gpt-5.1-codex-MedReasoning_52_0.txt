Here's a robust autopilot implementation designed to handle all 25 challenging freeway scenarios. It performs sensor fusion across redundant sensors, maintains persistent state for filtering/PID control, handles obstacles, lane keeping, traction limits, and gracefully degrades under sensor failures.

```python
import math

# Persistent controller memory
autopilot_memory = {
    "speed_est_kmh": 0.0,
    "speed_error_int": 0.0,
    "lane_offset_filtered": 0.0,
    "last_speed_limit": 130.0,
    "mu_filtered": 0.9,
    "yaw_rate_dps": 0.0,
    "ttc": None,
    "desired_lane_offset": 0.0,
}


def _clamp(value, low, high):
    return max(low, min(high, value))


def _valid_number(val):
    return isinstance(val, (int, float)) and math.isfinite(val)


def _median(values):
    if not values:
        return None
    values = sorted(values)
    mid = len(values) // 2
    if len(values) % 2:
        return values[mid]
    return 0.5 * (values[mid - 1] + values[mid])


def autopilot_step(state: dict, dt: float) -> dict:
    global autopilot_memory

    if dt is None or dt <= 0:
        dt = state.get("dt", 0.05)
    dt = _clamp(dt, 0.01, 0.2)

    # ---- Speed fusion ----
    last_speed = autopilot_memory.get("speed_est_kmh", 0.0)
    speed_candidates = []

    def consider_speed(value, valid_flag=None):
        if not _valid_number(value):
            return
        if not (0 <= value <= 300):
            return
        if valid_flag is None:
            speed_candidates.append(value)
        elif valid_flag:
            speed_candidates.append(value)
        else:
            if abs(value - last_speed) < 15:
                speed_candidates.append(value)

    for wheel in ["fl", "fr", "rl", "rr"]:
        consider_speed(state.get(f"wheel_speed_{wheel}_kmh"),
                       state.get(f"wheel_speed_{wheel}_valid"))

    for gps_idx in [1, 2]:
        consider_speed(state.get(f"gps_{gps_idx}_speed_kmh"),
                       state.get(f"gps_{gps_idx}_valid"))

    for veh_idx in [1, 2]:
        consider_speed(state.get(f"vehicle_speed_{veh_idx}_kmh"), None)

    if speed_candidates:
        fused_speed = _median(speed_candidates)
        if _valid_number(last_speed):
            fused_speed = 0.6 * last_speed + 0.4 * fused_speed
    else:
        fused_speed = last_speed
    autopilot_memory["speed_est_kmh"] = fused_speed
    current_speed = fused_speed

    # ---- Yaw fusion ----
    yaw_candidates = []
    last_yaw = autopilot_memory.get("yaw_rate_dps", 0.0)

    def consider_yaw(value, valid_flag=None):
        if not _valid_number(value):
            return
        if abs(value) > 90:
            return
        if valid_flag is None or valid_flag:
            yaw_candidates.append(value)
        elif abs(value - last_yaw) < 10:
            yaw_candidates.append(value)

    for idx in [1, 2, 3]:
        consider_yaw(state.get(f"gyro_{idx}_yaw_rate_dps"),
                     state.get(f"gyro_{idx}_valid"))
    yaw_rate = _median(yaw_candidates) if yaw_candidates else last_yaw
    autopilot_memory["yaw_rate_dps"] = yaw_rate

    # ---- Lane estimation ----
    lane_candidates = []
    for idx in [1, 2, 3]:
        lane = state.get(f"vision_{idx}_lane_offset_m")
        if _valid_number(lane) and abs(lane) <= 5:
            lane_candidates.append(lane)

        left = state.get(f"vision_{idx}_left_line_dist_m")
        right = state.get(f"vision_{idx}_right_line_dist_m")
        left_valid = state.get(f"vision_{idx}_left_line_valid")
        right_valid = state.get(f"vision_{idx}_right_line_valid")
        if _valid_number(left) and _valid_number(right) and (left_valid or right_valid):
            derived = 0.5 * (left - right)
            if abs(derived) <= 5:
                lane_candidates.append(derived)

    prev_lane = autopilot_memory.get("lane_offset_filtered", 0.0)
    if lane_candidates:
        lane_measure = _median(lane_candidates)
        lane_filtered = prev_lane + 0.3 * (lane_measure - prev_lane)
    else:
        lane_filtered = prev_lane
    autopilot_memory["lane_offset_filtered"] = lane_filtered

    # ---- Speed limit ----
    speed_limit_candidates = []
    for idx in [1, 2, 3]:
        if state.get(f"vision_{idx}_speed_limit_valid"):
            val = state.get(f"vision_{idx}_speed_limit_kmh")
            if _valid_number(val):
                speed_limit_candidates.append(val)
    if speed_limit_candidates:
        autopilot_memory["last_speed_limit"] = min(speed_limit_candidates)
    speed_limit = autopilot_memory.get("last_speed_limit", 130.0)

    # ---- Traction ----
    measured_mu = state.get("estimated_mu")
    if not _valid_number(measured_mu):
        measured_mu = autopilot_memory.get("mu_filtered", 0.9)
    mu_filtered = autopilot_memory.get("mu_filtered", 0.9)
    mu_filtered = 0.7 * mu_filtered + 0.3 * measured_mu
    autopilot_memory["mu_filtered"] = mu_filtered

    # ---- Obstacle detection ----
    obstacle_dist = None
    obstacle_rel_speed = None
    for idx in [1, 2, 3]:
        detected = state.get(f"vision_{idx}_obstacle_detected")
        valid = state.get(f"vision_{idx}_obstacle_valid")
        dist = state.get(f"vision_{idx}_obstacle_dist_m")
        rel = state.get(f"vision_{idx}_obstacle_rel_speed_kmh")
        if _valid_number(dist) and dist > 0:
            if detected or valid:
                if obstacle_dist is None or dist < obstacle_dist:
                    obstacle_dist = dist
                    obstacle_rel_speed = rel

    for angle in ["+005", "-005", "+015", "-015", "+030", "-030"]:
        dist = state.get(f"lidar_{angle}_deg_range_m")
        valid = state.get(f"lidar_{angle}_deg_valid")
        if _valid_number(dist) and dist > 0:
            if valid or dist < 60:
                if obstacle_dist is None or dist < obstacle_dist:
                    obstacle_dist = dist
                    obstacle_rel_speed = None

    autopilot_memory["obstacle_dist"] = obstacle_dist

    # ---- Desired lane offset ----
    current_lane = state.get("current_lane", 1)
    target_lane = state.get("target_lane", current_lane)
    desired_lane_offset = _clamp((target_lane - current_lane) * 3.7, -7.4, 7.4)
    autopilot_memory["desired_lane_offset"] = desired_lane_offset
    lane_error = lane_filtered - desired_lane_offset

    # ---- Desired speed ----
    target_speed = state.get("target_speed_kmh")
    if not _valid_number(target_speed):
        target_speed = speed_limit
    desired_speed = min(target_speed, speed_limit + 2.0)
    if not lane_candidates:
        desired_speed = min(desired_speed, 90.0)

    traction_factor = _clamp(0.35 + 0.65 * mu_filtered, 0.35, 1.0)
    desired_speed *= traction_factor

    hard_brake = False
    comfort_decel = 3.0
    if _valid_number(obstacle_dist) and obstacle_dist < 150:
        safe_speed_mps = math.sqrt(max(0.0, 2.0 * comfort_decel * max(obstacle_dist - 5.0, 0.0)))
        safe_speed_kmh = safe_speed_mps * 3.6
        desired_speed = min(desired_speed, safe_speed_kmh)
        if obstacle_dist < 12:
            hard_brake = True
        if _valid_number(obstacle_rel_speed):
            closing_kmh = max(0.0, -obstacle_rel_speed)
            if closing_kmh > 0:
                ttc = obstacle_dist / (closing_kmh / 3.6)
                autopilot_memory["ttc"] = ttc
                if ttc < 3.0:
                    desired_speed = min(desired_speed, current_speed - 20)
                    hard_brake = True
            else:
                autopilot_memory["ttc"] = None
        else:
            autopilot_memory["ttc"] = None
    else:
        autopilot_memory["ttc"] = None

    if obstacle_dist and obstacle_dist < 6:
        desired_speed = 0.0
        hard_brake = True

    # ---- Speed control ----
    speed_error = desired_speed - current_speed
    speed_int = autopilot_memory.get("speed_error_int", 0.0) + speed_error * dt
    speed_int = _clamp(speed_int, -400.0, 400.0)
    autopilot_memory["speed_error_int"] = speed_int

    kp_speed = 0.08
    ki_speed = 0.02
    acc_cmd = kp_speed * speed_error + ki_speed * speed_int

    throttle = _clamp(acc_cmd, 0.0, 1.0)
    brake = _clamp(-acc_cmd * 0.6, 0.0, 1.0)

    if desired_speed < 1.0:
        brake = max(brake, 0.4)
        throttle = 0.0

    if hard_brake:
        brake = max(brake, 0.8)
        throttle = 0.0

    throttle = min(throttle, _clamp(mu_filtered + 0.1, 0.2, 1.0))
    if state.get("rain_sensor"):
        throttle = min(throttle, 0.6)
        brake = min(brake, 0.9)

    if brake > 0.2:
        throttle = min(throttle, 0.2)
    if brake > 0.4:
        throttle = 0.0

    # ---- Steering control ----
    kp_lane = 0.35
    kd_yaw = 0.02
    yaw_rate_rad = math.radians(yaw_rate)

    steering_cmd = -kp_lane * lane_error - kd_yaw * yaw_rate_rad
    steering_cmd = _clamp(steering_cmd, -1.0, 1.0)

    return {
        "steering": steering_cmd,
        "throttle": throttle,
        "brake": brake,
    }
```

This autopilot fuses redundant sensors, stabilizes with filtering/PID, reacts to obstacles, adapts to lane-change commands, enforces speed limits, and adjusts behavior under low traction—all while handling sensor dropouts and failures gracefully.
