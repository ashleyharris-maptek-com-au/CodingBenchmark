OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1
%glsl = OpExtInstImport "GLSL.std.450"
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%uint_6 = OpConstant %uint 6
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_Input_arr_pv Input
%ptr_Input_v4 = OpTypePointer Input %v4float
%arr_v3 = OpTypeArray %v3float %uint_6
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3
%worldPos = OpVariable %ptr_Input_arr_v3 Input
%normal = OpVariable %ptr_Input_arr_v3 Input
%color = OpVariable %ptr_Input_arr_v3 Input
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Output_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_pv Output
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3 Output
%void_fn = OpTypeFunction %void
%emit_fn = OpTypeFunction %void %int %int
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_5 = OpConstant %float 0.5
%float_0_004 = OpConstant %float 0.004
%float_eps = OpConstant %float 0.000001

%EmitEdge = OpFunction %void None %emit_fn
%param_i0 = OpFunctionParameter %int
%param_i1 = OpFunctionParameter %int
%edge_entry = OpLabel
%p0_ptr = OpAccessChain %ptr_Input_v4 %gl_in %param_i0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_Input_v4 %gl_in %param_i1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%col0_ptr = OpAccessChain %ptr_Input_v3 %color %param_i0
%col0 = OpLoad %v3float %col0_ptr
%col1_ptr = OpAccessChain %ptr_Input_v3 %color %param_i1
%col1 = OpLoad %v3float %col1_ptr
%sumColor = OpFAdd %v3float %col0 %col1
%avgColor = OpVectorTimesScalar %v3float %sumColor %float_0_5
%xy0 = OpVectorShuffle %v2float %p0 %p0 0 1
%xy1 = OpVectorShuffle %v2float %p1 %p1 0 1
%w0 = OpCompositeExtract %float %p0 3
%w1 = OpCompositeExtract %float %p1 3
%z0 = OpCompositeExtract %float %p0 2
%z1 = OpCompositeExtract %float %p1 2
%invW0 = OpFDiv %float %float_1 %w0
%invW1 = OpFDiv %float %float_1 %w1
%xy0_ndc = OpVectorTimesScalar %v2float %xy0 %invW0
%xy1_ndc = OpVectorTimesScalar %v2float %xy1 %invW1
%dir = OpFSub %v2float %xy1_ndc %xy0_ndc
%len2 = OpDot %float %dir %dir
%len = OpExtInst %float %glsl Sqrt %len2
%len_clamped = OpExtInst %float %glsl FMax %len %float_eps
%invLen = OpFDiv %float %float_1 %len_clamped
%dirNorm = OpVectorTimesScalar %v2float %dir %invLen
%dirX = OpCompositeExtract %float %dirNorm 0
%dirY = OpCompositeExtract %float %dirNorm 1
%negDirY = OpFNegate %float %dirY
%perp = OpCompositeConstruct %v2float %negDirY %dirX
%offset = OpVectorTimesScalar %v2float %perp %float_0_004
%offset0_clip = OpVectorTimesScalar %v2float %offset %w0
%offset1_clip = OpVectorTimesScalar %v2float %offset %w1
%xy0_plus = OpFAdd %v2float %xy0 %offset0_clip
%xy0_minus = OpFSub %v2float %xy0 %offset0_clip
%xy1_plus = OpFAdd %v2float %xy1 %offset1_clip
%xy1_minus = OpFSub %v2float %xy1 %offset1_clip
%ptr_gl_pos = OpAccessChain %ptr_Output_v4 %gl_out %int_0
%xy0_plus_x = OpCompositeExtract %float %xy0_plus 0
%xy0_plus_y = OpCompositeExtract %float %xy0_plus 1
%vtx0 = OpCompositeConstruct %v4float %xy0_plus_x %xy0_plus_y %z0 %w0
OpStore %ptr_gl_pos %vtx0
OpStore %outColor %avgColor
OpEmitVertex
%xy1_plus_x = OpCompositeExtract %float %xy1_plus 0
%xy1_plus_y = OpCompositeExtract %float %xy1_plus 1
%vtx1 = OpCompositeConstruct %v4float %xy1_plus_x %xy1_plus_y %z1 %w1
OpStore %ptr_gl_pos %vtx1
OpStore %outColor %avgColor
OpEmitVertex
%xy0_minus_x = OpCompositeExtract %float %xy0_minus 0
%xy0_minus_y = OpCompositeExtract %float %xy0_minus 1
%vtx2 = OpCompositeConstruct %v4float %xy0_minus_x %xy0_minus_y %z0 %w0
OpStore %ptr_gl_pos %vtx2
OpStore %outColor %avgColor
OpEmitVertex
%xy1_minus_x = OpCompositeExtract %float %xy1_minus 0
%xy1_minus_y = OpCompositeExtract %float %xy1_minus 1
%vtx3 = OpCompositeConstruct %v4float %xy1_minus_x %xy1_minus_y %z1 %w1
OpStore %ptr_gl_pos %vtx3
OpStore %outColor %avgColor
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd

%main = OpFunction %void None %void_fn
%main_entry = OpLabel
OpFunctionCall %void %EmitEdge %int_0 %int_2
OpFunctionCall %void %EmitEdge %int_2 %int_4
