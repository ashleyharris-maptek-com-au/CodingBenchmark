```rust
use std::io::{self, Read, Write};

struct FastInput {
    data: Vec<u8>,
    pos: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data).unwrap();
        Self { data, pos: 0 }
    }

    fn next_usize(&mut self) -> usize {
        while self.pos < self.data.len() && self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        let start = self.pos;
        while self.pos < self.data.len() && !self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        let slice = &self.data[start..self.pos];
        std::str::from_utf8(slice).unwrap().parse::<usize>().unwrap()
    }

    fn next_u32(&mut self) -> u32 {
        self.next_usize() as u32
    }
}

fn main() {
    let mut input = FastInput::new();
    let n = input.next_usize();
    let m = input.next_usize();

    let mut edges: Vec<(u32, u32)> = Vec::with_capacity(m);
    for _ in 0..m {
        let u = input.next_u32();
        let v = input.next_u32();
        edges.push((u, v));
    }

    let mut in_cover = vec![false; n];
    for &(u, v) in &edges {
        let ui = u as usize;
        let vi = v as usize;
        if in_cover[ui] || in_cover[vi] {
            continue;
        }
        in_cover[ui] = true;
        in_cover[vi] = true;
    }

    let mut cover = Vec::new();
    for (idx, &flag) in in_cover.iter().enumerate() {
        if flag {
            cover.push(idx);
        }
    }

    let mut out = io::BufWriter::new(io::stdout());
    writeln!(out, "{}", cover.len()).unwrap();
    if !cover.is_empty() {
        for (i, v) in cover.iter().enumerate() {
            if i > 0 {
                write!(out, " ").unwrap();
            }
            write!(out, "{}", v).unwrap();
        }
        writeln!(out).unwrap();
    } else {
        writeln!(out).unwrap();
    }
}
```
