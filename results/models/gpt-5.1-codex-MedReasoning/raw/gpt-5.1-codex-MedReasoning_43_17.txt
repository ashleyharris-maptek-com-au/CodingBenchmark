struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diff = saturate(dot(N, L));
    float shade = 0.4 + 0.6 * diff;

    float2 scaledUV = input.uv * float2(8.0, 16.0);
    float row = floor(scaledUV.y);
    float rowParity = row - 2.0 * floor(row * 0.5);
    scaledUV.x += rowParity * 0.5;
    float2 cell = frac(scaledUV);

    float mortarX = 0.05;
    float mortarY = 0.1;
    float halfX = mortarX * 0.5;
    float halfY = mortarY * 0.5;
    float isMortar = (cell.x < halfX || cell.x > 1.0 - halfX || cell.y < halfY || cell.y > 1.0 - halfY) ? 1.0 : 0.0;

    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 baseColor = lerp(brickColor, mortarColor, isMortar);
    float3 finalColor = shade * baseColor;

    return float4(finalColor, 1.0);
}
