```python
import sys
import math
import random
from functools import lru_cache


def main():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)

    try:
        n = int(next(it))
        k = int(next(it))
        capacity = float(next(it))
        depot_x = float(next(it))
        depot_y = float(next(it))
    except StopIteration:
        return

    coords = []
    demands = []
    for _ in range(n):
        try:
            x = float(next(it))
            y = float(next(it))
            d = float(next(it))
        except StopIteration:
            break
        coords.append((x, y))
        demands.append(d)

    if n == 0:
        print("0.0")
        print("0")
        return

    random.seed(0)
    cap_eps = 1e-9
    target_routes = k if k > 0 else (0 if n == 0 else 1)

    dist0 = [math.hypot(x - depot_x, y - depot_y) for (x, y) in coords]

    @lru_cache(maxsize=None)
    def dist(i, j):
        xi, yi = coords[i]
        xj, yj = coords[j]
        return math.hypot(xi - xj, yi - yj)

    def dist_point(a, b):
        if a == -1 and b == -1:
            return 0.0
        if a == -1:
            return dist0[b]
        if b == -1:
            return dist0[a]
        return dist(a, b)

    def route_cost(seq):
        if not seq:
            return 0.0
        cost = dist0[seq[0]] + dist0[seq[-1]]
        for u, v in zip(seq, seq[1:]):
            cost += dist(u, v)
        return cost

    routes = [{'seq': [i], 'load': demands[i], 'active': True} for i in range(n)]
    route_of = list(range(n))

    def active_route_ids():
        return [idx for idx, r in enumerate(routes) if r['active'] and r['seq']]

    def count_active():
        return len(active_route_ids())

    def attempt_merge_nodes(a, b):
        ri = route_of[a]
        rj = route_of[b]
        if ri == rj:
            return False
        route_i = routes[ri]
        route_j = routes[rj]
        if not route_i['active'] or not route_j['active']:
            return False
        seq_i = route_i['seq']
        seq_j = route_j['seq']
        if not seq_i or not seq_j:
            return False
        if seq_i[0] != a and seq_i[-1] != a:
            return False
        if seq_j[0] != b and seq_j[-1] != b:
            return False
        if route_i['load'] + route_j['load'] > capacity + cap_eps:
            return False

        merged_seq = None
        if seq_i[-1] == a and seq_j[0] == b:
            merged_seq = seq_i + seq_j
        elif seq_i[0] == a and seq_j[-1] == b:
            merged_seq = seq_j + seq_i
        elif seq_i[0] == a and seq_j[0] == b:
            merged_seq = seq_i[::-1] + seq_j
        elif seq_i[-1] == a and seq_j[-1] == b:
            merged_seq = seq_i + seq_j[::-1]
        else:
            return False

        route_i['seq'] = merged_seq
        route_i['load'] = route_i['load'] + route_j['load']
        for node in merged_seq:
            route_of[node] = ri
        route_j['seq'] = []
        route_j['load'] = 0.0
        route_j['active'] = False
        return True

    def best_merge_pair(ri, rj, cost_i, cost_j):
        load_sum = routes[ri]['load'] + routes[rj]['load']
        if load_sum > capacity + cap_eps:
            return None
        base_cost = cost_i + cost_j
        best = None
        for first_id, second_id in ((ri, rj), (rj, ri)):
            seq_first = routes[first_id]['seq']
            seq_second = routes[second_id]['seq']
            for rev_first in (False, True):
                first = seq_first[::-1] if rev_first else seq_first
                for rev_second in (False, True):
                    second = seq_second[::-1] if rev_second else seq_second
                    new_seq = first + second
                    new_cost = route_cost(new_seq)
                    delta = new_cost - base_cost
                    if best is None or delta < best[0]:
                        best = (delta, first_id, second_id, new_seq)
        return best

    def attempt_best_merge():
        ids = active_route_ids()
        if len(ids) < 2:
            return False
        cost_cache = {idx: route_cost(routes[idx]['seq']) for idx in ids}
        best = None
        for i in range(len(ids)):
            ri = ids[i]
            for j in range(i + 1, len(ids)):
                rj = ids[j]
                candidate = best_merge_pair(ri, rj, cost_cache[ri], cost_cache[rj])
                if candidate is None:
                    continue
                if best is None or candidate[0] < best[0]:
                    best = candidate
        if best is None:
            return False
        _, base_id, remove_id, new_seq = best
        new_load = routes[base_id]['load'] + routes[remove_id]['load']
        routes[base_id]['seq'] = new_seq
        routes[base_id]['load'] = new_load
        for node in new_seq:
            route_of[node] = base_id
        routes[remove_id]['seq'] = []
        routes[remove_id]['load'] = 0.0
        routes[remove_id]['active'] = False
        return True

    def insertion_delta(seq, cust, pos):
        if not seq:
            return dist0[cust] * 2.0
        if pos == 0:
            return dist_point(-1, cust) + dist_point(cust, seq[0]) - dist_point(-1, seq[0])
        if pos == len(seq):
            return dist_point(seq[-1], cust) + dist_point(cust, -1) - dist_point(seq[-1], -1)
        prev_node = seq[pos - 1]
        next_node = seq[pos]
        return dist_point(prev_node, cust) + dist_point(cust, next_node) - dist_point(prev_node, next_node)

    def best_insertion(seq, cust):
        if not seq:
            return dist0[cust] * 2.0, 0
        best_delta = None
        best_pos = 0
        for pos in range(len(seq) + 1):
            delta = insertion_delta(seq, cust, pos)
            if best_delta is None or delta < best_delta:
                best_delta = delta
                best_pos = pos
        return best_delta, best_pos

    def clone_routes():
        cloned = []
        for r in routes:
            cloned.append({'seq': r['seq'][:], 'load': r['load'], 'active': r['active']})
        return cloned

    def relocate_with_order(rid, order):
        new_routes = clone_routes()
        new_route_of = route_of[:]
        route = new_routes[rid]
        if not route['active']:
            return False, None, None
        route['active'] = False
        route['seq'] = []
        route['load'] = 0.0
        for cust in order:
            best_move = None
            for idx, target in enumerate(new_routes):
                if idx == rid or not target['active']:
                    continue
                if target['load'] + demands[cust] > capacity + cap_eps:
                    continue
                delta, pos = best_insertion(target['seq'], cust)
                if best_move is None or delta < best_move[0]:
                    best_move = (delta, idx, pos)
            if best_move is None:
                return False, None, None
            _, dest_idx, pos = best_move
            target_route = new_routes[dest_idx]
            target_route['seq'].insert(pos, cust)
            target_route['load'] += demands[cust]
            new_route_of[cust] = dest_idx
        return True, new_routes, new_route_of

    def attempt_route_relocation():
        ids = active_route_ids()
        if len(ids) <= target_routes:
            return False
        ids.sort(key=lambda idx: routes[idx]['load'])
        for rid in ids:
            seq = routes[rid]['seq']
            if not seq:
                continue
            orders = [
                seq[:],
                sorted(seq, key=lambda c: -demands[c]),
                sorted(seq, key=lambda c: demands[c]),
            ]
            if len(seq) > 2:
                rand_order = seq[:]
                random.shuffle(rand_order)
                orders.append(rand_order)
                rand_order2 = seq[:]
                random.shuffle(rand_order2)
                orders.append(rand_order2)
            seen = set()
            for order in orders:
                key = tuple(order)
                if key in seen:
                    continue
                seen.add(key)
                success, new_routes, new_route_of = relocate_with_order(rid, order)
                if success:
                    routes[:] = new_routes
                    route_of[:] = new_route_of
                    return True
        return False

    def two_opt_route(seq):
        seq = seq[:]
        n = len(seq)
        if n < 3:
            return seq
        improved = True
        while improved:
            improved = False
            for i in range(n - 1):
                for j in range(i + 1, n):
                    if j - i <= 1:
                        continue
                    prev_i = -1 if i == 0 else seq[i - 1]
                    next_j = -1 if j == n - 1 else seq[j + 1]
                    a = prev_i
                    b = seq[i]
                    c = seq[j]
                    d = next_j
                    delta = dist_point(a, c) + dist_point(b, d) - dist_point(a, b) - dist_point(c, d)
                    if delta < -1e-9:
                        seq[i:j + 1] = reversed(seq[i:j + 1])
                        improved = True
                        break
                if improved:
                    break
        return seq

    def optimize_routes():
        for idx, r in enumerate(routes):
            if r['active'] and len(r['seq']) >= 3:
                new_seq = two_opt_route(r['seq'])
                routes[idx]['seq'] = new_seq
                for cust in new_seq:
                    route_of[cust] = idx

    def compute_active_sequences():
        return [r['seq'][:] for r in routes if r['active'] and r['seq']]

    def solution_is_valid():
        seqs = compute_active_sequences()
        if len(seqs) > k:
            return False
        seen = []
        for seq in seqs:
            seen.extend(seq)
        if len(seen) != n:
            return False
        if len(set(seen)) != n:
            return False
        return True

    def build_route_for_customers(customers):
        if not customers:
            return []
        remaining = set(customers)
        current = max(remaining, key=lambda c: dist0[c])
        route = [current]
        remaining.remove(current)
        while remaining:
            nxt = min(remaining, key=lambda c: dist(current, c))
            route.append(nxt)
            remaining.remove(nxt)
        return route

    def pack_order(order):
        bins = []
        for cust in order:
            best_idx = -1
            best_remaining = None
            for idx, b in enumerate(bins):
                if b['load'] + demands[cust] <= capacity + cap_eps:
                    remaining = capacity - (b['load'] + demands[cust])
                    if best_remaining is None or remaining < best_remaining:
                        best_remaining = remaining
                        best_idx = idx
            if best_idx == -1:
                if len(bins) < k:
                    bins.append({'customers': [cust], 'load': demands[cust]})
                else:
                    return None
            else:
                bins[best_idx]['customers'].append(cust)
                bins[best_idx]['load'] += demands[cust]
        return bins

    def build_partition_routes():
        if k <= 0:
            return None
        orders = []
        indices = list(range(n))
        angle_order = sorted(indices, key=lambda i: math.atan2(coords[i][1] - depot_y, coords[i][0] - depot_x))
        orders.append(angle_order)
        orders.append(sorted(indices, key=lambda i: -demands[i]))
        orders.append(sorted(indices, key=lambda i: demands[i]))
        orders.append(sorted(indices, key=lambda i: coords[i][0]))
        orders.append(sorted(indices, key=lambda i: coords[i][1]))
        random_trials = max(200, min(800, n * 4))
        for _ in range(random_trials):
            perm = indices[:]
            random.shuffle(perm)
            orders.append(perm)
        tried = set()
        for order in orders:
            key = tuple(order)
            if key in tried:
                continue
            tried.add(key)
            bins = pack_order(order)
            if bins is None:
                continue
            sequences = []
            for b in bins:
                if not b['customers']:
                    continue
                seq = build_route_for_customers(b['customers'])
                sequences.append(seq)
            if len(sequences) <= k:
                return sequences
        return None

    def set_routes_from_sequences(sequences):
        new_routes = []
        new_route_of = [-1] * n
        for seq in sequences:
            load = sum(demands[c] for c in seq)
            new_routes.append({'seq': seq[:], 'load': load, 'active': True})
            rid = len(new_routes) - 1
            for cust in seq:
                new_route_of[cust] = rid
        routes[:] = new_routes
        route_of[:] = new_route_of

    # Build initial solution with Clarke-Wright savings
    savings = []
    for i in range(n):
        for j in range(i + 1, n):
            s = dist0[i] + dist0[j] - dist(i, j)
            savings.append((s, i, j))
    savings.sort(reverse=True, key=lambda x: x[0])
    for _, i, j in savings:
        attempt_merge_nodes(i, j)

    # Reduce routes via merges and relocations
    iteration_limit = max(10, len(routes) * 10)
    iteration = 0
    while count_active() > target_routes and iteration < iteration_limit:
        iteration += 1
        if attempt_best_merge():
            continue
        if attempt_route_relocation():
            continue
        break

    if count_active() > k and k > 0:
        partition = build_partition_routes()
        if partition:
            set_routes_from_sequences(partition)

    optimize_routes()

    if not solution_is_valid():
        partition = build_partition_routes()
        if partition:
            set_routes_from_sequences(partition)
            optimize_routes()

    sequences = compute_active_sequences()
    if (len(sequences) > k or len(sequences) == 0 or
            sorted([cust for seq in sequences for cust in seq]) != list(range(n))):
        partition = build_partition_routes()
        if partition:
            set_routes_from_sequences(partition)
            optimize_routes()
            sequences = compute_active_sequences()

    sequences = compute_active_sequences()
    if len(sequences) == 0:
        # Fallback: each customer alone (only safe if k >= n)
        sequences = [[i] for i in range(n)]
        set_routes_from_sequences(sequences)

    sequences = compute_active_sequences()
    total_distance = sum(route_cost(seq) for seq in sequences)

    print(f"{total_distance:.6f}")
    print(len(sequences))
    for seq in sequences:
        print(" ".join(str(idx) for idx in seq))


if __name__ == "__main__":
    main()
```
