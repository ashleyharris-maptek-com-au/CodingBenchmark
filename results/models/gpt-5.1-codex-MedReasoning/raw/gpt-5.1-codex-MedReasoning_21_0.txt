#include <algorithm>
#include <cstddef>
#include <deque>
#include <functional>
#include <iostream>
#include <limits>
#include <optional>
#include <queue>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using Position = std::vector<int>;

struct PositionHash {
    std::size_t operator()(const Position& pos) const noexcept {
        std::size_t h = 0xcbf29ce484222325ULL;
        for (int value : pos) {
            std::size_t component = std::hash<long long>{}(static_cast<long long>(value));
            h ^= component + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        }
        return h;
    }
};

bool withinBounds(const Position& pos, const std::vector<int>& bounds) {
    for (std::size_t i = 0; i < bounds.size(); ++i) {
        if (pos[i] < 0 || pos[i] >= bounds[i]) {
            return false;
        }
    }
    return true;
}

long long manhattan(const Position& a, const Position& b) {
    long long sum = 0;
    for (std::size_t i = 0; i < a.size(); ++i) {
        long long diff = static_cast<long long>(a[i]) - static_cast<long long>(b[i]);
        sum += diff >= 0 ? diff : -diff;
    }
    return sum;
}

bool lexicographicalLess(const Position& lhs, const Position& rhs) {
    for (std::size_t i = 0; i < lhs.size(); ++i) {
        if (lhs[i] != rhs[i]) {
            return lhs[i] < rhs[i];
        }
    }
    return false;
}

int boundaryPenalty(const Position& pos, const std::vector<int>& bounds) {
    int penalty = 0;
    for (std::size_t i = 0; i < bounds.size(); ++i) {
        if (bounds[i] <= 0) {
            continue;
        }
        if (pos[i] == 0 || pos[i] == bounds[i] - 1) {
            ++penalty;
        }
    }
    return penalty;
}

struct FoodStore {
    std::vector<Position> items;
    std::vector<char> alive;
    std::unordered_map<Position, int, PositionHash> index;
    int aliveCount = 0;

    void reserve(std::size_t count) {
        items.reserve(count);
        alive.reserve(count);
        index.reserve(count * 2 + 1);
    }

    void add(const Position& pos) {
        items.push_back(pos);
        alive.push_back(1);
        index[items.back()] = static_cast<int>(items.size()) - 1;
        ++aliveCount;
    }

    bool contains(const Position& pos) const {
        auto it = index.find(pos);
        if (it == index.end()) {
            return false;
        }
        int idx = it->second;
        if (idx < 0 || idx >= static_cast<int>(alive.size())) {
            return false;
        }
        return alive[idx] != 0;
    }

    bool consume(const Position& pos) {
        auto it = index.find(pos);
        if (it == index.end()) {
            return false;
        }
        int idx = it->second;
        if (!alive[idx]) {
            return false;
        }
        alive[idx] = 0;
        index.erase(it);
        --aliveCount;
        return true;
    }

    bool empty() const {
        return aliveCount == 0;
    }

    const Position* closestTarget(const Position& from) const {
        if (aliveCount == 0) {
            return nullptr;
        }
        long long bestDist = std::numeric_limits<long long>::max();
        const Position* best = nullptr;
        for (std::size_t i = 0; i < items.size(); ++i) {
            if (!alive[i]) {
                continue;
            }
            long long dist = manhattan(from, items[i]);
            if (dist < bestDist) {
                bestDist = dist;
                best = &items[i];
            } else if (dist == bestDist && best != nullptr && lexicographicalLess(items[i], *best)) {
                best = &items[i];
            }
        }
        return best;
    }
};

void seedSnakeBody(int desiredLength,
                   const Position& head,
                   const std::vector<int>& bounds,
                   const std::unordered_set<Position, PositionHash>& obstacles,
                   const FoodStore& foods,
                   std::deque<Position>& snake,
                   std::unordered_set<Position, PositionHash>& snakeCells) {
    snake.clear();
    snakeCells.clear();
    snake.push_back(head);
    snakeCells.insert(head);

    Position tailBuilder = head;
    const int dirOptions[2] = {-1, 1};

    for (int segment = 1; segment < desiredLength; ++segment) {
        bool placed = false;
        for (int axis = 0; axis < static_cast<int>(bounds.size()) && !placed; ++axis) {
            for (int dir : dirOptions) {
                Position candidate = tailBuilder;
                candidate[axis] += dir;
                if (!withinBounds(candidate, bounds)) {
                    continue;
                }
                if (obstacles.find(candidate) != obstacles.end()) {
                    continue;
                }
                if (snakeCells.find(candidate) != snakeCells.end()) {
                    continue;
                }
                if (foods.contains(candidate)) {
                    continue;
                }
                snake.push_back(candidate);
                snakeCells.insert(candidate);
                tailBuilder = candidate;
                placed = true;
                break;
            }
        }
        if (!placed) {
            break;
        }
    }
}

std::optional<std::pair<int, int>> findFirstStepBfs(
    const Position& head,
    const Position& target,
    const std::vector<int>& bounds,
    const std::unordered_set<Position, PositionHash>& obstacles,
    const std::unordered_set<Position, PositionHash>& snakeCells,
    const Position& tail,
    std::size_t reserveHint) {

    const int dirOptions[2] = {-1, 1};
    std::queue<Position> q;
    q.push(head);

    std::size_t reserveSize = std::max<std::size_t>(reserveHint, 16);

    std::unordered_set<Position, PositionHash> visited;
    visited.reserve(reserveSize);
    visited.insert(head);

    std::unordered_map<Position, std::pair<int, int>, PositionHash> firstMove;
    firstMove.reserve(reserveSize);
    firstMove.emplace(head, std::make_pair(-1, 0));

    while (!q.empty()) {
        Position current = q.front();
        q.pop();

        if (current == target) {
            auto it = firstMove.find(current);
            if (it != firstMove.end() && it->second.first != -1) {
                return it->second;
            }
            return std::nullopt;
        }

        auto parentMoveIt = firstMove.find(current);
        std::pair<int, int> inherited = {-1, 0};
        if (parentMoveIt != firstMove.end()) {
            inherited = parentMoveIt->second;
        }

        for (int axis = 0; axis < static_cast<int>(bounds.size()); ++axis) {
            for (int dir : dirOptions) {
                Position next = current;
                next[axis] += dir;

                if (!withinBounds(next, bounds)) {
                    continue;
                }
                if (obstacles.find(next) != obstacles.end()) {
                    continue;
                }

                bool occupied = snakeCells.find(next) != snakeCells.end();
                bool isTailCell = (next == tail);

                if (occupied && !isTailCell && !(next == target)) {
                    continue;
                }

                if (visited.insert(next).second) {
                    if (current == head) {
                        firstMove[next] = {axis, dir};
                    } else if (inherited.first != -1) {
                        firstMove[next] = inherited;
                    } else {
                        firstMove[next] = {axis, dir};
                    }
                    q.push(next);
                }
            }
        }
    }

    return std::nullopt;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int dimension = 0;
    long long maxTurns = 0;
    if (!(std::cin >> dimension >> maxTurns)) {
        return 0;
    }
    if (dimension <= 0) {
        return 0;
    }

    std::vector<int> bounds(dimension);
    for (int i = 0; i < dimension; ++i) {
        std::cin >> bounds[i];
    }

    int declaredLength = 1;
    std::cin >> declaredLength;
    if (declaredLength <= 0) {
        declaredLength = 1;
    }

    Position head(dimension);
    for (int i = 0; i < dimension; ++i) {
        std::cin >> head[i];
    }

    int foodCount = 0;
    std::cin >> foodCount;
    FoodStore foods;
    foods.reserve(foodCount);
    for (int i = 0; i < foodCount; ++i) {
        Position foodPos(dimension);
        for (int d = 0; d < dimension; ++d) {
            std::cin >> foodPos[d];
        }
        foods.add(foodPos);
    }

    int obstacleCount = 0;
    std::cin >> obstacleCount;
    std::unordered_set<Position, PositionHash> obstacles;
    obstacles.reserve(static_cast<std::size_t>(obstacleCount) * 2 + 1);
    for (int i = 0; i < obstacleCount; ++i) {
        Position obs(dimension);
        for (int d = 0; d < dimension; ++d) {
            std::cin >> obs[d];
        }
        obstacles.insert(obs);
    }

    std::deque<Position> snake;
    std::unordered_set<Position, PositionHash> snakeCells;
    snakeCells.reserve(static_cast<std::size_t>(declaredLength) + static_cast<std::size_t>(foodCount) +
                       static_cast<std::size_t>(obstacleCount) + 8);
    seedSnakeBody(declaredLength, head, bounds, obstacles, foods, snake, snakeCells);
    if (snake.empty()) {
        snake.push_back(head);
        snakeCells.insert(head);
    }

    constexpr std::size_t BFS_LIMIT = 120000;
    std::size_t approxVolume = 1;
    bool boardSmallEnough = true;
    for (int size : bounds) {
        int clampSize = std::max(size, 1);
        if (approxVolume > BFS_LIMIT / static_cast<std::size_t>(clampSize)) {
            boardSmallEnough = false;
            break;
        }
        approxVolume *= static_cast<std::size_t>(clampSize);
    }
    std::size_t bfsReserve = boardSmallEnough ? std::max<std::size_t>(approxVolume, 16) : 16;

    long long turns = 0;
    while (turns < maxTurns && !foods.empty()) {
        Position currentHead = snake.front();
        const Position* target = foods.closestTarget(currentHead);

        int slotCount = std::max(1, dimension) * 2;
        std::vector<std::pair<int, int>> candidates;
        candidates.reserve(slotCount);
        std::vector<char> used(slotCount, 0);

        auto pushCandidate = [&](int axis, int dir) {
            if (axis < 0 || axis >= dimension) {
                return;
            }
            int idx = axis * 2 + (dir > 0 ? 1 : 0);
            if (idx < 0 || idx >= slotCount) {
                return;
            }
            if (!used[idx]) {
                used[idx] = 1;
                candidates.emplace_back(axis, dir);
            }
        };

        if (target != nullptr) {
            if (boardSmallEnough) {
                auto bfsMove = findFirstStepBfs(currentHead, *target, bounds, obstacles, snakeCells, snake.back(), bfsReserve);
                if (bfsMove) {
                    pushCandidate(bfsMove->first, bfsMove->second);
                }
            }

            struct AxisPreference {
                int axis;
                int dir;
                int weight;
            };
            std::vector<AxisPreference> prefs;
            prefs.reserve(dimension);
            for (int axis = 0; axis < dimension; ++axis) {
                int diff = (*target)[axis] - currentHead[axis];
                if (diff != 0) {
                    prefs.push_back({axis, diff > 0 ? 1 : -1, std::abs(diff)});
                }
            }
            std::sort(prefs.begin(), prefs.end(), [](const AxisPreference& a, const AxisPreference& b) {
                if (a.weight != b.weight) {
                    return a.weight > b.weight;
                }
                return a.axis < b.axis;
            });
            for (const auto& pref : prefs) {
                pushCandidate(pref.axis, pref.dir);
            }
        }

        for (int axis = 0; axis < dimension; ++axis) {
            pushCandidate(axis, 1);
            pushCandidate(axis, -1);
        }

        bool foundSafeMove = false;
        int selectedAxis = 0;
        int selectedDir = 1;
        long long bestScore = std::numeric_limits<long long>::max();
        const Position& tail = snake.back();

        for (const auto& cand : candidates) {
            Position next = currentHead;
            next[cand.first] += cand.second;

            if (!withinBounds(next, bounds)) {
                continue;
            }
            if (obstacles.find(next) != obstacles.end()) {
                continue;
            }

            bool hasFood = foods.contains(next);
            auto occIt = snakeCells.find(next);
            if (occIt != snakeCells.end()) {
                if (!( !hasFood && next == tail )) {
                    continue;
                }
            }

            long long cost = target ? manhattan(next, *target) : 0;
            int border = boundaryPenalty(next, bounds);
            cost = cost * 1024 + static_cast<long long>(border) * 16 +
                   static_cast<long long>(cand.first) * 2 + (cand.second > 0 ? 0 : 1);

            if (cost < bestScore) {
                bestScore = cost;
                selectedAxis = cand.first;
                selectedDir = cand.second;
                foundSafeMove = true;
            }
        }

        if (foundSafeMove) {
            std::cout << selectedAxis << ' ' << selectedDir << std::endl;

            Position newHead = currentHead;
            newHead[selectedAxis] += selectedDir;
            bool ateFood = foods.contains(newHead);

            snake.push_front(newHead);
            snakeCells.insert(newHead);

            if (ateFood) {
                foods.consume(newHead);
            } else {
                Position tailPos = snake.back();
                snake.pop_back();
                snakeCells.erase(tailPos);
            }

            ++turns;
            continue;
        }

        bool fallbackFound = false;
        int fallbackAxis = 0;
        int fallbackDir = 1;

        for (const auto& cand : candidates) {
            Position next = currentHead;
            next[cand.first] += cand.second;
            if (!withinBounds(next, bounds)) {
                continue;
            }
            if (obstacles.find(next) != obstacles.end()) {
                continue;
            }
            fallbackAxis = cand.first;
            fallbackDir = cand.second;
            fallbackFound = true;
            break;
        }

        if (!fallbackFound) {
            for (int axis = 0; axis < dimension && !fallbackFound; ++axis) {
                for (int dir : {1, -1}) {
                    Position next = currentHead;
                    next[axis] += dir;
                    if (!withinBounds(next, bounds)) {
                        continue;
                    }
                    fallbackAxis = axis;
                    fallbackDir = dir;
                    fallbackFound = true;
                    break;
                }
            }
        }

        if (!fallbackFound) {
            break;
        }

        std::cout << fallbackAxis << ' ' << fallbackDir << std::endl;
        break;
    }

    return 0;
}
