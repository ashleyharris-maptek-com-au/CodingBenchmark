use std::io::{self, Read};

fn dfs(
    idx: usize,
    n: usize,
    target: usize,
    subset: &mut Vec<usize>,
    in_subset: &mut [bool],
    edges: &[(usize, usize)],
    best_cut: &mut usize,
    best_subset: &mut Vec<usize>,
) {
    if *best_cut == 0 {
        return;
    }
    if subset.len() == target {
        let mut cut = 0usize;
        for &(u, v) in edges {
            if in_subset[u] ^ in_subset[v] {
                cut += 1;
            }
        }
        if cut < *best_cut {
            *best_cut = cut;
            best_subset.clear();
            best_subset.extend(subset.iter().copied());
        }
        return;
    }
    if idx >= n {
        return;
    }

    let remaining_vertices = n - idx;
    let remaining_needed = target - subset.len();
    if remaining_vertices < remaining_needed {
        return;
    }

    if remaining_vertices == remaining_needed {
        in_subset[idx] = true;
        subset.push(idx);
        dfs(
            idx + 1,
            n,
            target,
            subset,
            in_subset,
            edges,
            best_cut,
            best_subset,
        );
        subset.pop();
        in_subset[idx] = false;
        return;
    }

    in_subset[idx] = true;
    subset.push(idx);
    dfs(
        idx + 1,
        n,
        target,
        subset,
        in_subset,
        edges,
        best_cut,
        best_subset,
    );
    subset.pop();
    in_subset[idx] = false;

    dfs(
        idx + 1,
        n,
        target,
        subset,
        in_subset,
        edges,
        best_cut,
        best_subset,
    );
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }
    let mut it = input.split_whitespace();
    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();
    assert!(n % 2 == 0, "number of vertices must be even");

    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        edges.push((u, v));
    }

    if n == 0 {
        println!("0");
        println!();
        return;
    }

    let target = n / 2;
    let mut subset = Vec::with_capacity(target);
    let mut in_subset = vec![false; n];
    let mut best_subset = Vec::with_capacity(target);
    let mut best_cut = usize::MAX;

    subset.push(0);
    in_subset[0] = true;

    dfs(
        1,
        n,
        target,
        &mut subset,
        &mut in_subset,
        &edges,
        &mut best_cut,
        &mut best_subset,
    );

    best_subset.sort_unstable();

    println!("{}", best_cut);
    if !best_subset.is_empty() {
        for (i, v) in best_subset.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", v);
        }
    }
    println!();
}
