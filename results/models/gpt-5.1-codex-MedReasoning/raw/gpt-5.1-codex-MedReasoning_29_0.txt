```csharp
using System;
using System.IO;
using System.Collections.Generic;
using System.Diagnostics;
using System.Buffers;
using System.Numerics;

class Program
{
    const int EXACT_VERTEX_THRESHOLD = 600;
    const long EXACT_EDGE_THRESHOLD = 300_000;

    static void Main()
    {
        var reader = new FastReader();
        int n = reader.NextInt();
        int m = reader.NextInt();

        if (n <= 0)
        {
            for (int i = 0; i < m; i++)
            {
                reader.NextInt();
                reader.NextInt();
            }
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }

        var temp = new List<int>[n];
        long actualEdges = 0;
        for (int i = 0; i < m; i++)
        {
            int u = reader.NextInt();
            int v = reader.NextInt();
            if (u == v || u < 0 || v < 0 || u >= n || v >= n) continue;

            (temp[u] ??= new List<int>()).Add(v);
            (temp[v] ??= new List<int>()).Add(u);
            actualEdges++;
        }

        var adj = new int[n][];
        var deg = new int[n];
        for (int i = 0; i < n; i++)
        {
            var list = temp[i];
            if (list == null || list.Count == 0)
            {
                adj[i] = Array.Empty<int>();
                deg[i] = 0;
                continue;
            }

            int[] arr = list.ToArray();
            Array.Sort(arr);
            int unique = 1;
            int last = arr[0];
            for (int j = 1; j < arr.Length; j++)
            {
                int val = arr[j];
                if (val == last) continue;
                arr[unique++] = val;
                last = val;
            }
            if (unique != arr.Length)
            {
                int[] trimmed = new int[unique];
                Array.Copy(arr, trimmed, unique);
                arr = trimmed;
            }
            adj[i] = arr;
            deg[i] = unique;
            temp[i] = null;
        }

        var bestClique = HeuristicSearch(adj, deg);
        if (bestClique.Count == 0)
        {
            bestClique.Add(0);
        }
        int bestSize = bestClique.Count;

        bool useExact = adj.Length <= EXACT_VERTEX_THRESHOLD && actualEdges <= EXACT_EDGE_THRESHOLD;
        if (useExact)
        {
            var exact = ExactClique(adj, bestClique);
            if (exact.Count > bestSize)
            {
                bestClique = exact;
                bestSize = exact.Count;
            }
        }

        bestClique.Sort();
        Console.WriteLine(bestClique.Count);
        if (bestClique.Count > 0)
        {
            Console.WriteLine(string.Join(" ", bestClique));
        }
        else
        {
            Console.WriteLine();
        }
    }

    static List<int> HeuristicSearch(int[][] adj, int[] deg)
    {
        int n = adj.Length;
        if (n == 0) return new List<int>();

        int[] order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        Array.Sort(order, new DegreeComparer(deg));

        var best = new List<int> { order[0] };
        int bestSize = 1;

        var rng = new Random(1);
        int maxStarts = Math.Max(1, DetermineDeterministicStarts(n));
        int maxCandidates = Math.Max(1, DetermineMaxCandidateSize(n));
        int randomTries = Math.Max(0, DetermineRandomTrials(n));
        TimeSpan timeLimit = DetermineHeuristicTimeLimit(n);
        var sw = Stopwatch.StartNew();

        int deterministic = Math.Min(maxStarts, n);
        for (int idx = 0; idx < deterministic; idx++)
        {
            int v = order[idx];
            if (deg[v] + 1 <= bestSize) break;

            var clique = BuildGreedyClique(v, adj, deg, rng, maxCandidates);
            if (clique.Count > bestSize)
            {
                best = clique;
                bestSize = clique.Count;
            }

            if (sw.Elapsed >= timeLimit) break;
        }

        if (sw.Elapsed < timeLimit)
        {
            for (int t = 0; t < randomTries && sw.Elapsed < timeLimit; t++)
            {
                int v = rng.Next(n);
                if (deg[v] + 1 <= bestSize) continue;

                var clique = BuildGreedyClique(v, adj, deg, rng, maxCandidates);
                if (clique.Count > bestSize)
                {
                    best = clique;
                    bestSize = clique.Count;
                }
            }
        }

        return best;
    }

    static List<int> BuildGreedyClique(int start, int[][] adj, int[] deg, Random rng, int maxCandidateSize)
    {
        var clique = new List<int>(Math.Max(2, Math.Min(deg[start] + 1, 256)))
        {
            start
        };

        var pool = ArrayPool<int>.Shared;
        var cand = CandidateSet.FromArray(adj[start]);

        while (cand.Length > 0)
        {
            if (maxCandidateSize > 0 && cand.Length > maxCandidateSize)
            {
                var sampled = SampleCandidates(cand, maxCandidateSize, rng, pool);
                cand.Release(pool);
                cand = sampled;
            }

            int bestVertex = cand.Data[0];
            int bestDeg = deg[bestVertex];
            for (int i = 1; i < cand.Length; i++)
            {
                int v = cand.Data[i];
                int d = deg[v];
                if (d > bestDeg)
                {
                    bestDeg = d;
                    bestVertex = v;
                }
            }

            clique.Add(bestVertex);

            var next = IntersectWithNeighbor(cand, adj[bestVertex], pool);
            cand.Release(pool);
            cand = next;
        }

        cand.Release(pool);
        return clique;
    }

    static CandidateSet SampleCandidates(CandidateSet source, int limit, Random rng, ArrayPool<int> pool)
    {
        int length = source.Length;
        int[] buffer = pool.Rent(limit);
        var data = source.Data;
        int i = 0;
        for (; i < limit; i++)
        {
            buffer[i] = data[i];
        }
        for (; i < length; i++)
        {
            int j = rng.Next(i + 1);
            if (j < limit)
            {
                buffer[j] = data[i];
            }
        }
        Array.Sort(buffer, 0, limit);
        return new CandidateSet { Data = buffer, Length = limit, FromPool = true };
    }

    static CandidateSet IntersectWithNeighbor(CandidateSet cand, int[] neighbor, ArrayPool<int> pool)
    {
        if (cand.Length == 0 || neighbor.Length == 0) return CandidateSet.Empty;

        int maxLen = Math.Min(cand.Length, neighbor.Length);
        int[] buffer = pool.Rent(maxLen);
        int len = 0;
        var candData = cand.Data;
        int candLen = cand.Length;
        int i = 0, j = 0;
        while (i < candLen && j < neighbor.Length)
        {
            int a = candData[i];
            int b = neighbor[j];
            if (a == b)
            {
                buffer[len++] = a;
                i++;
                j++;
            }
            else if (a < b)
            {
                i++;
            }
            else
            {
                j++;
            }
        }

        if (len == 0)
        {
            pool.Return(buffer);
            return CandidateSet.Empty;
        }

        return new CandidateSet { Data = buffer, Length = len, FromPool = true };
    }

    static List<int> ExactClique(int[][] adj, List<int> initialBest)
    {
        int n = adj.Length;
        int wordCount = (n + 63) >> 6;
        var matrix = new ulong[n][];
        for (int i = 0; i < n; i++)
        {
            var bits = new ulong[wordCount];
            var neighbors = adj[i];
            for (int j = 0; j < neighbors.Length; j++)
            {
                int v = neighbors[j];
                int word = v >> 6;
                bits[word] |= 1UL << (v & 63);
            }
            matrix[i] = bits;
        }

        var P = new ulong[wordCount];
        for (int v = 0; v < n; v++)
        {
            SetBit(P, v);
        }
        var X = new ulong[wordCount];
        var current = new List<int>();
        var best = initialBest != null ? new List<int>(initialBest) : new List<int>();
        int bestSize = best.Count;

        var watch = Stopwatch.StartNew();
        TimeSpan limit = DetermineExactTimeLimit(n);
        bool aborted = false;
        Expand(matrix, P, X, current, ref best, ref bestSize, wordCount, watch, limit, ref aborted);
        return best;
    }

    static void Expand(
        ulong[][] adjacencyBits,
        ulong[] P,
        ulong[] X,
        List<int> current,
        ref List<int> best,
        ref int bestSize,
        int wordCount,
        Stopwatch watch,
        TimeSpan limit,
        ref bool aborted)
    {
        if (aborted) return;
        if (watch.Elapsed > limit)
        {
            aborted = true;
            return;
        }

        int pCount = CountBits(P, wordCount);
        if (current.Count + pCount <= bestSize) return;

        if (IsZero(P, wordCount) && IsZero(X, wordCount))
        {
            if (current.Count > bestSize)
            {
                best = new List<int>(current);
                bestSize = current.Count;
            }
            return;
        }

        int pivot = ChoosePivot(adjacencyBits, P, X, wordCount);
        ulong[] cand = (ulong[])P.Clone();
        if (pivot != -1)
        {
            var neigh = adjacencyBits[pivot];
            for (int i = 0; i < wordCount; i++)
            {
                cand[i] &= ~neigh[i];
            }
        }

        for (int wi = 0; wi < wordCount; wi++)
        {
            ulong word = cand[wi];
            while (word != 0)
            {
                if (aborted) return;
                if (watch.Elapsed > limit)
                {
                    aborted = true;
                    return;
                }

                int bit = BitOperations.TrailingZeroCount(word);
                int v = (wi << 6) + bit;
                word &= word - 1;
                if (v >= adjacencyBits.Length) continue;

                current.Add(v);
                var Pnext = And(P, adjacencyBits[v], wordCount);
                var Xnext = And(X, adjacencyBits[v], wordCount);
                Expand(adjacencyBits, Pnext, Xnext, current, ref best, ref bestSize, wordCount, watch, limit, ref aborted);
                current.RemoveAt(current.Count - 1);

                ClearBit(P, v);
                SetBit(X, v);

                if (current.Count + CountBits(P, wordCount) <= bestSize) return;
            }
        }
    }

    static int CountBits(ulong[] bits, int wordCount)
    {
        int sum = 0;
        for (int i = 0; i < wordCount; i++)
        {
            sum += BitOperations.PopCount(bits[i]);
        }
        return sum;
    }

    static bool IsZero(ulong[] bits, int wordCount)
    {
        for (int i = 0; i < wordCount; i++)
        {
            if (bits[i] != 0UL) return false;
        }
        return true;
    }

    static int ChoosePivot(ulong[][] matrix, ulong[] P, ulong[] X, int wordCount)
    {
        int pivot = -1;
        int best = -1;
        for (int i = 0; i < wordCount; i++)
        {
            ulong unionWord = P[i] | X[i];
            while (unionWord != 0)
            {
                int bit = BitOperations.TrailingZeroCount(unionWord);
                int v = (i << 6) + bit;
                unionWord &= unionWord - 1;
                if (v >= matrix.Length) continue;

                int count = CountBitsAnd(P, matrix[v], wordCount);
                if (count > best)
                {
                    best = count;
                    pivot = v;
                }
            }
        }
        return pivot;
    }

    static int CountBitsAnd(ulong[] a, ulong[] b, int wordCount)
    {
        int sum = 0;
        for (int i = 0; i < wordCount; i++)
        {
            sum += BitOperations.PopCount(a[i] & b[i]);
        }
        return sum;
    }

    static ulong[] And(ulong[] a, ulong[] b, int wordCount)
    {
        var res = new ulong[wordCount];
        for (int i = 0; i < wordCount; i++)
        {
            res[i] = a[i] & b[i];
        }
        return res;
    }

    static void SetBit(ulong[] bits, int index)
    {
        int word = index >> 6;
        if (word < bits.Length)
        {
            bits[word] |= 1UL << (index & 63);
        }
    }

    static void ClearBit(ulong[] bits, int index)
    {
        int word = index >> 6;
        if (word < bits.Length)
        {
            bits[word] &= ~(1UL << (index & 63));
        }
    }

    static int DetermineDeterministicStarts(int n)
    {
        if (n <= 2000) return n;
        if (n <= 10_000) return 2500;
        if (n <= 50_000) return 1500;
        if (n <= 200_000) return 900;
        if (n <= 500_000) return 600;
        return 400;
    }

    static int DetermineMaxCandidateSize(int n)
    {
        if (n <= 2000) return 20000;
        if (n <= 10_000) return 8000;
        if (n <= 50_000) return 4000;
        if (n <= 200_000) return 2500;
        if (n <= 500_000) return 1500;
        return 900;
    }

    static int DetermineRandomTrials(int n)
    {
        if (n <= 5000) return Math.Min(300, n);
        if (n <= 20_000) return 300;
        if (n <= 100_000) return 200;
        if (n <= 500_000) return 150;
        return 120;
    }

    static TimeSpan DetermineHeuristicTimeLimit(int n)
    {
        if (n <= 5000) return TimeSpan.FromSeconds(2);
        if (n <= 20_000) return TimeSpan.FromSeconds(4);
        if (n <= 100_000) return TimeSpan.FromSeconds(6);
        if (n <= 500_000) return TimeSpan.FromSeconds(9);
        return TimeSpan.FromSeconds(12);
    }

    static TimeSpan DetermineExactTimeLimit(int n)
    {
        if (n <= 200) return TimeSpan.FromSeconds(15);
        if (n <= 400) return TimeSpan.FromSeconds(25);
        return TimeSpan.FromSeconds(40);
    }

    struct CandidateSet
    {
        public int[] Data;
        public int Length;
        public bool FromPool;

        public static CandidateSet FromArray(int[] data) =>
            new CandidateSet { Data = data, Length = data.Length, FromPool = false };

        public static CandidateSet Empty =>
            new CandidateSet { Data = Array.Empty<int>(), Length = 0, FromPool = false };

        public void Release(ArrayPool<int> pool)
        {
            if (FromPool && Data != null)
            {
                pool.Return(Data);
            }
        }
    }

    sealed class DegreeComparer : IComparer<int>
    {
        private readonly int[] _deg;
        public DegreeComparer(int[] deg) => _deg = deg;

        public int Compare(int x, int y)
        {
            int cmp = _deg[y].CompareTo(_deg[x]);
            if (cmp != 0) return cmp;
            return x.CompareTo(y);
        }
    }

    sealed class FastReader
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _ptr;
        private int _len;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            while ((c = Read()) != -1 && c <= ' ') ;
            if (c == -1) return 0;
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }
}
```
