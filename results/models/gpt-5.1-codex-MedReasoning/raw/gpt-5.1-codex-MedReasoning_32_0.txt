```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <limits>
#include <algorithm>
#include <numeric>
#include <functional>

struct AdjEdge {
    int to;
    long long w;
    int id;
};

struct DSU {
    std::vector<int> parent, rnk;
    explicit DSU(int n) : parent(n), rnk(n, 0) {
        std::iota(parent.begin(), parent.end(), 0);
    }
    int find(int x) {
        if (parent[x] != x) parent[x] = find(parent[x]);
        return parent[x];
    }
    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (rnk[a] < rnk[b]) std::swap(a, b);
        parent[b] = a;
        if (rnk[a] == rnk[b]) ++rnk[a];
        return true;
    }
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    
    const long long INF = std::numeric_limits<long long>::max() / 4;
    
    int n, m, k;
    if (!(std::cin >> n >> m >> k)) return 0;
    
    std::vector<std::vector<AdjEdge>> graph(n);
    std::vector<std::pair<int, int>> edges(m);
    std::vector<long long> weights(m);
    
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        std::cin >> u >> v >> w;
        edges[i] = {u, v};
        weights[i] = w;
        graph[u].push_back({v, w, i});
        graph[v].push_back({u, w, i});
    }
    
    std::vector<int> terminals(k);
    for (int i = 0; i < k; ++i) std::cin >> terminals[i];
    
    if (k <= 1) {
        std::cout << 0 << " " << 0 << "\n";
        return 0;
    }
    
    std::vector<std::vector<long long>> terminalDist(k, std::vector<long long>(n, INF));
    
    auto dijkstra = [&](int source,
                        std::vector<long long>& dist,
                        std::vector<int>* parent_node,
                        std::vector<int>* parent_edge,
                        int target) {
        std::fill(dist.begin(), dist.end(), INF);
        if (parent_node) std::fill(parent_node->begin(), parent_node->end(), -1);
        if (parent_edge) std::fill(parent_edge->begin(), parent_edge->end(), -1);
        
        using State = std::pair<long long, int>;
        std::priority_queue<State, std::vector<State>, std::greater<State>> pq;
        dist[source] = 0;
        pq.push({0, source});
        
        while (!pq.empty()) {
            auto [d, u] = pq.top();
            pq.pop();
            if (d != dist[u]) continue;
            if (target != -1 && u == target) break;
            for (const auto& e : graph[u]) {
                long long nd = d + e.w;
                if (nd < dist[e.to]) {
                    dist[e.to] = nd;
                    if (parent_node) (*parent_node)[e.to] = u;
                    if (parent_edge) (*parent_edge)[e.to] = e.id;
                    pq.push({nd, e.to});
                }
            }
        }
    };
    
    for (int i = 0; i < k; ++i) {
        dijkstra(terminals[i], terminalDist[i], nullptr, nullptr, -1);
    }
    
    std::vector<long long> key(k, INF);
    std::vector<int> parent(k, -1);
    std::vector<char> used(k, false);
    key[0] = 0;
    
    for (int iter = 0; iter < k; ++iter) {
        int u = -1;
        for (int i = 0; i < k; ++i) {
            if (!used[i] && (u == -1 || key[i] < key[u])) u = i;
        }
        if (u == -1 || key[u] >= INF / 2) {
            std::cout << 0 << " " << 0 << "\n";
            return 0;
        }
        used[u] = true;
        for (int v = 0; v < k; ++v) {
            if (used[v]) continue;
            long long w = terminalDist[u][terminals[v]];
            if (w < key[v]) {
                key[v] = w;
                parent[v] = u;
            }
        }
    }
    
    std::vector<std::pair<int, int>> metricEdges;
    for (int i = 1; i < k; ++i) {
        if (parent[i] == -1) {
            std::cout << 0 << " " << 0 << "\n";
            return 0;
        }
        metricEdges.push_back({i, parent[i]});
    }
    
    std::vector<char> inSteiner(m, false);
    std::vector<long long> distTmp(n, INF);
    std::vector<int> parentNode(n, -1), parentEdge(n, -1);
    
    for (auto [childIdx, parentIdx] : metricEdges) {
        int src = terminals[childIdx];
        int dst = terminals[parentIdx];
        dijkstra(src, distTmp, &parentNode, &parentEdge, dst);
        if (distTmp[dst] >= INF / 2) {
            std::cout << 0 << " " << 0 << "\n";
            return 0;
        }
        int cur = dst;
        while (cur != src) {
            int eid = parentEdge[cur];
            if (eid < 0) {
                std::cout << 0 << " " << 0 << "\n";
                return 0;
            }
            inSteiner[eid] = true;
            cur = parentNode[cur];
            if (cur < 0) {
                std::cout << 0 << " " << 0 << "\n";
                return 0;
            }
        }
    }
    
    std::vector<int> candidateEdges;
    for (int i = 0; i < m; ++i) {
        if (inSteiner[i]) candidateEdges.push_back(i);
    }
    if (candidateEdges.empty()) {
        std::cout << 0 << " " << 0 << "\n";
        return 0;
    }
    
    std::sort(candidateEdges.begin(), candidateEdges.end(), [&](int a, int b) {
        if (weights[a] != weights[b]) return weights[a] < weights[b];
        return a < b;
    });
    
    DSU dsu(n);
    std::vector<int> finalEdges;
    long long totalWeight = 0;
    for (int id : candidateEdges) {
        int u = edges[id].first;
        int v = edges[id].second;
        if (dsu.unite(u, v)) {
            finalEdges.push_back(id);
            totalWeight += weights[id];
        }
    }
    
    int root = dsu.find(terminals[0]);
    for (int t : terminals) {
        if (dsu.find(t) != root) {
            std::cout << 0 << " " << 0 << "\n";
            return 0;
        }
    }
    
    std::cout << totalWeight << " " << finalEdges.size() << "\n";
    for (int id : finalEdges) {
        std::cout << edges[id].first << " " << edges[id].second << "\n";
    }
    
    return 0;
}
```
