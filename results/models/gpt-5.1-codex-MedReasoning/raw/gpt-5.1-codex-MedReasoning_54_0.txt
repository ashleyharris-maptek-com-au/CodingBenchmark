```python
import math
from statistics import median

# Persistent memory across sim steps (reset between scenarios / go-arounds)
_AP_MEM = {}


def _clamp(val, lo, hi):
    return max(lo, min(hi, val))


def _is_valid(value):
    if value is None:
        return False
    if isinstance(value, float) and math.isnan(value):
        return False
    return True


def _fuse(values, valids=None, default=0.0):
    data = []
    for idx, val in enumerate(values):
        if not _is_valid(val):
            continue
        valid_flag = True
        if valids is not None and idx < len(valids):
            valid_flag = bool(valids[idx])
        if valid_flag:
            data.append(val)
    if data:
        data.sort()
        mid = len(data) // 2
        if len(data) % 2:
            return data[mid]
        return 0.5 * (data[mid - 1] + data[mid])
    for val in values:
        if _is_valid(val):
            return val
    return default


def _wrap_angle(angle):
    if not _is_valid(angle):
        return 0.0
    while angle > 180.0:
        angle -= 360.0
    while angle < -180.0:
        angle += 360.0
    return angle


def _bool(val):
    if isinstance(val, bool):
        return val
    if val is None:
        return False
    return bool(val)


def autopilot_step(state: dict, dt: float) -> dict:
    global _AP_MEM
    if not _AP_MEM:
        _AP_MEM = {
            "phase": "init",
            "time": 0.0,
            "throttle_int": 0.0,
            "runway_elev": None,
            "unstable_timer": 0.0,
            "flare_start_alt": None,
            "flare_start_time": None,
            "ga_reason": None,
        }
    mem = _AP_MEM
    mem["time"] += dt

    commands = {}

    # --- Sensor fusion ---
    glideslope = _fuse(
        [
            state.get("glideslope_1_dots"),
            state.get("glideslope_2_dots"),
            state.get("glideslope_3_dots"),
        ],
        [
            state.get("glideslope_1_valid", 1),
            state.get("glideslope_2_valid", 1),
            state.get("glideslope_3_valid", 1),
        ],
        0.0,
    )
    localizer = _fuse(
        [
            state.get("localizer_1_dots"),
            state.get("localizer_2_dots"),
            state.get("localizer_3_dots"),
        ],
        [
            state.get("localizer_1_valid", 1),
            state.get("localizer_2_valid", 1),
            state.get("localizer_3_valid", 1),
        ],
        0.0,
    )

    ias = _fuse(
        [state.get("ias_1_kt"), state.get("ias_2_kt"), state.get("ias_3_kt")],
        [state.get("ias_1_valid", 1), state.get("ias_2_valid", 1), state.get("ias_3_valid", 1)],
        state.get("groundspeed_kt", 140.0),
    )

    tas = _fuse(
        [state.get("tas_1_kt"), state.get("tas_2_kt"), state.get("tas_3_kt")],
        None,
        ias,
    )

    pitch = _fuse(
        [state.get("irs_1_pitch_deg"), state.get("irs_2_pitch_deg"), state.get("irs_3_pitch_deg")],
        [state.get("irs_1_valid", 1), state.get("irs_2_valid", 1), state.get("irs_3_valid", 1)],
        3.0,
    )
    roll = _fuse(
        [state.get("irs_1_roll_deg"), state.get("irs_2_roll_deg"), state.get("irs_3_roll_deg")],
        [state.get("irs_1_valid", 1), state.get("irs_2_valid", 1), state.get("irs_3_valid", 1)],
        0.0,
    )

    heading = _fuse(
        [state.get("irs_1_heading_deg"), state.get("irs_2_heading_deg"), state.get("irs_3_heading_deg")],
        [state.get("irs_1_valid", 1), state.get("irs_2_valid", 1), state.get("irs_3_valid", 1)],
        state.get("target_hdg_deg", 0.0),
    )
    track = _fuse(
        [state.get("irs_1_track_deg"), state.get("irs_2_track_deg"), state.get("irs_3_track_deg")],
        [state.get("irs_1_valid", 1), state.get("irs_2_valid", 1), state.get("irs_3_valid", 1)],
        heading,
    )

    vs = _fuse(
        [state.get("irs_1_vs_fpm"), state.get("irs_2_vs_fpm"), state.get("irs_3_vs_fpm")],
        [state.get("irs_1_valid", 1), state.get("irs_2_valid", 1), state.get("irs_3_valid", 1)],
        -700.0,
    )

    groundspeed = state.get("groundspeed_kt", _fuse([state.get("irs_1_gs_kt"), state.get("irs_2_gs_kt"), state.get("irs_3_gs_kt")], None, tas))

    baro_alt = _fuse(
        [state.get("baro_alt_1_ft"), state.get("baro_alt_2_ft"), state.get("baro_alt_3_ft")],
        [state.get("baro_alt_1_valid", 1), state.get("baro_alt_2_valid", 1), state.get("baro_alt_3_valid", 1)],
        state.get("gps_1_alt_ft", 0.0),
    )

    radio_alt_candidates = [
        state.get("radio_alt_ft"),
        state.get("ground_radio_alt_ft"),
        state.get("ground_radio_alt"),
        state.get("Ground_radio_alt_ft"),
        state.get("radio_alt_ft"),  # duplicates safe
    ]
    radio_alt = None
    for val in radio_alt_candidates:
        if _is_valid(val):
            radio_alt = val
            break

    dme = _fuse([state.get("dme_1_nm"), state.get("dme_2_nm")], None, None)
    if not _is_valid(dme):
        dme = None

    # Estimate runway elevation once radio alt + baro available
    if _is_valid(baro_alt) and _is_valid(radio_alt) and radio_alt > 5.0:
        mem["runway_elev"] = baro_alt - radio_alt

    # Altitude AGL estimate
    if _is_valid(radio_alt):
        alt_agl = radio_alt
    elif mem.get("runway_elev") is not None:
        alt_agl = max(0.0, baro_alt - mem["runway_elev"])
    elif dme is not None:
        alt_agl = max(0.0, dme * 318.0)  # ~3° path
    else:
        alt_agl = max(0.0, baro_alt)

    target_vref = state.get("target_vref_kt") or state.get("target_ias_kt") or 140.0
    target_vs = state.get("target_vs_fpm", -700.0)

    weight_on_wheels = _bool(state.get("weight_on_wheels")) or _bool(state.get("on_ground"))
    below_dh = _bool(state.get("below_decision_height"))

    # Engine availability
    engine1_running = _bool(state.get("engine_1_running")) or (state.get("engine_1_n1_pct", 0.0) > 20.0)
    engine2_running = _bool(state.get("engine_2_running")) or (state.get("engine_2_n1_pct", 0.0) > 20.0)
    engines_available = int(engine1_running) + int(engine2_running)

    # --- Phase management ---
    if mem["phase"] == "init":
        mem["phase"] = "approach"

    if mem["phase"] != "go_around":
        unstable = False
        gs_tol = 1.4 if alt_agl > 1000 else 1.0
        loc_tol = 1.5 if alt_agl > 1000 else 1.1
        speed_low = target_vref - 15.0
        speed_high = target_vref + 25.0

        if abs(glideslope) > gs_tol or abs(localizer) > loc_tol:
            unstable = True
        if ias < speed_low or ias > speed_high:
            unstable = True
        if alt_agl < 100 and abs(glideslope) > 1.0:
            unstable = True
        if below_dh and (abs(localizer) > 1.0 or abs(glideslope) > 1.0):
            unstable = True
        # Requested go-around (target altitude > 300 ft near runway)
        if state.get("target_alt_ft", 0.0) > 300.0 and alt_agl < 400.0:
            unstable = True

        # Can't go around without engines
        if engines_available == 0:
            unstable = False

        if unstable:
            mem["unstable_timer"] += dt
            if mem["unstable_timer"] > 2.0:
                mem["phase"] = "go_around"
                mem["ga_reason"] = "unstable"
        else:
            mem["unstable_timer"] = max(0.0, mem["unstable_timer"] - dt * 0.5)

    # Transition to flare / touchdown / rollout
    if mem["phase"] not in ("go_around", "rollout"):
        if mem["phase"] == "approach" and alt_agl < 60.0:
            mem["phase"] = "flare"
            mem["flare_start_alt"] = alt_agl
            mem["flare_start_time"] = mem["time"]
        if mem["phase"] == "flare" and (alt_agl < 10.0 or weight_on_wheels):
            mem["phase"] = "touchdown"
        if mem["phase"] == "touchdown" and (weight_on_wheels or alt_agl < 5.0):
            mem["phase"] = "rollout"

    # Desired flap/gear schedule
    flaps = 20.0
    if alt_agl < 1500:
        flaps = 25.0
    if alt_agl < 900:
        flaps = 30.0
    if alt_agl < 500:
        flaps = 35.0
    if alt_agl < 300:
        flaps = 38.0
    flaps = _clamp(flaps, 0.0, 40.0)
    commands["flaps"] = flaps

    gear = 1.0 if alt_agl < 2000.0 else 0.0
    commands["gear"] = gear

    commands["autobrake"] = 0.7 if gear > 0.5 else 0.0

    # Default speedbrake/spoiler
    commands["speedbrake"] = 0.0
    commands["spoiler"] = 0.0

    # --- GO AROUND handling ---
    if mem["phase"] == "go_around":
        commands["throttle_1"] = 1.0 if engine1_running else 0.0
        commands["throttle_2"] = 1.0 if engine2_running else 0.0
        commands["elevator"] = 0.55
        commands["aileron"] = _clamp(-roll * 0.04 - localizer * 0.2, -0.8, 0.8)
        beta = _wrap_angle(heading - track)
        commands["rudder"] = _clamp(-beta * 0.02 - localizer * 0.05, -0.5, 0.5)
        commands["speedbrake"] = 0.0
        commands["spoiler"] = 0.0
        commands["flaps"] = min(flaps, 20.0)
        commands["gear"] = 1.0
        commands["reverse_1"] = 0.0
        commands["reverse_2"] = 0.0
        return commands

    # --- Approach / flare controls ---
    approach_margin = 5.0
    desired_speed = target_vref + approach_margin
    if mem["phase"] in ("flare", "touchdown"):
        desired_speed = target_vref
    if engines_available == 0:
        # Glide: preserve extra speed to avoid stall
        desired_speed = max(target_vref + 15.0, ias)

    ias_error = desired_speed - ias
    mem["throttle_int"] = _clamp(mem["throttle_int"] + ias_error * dt * 0.005, -0.2, 0.2)
    throttle_base = 0.45
    throttle = throttle_base + 0.012 * ias_error + mem["throttle_int"] + (target_vs - vs) * 0.0005
    throttle = _clamp(throttle, 0.0, 1.0)

    if mem["phase"] == "flare":
        flare_progress = _clamp((50.0 - alt_agl) / 50.0, 0.0, 1.0)
        throttle *= 0.3
        throttle = min(throttle, 0.25)
    if mem["phase"] in ("touchdown", "rollout"):
        throttle = 0.0

    commands["throttle_1"] = throttle if engine1_running else 0.0
    commands["throttle_2"] = throttle if engine2_running else 0.0

    # Elevator control
    gs_error = _clamp(glideslope, -2.5, 2.5)
    vs_error = (target_vs - vs) / 1000.0
    desired_pitch = 2.5 + 0.01 * (desired_speed - target_vref)
    pitch_error = desired_pitch - pitch
    elevator = _clamp(gs_error * 0.35 + vs_error * 0.2 + pitch_error * 0.08, -0.9, 0.9)

    if mem["phase"] == "flare":
        flare_bias = _clamp((30.0 - alt_agl) / 30.0, 0.0, 1.0)
        elevator += flare_bias * 0.25
    if mem["phase"] in ("touchdown", "rollout"):
        elevator *= 0.3

    commands["elevator"] = _clamp(elevator, -1.0, 1.0)

    # Aileron / roll control
    aileron = _clamp(localizer * 0.4 - roll * 0.03, -1.0, 1.0)
    commands["aileron"] = aileron

    # Rudder / yaw (crab into crosswind)
    beta = _wrap_angle(heading - track)
    rudder = _clamp(-beta * 0.02 - localizer * 0.05, -0.5, 0.5)
    if mem["phase"] in ("flare", "touchdown", "rollout"):
        # De-crab more aggressively near ground
        rudder = _clamp(rudder - localizer * 0.1, -0.6, 0.6)
    commands["rudder"] = rudder

    # --- Landing / rollout logic ---
    if mem["phase"] == "rollout":
        commands["speedbrake"] = 1.0
        commands["spoiler"] = 1.0
        commands["reverse_1"] = 0.0
        commands["reverse_2"] = 0.0
        if engines_available and groundspeed > 40.0:
            commands["reverse_1"] = 0.7 if engine1_running else 0.0
            commands["reverse_2"] = 0.7 if engine2_running else 0.0
        commands["throttle_1"] = 0.0
        commands["throttle_2"] = 0.0
        commands["autobrake"] = 1.0
    else:
        commands["reverse_1"] = 0.0
        commands["reverse_2"] = 0.0

    return commands
```
