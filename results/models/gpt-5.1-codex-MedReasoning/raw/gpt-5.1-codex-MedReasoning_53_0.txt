```python
import math

MEAN_MOTION = 0.00113  # rad/s at ~400 km
N2 = MEAN_MOTION ** 2
BASE_MAX_ACCEL = 0.025  # m/s^2 equivalent for thrust = 1.0 (tunable)
POS_GAINS = {'x': 0.02, 'y': 0.015, 'z': 0.02}
VEL_GAINS = {'x': 0.08, 'y': 0.07, 'z': 0.08}
PHASE_GAIN_FAR = 0.20
PHASE_GAIN_MID = 0.12
PHASE_LIMIT = 80.0  # meters radial bias limit
BRAKE_GAIN = 0.75
POS_STALE_LIMIT = 20.0  # seconds
VEL_STALE_LIMIT = 10.0  # seconds
RANGE_STALE_LIMIT = 35.0  # seconds
ATT_RATE_LIMIT = 0.035
_AXES = ('x', 'y', 'z')

_POS_SOURCES = [
    ('radar_rel_pos_{axis}_m', 'radar_rel_{axis}_valid'),
    ('lidar_rel_pos_{axis}_m', 'lidar_rel_{axis}_valid'),
]

_VEL_SOURCES = [
    ('radar_rel_vel_{axis}_ms', 'radar_relv_{axis}_valid'),
    ('lidar_rel_vel_{axis}_ms', 'lidar_relv_{axis}_valid'),
]

_AP_MEM = None


def _init_memory():
    return {
        'pos': [0.0, 0.0, 0.0],
        'vel': [0.0, 0.0, 0.0],
        'pos_valid': [False, False, False],
        'vel_valid': [False, False, False],
        'pos_age': [POS_STALE_LIMIT + 1.0] * 3,
        'vel_age': [VEL_STALE_LIMIT + 1.0] * 3,
        'range': None,
        'range_valid': False,
        'range_age': RANGE_STALE_LIMIT + 1.0,
    }


def _clamp(value, minimum, maximum):
    return max(minimum, min(maximum, value))


def _clamp_abs(value, limit):
    return _clamp(value, -limit, limit)


def _sanitize_dt(dt):
    try:
        val = float(dt)
        if not math.isfinite(val) or val <= 0.0:
            return 1.0
        return val
    except (TypeError, ValueError):
        return 1.0


def _get_valid_value(state, value_key, valid_key=None):
    if valid_key is not None:
        if not bool(state.get(valid_key)):
            return None
    value = state.get(value_key)
    if value is None:
        return None
    try:
        val = float(value)
    except (TypeError, ValueError):
        return None
    if not math.isfinite(val):
        return None
    return val


def _fetch_vector(state, kind, dt):
    global _AP_MEM
    sources = _POS_SOURCES if kind == 'pos' else _VEL_SOURCES
    mem_key = kind
    mem_valid_key = f'{kind}_valid'
    age_key = f'{kind}_age'
    stale_limit = POS_STALE_LIMIT if kind == 'pos' else VEL_STALE_LIMIT

    values, valids = [], []
    for idx, axis in enumerate(_AXES):
        axis_measurements = []
        for value_fmt, valid_fmt in sources:
            measurement = _get_valid_value(state, value_fmt.format(axis=axis), valid_fmt.format(axis=axis))
            if measurement is not None:
                axis_measurements.append(measurement)

        _AP_MEM[age_key][idx] = min(_AP_MEM[age_key][idx] + dt, stale_limit + 5.0)

        if axis_measurements:
            axis_value = sum(axis_measurements) / len(axis_measurements)
            axis_valid = True
            _AP_MEM[mem_key][idx] = axis_value
            _AP_MEM[mem_valid_key][idx] = True
            _AP_MEM[age_key][idx] = 0.0
        elif _AP_MEM[mem_valid_key][idx] and _AP_MEM[age_key][idx] < stale_limit:
            axis_value = _AP_MEM[mem_key][idx]
            axis_valid = True
        else:
            axis_value = 0.0
            axis_valid = False
            _AP_MEM[mem_valid_key][idx] = False
        values.append(axis_value)
        valids.append(axis_valid)
    return values, valids


def _median(values):
    if not values:
        return None
    sorted_vals = sorted(values)
    mid = len(sorted_vals) // 2
    if len(sorted_vals) % 2 == 1:
        return sorted_vals[mid]
    return 0.5 * (sorted_vals[mid - 1] + sorted_vals[mid])


def _range_measurement(state, pos, pos_valid):
    candidates = []
    for i in range(1, 4):
        val = _get_valid_value(state, f'range_{i}_m', f'range_{i}_valid')
        if val is not None and val > 0.0:
            candidates.append(val)

    target_range = _get_valid_value(state, 'target_dock_range_m')
    if target_range is not None and target_range > 0.0:
        candidates.append(target_range)

    if bool(state.get('dock_camera_active')):
        cam_range = state.get('dock_camera_range_m')
        if cam_range is not None:
            try:
                cam_val = float(cam_range)
            except (TypeError, ValueError):
                cam_val = None
            if cam_val is not None and math.isfinite(cam_val) and cam_val > 0.0:
                candidates.append(cam_val)

    if pos_valid and all(pos_valid):
        los = math.sqrt(pos[0] ** 2 + pos[1] ** 2 + pos[2] ** 2)
        candidates.append(los)

    return _median(candidates)


def _speed_limit(distance):
    if distance is None:
        return 0.2
    if distance > 1500:
        return 1.2
    if distance > 800:
        return 0.9
    if distance > 400:
        return 0.6
    if distance > 250:
        return 0.45
    if distance > 120:
        return 0.3
    if distance > 60:
        return 0.20
    if distance > 30:
        return 0.14
    if distance > 15:
        return 0.09
    if distance > 8:
        return 0.06
    if distance > 4:
        return 0.045
    return 0.025


def _closing_limit(distance):
    if distance is None:
        return 0.3
    if distance > 500:
        return 0.6
    if distance > 250:
        return 0.45
    if distance > 120:
        return 0.30
    if distance > 60:
        return 0.20
    if distance > 30:
        return 0.15
    if distance > 15:
        return 0.09
    if distance > 8:
        return 0.06
    if distance > 4:
        return 0.045
    return 0.03


def _fuel_thrust_limit(state):
    pct = _get_valid_value(state, 'fuel_remaining_pct')
    if pct is None:
        return 1.0
    if pct < 5.0:
        return 0.35
    if pct < 10.0:
        return 0.45
    if pct < 20.0:
        return 0.65
    if pct < 35.0:
        return 0.85
    return 1.0


def _zero_output():
    return {
        'thrust_x': 0.0,
        'thrust_y': 0.0,
        'thrust_z': 0.0,
        'att_rate_roll': 0.0,
        'att_rate_pitch': 0.0,
        'att_rate_yaw': 0.0,
    }


def autopilot_step(state: dict, dt: float) -> dict:
    global _AP_MEM
    if _AP_MEM is None:
        _AP_MEM = _init_memory()

    step_dt = _sanitize_dt(dt)

    pos, pos_valid = _fetch_vector(state, 'pos', step_dt)
    vel, vel_valid = _fetch_vector(state, 'vel', step_dt)

    range_meas = _range_measurement(state, pos, pos_valid)
    _AP_MEM['range_age'] = min(_AP_MEM['range_age'] + step_dt, RANGE_STALE_LIMIT + 5.0)
    if range_meas is not None:
        distance = max(range_meas, 0.01)
        _AP_MEM['range'] = distance
        _AP_MEM['range_valid'] = True
        _AP_MEM['range_age'] = 0.0
    elif _AP_MEM['range_valid'] and _AP_MEM['range_age'] < RANGE_STALE_LIMIT:
        distance = _AP_MEM['range']
    else:
        distance = None
        _AP_MEM['range_valid'] = False

    if distance is None:
        return _zero_output()

    pos_all_valid = all(pos_valid)
    vel_all_valid = all(vel_valid)

    rel_speed = math.sqrt(vel[0] ** 2 + vel[1] ** 2 + vel[2] ** 2) if vel_all_valid else None

    closing_rate = None
    unit_vector = None
    if pos_all_valid and vel_all_valid and distance > 0.001:
        inv_dist = 1.0 / distance
        closing_rate = -(pos[0] * vel[0] + pos[1] * vel[1] + pos[2] * vel[2]) * inv_dist
        unit_vector = [pos[0] * inv_dist, pos[1] * inv_dist, pos[2] * inv_dist]

    speed_limit = _speed_limit(distance)
    radial_limit = min(0.35, max(0.05, 0.6 * speed_limit + 0.02))
    cross_limit = min(0.3, max(0.04, 0.6 * speed_limit + 0.015))

    phase_bias = 0.0
    if pos_valid[1] and distance > 80.0:
        phase_gain = PHASE_GAIN_FAR if distance > 400.0 else PHASE_GAIN_MID
        desired_bias = phase_gain * pos[1]
        phase_bias = _clamp(desired_bias, -PHASE_LIMIT, PHASE_LIMIT)

    v_refs = {'x': 0.0, 'y': 0.0, 'z': 0.0}

    if pos_valid[0] or vel_valid[0]:
        x_error = pos[0] - phase_bias if pos_valid[0] else -phase_bias
        v_refs['x'] = _clamp_abs(-POS_GAINS['x'] * x_error, radial_limit)

    if pos_valid[1] or vel_valid[1]:
        y_value = pos[1] if pos_valid[1] else 0.0
        v_refs['y'] = _clamp_abs(-POS_GAINS['y'] * y_value, speed_limit)
        if pos_valid[1]:
            min_close = min(speed_limit, 0.02 if distance > 50.0 else (0.015 if distance > 15.0 else 0.008))
            if y_value < -5.0 and distance > 30.0:
                v_refs['y'] = max(v_refs['y'], min_close)
            elif y_value > 5.0 and distance > 30.0:
                v_refs['y'] = min(v_refs['y'], -min_close)

    if pos_valid[2] or vel_valid[2]:
        z_value = pos[2] if pos_valid[2] else 0.0
        v_refs['z'] = _clamp_abs(-POS_GAINS['z'] * z_value, cross_limit)

    acc = {'x': 0.0, 'y': 0.0, 'z': 0.0}
    for idx, axis in enumerate(_AXES):
        if vel_valid[idx]:
            acc[axis] = VEL_GAINS[axis] * (v_refs[axis] - vel[idx])

    if pos_valid[0]:
        acc['x'] += 3.0 * N2 * pos[0]
    if vel_valid[1]:
        acc['x'] += 2.0 * MEAN_MOTION * vel[1]
    if vel_valid[0]:
        acc['y'] -= 2.0 * MEAN_MOTION * vel[0]
    if pos_valid[2]:
        acc['z'] += N2 * pos[2]

    if closing_rate is not None and unit_vector is not None:
        close_limit = _closing_limit(distance)
        if closing_rate > close_limit:
            over = closing_rate - close_limit
            brake = BRAKE_GAIN * over
            for idx, axis in enumerate(_AXES):
                if pos_valid[idx]:
                    acc[axis] -= brake * unit_vector[idx]
        elif closing_rate < -close_limit and distance > 20.0:
            shortage = min(0.05, (-close_limit - closing_rate))
            boost = 0.4 * shortage
            for idx, axis in enumerate(_AXES):
                if pos_valid[idx]:
                    acc[axis] += boost * unit_vector[idx]

    thrust_limit = _fuel_thrust_limit(state)
    if distance < 10.0:
        thrust_limit = min(thrust_limit, 0.7)
    if distance < 5.0:
        thrust_limit = min(thrust_limit, 0.5)
    if distance < 2.5:
        thrust_limit = min(thrust_limit, 0.35)

    thrust_cmds = {}
    for idx, axis in enumerate(_AXES):
        if not (pos_valid[idx] or vel_valid[idx]):
            thrust_cmds[axis] = 0.0
            continue
        axis_acc = acc[axis]
        thr = _clamp(axis_acc / BASE_MAX_ACCEL, -1.0, 1.0)
        thr = _clamp(thr, -thrust_limit, thrust_limit)
        if abs(thr) < 0.005:
            thr = 0.0
        thrust_cmds[axis] = thr

    attitude_cmd = {'roll': 0.0, 'pitch': 0.0, 'yaw': 0.0}
    if bool(state.get('dock_camera_active', False)):
        bx = _get_valid_value(state, 'dock_camera_bearing_x', 'dock_camera_bearing_x_valid')
        by = _get_valid_value(state, 'dock_camera_bearing_y', 'dock_camera_bearing_y_valid')
        if bx is not None:
            attitude_cmd['yaw'] = _clamp(-0.4 * bx, -ATT_RATE_LIMIT, ATT_RATE_LIMIT)
        if by is not None:
            attitude_cmd['pitch'] = _clamp(0.4 * by, -ATT_RATE_LIMIT, ATT_RATE_LIMIT)

    return {
        'thrust_x': thrust_cmds.get('x', 0.0),
        'thrust_y': thrust_cmds.get('y', 0.0),
        'thrust_z': thrust_cmds.get('z', 0.0),
        'att_rate_roll': attitude_cmd['roll'],
        'att_rate_pitch': attitude_cmd['pitch'],
        'att_rate_yaw': attitude_cmd['yaw'],
    }
```
