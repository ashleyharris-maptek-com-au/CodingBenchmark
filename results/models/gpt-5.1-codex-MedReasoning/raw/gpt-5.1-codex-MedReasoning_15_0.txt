```rust
use std::io::{self, Read};

const BASE_VERTS: [[f64; 3]; 4] = [
    [1.0, 0.0, -0.707],
    [-1.0, 0.0, -0.707],
    [0.0, 1.0, 0.707],
    [0.0, -1.0, 0.707],
];
const SQRT_ONE_POINT_FIVE: f64 = 1.224_744_871_391_589;
const EPS: f64 = 1e-9;

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut iter = input.split_whitespace();

    let nv: usize = iter.next().unwrap_or("0").parse().unwrap_or(0);
    if nv == 0 {
        println!("0");
        return;
    }

    let mut polygon = Vec::with_capacity(nv);
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for _ in 0..nv {
        let x: f64 = iter.next().unwrap_or("0").parse().unwrap_or(0.0);
        let y: f64 = iter.next().unwrap_or("0").parse().unwrap_or(0.0);
        polygon.push((x, y));
        min_x = min_x.min(x);
        max_x = max_x.max(x);
        min_y = min_y.min(y);
        max_y = max_y.max(y);
    }

    let sx: f64 = iter.next().unwrap_or("0").parse().unwrap_or(0.0);
    let sy: f64 = iter.next().unwrap_or("0").parse().unwrap_or(0.0);
    let sz: f64 = iter.next().unwrap_or("-1").parse().unwrap_or(-1.0);

    let mut sun = (sx, sy, sz);
    if sun.0 == 0.0 && sun.1 == 0.0 && sun.2 == 0.0 {
        sun = (0.0, 0.0, -1.0);
    }
    if sun.2 > 0.0 {
        sun.0 = -sun.0;
        sun.1 = -sun.1;
        sun.2 = -sun.2;
    }
    if sun.2 > -1e-6 {
        sun.2 = -1e-6;
    }

    let width = (max_x - min_x).abs();
    let height = (max_y - min_y).abs();
    let mut scale = choose_scale(width.max(height));
    if nv > 400 {
        scale *= 0.7;
    } else if nv > 120 {
        scale *= 0.85;
    }
    scale = scale.clamp(0.3, 10.0);

    let mut pad = 0.0;
    for _ in 0..6 {
        pad = compute_inradius(scale, sun).max(1e-6);
        let span_x = width + 2.0 * pad;
        let span_y = height + 2.0 * pad;
        let approx_cols = (span_x / pad).ceil().max(1.0);
        let approx_rows = (span_y / pad).ceil().max(1.0);
        if approx_cols * approx_rows <= 60_000.0 || scale >= 9.5 {
            break;
        }
        let new_scale = (scale * 1.25).min(10.0);
        if (new_scale - scale).abs() < 1e-6 {
            break;
        }
        scale = new_scale;
    }
    pad = compute_inradius(scale, sun).max(1e-6);
    let step = pad;

    let y_start = min_y - pad;
    let y_end = max_y + pad;
    let total_rows = ((y_end - y_start) / step).ceil().max(0.0) as usize + 2;

    let mut centers = Vec::new();
    let approx_cols = ((width + 2.0 * pad) / step).ceil().max(1.0);
    let approx_rows = ((height + 2.0 * pad) / step).ceil().max(1.0);
    let reserve = (approx_cols * approx_rows).min(200_000.0) as usize;
    centers.reserve(reserve);

    if nv < 3 {
        for row in 0..=total_rows {
            let y = y_start + row as f64 * step;
            if y > y_end + EPS {
                break;
            }
            let x_start = min_x - pad;
            let x_end = max_x + pad;
            let cols = ((x_end - x_start) / step).ceil().max(0.0) as usize + 2;
            for col in 0..=cols {
                let x = x_start + col as f64 * step;
                if x > x_end + EPS {
                    break;
                }
                centers.push((x, y));
            }
        }
    } else {
        for row in 0..=total_rows {
            let y = y_start + row as f64 * step;
            if y > y_end + EPS {
                break;
            }
            let mut sample_y = y.clamp(min_y, max_y);
            if height > 1e-9 {
                if (sample_y - max_y).abs() < 1e-9 {
                    sample_y = (max_y - 1e-9).max(min_y);
                } else if (sample_y - min_y).abs() < 1e-9 {
                    sample_y = (min_y + 1e-9).min(max_y);
                }
            }
            let mut intervals = scanline_intervals(&polygon, sample_y);
            if intervals.is_empty() {
                intervals.push((min_x, max_x));
            }
            for (mut left, mut right) in intervals {
                if left > right {
                    let tmp = left;
                    left = right;
                    right = tmp;
                }
                let start = left - pad;
                let end = right + pad;
                if end + EPS < start {
                    continue;
                }
                let cols = ((end - start) / step).ceil().max(0.0) as usize + 2;
                for col in 0..=cols {
                    let x = start + col as f64 * step;
                    if x > end + EPS {
                        break;
                    }
                    centers.push((x, y));
                }
            }
        }
    }

    if centers.is_empty() {
        let (sum_x, sum_y) = polygon
            .iter()
            .fold((0.0, 0.0), |acc, &(x, y)| (acc.0 + x, acc.1 + y));
        centers.push((sum_x / nv as f64, sum_y / nv as f64));
    }

    let d_unit = normalize(sun);
    let radius = scale * SQRT_ONE_POINT_FIVE;
    let sep = radius * 2.6;
    let count = centers.len();
    let z_min = -0.707 * scale;
    let clearance = 0.05 * scale;
    let mut base_lift = 0.0;
    if d_unit.2 < -1e-3 {
        let needed = (clearance - z_min) / (-d_unit.2);
        let extra = if count > 0 {
            ((count - 1) as f64) * sep
        } else {
            0.0
        };
        base_lift = needed + extra;
    }

    let mut out = String::new();
    out.push_str(&format!("{}\n", count));
    for (idx, &(cx, cy)) in centers.iter().enumerate() {
        let factor = idx as f64 * sep - base_lift;
        let px = cx + d_unit.0 * factor;
        let py = cy + d_unit.1 * factor;
        let pz = d_unit.2 * factor;
        out.push_str(&format!(
            "{:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6}\n",
            px, py, pz, 1.0, 0.0, 0.0, 0.0, scale
        ));
    }
    print!("{out}");
}

fn choose_scale(max_dim: f64) -> f64 {
    if !max_dim.is_finite() {
        return 1.0;
    }
    let mut s = if max_dim < 1.0 {
        max_dim * 0.8 + 0.3
    } else if max_dim < 6.0 {
        max_dim / 1.05
    } else if max_dim < 20.0 {
        (max_dim / 1.02).min(10.0)
    } else {
        10.0
    };
    s = s.clamp(0.3, 10.0);
    s
}

fn compute_inradius(scale: f64, sun: (f64, f64, f64)) -> f64 {
    let mut pts = Vec::with_capacity(4);
    for v in BASE_VERTS {
        let p = (v[0] * scale, v[1] * scale, v[2] * scale);
        pts.push(project_point(p, sun));
    }
    let cx = pts.iter().map(|p| p.0).sum::<f64>() / pts.len() as f64;
    let cy = pts.iter().map(|p| p.1).sum::<f64>() / pts.len() as f64;

    let mut min_dist = f64::INFINITY;
    for i in 0..pts.len() {
        let ax = pts[i].0 - cx;
        let ay = pts[i].1 - cy;
        let bx = pts[(i + 1) % pts.len()].0 - cx;
        let by = pts[(i + 1) % pts.len()].1 - cy;
        let edge_x = bx - ax;
        let edge_y = by - ay;
        let len = (edge_x * edge_x + edge_y * edge_y).sqrt();
        if len < 1e-9 {
            continue;
        }
        let cross = edge_x * (-ay) - edge_y * (-ax);
        let dist = cross.abs() / len;
        if dist < min_dist {
            min_dist = dist;
        }
    }
    if !min_dist.is_finite() || min_dist <= 1e-6 {
        scale * 0.3
    } else {
        min_dist
    }
}

fn project_point(p: (f64, f64, f64), sun: (f64, f64, f64)) -> (f64, f64) {
    let dz = if sun.2.abs() < 1e-9 {
        if sun.2 >= 0.0 {
            -1e-9
        } else {
            sun.2
        }
    } else {
        sun.2
    };
    let t = -p.2 / dz;
    (p.0 + sun.0 * t, p.1 + sun.1 * t)
}

fn scanline_intervals(poly: &[(f64, f64)], y: f64) -> Vec<(f64, f64)> {
    let n = poly.len();
    let mut xs = Vec::new();
    if n < 3 {
        return xs;
    }
    for i in 0..n {
        let (x1, y1) = poly[i];
        let (x2, y2) = poly[(i + 1) % n];
        if (y1 - y2).abs() < 1e-12 {
            continue;
        }
        let (low, high) = if y1 < y2 { (y1, y2) } else { (y2, y1) };
        if y < low || y >= high {
            continue;
        }
        let t = (y - y1) / (y2 - y1);
        let x = x1 + t * (x2 - x1);
        if x.is_finite() {
            xs.push(x);
        }
    }
    xs.sort_by(|a, b| a.partial_cmp(b).unwrap());
    let mut intervals = Vec::with_capacity(xs.len() / 2);
    let mut i = 0;
    while i + 1 < xs.len() {
        let left = xs[i];
        let right = xs[i + 1];
        if right > left {
            intervals.push((left, right));
        }
        i += 2;
    }
    intervals
}

fn normalize(v: (f64, f64, f64)) -> (f64, f64, f64) {
    let len = (v.0 * v.0 + v.1 * v.1 + v.2 * v.2).sqrt();
    if len < 1e-9 {
        (0.0, 0.0, -1.0)
    } else {
        (v.0 / len, v.1 / len, v.2 / len)
    }
}
```
