inline bool stepAllowed(int from, int to, int dir) {
    if (to < 0 || to >= 64) return false;
    int fromFile = from & 7;
    int toFile = to & 7;
    switch (dir) {
        case 8: case -8:
            return toFile == fromFile;
        case 1: case 9: case -7:
            return toFile - fromFile == 1;
        case -1: case 7: case -9:
            return toFile - fromFile == -1;
        default:
            return false;
    }
}

void Board::parseFEN(const string &fen) {
    squares.fill(0);
    castlingRights = 0;
    enPassant = -1;
    halfmoveClock = 0;
    fullmoveNumber = 1;
    whiteToMove = true;
    whiteKingPos = -1;
    blackKingPos = -1;
    stateStack.clear();

    stringstream ss(fen);
    string boardPart, sidePart, castlePart, enPassantPart, halfmovePart, fullmovePart;
    ss >> boardPart >> sidePart >> castlePart >> enPassantPart >> halfmovePart >> fullmovePart;

    int rank = 7;
    int file = 0;
    for (char c : boardPart) {
        if (c == '/') {
            rank--;
            file = 0;
            continue;
        }
        if (isdigit(static_cast<unsigned char>(c))) {
            file += c - '0';
        } else {
            bool isWhite = isupper(static_cast<unsigned char>(c));
            int value = 0;
            switch (tolower(static_cast<unsigned char>(c))) {
                case 'p': value = 1; break;
                case 'n': value = 2; break;
                case 'b': value = 3; break;
                case 'r': value = 4; break;
                case 'q': value = 5; break;
                case 'k': value = 6; break;
                default: value = 0; break;
            }
            int piece = isWhite ? value : -value;
            int square = rank * 8 + file;
            squares[square] = static_cast<int8_t>(piece);
            if (piece == 6) whiteKingPos = square;
            if (piece == -6) blackKingPos = square;
            file++;
        }
    }

    whiteToMove = (sidePart == "w");

    if (castlePart.find('K') != string::npos) castlingRights |= 1;
    if (castlePart.find('Q') != string::npos) castlingRights |= 2;
    if (castlePart.find('k') != string::npos) castlingRights |= 4;
    if (castlePart.find('q') != string::npos) castlingRights |= 8;

    if (enPassantPart != "-" && enPassantPart.size() == 2) {
        int f = enPassantPart[0] - 'a';
        int r = enPassantPart[1] - '1';
        if (f >= 0 && f < 8 && r >= 0 && r < 8) {
            enPassant = r * 8 + f;
        } else {
            enPassant = -1;
        }
    } else {
        enPassant = -1;
    }

    auto toIntOrDefault = [](const string &s, int def) {
        if (s.empty()) return def;
        try {
            return stoi(s);
        } catch (...) {
            return def;
        }
    };

    halfmoveClock = toIntOrDefault(halfmovePart, 0);
    fullmoveNumber = toIntOrDefault(fullmovePart, 1);

    if (whiteKingPos == -1 || blackKingPos == -1) {
        for (int i = 0; i < 64; ++i) {
            if (squares[i] == 6) whiteKingPos = i;
            if (squares[i] == -6) blackKingPos = i;
        }
    }
}

void Board::generateMoves(MoveList &list, bool capturesOnly) const {
    list.clear();
    bool white = whiteToMove;
    auto addMove = [&](int from, int to, bool capture, bool ep, bool castle,
                       bool doublePush, int promo, int capturedType) {
        Move mv;
        mv.from = static_cast<uint8_t>(from);
        mv.to = static_cast<uint8_t>(to);
        mv.piece = squares[from];
        mv.isCapture = capture;
        mv.isEnPassant = ep;
        mv.isCastle = castle;
        mv.isDoublePush = doublePush;
        mv.promo = static_cast<uint8_t>(promo);
        mv.captured = static_cast<uint8_t>(capturedType);
        if (list.count < MAX_MOVES) {
            list.add(mv);
        }
    };

    for (int sq = 0; sq < 64; ++sq) {
        int piece = squares[sq];
        if (piece == 0) continue;
        if (white && piece < 0) continue;
        if (!white && piece > 0) continue;
        int absPiece = std::abs(piece);

        if (absPiece == 1) {
            if (white) {
                int forward = sq + 8;
                if (forward < 64 && squares[forward] == 0) {
                    bool promotion = forward >= 56;
                    if (!capturesOnly || promotion) {
                        if (promotion) {
                            int promos[4] = {5, 4, 3, 2};
                            for (int pr : promos) addMove(sq, forward, false, false, false, false, pr, 0);
                        } else {
                            addMove(sq, forward, false, false, false, false, 0, 0);
                        }
                    }
                    if (!capturesOnly && sq / 8 == 1) {
                        int dbl = sq + 16;
                        if (squares[dbl] == 0) {
                            addMove(sq, dbl, false, false, false, true, 0, 0);
                        }
                    }
                }
                if (sq % 8 != 0) {
                    int to = sq + 7;
                    if (to < 64) {
                        int target = squares[to];
                        bool capture = target < 0;
                        bool ep = (enPassant == to);
                        if ((capture || ep) && (!capturesOnly || capture || ep)) {
                            bool promotion = to >= 56;
                            int capturedType = capture ? std::abs(target) : 1;
                            if (promotion) {
                                int promos[4] = {5, 4, 3, 2};
                                for (int pr : promos) addMove(sq, to, true, ep, false, false, pr, capturedType);
                            } else {
                                addMove(sq, to, true, ep, false, false, 0, capturedType);
                            }
                        }
                    }
                }
                if (sq % 8 != 7) {
                    int to = sq + 9;
                    if (to < 64) {
                        int target = squares[to];
                        bool capture = target < 0;
                        bool ep = (enPassant == to);
                        if ((capture || ep) && (!capturesOnly || capture || ep)) {
                            bool promotion = to >= 56;
                            int capturedType = capture ? std::abs(target) : 1;
                            if (promotion) {
                                int promos[4] = {5, 4, 3, 2};
                                for (int pr : promos) addMove(sq, to, true, ep, false, false, pr, capturedType);
                            } else {
                                addMove(sq, to, true, ep, false, false, 0, capturedType);
                            }
                        }
                    }
                }
            } else {
                int forward = sq - 8;
                if (forward >= 0 && squares[forward] == 0) {
                    bool promotion = forward < 8;
                    if (!capturesOnly || promotion) {
                        if (promotion) {
                            int promos[4] = {5, 4, 3, 2};
                            for (int pr : promos) addMove(sq, forward, false, false, false, false, pr, 0);
                        } else {
                            addMove(sq, forward, false, false, false, false, 0, 0);
                        }
                    }
                    if (!capturesOnly && sq / 8 == 6) {
                        int dbl = sq - 16;
                        if (squares[dbl] == 0) {
                            addMove(sq, dbl, false, false, false, true, 0, 0);
                        }
                    }
                }
                if (sq % 8 != 7) {
                    int to = sq - 7;
                    if (to >= 0) {
                        int target = squares[to];
                        bool capture = target > 0;
                        bool ep = (enPassant == to);
                        if ((capture || ep) && (!capturesOnly || capture || ep)) {
                            bool promotion = to < 8;
                            int capturedType = capture ? std::abs(target) : 1;
                            if (promotion) {
                                int promos[4] = {5, 4, 3, 2};
                                for (int pr : promos) addMove(sq, to, true, ep, false, false, pr, capturedType);
                            } else {
                                addMove(sq, to, true, ep, false, false, 0, capturedType);
                            }
                        }
                    }
                }
                if (sq % 8 != 0) {
                    int to = sq - 9;
                    if (to >= 0) {
                        int target = squares[to];
                        bool capture = target > 0;
                        bool ep = (enPassant == to);
                        if ((capture || ep) && (!capturesOnly || capture || ep)) {
                            bool promotion = to < 8;
                            int capturedType = capture ? std::abs(target) : 1;
                            if (promotion) {
                                int promos[4] = {5, 4, 3, 2};
                                for (int pr : promos) addMove(sq, to, true, ep, false, false, pr, capturedType);
                            } else {
                                addMove(sq, to, true, ep, false, false, 0, capturedType);
                            }
                        }
                    }
                }
            }
            continue;
        }

        if (absPiece == 2) {
            for (int dest : knightMoves[sq]) {
                int target = squares[dest];
                if (target == 0) {
                    if (!capturesOnly) addMove(sq, dest, false, false, false, false, 0, 0);
                } else if ((white && target < 0) || (!white && target > 0)) {
                    addMove(sq, dest, true, false, false, false, 0, std::abs(target));
                }
            }
            continue;
        }

        auto handleSliding = [&](const vector<int> &dirs) {
            for (int dir : dirs) {
                int cur = sq;
                while (true) {
                    int next = cur + dir;
                    if (!stepAllowed(cur, next, dir)) break;
                    cur = next;
                    int target = squares[cur];
                    if (target == 0) {
                        if (!capturesOnly) addMove(sq, cur, false, false, false, false, 0, 0);
                    } else {
                        if ((white && target < 0) || (!white && target > 0)) {
                            addMove(sq, cur, true, false, false, false, 0, std::abs(target));
                        }
                        break;
                    }
                }
            }
        };

        if (absPiece == 3) {
            static const vector<int> dirs = {9, 7, -7, -9};
            handleSliding(dirs);
            continue;
        }

        if (absPiece == 4) {
            static const vector<int> dirs = {8, -8, 1, -1};
            handleSliding(dirs);
            continue;
        }

        if (absPiece == 5) {
            static const vector<int> dirsB = {9, 7, -7, -9};
            static const vector<int> dirsR = {8, -8, 1, -1};
            handleSliding(dirsB);
            handleSliding(dirsR);
            continue;
        }

        if (absPiece == 6) {
            for (int dest : kingMoves[sq]) {
                int target = squares[dest];
                if (target == 0) {
                    if (!capturesOnly) addMove(sq, dest, false, false, false, false, 0, 0);
                } else if ((white && target < 0) || (!white && target > 0)) {
                    addMove(sq, dest, true, false, false, false, 0, std::abs(target));
                }
            }
            if (!capturesOnly) {
                if (white && sq == 4) {
                    if ((castlingRights & 1) && squares[5] == 0 && squares[6] == 0 &&
                        !isSquareAttacked(4, false) && !isSquareAttacked(5, false) && !isSquareAttacked(6, false)) {
                        addMove(4, 6, false, false, true, false, 0, 0);
                    }
                    if ((castlingRights & 2) && squares[3] == 0 && squares[2] == 0 && squares[1] == 0 &&
                        !isSquareAttacked(4, false) && !isSquareAttacked(3, false) && !isSquareAttacked(2, false)) {
                        addMove(4, 2, false, false, true, false, 0, 0);
                    }
                } else if (!white && sq == 60) {
                    if ((castlingRights & 4) && squares[61] == 0 && squares[62] == 0 &&
                        !isSquareAttacked(60, true) && !isSquareAttacked(61, true) && !isSquareAttacked(62, true)) {
                        addMove(60, 62, false, false, true, false, 0, 0);
                    }
                    if ((castlingRights & 8) && squares[59] == 0 && squares[58] == 0 && squares[57] == 0 &&
                        !isSquareAttacked(60, true) && !isSquareAttacked(59, true) && !isSquareAttacked(58, true)) {
                        addMove(60, 58, false, false, true, false, 0, 0);
                    }
                }
            }
        }
    }
}

bool Board::isSquareAttacked(int sq, bool byWhite) const {
    if (byWhite) {
        if (sq % 8 != 0) {
            int p = sq - 9;
            if (p >= 0 && squares[p] == 1) return true;
        }
        if (sq % 8 != 7) {
            int p = sq - 7;
            if (p >= 0 && squares[p] == 1) return true;
        }
    } else {
        if (sq % 8 != 7) {
            int p = sq + 9;
            if (p < 64 && squares[p] == -1) return true;
        }
        if (sq % 8 != 0) {
            int p = sq + 7;
            if (p < 64 && squares[p] == -1) return true;
        }
    }

    for (int pos : knightMoves[sq]) {
        int piece = squares[pos];
        if (piece == 0) continue;
        if (byWhite) {
            if (piece == 2) return true;
        } else {
            if (piece == -2) return true;
        }
    }

    static const vector<int> bishopDirs = {9, 7, -7, -9};
    static const vector<int> rookDirs = {8, -8, 1, -1};

    auto checkSlider = [&](const vector<int> &dirs, int pieceType) {
        for (int dir : dirs) {
            int cur = sq;
            while (true) {
                int next = cur + dir;
                if (!stepAllowed(cur, next, dir)) break;
                cur = next;
                int target = squares[cur];
                if (target == 0) continue;
                if (byWhite) {
                    if (target > 0) {
                        int absPiece = std::abs(target);
                        if (absPiece == pieceType || absPiece == 5) return true;
                    }
                } else {
                    if (target < 0) {
                        int absPiece = std::abs(target);
                        if (absPiece == pieceType || absPiece == 5) return true;
                    }
                }
                break;
            }
        }
        return false;
    };

    if (checkSlider(bishopDirs, 3)) return true;
    if (checkSlider(rookDirs, 4)) return true;

    for (int pos : kingMoves[sq]) {
        int piece = squares[pos];
        if (byWhite) {
            if (piece == 6) return true;
        } else {
            if (piece == -6) return true;
        }
    }
    return false;
}

bool Board::inCheck(bool white) const {
    int kingSq = white ? whiteKingPos : blackKingPos;
    if (kingSq < 0) return false;
    return isSquareAttacked(kingSq, !white);
}

bool Board::makeMove(const Move &mv) {
    bool white = whiteToMove;
    int from = mv.from;
    int to = mv.to;
    int piece = squares[from];
    if (piece == 0) return false;
    if (white && piece < 0) return false;
    if (!white && piece > 0) return false;

    int captured = 0;
    int capSquare = to;
    if (mv.isEnPassant) {
        capSquare = white ? to - 8 : to + 8;
        if (capSquare < 0 || capSquare >= 64) return false;
        captured = squares[capSquare];
        if (captured == 0) return false;
    } else {
        captured = squares[to];
    }

    GameState st;
    st.castlingRights = castlingRights;
    st.enPassant = enPassant;
    st.halfmoveClock = halfmoveClock;
    st.fullmoveNumber = fullmoveNumber;
    st.whiteKingPos = whiteKingPos;
    st.blackKingPos = blackKingPos;
    st.move = mv;
    st.capturedPiece = captured;
    stateStack.push_back(st);

    squares[from] = 0;
    if (mv.isEnPassant) {
        squares[capSquare] = 0;
    }

    if (mv.isCastle) {
        if (white) {
            if (to == 6) {
                squares[5] = squares[7];
                squares[7] = 0;
            } else if (to == 2) {
                squares[3] = squares[0];
                squares[0] = 0;
            }
        } else {
            if (to == 62) {
                squares[61] = squares[63];
                squares[63] = 0;
            } else if (to == 58) {
                squares[59] = squares[56];
                squares[56] = 0;
            }
        }
    }

    int placedPiece = mv.piece;
    if (mv.promo) {
        placedPiece = (mv.piece > 0) ? mv.promo : -mv.promo;
    }
    squares[to] = static_cast<int8_t>(placedPiece);

    if (std::abs(mv.piece) == 6) {
        if (white) whiteKingPos = to;
        else blackKingPos = to;
    }

    castlingRights &= ~castlingMask[from];
    castlingRights &= ~castlingMask[to];

    enPassant = -1;
    if (mv.isDoublePush) {
        enPassant = white ? (to - 8) : (to + 8);
    }

    if (std::abs(mv.piece) == 1 || captured != 0) {
        halfmoveClock = 0;
    } else {
        halfmoveClock++;
    }

    if (!white) {
        fullmoveNumber++;
    }

    whiteToMove = !white;

    int kingSq = white ? whiteKingPos : blackKingPos;
    if (kingSq >= 0 && isSquareAttacked(kingSq, whiteToMove)) {
        undoMove();
        return false;
    }

    return true;
}

void Board::undoMove() {
    if (stateStack.empty()) return;
    GameState st = stateStack.back();
    stateStack.pop_back();
    Move mv = st.move;

    whiteToMove = !whiteToMove;
    castlingRights = st.castlingRights;
    enPassant = st.enPassant;
    halfmoveClock = st.halfmoveClock;
    fullmoveNumber = st.fullmoveNumber;
    whiteKingPos = st.whiteKingPos;
    blackKingPos = st.blackKingPos;

    int from = mv.from;
    int to = mv.to;

    squares[from] = mv.piece;

    if (mv.isCastle) {
        if (to == 6) {
            squares[7] = squares[5];
            squares[5] = 0;
        } else if (to == 2) {
            squares[0] = squares[3];
            squares[3] = 0;
        } else if (to == 62) {
            squares[63] = squares[61];
            squares[61] = 0;
        } else if (to == 58) {
            squares[56] = squares[59];
            squares[59] = 0;
        }
    }

    if (mv.isEnPassant) {
        squares[to] = 0;
        int capSq = whiteToMove ? to - 8 : to + 8;
        squares[capSq] = static_cast<int8_t>(st.capturedPiece);
    } else {
        squares[to] = static_cast<int8_t>(st.capturedPiece);
    }
}

int evaluateBoard(const Board &board) {
    int score = 0;
    for (int sq = 0; sq < 64; ++sq) {
        int piece = board.squares[sq];
        if (piece == 0) continue;
        int absPiece = std::abs(piece);
        int val = pieceValues[absPiece];
        const auto &table = *PIECE_TABLES[absPiece];
        if (piece > 0) {
            score += val + table[sq];
        } else {
            score -= val + table[mirrorSquare(sq)];
        }
    }
    return board.whiteToMove ? score : -score;
}

class SearchEngine {
public:
    Move searchBestMove(Board &board, double timeLimitSeconds);

private:
    bool stop = false;
    double timeLimit = 1.0;
    chrono::steady_clock::time_point start;
    long long nodes = 0;
    Move killerMoves[MAX_PLY][2];
    bool killerUsed[MAX_PLY][2]{};
    int history[2][64][64]{};

    int negamax(Board &board, int depth, int alpha, int beta, int ply);
    int quiescence(Board &board, int alpha, int beta, int ply);
    void orderMoves(MoveList &list, int ply);
    int scoreMove(const Move &mv, int ply) const;
    void addKiller(const Move &mv, int ply);
    void updateHistory(const Move &mv, int depth);
    bool timeExceeded() const;
};

bool SearchEngine::timeExceeded() const {
    double elapsed = chrono::duration<double>(chrono::steady_clock::now() - start).count();
    return elapsed >= timeLimit;
}

void SearchEngine::orderMoves(MoveList &list, int ply) {
    auto begin = list.moves.begin();
    auto end = begin + list.count;
    std::sort(begin, end, [&](const Move &a, const Move &b) {
        return scoreMove(a, ply) > scoreMove(b, ply);
    });
}

int SearchEngine::scoreMove(const Move &mv, int ply) const {
    if (mv.isCapture) {
        int capturedVal = pieceValues[mv.captured];
        int attackerVal = pieceValues[std::abs(mv.piece)];
        return 1'000'000 + capturedVal * 10 - attackerVal + (mv.promo ? 100 : 0);
    }
    if (mv.promo) {
        return 900'000 + pieceValues[mv.promo];
    }
    if (ply < MAX_PLY) {
        for (int i = 0; i < 2; ++i) {
            if (killerUsed[ply][i] && killerMoves[ply][i] == mv) {
                return 800'000 - i;
            }
        }
    }
    int colorIndex = mv.piece > 0 ? 0 : 1;
    return history[colorIndex][mv.from][mv.to];
}

void SearchEngine::addKiller(const Move &mv, int ply) {
    if (ply >= MAX_PLY) return;
    if (killerUsed[ply][0] && killerMoves[ply][0] == mv) return;
    killerMoves[ply][1] = killerMoves[ply][0];
    killerUsed[ply][1] = killerUsed[ply][0];
    killerMoves[ply][0] = mv;
    killerUsed[ply][0] = true;
}

void SearchEngine::updateHistory(const Move &mv, int depth) {
    int colorIndex = mv.piece > 0 ? 0 : 1;
    history[colorIndex][mv.from][mv.to] += depth * depth;
    if (history[colorIndex][mv.from][mv.to] > 1'000'000) {
        history[colorIndex][mv.from][mv.to] = 1'000'000;
    }
}

int SearchEngine::quiescence(Board &board, int alpha, int beta, int ply) {
    if (stop) return 0;
    nodes++;
    if ((nodes & 2047) == 0 && timeExceeded()) {
        stop = true;
        return 0;
    }

    int standPat = evaluateBoard(board);
    if (standPat >= beta) return beta;
    if (standPat > alpha) alpha = standPat;
    if (ply >= MAX_PLY - 1) return standPat;

    MoveList moves;
    board.generateMoves(moves, true);
    orderMoves(moves, ply);

    for (int i = 0; i < moves.count; ++i) {
        const Move &mv = moves.moves[i];
        if (!board.makeMove(mv)) continue;
        int score = -quiescence(board, -beta, -alpha, ply + 1);
        board.undoMove();

        if (stop) return 0;
        if (score >= beta) return beta;
        if (score > alpha) alpha = score;
    }
    return alpha;
}

int SearchEngine::negamax(Board &board, int depth, int alpha, int beta, int ply) {
    if (stop) return 0;
    nodes++;
    if ((nodes & 2047) == 0 && timeExceeded()) {
        stop = true;
        return 0;
    }

    bool inCheck = board.inCheck(board.whiteToMove);
    if (depth <= 0) {
        return quiescence(board, alpha, beta, ply);
    }
    if (ply >= MAX_PLY - 1) {
        return evaluateBoard(board);
    }

    MoveList moves;
    board.generateMoves(moves);
    orderMoves(moves, ply);

    int originalAlpha = alpha;
    bool anyLegal = false;
    Move bestMove;

    for (int i = 0; i < moves.count; ++i) {
        const Move &mv = moves.moves[i];
        if (!board.makeMove(mv)) continue;
        anyLegal = true;
        int score = -negamax(board, depth - 1, -beta, -alpha, ply + 1);
        board.undoMove();

        if (stop) return 0;

        if (score >= beta) {
            if (!mv.isCapture && mv.promo == 0) {
                addKiller(mv, ply);
                updateHistory(mv, depth);
            }
            return beta;
        }
        if (score > alpha) {
            alpha = score;
            bestMove = mv;
        }
    }

    if (!anyLegal) {
        if (inCheck) {
            return -MATE_VALUE + ply;
        }
        return 0;
    }

    if (alpha > originalAlpha && !bestMove.isCapture && bestMove.promo == 0) {
        updateHistory(bestMove, depth);
    }

    return alpha;
}

Move SearchEngine::searchBestMove(Board &board, double timeLimitSeconds) {
    timeLimit = max(0.05, timeLimitSeconds * 0.9);
    start = chrono::steady_clock::now();
    stop = false;
    nodes = 0;
    memset(history, 0, sizeof(history));
    memset(killerUsed, 0, sizeof(killerUsed));

    Move best;
    bool hasBest = false;
    int bestScore = -INF;

    for (int depth = 1; depth <= 64; ++depth) {
        MoveList moves;
        board.generateMoves(moves);
        orderMoves(moves, 0);

        int alpha = -INF;
        Move iterationBest;
        bool found = false;

        for (int i = 0; i < moves.count; ++i) {
            const Move &mv = moves.moves[i];
            if (!board.makeMove(mv)) continue;
            int score = -negamax(board, depth - 1, -INF, INF, 1);
            board.undoMove();

            if (stop) break;

            if (!found || score > alpha) {
                alpha = score;
                iterationBest = mv;
                found = true;
            }
        }

        if (found) {
            best = iterationBest;
            hasBest = true;
            bestScore = alpha;
        }

        if (stop) break;

        if (hasBest && (bestScore > MATE_VALUE - MAX_PLY || bestScore < -MATE_VALUE + MAX_PLY)) {
            break;
        }
    }

    if (!hasBest) {
        MoveList fallback;
        board.generateMoves(fallback);
        for (int i = 0; i < fallback.count; ++i) {
            const Move &mv = fallback.moves[i];
            if (board.makeMove(mv)) {
                board.undoMove();
                best = mv;
                hasBest = true;
                break;
            }
        }
    }

    return best;
}

string squareToString(int sq) {
    string s;
    s += static_cast<char>('a' + (sq % 8));
    s += static_cast<char>('1' + (sq / 8));
    return s;
}

char pieceToChar(int type) {
    switch (type) {
        case 1: return 'p';
        case 2: return 'N';
        case 3: return 'B';
        case 4: return 'R';
        case 5: return 'Q';
        case 6: return 'K';
        default: return '?';
    }
}

string formatMove(const Board &board, const Move &mv) {
    if (!mv.isValid()) return "resign";
    if (mv.isCastle) {
        if (mv.to == 6 || mv.to == 62) return "O-O";
        return "O-O-O";
    }
    string res;
    res += pieceToChar(std::abs(mv.piece));
    res += squareToString(mv.from);
    if (mv.isCapture) res += 'x';
    res += squareToString(mv.to);
    if (mv.isEnPassant) res += "ep";
    if (mv.promo) {
        res += '/';
        res += pieceToChar(mv.promo);
    }

    Board copy = board;
    if (copy.makeMove(mv)) {
        bool givesCheck = copy.inCheck(copy.whiteToMove);
        if (givesCheck) {
            MoveList replies;
            copy.generateMoves(replies);
            bool hasReply = false;
            for (int i = 0; i < replies.count; ++i) {
                if (copy.makeMove(replies.moves[i])) {
                    hasReply = true;
                    copy.undoMove();
                    break;
                }
            }
            if (!hasReply) res += '#';
            else res += '+';
        }
    }
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    initMoveTables();

    string timeoutLine;
    if (!getline(cin, timeoutLine)) return 0;
    string fen;
    while (fen.empty() && getline(cin, fen)) {
        if (!fen.empty()) break;
    }
    if (fen.empty()) return 0;

    double timeoutSeconds = 5.0;
    try {
        timeoutSeconds = stod(timeoutLine);
    } catch (...) {
        timeoutSeconds = 5.0;
    }

    Board board;
    board.parseFEN(fen);

    SearchEngine engine;
    Move best = engine.searchBestMove(board, timeoutSeconds);
    string notation = formatMove(board, best);
    cout << notation << '\n';
    return 0;
}
