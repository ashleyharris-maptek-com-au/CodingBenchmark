; SPIR-V
; Version: 1.0
; Generator: 0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inUV %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %inUV Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%functype = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%bool = OpTypeBool
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%float_3 = OpConstant %float 3
%float_4 = OpConstant %float 4
%float_0_5 = OpConstant %float 0.5
%float_20 = OpConstant %float 20
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_20 = OpConstant %int 20
%v2scale = OpConstantComposite %v2float %float_3 %float_2
%v2offset = OpConstantComposite %v2float %float_2 %float_1
%v2zero = OpConstantComposite %v2float %float_0 %float_0

%inUV = OpVariable %_ptr_Input_v2float Input
%outColor = OpVariable %_ptr_Output_v4float Output

%main = OpFunction %void None %functype
%entry = OpLabel
%uv = OpLoad %v2float %inUV
%scaled = OpFMul %v2float %uv %v2scale
%c = OpFSub %v2float %scaled %v2offset
OpBranch %loop

%loop = OpLabel
%iter = OpPhi %int %int_0 %entry %iterPlus %continue
%z = OpPhi %v2float %v2zero %entry %zNew %continue
%condMax = OpSGreaterThanEqual %bool %iter %int_20
OpLoopMerge %merge %continue None
OpBranchConditional %condMax %merge %body

%body = OpLabel
%zx = OpCompositeExtract %float %z 0
%zy = OpCompositeExtract %float %z 1
%zx2 = OpFMul %float %zx %zx
%zy2 = OpFMul %float %zy %zy
%realTemp = OpFSub %float %zx2 %zy2
%cx = OpCompositeExtract %float %c 0
%real = OpFAdd %float %realTemp %cx
%mul = OpFMul %float %zx %zy
%twoMul = OpFMul %float %mul %float_2
%cy = OpCompositeExtract %float %c 1
%imag = OpFAdd %float %twoMul %cy
%zNew = OpCompositeConstruct %v2float %real %imag
%realSq = OpFMul %float %real %real
%imagSq = OpFMul %float %imag %imag
%mag2 = OpFAdd %float %realSq %imagSq
%iterPlus = OpIAdd %int %iter %int_1
%condEscape = OpFOrdGreaterThan %bool %mag2 %float_4
OpBranchConditional %condEscape %merge %continue

%continue = OpLabel
OpBranch %loop

%merge = OpLabel
%finalIter = OpPhi %int %iter %loop %iterPlus %body
%iterFloat = OpConvertSToF %float %finalIter
%t = OpFDiv %float %iterFloat %float_20
%halfT = OpFMul %float %t %float_0_5
%invT = OpFSub %float %float_1 %t
%color = OpCompositeConstruct %v4float %t %halfT %invT %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
