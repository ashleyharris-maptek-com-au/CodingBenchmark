```glsl
#version 450

layout(local_size_x = 256) in;

// in/out buffers
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;     // counts.x = N
    vec4 radii;       // radii.x = dt, radii.y = sepRadius, radii.z = alignRadius, radii.w = cohRadius
    vec4 weights;     // weights.x = sepWeight, weights.y = alignWeight, weights.z = cohWeight, weights.w = maxSpeed
    vec4 bounds;      // bounds.x = boundSize, bounds.y = boundForce
};

void main() {
    uint gid = gl_GlobalInvocationID.x;
    uint boidCount = counts.x;
    if (gid >= boidCount) {
        return;
    }

    uint baseIndex = gid * 2u;
    vec3 position = inData[baseIndex].xyz;
    vec3 velocity = inData[baseIndex + 1u].xyz;

    float dt         = radii.x;
    float sepRadius  = radii.y;
    float alignRadius = radii.z;
    float cohRadius  = radii.w;

    float sepWeight   = weights.x;
    float alignWeight = weights.y;
    float cohWeight   = weights.z;
    float maxSpeed    = weights.w;

    float boundSize  = bounds.x;
    float boundForce = bounds.y;

    float sepRadius2   = sepRadius * sepRadius;
    float alignRadius2 = alignRadius * alignRadius;
    float cohRadius2   = cohRadius * cohRadius;

    vec3 separation = vec3(0.0);
    vec3 alignment  = vec3(0.0);
    vec3 cohesion   = vec3(0.0);

    float sepCount = 0.0;
    float alignCount = 0.0;
    float cohCount = 0.0;

    for (uint j = 0u; j < boidCount; ++j) {
        if (j == gid) {
            continue;
        }

        uint otherIdx = j * 2u;
        vec3 otherPos = inData[otherIdx].xyz;
        vec3 diff = otherPos - position;
        float dist2 = dot(diff, diff);
        if (dist2 <= 0.0) {
            continue;
        }

        if (dist2 < sepRadius2) {
            float dist = sqrt(dist2);
            vec3 dir = (position - otherPos) / dist; // push away
            separation += dir;
            sepCount += 1.0;
        }

        if (dist2 < alignRadius2) {
            vec3 otherVel = inData[otherIdx + 1u].xyz;
            alignment += otherVel;
            alignCount += 1.0;
        }

        if (dist2 < cohRadius2) {
            cohesion += otherPos;
            cohCount += 1.0;
        }
    }

    vec3 accel = vec3(0.0);

    if (sepCount > 0.0) {
        vec3 sepAvg = separation / sepCount;
        accel += sepAvg * sepWeight;
    }

    if (alignCount > 0.0) {
        vec3 alignAvg = alignment / alignCount;
        vec3 alignForce = alignAvg - velocity;
        accel += alignForce * alignWeight;
    }

    if (cohCount > 0.0) {
        vec3 centerPos = cohesion / cohCount;
        vec3 cohForce = centerPos - position;
        accel += cohForce * cohWeight;
    }

    // Boundary force
    float boundThreshold = boundSize * 0.4;
    float boundRange = boundSize * 0.5;

    for (int axis = 0; axis < 3; ++axis) {
        float coord = position[axis];
        if (coord > boundThreshold) {
            float t = (coord - boundThreshold) / boundRange;
            accel[axis] -= boundForce * clamp(t, 0.0, 1.0);
        } else if (coord < -boundThreshold) {
            float t = (-boundThreshold - coord) / boundRange;
            accel[axis] += boundForce * clamp(t, 0.0, 1.0);
        }
    }

    velocity += accel * dt;

    float speed = length(velocity);
    if (speed > maxSpeed && speed > 0.0) {
        velocity = (velocity / speed) * maxSpeed;
    }

    position += velocity * dt;

    outData[baseIndex] = vec4(position, 0.0);
    outData[baseIndex + 1u] = vec4(velocity, 0.0);
}
```
