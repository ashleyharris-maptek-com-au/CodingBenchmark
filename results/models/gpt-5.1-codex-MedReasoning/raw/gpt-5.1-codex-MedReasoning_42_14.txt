OpCapability Geometry
OpMemoryModel Logical GLSL450
%glsl = OpExtInstImport "GLSL.std.450"
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_0_33333334 = OpConstant %float 0.33333334
%vec3_one = OpConstantComposite %v3float %float_1 %float_1 %float_1
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_input_v4float = OpTypePointer Input %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_output_glpv = OpTypePointer Output %gl_PerVertex_out
%ptr_output_v4float = OpTypePointer Output %v4float
%arr_v3float_6 = OpTypeArray %v3float %uint_6
%ptr_input_arr_v3 = OpTypePointer Input %arr_v3float_6
%ptr_input_v3float = OpTypePointer Input %v3float
%ptr_output_v3float = OpTypePointer Output %v3float
%gl_in = OpVariable %ptr_input_arr_pv Input
%worldPos = OpVariable %ptr_input_arr_v3 Input
%normal = OpVariable %ptr_input_arr_v3 Input
%color = OpVariable %ptr_input_arr_v3 Input
%gl_out = OpVariable %ptr_output_glpv Output
%outColor = OpVariable %ptr_output_v3float Output
%main = OpFunction %void None %func
%label = OpLabel
%p0_ptr = OpAccessChain %ptr_input_v3float %worldPos %int_0
%p0 = OpLoad %v3float %p0_ptr
%p2_ptr = OpAccessChain %ptr_input_v3float %worldPos %int_2
%p2 = OpLoad %v3float %p2_ptr
%p4_ptr = OpAccessChain %ptr_input_v3float %worldPos %int_4
%p4 = OpLoad %v3float %p4_ptr
%edge1 = OpFSub %v3float %p2 %p0
%edge2 = OpFSub %v3float %p4 %p0
%edge1_x = OpCompositeExtract %float %edge1 0
%edge1_y = OpCompositeExtract %float %edge1 1
%edge1_z = OpCompositeExtract %float %edge1 2
%edge2_x = OpCompositeExtract %float %edge2 0
%edge2_y = OpCompositeExtract %float %edge2 1
%edge2_z = OpCompositeExtract %float %edge2 2
%temp_x1 = OpFMul %float %edge1_y %edge2_z
%temp_x2 = OpFMul %float %edge1_z %edge2_y
%cross_x = OpFSub %float %temp_x1 %temp_x2
%temp_y1 = OpFMul %float %edge1_z %edge2_x
%temp_y2 = OpFMul %float %edge1_x %edge2_z
%cross_y = OpFSub %float %temp_y1 %temp_y2
%temp_z1 = OpFMul %float %edge1_x %edge2_y
%temp_z2 = OpFMul %float %edge1_y %edge2_x
%cross_z = OpFSub %float %temp_z1 %temp_z2
%cross = OpCompositeConstruct %v3float %cross_x %cross_y %cross_z
%faceNorm = OpExtInst %v3float %glsl Normalize %cross
%lightNorm = OpExtInst %v3float %glsl Normalize %vec3_one
%ndotl = OpDot %float %faceNorm %lightNorm
%diff = OpExtInst %float %glsl FMax %ndotl %float_0
%col0_ptr = OpAccessChain %ptr_input_v3float %color %int_0
%col0 = OpLoad %v3float %col0_ptr
%col2_ptr = OpAccessChain %ptr_input_v3float %color %int_2
%col2 = OpLoad %v3float %col2_ptr
%col4_ptr = OpAccessChain %ptr_input_v3float %color %int_4
%col4 = OpLoad %v3float %col4_ptr
%col01 = OpFAdd %v3float %col0 %col2
%colSum = OpFAdd %v3float %col01 %col4
%baseColor = OpVectorTimesScalar %v3float %colSum %float_0_33333334
%scaledDiffuse = OpFMul %float %float_0_8 %diff
%factor = OpFAdd %float %float_0_2 %scaledDiffuse
%finalColor = OpVectorTimesScalar %v3float %baseColor %factor
%out_pos_ptr = OpAccessChain %ptr_output_v4float %gl_out %int_0
%in_pos0_ptr = OpAccessChain %ptr_input_v4float %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %in_pos0_ptr
OpStore %out_pos_ptr %pos0
OpStore %outColor %finalColor
OpEmitVertex
%in_pos2_ptr = OpAccessChain %ptr_input_v4float %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %in_pos2_ptr
OpStore %out_pos_ptr %pos2
OpStore %outColor %finalColor
OpEmitVertex
%in_pos4_ptr = OpAccessChain %ptr_input_v4float %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %in_pos4_ptr
OpStore %out_pos_ptr %pos4
OpStore %outColor %finalColor
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
