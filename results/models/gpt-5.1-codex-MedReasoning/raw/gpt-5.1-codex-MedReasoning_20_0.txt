use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read};
use std::str::FromStr;

const BRUTE_FORCE_LIMIT: usize = 8_000;

fn main() {
    let mut scanner = FastScanner::new();
    let num_holes = match scanner.next_parse::<usize>() {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let num_properties = match scanner.next_parse::<usize>() {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut property_names = Vec::with_capacity(num_properties);
    for idx in 0..num_properties {
        let name = scanner
            .next_owned()
            .unwrap_or_else(|| format!("property_{idx}"));
        property_names.push(name);
    }

    let mut property_values = vec![Vec::<f64>::new(); num_properties];
    let mut global_stats = vec![RunningStat::default(); num_properties];
    let mut hole_names = Vec::<String>::with_capacity(num_holes);
    let mut sample_hole_idx = Vec::<usize>::new();
    let mut sample_idx_in_hole = Vec::<u32>::new();
    let mut positions_x = Vec::<f64>::new();
    let mut positions_y = Vec::<f64>::new();
    let mut positions_z = Vec::<f64>::new();

    let mut min_x = f64::INFINITY;
    let mut min_y = f64::INFINITY;
    let mut min_z = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    let mut max_z = f64::NEG_INFINITY;

    let mut interval_sum = 0.0;
    let mut interval_cnt = 0usize;
    let mut length_interval_sum = 0.0;
    let mut length_interval_cnt = 0usize;

    for _ in 0..num_holes {
        let hole_id = match scanner.next_owned() {
            Some(v) => v,
            None => break,
        };
        hole_names.push(hole_id);
        let hole_index = hole_names.len() - 1;

        let start_x = sanitize(scanner.next_parse::<f64>().unwrap_or(0.0));
        let start_y = sanitize(scanner.next_parse::<f64>().unwrap_or(0.0));
        let start_z = sanitize(scanner.next_parse::<f64>().unwrap_or(0.0));
        let dir_x = sanitize(scanner.next_parse::<f64>().unwrap_or(0.0));
        let dir_y = sanitize(scanner.next_parse::<f64>().unwrap_or(0.0));
        let dir_z = sanitize(scanner.next_parse::<f64>().unwrap_or(-1.0));
        let length = sanitize(scanner.next_parse::<f64>().unwrap_or(0.0));
        let num_samples = scanner.next_parse::<usize>().unwrap_or(0);

        positions_x.reserve(num_samples);
        positions_y.reserve(num_samples);
        positions_z.reserve(num_samples);
        sample_hole_idx.reserve(num_samples);
        sample_idx_in_hole.reserve(num_samples);
        for column in property_values.iter_mut() {
            column.reserve(num_samples);
        }

        let dir_unit = normalize_direction(dir_x, dir_y, dir_z);
        let mut prev_depth_valid: Option<f64> = None;

        for sample_local_idx in 0..num_samples {
            let depth_measure = parse_measurement(scanner.next_str());
            let (depth, is_valid_depth) = if depth_measure.is_finite() {
                (depth_measure, true)
            } else {
                (prev_depth_valid.unwrap_or(0.0), false)
            };

            if is_valid_depth {
                if let Some(prev) = prev_depth_valid {
                    interval_sum += (depth_measure - prev).abs();
                    interval_cnt += 1;
                }
                prev_depth_valid = Some(depth_measure);
            }

            let px = start_x + dir_unit[0] * depth;
            let py = start_y + dir_unit[1] * depth;
            let pz = start_z + dir_unit[2] * depth;

            if px.is_finite() {
                min_x = min_x.min(px);
                max_x = max_x.max(px);
            }
            if py.is_finite() {
                min_y = min_y.min(py);
                max_y = max_y.max(py);
            }
            if pz.is_finite() {
                min_z = min_z.min(pz);
                max_z = max_z.max(pz);
            }

            positions_x.push(px);
            positions_y.push(py);
            positions_z.push(pz);
            sample_hole_idx.push(hole_index);
            sample_idx_in_hole.push(sample_local_idx as u32);

            for prop_idx in 0..num_properties {
                let value = parse_measurement(scanner.next_str());
                property_values[prop_idx].push(value);
                if value.is_finite() {
                    global_stats[prop_idx].add(value);
                }
            }
        }

        if num_samples > 0 && length.abs() > 1e-6 {
            length_interval_sum += length.abs() / num_samples as f64;
            length_interval_cnt += 1;
        }
    }

    let total_samples = positions_x.len();
    if total_samples == 0 || num_properties == 0 {
        println!("0");
        return;
    }
    if !global_stats.iter().any(|s| s.count > 0) {
        println!("0");
        return;
    }

    let interval_estimate = if interval_cnt > 0 {
        Some((interval_sum / interval_cnt as f64).max(0.05))
    } else if length_interval_cnt > 0 {
        Some((length_interval_sum / length_interval_cnt as f64).max(0.05))
    } else {
        None
    };

    let radius = estimate_radius(
        total_samples,
        min_x,
        max_x,
        min_y,
        max_y,
        min_z,
        max_z,
        interval_estimate,
    );
    let radius2 = radius * radius;

    let min_neighbors = if total_samples < 5_000 {
        3
    } else if total_samples < 20_000 {
        4
    } else {
        6
    };
    let local_threshold = if total_samples < 5_000 {
        2.8
    } else if total_samples < 50_000 {
        3.2
    } else if total_samples < 200_000 {
        3.6
    } else {
        4.0
    };
    let global_threshold = local_threshold + 0.6;
    let global_min_samples = if total_samples < 5_000 { 20 } else { 40 };

    let mut neighbor_aggs = vec![NeighborAgg::default(); num_properties];
    let mut active_props = Vec::<usize>::with_capacity(num_properties);
    let mut suspects = Vec::<Suspect>::new();

    if total_samples <= BRUTE_FORCE_LIMIT {
        for sample_idx in 0..total_samples {
            if !prepare_sample(
                sample_idx,
                &property_values,
                &mut neighbor_aggs,
                &mut active_props,
            ) {
                continue;
            }
            let px = positions_x[sample_idx];
            let py = positions_y[sample_idx];
            let pz = positions_z[sample_idx];

            for other_idx in 0..total_samples {
                if other_idx == sample_idx {
                    continue;
                }
                let dx = px - positions_x[other_idx];
                let dy = py - positions_y[other_idx];
                let dz = pz - positions_z[other_idx];
                let dist2 = dx * dx + dy * dy + dz * dz;
                if dist2 > radius2 {
                    continue;
                }
                for &prop_idx in &active_props {
                    let val = property_values[prop_idx][other_idx];
                    if val.is_finite() {
                        neighbor_aggs[prop_idx].add(val);
                    }
                }
            }

            evaluate_sample(
                sample_idx,
                &active_props,
                &neighbor_aggs,
                &property_values,
                &global_stats,
                min_neighbors,
                local_threshold,
                global_threshold,
                global_min_samples,
                &sample_hole_idx,
                &sample_idx_in_hole,
                &mut suspects,
            );
        }
    } else {
        let cell_size = radius.max(1.0);
        let inv_cell = 1.0 / cell_size;
        let mut grid: HashMap<CellKey, Vec<usize>> =
            HashMap::with_capacity(total_samples.saturating_mul(2).saturating_add(1));
        for idx in 0..total_samples {
            let key = cell_from_coords(
                positions_x[idx],
                positions_y[idx],
                positions_z[idx],
                inv_cell,
            );
            grid.entry(key).or_insert_with(Vec::new).push(idx);
        }

        for sample_idx in 0..total_samples {
            if !prepare_sample(
                sample_idx,
                &property_values,
                &mut neighbor_aggs,
                &mut active_props,
            ) {
                continue;
            }
            let px = positions_x[sample_idx];
            let py = positions_y[sample_idx];
            let pz = positions_z[sample_idx];
            let base_key = cell_from_coords(px, py, pz, inv_cell);

            for dx in -1..=1 {
                for dy in -1..=1 {
                    for dz in -1..=1 {
                        let neighbor_key = CellKey {
                            x: base_key.x + dx,
                            y: base_key.y + dy,
                            z: base_key.z + dz,
                        };
                        if let Some(indices) = grid.get(&neighbor_key) {
                            for &other_idx in indices {
                                if other_idx == sample_idx {
                                    continue;
                                }
                                let ddx = px - positions_x[other_idx];
                                let ddy = py - positions_y[other_idx];
                                let ddz = pz - positions_z[other_idx];
                                let dist2 = ddx * ddx + ddy * ddy + ddz * ddz;
                                if dist2 > radius2 {
                                    continue;
                                }
                                for &prop_idx in &active_props {
                                    let val = property_values[prop_idx][other_idx];
                                    if val.is_finite() {
                                        neighbor_aggs[prop_idx].add(val);
                                    }
                                }
                            }
                        }
                    }
                }
            }

            evaluate_sample(
                sample_idx,
                &active_props,
                &neighbor_aggs,
                &property_values,
                &global_stats,
                min_neighbors,
                local_threshold,
                global_threshold,
                global_min_samples,
                &sample_hole_idx,
                &sample_idx_in_hole,
                &mut suspects,
            );
        }
    }

    suspects.sort_by(|a, b| {
        b.confidence
            .partial_cmp(&a.confidence)
            .unwrap_or(Ordering::Equal)
    });

    println!("{}", suspects.len());
    for suspect in suspects {
        let hole_name = &hole_names[suspect.hole_idx];
        let property_name = &property_names[suspect.property_idx];
        let sample_number = suspect.sample_idx as usize + 1;
        println!(
            "{} {} {} {:.4}",
            hole_name, sample_number, property_name, suspect.confidence
        );
    }
}

fn sanitize(value: f64) -> f64 {
    if value.is_finite() {
        value
    } else {
        0.0
    }
}

fn parse_measurement(token: Option<&str>) -> f64 {
    if let Some(raw) = token {
        let trimmed = raw.trim_matches(|c| matches!(c, '"' | '\'' | ',')).trim();
        if trimmed.is_empty() || is_missing_literal(trimmed) {
            f64::NAN
        } else {
            trimmed.parse::<f64>().unwrap_or(f64::NAN)
        }
    } else {
        f64::NAN
    }
}

fn is_missing_literal(token: &str) -> bool {
    token.eq_ignore_ascii_case("na")
        || token.eq_ignore_ascii_case("nan")
        || token.eq_ignore_ascii_case("null")
        || token.eq_ignore_ascii_case("none")
        || token.eq_ignore_ascii_case("missing")
        || token.eq_ignore_ascii_case("inf")
        || token.eq_ignore_ascii_case("infinity")
        || token == "-"
        || token == "--"
        || token == "?"
        || token == "*"
}

fn normalize_direction(dx: f64, dy: f64, dz: f64) -> [f64; 3] {
    let mut x = dx;
    let mut y = dy;
    let mut z = dz;
    if !x.is_finite() {
        x = 0.0;
    }
    if !y.is_finite() {
        y = 0.0;
    }
    if !z.is_finite() {
        z = -1.0;
    }
    let norm = (x * x + y * y + z * z).sqrt();
    if !norm.is_finite() || norm < 1e-6 {
        [0.0, 0.0, -1.0]
    } else {
        [x / norm, y / norm, z / norm]
    }
}

fn estimate_radius(
    total_samples: usize,
    min_x: f64,
    max_x: f64,
    min_y: f64,
    max_y: f64,
    min_z: f64,
    max_z: f64,
    interval_estimate: Option<f64>,
) -> f64 {
    if total_samples == 0 {
        return 10.0;
    }
    let span_x = if min_x.is_finite() && max_x.is_finite() {
        (max_x - min_x).abs().max(1.0)
    } else {
        1.0
    };
    let span_y = if min_y.is_finite() && max_y.is_finite() {
        (max_y - min_y).abs().max(1.0)
    } else {
        1.0
    };
    let span_z = if min_z.is_finite() && max_z.is_finite() {
        (max_z - min_z).abs().max(1.0)
    } else {
        1.0
    };

    let sample_factor = if total_samples == 0 {
        1.0
    } else {
        total_samples as f64
    };

    let mut bbox_based = ((span_x * span_y * span_z) / sample_factor).cbrt();
    if !bbox_based.is_finite() || bbox_based <= 0.0 {
        bbox_based = 10.0;
    }

    let mut radius = bbox_based;
    if let Some(interval) = interval_estimate {
        if interval.is_finite() && interval > 0.0 {
            let interval_based = (interval * 3.0).clamp(interval * 1.5, interval * 6.0);
            radius = interval_based.max(bbox_based * 0.35);
        }
    }

    if total_samples > 200_000 {
        radius *= 1.25;
    } else if total_samples < 5_000 {
        radius *= 0.85;
    }

    radius.clamp(3.0, 150.0)
}

fn prepare_sample(
    sample_idx: usize,
    property_values: &[Vec<f64>],
    neighbor_aggs: &mut [NeighborAgg],
    active_props: &mut Vec<usize>,
) -> bool {
    active_props.clear();
    for (prop_idx, agg) in neighbor_aggs.iter_mut().enumerate() {
        agg.reset();
        if property_values[prop_idx][sample_idx].is_finite() {
            active_props.push(prop_idx);
        }
    }
    !active_props.is_empty()
}

fn evaluate_sample(
    sample_idx: usize,
    active_props: &[usize],
    neighbor_aggs: &[NeighborAgg],
    property_values: &[Vec<f64>],
    global_stats: &[RunningStat],
    min_neighbors: usize,
    local_threshold: f64,
    global_threshold: f64,
    global_min_samples: usize,
    sample_hole_idx: &[usize],
    sample_idx_in_hole: &[u32],
    suspects: &mut Vec<Suspect>,
) {
    for &prop_idx in active_props {
        let sample_value = property_values[prop_idx][sample_idx];
        let mut confidence = None;

        let agg = &neighbor_aggs[prop_idx];
        if (agg.count as usize) >= min_neighbors {
            let local_conf = compute_confidence(sample_value, agg.mean, agg.std_dev());
            if local_conf >= local_threshold {
                confidence = Some(local_conf);
            }
        }

        if confidence.is_none() {
            let global = &global_stats[prop_idx];
            if (global.count as usize) >= global_min_samples {
                if let Some(var) = global.variance() {
                    let std = var.sqrt();
                    let global_conf = compute_confidence(sample_value, global.mean, std);
                    if global_conf >= global_threshold {
                        confidence = Some(global_conf);
                    }
                }
            }
        }

        if let Some(conf) = confidence {
            suspects.push(Suspect {
                hole_idx: sample_hole_idx[sample_idx],
                sample_idx: sample_idx_in_hole[sample_idx],
                property_idx: prop_idx,
                confidence: conf,
            });
        }
    }
}

fn compute_confidence(value: f64, mean: f64, std_dev: f64) -> f64 {
    let diff = (value - mean).abs();
    let scale = (value.abs().max(mean.abs())) * 0.02;
    let denom = std_dev.max(scale).max(1e-6);
    diff / denom
}

fn cell_from_coords(x: f64, y: f64, z: f64, inv_size: f64) -> CellKey {
    CellKey {
        x: quantize_coord(x, inv_size),
        y: quantize_coord(y, inv_size),
        z: quantize_coord(z, inv_size),
    }
}

fn quantize_coord(value: f64, inv_size: f64) -> i64 {
    if value.is_finite() {
        (value * inv_size).floor() as i64
    } else if value.is_sign_negative() {
        i64::MIN / 4
    } else {
        i64::MAX / 4
    }
}

struct FastScanner {
    buf: Vec<u8>,
    idx: usize,
}

impl FastScanner {
    fn new() -> Self {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).unwrap();
        Self { buf: input, idx: 0 }
    }

    fn next_str(&mut self) -> Option<&str> {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let start = self.idx;
        while self.idx < self.buf.len() && !self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        let end = self.idx;
        Some(unsafe { std::str::from_utf8_unchecked(&self.buf[start..end]) })
    }

    fn next_parse<T: FromStr>(&mut self) -> Option<T> {
        self.next_str().and_then(|s| s.parse::<T>().ok())
    }

    fn next_owned(&mut self) -> Option<String> {
        self.next_str().map(|s| s.to_string())
    }
}

#[derive(Clone, Copy, Default)]
struct RunningStat {
    count: u64,
    mean: f64,
    m2: f64,
}

impl RunningStat {
    fn add(&mut self, value: f64) {
        self.count += 1;
        let count_f = self.count as f64;
        let delta = value - self.mean;
        self.mean += delta / count_f;
        let delta2 = value - self.mean;
        self.m2 += delta * delta2;
    }

    fn variance(&self) -> Option<f64> {
        if self.count > 1 {
            Some((self.m2 / (self.count as f64 - 1.0)).max(0.0))
        } else {
            None
        }
    }
}

#[derive(Clone, Copy, Default)]
struct NeighborAgg {
    count: u32,
    mean: f64,
    m2: f64,
}

impl NeighborAgg {
    fn reset(&mut self) {
        self.count = 0;
        self.mean = 0.0;
        self.m2 = 0.0;
    }

    fn add(&mut self, value: f64) {
        self.count += 1;
        let count_f = self.count as f64;
        let delta = value - self.mean;
        self.mean += delta / count_f;
        let delta2 = value - self.mean;
        self.m2 += delta * delta2;
    }

    fn std_dev(&self) -> f64 {
        if self.count > 1 {
            (self.m2 / (self.count as f64 - 1.0)).max(0.0).sqrt()
        } else {
            0.0
        }
    }
}

#[derive(Clone, Copy, Hash, Eq, PartialEq)]
struct CellKey {
    x: i64,
    y: i64,
    z: i64,
}

struct Suspect {
    hole_idx: usize,
    sample_idx: u32,
    property_idx: usize,
    confidence: f64,
}
