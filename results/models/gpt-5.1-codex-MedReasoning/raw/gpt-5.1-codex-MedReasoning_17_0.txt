using System;
using System.Collections.Generic;

public static class Program
{
    private static readonly char[] SplitSeparators = { ' ', '\t' };

    public static void Main()
    {
        string? headerLine = ReadNonEmptyLine();
        if (headerLine == null)
        {
            Console.WriteLine(0);
            return;
        }

        int[] header = ParseInts(headerLine);
        if (header.Length < 4)
        {
            Console.WriteLine(0);
            return;
        }

        int n = header[0];
        int containerW = header[1];
        int containerH = header[2];
        int containerD = header[3];

        var boxes = new List<BoxToPack>(Math.Max(n, 0));
        for (int i = 0; i < n; i++)
        {
            string? line = ReadNonEmptyLine();
            if (line == null)
            {
                boxes.Add(new BoxToPack(i, 0, 0, 0));
                continue;
            }

            int[] dims = ParseInts(line);
            int bw = dims.Length > 0 ? dims[0] : 0;
            int bh = dims.Length > 1 ? dims[1] : 0;
            int bd = dims.Length > 2 ? dims[2] : 0;
            boxes.Add(new BoxToPack(i, bw, bh, bd));
        }

        if (n <= 0 || containerW <= 0 || containerH <= 0 || containerD <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var placements = new List<Placement>(n);
        var spaces = new List<EmptySpace>(Math.Max(4, n * 2))
        {
            new EmptySpace(0, 0, 0, containerW, containerH, containerD)
        };

        boxes.Sort(CompareByVolumeDesc);

        var leftovers = new List<BoxToPack>();
        foreach (var box in boxes)
        {
            if (box.Rotations.Length == 0)
            {
                continue;
            }

            if (!TryPlaceBox(box, spaces, placements))
            {
                leftovers.Add(box);
            }
        }

        if (leftovers.Count > 0 && spaces.Count > 0)
        {
            leftovers.Sort(CompareByVolumeAsc);
            var retry = new List<BoxToPack>();

            foreach (var box in leftovers)
            {
                if (box.Rotations.Length == 0)
                {
                    continue;
                }

                if (!TryPlaceBox(box, spaces, placements))
                {
                    retry.Add(box);
                }
            }

            if (retry.Count > 0 && spaces.Count > 0)
            {
                retry.Sort((a, b) => a.MaxSide.CompareTo(b.MaxSide));
                foreach (var box in retry)
                {
                    if (box.Rotations.Length == 0)
                    {
                        continue;
                    }

                    TryPlaceBox(box, spaces, placements);
                }
            }
        }

        Console.WriteLine(placements.Count);
        foreach (var placement in placements)
        {
            Console.WriteLine($"{placement.BoxIndex} {placement.X} {placement.Y} {placement.Z}");
        }
    }

    private static string? ReadNonEmptyLine()
    {
        string? line;
        while ((line = Console.ReadLine()) != null)
        {
            if (!string.IsNullOrWhiteSpace(line))
            {
                return line;
            }
        }

        return null;
    }

    private static int[] ParseInts(string line)
    {
        var parts = line.Split(SplitSeparators, StringSplitOptions.RemoveEmptyEntries);
        var result = new int[parts.Length];
        for (int i = 0; i < parts.Length; i++)
        {
            int.TryParse(parts[i], out result[i]);
        }

        return result;
    }

    private static bool TryPlaceBox(BoxToPack box, List<EmptySpace> spaces, List<Placement> placements)
    {
        if (spaces.Count == 0 || box.Rotations.Length == 0)
        {
            return false;
        }

        int bestSpaceIndex = -1;
        Rotation bestRotation = default;
        long bestScore = long.MaxValue;
        bool perfect = false;

        int spaceCount = spaces.Count;
        long boxVolume = box.Volume;
        var rotations = box.Rotations;

        for (int i = 0; i < spaceCount && !perfect; i++)
        {
            var space = spaces[i];
            if (space.Volume < boxVolume)
            {
                continue;
            }

            for (int r = 0; r < rotations.Length; r++)
            {
                var rot = rotations[r];
                if (rot.W > space.W || rot.H > space.H || rot.D > space.D)
                {
                    continue;
                }

                long slackX = space.W - rot.W;
                long slackY = space.H - rot.H;
                long slackZ = space.D - rot.D;
                long leftover = space.Volume - boxVolume;
                long maxSlack = Math.Max(Math.Max(slackX, slackY), slackZ);

                long scorePrimary = leftover * 512L + (slackX + slackY + slackZ) * 32L + maxSlack;
                long positionalBias = ((long)space.Z << 12) + ((long)space.Y << 6) + space.X;
                long score = (scorePrimary << 12) + positionalBias;

                if (score < bestScore)
                {
                    bestScore = score;
                    bestSpaceIndex = i;
                    bestRotation = rot;

                    if (leftover == 0 && slackX == 0 && slackY == 0 && slackZ == 0)
                    {
                        perfect = true;
                        break;
                    }
                }
            }
        }

        if (bestSpaceIndex == -1)
        {
            return false;
        }

        var chosen = spaces[bestSpaceIndex];
        int lastIndex = spaces.Count - 1;
        spaces[bestSpaceIndex] = spaces[lastIndex];
        spaces.RemoveAt(lastIndex);

        placements.Add(new Placement(box.Index, chosen.X, chosen.Y, chosen.Z,
                                     bestRotation.W, bestRotation.H, bestRotation.D));

        AddSpace(spaces, chosen.X + bestRotation.W, chosen.Y, chosen.Z,
                 chosen.W - bestRotation.W, chosen.H, chosen.D);
        AddSpace(spaces, chosen.X, chosen.Y + bestRotation.H, chosen.Z,
                 bestRotation.W, chosen.H - bestRotation.H, chosen.D);
        AddSpace(spaces, chosen.X, chosen.Y, chosen.Z + bestRotation.D,
                 bestRotation.W, bestRotation.H, chosen.D - bestRotation.D);

        return true;
    }

    private static void AddSpace(List<EmptySpace> spaces, int x, int y, int z, int w, int h, int d)
    {
        if (w <= 0 || h <= 0 || d <= 0)
        {
            return;
        }

        spaces.Add(new EmptySpace(x, y, z, w, h, d));
    }

    private static int CompareByVolumeDesc(BoxToPack a, BoxToPack b)
    {
        int cmp = b.Volume.CompareTo(a.Volume);
        if (cmp != 0) return cmp;

        cmp = b.MaxSide.CompareTo(a.MaxSide);
        if (cmp != 0) return cmp;

        return a.Index.CompareTo(b.Index);
    }

    private static int CompareByVolumeAsc(BoxToPack a, BoxToPack b)
    {
        int cmp = a.Volume.CompareTo(b.Volume);
        if (cmp != 0) return cmp;

        cmp = a.MaxSide.CompareTo(b.MaxSide);
        if (cmp != 0) return cmp;

        return a.Index.CompareTo(b.Index);
    }
}

public readonly struct Rotation
{
    public int W { get; }
    public int H { get; }
    public int D { get; }

    public Rotation(int w, int h, int d)
    {
        W = w;
        H = h;
        D = d;
    }
}

public struct EmptySpace
{
    public int X;
    public int Y;
    public int Z;
    public int W;
    public int H;
    public int D;
    public long Volume;

    public EmptySpace(int x, int y, int z, int w, int h, int d)
    {
        X = x;
        Y = y;
        Z = z;
        W = w;
        H = h;
        D = d;
        Volume = (long)w * h * d;
    }
}

public struct Placement
{
    public int BoxIndex;
    public int X;
    public int Y;
    public int Z;
    public int W;
    public int H;
    public int D;

    public Placement(int boxIndex, int x, int y, int z, int w, int h, int d)
    {
        BoxIndex = boxIndex;
        X = x;
        Y = y;
        Z = z;
        W = w;
        H = h;
        D = d;
    }
}

public class BoxToPack
{
    public int Index { get; }
    public int W { get; }
    public int H { get; }
    public int D { get; }
    public long Volume { get; }
    public int MaxSide { get; }
    public Rotation[] Rotations { get; }

    public BoxToPack(int index, int w, int h, int d)
    {
        Index = index;
        W = w;
        H = h;
        D = d;
        MaxSide = Math.Max(Math.Max(w, h), d);

        if (w <= 0 || h <= 0 || d <= 0)
        {
            Volume = 0;
            Rotations = Array.Empty<Rotation>();
        }
        else
        {
            Volume = (long)w * h * d;
            Rotations = BuildRotations(w, h, d);
        }
    }

    private static Rotation[] BuildRotations(int w, int h, int d)
    {
        var combos = new (int, int, int)[6]
        {
            (w, h, d),
            (w, d, h),
            (h, w, d),
            (h, d, w),
            (d, w, h),
            (d, h, w)
        };

        var list = new List<Rotation>(6);
        for (int i = 0; i < combos.Length; i++)
        {
            var combo = combos[i];
            if (combo.Item1 <= 0 || combo.Item2 <= 0 || combo.Item3 <= 0)
            {
                continue;
            }

            bool duplicate = false;
            for (int j = 0; j < list.Count; j++)
            {
                var existing = list[j];
                if (existing.W == combo.Item1 &&
                    existing.H == combo.Item2 &&
                    existing.D == combo.Item3)
                {
                    duplicate = true;
                    break;
                }
            }

            if (!duplicate)
            {
                list.Add(new Rotation(combo.Item1, combo.Item2, combo.Item3));
            }
        }

        return list.ToArray();
    }
}
