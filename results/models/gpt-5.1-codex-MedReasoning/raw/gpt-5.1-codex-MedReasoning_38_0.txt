import sys

def main():
    sys.setrecursionlimit(1_000_000)
    lines = sys.stdin.read().splitlines()

    idx = 0
    while idx < len(lines) and not lines[idx].strip():
        idx += 1
    if idx == len(lines):
        print("NO SOLUTION")
        return

    parts = lines[idx].split()
    if len(parts) < 2:
        print("NO SOLUTION")
        return

    try:
        U = int(parts[0])
        S = int(parts[1])
    except ValueError:
        print("NO SOLUTION")
        return

    if U < 0 or S < 0:
        print("NO SOLUTION")
        return

    set_lines = []
    line_idx = idx + 1
    for _ in range(S):
        if line_idx < len(lines):
            set_lines.append(lines[line_idx])
            line_idx += 1
        else:
            set_lines.append('')

    sets_by_element = [[] for _ in range(U)]
    set_bits = []

    for s_idx, line in enumerate(set_lines):
        bits = 0
        seen = set()
        stripped = line.strip()
        if stripped:
            for token in stripped.split():
                try:
                    elem = int(token)
                except ValueError:
                    print("NO SOLUTION")
                    return
                if 0 <= elem < U and elem not in seen:
                    seen.add(elem)
                    bits |= 1 << elem
                    sets_by_element[elem].append(s_idx)
        set_bits.append(bits)

    for elem in range(U):
        if not sets_by_element[elem]:
            print("NO SOLUTION")
            return
        sets_by_element[elem].sort()

    full_mask = (1 << U) - 1
    memo = set()

    def select_element(uncovered):
        uc = uncovered
        best_elem = -1
        best_candidates = None
        best_count = None
        while uc:
            bit = uc & -uc
            elem = bit.bit_length() - 1
            candidates = []
            for idx_set in sets_by_element[elem]:
                subset = set_bits[idx_set]
                if (subset & uncovered) == subset:
                    candidates.append(idx_set)
            if not candidates:
                return elem, []
            cnt = len(candidates)
            if best_count is None or cnt < best_count:
                best_elem = elem
                best_candidates = candidates
                best_count = cnt
                if cnt == 1:
                    break
            uc ^= bit
        if best_candidates is None:
            return -1, []
        return best_elem, best_candidates

    def dfs(uncovered):
        if uncovered == 0:
            return []
        if uncovered in memo:
            return None
        elem, candidates = select_element(uncovered)
        if not candidates:
            memo.add(uncovered)
            return None
        for idx_set in candidates:
            subset = set_bits[idx_set]
            result = dfs(uncovered ^ subset)
            if result is not None:
                result.append(idx_set)
                return result
        memo.add(uncovered)
        return None

    solution = dfs(full_mask)
    if solution is None:
        print("NO SOLUTION")
    else:
        solution.sort()
        print("SOLUTION")
        if solution:
            print(" ".join(map(str, solution)))
        else:
            print()

if __name__ == "__main__":
    main()
