OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
%glsl = OpExtInstImport "GLSL.std.450"
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_2 = OpConstant %uint 2
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_Input_v4 = OpTypePointer Input %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4 = OpTypePointer Output %v4float
%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Output_v3 = OpTypePointer Output %v3float
%float_0 = OpConstant %float 0
%float_0_03 = OpConstant %float 0.03
%float_1e_6 = OpConstant %float 0.000001
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%gl_in = OpVariable %ptr_Input_arr_pv Input
%worldPos = OpVariable %ptr_Input_arr_v3 Input
%normal = OpVariable %ptr_Input_arr_v3 Input
%color = OpVariable %ptr_Input_arr_v3 Input
%gl_out = OpVariable %ptr_Output_pv Output
%outColor = OpVariable %ptr_Output_v3 Output
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
%main = OpFunction %void None %func
%entry = OpLabel
%p0ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0ptr
%p1ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1ptr
%dir = OpFSub %v4float %p1 %p0
%dx = OpCompositeExtract %float %dir 0
%dy = OpCompositeExtract %float %dir 1
%dx2 = OpFMul %float %dx %dx
%dy2 = OpFMul %float %dy %dy
%len2 = OpFAdd %float %dx2 %dy2
%len = OpExtInst %float %glsl Sqrt %len2
%lenSafe = OpExtInst %float %glsl FMax %len %float_1e_6
%scale = OpFDiv %float %float_0_03 %lenSafe
%negDy = OpFNegate %float %dy
%perp_x = OpFMul %float %negDy %scale
%perp_y = OpFMul %float %dx %scale
%perp = OpCompositeConstruct %v4float %perp_x %perp_y %float_0 %float_0
%p0_plus = OpFAdd %v4float %p0 %perp
%p0_minus = OpFSub %v4float %p0 %perp
%p1_plus = OpFAdd %v4float %p1 %perp
%p1_minus = OpFSub %v4float %p1 %perp
%col0ptr = OpAccessChain %ptr_Input_v3 %color %int_0
%c0 = OpLoad %v3float %col0ptr
%col1ptr = OpAccessChain %ptr_Input_v3 %color %int_1
%c1 = OpLoad %v3float %col1ptr
%outPosPtr = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %outPosPtr %p0_plus
OpStore %outColor %c0
OpEmitVertex
OpStore %outPosPtr %p0_minus
OpStore %outColor %c0
OpEmitVertex
OpStore %outPosPtr %p1_plus
OpStore %outColor %c1
OpEmitVertex
OpStore %outPosPtr %p1_minus
OpStore %outColor %c1
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
