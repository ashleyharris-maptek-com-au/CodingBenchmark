OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
%void = OpTypeVoid
%functype = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%uint = OpTypeInt 32 0
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_04 = OpConstant %float 0.04
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_Input_v4 = OpTypePointer Input %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_Output_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4 = OpTypePointer Output %v4float
%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Output_v3 = OpTypePointer Output %v3float
OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0
%gl_in = OpVariable %ptr_Input_arr_pv Input
%worldPos_in = OpVariable %ptr_Input_arr_v3 Input
%normal_in = OpVariable %ptr_Input_arr_v3 Input
%color_in = OpVariable %ptr_Input_arr_v3 Input
%gl_out = OpVariable %ptr_Output_pv Output
%outColor = OpVariable %ptr_Output_v3 Output
%main = OpFunction %void None %functype
%label = OpLabel
%pos_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr
%col_ptr = OpAccessChain %ptr_Input_v3 %color_in %int_0
%col = OpLoad %v3float %col_ptr
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1
%x_minus = OpFSub %float %x %float_0_04
%x_plus = OpFAdd %float %x %float_0_04
%y_minus = OpFSub %float %y %float_0_04
%y_plus = OpFAdd %float %y %float_0_04
%pos_out_ptr = OpAccessChain %ptr_Output_v4 %gl_out %int_0
; Vertex 0
%v0 = OpCompositeConstruct %v4float %x_minus %y_minus %float_0 %float_1

%v1 = OpCompositeConstruct %v4float %x_plus %y_minus %float_0 %float_1
OpStore %pos_out_ptr %v1
OpStore %outColor %col
OpEmitVertex
; Vertex 2
%v2 = OpCompositeConstruct %v4float %x_minus %y_plus %float_0 %float_1
OpStore %pos_out_ptr %v2
OpStore %outColor %col
OpEmitVertex
; Vertex 3
%v3 = OpCompositeConstruct %v4float %x_plus %y_plus %float_0 %float_1
OpStore %pos_out_ptr %v3
OpStore %outColor %col
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
