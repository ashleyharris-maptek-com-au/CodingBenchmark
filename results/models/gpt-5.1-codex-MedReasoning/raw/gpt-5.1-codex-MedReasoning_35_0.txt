use std::io::{self, Read};

fn coverage_gain(node: usize, dominated: &[bool], adj: &[Vec<usize>]) -> usize {
    let mut gain = 0;
    if !dominated[node] {
        gain += 1;
    }
    for &nb in &adj[node] {
        if !dominated[nb] {
            gain += 1;
        }
    }
    gain
}

fn main() {
    // Read entire input
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() || input.trim().is_empty() {
        println!("0\n");
        return;
    }
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(v) => v.parse().unwrap_or(0),
        None => {
            println!("0\n");
            return;
        }
    };
    let m: usize = it.next().unwrap_or("0").parse().unwrap_or(0);

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        if let (Some(us), Some(vs)) = (it.next(), it.next()) {
            let mut u: usize = us.parse().unwrap_or(0);
            let mut v: usize = vs.parse().unwrap_or(0);
            if u >= n || v >= n {
                continue;
            }
            if u == v {
                continue; // Ignore self-loops; they don't affect dominance
            }
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    for neighbors in &mut adj {
        neighbors.sort_unstable();
        neighbors.dedup();
    }

    if n == 0 {
        println!("0\n");
        return;
    }

    // Order vertices by degree (descending)
    let mut order: Vec<usize> = (0..n).collect();
    order.sort_unstable_by(|&a, &b| {
        let da = adj[a].len();
        let db = adj[b].len();
        db.cmp(&da).then_with(|| a.cmp(&b))
    });

    let mut dominated = vec![false; n];
    let mut solution = Vec::<usize>::new();

    for &v in &order {
        if dominated[v] {
            continue;
        }
        let mut best = v;
        let mut best_gain = coverage_gain(v, &dominated, &adj);

        for &nb in &adj[v] {
            let gain = coverage_gain(nb, &dominated, &adj);
            if gain > best_gain
                || (gain == best_gain && adj[nb].len() > adj[best].len())
                || (gain == best_gain && adj[nb].len() == adj[best].len() && nb < best)
            {
                best = nb;
                best_gain = gain;
            }
        }

        // Ensure we add something even if gain is zero (shouldn't happen, but be safe)
        if best_gain == 0 {
            best = v;
        }

        if !dominated[best] {
            dominated[best] = true;
        }
        for &nb in &adj[best] {
            dominated[nb] = true;
        }
        solution.push(best);
    }

    // Cover any remaining undominated vertices
    for v in 0..n {
        if !dominated[v] {
            dominated[v] = true;
            for &nb in &adj[v] {
                dominated[nb] = true;
            }
            solution.push(v);
        }
    }

    // Build coverage counts for reduction phase
    let mut in_solution = vec![false; n];
    for &s in &solution {
        in_solution[s] = true;
    }

    let mut coverage_count = vec![0u32; n];
    for &s in &solution {
        coverage_count[s] += 1;
        for &nb in &adj[s] {
            coverage_count[nb] += 1;
        }
    }

    // Iteratively try to remove redundant vertices
    let mut changed = true;
    while changed {
        changed = false;
        let current_solution: Vec<usize> = in_solution
            .iter()
            .enumerate()
            .filter_map(|(idx, &flag)| if flag { Some(idx) } else { None })
            .collect();

        for &s in &current_solution {
            if !in_solution[s] {
                continue;
            }
            if coverage_count[s] <= 1 {
                continue;
            }
            let mut can_remove = true;
            for &nb in &adj[s] {
                if coverage_count[nb] <= 1 {
                    can_remove = false;
                    break;
                }
            }
            if !can_remove {
                continue;
            }
            in_solution[s] = false;
            coverage_count[s] -= 1;
            for &nb in &adj[s] {
                coverage_count[nb] -= 1;
            }
            changed = true;
        }
    }

    let final_solution: Vec<usize> = in_solution
        .iter()
        .enumerate()
        .filter_map(|(idx, &flag)| if flag { Some(idx) } else { None })
        .collect();

    println!("{}", final_solution.len());
    if !final_solution.is_empty() {
        for (i, v) in final_solution.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", v);
        }
        println!();
    } else {
        println!();
    }
}
