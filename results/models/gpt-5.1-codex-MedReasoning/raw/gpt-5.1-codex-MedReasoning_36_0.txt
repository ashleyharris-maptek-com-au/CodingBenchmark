#include <iostream>
#include <vector>
#include <algorithm>
#include <bit>
#include <cstdint>
#include <queue>
#include <utility>

struct MaxClique {
    int n;
    int blocks;
    const std::vector<std::vector<uint64_t>>& adj;
    std::vector<int> current;
    std::vector<int> best;
    int bestSize;

    MaxClique(int n, int blocks, const std::vector<std::vector<uint64_t>>& adj)
        : n(n), blocks(blocks), adj(adj), current(), best(), bestSize(0) {
        current.reserve(n);
        best.reserve(n);
    }

    void updateBest() {
        if (static_cast<int>(current.size()) > bestSize) {
            bestSize = static_cast<int>(current.size());
            best = current;
        }
    }

    bool isEmpty(const std::vector<uint64_t>& bs) const {
        for (uint64_t x : bs) {
            if (x) return false;
            }
        return true;
    }

    void nodesFromBitset(const std::vector<uint64_t>& bs, std::vector<int>& nodes) const {
        nodes.clear();
        int total = 0;
        for (uint64_t block : bs) {
            total += std::popcount(block);
        }
        nodes.reserve(total);
        for (int b = 0; b < blocks; ++b) {
            uint64_t bits = bs[b];
            while (bits) {
                int offset = std::countr_zero(bits);
                nodes.push_back((b << 6) + offset);
                bits &= bits - 1;
            }
        }
    }

    void colorOrder(std::vector<int> nodes, std::vector<int>& order, std::vector<int>& colors) const {
        order.clear();
        colors.clear();
        order.reserve(nodes.size());
        colors.reserve(nodes.size());
        std::vector<uint64_t> used(blocks);
        std::vector<int> next;
        int color = 0;
        while (!nodes.empty()) {
            ++color;
            std::fill(used.begin(), used.end(), 0ULL);
            next.clear();
            next.reserve(nodes.size());
            for (int v : nodes) {
                bool conflict = false;
                for (int b = 0; b < blocks; ++b) {
                    if (adj[v][b] & used[b]) {
                        conflict = true;
                        break;
                    }
                }
                if (!conflict) {
                    order.push_back(v);
                    colors.push_back(color);
                    int blockIndex = v >> 6;
                    used[blockIndex] |= (1ULL << (v & 63));
                } else {
                    next.push_back(v);
                }
            }
            nodes.swap(next);
        }
    }

    void expand(std::vector<uint64_t> cand) {
        std::vector<int> nodes;
        nodesFromBitset(cand, nodes);
        if (nodes.empty()) {
            updateBest();
            return;
        }
        std::vector<int> order;
        std::vector<int> colors;
        colorOrder(std::move(nodes), order, colors);
        for (int idx = static_cast<int>(order.size()) - 1; idx >= 0; --idx) {
            int v = order[idx];
            if (static_cast<int>(current.size()) + colors[idx] <= bestSize) {
                break;
            }
            current.push_back(v);
            std::vector<uint64_t> newCand(blocks);
            for (int b = 0; b < blocks; ++b) {
                newCand[b] = cand[b] & adj[v][b];
            }
            if (isEmpty(newCand)) {
                updateBest();
            } else {
                expand(std::move(newCand));
            }
            current.pop_back();
            int blockIndex = v >> 6;
            uint64_t bit = 1ULL << (v & 63);
            cand[blockIndex] &= ~bit;
        }
    }

    std::vector<int> solve(const std::vector<uint64_t>& initial) {
        best.clear();
        current.clear();
        bestSize = 0;
        expand(initial);
        return best;
    }
};

bool isBipartite(const std::vector<std::vector<int>>& graph,
                 const std::vector<char>& forbidden,
                 std::vector<int>& color) {
    int n = static_cast<int>(graph.size());
    color.assign(n, -1);
    std::queue<int> q;
    for (int i = 0; i < n; ++i) {
        if (forbidden[i] || color[i] != -1) continue;
        color[i] = 0;
        q.push(i);
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (int v : graph[u]) {
                if (forbidden[v]) continue;
                if (color[v] == -1) {
                    color[v] = color[u] ^ 1;
                    q.push(v);
                } else if (color[v] == color[u]) {
                    return false;
                }
            }
        }
    }
    return true;
}

std::vector<int> maximumIndependentSetBipartite(
    const std::vector<std::vector<int>>& graph,
    const std::vector<int>& color,
    const std::vector<char>& forbidden) {
    int n = static_cast<int>(graph.size());
    std::vector<int> leftIndex(n, -1), rightIndex(n, -1);
    std::vector<int> leftNodes, rightNodes;
    leftNodes.reserve(n);
    rightNodes.reserve(n);
    for (int v = 0; v < n; ++v) {
        if (forbidden[v]) continue;
        if (color[v] == 0) {
            leftIndex[v] = static_cast<int>(leftNodes.size());
            leftNodes.push_back(v);
        } else {
            rightIndex[v] = static_cast<int>(rightNodes.size());
            rightNodes.push_back(v);
        }
    }
    int L = static_cast<int>(leftNodes.size());
    int R = static_cast<int>(rightNodes.size());
    std::vector<std::vector<int>> adjLeft(L);
    for (int idx = 0; idx < L; ++idx) {
        int v = leftNodes[idx];
        adjLeft[idx].reserve(graph[v].size());
        for (int nb : graph[v]) {
            if (forbidden[nb]) continue;
            int rid = rightIndex[nb];
            if (rid >= 0) {
                adjLeft[idx].push_back(rid);
            }
        }
    }
    std::vector<int> pairU(L, -1), pairV(R, -1), dist(L);
    auto bfs = [&]() -> bool {
        std::queue<int> q;
        for (int u = 0; u < L; ++u) {
            if (pairU[u] == -1) {
                dist[u] = 0;
                q.push(u);
            } else {
                dist[u] = -1;
            }
        }
        bool reachableFree = false;
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (int v : adjLeft[u]) {
                int mate = pairV[v];
                if (mate != -1 && dist[mate] == -1) {
                    dist[mate] = dist[u] + 1;
                    q.push(mate);
                }
                if (mate == -1) {
                    reachableFree = true;
                }
            }
        }
        return reachableFree;
    };
    auto dfs = [&](auto&& self, int u) -> bool {
        for (int v : adjLeft[u]) {
            int mate = pairV[v];
            if (mate == -1 || (dist[mate] == dist[u] + 1 && self(self, mate))) {
                pairU[u] = v;
                pairV[v] = u;
                return true;
            }
        }
        dist[u] = -1;
        return false;
    };
    while (bfs()) {
        for (int u = 0; u < L; ++u) {
            if (pairU[u] == -1) {
                dfs(dfs, u);
            }
        }
    }
    std::vector<char> visitedLeft(L, 0), visitedRight(R, 0);
    std::queue<int> q;
    for (int u = 0; u < L; ++u) {
        if (pairU[u] == -1) {
            visitedLeft[u] = 1;
            q.push(u);
        }
    }
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        for (int v : adjLeft[u]) {
            if (pairU[u] == v) continue;
            if (visitedRight[v]) continue;
            visitedRight[v] = 1;
            int mate = pairV[v];
            if (mate != -1 && !visitedLeft[mate]) {
                visitedLeft[mate] = 1;
                q.push(mate);
            }
        }
    }
    std::vector<int> independent;
    independent.reserve(n);
    for (int i = 0; i < L; ++i) {
        if (visitedLeft[i]) {
            independent.push_back(leftNodes[i]);
        }
    }
    for (int i = 0; i < R; ++i) {
        if (!visitedRight[i]) {
            independent.push_back(rightNodes[i]);
        }
    }
    return independent;
}

void printAnswer(const std::vector<int>& nodes) {
    std::cout << nodes.size() << "\n";
    if (!nodes.empty()) {
        for (std::size_t i = 0; i < nodes.size(); ++i) {
            if (i) std::cout << ' ';
            std::cout << nodes[i];
        }
    }
    std::cout << "\n";
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n;
    long long m;
    if (!(std::cin >> n >> m)) {
        return 0;
    }
    if (n == 0) {
        std::cout << 0 << "\n\n";
        return 0;
    }

    std::vector<std::vector<int>> graph(n);
    std::vector<char> forbidden(n, 0);
    for (long long i = 0; i < m; ++i) {
        int u, v;
        std::cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) {
            forbidden[u] = 1;
            continue;
        }
        graph[u].push_back(v);
        graph[v].push_back(u);
    }

    bool hadForbidden = false;
    for (char f : forbidden) {
        if (f) {
            hadForbidden = true;
            break;
        }
    }
    if (hadForbidden) {
        for (int u = 0; u < n; ++u) {
            if (forbidden[u]) {
                graph[u].clear();
            } else if (!graph[u].empty()) {
                auto& vec = graph[u];
                vec.erase(std::remove_if(vec.begin(), vec.end(),
                                         [&](int v) { return forbidden[v]; }),
                          vec.end());
            }
        }
    }

    std::vector<int> forced;
    forced.reserve(n);
    for (int i = 0; i < n; ++i) {
        if (!forbidden[i] && graph[i].empty()) {
            forbidden[i] = 1;
            forced.push_back(i);
        }
    }

    bool hasActive = false;
    for (int i = 0; i < n; ++i) {
        if (!forbidden[i]) {
            hasActive = true;
            break;
        }
    }
    if (!hasActive) {
        std::sort(forced.begin(), forced.end());
        printAnswer(forced);
        return 0;
    }

    std::vector<int> color;
    if (isBipartite(graph, forbidden, color)) {
        auto result = maximumIndependentSetBipartite(graph, color, forbidden);
        result.insert(result.end(), forced.begin(), forced.end());
        std::sort(result.begin(), result.end());
        printAnswer(result);
        return 0;
    }

    std::vector<int> activeVertices;
    std::vector<int> activeId(n, -1);
    activeVertices.reserve(n);
    for (int i = 0; i < n; ++i) {
        if (!forbidden[i]) {
            activeId[i] = static_cast<int>(activeVertices.size());
            activeVertices.push_back(i);
        }
    }
    int activeN = static_cast<int>(activeVertices.size());
    if (activeN == 0) {
        std::vector<int> result = forced;
        std::sort(result.begin(), result.end());
        printAnswer(result);
        return 0;
    }
    int blocks = (activeN + 63) / 64;
    std::vector<uint64_t> universal(blocks, ~0ULL);
    if ((activeN & 63) != 0) {
        universal.back() = (1ULL << (activeN & 63)) - 1ULL;
    }
    std::vector<std::vector<uint64_t>> comp(activeN, std::vector<uint64_t>(blocks, 0ULL));
    for (int idx = 0; idx < activeN; ++idx) {
        comp[idx] = universal;
        int blockIndex = idx >> 6;
        comp[idx][blockIndex] &= ~(1ULL << (idx & 63));
        int orig = activeVertices[idx];
        for (int nb : graph[orig]) {
            int nbId = activeId[nb];
            if (nbId == -1) continue;
            comp[idx][nbId >> 6] &= ~(1ULL << (nbId & 63));
        }
    }

    MaxClique solver(activeN, blocks, comp);
    auto cliqueIndices = solver.solve(universal);
    std::vector<int> answer;
    answer.reserve(cliqueIndices.size() + forced.size());
    for (int idx : cliqueIndices) {
        answer.push_back(activeVertices[idx]);
    }
    answer.insert(answer.end(), forced.begin(), forced.end());
    std::sort(answer.begin(), answer.end());
    printAnswer(answer);
    return 0;
}
