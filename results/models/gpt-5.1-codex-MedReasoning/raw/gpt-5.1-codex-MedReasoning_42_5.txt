%arrow_y = OpCompositeExtract %float %arrow_vec2 1
%offset_arrow = OpCompositeConstruct %v4float %arrow_x %arrow_y %float_0 %float_0
%neg_offset_arrow = OpFNegate %v4float %offset_arrow

%ptr_c0 = OpAccessChain %ptr_Input_v3 %color %int_0
