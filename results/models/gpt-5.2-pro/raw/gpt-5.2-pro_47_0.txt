```wgsl
@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>             params   : vec4<u32>;
// params.x = difficulty, params.y = range_start, params.z = range_size

fn rotl32(x : u32, s : u32) -> u32 {
    return (x << s) | (x >> (32u - s));
}

fn trailing_zeros_64(h0 : u32, h1 : u32) -> u32 {
    if (h0 != 0u) {
        return countTrailingZeros(h0);
    }
    if (h1 != 0u) {
        return 32u + countTrailingZeros(h1);
    }
    return 64u;
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    // Guard: out of range
    if (gid.x >= params.z) {
        return;
    }

    let difficulty : u32 = params.x;
    let nonce : u32 = params.y + gid.x;

    // state[0..15] = baseData[0..15] ^ (nonce + i)
    var state : array<u32, 16>;
    state[0]  = baseData[0]  ^ (nonce + 0u);
    state[1]  = baseData[1]  ^ (nonce + 1u);
    state[2]  = baseData[2]  ^ (nonce + 2u);
    state[3]  = baseData[3]  ^ (nonce + 3u);
    state[4]  = baseData[4]  ^ (nonce + 4u);
    state[5]  = baseData[5]  ^ (nonce + 5u);
    state[6]  = baseData[6]  ^ (nonce + 6u);
    state[7]  = baseData[7]  ^ (nonce + 7u);
    state[8]  = baseData[8]  ^ (nonce + 8u);
    state[9]  = baseData[9]  ^ (nonce + 9u);
    state[10] = baseData[10] ^ (nonce + 10u);
    state[11] = baseData[11] ^ (nonce + 11u);
    state[12] = baseData[12] ^ (nonce + 12u);
    state[13] = baseData[13] ^ (nonce + 13u);
    state[14] = baseData[14] ^ (nonce + 14u);
    state[15] = baseData[15] ^ (nonce + 15u);

    // 8 rounds; inner quarter-round groups explicitly unrolled
    for (var r : u32 = 0u; r < 8u; r = r + 1u) {
        // Group (0,1,2,3)
        var a0 : u32 = state[0];
        var b0 : u32 = state[1];
        var c0 : u32 = state[2];
        var d0 : u32 = state[3];
        a0 = a0 + b0; d0 = d0 ^ a0; d0 = rotl32(d0, 16u);
        c0 = c0 + d0; b0 = b0 ^ c0; b0 = rotl32(b0, 12u);
        a0 = a0 + b0; d0 = d0 ^ a0; d0 = rotl32(d0, 8u);
        c0 = c0 + d0; b0 = b0 ^ c0; b0 = rotl32(b0, 7u);
        state[0] = a0; state[1] = b0; state[2] = c0; state[3] = d0;

        // Group (4,5,6,7)
        var a1 : u32 = state[4];
        var b1 : u32 = state[5];
        var c1 : u32 = state[6];
        var d1 : u32 = state[7];
        a1 = a1 + b1; d1 = d1 ^ a1; d1 = rotl32(d1, 16u);
        c1 = c1 + d1; b1 = b1 ^ c1; b1 = rotl32(b1, 12u);
        a1 = a1 + b1; d1 = d1 ^ a1; d1 = rotl32(d1, 8u);
        c1 = c1 + d1; b1 = b1 ^ c1; b1 = rotl32(b1, 7u);
        state[4] = a1; state[5] = b1; state[6] = c1; state[7] = d1;

        // Group (8,9,10,11)
        var a2 : u32 = state[8];
        var b2 : u32 = state[9];
        var c2 : u32 = state[10];
        var d2 : u32 = state[11];
        a2 = a2 + b2; d2 = d2 ^ a2; d2 = rotl32(d2, 16u);
        c2 = c2 + d2; b2 = b2 ^ c2; b2 = rotl32(b2, 12u);
        a2 = a2 + b2; d2 = d2 ^ a2; d2 = rotl32(d2, 8u);
        c2 = c2 + d2; b2 = b2 ^ c2; b2 = rotl32(b2, 7u);
        state[8] = a2; state[9] = b2; state[10] = c2; state[11] = d2;

        // Group (12,13,14,15)
        var a3 : u32 = state[12];
        var b3 : u32 = state[13];
        var c3 : u32 = state[14];
        var d3 : u32 = state[15];
        a3 = a3 + b3; d3 = d3 ^ a3; d3 = rotl32(d3, 16u);
        c3 = c3 + d3; b3 = b3 ^ c3; b3 = rotl32(b3, 12u);
        a3 = a3 + b3; d3 = d3 ^ a3; d3 = rotl32(d3, 8u);
        c3 = c3 + d3; b3 = b3 ^ c3; b3 = rotl32(b3, 7u);
        state[12] = a3; state[13] = b3; state[14] = c3; state[15] = d3;
    }

    // Fold 16 words to 2
    let h0 : u32 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
    let h1 : u32 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];

    // Trailing zero count in 64-bit value [h0 | (h1 << 32)]
    let tz : u32 = trailing_zeros_64(h0, h1);

    if (tz >= difficulty) {
        // Race writes are acceptable
        result[0] = 1u;
        result[1] = nonce;
        result[2] = h0;
        result[3] = h1;
    }
}
```
