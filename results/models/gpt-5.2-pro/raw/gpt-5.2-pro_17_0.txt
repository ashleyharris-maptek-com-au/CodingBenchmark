```csharp
using System;
using System.Collections.Generic;
using System.Text;

public static class Program
{
    // ---------------- Fast Scanner ----------------
    private sealed class FastScanner
    {
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = Console.OpenStandardInput().Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = ReadByte(); } while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }
            int x = 0;
            while (c > 32)
            {
                x = x * 10 + (c - '0');
                c = ReadByte();
            }
            return x * sign;
        }
    }

    // ---------------- Data Structures ----------------
    private readonly struct Orientation
    {
        public readonly int W, H, D;
        public Orientation(int w, int h, int d) { W = w; H = h; D = d; }
    }

    private sealed class Box
    {
        public int Index;
        public int A, B, C;
        public long Volume;
        public Orientation[] Orients = Array.Empty<Orientation>();
        public int MinSide, MaxSide;

        public void PrecomputeOrients()
        {
            // generate up to 6 unique permutations
            Span<Orientation> tmp = stackalloc Orientation[6];
            tmp[0] = new Orientation(A, B, C);
            tmp[1] = new Orientation(A, C, B);
            tmp[2] = new Orientation(B, A, C);
            tmp[3] = new Orientation(B, C, A);
            tmp[4] = new Orientation(C, A, B);
            tmp[5] = new Orientation(C, B, A);

            var list = new List<Orientation>(6);
            for (int i = 0; i < 6; i++)
            {
                var o = tmp[i];
                bool dup = false;
                for (int j = 0; j < list.Count; j++)
                {
                    var p = list[j];
                    if (p.W == o.W && p.H == o.H && p.D == o.D) { dup = true; break; }
                }
                if (!dup) list.Add(o);
            }
            Orients = list.ToArray();

            MinSide = Math.Min(A, Math.Min(B, C));
            MaxSide = Math.Max(A, Math.Max(B, C));
        }
    }

    private readonly struct Placement
    {
        public readonly int Index;
        public readonly int X, Y, Z;
        public Placement(int index, int x, int y, int z) { Index = index; X = x; Y = y; Z = z; }
    }

    private readonly struct Space
    {
        public readonly int X, Y, Z;
        public readonly int W, H, D;
        public Space(int x, int y, int z, int w, int h, int d) { X = x; Y = y; Z = z; W = w; H = h; D = d; }
        public long Volume => (long)W * H * D;
    }

    private sealed class PackResult
    {
        public List<Placement> Placements = new List<Placement>();
        public int Count;
        public long PackedVolume;
    }

    // ---------------- Main ----------------
    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int W = fs.NextInt();
        int H = fs.NextInt();
        int D = fs.NextInt();

        var boxes = new List<Box>(n);
        for (int i = 0; i < n; i++)
        {
            int a = fs.NextInt();
            int b = fs.NextInt();
            int c = fs.NextInt();

            var box = new Box { Index = i, A = a, B = b, C = c };
            box.Volume = (long)a * b * c;
            box.PrecomputeOrients();

            if (CanFitAnyOrientation(box, W, H, D))
                boxes.Add(box);
        }

        if (boxes.Count == 0 || W <= 0 || H <= 0 || D <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        // Strategy selection: always try 2 fast shelf passes.
        // Additionally, for smaller N, try guillotine (more powerful but heavier).
        PackResult best = null!;

        // Shelf: small-first
        var shelf1 = ShelfPack(CloneAndSort(boxes, CompareSmallFirst), W, H, D);
        best = ChooseBetter(best, shelf1);

        // Shelf: large-first
        var shelf2 = ShelfPack(CloneAndSort(boxes, CompareLargeFirst), W, H, D);
        best = ChooseBetter(best, shelf2);

        // Guillotine for smaller instances (adaptivity)
        int m = boxes.Count;
        if (m <= 4000)
        {
            var g1 = GuillotinePack(CloneAndSort(boxes, CompareSmallFirst), W, H, D);
            best = ChooseBetter(best, g1);
        }
        if (m <= 1500)
        {
            var g2 = GuillotinePack(CloneAndSort(boxes, CompareLargeFirst), W, H, D);
            best = ChooseBetter(best, g2);
        }

        // Output
        var sb = new StringBuilder(best.Placements.Count * 20);
        sb.Append(best.Count).Append('\n');
        for (int i = 0; i < best.Placements.Count; i++)
        {
            var p = best.Placements[i];
            sb.Append(p.Index).Append(' ')
              .Append(p.X).Append(' ')
              .Append(p.Y).Append(' ')
              .Append(p.Z).Append('\n');
        }
        Console.Write(sb.ToString());
    }

    // ---------------- Helpers ----------------
    private static bool CanFitAnyOrientation(Box b, int W, int H, int D)
    {
        foreach (var o in b.Orients)
            if (o.W <= W && o.H <= H && o.D <= D) return true;
        return false;
    }

    private static List<Box> CloneAndSort(List<Box> boxes, Comparison<Box> cmp)
    {
        var arr = new List<Box>(boxes);
        arr.Sort(cmp);
        return arr;
    }

    private static int CompareSmallFirst(Box x, Box y)
    {
        // Primary: smaller volume first (maximize count tendency)
        int c = x.Volume.CompareTo(y.Volume);
        if (c != 0) return c;
        // Then smaller max side
        c = x.MaxSide.CompareTo(y.MaxSide);
        if (c != 0) return c;
        // Then smaller min side
        c = x.MinSide.CompareTo(y.MinSide);
        if (c != 0) return c;
        return x.Index.CompareTo(y.Index);
    }

    private static int CompareLargeFirst(Box x, Box y)
    {
        // Primary: larger max side first (reduces early fragmentation for shelf)
        int c = y.MaxSide.CompareTo(x.MaxSide);
        if (c != 0) return c;
        // Then larger volume
        c = y.Volume.CompareTo(x.Volume);
        if (c != 0) return c;
        return x.Index.CompareTo(y.Index);
    }

    private static PackResult ChooseBetter(PackResult? best, PackResult cand)
    {
        if (best == null) return cand;
        if (cand.Count != best.Count) return cand.Count > best.Count ? cand : best;
        if (cand.PackedVolume != best.PackedVolume) return cand.PackedVolume > best.PackedVolume ? cand : best;
        // deterministic tie-breaker: fewer placements list realloc? not needed, keep best.
        return best;
    }

    // ---------------- Strategy 1: 3D Shelf (fast, scalable) ----------------
    private static PackResult ShelfPack(List<Box> boxes, int W, int H, int D)
    {
        int x = 0, y = 0, z = 0;
        int rowHeight = 0;
        int layerDepth = 0;

        var res = new PackResult();
        res.Placements = new List<Placement>(Math.Min(boxes.Count, 16384));

        for (int i = 0; i < boxes.Count; i++)
        {
            var b = boxes[i];

            bool placed = false;
            while (!placed)
            {
                if (z >= D) break;

                int availW = W - x;
                int availH = H - y;
                int availD = D - z;

                if (availW <= 0)
                {
                    // new row
                    y += rowHeight;
                    x = 0;
                    rowHeight = 0;
                    continue;
                }
                if (availH <= 0)
                {
                    // new layer
                    z += layerDepth;
                    x = 0; y = 0;
                    rowHeight = 0;
                    layerDepth = 0;
                    continue;
                }
                if (availD <= 0) break;

                // Choose best orientation to fit current row/layer with minimal expansion
                int bestW = 0, bestH = 0, bestD = 0;
                bool found = false;

                int bestDeltaLayer = int.MaxValue;
                int bestDeltaRow = int.MaxValue;
                int bestLeftW = int.MaxValue;
                int bestLeftH = int.MaxValue;
                int bestLeftD = int.MaxValue;

                foreach (var o in b.Orients)
                {
                    if (o.W > availW) continue;

                    int newRowHeight = rowHeight;
                    int deltaRow = 0;
                    if (o.H > newRowHeight)
                    {
                        newRowHeight = o.H;
                        deltaRow = o.H - rowHeight;
                    }
                    if (newRowHeight > availH) continue;

                    int newLayerDepth = layerDepth;
                    int deltaLayer = 0;
                    if (o.D > newLayerDepth)
                    {
                        newLayerDepth = o.D;
                        deltaLayer = o.D - layerDepth;
                    }
                    if (newLayerDepth > availD) continue;

                    int leftW = availW - o.W;
                    int leftH = availH - newRowHeight;
                    int leftD = availD - newLayerDepth;

                    // Lexicographic: minimize layer expansion, then row expansion, then leftover width
                    if (!found ||
                        deltaLayer < bestDeltaLayer ||
                        (deltaLayer == bestDeltaLayer && deltaRow < bestDeltaRow) ||
                        (deltaLayer == bestDeltaLayer && deltaRow == bestDeltaRow && leftW < bestLeftW) ||
                        (deltaLayer == bestDeltaLayer && deltaRow == bestDeltaRow && leftW == bestLeftW && leftH < bestLeftH) ||
                        (deltaLayer == bestDeltaLayer && deltaRow == bestDeltaRow && leftW == bestLeftW && leftH == bestLeftH && leftD < bestLeftD))
                    {
                        found = true;
                        bestDeltaLayer = deltaLayer;
                        bestDeltaRow = deltaRow;
                        bestLeftW = leftW;
                        bestLeftH = leftH;
                        bestLeftD = leftD;
                        bestW = o.W; bestH = o.H; bestD = o.D;
                    }
                }

                if (!found)
                {
                    // Try moving to next row; if impossible, next layer
                    if (rowHeight > 0 && y + rowHeight < H)
                    {
                        y += rowHeight;
                        x = 0;
                        rowHeight = 0;
                        continue;
                    }
                    if (layerDepth > 0 && z + layerDepth < D)
                    {
                        z += layerDepth;
                        x = 0; y = 0;
                        rowHeight = 0;
                        layerDepth = 0;
                        continue;
                    }
                    // Can't advance (or empty row/layer): stop
                    break;
                }

                // Place
                res.Placements.Add(new Placement(b.Index, x, y, z));
                res.Count++;
                res.PackedVolume += (long)bestW * bestH * bestD;

                x += bestW;
                if (bestH > rowHeight) rowHeight = bestH;
                if (bestD > layerDepth) layerDepth = bestD;

                placed = true;
            }
        }

        return res;
    }

    // ---------------- Strategy 2: Guillotine Free-Spaces (better quality for small N) ----------------
    private static PackResult GuillotinePack(List<Box> boxes, int W, int H, int D)
    {
        var res = new PackResult();
        res.Placements = new List<Placement>(Math.Min(boxes.Count, 8192));

        var spaces = new List<Space>(2048) { new Space(0, 0, 0, W, H, D) };

        // Adaptive caps to keep runtime controlled
        int maxSpaces = Math.Clamp(boxes.Count * 3, 1024, 12000);

        for (int bi = 0; bi < boxes.Count; bi++)
        {
            var b = boxes[bi];

            int bestSpaceIdx = -1;
            int bestW = 0, bestH = 0, bestD = 0;

            int bestZ = int.MaxValue, bestY = int.MaxValue, bestX = int.MaxValue;
            long bestWaste = long.MaxValue;
            int bestMaxSlack = int.MaxValue;

            // Scan spaces (small N => OK)
            for (int si = 0; si < spaces.Count; si++)
            {
                var s = spaces[si];
                // position preference (bottom-left-back): skip if already worse on coordinates
                // (still allow if coordinate same but better waste)
                foreach (var o in b.Orients)
                {
                    if (o.W > s.W || o.H > s.H || o.D > s.D) continue;

                    long waste = s.Volume - (long)o.W * o.H * o.D;
                    int maxSlack = Math.Max(s.W - o.W, Math.Max(s.H - o.H, s.D - o.D));

                    bool better =
                        (s.Z < bestZ) ||
                        (s.Z == bestZ && s.Y < bestY) ||
                        (s.Z == bestZ && s.Y == bestY && s.X < bestX) ||
                        (s.Z == bestZ && s.Y == bestY && s.X == bestX && waste < bestWaste) ||
                        (s.Z == bestZ && s.Y == bestY && s.X == bestX && waste == bestWaste && maxSlack < bestMaxSlack);

                    if (better)
                    {
                        bestSpaceIdx = si;
                        bestW = o.W; bestH = o.H; bestD = o.D;
                        bestZ = s.Z; bestY = s.Y; bestX = s.X;
                        bestWaste = waste;
                        bestMaxSlack = maxSlack;
                    }
                }
            }

            if (bestSpaceIdx < 0) continue;

            // Place at chosen space origin
            var chosen = spaces[bestSpaceIdx];
            res.Placements.Add(new Placement(b.Index, chosen.X, chosen.Y, chosen.Z));
            res.Count++;
            res.PackedVolume += (long)bestW * bestH * bestD;

            // Remove chosen space (swap remove)
            int last = spaces.Count - 1;
            spaces[bestSpaceIdx] = spaces[last];
            spaces.RemoveAt(last);

            // Split chosen space into 3 disjoint spaces (XYZ pattern)
            int dx = chosen.W - bestW;
            int dy = chosen.H - bestH;
            int dz = chosen.D - bestD;

            if (dx > 0) spaces.Add(new Space(chosen.X + bestW, chosen.Y, chosen.Z, dx, chosen.H, chosen.D));
            if (dy > 0) spaces.Add(new Space(chosen.X, chosen.Y + bestH, chosen.Z, bestW, dy, chosen.D));
            if (dz > 0) spaces.Add(new Space(chosen.X, chosen.Y, chosen.Z + bestD, bestW, bestH, dz));

            // Keep spaces under control (cap + order by accessibility)
            if (spaces.Count > maxSpaces)
            {
                spaces.Sort(CompareSpacesForCap);
                if (spaces.Count > maxSpaces) spaces.RemoveRange(maxSpaces, spaces.Count - maxSpaces);
            }
        }

        return res;
    }

    private static int CompareSpacesForCap(Space a, Space b)
    {
        // Prefer smaller Z,Y,X (more "reachable"), then larger volume
        int c = a.Z.CompareTo(b.Z);
        if (c != 0) return c;
        c = a.Y.CompareTo(b.Y);
        if (c != 0) return c;
        c = a.X.CompareTo(b.X);
        if (c != 0) return c;
        // keep larger spaces first when equal coordinate
        return b.Volume.CompareTo(a.Volume);
    }
}
```
