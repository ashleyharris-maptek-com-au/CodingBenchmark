// Maximum Clique solver (hybrid exact/heuristic) for undirected graphs.
// - Exact for: small graphs via bitset branch&bound (Tomita-style coloring bound, iterative stack to avoid recursion overflow)
// - Exact for: large sparse graphs with low degeneracy via degeneracy-based branch&bound on small candidate sets
// - Fallback: if degeneracy too high (typically dense but large), runs exact on induced subgraph of top-degree vertices + heuristics
//
// Reads:
//   n m
//   u v   (0-indexed, undirected)
// Writes:
//   k
//   v1 v2 ... vk
//
// Notes:
// - Always outputs a valid clique.
// - Attempts to be exact where feasible; includes time guard (5 minutes).

using System;
using System.Buffers;
using System.Collections.Generic;
using System.Diagnostics;
using System.Numerics;
using System.Runtime.CompilerServices;
using System.Threading.Tasks;

public static class Program
{
    // --------------------------- Time control ---------------------------
    static readonly Stopwatch Sw = Stopwatch.StartNew();
    static long TimeLimitMs = 295_000; // keep some slack vs 5 minutes
    static volatile bool TimeUp = false;

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    static void CheckTime()
    {
        if (!TimeUp && Sw.ElapsedMilliseconds > TimeLimitMs) TimeUp = true;
    }

    // --------------------------- Fast input ---------------------------
    sealed class FastScanner
    {
        private readonly byte[] _buf;
        private int _len, _ptr;
        private readonly System.IO.Stream _in;

        public FastScanner(int bufSize = 1 << 20)
        {
            _buf = new byte[bufSize];
            _in = Console.OpenStandardInput();
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _in.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do { c = Read(); } while (c <= (byte)' ' && c != 0);
            int sign = 1;
            if (c == (byte)'-') { sign = -1; c = Read(); }
            int x = 0;
            while (c > (byte)' ')
            {
                x = x * 10 + (c - (byte)'0');
                c = Read();
            }
            return x * sign;
        }
    }

    // --------------------------- CSR graph ---------------------------
    static int[] Off;     // length n+1
    static int[] Adj;     // length 2m, sorted within each vertex segment
    static int[] Deg;     // length n
    static int N, M;

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    static ReadOnlySpan<int> NeighSpan(int v) => new ReadOnlySpan<int>(Adj, Off[v], Off[v + 1] - Off[v]);

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    static bool IsNeighbor(int u, int v)
    {
        // binary search v in adjacency list of u
        int lo = Off[u], hi = Off[u + 1] - 1;
        while (lo <= hi)
        {
            int mid = (lo + hi) >> 1;
            int x = Adj[mid];
            if (x == v) return true;
            if (x < v) lo = mid + 1;
            else hi = mid - 1;
        }
        return false;
    }

    // --------------------------- Global best clique ---------------------------
    static int BestSize = 0;
    static int[] BestClique = Array.Empty<int>();
    static readonly object BestLock = new object();

    static void TryUpdateBest(ReadOnlySpan<int> clique)
    {
        int k = clique.Length;
        if (k <= BestSize) return;
        lock (BestLock)
        {
            if (k <= BestSize) return;
            BestSize = k;
            BestClique = clique.ToArray();
        }
    }

    // --------------------------- Heuristic: greedy clique from a start vertex ---------------------------
    static int[] GreedyCliqueFrom(int start)
    {
        // clique = {start}, candidates = N(start), iteratively pick high-degree vertex in candidates and intersect.
        var clique = new List<int>(16) { start };

        int[] cand = null!;
        int candLen;

        // copy neighbors
        var ns = NeighSpan(start);
        candLen = ns.Length;
        if (candLen == 0) return clique.ToArray();
        cand = ArrayPool<int>.Shared.Rent(candLen);
        ns.CopyTo(cand);

        while (candLen > 0 && !TimeUp)
        {
            // pick vertex in candidates with max degree
            int best = cand[0];
            int bestDeg = Deg[best];
            for (int i = 1; i < candLen; i++)
            {
                int v = cand[i];
                int d = Deg[v];
                if (d > bestDeg)
                {
                    bestDeg = d;
                    best = v;
                }
            }

            clique.Add(best);

            // intersect cand with N(best)
            var nb = NeighSpan(best);
            int[] next = ArrayPool<int>.Shared.Rent(Math.Min(candLen, nb.Length));
            int iCand = 0, iNb = 0, outLen = 0;

            // both sorted
            while (iCand < candLen && iNb < nb.Length)
            {
                int a = cand[iCand];
                int b = nb[iNb];
                if (a == best) { iCand++; continue; } // skip itself if present
                if (a == b) { next[outLen++] = a; iCand++; iNb++; }
                else if (a < b) iCand++;
                else iNb++;
            }

            ArrayPool<int>.Shared.Return(cand);
            cand = next;
            candLen = outLen;

            // cheap bound: if cannot beat current best, can stop early
            if (clique.Count + candLen <= BestSize) break;
            CheckTime();
        }

        ArrayPool<int>.Shared.Return(cand);
        return clique.ToArray();
    }

    static void RunParallelGreedySeeded(int[] seeds)
    {
        int p = Math.Max(1, Environment.ProcessorCount);
        Parallel.For(0, seeds.Length, new ParallelOptions { MaxDegreeOfParallelism = p }, i =>
        {
            if (TimeUp) return;
            var c = GreedyCliqueFrom(seeds[i]);
            TryUpdateBest(c);
        });
    }

    // --------------------------- Top-K by degree ---------------------------
    static int[] TopKByDegree(int k)
    {
        k = Math.Min(k, N);
        if (k <= 0) return Array.Empty<int>();

        // For small N, just sort.
        if (N <= 300_000)
        {
            int[] verts = new int[N];
            for (int i = 0; i < N; i++) verts[i] = i;
            Array.Sort(verts, (a, b) => Deg[b].CompareTo(Deg[a]));
            if (verts.Length == k) return verts;
            int[] res = new int[k];
            Array.Copy(verts, res, k);
            return res;
        }

        // For large N, maintain a min-heap of size k.
        var heap = new MinHeap(k);
        for (int v = 0; v < N; v++)
        {
            heap.Push(v, Deg[v]);
        }
        return heap.ToArrayDescending();
    }

    sealed class MinHeap
    {
        private readonly int _cap;
        private int _size;
        private readonly int[] _v;
        private readonly int[] _key; // degree

        public MinHeap(int cap)
        {
            _cap = cap;
            _v = new int[cap];
            _key = new int[cap];
            _size = 0;
        }

        public void Push(int v, int key)
        {
            if (_cap == 0) return;
            if (_size < _cap)
            {
                int i = _size++;
                _v[i] = v; _key[i] = key;
                SiftUp(i);
            }
            else if (key > _key[0])
            {
                _v[0] = v; _key[0] = key;
                SiftDown(0);
            }
        }

        private void SiftUp(int i)
        {
            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (_key[p] <= _key[i]) break;
                Swap(p, i);
                i = p;
            }
        }

        private void SiftDown(int i)
        {
            while (true)
            {
                int l = i * 2 + 1;
                if (l >= _size) break;
                int r = l + 1;
                int m = (r < _size && _key[r] < _key[l]) ? r : l;
                if (_key[i] <= _key[m]) break;
                Swap(i, m);
                i = m;
            }
        }

        private void Swap(int a, int b)
        {
            ( _v[a], _v[b] ) = ( _v[b], _v[a] );
            ( _key[a], _key[b] ) = ( _key[b], _key[a] );
        }

        public int[] ToArrayDescending()
        {
            // heap contains top-k but unsorted; sort descending by key
            int[] arr = new int[_size];
            Array.Copy(_v, arr, _size);
            Array.Sort(arr, (a, b) => Deg[b].CompareTo(Deg[a]));
            return arr;
        }
    }

    // --------------------------- BitSet (pooled) ---------------------------
    sealed class PooledBitSet : IDisposable
    {
        public ulong[] W;
        public readonly int Words;
        private bool _disposed;

        public PooledBitSet(int words)
        {
            Words = words;
            W = ArrayPool<ulong>.Shared.Rent(words);
            Array.Clear(W, 0, words);
        }

        public PooledBitSet(int words, bool noClear)
        {
            Words = words;
            W = ArrayPool<ulong>.Shared.Rent(words);
            if (!noClear) Array.Clear(W, 0, words);
        }

        public void Dispose()
        {
            if (_disposed) return;
            _disposed = true;
            ArrayPool<ulong>.Shared.Return(W);
            W = Array.Empty<ulong>();
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public bool IsEmpty()
        {
            for (int i = 0; i < Words; i++) if (W[i] != 0UL) return false;
            return true;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void SetAll(int nBits)
        {
            int full = nBits >> 6;
            int rem = nBits & 63;
            for (int i = 0; i < full; i++) W[i] = ~0UL;
            if (rem != 0) W[full] = (1UL << rem) - 1UL;
            for (int i = full + (rem != 0 ? 1 : 0); i < Words; i++) W[i] = 0UL;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void ClearBit(int bit)
        {
            W[bit >> 6] &= ~(1UL << (bit & 63));
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void AndWith(ulong[] other)
        {
            for (int i = 0; i < Words; i++) W[i] &= other[i];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void AndNotWith(ulong[] other)
        {
            for (int i = 0; i < Words; i++) W[i] &= ~other[i];
        }

        public PooledBitSet Clone()
        {
            var b = new PooledBitSet(Words, noClear: true);
            Array.Copy(W, b.W, Words);
            return b;
        }

        public PooledBitSet IntersectWith(ulong[] other)
        {
            var b = new PooledBitSet(Words, noClear: true);
            for (int i = 0; i < Words; i++) b.W[i] = W[i] & other[i];
            return b;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int PopFirst()
        {
            // returns index of a set bit and clears it; assumes non-empty
            for (int wi = 0; wi < Words; wi++)
            {
                ulong x = W[wi];
                if (x == 0UL) continue;
                int tz = BitOperations.TrailingZeroCount(x);
                W[wi] = x & (x - 1);
                return (wi << 6) + tz;
            }
            return -1;
        }

        public int CountBits()
        {
            int c = 0;
            for (int i = 0; i < Words; i++) c += BitOperations.PopCount(W[i]);
            return c;
        }
    }

    // --------------------------- Bitset exact solver (iterative Tomita-style) ---------------------------
    sealed class BitsetMaxCliqueSolver
    {
        private readonly int _n;
        private readonly int _words;
        private readonly ulong[][] _adj; // per vertex bitset length _words

        // Work stacks
        private int[] _cliqueStack;

        public BitsetMaxCliqueSolver(int n, ulong[][] adjBits)
        {
            _n = n;
            _adj = adjBits;
            _words = (n + 63) >> 6;
            _cliqueStack = new int[Math.Max(64, n)]; // will be resized if needed
        }

        private struct Frame
        {
            public PooledBitSet P;          // mutable candidate set for this frame
            public int[] Order;             // vertices in coloring order
            public int[] Color;             // color bound per Order index
            public int Count;               // number of entries
            public int Index;               // current index (descending)
            public bool HasColoring;        // whether Order/Color allocated
        }

        public void Solve()
        {
            // root P = all vertices
            using var rootP = new PooledBitSet(_words);
            rootP.SetAll(_n);

            // iterative DFS stack
            Frame[] stack = new Frame[128];
            int sp = 0;

            // current clique size tracked by sp-1? no: separate
            int cSize = 0;

            // push root frame (clone rootP as owned by frame)
            stack[sp++] = new Frame
            {
                P = rootP.Clone(),
                Order = Array.Empty<int>(),
                Color = Array.Empty<int>(),
                Count = 0,
                Index = -1,
                HasColoring = false
            };

            while (sp > 0 && !TimeUp)
            {
                CheckTime();
                ref Frame f = ref stack[sp - 1];

                // If just entered frame, handle base case / coloring creation
                if (!f.HasColoring)
                {
                    if (f.P.IsEmpty())
                    {
                        // current clique is maximal in this branch
                        if (cSize > BestSize)
                        {
                            EnsureCliqueStackCapacity(cSize);
                            TryUpdateBest(new ReadOnlySpan<int>(_cliqueStack, 0, cSize));
                        }

                        // pop frame
                        f.P.Dispose();
                        sp--;
                        if (sp > 0) cSize--; // returning from child => remove last chosen vertex
                        continue;
                    }

                    // pruning with trivial bound
                    int pCount = f.P.CountBits();
                    if (cSize + pCount <= BestSize)
                    {
                        f.P.Dispose();
                        sp--;
                        if (sp > 0) cSize--;
                        continue;
                    }

                    // compute greedy coloring bound
                    CreateColoring(ref f);
                    f.Index = f.Count - 1;
                    f.HasColoring = true;
                    continue;
                }

                // main loop: pick next vertex to branch on
                if (f.Index < 0)
                {
                    // finished this frame
                    if (f.Order.Length != 0) ArrayPool<int>.Shared.Return(f.Order);
                    if (f.Color.Length != 0) ArrayPool<int>.Shared.Return(f.Color);
                    f.P.Dispose();
                    sp--;
                    if (sp > 0) cSize--;
                    continue;
                }

                int idx = f.Index;
                int v = f.Order[idx];
                int bound = f.Color[idx];
                f.Index--;

                // upper bound check (Tomita): if current clique + bound <= best => stop exploring remaining
                if (cSize + bound <= BestSize)
                {
                    f.Index = -1; // force pop
                    continue;
                }

                // Branch include v: C := C ∪ {v}, P' := P ∩ N(v)
                // Also remove v from current frame P for subsequent branches
                f.P.ClearBit(v);
                var childP = f.P.IntersectWith(_adj[v]);

                // push child frame
                EnsureCliqueStackCapacity(cSize + 1);
                _cliqueStack[cSize] = v;
                cSize++;

                if (sp >= stack.Length) Array.Resize(ref stack, stack.Length * 2);
                stack[sp++] = new Frame
                {
                    P = childP,
                    Order = Array.Empty<int>(),
                    Color = Array.Empty<int>(),
                    Count = 0,
                    Index = -1,
                    HasColoring = false
                };
            }

            // cleanup any remaining frames if time up
            while (sp > 0)
            {
                ref Frame f = ref stack[--sp];
                if (f.HasColoring)
                {
                    if (f.Order.Length != 0) ArrayPool<int>.Shared.Return(f.Order);
                    if (f.Color.Length != 0) ArrayPool<int>.Shared.Return(f.Color);
                }
                f.P.Dispose();
            }
        }

        private void EnsureCliqueStackCapacity(int needed)
        {
            if (needed <= _cliqueStack.Length) return;
            int newCap = _cliqueStack.Length;
            while (newCap < needed) newCap *= 2;
            Array.Resize(ref _cliqueStack, newCap);
        }

        private void CreateColoring(ref Frame f)
        {
            // Greedy coloring using bitsets:
            // Repeatedly build independent sets (colors) from remaining vertices.
            int k = f.P.CountBits();
            int[] order = ArrayPool<int>.Shared.Rent(k);
            int[] color = ArrayPool<int>.Shared.Rent(k);
            int outIdx = 0;

            using var pCopy = f.P.Clone();
            int col = 0;

            while (!pCopy.IsEmpty() && !TimeUp)
            {
                CheckTime();
                col++;

                using var q = pCopy.Clone(); // vertices available for this color (will be reduced by neighbors)
                while (!q.IsEmpty())
                {
                    int v = q.PopFirst();
                    order[outIdx] = v;
                    color[outIdx] = col;
                    outIdx++;

                    pCopy.ClearBit(v);
                    q.AndNotWith(_adj[v]);
                }
            }

            f.Order = order;
            f.Color = color;
            f.Count = outIdx;
        }
    }

    // --------------------------- Build adjacency bitsets for small graphs ---------------------------
    static ulong[][] BuildAdjBitsets(int n)
    {
        int words = (n + 63) >> 6;
        var adjBits = new ulong[n][];
        for (int i = 0; i < n; i++) adjBits[i] = new ulong[words];

        for (int u = 0; u < n; u++)
        {
            var ns = NeighSpan(u);
            foreach (int v in ns)
            {
                adjBits[u][v >> 6] |= 1UL << (v & 63);
            }
        }
        return adjBits;
    }

    // --------------------------- Degeneracy ordering (O(n+m)) ---------------------------
    static (int[] order, int degeneracy) DegeneracyOrder()
    {
        int n = N;
        int[] deg = new int[n];
        Array.Copy(Deg, deg, n);

        int maxDeg = 0;
        for (int i = 0; i < n; i++) if (deg[i] > maxDeg) maxDeg = deg[i];

        int[] head = new int[maxDeg + 1];
        Array.Fill(head, -1);
        int[] next = new int[n];
        int[] prev = new int[n];
        Array.Fill(next, -1);
        Array.Fill(prev, -1);

        void BucketInsert(int v, int d)
        {
            int h = head[d];
            next[v] = h;
            prev[v] = -1;
            if (h != -1) prev[h] = v;
            head[d] = v;
        }

        void BucketRemove(int v, int d)
        {
            int p = prev[v];
            int nx = next[v];
            if (p != -1) next[p] = nx;
            else head[d] = nx;
            if (nx != -1) prev[nx] = p;
            prev[v] = next[v] = -1;
        }

        for (int v = 0; v < n; v++) BucketInsert(v, deg[v]);

        int[] order = new int[n];
        bool[] removed = new bool[n];
        int minD = 0;
        int degeneracy = 0;

        for (int idx = 0; idx < n; idx++)
        {
            while (minD < head.Length && head[minD] == -1) minD++;
            if (minD >= head.Length) break;

            int v = head[minD];
            BucketRemove(v, minD);
            removed[v] = true;
            order[idx] = v;
            if (minD > degeneracy) degeneracy = minD;

            // decrease degrees of remaining neighbors
            var ns = NeighSpan(v);
            foreach (int u in ns)
            {
                if (removed[u]) continue;
                int du = deg[u];
                BucketRemove(u, du);
                deg[u] = du - 1;
                BucketInsert(u, du - 1);
            }

            CheckTime();
            if (TimeUp) break;
        }

        return (order, degeneracy);
    }

    // --------------------------- Degeneracy-based exact search for large sparse graphs ---------------------------
    // For each vertex v in degeneracy order, search maximum clique in subgraph induced by {v} ∪ N+(v)
    // Candidate sets are size <= degeneracy, so recursion depth is bounded and manageable.
    static void SolveByDegeneracyExact(int[] order)
    {
        int n = N;
        int[] pos = new int[n];
        for (int i = 0; i < n; i++) pos[order[i]] = i;

        // Estimate degeneracy upper bound for stack sizing: maximum |N+(v)| across v
        int maxLater = 0;
        for (int i = 0; i < n; i++)
        {
            int v = order[i];
            int cnt = 0;
            var ns = NeighSpan(v);
            foreach (int u in ns) if (pos[u] > i) cnt++;
            if (cnt > maxLater) maxLater = cnt;
        }
        int maxDepth = maxLater + 1;
        int[] cliqueStack = new int[Math.Max(4, maxDepth + 2)];

        void Expand(int cSize, int[] P, int pLen)
        {
            if (TimeUp) return;
            if (pLen == 0)
            {
                if (cSize > BestSize) TryUpdateBest(new ReadOnlySpan<int>(cliqueStack, 0, cSize));
                return;
            }

            if (cSize + pLen <= BestSize) return;

            // choose pivot in P maximizing neighbors in P
            int pivot = P[0];
            int pivotMax = -1;
            for (int i = 0; i < pLen; i++)
            {
                int u = P[i];
                int cnt = 0;
                for (int j = 0; j < pLen; j++)
                    if (j != i && IsNeighbor(u, P[j])) cnt++;
                if (cnt > pivotMax) { pivotMax = cnt; pivot = u; }
            }

            // Iterate over vertices in P \ N(pivot)
            int iIdx = 0;
            while (iIdx < pLen)
            {
                CheckTime();
                if (TimeUp) return;

                int v = P[iIdx];
                if (IsNeighbor(pivot, v))
                {
                    iIdx++;
                    continue;
                }

                // Build P2 = P ∩ N(v), excluding v itself
                int[] P2 = ArrayPool<int>.Shared.Rent(pLen);
                int p2Len = 0;
                for (int j = 0; j < pLen; j++)
                {
                    int w = P[j];
                    if (w == v) continue;
                    if (IsNeighbor(v, w)) P2[p2Len++] = w;
                }

                cliqueStack[cSize] = v;
                Expand(cSize + 1, P2, p2Len);

                ArrayPool<int>.Shared.Return(P2);

                // Remove v from P (swap with last)
                pLen--;
                P[iIdx] = P[pLen];
                // do not increment iIdx; need to process swapped-in element
            }
        }

        for (int i = 0; i < n && !TimeUp; i++)
        {
            CheckTime();
            int v = order[i];

            // Build P = N+(v)
            var ns = NeighSpan(v);
            int laterCount = 0;
            foreach (int u in ns) if (pos[u] > i) laterCount++;

            if (1 + laterCount <= BestSize) continue;
            int[] P = ArrayPool<int>.Shared.Rent(laterCount);
            int pLen = 0;
            foreach (int u in ns) if (pos[u] > i) P[pLen++] = u;

            cliqueStack[0] = v;
            Expand(1, P, pLen);

            ArrayPool<int>.Shared.Return(P);
        }
    }

    // --------------------------- Fallback exact-on-induced-subgraph of top-degree vertices ---------------------------
    static void SolveOnInducedTopK(int k)
    {
        k = Math.Min(k, N);
        if (k <= BestSize) return;

        int[] subset = TopKByDegree(k);
        int[] map = new int[N];
        Array.Fill(map, -1);
        for (int i = 0; i < subset.Length; i++) map[subset[i]] = i;

        int n2 = subset.Length;
        int words = (n2 + 63) >> 6;
        ulong[][] adj2 = new ulong[n2][];
        for (int i = 0; i < n2; i++) adj2[i] = new ulong[words];

        // Build bitset adjacency for induced subgraph
        for (int i = 0; i < n2; i++)
        {
            int u = subset[i];
            var ns = NeighSpan(u);
            foreach (int v in ns)
            {
                int j = map[v];
                if (j >= 0)
                {
                    adj2[i][j >> 6] |= 1UL << (j & 63);
                }
            }
            CheckTime();
            if (TimeUp) break;
        }

        // Run exact bitset solver on induced subgraph
        var solver = new BitsetMaxCliqueSolver(n2, adj2);
        solver.Solve();

        // Translate best clique if it improved (it will have indices in [0..n2))
        // However global BestClique may have been improved by other methods; we only translate if needed by checking validity.
        // If the best clique currently consists entirely of vertices < n2, it might already be from this induced run.
        // To be safe: if BestClique contains any < n2, it's ambiguous; we don't attempt reverse mapping.
        // Instead we attempt to rebuild a clique found by greedy restricted to subset (always valid) and update if bigger.
        // Practical: the induced solver itself updated global best only with indices (wrong IDs), so we must avoid that.
        // Therefore: do not allow induced solver to write global best directly.
        // => We run induced solver in an isolated best tracker.
        //
        // Implement isolated run:
        // (This method is kept simple: re-run solver with a local capture by temporarily swapping globals.)

        // Re-run with isolated best:
        int savedBestSize = BestSize;
        int[] savedBestClique = BestClique;

        BestSize = 0;
        BestClique = Array.Empty<int>();
        try
        {
            var solver2 = new BitsetMaxCliqueSolver(n2, adj2);
            solver2.Solve();
            int[] bestLocal = BestClique;
            int bestLocalSize = BestSize;

            // restore globals, then map and update
            BestSize = savedBestSize;
            BestClique = savedBestClique;

            if (bestLocalSize > BestSize)
            {
                int[] mapped = new int[bestLocalSize];
                for (int i = 0; i < bestLocalSize; i++) mapped[i] = subset[bestLocal[i]];
                TryUpdateBest(mapped);
            }
        }
        finally
        {
            // ensure globals restored if something goes wrong
            BestSize = savedBestSize;
            BestClique = savedBestClique;
        }
    }

    // --------------------------- Main ---------------------------
    public static void Main()
    {
        var fs = new FastScanner();

        N = fs.NextInt();
        M = fs.NextInt();

        if (N <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        // Quick complete-graph upper bound check needs 64-bit
        long maxEdges = (long)N * (N - 1) / 2;

        Deg = new int[N];

        // Read edges into temporary arrays for CSR build
        int[] eu = new int[M];
        int[] ev = new int[M];

        for (int i = 0; i < M; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if ((uint)u >= (uint)N || (uint)v >= (uint)N || u == v)
            {
                // ignore invalid/self-loops
                eu[i] = 0; ev[i] = 0;
                continue;
            }
            eu[i] = u;
            ev[i] = v;
            Deg[u]++;
            Deg[v]++;
        }

        // Complete graph: immediate answer
        if ((long)M == maxEdges)
        {
            Console.WriteLine(N);
            for (int i = 0; i < N; i++)
            {
                if (i != 0) Console.Write(' ');
                Console.Write(i);
            }
            Console.WriteLine();
            return;
        }

        // Build CSR
        Off = new int[N + 1];
        Off[0] = 0;
        for (int i = 0; i < N; i++) Off[i + 1] = Off[i] + Deg[i];
        Adj = new int[Off[N]];

        int[] cur = new int[N];
        Array.Copy(Off, cur, N);

        for (int i = 0; i < M; i++)
        {
            int u = eu[i], v = ev[i];
            if (u == v) continue;
            Adj[cur[u]++] = v;
            Adj[cur[v]++] = u;
        }

        // Sort adjacency lists
        for (int v = 0; v < N; v++)
        {
            int start = Off[v];
            int len = Off[v + 1] - start;
            if (len > 1) Array.Sort(Adj, start, len);
        }

        // Initial best: at least one vertex
        BestSize = 1;
        BestClique = new[] { 0 };

        // Heuristic lower bound (parallel greedy from top-degree seeds)
        {
            int seedCount = Math.Min(N, 256 * Math.Max(1, Environment.ProcessorCount));
            int[] seeds = TopKByDegree(seedCount);
            RunParallelGreedySeeded(seeds);
        }

        // Choose exact strategy
        const int BitsetNThreshold = 5000;       // bitset solver is strong for dense/small; keep small to avoid memory and deep search
        const int DegeneracyThreshold = 250;     // exact degeneracy-based is feasible if degeneracy is small

        if (!TimeUp && N <= BitsetNThreshold)
        {
            // Exact bitset solver on full graph
            var adjBits = BuildAdjBitsets(N);
            var solver = new BitsetMaxCliqueSolver(N, adjBits);
            solver.Solve();
        }
        else if (!TimeUp)
        {
            // Degeneracy ordering + exact if degeneracy small
            var (order, degeneracy) = DegeneracyOrder();

            if (!TimeUp && degeneracy <= DegeneracyThreshold)
            {
                SolveByDegeneracyExact(order);
            }
            else
            {
                // Fallback: exact on induced top-K (degree) subgraph + more greedy
                // (common when graph is not extremely sparse but input is too large for bitsets)
                SolveOnInducedTopK(Math.Min(8000, N));

                // One more heuristic pass from updated best seeds if time remains
                if (!TimeUp)
                {
                    int seedCount = Math.Min(N, 512 * Math.Max(1, Environment.ProcessorCount));
                    int[] seeds = TopKByDegree(seedCount);
                    RunParallelGreedySeeded(seeds);
                }
            }
        }

        // Ensure output clique is valid (constructively it should be; if something weird happened, repair by greedy filtering)
        if (BestClique.Length > 1 && !IsClique(BestClique))
        {
            BestClique = RepairToClique(BestClique);
            BestSize = BestClique.Length;
        }

        // Output
        Console.WriteLine(BestClique.Length);
        if (BestClique.Length == 0)
        {
            Console.WriteLine();
        }
        else
        {
            for (int i = 0; i < BestClique.Length; i++)
            {
                if (i != 0) Console.Write(' ');
                Console.Write(BestClique[i]);
            }
            Console.WriteLine();
        }
    }

    static bool IsClique(int[] verts)
    {
        for (int i = 0; i < verts.Length; i++)
            for (int j = i + 1; j < verts.Length; j++)
                if (!IsNeighbor(verts[i], verts[j])) return false;
        return true;
    }

    static int[] RepairToClique(int[] verts)
    {
        // Greedily keep vertices that connect to all already kept ones.
        // Sort by descending degree to maximize retained size.
        Array.Sort(verts, (a, b) => Deg[b].CompareTo(Deg[a]));
        var kept = new List<int>(verts.Length);
        foreach (var v in verts)
        {
            bool ok = true;
            for (int i = 0; i < kept.Count; i++)
            {
                if (!IsNeighbor(v, kept[i])) { ok = false; break; }
            }
            if (ok) kept.Add(v);
        }
        return kept.ToArray();
    }
