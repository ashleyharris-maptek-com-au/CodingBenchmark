#include <iostream>
#include <vector>
#include <cmath>
#include <cstdint>
#include <unordered_map>
#include <algorithm>
#include <iomanip>
#include <limits>

using std::vector;

struct Vec3 {
    double x=0, y=0, z=0;
    Vec3() = default;
    Vec3(double X,double Y,double Z):x(X),y(Y),z(Z){}
    Vec3 operator+(const Vec3& o) const { return {x+o.x,y+o.y,z+o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x-o.x,y-o.y,z-o.z}; }
    Vec3 operator*(double s) const { return {x*s,y*s,z*s}; }
    Vec3 operator/(double s) const { return {x/s,y/s,z/s}; }
};
static inline double dot(const Vec3& a, const Vec3& b){ return a.x*b.x + a.y*b.y + a.z*b.z; }
static inline Vec3 cross(const Vec3& a, const Vec3& b){
    return {a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x};
}
static inline double norm2(const Vec3& v){ return dot(v,v); }
static inline double norm(const Vec3& v){ return std::sqrt(norm2(v)); }

struct Face {
    int a=-1,b=-1,c=-1;
    Vec3 n;          // unit outward normal
    double off=0.0;  // plane: dot(n, x) = off, inside is dot(n,x) <= off
    bool deleted=false;
    vector<int> outside; // indices of points outside this face
    int farIdx=-1;
    double farDist=0.0;
};

static inline uint64_t edgeKey(int u, int v){
    return (uint64_t)(uint32_t)u << 32 | (uint32_t)v;
}

static inline bool nearlyEqual(const Vec3& a, const Vec3& b, double eps){
    return std::abs(a.x-b.x)<=eps && std::abs(a.y-b.y)<=eps && std::abs(a.z-b.z)<=eps;
}

static inline double signedDistance(const Face& f, const Vec3& p){
    return dot(f.n, p) - f.off;
}

static bool makeFace(Face &f, int a, int b, int c, const vector<Vec3>& pts, const Vec3& interior, double eps)
{
    f.a=a; f.b=b; f.c=c;
    Vec3 A = pts[a], B = pts[b], C = pts[c];
    Vec3 nn = cross(B-A, C-A);
    double len = norm(nn);
    if (len <= eps) return false;
    Vec3 nunit = nn / len;
    double off = dot(nunit, A);

    // Ensure interior point is on the inside side (<=)
    if (dot(nunit, interior) > off + 10*eps) {
        std::swap(f.b, f.c);
        nn = nn * -1.0;
        nunit = nunit * -1.0;
        off = dot(nunit, pts[f.a]);
    }
    f.n = nunit;
    f.off = off;
    f.deleted = false;
    f.outside.clear();
    f.farIdx = -1;
    f.farDist = 0.0;
    return true;
}

// Very standard QuickHull-style convex hull triangulation for 3D points (convex hull).
// Returns outward oriented triangular faces for the convex hull.
// If degenerate (coplanar/collinear), returns empty.
static vector<Face> convexHull3D(const vector<Vec3>& pts)
{
    const double EPS = 1e-12;

    int n = (int)pts.size();
    if (n < 4) return {};

    // 1) Build initial simplex (tetrahedron)
    int iMinX=0, iMaxX=0;
    for(int i=1;i<n;i++){
        if (pts[i].x < pts[iMinX].x) iMinX=i;
        if (pts[i].x > pts[iMaxX].x) iMaxX=i;
    }
    if (iMinX == iMaxX) return {};

    // farthest from line (iMinX - iMaxX)
    Vec3 A = pts[iMinX], B = pts[iMaxX];
    Vec3 AB = B - A;
    double ab2 = norm2(AB);
    int iFarLine = -1;
    double bestLine = -1.0;
    for(int i=0;i<n;i++){
        if (i==iMinX || i==iMaxX) continue;
        Vec3 AP = pts[i] - A;
        Vec3 cr = cross(AB, AP);
        double d2 = norm2(cr) / (ab2 > EPS ? ab2 : 1.0);
        if (d2 > bestLine){
            bestLine = d2;
            iFarLine = i;
        }
    }
    if (iFarLine < 0 || bestLine <= EPS) return {};

    // farthest from plane (iMinX, iMaxX, iFarLine)
    int iFarPlane = -1;
    double bestPlane = -1.0;
    Vec3 C = pts[iFarLine];
    Vec3 n0 = cross(B-A, C-A);
    double n0len = norm(n0);
    if (n0len <= EPS) return {};
    Vec3 n0u = n0 / n0len;
    double off0 = dot(n0u, A);
    for(int i=0;i<n;i++){
        if (i==iMinX || i==iMaxX || i==iFarLine) continue;
        double d = std::abs(dot(n0u, pts[i]) - off0);
        if (d > bestPlane){
            bestPlane = d;
            iFarPlane = i;
        }
    }
    if (iFarPlane < 0 || bestPlane <= 1e-10) return {}; // coplanar

    int i0=iMinX, i1=iMaxX, i2=iFarLine, i3=iFarPlane;

    // interior point = centroid of initial tetra
    Vec3 interior = (pts[i0] + pts[i1] + pts[i2] + pts[i3]) / 4.0;

    // 2) Create 4 faces of initial tetra (oriented outward)
    vector<Face> faces;
    faces.reserve(2*n);

    auto addFace = [&](int a,int b,int c)->int{
        Face f;
        if(!makeFace(f,a,b,c,pts,interior,1e-14)) return -1;
        faces.push_back(std::move(f));
        return (int)faces.size()-1;
    };

    // Tetra faces; orientation will be corrected using interior point
    int f0 = addFace(i0,i1,i2);
    int f1 = addFace(i0,i3,i1);
    int f2 = addFace(i0,i2,i3);
    int f3 = addFace(i1,i3,i2);
    if (f0<0 || f1<0 || f2<0 || f3<0) return {};

    // Helper: rebuild edge->face map for active faces
    auto rebuildEdgeMap = [&](std::unordered_map<uint64_t,int>& edge2face){
        edge2face.clear();
        edge2face.reserve(faces.size()*3);
        for(int fi=0; fi<(int)faces.size(); ++fi){
            if (faces[fi].deleted) continue;
            const Face& f = faces[fi];
            edge2face[edgeKey(f.a,f.b)] = fi;
            edge2face[edgeKey(f.b,f.c)] = fi;
            edge2face[edgeKey(f.c,f.a)] = fi;
        }
    };

    // 3) Assign points to faces' outside lists
    vector<char> isSimplex(n, 0);
    isSimplex[i0]=isSimplex[i1]=isSimplex[i2]=isSimplex[i3]=1;

    for(int pi=0; pi<n; ++pi){
        if(isSimplex[pi]) continue;
        double maxD = 0.0;
        int bestF = -1;
        for(int fi=0; fi<(int)faces.size(); ++fi){
            if (faces[fi].deleted) continue;
            double d = signedDistance(faces[fi], pts[pi]);
            if (d > 1e-12 && d > maxD){
                maxD = d; bestF = fi;
            }
        }
        if(bestF >= 0){
            faces[bestF].outside.push_back(pi);
            if (maxD > faces[bestF].farDist){
                faces[bestF].farDist = maxD;
                faces[bestF].farIdx = pi;
            }
        }
    }

    // 4) QuickHull expansion
    std::unordered_map<uint64_t,int> edge2face;
    rebuildEdgeMap(edge2face);

    while(true){
        // Pick a face with outside points; heuristic: largest farDist
        int seedFace = -1;
        double best = 0.0;
        for(int fi=0; fi<(int)faces.size(); ++fi){
            if (faces[fi].deleted) continue;
            if (faces[fi].outside.empty()) continue;
            if (faces[fi].farIdx < 0) continue;
            if (faces[fi].farDist > best){
                best = faces[fi].farDist;
                seedFace = fi;
            }
        }
        if(seedFace < 0) break;

        int pIdx = faces[seedFace].farIdx;
        Vec3 P = pts[pIdx];

        // Find all visible faces from P
        vector<char> vis(faces.size(), 0);
        vector<int> visibleFaces;
        visibleFaces.reserve(64);

        for(int fi=0; fi<(int)faces.size(); ++fi){
            if (faces[fi].deleted) continue;
            if (signedDistance(faces[fi], P) > 1e-12){
                vis[fi] = 1;
                visibleFaces.push_back(fi);
            }
        }
        if (visibleFaces.empty()){
            // should not happen but avoid infinite loop
            faces[seedFace].outside.clear();
            faces[seedFace].farIdx = -1;
            faces[seedFace].farDist = 0.0;
            continue;
        }

        // Collect horizon edges (as directed edges) and collect points to reassign
        vector<std::pair<int,int>> horizon;
        horizon.reserve(visibleFaces.size()*2);

        vector<int> toReassign;
        toReassign.reserve(1024);

        for(int fi: visibleFaces){
            Face& f = faces[fi];

            // collect outside points to reassign (excluding pIdx)
            for(int q : f.outside){
                if (q != pIdx) toReassign.push_back(q);
            }

            // Horizon detection via adjacency using edge map:
            // For each edge (u->v) of visible face, if adjacent across (v->u) is not visible, it's horizon.
            int u0=f.a, u1=f.b, u2=f.c;
            int eu[3] = {u0,u1,u2};
            int ev[3] = {u1,u2,u0};
            for(int e=0;e<3;e++){
                int u = eu[e], v = ev[e];
                auto it = edge2face.find(edgeKey(v,u)); // reverse edge
                int adj = (it==edge2face.end()) ? -1 : it->second;
                if (adj < 0 || faces[adj].deleted || !vis[adj]){
                    // store reversed edge so (u,v,p) orientation can be fixed with interior point
                    horizon.emplace_back(v,u);
                }
            }

            f.deleted = true;
            f.outside.clear();
            f.farIdx = -1;
            f.farDist = 0.0;
        }

        // Rebuild edge map after deletions (simpler/robust)
        rebuildEdgeMap(edge2face);

        // Create new faces from horizon edges to pIdx
        vector<int> newFaceIds;
        newFaceIds.reserve(horizon.size());
        for(auto [u,v] : horizon){
            Face nf;
            if(!makeFace(nf, u, v, pIdx, pts, interior, 1e-14)) continue;
            faces.push_back(std::move(nf));
            newFaceIds.push_back((int)faces.size()-1);
        }

        // Rebuild edge map to include new faces
        rebuildEdgeMap(edge2face);

        // Reassign points from removed faces to the new faces
        // (only consider new faces to reduce work)
        for(int qIdx : toReassign){
            double maxD = 0.0;
            int bestF = -1;
            const Vec3& Q = pts[qIdx];
            for(int fi : newFaceIds){
                if (faces[fi].deleted) continue;
                double d = signedDistance(faces[fi], Q);
                if (d > 1e-12 && d > maxD){
                    maxD = d;
                    bestF = fi;
                }
            }
            if(bestF >= 0){
                Face& bf = faces[bestF];
                bf.outside.push_back(qIdx);
                if (maxD > bf.farDist){
                    bf.farDist = maxD;
                    bf.farIdx = qIdx;
                }
            }
        }
    }

    // Extract active faces
    vector<Face> out;
    out.reserve(faces.size());
    for(auto &f : faces){
        if (!f.deleted) out.push_back(std::move(f));
    }
    return out;
}

static double polyVolumeFromFaces(const vector<Vec3>& pts, const vector<Face>& faces, const Vec3& interior)
{
    long double V = 0.0L;
    for(const auto& f : faces){
        const Vec3& a = pts[f.a];
        const Vec3& b = pts[f.b];
        const Vec3& c = pts[f.c];
        Vec3 A = a - interior;
        Vec3 B = b - interior;
        Vec3 C = c - interior;
        long double v6 = (long double)dot(A, cross(B, C));
        V += v6 / 6.0L;
    }
    return (double)std::abs((double)V);
}

int main(){
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int V;
    double edgeLen;
    if(!(std::cin >> V >> edgeLen)){
        return 0;
    }
    vector<Vec3> inPts;
    inPts.reserve(std::max(0, V));
    for(int i=0;i<V;i++){
        double x,y,z;
        std::cin >> x >> y >> z;
        inPts.emplace_back(x,y,z);
    }
    if ((int)inPts.size() < 4 || edgeLen <= 0){
        std::cout << 0 << "\n";
        return 0;
    }

    // Remove exact duplicates (helps hull stability). Use tiny epsilon exact-ish.
    const double DE = 0.0; // exact duplicates only
    std::sort(inPts.begin(), inPts.end(), [](const Vec3& a, const Vec3& b){
        if (a.x != b.x) return a.x < b.x;
        if (a.y != b.y) return a.y < b.y;
        return a.z < b.z;
    });
    vector<Vec3> pts;
    pts.reserve(inPts.size());
    for(const auto& p : inPts){
        if (pts.empty() || !nearlyEqual(pts.back(), p, DE)) pts.push_back(p);
    }

    // Build convex hull
    vector<Face> faces = convexHull3D(pts);
    if (faces.empty()){
        std::cout << 0 << "\n";
        return 0;
    }

    // An interior point: average of vertices (should lie inside for convex hull if vertices well-spread;
    // but for safety, compute average of all points, then it's inside convex hull for convex combination).
    Vec3 interior(0,0,0);
    for(const auto& p: pts) interior = interior + p;
    interior = interior / (double)pts.size();

    // If volume ~ 0, degenerate
    double vol = polyVolumeFromFaces(pts, faces, interior);
    if (!(vol > 1e-15)){
        std::cout << 0 << "\n";
        return 0;
    }

    // Regular tetrahedron circumradius: R = e*sqrt(3/8)
    const double R = edgeLen * std::sqrt(3.0/8.0);

    // Bounding box of container
    Vec3 mn(+std::numeric_limits<double>::infinity(),
            +std::numeric_limits<double>::infinity(),
            +std::numeric_limits<double>::infinity());
    Vec3 mx(-std::numeric_limits<double>::infinity(),
            -std::numeric_limits<double>::infinity(),
            -std::numeric_limits<double>::infinity());
    for(const auto& p : pts){
        mn.x = std::min(mn.x, p.x); mn.y = std::min(mn.y, p.y); mn.z = std::min(mn.z, p.z);
        mx.x = std::max(mx.x, p.x); mx.y = std::max(mx.y, p.y); mx.z = std::max(mx.z, p.z);
    }

    // If shrunken bbox invalid, no placement
    Vec3 smn(mn.x + R, mn.y + R, mn.z + R);
    Vec3 smx(mx.x - R, mx.y - R, mx.z - R);
    if (!(smn.x <= smx.x && smn.y <= smx.y && smn.z <= smx.z)){
        std::cout << 0 << "\n";
        return 0;
    }

    // FCC lattice parameters for sphere centers of radius R:
    // conventional cubic cell side a0 = 2*sqrt(2)*R with 4 points per cell.
    const double a0 = 2.0 * std::sqrt(2.0) * R;

    // Adaptive scaling to keep runtime/output bounded on huge containers.
    // Note: This sacrifices optimality at extreme scales but remains correct and fast.
    const std::size_t CAP_POINTS = 300000;  // keep output manageable
    double estPoints = vol * 4.0 / (a0*a0*a0);
    double scale = 1.0;
    if (estPoints > (double)CAP_POINTS){
        scale = std::cbrt(estPoints / (double)CAP_POINTS);
        if (scale < 1.0) scale = 1.0;
    }
    double a = a0 * scale;

    // Also guard against massive triple loops even if volume estimate is misleading
    auto safeCeil = [](double x)->int64_t{
        return (int64_t)std::ceil(x);
    };
    auto safeFloor = [](double x)->int64_t{
        return (int64_t)std::floor(x);
    };

    // FCC basis offsets
    const Vec3 basis[4] = {
        {0,0,0},
        {0,a*0.5,a*0.5},
        {a*0.5,0,a*0.5},
        {a*0.5,a*0.5,0}
    };

    // Determine index ranges (pad a bit for safety due to basis offsets)
    int64_t ix0 = safeFloor(smn.x / a) - 2, ix1 = safeCeil(smx.x / a) + 2;
    int64_t iy0 = safeFloor(smn.y / a) - 2, iy1 = safeCeil(smx.y / a) + 2;
    int64_t iz0 = safeFloor(smn.z / a) - 2, iz1 = safeCeil(smx.z / a) + 2;

    // If still enormous, increase scale more
    long double nx = (long double)(ix1 - ix0 + 1);
    long double ny = (long double)(iy1 - iy0 + 1);
    long double nz = (long double)(iz1 - iz0 + 1);
    long double estIters = nx*ny*nz;
    const long double MAX_ITERS = 6e7L; // about 60 million cells => 240 million FCC points checks worst-case
    if (estIters > MAX_ITERS && estIters > 0){
        double extra = std::cbrt((double)(estIters / MAX_ITERS));
        if (extra > 1.0){
            scale *= extra;
            a = a0 * scale;
        }
        // recompute ranges
        ix0 = safeFloor(smn.x / a) - 2; ix1 = safeCeil(smx.x / a) + 2;
        iy0 = safeFloor(smn.y / a) - 2; iy1 = safeCeil(smx.y / a) + 2;
        iz0 = safeFloor(smn.z / a) - 2; iz1 = safeCeil(smx.z / a) + 2;
    }

    // Precompute face planes for shrunken-inside test: dot(n, p) <= off - R
    // (since if the circumsphere is inside, the tetrahedron is inside).
    const double EPS_IN = 1e-10;
    struct Plane { Vec3 n; double offShrink; };
    vector<Plane> planes;
    planes.reserve(faces.size());
    for(const auto& f : faces){
        planes.push_back({f.n, f.off - R});
    }

    auto insideShrunken = [&](const Vec3& p)->bool{
        // AABB early reject
        if (p.x < smn.x-EPS_IN || p.x > smx.x+EPS_IN) return false;
        if (p.y < smn.y-EPS_IN || p.y > smx.y+EPS_IN) return false;
        if (p.z < smn.z-EPS_IN || p.z > smx.z+EPS_IN) return false;
        for(const auto& pl : planes){
            if (dot(pl.n, p) > pl.offShrink + EPS_IN) return false;
        }
        return true;
    };

    // Generate placements (tetrahedra centered at FCC lattice points; fixed orientation)
    struct Pose { Vec3 c; };
    vector<Pose> poses;
    poses.reserve(std::min<std::size_t>(CAP_POINTS, 10000));

    for(int64_t iz=iz0; iz<=iz1; ++iz){
        double z0 = (double)iz * a;
        for(int64_t iy=iy0; iy<=iy1; ++iy){
            double y0 = (double)iy * a;
            for(int64_t ix=ix0; ix<=ix1; ++ix){
                double x0 = (double)ix * a;
                Vec3 cell(x0,y0,z0);
                for(int bi=0; bi<4; ++bi){
                    Vec3 p = cell + basis[bi];
                    if (insideShrunken(p)){
                        poses.push_back({p});
                        if (poses.size() >= CAP_POINTS) goto done;
                    }
                }
            }
        }
    }
done:

    std::cout.setf(std::ios::fixed);
    std::cout << poses.size() << "\n";
    std::cout << std::setprecision(10);
    for(const auto& ps : poses){
        // cx cy cz rx ry rz  (Euler angles in radians)
        // Orientation doesn't matter for the sphere-guaranteed non-overlap approach, so output 0 0 0.
        std::cout << ps.c.x << " " << ps.c.y << " " << ps.c.z << " 0 0 0\n";
    }
    return 0;
}
