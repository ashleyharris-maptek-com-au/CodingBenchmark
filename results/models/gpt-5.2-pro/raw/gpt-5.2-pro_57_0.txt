'use strict';

const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

const W = data.width | 0;
const H = data.height | 0;
const N = (W * H) | 0;

const grid = data.grid;
const ignitions = data.ignitions || [];
const towns = data.towns || [];
let cutBudget = (data.cutBudget ?? 0) | 0;

function inBounds(x, y) {
  return x >= 0 && x < W && y >= 0 && y < H;
}
function idxOf(x, y) {
  return (y * W + x) | 0;
}

class Fenwick {
  constructor(n) {
    this.n = n | 0;
    this.bit = new Float64Array(this.n + 1);
  }
  add(i, delta) {
    for (let x = i | 0; x <= this.n; x += x & -x) this.bit[x] += delta;
  }
  sum(i) {
    let res = 0.0;
    for (let x = i | 0; x > 0; x -= x & -x) res += this.bit[x];
    return res;
  }
  rangeSum(l, r) {
    if (r < l) return 0.0;
    return this.sum(r) - this.sum(l - 1);
  }
}

class MaxHeap {
  constructor() {
    this.ids = [];
    this.keys = [];
  }
  size() { return this.ids.length; }
  push(id, key) {
    let i = this.ids.length;
    this.ids.push(id);
    this.keys.push(key);
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.keys[p] >= key) break;
      this.ids[i] = this.ids[p];
      this.keys[i] = this.keys[p];
      i = p;
    }
    this.ids[i] = id;
    this.keys[i] = key;
  }
  pop() {
    const n = this.ids.length;
    if (n === 0) return null;
    const topId = this.ids[0];
    const topKey = this.keys[0];
    const lastId = this.ids.pop();
    const lastKey = this.keys.pop();
    if (n > 1) {
      let i = 0;
      while (true) {
        let l = i * 2 + 1;
        if (l >= n - 1) break;
        let r = l + 1;
        let c = (r < n - 1 && this.keys[r] > this.keys[l]) ? r : l;
        if (this.keys[c] <= lastKey) break;
        this.ids[i] = this.ids[c];
        this.keys[i] = this.keys[c];
        i = c;
      }
      this.ids[i] = lastId;
      this.keys[i] = lastKey;
    }
    return [topId, topKey];
  }
}

function lowerBound(arr, x) {
  let lo = 0, hi = arr.length;
  while (lo < hi) {
    const mid = (lo + hi) >> 1;
    if (arr[mid] < x) lo = mid + 1;
    else hi = mid;
  }
  return lo;
}

// 1) Build burnable mask.
const burnable = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = grid[y];
  const base = (y * W) | 0;
  for (let x = 0; x < W; x++) {
    burnable[base + x] = (row.charCodeAt(x) === 46) ? 1 : 0; // '.'
  }
}

// Mark ignition & town cells (for constraints / fallback).
const isIgnitionCell = new Uint8Array(N);
for (let i = 0; i < ignitions.length; i++) {
  const x = ignitions[i][0] | 0, y = ignitions[i][1] | 0;
  if (!inBounds(x, y)) continue;
  const c = idxOf(x, y);
  if (burnable[c]) isIgnitionCell[c] = 1;
}
const isTownCell = new Uint8Array(N);
for (let i = 0; i < towns.length; i++) {
  const x = towns[i].x | 0, y = towns[i].y | 0;
  if (!inBounds(x, y)) continue;
  const c = idxOf(x, y);
  isTownCell[c] = 1;
}

// 2) BFS from ignitions to mark reachable burnable cells, assigning compact node ids.
const idMap = new Int32Array(N);
idMap.fill(-1);

// Preallocate worst-case.
const queue = new Int32Array(N);
const cellsAll = new Int32Array(N);

let head = 0, tail = 0, M = 0;

// Seed with ignitions.
for (let i = 0; i < ignitions.length; i++) {
  const x = ignitions[i][0] | 0, y = ignitions[i][1] | 0;
  if (!inBounds(x, y)) continue;
  const c = idxOf(x, y);
  if (!burnable[c]) continue;
  if (idMap[c] !== -1) continue;
  idMap[c] = M;
  cellsAll[M] = c;
  M++;
  queue[tail++] = c;
}

// If no valid ignition cells, no fire spreads; answer empty.
if (M === 0 || cutBudget <= 0 && towns.length === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const lastRowStart = (N - W) | 0;
while (head < tail) {
  const c = queue[head++];
  const x = c % W;

  // left
  if (x > 0) {
    const n = c - 1;
    if (burnable[n] && idMap[n] === -1) {
      idMap[n] = M;
      cellsAll[M] = n;
      M++;
      queue[tail++] = n;
    }
  }
  // right
  if (x + 1 < W) {
    const n = c + 1;
    if (burnable[n] && idMap[n] === -1) {
      idMap[n] = M;
      cellsAll[M] = n;
      M++;
      queue[tail++] = n;
    }
  }
  // up
  if (c >= W) {
    const n = c - W;
    if (burnable[n] && idMap[n] === -1) {
      idMap[n] = M;
      cellsAll[M] = n;
      M++;
      queue[tail++] = n;
    }
  }
  // down
  if (c < lastRowStart) {
    const n = c + W;
    if (burnable[n] && idMap[n] === -1) {
      idMap[n] = M;
      cellsAll[M] = n;
      M++;
      queue[tail++] = n;
    }
  }
}

const cells = cellsAll.subarray(0, M);

// 3) Create per-node arrays: sources + town values.
const isSource = new Uint8Array(M);
for (let i = 0; i < ignitions.length; i++) {
  const x = ignitions[i][0] | 0, y = ignitions[i][1] | 0;
  if (!inBounds(x, y)) continue;
  const c = idxOf(x, y);
  const id = idMap[c];
  if (id >= 0) isSource[id] = 1;
}

const townVal = new Float64Array(M);
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  const x = t.x | 0, y = t.y | 0;
  if (!inBounds(x, y)) continue;
  const c = idxOf(x, y);
  const id = idMap[c];
  if (id < 0) continue; // not threatened (unreachable or non-burnable)
  const v = +t.value;
  if (v > 0) townVal[id] += v;
}

// Quick exit: no threatened town value.
let threatenedTotal = 0.0;
for (let i = 0; i < M; i++) threatenedTotal += townVal[i];
if (threatenedTotal <= 0 || cutBudget <= 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

// 4) Tarjan articulation data (iterative DFS) over reachable subgraph.
//    We record, for each node v, a list of DFS-child-subtree intervals that would be disconnected from sources
//    by removing v (and that subtree contains no sources). Those intervals carry the town value potentially saved.
const disc = new Int32Array(M);
const low = new Int32Array(M);
const parent = new Int32Array(M);
parent.fill(-1);

const tin = new Int32Array(M);
const tout = new Int32Array(M);
const nodeAtTin = new Int32Array(M + 1);

const subtreeSum = new Float64Array(M);
const subtreeHasSource = new Uint8Array(M);

const headInt = new Int32Array(M);
headInt.fill(-1);
const intL = new Int32Array(M);
const intR = new Int32Array(M);
const intNext = new Int32Array(M);
let intCnt = 0;

const benefit = new Float64Array(M);

const stack = new Int32Array(M);
const stackDir = new Uint8Array(M);
let sp = 0;

let time = 0;
let ttime = 0;

// Start DFS from each source node to cover all BFS-reachable components.
for (let s = 0; s < M; s++) {
  if (!isSource[s] || disc[s] !== 0) continue;

  stack[sp] = s;
  stackDir[sp] = 0;
  parent[s] = -1;
  sp++;

  while (sp > 0) {
    const u = stack[sp - 1];
    let dir = stackDir[sp - 1];

    if (dir === 0 && disc[u] === 0) {
      disc[u] = low[u] = ++time;
      tin[u] = ++ttime;
      nodeAtTin[ttime] = u;
      subtreeSum[u] = townVal[u];
      subtreeHasSource[u] = isSource[u];
    }

    if (dir === 4) {
      // exit u
      tout[u] = ttime;
      sp--;

      const p = parent[u];
      if (p !== -1) {
        // Accumulate to parent
        subtreeSum[p] += subtreeSum[u];
        if (subtreeHasSource[u]) subtreeHasSource[p] = 1;

        // Articulation separation condition for p -> u
        if (low[u] >= disc[p] && !subtreeHasSource[u]) {
          intL[intCnt] = tin[u];
          intR[intCnt] = tout[u];
          intNext[intCnt] = headInt[p];
          headInt[p] = intCnt;
          benefit[p] += subtreeSum[u];
          intCnt++;
        }

        // Low-link update
        if (low[u] < low[p]) low[p] = low[u];
      }

      continue;
    }

    // Process next neighbor direction
    stackDir[sp - 1] = (dir + 1) & 255;

    const c = cells[u];
    const x = c % W;

    let nc = -1;
    if (dir === 0) {
      if (x > 0) nc = c - 1;
    } else if (dir === 1) {
      if (x + 1 < W) nc = c + 1;
    } else if (dir === 2) {
      if (c >= W) nc = c - W;
    } else {
      if (c < lastRowStart) nc = c + W;
    }
    if (nc < 0) continue;

    const v = idMap[nc];
    if (v < 0) continue; // not in reachable subgraph

    if (disc[v] === 0) {
      parent[v] = u;
      stack[sp] = v;
      stackDir[sp] = 0;
      sp++;
    } else if (v !== parent[u]) {
      const dv = disc[v];
      if (dv < low[u]) low[u] = dv;
    }
  }
}

// 5) Prepare town values by DFS-order position for dynamic coverage bookkeeping.
const townValAtPos = new Float64Array(M + 1);
for (let u = 0; u < M; u++) {
  const v = townVal[u];
  if (v > 0) townValAtPos[tin[u]] += v;
}

const townPosArr = [];
const townValArr = [];
for (let p = 1; p <= M; p++) {
  const v = townValAtPos[p];
  if (v > 0) {
    townPosArr.push(p);
    townValArr.push(v);
  }
}
const T = townPosArr.length;

// Fenwick tracks remaining (not-yet-saved-by-our-plan) town value.
const bit = new Fenwick(M);
for (let i = 0; i < T; i++) bit.add(townPosArr[i], townValArr[i]);

// DSU over town indices for fast removal over intervals.
const nextTown = new Int32Array(T + 1);
for (let i = 0; i <= T; i++) nextTown[i] = i;
const removedTown = new Uint8Array(T);

function findTown(i) {
  let r = i;
  while (nextTown[r] !== r) r = nextTown[r];
  while (nextTown[i] !== i) {
    const p = nextTown[i];
    nextTown[i] = r;
    i = p;
  }
  return r;
}
function removeTownIndex(i) {
  if (i >= T) return;
  if (removedTown[i]) return;
  removedTown[i] = 1;
  nextTown[i] = findTown(i + 1);
}

// 6) Greedy selection of high-yield articulation cuts (lazy heap).
const cutCell = new Uint8Array(N);
const chosenNode = new Uint8Array(M);

const cutHeap = new MaxHeap();
for (let v = 0; v < M; v++) {
  if (benefit[v] <= 0) continue;
  if (isSource[v]) continue;         // disallow cutting ignition/source cells
  if (townVal[v] > 0) continue;      // avoid cutting town cells
  cutHeap.push(v, benefit[v]);
}

function currentBenefit(v) {
  let sum = 0.0;
  for (let e = headInt[v]; e !== -1; e = intNext[e]) {
    sum += bit.rangeSum(intL[e], intR[e]);
  }
  return sum;
}

function coverInterval(L, R) {
  // Remove all town positions in [L, R] from BIT using DSU iteration.
  let idx = lowerBound(townPosArr, L);
  idx = findTown(idx);
  while (idx < T) {
    const pos = townPosArr[idx];
    if (pos > R) break;
    const val = townValArr[idx];
    if (!removedTown[idx] && val > 0) {
      bit.add(pos, -val);
      removeTownIndex(idx);
    } else {
      removeTownIndex(idx);
    }
    idx = findTown(idx);
  }
}

const firebreaks = [];
let remainingBudget = cutBudget;

function addCutCell(cellIdx) {
  if (cellIdx < 0 || cellIdx >= N) return false;
  if (!burnable[cellIdx]) return false;
  if (cutCell[cellIdx]) return false;
  cutCell[cellIdx] = 1;
  const x = cellIdx % W;
  const y = (cellIdx / W) | 0;
  firebreaks.push([x, y]);
  return true;
}

// Articulation greedy loop
while (remainingBudget > 0 && cutHeap.size() > 0) {
  const top = cutHeap.pop();
  if (!top) break;
  const v = top[0];
  const key = top[1];

  if (chosenNode[v]) continue;

  const cb = currentBenefit(v);
  if (cb <= 0) continue;

  // Lazy update
  if (cb + 1e-9 < key) {
    cutHeap.push(v, cb);
    continue;
  }

  const cellIdx = cells[v];
  // Safety constraints (should already be satisfied)
  if (isIgnitionCell[cellIdx]) continue;
  if (isTownCell[cellIdx]) continue;

  if (addCutCell(cellIdx)) {
    chosenNode[v] = 1;
    remainingBudget--;

    // Cover all saved subtrees' towns so they won't be double-counted.
    for (let e = headInt[v]; e !== -1; e = intNext[e]) {
      coverInterval(intL[e], intR[e]);
    }
  } else {
    chosenNode[v] = 1;
  }
}

// 7) Fallback: surround high-value remaining towns by cutting their 4-neighbors (if possible within budget).
// This guarantees those towns won't burn, even if the area is highly connected (no articulation bottlenecks).
if (remainingBudget > 0 && T > 0) {
  const townHeap = new MaxHeap();
  for (let i = 0; i < T; i++) {
    if (!removedTown[i]) townHeap.push(i, townValArr[i]);
  }

  while (remainingBudget > 0 && townHeap.size() > 0) {
    const top = townHeap.pop();
    if (!top) break;
    const ti = top[0];

    if (ti < 0 || ti >= T) continue;
    if (removedTown[ti]) continue;

    const pos = townPosArr[ti];
    const node = nodeAtTin[pos] | 0;
    if (node < 0 || node >= M) {
      // Shouldn't happen; just mark removed to avoid loops.
      removeTownIndex(ti);
      continue;
    }

    const townCellIdx = cells[node];
    const x = townCellIdx % W;

    // Determine required neighbor cuts to fully isolate the town cell.
    const neighborsToCut = [];
    let impossible = false;

    function considerNeighbor(nc) {
      if (nc < 0 || nc >= N) return;
      if (!burnable[nc]) return;          // already blocked
      if (cutCell[nc]) return;            // already blocked by our cut
      if (isIgnitionCell[nc]) { impossible = true; return; } // don't cut ignition
      if (isTownCell[nc]) { impossible = true; return; }     // don't cut other town
      neighborsToCut.push(nc);
    }

    // left/right/up/down
    if (x > 0) considerNeighbor(townCellIdx - 1);
    if (x + 1 < W) considerNeighbor(townCellIdx + 1);
    if (townCellIdx >= W) considerNeighbor(townCellIdx - W);
    if (townCellIdx < lastRowStart) considerNeighbor(townCellIdx + W);

    if (impossible) continue;
    if (neighborsToCut.length === 0) {
      // Already isolated by water/cuts; mark saved.
      bit.add(pos, -townValArr[ti]);
      removeTownIndex(ti);
      continue;
    }
    if (neighborsToCut.length > remainingBudget) continue;

    // Apply neighbor cuts
    let ok = true;
    for (let j = 0; j < neighborsToCut.length; j++) {
      if (remainingBudget <= 0) { ok = false; break; }
      if (addCutCell(neighborsToCut[j])) {
        remainingBudget--;
      } else {
        // If we couldn't add (already cut), it's still fine; does not consume budget.
      }
    }
    if (!ok) break;

    // Mark this town as saved in our bookkeeping.
    bit.add(pos, -townValArr[ti]);
    removeTownIndex(ti);
  }
}

process.stdout.write(JSON.stringify({ firebreaks }));
