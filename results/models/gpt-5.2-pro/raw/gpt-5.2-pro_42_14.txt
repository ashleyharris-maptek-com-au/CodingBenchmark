```spirv
; SPIR-V
; Version: 1.0
; Bound: 200
; Schema: 0
OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %in_color "color"
OpName %outColor "outColor"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_in 0 Offset 0

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 Offset 0

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func_void = OpTypeFunction %void

%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4

%uint_6 = OpConstant %uint 6

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_0_333333343 = OpConstant %float 0.333333343

%L_raw = OpConstantComposite %v3float %float_1 %float_1 %float_1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4float = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3float = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_out_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3float Output

%main = OpFunction %void None %func_void
%entry = OpLabel

; p0, p2, p4 (from worldPos[])
%p0_ptr = OpAccessChain %ptr_in_v3float %in_worldPos %int_0
%p0 = OpLoad %v3float %p0_ptr
%p2_ptr = OpAccessChain %ptr_in_v3float %in_worldPos %int_2
%p2 = OpLoad %v3float %p2_ptr
%p4_ptr = OpAccessChain %ptr_in_v3float %in_worldPos %int_4
%p4 = OpLoad %v3float %p4_ptr

%a = OpFSub %v3float %p2 %p0
%b = OpFSub %v3float %p4 %p0

; cross(a, b) manually
%a_x = OpCompositeExtract %float %a 0
%a_y = OpCompositeExtract %float %a 1
%a_z = OpCompositeExtract %float %a 2
%b_x = OpCompositeExtract %float %b 0
%b_y = OpCompositeExtract %float %b 1
%b_z = OpCompositeExtract %float %b 2

; cx = a.y*b.z - a.z*b.y
%ay_bz = OpFMul %float %a_y %b_z
%az_by = OpFMul %float %a_z %b_y
%cx = OpFSub %float %ay_bz %az_by

; cy = a.z*b.x - a.x*b.z
%az_bx = OpFMul %float %a_z %b_x
%ax_bz = OpFMul %float %a_x %b_z
%cy = OpFSub %float %az_bx %ax_bz

; cz = a.x*b.y - a.y*b.x
%ax_by = OpFMul %float %a_x %b_y
%ay_bx = OpFMul %float %a_y %b_x
%cz = OpFSub %float %ax_by %ay_bx

%cross = OpCompositeConstruct %v3float %cx %cy %cz
%N = OpExtInst %v3float %glsl Normalize %cross

%L = OpExtInst %v3float %glsl Normalize %L_raw

%dotNL = OpDot %float %N %L
%diff = OpExtInst %float %glsl FMax %dotNL %float_0

; base color = (color[0] + color[2] + color[4]) / 3
%c0_ptr = OpAccessChain %ptr_in_v3float %in_color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c2_ptr = OpAccessChain %ptr_in_v3float %in_color %int_2
%c2 = OpLoad %v3float %c2_ptr
%c4_ptr = OpAccessChain %ptr_in_v3float %in_color %int_4
%c4 = OpLoad %v3float %c4_ptr

%sum02 = OpFAdd %v3float %c0 %c2
%sum024 = OpFAdd %v3float %sum02 %c4
%base = OpVectorTimesScalar %v3float %sum024 %float_0_333333343

; final = base * (0.2 + 0.8 * diff)
%diff_08 = OpFMul %float %float_0_8 %diff
%factor = OpFAdd %float %float_0_2 %diff_08
%finalColor = OpVectorTimesScalar %v3float %base %factor

; Emit vertex 0 (gl_in[0].gl_Position)
%pos0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%glpos_out_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %glpos_out_ptr %pos0
OpStore %outColor %finalColor
OpEmitVertex

; Emit vertex 2 (gl_in[2].gl_Position)
%pos2_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
OpStore %glpos_out_ptr %pos2
OpStore %outColor %finalColor
OpEmitVertex

; Emit vertex 4 (gl_in[4].gl_Position)
%pos4_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %pos4_ptr
OpStore %glpos_out_ptr %pos4
OpStore %outColor %finalColor
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
