```typescript
declare const require: any;

const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8");
const data = JSON.parse(input);

type Drone = { id: any; x: number; y: number };

const prob: number[][] = data.probability;
const drones: Drone[] = data.drones;
const maxStepsPerDrone: number = data.maxStepsPerDrone | 0;

const H = prob.length | 0;
const W = (H > 0 ? prob[0].length : 0) | 0;
const N = drones.length | 0;

const size = (W * H) | 0;

function idxOf(x: number, y: number): number {
  return (y * W + x) | 0;
}
function xOf(idx: number): number {
  return idx % W;
}
function yOf(idx: number): number {
  return (idx / W) | 0;
}
function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < W && y >= 0 && y < H;
}

// Flatten grid: obstacle + value
const obstacle = new Uint8Array(size);
const value = new Float32Array(size);

for (let y = 0; y < H; y++) {
  const row = prob[y];
  for (let x = 0; x < W; x++) {
    const p = row[x];
    const id = idxOf(x, y);
    if (p === -1) {
      obstacle[id] = 1;
      value[id] = 0;
    } else {
      obstacle[id] = 0;
      value[id] = p > 0 ? p : 0;
    }
  }
}

// Multi-source BFS to assign each free cell to nearest drone (region ownership).
// owner[id] = droneIndex, distOwner[id] = distance; obstacles keep owner=-1, dist=-2.
const owner = new Int32Array(size);
const distOwner = new Int32Array(size);
owner.fill(-1);
distOwner.fill(-1);

for (let i = 0; i < size; i++) {
  if (obstacle[i]) distOwner[i] = -2;
}

const queue = new Int32Array(size);
let qh = 0, qt = 0;

for (let i = 0; i < N; i++) {
  const sx = drones[i].x | 0;
  const sy = drones[i].y | 0;
  if (!inBounds(sx, sy)) continue;
  const sidx = idxOf(sx, sy);
  if (obstacle[sidx]) continue;
  // If multiple drones start same cell, pick smaller index.
  if (distOwner[sidx] === -1 || distOwner[sidx] > 0 || owner[sidx] > i) {
    distOwner[sidx] = 0;
    owner[sidx] = i;
    queue[qt++] = sidx;
  }
}

const dx4 = new Int32Array([1, -1, 0, 0]);
const dy4 = new Int32Array([0, 0, 1, -1]);

while (qh < qt) {
  const v = queue[qh++];
  const dv = distOwner[v];
  const ov = owner[v];
  const vx = xOf(v);
  const vy = yOf(v);

  for (let k = 0; k < 4; k++) {
    const nx = vx + dx4[k];
    const ny = vy + dy4[k];
    if (!inBounds(nx, ny)) continue;
    const ni = idxOf(nx, ny);
    if (obstacle[ni]) continue;

    const nd = dv + 1;
    const cd = distOwner[ni];
    const co = owner[ni];

    if (cd === -1 || nd < cd || (nd === cd && ov < co)) {
      distOwner[ni] = nd;
      owner[ni] = ov;
      queue[qt++] = ni;
    }
  }
}

// Optional blurred potential field to guide movement when no immediate gain.
// (Skip on very large grids to reduce memory/time.)
let blur: Float32Array | null = null;
const BLUR_RADIUS = 4;
const BLUR_SIZE_LIMIT = 3_000_000; // cells

if (size > 0 && size <= BLUR_SIZE_LIMIT && BLUR_RADIUS > 0) {
  const r = BLUR_RADIUS | 0;
  const tmp = new Float32Array(size);
  blur = new Float32Array(size);

  // Horizontal box blur (sum)
  for (let y = 0; y < H; y++) {
    let sum = 0;
    const rowBase = y * W;

    for (let x = -r; x <= r; x++) {
      if (x >= 0 && x < W) sum += value[rowBase + x];
    }
    for (let x = 0; x < W; x++) {
      tmp[rowBase + x] = sum;
      const xOut = x - r;
      const xIn = x + r + 1;
      if (xOut >= 0) sum -= value[rowBase + xOut];
      if (xIn < W) sum += value[rowBase + xIn];
    }
  }

  // Vertical box blur (sum of horizontal sums)
  for (let x = 0; x < W; x++) {
    let sum = 0;

    for (let y = -r; y <= r; y++) {
      if (y >= 0 && y < H) sum += tmp[y * W + x];
    }
    for (let y = 0; y < H; y++) {
      blur[y * W + x] = sum;
      const yOut = y - r;
      const yIn = y + r + 1;
      if (yOut >= 0) sum -= tmp[yOut * W + x];
      if (yIn < H) sum += tmp[yIn * W + x];
    }
  }
}

// Shared covered set (unique coverage across all drones).
const covered = new Uint8Array(size);

// Reusable arrays for bounded BFS with stamp technique.
const seenStamp = new Int32Array(size);
const parent = new Int32Array(size);
const distSeen = new Int16Array(size);
let stamp = 1;

const MAX_RADIUS = 70; // bounded BFS max distance for selecting a target
const NO_TARGET_COOLDOWN = 10;

function boundedBestTarget(
  startIdx: number,
  droneIndex: number,
  stepsRemaining: number
): number {
  const maxDist = Math.min(MAX_RADIUS, stepsRemaining) | 0;
  if (maxDist <= 0) return -1;

  stamp++;
  if (stamp === 0x7fffffff) {
    // Rare: reset stamps
    seenStamp.fill(0);
    stamp = 1;
  }

  const q: number[] = [];
  q.push(startIdx);
  seenStamp[startIdx] = stamp;
  parent[startIdx] = -1;
  distSeen[startIdx] = 0;

  let bestIdx = -1;
  let bestScore = 0;

  for (let head = 0; head < q.length; head++) {
    const v = q[head] | 0;
    const dv = distSeen[v] | 0;

    // Evaluate this cell as a target (skip start with dist 0 unless it is valuable and uncovered).
    if (!covered[v]) {
      const vv = value[v];
      if (vv > 0) {
        // Prefer staying in assigned region.
        const own = owner[v];
        const regionFactor = own === droneIndex ? 1.0 : (own >= 0 ? 0.2 : 0.4);
        const score = regionFactor * (vv / (dv + 1));
        if (score > bestScore) {
          bestScore = score;
          bestIdx = v;
        }
      }
    }

    if (dv >= maxDist) continue;

    const vx = xOf(v);
    const vy = yOf(v);

    for (let k = 0; k < 4; k++) {
      const nx = vx + dx4[k];
      const ny = vy + dy4[k];
      if (!inBounds(nx, ny)) continue;
      const ni = idxOf(nx, ny);
      if (obstacle[ni]) continue;
      if (seenStamp[ni] === stamp) continue;
      seenStamp[ni] = stamp;
      parent[ni] = v;
      distSeen[ni] = (dv + 1) as any;
      q.push(ni);
    }
  }

  return bestIdx;
}

function reconstructPath(startIdx: number, targetIdx: number): number[] {
  // Returns indices from start to target inclusive, using current `parent`.
  const rev: number[] = [];
  let cur = targetIdx;
  while (cur !== -1) {
    rev.push(cur);
    if (cur === startIdx) break;
    cur = parent[cur];
  }
  if (rev[rev.length - 1] !== startIdx) return []; // unreachable (shouldn't happen)
  rev.reverse();
  return rev;
}

function bestLocalMove(
  curIdx: number,
  droneIndex: number
): number {
  // Among stay + 4-neighbors, pick max heuristic.
  const cx = xOf(curIdx);
  const cy = yOf(curIdx);

  let best = curIdx;
  let bestScore = -1e30;

  // 5 moves: stay first
  for (let m = -1; m < 4; m++) {
    let nx = cx, ny = cy;
    if (m >= 0) {
      nx = cx + dx4[m];
      ny = cy + dy4[m];
      if (!inBounds(nx, ny)) continue;
    }
    const ni = idxOf(nx, ny);
    if (obstacle[ni]) continue;

    const gain = (!covered[ni] ? value[ni] : 0);
    const own = owner[ni];
    const regionFactor = own === droneIndex ? 1.0 : (own >= 0 ? 0.25 : 0.4);

    // Small drift toward higher-density areas even if already covered.
    const drift = blur ? 0.002 * blur[ni] : 0;

    const score = regionFactor * gain + drift;

    if (score > bestScore) {
      bestScore = score;
      best = ni;
    }
  }

  return best;
}

// Build paths
const paths: number[][][] = new Array(N);

for (let i = 0; i < N; i++) {
  const sx = drones[i].x | 0;
  const sy = drones[i].y | 0;

  let curIdx: number;
  if (!inBounds(sx, sy)) {
    // If invalid, just return an empty-ish path at (0,0) within bounds.
    const fx = Math.max(0, Math.min(W - 1, sx));
    const fy = Math.max(0, Math.min(H - 1, sy));
    curIdx = idxOf(fx, fy);
  } else {
    curIdx = idxOf(sx, sy);
  }

  // If start is on obstacle, attempt to stay (will violate constraints otherwise, but assume valid inputs).
  const path: number[][] = [];
  path.push([xOf(curIdx), yOf(curIdx)]);
  if (!obstacle[curIdx]) covered[curIdx] = 1;

  let stepsLeft = maxStepsPerDrone | 0;
  let cooldown = 0;

  while (stepsLeft > 0) {
    if (cooldown > 0) {
      // Move locally without BFS
      const nextIdx = bestLocalMove(curIdx, i);
      curIdx = nextIdx;
      path.push([xOf(curIdx), yOf(curIdx)]);
      if (!obstacle[curIdx]) covered[curIdx] = 1;
      stepsLeft--;
      cooldown--;
      continue;
    }

    // Find best target within bounded BFS
    const targetIdx = boundedBestTarget(curIdx, i, stepsLeft);

    if (targetIdx === -1 || targetIdx === curIdx) {
      // No target found (or we're already on it): do a few local steps
      cooldown = NO_TARGET_COOLDOWN;
      continue;
    }

    const seq = reconstructPath(curIdx, targetIdx);
    if (seq.length <= 1) {
      cooldown = NO_TARGET_COOLDOWN;
      continue;
    }

    // Follow the path toward the target
    // seq[0] == curIdx; execute moves along seq[1..]
    for (let t = 1; t < seq.length && stepsLeft > 0; t++) {
      curIdx = seq[t] | 0;
      path.push([xOf(curIdx), yOf(curIdx)]);
      if (!obstacle[curIdx]) covered[curIdx] = 1;
      stepsLeft--;
    }
  }

  paths[i] = path;
}

process.stdout.write(JSON.stringify({ paths }));
```
