```spirv
; SPIR-V
; Version: 1.0
; Bound: 86
; Schema: 0
OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void

%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%uint_3 = OpConstant %uint 3

%float_0 = OpConstant %float 0
%float_0_08 = OpConstant %float 0.08
%float_0_33333334 = OpConstant %float 0.33333334

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float

%arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float

%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %fn_void
%entry = OpLabel

; p0 = gl_in[0].gl_Position
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2

; p1 = gl_in[1].gl_Position
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2

; p2 = gl_in[2].gl_Position
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

; center = (p0+p1+p2)/3
%sum01 = OpFAdd %v3float %p0_xyz %p1_xyz
%sum012 = OpFAdd %v3float %sum01 %p2_xyz
%center = OpVectorTimesScalar %v3float %sum012 %float_0_33333334

; e1 = p1 - p0, e2 = p2 - p0
%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz

; N = normalize(cross(e1, e2))
%cr = OpExtInst %v3float %glsl Cross %e1 %e2
%N = OpExtInst %v3float %glsl Normalize %cr

; offset = N * 0.08
%offset3 = OpVectorTimesScalar %v3float %N %float_0_08
%ox = OpCompositeExtract %float %offset3 0
%oy = OpCompositeExtract %float %offset3 1
%oz = OpCompositeExtract %float %offset3 2
%offset4 = OpCompositeConstruct %v4float %ox %oy %oz %float_0

; ---- Emit vertex 0 ----
%pos0 = OpFAdd %v4float %p0 %offset4
%outpos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outpos_ptr %pos0
%col0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%col0 = OpLoad %v3float %col0_ptr
OpStore %outColor %col0
OpEmitVertex

; ---- Emit vertex 1 ----
%pos1 = OpFAdd %v4float %p1 %offset4
OpStore %outpos_ptr %pos1
%col1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%col1 = OpLoad %v3float %col1_ptr
OpStore %outColor %col1
OpEmitVertex

; ---- Emit vertex 2 ----
%pos2 = OpFAdd %v4float %p2 %offset4
OpStore %outpos_ptr %pos2
%col2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%col2 = OpLoad %v3float %col2_ptr
OpStore %outColor %col2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
