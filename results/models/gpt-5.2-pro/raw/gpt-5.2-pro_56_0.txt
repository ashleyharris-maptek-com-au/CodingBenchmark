```javascript
'use strict';

const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input ? JSON.parse(input) : {};

const depot = Array.isArray(data.depot) ? data.depot : [0, 0];
const packages = Array.isArray(data.packages) ? data.packages : [];
const n = packages.length;

if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

// Store points as index-based arrays: 0 = depot, 1..n = packages
const xs = new Float64Array(n + 1);
const ys = new Float64Array(n + 1);
const ids = new Array(n + 1);

xs[0] = +depot[0] || 0;
ys[0] = +depot[1] || 0;
ids[0] = null;

let minX = xs[0], maxX = xs[0], minY = ys[0], maxY = ys[0];
for (let i = 1; i <= n; i++) {
  const p = packages[i - 1];
  const x = +p.x || 0;
  const y = +p.y || 0;
  xs[i] = x;
  ys[i] = y;
  ids[i] = p.id;
  if (x < minX) minX = x;
  if (x > maxX) maxX = x;
  if (y < minY) minY = y;
  if (y > maxY) maxY = y;
}

// Morton (Z-order) code for initial tour
const spanX = maxX - minX;
const spanY = maxY - minY;
const invX = spanX > 0 ? 65535.0 / spanX : 0.0;
const invY = spanY > 0 ? 65535.0 / spanY : 0.0;

function part1by1_16(x) {
  // x: 16-bit
  x &= 0xFFFF;
  x = (x | (x << 8)) & 0x00FF00FF;
  x = (x | (x << 4)) & 0x0F0F0F0F;
  x = (x | (x << 2)) & 0x33333333;
  x = (x | (x << 1)) & 0x55555555;
  return x >>> 0;
}

const codes = new Uint32Array(n + 1); // only 1..n used
for (let i = 1; i <= n; i++) {
  let xi = spanX > 0 ? ((xs[i] - minX) * invX) : 0;
  let yi = spanY > 0 ? ((ys[i] - minY) * invY) : 0;
  if (xi < 0) xi = 0; else if (xi > 65535) xi = 65535;
  if (yi < 0) yi = 0; else if (yi > 65535) yi = 65535;
  const x16 = xi | 0;
  const y16 = yi | 0;
  const code = ((part1by1_16(x16) << 1) | part1by1_16(y16)) >>> 0;
  codes[i] = code;
}

// Initial order by Morton code
const order = new Array(n);
for (let k = 0; k < n; k++) order[k] = k + 1;
order.sort((a, b) => (codes[a] - codes[b]) || (a - b));

// Tour indices with depot endpoints fixed: [0, ...packages..., 0]
const L = n + 2;
const tour = new Int32Array(L);
tour[0] = 0;
for (let i = 0; i < n; i++) tour[i + 1] = order[i];
tour[L - 1] = 0;

function reverseRange(arr, lo, hi) {
  while (lo < hi) {
    const t = arr[lo];
    arr[lo++] = arr[hi];
    arr[hi--] = t;
  }
}

function distIdx(a, b) {
  const dx = xs[a] - xs[b];
  const dy = ys[a] - ys[b];
  return Math.sqrt(dx * dx + dy * dy);
}

// Local improvement: limited-window 2-opt (scales ~ O(n * window * passes))
const WINDOW = 32;
const PASSES = 2;
const EPS = 1e-12;

for (let pass = 0; pass < PASSES; pass++) {
  let improved = false;

  // i is start edge (i -> i+1), j is start of second edge (j -> j+1)
  // valid i: 0..L-4, valid j: i+2..L-2
  for (let i = 0; i <= L - 4; i++) {
    const a = tour[i];
    let b = tour[i + 1];

    const ax = xs[a], ay = ys[a];

    const maxJ = Math.min(L - 2, i + WINDOW);
    for (let j = i + 2; j <= maxJ; j++) {
      const c = tour[j];
      const d = tour[j + 1];

      // old edges: (a-b) + (c-d)
      // new edges: (a-c) + (b-d)
      const bx = xs[b], by = ys[b];
      const cx = xs[c], cy = ys[c];
      const dxv = xs[d], dyv = ys[d];

      const ab = Math.sqrt((ax - bx) * (ax - bx) + (ay - by) * (ay - by));
      const cd = Math.sqrt((cx - dxv) * (cx - dxv) + (cy - dyv) * (cy - dyv));
      const ac = Math.sqrt((ax - cx) * (ax - cx) + (ay - cy) * (ay - cy));
      const bd = Math.sqrt((bx - dxv) * (bx - dxv) + (by - dyv) * (by - dyv));

      if (ac + bd + EPS < ab + cd) {
        reverseRange(tour, i + 1, j);
        improved = true;
        b = tour[i + 1]; // update because tour changed
      }
    }
  }

  if (!improved) break;
}

// Output: single route (max routes constraint is "at most", and fewer routes never hurts in metric VRP)
const route = new Array(n);
for (let i = 0; i < n; i++) route[i] = ids[tour[i + 1]];

process.stdout.write(JSON.stringify({ routes: [route] }));
```
