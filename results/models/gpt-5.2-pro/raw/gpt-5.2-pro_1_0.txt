// TSP adaptive solver (C++20, MSVC-friendly, no non-standard headers)
//
// Notes on reality constraints:
// - For extremely large N, input size and output size dominate runtime. Printing N integers
//   (and even reading 2*N floating values) is physically impossible within 300 seconds for
//   astronomically large N on any system.
// - This program adapts algorithms by N and always outputs a valid permutation starting at 0.
// - For N beyond an in-memory threshold, it falls back to the identity order: 0..N-1
//   (constant memory, valid tour, minimal compute).

#include <algorithm>
#include <cstdint>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <limits>
#include <numeric>
#include <string>
#include <vector>

static inline bool isSpace(char c) {
    return c == ' ' || c == '\n' || c == '\r' || c == '\t' || c == '\f' || c == '\v';
}

class FastScanner {
public:
    explicit FastScanner(FILE* f = stdin) : f_(f) {
        refill();
    }

    bool readU64(uint64_t& out) {
        skipSpaces();
        if (eof_) return false;
        uint64_t x = 0;
        char c = peek();
        if (c < '0' || c > '9') return false;
        while (!eof_) {
            c = peek();
            if (c < '0' || c > '9') break;
            x = x * 10 + uint64_t(c - '0');
            advance();
        }
        out = x;
        return true;
    }

    bool readDouble(double& out) {
        skipSpaces();
        if (eof_) return false;

        bool neg = false;
        char c = peek();
        if (c == '+' || c == '-') {
            neg = (c == '-');
            advance();
        }

        // integer part
        double val = 0.0;
        bool any = false;
        while (!eof_) {
            c = peek();
            if (c < '0' || c > '9') break;
            any = true;
            val = val * 10.0 + double(c - '0');
            advance();
        }

        // fractional
        if (!eof_ && peek() == '.') {
            advance();
            double base = 1.0;
            while (!eof_) {
                c = peek();
                if (c < '0' || c > '9') break;
                any = true;
                base *= 0.1;
                val += double(c - '0') * base;
                advance();
            }
        }

        // exponent
        if (!eof_) {
            c = peek();
            if (c == 'e' || c == 'E') {
                advance();
                bool eneg = false;
                if (!eof_) {
                    c = peek();
                    if (c == '+' || c == '-') {
                        eneg = (c == '-');
                        advance();
                    }
                }
                int exp = 0;
                while (!eof_) {
                    c = peek();
                    if (c < '0' || c > '9') break;
                    exp = exp * 10 + int(c - '0');
                    advance();
                }
                if (eneg) exp = -exp;
                val *= pow10i(exp);
            }
        }

        if (!any) return false;
        out = neg ? -val : val;
        return true;
    }

private:
    FILE* f_;
    static constexpr size_t BUFSZ = 1 << 20;
    char buf_[BUFSZ];
    size_t pos_ = 0, len_ = 0;
    bool eof_ = false;

    void refill() {
        len_ = std::fread(buf_, 1, BUFSZ, f_);
        pos_ = 0;
        eof_ = (len_ == 0);
    }

    char peek() {
        if (pos_ >= len_) refill();
        return eof_ ? '\0' : buf_[pos_];
    }

    void advance() {
        ++pos_;
    }

    void skipSpaces() {
        while (!eof_) {
            char c = peek();
            if (!isSpace(c)) break;
            advance();
        }
    }

    static double pow10i(int e) {
        // Fast-ish pow10 using precomputed tables for |e|<=308 (double range).
        // Falls back to std::pow for large |e| (rare in typical coordinate inputs).
        static bool inited = false;
        static double p10[309];
        if (!inited) {
            p10[0] = 1.0;
            for (int i = 1; i <= 308; ++i) p10[i] = p10[i - 1] * 10.0;
            inited = true;
        }
        if (e == 0) return 1.0;
        if (e > 0) {
            if (e <= 308) return p10[e];
            return std::pow(10.0, (double)e);
        } else {
            int a = -e;
            if (a <= 308) return 1.0 / p10[a];
            return std::pow(10.0, (double)e);
        }
    }
};

class FastWriter {
public:
    explicit FastWriter(FILE* f = stdout) : f_(f) {
        buf_.reserve(1 << 20);
    }

    void writeU64(uint64_t x) {
        char tmp[32];
        int n = 0;
        if (x == 0) tmp[n++] = '0';
        else {
            while (x > 0) {
                tmp[n++] = char('0' + (x % 10));
                x /= 10;
            }
            std::reverse(tmp, tmp + n);
        }
        append(tmp, n);
    }

    void writeChar(char c) {
        buf_.push_back(c);
        if (buf_.size() >= FLUSH_THRESH) flush();
    }

    void flush() {
        if (!buf_.empty()) {
            std::fwrite(buf_.data(), 1, buf_.size(), f_);
            buf_.clear();
        }
    }

    ~FastWriter() { flush(); }

private:
    FILE* f_;
    std::string buf_;
    static constexpr size_t FLUSH_THRESH = 1 << 20;

    void append(const char* s, size_t n) {
        if (buf_.size() + n >= FLUSH_THRESH) flush();
        if (n >= FLUSH_THRESH) {
            std::fwrite(s, 1, n, f_);
            return;
        }
        buf_.append(s, n);
    }
};

static inline double dist2(const std::vector<float>& xs, const std::vector<float>& ys, uint32_t a, uint32_t b) {
    double dx = double(xs[a]) - double(xs[b]);
    double dy = double(ys[a]) - double(ys[b]);
    return dx * dx + dy * dy;
}

// Morton interleave helpers (32-bit -> 64-bit).
static inline uint64_t dilate32(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8))  & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4))  & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2))  & 0x3333333333333333ULL;
    v = (v | (v << 1))  & 0x5555555555555555ULL;
    return v;
}

static inline uint64_t morton2D(uint32_t x, uint32_t y) {
    return (dilate32(y) << 1) | dilate32(x);
}

// Stable radix sort on 64-bit keys, carrying a uint32 index.
static void radixSort64(std::vector<uint64_t>& keys, std::vector<uint32_t>& idx) {
    const size_t n = keys.size();
    std::vector<uint64_t> k2(n);
    std::vector<uint32_t> i2(n);

    constexpr int PASSES = 8; // 8 * 8 bits = 64
    uint32_t count[256];

    for (int pass = 0; pass < PASSES; ++pass) {
        std::memset(count, 0, sizeof(count));
        const int shift = pass * 8;

        for (size_t i = 0; i < n; ++i) {
            uint8_t b = uint8_t((keys[i] >> shift) & 0xFFULL);
            ++count[b];
        }
        uint32_t sum = 0;
        for (int b = 0; b < 256; ++b) {
            uint32_t c = count[b];
            count[b] = sum;
            sum += c;
        }
        for (size_t i = 0; i < n; ++i) {
            uint8_t b = uint8_t((keys[i] >> shift) & 0xFFULL);
            uint32_t p = count[b]++;
            k2[p] = keys[i];
            i2[p] = idx[i];
        }
        keys.swap(k2);
        idx.swap(i2);
    }
}

// Rotate tour so that city 0 is first.
static void rotateToZero(std::vector<uint32_t>& order) {
    auto it = std::find(order.begin(), order.end(), 0u);
    if (it != order.end()) std::rotate(order.begin(), it, order.end());
}

// Exact solver for N<=12 (Held-Karp path cycle, then rotate to start at 0).
static std::vector<uint32_t> solveExactHeldKarp(const std::vector<float>& xs, const std::vector<float>& ys) {
    const int n = (int)xs.size();
    const int FULL = 1 << n;
    const double INF = std::numeric_limits<double>::infinity();

    std::vector<std::vector<double>> d(n, std::vector<double>(n, 0.0));
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < n; ++j)
            d[i][j] = std::sqrt(dist2(xs, ys, (uint32_t)i, (uint32_t)j));

    std::vector<double> dp(FULL * n, INF);
    std::vector<int> parent(FULL * n, -1);

    auto at = [&](int mask, int last) -> double& { return dp[mask * n + last]; };
    auto par = [&](int mask, int last) -> int& { return parent[mask * n + last]; };

    // Force starting at 0
    at(1 << 0, 0) = 0.0;

    for (int mask = 0; mask < FULL; ++mask) {
        if (!(mask & 1)) continue; // must include 0
        for (int last = 0; last < n; ++last) {
            double cur = at(mask, last);
            if (!(cur < INF)) continue;
            for (int nxt = 0; nxt < n; ++nxt) {
                if (mask & (1 << nxt)) continue;
                int nmask = mask | (1 << nxt);
                double cand = cur + d[last][nxt];
                double& ref = at(nmask, nxt);
                if (cand < ref) {
                    ref = cand;
                    par(nmask, nxt) = last;
                }
            }
        }
    }

    // close cycle to 0
    int bestLast = -1;
    double best = INF;
    int fullMask = FULL - 1;
    for (int last = 0; last < n; ++last) {
        if (last == 0) continue;
        double cand = at(fullMask, last) + d[last][0];
        if (cand < best) {
            best = cand;
            bestLast = last;
        }
    }
    if (n == 1) return {0};

    // Reconstruct path visiting all, ending at bestLast, starting at 0
    std::vector<uint32_t> rev;
    int mask = fullMask;
    int last = bestLast;
    while (last != -1) {
        rev.push_back((uint32_t)last);
        int p = par(mask, last);
        mask ^= (1 << last);
        last = p;
    }
    std::reverse(rev.begin(), rev.end()); // starts with 0

    // rev is Hamiltonian path from 0; interpret as tour with implicit return
    rotateToZero(rev);
    return rev;
}

// O(N^2) nearest neighbor (good for N up to a few thousand).
static std::vector<uint32_t> solveNearestNeighbor(const std::vector<float>& xs, const std::vector<float>& ys) {
    const uint32_t n = (uint32_t)xs.size();
    std::vector<uint32_t> order;
    order.reserve(n);
    std::vector<uint8_t> used(n, 0);

    uint32_t cur = 0;
    order.push_back(cur);
    used[cur] = 1;

    for (uint32_t step = 1; step < n; ++step) {
        uint32_t best = UINT32_MAX;
        double bestD = std::numeric_limits<double>::infinity();
        for (uint32_t j = 0; j < n; ++j) {
            if (used[j]) continue;
            double d = dist2(xs, ys, cur, j);
            if (d < bestD) {
                bestD = d;
                best = j;
            }
        }
        cur = best;
        used[cur] = 1;
        order.push_back(cur);
    }
    rotateToZero(order);
    return order;
}

// 2-opt improvement on a cyclic tour stored in order[].
// For N<=5000 we can do a couple of full passes; for larger we do bounded window.
static void twoOptFull(std::vector<uint32_t>& order,
                       const std::vector<float>& xs,
                       const std::vector<float>& ys,
                       int maxPasses) {
    const int n = (int)order.size();
    if (n < 4) return;

    auto dist = [&](int i, int j) -> double {
        return std::sqrt(dist2(xs, ys, order[i], order[j]));
    };

    for (int pass = 0; pass < maxPasses; ++pass) {
        bool improved = false;
        for (int i = 0; i < n - 1; ++i) {
            int i2 = (i + 1) % n;
            for (int j = i + 2; j < n; ++j) {
                int j2 = (j + 1) % n;
                if (i == 0 && j2 == 0) continue; // don't break the cycle in a degenerate way
                // edges: (i,i2) and (j,j2) -> (i,j) and (i2,j2)
                double before = dist(i, i2) + dist(j, j2);
                double after  = dist(i, j) + dist(i2, j2);
                if (after + 1e-12 < before) {
                    // reverse segment [i2..j]
                    std::reverse(order.begin() + i2, order.begin() + j + 1);
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }
    rotateToZero(order);
}

static void twoOptWindow(std::vector<uint32_t>& order,
                         const std::vector<float>& xs,
                         const std::vector<float>& ys,
                         int window,
                         int passes) {
    const int n = (int)order.size();
    if (n < 4) return;

    auto dist = [&](int i, int j) -> double {
        return std::sqrt(dist2(xs, ys, order[i], order[j]));
    };

    for (int pass = 0; pass < passes; ++pass) {
        bool improved = false;
        for (int i = 0; i < n - 1; ++i) {
            int i2 = i + 1;
            int jMax = std::min(n - 1, i + window);
            for (int j = i + 2; j <= jMax; ++j) {
                int j2 = (j + 1 < n) ? (j + 1) : 0;
                if (i == 0 && j2 == 0) continue;
                double before = dist(i, i2) + dist(j, j2);
                double after  = dist(i, j) + dist(i2, j2);
                if (after + 1e-12 < before) {
                    std::reverse(order.begin() + i2, order.begin() + j + 1);
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }
    rotateToZero(order);
}

static std::vector<uint32_t> solveMortonOrder(const std::vector<float>& xs, const std::vector<float>& ys) {
    const size_t n = xs.size();
    std::vector<uint64_t> keys(n);
    std::vector<uint32_t> idx(n);

    float minx = std::numeric_limits<float>::infinity();
    float maxx = -std::numeric_limits<float>::infinity();
    float miny = std::numeric_limits<float>::infinity();
    float maxy = -std::numeric_limits<float>::infinity();

    for (size_t i = 0; i < n; ++i) {
        minx = std::min(minx, xs[i]); maxx = std::max(maxx, xs[i]);
        miny = std::min(miny, ys[i]); maxy = std::max(maxy, ys[i]);
    }

    const double dx = double(maxx) - double(minx);
    const double dy = double(maxy) - double(miny);
    const double invx = (dx > 0) ? (1.0 / dx) : 0.0;
    const double invy = (dy > 0) ? (1.0 / dy) : 0.0;

    for (size_t i = 0; i < n; ++i) idx[i] = (uint32_t)i;

    for (size_t i = 0; i < n; ++i) {
        double nx = (double(xs[i]) - double(minx)) * invx; // ~[0,1]
        double ny = (double(ys[i]) - double(miny)) * invy;
        if (nx < 0) nx = 0; if (nx > 1) nx = 1;
        if (ny < 0) ny = 0; if (ny > 1) ny = 1;
        uint32_t qx = (uint32_t)std::llround(nx * 4294967295.0);
        uint32_t qy = (uint32_t)std::llround(ny * 4294967295.0);
        keys[i] = morton2D(qx, qy);
    }

    if (n >= 200000) {
        radixSort64(keys, idx);
    } else {
        std::sort(idx.begin(), idx.end(), [&](uint32_t a, uint32_t b) {
            return keys[a] < keys[b];
        });
    }

    rotateToZero(idx);
    return idx;
}

int main() {
    FastScanner in(stdin);
    FastWriter out(stdout);

    uint64_t N64 = 0;
    if (!in.readU64(N64)) return 0;

    // Trivial cases
    if (N64 == 0) {
        // Undefined by problem statement; output nothing.
        return 0;
    }
    if (N64 == 1) {
        out.writeU64(0);
        out.writeChar('\n');
        return 0;
    }

    // If N is too large to store comfortably, output identity permutation.
    // (Also avoids size_t overflow in vectors, and avoids memory blow-ups.)
    // Tune as needed; with 32GB RAM this is a conservative but safe ceiling.
    constexpr uint64_t MAX_STORE = 25000000ULL; // 25 million
    if (N64 > MAX_STORE || N64 > (uint64_t)std::numeric_limits<size_t>::max()) {
        // Fast valid output: 0 1 2 ... N-1
        // (We do not read coordinates; consuming stdin isn't necessary for correctness.)
        for (uint64_t i = 0; i < N64; ++i) {
            if (i) out.writeChar(' ');
            out.writeU64(i);
        }
        out.writeChar('\n');
        return 0;
    }

    const size_t N = (size_t)N64;
    std::vector<float> xs(N), ys(N);

    for (size_t i = 0; i < N; ++i) {
        double x, y;
        if (!in.readDouble(x)) x = 0.0;
        if (!in.readDouble(y)) y = 0.0;
        xs[i] = (float)x;
        ys[i] = (float)y;
    }

    std::vector<uint32_t> order;

    // Adaptive strategy by N
    if (N <= 12) {
        order = solveExactHeldKarp(xs, ys);
    } else if (N <= 3000) {
        order = solveNearestNeighbor(xs, ys);
        // Improve with 2-opt; full passes are ok at this size.
        if (N <= 2000) twoOptFull(order, xs, ys, 2);
        else twoOptWindow(order, xs, ys, 48, 2);
    } else if (N <= 200000) {
        order = solveMortonOrder(xs, ys);
        // Quick local improvement; bounded to keep runtime predictable.
        twoOptWindow(order, xs, ys, 32, 2);
    } else {
        order = solveMortonOrder(xs, ys);
        // No further optimization at large sizes.
    }

    // Output permutation starting at 0.
    // Ensure it is exactly N numbers.
    if (order.empty()) {
        // Fallback (shouldn't happen)
        for (size_t i = 0; i < N; ++i) order.push_back((uint32_t)i);
        rotateToZero(order);
    }

    // Write
    for (size_t i = 0; i < order.size(); ++i) {
        if (i) out.writeChar(' ');
        out.writeU64(order[i]);
    }
    out.writeChar('\n');
    return 0;
}
