import math
from typing import Dict, Tuple, Optional, List

# ------------------------ module-persistent autopilot memory ------------------------
_AP = {
    "init": False,
    "r": [0.0, 0.0, 0.0],     # estimated relative position LVLH [m] (x radial, y along-track, z cross-track)
    "v": [0.0, 0.0, 0.0],     # estimated relative velocity LVLH [m/s]
    "age_pos": 1e9,          # seconds since last good position measurement
    "age_vel": 1e9,          # seconds since last good velocity measurement
    "last_range": None,      # last scalar range estimate [m]
}

# ------------------------ small utilities ------------------------
def _clamp(x: float, lo: float, hi: float) -> float:
    return lo if x < lo else hi if x > hi else x

def _norm3(a: List[float]) -> float:
    return math.sqrt(a[0]*a[0] + a[1]*a[1] + a[2]*a[2])

def _dot3(a: List[float], b: List[float]) -> float:
    return a[0]*b[0] + a[1]*b[1] + a[2]*b[2]

def _add3(a: List[float], b: List[float]) -> List[float]:
    return [a[0]+b[0], a[1]+b[1], a[2]+b[2]]

def _sub3(a: List[float], b: List[float]) -> List[float]:
    return [a[0]-b[0], a[1]-b[1], a[2]-b[2]]

def _mul3(a: List[float], s: float) -> List[float]:
    return [a[0]*s, a[1]*s, a[2]*s]

def _unit3(a: List[float]) -> List[float]:
    n = _norm3(a)
    if n < 1e-9:
        return [1.0, 0.0, 0.0]
    return [a[0]/n, a[1]/n, a[2]/n]

def _median(vals: List[float]) -> Optional[float]:
    vals = [v for v in vals if v is not None and math.isfinite(v)]
    if not vals:
        return None
    vals.sort()
    m = len(vals)//2
    return vals[m] if (len(vals) % 2 == 1) else 0.5*(vals[m-1] + vals[m])

def _robust_vec(cands: List[Tuple[List[float], float]]) -> Optional[List[float]]:
    """
    Weighted robust-ish merge:
      - if only one -> return it
      - else do weighted average, but downweight outliers (one reweight pass)
    """
    cands = [(v, w) for (v, w) in cands if v is not None and w > 0]
    if not cands:
        return None
    if len(cands) == 1:
        return list(cands[0][0])

    # initial weighted mean
    sw = sum(w for _, w in cands)
    mu = [0.0, 0.0, 0.0]
    for v, w in cands:
        mu[0] += w * v[0]; mu[1] += w * v[1]; mu[2] += w * v[2]
    mu = [mu[0]/sw, mu[1]/sw, mu[2]/sw]

    # reweight by distance to mean (Huber-like)
    dists = [(_norm3(_sub3(v, mu)), w) for v, w in cands]
    med_d = _median([d for d, _ in dists]) or 0.0
    scale = max(1e-6, 2.5 * med_d + 1e-6)

    mu2 = [0.0, 0.0, 0.0]
    sw2 = 0.0
    for (v, w), (d, _) in zip(cands, dists):
        rw = w * (1.0 if d <= scale else scale / d)
        sw2 += rw
        mu2[0] += rw * v[0]; mu2[1] += rw * v[1]; mu2[2] += rw * v[2]
    if sw2 <= 0:
        return list(mu)
    return [mu2[0]/sw2, mu2[1]/sw2, mu2[2]/sw2]

def _get(state: Dict, key: str, default=None):
    return state[key] if key in state else default

def _valid(state: Dict, key: str) -> bool:
    v = _get(state, key, False)
    return bool(v)

# ------------------------ sensor fusion ------------------------
def _collect_rel_pos_vel_measurements(state: Dict) -> Tuple[Optional[List[float]], Optional[List[float]], Optional[float], Optional[List[float]]]:
    """
    Returns:
      pos_meas [x,y,z] in m (LVLH), vel_meas [vx,vy,vz] in m/s, range_meas in m, cam_pos_meas in m (if available)
    """
    # Radar
    radar_pos = None
    radar_vel = None
    if _valid(state, "radar_rel_x_valid") and _valid(state, "radar_rel_y_valid") and _valid(state, "radar_rel_z_valid"):
        radar_pos = [
            float(_get(state, "radar_rel_pos_x_m", 0.0)),
            float(_get(state, "radar_rel_pos_y_m", 0.0)),
            float(_get(state, "radar_rel_pos_z_m", 0.0)),
        ]
    if _valid(state, "radar_relv_x_valid") and _valid(state, "radar_relv_y_valid") and _valid(state, "radar_relv_z_valid"):
        radar_vel = [
            float(_get(state, "radar_rel_vel_x_ms", 0.0)),
            float(_get(state, "radar_rel_vel_y_ms", 0.0)),
            float(_get(state, "radar_rel_vel_z_ms", 0.0)),
        ]

    # Lidar
    lidar_pos = None
    lidar_vel = None
    if _valid(state, "lidar_rel_x_valid") and _valid(state, "lidar_rel_y_valid") and _valid(state, "lidar_rel_z_valid"):
        lidar_pos = [
            float(_get(state, "lidar_rel_pos_x_m", 0.0)),
            float(_get(state, "lidar_rel_pos_y_m", 0.0)),
            float(_get(state, "lidar_rel_pos_z_m", 0.0)),
        ]
    if _valid(state, "lidar_relv_x_valid") and _valid(state, "lidar_relv_y_valid") and _valid(state, "lidar_relv_z_valid"):
        lidar_vel = [
            float(_get(state, "lidar_rel_vel_x_ms", 0.0)),
            float(_get(state, "lidar_rel_vel_y_ms", 0.0)),
            float(_get(state, "lidar_rel_vel_z_ms", 0.0)),
        ]

    # Docking camera (within 500 m, provides bearing + range; assume bearing in LVLH axes)
    cam_pos = None
    if bool(_get(state, "dock_camera_active", False)):
        bxv = _valid(state, "dock_camera_bearing_x_valid")
        byv = _valid(state, "dock_camera_bearing_y_valid")
        bzv = _valid(state, "dock_camera_bearing_z_valid")
        rng = _get(state, "dock_camera_range_m", None)
        if bxv and byv and bzv and rng is not None and math.isfinite(float(rng)) and float(rng) > 0:
            b = [float(_get(state, "dock_camera_bearing_x", 0.0)),
                 float(_get(state, "dock_camera_bearing_y", 0.0)),
                 float(_get(state, "dock_camera_bearing_z", 0.0))]
            # If bearing not normalized, normalize.
            b = _unit3(b)
            cam_pos = _mul3(b, float(rng))

    # Range finders: robust scalar range (used to correct magnitude)
    ranges = []
    if _valid(state, "range_1_valid"):
        r = _get(state, "range_1_m", None)
        if r is not None and math.isfinite(float(r)):
            ranges.append(float(r))
    if _valid(state, "range_2_valid"):
        r = _get(state, "range_2_m", None)
        if r is not None and math.isfinite(float(r)):
            ranges.append(float(r))
    if _valid(state, "range_3_valid"):
        r = _get(state, "range_3_m", None)
        if r is not None and math.isfinite(float(r)):
            ranges.append(float(r))
    range_meas = _median(ranges)

    # Merge position sources with range-aware weighting.
    pos_cands = []
    vel_cands = []

    # Weighting intuition:
    # - Radar: good far, OK near
    # - Lidar: excellent near, may degrade far
    # - Camera: good within 500 m for line-of-sight to port
    if radar_pos is not None:
        pos_cands.append((radar_pos, 1.0))
    if lidar_pos is not None:
        pos_cands.append((lidar_pos, 1.3))
    if cam_pos is not None:
        pos_cands.append((cam_pos, 1.2))

    if radar_vel is not None:
        vel_cands.append((radar_vel, 1.0))
    if lidar_vel is not None:
        vel_cands.append((lidar_vel, 1.2))

    pos_meas = _robust_vec(pos_cands)
    vel_meas = _robust_vec(vel_cands)
    return pos_meas, vel_meas, range_meas, cam_pos

def _alpha_beta_update(r: List[float], v: List[float], pos_meas: Optional[List[float]], vel_meas: Optional[List[float]],
                       range_meas: Optional[float], dt: float,
                       age_pos: float, age_vel: float) -> Tuple[List[float], List[float], float, float]:
    """
    Simple alpha-beta(-gamma) style tracker with magnitude correction using scalar range sensors.
    """
    dt = max(1e-3, float(dt))

    # Predict
    r_pred = _add3(r, _mul3(v, dt))
    v_pred = list(v)

    # Gain scheduling: trust measurements more when recently valid, and when close in.
    # If measurements have been missing, reduce gains to avoid latching onto spikes.
    base_alpha = 0.65
    base_beta = 0.25
    base_gamma = 0.55  # for direct velocity measurement blend

    # Update with position measurement
    if pos_meas is not None:
        innov = _sub3(pos_meas, r_pred)
        # Slightly lower gains if we've been "blind"
        blind_factor = _clamp(1.0 / (1.0 + 0.08 * age_pos), 0.25, 1.0)
        alpha = base_alpha * blind_factor
        beta = base_beta * blind_factor

        r_upd = _add3(r_pred, _mul3(innov, alpha))
        v_upd = _add3(v_pred, _mul3(innov, beta / dt))
        age_pos = 0.0
    else:
        r_upd = r_pred
        v_upd = v_pred
        age_pos += dt

    # Update with velocity measurement
    if vel_meas is not None:
        blind_factor = _clamp(1.0 / (1.0 + 0.08 * age_vel), 0.25, 1.0)
        gamma = base_gamma * blind_factor
        v_upd = _add3(v_upd, _mul3(_sub3(vel_meas, v_upd), gamma))
        age_vel = 0.0
    else:
        age_vel += dt

    # Scalar range correction (helps if vector sensors flicker)
    if range_meas is not None and math.isfinite(range_meas) and range_meas > 0:
        rmag = _norm3(r_upd)
        # only apply if not wildly inconsistent (reject big outliers)
        if rmag > 1e-3 and abs(rmag - range_meas) < max(30.0, 0.35 * rmag):
            # pull magnitude gently towards scalar range
            u = _unit3(r_upd)
            r_target = _mul3(u, float(range_meas))
            r_upd = _add3(r_upd, _mul3(_sub3(r_target, r_upd), 0.35))

    return r_upd, v_upd, age_pos, age_vel

# ------------------------ attitude hold (rate damping) ------------------------
def _gyro_rate_estimate(state: Dict) -> Tuple[float, float, float]:
    """
    Robust median of 3 gyros. Input keys are deg/s; output rad/s.
    """
    def gather(axis: str) -> Optional[float]:
        vals = []
        for i in (1, 2, 3):
            vkey = f"gyro_{i}_{axis}_dps"
            okkey = f"gyro_{i}_{axis}_valid"
            if _valid(state, okkey):
                v = _get(state, vkey, None)
                if v is not None and math.isfinite(float(v)):
                    vals.append(float(v))
        med = _median(vals)
        if med is None:
            return None
        return med * math.pi / 180.0

    pitch = gather("pitch")
    roll  = gather("roll")
    yaw   = gather("yaw")

    # If missing, assume 0 (better than NaN for a rate-hold)
    return (roll or 0.0), (pitch or 0.0), (yaw or 0.0)

# ------------------------ guidance & control ------------------------
def _approach_speed_limit(range_m: float) -> float:
    # Conservative limits aimed at the success criteria and "final gentle" requirement.
    if range_m > 200:
        return 0.7
    if range_m > 50:
        return 0.35
    if range_m > 10:
        return 0.18
    if range_m > 2:
        return 0.08
    return 0.05

def _fuel_scale(state: Dict) -> float:
    kg = _get(state, "fuel_remaining_kg", None)
    pct = _get(state, "fuel_remaining_pct", None)
    s = 1.0
    if kg is not None and math.isfinite(float(kg)):
        if float(kg) < 60.0:
            s *= 0.35
        elif float(kg) < 120.0:
            s *= 0.55
    if pct is not None and math.isfinite(float(pct)):
        if float(pct) < 0.12:
            s *= 0.6
    return _clamp(s, 0.25, 1.0)

def _compute_thrust_cmd(r: List[float], v: List[float], state: Dict) -> List[float]:
    """
    Output command in normalized thrust units [-1..+1] in LVLH axes.
    Implements:
      - FAR (>200m): "phasing" along-track using orbital insight (if behind -> retrograde)
      - MID/CLOSE/FINAL: increasingly direct PD, with strict speed limiting
    """
    x, y, z = r
    vx, vy, vz = v
    rng = max(1e-6, _norm3(r))

    # If we have docking camera, use it as a strong hint to null lateral (x,z) errors near the port.
    # (Cam measurement already fused into r, but we tighten corridor when camera is active.)
    cam_active = bool(_get(state, "dock_camera_active", False))

    # Phase selection
    if rng > 200.0:
        phase = "FAR"
    elif rng > 50.0:
        phase = "MID"
    elif rng > 10.0:
        phase = "CLOSE"
    else:
        phase = "FINAL"

    fuel_s = _fuel_scale(state)

    # Hard cap on total authority to reduce wasted propellant + oscillations
    if phase == "FAR":
        umax = 0.18
    elif phase == "MID":
        umax = 0.22
    elif phase == "CLOSE":
        umax = 0.16
    else:
        umax = 0.10
    umax *= fuel_s

    # --- Cross-track & radial: keep small, damp velocity (avoid exciting oscillations) ---
    # corridor tightens as we get closer and if camera is active
    if phase == "FAR":
        corr_xz = 25.0
    elif phase == "MID":
        corr_xz = 8.0
    elif phase == "CLOSE":
        corr_xz = 2.5
    else:
        corr_xz = 0.7
    if cam_active and rng < 80:
        corr_xz *= 0.7

    # soft-saturation position error to avoid huge commands from large offsets
    ex = _clamp(x / max(1.0, corr_xz), -3.0, 3.0)
    ez = _clamp(z / max(1.0, corr_xz), -3.0, 3.0)

    # Gains tuned to be gentle (normalized thruster units are unknown -> keep small)
    kpx = 0.10 if phase in ("CLOSE", "FINAL") else 0.06
    kdx = 0.22 if phase in ("CLOSE", "FINAL") else 0.14
    kpz = kpx
    kdz = kdx

    ux = -kpx * ex - kdx * vx
    uz = -kpz * ez - kdz * vz

    # --- Along-track (y): apply orbital-mechanics-aware phasing far away ---
    # CRITICAL: if behind (y<0), use RETROGRADE (-Y) to lower orbit and catch up.
    # If ahead (y>0), use PROGRADE (+Y) to raise orbit and let station catch up.
    if phase == "FAR":
        # "phasing burn" magnitude based mostly on along-track separation
        # Keep small to avoid waste; damp vy to prevent runaway closing.
        y_scale = max(200.0, min(2500.0, abs(y)))
        base = _clamp(abs(y) / y_scale, 0.0, 1.0)

        # sign(y): behind -> negative thrust_y (retrograde), ahead -> positive (prograde)
        phasing = 0.14 * (1.0 if y > 0 else -1.0) * base

        # If we are already closing rapidly (vy toward reducing |y|), back off
        # For behind (y<0): "closing" means vy > 0 (y increasing toward 0)
        # For ahead (y>0): "closing" means vy < 0
        closing_component = (-vy) if y > 0 else (vy)
        phasing *= _clamp(1.0 - 0.6 * max(0.0, closing_component), 0.1, 1.0)

        # Damping (always opposes current vy)
        uy = phasing - 0.10 * vy

        # Do not try to "thrust toward station" in y at long range; let orbital dynamics do it.
        # Also avoid too much y-thrust when close enough that linear control takes over.
        if abs(y) < 80.0:
            uy *= 0.4

    else:
        # MID/CLOSE/FINAL: direct approach with speed limit
        vlim = _approach_speed_limit(rng)

        # Desired velocity: point toward origin, but prefer V-bar-ish (mostly y) when still mid-range.
        udir = _unit3(_mul3(r, -1.0))  # toward target
        v_des = _mul3(udir, vlim)

        # If we're mid-range and mostly behind/ahead, bias toward along-track closure and keep x,z small.
        if phase == "MID" and abs(y) > (abs(x) + abs(z)) and abs(y) > 20:
            v_des[0] *= 0.6
            v_des[2] *= 0.6

        # PD in velocity and position (position term kept weak; helps reject biases/uncommanded thrust)
        kpy = 0.08 if phase == "MID" else 0.10
        kdy = 0.28 if phase in ("CLOSE", "FINAL") else 0.22

        # Use a small position term in y to eliminate steady offsets; in close range, this is safe.
        uy = -kpy * _clamp(y / max(1.0, (40.0 if phase == "MID" else 10.0)), -3.0, 3.0) - kdy * (vy - v_des[1])

        # Also apply speed-brake if total speed exceeds limit significantly
        speed = _norm3(v)
        if speed > (1.6 * vlim + 0.05):
            # brake opposite velocity
            b = _unit3(v)
            ux += -0.25 * b[0]
            uy += -0.25 * b[1]
            uz += -0.25 * b[2]

    # Clip per-axis and total (to reduce propellant waste)
    ux = _clamp(ux, -umax, umax)
    uy = _clamp(uy, -umax, umax)
    uz = _clamp(uz, -umax, umax)

    # Total norm cap
    mag = math.sqrt(ux*ux + uy*uy + uz*uz)
    if mag > umax and mag > 1e-9:
        s = umax / mag
        ux *= s; uy *= s; uz *= s

    return [ux, uy, uz]

# ------------------------ public API ------------------------
def autopilot_step(state: dict, dt: float) -> dict:
    global _AP
    dt = float(dt) if dt is not None else 0.1
    dt = max(1e-3, min(dt, 1.0))  # guard against absurd dt

    # Initialize estimator from best available measurement
    pos_meas, vel_meas, range_meas, cam_pos = _collect_rel_pos_vel_measurements(state)

    if not _AP["init"]:
        if pos_meas is not None:
            _AP["r"] = list(pos_meas)
        else:
            # If only camera or scalar range exists, best effort:
            if cam_pos is not None:
                _AP["r"] = list(cam_pos)
            elif range_meas is not None:
                _AP["r"] = [0.0, -float(range_meas), 0.0]  # assume "behind" if nothing else
        if vel_meas is not None:
            _AP["v"] = list(vel_meas)
        else:
            _AP["v"] = [0.0, 0.0, 0.0]
        _AP["init"] = True
        _AP["age_pos"] = 0.0 if pos_meas is not None else 1e9
        _AP["age_vel"] = 0.0 if vel_meas is not None else 1e9

    # Filter update
    r, v, age_pos, age_vel = _alpha_beta_update(
        _AP["r"], _AP["v"], pos_meas, vel_meas, range_meas, dt, _AP["age_pos"], _AP["age_vel"]
    )
    _AP["r"], _AP["v"] = r, v
    _AP["age_pos"], _AP["age_vel"] = age_pos, age_vel
    _AP["last_range"] = _norm3(r)

    # Guidance & thrust
    thrust = _compute_thrust_cmd(r, v, state)
    thrust_x, thrust_y, thrust_z = thrust[0], thrust[1], thrust[2]

    # Attitude: rate hold (damp to 0). (We are not given a docking attitude target.)
    roll_rate, pitch_rate, yaw_rate = _gyro_rate_estimate(state)
    # P controller on measured body rates to command attitude rates (within limits)
    # Output is desired att rate; here we command opposite the current rate to stop rotation.
    k_rate = 0.85
    att_rate_roll  = _clamp(-k_rate * roll_rate,  -0.035, 0.035)
    att_rate_pitch = _clamp(-k_rate * pitch_rate, -0.035, 0.035)
    att_rate_yaw   = _clamp(-k_rate * yaw_rate,   -0.035, 0.035)

    return {
        "thrust_x": float(_clamp(thrust_x, -1.0, 1.0)),
        "thrust_y": float(_clamp(thrust_y, -1.0, 1.0)),
        "thrust_z": float(_clamp(thrust_z, -1.0, 1.0)),
        "att_rate_roll": float(att_rate_roll),
        "att_rate_pitch": float(att_rate_pitch),
        "att_rate_yaw": float(att_rate_yaw),
    }
