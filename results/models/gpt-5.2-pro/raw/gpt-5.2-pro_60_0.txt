```javascript
'use strict';

const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input ? JSON.parse(input) : {};

const machineCount = Math.max(1, data.machineCount | 0);
const tasks = Array.isArray(data.tasks) ? data.tasks : [];
const n = tasks.length;

if (n === 0) {
  process.stdout.write(JSON.stringify({ order: [] }));
  process.exit(0);
}

// ---------- Build id -> index ----------
const idToIdx = new Map();
const ids = new Array(n);
const idStr = new Array(n);
for (let i = 0; i < n; i++) {
  const id = tasks[i].id;
  ids[i] = id;
  idStr[i] = String(id);
  idToIdx.set(id, i);
}

// ---------- Read basic fields, count edges ----------
const dur = new Float64Array(n);
const weight = new Float64Array(n);
const deadline = new Float64Array(n);
const hasDeadline = new Uint8Array(n);

let totalDur = 0;
let maxDur = 0;
let maxDeadline = -Infinity;

const indeg = new Int32Array(n);
const outCount = new Int32Array(n);

for (let i = 0; i < n; i++) {
  const t = tasks[i];

  const d = Number.isFinite(t.duration) ? t.duration : 1;
  dur[i] = d;
  totalDur += d;
  if (d > maxDur) maxDur = d;

  const w = Number.isFinite(t.weight) ? t.weight : 1;
  weight[i] = w > 0 ? w : 1;

  const dl = t.deadline;
  if (Number.isFinite(dl)) {
    deadline[i] = dl;
    hasDeadline[i] = 1;
    if (dl > maxDeadline) maxDeadline = dl;
  } else {
    deadline[i] = NaN;
    hasDeadline[i] = 0;
  }

  const deps = Array.isArray(t.deps) ? t.deps : [];
  indeg[i] = deps.length;

  for (let k = 0; k < deps.length; k++) {
    const depId = deps[k];
    const depIdx = idToIdx.get(depId);
    if (depIdx !== undefined) {
      outCount[depIdx]++;
    } else {
      // Unknown dependency id: ignore (keeps solver robust on imperfect inputs)
      // but still counts as dependency in indeg; better: treat as absent.
      indeg[i]--;
    }
  }
}

if (!Number.isFinite(maxDeadline)) maxDeadline = 0;
const pseudoDeadline =
  maxDeadline +
  Math.ceil(totalDur / Math.max(1, machineCount)) +
  maxDur;

// ---------- Build CSR adjacency (outgoing edges) ----------
const outStart = new Int32Array(n + 1);
for (let i = 0; i < n; i++) outStart[i + 1] = outStart[i] + outCount[i];
const m = outStart[n];
const outEdges = new Int32Array(m);
const cursor = outStart.slice(0, n);

for (let i = 0; i < n; i++) {
  const deps = Array.isArray(tasks[i].deps) ? tasks[i].deps : [];
  for (let k = 0; k < deps.length; k++) {
    const depIdx = idToIdx.get(deps[k]);
    if (depIdx === undefined) continue;
    const pos = cursor[depIdx]++;
    outEdges[pos] = i;
  }
}

// ---------- Topological order (for b-level computation) ----------
const indegTopo = indeg.slice();
const q = new Int32Array(n);
let qh = 0, qt = 0;
for (let i = 0; i < n; i++) if (indegTopo[i] === 0) q[qt++] = i;

const topo = new Int32Array(n);
let tk = 0;
while (qh < qt) {
  const v = q[qh++];
  topo[tk++] = v;
  for (let e = outStart[v]; e < outStart[v + 1]; e++) {
    const u = outEdges[e];
    const nd = --indegTopo[u];
    if (nd === 0) q[qt++] = u;
  }
}
// If tk < n there is a cycle; we still proceed with partial topo.

// ---------- Compute b-level (longest path to exit, including own duration) ----------
const bLevel = new Float64Array(n);
for (let ii = tk - 1; ii >= 0; ii--) {
  const v = topo[ii];
  let best = 0;
  for (let e = outStart[v]; e < outStart[v + 1]; e++) {
    const u = outEdges[e];
    const bl = bLevel[u];
    if (bl > best) best = bl;
  }
  bLevel[v] = dur[v] + best;
}
// Nodes in cycles (if any) keep bLevel = 0; that’s fine for fallback ordering.

// ---------- Priority key: latestStart = effectiveDeadline - bLevel ----------
const latestStart = new Float64Array(n);
for (let i = 0; i < n; i++) {
  const effDl = hasDeadline[i] ? deadline[i] : pseudoDeadline;
  latestStart[i] = effDl - bLevel[i];
}

// ---------- Binary heap for ready tasks (indices) ----------
class IntHeap {
  constructor(less) {
    this.a = [];
    this.less = less; // less(x,y) => x should be above y
  }
  size() { return this.a.length; }
  peek() { return this.a[0]; }
  push(x) {
    const a = this.a;
    a.push(x);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (!this.less(a[i], a[p])) break;
      const tmp = a[i]; a[i] = a[p]; a[p] = tmp;
      i = p;
    }
  }
  pop() {
    const a = this.a;
    const root = a[0];
    const last = a.pop();
    if (a.length) {
      a[0] = last;
      let i = 0;
      for (;;) {
        const l = i * 2 + 1;
        if (l >= a.length) break;
        const r = l + 1;
        let c = l;
        if (r < a.length && this.less(a[r], a[l])) c = r;
        if (!this.less(a[c], a[i])) break;
        const tmp = a[i]; a[i] = a[c]; a[c] = tmp;
        i = c;
      }
    }
    return root;
  }
}

// Compare by (latestStart/weight) ascending, then bLevel desc, then weight desc, then dur asc, then id
function readyLess(i, j) {
  // Compare (latestStart[i] / weight[i]) < (latestStart[j] / weight[j]) via cross-multiply
  const lhs = latestStart[i] * weight[j];
  const rhs = latestStart[j] * weight[i];
  if (lhs !== rhs) return lhs < rhs;

  const bi = bLevel[i], bj = bLevel[j];
  if (bi !== bj) return bi > bj;

  const wi = weight[i], wj = weight[j];
  if (wi !== wj) return wi > wj;

  const di = dur[i], dj = dur[j];
  if (di !== dj) return di < dj;

  return idStr[i] < idStr[j];
}

// ---------- Produce a topological order with the priority heap ----------
const rem = indeg.slice();
const ready = new IntHeap(readyLess);
for (let i = 0; i < n; i++) if (rem[i] === 0) ready.push(i);

const order = new Array(n);
let outPos = 0;

while (ready.size()) {
  const v = ready.pop();
  order[outPos++] = ids[v];
  for (let e = outStart[v]; e < outStart[v + 1]; e++) {
    const u = outEdges[e];
    const nd = --rem[u];
    if (nd === 0) ready.push(u);
  }
}

// Cycle / invalid deps fallback: append remaining in deterministic order
if (outPos < n) {
  const left = [];
  for (let i = 0; i < n; i++) if (rem[i] > 0) left.push(i);
  left.sort((a, b) => (idStr[a] < idStr[b] ? -1 : idStr[a] > idStr[b] ? 1 : 0));
  for (let k = 0; k < left.length; k++) order[outPos++] = ids[left[k]];
}

process.stdout.write(JSON.stringify({ order }));
```
