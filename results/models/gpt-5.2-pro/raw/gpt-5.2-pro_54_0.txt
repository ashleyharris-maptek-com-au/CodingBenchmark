```python
# Autoland / ILS autopilot (simulation-oriented).
# Not for real aircraft use.

from __future__ import annotations
import math
from typing import Dict, Any, Optional, List, Tuple

# ---------- persistent module state (reset by the harness between scenarios/attempts) ----------
_AP = {
    "init": False,
    "mode": "APPROACH",  # APPROACH | GO_AROUND | ROLLOUT
    "loc_f": 0.0,
    "gs_f": 0.0,
    "loc_last": 0.0,
    "gs_last": 0.0,
    "loc_int": 0.0,
    "gs_int": 0.0,
    "spd_int": 0.0,
    "unstable_time": 0.0,
    "flare_started": False,
    "last_ra": None,
}


# ---------- helpers ----------
def _clamp(x: float, lo: float, hi: float) -> float:
    return lo if x < lo else hi if x > hi else x


def _isfinite(x: Any) -> bool:
    return isinstance(x, (int, float)) and math.isfinite(x)


def _wrap180(deg: float) -> float:
    # returns in [-180, 180)
    deg = (deg + 180.0) % 360.0 - 180.0
    return deg


def _median(xs: List[float]) -> float:
    xs = sorted(xs)
    n = len(xs)
    if n == 0:
        return 0.0
    if n % 2 == 1:
        return xs[n // 2]
    return 0.5 * (xs[n // 2 - 1] + xs[n // 2])


def _robust_vote(values: List[float], last: Optional[float], jump_gate: float, pair_gate: float) -> Optional[float]:
    """
    Pick a robust estimate:
      - drop values that jump too far from last (if last exists)
      - if >=2 remaining, prefer the closest pair if they agree enough
      - else median
    """
    vs = [v for v in values if _isfinite(v)]
    if not vs:
        return None

    if last is not None and _isfinite(last):
        vs2 = [v for v in vs if abs(v - last) <= jump_gate]
        if vs2:
            vs = vs2  # only gate if it doesn't remove everything

    if len(vs) == 1:
        return vs[0]

    # find closest pair
    best = None
    best_d = 1e9
    for i in range(len(vs)):
        for j in range(i + 1, len(vs)):
            d = abs(vs[i] - vs[j])
            if d < best_d:
                best_d = d
                best = (vs[i], vs[j])
    if best is not None and best_d <= pair_gate:
        return 0.5 * (best[0] + best[1])

    return _median(vs)


def _fuse_3(state: Dict[str, Any], base: str, suffix: str, last: Optional[float],
            jump_gate: float, pair_gate: float) -> Optional[float]:
    vals = []
    for i in (1, 2, 3):
        v = state.get(f"{base}_{i}_{suffix}")
        valid = state.get(f"{base}_{i}_valid", True)
        if valid and _isfinite(v):
            vals.append(float(v))
    return _robust_vote(vals, last=last, jump_gate=jump_gate, pair_gate=pair_gate)


def _fuse_irs(state: Dict[str, Any], field: str, last: Optional[float] = None) -> Optional[float]:
    vals = []
    for i in (1, 2, 3):
        valid = state.get(f"irs_{i}_valid", True)
        v = state.get(f"irs_{i}_{field}")
        if valid and _isfinite(v):
            vals.append(float(v))
    # IRS can legitimately change quickly in turbulence; use looser gating
    return _robust_vote(vals, last=last, jump_gate=50.0 if field.endswith("_deg") else 5000.0, pair_gate=5.0 if field.endswith("_deg") else 200.0)


def _fuse_baro_alt(state: Dict[str, Any], last: Optional[float]) -> Optional[float]:
    vals = []
    for i in (1, 2, 3):
        valid = state.get(f"baro_alt_{i}_valid", True)
        v = state.get(f"baro_alt_{i}_ft")
        if valid and _isfinite(v):
            vals.append(float(v))
    return _robust_vote(vals, last=last, jump_gate=200.0, pair_gate=30.0)


def _fuse_ias(state: Dict[str, Any], last: Optional[float]) -> Optional[float]:
    vals = []
    for i in (1, 2, 3):
        valid = state.get(f"ias_{i}_valid", True)
        v = state.get(f"ias_{i}_kt")
        if valid and _isfinite(v):
            vals.append(float(v))
    return _robust_vote(vals, last=last, jump_gate=30.0, pair_gate=8.0)


def _get_dme(state: Dict[str, Any]) -> Optional[float]:
    d1 = state.get("dme_1_nm")
    d2 = state.get("dme_2_nm")
    ds = [d for d in (d1, d2) if _isfinite(d)]
    if not ds:
        return None
    return float(min(ds))


def _radio_alt_valid(ra: Any) -> bool:
    # no explicit valid flag; treat obviously-wrong values as invalid
    if not _isfinite(ra):
        return False
    ra = float(ra)
    return -20.0 <= ra <= 5000.0


# ---------- main autopilot ----------
def autopilot_step(state: dict, dt: float) -> dict:
    global _AP
    dt = float(dt) if _isfinite(dt) and dt > 0 else 0.02

    if not _AP["init"]:
        _AP["init"] = True
        _AP["mode"] = "APPROACH"
        _AP["loc_f"] = 0.0
        _AP["gs_f"] = 0.0
        _AP["loc_last"] = 0.0
        _AP["gs_last"] = 0.0
        _AP["loc_int"] = 0.0
        _AP["gs_int"] = 0.0
        _AP["spd_int"] = 0.0
        _AP["unstable_time"] = 0.0
        _AP["flare_started"] = False
        _AP["last_ra"] = None

    # --- fused/nav inputs ---
    loc = _fuse_3(state, "localizer", "dots", last=_AP["loc_last"], jump_gate=1.8, pair_gate=0.7)
    gs = _fuse_3(state, "glideslope", "dots", last=_AP["gs_last"], jump_gate=1.8, pair_gate=0.7)

    # low-pass filter to handle flicker/turbulence
    tau = 0.7  # seconds
    alpha = _clamp(dt / (tau + dt), 0.0, 1.0)
    if loc is not None:
        _AP["loc_f"] = (1 - alpha) * _AP["loc_f"] + alpha * loc
        _AP["loc_last"] = loc
    # if invalid: hold last filtered value (do not slam to 0)

    if gs is not None:
        _AP["gs_f"] = (1 - alpha) * _AP["gs_f"] + alpha * gs
        _AP["gs_last"] = gs

    loc_f = float(_AP["loc_f"])
    gs_f = float(_AP["gs_f"])

    ra_raw = state.get("radio_alt_ft")
    ra_ok = _radio_alt_valid(ra_raw)
    ra = float(ra_raw) if ra_ok else None

    dme = _get_dme(state)

    roll = _fuse_irs(state, "roll_deg", last=None) or 0.0
    pitch = _fuse_irs(state, "pitch_deg", last=None) or 0.0
    vs_fpm = _fuse_irs(state, "vs_fpm", last=None) or 0.0
    track = _fuse_irs(state, "track_deg", last=None)  # can be None
    hdg_tgt = state.get("target_hdg_deg")
    hdg_tgt = float(hdg_tgt) if _isfinite(hdg_tgt) else (track if track is not None else 0.0)

    ias = _fuse_ias(state, last=None)
    if ias is None:
        ias = float(state.get("target_ias_kt", 140.0)) if _isfinite(state.get("target_ias_kt")) else 140.0

    groundspeed = state.get("groundspeed_kt")
    groundspeed = float(groundspeed) if _isfinite(groundspeed) else ias

    wow = bool(state.get("weight_on_wheels", False)) or bool(state.get("on_ground", False))

    # engine status
    e1_run = bool(state.get("engine_1_running", True))
    e2_run = bool(state.get("engine_2_running", True))
    any_engine = e1_run or e2_run
    dual_engine_out = (not e1_run) and (not e2_run)

    # --- phase/mode logic ---
    # flare trigger: primarily RA, else DME fallback if RA failed
    flare_now = False
    if wow:
        flare_now = False
    elif ra_ok and ra is not None:
        flare_now = ra < 50.0
    else:
        # RA failed: approximate flare very near threshold
        # (DME can be 0-ish at threshold; choose conservative start)
        flare_now = (dme is not None and dme < 0.18)

    if wow:
        _AP["mode"] = "ROLLOUT"
    elif _AP["mode"] != "GO_AROUND":
        _AP["mode"] = "APPROACH"

    # --- configuration targets ---
    vref = state.get("target_vref_kt")
    vref = float(vref) if _isfinite(vref) else 137.0

    # simple wind additive estimate: if GS significantly lower than IAS, assume headwind
    headwind_est = _clamp(ias - groundspeed, 0.0, 30.0)
    wind_add = _clamp(0.5 * headwind_est, 0.0, 10.0)

    # approach speed schedule
    if wow or _AP["mode"] == "ROLLOUT":
        v_tgt = max(0.0, min(80.0, ias))  # irrelevant after touchdown
    elif flare_now:
        v_tgt = vref + 0.0
    else:
        v_tgt = vref + 5.0 + wind_add

    # flap/gear schedule using DME/RA (keep simple and robust)
    flaps_cmd = 30.0
    gear_cmd = 1.0
    if _AP["mode"] == "GO_AROUND":
        # go-around configuration (typical: reduce to a safe intermediate flap)
        flaps_cmd = 15.0
        gear_cmd = 0.0 if (ra_ok and ra is not None and ra > 100.0) else 1.0
    elif wow:
        flaps_cmd = 40.0
        gear_cmd = 1.0
    else:
        if (dme is not None and dme > 8.0) and (not (ra_ok and ra is not None and ra < 2500.0)):
            flaps_cmd = 5.0
            gear_cmd = 0.0
        elif (dme is not None and dme > 5.0) and (not (ra_ok and ra is not None and ra < 1800.0)):
            flaps_cmd = 15.0
            gear_cmd = 0.0
        else:
            flaps_cmd = 30.0
            gear_cmd = 1.0
        # if close-in and slow enough, command full landing flaps
        if (ra_ok and ra is not None and ra < 700.0) or (dme is not None and dme < 3.5):
            if ias <= vref + 15.0:
                flaps_cmd = 40.0

    # speedbrake usage: only to shed excess speed when high on energy, never in flare/rollout
    speedbrake_cmd = 0.0
    if _AP["mode"] == "APPROACH" and (not flare_now):
        if (ra_ok and ra is not None and ra > 250.0) or (ra is None):
            if ias > v_tgt + 18.0:
                speedbrake_cmd = _clamp((ias - (v_tgt + 18.0)) / 25.0, 0.0, 0.6)

    # --- stability / go-around decision ---
    # Define "stable" mainly for below 1000 ft AGL if RA valid, else below ~3nm.
    below_1000 = (ra_ok and ra is not None and ra < 1000.0) or (ra is None and dme is not None and dme < 3.0)
    below_DH = bool(state.get("below_decision_height", False))

    loc_ok_for_stable = abs(loc_f) < 1.2
    gs_ok_for_stable = abs(gs_f) < 1.2
    spd_ok_for_stable = (ias < v_tgt + 12.0) and (ias > vref - 8.0)
    sink_ok_for_stable = (vs_fpm > -1400.0)  # avoid extreme sink on short final
    config_ok = (gear_cmd >= 0.5) and (flaps_cmd >= 25.0)

    stable = loc_ok_for_stable and gs_ok_for_stable and spd_ok_for_stable and sink_ok_for_stable and config_ok

    # Only accumulate unstable time in the region where it matters.
    if _AP["mode"] == "APPROACH" and below_1000 and not stable and not wow:
        _AP["unstable_time"] += dt
    else:
        _AP["unstable_time"] = max(0.0, _AP["unstable_time"] - 2.0 * dt)

    # Go-around rules:
    # - If below DH and unstable for > ~1.5s, go-around (unless dual engine failure).
    # - Also go-around if grossly off scale.
    grossly_unstable = (abs(loc_f) > 2.5) or (abs(gs_f) > 2.5) or (vs_fpm < -2000.0 and (ra_ok and ra is not None and ra < 800.0))
    allow_go_around = any_engine and (not dual_engine_out)

    if _AP["mode"] != "GO_AROUND" and allow_go_around and not wow:
        if (below_DH and _AP["unstable_time"] > 1.5) or (below_1000 and grossly_unstable and _AP["unstable_time"] > 0.5):
            _AP["mode"] = "GO_AROUND"
            _AP["flare_started"] = False
            # reset integrators to avoid fighting the maneuver
            _AP["loc_int"] = 0.0
            _AP["gs_int"] = 0.0
            _AP["spd_int"] = 0.0

    # --- control laws ---
    # Lateral: LOC -> bank demand; add small track alignment term.
    track_err = 0.0
    if track is not None and _isfinite(track):
        track_err = _wrap180(float(hdg_tgt) - float(track))

    if _AP["mode"] == "ROLLOUT":
        desired_bank = 0.0
    elif _AP["mode"] == "GO_AROUND":
        desired_bank = _clamp(-0.8 * track_err, -10.0, 10.0)
    else:
        # localizer: positive dots = right of course -> bank left (negative)
        desired_bank = _clamp(-10.0 * loc_f - 0.4 * track_err, -25.0, 25.0)

    # Aileron to achieve desired bank
    bank_err = desired_bank - roll
    # integrate only when not saturated and not in flare/rollout
    if _AP["mode"] == "APPROACH" and not flare_now:
        _AP["loc_int"] = _clamp(_AP["loc_int"] + dt * _clamp(loc_f, -2.0, 2.0), -10.0, 10.0)
    else:
        _AP["loc_int"] *= (1.0 - _clamp(2.0 * dt, 0.0, 1.0))

    aileron = 0.045 * bank_err + 0.015 * _AP["loc_int"]
    aileron = _clamp(aileron, -1.0, 1.0)

    # Rudder: track alignment + asymmetric thrust compensation; more authority on rollout
    thrust1 = state.get("engine_1_thrust_pct")
    thrust2 = state.get("engine_2_thrust_pct")
    t1 = float(thrust1) if _isfinite(thrust1) else (50.0 if e1_run else 0.0)
    t2 = float(thrust2) if _isfinite(thrust2) else (50.0 if e2_run else 0.0)
    asym = _clamp((t2 - t1) / 100.0, -1.0, 1.0)  # + means right engine stronger => yaw left needed (negative rudder)
    rudder_asym = _clamp(-0.35 * asym, -0.4, 0.4)

    if _AP["mode"] == "ROLLOUT":
        rudder = _clamp(0.05 * track_err + rudder_asym, -1.0, 1.0)
    elif _AP["mode"] == "GO_AROUND":
        rudder = _clamp(0.02 * track_err + rudder_asym, -0.6, 0.6)
    else:
        rudder = _clamp(0.015 * track_err + rudder_asym, -0.5, 0.5)

    # Vertical: GS -> desired VS; elevator to track VS + GS bias.
    tgt_vs = state.get("target_vs_fpm")
    tgt_vs = float(tgt_vs) if _isfinite(tgt_vs) else -700.0

    if _AP["mode"] == "GO_AROUND":
        desired_vs = +1200.0
    elif flare_now and not wow:
        # flare: progressively reduce sink
        if ra_ok and ra is not None:
            flare_prog = _clamp((50.0 - ra) / 50.0, 0.0, 1.0)
        else:
            flare_prog = 1.0  # if we're using DME fallback, commit to flare
        desired_vs = (1.0 - flare_prog) * tgt_vs + flare_prog * (-60.0)
    else:
        # if above GS (gs_f>0), command more descent (more negative VS)
        desired_vs = tgt_vs - 240.0 * _clamp(gs_f, -2.5, 2.5)

    vs_err = desired_vs - vs_fpm

    # integrate for steady biases (jammed elevator / heavy landing)
    if _AP["mode"] == "APPROACH" and not flare_now:
        _AP["gs_int"] = _clamp(_AP["gs_int"] + dt * _clamp(-gs_f, -1.5, 1.5), -0.8, 0.8)
    else:
        _AP["gs_int"] *= (1.0 - _clamp(3.0 * dt, 0.0, 1.0))

    # elevator: positive => pitch up
    elevator = 0.0032 * vs_err + (-0.34 * gs_f) + _AP["gs_int"]

    # flare shaping (add a gentle pitch-up as we enter flare)
    if flare_now and not wow and _AP["mode"] == "APPROACH":
        _AP["flare_started"] = True
    if _AP["mode"] == "APPROACH" and _AP["flare_started"] and not wow:
        # keep this modest to avoid ballooning
        elevator += 0.10

    if _AP["mode"] == "GO_AROUND":
        # pitch to a safe go-around attitude (roughly)
        elevator = max(elevator, 0.35)

    elevator = _clamp(elevator, -1.0, 1.0)

    # Trim: slowly bias out sustained elevator demand when not near the ground
    elevator_trim = 0.0
    if _AP["mode"] == "APPROACH" and (ra_ok and ra is not None and ra > 300.0):
        # Use a slow integrator to keep elevator from sitting at a constant offset.
        # (Output trim is independent; simulator interprets elevator as relative to trim.)
        trim = state.get("elevator_trim", 0.0)
        trim = float(trim) if _isfinite(trim) else 0.0
        elevator_trim = _clamp(trim + 0.04 * dt * _clamp(elevator, -0.6, 0.6), -1.0, 1.0)
    else:
        # don't chase trim in flare/rollout/go-around
        elevator_trim = 0.0

    # Throttle (autothrottle-like PI on IAS), with flare idle and go-around TOGA
    speed_err = v_tgt - ias

    if _AP["mode"] == "ROLLOUT":
        thr = 0.0
    elif _AP["mode"] == "GO_AROUND":
        thr = 1.0
    elif flare_now:
        # smoothly close thrust; never increase in flare
        if ra_ok and ra is not None:
            idle_prog = _clamp((40.0 - ra) / 25.0, 0.0, 1.0)  # start pulling at ~40ft
        else:
            idle_prog = 1.0
        thr = (1.0 - idle_prog) * 0.25 + idle_prog * 0.0
    else:
        _AP["spd_int"] = _clamp(_AP["spd_int"] + dt * _clamp(speed_err, -15.0, 15.0), -0.35, 0.35)
        # base thrust depends on descent; a bit more if sinking fast
        base = 0.42 + _clamp((-vs_fpm - 700.0) / 4000.0, -0.05, 0.18)
        thr = base + 0.018 * speed_err + _AP["spd_int"]
        thr = _clamp(thr, 0.0, 1.0)

    # Split throttles for engine-out: command 0 on dead engine, thr on live engine.
    throttle_1 = thr if e1_run else 0.0
    throttle_2 = thr if e2_run else 0.0

    # Ground/rollout devices
    spoiler = 0.0
    autobrake = 0.0
    reverse_1 = 0.0
    reverse_2 = 0.0

    if wow or _AP["mode"] == "ROLLOUT":
        spoiler = 1.0
        speedbrake_cmd = 1.0
        # stronger braking if still fast
        gs = float(state.get("groundspeed_kt", 0.0)) if _isfinite(state.get("groundspeed_kt")) else 0.0
        autobrake = 0.85 if gs > 120.0 else 0.65
        if gs > 60.0:
            reverse_1 = 1.0 if e1_run else 0.0
            reverse_2 = 1.0 if e2_run else 0.0
        elif gs > 30.0:
            reverse_1 = 0.4 if e1_run else 0.0
            reverse_2 = 0.4 if e2_run else 0.0
        else:
            reverse_1 = 0.0
            reverse_2 = 0.0

    # In go-around, ensure drag devices are stowed
    if _AP["mode"] == "GO_AROUND":
        speedbrake_cmd = 0.0
        spoiler = 0.0
        reverse_1 = reverse_2 = 0.0
        autobrake = 0.0

    # Final command package
    return {
        "elevator": float(elevator),
        "aileron": float(aileron),
        "rudder": float(rudder),
        "throttle_1": float(_clamp(throttle_1, 0.0, 1.0)),
        "throttle_2": float(_clamp(throttle_2, 0.0, 1.0)),
        "elevator_trim": float(elevator_trim),
        "flaps": float(_clamp(flaps_cmd, 0.0, 40.0)),
        "gear": float(1.0 if gear_cmd >= 0.5 else 0.0),
        "speedbrake": float(_clamp(speedbrake_cmd, 0.0, 1.0)),
        "spoiler": float(1.0 if spoiler >= 0.5 else 0.0),
        "autobrake": float(_clamp(autobrake, 0.0, 1.0)),
        "reverse_1": float(_clamp(reverse_1, 0.0, 1.0)),
        "reverse_2": float(_clamp(reverse_2, 0.0, 1.0)),
    }
```
