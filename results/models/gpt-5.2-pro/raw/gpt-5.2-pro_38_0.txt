import sys

sys.setrecursionlimit(1_000_000)

def main() -> None:
    data = sys.stdin.read().splitlines()
    if not data:
        return

    U, S = map(int, data[0].split())
    row_cols = []
    col_rows = [set() for _ in range(U)]

    # Read S sets (allow empty lines => empty set)
    for i in range(S):
        line = data[i + 1].strip() if i + 1 < len(data) else ""
        if line == "":
            cols = []
        else:
            cols = list(map(int, line.split()))
        cols = sorted(set(cols))
        row_cols.append(cols)
        for c in cols:
            if 0 <= c < U:
                col_rows[c].add(i)
            else:
                # Invalid element id => no solution can match spec
                print("NO SOLUTION")
                return

    active = set(range(U))

    # Quick fail: any column with no covering rows
    for c in range(U):
        if len(col_rows[c]) == 0:
            print("NO SOLUTION")
            return

    solution = []

    def cover_row(r: int):
        removed_cols = []
        removed_entries = []  # list of (col, row) removed from col_rows[col]

        for j in row_cols[r]:
            if j not in active:
                continue
            active.remove(j)
            removed_cols.append(j)

            # Remove all rows that hit column j from all other active columns they touch
            for i in list(col_rows[j]):
                for k in row_cols[i]:
                    if k == j:
                        continue
                    if k in active and i in col_rows[k]:
                        col_rows[k].remove(i)
                        removed_entries.append((k, i))

        return removed_cols, removed_entries

    def uncover(removed_cols, removed_entries):
        for k, i in reversed(removed_entries):
            col_rows[k].add(i)
        for j in reversed(removed_cols):
            active.add(j)

    def search() -> bool:
        if not active:
            return True

        # Choose the active column with the fewest remaining rows (deterministic ties)
        c = min(active, key=lambda x: (len(col_rows[x]), x))
        rows = col_rows[c]
        if not rows:
            return False

        for r in sorted(rows):
            solution.append(r)
            removed_cols, removed_entries = cover_row(r)
            if search():
                return True
            uncover(removed_cols, removed_entries)
            solution.pop()

        return False

    ok = search()
    if not ok:
        print("NO SOLUTION")
        return

    solution_sorted = sorted(solution)
    print("SOLUTION")
    print(" ".join(map(str, solution_sorted)))

if __name__ == "__main__":
    main()
