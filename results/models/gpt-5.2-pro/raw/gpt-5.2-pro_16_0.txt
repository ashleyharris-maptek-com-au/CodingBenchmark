```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <cstdint>
#include <limits>
#include <algorithm>
#include <random>
#include <chrono>

using i64 = long long;

struct Interval {
    i64 s, e;
};

enum class Heuristic : int {
    LRPT = 0,   // longest remaining processing time (tie-break)
    SPT  = 1,   // shortest processing time
    LPT  = 2,   // longest processing time
    RAND = 3    // random tie-break
};

struct Problem {
    int J = 0, M = 0;
    std::vector<int> T; // tasks per job
    std::vector<std::vector<int>> mach;      // [j][t]
    std::vector<std::vector<i64>> dur;       // [j][t]
    std::vector<std::vector<i64>> rem;       // suffix sums remaining proc time [j][t]
    i64 totalOps = 0;
};

struct ScheduleResult {
    i64 makespan = std::numeric_limits<i64>::max();
    std::vector<std::vector<i64>> start; // [j][t]
};

struct Node {
    int j, t;
    i64 est;      // last computed earliest start
    i64 key2;     // heuristic tie-breaker key (lower is better)
    uint32_t rnd; // random tie-breaker
};

struct NodeCmp {
    bool operator()(const Node& a, const Node& b) const {
        if (a.est != b.est) return a.est > b.est;   // min-heap by est
        if (a.key2 != b.key2) return a.key2 > b.key2;
        return a.rnd > b.rnd;
    }
};

struct RunConfig {
    bool useGaps = false;       // true: insert into machine gaps (small/medium)
    Heuristic heu = Heuristic::LRPT;
    bool randomize = true;      // whether to include rnd tie-breaker
};

struct RunState {
    const Problem* P = nullptr;
    RunConfig cfg;

    std::vector<i64> jobReady;          // completion time of last scheduled op per job
    std::vector<i64> machineReady;      // only used if !useGaps
    std::vector<std::vector<Interval>> macIv; // only used if useGaps

    std::vector<std::vector<i64>> start;
};

static i64 earliestOnMachineSerial(const RunState& st, int m, i64 r) {
    return std::max(r, st.machineReady[m]);
}

static i64 earliestOnMachineWithGaps(const RunState& st, int m, i64 r, i64 d) {
    const auto& v = st.macIv[m];
    if (v.empty()) return r;

    // Before first
    if (r + d <= v.front().s) return r;

    // Between intervals
    for (size_t i = 0; i + 1 < v.size(); ++i) {
        i64 cand = std::max(r, v[i].e);
        if (cand + d <= v[i + 1].s) return cand;
    }

    // After last
    return std::max(r, v.back().e);
}

static void insertInterval(std::vector<Interval>& v, Interval in) {
    auto it = std::lower_bound(v.begin(), v.end(), in.s,
        [](const Interval& a, i64 s) { return a.s < s; });
    v.insert(it, in);
}

static i64 computeKey2(const Problem& P, int j, int t, Heuristic heu, std::mt19937& rng) {
    i64 d = P.dur[j][t];
    switch (heu) {
        case Heuristic::SPT:  return d;
        case Heuristic::LPT:  return -d;
        case Heuristic::LRPT: return -P.rem[j][t];
        case Heuristic::RAND: return 0;
        default: return -P.rem[j][t];
    }
}

static i64 scheduleOnce(const Problem& P, const RunConfig& cfg, std::mt19937& rng,
                        std::vector<std::vector<i64>>& outStart)
{
    RunState st;
    st.P = &P;
    st.cfg = cfg;
    st.jobReady.assign(P.J, 0);
    st.start.assign(P.J, std::vector<i64>());
    for (int j = 0; j < P.J; ++j) st.start[j].assign(P.T[j], 0);

    if (cfg.useGaps) {
        st.macIv.assign(P.M, {});
    } else {
        st.machineReady.assign(P.M, 0);
    }

    std::priority_queue<Node, std::vector<Node>, NodeCmp> pq;

    auto pushOp = [&](int j, int t) {
        int m = P.mach[j][t];
        i64 d = P.dur[j][t];
        i64 r = st.jobReady[j];
        i64 est = cfg.useGaps ? earliestOnMachineWithGaps(st, m, r, d)
                              : earliestOnMachineSerial(st, m, r);
        Node nd;
        nd.j = j; nd.t = t;
        nd.est = est;
        nd.key2 = computeKey2(P, j, t, cfg.heu, rng);
        nd.rnd = cfg.randomize ? static_cast<uint32_t>(rng()) : 0u;
        pq.push(nd);
    };

    // Initially, first op of each job is ready (if any).
    for (int j = 0; j < P.J; ++j) {
        if (P.T[j] > 0) pushOp(j, 0);
    }

    i64 scheduledOps = 0;
    i64 makespan = 0;

    while (!pq.empty()) {
        Node nd = pq.top(); pq.pop();
        int j = nd.j, t = nd.t;
        int m = P.mach[j][t];
        i64 d = P.dur[j][t];
        i64 r = st.jobReady[j];

        i64 estNow = cfg.useGaps ? earliestOnMachineWithGaps(st, m, r, d)
                                 : earliestOnMachineSerial(st, m, r);

        // Lazy update: earliest start can only increase as we add operations.
        if (estNow != nd.est) {
            nd.est = estNow;
            // Keep key2 & rnd; only fix est.
            pq.push(nd);
            continue;
        }

        // Commit schedule
        i64 s = estNow;
        i64 e = s + d;
        st.start[j][t] = s;

        st.jobReady[j] = e;
        makespan = std::max(makespan, e);

        if (cfg.useGaps) {
            insertInterval(st.macIv[m], Interval{s, e});
        } else {
            st.machineReady[m] = e;
        }

        ++scheduledOps;
        if (t + 1 < P.T[j]) pushOp(j, t + 1);
    }

    outStart = std::move(st.start);
    return makespan;
}

static Problem readProblem() {
    Problem P;
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    if (!(std::cin >> P.J >> P.M)) return P;
    P.T.assign(P.J, 0);
    P.mach.assign(P.J, {});
    P.dur.assign(P.J, {});

    for (int j = 0; j < P.J; ++j) {
        int Tj;
        std::cin >> Tj;
        P.T[j] = Tj;
        P.mach[j].resize(Tj);
        P.dur[j].resize(Tj);
        for (int t = 0; t < Tj; ++t) {
            int m; i64 d;
            std::cin >> m >> d;
            // Basic sanity (still schedule even if weird input)
            if (m < 0) m = 0;
            if (m >= P.M) m = P.M ? (m % P.M) : 0;
            if (d < 0) d = 0;
            P.mach[j][t] = m;
            P.dur[j][t] = d;
            P.totalOps++;
        }
    }

    // Precompute remaining processing time suffix sums per job for LRPT tie-break.
    P.rem.assign(P.J, {});
    for (int j = 0; j < P.J; ++j) {
        int Tj = P.T[j];
        P.rem[j].assign(Tj + 1, 0);
        for (int t = Tj - 1; t >= 0; --t) {
            P.rem[j][t] = P.rem[j][t + 1] + P.dur[j][t];
        }
        if (Tj > 0) P.rem[j].pop_back(); // keep size Tj (index t valid)
    }

    return P;
}

int main() {
    Problem P = readProblem();
    if (P.J <= 0) return 0;

    const i64 nOps = P.totalOps;

    // Adaptive strategy selection
    bool useGaps = false;
    int iterations = 1;
    auto budget = std::chrono::milliseconds(200);

    if (nOps <= 300) {
        useGaps = true;
        iterations = std::min<i64>(5000, 50 * nOps + 200);
        budget = std::chrono::milliseconds(3000);
    } else if (nOps <= 5000) {
        useGaps = true;
        iterations = 250;
        budget = std::chrono::milliseconds(2000);
    } else if (nOps <= 50000) {
        useGaps = false;
        iterations = 30;
        budget = std::chrono::milliseconds(1200);
    } else {
        useGaps = false;
        iterations = 5;
        budget = std::chrono::milliseconds(500);
    }

    // RNG
    uint64_t seed = (uint64_t)std::chrono::high_resolution_clock::now().time_since_epoch().count();
    seed ^= (uint64_t)nOps * 0x9E3779B97F4A7C15ULL;
    seed ^= (uint64_t)P.J * 0xBF58476D1CE4E5B9ULL;
    seed ^= (uint64_t)P.M * 0x94D049BB133111EBULL;
    std::mt19937 rng((uint32_t)(seed ^ (seed >> 32)));

    ScheduleResult best;
    best.start.assign(P.J, {});
    for (int j = 0; j < P.J; ++j) best.start[j].assign(P.T[j], 0);

    auto t0 = std::chrono::steady_clock::now();

    // A small portfolio of heuristics
    const Heuristic portfolio[] = {
        Heuristic::LRPT,
        Heuristic::SPT,
        Heuristic::LPT,
        Heuristic::RAND
    };

    for (int it = 0; it < iterations; ++it) {
        auto now = std::chrono::steady_clock::now();
        if (now - t0 > budget) break;

        RunConfig cfg;
        cfg.useGaps = useGaps;

        // Mix deterministic + randomized selection
        Heuristic base = portfolio[it % (int)(sizeof(portfolio) / sizeof(portfolio[0]))];
        if (nOps > 50000) base = Heuristic::LRPT; // keep stable & fast for huge cases
        cfg.heu = base;
        cfg.randomize = (base == Heuristic::RAND) || (it > 0); // add randomness after first run

        std::vector<std::vector<i64>> start;
        i64 ms = scheduleOnce(P, cfg, rng, start);
        if (ms < best.makespan) {
            best.makespan = ms;
            best.start = std::move(start);
        }
    }

    // Output: for each job, print (start machine) pairs for each task
    for (int j = 0; j < P.J; ++j) {
        for (int t = 0; t < P.T[j]; ++t) {
            if (t) std::cout << ' ';
            std::cout << best.start[j][t] << ' ' << P.mach[j][t];
        }
        std::cout << "\n";
    }
    return 0;
}
```
