'use strict';

const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input.length ? JSON.parse(input) : {};
const prob = data.probability || [];
const dronesIn = data.drones || [];
const maxStepsPerDrone = Math.max(0, data.maxStepsPerDrone | 0);

const H = prob.length | 0;
const W = H > 0 ? (prob[0].length | 0) : 0;
const N = W * H;

function idxOf(x, y) { return y * W + x; }
function xOf(idx) { return idx % W; }
function yOf(idx) { return (idx / W) | 0; }

const walkable = new Uint8Array(N);
const weight = new Float64Array(N);

for (let y = 0; y < H; y++) {
  const row = prob[y];
  for (let x = 0; x < W; x++) {
    const v = row[x];
    const id = idxOf(x, y);
    if (v === -1) {
      walkable[id] = 0;
      weight[id] = 0;
    } else {
      walkable[id] = 1;
      weight[id] = v > 0 ? v : 0;
    }
  }
}

const visited = new Uint8Array(N);

// Simple LCG for tie-breaking
let rngState = 0x12345678;
function rand01() {
  rngState = (Math.imul(rngState, 1664525) + 1013904223) >>> 0;
  return rngState / 4294967296;
}

// -------- Top-K weighted cells (global candidates) --------
class MinHeapTopK {
  constructor() { this.idx = []; this.w = []; }
  size() { return this.idx.length; }
  minW() { return this.w[0]; }
  push(i, wi) {
    let aIdx = this.idx, aW = this.w;
    aIdx.push(i); aW.push(wi);
    this._siftUp(aIdx.length - 1);
  }
  replaceMin(i, wi) {
    this.idx[0] = i; this.w[0] = wi;
    this._siftDown(0);
  }
  _siftUp(pos) {
    const aIdx = this.idx, aW = this.w;
    while (pos > 0) {
      const p = (pos - 1) >> 1;
      if (aW[p] <= aW[pos]) break;
      [aW[p], aW[pos]] = [aW[pos], aW[p]];
      [aIdx[p], aIdx[pos]] = [aIdx[pos], aIdx[p]];
      pos = p;
    }
  }
  _siftDown(pos) {
    const aIdx = this.idx, aW = this.w;
    const n = aIdx.length;
    while (true) {
      let l = pos * 2 + 1;
      if (l >= n) break;
      let r = l + 1;
      let m = (r < n && aW[r] < aW[l]) ? r : l;
      if (aW[pos] <= aW[m]) break;
      [aW[pos], aW[m]] = [aW[m], aW[pos]];
      [aIdx[pos], aIdx[m]] = [aIdx[m], aIdx[pos]];
      pos = m;
    }
  }
  toArray() {
    const out = new Array(this.size());
    for (let i = 0; i < out.length; i++) out[i] = { idx: this.idx[i], w: this.w[i] };
    return out;
  }
}

const TOP_K = Math.min(50000, N); // cap memory/time
const heapTop = new MinHeapTopK();

for (let i = 0; i < N; i++) {
  const w = weight[i];
  if (w <= 0 || walkable[i] === 0) continue;
  if (heapTop.size() < TOP_K) heapTop.push(i, w);
  else if (w > heapTop.minW()) heapTop.replaceMin(i, w);
}

const topCells = heapTop.toArray();
topCells.sort((a, b) => b.w - a.w);

// -------- A* (with reusable typed arrays) --------
class MinHeapOpen {
  constructor() { this.idx = []; this.f = []; }
  clear() { this.idx.length = 0; this.f.length = 0; }
  size() { return this.idx.length; }
  push(i, fi) {
    const aIdx = this.idx, aF = this.f;
    aIdx.push(i); aF.push(fi);
    let pos = aIdx.length - 1;
    while (pos > 0) {
      const p = (pos - 1) >> 1;
      if (aF[p] <= aF[pos]) break;
      [aF[p], aF[pos]] = [aF[pos], aF[p]];
      [aIdx[p], aIdx[pos]] = [aIdx[pos], aIdx[p]];
      pos = p;
    }
  }
  pop() {
    const aIdx = this.idx, aF = this.f;
    const n = aIdx.length;
    const out = aIdx[0];
    if (n === 1) { aIdx.pop(); aF.pop(); return out; }
    aIdx[0] = aIdx.pop();
    aF[0] = aF.pop();
    let pos = 0;
    while (true) {
      let l = pos * 2 + 1;
      if (l >= aIdx.length) break;
      let r = l + 1;
      let m = (r < aIdx.length && aF[r] < aF[l]) ? r : l;
      if (aF[pos] <= aF[m]) break;
      [aF[pos], aF[m]] = [aF[m], aF[pos]];
      [aIdx[pos], aIdx[m]] = [aIdx[m], aIdx[pos]];
      pos = m;
    }
    return out;
  }
}

const gScore = new Float32Array(N);
const parent = new Int32Array(N);
const mark = new Int32Array(N);
let searchId = 1;
const open = new MinHeapOpen();

function manhattan(aIdx, bIdx) {
  const ax = aIdx % W, ay = (aIdx / W) | 0;
  const bx = bIdx % W, by = (bIdx / W) | 0;
  const dx = ax - bx, dy = ay - by;
  return (dx < 0 ? -dx : dx) + (dy < 0 ? -dy : dy);
}

function reconstructPath(startIdx, endIdx) {
  // returns array of idx from start(excluded) -> end(included)
  const rev = [];
  let cur = endIdx;
  while (cur !== -1 && cur !== startIdx) {
    rev.push(cur);
    cur = parent[cur];
  }
  rev.reverse();
  return rev;
}

function aStarToward(startIdx, goalIdx, expansionsLimit) {
  if (startIdx === goalIdx) return [];
  if (walkable[goalIdx] === 0) return null;

  const sid = (searchId = (searchId + 1) | 0);
  open.clear();

  mark[startIdx] = sid;
  gScore[startIdx] = 0;
  parent[startIdx] = -1;
  open.push(startIdx, manhattan(startIdx, goalIdx));

  let bestNode = startIdx;
  let bestH = manhattan(startIdx, goalIdx);

  let expansions = 0;

  while (open.size() > 0) {
    const cur = open.pop();
    expansions++;
    if (expansions > expansionsLimit) break;

    const curH = manhattan(cur, goalIdx);
    if (curH < bestH) { bestH = curH; bestNode = cur; }

    if (cur === goalIdx) {
      return reconstructPath(startIdx, goalIdx);
    }

    const curG = gScore[cur];

    const x = cur % W;
    const y = (cur / W) | 0;

    // neighbors: L,R,U,D
    // left
    if (x > 0) {
      const nb = cur - 1;
      if (walkable[nb]) {
        const ng = curG + 1;
        if (mark[nb] !== sid || ng < gScore[nb]) {
          mark[nb] = sid;
          gScore[nb] = ng;
          parent[nb] = cur;
          open.push(nb, ng + manhattan(nb, goalIdx));
        }
      }
    }
    // right
    if (x + 1 < W) {
      const nb = cur + 1;
      if (walkable[nb]) {
        const ng = curG + 1;
        if (mark[nb] !== sid || ng < gScore[nb]) {
          mark[nb] = sid;
          gScore[nb] = ng;
          parent[nb] = cur;
          open.push(nb, ng + manhattan(nb, goalIdx));
        }
      }
    }
    // up
    if (y > 0) {
      const nb = cur - W;
      if (walkable[nb]) {
        const ng = curG + 1;
        if (mark[nb] !== sid || ng < gScore[nb]) {
          mark[nb] = sid;
          gScore[nb] = ng;
          parent[nb] = cur;
          open.push(nb, ng + manhattan(nb, goalIdx));
        }
      }
    }
    // down
    if (y + 1 < H) {
      const nb = cur + W;
      if (walkable[nb]) {
        const ng = curG + 1;
        if (mark[nb] !== sid || ng < gScore[nb]) {
          mark[nb] = sid;
          gScore[nb] = ng;
          parent[nb] = cur;
          open.push(nb, ng + manhattan(nb, goalIdx));
        }
      }
    }
  }

  // Not reached: return path toward best node (closest to goal among explored)
  if (bestNode !== startIdx) return reconstructPath(startIdx, bestNode);
  return null;
}

// -------- Local greedy fallback --------
function localGreedyMove(curIdx) {
  const x = curIdx % W;
  const y = (curIdx / W) | 0;

  // candidates: stay, L, R, U, D
  const cand = [];
  cand.push(curIdx);
  if (x > 0) cand.push(curIdx - 1);
  if (x + 1 < W) cand.push(curIdx + 1);
  if (y > 0) cand.push(curIdx - W);
  if (y + 1 < H) cand.push(curIdx + W);

  let best = curIdx;
  let bestScore = -1e100;

  for (let i = 0; i < cand.length; i++) {
    const nb = cand[i];
    if (!walkable[nb]) continue;

    let s = 0;
    if (!visited[nb]) s += weight[nb];

    // small neighborhood bonus (encourages heading toward value even if immediate is 0)
    const nx = nb % W;
    const ny = (nb / W) | 0;

    // 4-neighborhood around nb
    let bonus = 0;
    if (nx > 0) { const t = nb - 1; if (walkable[t] && !visited[t]) bonus += weight[t]; }
    if (nx + 1 < W) { const t = nb + 1; if (walkable[t] && !visited[t]) bonus += weight[t]; }
    if (ny > 0) { const t = nb - W; if (walkable[t] && !visited[t]) bonus += weight[t]; }
    if (ny + 1 < H) { const t = nb + W; if (walkable[t] && !visited[t]) bonus += weight[t]; }

    s += 0.15 * bonus;

    // tiny randomness to break ties deterministically-ish
    s += 1e-9 * rand01();

    if (s > bestScore) {
      bestScore = s;
      best = nb;
    }
  }

  return best;
}

// -------- Target selection --------
const reservedTargets = new Set();

function pickTarget(curIdx, maxChecks) {
  let bestIdx = -1;
  let bestUtil = -1;

  const curX = curIdx % W;
  const curY = (curIdx / W) | 0;

  let checked = 0;
  for (let i = 0; i < topCells.length && checked < maxChecks; i++) {
    const c = topCells[i];
    const tIdx = c.idx;

    if (visited[tIdx]) continue;
    if (reservedTargets.has(tIdx)) continue;
    // walkable should be true, but keep safe
    if (!walkable[tIdx]) continue;

    const tx = tIdx % W;
    const ty = (tIdx / W) | 0;
    const dx = tx - curX, dy = ty - curY;
    const dist = (dx < 0 ? -dx : dx) + (dy < 0 ? -dy : dy);

    const util = c.w / (dist + 1);

    if (util > bestUtil) {
      bestUtil = util;
      bestIdx = tIdx;
    }

    // Since c.w is descending and util <= c.w (because dist+1 >= 1),
    // once c.w can't beat current bestUtil, nothing later can.
    if (c.w <= bestUtil) break;

    checked++;
  }

  return bestIdx;
}

// -------- Simulation --------
const drones = dronesIn.map((d, i) => {
  const x = d.x | 0, y = d.y | 0;
  let startIdx = 0;
  if (x >= 0 && x < W && y >= 0 && y < H) startIdx = idxOf(x, y);
  // If start is obstacle, try to find nearest walkable by expanding radius (cheap fallback)
  if (N > 0 && !walkable[startIdx]) {
    let found = -1;
    const maxR = 50;
    for (let r = 1; r <= maxR && found === -1; r++) {
      for (let dy = -r; dy <= r; dy++) {
        const yy = y + dy;
        if (yy < 0 || yy >= H) continue;
        const rem = r - (dy < 0 ? -dy : dy);
        const x1 = x - rem, x2 = x + rem;
        if (x1 >= 0 && x1 < W) {
          const ii = idxOf(x1, yy);
          if (walkable[ii]) { found = ii; break; }
        }
        if (x2 !== x1 && x2 >= 0 && x2 < W) {
          const ii = idxOf(x2, yy);
          if (walkable[ii]) { found = ii; break; }
        }
      }
    }
    if (found !== -1) startIdx = found;
  }

  return {
    i,
    pos: startIdx,
    path: [[xOf(startIdx), yOf(startIdx)]],
    target: -1,
    route: [],   // array of idx steps to follow
    rpos: 0
  };
});

// Mark starting cells as visited (unique coverage)
for (const dr of drones) visited[dr.pos] = 1;

function ensureRoute(dr, remainingSteps) {
  // If current target was already visited by someone else, drop it
  if (dr.target !== -1 && visited[dr.target] && dr.target !== dr.pos) {
    reservedTargets.delete(dr.target);
    dr.target = -1;
    dr.route = [];
    dr.rpos = 0;
  }

  // Have usable route
  if (dr.rpos < dr.route.length) return;

  // Need new target/route
  if (remainingSteps <= 0) return;

  // Pick a high-value target and plan toward it
  const maxChecks = 3000; // per retarget attempt
  const t = pickTarget(dr.pos, maxChecks);
  if (t === -1) return;

  reservedTargets.add(t);
  dr.target = t;

  // Expansion budget scales a bit with remaining steps, but capped
  const baseLimit = 15000;
  const limit = Math.min(60000, baseLimit + remainingSteps * 20);

  const route = aStarToward(dr.pos, t, limit);
  if (!route || route.length === 0) {
    reservedTargets.delete(t);
    dr.target = -1;
    dr.route = [];
    dr.rpos = 0;
    return;
  }

  dr.route = route;
  dr.rpos = 0;
}

for (let step = 0; step < maxStepsPerDrone; step++) {
  const remaining = maxStepsPerDrone - step;

  // Refresh routes if needed
  for (const dr of drones) {
    ensureRoute(dr, remaining);
  }

  // Move each drone
  for (const dr of drones) {
    let next = dr.pos;

    if (dr.rpos < dr.route.length) {
      next = dr.route[dr.rpos++];
      // If route step invalid (shouldn't happen), fall back
      if (!walkable[next] || manhattan(next, dr.pos) > 1) next = localGreedyMove(dr.pos);
    } else {
      next = localGreedyMove(dr.pos);
    }

    dr.pos = next;
    dr.path.push([xOf(next), yOf(next)]);
    if (!visited[next]) visited[next] = 1;

    if (dr.target !== -1 && dr.pos === dr.target) {
      reservedTargets.delete(dr.target);
      dr.target = -1;
      dr.route = [];
      dr.rpos = 0;
    }
  }
}

const out = {
  paths: drones.map(dr => dr.path)
};

process.stdout.write(JSON.stringify(out));
