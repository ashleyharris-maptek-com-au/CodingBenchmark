```spirv
; SPIR-V
; Version: 1.0
; Generator: hand-written
; Bound: 200
OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

OpSource GLSL 450

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_in 0 Offset 0
OpDecorate %arr_pv_in ArrayStride 16

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 Offset 0

OpDecorate %arr_v3_2 ArrayStride 16
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%fn_void = OpTypeFunction %void

%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%uint_2 = OpConstant %uint 2

%f0 = OpConstant %float 0
%f1 = OpConstant %float 1
%f075 = OpConstant %float 0.75
%f_thick = OpConstant %float 0.015
%f_arrow = OpConstant %float 0.05

; gl_PerVertex in/out blocks
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float

; User varyings (arrays sized to input primitive vertex count = 2)
%arr_v3_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_2
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v3 = OpTypePointer Output %v3float

; Interface variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

; Main
%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load p0, p1 from gl_in[].gl_Position
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr

; Load colors c0, c1 from color[]
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

; dir = p1 - p0 (use xy for perpendicular)
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir_xy = OpFSub %v2float %p1_xy %p0_xy

%dir_x = OpCompositeExtract %float %dir_xy 0
%dir_y = OpCompositeExtract %float %dir_xy 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_raw = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp = OpExtInst %v2float %glsl Normalize %perp_raw

; mid = p0 + 0.75*(p1-p0)
%diff4 = OpFSub %v4float %p1 %p0
%diff4_075 = OpVectorTimesScalar %v4float %diff4 %f075
%mid = OpFAdd %v4float %p0 %diff4_075

; offsets (vec2) -> (vec4)
%offS2 = OpVectorTimesScalar %v2float %perp %f_thick
%offA2 = OpVectorTimesScalar %v2float %perp %f_arrow

%offS_x = OpCompositeExtract %float %offS2 0
%offS_y = OpCompositeExtract %float %offS2 1
%offA_x = OpCompositeExtract %float %offA2 0
%offA_y = OpCompositeExtract %float %offA2 1

%offS4 = OpCompositeConstruct %v4float %offS_x %offS_y %f0 %f0
%offA4 = OpCompositeConstruct %v4float %offA_x %offA_y %f0 %f0

; Shaft quad as triangle strip: p0+/-offS, mid+/-offS (4 verts), color c0
%p0_plus = OpFAdd %v4float %p0 %offS4
%p0_minus = OpFSub %v4float %p0 %offS4
%mid_plus = OpFAdd %v4float %mid %offS4
%mid_minus = OpFSub %v4float %mid %offS4

%glpos_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %glpos_out_ptr %p0_plus
OpStore %outColor %c0
OpEmitVertex

OpStore %glpos_out_ptr %p0_minus
OpStore %outColor %c0
OpEmitVertex

OpStore %glpos_out_ptr %mid_plus
OpStore %outColor %c0
OpEmitVertex

OpStore %glpos_out_ptr %mid_minus
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

; Arrowhead as triangle strip: mid+offA, mid-offA, p1 (3 verts), color c1
%mid_plusA = OpFAdd %v4float %mid %offA4
%mid_minusA = OpFSub %v4float %mid %offA4

OpStore %glpos_out_ptr %mid_plusA
OpStore %outColor %c1
OpEmitVertex

OpStore %glpos_out_ptr %mid_minusA
OpStore %outColor %c1
OpEmitVertex

OpStore %glpos_out_ptr %p1
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
