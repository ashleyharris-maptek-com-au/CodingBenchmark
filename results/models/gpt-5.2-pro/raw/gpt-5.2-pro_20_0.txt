#[derive(Clone, Copy, Debug)]
struct Vec3 {
    x: f32,
    y: f32,
    z: f32,
}
impl Vec3 {
    #[inline]
    fn sub(self, o: Vec3) -> Vec3 {
        Vec3 { x: self.x - o.x, y: self.y - o.y, z: self.z - o.z }
    }
    #[inline]
    fn add(self, o: Vec3) -> Vec3 {
        Vec3 { x: self.x + o.x, y: self.y + o.y, z: self.z + o.z }
    }
    #[inline]
    fn mul(self, s: f32) -> Vec3 {
        Vec3 { x: self.x * s, y: self.y * s, z: self.z * s }
    }
    #[inline]
    fn dot(self, o: Vec3) -> f32 {
        self.x * o.x + self.y * o.y + self.z * o.z
    }
    #[inline]
    fn norm(self) -> f32 {
        self.dot(self).sqrt()
    }
}

struct FastInput {
    buf: Vec<u8>,
    i: usize,
}
impl FastInput {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, i: 0 }
    }

    #[inline]
    fn skip_ws(&mut self) {
        while self.i < self.buf.len() && self.buf[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
    }

    #[inline]
    fn next_token(&mut self) -> Option<&[u8]> {
        self.skip_ws();
        if self.i >= self.buf.len() {
            return None;
        }
        let start = self.i;
        while self.i < self.buf.len() && !self.buf[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
        Some(&self.buf[start..self.i])
    }

    fn next_string(&mut self) -> Option<String> {
        let t = self.next_token()?;
        Some(unsafe { std::str::from_utf8_unchecked(t) }.to_string())
    }

    fn next_usize(&mut self) -> Option<usize> {
        let t = self.next_token()?;
        parse_usize(t)
    }

    fn next_f32(&mut self) -> Option<f32> {
        let t = self.next_token()?;
        Some(parse_f32_lossy(t))
    }
}

fn parse_usize(t: &[u8]) -> Option<usize> {
    if t.is_empty() {
        return None;
    }
    let mut v: usize = 0;
    for &c in t {
        if !(b'0'..=b'9').contains(&c) {
            return None;
        }
        v = v.saturating_mul(10).saturating_add((c - b'0') as usize);
    }
    Some(v)
}

#[inline]
fn is_na_token(t: &[u8]) -> bool {
    if t.is_empty() {
        return true;
    }
    // Case-insensitive check for NA/NAN
    if t.len() == 2 {
        let a = t[0] | 0x20;
        let b = t[1] | 0x20;
        return a == b'n' && b == b'a';
    }
    if t.len() == 3 {
        let a = t[0] | 0x20;
        let b = t[1] | 0x20;
        let c = t[2] | 0x20;
        return a == b'n' && b == b'a' && c == b'n';
    }
    false
}

/// Fast-ish float parser: [-]digits[.digits][e[-]digits]
/// Returns NaN on invalid tokens (including "NA"/"NaN").
fn parse_f32_lossy(t: &[u8]) -> f32 {
    if is_na_token(t) {
        return f32::NAN;
    }
    let s = unsafe { std::str::from_utf8_unchecked(t) };
    // Quick path: try standard parse (still fairly fast in practice).
    // If it fails, return NaN.
    match s.parse::<f32>() {
        Ok(v) => v,
        Err(_) => f32::NAN,
    }
}

#[derive(Clone)]
struct Suspect {
    conf: f64,
    hole: u32,
    sample_idx: u32,
    prop: usize,
}
impl PartialEq for Suspect {
    fn eq(&self, other: &Self) -> bool {
        self.conf.to_bits() == other.conf.to_bits()
            && self.hole == other.hole
            && self.sample_idx == other.sample_idx
            && self.prop == other.prop
    }
}
impl Eq for Suspect {}
impl PartialOrd for Suspect {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}
impl Ord for Suspect {
    fn cmp(&self, other: &Self) -> Ordering {
        // Natural ordering by confidence, then tie-breakers.
        // Use total_cmp for floats.
        let c = self.conf.total_cmp(&other.conf);
        if c != Ordering::Equal {
            return c;
        }
        let c2 = self.hole.cmp(&other.hole);
        if c2 != Ordering::Equal {
            return c2;
        }
        let c3 = self.sample_idx.cmp(&other.sample_idx);
        if c3 != Ordering::Equal {
            return c3;
        }
        self.prop.cmp(&other.prop)
    }
}

enum GridIndex {
    Dense(DenseGrid),
    Sparse(SparseGrid),
}

struct DenseGrid {
    min: Vec3,
    inv_cell: f32,
    nx: u32,
    ny: u32,
    nz: u32,
    offsets: Vec<u32>, // len = num_cells + 1
    points: Vec<u32>,  // len = N, point indices grouped by cell
}

struct SparseGrid {
    min: Vec3,
    inv_cell: f32,
    // Packed key: (ix<<42)|(iy<<21)|iz  (each < 2^21)
    keys: Vec<u64>,     // unique sorted
    starts: Vec<u32>,   // len = keys.len()+1 (starts, with last = points.len())
    points: Vec<u32>,   // point indices grouped by key
}

#[inline]
fn compute_cell_size(min: Vec3, max: Vec3, n: usize) -> f32 {
    let dx = (max.x - min.x).max(0.0) as f64;
    let dy = (max.y - min.y).max(0.0) as f64;
    let dz = (max.z - min.z).max(0.0) as f64;
    let diag = (dx * dx + dy * dy + dz * dz).sqrt();
    if n <= 1 || !diag.is_finite() || diag == 0.0 {
        return 1.0;
    }
    let cbrt_n = (n as f64).cbrt().max(1.0);
    // Slightly larger than average spacing to keep per-cell population small.
    let mut cell = (diag / cbrt_n) * 2.2;
    if !cell.is_finite() || cell <= 0.0 {
        cell = 1.0;
    }
    // Clamp to avoid absurd values on pathological inputs.
    cell = cell.clamp(1e-3, (diag as f32).max(1.0) as f64);
    cell as f32
}

#[inline]
fn cell_coord(x: f32, min_x: f32, inv_cell: f32, max_c: u32) -> u32 {
    let mut v = ((x - min_x) * inv_cell) as i64; // trunc toward 0; x>=min -> floor
    if v < 0 {
        v = 0;
    }
    let mut u = v as u64;
    if u > max_c as u64 {
        u = max_c as u64;
    }
    u as u32
}

fn build_grid_index(pos: &[Vec3], min: Vec3, max: Vec3) -> GridIndex {
    let n = pos.len();
    let cell = compute_cell_size(min, max, n);
    let inv_cell = 1.0f32 / cell;

    // Compute grid extents.
    let span_x = (max.x - min.x).max(0.0);
    let span_y = (max.y - min.y).max(0.0);
    let span_z = (max.z - min.z).max(0.0);

    let nx = (span_x * inv_cell).floor() as u32 + 1;
    let ny = (span_y * inv_cell).floor() as u32 + 1;
    let nz = (span_z * inv_cell).floor() as u32 + 1;

    let nx = nx.max(1);
    let ny = ny.max(1);
    let nz = nz.max(1);

    let num_cells_u64 = (nx as u64) * (ny as u64) * (nz as u64);

    // Dense grid only if not too many cells.
    // (Memory for counts+offsets+cursor ~ 12 bytes per cell during build.)
    const MAX_DENSE_CELLS: u64 = 12_000_000;

    if num_cells_u64 <= MAX_DENSE_CELLS {
        let num_cells = num_cells_u64 as usize;

        let mut counts = vec![0u32; num_cells];

        // First pass: count.
        for &p in pos {
            let ix = cell_coord(p.x, min.x, inv_cell, nx - 1);
            let iy = cell_coord(p.y, min.y, inv_cell, ny - 1);
            let iz = cell_coord(p.z, min.z, inv_cell, nz - 1);
            let cid = ((ix as u64) * (ny as u64) + (iy as u64)) * (nz as u64) + (iz as u64);
            counts[cid as usize] = counts[cid as usize].saturating_add(1);
        }

        // Prefix sum offsets.
        let mut offsets = vec![0u32; num_cells + 1];
        for i in 0..num_cells {
            offsets[i + 1] = offsets[i].saturating_add(counts[i]);
        }

        let mut cursor = offsets[..num_cells].to_vec();
        let mut points = vec![0u32; n];

        // Second pass: fill.
        for (idx, &p) in pos.iter().enumerate() {
            let ix = cell_coord(p.x, min.x, inv_cell, nx - 1);
            let iy = cell_coord(p.y, min.y, inv_cell, ny - 1);
            let iz = cell_coord(p.z, min.z, inv_cell, nz - 1);
            let cid = ((ix as u64) * (ny as u64) + (iy as u64)) * (nz as u64) + (iz as u64);
            let cid = cid as usize;
            let at = cursor[cid] as usize;
            points[at] = idx as u32;
            cursor[cid] = cursor[cid].saturating_add(1);
        }

        GridIndex::Dense(DenseGrid { min, inv_cell, nx, ny, nz, offsets, points })
    } else {
        // Sparse sorted keys
        // Pack keys (assumes ix,iy,iz reasonably bounded; with our cell sizing they almost always are).
        // If a dimension exceeds 2^21, we still pack but will clamp (may cause collisions).
        // On such pathological cases, detection quality degrades but program remains safe.
        let max_pack: u32 = (1u32 << 21) - 1;

        let mut pairs: Vec<(u64, u32)> = Vec::with_capacity(n);
        for (i, &p) in pos.iter().enumerate() {
            let mut ix = ((p.x - min.x) * inv_cell) as i64;
            let mut iy = ((p.y - min.y) * inv_cell) as i64;
            let mut iz = ((p.z - min.z) * inv_cell) as i64;
            if ix < 0 { ix = 0; }
            if iy < 0 { iy = 0; }
            if iz < 0 { iz = 0; }
            let ix = (ix as u64).min(max_pack as u64) as u64;
            let iy = (iy as u64).min(max_pack as u64) as u64;
            let iz = (iz as u64).min(max_pack as u64) as u64;
            let key = (ix << 42) | (iy << 21) | iz;
            pairs.push((key, i as u32));
        }
        pairs.sort_unstable_by_key(|p| p.0);

        let mut keys: Vec<u64> = Vec::new();
        let mut starts: Vec<u32> = Vec::new();
        let mut points: Vec<u32> = Vec::with_capacity(n);

        let mut last_key: Option<u64> = None;
        for (i, (k, idx)) in pairs.into_iter().enumerate() {
            if last_key != Some(k) {
                keys.push(k);
                starts.push(i as u32);
                last_key = Some(k);
            }
            points.push(idx);
        }
        starts.push(points.len() as u32);

        GridIndex::Sparse(SparseGrid { min, inv_cell, keys, starts, points })
    }
}

#[inline]
fn dense_cell_id(g: &DenseGrid, p: Vec3) -> (u32, u32, u32, u64) {
    let ix = cell_coord(p.x, g.min.x, g.inv_cell, g.nx - 1);
    let iy = cell_coord(p.y, g.min.y, g.inv_cell, g.ny - 1);
    let iz = cell_coord(p.z, g.min.z, g.inv_cell, g.nz - 1);
    let cid = ((ix as u64) * (g.ny as u64) + (iy as u64)) * (g.nz as u64) + (iz as u64);
    (ix, iy, iz, cid)
}

#[inline]
fn sparse_key(g: &SparseGrid, p: Vec3) -> (u32, u32, u32, u64) {
    let max_pack: u32 = (1u32 << 21) - 1;
    let mut ix = ((p.x - g.min.x) * g.inv_cell) as i64;
    let mut iy = ((p.y - g.min.y) * g.inv_cell) as i64;
    let mut iz = ((p.z - g.min.z) * g.inv_cell) as i64;
    if ix < 0 { ix = 0; }
    if iy < 0 { iy = 0; }
    if iz < 0 { iz = 0; }
    let ixu = (ix as u64).min(max_pack as u64) as u32;
    let iyu = (iy as u64).min(max_pack as u64) as u32;
    let izu = (iz as u64).min(max_pack as u64) as u32;
    let key = ((ixu as u64) << 42) | ((iyu as u64) << 21) | (izu as u64);
    (ixu, iyu, izu, key)
}

#[inline]
fn sparse_range(g: &SparseGrid, key: u64) -> Option<(u32, u32)> {
    match g.keys.binary_search(&key) {
        Ok(i) => Some((g.starts[i], g.starts[i + 1])),
        Err(_) => None,
    }
}

fn gather_neighbors(
    grid: &GridIndex,
    pos: &[Vec3],
    i: usize,
    k: usize,
    max_r: i32,
    max_candidates: usize,
    cand_buf: &mut Vec<(f32, u32)>,
    out_neighbors: &mut Vec<u32>,
) {
    cand_buf.clear();
    out_neighbors.clear();

    let pi = pos[i];

    match grid {
        GridIndex::Dense(g) => {
            let (ix0, iy0, iz0, _cid0) = dense_cell_id(g, pi);
            let ix0 = ix0 as i32;
            let iy0 = iy0 as i32;
            let iz0 = iz0 as i32;

            'rings: for r in 0..=max_r {
                let mut pushed_any = false;
                for dx in -r..=r {
                    let ix = ix0 + dx;
                    if ix < 0 || ix >= g.nx as i32 { continue; }
                    for dy in -r..=r {
                        let iy = iy0 + dy;
                        if iy < 0 || iy >= g.ny as i32 { continue; }
                        for dz in -r..=r {
                            let iz = iz0 + dz;
                            if iz < 0 || iz >= g.nz as i32 { continue; }
                            let cid = ((ix as u64) * (g.ny as u64) + (iy as u64)) * (g.nz as u64) + (iz as u64);
                            let cid = cid as usize;
                            let start = g.offsets[cid] as usize;
                            let end = g.offsets[cid + 1] as usize;

                            // Cap per-cell scan (keeps worst-case bounded if a cell is too dense)
                            let mut taken = 0usize;
                            for &pj_u32 in &g.points[start..end] {
                                if taken >= 256 { break; }
                                let pj = pj_u32 as usize;
                                if pj == i { continue; }
                                let d = pos[pj].sub(pi);
                                let dist2 = d.dot(d);
                                cand_buf.push((dist2, pj_u32));
                                taken += 1;
                                pushed_any = true;
                                if cand_buf.len() >= max_candidates {
                                    break 'rings;
                                }
                            }
                        }
                    }
                }

                // If we have enough, stop early.
                if pushed_any && cand_buf.len() >= k * 4 {
                    break;
                }
            }
        }
        GridIndex::Sparse(g) => {
            let (ix0, iy0, iz0, _k0) = sparse_key(g, pi);
            let ix0 = ix0 as i32;
            let iy0 = iy0 as i32;
            let iz0 = iz0 as i32;

            for r in 0..=max_r {
                for dx in -r..=r {
                    let ix = ix0 + dx;
                    if ix < 0 { continue; }
                    for dy in -r..=r {
                        let iy = iy0 + dy;
                        if iy < 0 { continue; }
                        for dz in -r..=r {
                            let iz = iz0 + dz;
                            if iz < 0 { continue; }
                            let ixu = (ix as u32).min((1u32 << 21) - 1);
                            let iyu = (iy as u32).min((1u32 << 21) - 1);
                            let izu = (iz as u32).min((1u32 << 21) - 1);
                            let key = ((ixu as u64) << 42) | ((iyu as u64) << 21) | (izu as u64);

                            if let Some((s, e)) = sparse_range(g, key) {
                                let s = s as usize;
                                let e = e as usize;
                                let mut taken = 0usize;
                                for &pj_u32 in &g.points[s..e] {
                                    if taken >= 256 { break; }
                                    let pj = pj_u32 as usize;
                                    if pj == i { continue; }
                                    let d = pos[pj].sub(pi);
                                    let dist2 = d.dot(d);
                                    cand_buf.push((dist2, pj_u32));
                                    taken += 1;
                                    if cand_buf.len() >= max_candidates {
                                        break;
                                    }
                                }
                            }
                            if cand_buf.len() >= max_candidates {
                                break;
                            }
                        }
                        if cand_buf.len() >= max_candidates {
                            break;
                        }
                    }
                    if cand_buf.len() >= max_candidates {
                        break;
                    }
                }
                if cand_buf.len() >= k * 4 {
                    break;
                }
            }
        }
    }

    // Sort by distance and take first k.
    cand_buf.sort_unstable_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(Ordering::Equal));

    out_neighbors.reserve(k);
    for &(_d2, idx) in cand_buf.iter().take(k * 2) {
        if out_neighbors.len() >= k {
            break;
        }
        out_neighbors.push(idx);
    }
}

fn median_sorted(vals: &[f32]) -> f32 {
    let n = vals.len();
    if n == 0 {
        return f32::NAN;
    }
    if n & 1 == 1 {
        vals[n / 2]
    } else {
        (vals[n / 2 - 1] + vals[n / 2]) * 0.5
    }
}

fn main() {
    let mut input = FastInput::new();

    let _num_holes = match input.next_usize() {
        Some(v) => v,
        None => {
            // Empty input
            println!("0");
            return;
        }
    };
    let num_props = match input.next_usize() {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut prop_names = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        prop_names.push(input.next_string().unwrap_or_else(|| "prop".to_string()));
    }

    // Hole ID interning
    let mut hole_map: HashMap<String, u32> = HashMap::new();
    let mut hole_ids: Vec<String> = Vec::new();

    // Global sample storage (flattened)
    let mut pos: Vec<Vec3> = Vec::new();
    let mut hole_of: Vec<u32> = Vec::new();
    let mut sample_idx_of: Vec<u32> = Vec::new();
    let mut values: Vec<Vec<f32>> = (0..num_props).map(|_| Vec::<f32>::new()).collect();

    // Track bbox
    let mut min = Vec3 { x: f32::INFINITY, y: f32::INFINITY, z: f32::INFINITY };
    let mut max = Vec3 { x: f32::NEG_INFINITY, y: f32::NEG_INFINITY, z: f32::NEG_INFINITY };

    // Track global valid counts to skip mostly-missing properties
    let mut prop_valid_counts: Vec<u32> = vec![0; num_props];

    // Read holes until EOF
    loop {
        let hole_id = match input.next_string() {
            Some(s) => s,
            None => break,
        };

        let sx = input.next_f32().unwrap_or(f32::NAN);
        let sy = input.next_f32().unwrap_or(f32::NAN);
        let sz = input.next_f32().unwrap_or(f32::NAN);
        let dx = input.next_f32().unwrap_or(0.0);
        let dy = input.next_f32().unwrap_or(0.0);
        let dz = input.next_f32().unwrap_or(1.0);
        let _len = input.next_f32().unwrap_or(f32::NAN);
        let num_samples = input.next_usize().unwrap_or(0);

        let hid = if let Some(&id) = hole_map.get(&hole_id) {
            id
        } else {
            let id = hole_ids.len() as u32;
            hole_map.insert(hole_id.clone(), id);
            hole_ids.push(hole_id);
            id
        };

        let start = Vec3 { x: sx, y: sy, z: sz };
        let dir = Vec3 { x: dx, y: dy, z: dz };
        let dir_n = dir.norm();
        let dir_u = if dir_n.is_finite() && dir_n > 0.0 {
            dir.mul(1.0 / dir_n)
        } else {
            Vec3 { x: 0.0, y: 0.0, z: 1.0 }
        };

        // Read samples
        for sidx in 0..num_samples {
            // Each sample line: depth + num_props values
            let depth = input.next_f32().unwrap_or(f32::NAN);
            let p = if depth.is_finite() && sx.is_finite() && sy.is_finite() && sz.is_finite() {
                start.add(dir_u.mul(depth))
            } else {
                // If bad geometry, keep NaN positions; these will yield weak neighbor sets.
                Vec3 { x: f32::NAN, y: f32::NAN, z: f32::NAN }
            };

            // Store
            pos.push(p);
            hole_of.push(hid);
            sample_idx_of.push(sidx as u32);

            if p.x.is_finite() {
                min.x = min.x.min(p.x);
                max.x = max.x.max(p.x);
            }
            if p.y.is_finite() {
                min.y = min.y.min(p.y);
                max.y = max.y.max(p.y);
            }
            if p.z.is_finite() {
                min.z = min.z.min(p.z);
                max.z = max.z.max(p.z);
            }

            for pi in 0..num_props {
                let v = input.next_f32().unwrap_or(f32::NAN);
                if v.is_finite() {
                    prop_valid_counts[pi] = prop_valid_counts[pi].saturating_add(1);
                }
                values[pi].push(v);
            }
        }
    }

    let n = pos.len();
    if n == 0 || num_props == 0 {
        println!("0");
        return;
    }

    // If bbox is completely invalid (all NaN), bail gracefully.
    if !min.x.is_finite() || !min.y.is_finite() || !min.z.is_finite() || !max.x.is_finite() || !max.y.is_finite() || !max.z.is_finite() {
        println!("0");
        return;
    }

    let work = (n as u64) * (num_props as u64);

    // Adaptive strategy:
    // - Robust (median/MAD) when total work is small enough
    // - Faster (mean/std) for larger datasets
    let use_robust = work <= 3_000_000 && n <= 400_000;

    let (k, max_r, max_candidates, threshold) = if n <= 100_000 {
        (24usize, 3i32, 1200usize, if use_robust { 6.0 } else { 5.0 })
    } else if n <= 1_000_000 {
        (20usize, 2i32, 900usize, if use_robust { 6.0 } else { 5.0 })
    } else if n <= 3_000_000 {
        (16usize, 2i32, 700usize, 5.0)
    } else {
        (14usize, 1i32, 500usize, 5.0)
    };

    // Limit output size: keep only top-N suspects by confidence.
    let max_reports: usize = if work <= 200_000 { work as usize } else { 20_000 };

    // Skip properties that are almost entirely missing.
    let mut prop_enabled = vec![true; num_props];
    for p in 0..num_props {
        if prop_valid_counts[p] < 16 {
            prop_enabled[p] = false;
        }
    }

    // Build spatial index
    let grid = build_grid_index(&pos, min, max);

    // Buffers reused per point
    let mut cand_buf: Vec<(f32, u32)> = Vec::new();
    let mut neighbors: Vec<u32> = Vec::new();
    let mut neigh_vals: Vec<f32> = Vec::with_capacity(k);
    let mut devs: Vec<f32> = Vec::with_capacity(k);

    let mut heap: BinaryHeap<Reverse<Suspect>> = BinaryHeap::new();

    for i in 0..n {
        let pi = pos[i];
        if !pi.x.is_finite() || !pi.y.is_finite() || !pi.z.is_finite() {
            continue;
        }

        gather_neighbors(
            &grid,
            &pos,
            i,
            k,
            max_r,
            max_candidates,
            &mut cand_buf,
            &mut neighbors,
        );

        if neighbors.len() < 8 {
            continue; // not enough local context
        }

        for p in 0..num_props {
            if !prop_enabled[p] {
                continue;
            }
            let x = values[p][i];
            if !x.is_finite() {
                continue;
            }

            if use_robust {
                neigh_vals.clear();
                for &nb in &neighbors {
                    let v = values[p][nb as usize];
                    if v.is_finite() {
                        neigh_vals.push(v);
                    }
                }
                if neigh_vals.len() < 8 {
                    continue;
                }
                neigh_vals.sort_unstable_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
                let med = median_sorted(&neigh_vals);

                devs.clear();
                devs.extend(neigh_vals.iter().map(|&v| (v - med).abs()));
                devs.sort_unstable_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
                let mad = median_sorted(&devs);

                let diff = (x - med).abs() as f64;
                let scale = (1.4826f64) * (mad as f64);
                let eps = 1e-6f64 * ((med as f64).abs() + 1.0);

                let z = if scale + eps > 0.0 {
                    diff / (scale + eps)
                } else if diff > 0.0 {
                    1e9
                } else {
                    0.0
                };

                if z > threshold && diff > 0.0 {
                    let s = Suspect {
                        conf: z,
                        hole: hole_of[i],
                        sample_idx: sample_idx_of[i],
                        prop: p,
                    };
                    heap.push(Reverse(s));
                    if heap.len() > max_reports {
                        heap.pop();
                    }
                }
            } else {
                // Fast mean/std
                let mut cnt: u32 = 0;
                let mut sum: f64 = 0.0;
                let mut sumsq: f64 = 0.0;
                for &nb in &neighbors {
                    let v = values[p][nb as usize];
                    if v.is_finite() {
                        let vf = v as f64;
                        cnt += 1;
                        sum += vf;
                        sumsq += vf * vf;
                    }
                }
                if cnt < 8 {
                    continue;
                }
                let mean = sum / (cnt as f64);
                let var = (sumsq / (cnt as f64) - mean * mean).max(0.0);
                let std = var.sqrt();

                let diff = (x as f64 - mean).abs();
                let eps = 1e-6f64 * (mean.abs() + 1.0);
                let z = if std + eps > 0.0 {
                    diff / (std + eps)
                } else if diff > 0.0 {
                    1e9
                } else {
                    0.0
                };

                if z > threshold && diff > 0.0 {
                    let s = Suspect {
                        conf: z,
                        hole: hole_of[i],
                        sample_idx: sample_idx_of[i],
                        prop: p,
                    };
                    heap.push(Reverse(s));
                    if heap.len() > max_reports {
                        heap.pop();
                    }
                }
            }
        }
    }

    // Extract and sort descending by confidence
    let mut out: Vec<Suspect> = Vec::with_capacity(heap.len());
    while let Some(Reverse(s)) = heap.pop() {
        out.push(s);
    }
    out.sort_unstable_by(|a, b| b.conf.total_cmp(&a.conf));

    // Print
    let mut s = String::new();
    s.push_str(&format!("{}\n", out.len()));
    for r in out {
        let hid = hole_ids.get(r.hole as usize).map(|x| x.as_str()).unwrap_or("hole");
        let pname = prop_names.get(r.prop).map(|x| x.as_str()).unwrap_or("prop");
        s.push_str(hid);
        s.push(' ');
        s.push_str(&r.sample_idx.to_string());
        s.push(' ');
        s.push_str(pname);
        s.push(' ');
        s.push_str(&format!("{:.6}\n", r.conf));
    }
    print!("{s}");
}
