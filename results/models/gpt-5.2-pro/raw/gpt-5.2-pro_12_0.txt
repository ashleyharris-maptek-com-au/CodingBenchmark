using System;
using System.IO;
using System.Globalization;
using System.Collections.Generic;

public static class Program
{
    // ---------------- Fast scanner ----------------
    private sealed class FastScanner
    {
        private readonly Stream _s;
        private readonly byte[] _buf;
        private int _len, _ptr;

        public FastScanner(Stream s, int bufSize = 1 << 20)
        {
            _s = s;
            _buf = new byte[bufSize];
        }

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buf[_ptr++];
        }

        public bool TryNextDouble(out double val)
        {
            val = 0;
            byte c;
            do
            {
                c = Read();
                if (c == 0) return false;
            } while (c <= (byte)' ');

            bool neg = false;
            if (c == (byte)'-')
            {
                neg = true;
                c = Read();
            }

            long ip = 0;
            while (c >= (byte)'0' && c <= (byte)'9')
            {
                ip = ip * 10 + (c - (byte)'0');
                c = Read();
            }

            double fp = 0;
            double div = 1;
            if (c == (byte)'.')
            {
                c = Read();
                while (c >= (byte)'0' && c <= (byte)'9')
                {
                    fp = fp * 10 + (c - (byte)'0');
                    div *= 10;
                    c = Read();
                }
            }

            double x = ip + (fp / div);

            // exponent
            if (c == (byte)'e' || c == (byte)'E')
            {
                c = Read();
                bool eneg = false;
                if (c == (byte)'-') { eneg = true; c = Read(); }
                else if (c == (byte)'+') { c = Read(); }

                int e = 0;
                while (c >= (byte)'0' && c <= (byte)'9')
                {
                    e = e * 10 + (c - (byte)'0');
                    c = Read();
                }

                double pow10 = Math.Pow(10, eneg ? -e : e);
                x *= pow10;
            }

            val = neg ? -x : x;
            return true;
        }

        public int NextInt()
        {
            if (!TryNextDouble(out double d)) throw new EndOfStreamException();
            return (int)d;
        }

        public double NextDouble()
        {
            if (!TryNextDouble(out double d)) throw new EndOfStreamException();
            return d;
        }
    }

    // ---------------- Geometry ----------------
    private struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
    }

    private struct Orientation
    {
        public double Qw, Qx, Qy, Qz;

        // If IsPerm is true, rotation is just axis permutation + sign:
        // x' = s0 * v[p0], y' = s1 * v[p1], z' = s2 * v[p2]
        public bool IsPerm;
        public int P0, P1, P2;
        public int S0, S1, S2;

        // If IsPerm is false, use matrix
        public double M00, M01, M02;
        public double M10, M11, M12;
        public double M20, M21, M22;
    }

    private static Orientation MakePermOrientation(int p0, int p1, int p2, int s0, int s1, int s2)
    {
        // Build rotation matrix for permutation/sign and convert to quaternion.
        // v' = M v, where M row i picks component p_i with sign s_i.
        double m00 = (p0 == 0) ? s0 : 0;
        double m01 = (p0 == 1) ? s0 : 0;
        double m02 = (p0 == 2) ? s0 : 0;

        double m10 = (p1 == 0) ? s1 : 0;
        double m11 = (p1 == 1) ? s1 : 0;
        double m12 = (p1 == 2) ? s1 : 0;

        double m20 = (p2 == 0) ? s2 : 0;
        double m21 = (p2 == 1) ? s2 : 0;
        double m22 = (p2 == 2) ? s2 : 0;

        var q = QuaternionFromMatrix(m00, m01, m02, m10, m11, m12, m20, m21, m22);

        return new Orientation
        {
            IsPerm = true,
            P0 = p0, P1 = p1, P2 = p2,
            S0 = s0, S1 = s1, S2 = s2,
            Qw = q.qw, Qx = q.qx, Qy = q.qy, Qz = q.qz
        };
    }

    private static Orientation MakeGeneralOrientation(double qw, double qx, double qy, double qz)
    {
        NormalizeQuaternion(ref qw, ref qx, ref qy, ref qz);
        QuaternionToMatrix(qw, qx, qy, qz,
            out double m00, out double m01, out double m02,
            out double m10, out double m11, out double m12,
            out double m20, out double m21, out double m22);

        return new Orientation
        {
            IsPerm = false,
            Qw = qw, Qx = qx, Qy = qy, Qz = qz,
            M00 = m00, M01 = m01, M02 = m02,
            M10 = m10, M11 = m11, M12 = m12,
            M20 = m20, M21 = m21, M22 = m22
        };
    }

    private static void NormalizeQuaternion(ref double qw, ref double qx, ref double qy, ref double qz)
    {
        double n = Math.Sqrt(qw * qw + qx * qx + qy * qy + qz * qz);
        if (n <= 0) { qw = 1; qx = qy = qz = 0; return; }
        qw /= n; qx /= n; qy /= n; qz /= n;
        // Optional canonicalization: keep w >= 0 to reduce equivalent representations
        if (qw < 0) { qw = -qw; qx = -qx; qy = -qy; qz = -qz; }
    }

    private static void QuaternionToMatrix(
        double qw, double qx, double qy, double qz,
        out double m00, out double m01, out double m02,
        out double m10, out double m11, out double m12,
        out double m20, out double m21, out double m22)
    {
        // Standard conversion
        double xx = qx * qx, yy = qy * qy, zz = qz * qz;
        double xy = qx * qy, xz = qx * qz, yz = qy * qz;
        double wx = qw * qx, wy = qw * qy, wz = qw * qz;

        m00 = 1 - 2 * (yy + zz);
        m01 = 2 * (xy - wz);
        m02 = 2 * (xz + wy);

        m10 = 2 * (xy + wz);
        m11 = 1 - 2 * (xx + zz);
        m12 = 2 * (yz - wx);

        m20 = 2 * (xz - wy);
        m21 = 2 * (yz + wx);
        m22 = 1 - 2 * (xx + yy);
    }

    private static (double qw, double qx, double qy, double qz) QuaternionFromMatrix(
        double m00, double m01, double m02,
        double m10, double m11, double m12,
        double m20, double m21, double m22)
    {
        double tr = m00 + m11 + m22;
        double qw, qx, qy, qz;

        if (tr > 0)
        {
            double s = Math.Sqrt(tr + 1.0) * 2.0; // s=4*qw
            qw = 0.25 * s;
            qx = (m21 - m12) / s;
            qy = (m02 - m20) / s;
            qz = (m10 - m01) / s;
        }
        else if (m00 > m11 && m00 > m22)
        {
            double s = Math.Sqrt(1.0 + m00 - m11 - m22) * 2.0; // s=4*qx
            qw = (m21 - m12) / s;
            qx = 0.25 * s;
            qy = (m01 + m10) / s;
            qz = (m02 + m20) / s;
        }
        else if (m11 > m22)
        {
            double s = Math.Sqrt(1.0 + m11 - m00 - m22) * 2.0; // s=4*qy
            qw = (m02 - m20) / s;
            qx = (m01 + m10) / s;
            qy = 0.25 * s;
            qz = (m12 + m21) / s;
        }
        else
        {
            double s = Math.Sqrt(1.0 + m22 - m00 - m11) * 2.0; // s=4*qz
            qw = (m10 - m01) / s;
            qx = (m02 + m20) / s;
            qy = (m12 + m21) / s;
            qz = 0.25 * s;
        }

        NormalizeQuaternion(ref qw, ref qx, ref qy, ref qz);
        return (qw, qx, qy, qz);
    }

    private static (double qw, double qx, double qy, double qz) RandomUnitQuaternion(Random rng)
    {
        // Uniform random quaternion
        // (x,y,z,w) form from Shoemake
        double u1 = rng.NextDouble();
        double u2 = rng.NextDouble();
        double u3 = rng.NextDouble();

        double a = Math.Sqrt(1 - u1);
        double b = Math.Sqrt(u1);
        double t1 = 2 * Math.PI * u2;
        double t2 = 2 * Math.PI * u3;

        double qx = a * Math.Sin(t1);
        double qy = a * Math.Cos(t1);
        double qz = b * Math.Sin(t2);
        double qw = b * Math.Cos(t2);

        NormalizeQuaternion(ref qw, ref qx, ref qy, ref qz);
        return (qw, qx, qy, qz);
    }

    private static List<Orientation> BuildPerm24Orientations()
    {
        // All right-handed axis permutations with sign flips -> 24 rotations
        var list = new List<Orientation>(24);

        int[][] perms = new[]
        {
            new[]{0,1,2},
            new[]{0,2,1},
            new[]{1,0,2},
            new[]{1,2,0},
            new[]{2,0,1},
            new[]{2,1,0},
        };

        foreach (var p in perms)
        {
            int parity = PermutationParity(p[0], p[1], p[2]); // +1 even, -1 odd
            for (int s0 = -1; s0 <= 1; s0 += 2)
            for (int s1 = -1; s1 <= 1; s1 += 2)
            for (int s2 = -1; s2 <= 1; s2 += 2)
            {
                int det = parity * s0 * s1 * s2;
                if (det != 1) continue; // keep right-handed rotations only
                list.Add(MakePermOrientation(p[0], p[1], p[2], s0, s1, s2));
            }
        }

        return list; // should be 24
    }

    private static int PermutationParity(int a, int b, int c)
    {
        // parity of permutation (a,b,c) of (0,1,2)
        // even -> +1, odd -> -1
        int inv = 0;
        if (a > b) inv++;
        if (a > c) inv++;
        if (b > c) inv++;
        return (inv % 2 == 0) ? 1 : -1;
    }

    // ---------------- Solver ----------------
    private static void ComputeBoundsPerm(
        Orientation o,
        double[] minComp, double[] maxComp,
        out double minX, out double maxX,
        out double minY, out double maxY,
        out double minZ, out double maxZ)
    {
        // axis X'
        GetSignedBounds(o.P0, o.S0, minComp, maxComp, out minX, out maxX);
        GetSignedBounds(o.P1, o.S1, minComp, maxComp, out minY, out maxY);
        GetSignedBounds(o.P2, o.S2, minComp, maxComp, out minZ, out maxZ);
    }

    private static void GetSignedBounds(int comp, int sign, double[] minComp, double[] maxComp, out double mn, out double mx)
    {
        if (sign >= 0)
        {
            mn = minComp[comp];
            mx = maxComp[comp];
        }
        else
        {
            mn = -maxComp[comp];
            mx = -minComp[comp];
        }
    }

    private static void ComputeBoundsGeneral(Orientation o, Vec3[] verts,
        out double minX, out double maxX,
        out double minY, out double maxY,
        out double minZ, out double maxZ)
    {
        double mnx = double.PositiveInfinity, mxx = double.NegativeInfinity;
        double mny = double.PositiveInfinity, mxy = double.NegativeInfinity;
        double mnz = double.PositiveInfinity, mxz = double.NegativeInfinity;

        double m00 = o.M00, m01 = o.M01, m02 = o.M02;
        double m10 = o.M10, m11 = o.M11, m12 = o.M12;
        double m20 = o.M20, m21 = o.M21, m22 = o.M22;

        for (int i = 0; i < verts.Length; i++)
        {
            var v = verts[i];
            double x = m00 * v.X + m01 * v.Y + m02 * v.Z;
            double y = m10 * v.X + m11 * v.Y + m12 * v.Z;
            double z = m20 * v.X + m21 * v.Y + m22 * v.Z;

            if (x < mnx) mnx = x; if (x > mxx) mxx = x;
            if (y < mny) mny = y; if (y > mxy) mxy = y;
            if (z < mnz) mnz = z; if (z > mxz) mxz = z;
        }

        minX = mnx; maxX = mxx;
        minY = mny; maxY = mxy;
        minZ = mnz; maxZ = mxz;
    }

    private static long SafeMul3(long a, long b, long c)
    {
        // For these problem sizes, overflow is unlikely, but keep it safe-ish.
        checked { return a * b * c; }
    }

    public static void Main()
    {
        CultureInfo inv = CultureInfo.InvariantCulture;
        var fs = new FastScanner(Console.OpenStandardInput());

        int V = fs.NextInt();
        int F = fs.NextInt();

        var verts = new Vec3[V];

        double minX0 = double.PositiveInfinity, maxX0 = double.NegativeInfinity;
        double minY0 = double.PositiveInfinity, maxY0 = double.NegativeInfinity;
        double minZ0 = double.PositiveInfinity, maxZ0 = double.NegativeInfinity;

        for (int i = 0; i < V; i++)
        {
            double x = fs.NextDouble();
            double y = fs.NextDouble();
            double z = fs.NextDouble();
            verts[i] = new Vec3(x, y, z);

            if (x < minX0) minX0 = x; if (x > maxX0) maxX0 = x;
            if (y < minY0) minY0 = y; if (y > maxY0) maxY0 = y;
            if (z < minZ0) minZ0 = z; if (z > maxZ0) maxZ0 = z;
        }

        // Faces are not used by this AABB-based packer, but we must consume input.
        for (int i = 0; i < F; i++)
        {
            _ = fs.NextInt();
            _ = fs.NextInt();
            _ = fs.NextInt();
        }

        double W = fs.NextDouble();
        double H = fs.NextDouble();
        double D = fs.NextDouble();

        double[] minComp = { minX0, minY0, minZ0 };
        double[] maxComp = { maxX0, maxY0, maxZ0 };

        double maxDim = Math.Max(W, Math.Max(H, D));
        double vol = W * H * D;

        // --- Adaptive strategy selection ---
        // Always include fast 24 right-angle orientations (cheap bounds via component mins/maxs).
        var orientations = BuildPerm24Orientations();

        // For small/medium problems (smaller container and not too many vertices),
        // also sample random orientations, because a tighter AABB can dramatically increase count.
        bool allowRandom = (maxDim <= 40.0 && V <= 20000) || (vol <= 20000.0 && V <= 50000);
        if (allowRandom)
        {
            // Cap total work roughly by limiting samples as V grows.
            int budgetOps = 5_000_000; // V * samples
            int samples = Math.Max(30, Math.Min(250, budgetOps / Math.Max(1, V)));

            var rng = new Random(1);
            for (int i = 0; i < samples; i++)
            {
                var q = RandomUnitQuaternion(rng);
                orientations.Add(MakeGeneralOrientation(q.qw, q.qx, q.qy, q.qz));
            }

            // Also include a few deterministic non-90deg rotations (45° around axes) to help some cases.
            double s2 = Math.Sqrt(0.5);
            orientations.Add(MakeGeneralOrientation(s2, s2, 0, 0)); // 90° around X actually (still ok)
            orientations.Add(MakeGeneralOrientation(s2, 0, s2, 0)); // 90° around Y
            orientations.Add(MakeGeneralOrientation(s2, 0, 0, s2)); // 90° around Z
        }

        // --- Evaluate best orientation by maximum axis-aligned tiling count ---
        const double eps = 1e-12;

        long bestCount = 0;
        Orientation bestOri = default;
        double bestDx = 0, bestDy = 0, bestDz = 0;
        double bestMinX = 0, bestMinY = 0, bestMinZ = 0;
        long bestNx = 0, bestNy = 0, bestNz = 0;

        for (int i = 0; i < orientations.Count; i++)
        {
            var o = orientations[i];

            double minX, maxX, minY, maxY, minZ, maxZ;
            if (o.IsPerm)
                ComputeBoundsPerm(o, minComp, maxComp, out minX, out maxX, out minY, out maxY, out minZ, out maxZ);
            else
                ComputeBoundsGeneral(o, verts, out minX, out maxX, out minY, out maxY, out minZ, out maxZ);

            double dx = maxX - minX;
            double dy = maxY - minY;
            double dz = maxZ - minZ;

            if (!(dx > eps && dy > eps && dz > eps)) continue;

            long nx = (dx <= W + eps) ? (long)Math.Floor((W + eps) / dx) : 0;
            long ny = (dy <= H + eps) ? (long)Math.Floor((H + eps) / dy) : 0;
            long nz = (dz <= D + eps) ? (long)Math.Floor((D + eps) / dz) : 0;

            if (nx <= 0 || ny <= 0 || nz <= 0) continue;

            long count;
            try { count = SafeMul3(nx, ny, nz); }
            catch { continue; }

            // Tie-breaker: prefer smaller cell volume (less wasted space).
            double cellVol = dx * dy * dz;

            if (count > bestCount ||
               (count == bestCount && count > 0 && cellVol < bestDx * bestDy * bestDz))
            {
                bestCount = count;
                bestOri = o;
                bestDx = dx; bestDy = dy; bestDz = dz;
                bestMinX = minX; bestMinY = minY; bestMinZ = minZ;
                bestNx = nx; bestNy = ny; bestNz = nz;
            }
        }

        // --- Output ---
        using var outStream = Console.OpenStandardOutput();
        using var writer = new StreamWriter(outStream, System.Text.Encoding.ASCII, bufferSize: 1 << 20);
        writer.NewLine = "\n";

        if (bestCount <= 0)
        {
            writer.WriteLine("0");
            writer.Flush();
            return;
        }

        writer.WriteLine(bestCount.ToString(inv));

        // Quaternion constant per placement
        string qStr =
            bestOri.Qw.ToString("G17", inv) + " " +
            bestOri.Qx.ToString("G17", inv) + " " +
            bestOri.Qy.ToString("G17", inv) + " " +
            bestOri.Qz.ToString("G17", inv);

        // Place each copy so its rotated AABB occupies a unique grid cell:
        // cell min at (ix*dx, iy*dy, iz*dz), so translation = cellMin - rotatedMin.
        // This guarantees: inside container and no overlaps (by non-overlapping AABBs).
        for (long iz = 0; iz < bestNz; iz++)
        {
            double baseZ = iz * bestDz - bestMinZ;
            for (long iy = 0; iy < bestNy; iy++)
            {
                double baseY = iy * bestDy - bestMinY;
                for (long ix = 0; ix < bestNx; ix++)
                {
                    double baseX = ix * bestDx - bestMinX;

                    writer.Write(baseX.ToString("G17", inv));
                    writer.Write(' ');
                    writer.Write(baseY.ToString("G17", inv));
                    writer.Write(' ');
                    writer.Write(baseZ.ToString("G17", inv));
                    writer.Write(' ');
                    writer.WriteLine(qStr);
                }
            }
        }

        writer.Flush();
    }
}
