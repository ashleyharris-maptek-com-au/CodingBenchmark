```spirv
; SPIR-V
; Version: 1.0
; Generator: hand-written
; Bound: 200
OpCapability Shader
OpCapability Geometry
%GLSLstd450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %emitEdgeQuad "emitEdgeQuad"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"
OpName %inColor "color"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_in 0 Offset 0

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 Offset 0

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_6 = OpConstant %uint 6
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%arr_v3_in = OpTypeArray %v3float %uint_6

%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float

%ptr_Output_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_Input_arr_pv Input
%inWorldPos = OpVariable %ptr_Input_arr_v3 Input
%inNormal = OpVariable %ptr_Input_arr_v3 Input
%inColor = OpVariable %ptr_Input_arr_v3 Input

%gl_out = OpVariable %ptr_Output_pv Output
%outColor = OpVariable %ptr_Output_v3float Output

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_5 = OpConstant %float 0.5
%float_halfThickness = OpConstant %float 0.004

%v4_0 = OpConstantComposite %v4float %float_0 %float_0 %float_0 %float_0

%func_void = OpTypeFunction %void
%func_emit = OpTypeFunction %void %v4float %v4float %v3float %v3float

%emitEdgeQuad = OpFunction %void None %func_emit
%clipA = OpFunctionParameter %v4float
%clipB = OpFunctionParameter %v4float
%colA = OpFunctionParameter %v3float
%colB = OpFunctionParameter %v3float
%emit_entry = OpLabel

; avgColor = (colA + colB) * 0.5
%colSum = OpFAdd %v3float %colA %colB
%avgColor = OpVectorTimesScalar %v3float %colSum %float_0_5

; Extract clip.xy and w for A
%wA = OpCompositeExtract %float %clipA 3
%xyA = OpVectorShuffle %v2float %clipA %clipA 0 1
%wA2 = OpCompositeConstruct %v2float %wA %wA
%ndcA = OpFDiv %v2float %xyA %wA2

; Extract clip.xy and w for B
%wB = OpCompositeExtract %float %clipB 3
%xyB = OpVectorShuffle %v2float %clipB %clipB 0 1
%wB2 = OpCompositeConstruct %v2float %wB %wB
%ndcB = OpFDiv %v2float %xyB %wB2

; dir = normalize(ndcB - ndcA)
%dir = OpFSub %v2float %ndcB %ndcA
%dirDot = OpDot %float %dir %dir
%dirLen = OpExtInst %float %GLSLstd450 Sqrt %dirDot
%invLen = OpFDiv %float %float_1 %dirLen
%dirN = OpVectorTimesScalar %v2float %dir %invLen

; perp = (-dirN.y, dirN.x)
%dirNx = OpCompositeExtract %float %dirN 0
%dirNy = OpCompositeExtract %float %dirN 1
%negDirNy = OpFNegate %float %dirNy
%perp = OpCompositeConstruct %v2float %negDirNy %dirNx

; offsetNDC = perp * halfThickness
%offsetNDC = OpVectorTimesScalar %v2float %perp %float_halfThickness

; offsetClipA2 = offsetNDC * wA, offsetClipB2 = offsetNDC * wB
%offsetClipA2 = OpVectorTimesScalar %v2float %offsetNDC %wA
%offsetClipB2 = OpVectorTimesScalar %v2float %offsetNDC %wB

; Build v4 offsets (x,y,0,0)
%offAx = OpCompositeExtract %float %offsetClipA2 0
%offAy = OpCompositeExtract %float %offsetClipA2 1
%offA4 = OpCompositeConstruct %v4float %offAx %offAy %float_0 %float_0

%offBx = OpCompositeExtract %float %offsetClipB2 0
%offBy = OpCompositeExtract %float %offsetClipB2 1
%offB4 = OpCompositeConstruct %v4float %offBx %offBy %float_0 %float_0

; Quad vertices in triangle strip order: A+, A-, B+, B-
%posAplus = OpFAdd %v4float %clipA %offA4
%posAminus = OpFSub %v4float %clipA %offA4
%posBplus = OpFAdd %v4float %clipB %offB4
%posBminus = OpFSub %v4float %clipB %offB4

%outPosPtr = OpAccessChain %ptr_Output_v4float %gl_out %int_0

; Emit A+
OpStore %outPosPtr %posAplus
OpStore %outColor %avgColor
OpEmitVertex

; Emit A-
OpStore %outPosPtr %posAminus
OpStore %outColor %avgColor
OpEmitVertex

; Emit B+
OpStore %outPosPtr %posBplus
OpStore %outColor %avgColor
OpEmitVertex

; Emit B-
OpStore %outPosPtr %posBminus
OpStore %outColor %avgColor
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd

%main = OpFunction %void None %func_void
%main_entry = OpLabel

; Load main triangle clip positions from gl_in indices 0,2,4 (member 0 = Position)
%p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p2_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%p4_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_4 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p2 = OpLoad %v4float %p2_ptr
%p4 = OpLoad %v4float %p4_ptr

; Load endpoint colors from location=2 array indices 0,2,4
%c0_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_0
%c2_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_2
%c4_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_4
%c0 = OpLoad %v3float %c0_ptr
%c2 = OpLoad %v3float %c2_ptr
%c4 = OpLoad %v3float %c4_ptr

; Emit 3 edge quads: (0->2), (2->4), (4->0)
OpFunctionCall %void %emitEdgeQuad %p0 %p2 %c0 %c2
OpFunctionCall %void %emitEdgeQuad %p2 %p4 %c2 %c4
OpFunctionCall %void %emitEdgeQuad %p4 %p0 %c4 %c0

OpReturn
OpFunctionEnd
```
