use std::collections::HashMap;
use std::f64::consts::PI;
use std::hash::{BuildHasherDefault, Hasher};
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy, Debug)]
struct Aabb {
    minx: f64,
    miny: f64,
    maxx: f64,
    maxy: f64,
}

impl Aabb {
    fn empty() -> Self {
        Self {
            minx: f64::INFINITY,
            miny: f64::INFINITY,
            maxx: f64::NEG_INFINITY,
            maxy: f64::NEG_INFINITY,
        }
    }
    fn from_pts(pts: &[Pt]) -> Self {
        let mut b = Self::empty();
        for &p in pts {
            b.minx = b.minx.min(p.x);
            b.miny = b.miny.min(p.y);
            b.maxx = b.maxx.max(p.x);
            b.maxy = b.maxy.max(p.y);
        }
        b
    }
    fn width(&self) -> f64 {
        self.maxx - self.minx
    }
    fn height(&self) -> f64 {
        self.maxy - self.miny
    }
    fn translate(&self, tx: f64, ty: f64) -> Self {
        Self {
            minx: self.minx + tx,
            miny: self.miny + ty,
            maxx: self.maxx + tx,
            maxy: self.maxy + ty,
        }
    }
    fn intersects(&self, o: &Aabb, eps: f64) -> bool {
        !(self.maxx <= o.minx + eps
            || self.minx >= o.maxx - eps
            || self.maxy <= o.miny + eps
            || self.miny >= o.maxy - eps)
    }
    fn contains_aabb(&self, o: &Aabb, eps: f64) -> bool {
        self.minx <= o.minx + eps
            && self.miny <= o.miny + eps
            && self.maxx >= o.maxx - eps
            && self.maxy >= o.maxy - eps
    }
}

#[derive(Clone, Debug)]
struct Edge {
    a: Pt,
    b: Pt,
    bb: Aabb,
}

fn edges_of(poly: &[Pt]) -> Vec<Edge> {
    let n = poly.len();
    let mut out = Vec::with_capacity(n);
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        let bb = Aabb {
            minx: a.x.min(b.x),
            miny: a.y.min(b.y),
            maxx: a.x.max(b.x),
            maxy: a.y.max(b.y),
        };
        out.push(Edge { a, b, bb });
    }
    out
}

// ---------------- Fast hash (FxHasher-like) ----------------

#[derive(Default)]
struct FxHasher {
    hash: u64,
}
impl Hasher for FxHasher {
    fn write(&mut self, bytes: &[u8]) {
        // simple 64-bit FNV-1a
        let mut h = if self.hash == 0 {
            0xcbf29ce484222325u64
        } else {
            self.hash
        };
        for &b in bytes {
            h ^= b as u64;
            h = h.wrapping_mul(0x100000001b3);
        }
        self.hash = h;
    }
    fn write_u64(&mut self, i: u64) {
        let mut h = if self.hash == 0 {
            0xcbf29ce484222325u64
        } else {
            self.hash
        };
        h ^= i;
        h = h.wrapping_mul(0x100000001b3);
        self.hash = h;
    }
    fn finish(&self) -> u64 {
        self.hash
    }
}

type FxBuildHasher = BuildHasherDefault<FxHasher>;
type FxHashMap<K, V> = HashMap<K, V, FxBuildHasher>;

// ---------------- Spatial hash grid ----------------

#[derive(Clone)]
struct SpatialHash {
    cell: f64,
    map: FxHashMap<u64, Vec<usize>>,
}

impl SpatialHash {
    fn new(cell: f64) -> Self {
        Self {
            cell: cell.max(1e-9),
            map: FxHashMap::default(),
        }
    }
    #[inline]
    fn key(ix: i32, iy: i32) -> u64 {
        ((ix as u32) as u64) << 32 | ((iy as u32) as u64)
    }
    fn cell_range(&self, bb: &Aabb) -> (i32, i32, i32, i32) {
        let ix0 = (bb.minx / self.cell).floor() as i32;
        let iy0 = (bb.miny / self.cell).floor() as i32;
        let ix1 = (bb.maxx / self.cell).floor() as i32;
        let iy1 = (bb.maxy / self.cell).floor() as i32;
        (ix0, iy0, ix1, iy1)
    }
    fn insert_aabb(&mut self, bb: &Aabb, idx: usize) {
        let (ix0, iy0, ix1, iy1) = self.cell_range(bb);
        for iy in iy0..=iy1 {
            for ix in ix0..=ix1 {
                self.map.entry(Self::key(ix, iy)).or_default().push(idx);
            }
        }
    }
    fn query_aabb(&self, bb: &Aabb, out: &mut Vec<usize>) {
        out.clear();
        let (ix0, iy0, ix1, iy1) = self.cell_range(bb);
        for iy in iy0..=iy1 {
            for ix in ix0..=ix1 {
                if let Some(v) = self.map.get(&Self::key(ix, iy)) {
                    out.extend_from_slice(v);
                }
            }
        }
        out.sort_unstable();
        out.dedup();
    }
}

// ---------------- Geometry predicates ----------------

const EPS: f64 = 1e-9;

#[inline]
fn cross(ax: f64, ay: f64, bx: f64, by: f64) -> f64 {
    ax * by - ay * bx
}

#[inline]
fn orient(a: Pt, b: Pt, c: Pt) -> f64 {
    cross(b.x - a.x, b.y - a.y, c.x - a.x, c.y - a.y)
}

#[inline]
fn on_segment(a: Pt, b: Pt, p: Pt) -> bool {
    let minx = a.x.min(b.x) - 1e-10;
    let maxx = a.x.max(b.x) + 1e-10;
    let miny = a.y.min(b.y) - 1e-10;
    let maxy = a.y.max(b.y) + 1e-10;
    (p.x >= minx && p.x <= maxx && p.y >= miny && p.y <= maxy) && orient(a, b, p).abs() <= 1e-10
}

#[derive(Clone, Copy, Debug, PartialEq, Eq)]
enum SegIsect {
    None,
    Touch,          // only at endpoints or tangential single-point touch
    Proper,         // strict crossing (interior intersection)
    ColinearOverlap // overlapping with non-zero length
}

fn seg_intersect_type(a: Pt, b: Pt, c: Pt, d: Pt) -> SegIsect {
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);

    let s1 = o1.signum();
    let s2 = o2.signum();
    let s3 = o3.signum();
    let s4 = o4.signum();

    // Colinear cases
    if o1.abs() <= EPS && o2.abs() <= EPS && o3.abs() <= EPS && o4.abs() <= EPS {
        // project overlap on x or y
        let (ax0, ax1) = (a.x.min(b.x), a.x.max(b.x));
        let (ay0, ay1) = (a.y.min(b.y), a.y.max(b.y));
        let (cx0, cx1) = (c.x.min(d.x), c.x.max(d.x));
        let (cy0, cy1) = (c.y.min(d.y), c.y.max(d.y));
        let ox = ax0.max(cx0) <= ax1.min(cx1) + 1e-12;
        let oy = ay0.max(cy0) <= ay1.min(cy1) + 1e-12;
        if ox && oy {
            // Distinguish point-touch vs overlap: check if intersection is more than a single point
            // Use 1D overlap length on the dominant axis.
            let (l0, r0, l1, r1) = if (ax1 - ax0) >= (ay1 - ay0) {
                (ax0, ax1, cx0, cx1)
            } else {
                (ay0, ay1, cy0, cy1)
            };
            let inter_l = l0.max(l1);
            let inter_r = r0.min(r1);
            if inter_r - inter_l > 1e-10 {
                SegIsect::ColinearOverlap
            } else {
                SegIsect::Touch
            }
        } else {
            SegIsect::None
        }
    } else {
        // Endpoint touches
        if o1.abs() <= EPS && on_segment(a, b, c) {
            return SegIsect::Touch;
        }
        if o2.abs() <= EPS && on_segment(a, b, d) {
            return SegIsect::Touch;
        }
        if o3.abs() <= EPS && on_segment(c, d, a) {
            return SegIsect::Touch;
        }
        if o4.abs() <= EPS && on_segment(c, d, b) {
            return SegIsect::Touch;
        }

        // Proper intersection
        if s1 * s2 < 0.0 && s3 * s4 < 0.0 {
            SegIsect::Proper
        } else {
            SegIsect::None
        }
    }
}

fn point_on_poly_edge(p: Pt, poly: &[Pt]) -> bool {
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if on_segment(a, b, p) {
            return true;
        }
    }
    false
}

// Ray casting; inclusive means boundary counts as inside.
fn point_in_poly_inclusive(p: Pt, poly: &[Pt]) -> bool {
    if point_on_poly_edge(p, poly) {
        return true;
    }
    let mut inside = false;
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        // check if edge straddles ray
        let (ay, by) = (a.y, b.y);
        let (ax, bx) = (a.x, b.x);
        let cond = (ay > p.y) != (by > p.y);
        if cond {
            let xint = (bx - ax) * (p.y - ay) / (by - ay + 0.0) + ax;
            if xint > p.x {
                inside = !inside;
            }
        }
    }
    inside
}

fn point_in_poly_strict(p: Pt, poly: &[Pt]) -> bool {
    if point_on_poly_edge(p, poly) {
        return false;
    }
    point_in_poly_inclusive(p, poly)
}

// Polygon overlap (interior overlap). Touching at a point/edge is allowed.
fn polys_overlap(poly_a: &[Pt], edges_a: &[Edge], bb_a: &Aabb, poly_b: &[Pt], edges_b: &[Edge], bb_b: &Aabb) -> bool {
    if !bb_a.intersects(bb_b, 0.0) {
        return false;
    }

    // Edge intersections
    // Iterate smaller edge list outside for fewer bbox checks
    let (outer_poly, outer_edges, inner_poly, inner_edges) = if edges_a.len() <= edges_b.len() {
        (poly_a, edges_a, poly_b, edges_b)
    } else {
        (poly_b, edges_b, poly_a, edges_a)
    };

    for e1 in outer_edges {
        for e2 in inner_edges {
            if !e1.bb.intersects(&e2.bb, 0.0) {
                continue;
            }
            match seg_intersect_type(e1.a, e1.b, e2.a, e2.b) {
                SegIsect::Proper | SegIsect::ColinearOverlap => return true,
                SegIsect::Touch | SegIsect::None => {}
            }
        }
    }

    // Containment (strict): one vertex strictly inside other
    if point_in_poly_strict(outer_poly[0], inner_poly) {
        return true;
    }
    if point_in_poly_strict(inner_poly[0], outer_poly) {
        return true;
    }
    false
}

// Check piece polygon is inside stock polygon (touch allowed).
// Uses: (1) all vertices inside/inclusive; (2) no proper crossings with stock boundary.
fn poly_inside_stock(
    piece: &[Pt],
    piece_edges: &[Edge],
    stock: &[Pt],
    stock_edge_grid: &SpatialHash,
    stock_edges: &[Edge],
) -> bool {
    // vertices inside
    for &p in piece {
        if !point_in_poly_inclusive(p, stock) {
            return false;
        }
    }

    // no proper crossings with boundary
    let mut cand = Vec::<usize>::new();
    for e in piece_edges {
        stock_edge_grid.query_aabb(&e.bb, &mut cand);
        for &ei in &cand {
            let se = &stock_edges[ei];
            if !e.bb.intersects(&se.bb, 0.0) {
                continue;
            }
            if seg_intersect_type(e.a, e.b, se.a, se.b) == SegIsect::Proper {
                return false;
            }
        }
    }
    true
}

#[derive(Clone, Copy, Debug)]
enum Mode {
    Quality,
    Balanced,
    Fast,
}

fn shoelace_area(poly: &[Pt]) -> f64 {
    let n = poly.len();
    let mut s = 0.0;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        s += a.x * b.y - a.y * b.x;
    }
    0.5 * s.abs()
}

fn rotate_pt(p: Pt, ang: f64) -> Pt {
    let (s, c) = ang.sin_cos();
    Pt {
        x: c * p.x - s * p.y,
        y: s * p.x + c * p.y,
    }
}

fn norm_angle(mut a: f64) -> f64 {
    while a >= PI {
        a -= 2.0 * PI;
    }
    while a < -PI {
        a += 2.0 * PI;
    }
    a
}

fn angles_for_polygon(poly: &[Pt], mode: Mode) -> Vec<f64> {
    match mode {
        Mode::Fast => vec![0.0],
        Mode::Balanced => {
            // a few canonical angles (often enough for rectangles-ish problems)
            vec![0.0, 0.5 * PI, PI, 1.5 * PI]
        }
        Mode::Quality => {
            // derive angles to align edges to x-axis; dedup by coarse rounding
            let n = poly.len();
            let mut angs: Vec<f64> = Vec::with_capacity(n + 4);
            angs.push(0.0);
            for i in 0..n {
                let a = poly[i];
                let b = poly[(i + 1) % n];
                let dx = b.x - a.x;
                let dy = b.y - a.y;
                if dx.abs() + dy.abs() < 1e-12 {
                    continue;
                }
                let ang = norm_angle(-dy.atan2(dx));
                angs.push(ang);
                angs.push(norm_angle(ang + 0.5 * PI));
            }
            // dedup by binning
            angs.sort_by(|a, b| a.partial_cmp(b).unwrap());
            let mut uniq = Vec::new();
            let mut last = None::<f64>;
            for &a in &angs {
                let k = (a * 1000.0).round() / 1000.0;
                if last.map_or(true, |l| (k - l).abs() > 1e-12) {
                    uniq.push(k);
                    last = Some(k);
                }
            }
            // limit to a reasonable count
            let max_angles = 12usize;
            if uniq.len() <= max_angles {
                uniq
            } else {
                // sample evenly
                let mut sampled = Vec::with_capacity(max_angles);
                for i in 0..max_angles {
                    let idx = i * (uniq.len() - 1) / (max_angles - 1);
                    sampled.push(uniq[idx]);
                }
                sampled.sort_by(|a, b| a.partial_cmp(b).unwrap());
                sampled.dedup_by(|a, b| (*a - *b).abs() < 1e-12);
                sampled
            }
        }
    }
}

fn build_variants(poly: &[Pt], angles: &[f64]) -> Vec<Variant> {
    let mut out = Vec::with_capacity(angles.len());
    for &a in angles {
        let mut v = Vec::with_capacity(poly.len());
        for &p in poly {
            v.push(rotate_pt(p, a));
        }
        let aabb = Aabb::from_pts(&v);
        let edges = edges_of(&v);
        out.push(Variant {
            angle: a,
            verts: v,
            aabb,
            edges,
        });
    }
    out
}

// deterministic xorshift RNG (for fallback search)
#[derive(Clone)]
struct Rng64 {
    s: u64,
}
impl Rng64 {
    fn new(seed: u64) -> Self {
        Self { s: seed.max(1) }
    }
    fn next_u64(&mut self) -> u64 {
        let mut x = self.s;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.s = x;
        x
    }
    fn next_f64(&mut self) -> f64 {
        // [0,1)
        let u = self.next_u64() >> 11; // 53 bits
        (u as f64) * (1.0 / ((1u64 << 53) as f64))
    }
}

fn try_place_in_bin(
    piece: &Piece,
    bin: &mut StockBin,
    tpl: &StockTemplate,
    mode: Mode,
    stock_index: usize,
    rng: &mut Rng64,
) -> Option<Placement> {
    let stock_bb = tpl.aabb;

    let scan_limit = match mode {
        Mode::Fast => 24usize,
        Mode::Balanced => 120usize,
        Mode::Quality => 500usize,
    };

    let random_limit = match mode {
        Mode::Fast => 40usize,
        Mode::Balanced => 200usize,
        Mode::Quality => 1200usize,
    };

    let mut near = Vec::<usize>::new();

    for var in &piece.variants {
        // quick reject: variant bbox can't fit stock bbox at all
        if var.aabb.width() > stock_bb.width() + 1e-9 || var.aabb.height() > stock_bb.height() + 1e-9 {
            continue;
        }

        // Candidate positions:
        // 1) Shelf position
        // We will attempt a few local adjustments near the shelf too.
        let mut candidates: Vec<(f64, f64)> = Vec::new();

        // helper: compute translation to put variant AABB min at (target_x, target_y)
        let mut push_by_min = |target_minx: f64, target_miny: f64, cands: &mut Vec<(f64, f64)>| {
            let tx = target_minx - var.aabb.minx;
            let ty = target_miny - var.aabb.miny;
            cands.push((tx, ty));
        };

        let mut shelf_x = bin.shelf_x;
        let mut shelf_y = bin.shelf_y;
        let mut shelf_h = bin.shelf_h;

        // ensure shelf is within stock
        if shelf_x.is_nan() || shelf_y.is_nan() || shelf_h.is_nan() {
            shelf_x = 0.0;
            shelf_y = 0.0;
            shelf_h = 0.0;
        }

        // wrap to next shelf if needed
        if shelf_x + var.aabb.width() > stock_bb.width() + 1e-9 {
            shelf_x = 0.0;
            shelf_y += shelf_h;
            shelf_h = 0.0;
        }

        // base shelf candidate
        let base_minx = stock_bb.minx + shelf_x;
        let base_miny = stock_bb.miny + shelf_y;
        push_by_min(base_minx, base_miny, &mut candidates);

        // local nudges (helps when exact polygon constraints fail)
        let step = (tpl.cell * 0.5).max(1e-3);
        for dy in 0..3 {
            for dx in 0..3 {
                let mx = base_minx + (dx as f64) * step;
                let my = base_miny + (dy as f64) * step;
                push_by_min(mx, my, &mut candidates);
            }
        }

        // 2) Coarse scan if needed
        let w = var.aabb.width();
        let h = var.aabb.height();
        let sx_step = (w * 0.5).max(tpl.cell * 0.75).max(1e-3);
        let sy_step = (h * 0.5).max(tpl.cell * 0.75).max(1e-3);

        let mut scanned = 0usize;
        let mut y = stock_bb.miny;
        while y + h <= stock_bb.maxy + 1e-9 && scanned < scan_limit {
            let mut x = stock_bb.minx;
            while x + w <= stock_bb.maxx + 1e-9 && scanned < scan_limit {
                push_by_min(x, y, &mut candidates);
                scanned += 1;
                x += sx_step;
            }
            y += sy_step;
        }

        // 3) Random fallback
        for _ in 0..random_limit {
            let rx = stock_bb.minx + rng.next_f64() * (stock_bb.width() - w).max(0.0);
            let ry = stock_bb.miny + rng.next_f64() * (stock_bb.height() - h).max(0.0);
            push_by_min(rx, ry, &mut candidates);
        }

        // Evaluate candidates
        let mut tmp_poly: Vec<Pt> = Vec::with_capacity(var.verts.len());
        let mut tmp_edges: Vec<Edge> = Vec::with_capacity(var.edges.len());

        for (tx, ty) in candidates {
            let bb_t = var.aabb.translate(tx, ty);
            if !stock_bb.contains_aabb(&bb_t, 1e-9) {
                continue;
            }

            // transform polygon into tmp_poly
            tmp_poly.clear();
            for &p in &var.verts {
                tmp_poly.push(Pt { x: p.x + tx, y: p.y + ty });
            }
            // transform edges into tmp_edges (reuse endpoints from tmp_poly)
            tmp_edges.clear();
            for i in 0..tmp_poly.len() {
                let a = tmp_poly[i];
                let b = tmp_poly[(i + 1) % tmp_poly.len()];
                tmp_edges.push(Edge {
                    a,
                    b,
                    bb: Aabb {
                        minx: a.x.min(b.x),
                        miny: a.y.min(b.y),
                        maxx: a.x.max(b.x),
                        maxy: a.y.max(b.y),
                    },
                });
            }

            if !poly_inside_stock(&tmp_poly, &tmp_edges, &tpl.poly, &tpl.stock_edge_grid, &tpl.stock_edges) {
                continue;
            }

            // overlap check with nearby pieces
            bin.grid.query_aabb(&bb_t, &mut near);
            let mut ok = true;
            for &pi in &near {
                let other = &bin.placed[pi];
                if !bb_t.intersects(&other.aabb, 0.0) {
                    continue;
                }
                if polys_overlap(&tmp_poly, &tmp_edges, &bb_t, &other.poly, &other.edges, &other.aabb) {
                    ok = false;
                    break;
                }
            }
            if !ok {
                continue;
            }

            // success: commit
            let placed_idx = bin.placed.len();
            bin.placed.push(Placed {
                aabb: bb_t,
                poly: tmp_poly.clone(),
                edges: tmp_edges.clone(),
            });
            bin.grid.insert_aabb(&bb_t, placed_idx);

            // update shelf using the (possibly wrapped) shelf state we computed
            // (use the shelf position corresponding to the base candidate if it was shelf-aligned;
            // otherwise just advance by bbox width at least to avoid stalling)
            // We conservatively update from current bin shelf values:
            let mut nx = bin.shelf_x;
            let mut ny = bin.shelf_y;
            let mut nh = bin.shelf_h;
            if nx + w > stock_bb.width() + 1e-9 {
                nx = 0.0;
                ny += nh;
                nh = 0.0;
            }
            // if candidate appears to be placed at the shelf origin-ish, advance shelf; else still advance a bit
            // This keeps the packer progressing even when scan/random picks succeed.
            nx = (nx + w).min(stock_bb.width());
            nh = nh.max(h);
            bin.shelf_x = nx;
            bin.shelf_y = ny;
            bin.shelf_h = nh;

            return Some(Placement {
                stock: stock_index,
                tx,
                ty,
                rot: var.angle,
            });
        }
    }

    None
}

fn make_stock_template(stock_poly: Vec<Pt>, cell: f64) -> StockTemplate {
    let aabb = Aabb::from_pts(&stock_poly);
    let stock_edges = edges_of(&stock_poly);

    // Build edge grid
    let mut grid = SpatialHash::new(cell);
    for (i, e) in stock_edges.iter().enumerate() {
        grid.insert_aabb(&e.bb, i);
    }

    StockTemplate {
        poly: stock_poly,
        aabb,
        stock_edges,
        stock_edge_grid: grid,
        cell,
    }
}

fn choose_mode(np: usize, total_vertices: usize) -> Mode {
    if np <= 120 && total_vertices <= 25_000 {
        Mode::Quality
    } else if np <= 1200 && total_vertices <= 250_000 {
        Mode::Balanced
    } else {
        Mode::Fast
    }
}

// ---------------- Input parsing ----------------

struct Scanner {
    b: Vec<u8>,
    i: usize,
}
impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self { b: input.into_bytes(), i: 0 }
    }
    fn skip_ws(&mut self) {
        while self.i < self.b.len() && self.b[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
    }
    fn next_usize(&mut self) -> usize {
        self.skip_ws();
        let mut s = 0usize;
        while self.i < self.b.len() && !self.b[self.i].is_ascii_whitespace() {
            s = s * 10 + (self.b[self.i] - b'0') as usize;
            self.i += 1;
        }
        s
    }
    fn next_f64(&mut self) -> f64 {
        self.skip_ws();
        let start = self.i;
        while self.i < self.b.len() && !self.b[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
        // SAFETY: slice is valid UTF-8 for numeric tokens in the input
        let s = unsafe { std::str::from_utf8_unchecked(&self.b[start..self.i]) };
        s.parse::<f64>().unwrap()
    }
}

fn main() {
    let mut sc = Scanner::new();

    let sv = sc.next_usize();
    let np = sc.next_usize();

    let mut stock_poly = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x = sc.next_f64();
        let y = sc.next_f64();
        stock_poly.push(Pt { x, y });
    }

    let mut raw_pieces: Vec<Vec<Pt>> = Vec::with_capacity(np);
    let mut total_vertices = 0usize;
    for _ in 0..np {
        let pv = sc.next_usize();
        total_vertices += pv;
        let mut poly = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x = sc.next_f64();
            let y = sc.next_f64();
            poly.push(Pt { x, y });
        }
        raw_pieces.push(poly);
    }

    let mode = choose_mode(np, total_vertices);

    // Choose a global grid cell size from stock bbox
    let stock_bb = Aabb::from_pts(&stock_poly);
    let diag_scale = (stock_bb.width() * stock_bb.height()).abs().sqrt().max(1.0);
    let cell = match mode {
        Mode::Fast => (diag_scale / 40.0).max(1e-3),
        Mode::Balanced => (diag_scale / 55.0).max(1e-3),
        Mode::Quality => (diag_scale / 70.0).max(1e-3),
    };

    let tpl = make_stock_template(stock_poly, cell);

    // Build pieces with variants
    let mut pieces: Vec<Piece> = Vec::with_capacity(np);
    for (idx, poly) in raw_pieces.iter().enumerate() {
        let area = shoelace_area(poly);
        let angs = angles_for_polygon(poly, mode);
        let variants = build_variants(poly, &angs);
        pieces.push(Piece {
            orig_idx: idx,
            area,
            variants,
        });
    }

    // Sort pieces by descending area (FFD)
    let mut order: Vec<usize> = (0..np).collect();
    order.sort_by(|&i, &j| pieces[j].area.partial_cmp(&pieces[i].area).unwrap());

    let mut placements: Vec<Option<Placement>> = vec![None; np];
    let mut bins: Vec<StockBin> = Vec::new();

    let mut rng = Rng64::new(0x1234_5678_9abc_def0);

    for &pi in &order {
        // Try existing bins first
        let mut placed = None;
        for bi in 0..bins.len() {
            if let Some(pl) = try_place_in_bin(&pieces[pi], &mut bins[bi], &tpl, mode, bi, &mut rng) {
                placed = Some(pl);
                break;
            }
        }

        // If not placed, open a new stock and place there
        if placed.is_none() {
            let mut new_bin = StockBin {
                placed: Vec::new(),
                grid: SpatialHash::new(tpl.cell),
                shelf_x: 0.0,
                shelf_y: 0.0,
                shelf_h: 0.0,
            };
            let new_index = bins.len();
            placed = try_place_in_bin(&pieces[pi], &mut new_bin, &tpl, mode, new_index, &mut rng);
            bins.push(new_bin);
        }

        let pl = placed.unwrap_or_else(|| {
            // Last-resort fallback: output something deterministic (may be invalid if instance is infeasible).
            Placement {
                stock: 0,
                tx: 0.0,
                ty: 0.0,
                rot: 0.0,
            }
        });

        placements[pieces[pi].orig_idx] = Some(pl);
    }

    // Output
    println!("{}", bins.len());
    for i in 0..np {
        let p = placements[i].unwrap();
        println!("{} {:.10} {:.10} {:.10}", p.stock, p.tx, p.ty, p.rot);
    }
}
