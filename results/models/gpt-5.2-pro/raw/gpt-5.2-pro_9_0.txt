using System;
using System.IO;
using System.Runtime.CompilerServices;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        int n = fs.NextInt();
        int stockLen = fs.NextInt();

        if (n <= 0)
        {
            Console.Write("0\n");
            return;
        }

        // Pack (length desc, index asc) into a single sortable key.
        // We sort ascending by (int.MaxValue - len) -> effectively len descending.
        ulong[] items = new ulong[n];
        for (int i = 0; i < n; i++)
        {
            int len = fs.NextInt();
            // If input is invalid (len > stockLen), the problem is infeasible. We assume valid as per statement.
            int high = int.MaxValue - len;
            items[i] = ((ulong)(uint)high << 32) | (uint)i;
        }

        Array.Sort(items);

        // Cut assignment list per bin using linked lists over cut indices (memory-lean):
        // binHead[binId] is a cut index, and nextInBin[cutIndex] links to next cut in that bin.
        int[] binHead = new int[n];
        int[] nextInBin = new int[n];
        Array.Fill(binHead, -1);
        Array.Fill(nextInBin, -1);

        // Treap stores remaining capacity -> stack of bins having that remaining.
        // Use +1 encoding for bin links to allow 0 as "null" without clashing with binId=0.
        int[] binNext = new int[n]; // stores next binId+1, 0 means null

        var treap = new RemainingTreap(capacity: Math.Min(n, 1 << 20), binNext: binNext);

        int binCount = 0;

        for (int i = 0; i < n; i++)
        {
            ulong packed = items[i];
            int high = (int)(packed >> 32);
            int len = int.MaxValue - high;
            int cutIdx = (int)(packed & 0xFFFFFFFFu);

            int binId, rem;
            if (!treap.TryPopLowerBound(len, out binId, out rem))
            {
                binId = binCount++;
                rem = stockLen;
            }

            // Assign this cut to bin
            nextInBin[cutIdx] = binHead[binId];
            binHead[binId] = cutIdx;

            int newRem = rem - len;
            if (newRem > 0)
                treap.Insert(newRem, binId);
        }

        var w = new FastWriter(Console.OpenStandardOutput(), 1 << 20);
        w.WriteInt(binCount);
        w.WriteByte((byte)'\n');

        for (int b = 0; b < binCount; b++)
        {
            int c = binHead[b];
            if (c < 0)
            {
                w.WriteByte((byte)'\n');
                continue;
            }

            // Print space-separated cut indices (order is arbitrary; this is LIFO within each bin)
            w.WriteInt(c);
            c = nextInBin[c];
            while (c >= 0)
            {
                w.WriteByte((byte)' ');
                w.WriteInt(c);
                c = nextInBin[c];
            }
            w.WriteByte((byte)'\n');
        }

        w.Flush();
    }

    // ---------------- Fast Scanner ----------------
    private sealed class FastScanner
    {
        private readonly Stream _s;
        private readonly byte[] _buf;
        private int _len, _ptr;

        public FastScanner(Stream s, int bufSize = 1 << 20)
        {
            _s = s;
            _buf = new byte[bufSize];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buf[_ptr++];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int NextInt()
        {
            byte c;
            do c = Read(); while (c <= (byte)' ');

            int sign = 1;
            if (c == (byte)'-')
            {
                sign = -1;
                c = Read();
            }

            int x = 0;
            while (c > (byte)' ')
            {
                x = x * 10 + (c - (byte)'0');
                c = Read();
            }
            return x * sign;
        }
    }

    // ---------------- Fast Writer ----------------
    private sealed class FastWriter
    {
        private readonly Stream _s;
        private readonly byte[] _buf;
        private int _pos;

        public FastWriter(Stream s, int bufSize)
        {
            _s = s;
            _buf = new byte[bufSize];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteByte(byte b)
        {
            if (_pos == _buf.Length) Flush();
            _buf[_pos++] = b;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteInt(int x)
        {
            if (x == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            if (x < 0)
            {
                WriteByte((byte)'-');
                x = -x;
            }

            // Max 10 digits for int
            Span<byte> tmp = stackalloc byte[10];
            int n = 0;
            while (x > 0)
            {
                tmp[n++] = (byte)('0' + (x % 10));
                x /= 10;
            }
            for (int i = n - 1; i >= 0; i--) WriteByte(tmp[i]);
        }

        public void Flush()
        {
            if (_pos > 0)
            {
                _s.Write(_buf, 0, _pos);
                _pos = 0;
            }
        }
    }

    // ---------------- Treap: remaining capacity -> stack of binIds ----------------
    private sealed class RemainingTreap
    {
        // Node arrays, 1-based indices; 0 = null
        private int[] _key;
        private int[] _prio;
        private int[] _left;
        private int[] _right;
        private int[] _headBinPlus1; // stack head: binId+1, 0 = empty

        private int _root;
        private int _nodes;

        private ulong _rng;

        private readonly int[] _binNextPlus1; // binId -> next binId+1, 0 = null

        public RemainingTreap(int capacity, int[] binNext)
        {
            capacity = Math.Max(16, capacity);
            _key = new int[capacity + 1];
            _prio = new int[capacity + 1];
            _left = new int[capacity + 1];
            _right = new int[capacity + 1];
            _headBinPlus1 = new int[capacity + 1];
            _root = 0;
            _nodes = 0;
            _rng = (ulong)Environment.TickCount64 * 11400714819323198485UL + 12345UL;
            _binNextPlus1 = binNext;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int NextPrio()
        {
            // xorshift64*
            _rng ^= _rng >> 12;
            _rng ^= _rng << 25;
            _rng ^= _rng >> 27;
            ulong z = _rng * 2685821657736338717UL;
            return (int)(z & 0x7FFFFFFF);
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private void EnsureNodeCapacity()
        {
            if (_nodes + 1 < _key.Length) return;
            int newCap = _key.Length * 2;

            Array.Resize(ref _key, newCap);
            Array.Resize(ref _prio, newCap);
            Array.Resize(ref _left, newCap);
            Array.Resize(ref _right, newCap);
            Array.Resize(ref _headBinPlus1, newCap);
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int NewNode(int key, int binId)
        {
            EnsureNodeCapacity();
            int id = ++_nodes;
            _key[id] = key;
            _prio[id] = NextPrio();
            _left[id] = 0;
            _right[id] = 0;
            _headBinPlus1[id] = binId + 1;
            return id;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private void PushBin(int node, int binId)
        {
            int head = _headBinPlus1[node]; // binId+1
            _binNextPlus1[binId] = head;
            _headBinPlus1[node] = binId + 1;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private static int RotateRight(int p, int[] left, int[] right)
        {
            int q = left[p];
            left[p] = right[q];
            right[q] = p;
            return q;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private static int RotateLeft(int p, int[] left, int[] right)
        {
            int q = right[p];
            right[p] = left[q];
            left[q] = p;
            return q;
        }

        public void Insert(int rem, int binId)
        {
            _root = Insert(_root, rem, binId);
        }

        private int Insert(int node, int key, int binId)
        {
            if (node == 0)
                return NewNode(key, binId);

            int nk = _key[node];
            if (key == nk)
            {
                PushBin(node, binId);
                return node;
            }

            if (key < nk)
            {
                int l = Insert(_left[node], key, binId);
                _left[node] = l;
                if (_prio[l] < _prio[node])
                    node = RotateRight(node, _left, _right);
            }
            else
            {
                int r = Insert(_right[node], key, binId);
                _right[node] = r;
                if (_prio[r] < _prio[node])
                    node = RotateLeft(node, _left, _right);
            }
            return node;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int Merge(int a, int b)
        {
            if (a == 0) return b;
            if (b == 0) return a;

            if (_prio[a] < _prio[b])
            {
                _right[a] = Merge(_right[a], b);
                return a;
            }
            else
            {
                _left[b] = Merge(a, _left[b]);
                return b;
            }
        }

        private int Delete(int node, int key)
        {
            if (node == 0) return 0;

            int nk = _key[node];
            if (key == nk)
                return Merge(_left[node], _right[node]);

            if (key < nk) _left[node] = Delete(_left[node], key);
            else _right[node] = Delete(_right[node], key);
            return node;
        }

        // Pop a bin whose remaining capacity is the smallest >= need (best-fit).
        public bool TryPopLowerBound(int need, out int binId, out int rem)
        {
            int cur = _root;
            int cand = 0;

            while (cur != 0)
            {
                if (_key[cur] >= need)
                {
                    cand = cur;
                    cur = _left[cur];
                }
                else
                {
                    cur = _right[cur];
                }
            }

            if (cand == 0)
            {
                binId = 0;
                rem = 0;
                return false;
            }

            rem = _key[cand];

            int headPlus1 = _headBinPlus1[cand]; // binId+1
            binId = headPlus1 - 1;
            _headBinPlus1[cand] = _binNextPlus1[binId];
            _binNextPlus1[binId] = 0;

            if (_headBinPlus1[cand] == 0)
                _root = Delete(_root, rem);

            return true;
        }
    }
}
