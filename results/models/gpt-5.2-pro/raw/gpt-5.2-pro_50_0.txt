```glsl
#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

float safeInv(float x) { return 1.0 / max(x, 1e-8); }

void main()
{
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) return;

    uint i2 = i * 2u;

    vec3 pos = inData[i2 + 0u].xyz;
    vec3 vel = inData[i2 + 1u].xyz;

    float dt       = radii.x;
    float sepR     = radii.y;
    float alignR   = radii.z;
    float cohR     = radii.w;

    float sepW     = weights.x;
    float alignW   = weights.y;
    float cohW     = weights.z;
    float maxSpeed = weights.w;

    float boundSize  = bounds.x;
    float boundForce = bounds.y;

    float sepR2   = sepR   * sepR;
    float alignR2 = alignR * alignR;
    float cohR2   = cohR   * cohR;

    vec3 sepAcc = vec3(0.0);
    vec3 aliAcc = vec3(0.0);
    vec3 cohAcc = vec3(0.0);

    uint sepCount = 0u;
    uint aliCount = 0u;
    uint cohCount = 0u;

    // Naive O(N^2) neighborhood scan
    for (uint j = 0u; j < N; ++j)
    {
        if (j == i) continue;

        uint j2 = j * 2u;
        vec3 op = inData[j2 + 0u].xyz;
        vec3 ov = inData[j2 + 1u].xyz;

        vec3 d  = pos - op;
        float d2 = dot(d, d);
        if (d2 <= 1e-12) continue; // skip zero-distance neighbors

        if (d2 < sepR2)
        {
            // Separation: push away more strongly when closer.
            // Use 1/(distance^2) style weighting to avoid jitter and emphasize close neighbors.
            float invd2 = safeInv(d2);
            sepAcc += d * invd2;
            sepCount++;
        }

        if (d2 < alignR2)
        {
            // Alignment: match neighbor velocities
            aliAcc += ov;
            aliCount++;
        }

        if (d2 < cohR2)
        {
            // Cohesion: move toward neighbor center of mass
            cohAcc += op;
            cohCount++;
        }
    }

    // Convert accumulations into steering accelerations (average, then desired-change)
    if (sepCount > 0u)
    {
        sepAcc *= (1.0 / float(sepCount));
        // sepAcc already points away; keep as-is
    }
    else
    {
        sepAcc = vec3(0.0);
    }

    if (aliCount > 0u)
    {
        vec3 avgVel = aliAcc * (1.0 / float(aliCount));
        aliAcc = (avgVel - vel);
    }
    else
    {
        aliAcc = vec3(0.0);
    }

    if (cohCount > 0u)
    {
        vec3 avgPos = cohAcc * (1.0 / float(cohCount));
        cohAcc = (avgPos - pos);
    }
    else
    {
        cohAcc = vec3(0.0);
    }

    // Boundary force: if beyond boundSize*0.4, push toward center
    float limit = boundSize * 0.4;
    float half  = boundSize * 0.5;

    vec3 boundAcc = vec3(0.0);

    if (pos.x >  limit) boundAcc.x -= boundForce * ((pos.x -  limit) * safeInv(half));
    if (pos.x < -limit) boundAcc.x += boundForce * ((-limit - pos.x) * safeInv(half));

    if (pos.y >  limit) boundAcc.y -= boundForce * ((pos.y -  limit) * safeInv(half));
    if (pos.y < -limit) boundAcc.y += boundForce * ((-limit - pos.y) * safeInv(half));

    if (pos.z >  limit) boundAcc.z -= boundForce * ((pos.z -  limit) * safeInv(half));
    if (pos.z < -limit) boundAcc.z += boundForce * ((-limit - pos.z) * safeInv(half));

    // Total acceleration
    vec3 accel =
        sepAcc * sepW +
        aliAcc * alignW +
        cohAcc * cohW +
        boundAcc;

    // Integrate velocity and clamp to maxSpeed
    vel += accel * dt;

    float spd2 = dot(vel, vel);
    float maxSpd2 = maxSpeed * maxSpeed;
    if (spd2 > maxSpd2)
    {
        vel *= (maxSpeed * inversesqrt(spd2));
    }

    // Integrate position
    pos += vel * dt;

    // Write out
    outData[i2 + 0u] = vec4(pos, 0.0);
    outData[i2 + 1u] = vec4(vel, 0.0);
}
```
