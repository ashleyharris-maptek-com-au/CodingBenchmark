```rust
use std::cmp::Ordering;
use std::io::{self, Read};

#[derive(Clone)]
struct Scanner {
    input: Vec<u8>,
    idx: usize,
}
impl Scanner {
    fn new() -> Self {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).unwrap();
        Self { input, idx: 0 }
    }
    fn next_usize(&mut self) -> usize {
        while self.idx < self.input.len() && self.input[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        let mut x = 0usize;
        while self.idx < self.input.len() && !self.input[self.idx].is_ascii_whitespace() {
            x = x * 10 + (self.input[self.idx] - b'0') as usize;
            self.idx += 1;
        }
        x
    }
}

fn assign_vertex(
    v: usize,
    side_a: bool, // true => A, false => B
    assignment: &mut [i8], // -1 undecided, 1 A, 0 B
    to_a: &mut [i32],
    to_b: &mut [i32],
    adj: &[Vec<usize>],
    current_cut: &mut i64,
) -> (i64, Vec<usize>) {
    let mut delta_cut: i64 = 0;
    let mut updated_undecided: Vec<usize> = Vec::new();

    let my_side = if side_a { 1i8 } else { 0i8 };
    assignment[v] = my_side;

    for &u in &adj[v] {
        match assignment[u] {
            -1 => {
                if side_a {
                    to_a[u] += 1;
                } else {
                    to_b[u] += 1;
                }
                // record once per adjacency occurrence (handles multi-edges correctly)
                updated_undecided.push(u);
            }
            0 => {
                if side_a {
                    delta_cut += 1;
                }
            }
            1 => {
                if !side_a {
                    delta_cut += 1;
                }
            }
            _ => unreachable!(),
        }
    }

    *current_cut += delta_cut;
    (delta_cut, updated_undecided)
}

fn unassign_vertex(
    v: usize,
    side_a: bool,
    assignment: &mut [i8],
    to_a: &mut [i32],
    to_b: &mut [i32],
    current_cut: &mut i64,
    delta_cut: i64,
    updated_undecided: Vec<usize>,
) {
    for u in updated_undecided {
        if side_a {
            to_a[u] -= 1;
        } else {
            to_b[u] -= 1;
        }
    }
    assignment[v] = -1;
    *current_cut -= delta_cut;
}

fn lower_bound(
    idx: usize,
    order: &[usize],
    k: usize,
    a_cnt: usize,
    b_cnt: usize,
    to_a: &[i32],
    to_b: &[i32],
    current_cut: i64,
) -> i64 {
    let rem = &order[idx..];
    let rem_len = rem.len();

    let ra = k as i32 - a_cnt as i32; // remaining slots in A
    let rb = k as i32 - b_cnt as i32; // remaining slots in B
    if ra < 0 || rb < 0 || (ra as usize) > rem_len || (rb as usize) > rem_len {
        return i64::MAX / 4; // infeasible
    }
    if rem_len == 0 {
        return current_cut;
    }

    // If all remaining forced to one side, we can compute exact additional cut from decided<->undecided edges.
    if ra == 0 {
        let mut sum = 0i64;
        for &v in rem {
            sum += to_a[v] as i64; // v in B cuts edges to decided A
        }
        return current_cut + sum;
    }
    if rb == 0 {
        let mut sum = 0i64;
        for &v in rem {
            sum += to_b[v] as i64; // v in A cuts edges to decided B
        }
        return current_cut + sum;
    }

    // General bound: minimize decided<->undecided cut subject to choosing exactly ra vertices into A,
    // ignoring undecided<->undecided edges (nonnegative, so safe lower bound).
    let ra = ra as usize;

    let mut base_sum_to_a = 0i64;
    let mut deltas: Vec<i32> = Vec::with_capacity(rem_len);
    for &v in rem {
        base_sum_to_a += to_a[v] as i64; // cost if v goes to B
        deltas.push(to_b[v] - to_a[v]); // extra cost if v goes to A instead
    }

    // Add smallest ra deltas.
    deltas.select_nth_unstable(ra - 1);
    let mut add = 0i64;
    for i in 0..ra {
        add += deltas[i] as i64;
    }

    current_cut + base_sum_to_a + add
}

fn main() {
    let mut sc = Scanner::new();
    let n = sc.next_usize();
    let m = sc.next_usize();
    let k = n / 2;

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);

    for _ in 0..m {
        let u = sc.next_usize();
        let v = sc.next_usize();
        if u == v {
            // self-loop never contributes to cut; ignore
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
        edges.push((u, v));
    }

    // Vertex order heuristic (does not affect correctness).
    let mut order: Vec<usize> = (0..n).collect();
    order.sort_unstable_by(|&a, &b| {
        let da = adj[a].len();
        let db = adj[b].len();
        db.cmp(&da).then_with(|| a.cmp(&b))
    });

    // Initial upper bound: put first k vertices (by order) in A.
    let mut in_a0 = vec![false; n];
    for i in 0..k {
        in_a0[order[i]] = true;
    }
    let mut best_cut: i64 = 0;
    for &(u, v) in &edges {
        if in_a0[u] ^ in_a0[v] {
            best_cut += 1;
        }
    }
    let mut best_assignment: Vec<i8> = vec![-1; n];
    for v in 0..n {
        best_assignment[v] = if in_a0[v] { 1 } else { 0 };
    }

    // Branch and bound search.
    let mut assignment: Vec<i8> = vec![-1; n];
    let mut to_a: Vec<i32> = vec![0; n];
    let mut to_b: Vec<i32> = vec![0; n];
    let mut current_cut: i64 = 0;

    // Symmetry breaking: force order[0] into A.
    let v0 = order[0];
    let (_dc0, _upd0) = assign_vertex(
        v0, true, &mut assignment, &mut to_a, &mut to_b, &adj, &mut current_cut,
    );
    let mut a_cnt: usize = 1;
    let mut b_cnt: usize = 0;

    fn dfs(
        idx: usize,
        n: usize,
        k: usize,
        order: &[usize],
        adj: &[Vec<usize>],
        assignment: &mut [i8],
        to_a: &mut [i32],
        to_b: &mut [i32],
        current_cut: &mut i64,
        a_cnt: &mut usize,
        b_cnt: &mut usize,
        best_cut: &mut i64,
        best_assignment: &mut Vec<i8>,
    ) {
        if idx == n {
            if *a_cnt == k && *b_cnt == k && *current_cut < *best_cut {
                *best_cut = *current_cut;
                *best_assignment = assignment.to_vec();
            }
            return;
        }

        let rem_len = n - idx;
        let ra = k as i32 - *a_cnt as i32;
        let rb = k as i32 - *b_cnt as i32;
        if ra < 0 || rb < 0 || (ra as usize) > rem_len || (rb as usize) > rem_len {
            return;
        }

        // Forced completion shortcut.
        if ra == 0 || rb == 0 {
            let force_a = rb == 0;
            let mut final_cut = *current_cut;
            for &v in &order[idx..] {
                if force_a {
                    final_cut += to_b[v] as i64;
                } else {
                    final_cut += to_a[v] as i64;
                }
            }
            if final_cut < *best_cut {
                *best_cut = final_cut;
                let mut cand = assignment.to_vec();
                for &v in &order[idx..] {
                    cand[v] = if force_a { 1 } else { 0 };
                }
                *best_assignment = cand;
            }
            return;
        }

        let lb = lower_bound(idx, order, k, *a_cnt, *b_cnt, to_a, to_b, *current_cut);
        if lb >= *best_cut {
            return;
        }

        let v = order[idx];

        // Branch ordering: try the side with smaller immediate additional cut to already-decided vertices.
        // If v in A adds edges to decided B => to_b[v]; if v in B adds edges to decided A => to_a[v].
        let cost_a = to_b[v];
        let cost_b = to_a[v];

        let mut branches = [(true, cost_a), (false, cost_b)];
        branches.sort_unstable_by(|a, b| a.1.cmp(&b.1));

        for (side_a, _) in branches {
            if side_a && *a_cnt == k {
                continue;
            }
            if !side_a && *b_cnt == k {
                continue;
            }

            let (dc, upd) =
                assign_vertex(v, side_a, assignment, to_a, to_b, adj, current_cut);
            if side_a {
                *a_cnt += 1;
            } else {
                *b_cnt += 1;
            }

            dfs(
                idx + 1,
                n,
                k,
                order,
                adj,
                assignment,
                to_a,
                to_b,
                current_cut,
                a_cnt,
                b_cnt,
                best_cut,
                best_assignment,
            );

            if side_a {
                *a_cnt -= 1;
            } else {
                *b_cnt -= 1;
            }
            unassign_vertex(v, side_a, assignment, to_a, to_b, current_cut, dc, upd);
        }
    }

    dfs(
        1,
        n,
        k,
        &order,
        &adj,
        &mut assignment,
        &mut to_a,
        &mut to_b,
        &mut current_cut,
        &mut a_cnt,
        &mut b_cnt,
        &mut best_cut,
        &mut best_assignment,
    );

    // Build partition A and recompute cut to guarantee consistency.
    let mut in_a = vec![false; n];
    let mut part_a: Vec<usize> = Vec::with_capacity(k);
    for v in 0..n {
        if best_assignment[v] == 1 {
            in_a[v] = true;
            part_a.push(v);
        }
    }

    // If something went wrong, fall back (shouldn't happen).
    if part_a.len() != k {
        // Make a valid partition deterministically.
        part_a.clear();
        for v in 0..n {
            if part_a.len() < k {
                part_a.push(v);
                in_a[v] = true;
            } else {
                in_a[v] = false;
            }
        }
    }

    let mut cut2: i64 = 0;
    for &(u, v) in &edges {
        if in_a[u] ^ in_a[v] {
            cut2 += 1;
        }
    }

    let mut out = String::new();
    out.push_str(&format!("{}\n", cut2));
    for (i, v) in part_a.iter().enumerate() {
        if i > 0 {
            out.push(' ');
        }
        out.push_str(&v.to_string());
    }
    out.push('\n');
    print!("{out}");
}
```
