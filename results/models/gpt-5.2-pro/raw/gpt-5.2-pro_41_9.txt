; SPIR-V
; Version: 1.0
; Bound: 200
; Schema: 0
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %ubo %outColor
OpExecutionMode %main OriginUpperLeft

OpName %main "main"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"
OpName %inUV "uv"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpName %ubo "ubo"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%v2int = OpTypeVector %int 2
%mat4 = OpTypeMatrix %v4float 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptrUniformUBO = OpTypePointer Uniform %UBO
%ptrUniformv4 = OpTypePointer Uniform %v4float

%ptrInputv3 = OpTypePointer Input %v3float
%ptrInputv2 = OpTypePointer Input %v2float
%ptrOutputv4 = OpTypePointer Output %v4float

%fun = OpTypeFunction %void

%inWorldPos = OpVariable %ptrInputv3 Input
%inNormal = OpVariable %ptrInputv3 Input
%inUV = OpVariable %ptrInputv2 Input
%ubo = OpVariable %ptrUniformUBO Uniform
%outColor = OpVariable %ptrOutputv4 Output

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_3 = OpConstant %int 3

%f0 = OpConstant %float 0.0
%f02 = OpConstant %float 0.2
%f05 = OpConstant %float 0.5
%f09 = OpConstant %float 0.9
%f1 = OpConstant %float 1.0
%f8 = OpConstant %float 8.0

%v2f8 = OpConstantComposite %v2float %f8 %f8

%main = OpFunction %void None %fun
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %v3float %inWorldPos
%normal = OpLoad %v3float %inNormal
%uv = OpLoad %v2float %inUV

; Checkerboard base value: 8x8, alternating 0.9 / 0.2
%uvScaled = OpFMul %v2float %uv %v2f8
%uvFloor = OpExtInst %v2float %glsl Floor %uvScaled
%uvInt = OpConvertFToS %v2int %uvFloor
%ix = OpCompositeExtract %int %uvInt 0
%iy = OpCompositeExtract %int %uvInt 1
%sum = OpIAdd %int %ix %iy
%par = OpBitwiseAnd %int %sum %int_1
%isEven = OpIEqual %bool %par %int_0
%base = OpSelect %float %isEven %f09 %f02

; Light direction and Lambert term
%lightPosPtr = OpAccessChain %ptrUniformv4 %ubo %int_3
%lightPos4 = OpLoad %v4float %lightPosPtr
%lightPos3 = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2
%Lraw = OpFSub %v3float %lightPos3 %worldPos
%L = OpExtInst %v3float %glsl Normalize %Lraw
%N = OpExtInst %v3float %glsl Normalize %normal
%dotNL = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl FMax %dotNL %f0

; Shading factor: 0.5 + 0.5*diffuse
%halfDiff = OpFMul %float %f05 %diffuse
%shade = OpFAdd %float %f05 %halfDiff

; Final color
%val = OpFMul %float %base %shade
%rgb = OpCompositeConstruct %v3float %val %val %val
%out = OpCompositeConstruct %v4float %val %val %val %f1
OpStore %outColor %out

OpReturn
OpFunctionEnd
