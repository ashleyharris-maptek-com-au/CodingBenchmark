use std::cmp::Ordering;
use std::f64::consts::PI;
use std::io::{self, Read, Write};

#[derive(Clone, Copy, Debug, Default)]
struct Vec2 {
    x: f64,
    y: f64,
}
impl Vec2 {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }
    fn add(self, o: Vec2) -> Self {
        Self::new(self.x + o.x, self.y + o.y)
    }
    fn sub(self, o: Vec2) -> Self {
        Self::new(self.x - o.x, self.y - o.y)
    }
    fn mul(self, k: f64) -> Self {
        Self::new(self.x * k, self.y * k)
    }
    fn dot(self, o: Vec2) -> f64 {
        self.x * o.x + self.y * o.y
    }
    fn cross(self, o: Vec2) -> f64 {
        self.x * o.y - self.y * o.x
    }
    fn len(self) -> f64 {
        self.dot(self).sqrt()
    }
}

#[derive(Clone, Copy, Debug, Default)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}
impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Self { x, y, z }
    }
    fn add(self, o: Vec3) -> Self {
        Self::new(self.x + o.x, self.y + o.y, self.z + o.z)
    }
    fn sub(self, o: Vec3) -> Self {
        Self::new(self.x - o.x, self.y - o.y, self.z - o.z)
    }
    fn mul(self, k: f64) -> Self {
        Self::new(self.x * k, self.y * k, self.z * k)
    }
    fn dot(self, o: Vec3) -> f64 {
        self.x * o.x + self.y * o.y + self.z * o.z
    }
    fn len(self) -> f64 {
        self.dot(self).sqrt()
    }
    fn norm(self) -> Vec3 {
        let l = self.len();
        if l > 0.0 {
            self.mul(1.0 / l)
        } else {
            self
        }
    }
}

#[derive(Clone, Copy, Debug)]
struct Quat {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}
impl Quat {
    fn identity() -> Self {
        Self {
            w: 1.0,
            x: 0.0,
            y: 0.0,
            z: 0.0,
        }
    }
    fn from_axis_angle(axis: Vec3, angle_rad: f64) -> Self {
        let a = axis.norm();
        let h = 0.5 * angle_rad;
        let s = h.sin();
        Self {
            w: h.cos(),
            x: a.x * s,
            y: a.y * s,
            z: a.z * s,
        }
    }
    fn mul(self, o: Quat) -> Quat {
        Quat {
            w: self.w * o.w - self.x * o.x - self.y * o.y - self.z * o.z,
            x: self.w * o.x + self.x * o.w + self.y * o.z - self.z * o.y,
            y: self.w * o.y - self.x * o.z + self.y * o.w + self.z * o.x,
            z: self.w * o.z + self.x * o.y - self.y * o.x + self.z * o.w,
        }
    }
    fn conj(self) -> Quat {
        Quat {
            w: self.w,
            x: -self.x,
            y: -self.y,
            z: -self.z,
        }
    }
    fn rotate_vec3(self, v: Vec3) -> Vec3 {
        // q * (0,v) * q^-1
        let qv = Quat {
            w: 0.0,
            x: v.x,
            y: v.y,
            z: v.z,
        };
        let qi = self.conj(); // unit quats only here
        let r = self.mul(qv).mul(qi);
        Vec3::new(r.x, r.y, r.z)
    }
}

struct Scanner {
    b: Vec<u8>,
    i: usize,
}
impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self {
            b: input.into_bytes(),
            i: 0,
        }
    }
    fn skip_ws(&mut self) {
        while self.i < self.b.len() && self.b[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
    }
    fn next_i64(&mut self) -> i64 {
        self.skip_ws();
        let mut sgn = 1i64;
        if self.i < self.b.len() && self.b[self.i] == b'-' {
            sgn = -1;
            self.i += 1;
        }
        let mut v = 0i64;
        while self.i < self.b.len() {
            let c = self.b[self.i];
            if !c.is_ascii_digit() {
                break;
            }
            v = v * 10 + (c - b'0') as i64;
            self.i += 1;
        }
        sgn * v
    }
    fn next_f64(&mut self) -> f64 {
        self.skip_ws();
        let start = self.i;
        while self.i < self.b.len() && !self.b[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
        let s = std::str::from_utf8(&self.b[start..self.i]).unwrap();
        s.parse::<f64>().unwrap_or(0.0)
    }
}

fn convex_hull(mut pts: Vec<Vec2>) -> Vec<Vec2> {
    // Monotonic chain. Works with <=4 points too.
    pts.sort_by(|a, b| {
        match a.x.partial_cmp(&b.x).unwrap_or(Ordering::Equal) {
            Ordering::Equal => a.y.partial_cmp(&b.y).unwrap_or(Ordering::Equal),
            o => o,
        }
    });
    pts.dedup_by(|a, b| (a.x == b.x) && (a.y == b.y));
    if pts.len() <= 2 {
        return pts;
    }
    let mut lower: Vec<Vec2> = Vec::with_capacity(pts.len());
    for &p in &pts {
        while lower.len() >= 2 {
            let a = lower[lower.len() - 2];
            let b = lower[lower.len() - 1];
            if (b.sub(a)).cross(p.sub(a)) <= 0.0 {
                lower.pop();
            } else {
                break;
            }
        }
        lower.push(p);
    }
    let mut upper: Vec<Vec2> = Vec::with_capacity(pts.len());
    for &p in pts.iter().rev() {
        while upper.len() >= 2 {
            let a = upper[upper.len() - 2];
            let b = upper[upper.len() - 1];
            if (b.sub(a)).cross(p.sub(a)) <= 0.0 {
                upper.pop();
            } else {
                break;
            }
        }
        upper.push(p);
    }
    lower.pop();
    upper.pop();
    lower.extend(upper);
    lower
}

fn project_to_plane(p: Vec3, sun: Vec3) -> Vec2 {
    // Projection along sun direction onto z=0:
    // x' = x - z*sx/sz, y' = y - z*sy/sz
    Vec2::new(p.x - p.z * sun.x / sun.z, p.y - p.z * sun.y / sun.z)
}

fn inradius_about_center_ccw(hull: &[Vec2], center: Vec2) -> f64 {
    if hull.len() < 3 {
        return 0.0;
    }
    let mut r = f64::INFINITY;
    for i in 0..hull.len() {
        let a = hull[i];
        let b = hull[(i + 1) % hull.len()];
        let e = b.sub(a);
        let el = e.len();
        if el <= 0.0 {
            continue;
        }
        let dist = (e.cross(center.sub(a)) / el).abs();
        if dist < r {
            r = dist;
        }
    }
    if r.is_finite() { r } else { 0.0 }
}

fn main() {
    let mut sc = Scanner::new();
    let nv = sc.next_i64() as usize;
    if nv < 3 {
        let mut out = io::BufWriter::new(io::stdout().lock());
        writeln!(out, "0").unwrap();
        return;
    }

    let mut minx = f64::INFINITY;
    let mut maxx = -f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxy = -f64::INFINITY;

    let mut poly: Vec<Vec2> = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x = sc.next_f64();
        let y = sc.next_f64();
        poly.push(Vec2::new(x, y));
        minx = minx.min(x);
        maxx = maxx.max(x);
        miny = miny.min(y);
        maxy = maxy.max(y);
    }

    let mut sun = Vec3::new(sc.next_f64(), sc.next_f64(), sc.next_f64());
    if !sun.z.is_finite() || sun.z.abs() < 1e-9 {
        // Nearly parallel to plane: fall back to vertical sun to avoid numeric blowups.
        sun = Vec3::new(0.0, 0.0, -1.0);
    }
    // Keep exact components for projection formula (uses sun.z); magnitude irrelevant.

    // Reference tetra vertices.
    let ref_v = [
        Vec3::new(1.0, 0.0, -0.707),
        Vec3::new(-1.0, 0.0, -0.707),
        Vec3::new(0.0, 1.0, 0.707),
        Vec3::new(0.0, -1.0, 0.707),
    ];

    // Always use maximum scale to reduce count (constraint: <= 10).
    let scale = 10.0_f64;

    // Candidate orientations; pick one maximizing a *guaranteed inscribed disk radius*
    // (about the mean of hull vertices).
    let yaw45 = Quat::from_axis_angle(Vec3::new(0.0, 0.0, 1.0), PI / 4.0);
    let yaw90 = Quat::from_axis_angle(Vec3::new(0.0, 0.0, 1.0), PI / 2.0);
    let pitch35 = Quat::from_axis_angle(Vec3::new(0.0, 1.0, 0.0), 35.0_f64.to_radians());
    let roll35 = Quat::from_axis_angle(Vec3::new(1.0, 0.0, 0.0), 35.0_f64.to_radians());

    let candidates: [Quat; 10] = [
        Quat::identity(),
        yaw45,
        yaw90,
        pitch35,
        roll35,
        yaw45.mul(pitch35),
        yaw45.mul(roll35),
        pitch35.mul(roll35),
        yaw90.mul(pitch35),
        yaw90.mul(roll35),
    ];

    let mut best_q = Quat::identity();
    let mut best_c0 = Vec2::new(0.0, 0.0);
    let mut best_r = 0.0_f64;
    let mut best_min_vz = 0.0_f64;
    let mut best_max_vz = 0.0_f64;

    for &q in &candidates {
        let mut proj_pts: Vec<Vec2> = Vec::with_capacity(4);
        let mut min_vz = f64::INFINITY;
        let mut max_vz = -f64::INFINITY;

        for &v in &ref_v {
            let vr = q.rotate_vec3(v.mul(scale));
            min_vz = min_vz.min(vr.z);
            max_vz = max_vz.max(vr.z);
            proj_pts.push(project_to_plane(vr, sun));
        }
        let hull = convex_hull(proj_pts);
        if hull.len() < 3 {
            continue;
        }
        let mut c0 = Vec2::new(0.0, 0.0);
        for &p in &hull {
            c0 = c0.add(p);
        }
        c0 = c0.mul(1.0 / (hull.len() as f64));
        let r = inradius_about_center_ccw(&hull, c0);

        if r > best_r {
            best_r = r;
            best_q = q;
            best_c0 = c0;
            best_min_vz = min_vz;
            best_max_vz = max_vz;
        }
    }

    // Safety fallback if something degenerated.
    if !(best_r.is_finite()) || best_r <= 1e-9 {
        best_q = Quat::identity();
        best_r = 1.0;
        best_c0 = Vec2::new(0.0, 0.0);
        best_min_vz = -0.707 * scale;
        best_max_vz = 0.707 * scale;
    }

    // Adaptive strategy:
    // - If polygon fits into one guaranteed disk => 1 tetrahedron.
    // - Else cover polygon AABB by a triangular-lattice disk covering (fewer pieces).
    let mut poly_center = Vec2::new(0.0, 0.0);
    for &p in &poly {
        poly_center = poly_center.add(p);
    }
    poly_center = poly_center.mul(1.0 / (nv as f64));

    let mut max_d = 0.0_f64;
    for &p in &poly {
        let d = p.sub(poly_center).len();
        if d > max_d {
            max_d = d;
        }
    }

    let mut centers: Vec<Vec2> = Vec::new();

    // Slight shrink for robust "fits in one" decision.
    if max_d <= best_r * 0.95 {
        centers.push(poly_center);
    } else {
        let width = (maxx - minx).max(0.0);
        let height = (maxy - miny).max(0.0);

        // Triangular lattice covering:
        // Use lattice constant a = r*sqrt(3) => covering radius r.
        // Make it denser by factor < 1 for numeric robustness.
        let sqrt3 = 3.0_f64.sqrt();
        let density = 0.98_f64;
        let step_x = best_r * sqrt3 * density;
        let step_y = best_r * 1.5 * density;

        // Expand bounds so disks cover edges of the AABB.
        let pad = best_r;
        let minx_e = minx - pad;
        let maxx_e = maxx + pad;
        let miny_e = miny - pad;
        let maxy_e = maxy + pad;

        // If extremely thin in one dimension, do a 1D strip (adaptive).
        if width < best_r * 0.5 || height < best_r * 0.5 {
            if width >= height {
                // along x
                let mut x = minx_e;
                while x <= maxx_e + 1e-12 {
                    centers.push(Vec2::new(x, (miny + maxy) * 0.5));
                    x += step_x.max(1e-6);
                }
            } else {
                // along y
                let mut y = miny_e;
                while y <= maxy_e + 1e-12 {
                    centers.push(Vec2::new((minx + maxx) * 0.5, y));
                    y += step_y.max(1e-6);
                }
            }
        } else {
            let mut row = 0usize;
            let mut y = miny_e;
            while y <= maxy_e + 1e-12 {
                let x0 = minx_e + if (row & 1) == 1 { 0.5 * step_x } else { 0.0 };
                let mut x = x0;
                while x <= maxx_e + 1e-12 {
                    centers.push(Vec2::new(x, y));
                    x += step_x.max(1e-6);
                }
                row += 1;
                y += step_y.max(1e-6);
            }
        }
    }

    // Non-intersection guarantee:
    // Put each tetrahedron on its own z-slab (disjoint z-intervals).
    // This works for any sun direction; we compensate x,y so shadow location stays fixed.
    let span_z = (best_max_vz - best_min_vz).abs();
    let zsep = span_z + 1.0; // > span => disjoint z ranges

    // Ensure all vertices lie on the correct side of z=0 so rays intersect the plane:
    // Need (vertex_z / sun.z) <= 0 for all vertices => vertex_z has opposite sign to sun.z.
    let margin = 0.1;
    let (z0, z_dir) = if sun.z < 0.0 {
        // Rays go downward; tetrahedra must be at z >= 0
        (-best_min_vz + margin, 1.0)
    } else {
        // Rays go upward; tetrahedra must be at z <= 0
        (-best_max_vz - margin, -1.0)
    };

    let mut out = io::BufWriter::new(io::stdout().lock());
    writeln!(out, "{}", centers.len()).unwrap();

    for (i, &p) in centers.iter().enumerate() {
        let pz = z0 + z_dir * (i as f64) * zsep;

        // Place so the guaranteed disk center (best_c0) moves to desired 2D position p.
        // Translation effect on shadow is (tx - tz*sx/sz, ty - tz*sy/sz).
        let dx = p.x - best_c0.x;
        let dy = p.y - best_c0.y;
        let px = dx + pz * (sun.x / sun.z);
        let py = dy + pz * (sun.y / sun.z);

        writeln!(
            out,
            "{:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10}",
            px, py, pz, best_q.w, best_q.x, best_q.y, best_q.z, scale
        )
        .unwrap();
    }
}
