using System;
using System.Diagnostics;
using System.Globalization;

public static class Program
{
    // ---------------------- Math / Vector ----------------------
    const double TwoPi = Math.PI * 2.0;

    static double Clamp(double x, double lo, double hi) => x < lo ? lo : (x > hi ? hi : x);

    static double Wrap0To2Pi(double a)
    {
        a %= TwoPi;
        if (a < 0) a += TwoPi;
        return a;
    }

    static double AngleDiff01Pi(double a, double b)
    {
        double d = (a - b) % TwoPi;
        if (d < -Math.PI) d += TwoPi;
        else if (d > Math.PI) d -= TwoPi;
        return Math.Abs(d);
    }

    readonly struct Vec3
    {
        public readonly double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(double s, Vec3 v) => new Vec3(s * v.X, s * v.Y, s * v.Z);

        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public Vec3 Unit()
        {
            double n = Norm();
            if (n <= 0) return new Vec3(0, 0, 0);
            return (1.0 / n) * this;
        }

        public static double Dot(Vec3 a, Vec3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vec3 Cross(Vec3 a, Vec3 b) =>
            new Vec3(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
    }

    // ---------------------- Parsing ----------------------
    sealed class FastScanner
    {
        readonly string _s;
        int _i;

        public FastScanner(string s) { _s = s; _i = 0; }

        public bool TryReadDouble(out double v)
        {
            v = 0;
            int n = _s.Length;
            while (_i < n && char.IsWhiteSpace(_s[_i])) _i++;
            if (_i >= n) return false;

            int start = _i;
            while (_i < n && !char.IsWhiteSpace(_s[_i])) _i++;
            var token = _s.AsSpan(start, _i - start);
            return double.TryParse(token, NumberStyles.Float, CultureInfo.InvariantCulture, out v);
        }

        public bool TryReadInt(out int v)
        {
            v = 0;
            double d;
            if (!TryReadDouble(out d)) return false;
            v = (int)d;
            return true;
        }
    }

    // ---------------------- Orbit ----------------------
    struct Orbit
    {
        public Vec3 R, V;

        public Vec3 HUnit;     // plane normal (unit angular momentum)
        public Vec3 EVec;      // eccentricity vector
        public double E;       // eccentricity
        public double A;       // semi-major axis (km), can be <0 for hyperbolic
        public double I;       // inclination (rad)
        public double Raan;    // RAAN (rad)
        public double Argp;    // arg of periapsis (rad)
        public double Nu;      // true anomaly (rad)
        public double M;       // mean anomaly (rad)
        public double Lambda;  // mean longitude approx (rad)
        public double RMean;   // characteristic radius (km)
    }

    static Orbit FromState(Vec3 r, Vec3 v, double mu)
    {
        // Classical elements from state; robust-ish for near-circular / equatorial.
        var h = Vec3.Cross(r, v);
        double hmag = h.Norm();
        var hUnit = (hmag > 0) ? (1.0 / hmag) * h : new Vec3(0, 0, 1);

        double rmag = r.Norm();
        double vmag = v.Norm();

        // Eccentricity vector
        var evec = (1.0 / mu) * Vec3.Cross(v, h) - (1.0 / rmag) * r;
        double e = evec.Norm();

        // Energy -> semi-major axis
        double energy = vmag * vmag * 0.5 - mu / rmag;
        double a = (Math.Abs(energy) > 1e-12) ? (-mu / (2.0 * energy)) : 1e18;

        // Inclination
        double inc = Math.Acos(Clamp(hUnit.Z, -1, 1));

        // Node vector
        var k = new Vec3(0, 0, 1);
        var nvec = Vec3.Cross(k, h);
        double nmag = nvec.Norm();

        // RAAN
        double raan = 0.0;
        if (nmag > 1e-12)
            raan = Wrap0To2Pi(Math.Atan2(nvec.Y, nvec.X));

        // Argument of periapsis
        double argp = 0.0;
        if (nmag > 1e-12 && e > 1e-10)
        {
            double c = Clamp(Vec3.Dot(nvec, evec) / (nmag * e), -1, 1);
            argp = Math.Acos(c);
            if (evec.Z < 0) argp = TwoPi - argp;
            argp = Wrap0To2Pi(argp);
        }

        // True anomaly
        double nu = 0.0;
        if (e > 1e-10)
        {
            double c = Clamp(Vec3.Dot(evec, r) / (e * rmag), -1, 1);
            nu = Math.Acos(c);
            if (Vec3.Dot(r, v) < 0) nu = TwoPi - nu;
        }
        else
        {
            // near circular: measure from node if possible, else inertial longitude
            if (nmag > 1e-12)
            {
                double c = Clamp(Vec3.Dot(nvec, r) / (nmag * rmag), -1, 1);
                nu = Math.Acos(c);
                if (r.Z < 0) nu = TwoPi - nu;
            }
            else
            {
                nu = Wrap0To2Pi(Math.Atan2(r.Y, r.X));
            }
        }
        nu = Wrap0To2Pi(nu);

        // Eccentric anomaly E and mean anomaly M (elliptic only; otherwise just use nu as proxy)
        double M = nu;
        if (a > 0 && e < 1.0 - 1e-10)
        {
            double sinNu2 = Math.Sin(nu * 0.5);
            double cosNu2 = Math.Cos(nu * 0.5);
            double Ean = 2.0 * Math.Atan2(Math.Sqrt(1 - e) * sinNu2, Math.Sqrt(1 + e) * cosNu2);
            Ean = Wrap0To2Pi(Ean);
            M = Wrap0To2Pi(Ean - e * Math.Sin(Ean));
        }

        // Characteristic radius:
        // for ellipse, average radius over true anomaly is a*(1 + e^2/2); for anything else, just rmag.
        double rMean = (a > 0 && e < 1.0) ? (a * (1.0 + 0.5 * e * e)) : rmag;
        if (!(rMean > 1)) rMean = rmag;

        double lambda = Wrap0To2Pi(raan + argp + M);

        return new Orbit
        {
            R = r,
            V = v,
            HUnit = hUnit,
            EVec = evec,
            E = e,
            A = a,
            I = inc,
            Raan = raan,
            Argp = argp,
            Nu = nu,
            M = M,
            Lambda = lambda,
            RMean = rMean
        };
    }

    // ---------------------- Cost metric (heuristic delta-V proxy) ----------------------
    static double HohmannLike(double mu, double r1, double r2)
    {
        r1 = Math.Max(1.0, r1);
        r2 = Math.Max(1.0, r2);
        if (Math.Abs(r1 - r2) < 1e-9) return 0.0;

        double v1 = Math.Sqrt(mu / r1);
        double v2 = Math.Sqrt(mu / r2);
        double sum = r1 + r2;

        double dv1 = v1 * (Math.Sqrt(2.0 * r2 / sum) - 1.0);
        double dv2 = v2 * (1.0 - Math.Sqrt(2.0 * r1 / sum));
        return Math.Abs(dv1) + Math.Abs(dv2);
    }

    static double OrbitTransferCost(in Orbit a, in Orbit b, double mu)
    {
        double r1 = a.RMean;
        double r2 = b.RMean;
        double rAvg = 0.5 * (r1 + r2);
        rAvg = Math.Max(1.0, rAvg);

        // Radial/energy change (Hohmann-like)
        double dvRad = HohmannLike(mu, r1, r2);

        // Plane change (angle between angular momentum vectors)
        double dot = Clamp(Vec3.Dot(a.HUnit, b.HUnit), -1, 1);
        double dPlane = Math.Acos(dot);
        double vScale = Math.Sqrt(mu / rAvg);
        double dvPlane = 2.0 * vScale * Math.Sin(0.5 * dPlane);

        // Shape change (eccentricity vector difference)
        Vec3 de = a.EVec - b.EVec;
        double deMag = de.Norm();
        double dvShape = 0.5 * vScale * deMag;

        // Phasing-ish penalty (mean longitude difference)
        double dLam = AngleDiff01Pi(a.Lambda, b.Lambda);
        double dvPhase = 0.04 * vScale * dLam; // keep small; just a tie-breaker

        // Weighting: keep plane/energy dominant, then shape, then phase.
        return dvRad + 0.9 * dvPlane + 0.6 * dvShape + dvPhase;
    }

    // Edge cost with special nodes:
    // u == -1 => start, v in [0..N-1]
    // v == -2 => end marker => edge cost 0
    static double Edge(int u, int v, double[] costStart, double[,] cost)
    {
        if (v == -2) return 0.0;
        if (u == -1) return costStart[v];
        return cost[u, v];
    }

    // ---------------------- Exact DP (path TSP from start, no return) ----------------------
    static int[] SolveExactDP(double[] costStart, double[,] cost, int n)
    {
        int size = 1 << n;
        int stride = n;
        double INF = 1e100;

        double[] dp = new double[size * stride];
        int[] parent = new int[size * stride];

        for (int i = 0; i < dp.Length; i++) { dp[i] = INF; parent[i] = -1; }

        for (int j = 0; j < n; j++)
        {
            int mask = 1 << j;
            dp[mask * stride + j] = costStart[j];
            parent[mask * stride + j] = -1;
        }

        for (int mask = 1; mask < size; mask++)
        {
            for (int j = 0; j < n; j++)
            {
                if (((mask >> j) & 1) == 0) continue;
                double cur = dp[mask * stride + j];
                if (cur >= INF * 0.5) continue;

                int rem = mask ^ (1 << j);
                if (rem == 0) continue;

                // We want dp[mask][j] = min_k dp[mask\{j}][k] + cost[k][j]
                // But we already built forward; do reverse relaxation:
            }
        }

        // Build properly with forward transitions
        for (int mask = 1; mask < size; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                if (((mask >> last) & 1) == 0) continue;
                double cur = dp[mask * stride + last];
                if (cur >= INF * 0.5) continue;

                int notVisited = (~mask) & (size - 1);
                while (notVisited != 0)
                {
                    int lsb = notVisited & -notVisited;
                    int nxt = BitIndex(lsb);
                    notVisited ^= lsb;

                    int nmask = mask | (1 << nxt);
                    double cand = cur + cost[last, nxt];
                    int idx = nmask * stride + nxt;
                    if (cand < dp[idx])
                    {
                        dp[idx] = cand;
                        parent[idx] = last;
                    }
                }
            }
        }

        int full = size - 1;
        double best = INF;
        int bestEnd = 0;
        for (int j = 0; j < n; j++)
        {
            double val = dp[full * stride + j];
            if (val < best) { best = val; bestEnd = j; }
        }

        int[] route = new int[n];
        int mask2 = full;
        int curNode = bestEnd;
        for (int pos = n - 1; pos >= 0; pos--)
        {
            route[pos] = curNode;
            int prev = parent[mask2 * stride + curNode];
            mask2 ^= 1 << curNode;
            curNode = prev;
            if (pos == 0) break;
        }
        return route;
    }

    static int BitIndex(int x)
    {
        // x is power-of-two
        int idx = 0;
        while ((x >>= 1) != 0) idx++;
        return idx;
    }

    // ---------------------- Heuristic ----------------------
    static double RouteCost(int[] route, double[] costStart, double[,] cost)
    {
        int n = route.Length;
        if (n == 0) return 0;
        double s = costStart[route[0]];
        for (int i = 0; i + 1 < n; i++)
            s += cost[route[i], route[i + 1]];
        return s;
    }

    static int[] BuildGreedyRoute(int n, double[] costStart, double[,] cost, Random rng, bool randomized)
    {
        int[] route = new int[n];
        bool[] used = new bool[n];

        int first;
        if (!randomized)
        {
            // deterministic: cheapest from start
            first = 0;
            double best = costStart[0];
            for (int j = 1; j < n; j++)
            {
                if (costStart[j] < best) { best = costStart[j]; first = j; }
            }
        }
        else
        {
            // biased random: choose among top K from start
            int K = Math.Min(8, n);
            int[] cand = new int[K];
            double[] candC = new double[K];
            for (int k = 0; k < K; k++) { cand[k] = -1; candC[k] = 1e100; }

            for (int j = 0; j < n; j++)
            {
                double c = costStart[j];
                int worst = 0;
                for (int k = 1; k < K; k++) if (candC[k] > candC[worst]) worst = k;
                if (c < candC[worst])
                {
                    candC[worst] = c;
                    cand[worst] = j;
                }
            }

            // Pick using exponential bias toward better
            Array.Sort(candC, cand);
            int pick = (int)Math.Floor(Math.Pow(rng.NextDouble(), 2.5) * K);
            if (pick >= K) pick = K - 1;
            first = cand[Math.Max(0, pick)];
            if (first < 0) first = rng.Next(n);
        }

        route[0] = first;
        used[first] = true;
        int last = first;

        for (int pos = 1; pos < n; pos++)
        {
            int next = -1;

            if (!randomized || n <= 25)
            {
                double best = 1e100;
                for (int j = 0; j < n; j++)
                {
                    if (used[j]) continue;
                    double c = cost[last, j];
                    if (c < best) { best = c; next = j; }
                }
            }
            else
            {
                // randomized nearest neighbor: choose among top K candidates
                int K = Math.Min(12, n - pos);
                int[] cand = new int[K];
                double[] candC = new double[K];
                for (int k = 0; k < K; k++) { cand[k] = -1; candC[k] = 1e100; }

                for (int j = 0; j < n; j++)
                {
                    if (used[j]) continue;
                    double c = cost[last, j];
                    int worst = 0;
                    for (int k = 1; k < K; k++) if (candC[k] > candC[worst]) worst = k;
                    if (c < candC[worst])
                    {
                        candC[worst] = c;
                        cand[worst] = j;
                    }
                }

                Array.Sort(candC, cand);
                int pick = (int)Math.Floor(Math.Pow(rng.NextDouble(), 2.2) * K);
                if (pick >= K) pick = K - 1;
                next = cand[Math.Max(0, pick)];
                if (next < 0)
                {
                    // fallback
                    for (int j = 0; j < n; j++) if (!used[j]) { next = j; break; }
                }
            }

            route[pos] = next;
            used[next] = true;
            last = next;
        }

        return route;
    }

    static void TwoOptImprove(int[] route, double[] costStart, double[,] cost, Stopwatch sw, double timeLimitSeconds)
    {
        int n = route.Length;
        if (n < 4) return;

        int maxPasses = n <= 40 ? 60 : 25;
        for (int pass = 0; pass < maxPasses; pass++)
        {
            if (sw.Elapsed.TotalSeconds > timeLimitSeconds) return;

            bool improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                if (sw.Elapsed.TotalSeconds > timeLimitSeconds) return;

                int prevNode = (i == 0) ? -1 : route[i - 1];
                int a = route[i];

                for (int k = i + 1; k < n; k++)
                {
                    int b = route[k];
                    int nextNode = (k == n - 1) ? -2 : route[k + 1];

                    double oldCost = Edge(prevNode, a, costStart, cost) + Edge(b, nextNode, costStart, cost);
                    double newCost = Edge(prevNode, b, costStart, cost) + Edge(a, nextNode, costStart, cost);

                    if (newCost + 1e-12 < oldCost)
                    {
                        Reverse(route, i, k);
                        improved = true;
                        goto NextOuter; // restart scanning (first-improvement)
                    }
                }
            }
        NextOuter:
            if (!improved) break;
        }
    }

    static void Reverse(int[] a, int i, int k)
    {
        while (i < k)
        {
            int t = a[i];
            a[i] = a[k];
            a[k] = t;
            i++; k--;
        }
    }

    static void Relocate1Improve(int[] route, double[] costStart, double[,] cost, Stopwatch sw, double timeLimitSeconds)
    {
        int n = route.Length;
        if (n < 4) return;

        // A couple of steepest-descent passes
        int[] buf = new int[Math.Max(0, n - 1)];
        for (int pass = 0; pass < 2; pass++)
        {
            if (sw.Elapsed.TotalSeconds > timeLimitSeconds) return;

            double bestDelta = 0.0;
            int bestFrom = -1, bestTo = -1;

            for (int from = 0; from < n; from++)
            {
                if (sw.Elapsed.TotalSeconds > timeLimitSeconds) return;

                int x = route[from];

                // Build route without x into buf
                int m = 0;
                for (int i = 0; i < n; i++)
                    if (i != from) buf[m++] = route[i];

                int prev = (from == 0) ? -1 : route[from - 1];
                int next = (from == n - 1) ? -2 : route[from + 1];

                // Removal change (new - old)
                double removalDelta = Edge(prev, next, costStart, cost) - Edge(prev, x, costStart, cost) - Edge(x, next, costStart, cost);

                // Try all insertion positions into buf (length n-1), producing length n
                // insert position 'to' in [0..n-1] in final route
                for (int to = 0; to <= n - 1; to++)
                {
                    // Inserting back into same position is allowed but won't improve; skip roughly:
                    // (we skip exact equality by checking if to == from)
                    if (to == from) continue;

                    int a = (to == 0) ? -1 : buf[to - 1];
                    int b = (to == n - 1) ? -2 : buf[to];

                    double insertionDelta = -Edge(a, b, costStart, cost) + Edge(a, x, costStart, cost) + Edge(x, b, costStart, cost);
                    double delta = removalDelta + insertionDelta;

                    if (delta < bestDelta - 1e-12)
                    {
                        bestDelta = delta;
                        bestFrom = from;
                        bestTo = to;
                    }
                }
            }

            if (bestFrom < 0) break;

            // Apply move
            int node = route[bestFrom];
            if (bestTo < bestFrom)
            {
                // shift right [bestTo..bestFrom-1]
                for (int i = bestFrom; i > bestTo; i--) route[i] = route[i - 1];
                route[bestTo] = node;
            }
            else
            {
                // bestTo > bestFrom: after removing, indices shift left.
                for (int i = bestFrom; i < bestTo - 1; i++) route[i] = route[i + 1];
                route[bestTo - 1] = node;
            }
        }
    }

    static int[] SolveHeuristic(double[] costStart, double[,] cost, int n, Stopwatch sw)
    {
        var rng = new Random(1234567);

        int restarts;
        if (n <= 20) restarts = 6;
        else if (n <= 40) restarts = 12;
        else if (n <= 70) restarts = 20;
        else restarts = 28;

        // Keep some deterministic + randomized starts
        int[] bestRoute = BuildGreedyRoute(n, costStart, cost, rng, randomized: false);
        TwoOptImprove(bestRoute, costStart, cost, sw, timeLimitSeconds: 280.0);
        Relocate1Improve(bestRoute, costStart, cost, sw, timeLimitSeconds: 280.0);
        double bestCost = RouteCost(bestRoute, costStart, cost);

        for (int r = 0; r < restarts; r++)
        {
            if (sw.Elapsed.TotalSeconds > 280.0) break;

            bool randomized = true;
            int[] route = BuildGreedyRoute(n, costStart, cost, rng, randomized);
            TwoOptImprove(route, costStart, cost, sw, timeLimitSeconds: 280.0);
            Relocate1Improve(route, costStart, cost, sw, timeLimitSeconds: 280.0);

            double c = RouteCost(route, costStart, cost);
            if (c < bestCost)
            {
                bestCost = c;
                bestRoute = route;
            }
        }

        // One last polish
        TwoOptImprove(bestRoute, costStart, cost, sw, timeLimitSeconds: 295.0);
        return bestRoute;
    }

    // ---------------------- Main ----------------------
    public static void Main()
    {
        var sw = Stopwatch.StartNew();

        string input = Console.In.ReadToEnd();
        var fs = new FastScanner(input);

        if (!fs.TryReadInt(out int N)) return;
        if (!fs.TryReadDouble(out double mu)) mu = 398600.4418;

        // Start orbit
        if (!fs.TryReadDouble(out double sx)) return;
        fs.TryReadDouble(out double sy);
        fs.TryReadDouble(out double sz);
        fs.TryReadDouble(out double svx);
        fs.TryReadDouble(out double svy);
        fs.TryReadDouble(out double svz);

        Orbit start = FromState(new Vec3(sx, sy, sz), new Vec3(svx, svy, svz), mu);

        Orbit[] st = new Orbit[N];
        for (int i = 0; i < N; i++)
        {
            fs.TryReadDouble(out double x);
            fs.TryReadDouble(out double y);
            fs.TryReadDouble(out double z);
            fs.TryReadDouble(out double vx);
            fs.TryReadDouble(out double vy);
            fs.TryReadDouble(out double vz);
            st[i] = FromState(new Vec3(x, y, z), new Vec3(vx, vy, vz), mu);
        }

        // Precompute costs
        double[] costStart = new double[N];
        for (int j = 0; j < N; j++)
            costStart[j] = OrbitTransferCost(start, st[j], mu);

        double[,] cost = new double[N, N];
        for (int i = 0; i < N; i++)
        {
            cost[i, i] = 0;
            for (int j = i + 1; j < N; j++)
            {
                double c = OrbitTransferCost(st[i], st[j], mu);
                cost[i, j] = c;
                cost[j, i] = c; // keep symmetric
            }
        }

        int[] route;
        if (N <= 15)
        {
            route = SolveExactDP(costStart, cost, N);
        }
        else
        {
            route = SolveHeuristic(costStart, cost, N, sw);
        }

        // Output station indices 0..N-1
        for (int i = 0; i < route.Length; i++)
        {
            if (i != 0) Console.Write(' ');
            Console.Write(route[i]);
        }
        Console.WriteLine();
    }
}
