; SPIR-V
; Version: 1.0
; Bound: 100
; Schema: 0
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%fn = OpTypeFunction %void

%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

%ptrInputv3float = OpTypePointer Input %v3float
%ptrOutputv4float = OpTypePointer Output %v4float
%ptrUniformUBO = OpTypePointer Uniform %UBO
%ptrUniformv4float = OpTypePointer Uniform %v4float

%inWorldPos = OpVariable %ptrInputv3float Input
%inNormal = OpVariable %ptrInputv3float Input
%outColor = OpVariable %ptrOutputv4float Output
%ubo = OpVariable %ptrUniformUBO Uniform

%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_2 = OpConstant %float 0.2
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8

%baseColor = OpConstantComposite %v3float %float_0_2 %float_0_3 %float_0_8
%rimTint = OpConstantComposite %v3float %float_1 %float_0_8 %float_0_5

%main = OpFunction %void None %fn
%entry = OpLabel

%worldPos = OpLoad %v3float %inWorldPos
%normIn = OpLoad %v3float %inNormal
%N = OpExtInst %v3float %glsl Normalize %normIn

%ptrLight = OpAccessChain %ptrUniformv4float %ubo %int_3
%light4 = OpLoad %v4float %ptrLight
%light3 = OpVectorShuffle %v3float %light4 %light4 0 1 2
%Lvec = OpFSub %v3float %light3 %worldPos
%L = OpExtInst %v3float %glsl Normalize %Lvec

%ptrCam = OpAccessChain %ptrUniformv4float %ubo %int_4
%cam4 = OpLoad %v4float %ptrCam
%cam3 = OpVectorShuffle %v3float %cam4 %cam4 0 1 2
%Vvec = OpFSub %v3float %cam3 %worldPos
%V = OpExtInst %v3float %glsl Normalize %Vvec

%ndotl_raw = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl FMax %ndotl_raw %float_0

%ndotv_raw = OpDot %float %N %V
%ndotv = OpExtInst %float %glsl FMax %ndotv_raw %float_0
%oneMinus = OpFSub %float %float_1 %ndotv
%rim2 = OpFMul %float %oneMinus %oneMinus
%rim = OpFMul %float %rim2 %oneMinus

%diffScaled = OpFMul %float %float_0_5 %diffuse
%baseFactor = OpFAdd %float %float_0_3 %diffScaled

%baseTerm = OpVectorTimesScalar %v3float %baseColor %baseFactor
%rimTerm = OpVectorTimesScalar %v3float %rimTint %rim
%rgb = OpFAdd %v3float %baseTerm %rimTerm

%r = OpCompositeExtract %float %rgb 0
%g = OpCompositeExtract %float %rgb 1
%b = OpCompositeExtract %float %rgb 2
%outV = OpCompositeConstruct %v4float %r %g %b %float_1
OpStore %outColor %outV

OpReturn
OpFunctionEnd
