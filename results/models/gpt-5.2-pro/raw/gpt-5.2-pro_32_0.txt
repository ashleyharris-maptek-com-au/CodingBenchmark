#include <algorithm>
#include <cassert>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <unordered_set>
#include <utility>
#include <vector>

using ll = long long;
static constexpr ll INF = (std::numeric_limits<ll>::max() / 4);

struct InputEdge {
    int u, v;
    ll w;
};

struct AdjEdge {
    int to;
    ll w;
    int id;
};

struct DSU {
    std::vector<int> p, r;
    DSU() = default;
    explicit DSU(int n) : p(n), r(n, 0) { std::iota(p.begin(), p.end(), 0); }
    int find(int a) { return p[a] == a ? a : (p[a] = find(p[a])); }
    bool unite(int a, int b) {
        a = find(a); b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) std::swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) r[a]++;
        return true;
    }
};

static void dijkstra_single_source(
    int n,
    const std::vector<std::vector<AdjEdge>>& g,
    int src,
    std::vector<ll>& dist,
    std::vector<int>& prevV,
    std::vector<int>& prevE
) {
    dist.assign(n, INF);
    prevV.assign(n, -1);
    prevE.assign(n, -1);

    using P = std::pair<ll, int>;
    std::priority_queue<P, std::vector<P>, std::greater<P>> pq;
    dist[src] = 0;
    pq.push({0, src});

    while (!pq.empty()) {
        auto [d, v] = pq.top(); pq.pop();
        if (d != dist[v]) continue;
        for (const auto& e : g[v]) {
            ll nd = d + e.w;
            if (nd < dist[e.to]) {
                dist[e.to] = nd;
                prevV[e.to] = v;
                prevE[e.to] = e.id;
                pq.push({nd, e.to});
            }
        }
    }
}

static void dijkstra_multi_source(
    int n,
    const std::vector<std::vector<AdjEdge>>& g,
    const std::vector<int>& sources,
    std::vector<ll>& dist,
    std::vector<int>& prevV,
    std::vector<int>& prevE
) {
    dist.assign(n, INF);
    prevV.assign(n, -1);
    prevE.assign(n, -1);

    using P = std::pair<ll, int>;
    std::priority_queue<P, std::vector<P>, std::greater<P>> pq;

    for (int s : sources) {
        if (dist[s] > 0) {
            dist[s] = 0;
            pq.push({0, s});
        }
    }

    while (!pq.empty()) {
        auto [d, v] = pq.top(); pq.pop();
        if (d != dist[v]) continue;
        for (const auto& e : g[v]) {
            ll nd = d + e.w;
            if (nd < dist[e.to]) {
                dist[e.to] = nd;
                prevV[e.to] = v;
                prevE[e.to] = e.id;
                pq.push({nd, e.to});
            }
        }
    }
}

static void add_path_edges_to_used(
    int src,
    int tgt,
    const std::vector<int>& prevV,
    const std::vector<int>& prevE,
    std::vector<char>& usedEdge
) {
    int cur = tgt;
    while (cur != src) {
        int pe = prevE[cur];
        int pv = prevV[cur];
        if (pe < 0 || pv < 0) break; // unreachable / broken
        usedEdge[pe] = 1;
        cur = pv;
    }
}

static void add_path_until_reach_marked(
    int start,
    const std::vector<int>& prevV,
    const std::vector<int>& prevE,
    const std::vector<char>& isMarkedSource,
    std::vector<char>& usedEdge,
    std::vector<char>& inTreeVertex
) {
    int cur = start;
    while (cur >= 0 && !isMarkedSource[cur]) {
        int pe = prevE[cur];
        int pv = prevV[cur];
        if (pe < 0 || pv < 0) break;
        usedEdge[pe] = 1;
        inTreeVertex[cur] = 1;
        cur = pv;
    }
    if (cur >= 0) inTreeVertex[cur] = 1;
}

static std::vector<int> kruskal_on_used_edges(
    int n,
    const std::vector<InputEdge>& edges,
    const std::vector<char>& usedEdge
) {
    std::vector<int> ids;
    ids.reserve(edges.size());
    for (int i = 0; i < (int)edges.size(); i++) if (usedEdge[i]) ids.push_back(i);

    std::sort(ids.begin(), ids.end(), [&](int a, int b) {
        if (edges[a].w != edges[b].w) return edges[a].w < edges[b].w;
        if (edges[a].u != edges[b].u) return edges[a].u < edges[b].u;
        return edges[a].v < edges[b].v;
    });

    DSU dsu(n);
    std::vector<int> picked;
    picked.reserve(ids.size());
    for (int id : ids) {
        if (dsu.unite(edges[id].u, edges[id].v)) picked.push_back(id);
    }
    return picked;
}

static std::vector<int> prune_nonterminal_leaves(
    int n,
    const std::vector<InputEdge>& edges,
    const std::vector<int>& treeEdgeIds,
    const std::vector<char>& isTerminal
) {
    // Build adjacency for the current tree.
    int E = (int)treeEdgeIds.size();
    std::vector<std::vector<std::pair<int,int>>> adj(n);
    adj.assign(n, {});
    adj.shrink_to_fit();
    adj.assign(n, {});

    std::vector<int> deg(n, 0);
    std::vector<char> active(E, 1);

    for (int i = 0; i < E; i++) {
        auto &e = edges[treeEdgeIds[i]];
        adj[e.u].push_back({e.v, i});
        adj[e.v].push_back({e.u, i});
        deg[e.u]++;
        deg[e.v]++;
    }

    std::queue<int> q;
    for (int v = 0; v < n; v++) {
        if (!isTerminal[v] && deg[v] == 1) q.push(v);
    }

    while (!q.empty()) {
        int v = q.front(); q.pop();
        if (isTerminal[v] || deg[v] != 1) continue;

        // Find its only active incident edge
        int remEdgeIndex = -1;
        int nei = -1;
        for (auto [to, ei] : adj[v]) {
            if (active[ei]) { remEdgeIndex = ei; nei = to; break; }
        }
        if (remEdgeIndex < 0) continue;

        active[remEdgeIndex] = 0;
        deg[v]--;
        deg[nei]--;
        if (!isTerminal[nei] && deg[nei] == 1) q.push(nei);
    }

    std::vector<int> pruned;
    pruned.reserve(E);
    for (int i = 0; i < E; i++) if (active[i]) pruned.push_back(treeEdgeIds[i]);
    return pruned;
}

static bool check_terminals_connected(
    int n,
    const std::vector<InputEdge>& edges,
    const std::vector<int>& edgeIds,
    const std::vector<int>& terminals
) {
    if (terminals.empty()) return true;
    DSU dsu(n);
    for (int id : edgeIds) dsu.unite(edges[id].u, edges[id].v);
    int root = dsu.find(terminals[0]);
    for (int t : terminals) if (dsu.find(t) != root) return false;
    return true;
}

// ----- Exact DP (Dreyfus-Wagner style) for small instances -----
struct DPState {
    int predV;   // predecessor vertex when last step is via an edge; -1 if none
    int predE;   // predecessor edge id; -1 if none
    int split;   // subset used for merge at same vertex; -1 if none
};

static bool steiner_exact_dp(
    int n,
    const std::vector<std::vector<AdjEdge>>& g,
    const std::vector<int>& terminals,
    const std::vector<InputEdge>& edges,
    std::vector<char>& usedEdgeOut
) {
    int k = (int)terminals.size();
    if (k <= 1) {
        usedEdgeOut.assign(edges.size(), 0);
        return true;
    }
    if (k > 18) return false;

    const int FULL = (1 << k);
    const uint64_t states = (uint64_t)FULL * (uint64_t)n;
    // Safety threshold (memory/time). Adjust conservatively.
    if (states > 3'000'000ULL) return false;

    std::vector<ll> dp(states, INF);
    std::vector<DPState> st(states);
    for (uint64_t i = 0; i < states; i++) st[i] = DPState{-1, -1, -1};

    auto idx = [&](int mask, int v) -> uint64_t { return (uint64_t)mask * (uint64_t)n + (uint64_t)v; };

    // Initialize: dp[1<<i][term_i] = 0
    for (int i = 0; i < k; i++) {
        dp[idx(1 << i, terminals[i])] = 0;
    }

    // For each mask: do subset merging then Dijkstra relaxation (multi-source with current dp)
    std::vector<ll> dist;
    std::vector<char> inPQ; inPQ.reserve(n);

    for (int mask = 1; mask < FULL; mask++) {
        // Merge
        if ((mask & (mask - 1)) != 0) {
            for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
                int other = mask ^ sub;
                if (sub > other) continue; // avoid duplicate partitions
                for (int v = 0; v < n; v++) {
                    ll a = dp[idx(sub, v)];
                    ll b = dp[idx(other, v)];
                    if (a == INF || b == INF) continue;
                    ll cand = a + b;
                    uint64_t id = idx(mask, v);
                    if (cand < dp[id]) {
                        dp[id] = cand;
                        st[id].split = sub;
                        st[id].predV = -1;
                        st[id].predE = -1;
                    }
                }
            }
        }

        // Dijkstra with initial distances dp[mask][v]
        using P = std::pair<ll, int>;
        std::priority_queue<P, std::vector<P>, std::greater<P>> pq;
        for (int v = 0; v < n; v++) {
            ll d0 = dp[idx(mask, v)];
            if (d0 < INF) pq.push({d0, v});
        }

        while (!pq.empty()) {
            auto [d, v] = pq.top(); pq.pop();
            uint64_t idv = idx(mask, v);
            if (d != dp[idv]) continue;
            for (const auto& e : g[v]) {
                uint64_t idto = idx(mask, e.to);
                ll nd = d + e.w;
                if (nd < dp[idto]) {
                    dp[idto] = nd;
                    st[idto].predV = v;
                    st[idto].predE = e.id;
                    st[idto].split = -1;
                    pq.push({nd, e.to});
                }
            }
        }
    }

    int all = FULL - 1;
    ll best = INF;
    int bestV = -1;
    for (int v = 0; v < n; v++) {
        ll d = dp[idx(all, v)];
        if (d < best) { best = d; bestV = v; }
    }
    if (bestV < 0 || best >= INF) return false;

    usedEdgeOut.assign(edges.size(), 0);

    // Reconstruct recursively
    std::vector<std::pair<int,int>> stack; // (mask, v)
    stack.push_back({all, bestV});

    while (!stack.empty()) {
        auto [mask, v] = stack.back();
        stack.pop_back();

        uint64_t id = idx(mask, v);
        if (st[id].predV != -1) {
            int pv = st[id].predV;
            int pe = st[id].predE;
            if (pe >= 0) usedEdgeOut[pe] = 1;
            stack.push_back({mask, pv});
            continue;
        }
        if (st[id].split != -1) {
            int sub = st[id].split;
            int other = mask ^ sub;
            stack.push_back({sub, v});
            stack.push_back({other, v});
            continue;
        }
        // base: should be a terminal state with v as some source (no predecessor)
    }

    return true;
}

// ----- Heuristics -----

static bool steiner_kmb(
    int n,
    const std::vector<std::vector<AdjEdge>>& g,
    const std::vector<int>& terminals,
    const std::vector<InputEdge>& edges,
    std::vector<char>& usedEdgeOut
) {
    int k = (int)terminals.size();
    if (k <= 1) {
        usedEdgeOut.assign(edges.size(), 0);
        return true;
    }

    // All-pairs (terminal-to-terminal) distances via k Dijkstra
    std::vector<std::vector<ll>> distTT(k, std::vector<ll>(k, INF));
    std::vector<ll> dist;
    std::vector<int> prevV, prevE;

    for (int i = 0; i < k; i++) {
        dijkstra_single_source(n, g, terminals[i], dist, prevV, prevE);
        for (int j = 0; j < k; j++) distTT[i][j] = dist[terminals[j]];
    }

    // If some terminal unreachable from others, KMB fails
    for (int i = 0; i < k; i++) {
        if (distTT[i][i] != 0) return false;
    }

    // Prim MST on terminal complete graph
    std::vector<ll> key(k, INF);
    std::vector<int> parent(k, -1);
    std::vector<char> inMST(k, 0);
    key[0] = 0;
    for (int it = 0; it < k; it++) {
        int u = -1;
        for (int i = 0; i < k; i++) if (!inMST[i] && (u == -1 || key[i] < key[u])) u = i;
        if (u == -1 || key[u] >= INF) return false;
        inMST[u] = 1;
        for (int v = 0; v < k; v++) if (!inMST[v]) {
            if (distTT[u][v] < key[v]) {
                key[v] = distTT[u][v];
                parent[v] = u;
            }
        }
    }

    usedEdgeOut.assign(edges.size(), 0);

    // For each MST edge (term v - term parent[v]), expand shortest path in original graph.
    for (int v = 1; v < k; v++) {
        int u = parent[v];
        if (u < 0) return false;
        // rerun Dijkstra from terminals[u] and reconstruct path to terminals[v]
        dijkstra_single_source(n, g, terminals[u], dist, prevV, prevE);
        if (dist[terminals[v]] >= INF) return false;
        add_path_edges_to_used(terminals[u], terminals[v], prevV, prevE, usedEdgeOut);
    }
    return true;
}

// Simple fallback: pick a root terminal, run Dijkstra once, union of predecessor paths to root.
static bool steiner_root_shortest_path_tree(
    int n,
    const std::vector<std::vector<AdjEdge>>& g,
    const std::vector<int>& terminals,
    const std::vector<InputEdge>& edges,
    std::vector<char>& usedEdgeOut
) {
    int k = (int)terminals.size();
    usedEdgeOut.assign(edges.size(), 0);
    if (k <= 1) return true;

    int root = terminals[0];
    std::vector<ll> dist;
    std::vector<int> prevV, prevE;
    dijkstra_single_source(n, g, root, dist, prevV, prevE);

    for (int t : terminals) {
        if (dist[t] >= INF) return false;
        int cur = t;
        while (cur != root) {
            int pe = prevE[cur];
            int pv = prevV[cur];
            if (pe < 0 || pv < 0) return false;
            usedEdgeOut[pe] = 1;
            cur = pv;
        }
    }
    return true;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, m, k;
    if (!(std::cin >> n >> m >> k)) return 0;

    std::vector<InputEdge> edges;
    edges.reserve(m);
    std::vector<std::vector<AdjEdge>> g(n);

    for (int i = 0; i < m; i++) {
        int u, v;
        ll w;
        std::cin >> u >> v >> w;
        edges.push_back({u, v, w});
        g[u].push_back({v, w, i});
        g[v].push_back({u, w, i});
    }

    std::vector<int> terminals(k);
    for (int i = 0; i < k; i++) std::cin >> terminals[i];

    if (k <= 1) {
        std::cout << 0 << " " << 0 << "\n";
        return 0;
    }

    std::vector<char> isTerminal(n, 0);
    for (int t : terminals) if (0 <= t && t < n) isTerminal[t] = 1;

    std::vector<char> usedEdge(m, 0);
    bool ok = false;

    // Strategy selection
    // 1) Exact DP if very small (both k and state space)
    if (!ok) {
        ok = steiner_exact_dp(n, g, terminals, edges, usedEdge);
    }
    // 2) KMB heuristic if moderate terminal count
    if (!ok && k <= 2000) {
        ok = steiner_kmb(n, g, terminals, edges, usedEdge);
    }
    // 3) Always-works fast fallback
    if (!ok) {
        ok = steiner_root_shortest_path_tree(n, g, terminals, edges, usedEdge);
    }

    // If still not ok, output empty (no feasible tree)
    if (!ok) {
        std::cout << 0 << " " << 0 << "\n";
        return 0;
    }

    // Make it acyclic & connected (within the used subgraph) via Kruskal, then prune.
    std::vector<int> treeEdges = kruskal_on_used_edges(n, edges, usedEdge);

    // If Kruskal result doesn't connect all terminals, fall back to root-SPT and retry
    if (!check_terminals_connected(n, edges, treeEdges, terminals)) {
        std::vector<char> used2;
        if (steiner_root_shortest_path_tree(n, g, terminals, edges, used2)) {
            treeEdges = kruskal_on_used_edges(n, edges, used2);
        }
    }

    // Prune non-terminal leaves to reduce weight.
    treeEdges = prune_nonterminal_leaves(n, edges, treeEdges, isTerminal);

    // Final sanity: if disconnected, try fallback without prune.
    if (!check_terminals_connected(n, edges, treeEdges, terminals)) {
        std::vector<char> used2;
        if (steiner_root_shortest_path_tree(n, g, terminals, edges, used2)) {
            treeEdges = kruskal_on_used_edges(n, edges, used2);
            treeEdges = prune_nonterminal_leaves(n, edges, treeEdges, isTerminal);
        }
    }

    // Output
    ll totalW = 0;
    for (int id : treeEdges) totalW += edges[id].w;

    std::cout << totalW << " " << (int)treeEdges.size() << "\n";
    for (int id : treeEdges) {
        std::cout << edges[id].u << " " << edges[id].v << "\n";
    }
    return 0;
}
