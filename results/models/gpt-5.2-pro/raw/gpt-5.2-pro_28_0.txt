```rust
use std::io::{self, Read, Write};

struct Scanner {
    buf: Vec<u8>,
    i: usize,
}

impl Scanner {
    fn new() -> io::Result<Self> {
        let mut buf = Vec::new();
        io::stdin().lock().read_to_end(&mut buf)?;
        Ok(Self { buf, i: 0 })
    }

    #[inline]
    fn next_usize(&mut self) -> Option<usize> {
        while self.i < self.buf.len() && self.buf[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
        if self.i >= self.buf.len() {
            return None;
        }
        let mut x: usize = 0;
        while self.i < self.buf.len() {
            let b = self.buf[self.i];
            if !b.is_ascii_digit() {
                break;
            }
            x = x
                .saturating_mul(10)
                .saturating_add((b - b'0') as usize);
            self.i += 1;
        }
        Some(x)
    }
}

#[inline]
fn push_usize(out: &mut Vec<u8>, mut x: usize) {
    if x == 0 {
        out.push(b'0');
        return;
    }
    // Enough for 64-bit usize: max 20 digits.
    let mut tmp = [0u8; 20];
    let mut k = 0usize;
    while x > 0 {
        tmp[k] = b'0' + (x % 10) as u8;
        x /= 10;
        k += 1;
    }
    for j in (0..k).rev() {
        out.push(tmp[j]);
    }
}

fn main() -> io::Result<()> {
    let mut sc = Scanner::new()?;

    let n = match sc.next_usize() {
        Some(v) => v,
        None => return Ok(()),
    };
    let m = sc.next_usize().unwrap_or(0);

    // Greedy maximal matching => 2-approx vertex cover:
    // take both endpoints of each matched edge.
    let mut chosen = vec![0u8; n];

    for _ in 0..m {
        let u = sc.next_usize().unwrap_or(0) as usize;
        let v = sc.next_usize().unwrap_or(0) as usize;

        if u < n && v < n && chosen[u] == 0 && chosen[v] == 0 {
            chosen[u] = 1;
            chosen[v] = 1;
        }
    }

    let mut cover: Vec<usize> = Vec::new();
    cover.reserve(n / 2);
    for (idx, &c) in chosen.iter().enumerate() {
        if c != 0 {
            cover.push(idx);
        }
    }

    // Output
    let mut out: Vec<u8> = Vec::with_capacity(32 + cover.len().saturating_mul(9));
    push_usize(&mut out, cover.len());
    out.push(b'\n');

    for (i, &v) in cover.iter().enumerate() {
        if i != 0 {
            out.push(b' ');
        }
        push_usize(&mut out, v);
    }
    out.push(b'\n');

    io::stdout().lock().write_all(&out)?;
    Ok(())
}
```
