#include <bits/stdc++.h>
using namespace std;

struct DynBitset {
    int n = 0;
    int blocks = 0;
    uint64_t lastMask = ~0ULL;
    vector<uint64_t> b;

    DynBitset() = default;
    DynBitset(int n_, int blocks_, uint64_t lastMask_, bool fillZeros = true)
        : n(n_), blocks(blocks_), lastMask(lastMask_), b(blocks_, fillZeros ? 0ULL : ~0ULL) {
        if (!b.empty()) b.back() &= lastMask;
    }

    static DynBitset full(int n_, int blocks_, uint64_t lastMask_) {
        DynBitset s(n_, blocks_, lastMask_, false);
        return s;
    }

    inline void trim() {
        if (!b.empty()) b.back() &= lastMask;
    }

    inline bool any() const {
        for (uint64_t x : b) if (x) return true;
        return false;
    }
    inline bool none() const { return !any(); }

    inline void set(int i) {
        b[i >> 6] |= (1ULL << (i & 63));
    }
    inline void reset(int i) {
        b[i >> 6] &= ~(1ULL << (i & 63));
    }
    inline bool test(int i) const {
        return (b[i >> 6] >> (i & 63)) & 1ULL;
    }

    inline int pop_lsb() {
        for (int bi = 0; bi < blocks; ++bi) {
            uint64_t x = b[bi];
            if (!x) continue;
            int t = __builtin_ctzll(x);
            b[bi] = x & (x - 1);
            return (bi << 6) + t;
        }
        return -1;
    }

    inline void andWith(const DynBitset& o) {
        for (int i = 0; i < blocks; ++i) b[i] &= o.b[i];
    }
    inline void andNotWith(const DynBitset& o) {
        for (int i = 0; i < blocks; ++i) b[i] &= ~o.b[i];
        trim();
    }

    friend DynBitset operator&(const DynBitset& a, const DynBitset& c) {
        DynBitset r = a;
        r.andWith(c);
        return r;
    }
};

static int N, M, B;
static uint64_t LAST_MASK;

static vector<long long> w;
static vector<DynBitset> adjH;  // complement graph adjacency (for clique)
static DynBitset POS;           // vertices allowed (weight>0 and not forced to 0)

static long long bestW = 0;
static vector<int> bestClique;

static void colorSortWeighted(const DynBitset& P, vector<int>& order, vector<long long>& bounds) {
    // Tomita-style: repeatedly extract a greedy maximal independent set -> one color class.
    DynBitset U = P;

    order.clear();
    bounds.clear();
    vector<int> colorOf;              // parallel to order[]
    vector<long long> maxWPerColor;   // 1-based index of colors

    int color = 0;
    while (U.any()) {
        ++color;
        DynBitset Q = U;
        long long mx = 0;

        while (Q.any()) {
            int v = Q.pop_lsb();
            order.push_back(v);
            colorOf.push_back(color);
            mx = max(mx, w[v]);

            U.reset(v);

            // keep only vertices non-adjacent to v within this color class
            Q.andNotWith(adjH[v]);
            // and keep only those still in U
            Q.andWith(U);
        }
        maxWPerColor.push_back(mx);
    }

    // prefix sums of per-color maxima
    vector<long long> pref(color + 1, 0);
    for (int c = 1; c <= color; ++c) pref[c] = pref[c - 1] + maxWPerColor[c - 1];

    bounds.resize(order.size());
    for (size_t i = 0; i < order.size(); ++i) bounds[i] = pref[colorOf[i]];
    // bounds is nondecreasing due to color-grouped order
}

static void expand(DynBitset P, vector<int>& C, long long CW) {
    if (CW > bestW) {
        bestW = CW;
        bestClique = C;
    }
    if (P.none()) return;

    vector<int> order;
    vector<long long> bounds;
    colorSortWeighted(P, order, bounds);

    for (int i = (int)order.size() - 1; i >= 0; --i) {
        int v = order[i];

        // Upper bound check for remaining vertices in order[0..i]
        if (CW + bounds[i] <= bestW) return;

        // Include v
        C.push_back(v);
        long long CW2 = CW + w[v];
        DynBitset P2 = P & adjH[v]; // must be adjacent to v in complement => clique property
        expand(P2, C, CW2);
        C.pop_back();

        // Exclude v
        P.reset(v);
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> N >> M;
    w.assign(N, 0);
    for (int i = 0; i < N; ++i) cin >> w[i];

    B = (N + 63) / 64;
    if (N % 64 == 0) LAST_MASK = ~0ULL;
    else LAST_MASK = (1ULL << (N % 64)) - 1ULL;

    vector<DynBitset> adjG(N, DynBitset(N, B, LAST_MASK));
    vector<char> forcedZero(N, 0);

    for (int i = 0; i < M; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= N || v < 0 || v >= N) continue; // defensive
        if (u == v) {
            forcedZero[u] = 1; // 2*x_u <= 1 => x_u = 0
            continue;
        }
        adjG[u].set(v);
        adjG[v].set(u);
    }

    // Build complement adjacency adjH
    DynBitset ALL = DynBitset::full(N, B, LAST_MASK);
    adjH.assign(N, DynBitset(N, B, LAST_MASK));
    for (int i = 0; i < N; ++i) {
        adjH[i] = ALL;
        adjH[i].reset(i);
        adjH[i].andNotWith(adjG[i]); // edges where NOT adjacent in G (and not self)
    }

    // Allowed vertices: positive weight and not forced to 0
    POS = DynBitset(N, B, LAST_MASK);
    for (int i = 0; i < N; ++i) {
        if (!forcedZero[i] && w[i] > 0) POS.set(i);
    }

    // Solve maximum weight clique in complement graph (=> maximum weight independent set in original)
    DynBitset P0 = POS;
    vector<int> C;
    bestW = 0;
    bestClique.clear();
    expand(P0, C, 0);

    // Output independent set in original graph: x_i = 1 for vertices in bestClique
    vector<int> x(N, 0);
    for (int v : bestClique) x[v] = 1;

    long long obj = 0;
    for (int i = 0; i < N; ++i) obj += (long long)x[i] * w[i];

    cout << obj << "\n";
    for (int i = 0; i < N; ++i) {
        if (i) cout << ' ';
        cout << x[i];
    }
    cout << "\n";
    return 0;
}
