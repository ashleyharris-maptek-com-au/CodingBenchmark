```csharp
using System;
using System.Diagnostics;
using System.Text;

public class Program
{
    // ---------- Fast scanner ----------
    private sealed class FastScanner
    {
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = Console.OpenStandardInput().Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return int.MinValue;
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int x = 0;
            while (c > ' ')
            {
                x = x * 10 + (c - '0');
                c = ReadByte();
            }
            return x * sign;
        }
    }

    // ---------- Helpers ----------
    static long ComputeCost(int n, int[] flow, int[] dist, int[] perm)
    {
        long cost = 0;
        for (int i = 0; i < n; i++)
        {
            int li = perm[i];
            int iN = i * n;
            int liN = li * n;
            for (int j = 0; j < n; j++)
            {
                cost += (long)flow[iN + j] * dist[liN + perm[j]];
            }
        }
        return cost;
    }

    // Swap delta for swapping facilities a and b (i.e., swapping their assigned locations perm[a] and perm[b])
    static long SwapDelta(int n, int[] flow, int[] dist, int[] perm, int a, int b)
    {
        if (a == b) return 0;
        int la = perm[a];
        int lb = perm[b];

        long delta = 0;

        int aN = a * n;
        int bN = b * n;
        int laN = la * n;
        int lbN = lb * n;

        // Base terms (a,a), (b,b), (a,b), (b,a)
        delta += (long)(flow[aN + a] - flow[bN + b]) * (dist[lbN + lb] - dist[laN + la]);
        delta += (long)(flow[aN + b] - flow[bN + a]) * (dist[lbN + la] - dist[laN + lb]);

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;

            int lk = perm[k];
            int kN = k * n;
            int lkN = lk * n;

            // (k,a)-(k,b) with (lk,lb)-(lk,la)
            delta += (long)(flow[kN + a] - flow[kN + b]) * (dist[lkN + lb] - dist[lkN + la]);
            // (a,k)-(b,k) with (lb,lk)-(la,lk)
            delta += (long)(flow[aN + k] - flow[bN + k]) * (dist[lbN + lk] - dist[laN + lk]);
        }

        return delta;
    }

    static void Swap(int[] perm, int a, int b)
    {
        int t = perm[a];
        perm[a] = perm[b];
        perm[b] = t;
    }

    static int[] InitialGreedy(int n, int[] flow, int[] dist)
    {
        long[] flowScore = new long[n];
        long[] distScore = new long[n];

        // flowScore[i] = sum_j (flow[i][j] + flow[j][i])
        for (int i = 0; i < n; i++)
        {
            long s = 0;
            int iN = i * n;
            for (int j = 0; j < n; j++)
            {
                s += flow[iN + j];
                s += flow[j * n + i];
            }
            flowScore[i] = s;
        }

        // distScore[l] = sum_j (dist[l][j] + dist[j][l])
        for (int l = 0; l < n; l++)
        {
            long s = 0;
            int lN = l * n;
            for (int j = 0; j < n; j++)
            {
                s += dist[lN + j];
                s += dist[j * n + l];
            }
            distScore[l] = s;
        }

        int[] fac = new int[n];
        int[] loc = new int[n];
        for (int i = 0; i < n; i++) { fac[i] = i; loc[i] = i; }

        Array.Sort(fac, (a, b) => flowScore[b].CompareTo(flowScore[a]));  // descending
        Array.Sort(loc, (a, b) => distScore[a].CompareTo(distScore[b]));  // ascending

        int[] perm = new int[n];
        for (int t = 0; t < n; t++)
            perm[fac[t]] = loc[t];

        return perm;
    }

    // Recompute a "hot list" of facilities with high current outgoing cost contribution, to bias search.
    static int[] RecomputeHotFacilities(int n, int[] flow, int[] dist, int[] perm, int hotCount)
    {
        long[] contrib = new long[n];
        for (int i = 0; i < n; i++)
        {
            int li = perm[i];
            int iN = i * n;
            int liN = li * n;
            long s = 0;
            for (int j = 0; j < n; j++)
                s += (long)flow[iN + j] * dist[liN + perm[j]];
            contrib[i] = s;
        }

        int[] idx = new int[n];
        for (int i = 0; i < n; i++) idx[i] = i;
        Array.Sort(idx, (a, b) => contrib[b].CompareTo(contrib[a])); // descending

        if (hotCount >= n) return idx;
        int[] hot = new int[hotCount];
        Array.Copy(idx, hot, hotCount);
        return hot;
    }

    // ---------- Main solver ----------
    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        if (n == int.MinValue || n <= 0)
        {
            // No valid instance
            return;
        }

        int nn = n * n;
        int[] flow = new int[nn];
        int[] dist = new int[nn];

        for (int i = 0; i < nn; i++) flow[i] = fs.NextInt();
        for (int i = 0; i < nn; i++) dist[i] = fs.NextInt();

        // Initial solution
        int[] perm = InitialGreedy(n, flow, dist);

        // Search parameters
        var rng = new Random(1);
        var sw = Stopwatch.StartNew();

        // Keep some time for final recompute + output
        const int maxMs = 290_000; // ~4m50s

        long cost = ComputeCost(n, flow, dist, perm);
        long bestCost = cost;
        int[] bestPerm = (int[])perm.Clone();

        int hotCount = Math.Min(n, 200);
        int[] hot = RecomputeHotFacilities(n, flow, dist, perm, hotCount);

        // K attempts per "step" (first improvement). Larger K for smaller n.
        int K = Math.Min(250, Math.Max(30, 5000 / Math.Max(1, n)));

        int stagnation = 0;
        long nextRefreshMs = 5000;

        while (sw.ElapsedMilliseconds < maxMs)
        {
            // Periodic refresh: recompute cost (eliminate drift), update best, update "hot" list.
            if (sw.ElapsedMilliseconds >= nextRefreshMs)
            {
                cost = ComputeCost(n, flow, dist, perm);
                if (cost < bestCost)
                {
                    bestCost = cost;
                    Array.Copy(perm, bestPerm, n);
                }
                hot = RecomputeHotFacilities(n, flow, dist, perm, hotCount);
                nextRefreshMs += 5000;
            }

            bool improved = false;

            for (int t = 0; t < K; t++)
            {
                int a;
                if (hot.Length > 0 && rng.NextDouble() < 0.70)
                    a = hot[rng.Next(hot.Length)];
                else
                    a = rng.Next(n);

                int b = rng.Next(n - 1);
                if (b >= a) b++;

                long delta = SwapDelta(n, flow, dist, perm, a, b);
                if (delta < 0)
                {
                    Swap(perm, a, b);
                    cost += delta;
                    improved = true;
                    stagnation = 0;

                    if (cost < bestCost)
                    {
                        bestCost = cost;
                        Array.Copy(perm, bestPerm, n);
                    }
                    break; // first improvement
                }
            }

            if (!improved)
            {
                stagnation++;

                // If stuck, apply a small perturbation (random swaps) to escape local minima.
                if (stagnation > 5000)
                {
                    int kicks = Math.Min(30, Math.Max(5, n / 100));
                    for (int t = 0; t < kicks; t++)
                    {
                        int a = rng.Next(n);
                        int b = rng.Next(n - 1);
                        if (b >= a) b++;

                        long delta = SwapDelta(n, flow, dist, perm, a, b);
                        Swap(perm, a, b);
                        cost += delta; // accept even if worse
                    }
                    stagnation = 0;
                }
            }
        }

        // Output the best found solution (recompute exact cost for safety)
        long finalCost = ComputeCost(n, flow, dist, bestPerm);

        var sb = new StringBuilder();
        sb.AppendLine(finalCost.ToString());
        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(bestPerm[i]);
        }
        sb.AppendLine();

        Console.Write(sb.ToString());
    }
}
```
