```rust
use std::collections::VecDeque;
use std::io::{self, BufRead, BufReader, BufWriter, Write};

const UP: u8 = 0;
const RIGHT: u8 = 1;
const DOWN: u8 = 2;
const LEFT: u8 = 3;

const VIS_A: u8 = 1 << 2; // visited from A side
const VIS_B: u8 = 1 << 5; // visited from B side

#[inline]
fn set_wall(bits: &mut [u64], idx: usize) {
    let w = idx >> 6;
    let b = idx & 63;
    bits[w] |= 1u64 << b;
}

#[inline]
fn is_wall(bits: &[u64], idx: usize) -> bool {
    let w = idx >> 6;
    let b = idx & 63;
    ((bits[w] >> b) & 1) != 0
}

#[inline]
fn set_a(cell: &mut u8, dir: u8) {
    // bits 0..1 = dirA, bit2 = visA
    *cell = (*cell & !0x03) | (dir & 0x03);
    *cell |= VIS_A;
}

#[inline]
fn set_b(cell: &mut u8, dir: u8) {
    // bits 3..4 = dirB, bit5 = visB
    *cell = (*cell & !(0x03 << 3)) | ((dir & 0x03) << 3);
    *cell |= VIS_B;
}

#[inline]
fn dir_a(cell: u8) -> u8 {
    cell & 0x03
}

#[inline]
fn dir_b(cell: u8) -> u8 {
    (cell >> 3) & 0x03
}

#[inline]
fn step_parent(idx: usize, dir: u8, w: usize) -> usize {
    match dir {
        UP => idx - w,
        RIGHT => idx + 1,
        DOWN => idx + w,
        LEFT => idx - 1,
        _ => unreachable!(),
    }
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());

    // Read first line: H W
    let mut first = String::new();
    if reader.read_line(&mut first)? == 0 {
        return Ok(());
    }
    let mut it = first.split_whitespace();
    let h: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => {
            // invalid input
            return Ok(());
        }
    };
    let w: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => {
            // invalid input
            return Ok(());
        }
    };

    let n = match h.checked_mul(w) {
        Some(v) => v,
        None => {
            // too large
            return Ok(());
        }
    };

    if n == 0 {
        let mut out = BufWriter::new(io::stdout().lock());
        writeln!(out, "0")?;
        return Ok(());
    }

    // We store indices as u32 for compact queues; ensure it fits.
    if n > (u32::MAX as usize) {
        // Not expected under given constraints; cannot index with u32 efficiently.
        let mut out = BufWriter::new(io::stdout().lock());
        writeln!(out, "0")?;
        return Ok(());
    }

    let wall_words = (n + 63) / 64;
    let mut walls = vec![0u64; wall_words];

    // parent/visited packed:
    // bits 0..1 : dirA (direction to parent in A-tree)
    // bit 2    : visitedA
    // bits 3..4: dirB (direction to parent in B-tree)
    // bit 5    : visitedB
    let mut mark = vec![0u8; n];

    let mut start: Option<u32> = None;
    let mut goal: Option<u32> = None;

    // Read H maze rows preserving trailing spaces
    let mut line_buf: Vec<u8> = Vec::with_capacity(w + 2);

    for y in 0..h {
        line_buf.clear();
        // read_until keeps all bytes including spaces; trims nothing
        let bytes_read = reader.read_until(b'\n', &mut line_buf)?;
        if bytes_read == 0 && y < h {
            // Unexpected EOF; treat missing cells as spaces
            // (we'll just parse what we have)
        }
        // Remove trailing '\n' and optional '\r'
        while matches!(line_buf.last(), Some(b'\n') | Some(b'\r')) {
            line_buf.pop();
        }

        for x in 0..w {
            let ch = *line_buf.get(x).unwrap_or(&b' ');
            let idx = y * w + x;
            match ch {
                b'#' => set_wall(&mut walls, idx),
                b'A' => start = Some(idx as u32),
                b'B' => goal = Some(idx as u32),
                b' ' => {}
                _ => {
                    // Treat any unknown as open.
                }
            }
        }
    }

    let (start, goal) = match (start, goal) {
        (Some(a), Some(b)) => (a, b),
        _ => {
            let mut out = BufWriter::new(io::stdout().lock());
            writeln!(out, "0")?;
            return Ok(());
        }
    };

    if start == goal {
        let mut out = BufWriter::new(io::stdout().lock());
        writeln!(out, "1")?;
        let sx = (start as usize) % w;
        let sy = (start as usize) / w;
        writeln!(out, "{} {}", sx, sy)?;
        return Ok(());
    }

    // Bidirectional BFS
    let mut qa: VecDeque<u32> = VecDeque::new();
    let mut qb: VecDeque<u32> = VecDeque::new();

    mark[start as usize] |= VIS_A;
    mark[goal as usize] |= VIS_B;
    qa.push_back(start);
    qb.push_back(goal);

    let mut meet: Option<u32> = None;

    'search: while !qa.is_empty() && !qb.is_empty() {
        // Expand the smaller frontier for better performance
        let expand_a = qa.len() <= qb.len();

        if expand_a {
            let cur = qa.pop_front().unwrap();
            let cur_u = cur as usize;

            if (mark[cur_u] & VIS_B) != 0 {
                meet = Some(cur);
                break;
            }

            let x = cur_u % w;

            // Up
            if cur_u >= w {
                let nei = cur_u - w;
                if !is_wall(&walls, nei) {
                    if (mark[nei] & VIS_A) == 0 {
                        set_a(&mut mark[nei], DOWN);
                        qa.push_back(nei as u32);
                    }
                    if (mark[nei] & VIS_B) != 0 {
                        meet = Some(nei as u32);
                        break 'search;
                    }
                }
            }
            // Down
            if cur_u + w < n {
                let nei = cur_u + w;
                if !is_wall(&walls, nei) {
                    if (mark[nei] & VIS_A) == 0 {
                        set_a(&mut mark[nei], UP);
                        qa.push_back(nei as u32);
                    }
                    if (mark[nei] & VIS_B) != 0 {
                        meet = Some(nei as u32);
                        break 'search;
                    }
                }
            }
            // Left
            if x != 0 {
                let nei = cur_u - 1;
                if !is_wall(&walls, nei) {
                    if (mark[nei] & VIS_A) == 0 {
                        set_a(&mut mark[nei], RIGHT);
                        qa.push_back(nei as u32);
                    }
                    if (mark[nei] & VIS_B) != 0 {
                        meet = Some(nei as u32);
                        break 'search;
                    }
                }
            }
            // Right
            if x + 1 < w {
                let nei = cur_u + 1;
                if !is_wall(&walls, nei) {
                    if (mark[nei] & VIS_A) == 0 {
                        set_a(&mut mark[nei], LEFT);
                        qa.push_back(nei as u32);
                    }
                    if (mark[nei] & VIS_B) != 0 {
                        meet = Some(nei as u32);
                        break 'search;
                    }
                }
            }
        } else {
            let cur = qb.pop_front().unwrap();
            let cur_u = cur as usize;

            if (mark[cur_u] & VIS_A) != 0 {
                meet = Some(cur);
                break;
            }

            let x = cur_u % w;

            // Up
            if cur_u >= w {
                let nei = cur_u - w;
                if !is_wall(&walls, nei) {
                    if (mark[nei] & VIS_B) == 0 {
                        set_b(&mut mark[nei], DOWN);
                        qb.push_back(nei as u32);
                    }
                    if (mark[nei] & VIS_A) != 0 {
                        meet = Some(nei as u32);
                        break 'search;
                    }
                }
            }
            // Down
            if cur_u + w < n {
                let nei = cur_u + w;
                if !is_wall(&walls, nei) {
                    if (mark[nei] & VIS_B) == 0 {
                        set_b(&mut mark[nei], UP);
                        qb.push_back(nei as u32);
                    }
                    if (mark[nei] & VIS_A) != 0 {
                        meet = Some(nei as u32);
                        break 'search;
                    }
                }
            }
            // Left
            if x != 0 {
                let nei = cur_u - 1;
                if !is_wall(&walls, nei) {
                    if (mark[nei] & VIS_B) == 0 {
                        set_b(&mut mark[nei], RIGHT);
                        qb.push_back(nei as u32);
                    }
                    if (mark[nei] & VIS_A) != 0 {
                        meet = Some(nei as u32);
                        break 'search;
                    }
                }
            }
            // Right
            if x + 1 < w {
                let nei = cur_u + 1;
                if !is_wall(&walls, nei) {
                    if (mark[nei] & VIS_B) == 0 {
                        set_b(&mut mark[nei], LEFT);
                        qb.push_back(nei as u32);
                    }
                    if (mark[nei] & VIS_A) != 0 {
                        meet = Some(nei as u32);
                        break 'search;
                    }
                }
            }
        }
    }

    let mut out = BufWriter::new(io::stdout().lock());

    let meet = match meet {
        Some(m) => m,
        None => {
            writeln!(out, "0")?;
            return Ok(());
        }
    };

    // Reconstruct A -> meet
    let mut a_rev: Vec<u32> = Vec::new();
    let mut cur = meet as usize;
    a_rev.push(cur as u32);
    while cur != start as usize {
        let cell = mark[cur];
        // If meet was found by touching from B side first, it should still have VIS_A set by connection step.
        if (cell & VIS_A) == 0 {
            // Should not happen, but avoid infinite loop.
            writeln!(out, "0")?;
            return Ok(());
        }
        let d = dir_a(cell);
        cur = step_parent(cur, d, w);
        a_rev.push(cur as u32);
    }
    a_rev.reverse(); // now start .. meet

    // Reconstruct meet -> B (goal)
    let mut b_fwd: Vec<u32> = Vec::new();
    cur = meet as usize;
    b_fwd.push(cur as u32);
    while cur != goal as usize {
        let cell = mark[cur];
        if (cell & VIS_B) == 0 {
            // Should not happen, but avoid infinite loop.
            writeln!(out, "0")?;
            return Ok(());
        }
        let d = dir_b(cell);
        cur = step_parent(cur, d, w);
        b_fwd.push(cur as u32);
    }

    // Merge, skipping duplicate meet
    let total_len = a_rev.len() + b_fwd.len().saturating_sub(1);
    writeln!(out, "{}", total_len)?;

    // Output coordinates as "x y" (x=column, y=row), 0-indexed from top-left.
    for &idx in &a_rev {
        let u = idx as usize;
        let x = u % w;
        let y = u / w;
        writeln!(out, "{} {}", x, y)?;
    }
    for &idx in b_fwd.iter().skip(1) {
        let u = idx as usize;
        let x = u % w;
        let y = u / w;
        writeln!(out, "{} {}", x, y)?;
    }

    Ok(())
}
```
