```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    float t = clamp(input.worldPos.y * 0.5f + 0.5f, 0.0f, 1.0f);
    float3 bottom = float3(0.1f, 0.4f, 0.1f);
    float3 top    = float3(1.0f, 1.0f, 1.0f);
    float3 rgb = lerp(bottom, top, t);
    return float4(rgb, 1.0f);
}
```
