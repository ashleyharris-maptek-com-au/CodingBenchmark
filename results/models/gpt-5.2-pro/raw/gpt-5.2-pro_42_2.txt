OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"

; Types

%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3_in

%e1 = OpFAdd %v4float %centerPos %off1
%e2 = OpFAdd %v4float %centerPos %off2
%e3 = OpFAdd %v4float %centerPos %off3
%e4 = OpFAdd %v4float %centerPos %off4
%e5 = OpFAdd %v4float %centerPos %off5

%ptr_out_pos = OpAccessChain %ptr_Output_v4 %gl_out %int_0

OpStore %outColor %centerColor
OpEmitVertex

OpStore %outColor %edgeColor

OpEndPrimitive
