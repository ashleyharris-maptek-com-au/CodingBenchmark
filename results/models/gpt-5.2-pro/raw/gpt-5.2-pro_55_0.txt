#include <iostream>

using namespace std;

static constexpr int INF = 1'000'000;
static constexpr int MATE = 30'000;
static constexpr int MAXPLY = 128;

enum Color { WHITE = 0, BLACK = 1 };
enum PieceType { EMPTY=0, PAWN=1, KNIGHT=2, BISHOP=3, ROOK=4, QUEEN=5, KING=6 };

static inline int fileOf(int sq) { return sq & 7; }
static inline int rankOf(int sq) { return sq >> 3; }
static inline int mirrorSq(int sq) { return (7 - rankOf(sq)) * 8 + fileOf(sq); }

static inline string sqToStr(int sq) {
    string s;
    s += char('a' + fileOf(sq));
    s += char('1' + rankOf(sq));
    return s;
}

static inline int strToSq(const string& s) {
    if (s.size() != 2) return -1;
    int f = s[0] - 'a', r = s[1] - '1';
    if (f < 0 || f > 7 || r < 0 || r > 7) return -1;
    return r * 8 + f;
}

static inline char pieceLetterForMove(int piece) {
    int pt = std::abs(piece);
    if (pt == PAWN) return 'p';
    if (pt == KNIGHT) return 'N';
    if (pt == BISHOP) return 'B';
    if (pt == ROOK) return 'R';
    if (pt == QUEEN) return 'Q';
    return 'K';
}

static inline char promoLetter(PieceType pt) {
    switch (pt) {
        case QUEEN: return 'Q';
        case ROOK: return 'R';
        case BISHOP: return 'B';
        case KNIGHT: return 'N';
        default: return 'Q';
    }
}

struct Move {
    uint8_t from = 0, to = 0;
    uint8_t promo = 0;      // 0 none, else PieceType
    uint8_t flags = 0;      // bits: 1 capture, 2 ep, 4 castle, 8 dblpush
};

static inline bool operator==(const Move& a, const Move& b) {
    return a.from==b.from && a.to==b.to && a.promo==b.promo && a.flags==b.flags;
}

struct Undo {
    int captured = 0;
    int castling = 0;
    int epSq = -1;
    int halfmove = 0;
    uint64_t hash = 0;
    int kingSqW = -1, kingSqB = -1;
};

struct Zobrist {
    uint64_t piece[12][64]{};
    uint64_t side{};
    uint64_t castling[16]{};
    uint64_t epFile[9]{}; // 0..7 file, 8 = none

    Zobrist() {
        std::mt19937_64 rng(0xC0FFEE123456789ULL);
        auto r64 = [&](){ return rng(); };
        for (int p=0;p<12;p++)
            for (int s=0;s<64;s++)
                piece[p][s] = r64();
        side = r64();
        for (int i=0;i<16;i++) castling[i] = r64();
        for (int i=0;i<9;i++) epFile[i] = r64();
    }

    static inline int pieceIndex(int piece) {
        // white: +1..+6 -> 0..5 ; black: -1..-6 -> 6..11
        int pt = std::abs(piece) - 1;
        int color = (piece > 0) ? 0 : 1;
        return color*6 + pt;
    }
};

struct TTEntry {
    uint64_t key = 0;
    int16_t depth = -1;
    int16_t score = 0;
    uint8_t flag = 0; // 0 exact, 1 alpha, 2 beta
    uint32_t movePack = 0;
};

static inline uint32_t packMove(const Move& m) {
    // from 6, to 6, promo 3, flags 8
    return (uint32_t)m.from | ((uint32_t)m.to << 6) | ((uint32_t)m.promo << 12) | ((uint32_t)m.flags << 15);
}
static inline Move unpackMove(uint32_t x) {
    Move m;
    m.from = (uint8_t)(x & 63);
    m.to = (uint8_t)((x >> 6) & 63);
    m.promo = (uint8_t)((x >> 12) & 7);
    m.flags = (uint8_t)((x >> 15) & 255);
    return m;
}

struct Position {
    array<int,64> b{};
    int sideToMove = WHITE;
    int castling = 0; // bits: 1 WK, 2 WQ, 4 BK, 8 BQ
    int epSq = -1;
    int halfmoveClock = 0;
    int fullmoveNumber = 1;
    int kingSq[2]{-1,-1};
    uint64_t hash = 0;

    Zobrist* Z = nullptr;

    void clear() {
        b.fill(0);
        sideToMove = WHITE;
        castling = 0;
        epSq = -1;
        halfmoveClock = 0;
        fullmoveNumber = 1;
        kingSq[0]=kingSq[1]=-1;
        hash = 0;
    }

    uint64_t computeHash() const {
        uint64_t h = 0;
        for (int sq=0;sq<64;sq++) {
            int pc = b[sq];
            if (pc != 0) h ^= Z->piece[Zobrist::pieceIndex(pc)][sq];
        }
        if (sideToMove == BLACK) h ^= Z->side;
        h ^= Z->castling[castling & 15];
        int epIdx = (epSq == -1) ? 8 : fileOf(epSq);
        h ^= Z->epFile[epIdx];
        return h;
    }

    bool parseFEN(const string& fen) {
        clear();
        istringstream iss(fen);
        string placement, stm, cast, ep;
        if (!(iss >> placement >> stm >> cast >> ep)) return false;
        iss >> halfmoveClock >> fullmoveNumber;

        int r = 7, f = 0;
        for (char c : placement) {
            if (c == '/') { r--; f = 0; continue; }
            if (c >= '1' && c <= '8') { f += (c - '0'); continue; }
            if (f > 7 || r < 0) return false;
            int sq = r*8 + f;
            int piece = 0;
            switch (c) {
                case 'P': piece = +PAWN; break;
                case 'N': piece = +KNIGHT; break;
                case 'B': piece = +BISHOP; break;
                case 'R': piece = +ROOK; break;
                case 'Q': piece = +QUEEN; break;
                case 'K': piece = +KING; break;
                case 'p': piece = -PAWN; break;
                case 'n': piece = -KNIGHT; break;
                case 'b': piece = -BISHOP; break;
                case 'r': piece = -ROOK; break;
                case 'q': piece = -QUEEN; break;
                case 'k': piece = -KING; break;
                default: return false;
            }
            b[sq] = piece;
            if (std::abs(piece) == KING) {
                kingSq[piece > 0 ? WHITE : BLACK] = sq;
            }
            f++;
        }

        sideToMove = (stm == "w") ? WHITE : BLACK;

        castling = 0;
        if (cast != "-") {
            for (char c : cast) {
                if (c == 'K') castling |= 1;
                else if (c == 'Q') castling |= 2;
                else if (c == 'k') castling |= 4;
                else if (c == 'q') castling |= 8;
            }
        }

        if (ep == "-") epSq = -1;
        else epSq = strToSq(ep);

        if (kingSq[WHITE] < 0 || kingSq[BLACK] < 0) return false;
        hash = computeHash();
        return true;
    }
};

struct Engine {
    Zobrist Z;
    Position pos;

    vector<uint64_t> repStack; // hash history for repetition

    vector<TTEntry> tt;
    uint64_t ttMask = 0;

    array<array<int,MAXPLY>,2> killer{};
    array<array<int,64>,2> history{}; // [color][toSq] small/simple

    chrono::steady_clock::time_point endTime;
    bool stop = false;
    uint64_t nodes = 0;

    Engine() {
        pos.Z = &Z;
        // TT size ~ 2^21 entries (~2 million) reasonable memory
        uint64_t sizePow2 = 1ULL << 21;
        tt.resize((size_t)sizePow2);
        ttMask = sizePow2 - 1;
        for (auto& e : tt) e = TTEntry{};
    }

    bool timeUp() {
        return chrono::steady_clock::now() >= endTime;
    }

    static inline int pieceColor(int pc) { return pc > 0 ? WHITE : BLACK; }

    bool squareAttacked(const Position& p, int sq, int byColor) const {
        // pawns
        if (byColor == WHITE) {
            int r = rankOf(sq), f = fileOf(sq);
            if (r > 0) {
                if (f > 0 && p.b[sq - 9] == +PAWN) return true;
                if (f < 7 && p.b[sq - 7] == +PAWN) return true;
            }
        } else {
            int r = rankOf(sq), f = fileOf(sq);
            if (r < 7) {
                if (f > 0 && p.b[sq + 7] == -PAWN) return true;
                if (f < 7 && p.b[sq + 9] == -PAWN) return true;
            }
        }

        // knights
        static const int kOff[8] = {+17,+15,+10,+6,-17,-15,-10,-6};
        for (int d : kOff) {
            int s2 = sq + d;
            if (s2 < 0 || s2 >= 64) continue;
            int f1 = fileOf(sq), f2 = fileOf(s2);
            int df = std::abs(f2 - f1);
            if (df != 1 && df != 2) continue;
            int pc = p.b[s2];
            if (pc == 0) continue;
            if (pieceColor(pc) == byColor && std::abs(pc) == KNIGHT) return true;
        }

        // king
        static const int kingOff[8] = {+1,-1,+8,-8,+9,+7,-7,-9};
        for (int d : kingOff) {
            int s2 = sq + d;
            if (s2 < 0 || s2 >= 64) continue;
            int f1=fileOf(sq), f2=fileOf(s2);
            if (std::abs(f2-f1) > 1) continue;
            int pc = p.b[s2];
            if (pc && pieceColor(pc) == byColor && std::abs(pc) == KING) return true;
        }

        // sliders
        auto ray = [&](int step, int want1, int want2)->bool{
            int s = sq;
            while (true) {
                int s2 = s + step;
                if (s2 < 0 || s2 >= 64) return false;
                int f1=fileOf(s), f2=fileOf(s2);
                if (std::abs(f2 - f1) > 1 && (step==+1 || step==-1 || step==+9 || step==-9 || step==+7 || step==-7))
                    return false;
                // handle wrap for orthogonal too
                if ((step==+1 || step==-1 || step==+9 || step==-9 || step==+7 || step==-7) && std::abs(f2-f1) != 1 && (step==+1 || step==-1 || step==+9 || step==-9 || step==+7 || step==-7))
                    ; // already ok
                if ((step==+1 || step==-1) && rankOf(s2)!=rankOf(s)) return false;

                s = s2;
                int pc = p.b[s];
                if (!pc) continue;
                if (pieceColor(pc) != byColor) return false;
                int pt = std::abs(pc);
                return (pt == want1 || pt == want2);
            }
        };

        // rook/queen
        if (ray(+1, ROOK, QUEEN)) return true;
        if (ray(-1, ROOK, QUEEN)) return true;
        if (ray(+8, ROOK, QUEEN)) return true;
        if (ray(-8, ROOK, QUEEN)) return true;
        // bishop/queen
        if (ray(+9, BISHOP, QUEEN)) return true;
        if (ray(-9, BISHOP, QUEEN)) return true;
        if (ray(+7, BISHOP, QUEEN)) return true;
        if (ray(-7, BISHOP, QUEEN)) return true;

        return false;
    }

    bool inCheck(const Position& p, int color) const {
        return squareAttacked(p, p.kingSq[color], color ^ 1);
    }

    void genPseudoMoves(const Position& p, vector<Move>& moves) const {
        moves.clear();
        int us = p.sideToMove;
        int them = us ^ 1;

        auto addMove = [&](int from, int to, bool cap, uint8_t promo=0, uint8_t extraFlags=0) {
            Move m;
            m.from = (uint8_t)from;
            m.to = (uint8_t)to;
            m.promo = promo;
            m.flags = (cap ? 1 : 0) | extraFlags;
            moves.push_back(m);
        };

        for (int from = 0; from < 64; from++) {
            int pc = p.b[from];
            if (!pc) continue;
            if (pieceColor(pc) != us) continue;
            int pt = std::abs(pc);

            if (pt == PAWN) {
                int r = rankOf(from), f = fileOf(from);
                int dir = (us == WHITE) ? +8 : -8;
                int startRank = (us == WHITE) ? 1 : 6;
                int promoRank = (us == WHITE) ? 6 : 1; // from rank before last
                int lastRank = (us == WHITE) ? 7 : 0;

                int one = from + dir;
                if (one >= 0 && one < 64 && p.b[one] == 0) {
                    if (r == promoRank) {
                        addMove(from, one, false, QUEEN);
                        addMove(from, one, false, ROOK);
                        addMove(from, one, false, BISHOP);
                        addMove(from, one, false, KNIGHT);
                    } else {
                        addMove(from, one, false);
                        // double
                        int two = from + 2*dir;
                        if (r == startRank && two >= 0 && two < 64 && p.b[two] == 0) {
                            addMove(from, two, false, 0, 8);
                        }
                    }
                }
                // captures
                int capL = (us == WHITE) ? from + 7 : from - 9;
                int capR = (us == WHITE) ? from + 9 : from - 7;
                if (f > 0 && capL >= 0 && capL < 64) {
                    if (p.b[capL] && pieceColor(p.b[capL]) == them) {
                        if (r == promoRank) {
                            addMove(from, capL, true, QUEEN);
                            addMove(from, capL, true, ROOK);
                            addMove(from, capL, true, BISHOP);
                            addMove(from, capL, true, KNIGHT);
                        } else addMove(from, capL, true);
                    }
                    if (p.epSq == capL) {
                        addMove(from, capL, true, 0, 2);
                    }
                }
                if (f < 7 && capR >= 0 && capR < 64) {
                    if (p.b[capR] && pieceColor(p.b[capR]) == them) {
                        if (r == promoRank) {
                            addMove(from, capR, true, QUEEN);
                            addMove(from, capR, true, ROOK);
                            addMove(from, capR, true, BISHOP);
                            addMove(from, capR, true, KNIGHT);
                        } else addMove(from, capR, true);
                    }
                    if (p.epSq == capR) {
                        addMove(from, capR, true, 0, 2);
                    }
                }
            } else if (pt == KNIGHT) {
                static const int off[8] = {+17,+15,+10,+6,-17,-15,-10,-6};
                for (int d : off) {
                    int to = from + d;
                    if (to < 0 || to >= 64) continue;
                    int df = std::abs(fileOf(to) - fileOf(from));
                    if (df != 1 && df != 2) continue;
                    int occ = p.b[to];
                    if (!occ) addMove(from, to, false);
                    else if (pieceColor(occ) == them) addMove(from, to, true);
                }
            } else if (pt == BISHOP || pt == ROOK || pt == QUEEN) {
                static const int bishopDirs[4] = {+9,+7,-7,-9};
                static const int rookDirs[4] = {+1,-1,+8,-8};

                auto slide = [&](int step) {
                    int s = from;
                    while (true) {
                        int to = s + step;
                        if (to < 0 || to >= 64) break;
                        if ((step == +1 || step == -1) && rankOf(to) != rankOf(s)) break;
                        if ((step == +9 || step == -9 || step == +7 || step == -7) &&
                            std::abs(fileOf(to) - fileOf(s)) != 1) break;

                        int occ = p.b[to];
                        if (!occ) {
                            addMove(from, to, false);
                            s = to;
                            continue;
                        } else {
                            if (pieceColor(occ) == them) addMove(from, to, true);
                            break;
                        }
                    }
                };

                if (pt == BISHOP || pt == QUEEN) for (int d : bishopDirs) slide(d);
                if (pt == ROOK || pt == QUEEN) for (int d : rookDirs) slide(d);

            } else if (pt == KING) {
                static const int off[8] = {+1,-1,+8,-8,+9,+7,-7,-9};
                for (int d : off) {
                    int to = from + d;
                    if (to < 0 || to >= 64) continue;
                    if (std::abs(fileOf(to) - fileOf(from)) > 1) continue;
                    int occ = p.b[to];
                    if (!occ) addMove(from, to, false);
                    else if (pieceColor(occ) == them) addMove(from, to, true);
                }

                // castling
                if (us == WHITE) {
                    if ((p.castling & 1) && p.b[strToSq("f1")] == 0 && p.b[strToSq("g1")] == 0) {
                        // squares not attacked: e1 f1 g1
                        int e1 = strToSq("e1"), f1 = strToSq("f1"), g1 = strToSq("g1");
                        if (!squareAttacked(p, e1, them) && !squareAttacked(p, f1, them) && !squareAttacked(p, g1, them))
                            addMove(e1, g1, false, 0, 4);
                    }
                    if ((p.castling & 2) && p.b[strToSq("d1")] == 0 && p.b[strToSq("c1")] == 0 && p.b[strToSq("b1")] == 0) {
                        int e1 = strToSq("e1"), d1 = strToSq("d1"), c1 = strToSq("c1");
                        if (!squareAttacked(p, e1, them) && !squareAttacked(p, d1, them) && !squareAttacked(p, c1, them))
                            addMove(e1, c1, false, 0, 4);
                    }
                } else {
                    if ((p.castling & 4) && p.b[strToSq("f8")] == 0 && p.b[strToSq("g8")] == 0) {
                        int e8 = strToSq("e8"), f8 = strToSq("f8"), g8 = strToSq("g8");
                        if (!squareAttacked(p, e8, them) && !squareAttacked(p, f8, them) && !squareAttacked(p, g8, them))
                            addMove(e8, g8, false, 0, 4);
                    }
                    if ((p.castling & 8) && p.b[strToSq("d8")] == 0 && p.b[strToSq("c8")] == 0 && p.b[strToSq("b8")] == 0) {
                        int e8 = strToSq("e8"), d8 = strToSq("d8"), c8 = strToSq("c8");
                        if (!squareAttacked(p, e8, them) && !squareAttacked(p, d8, them) && !squareAttacked(p, c8, them))
                            addMove(e8, c8, false, 0, 4);
                    }
                }
            }
        }
    }

    bool makeMove(Position& p, const Move& m, Undo& u) {
        u.captured = 0;
        u.castling = p.castling;
        u.epSq = p.epSq;
        u.halfmove = p.halfmoveClock;
        u.hash = p.hash;
        u.kingSqW = p.kingSq[WHITE];
        u.kingSqB = p.kingSq[BLACK];

        int us = p.sideToMove;
        int them = us ^ 1;

        int from = m.from, to = m.to;
        int pc = p.b[from];
        if (!pc) return false;

        bool isCapture = (m.flags & 1) != 0;
        bool isEP = (m.flags & 2) != 0;
        bool isCastle = (m.flags & 4) != 0;
        bool isDouble = (m.flags & 8) != 0;

        // hash remove ep/castling/side
        p.hash ^= Z.epFile[(p.epSq == -1) ? 8 : fileOf(p.epSq)];
        p.hash ^= Z.castling[p.castling & 15];
        if (p.sideToMove == BLACK) p.hash ^= Z.side;

        // reset ep
        p.epSq = -1;

        // halfmove
        if (std::abs(pc) == PAWN || isCapture) p.halfmoveClock = 0;
        else p.halfmoveClock++;

        // capture handling
        int capturedSq = to;
        if (isEP) {
            capturedSq = (us == WHITE) ? (to - 8) : (to + 8);
        }
        int captured = p.b[capturedSq];
        if (isCapture && captured == 0 && !isEP) {
            // inconsistent flags
            return false;
        }
        if (isEP) {
            if (captured == 0 || std::abs(captured) != PAWN || pieceColor(captured) != them) return false;
        }
        u.captured = captured;

        // update hash: remove moving piece from 'from'
        p.hash ^= Z.piece[Zobrist::pieceIndex(pc)][from];
        if (captured) p.hash ^= Z.piece[Zobrist::pieceIndex(captured)][capturedSq];

        // move piece on board
        p.b[from] = 0;
        if (isEP) p.b[capturedSq] = 0;

        int placed = pc;

        // promotion
        if (m.promo && std::abs(pc) == PAWN) {
            int promotedType = m.promo;
            placed = (us == WHITE) ? promotedType : -promotedType;
        }

        // castling rook move
        if (isCastle) {
            // king moves e-file to g/c
            if (us == WHITE) {
                if (to == strToSq("g1")) {
                    // rook h1->f1
                    int rh1 = strToSq("h1"), rf1 = strToSq("f1");
                    int rook = p.b[rh1];
                    p.b[rh1] = 0;
                    p.b[rf1] = rook;
                    p.hash ^= Z.piece[Zobrist::pieceIndex(rook)][rh1];
                    p.hash ^= Z.piece[Zobrist::pieceIndex(rook)][rf1];
                } else if (to == strToSq("c1")) {
                    int ra1 = strToSq("a1"), rd1 = strToSq("d1");
                    int rook = p.b[ra1];
                    p.b[ra1] = 0;
                    p.b[rd1] = rook;
                    p.hash ^= Z.piece[Zobrist::pieceIndex(rook)][ra1];
                    p.hash ^= Z.piece[Zobrist::pieceIndex(rook)][rd1];
                }
            } else {
                if (to == strToSq("g8")) {
                    int rh8 = strToSq("h8"), rf8 = strToSq("f8");
                    int rook = p.b[rh8];
                    p.b[rh8] = 0;
                    p.b[rf8] = rook;
                    p.hash ^= Z.piece[Zobrist::pieceIndex(rook)][rh8];
                    p.hash ^= Z.piece[Zobrist::pieceIndex(rook)][rf8];
                } else if (to == strToSq("c8")) {
                    int ra8 = strToSq("a8"), rd8 = strToSq("d8");
                    int rook = p.b[ra8];
                    p.b[ra8] = 0;
                    p.b[rd8] = rook;
                    p.hash ^= Z.piece[Zobrist::pieceIndex(rook)][ra8];
                    p.hash ^= Z.piece[Zobrist::pieceIndex(rook)][rd8];
                }
            }
        }

        p.b[to] = placed;
        p.hash ^= Z.piece[Zobrist::pieceIndex(placed)][to];

        // set ep square after double pawn push
        if (isDouble && std::abs(pc) == PAWN) {
            p.epSq = (us == WHITE) ? (to - 8) : (to + 8);
        }

        // update king sq
        if (std::abs(placed) == KING) {
            p.kingSq[us] = to;
        }

        // update castling rights
        auto clearCastling = [&](int mask) { p.castling &= ~mask; };

        // king move clears both sides
        if (std::abs(pc) == KING) {
            if (us == WHITE) clearCastling(1|2);
            else clearCastling(4|8);
        }
        // rook move clears side
        if (std::abs(pc) == ROOK) {
            if (from == strToSq("h1")) clearCastling(1);
            if (from == strToSq("a1")) clearCastling(2);
            if (from == strToSq("h8")) clearCastling(4);
            if (from == strToSq("a8")) clearCastling(8);
        }
        // rook captured clears opponent side
        if (captured && std::abs(captured) == ROOK) {
            if (capturedSq == strToSq("h1")) clearCastling(1);
            if (capturedSq == strToSq("a1")) clearCastling(2);
            if (capturedSq == strToSq("h8")) clearCastling(4);
            if (capturedSq == strToSq("a8")) clearCastling(8);
        }

        // hash add new castling/ep/side
        p.hash ^= Z.castling[p.castling & 15];
        p.hash ^= Z.epFile[(p.epSq == -1) ? 8 : fileOf(p.epSq)];
        p.sideToMove ^= 1;
        if (p.sideToMove == BLACK) p.hash ^= Z.side;

        // legality: our king not in check
        if (inCheck(p, us)) {
            unmakeMove(p, m, u);
            return false;
        }
        return true;
    }

    void unmakeMove(Position& p, const Move& m, const Undo& u) {
        // simplest: restore everything except board needs reconstruct? We'll restore board by reverse move,
        // but also we saved hash/king/castling/ep/halfmove so can safely just reverse board state.
        // We'll do explicit reverse for board, then restore state fields from Undo for safety.
        int us = p.sideToMove ^ 1; // side who made move
        int them = us ^ 1;
        (void)them;

        int from = m.from, to = m.to;
        bool isCapture = (m.flags & 1) != 0;
        bool isEP = (m.flags & 2) != 0;
        bool isCastle = (m.flags & 4) != 0;

        // revert side to move first to interpret pieces
        p.sideToMove ^= 1;

        int movedPiece = p.b[to];
        p.b[to] = 0;

        // undo castling rook
        if (isCastle) {
            if (us == WHITE) {
                if (to == strToSq("g1")) {
                    int rh1 = strToSq("h1"), rf1 = strToSq("f1");
                    p.b[rh1] = p.b[rf1];
                    p.b[rf1] = 0;
                } else if (to == strToSq("c1")) {
                    int ra1 = strToSq("a1"), rd1 = strToSq("d1");
                    p.b[ra1] = p.b[rd1];
                    p.b[rd1] = 0;
                }
            } else {
                if (to == strToSq("g8")) {
                    int rh8 = strToSq("h8"), rf8 = strToSq("f8");
                    p.b[rh8] = p.b[rf8];
                    p.b[rf8] = 0;
                } else if (to == strToSq("c8")) {
                    int ra8 = strToSq("a8"), rd8 = strToSq("d8");
                    p.b[ra8] = p.b[rd8];
                    p.b[rd8] = 0;
                }
            }
        }

        // undo promotion: movedPiece may be promoted piece, restore pawn
        int origPiece = movedPiece;
        if (m.promo) {
            origPiece = (us == WHITE) ? +PAWN : -PAWN;
        }
        p.b[from] = origPiece;

        // restore captured
        if (isCapture) {
            int capSq = to;
            if (isEP) capSq = (us == WHITE) ? (to - 8) : (to + 8);
            p.b[capSq] = u.captured;
        }

        // restore state
        p.castling = u.castling;
        p.epSq = u.epSq;
        p.halfmoveClock = u.halfmove;
        p.hash = u.hash;
        p.kingSq[WHITE] = u.kingSqW;
        p.kingSq[BLACK] = u.kingSqB;
    }

    bool isRepetition(uint64_t h) const {
        int count = 0;
        for (int i = (int)repStack.size() - 1; i >= 0; --i) {
            if (repStack[i] == h) {
                count++;
                if (count >= 2) return true; // third occurrence overall incl current
            }
        }
        return false;
    }

    int eval(const Position& p) const {
        // material + simple piece-square tables (white perspective)
        static const int val[7] = {0, 100, 320, 330, 500, 900, 0};

        static const int pstPawn[64] = {
             0,  0,  0,  0,  0,  0,  0,  0,
             5, 10, 10,-10,-10, 10, 10,  5,
             5, -5,-10,  0,  0,-10, -5,  5,
             0,  0,  0, 20, 20,  0,  0,  0,
             5,  5, 10, 25, 25, 10,  5,  5,
            10, 10, 20, 30, 30, 20, 10, 10,
            50, 50, 50, 50, 50, 50, 50, 50,
             0,  0,  0,  0,  0,  0,  0,  0
        };
        static const int pstKnight[64] = {
            -50,-40,-30,-30,-30,-30,-40,-50,
            -40,-20,  0,  0,  0,  0,-20,-40,
            -30,  0, 10, 15, 15, 10,  0,-30,
            -30,  5, 15, 20, 20, 15,  5,-30,
            -30,  0, 15, 20, 20, 15,  0,-30,
            -30,  5, 10, 15, 15, 10,  5,-30,
            -40,-20,  0,  5,  5,  0,-20,-40,
            -50,-40,-30,-30,-30,-30,-40,-50
        };
        static const int pstBishop[64] = {
            -20,-10,-10,-10,-10,-10,-10,-20,
            -10,  0,  0,  0,  0,  0,  0,-10,
            -10,  0,  5, 10, 10,  5,  0,-10,
            -10,  5,  5, 10, 10,  5,  5,-10,
            -10,  0, 10, 10, 10, 10,  0,-10,
            -10, 10, 10, 10, 10, 10, 10,-10,
            -10,  5,  0,  0,  0,  0,  5,-10,
            -20,-10,-10,-10,-10,-10,-10,-20
        };
        static const int pstRook[64] = {
              0,  0,  0,  5,  5,  0,  0,  0,
             -5,  0,  0,  0,  0,  0,  0, -5,
             -5,  0,  0,  0,  0,  0,  0, -5,
             -5,  0,  0,  0,  0,  0,  0, -5,
             -5,  0,  0,  0,  0,  0,  0, -5,
             -5,  0,  0,  0,  0,  0,  0, -5,
              5, 10, 10, 10, 10, 10, 10,  5,
              0,  0,  0,  0,  0,  0,  0,  0
        };
        static const int pstQueen[64] = {
            -20,-10,-10, -5, -5,-10,-10,-20,
            -10,  0,  0,  0,  0,  0,  0,-10,
            -10,  0,  5,  5,  5,  5,  0,-10,
             -5,  0,  5,  5,  5,  5,  0, -5,
              0,  0,  5,  5,  5,  5,  0, -5,
            -10,  5,  5,  5,  5,  5,  0,-10,
            -10,  0,  5,  0,  0,  0,  0,-10,
            -20,-10,-10, -5, -5,-10,-10,-20
        };
        static const int pstKing[64] = {
            -30,-40,-40,-50,-50,-40,-40,-30,
            -30,-40,-40,-50,-50,-40,-40,-30,
            -30,-40,-40,-50,-50,-40,-40,-30,
            -30,-40,-40,-50,-50,-40,-40,-30,
            -20,-30,-30,-40,-40,-30,-30,-20,
            -10,-20,-20,-20,-20,-20,-20,-10,
             20, 20,  0,  0,  0,  0, 20, 20,
             20, 30, 10,  0,  0, 10, 30, 20
        };

        auto pst = [&](int pt, int sq, int color)->int{
            int s = (color == WHITE) ? sq : mirrorSq(sq);
            switch (pt) {
                case PAWN: return pstPawn[s];
                case KNIGHT: return pstKnight[s];
                case BISHOP: return pstBishop[s];
                case ROOK: return pstRook[s];
                case QUEEN: return pstQueen[s];
                case KING: return pstKing[s];
            }
            return 0;
        };

        int score = 0;
        for (int sq=0; sq<64; sq++) {
            int pc = p.b[sq];
            if (!pc) continue;
            int c = (pc > 0) ? WHITE : BLACK;
            int pt = std::abs(pc);
            int s = val[pt] + pst(pt, sq, c);
            score += (c == WHITE) ? s : -s;
        }
        return (p.sideToMove == WHITE) ? score : -score; // negamax view
    }

    TTEntry* ttProbe(uint64_t key) {
        TTEntry& e = tt[(size_t)(key & ttMask)];
        if (e.key == key) return &e;
        return nullptr;
    }
    void ttStore(uint64_t key, int depth, int score, int flag, const Move& best) {
        TTEntry& e = tt[(size_t)(key & ttMask)];
        // replacement: always replace if deeper or empty/different
        if (e.key != key || depth >= e.depth) {
            e.key = key;
            e.depth = (int16_t)depth;
            e.score = (int16_t)std::clamp(score, -32767, 32767);
            e.flag = (uint8_t)flag;
            e.movePack = packMove(best);
        }
    }

    static inline int mvvLvaScore(const Position& p, const Move& m) {
        if ((m.flags & 1) == 0) return 0;
        int to = m.to;
        int from = m.from;
        int victim = p.b[to];
        if (m.flags & 2) {
            // ep victim is pawn
            victim = (p.sideToMove == WHITE) ? -PAWN : +PAWN;
        }
        int attacker = p.b[from];
        int v = std::abs(victim);
        int a = std::abs(attacker);
        static const int vv[7] = {0,100,320,330,500,900,20000};
        return 10'000 + (vv[v] * 10 - vv[a]);
    }

    int quiescence(Position& p, int alpha, int beta, int ply) {
        if ((nodes++ & 2047ULL) == 0) {
            if (timeUp()) { stop = true; return alpha; }
        }
        int stand = eval(p);
        if (stand >= beta) return beta;
        if (stand > alpha) alpha = stand;

        vector<Move> moves;
        genPseudoMoves(p, moves);

        // only captures/promotions in qsearch
        vector<pair<int, Move>> caps;
        caps.reserve(moves.size());
        for (auto& m : moves) {
            if ((m.flags & 1) || m.promo) {
                caps.push_back({mvvLvaScore(p, m) + (m.promo ? 5000 : 0), m});
            }
        }
        sort(caps.begin(), caps.end(), [](auto& a, auto& b){ return a.first > b.first; });

        for (auto& it : caps) {
            Move m = it.second;
            Undo u;
            if (!makeMove(p, m, u)) continue;
            repStack.push_back(p.hash);

            int score = -quiescence(p, -beta, -alpha, ply+1);

            repStack.pop_back();
            unmakeMove(p, m, u);

            if (stop) return alpha;
            if (score >= beta) return beta;
            if (score > alpha) alpha = score;
        }
        return alpha;
    }

    int search(Position& p, int depth, int alpha, int beta, int ply, Move& pvMoveOut) {
        pvMoveOut = Move{};
        if ((nodes++ & 2047ULL) == 0) {
            if (timeUp()) { stop = true; return alpha; }
        }

        if (p.halfmoveClock >= 100) return 0;
        if (isRepetition(p.hash)) return 0;

        int alphaOrig = alpha;

        // TT
        Move ttMove{};
        if (TTEntry* e = ttProbe(p.hash)) {
            if (e->depth >= depth) {
                int sc = e->score;
                if (e->flag == 0) return sc;
                if (e->flag == 1) alpha = max(alpha, sc);
                else if (e->flag == 2) beta = min(beta, sc);
                if (alpha >= beta) return sc;
            }
            ttMove = unpackMove(e->movePack);
        }

        if (depth <= 0) {
            return quiescence(p, alpha, beta, ply);
        }

        vector<Move> moves;
        genPseudoMoves(p, moves);

        // order moves: TT move first, then captures, then killers/history
        vector<pair<int, Move>> ordered;
        ordered.reserve(moves.size());
        for (auto& m : moves) {
            int score = 0;
            if (m == ttMove) score += 1'000'000;
            score += mvvLvaScore(p, m);
            if ((m.flags & 4) != 0) score += 50;
            if (m.promo) score += 9000 + (int)m.promo;
            // killers/history for quiet moves
            if ((m.flags & 1) == 0 && !m.promo) {
                if (killer[p.sideToMove][ply] == (int)packMove(m)) score += 4000;
                score += history[p.sideToMove][m.to];
            }
            ordered.push_back({score, m});
        }
        sort(ordered.begin(), ordered.end(), [](auto& a, auto& b){ return a.first > b.first; });

        bool foundLegal = false;
        Move bestMove{};
        int bestScore = -INF;

        for (auto& it : ordered) {
            Move m = it.second;
            Undo u;
            if (!makeMove(p, m, u)) continue;
            foundLegal = true;

            repStack.push_back(p.hash);

            Move childPV{};
            int score = -search(p, depth - 1, -beta, -alpha, ply + 1, childPV);

            repStack.pop_back();
            unmakeMove(p, m, u);

            if (stop) return alpha;

            if (score > bestScore) {
                bestScore = score;
                bestMove = m;
                pvMoveOut = m;
            }

            if (score > alpha) {
                alpha = score;
                pvMoveOut = m;
                // update history on quiet move
                if ((m.flags & 1) == 0 && !m.promo && (m.flags & 4) == 0) {
                    history[p.sideToMove][m.to] += depth * depth;
                    history[p.sideToMove][m.to] = min(history[p.sideToMove][m.to], 200000);
                }
            }
            if (alpha >= beta) {
                // beta cutoff: update killer for quiet
                if ((m.flags & 1) == 0 && !m.promo) {
                    killer[p.sideToMove][ply] = (int)packMove(m);
                }
                break;
            }
        }

        if (!foundLegal) {
            if (inCheck(p, p.sideToMove)) return -MATE + ply; // checkmated
            return 0; // stalemate
        }

        // store TT
        int flag = 0;
        if (bestScore <= alphaOrig) flag = 1;      // alpha
        else if (bestScore >= beta) flag = 2;      // beta
        else flag = 0;                              // exact
        ttStore(p.hash, depth, bestScore, flag, bestMove);

        return bestScore;
    }

    int searchRoot(Position& p, int depth, Move& bestMove) {
        vector<Move> moves;
        genPseudoMoves(p, moves);

        // initial best
        bestMove = Move{};
        int bestScore = -INF;

        // order root: try TT best first if present
        Move ttMove{};
        if (TTEntry* e = ttProbe(p.hash)) ttMove = unpackMove(e->movePack);

        vector<pair<int, Move>> ordered;
        ordered.reserve(moves.size());
        for (auto& m : moves) {
            int sc = 0;
            if (m == ttMove) sc += 1'000'000;
            sc += mvvLvaScore(p, m);
            if (m.promo) sc += 9000 + (int)m.promo;
            if ((m.flags & 4) != 0) sc += 50;
            ordered.push_back({sc, m});
        }
        sort(ordered.begin(), ordered.end(), [](auto& a, auto& b){ return a.first > b.first; });

        int alpha = -INF, beta = INF;

        bool found = false;
        for (auto& it : ordered) {
            Move m = it.second;
            Undo u;
            if (!makeMove(p, m, u)) continue;
            found = true;

            repStack.push_back(p.hash);

            Move pv{};
            int score = -search(p, depth - 1, -beta, -alpha, 1, pv);

            repStack.pop_back();
            unmakeMove(p, m, u);

            if (stop) break;

            if (score > bestScore) {
                bestScore = score;
                bestMove = m;
            }
            alpha = max(alpha, score);
        }

        if (!found) return -MATE;
        return bestScore;
    }

    string moveToNotation(Position& p, const Move& m) {
        // castling special output
        if (m.flags & 4) {
            // to file g => O-O else O-O-O
            return (fileOf(m.to) == 6) ? "O-O" : "O-O-O";
        }

        int pc = p.b[m.from];
        char pl = pieceLetterForMove(pc);

        bool isCapture = (m.flags & 1) != 0;
        bool isEP = (m.flags & 2) != 0;

        string out;
        out += pl;
        out += sqToStr(m.from);
        out += (isCapture ? "x" : "");
        out += sqToStr(m.to);
        if (m.promo) {
            out += "/";
            out += promoLetter((PieceType)m.promo);
        }
        if (isEP) out += "ep";

        // add check/mate
        Undo u;
        if (makeMove(p, m, u)) {
            bool chk = inCheck(p, p.sideToMove); // now opponent to move
            if (chk) {
                // determine if mate: opponent has any legal moves
                vector<Move> replies;
                genPseudoMoves(p, replies);
                bool hasLegal = false;
                for (auto& r : replies) {
                    Undo u2;
                    if (makeMove(p, r, u2)) {
                        hasLegal = true;
                        unmakeMove(p, r, u2);
                        break;
                    }
                }
                out += hasLegal ? "+" : "#";
            }
            unmakeMove(p, m, u);
        }
        return out;
    }

    Move think(double timeoutSeconds) {
        stop = false;
        nodes = 0;

        // time management
        auto now = chrono::steady_clock::now();
        auto ms = (int64_t)(timeoutSeconds * 1000.0);
        // keep a small buffer
        ms = (ms > 50) ? (ms - 30) : ms;
        endTime = now + chrono::milliseconds(ms);

        repStack.clear();
        repStack.push_back(pos.hash);

        Move best{};
        Move lastBest{};
        int maxDepth = 64;

        for (int depth=1; depth<=maxDepth; depth++) {
            Move bm{};
            int sc = searchRoot(pos, depth, bm);
            if (stop) break;
            lastBest = bm;

            // if we found forced mate close, can stop early
            if (std::abs(sc) > MATE - 200) break;
        }
        return lastBest;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    double timeout = 5.0;
    string fen;

    string line1;
    if (!std::getline(cin, line1)) return 0;
    if (!line1.empty()) timeout = std::stod(line1);

    if (!std::getline(cin, fen)) return 0;
    // allow trailing spaces / extra lines
    while (!fen.empty() && (fen.back()=='\r' || fen.back()=='\n')) fen.pop_back();

    Engine eng;
    if (!eng.pos.parseFEN(fen)) {
        cout << "0000\n";
        return 0;
    }

    Move best = eng.think(timeout);

    // If no move found (should happen only in mate/stalemate), output 0000
    vector<Move> test;
    eng.genPseudoMoves(eng.pos, test);
    bool anyLegal = false;
    for (auto& m : test) {
        Undo u;
        if (eng.makeMove(eng.pos, m, u)) {
            anyLegal = true;
            eng.unmakeMove(eng.pos, m, u);
            break;
        }
    }
    if (!anyLegal) {
        cout << "0000\n";
        return 0;
    }

    // If the chosen move is illegal (shouldn't), pick first legal
    {
        Undo u;
        if (!eng.makeMove(eng.pos, best, u)) {
            for (auto& m : test) {
                Undo u2;
                if (eng.makeMove(eng.pos, m, u2)) {
                    eng.unmakeMove(eng.pos, m, u2);
                    best = m;
                    break;
                }
            }
        } else {
            eng.unmakeMove(eng.pos, best, u);
        }
    }

    cout << eng.moveToNotation(eng.pos, best) << "\n";
    return 0;
}
