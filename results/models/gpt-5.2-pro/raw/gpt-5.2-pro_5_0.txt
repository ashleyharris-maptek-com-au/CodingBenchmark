```rust
use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read, Write};

#[derive(Clone)]
struct Scanner {
    buf: Vec<u8>,
    i: usize,
}
impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self { buf: input.into_bytes(), i: 0 }
    }
    fn next_i64(&mut self) -> i64 {
        while self.i < self.buf.len() && self.buf[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
        let mut sign = 1i64;
        if self.i < self.buf.len() && self.buf[self.i] == b'-' {
            sign = -1;
            self.i += 1;
        }
        let mut v = 0i64;
        while self.i < self.buf.len() && !self.buf[self.i].is_ascii_whitespace() {
            v = v * 10 + (self.buf[self.i] - b'0') as i64;
            self.i += 1;
        }
        v * sign
    }
    fn next_usize(&mut self) -> usize {
        self.next_i64() as usize
    }
}

fn main() {
    let mut sc = Scanner::new();
    let w = sc.next_usize();
    let h = sc.next_usize();
    let obs_cnt = sc.next_usize();
    let require_cycle = sc.next_usize() != 0;

    let total_u128 = (w as u128) * (h as u128);
    let mut obstacles: Vec<(usize, usize)> = Vec::with_capacity(obs_cnt);
    for _ in 0..obs_cnt {
        let x = sc.next_usize();
        let y = sc.next_usize();
        obstacles.push((x, y));
    }

    let mut out = io::BufWriter::new(io::stdout());

    // If dimensions are too large to even index, we can only handle the obstacle-free constructive cases.
    if total_u128 > (usize::MAX as u128) {
        if obs_cnt != 0 {
            // Cannot even represent the grid in memory/indexing.
            writeln!(out, "0").unwrap();
            return;
        }
        // Obstacle-free, we can still stream-construct.
        solve_obstacle_free_streaming(&mut out, w as u64, h as u64, require_cycle);
        return;
    }

    let total = (w * h) as usize;

    // Obstacle-free: stream constructive solution (fast, scales, no extra memory).
    if obs_cnt == 0 {
        solve_obstacle_free_streaming(&mut out, w as u64, h as u64, require_cycle);
        return;
    }

    // For obstacle cases we need a concrete representation; this is intended for up to a few thousand cells.
    // (The Hamiltonian problem with obstacles is NP-complete in general.)
    if total > 5_000_000 {
        // Not realistically solvable with the exact/branching approach below, and storing full arrays is large.
        // We only output 0 when we can prove impossibility; here we cannot, but such instances are outside
        // realistic IO/time anyway. Still, keep behavior deterministic.
        writeln!(out, "0").unwrap();
        return;
    }

    let mut blocked = vec![false; total];
    for &(x, y) in &obstacles {
        if x < w && y < h {
            blocked[y * w + x] = true;
        }
    }
    let start_idx = 0usize; // (0,0)
    if blocked[start_idx] {
        writeln!(out, "0").unwrap();
        return;
    }

    // Build mapping from grid cell -> node index in compact graph.
    let mut grid_to_node = vec![i32::MIN; total];
    let mut xs: Vec<i32> = Vec::new();
    let mut ys: Vec<i32> = Vec::new();

    for y in 0..h {
        for x in 0..w {
            if !blocked[y * w + x] {
                let id = xs.len() as i32;
                grid_to_node[y * w + x] = id;
                xs.push(x as i32);
                ys.push(y as i32);
            }
        }
    }

    let n = xs.len();
    if n == 0 {
        writeln!(out, "0").unwrap();
        return;
    }
    if n == 1 {
        if require_cycle {
            writeln!(out, "0").unwrap();
        } else {
            writeln!(out, "1").unwrap();
            writeln!(out, "0 0").unwrap();
        }
        return;
    }

    let start_node = grid_to_node[0] as usize;

    // Build adjacency list (grid degree <=4).
    let mut adj: Vec<Vec<usize>> = vec![Vec::with_capacity(4); n];
    for i in 0..n {
        let x = xs[i] as i32;
        let y = ys[i] as i32;
        let nbrs = [(x - 1, y), (x + 1, y), (x, y - 1), (x, y + 1)];
        for (nx, ny) in nbrs {
            if nx >= 0 && ny >= 0 && (nx as usize) < w && (ny as usize) < h {
                let gi = (ny as usize) * w + (nx as usize);
                let nid = grid_to_node[gi];
                if nid != i32::MIN {
                    adj[i].push(nid as usize);
                }
            }
        }
    }

    // Quick necessary conditions (sound impossibility checks).
    if !check_connected(n, &adj, start_node) {
        writeln!(out, "0").unwrap();
        return;
    }

    // Bipartite color counts and parity constraints.
    let mut black = 0usize;
    let mut white = 0usize;
    for i in 0..n {
        let c = ((xs[i] + ys[i]) & 1) as i32;
        if c == 0 { black += 1 } else { white += 1 }
    }
    if require_cycle {
        // Hamiltonian cycle (as requested: last adjacent to start) implies even number of vertices.
        // In bipartite graphs, any cycle alternates colors, so black==white is necessary.
        if black != white {
            writeln!(out, "0").unwrap();
            return;
        }
        // Start must have degree >= 2 in the free-cell graph.
        if adj[start_node].len() < 2 {
            writeln!(out, "0").unwrap();
            return;
        }
    } else {
        // For Hamiltonian path in bipartite graph, |black-white| <= 1 is necessary.
        let diff = if black > white { black - white } else { white - black };
        if diff > 1 {
            writeln!(out, "0").unwrap();
            return;
        }
        // Additionally, start color must be compatible with counts:
        // If black==white+1, both endpoints are black; if white==black+1, both endpoints are white; else opposite.
        // Start is at (0,0) => black, so white==black+1 is impossible.
        if white == black + 1 {
            writeln!(out, "0").unwrap();
            return;
        }
    }

    // Degree-1 necessary condition (sound):
    // In any Hamiltonian path, there can be at most 2 vertices of degree 1 in the graph.
    // If cycle required, degree-1 vertices make it impossible.
    let mut deg1 = 0usize;
    for i in 0..n {
        if adj[i].len() == 0 {
            writeln!(out, "0").unwrap();
            return;
        }
        if adj[i].len() == 1 {
            deg1 += 1;
        }
    }
    if require_cycle {
        if deg1 != 0 {
            writeln!(out, "0").unwrap();
            return;
        }
    } else {
        if deg1 > 2 {
            writeln!(out, "0").unwrap();
            return;
        }
        // If exactly 2 degree-1 vertices exist, start must be one of them (sound).
        if deg1 == 2 && adj[start_node].len() != 1 {
            writeln!(out, "0").unwrap();
            return;
        }
    }

    // Solve:
    let path_opt = if n <= 64 {
        solve_bitmask(n, &adj, start_node, require_cycle, &xs, &ys)
    } else {
        solve_dfs_pruned(n, &adj, start_node, require_cycle, &xs, &ys)
    };

    match path_opt {
        None => {
            writeln!(out, "0").unwrap();
        }
        Some(path_nodes) => {
            writeln!(out, "{}", path_nodes.len()).unwrap();
            for &u in &path_nodes {
                writeln!(out, "{} {}", xs[u], ys[u]).unwrap();
            }
        }
    }
}

fn solve_obstacle_free_streaming(out: &mut dyn Write, w: u64, h: u64, require_cycle: bool) {
    let total = w as u128 * h as u128;
    if total == 0 {
        writeln!(out, "0").unwrap();
        return;
    }
    if total == 1 {
        if require_cycle {
            writeln!(out, "0").unwrap();
        } else {
            writeln!(out, "1").unwrap();
            writeln!(out, "0 0").unwrap();
        }
        return;
    }

    if !require_cycle {
        writeln!(out, "{}", total).unwrap();
        for y in 0..h {
            if y % 2 == 0 {
                for x in 0..w {
                    writeln!(out, "{} {}", x, y).unwrap();
                }
            } else {
                for x in (0..w).rev() {
                    writeln!(out, "{} {}", x, y).unwrap();
                }
            }
        }
        return;
    }

    // Cycle requested: exists for obstacle-free grid iff both dims > 1 and total is even.
    if w == 1 || h == 1 || (total % 2 == 1) {
        writeln!(out, "0").unwrap();
        return;
    }

    // Construct a Hamiltonian cycle order but output as a Hamiltonian path whose last node is adjacent to (0,0).
    // If w even: use the explicit construction; else (w odd, h even) transpose.
    writeln!(out, "{}", total).unwrap();
    if w % 2 == 0 {
        print_cycle_even_width(out, w, h, false);
    } else {
        // w odd, h even: transpose and swap coordinates when printing.
        print_cycle_even_width(out, h, w, true);
    }
}

/// Prints a Hamiltonian "cycle order" for an even-width w x h grid (w even, h>=2),
/// starting at (0,0), ending at (1,0), so the last cell is adjacent to start.
/// If transposed=true, the generated coordinates (x',y') are output as (y',x').
fn print_cycle_even_width(out: &mut dyn Write, w: u64, h: u64, transposed: bool) {
    debug_assert!(w % 2 == 0 && w >= 2 && h >= 2);

    let mut emit = |x: u64, y: u64| {
        if !transposed {
            writeln!(out, "{} {}", x, y).unwrap();
        } else {
            writeln!(out, "{} {}", y, x).unwrap();
        }
    };

    // Column 0: (0,0) down to (0,h-1)
    for y in 0..h {
        emit(0, y);
    }

    // Columns 1..w-1: cover rows 1..h-1 in a serpentine, leaving row 0 for the end sweep.
    for x in 1..w {
        if x % 2 == 1 {
            // Start at (x,h-1) then go up to (x,1).
            emit(x, h - 1);
            if h > 2 {
                for y in (1..=(h - 2)).rev() {
                    emit(x, y);
                }
            }
        } else {
            // Start at (x,1) then go down to (x,h-1).
            emit(x, 1);
            for y in 2..h {
                emit(x, y);
            }
        }
    }

    // Now at (w-1,1). Go to (w-1,0), then sweep top row back to (1,0).
    emit(w - 1, 0);
    if w > 2 {
        for x in (1..=(w - 2)).rev() {
            emit(x, 0);
        }
    }
    // End is (1,0), which is adjacent to (0,0).
}

fn check_connected(n: usize, adj: &[Vec<usize>], start: usize) -> bool {
    let mut q = Vec::with_capacity(n);
    let mut seen = vec![false; n];
    seen[start] = true;
    q.push(start);
    let mut qi = 0usize;
    while qi < q.len() {
        let u = q[qi];
        qi += 1;
        for &v in &adj[u] {
            if !seen[v] {
                seen[v] = true;
                q.push(v);
            }
        }
    }
    q.len() == n
}

fn solve_bitmask(
    n: usize,
    adj: &[Vec<usize>],
    start: usize,
    require_cycle: bool,
    xs: &[i32],
    ys: &[i32],
) -> Option<Vec<usize>> {
    debug_assert!(n <= 64);
    let all_mask = if n == 64 { u64::MAX } else { (1u64 << n) - 1 };

    let mut neigh = vec![0u64; n];
    for u in 0..n {
        let mut m = 0u64;
        for &v in &adj[u] {
            m |= 1u64 << v;
        }
        neigh[u] = m;
    }

    let mut path = vec![0usize; n];
    path[0] = start;

    let mut memo_dead: HashMap<u128, ()> = HashMap::new();

    fn is_adjacent(adj: &[Vec<usize>], a: usize, b: usize) -> bool {
        adj[a].iter().any(|&v| v == b)
    }

    fn dfs(
        n: usize,
        adj: &[Vec<usize>],
        neigh: &[u64],
        start: usize,
        require_cycle: bool,
        all_mask: u64,
        pos: usize,
        visited: u64,
        step: usize,
        path: &mut [usize],
        memo_dead: &mut HashMap<u128, ()>,
    ) -> bool {
        if step == n {
            return if require_cycle {
                is_adjacent(adj, pos, start)
            } else {
                true
            };
        }

        let key = (visited as u128) | ((pos as u128) << 64);
        if memo_dead.contains_key(&key) {
            return false;
        }

        // Candidates: unvisited neighbors of pos.
        let mut cand_mask = neigh[pos] & (!visited);
        if cand_mask == 0 {
            memo_dead.insert(key, ());
            return false;
        }

        // Heuristic: iterate candidates sorted by onward degree.
        let mut candidates: [usize; 4] = [0; 4];
        let mut cc = 0usize;
        while cand_mask != 0 {
            let b = cand_mask.trailing_zeros() as usize;
            cand_mask &= cand_mask - 1;
            candidates[cc] = b;
            cc += 1;
        }

        let remaining_mask = ((!visited) & all_mask) | (1u64 << pos);

        // Degree/connectivity pruning (sound).
        // Connectivity: remaining vertices (pos + unvisited) must be connected via remaining edges.
        // (Start is not included even in cycle mode; it cannot be traversed again.)
        let rem_count = n - (step - 1);
        if !connected_mask(pos, neigh, remaining_mask, rem_count) {
            memo_dead.insert(key, ());
            return false;
        }

        if require_cycle {
            // At most one unvisited node may have degree 1 among remaining excluding start-edge,
            // and if exists it must be adjacent to start (can be the final node closing to start).
            let mut deg1_unvisited = 0usize;
            for u in 0..n {
                if (visited >> u) & 1 == 1 {
                    continue;
                }
                let deg_excl = (neigh[u] & remaining_mask).count_ones() as usize;
                if deg_excl == 0 {
                    memo_dead.insert(key, ());
                    return false;
                }
                if deg_excl == 1 {
                    deg1_unvisited += 1;
                    if deg1_unvisited > 1 {
                        memo_dead.insert(key, ());
                        return false;
                    }
                    if !is_adjacent(adj, u, start) {
                        memo_dead.insert(key, ());
                        return false;
                    }
                }
            }
        } else {
            let mut deg1_unvisited = 0usize;
            for u in 0..n {
                if (visited >> u) & 1 == 1 {
                    continue;
                }
                let deg = (neigh[u] & remaining_mask).count_ones() as usize;
                if deg == 0 {
                    memo_dead.insert(key, ());
                    return false;
                }
                if deg == 1 {
                    deg1_unvisited += 1;
                    if deg1_unvisited > 1 {
                        memo_dead.insert(key, ());
                        return false;
                    }
                }
            }
        }

        // Sort candidates by onward degree (Warnsdorff-like).
        // Onward degree counts neighbors among unvisited after stepping there.
        let mut cand_vec = candidates[..cc].to_vec();
        cand_vec.sort_by(|&a, &b| {
            let da = (neigh[a] & (!visited)).count_ones();
            let db = (neigh[b] & (!visited)).count_ones();
            da.cmp(&db)
        });

        for &nx in &cand_vec {
            // If we're about to finish in cycle mode, enforce final adjacency quickly.
            if require_cycle && step + 1 == n && !is_adjacent(adj, nx, start) {
                continue;
            }

            path[step] = nx;
            let visited2 = visited | (1u64 << nx);
            if dfs(
                n,
                adj,
                neigh,
                start,
                require_cycle,
                all_mask,
                nx,
                visited2,
                step + 1,
                path,
                memo_dead,
            ) {
                return true;
            }
        }

        memo_dead.insert(key, ());
        false
    }

    fn connected_mask(pos: usize, neigh: &[u64], remaining_mask: u64, want: usize) -> bool {
        let mut seen = 0u64;
        let mut frontier = 1u64 << pos;
        while frontier != 0 {
            let b = frontier & (!frontier + 1);
            frontier ^= b;
            seen |= b;
            let u = b.trailing_zeros() as usize;
            let nxt = neigh[u] & remaining_mask & (!seen);
            frontier |= nxt;
        }
        seen.count_ones() as usize == want
    }

    let visited0 = 1u64 << start;
    let ok = dfs(
        n,
        adj,
        &neigh,
        start,
        require_cycle,
        all_mask,
        start,
        visited0,
        1,
        &mut path,
        &mut memo_dead,
    );
    if ok { Some(path) } else { None }
}

fn solve_dfs_pruned(
    n: usize,
    adj: &[Vec<usize>],
    start: usize,
    require_cycle: bool,
    xs: &[i32],
    ys: &[i32],
) -> Option<Vec<usize>> {
    let _ = (xs, ys); // (kept for potential future heuristics; silence unused warnings)

    let mut visited = vec![false; n];
    let mut path = vec![0usize; n];
    visited[start] = true;
    path[0] = start;

    fn is_adjacent(adj: &[Vec<usize>], a: usize, b: usize) -> bool {
        adj[a].iter().any(|&v| v == b)
    }

    fn remaining_connected(curr: usize, visited: &[bool], adj: &[Vec<usize>], remaining: usize) -> bool {
        let n = visited.len();
        let mut q = Vec::with_capacity(remaining);
        let mut seen = vec![false; n];
        seen[curr] = true;
        q.push(curr);
        let mut qi = 0usize;
        while qi < q.len() {
            let u = q[qi];
            qi += 1;
            for &v in &adj[u] {
                if seen[v] {
                    continue;
                }
                if !visited[v] {
                    seen[v] = true;
                    q.push(v);
                }
            }
        }
        q.len() == remaining
    }

    fn prune_ok(curr: usize, step: usize, visited: &[bool], adj: &[Vec<usize>], start: usize, require_cycle: bool) -> bool {
        let n = visited.len();

        // Remaining set size is: curr + all unvisited = n - (step-1)
        let remaining = n - (step - 1);

        // Degree-based sound pruning.
        if require_cycle {
            // In cycle-as-path-to-neighbor-of-start, there is exactly one special "final" node that uses the edge to start.
            // So among unvisited nodes, at most one may have remaining-degree 1 (excluding the start-edge),
            // and if it exists, it must be adjacent to start.
            let mut deg1_unvisited = 0usize;

            // Current must have at least one unvisited neighbor unless finished.
            if step < n {
                let curr_deg = adj[curr].iter().filter(|&&v| !visited[v]).count();
                if curr_deg == 0 {
                    return false;
                }
            }

            for u in 0..n {
                if visited[u] {
                    continue;
                }
                let mut deg_excl = 0usize;
                for &v in &adj[u] {
                    if !visited[v] || v == curr {
                        // v==curr is always visited, but allowed as the remaining path endpoint.
                        if v == curr || !visited[v] {
                            deg_excl += 1;
                        }
                    }
                }
                if deg_excl == 0 {
                    return false;
                }
                if deg_excl == 1 {
                    deg1_unvisited += 1;
                    if deg1_unvisited > 1 {
                        return false;
                    }
                    if !is_adjacent(adj, u, start) {
                        return false;
                    }
                }
            }
        } else {
            // Simple path: among unvisited nodes, at most one may have remaining-degree 1 (forced end), since curr is the other end.
            let mut deg1_unvisited = 0usize;

            if step < n {
                let curr_deg = adj[curr].iter().filter(|&&v| !visited[v]).count();
                if curr_deg == 0 {
                    return false;
                }
            }

            for u in 0..n {
                if visited[u] {
                    continue;
                }
                let mut deg = 0usize;
                for &v in &adj[u] {
                    if !visited[v] || v == curr {
                        if v == curr || !visited[v] {
                            deg += 1;
                        }
                    }
                }
                if deg == 0 {
                    return false;
                }
                if deg == 1 {
                    deg1_unvisited += 1;
                    if deg1_unvisited > 1 {
                        return false;
                    }
                }
            }
        }

        // Connectivity pruning: remaining graph induced by (curr + unvisited) must be connected.
        remaining_connected(curr, visited, adj, remaining)
    }

    fn dfs(
        n: usize,
        adj: &[Vec<usize>],
        start: usize,
        require_cycle: bool,
        visited: &mut [bool],
        path: &mut [usize],
        curr: usize,
        step: usize,
    ) -> bool {
        if step == n {
            return if require_cycle {
                is_adjacent(adj, curr, start)
            } else {
                true
            };
        }

        // Collect candidates (unvisited neighbors).
        let mut cand: [usize; 4] = [usize::MAX; 4];
        let mut cc = 0usize;
        for &v in &adj[curr] {
            if !visited[v] {
                cand[cc] = v;
                cc += 1;
            }
        }
        if cc == 0 {
            return false;
        }

        // Sort candidates by onward degree (Warnsdorff).
        let mut cand_vec = cand[..cc].to_vec();
        cand_vec.sort_by(|&a, &b| {
            let da = adj[a].iter().filter(|&&u| !visited[u]).count();
            let db = adj[b].iter().filter(|&&u| !visited[u]).count();
            da.cmp(&db)
        });

        // Small tie-break: prefer staying in same row/col (often helps in grids).
        cand_vec.sort_by(|&a, &b| {
            let da = adj[a].iter().filter(|&&u| !visited[u]).count();
            let db = adj[b].iter().filter(|&&u| !visited[u]).count();
            if da != db {
                return da.cmp(&db);
            }
            // Stable-ish ordering to keep determinism
            a.cmp(&b)
        });

        for &nx in &cand_vec {
            // If we're choosing the last node in cycle mode, it must be adjacent to start.
            if require_cycle && step + 1 == n && !is_adjacent(adj, nx, start) {
                continue;
            }

            visited[nx] = true;
            path[step] = nx;

            if prune_ok(nx, step + 1, visited, adj, start, require_cycle) {
                if dfs(n, adj, start, require_cycle, visited, path, nx, step + 1) {
                    return true;
                }
            }

            visited[nx] = false;
        }
        false
    }

    // A little extra: early-prune before starting recursion.
    if !prune_ok(start, 1, &visited, adj, start, require_cycle) {
        return None;
    }

    if dfs(n, adj, start, require_cycle, &mut visited, &mut path, start, 1) {
        Some(path)
    } else {
        None
    }
}
```
