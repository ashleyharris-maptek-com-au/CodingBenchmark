```spirv
; SPIR-V
; Version: 1.0
; Generator: hand-written
; Bound: 200
OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void

%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_2 = OpConstant %uint 2
%arr_pv_in_2 = OpTypeArray %gl_PerVertex_in %uint_2
%arr_v3_in_2 = OpTypeArray %v3float %uint_2

%ptr_Input_arr_pv_in_2 = OpTypePointer Input %arr_pv_in_2
%ptr_Input_gl_PerVertex_in = OpTypePointer Input %gl_PerVertex_in
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_arr_v3_in_2 = OpTypePointer Input %arr_v3_in_2
%ptr_Input_v3float = OpTypePointer Input %v3float

%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_Input_arr_pv_in_2 Input
%worldPos = OpVariable %ptr_Input_arr_v3_in_2 Input
%normal = OpVariable %ptr_Input_arr_v3_in_2 Input
%color = OpVariable %ptr_Input_arr_v3_in_2 Input

%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %ptr_Output_v3float Output

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_3 = OpConstant %int 3

%float_1 = OpConstant %float 1
%float_0_04 = OpConstant %float 0.04
%float_0_005 = OpConstant %float 0.005

%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load endpoint positions from gl_in[0].gl_Position and gl_in[1].gl_Position
%p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%p0_clip = OpLoad %v4float %p0_ptr
%p1_clip = OpLoad %v4float %p1_ptr

; Load endpoint colors from color[0], color[1]
%c0_ptr = OpAccessChain %ptr_Input_v3float %color %int_0
%c1_ptr = OpAccessChain %ptr_Input_v3float %color %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

; Extract w and z
%p0_w = OpCompositeExtract %float %p0_clip 3
%p1_w = OpCompositeExtract %float %p1_clip 3
%p0_z = OpCompositeExtract %float %p0_clip 2
%p1_z = OpCompositeExtract %float %p1_clip 2

; Extract xy
%p0_xy = OpVectorShuffle %v2float %p0_clip %p0_clip 0 1
%p1_xy = OpVectorShuffle %v2float %p1_clip %p1_clip 0 1

; Convert to NDC: xy / w
%invw0 = OpFDiv %float %float_1 %p0_w
%invw1 = OpFDiv %float %float_1 %p1_w
%p0_ndc = OpVectorTimesScalar %v2float %p0_xy %invw0
%p1_ndc = OpVectorTimesScalar %v2float %p1_xy %invw1

; Direction and perpendicular in NDC
%dir = OpFSub %v2float %p1_ndc %p0_ndc
%dir_n = OpExtInst %v2float %glsl Normalize %dir
%dir_x = OpCompositeExtract %float %dir_n 0
%dir_y = OpCompositeExtract %float %dir_n 1
%neg_dir_y = OpFNegate %float %dir_y
%perp = OpCompositeConstruct %v2float %neg_dir_y %dir_x

; Thicknesses
%off0 = OpVectorTimesScalar %v2float %perp %float_0_04
%off1 = OpVectorTimesScalar %v2float %perp %float_0_005

; NDC ribbon vertices
%p0_plus_ndc = OpFAdd %v2float %p0_ndc %off0
%p0_minus_ndc = OpFSub %v2float %p0_ndc %off0
%p1_plus_ndc = OpFAdd %v2float %p1_ndc %off1
%p1_minus_ndc = OpFSub %v2float %p1_ndc %off1

; Back to clip: xy = ndc * w (preserve z,w)
%p0_plus_xy = OpVectorTimesScalar %v2float %p0_plus_ndc %p0_w
%p0_minus_xy = OpVectorTimesScalar %v2float %p0_minus_ndc %p0_w
%p1_plus_xy = OpVectorTimesScalar %v2float %p1_plus_ndc %p1_w
%p1_minus_xy = OpVectorTimesScalar %v2float %p1_minus_ndc %p1_w

%p0_plus_x = OpCompositeExtract %float %p0_plus_xy 0
%p0_plus_y = OpCompositeExtract %float %p0_plus_xy 1
%p0_minus_x = OpCompositeExtract %float %p0_minus_xy 0
%p0_minus_y = OpCompositeExtract %float %p0_minus_xy 1

%p1_plus_x = OpCompositeExtract %float %p1_plus_xy 0
%p1_plus_y = OpCompositeExtract %float %p1_plus_xy 1
%p1_minus_x = OpCompositeExtract %float %p1_minus_xy 0
%p1_minus_y = OpCompositeExtract %float %p1_minus_xy 1

%v0_clip = OpCompositeConstruct %v4float %p0_plus_x %p0_plus_y %p0_z %p0_w
%v1_clip = OpCompositeConstruct %v4float %p0_minus_x %p0_minus_y %p0_z %p0_w
%v2_clip = OpCompositeConstruct %v4float %p1_plus_x %p1_plus_y %p1_z %p1_w
%v3_clip = OpCompositeConstruct %v4float %p1_minus_x %p1_minus_y %p1_z %p1_w

; Output member pointer for gl_Position
%glpos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0

; Emit 4 vertices as triangle strip with tapered thickness
OpStore %glpos_ptr %v0_clip
OpStore %outColor %c0
OpEmitVertex

OpStore %glpos_ptr %v1_clip
OpStore %outColor %c0
OpEmitVertex

OpStore %glpos_ptr %v2_clip
OpStore %outColor %c1
OpEmitVertex

OpStore %glpos_ptr %v3_clip
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
