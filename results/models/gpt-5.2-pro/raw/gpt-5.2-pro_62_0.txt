/* eslint-disable @typescript-eslint/no-unused-vars */
declare const require: any;

const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8");
const data = JSON.parse(input);

type Town = { id: any; x: number; y: number; value: number };

const W: number = data.width | 0;
const H: number = data.height | 0;
const grid: string[] = data.grid || [];
const ignitionsIn: [number, number][] = data.ignitions || [];
const townsIn: Town[] = data.towns || [];
const cutBudget: number = Math.max(0, data.cutBudget | 0);

const WH = W * H;
const SUPER = WH;
const totalNodes = WH + 1;

function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < W && y >= 0 && y < H;
}
function idxOf(x: number, y: number): number {
  return y * W + x;
}

// 0 = blocked (~), 1 = forest (.)
const forest = new Uint8Array(WH);
for (let y = 0; y < H; y++) {
  const row = grid[y] ?? "";
  for (let x = 0; x < W; x++) {
    const c = row.charCodeAt(x);
    forest[idxOf(x, y)] = c === 46 /* '.' */ ? 1 : 0;
  }
}

const isIgn = new Uint8Array(WH);
const ignitionIdx: number[] = [];
{
  const seen = new Uint8Array(WH);
  for (let i = 0; i < ignitionsIn.length; i++) {
    const [x, y] = ignitionsIn[i];
    if (!inBounds(x, y)) continue;
    const id = idxOf(x, y);
    if (!forest[id]) continue;
    if (seen[id]) continue;
    seen[id] = 1;
    isIgn[id] = 1;
    ignitionIdx.push(id);
  }
}

// Town values per cell (sum if multiple)
const townVal = new Float64Array(WH);
for (let i = 0; i < townsIn.length; i++) {
  const t = townsIn[i];
  const x = t.x | 0,
    y = t.y | 0;
  if (!inBounds(x, y)) continue;
  const id = idxOf(x, y);
  if (!forest[id]) continue;
  const v = Number(t.value) || 0;
  if (v > 0) townVal[id] += v;
}

if (cutBudget === 0 || ignitionIdx.length === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

/** Segment tree (power-of-two) with range-set-to-zero lazy flag */
class SegTreeZero {
  size: number;
  sum: Float64Array;
  lazyZero: Uint8Array;

  constructor(values1Based: Float64Array, n1: number) {
    let size = 1;
    while (size < n1 + 1) size <<= 1; // we use indices 1..n
    this.size = size;
    this.sum = new Float64Array(size * 2);
    this.lazyZero = new Uint8Array(size * 2);

    // leaves: position size + i
    for (let i = 1; i <= n1; i++) this.sum[size + i] = values1Based[i];
    for (let i = size - 1; i >= 1; i--) this.sum[i] = this.sum[i << 1] + this.sum[(i << 1) | 1];
  }

  private push(node: number) {
    if (!this.lazyZero[node]) return;
    const lc = node << 1;
    const rc = lc | 1;
    this.sum[lc] = 0;
    this.sum[rc] = 0;
    this.lazyZero[lc] = 1;
    this.lazyZero[rc] = 1;
    this.lazyZero[node] = 0;
  }

  query(l: number, r: number): number {
    if (l > r) return 0;
    return this._query(1, 1, this.size, l, r);
  }

  private _query(node: number, nl: number, nr: number, ql: number, qr: number): number {
    if (ql <= nl && nr <= qr) return this.sum[node];
    if (nr < ql || qr < nl) return 0;
    this.push(node);
    const mid = (nl + nr) >> 1;
    return this._query(node << 1, nl, mid, ql, qr) + this._query((node << 1) | 1, mid + 1, nr, ql, qr);
  }

  setZero(l: number, r: number) {
    if (l > r) return;
    this._setZero(1, 1, this.size, l, r);
  }

  private _setZero(node: number, nl: number, nr: number, ql: number, qr: number) {
    if (ql <= nl && nr <= qr) {
      this.sum[node] = 0;
      this.lazyZero[node] = 1;
      return;
    }
    if (nr < ql || qr < nl) return;
    this.push(node);
    const mid = (nl + nr) >> 1;
    this._setZero(node << 1, nl, mid, ql, qr);
    this._setZero((node << 1) | 1, mid + 1, nr, ql, qr);
    this.sum[node] = this.sum[node << 1] + this.sum[(node << 1) | 1];
  }

  total(): number {
    return this.sum[1];
  }
}

/** Dynamic typed arrays for articulation-separated subtree intervals per cut-node. */
class IntervalStore {
  start: Int32Array;
  end: Int32Array;
  next: Int32Array;
  len: number;

  constructor(cap: number) {
    this.start = new Int32Array(cap);
    this.end = new Int32Array(cap);
    this.next = new Int32Array(cap);
    this.len = 0;
  }

  push(s: number, e: number, nxt: number): number {
    let i = this.len;
    if (i >= this.start.length) {
      const ncap = Math.max(1024, this.start.length << 1);
      const ns = new Int32Array(ncap);
      const ne = new Int32Array(ncap);
      const nn = new Int32Array(ncap);
      ns.set(this.start);
      ne.set(this.end);
      nn.set(this.next);
      this.start = ns;
      this.end = ne;
      this.next = nn;
    }
    this.start[i] = s | 0;
    this.end[i] = e | 0;
    this.next[i] = nxt | 0;
    this.len = i + 1;
    return i;
  }
}

/** Max-heap keyed by score */
class MaxHeap {
  u: Int32Array;
  s: Float64Array;
  n: number;

  constructor(cap: number) {
    this.u = new Int32Array(Math.max(1, cap));
    this.s = new Float64Array(Math.max(1, cap));
    this.n = 0;
  }

  private grow() {
    const ncap = Math.max(4, this.u.length << 1);
    const nu = new Int32Array(ncap);
    const ns = new Float64Array(ncap);
    nu.set(this.u);
    ns.set(this.s);
    this.u = nu;
    this.s = ns;
  }

  push(node: number, score: number) {
    if (this.n >= this.u.length) this.grow();
    let i = this.n++;
    this.u[i] = node | 0;
    this.s[i] = score;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.s[p] >= score) break;
      this.u[i] = this.u[p];
      this.s[i] = this.s[p];
      i = p;
    }
    this.u[i] = node | 0;
    this.s[i] = score;
  }

  pop(): { node: number; score: number } | null {
    if (this.n === 0) return null;
    const node = this.u[0];
    const score = this.s[0];
    const lastU = this.u[--this.n];
    const lastS = this.s[this.n];
    if (this.n > 0) {
      let i = 0;
      while (true) {
        const l = i * 2 + 1;
        if (l >= this.n) break;
        const r = l + 1;
        let c = l;
        if (r < this.n && this.s[r] > this.s[l]) c = r;
        if (this.s[c] <= lastS) break;
        this.u[i] = this.u[c];
        this.s[i] = this.s[c];
        i = c;
      }
      this.u[i] = lastU;
      this.s[i] = lastS;
    }
    return { node, score };
  }

  get size(): number {
    return this.n;
  }
}

/** Min-heap keyed by score (for top-K selection) */
class MinHeap {
  u: Int32Array;
  s: Float64Array;
  n: number;

  constructor(cap: number) {
    this.u = new Int32Array(Math.max(1, cap));
    this.s = new Float64Array(Math.max(1, cap));
    this.n = 0;
  }

  push(node: number, score: number) {
    let i = this.n++;
    if (this.n > this.u.length) {
      const ncap = Math.max(4, this.u.length << 1);
      const nu = new Int32Array(ncap);
      const ns = new Float64Array(ncap);
      nu.set(this.u);
      ns.set(this.s);
      this.u = nu;
      this.s = ns;
    }
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.s[p] <= score) break;
      this.u[i] = this.u[p];
      this.s[i] = this.s[p];
      i = p;
    }
    this.u[i] = node | 0;
    this.s[i] = score;
  }

  pop(): { node: number; score: number } | null {
    if (this.n === 0) return null;
    const node = this.u[0];
    const score = this.s[0];
    const lastU = this.u[--this.n];
    const lastS = this.s[this.n];
    if (this.n > 0) {
      let i = 0;
      while (true) {
        const l = i * 2 + 1;
        if (l >= this.n) break;
        const r = l + 1;
        let c = l;
        if (r < this.n && this.s[r] < this.s[l]) c = r;
        if (this.s[c] >= lastS) break;
        this.u[i] = this.u[c];
        this.s[i] = this.s[c];
        i = c;
      }
      this.u[i] = lastU;
      this.s[i] = lastS;
    }
    return { node, score };
  }

  peekScore(): number {
    return this.n ? this.s[0] : Number.POSITIVE_INFINITY;
  }

  get size(): number {
    return this.n;
  }
}

// --- Iterative DFS for articulation intervals (Tarjan low-link) rooted at SUPER ---
const disc = new Int32Array(totalNodes);
const low = new Int32Array(totalNodes);
const parent = new Int32Array(totalNodes);
parent.fill(-1);

const tin = new Int32Array(totalNodes);
const tout = new Int32Array(totalNodes);
const eulerToNode = new Int32Array(totalNodes + 1); // tin ranges 1..timeEuler

const subTown = new Float64Array(totalNodes);
const subIgn = new Uint8Array(totalNodes);

const head = new Int32Array(totalNodes);
head.fill(-1);

const potential = new Float64Array(totalNodes); // sum of (town in separable child subtrees), excludes own town

const intervals = new IntervalStore(1024);

// DFS stack (explicit)
const stackU = new Int32Array(totalNodes);
const stackNext = new Uint8Array(totalNodes);

let sp = 0;
stackU[0] = SUPER;
stackNext[0] = 0;

let timeDisc = 0;
let timeEuler = 0;

function nextNeighbor(u: number, next: number): { v: number; next: number } {
  if (u === SUPER) {
    if (next >= ignitionIdx.length) return { v: -1, next };
    const v = ignitionIdx[next] | 0;
    return { v, next: (next + 1) | 0 };
  }

  const id = u | 0;
  // 0..3 cardinal neighbors, 4 => SUPER edge if ignition
  let n = next | 0;
  while (n < 5) {
    if (n < 4) {
      // boundary checks avoiding division when possible
      const mod = id % W;
      let v = -1;
      if (n === 0) {
        if (mod !== 0) v = id - 1;
      } else if (n === 1) {
        if (mod !== W - 1) v = id + 1;
      } else if (n === 2) {
        if (id >= W) v = id - W;
      } else {
        if (id < WH - W) v = id + W;
      }
      n++;
      if (v >= 0 && forest[v]) return { v, next: n };
      continue;
    } else {
      // SUPER edge
      n++;
      if (isIgn[id]) return { v: SUPER, next: n };
      continue;
    }
  }
  return { v: -1, next: n };
}

// Run DFS from SUPER
while (sp >= 0) {
  const u = stackU[sp];

  if (disc[u] === 0) {
    disc[u] = low[u] = ++timeDisc;
    tin[u] = ++timeEuler;
    eulerToNode[timeEuler] = u;
    if (u === SUPER) {
      subTown[u] = 0;
      subIgn[u] = 1; // treat as source-bearing
    } else {
      subTown[u] = townVal[u];
      subIgn[u] = isIgn[u];
    }
  }

  const nn = nextNeighbor(u, stackNext[sp]);
  stackNext[sp] = nn.next as any;

  const v = nn.v;
  if (v !== -1) {
    if (disc[v] === 0) {
      parent[v] = u;
      sp++;
      stackU[sp] = v;
      stackNext[sp] = 0;
    } else if (v !== parent[u]) {
      const dv = disc[v];
      if (dv < low[u]) low[u] = dv;
    }
  } else {
    // finish u
    tout[u] = timeEuler;
    const p = parent[u];
    sp--;
    if (p !== -1) {
      if (low[u] < low[p]) low[p] = low[u];
      subTown[p] += subTown[u];
      if (subIgn[u]) subIgn[p] = 1;

      // If u is a DFS-tree child of p that would be disconnected from sources by removing p:
      // low[u] >= disc[p] (and we ignore p=SUPER)
      if (p !== SUPER && low[u] >= disc[p]) {
        if (!subIgn[u] && subTown[u] > 0) {
          // store as a separable interval for p
          const oldHead = head[p];
          const ii = intervals.push(tin[u], tout[u], oldHead);
          head[p] = ii;
          potential[p] += subTown[u];
        }
      }
    }
  }
}

const visitedN = timeEuler; // includes SUPER and all ignition-reachable forest cells

// Build segment tree over Euler order with threatened town values (only visited nodes)
const base = new Float64Array(visitedN + 1);
for (let t = 1; t <= visitedN; t++) {
  const node = eulerToNode[t];
  if (node !== SUPER) base[t] = townVal[node];
}
const seg = new SegTreeZero(base, visitedN);

// Candidate selection
const candidateMark = new Uint8Array(WH);
const candidates: number[] = [];

// Always include threatened town-cells as candidates (cutting on the town itself saves it under this model)
for (let t = 1; t <= visitedN; t++) {
  const node = eulerToNode[t];
  if (node === SUPER) continue;
  if (isIgn[node]) continue; // disallow cutting ignition
  if (townVal[node] > 0) {
    if (!candidateMark[node]) {
      candidateMark[node] = 1;
      candidates.push(node);
    }
  }
}

// Include articulation candidates (plus any node with separable intervals)
let K = Math.max(50000, cutBudget * 20);
if (K > 200000) K = 200000;
const topK = new MinHeap(Math.min(K, 1024));

let articulationCount = 0;
for (let t = 1; t <= visitedN; t++) {
  const node = eulerToNode[t];
  if (node === SUPER) continue;
  if (isIgn[node]) continue;
  if (head[node] !== -1) {
    articulationCount++;
    const pot = potential[node] + townVal[node]; // include own town if any
    if (pot <= 0) continue;
    if (topK.size < K) {
      topK.push(node, pot);
    } else if (pot > topK.peekScore()) {
      topK.pop();
      topK.push(node, pot);
    }
  }
}

// Drain topK into candidates
while (topK.size) {
  const it = topK.pop()!;
  const node = it.node;
  if (node >= 0 && node < WH && !candidateMark[node]) {
    candidateMark[node] = 1;
    candidates.push(node);
  }
}

// Greedy selection using lazy-recomputed max-heap
function computeScore(u: number): number {
  const tu = tin[u];
  if (tu === 0) return 0;
  let s = seg.query(tu, tu); // own town value (if any)
  for (let e = head[u]; e !== -1; e = intervals.next[e]) {
    s += seg.query(intervals.start[e], intervals.end[e]);
  }
  return s;
}

function applyCut(u: number) {
  const tu = tin[u];
  if (tu !== 0) seg.setZero(tu, tu);
  for (let e = head[u]; e !== -1; e = intervals.next[e]) {
    seg.setZero(intervals.start[e], intervals.end[e]);
  }
}

const heap = new MaxHeap(Math.max(4, candidates.length));
for (let i = 0; i < candidates.length; i++) {
  const u = candidates[i];
  const s = computeScore(u);
  if (s > 0) heap.push(u, s);
}

const chosen = new Uint8Array(WH);
const firebreaks: number[][] = [];

for (let k = 0; k < cutBudget && heap.size > 0; ) {
  const top = heap.pop()!;
  const u = top.node;
  if (u < 0 || u >= WH) continue;
  if (chosen[u]) continue;
  if (!forest[u] || isIgn[u]) continue; // must be a forest cell, and not ignition

  const real = computeScore(u);
  if (real <= 0) continue;

  // Lazy update: if stale, reinsert with updated score.
  if (real + 1e-9 < top.score) {
    heap.push(u, real);
    continue;
  }

  // Take it
  chosen[u] = 1;
  applyCut(u);
  firebreaks.push([u % W, (u / W) | 0]);
  k++;
}

process.stdout.write(JSON.stringify({ firebreaks }));
