use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read, Write};

const MAX_MIM_N: usize = 44;
const BITSET_TARGET_LIMIT: u32 = 25_000_000; // exact DP with parent pointers
const BUCKET_TARGET_LIMIT: u32 = 2_000_000; // use counting/buckets (no sort) under this
const RANGE_BITSET_LIMIT: i64 = 20_000_000; // exact DP with negatives if total sum range is small
const HASH_DP_N_LIMIT: usize = 6000; // exact HashMap DP for moderate sizes (fallback)

#[derive(Clone, Copy)]
struct Item {
    weight: u32,
    group: u32,
    take: u32,
}

struct FastInput {
    buf: Vec<u8>,
    pos: usize,
}
impl FastInput {
    fn new() -> io::Result<Self> {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf)?;
        Ok(Self { buf, pos: 0 })
    }
    fn next_i64(&mut self) -> Option<i64> {
        while self.pos < self.buf.len() && self.buf[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        if self.pos >= self.buf.len() {
            return None;
        }
        let mut sign = 1i64;
        if self.buf[self.pos] == b'-' {
            sign = -1;
            self.pos += 1;
        } else if self.buf[self.pos] == b'+' {
            self.pos += 1;
        }
        let mut x = 0i64;
        while self.pos < self.buf.len() {
            let c = self.buf[self.pos];
            if !c.is_ascii_digit() {
                break;
            }
            x = x * 10 + (c - b'0') as i64;
            self.pos += 1;
        }
        Some(sign * x)
    }
}

fn gcd_i64(mut a: i64, mut b: i64) -> i64 {
    if a < 0 {
        a = -a;
    }
    if b < 0 {
        b = -b;
    }
    while b != 0 {
        let r = a % b;
        a = b;
        b = r;
    }
    a
}

fn output_yes(indices: &[usize]) -> io::Result<()> {
    let mut out = io::BufWriter::new(io::stdout());
    writeln!(out, "YES")?;
    if indices.is_empty() {
        writeln!(out)?;
        return Ok(());
    }
    for (i, idx) in indices.iter().enumerate() {
        if i != 0 {
            out.write_all(b" ")?;
        }
        write!(out, "{idx}")?;
    }
    writeln!(out)?;
    Ok(())
}

fn output_no() -> io::Result<()> {
    let mut out = io::BufWriter::new(io::stdout());
    writeln!(out, "NO")?;
    Ok(())
}

/* ---------------- Meet-in-the-middle (exact, supports negatives) ---------------- */

fn meet_in_middle(values: &[(i64, usize)], target: i64) -> Option<Vec<usize>> {
    let n = values.len();
    let mid = n / 2;
    let (left, right) = values.split_at(mid);

    let ln = left.len();
    let rn = right.len();
    if ln > 22 || rn > 22 {
        // Still OK often, but cap to avoid memory blowups.
        return None;
    }

    let lsz = 1usize << ln;
    let rsz = 1usize << rn;

    let mut lsums: Vec<(i64, u32)> = Vec::with_capacity(lsz);
    for mask in 0u32..(lsz as u32) {
        let mut s = 0i64;
        for i in 0..ln {
            if ((mask >> i) & 1) != 0 {
                s += left[i].0;
            }
        }
        lsums.push((s, mask));
    }

    let mut rsums: Vec<(i64, u32)> = Vec::with_capacity(rsz);
    for mask in 0u32..(rsz as u32) {
        let mut s = 0i64;
        for i in 0..rn {
            if ((mask >> i) & 1) != 0 {
                s += right[i].0;
            }
        }
        rsums.push((s, mask));
    }

    rsums.sort_unstable_by(|a, b| a.0.cmp(&b.0));

    for (ls, lm) in lsums {
        let need = target - ls;
        let pos = rsums.binary_search_by(|x| x.0.cmp(&need));
        if let Ok(j) = pos {
            let rm = rsums[j].1;
            let mut ans = Vec::new();
            for i in 0..ln {
                if ((lm >> i) & 1) != 0 {
                    ans.push(left[i].1);
                }
            }
            for i in 0..rn {
                if ((rm >> i) & 1) != 0 {
                    ans.push(right[i].1);
                }
            }
            return Some(ans);
        }
    }
    None
}

/* ---------------- Bitset DP (exact, non-negative) with parent reconstruction ---------------- */

fn bitset_dp_core(
    target: u32,
    items: &[Item],
    mut pop_take: impl FnMut(u32, u32) -> Vec<usize>,
) -> Option<Vec<usize>> {
    let t = target as usize;
    let words = (t + 64) / 64;
    let last_bits = (t + 1) % 64;
    let last_mask = if last_bits == 0 {
        !0u64
    } else {
        (1u64 << last_bits) - 1
    };

    let mut bits = vec![0u64; words];
    bits[0] = 1u64;

    let mut tmp = vec![0u64; words];

    let mut parent_item = vec![u32::MAX; t + 1];
    let mut prev_sum = vec![0u32; t + 1];

    let target_word = t / 64;
    let target_bit = (t % 64) as u32;

    let mut found = false;

    for (item_idx, it) in items.iter().enumerate() {
        let w = it.weight as usize;
        if w == 0 || w > t {
            continue;
        }

        tmp.fill(0);

        let ws = w / 64;
        let bs = (w % 64) as u32;
        if ws >= words {
            continue;
        }

        if bs == 0 {
            for i in 0..(words - ws) {
                tmp[i + ws] = bits[i];
            }
        } else {
            let rshift = 64 - bs;
            let mut carry = 0u64;
            for i in 0..(words - ws) {
                let src = bits[i];
                tmp[i + ws] = (src << bs) | carry;
                carry = src >> rshift;
            }
        }

        for j in 0..words {
            let shifted = tmp[j];
            let old = bits[j];
            let neww = shifted & !old;
            bits[j] = old | shifted;
            if j == words - 1 {
                bits[j] &= last_mask;
            }

            if neww != 0 {
                let mut x = neww;
                while x != 0 {
                    let tz = x.trailing_zeros() as usize;
                    let s = j * 64 + tz;
                    if s <= t {
                        // Only the first time we reach s, store parent.
                        if parent_item[s] == u32::MAX {
                            parent_item[s] = item_idx as u32;
                            prev_sum[s] = (s - w) as u32;
                            if j == target_word && tz as u32 == target_bit {
                                found = true;
                            }
                        }
                    }
                    x &= x - 1;
                }
            }

            if found && j >= target_word {
                break;
            }
        }

        if found {
            break;
        }
    }

    if !found {
        return None;
    }

    // Reconstruct selected items from parent pointers.
    let mut chosen: Vec<(u32, u32)> = Vec::new(); // (group, take)
    let mut s = t;
    while s != 0 {
        let pi = parent_item[s];
        if pi == u32::MAX {
            // Should not happen if found=true, but be safe.
            return None;
        }
        let it = items[pi as usize];
        chosen.push((it.group, it.take));
        s = prev_sum[s] as usize;
    }

    // Expand items into original indices.
    let mut ans: Vec<usize> = Vec::new();
    for (g, take) in chosen {
        ans.extend(pop_take(g, take));
    }
    Some(ans)
}

fn bitset_positive_bucket(pairs: &[(u32, u32)], target: u32) -> Option<Vec<usize>> {
    let t = target as usize;
    let mut counts = vec![0u32; t + 1];
    let mut head = vec![-1i32; t + 1];

    let mut next: Vec<i32> = Vec::with_capacity(pairs.len());
    let mut idxs: Vec<u32> = Vec::with_capacity(pairs.len());

    let mut sum_all: u64 = 0;
    let mut g = 0u32;
    let mut gcdv: i64 = 0;

    for &(v, idx) in pairs {
        if v == 0 || v > target {
            continue;
        }
        sum_all += v as u64;
        gcdv = if gcdv == 0 { v as i64 } else { gcd_i64(gcdv, v as i64) };
        let node = idxs.len();
        idxs.push(idx);
        next.push(head[v as usize]);
        head[v as usize] = node as i32;
        counts[v as usize] += 1;
        g = g.wrapping_add(1);
    }

    if target != 0 && sum_all < target as u64 {
        return None;
    }
    if gcdv != 0 && (target as i64) % gcdv != 0 {
        return None;
    }
    if g == 0 {
        return None;
    }

    // Map value -> group_id (compacting distinct values).
    let mut group_id_for_value = vec![-1i32; t + 1];
    let mut group_value: Vec<u32> = Vec::new();
    for v in 1..=t {
        if counts[v] != 0 {
            group_id_for_value[v] = group_value.len() as i32;
            group_value.push(v as u32);
        }
    }

    // Build items with binary decomposition, capping unusable counts.
    let mut items: Vec<Item> = Vec::new();
    for (gid, &v) in group_value.iter().enumerate() {
        let v_us = v as usize;
        let mut c = counts[v_us];
        let max_use = (target / v).max(0);
        if c > max_use {
            c = max_use;
        }
        let mut k = 1u32;
        let mut remaining = c;
        while remaining != 0 {
            let take = k.min(remaining);
            let weight = v.saturating_mul(take);
            if weight != 0 && weight <= target {
                items.push(Item {
                    weight,
                    group: gid as u32,
                    take,
                });
            }
            remaining -= take;
            k = k.saturating_mul(2);
            if k == 0 {
                break;
            }
        }
    }

    bitset_dp_core(target, &items, |group, take| {
        let v = group_value[group as usize] as usize;
        let mut out = Vec::with_capacity(take as usize);
        for _ in 0..take {
            let node = head[v];
            if node < 0 {
                break;
            }
            let ni = node as usize;
            out.push(idxs[ni] as usize);
            head[v] = next[ni];
        }
        out
    })
}

fn bitset_positive_sort(mut pairs: Vec<(u32, u32)>, target: u32) -> Option<Vec<usize>> {
    // Sort by value to group duplicates without a big counts/head array.
    pairs.sort_unstable_by(|a, b| a.0.cmp(&b.0));

    let mut sum_all: u64 = 0;
    let mut gcdv: i64 = 0;
    for &(v, _) in &pairs {
        if v == 0 || v > target {
            continue;
        }
        sum_all += v as u64;
        gcdv = if gcdv == 0 { v as i64 } else { gcd_i64(gcdv, v as i64) };
    }
    if target != 0 && sum_all < target as u64 {
        return None;
    }
    if gcdv != 0 && (target as i64) % gcdv != 0 {
        return None;
    }

    struct Group {
        value: u32,
        start: usize,
        end: usize,
        cursor: usize,
    }

    let mut groups: Vec<Group> = Vec::new();
    let mut i = 0usize;
    while i < pairs.len() {
        let v = pairs[i].0;
        let mut j = i + 1;
        while j < pairs.len() && pairs[j].0 == v {
            j += 1;
        }
        if v != 0 && v <= target {
            groups.push(Group {
                value: v,
                start: i,
                end: j,
                cursor: j,
            });
        }
        i = j;
    }

    // Build items with binary decomposition, capping unusable counts.
    let mut items: Vec<Item> = Vec::new();
    for (gid, gr) in groups.iter().enumerate() {
        let v = gr.value;
        let count = (gr.end - gr.start) as u32;
        let max_use = (target / v).max(0);
        let mut remaining = count.min(max_use);

        let mut k = 1u32;
        while remaining != 0 {
            let take = k.min(remaining);
            let weight = v.saturating_mul(take);
            if weight != 0 && weight <= target {
                items.push(Item {
                    weight,
                    group: gid as u32,
                    take,
                });
            }
            remaining -= take;
            k = k.saturating_mul(2);
            if k == 0 {
                break;
            }
        }
    }

    bitset_dp_core(target, &items, |group, take| {
        let g = &mut groups[group as usize];
        let mut out = Vec::with_capacity(take as usize);
        for _ in 0..take {
            if g.cursor == g.start {
                break;
            }
            g.cursor -= 1;
            out.push(pairs[g.cursor].1 as usize);
        }
        out
    })
}

/* ---------------- Offset range bitset DP (exact, supports negatives) ---------------- */

fn range_bitset_dp(values: &[(i64, usize)], target: i64) -> Option<Vec<usize>> {
    let mut min_sum = 0i64;
    let mut max_sum = 0i64;
    for &(x, _) in values {
        if x < 0 {
            min_sum += x;
        } else {
            max_sum += x;
        }
    }
    let range = max_sum - min_sum;
    if range < 0 || range > RANGE_BITSET_LIMIT {
        return None;
    }
    if target < min_sum || target > max_sum {
        return None;
    }

    let offset = -min_sum;
    let size = range as usize; // sums from min_sum..=max_sum -> size+1
    let words = (size + 64) / 64;
    let last_bits = (size + 1) % 64;
    let last_mask = if last_bits == 0 {
        !0u64
    } else {
        (1u64 << last_bits) - 1
    };

    let mut bits = vec![0u64; words];
    bits[(offset as usize) / 64] |= 1u64 << ((offset as usize) % 64);

    let mut tmp = vec![0u64; words];

    let mut parent_idx = vec![u32::MAX; size + 1];
    let mut prev_pos = vec![0u32; size + 1];

    let target_pos = (target + offset) as usize;

    for (i, &(x, _orig_idx)) in values.iter().enumerate() {
        if x == 0 {
            continue;
        }
        tmp.fill(0);

        if x > 0 {
            let k = x as usize;
            let ws = k / 64;
            let bs = (k % 64) as u32;
            if ws < words {
                if bs == 0 {
                    for w in 0..(words - ws) {
                        tmp[w + ws] = bits[w];
                    }
                } else {
                    let rshift = 64 - bs;
                    let mut carry = 0u64;
                    for w in 0..(words - ws) {
                        let src = bits[w];
                        tmp[w + ws] = (src << bs) | carry;
                        carry = src >> rshift;
                    }
                }
            }
        } else {
            let k = (-x) as usize;
            let ws = k / 64;
            let bs = (k % 64) as u32;
            if ws < words {
                if bs == 0 {
                    for w in ws..words {
                        tmp[w - ws] = bits[w];
                    }
                } else {
                    let lshift = 64 - bs;
                    let mut carry = 0u64;
                    for w in (ws..words).rev() {
                        let src = bits[w];
                        tmp[w - ws] = (src >> bs) | carry;
                        carry = src << lshift;
                    }
                }
            }
        }

        for w in 0..words {
            let shifted = tmp[w];
            let old = bits[w];
            let neww = shifted & !old;
            bits[w] = old | shifted;
            if w == words - 1 {
                bits[w] &= last_mask;
            }
            if neww != 0 {
                let mut y = neww;
                while y != 0 {
                    let tz = y.trailing_zeros() as usize;
                    let pos = w * 64 + tz;
                    if pos <= size && parent_idx[pos] == u32::MAX {
                        parent_idx[pos] = i as u32;
                        let prev = (pos as i64) - x;
                        if prev < 0 || prev > size as i64 {
                            // Ignore impossible predecessor.
                        } else {
                            prev_pos[pos] = prev as u32;
                        }
                    }
                    y &= y - 1;
                }
            }
        }

        let word = target_pos / 64;
        let bit = target_pos % 64;
        if ((bits[word] >> bit) & 1) != 0 {
            // We can stop early; reconstruction uses stored parents.
            break;
        }
    }

    {
        let word = target_pos / 64;
        let bit = target_pos % 64;
        if ((bits[word] >> bit) & 1) == 0 {
            return None;
        }
    }

    // Reconstruct.
    let mut ans: Vec<usize> = Vec::new();
    let mut pos = target_pos;
    let start_pos = offset as usize;
    while pos != start_pos {
        let pi = parent_idx[pos];
        if pi == u32::MAX {
            return None;
        }
        let idx = values[pi as usize].1;
        ans.push(idx);
        pos = prev_pos[pos] as usize;
    }
    Some(ans)
}

/* ---------------- HashMap DP (exact, small/medium) ---------------- */

fn hash_dp(values: &[(i64, usize)], target: i64) -> Option<Vec<usize>> {
    // dp[sum] = (prev_sum, used_original_index)
    let mut dp: HashMap<i64, (i64, usize)> = HashMap::new();
    dp.insert(0, (0, usize::MAX));

    for &(x, orig_idx) in values {
        let keys: Vec<i64> = dp.keys().copied().collect();
        for s in keys {
            let ns = s + x;
            if !dp.contains_key(&ns) {
                dp.insert(ns, (s, orig_idx));
                if ns == target {
                    break;
                }
            }
        }
        if dp.contains_key(&target) {
            break;
        }
    }

    if !dp.contains_key(&target) {
        return None;
    }

    let mut ans: Vec<usize> = Vec::new();
    let mut cur = target;
    while cur != 0 {
        let (prev, idx) = dp.get(&cur).copied()?;
        if idx == usize::MAX {
            break;
        }
        ans.push(idx);
        cur = prev;
    }
    Some(ans)
}

/* ---------------- Deterministic heuristic (finds YES when lucky; never claims NO incorrectly) ---------------- */

fn greedy_try_positive(mut pairs: Vec<(u32, u32)>, target: u32) -> Option<Vec<usize>> {
    pairs.retain(|&(v, _)| v != 0 && v <= target);
    pairs.sort_unstable_by(|a, b| b.0.cmp(&a.0)); // descending by value
    let mut rem = target as i64;
    let mut ans: Vec<usize> = Vec::new();
    for (v, idx) in pairs {
        if rem <= 0 {
            break;
        }
        let vv = v as i64;
        if vv <= rem {
            rem -= vv;
            ans.push(idx as usize);
        }
    }
    if rem == 0 {
        Some(ans)
    } else {
        None
    }
}

fn main() -> io::Result<()> {
    let mut input = FastInput::new()?;
    let n_i64 = match input.next_i64() {
        Some(x) => x,
        None => return output_no(),
    };
    let target = match input.next_i64() {
        Some(x) => x,
        None => return output_no(),
    };
    if n_i64 < 0 {
        return output_no();
    }
    let n = n_i64 as usize;

    if target == 0 {
        // Empty subset always works.
        return output_yes(&[]);
    }

    // Small: exact MiM, store all.
    if n <= MAX_MIM_N {
        let mut vals: Vec<(i64, usize)> = Vec::with_capacity(n);
        for i in 0..n {
            let x = match input.next_i64() {
                Some(v) => v,
                None => 0,
            };
            vals.push((x, i));
        }
        if let Some(ans) = meet_in_middle(&vals, target) {
            return output_yes(&ans);
        }
        return output_no();
    }

    // General reading:
    // - Always track if any negative.
    // - For the non-negative/bitset path, only keep (value, index) for 1..=target when target>0.
    // - For negative-capable exact paths, we only store all values when n is not enormous.
    let mut has_neg = false;
    let mut all_nonneg = true;

    let mut single_hit: Option<usize> = None;

    let mut pairs_pos: Vec<(u32, u32)> = Vec::new(); // (value, index) for positive DP
    let mut values_for_exact: Vec<(i64, usize)> = Vec::new(); // for negative/range/hash DP (limited)

    // If target not representable as u32 or negative, bitset positive DP won't be used.
    let target_u32_opt = if target > 0 && target <= u32::MAX as i64 {
        Some(target as u32)
    } else {
        None
    };

    // Keep all values for exact fallback only up to this n (memory guard).
    let store_all_limit = 220_000usize;
    let store_all = n <= store_all_limit;

    for i in 0..n {
        let x = match input.next_i64() {
            Some(v) => v,
            None => 0,
        };

        if x < 0 {
            has_neg = true;
            all_nonneg = false;
        }
        if x < 0 {
            all_nonneg = false;
        }

        if store_all {
            values_for_exact.push((x, i));
        }

        if let Some(tu) = target_u32_opt {
            if x == target {
                single_hit = Some(i);
            }
            if x > 0 && x <= tu as i64 {
                pairs_pos.push((x as u32, i as u32));
            }
        }
    }

    if let Some(idx) = single_hit {
        return output_yes(&[idx]);
    }

    // If we can do exact DP with negatives and we stored values, try that first.
    if store_all && has_neg {
        // Try exact offset-range bitset if the total sum range is small.
        if let Some(ans) = range_bitset_dp(&values_for_exact, target) {
            return output_yes(&ans);
        }
        // Try exact HashMap DP for moderate sizes.
        if values_for_exact.len() <= HASH_DP_N_LIMIT {
            if let Some(ans) = hash_dp(&values_for_exact, target) {
                return output_yes(&ans);
            }
        }
        // Deterministic MiM on a chosen subset (best-effort exact YES finder).
        // Take up to 44 largest-magnitude elements.
        let mut sample = values_for_exact.clone();
        sample.sort_unstable_by(|a, b| b.0.abs().cmp(&a.0.abs()));
        sample.truncate(MAX_MIM_N);
        if let Some(ans) = meet_in_middle(&sample, target) {
            return output_yes(&ans);
        }
        // Cannot prove NO efficiently here.
        return output_no();
    }

    // Non-negative fast path (exact) when target is manageable.
    if all_nonneg {
        if target < 0 {
            return output_no();
        }
        let tu = match target_u32_opt {
            Some(v) => v,
            None => {
                // Target too large for bitset; try heuristic YES finder.
                if let Some(ans) = greedy_try_positive(pairs_pos, u32::MAX) {
                    return output_yes(&ans);
                }
                return output_no();
            }
        };

        if pairs_pos.is_empty() {
            return output_no();
        }

        if tu <= BITSET_TARGET_LIMIT {
            // Choose bucketing vs sorting grouping.
            let ans = if tu <= BUCKET_TARGET_LIMIT {
                bitset_positive_bucket(&pairs_pos, tu)
            } else {
                bitset_positive_sort(pairs_pos, tu)
            };
            if let Some(indices) = ans {
                return output_yes(&indices);
            }
            return output_no();
        }

        // Target too big for exact bitset DP: try deterministic heuristic(s) that can still find YES.
        // 1) Greedy.
        if let Some(indices) = greedy_try_positive(pairs_pos.clone(), tu) {
            return output_yes(&indices);
        }
        // 2) MiM on top values (best-effort YES finder).
        let mut sample: Vec<(i64, usize)> = pairs_pos
            .into_iter()
            .map(|(v, idx)| (v as i64, idx as usize))
            .collect();
        sample.sort_unstable_by(|a, b| b.0.cmp(&a.0));
        sample.truncate(MAX_MIM_N);
        if let Some(ans) = meet_in_middle(&sample, target) {
            return output_yes(&ans);
        }
        return output_no();
    }

    // Mixed signs but we didn't store all: best-effort (cannot prove NO).
    if store_all {
        // (Should have been handled above), but keep safe.
        if let Some(ans) = hash_dp(&values_for_exact, target) {
            return output_yes(&ans);
        }
    }
    output_no()
}
