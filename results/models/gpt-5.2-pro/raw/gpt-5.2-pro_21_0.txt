#include <iostream>
#include <vector>
#include <array>
#include <deque>
#include <unordered_set>
#include <unordered_map>
#include <queue>
#include <algorithm>
#include <cstdint>
#include <limits>

using namespace std;

static constexpr int MAXD = 20;

static int GD = 0; // global dimension count

struct Pos {
    array<int, MAXD> c{};

    bool operator==(const Pos& other) const noexcept {
        for (int i = 0; i < GD; ++i) {
            if (c[i] != other.c[i]) return false;
        }
        return true;
    }
};

static inline uint64_t splitmix64(uint64_t x) noexcept {
    x += 0x9e3779b97f4a7c15ULL;
    x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ULL;
    x = (x ^ (x >> 27)) * 0x94d049bb133111ebULL;
    return x ^ (x >> 31);
}

struct PosHash {
    size_t operator()(const Pos& p) const noexcept {
        uint64_t h = 0x243f6a8885a308d3ULL;
        for (int i = 0; i < GD; ++i) {
            uint64_t x = (uint64_t)(int64_t)p.c[i];
            h = splitmix64(h ^ (x + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2)));
        }
        return (size_t)h;
    }
};

struct Move {
    int axis = 0;
    int dir = +1; // -1 or +1
};

static inline int manhattan(const Pos& a, const Pos& b) {
    int s = 0;
    for (int i = 0; i < GD; ++i) s += std::abs(a.c[i] - b.c[i]);
    return s;
}

static inline bool inBounds(const Pos& p, const vector<int>& bounds) {
    for (int i = 0; i < GD; ++i) {
        if (p.c[i] < 0 || p.c[i] >= bounds[i]) return false;
    }
    return true;
}

static inline Pos moved(const Pos& p, int axis, int dir) {
    Pos q = p;
    q.c[axis] += dir;
    return q;
}

static inline vector<Move> allMoves() {
    vector<Move> ms;
    ms.reserve((size_t)GD * 2);
    for (int a = 0; a < GD; ++a) {
        ms.push_back({a, -1});
        ms.push_back({a, +1});
    }
    return ms;
}

struct ParentInfo {
    Pos prev{};
    Move mv{};
    bool has = false;
};

static bool isLegalNext(
    const Pos& next,
    const vector<int>& bounds,
    const unordered_set<Pos, PosHash>& obstacles,
    const unordered_set<Pos, PosHash>& occBody,
    const Pos& tail,
    bool willEat
) {
    if (!inBounds(next, bounds)) return false;
    if (obstacles.find(next) != obstacles.end()) return false;

    auto it = occBody.find(next);
    if (it == occBody.end()) return true;

    // In snake, stepping into the current tail cell is allowed ONLY if the tail moves away this turn (i.e., not eating)
    if (!willEat && next == tail) return true;

    return false;
}

static Move chooseSafeMove(
    const Pos& head,
    const vector<int>& bounds,
    const unordered_set<Pos, PosHash>& obstacles,
    const unordered_set<Pos, PosHash>& foods,
    const deque<Pos>& body,
    const unordered_set<Pos, PosHash>& occBody,
    const Pos* maybeTarget
) {
    const vector<Move> ms = allMoves();
    const Pos tail = body.back();

    Move best{0, +1};
    bool found = false;
    int bestScore = std::numeric_limits<int>::min();

    for (const Move& mv : ms) {
        Pos nh = moved(head, mv.axis, mv.dir);
        bool willEat = (foods.find(nh) != foods.end());
        if (!isLegalNext(nh, bounds, obstacles, occBody, tail, willEat)) continue;

        // score = prefer more exits + prefer moving toward target (if any) + avoid immediate dead ends
        int degree = 0;
        for (const Move& mv2 : ms) {
            Pos nn = moved(nh, mv2.axis, mv2.dir);
            bool eat2 = (foods.find(nn) != foods.end());
            // approximate occupancy after first move:
            // - new head occupies nh
            // - if not eating, tail cell freed
            // We'll do a conservative check: avoid obstacles and avoid current body except allow tail if not eating.
            // Also avoid going back into nh itself is naturally excluded by occ check.
            bool ok = true;
            if (!inBounds(nn, bounds)) ok = false;
            else if (obstacles.find(nn) != obstacles.end()) ok = false;
            else {
                // if nn is current body cell:
                if (occBody.find(nn) != occBody.end()) {
                    if (!( (!willEat) && nn == tail )) ok = false;
                }
                // also nn could be nh (new head), which would be self-collision in 1 step
                if (nn == nh) ok = false;
            }
            if (ok) degree++;
        }

        int distScore = 0;
        if (maybeTarget) {
            int d0 = manhattan(head, *maybeTarget);
            int d1 = manhattan(nh, *maybeTarget);
            distScore = (d0 - d1); // positive if closer
        }

        int score = degree * 100 + distScore * 10;
        if (degree == 0) score -= 1000000; // avoid forcing death next turn

        if (!found || score > bestScore) {
            found = true;
            bestScore = score;
            best = mv;
        }
    }

    if (!found) return {0, +1}; // no legal move; output something
    return best;
}

static bool astarPlan(
    const Pos& start,
    const Pos& goal,
    const vector<int>& bounds,
    const unordered_set<Pos, PosHash>& obstacles,
    const unordered_set<Pos, PosHash>& occBody,
    const Pos& tailTreatAsFree,
    int nodeCap,
    vector<Move>& outPath
) {
    outPath.clear();
    if (start == goal) return true;

    struct QNode {
        int f, g;
        Pos p;
    };
    struct Cmp {
        bool operator()(const QNode& a, const QNode& b) const {
            if (a.f != b.f) return a.f > b.f;
            return a.g > b.g;
        }
    };

    priority_queue<QNode, vector<QNode>, Cmp> pq;
    unordered_map<Pos, int, PosHash> gScore;
    unordered_map<Pos, ParentInfo, PosHash> parent;

    gScore.reserve((size_t)nodeCap * 2);
    parent.reserve((size_t)nodeCap * 2);

    auto h = [&](const Pos& p) { return manhattan(p, goal); };

    gScore[start] = 0;
    pq.push({h(start), 0, start});

    const vector<Move> ms = allMoves();
    int expanded = 0;

    while (!pq.empty() && expanded < nodeCap) {
        QNode cur = pq.top();
        pq.pop();

        auto itg = gScore.find(cur.p);
        if (itg == gScore.end() || itg->second != cur.g) continue; // outdated

        expanded++;
        if (cur.p == goal) {
            // reconstruct
            vector<Move> rev;
            Pos p = goal;
            while (!(p == start)) {
                auto itp = parent.find(p);
                if (itp == parent.end() || !itp->second.has) break;
                rev.push_back(itp->second.mv);
                p = itp->second.prev;
            }
            if (!(p == start)) return false;
            reverse(rev.begin(), rev.end());
            outPath = std::move(rev);
            return true;
        }

        for (const Move& mv : ms) {
            Pos np = moved(cur.p, mv.axis, mv.dir);
            if (!inBounds(np, bounds)) continue;
            if (obstacles.find(np) != obstacles.end()) continue;

            // A* uses static occupancy approximation:
            // treat body cells blocked, but allow current tail cell as free (it will likely move away).
            if (occBody.find(np) != occBody.end() && !(np == tailTreatAsFree)) continue;

            int ng = cur.g + 1;
            auto it = gScore.find(np);
            if (it == gScore.end() || ng < it->second) {
                gScore[np] = ng;
                ParentInfo pi;
                pi.prev = cur.p;
                pi.mv = mv;
                pi.has = true;
                parent[np] = pi;
                int nf = ng + h(np);
                pq.push({nf, ng, np});
            }
        }
    }
    return false;
}

static bool buildInitialBody(
    int length,
    const Pos& head,
    const vector<int>& bounds,
    const unordered_set<Pos, PosHash>& obstacles,
    const unordered_set<Pos, PosHash>& foods,
    deque<Pos>& bodyOut
) {
    bodyOut.clear();
    if (length <= 1) {
        bodyOut.push_back(head);
        return true;
    }

    // Try a straight line body: head, head - dir*e_axis, head - 2*dir*e_axis, ...
    for (int axis = 0; axis < GD; ++axis) {
        for (int dir : {+1, -1}) {
            bool ok = true;
            deque<Pos> tmp;
            unordered_set<Pos, PosHash> seen;
            seen.reserve((size_t)length * 2);

            for (int k = 0; k < length; ++k) {
                Pos p = head;
                p.c[axis] -= k * dir;
                if (!inBounds(p, bounds)) { ok = false; break; }
                if (obstacles.find(p) != obstacles.end()) { ok = false; break; }
                // Prefer not to start with body on food (except possibly head), but allow if necessary.
                if (k != 0 && foods.find(p) != foods.end()) { ok = false; break; }
                if (seen.find(p) != seen.end()) { ok = false; break; }
                seen.insert(p);
                tmp.push_back(p);
            }
            if (ok) {
                bodyOut = std::move(tmp); // head at front already
                return true;
            }
        }
    }

    // Fallback: length=1 only
    bodyOut.push_back(head);
    return false;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int D;
    int maxTurns;
    if (!(cin >> D >> maxTurns)) return 0;
    GD = D;

    vector<int> bounds(D);
    for (int i = 0; i < D; ++i) cin >> bounds[i];

    int snakeLen;
    Pos head{};
    cin >> snakeLen;
    for (int i = 0; i < D; ++i) cin >> head.c[i];

    int F;
    cin >> F;
    vector<Pos> foodsVec;
    foodsVec.reserve((size_t)F);
    unordered_set<Pos, PosHash> foods;
    foods.reserve((size_t)F * 2 + 8);
    for (int i = 0; i < F; ++i) {
        Pos p{};
        for (int j = 0; j < D; ++j) cin >> p.c[j];
        foodsVec.push_back(p);
        foods.insert(p);
    }

    int O;
    cin >> O;
    unordered_set<Pos, PosHash> obstacles;
    obstacles.reserve((size_t)O * 2 + 8);
    for (int i = 0; i < O; ++i) {
        Pos p{};
        for (int j = 0; j < D; ++j) cin >> p.c[j];
        obstacles.insert(p);
    }

    // Build initial body.
    deque<Pos> body; // front=head, back=tail
    buildInitialBody(snakeLen, head, bounds, obstacles, foods, body);

    unordered_set<Pos, PosHash> occBody;
    occBody.reserve((size_t)body.size() * 2 + 8);
    for (const Pos& p : body) occBody.insert(p);

    // If head starts on food, eat it immediately.
    if (foods.find(head) != foods.end()) {
        foods.erase(head);
    }

    // Planning state
    bool haveTarget = false;
    Pos target{};
    deque<Move> plan; // cached path moves toward target

    auto chooseTargetNearest = [&]() -> bool {
        if (foods.empty()) return false;
        int bestDist = std::numeric_limits<int>::max();
        bool found = false;
        Pos best{};
        for (const Pos& fpos : foods) {
            int d = manhattan(head, fpos);
            if (!found || d < bestDist) {
                found = true;
                bestDist = d;
                best = fpos;
            }
        }
        if (found) {
            target = best;
            haveTarget = true;
            return true;
        }
        return false;
    };

    auto ensureTarget = [&]() {
        if (haveTarget) {
            if (foods.find(target) == foods.end()) haveTarget = false;
        }
        if (!haveTarget) {
            chooseTargetNearest();
        }
    };

    auto tryMakePlan = [&]() -> bool {
        plan.clear();
        ensureTarget();
        if (!haveTarget) return false;

        // Node cap: keep bounded for huge spaces, but allow more in small/medium.
        int nodeCap = 80000;
        // If very small bounds product, allow larger.
        long long prod = 1;
        bool prodTooBig = false;
        for (int i = 0; i < D; ++i) {
            if (bounds[i] <= 0) { prodTooBig = true; break; }
            if (prod > (long long)3e6 / bounds[i]) { prodTooBig = true; break; }
            prod *= bounds[i];
        }
        if (!prodTooBig) {
            nodeCap = (int)min<long long>(250000, max<long long>(80000, prod));
        }

        vector<Move> path;
        const Pos tail = body.back();
        bool ok = astarPlan(head, target, bounds, obstacles, occBody, tail, nodeCap, path);
        if (!ok || path.empty()) return false;

        for (const Move& mv : path) plan.push_back(mv);
        return true;
    };

    const vector<Move> ms = allMoves();

    for (int turn = 0; turn < maxTurns; ++turn) {
        // If food done, just survive (still output moves up to maxTurns).
        ensureTarget();

        Move chosen{0, +1};

        // Attempt to use cached plan if possible
        bool usedPlan = false;
        if (!plan.empty()) {
            Move mv = plan.front();
            Pos nh = moved(head, mv.axis, mv.dir);
            bool willEat = (foods.find(nh) != foods.end());
            Pos tail = body.back();
            if (isLegalNext(nh, bounds, obstacles, occBody, tail, willEat)) {
                chosen = mv;
                usedPlan = true;
                plan.pop_front();
            } else {
                plan.clear(); // invalidate
            }
        }

        if (!usedPlan) {
            bool planned = false;
            if (haveTarget) planned = tryMakePlan();

            if (planned && !plan.empty()) {
                Move mv = plan.front();
                Pos nh = moved(head, mv.axis, mv.dir);
                bool willEat = (foods.find(nh) != foods.end());
                Pos tail = body.back();
                if (isLegalNext(nh, bounds, obstacles, occBody, tail, willEat)) {
                    chosen = mv;
                    plan.pop_front();
                } else {
                    plan.clear();
                    chosen = chooseSafeMove(head, bounds, obstacles, foods, body, occBody, haveTarget ? &target : nullptr);
                }
            } else {
                chosen = chooseSafeMove(head, bounds, obstacles, foods, body, occBody, haveTarget ? &target : nullptr);
            }
        }

        // Output move
        cout << chosen.axis << " " << (chosen.dir < 0 ? -1 : 1) << "\n";

        // Simulate move locally (so subsequent moves are consistent)
        Pos nextHead = moved(head, chosen.axis, chosen.dir);
        bool willEat = (foods.find(nextHead) != foods.end());
        Pos tail = body.back();

        // If illegal, we still simulate a death-stop by not changing state (cannot do more).
        // But we already tried hard to avoid it.
        if (!isLegalNext(nextHead, bounds, obstacles, occBody, tail, willEat)) {
            // continue outputting something; state frozen
            plan.clear();
            continue;
        }

        // apply
        body.push_front(nextHead);
        occBody.insert(nextHead);
        head = nextHead;

        if (willEat) {
            foods.erase(nextHead);
            // grew: don't pop tail
            plan.clear();       // target may change / path invalid
            haveTarget = false; // force re-pick
        } else {
            Pos oldTail = body.back();
            body.pop_back();
            occBody.erase(oldTail);
        }
    }

    return 0;
}
