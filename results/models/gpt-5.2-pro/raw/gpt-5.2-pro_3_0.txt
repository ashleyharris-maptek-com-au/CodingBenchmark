use std::cmp::Ordering;
use std::io::{self, Read, Write};

#[derive(Clone, Copy, Debug)]
struct BBox {
    minx: f32,
    maxx: f32,
    miny: f32,
    maxy: f32,
}
impl BBox {
    fn width(&self) -> f32 {
        self.maxx - self.minx
    }
    fn height(&self) -> f32 {
        self.maxy - self.miny
    }
}

struct Scratch {
    da: Vec<i32>,
    db: Vec<i32>,
    dc: Vec<i32>,
    dd: Vec<i32>,

    tmpx: Vec<f32>,
    tmpy: Vec<f32>,

    fx: Vec<f32>,
    fy: Vec<f32>,
}

impl Scratch {
    fn new() -> Self {
        Self {
            da: Vec::new(),
            db: Vec::new(),
            dc: Vec::new(),
            dd: Vec::new(),
            tmpx: Vec::new(),
            tmpy: Vec::new(),
            fx: Vec::new(),
            fy: Vec::new(),
        }
    }
    fn ensure_n(&mut self, n: usize) {
        self.da.resize(n, 0);
        self.db.resize(n, 0);
        self.dc.resize(n, 0);
        self.dd.resize(n, 0);
        self.tmpx.resize(n, 0.0);
        self.tmpy.resize(n, 0.0);
        self.fx.resize(n, 0.0);
        self.fy.resize(n, 0.0);
    }
}

#[inline]
fn splitmix64(mut x: u64) -> u64 {
    x = x.wrapping_add(0x9E3779B97F4A7C15);
    let mut z = x;
    z = (z ^ (z >> 30)).wrapping_mul(0xBF58476D1CE4E5B9);
    z = (z ^ (z >> 27)).wrapping_mul(0x94D049BB133111EB);
    z ^ (z >> 31)
}
#[inline]
fn jitter01_from_id(id: u32, salt: u64) -> f32 {
    // deterministic in [0,1)
    let h = splitmix64((id as u64) ^ salt);
    let v = (h >> 40) as u32; // 24 bits
    (v as f32) * (1.0 / ((1u32 << 24) as f32))
}
#[inline]
fn jitter_signed(id: u32, salt: u64) -> f32 {
    jitter01_from_id(id, salt) - 0.5
}

struct FastInput {
    data: Vec<u8>,
    idx: usize,
}
impl FastInput {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data).unwrap();
        Self { data, idx: 0 }
    }
    #[inline]
    fn skip_ws(&mut self) {
        while self.idx < self.data.len() && self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
    }
    #[inline]
    fn next_usize(&mut self) -> Option<usize> {
        self.skip_ws();
        if self.idx >= self.data.len() {
            return None;
        }
        let mut x: usize = 0;
        while self.idx < self.data.len() {
            let c = self.data[self.idx];
            if !c.is_ascii_digit() {
                break;
            }
            x = x.saturating_mul(10).saturating_add((c - b'0') as usize);
            self.idx += 1;
        }
        Some(x)
    }
}

fn bfs_fill_global(
    start: usize,
    offsets: &[usize],
    adj: &[u32],
    seen: &mut [u32],
    dist: &mut [i32],
    gen: u32,
) -> (usize, i32) {
    let mut q: Vec<u32> = Vec::new();
    q.push(start as u32);
    let mut head = 0usize;

    seen[start] = gen;
    dist[start] = 0;

    let mut far = start;
    let mut far_d: i32 = 0;

    while head < q.len() {
        let u = q[head] as usize;
        head += 1;

        let du = dist[u];
        if du > far_d {
            far_d = du;
            far = u;
        }

        let a = offsets[u];
        let b = offsets[u + 1];
        for &vv in &adj[a..b] {
            let v = vv as usize;
            if seen[v] != gen {
                seen[v] = gen;
                dist[v] = du + 1;
                q.push(vv);
            }
        }
    }
    (far, far_d)
}

fn copy_dist_for_nodes(nodes: &[u32], dist_global: &[i32], out: &mut [i32]) {
    for (i, &u) in nodes.iter().enumerate() {
        out[i] = dist_global[u as usize];
    }
}

fn bbox_for_nodes(nodes: &[u32], x: &[f32], y: &[f32]) -> BBox {
    let mut minx = f32::INFINITY;
    let mut maxx = f32::NEG_INFINITY;
    let mut miny = f32::INFINITY;
    let mut maxy = f32::NEG_INFINITY;

    for &u in nodes {
        let i = u as usize;
        let xi = x[i];
        let yi = y[i];
        if xi < minx {
            minx = xi;
        }
        if xi > maxx {
            maxx = xi;
        }
        if yi < miny {
            miny = yi;
        }
        if yi > maxy {
            maxy = yi;
        }
    }
    if !minx.is_finite() {
        minx = 0.0;
        maxx = 0.0;
        miny = 0.0;
        maxy = 0.0;
    }
    BBox {
        minx,
        maxx,
        miny,
        maxy,
    }
}

fn laplacian_smooth(
    nodes: &[u32],
    offsets: &[usize],
    adj: &[u32],
    x: &mut [f32],
    y: &mut [f32],
    scratch: &mut Scratch,
    iters: usize,
    alpha: f32,
) {
    if nodes.len() <= 1 || iters == 0 {
        return;
    }
    scratch.ensure_n(nodes.len());

    for _ in 0..iters {
        for (li, &uu) in nodes.iter().enumerate() {
            let u = uu as usize;
            let a = offsets[u];
            let b = offsets[u + 1];
            let mut sx = 0.0f32;
            let mut sy = 0.0f32;
            let mut cnt = 0u32;

            for &vv in &adj[a..b] {
                let v = vv as usize;
                sx += x[v];
                sy += y[v];
                cnt += 1;
            }
            if cnt > 0 {
                let inv = 1.0 / (cnt as f32);
                let ax = sx * inv;
                let ay = sy * inv;
                scratch.tmpx[li] = x[u] * (1.0 - alpha) + ax * alpha;
                scratch.tmpy[li] = y[u] * (1.0 - alpha) + ay * alpha;
            } else {
                scratch.tmpx[li] = x[u];
                scratch.tmpy[li] = y[u];
            }
        }
        for (li, &uu) in nodes.iter().enumerate() {
            let u = uu as usize;
            x[u] = scratch.tmpx[li];
            y[u] = scratch.tmpy[li];
        }
    }
}

fn force_directed_fr(
    nodes: &[u32],
    offsets: &[usize],
    adj: &[u32],
    x: &mut [f32],
    y: &mut [f32],
    scratch: &mut Scratch,
    mark: &mut [i32],
    iters: usize,
    base_k: f32,
) {
    let n = nodes.len();
    if n <= 1 || iters == 0 {
        return;
    }
    scratch.ensure_n(n);

    // Build local index map in `mark`
    for (i, &u) in nodes.iter().enumerate() {
        mark[u as usize] = i as i32;
    }

    let mut t = (n as f32).sqrt() * base_k * 0.15;
    let k = base_k.max(1e-3);
    let kk = k * k;

    for _ in 0..iters {
        // reset forces
        for i in 0..n {
            scratch.fx[i] = 0.0;
            scratch.fy[i] = 0.0;
        }

        // repulsive O(n^2)
        for i in 0..n {
            let ui = nodes[i] as usize;
            let xi = x[ui];
            let yi = y[ui];
            for j in (i + 1)..n {
                let uj = nodes[j] as usize;
                let dx = xi - x[uj];
                let dy = yi - y[uj];
                let dist2 = dx * dx + dy * dy + 1e-6;
                let dist = dist2.sqrt();
                let f = kk / dist; // FR repulsion
                let rx = dx / dist * f;
                let ry = dy / dist * f;
                scratch.fx[i] += rx;
                scratch.fy[i] += ry;
                scratch.fx[j] -= rx;
                scratch.fy[j] -= ry;
            }
        }

        // attractive along edges (count each undirected edge once via u < v)
        for &uu in nodes {
            let u = uu as usize;
            let iu = mark[u] as i32;
            if iu < 0 {
                continue;
            }
            let iu = iu as usize;
            let a = offsets[u];
            let b = offsets[u + 1];
            for &vv in &adj[a..b] {
                let v = vv as usize;
                if u >= v {
                    continue;
                }
                let iv = mark[v];
                if iv < 0 {
                    continue;
                }
                let iv = iv as usize;

                let dx = x[u] - x[v];
                let dy = y[u] - y[v];
                let dist2 = dx * dx + dy * dy + 1e-6;
                let dist = dist2.sqrt();
                let f = dist2 / k; // FR attraction
                let ax = dx / dist * f;
                let ay = dy / dist * f;

                scratch.fx[iu] -= ax;
                scratch.fy[iu] -= ay;
                scratch.fx[iv] += ax;
                scratch.fy[iv] += ay;
            }
        }

        // apply moves with temperature
        for i in 0..n {
            let u = nodes[i] as usize;
            let dx = scratch.fx[i];
            let dy = scratch.fy[i];
            let disp2 = dx * dx + dy * dy;
            if disp2 <= 1e-12 {
                continue;
            }
            let disp = disp2.sqrt();
            let step = disp.min(t);
            x[u] += dx / disp * step;
            y[u] += dy / disp * step;
        }

        t *= 0.90;
        if t < 1e-3 {
            break;
        }
    }

    // clear marks
    for &u in nodes {
        mark[u as usize] = -1;
    }
}

fn layout_component(
    nodes: &[u32],
    offsets: &[usize],
    adj: &[u32],
    seen: &mut [u32],
    dist: &mut [i32],
    gen: &mut u32,
    x: &mut [f32],
    y: &mut [f32],
    scratch: &mut Scratch,
    mark: &mut [i32],
    force_budget: &mut i64,
) -> BBox {
    let n = nodes.len();
    if n == 0 {
        return BBox {
            minx: 0.0,
            maxx: 0.0,
            miny: 0.0,
            maxy: 0.0,
        };
    }
    if n == 1 {
        let u = nodes[0] as usize;
        x[u] = 0.0;
        y[u] = 0.0;
        return BBox {
            minx: 0.0,
            maxx: 0.0,
            miny: 0.0,
            maxy: 0.0,
        };
    }

    scratch.ensure_n(n);

    // Landmark BFS embedding (4 BFS total)
    let seed = nodes[0] as usize;

    *gen = gen.wrapping_add(1);
    let (a, _) = bfs_fill_global(seed, offsets, adj, seen, dist, *gen);

    *gen = gen.wrapping_add(1);
    let (b, _) = bfs_fill_global(a, offsets, adj, seen, dist, *gen);
    copy_dist_for_nodes(nodes, dist, &mut scratch.da);

    *gen = gen.wrapping_add(1);
    let (_tmp, _) = bfs_fill_global(b, offsets, adj, seen, dist, *gen);
    copy_dist_for_nodes(nodes, dist, &mut scratch.db);

    let mut best_i = 0usize;
    let mut best_val: i64 = i64::MIN;
    for i in 0..n {
        let v = (scratch.da[i] as i64) + (scratch.db[i] as i64);
        if v > best_val {
            best_val = v;
            best_i = i;
        }
    }
    let c = nodes[best_i] as usize;

    *gen = gen.wrapping_add(1);
    let (d, _) = bfs_fill_global(c, offsets, adj, seen, dist, *gen);
    copy_dist_for_nodes(nodes, dist, &mut scratch.dc);

    *gen = gen.wrapping_add(1);
    let (_tmp2, _) = bfs_fill_global(d, offsets, adj, seen, dist, *gen);
    copy_dist_for_nodes(nodes, dist, &mut scratch.dd);

    // Convert distances -> coordinates
    let mut max_abs_x = 0.0f32;
    let mut max_abs_y = 0.0f32;
    for i in 0..n {
        let xi = (scratch.db[i] - scratch.da[i]) as f32;
        let yi = (scratch.dd[i] - scratch.dc[i]) as f32;
        scratch.tmpx[i] = xi;
        scratch.tmpy[i] = yi;
        max_abs_x = max_abs_x.max(xi.abs());
        max_abs_y = max_abs_y.max(yi.abs());
    }

    // Normalize and scale to roughly sqrt(n)
    let base = (n as f32).sqrt() * 10.0;
    if max_abs_x < 1e-6 {
        max_abs_x = 1.0;
    }
    if max_abs_y < 1e-6 {
        // fallback y: use one landmark distance + jitter to separate ties
        let mut maxd = 1i32;
        for &d0 in &scratch.da {
            maxd = maxd.max(d0);
        }
        let inv = 1.0 / (maxd as f32).max(1.0);
        for i in 0..n {
            scratch.tmpy[i] = (scratch.da[i] as f32) * inv * 2.0 - 1.0;
        }
        max_abs_y = 1.0;
    }

    for i in 0..n {
        let u = nodes[i] as usize;
        let mut xi = scratch.tmpx[i] / max_abs_x * base;
        let mut yi = scratch.tmpy[i] / max_abs_y * base;

        // deterministic tiny jitter (break symmetries)
        let ju = nodes[i];
        xi += jitter_signed(ju, 0xA1B2C3D4E5F60718) * (base * 0.01);
        yi += jitter_signed(ju, 0x1122334455667788) * (base * 0.01);

        x[u] = xi;
        y[u] = yi;
    }

    // Light smoothing for mid/large graphs; more for medium components
    let smooth_iters = if n <= 50_000 { 3 } else { 1 };
    laplacian_smooth(nodes, offsets, adj, x, y, scratch, smooth_iters, 0.45);

    // Optional force-directed refinement for small components under a global budget
    // (O(n^2) repulsion is great for tiny graphs, dangerous for many components.)
    if n <= 1200 {
        let mut iters = if n <= 250 { 40 } else if n <= 700 { 24 } else { 14 };
        let est_cost = (n as i64) * (n as i64) * (iters as i64);
        if est_cost > *force_budget {
            // Reduce iterations to stay within budget
            let denom = (n as i64) * (n as i64);
            if denom > 0 {
                iters = ((*force_budget / denom) as usize).min(iters);
            } else {
                iters = 0;
            }
        }
        if iters >= 6 {
            *force_budget -= (n as i64) * (n as i64) * (iters as i64);
            force_directed_fr(nodes, offsets, adj, x, y, scratch, mark, iters, 10.0);
            // a tiny post-smooth to reduce edge wiggle
            laplacian_smooth(nodes, offsets, adj, x, y, scratch, 1, 0.25);
        }
    }

    bbox_for_nodes(nodes, x, y)
}

fn main() {
    let mut inp = FastInput::new();
    let n = match inp.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m = inp.next_usize().unwrap_or(0);

    // Read edges, ignore self-loops for layout.
    let mut deg = vec![0u32; n];
    let mut eu: Vec<u32> = Vec::with_capacity(m);
    let mut ev: Vec<u32> = Vec::with_capacity(m);

    for _ in 0..m {
        let Some(uu) = inp.next_usize() else { break };
        let Some(vv) = inp.next_usize() else { break };
        if uu >= n || vv >= n {
            continue;
        }
        if uu == vv {
            continue;
        }
        let u = uu as u32;
        let v = vv as u32;
        eu.push(u);
        ev.push(v);
        deg[uu] += 1;
        deg[vv] += 1;
    }

    // CSR adjacency
    let mut offsets = vec![0usize; n + 1];
    for i in 0..n {
        offsets[i + 1] = offsets[i] + deg[i] as usize;
    }
    let mut adj = vec![0u32; offsets[n]];
    let mut cur = offsets.clone();
    for i in 0..eu.len() {
        let u = eu[i] as usize;
        let v = ev[i] as usize;

        let pu = cur[u];
        adj[pu] = v as u32;
        cur[u] += 1;

        let pv = cur[v];
        adj[pv] = u as u32;
        cur[v] += 1;
    }

    // Find connected components
    let mut visited = vec![false; n];
    let mut comp_offsets: Vec<usize> = Vec::new();
    let mut comp_sizes: Vec<usize> = Vec::new();
    let mut comp_nodes: Vec<u32> = Vec::with_capacity(n);

    let mut stack: Vec<usize> = Vec::new();

    for s in 0..n {
        if visited[s] {
            continue;
        }
        comp_offsets.push(comp_nodes.len());

        visited[s] = true;
        stack.clear();
        stack.push(s);

        while let Some(u) = stack.pop() {
            comp_nodes.push(u as u32);
            let a = offsets[u];
            let b = offsets[u + 1];
            for &vv in &adj[a..b] {
                let v = vv as usize;
                if !visited[v] {
                    visited[v] = true;
                    stack.push(v);
                }
            }
        }

        let start = *comp_offsets.last().unwrap();
        comp_sizes.push(comp_nodes.len() - start);
    }

    let comps = comp_sizes.len();

    // Global arrays
    let mut x = vec![0.0f32; n];
    let mut y = vec![0.0f32; n];

    let mut seen = vec![0u32; n];
    let mut dist = vec![0i32; n];
    let mut gen: u32 = 1;

    let mut scratch = Scratch::new();
    let mut mark = vec![-1i32; n];

    // Budget for O(n^2) refinement (approx pair-ops), tuned for 30s worst-case.
    let mut force_budget: i64 = 180_000_000;

    // Layout each component around origin, store bbox
    let mut bboxes: Vec<BBox> = Vec::with_capacity(comps);
    for ci in 0..comps {
        let start = comp_offsets[ci];
        let size = comp_sizes[ci];
        let nodes = &comp_nodes[start..start + size];
        let bb = layout_component(
            nodes,
            &offsets,
            &adj,
            &mut seen,
            &mut dist,
            &mut gen,
            &mut x,
            &mut y,
            &mut scratch,
            &mut mark,
            &mut force_budget,
        );
        bboxes.push(bb);
    }

    // Pack components so they don't overlap too badly
    let total_n = n as f32;
    let max_row_width = total_n.sqrt() * 200.0 + 1500.0;
    let margin = 60.0f32;

    let mut order: Vec<usize> = (0..comps).collect();
    order.sort_by(|&a, &b| {
        let aa = bboxes[a].width() * bboxes[a].height();
        let bb = bboxes[b].width() * bboxes[b].height();
        bb.partial_cmp(&aa).unwrap_or(Ordering::Equal)
    });

    let mut cur_x = 0.0f32;
    let mut cur_y = 0.0f32;
    let mut row_h = 0.0f32;

    for &ci in &order {
        let bb = bboxes[ci];
        let w = bb.width().max(0.0);
        let h = bb.height().max(0.0);

        if cur_x > 0.0 && cur_x + w + margin > max_row_width {
            cur_x = 0.0;
            cur_y += row_h + margin;
            row_h = 0.0;
        }

        let ox = cur_x - bb.minx + margin;
        let oy = cur_y - bb.miny + margin;

        let start = comp_offsets[ci];
        let size = comp_sizes[ci];
        for &u in &comp_nodes[start..start + size] {
            let i = u as usize;
            x[i] += ox;
            y[i] += oy;
        }

        cur_x += w + margin;
        row_h = row_h.max(h);
    }

    // Output: N lines "x y"
    let mut out = io::BufWriter::new(io::stdout());
    for i in 0..n {
        // Use a fixed precision; judge typically accepts any float format
        writeln!(out, "{:.6} {:.6}", x[i], y[i]).unwrap();
    }
}
