; SPIR-V
; Version: 1.0

OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inColor %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float

%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%fn = OpTypeFunction %void

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%f0 = OpConstant %float 0
%f1 = OpConstant %float 1
%f0_05 = OpConstant %float 0.05
%f0_1 = OpConstant %float 0.1
%f32 = OpConstant %float 32

%zero3 = OpConstantComposite %v3float %f0 %f0 %f0
%one3 = OpConstantComposite %v3float %f1 %f1 %f1
%ambient = OpConstantComposite %v3float %f0_1 %f0_05 %f0_05

%inWorldPos = OpVariable %_ptr_Input_v3float Input
%inNormal = OpVariable %_ptr_Input_v3float Input
%inColor = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output

%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%main = OpFunction %void None %fn
%entry = OpLabel

%worldPos = OpLoad %v3float %inWorldPos
%n_in = OpLoad %v3float %inNormal
%N = OpExtInst %v3float %glsl Normalize %n_in
%vtxColor = OpLoad %v3float %inColor

%lightPos_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lightPos4 = OpLoad %v4float %lightPos_ptr
%lightPos3 = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2

%cameraPos_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_4
%cameraPos4 = OpLoad %v4float %cameraPos_ptr
%cameraPos3 = OpVectorShuffle %v3float %cameraPos4 %cameraPos4 0 1 2

%Ldir = OpFSub %v3float %lightPos3 %worldPos
%L = OpExtInst %v3float %glsl Normalize %Ldir

%Vdir = OpFSub %v3float %cameraPos3 %worldPos
%V = OpExtInst %v3float %glsl Normalize %Vdir

%NdotL = OpDot %float %N %L
%NdotL_pos = OpExtInst %float %glsl FMax %NdotL %f0
%diffuse = OpVectorTimesScalar %v3float %vtxColor %NdotL_pos

%negL = OpFNegate %v3float %L
%R = OpExtInst %v3float %glsl Reflect %negL %N
%RdotV = OpDot %float %R %V
%RdotV_pos = OpExtInst %float %glsl FMax %RdotV %f0
%specPow = OpExtInst %float %glsl Pow %RdotV_pos %f32
%specular = OpVectorTimesScalar %v3float %one3 %specPow

%amb_diff = OpFAdd %v3float %ambient %diffuse
%sum = OpFAdd %v3float %amb_diff %specular
%clamped = OpExtInst %v3float %glsl FClamp %sum %zero3 %one3

%r = OpCompositeExtract %float %clamped 0
%g = OpCompositeExtract %float %clamped 1
%b = OpCompositeExtract %float %clamped 2
%out = OpCompositeConstruct %v4float %r %g %b %f1
OpStore %outColor %out

OpReturn
OpFunctionEnd
