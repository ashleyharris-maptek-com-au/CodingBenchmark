```csharp
using System;
using System.IO;

public static class Program
{
    // Chinese Postman (route inspection) solver that scales to very large graphs.
    // We only need ANY closed walk starting/ending at 0 that traverses every edge at least once.
    //
    // Strategy (fast, scalable, always works for connected undirected graphs):
    // Convert each undirected edge {u,v} into TWO directed arcs (u->v) and (v->u).
    // Then every node has indegree == outdegree, so an Eulerian circuit exists (if the graph is connected).
    // Compute an Euler circuit with iterative Hierholzer in O(N+M) time and O(N+M) memory.
    //
    // Output: the node sequence of the Euler circuit (no length prefix).

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        int n = fs.NextInt();
        int m = fs.NextInt();

        if (n <= 0)
        {
            // Degenerate: nothing to do.
            return;
        }

        if (m <= 0)
        {
            // No edges: route is just staying at 0 (starts and ends at 0).
            Console.Write("0");
            return;
        }

        int arcCount = checked(2 * m);

        int[] head = new int[n];
        Array.Fill(head, -1);

        int[] to = new int[arcCount];
        int[] next = new int[arcCount];

        int idx = 0;
        void AddArc(int u, int v)
        {
            to[idx] = v;
            next[idx] = head[u];
            head[u] = idx;
            idx++;
        }

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            _ = fs.NextInt(); // weight ignored (we do not optimize total length; only need a valid closed walk)
            AddArc(u, v);
            AddArc(v, u);
        }

        // Iterative Hierholzer for directed graph
        int[] cur = new int[n];
        Array.Copy(head, cur, n);

        // byte[] is a bit smaller/faster than bool[] for huge instances.
        byte[] used = new byte[arcCount];

        int[] stack = new int[arcCount + 1];
        int sp = 0;
        stack[sp++] = 0;

        int[] routeRev = new int[arcCount + 1]; // nodes in reverse order
        int rlen = 0;

        while (sp > 0)
        {
            int v = stack[sp - 1];
            int e = cur[v];

            while (e != -1 && used[e] != 0) e = next[e];
            cur[v] = e;

            if (e == -1)
            {
                // No more outgoing unused arcs: add vertex to circuit
                routeRev[rlen++] = v;
                sp--;
            }
            else
            {
                // Use arc e
                used[e] = 1;
                cur[v] = next[e];   // advance adjacency pointer
                stack[sp++] = to[e];
            }
        }

        // If the graph is connected (in the undirected sense) and node 0 is in that component,
        // then rlen == arcCount + 1 and all edges are covered.
        // For disconnected graphs, covering all edges in one closed walk is impossible; this will cover
        // the component reachable from 0.

        var fw = new FastWriter(Console.OpenStandardOutput());

        for (int i = rlen - 1; i >= 0; i--)
        {
            fw.WriteInt(routeRev[i]);
            if (i != 0) fw.WriteByte((byte)' ');
        }

        fw.WriteByte((byte)'\n');
        fw.Flush();
    }

    // -------- Fast I/O --------

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buf;
        private int _len, _ptr;

        public FastScanner(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buf = new byte[bufferSize];
        }

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do { c = Read(); } while (c <= (byte)' ');

            int sign = 1;
            if (c == (byte)'-')
            {
                sign = -1;
                c = Read();
            }

            int x = 0;
            while (c > (byte)' ')
            {
                x = (x * 10) + (c - (byte)'0');
                c = Read();
            }
            return x * sign;
        }
    }

    private sealed class FastWriter
    {
        private readonly Stream _stream;
        private readonly byte[] _buf;
        private int _ptr;

        public FastWriter(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buf = new byte[bufferSize];
        }

        public void WriteByte(byte b)
        {
            if (_ptr >= _buf.Length) Flush();
            _buf[_ptr++] = b;
        }

        public void WriteInt(int x)
        {
            if (x == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            if (x < 0)
            {
                WriteByte((byte)'-');
                x = -x;
            }

            // Write digits to a small temp buffer in reverse, then flush them.
            Span<byte> tmp = stackalloc byte[11]; // enough for int32
            int n = 0;
            while (x > 0)
            {
                int q = x / 10;
                tmp[n++] = (byte)('0' + (x - q * 10));
                x = q;
            }
            for (int i = n - 1; i >= 0; i--) WriteByte(tmp[i]);
        }

        public void Flush()
        {
            if (_ptr > 0)
            {
                _stream.Write(_buf, 0, _ptr);
                _ptr = 0;
            }
        }
    }
}
```
