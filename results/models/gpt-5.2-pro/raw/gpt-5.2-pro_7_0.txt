struct FastScanner {
    static constexpr size_t BUFSZ = 1 << 20;
    size_t idx = 0, size = 0;
    char buf[BUFSZ];

    inline char readChar() {
        if (idx >= size) {
            size = std::fread(buf, 1, BUFSZ, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    template <class T>
    bool readInt(T& out) {
        char c;
        do {
            c = readChar();
            if (!c) return false;
        } while (c <= ' ');

        bool neg = false;
        if (c == '-') {
            neg = true;
            c = readChar();
        }
        T val = 0;
        while (c > ' ') {
            val = val * 10 + (c - '0');
            c = readChar();
        }
        out = neg ? -val : val;
        return true;
    }

    bool readDouble(double& out) {
        char c;
        do {
            c = readChar();
            if (!c) return false;
        } while (c <= ' ');

        char s[128];
        int n = 0;
        while (c > ' ') {
            if (n < (int)sizeof(s) - 1) s[n++] = c;
            c = readChar();
        }
        s[n] = 0;
        out = std::strtod(s, nullptr);
        return true;
    }
};

struct FastWriter {
    static constexpr size_t BUFSZ = 1 << 20;
    std::vector<char> buf;

    FastWriter() { buf.reserve(BUFSZ); }

    inline void flush() {
        if (!buf.empty()) {
            std::fwrite(buf.data(), 1, buf.size(), stdout);
            buf.clear();
        }
    }

    ~FastWriter() { flush(); }

    inline void putChar(char c) {
        buf.push_back(c);
        if (buf.size() >= BUFSZ) flush();
    }

    inline void putStr(const char* s) {
        while (*s) putChar(*s++);
    }

    inline void putInt(int v) {
        char tmp[64];
        auto [p, ec] = std::to_chars(tmp, tmp + 64, v);
        (void)ec;
        for (char* q = tmp; q < p; ++q) putChar(*q);
    }

    inline void putUInt(uint32_t v) {
        char tmp[64];
        auto [p, ec] = std::to_chars(tmp, tmp + 64, v);
        (void)ec;
        for (char* q = tmp; q < p; ++q) putChar(*q);
    }

    inline void putInt64(int64_t v) {
        char tmp[64];
        auto [p, ec] = std::to_chars(tmp, tmp + 64, v);
        (void)ec;
        for (char* q = tmp; q < p; ++q) putChar(*q);
    }

    inline void putDouble(double x) {
        // general with enough precision for robust round-trip
        char tmp[128];
        auto [p, ec] = std::to_chars(tmp, tmp + 128, x, std::chars_format::general, 17);
        if (ec != std::errc{}) {
            // fallback
            int n = std::snprintf(tmp, sizeof(tmp), "%.17g", x);
            p = tmp + n;
        }
        for (char* q = tmp; q < p; ++q) putChar(*q);
    }
};

struct AABB {
    Vec3 mn, mx;
    AABB() {
        mn = { std::numeric_limits<double>::infinity(),
               std::numeric_limits<double>::infinity(),
               std::numeric_limits<double>::infinity() };
        mx = { -std::numeric_limits<double>::infinity(),
               -std::numeric_limits<double>::infinity(),
               -std::numeric_limits<double>::infinity() };
    }
    void expand(const Vec3& p) {
        mn.x = std::min(mn.x,p.x); mn.y = std::min(mn.y,p.y); mn.z = std::min(mn.z,p.z);
        mx.x = std::max(mx.x,p.x); mx.y = std::max(mx.y,p.y); mx.z = std::max(mx.z,p.z);
    }
    void expand(const AABB& b) { expand(b.mn); expand(b.mx); }
};

static inline bool aabbOverlap(const AABB& a, const AABB& b) {
    if (a.mx.x < b.mn.x || a.mn.x > b.mx.x) return false;
    if (a.mx.y < b.mn.y || a.mn.y > b.mx.y) return false;
    if (a.mx.z < b.mn.z || a.mn.z > b.mx.z) return false;
    return true;
}

static inline bool rayAabbIntersect(const Vec3& ro, const Vec3& rd, const AABB& b, double& tminOut, double& tmaxOut) {
    double tmin = -std::numeric_limits<double>::infinity();
    double tmax =  std::numeric_limits<double>::infinity();
    auto slab = [&](double roC, double rdC, double mnC, double mxC) {
        if (std::abs(rdC) < 1e-30) {
            if (roC < mnC || roC > mxC) return false;
            return true;
        }
        double inv = 1.0 / rdC;
        double t1 = (mnC - roC) * inv;
        double t2 = (mxC - roC) * inv;
        if (t1 > t2) std::swap(t1,t2);
        tmin = std::max(tmin, t1);
        tmax = std::min(tmax, t2);
        return tmin <= tmax;
    };
    if (!slab(ro.x, rd.x, b.mn.x, b.mx.x)) return false;
    if (!slab(ro.y, rd.y, b.mn.y, b.mx.y)) return false;
    if (!slab(ro.z, rd.z, b.mn.z, b.mx.z)) return false;
    tminOut = tmin; tmaxOut = tmax;
    return true;
}

static inline AABB triAabb(const Vec3& p0, const Vec3& p1, const Vec3& p2) {

struct BVH {
    const std::vector<Vec3>* V = nullptr;
    const std::vector<Tri>*  F = nullptr;
    std::vector<uint32_t>    indices;
    std::vector<BVHNode>     nodes;
    uint32_t leafSize = 8;

    AABB triBoxByIndex(uint32_t idx) const {
        const Tri& t = (*F)[idx];
        return triAabb((*V)[t.a], (*V)[t.b], (*V)[t.c]);
    }
    Vec3 triCentroidByIndex(uint32_t idx) const {
        const Tri& t = (*F)[idx];
        Vec3 c = ((*V)[t.a] + (*V)[t.b] + (*V)[t.c]) / 3.0;
        return c;
    }

    uint32_t buildNode(uint32_t start, uint32_t count) {
        uint32_t nodeId = (uint32_t)nodes.size();
        nodes.push_back(BVHNode{});
        BVHNode& node = nodes.back();
        node.start = start;
        node.count = 0;

        AABB bb;
        AABB cb;
        for (uint32_t i = 0; i < count; ++i) {
            uint32_t triIdx = indices[start + i];
            AABB tb = triBoxByIndex(triIdx);
            bb.expand(tb);
            Vec3 c = triCentroidByIndex(triIdx);
            cb.expand(c);
        }
        node.box = bb;

        if (count <= leafSize) {
            node.count = count;
            return nodeId;
        }

        Vec3 extent = {cb.mx.x - cb.mn.x, cb.mx.y - cb.mn.y, cb.mx.z - cb.mn.z};
        int axis = 0;
        if (extent.y > extent.x) axis = 1;
        if ((axis == 0 ? extent.z : extent.z) > (axis == 0 ? extent.x : extent.y)) {
            if (extent.z > (axis==0?extent.x:extent.y)) axis = 2;
        }

        uint32_t mid = start + count / 2;
        auto nthCmp = [&](uint32_t i, uint32_t j) {
            Vec3 ci = triCentroidByIndex(i);
            Vec3 cj = triCentroidByIndex(j);
            if (axis == 0) return ci.x < cj.x;
            if (axis == 1) return ci.y < cj.y;
            return ci.z < cj.z;
        };
        std::nth_element(indices.begin() + start, indices.begin() + mid, indices.begin() + start + count,
            [&](uint32_t ia, uint32_t ib){ return nthCmp(ia, ib); });

        uint32_t leftCount = mid - start;
        uint32_t rightCount = count - leftCount;
        node.left  = buildNode(start, leftCount);
        node.right = buildNode(mid, rightCount);
        return nodeId;
    }

    void build(const std::vector<Vec3>& verts, const std::vector<Tri>& tris, uint32_t leaf=8) {
        V = &verts; F = &tris; leafSize = leaf;
        indices.resize(tris.size());
        std::iota(indices.begin(), indices.end(), 0u);
        nodes.clear();
        nodes.reserve(tris.size() * 2);
        buildNode(0, (uint32_t)indices.size());
    }
};

static inline bool rayTriIntersectMT(const Vec3& ro, const Vec3& rd,
                                    const Vec3& p0, const Vec3& p1, const Vec3& p2,
                                    double& t, double& u, double& v) {
    // Moller-Trumbore
    Vec3 e1 = p1 - p0;
    Vec3 e2 = p2 - p0;
    Vec3 pvec = cross(rd, e2);
    double det = dot(e1, pvec);
    if (std::abs(det) < 1e-18) return false;
    double invDet = 1.0 / det;

    Vec3 tvec = ro - p0;
    u = dot(tvec, pvec) * invDet;
    if (u < 0.0 || u > 1.0) return false;

    Vec3 qvec = cross(tvec, e1);
    v = dot(rd, qvec) * invDet;
    if (v < 0.0 || u + v > 1.0) return false;

    t = dot(e2, qvec) * invDet;
    return t >= 0.0;
}

static inline bool segTriIntersect(const Vec3& a, const Vec3& b,
                                   const Vec3& p0, const Vec3& p1, const Vec3& p2,
                                   Vec3& hit) {
    Vec3 rd = b - a;
    double len = norm(rd);
    if (len < 1e-30) return false;
    rd = rd / len;
    double t,u,v;
    if (!rayTriIntersectMT(a, rd, p0,p1,p2, t,u,v)) return false;
    if (t < -1e-12 || t > len + 1e-12) return false;
    hit = a + rd * t;
    return true;
}

static inline bool pointInTriBary(const Vec3& p, const Vec3& a, const Vec3& b, const Vec3& c, double eps) {
    // assumes p on plane of triangle approximately
    Vec3 v0 = b - a, v1 = c - a, v2 = p - a;
    double d00 = dot(v0,v0);
    double d01 = dot(v0,v1);
    double d11 = dot(v1,v1);
    double d20 = dot(v2,v0);
    double d21 = dot(v2,v1);
    double denom = d00*d11 - d01*d01;
    if (std::abs(denom) < 1e-30) return false;
    double inv = 1.0 / denom;
    double v = (d11*d20 - d01*d21)*inv;
    double w = (d00*d21 - d01*d20)*inv;
    double u = 1.0 - v - w;
    return u >= -eps && v >= -eps && w >= -eps;
}

static inline double distPointSegment2(const Vec3& p, const Vec3& a, const Vec3& b) {
    Vec3 ab = b - a;
    double t = dot(p - a, ab) / std::max(1e-300, dot(ab,ab));
    t = std::min(1.0, std::max(0.0, t));
    Vec3 q = a + ab * t;
    return norm2(p - q);
}

static inline int chooseDropAxis(const Vec3& p0, const Vec3& p1, const Vec3& p2) {
    Vec3 n = cross(p1 - p0, p2 - p0);
    n.x = std::abs(n.x); n.y = std::abs(n.y); n.z = std::abs(n.z);
    // Drop the largest component axis (best projection area)
    if (n.x >= n.y && n.x >= n.z) return 0;
    if (n.y >= n.x && n.y >= n.z) return 1;
    return 2;
}

static inline double polyArea2(const std::vector<int>& poly, const std::vector<Vec2>& pts) {
    double a = 0;
    int n = (int)poly.size();
    for (int i = 0; i < n; ++i) {
        const Vec2& p = pts[poly[i]];
        const Vec2& q = pts[poly[(i+1)%n]];
        a += p.x*q.y - p.y*q.x;
    }
    return a * 0.5;
}

static inline bool pointInTri2D(const Vec2& p, const Vec2& a, const Vec2& b, const Vec2& c, double eps) {
    // barycentric sign method
    Vec2 ab = b - a, bc = c - b, ca = a - c;
    Vec2 ap = p - a, bp = p - b, cp = p - c;
    double c1 = cross2(ab, ap);
    double c2 = cross2(bc, bp);
    double c3 = cross2(ca, cp);
    bool hasNeg = (c1 < -eps) || (c2 < -eps) || (c3 < -eps);
    bool hasPos = (c1 > eps) || (c2 > eps) || (c3 > eps);
    return !(hasNeg && hasPos);
}

static void removeCollinear(std::vector<int>& poly, const std::vector<Vec2>& pts, double eps) {
    bool changed = true;
    while (changed && poly.size() > 3) {
        changed = false;
        int n = (int)poly.size();
        for (int i = 0; i < n && poly.size() > 3; ++i) {
            int ip = poly[(i - 1 + n) % n];
            int ic = poly[i];
            int in = poly[(i + 1) % n];
            Vec2 a = pts[ip], b = pts[ic], c = pts[in];
            Vec2 ab{b.x - a.x, b.y - a.y};
            Vec2 bc{c.x - b.x, c.y - b.y};
            double cr = cross2(ab, bc);
            if (std::abs(cr) <= eps) {
                poly.erase(poly.begin() + i);
                changed = true;
                break;
            }
        }
    }
}

// Ear clipping triangulation for simple CCW polygon
static void triangulateEarClipping(const std::vector<int>& polyIn,
                                   const std::vector<Vec2>& pts,
                                   std::vector<std::array<int,3>>& outTris,
                                   double eps) {
    std::vector<int> poly = polyIn;
    removeCollinear(poly, pts, eps);

    if (poly.size() < 3) return;
    if (poly.size() == 3) {
        outTris.push_back({poly[0], poly[1], poly[2]});
        return;
    }

    // Ensure CCW
    if (polyArea2(poly, pts) < 0) {
        std::reverse(poly.begin(), poly.end());
    }

    int guard = 0;
    while (poly.size() > 3 && guard++ < 10000) {
        bool cut = false;
        int n = (int)poly.size();
        for (int i = 0; i < n; ++i) {
            int ip = poly[(i - 1 + n) % n];
            int ic = poly[i];
            int in = poly[(i + 1) % n];
            Vec2 a = pts[ip], b = pts[ic], c = pts[in];
            Vec2 ab = b - a, bc = c - b;
            double cr = cross2(ab, bc);
            if (cr <= eps) continue; // reflex or flat

            // check if any other point inside ear
            bool anyInside = false;
            for (int j = 0; j < n; ++j) {
                int pj = poly[j];
                if (pj == ip || pj == ic || pj == in) continue;
                if (pointInTri2D(pts[pj], a, b, c, eps)) { anyInside = true; break; }
            }
            if (anyInside) continue;

            outTris.push_back({ip, ic, in});
            poly.erase(poly.begin() + i);
            cut = true;
            break;
        }
        if (!cut) {
            // Fallback fan triangulation (should not happen for clean PSLG faces)
            int v0 = poly[0];
            for (size_t i = 1; i + 1 < poly.size(); ++i)
                outTris.push_back({v0, poly[i], poly[i+1]});
            return;
        }
    }
    if (poly.size() == 3) outTris.push_back({poly[0], poly[1], poly[2]});
}

struct HalfEdge {
    int from, to;
    int twin;
    int next;
    bool used;
    double ang;
};

static inline double angle2pi(double a) {
    const double PI = 3.1415926535897932384626433832795;
    while (a < 0) a += 2*PI;
    while (a >= 2*PI) a -= 2*PI;
    return a;
}

// Build faces from planar embedded graph in 2D, returns faces as cycles of local vertex ids (CCW interior)
static void planarFacesFromEdges(
    int nVerts,
    const std::vector<Vec2>& pts,
    const std::vector<std::pair<int,int>>& undirectedEdges,
    std::vector<std::vector<int>>& outFaces,
    double epsArea)
{
    // Build halfedges
    std::vector<HalfEdge> H;
    H.reserve(undirectedEdges.size() * 2);
    std::vector<std::vector<int>> outAt(nVerts);

    auto addHalf = [&](int u, int v) -> int {
        HalfEdge he;
        he.from = u; he.to = v;
        he.twin = -1; he.next = -1; he.used = false;
        Vec2 d{pts[v].x - pts[u].x, pts[v].y - pts[u].y};
        he.ang = angle2pi(std::atan2(d.y, d.x));
        int id = (int)H.size();
        H.push_back(he);
        outAt[u].push_back(id);
        return id;
    };

    for (auto [u,v] : undirectedEdges) {
        int e0 = addHalf(u,v);
        int e1 = addHalf(v,u);
        H[e0].twin = e1;
        H[e1].twin = e0;
    }

    // Sort outgoing at each vertex by angle CCW
    for (int v = 0; v < nVerts; ++v) {
        auto& lst = outAt[v];
        std::sort(lst.begin(), lst.end(), [&](int a, int b){
            return H[a].ang < H[b].ang;
        });
    }

    // For each halfedge, set next = prev(twin) in CCW order around destination
    // Find position of twin in outAt[to], then take previous (clockwise neighbor).
    std::vector<int> posInOut(H.size(), -1);
    for (int v = 0; v < nVerts; ++v) {
        auto& lst = outAt[v];
        for (int i = 0; i < (int)lst.size(); ++i) posInOut[lst[i]] = i;
    }

    for (int eid = 0; eid < (int)H.size(); ++eid) {
        int v = H[eid].to;
        int twin = H[eid].twin; // from v to from
        auto& lst = outAt[v];
        int pos = posInOut[twin];
        int prevPos = (pos - 1 + (int)lst.size()) % (int)lst.size();
        H[eid].next = lst[prevPos];
    }

    // Traverse faces
    for (int eid = 0; eid < (int)H.size(); ++eid) {
        if (H[eid].used) continue;

        std::vector<int> cycle;
        int cur = eid;
        int guard = 0;
        while (!H[cur].used && guard++ < 100000) {
            H[cur].used = true;
            cycle.push_back(H[cur].from);
            cur = H[cur].next;
            if (cur == eid) break;
        }
        if (cycle.size() < 3) continue;

        // compute signed area
        double a = 0;
        for (size_t i = 0; i < cycle.size(); ++i) {
            const Vec2& p = pts[cycle[i]];
            const Vec2& q = pts[cycle[(i+1)%cycle.size()]];
            a += p.x*q.y - p.y*q.x;
        }
        a *= 0.5;

        if (a > epsArea) {
            outFaces.push_back(std::move(cycle));
        }
    }
}

static inline uint64_t hashPoint(const Vec3& p) {
    auto hmix = [](uint64_t x) {
        x ^= x >> 33;
        x *= 0xff51afd7ed558ccdULL;
        x ^= x >> 33;
        x *= 0xc4ceb9fe1a85ec53ULL;
        x ^= x >> 33;
        return x;
    };
    uint64_t a = (uint64_t)std::llround(p.x * 1000003.0);
    uint64_t b = (uint64_t)std::llround(p.y * 1000033.0);
    uint64_t c = (uint64_t)std::llround(p.z * 1000211.0);
    return hmix(a ^ (hmix(b) << 1) ^ (hmix(c) << 2));
}

static inline Vec3 makeRayDir(const Vec3& p) {
    // base dir + tiny deterministic perturb to reduce degeneracies
    Vec3 d{1.0, 0.23456789, 0.34567891};
    uint64_t h = hashPoint(p);
    auto u01 = [&](uint64_t& s)->double{
        s ^= s >> 12; s ^= s << 25; s ^= s >> 27;
        uint64_t r = s * 2685821657736338717ULL;
        return (r >> 11) * (1.0 / 9007199254740992.0); // /2^53
    };
    double jx = (u01(h) - 0.5) * 1e-7;
    double jy = (u01(h) - 0.5) * 1e-7;
    double jz = (u01(h) - 0.5) * 1e-7;
    d.x += jx; d.y += jy; d.z += jz;
    double l = norm(d);
    return d / std::max(1e-300, l);
}

static int rayCountIntersections(const BVH& bvh, const Vec3& ro, const Vec3& rd,
                                 const std::vector<Vec3>& V, const std::vector<Tri>& F) {
    int cnt = 0;
    std::vector<uint32_t> stack;
    stack.reserve(64);
    stack.push_back(0);

    const double baryEps = 1e-12;

    while (!stack.empty()) {
        uint32_t nid = stack.back(); stack.pop_back();
        const BVHNode& n = bvh.nodes[nid];
        double tmin, tmax;
        if (!rayAabbIntersect(ro, rd, n.box, tmin, tmax)) continue;
        if (tmax < 0) continue;

        if (n.count) {
            for (uint32_t i = 0; i < n.count; ++i) {
                uint32_t triIdx = bvh.indices[n.start + i];
                const Tri& t = F[triIdx];
                const Vec3& p0 = V[t.a];
                const Vec3& p1 = V[t.b];
                const Vec3& p2 = V[t.c];
                double tt,u,v;
                if (rayTriIntersectMT(ro, rd, p0,p1,p2, tt,u,v)) {
                    if (tt <= 1e-12) continue;
                    // avoid counting hits exactly on edges/vertices
                    if (u <= baryEps || v <= baryEps || (u+v) >= 1.0 - baryEps) continue;
                    cnt++;
                }
            }
        } else {
            stack.push_back(n.left);
            stack.push_back(n.right);
        }
    }
    return cnt;
}

static bool pointInsideMesh(const BVH& bvh, const Vec3& p,
                            const std::vector<Vec3>& V, const std::vector<Tri>& F) {
    Vec3 rd = makeRayDir(p);
    Vec3 ro = p + rd * 1e-10; // tiny offset
    int cnt = rayCountIntersections(bvh, ro, rd, V, F);
    return (cnt & 1) != 0;
}

static bool triTriIntersectionSegment_approx(
    const Vec3& a0, const Vec3& a1, const Vec3& a2,
    const Vec3& b0, const Vec3& b1, const Vec3& b2,
    Vec3& outP, Vec3& outQ)
{
    // Collect intersection points from segment-triangle tests + near-coplanar vertex-in-tri.
    std::vector<Vec3> pts;
    pts.reserve(12);

    auto addUnique = [&](const Vec3& p) {
        for (auto& q : pts) {
            if (norm2(p - q) <= 1e-20) return;
        }
        pts.push_back(p);
    };

    Vec3 hit;
    // edges of A against tri B
    if (segTriIntersect(a0,a1, b0,b1,b2, hit)) addUnique(hit);
    if (segTriIntersect(a1,a2, b0,b1,b2, hit)) addUnique(hit);
    if (segTriIntersect(a2,a0, b0,b1,b2, hit)) addUnique(hit);
    // edges of B against tri A
    if (segTriIntersect(b0,b1, a0,a1,a2, hit)) addUnique(hit);
    if (segTriIntersect(b1,b2, a0,a1,a2, hit)) addUnique(hit);
    if (segTriIntersect(b2,b0, a0,a1,a2, hit)) addUnique(hit);

    // handle vertex-on-triangle plane (rare), with tolerance
    auto planeDist = [](const Vec3& p, const Vec3& p0, const Vec3& n)->double{
        return dot(p - p0, n);
    };
    Vec3 nb = cross(b1 - b0, b2 - b0);
    double nbLen = norm(nb);
    if (nbLen > 1e-30) {
        Vec3 nbN = nb / nbLen;
        double d0 = planeDist(a0, b0, nbN);
        double d1 = planeDist(a1, b0, nbN);
        double d2 = planeDist(a2, b0, nbN);
        if (std::abs(d0) < 1e-10 && pointInTriBary(a0,b0,b1,b2,1e-10)) addUnique(a0);
        if (std::abs(d1) < 1e-10 && pointInTriBary(a1,b0,b1,b2,1e-10)) addUnique(a1);
        if (std::abs(d2) < 1e-10 && pointInTriBary(a2,b0,b1,b2,1e-10)) addUnique(a2);
    }
    Vec3 na = cross(a1 - a0, a2 - a0);
    double naLen = norm(na);
    if (naLen > 1e-30) {
        Vec3 naN = na / naLen;
        double d0 = dot(b0 - a0, naN);
        double d1 = dot(b1 - a0, naN);
        double d2 = dot(b2 - a0, naN);
        if (std::abs(d0) < 1e-10 && pointInTriBary(b0,a0,a1,a2,1e-10)) addUnique(b0);
        if (std::abs(d1) < 1e-10 && pointInTriBary(b1,a0,a1,a2,1e-10)) addUnique(b1);
        if (std::abs(d2) < 1e-10 && pointInTriBary(b2,a0,a1,a2,1e-10)) addUnique(b2);
    }

    if (pts.size() < 2) return false;

    // choose farthest pair as segment
    double best = -1;
    int bi=0,bj=1;
    for (int i = 0; i < (int)pts.size(); ++i) {
        for (int j = i+1; j < (int)pts.size(); ++j) {
            double d = norm2(pts[i] - pts[j]);
            if (d > best) { best = d; bi=i; bj=j; }
        }
    }
    if (best < 1e-24) return false; // almost a point touch
    outP = pts[bi];
    outQ = pts[bj];
    return true;
}

static void bvhIntersectCollectCuts(
    const BVH& A, const std::vector<Vec3>& VA, const std::vector<Tri>& FA,
    const BVH& B, const std::vector<Vec3>& VB, const std::vector<Tri>& FB,
    std::vector<CutTmp>& cutsA,
    std::vector<CutTmp>& cutsB)
{
    if (A.nodes.empty() || B.nodes.empty()) return;
    if (!aabbOverlap(A.nodes[0].box, B.nodes[0].box)) return;

    std::vector<std::pair<uint32_t,uint32_t>> st;
    st.reserve(256);
    st.push_back({0,0});

    Vec3 p,q;

    while (!st.empty()) {
        auto [na, nb] = st.back(); st.pop_back();
        const BVHNode& a = A.nodes[na];
        const BVHNode& b = B.nodes[nb];
        if (!aabbOverlap(a.box, b.box)) continue;

        bool aLeaf = a.count != 0;
        bool bLeaf = b.count != 0;

        if (aLeaf && bLeaf) {
            for (uint32_t ia = 0; ia < a.count; ++ia) {
                uint32_t triAIdx = A.indices[a.start + ia];
                const Tri& ta = FA[triAIdx];
                const Vec3& a0 = VA[ta.a];
                const Vec3& a1 = VA[ta.b];
                const Vec3& a2 = VA[ta.c];

                AABB boxA = triAabb(a0,a1,a2);

                for (uint32_t ib = 0; ib < b.count; ++ib) {
                    uint32_t triBIdx = B.indices[b.start + ib];
                    const Tri& tb = FB[triBIdx];
                    const Vec3& b0 = VB[tb.a];
                    const Vec3& b1 = VB[tb.b];
                    const Vec3& b2 = VB[tb.c];

                    AABB boxB = triAabb(b0,b1,b2);
                    if (!aabbOverlap(boxA, boxB)) continue;

                    if (triTriIntersectionSegment_approx(a0,a1,a2, b0,b1,b2, p,q)) {
                        cutsA.push_back(CutTmp{triAIdx, p,q});
                        cutsB.push_back(CutTmp{triBIdx, p,q});
                    }
                }
            }
        } else {
            // descend the larger node
            if (aLeaf && !bLeaf) {
                st.push_back({na, b.left});
                st.push_back({na, b.right});
            } else if (!aLeaf && bLeaf) {
                st.push_back({a.left, nb});
                st.push_back({a.right, nb});
            } else {
                // both internal: descend both (heuristic)
                double va = (a.box.mx.x-a.box.mn.x)*(a.box.mx.y-a.box.mn.y)*(a.box.mx.z-a.box.mn.z);
                double vb = (b.box.mx.x-b.box.mn.x)*(b.box.mx.y-b.box.mn.y)*(b.box.mx.z-b.box.mn.z);
                if (va > vb) {
                    st.push_back({a.left, nb});
                    st.push_back({a.right, nb});
                } else {
                    st.push_back({na, b.left});
                    st.push_back({na, b.right});
                }
            }
        }
    }
}

static void buildCSR(uint32_t triCount, const std::vector<Cut>& cuts, std::vector<uint32_t>& offsets, std::vector<Cut>& sorted) {
    offsets.assign(triCount + 1, 0);
    for (auto& c : cuts) offsets[c.tri + 1]++;
    for (uint32_t i = 1; i <= triCount; ++i) offsets[i] += offsets[i-1];
    sorted.resize(cuts.size());
    std::vector<uint32_t> cur = offsets;
    for (auto& c : cuts) {
        uint32_t pos = cur[c.tri]++;
        sorted[pos] = c;
    }
}

static void processMeshPiecesUnion(
    const std::vector<Vec3>& allVerts,
    const std::vector<Tri>& trisThis,
    uint32_t triOffsetIdx, // 0 for A, VA for B (already applied in trisThis indices if desired; here trisThis refers to allVerts indices)
    const std::vector<uint32_t>& cutOffsets,
    const std::vector<Cut>& cutSorted,
    const BVH& otherBVH,
    const std::vector<Vec3>& otherV,
    const std::vector<Tri>& otherF,
    uint32_t beginTri, uint32_t endTri,
    std::vector<Tri>& outFaces)
{
    (void)triOffsetIdx; // indices already global in trisThis
    const double epsEdge = 1e-20; // for distance^2 checks (uses quantization already)
    const double eps2D = 1e-14;
    const double epsArea = 1e-18;

    std::vector<uint32_t> localGlobal; localGlobal.reserve(32);
    std::unordered_map<uint32_t,int> g2l; g2l.reserve(64);

    for (uint32_t ti = beginTri; ti < endTri; ++ti) {
        const Tri& T = trisThis[ti];
        uint32_t v0 = T.a, v1 = T.b, v2 = T.c;
        const Vec3& p0 = allVerts[v0];
        const Vec3& p1 = allVerts[v1];
        const Vec3& p2 = allVerts[v2];

        // gather cut segments
        uint32_t s = cutOffsets[ti];
        uint32_t e = cutOffsets[ti+1];

        if (s == e) {
            Vec3 cen = (p0+p1+p2)/3.0;
            if (!pointInsideMesh(otherBVH, cen, otherV, otherF)) {
                outFaces.push_back(T);
            }
            continue;
        }

        // collect boundary points per edge (by global indices)
        std::vector<std::pair<double,uint32_t>> edgePts[3];
        edgePts[0].reserve(8); edgePts[1].reserve(8); edgePts[2].reserve(8);

        auto addEdgePt = [&](int edge, uint32_t g, double t) {
            edgePts[edge].push_back({t, g});
        };

        // Always include triangle corners
        addEdgePt(0, v0, 0.0); addEdgePt(0, v1, 1.0);
        addEdgePt(1, v1, 0.0); addEdgePt(1, v2, 1.0);
        addEdgePt(2, v2, 0.0); addEdgePt(2, v0, 1.0);

        auto edgeParam = [&](const Vec3& a, const Vec3& b, const Vec3& p)->double{
            Vec3 ab = b - a;
            return dot(p - a, ab) / std::max(1e-300, dot(ab,ab));
        };

        // internal edges list in global indices
        std::vector<std::pair<uint32_t,uint32_t>> internalSegs;
        internalSegs.reserve(e - s);

        // assign endpoints to boundary edges
        for (uint32_t k = s; k < e; ++k) {
            uint32_t gu = cutSorted[k].u;
            uint32_t gv = cutSorted[k].v;
            const Vec3& pu = allVerts[gu];
            const Vec3& pv = allVerts[gv];

            // Determine on which boundary edge each endpoint lies (should be exactly on one)
            auto findEdge = [&](const Vec3& pp)->int{
                double d01 = distPointSegment2(pp, p0, p1);
                double d12 = distPointSegment2(pp, p1, p2);
                double d20 = distPointSegment2(pp, p2, p0);
                double md = std::min({d01,d12,d20});
                if (md > epsEdge) {
                    // numerical issues: pick min anyway
                }
                if (md == d01) return 0;
                if (md == d12) return 1;
                return 2;
            };
            int eu = findEdge(pu);
            int ev = findEdge(pv);

            // compute t along that edge
            if (eu == 0) addEdgePt(0, gu, edgeParam(p0,p1,pu));
            else if (eu == 1) addEdgePt(1, gu, edgeParam(p1,p2,pu));
            else addEdgePt(2, gu, edgeParam(p2,p0,pu));

            if (ev == 0) addEdgePt(0, gv, edgeParam(p0,p1,pv));
            else if (ev == 1) addEdgePt(1, gv, edgeParam(p1,p2,pv));
            else addEdgePt(2, gv, edgeParam(p2,p0,pv));

            // add as internal only if endpoints on different boundary edges (true cut chord)
            if (gu != gv) {
                if (eu != ev) internalSegs.push_back({gu, gv});
            }
        }

        // Unique+sort edge points
        for (int ed = 0; ed < 3; ++ed) {
            auto& v = edgePts[ed];
            std::sort(v.begin(), v.end(), [](auto& A, auto& B){
                if (A.first != B.first) return A.first < B.first;
                return A.second < B.second;
            });
            // clamp and unique by global index (and near-identical t)
            std::vector<std::pair<double,uint32_t>> u;
            u.reserve(v.size());
            for (auto& it : v) {
                double t = std::min(1.0, std::max(0.0, it.first));
                uint32_t g = it.second;
                if (!u.empty() && u.back().second == g) continue;
                u.push_back({t,g});
            }
            v.swap(u);
        }

        // Build local vertices: collect all unique globals from boundary and internal segments
        localGlobal.clear();
        g2l.clear();

        auto addLocal = [&](uint32_t g)->int{
            auto it = g2l.find(g);
            if (it != g2l.end()) return it->second;
            int id = (int)localGlobal.size();
            localGlobal.push_back(g);
            g2l.emplace(g,id);
            return id;
        };

        for (int ed=0; ed<3; ++ed) for (auto& pr : edgePts[ed]) addLocal(pr.second);
        for (auto& seg : internalSegs) { addLocal(seg.first); addLocal(seg.second); }

        // local 2D coords
        int drop = chooseDropAxis(p0,p1,p2);
        std::vector<Vec2> pts2D(localGlobal.size());
        for (size_t i = 0; i < localGlobal.size(); ++i) {
            pts2D[i] = project2D(allVerts[localGlobal[i]], drop);
        }

        // collect undirected edges in local indices
        struct EKey { int a,b; };
        struct EHash { size_t operator()(const EKey& k) const noexcept {
            return (size_t)((uint32_t)k.a * 2654435761u) ^ (size_t)((uint32_t)k.b * 40503u);
        }};
        struct EEq { bool operator()(const EKey& x, const EKey& y) const noexcept { return x.a==y.a && x.b==y.b; } };

        std::unordered_map<EKey, char, EHash, EEq> edgeSet;
        edgeSet.reserve(128);

        auto addUndirEdge = [&](int a, int b) {
            if (a == b) return;
            if (a > b) std::swap(a,b);
            EKey k{a,b};
            edgeSet.emplace(k, 1);
        };

        // boundary subdiv edges
        auto connectSeq = [&](const std::vector<std::pair<double,uint32_t>>& seq) {
            for (size_t i = 0; i + 1 < seq.size(); ++i) {
                int la = addLocal(seq[i].second);
                int lb = addLocal(seq[i+1].second);
                addUndirEdge(la, lb);
            }
        };
        connectSeq(edgePts[0]);
        connectSeq(edgePts[1]);
        connectSeq(edgePts[2]);

        // internal segments
        for (auto& seg : internalSegs) {
            int la = addLocal(seg.first);
            int lb = addLocal(seg.second);
            addUndirEdge(la, lb);
        }

        std::vector<std::pair<int,int>> edges;
        edges.reserve(edgeSet.size());
        for (auto& kv : edgeSet) edges.push_back({kv.first.a, kv.first.b});

        // Extract faces
        std::vector<std::vector<int>> faces;
        faces.reserve(8);
        planarFacesFromEdges((int)localGlobal.size(), pts2D, edges, faces, epsArea);

        // Triangulate faces and classify
        for (auto& poly : faces) {
            // Triangulate polygon in local ids
            std::vector<std::array<int,3>> trisLocal;
            trisLocal.reserve(poly.size() - 2);
            triangulateEarClipping(poly, pts2D, trisLocal, eps2D);

            for (auto& tl : trisLocal) {
                uint32_t gA = localGlobal[tl[0]];
                uint32_t gB = localGlobal[tl[1]];
                uint32_t gC = localGlobal[tl[2]];
                const Vec3& A3 = allVerts[gA];
                const Vec3& B3 = allVerts[gB];
                const Vec3& C3 = allVerts[gC];
                Vec3 cen = (A3 + B3 + C3) / 3.0;
                if (!pointInsideMesh(otherBVH, cen, otherV, otherF)) {
                    outFaces.push_back(Tri{gA,gB,gC});
                }
            }
        }
    }
}

int main() {
    FastScanner fs;
    uint32_t VA, FA, VB, FB;
    if (!fs.readInt(VA)) return 0;
    fs.readInt(FA);
    fs.readInt(VB);
    fs.readInt(FB);

    std::vector<Vec3> vertsA(VA), vertsB(VB);
    std::vector<Tri> facesA(FA), facesB(FB);

    for (uint32_t i=0;i<VA;++i) {
        fs.readDouble(vertsA[i].x);
        fs.readDouble(vertsA[i].y);
        fs.readDouble(vertsA[i].z);
    }
    for (uint32_t i=0;i<FA;++i) {
        uint32_t a,b,c;
        fs.readInt(a); fs.readInt(b); fs.readInt(c);
        facesA[i] = Tri{a,b,c};
    }
    for (uint32_t i=0;i<VB;++i) {
        fs.readDouble(vertsB[i].x);
        fs.readDouble(vertsB[i].y);
        fs.readDouble(vertsB[i].z);
    }
    for (uint32_t i=0;i<FB;++i) {
        uint32_t a,b,c;
        fs.readInt(a); fs.readInt(b); fs.readInt(c);
        facesB[i] = Tri{a,b,c};
    }

    // Build BVHs on original meshes
    BVH bvhA, bvhB;
    bvhA.build(vertsA, facesA, 8);
    bvhB.build(vertsB, facesB, 8);

    // Collect intersection cut segments (temporary, with double endpoints)
    std::vector<CutTmp> cutsATmp, cutsBTmp;
    cutsATmp.reserve(1024);
    cutsBTmp.reserve(1024);

    bvhIntersectCollectCuts(bvhA, vertsA, facesA, bvhB, vertsB, facesB, cutsATmp, cutsBTmp);

    // Fast "no intersection" handling
    if (cutsATmp.empty() && cutsBTmp.empty()) {
        // Determine containment / disjoint
        bool aInB = false, bInA = false;
        if (!vertsA.empty()) aInB = pointInsideMesh(bvhB, vertsA[0], vertsB, facesB);
        if (!vertsB.empty()) bInA = pointInsideMesh(bvhA, vertsB[0], vertsA, facesA);

        FastWriter fw;

        if (aInB && !bInA) {
            // union is B
            fw.putUInt((uint32_t)vertsB.size()); fw.putChar(' ');
            fw.putUInt((uint32_t)facesB.size()); fw.putChar('\n');
            for (auto& v : vertsB) {
                fw.putDouble(v.x); fw.putChar(' ');
                fw.putDouble(v.y); fw.putChar(' ');
                fw.putDouble(v.z); fw.putChar('\n');
            }
            for (auto& f : facesB) {
                fw.putUInt(f.a); fw.putChar(' ');
                fw.putUInt(f.b); fw.putChar(' ');
                fw.putUInt(f.c); fw.putChar('\n');
            }
            return 0;
        }
        if (bInA && !aInB) {
            // union is A
            fw.putUInt((uint32_t)vertsA.size()); fw.putChar(' ');
            fw.putUInt((uint32_t)facesA.size()); fw.putChar('\n');
            for (auto& v : vertsA) {
                fw.putDouble(v.x); fw.putChar(' ');
                fw.putDouble(v.y); fw.putChar(' ');
                fw.putDouble(v.z); fw.putChar('\n');
            }
            for (auto& f : facesA) {
                fw.putUInt(f.a); fw.putChar(' ');
                fw.putUInt(f.b); fw.putChar(' ');
                fw.putUInt(f.c); fw.putChar('\n');
            }
            return 0;
        }

        // disjoint (or nested weird): concatenate
        uint32_t Vout = VA + VB;
        uint32_t Fout = FA + FB;
        fw.putUInt(Vout); fw.putChar(' ');
        fw.putUInt(Fout); fw.putChar('\n');
        for (auto& v : vertsA) {
            fw.putDouble(v.x); fw.putChar(' ');
            fw.putDouble(v.y); fw.putChar(' ');
            fw.putDouble(v.z); fw.putChar('\n');
        }
        for (auto& v : vertsB) {
            fw.putDouble(v.x); fw.putChar(' ');
            fw.putDouble(v.y); fw.putChar(' ');
            fw.putDouble(v.z); fw.putChar('\n');
        }
        for (auto& f : facesA) {
            fw.putUInt(f.a); fw.putChar(' ');
            fw.putUInt(f.b); fw.putChar(' ');
            fw.putUInt(f.c); fw.putChar('\n');
        }
        for (auto& f : facesB) {
            fw.putUInt(f.a + VA); fw.putChar(' ');
            fw.putUInt(f.b + VA); fw.putChar(' ');
            fw.putUInt(f.c + VA); fw.putChar('\n');
        }
        return 0;
    }

    // Build global vertex list: A verts + B verts + unique quantized intersection endpoints
    AABB bb;
    for (auto& v : vertsA) bb.expand(v);
    for (auto& v : vertsB) bb.expand(v);
    Vec3 diag = bb.mx - bb.mn;
    double diagLen = std::max(1e-12, norm(diag));
    double eps = diagLen * 1e-9 + 1e-12;
    double invEps = 1.0 / eps;

    std::vector<std::pair<QuantKey, Vec3>> ip;
    ip.reserve((cutsATmp.size() + cutsBTmp.size()) * 2);

    auto pushPt = [&](const Vec3& p) {
        ip.push_back({quantize(p, invEps), p});
    };
    for (auto& c : cutsATmp) { pushPt(c.p); pushPt(c.q); }
    for (auto& c : cutsBTmp) { pushPt(c.p); pushPt(c.q); }

    std::sort(ip.begin(), ip.end(), [](auto& A, auto& B){
        if (A.first.x != B.first.x) return A.first.x < B.first.x;
        if (A.first.y != B.first.y) return A.first.y < B.first.y;
        return A.first.z < B.first.z;
    });
    ip.erase(std::unique(ip.begin(), ip.end(), [](auto& A, auto& B){
        return A.first == B.first;
    }), ip.end());

    std::vector<Vec3> allVerts;
    allVerts.reserve((size_t)VA + VB + ip.size());
    allVerts.insert(allVerts.end(), vertsA.begin(), vertsA.end());
    allVerts.insert(allVerts.end(), vertsB.begin(), vertsB.end());

    uint32_t baseNew = (uint32_t)allVerts.size();
    allVerts.resize(allVerts.size() + ip.size());
    for (size_t i = 0; i < ip.size(); ++i) allVerts[baseNew + (uint32_t)i] = ip[i].second;

    std::unordered_map<QuantKey, uint32_t, KeyHash> keyToIdx;
    keyToIdx.reserve(ip.size() * 2 + 1);
    for (size_t i = 0; i < ip.size(); ++i) {
        keyToIdx.emplace(ip[i].first, baseNew + (uint32_t)i);
    }

    // Convert cuts to indexed form (global allVerts indices)
    std::vector<Cut> cutsA, cutsB;
    cutsA.reserve(cutsATmp.size());
    cutsB.reserve(cutsBTmp.size());

    auto mapPt = [&](const Vec3& p)->uint32_t {
        QuantKey k = quantize(p, invEps);
        auto it = keyToIdx.find(k);
        if (it == keyToIdx.end()) {
            // should not happen; fallback: append (rare)
            uint32_t idx = (uint32_t)allVerts.size();
            allVerts.push_back(p);
            keyToIdx.emplace(k, idx);
            return idx;
        }
        return it->second;
    };

    for (auto& c : cutsATmp) {
        uint32_t u = mapPt(c.p);
        uint32_t v = mapPt(c.q);
        if (u != v) cutsA.push_back(Cut{c.tri, u, v});
    }
    for (auto& c : cutsBTmp) {
        uint32_t u = mapPt(c.p);
        uint32_t v = mapPt(c.q);
        if (u != v) cutsB.push_back(Cut{c.tri, u, v});
    }

    // Build global-indexed triangle arrays for both meshes
    std::vector<Tri> facesA_global = facesA;
    std::vector<Tri> facesB_global = facesB;
    for (auto& t : facesB_global) { t.a += VA; t.b += VA; t.c += VA; }

    // CSR for cuts
    std::vector<uint32_t> offA, offB;
    std::vector<Cut> sortedA, sortedB;
    buildCSR(FA, cutsA, offA, sortedA);
    buildCSR(FB, cutsB, offB, sortedB);

    // Parallel processing over triangles (pieces generation + inside/outside filtering)
    unsigned hw = std::max(1u, std::thread::hardware_concurrency());
    unsigned threads = std::min(12u, hw);

    std::vector<std::vector<Tri>> facesOutA(threads), facesOutB(threads);

    auto runMesh = [&](bool isA) {
        const auto& trisThis = isA ? facesA_global : facesB_global;
        uint32_t triCount = isA ? FA : FB;
        const auto& cutOffsets = isA ? offA : offB;
        const auto& cutSorted = isA ? sortedA : sortedB;
        const BVH& otherBVH = isA ? bvhB : bvhA;
        const auto& otherV = isA ? vertsB : vertsA;
        const auto& otherF = isA ? facesB : facesA;
        auto& buckets = isA ? facesOutA : facesOutB;

        std::vector<std::thread> pool;
        pool.reserve(threads);

        for (unsigned t = 0; t < threads; ++t) {
            uint32_t begin = (uint32_t)((uint64_t)triCount * t / threads);
            uint32_t end   = (uint32_t)((uint64_t)triCount * (t+1) / threads);
            pool.emplace_back([&, t, begin, end]{
                buckets[t].reserve((end - begin) * 2);
                processMeshPiecesUnion(
                    allVerts,
                    trisThis,
                    0,
                    cutOffsets,
                    cutSorted,
                    otherBVH,
                    otherV,
                    otherF,
                    begin, end,
                    buckets[t]
                );
            });
        }
        for (auto& th : pool) th.join();
    };

    runMesh(true);
    runMesh(false);

    std::vector<Tri> outFaces;
    size_t totalFaces = 0;
    for (auto& v : facesOutA) totalFaces += v.size();
    for (auto& v : facesOutB) totalFaces += v.size();
    outFaces.reserve(totalFaces);
    for (auto& v : facesOutA) { outFaces.insert(outFaces.end(), v.begin(), v.end()); std::vector<Tri>().swap(v); }
    for (auto& v : facesOutB) { outFaces.insert(outFaces.end(), v.begin(), v.end()); std::vector<Tri>().swap(v); }

    // Output
    FastWriter fw;
    fw.putUInt((uint32_t)allVerts.size()); fw.putChar(' ');
    fw.putUInt((uint32_t)outFaces.size()); fw.putChar('\n');
    for (auto& v : allVerts) {
        fw.putDouble(v.x); fw.putChar(' ');
        fw.putDouble(v.y); fw.putChar(' ');
        fw.putDouble(v.z); fw.putChar('\n');
    }
    for (auto& f : outFaces) {
        fw.putUInt(f.a); fw.putChar(' ');
        fw.putUInt(f.b); fw.putChar(' ');
        fw.putUInt(f.c); fw.putChar('\n');
    }
    return 0;
}
