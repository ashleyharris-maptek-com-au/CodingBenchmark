```javascript
'use strict';

const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input.length ? JSON.parse(input) : {};

const grid = data.grid || [];
const robotsIn = data.robots || [];
const jobsIn = data.jobs || [];

const H = grid.length;
const W = H ? grid[0].length : 0;
const N = W * H;

function idxOf(x, y) { return y * W + x; }

// Build open-cell map
const open = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = grid[y];
  for (let x = 0; x < W; x++) {
    open[idxOf(x, y)] = (row.charCodeAt(x) === 46) ? 1 : 0; // '.' => 1, else 0
  }
}

// Precompute x coordinate per index to avoid % in BFS
const xOf = new Uint16Array(N);
for (let y = 0; y < H; y++) {
  let base = y * W;
  for (let x = 0; x < W; x++) xOf[base + x] = x;
}

// Robots by id
let maxRobotId = -1;
let minRobotId = Infinity;
const robotById = new Map();
for (const r of robotsIn) {
  const id = r.id | 0;
  maxRobotId = Math.max(maxRobotId, id);
  minRobotId = Math.min(minRobotId, id);
  robotById.set(id, { id, x: r.x | 0, y: r.y | 0 });
}
if (!Number.isFinite(minRobotId)) minRobotId = 0;

// Ensure output has indexes up to maxRobotId (if none, still output [])
const assignments = Array.from({ length: Math.max(0, maxRobotId + 1) }, () => []);

// If grid is empty or no robots, assign everything to robot 0 index if possible
if (N === 0 || robotById.size === 0) {
  if (assignments.length === 0) assignments.push([]);
  for (const j of jobsIn) assignments[0].push(j.id);
  process.stdout.write(JSON.stringify({ assignments }));
  process.exit(0);
}

// Treat robot/job cells as walkable (defensive)
for (const r of robotById.values()) {
  if (r.x >= 0 && r.x < W && r.y >= 0 && r.y < H) open[idxOf(r.x, r.y)] = 1;
}
for (const j of jobsIn) {
  const x = j.x | 0, y = j.y | 0;
  if (x >= 0 && x < W && y >= 0 && y < H) open[idxOf(x, y)] = 1;
}

// Multi-source BFS to get nearest robot owner and distance
const distNearest = new Int32Array(N);
distNearest.fill(-1);
const owner = new Int32Array(N);
owner.fill(-1);

const queue = new Int32Array(N);
let qh = 0, qt = 0;

// Initialize sources in ascending robot id for deterministic tie-breaking
const robotIds = Array.from(robotById.keys()).sort((a, b) => a - b);
for (const id of robotIds) {
  const r = robotById.get(id);
  if (r.x < 0 || r.x >= W || r.y < 0 || r.y >= H) continue;
  const s = idxOf(r.x, r.y);
  if (distNearest[s] === -1) {
    distNearest[s] = 0;
    owner[s] = id;
    queue[qt++] = s;
  } else if (distNearest[s] === 0 && id < owner[s]) {
    owner[s] = id;
  }
}

while (qh < qt) {
  const v = queue[qh++];
  const dv = distNearest[v] + 1;
  const xv = xOf[v];

  // left
  if (xv > 0) {
    const nb = v - 1;
    if (open[nb]) {
      if (distNearest[nb] === -1) {
        distNearest[nb] = dv;
        owner[nb] = owner[v];
        queue[qt++] = nb;
      }
    }
  }
  // right
  if (xv + 1 < W) {
    const nb = v + 1;
    if (open[nb]) {
      if (distNearest[nb] === -1) {
        distNearest[nb] = dv;
        owner[nb] = owner[v];
        queue[qt++] = nb;
      }
    }
  }
  // up
  if (v >= W) {
    const nb = v - W;
    if (open[nb]) {
      if (distNearest[nb] === -1) {
        distNearest[nb] = dv;
        owner[nb] = owner[v];
        queue[qt++] = nb;
      }
    }
  }
  // down
  if (v < N - W) {
    const nb = v + W;
    if (open[nb]) {
      if (distNearest[nb] === -1) {
        distNearest[nb] = dv;
        owner[nb] = owner[v];
        queue[qt++] = nb;
      }
    }
  }
}

// Jobs per robot
const jobsPerRobot = Array.from({ length: Math.max(0, maxRobotId + 1) }, () => []);
let maxPrio = 1;
for (const j of jobsIn) {
  const p = (j.priority == null ? 0 : +j.priority);
  if (Number.isFinite(p)) maxPrio = Math.max(maxPrio, p);
}

for (const j of jobsIn) {
  const x = j.x | 0, y = j.y | 0;
  let cell = -1;
  if (x >= 0 && x < W && y >= 0 && y < H) cell = idxOf(x, y);

  let rid = minRobotId;
  let d0 = 1e9;
  if (cell >= 0) {
    const o = owner[cell];
    if (o !== -1 && o <= maxRobotId) {
      rid = o;
      d0 = distNearest[cell];
    }
  }
  const pr = (j.priority == null ? 0 : +j.priority);
  const prioNorm = (Number.isFinite(pr) ? pr : 0) / maxPrio;

  // If rid index doesn't exist (sparse ids), clamp to minRobotId when possible
  if (rid < 0 || rid > maxRobotId || !jobsPerRobot[rid]) rid = minRobotId;

  jobsPerRobot[rid].push({
    id: j.id,
    idStr: String(j.id),
    x, y,
    cell,
    distFromStart: (d0 >= 0 ? d0 : 1e9),
    prioNorm
  });
}

// Local BFS helper with stamping (for early steps / small instances)
const visitStamp = new Int32Array(N);
const distWork = new Int32Array(N);
const markStamp = new Int32Array(N);
const markId = new Int32Array(N);
let stamp = 0;

function bfsChooseAmongCandidates(startCell, candCells, candPrioNorm, prioBias) {
  // Returns: {bestIdx: i, bestDist: d} where i is index in candCells (0..k-1)
  stamp++;
  const st = stamp;

  const k = candCells.length;
  for (let i = 0; i < k; i++) {
    const c = candCells[i];
    if (c >= 0 && c < N) {
      markStamp[c] = st;
      markId[c] = i;
    }
  }

  let head = 0, tail = 0;
  queue[tail++] = startCell;
  visitStamp[startCell] = st;
  distWork[startCell] = 0;

  let bestDist = Infinity;
  let bestI = -1;
  let bestScore = Infinity;

  while (head < tail) {
    const v = queue[head++];
    const dv = distWork[v];
    if (dv > bestDist) break;

    if (markStamp[v] === st) {
      const i = markId[v];
      // score: distance with a mild priority preference
      const score = dv - candPrioNorm[i] * prioBias;
      if (score < bestScore) {
        bestScore = score;
        bestDist = dv;
        bestI = i;
      } else if (score === bestScore) {
        // tie-breaker: lower raw distance
        if (dv < bestDist) {
          bestDist = dv;
          bestI = i;
        }
      }
    }

    const nd = dv + 1;
    const xv = xOf[v];

    // left
    if (xv > 0) {
      const nb = v - 1;
      if (open[nb] && visitStamp[nb] !== st) {
        visitStamp[nb] = st; distWork[nb] = nd; queue[tail++] = nb;
      }
    }
    // right
    if (xv + 1 < W) {
      const nb = v + 1;
      if (open[nb] && visitStamp[nb] !== st) {
        visitStamp[nb] = st; distWork[nb] = nd; queue[tail++] = nb;
      }
    }
    // up
    if (v >= W) {
      const nb = v - W;
      if (open[nb] && visitStamp[nb] !== st) {
        visitStamp[nb] = st; distWork[nb] = nd; queue[tail++] = nb;
      }
    }
    // down
    if (v < N - W) {
      const nb = v + W;
      if (open[nb] && visitStamp[nb] !== st) {
        visitStamp[nb] = st; distWork[nb] = nd; queue[tail++] = nb;
      }
    }
  }

  return { bestIdx: bestI, bestDist: bestDist === Infinity ? -1 : bestDist };
}

function absInt(a) { return a < 0 ? -a : a; }

function pickCandidatesByApprox(remaining, curX, curY, k, prioBias) {
  // Returns arrays {pos[], cells[], prioNorm[], approxScore[]}, size <= k
  const pos = [];
  const cells = [];
  const prios = [];
  const approx = [];

  for (let p = 0; p < remaining.length; p++) {
    const j = remaining[p];
    const man = absInt(j.x - curX) + absInt(j.y - curY);
    const score = man - j.prioNorm * prioBias;

    // insert into sorted arrays (ascending score), keep length <= k
    let ins = pos.length;
    while (ins > 0 && score < approx[ins - 1]) ins--;
    if (ins < k) {
      pos.splice(ins, 0, p);
      cells.splice(ins, 0, j.cell);
      prios.splice(ins, 0, j.prioNorm);
      approx.splice(ins, 0, score);
      if (pos.length > k) {
        pos.pop(); cells.pop(); prios.pop(); approx.pop();
      }
    }
  }
  return { pos, cells, prioNorm: prios, approxScore: approx };
}

function compareRest(a, b) {
  if (a.distFromStart !== b.distFromStart) return a.distFromStart - b.distFromStart;
  // higher priority first as a mild tie-breaker
  if (a.prioNorm !== b.prioNorm) return b.prioNorm - a.prioNorm;
  return a.idStr < b.idStr ? -1 : (a.idStr > b.idStr ? 1 : 0);
}

const totalJobs = jobsIn.length;
const smallInstance = totalJobs <= 250;

for (let rid = 0; rid <= maxRobotId; rid++) {
  const r = robotById.get(rid);
  const jobs = jobsPerRobot[rid] || [];
  if (!r || jobs.length === 0) continue;

  // Routing
  const remaining = jobs.slice(); // we will mutate
  let curX = r.x, curY = r.y;
  let curCell = (curX >= 0 && curX < W && curY >= 0 && curY < H) ? idxOf(curX, curY) : -1;
  if (curCell >= 0) open[curCell] = 1;

  const out = assignments[rid];
  const prioBias = 0.5; // mild influence
  const candK = 10;

  const bfsStepsCap = smallInstance ? remaining.length : Math.min(20, remaining.length);

  for (let step = 0; step < bfsStepsCap && remaining.length > 0; step++) {
    if (curCell < 0 || curCell >= N) break; // fallback to sorting rest
    const cands = pickCandidatesByApprox(remaining, curX, curY, candK, prioBias);
    if (cands.pos.length === 0) break;

    let chosenPosInRemaining = cands.pos[0]; // fallback: best approx

    // If candidate cells are invalid, just use approx
    let allBad = true;
    for (let i = 0; i < cands.cells.length; i++) {
      const cc = cands.cells[i];
      if (cc >= 0 && cc < N) { allBad = false; break; }
    }

    if (!allBad) {
      const res = bfsChooseAmongCandidates(curCell, cands.cells, cands.prioNorm, prioBias);
      if (res.bestIdx >= 0) {
        chosenPosInRemaining = cands.pos[res.bestIdx];
      }
    }

    const chosen = remaining[chosenPosInRemaining];
    out.push(chosen.id);

    curX = chosen.x;
    curY = chosen.y;
    curCell = chosen.cell;

    // remove chosen by swap-pop
    const last = remaining.length - 1;
    remaining[chosenPosInRemaining] = remaining[last];
    remaining.pop();
  }

  if (remaining.length > 0) {
    remaining.sort(compareRest);
    for (const j of remaining) out.push(j.id);
  }
}

process.stdout.write(JSON.stringify({ assignments }));
```
