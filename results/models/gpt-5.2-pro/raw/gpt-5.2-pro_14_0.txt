```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    // Cell states in grid[y][x]
    private const byte D0 = 0;   // 0..8 are digits
    private const byte UNKNOWN = 9;
    private const byte MINE = 10;      // flagged '*', or inferred mine
    private const byte SAFE = 11;      // inferred safe (was UNKNOWN)

    private static int H, W;
    private static byte[][] grid = null!;
    private static readonly int[] dx = { -1, 0, 1, -1, 1, -1, 0, 1 };
    private static readonly int[] dy = { -1, -1, -1, 0, 0, 1, 1, 1 };

    private static readonly List<int> safeCells = new(1024); // positions (y*W+x) that were UNKNOWN and proven SAFE

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        H = fs.ReadInt();
        W = fs.ReadInt();

        grid = new byte[H][];
        var digitPositions = new List<int>(capacity: Math.Min(H * W, 1_000_000)); // only used for "small board" mode

        long area = (long)H * W;
        bool smallBoardMode = area <= 5_000_000; // adaptive threshold

        for (int y = 0; y < H; y++)
        {
            var row = new byte[W];
            grid[y] = row;

            fs.ReadRowInto(row, W);

            if (smallBoardMode)
            {
                for (int x = 0; x < W; x++)
                {
                    byte s = row[x];
                    if (s <= 8) digitPositions.Add(y * W + x);
                }
            }
        }

        if (!smallBoardMode)
        {
            // Huge boards: do a single full-scan propagation (plus queue ripple),
            // then stop (advanced CSP is skipped for scalability).
            RunBasicPropagationFullScan();
        }
        else
        {
            // Small/medium boards: iterate basic propagation + advanced frontier reasoning.
            // Each iteration is correct; we stop at fixpoint.
            const int MaxIterations = 50;
            for (int iter = 0; iter < MaxIterations; iter++)
            {
                bool changed1 = RunBasicPropagationSeeded(digitPositions);
                bool changed2 = RunAdvancedInference(digitPositions);
                if (!changed1 && !changed2) break;
            }
        }

        // Output all proven-safe cells (only those that were ' ' in input, i.e., UNKNOWN->SAFE)
        var sb = new StringBuilder(safeCells.Count * 12 + 32);
        sb.AppendLine(safeCells.Count.ToString());
        foreach (int pos in safeCells)
        {
            int x = pos % W;
            int y = pos / W;
            sb.Append(x).Append(' ').Append(y).Append('\n');
        }
        Console.Out.Write(sb.ToString());
    }

    // ---------------- Basic deterministic propagation (always sound) ----------------

    private static bool RunBasicPropagationSeeded(List<int> seedDigits)
    {
        // Queue is a growing list with head index; duplicates are allowed (keeps memory low vs. "inQueue" arrays).
        var queue = new List<int>(seedDigits.Count + 1024);
        queue.AddRange(seedDigits);

        bool anyChanged = false;
        int head = 0;
        while (head < queue.Count)
        {
            int pos = queue[head++];
            int x = pos % W;
            int y = pos / W;
            byte s = grid[y][x];
            if (s > 8) continue; // not a digit
            if (ProcessDigitCell(pos, queue)) anyChanged = true;
        }
        return anyChanged;
    }

    private static void RunBasicPropagationFullScan()
    {
        // For very large boards: scan all digits once, enqueue ripples, then process queue until empty.
        var queue = new List<int>(capacity: 1024);

        for (int y = 0; y < H; y++)
        {
            var row = grid[y];
            for (int x = 0; x < W; x++)
            {
                byte s = row[x];
                if (s <= 8)
                {
                    ProcessDigitCell(y * W + x, queue);
                }
            }
        }

        int head = 0;
        while (head < queue.Count)
        {
            int pos = queue[head++];
            int x = pos % W;
            int y = pos / W;
            byte s = grid[y][x];
            if (s > 8) continue;
            ProcessDigitCell(pos, queue);
        }
    }

    private static bool ProcessDigitCell(int digitPos, List<int> queue)
    {
        int x = digitPos % W;
        int y = digitPos / W;

        byte digit = grid[y][x]; // 0..8
        int flagged = 0;

        int unkCount = 0;
        Span<int> unk = stackalloc int[8];

        for (int k = 0; k < 8; k++)
        {
            int nx = x + dx[k];
            int ny = y + dy[k];
            if ((uint)nx >= (uint)W || (uint)ny >= (uint)H) continue;

            byte st = grid[ny][nx];
            if (st == MINE) flagged++;
            else if (st == UNKNOWN)
            {
                unk[unkCount++] = ny * W + nx;
            }
        }

        int remaining = digit - flagged;
        if (remaining < 0) return false; // inconsistent board; ignore deductions from this digit

        bool changed = false;

        if (remaining == 0 && unkCount > 0)
        {
            for (int i = 0; i < unkCount; i++)
            {
                int p = unk[i];
                if (MarkSafe(p))
                {
                    changed = true;
                    EnqueueAdjacentDigits(p, queue);
                }
            }
        }
        else if (remaining == unkCount && remaining > 0)
        {
            for (int i = 0; i < unkCount; i++)
            {
                int p = unk[i];
                if (MarkMine(p))
                {
                    changed = true;
                    EnqueueAdjacentDigits(p, queue);
                }
            }
        }

        return changed;
    }

    private static void EnqueueAdjacentDigits(int cellPos, List<int> queue)
    {
        int x = cellPos % W;
        int y = cellPos / W;

        for (int k = 0; k < 8; k++)
        {
            int nx = x + dx[k];
            int ny = y + dy[k];
            if ((uint)nx >= (uint)W || (uint)ny >= (uint)H) continue;

            byte st = grid[ny][nx];
            if (st <= 8) queue.Add(ny * W + nx);
        }
    }

    private static bool MarkSafe(int pos)
    {
        int x = pos % W;
        int y = pos / W;
        if (grid[y][x] != UNKNOWN) return false;
        grid[y][x] = SAFE;
        safeCells.Add(pos);
        return true;
    }

    private static bool MarkMine(int pos)
    {
        int x = pos % W;
        int y = pos / W;
        if (grid[y][x] != UNKNOWN) return false;
        grid[y][x] = MINE;
        return true;
    }

    // ---------------- Advanced inference for small/medium boards ----------------
    // Uses:
    //  1) Component extraction on the frontier (unknown cells adjacent to digits)
    //  2) Exact backtracking for tiny components
    //  3) Subset rule (constraint inclusion) for small components
    //
    // Always sound; never guesses. Skips too-large frontiers to stay scalable.

    private readonly struct Constraint
    {
        public readonly int[] Vars; // global var ids
        public readonly int Mines;  // mines among Vars

        public Constraint(int[] vars, int mines)
        {
            Vars = vars;
            Mines = mines;
        }
    }

    private static bool RunAdvancedInference(List<int> digitPositions)
    {
        // Build frontier variables and constraints from current grid state.
        // Each constraint is a digit cell: remaining mines among its adjacent UNKNOWN cells.
        var varMap = new Dictionary<int, int>(capacity: 1024); // cellPos -> varId
        var varPos = new List<int>(capacity: 1024);            // varId -> cellPos
        var constraints = new List<Constraint>(capacity: 1024);

        // Build constraints
        for (int idx = 0; idx < digitPositions.Count; idx++)
        {
            int dpos = digitPositions[idx];
            int x = dpos % W;
            int y = dpos / W;
            byte s = grid[y][x];
            if (s > 8) continue;

            int flagged = 0;
            int unkCount = 0;
            Span<int> unkPos = stackalloc int[8];

            for (int k = 0; k < 8; k++)
            {
                int nx = x + dx[k];
                int ny = y + dy[k];
                if ((uint)nx >= (uint)W || (uint)ny >= (uint)H) continue;

                byte st = grid[ny][nx];
                if (st == MINE) flagged++;
                else if (st == UNKNOWN)
                    unkPos[unkCount++] = ny * W + nx;
            }

            if (unkCount == 0) continue;

            int remaining = s - flagged;
            if ((uint)remaining > (uint)unkCount) continue; // inconsistent; ignore

            // Basic rules should have already stabilized, but keep these safe checks anyway.
            if (remaining == 0)
            {
                for (int i = 0; i < unkCount; i++) MarkSafe(unkPos[i]);
                continue;
            }
            if (remaining == unkCount)
            {
                for (int i = 0; i < unkCount; i++) MarkMine(unkPos[i]);
                continue;
            }

            int[] vars = new int[unkCount];
            for (int i = 0; i < unkCount; i++)
            {
                int p = unkPos[i];
                if (!varMap.TryGetValue(p, out int vid))
                {
                    vid = varPos.Count;
                    varMap.Add(p, vid);
                    varPos.Add(p);
                }
                vars[i] = vid;
            }
            Array.Sort(vars); // needed for subset checks
            constraints.Add(new Constraint(vars, remaining));
        }

        // Too large? Skip advanced.
        if (varPos.Count > 200_000 || constraints.Count > 200_000)
            return false;

        if (varPos.Count == 0 || constraints.Count == 0)
            return false;

        // Build var->constraints incidence for component extraction.
        var varToCons = new List<int>[varPos.Count];
        for (int i = 0; i < varToCons.Length; i++) varToCons[i] = new List<int>(4);

        for (int ci = 0; ci < constraints.Count; ci++)
        {
            var c = constraints[ci];
            for (int j = 0; j < c.Vars.Length; j++)
                varToCons[c.Vars[j]].Add(ci);
        }

        var visVar = new bool[varPos.Count];
        var visCon = new bool[constraints.Count];

        bool anyChanged = false;

        // Component BFS on bipartite graph (vars <-> constraints)
        var varStack = new Stack<int>(128);
        var conStack = new Stack<int>(128);
        var compVars = new List<int>(128);
        var compCons = new List<int>(128);

        for (int startVar = 0; startVar < varPos.Count; startVar++)
        {
            if (visVar[startVar]) continue;

            compVars.Clear();
            compCons.Clear();
            varStack.Clear();
            conStack.Clear();

            visVar[startVar] = true;
            varStack.Push(startVar);

            while (varStack.Count > 0 || conStack.Count > 0)
            {
                while (varStack.Count > 0)
                {
                    int v = varStack.Pop();
                    compVars.Add(v);
                    foreach (int ci in varToCons[v])
                    {
                        if (!visCon[ci])
                        {
                            visCon[ci] = true;
                            conStack.Push(ci);
                        }
                    }
                }

                while (conStack.Count > 0)
                {
                    int ci = conStack.Pop();
                    compCons.Add(ci);
                    var c = constraints[ci];
                    for (int j = 0; j < c.Vars.Length; j++)
                    {
                        int v = c.Vars[j];
                        if (!visVar[v])
                        {
                            visVar[v] = true;
                            varStack.Push(v);
                        }
                    }
                }
            }

            // Choose strategy by component size
            int nVars = compVars.Count;
            int nCons = compCons.Count;

            if (nVars <= 20 && nCons <= 64)
            {
                if (ExactSolveComponent(compVars, compCons, constraints, varPos))
                    anyChanged = true;
            }
            else if (nVars <= 200 && nCons <= 400)
            {
                // Subset inference; if it changes anything, we stop early (constraints become stale)
                if (SubsetInferComponent(compCons, constraints, varPos))
                {
                    anyChanged = true;
                    break;
                }
            }
        }

        return anyChanged;
    }

    private static bool ExactSolveComponent(
        List<int> compVarsGlobal,
        List<int> compConsGlobal,
        List<Constraint> allConstraints,
        List<int> varPos)
    {
        // Map global varId -> local [0..n-1]
        int n = compVarsGlobal.Count;
        var map = new Dictionary<int, int>(n * 2);
        for (int i = 0; i < n; i++) map[compVarsGlobal[i]] = i;

        // Build local constraints (vars are local indices)
        int m = compConsGlobal.Count;
        int[][] consVars = new int[m][];
        int[] consMines = new int[m];
        for (int i = 0; i < m; i++)
        {
            var c = allConstraints[compConsGlobal[i]];
            int[] lv = new int[c.Vars.Length];
            for (int j = 0; j < c.Vars.Length; j++)
                lv[j] = map[c.Vars[j]];
            Array.Sort(lv);
            consVars[i] = lv;
            consMines[i] = c.Mines;
        }

        // var -> constraints
        var varToCons = new List<int>[n];
        for (int i = 0; i < n; i++) varToCons[i] = new List<int>(4);
        for (int ci = 0; ci < m; ci++)
            for (int j = 0; j < consVars[ci].Length; j++)
                varToCons[consVars[ci][j]].Add(ci);

        // Order vars by degree descending (heuristic)
        int[] order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        Array.Sort(order, (a, b) => varToCons[b].Count.CompareTo(varToCons[a].Count));

        int[] assignedCount = new int[m];
        int[] assignedMines = new int[m];
        int[] assignment = new int[n]; // local var -> 0/1
        long solutions = 0;
        long[] mineCount = new long[n];

        // Backtracking
        bool Dfs(int idx)
        {
            if (idx == n)
            {
                solutions++;
                for (int i = 0; i < n; i++)
                    if (assignment[i] == 1) mineCount[i]++;
                return true;
            }

            int v = order[idx];

            // Try 0 then 1
            for (int val = 0; val <= 1; val++)
            {
                assignment[v] = val;

                bool ok = true;
                var touched = varToCons[v];
                for (int ti = 0; ti < touched.Count; ti++)
                {
                    int ci = touched[ti];
                    assignedCount[ci]++;
                    if (val == 1) assignedMines[ci]++;

                    int mines = consMines[ci];
                    int aM = assignedMines[ci];
                    int aC = assignedCount[ci];
                    int total = consVars[ci].Length;

                    // feasibility:
                    if (aM > mines) ok = false;
                    else if (aM + (total - aC) < mines) ok = false;
                }

                if (ok)
                    Dfs(idx + 1);

                // undo
                for (int ti = 0; ti < touched.Count; ti++)
                {
                    int ci = touched[ti];
                    if (val == 1) assignedMines[ci]--;
                    assignedCount[ci]--;
                }
            }

            return true;
        }

        Dfs(0);

        if (solutions == 0) return false; // inconsistent component; ignore

        bool changed = false;
        for (int localVar = 0; localVar < n; localVar++)
        {
            long minesHere = mineCount[localVar];
            int globalVar = compVarsGlobal[localVar];
            int cellPos = varPos[globalVar];

            if (minesHere == 0)
            {
                if (MarkSafe(cellPos)) changed = true;
            }
            else if (minesHere == solutions)
            {
                if (MarkMine(cellPos)) changed = true;
            }
        }
        return changed;
    }

    private static bool SubsetInferComponent(
        List<int> compConsGlobal,
        List<Constraint> allConstraints,
        List<int> varPos)
    {
        // Apply classic subset rule between constraints A and B:
        // If Vars(A) ⊆ Vars(B):
        //    Mines(B)-Mines(A) mines in (Vars(B)\Vars(A))
        //    if deltaMines == 0 => those extras are SAFE
        //    if deltaMines == |extras| => those extras are MINES
        //
        // Constraints are tiny (<=8 vars), so O(m^2 * 8) is fine here.
        int m = compConsGlobal.Count;

        for (int i = 0; i < m; i++)
        {
            var A = allConstraints[compConsGlobal[i]];
            for (int j = i + 1; j < m; j++)
            {
                var B = allConstraints[compConsGlobal[j]];

                if (TrySubsetDeduce(A, B, varPos)) return true;
                if (TrySubsetDeduce(B, A, varPos)) return true;
            }
        }

        return false;
    }

    private static bool TrySubsetDeduce(in Constraint small, in Constraint big, List<int> varPos)
    {
        // Check if small.Vars ⊆ big.Vars (both sorted), and deduce on big\small.
        var a = small.Vars;
        var b = big.Vars;
        if (a.Length >= b.Length) return false;

        // Two-pointer subset + collect extras from b
        Span<int> extras = stackalloc int[8];
        int eCount = 0;

        int i = 0, j = 0;
        while (i < a.Length && j < b.Length)
        {
            if (a[i] == b[j])
            {
                i++; j++;
            }
            else if (a[i] > b[j])
            {
                extras[eCount++] = b[j];
                j++;
            }
            else
            {
                return false; // a has an element not in b
            }
        }

        if (i != a.Length) return false; // not all matched
        while (j < b.Length) extras[eCount++] = b[j++];

        if (eCount == 0) return false;

        int deltaM = big.Mines - small.Mines;
        if (deltaM < 0 || deltaM > eCount) return false; // inconsistent; ignore

        bool changed = false;
        if (deltaM == 0)
        {
            for (int k = 0; k < eCount; k++)
            {
                int globalVar = extras[k];
                int cellPos = varPos[globalVar];
                if (MarkSafe(cellPos)) changed = true;
            }
        }
        else if (deltaM == eCount)
        {
            for (int k = 0; k < eCount; k++)
            {
                int globalVar = extras[k];
                int cellPos = varPos[globalVar];
                if (MarkMine(cellPos)) changed = true;
            }
        }

        return changed;
    }

    // ---------------- Fast input ----------------

    private sealed class FastScanner
    {
        private readonly Stream _s;
        private readonly byte[] _buf;
        private int _len, _ptr;

        public FastScanner(Stream s, int bufferSize = 1 << 20)
        {
            _s = s;
            _buf = new byte[bufferSize];
        }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }

        public int ReadInt()
        {
            int c;
            do { c = ReadByte(); } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int v = 0;
            while (c > 32 && c >= 0)
            {
                v = v * 10 + (c - '0');
                c = ReadByte();
            }
            return v * sign;
        }

        public void ReadRowInto(byte[] row, int w)
        {
            int i = 0;

            // Read until we fill w columns; if line ends early, pad with UNKNOWN.
            while (i < w)
            {
                int c = ReadByte();
                if (c < 0)
                {
                    // EOF: pad
                    while (i < w) row[i++] = UNKNOWN;
                    return;
                }

                if (c == '\r') continue;
                if (c == '\n')
                {
                    while (i < w) row[i++] = UNKNOWN;
                    return;
                }

                row[i++] = MapCell((byte)c);
            }

            // Consume rest of the line (if any) until newline
            while (true)
            {
                int c = ReadByte();
                if (c < 0) return;
                if (c == '\n') return;
                if (c == '\r') continue;
            }
        }

        private static byte MapCell(byte c)
        {
            if (c == (byte)' ') return UNKNOWN;
            if (c == (byte)'*') return MINE;
            if (c >= (byte)'0' && c <= (byte)'8') return (byte)(c - (byte)'0');
            // Treat anything unexpected as UNKNOWN (safe, conservative)
            return UNKNOWN;
        }
    }
}
```
