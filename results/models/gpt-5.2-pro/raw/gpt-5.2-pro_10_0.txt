#include <iostream>
#include <vector>
#include <algorithm>
#include <map>
#include <cstdint>
#include <limits>
#include <utility>

using std::cin;
using std::cout;
using std::vector;

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0; // 0 = as-given, 1 = rotated 90 deg
};

struct InputRect {
    int w = 0, h = 0;
    int idx = -1;
};

static inline bool containsRect(int ax, int ay, int aw, int ah, int bx, int by, int bw, int bh) {
    return bx >= ax && by >= ay && (bx + bw) <= (ax + aw) && (by + bh) <= (ay + ah);
}

/* ------------------------ Small-mode: MaxRects ------------------------ */
struct MRRect {
    int x, y, w, h;
};

struct MRScore {
    int shortSideFit;
    int longSideFit;
    bool fits;
};

class MaxRectsBin {
public:
    MaxRectsBin(int W, int H) : bw(W), bh(H) {
        freeRects.push_back({0, 0, bw, bh});
    }

    // Find best placement for (w,h) in current bin; does not modify.
    bool findBest(int w, int h, MRRect& out, int& outRot, MRScore& outScore) const {
        outScore.fits = false;
        outScore.shortSideFit = std::numeric_limits<int>::max();
        outScore.longSideFit = std::numeric_limits<int>::max();

        MRRect bestNode{0,0,0,0};
        int bestRot = 0;

        for (const auto& fr : freeRects) {
            // Not rotated
            if (w <= fr.w && h <= fr.h) {
                int leftoverH = fr.w - w;
                int leftoverV = fr.h - h;
                int ssf = std::min(leftoverH, leftoverV);
                int lsf = std::max(leftoverH, leftoverV);
                if (!outScore.fits || ssf < outScore.shortSideFit || (ssf == outScore.shortSideFit && lsf < outScore.longSideFit)) {
                    outScore = {ssf, lsf, true};
                    bestNode = {fr.x, fr.y, w, h};
                    bestRot = 0;
                }
            }
            // Rotated
            if (h <= fr.w && w <= fr.h) {
                int rw = h, rh = w;
                int leftoverH = fr.w - rw;
                int leftoverV = fr.h - rh;
                int ssf = std::min(leftoverH, leftoverV);
                int lsf = std::max(leftoverH, leftoverV);
                if (!outScore.fits || ssf < outScore.shortSideFit || (ssf == outScore.shortSideFit && lsf < outScore.longSideFit)) {
                    outScore = {ssf, lsf, true};
                    bestNode = {fr.x, fr.y, rw, rh};
                    bestRot = 1;
                }
            }
        }

        if (!outScore.fits) return false;
        out = bestNode;
        outRot = bestRot;
        return true;
    }

    // Insert a rectangle at the computed best position; modifies free list.
    bool insertAt(const MRRect& placed) {
        bool anySplit = false;
        vector<MRRect> newFree;
        newFree.reserve(freeRects.size() + 8);

        for (const auto& fr : freeRects) {
            if (!intersects(fr, placed)) {
                newFree.push_back(fr);
                continue;
            }
            anySplit = true;
            splitFree(fr, placed, newFree);
        }

        if (!anySplit) return false; // should not happen if chosen from freeRects

        freeRects.swap(newFree);
        pruneFreeList();
        return true;
    }

private:
    int bw, bh;
    vector<MRRect> freeRects;

    static inline bool intersects(const MRRect& a, const MRRect& b) {
        if (b.x >= a.x + a.w) return false;
        if (b.x + b.w <= a.x) return false;
        if (b.y >= a.y + a.h) return false;
        if (b.y + b.h <= a.y) return false;
        return true;
    }

    static inline void splitFree(const MRRect& fr, const MRRect& used, vector<MRRect>& out) {
        // Left
        if (used.x > fr.x) {
            int w = used.x - fr.x;
            if (w > 0) out.push_back({fr.x, fr.y, w, fr.h});
        }
        // Right
        int frRight = fr.x + fr.w;
        int usedRight = used.x + used.w;
        if (usedRight < frRight) {
            int w = frRight - usedRight;
            if (w > 0) out.push_back({usedRight, fr.y, w, fr.h});
        }
        // Bottom
        if (used.y > fr.y) {
            int h = used.y - fr.y;
            if (h > 0) out.push_back({fr.x, fr.y, fr.w, h});
        }
        // Top
        int frTop = fr.y + fr.h;
        int usedTop = used.y + used.h;
        if (usedTop < frTop) {
            int h = frTop - usedTop;
            if (h > 0) out.push_back({fr.x, usedTop, fr.w, h});
        }
    }

    void pruneFreeList() {
        // Remove any free rect that is fully contained in another.
        for (size_t i = 0; i < freeRects.size(); ++i) {
            for (size_t j = i + 1; j < freeRects.size(); ) {
                const auto& A = freeRects[i];
                const auto& B = freeRects[j];
                if (containsRect(A.x, A.y, A.w, A.h, B.x, B.y, B.w, B.h)) {
                    freeRects.erase(freeRects.begin() + static_cast<std::ptrdiff_t>(j));
                    continue;
                }
                if (containsRect(B.x, B.y, B.w, B.h, A.x, A.y, A.w, A.h)) {
                    freeRects.erase(freeRects.begin() + static_cast<std::ptrdiff_t>(i));
                    --i;
                    break;
                }
                ++j;
            }
        }
    }
};

struct PackResult {
    int numBoards = 0;
    vector<Placement> placements;
};

static PackResult solveSmall_MaxRectsMultiTry(const vector<InputRect>& in, int BW, int BH) {
    auto run = [&](int mode) -> PackResult {
        // mode 0: sort by area desc, mode 1: sort by maxside desc then area
        vector<InputRect> rects = in;
        std::sort(rects.begin(), rects.end(), [&](const InputRect& a, const InputRect& b) {
            int64_t aa = int64_t(a.w) * a.h;
            int64_t bb = int64_t(b.w) * b.h;
            int am = std::max(a.w, a.h);
            int bm = std::max(b.w, b.h);
            if (mode == 0) {
                if (aa != bb) return aa > bb;
                if (am != bm) return am > bm;
                return a.idx < b.idx;
            } else {
                if (am != bm) return am > bm;
                if (aa != bb) return aa > bb;
                return a.idx < b.idx;
            }
        });

        PackResult res;
        res.placements.assign(in.size(), Placement{});

        vector<MaxRectsBin> bins;
        bins.reserve(128);

        for (const auto& r : rects) {
            bool fit0 = (r.w <= BW && r.h <= BH);
            bool fit1 = (r.h <= BW && r.w <= BH);
            if (!fit0 && !fit1) {
                // Unplaceable; leave invalid (should not happen in valid tasks)
                continue;
            }

            int bestBin = -1;
            MRRect bestNode{0,0,0,0};
            int bestRot = 0;
            MRScore bestScore{0,0,false};

            // Best-fit across existing bins
            for (int bi = 0; bi < (int)bins.size(); ++bi) {
                MRRect node;
                int rot;
                MRScore score;
                if (bins[bi].findBest(r.w, r.h, node, rot, score)) {
                    if (bestBin < 0 ||
                        score.shortSideFit < bestScore.shortSideFit ||
                        (score.shortSideFit == bestScore.shortSideFit && score.longSideFit < bestScore.longSideFit)) {
                        bestBin = bi;
                        bestNode = node;
                        bestRot = rot;
                        bestScore = score;
                    }
                }
            }

            if (bestBin < 0) {
                // Open new bin
                bins.emplace_back(BW, BH);
                int bi = (int)bins.size() - 1;
                MRRect node;
                int rot;
                MRScore score;
                bool ok = bins[bi].findBest(r.w, r.h, node, rot, score);
                if (!ok) {
                    // should not happen if fits0/fit1 true and bin is empty
                    continue;
                }
                bins[bi].insertAt(node);
                res.placements[r.idx] = Placement{bi, node.x, node.y, rot};
            } else {
                bins[bestBin].insertAt(bestNode);
                res.placements[r.idx] = Placement{bestBin, bestNode.x, bestNode.y, bestRot};
            }
        }

        res.numBoards = (int)bins.size();
        return res;
    };

    PackResult a = run(0);
    PackResult b = run(1);

    // Prefer fewer boards; tie-break by total waste proxy (unused boards count already; keep a)
    if (b.numBoards < a.numBoards) return b;
    return a;
}

/* ------------------------ Large-mode: Global best-fit shelves ------------------------ */

struct OrientedRect {
    int w = 0, h = 0;
    int rot = 0;
    int idx = -1;
};

struct Shelf {
    int y = 0;
    int height = 0;
    int usedW = 0;

    bool inMap = false;
    std::multimap<int, std::pair<int,int>>::iterator it; // remaining width -> (binId, shelfId)
};

struct Bin {
    int usedH = 0;
    int remH = 0;

    vector<Shelf> shelves;

    bool inMap = false;
    std::multimap<int, int>::iterator it; // remaining height -> binId
};

static PackResult solveLarge_GlobalShelves(const vector<InputRect>& in, int BW, int BH) {
    vector<OrientedRect> rects;
    rects.reserve(in.size());

    for (auto &r : in) {
        bool fit0 = (r.w <= BW && r.h <= BH);
        bool fit1 = (r.h <= BW && r.w <= BH);
        if (!fit0 && !fit1) {
            // Unplaceable (should not happen in valid tasks)
            // Put a dummy; will remain unplaced.
            rects.push_back({r.w, r.h, 0, r.idx});
            continue;
        }
        int w0 = r.w, h0 = r.h;
        int w1 = r.h, h1 = r.w;

        // Choose orientation that makes height as large as possible (to support decreasing-height shelf packing),
        // while still fitting within board.
        int chosenW, chosenH, chosenRot;
        if (fit0 && fit1) {
            // Prefer height=max(w,h), width=min(w,h) if that orientation fits.
            if (std::max(r.w, r.h) <= BH && std::min(r.w, r.h) <= BW) {
                chosenH = std::max(r.w, r.h);
                chosenW = std::min(r.w, r.h);
                // Determine rot relative to input
                if (chosenW == r.w && chosenH == r.h) chosenRot = 0;
                else chosenRot = 1;
            } else {
                // Otherwise choose whichever has larger height among fitting
                if (h0 >= h1) { chosenW = w0; chosenH = h0; chosenRot = 0; }
                else { chosenW = w1; chosenH = h1; chosenRot = 1; }
            }
        } else if (fit0) {
            chosenW = w0; chosenH = h0; chosenRot = 0;
        } else {
            chosenW = w1; chosenH = h1; chosenRot = 1;
        }

        rects.push_back({chosenW, chosenH, chosenRot, r.idx});
    }

    std::sort(rects.begin(), rects.end(), [&](const OrientedRect& a, const OrientedRect& b) {
        if (a.h != b.h) return a.h > b.h;
        if (a.w != b.w) return a.w > b.w;
        return a.idx < b.idx;
    });

    PackResult res;
    res.placements.assign(in.size(), Placement{});

    vector<Bin> bins;
    bins.reserve(256);

    // Global maps:
    // - shelves: remaining width -> (binId, shelfId)
    // - bins: remaining height -> binId (eligible for creating a new shelf)
    std::multimap<int, std::pair<int,int>> shelfByRemW;
    std::multimap<int, int> binByRemH;

    auto addBinToMap = [&](int binId) {
        Bin& b = bins[binId];
        if (b.inMap) {
            binByRemH.erase(b.it);
            b.inMap = false;
        }
        if (b.remH > 0) {
            b.it = binByRemH.insert({b.remH, binId});
            b.inMap = true;
        }
    };

    auto addShelfToMap = [&](int binId, int shelfId) {
        Shelf& s = bins[binId].shelves[shelfId];
        if (s.inMap) {
            shelfByRemW.erase(s.it);
            s.inMap = false;
        }
        int remW = BW - s.usedW;
        if (remW > 0) {
            s.it = shelfByRemW.insert({remW, {binId, shelfId}});
            s.inMap = true;
        }
    };

    for (const auto& r : rects) {
        // Skip impossible ones (should not occur)
        if (r.w > BW || r.h > BH) continue;

        // 1) Best-fit into any existing shelf by remaining width (height monotonic ensures shelf.height >= r.h)
        auto itShelf = shelfByRemW.lower_bound(r.w);
        if (itShelf != shelfByRemW.end()) {
            int binId = itShelf->second.first;
            int shelfId = itShelf->second.second;

            Shelf& s = bins[binId].shelves[shelfId];

            // Safety check: in this large-mode design this should always hold.
            if (r.h <= s.height && s.usedW + r.w <= BW) {
                int x = s.usedW;
                int y = s.y;
                s.usedW += r.w;

                // Update shelf key
                addShelfToMap(binId, shelfId);

                res.placements[r.idx] = Placement{binId, x, y, r.rot};
                continue;
            } else {
                // If it happens due to numerical/logic mismatch, fall through to new shelf.
                // Remove this shelf from map to avoid repeated bad hits.
                shelfByRemW.erase(itShelf);
                s.inMap = false;
            }
        }

        // 2) Create a new shelf in an existing bin with enough remaining height (best-fit by remaining height)
        auto itBin = binByRemH.lower_bound(r.h);
        if (itBin != binByRemH.end()) {
            int binId = itBin->second;
            Bin& b = bins[binId];

            // Update bin map entry (will reinsert after changing remH)
            if (b.inMap) {
                binByRemH.erase(b.it);
                b.inMap = false;
            }

            int y = b.usedH;
            int shelfId = (int)b.shelves.size();
            b.shelves.push_back(Shelf{y, r.h, r.w});

            b.usedH += r.h;
            b.remH = BH - b.usedH;

            // Reinsert bin to map if still has height
            addBinToMap(binId);

            // Insert shelf to shelf map
            addShelfToMap(binId, shelfId);

            res.placements[r.idx] = Placement{binId, 0, y, r.rot};
            continue;
        }

        // 3) Open a new bin
        int binId = (int)bins.size();
        bins.push_back(Bin{0, BH, {}});

        Bin& b = bins.back();
        b.usedH = r.h;
        b.remH = BH - b.usedH;
        b.shelves.push_back(Shelf{0, r.h, r.w});

        addBinToMap(binId);
        addShelfToMap(binId, 0);

        res.placements[r.idx] = Placement{binId, 0, 0, r.rot};
    }

    res.numBoards = (int)bins.size();
    return res;
}

/* ------------------------ Main ------------------------ */

int main() {
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N = 0;
    int BW = 0, BH = 0;
    if (!(cin >> N >> BW >> BH)) return 0;

    vector<InputRect> rects;
    rects.reserve((size_t)N);

    for (int i = 0; i < N; ++i) {
        int w, h;
        cin >> w >> h;
        rects.push_back({w, h, i});
    }

    PackResult res;
    // Adaptive strategy:
    // - Small N: higher-quality MaxRects (two orderings)
    // - Large N: very fast global best-fit shelves with global maps (O(N log S))
    const int SMALL_THRESHOLD = 3000;
    if (N <= SMALL_THRESHOLD) {
        res = solveSmall_MaxRectsMultiTry(rects, BW, BH);
    } else {
        res = solveLarge_GlobalShelves(rects, BW, BH);
    }

    cout << res.numBoards << "\n";
    // Output one line per rectangle in input order
    // board_index is 0-based: [0, numBoards-1]
    for (int i = 0; i < N; ++i) {
        const Placement& p = res.placements[i];
        // If something went wrong/unplaceable, still print a line (but validity would fail).
        cout << p.board << " " << p.x << " " << p.y << " " << p.rotated << "\n";
    }

    return 0;
}
