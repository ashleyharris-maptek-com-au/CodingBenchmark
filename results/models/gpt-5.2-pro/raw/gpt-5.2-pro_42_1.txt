; SPIR-V
; Version: 1.0
; Generator: hand-written
; Bound: 100
OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_in 0 Offset 0

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 Offset 0

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void

%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%float_h = OpConstant %float 0.05

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%arr_v3_in = OpTypeArray %v3float %uint_1

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %fn_void
%label = OpLabel

; Load input position gl_in[0].gl_Position
%inPosPtr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%inPos = OpLoad %v4float %inPosPtr
%x = OpCompositeExtract %float %inPos 0
%y = OpCompositeExtract %float %inPos 1
%z = OpCompositeExtract %float %inPos 2
%w = OpCompositeExtract %float %inPos 3

; Load input color color[0]
%inColPtr = OpAccessChain %ptr_in_v3 %color %int_0
%inCol = OpLoad %v3float %inColPtr

; Common output pointers
%outPosPtr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; ---- Vertex 0: Top (x, y+h) ----
%y_top = OpFAdd %float %y %float_h
%pos_top = OpCompositeConstruct %v4float %x %y_top %z %w
OpStore %outPosPtr %pos_top
OpStore %outColor %inCol
OpEmitVertex

; ---- Vertex 1: Right (x+h, y) ----
%x_right = OpFAdd %float %x %float_h
%pos_right = OpCompositeConstruct %v4float %x_right %y %z %w
OpStore %outPosPtr %pos_right
OpStore %outColor %inCol
OpEmitVertex

; ---- Vertex 2: Left (x-h, y) ----
%x_left = OpFSub %float %x %float_h
%pos_left = OpCompositeConstruct %v4float %x_left %y %z %w
OpStore %outPosPtr %pos_left
OpStore %outColor %inCol
OpEmitVertex

; ---- Vertex 3: Bottom (x, y-h) ----
%y_bottom = OpFSub %float %y %float_h
%pos_bottom = OpCompositeConstruct %v4float %x %y_bottom %z %w
OpStore %outPosPtr %pos_bottom
OpStore %outColor %inCol
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
