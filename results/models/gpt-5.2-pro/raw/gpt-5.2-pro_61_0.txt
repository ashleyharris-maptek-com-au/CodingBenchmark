declare const require: any;

const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8");
const data = JSON.parse(input);

type Pkg = { id: any; x: number; y: number };

const depot: [number, number] = data.depot;
const depotX = depot[0];
const depotY = depot[1];

const packages: Pkg[] = Array.isArray(data.packages) ? data.packages : [];
const n = packages.length;

let truckCount: number = Number.isFinite(data.truckCount) ? Math.floor(data.truckCount) : 1;
if (truckCount < 1) truckCount = 1;

if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

// Global arrays for speed
const ids: any[] = new Array(n);
const xs: number[] = new Array(n);
const ys: number[] = new Array(n);

for (let i = 0; i < n; i++) {
  const p = packages[i];
  ids[i] = p.id;
  xs[i] = p.x;
  ys[i] = p.y;
}

function distNode(a: number, b: number): number {
  // a/b: -1 means depot, otherwise package index
  const ax = a === -1 ? depotX : xs[a];
  const ay = a === -1 ? depotY : ys[a];
  const bx = b === -1 ? depotX : xs[b];
  const by = b === -1 ? depotY : ys[b];
  return Math.hypot(ax - bx, ay - by);
}

function twoOptBounded(route: number[]): void {
  const m = route.length;
  if (m < 4) return;

  // Tune: fewer checks for large routes; deterministic.
  const step = m > 400 ? Math.ceil(m / 200) : m > 200 ? 2 : 1;
  const maxChecks = Math.min(60000, Math.floor((m * m) / 2));
  const maxIter = m > 400 ? 2 : m > 200 ? 3 : 10;

  let checks = 0;
  for (let iter = 0; iter < maxIter; iter++) {
    let improved = false;

    for (let i = 0; i < m - 1; i += step) {
      const a = i === 0 ? -1 : route[i - 1];
      const b = route[i];

      for (let j = i + 1; j < m; j += step) {
        // Skip trivial adjacent swaps that don't change edges meaningfully
        if (j === i) continue;

        const c = route[j];
        const d = j === m - 1 ? -1 : route[j + 1];

        // Old edges: (a-b) + (c-d), New edges: (a-c) + (b-d)
        const oldLen = distNode(a, b) + distNode(c, d);
        const newLen = distNode(a, c) + distNode(b, d);

        checks++;
        if (newLen + 1e-12 < oldLen) {
          // Reverse segment [i..j]
          for (let l = i, r = j; l < r; l++, r--) {
            const tmp = route[l];
            route[l] = route[r];
            route[r] = tmp;
          }
          improved = true;
          break;
        }

        if (checks >= maxChecks) break;
      }
      if (improved || checks >= maxChecks) break;
    }

    if (!improved || checks >= maxChecks) break;
  }
}

function nearestNeighborSmall(route: number[]): number[] {
  const m = route.length;
  if (m <= 2) return route.slice();

  // Deterministic NN restricted to nodes in this route
  const unvisited: boolean[] = new Array(m).fill(true);
  const ordered: number[] = [];
  let curNode = -1; // depot

  for (let t = 0; t < m; t++) {
    let bestIdx = -1;
    let bestD = Infinity;

    for (let i = 0; i < m; i++) {
      if (!unvisited[i]) continue;
      const node = route[i];
      const d = distNode(curNode, node);
      if (d < bestD - 1e-15) {
        bestD = d;
        bestIdx = i;
      } else if (Math.abs(d - bestD) <= 1e-15) {
        // Tie-break deterministically by id stringified, then index
        const a = String(ids[node]);
        const b = bestIdx >= 0 ? String(ids[route[bestIdx]]) : "";
        if (a < b) bestIdx = i;
      }
    }

    unvisited[bestIdx] = false;
    const chosen = route[bestIdx];
    ordered.push(chosen);
    curNode = chosen;
  }

  return ordered;
}

// 1) Sweep clustering around depot
type Item = { idx: number; ang: number; r2: number };
const items: Item[] = new Array(n);

for (let i = 0; i < n; i++) {
  const dx = xs[i] - depotX;
  const dy = ys[i] - depotY;
  let ang = Math.atan2(dy, dx);
  if (ang < 0) ang += Math.PI * 2;
  const r2 = dx * dx + dy * dy;
  items[i] = { idx: i, ang, r2 };
}

items.sort((a, b) => {
  if (a.ang !== b.ang) return a.ang - b.ang;
  if (a.r2 !== b.r2) return a.r2 - b.r2;
  const sa = String(ids[a.idx]);
  const sb = String(ids[b.idx]);
  if (sa < sb) return -1;
  if (sa > sb) return 1;
  return a.idx - b.idx;
});

const routeCount = Math.min(truckCount, n);
const base = Math.floor(n / routeCount);
const extra = n % routeCount;

const routesIdx: number[][] = [];
let pos = 0;
for (let r = 0; r < routeCount; r++) {
  const size = base + (r < extra ? 1 : 0);
  const segment: number[] = new Array(size);
  for (let t = 0; t < size; t++) segment[t] = items[pos++].idx;

  // 2) In-route ordering improvement (bounded; deterministic)
  let ordered = segment;
  if (size <= 80) ordered = nearestNeighborSmall(segment);
  twoOptBounded(ordered);

  routesIdx.push(ordered);
}

// Output package IDs in visiting order for each route
const outRoutes: any[][] = routesIdx.map((rt) => rt.map((i) => ids[i]));
process.stdout.write(JSON.stringify({ routes: outRoutes }));
