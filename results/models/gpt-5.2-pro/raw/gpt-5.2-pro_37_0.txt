using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    // ---------- FastScanner ----------
    private sealed class FastScanner
    {
        private readonly Stream _s;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream s) { _s = s; }

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do { c = Read(); } while (c <= 32 && c != 0);

            int sign = 1;
            if (c == (byte)'-') { sign = -1; c = Read(); }

            int x = 0;
            while (c > 32)
            {
                x = x * 10 + (c - (byte)'0');
                c = Read();
            }
            return x * sign;
        }
    }

    // ---------- DFVS solver on one SCC (local graph) ----------
    private sealed class DfvsSolver
    {
        private readonly int n;
        private readonly int[] outStart, outAdj;
        private readonly int[] inStart, inAdj;

        private readonly bool[] hasSelfLoop;

        private readonly bool[] active;
        private readonly int[] inDeg, outDeg;
        private int activeCount;

        private readonly List<int> current = new();
        private List<int> best = new();
        private int bestSize;

        private struct TrailEntry
        {
            public byte Kind; // 0=active, 1=deg
            public int V;
            public bool OldActive;
            public int OldActiveCount;
            public int OldIn, OldOut;
        }
        private readonly List<TrailEntry> trail = new(1 << 14);

        // temp arrays for SCC/cycle detection
        private readonly bool[] visited;
        private readonly int[] order;
        private readonly int[] compId;
        private readonly int[] compSize;
        private readonly bool[] compHasCycle;
        private readonly int[] stackV;
        private readonly int[] stackI;

        private readonly int[] color;
        private readonly int[] parent;

        public DfvsSolver(int n, int[] outStart, int[] outAdj, int[] inStart, int[] inAdj, bool[] hasSelfLoop)
        {
            this.n = n;
            this.outStart = outStart;
            this.outAdj = outAdj;
            this.inStart = inStart;
            this.inAdj = inAdj;
            this.hasSelfLoop = hasSelfLoop;

            active = new bool[n];
            inDeg = new int[n];
            outDeg = new int[n];
            visited = new bool[n];
            order = new int[n];
            compId = new int[n];
            compSize = new int[n];
            compHasCycle = new bool[n];
            stackV = new int[n];
            stackI = new int[n];
            color = new int[n];
            parent = new int[n];

            ResetState();
        }

        private void ResetState()
        {
            Array.Fill(active, true);
            activeCount = n;
            for (int v = 0; v < n; v++)
            {
                outDeg[v] = outStart[v + 1] - outStart[v];
                inDeg[v] = inStart[v + 1] - inStart[v];
            }
            trail.Clear();
            current.Clear();
        }

        public List<int> Solve()
        {
            // Upper bound: greedy
            var greedy = GreedyUpperBound();
            best = greedy;
            bestSize = greedy.Count;

            // Exact search with BnB
            ResetState();
            Dfs();

            return best;
        }

        private List<int> GreedyUpperBound()
        {
            // Work on local clones
            bool[] a = new bool[n];
            int[] inD = new int[n];
            int[] outD = new int[n];
            Array.Fill(a, true);
            for (int v = 0; v < n; v++)
            {
                inD[v] = inStart[v + 1] - inStart[v];
                outD[v] = outStart[v + 1] - outStart[v];
            }

            var removed = new List<int>();

            // forced self-loops first
            for (int v = 0; v < n; v++)
            {
                if (a[v] && hasSelfLoop[v])
                    GreedyDeactivate(v, a, inD, outD, removed, countRemoval: true);
            }
            GreedyPeel(a, inD, outD);

            while (true)
            {
                var cycle = GreedyFindAnyCycle(a);
                if (cycle == null) break;

                int bestV = cycle[0];
                int bestScore = inD[bestV] + outD[bestV];
                for (int i = 1; i < cycle.Count; i++)
                {
                    int v = cycle[i];
                    int sc = inD[v] + outD[v];
                    if (sc > bestScore)
                    {
                        bestScore = sc;
                        bestV = v;
                    }
                }

                GreedyDeactivate(bestV, a, inD, outD, removed, countRemoval: true);

                // forced self-loops created? (self-loop property doesn't change, but vertex may still exist)
                for (int v = 0; v < n; v++)
                {
                    if (a[v] && hasSelfLoop[v])
                        GreedyDeactivate(v, a, inD, outD, removed, countRemoval: true);
                }

                GreedyPeel(a, inD, outD);
            }

            return removed;

            void GreedyDeactivate(int v, bool[] a2, int[] inD2, int[] outD2, List<int> rem, bool countRemoval)
            {
                if (!a2[v]) return;
                a2[v] = false;
                if (countRemoval) rem.Add(v);

                for (int ei = outStart[v]; ei < outStart[v + 1]; ei++)
                {
                    int w = outAdj[ei];
                    if (a2[w]) inD2[w]--;
                }
                for (int ei = inStart[v]; ei < inStart[v + 1]; ei++)
                {
                    int w = inAdj[ei];
                    if (a2[w]) outD2[w]--;
                }
            }

            void GreedyPeel(bool[] a2, int[] inD2, int[] outD2)
            {
                int[] q = new int[n];
                int qh = 0, qt = 0;
                for (int v = 0; v < n; v++)
                    if (a2[v] && (inD2[v] == 0 || outD2[v] == 0))
                        q[qt++] = v;

                while (qh < qt)
                {
                    int v = q[qh++];
                    if (!a2[v]) continue;
                    if (inD2[v] != 0 && outD2[v] != 0) continue;

                    a2[v] = false;

                    for (int ei = outStart[v]; ei < outStart[v + 1]; ei++)
                    {
                        int w = outAdj[ei];
                        if (!a2[w]) continue;
                        inD2[w]--;
                        if (inD2[w] == 0 || outD2[w] == 0) q[qt++] = w;
                    }
                    for (int ei = inStart[v]; ei < inStart[v + 1]; ei++)
                    {
                        int w = inAdj[ei];
                        if (!a2[w]) continue;
                        outD2[w]--;
                        if (inD2[w] == 0 || outD2[w] == 0) q[qt++] = w;
                    }
                }
            }

            List<int>? GreedyFindAnyCycle(bool[] a2)
            {
                Array.Fill(color, 0);
                Array.Fill(parent, -1);

                int top;
                for (int s = 0; s < n; s++)
                {
                    if (!a2[s] || color[s] != 0) continue;

                    top = 0;
                    stackV[top] = s;
                    stackI[top] = outStart[s];
                    top++;
                    parent[s] = -1;

                    while (top > 0)
                    {
                        int v = stackV[top - 1];
                        if (color[v] == 0) color[v] = 1;

                        int it = stackI[top - 1];
                        if (it < outStart[v + 1])
                        {
                            int w = outAdj[it];
                            stackI[top - 1] = it + 1;
                            if (!a2[w]) continue;

                            if (color[w] == 0)
                            {
                                parent[w] = v;
                                stackV[top] = w;
                                stackI[top] = outStart[w];
                                top++;
                            }
                            else if (color[w] == 1)
                            {
                                // cycle found, reconstruct vertices
                                var cyc = new List<int>();
                                cyc.Add(w);
                                int x = v;
                                while (x != w && x != -1)
                                {
                                    cyc.Add(x);
                                    x = parent[x];
                                }
                                return cyc;
                            }
                        }
                        else
                        {
                            color[v] = 2;
                            top--;
                        }
                    }
                }
                return null;
            }
        }

        private void Dfs()
        {
            int trailMark = trail.Count;
            int curMark = current.Count;

            ReducePeelAndForced();

            if (current.Count >= bestSize)
            {
                UndoTo(trailMark);
                current.RemoveRange(curMark, current.Count - curMark);
                return;
            }

            ComputeSccLowerBoundAndChoose(out int cyclicCount, out int chosenComp);

            if (cyclicCount == 0)
            {
                // acyclic
                if (current.Count < bestSize)
                {
                    bestSize = current.Count;
                    best = new List<int>(current);
                }
                UndoTo(trailMark);
                current.RemoveRange(curMark, current.Count - curMark);
                return;
            }

            // admissible lower bound: at least 1 removal per cyclic SCC
            if (current.Count + cyclicCount >= bestSize)
            {
                UndoTo(trailMark);
                current.RemoveRange(curMark, current.Count - curMark);
                return;
            }

            // find a cycle inside chosenComp
            var cycle = FindCycleInComponent(chosenComp);
            if (cycle.Count == 0)
            {
                // should not happen, but treat as acyclic
                if (current.Count < bestSize)
                {
                    bestSize = current.Count;
                    best = new List<int>(current);
                }
                UndoTo(trailMark);
                current.RemoveRange(curMark, current.Count - curMark);
                return;
            }

            // Branch order: higher degree first
            cycle.Sort((a, b) => (inDeg[b] + outDeg[b]).CompareTo(inDeg[a] + outDeg[a]));

            // Branch on each vertex in cycle
            for (int i = 0; i < cycle.Count; i++)
            {
                int v = cycle[i];
                if (!active[v]) continue;

                int tMark = trail.Count;
                int cMark = current.Count;

                DeactivateInternal(v, queue: null, ref UnsafeDummyRef.Int); // actual removal
                current.Add(v);

                Dfs();

                UndoTo(tMark);
                current.RemoveRange(cMark, current.Count - cMark);

                if (current.Count + 1 >= bestSize) break; // mild pruning for this loop
            }

            UndoTo(trailMark);
            current.RemoveRange(curMark, current.Count - curMark);
        }

        private void ReducePeelAndForced()
        {
            // forced self-loops (must be in any DFVS)
            // do it first and after peeling because peeling can expose nothing but forced removals don't depend on degrees
            bool changed;
            do
            {
                changed = false;

                ReducePeel();

                for (int v = 0; v < n; v++)
                {
                    if (active[v] && hasSelfLoop[v])
                    {
                        DeactivateInternal(v, queue: null, ref UnsafeDummyRef.Int);
                        current.Add(v);
                        changed = true;
                    }
                }
            } while (changed);
        }

        private void ReducePeel()
        {
            int[] q = stackV; // reuse stackV as queue buffer
            int qh = 0, qt = 0;

            for (int v = 0; v < n; v++)
                if (active[v] && (inDeg[v] == 0 || outDeg[v] == 0))
                    q[qt++] = v;

            while (qh < qt)
            {
                int v = q[qh++];
                if (!active[v]) continue;
                if (inDeg[v] != 0 && outDeg[v] != 0) continue;

                DeactivateInternal(v, q, ref qt); // peel: not counted as removal
            }
        }

        private void DeactivateInternal(int v, int[]? queue, ref int qt)
        {
            // active change
            trail.Add(new TrailEntry { Kind = 0, V = v, OldActive = active[v], OldActiveCount = activeCount });
            active[v] = false;
            activeCount--;

            // update neighbors
            for (int ei = outStart[v]; ei < outStart[v + 1]; ei++)
            {
                int w = outAdj[ei];
                if (!active[w]) continue;
                trail.Add(new TrailEntry { Kind = 1, V = w, OldIn = inDeg[w], OldOut = outDeg[w] });
                inDeg[w]--;
                if (queue != null && (inDeg[w] == 0 || outDeg[w] == 0)) queue[qt++] = w;
            }
            for (int ei = inStart[v]; ei < inStart[v + 1]; ei++)
            {
                int w = inAdj[ei];
                if (!active[w]) continue;
                trail.Add(new TrailEntry { Kind = 1, V = w, OldIn = inDeg[w], OldOut = outDeg[w] });
                outDeg[w]--;
                if (queue != null && (inDeg[w] == 0 || outDeg[w] == 0)) queue[qt++] = w;
            }
        }

        private void UndoTo(int mark)
        {
            for (int i = trail.Count - 1; i >= mark; i--)
            {
                var e = trail[i];
                if (e.Kind == 0)
                {
                    active[e.V] = e.OldActive;
                    activeCount = e.OldActiveCount;
                }
                else
                {
                    inDeg[e.V] = e.OldIn;
                    outDeg[e.V] = e.OldOut;
                }
            }
            if (trail.Count > mark) trail.RemoveRange(mark, trail.Count - mark);
        }

        private void ComputeSccLowerBoundAndChoose(out int cyclicCount, out int chosenComp)
        {
            // Kosaraju on active vertices
            Array.Fill(visited, false);
            int ord = 0;

            // 1st pass: order by finish time
            for (int s = 0; s < n; s++)
            {
                if (!active[s] || visited[s]) continue;

                int top = 0;
                stackV[top] = s;
                stackI[top] = outStart[s];
                visited[s] = true;
                top++;

                while (top > 0)
                {
                    int v = stackV[top - 1];
                    int it = stackI[top - 1];

                    if (it < outStart[v + 1])
                    {
                        int w = outAdj[it];
                        stackI[top - 1] = it + 1;
                        if (!active[w] || visited[w]) continue;
                        visited[w] = true;
                        stackV[top] = w;
                        stackI[top] = outStart[w];
                        top++;
                    }
                    else
                    {
                        top--;
                        order[ord++] = v;
                    }
                }
            }

            // 2nd pass: reverse graph assign components
            for (int i = 0; i < n; i++) compId[i] = -1;

            int compCnt = 0;
            for (int idx = ord - 1; idx >= 0; idx--)
            {
                int s = order[idx];
                if (!active[s] || compId[s] != -1) continue;

                compSize[compCnt] = 0;
                compHasCycle[compCnt] = false;

                int top = 0;
                stackV[top++] = s;
                compId[s] = compCnt;

                while (top > 0)
                {
                    int v = stackV[--top];
                    compSize[compCnt]++;

                    for (int ei = inStart[v]; ei < inStart[v + 1]; ei++)
                    {
                        int w = inAdj[ei];
                        if (!active[w] || compId[w] != -1) continue;
                        compId[w] = compCnt;
                        stackV[top++] = w;
                    }
                }

                if (compSize[compCnt] > 1) compHasCycle[compCnt] = true;
                compCnt++;
            }

            // detect self-loop SCCs (size==1) as cyclic
            for (int v = 0; v < n; v++)
            {
                if (!active[v]) continue;
                int c = compId[v];
                if (c < 0) continue;
                if (compSize[c] == 1 && !compHasCycle[c] && hasSelfLoop[v])
                    compHasCycle[c] = true;
            }

            cyclicCount = 0;
            chosenComp = -1;
            int bestSz = -1;
            for (int c = 0; c < compCnt; c++)
            {
                if (!compHasCycle[c]) continue;
                cyclicCount++;
                if (compSize[c] > bestSz)
                {
                    bestSz = compSize[c];
                    chosenComp = c;
                }
            }
        }

        private List<int> FindCycleInComponent(int comp)
        {
            Array.Fill(color, 0);
            Array.Fill(parent, -1);

            for (int s = 0; s < n; s++)
            {
                if (!active[s] || compId[s] != comp || color[s] != 0) continue;

                int top = 0;
                stackV[top] = s;
                stackI[top] = outStart[s];
                top++;
                parent[s] = -1;

                while (top > 0)
                {
                    int v = stackV[top - 1];
                    if (color[v] == 0) color[v] = 1;

                    int it = stackI[top - 1];
                    if (it < outStart[v + 1])
                    {
                        int w = outAdj[it];
                        stackI[top - 1] = it + 1;
                        if (!active[w] || compId[w] != comp) continue;

                        if (color[w] == 0)
                        {
                            parent[w] = v;
                            stackV[top] = w;
                            stackI[top] = outStart[w];
                            top++;
                        }
                        else if (color[w] == 1)
                        {
                            // found back edge v->w, reconstruct cycle vertices
                            var cyc = new List<int>();
                            cyc.Add(w);
                            int x = v;
                            while (x != w && x != -1)
                            {
                                cyc.Add(x);
                                x = parent[x];
                            }
                            return cyc;
                        }
                    }
                    else
                    {
                        color[v] = 2;
                        top--;
                    }
                }
            }

            return new List<int>();
        }

        private static class UnsafeDummyRef
        {
            // Used only to satisfy ref int parameter when no queue is used.
            public static int Int;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();

        int[] U = new int[m];
        int[] V = new int[m];
        int[] outCnt = new int[n];
        int[] inCnt = new int[n];

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            U[i] = u;
            V[i] = v;
            outCnt[u]++;
            inCnt[v]++;
        }

        // Build CSR adjacency for global graph
        int[] outStart = new int[n + 1];
        int[] inStart = new int[n + 1];
        for (int i = 0; i < n; i++)
        {
            outStart[i + 1] = outStart[i] + outCnt[i];
            inStart[i + 1] = inStart[i] + inCnt[i];
        }

        int[] outAdj = new int[m];
        int[] inAdj = new int[m];
        int[] outPos = (int[])outStart.Clone();
        int[] inPos = (int[])inStart.Clone();

        for (int i = 0; i < m; i++)
        {
            int u = U[i], v = V[i];
            outAdj[outPos[u]++] = v;
            inAdj[inPos[v]++] = u;
        }

        // Kosaraju SCC on global graph (iterative)
        bool[] visited = new bool[n];
        int[] order = new int[n];
        int ord = 0;
        int[] stackV = new int[n];
        int[] stackI = new int[n];

        for (int s = 0; s < n; s++)
        {
            if (visited[s]) continue;
            int top = 0;
            stackV[top] = s;
            stackI[top] = outStart[s];
            visited[s] = true;
            top++;

            while (top > 0)
            {
                int v = stackV[top - 1];
                int it = stackI[top - 1];
                if (it < outStart[v + 1])
                {
                    int w = outAdj[it];
                    stackI[top - 1] = it + 1;
                    if (visited[w]) continue;
                    visited[w] = true;
                    stackV[top] = w;
                    stackI[top] = outStart[w];
                    top++;
                }
                else
                {
                    top--;
                    order[ord++] = v;
                }
            }
        }

        int[] compId = new int[n];
        Array.Fill(compId, -1);
        int compCnt = 0;
        int[] compSize = new int[n];

        for (int idx = ord - 1; idx >= 0; idx--)
        {
            int s = order[idx];
            if (compId[s] != -1) continue;

            int top = 0;
            stackV[top++] = s;
            compId[s] = compCnt;
            int sz = 0;

            while (top > 0)
            {
                int v = stackV[--top];
                sz++;
                for (int ei = inStart[v]; ei < inStart[v + 1]; ei++)
                {
                    int w = inAdj[ei];
                    if (compId[w] != -1) continue;
                    compId[w] = compCnt;
                    stackV[top++] = w;
                }
            }

            compSize[compCnt] = sz;
            compCnt++;
        }

        // Build components arrays
        int[] compFill = new int[compCnt];
        for (int c = 0; c < compCnt; c++) compFill[c] = 0;

        int[][] comps = new int[compCnt][];
        for (int c = 0; c < compCnt; c++) comps[c] = new int[compSize[c]];

        for (int v = 0; v < n; v++)
        {
            int c = compId[v];
            comps[c][compFill[c]++] = v;
        }

        // Map arrays for per-SCC building
        int[] seen = new int[n];
        int[] toLocal = new int[n];
        int token = 1;

        var answer = new List<int>();

        for (int c = 0; c < compCnt; c++)
        {
            int[] verts = comps[c];
            if (verts.Length == 1)
            {
                int v = verts[0];
                bool selfLoop = false;
                for (int ei = outStart[v]; ei < outStart[v + 1]; ei++)
                {
                    if (outAdj[ei] == v) { selfLoop = true; break; }
                }
                if (selfLoop) answer.Add(v);
                continue;
            }

            // Prepare local mapping
            int sccN = verts.Length;
            int myTok = token++;
            for (int i = 0; i < sccN; i++)
            {
                int gv = verts[i];
                seen[gv] = myTok;
                toLocal[gv] = i;
            }

            // Count internal edges and self-loops
            int[] locOutCnt = new int[sccN];
            int[] locInCnt = new int[sccN];
            bool[] hasSelfLoop = new bool[sccN];
            int internalEdges = 0;

            for (int li = 0; li < sccN; li++)
            {
                int u = verts[li];
                for (int ei = outStart[u]; ei < outStart[u + 1]; ei++)
                {
                    int v = outAdj[ei];
                    if (seen[v] != myTok) continue;
                    int lv = toLocal[v];
                    locOutCnt[li]++;
                    locInCnt[lv]++;
                    internalEdges++;
                    if (li == lv) hasSelfLoop[li] = true;
                }
            }

            // Build local CSR
            int[] lOutStart = new int[sccN + 1];
            int[] lInStart = new int[sccN + 1];
            for (int i = 0; i < sccN; i++)
            {
                lOutStart[i + 1] = lOutStart[i] + locOutCnt[i];
                lInStart[i + 1] = lInStart[i] + locInCnt[i];
            }

            int[] lOutAdj = new int[internalEdges];
            int[] lInAdj = new int[internalEdges];
            int[] lOutPos = (int[])lOutStart.Clone();
            int[] lInPos = (int[])lInStart.Clone();

            for (int lu = 0; lu < sccN; lu++)
            {
                int gu = verts[lu];
                for (int ei = outStart[gu]; ei < outStart[gu + 1]; ei++)
                {
                    int gv = outAdj[ei];
                    if (seen[gv] != myTok) continue;
                    int lv = toLocal[gv];
                    lOutAdj[lOutPos[lu]++] = lv;
                    lInAdj[lInPos[lv]++] = lu;
                }
            }

            // Solve exactly on this SCC
            var solver = new DfvsSolver(sccN, lOutStart, lOutAdj, lInStart, lInAdj, hasSelfLoop);
            List<int> removedLocal = solver.Solve();

            for (int i = 0; i < removedLocal.Count; i++)
                answer.Add(verts[removedLocal[i]]);
        }

        // Output
        using var sw = new StreamWriter(Console.OpenStandardOutput());
        sw.WriteLine(answer.Count);
        if (answer.Count == 0)
        {
            sw.WriteLine();
        }
        else
        {
            for (int i = 0; i < answer.Count; i++)
            {
                if (i > 0) sw.Write(' ');
                sw.Write(answer[i]);
            }
            sw.WriteLine();
        }
    }
