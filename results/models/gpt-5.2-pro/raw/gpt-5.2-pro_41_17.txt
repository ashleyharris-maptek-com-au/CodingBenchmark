; SPIR-V
; Version: 1.0
; Bound: 200
; Schema: 0
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%int = OpTypeInt 32 1

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

%ptrInputv3 = OpTypePointer Input %v3float
%ptrInputv2 = OpTypePointer Input %v2float
%ptrOutputv4 = OpTypePointer Output %v4float
%ptrUniformUBO = OpTypePointer Uniform %UBO
%ptrUniformv4 = OpTypePointer Uniform %v4float

%inWorldPos = OpVariable %ptrInputv3 Input
%inNormal = OpVariable %ptrInputv3 Input
%inUV = OpVariable %ptrInputv2 Input
%outColor = OpVariable %ptrOutputv4 Output
%uboVar = OpVariable %ptrUniformUBO Uniform

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_05 = OpConstant %float 0.05
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_0_6 = OpConstant %float 0.6
%float_0_7 = OpConstant %float 0.7
%float_0_2 = OpConstant %float 0.2
%float_0_9 = OpConstant %float 0.9
%float_0_95 = OpConstant %float 0.95
%float_8 = OpConstant %float 8
%float_16 = OpConstant %float 16

%int_3 = OpConstant %int 3

%scale = OpConstantComposite %v2float %float_8 %float_16
%mortarColor = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_0_7
%brickColor = OpConstantComposite %v3float %float_0_7 %float_0_2 %float_0_1

%fn = OpTypeFunction %void

%main = OpFunction %void None %fn
%label = OpLabel

%wp = OpLoad %v3float %inWorldPos
%n_in = OpLoad %v3float %inNormal
%uv = OpLoad %v2float %inUV

%n = OpExtInst %v3float %glsl Normalize %n_in

%ptrLightPos = OpAccessChain %ptrUniformv4 %uboVar %int_3
%lightPos4 = OpLoad %v4float %ptrLightPos
%lightPos3 = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2

%ldir_raw = OpFSub %v3float %lightPos3 %wp
%ldir = OpExtInst %v3float %glsl Normalize %ldir_raw

%ndotl = OpDot %float %n %ldir
%diffuse = OpExtInst %float %glsl FMax %ndotl %float_0
%shadeMul = OpFMul %float %float_0_6 %diffuse
%shade = OpFAdd %float %float_0_4 %shadeMul

%uvScaled = OpFMul %v2float %uv %scale
%uvScaled_x = OpCompositeExtract %float %uvScaled 0
%uvScaled_y = OpCompositeExtract %float %uvScaled 1

%row = OpExtInst %float %glsl Floor %uvScaled_y
%halfRow = OpFMul %float %row %float_0_5
%rowFr = OpExtInst %float %glsl Fract %halfRow
%xShift = OpFAdd %float %uvScaled_x %rowFr

%shifted = OpCompositeConstruct %v2float %xShift %uvScaled_y
%cell = OpExtInst %v2float %glsl Fract %shifted
%fx = OpCompositeExtract %float %cell 0
%fy = OpCompositeExtract %float %cell 1

%fx_lt = OpFOrdLessThan %bool %fx %float_0_05
%fx_gt = OpFOrdGreaterThan %bool %fx %float_0_95
%condX = OpLogicalOr %bool %fx_lt %fx_gt

%fy_lt = OpFOrdLessThan %bool %fy %float_0_1
%fy_gt = OpFOrdGreaterThan %bool %fy %float_0_9
%condY = OpLogicalOr %bool %fy_lt %fy_gt

%mortar = OpLogicalOr %bool %condX %condY
%base = OpSelect %v3float %mortar %mortarColor %brickColor

%shadeVec = OpCompositeConstruct %v3float %shade %shade %shade
%rgb = OpFMul %v3float %base %shadeVec

%r = OpCompositeExtract %float %rgb 0
%g = OpCompositeExtract %float %rgb 1
%b = OpCompositeExtract %float %rgb 2
%out = OpCompositeConstruct %v4float %r %g %b %float_1
OpStore %outColor %out

OpReturn
OpFunctionEnd
