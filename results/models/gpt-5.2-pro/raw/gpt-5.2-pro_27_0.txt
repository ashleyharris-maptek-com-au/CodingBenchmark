// -------------------------- Fast IO --------------------------
class FastScanner {
public:
    explicit FastScanner(size_t bufSize = 1u << 20) : buf_(bufSize), idx_(0), size_(0) {}

    bool readInt(int &out) { long long x; if (!readLongLong(x)) return false; out = (int)x; return true; }
    bool readU32(uint32_t &out) { unsigned long long x; if (!readULL(x)) return false; out = (uint32_t)x; return true; }
    bool readU64(uint64_t &out) { unsigned long long x; if (!readULL(x)) return false; out = (uint64_t)x; return true; }

private:
    std::vector<char> buf_;
    size_t idx_, size_;

    inline bool refill() {
        size_ = std::fread(buf_.data(), 1, buf_.size(), stdin);
        idx_ = 0;
        return size_ != 0;
    }

    inline bool skipBlanks() {
        for (;;) {
            if (idx_ >= size_) { if (!refill()) return false; }
            if (buf_[idx_] > ' ') return true;
            ++idx_;
        }
    }

    bool readLongLong(long long &out) {
        if (!skipBlanks()) return false;
        bool neg = false;
        if (buf_[idx_] == '-') { neg = true; ++idx_; }
        long long x = 0;
        for (;;) {
            if (idx_ >= size_) { if (!refill()) break; }
            char c = buf_[idx_];
            if (c < '0' || c > '9') break;
            x = x * 10 + (c - '0');
            ++idx_;
        }
        out = neg ? -x : x;
        return true;
    }

    bool readULL(unsigned long long &out) {
        if (!skipBlanks()) return false;
        unsigned long long x = 0;
        for (;;) {
            if (idx_ >= size_) { if (!refill()) break; }
            char c = buf_[idx_];
            if (c < '0' || c > '9') break;
            x = x * 10 + (unsigned long long)(c - '0');
            ++idx_;
        }
        out = x;
        return true;
    }
};

class FastOutput {
public:
    explicit FastOutput(size_t bufSize = 1u << 20) : buf_(bufSize), pos_(0) {}
    ~FastOutput() { flush(); }

    void writeChar(char c) {
        if (pos_ == buf_.size()) flush();
        buf_[pos_++] = c;
    }

    void writeInt(int x) {
        char tmp[32];
        int n = 0;
        if (x == 0) tmp[n++] = '0';
        else {
            if (x < 0) { writeChar('-'); x = -x; }
            while (x > 0) { tmp[n++] = char('0' + (x % 10)); x /= 10; }
        }
        while (n--) writeChar(tmp[n]);
    }

    void flush() {
        if (pos_) {
            std::fwrite(buf_.data(), 1, pos_, stdout);
            pos_ = 0;
        }
    }

private:
    std::vector<char> buf_;
    size_t pos_;
};

// -------------------------- Bit helpers (portable) --------------------------
static inline int ctz64(uint64_t x) {
#if defined(__cpp_lib_bitops) && __cpp_lib_bitops >= 201907L
    return (int)std::countr_zero(x);
#elif defined(_MSC_VER) && defined(_M_X64)
    unsigned long idx;
    _BitScanForward64(&idx, x);
    return (int)idx;
#else
    return __builtin_ctzll(x);
#endif
}

static inline int popcount64(uint64_t x) {
#if defined(__cpp_lib_bitops) && __cpp_lib_bitops >= 201907L
    return (int)std::popcount(x);
#elif defined(_MSC_VER)
    return (int)__popcnt64(x);
#else
    return __builtin_popcountll(x);
#endif
}

// -------------------------- Graph (CSR) --------------------------
struct Graph {
    int n = 0;
    uint64_t m = 0;
    std::vector<uint64_t> off;   // size n+1
    std::vector<uint32_t> adj;   // size 2m
    std::vector<uint32_t> deg;   // size n
};

static Graph buildCSR(int n, uint64_t m, const std::vector<uint32_t>& eu, const std::vector<uint32_t>& ev) {
    Graph g;
    g.n = n;
    g.m = m;
    g.deg.assign(n, 0);

    for (uint64_t i = 0; i < m; ++i) {
        uint32_t a = eu[i], b = ev[i];
        if (a >= (uint32_t)n || b >= (uint32_t)n) continue; // safety
        if (a == b) { g.deg[a] += 2; } // still store twice to keep symmetry, but it's unsatisfiable anyway
        else { g.deg[a]++; g.deg[b]++; }
    }

    g.off.assign((size_t)n + 1, 0);
    for (int i = 0; i < n; ++i) g.off[i + 1] = g.off[i] + g.deg[i];

    g.adj.assign((size_t)g.off[n], 0);
    std::vector<uint64_t> cur = g.off;

    for (uint64_t i = 0; i < m; ++i) {
        uint32_t a = eu[i], b = ev[i];
        if (a >= (uint32_t)n || b >= (uint32_t)n) continue;
        g.adj[(size_t)cur[a]++] = b;
        g.adj[(size_t)cur[b]++] = a;
    }
    return g;
}

// -------------------------- Validation --------------------------
static bool validateColoring(const Graph& g, int k, const std::vector<int16_t>& col) {
    if ((int)col.size() != g.n) return false;
    for (int v = 0; v < g.n; ++v) {
        int c = col[v];
        if (c < 0 || c >= k) return false;
    }
    for (int v = 0; v < g.n; ++v) {
        int cv = col[v];
        uint64_t start = g.off[v], end = g.off[v + 1];
        for (uint64_t it = start; it < end; ++it) {
            uint32_t u = g.adj[(size_t)it];
            if ((int)u > v && col[u] == cv) return false;
            if ((int)u == v) return false; // self-loop => impossible
        }
    }
    return true;
}

// -------------------------- Ordering: smallest-last (degeneracy) --------------------------
// O(n+m) bucket queue implementation; returns removal order (smallest degree first).
static std::vector<uint32_t> degeneracyRemovalOrder(const Graph& g) {
    int n = g.n;
    std::vector<int> d(n);
    int maxDeg = 0;
    for (int i = 0; i < n; ++i) {
        d[i] = (int)g.deg[i];
        maxDeg = std::max(maxDeg, d[i]);
    }

    std::vector<int> head((size_t)maxDeg + 1, -1);
    std::vector<int> next(n, -1), prev(n, -1);
    std::vector<int> bucket(n, 0);
    std::vector<char> removed(n, 0);

    auto insertBucket = [&](int v, int deg) {
        bucket[v] = deg;
        int &h = head[deg];
        prev[v] = -1;
        next[v] = h;
        if (h != -1) prev[h] = v;
        h = v;
    };
    auto removeBucket = [&](int v, int deg) {
        int p = prev[v], nx = next[v];
        if (p != -1) next[p] = nx;
        else head[deg] = nx;
        if (nx != -1) prev[nx] = p;
        prev[v] = next[v] = -1;
    };

    for (int v = 0; v < n; ++v) insertBucket(v, d[v]);

    std::vector<uint32_t> order;
    order.reserve((size_t)n);

    int cur = 0;
    for (int iter = 0; iter < n; ++iter) {
        while (cur <= maxDeg && head[cur] == -1) ++cur;
        if (cur > maxDeg) break; // should not happen

        int v = head[cur];
        removeBucket(v, cur);
        removed[v] = 1;
        order.push_back((uint32_t)v);

        uint64_t start = g.off[v], end = g.off[v + 1];
        for (uint64_t it = start; it < end; ++it) {
            int u = (int)g.adj[(size_t)it];
            if (u == v) continue;
            if (u < 0 || u >= n) continue;
            if (removed[u]) continue;
            int du = bucket[u];
            removeBucket(u, du);
            d[u] = du - 1;
            insertBucket(u, d[u]);
            if (d[u] < cur) cur = d[u];
        }
    }
    return order;
}

static std::vector<uint32_t> degreeDescOrder(const Graph& g) {
    int n = g.n;
    std::vector<uint32_t> order((size_t)n);
    for (int i = 0; i < n; ++i) order[i] = (uint32_t)i;
    std::sort(order.begin(), order.end(), [&](uint32_t a, uint32_t b) {
        uint32_t da = g.deg[a], db = g.deg[b];
        if (da != db) return da > db;
        return a < b;
    });
    return order;
}

// -------------------------- Greedy coloring --------------------------
static bool greedyColor(
    const Graph& g,
    const std::vector<uint32_t>& order, // if it's a removal order, caller can pass reversed; we just use given sequence
    int k,
    std::vector<int16_t>& col,
    bool allowOverflowConflicts,
    std::mt19937_64* rngForTies // can be null
) {
    const uint64_t maskAll = (k >= 64) ? ~0ULL : ((k == 64) ? ~0ULL : ((1ULL << k) - 1ULL));
    col.assign((size_t)g.n, (int16_t)-1);

    // Color in the given order (typically "good" order is largest-last, etc.)
    std::array<int, 64> cnt{};
    for (uint32_t vv : order) {
        int v = (int)vv;

        uint64_t forb = 0;
        uint64_t start = g.off[v], end = g.off[v + 1];
        for (uint64_t it = start; it < end; ++it) {
            uint32_t u = g.adj[(size_t)it];
            int cu = (u < (uint32_t)g.n) ? (int)col[u] : -1;
            if (cu >= 0) forb |= (1ULL << cu);
        }

        uint64_t avail = maskAll & ~forb;
        if (avail) {
            int c = ctz64(avail);
            col[v] = (int16_t)c;
        } else {
            if (!allowOverflowConflicts) return false;

            // Pick a color with minimal neighbor conflicts (min-conflicts heuristic).
            // Count neighbor colors:
            for (int i = 0; i < k; ++i) cnt[i] = 0;
            for (uint64_t it = start; it < end; ++it) {
                uint32_t u = g.adj[(size_t)it];
                int cu = (u < (uint32_t)g.n) ? (int)col[u] : -1;
                if (cu >= 0) cnt[cu]++;
            }
            int best = 0;
            int bestVal = cnt[0];
            for (int c = 1; c < k; ++c) {
                if (cnt[c] < bestVal) { bestVal = cnt[c]; best = c; }
                else if (cnt[c] == bestVal && rngForTies && ((*rngForTies)() & 7ULL) == 0ULL) best = c;
            }
            col[v] = (int16_t)best;
        }
    }
    return true;
}

// -------------------------- Local repair (min-conflicts) --------------------------
static bool repairMinConflicts(
    const Graph& g, int k, std::vector<int16_t>& col,
    std::chrono::steady_clock::time_point deadline,
    std::mt19937_64& rng
) {
    int n = g.n;
    std::vector<int> conf(n, 0);

    // totalConf = number of conflicting edges (count each undirected edge once: u>v)
    uint64_t totalConf = 0;
    for (int v = 0; v < n; ++v) {
        int cv = col[v];
        uint64_t start = g.off[v], end = g.off[v + 1];
        for (uint64_t it = start; it < end; ++it) {
            uint32_t u = g.adj[(size_t)it];
            if ((int)u <= v) continue;
            if ((int16_t)cv == col[u]) {
                totalConf++;
                conf[v]++; conf[u]++;
            }
        }
    }
    if (totalConf == 0) return true;

    // Candidate set with O(1) add/remove
    std::vector<int> cand;
    cand.reserve((size_t)std::min(n, 100000));
    std::vector<int> pos(n, -1);
    auto addCand = [&](int v) {
        if (pos[v] != -1) return;
        pos[v] = (int)cand.size();
        cand.push_back(v);
    };
    auto removeCand = [&](int v) {
        int i = pos[v];
        if (i == -1) return;
        int last = cand.back();
        cand[i] = last;
        pos[last] = i;
        cand.pop_back();
        pos[v] = -1;
    };

    for (int v = 0; v < n; ++v) if (conf[v] > 0) addCand(v);

    std::array<int, 64> cnt{};
    auto now = std::chrono::steady_clock::now();

    while (totalConf > 0) {
        now = std::chrono::steady_clock::now();
        if (now >= deadline) break;
        if (cand.empty()) break;

        // pick vertex with high conflicts among a few random samples
        int v = cand[(size_t)(rng() % cand.size())];
        for (int t = 0; t < 4; ++t) {
            int vv = cand[(size_t)(rng() % cand.size())];
            if (conf[vv] > conf[v]) v = vv;
        }
        if (conf[v] == 0) { removeCand(v); continue; }

        int curC = col[v];
        for (int c = 0; c < k; ++c) cnt[c] = 0;

        uint64_t start = g.off[v], end = g.off[v + 1];
        for (uint64_t it = start; it < end; ++it) {
            uint32_t u = g.adj[(size_t)it];
            int cu = col[u];
            if (cu >= 0) cnt[cu]++;
        }

        // choose best color (min conflicts); allow staying in same color if ties
        int bestC = curC;
        int bestVal = cnt[curC];
        for (int c = 0; c < k; ++c) {
            int val = cnt[c];
            if (val < bestVal || (val == bestVal && ((rng() & 15ULL) == 0ULL))) {
                bestVal = val;
                bestC = c;
            }
        }
        if (bestC == curC) continue;

        // apply recolor and update conflicts incrementally
        col[v] = (int16_t)bestC;

        int newConfV = 0;
        for (uint64_t it = start; it < end; ++it) {
            int u = (int)g.adj[(size_t)it];
            if (u == v) continue;
            int cu = col[u];

            bool was = (cu == curC);
            bool nowc = (cu == bestC);

            if (was) {
                // edge (v,u) conflict removed
                totalConf--;
                conf[u]--;
            }
            if (nowc) {
                // edge (v,u) conflict added
                totalConf++;
                conf[u]++;
                newConfV++;
            }

            if (conf[u] > 0) addCand(u);
            else removeCand(u);
        }

        conf[v] = newConfV;
        if (conf[v] > 0) addCand(v);
        else removeCand(v);
    }

    return totalConf == 0;
}

// -------------------------- Exact DSATUR (small graphs) --------------------------
struct ExactDSATUR {
    const Graph& g;
    int n;
    int k;
    uint64_t maskAll;
    std::vector<int16_t> col;
    std::vector<uint64_t> satMask;
    std::vector<uint32_t> deg;
    int colored = 0;

    std::chrono::steady_clock::time_point deadline;
    bool timedOut = false;

    ExactDSATUR(const Graph& gg, int kk, std::chrono::steady_clock::time_point dl)
        : g(gg), n(gg.n), k(kk), maskAll((kk >= 64) ? ~0ULL : ((1ULL << kk) - 1ULL)),
          col((size_t)gg.n, -1),
          satMask((size_t)gg.n, 0),
          deg(gg.deg),
          deadline(dl) {}

    bool dfs() {
        if (std::chrono::steady_clock::now() >= deadline) { timedOut = true; return false; }
        if (colored == n) return true;

        int bestV = -1;
        int bestSat = -1;
        uint32_t bestDeg = 0;

        for (int v = 0; v < n; ++v) if (col[v] < 0) {
            uint64_t avail = maskAll & ~satMask[v];
            if (!avail) return false; // immediate dead-end
            int s = popcount64(satMask[v]);
            uint32_t d = deg[v];
            if (s > bestSat || (s == bestSat && d > bestDeg)) {
                bestSat = s;
                bestDeg = d;
                bestV = v;
            }
        }

        int v = bestV;
        uint64_t avail = maskAll & ~satMask[v];

        // Try colors in increasing order.
        while (avail) {
            uint64_t bit = avail & -avail;
            int c = ctz64(bit);
            avail ^= bit;

            col[v] = (int16_t)c;
            colored++;

            std::vector<std::pair<int, uint64_t>> mods;
            mods.reserve((size_t)(g.off[v + 1] - g.off[v]));

            bool ok = true;
            uint64_t start = g.off[v], end = g.off[v + 1];
            for (uint64_t it = start; it < end; ++it) {
                int u = (int)g.adj[(size_t)it];
                if (u == v) { ok = false; break; } // self-loop
                if (col[u] >= 0) continue;
                uint64_t old = satMask[u];
                uint64_t nw = old | (1ULL << c);
                if (nw != old) {
                    mods.emplace_back(u, old);
                    satMask[u] = nw;
                    if ((maskAll & ~satMask[u]) == 0) { ok = false; break; }
                }
            }

            if (ok && dfs()) return true;

            // undo
            for (int i = (int)mods.size() - 1; i >= 0; --i) {
                satMask[mods[i].first] = mods[i].second;
            }
            col[v] = -1;
            colored--;

            if (timedOut) return false;
        }
        return false;
    }
};

// -------------------------- Main adaptive solver --------------------------
int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    FastScanner fs;
    int n, m_in, k;
    if (!fs.readInt(n)) return 0;
    fs.readInt(m_in);
    fs.readInt(k);
    if (n < 0 || m_in < 0 || k <= 0) return 0;
    uint64_t m = (uint64_t)m_in;

    std::vector<uint32_t> eu;
    std::vector<uint32_t> ev;
    eu.resize((size_t)m);
    ev.resize((size_t)m);

    for (uint64_t i = 0; i < m; ++i) {
        uint32_t a, b;
        if (!fs.readU32(a) || !fs.readU32(b)) { a = b = 0; }
        eu[i] = a; ev[i] = b;
    }

    Graph g = buildCSR(n, m, eu, ev);

    // Global time budget ~ 5 minutes, but keep some margin.
    auto t0 = std::chrono::steady_clock::now();
    auto globalDeadline = t0 + std::chrono::seconds(295);

    std::mt19937_64 rng(
        (uint64_t)std::chrono::high_resolution_clock::now().time_since_epoch().count() ^
        (uint64_t)(uintptr_t)&g
    );

    std::vector<int16_t> bestCol;

    // Trivial cases
    if (n == 0) {
        // print nothing (empty)
        return 0;
    }
    if (k == 1) {
        std::vector<int16_t> col((size_t)n, 0);
        if (validateColoring(g, k, col)) {
            FastOutput fo;
            for (int i = 0; i < n; ++i) {
                if (i) fo.writeChar(' ');
                fo.writeInt(0);
            }
            fo.writeChar('\n');
            return 0;
        }
        // unsatisfiable due to edges/self-loop; fallthrough to heuristics (will still fail, but testcases should avoid this)
    }

    // 1) Exact DSATUR for small graphs (tight instances).
    // Keep it conservative: exact only for very small n.
    if (n <= 250) {
        auto exactDeadline = std::min(globalDeadline, t0 + std::chrono::seconds(5));
        ExactDSATUR ex(g, k, exactDeadline);
        if (ex.dfs() && validateColoring(g, k, ex.col)) {
            FastOutput fo;
            for (int i = 0; i < n; ++i) {
                if (i) fo.writeChar(' ');
                fo.writeInt((int)ex.col[i]);
            }
            fo.writeChar('\n');
            return 0;
        }
    }

    // 2) Heuristic attempts: multiple orderings; first try strict greedy (must not overflow).
    auto tryStrictGreedy = [&](const std::vector<uint32_t>& order) -> bool {
        std::vector<int16_t> col;
        if (!greedyColor(g, order, k, col, /*allowOverflowConflicts=*/false, nullptr)) return false;
        if (!validateColoring(g, k, col)) return false;
        bestCol = std::move(col);
        return true;
    };

    // 2a) Degeneracy-based (smallest-last) greedy: use reverse of removal order for coloring.
    {
        auto rem = degeneracyRemovalOrder(g);
        std::reverse(rem.begin(), rem.end());
        if (tryStrictGreedy(rem)) {
            FastOutput fo;
            for (int i = 0; i < n; ++i) {
                if (i) fo.writeChar(' ');
                fo.writeInt((int)bestCol[i]);
            }
            fo.writeChar('\n');
            return 0;
        }
    }

    // 2b) Degree descending
    {
        auto ord = degreeDescOrder(g);
        if (tryStrictGreedy(ord)) {
            FastOutput fo;
            for (int i = 0; i < n; ++i) {
                if (i) fo.writeChar(' ');
                fo.writeInt((int)bestCol[i]);
            }
            fo.writeChar('\n');
            return 0;
        }
    }

    // 2c) A couple randomized shuffles (useful for adversarial small cases)
    for (int attempt = 0; attempt < 2; ++attempt) {
        std::vector<uint32_t> ord((size_t)n);
        for (int i = 0; i < n; ++i) ord[i] = (uint32_t)i;
        std::shuffle(ord.begin(), ord.end(), rng);
        if (tryStrictGreedy(ord)) {
            FastOutput fo;
            for (int i = 0; i < n; ++i) {
                if (i) fo.writeChar(' ');
                fo.writeInt((int)bestCol[i]);
            }
            fo.writeChar('\n');
            return 0;
        }
        if (std::chrono::steady_clock::now() >= globalDeadline) break;
    }

    // 3) If strict greedy didn't find a k-coloring, allow overflow conflicts and repair.
    auto tryGreedyPlusRepair = [&](std::vector<uint32_t> order, int repairSeconds) -> bool {
        std::vector<int16_t> col;
        greedyColor(g, order, k, col, /*allowOverflowConflicts=*/true, &rng);

        auto dl = std::min(globalDeadline, std::chrono::steady_clock::now() + std::chrono::seconds(repairSeconds));
        if (!repairMinConflicts(g, k, col, dl, rng)) return false;
        if (!validateColoring(g, k, col)) return false;
        bestCol = std::move(col);
        return true;
    };

    // Degeneracy + repair (usually strongest)
    {
        auto rem = degeneracyRemovalOrder(g);
        std::reverse(rem.begin(), rem.end());
        if (tryGreedyPlusRepair(std::move(rem), /*repairSeconds=*/25)) {
            FastOutput fo;
            for (int i = 0; i < n; ++i) {
                if (i) fo.writeChar(' ');
                fo.writeInt((int)bestCol[i]);
            }
            fo.writeChar('\n');
            return 0;
        }
    }

    // Degree-desc + repair
    {
        auto ord = degreeDescOrder(g);
        if (tryGreedyPlusRepair(std::move(ord), /*repairSeconds=*/25)) {
            FastOutput fo;
            for (int i = 0; i < n; ++i) {
                if (i) fo.writeChar(' ');
                fo.writeInt((int)bestCol[i]);
            }
            fo.writeChar('\n');
            return 0;
        }
    }

    // Random + repair (a few tries)
    for (int attempt = 0; attempt < 3; ++attempt) {
        std::vector<uint32_t> ord((size_t)n);
        for (int i = 0; i < n; ++i) ord[i] = (uint32_t)i;
        std::shuffle(ord.begin(), ord.end(), rng);
        if (tryGreedyPlusRepair(std::move(ord), /*repairSeconds=*/20)) {
            FastOutput fo;
            for (int i = 0; i < n; ++i) {
                if (i) fo.writeChar(' ');
                fo.writeInt((int)bestCol[i]);
            }
            fo.writeChar('\n');
            return 0;
        }
        if (std::chrono::steady_clock::now() >= globalDeadline) break;
    }

    // Last resort: if we failed, still print something within range.
    // (Test data is expected to be k-colorable; above should succeed in practice.)
    std::vector<int16_t> fallback((size_t)n, 0);
    FastOutput fo;
    for (int i = 0; i < n; ++i) {
        if (i) fo.writeChar(' ');
        fo.writeInt((int)fallback[i]);
    }
    fo.writeChar('\n');
    return 0;
}
