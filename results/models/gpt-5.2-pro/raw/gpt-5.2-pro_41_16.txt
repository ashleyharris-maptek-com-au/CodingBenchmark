; SPIR-V
; Version: 1.0
; Bound: 100
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_uv %out_outColor
OpExecutionMode %main OriginUpperLeft

OpName %main "main"
OpName %in_uv "uv"
OpName %out_outColor "outColor"

OpDecorate %in_uv Location 2
OpDecorate %out_outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Function_v2float = OpTypePointer Function %v2float
%ptr_Function_int = OpTypePointer Function %int

%func = OpTypeFunction %void

%float_0 = OpConstant %float 0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%float_3 = OpConstant %float 3
%float_4 = OpConstant %float 4
%float_20 = OpConstant %float 20

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_20 = OpConstant %int 20

%v2float_0_0 = OpConstantComposite %v2float %float_0 %float_0

%in_uv = OpVariable %ptr_Input_v2float Input
%out_outColor = OpVariable %ptr_Output_v4float Output

%main = OpFunction %void None %func
%entry = OpLabel

%z = OpVariable %ptr_Function_v2float Function %v2float_0_0
%i = OpVariable %ptr_Function_int Function %int_0

%uv_val = OpLoad %v2float %in_uv
%uv_x = OpCompositeExtract %float %uv_val 0
%uv_y = OpCompositeExtract %float %uv_val 1

%cx_mul = OpFMul %float %uv_x %float_3
%cx = OpFSub %float %cx_mul %float_2
%cy_mul = OpFMul %float %uv_y %float_2
%cy = OpFSub %float %cy_mul %float_1

OpBranch %loop_header

%loop_header = OpLabel
OpLoopMerge %loop_merge %loop_continue None

%i_val = OpLoad %int %i
%cond_i = OpSLessThan %bool %i_val %int_20

%z_val = OpLoad %v2float %z
%zz = OpDot %float %z_val %z_val
%cond_mag = OpFOrdLessThanEqual %bool %zz %float_4

%cond = OpLogicalAnd %bool %cond_i %cond_mag
OpBranchConditional %cond %loop_body %loop_merge

%loop_body = OpLabel
%zx = OpCompositeExtract %float %z_val 0
%zy = OpCompositeExtract %float %z_val 1

%zx2 = OpFMul %float %zx %zx
%zy2 = OpFMul %float %zy %zy
%real = OpFSub %float %zx2 %zy2

%zxzy = OpFMul %float %zx %zy
%imag = OpFMul %float %zxzy %float_2

%newx = OpFAdd %float %real %cx
%newy = OpFAdd %float %imag %cy
%newz = OpCompositeConstruct %v2float %newx %newy
OpStore %z %newz

%i_next = OpIAdd %int %i_val %int_1
OpStore %i %i_next

OpBranch %loop_continue

%loop_continue = OpLabel
OpBranch %loop_header

%loop_merge = OpLabel
%i_final = OpLoad %int %i
%i_final_f = OpConvertSToF %float %i_final
%t = OpFDiv %float %i_final_f %float_20

%g = OpFMul %float %t %float_0_5
%b = OpFSub %float %float_1 %t
%out_col = OpCompositeConstruct %v4float %t %g %b %float_1
OpStore %out_outColor %out_col

OpReturn
OpFunctionEnd
