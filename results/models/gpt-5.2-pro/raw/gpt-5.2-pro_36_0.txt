#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <cstdint>
#include <bit>
#include <limits>

using namespace std;

struct FastBitset {
    int n = 0;
    int words = 0;
    uint64_t lastMask = ~0ULL;
    vector<uint64_t> a;

    FastBitset() = default;
    explicit FastBitset(int nBits, bool allZero = true) { init(nBits, allZero); }

    void init(int nBits, bool allZero = true) {
        n = nBits;
        words = (n + 63) / 64;
        a.assign(words, allZero ? 0ULL : ~0ULL);
        if (n % 64) lastMask = (1ULL << (n % 64)) - 1ULL;
        else lastMask = ~0ULL;
        if (!a.empty()) a.back() &= lastMask;
    }

    static FastBitset full(int nBits) {
        FastBitset b(nBits, false);
        return b;
    }

    inline void trimLast() {
        if (!a.empty()) a.back() &= lastMask;
    }

    inline void setBit(int i) {
        a[i >> 6] |= 1ULL << (i & 63);
    }
    inline void resetBit(int i) {
        a[i >> 6] &= ~(1ULL << (i & 63));
    }
    inline bool testBit(int i) const {
        return (a[i >> 6] >> (i & 63)) & 1ULL;
    }

    inline bool any() const {
        for (uint64_t w : a) if (w) return true;
        return false;
    }
    inline int popcount() const {
        int s = 0;
        for (uint64_t w : a) s += (int)std::popcount(w);
        return s;
    }

    // index of least significant set bit (assumes any()==true)
    inline int firstOne() const {
        for (int wi = 0; wi < words; ++wi) {
            uint64_t w = a[wi];
            if (w) {
                return (wi << 6) + (int)std::countr_zero(w);
            }
        }
        return -1;
    }

    inline void andWith(const FastBitset& o) {
        for (int i = 0; i < words; ++i) a[i] &= o.a[i];
    }
    inline void andNotWith(const FastBitset& o) {
        for (int i = 0; i < words; ++i) a[i] &= ~o.a[i];
        // no need to trim because we only clear bits
    }
};

struct MaxCliqueTomita {
    int n = 0;
    vector<FastBitset> adj; // adjacency in the graph where we search clique
    int bestSize = 0;
    vector<int> bestClique;

    explicit MaxCliqueTomita(vector<FastBitset> adjacency)
        : n((int)adjacency.size()), adj(std::move(adjacency)) {}

    void greedyInitialBound() {
        FastBitset P = FastBitset::full(n);
        vector<int> C;
        while (P.any()) {
            int bestV = -1;
            int bestDeg = -1;
            // choose v with maximum degree within P
            for (int v = 0; v < n; ++v) {
                if (!P.testBit(v)) continue;
                FastBitset tmp = P;
                tmp.andWith(adj[v]);
                int deg = tmp.popcount();
                if (deg > bestDeg) {
                    bestDeg = deg;
                    bestV = v;
                }
            }
            if (bestV == -1) break;
            C.push_back(bestV);
            P.andWith(adj[bestV]);
        }
        if ((int)C.size() > bestSize) {
            bestSize = (int)C.size();
            bestClique = C;
        }
    }

    void colorSort(const FastBitset& P, vector<int>& order, vector<int>& colors) {
        order.clear();
        colors.clear();
        FastBitset R = P;
        int color = 0;

        while (R.any()) {
            ++color;
            FastBitset Q = R;
            while (Q.any()) {
                int v = Q.firstOne();
                Q.resetBit(v);
                R.resetBit(v);
                // same color class must be an independent set => remove neighbors
                Q.andNotWith(adj[v]); // Q &= ~adj[v]
                order.push_back(v);
                colors.push_back(color);
            }
        }
    }

    void expand(FastBitset& P, vector<int>& C) {
        if (!P.any()) {
            if ((int)C.size() > bestSize) {
                bestSize = (int)C.size();
                bestClique = C;
            }
            return;
        }
        if ((int)C.size() + P.popcount() <= bestSize) return;

        vector<int> order, colors;
        colorSort(P, order, colors);

        for (int i = (int)order.size() - 1; i >= 0; --i) {
            int v = order[i];
            if ((int)C.size() + colors[i] <= bestSize) return;

            C.push_back(v);
            FastBitset P2 = P;
            P2.andWith(adj[v]);
            expand(P2, C);
            C.pop_back();

            P.resetBit(v);
        }
    }

    vector<int> solve() {
        bestSize = 0;
        bestClique.clear();

        greedyInitialBound();

        FastBitset P = FastBitset::full(n);
        vector<int> C;
        expand(P, C);
        return bestClique;
    }
};

struct HopcroftKarp {
    int nL, nR;
    const vector<vector<int>>* g; // from left to right
    vector<int> matchL, matchR, dist;
    static constexpr int INF = 1e9;

    HopcroftKarp(int nL_, int nR_, const vector<vector<int>>& g_)
        : nL(nL_), nR(nR_), g(&g_),
          matchL(nL, -1), matchR(nR, -1), dist(nL, INF) {}

    bool bfs() {
        queue<int> q;
        bool found = false;
        for (int i = 0; i < nL; ++i) {
            if (matchL[i] == -1) {
                dist[i] = 0;
                q.push(i);
            } else {
                dist[i] = INF;
            }
        }
        while (!q.empty()) {
            int u = q.front(); q.pop();
            for (int v : (*g)[u]) {
                int u2 = matchR[v];
                if (u2 == -1) {
                    found = true;
                } else if (dist[u2] == INF) {
                    dist[u2] = dist[u] + 1;
                    q.push(u2);
                }
            }
        }
        return found;
    }

    bool dfsIter(int root, vector<int>& it) {
        vector<int> stL;
        vector<int> stR;
        stL.reserve(64);
        stR.reserve(64);

        stL.push_back(root);

        while (!stL.empty()) {
            int u = stL.back();
            bool advanced = false;

            auto& adj = (*g)[u];
            for (int &ei = it[u]; ei < (int)adj.size(); ++ei) {
                int v = adj[ei];
                int u2 = matchR[v];

                if (u2 == -1 || dist[u2] == dist[u] + 1) {
                    // move past this edge for next time we are at u
                    ++ei;

                    stR.push_back(v);
                    if (u2 == -1) {
                        // augment along stacks (forward)
                        for (int k = 0; k < (int)stL.size(); ++k) {
                            int uu = stL[k];
                            int vv = stR[k];
                            matchL[uu] = vv;
                            matchR[vv] = uu;
                        }
                        return true;
                    } else {
                        stL.push_back(u2);
                        advanced = true;
                        break;
                    }
                }
            }

            if (advanced) continue;

            // dead end
            dist[u] = INF;
            stL.pop_back();
            if (!stR.empty()) stR.pop_back();
        }
        return false;
    }

    int maxMatching() {
        int matching = 0;
        vector<int> it(nL, 0);
        while (bfs()) {
            fill(it.begin(), it.end(), 0);
            for (int i = 0; i < nL; ++i) {
                if (matchL[i] == -1) {
                    if (dfsIter(i, it)) ++matching;
                }
            }
        }
        return matching;
    }
};

static vector<int> solveBipartiteComponentMIS(
    const vector<int>& compVerts,
    const vector<vector<int>>& adjList,
    vector<int8_t>& color, // color[v] in {0,1} for v in component
    int nTotal
) {
    // build left/right sets
    vector<int> leftG, rightG;
    leftG.reserve(compVerts.size());
    rightG.reserve(compVerts.size());
    for (int v : compVerts) {
        if (color[v] == 0) leftG.push_back(v);
        else rightG.push_back(v);
    }
    int nL = (int)leftG.size();
    int nR = (int)rightG.size();

    vector<int> lpos(nTotal, -1), rpos(nTotal, -1);
    for (int i = 0; i < nL; ++i) lpos[leftG[i]] = i;
    for (int i = 0; i < nR; ++i) rpos[rightG[i]] = i;

    vector<vector<int>> gL(nL);
    for (int uGlobal : leftG) {
        int u = lpos[uGlobal];
        auto& vec = gL[u];
        for (int vGlobal : adjList[uGlobal]) {
            int rv = rpos[vGlobal];
            if (rv != -1) vec.push_back(rv);
        }
    }

    HopcroftKarp hk(nL, nR, gL);
    hk.maxMatching();

    // Konig: find min vertex cover via alternating BFS from unmatched left vertices
    vector<char> visL(nL, false), visR(nR, false);
    queue<int> q;

    for (int l = 0; l < nL; ++l) {
        if (hk.matchL[l] == -1) {
            visL[l] = true;
            q.push(l);
        }
    }

    while (!q.empty()) {
        int l = q.front(); q.pop();
        for (int r : gL[l]) {
            if (hk.matchL[l] == r) continue; // only unmatched edges L->R
            if (!visR[r]) {
                visR[r] = true;
                int l2 = hk.matchR[r]; // matched edge R->L
                if (l2 != -1 && !visL[l2]) {
                    visL[l2] = true;
                    q.push(l2);
                }
            }
        }
    }

    // Max independent set = (Z_L) U (R \ Z_R)
    vector<int> mis;
    mis.reserve(compVerts.size());
    for (int l = 0; l < nL; ++l) if (visL[l]) mis.push_back(leftG[l]);
    for (int r = 0; r < nR; ++r) if (!visR[r]) mis.push_back(rightG[r]);

    return mis;
}

static bool isBipartiteComponent(
    const vector<int>& compVerts,
    const vector<vector<int>>& adjList,
    vector<int8_t>& color
) {
    for (int v : compVerts) color[v] = -1;
    queue<int> q;

    for (int start : compVerts) {
        if (color[start] != -1) continue;
        color[start] = 0;
        q.push(start);

        while (!q.empty()) {
            int u = q.front(); q.pop();
            for (int v : adjList[u]) {
                if (color[v] == -1) {
                    color[v] = (int8_t)(color[u] ^ 1);
                    q.push(v);
                } else if (color[v] == color[u]) {
                    return false;
                }
            }
        }
    }
    return true;
}

static vector<int> solveNonBipartiteComponentMIS_ByCliqueInComplement(
    const vector<int>& compVerts,
    const vector<vector<int>>& adjList
) {
    int sz = (int)compVerts.size();

    // order vertices by descending degree in complement (i.e., ascending degree in original)
    vector<pair<int,int>> ord;
    ord.reserve(sz);
    for (int i = 0; i < sz; ++i) {
        int u = compVerts[i];
        int degG = (int)adjList[u].size();
        ord.push_back({degG, u});
    }
    sort(ord.begin(), ord.end()); // smaller original degree first => larger complement degree first

    vector<int> verts;
    verts.reserve(sz);
    for (auto &p : ord) verts.push_back(p.second);

    int nTotal = 0; // not needed here
    (void)nTotal;

    // global->local map (local indices follow 'verts' order)
    int maxIndex = 0;
    for (int u : verts) maxIndex = max(maxIndex, u);
    // use a map sized by (max vertex index + 1) is not safe globally; build via unordered?:
    // But we do not know maximum index range here; instead pass n from outside usually.
    // We'll instead build using vector<int> sized by (max vertex +1) BUT graph vertices are 0..n-1,
    // so maxIndex is within that range and vector is fine.
    vector<int> pos(maxIndex + 1, -1);
    for (int i = 0; i < sz; ++i) pos[verts[i]] = i;

    // Build adjacency in original graph (within component) as bitsets
    vector<FastBitset> adjG(sz, FastBitset(sz));
    for (int i = 0; i < sz; ++i) adjG[i].init(sz, true);

    for (int i = 0; i < sz; ++i) {
        int u = verts[i];
        for (int v : adjList[u]) {
            if (v >= 0 && v < (int)pos.size()) {
                int j = pos[v];
                if (j != -1 && j != i) adjG[i].setBit(j);
            }
        }
    }

    // Complement graph adjacency for clique search:
    // edge in complement iff NOT edge in original (excluding self)
    FastBitset all = FastBitset::full(sz);
    vector<FastBitset> adjH(sz, FastBitset(sz));
    for (int i = 0; i < sz; ++i) {
        adjH[i] = all;
        adjH[i].andNotWith(adjG[i]); // remove original neighbors
        adjH[i].resetBit(i);         // remove self
    }

    MaxCliqueTomita mc(std::move(adjH));
    vector<int> cliqueLocal = mc.solve();

    vector<int> mis;
    mis.reserve(cliqueLocal.size());
    for (int li : cliqueLocal) mis.push_back(verts[li]);
    return mis;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    int m;
    if (!(cin >> n >> m)) return 0;

    vector<char> hasSelfLoop(n, false);
    vector<pair<int,int>> edges;
    edges.reserve(m);

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue; // ignore invalid input silently
        if (u == v) {
            hasSelfLoop[u] = true;
        } else {
            edges.push_back({u, v});
        }
    }

    vector<char> active(n, true);
    for (int i = 0; i < n; ++i) if (hasSelfLoop[i]) active[i] = false;

    vector<vector<int>> adj(n);
    adj.assign(n, {});
    for (auto [u, v] : edges) {
        if (!active[u] || !active[v]) continue; // vertices with self-loop can never be chosen; ignore them entirely
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    // Find connected components on active vertices
    vector<char> vis(n, false);
    vector<int8_t> color(n, -1);

    vector<int> answer;
    answer.reserve(n);

    vector<int> q;
    q.reserve(n);

    for (int i = 0; i < n; ++i) {
        if (!active[i] || vis[i]) continue;

        // BFS component
        vector<int> comp;
        q.clear();
        q.push_back(i);
        vis[i] = true;

        for (size_t qi = 0; qi < q.size(); ++qi) {
            int u = q[qi];
            comp.push_back(u);
            for (int v : adj[u]) {
                if (!vis[v]) {
                    vis[v] = true;
                    q.push_back(v);
                }
            }
        }

        if (comp.size() == 1) {
            answer.push_back(comp[0]);
            continue;
        }

        bool bip = isBipartiteComponent(comp, adj, color);
        if (bip) {
            vector<int> mis = solveBipartiteComponentMIS(comp, adj, color, n);
            answer.insert(answer.end(), mis.begin(), mis.end());
        } else {
            // Exact MIS on general component via max clique in complement
            // (Works well when non-bipartite components are not huge.)
            vector<int> mis = solveNonBipartiteComponentMIS_ByCliqueInComplement(comp, adj);
            answer.insert(answer.end(), mis.begin(), mis.end());
        }
    }

    // Output
    cout << answer.size() << "\n";
    for (size_t i = 0; i < answer.size(); ++i) {
        if (i) cout << ' ';
        cout << answer[i];
    }
    cout << "\n";
    return 0;
}
