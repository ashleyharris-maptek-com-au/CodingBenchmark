#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <cstdint>
#include <limits>

using std::int32_t;
using std::uint32_t;
using std::uint64_t;

static inline uint64_t splitmix64(uint64_t x) {
    // https://prng.di.unimi.it/splitmix64.c
    x += 0x9e3779b97f4a7c15ull;
    x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ull;
    x = (x ^ (x >> 27)) * 0x94d049bb133111ebull;
    return x ^ (x >> 31);
}

static inline size_t nextPow2(size_t x) {
    if (x <= 1) return 1;
    --x;
    x |= x >> 1;
    x |= x >> 2;
    x |= x >> 4;
    x |= x >> 8;
    x |= x >> 16;
#if SIZE_MAX > UINT32_MAX
    x |= x >> 32;
#endif
    return x + 1;
}

struct Edge {
    int32_t to;
    int32_t next;
    unsigned char ch;
};

struct TransHash {
    static constexpr uint64_t EMPTY = std::numeric_limits<uint64_t>::max();

    std::vector<uint64_t> keys;     // key = (state<<8) | ch
    std::vector<int32_t>  edgeIdx;  // value = index into edges
    size_t mask = 0;
    size_t filled = 0;

    void init(size_t expectedEntries) {
        size_t cap = nextPow2(std::max<size_t>(8, (expectedEntries * 10) / 7 + 1)); // ~0.7 load
        keys.assign(cap, EMPTY);
        edgeIdx.assign(cap, -1);
        mask = cap - 1;
        filled = 0;
    }

    void rehash(size_t newCap) {
        newCap = nextPow2(std::max<size_t>(8, newCap));
        std::vector<uint64_t> oldKeys = std::move(keys);
        std::vector<int32_t> oldVals = std::move(edgeIdx);

        keys.assign(newCap, EMPTY);
        edgeIdx.assign(newCap, -1);
        mask = newCap - 1;
        filled = 0;

        for (size_t i = 0; i < oldKeys.size(); ++i) {
            uint64_t k = oldKeys[i];
            if (k == EMPTY) continue;
            int32_t v = oldVals[i];
            // insert
            size_t pos = (size_t)splitmix64(k) & mask;
            while (keys[pos] != EMPTY) pos = (pos + 1) & mask;
            keys[pos] = k;
            edgeIdx[pos] = v;
            ++filled;
        }
    }

    void ensureOneMore() {
        // if (filled+1) / cap > 0.7 => (filled+1)*10 > cap*7
        size_t cap = keys.size();
        if ((filled + 1) * 10 > cap * 7) rehash(cap * 2);
    }

    inline uint64_t makeKey(int32_t state, unsigned char ch) const {
        return (uint64_t)((uint32_t)state) << 8 | (uint64_t)ch;
    }

    int32_t find(uint64_t k) const {
        size_t pos = (size_t)splitmix64(k) & mask;
        while (true) {
            uint64_t kk = keys[pos];
            if (kk == EMPTY) return -1;
            if (kk == k) return edgeIdx[pos];
            pos = (pos + 1) & mask;
        }
    }

    int32_t find(int32_t state, unsigned char ch) const {
        return find(makeKey(state, ch));
    }

    void insert(uint64_t k, int32_t eidx) {
        ensureOneMore();
        size_t pos = (size_t)splitmix64(k) & mask;
        while (keys[pos] != EMPTY) {
            if (keys[pos] == k) { // overwrite (shouldn't generally happen for SAM build, but safe)
                edgeIdx[pos] = eidx;
                return;
            }
            pos = (pos + 1) & mask;
        }
        keys[pos] = k;
        edgeIdx[pos] = eidx;
        ++filled;
    }

    void insert(int32_t state, unsigned char ch, int32_t eidx) {
        insert(makeKey(state, ch), eidx);
    }
};

struct SAM {
    std::vector<int32_t> len;
    std::vector<int32_t> link;
    std::vector<int32_t> head;
    std::vector<int32_t> firstpos;
    std::vector<Edge> edges;
    TransHash h;

    int32_t last = 0;

    void init(size_t baseLen) {
        // reserve states ~ 2*baseLen
        size_t stCap = std::max<size_t>(2, 2 * baseLen + 1);
        len.reserve(stCap);
        link.reserve(stCap);
        head.reserve(stCap);
        firstpos.reserve(stCap);

        edges.reserve(std::max<size_t>(4, 2 * baseLen + 1));

        // expected transitions O(n) (constant factor). Choose ~2n for a start.
        h.init(std::max<size_t>(16, 2 * baseLen + 1));

        // state 0
        len.push_back(0);
        link.push_back(-1);
        head.push_back(-1);
        firstpos.push_back(-1);
        last = 0;
    }

    inline int32_t newState(int32_t l, int32_t lk, int32_t fp) {
        int32_t id = (int32_t)len.size();
        len.push_back(l);
        link.push_back(lk);
        head.push_back(-1);
        firstpos.push_back(fp);
        return id;
    }

    inline int32_t getEdgeIdx(int32_t v, unsigned char c) const {
        return h.find(v, c);
    }

    inline int32_t nextState(int32_t v, unsigned char c) const {
        int32_t e = h.find(v, c);
        return (e < 0) ? -1 : edges[(size_t)e].to;
    }

    inline void addTransition(int32_t from, unsigned char c, int32_t to) {
        Edge e;
        e.to = to;
        e.ch = c;
        e.next = head[(size_t)from];
        int32_t idx = (int32_t)edges.size();
        edges.push_back(e);
        head[(size_t)from] = idx;
        h.insert(from, c, idx);
    }

    inline void copyTransitions(int32_t from, int32_t to) {
        for (int32_t e = head[(size_t)from]; e != -1; e = edges[(size_t)e].next) {
            addTransition(to, edges[(size_t)e].ch, edges[(size_t)e].to);
        }
    }

    void extend(unsigned char c, int32_t pos) {
        int32_t cur = newState(len[(size_t)last] + 1, 0, pos);
        int32_t p = last;

        while (p != -1 && getEdgeIdx(p, c) == -1) {
            addTransition(p, c, cur);
            p = link[(size_t)p];
        }

        if (p == -1) {
            link[(size_t)cur] = 0;
        } else {
            int32_t eidx = getEdgeIdx(p, c);
            int32_t q = edges[(size_t)eidx].to;
            if (len[(size_t)p] + 1 == len[(size_t)q]) {
                link[(size_t)cur] = q;
            } else {
                int32_t clone = newState(len[(size_t)p] + 1, link[(size_t)q], firstpos[(size_t)q]);
                copyTransitions(q, clone);

                while (p != -1) {
                    int32_t ei = getEdgeIdx(p, c);
                    if (ei == -1) break;
                    if (edges[(size_t)ei].to != q) break;
                    edges[(size_t)ei].to = clone;
                    p = link[(size_t)p];
                }

                link[(size_t)q] = clone;
                link[(size_t)cur] = clone;
            }
        }
        last = cur;
    }
};

// Classic DP for 2 strings (small only).
static std::string lcs2_dp(const std::string& a, const std::string& b) {
    if (a.empty() || b.empty()) return {};
    const size_t n = a.size(), m = b.size();
    std::vector<int32_t> prev(m + 1, 0), cur(m + 1, 0);
    int32_t best = 0;
    size_t bestEnd = 0; // end position in a

    for (size_t i = 1; i <= n; ++i) {
        for (size_t j = 1; j <= m; ++j) {
            if ((unsigned char)a[i - 1] == (unsigned char)b[j - 1]) {
                cur[j] = prev[j - 1] + 1;
                if (cur[j] > best) {
                    best = cur[j];
                    bestEnd = i;
                }
            } else {
                cur[j] = 0;
            }
        }
        std::swap(prev, cur);
    }
    if (best == 0) return {};
    return a.substr(bestEnd - (size_t)best, (size_t)best);
}

static inline void strip_trailing_cr(std::string& s) {
    if (!s.empty() && s.back() == '\r') s.pop_back();
}

// Radix sort indices by sam.len[idx] ascending (2-pass 16-bit).
static void radixSortByLen(std::vector<uint32_t>& a, const std::vector<int32_t>& keyLen) {
    const size_t n = a.size();
    if (n <= 1) return;

    std::vector<uint32_t> tmp(n);
    static thread_local std::vector<uint32_t> cnt(65536);

    // low 16
    std::fill(cnt.begin(), cnt.end(), 0);
    for (size_t i = 0; i < n; ++i) {
        uint32_t k = (uint32_t)keyLen[(size_t)a[i]];
        ++cnt[k & 0xFFFFu];
    }
    uint32_t sum = 0;
    for (size_t i = 0; i < 65536; ++i) {
        uint32_t c = cnt[i];
        cnt[i] = sum;
        sum += c;
    }
    for (size_t i = 0; i < n; ++i) {
        uint32_t x = a[i];
        uint32_t k = (uint32_t)keyLen[(size_t)x];
        tmp[cnt[k & 0xFFFFu]++] = x;
    }

    // high 16
    std::fill(cnt.begin(), cnt.end(), 0);
    for (size_t i = 0; i < n; ++i) {
        uint32_t k = (uint32_t)keyLen[(size_t)tmp[i]];
        ++cnt[(k >> 16) & 0xFFFFu];
    }
    sum = 0;
    for (size_t i = 0; i < 65536; ++i) {
        uint32_t c = cnt[i];
        cnt[i] = sum;
        sum += c;
    }
    for (size_t i = 0; i < n; ++i) {
        uint32_t x = tmp[i];
        uint32_t k = (uint32_t)keyLen[(size_t)x];
        a[cnt[(k >> 16) & 0xFFFFu]++] = x;
    }
}

static void buildLenOrderCounting(const SAM& sam, std::vector<uint32_t>& orderAsc) {
    // Counting sort states by len (ascending). Memory O(maxLen).
    int32_t maxL = 0;
    for (int32_t l : sam.len) if (l > maxL) maxL = l;
    std::vector<uint32_t> cnt((size_t)maxL + 1, 0);
    orderAsc.resize(sam.len.size());

    for (size_t i = 0; i < sam.len.size(); ++i) ++cnt[(size_t)sam.len[i]];
    uint32_t sum = 0;
    for (size_t i = 0; i < cnt.size(); ++i) {
        uint32_t c = cnt[i];
        cnt[i] = sum;
        sum += c;
    }
    for (uint32_t v = 0; v < (uint32_t)sam.len.size(); ++v) {
        orderAsc[(size_t)cnt[(size_t)sam.len[(size_t)v]]++] = v;
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N;
    if (!(std::cin >> N)) return 0;
    std::string dummy;
    std::getline(std::cin, dummy);

    if (N <= 0) {
        std::cout << "\n";
        return 0;
    }

    std::vector<std::string> s;
    s.resize((size_t)N);
    size_t baseIdx = 0;
    size_t baseLen = std::numeric_limits<size_t>::max();

    for (int i = 0; i < N; ++i) {
        std::getline(std::cin, s[(size_t)i]);
        strip_trailing_cr(s[(size_t)i]);
        if (s[(size_t)i].size() < baseLen) {
            baseLen = s[(size_t)i].size();
            baseIdx = (size_t)i;
        }
    }

    if (N == 1) {
        std::cout << s[0] << "\n";
        return 0;
    }

    if (baseLen == 0) {
        std::cout << "\n";
        return 0;
    }

    // Small-case specialized DP for 2 strings only.
    if (N == 2) {
        const std::string& a = s[0];
        const std::string& b = s[1];
        // DP threshold: keep it very small to avoid O(n*m) blowups.
        if (a.size() <= 6000 && b.size() <= 6000) {
            std::string ans = lcs2_dp(a, b);
            std::cout << ans << "\n";
            return 0;
        }
    }

    // Put shortest string first (base).
    if (baseIdx != 0) std::swap(s[0], s[baseIdx]);
    const std::string& base = s[0];

    // Build SAM over base.
    SAM sam;
    sam.init(base.size());
    for (int32_t i = 0; i < (int32_t)base.size(); ++i) {
        sam.extend((unsigned char)base[(size_t)i], i);
    }

    const uint32_t numStates = (uint32_t)sam.len.size();

    // best[v] = min over processed strings of (max match length in that string for state v)
    std::vector<int32_t> best(numStates, 0);
    for (uint32_t v = 0; v < numStates; ++v) best[v] = sam.len[(size_t)v];
    best[0] = 0;

    // mx[v] = per-string temporary (max match length ending in v, then propagated)
    std::vector<int32_t> mx(numStates, 0);

    // Active states where best[v] > 0 (monotonically shrinks); we never need to revisit eliminated states.
    std::vector<uint32_t> active, newActive;
    active.reserve(numStates);
    for (uint32_t v = 1; v < numStates; ++v) active.push_back(v);

    // For touched-mode: vis[v] == epoch means v is in the closure set (touched + suffix ancestors) for current string.
    std::vector<uint32_t> vis(numStates, 0);
    uint32_t epoch = 1;

    std::vector<uint32_t> touched;
    touched.reserve(1 << 20);

    // For full-scan mode: order of states by len ascending (built lazily).
    bool orderBuilt = false;
    std::vector<uint32_t> orderAsc;

    // Process other strings in ascending length as a heuristic.
    std::vector<size_t> idx;
    idx.reserve((size_t)N - 1);
    for (int i = 1; i < N; ++i) idx.push_back((size_t)i);
    std::sort(idx.begin(), idx.end(), [&](size_t i, size_t j) { return s[i].size() < s[j].size(); });

    auto traverseString = [&](const std::string& str, bool markTouched) {
        int32_t v = 0;
        int32_t l = 0;

        if (markTouched) {
            ++epoch;
            if (epoch == 0) { // extremely unlikely wraparound; reset vis
                std::fill(vis.begin(), vis.end(), 0);
                epoch = 1;
            }
            touched.clear();
        }

        for (unsigned char c : str) {
            int32_t nxt = sam.nextState(v, c);
            while (v != 0 && nxt == -1) {
                v = sam.link[(size_t)v];
                l = sam.len[(size_t)v];
                nxt = sam.nextState(v, c);
            }
            if (nxt != -1) {
                v = nxt;
                ++l;
            } else {
                v = 0;
                l = 0;
            }
            if (l > mx[(size_t)v]) mx[(size_t)v] = l;
            if (markTouched && v != 0 && vis[(size_t)v] != epoch) {
                vis[(size_t)v] = epoch;
                touched.push_back((uint32_t)v);
            }
        }
    };

    // Strategy thresholds (heuristics).
    // If active is large, full scan is typically faster than building closures + radix sorting.
    auto wantFullScan = [&](size_t activeSz) -> bool {
        if (numStates <= 2'000'000u) return true;          // small SAM: full scan is fine
        if ((size_t)N <= 10) return true;                  // few strings: full scan ok
        if (activeSz * 10 >= (size_t)numStates * 8) return true; // active >= 80%
        return false;
    };

    for (size_t k = 0; k < idx.size(); ++k) {
        const std::string& curStr = s[idx[k]];
        if (active.empty()) break;
        if (curStr.empty()) { // then no non-empty common substring can exist
            active.clear();
            break;
        }

        if (wantFullScan(active.size())) {
            if (!orderBuilt) {
                buildLenOrderCounting(sam, orderAsc);
                orderBuilt = true;
            }

            traverseString(curStr, /*markTouched=*/false);

            // Propagate mx via suffix links (descending len) and update best in the same pass.
            newActive.clear();
            newActive.reserve(active.size());

            for (size_t ii = orderAsc.size(); ii-- > 0;) {
                uint32_t v = orderAsc[ii];
                int32_t p = sam.link[(size_t)v];

                if (p != -1) {
                    int32_t cand = mx[(size_t)v];
                    int32_t pl = sam.len[(size_t)p];
                    if (cand > pl) cand = pl;
                    if (cand > mx[(size_t)p]) mx[(size_t)p] = cand;
                }

                if (v != 0) {
                    int32_t b = best[(size_t)v];
                    int32_t m = mx[(size_t)v];
                    if (m < b) b = m;
                    best[(size_t)v] = b;
                    if (b > 0) newActive.push_back(v);
                }
                mx[(size_t)v] = 0; // clear for next string
            }

            active.swap(newActive);
        } else {
            // Touched-mode: only update currently-active states, and only propagate on closure of visited states.
            traverseString(curStr, /*markTouched=*/true);

            // Add suffix-link ancestors to make a closed set, so one descending pass is sufficient.
            for (size_t i = 0; i < touched.size(); ++i) {
                uint32_t v = touched[i];
                int32_t p = sam.link[(size_t)v];
                while (p != -1) {
                    uint32_t up = (uint32_t)p;
                    if (up == 0) break;
                    if (vis[(size_t)up] == epoch) break;
                    vis[(size_t)up] = epoch;
                    touched.push_back(up);
                    p = sam.link[(size_t)up];
                }
            }

            // Sort closure by len ascending, then traverse descending to propagate.
            radixSortByLen(touched, sam.len);

            for (size_t ii = touched.size(); ii-- > 0;) {
                uint32_t v = touched[ii];
                int32_t p = sam.link[(size_t)v];
                if (p == -1 || p == 0) continue;
                int32_t cand = mx[(size_t)v];
                int32_t pl = sam.len[(size_t)p];
                if (cand > pl) cand = pl;
                if (cand > mx[(size_t)p]) mx[(size_t)p] = cand;
            }

            // Update only active states; states not in closure have mx=0 => best becomes 0 and are dropped.
            newActive.clear();
            newActive.reserve(active.size());
            for (uint32_t v : active) {
                int32_t m = (vis[(size_t)v] == epoch) ? mx[(size_t)v] : 0;
                int32_t b = best[(size_t)v];
                if (m < b) b = m;
                best[(size_t)v] = b;
                if (b > 0) newActive.push_back(v);
            }
            active.swap(newActive);

            // Clear mx for touched closure only.
            for (uint32_t v : touched) mx[(size_t)v] = 0;
        }
    }

    // Find best answer among active states (or all states if active empty -> length 0).
    int32_t ansLen = 0;
    uint32_t ansState = 0;

    // If active is empty, answer is empty. Otherwise scan active.
    for (uint32_t v : active) {
        int32_t b = best[(size_t)v];
        if (b > ansLen) {
            ansLen = b;
            ansState = v;
        }
    }

    if (ansLen <= 0) {
        std::cout << "\n";
        return 0;
    }

    int32_t endPos = sam.firstpos[(size_t)ansState];
    int32_t startPos = endPos - ansLen + 1;
    if (startPos < 0) startPos = 0;

    std::cout.write(base.data() + startPos, (std::streamsize)ansLen);
    std::cout << "\n";
    return 0;
}
