```python
#!/usr/bin/env python3
# 3-SAT heuristic solver (WalkSAT-style local search), optimized for fast I/O and
# memory-efficient storage.
#
# Notes:
# - 3-SAT is NP-complete; this is a practical heuristic. The prompt guarantees
#   instances are satisfiable; this aims to find a satisfying assignment quickly.
# - Stores all literals and builds variable→occurrence lists for fast incremental
#   updates during local search.

import sys
import time
from array import array


class FastScanner:
    __slots__ = ("_fd", "_buf", "_mv", "_n", "_i")

    def __init__(self, fd, buf_size=1 << 22):
        self._fd = fd
        self._buf = bytearray(buf_size)
        self._mv = memoryview(self._buf)
        self._n = 0
        self._i = 0

    def _refill(self):
        self._n = self._fd.readinto(self._buf)
        self._i = 0
        return self._n

    def next_int(self):
        n = self._n
        i = self._i
        mv = self._mv

        while True:
            if i >= n:
                if not self._refill():
                    return None
                n = self._n
                i = self._i

            # skip whitespace
            while i < n and mv[i] <= 32:
                i += 1
            if i < n:
                break

        sign = 1
        if mv[i] == 45:  # '-'
            sign = -1
            i += 1

        x = 0
        while True:
            if i >= n:
                self._i = i
                if not self._refill():
                    break
                n = self._n
                i = self._i
            c = mv[i]
            if c <= 32:
                break
            x = x * 10 + (c - 48)
            i += 1

        self._i = i
        return sign * x


# Very fast PRNG (xorshift64)
MASK64 = (1 << 64) - 1


class XorShift64:
    __slots__ = ("s",)

    def __init__(self, seed):
        self.s = (seed & MASK64) or 88172645463393265

    def next(self):
        x = self.s
        x ^= (x << 7) & MASK64
        x ^= (x >> 9) & MASK64
        self.s = x & MASK64
        return self.s

    def randbelow(self, n: int) -> int:
        return self.next() % n


def main():
    fs = FastScanner(sys.stdin.buffer)

    n = fs.next_int()
    m = fs.next_int()
    if n is None or m is None:
        return
    if n < 0 or m < 0:
        return

    total_lits = 3 * m

    # literals in clause order (flat): clause cid uses lits[3*cid : 3*cid+3]
    lits = array("i", [0]) * total_lits

    # diff[v] = (#positive occurrences) - (#negative occurrences), for majority init
    diff = array("i", [0]) * (n + 1)

    # occ_count[v] = total occurrences of variable v (pos+neg), for adjacency build
    occ_count = array("I", [0]) * (n + 1)

    # Read all literals; update counts
    for p in range(total_lits):
        lit = fs.next_int()
        if lit is None:
            # malformed / early EOF
            lit = 1
        lits[p] = lit
        if lit > 0:
            v = lit
            diff[v] += 1
        else:
            v = -lit
            diff[v] -= 1
        occ_count[v] += 1

    # Initial assignment: majority sign for each variable (ties -> True)
    assign = bytearray(n + 1)  # 0/1
    for v in range(1, n + 1):
        assign[v] = 1 if diff[v] >= 0 else 0
    del diff

    # Build offsets (CSR) for variable->occurrences
    offs = array("I", [0]) * (n + 2)
    s = 0
    for v in range(1, n + 1):
        offs[v] = s
        s += occ_count[v]
    offs[n + 1] = s  # should be total_lits

    # For each occurrence, store packed entry: (clause_id << 1) | sign_bit
    # sign_bit: 1 for positive literal (x_v), 0 for negative literal (~x_v)
    occ_entries = array("I", [0]) * s

    # Reuse occ_count as write cursor: cursor[v] starts at offs[v]
    for v in range(1, n + 1):
        occ_count[v] = offs[v]

    # Fill adjacency by scanning lits
    for p in range(total_lits):
        lit = lits[p]
        if lit > 0:
            v = lit
            sign_bit = 1
        else:
            v = -lit
            sign_bit = 0
        cid = p // 3
        entry = (cid << 1) | sign_bit

        k = occ_count[v]
        occ_entries[k] = entry
        occ_count[v] = k + 1

    del occ_count

    # Clause true-literal counts (0..3) and unsat tracking
    num_true = bytearray(m)
    is_unsat = bytearray(m)
    unsat = array("I")
    unsat_count = 0

    p = 0
    for cid in range(m):
        t = 0

        lit0 = lits[p]
        if lit0 > 0:
            t += assign[lit0]
        else:
            t += assign[-lit0] ^ 1

        lit1 = lits[p + 1]
        if lit1 > 0:
            t += assign[lit1]
        else:
            t += assign[-lit1] ^ 1

        lit2 = lits[p + 2]
        if lit2 > 0:
            t += assign[lit2]
        else:
            t += assign[-lit2] ^ 1

        num_true[cid] = t
        if t == 0:
            is_unsat[cid] = 1
            unsat.append(cid)
            unsat_count += 1

        p += 3

    if unsat_count:
        rng = XorShift64(seed=(time.time_ns() ^ (n * 1315423911) ^ (m * 2654435761)))
        start_time = time.monotonic()
        deadline = start_time + 295.0  # keep margin under 5 minutes

        noise_pct = 35  # percent probability to pick random var from clause
        steps = 0
        last_check = 0
        last_unsat = unsat_count

        # Helper: compute break count for flipping variable v
        def break_count(v: int) -> int:
            av = assign[v]  # 0/1
            cnt = 0
            a = offs[v]
            b = offs[v + 1]
            for i in range(a, b):
                entry = occ_entries[i]
                cid2 = entry >> 1
                if num_true[cid2] == 1:
                    sign_bit2 = entry & 1
                    # This literal is currently true iff assign[v] == sign_bit (pos needs 1, neg needs 0)
                    if av == sign_bit2:
                        cnt += 1
            return cnt

        # Local search
        while unsat_count and time.monotonic() < deadline:
            # pick a currently-unsatisfied clause id
            while True:
                cid = unsat[rng.randbelow(len(unsat))]
                if is_unsat[cid]:
                    break

            base = 3 * cid
            l0 = lits[base]
            l1 = lits[base + 1]
            l2 = lits[base + 2]

            v0 = l0 if l0 > 0 else -l0
            v1 = l1 if l1 > 0 else -l1
            v2 = l2 if l2 > 0 else -l2

            # Choose variable to flip
            if (rng.next() % 100) < noise_pct:
                r = rng.next() % 3
                v = v0 if r == 0 else (v1 if r == 1 else v2)
            else:
                b0 = break_count(v0)
                b1 = break_count(v1)
                b2 = break_count(v2)
                mb = b0
                v = v0
                # manual min with random tie-breaking
                if b1 < mb or (b1 == mb and (rng.next() & 1)):
                    mb = b1
                    v = v1
                if b2 < mb or (b2 == mb and (rng.next() & 1)):
                    v = v2

            # Flip v and incrementally update affected clauses
            old = assign[v]
            assign[v] = old ^ 1

            a = offs[v]
            b = offs[v + 1]
            for i in range(a, b):
                entry = occ_entries[i]
                cid2 = entry >> 1
                sign_bit = entry & 1

                nt = num_true[cid2]
                before_true = 1 if (old == sign_bit) else 0  # literal truth toggles
                new_nt = nt - 1 if before_true else nt + 1
                num_true[cid2] = new_nt

                if nt == 0 and new_nt == 1:
                    if is_unsat[cid2]:
                        is_unsat[cid2] = 0
                        unsat_count -= 1
                elif nt == 1 and new_nt == 0:
                    if not is_unsat[cid2]:
                        is_unsat[cid2] = 1
                        unsat.append(cid2)
                        unsat_count += 1

            steps += 1

            # Periodic cleanup/perturbation
            if steps - last_check >= 1_000_000:
                # Compact unsat list if it has many stale entries
                if len(unsat) > 4 * unsat_count + 4096:
                    new_unsat = array("I")
                    new_unsat_extend = new_unsat.append
                    for c in unsat:
                        if is_unsat[c]:
                            new_unsat_extend(c)
                    unsat = new_unsat

                # If no improvement, do small random perturbation using unsat clauses
                if unsat_count >= last_unsat:
                    k = 5000
                    if len(unsat) and unsat_count:
                        for _ in range(k):
                            c = unsat[rng.randbelow(len(unsat))]
                            if not is_unsat[c]:
                                continue
                            b0 = 3 * c
                            lit = lits[b0 + (rng.next() % 3)]
                            vv = lit if lit > 0 else -lit

                            old2 = assign[vv]
                            assign[vv] = old2 ^ 1
                            aa = offs[vv]
                            bb = offs[vv + 1]
                            for j in range(aa, bb):
                                e2 = occ_entries[j]
                                c2 = e2 >> 1
                                sb2 = e2 & 1
                                nt2 = num_true[c2]
                                before2 = 1 if (old2 == sb2) else 0
                                new2 = nt2 - 1 if before2 else nt2 + 1
                                num_true[c2] = new2
                                if nt2 == 0 and new2 == 1:
                                    if is_unsat[c2]:
                                        is_unsat[c2] = 0
                                        unsat_count -= 1
                                elif nt2 == 1 and new2 == 0:
                                    if not is_unsat[c2]:
                                        is_unsat[c2] = 1
                                        unsat.append(c2)
                                        unsat_count += 1

                last_unsat = unsat_count
                last_check = steps

    # Output (assumes satisfiable and solved; per prompt instances are satisfiable)
    out = sys.stdout.buffer
    out.write(b"SAT\n")

    # Stream assignment as "0/1 " bytes, chunked to avoid huge temporary strings
    chunk = 1_000_000
    i = 1
    while i <= n:
        j = i + chunk
        if j > n + 1:
            j = n + 1
        k = j - i
        buf = bytearray(2 * k)
        mv = memoryview(buf)
        t = 0
        # fill "v " pairs
        for val in assign[i:j]:
            mv[t] = 49 if val else 48  # '1' or '0'
            mv[t + 1] = 32            # space
            t += 2
        if j == n + 1:
            buf[-1] = 10  # newline instead of trailing space
        out.write(buf)
        i = j


if __name__ == "__main__":
    main()
```
