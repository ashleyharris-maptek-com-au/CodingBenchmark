```rust
use std::cmp::Ordering;
use std::collections::BinaryHeap;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug, Eq, PartialEq)]
struct Item {
    gain: i32,
    v: usize,
    ver: u64,
}

impl Ord for Item {
    fn cmp(&self, other: &Self) -> Ordering {
        // Max-heap by gain, then by vertex id for tie-break stability.
        self.gain
            .cmp(&other.gain)
            .then_with(|| other.v.cmp(&self.v)) // smaller v is "bigger" -> deterministic
            .then_with(|| self.ver.cmp(&other.ver))
    }
}
impl PartialOrd for Item {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}
impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self {
            buf: input.into_bytes(),
            idx: 0,
        }
    }
    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let mut x: usize = 0;
        while self.idx < self.buf.len() && !self.buf[self.idx].is_ascii_whitespace() {
            x = x * 10 + (self.buf[self.idx] - b'0') as usize;
            self.idx += 1;
        }
        Some(x)
    }
}

fn is_dominating_set(n: usize, adj: &[Vec<usize>], set: &[usize]) -> bool {
    if n == 0 {
        return true;
    }
    let mut dom = vec![false; n];
    for &v in set {
        dom[v] = true;
        for &u in &adj[v] {
            dom[u] = true;
        }
    }
    dom.into_iter().all(|x| x)
}

fn main() {
    let mut sc = Scanner::new();
    let n = match sc.next_usize() {
        Some(x) => x,
        None => {
            // Empty input -> output empty dominating set
            println!("0\n");
            return;
        }
    };
    let m = sc.next_usize().unwrap_or(0);

    if n == 0 {
        println!("0\n");
        return;
    }

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u = match sc.next_usize() {
            Some(x) => x,
            None => break,
        };
        let v = match sc.next_usize() {
            Some(x) => x,
            None => break,
        };
        if u >= n || v >= n {
            continue;
        }
        if u == v {
            // Self-loops don't change domination; ignore.
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    // Deduplicate adjacency lists (important for correctness of counts/pruning).
    for a in &mut adj {
        a.sort_unstable();
        a.dedup();
    }

    // --- Greedy construction with dynamic "undominated coverage" gains ---
    let mut dominated = vec![false; n];
    let mut in_set = vec![false; n];

    let mut gain: Vec<i32> = (0..n).map(|v| (adj[v].len() as i32) + 1).collect();
    let mut ver: Vec<u64> = vec![0; n];

    let mut heap = BinaryHeap::new();
    for v in 0..n {
        heap.push(Item {
            gain: gain[v],
            v,
            ver: ver[v],
        });
    }

    let mut undom_left = n;
    let mut set: Vec<usize> = Vec::new();

    while undom_left > 0 {
        let item = loop {
            let top = heap.pop().expect("Heap exhausted while vertices remain undominated");
            if in_set[top.v] {
                continue;
            }
            if top.ver != ver[top.v] || top.gain != gain[top.v] {
                continue; // stale
            }
            break top;
        };

        if item.gain <= 0 {
            // Should not happen; as long as there is an undominated vertex u, choosing u yields gain>=1.
            // Fallback: add any undominated vertex.
            if let Some(u) = (0..n).find(|&u| !dominated[u] && !in_set[u]) {
                in_set[u] = true;
                set.push(u);
                // Dominate u and update bookkeeping (same codepath as below would).
                let v = u;
                // Apply domination of closed neighborhood of v:
                // For each newly dominated vertex x, decrement gain for all vertices in closed neighborhood of x.
                let mut newly_dominated: Vec<usize> = Vec::new();
                if !dominated[v] {
                    dominated[v] = true;
                    undom_left -= 1;
                    newly_dominated.push(v);
                }
                for &u2 in &adj[v] {
                    if !dominated[u2] {
                        dominated[u2] = true;
                        undom_left -= 1;
                        newly_dominated.push(u2);
                    }
                }
                for x in newly_dominated {
                    // For each dominator candidate y in N[x] U {x}, u=x is no longer undominated -> gain[y]--
                    // (only if y not selected yet).
                    if !in_set[x] {
                        if gain[x] > 0 {
                            gain[x] -= 1;
                        }
                        ver[x] += 1;
                        heap.push(Item {
                            gain: gain[x],
                            v: x,
                            ver: ver[x],
                        });
                    }
                    for &y in &adj[x] {
                        if in_set[y] {
                            continue;
                        }
                        if gain[y] > 0 {
                            gain[y] -= 1;
                        }
                        ver[y] += 1;
                        heap.push(Item {
                            gain: gain[y],
                            v: y,
                            ver: ver[y],
                        });
                    }
                }
                continue;
            } else {
                break;
            }
        }

        let v = item.v;
        in_set[v] = true;
        set.push(v);

        // Mark dominated vertices in closed neighborhood of v, and for each newly dominated vertex x,
        // decrement gains for all y in closed neighborhood of x.
        let mut newly_dominated: Vec<usize> = Vec::new();

        if !dominated[v] {
            dominated[v] = true;
            undom_left -= 1;
            newly_dominated.push(v);
        }
        for &u in &adj[v] {
            if !dominated[u] {
                dominated[u] = true;
                undom_left -= 1;
                newly_dominated.push(u);
            }
        }

        for x in newly_dominated {
            // Update y = x (self)
            if !in_set[x] {
                if gain[x] > 0 {
                    gain[x] -= 1;
                }
                ver[x] += 1;
                heap.push(Item {
                    gain: gain[x],
                    v: x,
                    ver: ver[x],
                });
            }
            // Update neighbors y of x
            for &y in &adj[x] {
                if in_set[y] {
                    continue;
                }
                if gain[y] > 0 {
                    gain[y] -= 1;
                }
                ver[y] += 1;
                heap.push(Item {
                    gain: gain[y],
                    v: y,
                    ver: ver[y],
                });
            }
        }
    }

    // Safety check: ensure valid dominating set (should always be true).
    if !is_dominating_set(n, &adj, &set) {
        // Very conservative fallback: add all vertices (always dominating).
        set = (0..n).collect();
        for &v in &set {
            in_set[v] = true;
        }
    }

    // --- Prune redundant vertices (simple, fast local improvement) ---
    // dom_count[u] = how many chosen vertices dominate u.
    let mut dom_count = vec![0u32; n];
    for &v in &set {
        dom_count[v] += 1;
        for &u in &adj[v] {
            dom_count[u] += 1;
        }
    }

    let mut i = 0usize;
    while i < set.len() {
        let v = set[i];

        // v is removable if every vertex in its closed neighborhood has dom_count >= 2.
        let mut removable = dom_count[v] >= 2;
        if removable {
            for &u in &adj[v] {
                if dom_count[u] < 2 {
                    removable = false;
                    break;
                }
            }
        }

        if removable {
            // Remove v
            in_set[v] = false;
            dom_count[v] -= 1;
            for &u in &adj[v] {
                dom_count[u] -= 1;
            }
            set.swap_remove(i);
        } else {
            i += 1;
        }
    }

    // Final safety check.
    if !is_dominating_set(n, &adj, &set) {
        // Add any uncovered vertices (should not happen, but keep output valid).
        let mut dom = vec![false; n];
        for &v in &set {
            dom[v] = true;
            for &u in &adj[v] {
                dom[u] = true;
            }
        }
        for u in 0..n {
            if !dom[u] {
                set.push(u);
                dom[u] = true;
                for &w in &adj[u] {
                    dom[w] = true;
                }
            }
        }
    }

    // Output
    set.sort_unstable();
    println!("{}", set.len());
    if set.is_empty() {
        println!();
    } else {
        let mut out = String::new();
        for (k, v) in set.iter().enumerate() {
            if k > 0 {
                out.push(' ');
            }
            out.push_str(&v.to_string());
        }
        println!("{out}");
    }
}
```
