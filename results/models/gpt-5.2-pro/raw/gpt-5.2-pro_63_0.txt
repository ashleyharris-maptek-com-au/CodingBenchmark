```typescript
declare const require: any;
const fs = require("fs");

type Robot = { id: number; x: number; y: number };
type Job = { id: any; x: number; y: number; priority: number };

const input: string = fs.readFileSync(0, "utf8");
const data = JSON.parse(input);

const grid: string[] = data.grid;
const robots: Robot[] = data.robots;
const jobs: Job[] = data.jobs;

const H = grid.length | 0;
const W = (H > 0 ? grid[0].length : 0) | 0;
const N = (W * H) | 0;

function cellIndex(x: number, y: number): number {
  return (y * W + x) | 0;
}

function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < W && y >= 0 && y < H;
}

// Open cell mask
const open = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = grid[y];
  // Assume consistent width; if not, treat missing as blocked.
  for (let x = 0; x < W; x++) {
    const c = row.charCodeAt(x);
    // '.' => 46, '#' => 35
    open[cellIndex(x, y)] = c === 46 ? 1 : 0;
  }
}

// Robot id range
let maxRobotId = -1;
let minActiveRobotId = 2147483647;
const activeRobotIds: number[] = [];
for (let i = 0; i < robots.length; i++) {
  const rid = robots[i].id | 0;
  if (rid > maxRobotId) maxRobotId = rid;
  if (rid < minActiveRobotId) minActiveRobotId = rid;
  activeRobotIds.push(rid);
}
if (maxRobotId < 0) {
  // No robots; still must output something.
  process.stdout.write(JSON.stringify({ assignments: [] }));
  process.exit(0);
}
if (minActiveRobotId === 2147483647) minActiveRobotId = 0;

// Per-cell doubly-linked list of robots currently on that cell
const cellHead = new Int32Array(N);
cellHead.fill(-1);

const robotPos = new Int32Array(maxRobotId + 1);
robotPos.fill(-1);

const robotNext = new Int32Array(maxRobotId + 1);
robotNext.fill(-1);

const robotPrev = new Int32Array(maxRobotId + 1);
robotPrev.fill(-1);

function addRobotToCell(rid: number, cell: number): void {
  robotPos[rid] = cell;
  const head = cellHead[cell];
  robotPrev[rid] = -1;
  robotNext[rid] = head;
  if (head !== -1) robotPrev[head] = rid;
  cellHead[cell] = rid;
}

function removeRobotFromCell(rid: number): void {
  const cell = robotPos[rid];
  if (cell < 0) return;
  const p = robotPrev[rid];
  const n = robotNext[rid];
  if (p !== -1) robotNext[p] = n;
  else cellHead[cell] = n;
  if (n !== -1) robotPrev[n] = p;
  robotPrev[rid] = -1;
  robotNext[rid] = -1;
  robotPos[rid] = -1;
}

function moveRobot(rid: number, newCell: number): void {
  removeRobotFromCell(rid);
  addRobotToCell(rid, newCell);
}

function minRobotInCell(cell: number): number {
  let r = cellHead[cell];
  let best = 2147483647;
  while (r !== -1) {
    if (r < best) best = r;
    r = robotNext[r];
  }
  return best === 2147483647 ? -1 : best;
}

// Initialize robots
for (let i = 0; i < robots.length; i++) {
  const r = robots[i];
  const rid = r.id | 0;
  if (!inBounds(r.x, r.y)) continue;
  const c = cellIndex(r.x, r.y);
  if (open[c] === 0) continue;
  addRobotToCell(rid, c);
}

// Assignments array: index is robot id
const assignments: any[][] = new Array(maxRobotId + 1);
for (let i = 0; i <= maxRobotId; i++) assignments[i] = [];

// Deterministic job order: priority desc, then id asc (string), then y, then x
jobs.sort((a, b) => {
  const pa = +a.priority;
  const pb = +b.priority;
  if (pa !== pb) return pb - pa;
  const sa = String(a.id);
  const sb = String(b.id);
  if (sa < sb) return -1;
  if (sa > sb) return 1;
  if (a.y !== b.y) return a.y - b.y;
  return a.x - b.x;
});

// BFS workspace
const visitedStamp = new Uint32Array(N);
const dist = new Int32Array(N);
const queue = new Int32Array(N);
let stamp = 1;

function bumpStamp(): void {
  stamp = (stamp + 1) >>> 0;
  if (stamp === 0) {
    visitedStamp.fill(0);
    stamp = 1;
  }
}

function findNearestRobotFromJobCell(src: number): number {
  if (src < 0 || src >= N || open[src] === 0) return minActiveRobotId;

  // Distance 0 case
  if (cellHead[src] !== -1) {
    const r0 = minRobotInCell(src);
    return r0 !== -1 ? r0 : minActiveRobotId;
  }

  bumpStamp();
  let head = 0;
  let tail = 0;

  queue[tail++] = src;
  visitedStamp[src] = stamp;
  dist[src] = 0;

  let bestDist = -1;
  let bestRobot = -1;

  while (head < tail) {
    const u = queue[head++];
    const du = dist[u];

    // If we already have a best distance, don't expand from nodes at/after it.
    if (bestDist !== -1 && du >= bestDist) {
      // Still consider robots on this node (in case we didn't check on enqueue).
      if (cellHead[u] !== -1 && du === bestDist) {
        const mr = minRobotInCell(u);
        if (mr !== -1 && (bestRobot === -1 || mr < bestRobot)) bestRobot = mr;
      }
      continue;
    }

    const ux = u % W;

    // Neighbor helper
    const tryVisit = (v: number) => {
      if (open[v] === 0) return;
      if (visitedStamp[v] === stamp) return;
      const nd = du + 1;
      if (bestDist !== -1 && nd > bestDist) return;

      visitedStamp[v] = stamp;
      dist[v] = nd;
      queue[tail++] = v;

      if (cellHead[v] !== -1) {
        const mr = minRobotInCell(v);
        if (mr !== -1) {
          if (bestDist === -1) {
            bestDist = nd;
            bestRobot = mr;
          } else if (nd === bestDist && mr < bestRobot) {
            bestRobot = mr;
          }
        }
      }
    };

    if (ux > 0) tryVisit(u - 1);
    if (ux + 1 < W) tryVisit(u + 1);
    if (u >= W) tryVisit(u - W);
    if (u + W < N) tryVisit(u + W);
  }

  return bestRobot !== -1 ? bestRobot : minActiveRobotId;
}

// Main greedy assignment
for (let i = 0; i < jobs.length; i++) {
  const j = jobs[i];
  const x = j.x | 0;
  const y = j.y | 0;
  const jc = inBounds(x, y) ? cellIndex(x, y) : -1;

  const rid = findNearestRobotFromJobCell(jc) | 0;
  assignments[rid].push(j.id);

  if (jc >= 0 && jc < N && open[jc] === 1) {
    // Move robot to job location (sequence order matters for later distances)
    moveRobot(rid, jc);
  }
}

process.stdout.write(JSON.stringify({ assignments }));
```
