```spirv
; SPIR-V
; Version: 1.0
; Generator: hand-written
; Bound: 200
OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn = OpTypeFunction %void

%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%uint_4 = OpConstant %uint 4
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_thicknessHalf = OpConstant %float 0.01
%float_oneThird = OpConstant %float 0.333333343
%float_twoThird = OpConstant %float 0.666666687

%main = OpFunction %void None %fn
%entry = OpLabel

; Load positions: p0=prev, p1=start, p2=end, p3=next
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr
%p3 = OpLoad %v4float %p3_ptr

; Load endpoint colors: c1 = color[1], c2 = color[2]
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c1 = OpLoad %v3float %c1_ptr
%c2 = OpLoad %v3float %c2_ptr

; Compute adjacency tangents in XY:
; tanStart = normalize( (p2 - p0).xy )
; tanEnd   = normalize( (p3 - p1).xy )
%dp20 = OpFSub %v4float %p2 %p0
%dp31 = OpFSub %v4float %p3 %p1
%dp20_xy = OpVectorShuffle %v2float %dp20 %dp20 0 1
%dp31_xy = OpVectorShuffle %v2float %dp31 %dp31 0 1
%tanStart = OpExtInst %v2float %glsl Normalize %dp20_xy
%tanEnd = OpExtInst %v2float %glsl Normalize %dp31_xy

; Segment delta for interpolation
%seg = OpFSub %v4float %p2 %p1
%dc = OpFSub %v3float %c2 %c1

; ============================================================
; Dash 1: t0 = 0, t1 = 1/3
; ============================================================

; Endpoint A (t = 0)
%tA = OpCopyObject %float %float_0
%pA_off = OpVectorTimesScalar %v4float %seg %tA
%pA = OpFAdd %v4float %p1 %pA_off

%tanDiff = OpFSub %v2float %tanEnd %tanStart
%tanA_off = OpVectorTimesScalar %v2float %tanDiff %tA
%tanA_raw = OpFAdd %v2float %tanStart %tanA_off
%tanA = OpExtInst %v2float %glsl Normalize %tanA_raw

%tanA_x = OpCompositeExtract %float %tanA 0
%tanA_y = OpCompositeExtract %float %tanA 1
%perpA_x = OpFNegate %float %tanA_y
%perpA_y = OpCopyObject %float %tanA_x
%perpA = OpCompositeConstruct %v2float %perpA_x %perpA_y

%pA_w = OpCompositeExtract %float %pA 3
%scaleA0 = OpFMul %float %float_thicknessHalf %pA_w
%offA2 = OpVectorTimesScalar %v2float %perpA %scaleA0
%offA_x = OpCompositeExtract %float %offA2 0
%offA_y = OpCompositeExtract %float %offA2 1
%offA4 = OpCompositeConstruct %v4float %offA_x %offA_y %float_0 %float_0
%pA_plus = OpFAdd %v4float %pA %offA4
%pA_minus = OpFSub %v4float %pA %offA4

%cA_off = OpVectorTimesScalar %v3float %dc %tA
%cA = OpFAdd %v3float %c1 %cA_off

; Endpoint B (t = 1/3)
%tB = OpCopyObject %float %float_oneThird
%pB_off = OpVectorTimesScalar %v4float %seg %tB
%pB = OpFAdd %v4float %p1 %pB_off

%tanB_off = OpVectorTimesScalar %v2float %tanDiff %tB
%tanB_raw = OpFAdd %v2float %tanStart %tanB_off
%tanB = OpExtInst %v2float %glsl Normalize %tanB_raw

%tanB_x = OpCompositeExtract %float %tanB 0
%tanB_y = OpCompositeExtract %float %tanB 1
%perpB_x = OpFNegate %float %tanB_y
%perpB_y = OpCopyObject %float %tanB_x
%perpB = OpCompositeConstruct %v2float %perpB_x %perpB_y

%pB_w = OpCompositeExtract %float %pB 3
%scaleB0 = OpFMul %float %float_thicknessHalf %pB_w
%offB2 = OpVectorTimesScalar %v2float %perpB %scaleB0
%offB_x = OpCompositeExtract %float %offB2 0
%offB_y = OpCompositeExtract %float %offB2 1
%offB4 = OpCompositeConstruct %v4float %offB_x %offB_y %float_0 %float_0
%pB_plus = OpFAdd %v4float %pB %offB4
%pB_minus = OpFSub %v4float %pB %offB4

%cB_off = OpVectorTimesScalar %v3float %dc %tB
%cB = OpFAdd %v3float %c1 %cB_off

; Emit quad as triangle strip: A+, A-, B+, B-
%outPosPtr = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %outPosPtr %pA_plus
OpStore %outColor %cA
OpEmitVertex

OpStore %outPosPtr %pA_minus
OpStore %outColor %cA
OpEmitVertex

OpStore %outPosPtr %pB_plus
OpStore %outColor %cB
OpEmitVertex

OpStore %outPosPtr %pB_minus
OpStore %outColor %cB
OpEmitVertex

OpEndPrimitive

; ============================================================
; Dash 2: t0 = 2/3, t1 = 1
; ============================================================

; Endpoint C (t = 2/3)
%tC = OpCopyObject %float %float_twoThird
%pC_off = OpVectorTimesScalar %v4float %seg %tC
%pC = OpFAdd %v4float %p1 %pC_off

%tanC_off = OpVectorTimesScalar %v2float %tanDiff %tC
%tanC_raw = OpFAdd %v2float %tanStart %tanC_off
%tanC = OpExtInst %v2float %glsl Normalize %tanC_raw

%tanC_x = OpCompositeExtract %float %tanC 0
%tanC_y = OpCompositeExtract %float %tanC 1
%perpC_x = OpFNegate %float %tanC_y
%perpC_y = OpCopyObject %float %tanC_x
%perpC = OpCompositeConstruct %v2float %perpC_x %perpC_y

%pC_w = OpCompositeExtract %float %pC 3
%scaleC0 = OpFMul %float %float_thicknessHalf %pC_w
%offC2 = OpVectorTimesScalar %v2float %perpC %scaleC0
%offC_x = OpCompositeExtract %float %offC2 0
%offC_y = OpCompositeExtract %float %offC2 1
%offC4 = OpCompositeConstruct %v4float %offC_x %offC_y %float_0 %float_0
%pC_plus = OpFAdd %v4float %pC %offC4
%pC_minus = OpFSub %v4float %pC %offC4

%cC_off = OpVectorTimesScalar %v3float %dc %tC
%cC = OpFAdd %v3float %c1 %cC_off

; Endpoint D (t = 1)
%tD = OpCopyObject %float %float_1
%pD_off = OpVectorTimesScalar %v4float %seg %tD
%pD = OpFAdd %v4float %p1 %pD_off

%tanD_off = OpVectorTimesScalar %v2float %tanDiff %tD
%tanD_raw = OpFAdd %v2float %tanStart %tanD_off
%tanD = OpExtInst %v2float %glsl Normalize %tanD_raw

%tanD_x = OpCompositeExtract %float %tanD 0
%tanD_y = OpCompositeExtract %float %tanD 1
%perpD_x = OpFNegate %float %tanD_y
%perpD_y = OpCopyObject %float %tanD_x
%perpD = OpCompositeConstruct %v2float %perpD_x %perpD_y

%pD_w = OpCompositeExtract %float %pD 3
%scaleD0 = OpFMul %float %float_thicknessHalf %pD_w
%offD2 = OpVectorTimesScalar %v2float %perpD %scaleD0
%offD_x = OpCompositeExtract %float %offD2 0
%offD_y = OpCompositeExtract %float %offD2 1
%offD4 = OpCompositeConstruct %v4float %offD_x %offD_y %float_0 %float_0
%pD_plus = OpFAdd %v4float %pD %offD4
%pD_minus = OpFSub %v4float %pD %offD4

%cD_off = OpVectorTimesScalar %v3float %dc %tD
%cD = OpFAdd %v3float %c1 %cD_off

; Emit quad as triangle strip: C+, C-, D+, D-
OpStore %outPosPtr %pC_plus
OpStore %outColor %cC
OpEmitVertex

OpStore %outPosPtr %pC_minus
OpStore %outColor %cC
OpEmitVertex

OpStore %outPosPtr %pD_plus
OpStore %outColor %cD
OpEmitVertex

OpStore %outPosPtr %pD_minus
OpStore %outColor %cD
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
