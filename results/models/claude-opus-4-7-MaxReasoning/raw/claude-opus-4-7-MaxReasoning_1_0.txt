#include <cstdio>
#include <cstdlib>
#include <cstdint>
#include <cstring>
#include <vector>
#include <algorithm>
#include <cmath>
#include <chrono>

using namespace std;
using Clock = chrono::steady_clock;

// ===================== Fast input =====================
static char ibuf[1 << 22];
static int  ipos = 0, ilen = 0;

static inline int rch() {
    if (ipos >= ilen) {
        ilen = (int)fread(ibuf, 1, sizeof(ibuf), stdin);
        ipos = 0;
        if (ilen <= 0) return -1;
    }
    return ibuf[ipos++];
}

static int readInt() {
    int c, n = 0, s = 1;
    do { c = rch(); } while (c != -1 && (c == ' ' || c == '\n' || c == '\r' || c == '\t'));
    if (c == '-') { s = -1; c = rch(); }
    while (c >= '0' && c <= '9') { n = n * 10 + (c - '0'); c = rch(); }
    return n * s;
}

static double readDouble() {
    int c;
    do { c = rch(); } while (c != -1 && (c == ' ' || c == '\n' || c == '\r' || c == '\t'));
    double sign = 1.0;
    if (c == '-') { sign = -1.0; c = rch(); }
    else if (c == '+') { c = rch(); }
    double val = 0.0;
    while (c >= '0' && c <= '9') { val = val * 10.0 + (c - '0'); c = rch(); }
    if (c == '.') {
        c = rch();
        double frac = 0.1;
        while (c >= '0' && c <= '9') {
            val += (c - '0') * frac;
            frac *= 0.1;
            c = rch();
        }
    }
    if (c == 'e' || c == 'E') {
        c = rch();
        int es = 1;
        if (c == '-') { es = -1; c = rch(); }
        else if (c == '+') { c = rch(); }
        int e = 0;
        while (c >= '0' && c <= '9') { e = e * 10 + (c - '0'); c = rch(); }
        double m = 1.0;
        for (int i = 0; i < e; i++) m *= 10.0;
        val = (es > 0) ? val * m : val / m;
    }
    return sign * val;
}

// ===================== Fast output =====================
static char obuf[1 << 22];
static int  opos = 0;

static inline void flushOut() {
    if (opos > 0) { fwrite(obuf, 1, opos, stdout); opos = 0; }
}
static inline void wch(char c) {
    if (opos >= (int)sizeof(obuf)) flushOut();
    obuf[opos++] = c;
}
static inline void writeInt(int n) {
    if (opos + 12 >= (int)sizeof(obuf)) flushOut();
    if (n < 0) { obuf[opos++] = '-'; n = -n; }
    if (n == 0) { obuf[opos++] = '0'; return; }
    char tmp[12]; int l = 0;
    while (n > 0) { tmp[l++] = (char)('0' + (n % 10)); n /= 10; }
    while (l > 0) obuf[opos++] = tmp[--l];
}

// ===================== Hilbert curve mapping =====================
// Standard XY->D for Hilbert curve of order `order` (grid is 2^order x 2^order).
static inline uint64_t hilbertXY2D(uint32_t order, uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (uint32_t s = 1u << (order - 1); s > 0; s >>= 1) {
        uint32_t rx = (x & s) ? 1u : 0u;
        uint32_t ry = (y & s) ? 1u : 0u;
        d += (uint64_t)s * (uint64_t)s * ((3u * rx) ^ ry);
        if (ry == 0u) {
            if (rx == 1u) {
                x = s - 1u - x;
                y = s - 1u - y;
            }
            uint32_t t = x; x = y; y = t;
        }
    }
    return d;
}

int main() {
    auto t0 = Clock::now();
    auto elapsed = [&]() -> double {
        return chrono::duration<double>(Clock::now() - t0).count();
    };

    int N = readInt();
    if (N <= 0) { flushOut(); return 0; }

    vector<float> xs((size_t)N), ys((size_t)N);
    for (int i = 0; i < N; i++) {
        xs[i] = (float)readDouble();
        ys[i] = (float)readDouble();
    }

    if (N == 1) {
        writeInt(0);
        wch('\n');
        flushOut();
        return 0;
    }

    vector<int> tour((size_t)N);

    // Bounding box
    float minX = xs[0], maxX = xs[0], minY = ys[0], maxY = ys[0];
    for (int i = 1; i < N; i++) {
        float x = xs[i], y = ys[i];
        if (x < minX) minX = x; else if (x > maxX) maxX = x;
        if (y < minY) minY = y; else if (y > maxY) maxY = y;
    }
    float rngX = maxX - minX;
    float rngY = maxY - minY;
    float rng  = rngX > rngY ? rngX : rngY;
    if (rng < 1e-9f) rng = 1.0f;

    auto sqd = [&](int a, int b) -> float {
        float dx = xs[a] - xs[b];
        float dy = ys[a] - ys[b];
        return dx * dx + dy * dy;
    };
    auto dist = [&](int a, int b) -> float {
        return sqrtf(sqd(a, b));
    };

    // ===================== Strategy selection =====================
    // Huge N: Hilbert curve only (O(N log N)).
    // Medium N: nearest-neighbor with spatial grid.
    // Small N: NN + 2-opt local search.
    const int HILBERT_THRESHOLD = 100000;
    const int TWO_OPT_THRESHOLD = 5000;

    if (N >= HILBERT_THRESHOLD) {
        // -------- Hilbert curve sort --------
        const uint32_t ORDER  = 16;
        const uint32_t maxCo  = (1u << ORDER) - 1u;
        const float    scale  = (float)maxCo / rng;

        vector<pair<uint64_t, int>> sv((size_t)N);
        for (int i = 0; i < N; i++) {
            int hx = (int)((xs[i] - minX) * scale);
            int hy = (int)((ys[i] - minY) * scale);
            if (hx < 0) hx = 0;
            if (hy < 0) hy = 0;
            if ((uint32_t)hx > maxCo) hx = (int)maxCo;
            if ((uint32_t)hy > maxCo) hy = (int)maxCo;
            sv[i] = { hilbertXY2D(ORDER, (uint32_t)hx, (uint32_t)hy), i };
        }
        sort(sv.begin(), sv.end());

        for (int i = 0; i < N; i++) tour[i] = sv[i].second;

        // Rotate so city 0 is first
        int zp = 0;
        for (int i = 0; i < N; i++) if (tour[i] == 0) { zp = i; break; }
        if (zp != 0) rotate(tour.begin(), tour.begin() + zp, tour.end());

        sv.clear();
        sv.shrink_to_fit();
    } else {
        // -------- Nearest-neighbor with spatial grid --------
        int gd = (int)sqrt((double)N / 4.0);
        if (gd < 1) gd = 1;
        if (gd > 1024) gd = 1024;
        float cs  = rng / (float)gd;
        if (cs < 1e-9f) cs = 1.0f;
        float ics = 1.0f / cs;
        int gn = gd * gd;

        vector<vector<int>> grid((size_t)gn);

        auto cellIdx = [&](int i) -> int {
            int gx = (int)((xs[i] - minX) * ics);
            int gy = (int)((ys[i] - minY) * ics);
            if (gx < 0) gx = 0; else if (gx >= gd) gx = gd - 1;
            if (gy < 0) gy = 0; else if (gy >= gd) gy = gd - 1;
            return gy * gd + gx;
        };

        for (int i = 0; i < N; i++) grid[cellIdx(i)].push_back(i);

        tour[0] = 0;
        // Remove city 0 from its cell (swap-pop)
        {
            auto& cl = grid[cellIdx(0)];
            for (size_t k = 0; k < cl.size(); k++) {
                if (cl[k] == 0) { cl[k] = cl.back(); cl.pop_back(); break; }
            }
        }
        int cur = 0;

        for (int step = 1; step < N; step++) {
            float cx = xs[cur], cy = ys[cur];
            int gx = (int)((cx - minX) * ics);
            int gy = (int)((cy - minY) * ics);
            if (gx < 0) gx = 0; else if (gx >= gd) gx = gd - 1;
            if (gy < 0) gy = 0; else if (gy >= gd) gy = gd - 1;

            int   best  = -1;
            float bestD = 1e30f;

            int maxR = gd;
            for (int r = 0; r <= maxR; r++) {
                int x0 = gx - r, x1 = gx + r;
                int y0 = gy - r, y1 = gy + r;
                int cx0 = x0 < 0 ? 0 : x0;
                int cx1 = x1 >= gd ? gd - 1 : x1;

                if (r == 0) {
                    auto& cl = grid[gy * gd + gx];
                    for (int j : cl) {
                        float d = sqd(cur, j);
                        if (d < bestD) { bestD = d; best = j; }
                    }
                } else {
                    // Top edge row
                    if (y0 >= 0) {
                        int rs = y0 * gd;
                        for (int x = cx0; x <= cx1; x++) {
                            auto& cl = grid[rs + x];
                            for (int j : cl) {
                                float d = sqd(cur, j);
                                if (d < bestD) { bestD = d; best = j; }
                            }
                        }
                    }
                    // Bottom edge row
                    if (y1 < gd && y1 != y0) {
                        int rs = y1 * gd;
                        for (int x = cx0; x <= cx1; x++) {
                            auto& cl = grid[rs + x];
                            for (int j : cl) {
                                float d = sqd(cur, j);
                                if (d < bestD) { bestD = d; best = j; }
                            }
                        }
                    }
                    // Middle rows: only left/right columns
                    int iy0 = (y0 + 1 > 0) ? (y0 + 1) : 0;
                    int iy1 = (y1 - 1 < gd) ? (y1 - 1) : (gd - 1);
                    if (iy0 < 0) iy0 = 0;
                    if (iy1 >= gd) iy1 = gd - 1;
                    for (int y = iy0; y <= iy1; y++) {
                        int rs = y * gd;
                        if (x0 >= 0) {
                            auto& cl = grid[rs + x0];
                            for (int j : cl) {
                                float d = sqd(cur, j);
                                if (d < bestD) { bestD = d; best = j; }
                            }
                        }
                        if (x1 < gd && x1 != x0) {
                            auto& cl = grid[rs + x1];
                            for (int j : cl) {
                                float d = sqd(cur, j);
                                if (d < bestD) { bestD = d; best = j; }
                            }
                        }
                    }
                }

                if (best != -1) {
                    // After searching Chebyshev ring r, nearest unseen city
                    // can be no closer than r * cs.
                    float minNext = (float)r * cs;
                    if (bestD <= minNext * minNext) break;
                }
            }

            // Safety fallback (shouldn't trigger)
            if (best == -1) {
                for (int g = 0; g < gn; g++) {
                    if (!grid[g].empty()) {
                        best = grid[g].front();
                        break;
                    }
                }
            }

            tour[step] = best;
            {
                auto& cl = grid[cellIdx(best)];
                for (size_t k = 0; k < cl.size(); k++) {
                    if (cl[k] == best) { cl[k] = cl.back(); cl.pop_back(); break; }
                }
            }
            cur = best;
        }

        grid.clear();
        grid.shrink_to_fit();

        // -------- 2-opt local search (small N only) --------
        if (N <= TWO_OPT_THRESHOLD) {
            double budget;
            if (N <= 200)        budget = 30.0;
            else if (N <= 1000)  budget = 90.0;
            else if (N <= 3000)  budget = 180.0;
            else                 budget = 240.0;

            int maxRounds = 60;
            bool improved = true;
            int rounds = 0;
            while (improved && rounds < maxRounds && elapsed() < budget) {
                improved = false;
                rounds++;
                for (int i = 0; i < N - 1; i++) {
                    if ((i & 1023) == 0 && elapsed() >= budget) break;
                    int a   = tour[i];
                    int b   = tour[i + 1];
                    float dab = dist(a, b);
                    for (int j = i + 2; j < N; j++) {
                        int c = tour[j];
                        int d = (j + 1 < N) ? tour[j + 1] : tour[0];
                        if (a == d) continue;
                        float dac = dist(a, c);
                        float dbd = dist(b, d);
                        float dcd = dist(c, d);
                        if (dac + dbd + 1e-9f < dab + dcd) {
                            reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                            improved = true;
                            b   = tour[i + 1];
                            dab = dist(a, b);
                        }
                    }
                }
            }
        }
    }

    // ===================== Output =====================
    for (int i = 0; i < N; i++) {
        if (i > 0) wch(' ');
        writeInt(tour[i]);
    }
    wch('\n');
    flushOut();
    return 0;
}
