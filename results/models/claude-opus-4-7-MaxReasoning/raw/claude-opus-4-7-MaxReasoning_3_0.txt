use std::io::{self, Read, Write, BufWriter};

struct Rng { state: u64 }
impl Rng {
    fn new(seed: u64) -> Self { Rng { state: seed | 1 } }
    fn next_u64(&mut self) -> u64 {
        self.state = self.state.wrapping_add(0x9E3779B97F4A7C15);
        let mut z = self.state;
        z = (z ^ (z >> 30)).wrapping_mul(0xBF58476D1CE4E5B9);
        z = (z ^ (z >> 27)).wrapping_mul(0x94D049BB133111EB);
        z ^ (z >> 31)
    }
    fn next_f64(&mut self) -> f64 {
        ((self.next_u64() >> 11) as f64) / ((1u64 << 53) as f64)
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut it = input.split_ascii_whitespace();
    let n: usize = it.next().and_then(|s| s.parse().ok()).unwrap_or(0);
    let m: usize = it.next().and_then(|s| s.parse().ok()).unwrap_or(0);

    let mut edges: Vec<(u32, u32)> = Vec::with_capacity(m);
    for _ in 0..m {
        let u_opt = it.next().and_then(|s| s.parse::<usize>().ok());
        let v_opt = it.next().and_then(|s| s.parse::<usize>().ok());
        let (u, v) = match (u_opt, v_opt) {
            (Some(u), Some(v)) => (u, v),
            _ => break,
        };
        if u >= n || v >= n || u == v { continue; }
        let (a, b) = if u < v { (u as u32, v as u32) } else { (v as u32, u as u32) };
        edges.push((a, b));
    }
    edges.sort_unstable();
    edges.dedup();

    let positions = layout(n, &edges);

    writeln!(out, "{}", n).unwrap();
    for &(x, y) in &positions {
        writeln!(out, "{:.6} {:.6}", x, y).unwrap();
    }
}

fn layout(n: usize, edges: &[(u32, u32)]) -> Vec<(f64, f64)> {
    if n == 0 { return Vec::new(); }
    if n == 1 { return vec![(0.0, 0.0)]; }
    if n == 2 { return vec![(-0.5, 0.0), (0.5, 0.0)]; }

    let k = 1.0_f64;
    let mut pos = init_positions(n, k);
    let iters = pick_iters(n);

    if n <= 800 {
        fr_direct(&mut pos, edges, iters, k);
    } else {
        fr_grid(&mut pos, edges, iters, k);
    }

    center(&mut pos);
    pos
}

fn pick_iters(n: usize) -> usize {
    match n {
        0..=30 => 1500,
        31..=100 => 900,
        101..=500 => 500,
        501..=2000 => 350,
        2001..=10000 => 220,
        10001..=50000 => 140,
        50001..=100000 => 90,
        _ => 60,
    }
}

fn init_positions(n: usize, k: f64) -> Vec<(f64, f64)> {
    let mut rng = Rng::new(0xC0FFEE);
    let w = (n as f64).sqrt() * k;
    let mut pos = Vec::with_capacity(n);
    for _ in 0..n {
        let x = (rng.next_f64() - 0.5) * w;
        let y = (rng.next_f64() - 0.5) * w;
        pos.push((x, y));
    }
    pos
}

fn center(pos: &mut [(f64, f64)]) {
    if pos.is_empty() { return; }
    let mut cx = 0.0_f64;
    let mut cy = 0.0_f64;
    for &(x, y) in pos.iter() {
        cx += x;
        cy += y;
    }
    cx /= pos.len() as f64;
    cy /= pos.len() as f64;
    for p in pos.iter_mut() {
        p.0 -= cx;
        p.1 -= cy;
    }
}

fn fr_direct(pos: &mut [(f64, f64)], edges: &[(u32, u32)], iters: usize, k: f64) {
    let n = pos.len();
    let mut disp_x = vec![0.0_f64; n];
    let mut disp_y = vec![0.0_f64; n];
    let init_t = (n as f64).sqrt() * k * 0.4;
    let k2 = k * k;
    let mut rng = Rng::new(7);

    for iter in 0..iters {
        for v in disp_x.iter_mut() { *v = 0.0; }
        for v in disp_y.iter_mut() { *v = 0.0; }

        for i in 0..n {
            let pix = pos[i].0;
            let piy = pos[i].1;
            let mut dxi = 0.0_f64;
            let mut dyi = 0.0_f64;
            for j in (i + 1)..n {
                let dx = pix - pos[j].0;
                let dy = piy - pos[j].1;
                let dist2 = dx * dx + dy * dy;
                if dist2 < 1e-9 {
                    let r1 = (rng.next_f64() - 0.5) * 0.01;
                    let r2 = (rng.next_f64() - 0.5) * 0.01;
                    dxi += r1;
                    dyi += r2;
                    disp_x[j] -= r1;
                    disp_y[j] -= r2;
                    continue;
                }
                let f_over_dist = k2 / dist2;
                let fx = dx * f_over_dist;
                let fy = dy * f_over_dist;
                dxi += fx;
                dyi += fy;
                disp_x[j] -= fx;
                disp_y[j] -= fy;
            }
            disp_x[i] += dxi;
            disp_y[i] += dyi;
        }

        for &(u, v) in edges {
            let u = u as usize;
            let v = v as usize;
            let dx = pos[u].0 - pos[v].0;
            let dy = pos[u].1 - pos[v].1;
            let dist2 = dx * dx + dy * dy;
            if dist2 < 1e-9 { continue; }
            let dist = dist2.sqrt();
            let fx = dx * dist / k;
            let fy = dy * dist / k;
            disp_x[u] -= fx;
            disp_y[u] -= fy;
            disp_x[v] += fx;
            disp_y[v] += fy;
        }

        let t = init_t * (1.0 - (iter as f64) / (iters as f64)).max(0.005);

        for i in 0..n {
            let dx = disp_x[i];
            let dy = disp_y[i];
            let mag2 = dx * dx + dy * dy;
            if mag2 < 1e-18 { continue; }
            let mag = mag2.sqrt();
            let lim = mag.min(t);
            let scale = lim / mag;
            pos[i].0 += dx * scale;
            pos[i].1 += dy * scale;
        }
    }
}

fn fr_grid(pos: &mut [(f64, f64)], edges: &[(u32, u32)], iters: usize, k: f64) {
    let n = pos.len();
    let mut disp_x = vec![0.0_f64; n];
    let mut disp_y = vec![0.0_f64; n];
    let init_t = (n as f64).sqrt() * k * 0.4;
    let k2 = k * k;
    let cell_size_base = k * 2.0;

    let mut bucket_start: Vec<u32> = Vec::new();
    let mut bucket_pos: Vec<u32> = vec![0u32; n];
    let mut tmp: Vec<u32> = Vec::new();
    let mut rng = Rng::new(7);

    for iter in 0..iters {
        for v in disp_x.iter_mut() { *v = 0.0; }
        for v in disp_y.iter_mut() { *v = 0.0; }

        let mut min_x = f64::INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut max_y = f64::NEG_INFINITY;
        for &(x, y) in pos.iter() {
            if x < min_x { min_x = x; }
            if y < min_y { min_y = y; }
            if x > max_x { max_x = x; }
            if y > max_y { max_y = y; }
        }
        if !min_x.is_finite() {
            min_x = 0.0; min_y = 0.0; max_x = 1.0; max_y = 1.0;
        }

        let mut cs = cell_size_base;
        let mut gw = (((max_x - min_x) / cs) as usize + 2).max(2);
        let mut gh = (((max_y - min_y) / cs) as usize + 2).max(2);

        let max_cells = 4_000_000usize;
        if gw.saturating_mul(gh) > max_cells {
            let total = (gw as f64) * (gh as f64);
            let scale = (total / max_cells as f64).sqrt();
            cs *= scale;
            gw = (((max_x - min_x) / cs) as usize + 2).max(2);
            gh = (((max_y - min_y) / cs) as usize + 2).max(2);
        }
        let ncells = gw * gh;

        bucket_start.clear();
        bucket_start.resize(ncells + 1, 0);
        tmp.clear();
        tmp.resize(ncells, 0);

        for &(x, y) in pos.iter() {
            let gx = (((x - min_x) / cs) as usize).min(gw - 1);
            let gy = (((y - min_y) / cs) as usize).min(gh - 1);
            bucket_start[gy * gw + gx + 1] += 1;
        }
        for i in 1..bucket_start.len() {
            bucket_start[i] += bucket_start[i - 1];
        }
        for (i, &(x, y)) in pos.iter().enumerate() {
            let gx = (((x - min_x) / cs) as usize).min(gw - 1);
            let gy = (((y - min_y) / cs) as usize).min(gh - 1);
            let cell = gy * gw + gx;
            bucket_pos[bucket_start[cell] as usize + tmp[cell] as usize] = i as u32;
            tmp[cell] += 1;
        }

        for gy in 0..gh {
            let ylo = gy.saturating_sub(1);
            let yhi = (gy + 1).min(gh - 1);
            for gx in 0..gw {
                let cell = gy * gw + gx;
                let cs_start = bucket_start[cell] as usize;
                let cs_end = bucket_start[cell + 1] as usize;
                if cs_start == cs_end { continue; }

                let xlo = gx.saturating_sub(1);
                let xhi = (gx + 1).min(gw - 1);

                for i_idx in cs_start..cs_end {
                    let i = bucket_pos[i_idx] as usize;
                    let pix = pos[i].0;
                    let piy = pos[i].1;
                    let mut dxi = 0.0_f64;
                    let mut dyi = 0.0_f64;

                    for ny in ylo..=yhi {
                        for nx in xlo..=xhi {
                            let ncell = ny * gw + nx;
                            let ns = bucket_start[ncell] as usize;
                            let ne = bucket_start[ncell + 1] as usize;
                            for j_idx in ns..ne {
                                let j = bucket_pos[j_idx] as usize;
                                if i == j { continue; }
                                let dx = pix - pos[j].0;
                                let dy = piy - pos[j].1;
                                let dist2 = dx * dx + dy * dy;
                                if dist2 < 1e-9 {
                                    dxi += (rng.next_f64() - 0.5) * 0.01;
                                    dyi += (rng.next_f64() - 0.5) * 0.01;
                                    continue;
                                }
                                let f_over_dist = k2 / dist2;
                                dxi += dx * f_over_dist;
                                dyi += dy * f_over_dist;
                            }
                        }
                    }
                    disp_x[i] += dxi;
                    disp_y[i] += dyi;
                }
            }
        }

        for &(u, v) in edges {
            let u = u as usize;
            let v = v as usize;
            let dx = pos[u].0 - pos[v].0;
            let dy = pos[u].1 - pos[v].1;
            let dist2 = dx * dx + dy * dy;
            if dist2 < 1e-9 { continue; }
            let dist = dist2.sqrt();
            let fx = dx * dist / k;
            let fy = dy * dist / k;
            disp_x[u] -= fx;
            disp_y[u] -= fy;
            disp_x[v] += fx;
            disp_y[v] += fy;
        }

        let t = init_t * (1.0 - (iter as f64) / (iters as f64)).max(0.005);

        for i in 0..n {
            let dx = disp_x[i];
            let dy = disp_y[i];
            let mag2 = dx * dx + dy * dy;
            if mag2 < 1e-18 { continue; }
            let mag = mag2.sqrt();
            let lim = mag.min(t);
            let scale = lim / mag;
            pos[i].0 += dx * scale;
            pos[i].1 += dy * scale;
        }
    }
}
