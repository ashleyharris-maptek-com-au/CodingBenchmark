#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <cstring>
#include <vector>
#include <algorithm>
#include <numeric>
#include <chrono>

using namespace std;
using Clock = chrono::steady_clock;

struct Point { float x, y; };

static inline double distSq(const Point& a, const Point& b) {
    double dx = (double)a.x - b.x, dy = (double)a.y - b.y;
    return dx*dx + dy*dy;
}
static inline double dist(const Point& a, const Point& b) {
    return sqrt(distSq(a, b));
}

struct GridHash {
    int gw, gh;
    float minX, minY, cs;
    vector<int> starts, order;

    void build(const Point* pts, int n) {
        minX = pts[0].x; minY = pts[0].y;
        float maxX = minX, maxY = minY;
        for (int i = 1; i < n; i++) {
            if (pts[i].x < minX) minX = pts[i].x;
            if (pts[i].x > maxX) maxX = pts[i].x;
            if (pts[i].y < minY) minY = pts[i].y;
            if (pts[i].y > maxY) maxY = pts[i].y;
        }
        float rx = maxX - minX, ry = maxY - minY;
        int target = max(1, (int)sqrt((double)n / 5.0));
        if (rx < 1e-9f && ry < 1e-9f) { gw = gh = 1; cs = 1.0f; }
        else if (rx < 1e-9f) { gw = 1; gh = target; cs = ry / gh + 1e-9f; }
        else if (ry < 1e-9f) { gw = target; gh = 1; cs = rx / gw + 1e-9f; }
        else {
            float asp = rx / ry;
            gw = max(1, (int)(target * sqrt(asp)));
            gh = max(1, (int)(target / sqrt(asp)));
            cs = max(rx / gw, ry / gh) + 1e-9f;
        }
        long long tot = (long long)gw * gh;
        if (tot > 40000000LL) {
            double s = sqrt(40000000.0 / tot);
            gw = max(1, (int)(gw * s));
            gh = max(1, (int)(gh * s));
            cs = max(rx / gw, ry / gh) + 1e-9f;
        }
        tot = (long long)gw * gh;
        vector<int> cnt(tot + 1, 0);
        order.resize(n);
        for (int i = 0; i < n; i++) {
            int cx = min((int)((pts[i].x - minX) / cs), gw - 1);
            int cy = min((int)((pts[i].y - minY) / cs), gh - 1);
            cnt[cy * gw + cx]++;
        }
        starts.resize(tot + 1);
        starts[0] = 0;
        for (int i = 0; i < (int)tot; i++) starts[i+1] = starts[i] + cnt[i];
        memset(cnt.data(), 0, sizeof(int) * tot);
        for (int i = 0; i < n; i++) {
            int cx = min((int)((pts[i].x - minX) / cs), gw - 1);
            int cy = min((int)((pts[i].y - minY) / cs), gh - 1);
            int cell = cy * gw + cx;
            order[starts[cell] + cnt[cell]] = i;
            cnt[cell]++;
        }
    }

    int findNearest(const Point* pts, int qi, const vector<bool>& used) const {
        float qx = pts[qi].x, qy = pts[qi].y;
        int cx = min(max((int)((qx - minX) / cs), 0), gw - 1);
        int cy = min(max((int)((qy - minY) / cs), 0), gh - 1);
        int best = -1;
        double bestD = 1e30;
        int maxR = max(gw, gh);
        for (int r = 0; r <= maxR; r++) {
            int x0 = max(0, cx - r), x1 = min(gw - 1, cx + r);
            int y0 = max(0, cy - r), y1 = min(gh - 1, cy + r);
            for (int gy = y0; gy <= y1; gy++) {
                for (int gx = x0; gx <= x1; gx++) {
                    if (r > 0 && gx > x0 && gx < x1 && gy > y0 && gy < y1) continue;
                    int cell = gy * gw + gx;
                    int s = starts[cell], e = starts[cell + 1];
                    for (int k = s; k < e; k++) {
                        int idx = order[k];
                        if (used[idx]) continue;
                        double d = distSq(pts[qi], pts[idx]);
                        if (d < bestD) { bestD = d; best = idx; }
                    }
                }
            }
            if (best >= 0 && r > 0) {
                double realD = sqrt(bestD);
                if (realD < (double)(r - 1) * cs) break;
            }
        }
        return best;
    }
};

vector<int> greedyNN_simple(const Point* pts, int n) {
    vector<bool> used(n, false);
    vector<int> tour; tour.reserve(n);
    tour.push_back(0); used[0] = true;
    for (int i = 1; i < n; i++) {
        int cur = tour.back(), bi = -1;
        double bd = 1e30;
        for (int j = 0; j < n; j++) {
            if (!used[j]) {
                double d = distSq(pts[cur], pts[j]);
                if (d < bd) { bd = d; bi = j; }
            }
        }
        tour.push_back(bi); used[bi] = true;
    }
    return tour;
}

vector<int> greedyNN_grid(const Point* pts, int n) {
    GridHash grid;
    grid.build(pts, n);
    vector<bool> used(n, false);
    vector<int> tour; tour.reserve(n);
    tour.push_back(0); used[0] = true;
    for (int i = 1; i < n; i++) {
        int cur = tour.back();
        int next = grid.findNearest(pts, cur, used);
        if (next < 0) {
            for (int j = 0; j < n; j++) if (!used[j]) { next = j; break; }
        }
        tour.push_back(next); used[next] = true;
    }
    return tour;
}

void twoOpt(vector<int>& tour, const Point* pts, int maxIters, double tlSec, Clock::time_point t0) {
    int n = (int)tour.size();
    if (n < 4) return;
    for (int iter = 0; iter < maxIters; iter++) {
        bool improved = false;
        for (int i = 0; i < n - 2; i++) {
            for (int j = i + 2; j < n; j++) {
                if (i == 0 && j == n - 1) continue;
                double oldD = dist(pts[tour[i]], pts[tour[i+1]]) + dist(pts[tour[j]], pts[tour[(j+1)%n]]);
                double newD = dist(pts[tour[i]], pts[tour[j]]) + dist(pts[tour[i+1]], pts[tour[(j+1)%n]]);
                if (newD < oldD - 1e-10) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                }
            }
            if (i % 200 == 0) {
                double el = chrono::duration<double>(Clock::now() - t0).count();
                if (el > tlSec) return;
            }
        }
        if (!improved) break;
    }
}

void twoOptLimited(vector<int>& tour, const Point* pts, int n, double tlSec, Clock::time_point t0) {
    if (n < 4) return;
    int window = min(n, 200);
    if (n > 100000) window = 50;
    if (n > 500000) window = 20;
    for (int pass = 0; pass < 3; pass++) {
        bool improved = false;
        for (int i = 0; i < n - 2; i++) {
            int jmax = min(n - 1, i + window);
            for (int j = i + 2; j <= jmax; j++) {
                double oldD = dist(pts[tour[i]], pts[tour[i+1]]) + dist(pts[tour[j]], pts[tour[(j+1)%n]]);
                double newD = dist(pts[tour[i]], pts[tour[j]]) + dist(pts[tour[i+1]], pts[tour[(j+1)%n]]);
                if (newD < oldD - 1e-10) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                }
            }
            if (i % 5000 == 0) {
                double el = chrono::duration<double>(Clock::now() - t0).count();
                if (el > tlSec) return;
            }
        }
        if (!improved) break;
    }
}

static char buf[1 << 22];
static int bufPos = 0, bufLen = 0;
static inline int readChar() {
    if (bufPos >= bufLen) {
        bufLen = (int)fread(buf, 1, sizeof(buf), stdin);
        bufPos = 0;
        if (bufLen == 0) return -1;
    }
    return (unsigned char)buf[bufPos++];
}
static inline int readInt() {
    int c = readChar(), s = 1, x = 0;
    while (c == ' ' || c == '\n' || c == '\r') c = readChar();
    if (c == '-') { s = -1; c = readChar(); }
    while (c >= '0' && c <= '9') { x = x * 10 + c - '0'; c = readChar(); }
    return x * s;
}
static inline float readFloat() {
    int c = readChar();
    while (c == ' ' || c == '\n' || c == '\r') c = readChar();
    double sign = 1.0, val = 0.0;
    if (c == '-') { sign = -1.0; c = readChar(); }
    while (c >= '0' && c <= '9') { val = val * 10 + (c - '0'); c = readChar(); }
    if (c == '.') {
        c = readChar();
        double frac = 0.1;
        while (c >= '0' && c <= '9') { val += (c - '0') * frac; frac *= 0.1; c = readChar(); }
    }
    if (c == 'e' || c == 'E') {
        c = readChar();
        int esign = 1, exp = 0;
        if (c == '-') { esign = -1; c = readChar(); }
        else if (c == '+') { c = readChar(); }
        while (c >= '0' && c <= '9') { exp = exp * 10 + (c - '0'); c = readChar(); }
        val *= pow(10.0, esign * exp);
    }
    return (float)(sign * val);
}

static char outBuf[1 << 22];
static int outPos = 0;
static inline void flushOut() { fwrite(outBuf, 1, outPos, stdout); outPos = 0; }
static inline void writeInt(int x) {
    if (outPos > (1 << 22) - 20) flushOut();
    if (x == 0) { outBuf[outPos++] = '0'; return; }
    char tmp[12]; int len = 0;
    while (x > 0) { tmp[len++] = '0' + x % 10; x /= 10; }
    for (int i = len - 1; i >= 0; i--) outBuf[outPos++] = tmp[i];
}

int main() {
    auto t0 = Clock::now();
    int n = readInt();
    vector<Point> pts(n);
    for (int i = 0; i < n; i++) {
        pts[i].x = readFloat();
        pts[i].y = readFloat();
    }

    if (n <= 1) {
        for (int i = 0; i < n; i++) { if (i) printf(" "); printf("%d", i); }
        printf("\n");
        return 0;
    }

    vector<int> tour;
    double elapsed = chrono::duration<double>(Clock::now() - t0).count();

    if (n <= 8000) {
        tour = greedyNN_simple(pts.data(), n);
        double tl = 270.0;
        if (n <= 3000) {
            twoOpt(tour, pts.data(), 50, tl, t0);
        } else {
            twoOpt(tour, pts.data(), 3, tl, t0);
        }
    } else if (n <= 1000000) {
        tour = greedyNN_grid(pts.data(), n);
        elapsed = chrono::duration<double>(Clock::now() - t0).count();
        double tl = 260.0 - elapsed;
        if (tl > 5.0) {
            twoOptLimited(tour, pts.data(), n, 260.0, t0);
        }
    } else {
        tour = greedyNN_grid(pts.data(), n);
    }

    for (int i = 0; i < n; i++) {
        if (i) outBuf[outPos++] = ' ';
        writeInt(tour[i]);
    }
    outBuf[outPos++] = '\n';
    flushOut();

    return 0;
}
