#include <cstdio>
#include <cstdlib>
#include <cstdint>
#include <vector>
#include <algorithm>
#include <chrono>
#include <utility>
#include <cstring>

using namespace std;
using namespace std::chrono;

struct Rect {
    int w, h;
    int id;
};

struct Placement {
    int board;
    int x, y;
    int rotated;
};

struct SkyNode {
    int x, y, w;
};

struct Board {
    int W, H;
    vector<SkyNode> sky;
    Board(int W_, int H_) : W(W_), H(H_) {
        sky.push_back({0, 0, W_});
    }
};

struct FitResult {
    int x, y;
    int score1, score2;
    int idx;
    bool ok;
};

static FitResult skylineFit(const Board& b, int rw, int rh) {
    FitResult best;
    best.ok = false;
    best.score1 = INT32_MAX;
    best.score2 = INT32_MAX;
    int n = (int)b.sky.size();
    for (int i = 0; i < n; i++) {
        int x = b.sky[i].x;
        if (x + rw > b.W) continue;
        int widthLeft = rw;
        int j = i;
        int y = b.sky[i].y;
        while (widthLeft > 0 && j < n) {
            if (b.sky[j].y > y) y = b.sky[j].y;
            if (y + rh > b.H) break;
            widthLeft -= b.sky[j].w;
            j++;
        }
        if (widthLeft > 0) continue;
        if (y + rh > b.H) continue;
        int s1 = y + rh;
        int s2 = x;
        if (s1 < best.score1 || (s1 == best.score1 && s2 < best.score2)) {
            best.score1 = s1;
            best.score2 = s2;
            best.x = x;
            best.y = y;
            best.idx = i;
            best.ok = true;
        }
    }
    return best;
}

static void skylineAdd(Board& b, int idx, int x, int y, int rw, int rh) {
    SkyNode newNode{ x, y + rh, rw };
    b.sky.insert(b.sky.begin() + idx, newNode);
    int i = idx + 1;
    while (i < (int)b.sky.size()) {
        SkyNode& cur = b.sky[i];
        SkyNode& prev = b.sky[i - 1];
        if (cur.x < prev.x + prev.w) {
            int shrink = prev.x + prev.w - cur.x;
            if (cur.w <= shrink) {
                b.sky.erase(b.sky.begin() + i);
                continue;
            } else {
                cur.x += shrink;
                cur.w -= shrink;
                break;
            }
        } else {
            break;
        }
    }
    for (int k = 0; k + 1 < (int)b.sky.size();) {
        if (b.sky[k].y == b.sky[k + 1].y) {
            b.sky[k].w += b.sky[k + 1].w;
            b.sky.erase(b.sky.begin() + k + 1);
        } else k++;
    }
}

int main() {
    auto t0 = high_resolution_clock::now();
    int N, BW, BH;
    if (scanf("%d %d %d", &N, &BW, &BH) != 3) return 0;
    vector<Rect> rects(N);
    for (int i = 0; i < N; i++) {
        scanf("%d %d", &rects[i].w, &rects[i].h);
        rects[i].id = i;
    }

    vector<int> order(N);
    for (int i = 0; i < N; i++) order[i] = i;
    sort(order.begin(), order.end(), [&](int a, int b) {
        int ma = max(rects[a].w, rects[a].h);
        int mb = max(rects[b].w, rects[b].h);
        if (ma != mb) return ma > mb;
        long long aa = (long long)rects[a].w * rects[a].h;
        long long bb = (long long)rects[b].w * rects[b].h;
        return aa > bb;
    });

    vector<Board> boards;
    vector<Placement> placements(N);

    auto timeLeft = [&]() {
        auto t = high_resolution_clock::now();
        double el = duration_cast<milliseconds>(t - t0).count() / 1000.0;
        return 28.0 - el;
    };

    for (int oi = 0; oi < N; oi++) {
        int ri = order[oi];
        int rw = rects[ri].w, rh = rects[ri].h;

        int bestBoard = -1;
        int bestRot = 0;
        FitResult bestFit;
        bestFit.ok = false;
        bestFit.score1 = INT32_MAX;
        bestFit.score2 = INT32_MAX;

        int startB = 0;
        int maxBoardsCheck = (int)boards.size();
        if (N > 5000) {
            maxBoardsCheck = min((int)boards.size(), 8);
            startB = (int)boards.size() - maxBoardsCheck;
            if (startB < 0) startB = 0;
        }

        for (int bi = startB; bi < (int)boards.size(); bi++) {
            if (rw <= boards[bi].W && rh <= boards[bi].H) {
                FitResult f = skylineFit(boards[bi], rw, rh);
                if (f.ok && (f.score1 < bestFit.score1 || (f.score1 == bestFit.score1 && f.score2 < bestFit.score2))) {
                    bestFit = f;
                    bestBoard = bi;
                    bestRot = 0;
                }
            }
            if (rw != rh && rh <= boards[bi].W && rw <= boards[bi].H) {
                FitResult f = skylineFit(boards[bi], rh, rw);
                if (f.ok && (f.score1 < bestFit.score1 || (f.score1 == bestFit.score1 && f.score2 < bestFit.score2))) {
                    bestFit = f;
                    bestBoard = bi;
                    bestRot = 1;
                }
            }
            if (timeLeft() < 1.0) break;
        }

        if (!bestFit.ok) {
            boards.emplace_back(BW, BH);
            int bi = (int)boards.size() - 1;
            int useRot = 0;
            int useW = rw, useH = rh;
            bool fitsNorm = (rw <= BW && rh <= BH);
            bool fitsRot = (rh <= BW && rw <= BH);
            if (!fitsNorm && fitsRot) {
                useRot = 1; useW = rh; useH = rw;
            } else if (fitsNorm && fitsRot) {
                if (rw >= rh) { useRot = 0; useW = rw; useH = rh; }
                else { useRot = 1; useW = rh; useH = rw; }
            } else if (!fitsNorm && !fitsRot) {
                placements[ri] = {bi, 0, 0, 0};
                continue;
            }
            FitResult f = skylineFit(boards[bi], useW, useH);
            if (!f.ok) {
                placements[ri] = {bi, 0, 0, useRot};
                continue;
            }
            skylineAdd(boards[bi], f.idx, f.x, f.y, useW, useH);
            placements[ri] = { bi, f.x, f.y, useRot };
        } else {
            int useW = bestRot ? rh : rw;
            int useH = bestRot ? rw : rh;
            skylineAdd(boards[bestBoard], bestFit.idx, bestFit.x, bestFit.y, useW, useH);
            placements[ri] = { bestBoard, bestFit.x, bestFit.y, bestRot };
        }
    }

    printf("%d\n", (int)boards.size());
    for (int i = 0; i < N; i++) {
        Placement& p = placements[i];
        int b = p.board;
        if (b < 0) b = 0;
        printf("%d %d %d %d\n", b, p.x, p.y, p.rotated);
    }
    return 0;
}
