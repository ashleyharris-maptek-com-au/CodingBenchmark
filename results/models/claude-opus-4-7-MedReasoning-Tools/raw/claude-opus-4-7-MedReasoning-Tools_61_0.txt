declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Pkg { id: any; x: number; y: number; }
const depot: [number, number] = data.depot;
const truckCount: number = data.truckCount;
const packages: Pkg[] = data.packages;
const n = packages.length;

function dist(ax: number, ay: number, bx: number, by: number): number {
  const dx = ax - bx, dy = ay - by;
  return Math.sqrt(dx * dx + dy * dy);
}

const dx = depot[0], dy = depot[1];

if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const d0: number[] = new Array(n);
for (let i = 0; i < n; i++) d0[i] = dist(packages[i].x, packages[i].y, dx, dy);

const routeOf: number[] = new Array(n);
const next: number[] = new Array(n).fill(-1);
const prev: number[] = new Array(n).fill(-1);
const head: number[] = new Array(n);
const tail: number[] = new Array(n);
const size: number[] = new Array(n).fill(1);
let numRoutes = n;
for (let i = 0; i < n; i++) { routeOf[i] = i; head[i] = i; tail[i] = i; }

const MAX_FULL = 3000;
interface Saving { i: number; j: number; s: number; }
let savings: Saving[];

if (n <= MAX_FULL) {
  savings = [];
  for (let i = 0; i < n; i++) {
    const pi = packages[i];
    for (let j = i + 1; j < n; j++) {
      const pj = packages[j];
      const s = d0[i] + d0[j] - dist(pi.x, pi.y, pj.x, pj.y);
      if (s > 0) savings.push({ i, j, s });
    }
  }
} else {
  const K = Math.min(30, n - 1);
  savings = [];
  for (let i = 0; i < n; i++) {
    const pi = packages[i];
    const cand: { j: number; d: number }[] = [];
    for (let j = 0; j < n; j++) {
      if (j === i) continue;
      const d = dist(pi.x, pi.y, packages[j].x, packages[j].y);
      cand.push({ j, d });
    }
    cand.sort((a, b) => a.d - b.d);
    for (let k = 0; k < K; k++) {
      const j = cand[k].j;
      if (j > i) {
        const s = d0[i] + d0[j] - cand[k].d;
        if (s > 0) savings.push({ i, j, s });
      }
    }
  }
}

savings.sort((a, b) => b.s - a.s);

for (const sv of savings) {
  if (numRoutes <= truckCount) break;
  const i = sv.i, j = sv.j;
  const ri = routeOf[i], rj = routeOf[j];
  if (ri === rj) continue;
  const iAtHead = head[ri] === i;
  const iAtTail = tail[ri] === i;
  const jAtHead = head[rj] === j;
  const jAtTail = tail[rj] === j;
  if (!(iAtHead || iAtTail)) continue;
  if (!(jAtHead || jAtTail)) continue;

  if (iAtHead && !iAtTail) {
    let cur = head[ri];
    while (cur !== -1) {
      const nx = next[cur];
      next[cur] = prev[cur];
      prev[cur] = nx;
      cur = nx;
    }
    const oldHead = head[ri];
    head[ri] = tail[ri];
    tail[ri] = oldHead;
  }
  if (jAtTail && !jAtHead) {
    let cur = head[rj];
    while (cur !== -1) {
      const nx = next[cur];
      next[cur] = prev[cur];
      prev[cur] = nx;
      cur = nx;
    }
    const oldHead = head[rj];
    head[rj] = tail[rj];
    tail[rj] = oldHead;
  }

  next[tail[ri]] = head[rj];
  prev[head[rj]] = tail[ri];
  tail[ri] = tail[rj];
  size[ri] += size[rj];

  let cur = head[rj];
  while (cur !== -1) {
    routeOf[cur] = ri;
    cur = next[cur];
  }
  numRoutes--;
}

const routeMap = new Map<number, number[]>();
for (let i = 0; i < n; i++) {
  const r = routeOf[i];
  if (!routeMap.has(r)) {
    const arr: number[] = [];
    let cur = head[r];
    while (cur !== -1) {
      arr.push(cur);
      cur = next[cur];
    }
    routeMap.set(r, arr);
  }
}

let routes: number[][] = Array.from(routeMap.values());

while (routes.length > truckCount) {
  routes.sort((a, b) => a.length - b.length);
  const small = routes.shift()!;
  routes[0] = routes[0].concat(small);
}

function twoOpt(route: number[]): number[] {
  const len = route.length;
  if (len < 4) return route;
  const xs = new Array(len + 2);
  const ys = new Array(len + 2);
  xs[0] = dx; ys[0] = dy;
  for (let i = 0; i < len; i++) {
    xs[i + 1] = packages[route[i]].x;
    ys[i + 1] = packages[route[i]].y;
  }
  xs[len + 1] = dx; ys[len + 1] = dy;

  const r = route.slice();
  let improved = true;
  const maxIter = 50;
  let iter = 0;
  while (improved && iter < maxIter) {
    improved = false;
    iter++;
    outer: for (let i = 0; i < len - 1; i++) {
      const a1x = xs[i], a1y = ys[i];
      const a2x = xs[i + 1], a2y = ys[i + 1];
      const dA = dist(a1x, a1y, a2x, a2y);
      for (let k = i + 1; k < len; k++) {
        const b1x = xs[k], b1y = ys[k];
        const b2x = xs[k + 1], b2y = ys[k + 1];
        const dB = dist(b1x, b1y, b2x, b2y);
        const newA = dist(a1x, a1y, b1x, b1y);
        const newB = dist(a2x, a2y, b2x, b2y);
        if (newA + newB + 1e-12 < dA + dB) {
          let lo = i, hi = k - 1;
          while (lo < hi) {
            const tmp = r[lo]; r[lo] = r[hi]; r[hi] = tmp;
            lo++; hi--;
          }
          lo = i + 1; hi = k;
          while (lo < hi) {
            let t = xs[lo]; xs[lo] = xs[hi]; xs[hi] = t;
            t = ys[lo]; ys[lo] = ys[hi]; ys[hi] = t;
            lo++; hi--;
          }
          improved = true;
          break outer;
        }
      }
    }
  }
  return r;
}

const totalLen = routes.reduce((a, r) => a + r.length, 0);
if (totalLen <= 2000) {
  routes = routes.map(r => twoOpt(r));
}

const outRoutes = routes.map(r => r.map(idx => packages[idx].id));
const finalRoutes = outRoutes.filter(r => r.length > 0);

process.stdout.write(JSON.stringify({ routes: finalRoutes }));
