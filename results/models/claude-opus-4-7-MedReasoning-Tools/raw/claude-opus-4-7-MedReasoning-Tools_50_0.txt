#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 radii;
    vec4 weights;
    vec4 bounds;
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) {
        return;
    }

    float dt          = radii.x;
    float sepRadius   = radii.y;
    float alignRadius = radii.z;
    float cohRadius   = radii.w;

    float sepWeight   = weights.x;
    float alignWeight = weights.y;
    float cohWeight   = weights.z;
    float maxSpeed    = weights.w;

    float boundSize  = bounds.x;
    float boundForce = bounds.y;

    vec3 pos = inData[i * 2u].xyz;
    vec3 vel = inData[i * 2u + 1u].xyz;

    vec3 sepAccum   = vec3(0.0);
    vec3 alignAccum = vec3(0.0);
    vec3 cohAccum   = vec3(0.0);

    int sepCount   = 0;
    int alignCount = 0;
    int cohCount   = 0;

    float sepR2   = sepRadius   * sepRadius;
    float alignR2 = alignRadius * alignRadius;
    float cohR2   = cohRadius   * cohRadius;

    for (uint j = 0u; j < N; ++j) {
        if (j == i) {
            continue;
        }
        vec3 otherPos = inData[j * 2u].xyz;
        vec3 otherVel = inData[j * 2u + 1u].xyz;

        vec3 diff = pos - otherPos;
        float dist2 = dot(diff, diff);
        if (dist2 <= 0.0) {
            continue;
        }

        if (dist2 < sepR2) {
            float dist = sqrt(dist2);
            sepAccum += diff / dist;
            sepCount++;
        }
        if (dist2 < alignR2) {
            alignAccum += otherVel;
            alignCount++;
        }
        if (dist2 < cohR2) {
            cohAccum += otherPos;
            cohCount++;
        }
    }

    vec3 sepForce   = vec3(0.0);
    vec3 alignForce = vec3(0.0);
    vec3 cohForce   = vec3(0.0);

    if (sepCount > 0) {
        sepForce = (sepAccum / float(sepCount)) * sepWeight;
    }
    if (alignCount > 0) {
        vec3 avgVel = alignAccum / float(alignCount);
        alignForce = (avgVel - vel) * alignWeight;
    }
    if (cohCount > 0) {
        vec3 center = cohAccum / float(cohCount);
        cohForce = (center - pos) * cohWeight;
    }

    vec3 accel = sepForce + alignForce + cohForce;

    float threshold = boundSize * 0.4;
    float scale     = boundSize * 0.5;
    if (pos.x >  threshold) accel.x -= boundForce * (pos.x / scale);
    if (pos.x < -threshold) accel.x -= boundForce * (pos.x / scale);
    if (pos.y >  threshold) accel.y -= boundForce * (pos.y / scale);
    if (pos.y < -threshold) accel.y -= boundForce * (pos.y / scale);
    if (pos.z >  threshold) accel.z -= boundForce * (pos.z / scale);
    if (pos.z < -threshold) accel.z -= boundForce * (pos.z / scale);

    vel += accel * dt;

    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = (vel / speed) * maxSpeed;
    }

    pos += vel * dt;

    outData[i * 2u]      = vec4(pos, 0.0);
    outData[i * 2u + 1u] = vec4(vel, 0.0);
}
