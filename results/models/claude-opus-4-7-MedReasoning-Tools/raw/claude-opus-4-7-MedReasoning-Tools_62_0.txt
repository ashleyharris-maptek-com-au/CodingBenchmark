declare const require: any;
declare const process: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Town { id: any; x: number; y: number; value: number; }

const W: number = data.width;
const H: number = data.height;
const gridRows: string[] = data.grid;
const ignitions: [number, number][] = data.ignitions;
const towns: Town[] = data.towns;
const cutBudget: number = data.cutBudget;

const blocked = new Uint8Array(W * H);
for (let y = 0; y < H; y++) {
  const row = gridRows[y];
  for (let x = 0; x < W; x++) {
    if (row.charCodeAt(x) !== 46) blocked[y * W + x] = 1;
  }
}

function idx(x: number, y: number) { return y * W + x; }
function inBounds(x: number, y: number) { return x >= 0 && y >= 0 && x < W && y < H; }

function computeBurned(blockedArr: Uint8Array): Uint8Array {
  const burned = new Uint8Array(W * H);
  const queue: number[] = [];
  let head = 0;
  for (const ig of ignitions) {
    const ix = ig[0], iy = ig[1];
    if (!inBounds(ix, iy)) continue;
    const k = idx(ix, iy);
    if (blockedArr[k]) continue;
    if (!burned[k]) { burned[k] = 1; queue.push(k); }
  }
  while (head < queue.length) {
    const k = queue[head++];
    const x = k % W;
    const y = (k - x) / W;
    if (x + 1 < W) { const n = k + 1; if (!blockedArr[n] && !burned[n]) { burned[n] = 1; queue.push(n); } }
    if (x - 1 >= 0) { const n = k - 1; if (!blockedArr[n] && !burned[n]) { burned[n] = 1; queue.push(n); } }
    if (y + 1 < H) { const n = k + W; if (!blockedArr[n] && !burned[n]) { burned[n] = 1; queue.push(n); } }
    if (y - 1 >= 0) { const n = k - W; if (!blockedArr[n] && !burned[n]) { burned[n] = 1; queue.push(n); } }
  }
  return burned;
}

const work = new Uint8Array(blocked);
const cuts: [number, number][] = [];
let budget = cutBudget;

let burned = computeBurned(work);

function neighborsToBlock(x: number, y: number, blk: Uint8Array): number[] {
  const out: number[] = [];
  if (x + 1 < W && !blk[idx(x + 1, y)]) out.push(idx(x + 1, y));
  if (x - 1 >= 0 && !blk[idx(x - 1, y)]) out.push(idx(x - 1, y));
  if (y + 1 < H && !blk[idx(x, y + 1)]) out.push(idx(x, y + 1));
  if (y - 1 >= 0 && !blk[idx(x, y - 1)]) out.push(idx(x, y - 1));
  return out;
}

while (budget > 0) {
  type Cand = { town: Town; cells: number[]; cost: number; ratio: number; };
  const cands: Cand[] = [];
  for (const t of towns) {
    if (!inBounds(t.x, t.y)) continue;
    const tk = idx(t.x, t.y);
    if (!burned[tk]) continue;
    if (work[tk]) continue;
    if (t.value <= 0) continue;
    const ring = neighborsToBlock(t.x, t.y, work);
    if (ring.length === 0) continue;
    if (ring.length > budget) continue;
    cands.push({ town: t, cells: ring, cost: ring.length, ratio: t.value / ring.length });
  }
  if (cands.length === 0) break;
  cands.sort((a, b) => b.ratio - a.ratio);
  const best = cands[0];
  for (const c of best.cells) {
    if (!work[c]) {
      work[c] = 1;
      const x = c % W;
      const y = (c - x) / W;
      cuts.push([x, y]);
    }
  }
  budget -= best.cost;
  burned = computeBurned(work);
}

process.stdout.write(JSON.stringify({ firebreaks: cuts }));
