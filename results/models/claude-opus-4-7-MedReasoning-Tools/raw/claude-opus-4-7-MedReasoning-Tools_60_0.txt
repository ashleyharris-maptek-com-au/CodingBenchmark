'use strict';

const fs = require('fs');
const data = JSON.parse(fs.readFileSync(0, 'utf8'));

const machineCount = data.machineCount | 0;
const tasks = data.tasks || [];
const n = tasks.length;

const idx = new Map();
for (let i = 0; i < n; i++) idx.set(tasks[i].id, i);

const deps = new Array(n);
const children = new Array(n);
for (let i = 0; i < n; i++) { deps[i] = []; children[i] = []; }
for (let i = 0; i < n; i++) {
  const t = tasks[i];
  const d = t.deps || [];
  for (let k = 0; k < d.length; k++) {
    const j = idx.get(d[k]);
    if (j !== undefined) {
      deps[i].push(j);
      children[j].push(i);
    }
  }
}

const indeg = new Array(n).fill(0);
for (let i = 0; i < n; i++) indeg[i] = deps[i].length;
const topo = [];
{
  const q = [];
  for (let i = 0; i < n; i++) if (indeg[i] === 0) q.push(i);
  let head = 0;
  while (head < q.length) {
    const u = q[head++];
    topo.push(u);
    const ch = children[u];
    for (let k = 0; k < ch.length; k++) {
      const v = ch[k];
      if (--indeg[v] === 0) q.push(v);
    }
  }
}

const rank = new Array(n).fill(0);
const downWeight = new Array(n).fill(0);
const minDeadline = new Array(n).fill(Infinity);

for (let p = topo.length - 1; p >= 0; p--) {
  const u = topo[p];
  const t = tasks[u];
  const dur = t.duration || 0;
  const w = t.weight || 0;
  const dl = (t.deadline !== undefined && t.deadline !== null) ? t.deadline : Infinity;
  let maxChild = 0;
  let sumW = w;
  let md = dl;
  const ch = children[u];
  for (let k = 0; k < ch.length; k++) {
    const v = ch[k];
    if (rank[v] > maxChild) maxChild = rank[v];
    sumW += downWeight[v];
    if (minDeadline[v] < md) md = minDeadline[v];
  }
  rank[u] = dur + maxChild;
  downWeight[u] = sumW;
  minDeadline[u] = md;
}

const priority = new Array(n);
for (let i = 0; i < n; i++) {
  const dl = minDeadline[i];
  const dlTerm = (dl === Infinity) ? 0 : -dl;
  priority[i] = rank[i] * 1.0 + downWeight[i] * 0.5 + dlTerm * 0.25;
}

class Heap {
  constructor(cmp) { this.a = []; this.cmp = cmp; }
  push(x) {
    const a = this.a; a.push(x);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.cmp(a[i], a[p]) < 0) { const t = a[i]; a[i] = a[p]; a[p] = t; i = p; }
      else break;
    }
  }
  pop() {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0]; const last = a.pop();
    if (a.length > 0) {
      a[0] = last;
      let i = 0; const n2 = a.length;
      while (true) {
        const l = 2 * i + 1, r = l + 1;
        let s = i;
        if (l < n2 && this.cmp(a[l], a[s]) < 0) s = l;
        if (r < n2 && this.cmp(a[r], a[s]) < 0) s = r;
        if (s !== i) { const t = a[i]; a[i] = a[s]; a[s] = t; i = s; }
        else break;
      }
    }
    return top;
  }
  size() { return this.a.length; }
}

const ready = new Heap((x, y) => y.p - x.p);
for (let i = 0; i < n; i++) {
  if (remainingInit(i) === 0) ready.push({ i, p: priority[i] });
}
function remainingInit(i) { return deps[i].length; }

const machineFree = new Array(machineCount).fill(0);
const finish = new Array(n).fill(0);
const remaining = new Array(n);
for (let i = 0; i < n; i++) remaining[i] = deps[i].length;

const schedule = [];
let scheduled = 0;

while (scheduled < n) {
  if (ready.size() === 0) break;
  let bestM = 0;
  for (let m = 1; m < machineCount; m++) if (machineFree[m] < machineFree[bestM]) bestM = m;

  const item = ready.pop();
  const u = item.i;
  const t = tasks[u];
  let earliest = machineFree[bestM];
  const dps = deps[u];
  for (let k = 0; k < dps.length; k++) {
    const f = finish[dps[k]];
    if (f > earliest) earliest = f;
  }
  const start = earliest;
  const end = start + (t.duration || 0);
  finish[u] = end;
  machineFree[bestM] = end;
  schedule.push({ id: t.id, start: start, machine: bestM });
  scheduled++;

  const ch = children[u];
  for (let k = 0; k < ch.length; k++) {
    const v = ch[k];
    if (--remaining[v] === 0) ready.push({ i: v, p: priority[v] });
  }
}

process.stdout.write(JSON.stringify({ schedule: schedule }));
