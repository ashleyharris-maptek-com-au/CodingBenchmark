#include <bits/stdc++.h>
using namespace std;

int n, m;
vector<vector<int>> adj;
vector<double> c;

struct Solver {
    vector<int> nodes;
    vector<char> alive;
    vector<char> chosen;
    double compBest;
    vector<char> compBestChoice;

    int pickVertex() {
        int best = -1, bestDeg = -1;
        for (int v : nodes) {
            if (!alive[v]) continue;
            int d = 0;
            for (int u : adj[v]) if (alive[u]) d++;
            if (d > bestDeg) { bestDeg = d; best = v; }
        }
        return best;
    }

    void recurse(double curSum) {
        vector<pair<int,int>> undoAlive;
        vector<pair<int,int>> undoChosen;
        auto setAlive = [&](int v, int s) {
            undoAlive.push_back({v, alive[v]});
            alive[v] = s;
        };
        auto setChosen = [&](int v, int s) {
            undoChosen.push_back({v, chosen[v]});
            chosen[v] = s;
        };

        bool changed = true;
        double addSum = 0;
        while (changed) {
            changed = false;
            for (int v : nodes) {
                if (!alive[v]) continue;
                int deg = 0; int singleN = -1;
                for (int u : adj[v]) if (alive[u]) { deg++; singleN = u; if (deg>1) break; }
                if (deg == 0) {
                    if (c[v] > 0) { setChosen(v, 1); addSum += c[v]; }
                    else { setChosen(v, 0); }
                    setAlive(v, 0);
                    changed = true;
                } else if (deg == 1) {
                    int u = singleN;
                    if (c[v] >= c[u]) {
                        if (c[v] >= 0) {
                            setChosen(v, 1); addSum += c[v];
                            setAlive(v, 0);
                            setChosen(u, 0); setAlive(u, 0);
                        } else {
                            setChosen(v, 0); setAlive(v, 0);
                            setChosen(u, 0); setAlive(u, 0);
                        }
                        changed = true;
                    }
                }
            }
        }

        curSum += addSum;

        double ub = curSum;
        for (int v : nodes) if (alive[v] && c[v] > 0) ub += c[v];

        if (ub <= compBest + 1e-12) {
            for (int i = (int)undoChosen.size()-1; i >= 0; i--) chosen[undoChosen[i].first] = undoChosen[i].second;
            for (int i = (int)undoAlive.size()-1; i >= 0; i--) alive[undoAlive[i].first] = undoAlive[i].second;
            return;
        }

        int v = pickVertex();
        if (v == -1) {
            if (curSum > compBest) {
                compBest = curSum;
                for (int w : nodes) compBestChoice[w] = chosen[w];
            }
            for (int i = (int)undoChosen.size()-1; i >= 0; i--) chosen[undoChosen[i].first] = undoChosen[i].second;
            for (int i = (int)undoAlive.size()-1; i >= 0; i--) alive[undoAlive[i].first] = undoAlive[i].second;
            return;
        }

        vector<pair<int,int>> b1Alive, b1Chosen;
        auto setAlive2 = [&](int x, int s) { b1Alive.push_back({x, alive[x]}); alive[x] = s; };
        auto setChosen2 = [&](int x, int s) { b1Chosen.push_back({x, chosen[x]}); chosen[x] = s; };

        setChosen2(v, 1);
        setAlive2(v, 0);
        for (int u : adj[v]) if (alive[u]) { setChosen2(u, 0); setAlive2(u, 0); }
        recurse(curSum + c[v]);
        for (int i = (int)b1Chosen.size()-1; i >= 0; i--) chosen[b1Chosen[i].first] = b1Chosen[i].second;
        for (int i = (int)b1Alive.size()-1; i >= 0; i--) alive[b1Alive[i].first] = b1Alive[i].second;

        b1Alive.clear(); b1Chosen.clear();
        setChosen2(v, 0);
        setAlive2(v, 0);
        recurse(curSum);
        for (int i = (int)b1Chosen.size()-1; i >= 0; i--) chosen[b1Chosen[i].first] = b1Chosen[i].second;
        for (int i = (int)b1Alive.size()-1; i >= 0; i--) alive[b1Alive[i].first] = b1Alive[i].second;

        for (int i = (int)undoChosen.size()-1; i >= 0; i--) chosen[undoChosen[i].first] = undoChosen[i].second;
        for (int i = (int)undoAlive.size()-1; i >= 0; i--) alive[undoAlive[i].first] = undoAlive[i].second;
    }
};

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cin >> n >> m;
    c.assign(n, 0);
    for (int i = 0; i < n; i++) cin >> c[i];
    adj.assign(n, {});
    for (int i = 0; i < m; i++) {
        int u, v; cin >> u >> v;
        if (u == v) continue;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    for (int i = 0; i < n; i++) {
        sort(adj[i].begin(), adj[i].end());
        adj[i].erase(unique(adj[i].begin(), adj[i].end()), adj[i].end());
    }

    vector<int> assign(n, 0);
    double totalObj = 0;

    vector<int> comp(n, -1);
    for (int s = 0; s < n; s++) {
        if (comp[s] != -1) continue;
        vector<int> nodes;
        queue<int> q; q.push(s); comp[s] = s;
        while (!q.empty()) {
            int u = q.front(); q.pop();
            nodes.push_back(u);
            for (int v : adj[u]) if (comp[v] == -1) { comp[v] = s; q.push(v); }
        }

        Solver sv;
        sv.nodes = nodes;
        sv.alive.assign(n, 0);
        sv.chosen.assign(n, 0);
        for (int v : nodes) sv.alive[v] = 1;
        sv.compBest = -1e18;
        sv.compBestChoice.assign(n, 0);

        vector<int> order = nodes;
        sort(order.begin(), order.end(), [&](int a, int b){ return c[a] > c[b]; });
        vector<char> taken(n, 0), blocked(n, 0);
        double greedy = 0;
        for (int v : order) {
            if (c[v] <= 0) break;
            if (blocked[v]) continue;
            taken[v] = 1; greedy += c[v];
            for (int u : adj[v]) blocked[u] = 1;
        }
        sv.compBest = greedy;
        for (int v : nodes) sv.compBestChoice[v] = taken[v];

        sv.recurse(0);

        totalObj += sv.compBest;
        for (int v : nodes) assign[v] = sv.compBestChoice[v];
    }

    bool allInt = true;
    for (int i = 0; i < n; i++) {
        if (fabs(c[i] - round(c[i])) > 1e-9) { allInt = false; break; }
    }
    if (allInt) {
        cout << (long long)llround(totalObj) << "\n";
    } else {
        cout << setprecision(15) << totalObj << "\n";
    }
    for (int i = 0; i < n; i++) {
        cout << (int)assign[i];
        if (i+1 < n) cout << ' ';
    }
    cout << "\n";
    return 0;
}
