use std::io::{self, Read, Write, BufWriter};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let bytes = input.as_bytes();
    let mut i = 0usize;
    let len = bytes.len();

    while i < len && (bytes[i] == b' ' || bytes[i] == b'\r' || bytes[i] == b'\n' || bytes[i] == b'\t') { i += 1; }
    let mut h: usize = 0;
    while i < len && bytes[i] >= b'0' && bytes[i] <= b'9' {
        h = h * 10 + (bytes[i] - b'0') as usize;
        i += 1;
    }
    while i < len && (bytes[i] == b' ' || bytes[i] == b'\t') { i += 1; }
    let mut w: usize = 0;
    while i < len && bytes[i] >= b'0' && bytes[i] <= b'9' {
        w = w * 10 + (bytes[i] - b'0') as usize;
        i += 1;
    }
    while i < len && bytes[i] != b'\n' { i += 1; }
    if i < len { i += 1; }

    if h == 0 || w == 0 {
        writeln!(out, "0").unwrap();
        return;
    }

    let total = h * w;
    let mut grid: Vec<u8> = vec![b'#'; total];
    let mut start: usize = usize::MAX;
    let mut goal: usize = usize::MAX;

    for r in 0..h {
        let row_start = r * w;
        let mut c = 0usize;
        while c < w && i < len && bytes[i] != b'\n' && bytes[i] != b'\r' {
            let ch = bytes[i];
            grid[row_start + c] = ch;
            if ch == b'A' { start = row_start + c; }
            else if ch == b'B' { goal = row_start + c; }
            c += 1;
            i += 1;
        }
        while i < len && bytes[i] != b'\n' { i += 1; }
        if i < len { i += 1; }
    }

    if start == usize::MAX || goal == usize::MAX {
        writeln!(out, "0").unwrap();
        return;
    }

    if start == goal {
        writeln!(out, "1").unwrap();
        let x = start % w;
        let y = start / w;
        writeln!(out, "{} {}", x, y).unwrap();
        return;
    }

    // parent_dir: 0=unvisited, 1=came from up, 2=from down, 3=from left, 4=from right, 5=start
    let mut parent_dir: Vec<u8> = vec![0u8; total];
    parent_dir[start] = 5;

    let mut queue: Vec<usize> = Vec::with_capacity(64);
    queue.push(start);
    let mut head = 0usize;

    let mut found = false;
    while head < queue.len() {
        let idx = queue[head];
        head += 1;
        if idx == goal { found = true; break; }

        let r = idx / w;
        let c = idx - r * w;

        if r > 0 {
            let n = idx - w;
            if parent_dir[n] == 0 && grid[n] != b'#' {
                parent_dir[n] = 2;
                queue.push(n);
            }
        }
        if r + 1 < h {
            let n = idx + w;
            if parent_dir[n] == 0 && grid[n] != b'#' {
                parent_dir[n] = 1;
                queue.push(n);
            }
        }
        if c > 0 {
            let n = idx - 1;
            if parent_dir[n] == 0 && grid[n] != b'#' {
                parent_dir[n] = 4;
                queue.push(n);
            }
        }
        if c + 1 < w {
            let n = idx + 1;
            if parent_dir[n] == 0 && grid[n] != b'#' {
                parent_dir[n] = 3;
                queue.push(n);
            }
        }
    }

    if !found {
        writeln!(out, "0").unwrap();
        return;
    }

    let mut path: Vec<usize> = Vec::new();
    let mut cur = goal;
    loop {
        path.push(cur);
        if cur == start { break; }
        let d = parent_dir[cur];
        match d {
            1 => cur -= w,
            2 => cur += w,
            3 => cur -= 1,
            4 => cur += 1,
            _ => break,
        }
    }
    path.reverse();

    writeln!(out, "{}", path.len()).unwrap();
    let mut buf = String::with_capacity(path.len() * 12);
    for &p in &path {
        let x = p % w;
        let y = p / w;
        buf.push_str(&x.to_string());
        buf.push(' ');
        buf.push_str(&y.to_string());
        buf.push('\n');
    }
    out.write_all(buf.as_bytes()).unwrap();
}
