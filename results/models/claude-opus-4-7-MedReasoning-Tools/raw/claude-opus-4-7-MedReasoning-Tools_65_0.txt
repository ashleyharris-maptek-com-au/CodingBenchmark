declare const require: any;
declare const process: any;

interface Task {
  id: number | string;
  duration: number;
  deps: (number | string)[];
  deadline: number;
  weight: number;
}

interface Input {
  machineCount: number;
  tasks: Task[];
}

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data: Input = JSON.parse(input);

const machineCount: number = Math.max(1, (data.machineCount | 0) || 1);
const tasks: Task[] = data.tasks || [];
const n = tasks.length;

const idToIdx = new Map<any, number>();
for (let i = 0; i < n; i++) idToIdx.set(tasks[i].id, i);

const deps: number[][] = new Array(n);
const succ: number[][] = new Array(n);
const indeg: number[] = new Array(n).fill(0);
for (let i = 0; i < n; i++) { deps[i] = []; succ[i] = []; }
for (let i = 0; i < n; i++) {
  const dlist = tasks[i].deps || [];
  for (const d of dlist) {
    const j = idToIdx.get(d);
    if (j === undefined) continue;
    deps[i].push(j);
    succ[j].push(i);
    indeg[i]++;
  }
}

// Topological order
const order: number[] = [];
{
  const q: number[] = [];
  const tmpIn = indeg.slice();
  for (let i = 0; i < n; i++) if (tmpIn[i] === 0) q.push(i);
  let head = 0;
  while (head < q.length) {
    const u = q[head++];
    order.push(u);
    for (const v of succ[u]) if (--tmpIn[v] === 0) q.push(v);
  }
}
if (order.length < n) {
  const seen = new Set(order);
  for (let i = 0; i < n; i++) if (!seen.has(i)) order.push(i);
}

// Critical path length (including own duration)
const cp: number[] = new Array(n).fill(0);
for (let k = order.length - 1; k >= 0; k--) {
  const u = order[k];
  let best = 0;
  for (const v of succ[u]) if (cp[v] > best) best = cp[v];
  cp[u] = best + (tasks[u].duration || 0);
}

// Earliest start ignoring machine contention
const est: number[] = new Array(n).fill(0);
for (const u of order) {
  let e = 0;
  for (const p of deps[u]) {
    const f = est[p] + (tasks[p].duration || 0);
    if (f > e) e = f;
  }
  est[u] = e;
}

// Sum of successor weights downstream (weighted critical-path style heuristic)
const downWeight: number[] = new Array(n).fill(0);
for (let k = order.length - 1; k >= 0; k--) {
  const u = order[k];
  let s = (tasks[u].weight || 0);
  for (const v of succ[u]) s += downWeight[v];
  downWeight[u] = s;
}

function priority(i: number): number {
  const t = tasks[i];
  const w = t.weight || 0;
  const d = t.duration || 1;
  const dl = (typeof t.deadline === 'number' && isFinite(t.deadline)) ? t.deadline : 1e15;
  const slack = dl - est[i] - (t.duration || 0);
  return cp[i] * 2 + downWeight[i] * 1.5 + w * 3 + (w / d) * 5 - slack * 0.2;
}

class MinHeap<T> {
  arr: T[] = [];
  cmp: (a: T, b: T) => number;
  constructor(cmp: (a: T, b: T) => number) { this.cmp = cmp; }
  size() { return this.arr.length; }
  push(x: T) {
    this.arr.push(x);
    let i = this.arr.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.cmp(this.arr[i], this.arr[p]) < 0) {
        const t = this.arr[i]; this.arr[i] = this.arr[p]; this.arr[p] = t;
        i = p;
      } else break;
    }
  }
  pop(): T | undefined {
    if (this.arr.length === 0) return undefined;
    const top = this.arr[0];
    const last = this.arr.pop()!;
    if (this.arr.length > 0) {
      this.arr[0] = last;
      let i = 0;
      const sz = this.arr.length;
      while (true) {
        const l = 2 * i + 1, r = 2 * i + 2;
        let best = i;
        if (l < sz && this.cmp(this.arr[l], this.arr[best]) < 0) best = l;
        if (r < sz && this.cmp(this.arr[r], this.arr[best]) < 0) best = r;
        if (best !== i) {
          const t = this.arr[i]; this.arr[i] = this.arr[best]; this.arr[best] = t;
          i = best;
        } else break;
      }
    }
    return top;
  }
}

const finishTime: number[] = new Array(n).fill(0);
const remainingIn: number[] = indeg.slice();

const ready = new MinHeap<number>((a, b) => priority(b) - priority(a));
for (let i = 0; i < n; i++) if (remainingIn[i] === 0) ready.push(i);

const machHeap = new MinHeap<{ time: number; m: number }>((a, b) => a.time - b.time || a.m - b.m);
for (let m = 0; m < machineCount; m++) machHeap.push({ time: 0, m });

function earliestForDeps(i: number): number {
  let e = 0;
  for (const p of deps[i]) if (finishTime[p] > e) e = finishTime[p];
  return e;
}

const result: { id: any; start: number; machine: number }[] = [];

while (ready.size() > 0) {
  const machEvt = machHeap.pop()!;
  const m = machEvt.m;
  const mTime = machEvt.time;
  const best = ready.pop()!;
  const depReady = earliestForDeps(best);
  const start = Math.max(mTime, depReady);
  const dur = tasks[best].duration || 0;
  const fin = start + dur;
  finishTime[best] = fin;
  result.push({ id: tasks[best].id, start, machine: m });
  machHeap.push({ time: fin, m });
  for (const v of succ[best]) {
    if (--remainingIn[v] === 0) ready.push(v);
  }
}

process.stdout.write(JSON.stringify({ schedule: result }));
