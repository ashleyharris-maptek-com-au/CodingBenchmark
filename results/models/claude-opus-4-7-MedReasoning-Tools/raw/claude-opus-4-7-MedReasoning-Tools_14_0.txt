using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class MinesweeperSolver
{
    static int H, W;
    static byte[] grid;
    static int[] dx = { -1, 0, 1, -1, 1, -1, 0, 1 };
    static int[] dy = { -1, -1, -1, 0, 0, 1, 1, 1 };

    const byte UNKNOWN = 9;
    const byte MINE = 10;

    static int Idx(int x, int y) => y * W + x;

    static void Main()
    {
        var sin = Console.In;
        string firstLine = sin.ReadLine();
        if (string.IsNullOrEmpty(firstLine)) { Console.WriteLine(0); return; }
        var parts = firstLine.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        H = int.Parse(parts[0]);
        W = int.Parse(parts[1]);
        long total = (long)H * W;
        grid = new byte[total];

        for (int y = 0; y < H; y++)
        {
            string line = sin.ReadLine() ?? "";
            if (line.Length < W) line = line.PadRight(W, ' ');
            for (int x = 0; x < W; x++)
            {
                char c = line[x];
                if (c == ' ') grid[Idx(x, y)] = UNKNOWN;
                else if (c == '*') grid[Idx(x, y)] = MINE;
                else if (c >= '0' && c <= '8') grid[Idx(x, y)] = (byte)(c - '0');
                else grid[Idx(x, y)] = UNKNOWN;
            }
        }

        byte[] safeMask = new byte[total];
        byte[] mineMask = new byte[total];

        for (long i = 0; i < total; i++)
            if (grid[i] == MINE) mineMask[i] = 1;

        long boardSize = total;
        bool useSubsetReasoning = boardSize <= 2_000_000;

        SimplePropagate(safeMask, mineMask);

        if (useSubsetReasoning)
        {
            bool outerChanged = true;
            int subsetPasses = 0;
            while (outerChanged && subsetPasses < 10)
            {
                outerChanged = false;
                subsetPasses++;

                var consCells = new List<int>();
                var consMines = new List<int>();
                var consNbrs = new List<int[]>();

                for (int y = 0; y < H; y++)
                {
                    for (int x = 0; x < W; x++)
                    {
                        int idx = Idx(x, y);
                        byte v = grid[idx];
                        if (v > 8) continue;
                        int flagged = 0;
                        var nbrs = new List<int>(8);
                        for (int k = 0; k < 8; k++)
                        {
                            int nx = x + dx[k], ny = y + dy[k];
                            if (nx < 0 || ny < 0 || nx >= W || ny >= H) continue;
                            int ni = Idx(nx, ny);
                            if (mineMask[ni] == 1) flagged++;
                            else if (grid[ni] == UNKNOWN && safeMask[ni] == 0) nbrs.Add(ni);
                        }
                        if (nbrs.Count == 0) continue;
                        nbrs.Sort();
                        consCells.Add(idx);
                        consMines.Add(v - flagged);
                        consNbrs.Add(nbrs.ToArray());
                    }
                }

                int M = consCells.Count;
                if (M == 0) break;

                var cellToCons = new Dictionary<int, List<int>>();
                for (int i = 0; i < M; i++)
                {
                    foreach (int c in consNbrs[i])
                    {
                        if (!cellToCons.TryGetValue(c, out var lst))
                        {
                            lst = new List<int>();
                            cellToCons[c] = lst;
                        }
                        lst.Add(i);
                    }
                }

                bool localChanged = false;
                var visitedPair = new HashSet<long>();
                for (int i = 0; i < M; i++)
                {
                    var nbrsI = consNbrs[i];
                    var candidates = new HashSet<int>();
                    foreach (int c in nbrsI)
                        foreach (int j in cellToCons[c])
                            if (j != i) candidates.Add(j);

                    foreach (int j in candidates)
                    {
                        long key = ((long)Math.Min(i, j) << 32) | (uint)Math.Max(i, j);
                        if (!visitedPair.Add(key)) continue;
                        var nbrsJ = consNbrs[j];
                        bool iSubJ = IsSubset(nbrsI, nbrsJ);
                        bool jSubI = !iSubJ && IsSubset(nbrsJ, nbrsI);
                        if (!iSubJ && !jSubI) continue;
                        int aIdx = iSubJ ? i : j;
                        int bIdx = iSubJ ? j : i;
                        int diffCount = consNbrs[bIdx].Length - consNbrs[aIdx].Length;
                        if (diffCount == 0) continue;
                        int diffMines = consMines[bIdx] - consMines[aIdx];
                        if (diffMines < 0) continue;
                        if (diffMines == 0)
                        {
                            foreach (int c in consNbrs[bIdx])
                            {
                                if (Array.BinarySearch(consNbrs[aIdx], c) < 0)
                                {
                                    if (safeMask[c] == 0 && mineMask[c] == 0)
                                    {
                                        safeMask[c] = 1;
                                        localChanged = true;
                                    }
                                }
                            }
                        }
                        else if (diffMines == diffCount)
                        {
                            foreach (int c in consNbrs[bIdx])
                            {
                                if (Array.BinarySearch(consNbrs[aIdx], c) < 0)
                                {
                                    if (safeMask[c] == 0 && mineMask[c] == 0)
                                    {
                                        mineMask[c] = 1;
                                        localChanged = true;
                                    }
                                }
                            }
                        }
                    }
                }

                if (localChanged)
                {
                    outerChanged = true;
                    SimplePropagate(safeMask, mineMask);
                }
            }
        }

        var output = new StringBuilder();
        long count = 0;
        for (int y = 0; y < H; y++)
        {
            for (int x = 0; x < W; x++)
            {
                int idx = Idx(x, y);
                if (safeMask[idx] == 1 && grid[idx] == UNKNOWN && mineMask[idx] == 0)
                {
                    output.Append(x).Append(' ').Append(y).Append('\n');
                    count++;
                }
            }
        }
        var sb = new StringBuilder();
        sb.Append(count).Append('\n').Append(output);
        Console.Out.Write(sb.ToString());
    }

    static void SimplePropagate(byte[] safeMask, byte[] mineMask)
    {
        bool changed = true;
        while (changed)
        {
            changed = false;
            for (int y = 0; y < H; y++)
            {
                for (int x = 0; x < W; x++)
                {
                    int idx = Idx(x, y);
                    byte v = grid[idx];
                    if (v > 8) continue;
                    int n = v;
                    int flagged = 0;
                    int unknownCount = 0;
                    for (int k = 0; k < 8; k++)
                    {
                        int nx = x + dx[k], ny = y + dy[k];
                        if (nx < 0 || ny < 0 || nx >= W || ny >= H) continue;
                        int ni = Idx(nx, ny);
                        if (mineMask[ni] == 1) flagged++;
                        else if (grid[ni] == UNKNOWN && safeMask[ni] == 0) unknownCount++;
                    }
                    if (unknownCount == 0) continue;
                    int remaining = n - flagged;
                    if (remaining == 0)
                    {
                        for (int k = 0; k < 8; k++)
                        {
                            int nx = x + dx[k], ny = y + dy[k];
                            if (nx < 0 || ny < 0 || nx >= W || ny >= H) continue;
                            int ni = Idx(nx, ny);
                            if (grid[ni] == UNKNOWN && mineMask[ni] == 0 && safeMask[ni] == 0)
                            { safeMask[ni] = 1; changed = true; }
                        }
                    }
                    else if (remaining == unknownCount)
                    {
                        for (int k = 0; k < 8; k++)
                        {
                            int nx = x + dx[k], ny = y + dy[k];
                            if (nx < 0 || ny < 0 || nx >= W || ny >= H) continue;
                            int ni = Idx(nx, ny);
                            if (grid[ni] == UNKNOWN && mineMask[ni] == 0 && safeMask[ni] == 0)
                            { mineMask[ni] = 1; changed = true; }
                        }
                    }
                }
            }
        }
    }

    static bool IsSubset(int[] a, int[] b)
    {
        if (a.Length > b.Length) return false;
        int i = 0, j = 0;
        while (i < a.Length && j < b.Length)
        {
            if (a[i] == b[j]) { i++; j++; }
            else if (a[i] > b[j]) j++;
            else return false;
        }
        return i == a.Length;
    }
}
