               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color %gl_out %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
     %fnvoid = OpTypeFunction %void
      %float = OpTypeFloat 32
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4

      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
     %uint_2 = OpConstant %uint 2
    %f_0     = OpConstant %float 0.0
    %f_1     = OpConstant %float 1.0
    %f_thick = OpConstant %float 0.03

%gl_PerVertex_in  = OpTypeStruct %v4float
%arr_pv_in        = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv    = OpTypePointer Input %arr_pv_in
       %gl_in    = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4        = OpTypePointer Input %v4float

%arr_v3_in        = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3    = OpTypePointer Input %arr_v3_in
       %color    = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3        = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv       = OpTypePointer Output %gl_PerVertex_out
       %gl_out   = OpVariable %ptr_out_pv Output
%ptr_out_v4       = OpTypePointer Output %v4float

%ptr_out_v3       = OpTypePointer Output %v3float
     %outColor   = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fnvoid
      %entry = OpLabel

      ; Read p0 and p1
      %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
      %p0     = OpLoad %v4float %p0_ptr
      %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
      %p1     = OpLoad %v4float %p1_ptr

      ; Read colors
      %c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
      %c0     = OpLoad %v3float %c0_ptr
      %c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
      %c1     = OpLoad %v3float %c1_ptr

      ; dir.xy = p1.xy - p0.xy
      %p0x = OpCompositeExtract %float %p0 0
      %p0y = OpCompositeExtract %float %p0 1
      %p1x = OpCompositeExtract %float %p1 0
      %p1y = OpCompositeExtract %float %p1 1
      %dx  = OpFSub %float %p1x %p0x
      %dy  = OpFSub %float %p1y %p0y

      ; perp_raw = (-dy, dx)
      %ndy = OpFNegate %float %dy
      %perp_raw = OpCompositeConstruct %v2float %ndy %dx

      ; perp = normalize(perp_raw) * thickness
      %perp_n = OpExtInst %v2float %1 Normalize %perp_raw
      %perp   = OpVectorTimesScalar %v2float %perp_n %f_thick
      %perpx  = OpCompositeExtract %float %perp 0
      %perpy  = OpCompositeExtract %float %perp 1

      ; Extract p0/p1 z and w
      %p0z = OpCompositeExtract %float %p0 2
      %p0w = OpCompositeExtract %float %p0 3
      %p1z = OpCompositeExtract %float %p1 2
      %p1w = OpCompositeExtract %float %p1 3

      ; v0 = p0 + (perp.x, perp.y, 0, 0), color = c0
      %v0x = OpFAdd %float %p0x %perpx
      %v0y = OpFAdd %float %p0y %perpy
      %v0  = OpCompositeConstruct %v4float %v0x %v0y %p0z %p0w
      %gp0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %gp0_ptr %v0
               OpStore %outColor %c0
               OpEmitVertex

      ; v1 = p0 - perp, color = c0
      %v1x = OpFSub %float %p0x %perpx
      %v1y = OpFSub %float %p0y %perpy
      %v1  = OpCompositeConstruct %v4float %v1x %v1y %p0z %p0w
      %gp1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %gp1_ptr %v1
               OpStore %outColor %c0
               OpEmitVertex

      ; v2 = p1 + perp, color = c1
      %v2x = OpFAdd %float %p1x %perpx
      %v2y = OpFAdd %float %p1y %perpy
      %v2  = OpCompositeConstruct %v4float %v2x %v2y %p1z %p1w
      %gp2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %gp2_ptr %v2
               OpStore %outColor %c1
               OpEmitVertex

      ; v3 = p1 - perp, color = c1
      %v3x = OpFSub %float %p1x %perpx
      %v3y = OpFSub %float %p1y %perpy
      %v3  = OpCompositeConstruct %v4float %v3x %v3y %p1z %p1w
      %gp3_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %gp3_ptr %v3
               OpStore %outColor %c1
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
