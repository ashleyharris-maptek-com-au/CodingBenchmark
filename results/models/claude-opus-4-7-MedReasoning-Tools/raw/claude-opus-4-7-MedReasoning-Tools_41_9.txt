; SPIR-V
; Version: 1.0
               OpCapability Shader
       %glsl = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %color Location 4
               OpDecorate %outColor Location 0

               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

       %void = OpTypeVoid
     %fnVoid = OpTypeFunction %void
      %float = OpTypeFloat 32
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
       %vec2 = OpTypeVector %float 2
       %vec3 = OpTypeVector %float 3
       %vec4 = OpTypeVector %float 4
       %mat4 = OpTypeMatrix %vec4 4
       %bool = OpTypeBool

        %UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
   %_ptr_UBO = OpTypePointer Uniform %UBO
        %ubo = OpVariable %_ptr_UBO Uniform
%_ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Input_vec2 = OpTypePointer Input %vec2
%_ptr_Output_vec4 = OpTypePointer Output %vec4

   %worldPos = OpVariable %_ptr_Input_vec3 Input
     %normal = OpVariable %_ptr_Input_vec3 Input
         %uv = OpVariable %_ptr_Input_vec2 Input
    %tangent = OpVariable %_ptr_Input_vec3 Input
      %color = OpVariable %_ptr_Input_vec3 Input
   %outColor = OpVariable %_ptr_Output_vec4 Output

       %f_0 = OpConstant %float 0.0
       %f_1 = OpConstant %float 1.0
     %f_0_5 = OpConstant %float 0.5
     %f_0_9 = OpConstant %float 0.9
     %f_0_2 = OpConstant %float 0.2
       %f_8 = OpConstant %float 8.0
       %i_3 = OpConstant %int 3
       %u_2 = OpConstant %uint 2
       %u_0 = OpConstant %uint 0

       %main = OpFunction %void None %fnVoid
      %label = OpLabel

       %nrm0 = OpLoad %vec3 %normal
        %nrm = OpExtInst %vec3 %glsl Normalize %nrm0
       %wpos = OpLoad %vec3 %worldPos

   %lightPtr = OpAccessChain %_ptr_Uniform_vec4 %ubo %i_3
     %light4 = OpLoad %vec4 %lightPtr
      %light = OpVectorShuffle %vec3 %light4 %light4 0 1 2

      %ldir0 = OpFSub %vec3 %light %wpos
       %ldir = OpExtInst %vec3 %glsl Normalize %ldir0

      %ndotl = OpDot %float %nrm %ldir
       %diff = OpExtInst %float %glsl FMax %ndotl %f_0
     %scaled = OpFMul %float %diff %f_0_5
      %shade = OpFAdd %float %scaled %f_0_5

        %uvv = OpLoad %vec2 %uv
        %uvs = OpVectorTimesScalar %vec2 %uvv %f_8
        %uvf = OpExtInst %vec2 %glsl Floor %uvs
         %ux = OpCompositeExtract %float %uvf 0
         %uy = OpCompositeExtract %float %uvf 1
        %sum = OpFAdd %float %ux %uy
       %sumi = OpConvertFToS %int %sum
       %sumu = OpBitcast %uint %sumi
        %mod = OpUMod %uint %sumu %u_2
       %iseq = OpIEqual %bool %mod %u_0
       %cval = OpSelect %float %iseq %f_0_9 %f_0_2

        %lit = OpFMul %float %cval %shade
        %r   = OpFMul %float %lit %f_1
        %out4 = OpCompositeConstruct %vec4 %r %r %r %f_1
               OpStore %outColor %out4
               OpReturn
               OpFunctionEnd
