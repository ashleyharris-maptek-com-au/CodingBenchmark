using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

class Body
{
    public string Name;
    public double Mass;
    public double[] R = new double[3];
    public double[] V = new double[3];
}

class Program
{
    const double G = 6.67430e-11;
    const double DAY = 86400.0;

    static List<Body> Bodies;
    static int AsteroidIdx, EarthIdx, SunIdx;

    static void Main()
    {
        var sin = Console.In.ReadToEnd();
        var toks = sin.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int p = 0;
        int numBodies = int.Parse(toks[p++], CultureInfo.InvariantCulture);
        double warningDays = double.Parse(toks[p++], CultureInfo.InvariantCulture);
        double impactorMass = double.Parse(toks[p++], CultureInfo.InvariantCulture);
        double dvBudget = double.Parse(toks[p++], CultureInfo.InvariantCulture);
        double asteroidMass = double.Parse(toks[p++], CultureInfo.InvariantCulture);
        double t0Unix = double.Parse(toks[p++], CultureInfo.InvariantCulture);

        Bodies = new List<Body>();
        for (int i = 0; i < numBodies; i++)
        {
            var b = new Body();
            b.Name = toks[p++];
            b.Mass = double.Parse(toks[p++], CultureInfo.InvariantCulture);
            for (int k = 0; k < 3; k++) b.R[k] = double.Parse(toks[p++], CultureInfo.InvariantCulture);
            for (int k = 0; k < 3; k++) b.V[k] = double.Parse(toks[p++], CultureInfo.InvariantCulture);
            Bodies.Add(b);
        }

        AsteroidIdx = Bodies.FindIndex(b => b.Name == "Asteroid");
        EarthIdx = Bodies.FindIndex(b => b.Name == "Earth");
        SunIdx = Bodies.FindIndex(b => b.Name == "Sun");

        double impactTimeSec = FindEarthImpactTime(warningDays * DAY * 1.5);
        double interceptTimeSec = impactTimeSec * 0.7;
        if (interceptTimeSec < 20 * DAY) interceptTimeSec = Math.Max(impactTimeSec * 0.5, impactTimeSec - 5 * DAY);
        if (interceptTimeSec <= 0) interceptTimeSec = Math.Max(impactTimeSec * 0.5, 1 * DAY);

        double maxLaunchSec = Math.Min(warningDays * DAY, interceptTimeSec - 5 * DAY);
        if (maxLaunchSec < 0) maxLaunchSec = 0;

        var astStateInt = PropagateBodyTo(AsteroidIdx, interceptTimeSec);

        double bestLaunch = 0;
        double bestDv = double.MaxValue;
        double[] bestDvVec = new double[] { 0, 0, 0 };
        double muSun = G * (SunIdx >= 0 ? Bodies[SunIdx].Mass : 1.989e30);

        int N = 50;
        for (int i = 0; i <= N; i++)
        {
            double t = maxLaunchSec * i / N;
            double tof = interceptTimeSec - t;
            if (tof < 3 * DAY) continue;
            var earthState = PropagateBodyTo(EarthIdx, t);
            double[] sunR0 = SunIdx >= 0 ? PropagateBodyTo(SunIdx, t).Item1 : new double[3];
            double[] sunV0 = SunIdx >= 0 ? PropagateBodyTo(SunIdx, t).Item2 : new double[3];
            double[] sunRI = SunIdx >= 0 ? PropagateBodyTo(SunIdx, interceptTimeSec).Item1 : new double[3];
            double[] r1 = Sub(earthState.Item1, sunR0);
            double[] r2 = Sub(astStateInt.Item1, sunRI);
            double[] v1, v2;
            bool ok = LambertSolve(r1, r2, tof, muSun, false, out v1, out v2);
            if (!ok) continue;
            double[] vDep = Add(v1, sunV0);
            double[] dvVec = Sub(vDep, earthState.Item2);
            double dv = Norm(dvVec);
            if (dv < bestDv)
            {
                bestDv = dv;
                bestLaunch = t;
                bestDvVec = dvVec;
            }
        }

        if (bestDv > dvBudget * 0.9 || bestDv == double.MaxValue)
        {
            // Fallback direct burn
            bestLaunch = 0;
            var e0 = PropagateBodyTo(EarthIdx, 0);
            double[] dir = Sub(astStateInt.Item1, e0.Item1);
            double d = Norm(dir);
            double speed = Math.Min(dvBudget * 0.6, d / interceptTimeSec);
            bestDvVec = new[] { dir[0] / d * speed, dir[1] / d * speed, dir[2] / d * speed };
        }

        double remaining = Math.Max(0, dvBudget - Norm(bestDvVec));
        double[] aV = astStateInt.Item2;
        double aVn = Norm(aV);
        double[] impactDv = aVn > 0
            ? new[] { aV[0] / aVn * remaining, aV[1] / aVn * remaining, aV[2] / aVn * remaining }
            : new double[3];

        var sb = new System.Text.StringBuilder();
        sb.AppendLine("2");
        sb.AppendLine(Fmt(bestLaunch / DAY, bestDvVec));
        sb.AppendLine(Fmt(interceptTimeSec / DAY, impactDv));
        Console.Write(sb.ToString());
    }

    static string Fmt(double t, double[] dv) =>
        string.Format(CultureInfo.InvariantCulture, "{0:R} {1:R} {2:R} {3:R}", t, dv[0], dv[1], dv[2]);

    static Tuple<double[], double[]> PropagateBodyTo(int idx, double tSec)
    {
        var b = Bodies[idx];
        if (idx == SunIdx)
            return Tuple.Create(
                new[] { b.R[0] + b.V[0] * tSec, b.R[1] + b.V[1] * tSec, b.R[2] + b.V[2] * tSec },
                (double[])b.V.Clone());
        double[] sunR = SunIdx >= 0 ? Bodies[SunIdx].R : new double[3];
        double[] sunV = SunIdx >= 0 ? Bodies[SunIdx].V : new double[3];
        double muSun = G * (SunIdx >= 0 ? Bodies[SunIdx].Mass : 1.989e30);
        double[] r0 = Sub(b.R, sunR);
        double[] v0 = Sub(b.V, sunV);
        double[] rT, vT;
        KeplerPropagate(r0, v0, muSun, tSec, out rT, out vT);
        double[] sunRt = { sunR[0] + sunV[0] * tSec, sunR[1] + sunV[1] * tSec, sunR[2] + sunV[2] * tSec };
        return Tuple.Create(Add(rT, sunRt), Add(vT, sunV));
    }

    static double FindEarthImpactTime(double maxSec)
    {
        double bestT = maxSec;
        double bestD = double.MaxValue;
        int N = 400;
        for (int i = 0; i <= N; i++)
        {
            double t = maxSec * i / N;
            var e = PropagateBodyTo(EarthIdx, t);
            var a = PropagateBodyTo(AsteroidIdx, t);
            double d = Norm(Sub(e.Item1, a.Item1));
            if (d < bestD) { bestD = d; bestT = t; }
        }
        return bestT;
    }

    static double[] Sub(double[] a, double[] b) => new[] { a[0] - b[0], a[1] - b[1], a[2] - b[2] };
    static double[] Add(double[] a, double[] b) => new[] { a[0] + b[0], a[1] + b[1], a[2] + b[2] };
    static double Dot(double[] a, double[] b) => a[0] * b[0] + a[1] * b[1] + a[2] * b[2];
    static double[] Cross(double[] a, double[] b) => new[] { a[1] * b[2] - a[2] * b[1], a[2] * b[0] - a[0] * b[2], a[0] * b[1] - a[1] * b[0] };
    static double Norm(double[] a) => Math.Sqrt(Dot(a, a));

    static void KeplerPropagate(double[] r0, double[] v0, double mu, double dt, out double[] r, out double[] v)
    {
        double r0n = Norm(r0);
        double v0n = Norm(v0);
        double alpha = 2.0 / r0n - v0n * v0n / mu;
        double sqrtMu = Math.Sqrt(mu);
        double chi;
        if (alpha > 1e-12) chi = sqrtMu * dt * alpha;
        else chi = sqrtMu * dt / r0n;

        double rdotv = Dot(r0, v0);
        double c2 = 0.5, c3 = 1.0 / 6.0, psi = 0, rNew = r0n;
        for (int it = 0; it < 300; it++)
        {
            psi = chi * chi * alpha;
            StumpffCS(psi, out c2, out c3);
            rNew = chi * chi * c2 + rdotv / sqrtMu * chi * (1 - psi * c3) + r0n * (1 - psi * c2);
            double F = rdotv / sqrtMu * chi * chi * c2 + (1 - r0n * alpha) * chi * chi * chi * c3 + r0n * chi - sqrtMu * dt;
            if (Math.Abs(rNew) < 1e-12) break;
            double dchi = F / rNew;
            chi -= dchi;
            if (Math.Abs(dchi) < 1e-10) break;
        }
        psi = chi * chi * alpha;
        StumpffCS(psi, out c2, out c3);
        double f = 1 - chi * chi / r0n * c2;
        double g = dt - chi * chi * chi / sqrtMu * c3;
        r = new[] { f * r0[0] + g * v0[0], f * r0[1] + g * v0[1], f * r0[2] + g * v0[2] };
        double rn = Norm(r);
        if (rn < 1e-12) rn = 1e-12;
        double fdot = sqrtMu / (rn * r0n) * chi * (psi * c3 - 1);
        double gdot = 1 - chi * chi / rn * c2;
        v = new[] { fdot * r0[0] + gdot * v0[0], fdot * r0[1] + gdot * v0[1], fdot * r0[2] + gdot * v0[2] };
    }

    static void StumpffCS(double psi, out double c, out double s)
    {
        if (psi > 1e-6)
        {
            double sp = Math.Sqrt(psi);
            c = (1 - Math.Cos(sp)) / psi;
            s = (sp - Math.Sin(sp)) / (sp * psi);
        }
        else if (psi < -1e-6)
        {
            double sp = Math.Sqrt(-psi);
            c = (1 - Math.Cosh(sp)) / psi;
            s = (Math.Sinh(sp) - sp) / (sp * (-psi));
        }
        else
        {
            c = 0.5; s = 1.0 / 6.0;
        }
    }

    static bool LambertSolve(double[] r1, double[] r2, double tof, double mu, bool longWay, out double[] v1, out double[] v2)
    {
        v1 = new double[3]; v2 = new double[3];
        double r1n = Norm(r1), r2n = Norm(r2);
        double cosdnu = Dot(r1, r2) / (r1n * r2n);
        cosdnu = Math.Max(-1, Math.Min(1, cosdnu));
        double[] cr = Cross(r1, r2);
        double sindnu = Norm(cr) / (r1n * r2n);
        if (cr[2] < 0) sindnu = -sindnu;
        double dnu = Math.Atan2(sindnu, cosdnu);
        if (dnu < 0) dnu += 2 * Math.PI;
        if (longWay) dnu = 2 * Math.PI - dnu;
        double sinD = Math.Sin(dnu);
        double cosD = Math.Cos(dnu);
        if (Math.Abs(1 - cosD) < 1e-12) return false;
        double A = sinD * Math.Sqrt(r1n * r2n / (1 - cosD));
        if (Math.Abs(A) < 1e-6) return false;
        double signA = Math.Sign(A);

        double psiLo = -4 * Math.PI * Math.PI, psiHi = 4 * Math.PI * Math.PI;
        double psi = 0, c2 = 0.5, c3 = 1.0 / 6.0;
        double y = 0;
        for (int it = 0; it < 200; it++)
        {
            StumpffCS(psi, out c2, out c3);
            y = r1n + r2n + A * (psi * c3 - 1) / Math.Sqrt(c2);
            if (A > 0 && y < 0)
            {
                for (int j = 0; j < 30 && y < 0; j++)
                {
                    psi += 0.1;
                    StumpffCS(psi, out c2, out c3);
                    y = r1n + r2n + A * (psi * c3 - 1) / Math.Sqrt(c2);
                }
                psiLo = psi;
            }
            if (y <= 0) return false;
            double chi = Math.Sqrt(y / c2);
            double tn = (chi * chi * chi * c3 + A * Math.Sqrt(y)) / Math.Sqrt(mu);
            if (tn <= tof) psiLo = psi; else psiHi = psi;
            double psiNew = (psiLo + psiHi) / 2;
            if (Math.Abs(tn - tof) < 1e-4 * tof) { psi = psiNew; break; }
            psi = psiNew;
        }
        StumpffCS(psi, out c2, out c3);
        y = r1n + r2n + A * (psi * c3 - 1) / Math.Sqrt(c2);
        if (y <= 0) return false;
        double f = 1 - y / r1n;
        double g = A * Math.Sqrt(y / mu);
        double gdot = 1 - y / r2n;
        if (Math.Abs(g) < 1e-12) return false;
        for (int k = 0; k < 3; k++)
        {
            v1[k] = (r2[k] - f * r1[k]) / g;
            v2[k] = (gdot * r2[k] - r1[k]) / g;
        }
        return true;
    }
}
