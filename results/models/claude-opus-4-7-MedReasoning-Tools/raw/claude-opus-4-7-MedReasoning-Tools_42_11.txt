; SPIR-V
; Version: 1.0
               OpCapability Geometry
        %ext = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 8

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
     %fnvoid = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0

      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
     %uint_4 = OpConstant %uint 4
    %f_zero  = OpConstant %float 0.0
    %f_one   = OpConstant %float 1.0
    %f_thick = OpConstant %float 0.02
    %f_1_3   = OpConstant %float 0.33333333
    %f_2_3   = OpConstant %float 0.66666667
    %f_eps   = OpConstant %float 0.0000001

  %v4_t13 = OpConstantComposite %v4float %f_1_3 %f_1_3 %f_1_3 %f_1_3
  %v4_t23 = OpConstantComposite %v4float %f_2_3 %f_2_3 %f_2_3 %f_2_3
  %v2_t13 = OpConstantComposite %v2float %f_1_3 %f_1_3
  %v2_t23 = OpConstantComposite %v2float %f_2_3 %f_2_3
  %v3_t13 = OpConstantComposite %v3float %f_1_3 %f_1_3 %f_1_3
  %v3_t23 = OpConstantComposite %v3float %f_2_3 %f_2_3 %f_2_3

%gl_PerVertex_in   = OpTypeStruct %v4float
%arr_pv_in         = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv     = OpTypePointer Input %arr_pv_in
%gl_in             = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4         = OpTypePointer Input %v4float

%arr_v3_4          = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3     = OpTypePointer Input %arr_v3_4
%inColor           = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3         = OpTypePointer Input %v3float

%gl_PerVertex_out  = OpTypeStruct %v4float
%ptr_out_pv        = OpTypePointer Output %gl_PerVertex_out
%gl_out            = OpVariable %ptr_out_pv Output
%ptr_out_v4        = OpTypePointer Output %v4float
%ptr_out_v3        = OpTypePointer Output %v3float
%outColor          = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fnvoid
      %entry = OpLabel

  %p0p = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
  %p0  = OpLoad %v4float %p0p
  %p1p = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
  %p1  = OpLoad %v4float %p1p
  %p2p = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
  %p2  = OpLoad %v4float %p2p
  %p3p = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
  %p3  = OpLoad %v4float %p3p

  %c1p = OpAccessChain %ptr_in_v3 %inColor %int_1
  %c1  = OpLoad %v3float %c1p
  %c2p = OpAccessChain %ptr_in_v3 %inColor %int_2
  %c2  = OpLoad %v3float %c2p

  ; tangent at p1 = p2 - p0  ; tangent at p2 = p3 - p1
  %t1v4 = OpFSub %v4float %p2 %p0
  %t1x  = OpCompositeExtract %float %t1v4 0
  %t1y  = OpCompositeExtract %float %t1v4 1
  %nt1y = OpFNegate %float %t1y
  %perp1_raw = OpCompositeConstruct %v2float %nt1y %t1x
  %perp1_len = OpExtInst %float %ext Length %perp1_raw
  %perp1_safe = OpExtInst %float %ext FMax %perp1_len %f_eps
  %perp1_safe_v2 = OpCompositeConstruct %v2float %perp1_safe %perp1_safe
  %perp1_n   = OpFDiv %v2float %perp1_raw %perp1_safe_v2
  %perp1     = OpVectorTimesScalar %v2float %perp1_n %f_thick

  %t2v4 = OpFSub %v4float %p3 %p1
  %t2x  = OpCompositeExtract %float %t2v4 0
  %t2y  = OpCompositeExtract %float %t2v4 1
  %nt2y = OpFNegate %float %t2y
  %perp2_raw = OpCompositeConstruct %v2float %nt2y %t2x
  %perp2_len = OpExtInst %float %ext Length %perp2_raw
  %perp2_safe = OpExtInst %float %ext FMax %perp2_len %f_eps
  %perp2_safe_v2 = OpCompositeConstruct %v2float %perp2_safe %perp2_safe
  %perp2_n   = OpFDiv %v2float %perp2_raw %perp2_safe_v2
  %perp2     = OpVectorTimesScalar %v2float %perp2_n %f_thick

  ; Interpolated positions along p1->p2
  %posA = %p1
  %posB = OpExtInst %v4float %ext FMix %p1 %p2 %v4_t13
  %posC = OpExtInst %v4float %ext FMix %p1 %p2 %v4_t23
  %posD = %p2

  ; Interpolated perpendiculars
  %perpA = %perp1
  %perpB = OpExtInst %v2float %ext FMix %perp1 %perp2 %v2_t13
  %perpC = OpExtInst %v2float %ext FMix %perp1 %perp2 %v2_t23
  %perpD = %perp2

  ; Interpolated colors
  %colA = %c1
  %colB = OpExtInst %v3float %ext FMix %c1 %c2 %v3_t13
  %colC = OpExtInst %v3float %ext FMix %c1 %c2 %v3_t23
  %colD = %c2

  ; ===== Quad 1: from t=0 (A) to t=1/3 (B) =====
  %A_xy   = OpVectorShuffle %v2float %posA %posA 0 1
  %A_zw   = OpVectorShuffle %v2float %posA %posA 2 3
  %A_neg  = OpFSub %v2float %A_xy %perpA
  %A_pos  = OpFAdd %v2float %A_xy %perpA
  %A_zw_x = OpCompositeExtract %float %A_zw 0
  %A_zw_y = OpCompositeExtract %float %A_zw 1
  %A_neg_x = OpCompositeExtract %float %A_neg 0
  %A_neg_y = OpCompositeExtract %float %A_neg 1
  %A_pos_x = OpCompositeExtract %float %A_pos 0
  %A_pos_y = OpCompositeExtract %float %A_pos 1
  %vA0 = OpCompositeConstruct %v4float %A_neg_x %A_neg_y %A_zw_x %A_zw_y
  %vA1 = OpCompositeConstruct %v4float %A_pos_x %A_pos_y %A_zw_x %A_zw_y

  %B_xy   = OpVectorShuffle %v2float %posB %posB 0 1
  %B_zw   = OpVectorShuffle %v2float %posB %posB 2 3
  %B_neg  = OpFSub %v2float %B_xy %perpB
  %B_pos  = OpFAdd %v2float %B_xy %perpB
  %B_zw_x = OpCompositeExtract %float %B_zw 0
  %B_zw_y = OpCompositeExtract %float %B_zw 1
  %B_neg_x = OpCompositeExtract %float %B_neg 0
  %B_neg_y = OpCompositeExtract %float %B_neg 1
  %B_pos_x = OpCompositeExtract %float %B_pos 0
  %B_pos_y = OpCompositeExtract %float %B_pos 1
  %vB0 = OpCompositeConstruct %v4float %B_neg_x %B_neg_y %B_zw_x %B_zw_y
  %vB1 = OpCompositeConstruct %v4float %B_pos_x %B_pos_y %B_zw_x %B_zw_y

  ; emit vA0
  %o_pA0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
           OpStore %o_pA0 %vA0
           OpStore %outColor %colA
           OpEmitVertex
  ; emit vA1
  %o_pA1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
           OpStore %o_pA1 %vA1
           OpStore %outColor %colA
           OpEmitVertex
  ; emit vB0
  %o_pB0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
           OpStore %o_pB0 %vB0
           OpStore %outColor %colB
           OpEmitVertex
  ; emit vB1
  %o_pB1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
           OpStore %o_pB1 %vB1
           OpStore %outColor %colB
           OpEmitVertex
           OpEndPrimitive

  ; ===== Quad 2: from t=2/3 (C) to t=1 (D) =====
  %C_xy   = OpVectorShuffle %v2float %posC %posC 0 1
  %C_zw   = OpVectorShuffle %v2float %posC %posC 2 3
  %C_neg  = OpFSub %v2float %C_xy %perpC
  %C_pos  = OpFAdd %v2float %C_xy %perpC
  %C_zw_x = OpCompositeExtract %float %C_zw 0
  %C_zw_y = OpCompositeExtract %float %C_zw 1
  %C_neg_x = OpCompositeExtract %float %C_neg 0
  %C_neg_y = OpCompositeExtract %float %C_neg 1
  %C_pos_x = OpCompositeExtract %float %C_pos 0
  %C_pos_y = OpCompositeExtract %float %C_pos 1
  %vC0 = OpCompositeConstruct %v4float %C_neg_x %C_neg_y %C_zw_x %C_zw_y
  %vC1 = OpCompositeConstruct %v4float %C_pos_x %C_pos_y %C_zw_x %C_zw_y

  %D_xy   = OpVectorShuffle %v2float %posD %posD 0 1
  %D_zw   = OpVectorShuffle %v2float %posD %posD 2 3
  %D_neg  = OpFSub %v2float %D_xy %perpD
  %D_pos  = OpFAdd %v2float %D_xy %perpD
  %D_zw_x = OpCompositeExtract %float %D_zw 0
  %D_zw_y = OpCompositeExtract %float %D_zw 1
  %D_neg_x = OpCompositeExtract %float %D_neg 0
  %D_neg_y = OpCompositeExtract %float %D_neg 1
  %D_pos_x = OpCompositeExtract %float %D_pos 0
  %D_pos_y = OpCompositeExtract %float %D_pos 1
  %vD0 = OpCompositeConstruct %v4float %D_neg_x %D_neg_y %D_zw_x %D_zw_y
  %vD1 = OpCompositeConstruct %v4float %D_pos_x %D_pos_y %D_zw_x %D_zw_y

  %o_pC0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
           OpStore %o_pC0 %vC0
           OpStore %outColor %colC
           OpEmitVertex
  %o_pC1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
           OpStore %o_pC1 %vC1
           OpStore %outColor %colC
           OpEmitVertex
  %o_pD0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
           OpStore %o_pD0 %vD0
           OpStore %outColor %colD
           OpEmitVertex
  %o_pD1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
           OpStore %o_pD1 %vD1
           OpStore %outColor %colD
           OpEmitVertex
           OpEndPrimitive

               OpReturn
               OpFunctionEnd
