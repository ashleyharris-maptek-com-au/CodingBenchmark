declare const require: any;
declare const process: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const prob: number[][] = data.probability;
const drones: { id: any; x: number; y: number }[] = data.drones;
const maxSteps: number = data.maxStepsPerDrone;

const H = prob.length;
const W = H > 0 ? prob[0].length : 0;

function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < W && y >= 0 && y < H;
}

const claimed: Uint8Array = new Uint8Array(W * H);

const DX = [0, 1, 0, -1, 0];
const DY = [-1, 0, 1, 0, 0];

const totalCells = W * H;
// Budget beam width so total work is bounded
const stepsBudget = Math.max(1, maxSteps);
let K_BEAM = Math.max(4, Math.min(48, Math.floor(4000 / stepsBudget)));
if (totalCells > 10000) K_BEAM = Math.min(K_BEAM, 16);

function planDrone(sx: number, sy: number, steps: number): number[][] {
  type State = {
    x: number;
    y: number;
    score: number;
    visited: Set<number>;
    path: number[];
  };

  const startKey = sy * W + sx;
  const startScore = (!claimed[startKey] && prob[sy][sx] > 0) ? prob[sy][sx] : 0;
  const startVisited = new Set<number>();
  startVisited.add(startKey);

  let beam: State[] = [{
    x: sx,
    y: sy,
    score: startScore,
    visited: startVisited,
    path: [startKey],
  }];

  for (let s = 0; s < steps; s++) {
    const next: State[] = [];
    for (const st of beam) {
      for (let d = 0; d < 5; d++) {
        const nx = st.x + DX[d];
        const ny = st.y + DY[d];
        if (!inBounds(nx, ny)) continue;
        const v = prob[ny][nx];
        if (v < 0) continue;
        const key = ny * W + nx;
        let addScore = 0;
        if (!st.visited.has(key) && !claimed[key] && v > 0) {
          addScore = v;
        }
        const newVisited = new Set(st.visited);
        newVisited.add(key);
        const newPath = st.path.slice();
        newPath.push(key);
        next.push({
          x: nx,
          y: ny,
          score: st.score + addScore,
          visited: newVisited,
          path: newPath,
        });
      }
    }
    if (next.length === 0) break;
    // Keep best state per ending cell
    const byCell = new Map<number, State>();
    for (const st of next) {
      const k = st.y * W + st.x;
      const cur = byCell.get(k);
      if (!cur || st.score > cur.score) byCell.set(k, st);
    }
    const arr = Array.from(byCell.values());
    arr.sort((a, b) => b.score - a.score);
    beam = arr.slice(0, K_BEAM);
  }

  let best = beam[0];
  for (const st of beam) if (st.score > best.score) best = st;

  const out: number[][] = [];
  for (const k of best.path) {
    const x = k % W;
    const y = (k - x) / W;
    out.push([x, y]);
  }
  for (const k of best.path) claimed[k] = 1;
  return out;
}

const paths: number[][][] = [];
for (const d of drones) {
  if (!inBounds(d.x, d.y) || prob[d.y][d.x] < 0) {
    paths.push([[d.x, d.y]]);
    continue;
  }
  paths.push(planDrone(d.x, d.y, maxSteps));
}

process.stdout.write(JSON.stringify({ paths }));
