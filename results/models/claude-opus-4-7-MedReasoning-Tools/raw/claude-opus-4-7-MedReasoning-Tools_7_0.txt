// CSG Union via BSP (Evan Wallace style), with fast-path for disjoint inputs
// and an AABB-based candidate filter so triangles far from the intersection
// region are passed through unchanged. Single-file C++20, MSVC-compatible.

#include <cstdio>
#include <cstdlib>
#include <cstdint>
#include <cmath>
#include <vector>
#include <array>
#include <string>
#include <unordered_map>
#include <algorithm>
#include <memory>

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double a, double b, double c) : x(a), y(b), z(c) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    double dot(const Vec3& o) const { return x*o.x + y*o.y + z*o.z; }
    Vec3 cross(const Vec3& o) const {
        return { y*o.z - z*o.y, z*o.x - x*o.z, x*o.y - y*o.x };
    }
    double norm() const { return std::sqrt(x*x + y*y + z*z); }
    Vec3 unit() const { double n = norm(); return n > 0 ? Vec3(x/n, y/n, z/n) : Vec3(0,0,0); }
};

static inline Vec3 lerp(const Vec3& a, const Vec3& b, double t) { return a + (b - a) * t; }

struct Plane {
    Vec3 n;
    double w;
    static constexpr double EPS = 1e-7;
    Plane() : w(0) {}
    Plane(const Vec3& a, const Vec3& b, const Vec3& c) {
        n = (b - a).cross(c - a).unit();
        w = n.dot(a);
    }
    void flip() { n = Vec3(-n.x, -n.y, -n.z); w = -w; }
};

struct Polygon {
    std::vector<Vec3> v;
    Plane plane;
    Polygon() = default;
    Polygon(std::vector<Vec3> verts) {
        v = std::move(verts);
        if (v.size() >= 3) plane = Plane(v[0], v[1], v[2]);
    }
    void flip() {
        std::reverse(v.begin(), v.end());
        plane.flip();
    }
};

static void splitPolygon(const Plane& plane, const Polygon& poly,
                         std::vector<Polygon>& coplanarFront,
                         std::vector<Polygon>& coplanarBack,
                         std::vector<Polygon>& front,
                         std::vector<Polygon>& back)
{
    const double EPS = Plane::EPS;
    enum { CO = 0, FR = 1, BA = 2, SP = 3 };
    int polygonType = 0;
    int n = (int)poly.v.size();
    std::vector<int> types(n);
    for (int i = 0; i < n; i++) {
        double t = plane.n.dot(poly.v[i]) - plane.w;
        int type = (t < -EPS) ? BA : (t > EPS) ? FR : CO;
        polygonType |= type;
        types[i] = type;
    }
    if (polygonType == CO) {
        if (plane.n.dot(poly.plane.n) > 0) coplanarFront.push_back(poly);
        else coplanarBack.push_back(poly);
    } else if (polygonType == FR) {
        front.push_back(poly);
    } else if (polygonType == BA) {
        back.push_back(poly);
    } else {
        std::vector<Vec3> f, b;
        f.reserve(n + 1); b.reserve(n + 1);
        for (int i = 0; i < n; i++) {
            int j = (i + 1) % n;
            int ti = types[i], tj = types[j];
            const Vec3& vi = poly.v[i];
            const Vec3& vj = poly.v[j];
            if (ti != BA) f.push_back(vi);
            if (ti != FR) b.push_back(vi);
            if ((ti | tj) == SP) {
                double denom = plane.n.dot(vj - vi);
                if (std::fabs(denom) < 1e-30) continue;
                double t = (plane.w - plane.n.dot(vi)) / denom;
                Vec3 vx = lerp(vi, vj, t);
                f.push_back(vx);
                b.push_back(vx);
            }
        }
        if (f.size() >= 3) front.push_back(Polygon(std::move(f)));
        if (b.size() >= 3) back.push_back(Polygon(std::move(b)));
    }
}

struct Node {
    Plane plane;
    bool hasPlane = false;
    std::unique_ptr<Node> front, back;
    std::vector<Polygon> polygons;

    void invert() {
        for (auto& p : polygons) p.flip();
        plane.flip();
        if (front) front->invert();
        if (back) back->invert();
        std::swap(front, back);
    }

    std::vector<Polygon> clipPolygons(std::vector<Polygon> polys) const {
        if (!hasPlane) return polys;
        std::vector<Polygon> f, b;
        for (auto& p : polys) {
            splitPolygon(plane, p, f, b, f, b);
        }
        if (front) f = front->clipPolygons(std::move(f));
        if (back) b = back->clipPolygons(std::move(b));
        else b.clear();
        f.insert(f.end(), std::make_move_iterator(b.begin()), std::make_move_iterator(b.end()));
        return f;
    }

    void clipTo(const Node& other) {
        polygons = other.clipPolygons(std::move(polygons));
        if (front) front->clipTo(other);
        if (back) back->clipTo(other);
    }

    void allPolygons(std::vector<Polygon>& out) const {
        out.insert(out.end(), polygons.begin(), polygons.end());
        if (front) front->allPolygons(out);
        if (back) back->allPolygons(out);
    }

    void build(std::vector<Polygon> polys) {
        if (polys.empty()) return;
        if (!hasPlane) {
            plane = polys[0].plane;
            hasPlane = true;
        }
        std::vector<Polygon> f, b;
        for (auto& p : polys) {
            splitPolygon(plane, p, polygons, polygons, f, b);
        }
        if (!f.empty()) {
            if (!front) front = std::make_unique<Node>();
            front->build(std::move(f));
        }
        if (!b.empty()) {
            if (!back) back = std::make_unique<Node>();
            back->build(std::move(b));
        }
    }
};

struct AABB {
    Vec3 mn, mx;
    bool valid = false;
    void expand(const Vec3& v) {
        if (!valid) { mn = mx = v; valid = true; return; }
        mn.x = std::min(mn.x, v.x); mn.y = std::min(mn.y, v.y); mn.z = std::min(mn.z, v.z);
        mx.x = std::max(mx.x, v.x); mx.y = std::max(mx.y, v.y); mx.z = std::max(mx.z, v.z);
    }
    bool overlaps(const AABB& o, double eps = 1e-9) const {
        if (!valid || !o.valid) return false;
        if (mx.x < o.mn.x - eps || mn.x > o.mx.x + eps) return false;
        if (mx.y < o.mn.y - eps || mn.y > o.mx.y + eps) return false;
        if (mx.z < o.mn.z - eps || mn.z > o.mx.z + eps) return false;
        return true;
    }
};

static std::string readAll() {
    std::string s;
    s.reserve(1 << 20);
    char buf[1 << 16];
    size_t n;
    while ((n = fread(buf, 1, sizeof(buf), stdin)) > 0) s.append(buf, buf + n);
    return s;
}

struct Parser {
    const char* p;
    const char* end;
    void skip() {
        while (p < end && (*p == ' ' || *p == '\t' || *p == '\r' || *p == '\n')) ++p;
    }
    long long readInt() {
        skip();
        bool neg = false;
        if (p < end && (*p == '-' || *p == '+')) { if (*p == '-') neg = true; ++p; }
        long long v = 0;
        while (p < end && *p >= '0' && *p <= '9') { v = v * 10 + (*p - '0'); ++p; }
        return neg ? -v : v;
    }
    double readDouble() {
        skip();
        char* endp;
        double v = std::strtod(p, &endp);
        p = endp;
        return v;
    }
};

int main() {
    std::string data = readAll();
    Parser ps{ data.data(), data.data() + data.size() };

    long long VA = ps.readInt();
    long long FA = ps.readInt();
    long long VB = ps.readInt();
    long long FB = ps.readInt();

    std::vector<Vec3> va((size_t)VA), vb((size_t)VB);
    std::vector<std::array<int, 3>> fa((size_t)FA), fb((size_t)FB);

    AABB ba, bb;
    for (long long i = 0; i < VA; i++) {
        double x = ps.readDouble(), y = ps.readDouble(), z = ps.readDouble();
        va[(size_t)i] = { x, y, z };
        ba.expand(va[(size_t)i]);
    }
    for (long long i = 0; i < FA; i++) {
        fa[(size_t)i] = { (int)ps.readInt(), (int)ps.readInt(), (int)ps.readInt() };
    }
    for (long long i = 0; i < VB; i++) {
        double x = ps.readDouble(), y = ps.readDouble(), z = ps.readDouble();
        vb[(size_t)i] = { x, y, z };
        bb.expand(vb[(size_t)i]);
    }
    for (long long i = 0; i < FB; i++) {
        fb[(size_t)i] = { (int)ps.readInt(), (int)ps.readInt(), (int)ps.readInt() };
    }

    auto emitConcat = [&]() {
        long long V = VA + VB;
        long long F = FA + FB;
        std::string out;
        out.reserve((size_t)(V * 40 + F * 20 + 64));
        char buf[128];
        int n = std::snprintf(buf, sizeof(buf), "%lld %lld\n", V, F);
        out.append(buf, buf + n);
        for (auto& v : va) {
            n = std::snprintf(buf, sizeof(buf), "%.17g %.17g %.17g\n", v.x, v.y, v.z);
            out.append(buf, buf + n);
        }
        for (auto& v : vb) {
            n = std::snprintf(buf, sizeof(buf), "%.17g %.17g %.17g\n", v.x, v.y, v.z);
            out.append(buf, buf + n);
        }
        for (auto& f : fa) {
            n = std::snprintf(buf, sizeof(buf), "%d %d %d\n", f[0], f[1], f[2]);
            out.append(buf, buf + n);
        }
        for (auto& f : fb) {
            n = std::snprintf(buf, sizeof(buf), "%d %d %d\n",
                              f[0] + (int)VA, f[1] + (int)VA, f[2] + (int)VA);
            out.append(buf, buf + n);
        }
        fwrite(out.data(), 1, out.size(), stdout);
    };

    if (!ba.overlaps(bb)) {
        emitConcat();
        return 0;
    }

    auto triAABB = [](const Vec3& a, const Vec3& b, const Vec3& c) {
        AABB x; x.expand(a); x.expand(b); x.expand(c); return x;
    };

    AABB inter;
    inter.valid = true;
    inter.mn = { std::max(ba.mn.x, bb.mn.x), std::max(ba.mn.y, bb.mn.y), std::max(ba.mn.z, bb.mn.z) };
    inter.mx = { std::min(ba.mx.x, bb.mx.x), std::min(ba.mx.y, bb.mx.y), std::min(ba.mx.z, bb.mx.z) };
    double pad = 1e-6 * std::max({ ba.mx.x - ba.mn.x, ba.mx.y - ba.mn.y, ba.mx.z - ba.mn.z, 1.0 });
    inter.mn = inter.mn - Vec3(pad, pad, pad);
    inter.mx = inter.mx + Vec3(pad, pad, pad);

    std::vector<char> aCandidate((size_t)FA, 0), bCandidate((size_t)FB, 0);
    for (long long i = 0; i < FA; i++) {
        AABB t = triAABB(va[fa[(size_t)i][0]], va[fa[(size_t)i][1]], va[fa[(size_t)i][2]]);
        if (t.overlaps(inter)) aCandidate[(size_t)i] = 1;
    }
    for (long long i = 0; i < FB; i++) {
        AABB t = triAABB(vb[fb[(size_t)i][0]], vb[fb[(size_t)i][1]], vb[fb[(size_t)i][2]]);
        if (t.overlaps(inter)) bCandidate[(size_t)i] = 1;
    }

    std::vector<Polygon> polyA, polyB;
    polyA.reserve((size_t)std::count(aCandidate.begin(), aCandidate.end(), 1));
    polyB.reserve((size_t)std::count(bCandidate.begin(), bCandidate.end(), 1));
    for (long long i = 0; i < FA; i++) if (aCandidate[(size_t)i]) {
        auto& f = fa[(size_t)i];
        std::vector<Vec3> v = { va[f[0]], va[f[1]], va[f[2]] };
        Polygon p(std::move(v));
        if (p.plane.n.norm() > 0.5) polyA.push_back(std::move(p));
    }
    for (long long i = 0; i < FB; i++) if (bCandidate[(size_t)i]) {
        auto& f = fb[(size_t)i];
        std::vector<Vec3> v = { vb[f[0]], vb[f[1]], vb[f[2]] };
        Polygon p(std::move(v));
        if (p.plane.n.norm() > 0.5) polyB.push_back(std::move(p));
    }

    Node A, B;
    A.build(polyA);
    B.build(polyB);
    A.clipTo(B);
    B.clipTo(A);
    B.invert();
    B.clipTo(A);
    B.invert();
    std::vector<Polygon> bPolys;
    B.allPolygons(bPolys);
    A.build(std::move(bPolys));

    std::vector<Polygon> unionPolys;
    A.allPolygons(unionPolys);

    struct VKey { int64_t x, y, z; bool operator==(const VKey& o) const { return x == o.x && y == o.y && z == o.z; } };
    struct VKeyHash {
        size_t operator()(const VKey& k) const {
            uint64_t h = (uint64_t)k.x * 1469598103934665603ULL ^ ((uint64_t)k.y + 0x9e3779b97f4a7c15ULL);
            h ^= ((uint64_t)k.z + 0x517cc1b727220a95ULL) + (h << 6) + (h >> 2);
            return (size_t)h;
        }
    };

    AABB scene = ba;
    scene.expand(bb.mn); scene.expand(bb.mx);
    double scale = std::max({ scene.mx.x - scene.mn.x, scene.mx.y - scene.mn.y, scene.mx.z - scene.mn.z, 1.0 });
    double q = 1e9 / scale;

    std::unordered_map<VKey, int, VKeyHash> vmap;
    std::vector<Vec3> outV;
    std::vector<std::array<int, 3>> outF;
    outV.reserve((size_t)(VA + VB));
    outF.reserve((size_t)(FA + FB));

    auto addVertex = [&](const Vec3& v) -> int {
        VKey k{ (int64_t)std::llround(v.x * q), (int64_t)std::llround(v.y * q), (int64_t)std::llround(v.z * q) };
        auto it = vmap.find(k);
        if (it != vmap.end()) return it->second;
        int idx = (int)outV.size();
        outV.push_back(v);
        vmap.emplace(k, idx);
        return idx;
    };

    for (long long i = 0; i < FA; i++) if (!aCandidate[(size_t)i]) {
        auto& f = fa[(size_t)i];
        int i0 = addVertex(va[f[0]]);
        int i1 = addVertex(va[f[1]]);
        int i2 = addVertex(va[f[2]]);
        if (i0 != i1 && i1 != i2 && i0 != i2) outF.push_back({ i0, i1, i2 });
    }
    for (long long i = 0; i < FB; i++) if (!bCandidate[(size_t)i]) {
        auto& f = fb[(size_t)i];
        int i0 = addVertex(vb[f[0]]);
        int i1 = addVertex(vb[f[1]]);
        int i2 = addVertex(vb[f[2]]);
        if (i0 != i1 && i1 != i2 && i0 != i2) outF.push_back({ i0, i1, i2 });
    }
    for (auto& p : unionPolys) {
        if (p.v.size() < 3) continue;
        int i0 = addVertex(p.v[0]);
        for (size_t k = 1; k + 1 < p.v.size(); k++) {
            int i1 = addVertex(p.v[k]);
            int i2 = addVertex(p.v[k + 1]);
            if (i0 != i1 && i1 != i2 && i0 != i2) outF.push_back({ i0, i1, i2 });
        }
    }

    std::string out;
    out.reserve(outV.size() * 40 + outF.size() * 20 + 64);
    char buf[128];
    int n = std::snprintf(buf, sizeof(buf), "%zu %zu\n", outV.size(), outF.size());
    out.append(buf, buf + n);
    for (auto& v : outV) {
        n = std::snprintf(buf, sizeof(buf), "%.17g %.17g %.17g\n", v.x, v.y, v.z);
        out.append(buf, buf + n);
    }
    for (auto& f : outF) {
        n = std::snprintf(buf, sizeof(buf), "%d %d %d\n", f[0], f[1], f[2]);
        out.append(buf, buf + n);
    }
    fwrite(out.data(), 1, out.size(), stdout);
    return 0;
}
