using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Numerics;
using System.Text;
using System.Threading;
using System.Threading.Tasks;

class Program
{
    static int N;
    static int W;
    static ulong[][] Adj;
    static int[] Deg;
    static List<int>[] AdjList;
    static bool UseList = false;

    static Stopwatch Watch = new Stopwatch();
    static long TimeLimitMs = 4 * 60 * 1000 + 30 * 1000;

    static int BestSize = 0;
    static int[] BestClique = Array.Empty<int>();
    static readonly object BestLock = new object();
    static volatile bool TimeUp = false;

    static void Main()
    {
        Watch.Start();
        ReadInput();

        if (N == 0)
        {
            Console.Out.Write("0\n\n");
            return;
        }

        var heur = GreedyHeuristic();
        UpdateBest(heur);

        var heurTask = Task.Run(() =>
        {
            try { var h2 = LocalSearchHeuristic(2000); UpdateBest(h2); } catch { }
        });

        if (!UseList && N <= 3000)
        {
            try { ExactSearchParallel(); } catch { }
        }
        else
        {
            try { var h3 = LocalSearchHeuristic(8000); UpdateBest(h3); } catch { }
        }

        try { heurTask.Wait(500); } catch { }
        WriteOutput();
    }

    static void UpdateBest(int[] cand)
    {
        if (cand == null) return;
        lock (BestLock)
        {
            if (cand.Length > BestSize)
            {
                BestSize = cand.Length;
                BestClique = cand;
            }
        }
    }

    static bool TimeExceeded()
    {
        if (TimeUp) return true;
        if (Watch.ElapsedMilliseconds > TimeLimitMs) { TimeUp = true; return true; }
        return false;
    }

    static void ReadInput()
    {
        using var sr = new StreamReader(Console.OpenStandardInput(), Encoding.ASCII, false, 1 << 16);
        string firstLine;
        do { firstLine = sr.ReadLine(); if (firstLine == null) { N = 0; return; } } while (firstLine.Length == 0);
        var parts = firstLine.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        N = int.Parse(parts[0]);

        W = (N + 63) >> 6;
        bool bigBitset = (long)N * W > 100_000_000L;

        Adj = new ulong[N][];
        if (!bigBitset)
        {
            for (int i = 0; i < N; i++) Adj[i] = new ulong[W];
        }
        else
        {
            UseList = true;
            AdjList = new List<int>[N];
            for (int i = 0; i < N; i++) AdjList[i] = new List<int>();
        }

        Deg = new int[N];

        string line;
        while ((line = sr.ReadLine()) != null)
        {
            if (line.Length == 0) continue;
            int u = 0, v = 0, idx = 0, len = line.Length;
            while (idx < len && (line[idx] == ' ' || line[idx] == '\t')) idx++;
            if (idx >= len) continue;
            while (idx < len && line[idx] >= '0' && line[idx] <= '9') { u = u * 10 + (line[idx] - '0'); idx++; }
            while (idx < len && (line[idx] == ' ' || line[idx] == '\t')) idx++;
            while (idx < len && line[idx] >= '0' && line[idx] <= '9') { v = v * 10 + (line[idx] - '0'); idx++; }
            if (u == v) continue;
            if ((uint)u >= (uint)N || (uint)v >= (uint)N) continue;

            if (UseList)
            {
                AdjList[u].Add(v);
                AdjList[v].Add(u);
            }
            else
            {
                ulong mu = 1UL << (v & 63);
                ulong mv = 1UL << (u & 63);
                int wu = v >> 6, wv = u >> 6;
                if ((Adj[u][wu] & mu) == 0)
                {
                    Adj[u][wu] |= mu;
                    Adj[v][wv] |= mv;
                    Deg[u]++;
                    Deg[v]++;
                }
            }
        }

        if (UseList)
        {
            Parallel.For(0, N, i =>
            {
                var s = new HashSet<int>(AdjList[i]);
                var lst = new List<int>(s);
                lst.Sort();
                AdjList[i] = lst;
                Deg[i] = lst.Count;
            });
        }
    }

    static bool HasEdge(int u, int v)
    {
        if (UseList) return AdjList[u].BinarySearch(v) >= 0;
        return (Adj[u][v >> 6] & (1UL << (v & 63))) != 0;
    }

    static int[] GreedyHeuristic()
    {
        if (N == 0) return Array.Empty<int>();
        var order = new int[N];
        for (int i = 0; i < N; i++) order[i] = i;
        Array.Sort(order, (a, b) => Deg[b].CompareTo(Deg[a]));

        int[] best = Array.Empty<int>();
        int tries = Math.Min(N, 32);
        for (int t = 0; t < tries; t++)
        {
            if (TimeExceeded()) break;
            int start = order[t];
            var clique = new List<int> { start };
            List<int> cand;
            if (UseList) cand = new List<int>(AdjList[start]);
            else
            {
                cand = new List<int>();
                var row = Adj[start];
                for (int w = 0; w < W; w++)
                {
                    ulong b = row[w];
                    while (b != 0)
                    {
                        int bit = BitOperations.TrailingZeroCount(b);
                        b &= b - 1;
                        int vv = (w << 6) + bit;
                        if (vv < N && vv != start) cand.Add(vv);
                    }
                }
            }
            cand.Sort((a, b) => Deg[b].CompareTo(Deg[a]));
            while (cand.Count > 0)
            {
                int v = cand[0];
                clique.Add(v);
                var nc = new List<int>(cand.Count);
                if (UseList)
                {
                    var nb = AdjList[v];
                    foreach (var u in cand)
                    {
                        if (u == v) continue;
                        if (nb.BinarySearch(u) >= 0) nc.Add(u);
                    }
                }
                else
                {
                    var row = Adj[v];
                    foreach (var u in cand)
                    {
                        if (u == v) continue;
                        if ((row[u >> 6] & (1UL << (u & 63))) != 0) nc.Add(u);
                    }
                }
                cand = nc;
            }
            if (clique.Count > best.Length) best = clique.ToArray();
        }
        return best;
    }

    static int[] LocalSearchHeuristic(int iterations)
    {
        if (N == 0) return Array.Empty<int>();
        int[] best;
        lock (BestLock) best = BestClique == null ? Array.Empty<int>() : (int[])BestClique.Clone();
        var rng = new Random(20260514);

        int[] topByDeg;
        {
            var idx = new int[N];
            for (int i = 0; i < N; i++) idx[i] = i;
            Array.Sort(idx, (a, b) => Deg[b].CompareTo(Deg[a]));
            int k = Math.Min(N, 200);
            topByDeg = new int[k];
            Array.Copy(idx, topByDeg, k);
        }

        for (int iter = 0; iter < iterations; iter++)
        {
            if (TimeExceeded()) break;
            int start = rng.NextDouble() < 0.7 ? topByDeg[rng.Next(topByDeg.Length)] : rng.Next(N);

            var clique = new List<int> { start };
            List<int> cand;
            if (UseList) cand = new List<int>(AdjList[start]);
            else
            {
                cand = new List<int>();
                var row = Adj[start];
                for (int w = 0; w < W; w++)
                {
                    ulong b = row[w];
                    while (b != 0)
                    {
                        int bit = BitOperations.TrailingZeroCount(b);
                        b &= b - 1;
                        int vv = (w << 6) + bit;
                        if (vv < N && vv != start) cand.Add(vv);
                    }
                }
            }

            while (cand.Count > 0)
            {
                int pick = cand[rng.Next(cand.Count)];
                clique.Add(pick);
                var nc = new List<int>(cand.Count);
                if (UseList)
                {
                    var nb = AdjList[pick];
                    foreach (var u in cand)
                    {
                        if (u == pick) continue;
                        if (nb.BinarySearch(u) >= 0) nc.Add(u);
                    }
                }
                else
                {
                    var row = Adj[pick];
                    foreach (var u in cand)
                    {
                        if (u == pick) continue;
                        if ((row[u >> 6] & (1UL << (u & 63))) != 0) nc.Add(u);
                    }
                }
                cand = nc;
            }
            if (clique.Count > best.Length)
            {
                best = clique.ToArray();
                UpdateBest(best);
            }
        }
        return best;
    }

    // ----- Exact branch and bound with greedy coloring (MCS-like) -----
    static int[] OrderGlobal;

    static void ExactSearchParallel()
    {
        if (N == 0 || UseList) return;

        OrderGlobal = new int[N];
        for (int i = 0; i < N; i++) OrderGlobal[i] = i;
        Array.Sort(OrderGlobal, (a, b) => Deg[b].CompareTo(Deg[a]));

        int threads = Math.Max(1, Environment.ProcessorCount - 1);
        int nextIdx = -1;
        var po = new ParallelOptions { MaxDegreeOfParallelism = threads };
        Parallel.For(0, threads, po, _ =>
        {
            var state = new SearchState(W, N);
            while (true)
            {
                int i = Interlocked.Increment(ref nextIdx);
                if (i >= N) break;
                if (TimeExceeded()) break;
                int v = OrderGlobal[i];

                var laterMask = new ulong[W];
                for (int j = i + 1; j < N; j++)
                {
                    int u = OrderGlobal[j];
                    laterMask[u >> 6] |= 1UL << (u & 63);
                }
                var P = new ulong[W];
                var nrow = Adj[v];
                for (int w = 0; w < W; w++) P[w] = nrow[w] & laterMask[w];

                state.CurrentClique[0] = v;
                state.CurrentSize = 1;
                int gBest; lock (BestLock) gBest = BestSize;
                state.LocalBest = gBest;
                state.LocalBestClique = null;

                Expand(state, P);

                if (state.LocalBestClique != null) UpdateBest(state.LocalBestClique);
            }
        });
    }

    class SearchState
    {
        public int[] CurrentClique;
        public int CurrentSize;
        public int LocalBest;
        public int[] LocalBestClique;
        public SearchState(int w, int n)
        {
            CurrentClique = new int[n + 1];
        }
    }

    static void Expand(SearchState s, ulong[] P)
    {
        if (TimeExceeded()) return;

        int pc = PopCount(P);
        if (pc == 0)
        {
            int gBest; lock (BestLock) gBest = BestSize;
            int bound = Math.Max(s.LocalBest, gBest);
            if (s.CurrentSize > bound)
            {
                s.LocalBest = s.CurrentSize;
                var c = new int[s.CurrentSize];
                Array.Copy(s.CurrentClique, c, s.CurrentSize);
                s.LocalBestClique = c;
            }
            return;
        }

        var verts = new int[pc];
        var colors = new int[pc];
        ColorSort(P, verts, colors);

        int gBest2; lock (BestLock) gBest2 = BestSize;
        int curBound = Math.Max(s.LocalBest, gBest2);

        for (int i = verts.Length - 1; i >= 0; i--)
        {
            if (s.CurrentSize + colors[i] <= curBound) return;
            int v = verts[i];
            s.CurrentClique[s.CurrentSize++] = v;

            var newP = new ulong[W];
            var nrow = Adj[v];
            for (int w = 0; w < W; w++) newP[w] = P[w] & nrow[w];
            newP[v >> 6] &= ~(1UL << (v & 63));

            Expand(s, newP);

            s.CurrentSize--;
            P[v >> 6] &= ~(1UL << (v & 63));

            lock (BestLock) gBest2 = BestSize;
            curBound = Math.Max(s.LocalBest, gBest2);

            if (TimeExceeded()) return;
        }
    }

    static int PopCount(ulong[] a)
    {
        int s = 0;
        for (int i = 0; i < a.Length; i++) s += BitOperations.PopCount(a[i]);
        return s;
    }

    static void ColorSort(ulong[] P, int[] outVerts, int[] outColors)
    {
        int n = outVerts.Length;
        var Q = new ulong[W];
        Array.Copy(P, Q, W);
        int color = 0, idx = 0;
        var R = new ulong[W];
        while (idx < n)
        {
            color++;
            Array.Copy(Q, R, W);
            while (true)
            {
                int v = -1;
                for (int w = 0; w < W; w++)
                {
                    ulong b = R[w];
                    if (b != 0) { v = (w << 6) + BitOperations.TrailingZeroCount(b); break; }
                }
                if (v < 0) break;
                outVerts[idx] = v;
                outColors[idx] = color;
                idx++;
                ulong mask = 1UL << (v & 63);
                int wv = v >> 6;
                Q[wv] &= ~mask;
                R[wv] &= ~mask;
                var nrow = Adj[v];
                for (int w = 0; w < W; w++) R[w] &= ~nrow[w];
                if (idx >= n) break;
            }
        }
    }

    static void WriteOutput()
    {
        int[] clique;
        int size;
        lock (BestLock) { clique = BestClique; size = BestSize; }
        if (size == 0 && N > 0) { clique = new int[] { 0 }; size = 1; }
        var sb = new StringBuilder();
        sb.Append(size).Append('\n');
        for (int i = 0; i < size; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(clique[i]);
        }
        sb.Append('\n');
        Console.Out.Write(sb.ToString());
        Console.Out.Flush();
    }
}
