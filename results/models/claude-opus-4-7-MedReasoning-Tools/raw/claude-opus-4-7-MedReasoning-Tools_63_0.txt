declare const require: any;
declare const process: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid: string[] = data.grid;
const robots: { id: number; x: number; y: number }[] = data.robots;
const jobs: { id: number; x: number; y: number; priority: number }[] = data.jobs;

const H = grid.length;
const W = H > 0 ? grid[0].length : 0;

function bfs(sx: number, sy: number): Int32Array {
  const dist = new Int32Array(H * W);
  dist.fill(-1);
  if (sx < 0 || sy < 0 || sx >= W || sy >= H) return dist;
  if (grid[sy].charCodeAt(sx) === 35) return dist;
  const queue = new Int32Array(H * W);
  let head = 0, tail = 0;
  const start = sy * W + sx;
  dist[start] = 0;
  queue[tail++] = start;
  while (head < tail) {
    const cur = queue[head++];
    const cx = cur % W;
    const cy = (cur - cx) / W;
    const d = dist[cur] + 1;
    if (cx + 1 < W) {
      const n = cur + 1;
      if (dist[n] === -1 && grid[cy].charCodeAt(cx + 1) !== 35) {
        dist[n] = d; queue[tail++] = n;
      }
    }
    if (cx - 1 >= 0) {
      const n = cur - 1;
      if (dist[n] === -1 && grid[cy].charCodeAt(cx - 1) !== 35) {
        dist[n] = d; queue[tail++] = n;
      }
    }
    if (cy + 1 < H) {
      const n = cur + W;
      if (dist[n] === -1 && grid[cy + 1].charCodeAt(cx) !== 35) {
        dist[n] = d; queue[tail++] = n;
      }
    }
    if (cy - 1 >= 0) {
      const n = cur - W;
      if (dist[n] === -1 && grid[cy - 1].charCodeAt(cx) !== 35) {
        dist[n] = d; queue[tail++] = n;
      }
    }
  }
  return dist;
}

// Greedy: sort jobs by priority desc, then by id asc; assign each to nearest robot by BFS distance from robot's current position. Update robot position.
const jobOrder = jobs.slice().sort((a, b) => {
  if (b.priority !== a.priority) return b.priority - a.priority;
  return a.id - b.id;
});

const rx: number[] = robots.map(r => r.x);
const ry: number[] = robots.map(r => r.y);

let maxRobotId = 0;
for (const r of robots) if (r.id > maxRobotId) maxRobotId = r.id;
const assignments: number[][] = [];
for (let i = 0; i <= maxRobotId; i++) assignments.push([]);

for (const job of jobOrder) {
  const dist = bfs(job.x, job.y);
  let bestIdx = -1;
  let bestDist = Infinity;
  let bestRobotId = Infinity;
  for (let i = 0; i < robots.length; i++) {
    const idx = ry[i] * W + rx[i];
    const d = dist[idx];
    if (d < 0) continue;
    if (d < bestDist || (d === bestDist && robots[i].id < bestRobotId)) {
      bestDist = d;
      bestIdx = i;
      bestRobotId = robots[i].id;
    }
  }
  if (bestIdx === -1) {
    // Fallback: pick robot with smallest id
    bestIdx = 0;
    for (let i = 1; i < robots.length; i++) {
      if (robots[i].id < robots[bestIdx].id) bestIdx = i;
    }
  }
  assignments[robots[bestIdx].id].push(job.id);
  rx[bestIdx] = job.x;
  ry[bestIdx] = job.y;
}

process.stdout.write(JSON.stringify({ assignments }));
