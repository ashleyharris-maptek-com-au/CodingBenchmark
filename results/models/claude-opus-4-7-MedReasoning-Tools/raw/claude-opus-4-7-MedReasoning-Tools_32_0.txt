#include <iostream>
#include <vector>
#include <queue>
#include <tuple>
#include <algorithm>
#include <cstdint>
#include <cstdio>
#include <unordered_map>
#include <limits>

using namespace std;

typedef long long ll;
const ll INF = (ll)1e18;

struct Edge {
    int to;
    ll w;
};

int N, M, K;
vector<vector<Edge>> adj;
vector<int> terminals;

static ll edgeWeight(int u, int v) {
    for (auto& e : adj[u]) if (e.to == v) return e.w;
    return 0;
}

void dijkstra(int src, vector<ll>& dist, vector<int>& parent) {
    dist.assign(N, INF);
    parent.assign(N, -1);
    dist[src] = 0;
    priority_queue<pair<ll,int>, vector<pair<ll,int>>, greater<>> pq;
    pq.push({0, src});
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;
        for (auto& e : adj[u]) {
            if (d + e.w < dist[e.to]) {
                dist[e.to] = d + e.w;
                parent[e.to] = u;
                pq.push({dist[e.to], e.to});
            }
        }
    }
}

struct DSU {
    vector<int> p, r;
    DSU(int n): p(n), r(n, 0) { for (int i = 0; i < n; i++) p[i] = i; }
    int find(int x) { while (p[x] != x) { p[x] = p[p[x]]; x = p[x]; } return x; }
    bool unite(int a, int b) {
        a = find(a); b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) r[a]++;
        return true;
    }
};

pair<ll, vector<pair<int,int>>> mstHeuristic() {
    int T = (int)terminals.size();
    vector<vector<ll>> tdist(T);
    vector<vector<int>> tparent(T);
    for (int i = 0; i < T; i++) {
        dijkstra(terminals[i], tdist[i], tparent[i]);
    }

    // Prim's MST on metric closure of terminals
    vector<ll> key(T, INF);
    vector<int> par(T, -1);
    vector<bool> inMST(T, false);
    key[0] = 0;
    for (int it = 0; it < T; it++) {
        int u = -1;
        ll best = INF;
        for (int i = 0; i < T; i++) {
            if (!inMST[i] && key[i] < best) { best = key[i]; u = i; }
        }
        if (u == -1) break;
        inMST[u] = true;
        for (int v = 0; v < T; v++) {
            if (!inMST[v]) {
                ll d = tdist[u][terminals[v]];
                if (d < key[v]) { key[v] = d; par[v] = u; }
            }
        }
    }

    // Collect edges along chosen shortest paths
    vector<tuple<ll,int,int>> subEdges;
    auto addPath = [&](int ti, int tj) {
        int v = terminals[tj];
        while (tparent[ti][v] != -1) {
            int u = tparent[ti][v];
            subEdges.push_back({edgeWeight(u, v), u, v});
            v = u;
        }
    };
    for (int i = 1; i < T; i++) {
        if (par[i] != -1) addPath(par[i], i);
    }

    // MST over collected edges (dedupes)
    sort(subEdges.begin(), subEdges.end());
    DSU dsu(N);
    vector<pair<int,int>> treeEdges;
    ll total = 0;
    for (auto& [w, u, v] : subEdges) {
        if (dsu.unite(u, v)) {
            treeEdges.push_back({u, v});
            total += w;
        }
    }

    // Build tree adjacency, prune non-terminal leaves
    vector<unordered_map<int, ll>> tadj(N);
    for (auto& [u, v] : treeEdges) {
        ll w = edgeWeight(u, v);
        tadj[u][v] = w;
        tadj[v][u] = w;
    }
    vector<bool> isTerm(N, false);
    for (int t : terminals) isTerm[t] = true;

    bool changed = true;
    while (changed) {
        changed = false;
        for (int u = 0; u < N; u++) {
            if (!isTerm[u] && tadj[u].size() == 1) {
                auto it = tadj[u].begin();
                int v = it->first;
                ll w = it->second;
                total -= w;
                tadj[v].erase(u);
                tadj[u].clear();
                changed = true;
            }
        }
    }

    vector<pair<int,int>> finalEdges;
    for (int u = 0; u < N; u++) {
        for (auto& kv : tadj[u]) {
            int v = kv.first;
            if (u < v) finalEdges.push_back({u, v});
        }
    }
    return {total, finalEdges};
}

pair<ll, vector<pair<int,int>>> dreyfusWagner() {
    int T = (int)terminals.size();
    int full = 1 << T;
    vector<vector<ll>> dp(full, vector<ll>(N, INF));
    vector<vector<int>> ptype(full, vector<int>(N, 0));
    vector<vector<int>> pS(full, vector<int>(N, 0));
    vector<vector<int>> pV(full, vector<int>(N, 0));

    for (int i = 0; i < T; i++) {
        int s = 1 << i;
        dp[s][terminals[i]] = 0;
        ptype[s][terminals[i]] = 0;
    }

    for (int S = 1; S < full; S++) {
        // Skip singletons for split step (no proper non-empty splits)
        // Split step
        for (int v = 0; v < N; v++) {
            for (int S1 = (S - 1) & S; S1 > 0; S1 = (S1 - 1) & S) {
                int S2 = S ^ S1;
                if (S2 == 0) continue;
                if (dp[S1][v] >= INF || dp[S2][v] >= INF) continue;
                ll val = dp[S1][v] + dp[S2][v];
                if (val < dp[S][v]) {
                    dp[S][v] = val;
                    ptype[S][v] = 1;
                    pS[S][v] = S1;
                    pV[S][v] = v;
                }
            }
        }
        // Relax via shortest path
        priority_queue<pair<ll,int>, vector<pair<ll,int>>, greater<>> pq;
        for (int v = 0; v < N; v++) {
            if (dp[S][v] < INF) pq.push({dp[S][v], v});
        }
        while (!pq.empty()) {
            auto [d, u] = pq.top(); pq.pop();
            if (d > dp[S][u]) continue;
            for (auto& e : adj[u]) {
                ll nd = d + e.w;
                if (nd < dp[S][e.to]) {
                    dp[S][e.to] = nd;
                    ptype[S][e.to] = 2;
                    pS[S][e.to] = S;
                    pV[S][e.to] = u;
                    pq.push({nd, e.to});
                }
            }
        }
    }

    int FULL = full - 1;
    int bestV = 0;
    ll bestC = INF;
    for (int v = 0; v < N; v++) {
        if (dp[FULL][v] < bestC) { bestC = dp[FULL][v]; bestV = v; }
    }

    vector<pair<int,int>> result;
    vector<pair<int,int>> stk;
    stk.push_back({FULL, bestV});
    while (!stk.empty()) {
        auto [S, v] = stk.back(); stk.pop_back();
        int t = ptype[S][v];
        if (t == 0) continue;
        if (t == 1) {
            int S1 = pS[S][v];
            int S2 = S ^ S1;
            if (S1) stk.push_back({S1, v});
            if (S2) stk.push_back({S2, v});
        } else {
            int u = pV[S][v];
            result.push_back({u, v});
            stk.push_back({S, u});
        }
    }

    DSU dsu(N);
    vector<pair<int,int>> finalE;
    ll total = 0;
    for (auto& [u, v] : result) {
        if (dsu.unite(u, v)) {
            finalE.push_back({u, v});
            total += edgeWeight(u, v);
        }
    }
    return {total, finalE};
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> N >> M >> K)) return 0;
    adj.assign(N, {});
    for (int i = 0; i < M; i++) {
        int u, v; ll w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }
    terminals.resize(K);
    for (int i = 0; i < K; i++) cin >> terminals[i];

    if (K <= 1) {
        cout << "0 0\n";
        return 0;
    }

    pair<ll, vector<pair<int,int>>> ans;
    if (K <= 12 && (ll)N * (1LL << K) <= 8000000LL) {
        ans = dreyfusWagner();
    } else {
        ans = mstHeuristic();
    }

    cout << ans.first << " " << ans.second.size() << "\n";
    for (auto& pr : ans.second) {
        cout << pr.first << " " << pr.second << "\n";
    }
    return 0;
}
