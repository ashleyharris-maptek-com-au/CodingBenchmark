const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depot = data.depot;
const K = data.truckCount;
const pkgs = data.packages;
const N = pkgs.length;

const dx = depot[0], dy = depot[1];
const px = new Float64Array(N), py = new Float64Array(N);
const ids = new Array(N);
for (let i = 0; i < N; i++) {
  px[i] = pkgs[i].x;
  py[i] = pkgs[i].y;
  ids[i] = pkgs[i].id;
}

function dist(ax, ay, bx, by) {
  const dxv = ax - bx, dyv = ay - by;
  return Math.sqrt(dxv * dxv + dyv * dyv);
}

const d0 = new Float64Array(N);
for (let i = 0; i < N; i++) d0[i] = dist(dx, dy, px[i], py[i]);

if (N === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

function buildRoutesCW() {
  const route = new Int32Array(N);
  const next = new Int32Array(N);
  const prev = new Int32Array(N);
  const isStart = new Uint8Array(N);
  const isEnd = new Uint8Array(N);
  const routeSize = new Int32Array(N);
  for (let i = 0; i < N; i++) {
    route[i] = i;
    next[i] = -1;
    prev[i] = -1;
    isStart[i] = 1;
    isEnd[i] = 1;
    routeSize[i] = 1;
  }
  let numRoutes = N;

  const KNN = Math.min(N - 1, N > 2000 ? 25 : (N > 500 ? 50 : N - 1));

  const savings = [];
  if (KNN === N - 1 && N <= 1500) {
    for (let i = 0; i < N; i++) {
      for (let j = i + 1; j < N; j++) {
        const s = d0[i] + d0[j] - dist(px[i], py[i], px[j], py[j]);
        if (s > 0) savings.push([s, i, j]);
      }
    }
  } else {
    const seen = new Set();
    for (let i = 0; i < N; i++) {
      const cand = [];
      for (let j = 0; j < N; j++) {
        if (j === i) continue;
        cand.push([dist(px[i], py[i], px[j], py[j]), j]);
      }
      cand.sort((a, b) => a[0] - b[0]);
      const lim = Math.min(KNN, cand.length);
      for (let k = 0; k < lim; k++) {
        const j = cand[k][1];
        const a = i < j ? i : j;
        const b = i < j ? j : i;
        const key = a * N + b;
        if (seen.has(key)) continue;
        seen.add(key);
        const s = d0[a] + d0[b] - cand[k][0];
        if (s > 0) savings.push([s, a, b]);
      }
    }
  }

  savings.sort((a, b) => b[0] - a[0]);

  const minRoutes = Math.min(K, N);

  function reverseRoute(startNode) {
    const nodes = [];
    let cur = startNode;
    while (cur !== -1) { nodes.push(cur); cur = next[cur]; }
    for (let k = 0; k < nodes.length; k++) {
      const n = nodes[k];
      const t = next[n]; next[n] = prev[n]; prev[n] = t;
    }
    const oldEnd = nodes[nodes.length - 1];
    isStart[startNode] = 0; isEnd[startNode] = 1;
    isStart[oldEnd] = 1; isEnd[oldEnd] = 0;
  }

  for (let idx = 0; idx < savings.length; idx++) {
    if (numRoutes <= minRoutes) break;
    const ent = savings[idx];
    const i = ent[1], j = ent[2];
    if (route[i] === route[j]) continue;
    if (!(isStart[i] || isEnd[i])) continue;
    if (!(isStart[j] || isEnd[j])) continue;

    const ri = route[i], rj = route[j];

    if (!isEnd[i]) {
      reverseRoute(i);
    }
    if (!isStart[j]) {
      let f = j;
      while (next[f] !== -1) f = next[f];
      const allNodes = [];
      let s = f;
      while (s !== -1) { allNodes.push(s); s = prev[s]; }
      for (let k = 0; k < allNodes.length; k++) {
        const n = allNodes[k];
        const t = next[n]; next[n] = prev[n]; prev[n] = t;
      }
      isStart[j] = 1; isEnd[j] = 0;
      isStart[f] = 0; isEnd[f] = 1;
    }

    next[i] = j;
    prev[j] = i;
    isEnd[i] = 0;
    isStart[j] = 0;

    let cur = j;
    while (cur !== -1) {
      route[cur] = ri;
      cur = next[cur];
    }
    routeSize[ri] += routeSize[rj];
    numRoutes--;
  }

  const routes = [];
  const visited = new Uint8Array(N);
  for (let i = 0; i < N; i++) {
    if (visited[i]) continue;
    if (!isStart[i]) continue;
    const r = [];
    let cur = i;
    while (cur !== -1) {
      visited[cur] = 1;
      r.push(cur);
      cur = next[cur];
    }
    routes.push(r);
  }
  for (let i = 0; i < N; i++) {
    if (!visited[i]) routes.push([i]);
  }

  return routes;
}

let routes = buildRoutesCW();

function routeDist(r) {
  if (r.length === 0) return 0;
  let d = d0[r[0]] + d0[r[r.length - 1]];
  for (let i = 0; i + 1 < r.length; i++) d += dist(px[r[i]], py[r[i]], px[r[i + 1]], py[r[i + 1]]);
  return d;
}

while (routes.length > K) {
  routes.sort((a, b) => a.length - b.length);
  const a = routes.shift();
  const b = routes.shift();
  const aRev = a.slice().reverse();
  const bRev = b.slice().reverse();
  const opts = [a.concat(b), a.concat(bRev), aRev.concat(b), aRev.concat(bRev)];
  let best = opts[0], bd = routeDist(opts[0]);
  for (let i = 1; i < 4; i++) { const d = routeDist(opts[i]); if (d < bd) { bd = d; best = opts[i]; } }
  routes.unshift(best);
}

function twoOpt(route) {
  const n = route.length;
  if (n < 4) return route;
  const rx = new Float64Array(n + 2);
  const ry = new Float64Array(n + 2);
  rx[0] = dx; ry[0] = dy;
  for (let i = 0; i < n; i++) { rx[i + 1] = px[route[i]]; ry[i + 1] = py[route[i]]; }
  rx[n + 1] = dx; ry[n + 1] = dy;
  const order = new Int32Array(n + 2);
  for (let i = 0; i < n + 2; i++) order[i] = i;
  let improved = true;
  let passes = 0;
  const maxPasses = n > 1000 ? 2 : (n > 300 ? 5 : 20);
  while (improved && passes < maxPasses) {
    improved = false;
    passes++;
    for (let i = 1; i < n; i++) {
      const a = order[i - 1], b = order[i];
      const ax = rx[a], ay = ry[a], bx = rx[b], by = ry[b];
      const dab = dist(ax, ay, bx, by);
      for (let j = i + 1; j < n + 1; j++) {
        const c = order[j], d = order[j + 1];
        const cx = rx[c], cy = ry[c], dxv = rx[d], dyv = ry[d];
        const dcd = dist(cx, cy, dxv, dyv);
        const dac = dist(ax, ay, cx, cy);
        const dbd = dist(bx, by, dxv, dyv);
        if (dac + dbd + 1e-12 < dab + dcd) {
          let lo = i, hi = j;
          while (lo < hi) { const t = order[lo]; order[lo] = order[hi]; order[hi] = t; lo++; hi--; }
          improved = true;
        }
      }
    }
  }
  const out = new Array(n);
  for (let i = 0; i < n; i++) out[i] = route[order[i + 1] - 1];
  return out;
}

if (N <= 5000) {
  for (let i = 0; i < routes.length; i++) {
    if (routes[i].length <= 2000) routes[i] = twoOpt(routes[i]);
  }
}

const outRoutes = routes.map(r => r.map(idx => ids[idx]));
process.stdout.write(JSON.stringify({ routes: outRoutes }));
