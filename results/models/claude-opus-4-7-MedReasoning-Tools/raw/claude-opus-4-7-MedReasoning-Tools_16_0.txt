#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <vector>
#include <string>
#include <algorithm>
#include <queue>
#include <random>
#include <chrono>
#include <numeric>
#include <climits>

using namespace std;
using ll = long long;

struct Task {
    int machine;
    int duration;
};

int J, M;
vector<vector<Task>> jobs;
vector<vector<int>> startTimes;

static vector<char> inbuf_v;
static int inpos = 0, inlen = 0;

static inline int readInt() {
    while (inpos < inlen && (inbuf_v[inpos] < '0' || inbuf_v[inpos] > '9') && inbuf_v[inpos] != '-') inpos++;
    int sign = 1;
    if (inpos < inlen && inbuf_v[inpos] == '-') { sign = -1; inpos++; }
    int x = 0;
    while (inpos < inlen && inbuf_v[inpos] >= '0' && inbuf_v[inpos] <= '9') {
        x = x * 10 + (inbuf_v[inpos++] - '0');
    }
    return x * sign;
}

void readInput() {
    inbuf_v.resize(1 << 16);
    inlen = 0;
    while (true) {
        if ((int)inbuf_v.size() - inlen < (1 << 15)) inbuf_v.resize(inbuf_v.size() * 2);
        int n = (int)fread(inbuf_v.data() + inlen, 1, inbuf_v.size() - inlen, stdin);
        if (n <= 0) break;
        inlen += n;
    }
    J = readInt();
    M = readInt();
    jobs.assign(J, {});
    for (int j = 0; j < J; j++) {
        int T = readInt();
        jobs[j].resize(T);
        for (int i = 0; i < T; i++) {
            jobs[j][i].machine = readInt();
            jobs[j][i].duration = readInt();
        }
    }
}

// Greedy list scheduler: pick ready op with earliest start; tiebreak by rule.
// Rules: 0=SPT,1=LPT,2=MWKR,3=LWKR,4=FIFO,5=random
ll greedySchedule(int rule, mt19937 *rng = nullptr) {
    vector<int> nextOp(J, 0);
    vector<ll> jobAvail(J, 0);
    vector<ll> machAvail(M, 0);
    startTimes.assign(J, {});
    for (int j = 0; j < J; j++) startTimes[j].assign((int)jobs[j].size(), 0);

    vector<ll> remWork(J, 0);
    for (int j = 0; j < J; j++)
        for (auto &t : jobs[j]) remWork[j] += t.duration;

    int totalOps = 0;
    for (int j = 0; j < J; j++) totalOps += (int)jobs[j].size();

    int scheduled = 0;
    ll makespan = 0;

    while (scheduled < totalOps) {
        ll bestStart = LLONG_MAX;
        for (int j = 0; j < J; j++) {
            if (nextOp[j] >= (int)jobs[j].size()) continue;
            int m = jobs[j][nextOp[j]].machine;
            ll s = max(jobAvail[j], machAvail[m]);
            if (s < bestStart) bestStart = s;
        }
        int chosen = -1;
        ll chosenKey = LLONG_MAX;
        ll chosenKey2 = LLONG_MAX;
        for (int j = 0; j < J; j++) {
            if (nextOp[j] >= (int)jobs[j].size()) continue;
            int m = jobs[j][nextOp[j]].machine;
            ll s = max(jobAvail[j], machAvail[m]);
            if (s != bestStart) continue;
            int dur = jobs[j][nextOp[j]].duration;
            ll key;
            switch (rule) {
                case 0: key = dur; break;
                case 1: key = -(ll)dur; break;
                case 2: key = -remWork[j]; break;
                case 3: key = remWork[j]; break;
                case 4: key = j; break;
                default: key = rng ? (ll)((*rng)()) : j; break;
            }
            ll key2 = j;
            if (key < chosenKey || (key == chosenKey && key2 < chosenKey2)) {
                chosenKey = key;
                chosenKey2 = key2;
                chosen = j;
            }
        }
        if (chosen < 0) break;
        int j = chosen;
        int i = nextOp[j];
        int m = jobs[j][i].machine;
        int d = jobs[j][i].duration;
        ll s = max(jobAvail[j], machAvail[m]);
        startTimes[j][i] = (int)s;
        jobAvail[j] = s + d;
        machAvail[m] = s + d;
        remWork[j] -= d;
        nextOp[j]++;
        scheduled++;
        if (s + d > makespan) makespan = s + d;
    }
    return makespan;
}

// Giffler-Thompson active schedule generation
ll gtSchedule(int rule, mt19937 *rng = nullptr) {
    vector<int> nextOp(J, 0);
    vector<ll> jobAvail(J, 0);
    vector<ll> machAvail(M, 0);
    startTimes.assign(J, {});
    for (int j = 0; j < J; j++) startTimes[j].assign((int)jobs[j].size(), 0);

    vector<ll> remWork(J, 0);
    for (int j = 0; j < J; j++)
        for (auto &t : jobs[j]) remWork[j] += t.duration;

    int totalOps = 0;
    for (int j = 0; j < J; j++) totalOps += (int)jobs[j].size();

    int scheduled = 0;
    ll makespan = 0;

    while (scheduled < totalOps) {
        ll minC = LLONG_MAX;
        int minCmach = -1;
        for (int j = 0; j < J; j++) {
            if (nextOp[j] >= (int)jobs[j].size()) continue;
            int m = jobs[j][nextOp[j]].machine;
            ll s = max(jobAvail[j], machAvail[m]);
            ll c = s + jobs[j][nextOp[j]].duration;
            if (c < minC) {
                minC = c;
                minCmach = m;
            }
        }
        if (minCmach < 0) break;
        ll minCstar = minC;

        int chosen = -1;
        ll chosenKey = LLONG_MAX;
        ll chosenKey2 = LLONG_MAX;
        for (int j = 0; j < J; j++) {
            if (nextOp[j] >= (int)jobs[j].size()) continue;
            int m = jobs[j][nextOp[j]].machine;
            if (m != minCmach) continue;
            ll s = max(jobAvail[j], machAvail[m]);
            if (s >= minCstar) continue;
            int dur = jobs[j][nextOp[j]].duration;
            ll key;
            switch (rule) {
                case 0: key = dur; break;
                case 1: key = -(ll)dur; break;
                case 2: key = -remWork[j]; break;
                case 3: key = remWork[j]; break;
                case 4: key = j; break;
                default: key = rng ? (ll)((*rng)()) : j; break;
            }
            ll key2 = j;
            if (key < chosenKey || (key == chosenKey && key2 < chosenKey2)) {
                chosenKey = key;
                chosenKey2 = key2;
                chosen = j;
            }
        }
        if (chosen < 0) break;
        int j = chosen;
        int i = nextOp[j];
        int m = jobs[j][i].machine;
        int d = jobs[j][i].duration;
        ll s = max(jobAvail[j], machAvail[m]);
        startTimes[j][i] = (int)s;
        jobAvail[j] = s + d;
        machAvail[m] = s + d;
        remWork[j] -= d;
        nextOp[j]++;
        scheduled++;
        if (s + d > makespan) makespan = s + d;
    }
    return makespan;
}

int main() {
    auto t0 = chrono::steady_clock::now();
    readInput();

    ll totalOps = 0;
    for (int j = 0; j < J; j++) totalOps += (ll)jobs[j].size();

    double budgetSec;
    if (totalOps < 200) budgetSec = 1.5;
    else if (totalOps < 2000) budgetSec = 2.5;
    else if (totalOps < 20000) budgetSec = 3.5;
    else budgetSec = 5.0;

    vector<int> rules = {2, 1, 0, 3, 4};

    ll bestMakespan = LLONG_MAX;
    vector<vector<int>> bestStarts;

    auto tryUpdate = [&](ll ms) {
        if (ms < bestMakespan) {
            bestMakespan = ms;
            bestStarts = startTimes;
        }
    };

    for (int r : rules) {
        ll ms = gtSchedule(r);
        tryUpdate(ms);
        auto t1 = chrono::steady_clock::now();
        if (chrono::duration<double>(t1 - t0).count() > budgetSec * 0.5) break;
    }

    for (int r : rules) {
        auto t1 = chrono::steady_clock::now();
        if (chrono::duration<double>(t1 - t0).count() > budgetSec * 0.7) break;
        ll ms = greedySchedule(r);
        tryUpdate(ms);
    }

    mt19937 rng(12345);
    int trials = 0;
    while (true) {
        auto t1 = chrono::steady_clock::now();
        if (chrono::duration<double>(t1 - t0).count() > budgetSec) break;
        ll ms = gtSchedule(5, &rng);
        tryUpdate(ms);
        trials++;
        if (trials > 10000) break;
    }

    if (bestStarts.empty()) bestStarts = startTimes;

    string out;
    out.reserve((size_t)totalOps * 16);
    char buf[64];
    for (int j = 0; j < J; j++) {
        int T = (int)jobs[j].size();
        for (int i = 0; i < T; i++) {
            if (i) out.push_back(' ');
            int n = snprintf(buf, sizeof(buf), "%d %d", bestStarts[j][i], jobs[j][i].machine);
            out.append(buf, n);
        }
        out.push_back('\n');
    }
    fwrite(out.data(), 1, out.size(), stdout);
    return 0;
}
