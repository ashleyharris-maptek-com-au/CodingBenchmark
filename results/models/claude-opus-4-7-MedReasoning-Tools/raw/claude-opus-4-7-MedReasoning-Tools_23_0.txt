use std::io::{self, BufRead, Write, BufWriter};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Default, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

#[derive(Default)]
struct MapInfo {
    width: usize,
    height: usize,
    gravity: f64,
    max_thrust: f64,
    _fuel: f64,
    _start_x: f64,
    _start_y: f64,
    target_x: f64,
    target_y: f64,
    grid: Vec<u8>, // 0=empty,1=obstacle,2=landing
    ground: Vec<i32>, // first solid/landing row per column from top
}

impl MapInfo {
    fn col_ground(&self, x: i32) -> f64 {
        if x < 0 || x >= self.width as i32 {
            return self.height as f64;
        }
        self.ground[x as usize] as f64
    }
}

fn read_initial<R: BufRead>(rdr: &mut R) -> io::Result<MapInfo> {
    let mut buf = String::new();
    let mut header = String::new();
    loop {
        buf.clear();
        if rdr.read_line(&mut buf)? == 0 { break; }
        let s = buf.trim();
        if !s.is_empty() { header = s.to_string(); break; }
    }
    let hp: Vec<&str> = header.split_whitespace().collect();
    let width: usize = hp.get(0).and_then(|v| v.parse().ok()).unwrap_or(100);
    let height: usize = hp.get(1).and_then(|v| v.parse().ok()).unwrap_or(50);
    let gravity: f64 = hp.get(2).and_then(|v| v.parse().ok()).unwrap_or(1.6);
    let max_thrust: f64 = hp.get(3).and_then(|v| v.parse().ok()).unwrap_or(15.0);
    let fuel: f64 = hp.get(4).and_then(|v| v.parse().ok()).unwrap_or(100.0);

    let mut line2 = String::new();
    loop {
        buf.clear();
        if rdr.read_line(&mut buf)? == 0 { break; }
        let s = buf.trim();
        if !s.is_empty() { line2 = s.to_string(); break; }
    }
    let lp: Vec<&str> = line2.split_whitespace().collect();
    let start_x: f64 = lp.get(0).and_then(|v| v.parse().ok()).unwrap_or(0.0);
    let start_y: f64 = lp.get(1).and_then(|v| v.parse().ok()).unwrap_or(0.0);
    let target_x: f64 = lp.get(2).and_then(|v| v.parse().ok()).unwrap_or(0.0);
    let target_y: f64 = lp.get(3).and_then(|v| v.parse().ok()).unwrap_or(0.0);

    let mut grid = vec![0u8; width * height];
    let mut rows = 0usize;
    while rows < height {
        buf.clear();
        if rdr.read_line(&mut buf)? == 0 { break; }
        let line = buf.trim_end_matches(&['\r', '\n'][..]);
        if line.is_empty() { continue; }
        let bytes = line.as_bytes();
        for x in 0..width {
            let c = if x < bytes.len() { bytes[x] as char } else { '.' };
            grid[rows * width + x] = match c {
                '#' => 1,
                'T' | 'L' => 2,
                _ => 0,
            };
        }
        rows += 1;
    }

    let mut ground = vec![height as i32; width];
    for x in 0..width {
        for y in 0..height {
            let v = grid[y * width + x];
            if v == 1 || v == 2 {
                ground[x] = y as i32;
                break;
            }
        }
    }

    Ok(MapInfo {
        width, height, gravity, max_thrust,
        _fuel: fuel, _start_x: start_x, _start_y: start_y,
        target_x, target_y, grid, ground,
    })
}

fn clamp(v: f64, lo: f64, hi: f64) -> f64 {
    if v < lo { lo } else if v > hi { hi } else { v }
}

fn compute_control(s: &State, m: &MapInfo) -> (f64, f64) {
    let g = m.gravity;
    let max_t = m.max_thrust.max(1e-6);
    let hover = (g / max_t).min(1.0);

    let dx = m.target_x - s.x;
    let dy = m.target_y - s.y;

    let a_brake = (max_t - g).max(0.5);

    // Desired descent rate using sqrt profile to ensure we can brake in time
    let mut desired_vy = if dy > 0.5 {
        let v = (2.0 * 0.5 * a_brake * dy).sqrt();
        clamp(v, 0.5, 10.0)
    } else if dy < -0.5 {
        -1.0
    } else {
        clamp(dy, -0.5, 0.5)
    };
    if dy.abs() < 1.5 {
        desired_vy = clamp(dy * 1.5, -1.0, 1.0);
    }

    let desired_vx = clamp(dx * 0.6, -6.0, 6.0);

    let err_vy = desired_vy - s.vy;
    let needed_up_acc = g + (-err_vy) * 2.0;

    let err_vx = desired_vx - s.vx;
    let needed_h_acc = err_vx * 1.5;

    let a_up = needed_up_acc.max(0.0);
    let a_h = needed_h_acc;

    let mut desired_angle = if a_up > 1e-3 || a_h.abs() > 1e-3 {
        a_h.atan2(a_up)
    } else { 0.0 };
    desired_angle = clamp(desired_angle, -0.9, 0.9);
    if dy < 6.0 { desired_angle = clamp(desired_angle, -0.35, 0.35); }
    if dy < 2.5 { desired_angle = clamp(desired_angle, -0.15, 0.15); }
    if dy < 1.0 { desired_angle = clamp(desired_angle, -0.05, 0.05); }

    let ang_err = desired_angle - s.angle;
    let turn = clamp(ang_err * 5.0 - s.angular_vel * 1.8, -1.0, 1.0);

    let facing_x = s.angle.sin();
    let facing_y = -s.angle.cos();
    let needed = a_h * facing_x + (-a_up) * facing_y;
    let mut thrust = clamp(needed / max_t, 0.0, 1.0);

    let col = clamp(s.x.round(), 0.0, (m.width as f64) - 1.0) as i32;
    let ground_y = m.col_ground(col);
    let dist_to_ground = (ground_y - s.y).max(0.0);
    if s.vy > 0.0 && dist_to_ground > 0.0 {
        let bd = (s.vy * s.vy) / (2.0 * a_brake);
        if bd > dist_to_ground - 1.2 {
            thrust = 1.0;
        }
    }

    if dy.abs() < 1.0 && s.vy.abs() < 1.0 && dx.abs() < 1.0 {
        thrust = clamp(hover * 0.9, 0.0, 1.0);
    }

    (thrust, turn)
}

fn main() {
    let stdout = io::stdout();
    let stdout = Arc::new(Mutex::new(BufWriter::new(stdout.lock())));

    let state: Arc<Mutex<Option<State>>> = Arc::new(Mutex::new(None));
    let map: Arc<Mutex<Option<MapInfo>>> = Arc::new(Mutex::new(None));
    let ready = Arc::new(Mutex::new(false));

    let state_w = state.clone();
    let map_w = map.clone();
    let ready_w = ready.clone();
    thread::spawn(move || {
        let stdin = io::stdin();
        let mut rdr = stdin.lock();
        if let Ok(m) = read_initial(&mut rdr) {
            *map_w.lock().unwrap() = Some(m);
            *ready_w.lock().unwrap() = true;
        }
        let mut buf = String::new();
        loop {
            buf.clear();
            let n = match rdr.read_line(&mut buf) {
                Ok(n) => n,
                Err(_) => break,
            };
            if n == 0 { break; }
            let line = buf.trim();
            if line.is_empty() { continue; }
            let parts: Vec<&str> = line.split_whitespace().collect();
            if parts.len() < 7 { continue; }
            let p = |i: usize| -> f64 { parts[i].parse().unwrap_or(0.0) };
            let s = State {
                x: p(0), y: p(1), vx: p(2), vy: p(3),
                angle: p(4), angular_vel: p(5), fuel: p(6),
            };
            *state_w.lock().unwrap() = Some(s);
        }
    });

    // Output loop at 50Hz
    let period = Duration::from_millis(20);
    let mut next = Instant::now() + period;
    let start = Instant::now();
    loop {
        let s_opt = *state.lock().unwrap();
        let m_guard = map.lock().unwrap();
        let (thrust, turn) = match (s_opt, m_guard.as_ref()) {
            (Some(s), Some(m)) => compute_control(&s, m),
            _ => (0.0, 0.0),
        };
        drop(m_guard);

        {
            let mut out = stdout.lock().unwrap();
            let _ = writeln!(out, "{:.3} {:.3}", thrust, turn);
            let _ = out.flush();
        }

        let now = Instant::now();
        if next > now { thread::sleep(next - now); }
        next += period;
        if next < Instant::now() { next = Instant::now() + period; }

        if start.elapsed() > Duration::from_secs(60 * 60) { break; }
    }
}
