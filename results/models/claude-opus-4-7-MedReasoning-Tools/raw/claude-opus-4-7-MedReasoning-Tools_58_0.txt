const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.grid;
const H = grid.length;
const W = H > 0 ? grid[0].length : 0;
const robots = data.robots;
const jobs = data.jobs;

const blocked = new Uint8Array(H * W);
for (let y = 0; y < H; y++) {
  const row = grid[y];
  for (let x = 0; x < W; x++) {
    if (row.charCodeAt(x) === 35) blocked[y * W + x] = 1;
  }
}

const robotPos = robots.map(r => ({ x: r.x, y: r.y, id: r.id }));
const numRobots = robots.length;
const assignments = [];
for (let i = 0; i < numRobots; i++) assignments.push([]);
// Map robot.id -> index in robotPos (in case ids aren't 0..N-1)
// Per spec, "array index is the robot id" implies ids are 0..N-1
const idToIdx = new Map();
for (let i = 0; i < numRobots; i++) idToIdx.set(robotPos[i].id, i);

const sortedJobs = jobs.slice().sort((a, b) => {
  if (b.priority !== a.priority) return b.priority - a.priority;
  return a.id - b.id;
});

const distBuf = new Int32Array(H * W);
const queue = new Int32Array(H * W);

function bfsFromJobFindNearestRobot(jx, jy) {
  for (let i = 0; i < H * W; i++) distBuf[i] = -1;
  const start = jy * W + jx;

  // Build robot cell lookup: cellIndex -> sorted array of robot indices
  const robotCells = new Map();
  for (let i = 0; i < numRobots; i++) {
    const r = robotPos[i];
    const c = r.y * W + r.x;
    if (!robotCells.has(c)) robotCells.set(c, []);
    robotCells.get(c).push(i);
  }

  const pickFromCell = (cell) => {
    const idxs = robotCells.get(cell);
    let pick = idxs[0];
    for (const i of idxs) if (i < pick) pick = i;
    return pick;
  };

  if (blocked[start]) {
    let best = -1, bestD = Infinity;
    for (const [cell, idxs] of robotCells) {
      const ry = (cell / W) | 0;
      const rx = cell - ry * W;
      const d = Math.abs(rx - jx) + Math.abs(ry - jy);
      let candIdx = idxs[0];
      for (const i of idxs) if (i < candIdx) candIdx = i;
      if (d < bestD || (d === bestD && candIdx < best)) {
        bestD = d;
        best = candIdx;
      }
    }
    return { robotIdx: best, dist: bestD };
  }

  distBuf[start] = 0;
  if (robotCells.has(start)) {
    return { robotIdx: pickFromCell(start), dist: 0 };
  }

  let head = 0, tail = 0;
  queue[tail++] = start;
  while (head < tail) {
    const cur = queue[head++];
    const cy = (cur / W) | 0;
    const cx = cur - cy * W;
    const nd = distBuf[cur] + 1;

    if (cx > 0) {
      const n = cur - 1;
      if (distBuf[n] === -1 && !blocked[n]) {
        distBuf[n] = nd;
        if (robotCells.has(n)) return { robotIdx: pickFromCell(n), dist: nd };
        queue[tail++] = n;
      }
    }
    if (cx < W - 1) {
      const n = cur + 1;
      if (distBuf[n] === -1 && !blocked[n]) {
        distBuf[n] = nd;
        if (robotCells.has(n)) return { robotIdx: pickFromCell(n), dist: nd };
        queue[tail++] = n;
      }
    }
    if (cy > 0) {
      const n = cur - W;
      if (distBuf[n] === -1 && !blocked[n]) {
        distBuf[n] = nd;
        if (robotCells.has(n)) return { robotIdx: pickFromCell(n), dist: nd };
        queue[tail++] = n;
      }
    }
    if (cy < H - 1) {
      const n = cur + W;
      if (distBuf[n] === -1 && !blocked[n]) {
        distBuf[n] = nd;
        if (robotCells.has(n)) return { robotIdx: pickFromCell(n), dist: nd };
        queue[tail++] = n;
      }
    }
  }

  // Unreachable: lowest-index robot (deterministic fallback)
  let best = 0;
  let bestD = Math.abs(robotPos[0].x - jx) + Math.abs(robotPos[0].y - jy);
  for (let i = 1; i < numRobots; i++) {
    const d = Math.abs(robotPos[i].x - jx) + Math.abs(robotPos[i].y - jy);
    if (d < bestD) { bestD = d; best = i; }
  }
  return { robotIdx: best, dist: bestD };
}

for (const job of sortedJobs) {
  const { robotIdx } = bfsFromJobFindNearestRobot(job.x, job.y);
  const ri = robotIdx >= 0 ? robotIdx : 0;
  const rid = robotPos[ri].id;
  if (rid >= 0 && rid < numRobots) {
    assignments[rid].push(job.id);
  } else {
    assignments[ri].push(job.id);
  }
  robotPos[ri].x = job.x;
  robotPos[ri].y = job.y;
}

process.stdout.write(JSON.stringify({ assignments }));
