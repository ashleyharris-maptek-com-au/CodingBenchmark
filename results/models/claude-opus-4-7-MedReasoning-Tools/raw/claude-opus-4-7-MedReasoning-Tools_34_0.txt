import sys
import math


def dist(a, b):
    return math.hypot(a[0] - b[0], a[1] - b[1])


def route_distance(route, depot, coords):
    if not route:
        return 0.0
    d = dist(depot, coords[route[0]])
    for i in range(len(route) - 1):
        d += dist(coords[route[i]], coords[route[i + 1]])
    d += dist(coords[route[-1]], depot)
    return d


def two_opt_idx(route, depot, coords):
    if len(route) < 3:
        return route
    r = list(route)
    n = len(r)
    improved = True
    iters = 0
    while improved and iters < 50:
        improved = False
        iters += 1
        for i in range(n - 1):
            for j in range(i + 1, n):
                a = depot if i == 0 else coords[r[i - 1]]
                b = coords[r[i]]
                c = coords[r[j]]
                d = depot if j == n - 1 else coords[r[j + 1]]
                delta = (dist(a, c) + dist(b, d)) - (dist(a, b) + dist(c, d))
                if delta < -1e-9:
                    r[i:j + 1] = r[i:j + 1][::-1]
                    improved = True
    return r


def solve():
    data = sys.stdin.read().split()
    idx = 0
    N = int(data[idx]); idx += 1
    K = int(data[idx]); idx += 1
    CAP = float(data[idx]); idx += 1
    depot = (float(data[idx]), float(data[idx + 1])); idx += 2
    coords = []
    demands = []
    for i in range(N):
        x = float(data[idx]); idx += 1
        y = float(data[idx]); idx += 1
        d = float(data[idx]); idx += 1
        coords.append((x, y))
        demands.append(d)

    # Clarke-Wright savings algorithm
    route_of = list(range(N))
    routes = [[i] for i in range(N)]
    loads = list(demands)
    active = [True] * N

    savings = []
    for i in range(N):
        di0 = dist(depot, coords[i])
        for j in range(i + 1, N):
            s = di0 + dist(depot, coords[j]) - dist(coords[i], coords[j])
            if s > 0:
                savings.append((s, i, j))
    savings.sort(reverse=True)

    for s, i, j in savings:
        ri = route_of[i]
        rj = route_of[j]
        if ri == rj:
            continue
        if not active[ri] or not active[rj]:
            continue
        if loads[ri] + loads[rj] > CAP + 1e-9:
            continue
        route_i = routes[ri]
        route_j = routes[rj]
        if i == route_i[0]:
            ri_rev = True
        elif i == route_i[-1]:
            ri_rev = False
        else:
            continue
        if j == route_j[0]:
            rj_rev = False
        elif j == route_j[-1]:
            rj_rev = True
        else:
            continue

        new_i = route_i[::-1] if ri_rev else route_i
        new_j = route_j[::-1] if rj_rev else route_j
        merged = new_i + new_j
        routes[ri] = merged
        loads[ri] = loads[ri] + loads[rj]
        active[rj] = False
        for c in new_j:
            route_of[c] = ri

    final_routes = [routes[i] for i in range(len(routes)) if active[i]]

    # If too many routes, try merging while feasible
    while len(final_routes) > K:
        merged = False
        final_routes.sort(key=lambda r: sum(demands[c] for c in r))
        for a in range(len(final_routes)):
            for b in range(a + 1, len(final_routes)):
                la = sum(demands[c] for c in final_routes[a])
                lb = sum(demands[c] for c in final_routes[b])
                if la + lb <= CAP + 1e-9:
                    final_routes[a] = final_routes[a] + final_routes[b]
                    final_routes.pop(b)
                    merged = True
                    break
            if merged:
                break
        if not merged:
            break

    # 2-opt improvement per route
    final_routes = [two_opt_idx(r, depot, coords) for r in final_routes]

    total = sum(route_distance(r, depot, coords) for r in final_routes)

    out = []
    out.append(f"{total:.6f}")
    out.append(str(len(final_routes)))
    for r in final_routes:
        out.append(" ".join(str(c) for c in r))
    sys.stdout.write("\n".join(out) + "\n")


if __name__ == "__main__":
    solve()
