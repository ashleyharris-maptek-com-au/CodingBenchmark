use std::io::{self, Read, Write, BufWriter};

fn read_input() -> (usize, Vec<Vec<u32>>) {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_ascii_whitespace();
    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();
    let mut adj: Vec<Vec<u32>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v { continue; }
        adj[u].push(v as u32);
        adj[v].push(u as u32);
    }
    for a in adj.iter_mut() {
        a.sort_unstable();
        a.dedup();
    }
    (n, adj)
}

fn greedy(n: usize, adj: &[Vec<u32>]) -> Vec<u32> {
    let mut dominated = vec![false; n];
    let mut in_set = vec![false; n];
    let mut set: Vec<u32> = Vec::new();

    let mut gain: Vec<i32> = (0..n).map(|v| 1 + adj[v].len() as i32).collect();

    let max_gain = gain.iter().copied().max().unwrap_or(0) as usize;
    let mut buckets: Vec<Vec<u32>> = vec![Vec::new(); max_gain + 1];
    for v in 0..n {
        let g = gain[v] as usize;
        buckets[g].push(v as u32);
    }
    let mut cur = max_gain as i32;
    let mut remaining = n;

    while remaining > 0 {
        while cur >= 0 {
            let cu = cur as usize;
            while let Some(&v) = buckets[cu].last() {
                if gain[v as usize] == cur && !in_set[v as usize] {
                    break;
                }
                buckets[cu].pop();
            }
            if buckets[cu].is_empty() {
                cur -= 1;
            } else {
                break;
            }
        }
        if cur < 0 { break; }
        let v = buckets[cur as usize].pop().unwrap() as usize;
        if in_set[v] { continue; }
        in_set[v] = true;
        set.push(v as u32);

        let mut newly: Vec<u32> = Vec::new();
        if !dominated[v] {
            dominated[v] = true;
            newly.push(v as u32);
            remaining -= 1;
        }
        for &u in &adj[v] {
            let uu = u as usize;
            if !dominated[uu] {
                dominated[uu] = true;
                newly.push(u);
                remaining -= 1;
            }
        }
        for &w in &newly {
            let ww = w as usize;
            if !in_set[ww] {
                gain[ww] -= 1;
                let g = gain[ww];
                if g >= 0 {
                    buckets[g as usize].push(w);
                }
            }
            for &x in &adj[ww] {
                let xx = x as usize;
                if !in_set[xx] {
                    gain[xx] -= 1;
                    let g = gain[xx];
                    if g >= 0 {
                        buckets[g as usize].push(x);
                    }
                }
            }
        }
    }

    set
}

fn prune(n: usize, adj: &[Vec<u32>], set: Vec<u32>) -> Vec<u32> {
    let mut in_set = vec![false; n];
    for &v in &set { in_set[v as usize] = true; }
    let mut dom_count = vec![0i32; n];
    for v in 0..n {
        if in_set[v] { dom_count[v] += 1; }
        for &u in &adj[v] {
            if in_set[u as usize] { dom_count[v] += 1; }
        }
    }
    let mut order: Vec<u32> = set.clone();
    order.sort_by_key(|&v| adj[v as usize].len());
    let mut result_in = in_set.clone();
    for &v in &order {
        let vv = v as usize;
        if !result_in[vv] { continue; }
        let mut ok = dom_count[vv] - 1 >= 1;
        if ok {
            for &u in &adj[vv] {
                if dom_count[u as usize] - 1 < 1 {
                    ok = false;
                    break;
                }
            }
        }
        if ok {
            result_in[vv] = false;
            dom_count[vv] -= 1;
            for &u in &adj[vv] {
                dom_count[u as usize] -= 1;
            }
        }
    }
    (0..n as u32).filter(|&v| result_in[v as usize]).collect()
}

fn main() {
    let (n, adj) = read_input();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    if n == 0 {
        writeln!(out, "0").unwrap();
        writeln!(out).unwrap();
        return;
    }

    let mut set = greedy(n, &adj);
    set = prune(n, &adj, set);

    writeln!(out, "{}", set.len()).unwrap();
    let strs: Vec<String> = set.iter().map(|v| v.to_string()).collect();
    writeln!(out, "{}", strs.join(" ")).unwrap();
}
