use std::io::{self, Read, Write, BufWriter};

fn dfs(
    idx: usize,
    n: usize,
    half: usize,
    adj: &Vec<Vec<usize>>,
    side: &mut Vec<u8>,
    count_a: &mut usize,
    count_b: &mut usize,
    cur_cut: &mut usize,
    na: &mut Vec<i32>,
    nb: &mut Vec<i32>,
    best_cut: &mut usize,
    best_side: &mut Vec<u8>,
) {
    if *cur_cut >= *best_cut {
        return;
    }
    if idx == n {
        if *count_a == half {
            *best_cut = *cur_cut;
            *best_side = side.clone();
        }
        return;
    }
    let mut lb_extra: usize = 0;
    for j in idx..n {
        let a = na[j];
        let b = nb[j];
        if a > 0 && b > 0 {
            lb_extra += a.min(b) as usize;
        }
    }
    if *cur_cut + lb_extra >= *best_cut {
        return;
    }

    let remaining = n - idx;
    let restrict = idx == 0;
    let order: [u8; 2] = if *count_a <= *count_b { [1, 2] } else { [2, 1] };

    for &s_try in &order {
        if restrict && s_try != 1 {
            continue;
        }
        if s_try == 1 {
            if *count_a >= half {
                continue;
            }
            if (half - *count_b).saturating_sub(0) > remaining.saturating_sub(1) + 1 {
                // unused guard
            }
            if half > *count_b + remaining - 1 + 1 {
                // not needed; main check below
            }
            // Feasibility: B side must still be fillable
            let b_need = half - *count_b;
            let rem_after = remaining - 1;
            if b_need > rem_after {
                continue;
            }
        } else {
            if *count_b >= half {
                continue;
            }
            let a_need = half - *count_a;
            let rem_after = remaining - 1;
            if a_need > rem_after {
                continue;
            }
        }

        side[idx] = s_try;
        let added: usize = if s_try == 1 {
            *count_a += 1;
            nb[idx] as usize
        } else {
            *count_b += 1;
            na[idx] as usize
        };
        *cur_cut += added;

        for &w in &adj[idx] {
            if side[w] == 0 {
                if s_try == 1 {
                    na[w] += 1;
                } else {
                    nb[w] += 1;
                }
            }
        }

        dfs(idx + 1, n, half, adj, side, count_a, count_b, cur_cut, na, nb, best_cut, best_side);

        for &w in &adj[idx] {
            if side[w] == 0 {
                if s_try == 1 {
                    na[w] -= 1;
                } else {
                    nb[w] -= 1;
                }
            }
        }
        *cur_cut -= added;
        if s_try == 1 {
            *count_a -= 1;
        } else {
            *count_b -= 1;
        }
        side[idx] = 0;
    }
}

fn main() {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut it = s.split_ascii_whitespace();
    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    let half = n / 2;

    let mut side = vec![0u8; n];
    let mut count_a: usize = 0;
    let mut count_b: usize = 0;
    let mut cur_cut: usize = 0;

    let mut best_cut: usize = usize::MAX;
    let mut best_side: Vec<u8> = vec![0u8; n];
    // Initialize a feasible solution: first half = A
    for i in 0..n {
        if i < half {
            best_side[i] = 1;
        } else {
            best_side[i] = 2;
        }
    }
    // compute its cut
    {
        let mut c = 0usize;
        for u in 0..n {
            for &v in &adj[u] {
                if u < v && best_side[u] != best_side[v] {
                    c += 1;
                }
            }
        }
        best_cut = c;
    }

    let mut na = vec![0i32; n];
    let mut nb = vec![0i32; n];

    dfs(
        0, n, half, &adj,
        &mut side, &mut count_a, &mut count_b, &mut cur_cut,
        &mut na, &mut nb,
        &mut best_cut, &mut best_side,
    );

    let mut part_a: Vec<usize> = Vec::with_capacity(half);
    for i in 0..n {
        if best_side[i] == 1 {
            part_a.push(i);
        }
    }

    writeln!(out, "{}", best_cut).unwrap();
    let strs: Vec<String> = part_a.iter().map(|x| x.to_string()).collect();
    writeln!(out, "{}", strs.join(" ")).unwrap();
}
