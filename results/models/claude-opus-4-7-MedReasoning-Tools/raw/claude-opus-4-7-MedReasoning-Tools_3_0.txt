use std::io::{self, Read, Write, BufWriter};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut it = input.split_ascii_whitespace();
    let n: usize = match it.next() { Some(s) => s.parse().unwrap_or(0), None => 0 };
    let m: usize = match it.next() { Some(s) => s.parse().unwrap_or(0), None => 0 };

    if n == 0 {
        return;
    }

    let mut adj: Vec<Vec<u32>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = match it.next() { Some(s) => s.parse().unwrap_or(0), None => break };
        let v: usize = match it.next() { Some(s) => s.parse().unwrap_or(0), None => break };
        if u >= n || v >= n || u == v { continue; }
        adj[u].push(v as u32);
        adj[v].push(u as u32);
    }
    for a in adj.iter_mut() {
        a.sort_unstable();
        a.dedup();
    }

    let mut x = vec![0.0f64; n];
    let mut y = vec![0.0f64; n];

    let area = (n as f64).max(16.0);
    let side = area.sqrt();
    let k = side / ((n as f64).sqrt().max(1.0));

    let mut seed: u64 = 0x9E3779B97F4A7C15;
    let mut rng = || {
        seed ^= seed << 13;
        seed ^= seed >> 7;
        seed ^= seed << 17;
        (seed as f64) / (u64::MAX as f64)
    };

    for i in 0..n {
        x[i] = (rng() - 0.5) * side;
        y[i] = (rng() - 0.5) * side;
    }

    let (iters, use_grid) = if n <= 200 {
        (300, false)
    } else if n <= 1500 {
        (200, true)
    } else if n <= 10000 {
        (120, true)
    } else {
        (60, true)
    };

    let k2 = k * k;
    let mut t = side / 4.0;
    let cool = t / (iters as f64 + 1.0);

    let mut disp_x = vec![0.0f64; n];
    let mut disp_y = vec![0.0f64; n];

    for _iter in 0..iters {
        for i in 0..n {
            disp_x[i] = 0.0;
            disp_y[i] = 0.0;
        }

        if !use_grid {
            for i in 0..n {
                let xi = x[i]; let yi = y[i];
                for j in (i+1)..n {
                    let mut dx = xi - x[j];
                    let mut dy = yi - y[j];
                    let mut d2 = dx*dx + dy*dy;
                    if d2 < 1e-6 {
                        dx = (rng() - 0.5) * 0.01;
                        dy = (rng() - 0.5) * 0.01;
                        d2 = dx*dx + dy*dy + 1e-6;
                    }
                    let d = d2.sqrt();
                    let force = k2 / d;
                    let fx = dx / d * force;
                    let fy = dy / d * force;
                    disp_x[i] += fx; disp_y[i] += fy;
                    disp_x[j] -= fx; disp_y[j] -= fy;
                }
            }
        } else {
            let mut min_x = f64::INFINITY; let mut max_x = f64::NEG_INFINITY;
            let mut min_y = f64::INFINITY; let mut max_y = f64::NEG_INFINITY;
            for i in 0..n {
                if x[i] < min_x { min_x = x[i]; }
                if x[i] > max_x { max_x = x[i]; }
                if y[i] < min_y { min_y = y[i]; }
                if y[i] > max_y { max_y = y[i]; }
            }
            let mut cell = (2.0 * k).max(1e-3);
            let mut gw = (((max_x - min_x) / cell).ceil() as i64 + 1).max(1) as usize;
            let mut gh = (((max_y - min_y) / cell).ceil() as i64 + 1).max(1) as usize;
            let max_cells = 4_000_000usize;
            if gw.saturating_mul(gh) > max_cells {
                let scale = ((gw as f64 * gh as f64) / max_cells as f64).sqrt();
                cell *= scale;
                gw = (((max_x - min_x) / cell).ceil() as i64 + 1).max(1) as usize;
                gh = (((max_y - min_y) / cell).ceil() as i64 + 1).max(1) as usize;
            }

            let mut grid: Vec<Vec<u32>> = vec![Vec::new(); gw * gh];
            let mut cx = vec![0u32; n];
            let mut cy = vec![0u32; n];
            for i in 0..n {
                let xi = (((x[i] - min_x) / cell) as i64).max(0).min(gw as i64 - 1) as usize;
                let yi = (((y[i] - min_y) / cell) as i64).max(0).min(gh as i64 - 1) as usize;
                cx[i] = xi as u32; cy[i] = yi as u32;
                grid[yi * gw + xi].push(i as u32);
            }
            for i in 0..n {
                let xi = x[i]; let yi = y[i];
                let gx = cx[i] as i64; let gy = cy[i] as i64;
                let mut fx_sum = 0.0; let mut fy_sum = 0.0;
                for dy_c in -1..=1i64 {
                    let ny = gy + dy_c;
                    if ny < 0 || ny >= gh as i64 { continue; }
                    for dx_c in -1..=1i64 {
                        let nx = gx + dx_c;
                        if nx < 0 || nx >= gw as i64 { continue; }
                        let cell_idx = (ny as usize) * gw + (nx as usize);
                        for &j_u in &grid[cell_idx] {
                            let j = j_u as usize;
                            if j == i { continue; }
                            let mut dxv = xi - x[j];
                            let mut dyv = yi - y[j];
                            let mut d2 = dxv*dxv + dyv*dyv;
                            if d2 < 1e-6 {
                                dxv = (rng() - 0.5) * 0.01;
                                dyv = (rng() - 0.5) * 0.01;
                                d2 = dxv*dxv + dyv*dyv + 1e-6;
                            }
                            let d = d2.sqrt();
                            if d > 2.0 * cell { continue; }
                            let force = k2 / d;
                            fx_sum += dxv / d * force;
                            fy_sum += dyv / d * force;
                        }
                    }
                }
                disp_x[i] += fx_sum;
                disp_y[i] += fy_sum;
            }
        }

        for u in 0..n {
            for &v_u in &adj[u] {
                let v = v_u as usize;
                if v <= u { continue; }
                let mut dx = x[u] - x[v];
                let mut dy = y[u] - y[v];
                let mut d2 = dx*dx + dy*dy;
                if d2 < 1e-6 {
                    dx = 0.01; dy = 0.0; d2 = dx*dx + dy*dy;
                }
                let d = d2.sqrt();
                let force = d2 / k;
                let fx = dx / d * force;
                let fy = dy / d * force;
                disp_x[u] -= fx; disp_y[u] -= fy;
                disp_x[v] += fx; disp_y[v] += fy;
            }
        }

        for i in 0..n {
            let dx = disp_x[i]; let dy = disp_y[i];
            let d = (dx*dx + dy*dy).sqrt();
            if d > 1e-9 {
                let limited = d.min(t);
                x[i] += dx / d * limited;
                y[i] += dy / d * limited;
            }
        }

        t -= cool;
        if t < 0.001 { t = 0.001; }
    }

    writeln!(out, "{}", n).ok();
    for i in 0..n {
        writeln!(out, "{:.6} {:.6}", x[i], y[i]).ok();
    }
}
