#include <bits/stdc++.h>
using namespace std;

// Maximum Independent Set solver.
// MIS on G == Max Clique on complement(G).
// Approach:
//   - Read graph.
//   - Reductions: isolated vertices (include), degree-1 pendants (include leaf, drop neighbor).
//   - Split into connected components.
//   - Solve each component with bitset Tomita-style max-clique B&B
//     on the component's complement.

using u64 = unsigned long long;

struct Bitset {
    int nWords;
    vector<u64> w;
    Bitset() : nWords(0) {}
    Bitset(int n) : nWords((n + 63) >> 6), w((n + 63) >> 6, 0ULL) {}
    inline void set(int i)   { w[i >> 6] |=  (1ULL << (i & 63)); }
    inline void clr(int i)   { w[i >> 6] &= ~(1ULL << (i & 63)); }
    inline bool test(int i) const { return (w[i >> 6] >> (i & 63)) & 1ULL; }
    inline int popcount() const {
        int c = 0;
        for (auto x : w) c += __builtin_popcountll(x);
        return c;
    }
    inline int firstBit() const {
        for (int i = 0; i < nWords; ++i)
            if (w[i]) return (i << 6) + __builtin_ctzll(w[i]);
        return -1;
    }
};

struct MCSolver {
    int N;
    vector<Bitset> adjC; // complement adjacency in local indexing
    vector<int> best;
    vector<int> curr;
    int bestSize;

    void color(Bitset P, vector<int>& order, vector<int>& colors) {
        order.clear();
        colors.clear();
        int k = 0;
        while (P.firstBit() != -1) {
            ++k;
            Bitset Q = P;
            while (true) {
                int v = Q.firstBit();
                if (v < 0) break;
                order.push_back(v);
                colors.push_back(k);
                P.clr(v);
                Q.clr(v);
                for (int i = 0; i < Q.nWords; ++i)
                    Q.w[i] &= ~adjC[v].w[i];
            }
        }
    }

    void expand(Bitset P) {
        vector<int> order, colors;
        color(P, order, colors);
        for (int i = (int)order.size() - 1; i >= 0; --i) {
            if ((int)curr.size() + colors[i] <= bestSize) return;
            int v = order[i];
            curr.push_back(v);
            Bitset newP(N);
            for (int j = 0; j < P.nWords; ++j)
                newP.w[j] = P.w[j] & adjC[v].w[j];
            if (newP.firstBit() < 0) {
                if ((int)curr.size() > bestSize) {
                    bestSize = (int)curr.size();
                    best = curr;
                }
            } else {
                expand(newP);
            }
            curr.pop_back();
            P.clr(v);
        }
    }

    vector<int> solve() {
        bestSize = 0;
        best.clear();
        curr.clear();
        Bitset P(N);
        for (int i = 0; i < N; ++i) P.set(i);
        if (N > 0) expand(P);
        return best;
    }
};

int n, m;
vector<vector<int>> adj;
vector<char> removed;
vector<char> inSet;

void reduceGraph() {
    bool changed = true;
    while (changed) {
        changed = false;
        for (int v = 0; v < n; ++v) {
            if (removed[v]) continue;
            int deg = 0;
            int onlyNb = -1;
            for (int u : adj[v]) {
                if (!removed[u]) {
                    ++deg;
                    onlyNb = u;
                    if (deg > 1) break;
                }
            }
            if (deg == 0) {
                inSet[v] = 1;
                removed[v] = 1;
                changed = true;
            } else if (deg == 1) {
                inSet[v] = 1;
                removed[v] = 1;
                removed[onlyNb] = 1;
                changed = true;
            }
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> n >> m)) return 0;
    adj.assign(n, {});
    for (int i = 0; i < m; ++i) {
        int u, v;
        if (!(cin >> u >> v)) break;
        if (u == v) continue;
        if (u < 0 || v < 0 || u >= n || v >= n) continue;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    for (int i = 0; i < n; ++i) {
        auto& a = adj[i];
        sort(a.begin(), a.end());
        a.erase(unique(a.begin(), a.end()), a.end());
        a.erase(remove(a.begin(), a.end(), i), a.end());
    }

    removed.assign(n, 0);
    inSet.assign(n, 0);

    reduceGraph();

    vector<int> compId(n, -1);
    vector<vector<int>> comps;
    for (int s = 0; s < n; ++s) {
        if (removed[s] || compId[s] != -1) continue;
        vector<int> stack = {s};
        compId[s] = (int)comps.size();
        vector<int> verts;
        while (!stack.empty()) {
            int v = stack.back(); stack.pop_back();
            verts.push_back(v);
            for (int u : adj[v]) {
                if (removed[u]) continue;
                if (compId[u] == -1) {
                    compId[u] = compId[s];
                    stack.push_back(u);
                }
            }
        }
        comps.push_back(verts);
    }

    for (auto& comp : comps) {
        int k = (int)comp.size();
        if (k == 1) {
            inSet[comp[0]] = 1;
            continue;
        }
        unordered_map<int,int> g2l;
        g2l.reserve(k * 2);
        for (int i = 0; i < k; ++i) g2l[comp[i]] = i;

        MCSolver mc;
        mc.N = k;
        mc.adjC.assign(k, Bitset(k));

        int words = (k + 63) >> 6;
        vector<u64> full(words, 0ULL);
        for (int i = 0; i < k; ++i) full[i >> 6] |= (1ULL << (i & 63));

        for (int i = 0; i < k; ++i) {
            mc.adjC[i].w = full;
            mc.adjC[i].clr(i);
        }
        for (int i = 0; i < k; ++i) {
            int gv = comp[i];
            for (int gu : adj[gv]) {
                if (removed[gu]) continue;
                auto it = g2l.find(gu);
                if (it == g2l.end()) continue;
                mc.adjC[i].clr(it->second);
            }
        }

        vector<int> clique = mc.solve();
        for (int loc : clique) inSet[comp[loc]] = 1;
    }

    vector<int> result;
    for (int i = 0; i < n; ++i) if (inSet[i]) result.push_back(i);

    cout << result.size() << "\n";
    for (size_t i = 0; i < result.size(); ++i) {
        if (i) cout << ' ';
        cout << result[i];
    }
    cout << "\n";
    return 0;
}
