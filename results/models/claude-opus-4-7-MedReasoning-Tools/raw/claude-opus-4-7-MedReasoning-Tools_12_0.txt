using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Solver
{
    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
    }

    public static void Main()
    {
        var ci = CultureInfo.InvariantCulture;
        using var reader = new StreamReader(Console.OpenStandardInput());
        string all = reader.ReadToEnd();
        var tokens = all.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        int V = int.Parse(tokens[idx++], ci);
        int F = int.Parse(tokens[idx++], ci);
        var verts = new Vec3[V];
        for (int i = 0; i < V; i++)
        {
            double x = double.Parse(tokens[idx++], ci);
            double y = double.Parse(tokens[idx++], ci);
            double z = double.Parse(tokens[idx++], ci);
            verts[i] = new Vec3(x, y, z);
        }
        // skip faces (count only; we use AABB-based packing)
        for (int i = 0; i < F; i++) { idx += 3; }
        double W = double.Parse(tokens[idx++], ci);
        double H = double.Parse(tokens[idx++], ci);
        double D = double.Parse(tokens[idx++], ci);

        // AABB of input polyhedron
        double minX = double.PositiveInfinity, minY = double.PositiveInfinity, minZ = double.PositiveInfinity;
        double maxX = double.NegativeInfinity, maxY = double.NegativeInfinity, maxZ = double.NegativeInfinity;
        for (int i = 0; i < V; i++)
        {
            if (verts[i].X < minX) minX = verts[i].X;
            if (verts[i].Y < minY) minY = verts[i].Y;
            if (verts[i].Z < minZ) minZ = verts[i].Z;
            if (verts[i].X > maxX) maxX = verts[i].X;
            if (verts[i].Y > maxY) maxY = verts[i].Y;
            if (verts[i].Z > maxZ) maxZ = verts[i].Z;
        }
        double bx = maxX - minX;
        double by = maxY - minY;
        double bz = maxZ - minZ;

        double eps = 1e-9;
        var sb = new StringBuilder();

        if (bx <= eps || by <= eps || bz <= eps || W <= eps || H <= eps || D <= eps)
        {
            Console.WriteLine("0");
            return;
        }

        // Strategy: try the 6 axis-aligned permutations of the AABB dimensions
        // (each corresponds to a 90-degree rotation), pick the one that yields
        // the maximum count of identity-tiling copies. This is adaptive and
        // scales to any container size in O(6) work plus output.
        // Quaternions for the 6 axis-aligned orderings of (X,Y,Z) -> mapped axes:
        // We rotate the polyhedron so its local-x maps to chosen world axis, etc.
        // Use simple discrete rotations: identity and 90-deg rotations.
        // We enumerate the 6 permutations of (bx,by,bz) and remember the rotation quaternion.

        var perms = new (double sx, double sy, double sz, double qw, double qx, double qy, double qz, int kind)[]
        {
            (bx, by, bz, 1, 0, 0, 0, 0),                       // identity
            (by, bx, bz, 0.7071067811865476, 0, 0, 0.7071067811865476, 1),  // 90 about Z: swaps x,y (with sign)
            (bz, by, bx, 0.7071067811865476, 0, 0.7071067811865476, 0, 2),  // 90 about Y: swaps x,z
            (bx, bz, by, 0.7071067811865476, 0.7071067811865476, 0, 0, 3),  // 90 about X: swaps y,z
            (by, bz, bx, 0.5, 0.5, 0.5, 0.5, 4),               // 120 rotation (cycles x->y->z->x)
            (bz, bx, by, 0.5, -0.5, -0.5, -0.5, 5),            // 240 rotation
        };

        long bestCount = -1;
        int bestKind = 0;
        double bestSX = bx, bestSY = by, bestSZ = bz;
        double bestQW = 1, bestQX = 0, bestQY = 0, bestQZ = 0;

        foreach (var p in perms)
        {
            if (p.sx > W + eps || p.sy > H + eps || p.sz > D + eps) continue;
            long nx = (long)Math.Floor((W + eps) / p.sx);
            long ny = (long)Math.Floor((H + eps) / p.sy);
            long nz = (long)Math.Floor((D + eps) / p.sz);
            if (nx <= 0 || ny <= 0 || nz <= 0) continue;
            // Cap individual factors to avoid overflow in count
            long count;
            try { count = checked(nx * ny * nz); }
            catch (OverflowException) { count = long.MaxValue; }
            if (count > bestCount)
            {
                bestCount = count;
                bestKind = p.kind;
                bestSX = p.sx; bestSY = p.sy; bestSZ = p.sz;
                bestQW = p.qw; bestQX = p.qx; bestQY = p.qy; bestQZ = p.qz;
            }
        }

        if (bestCount <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        // Cap the output to a reasonable count to fit time/memory constraints.
        long cap = 2_000_000;
        if (bestCount > cap)
        {
            // Reduce dimensions proportionally to fit cap.
            long nx = (long)Math.Floor(W / bestSX);
            long ny = (long)Math.Floor(H / bestSY);
            long nz = (long)Math.Floor(D / bestSZ);
            // shrink the dimension with the largest count first
            while ((long)nx * ny * nz > cap)
            {
                if (nx >= ny && nx >= nz && nx > 1) nx--;
                else if (ny >= nz && ny > 1) ny--;
                else if (nz > 1) nz--;
                else break;
            }
            bestCount = nx * ny * nz;
            WriteOutput(sb, ci, nx, ny, nz, minX, minY, minZ, bestSX, bestSY, bestSZ,
                bestQW, bestQX, bestQY, bestQZ, bestKind, verts);
        }
        else
        {
            long nx = (long)Math.Floor(W / bestSX);
            long ny = (long)Math.Floor(H / bestSY);
            long nz = (long)Math.Floor(D / bestSZ);
            WriteOutput(sb, ci, nx, ny, nz, minX, minY, minZ, bestSX, bestSY, bestSZ,
                bestQW, bestQX, bestQY, bestQZ, bestKind, verts);
        }

        var stdout = Console.OpenStandardOutput();
        var bytes = Encoding.ASCII.GetBytes(sb.ToString());
        stdout.Write(bytes, 0, bytes.Length);
        stdout.Flush();
    }

    static void WriteOutput(StringBuilder sb, CultureInfo ci,
        long nx, long ny, long nz,
        double minX, double minY, double minZ,
        double sx, double sy, double sz,
        double qw, double qx, double qy, double qz,
        int kind, Vec3[] verts)
    {
        // Compute the AABB of the *rotated* polyhedron so we can translate it
        // into each grid cell with corner at world (i*sx, j*sy, k*sz).
        double rMinX = double.PositiveInfinity, rMinY = double.PositiveInfinity, rMinZ = double.PositiveInfinity;
        for (int v = 0; v < verts.Length; v++)
        {
            double x = verts[v].X, y = verts[v].Y, z = verts[v].Z;
            double rx, ry, rz;
            RotateByQuat(x, y, z, qw, qx, qy, qz, out rx, out ry, out rz);
            if (rx < rMinX) rMinX = rx;
            if (ry < rMinY) rMinY = ry;
            if (rz < rMinZ) rMinZ = rz;
        }

        long total = nx * ny * nz;
        sb.Append(total).Append('\n');

        string qStr = qw.ToString("R", ci) + " " + qx.ToString("R", ci) + " " +
                      qy.ToString("R", ci) + " " + qz.ToString("R", ci);

        for (long k = 0; k < nz; k++)
        {
            double tz = -rMinZ + k * sz;
            string tzs = tz.ToString("R", ci);
            for (long j = 0; j < ny; j++)
            {
                double ty = -rMinY + j * sy;
                string tys = ty.ToString("R", ci);
                for (long i = 0; i < nx; i++)
                {
                    double tx = -rMinX + i * sx;
                    sb.Append(tx.ToString("R", ci)).Append(' ');
                    sb.Append(tys).Append(' ');
                    sb.Append(tzs).Append(' ');
                    sb.Append(qStr).Append('\n');
                }
            }
        }
    }

    static void RotateByQuat(double x, double y, double z,
        double qw, double qx, double qy, double qz,
        out double rx, out double ry, out double rz)
    {
        // p' = q * p * q^-1
        // Using standard formula
        double tx = 2.0 * (qy * z - qz * y);
        double ty = 2.0 * (qz * x - qx * z);
        double tz = 2.0 * (qx * y - qy * x);
        rx = x + qw * tx + (qy * tz - qz * ty);
        ry = y + qw * ty + (qz * tx - qx * tz);
        rz = z + qw * tz + (qx * ty - qy * tx);
    }
}
