// Minimum Edge Cut solver
// Adaptive strategy:
//   1) Find a bridge (cut edge) via iterative Tarjan -> answer = 1 edge.
//   2) Otherwise (2-edge-connected), compute global minimum edge cut:
//      - Small/medium graphs: Stoer-Wagner (parallel-edge counts as weights).
//      - Large graphs: Karger randomized contraction trials + min-degree star upper bound.
//   3) Reconstruct actual edges crossing the partition.
//
// Build: cl /std:c++20 /O2 /EHsc solver.cpp
//
#include <cstdio>
#include <cstdlib>
#include <cstdint>
#include <cstring>
#include <vector>
#include <queue>
#include <stack>
#include <algorithm>
#include <random>
#include <chrono>
#include <limits>
#include <utility>

using namespace std;

typedef int32_t i32;
typedef int64_t i64;
typedef uint64_t u64;

// ---------------- Fast IO ----------------
static char inbuf[1 << 22];
static int inpos = 0, inlen = 0;
static inline int rd_char() {
    if (inpos >= inlen) {
        inlen = (int)fread(inbuf, 1, sizeof(inbuf), stdin);
        inpos = 0;
        if (inlen <= 0) return -1;
    }
    return inbuf[inpos++];
}
static inline bool rd_i64(i64& out) {
    int c = rd_char();
    while (c != -1 && (c == ' ' || c == '\n' || c == '\r' || c == '\t')) c = rd_char();
    if (c == -1) return false;
    bool neg = false;
    if (c == '-') { neg = true; c = rd_char(); }
    i64 v = 0;
    while (c >= '0' && c <= '9') { v = v * 10 + (c - '0'); c = rd_char(); }
    out = neg ? -v : v;
    return true;
}

// ---------------- CSR Graph ----------------
struct Graph {
    i64 N = 0, M = 0;
    vector<i64> head; // size N+1
    vector<i32> nxt;  // size 2M
    vector<i64> eid;  // size 2M
    vector<pair<i32,i32>> edges;
};

static void build_csr(Graph& G, const vector<pair<i32,i32>>& edges) {
    G.M = (i64)edges.size();
    G.edges = edges;
    vector<i64> deg(G.N + 1, 0);
    for (auto& e : edges) { deg[e.first]++; deg[e.second]++; }
    G.head.assign(G.N + 1, 0);
    for (i64 i = 0; i < G.N; ++i) G.head[i+1] = G.head[i] + deg[i];
    vector<i64> cur = G.head;
    G.nxt.assign(2 * G.M, 0);
    G.eid.assign(2 * G.M, 0);
    for (i64 i = 0; i < G.M; ++i) {
        i32 u = edges[i].first, v = edges[i].second;
        G.nxt[cur[u]] = v; G.eid[cur[u]] = i; cur[u]++;
        G.nxt[cur[v]] = u; G.eid[cur[v]] = i; cur[v]++;
    }
}

// ---------------- Bridge via iterative Tarjan ----------------
static i64 find_one_bridge(const Graph& G) {
    i64 N = G.N;
    vector<i32> disc(N, -1), low(N, 0);
    vector<i64> parentEdge(N, -1);
    vector<i64> it(N, 0);
    i32 timer = 0;
    for (i64 s = 0; s < N; ++s) {
        if (disc[s] != -1) continue;
        vector<i64> st;
        st.push_back(s);
        disc[s] = low[s] = timer++;
        it[s] = G.head[s];
        parentEdge[s] = -1;
        while (!st.empty()) {
            i64 u = st.back();
            if (it[u] < G.head[u+1]) {
                i64 i = it[u]++;
                i32 v = G.nxt[i];
                i64 eIdx = G.eid[i];
                if (eIdx == parentEdge[u]) continue;
                if (disc[v] == -1) {
                    disc[v] = low[v] = timer++;
                    parentEdge[v] = eIdx;
                    it[v] = G.head[v];
                    st.push_back(v);
                } else {
                    if (disc[v] < low[u]) low[u] = disc[v];
                }
            } else {
                st.pop_back();
                if (!st.empty()) {
                    i64 p = st.back();
                    if (low[u] < low[p]) low[p] = low[u];
                    if (low[u] > disc[p]) return parentEdge[u];
                }
            }
        }
    }
    return -1;
}

// ---------------- Min-degree trivial upper bound ----------------
static i64 min_degree_cut(const Graph& G, i32& bestVertex) {
    i64 best = std::numeric_limits<i64>::max();
    bestVertex = -1;
    for (i64 u = 0; u < G.N; ++u) {
        i64 d = G.head[u+1] - G.head[u];
        if (d < best) { best = d; bestVertex = (i32)u; }
    }
    return best;
}

// ---------------- Stoer-Wagner (O(N^3)) ----------------
struct SWResult { i64 cut; vector<char> inA; };

static SWResult stoer_wagner(i64 N, vector<vector<i64>> W) {
    vector<vector<i32>> groups(N);
    for (i64 i = 0; i < N; ++i) groups[i].push_back((i32)i);
    vector<char> alive(N, 1);

    i64 bestCut = std::numeric_limits<i64>::max();
    vector<i32> bestSet;

    i64 aliveCount = N;
    while (aliveCount > 1) {
        vector<char> inA(N, 0);
        vector<i64> w(N, 0);
        i32 start = -1;
        for (i32 i = 0; i < N; ++i) if (alive[i]) { start = i; break; }
        inA[start] = 1;
        for (i32 i = 0; i < N; ++i) if (alive[i] && !inA[i]) w[i] = W[start][i];

        i32 prev = start, last = start;
        i64 cutOfPhase = 0;
        for (i64 k = 1; k < aliveCount; ++k) {
            i32 sel = -1;
            i64 bw = std::numeric_limits<i64>::min();
            for (i32 i = 0; i < N; ++i) if (alive[i] && !inA[i]) {
                if (w[i] > bw) { bw = w[i]; sel = i; }
            }
            inA[sel] = 1;
            prev = last;
            last = sel;
            cutOfPhase = bw;
            for (i32 i = 0; i < N; ++i) if (alive[i] && !inA[i]) {
                w[i] += W[sel][i];
            }
        }
        if (cutOfPhase < bestCut) {
            bestCut = cutOfPhase;
            bestSet = groups[last];
        }
        for (i32 v = 0; v < N; ++v) if (alive[v] && v != prev && v != last) {
            W[prev][v] += W[last][v];
            W[v][prev] += W[v][last];
        }
        for (i32 g : groups[last]) groups[prev].push_back(g);
        groups[last].clear();
        alive[last] = 0;
        aliveCount--;
    }

    SWResult r;
    r.cut = bestCut;
    r.inA.assign(N, 0);
    for (i32 v : bestSet) r.inA[v] = 1;
    return r;
}

// ---------------- Karger random contraction ----------------
struct DSU {
    vector<i32> p, r;
    DSU(i64 n) : p(n), r(n, 0) { for (i64 i = 0; i < n; ++i) p[i] = (i32)i; }
    i32 find(i32 x) { while (p[x] != x) { p[x] = p[p[x]]; x = p[x]; } return x; }
    bool unite(i32 a, i32 b) {
        a = find(a); b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) r[a]++;
        return true;
    }
};

static i64 karger_once(const Graph& G, mt19937_64& rng, vector<char>& outSide) {
    i64 N = G.N, M = G.M;
    vector<i32> order(M);
    for (i64 i = 0; i < M; ++i) order[i] = (i32)i;
    shuffle(order.begin(), order.end(), rng);

    DSU dsu(N);
    i64 components = N;
    i64 ei = 0;
    while (components > 2 && ei < M) {
        i32 idx = order[ei++];
        i32 u = G.edges[idx].first, v = G.edges[idx].second;
        if (dsu.unite(u, v)) components--;
    }
    if (components < 2) return std::numeric_limits<i64>::max();

    i32 repA = -1;
    for (i64 i = 0; i < N; ++i) {
        i32 r = dsu.find((i32)i);
        if (repA == -1) { repA = r; break; }
    }

    outSide.assign(N, 0);
    for (i64 i = 0; i < N; ++i) {
        if (dsu.find((i32)i) == repA) outSide[i] = 1;
    }
    i64 cnt = 0;
    for (i64 i = 0; i < M; ++i) {
        if (outSide[G.edges[i].first] != outSide[G.edges[i].second]) cnt++;
    }
    return cnt;
}

static vector<i64> cut_edge_indices(const Graph& G, const vector<char>& inA) {
    vector<i64> out;
    for (i64 i = 0; i < G.M; ++i) {
        if (inA[G.edges[i].first] != inA[G.edges[i].second]) out.push_back(i);
    }
    return out;
}

static void star_cut(const Graph& G, i32 v, vector<char>& inA) {
    inA.assign(G.N, 1);
    inA[v] = 0;
}

int main() {
    i64 N, M;
    if (!rd_i64(N) || !rd_i64(M)) { printf("0\n"); return 0; }

    vector<pair<i32,i32>> edges;
    edges.reserve((size_t)M);
    for (i64 i = 0; i < M; ++i) {
        i64 u, v;
        if (!rd_i64(u) || !rd_i64(v)) break;
        if (u == v) continue;
        edges.emplace_back((i32)u, (i32)v);
    }

    Graph G;
    G.N = N;
    build_csr(G, edges);

    if (N <= 1 || G.M == 0) { printf("0\n"); return 0; }

    // 1) Bridge check
    i64 bridgeEid = find_one_bridge(G);
    if (bridgeEid >= 0) {
        auto& e = G.edges[bridgeEid];
        printf("1\n%d %d\n", e.first, e.second);
        return 0;
    }

    // 2) Upper bound from min degree
    i32 minDegV = -1;
    i64 ub = min_degree_cut(G, minDegV);

    vector<char> bestSide;
    star_cut(G, minDegV, bestSide);
    i64 bestK = ub;

    // 3) Pick algorithm by scale
    bool useSW = (N <= 1500) && (G.M <= 2'000'000);

    if (useSW) {
        vector<vector<i64>> W(N, vector<i64>(N, 0));
        for (i64 i = 0; i < G.M; ++i) {
            i32 u = G.edges[i].first, v = G.edges[i].second;
            W[u][v]++; W[v][u]++;
        }
        SWResult sw = stoer_wagner(N, std::move(W));
        if (sw.cut < bestK) {
            bestK = sw.cut;
            bestSide = sw.inA;
        }
    } else {
        mt19937_64 rng((u64)chrono::steady_clock::now().time_since_epoch().count());
        auto t0 = chrono::steady_clock::now();
        int trialsCap;
        if (N <= 10000) trialsCap = 200;
        else if (N <= 100000) trialsCap = 40;
        else if (N <= 1000000) trialsCap = 8;
        else trialsCap = 2;

        for (int t = 0; t < trialsCap; ++t) {
            vector<char> side;
            i64 c = karger_once(G, rng, side);
            if (c < bestK) {
                bestK = c;
                bestSide = side;
            }
            auto t1 = chrono::steady_clock::now();
            double elapsed = chrono::duration<double>(t1 - t0).count();
            if (elapsed > 8.0) break;
        }
    }

    vector<i64> cutIdx = cut_edge_indices(G, bestSide);
    if (cutIdx.empty()) {
        star_cut(G, minDegV, bestSide);
        cutIdx = cut_edge_indices(G, bestSide);
        bestK = (i64)cutIdx.size();
    }

    char outbuf[64];
    int n = snprintf(outbuf, sizeof(outbuf), "%lld\n", (long long)cutIdx.size());
    fwrite(outbuf, 1, n, stdout);
    for (i64 idx : cutIdx) {
        auto& e = G.edges[idx];
        n = snprintf(outbuf, sizeof(outbuf), "%d %d\n", e.first, e.second);
        fwrite(outbuf, 1, n, stdout);
    }
    return 0;
}
