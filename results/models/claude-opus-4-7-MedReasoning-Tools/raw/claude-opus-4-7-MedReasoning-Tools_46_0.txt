#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 physics;
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) {
        return;
    }

    float dt        = physics.x;
    float softening = physics.y;
    float G         = physics.z;
    float soft2     = softening * softening;

    vec4 posMass_i = inData[i * 2u + 0u];
    vec4 velPad_i  = inData[i * 2u + 1u];

    vec3 pos_i = posMass_i.xyz;
    vec3 vel_i = velPad_i.xyz;

    vec3 acc = vec3(0.0);

    for (uint j = 0u; j < N; ++j) {
        if (j == i) {
            continue;
        }
        vec4 posMass_j = inData[j * 2u + 0u];
        vec3 r = posMass_j.xyz - pos_i;
        float mass_j = posMass_j.w;

        float distSqr = dot(r, r) + soft2;
        float invDist  = inversesqrt(distSqr);
        float invDist3 = invDist * invDist * invDist;

        acc += G * mass_j * r * invDist3;
    }

    vec3 newVel = vel_i + acc * dt;
    vec3 newPos = pos_i + newVel * dt;

    outData[i * 2u + 0u] = vec4(newPos, posMass_i.w);
    outData[i * 2u + 1u] = vec4(newVel, 0.0);
}
