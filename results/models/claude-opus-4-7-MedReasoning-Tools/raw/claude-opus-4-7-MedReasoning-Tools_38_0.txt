import sys

def run():
    text = sys.stdin.read().splitlines()
    if not text:
        print("NO SOLUTION")
        return
    header = text[0].split()
    U = int(header[0]); S = int(header[1])
    sets = []
    for i in range(1, 1 + S):
        if i < len(text):
            parts = text[i].split()
            sets.append(frozenset(int(x) for x in parts))
        else:
            sets.append(frozenset())

    universe = frozenset(range(U))

    elem_to_sets = {e: [] for e in range(U)}
    for i, s in enumerate(sets):
        for e in s:
            if 0 <= e < U:
                elem_to_sets[e].append(i)

    chosen = []
    found = []

    def solve(remaining):
        if not remaining:
            found.append(list(chosen))
            return True
        e = min(remaining, key=lambda x: len(elem_to_sets[x]))
        candidates = sorted(i for i in elem_to_sets[e] if sets[i].issubset(remaining))
        for i in candidates:
            chosen.append(i)
            if solve(remaining - sets[i]):
                return True
            chosen.pop()
        return False

    if solve(universe):
        sol = sorted(found[0])
        print("SOLUTION")
        print(" ".join(str(x) for x in sol))
    else:
        print("NO SOLUTION")

run()
