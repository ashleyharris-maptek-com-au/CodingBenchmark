const fs = require('fs');
const data = JSON.parse(fs.readFileSync(0, 'utf8'));

const W = data.width, H = data.height;
const grid = data.grid.map(r => r.split(''));
const ignitions = data.ignitions || [];
const towns = data.towns || [];
const budget = data.cutBudget | 0;
const N = W * H;
const INF = 1e9;

const idx = (x, y) => y * W + x;

const isIgn = new Uint8Array(N);
for (const [x, y] of ignitions) {
  if (x >= 0 && y >= 0 && x < W && y < H) isIgn[idx(x, y)] = 1;
}

function simulate(cutSet) {
  const burned = new Uint8Array(N);
  const queue = new Int32Array(N);
  let qh = 0, qt = 0;
  for (const [x, y] of ignitions) {
    if (x < 0 || y < 0 || x >= W || y >= H) continue;
    const i = idx(x, y);
    if (grid[y][x] === '.' && !cutSet[i] && !burned[i]) {
      burned[i] = 1;
      queue[qt++] = i;
    }
  }
  while (qh < qt) {
    const c = queue[qh++];
    const x = c % W, y = (c - x) / W;
    if (x + 1 < W) {
      const ni = c + 1;
      if (!burned[ni] && grid[y][x + 1] === '.' && !cutSet[ni]) { burned[ni] = 1; queue[qt++] = ni; }
    }
    if (x > 0) {
      const ni = c - 1;
      if (!burned[ni] && grid[y][x - 1] === '.' && !cutSet[ni]) { burned[ni] = 1; queue[qt++] = ni; }
    }
    if (y + 1 < H) {
      const ni = c + W;
      if (!burned[ni] && grid[y + 1][x] === '.' && !cutSet[ni]) { burned[ni] = 1; queue[qt++] = ni; }
    }
    if (y > 0) {
      const ni = c - W;
      if (!burned[ni] && grid[y - 1][x] === '.' && !cutSet[ni]) { burned[ni] = 1; queue[qt++] = ni; }
    }
  }
  return burned;
}

const baseBurned = simulate(new Uint8Array(N));
const burnableTowns = [];
for (const t of towns) {
  if (t.x < 0 || t.y < 0 || t.x >= W || t.y >= H) continue;
  const i = idx(t.x, t.y);
  if (baseBurned[i]) burnableTowns.push({ id: t.id, x: t.x, y: t.y, value: t.value || 0, i });
}
burnableTowns.sort((a, b) => b.value - a.value);

const forestIdx = new Int32Array(N).fill(-1);
let forestCount = 0;
for (let y = 0; y < H; y++) {
  for (let x = 0; x < W; x++) {
    if (grid[y][x] === '.') forestIdx[idx(x, y)] = forestCount++;
  }
}

function buildGraph(protectedSet) {
  const numNodes = 2 * forestCount + 2;
  const S = 2 * forestCount, T = 2 * forestCount + 1;
  const head = new Int32Array(numNodes).fill(-1);
  const eNext = [];
  const eTo = [];
  const eCap = [];

  function addEdge(u, v, cap) {
    eNext.push(head[u]); eTo.push(v); eCap.push(cap); head[u] = eTo.length - 1;
    eNext.push(head[v]); eTo.push(u); eCap.push(0); head[v] = eTo.length - 1;
  }

  for (let y = 0; y < H; y++) {
    for (let x = 0; x < W; x++) {
      if (grid[y][x] !== '.') continue;
      const cellI = idx(x, y);
      const k = forestIdx[cellI];
      const inN = 2 * k, outN = 2 * k + 1;
      let cap = 1;
      if (isIgn[cellI] || protectedSet[cellI]) cap = INF;
      addEdge(inN, outN, cap);
      if (x + 1 < W && grid[y][x + 1] === '.') addEdge(outN, 2 * forestIdx[cellI + 1], INF);
      if (x > 0 && grid[y][x - 1] === '.') addEdge(outN, 2 * forestIdx[cellI - 1], INF);
      if (y + 1 < H && grid[y + 1][x] === '.') addEdge(outN, 2 * forestIdx[cellI + W], INF);
      if (y > 0 && grid[y - 1][x] === '.') addEdge(outN, 2 * forestIdx[cellI - W], INF);
    }
  }
  for (const [x, y] of ignitions) {
    if (x < 0 || y < 0 || x >= W || y >= H) continue;
    if (grid[y][x] !== '.') continue;
    addEdge(S, 2 * forestIdx[idx(x, y)], INF);
  }
  for (let i = 0; i < N; i++) {
    if (protectedSet[i]) {
      const yy = (i / W) | 0;
      const xx = i % W;
      if (grid[yy][xx] === '.') addEdge(2 * forestIdx[i] + 1, T, INF);
    }
  }
  return { numNodes, S, T, head, eNext, eTo, eCap };
}

function maxFlow(g, cap) {
  const { numNodes, S, T, head, eNext, eTo, eCap } = g;
  let flow = 0;
  const parent = new Int32Array(numNodes);
  const parentEdge = new Int32Array(numNodes);
  const visited = new Uint8Array(numNodes);
  const queue = new Int32Array(numNodes);
  while (true) {
    visited.fill(0);
    visited[S] = 1;
    queue[0] = S;
    let qh = 0, qt = 1;
    while (qh < qt && !visited[T]) {
      const v = queue[qh++];
      for (let e = head[v]; e !== -1; e = eNext[e]) {
        const to = eTo[e];
        if (eCap[e] > 0 && !visited[to]) {
          visited[to] = 1;
          parent[to] = v;
          parentEdge[to] = e;
          if (to === T) { qh = qt; break; }
          queue[qt++] = to;
        }
      }
    }
    if (!visited[T]) break;
    let pathFlow = INF;
    let cur = T;
    while (cur !== S) {
      const e = parentEdge[cur];
      if (eCap[e] < pathFlow) pathFlow = eCap[e];
      cur = parent[cur];
    }
    cur = T;
    while (cur !== S) {
      const e = parentEdge[cur];
      eCap[e] -= pathFlow;
      eCap[e ^ 1] += pathFlow;
      cur = parent[cur];
    }
    flow += pathFlow;
    if (flow > cap) return flow;
  }
  return flow;
}

function extractCut(g) {
  const { numNodes, S, head, eNext, eTo, eCap } = g;
  const reach = new Uint8Array(numNodes);
  reach[S] = 1;
  const queue = new Int32Array(numNodes);
  queue[0] = S;
  let qh = 0, qt = 1;
  while (qh < qt) {
    const v = queue[qh++];
    for (let e = head[v]; e !== -1; e = eNext[e]) {
      const to = eTo[e];
      if (eCap[e] > 0 && !reach[to]) {
        reach[to] = 1;
        queue[qt++] = to;
      }
    }
  }
  const cuts = [];
  for (let y = 0; y < H; y++) {
    for (let x = 0; x < W; x++) {
      if (grid[y][x] !== '.') continue;
      const cellI = idx(x, y);
      const k = forestIdx[cellI];
      if (reach[2 * k] && !reach[2 * k + 1]) cuts.push([x, y]);
    }
  }
  return cuts;
}

const protectedSet = new Uint8Array(N);
let bestCuts = [];
let bestProtected = [];

for (const t of burnableTowns) {
  protectedSet[t.i] = 1;
  const g = buildGraph(protectedSet);
  const flow = maxFlow(g, budget);
  if (flow <= budget) {
    bestCuts = extractCut(g);
    bestProtected.push(t);
  } else {
    protectedSet[t.i] = 0;
  }
}

if (bestCuts.length > budget) {
  const cutMask = new Uint8Array(N);
  for (const [x, y] of bestCuts) cutMask[idx(x, y)] = 1;
  function townsSaved(mask) {
    const b = simulate(mask);
    let saved = 0;
    for (const t of bestProtected) if (!b[t.i]) saved++;
    return saved;
  }
  const baseline = townsSaved(cutMask);
  const newCuts = bestCuts.slice();
  for (let k = newCuts.length - 1; k >= 0 && newCuts.length > budget; k--) {
    const [x, y] = newCuts[k];
    const i = idx(x, y);
    cutMask[i] = 0;
    if (townsSaved(cutMask) === baseline) {
      newCuts.splice(k, 1);
    } else {
      cutMask[i] = 1;
    }
  }
  bestCuts = newCuts.slice(0, budget);
}

process.stdout.write(JSON.stringify({ firebreaks: bestCuts }));
