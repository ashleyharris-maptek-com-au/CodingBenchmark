const fs = require('fs');

function main() {
  const data = JSON.parse(fs.readFileSync(0, 'utf8'));
  const prob = data.probability;
  const drones = data.drones;
  const maxSteps = data.maxStepsPerDrone;
  const H = prob.length;
  const W = H > 0 ? prob[0].length : 0;
  const D = drones.length;

  const isObs = (x, y) => prob[y][x] < 0;

  // claimed[y*W+x] = 1 if collected by some drone's path already
  const claimed = new Uint8Array(W * H);

  const paths = [];
  const moves = [[0,0],[1,0],[-1,0],[0,1],[0,-1]];

  for (let di = 0; di < D; di++) {
    const drone = drones[di];
    let cx = drone.x, cy = drone.y;
    const path = [[cx, cy]];

    if (!isObs(cx, cy) && prob[cy][cx] > 0) {
      claimed[cx + cy * W] = 1;
    }

    // Choose horizon adaptively based on grid/steps
    const cells = W * H;
    let horizon;
    if (cells <= 400) horizon = 8;
    else if (cells <= 2500) horizon = 6;
    else if (cells <= 10000) horizon = 5;
    else horizon = 4;
    horizon = Math.min(horizon, maxSteps);

    const beamWidth = cells <= 2500 ? 16 : 10;

    for (let step = 0; step < maxSteps; step++) {
      // Beam search
      let beam = [];
      // Initial expansion from current
      for (let m = 0; m < 5; m++) {
        const nx = cx + moves[m][0], ny = cy + moves[m][1];
        if (nx < 0 || ny < 0 || nx >= W || ny >= H) continue;
        if (isObs(nx, ny)) continue;
        const idx = nx + ny * W;
        let s = 0;
        const localSet = new Set();
        if (!claimed[idx] && prob[ny][nx] > 0) {
          s += prob[ny][nx];
          localSet.add(idx);
        }
        beam.push({ x: nx, y: ny, score: s, claimed: localSet, firstMove: m });
      }

      for (let h = 1; h < horizon; h++) {
        const next = [];
        for (const state of beam) {
          for (let m = 0; m < 5; m++) {
            const nx = state.x + moves[m][0], ny = state.y + moves[m][1];
            if (nx < 0 || ny < 0 || nx >= W || ny >= H) continue;
            if (isObs(nx, ny)) continue;
            const idx = nx + ny * W;
            let s = state.score;
            let newSet = state.claimed;
            if (!claimed[idx] && !state.claimed.has(idx) && prob[ny][nx] > 0) {
              s += prob[ny][nx];
              newSet = new Set(state.claimed);
              newSet.add(idx);
            }
            next.push({ x: nx, y: ny, score: s, claimed: newSet, firstMove: state.firstMove });
          }
        }
        if (next.length === 0) break;
        // Dedupe by (x,y,firstMove) keeping best score
        next.sort((a, b) => b.score - a.score);
        const seen = new Set();
        const pruned = [];
        for (const st of next) {
          const key = st.x * 73856093 ^ st.y * 19349663 ^ st.firstMove * 83492791;
          if (seen.has(key)) continue;
          seen.add(key);
          pruned.push(st);
          if (pruned.length >= beamWidth) break;
        }
        beam = pruned;
      }

      if (beam.length === 0) {
        path.push([cx, cy]);
        continue;
      }

      let bestState = beam[0];
      for (const st of beam) {
        if (st.score > bestState.score) bestState = st;
      }

      const m = bestState.firstMove;
      cx = cx + moves[m][0];
      cy = cy + moves[m][1];
      const idx = cx + cy * W;
      if (!claimed[idx] && prob[cy][cx] > 0) claimed[idx] = 1;
      path.push([cx, cy]);
    }

    paths.push(path);
  }

  process.stdout.write(JSON.stringify({ paths }));
}

main();
