using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

class TetrisPlayer
{
    static int W, H;
    static int[] board;

    static Dictionary<char, int[][][]> pieces = new Dictionary<char, int[][][]>();
    static Dictionary<char, int[]> pieceWidths = new Dictionary<char, int[]>();
    static Dictionary<char, int[]> pieceHeights = new Dictionary<char, int[]>();

    static void InitPieces()
    {
        AddPiece('I', new int[][][] {
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{0,2}, new[]{0,3} },
            new int[][] { new[]{0,0}, new[]{1,0}, new[]{2,0}, new[]{3,0} },
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{0,2}, new[]{0,3} },
            new int[][] { new[]{0,0}, new[]{1,0}, new[]{2,0}, new[]{3,0} },
        });
        AddPiece('O', new int[][][] {
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{1,0}, new[]{1,1} },
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{1,0}, new[]{1,1} },
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{1,0}, new[]{1,1} },
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{1,0}, new[]{1,1} },
        });
        AddPiece('T', new int[][][] {
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{0,2}, new[]{1,1} },
            new int[][] { new[]{0,0}, new[]{1,0}, new[]{2,0}, new[]{1,1} },
            new int[][] { new[]{1,0}, new[]{1,1}, new[]{1,2}, new[]{0,1} },
            new int[][] { new[]{0,1}, new[]{1,1}, new[]{2,1}, new[]{1,0} },
        });
        AddPiece('S', new int[][][] {
            new int[][] { new[]{0,1}, new[]{0,2}, new[]{1,0}, new[]{1,1} },
            new int[][] { new[]{0,0}, new[]{1,0}, new[]{1,1}, new[]{2,1} },
            new int[][] { new[]{0,1}, new[]{0,2}, new[]{1,0}, new[]{1,1} },
            new int[][] { new[]{0,0}, new[]{1,0}, new[]{1,1}, new[]{2,1} },
        });
        AddPiece('Z', new int[][][] {
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{1,1}, new[]{1,2} },
            new int[][] { new[]{0,1}, new[]{1,0}, new[]{1,1}, new[]{2,0} },
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{1,1}, new[]{1,2} },
            new int[][] { new[]{0,1}, new[]{1,0}, new[]{1,1}, new[]{2,0} },
        });
        AddPiece('J', new int[][][] {
            new int[][] { new[]{0,0}, new[]{1,0}, new[]{1,1}, new[]{1,2} },
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{1,0}, new[]{2,0} },
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{0,2}, new[]{1,2} },
            new int[][] { new[]{0,1}, new[]{1,1}, new[]{2,0}, new[]{2,1} },
        });
        AddPiece('L', new int[][][] {
            new int[][] { new[]{0,2}, new[]{1,0}, new[]{1,1}, new[]{1,2} },
            new int[][] { new[]{0,0}, new[]{1,0}, new[]{2,0}, new[]{2,1} },
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{0,2}, new[]{1,0} },
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{1,1}, new[]{2,1} },
        });
    }

    static void AddPiece(char c, int[][][] rots)
    {
        pieces[c] = rots;
        int[] widths = new int[4];
        int[] heights = new int[4];
        for (int r = 0; r < 4; r++)
        {
            int maxC = 0, maxR = 0;
            foreach (var cell in rots[r])
            {
                if (cell[0] > maxR) maxR = cell[0];
                if (cell[1] > maxC) maxC = cell[1];
            }
            widths[r] = maxC + 1;
            heights[r] = maxR + 1;
        }
        pieceWidths[c] = widths;
        pieceHeights[c] = heights;
    }

    static int DropRow(int[] bd, char piece, int rot, int col)
    {
        var cells = pieces[piece][rot];
        int pw = pieceWidths[piece][rot];
        int ph = pieceHeights[piece][rot];
        if (col < 0 || col + pw > W) return -1;

        int[] bottomOffset = new int[pw];
        for (int i = 0; i < pw; i++) bottomOffset[i] = -1;
        foreach (var cell in cells)
        {
            int dc = cell[1];
            int dr = cell[0];
            if (dr > bottomOffset[dc]) bottomOffset[dc] = dr;
        }

        int r0 = -ph;
        for (int dc = 0; dc < pw; dc++)
        {
            if (bottomOffset[dc] < 0) continue;
            int bcol = col + dc;
            int gr = H;
            for (int r = 0; r < H; r++)
            {
                if ((bd[r] & (1 << bcol)) != 0) { gr = r; break; }
            }
            int candidate = gr - 1 - bottomOffset[dc];
            if (candidate > r0) r0 = candidate;
        }

        foreach (var cell in cells)
        {
            int rr = r0 + cell[0];
            int cc = col + cell[1];
            if (rr < 0) return -2;
            if (rr >= H) return -1;
            if ((bd[rr] & (1 << cc)) != 0) return -1;
        }
        return r0;
    }

    static int ApplyPiece(int[] bd, char piece, int rot, int col, int r0)
    {
        var cells = pieces[piece][rot];
        foreach (var cell in cells)
        {
            int rr = r0 + cell[0];
            int cc = col + cell[1];
            if (rr >= 0 && rr < H) bd[rr] |= (1 << cc);
        }
        int full = (1 << W) - 1;
        int cleared = 0;
        int writeR = H - 1;
        int[] newBoard = new int[H];
        for (int r = H - 1; r >= 0; r--)
        {
            if (bd[r] == full) { cleared++; continue; }
            newBoard[writeR--] = bd[r];
        }
        while (writeR >= 0) newBoard[writeR--] = 0;
        Array.Copy(newBoard, bd, H);
        return cleared;
    }

    static int[] CloneBoard(int[] bd)
    {
        int[] nb = new int[H];
        Array.Copy(bd, nb, H);
        return nb;
    }

    static double Evaluate(int[] bd, int linesCleared)
    {
        int[] heights = new int[W];
        int aggregateHeight = 0;
        int holes = 0;
        int bumpiness = 0;
        int wells = 0;
        int rowTransitions = 0;
        int colTransitions = 0;
        for (int c = 0; c < W; c++)
        {
            int h = 0;
            for (int r = 0; r < H; r++)
            {
                if ((bd[r] & (1 << c)) != 0) { h = H - r; break; }
            }
            heights[c] = h;
            aggregateHeight += h;
            bool seenBlock = false;
            for (int r = 0; r < H; r++)
            {
                bool filled = (bd[r] & (1 << c)) != 0;
                if (filled) seenBlock = true;
                else if (seenBlock) holes++;
            }
        }
        for (int c = 0; c < W - 1; c++)
            bumpiness += Math.Abs(heights[c] - heights[c + 1]);
        for (int c = 0; c < W; c++)
        {
            int left = (c == 0) ? H : heights[c - 1];
            int right = (c == W - 1) ? H : heights[c + 1];
            int wellDepth = Math.Min(left, right) - heights[c];
            if (wellDepth > 0) wells += wellDepth * (wellDepth + 1) / 2;
        }
        for (int r = 0; r < H; r++)
        {
            int row = bd[r];
            int prev = 1;
            for (int c = 0; c < W; c++)
            {
                int cur = (row >> c) & 1;
                if (cur != prev) rowTransitions++;
                prev = cur;
            }
            if (prev != 1) rowTransitions++;
        }
        for (int c = 0; c < W; c++)
        {
            int prev = 1;
            for (int r = 0; r < H; r++)
            {
                int cur = ((bd[r] >> c) & 1);
                if (cur != prev) colTransitions++;
                prev = cur;
            }
            if (prev != 1) colTransitions++;
        }
        int maxHeight = 0;
        for (int c = 0; c < W; c++) if (heights[c] > maxHeight) maxHeight = heights[c];

        double score = 0;
        score += -0.510066 * aggregateHeight;
        score +=  0.760666 * linesCleared;
        score += -0.35663  * holes;
        score += -0.184483 * bumpiness;
        score += -0.3      * wells;
        score += -0.15     * rowTransitions;
        score += -0.15     * colTransitions;
        score += -0.05     * maxHeight;
        return score;
    }

    static int RotMax(char piece)
    {
        if (piece == 'O') return 1;
        if (piece == 'I' || piece == 'S' || piece == 'Z') return 2;
        return 4;
    }

    struct Move { public int Rot; public int Col; public double Score; public bool Valid; }

    static Move BestMove(int[] bd, char piece)
    {
        Move best = new Move { Valid = false, Score = double.NegativeInfinity };
        int rotMax = RotMax(piece);
        for (int rot = 0; rot < rotMax; rot++)
        {
            int pw = pieceWidths[piece][rot];
            for (int col = 0; col + pw <= W; col++)
            {
                int r0 = DropRow(bd, piece, rot, col);
                if (r0 < 0) continue;
                int[] nb = CloneBoard(bd);
                int cleared = ApplyPiece(nb, piece, rot, col, r0);
                double s = Evaluate(nb, cleared);
                if (!best.Valid || s > best.Score)
                {
                    best = new Move { Valid = true, Rot = rot, Col = col, Score = s };
                }
            }
        }
        return best;
    }

    static void Main()
    {
        InitPieces();
        var input = new StreamReader(Console.OpenStandardInput(), Encoding.ASCII);
        var output = new StreamWriter(Console.OpenStandardOutput(), Encoding.ASCII);
        output.AutoFlush = false;

        var tokenQueue = new Queue<string>();
        Func<string> nextToken = () =>
        {
            while (tokenQueue.Count == 0)
            {
                string line = input.ReadLine();
                if (line == null) return null;
                foreach (var t in line.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries))
                    tokenQueue.Enqueue(t);
            }
            return tokenQueue.Dequeue();
        };

        W = int.Parse(nextToken());
        H = int.Parse(nextToken());
        int N = int.Parse(nextToken());
        board = new int[H];

        for (int i = 0; i < N; i++)
        {
            string ptok = nextToken();
            if (ptok == null) break;
            char piece = ptok[0];
            if (!pieces.ContainsKey(piece)) continue;

            Move mv = BestMove(board, piece);
            int rot = mv.Valid ? mv.Rot : 0;
            int col = mv.Valid ? mv.Col : 0;

            output.Write(rot);
            output.Write(' ');
            output.Write(col);
            output.Write('\n');
            output.Flush();

            string result = nextToken();
            if (result == null) break;

            int r0 = DropRow(board, piece, rot, col);
            if (r0 >= 0) ApplyPiece(board, piece, rot, col, r0);

            if (result.StartsWith("gameover")) break;
        }
        output.Flush();
    }
}
