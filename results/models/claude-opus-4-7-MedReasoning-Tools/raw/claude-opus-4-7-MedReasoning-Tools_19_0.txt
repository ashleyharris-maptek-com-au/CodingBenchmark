using System;
using System.Collections.Generic;
using System.IO;
using System.Text;
using System.Globalization;

class MiningSolver
{
    static int XC, YC, ZC;
    static double XB, YB, ZB;
    static byte[,,] grid;
    static bool useFull;
    static int[,] topZ;
    static byte[,] topMass;
    static byte[,] topVal;

    static void Main()
    {
        var sin = Console.OpenStandardInput();
        var sout = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 16);

        var headerSb = new StringBuilder();
        int b;
        while ((b = sin.ReadByte()) != -1 && b != '\n')
        {
            if (b != '\r') headerSb.Append((char)b);
        }
        var parts = headerSb.ToString().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        XC = int.Parse(parts[0]);
        YC = int.Parse(parts[1]);
        ZC = int.Parse(parts[2]);
        XB = double.Parse(parts[3], CultureInfo.InvariantCulture);
        YB = double.Parse(parts[4], CultureInfo.InvariantCulture);
        ZB = double.Parse(parts[5], CultureInfo.InvariantCulture);

        long total = (long)XC * YC * ZC;
        useFull = total <= 50_000_000L;

        topZ = new int[XC, YC];
        topMass = new byte[XC, YC];
        topVal = new byte[XC, YC];
        if (useFull) grid = new byte[XC, YC, ZC];

        var br = new BufferedStream(sin, 1 << 16);
        for (int x = 0; x < XC; x++)
        {
            for (int y = 0; y < YC; y++)
            {
                int highest = -1;
                byte topD = 0;
                for (int z = 0; z < ZC; z++)
                {
                    int v = br.ReadByte();
                    if (v < 0) v = 0;
                    byte d = (byte)v;
                    if (useFull) grid[x, y, z] = d;
                    if (d != 0) { highest = z; topD = d; }
                }
                topZ[x, y] = highest + 1;
                if (highest >= 0)
                {
                    topMass[x, y] = (byte)(topD & 0x0F);
                    topVal[x, y] = (byte)(topD >> 4);
                }
            }
        }

        var ops = new List<string>();
        var candidates = new List<(int x, int y, int profit)>();
        for (int x = 0; x < XC; x++)
            for (int y = 0; y < YC; y++)
            {
                int val = topVal[x, y];
                int mass = topMass[x, y];
                if (val == 0 || mass == 0) continue;
                if (x == 0 && y == 0) continue;
                int dist = x + y;
                int revenue = val * 1000 * mass;
                int costEst = dist * mass;
                int profit = revenue - costEst;
                if (profit > 0) candidates.Add((x, y, profit));
            }
        candidates.Sort((a, b) => b.profit.CompareTo(a.profit));

        var sb = new StringBuilder();
        foreach (var c in candidates)
        {
            int sx = c.x, sy = c.y;
            if (topZ[sx, sy] == 0) continue;
            int digZ = topZ[sx, sy] - 1;

            var path = new List<(int x, int y)>();
            bool ok = TryPath(sx, sy, true, path);
            if (!ok) { path.Clear(); ok = TryPath(sx, sy, false, path); }
            if (!ok) continue;

            sb.Clear();
            sb.Append(path.Count);
            sb.Append(' '); sb.Append(sx);
            sb.Append(' '); sb.Append(sy);
            sb.Append(' '); sb.Append(digZ);
            foreach (var step in path)
            {
                sb.Append(' '); sb.Append(step.x);
                sb.Append(' '); sb.Append(step.y);
            }
            sb.Append(" 0 0");
            ops.Add(sb.ToString());

            topZ[sx, sy]--;
            if (useFull && topZ[sx, sy] > 0)
            {
                byte d = grid[sx, sy, topZ[sx, sy] - 1];
                topMass[sx, sy] = (byte)(d & 0x0F);
                topVal[sx, sy] = (byte)(d >> 4);
            }
            else
            {
                topMass[sx, sy] = 0;
                topVal[sx, sy] = 0;
            }
        }

        sout.WriteLine(ops.Count);
        foreach (var op in ops) sout.WriteLine(op);
        sout.Flush();
    }

    static bool TryPath(int sx, int sy, bool xFirst, List<(int, int)> path)
    {
        int cx = sx, cy = sy;
        int curZ = topZ[cx, cy] - 1;
        int guard = XC + YC + 5;
        while (cx != 0 || cy != 0)
        {
            int nx = cx, ny = cy;
            if (xFirst)
            {
                if (cx != 0) nx = cx - Math.Sign(cx);
                else ny = cy - Math.Sign(cy);
            }
            else
            {
                if (cy != 0) ny = cy - Math.Sign(cy);
                else nx = cx - Math.Sign(cx);
            }
            int nextZ = topZ[nx, ny];
            if (Math.Abs(nextZ - curZ) > 1) return false;
            path.Add((nx, ny));
            cx = nx; cy = ny; curZ = nextZ;
            if (--guard < 0) return false;
        }
        return true;
    }
}
