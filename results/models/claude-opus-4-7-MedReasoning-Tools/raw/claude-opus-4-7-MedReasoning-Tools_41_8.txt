; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
               OpExecutionMode %main OriginUpperLeft

               OpName %main "main"
               OpName %UBO "UBO"
               OpMemberName %UBO 0 "model"
               OpMemberName %UBO 1 "view"
               OpMemberName %UBO 2 "projection"
               OpMemberName %UBO 3 "lightPos"
               OpMemberName %UBO 4 "cameraPos"
               OpMemberName %UBO 5 "params"
               OpName %ubo "ubo"
               OpName %worldPos "worldPos"
               OpName %normal "normal"
               OpName %uv "uv"
               OpName %tangent "tangent"
               OpName %color "color"
               OpName %outColor "outColor"

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %color Location 4
               OpDecorate %outColor Location 0

               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
        %v2f = OpTypeVector %float 2
        %v3f = OpTypeVector %float 3
        %v4f = OpTypeVector %float 4
        %m4f = OpTypeMatrix %v4f 4

        %UBO = OpTypeStruct %m4f %m4f %m4f %v4f %v4f %v4f
%ptr_UBO_Uniform = OpTypePointer Uniform %UBO
        %ubo = OpVariable %ptr_UBO_Uniform Uniform

  %ptr_in_v3 = OpTypePointer Input %v3f
  %ptr_in_v2 = OpTypePointer Input %v2f
 %ptr_out_v4 = OpTypePointer Output %v4f
 %ptr_uni_v4 = OpTypePointer Uniform %v4f

   %worldPos = OpVariable %ptr_in_v3 Input
     %normal = OpVariable %ptr_in_v3 Input
         %uv = OpVariable %ptr_in_v2 Input
    %tangent = OpVariable %ptr_in_v3 Input
      %color = OpVariable %ptr_in_v3 Input
   %outColor = OpVariable %ptr_out_v4 Output

        %int = OpTypeInt 32 1
         %i3 = OpConstant %int 3
         %f0 = OpConstant %float 0.0
         %f1 = OpConstant %float 1.0
        %f02 = OpConstant %float 0.2
        %f04 = OpConstant %float 0.4
        %f07 = OpConstant %float 0.7
       %f025 = OpConstant %float 0.25
        %f05 = OpConstant %float 0.5
       %f075 = OpConstant %float 0.75
       %bool = OpTypeBool

       %main = OpFunction %void None %fn_void
        %lbl = OpLabel

       %n_in = OpLoad %v3f %normal
          %N = OpExtInst %v3f %1 Normalize %n_in
         %wp = OpLoad %v3f %worldPos
     %lp_ptr = OpAccessChain %ptr_uni_v4 %ubo %i3
        %lp4 = OpLoad %v4f %lp_ptr
        %lp3 = OpVectorShuffle %v3f %lp4 %lp4 0 1 2
       %ldir = OpFSub %v3f %lp3 %wp
          %L = OpExtInst %v3f %1 Normalize %ldir
      %d_raw = OpDot %float %N %L
          %d = OpExtInst %float %1 FMax %d_raw %f0

       %c075 = OpFOrdGreaterThanEqual %bool %d %f075
        %c05 = OpFOrdGreaterThanEqual %bool %d %f05
       %c025 = OpFOrdGreaterThanEqual %bool %d %f025

         %b1 = OpSelect %float %c025 %f04 %f02
         %b2 = OpSelect %float %c05  %f07 %b1
       %band = OpSelect %float %c075 %f1  %b2

         %vc = OpLoad %v3f %color
        %rgb = OpVectorTimesScalar %v3f %vc %band
       %rgbx = OpCompositeExtract %float %rgb 0
       %rgby = OpCompositeExtract %float %rgb 1
       %rgbz = OpCompositeExtract %float %rgb 2
       %rgba = OpCompositeConstruct %v4f %rgbx %rgby %rgbz %f1
               OpStore %outColor %rgba
               OpReturn
               OpFunctionEnd
