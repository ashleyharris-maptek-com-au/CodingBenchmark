OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_out %gl_in %outColor %inColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 7

               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %outColor Location 0
               OpDecorate %inColor Location 2

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4

      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
     %uint_2 = OpConstant %uint 2

   %float_0  = OpConstant %float 0.0
   %float_1  = OpConstant %float 1.0
 %float_075  = OpConstant %float 0.75
 %float_015  = OpConstant %float 0.015
 %float_005  = OpConstant %float 0.05

  %gl_PerVertex_out = OpTypeStruct %v4float
  %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
  %ptr_out_v4 = OpTypePointer Output %v4float

  %gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
  %ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
  %ptr_in_v4 = OpTypePointer Input %v4float

  %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

  %arr_v3_in = OpTypeArray %v3float %uint_2
  %ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
    %inColor = OpVariable %ptr_in_arr_v3 Input
  %ptr_in_v3 = OpTypePointer Input %v3float

       %main = OpFunction %void None %fn_void
      %entry = OpLabel

      %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
      %p0     = OpLoad %v4float %p0_ptr
      %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
      %p1     = OpLoad %v4float %p1_ptr

      %c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
      %c0     = OpLoad %v3float %c0_ptr
      %c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
      %c1     = OpLoad %v3float %c1_ptr

      %p0x = OpCompositeExtract %float %p0 0
      %p0y = OpCompositeExtract %float %p0 1
      %p0z = OpCompositeExtract %float %p0 2
      %p0w = OpCompositeExtract %float %p0 3
      %p1x = OpCompositeExtract %float %p1 0
      %p1y = OpCompositeExtract %float %p1 1
      %p1z = OpCompositeExtract %float %p1 2
      %p1w = OpCompositeExtract %float %p1 3

      %dx = OpFSub %float %p1x %p0x
      %dy = OpFSub %float %p1y %p0y

      %ndy = OpFNegate %float %dy
      %perp_raw = OpCompositeConstruct %v2float %ndy %dx
      %perp = OpExtInst %v2float %1 Normalize %perp_raw
      %px = OpCompositeExtract %float %perp 0
      %py = OpCompositeExtract %float %perp 1

      %tx = OpFMul %float %px %float_015
      %ty = OpFMul %float %py %float_015

      %dz = OpFSub %float %p1z %p0z
      %dw = OpFSub %float %p1w %p0w
      %mdx = OpFMul %float %dx %float_075
      %mdy = OpFMul %float %dy %float_075
      %mdz = OpFMul %float %dz %float_075
      %mdw = OpFMul %float %dw %float_075
      %midx = OpFAdd %float %p0x %mdx
      %midy = OpFAdd %float %p0y %mdy
      %midz = OpFAdd %float %p0z %mdz
      %midw = OpFAdd %float %p0w %mdw

      ; Shaft vertex 1: p0 + perp*0.015
      %v1x = OpFAdd %float %p0x %tx
      %v1y = OpFAdd %float %p0y %ty
      %v1 = OpCompositeConstruct %v4float %v1x %v1y %p0z %p0w
      %s1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
            OpStore %s1 %v1
            OpStore %outColor %c0
            OpEmitVertex

      ; Shaft vertex 2: p0 - perp*0.015
      %v2x = OpFSub %float %p0x %tx
      %v2y = OpFSub %float %p0y %ty
      %v2 = OpCompositeConstruct %v4float %v2x %v2y %p0z %p0w
      %s2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
            OpStore %s2 %v2
            OpStore %outColor %c0
            OpEmitVertex

      ; Shaft vertex 3: mid + perp*0.015
      %v3x = OpFAdd %float %midx %tx
      %v3y = OpFAdd %float %midy %ty
      %v3v = OpCompositeConstruct %v4float %v3x %v3y %midz %midw
      %s3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
            OpStore %s3 %v3v
            OpStore %outColor %c0
            OpEmitVertex

      ; Shaft vertex 4: mid - perp*0.015
      %v4x = OpFSub %float %midx %tx
      %v4y = OpFSub %float %midy %ty
      %v4v = OpCompositeConstruct %v4float %v4x %v4y %midz %midw
      %s4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
            OpStore %s4 %v4v
            OpStore %outColor %c0
            OpEmitVertex

            OpEndPrimitive

      ; Arrowhead offsets
      %ax = OpFMul %float %px %float_005
      %ay = OpFMul %float %py %float_005

      ; a1 = mid + perp*0.05
      %a1x = OpFAdd %float %midx %ax
      %a1y = OpFAdd %float %midy %ay
      %a1 = OpCompositeConstruct %v4float %a1x %a1y %midz %midw
      %as1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
             OpStore %as1 %a1
             OpStore %outColor %c1
             OpEmitVertex

      ; a2 = mid - perp*0.05
      %a2x = OpFSub %float %midx %ax
      %a2y = OpFSub %float %midy %ay
      %a2 = OpCompositeConstruct %v4float %a2x %a2y %midz %midw
      %as2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
             OpStore %as2 %a2
             OpStore %outColor %c1
             OpEmitVertex

      ; a3 = p1 (tip)
      %as3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
             OpStore %as3 %p1
             OpStore %outColor %c1
             OpEmitVertex

             OpEndPrimitive

             OpReturn
             OpFunctionEnd
