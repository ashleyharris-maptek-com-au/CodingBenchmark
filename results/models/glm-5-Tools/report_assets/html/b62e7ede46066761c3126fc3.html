using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

public class MaximumClique
{
    private static int n, m;
    private static bool[][] adj;
    private static int[] bestClique;
    private static int bestSize;
    private static readonly object lockObj = new object();
    private static Stopwatch stopwatch;
    private static int timeLimitMs = 280000; // 4.67 minutes
    private static bool timeoutReached = false;
    private static bool isDense;

    public static void Main()
    {
        stopwatch = Stopwatch.StartNew();
        ReadGraph();
        
        // Determine if graph is dense
        isDense = (double)m / (n * (n - 1) / 2.0) > 0.3;
        
        // Choose strategy based on graph size and density
        if (n <= 100)
        {
            ExactSmall();
        }
        else if (n <= 2000)
        {
            ParallelBranchAndBound();
        }
        else if (n <= 50000)
        {
            HybridStrategy();
        }
        else
        {
            LargeGraphStrategy();
        }
        
        OutputResult();
    }

    static void ReadGraph()
    {
        string[] header = Console.ReadLine().Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        n = int.Parse(header[0]);
        m = int.Parse(header[1]);

        // Initialize adjacency matrix as array of bool arrays
        adj = new bool[n][];
        for (int i = 0; i < n; i++)
        {
            adj[i] = new bool[n];
            adj[i][i] = true; // Self-loop for convenience
        }

        for (int i = 0; i < m; i++)
        {
            string[] edge = Console.ReadLine().Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            if (u >= 0 && u < n && v >= 0 && v < n && u != v)
            {
                adj[u][v] = true;
                adj[v][u] = true;
            }
        }
    }

    static void ExactSmall()
    {
        // Use branch and bound with coloring for exact solution
        var ordering = GetOrdering();
        BranchAndBoundExact(ordering);
    }

    static void ParallelBranchAndBound()
    {
        // Get ordering and partition work for parallel processing
        var ordering = GetOrdering();
        int numThreads = Math.Min(Environment.ProcessorCount, 16);
        
        // Process root branches in parallel
        Parallel.For(0, n, new ParallelOptions { MaxDegreeOfParallelism = numThreads },
            (i, state) =>
            {
                if (timeoutReached)
                {
                    state.Stop();
                    return;
                }

                int v = ordering[n - 1 - i];
                var clique = new List<int> { v };
                var candidates = new List<int>();
                
                for (int j = 0; j < n; j++)
                {
                    if (j != v && adj[v][j])
                        candidates.Add(j);
                }

                SearchParallel(clique, candidates, ordering);
            });
    }

    static void SearchParallel(List<int> clique, List<int> candidates, int[] ordering)
    {
        if (candidates.Count == 0)
        {
            lock (lockObj)
            {
                if (clique.Count > bestSize)
                {
                    bestSize = clique.Count;
                    bestClique = clique.ToArray();
                }
            }
            return;
        }

        if (stopwatch.ElapsedMilliseconds > timeLimitMs)
        {
            timeoutReached = true;
            return;
        }

        // Upper bound check
        int upperBound = clique.Count + candidates.Count;
        if (upperBound <= bestSize) return;

        // Use coloring for better pruning
        var colors = GreedyColoring(candidates);
        upperBound = clique.Count + colors.Values.Max();
        if (upperBound <= bestSize) return;

        // Try each candidate in order
        var sortedCandidates = candidates.OrderByDescending(c => colors[c]).ToList();
        
        foreach (var v in sortedCandidates)
        {
            if (timeoutReached) return;
            
            // Check if we can still improve
            if (clique.Count + colors[v] <= bestSize) break;

            var newClique = new List<int>(clique) { v };
            var newCandidates = candidates.Where(c => c != v && adj[v][c]).ToList();

            if (newClique.Count + newCandidates.Count > bestSize)
            {
                SearchParallel(newClique, newCandidates, ordering);
            }
        }
    }

    static void BranchAndBoundExact(int[] ordering)
    {
        var clique = new List<int>();
        var candidates = Enumerable.Range(0, n).ToList();
        SearchExact(clique, candidates, ordering);
    }

    static void SearchExact(List<int> clique, List<int> candidates, int[] ordering)
    {
        if (candidates.Count == 0)
        {
            if (clique.Count > bestSize)
            {
                bestSize = clique.Count;
                bestClique = clique.ToArray();
            }
            return;
        }

        if (stopwatch.ElapsedMilliseconds > timeLimitMs)
        {
            timeoutReached = true;
            return;
        }

        // Upper bound check
        int upperBound = clique.Count + candidates.Count;
        if (upperBound <= bestSize) return;

        // Use coloring for better pruning
        var colors = GreedyColoring(candidates);
        upperBound = clique.Count + colors.Values.Max();
        if (upperBound <= bestSize) return;

        // Sort by color (process highest first)
        var sortedCandidates = candidates.OrderByDescending(c => colors[c]).ToList();

        foreach (var v in sortedCandidates)
        {
            if (clique.Count + colors[v] <= bestSize) break;
            if (timeoutReached) return;

            var newClique = new List<int>(clique) { v };
            var newCandidates = candidates.Where(c => c != v && adj[v][c]).ToList();

            SearchExact(newClique, newCandidates, ordering);
        }
    }

    static void HybridStrategy()
    {
        // Start with heuristic solution
        GreedyClique();
        LocalSearch();

        // Run branch and bound with time limit
        var ordering = GetOrdering();
        
        Task.Run(() =>
        {
            BranchAndBoundExact(ordering);
        }).Wait(Math.Min(timeLimitMs, 120000));
    }

    static void LargeGraphStrategy()
    {
        // For very large graphs, use greedy + local search + parallel improvement
        GreedyClique();
        LocalSearch();
        
        if (stopwatch.ElapsedMilliseconds < timeLimitMs / 2)
        {
            ParallelImprovement();
        }
    }

    static void GreedyClique()
    {
        // Multiple greedy strategies
        var strategies = new List<Func<List<int>>>
        {
            GreedyByDegree,
            GreedyByOrdering,
            GreedyByColoring
        };

        foreach (var strategy in strategies)
        {
            if (timeoutReached) break;
            var clique = strategy();
            UpdateBest(clique);
        }
    }

    static List<int> GreedyByDegree()
    {
        var clique = new List<int>();
        var degrees = new int[n];
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                if (adj[i][j] && i != j) degrees[i]++;

        var candidates = Enumerable.Range(0, n).OrderByDescending(x => degrees[x]).ToList();

        while (candidates.Count > 0)
        {
            int v = candidates[0];
            clique.Add(v);
            candidates = candidates.Where(c => c != v && adj[v][c]).ToList();
        }

        return clique;
    }

    static List<int> GreedyByOrdering()
    {
        var clique = new List<int>();
        var ordering = GetOrdering();
        var remaining = new HashSet<int>(ordering);

        foreach (int v in ordering)
        {
            if (!remaining.Contains(v)) continue;
            
            var newClique = new List<int> { v };
            var neighbors = new HashSet<int>();
            
            for (int i = 0; i < n; i++)
            {
                if (i != v && adj[v][i]) neighbors.Add(i);
            }

            foreach (int u in ordering)
            {
                if (u != v && remaining.Contains(u) && neighbors.Contains(u))
                {
                    bool canAdd = true;
                    foreach (int c in newClique)
                    {
                        if (!adj[c][u])
                        {
                            canAdd = false;
                            break;
                        }
                    }
                    if (canAdd) newClique.Add(u);
                }
            }

            if (newClique.Count > clique.Count)
                clique = newClique;
        }

        return clique;
    }

    static List<int> GreedyByColoring()
    {
        var clique = new List<int>();
        var remaining = Enumerable.Range(0, n).ToList();

        while (remaining.Count > 0)
        {
            // Pick vertex with highest color number
            var colors = GreedyColoring(remaining);
            int v = remaining.OrderByDescending(c => colors[c]).First();
            
            clique.Add(v);
            remaining = remaining.Where(c => c != v && adj[v][c]).ToList();
        }

        return clique;
    }

    static void LocalSearch()
    {
        if (bestClique == null || bestClique.Length == 0) return;

        var cliqueSet = new HashSet<int>(bestClique);
        var candidates = new HashSet<int>();

        // Find candidates (vertices adjacent to all clique members)
        for (int i = 0; i < n; i++)
        {
            if (!cliqueSet.Contains(i))
            {
                bool adjacentToAll = true;
                foreach (int v in cliqueSet)
                {
                    if (!adj[i][v])
                    {
                        adjacentToAll = false;
                        break;
                    }
                }
                if (adjacentToAll) candidates.Add(i);
            }
        }

        // Try to add candidates
        while (candidates.Count > 0 && !timeoutReached)
        {
            int best = -1;
            int bestNeighborCount = -1;

            foreach (int c in candidates)
            {
                int count = 0;
                foreach (int other in candidates)
                {
                    if (other != c && adj[c][other]) count++;
                }
                if (count > bestNeighborCount)
                {
                    bestNeighborCount = count;
                    best = c;
                }
            }

            if (best >= 0)
            {
                cliqueSet.Add(best);
                candidates.Remove(best);
                // Update candidates
                var toRemove = new List<int>();
                foreach (int c in candidates)
                {
                    if (!adj[best][c]) toRemove.Add(c);
                }
                foreach (int c in toRemove) candidates.Remove(c);
            }
            else break;
        }

        UpdateBest(cliqueSet.ToList());
    }

    static void ParallelImprovement()
    {
        // Try to improve clique in parallel using swap operations
        if (bestClique == null || bestClique.Length == 0) return;

        var currentBest = bestClique.ToList();
        bool improved = true;
        int iterations = 0;
        int maxIterations = n / 10;

        while (improved && iterations < maxIterations && !timeoutReached)
        {
            improved = false;
            iterations++;

            var cliqueSet = new HashSet<int>(currentBest);

            // Try swapping each vertex out
            Parallel.For(0, currentBest.Count, (i, state) =>
            {
                if (timeoutReached || improved)
                    return;

                var testClique = new List<int>(currentBest);
                int removed = testClique[i];
                testClique.RemoveAt(i);

                // Find candidates to add instead
                var candidates = new List<int>();
                for (int v = 0; v < n; v++)
                {
                    if (v == removed) continue;
                    bool canAdd = true;
                    foreach (int c in testClique)
                    {
                        if (!adj[v][c])
                        {
                            canAdd = false;
                            break;
                        }
                    }
                    if (canAdd) candidates.Add(v);
                }

                // Try to add multiple candidates
                while (candidates.Count > 0)
                {
                    int best = candidates[0];
                    testClique.Add(best);
                    candidates = candidates.Where(c => c != best && adj[best][c]).ToList();
                }

                lock (lockObj)
                {
                    if (testClique.Count > currentBest.Count)
                    {
                        currentBest = testClique;
                        improved = true;
                    }
                }
            });
        }

        UpdateBest(currentBest);
    }

    static void UpdateBest(List<int> clique)
    {
        lock (lockObj)
        {
            if (clique.Count > bestSize)
            {
                bestSize = clique.Count;
                bestClique = clique.ToArray();
            }
        }
    }

    static int[] GetOrdering()
    {
        // Create ordering based on degrees and structure
        var degrees = new int[n];
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                if (adj[i][j] && i != j) degrees[i]++;

        if (isDense)
        {
            // For dense graphs, order by degree descending
            return Enumerable.Range(0, n).OrderByDescending(x => degrees[x]).ToArray();
        }
        else
        {
            // For sparse graphs, use more sophisticated ordering
            return Enumerable.Range(0, n)
                .OrderBy(x => degrees[x])
                .ThenBy(x => x)
                .ToArray();
        }
    }

    static Dictionary<int, int> GreedyColoring(List<int> vertices)
    {
        var colors = new Dictionary<int, int>();
        var usedColors = new HashSet<int>();

        foreach (int v in vertices)
        {
            usedColors.Clear();
            
            foreach (int u in vertices)
            {
                if (u != v && adj[v][u] && colors.ContainsKey(u))
                {
                    usedColors.Add(colors[u]);
                }
            }

            int color = 1;
            while (usedColors.Contains(color)) color++;
            colors[v] = color;
        }

        return colors;
    }

    static void OutputResult()
    {
        // Verify clique is valid
        if (bestClique != null && bestClique.Length > 0)
        {
            for (int i = 0; i < bestClique.Length; i++)
            {
                for (int j = i + 1; j < bestClique.Length; j++)
                {
                    if (!adj[bestClique[i]][bestClique[j]])
                    {
                        // Invalid clique - output single vertex
                        Console.WriteLine("1");
                        Console.WriteLine("0");
                        return;
                    }
                }
            }

            Console.WriteLine(bestClique.Length);
            Console.WriteLine(string.Join(" ", bestClique.OrderBy(x => x)));
        }
        else
        {
            Console.WriteLine("1");
            Console.WriteLine("0");
        }
    }
