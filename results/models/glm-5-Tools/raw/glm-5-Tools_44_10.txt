```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize interpolated normal
    vec3 N = normalize(normal);
    
    // Light direction (from surface to light)
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    
    // Lambertian diffuse
    float NdotL = max(dot(N, L), 0.0);
    float diffuse = NdotL;
    
    // Horizontal stripes based on uv.y
    float stripe = sin(uv.y * 3.14159 * 16.0);
    
    // Alternate between red (0.9, 0.1, 0.1) and blue (0.1, 0.1, 0.9)
    vec3 red = vec3(0.9, 0.1, 0.1);
    vec3 blue = vec3(0.1, 0.1, 0.9);
    vec3 baseColor = mix(red, blue, step(0.0, stripe));
    
    // Apply shading factor (0.4 + 0.6*diffuse)
    vec3 shadedColor = baseColor * (0.4 + 0.6 * diffuse);
    
    outColor = vec4(shadedColor, 1.0);
}
```
