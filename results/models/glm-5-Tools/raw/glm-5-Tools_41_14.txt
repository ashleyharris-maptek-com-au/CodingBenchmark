; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor %ubo
OpExecutionMode %main OriginUpperLeft

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%ptrInput3 = OpTypePointer Input %v3float
%ptrOutput4 = OpTypePointer Output %v4float

%worldPos = OpVariable %ptrInput3 Input
%normal = OpVariable %ptrInput3 Input
%outColor = OpVariable %ptrOutput4 Output

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptrUBO = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptrUBO Uniform

%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_3 = OpConstant %float 0.3
%float_0_7 = OpConstant %float 0.7
%float_1_5 = OpConstant %float 1.5
%float_3 = OpConstant %float 3
%float_0_8 = OpConstant %float 0.8
%float_0_2 = OpConstant %float 0.2
%float_0_7_2 = OpConstant %float 0.7

%baseR = OpConstant %float 0.8
%baseG = OpConstant %float 0.2
%baseB = OpConstant %float 0.2
%fogR = OpConstant %float 0.7
%fogG = OpConstant %float 0.7
%fogB = OpConstant %float 0.8

%main = OpFunction %void None %func
%entry = OpLabel

; Load inputs
%pos = OpLoad %v3float %worldPos
%norm = OpLoad %v3float %normal

; Normalize the interpolated normal
%normLen = OpExtInst %float %1 Length %norm
%normNorm = OpVectorTimesScalar %v3float %norm %normLen
%normNorm2 = OpFMul %float %normLen %normLen
%normNorm3 = OpFDiv %float %float_1 %normNorm2
%normNorm4 = OpVectorTimesScalar %v3float %norm %normNorm3

; Load lightPos (member 3)
%lightPtr = OpAccessChain %ptrUniformVec4 %ubo %int_3
%lightPos = OpLoad %v4float %lightPtr
%lightPos3 = OpVectorShuffle %v3float %lightPos %lightPos 0 1 2

; Load cameraPos (member 4)
%camPtr = OpAccessChain %ptrUniformVec4 %ubo %int_4
%camPos = OpLoad %v4float %camPtr
%camPos3 = OpVectorShuffle %v3float %camPos %camPos 0 1 2

; Light direction
%lightDir = OpFSub %v3float %lightPos3 %pos
%lightDirNorm = OpExtInst %v3float %1 Normalize %lightDir

; Diffuse = max(dot(norm, lightDir), 0)
%dotNL = OpDot %float %normNorm4 %lightDirNorm
%diffuse = OpExtInst %float %1 FMax %dotNL %float_0

; Lambertian factor = 0.3 + 0.7 * diffuse
%lamMul = OpFMul %float %float_0_7 %diffuse
%lamFactor = OpFAdd %float %float_0_3 %lamMul

; Base color = (0.8, 0.2, 0.2) * lambertian
%baseColorR = OpFMul %float %baseR %lamFactor
%baseColorG = OpFMul %float %baseG %lamFactor
%baseColorB = OpFMul %float %baseB %lamFactor
%baseColor = OpCompositeConstruct %v3float %baseColorR %baseColorG %baseColorB

; Distance from camera
%diffCam = OpFSub %v3float %camPos3 %pos
%dist = OpExtInst %float %1 Length %diffCam

; Fog factor = clamp((dist - 1.5) / 3.0, 0, 1)
%distMinus = OpFSub %float %dist %float_1_5
%fogRaw = OpFDiv %float %distMinus %float_3
%fogFactor = OpExtInst %float %1 FClamp %fogRaw %float_0 %float_1

; Fog color = (0.7, 0.7, 0.8)
%fogColor = OpCompositeConstruct %v3float %fogR %fogG %fogB

; Mix = baseColor * (1 - fogFactor) + fogColor * fogFactor
%oneMinusFog = OpFSub %float %float_1 %fogFactor
%baseMul = OpVectorTimesScalar %v3float %baseColor %oneMinusFog
%fogMul = OpVectorTimesScalar %v3float %fogColor %fogFactor
%finalColor = OpFAdd %v3float %baseMul %fogMul

; Output with alpha 1.0
%outVec = OpCompositeConstruct %v4float %finalColor %float_1
OpStore %outColor %outVec

OpReturn
OpFunctionEnd
