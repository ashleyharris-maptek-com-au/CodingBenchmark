import sys

def solve_exact_cover(universe_size, sets):
    """
    Solve exact cover problem using backtracking (Algorithm X).
    """
    if universe_size == 0:
        return []
    
    universe = set(range(universe_size))
    
    # Build reverse mapping: element -> sets that contain it
    element_to_sets = [set() for _ in range(universe_size)]
    for i, s in enumerate(sets):
        for e in s:
            if 0 <= e < universe_size:
                element_to_sets[e].add(i)
    
    selected = []
    
    def backtrack(remaining, available):
        if not remaining:
            return True
        
        # Choose element covered by fewest available sets (heuristic)
        chosen = min(remaining, key=lambda e: len(element_to_sets[e] & available))
        candidates = element_to_sets[chosen] & available
        
        if not candidates:
            return False
        
        for set_idx in list(candidates):
            selected.append(set_idx)
            
            # Find all conflicting sets (those sharing elements with chosen set)
            conflict = set()
            for e in sets[set_idx]:
                conflict |= element_to_sets[e]
            
            if backtrack(remaining - sets[set_idx], available - conflict):
                return True
            
            selected.pop()
        
        return False
    
    if backtrack(universe, set(range(len(sets)))):
        return sorted(selected)
    return None

**Key features of this solution:**

1. **Algorithm X with heuristic**: Uses backtracking with the "minimum remaining values" heuristic - always picks the element covered by the fewest available sets first, which prunes the search space effectively.

2. **Conflict detection**: When a set is chosen, all sets that share any element with it are removed from consideration (exact cover constraint).

4. **Edge cases handled**: Empty universe, empty sets, no solution cases.
