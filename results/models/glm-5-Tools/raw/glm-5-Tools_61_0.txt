Here's the complete Node.js TypeScript program for the delivery route planning task:

```typescript
declare const require: any; const fs = require('fs'); const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Package {
    id: number;
    x: number;
    y: number;
}

interface Input {
    depot: [number, number];
    truckCount: number;
    packages: Package[];
}

const inputData = data as Input;
const depot: [number, number] = inputData.depot;
const truckCount: number = inputData.truckCount;
const packages: Package[] = inputData.packages;

function solveVRP(depot: [number, number], packages: Package[], truckCount: number): number[][] {
    if (packages.length === 0) {
        return [];
    }

    const pkgMap = new Map<number, Package>();
    for (const p of packages) {
        pkgMap.set(p.id, p);
    }

    const n = packages.length;
    const pids = packages.map(p => p.id);

    if (truckCount >= n) {
        return pids.map(pid => [pid]);
    }

    const depotPos: [number, number] = [depot[0], depot[1]];

    // For small problems, try both sweep and savings, pick best
    if (n <= 500) {
        const routesSweep = sweepClustering(depot, pkgMap, pids, truckCount);
        const routesSavings = savingsAlgorithm(depot, pkgMap, pids, truckCount);

        const distSweep = totalDistance(routesSweep, depotPos, pkgMap);
        const distSavings = totalDistance(routesSavings, depotPos, pkgMap);

        if (distSavings < distSweep) {
            return routesSavings;
        }
        return routesSweep;
    } else {
        return sweepClustering(depot, pkgMap, pids, truckCount);
    }
}

function sweepClustering(
    depot: [number, number],
    pkgMap: Map<number, Package>,
    pids: number[],
    truckCount: number
): number[][] {
    const n = pids.length;
    const depotPos: [number, number] = [depot[0], depot[1]];

    // Sort by angle from depot
    const angles: [number, number][] = [];
    for (const pid of pids) {
        const p = pkgMap.get(pid)!;
        const angle = Math.atan2(p.y - depotPos[1], p.x - depotPos[0]);
        angles.push([angle, pid]);
    }

    angles.sort((a, b) => a[0] - b[0]);
    const sortedPids = angles.map(a => a[1]);

    // Partition into clusters
    const baseSize = Math.floor(n / truckCount);
    const remainder = n % truckCount;

    const routes: number[][] = [];
    let idx = 0;
    for (let t = 0; t < truckCount; t++) {
        const size = baseSize + (t < remainder ? 1 : 0);
        if (idx < n) {
            const cluster = sortedPids.slice(idx, idx + size);
            if (cluster.length > 0) {
                let route = nearestNeighbor(cluster, depotPos, pkgMap);
                route = twoOpt(route, depot, pkgMap);
                routes.push(route);
            }
            idx += size;
        }
    }

    return routes;
}

function savingsAlgorithm(
    depot: [number, number],
    pkgMap: Map<number, Package>,
    pids: number[],
    truckCount: number
): number[][] {
    const n = pids.length;
    const depotPos: [number, number] = [depot[0], depot[1]];

    // Calculate savings
    interface Saving {
        saving: number;
        i: number;
        j: number;
    }
    const savings: Saving[] = [];

    for (let i = 0; i < n; i++) {
        for (let j = i + 1; j < n; j++) {
            const pi = pkgMap.get(pids[i])!;
            const pj = pkgMap.get(pids[j])!;
            const dDi = Math.hypot(depotPos[0] - pi.x, depotPos[1] - pi.y);
            const dDj = Math.hypot(depotPos[0] - pj.x, depotPos[1] - pj.y);
            const dIj = Math.hypot(pi.x - pj.x, pi.y - pj.y);
            savings.push({ saving: dDi + dDj - dIj, i, j });
        }
    }

    savings.sort((a, b) => b.saving - a.saving);

    // Initialize routes
    const routes: number[][] = [];
    for (let i = 0; i < n; i++) {
        routes.push([pids[i]]);
    }

    const routeOf: number[] = [];
    for (let i = 0; i < n; i++) {
        routeOf.push(i);
    }

    const routeStart: number[] = [];
    const routeEnd: number[] = [];
    for (let i = 0; i < n; i++) {
        routeStart.push(i);
        routeEnd.push(i);
    }

    const active: boolean[] = [];
    for (let i = 0; i < n; i++) {
        active.push(true);
    }

    const pidToIdx = new Map<number, number>();
    for (let i = 0; i < n; i++) {
        pidToIdx.set(pids[i], i);
    }

    // Merge routes based on savings
    for (const { saving, i, j } of savings) {
        const ri = routeOf[i];
        const rj = routeOf[j];

        if (!active[ri] || !active[rj] || ri === rj) {
            continue;
        }

        const iAtStart = routeStart[ri] === i;
        const iAtEnd = routeEnd[ri] === i;
        const jAtStart = routeStart[rj] === j;
        const jAtEnd = routeEnd[rj] === j;

        if (!((iAtStart || iAtEnd) && (jAtStart || jAtEnd))) {
            continue;
        }

        let merged: number[] | null = null;
        let newStart = -1;
        let newEnd = -1;

        if (iAtEnd && jAtStart) {
            merged = routes[ri].concat(routes[rj]);
            newStart = routeStart[ri];
            newEnd = routeEnd[rj];
        } else if (iAtStart && jAtEnd) {
            merged = routes[rj].concat(routes[ri]);
            newStart = routeStart[rj];
            newEnd = routeEnd[ri];
        } else if (iAtEnd && jAtEnd) {
            merged = routes[ri].concat(routes[rj].slice().reverse());
            newStart = routeStart[ri];
            newEnd = routeStart[rj];
        } else if (iAtStart && jAtStart) {
            merged = routes[ri].slice().reverse().concat(routes[rj]);
            newStart = routeEnd[ri];
            newEnd = routeEnd[rj];
        }

        if (merged === null) {
            continue;
        }

        for (const pid of routes[rj]) {
            routeOf[pidToIdx.get(pid)!] = ri;
        }

        routes[ri] = merged;
        routeStart[ri] = newStart;
        routeEnd[ri] = newEnd;
        active[rj] = false;
    }

    const finalRoutes: number[][] = [];
    for (let i = 0; i < n; i++) {
        if (active[i]) {
            finalRoutes.push(routes[i]);
        }
    }

    // Apply 2-opt
    for (let i = 0; i < finalRoutes.length; i++) {
        finalRoutes[i] = twoOpt(finalRoutes[i], depot, pkgMap);
    }

    // Consolidate if needed
    while (finalRoutes.length > truckCount) {
        let bestCost = Infinity;
        let bestMerge: { i: number; j: number; merged: number[] } | null = null;

        for (let i = 0; i < finalRoutes.length; i++) {
            for (let j = i + 1; j < finalRoutes.length; j++) {
                const ri = finalRoutes[i];
                const rj = finalRoutes[j];

                for (const revI of [false, true]) {
                    for (const revJ of [false, true]) {
                        const testRi = revI ? ri.slice().reverse() : ri;
                        const testRj = revJ ? rj.slice().reverse() : rj;
                        const p1 = pkgMap.get(testRi[testRi.length - 1])!;
                        const p2 = pkgMap.get(testRj[0])!;
                        const cost = Math.hypot(p1.x - p2.x, p1.y - p2.y);
                        if (cost < bestCost) {
                            bestCost = cost;
                            bestMerge = { i, j, merged: testRi.concat(testRj) };
                        }
                    }
                }
            }
        }

        if (bestMerge) {
            finalRoutes[bestMerge.i] = twoOpt(bestMerge.merged, depot, pkgMap);
            finalRoutes.splice(bestMerge.j, 1);
        }
    }

    return finalRoutes;
}

function nearestNeighbor(
    pids: number[],
    depotPos: [number, number],
    pkgMap: Map<number, Package>
): number[] {
    if (pids.length === 0) {
        return [];
    }

    const remaining = new Set(pids);
    const route: number[] = [];
    let cx = depotPos[0];
    let cy = depotPos[1];

    while (remaining.size > 0) {
        let best = -1;
        let bestDist = Infinity;

        for (const pid of remaining) {
            const p = pkgMap.get(pid)!;
            const d = (cx - p.x) ** 2 + (cy - p.y) ** 2;
            if (d < bestDist) {
                bestDist = d;
                best = pid;
            }
        }

        route.push(best);
        remaining.delete(best);
        const p = pkgMap.get(best)!;
        cx = p.x;
        cy = p.y;
    }

    return route;
}

function twoOpt(
    route: number[],
    depot: [number, number],
    pkgMap: Map<number, Package>
): number[] {
    const n = route.length;
    if (n < 3) {
        return route;
    }

    const px: number[] = [];
    const py: number[] = [];
    for (const pid of route) {
        const p = pkgMap.get(pid)!;
        px.push(p.x);
        py.push(p.y);
    }
    const dx = depot[0];
    const dy = depot[1];

    let improved = true;
    while (improved) {
        improved = false;

        for (let i = 0; i < n - 1; i++) {
            for (let j = i + 2; j < n; j++) {
                let old = Math.hypot(px[i] - px[i + 1], py[i] - py[i + 1]);
                if (j < n - 1) {
                    old += Math.hypot(px[j] - px[j + 1], py[j] - py[j + 1]);
                } else {
                    old += Math.hypot(px[j] - dx, py[j] - dy);
                }

                let newDist = Math.hypot(px[i] - px[j], py[i] - py[j]);
                if (j < n - 1) {
                    newDist += Math.hypot(px[i + 1] - px[j + 1], py[i + 1] - py[j + 1]);
                } else {
                    newDist += Math.hypot(px[i + 1] - dx, py[i + 1] - dy);
                }

                if (newDist < old - 1e-9) {
                    // Reverse the segment
                    route = route.slice(0, i + 1).concat(route.slice(i + 1, j + 1).reverse()).concat(route.slice(j + 1));
                    
                    // Update px and py
                    const newPx = px.slice(0, i + 1).concat(px.slice(i + 1, j + 1).reverse()).concat(px.slice(j + 1));
                    const newPy = py.slice(0, i + 1).concat(py.slice(i + 1, j + 1).reverse()).concat(py.slice(j + 1));
                    px.length = 0;
                    py.length = 0;
                    px.push(...newPx);
                    py.push(...newPy);
                    
                    improved = true;
                    break;
                }
            }
            if (improved) {
                break;
            }
        }
    }

    return route;
}

function totalDistance(
    routes: number[][],
    depotPos: [number, number],
    pkgMap: Map<number, Package>
): number {
    let total = 0;
    for (const route of routes) {
        if (route.length === 0) continue;
        
        total += Math.hypot(
            depotPos[0] - pkgMap.get(route[0])!.x,
            depotPos[1] - pkgMap.get(route[0])!.y
        );
        
        for (let i = 0; i < route.length - 1; i++) {
            total += Math.hypot(
                pkgMap.get(route[i])!.x - pkgMap.get(route[i + 1])!.x,
                pkgMap.get(route[i])!.y - pkgMap.get(route[i + 1])!.y
            );
        }
        
        total += Math.hypot(
            depotPos[0] - pkgMap.get(route[route.length - 1])!.x,
            depotPos[1] - pkgMap.get(route[route.length - 1])!.y
        );
    }
    return total;
}

const result = solveVRP(depot, packages, truckCount);
console.log(JSON.stringify({ routes: result }));
```

This program implements a hybrid approach for solving the Vehicle Routing Problem:

1. **Sweep Algorithm**: Partitions packages by angle from the depot, creating clusters that are optimized with nearest neighbor and 2-opt.

2. **Clarke-Wright Savings Algorithm**: Merges routes based on savings (distance saved by combining two routes), with 2-opt optimization.

3. **Hybrid Selection**: For problems with ≤500 packages, both algorithms are tried and the better solution is returned. For larger problems, only the faster sweep algorithm is used.

4. **2-opt Local Search**: Optimizes each route by iteratively reversing segments that reduce total distance.

The algorithm scales efficiently with O(n²) for savings calculation and O(n²) per 2-opt iteration, making it suitable for large-scale problems while maintaining good solution quality.
