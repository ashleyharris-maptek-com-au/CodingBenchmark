```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize interpolated normal
    float3 N = normalize(input.normal);
    
    // Light direction (from surface to light)
    float3 L = normalize(lightPos.xyz - input.worldPos);
    
    // Lambertian diffuse
    float diffuse = max(dot(N, L), 0.0);
    
    // UV checkerboard at 8x8 frequency
    float2 checkerUV = floor(input.uv * 8.0);
    float checker = fmod(checkerUV.x + checkerUV.y, 2.0);
    
    // Alternating colors 0.9 and 0.2 by parity
    float checkerColor = lerp(0.9, 0.2, checker);
    
    // Apply shading factor (0.5 + 0.5*diffuse)
    float shadeFactor = 0.5 + 0.5 * diffuse;
    
    // Final color
    float3 finalColor = checkerColor * shadeFactor;
    
    return float4(finalColor, 1.0);
}
```
