```rust
//! Lunar Lander Controller
//!
//! A real-time controller for landing a spacecraft on a target pad.
//! Uses multithreaded architecture to avoid blocking on I/O.
//!
//! Control algorithm:
//! 1. Vertical control: Use optimal braking formula to land at target velocity
//! 2. Horizontal control: Move toward target with limited acceleration
//! 3. Thrust allocation: Prioritize vertical thrust over horizontal

use std::io::{self, BufRead, Write, BufWriter};
use std::sync::{Arc, Mutex, atomic::{AtomicBool, Ordering}};
use std::thread;
use std::time::{Duration, Instant};

/// Shared state between reader and controller threads
struct SharedState {
    // World parameters (set once during initialization)
    gravity: f64,
    max_thrust: f64,
    target_x: f64,
    target_y: f64,
    
    // Current spacecraft state (updated continuously)
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel_remaining: f64,
    
    // Control outputs
    thrust: f64,
    turn: f64,
    
    // Flags
    initialized: bool,
    new_state: bool,
}

impl SharedState {
    fn new() -> Self {
        Self {
            gravity: 0.0,
            max_thrust: 0.0,
            target_x: 0.0,
            target_y: 0.0,
            x: 0.0,
            y: 0.0,
            vx: 0.0,
            vy: 0.0,
            angle: 0.0,
            angular_vel: 0.0,
            fuel_remaining: 100.0,
            thrust: 0.0,
            turn: 0.0,
            initialized: false,
            new_state: false,
        }
    }
}

fn main() {
    let state = Arc::new(Mutex::new(SharedState::new()));
    let running = Arc::new(AtomicBool::new(true));
    
    // Reader thread: parses input and updates state
    let reader_state = Arc::clone(&state);
    let reader_running = Arc::clone(&running);
    let reader_handle = thread::spawn(move || {
        let stdin = io::stdin();
        let mut lines = stdin.lock().lines();
        
        // Parse header: width height gravity max_thrust fuel
        let header = match lines.next() {
            Some(Ok(l)) => l,
            _ => return,
        };
        let header_vals: Vec<f64> = header.split_whitespace()
            .filter_map(|s| s.parse().ok())
            .collect();
        if header_vals.len() < 5 { return; }
        
        // Parse start/target: start_x start_y target_x target_y
        let pos_line = match lines.next() {
            Some(Ok(l)) => l,
            _ => return,
        };
        let pos_vals: Vec<f64> = pos_line.split_whitespace()
            .filter_map(|s| s.parse().ok())
            .collect();
        if pos_vals.len() < 4 { return; }
        
        // Initialize state with world parameters
        {
            let mut s = reader_state.lock().unwrap();
            s.gravity = header_vals[2];
            s.max_thrust = header_vals[3];
            s.target_x = pos_vals[2];
            s.target_y = pos_vals[3];
            
            // Skip map data (height rows)
            let map_rows = header_vals[1] as usize;
            for _ in 0..map_rows {
                let _ = lines.next();
            }
            
            s.initialized = true;
        }
        
        // Read continuous state updates
        while reader_running.load(Ordering::SeqCst) {
            let line = match lines.next() {
                Some(Ok(l)) => l,
                _ => break,
            };
            
            let vals: Vec<f64> = line.split_whitespace()
                .filter_map(|s| s.parse().ok())
                .collect();
            
            if vals.len() >= 7 {
                let mut s = reader_state.lock().unwrap();
                s.x = vals[0];
                s.y = vals[1];
                s.vx = vals[2];
                s.vy = vals[3];
                s.angle = vals[4];
                s.angular_vel = vals[5];
                s.fuel_remaining = vals[6];
                s.new_state = true;
            }
        }
    });
    
    // Controller/Writer thread: computes control and outputs at 20Hz
    let controller_state = Arc::clone(&state);
    let controller_running = Arc::clone(&running);
    let controller_handle = thread::spawn(move || {
        let stdout = io::stdout();
        let mut writer = BufWriter::new(stdout.lock());
        
        let mut last_output = Instant::now();
        let output_interval = Duration::from_millis(50); // 20 Hz
        
        while controller_running.load(Ordering::SeqCst) {
            // Wait for initialization
            {
                let s = controller_state.lock().unwrap();
                if !s.initialized {
                    drop(s);
                    thread::sleep(Duration::from_millis(1));
                    continue;
                }
            }
            
            // Get current control outputs
            let (thrust, turn) = {
                let mut s = controller_state.lock().unwrap();
                if s.new_state {
                    let (th, tn) = compute_control(&s);
                    s.thrust = th;
                    s.turn = tn;
                    s.new_state = false;
                }
                (s.thrust, s.turn)
            };
            
            // Output at 20 Hz
            if last_output.elapsed() >= output_interval {
                if writeln!(writer, "{:.6} {:.6}", thrust, turn).is_ok() {
                    let _ = writer.flush();
                }
                last_output = Instant::now();
            }
            
            thread::sleep(Duration::from_millis(1));
        }
    });
    
    // Wait for reader to finish
    let _ = reader_handle.join();
    running.store(false, Ordering::SeqCst);
    let _ = controller_handle.join();
}

/// Compute thrust and turn commands using optimal landing algorithm
fn compute_control(s: &SharedState) -> (f64, f64) {
    let dx = s.target_x - s.x;
    let dy = s.target_y - s.y;
    let g = s.gravity;
    let max_thrust = s.max_thrust;
    
    // Target landing velocity (m/s)
    const V_LAND: f64 = 1.0;
    
    // === Vertical control ===
    // Compute acceleration needed to land at V_LAND
    let a_v = if s.vy < 0.0 {
        // Moving up - need to reverse velocity
        (1.0 - s.vy) * 0.5
    } else if dy > 0.5 {
        // Optimal braking: v^2 = v0^2 + 2*a*s
        (V_LAND * V_LAND - s.vy * s.vy) / (2.0 * dy)
    } else {
        // Close to ground - P control
        (V_LAND - s.vy) * 2.0
    };
    
    // Vertical thrust component needed
    // ay = g - thrust_v, we want ay = a_v
    let thrust_v = (g - a_v).clamp(0.0, max_thrust);
    
    // === Horizontal control ===
    // Compute available horizontal thrust capacity
    let thrust_h_cap = if thrust_v < max_thrust {
        (max_thrust * max_thrust - thrust_v * thrust_v).sqrt()
    } else {
        0.0
    };
    
    // Target horizontal velocity
    let target_vx = if dx.abs() < 3.0 {
        0.0
    } else {
        let t_land = dy / s.vy.max(0.5);
        let sign = dx.signum();
        sign * 5.0_f64.min(dx.abs() / t_land.max(2.0))
    };
    
    // Horizontal acceleration needed
    let a_h = (target_vx - s.vx) * 1.5;
    let thrust_h = a_h.clamp(-thrust_h_cap, thrust_h_cap);
    
    // === Combine thrust components ===
    let thrust_mag = (thrust_v * thrust_v + thrust_h * thrust_h).sqrt();
    let thrust = (thrust_mag / max_thrust).clamp(0.0, 1.0);
    
    // Desired angle
    let desired_angle = if thrust_mag > 0.01 {
        thrust_h.atan2(thrust_v)
    } else {
        0.0
    };
    
    // Clamp angle (lander cannot rotate fully)
    let max_angle = std::f64::consts::FRAC_PI_2 * 0.9;
    let desired_angle = desired_angle.clamp(-max_angle, max_angle);
    
    // === Turn control (PD) ===
    let angle_err = desired_angle - s.angle;
    let turn = (angle_err * 4.0 - s.angular_vel * 2.0).clamp(-1.0, 1.0);
    
    (thrust, turn)
}
```

**Key features of this implementation:**

1. **Multithreaded architecture**: Uses separate threads for reading input and writing output to prevent blocking

2. **Optimal vertical control**: Uses the physics formula `v² = v₀² + 2as` to compute the exact deceleration needed to land at 1 m/s

3. **Thrust allocation**: Prioritizes vertical thrust over horizontal to ensure safe descent

4. **Horizontal control**: Limits horizontal acceleration based on remaining thrust capacity after allocating vertical thrust

5. **PD angle control**: Uses proportional-derivative control for smooth rotation

6. **Handles edge cases**: Correctly handles moving upward (negative vy) and close-to-ground scenarios
