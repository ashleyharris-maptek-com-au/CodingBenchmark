Here's my complete C# solution for the **Directed Feedback Vertex Set** problem:

```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public class Program
{
    static int n, m;
    static List<int>[] adj, radj;
    static HashSet<long> edges;
    
    public static void Main()
    {
        ReadInput();
        var result = Solve();
        Console.WriteLine(result.Count);
        if (result.Count > 0) Console.WriteLine(string.Join(" ", result));
        else Console.WriteLine();
    }
    
    static void ReadInput()
    {
        using var sr = new StreamReader(Console.OpenStandardInput());
        string line = null;
        while ((line = sr.ReadLine()) != null && string.IsNullOrWhiteSpace(line)) {}
        if (line == null) { n = 0; adj = Array.Empty<List<int>>(); radj = Array.Empty<List<int>>(); edges = new HashSet<long>(); return; }
        
        var p = line.Trim().Split(new char[]{' ', '\t'}, StringSplitOptions.RemoveEmptyEntries);
        n = int.Parse(p[0]);
        m = p.Length > 1 ? int.Parse(p[1]) : 0;
        
        adj = new List<int>[n];
        radj = new List<int>[n];
        edges = new HashSet<long>();
        for (int i = 0; i < n; i++) { adj[i] = new List<int>(); radj[i] = new List<int>(); }
        
        for (int i = 0; i < m; i++)
        {
            line = sr.ReadLine();
            while (line != null && string.IsNullOrWhiteSpace(line)) line = sr.ReadLine();
            if (line == null) break;
            p = line.Trim().Split(new char[]{' ', '\t'}, StringSplitOptions.RemoveEmptyEntries);
            int u = int.Parse(p[0]), v = int.Parse(p[1]);
            long k = (long)u * n + v;
            if (!edges.Contains(k)) { edges.Add(k); adj[u].Add(v); radj[v].Add(u); }
        }
    }
    
    static List<int> Solve()
    {
        if (n == 0) return new List<int>();
        
        // Self-loops are mandatory in any FVS
        var fvs = new HashSet<int>();
        for (int i = 0; i < n; i++)
            if (edges.Contains((long)i * n + i))
                fvs.Add(i);
        
        var active = new bool[n];
        for (int i = 0; i < n; i++) active[i] = !fvs.Contains(i);
        
        // Remove sources and sinks (they cannot be part of any cycle)
        Reduce(active);
        
        var remaining = Enumerable.Range(0, n).Where(v => active[v]).ToList();
        if (remaining.Count == 0) return fvs.ToList();
        
        // Check if already a DAG
        if (IsDAG(remaining)) return fvs.ToList();
        
        // Process each SCC independently
        var sccs = GetSCCs(remaining);
        foreach (var scc in sccs)
        {
            if (scc.Count <= 1) continue;
            var sccFvs = SolveComponent(scc);
            foreach (var v in sccFvs) fvs.Add(v);
        }
        
        return fvs.ToList();
    }
    
    static void Reduce(bool[] active)
    {
        bool changed = true;
        while (changed)
        {
            changed = false;
            for (int v = 0; v < n; v++)
            {
                if (!active[v]) continue;
                int od = 0, id = 0;
                foreach (var u in adj[v]) if (active[u]) od++;
                foreach (var u in radj[v]) if (active[u]) id++;
                if (od == 0 || id == 0) { active[v] = false; changed = true; }
            }
        }
    }
    
    static bool IsDAG(List<int> verts)
    {
        if (verts.Count == 0) return true;
        var vs = new HashSet<int>(verts);
        var ind = new int[n];
        foreach (var u in verts) foreach (var v in adj[u]) if (vs.Contains(v)) ind[v]++;
        
        var q = new Queue<int>();
        foreach (var v in verts) if (ind[v] == 0) q.Enqueue(v);
        
        int cnt = 0;
        while (q.Count > 0) { var u = q.Dequeue(); cnt++; foreach (var v in adj[u]) if (vs.Contains(v) && --ind[v] == 0) q.Enqueue(v); }
        return cnt == verts.Count;
    }
    
    static List<List<int>> GetSCCs(List<int> verts)
    {
        var vs = new HashSet<int>(verts);
        var vis = new bool[n];
        var ord = new List<int>();
        
        void dfs1(int u) { vis[u] = true; foreach (var v in adj[u]) if (vs.Contains(v) && !vis[v]) dfs1(v); ord.Add(u); }
        foreach (var v in verts) if (!vis[v]) dfs1(v);
        
        var comp = new int[n];
        Array.Fill(comp, -1);
        var sccs = new List<List<int>>();
        
        void dfs2(int u, int c) { comp[u] = c; sccs[c].Add(u); foreach (var v in radj[u]) if (vs.Contains(v) && comp[v] == -1) dfs2(v, c); }
        for (int i = ord.Count - 1; i >= 0; i--) { int v = ord[i]; if (comp[v] == -1) { sccs.Add(new List<int>()); dfs2(v, sccs.Count - 1); } }
        return sccs;
    }
    
    static List<int> SolveComponent(List<int> comp)
    {
        if (comp.Count <= 1) return new List<int>();
        if (IsDAG(comp)) return new List<int>();
        
        // Use exact algorithm for moderate-sized components
        if (comp.Count <= 80) return ExactSolve(comp);
        
        // For larger components, use iterative compression
        return IterativeCompression(comp);
    }
    
    static List<int> ExactSolve(List<int> verts)
    {
        for (int k = 0; k <= verts.Count; k++)
        {
            var r = Branch(verts, k, new bool[n], new List<int>());
            if (r != null) return r;
        }
        return verts;
    }
    
    static List<int> Branch(List<int> verts, int k, bool[] rem, List<int> fvs)
    {
        if (fvs.Count > k) return null;
        
        var remaining = verts.Where(v => !rem[v]).ToList();
        
        // Apply reduction
        var vs = new HashSet<int>(remaining);
        bool reduced = true;
        while (reduced)
        {
            reduced = false;
            var toRemove = new List<int>();
            foreach (var v in vs)
            {
                int od = 0, id = 0;
                foreach (var u in adj[v]) if (vs.Contains(u)) od++;
                foreach (var u in radj[v]) if (vs.Contains(u)) id++;
                if (od == 0 || id == 0) toRemove.Add(v);
            }
            foreach (var v in toRemove) { vs.Remove(v); reduced = true; }
        }
        remaining = vs.ToList();
        
        if (remaining.Count == 0 || IsDAG(remaining)) return new List<int>(fvs);
        if (fvs.Count == k) return null;
        
        var cycle = FindCycle(remaining);
        if (cycle == null) return new List<int>(fvs);
        
        cycle = cycle.OrderByDescending(v => adj[v].Count + radj[v].Count).ToList();
        
        foreach (var v in cycle)
        {
            rem[v] = true;
            fvs.Add(v);
            var r = Branch(verts, k, rem, fvs);
            if (r != null) return r;
            fvs.RemoveAt(fvs.Count - 1);
            rem[v] = false;
        }
        
        return null;
    }
    
    static List<int> FindCycle(List<int> verts)
    {
        if (verts.Count == 0) return null;
        var vs = new HashSet<int>(verts);
        var state = new int[n];
        var parent = new int[n];
        Array.Fill(parent, -1);
        
        foreach (var start in verts)
        {
            if (state[start] != 0) continue;
            var stack = new Stack<int>();
            stack.Push(start);
            
            while (stack.Count > 0)
            {
                var u = stack.Peek();
                
                if (state[u] == 0)
                {
                    state[u] = 1;
                    foreach (var v in adj[u])
                    {
                        if (!vs.Contains(v)) continue;
                        if (state[v] == 1)
                        {
                            var cycle = new List<int> { v };
                            var c = u;
                            while (c != v) { cycle.Add(c); c = parent[c]; if (c == -1) { cycle = null; break; } }
                            if (cycle != null) return cycle;
                        }
                        else if (state[v] == 0) { parent[v] = u; stack.Push(v); }
                    }
                }
                else if (state[u] == 1) { state[u] = 2; stack.Pop(); }
                else stack.Pop();
            }
        }
        return null;
    }
    
    static List<int> IterativeCompression(List<int> verts)
    {
        if (verts.Count == 0) return new List<int>();
        
        var order = verts.OrderByDescending(v => adj[v].Count + radj[v].Count).ToList();
        var fvsSet = GreedyFVS(verts);
        
        for (int i = 0; i < order.Count; i++)
        {
            var Wi = order.Take(i + 1).ToHashSet();
            var currentWi = fvsSet.Intersect(Wi).ToList();
            
            for (int newSize = 0; newSize < currentWi.Count; newSize++)
            {
                var better = TryCompress(Wi.ToList(), currentWi, newSize);
                if (better != null)
                {
                    foreach (var v in currentWi) fvsSet.Remove(v);
                    foreach (var v in better) fvsSet.Add(v);
                    break;
                }
            }
        }
        
        for (int pass = 0; pass < 20; pass++)
        {
            var current = fvsSet.ToList();
            if (current.Count == 0) break;
            var better = TryCompress(verts, current, current.Count - 1);
            if (better != null) fvsSet = new HashSet<int>(better);
            else break;
        }
        
        return fvsSet.ToList();
    }
    
    static List<int> TryCompress(List<int> verts, List<int> currentFvs, int targetK)
    {
        var rem = new bool[n];
        foreach (var v in currentFvs) rem[v] = true;
        return BranchCompress(verts, targetK, rem, new List<int>());
    }
    
    static List<int> BranchCompress(List<int> verts, int k, bool[] rem, List<int> fvs)
    {
        if (fvs.Count > k) return null;
        
        var remaining = verts.Where(v => !rem[v]).ToList();
        if (IsDAG(remaining)) return new List<int>(fvs);
        if (fvs.Count == k) return null;
        
        var cycle = FindCycle(remaining);
        if (cycle == null) return new List<int>(fvs);
        
        cycle = cycle.OrderByDescending(v => adj[v].Count + radj[v].Count).ToList();
        
        foreach (var v in cycle)
        {
            rem[v] = true;
            fvs.Add(v);
            var r = BranchCompress(verts, k, rem, fvs);
            if (r != null) return r;
            fvs.RemoveAt(fvs.Count - 1);
            rem[v] = false;
        }
        return null;
    }
    
    static HashSet<int> GreedyFVS(List<int> verts)
    {
        var remaining = new HashSet<int>(verts);
        var fvs = new HashSet<int>();
        
        while (!IsDAG(remaining.ToList()))
        {
            int best = -1, bestDeg = -1;
            foreach (var v in remaining)
            {
                int d = adj[v].Count(u => remaining.Contains(u)) + radj[v].Count(u => remaining.Contains(u));
                if (d > bestDeg) { bestDeg = d; best = v; }
            }
            if (best == -1) break;
            fvs.Add(best);
            remaining.Remove(best);
        }
        
        bool improved = true;
        while (improved)
        {
            improved = false;
            foreach (var v in fvs.ToList())
            {
                remaining.Add(v);
                if (IsDAG(remaining.ToList())) { fvs.Remove(v); improved = true; }
                else remaining.Remove(v);
            }
        }
        
        return fvs;
    }
}
```

**Key algorithmic features:**

1. **Preprocessing**: Self-loops are mandatory in any FVS; sources and sinks are iteratively removed (they can't be in cycles)

2. **SCC Decomposition**: Each strongly connected component is solved independently

3. **Exact Branch-and-Bound**: For components up to 80 vertices, uses cycle-based branching with degree ordering for pruning

4. **Iterative Compression**: For larger components, implements the FPT algorithm that iteratively improves a greedy solution

5. **Greedy + Local Search**: Provides the initial solution for IC, with local search refinement
