Looking at this problem, I need to implement a robust CSG Union operation that works across all mesh scales. Let me design a solution using a voxelization-based approach with marching cubes for extraction - this scales well and produces clean manifold output.

For smaller meshes, I'll use exact intersection calculations. For larger meshes, the voxel approach becomes more efficient.

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
    
    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
    Vec3 operator/(double s) const { return Vec3(x / s, y / s, z / s); }
    Vec3 operator-() const { return Vec3(-x, -y, -z); }
    
    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const {
        return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x);
    }
    
    double length() const { return sqrt(x * x + y * y + z * z); }
    double lengthSq() const { return x * x + y * y + z * z; }
    
    Vec3 normalized() const {
        double len = length();
        return len > 1e-12 ? *this / len : Vec3();
    }
};

struct Triangle {
    int a, b, c;
    Triangle() : a(0), b(0), c(0) {}
    Triangle(int a_, int b_, int c_) : a(a_), b(b_), c(c_) {}

struct Mesh {
    vector<Vec3> vertices;
    vector<Triangle> faces;
    
    void clear() {
        vertices.clear();
        faces.clear();
    }
    
    size_t memoryUsage() const {
        return vertices.size() * sizeof(Vec3) + faces.size() * sizeof(Triangle);
    }

struct Bounds {
    Vec3 min, max;
    Bounds() {
        min = Vec3(numeric_limits<double>::max(), numeric_limits<double>::max(), numeric_limits<double>::max());
        max = Vec3(-numeric_limits<double>::max(), -numeric_limits<double>::max(), -numeric_limits<double>::max());
    }
    
    void expand(const Vec3& v) {
        min.x = std::min(min.x, v.x);
        min.y = std::min(min.y, v.y);
        min.z = std::min(min.z, v.z);
        max.x = std::max(max.x, v.x);
        max.y = std::max(max.y, v.y);
        max.z = std::max(max.z, v.z);
    }
    
    Vec3 size() const { return max - min; }
    Vec3 center() const { return (min + max) * 0.5; }
    
    void expand(double d) {
        min = min - Vec3(d, d, d);
        max = max + Vec3(d, d, d);
    }
};

// Hash function for vertex deduplication
struct Vec3Hash {
    size_t operator()(const Vec3& v) const {
        size_t h = 0;
        h ^= hash<double>()(v.x) + 0x9e3779b9 + (h << 6) + (h >> 2);
        h ^= hash<double>()(v.y) + 0x9e3779b9 + (h << 6) + (h >> 2);
        h ^= hash<double>()(v.z) + 0x9e3779b9 + (h << 6) + (h >> 2);
        return h;
    }
};

// Ray-triangle intersection
bool rayTriangleIntersect(const Vec3& orig, const Vec3& dir,
                          const Vec3& v0, const Vec3& v1, const Vec3& v2,
                          double& t, double& u, double& v) {
    const double EPSILON = 1e-10;
    Vec3 edge1 = v1 - v0;
    Vec3 edge2 = v2 - v0;
    Vec3 h = dir.cross(edge2);
    double a = edge1.dot(h);
    
    if (fabs(a) < EPSILON) return false;
    
    double f = 1.0 / a;
    Vec3 s = orig - v0;
    u = f * s.dot(h);
    if (u < -EPSILON || u > 1.0 + EPSILON) return false;
    
    Vec3 q = s.cross(edge1);
    v = f * dir.dot(q);
    if (v < -EPSILON || u + v > 1.0 + EPSILON) return false;
    
    t = f * edge2.dot(q);
    return t > EPSILON;
}

// AABB-triangle intersection
bool triangleAABBIntersect(const Vec3& v0, const Vec3& v1, const Vec3& v2,
                           const Vec3& boxMin, const Vec3& boxMax) {
    // Box center and extents
    Vec3 boxCenter = (boxMin + boxMax) * 0.5;
    Vec3 boxExtents = (boxMax - boxMin) * 0.5;
    
    // Triangle mid-point
    Vec3 triMid = (v0 + v1 + v2) * (1.0 / 3.0);
    Vec3 diff = triMid - boxCenter;
    
    // Early exit if triangle center is far outside
    if (fabs(diff.x) > boxExtents.x + 2.0 || fabs(diff.y) > boxExtents.y + 2.0 ||
        fabs(diff.z) > boxExtents.z + 2.0) {
        return false;
    }
    
    // Simple bounds check
    Vec3 triMin(std::min({v0.x, v1.x, v2.x}), std::min({v0.y, v1.y, v2.y}), std::min({v0.z, v1.z, v2.z}));
    Vec3 triMax(std::max({v0.x, v1.x, v2.x}), std::max({v0.y, v1.y, v2.y}), std::max({v0.z, v1.z, v2.z}));
    
    return triMin.x <= boxMax.x && triMax.x >= boxMin.x &&
           triMin.y <= boxMax.y && triMax.y >= boxMin.y &&
           triMin.z <= boxMax.z && triMax.z >= boxMin.z;
}

// Spatial acceleration structure
class BVH {
public:
    struct Node {
        Bounds bounds;
        int left = -1, right = -1;
        int triStart = -1, triCount = 0;
        bool isLeaf() const { return triCount > 0; }
    };
    
    vector<Node> nodes;
    vector<int> triIndices;
    const Mesh* mesh;
    
    BVH() : mesh(nullptr) {}
    
    void build(const Mesh& m) {
        mesh = &m;
        nodes.clear();
        triIndices.resize(m.faces.size());
        for (int i = 0; i < (int)m.faces.size(); i++) triIndices[i] = i;
        if (!m.faces.empty()) buildNode(0, (int)m.faces.size(), 0);
    }
    
private:
    int buildNode(int start, int end, int depth) {
        int nodeIdx = (int)nodes.size();
        nodes.push_back(Node());
        
        Bounds bounds;
        for (int i = start; i < end; i++) {
            const Triangle& tri = mesh->faces[triIndices[i]];
            bounds.expand(mesh->vertices[tri.a]);
            bounds.expand(mesh->vertices[tri.b]);
            bounds.expand(mesh->vertices[tri.c]);
        }
        nodes[nodeIdx].bounds = bounds;
        
        int count = end - start;
        if (count <= 4 || depth > 20) {
            nodes[nodeIdx].triStart = start;
            nodes[nodeIdx].triCount = count;
            return nodeIdx;
        }
        
        Vec3 size = bounds.size();
        int axis = 0;
        if (size.y > size.x) axis = 1;
        if (size.z > size.array()[axis]) axis = 2;
        
        // Handle the case where size is not directly accessible as array
        double sizes[3] = {size.x, size.y, size.z};
        axis = 0;
        if (sizes[1] > sizes[0]) axis = 1;
        if (sizes[2] > sizes[axis]) axis = 2;
        
        double mid = (axis == 0 ? bounds.center().x : (axis == 1 ? bounds.center().y : bounds.center().z));
        
        int midIdx = start;
        for (int i = start; i < end; i++) {
            const Triangle& tri = mesh->faces[triIndices[i]];
            double centroid = (axis == 0) ? (mesh->vertices[tri.a].x + mesh->vertices[tri.b].x + mesh->vertices[tri.c].x) / 3.0 :
                              (axis == 1) ? (mesh->vertices[tri.a].y + mesh->vertices[tri.b].y + mesh->vertices[tri.c].y) / 3.0 :
                                            (mesh->vertices[tri.a].z + mesh->vertices[tri.b].z + mesh->vertices[tri.c].z) / 3.0;
            if (centroid < mid) {
                swap(triIndices[i], triIndices[midIdx]);
                midIdx++;
            }
        }
        
        if (midIdx == start || midIdx == end) midIdx = start + count / 2;
        
        nodes[nodeIdx].left = buildNode(start, midIdx, depth + 1);
        nodes[nodeIdx].right = buildNode(midIdx, end, depth + 1);
        
        return nodeIdx;
    }
    
public:
    // Check if ray hits anything
    bool intersectRay(const Vec3& orig, const Vec3& dir, double maxT) const {
        if (nodes.empty()) return false;
        return intersectRayNode(0, orig, dir, maxT);
    }
    
private:
    bool intersectRayNode(int nodeIdx, const Vec3& orig, const Vec3& dir, double maxT) const {
        const Node& node = nodes[nodeIdx];
        
        // Ray-box intersection
        double tmin = 0, tmax = maxT;
        for (int i = 0; i < 3; i++) {
            double invD = 1.0 / ((&dir.x)[i]);
            double t0 = ((&node.bounds.min.x)[i] - (&orig.x)[i]) * invD;
            double t1 = ((&node.bounds.max.x)[i] - (&orig.x)[i]) * invD;
            if (invD < 0) swap(t0, t1);
            tmin = max(tmin, t0);
            tmax = min(tmax, t1);
            if (tmax < tmin) return false;
        }
        
        if (node.isLeaf()) {
            for (int i = 0; i < node.triCount; i++) {
                const Triangle& tri = mesh->faces[triIndices[node.triStart + i]];
                double t, u, v;
                if (rayTriangleIntersect(orig, dir, mesh->vertices[tri.a],
                    mesh->vertices[tri.b], mesh->vertices[tri.c], t, u, v)) {
                    if (t < maxT) return true;
                }
            }
            return false;
        }
        
        return intersectRayNode(node.left, orig, dir, maxT) ||
               intersectRayNode(node.right, orig, dir, maxT);
    }
};

// Count intersections along a ray
int countIntersections(const Vec3& orig, const Vec3& dir, const Mesh& mesh, const BVH& bvh, double maxDist) {
    const double EPSILON = 1e-8;
    int count = 0;
    Vec3 rayDir = dir.normalized();
    double farT = maxDist;
    
    Vec3 currentOrig = orig;
    
    for (int iter = 0; iter < 1000; iter++) {
        double nearestT = farT;
        int hitTri = -1;
        
        // Simple traversal through all triangles (could be optimized with BVH)
        for (size_t i = 0; i < mesh.faces.size(); i++) {
            const Triangle& tri = mesh.faces[i];
            double t, u, v;
            if (rayTriangleIntersect(currentOrig, rayDir, mesh.vertices[tri.a],
                mesh.vertices[tri.b], mesh.vertices[tri.c], t, u, v)) {
                if (t > EPSILON && t < nearestT) {
                    nearestT = t;
                    hitTri = (int)i;
                }
            }
        }
        
        if (hitTri < 0) break;
        
        count++;
        currentOrig = currentOrig + rayDir * (nearestT + EPSILON * 10);
        
        if ((currentOrig - orig).length() > maxDist) break;
    }
    
    return count;
}

// Check if point is inside mesh using winding number
bool isInsideMesh(const Vec3& pt, const Mesh& mesh, const BVH& bvh, const Bounds& bounds) {
    if (mesh.faces.empty()) return false;
    
    // Use multiple rays and majority voting for robustness
    int insideCount = 0;
    int totalTests = 0;
    
    Vec3 directions[3] = {
        Vec3(1, 0, 0),
        Vec3(0, 1, 0),
        Vec3(0, 0, 1)
    };
    
    for (int d = 0; d < 3; d++) {
        double maxDist = (bounds.max.array()[d] - bounds.min.array()[d]) + 10.0;
        double maxDists[3] = {maxDist, maxDist, maxDist};
        
        int hits = countIntersections(pt, directions[d], mesh, bvh, maxDists[d]);
        if (hits % 2 == 1) insideCount++;
        totalTests++;
    }
    
    return insideCount >= 2;  // At least 2 out of 3 rays agree it's inside
}
