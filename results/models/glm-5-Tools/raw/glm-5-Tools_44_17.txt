```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Brick pattern parameters
    vec2 brickScale = vec2(8.0, 16.0);
    vec2 mortarThickness = vec2(0.05, 0.1);
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);
    vec3 brickColor = vec3(0.7, 0.2, 0.1);
    
    // Scale UVs
    vec2 scaledUV = uv * brickScale;
    
    // Stagger every other row
    float row = floor(scaledUV.y);
    scaledUV.x += (mod(row, 2.0) < 0.5) ? 0.0 : 0.5;
    
    // Get brick-local coordinates
    vec2 brickUV = fract(scaledUV);
    
    // Check if we're in mortar
    float inMortarX = step(brickUV.x, mortarThickness.x) + step(1.0 - mortarThickness.x, brickUV.x);
    float inMortarY = step(brickUV.y, mortarThickness.y) + step(1.0 - mortarThickness.y, brickUV.y);
    float inMortar = max(inMortarX, inMortarY);
    
    // Blend between brick and mortar
    vec3 surfaceColor = mix(brickColor, mortarColor, inMortar);
    
    // Lambertian shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    
    // Apply shading factor
    vec3 shadedColor = surfaceColor * (0.4 + 0.6 * diffuse);
    
    outColor = vec4(shadedColor, 1.0);
}
```
