declare const require: any;
const fs = require('fs');

interface Task {
    id: number;
    duration: number;
    deps: number[];
    deadline: number;
    weight: number;
}

interface ScheduleEntry {
    id: number;
    start: number;
    machine: number;
}

interface Input {
    machineCount: number;
    tasks: Task[];
}

function solve(data: Input): ScheduleEntry[] {
    const { machineCount, tasks } = data;
    const n = tasks.length;
    
    if (n === 0) return [];
    
    // Build task lookup
    const taskMap = new Map<number, Task>();
    for (const t of tasks) taskMap.set(t.id, t);
    
    // Build reverse dependency graph and compute in-degree
    const dependents = new Map<number, number[]>();
    const inDegree = new Map<number, number>();
    
    for (const t of tasks) {
        inDegree.set(t.id, t.deps.length);
        for (const d of t.deps) {
            if (!dependents.has(d)) dependents.set(d, []);
            dependents.get(d)!.push(t.id);
        }
    }
    
    // Compute earliest start times via topological sort
    const earliestStart = new Map<number, number>();
    const tempDegree = new Map(inDegree);
    const queue: number[] = [];
    
    for (const t of tasks) {
        if (t.deps.length === 0) {
            queue.push(t.id);
            earliestStart.set(t.id, 0);
        }
    }
    
    let qIdx = 0;
    while (qIdx < queue.length) {
        const curr = queue[qIdx++];
        const currTask = taskMap.get(curr)!;
        const currEst = earliestStart.get(curr)!;
        const currFinish = currEst + currTask.duration;
        
        for (const next of (dependents.get(curr) || [])) {
            const newDeg = tempDegree.get(next)! - 1;
            tempDegree.set(next, newDeg);
            
            const newEst = Math.max(earliestStart.get(next) || 0, currFinish);
            earliestStart.set(next, newEst);
            
            if (newDeg === 0) {
                queue.push(next);
            }
        }
    }
    
    // Compute critical path length (longest path to end)
    const criticalPath = new Map<number, number>();
    const reverseOrder = queue.slice().reverse();
    
    for (const id of reverseOrder) {
        const task = taskMap.get(id)!;
        let maxPath = task.duration;
        for (const next of (dependents.get(id) || [])) {
            maxPath = Math.max(maxPath, task.duration + (criticalPath.get(next) || 0));
        }
        criticalPath.set(id, maxPath);
    }
    
    // Priority function - higher is better
    const getPriority = (id: number): number => {
        const t = taskMap.get(id)!;
        const w = t.weight;
        const d = Math.max(1, t.duration);
        const dl = t.deadline;
        const est = earliestStart.get(id) || 0;
        const cp = criticalPath.get(id) || 0;
        
        // Weight per unit time - higher weight/shorter duration = higher priority
        const wpt = w / d;
        
        // Deadline urgency
        let deadlineScore = 0;
        if (dl > 0) {
            const slack = dl - est;
            if (slack < d * 2) {
                deadlineScore = Math.max(0, 1000 - slack);
            }
        }
        
        // Critical path - longer critical path means more blocking
        const cpScore = cp / 10;
        
        return wpt * 1000 + deadlineScore + cpScore;
    };
    
    // Ready queue
    const ready: number[] = [];
    const remainingDeps = new Map(inDegree);
    
    // Initialize ready queue
    for (const t of tasks) {
        if (t.deps.length === 0) {
            ready.push(t.id);
        }
    }
    
    // Sort ready queue by priority
    const sortReady = () => {
        ready.sort((a, b) => getPriority(b) - getPriority(a));
    };
    sortReady();
    
    // Machine availability
    const machineFree = new Array(machineCount).fill(0);
    const finishTime = new Map<number, number>();
    
    const schedule: ScheduleEntry[] = [];
    
    while (ready.length > 0) {
        const taskId = ready.shift()!;
        const task = taskMap.get(taskId)!;
        
        // Compute actual start time
        let start = 0;
        for (const d of task.deps) {
            start = Math.max(start, finishTime.get(d) || 0);
        }
        
        // Find best machine (earliest available)
        let bestMachine = 0;
        let bestStart = Math.max(machineFree[0], start);
        
        for (let m = 1; m < machineCount; m++) {
            const cand = Math.max(machineFree[m], start);
            if (cand < bestStart) {
                bestStart = cand;
                bestMachine = m;
            }
        }
        
        const end = bestStart + task.duration;
        
        schedule.push({
            id: taskId,
            start: bestStart,
            machine: bestMachine
        });
        
        machineFree[bestMachine] = end;
        finishTime.set(taskId, end);
        
        // Update ready tasks
        const newReady: number[] = [];
        for (const next of (dependents.get(taskId) || [])) {
            const newDeg = remainingDeps.get(next)! - 1;
            remainingDeps.set(next, newDeg);
            if (newDeg === 0) {
                newReady.push(next);
            }
        }
        
        // Insert new ready tasks maintaining sorted order
        if (newReady.length > 0) {
            for (const id of newReady) {
                ready.push(id);
            }
            sortReady();
        }
    }
    
    return schedule;
}
