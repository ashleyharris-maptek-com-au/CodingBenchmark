Here's the complete SPIR-V assembly for the geometry shader that explodes each triangle outward along its face normal:

```spirv
; Generator: Khronos SPIR-V Tools Assembler

OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block

OpDecorate %gl_PerVertex_out Block

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%void_fn = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_3

%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%outColor = OpVariable %ptr_out_v3 Output

%1 = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %void_fn
%entry = OpLabel

; Access chains for input positions (gl_in[i].gl_Position)
%gl_in_0_pos = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%gl_in_1_pos = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%gl_in_2_pos = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0

; Access chains for per-vertex colors
%color_0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%color_1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%color_2_ptr = OpAccessChain %ptr_in_v3 %color %int_2

; Access chain for output position (gl_out.gl_Position)
%gl_out_0_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Load gl_Position for each of the 3 input vertices
%p0_v4 = OpLoad %v4float %gl_in_0_pos
%p1_v4 = OpLoad %v4float %gl_in_1_pos
%p2_v4 = OpLoad %v4float %gl_in_2_pos

; Extract xyz components from each position for geometric calculations
%p0_x = OpCompositeExtract %float %p0_v4 0
%p0_y = OpCompositeExtract %float %p0_v4 1
%p0_z = OpCompositeExtract %float %p0_v4 2
%pos0 = OpCompositeConstruct %v3float %p0_x %p0_y %p0_z

%p1_x = OpCompositeExtract %float %p1_v4 0
%p1_y = OpCompositeExtract %float %p1_v4 1
%p1_z = OpCompositeExtract %float %p1_v4 2
%pos1 = OpCompositeConstruct %v3float %p1_x %p1_y %p1_z

%p2_x = OpCompositeExtract %float %p2_v4 0
%p2_y = OpCompositeExtract %float %p2_v4 1
%p2_z = OpCompositeExtract %float %p2_v4 2
%pos2 = OpCompositeConstruct %v3float %p2_x %p2_y %p2_z

; Compute edges for face normal: e1 = p1 - p0, e2 = p2 - p0
%e1 = OpFSub %v3float %pos1 %pos0
%e2 = OpFSub %v3float %pos2 %pos0

; Compute cross product: cross(e1, e2) = (e1.y*e2.z - e1.z*e2.y, e1.z*e2.x - e1.x*e2.z, e1.x*e2.y - e1.y*e2.x)
%e1_x = OpCompositeExtract %float %e1 0
%e1_y = OpCompositeExtract %float %e1 1
%e1_z = OpCompositeExtract %float %e1 2
%e2_x = OpCompositeExtract %float %e2 0
%e2_y = OpCompositeExtract %float %e2 1
%e2_z = OpCompositeExtract %float %e2 2

%cross_x_0 = OpFMul %float %e1_y %e2_z
%cross_x_1 = OpFMul %float %e1_z %e2_y
%cross_x = OpFSub %float %cross_x_0 %cross_x_1

%cross_y_0 = OpFMul %float %e1_z %e2_x
%cross_y_1 = OpFMul %float %e1_x %e2_z
%cross_y = OpFSub %float %cross_y_0 %cross_y_1

%cross_z_0 = OpFMul %float %e1_x %e2_y
%cross_z_1 = OpFMul %float %e1_y %e2_x
%cross_z = OpFSub %float %cross_z_0 %cross_z_1

%cross_vec = OpCompositeConstruct %v3float %cross_x %cross_y %cross_z

; Normalize the face normal using GLSL.std.450
%face_normal = OpExtInst %v3float %1 Normalize %cross_vec

; Compute offset vector: offset = face_normal * 0.08
%offset = OpVectorTimesScalar %v3float %face_normal %float_0_08

; === Emit vertex 0 (shifted by offset) ===
%color0 = OpLoad %v3float %color_0_ptr
%new_pos0 = OpFAdd %v3float %pos0 %offset
%new_p0_x = OpCompositeExtract %float %new_pos0 0
%new_p0_y = OpCompositeExtract %float %new_pos0 1
%new_p0_z = OpCompositeExtract %float %new_pos0 2
%new_p0_w = OpCompositeExtract %float %p0_v4 3
%new_p0_v4 = OpCompositeConstruct %v4float %new_p0_x %new_p0_y %new_p0_z %new_p0_w
OpStore %gl_out_0_pos %new_p0_v4
OpStore %outColor %color0
OpEmitVertex

; === Emit vertex 1 (shifted by offset) ===
%color1 = OpLoad %v3float %color_1_ptr
%new_pos1 = OpFAdd %v3float %pos1 %offset
%new_p1_x = OpCompositeExtract %float %new_pos1 0
%new_p1_y = OpCompositeExtract %float %new_pos1 1
%new_p1_z = OpCompositeExtract %float %new_pos1 2
%new_p1_w = OpCompositeExtract %float %p1_v4 3
%new_p1_v4 = OpCompositeConstruct %v4float %new_p1_x %new_p1_y %new_p1_z %new_p1_w
OpStore %gl_out_0_pos %new_p1_v4
OpStore %outColor %color1
OpEmitVertex

; === Emit vertex 2 (shifted by offset) ===
%color2 = OpLoad %v3float %color_2_ptr
%new_pos2 = OpFAdd %v3float %pos2 %offset
%new_p2_x = OpCompositeExtract %float %new_pos2 0
%new_p2_y = OpCompositeExtract %float %new_pos2 1
%new_p2_z = OpCompositeExtract %float %new_pos2 2
%new_p2_w = OpCompositeExtract %float %p2_v4 3
%new_p2_v4 = OpCompositeConstruct %v4float %new_p2_x %new_p2_y %new_p2_z %new_p2_w
OpStore %gl_out_0_pos %new_p2_v4
OpStore %outColor %color2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```

**Key implementation details:**

1. **Input/Output Setup**: Properly declares `gl_PerVertex` blocks for input (array of 3 for triangles) and output (single struct). User-defined inputs (`worldPos`, `normal`, `color`) are arrays of size 3.

2. **Face Normal Computation**: 
   - Computes two edge vectors from the triangle vertices
   - Manually calculates the cross product: `cross(e1, e2) = (e1.y*e2.z - e1.z*e2.y, e1.z*e2.x - e1.x*e2.z, e1.x*e2.y - e1.y*e2.x)`
   - Normalizes using `GLSL.std.450` Normalize instruction

3. **Explosion Effect**: Each vertex position is offset by `face_normal * 0.08` in the direction the triangle is facing.

4. **Vertex Emission**: Each of the 3 vertices is emitted with its original color but shifted position, preserving the original `w` component for proper perspective interpolation.
