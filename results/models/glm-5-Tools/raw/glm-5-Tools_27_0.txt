#include <algorithm>

using namespace std;

int n, m, k;
vector<vector<int>> adj;
vector<int> color, degree;
vector<uint64_t> nc;  // neighbor colors bitmask
mt19937_64 rng(chrono::steady_clock::now().time_since_epoch().count());

auto t0 = chrono::steady_clock::now();
double el() { return chrono::duration<double>(chrono::steady_clock::now() - t0).count(); }

#ifdef _MSC_VER
#include <intrin.h>
#define PC(x) (int)__popcnt64(x)
#else
#define PC(x) __builtin_popcountll(x)
#endif

int gcol(uint64_t mask) {
    for (int c = 0; c < k; c++) 
        if (!(mask & (1ULL << c))) return c;
    return -1;
}

bool greedy(const vector<int>& ord) {
    for (int u : ord) {
        int c = gcol(nc[u]);
        if (c < 0) return false;
        color[u] = c;
        uint64_t b = 1ULL << c;
        for (int v : adj[u]) nc[v] |= b;
    }
    return true;
}

// DSATUR using set - O((n+m) log n)
bool dsatur() {
    vector<int> sat(n, 0);
    vector<bool> done(n, false);
    set<tuple<int,int,int>> pq;
    
    for (int u = 0; u < n; u++) pq.insert({0, -degree[u], u});
    
    while (!pq.empty()) {
        auto it = prev(pq.end());
        auto [s, d, u] = *it;
        pq.erase(it);
        if (done[u]) continue;
        
        int c = gcol(nc[u]);
        if (c < 0) return false;
        
        color[u] = c;
        done[u] = true;
        uint64_t b = 1ULL << c;
        
        for (int v : adj[u]) {
            if (!done[v] && !(nc[v] & b)) {
                pq.erase({sat[v], -degree[v], v});
                sat[v]++;
                nc[v] |= b;
                pq.insert({sat[v], -degree[v], v});
            }
        }
    }
    return true;
}

int bt_cnt = 0, bt_max = 3000000;
bool bt() {
    int b = -1, bs = -1, bd = -1;
    for (int u = 0; u < n; u++) {
        if (color[u] >= 0) continue;
        int s = PC(nc[u]);
        if (s > bs || (s == bs && degree[u] > bd)) { b = u; bs = s; bd = degree[u]; }
    }
    if (b < 0) return true;
    if (++bt_cnt > bt_max || el() > 40.0) return false;
    
    uint64_t f = nc[b];
    for (int c = 0; c < k; c++) {
        if (f & (1ULL << c)) continue;
        color[b] = c;
        uint64_t bit = 1ULL << c;
        vector<int> ch;
        for (int v : adj[b]) if (color[v] < 0 && !(nc[v] & bit)) { nc[v] |= bit; ch.push_back(v); }
        if (bt()) return true;
        for (int v : ch) nc[v] &= ~bit;
        color[b] = -1;
    }
    return false;
}

bool rgreedy(int t) {
    vector<int> ord(n);
    for (int i = 0; i < n; i++) ord[i] = i;
    for (int i = 0; i < t && el() < 80.0; i++) {
        fill(color.begin(), color.end(), -1);
        fill(nc.begin(), nc.end(), 0);
        shuffle(ord.begin(), ord.end(), rng);
        if (greedy(ord)) return true;
    }
    return false;
}

bool dgreedy() {
    vector<int> ord(n);
    for (int i = 0; i < n; i++) ord[i] = i;
    sort(ord.begin(), ord.end(), [](int a, int b) { return degree[a] > degree[b]; });
    return greedy(ord);
}

bool repair(int mx) {
    vector<int> cf(n, 0);
    int tot = 0;
    for (int u = 0; u < n; u++) for (int v : adj[u]) if (color[u] == color[v]) { cf[u]++; tot++; }
    tot /= 2;
    
    for (int it = 0; it < mx && tot > 0 && el() < 240.0; it++) {
        int u = -1;
        for (int i = 0; i < n; i++) if (cf[i] > 0) { u = i; break; }
        if (u < 0) break;
        
        int bc = color[u], bi = 0;
        for (int c = 0; c < k; c++) {
            if (c == color[u]) continue;
            int cnt = 0;
            for (int v : adj[u]) if (color[v] == c) cnt++;
            int imp = cf[u] - cnt;
            if (imp > bi) { bi = imp; bc = c; }
        }
        
        if (bc != color[u]) {
            int old = color[u];
            for (int v : adj[u]) {
                if (color[v] == old) { cf[v]--; tot--; }
                if (color[v] == bc) { cf[v]++; tot++; }
            }
            color[u] = bc;
            cf[u] = 0;
            for (int v : adj[u]) if (color[v] == bc) cf[u]++;
        } else {
            color[u] = rng() % k;
            cf[u] = 0;
            for (int v : adj[u]) if (color[v] == color[u]) cf[u]++;
        }
    }
    return tot == 0;
}
