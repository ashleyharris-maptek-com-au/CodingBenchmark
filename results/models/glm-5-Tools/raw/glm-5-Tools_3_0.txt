use std::io::{self, BufRead, BufWriter, Write};
use std::collections::{HashMap, HashSet, VecDeque};

/// Graph structure
struct Graph {
    n: usize,
    edges: Vec<(usize, usize)>,
    adj: Vec<Vec<usize>>,
}

impl Graph {
    fn new(n: usize) -> Self {
        Graph {
            n,
            edges: Vec::new(),
            adj: vec![Vec::new(); n],
        }
    }

    fn add_edge(&mut self, u: usize, v: usize) {
        if u < self.n && v < self.n && u != v {
            self.edges.push((u, v));
            self.adj[u].push(v);
            self.adj[v].push(u);
        }
    }

    fn density(&self) -> f64 {
        if self.n < 2 {
            return 0.0;
        }
        let max_edges = self.n * (self.n - 1) / 2;
        self.edges.len() as f64 / max_edges as f64
    }

    fn find_components(&self) -> Vec<Vec<usize>> {
        let mut visited = vec![false; self.n];
        let mut components = Vec::new();

        for start in 0..self.n {
            if visited[start] {
                continue;
            }
            let mut comp = Vec::new();
            let mut queue = VecDeque::new();
            queue.push_back(start);
            visited[start] = true;

            while let Some(u) = queue.pop_front() {
                comp.push(u);
                for &v in &self.adj[u] {
                    if !visited[v] {
                        visited[v] = true;
                        queue.push_back(v);
                    }
                }
            }
            components.push(comp);
        }
        components
    }

    fn avg_degree(&self) -> f64 {
        if self.n == 0 {
            0.0
        } else {
            (2 * self.edges.len()) as f64 / self.n as f64
        }
    }
}

/// 2D point
#[derive(Clone, Copy, Default)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn new(x: f64, y: f64) -> Self {
        Point { x, y }
    }

    fn distance(&self, other: &Point) -> f64 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        (dx * dx + dy * dy).sqrt().max(1e-10)
    }

    fn magnitude(&self) -> f64 {
        (self.x * self.x + self.y * self.y).sqrt().max(1e-10)
    }

    fn normalize(&mut self) {
        let m = self.magnitude();
        self.x /= m;
        self.y /= m;
    }
}

impl std::ops::Add for Point {
    type Output = Point;
    fn add(self, other: Point) -> Point {
        Point::new(self.x + other.x, self.y + other.y)
    }
}

impl std::ops::Sub for Point {
    type Output = Point;
    fn sub(self, other: Point) -> Point {
        Point::new(self.x - other.x, self.y - other.y)
    }
}

impl std::ops::Mul<f64> for Point {
    type Output = Point;
    fn mul(self, scalar: f64) -> Point {
        Point::new(self.x * scalar, self.y * scalar)
    }
}

/// Grid cell for spatial hashing
#[derive(Hash, Eq, PartialEq, Clone, Copy)]
struct GridCell(i64, i64);

/// Spatial hash grid for O(n) force calculation
struct SpatialGrid {
    cell_size: f64,
    cells: HashMap<GridCell, Vec<usize>>,
    min_x: f64,
    min_y: f64,
}

impl SpatialGrid {
    fn new(cell_size: f64) -> Self {
        SpatialGrid {
            cell_size,
            cells: HashMap::new(),
            min_x: 0.0,
            min_y: 0.0,
        }
    }

    fn build(&mut self, positions: &[Point]) {
        self.cells.clear();
        
        // Find bounds
        let mut min_x = f64::MAX;
        let mut min_y = f64::MAX;
        for p in positions {
            min_x = min_x.min(p.x);
            min_y = min_y.min(p.y);
        }
        self.min_x = min_x;
        self.min_y = min_y;

        // Populate cells
        for (i, p) in positions.iter().enumerate() {
            let cell = self.get_cell(p);
            self.cells.entry(cell).or_insert_with(Vec::new).push(i);
        }
    }

    fn get_cell(&self, p: &Point) -> GridCell {
        let cx = ((p.x - self.min_x) / self.cell_size).floor() as i64;
        let cy = ((p.y - self.min_y) / self.cell_size).floor() as i64;
        GridCell(cx, cy)
    }

    fn get_nearby(&self, p: &Point, radius: f64) -> Vec<usize> {
        let mut result = Vec::new();
        let cell_radius = (radius / self.cell_size).ceil() as i64;
        let center = self.get_cell(p);

        for dx in -cell_radius..=cell_radius {
            for dy in -cell_radius..=cell_radius {
                let cell = GridCell(center.0 + dx, center.1 + dy);
                if let Some(nodes) = self.cells.get(&cell) {
                    result.extend(nodes.iter().cloned());
                }
            }
        }
        result
    }
}

/// Force-directed layout parameters
struct LayoutParams {
    iterations: usize,
    temperature: f64,
    cooling: f64,
    repulsion: f64,
    attraction: f64,
    optimal_dist: f64,
}

impl LayoutParams {
    fn for_graph(n: usize, m: usize) -> Self {
        let area = (n as f64).sqrt() * 100.0;
        let density = if n > 1 { m as f64 / (n * (n - 1) / 2) as f64 } else { 0.0 };
        let avg_deg = if n > 0 { 2.0 * m as f64 / n as f64 } else { 0.0 };
        
        // Adaptive parameters based on graph size
        let (iterations, temperature, cooling) = if n < 100 {
            (200, area * 0.3, 0.95)
        } else if n < 500 {
            (100, area * 0.2, 0.92)
        } else if n < 2000 {
            (50, area * 0.15, 0.90)
        } else {
            (30, area * 0.1, 0.88)
        };

        let optimal_dist = if avg_deg > 0.0 {
            (area / avg_deg).sqrt() * 2.0
        } else {
            area.sqrt()
        };

        let repulsion = optimal_dist * optimal_dist;
        let attraction = 1.0 / optimal_dist;

        LayoutParams {
            iterations,
            temperature,
            cooling,
            repulsion,
            attraction,
            optimal_dist,
        }
    }
}

/// Initialize positions in a circle or spiral
fn initialize_positions(n: usize) -> Vec<Point> {
    let mut positions = Vec::with_capacity(n);
    
    if n == 0 {
        return positions;
    }

    // Spiral layout for better initial distribution
    let golden_angle = std::f64::consts::PI * (3.0 - (5.0_f64).sqrt());
    let radius_factor = 10.0_f64.sqrt();

    for i in 0..n {
        let theta = i as f64 * golden_angle;
        let r = (i as f64).sqrt() * radius_factor;
        positions.push(Point::new(r * theta.cos(), r * theta.sin()));
    }

    positions
}

/// Initialize positions using a spectral-like approach
fn initialize_spectral(graph: &Graph) -> Vec<Point> {
    let n = graph.n;
    if n == 0 {
        return Vec::new();
    }

    // Use degrees and BFS distances for initial placement
    let mut positions = Vec::with_capacity(n);
    
    // Find highest-degree node as reference
    let mut max_deg = 0;
    let mut hub = 0;
    for (i, adj) in graph.adj.iter().enumerate() {
        if adj.len() > max_deg {
            max_deg = adj.len();
            hub = i;
        }
    }

    // BFS from hub
    let mut dist = vec![usize::MAX; n];
    let mut queue = VecDeque::new();
    dist[hub] = 0;
    queue.push_back(hub);

    while let Some(u) = queue.pop_front() {
        for &v in &graph.adj[u] {
            if dist[v] == usize::MAX {
                dist[v] = dist[u] + 1;
                queue.push_back(v);
            }
        }
    }

    // Place nodes in concentric circles based on BFS distance
    let max_dist = dist.iter().cloned().filter(|&d| d != usize::MAX).max().unwrap_or(1).max(1);
    let mut layers: Vec<Vec<usize>> = vec![Vec::new(); max_dist + 1];
    
    for i in 0..n {
        if dist[i] != usize::MAX {
            layers[dist[i]].push(i);
        }
    }

    // Assign positions
    let mut result = vec![Point::default(); n];
    
    for (layer_idx, nodes) in layers.iter().enumerate() {
        let r = (layer_idx as f64 + 1.0) * 20.0;
        let angle_step = 2.0 * std::f64::consts::PI / (nodes.len().max(1) as f64);
        
        for (i, &node) in nodes.iter().enumerate() {
            let angle = i as f64 * angle_step;
            result[node] = Point::new(r * angle.cos(), r * angle.sin());
        }
    }

    // Handle disconnected nodes
    let mut unplaced_count = 0;
    for i in 0..n {
        if dist[i] == usize::MAX {
            let angle = unplaced_count as f64 * 0.5;
            let r = (max_dist as f64 + 2.0) * 20.0 + (unplaced_count as f64) * 5.0;
            result[i] = Point::new(r * angle.cos(), r * angle.sin());
            unplaced_count += 1;
        }
    }

    result
}

/// Fast force-directed layout with spatial hashing
fn force_directed_fast(graph: &Graph, positions: &mut Vec<Point>, params: &LayoutParams) {
    let n = graph.n;
    if n == 0 {
        return;
    }

    let mut displacements = vec![Point::default(); n];
    let mut grid = SpatialGrid::new(params.optimal_dist);
    let mut temperature = params.temperature;

    for iteration in 0..params.iterations {
        // Build spatial grid
        grid.build(positions);

        // Reset displacements
        for d in &mut displacements {
            *d = Point::default();
        }

        // Calculate repulsive forces using spatial hashing
        for i in 0..n {
            let nearby = grid.get_nearby(&positions[i], params.optimal_dist * 3.0);
            
            for &j in &nearby {
                if i >= j {
                    continue;
                }
                let delta = positions[i] - positions[j];
                let dist = delta.magnitude();
                
                if dist < 1e-10 {
                    continue;
                }

                let force = params.repulsion / (dist * dist);
                let disp = delta * (force / dist);
                displacements[i] = displacements[i] + disp;
                displacements[j] = displacements[j] - disp;
            }
        }

        // Calculate attractive forces along edges
        for &(u, v) in &graph.edges {
            let delta = positions[v] - positions[u];
            let dist = delta.magnitude();
            
            if dist < 1e-10 {
                continue;
            }

            let force = dist * params.attraction;
            let disp = delta * (force / dist);
            displacements[u] = displacements[u] + disp;
            displacements[v] = displacements[v] - disp;
        }

        // Apply displacements with temperature-based limiting
        for i in 0..n {
            let disp_mag = displacements[i].magnitude();
            if disp_mag > 1e-10 {
                let scale = temperature.min(disp_mag) / disp_mag;
                displacements[i] = displacements[i] * scale;
                positions[i] = positions[i] + displacements[i];
            }
        }

        // Cool down
        temperature *= params.cooling;

        // Early termination if temperature is very low
        if temperature < 0.01 {
            break;
        }
    }
}

/// Simple force-directed for small graphs (exact O(n²) computation)
fn force_directed_exact(graph: &Graph, positions: &mut Vec<Point>, params: &LayoutParams) {
    let n = graph.n;
    if n == 0 {
        return;
    }

    let mut displacements = vec![Point::default(); n];
    let mut temperature = params.temperature;

    for iteration in 0..params.iterations {
        // Reset displacements
        for d in &mut displacements {
            *d = Point::default();
        }

        // Repulsive forces - all pairs
        for i in 0..n {
            for j in (i + 1)..n {
                let delta = positions[i] - positions[j];
                let dist = delta.magnitude();
                
                if dist < 1e-10 {
                    // Add small random displacement
                    let angle = (i * 17 + j * 31) as f64;
                    displacements[i] = displacements[i] + Point::new(angle.cos() * 0.1, angle.sin() * 0.1);
                    continue;
                }

                let force = params.repulsion / (dist * dist);
                let disp = delta * (force / dist);
                displacements[i] = displacements[i] + disp;
                displacements[j] = displacements[j] - disp;
            }
        }

        // Attractive forces along edges
        for &(u, v) in &graph.edges {
            let delta = positions[v] - positions[u];
            let dist = delta.magnitude();
            
            if dist < 1e-10 {
                continue;
            }

            let force = dist * params.attraction;
            let disp = delta * (force / dist);
            displacements[u] = displacements[u] + disp;
            displacements[v] = displacements[v] - disp;
        }

        // Apply displacements
        for i in 0..n {
            let disp_mag = displacements[i].magnitude();
            if disp_mag > 1e-10 {
                let scale = temperature.min(disp_mag) / disp_mag;
                displacements[i] = displacements[i] * scale;
                positions[i] = positions[i] + displacements[i];
            }
        }

        temperature *= params.cooling;
    }
}

/// Grid-based layout for very large graphs
fn grid_layout(graph: &Graph, positions: &mut Vec<Point>) {
    let n = graph.n;
    if n == 0 {
        return;
    }

    // Sort nodes by degree (higher degree = more central)
    let mut nodes_by_degree: Vec<usize> = (0..n).collect();
    nodes_by_degree.sort_by(|&a, &b| graph.adj[b].len().cmp(&graph.adj[a].len()));

    // Place on grid in spiral order
    let grid_size = (n as f64).sqrt().ceil() as i64;
    let mut x = grid_size / 2;
    let mut y = grid_size / 2;
    let mut dx = 1;
    let mut dy = 0;
    let mut segment_len = 1;
    let mut segment_passed = 0;
    let mut turns = 0;

    for (idx, &node) in nodes_by_degree.iter().enumerate() {
        positions[node] = Point::new(x as f64 * 10.0, y as f64 * 10.0);

        // Spiral walk
        x += dx;
        y += dy;
        segment_passed += 1;

        if segment_passed == segment_len {
            segment_passed = 0;
            // Turn
            let new_dx = -dy;
            let new_dy = dx;
            dx = new_dx;
            dy = new_dy;
            turns += 1;
            if turns % 2 == 0 {
                segment_len += 1;
            }
        }

        // Bounds check
        if x < 0 || x >= grid_size || y < 0 || y >= grid_size {
            break;
        }
    }

    // Fill remaining with simple grid
    let mut fill_x = 0;
    let mut fill_y = 0;
    for node in 0..n {
        if positions[node].x == 0.0 && positions[node].y == 0.0 && node != nodes_by_degree.get(0).copied().unwrap_or(0) {
            positions[node] = Point::new(fill_x as f64 * 10.0, fill_y as f64 * 10.0);
            fill_x += 1;
            if fill_x >= grid_size {
                fill_x = 0;
                fill_y += 1;
            }
        }
    }
}

/// Normalize layout to reasonable coordinates
fn normalize_layout(positions: &mut Vec<Point>) {
    if positions.is_empty() {
        return;
    }

    // Find bounds
    let mut min_x = f64::MAX;
    let mut max_x = f64::MIN;
    let mut min_y = f64::MAX;
    let mut max_y = f64::MIN;

    for p in positions.iter() {
        min_x = min_x.min(p.x);
        max_x = max_x.max(p.x);
        min_y = min_y.min(p.y);
        max_y = max_y.max(p.y);
    }

    let width = (max_x - min_x).max(1.0);
    let height = (max_y - min_y).max(1.0);
    let scale = 1000.0 / width.max(height);
    let cx = (min_x + max_x) / 2.0;
    let cy = (min_y + max_y) / 2.0;

    for p in positions.iter_mut() {
        p.x = (p.x - cx) * scale + 500.0;
        p.y = (p.y - cy) * scale + 500.0;
    }
}

/// Handle disconnected components by spreading them apart
fn layout_components(graph: &Graph, positions: &mut Vec<Point>) {
    let components = graph.find_components();
    
    if components.len() <= 1 {
        return;
    }

    // Find bounding box for each component
    let mut bounds: Vec<(f64, f64, f64, f64)> = Vec::with_capacity(components.len());
    
    for comp in &components {
        if comp.is_empty() {
            bounds.push((0.0, 0.0, 0.0, 0.0));
            continue;
        }
        
        let mut min_x = f64::MAX;
        let mut max_x = f64::MIN;
        let mut min_y = f64::MAX;
        let mut max_y = f64::MIN;
        
        for &node in comp {
            let p = &positions[node];
            min_x = min_x.min(p.x);
            max_x = max_x.max(p.x);
            min_y = min_y.min(p.y);
            max_y = max_y.max(p.y);
        }
        
        bounds.push((min_x, max_x, min_y, max_y));
    }

    // Arrange components in a grid
    let num_comp = components.len();
    let grid_cols = (num_comp as f64).sqrt().ceil() as usize;
    let grid_rows = (num_comp + grid_cols - 1) / grid_cols;
    
    let mut cell_width = 0.0;
    let mut cell_height = 0.0;
    
    for (min_x, max_x, min_y, max_y) in &bounds {
        cell_width = cell_width.max(max_x - min_x);
        cell_height = cell_height.max(max_y - min_y);
    }
    
    cell_width = cell_width.max(100.0);
    cell_height = cell_height.max(100.0);
    
    // Add spacing between components
    let spacing = 50.0;
    
    for (idx, comp) in components.iter().enumerate() {
        let grid_x = idx % grid_cols;
        let grid_y = idx / grid_cols;
        
        let (min_x, _, min_y, _) = bounds[idx];
        let offset_x = (grid_x as f64) * (cell_width + spacing) - min_x;
        let offset_y = (grid_y as f64) * (cell_height + spacing) - min_y;
        
        for &node in comp {
            positions[node].x += offset_x;
            positions[node].y += offset_y;
        }
    }
}

/// Main layout function
fn compute_layout(graph: &Graph) -> Vec<Point> {
    let n = graph.n;
    let m = graph.edges.len();
    
    if n == 0 {
        return Vec::new();
    }

    // Choose algorithm based on graph size and density
    let density = graph.density();
    let avg_deg = graph.avg_degree();

    // Initialize positions
    let mut positions = if n < 1000 {
        // Use spectral-like initialization for medium/small graphs
        initialize_spectral(graph)
    } else {
        // Use spiral for large graphs
        initialize_positions(n)
    };

    // Get layout parameters
    let params = LayoutParams::for_graph(n, m);

    // Run force-directed layout
    if n < 200 {
        // Exact O(n²) for small graphs
        force_directed_exact(graph, &mut positions, &params);
    } else if n < 10000 {
        // Fast O(n log n) with spatial hashing
        force_directed_fast(graph, &mut positions, &params);
    } else {
        // Grid-based initial layout + short refinement for very large graphs
        grid_layout(graph, &mut positions);
        
        // Quick refinement
        let quick_params = LayoutParams {
            iterations: 15,
            temperature: 100.0,
            cooling: 0.9,
            repulsion: params.repulsion,
            attraction: params.attraction,
            optimal_dist: params.optimal_dist,
        };
        force_directed_fast(graph, &mut positions, &quick_params);
    }

    // Handle disconnected components
    layout_components(graph, &mut positions);

    // Normalize to reasonable coordinates
    normalize_layout(&mut positions);

    positions
}

fn main() {
    // Fast I/O
    let stdin = io::stdin();

    let mut stdout = BufWriter::new(stdout.lock());

    }
}
