use std::io::{self, BufRead, Write, BufWriter};
use std::collections::{HashSet, VecDeque};

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    
    let mut lines = stdin.lock().lines();
    
    // Parse header
    let header: Vec<usize> = lines.next().unwrap().unwrap()
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    
    let width = header[0];
    let height = header[1];
    let obstacle_count = header[2];
    let require_cycle = header[3] == 1;
    
    // Parse obstacles
    let mut obstacles = Vec::with_capacity(obstacle_count);
    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<usize> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        obstacles.push((coords[0], coords[1]));
    }
    
    // Solve
    let result = solve(width, height, &obstacles, require_cycle);
    
    match result {
        Some(path) => {
            writeln!(out, "{}", path.len()).unwrap();
            for (x, y) in path {
                writeln!(out, "{} {}", x, y).unwrap();
            }
        }
        None => {
            writeln!(out, "0").unwrap();
        }
    }
}

fn solve(width: usize, height: usize, obstacles: &[(usize, usize)], require_cycle: bool) -> Option<Vec<(usize, usize)>> {
    // Special case: (0,0) is blocked
    if obstacles.contains(&(0, 0)) {
        return None;
    }
    
    let total_cells = width * height;
    let n_free = total_cells - obstacles.len();
    
    // Special case: single cell
    if n_free == 1 && !require_cycle {
        return Some(vec![(0, 0)]);
    }
    
    if n_free == 0 {
        return None;
    }
    
    // Build obstacle set
    let obstacle_set: HashSet<_> = obstacles.iter().copied().collect();
    
    // Count parity (chessboard coloring)
    let mut black = 0usize;
    let mut white = 0usize;
    
    for y in 0..height {
        for x in 0..width {
            if !obstacle_set.contains(&(x, y)) {
                if (x + y) % 2 == 0 {
                    black += 1;
                } else {
                    white += 1;
                }
            }
        }
    }
    
    // Parity check: necessary condition for Hamiltonian path/cycle
    if require_cycle {
        if black != white {
            return None;
        }
    } else {
        if black < white || black - white > 1 {
            return None;
        }
    }
    
    // Connectivity check using BFS
    if !is_connected(width, height, &obstacle_set, n_free) {
        return None;
    }
    
    // Degree check: no cell should have degree 0
    // For cycle: no cell should have degree < 2
    if !check_degrees(width, height, &obstacle_set, require_cycle) {
        return None;
    }
    
    // Try pattern-based solutions for obstacle-free grids
    if obstacles.is_empty() {
        if let Some(path) = try_pattern(width, height, require_cycle) {
            return Some(path);
        }
    }
    
    // Fall back to backtracking with heuristics
    backtrack(width, height, &obstacle_set, n_free, require_cycle)
}

fn is_connected(width: usize, height: usize, obstacles: &HashSet<(usize, usize)>, expected: usize) -> bool {
    if expected == 0 {
        return true;
    }
    
    let mut visited = vec![false; width * height];
    let mut queue = VecDeque::new();
    let mut count = 0;
    
    queue.push_back((0usize, 0usize));
    
    while let Some((x, y)) = queue.pop_front() {
        if x >= width || y >= height {
            continue;
        }
        let idx = y * width + x;
        if visited[idx] {
            continue;
        }
        if obstacles.contains(&(x, y)) {
            continue;
        }
        
        visited[idx] = true;
        count += 1;
        
        if x > 0 { queue.push_back((x - 1, y)); }
        if x + 1 < width { queue.push_back((x + 1, y)); }
        if y > 0 { queue.push_back((x, y - 1)); }
        if y + 1 < height { queue.push_back((x, y + 1)); }
    }
    
    count == expected
}

fn check_degrees(width: usize, height: usize, obstacles: &HashSet<(usize, usize)>, require_cycle: bool) -> bool {
    let min_degree = if require_cycle { 2 } else { 1 };
    
    for y in 0..height {
        for x in 0..width {
            if obstacles.contains(&(x, y)) {
                continue;
            }
            
            let mut degree = 0;
            if x > 0 && !obstacles.contains(&(x - 1, y)) { degree += 1; }
            if x + 1 < width && !obstacles.contains(&(x + 1, y)) { degree += 1; }
            if y > 0 && !obstacles.contains(&(x, y - 1)) { degree += 1; }
            if y + 1 < height && !obstacles.contains(&(x, y + 1)) { degree += 1; }
            
            if degree < min_degree {
                return false;
            }
        }
    }
    true
}

fn try_pattern(width: usize, height: usize, require_cycle: bool) -> Option<Vec<(usize, usize)>> {
    // Handle 1xN and Nx1 grids (single row/column)
    if width == 1 || height == 1 {
        if require_cycle {
            return None; // No cycle possible on a line
        }
        if width == 1 {
            return Some((0..height).map(|y| (0, y)).collect());
        } else {
            return Some((0..width).map(|x| (x, 0)).collect());
        }
    }
    
    if require_cycle {
        // Cycle pattern for even x even grids
        if width % 2 == 0 && height % 2 == 0 {
            return Some(cycle_even_even(width, height));
        }
        // For odd x even or even x odd grids, cycles are possible by parity
        // but require more complex construction - fall through to backtracking
        return None;
    } else {
        // Simple snake pattern for paths (works for all grid sizes)
        return Some(snake_path(width, height));
    }
}

fn snake_path(width: usize, height: usize) -> Vec<(usize, usize)> {
    let mut path = Vec::with_capacity(width * height);
    
    for y in 0..height {
        if y % 2 == 0 {
            // Left to right
            for x in 0..width {
                path.push((x, y));
            }
        } else {
            // Right to left
            for x in (0..width).rev() {
                path.push((x, y));
            }
        }
    }
    
    path
}

fn cycle_even_even(width: usize, height: usize) -> Vec<(usize, usize)> {
    let mut path = Vec::with_capacity(width * height);
    
    // Go down column 0
    for y in 0..height {
        path.push((0, y));
    }
    
    if height > 1 {
        // Go right on bottom row
        for x in 1..width {
            path.push((x, height - 1));
        }
        
        // Snake back up through remaining rows
        for row in (1..height - 1).rev() {
            let offset = height - 1 - row;
            if offset % 2 == 1 {
                for x in (1..width).rev() {
                    path.push((x, row));
                }
            } else {
                for x in 1..width {
                    path.push((x, row));
                }
            }
        }
        
        // Row 0: go from right to column 1
        for x in (2..width).rev() {
            path.push((x, 0));
        }
        
        // End at (1, 0) which is adjacent to (0, 0)
        path.push((1, 0));
    }
    
    path
}

fn backtrack(
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    n_free: usize,
    require_cycle: bool,
) -> Option<Vec<(usize, usize)>> {
    // Grid states: 0 = free, 1 = obstacle, 2 = visited
    let mut grid = vec![0u8; width * height];
    for &(x, y) in obstacles {
        grid[y * width + x] = 1;
    }
    
    let mut path = Vec::with_capacity(n_free);
    path.push((0, 0));
    grid[0] = 2; // Mark starting cell as visited
    
    if search(&mut grid, width, height, &mut path, n_free, require_cycle) {
        Some(path)
    } else {
        None
    }
}

#[inline]
fn get_neighbors(x: usize, y: usize, width: usize, height: usize) -> [(i32, i32, bool); 4] {
    [
        (x as i32 - 1, y as i32, x > 0),
        (x as i32 + 1, y as i32, x + 1 < width),
        (x as i32, y as i32 - 1, y > 0),
        (x as i32, y as i32 + 1, y + 1 < height),
    ]
}

fn search(
    grid: &mut [u8],
    width: usize,
    height: usize,
    path: &mut Vec<(usize, usize)>,
    target_len: usize,
    require_cycle: bool,
) -> bool {
    if path.len() == target_len {
        if require_cycle {
            // Check if last cell is adjacent to (0, 0)
            let (lx, ly) = path.last().unwrap();
            let dist = (*lx as i32).abs() + (*ly as i32).abs();
            return dist == 1;
        }
        return true;
    }
    
    let (x, y) = *path.last().unwrap();
    
    // Collect valid neighbors with their degrees (for Warnsdorff ordering)
    let mut neighbors: [(usize, usize, usize); 4] = [(0, 0, 5); 4]; // 5 = sentinel
    let mut n_count = 0;
    
    for (nx, ny, valid) in get_neighbors(x, y, width, height) {
        if !valid { continue; }
        let nx = nx as usize;
        let ny = ny as usize;
        let idx = ny * width + nx;
        
        if grid[idx] != 0 { continue; }
        
        let degree = count_degree(grid, width, height, nx, ny);
        neighbors[n_count] = (nx, ny, degree);
        n_count += 1;
    }
    
    // Sort by degree (Warnsdorff's rule: prefer cells with fewer exit options)
    neighbors[..n_count].sort_by_key(|&(_, _, d)| d);
    
    for i in 0..n_count {
        let (nx, ny, _) = neighbors[i];
        let idx = ny * width + nx;
        
        grid[idx] = 2;
        path.push((nx, ny));
        
        // Quick pruning: check if any neighbor of current position becomes isolated
        if !creates_isolated_cell(grid, width, height, nx, ny, target_len - path.len()) {
            if search(grid, width, height, path, target_len, require_cycle) {
                return true;
            }
        }
        
        path.pop();
        grid[idx] = 0;
    }
    
    false
}

#[inline]
fn count_degree(grid: &[u8], width: usize, height: usize, x: usize, y: usize) -> usize {
    let mut degree = 0;
    if x > 0 && grid[y * width + (x - 1)] == 0 { degree += 1; }
    if x + 1 < width && grid[y * width + (x + 1)] == 0 { degree += 1; }
    if y > 0 && grid[(y - 1) * width + x] == 0 { degree += 1; }
    if y + 1 < height && grid[(y + 1) * width + x] == 0 { degree += 1; }
    degree
}

fn creates_isolated_cell(grid: &[u8], width: usize, height: usize, last_x: usize, last_y: usize, remaining: usize) -> bool {
    // Check neighbors of the last visited cell for potential isolation
    for (nx, ny, valid) in get_neighbors(last_x, last_y, width, height) {
        if !valid { continue; }
        let nx = nx as usize;
        let ny = ny as usize;
        let idx = ny * width + nx;
        
        if grid[idx] != 0 { continue; }
        
        // Count this cell's remaining neighbors
        let mut degree = 0;
        if nx > 0 && grid[ny * width + (nx - 1)] == 0 { degree += 1; }
        if nx + 1 < width && grid[ny * width + (nx + 1)] == 0 { degree += 1; }
        if ny > 0 && grid[(ny - 1) * width + nx] == 0 { degree += 1; }
        if ny + 1 < height && grid[(ny + 1) * width + nx] == 0 { degree += 1; }
        
        // If degree is 0 and there are more cells to visit, we have an isolated cell
        if degree == 0 && remaining > 1 {
            return true;
        }
    }
    
    false
}

## Algorithm Summary

This solver uses a **multi-tiered approach**:

### 1. **Fast Pruning** (O(n))
- **Parity check**: Uses chessboard coloring - counts black/white cells. For paths: `black ≥ white` and `black - white ≤ 1`. For cycles: `black == white`.
- **Connectivity check**: BFS from (0,0) to ensure all free cells are reachable.
- **Degree check**: Ensures no cell has degree 0 (or < 2 for cycles).

### 2. **Pattern-Based Solutions** (O(n))
For obstacle-free grids:
- **Snake pattern**: Simple left-right-left traversal for any size path.
- **Cycle pattern for even×even**: Modified serpentine that ends adjacent to (0,0).

### 3. **Backtracking with Heuristics** (Exponential but pruned)
- **Warnsdorff's rule**: Always visit cells with fewest remaining neighbors first.
- **Isolation pruning**: Detect when a move creates an isolated region.
- **Efficient data structures**: Flat array for grid state, inline neighbor computation.

### Performance Characteristics
- **Obstacle-free grids**: O(n) - direct pattern construction
- **Small grids (<400 cells)**: Fast with Warnsdorff heuristic
- **Large grids with obstacles**: Depends on obstacle arrangement; parity/connectivity checks eliminate most impossible cases quickly
