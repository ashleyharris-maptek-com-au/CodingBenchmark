```spirv
; SPIR-V Assembly for Geometry Shader: Line to Thick Ribbon
; Version 1.0, Vulkan

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations for gl_PerVertex input block
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Decorations for gl_PerVertex output block
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Input locations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

; Output location
OpDecorate %outColor Location 0

; UBO layout (set=0, binding=0)
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 204
OpMemberDecorate %UBO 5 Offset 216
OpMemberDecorate %UBO 6 Offset 220
OpMemberDecorate %UBO 7 Offset 224
OpMemberDecorate %UBO 8 Offset 228
OpMemberDecorate %UBO 9 Offset 232
OpMemberDecorate %UBO 10 Offset 236
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

; Constants
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0_03 = OpConstant %float 0.03

; gl_PerVertex input type (array of 2 for InputLines)
%gl_PerVertex_in = OpTypeStruct %v4float
%uint_2_const = OpConstant %uint 2
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2_const
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; gl_PerVertex output type
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Input arrays for user-defined inputs (size 2)
%arr_v3_2 = OpTypeArray %v3float %uint_2_const
%ptr_input_arr_v3 = OpTypePointer Input %arr_v3_2
%worldPos = OpVariable %ptr_input_arr_v3 Input
%normal = OpVariable %ptr_input_arr_v3 Input
%color = OpVariable %ptr_input_arr_v3 Input

; Output color
%ptr_output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_output_v3 Output

; UBO type
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %float %float %float %float %float %float
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%ubo_var = OpVariable %ptr_uniform_ubo Uniform

; Pointer types for access
%ptr_input_v4 = OpTypePointer Input %v4float
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_uniform_float = OpTypePointer Uniform %float
%ptr_uniform_v3 = OpTypePointer Uniform %v3float

%main = OpFunction %void None %func
%entry = OpLabel

; Load p0 = gl_in[0].gl_Position
%p0_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr

; Load p1 = gl_in[1].gl_Position
%p1_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

; Load c0 = color[0]
%c0_ptr = OpAccessChain %ptr_input_v3 %color %int_0
%c0 = OpLoad %v3float %c0_ptr

; Load c1 = color[1]
%c1_ptr = OpAccessChain %ptr_input_v3 %color %int_1
%c1 = OpLoad %v3float %c1_ptr

; Compute direction: dir = p1 - p0
%dir = OpFSub %v4float %p1 %p0

; Extract XY components for perpendicular calculation
%p0_xy = OpCompositeExtract %v2float %p0 0
%p0_zw = OpCompositeExtract %v2float %p0 2
%p1_xy = OpCompositeExtract %v2float %p1 0
%p1_zw = OpCompositeExtract %v2float %p1 2
%dir_xy = OpCompositeExtract %v2float %dir 0

; Compute perpendicular in 2D: perp = normalize(-dir.y, dir.x) * thickness
%neg_dir_y = OpCompositeExtract %float %dir_xy 1
%neg_dir_y_neg = OpFNegate %float %neg_dir_y
%dir_x = OpCompositeExtract %float %dir_xy 0
%perp_dir = OpCompositeConstruct %v2float %neg_dir_y_neg %dir_x

; Normalize perpendicular
%perp_dir_len_sq = OpDot %float %perp_dir %perp_dir
%perp_dir_len = OpExtInst %float %1 Sqrt %perp_dir_len_sq
%perp_dir_norm = OpFDiv %v2float %perp_dir %perp_dir_len

; Scale by thickness
%perp = OpVectorTimesScalar %v2float %perp_dir_norm %float_0_03

; Negate perpendicular for other side
%neg_perp = OpFNegate %v2float %perp

; Compute 4 corner positions
; corner0 = p0 + perp (p0_xy + perp, keep p0.zw)
%corner0_xy = OpFAdd %v2float %p0_xy %perp
%corner0 = OpCompositeConstruct %v4float %corner0_xy %p0_zw

; corner1 = p0 - perp
%corner1_xy = OpFAdd %v2float %p0_xy %neg_perp
%corner1 = OpCompositeConstruct %v4float %corner1_xy %p0_zw

; corner2 = p1 + perp
%corner2_xy = OpFAdd %v2float %p1_xy %perp
%corner2 = OpCompositeConstruct %v4float %corner2_xy %p1_zw

; corner3 = p1 - perp
%corner3_xy = OpFAdd %v2float %p1_xy %neg_perp
%corner3 = OpCompositeConstruct %v4float %corner3_xy %p1_zw

; Emit vertex 0: p0 + perp, color = c0
%out_pos0 = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %out_pos0 %corner0
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 1: p0 - perp, color = c0
%out_pos1 = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %out_pos1 %corner1
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 2: p1 + perp, color = c1
%out_pos2 = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %out_pos2 %corner2
OpStore %outColor %c1
OpEmitVertex

; Emit vertex 3: p1 - perp, color = c1
%out_pos3 = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %out_pos3 %corner3
OpStore %outColor %c1
OpEmitVertex

; End the triangle strip (quad complete)
OpEndPrimitive

OpReturn
OpFunctionEnd
```
