```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Text;

public class TetrisAI
{
    // Board state
    private int width;
    private int height;
    private bool[,] board;
    private int piecesPlaced;
    private int linesCleared;
    
    // Piece definitions: [piece_type][rotation][cell_index] = (row, col)
    private static readonly Dictionary<char, int[][][]> PieceShapes = new Dictionary<char, int[][][]>
    {
        {'I', new int[][][] {
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {0,2}, new int[] {0,3} },
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {2,0}, new int[] {3,0} },
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {0,2}, new int[] {0,3} },
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {2,0}, new int[] {3,0} }
        }},
        {'O', new int[][][] {
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {1,0}, new int[] {1,1} },
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {1,0}, new int[] {1,1} },
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {1,0}, new int[] {1,1} },
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {1,0}, new int[] {1,1} }
        }},
        {'T', new int[][][] {
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {0,2}, new int[] {1,1} },
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {2,0}, new int[] {1,1} },
            new int[][] { new int[] {0,1}, new int[] {1,0}, new int[] {1,1}, new int[] {1,2} },
            new int[][] { new int[] {0,1}, new int[] {1,0}, new int[] {1,1}, new int[] {2,1} }
        }},
        {'S', new int[][][] {
            new int[][] { new int[] {0,1}, new int[] {0,2}, new int[] {1,0}, new int[] {1,1} },
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {1,1}, new int[] {2,1} },
            new int[][] { new int[] {0,1}, new int[] {0,2}, new int[] {1,0}, new int[] {1,1} },
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {1,1}, new int[] {2,1} }
        }},
        {'Z', new int[][][] {
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {1,1}, new int[] {1,2} },
            new int[][] { new int[] {0,1}, new int[] {1,0}, new int[] {1,1}, new int[] {2,0} },
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {1,1}, new int[] {1,2} },
            new int[][] { new int[] {0,1}, new int[] {1,0}, new int[] {1,1}, new int[] {2,0} }
        }},
        {'J', new int[][][] {
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {1,1}, new int[] {1,2} },
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {1,0}, new int[] {2,0} },
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {0,2}, new int[] {1,2} },
            new int[][] { new int[] {0,1}, new int[] {1,1}, new int[] {2,0}, new int[] {2,1} }
        }},
        {'L', new int[][][] {
            new int[][] { new int[] {0,2}, new int[] {1,0}, new int[] {1,1}, new int[] {1,2} },
            new int[][] { new int[] {0,0}, new int[] {1,0}, new int[] {2,0}, new int[] {2,1} },
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {0,2}, new int[] {1,0} },
            new int[][] { new int[] {0,0}, new int[] {0,1}, new int[] {1,1}, new int[] {2,1} }
        }}
    };
    
    // Adaptive weights based on game state
    private double weightAggregateHeight = -0.51;
    private double weightCompleteLines = 0.76;
    private double weightHoles = -0.36;
    private double weightBumpiness = -0.18;
    private double weightDeepHoles = -0.25;
    private double weightWells = 0.15;
    
    public TetrisAI(int w, int h)
    {
        width = w;
        height = h;
        board = new bool[height, width];
        piecesPlaced = 0;
        linesCleared = 0;
    }
    
    public void PlacePiece(char pieceType, out int bestRotation, out int bestColumn)
    {
        bestRotation = 0;
        bestColumn = 0;
        double bestScore = double.NegativeInfinity;
        
        AdaptWeights();
        
        int[][] pieceRotations = PieceShapes[pieceType];
        
        for (int rotation = 0; rotation < 4; rotation++)
        {
            int[][] shape = pieceRotations[rotation];
            int pieceWidth = GetPieceWidth(shape);
            int maxCol = width - pieceWidth;
            
            for (int col = 0; col <= maxCol; col++)
            {
                int dropRow = SimulateDrop(shape, col);
                if (dropRow < 0) continue;
                
                double score = EvaluatePlacement(shape, dropRow, col);
                
                if (score > bestScore)
                {
                    bestScore = score;
                    bestRotation = rotation;
                    bestColumn = col;
                }
            }
        }
        
        ApplyPlacement(PieceShapes[pieceType][bestRotation], bestColumn);
        piecesPlaced++;
    }
    
    private void AdaptWeights()
    {
        int currentHoles = CountHoles();
        int currentHeight = GetMaxHeight();
        
        if (piecesPlaced < 20)
        {
            weightAggregateHeight = -0.51;
            weightCompleteLines = 0.76;
            weightHoles = -0.36;
            weightBumpiness = -0.18;
        }
        else if (currentHeight < height * 0.6)
        {
            weightAggregateHeight = -0.51;
            weightCompleteLines = 1.0;
            weightHoles = -0.5;
            weightBumpiness = -0.2;
        }
        else
        {
            weightAggregateHeight = -0.8;
            weightCompleteLines = 1.5;
            weightHoles = -0.7;
            weightBumpiness = -0.3;
            weightDeepHoles = -0.5;
        }
        
        if (width < 12)
        {
            weightCompleteLines *= 1.3;
            weightHoles *= 1.2;
        }
        
        if (width > 14)
        {
            weightBumpiness *= 1.2;
            weightWells = 0.2;
        }
    }
    
    private int GetPieceWidth(int[][] shape)
    {
        int maxCol = 0;
        foreach (var cell in shape)
            maxCol = Math.Max(maxCol, cell[1]);
        return maxCol + 1;
    }
    
    private int SimulateDrop(int[][] shape, int col)
    {
        int dropRow = height - 1;
        
        foreach (var cell in shape)
        {
            int pieceCol = col + cell[1];
            if (pieceCol >= width) return -1;
            
            int cellRow = cell[0];
            for (int row = 0; row < height; row++)
            {
                if (board[row, pieceCol])
                {
                    int landRow = row - 1 - cellRow;
                    dropRow = Math.Min(dropRow, landRow);
                    break;
                }
            }
        }
        return dropRow;
    }
    
    private double EvaluatePlacement(int[][] shape, int dropRow, int col)
    {
        var tempBoard = new bool[height, width];
        Array.Copy(board, tempBoard, board.Length);
        
        foreach (var cell in shape)
        {
            int row = dropRow + cell[0];
            int c = col + cell[1];
            if (row < 0 || row >= height || c < 0 || c >= width)
                return double.NegativeInfinity;
            tempBoard[row, c] = true;
        }
        
        int linesCleared = ClearLines(tempBoard);
        int aggregateHeight = CalculateAggregateHeight(tempBoard);
        int holes = CountHoles(tempBoard);
        int bumpiness = CalculateBumpiness(tempBoard);
        int deepHoles = CountDeepHoles(tempBoard);
        int wells = CountWells(tempBoard);
        
        double score = 0;
        score += weightAggregateHeight * aggregateHeight;
        score += weightCompleteLines * linesCleared;
        score += weightHoles * holes;
        score += weightBumpiness * bumpiness;
        score += weightDeepHoles * deepHoles;
        score += weightWells * wells;
        
        if (linesCleared == 4) score += 10;
        
        return score;
    }
    
    private void ApplyPlacement(int[][] shape, int col)
    {
        int dropRow = SimulateDrop(shape, col);
        
        foreach (var cell in shape)
        {
            int row = dropRow + cell[0];
            int c = col + cell[1];
            if (row >= 0 && row < height && c >= 0 && c < width)
                board[row, c] = true;
        }
        ClearLines(board);
    }
    
    private int ClearLines(bool[,] b)
    {
        int cleared = 0;
        for (int row = height - 1; row >= 0; row--)
        {
            bool complete = true;
            for (int col = 0; col < width; col++)
            {
                if (!b[row, col]) { complete = false; break; }
            }
            if (complete)
            {
                cleared++;
                for (int r = row; r > 0; r--)
                    for (int c = 0; c < width; c++)
                        b[r, c] = b[r - 1, c];
                for (int c = 0; c < width; c++)
                    b[0, c] = false;
                row++;
            }
        }
        linesCleared += cleared;
        return cleared;
    }
    
    private int CalculateAggregateHeight(bool[,] b)
    {
        int total = 0;
        for (int col = 0; col < width; col++)
            for (int row = 0; row < height; row++)
                if (b[row, col]) { total += height - row; break; }
        return total;
    }
    
    private int GetMaxHeight()
    {
        int maxH = 0;
        for (int col = 0; col < width; col++)
            for (int row = 0; row < height; row++)
                if (board[row, col]) { maxH = Math.Max(maxH, height - row); break; }
        return maxH;
    }
    
    private int CountHoles(bool[,] b)
    {
        int holes = 0;
        for (int col = 0; col < width; col++)
        {
            bool foundBlock = false;
            for (int row = 0; row < height; row++)
            {
                if (b[row, col]) foundBlock = true;
                else if (foundBlock) holes++;
            }
        }
        return holes;
    }
    
    private int CountHoles() => CountHoles(board);
    
    private int CalculateBumpiness(bool[,] b)
    {
        int bumpiness = 0, prevHeight = 0;
        for (int col = 0; col < width; col++)
        {
            int colHeight = 0;
            for (int row = 0; row < height; row++)
                if (b[row, col]) { colHeight = height - row; break; }
            if (col > 0) bumpiness += Math.Abs(colHeight - prevHeight);
            prevHeight = colHeight;
        }
        return bumpiness;
    }
    
    private int CountDeepHoles(bool[,] b)
    {
        int deepHoles = 0;
        for (int col = 0; col < width; col++)
        {
            int holeDepth = 0;
            bool foundBlock = false;
            for (int row = 0; row < height; row++)
            {
                if (b[row, col]) foundBlock = true;
                else if (foundBlock) holeDepth++;
            }
            if (holeDepth >= 3) deepHoles += holeDepth;
        }
        return deepHoles;
    }
    
    private int CountWells(bool[,] b)
    {
        int wells = 0;
        for (int col = 0; col < width; col++)
        {
            int leftHeight = (col == 0) ? height : GetColumnHeight(b, col - 1);
            int rightHeight = (col == width - 1) ? height : GetColumnHeight(b, col + 1);
            int colHeight = GetColumnHeight(b, col);
            if (colHeight < leftHeight - 1 && colHeight < rightHeight - 1)
                wells += (Math.Min(leftHeight, rightHeight) - colHeight);
        }
        return wells;
    }
    
    private int GetColumnHeight(bool[,] b, int col)
    {
        for (int row = 0; row < height; row++)
            if (b[row, col]) return height - row;
        return 0;
    }
}

public class Program
{
    public static void Main(string[] args)
    {
        string line = Console.ReadLine();
        while (string.IsNullOrWhiteSpace(line)) line = Console.ReadLine();
        
        string[] parts = line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int width = int.Parse(parts[0]);
        int height = int.Parse(parts[1]);
        int numPieces = int.Parse(parts[2]);
        
        var ai = new TetrisAI(width, height);
        
        for (int i = 0; i < numPieces; i++)
        {
            string pieceLine = Console.ReadLine();
            while (string.IsNullOrWhiteSpace(pieceLine)) pieceLine = Console.ReadLine();
            char pieceType = pieceLine.Trim()[0];
            
            int rotation, column;
            ai.PlacePiece(pieceType, out rotation, out column);
            
            Console.WriteLine($"{rotation} {column}");
            Console.Out.Flush();
            
            string result = Console.ReadLine();
            while (string.IsNullOrWhiteSpace(result)) result = Console.ReadLine();
            if (result.Trim().StartsWith("gameover")) break;
        }
    }
}
```

## Key Features:

1. **Adaptive Strategy**: Weights adjust based on:
   - Game phase (early/mid/late)
   - Board height (danger zone detection)
   - Board width (narrow vs wide strategies)

2. **Evaluation Heuristics**:
   - **Aggregate Height**: Lower is better (more room)
   - **Complete Lines**: Higher is better (scoring)
   - **Holes**: Fewer is better (cleaner board)
   - **Bumpiness**: Smoother is better (easier placement)
   - **Deep Holes**: Penalizes hard-to-fill gaps
   - **Wells**: Can be strategic for I-pieces

3. **Performance**: O(width × rotations) evaluations per piece, easily under 5 seconds

4. **Correctness**: All 7 Tetriminos with all 4 rotations properly defined
