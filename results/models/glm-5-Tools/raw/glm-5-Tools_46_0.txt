```glsl
#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint id = gl_GlobalInvocationID.x;
    uint N = counts.x;
    
    // Guard against out-of-bounds invocations
    if (id >= N) {
        return;
    }
    
    // Load current body's position, mass, and velocity
    vec4 posMass = inData[id * 2 + 0];
    vec4 velDummy = inData[id * 2 + 1];
    
    vec3 pos = posMass.xyz;
    float mass_i = posMass.w;
    vec3 vel = velDummy.xyz;
    
    // Extract physics parameters
    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;
    float softeningSq = softening * softening;
    
    // Compute gravitational acceleration from all other bodies
    vec3 acc = vec3(0.0);
    
    for (uint j = 0u; j < N; j++) {
        // Skip self-interaction
        if (j == id) {
            continue;
        }
        
        // Load other body's position and mass
        vec4 otherPosMass = inData[j * 2 + 0];
        vec3 pos_j = otherPosMass.xyz;
        float mass_j = otherPosMass.w;
        
        // Compute displacement vector
        vec3 r = pos_j - pos;
        
        // Compute distance squared with softening
        float distSq = dot(r, r) + softeningSq;
        
        // Compute inverse distance cubed: 1 / (distSq)^(3/2)
        float invDistCubed = 1.0 / (distSq * sqrt(distSq));
        
        // Accumulate acceleration: G * mass_j * r / |r|^3
        acc += G * mass_j * r * invDistCubed;
    }
    
    // Leapfrog/Euler integration
    // Update velocity: v' = v + a * dt
    vec3 newVel = vel + acc * dt;
    
    // Update position: x' = x + v * dt
    vec3 newPos = pos + newVel * dt;
    
    // Write output: position+mass and velocity
    outData[id * 2 + 0] = vec4(newPos, mass_i);
    outData[id * 2 + 1] = vec4(newVel, 0.0);
}
```

**Key Implementation Details:**

1. **Bounds Checking**: The shader immediately returns if `gl_GlobalInvocationID.x >= N`, preventing out-of-bounds buffer access.

2. **Data Layout**: Each body's data is accessed at indices `i*2` (position+mass) and `i*2+1` (velocity).

3. **Softened Gravity**: The distance calculation uses `|r|² + softening²` to prevent singularities when bodies get very close:
   ```
   acc = G * m_j * r / (|r|² + ε²)^(3/2)
   ```

4. **Self-Interaction Skip**: The loop explicitly skips `j == id` to prevent a body from gravitationally attracting itself.

5. **Integration Method**: Uses leapfrog/Euler scheme:
   - Velocity updated: `v' = v + a*dt`
   - Position updated: `x' = x + v'*dt` (uses updated velocity for better energy conservation)

6. **Output Preservation**: The mass is preserved in the output position vec4, and the velocity vec4's w-component is written as 0.0.
