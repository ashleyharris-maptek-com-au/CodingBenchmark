%entry = OpLabel

; Get UV

; Compute c = (uv.x * 3 - 2, uv.y * 2 - 1)
%uv_x = OpCompositeExtract %float %uv 0
%uv_y = OpCompositeExtract %float %uv 1

%cx_part1 = OpFMul %float %uv_x %float_3

; Initialize z = (0, 0), iterations = 0

%loop_cond = OpLabel
%zx = OpPhi %float %float_0 %entry %zx_new %loop_body

%zy2 = OpFMul %float %zy %zy

%escaped = OpFGreaterThan %bool %mag2 %float_4

%iter_lt_20 = OpSLessThan %bool %iter %int_20
%not_escaped = OpLogicalNot %bool %escaped
%loop_continue = OpLogicalAnd %bool %iter_lt_20 %not_escaped

OpLoopMerge %loop_exit %loop_body None
OpBranchConditional %loop_continue %loop_body %loop_exit

%loop_body = OpLabel

%zy_cy = OpFMul %float %zy %cy
%zx_zy = OpFMul %float %zx %zy

%zx_new = OpFAdd %float %zx2 %zy_cy

%zx_new = OpFSub %float %zx2 %cy

%zx_new = OpFAdd %float %zx_temp %cx

%iter_inc = OpIAdd %int %iter %int_1
OpBranch %loop_cond

%loop_exit = OpLabel

%iter_final = OpPhi %int %iter %loop_cond
