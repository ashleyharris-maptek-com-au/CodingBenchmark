Here's a complete GLSL compute shader implementing Reynolds' boid flocking rules:

```glsl
#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint gid = gl_GlobalInvocationID.x;
    uint N = counts.x;
    
    // Guard against out-of-bounds invocation
    if (gid >= N) return;
    
    // Extract current boid position and velocity
    vec3 pos = inData[gid * 2].xyz;
    vec3 vel = inData[gid * 2 + 1].xyz;
    
    // Unpack uniform parameters
    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;
    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;
    float boundSize = bounds.x;
    float boundForce = bounds.y;
    
    // Accumulators for Reynolds' rules
    vec3 separation = vec3(0.0);
    vec3 alignment = vec3(0.0);
    vec3 cohesion = vec3(0.0);
    
    // Neighbor counters for each rule (can be different due to different radii)
    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;
    
    // Iterate over all potential neighbors
    for (uint j = 0u; j < N; j++) {
        // Skip self
        if (j == gid) continue;
        
        vec3 neighborPos = inData[j * 2].xyz;
        vec3 neighborVel = inData[j * 2 + 1].xyz;
        
        // Calculate distance vector and squared distance
        vec3 diff = pos - neighborPos;
        float distSq = dot(diff, diff);
        
        // Skip zero-distance neighbors (shouldn't happen, but guard against it)
        if (distSq < 1e-6) continue;
        
        float dist = sqrt(distSq);
        
        // Separation: steer away from nearby neighbors
        if (dist < sepRadius) {
            // Weight by inverse distance - closer neighbors have more influence
            separation += diff / distSq;
            sepCount++;
        }
        
        // Alignment: match velocity with nearby neighbors
        if (dist < alignRadius) {
            alignment += neighborVel;
            alignCount++;
        }
        
        // Cohesion: steer toward center of mass of neighbors
        if (dist < cohRadius) {
            cohesion += neighborPos;
            cohCount++;
        }
    }
    
    // Compute separation force (average of weighted contributions)
    vec3 separationForce = vec3(0.0);
    if (sepCount > 0) {
        separationForce = separation / float(sepCount);
        separationForce = normalize(separationForce) * sepWeight;
    }
    
    // Compute alignment force (average neighbor velocity, normalized and weighted)
    vec3 alignmentForce = vec3(0.0);
    if (alignCount > 0) {
        alignmentForce = alignment / float(alignCount);
        alignmentForce = normalize(alignmentForce) * alignWeight;
    }
    
    // Compute cohesion force (steer toward center of mass)
    vec3 cohesionForce = vec3(0.0);
    if (cohCount > 0) {
        vec3 centerOfMass = cohesion / float(cohCount);
        cohesionForce = centerOfMass - pos;
        cohesionForce = normalize(cohesionForce) * cohWeight;
    }
    
    // Total acceleration from flocking rules
    vec3 acceleration = separationForce + alignmentForce + cohesionForce;
    
    // Boundary force: push boids back toward center when near bounds
    // Apply when component exceeds boundSize * 0.4
    float boundThreshold = boundSize * 0.4;
    float boundScale = boundSize * 0.5;
    
    vec3 boundAccel = vec3(0.0);
    if (abs(pos.x) > boundThreshold) {
        boundAccel.x = -sign(pos.x) * boundForce * (abs(pos.x) - boundThreshold) / boundScale;
    }
    if (abs(pos.y) > boundThreshold) {
        boundAccel.y = -sign(pos.y) * boundForce * (abs(pos.y) - boundThreshold) / boundScale;
    }
    if (abs(pos.z) > boundThreshold) {
        boundAccel.z = -sign(pos.z) * boundForce * (abs(pos.z) - boundThreshold) / boundScale;
    }
    
    acceleration += boundAccel;
    
    // Update velocity: v = v + a * dt
    vel += acceleration * dt;
    
    // Clamp velocity magnitude to maxSpeed
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = vel / speed * maxSpeed;
    }
    
    // Update position: p = p + v * dt
    pos += vel * dt;
    
    // Write updated position and velocity to output buffer
    outData[gid * 2] = vec4(pos, 0.0);
    outData[gid * 2 + 1] = vec4(vel, 0.0);
}
```

## Key Implementation Details

**Reynolds' Flocking Rules:**

1. **Separation**: Steers away from neighbors within `sepRadius`. The force is weighted by `1/distance²` so closer neighbors have stronger repulsion.

2. **Alignment**: Averages the velocities of neighbors within `alignRadius` and steers to match that average direction.

3. **Cohesion**: Computes the center of mass of neighbors within `cohRadius` and steers toward it.

**Boundary Handling:**
- When any position component exceeds `boundSize * 0.4`, a restoring force pushes the boid toward the center
- Force scales linearly with how far past the threshold the boid has traveled
- Direction is always toward the origin (center of the simulation volume)

**Velocity Clamping:**
- After applying acceleration, the velocity magnitude is clamped to `maxSpeed`
- Prevents boids from accelerating indefinitely

**Performance Note:**
This O(N²) algorithm is straightforward but doesn't scale well for large N. For production use with thousands of boids, consider implementing spatial hashing or a uniform grid to reduce neighbor searches to O(N) average case.
