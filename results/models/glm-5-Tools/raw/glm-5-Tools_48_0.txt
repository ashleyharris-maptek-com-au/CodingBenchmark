Here's the complete SPIR-V assembly compute shader for the D2Q9 Lattice Boltzmann simulation:

```spv
; SPIR-V
; Version: 1.0
; Generator: Khronos SPIR-V Tools Assembler
; Bound: 250
; Schema: 0

OpCapability Shader
OpCapability Int64
OpMemoryModel Logical GLSL450
OpEntryPoint GLCompute %main "main" %gl_GlobalInvocationID
OpExecutionMode %main LocalSize 16 16 1
OpSource GLSL 450

; Decorations
OpDecorate %gl_GlobalInvocationID BuiltIn GlobalInvocationId
OpDecorate %arr_storage ArrayStride 4
OpDecorate %buf_in Block
OpDecorate %buf_out Block
OpDecorate %buf_uniform Block

OpDecorate %var_in DescriptorSet 0
OpDecorate %var_in Binding 0
OpDecorate %var_in NonWritable

OpDecorate %var_out DescriptorSet 0
OpDecorate %var_out Binding 1

OpDecorate %var_uniform DescriptorSet 0
OpDecorate %var_uniform Binding 2

; Types
%void = OpTypeVoid
%void_fn = OpTypeFunction %void

%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32

%v2uint = OpTypeVector %uint 2
%v3uint = OpTypeVector %uint 3
%v4uint = OpTypeVector %uint 4
%v2float = OpTypeVector %float 2
%v4float = OpTypeVector %float 4

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%uint_5 = OpConstant %uint 5
%uint_6 = OpConstant %uint 6
%uint_7 = OpConstant %uint 7
%uint_8 = OpConstant %uint 8
%uint_9 = OpConstant %uint 9

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_n1 = OpConstant %int -1

; Constants for BGK collision
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_3 = OpConstant %float 3
%float_4_5 = OpConstant %float 4.5
%float_1_5 = OpConstant %float 1.5

; D2Q9 weights: w[0]=4/9, w[1-4]=1/9, w[5-8]=1/36
%float_w0 = OpConstant %float 0.444444444  ; 4/9
%float_w1 = OpConstant %float 0.111111111  ; 1/9
%float_w5 = OpConstant %float 0.027777778  ; 1/36

; Storage buffer types
%arr_storage = OpTypeRuntimeArray %float

%buf_in = OpTypeStruct %arr_storage
%ptr_storage_buf_in = OpTypePointer StorageBuffer %buf_in
%ptr_storage_float_in = OpTypePointer StorageBuffer %float

%buf_out = OpTypeStruct %arr_storage
%ptr_storage_buf_out = OpTypePointer StorageBuffer %buf_out
%ptr_storage_float_out = OpTypePointer StorageBuffer %float

; Uniform buffer: { uvec4(W, H, 0, 0), vec4(omega, 0, 0, 0) }
%uniform_struct = OpTypeStruct %v4uint %v4float
%ptr_uniform_struct = OpTypePointer Uniform %uniform_struct
%ptr_uniform_v4uint = OpTypePointer Uniform %v4uint
%ptr_uniform_v4float = OpTypePointer Uniform %v4float

; Variables
%var_in = OpVariable %ptr_storage_buf_in StorageBuffer
%var_out = OpVariable %ptr_storage_buf_out StorageBuffer
%var_uniform = OpVariable %ptr_uniform_struct Uniform

%gl_GlobalInvocationID = OpVariable %_ptr_Input_v3uint Input
%_ptr_Input_v3uint = OpTypePointer Input %v3uint
%_ptr_Input_uint = OpTypePointer Input %uint

%_ptr_Function_float = OpTypePointer Function %float
%_ptr_Function_v2float = OpTypePointer Function %v2float
%_ptr_Function_uint = OpTypePointer Function %uint
%_ptr_Function_int = OpTypePointer Function %int

; Functions
%load_f = OpTypeFunction %float %uint %uint  ; (base_idx, k) -> float

%main = OpFunction %void None %void_fn
%entry = OpLabel

    ; Local variables for f[0..8]
    %f0 = OpVariable %_ptr_Function_float Function
    %f1 = OpVariable %_ptr_Function_float Function
    %f2 = OpVariable %_ptr_Function_float Function
    %f3 = OpVariable %_ptr_Function_float Function
    %f4 = OpVariable %_ptr_Function_float Function
    %f5 = OpVariable %_ptr_Function_float Function
    %f6 = OpVariable %_ptr_Function_float Function
    %f7 = OpVariable %_ptr_Function_float Function
    %f8 = OpVariable %_ptr_Function_float Function

    ; Get global invocation ID
    %gid = OpLoad %v3uint %gl_GlobalInvocationID
    %x = OpCompositeExtract %uint %gid 0
    %y = OpCompositeExtract %uint %gid 1

    ; Load grid dimensions from uniform buffer
    %uniform_ptr0 = OpAccessChain %ptr_uniform_v4uint %var_uniform %int_0
    %dims = OpLoad %v4uint %uniform_ptr0
    %W = OpCompositeExtract %uint %dims 0
    %H = OpCompositeExtract %uint %dims 1

    ; Bounds check
    %x_lt_W = OpULessThan %bool %x %W
    %y_lt_H = OpULessThan %bool %y %H
    %in_bounds = OpLogicalAnd %bool %x_lt_W %y_lt_H
    OpSelectionMerge %bounds_merge None
    OpBranchConditional %in_bounds %do_work %bounds_merge

%do_work = OpLabel
    ; Load omega from uniform buffer
    %uniform_ptr1 = OpAccessChain %ptr_uniform_v4float %var_uniform %int_1
    %omega_vec = OpLoad %v4float %uniform_ptr1
    %omega = OpCompositeExtract %float %omega_vec 0

    ; Calculate base index: (y * W + x) * 9
    %y_W = OpIMul %uint %y %W
    %yW_x = OpIAdd %uint %y_W %x
    %base_idx = OpIMul %uint %yW_x %uint_9

    ; Load all 9 distribution functions
    ; f[0]
    %idx0 = OpIAdd %uint %base_idx %uint_0
    %ptr_f0 = OpAccessChain %ptr_storage_float_in %var_in %int_0 %idx0
    %val_f0 = OpLoad %float %ptr_f0
    OpStore %f0 %val_f0

    ; f[1]
    %idx1 = OpIAdd %uint %base_idx %uint_1
    %ptr_f1 = OpAccessChain %ptr_storage_float_in %var_in %int_0 %idx1
    %val_f1 = OpLoad %float %ptr_f1
    OpStore %f1 %val_f1

    ; f[2]
    %idx2 = OpIAdd %uint %base_idx %uint_2
    %ptr_f2 = OpAccessChain %ptr_storage_float_in %var_in %int_0 %idx2
    %val_f2 = OpLoad %float %ptr_f2
    OpStore %f2 %val_f2

    ; f[3]
    %idx3 = OpIAdd %uint %base_idx %uint_3
    %ptr_f3 = OpAccessChain %ptr_storage_float_in %var_in %int_0 %idx3
    %val_f3 = OpLoad %float %ptr_f3
    OpStore %f3 %val_f3

    ; f[4]
    %idx4 = OpIAdd %uint %base_idx %uint_4
    %ptr_f4 = OpAccessChain %ptr_storage_float_in %var_in %int_0 %idx4
    %val_f4 = OpLoad %float %ptr_f4
    OpStore %f4 %val_f4

    ; f[5]
    %idx5 = OpIAdd %uint %base_idx %uint_5
    %ptr_f5 = OpAccessChain %ptr_storage_float_in %var_in %int_0 %idx5
    %val_f5 = OpLoad %float %ptr_f5
    OpStore %f5 %val_f5

    ; f[6]
    %idx6 = OpIAdd %uint %base_idx %uint_6
    %ptr_f6 = OpAccessChain %ptr_storage_float_in %var_in %int_0 %idx6
    %val_f6 = OpLoad %float %ptr_f6
    OpStore %f6 %val_f6

    ; f[7]
    %idx7 = OpIAdd %uint %base_idx %uint_7
    %ptr_f7 = OpAccessChain %ptr_storage_float_in %var_in %int_0 %idx7
    %val_f7 = OpLoad %float %ptr_f7
    OpStore %f7 %val_f7

    ; f[8]
    %idx8 = OpIAdd %uint %base_idx %uint_8
    %ptr_f8 = OpAccessChain %ptr_storage_float_in %var_in %int_0 %idx8
    %val_f8 = OpLoad %float %ptr_f8
    OpStore %f8 %val_f8

    ; Compute macroscopic density rho = sum(f[k])
    %rho_01 = OpFAdd %float %val_f0 %val_f1
    %rho_012 = OpFAdd %float %rho_01 %val_f2
    %rho_0123 = OpFAdd %float %rho_012 %val_f3
    %rho_01234 = OpFAdd %float %rho_0123 %val_f4
    %rho_012345 = OpFAdd %float %rho_01234 %val_f5
    %rho_0123456 = OpFAdd %float %rho_012345 %val_f6
    %rho_01234567 = OpFAdd %float %rho_0123456 %val_f7
    %rho = OpFAdd %float %rho_01234567 %val_f8

    ; Compute momentum: jx = sum(cx[k] * f[k]), jy = sum(cy[k] * f[k])
    ; k=0: (0,0) - contributes nothing
    ; k=1: (1,0) -> +f1
    ; k=2: (0,1) -> +f2
    ; k=3: (-1,0) -> -f3
    ; k=4: (0,-1) -> -f4
    ; k=5: (1,1) -> +f5
    ; k=6: (-1,1) -> -f6 for x, +f6 for y
    ; k=7: (-1,-1) -> -f7 for both
    ; k=8: (1,-1) -> +f8 for x, -f8 for y

    %jx_1 = OpFSub %float %val_f1 %val_f3    ; f1 - f3
    %jx_5 = OpFAdd %float %jx_1 %val_f5      ; + f5
    %jx_6 = OpFSub %float %jx_5 %val_f6      ; - f6
    %jx_7 = OpFSub %float %jx_6 %val_f7      ; - f7
    %jx = OpFAdd %float %jx_7 %val_f8        ; + f8

    %jy_2 = OpFSub %float %val_f2 %val_f4    ; f2 - f4
    %jy_5 = OpFAdd %float %jy_2 %val_f5      ; + f5
    %jy_6 = OpFAdd %float %jy_5 %val_f6      ; + f6
    %jy_7 = OpFSub %float %jy_6 %val_f7      ; - f7
    %jy = OpFSub %float %jy_7 %val_f8        ; - f8

    ; Compute velocity: ux = jx / rho, uy = jy / rho
    %inv_rho = OpFDiv %float %float_1 %rho
    %ux = OpFMul %float %jx %inv_rho
    %uy = OpFMul %float %jy %inv_rho

    ; Precompute common term: u_sq = ux^2 + uy^2
    %ux2 = OpFMul %float %ux %ux
    %uy2 = OpFMul %float %uy %uy
    %u_sq = OpFAdd %float %ux2 %uy2

    ; Precompute: -1.5 * u_sq
    %u_sq_neg = OpFNegate %float %u_sq
    %term_common = OpFMul %float %float_1_5 %u_sq_neg

    ; ========================================
    ; BGK Collision for each k
    ; feq = w * rho * (1 + 3*cu + 4.5*cu^2 - 1.5*u_sq)
    ; f_out = f + omega * (feq - f)
    ; ========================================

    ; --- k=0: (cx,cy)=(0,0), w=4/9 ---
    ; cu = 0
    ; feq = w0 * rho * (1 + 0 + 0 + term_common)
    ;     = w0 * rho * (1 + term_common)
    %one_plus_term = OpFAdd %float %float_1 %term_common
    %w0_rho = OpFMul %float %float_w0 %rho
    %feq0 = OpFMul %float %w0_rho %one_plus_term
    %diff0 = OpFSub %float %feq0 %val_f0
    %omega_diff0 = OpFMul %float %omega %diff0
    %f_out0 = OpFAdd %float %val_f0 %omega_diff0

    ; --- k=1: (cx,cy)=(1,0), w=1/9 ---
    ; cu = ux
    %cu1 = OpFMul %float %float_3 %ux
    %cu1_sq = OpFMul %float %ux %ux
    %cu1_sq_scaled = OpFMul %float %float_4_5 %cu1_sq
    %term1 = OpFAdd %float %cu1 %cu1_sq_scaled
    %term1_full = OpFAdd %float %term1 %term_common
    %bracket1 = OpFAdd %float %float_1 %term1_full
    %w1_rho = OpFMul %float %float_w1 %rho
    %feq1 = OpFMul %float %w1_rho %bracket1
    %diff1 = OpFSub %float %feq1 %val_f1
    %omega_diff1 = OpFMul %float %omega %diff1
    %f_out1 = OpFAdd %float %val_f1 %omega_diff1

    ; --- k=2: (cx,cy)=(0,1), w=1/9 ---
    ; cu = uy
    %cu2 = OpFMul %float %float_3 %uy
    %cu2_sq = OpFMul %float %uy %uy
    %cu2_sq_scaled = OpFMul %float %float_4_5 %cu2_sq
    %term2 = OpFAdd %float %cu2 %cu2_sq_scaled
    %term2_full = OpFAdd %float %term2 %term_common
    %bracket2 = OpFAdd %float %float_1 %term2_full
    %feq2 = OpFMul %float %w1_rho %bracket2
    %diff2 = OpFSub %float %feq2 %val_f2
    %omega_diff2 = OpFMul %float %omega %diff2
    %f_out2 = OpFAdd %float %val_f2 %omega_diff2

    ; --- k=3: (cx,cy)=(-1,0), w=1/9 ---
    ; cu = -ux
    %neg_ux = OpFNegate %float %ux
    %cu3 = OpFMul %float %float_3 %neg_ux
    %cu3_sq = OpFMul %float %neg_ux %neg_ux
    %cu3_sq_scaled = OpFMul %float %float_4_5 %cu3_sq
    %term3 = OpFAdd %float %cu3 %cu3_sq_scaled
    %term3_full = OpFAdd %float %term3 %term_common
    %bracket3 = OpFAdd %float %float_1 %term3_full
    %feq3 = OpFMul %float %w1_rho %bracket3
    %diff3 = OpFSub %float %feq3 %val_f3
    %omega_diff3 = OpFMul %float %omega %diff3
    %f_out3 = OpFAdd %float %val_f3 %omega_diff3

    ; --- k=4: (cx,cy)=(0,-1), w=1/9 ---
    ; cu = -uy
    %neg_uy = OpFNegate %float %uy
    %cu4 = OpFMul %float %float_3 %neg_uy
    %cu4_sq = OpFMul %float %neg_uy %neg_uy
    %cu4_sq_scaled = OpFMul %float %float_4_5 %cu4_sq
    %term4 = OpFAdd %float %cu4 %cu4_sq_scaled
    %term4_full = OpFAdd %float %term4 %term_common
    %bracket4 = OpFAdd %float %float_1 %term4_full
    %feq4 = OpFMul %float %w1_rho %bracket4
    %diff4 = OpFSub %float %feq4 %val_f4
    %omega_diff4 = OpFMul %float %omega %diff4
    %f_out4 = OpFAdd %float %val_f4 %omega_diff4

    ; --- k=5: (cx,cy)=(1,1), w=1/36 ---
    ; cu = ux + uy
    %cu5_base = OpFAdd %float %ux %uy
    %cu5 = OpFMul %float %float_3 %cu5_base
    %cu5_sq = OpFMul %float %cu5_base %cu5_base
    %cu5_sq_scaled = OpFMul %float %float_4_5 %cu5_sq
    %term5 = OpFAdd %float %cu5 %cu5_sq_scaled
    %term5_full = OpFAdd %float %term5 %term_common
    %bracket5 = OpFAdd %float %float_1 %term5_full
    %w5_rho = OpFMul %float %float_w5 %rho
    %feq5 = OpFMul %float %w5_rho %bracket5
    %diff5 = OpFSub %float %feq5 %val_f5
    %omega_diff5 = OpFMul %float %omega %diff5
    %f_out5 = OpFAdd %float %val_f5 %omega_diff5

    ; --- k=6: (cx,cy)=(-1,1), w=1/36 ---
    ; cu = -ux + uy
    %cu6_base = OpFAdd %float %neg_ux %uy
    %cu6 = OpFMul %float %float_3 %cu6_base
    %cu6_sq = OpFMul %float %cu6_base %cu6_base
    %cu6_sq_scaled = OpFMul %float %float_4_5 %cu6_sq
    %term6 = OpFAdd %float %cu6 %cu6_sq_scaled
    %term6_full = OpFAdd %float %term6 %term_common
    %bracket6 = OpFAdd %float %float_1 %term6_full
    %feq6 = OpFMul %float %w5_rho %bracket6
    %diff6 = OpFSub %float %feq6 %val_f6
    %omega_diff6 = OpFMul %float %omega %diff6
    %f_out6 = OpFAdd %float %val_f6 %omega_diff6

    ; --- k=7: (cx,cy)=(-1,-1), w=1/36 ---
    ; cu = -ux - uy
    %cu7_base = OpFAdd %float %neg_ux %neg_uy
    %cu7 = OpFMul %float %float_3 %cu7_base
    %cu7_sq = OpFMul %float %cu7_base %cu7_base
    %cu7_sq_scaled = OpFMul %float %float_4_5 %cu7_sq
    %term7 = OpFAdd %float %cu7 %cu7_sq_scaled
    %term7_full = OpFAdd %float %term7 %term_common
    %bracket7 = OpFAdd %float %float_1 %term7_full
    %feq7 = OpFMul %float %w5_rho %bracket7
    %diff7 = OpFSub %float %feq7 %val_f7
    %omega_diff7 = OpFMul %float %omega %diff7
    %f_out7 = OpFAdd %float %val_f7 %omega_diff7

    ; --- k=8: (cx,cy)=(1,-1), w=1/36 ---
    ; cu = ux - uy
    %cu8_base = OpFAdd %float %ux %neg_uy
    %cu8 = OpFMul %float %float_3 %cu8_base
    %cu8_sq = OpFMul %float %cu8_base %cu8_base
    %cu8_sq_scaled = OpFMul %float %float_4_5 %cu8_sq
    %term8 = OpFAdd %float %cu8 %cu8_sq_scaled
    %term8_full = OpFAdd %float %term8 %term_common
    %bracket8 = OpFAdd %float %float_1 %term8_full
    %feq8 = OpFMul %float %w5_rho %bracket8
    %diff8 = OpFSub %float %feq8 %val_f8
    %omega_diff8 = OpFMul %float %omega %diff8
    %f_out8 = OpFAdd %float %val_f8 %omega_diff8

    ; ========================================
    ; Streaming: write f_out[k] to neighbor (x+cx[k], y+cy[k]) with periodic wrap
    ; ========================================

    ; Convert dimensions to int for signed modulo
    %W_int = OpConvertUToF %float %W  ; Need float for conversion path
    %H_int = OpConvertUToF %float %H
    ; Actually, we need to work with uint for modulo
    ; For periodic wrap: new_x = (x + cx + W) % W

    ; --- k=0: stream to (x, y) ---
    ; No offset, just write to same location
    %out_idx0 = OpIMul %uint %yW_x %uint_9
    %ptr_out0 = OpAccessChain %ptr_storage_float_out %var_out %int_0 %out_idx0
    OpStore %ptr_out0 %f_out0

    ; --- k=1: stream to (x+1, y) ---
    %xp1 = OpIAdd %uint %x %uint_1
    %xp1_W = OpUMod %uint %xp1 %W
    %yW_xp1 = OpIMul %uint %y %W
    %yW_xp1_add = OpIAdd %uint %yW_xp1 %xp1_W
    %out_idx1 = OpIMul %uint %yW_xp1_add %uint_9
    %ptr_out1 = OpAccessChain %ptr_storage_float_out %var_out %int_0 %out_idx1
    OpStore %ptr_out1 %f_out1

    ; --- k=2: stream to (x, y+1) ---
    %yp1 = OpIAdd %uint %y %uint_1
    %yp1_H = OpUMod %uint %yp1 %H
    %yp1_W = OpIMul %uint %yp1_H %W
    %yp1_W_x = OpIAdd %uint %yp1_W %x
    %out_idx2 = OpIMul %uint %yp1_W_x %uint_9
    %ptr_out2 = OpAccessChain %ptr_storage_float_out %var_out %int_0 %out_idx2
    OpStore %ptr_out2 %f_out2

    ; --- k=3: stream to (x-1, y) ---
    ; Need signed arithmetic for negative offset
    %x_int = OpConvertUToS %int %x
    %W_int_s = OpConvertUToS %int %W
    %xm1 = OpISub %int %x_int %int_1
    %xm1_W = OpIAdd %int %xm1 %W_int_s
    %xm1_mod = OpSMod %int %xm1_W %W_int_s
    %xm1_uint = OpConvertSToU %uint %xm1_mod
    %yW_xm1 = OpIMul %uint %y %W
    %yW_xm1_add = OpIAdd %uint %yW_xm1 %xm1_uint
    %out_idx3 = OpIMul %uint %yW_xm1_add %uint_9
    %ptr_out3 = OpAccessChain %ptr_storage_float_out %var_out %int_0 %out_idx3
    OpStore %ptr_out3 %f_out3

    ; --- k=4: stream to (x, y-1) ---
    %y_int = OpConvertUToS %int %y
    %H_int_s = OpConvertUToS %int %H
    %ym1 = OpISub %int %y_int %int_1
    %ym1_H = OpIAdd %int %ym1 %H_int_s
    %ym1_mod = OpSMod %int %ym1_H %H_int_s
    %ym1_uint = OpConvertSToU %uint %ym1_mod
    %ym1_W = OpIMul %uint %ym1_uint %W
    %ym1_W_x = OpIAdd %uint %ym1_W %x
    %out_idx4 = OpIMul %uint %ym1_W_x %uint_9
    %ptr_out4 = OpAccessChain %ptr_storage_float_out %var_out %int_0 %out_idx4
    OpStore %ptr_out4 %f_out4

    ; --- k=5: stream to (x+1, y+1) ---
    %xp1b = OpIAdd %uint %x %uint_1
    %xp1b_mod = OpUMod %uint %xp1b %W
    %yp1b = OpIAdd %uint %y %uint_1
    %yp1b_mod = OpUMod %uint %yp1b %H
    %yp1b_W = OpIMul %uint %yp1b_mod %W
    %yp1b_W_xp1 = OpIAdd %uint %yp1b_W %xp1b_mod
    %out_idx5 = OpIMul %uint %yp1b_W_xp1 %uint_9
    %ptr_out5 = OpAccessChain %ptr_storage_float_out %var_out %int_0 %out_idx5
    OpStore %ptr_out5 %f_out5

    ; --- k=6: stream to (x-1, y+1) ---
    %xm1b = OpISub %int %x_int %int_1
    %xm1b_W = OpIAdd %int %xm1b %W_int_s
    %xm1b_mod = OpSMod %int %xm1b_W %W_int_s
    %xm1b_uint = OpConvertSToU %uint %xm1b_mod
    %yp1c = OpIAdd %uint %y %uint_1
    %yp1c_mod = OpUMod %uint %yp1c %H
    %yp1c_W = OpIMul %uint %yp1c_mod %W
    %yp1c_W_xm1 = OpIAdd %uint %yp1c_W %xm1b_uint
    %out_idx6 = OpIMul %uint %yp1c_W_xm1 %uint_9
    %ptr_out6 = OpAccessChain %ptr_storage_float_out %var_out %int_0 %out_idx6
    OpStore %ptr_out6 %f_out6

    ; --- k=7: stream to (x-1, y-1) ---
    %xm1c = OpISub %int %x_int %int_1
    %xm1c_W = OpIAdd %int %xm1c %W_int_s
    %xm1c_mod = OpSMod %int %xm1c_W %W_int_s
    %xm1c_uint = OpConvertSToU %uint %xm1c_mod
    %ym1b = OpISub %int %y_int %int_1
    %ym1b_H = OpIAdd %int %ym1b %H_int_s
    %ym1b_mod = OpSMod %int %ym1b_H %H_int_s
    %ym1b_uint = OpConvertSToU %uint %ym1b_mod
    %ym1b_W = OpIMul %uint %ym1b_uint %W
    %ym1b_W_xm1 = OpIAdd %uint %ym1b_W %xm1c_uint
    %out_idx7 = OpIMul %uint %ym1b_W_xm1 %uint_9
    %ptr_out7 = OpAccessChain %ptr_storage_float_out %var_out %int_0 %out_idx7
    OpStore %ptr_out7 %f_out7

    ; --- k=8: stream to (x+1, y-1) ---
    %xp1c = OpIAdd %uint %x %uint_1
    %xp1c_mod = OpUMod %uint %xp1c %W
    %ym1c = OpISub %int %y_int %int_1
    %ym1c_H = OpIAdd %int %ym1c %H_int_s
    %ym1c_mod = OpSMod %int %ym1c_H %H_int_s
    %ym1c_uint = OpConvertSToU %uint %ym1c_mod
    %ym1c_W = OpIMul %uint %ym1c_uint %W
```
