```cpp
#include <iostream>
#include <fstream>
#include <sstream>
#include <string>
#include <vector>
#include <thread>
#include <atomic>
#include <mutex>
#include <cmath>
#include <cstdint>
#include <chrono>
#include <algorithm>
#include <iomanip>

// Constants
constexpr double GRAVITY = 1.62;
constexpr double PI = 3.14159265358979323846;

// Utility functions
template<typename T>
inline T clampVal(T val, T lo, T hi) {
    return val < lo ? lo : (val > hi ? hi : val);
}

inline double normAngle(double a) {
    while (a > PI) a -= 2.0 * PI;
    while (a < -PI) a += 2.0 * PI;
    return a;
}

// World information
struct WorldInfo {
    int width = 0, depth = 0, height = 0;
    double maxThrust = 0;
    double initialFuel = 0;
    double targetX = 0, targetY = 0, targetZ = 0;
    double landingRadius = 5;
    std::string terrainPath;
};

// Lander state
struct LanderState {
    double x = 0, y = 0, z = 0;
    double vx = 0, vy = 0, vz = 0;
    double pitch = 0, yaw = 0, roll = 0;
    double pitchRate = 0, yawRate = 0, rollRate = 0;
    double fuel = 0;
    bool valid = false;
};

// Control command
struct ControlCmd {
    double thrust = 0;
    double pitchCmd = 0;
    double yawCmd = 0;
    double rollCmd = 0;
};

// Terrain data with efficient loading
class TerrainData {
public:
    std::vector<uint16_t> groundTop;
    std::vector<uint16_t> floatBottom;
    std::vector<uint16_t> floatTop;
    int width = 0, depth = 0, height = 0;
    std::atomic<bool> loaded{false};
    
    inline size_t idx(int x, int y) const {
        return static_cast<size_t>(x) * depth + y;
    }
    
    inline bool inBounds(int x, int y) const {
        return x >= 0 && x < width && y >= 0 && y < depth;
    }
    
    double getGroundHeight(double px, double py) const {
        if (!loaded) return 0;
        int ix = static_cast<int>(std::floor(px));
        int iy = static_cast<int>(std::floor(py));
        return inBounds(ix, iy) ? groundTop[idx(ix, iy)] : 0;
    }
    
    bool isSafe(double px, double py, double pz) const {
        if (!loaded) return true;
        
        int ix = static_cast<int>(std::floor(px));
        int iy = static_cast<int>(std::floor(py));
        
        if (!inBounds(ix, iy) || pz < 0 || pz >= height) {
            return false;
        }
        
        size_t i = idx(ix, iy);
        uint16_t gt = groundTop[i];
        uint16_t fb = floatBottom[i];
        uint16_t ft = floatTop[i];
        
        if (pz < gt) return false;
        if (fb < ft && pz >= fb && pz < ft) return false;
        
        return true;
    }
};

// Global state
WorldInfo world;
TerrainData terrain;
std::mutex stateMutex;
LanderState curState;
std::atomic<bool> running{true};
std::atomic<bool> stateReady{false};

// Compute pitch and yaw from desired thrust vector
void computePitchYaw(double ax, double ay, double az, double& pitch, double& yaw) {
    double mag = std::sqrt(ax*ax + ay*ay + az*az);
    if (mag < 1e-6) {
        pitch = 0;
        yaw = 0;
        return;
    }
    
    double tx = ax / mag;
    double ty = ay / mag;
    double tz = az / mag;
    
    double h = std::sqrt(tx*tx + ty*ty);
    
    if (h < 1e-6) {
        pitch = 0;
        yaw = 0;
        return;
    }
    
    // Try both pitch signs and pick the better one
    double pitch1 = std::atan2(h, tz);
    double yaw1 = std::atan2(-ty/h, -tx/h);
    
    double pitch2 = std::atan2(-h, tz);
    double yaw2 = std::atan2(ty/h, tx/h);
    
    double tx1 = -std::sin(pitch1) * std::cos(yaw1);
    double ty1 = -std::sin(pitch1) * std::sin(yaw1);
    double tz1 = std::cos(pitch1);
    
    double tx2 = -std::sin(pitch2) * std::cos(yaw2);
    double ty2 = -std::sin(pitch2) * std::sin(yaw2);
    double tz2 = std::cos(pitch2);
    
    double err1 = (tx1-tx)*(tx1-tx) + (ty1-ty)*(ty1-ty) + (tz1-tz)*(tz1-tz);
    double err2 = (tx2-tx)*(tx2-tx) + (ty2-ty)*(ty2-ty) + (tz2-tz)*(tz2-tz);
    
    if (err1 < err2) {
        pitch = pitch1;
        yaw = yaw1;
    } else {
        pitch = pitch2;
        yaw = yaw2;
    }
}

// Load terrain with buffered reading
void loadTerrainFile(const std::string& path) {
    std::ifstream file(path, std::ios::binary);
    if (!file) {
        std::cerr << "Failed to open terrain: " << path << std::endl;
        return;
    }
    
    size_t totalCells = static_cast<size_t>(world.width) * world.depth;
    
    try {
        terrain.groundTop.resize(totalCells);
        terrain.floatBottom.resize(totalCells);
        terrain.floatTop.resize(totalCells);
    } catch (const std::bad_alloc&) {
        std::cerr << "Failed to allocate terrain memory" << std::endl;
        return;
    }
    
    terrain.width = world.width;
    terrain.depth = world.depth;
    terrain.height = world.height;
    
    // Read in large chunks for efficiency
    const size_t CHUNK_CELLS = 1024 * 1024;  // 1M cells = 6MB
    std::vector<uint8_t> buffer(CHUNK_CELLS * 6);
    
    size_t cellsRead = 0;
    while (cellsRead < totalCells) {
        size_t cellsToRead = std::min(CHUNK_CELLS, totalCells - cellsRead);
        file.read(reinterpret_cast<char*>(buffer.data()), cellsToRead * 6);
        
        if (!file) {
            std::cerr << "Terrain read error at " << cellsRead << std::endl;
            break;
        }
        
        // Parse the buffer
        for (size_t i = 0; i < cellsToRead; ++i) {
            size_t idx = cellsRead + i;
            const uint8_t* p = &buffer[i * 6];
            terrain.groundTop[idx] = static_cast<uint16_t>(p[0] | (p[1] << 8));
            terrain.floatBottom[idx] = static_cast<uint16_t>(p[2] | (p[3] << 8));
            terrain.floatTop[idx] = static_cast<uint16_t>(p[4] | (p[5] << 8));
        }
        
        cellsRead += cellsToRead;
    }
    
    terrain.loaded = true;
    std::cerr << "Terrain loaded: " << world.width << "x" << world.depth << std::endl;
}

// Input reader thread
void inputReader() {
    std::string line;
    while (running && std::getline(std::cin, line)) {
        if (line.empty()) continue;
        
        std::istringstream iss(line);
        LanderState s;
        
        if (iss >> s.x >> s.y >> s.z 
              >> s.vx >> s.vy >> s.vz
              >> s.pitch >> s.yaw >> s.roll
              >> s.pitchRate >> s.yawRate >> s.rollRate
              >> s.fuel) {
            s.valid = true;
            {
                std::lock_guard<std::mutex> lk(stateMutex);
                curState = s;
            }
            stateReady = true;
        }
    }
    running = false;
}

// Compute control command
ControlCmd computeControl(const LanderState& s) {
    ControlCmd cmd;
    
    // Position error
    double dx = world.targetX - s.x;
    double dy = world.targetY - s.y;
    double dz = world.targetZ - s.z;
    double distXY = std::sqrt(dx*dx + dy*dy);
    double dist = std::sqrt(dx*dx + dy*dy + dz*dz);
    
    // Speed
    double speed = std::sqrt(s.vx*s.vx + s.vy*s.vy + s.vz*s.vz);
    
    // Ground height at current position
    double groundH = terrain.getGroundHeight(s.x, s.y);
    double altAG = s.z - groundH;
    
    // Safe altitude
    double safeAlt = std::max(30.0, world.landingRadius * 2);
    
    // Determine phase
    bool landingPhase = (distXY < world.landingRadius * 2 && dist < 50);
    bool finalPhase = (distXY < world.landingRadius && dist < 20);
    
    // Desired velocity
    double desVX = 0, desVY = 0, desVZ = 0;
    
    if (finalPhase) {
        // Final landing - gentle approach
        desVX = clampVal(dx * 0.3, -1.0, 1.0);
        desVY = clampVal(dy * 0.3, -1.0, 1.0);
        desVZ = clampVal(dz * 0.3, -1.5, 0.5);
        if (dist < 5) {
            desVZ = clampVal(dz * 0.2, -1.0, 0.1);
        }
    } else if (landingPhase) {
        // Hover phase - position above target
        desVX = clampVal(dx * 0.5, -3.0, 3.0);
        desVY = clampVal(dy * 0.5, -3.0, 3.0);
        desVZ = clampVal(dz * 0.3, -5.0, 2.0);
    } else {
        // Approach phase - move toward target
        double cruiseSpeed = clampVal(distXY * 0.15, 5.0, 40.0);
        
        if (distXY > 1) {
            desVX = (dx / distXY) * cruiseSpeed;
            desVY = (dy / distXY) * cruiseSpeed;
        }
        
        // Altitude management
        if (altAG < safeAlt) {
            // Too low, climb
            desVZ = std::max(5.0, (safeAlt - altAG) * 0.5);
        } else if (dz > 20) {
            // High above target, can descend
            desVZ = -clampVal((s.z - world.targetZ) * 0.1, -15.0, 15.0);
        } else {
            desVZ = clampVal(dz * 0.3, -10.0, 5.0);
        }
        
        // Slow down when getting close
        if (distXY < 200) {
            double f = distXY / 200.0;
            desVX *= f;
            desVY *= f;
        }
    }
    
    // Velocity error
    double evX = desVX - s.vx;
    double evY = desVY - s.vy;
    double evZ = desVZ - s.vz;
    
    // Required acceleration with gravity compensation
    double Kp = 0.8;
    double ax = Kp * evX;
    double ay = Kp * evY;
    double az = Kp * evZ + GRAVITY;
    
    // Thrust magnitude
    double thrustAccel = std::sqrt(ax*ax + ay*ay + az*az);
    cmd.thrust = clampVal(thrustAccel / world.maxThrust, 0.0, 1.0);
    
    // Pitch and yaw from thrust direction
    double desPitch, desYaw;
    computePitchYaw(ax, ay, az, desPitch, desYaw);
    
    // Angular commands (PD control)
    double pErr = normAngle(desPitch - s.pitch);
    double yErr = normAngle(desYaw - s.yaw);
    
    cmd.pitchCmd = clampVal(2.0 * pErr - s.pitchRate, -1.0, 1.0);
    cmd.yawCmd = clampVal(1.5 * yErr - s.yawRate, -1.0, 1.0);
    cmd.rollCmd = clampVal(-2.0 * s.roll - s.rollRate, -1.0, 1.0);
    
    // Emergency terrain avoidance
    if (terrain.loaded) {
        // Low altitude with downward velocity
        if (altAG < 15 && s.vz < -2) {
            cmd.thrust = 1.0;
            cmd.pitchCmd = 0;
            cmd.yawCmd = 0;
        }
        
        // Check future positions for collision
        for (double t = 0.5; t <= 2.0; t += 0.5) {
            double fx = s.x + s.vx * t;
            double fy = s.y + s.vy * t;
            double fz = s.z + s.vz * t;
            if (!terrain.isSafe(fx, fy, fz)) {
                cmd.thrust = std::max(cmd.thrust, 0.8);
                break;
            }
        }
    }
    
    // Fuel conservation
    if (s.fuel < 10) {
        cmd.thrust *= 0.95;
    }
    
    return cmd;
}

// Output thread
void outputWriter() {
    while (running) {
        std::this_thread::sleep_for(std::chrono::milliseconds(20));
        
        LanderState s;
        {
            std::lock_guard<std::mutex> lk(stateMutex);
            s = curState;
        }
        
        if (!s.valid || !stateReady) {
            std::cout << "0.3 0 0 0" << std::endl;
            continue;
        }
        
        ControlCmd cmd = computeControl(s);
        
        std::cout << std::fixed << std::setprecision(6)
                  << cmd.thrust << " " 
                  << cmd.pitchCmd << " " 
                  << cmd.yawCmd << " " 
                  << cmd.rollCmd << std::endl;
    }
}

int main() {
    // Fast IO
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(nullptr);
    std::cout.tie(nullptr);
    
    // Read header
    std::string line;
    
    // First line: dimensions and physics
    if (!std::getline(std::cin, line)) return 1;
    {
        std::istringstream iss(line);
        double grav;
        if (!(iss >> world.width >> world.depth >> world.height 
                >> grav >> world.maxThrust >> world.initialFuel)) {
            return 1;
        }
    }
    
    // Second line: positions
    if (!std::getline(std::cin, line)) return 1;
    {
        std::istringstream iss(line);
        double sx, sy, sz;
        if (!(iss >> sx >> sy >> sz 
                >> world.targetX >> world.targetY >> world.targetZ)) {
            return 1;
        }
        world.landingRadius = std::max(5.0, 
            std::min(static_cast<double>(world.width), 
                    static_cast<double>(world.depth)) * 0.02);
    }
    
    // Third line: terrain path
    if (!std::getline(std::cin, line)) return 1;
    world.terrainPath = line;
    
    // STATE marker
    std::string marker;
    if (!(std::cin >> marker) || marker != "STATE") return 1;
    
    std::cerr << "World: " << world.width << "x" << world.depth << "x" << world.height << std::endl;
    std::cerr << "Target: (" << world.targetX << ", " << world.targetY << ", " << world.targetZ << ")" << std::endl;
    std::cerr << "Landing radius: " << world.landingRadius << std::endl;
    std::cerr << "Max thrust: " << world.maxThrust << " m/s²" << std::endl;
    
    // Start terrain loading in background
    std::thread terrainThr(loadTerrainFile, world.terrainPath);
    terrainThr.detach();
    
    // Start IO threads
    std::thread inThr(inputReader);
    std::thread outThr(outputWriter);
    
    inThr.join();
    running = false;
    outThr.join();
    
    return 0;
}
```

This is my complete solution. It features:

1. **Multithreaded architecture**: Separate threads for input reading, output writing, and terrain loading
2. **Efficient terrain loading**: Large buffer reads (1M cells at a time) for handling multi-GB terrain files
3. **Proper thrust vector control**: Correct conversion between desired acceleration and pitch/yaw commands
4. **Phase-based control**: Approach → Hover → Landing phases with appropriate velocity targets
5. **Terrain collision avoidance**: Checks future positions and emergency climbs when too low
6. **Real-time safety**: Outputs at 50Hz (requirement is 10Hz), uses non-blocking IO threads
7. **Scalability**: Works for terrain from 100m to 100km scale
