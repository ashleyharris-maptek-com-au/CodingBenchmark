Looking at this problem, I'll implement a solution that uses the exact Dreyfus-Wagner DP algorithm for small terminal sets (k ≤ 15-20) and falls back to a Steiner tree heuristic for larger instances.

**Approach:**
1. **Dreyfus-Wagner DP**: O(3^k × n + 2^k × n²) - optimal for small k
   - Computes minimum Steiner tree connecting subsets of terminals
   - Uses Floyd-Warshall for all-pairs shortest paths

2. **Heuristic (large k)**: Shortest Path Heuristic (SPH)
   - Iteratively connect terminals via shortest paths

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <tuple>
#include <unordered_set>
#include <cstdint>

using namespace std;

using ll = long long;
using vi = vector<int>;
using vll = vector<ll>;
using vvi = vector<vi>;
using vvll = vector<vll>;

const ll INF = LLONG_MAX / 4;

// Floyd-Warshall for all-pairs shortest paths
void floyd_warshall(int n, vvll& dist, vvi& nxt) {
    for (int k = 0; k < n; k++) {
        for (int i = 0; i < n; i++) {
            if (dist[i][k] == INF) continue;
            for (int j = 0; j < n; j++) {
                if (dist[k][j] == INF) continue;
                if (dist[i][k] + dist[k][j] < dist[i][j]) {
                    dist[i][j] = dist[i][k] + dist[k][j];
                    nxt[i][j] = nxt[i][k];
                }
            }
        }
    }
}

// Reconstruct path between u and v using next matrix
vi reconstruct_path(int u, int v, const vvi& nxt) {
    vi path;
    if (nxt[u][v] == -1) return path;
    path.push_back(u);
    while (u != v) {
        u = nxt[u][v];
        path.push_back(u);
    }
    return path;
}

// Dreyfus-Wagner algorithm for exact Steiner tree
ll dreyfus_wagner(int n, int k, const vi& terminals, const vvll& dist, const vvi& nxt,
                  vvi& result_edges) {
    if (k == 0) return 0;
    if (k == 1) { result_edges.clear(); return 0; }
    
    // dp[mask][v] = min Steiner tree connecting terminals in mask, rooted at v
    int K = min(k, 16);
    vector<vll> dp(1 << K, vll(n, INF));
    
    // Base case: single terminal
    for (int i = 0; i < K; i++) {
        int t = terminals[i];
        for (int v = 0; v < n; v++) {
            dp[1 << i][v] = dist[t][v];
        }
    }
    
    // DP transitions
    for (int mask = 1; mask < (1 << K); mask++) {
        if (__popcount(mask) == 1) continue;
        
        // For each root v
        for (int v = 0; v < n; v++) {
            ll best = INF;
            
            // Try all partitions mask1 + mask2 = mask
            for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
                int other = mask ^ sub;
                if (other == 0) continue;
                if (dp[sub][v] < INF && dp[other][v] < INF) {
                    best = min(best, dp[sub][v] + dp[other][v]);
                }
            }
            dp[mask][v] = best;
            
            // Propagate to other roots via shortest paths
            for (int u = 0; u < n; u++) {
                if (dist[v][u] < INF && best < INF) {
                    dp[mask][u] = min(dp[mask][u], best + dist[v][u]);
                }
            }
        }
    }
    
    ll min_cost = dp[(1 << K) - 1][terminals[0]];
    int best_root = terminals[0];
    for (int v = 0; v < n; v++) {
        if (dp[(1 << K) - 1][v] < min_cost) {
            min_cost = dp[(1 << K) - 1][v];
            best_root = v;
        }
    }
    
    // Reconstruct solution using backtracking
    unordered_set<long long> edge_set;
    auto add_edge = [&](int u, int v) {
        if (u > v) swap(u, v);
        edge_set.insert((ll)u * n + v);
    };
    
    function<void(int, int)> recover = [&](int mask, int v) {
        if (__popcount(mask) == 1) {
            int i = __countr_zero(mask);
            int t = terminals[i];
            vi path = reconstruct_path(v, t, nxt);
            for (int i = 0; i + 1 < (int)path.size(); i++) {
                add_edge(path[i], path[i + 1]);
            }
            return;
        }
        
        // Find valid partition
        for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
            int other = mask ^ sub;
            if (other == 0) continue;
            
            ll cost_sub = dp[sub][v];
            ll cost_other = dp[other][v];
            
            // Check if this partition gives optimal cost
            if (cost_sub + cost_other == dp[mask][v]) {
                recover(sub, v);
                recover(other, v);
                return;
            }
        }
        
        // If no partition found, check if we came from another root
        for (int u = 0; u < n; u++) {
            if (u == v) continue;
            if (dist[u][v] < INF && dp[mask][u] + dist[u][v] == dp[mask][v]) {
                vi path = reconstruct_path(u, v, nxt);
                for (int i = 0; i + 1 < (int)path.size(); i++) {
                    add_edge(path[i], path[i + 1]);
                }
                recover(mask, u);
                return;
            }
        }
    };
    
    if (min_cost < INF) {
        recover((1 << K) - 1, best_root);
    }
    
    result_edges.clear();
    for (auto& e : edge_set) {
        int u = (int)(e / n);
        int v = (int)(e % n);
        result_edges.push_back({u, v});
    }
    
    return min_cost;
}

// Dijkstra's algorithm
vll dijkstra(int n, int src, const vector<vector<pair<int, ll>>>& adj) {
    vll dist(n, INF);
    dist[src] = 0;
    priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<>> pq;
    pq.push({0, src});
    
    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (d > dist[u]) continue;
        for (auto& [v, w] : adj[u]) {
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
    return dist;
}

// Shortest Path Heuristic for Steiner tree
ll shortest_path_heuristic(int n, const vector<vector<pair<int, ll>>>& adj,
                           const vi& terminals, vvi& result_edges) {
    int k = (int)terminals.size();
    if (k == 0) return 0;
    
    vector<bool> in_tree(n, false);
    vector<bool> terminal_in_tree(k, false);
    unordered_set<ll> selected_edges;
    
    auto add_edge = [&](int u, int v) {
        if (u > v) swap(u, v);
        selected_edges.insert((ll)u * n + v);
    };
    
    // Start with first terminal
    in_tree[terminals[0]] = true;
    terminal_in_tree[0] = true;
    int terminals_connected = 1;
    
    while (terminals_connected < k) {
        // Find closest terminal not in tree
        ll best_dist = INF;
        vi best_path;
        int best_term = -1;
        
        for (int i = 0; i < k; i++) {
            if (terminal_in_tree[i]) continue;
            int t = terminals[i];
            
            // BFS/Dijkstra from tree to find path to terminal t
            vll dist(n, INF);
            vi parent(n, -1);
            priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<>> pq;
            
            for (int v = 0; v < n; v++) {
                if (in_tree[v]) {
                    dist[v] = 0;
                    pq.push({0, v});
                }
            }
            
            while (!pq.empty()) {
                auto [d, u] = pq.top();
                pq.pop();
                if (d > dist[u]) continue;
                if (u == t) break;
                for (auto& [v, w] : adj[u]) {
                    if (dist[u] + w < dist[v]) {
                        dist[v] = dist[u] + w;
                        parent[v] = u;
                        pq.push({dist[v], v});
                    }
                }
            }
            
            if (dist[t] < best_dist) {
                best_dist = dist[t];
                best_term = i;
                best_path.clear();
                int cur = t;
                while (parent[cur] != -1) {
                    best_path.push_back(cur);
                    cur = parent[cur];
                }
                best_path.push_back(cur);
                reverse(best_path.begin(), best_path.end());
            }
        }
        
        if (best_term == -1 || best_dist == INF) break;
        
        // Add path to tree
        for (int i = 0; i + 1 < (int)best_path.size(); i++) {
            int u = best_path[i];
            int v = best_path[i + 1];
            add_edge(u, v);
            in_tree[u] = true;
            in_tree[v] = true;
        }
        
        terminal_in_tree[best_term] = true;
        terminals_connected++;
    }
    
    result_edges.clear();
    for (auto& e : selected_edges) {
        int u = (int)(e / n);
        int v = (int)(e % n);
        result_edges.push_back({u, v});
    }
    
    // Compute total weight
    ll total = 0;
    for (auto& [u, v] : result_edges) {
        for (auto& [to, w] : adj[u]) {
            if (to == v) { total += w; break; }
        }
    }
    
    return total;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m, k;
    cin >> n >> m >> k;
    
    vector<vector<pair<int, ll>>> adj(n);
    vvll dist(n, vll(n, INF));
    vvi nxt(n, vi(n, -1));
    
    // Store original edges for output
    vector<tuple<int, int, ll>> edges;
    
    for (int i = 0; i < m; i++) {
        int u, v;
        ll w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
        edges.push_back({u, v, w});
        
        if (w < dist[u][v]) {
            dist[u][v] = dist[v][u] = w;
            nxt[u][v] = v;
            nxt[v][u] = u;
        }
    }
    
    vi terminals(k);
    for (int i = 0; i < k; i++) {
        cin >> terminals[i];
    }
    
    // Initialize diagonal for Floyd-Warshall
    for (int i = 0; i < n; i++) {
        dist[i][i] = 0;
        nxt[i][i] = i;
    }
    
    // Run Floyd-Warshall
    floyd_warshall(n, dist, nxt);
    
    ll total_weight = 0;
    vvi result_edges;
    
    const int DP_THRESHOLD = 18;
    
    if (k <= DP_THRESHOLD) {
        // Use exact Dreyfus-Wagner algorithm
        total_weight = dreyfus_wagner(n, k, terminals, dist, nxt, result_edges);
    }
    
    // If DP failed or too many terminals, use heuristic
    if (k > DP_THRESHOLD || total_weight >= INF) {
        total_weight = shortest_path_heuristic(n, adj, terminals, result_edges);
    }
    
    // Output result
    cout << total_weight << " " << result_edges.size() << "\n";
    for (auto& e : result_edges) {
        cout << e[0] << " " << e[1] << "\n";
    }
    
    return 0;
}
```

**Key Features:**

1. **Dreyfus-Wagner DP** (for k ≤ 18):
   - `dp[mask][v]` = minimum Steiner tree cost connecting terminals in `mask`, with tree rooted at vertex `v`
   - Time complexity: O(3^k × n) for DP transitions
   - Uses Floyd-Warshall precomputed shortest paths

2. **Shortest Path Heuristic** (for k > 18):
   - Iteratively connects the closest terminal to the current tree
   - Uses Dijkstra from all tree vertices simultaneously
   - Practical for large terminal sets

3. **Path Reconstruction**:
   - Uses next-pointer matrix from Floyd-Warshall
   - Backtracking through DP table for exact solution

4. **MSVC Compatibility**:
   - Uses `__popcount` and `__countr_zero` intrinsics supported by MSVC
   - Standard headers only (`<iostream>`, `<vector>`, etc.)
