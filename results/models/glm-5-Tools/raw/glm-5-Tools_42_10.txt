```spirv
; SPIR-V Assembly for Geometry Shader - Sine Wave Curve with Adjacency
; Vulkan 1.0

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

; Decorations for gl_PerVertex blocks
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Input variables decorations (locations 0, 1, 2)
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

; Output variable decoration
OpDecorate %outColor Location 0

; UBO layout (set=0, binding=0)
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 204
OpMemberDecorate %UBO 5 Offset 216
OpMemberDecorate %UBO 6 Offset 232
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_02 = OpConstant %float 0.02
%float_0_5 = OpConstant %float 0.5
%float_2 = OpConstant %float 2.0
%float_4 = OpConstant %float 4.0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%uint_4 = OpConstant %uint 4

; gl_PerVertex input type (array of 4 for InputLinesAdjacency)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in

; gl_PerVertex output type
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out

; Input arrays for user-defined inputs (size 4)
%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4

; Output color
%ptr_out_v3 = OpTypePointer Output %v3float

; UBO type
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO

; Runtime pointers
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_uf = OpTypePointer Uniform %float

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output
%ubo = OpVariable %ptr_ubo Uniform

%main = OpFunction %void None %func
%entry = OpLabel

; Get clip space positions from gl_in
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0

%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr
%p3 = OpLoad %v4float %p3_ptr

; Get world positions for tangent calculation
%w0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%w1_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_1
%w2_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_2
%w3_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_3

%w0 = OpLoad %v3float %w0_ptr
%w1 = OpLoad %v3float %w1_ptr
%w2 = OpLoad %v3float %w2_ptr
%w3 = OpLoad %v3float %w3_ptr

; Get colors for interpolation
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c1 = OpLoad %v3float %c1_ptr
%c2 = OpLoad %v3float %c2_ptr

; Get normals for perpendicular calculation
%n1_ptr = OpAccessChain %ptr_in_v3 %normal %int_1
%n2_ptr = OpAccessChain %ptr_in_v3 %normal %int_2
%n1 = OpLoad %v3float %n1_ptr
%n2 = OpLoad %v3float %n2_ptr

; Extract XY from clip space positions
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p3_xy = OpVectorShuffle %v2float %p3 %p3 0 1

; Calculate tangent at P1: direction from P0 to P2 (Catmull-Rom style)
%t1_dir = OpFSub %v2float %p2_xy %p0_xy

; Calculate tangent at P2: direction from P1 to P3 (Catmull-Rom style)
%t2_dir = OpFSub %v2float %p3_xy %p1_xy

; Calculate deltas for position interpolation
%delta_xy = OpFSub %v2float %p2_xy %p1_xy
%delta_z = OpFSub %float %p2 %p1
%delta_w = OpFSub %float %p2 %p1
%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3

; Subdivide into 4 segments (5 points: t = 0.0, 0.25, 0.5, 0.75, 1.0)

; === Point 0 (t = 0.0) ===
%t0 = OpCopyObject %float %float_0

; Interpolate tangent (lerp between t1_dir and t2_dir)
%t0_tangent = OpFMul %v2float %t1_dir %float_1

; Perpendicular: rotate tangent 90 degrees (-y, x)
%t0_perp_x = OpCompositeExtract %float %t0_tangent 1
%t0_perp_y_neg = OpCompositeExtract %float %t0_tangent 0
%t0_perp_y = OpFNegate %float %t0_perp_y_neg
%t0_perp = OpCompositeConstruct %v2float %t0_perp_x %t0_perp_y

; Normalize perpendicular
%t0_len_sq = OpDot %float %t0_perp %t0_perp
%t0_len = OpExtInst %float %1 Sqrt %t0_len_sq
%t0_perp_norm = OpFDiv %v2float %t0_perp %t0_len

; Scale by thickness
%t0_offset = OpFMul %v2float %t0_perp_norm %float_0_02

; Vertex 0: pos + offset
%v0_xy = OpFAdd %v2float %p1_xy %t0_offset
%v0_pos = OpCompositeConstruct %v4float %v0_xy %p1_z %p1_w
%v0_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v0_out_ptr %v0_pos
OpStore %outColor %c1
OpEmitVertex

; Vertex 1: pos - offset
%v1_xy = OpFSub %v2float %p1_xy %t0_offset
%v1_pos = OpCompositeConstruct %v4float %v1_xy %p1_z %p1_w
%v1_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v1_out_ptr %v1_pos
OpStore %outColor %c1
OpEmitVertex

; === Point 1 (t = 0.25) ===
%t1_val = OpFMul %float %float_0_5 %float_0_5
%t1_inv = OpFSub %float %float_1 %t1_val

; Interpolate position
%pos1_xy = OpFMul %v2float %delta_xy %t1_val
%pos1_xy_final = OpFAdd %v2float %p1_xy %pos1_xy

; Interpolate z and w
%pos1_z = OpFFma %float %delta_z %t1_val %p1_z
%pos1_w = OpFFma %float %delta_w %t1_val %p1_w

; Interpolate tangent
%t1_tangent_scaled = OpFMul %v2float %t2_dir %t1_val
%t1_tangent = OpFAdd %v2float %t1_dir %t1_tangent_scaled
%t1_inv_scaled = OpFMul %v2float %t1_tangent %t1_inv
%t1_tangent_final = OpFMul %v2float %t1_tangent %float_1

; Actually just lerp tangents
%t1_tan_scale = OpFSub %float %float_1 %t1_val
%t1_tan_1 = OpFMul %v2float %t1_dir %t1_tan_scale
%t1_tan_2 = OpFMul %v2float %t2_dir %t1_val
%t1_tangent_lerp = OpFAdd %v2float %t1_tan_1 %t1_tan_2

; Perpendicular
%t1_perp_x = OpCompositeExtract %float %t1_tangent_lerp 1
%t1_perp_y_neg = OpCompositeExtract %float %t1_tangent_lerp 0
%t1_perp_y = OpFNegate %float %t1_perp_y_neg
%t1_perp = OpCompositeConstruct %v2float %t1_perp_x %t1_perp_y
%t1_len_sq = OpDot %float %t1_perp %t1_perp
%t1_len = OpExtInst %float %1 Sqrt %t1_len_sq
%t1_perp_norm = OpFDiv %v2float %t1_perp %t1_len
%t1_offset = OpFMul %v2float %t1_perp_norm %float_0_02

; Color interpolation
%c1_val = OpFMul %v3float %c1 %t1_tan_scale
%c1_val2 = OpFMul %v3float %c2 %t1_val
%color1 = OpFAdd %v3float %c1_val %c1_val2

; Vertex 2: pos + offset
%v2_xy_new = OpFAdd %v2float %pos1_xy_final %t1_offset
%v2_pos = OpCompositeConstruct %v4float %v2_xy_new %pos1_z %pos1_w
%v2_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v2_out_ptr %v2_pos
OpStore %outColor %color1
OpEmitVertex

; Vertex 3: pos - offset
%v3_xy = OpFSub %v2float %pos1_xy_final %t1_offset
%v3_pos = OpCompositeConstruct %v4float %v3_xy %pos1_z %pos1_w
%v3_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v3_out_ptr %v3_pos
OpStore %outColor %color1
OpEmitVertex

; === Point 2 (t = 0.5) ===
%t2_val = OpCopyObject %float %float_0_5
%t2_inv = OpFSub %float %float_1 %t2_val

; Interpolate position
%pos2_xy = OpFMul %v2float %delta_xy %t2_val
%pos2_xy_final = OpFAdd %v2float %p1_xy %pos2_xy
%pos2_z = OpFFma %float %delta_z %t2_val %p1_z
%pos2_w = OpFFma %float %delta_w %t2_val %p1_w

; Interpolate tangent
%t2_tan_1 = OpFMul %v2float %t1_dir %t2_inv
%t2_tan_2 = OpFMul %v2float %t2_dir %t2_val
%t2_tangent_lerp = OpFAdd %v2float %t2_tan_1 %t2_tan_2

; Perpendicular
%t2_perp_x = OpCompositeExtract %float %t2_tangent_lerp 1
%t2_perp_y_neg = OpCompositeExtract %float %t2_tangent_lerp 0
%t2_perp_y = OpFNegate %float %t2_perp_y_neg
%t2_perp = OpCompositeConstruct %v2float %t2_perp_x %t2_perp_y
%t2_len_sq = OpDot %float %t2_perp %t2_perp
%t2_len = OpExtInst %float %1 Sqrt %t2_len_sq
%t2_perp_norm = OpFDiv %v2float %t2_perp %t2_len
%t2_offset = OpFMul %v2float %t2_perp_norm %float_0_02

; Color interpolation
%col2_1 = OpFMul %v3float %c1 %t2_inv
%col2_2 = OpFMul %v3float %c2 %t2_val
%color2 = OpFAdd %v3float %col2_1 %col2_2

; Vertex 4: pos + offset
%v4_xy = OpFAdd %v2float %pos2_xy_final %t2_offset
%v4_pos = OpCompositeConstruct %v4float %v4_xy %pos2_z %pos2_w
%v4_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v4_out_ptr %v4_pos
OpStore %outColor %color2
OpEmitVertex

; Vertex 5: pos - offset
%v5_xy = OpFSub %v2float %pos2_xy_final %t2_offset
%v5_pos = OpCompositeConstruct %v4float %v5_xy %pos2_z %pos2_w
%v5_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v5_out_ptr %v5_pos
OpStore %outColor %color2
OpEmitVertex

; === Point 3 (t = 0.75) ===
%t3_val = OpFMul %float %float_0_5 %float_0_5
%t3_val_scaled = OpFMul %float %t3_val %float_4
%t3_val_final = OpFAdd %float %float_0_5 %t3_val_scaled
%t3_inv = OpFSub %float %float_1 %t3_val_final

; Actually: t = 0.75
%t3_v = OpCopyObject %float %float_0_5
%t3_v_adj = OpFMul %float %float_2 %t3_v
%t3_v_final = OpFSub %float %t3_v_adj %float_0_5
; Let's use constant 0.75
%float_0_75_num = OpFAdd %float %float_0_5 %float_0_5
%float_0_75 = OpFMul %float %float_0_75_num %float_0_5
%t3_real = OpFAdd %float %float_0_5 %float_0_25
; Use proper constant calculation
%float_0_25 = OpFMul %float %float_0_5 %float_0_5
%t3_correct = OpFAdd %float %float_0_5 %float_0_25

; Interpolate position
%pos3_xy = OpFMul %v2float %delta_xy %t3_correct
%pos3_xy_final = OpFAdd %v2float %p1_xy %pos3_xy
%pos3_z = OpFFma %float %delta_z %t3_correct %p1_z
%pos3_w = OpFFma %float %delta_w %t3_correct %p1_w

; Interpolate tangent
%t3_inv_val = OpFSub %float %float_1 %t3_correct
%t3_tan_1 = OpFMul %v2float %t1_dir %t3_inv_val
%t3_tan_2 = OpFMul %v2float %t2_dir %t3_correct
%t3_tangent_lerp = OpFAdd %v2float %t3_tan_1 %t3_tan_2

; Perpendicular
%t3_perp_x = OpCompositeExtract %float %t3_tangent_lerp 1
%t3_perp_y_neg = OpCompositeExtract %float %t3_tangent_lerp 0
%t3_perp_y = OpFNegate %float %t3_perp_y_neg
%t3_perp = OpCompositeConstruct %v2float %t3_perp_x %t3_perp_y
%t3_len_sq = OpDot %float %t3_perp %t3_perp
%t3_len = OpExtInst %float %1 Sqrt %t3_len_sq
%t3_perp_norm = OpFDiv %v2float %t3_perp %t3_len
%t3_offset = OpFMul %v2float %t3_perp_norm %float_0_02

; Color interpolation
%col3_1 = OpFMul %v3float %c1 %t3_inv_val
%col3_2 = OpFMul %v3float %c2 %t3_correct
%color3 = OpFAdd %v3float %col3_1 %col3_2

; Vertex 6: pos + offset
%v6_xy = OpFAdd %v2float %pos3_xy_final %t3_offset
%v6_pos = OpCompositeConstruct %v4float %v6_xy %pos3_z %pos3_w
%v6_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v6_out_ptr %v6_pos
OpStore %outColor %color3
OpEmitVertex

; Vertex 7: pos - offset
%v7_xy = OpFSub %v2float %pos3_xy_final %t3_offset
%v7_pos = OpCompositeConstruct %v4float %v7_xy %pos3_z %pos3_w
%v7_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v7_out_ptr %v7_pos
OpStore %outColor %color3
OpEmitVertex

; === Point 4 (t = 1.0) ===
%t4 = OpCopyObject %float %float_1

; Position is p2
%p2_z = OpCompositeExtract %float %p2 2
%p2_w = OpCompositeExtract %float %p2 3

; Tangent at end is t2_dir
; Perpendicular
%t4_perp_x = OpCompositeExtract %float %t2_dir 1
%t4_perp_y_neg = OpCompositeExtract %float %t2_dir 0
%t4_perp_y = OpFNegate %float %t4_perp_y_neg
%t4_perp = OpCompositeConstruct %v2float %t4_perp_x %t4_perp_y
%t4_len_sq = OpDot %float %t4_perp %t4_perp
%t4_len = OpExtInst %float %1 Sqrt %t4_len_sq
%t4_perp_norm = OpFDiv %v2float %t4_perp %t4_len
%t4_offset = OpFMul %v2float %t4_perp_norm %float_0_02

; Vertex 8: pos + offset
%v8_xy = OpFAdd %v2float %p2_xy %t4_offset
%v8_pos = OpCompositeConstruct %v4float %v8_xy %p2_z %p2_w
%v8_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v8_out_ptr %v8_pos
OpStore %outColor %c2
OpEmitVertex

; Vertex 9: pos - offset
%v9_xy = OpFSub %v2float %p2_xy %t4_offset
%v9_pos = OpCompositeConstruct %v4float %v9_xy %p2_z %p2_w
%v9_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v9_out_ptr %v9_pos
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
