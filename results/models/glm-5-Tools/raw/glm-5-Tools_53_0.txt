import math

# ============================================================================
# MODULE-LEVEL STATE (persists across calls within a scenario)
# ============================================================================

_state = {
    'initialized': False,
    'phase': 'approach',
    'pos_estimate': [0.0, -200.0, 0.0],
    'vel_estimate': [0.0, 0.0, 0.0],
    'step': 0,
    'accum_time': 0.0,
    'last_thrust': [0.0, 0.0, 0.0],
    'pos_history': [],
    'detected_bias': [0.0, 0.0, 0.0],
}

# ============================================================================
# UTILITIES
# ============================================================================

def clamp(v, lo, hi):
    return max(lo, min(hi, v))

def sign(x):
    if x > 1e-9: return 1.0
    if x < -1e-9: return -1.0
    return 0.0

def norm(v):
    return math.sqrt(sum(x*x for x in v))

# ============================================================================
# SENSOR READING WITH REDUNDANCY
# ============================================================================

def read_relative_state(s):
    """Fuse radar and lidar for best estimate."""
    pos = [None, None, None]
    vel = [None, None, None]
    
    for i, ax in enumerate(['x', 'y', 'z']):
        # Position
        readings = []
        v = s.get(f'radar_rel_pos_{ax}_m', 0)
        if s.get(f'radar_rel_{ax}_valid', False): readings.append(v)
        v = s.get(f'lidar_rel_pos_{ax}_m', 0)
        if s.get(f'lidar_rel_{ax}_valid', False): readings.append(v)
        if readings: pos[i] = sum(readings) / len(readings)
        
        # Velocity
        readings = []
        v = s.get(f'radar_rel_vel_{ax}_ms', 0)
        if s.get(f'radarv_{ax}_valid', False): readings.append(v)
        v = s.get(f'lidar_rel_vel_{ax}_ms', 0)
        if s.get(f'lidarv_{ax}_valid', False): readings.append(v)
        if readings: vel[i] = sum(readings) / len(readings)
    
    return pos, vel

def read_range(s):
    """Get range from range finders."""
    vals = []
    for i in [1, 2, 3]:
        r = s.get(f'range_{i}_m', 0)
        if s.get(f'range_{i}_valid', False) and r > 0: 
            vals.append(r)
    return sum(vals)/len(vals) if vals else None

def read_dock_camera(s):
    """Get docking camera data."""
    if not s.get('dock_camera_active', False):
        return None, None
    b = [s.get(f'dock_camera_bearing_{ax}', 0) for ax in ['x','y','z']]
    v = [s.get(f'dock_camera_bearing_{ax}_valid', False) for ax in ['x','y','z']]
    r = s.get('dock_camera_range_m', None)
    if all(v) and r is not None: return b, r
    return None, None

def read_gyro_rates(s):
    """Get averaged gyro rates in rad/s."""
    rates = {'roll': [], 'pitch': [], 'yaw': []}
    for i in [1, 2, 3]:
        for ax in ['roll', 'pitch', 'yaw']:
            val = s.get(f'gyro_{i}_{ax}_dps', 0)
            if s.get(f'gyro_{i}_{ax}_valid', False):
                rates[ax].append(val * math.pi / 180)
    return {ax: sum(rates[ax])/len(rates[ax]) if rates[ax] else 0.0 for ax in rates}

def read_attitude(s):
    """Get averaged attitude from star trackers."""
    att = {'roll': [], 'pitch': [], 'yaw': []}
    for i in [1, 2]:
        for ax in ['roll', 'pitch', 'yaw']:
            val = s.get(f'star_tracker_{i}_{ax}_deg', 0)
            if s.get(f'star_tracker_{i}_{ax}_valid', False):
                att[ax].append(val)
    return {ax: sum(att[ax])/len(att[ax]) if att[ax] else 0.0 for ax in att}

# ============================================================================
# STATE ESTIMATION WITH DEAD RECKONING
# ============================================================================

def estimate_state(pos, vel, dt):
    """Update state estimate with dead reckoning fallback."""
    global _state
    
    if pos[0] is not None:
        _state['pos_estimate'] = pos[:]
    else:
        for i in range(3):
            _state['pos_estimate'][i] += _state['vel_estimate'][i] * dt
    
    if vel[0] is not None:
        _state['vel_estimate'] = vel[:]
    
    return _state['pos_estimate'][:], _state['vel_estimate'][:]

# ============================================================================
# CONTROL LAWS - RANGE DEPENDENT
# ============================================================================

def control_far(pos, vel, dt, fuel):
    """
    Far range (>200m): Orbital mechanics approach.
    
    CRITICAL: In orbit, relative motion is COUNTER-INTUITIVE!
    - If BEHIND (y < 0): need to SPEED UP → RETROGRADE thrust (-Y)
    - If AHEAD (y > 0): need to SLOW DOWN → PROGRADE thrust (+Y)
    """
    x, y, z = pos
    vx, vy, vz = vel
    tx, ty, tz = 0.0, 0.0, 0.0
    
    # Along-track (Y): Main approach - use orbital mechanics
    if abs(y) > 800:
        ty = sign(y) * 0.40  # If y>0 (ahead), +thrust to slow; if y<0, -thrust to catch up
    elif abs(y) > 500:
        ty = sign(y) * 0.30
    elif abs(y) > 300:
        ty = sign(y) * 0.20
    elif abs(y) > 200:
        ty = sign(y) * 0.12
    else:
        ty = sign(y) * 0.06
    
    # Radial (X): Correct offset
    if abs(x) > 100:
        tx = -sign(x) * 0.15
    elif abs(x) > 40:
        tx = -sign(x) * 0.08
    elif abs(x) > 15:
        tx = -sign(x) * 0.03
    else:
        tx = -x * 0.0015
    
    # Cross-track (Z): Correct offset
    if abs(z) > 100:
        tz = -sign(z) * 0.15
    elif abs(z) > 40:
        tz = -sign(z) * 0.08
    elif abs(z) > 15:
        tz = -sign(z) * 0.03
    else:
        tz = -z * 0.0015
    
    return tx, ty, tz

def control_medium(pos, vel, dt, fuel):
    """Medium range (50-200m): V-bar approach with CW dynamics."""
    x, y, z = pos
    vx, vy, vz = vel
    
    target_speed = 0.8 + 1.2 * min(1.0, abs(y) / 150.0)
    desired_vy = sign(-y) * target_speed if abs(y) > 10 else 0
    
    kp, kd = 0.003, 0.10
    ty = kp * (-y) + kd * (desired_vy - vy)
    tx = -kp * x - kd * vx
    tz = -kp * z - kd * vz
    
    max_t = 0.30
    return clamp(tx,-max_t,max_t), clamp(ty,-max_t,max_t), clamp(tz,-max_t,max_t)

def control_close(pos, vel, dt, fuel):
    """Close range (10-50m): Linear regime, careful approach."""
    x, y, z = pos
    vx, vy, vz = vel
    
    target_speed = 0.2 + 0.3 * min(1.0, abs(y) / 50.0)
    desired_vy = sign(-y) * target_speed if abs(y) > 3 else 0
    
    kp, kd = 0.005, 0.15
    ty = kp * (-y) + kd * (desired_vy - vy)
    tx = -kp * x - kd * vx
    tz = -kp * z - kd * vz
    
    max_t = 0.22
    return clamp(tx,-max_t,max_t), clamp(ty,-max_t,max_t), clamp(tz,-max_t,max_t)

def control_final(pos, vel, dt, dock_bearing, dock_range, fuel):
    """Final approach (<10m): Very gentle, use docking camera."""
    x, y, z = pos
    vx, vy, vz = vel
    
    # Blend in docking camera if available
    if dock_bearing is not None and dock_range is not None and dock_range < 12:
        w = 0.4
        x = w * (-dock_bearing[0] * dock_range) + (1-w) * x
        y = w * (-dock_bearing[1] * dock_range) + (1-w) * y
        z = w * (-dock_bearing[2] * dock_range) + (1-w) * z
    
    target_speed = 0.04 + 0.06 * min(1.0, abs(y) / 10.0)
    desired_vy = sign(-y) * target_speed if abs(y) > 0.5 else 0
    
    kp, kd = 0.008, 0.25
    ty = kp * (-y) + kd * (desired_vy - vy)
    tx = -kp * x - kd * vx
    tz = -kp * z - kd * vz
    
    max_t = 0.12
    return clamp(tx,-max_t,max_t), clamp(ty,-max_t,max_t), clamp(tz,-max_t,max_t)

# ============================================================================
# MAIN AUTOPILOT FUNCTION
# ============================================================================

def autopilot_step(state: dict, dt: float) -> dict:
    """Main autopilot step function."""
    global _state
    
    _state['step'] += 1
    _state['accum_time'] += dt
    
    if not _state['initialized']:
        _state['initialized'] = True
        _state['pos_estimate'] = [0.0, -200.0, 0.0]
        _state['vel_estimate'] = [0.0, 0.0, 0.0]
    
    # Read sensors
    rel_pos, rel_vel = read_relative_state(state)
    range_m = read_range(state)
    dock_bearing, dock_range = read_dock_camera(state)
    gyro_rates = read_gyro_rates(state)
    attitude = read_attitude(state)
    fuel = state.get('fuel_remaining_kg', 500)
    
    # Estimate state
    est_pos, est_vel = estimate_state(rel_pos, rel_vel, dt)
    
    if range_m is None:
        range_m = norm(est_pos)
    
    _state['pos_history'].append(est_pos[:])
    if len(_state['pos_history']) > 200:
        _state['pos_history'].pop(0)
    
    # Phase-based control
    if range_m > 200:
        tx, ty, tz = control_far(est_pos, est_vel, dt, fuel)
        _state['phase'] = 'far'
    elif range_m > 50:
        tx, ty, tz = control_medium(est_pos, est_vel, dt, fuel)
        _state['phase'] = 'medium'
    elif range_m > 10:
        tx, ty, tz = control_close(est_pos, est_vel, dt, fuel)
        _state['phase'] = 'close'
    else:
        tx, ty, tz = control_final(est_pos, est_vel, dt, dock_bearing, dock_range, fuel)
        _state['phase'] = 'final'
    
    # Fuel conservation
    if fuel < 35:
        scale = max(0.25, fuel / 35.0)
        tx *= scale
        ty *= scale
        tz *= scale
    
    # Attitude control: rate damping + alignment
    damp = 0.65
    att_r = -damp * gyro_rates['roll']
    att_p = -damp * gyro_rates['pitch']
    att_y = -damp * gyro_rates['yaw']
    
    if range_m < 25:
        att_p -= 0.001 * attitude['pitch']
        att_y -= 0.001 * attitude['yaw']
    
    max_rate = 0.028
    att_r = clamp(att_r, -max_rate, max_rate)
    att_p = clamp(att_p, -max_rate, max_rate)
    att_y = clamp(att_y, -max_rate, max_rate)
    
    _state['last_thrust'] = [tx, ty, tz]
    
    return {
        'thrust_x': tx,
        'thrust_y': ty,
        'thrust_z': tz,
        'att_rate_roll': att_r,
        'att_rate_pitch': att_p,
        'att_rate_yaw': att_y
    }

**Key Design Decisions:**

1. **Orbital Mechanics**: The critical insight is that to catch up when behind (y < 0), you thrust RETROGRADE (negative Y) to lower your orbit and speed up. Conversely, when ahead, thrust PROGRADE to raise orbit and slow down.

2. **Sensor Fusion**: Averages valid readings from radar and lidar for robustness against sensor failures.

3. **Dead Reckoning**: When sensors fail, propagates position using last known velocity estimate.

4. **Range-Based Phases**:
   - **Far (>200m)**: Aggressive orbital transfers
   - **Medium (50-200m)**: Controlled V-bar approach
   - **Close (10-50m)**: Linear regime, slower approach
   - **Final (<10m)**: Very gentle, uses docking camera

5. **Fuel Conservation**: Scales down thrust when fuel is low.

6. **Attitude Control**: Rate damping with alignment corrections when close to target.
