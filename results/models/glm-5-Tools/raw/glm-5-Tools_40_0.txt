#include <bits/stdc++.h>
using namespace std;

int n, m;
vector<long long> c;
vector<vector<int>> adj;
vector<int> best_assignment;
long long best_value;
vector<long long> suffix_pos;

void solve(vector<int>& assignment, int idx, long long current_val) {
    if (idx == n) {
        if (current_val > best_value) {
            best_value = current_val;
            for (int i = 0; i < n; i++) best_assignment[i] = assignment[i];
        }
        return;
    }
    
    // Upper bound pruning
    if (current_val + suffix_pos[idx] <= best_value) return;
    
    // Check if idx can be set to 1
    bool can_set_one = true;
    for (int v : adj[idx]) {
        if (v < idx && assignment[v] == 1) {
            can_set_one = false;
            break;
        }
    }
    
    // Try x[idx] = 1 if feasible and beneficial
    if (can_set_one && c[idx] > 0) {
        assignment[idx] = 1;
        solve(assignment, idx + 1, current_val + c[idx]);
        assignment[idx] = 0;
    }
    
    // Try x[idx] = 0
    assignment[idx] = 0;
    solve(assignment, idx + 1, current_val);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    cin >> n >> m;
    
    c.resize(n);
    adj.resize(n);
    best_assignment.resize(n, 0);
    suffix_pos.resize(n + 1, 0);
    
    for (int i = 0; i < n; i++) cin >> c[i];
    
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    
    for (int i = n - 1; i >= 0; i--) {
        suffix_pos[i] = suffix_pos[i + 1] + max(0LL, c[i]);
    }
    
    best_value = 0;
    vector<int> assignment(n, 0);
    solve(assignment, 0, 0);
    
    cout << best_value << "\n";
    for (int i = 0; i < n; i++) {

        if (i < n - 1) cout << " ";
    }

    return 0;
}
