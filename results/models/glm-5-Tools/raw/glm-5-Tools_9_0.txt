using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program

        using var reader = new StreamReader(Console.OpenStandardInput(), Encoding.UTF8, false, 65536);
        using var writer = new StreamWriter(Console.OpenStandardOutput(), Encoding.UTF8, false, 65536);
        
        // Parse header
        var header = reader.ReadLine();
        int sp = header.IndexOf(' ');
        int n = int.Parse(header.AsSpan(0, sp));
        int stockLen = int.Parse(header.AsSpan(sp + 1));
        
        // Parse cuts
        var cuts = new int[n];
        var idxs = new int[n];
        var rest = reader.ReadToEnd();
        int pos = 0, cnt = 0;
        
        while (cnt < n)
        {
            while (pos < rest.Length && rest[pos] <= ' ') pos++;
            if (pos >= rest.Length) break;
            int start = pos;
            while (pos < rest.Length && rest[pos] > ' ') pos++;
            cuts[cnt] = int.Parse(rest.AsSpan(start, pos - start));
            idxs[cnt] = cnt;
            cnt++;
        }
        
        // Sort descending by cut value
        Array.Sort(cuts, idxs, Comparer<int>.Create((a, b) => b.CompareTo(a)));
        
        // Segment tree for O(log n) first-fit
        int cap = 1;
        while (cap < n) cap *= 2;
        var tree = new int[cap * 2];
        var rems = new int[n];
        var binItems = new List<int>[n];
        var binCounts = new int[n];
        int numBins = 0;
        
        for (int i = 0; i < n; i++)
        {
            int cut = cuts[i];
            int origIdx = idxs[i];
            
            // Query first bin with remaining >= cut
            int binIdx = -1;
            if (numBins > 0)
            {
                int node = 1, start = 0, end = cap;
                while (end - start > 1 && tree[node] >= cut)
                {
                    int mid = (start + end) >> 1;
                    if (tree[node * 2] >= cut && start < numBins)
                    {
                        node = node * 2;
                        end = mid;
                    }
                    else
                    {
                        node = node * 2 + 1;
                        start = mid;
                    }
                }
                if (start < numBins && tree[cap + start] >= cut)
                    binIdx = start;
            }
            
            if (binIdx >= 0)
            {
                binItems[binIdx].Add(origIdx);
                rems[binIdx] -= cut;
                int tpos = cap + binIdx;
                tree[tpos] = rems[binIdx];
                for (tpos >>= 1; tpos >= 1; tpos >>= 1)
                    tree[tpos] = Math.Max(tree[tpos * 2], tree[tpos * 2 + 1]);
            }
            else
            {
                binItems[numBins] = new List<int>();
                binItems[numBins].Add(origIdx);
                rems[numBins] = stockLen - cut;
                int tpos = cap + numBins;
                tree[tpos] = stockLen - cut;
                for (tpos >>= 1; tpos >= 1; tpos >>= 1)
                    tree[tpos] = Math.Max(tree[tpos * 2], tree[tpos * 2 + 1]);
                numBins++;
            }
        }
        
        // Output
        writer.WriteLine(numBins);
        var sb = new StringBuilder();
        for (int i = 0; i < numBins; i++)
        {
            sb.Clear();
            var bin = binItems[i];
            for (int j = 0; j < bin.Count; j++)
            {
                if (j > 0) sb.Append(' ');
                sb.Append(bin[j]);
            }
            writer.WriteLine(sb);
        }
    }
}
