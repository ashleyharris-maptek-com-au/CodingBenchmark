; SPIR-V
; Version: 1.0
; Generator: Khronos SPIR-V Tools Assembler
; Bound: 100
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

; Types
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4

; Constants
     %uint_6 = OpConstant %uint 6
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_4 = OpConstant %int 4
 %float_1_0 = OpConstant %float 1.0
 %float_0_2 = OpConstant %float 0.2
 %float_0_8 = OpConstant %float 0.8
%float_n0_0 = OpConstant %float 0.0
  %lightDir = OpConstantComposite %v3float %float_1_0 %float_1_0 %float_1_0

; Input gl_PerVertex block (Position only)
%gl_PerVertex_in = OpTypeStruct %v4float
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
   %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
 %ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
       %gl_in = OpVariable %ptr_in_arr_pv Input

; Output gl_PerVertex block (Position only)
%gl_PerVertex_out = OpTypeStruct %v4float
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
  %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
      %gl_out = OpVariable %ptr_out_pv Output

; User-defined inputs (arrays of 6 for InputTrianglesAdjacency)
  %arr_v3_6 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_6
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
    %worldPos = OpVariable %ptr_in_arr_v3 Input
      %normal = OpVariable %ptr_in_arr_v3 Input
       %color = OpVariable %ptr_in_arr_v3 Input

; User-defined output
               OpDecorate %outColor Location 0
%ptr_out_v3 = OpTypePointer Output %v3float
    %outColor = OpVariable %ptr_out_v3 Output

; Pointer types for accessing struct members
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; Main function
       %main = OpFunction %void None %3
          %5 = OpLabel

; Load gl_in[0].gl_Position (xyz component)
   %p0_full = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
        %p0 = OpLoad %v4float %p0_full
       %p0x = OpCompositeExtract %float %p0 0
       %p0y = OpCompositeExtract %float %p0 1
       %p0z = OpCompositeExtract %float %p0 2

; Load gl_in[2].gl_Position (xyz component)
   %p2_full = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
        %p2 = OpLoad %v4float %p2_full
       %p2x = OpCompositeExtract %float %p2 0
       %p2y = OpCompositeExtract %float %p2 1
       %p2z = OpCompositeExtract %float %p2 2

; Load gl_in[4].gl_Position (xyz component)
   %p4_full = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
        %p4 = OpLoad %v4float %p4_full
       %p4x = OpCompositeExtract %float %p4 0
       %p4y = OpCompositeExtract %float %p4 1
       %p4z = OpCompositeExtract %float %p4 2

; Compute edge vectors: p2-p0 and p4-p0
    %e1x = OpFSub %float %p2x %p0x
    %e1y = OpFSub %float %p2y %p0y
    %e1z = OpFSub %float %p2z %p0z

    %e2x = OpFSub %float %p4x %p0x
    %e2y = OpFSub %float %p4y %p0y
    %e2z = OpFSub %float %p4z %p0z

; Compute cross product: cross(e1, e2) = cross(p2-p0, p4-p0)
; cross.x = e1.y * e2.z - e1.z * e2.y
; cross.y = e1.z * e2.x - e1.x * e2.z
; cross.z = e1.x * e2.y - e1.y * e2.x
   %tmp_x1 = OpFMul %float %e1y %e2z
   %tmp_x2 = OpFMul %float %e1z %e2y
      %cx = OpFSub %float %tmp_x1 %tmp_x2

   %tmp_y1 = OpFMul %float %e1z %e2x
   %tmp_y2 = OpFMul %float %e1x %e2z
      %cy = OpFSub %float %tmp_y1 %tmp_y2

   %tmp_z1 = OpFMul %float %e1x %e2y
   %tmp_z2 = OpFMul %float %e1y %e2x
      %cz = OpFSub %float %tmp_z1 %tmp_z2

; Build and normalize face normal
   %crossN = OpCompositeConstruct %v3float %cx %cy %cz
        %N = OpExtInst %v3float %1 Normalize %crossN

; Normalize light direction
        %L = OpExtInst %v3float %1 Normalize %lightDir

; Compute diffuse: max(dot(N, L), 0.0)
       %dot = OpDot %float %N %L
   %diffuse = OpExtInst %float %1 FMax %dot %float_n0_0

; Compute lighting factor: 0.2 + 0.8 * diffuse
  %diff_08 = OpFMul %float %float_0_8 %diffuse
   %factor = OpFAdd %float %float_0_2 %diff_08

; Load input colors for vertices 0, 2, 4
     %c0ptr = OpAccessChain %ptr_in_v3 %color %int_0
         %c0 = OpLoad %v3float %c0ptr
     %c2ptr = OpAccessChain %ptr_in_v3 %color %int_2
         %c2 = OpLoad %v3float %c2ptr
     %c4ptr = OpAccessChain %ptr_in_v3 %color %int_4
         %c4 = OpLoad %v3float %c4ptr

; Compute average color: (c0 + c2 + c4) / 3
     %sum1 = OpFAdd %v3float %c0 %c2
     %sum2 = OpFAdd %v3float %sum1 %c4
   %avgCol = OpVectorTimesScalar %v3float %sum2 %float_0_333333

; Compute final color: avgCol * factor
  %finalCol = OpVectorTimesScalar %v3float %avgCol %factor

; --- Emit vertex 0 ---
  %outPos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %outPos0 %p0
               OpStore %outColor %finalCol
               OpEmitVertex

; --- Emit vertex 2 ---
  %outPos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %outPos2 %p2
               OpStore %outColor %finalCol
               OpEmitVertex

; --- Emit vertex 4 ---
  %outPos4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %outPos4 %p4
               OpStore %outColor %finalCol
               OpEmitVertex

; End primitive (complete the triangle)
               OpEndPrimitive

               OpReturn
               OpFunctionEnd
