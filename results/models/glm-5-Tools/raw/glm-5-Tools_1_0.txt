Looking at this challenge, I need to create a TSP solver that adapts its strategy based on problem size. Let me design a solution that uses:

1. **Nearest Neighbor + 2-opt** for small-medium instances
2. **Fast greedy construction** for larger instances
3. **Morton curve approximation** for extreme scales

The key insight is that for millions of cities, we need O(N) or O(N log N) algorithms - any O(N²) approach will fail.

```cpp
#include <cstdio>
#include <cstdlib>

using namespace std;

struct City {
    double x, y;
    int id;
};

// Fast integer coordinates for morton curves
static inline uint64_t splitBy2(uint32_t a) {
    uint64_t x = a & 0x1fffff;
    x = (x | (x << 32)) & 0x1f00000000ffffULL;
    x = (x | (x << 16)) & 0x1f0000ff0000ffULL;
    x = (x | (x << 8)) & 0x100f00f00f00f00fULL;
    x = (x | (x << 4)) & 0x10c30c30c30c30c3ULL;
    x = (x | (x << 2)) & 0x1249249249249249ULL;
    return x;
}

static inline uint64_t morton2D(uint32_t x, uint32_t y) {
    return (splitBy2(y) << 1) + splitBy2(x);
}

inline double fastDist(const City& a, const City& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

inline double euclideanDist(const City& a, const City& b) {
    return sqrt(fastDist(a, b));
}

// Nearest Neighbor construction
void nearestNeighbor(const vector<City>& cities, vector<int>& tour) {
    int n = (int)cities.size();
    vector<bool> visited(n, false);
    tour[0] = 0;
    visited[0] = true;
    
    for (int i = 1; i < n; i++) {
        int last = tour[i - 1];
        double bestDist = 1e30;
        int bestNext = -1;
        
        for (int j = 0; j < n; j++) {
            if (!visited[j]) {
                double d = fastDist(cities[last], cities[j]);
                if (d < bestDist) {
                    bestDist = d;
                    bestNext = j;
                }
            }
        }
        tour[i] = bestNext;
        visited[bestNext] = true;
    }
}

// Simple 2-opt improvement
void twoOpt(const vector<City>& cities, vector<int>& tour, int maxIter) {
    int n = (int)tour.size();
    if (n < 4) return;
    
    for (int iter = 0; iter < maxIter; iter++) {
        bool improved = false;
        for (int i = 0; i < n - 2 && !improved; i++) {
            for (int j = i + 2; j < n && !improved; j++) {
                double d1 = fastDist(cities[tour[i]], cities[tour[i + 1]]);
                double d2 = fastDist(cities[tour[j]], cities[tour[(j + 1) % n]]);
                double d3 = fastDist(cities[tour[i]], cities[tour[j]]);
                double d4 = fastDist(cities[tour[i + 1]], cities[tour[(j + 1) % n]]);
                
                if (d3 + d4 < d1 + d2) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }
}

// Greedy edge construction - O(N log N) with union-find
class UnionFind {
    vector<int> parent, rank;
public:
    UnionFind(int n) : parent(n), rank(n, 0) {
        for (int i = 0; i < n; i++) parent[i] = i;
    }
    int find(int x) {
        if (parent[x] != x) parent[x] = find(parent[x]);
        return parent[x];
    }
    bool unite(int x, int y) {
        int px = find(x), py = find(y);
        if (px == py) return false;
        if (rank[px] < rank[py]) swap(px, py);
        parent[py] = px;
        if (rank[px] == rank[py]) rank[px]++;
        return true;
    }
};

struct Edge {
    int from, to;
    double len;
    bool operator<(const Edge& other) const { return len < other.len; }
};

void greedyEdgeConstruction(const vector<City>& cities, vector<int>& tour) {
    int n = (int)cities.size();
    
    // For large n, use sampling to keep memory reasonable
    int maxEdges = min(n * 20, 50000000);
    
    // Generate candidate edges
    vector<Edge> edges;
    edges.reserve(maxEdges);
    
    if (n <= 10000) {
        // Full edge list for small instances
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                edges.push_back({i, j, fastDist(cities[i], cities[j])});
            }
        }
    } else {
        // Sample edges for large instances
        mt19937 rng(42);
        uniform_int_distribution<int> dist(0, n - 1);
        
        for (int k = 0; k < maxEdges; k++) {
            int i = dist(rng);
            int j = dist(rng);
            if (i != j) {
                edges.push_back({min(i, j), max(i, j), fastDist(cities[i], cities[j])});
            }
        }
    }
    
    sort(edges.begin(), edges.end());
    
    // Build degree-constrained MST (each node degree <= 2)
    vector<int> degree(n, 0);
    vector<vector<int>> adj(n);
    UnionFind uf(n);
    
    for (auto& e : edges) {
        if (degree[e.from] < 2 && degree[e.to] < 2 && uf.find(e.from) != uf.find(e.to)) {
            adj[e.from].push_back(e.to);
            adj[e.to].push_back(e.from);
            degree[e.from]++;
            degree[e.to]++;
            uf.unite(e.from, e.to);
        }
    }
    
    // Connect remaining components
    for (int i = 0; i < n; i++) {
        if (degree[i] < 2) {
            double bestDist = 1e30;
            int bestJ = -1;
            int checks = min(n, 5000);
            for (int k = 0; k < checks; k++) {
                int j = (i + k * 10007 + 1) % n;
                if (j != i && degree[j] < 2) {
                    double d = fastDist(cities[i], cities[j]);
                    if (d < bestDist) {
                        bestDist = d;
                        bestJ = j;
                    }
                }
            }
            if (bestJ >= 0) {
                adj[i].push_back(bestJ);
                adj[bestJ].push_back(i);
                degree[i]++;
                degree[bestJ]++;
            }
        }
    }
    
    // Extract tour (it's a set of paths, need to connect them)
    // Find starting point (city 0 or a degree-1 vertex)
    int start = 0;
    
    tour.clear();
    tour.reserve(n);
    vector<bool> visited(n, false);
    
    int current = start;
    int prev = -1;
    
    while ((int)tour.size() < n) {
        tour.push_back(current);
        visited[current] = true;
        
        int next = -1;
        double bestDist = 1e30;
        
        for (int neighbor : adj[current]) {
            if (!visited[neighbor]) {
                double d = fastDist(cities[current], cities[neighbor]);
                if (d < bestDist) {
                    bestDist = d;
                    next = neighbor;
                }
            }
        }
        
        if (next == -1) {
            // Find any unvisited node
            for (int i = 0; i < n; i++) {
                if (!visited[i]) {
                    next = i;
                    break;
                }
            }
        }
        
        prev = current;
        current = next;
        if (current == -1) break;
    }
    
    // Ensure tour starts at 0
    if (!tour.empty() && tour[0] != 0) {
        int pos = find(tour.begin(), tour.end(), 0) - tour.begin();
        if (pos < (int)tour.size()) {
            rotate(tour.begin(), tour.begin() + pos, tour.end());
        } else {
            tour[0] = 0;
        }
    }
}

// Morton curve approximation for very large instances
void mortonCurveTour(vector<City>& cities, vector<int>& tour) {
    int n = (int)cities.size();
    
    // Find bounding box
    double minX = cities[0].x, maxX = cities[0].x;
    double minY = cities[0].y, maxY = cities[0].y;
    
    for (int i = 1; i < n; i++) {
        minX = min(minX, cities[i].x);
        maxX = max(maxX, cities[i].x);
        minY = min(minY, cities[i].y);
        maxY = max(maxY, cities[i].y);
    }
    
    double scaleX = (maxX > minX) ? (double)0xFFFFE / (maxX - minX) : 1.0;
    double scaleY = (maxY > minY) ? (double)0xFFFFE / (maxY - minY) : 1.0;
    
    // Assign morton codes
    vector<pair<uint64_t, int>> order(n);
    for (int i = 0; i < n; i++) {
        uint32_t ix = (uint32_t)((cities[i].x - minX) * scaleX);
        uint32_t iy = (uint32_t)((cities[i].y - minY) * scaleY);
        ix = min(ix, (uint32_t)0xFFFFF);
        iy = min(iy, (uint32_t)0xFFFFF);
        order[i] = {morton2D(ix, iy), i};
    }
    
    sort(order.begin(), order.end());
    
    tour.resize(n);
    for (int i = 0; i < n; i++) {
        tour[i] = order[i].second;
    }
    
    // Ensure tour starts at 0
    int pos = find(tour.begin(), tour.end(), 0) - tour.begin();
    if (pos < n) {
        rotate(tour.begin(), tour.begin() + pos, tour.end());
    }
}

// Check if tour is valid
bool validateTour(const vector<int>& tour, int n) {
    if ((int)tour.size() != n) return false;
    if (tour.empty()) return false;
    if (tour[0] != 0) return false;
    
    vector<bool> seen(n, false);
    for (int i : tour) {
        if (i < 0 || i >= n || seen[i]) return false;
        seen[i] = true;
    }
    return true;
}

    if (n == 1) {

    else if (n <= 12) {

        tour.resize(n);

        vector<int> bestTour = tour;
        double bestLen = 1e30;
        
        do {
            double len = 0;
            for (int i = 0; i < n - 1; i++) {
                len += euclideanDist(cities[tour[i]], cities[tour[i + 1]]);
            }
            len += euclideanDist(cities[tour[n - 1]], cities[tour[0]]);
            
            if (len < bestLen) {

```

## Key Design Decisions:

1. **Trivial (≤12 cities)**: Brute force with permutations - finds optimal solution

2. **Small-Medium (13-1000)**: Nearest Neighbor + 2-opt iterations - good quality, reasonable time

3. **Medium-Large (1001-50000)**: Nearest Neighbor + limited 2-opt - balances quality vs speed

4. **Large (50001-500000)**: Greedy edge construction with sampling - O(N log N) complexity

5. **Extreme/Epic (>500000)**: Morton curve (Z-order curve) approximation - O(N log N), very fast

The algorithm automatically selects the appropriate strategy based on N. Morton curves are particularly effective for spatial data, giving a reasonable approximation in O(N log N) time even for billions of points.
