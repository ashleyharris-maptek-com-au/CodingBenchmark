; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2 Input
%in_tangent = OpVariable %ptr_Input_vec3 Input
%in_color = OpVariable %ptr_Input_vec3 Input
%outColor = OpVariable %ptr_Output_vec4 Output
%ubo = OpVariable %ptr_Uniform_UBO Uniform

%float_0 = OpConstant %float 0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2

%main_func = OpTypeFunction %void
%main = OpFunction %void None %main_func
%entry = OpLabel

; Load normal and normalize it
%normal = OpLoad %vec3 %in_normal
%N = OpVectorNormalize %vec3 %normal

; Load camera position from UBO (offset 208 = member 4)
%cameraPos_ptr = OpAccessChain %ptr_Uniform_vec4 %ubo %float_4
%cameraPos = OpLoad %vec4 %cameraPos_ptr

; Load world position
%worldPos = OpLoad %vec3 %in_worldPos

; Compute V = normalize(cameraPos.xyz - worldPos)
%camPos_xyz = OpVectorShuffle %vec3 %cameraPos %cameraPos 0 1 2
%V_unnorm = OpFSub %vec3 %camPos_xyz %worldPos
%V = OpVectorNormalize %vec3 %V_unnorm

; Compute NdotV = max(dot(N, V), 0)
%NdotV_full = OpDot %float %N %V
%NdotV = OpFMax %float %NdotV_full %float_0

; Check if NdotV < 0.5
%cond = OpFOrdLessThan %bool %NdotV %float_0_5

; Branch to compute color
OpBranchConditional %cond %branch_lt %branch_ge

%branch_lt = OpLabel
; NdotV < 0.5: r=1, g=2*NdotV, b=0
%g_lt = OpFMul %float %float_2 %NdotV
%color_lt = OpCompositeConstruct %vec3 %float_1 %g_lt %float_0
OpBranch %merge

%branch_ge = OpLabel
; NdotV >= 0.5: t = (NdotV - 0.5) * 2
%t = OpFSub %float %NdotV %float_0_5
%t2 = OpFMul %float %t %float_2
; r = g = 1 - t2, b = t2
%r_ge = OpFSub %float %float_1 %t2
%g_ge = OpFSub %float %float_1 %t2
%color_ge = OpCompositeConstruct %vec3 %r_ge %g_ge %t2
OpBranch %merge

%merge = OpLabel
%color_rgb = OpPhi %vec3 %color_lt %branch_lt %color_ge %branch_ge
%color_rgba = OpCompositeConstruct %vec4 %color_rgb %float_1
OpStore %outColor %color_rgba
OpReturn
OpFunctionEnd
