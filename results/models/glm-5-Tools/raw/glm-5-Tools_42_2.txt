```spirv
; SPIR-V Assembly for Geometry Shader: Point to Hexagon
; Version 1.0, Vulkan

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %inColor %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

; Debug info
OpSource GLSL 450
OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_in "gl_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %inColor "inColor"
OpName %outColor "outColor"

; Decorations for gl_PerVertex blocks
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Decorations for user-defined inputs
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%void_fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

; Constants
%uint_1 = OpConstant %uint 1
%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%float_0_04 = OpConstant %float 0.04
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_8 = OpConstant %float 0.8
%float_60 = OpConstant %float 60.0
%float_2 = OpConstant %float 2.0
%float_360 = OpConstant %float 360.0
%float_0_5 = OpConstant %float 0.5

; PI = 3.14159265...
%float_pi = OpConstant %float 3.14159265

; gl_PerVertex types (input and output)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out

; Pointer types for accessing members
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v3 = OpTypePointer Output %v3float

; Input arrays
%arr_v3_1 = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_1

; Interface variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

; Main function
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Get center position from gl_in[0].gl_Position
%center_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%center_pos = OpLoad %v4float %center_pos_ptr

; Get color from inColor[0]
%color_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%base_color = OpLoad %v3float %color_ptr

; Calculate darker color for edges (multiply by 0.8)
%dark_color = OpVectorTimesScalar %v3float %base_color %float_0_8

; Extract x, y, z from center position
%cx = OpCompositeExtract %float %center_pos 0
%cy = OpCompositeExtract %float %center_pos 1
%cz = OpCompositeExtract %float %center_pos 2
%cw = OpCompositeExtract %float %center_pos 3

; Radius
%radius = OpCopyObject %float %float_0_04

; Hexagon angles: 0°, 60°, 120°, 180°, 240°, 300° (in radians)
; Precompute cos/sin for each angle
; Angle 0: cos=1, sin=0
%cos0 = OpCopyObject %float %float_1
%sin0 = OpCopyObject %float %float_0

; Angle 60° = pi/3 ≈ 1.04719758
%angle1 = OpFMul %float %float_pi %float_0_5
%angle1_div3 = OpFMul %float %angle1 %float_0_5
; Actually 60° = pi/3, let me compute properly
%pi_div_3 = OpConstant %float 1.04719758
%cos1 = OpCos %float %pi_div_3
%sin1 = OpSin %float %pi_div_3

; Angle 120° = 2*pi/3
%pi2_div_3 = OpConstant %float 2.0943951
%cos2 = OpCos %float %pi2_div_3
%sin2 = OpSin %float %pi2_div_3

; Angle 180° = pi
%cos3 = OpConstant %float -1.0
%sin3 = OpCopyObject %float %float_0

; Angle 240° = 4*pi/3
%pi4_div_3 = OpConstant %float 4.1887902
%cos4 = OpCos %float %pi4_div_3
%sin4 = OpSin %float %pi4_div_3

; Angle 300° = 5*pi/3
%pi5_div_3 = OpConstant %float 5.2359876
%cos5 = OpCos %float %pi5_div_3
%sin5 = OpSin %float %pi5_div_3

; ============================================
; Emit Triangle 0: center, edge0, edge1
; ============================================

; Vertex 0: center
%out_pos_ptr_0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr_0 %center_pos
OpStore %outColor %base_color
OpEmitVertex

; Vertex 1: edge0 (angle 0)
%ex1 = OpFMul %float %radius %cos0
%ey1 = OpFMul %float %radius %sin0
%px1 = OpFAdd %float %cx %ex1
%py1 = OpFAdd %float %cy %ey1
%pos1 = OpCompositeConstruct %v4float %px1 %py1 %cz %cw
OpStore %out_pos_ptr_0 %pos1
OpStore %outColor %dark_color
OpEmitVertex

; Vertex 2: edge1 (angle 60°)
%ex2 = OpFMul %float %radius %cos1
%ey2 = OpFMul %float %radius %sin1
%px2 = OpFAdd %float %cx %ex2
%py2 = OpFAdd %float %cy %ey2
%pos2 = OpCompositeConstruct %v4float %px2 %py2 %cz %cw
OpStore %out_pos_ptr_0 %pos2
OpStore %outColor %dark_color
OpEmitVertex

OpEndPrimitive

; ============================================
; Emit Triangle 1: center, edge1, edge2
; ============================================

; Vertex 3: center
OpStore %out_pos_ptr_0 %center_pos
OpStore %outColor %base_color
OpEmitVertex

; Vertex 4: edge1 (reuse pos2)
OpStore %out_pos_ptr_0 %pos2
OpStore %outColor %dark_color
OpEmitVertex

; Vertex 5: edge2 (angle 120°)
%ex5 = OpFMul %float %radius %cos2
%ey5 = OpFMul %float %radius %sin2
%px5 = OpFAdd %float %cx %ex5
%py5 = OpFAdd %float %cy %ey5
%pos5 = OpCompositeConstruct %v4float %px5 %py5 %cz %cw
OpStore %out_pos_ptr_0 %pos5
OpStore %outColor %dark_color
OpEmitVertex

OpEndPrimitive

; ============================================
; Emit Triangle 2: center, edge2, edge3
; ============================================

; Vertex 6: center
OpStore %out_pos_ptr_0 %center_pos
OpStore %outColor %base_color
OpEmitVertex

; Vertex 7: edge2 (reuse pos5)
OpStore %out_pos_ptr_0 %pos5
OpStore %outColor %dark_color
OpEmitVertex

; Vertex 8: edge3 (angle 180°)
%ex8 = OpFMul %float %radius %cos3
%ey8 = OpFMul %float %radius %sin3
%px8 = OpFAdd %float %cx %ex8
%py8 = OpFAdd %float %cy %ey8
%pos8 = OpCompositeConstruct %v4float %px8 %py8 %cz %cw
OpStore %out_pos_ptr_0 %pos8
OpStore %outColor %dark_color
OpEmitVertex

OpEndPrimitive

; ============================================
; Emit Triangle 3: center, edge3, edge4
; ============================================

; Vertex 9: center
OpStore %out_pos_ptr_0 %center_pos
OpStore %outColor %base_color
OpEmitVertex

; Vertex 10: edge3 (reuse pos8)
OpStore %out_pos_ptr_0 %pos8
OpStore %outColor %dark_color
OpEmitVertex

; Vertex 11: edge4 (angle 240°)
%ex11 = OpFMul %float %radius %cos4
%ey11 = OpFMul %float %radius %sin4
%px11 = OpFAdd %float %cx %ex11
%py11 = OpFAdd %float %cy %ey11
%pos11 = OpCompositeConstruct %v4float %px11 %py11 %cz %cw
OpStore %out_pos_ptr_0 %pos11
OpStore %outColor %dark_color
OpEmitVertex

OpEndPrimitive

; ============================================
; Emit Triangle 4: center, edge4, edge5
; ============================================

; Vertex 12: center
OpStore %out_pos_ptr_0 %center_pos
OpStore %outColor %base_color
OpEmitVertex

; Vertex 13: edge4 (reuse pos11)
OpStore %out_pos_ptr_0 %pos11
OpStore %outColor %dark_color
OpEmitVertex

; Vertex 14: edge5 (angle 300°)
%ex14 = OpFMul %float %radius %cos5
%ey14 = OpFMul %float %radius %sin5
%px14 = OpFAdd %float %cx %ex14
%py14 = OpFAdd %float %cy %ey14
%pos14 = OpCompositeConstruct %v4float %px14 %py14 %cz %cw
OpStore %out_pos_ptr_0 %pos14
OpStore %outColor %dark_color
OpEmitVertex

OpEndPrimitive

; ============================================
; Emit Triangle 5: center, edge5, edge0 (close hexagon)
; ============================================

; Vertex 15: center
OpStore %out_pos_ptr_0 %center_pos
OpStore %outColor %base_color
OpEmitVertex

; Vertex 16: edge5 (reuse pos14)
OpStore %out_pos_ptr_0 %pos14
OpStore %outColor %dark_color
OpEmitVertex

; Vertex 17: edge0 (reuse pos1 - angle 0, closing the hexagon)
OpStore %out_pos_ptr_0 %pos1
OpStore %outColor %dark_color
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
