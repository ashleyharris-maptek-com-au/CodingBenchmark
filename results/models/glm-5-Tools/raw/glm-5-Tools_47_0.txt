# WGSL Compute Shader for Hash Mining (Proof-of-Work)

```wgsl
@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>              params   : vec4<u32>;

// Rotate left for u32
fn rotl(x: u32, shift: u32) -> u32 {
    return (x << shift) | (x >> (32u - shift));
}

// Count trailing zero bits in a u32
fn count_trailing_zeros_u32(x: u32) -> u32 {
    if (x == 0u) {
        return 32u;
    }
    var n: u32 = 0u;
    var v: u32 = x;
    if ((v & 0xFFFFu) == 0u) { v >>= 16u; n += 16u; }
    if ((v & 0xFFu) == 0u)   { v >>= 8u;  n += 8u;  }
    if ((v & 0xFu) == 0u)    { v >>= 4u;  n += 4u;  }
    if ((v & 0x3u) == 0u)    { v >>= 2u;  n += 2u;  }
    if ((v & 0x1u) == 0u)    { n += 1u;  }
    return n;
}

// Count trailing zero bits of 64-bit value [h0 low | h1 high]
fn count_trailing_zeros_64(h0: u32, h1: u32) -> u32 {
    if (h0 != 0u) {
        return count_trailing_zeros_u32(h0);
    }
    return 32u + count_trailing_zeros_u32(h1);
}

// Apply one quarter-round to state[i], state[i+1], state[i+2], state[i+3]
fn quarter_round(state: ptr<function, array<u32, 16>>, i: u32) {
    let a_idx = i;
    let b_idx = i + 1u;
    let c_idx = i + 2u;
    let d_idx = i + 3u;
    
    (*state)[a_idx] += (*state)[b_idx];
    (*state)[d_idx] ^= (*state)[a_idx];
    (*state)[d_idx] = rotl((*state)[d_idx], 16u);
    
    (*state)[c_idx] += (*state)[d_idx];
    (*state)[b_idx] ^= (*state)[c_idx];
    (*state)[b_idx] = rotl((*state)[b_idx], 12u);
    
    (*state)[a_idx] += (*state)[b_idx];
    (*state)[d_idx] ^= (*state)[a_idx];
    (*state)[d_idx] = rotl((*state)[d_idx], 8u);
    
    (*state)[c_idx] += (*state)[d_idx];
    (*state)[b_idx] ^= (*state)[c_idx];
    (*state)[b_idx] = rotl((*state)[b_idx], 7u);
}

// Apply 8 rounds of mixing (all 4 quarter-rounds per round)
fn apply_mixing(state: ptr<function, array<u32, 16>>) {
    // Round 1
    quarter_round(state, 0u);
    quarter_round(state, 4u);
    quarter_round(state, 8u);
    quarter_round(state, 12u);
    
    // Round 2
    quarter_round(state, 0u);
    quarter_round(state, 4u);
    quarter_round(state, 8u);
    quarter_round(state, 12u);
    
    // Round 3
    quarter_round(state, 0u);
    quarter_round(state, 4u);
    quarter_round(state, 8u);
    quarter_round(state, 12u);
    
    // Round 4
    quarter_round(state, 0u);
    quarter_round(state, 4u);
    quarter_round(state, 8u);
    quarter_round(state, 12u);
    
    // Round 5
    quarter_round(state, 0u);
    quarter_round(state, 4u);
    quarter_round(state, 8u);
    quarter_round(state, 12u);
    
    // Round 6
    quarter_round(state, 0u);
    quarter_round(state, 4u);
    quarter_round(state, 8u);
    quarter_round(state, 12u);
    
    // Round 7
    quarter_round(state, 0u);
    quarter_round(state, 4u);
    quarter_round(state, 8u);
    quarter_round(state, 12u);
    
    // Round 8
    quarter_round(state, 0u);
    quarter_round(state, 4u);
    quarter_round(state, 8u);
    quarter_round(state, 12u);
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid: vec3<u32>) {
    let difficulty: u32 = params.x;
    let range_start: u32 = params.y;
    let range_size: u32 = params.z;
    
    // Guard: check if this thread is within range
    let tid: u32 = gid.x;
    if (tid >= range_size) {
        return;
    }
    
    // Compute my nonce
    let my_nonce: u32 = range_start + tid;
    
    // Initialize state from baseData
    var state: array<u32, 16>;
    state[0] = baseData[0];
    state[1] = baseData[1];
    state[2] = baseData[2];
    state[3] = baseData[3];
    state[4] = baseData[4];
    state[5] = baseData[5];
    state[6] = baseData[6];
    state[7] = baseData[7];
    state[8] = baseData[8];
    state[9] = baseData[9];
    state[10] = baseData[10];
    state[11] = baseData[11];
    state[12] = baseData[12];
    state[13] = baseData[13];
    state[14] = baseData[14];
    state[15] = baseData[15];
    
    // XOR with (nonce + i) for each element
    state[0] ^= (my_nonce + 0u);
    state[1] ^= (my_nonce + 1u);
    state[2] ^= (my_nonce + 2u);
    state[3] ^= (my_nonce + 3u);
    state[4] ^= (my_nonce + 4u);
    state[5] ^= (my_nonce + 5u);
    state[6] ^= (my_nonce + 6u);
    state[7] ^= (my_nonce + 7u);
    state[8] ^= (my_nonce + 8u);
    state[9] ^= (my_nonce + 9u);
    state[10] ^= (my_nonce + 10u);
    state[11] ^= (my_nonce + 11u);
    state[12] ^= (my_nonce + 12u);
    state[13] ^= (my_nonce + 13u);
    state[14] ^= (my_nonce + 14u);
    state[15] ^= (my_nonce + 15u);
    
    // Apply 8 rounds of mixing
    apply_mixing(&state);
    
    // Fold 16 words to 2 output words
    let h0: u32 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
    let h1: u32 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];
    
    // Count trailing zero bits of 64-bit hash [h0 | (h1 << 32)]
    let tz: u32 = count_trailing_zeros_64(h0, h1);
    
    // Check if difficulty requirement is met
    if (tz >= difficulty) {
        result[0] = 1u;         // Found flag
        result[1] = my_nonce;   // The nonce
        result[2] = h0;         // Low 32 bits of hash
        result[3] = h1;         // High 32 bits of hash
    }
}
```

## Explanation

### Key Components:

1. **`rotl()`**: Implements left rotation for 32-bit unsigned integers using WGSL's shift operators.

2. **`count_trailing_zeros_u32()`**: Counts trailing zeros in a single u32 using binary search on bit positions (O(log n) operations).

3. **`count_trailing_zeros_64()`**: Counts trailing zeros in a 64-bit value represented as two u32s. Checks h0 first; if all zeros, adds 32 and continues with h1.

4. **`quarter_round()`**: Applies the ChaCha-style quarter-round mixing to 4 consecutive state elements. Takes a pointer to the state array and starting index.

5. **`apply_mixing()`**: Applies 8 rounds of quarter-round operations to all 4 groups (indices 0-3, 4-7, 8-11, 12-15) explicitly unrolled.

6. **`main()`**: 
   - Reads parameters and computes the thread's assigned nonce
   - Initializes state by copying baseData and XORing with `(nonce + i)` for each element
   - Runs the hash function
   - Checks trailing zeros against difficulty
   - Writes result if successful (race conditions are acceptable per spec)

### Hash Function Verification:
- XORs each of the 16 state words with `(nonce + i)` where `i` is the index
- Applies 8 rounds of ChaCha-style mixing (quarter-rounds on consecutive groups)
- Folds to 2 output words by XORing even/odd indexed elements respectively
- `h0` is the low 32 bits (even indices), `h1` is high 32 bits (odd indices)
