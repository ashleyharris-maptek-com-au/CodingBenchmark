```spirv
; SPIR-V Assembly for Geometry Shader - Thick Smooth Curve with Miter Joins
; Version 1.0, Vulkan environment

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Debug info
OpSource GLSL 450
OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_in "gl_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"

; Decorations for gl_PerVertex input block
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Decorations for gl_PerVertex output block
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Decorations for user-defined inputs
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; Basic types
%void = OpTypeVoid
%void_fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; Integer types for indexing
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%uint = OpTypeInt 32 0
%uint_4 = OpConstant %uint 4

; Float constants
%float_0_025 = OpConstant %float 0.025
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1

; gl_PerVertex input struct (just Position)
%gl_PerVertex_in = OpTypeStruct %v4float

; Array of 4 gl_PerVertex_in for InputLinesAdjacency
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4

; Pointers
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; Arrays for user-defined inputs (size 4)
%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4

; Interface variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_out = OpVariable %ptr_out_pv Output

%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%outColor = OpVariable %ptr_out_v3 Output

; Main function
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Load all 4 positions from gl_in
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0

%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr
%p3 = OpLoad %v4float %p3_ptr

; Load colors from color[1] and color[2]
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%color1 = OpLoad %v3float %c1_ptr
%color2 = OpLoad %v3float %c2_ptr

; Compute tangent at start: t0 = normalize(p2.xy - p0.xy)
; Note: We work in screen space (xy), z and w are passed through
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
%t0_raw = OpFSub %v2float %p2_xy %p0_xy
%t0_len = OpLength %float %t0_raw
%t0_len_safe = OpFMax %float %t0_len %float_0
%t0 = OpVectorTimesScalar %v2float %t0_raw %t0_len_safe
%t0_len_recip = OpFDiv %float %float_1 %t0_len_safe
%t0 = OpVectorTimesScalar %v2float %t0_raw %t0_len_recip

; Compute tangent at end: t1 = normalize(p3.xy - p1.xy)
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p3_xy = OpVectorShuffle %v2float %p3 %p3 0 1
%t1_raw = OpFSub %v2float %p3_xy %p1_xy
%t1_len = OpLength %float %t1_raw
%t1_len_safe = OpFMax %float %t1_len %float_0
%t1_len_recip = OpFDiv %float %float_1 %t1_len_safe
%t1 = OpVectorTimesScalar %v2float %t1_raw %t1_len_recip

; Compute perpendiculars: perp = vec2(-t.y, t.x)
%t0_y = OpCompositeExtract %float %t0 1
%t0_x = OpCompositeExtract %float %t0 0
%perp0_x = OpFNegate %float %t0_y
%perp0 = OpCompositeConstruct %v2float %perp0_x %t0_x

%t1_y = OpCompositeExtract %float %t1 1
%t1_x = OpCompositeExtract %float %t1 0
%perp1_x = OpFNegate %float %t1_y
%perp1 = OpCompositeConstruct %v2float %perp1_x %t1_x

; Scale perpendiculars by thickness
%perp0_thick = OpVectorTimesScalar %v2float %perp0 %float_0_025
%perp1_thick = OpVectorTimesScalar %v2float %perp1 %float_0_025

; Extract p1.xy and p2.xy for offset calculations
%p1_xy_2 = OpVectorShuffle %v2float %p1 %p1 0 1
%p2_xy_2 = OpVectorShuffle %v2float %p2 %p2 0 1

; === Vertex 0: start + perp0*thickness ===
%v0_xy = OpFAdd %v2float %p1_xy_2 %perp0_thick
%v0 = OpCompositeConstruct %v4float %v0_xy %float_0 %float_1

; Store gl_Position
%gl_out_ptr_0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_ptr_0 %v0

; Store outColor
OpStore %outColor %color1

; Emit vertex 0
OpEmitVertex

; === Vertex 1: start - perp0*thickness ===
%perp0_neg = OpFNegate %v2float %perp0_thick
%v1_xy = OpFAdd %v2float %p1_xy_2 %perp0_neg
%v1 = OpCompositeConstruct %v4float %v1_xy %float_0 %float_1

; Store gl_Position
%gl_out_ptr_1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_ptr_1 %v1

; Store outColor
OpStore %outColor %color1

; Emit vertex 1
OpEmitVertex

; === Vertex 2: end + perp1*thickness ===
%v2_xy_new = OpFAdd %v2float %p2_xy_2 %perp1_thick
%v2_new = OpCompositeConstruct %v4float %v2_xy_new %float_0 %float_1

; Store gl_Position
%gl_out_ptr_2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_ptr_2 %v2_new

; Store outColor
OpStore %outColor %color2

; Emit vertex 2
OpEmitVertex

; === Vertex 3: end - perp1*thickness ===
%perp1_neg = OpFNegate %v2float %perp1_thick
%v3_xy_new = OpFAdd %v2float %p2_xy_2 %perp1_neg
%v3_new = OpCompositeConstruct %v4float %v3_xy_new %float_0 %float_1

; Store gl_Position
%gl_out_ptr_3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_ptr_3 %v3_new

; Store outColor
OpStore %outColor %color2

; Emit vertex 3
OpEmitVertex

; End the primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
```
