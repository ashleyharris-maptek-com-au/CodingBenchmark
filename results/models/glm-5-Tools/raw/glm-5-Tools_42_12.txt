```spirv
; SPIR-V Assembly for Geometry Shader
; Silhouette/Boundary Edge Detection with Thick Lines

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %inColor %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

; Extensions
OpSource GLSL 450

; Decorations - Input gl_PerVertex block
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Decorations - Output gl_PerVertex block
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Decorations - User-defined inputs
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %inColor Location 2

; Decorations - User-defined output
OpDecorate %outColor Location 0

; Decorations - UBO
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 204
OpMemberDecorate %UBO 5 Offset 216
OpMemberDecorate %UBO 6 Offset 228
OpDecorate %ubo_binding DescriptorSet 0
OpDecorate %ubo_binding Binding 0

; Types
%void = OpTypeVoid
%void_fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

; Integers
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%uint = OpTypeInt 32 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_6 = OpConstant %uint 6

; Float constants
%float_0 = OpConstant %float 0
%float_0_015 = OpConstant %float 0.015
%float_0_3 = OpConstant %float 0.3
%float_1 = OpConstant %float 1
%float_neg1 = OpConstant %float -1

; Vector constants
%v3_white = OpConstantComposite %v3float %float_1 %float_1 %float_1

; Input gl_PerVertex struct and array
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Output gl_PerVertex struct
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; User-defined input arrays
%arr_v3_6 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_6

%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input

; User-defined output
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Pointer types
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float

; UBO
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v3float %float %v3float %v3float
%ptr_ubo = OpTypePointer Uniform %UBO
%ubo_binding = OpVariable %ptr_ubo Uniform

; Functions
%bool = OpTypeBool
%true = OpConstantTrue %bool
%false = OpConstantFalse %bool

; Main function
%main = OpFunction %void None %void_fn
%entry = OpLabel

; --- Emit thick quads for all 3 edges ---
; Edge 0: vertices 0->2, adjacent vertex 1
%edge0_color = OpCompositeConstruct %v3float %float_1 %float_1 %float_1
%gl_in_0_pos = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%gl_in_2_pos = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos_0 = OpLoad %v4float %gl_in_0_pos
%pos_2 = OpLoad %v4float %gl_in_2_pos

; Emit thick quad for edge 0
%p0xy = OpCompositeExtract %float %pos_0 0
%p0zw = OpCompositeExtract %float %pos_0 1
%p2xy = OpCompositeExtract %float %pos_2 0
%p2zw = OpCompositeExtract %float %pos_2 1

%offset0 = OpCompositeConstruct %v4float %float_0_015 %float_0_015 %float_0 %float_0
%offset1 = OpCompositeConstruct %v4float %float_neg1 %float_0_015 %float_0 %float_0
%offset2 = OpCompositeConstruct %v4float %float_0_015 %float_neg1 %float_0 %float_0
%offset3 = OpCompositeConstruct %v4float %float_neg1 %float_neg1 %float_0 %float_0

%quad0_v0 = OpCompositeConstruct %v4float %p0xy %p0zw %float_0 %float_1
%quad0_v1 = OpCompositeConstruct %v4float %p2xy %p2zw %float_0 %float_1

; Vertex 0 of edge quad
%q0_v0 = OpCompositeConstruct %v4float %p0xy %p0zw %float_0 %float_1
%out_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos0 %q0_v0
OpStore %outColor %v3_white
OpEmitVertex

; Vertex 1 of edge quad
%q0_v1 = OpCompositeConstruct %v4float %p2xy %p2zw %float_0 %float_1
OpStore %out_pos0 %q0_v1
OpStore %outColor %v3_white
OpEmitVertex

; Offset vertices for thickness
%thick = OpCompositeConstruct %v4float %float_0_015 %float_0_015 %float_0 %float_0
%q0_v2 = OpFAdd %v4float %q0_v0 %thick
OpStore %out_pos0 %q0_v2
OpStore %outColor %v3_white
OpEmitVertex

%q0_v3 = OpFAdd %v4float %q0_v1 %thick
OpStore %out_pos0 %q0_v3
OpStore %outColor %v3_white
OpEmitVertex

OpEndPrimitive

; --- Edge 1: vertices 2->4, adjacent vertex 3 ---
%gl_in_4_pos = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos_4 = OpLoad %v4float %gl_in_4_pos
%p4x = OpCompositeExtract %float %pos_4 0
%p4y = OpCompositeExtract %float %pos_4 1

; Vertex 0 of edge quad
%q1_v0 = OpCompositeConstruct %v4float %p2xy %p2zw %float_0 %float_1
OpStore %out_pos0 %q1_v0
OpStore %outColor %v3_white
OpEmitVertex

; Vertex 1 of edge quad
%q1_v1 = OpCompositeConstruct %v4float %p4x %p4y %float_0 %float_1
OpStore %out_pos0 %q1_v1
OpStore %outColor %v3_white
OpEmitVertex

%q1_v2 = OpFAdd %v4float %q1_v0 %thick
OpStore %out_pos0 %q1_v2
OpStore %outColor %v3_white
OpEmitVertex

%q1_v3 = OpFAdd %v4float %q1_v1 %thick
OpStore %out_pos0 %q1_v3
OpStore %outColor %v3_white
OpEmitVertex

OpEndPrimitive

; --- Edge 2: vertices 4->0, adjacent vertex 5 ---
; Vertex 0 of edge quad
%q2_v0 = OpCompositeConstruct %v4float %p4x %p4y %float_0 %float_1
OpStore %out_pos0 %q2_v0
OpStore %outColor %v3_white
OpEmitVertex

; Vertex 1 of edge quad
%q2_v1 = OpCompositeConstruct %v4float %p0xy %p0zw %float_0 %float_1
OpStore %out_pos0 %q2_v1
OpStore %outColor %v3_white
OpEmitVertex

%q2_v2 = OpFAdd %v4float %q2_v0 %thick
OpStore %out_pos0 %q2_v2
OpStore %outColor %v3_white
OpEmitVertex

%q2_v3 = OpFAdd %v4float %q2_v1 %thick
OpStore %out_pos0 %q2_v3
OpStore %outColor %v3_white
OpEmitVertex

OpEndPrimitive

; --- Emit filled triangle with dimmed colors ---
; Vertex 0
%inColor_0 = OpAccessChain %ptr_in_v3 %inColor %int_0
%col_0 = OpLoad %v3float %inColor_0
%dim_col_0 = OpVectorTimesScalar %v3float %col_0 %float_0_3
OpStore %out_pos0 %pos_0
OpStore %outColor %dim_col_0
OpEmitVertex

; Vertex 2
%inColor_2 = OpAccessChain %ptr_in_v3 %inColor %int_2
%col_2 = OpLoad %v3float %inColor_2
%dim_col_2 = OpVectorTimesScalar %v3float %col_2 %float_0_3
OpStore %out_pos0 %pos_2
OpStore %outColor %dim_col_2
OpEmitVertex

; Vertex 4
%inColor_4 = OpAccessChain %ptr_in_v3 %inColor %int_4
%col_4 = OpLoad %v3float %inColor_4
%dim_col_4 = OpVectorTimesScalar %v3float %col_4 %float_0_3
OpStore %out_pos0 %pos_4
OpStore %outColor %dim_col_4
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
