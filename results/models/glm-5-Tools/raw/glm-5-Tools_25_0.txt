        static void IntegrateStep(Vector3[] positions, Vector3[] velocities, CelestialBody[] bodies, double dt)
        {

            for (int i = 0; i < n; i++)
            {
                var accel = new Vector3();
                for (int j = 0; j < n; j++)
                {
                    if (i == j) continue;
                    var r = positions[j] - positions[i];
                    double rMag = r.Magnitude;
                    if (rMag > 0)
                        accel = accel + r * (G * bodies[j].Mass / (rMag * rMag * rMag));
                }
                velocities[i] = velocities[i] + accel * (dt * 0.5);
            }
            
            // Drift: update positions
            for (int i = 0; i < n; i++)
            {
                positions[i] = positions[i] + velocities[i] * dt;
            }

            for (int i = 0; i < n; i++)

        static TrajectoryResult EvaluateSequence(List<CelestialBody> sequence, MissionInput input,
            CelestialBody earth, CelestialBody asteroid, CelestialBody sun, double impactTime)
        {
            var result = new TrajectoryResult();
            
            double currentTime = 0;
            var currentPos = earth.Position.Copy();
            var currentVel = earth.Velocity.Copy();
            double totalDV = 0;
            
            foreach (var planet in sequence)
            {
                double flybyTime = currentTime + input.WarningDays / (sequence.Count + 2);
                var planetPos = PredictPosition(planet, sun, flybyTime);
                
                var leg = SolveLambert(currentPos, planetPos, (flybyTime - currentTime) * DAY_SECONDS, MU_SUN);
                if (!leg.Success) return null;
                
                double legDV = (leg.DepartureVelocity - currentVel).Magnitude;
                if (totalDV + legDV > input.DeltaVBudget) return null;
                
                result.Burns.Add(new Burn { Time = currentTime, DeltaV = leg.DepartureVelocity - currentVel });
                totalDV += legDV;
                
                currentPos = planetPos;
                var planetVel = PredictVelocity(planet, sun, flybyTime);
                currentVel = planetVel + (leg.ArrivalVelocity - planetVel);
                currentTime = flybyTime;
            }
            
            var asteroidPos = PredictPosition(asteroid, sun, impactTime);
            double finalTof = impactTime - currentTime;
            if (finalTof < 10) return null;
            
            var finalLeg = SolveLambert(currentPos, asteroidPos, finalTof * DAY_SECONDS, MU_SUN);
            if (!finalLeg.Success) return null;
            
            double finalDV = (finalLeg.DepartureVelocity - currentVel).Magnitude;
            if (totalDV + finalDV > input.DeltaVBudget) return null;
            
            result.Burns.Add(new Burn { Time = currentTime, DeltaV = finalLeg.DepartureVelocity - currentVel });
            result.Burns.Add(new Burn { Time = impactTime - 0.1, DeltaV = new Vector3(0, 0, 0) });
            
            return result;
        }

        static TrajectoryResult SolveOptimalDirect(MissionInput input, CelestialBody earth,

                var asteroidPos = PredictPosition(asteroid, sun, impactTime);

                {
                    bestDV = dv;
                    bestLaunchTime = launchTime;
                    bestLaunchV = lambert.DepartureVelocity - earthVel;
                    bestTOF = tof;
                }
            }
            
            // Golden section search refinement
            double a = Math.Max(0, bestLaunchTime - input.WarningDays / 20);
            double b = Math.Min(input.WarningDays, bestLaunchTime + input.WarningDays / 20);
            double phi = (1 + Math.Sqrt(5)) / 2;
            
            for (int iter = 0; iter < 20 && b - a > 0.1; iter++)
            {
                double c = b - (b - a) / phi;
                double d = a + (b - a) / phi;
                
                double dvC = EvaluateLaunchTime(c, impactTime, earth, asteroid, sun);
                double dvD = EvaluateLaunchTime(d, impactTime, earth, asteroid, sun);
                
                if (dvC < dvD)
                    b = d;
                else
                    a = c;
            }
            
            double refinedLaunchTime = (a + b) / 2;
            double refinedTOF = impactTime - refinedLaunchTime;
            
            if (refinedTOF > 10)
            {
                var earthPosRefined = PredictPosition(earth, sun, refinedLaunchTime);
                var asteroidPosRefined = PredictPosition(asteroid, sun, impactTime);
                var lambertRefined = SolveLambert(earthPosRefined, asteroidPosRefined, refinedTOF * DAY_SECONDS, MU_SUN);
                
                if (lambertRefined.Success)
                {
                    var earthVelRefined = PredictVelocity(earth, sun, refinedLaunchTime);
                    bestLaunchTime = refinedLaunchTime;
                    bestLaunchV = lambertRefined.DepartureVelocity - earthVelRefined;
                    bestTOF = refinedTOF;
                }
            }
            
            result.Burns.Add(new Burn { Time = bestLaunchTime, DeltaV = bestLaunchV });
            
            int numCorrections = Math.Max(1, (int)(bestTOF / 200));
            for (int i = 1; i <= numCorrections; i++)
            {
                double correctionTime = bestLaunchTime + bestTOF * i / (numCorrections + 1);
                var correction = CalculateCorrectionBurn(input, earth, asteroid, sun,
                    bestLaunchTime, bestLaunchV, correctionTime, impactTime);
                result.Burns.Add(correction);
            }
            
            result.Burns.Add(new Burn { Time = impactTime - 0.01, DeltaV = new Vector3(0, 0, 0) });
            
            return result;
        }

        static double EvaluateLaunchTime(double launchTime, double impactTime, 
            CelestialBody earth, CelestialBody asteroid, CelestialBody sun)
        {
            double tof = impactTime - launchTime;
            if (tof < 10) return double.MaxValue;
            
            var earthPos = PredictPosition(earth, sun, launchTime);
            var asteroidPos = PredictPosition(asteroid, sun, impactTime);
            
            var lambert = SolveLambert(earthPos, asteroidPos, tof * DAY_SECONDS, MU_SUN);
            if (!lambert.Success) return double.MaxValue;
            
            var earthVel = PredictVelocity(earth, sun, launchTime);
            return (lambert.DepartureVelocity - earthVel).Magnitude;
        }

        static Burn CalculateCorrectionBurn(MissionInput input, CelestialBody earth, 
            CelestialBody asteroid, CelestialBody sun, double launchTime, Vector3 launchDV,
            double correctionTime, double impactTime)
        {
            var earthPos0 = PredictPosition(earth, sun, launchTime);
            var earthVel0 = PredictVelocity(earth, sun, launchTime);
            var scVel0 = earthVel0 + launchDV;
            
            var scPosAtCorrection = PropagateKepler(earthPos0, scVel0, (correctionTime - launchTime) * DAY_SECONDS, MU_SUN);
            var scVelAtCorrection = PropagateVelocity(earthPos0, scVel0, (correctionTime - launchTime) * DAY_SECONDS, MU_SUN);
            
            var asteroidPosAtImpact = PredictPosition(asteroid, sun, impactTime);
            
            var leg = SolveLambert(scPosAtCorrection, asteroidPosAtImpact, 
                (impactTime - correctionTime) * DAY_SECONDS, MU_SUN);
            
            if (leg.Success)
            {
                var correction = leg.DepartureVelocity - scVelAtCorrection;
                if (correction.Magnitude > 100)
                {
                    correction = correction.Normalize() * Math.Min(correction.Magnitude, input.DeltaVBudget * 0.1);
                }
                return new Burn { Time = correctionTime, DeltaV = correction };
            }
            
            return new Burn { Time = correctionTime, DeltaV = new Vector3(0.1, 0, 0) };
        }

        static LambertResult SolveLambert(Vector3 r1, Vector3 r2, double tof, double mu)
        {
            var result = new LambertResult();
            
            try
            {
                double r1Mag = r1.Magnitude;
                double r2Mag = r2.Magnitude;
                
                double cosDnu = Vector3.Dot(r1, r2) / (r1Mag * r2Mag);
                cosDnu = Math.Max(-1, Math.Min(1, cosDnu));
                double dnu = Math.Acos(cosDnu);
                
                if (Vector3.Cross(r1, r2).Z < 0)
                    dnu = 2 * Math.PI - dnu;
                
                double c = Math.Sqrt(r1Mag * r1Mag + r2Mag * r2Mag - 2 * r1Mag * r2Mag * cosDnu);
                double s = (r1Mag + r2Mag + c) / 2;
                
                double aMin = s / 2;
                double a = aMin * 1.2;
                
                for (int iter = 0; iter < 50; iter++)
                {
                    double beta = 2 * Math.Asin(Math.Sqrt(Math.Max(0, (s - c) / (2 * a))));
                    double dtEst = Math.Sqrt(a * a * a / mu) * ((dnu - Math.Sin(dnu)) + (beta - Math.Sin(beta)));
                    
                    if (Math.Abs(dtEst - tof) < 1)
                        break;
                    
                    double error = dtEst - tof;
                    double ddtDa = 1.5 * dtEst / a;
                    if (Math.Abs(ddtDa) > 1e-10)
                        a = a - error / ddtDa;
                    
                    if (a < aMin) a = aMin * 1.01;
                    if (a > aMin * 10) a = aMin * 5;
                }
                
                double p = 4 * a * (s - r1Mag) * (s - r2Mag) / Math.Max(c * c, 1e-10);
                
                double f = 1 - r2Mag / Math.Max(p, 1e-10) * (1 - cosDnu);
                double g = r1Mag * r2Mag * Math.Sin(dnu) / Math.Sqrt(mu * Math.Max(p, 1e-10));
                
                result.DepartureVelocity = r1 * (-1.0 / Math.Max(g, 1e-10)) + r2 * (1.0 / Math.Max(g, 1e-10));
                result.ArrivalVelocity = r1 * (f / Math.Max(g, 1e-10)) + r2 * (-1);
                
                result.Success = true;
                result.DeltaV = result.DepartureVelocity.Magnitude;
            }
            catch
            {
                result.Success = false;
            }
            
            return result;
        }

        static Vector3 PredictPosition(CelestialBody body, CelestialBody sun, double timeDays)
        {
            double dt = timeDays * DAY_SECONDS;
            return PropagateKepler(body.Position, body.Velocity, dt, MU_SUN);
        }

        static Vector3 PredictVelocity(CelestialBody body, CelestialBody sun, double timeDays)
        {
            double dt = timeDays * DAY_SECONDS;
            return PropagateVelocity(body.Position, body.Velocity, dt, MU_SUN);
        }

        static Vector3 PropagateKepler(Vector3 r0, Vector3 v0, double dt, double mu)
        {
            double r0Mag = r0.Magnitude;
            if (r0Mag < 1) return r0;
            
            double v0Mag = v0.Magnitude;
            double alpha = 2.0 / r0Mag - v0Mag * v0Mag / mu;
            
            double a = 1.0 / Math.Abs(alpha);
            double n = Math.Sqrt(mu / (a * a * a));
            double M = n * dt;
            double E = SolveKeplerSimple(M);
            
            double cosE = Math.Cos(E);
            double sinE = Math.Sin(E);
            
            double f = 1 - a / r0Mag * (1 - cosE);
            double g = dt;
            
            return r0 * f + v0 * g;
        }

        static Vector3 PropagateVelocity(Vector3 r0, Vector3 v0, double dt, double mu)
        {
            double r0Mag = r0.Magnitude;
            if (r0Mag < 1) return v0;
            
            double v0Mag = v0.Magnitude;
            double alpha = 2.0 / r0Mag - v0Mag * v0Mag / mu;
            
            double a = 1.0 / Math.Abs(alpha);
            double n = Math.Sqrt(mu / (a * a * a));
            double M = n * dt;
            double E = SolveKeplerSimple(M);
            
            double cosE = Math.Cos(E);
            
            double r = a * (1 - cosE);
            double fdot = -Math.Sqrt(mu * a) * Math.Sin(E) / Math.Max(r * r0Mag, 1e-10);
            double gdot = 1 - a / Math.Max(r, 1e-10) * (1 - cosE);
            
            return r0 * fdot + v0 * gdot;
        }

        static double SolveKeplerSimple(double M)
        {
            double E = M;
            for (int i = 0; i < 30; i++)
            {
                double sinE = Math.Sin(E);
                double dE = (M - E + sinE) / (1 + Math.Cos(E));
                E += dE;
                if (Math.Abs(dE) < 1e-12) break;
            }
            return E;
        }
    }

            );
        }

## Key Features of This Solver:

1. **Mission Complexity Assessment**: Automatically categorizes missions as Simple, Moderate, Complex, or Advanced based on warning time, delta-v budget, and asteroid mass.

2. **Multiple Solver Strategies**:
   - **Simple**: Direct Lambert transfer with grid search optimization
   - **Moderate**: 2D grid search over launch time and time-of-flight with trajectory corrections
   - **Complex**: Gravity assist evaluation with Venus/Mars/Jupiter flyby options
   - **Advanced**: Multi-flyby sequence optimization with planetary gravity assists

3. **Orbital Mechanics**:
   - Lambert problem solver for transfer orbit computation
   - Kepler propagation for state prediction
   - KDK (Kick-Drift-Kick) N-body integrator for impact time estimation

4. **Trajectory Optimization**:
   - Golden section search for launch window refinement
   - Multiple mid-course corrections based on transfer time
   - Delta-v budget management

5. **Output**: Provides burn times (days since t=0) and delta-v vectors in m/s for launch, corrections, and impact.
