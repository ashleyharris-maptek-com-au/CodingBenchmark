```python
import sys
from collections import deque

def solve():
    """
    3-SAT solver using DPLL with unit propagation.
    Optimized for large instances (up to 20M variables, 85M clauses).
    """
    # Read input efficiently
    data = sys.stdin.buffer.read().split()
    if len(data) < 2:
        return
    
    it = iter(data)
    num_vars = int(next(it))
    num_clauses = int(next(it))
    
    # Read clauses
    clauses = []
    for _ in range(num_clauses):
        clause = [int(next(it)), int(next(it)), int(next(it))]
        clauses.append(clause)
    
    n = num_vars
    m = len(clauses)
    
    # assignment[v] = 0 (False), 1 (True), or -1 (unassigned)
    assignment = [-1] * (n + 1)
    
    # Build occurrence lists for efficient propagation
    pos_in = [[] for _ in range(n + 1)]
    neg_in = [[] for _ in range(n + 1)]
    
    for ci, clause in enumerate(clauses):
        for lit in clause:
            v = abs(lit)
            if lit > 0:
                pos_in[v].append(ci)
            else:
                neg_in[v].append(ci)
    
    # Clause state tracking
    satisfied = [False] * m
    num_unassigned = [3] * m
    num_false = [0] * m
    
    # Backtracking stack: (var, val, changes_list)
    stack = []
    
    # Unit propagation queue
    unit_queue = deque()
    
    def assign(var, val):
        """Assign a variable and update clause states. Returns changes or None if conflict."""
        changes = []
        assignment[var] = val
        
        # Update clauses containing positive literal var
        for ci in pos_in[var]:
            if satisfied[ci]:
                continue
            old_sat = satisfied[ci]
            
            num_unassigned[ci] -= 1
            if val == 1:
                satisfied[ci] = True
            else:
                num_false[ci] += 1
            
            changes.append((ci, 1 if val == 0 else 0, old_sat))
            
            # Check for unit clause or conflict
            if not satisfied[ci]:
                if num_unassigned[ci] == 1 and num_false[ci] == 2:
                    unit_queue.append(ci)
                elif num_unassigned[ci] == 0 and num_false[ci] == 3:
                    return None  # Conflict
        
        # Update clauses containing negative literal -var
        for ci in neg_in[var]:
            if satisfied[ci]:
                continue
            old_sat = satisfied[ci]
            
            num_unassigned[ci] -= 1
            if val == 0:
                satisfied[ci] = True
            else:
                num_false[ci] += 1
            
            changes.append((ci, 1 if val == 1 else 0, old_sat))
            
            if not satisfied[ci]:
                if num_unassigned[ci] == 1 and num_false[ci] == 2:
                    unit_queue.append(ci)
                elif num_unassigned[ci] == 0 and num_false[ci] == 3:
                    return None  # Conflict
        
        return changes
    
    def unassign(var, changes):
        """Unassign a variable and restore clause states."""
        assignment[var] = -1
        for ci, df, old_sat in changes:
            num_unassigned[ci] += 1
            num_false[ci] -= df
            satisfied[ci] = old_sat
    
    def propagate():
        """Process unit propagation queue. Returns False if conflict."""
        while unit_queue:
            ci = unit_queue.popleft()
            if satisfied[ci]:
                continue
            if num_unassigned[ci] != 1 or num_false[ci] != 2:
                continue
            
            # Find the unassigned literal and assign it
            for lit in clauses[ci]:
                v = abs(lit)
                if assignment[v] == -1:
                    val = 1 if lit > 0 else 0
                    changes = assign(v, val)
                    if changes is None:
                        return False
                    stack.append((v, val, changes))
                    break
        return True
    
    def find_unassigned():
        """Find an unassigned variable."""
        for v in range(1, n + 1):
            if assignment[v] == -1:
                return v
        return -1
    
    # Main DPLL loop
    while True:
        # Unit propagation
        if not propagate():
            # Conflict - backtrack
            while stack:
                var, val, changes = stack.pop()
                unassign(var, changes)
                if val == 1:
                    # Try False
                    changes = assign(var, 0)
                    if changes is not None:
                        stack.append((var, 0, changes))
                        break
            else:
                # No more backtracking possible - unsatisfiable
                print("UNSAT")
                return
            continue
        
        # Find next unassigned variable
        var = find_unassigned()
        if var == -1:
            # All variables assigned - solution found
            print("SAT")
            print(" ".join(str(assignment[v]) for v in range(1, n + 1)))
            return
        
        # Branch: try True first
        changes = assign(var, 1)
        if changes is None:
            # Conflict with True - try False
            assignment[var] = -1
            changes = assign(var, 0)
            if changes is None:
                # Both values conflict - backtrack
                assignment[var] = -1
                while stack:
                    var2, val2, changes2 = stack.pop()
                    unassign(var2, changes2)
                    if val2 == 1:
                        changes = assign(var2, 0)
                        if changes is not None:
                            stack.append((var2, 0, changes))
                            break
                else:
                    print("UNSAT")
                    return
                continue
        
        if changes is not None:
            stack.append((var, 1, changes))

if __name__ == "__main__":
    solve()
```

**Key optimizations:**
1. **Efficient input parsing** using `sys.stdin.buffer.read().split()` for handling large inputs
2. **Occurrence lists** (`pos_in`, `neg_in`) for O(1) access to clauses containing a variable
3. **Incremental clause state tracking** - each clause tracks unassigned/false literal counts
4. **Unit propagation queue** using `deque` for O(1) operations
5. **Efficient backtracking** - only stores changes made, not full state snapshots
6. **Conflict detection** happens during assignment, not in a separate pass
