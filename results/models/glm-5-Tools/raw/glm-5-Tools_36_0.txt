#include <iostream>
#include <vector>
#include <algorithm>
#include <bitset>
#include <random>

using namespace std;

const int MAXN = 150;

// Graph representation
struct Graph {
    int n, m;
    vector<vector<int>> adj;
    vector<int> degree;
    
    Graph(int n = 0) : n(n), m(0), adj(n), degree(n, 0) {}
    
    void addEdge(int u, int v) {
        if (u == v || u < 0 || v < 0 || u >= n || v >= n) return;
        adj[u].push_back(v);
        adj[v].push_back(u);
        degree[u]++;
        degree[v]++;
        m++;
    }
    
    bool hasEdge(int u, int v) const {
        for (int w : adj[u]) if (w == v) return true;
        return false;
    }
};

// Exact branch and bound solver using bitsets
class ExactMISSolver {
public:
    int n;
    bitset<MAXN> adjBit[MAXN];
    bitset<MAXN> bestSet;
    int bestSize;
    long long nodes;
    long long maxNodes;
    
    void solve(bitset<MAXN> current, bitset<MAXN> candidates) {
        nodes++;
        if (nodes > maxNodes) return;
        
        int candidateCount = (int)candidates.count();
        if (candidateCount == 0) {
            int sz = (int)current.count();
            if (sz > bestSize) {
                bestSize = sz;
                bestSet = current;
            }
            return;
        }
        
        // Upper bound
        int currentSize = (int)current.count();
        if (currentSize + candidateCount <= bestSize) return;
        
        // Find vertex with maximum degree in candidates (for better pruning)
        int v = -1, maxDeg = -1;
        for (int i = 0; i < n; i++) {
            if (candidates[i]) {
                int deg = (int)(adjBit[i] & candidates).count();
                if (deg > maxDeg) {
                    maxDeg = deg;
                    v = i;
                }
            }
        }
        if (v == -1) return;
        
        // Branch 1: Include v
        bitset<MAXN> newCurrent = current;
        newCurrent[v] = 1;
        bitset<MAXN> newCandidates = candidates;
        newCandidates &= ~adjBit[v];
        newCandidates[v] = 0;
        solve(newCurrent, newCandidates);
        
        // Branch 2: Exclude v
        candidates[v] = 0;
        solve(current, candidates);
    }
    
    vector<int> findMIS(const Graph& g, long long maxNodesHint = 100000000LL) {
        n = min(g.n, MAXN);
        bestSize = 0;
        bestSet.reset();
        nodes = 0;
        maxNodes = maxNodesHint;
        
        for (int i = 0; i < n; i++) {
            adjBit[i].reset();
            for (int v : g.adj[i]) {
                if (v < MAXN) adjBit[i][v] = 1;
            }
        }
        
        bitset<MAXN> current, candidates;
        for (int i = 0; i < n; i++) candidates[i] = 1;
        
        solve(current, candidates);
        
        vector<int> result;
        for (int i = 0; i < n; i++) {
            if (bestSet[i]) result.push_back(i);
        }
        return result;
    }
};

// Greedy MIS with minimum degree heuristic
vector<int> greedyMIS(const Graph& g) {
    vector<bool> excluded(g.n, false);
    vector<int> result;
    
    // Sort vertices by degree
    vector<int> order(g.n);
    for (int i = 0; i < g.n; i++) order[i] = i;
    sort(order.begin(), order.end(), [&](int a, int b) {
        return g.degree[a] < g.degree[b];
    });
    
    for (int v : order) {
        if (!excluded[v]) {
            result.push_back(v);
            excluded[v] = true;
            for (int u : g.adj[v]) {
                excluded[u] = true;
            }
        }
    }
    return result;
}

// Randomized greedy with multiple trials
vector<int> randomizedGreedyMIS(const Graph& g, int trials = 50) {
    vector<int> best;
    mt19937 rng(12345);
    
    for (int t = 0; t < trials; t++) {
        vector<int> order(g.n);
        for (int i = 0; i < g.n; i++) order[i] = i;
        shuffle(order.begin(), order.end(), rng);
        
        vector<bool> excluded(g.n, false);
        vector<int> result;
        
        for (int v : order) {
            if (!excluded[v]) {
                result.push_back(v);
                excluded[v] = true;
                for (int u : g.adj[v]) excluded[u] = true;
            }
        }
        
        if (result.size() > best.size()) best = result;
    }
    
    // Also try degree-based orderings
    vector<int> degOrder(g.n);
    for (int i = 0; i < g.n; i++) degOrder[i] = i;
    
    // Min degree
    sort(degOrder.begin(), degOrder.end(), [&](int a, int b) {
        return g.degree[a] < g.degree[b];
    });
    vector<bool> excl(g.n, false);
    vector<int> res;
    for (int v : degOrder) {
        if (!excl[v]) {
            res.push_back(v);
            excl[v] = true;
            for (int u : g.adj[v]) excl[u] = true;
        }
    }
    if (res.size() > best.size()) best = res;
    
    // Max degree
    sort(degOrder.begin(), degOrder.end(), [&](int a, int b) {
        return g.degree[a] > g.degree[b];
    });
    excl.assign(g.n, false);
    res.clear();
    for (int v : degOrder) {
        if (!excl[v]) {
            res.push_back(v);
            excl[v] = true;
            for (int u : g.adj[v]) excl[u] = true;
        }
    }
    if (res.size() > best.size()) best = res;
    
    return best;
}

// Local search to improve solution
vector<int> localSearch(const Graph& g, vector<int> current, int maxIter = 5000) {
    if (current.empty()) return current;
    
    int n = g.n;
    vector<bool> inSet(n, false);
    vector<int> neighborCount(n, 0);
    
    for (int v : current) inSet[v] = true;
    for (int v : current) {
        for (int u : g.adj[v]) neighborCount[u]++;
    }
    
    bool improved = true;
    int iterations = 0;
    
    while (improved && iterations < maxIter) {
        improved = false;
        iterations++;
        
        // Try 2-for-1 swap: remove v, add u1 and u2
        for (int idx = 0; idx < (int)current.size() && !improved; idx++) {
            int v = current[idx];
            
            // Find vertices with exactly 1 neighbor in set (that neighbor being v)
            vector<int> candidates;
            for (int u : g.adj[v]) {
                if (!inSet[u] && neighborCount[u] == 1) {
                    candidates.push_back(u);
                }
            }
            
            // Try pairs of candidates that aren't adjacent
            for (int i = 0; i < (int)candidates.size() && !improved; i++) {
                for (int j = i + 1; j < (int)candidates.size() && !improved; j++) {
                    int u1 = candidates[i], u2 = candidates[j];
                    if (!g.hasEdge(u1, u2)) {
                        // Perform swap
                        inSet[v] = false;
                        inSet[u1] = true;
                        inSet[u2] = true;
                        
                        current[idx] = current.back();
                        current.pop_back();
                        current.push_back(u1);
                        current.push_back(u2);
                        
                        // Update neighbor counts
                        for (int w : g.adj[v]) neighborCount[w]--;
                        for (int w : g.adj[u1]) neighborCount[w]++;
                        for (int w : g.adj[u2]) neighborCount[w]++;
                        
                        improved = true;
                    }
                }
            }
        }
        
        // Try adding vertices with 0 neighbors in set
        if (!improved) {
            for (int v = 0; v < n; v++) {
                if (!inSet[v] && neighborCount[v] == 0) {
                    inSet[v] = true;
                    current.push_back(v);
                    for (int u : g.adj[v]) neighborCount[u]++;
                    improved = true;
                    break;
                }
            }
        }
    }
    
    return current;
}

// Main solver
vector<int> solveMIS(Graph& g) {
    int n = g.n;
    
    if (n == 0) return {};
    if (g.m == 0) {
        vector<int> result(n);
        for (int i = 0; i < n; i++) result[i] = i;
        return result;
    }
    
    // Collect isolated vertices
    vector<int> isolated;
    for (int i = 0; i < n; i++) {
        if (g.degree[i] == 0) isolated.push_back(i);
    }
    
    // If graph is small enough, use exact algorithm
    if (n <= MAXN) {
        ExactMISSolver solver;
        vector<int> result = solver.findMIS(g);
        for (int v : isolated) {
            result.push_back(v);
        }
        return result;
    }
    
    // For larger graphs, use heuristics
    vector<int> best = randomizedGreedyMIS(g, 100);
    best = localSearch(g, best, 10000);
    
    vector<int> greedy1 = greedyMIS(g);
    greedy1 = localSearch(g, greedy1, 5000);
    if (greedy1.size() > best.size()) best = greedy1;
    
    vector<int> greedy2 = greedyMISMaxDegree(g);
    greedy2 = localSearch(g, greedy2, 5000);
    if (greedy2.size() > best.size()) best = greedy2;
    
    // Add isolated vertices
    for (int v : isolated) {
        best.push_back(v);
    }
    
    return best;
}
