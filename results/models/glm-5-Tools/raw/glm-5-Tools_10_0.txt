#include <algorithm>
#include <vector>
#include <iostream>
#include <chrono>
#include <random>
#include <numeric>
#include <cstdint>
#include <cstring>

using namespace std;

// Rectangle structure
struct Rect {
    int w, h;
    int id;
    int original_idx;
    
    int area() const { return w * h; }
    int min_dim() const { return min(w, h); }
    int max_dim() const { return max(w, h); }
};

// Placed rectangle
struct Placement {
    int board_idx;
    int x, y;
    int rotated;
};

// Board with occupied space tracking
struct Board {
    int width, height;
    vector<int> occupied; // height profile for skyline algorithm
    
    Board(int w, int h) : width(w), height(h), occupied(w, 0) {}
    
    bool can_place(int x, int pw, int ph) {
        if (x + pw > width) return false;
        int max_h = 0;
        for (int i = x; i < x + pw && max_h < height; i++) {
            max_h = max(max_h, occupied[i]);
        }
        return max_h + ph <= height;
    }
    
    void place(int x, int pw, int ph) {
        int max_h = 0;
        for (int i = x; i < x + pw; i++) {
            max_h = max(max_h, occupied[i]);
        }
        for (int i = x; i < x + pw; i++) {
            occupied[i] = max_h + ph;
        }
    }
    
    int find_best_x(int pw, int ph, int& best_x) {
        best_x = -1;
        int best_h = INT32_MAX;
        
        for (int x = 0; x <= width - pw; x++) {
            int max_h = 0;
            for (int i = x; i < x + pw; i++) {
                max_h = max(max_h, occupied[i]);
            }
            if (max_h + ph <= height && max_h < best_h) {
                best_h = max_h;
                best_x = x;
            }
        }
        return best_x >= 0 ? 1 : 0;
    }
};

// Configuration
int N, BW, BH;
vector<Rect> rects;
vector<Placement> placements;
int total_area = 0;

// Timing
auto start_time = chrono::high_resolution_clock::now();
double time_limit = 29.0;

double elapsed() {
    auto now = chrono::high_resolution_clock::now();
    return chrono::duration<double>(now - start_time).count();
}

// Sort rectangles by decreasing area
vector<int> sort_by_area() {
    vector<int> order(N);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [](int a, int b) {
        return rects[a].area() > rects[b].area();
    });
    return order;
}

// Sort rectangles by decreasing min dimension
vector<int> sort_by_min_dim() {
    vector<int> order(N);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [](int a, int b) {
        if (rects[a].min_dim() != rects[b].min_dim())
            return rects[a].min_dim() > rects[b].min_dim();
        return rects[a].max_dim() > rects[b].max_dim();
    });
    return order;
}

// Sort rectangles by decreasing max dimension
vector<int> sort_by_max_dim() {
    vector<int> order(N);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [](int a, int b) {
        if (rects[a].max_dim() != rects[b].max_dim())
            return rects[a].max_dim() > rects[b].max_dim();
        return rects[a].min_dim() > rects[b].min_dim();
    });
    return order;
}

// Skyline packing with bottom-left placement
int skyline_pack(const vector<int>& order, vector<Placement>& result) {
    vector<Board> boards;
    result.assign(N, {-1, -1, -1, 0});
    
    for (int idx : order) {
        Rect& r = rects[idx];
        int pw = r.w, ph = r.h;
        bool placed = false;
        int best_board = -1, best_x = -1, best_y = INT32_MAX, best_rot = 0;
        
        // Try both orientations
        for (int rot = 0; rot <= 1; rot++) {
            if (rot) swap(pw, ph);
            if (pw > BW || ph > BH) continue;
            
            for (int b = 0; b < (int)boards.size(); b++) {
                int x;
                if (boards[b].find_best_x(pw, ph, x)) {
                    int y = 0;
                    for (int i = x; i < x + pw; i++) {
                        y = max(y, boards[b].occupied[i]);
                    }
                    if (y < best_y || (y == best_y && b < best_board)) {
                        best_y = y;
                        best_board = b;
                        best_x = x;
                        best_rot = rot;
                        placed = true;
                    }
                }
            }
            if (rot) swap(pw, ph);
        }
        
        if (!placed) {
            // Try orientations for new board
            for (int rot = 0; rot <= 1; rot++) {
                pw = r.w; ph = r.h;
                if (rot) swap(pw, ph);
                if (pw <= BW && ph <= BH) {
                    boards.push_back(Board(BW, BH));
                    best_board = boards.size() - 1;
                    best_x = 0;
                    best_rot = rot;
                    placed = true;
                    break;
                }
            }
        }
        
        if (!placed) {
            return INT32_MAX; // Cannot place this rectangle
        }
        
        pw = r.w; ph = r.h;
        if (best_rot) swap(pw, ph);
        boards[best_board].place(best_x, pw, ph);
        result[idx] = {best_board, best_x, best_y, best_rot};
    }
    
    return (int)boards.size();
}

// FFDH (First Fit Decreasing Height) packing
int ffdh_pack(vector<int> order, vector<Placement>& result) {
    sort(order.begin(), order.end(), [](int a, int b) {
        return rects[a].h > rects[b].h;
    });
    
    result.assign(N, {-1, -1, -1, 0});
    
    struct Shelf {
        int y, h, used_w;
    };
    
    vector<vector<Shelf>> boards;
    
    for (int idx : order) {
        Rect& r = rects[idx];
        bool placed = false;
        
        for (int rot = 0; rot <= 1 && !placed; rot++) {
            int pw = rot ? r.h : r.w;
            int ph = rot ? r.w : r.h;
            
            if (pw > BW || ph > BH) continue;
            
            // Try existing boards and shelves
            for (int b = 0; b < (int)boards.size() && !placed; b++) {
                for (auto& shelf : boards[b]) {
                    if (ph <= shelf.h && shelf.used_w + pw <= BW) {
                        result[idx] = {b, shelf.used_w, shelf.y, rot};
                        shelf.used_w += pw;
                        placed = true;
                        break;
                    }
                }
                
                // Try new shelf on this board
                if (!placed && !boards[b].empty()) {
                    int last_y = boards[b].back().y + boards[b].back().h;
                    if (last_y + ph <= BH) {
                        boards[b].push_back({last_y, ph, pw});
                        result[idx] = {b, 0, last_y, rot};
                        placed = true;
                    }
                }
            }
            
            // New board
            if (!placed) {
                boards.push_back(vector<Shelf>());
                boards.back().push_back({0, ph, pw});
                result[idx] = {(int)boards.size() - 1, 0, 0, rot};
                placed = true;
            }
        }
    }
    
    return (int)boards.size();
}

// NFDH (Next Fit Decreasing Height) packing - very fast
int nfdh_pack(vector<int> order, vector<Placement>& result) {
    sort(order.begin(), order.end(), [](int a, int b) {
        return rects[a].h > rects[b].h;
    });
    
    result.assign(N, {-1, -1, -1, 0});
    
    int board = 0;
    int shelf_y = 0;
    int shelf_h = 0;
    int x_in_shelf = 0;
    
    for (int idx : order) {
        Rect& r = rects[idx];
        bool placed = false;
        
        for (int rot = 0; rot <= 1 && !placed; rot++) {
            int pw = rot ? r.h : r.w;
            int ph = rot ? r.w : r.h;
            
            if (pw > BW || ph > BH) continue;
            
            // Try current shelf
            if (shelf_h > 0 && ph <= shelf_h && x_in_shelf + pw <= BW) {
                result[idx] = {board, x_in_shelf, shelf_y, rot};
                x_in_shelf += pw;
                placed = true;
            }
            // Try new shelf on current board
            else if (shelf_y + shelf_h + ph <= BH) {
                shelf_y += shelf_h;
                shelf_h = ph;
                x_in_shelf = pw;
                result[idx] = {board, 0, shelf_y, rot};
                placed = true;
            }
            // New board
            else if (ph <= BH && pw <= BW) {
                board++;
                shelf_y = 0;
                shelf_h = ph;
                x_in_shelf = pw;
                result[idx] = {board, 0, 0, rot};
                placed = true;
            }
        }
    }
    
    return board + 1;
}

// Guillotine packing - fast heuristic
int guillotine_pack(vector<int> order, vector<Placement>& result) {
    sort(order.begin(), order.end(), [](int a, int b) {
        return max(rects[a].h, rects[a].w) > max(rects[b].h, rects[b].w);
    });
    
    result.assign(N, {-1, -1, -1, 0});
    vector<pair<int, int>> shelves; // (y_start, height)
    vector<int> shelf_used_width;
    int num_boards = 1;
    
    int cur_y = 0;
    int cur_shelf_h = 0;
    
    for (int idx : order) {
        Rect& r = rects[idx];
        
        bool placed = false;
        
        for (int rot = 0; rot <= 1 && !placed; rot++) {
            int pw = rot ? r.h : r.w;
            int ph = rot ? r.w : r.h;
            
            if (pw > BW || ph > BH) continue;
            
            // Try each shelf
            for (int s = 0; s < (int)shelves.size() && !placed; s++) {
                if (ph <= shelves[s].second && shelf_used_width[s] + pw <= BW) {
                    result[idx] = {0, shelf_used_width[s], shelves[s].first, rot};
                    shelf_used_width[s] += pw;
                    placed = true;
                }
            }
            
            // Try new shelf on current board
            if (!placed && cur_y + ph <= BH) {
                shelves.push_back({cur_y, ph});
                shelf_used_width.push_back(pw);
                result[idx] = {0, 0, cur_y, rot};
                cur_y += ph;
                placed = true;
            }
        }
        
        if (!placed) {
            num_boards++;
            cur_y = 0;
            shelves.clear();
            shelf_used_width.clear();
            
            for (int rot = 0; rot <= 1 && !placed; rot++) {
                int pw = rot ? r.h : r.w;
                int ph = rot ? r.w : r.h;
                
                if (pw <= BW && ph <= BH) {
                    shelves.push_back({0, ph});
                    shelf_used_width.push_back(pw);
                    result[idx] = {num_boards - 1, 0, 0, rot};
                    cur_y = ph;
                    placed = true;
                }
            }
        }
    }
    
    return num_boards;
}

// Main solve function
void solve() {
    vector<Placement> best_result;
    int best_boards = INT32_MAX;
    
    // Strategy selection based on problem size
    bool is_small = N <= 100;
    bool is_large = N >= 1000;
    
    auto try_solution = [&](vector<Placement>& sol, int boards) {
        if (boards < best_boards) {
            best_boards = boards;
            best_result = sol;
            return true;
        }
        return false;
    };
    
    // Try different packing strategies
    vector<Placement> sol;
    int boards;
    
    if (!is_large) {
        // Skyline pack with different orderings
        auto order = sort_by_area();
        boards = skyline_pack(order, sol);
        try_solution(sol, boards);
        
        if (elapsed() < time_limit - 10.0) {
            order = sort_by_min_dim();
            boards = skyline_pack(order, sol);
            try_solution(sol, boards);
        }
        
        if (elapsed() < time_limit - 10.0) {
            order = sort_by_max_dim();
            boards = skyline_pack(order, sol);
            try_solution(sol, boards);
        }
    }
    
    // FFDH - fast and good quality
    if (elapsed() < time_limit - 5.0) {
        auto order = sort_by_area();
        boards = ffdh_pack(order, sol);
        try_solution(sol, boards);
    }
    
    // NFDH - very fast
    if (elapsed() < time_limit - 3.0) {
        auto order = sort_by_area();
        boards = nfdh_pack(order, sol);
        try_solution(sol, boards);
    }
    
    // Guillotine
    if (elapsed() < time_limit - 3.0) {
        auto order = sort_by_area();
        boards = guillotine_pack(order, sol);
        try_solution(sol, boards);
    }
    
    // Reindex boards to be consecutive
    if (!best_result.empty()) {
        vector<int> board_map(N + 100, -1);
        int next_board = 0;
        for (int i = 0; i < N; i++) {
            int b = best_result[i].board_idx;
            if (board_map[b] == -1) {
                board_map[b] = next_board++;
            }
            best_result[i].board_idx = board_map[b];
        }
        best_boards = next_board;
    }
    
    // Output result
    cout << best_boards << "\n";
    for (int i = 0; i < N; i++) {
        cout << best_result[i].board_idx << " " 
             << best_result[i].x << " " 
             << best_result[i].y << " " 
             << best_result[i].rotated << "\n";
    }
}

}

This solution features:

1. **Multiple Packing Algorithms**:
   - **Skyline Packing**: Uses bottom-left heuristic with height profile tracking - best quality for small/medium problems
   - **FFDH**: First Fit Decreasing Height - good balance of speed and quality
   - **NFDH**: Next Fit Decreasing Height - very fast for large problems
   - **Guillotine Packing**: Fast shelf-based heuristic

2. **Adaptive Strategy Selection**:
   - Small problems (N ≤ 100): Uses all algorithms including skyline
   - Large problems (N ≥ 1000): Uses fast shelf-based algorithms only

3. **Time Management**: Tracks elapsed time and stops trying additional strategies when approaching the 30-second limit

4. **Orientation Handling**: All algorithms try both orientations (rotated=0 or 1) for each rectangle

5. **Board Reindexing**: Ensures output boards are numbered 0 to num_boards-1 consecutively
