Here is my complete Python autopilot implementation for an A320-class twin-engine airliner:

```python
import math
from collections import deque

class PIDController:
    def __init__(self, kp, ki, kd, imax=1.0, output_limit=1.0, derivative_filter_tc=0.1):
        self.kp = kp
        self.ki = ki
        self.kd = kd
        self.imax = imax
        self.output_limit = output_limit
        self.derivative_filter_tc = derivative_filter_tc
        self.integral = 0.0
        self.prev_error = None
        self.prev_derivative = 0.0
        
    def compute(self, error, dt, rate=None):
        p_term = self.kp * error
        self.integral += error * dt
        self.integral = max(-self.imax, min(self.imax, self.integral))
        i_term = self.ki * self.integral
        
        if rate is not None:
            d_term = self.kd * rate
        elif self.prev_error is not None and dt > 0:
            raw_derivative = (error - self.prev_error) / dt
            alpha = dt / (self.derivative_filter_tc + dt)
            self.prev_derivative = alpha * raw_derivative + (1 - alpha) * self.prev_derivative
            d_term = self.kd * self.prev_derivative
        else:
            d_term = 0.0
        self.prev_error = error
        
        output = p_term + i_term + d_term
        return max(-self.output_limit, min(self.output_limit, output))

class AutopilotState:
    def __init__(self):
        self.pitch_pid = PIDController(0.03, 0.01, 0.02, imax=0.5, output_limit=1.0)
        self.roll_pid = PIDController(0.05, 0.015, 0.025, imax=0.3, output_limit=1.0)
        
        self.target_pitch = 0.0
        self.target_roll = 0.0
        self.target_throttle = 0.5
        self.target_vs = 0.0
        
        self.phase = "unknown"
        self.last_sim_time = None
        self.time_in_phase = 0.0
        
        self.stall_recovery = False
        self.overspeed_recovery = False
        self.engine_out = False
        self.failed_engine = None

_ap_state = AutopilotState()

def median_of_valid(values, valid_flags):
    valid_vals = [v for v, vf in zip(values, valid_flags) if vf and v is not None]
    if not valid_vals:
        return None
    valid_vals.sort()
    n = len(valid_vals)
    return valid_vals[n // 2] if n % 2 == 1 else (valid_vals[n//2 - 1] + valid_vals[n//2]) / 2

def detect_outlier_idx(values, valid_flags, threshold=2.0):
    valid_pairs = [(i, v) for i, (v, vf) in enumerate(zip(values, valid_flags)) if vf and v is not None]
    if len(valid_pairs) < 3:
        return []
    vals = [v for i, v in valid_pairs]
    mean = sum(vals) / len(vals)
    std = math.sqrt(sum((v - mean)**2 for v in vals) / len(vals))
    if std < 1e-6:
        return []
    return [idx for idx, v in valid_pairs if abs(v - mean) > threshold * std]

def normalize_angle(angle):
    while angle > 180: angle -= 360
    while angle < -180: angle += 360
    return angle

def autopilot_step(state: dict, dt: float) -> dict:
    global _ap_state
    
    sim_time = state.get('sim_time_s', 0)
    if _ap_state.last_sim_time is not None and sim_time < _ap_state.last_sim_time - 1:
        _ap_state = AutopilotState()
    _ap_state.last_sim_time = sim_time
    _ap_state.time_in_phase += dt
    
    # ===== SENSOR FUSION =====
    fused = {}
    
    # Barometric altitude (triple redundancy with voting)
    baro_vals = [state.get('baro_alt_1_ft'), state.get('baro_alt_2_ft'), state.get('baro_alt_3_ft')]
    baro_valid = [state.get('baro_alt_1_valid', False), state.get('baro_alt_2_valid', False), 
                  state.get('baro_alt_3_valid', False)]
    fused['baro_alt'] = median_of_valid(baro_vals, baro_valid)
    
    # Radar altitude
    radar_vals = [state.get('radar_alt_1_ft'), state.get('radar_alt_2_ft'), state.get('radar_alt_3_ft')]
    radar_valid = [state.get('radar_alt_1_valid', False), state.get('radar_alt_2_valid', False),
                   state.get('radar_alt_3_valid', False)]
    fused['radar_alt'] = median_of_valid(radar_vals, radar_valid)
    
    # GPS altitude
    gps_alt_vals = [state.get('gps_1_alt_ft'), state.get('gps_2_alt_ft'), state.get('gps_3_alt_ft')]
    gps_valid = [state.get('gps_1_valid', False), state.get('gps_2_valid', False), state.get('gps_3_valid', False)]
    fused['gps_alt'] = median_of_valid(gps_alt_vals, gps_valid)
    
    # Best altitude selection with cross-check
    radar_alt = fused['radar_alt']
    baro_alt = fused['baro_alt']
    gps_alt = fused['gps_alt']
    
    if radar_alt is not None and radar_alt < 2500:
        fused['altitude'] = radar_alt
    elif baro_alt is not None:
        fused['altitude'] = baro_alt
    elif gps_alt is not None:
        fused['altitude'] = gps_alt
    else:
        fused['altitude'] = 0
    
    # Airspeed with outlier detection (handles frozen/blocked sensors)
    ias_vals = [state.get('ias_1_kt'), state.get('ias_2_kt'), state.get('ias_3_kt')]
    ias_valid = [state.get('ias_1_valid', False), state.get('ias_2_valid', False), state.get('ias_3_valid', False)]
    for idx in detect_outlier_idx(ias_vals, ias_valid):
        ias_valid[idx] = False
    fused['ias'] = median_of_valid(ias_vals, ias_valid)
    if fused['ias'] is None:
        fused['ias'] = state.get('ground_speed_kt', 200)
    
    # Mach number
    mach_vals = [state.get('mach_1'), state.get('mach_2'), state.get('mach_3')]
    mach_valid = [m is not None for m in mach_vals]
    fused['mach'] = median_of_valid(mach_vals, mach_valid)
    
    # Angle of Attack
    aoa_vals = [state.get('aoa_1_deg'), state.get('aoa_2_deg'), state.get('aoa_3_deg')]
    aoa_valid = [state.get('aoa_1_valid', False), state.get('aoa_2_valid', False), state.get('aoa_3_valid', False)]
    fused['aoa'] = median_of_valid(aoa_vals, aoa_valid) or 0
    
    # IRS/INS attitude
    irs_valid = [state.get('irs_1_valid', False), state.get('irs_2_valid', False), state.get('irs_3_valid', False)]
    fused['pitch'] = median_of_valid([state.get('irs_1_pitch_deg'), state.get('irs_2_pitch_deg'), state.get('irs_3_pitch_deg')], irs_valid) or 0
    fused['roll'] = median_of_valid([state.get('irs_1_roll_deg'), state.get('irs_2_roll_deg'), state.get('irs_3_roll_deg')], irs_valid) or 0
    fused['heading'] = median_of_valid([state.get('irs_1_heading_deg'), state.get('irs_2_heading_deg'), state.get('irs_3_heading_deg')], irs_valid) or 0
    fused['vs'] = median_of_valid([state.get('irs_1_vs_fpm'), state.get('irs_2_vs_fpm'), state.get('irs_3_vs_fpm')], irs_valid) or 0
    
    # Gyro rates
    gyro_valid = [state.get('gyro_1_valid', False), state.get('gyro_2_valid', False), state.get('gyro_3_valid', False)]
    fused['pitch_rate'] = median_of_valid([state.get('gyro_1_pitch_rate_dps'), state.get('gyro_2_pitch_rate_dps'), state.get('gyro_3_pitch_rate_dps')], gyro_valid) or 0
    fused['roll_rate'] = median_of_valid([state.get('gyro_1_roll_rate_dps'), state.get('gyro_2_roll_rate_dps'), state.get('gyro_3_roll_rate_dps')], gyro_valid) or 0
    fused['yaw_rate'] = median_of_valid([state.get('gyro_1_yaw_rate_dps'), state.get('gyro_2_yaw_rate_dps'), state.get('gyro_3_yaw_rate_dps')], gyro_valid) or 0
    
    # Engine status
    fused['eng1_running'] = state.get('engine_1_running', True)
    fused['eng2_running'] = state.get('engine_2_running', True)
    fused['eng1_fire'] = state.get('engine_1_fire', False)
    fused['eng2_fire'] = state.get('engine_2_fire', False)
    fused['eng1_n1'] = state.get('engine_1_n1_pct', 0) or 0
    fused['eng2_n1'] = state.get('engine_2_n1_pct', 0) or 0
    
    # Other parameters
    fused['flap_left'] = state.get('flap_pos_left_deg', 0) or 0
    fused['on_ground'] = (state.get('weight_on_wheels_left', False) or state.get('weight_on_wheels_nose', False) or state.get('weight_on_wheels_right', False))
    fused['target_alt'] = state.get('target_alt_ft', 0)
    fused['target_hdg'] = state.get('target_hdg_deg', 0)
    fused['target_ias'] = state.get('target_ias_kt', 200)
    fused['target_vs'] = state.get('target_vs_fpm', None)
    
    # ===== EMERGENCY DETECTION =====
    flap = fused['flap_left']
    ias = fused['ias'] or 200
    mach = fused['mach'] or 0
    aoa = fused['aoa']
    
    # Stall detection (AOA varies with flap position)
    stall_aoa = 13 + flap * 0.1
    _ap_state.stall_recovery = aoa > stall_aoa
    
    # Overspeed detection (Vne ~365 KIAS / M0.82)
    _ap_state.overspeed_recovery = (ias > 360) or (mach > 0.81)
    
    # Engine failure detection
    eng1_failed = fused['eng1_fire'] or not fused['eng1_running'] or fused['eng1_n1'] < 10
    eng2_failed = fused['eng2_fire'] or not fused['eng2_running'] or fused['eng2_n1'] < 10
    
    if eng1_failed and eng2_failed:
        _ap_state.engine_out = True
        _ap_state.failed_engine = 'both'
    elif eng1_failed:
        _ap_state.engine_out = True
        _ap_state.failed_engine = 1
    elif eng2_failed:
        _ap_state.engine_out = True
        _ap_state.failed_engine = 2
    else:
        _ap_state.engine_out = False
        _ap_state.failed_engine = None
    
    # ===== FLIGHT PHASE DETERMINATION =====
    alt = fused['altitude'] or 0
    radar_alt = fused['radar_alt']
    vs = fused['vs'] or 0
    ias = fused['ias'] or 0
    on_ground = fused['on_ground']
    target_alt = fused['target_alt']
    
    eff_alt = radar_alt if radar_alt is not None and radar_alt < 5000 else alt
    
    if on_ground:
        _ap_state.phase = 'takeoff_roll' if ias > 60 else 'ground'
    elif eff_alt < 50:
        _ap_state.phase = 'takeoff' if vs > 100 else 'flare'
    elif eff_alt < 2000 and vs < -400:
        _ap_state.phase = 'approach'
    elif abs(alt - target_alt) < 200 and abs(vs) < 400:
        _ap_state.phase = 'cruise'
    elif vs > 400:
        _ap_state.phase = 'climb'
    elif vs < -400:
        _ap_state.phase = 'descent'
    else:
        _ap_state.phase = 'cruise'
    
    # ===== TARGET COMPUTATION =====
    heading = fused['heading'] or 0
    target_alt = fused['target_alt']
    target_hdg = fused['target_hdg']
    target_ias = fused['target_ias']
    
    # Emergency modes override normal control
    if _ap_state.stall_recovery:
        _ap_state.target_pitch = -10
        _ap_state.target_roll = 0
        _ap_state.target_throttle = 1.0
    elif _ap_state.overspeed_recovery:
        _ap_state.target_pitch = 10
        _ap_state.target_roll = 0
        _ap_state.target_throttle = 0.2
    elif _ap_state.engine_out:
        if _ap_state.failed_engine == 'both':
            _ap_state.target_pitch = 2  # Best glide
            _ap_state.target_roll = 0
            _ap_state.target_throttle = 0
        else:
            _ap_state.target_pitch = 5
            _ap_state.target_roll = 0
            _ap_state.target_throttle = 0.95
    else:
        # Normal operations - altitude capture with VS control
        alt_error = target_alt - alt
        
        if alt_error > 1500:
            _ap_state.target_vs = 2000
        elif alt_error > 300:
            _ap_state.target_vs = min(1800, alt_error * 1.2)
        elif alt_error < -1500:
            _ap_state.target_vs = -2000
        elif alt_error < -300:
            _ap_state.target_vs = max(-1800, alt_error * 1.2)
        else:
            _ap_state.target_vs = 0
        
        vs_error = _ap_state.target_vs - vs
        
        # Base pitch for phase
        base_pitch = 4 if _ap_state.phase == 'climb' else (-3 if _ap_state.phase == 'descent' else (-2 if _ap_state.phase == 'approach' else 0))
        _ap_state.target_pitch = base_pitch + vs_error * 0.006
        
        # Adjust pitch for airspeed
        ias_error = target_ias - ias
        if ias_error > 20:
            _ap_state.target_pitch -= 4
        elif ias_error > 10:
            _ap_state.target_pitch -= 2
        elif ias_error < -20:
            _ap_state.target_pitch += 3
        elif ias_error < -10:
            _ap_state.target_pitch += 1.5
        
        _ap_state.target_pitch = max(-12, min(15, _ap_state.target_pitch))
        
        # Roll from heading error
        hdg_error = normalize_angle(target_hdg - heading)
        _ap_state.target_roll = max(-25, min(25, hdg_error * 0.5))
        
        # Throttle from airspeed
        if on_ground:
            _ap_state.target_throttle = 0.95 if _ap_state.phase == 'takeoff_roll' else 0.1
        else:
            if ias_error > 15:
                _ap_state.target_throttle = min(1.0, _ap_state.target_throttle + 0.03)
            elif ias_error > 5:
                _ap_state.target_throttle = min(0.9, _ap_state.target_throttle + 0.015)
            elif ias_error < -15:
                _ap_state.target_throttle = max(0.25, _ap_state.target_throttle - 0.03)
            elif ias_error < -5:
                _ap_state.target_throttle = max(0.4, _ap_state.target_throttle - 0.015)
    
    # ===== CONTROL COMPUTATION =====
    controls = {}
    
    pitch = fused['pitch'] or 0
    roll = fused['roll'] or 0
    pitch_rate = fused['pitch_rate'] or 0
    roll_rate = fused['roll_rate'] or 0
    yaw_rate = fused['yaw_rate'] or 0
    ias = fused['ias'] or 200
    
    # Control authority varies with airspeed
    ias_factor = min(1.3, max(0.6, 180 / max(ias, 80)))
    
    # Elevator
    elevator = _ap_state.pitch_pid.compute(_ap_state.target_pitch - pitch, dt, -pitch_rate) * ias_factor
    controls['elevator'] = max(-1, min(1, elevator))
    
    # Aileron
    aileron = _ap_state.roll_pid.compute(_ap_state.target_roll - roll, dt, -roll_rate) * ias_factor
    controls['aileron'] = max(-1, min(1, aileron))
    
    # Rudder (yaw damping + engine-out compensation)
    rudder = -yaw_rate * 0.1
    if _ap_state.engine_out and _ap_state.failed_engine != 'both':
        rudder += 0.35 if _ap_state.failed_engine == 1 else -0.35
    controls['rudder'] = max(-1, min(1, rudder))
    
    # Throttles
    t1, t2 = _ap_state.target_throttle, _ap_state.target_throttle
    if fused['eng1_fire']: t1 = 0
    if fused['eng2_fire']: t2 = 0
    if _ap_state.engine_out and _ap_state.failed_engine != 'both':
        if _ap_state.failed_engine == 1: t1 = 0
        else: t2 = 0
    controls['throttle_1'] = max(0, min(1, t1))
    controls['throttle_2'] = max(0, min(1, t2))
    
    # Elevator trim
    controls['elevator_trim'] = 0.003 if elevator > 0.2 else (-0.003 if elevator < -0.2 else 0)
    
    # Flaps (phase and speed dependent)
    target_flap = 0
    if _ap_state.phase in ['approach', 'flare']:
        if ias < 200: target_flap = 15
        if ias < 180: target_flap = 25
        if ias < 160: target_flap = 35
        if ias < 145: target_flap = 40
    elif _ap_state.phase == 'takeoff':
        target_flap = 15
    if abs(target_flap - fused['flap_left']) > 5:
        controls['flaps'] = target_flap
    
    # Speedbrake
    if _ap_state.overspeed_recovery:
        controls['speedbrake'] = 0.7
    elif _ap_state.phase == 'descent' and ias > fused['target_ias'] + 15:
        controls['speedbrake'] = 0.4
    else:
        controls['speedbrake'] = 0
    
    # Gear
    eff_alt = fused['radar_alt'] if fused['radar_alt'] is not None else alt
    if eff_alt is not None and eff_alt < 2500 and _ap_state.phase in ['approach', 'descent']:
        controls['gear'] = 1
    elif _ap_state.phase in ['climb', 'cruise']:
        controls['gear'] = 0
    
    return controls
```

**Key Design Features:**

1. **Sensor Fusion**: Uses median voting across triple-redundant sensors with outlier detection to handle failed sensors that report invalid data with valid flags.

2. **Altitude Source Selection**: Prioritizes radar altitude below 2500ft for terrain awareness, then barometric, then GPS.

3. **PID Control**: Implements anti-windup and filtered derivative terms for stable control.

4. **Emergency Handling**:
   - **Stall recovery**: Nose down, full thrust
   - **Overspeed recovery**: Nose up, reduce thrust, deploy speedbrakes
   - **Engine fire**: Shut down affected engine, apply opposite rudder for asymmetric thrust
   - **Dual engine failure**: Establish best glide attitude

5. **Flight Phase Detection**: Automatically determines ground, takeoff, climb, cruise, descent, approach, and flare phases.

6. **Speed-dependent Control Authority**: Reduces control surface authority at higher airspeeds to prevent over-control.

7. **Coordinated Flight**: Uses yaw rate feedback for turn coordination and engine-out compensation.
