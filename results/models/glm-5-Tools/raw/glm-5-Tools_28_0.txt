```rust
use std::io::{self, BufRead, Write, BufWriter};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    
    let mut lines = stdin.lock().lines();
    
    let header: Vec<usize> = lines.next().unwrap().unwrap()
        .split_whitespace()
        .map(|x| x.parse().unwrap())
        .collect();
    let n = header[0];
    
    if n == 0 {
        writeln!(out, "0").unwrap();
        return;
    }
    
    // Build adjacency list
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    
    for line in lines {
        let line = line.unwrap();
        let mut iter = line.split_whitespace();
        let u: usize = iter.next().unwrap().parse().unwrap();
        let v: usize = iter.next().unwrap().parse().unwrap();
        if u != v && u < n && v < n {
            adj[u].push(v);
            adj[v].push(u);
        }
    }
    
    // Sort and deduplicate
    for neighbors in &mut adj {
        neighbors.sort_unstable();
        neighbors.dedup();
    }
    
    let mut in_cover = vec![false; n];
    let mut degree: Vec<i32> = adj.iter().map(|x| x.len() as i32).collect();
    let mut active = vec![true; n];
    
    // Kernelization: degree-1 vertex reduction
    let mut queue: VecDeque<usize> = VecDeque::new();
    for v in 0..n {
        if degree[v] == 1 {
            queue.push_back(v);
        }
    }
    
    while let Some(v) = queue.pop_front() {
        if !active[v] || degree[v] != 1 {
            continue;
        }
        if let Some(&neighbor) = adj[v].iter().find(|&&u| active[u]) {
            in_cover[neighbor] = true;
            active[v] = false;
            active[neighbor] = false;
            for &w in &adj[neighbor] {
                if active[w] {
                    degree[w] -= 1;
                    if degree[w] == 1 {
                        queue.push_back(w);
                    }
                }
            }
        }
    }
    
    // Effective degree for remaining vertices
    let eff_degree: Vec<usize> = (0..n)
        .map(|v| if active[v] { adj[v].iter().filter(|&&u| active[u]).count() } else { 0 })
        .collect();
    
    // Matching-based vertex selection
    let mut matched = vec![false; n];
    for v in 0..n {
        if !active[v] || matched[v] {
            continue;
        }
        if let Some(&u) = adj[v].iter()
            .filter(|&&u| active[u] && !matched[u])
            .max_by_key(|&&u| eff_degree[u])
        {
            matched[v] = true;
            matched[u] = true;
            in_cover[if eff_degree[v] >= eff_degree[u] { v } else { u }] = true;
        }
    }
    
    // Cover any remaining uncovered edges
    for v in 0..n {
        if !active[v] {
            continue;
        }
        for &u in &adj[v] {
            if active[u] && !in_cover[v] && !in_cover[u] {
                in_cover[if eff_degree[v] >= eff_degree[u] { v } else { u }] = true;
            }
        }
    }
    
    // Remove redundant vertices
    loop {
        let mut changed = false;
        for v in 0..n {
            if in_cover[v] && !adj[v].is_empty() && adj[v].iter().all(|&u| in_cover[u]) {
                in_cover[v] = false;
                changed = true;
            }
        }
        if !changed { break; }
    }
    
    // Output
    let cover: Vec<usize> = (0..n).filter(|&v| in_cover[v]).collect();
    writeln!(out, "{}", cover.len()).unwrap();
    if !cover.is_empty() {
        writeln!(out, "{}", cover.iter().map(|v| v.to_string()).collect::<Vec<_>>().join(" ")).unwrap();
    }
}
```

**Algorithm summary:**

1. **Kernelization**: Process degree-1 vertices optimally by including their neighbors in the cover. This cascades as vertices are removed.

2. **Matching-based selection**: Build a maximal matching on the remaining graph, but instead of adding both endpoints, choose the higher-degree vertex. This gives better solutions than the naive 2-approximation.

3. **Greedy completion**: Cover any remaining uncovered edges by selecting the higher-degree endpoint.

4. **Redundancy removal**: Iteratively remove vertices from the cover if all their neighbors are already covered.

**Complexity**: O(n + m) for graph construction, O(n + m) for kernelization, O(n + m) for matching, O(n·d) for redundancy removal where d is the max degree. Overall linear for sparse graphs.
