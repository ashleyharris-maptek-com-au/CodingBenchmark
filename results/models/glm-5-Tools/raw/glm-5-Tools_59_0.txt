const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const probability = data.probability;
const drones = data.drones;
const maxStepsPerDrone = data.maxStepsPerDrone;

const rows = probability.length;
const cols = probability[0].length;

const directions = [[0, 0], [1, 0], [-1, 0], [0, 1], [0, -1]];

function isValid(x, y) {
    return x >= 0 && x < cols && y >= 0 && y < rows && probability[y][x] !== -1;
}

// BFS to compute distances and find path to target
function bfs(startX, startY, targets, visited) {
    const dist = new Map();
    const parent = new Map();
    const queue = [[startX, startY, 0]];
    const seen = new Set();
    seen.add(`${startX},${startY}`);
    
    while (queue.length > 0) {
        const [x, y, d] = queue.shift();
        const key = `${x},${y}`;
        dist.set(key, d);
        
        for (let i = 1; i < directions.length; i++) {
            const [dx, dy] = directions[i];
            const nx = x + dx;
            const ny = y + dy;
            const nkey = `${nx},${ny}`;
            if (isValid(nx, ny) && !seen.has(nkey)) {
                seen.add(nkey);
                parent.set(nkey, key);
                queue.push([nx, ny, d + 1]);
            }
        }
    }
    return dist;
}

highValueCells.sort((a, b) => b.val - a.val);

// Plan targets for each drone using a greedy assignment
function assignTargets() {
    const assignments = new Array(drones.length).fill(null);
    const assigned = new Set();
    
    for (const cell of highValueCells) {
        if (assigned.has(`${cell.x},${cell.y}`)) continue;
        if (globalVisited.has(`${cell.x},${cell.y}`)) continue;
        
        // Find closest drone
        let bestDrone = -1;
        let bestDist = Infinity;
        
        for (let di = 0; di < drones.length; di++) {
            if (assignments[di]) continue;
            const dist = distMaps[di].get(`${cell.x},${cell.y}`);
            if (dist !== undefined && dist < bestDist) {
                bestDist = dist;
                bestDrone = di;
            }
        }
        
        if (bestDrone >= 0 && bestDist <= maxStepsPerDrone * 2) {
            assignments[bestDrone] = cell;
            assigned.add(`${cell.x},${cell.y}`);
        }
    }
    
    return assignments;
}

// A* pathfinding to target
function findPath(startX, startY, targetX, targetY, visited) {
    const openSet = [[startX, startY, 0, Math.abs(targetX - startX) + Math.abs(targetY - startY), null]];
    const cameFrom = new Map();
    const gScore = new Map();
    gScore.set(`${startX},${startY}`, 0);
    
    while (openSet.length > 0) {
        openSet.sort((a, b) => (a[2] + a[3]) - (b[2] + b[3]));
        const [x, y, g, h, dir] = openSet.shift();
        
        if (x === targetX && y === targetY) {
            // Reconstruct path
            const path = [];
            let curr = `${x},${y}`;
            while (curr) {
                const [cx, cy] = curr.split(',').map(Number);
                path.unshift([cx, cy]);
                curr = cameFrom.get(curr);
            }
            return path.slice(1); // Remove start position
        }
        
        const key = `${x},${y}`;
        
        for (let i = 1; i < directions.length; i++) {
            const [dx, dy] = directions[i];
            const nx = x + dx;
            const ny = y + dy;
            
            if (!isValid(nx, ny)) continue;
            
            const nkey = `${nx},${ny}`;
            const tentativeG = g + 1;
            
            if (!gScore.has(nkey) || tentativeG < gScore.get(nkey)) {
                cameFrom.set(nkey, key);
                gScore.set(nkey, tentativeG);
                const newH = Math.abs(targetX - nx) + Math.abs(targetY - ny);
                openSet.push([nx, ny, tentativeG, newH, i]);
            }
        }
    }
    
    return null;
}

// Main simulation loop - coordinated greedy with look-ahead
for (let step = 0; step < maxStepsPerDrone; step++) {
    const plannedMoves = [];
    const plannedCells = new Set();
    
    for (let di = 0; di < drones.length; di++) {
        const pos = positions[di];
        let bestMove = [0, 0];
        let bestScore = -Infinity;
        
        // Evaluate each possible move
        for (const [dx, dy] of directions) {
            const nx = pos.x + dx;
            const ny = pos.y + dy;
            
            if (!isValid(nx, ny)) continue;
            
            let score = 0;
            const key = `${nx},${ny}`;
            
            // Immediate value
            if (!globalVisited.has(key) && !plannedCells.has(key)) {
                score += probability[ny][nx];
            }
            
            // Look-ahead: evaluate potential future value from this position
            let futureValue = 0;
            const lookAhead = Math.min(6, maxStepsPerDrone - step);
            
            for (let dy2 = -lookAhead; dy2 <= lookAhead; dy2++) {
                for (let dx2 = -lookAhead; dx2 <= lookAhead; dx2++) {
                    const fx = nx + dx2;
                    const fy = ny + dy2;
                    if (!isValid(fx, fy)) continue;
                    const fkey = `${fx},${fy}`;

                    const val = probability[fy][fx];
                    if (val > 0) {
                        const dist = Math.abs(dx2) + Math.abs(dy2);
                        futureValue += val / (dist + 1);
                    }
                }
            }
            
            score += futureValue * 0.3;
            
            // Penalty for staying still
            if (dx === 0 && dy === 0) {
                score -= 0.01;
            }
            
            // Bonus for spreading out from other drones' planned positions
            let spreadBonus = 0;
            for (const otherMove of plannedMoves) {
                const dist = Math.abs(nx - otherMove.x) + Math.abs(ny - otherMove.y);
                if (dist > 0) {
                    spreadBonus += 0.01 / dist;
                }
            }
            score += spreadBonus;
            
            if (score > bestScore) {
                bestScore = score;
                bestMove = [dx, dy];
            }
        }
        
        const newX = pos.x + bestMove[0];
        const newY = pos.y + bestMove[1];
        plannedMoves.push({ x: newX, y: newY });
        plannedCells.add(`${newX},${newY}`);
    }
    
    // Execute planned moves
    for (let di = 0; di < drones.length; di++) {
        const move = plannedMoves[di];
        positions[di] = { x: move.x, y: move.y };
        paths[di].push([move.x, move.y]);
        globalVisited.add(`${move.x},${move.y}`);
    }
    
    // Periodically recompute distance maps
    if (step % 10 === 9) {
        distMaps = positions.map(p => bfs(p.x, p.y, null, null));
    }
}

This solution implements:

1. **Coordinated Greedy Pathfinding**: Each drone evaluates moves based on immediate value plus look-ahead potential
2. **Collision Avoidance**: Drones plan sequentially, avoiding cells already claimed by other drones
3. **Spread Bonus**: Encourages drones to spread out and cover more area
4. **Future Value Heuristic**: Weighs potential future cells by their distance
5. **Valid Move Checking**: Ensures all moves stay within bounds and avoid obstacles

The algorithm is O(steps × drones × lookArea) which should scale well for large grids.
