```spirv
; SPIR-V Assembly for Geometry Shader - Triangle Shrink
; Version 1.0, Vulkan environment

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorations for input gl_PerVertex block
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Decorations for output gl_PerVertex block
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Input variable locations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

; Output variable location
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%voidfn = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

; Constants
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%float_0_7 = OpConstant %float 0.7
%float_0_333333 = OpConstant %float 0.333333

; Input gl_PerVertex struct (just Position)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; Output gl_PerVertex struct
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; Input variables
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_3

%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

; Output variables
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

; Main function
%main = OpFunction %void None %voidfn
%entry = OpLabel

; Load gl_Position for all 3 input vertices
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0

%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr

; Extract xyz components for centroid calculation
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

; Compute centroid: C = (p0 + p1 + p2) / 3
%sum_p = OpFAdd %v3float %p0_xyz %p1_xyz
%sum_p_all = OpFAdd %v3float %sum_p %p2_xyz
%centroid = OpVectorTimesScalar %v3float %sum_p_all %float_0_333333

; Compute shrunk positions for each vertex
; new_pos = C + s * (p_i - C) = C + s * delta

; Vertex 0: shrunk position
%delta0 = OpFSub %v3float %p0_xyz %centroid
%scaled_delta0 = OpVectorTimesScalar %v3float %delta0 %float_0_7
%new_pos0_xyz = OpFAdd %v3float %centroid %scaled_delta0
%new_pos0 = OpCompositeConstruct %v4float %new_pos0_xyz %float_0_7

; Vertex 1: shrunk position
%delta1 = OpFSub %v3float %p1_xyz %centroid
%scaled_delta1 = OpVectorTimesScalar %v3float %delta1 %float_0_7
%new_pos1_xyz = OpFAdd %v3float %centroid %scaled_delta1
%new_pos1 = OpCompositeConstruct %v4float %new_pos1_xyz %float_0_7

; Vertex 2: shrunk position
%delta2 = OpFSub %v3float %p2_xyz %centroid
%scaled_delta2 = OpVectorTimesScalar %v3float %delta2 %float_0_7
%new_pos2_xyz = OpFAdd %v3float %centroid %scaled_delta2
%new_pos2 = OpCompositeConstruct %v4float %new_pos2_xyz %float_0_7

; Load input colors
%color0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%color1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%color2_ptr = OpAccessChain %ptr_in_v3 %color %int_2

%color0 = OpLoad %v3float %color0_ptr
%color1 = OpLoad %v3float %color1_ptr
%color2 = OpLoad %v3float %color2_ptr

; Emit vertex 0
%out_pos0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos0_ptr %new_pos0
OpStore %outColor %color0
OpEmitVertex

; Emit vertex 1
%out_pos1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos1_ptr %new_pos1
OpStore %outColor %color1
OpEmitVertex

; Emit vertex 2
%out_pos2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos2_ptr %new_pos2
OpStore %outColor %color2
OpEmitVertex

; End the primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
```
