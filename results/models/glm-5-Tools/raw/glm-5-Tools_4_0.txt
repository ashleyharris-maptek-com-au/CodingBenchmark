const double PI = 3.14159265358979323846;
const double EPS = 1e-9;

struct Vec3 {
    double x, y, z;
    
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
    
    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
    Vec3 operator/(double s) const { return Vec3(x / s, y / s, z / s); }
    
    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const {
        return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x);
    }
    
    double length() const { return sqrt(x * x + y * y + z * z); }
    double lengthSq() const { return x * x + y * y + z * z; }
    
    Vec3 normalized() const {
        double len = length();
        return len > 0 ? *this / len : Vec3();
    }
};

struct Face {
    array<int, 3> verts;
    Vec3 normal;
    double d;
    
    void computePlane(const vector<Vec3>& pts) {
        Vec3 v0 = pts[verts[0]];
        Vec3 v1 = pts[verts[1]];
        Vec3 v2 = pts[verts[2]];
        normal = (v1 - v0).cross(v2 - v0).normalized();
        d = -normal.dot(v0);
    }
    
    bool inside(const Vec3& p) const {
        return normal.dot(p) + d <= EPS;
    }
};

struct Polyhedron {
    vector<Vec3> verts;
    vector<Face> faces;
    Vec3 minB, maxB;
    
    void computeBounds() {
        minB = maxB = verts[0];
        for (const auto& v : verts) {
            minB.x = min(minB.x, v.x);
            minB.y = min(minB.y, v.y);
            minB.z = min(minB.z, v.z);
            maxB.x = max(maxB.x, v.x);
            maxB.y = max(maxB.y, v.y);
            maxB.z = max(maxB.z, v.z);
        }
    }
    
    bool contains(const Vec3& p) const {
        if (p.x < minB.x - EPS || p.x > maxB.x + EPS ||
            p.y < minB.y - EPS || p.y > maxB.y + EPS ||
            p.z < minB.z - EPS || p.z > maxB.z + EPS) {
            return false;
        }
        for (const auto& f : faces) {
            if (!f.inside(p)) return false;
        }
        return true;
    }
    
    bool containsSphere(const Vec3& p, double r) const {
        for (const auto& f : faces) {
            if (f.normal.dot(p) + f.d < -r - EPS) return false;
        }
        return true;
    }
};

class ConvexHullBuilder {
public:
    static vector<Face> build(const vector<Vec3>& pts) {
        int n = (int)pts.size();
        if (n < 4) return {};
        
        vector<int> start = findInitial(pts);
        if (start.empty()) return {};
        
        vector<Face> faces;
        addFace(faces, pts, start[0], start[1], start[2], pts[start[3]]);
        addFace(faces, pts, start[0], start[2], start[3], pts[start[1]]);
        addFace(faces, pts, start[0], start[3], start[1], pts[start[2]]);
        addFace(faces, pts, start[1], start[3], start[2], pts[start[0]]);
        
        set<int> added(start.begin(), start.end());
        for (int i = 0; i < n; i++) {
            if (added.count(i)) continue;
            addPoint(pts, faces, i);
        }
        
        return faces;
    }
    
private:
    static vector<int> findInitial(const vector<Vec3>& pts) {
        int n = (int)pts.size();
        
        int p0 = 0;
        for (int i = 1; i < n; i++) {
            if (pts[i].x < pts[p0].x) p0 = i;
        }
        
        int p1 = -1;
        double maxD = 0;
        for (int i = 0; i < n; i++) {
            if (i == p0) continue;
            double d = (pts[i] - pts[p0]).lengthSq();
            if (d > maxD) { maxD = d; p1 = i; }
        }
        if (p1 < 0) return {};
        
        int p2 = -1;
        maxD = 0;
        Vec3 line = (pts[p1] - pts[p0]).normalized();
        for (int i = 0; i < n; i++) {
            if (i == p0 || i == p1) continue;
            Vec3 toPt = pts[i] - pts[p0];
            Vec3 perp = toPt - line * toPt.dot(line);
            double d = perp.lengthSq();
            if (d > maxD) { maxD = d; p2 = i; }
        }
        if (p2 < 0) return {};
        
        int p3 = -1;
        maxD = 0;
        Vec3 normal = (pts[p1] - pts[p0]).cross(pts[p2] - pts[p0]).normalized();
        double base = normal.dot(pts[p0]);
        for (int i = 0; i < n; i++) {
            if (i == p0 || i == p1 || i == p2) continue;
            double d = abs(normal.dot(pts[i]) - base);
            if (d > maxD) { maxD = d; p3 = i; }
        }
        if (p3 < 0) return {};
        
        return {p0, p1, p2, p3};
    }
    
    static void addFace(vector<Face>& faces, const vector<Vec3>& pts, 
                        int a, int b, int c, const Vec3& interiorPt) {
        Face f;
        f.verts = {a, b, c};
        Vec3 v0 = pts[a], v1 = pts[b], v2 = pts[c];
        Vec3 n = (v1 - v0).cross(v2 - v0);
        if (n.dot(interiorPt - v0) > 0) {
            swap(f.verts[1], f.verts[2]);
        }
        f.computePlane(pts);
        faces.push_back(f);
    }
    
    static void addPoint(const vector<Vec3>& pts, vector<Face>& faces, int newPt) {
        Vec3 p = pts[newPt];
        
        vector<bool> visible(faces.size(), false);
        for (size_t i = 0; i < faces.size(); i++) {
            if (!faces[i].inside(p)) visible[i] = true;
        }
        
        bool anyVis = false;
        for (bool v : visible) if (v) { anyVis = true; break; }
        if (!anyVis) return;
        
        map<pair<int,int>, int> edgeCount;
        for (size_t i = 0; i < faces.size(); i++) {
            if (!visible[i]) continue;
            for (int j = 0; j < 3; j++) {
                int a = faces[i].verts[j];
                int b = faces[i].verts[(j+1)%3];
                if (a > b) swap(a, b);
                edgeCount[{a, b}]++;
            }
        }
        
        vector<pair<int,int>> horizon;
        for (auto& kv : edgeCount) {
            if (kv.second == 1) horizon.push_back(kv.first);
        }
        
        vector<Face> newFaces;
        for (size_t i = 0; i < faces.size(); i++) {
            if (!visible[i]) newFaces.push_back(faces[i]);
        }
        
        Vec3 center(0,0,0);
        int cnt = 0;
        for (auto& f : newFaces) {
            for (int v : f.verts) { center = center + pts[v]; cnt++; }
        }
        if (cnt > 0) center = center / cnt;
        
        for (auto& e : horizon) {
            Face f;
            f.verts = {e.first, e.second, newPt};
            Vec3 v0 = pts[e.first], v1 = pts[e.second], v2 = pts[newPt];
            Vec3 n = (v1 - v0).cross(v2 - v0);
            if (n.dot(center - v0) > 0) swap(f.verts[1], f.verts[2]);
            f.computePlane(pts);
            newFaces.push_back(f);
        }
        
        faces = newFaces;
    }
};

struct Tetrahedron {
    double edge;
    double height;
    double circumradius;
    double inradius;
    Vec3 baseVerts[4];
    
    Tetrahedron(double e) : edge(e) {
        height = e * sqrt(2.0 / 3.0);
        circumradius = e * sqrt(3.0 / 8.0);
        inradius = e / sqrt(24.0);
        
        // Regular tetrahedron vertices centered at origin
        double s = e / (2.0 * sqrt(2.0));
        baseVerts[0] = Vec3(s, s, s);
        baseVerts[1] = Vec3(s, -s, -s);
        baseVerts[2] = Vec3(-s, s, -s);
        baseVerts[3] = Vec3(-s, -s, s);
    }
    
    Vec3 rotateX(const Vec3& v, double a) const {
        double c = cos(a), s = sin(a);
        return Vec3(v.x, v.y*c - v.z*s, v.y*s + v.z*c);
    }
    Vec3 rotateY(const Vec3& v, double a) const {
        double c = cos(a), s = sin(a);
        return Vec3(v.x*c + v.z*s, v.y, -v.x*s + v.z*c);
    }
    Vec3 rotateZ(const Vec3& v, double a) const {
        double c = cos(a), s = sin(a);
        return Vec3(v.x*c - v.y*s, v.x*s + v.y*c, v.z);
    }
    
    void getVertices(const Vec3& center, double rx, double ry, double rz, Vec3 out[4]) const {
        for (int i = 0; i < 4; i++) {
            Vec3 v = baseVerts[i];
            v = rotateX(v, rx);
            v = rotateY(v, ry);
            v = rotateZ(v, rz);
            out[i] = v + center;
        }
    }
};

class TetPacker {
    const Polyhedron& poly;
    double edge;
    Tetrahedron tet;
    
public:
    TetPacker(const Polyhedron& p, double e) : poly(p), edge(e), tet(e) {}
    
    vector<array<double, 6>> solve() {
        vector<array<double, 6>> result;
        
        double spacing = tet.circumradius * 2.0;
        double margin = tet.circumradius + 0.001;
        
        Vec3 size = poly.maxB - poly.minB;
        double step = spacing * 1.01;
        
        int gx = max(1, (int)ceil(size.x / step));
        int gy = max(1, (int)ceil(size.y / step));
        int gz = max(1, (int)ceil(size.z / step));
        
        long long total = (long long)gx * gy * gz;
        if (total > 50000000) {
            step *= cbrt(total / 50000000.0);
            gx = max(1, (int)ceil(size.x / step));
            gy = max(1, (int)ceil(size.y / step));
            gz = max(1, (int)ceil(size.z / step));
        }
        
        double cellSize = spacing * 2;
        map<tuple<int,int,int>, vector<int>> grid;
        
        auto hashKey = [&](const Vec3& p) {
            return make_tuple((int)floor(p.x/cellSize), (int)floor(p.y/cellSize), (int)floor(p.z/cellSize));
        };
        
        vector<Vec3> centers;
        
        vector<array<double,3>> rotPat = {
            {0,0,0}, {PI/6,0,0}, {0,PI/6,0}, {0,0,PI/6},
            {PI/4,0,0}, {0,PI/4,0}, {0,0,PI/4}, {PI/6,PI/6,PI/6},
            {PI/3,0,0}, {0,PI/3,0}, {0,0,PI/3}, {PI/4,PI/4,PI/4}
        };
        
        auto start = chrono::steady_clock::now();
        const int maxMs = 28000;
        
        double offsets[4][3] = {
            {0, 0, 0}, {step/2, step/2, 0}, {step/2, 0, step/2}, {0, step/2, step/2}
        };
        
        for (int pass = 0; pass < 4; pass++) {
            double ox = offsets[pass][0];
            double oy = offsets[pass][1];
            double oz = offsets[pass][2];
            
            for (int iz = 0; iz < gz; iz++) {
                for (int iy = 0; iy < gy; iy++) {
                    for (int ix = 0; ix < gx; ix++) {
                        if ((ix+iy+iz) % 1000 == 0) {
                            auto elapsed = chrono::duration_cast<chrono::milliseconds>(
                                chrono::steady_clock::now() - start).count();
                            if (elapsed > maxMs) goto done;
                        }
                        
                        Vec3 c(
                            poly.minB.x + margin + ox + ix * step,
                            poly.minB.y + margin + oy + iy * step,
                            poly.minB.z + margin + oz + iz * step
                        );
                        
                        if (!poly.containsSphere(c, tet.circumradius)) continue;
                        
                        auto key = hashKey(c);
                        bool overlap = false;
                        for (int dx = -1; dx <= 1 && !overlap; dx++) {
                            for (int dy = -1; dy <= 1 && !overlap; dy++) {
                                for (int dz = -1; dz <= 1 && !overlap; dz++) {
                                    auto k = make_tuple(get<0>(key)+dx, get<1>(key)+dy, get<2>(key)+dz);
                                    auto it = grid.find(k);
                                    if (it != grid.end()) {
                                        for (int idx : it->second) {
                                            if ((c - centers[idx]).length() < spacing * 0.999) {
                                                overlap = true;
                                                break;
                                            }
                                        }
                                    }
                                }
                            }
                        }
                        if (overlap) continue;
                        
                        for (auto& rot : rotPat) {
                            Vec3 verts[4];
                            tet.getVertices(c, rot[0], rot[1], rot[2], verts);
                            
                            bool ok = true;
                            for (int i = 0; i < 4; i++) {
                                if (!poly.contains(verts[i])) { ok = false; break; }
                            }
                            
                            if (ok) {
                                int idx = (int)centers.size();
                                centers.push_back(c);
                                grid[key].push_back(idx);
                                result.push_back({c.x, c.y, c.z, rot[0], rot[1], rot[2]});
                                break;
                            }
                        }
                    }
                }
            }
        }
        
        done:
        return result;
    }
};
