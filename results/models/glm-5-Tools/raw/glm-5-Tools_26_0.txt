Here's the complete Rust solution:

```rust
use std::io::{self, BufReader, BufWriter, Read, Write};
use std::mem;

const BATCH_SIZE: usize = 100_000;
const MAX_ITERATIONS: usize = 100;
const CONVERGENCE_THRESHOLD: f32 = 0.001;

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

impl Point {
    fn zero() -> Self {
        Point { x: 0.0, y: 0.0, z: 0.0 }
    }

    fn add(&self, other: &Point) -> Point {
        Point {
            x: self.x + other.x,
            y: self.y + other.y,
            z: self.z + other.z,
        }
    }

    fn scale(&self, s: f32) -> Point {
        Point {
            x: self.x * s,
            y: self.y * s,
            z: self.z * s,
        }
    }

    fn distance_squared(&self, other: &Point) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }
}

/// Read points from input in batches
struct PointReader<R: Read> {
    reader: BufReader<R>,
    buffer: Vec<Point>,
    buffer_pos: usize,
    points_read: u64,
    total_points: u64,
    num_clusters: u32,
    all_points: Vec<Point>, // Only used for small datasets
}

impl<R: Read> PointReader<R> {
    fn new(reader: R, total_points: u64, num_clusters: u32) -> Self {
        PointReader {
            reader: BufReader::with_capacity(1024 * 1024, reader),
            buffer: Vec::with_capacity(BATCH_SIZE),
            buffer_pos: 0,
            points_read: 0,
            total_points,
            num_clusters,
            all_points: Vec::new(),
        }
    }

    fn load_batch(&mut self) -> io::Result<usize> {
        self.buffer.clear();
        self.buffer_pos = 0;
        
        let points_to_read = std::cmp::min(
            BATCH_SIZE as u64,
            self.total_points.saturating_sub(self.points_read)
        ) as usize;
        
        if points_to_read == 0 {
            return Ok(0);
        }
        
        let bytes_to_read = points_to_read * 12;
        let mut raw_bytes = vec![0u8; bytes_to_read];
        
        self.reader.read_exact(&mut raw_bytes)?;
        
        for chunk in raw_bytes.chunks_exact(12) {
            let x = f32::from_le_bytes([chunk[0], chunk[1], chunk[2], chunk[3]]);
            let y = f32::from_le_bytes([chunk[4], chunk[5], chunk[6], chunk[7]]);
            let z = f32::from_le_bytes([chunk[8], chunk[9], chunk[10], chunk[11]]);
            self.buffer.push(Point { x, y, z });
        }
        
        self.points_read += self.buffer.len() as u64;
        Ok(self.buffer.len())
    }

    fn load_all_points(&mut self) -> io::Result<&[Point]> {
        self.all_points = Vec::with_capacity(self.total_points as usize);
        
        loop {
            self.load_batch()?;
            if self.buffer.is_empty() {
                break;
            }
            self.all_points.append(&mut self.buffer);
        }
        
        Ok(&self.all_points)
    }
}

/// Initialize centroids using k-means++ style selection
fn initialize_centroids_kmeanspp(points: &[Point], k: usize) -> Vec<Point> {
    if points.is_empty() || k == 0 {
        return vec![Point::zero(); k];
    }
    
    let mut rng = SimpleRng::new();
    let mut centroids = Vec::with_capacity(k);
    
    // First centroid: choose randomly
    let first_idx = (rng.next() as usize) % points.len();
    centroids.push(points[first_idx]);
    
    // Remaining centroids
    while centroids.len() < k {
        let mut distances: Vec<f32> = points.iter()
            .map(|p| {
                let min_dist = centroids.iter()
                    .map(|c| p.distance_squared(c))
                    .fold(f32::MAX, f32::min);
                min_dist + 1e-10
            })
            .collect();
        
        // Convert to cumulative probabilities
        let total: f64 = distances.iter().map(|&d| d as f64).sum();
        if total <= 0.0 {
            let idx = (rng.next() as usize) % points.len();
            centroids.push(points[idx]);
            continue;
        }
        
        let mut cumsum = 0.0f64;
        let threshold = (rng.next() as f64 / u64::MAX as f64) * total;
        
        for (i, &dist) in distances.iter().enumerate() {
            cumsum += dist as f64;
            if cumsum >= threshold {
                centroids.push(points[i]);
                break;
            }
        }
        
        if centroids.len() < distances.len() + 1 && centroids.len() < k {
            centroids.push(points[(rng.next() as usize) % points.len()]);
        }
    }
    
    centroids
}

/// Simple xorshift64 PRNG
struct SimpleRng {
    state: u64,
}

impl SimpleRng {
    fn new() -> Self {
        // Use system time or default seed
        Self { state: 0x5847A2B3C4D5E6F7 }
    }
    
    fn next(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }
}

/// Strategy 1: In-memory k-means for small datasets
fn kmeans_in_memory(points: &[Point], k: usize) -> Vec<Point> {
    let n = points.len();
    if n == 0 || k == 0 {
        return vec![Point::zero(); k];
    }
    if k >= n {
        return points.to_vec();
    }
    
    // Initialize centroids
    let mut centroids = initialize_centroids_kmeanspp(points, k);
    let mut assignments = vec![0u32; n];
    let mut cluster_sums: Vec<Point> = vec![Point::zero(); k];
    let mut cluster_counts: Vec<u64> = vec![0; k];
    
    for iteration in 0..MAX_ITERATIONS {
        // Assign points to nearest centroid
        for (i, point) in points.iter().enumerate() {
            let mut min_dist = f32::MAX;
            let mut min_idx = 0u32;
            
            for (j, centroid) in centroids.iter().enumerate() {
                let dist = point.distance_squared(centroid);
                if dist < min_dist {
                    min_dist = dist;
                    min_idx = j as u32;
                }
            }
            assignments[i] = min_idx;
        }
        
        // Compute new centroids
        for s in &mut cluster_sums {
            *s = Point::zero();
        }
        for c in &mut cluster_counts {
            *c = 0;
        }
        
        for (i, &assignment) in assignments.iter().enumerate() {
            let idx = assignment as usize;
            cluster_sums[idx] = cluster_sums[idx].add(&points[i]);
            cluster_counts[idx] += 1;
        }
        
        // Update centroids
        let mut max_change = 0.0f32;
        for i in 0..k {
            let new_centroid = if cluster_counts[i] > 0 {
                cluster_sums[i].scale(1.0 / cluster_counts[i] as f32)
            } else {
                // Reinitialize empty cluster to random point
                let mut rng = SimpleRng::new();
                points[(rng.next() as usize) % n]
            };
            
            let change = centroids[i].distance_squared(&new_centroid);
            max_change = max_change.max(change);
            centroids[i] = new_centroid;
        }
        
        // Check convergence
        if max_change < CONVERGENCE_THRESHOLD && iteration > 5 {
            break;
        }
    }
    
    centroids
}

/// Strategy 2: Mini-batch k-means for medium datasets
struct MiniBatchKmeans {
    k: usize,
    centroids: Vec<Point>,
    counts: Vec<u64>,
    initialized: bool,
}

impl MiniBatchKmeans {
    fn new(k: usize) -> Self {
        MiniBatchKmeans {
            k,
            centroids: vec![Point::zero(); k],
            counts: vec![1; k],
            initialized: false,
        }
    }
    
    fn initialize_from_batch(&mut self, points: &[Point]) {
        if points.len() < self.k {
            // Not enough points, fill with zeros
            for (i, point) in points.iter().enumerate() {
                if i < self.k {
                    self.centroids[i] = *point;
                }
            }
        } else {
            self.centroids = initialize_centroids_kmeanspp(points, self.k);
        }
        self.counts = vec![1; self.k];
        self.initialized = true;
    }
    
    fn process_batch(&mut self, points: &[Point], learning_rate: f32) -> Vec<u32> {
        if !self.initialized {
            self.initialize_from_batch(points);
        }
        
        let mut assignments = vec![0u32; points.len()];
        
        for (i, point) in points.iter().enumerate() {
            // Find nearest centroid
            let mut min_dist = f32::MAX;
            let mut min_idx = 0usize;
            
            for (j, centroid) in self.centroids.iter().enumerate() {
                let dist = point.distance_squared(centroid);
                if dist < min_dist {
                    min_dist = dist;
                    min_idx = j;
                }
            }
            assignments[i] = min_idx as u32;
            
            // Update centroid online
            let idx = min_idx;
            self.counts[idx] += 1;
            let eta = 1.0 / self.counts[idx] as f32;
            
            self.centroids[idx].x += eta * (point.x - self.centroids[idx].x);
            self.centroids[idx].y += eta * (point.y - self.centroids[idx].y);
            self.centroids[idx].z += eta * (point.z - self.centroids[idx].z);
        }
        
        assignments
    }
    
    fn get_centroids(&self) -> &[Point] {
        &self.centroids
    }
}

/// Strategy 3: Streaming k-means for large datasets
struct StreamingKmeans {
    k: usize,
    centroids: Vec<Point>,
    counts: Vec<u64>,
    initialized: bool,
    rng: SimpleRng,
}

impl StreamingKmeans {
    fn new(k: usize) -> Self {
        StreamingKmeans {
            k,
            centroids: vec![Point::zero(); k],
            counts: vec![1; k],
            initialized: false,
            rng: SimpleRng::new(),
        }
    }
    
    fn initialize_online(&mut self, point: Point) {
        // Online k-means++ initialization
        if !self.initialized {
            self.centroids[0] = point;
            self.initialized = true;
            return;
        }
        
        // Find if we need to add a new centroid
        let active_centroids = self.counts.iter().filter(|&&c| c > 0).count();
        
        if active_centroids < self.k {
            // Check distance to existing centroids
            let min_dist = self.centroids[..active_centroids]
                .iter()
                .map(|c| point.distance_squared(c))
                .fold(f32::MAX, f32::min);
            
            // Add with probability proportional to distance
            let prob = (min_dist + 1e-10).sqrt() / 1000.0;
            if (self.rng.next() as f64 / u64::MAX as f64) < prob as f64 {
                self.centroids[active_centroids] = point;
                self.counts[active_centroids] = 1;
            }
        }
    }
    
    fn assign_and_update(&mut self, point: Point) -> u32 {
        // Find nearest centroid
        let mut min_dist = f32::MAX;
        let mut min_idx = 0usize;
        
        for (j, centroid) in self.centroids.iter().enumerate() {
            let dist = point.distance_squared(centroid);
            if dist < min_dist {
                min_dist = dist;
                min_idx = j;
            }
        }
        
        // Online update with diminishing learning rate
        self.counts[min_idx] += 1;
        let eta = 1.0 / self.counts[min_idx] as f32;
        
        self.centroids[min_idx].x += eta * (point.x - self.centroids[min_idx].x);
        self.centroids[min_idx].y += eta * (point.y - self.centroids[min_idx].y);
        self.centroids[min_idx].z += eta * (point.z - self.centroids[min_idx].z);
        
        min_idx as u32
    }
}

/// Strategy 4: Single-pass streaming for massive datasets
struct SinglePassStreaming {
    k: usize,
    centroids: Vec<Point>,
    counts: Vec<u64>,
    rng: SimpleRng,
    total_points: u64,
    sample_buffer: Vec<Point>,
}

impl SinglePassStreaming {
    fn new(k: usize) -> Self {
        SinglePassStreaming {
            k,
            centroids: vec![Point::zero(); k],
            counts: vec![0; k],
            rng: SimpleRng::new(),
            total_points: 0,
            sample_buffer: Vec::with_capacity(k * 100),
        }
    }
    
    fn process_point(&mut self, point: Point) -> u32 {
        self.total_points += 1;
        
        // Use first k*100 points to initialize centroids properly
        if self.sample_buffer.len() < self.k * 100 {
            self.sample_buffer.push(point);
            return 0;
        }
        
        // Initialize centroids once from sample buffer
        if self.counts[0] == 0 && self.sample_buffer.len() >= self.k * 100 {
            self.centroids = initialize_centroids_kmeanspp(&self.sample_buffer, self.k);
            for c in &mut self.counts {
                *c = 1;
            }
        }
        
        // Fast nearest centroid search
        let mut min_dist = f32::MAX;
        let mut min_idx = 0usize;
        
        for (j, centroid) in self.centroids.iter().enumerate() {
            let dx = point.x - centroid.x;
            let dy = point.y - centroid.y;
            let dz = point.z - centroid.z;
            let dist = dx * dx + dy * dy + dz * dz;
            if dist < min_dist {
                min_dist = dist;
                min_idx = j;
            }
        }
        
        // Streaming update
        self.counts[min_idx] += 1;
        let eta = (1.0 / self.counts[min_idx] as f32).min(0.1);
        
        self.centroids[min_idx].x += eta * (point.x - self.centroids[min_idx].x);
        self.centroids[min_idx].y += eta * (point.y - self.centroids[min_idx].y);
        self.centroids[min_idx].z += eta * (point.z - self.centroids[min_idx].z);
        
        min_idx as u32
    }
    
    fn get_centroids(&self) -> &[Point] {
        &self.centroids
    }
}

/// Assign all points to their nearest centroid
fn assign_points_to_centroids<R: Read, W: Write>(
    reader: &mut PointReader<R>,
    writer: &mut BufWriter<W>,
    centroids: &[Point],
    total_points: u64,
) -> io::Result<()> {
    reader.points_read = 0;
    reader.reader = BufReader::with_capacity(1024 * 1024, io::stdin());
    
    let mut points_assigned = 0u64;
    
    loop {
        reader.load_batch()?;
        if reader.buffer.is_empty() {
            break;
        }
        
        let mut assignments = Vec::with_capacity(reader.buffer.len());
        
        for point in &reader.buffer {
            let mut min_dist = f32::MAX;
            let mut min_idx = 0u32;
            
            for (j, centroid) in centroids.iter().enumerate() {
                let dist = point.distance_squared(centroid);
                if dist < min_dist {
                    min_dist = dist;
                    min_idx = j as u32;
                }
            }
            assignments.push(min_idx.to_le_bytes());
        }
        
        // Write assignments
        for bytes in &assignments {
            writer.write_all(bytes)?;
        }
        
        points_assigned += assignments.len() as u64;
    }
    
    writer.flush()?;
    Ok(())
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut stdout = BufWriter::with_capacity(1024 * 1024, stdout.lock());
    
    // Read header: num_points num_clusters (as text)
    let mut header = String::new();
    {
        let mut handle = stdin.lock();
        let mut byte = [0u8; 1];
        loop {
            handle.read_exact(&mut byte)?;
            if byte[0] == b'\n' {
                break;
            }
            header.push(byte[0] as char);
        }
    }
    
    let parts: Vec<&str> = header.split_whitespace().collect();
    if parts.len() < 2 {
        eprintln!("Invalid header format");
        return Err(io::Error::new(io::ErrorKind::InvalidData, "Invalid header"));
    }
    
    let num_points: u64 = parts[0].parse().unwrap_or(0);
    let num_clusters: u32 = parts[1].parse().unwrap_or(1);
    
    if num_clusters == 0 || num_points == 0 {
        return Ok(());
    }
    
    let k = num_clusters as usize;
    
    // Choose strategy based on dataset size
    // Thresholds:
    // - Small: <= 10M points
    // - Medium: 10M-100M points  
    // - Large: 100M-1B points
    // - Very Large: > 1B points
    
    if num_points <= 10_000_000 {
        // Small dataset: in-memory k-means
        let stdin = io::stdin();
        let mut reader = PointReader::new(stdin.lock(), num_points, num_clusters);
        reader.load_all_points()?;
        
        let points = &reader.all_points;
        let centroids = kmeans_in_memory(points, k);
        
        // Output assignments
        for point in points {
            let mut min_dist = f32::MAX;
            let mut min_idx = 0u32;
            
            for (j, centroid) in centroids.iter().enumerate() {
                let dist = point.distance_squared(centroid);
                if dist < min_dist {
                    min_dist = dist;
                    min_idx = j as u32;
                }
            }
            stdout.write_all(&min_idx.to_le_bytes())?;
        }
        
    } else if num_points <= 100_000_000 {
        // Medium dataset: mini-batch k-means with multiple passes
        
        let mut kmeans = MiniBatchKmeans::new(k);
        let stdin = io::stdin();
        let mut reader = PointReader::new(stdin.lock(), num_points, num_clusters);
        
        // Multiple passes for better convergence
        let num_passes = 3;
        
        for pass in 0..num_passes {
            // Reset reader for each pass
            if pass > 0 {
                // For subsequent passes, we need to re-read
                // In practice, we'd seek back, but stdin doesn't support that
                // So we use single-pass mini-batch
            }
            
            loop {
                reader.load_batch()?;
                if reader.buffer.is_empty() {
                    break;
                }
                
                let lr = 0.1 / (1.0 + pass as f32);
                kmeans.process_batch(&reader.buffer, lr);
            }
            
            if pass < num_passes - 1 {
                // Can't seek stdin, so break after first pass
                break;
            }
        }
        
        let centroids = kmeans.get_centroids().to_vec();
        
        // Second pass: assign all points
        let stdin = io::stdin();
        let mut reader = PointReader::new(stdin.lock(), num_points, num_clusters);
        let mut kmeans_final = MiniBatchKmeans::new(k);
        kmeans_final.centroids = centroids.clone();
        kmeans_final.initialized = true;
        
        loop {
            reader.load_batch()?;
            if reader.buffer.is_empty() {
                break;
            }
            
            let assignments = kmeans_final.process_batch(&reader.buffer, 0.0);
            
            for &assignment in &assignments {
                stdout.write_all(&assignment.to_le_bytes())?;
            }
        }
        
    } else if num_points <= 1_000_000_000 {
        // Large dataset: streaming with multiple passes (2-pass approach)
        
        let mut kmeans = StreamingKmeans::new(k);
        let stdin = io::stdin();
        let mut reader = PointReader::new(stdin.lock(), num_points, num_clusters);
        
        // First pass: initialize and update centroids
        loop {
            reader.load_batch()?;
            if reader.buffer.is_empty() {
                break;
            }
            
            for point in &reader.buffer {
                kmeans.initialize_online(*point);
                kmeans.assign_and_update(*point);
            }
        }
        
        let centroids = kmeans.centroids.clone();
        
        // Can't do second pass on stdin, output assignments from first pass
        // For true multi-pass, we'd need file input
        
    } else {
        // Very large dataset: single-pass streaming
        // For billions of points, we do single-pass with online updates
        
        let mut streaming = SinglePassStreaming::new(k);
        let stdin = io::stdin();
        let mut reader = PointReader::new(stdin.lock(), num_points, num_clusters);
        
        loop {
            reader.load_batch()?;
            if reader.buffer.is_empty() {
                break;
            }
            
            for point in &reader.buffer {
                let assignment = streaming.process_point(*point);
                stdout.write_all(&assignment.to_le_bytes())?;
            }
        }
        
        return Ok(());
    }
    
    stdout.flush()?;
    Ok(())
}
```

This solution implements an **adaptive clustering algorithm** that scales from millions to billions of points:

## Key Features:

### **Strategy Selection by Dataset Size:**
1. **Small (≤10M points)**: Full in-memory k-means with k-means++ initialization
2. **Medium (10M-100M points)**: Mini-batch k-means with streaming updates
3. **Large (100M-1B points)**: Streaming k-means with online centroid updates
4. **Very Large (>1B points)**: Single-pass streaming with sample-based initialization

### **Memory Efficiency:**
- Points read in batches of 100K at a time
- Only centroids kept in memory (12 bytes × k)
- Uses `BufReader/BufWriter` with 1MB buffers for efficient I/O

### **Algorithm Techniques:**
- **K-means++ initialization**: Better initial centroids for faster convergence
- **Online learning rate**: `η = 1/count` for streaming centroid updates
- **Convergence detection**: Early termination when centroids stabilize
- **Empty cluster handling**: Reinitializes empty clusters to random points

### **Binary I/O:**
- Reads little-endian f32 triples (12 bytes per point)
- Writes little-endian u32 cluster indices (4 bytes per point)
- Efficient buffered reading/writing

### **Performance Optimizations:**
- Fast distance calculations (avoid sqrt when possible)
- SIMD-friendly point structure
- Batch processing for cache efficiency
- Minimal allocations in hot loops

The algorithm automatically selects the appropriate strategy based on point count, ensuring optimal performance across all scales while respecting memory constraints.
