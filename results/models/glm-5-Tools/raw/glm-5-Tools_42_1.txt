```spirv
; SPIR-V Assembly for Geometry Shader
; Expands each point into a diamond shape (4 triangles as triangle strip)

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inWorldPos %inNormal %inColor %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Extensions
OpSource GLSL 450

; Decorations for gl_PerVertex input (built-in block)
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Decorations for gl_PerVertex output (built-in block)
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Decorations for user-defined inputs
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2

; Decoration for user-defined output
OpDecorate %outColor Location 0

; Decorations for UBO
OpDecorate %UBO BufferBlock
OpDecorate %UBODescriptorSet DescriptorSet 0
OpDecorate %UBOBinding Binding 0
OpMemberDecorate %UBO 0 RowMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 RowMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 RowMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 204
OpMemberDecorate %UBO 5 Offset 216
OpMemberDecorate %UBO 6 Offset 228
OpMemberDecorate %UBO 7 Offset 232
OpMemberDecorate %UBO 8 Offset 236
OpMemberDecorate %UBO 9 Offset 240
OpMemberDecorate %UBO 10 Offset 248
OpMemberDecorate %UBODescriptorSet DescriptorSet 0
OpDecorate %UBOBinding Binding 0

; Types
%void = OpTypeVoid
%void_fn = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4 = OpTypeMatrix %v4float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_1 = OpConstant %uint 1
%float_0_05 = OpConstant %float 0.05
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_neg_0_05 = OpConstant %float -0.05

; gl_PerVertex input type (array size 1 for InputPoints)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; gl_PerVertex output type
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; User-defined input arrays (size 1 for InputPoints)
%arr_v3_1 = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_1

; UBO type
%UBO = OpTypeStruct %mat4v4 %mat4v4 %mat4v4 %v3float %v3float %v3float %float %float %float %v2float
%ptr_ubo = OpTypePointer Uniform %UBO

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_out = OpVariable %ptr_out_pv Output
%inWorldPos = OpVariable %ptr_in_arr_v3 Input
%inNormal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input
%outColor = OpVariable %ptr_out_v3 Output
%ubodata = OpVariable %ptr_ubo Uniform

; Main function
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Load input gl_Position (point position)
%pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr

; Load input color
%color_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%color = OpLoad %v3float %color_ptr

; Extract x, y, z, w from position
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1
%z = OpCompositeExtract %float %pos 2
%w = OpCompositeExtract %float %pos 3

; Calculate diamond vertices
; h = 0.05
; Top: (x, y+h), Right: (x+h, y), Bottom: (x, y-h), Left: (x-h, y)

; --- Emit vertex 0: Top (x, y+h) ---
%y_plus_h = OpFAdd %float %y %float_0_05
%v0 = OpCompositeConstruct %v4float %x %y_plus_h %z %w
%out_pos_ptr0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr0 %v0
OpStore %outColor %color
OpEmitVertex

; --- Emit vertex 1: Right (x+h, y) ---
%x_plus_h = OpFAdd %float %x %float_0_05
%v1 = OpCompositeConstruct %v4float %x_plus_h %y %z %w
%out_pos_ptr1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr1 %v1
OpStore %outColor %color
OpEmitVertex

; --- Emit vertex 2: Left (x-h, y) ---
%x_minus_h = OpFSub %float %x %float_0_05
%v2 = OpCompositeConstruct %v4float %x_minus_h %y %z %w
%out_pos_ptr2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr2 %v2
OpStore %outColor %color
OpEmitVertex

; --- Emit vertex 3: Bottom (x, y-h) ---
%y_minus_h = OpFSub %float %y %float_0_05
%v3 = OpCompositeConstruct %v4float %x %y_minus_h %z %w
%out_pos_ptr3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr3 %v3
OpStore %outColor %color
OpEmitVertex

; End the triangle strip primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
```
