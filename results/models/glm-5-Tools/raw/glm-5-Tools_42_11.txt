```spirv
; SPIR-V Assembly: Dashed Thick Curve Geometry Shader
; InputLinesAdjacency -> OutputTriangleStrip, 8 output vertices

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; UBO decorations
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 204
OpMemberDecorate %UBO 5 Offset 216
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; Types
%void = OpTypeVoid
%void_fn = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%uint = OpTypeInt 32 0
%uint_4 = OpConstant %uint 4

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in

%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out

%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_in_v4 = OpTypePointer Input %v4float

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_02 = OpConstant %float 0.02
%float_0_5 = OpConstant %float 0.5
%float_2 = OpConstant %float 2.0
%float_0_333 = OpConstant %float 0.333333
%float_0_666 = OpConstant %float 0.666666
%float_m0_333 = OpConstant %float -0.333333
%float_m0_666 = OpConstant %float -0.666666

; UBO type
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output
%ubo_var = OpVariable %ptr_ubo Uniform

; Main function
%main = OpFunction %void None %void_fn
%entry = OpLabel

  ; Load gl_Position for all 4 vertices
  %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
  %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
  %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
  %p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
  
  %p0 = OpLoad %v4float %p0_ptr
  %p1 = OpLoad %v4float %p1_ptr
  %p2 = OpLoad %v4float %p2_ptr
  %p3 = OpLoad %v4float %p3_ptr
  
  ; Load colors for vertices 1 and 2
  %c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
  %c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
  %c1 = OpLoad %v3float %c1_ptr
  %c2 = OpLoad %v3float %c2_ptr
  
  ; Compute 2D positions (xy plane)
  %p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
  %p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
  %p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
  %p3_xy = OpVectorShuffle %v2float %p3 %p3 0 1
  
  ; Compute tangents
  %t01 = OpFSub %v2float %p1_xy %p0_xy
  %t12 = OpFSub %v2float %p2_xy %p1_xy
  %t23 = OpFSub %v2float %p3_xy %p2_xy
  
  ; Average tangents at endpoints
  %t1_avg = OpFAdd %v2float %t01 %t12
  %t2_avg = OpFAdd %v2float %t12 %t23
  
  ; Perpendiculars: (-y, x)
  %p1_perp = OpCompositeConstruct %v2float (OpCompositeExtract %float %t1_avg 1) (OpFNegate %float (OpCompositeExtract %float %t1_avg 0))
  %p2_perp = OpCompositeConstruct %v2float (OpCompositeExtract %float %t2_avg 1) (OpFNegate %float (OpCompositeExtract %float %t2_avg 0))
  
  ; Normalize perpendiculars
  %len1_sq = OpDot %float %p1_perp %p1_perp
  %len2_sq = OpDot %float %p2_perp %p2_perp
  %len1 = OpExtInst %float %1 Sqrt %len1_sq
  %len2 = OpExtInst %float %1 Sqrt %len2_sq
  %eps = OpConstant %float 0.0001
  %len1_safe = OpExtInst %float %1 FMax %len1 %eps
  %len2_safe = OpExtInst %float %1 FMax %len2 %eps
  %p1_n = OpVectorTimesScalar %v2float %p1_perp %len1_safe
  %p2_n = OpVectorTimesScalar %v2float %p2_perp %len2_safe
  %p1_norm = OpFDiv %v2float %p1_n (OpCompositeConstruct %v2float %len1_safe %len1_safe)
  %p2_norm = OpFDiv %v2float %p2_n (OpCompositeConstruct %v2float %len2_safe %len2_safe)
  
  ; Offset vectors (thickness/2)
  %half_thick = OpFMul %float %float_0_02 %float_0_5
  %off1 = OpVectorTimesScalar %v2float %p1_norm %half_thick
  %off2 = OpVectorTimesScalar %v2float %p2_norm %half_thick
  
  ; Segment direction and length
  %seg_dir = OpFSub %v2float %p2_xy %p1_xy
  
  ; ===== FIRST QUAD (first third) =====
  ; Parametric positions: t = 0 to t = 0.333
  
  ; t=0: pos = p1_xy
  %t0_off = OpVectorTimesScalar %v2float %off1 %float_1
  %t0_neg_off = OpVectorTimesScalar %v2float %off1 %float_m1
  
  ; t=0.333: interpolate offset and position
  %seg_part1 = OpVectorTimesScalar %v2float %seg_dir %float_0_333
  %t0_333_pos = OpFAdd %v2float %p1_xy %seg_part1
  %off_interp1 = OpFMul %float %float_0_333 %float_1
  %off1_scaled = OpVectorTimesScalar %v2float %off2 %off_interp1
  %off2_scaled = OpVectorTimesScalar %v2float %off1 (OpFSub %float %float_1 %off_interp1)
  %off_at_0_333 = OpFAdd %v2float %off2_scaled %off1_scaled
  
  ; Emit quad 1: vertices at t=0 and t=0.333
  ; v0: t=0, +offset
  %v0_2d = OpFAdd %v2float %p1_xy %off1
  %v0_z = OpCompositeExtract %float %p1 2
  %v0_w = OpCompositeExtract %float %p1 3
  %v0_4d = OpCompositeConstruct %v4float (OpCompositeExtract %float %v0_2d 0) (OpCompositeExtract %float %v0_2d 1) %v0_z %v0_w
  %outPos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
  OpStore %outPos_ptr %v0_4d
  OpStore %outColor %c1
  OpEmitVertex
  
  ; v1: t=0, -offset
  %v1_2d = OpFSub %v2float %p1_xy %off1
  %v1_4d = OpCompositeConstruct %v4float (OpCompositeExtract %float %v1_2d 0) (OpCompositeExtract %float %v1_2d 1) %v0_z %v0_w
  OpStore %outPos_ptr %v1_4d
  OpStore %outColor %c1
  OpEmitVertex
  
  ; v2: t=0.333, +offset
  %v2_2d = OpFAdd %v2float %t0_333_pos %off_at_0_333
  %v2_z = OpCompositeExtract %float %p2 2
  %v2_w = OpCompositeExtract %float %p2 3
  %v2_4d = OpCompositeConstruct %v4float (OpCompositeExtract %float %v2_2d 0) (OpCompositeExtract %float %v2_2d 1) %v2_z %v2_w
  OpStore %outPos_ptr %v2_4d
  ; Interpolate color
  %c_interp1 = OpFMul %float %float_0_333 %float_1
  %c2_part1 = OpVectorTimesScalar %v3float %c2 %c_interp1
  %c1_part1 = OpVectorTimesScalar %v3float %c1 (OpFSub %float %float_1 %c_interp1)
  %c_at_0_333 = OpFAdd %v3float %c1_part1 %c2_part1
  OpStore %outColor %c_at_0_333
  OpEmitVertex
  
  ; v3: t=0.333, -offset
  %v3_2d = OpFSub %v2float %t0_333_pos %off_at_0_333
  %v3_4d = OpCompositeConstruct %v4float (OpCompositeExtract %float %v3_2d 0) (OpCompositeExtract %float %v3_2d 1) %v2_z %v2_w
  OpStore %outPos_ptr %v3_4d
  OpStore %outColor %c_at_0_333
  OpEmitVertex
  
  OpEndPrimitive
  
  ; ===== SECOND QUAD (third third) =====
  ; Parametric positions: t = 0.666 to t = 1.0
  
  ; t=0.666: interpolate offset and position
  %seg_part2 = OpVectorTimesScalar %v2float %seg_dir %float_0_666
  %t0_666_pos = OpFAdd %v2float %p1_xy %seg_part2
  %off_interp2 = OpFMul %float %float_0_666 %float_1
  %off1_scaled2 = OpVectorTimesScalar %v2float %off2 %off_interp2
  %off2_scaled2 = OpVectorTimesScalar %v2float %off1 (OpFSub %float %float_1 %off_interp2)
  %off_at_0_666 = OpFAdd %v2float %off1_scaled2 %off2_scaled2
  
  ; t=1.0: at p2 with offset2
  %t1_pos = OpLoad %v4float %p2_ptr
  
  ; Emit quad 2: vertices at t=0.666 and t=1.0
  ; v4: t=0.666, +offset
  %v4_2d = OpFAdd %v2float %t0_666_pos %off_at_0_666
  %v4_z = OpCompositeExtract %float %p2 2
  %v4_w = OpCompositeExtract %float %p2 3
  %v4_4d = OpCompositeConstruct %v4float (OpCompositeExtract %float %v4_2d 0) (OpCompositeExtract %float %v4_2d 1) %v4_z %v4_w
  OpStore %outPos_ptr %v4_4d
  %c_interp2 = OpFMul %float %float_0_666 %float_1
  %c2_part2 = OpVectorTimesScalar %v3float %c2 %c_interp2
  %c1_part2 = OpVectorTimesScalar %v3float %c1 (OpFSub %float %float_1 %c_interp2)
  %c_at_0_666 = OpFAdd %v3float %c1_part2 %c2_part2
  OpStore %outColor %c_at_0_666
  OpEmitVertex
  
  ; v5: t=0.666, -offset
  %v5_2d = OpFSub %v2float %t0_666_pos %off_at_0_666
  %v5_4d = OpCompositeConstruct %v4float (OpCompositeExtract %float %v5_2d 0) (OpCompositeExtract %float %v5_2d 1) %v4_z %v4_w
  OpStore %outPos_ptr %v5_4d
  OpStore %outColor %c_at_0_666
  OpEmitVertex
  
  ; v6: t=1.0, +offset
  %v6_2d = OpFAdd %v2float %p2_xy %off2
  %v6_4d = OpCompositeConstruct %v4float (OpCompositeExtract %float %v6_2d 0) (OpCompositeExtract %float %v6_2d 1) %v4_z %v4_w
  OpStore %outPos_ptr %v6_4d
  OpStore %outColor %c2
  OpEmitVertex
  
  ; v7: t=1.0, -offset
  %v7_2d = OpFSub %v2float %p2_xy %off2
  %v7_4d = OpCompositeConstruct %v4float (OpCompositeExtract %float %v7_2d 0) (OpCompositeExtract %float %v7_2d 1) %v4_z %v4_w
  OpStore %outPos_ptr %v7_4d
  OpStore %outColor %c2
  OpEmitVertex
  
  OpEndPrimitive

  OpReturn
OpFunctionEnd
```
