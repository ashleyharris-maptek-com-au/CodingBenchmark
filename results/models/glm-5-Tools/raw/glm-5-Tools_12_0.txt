    // All 24 unique rotations of a cube (symmetry group)
    private static readonly Quaternion[] AllRotations = GenerateAllRotations();
    
    private static Quaternion[] GenerateAllRotations()
    {
        var rotations = new HashSet<Quaternion>(new QuaternionComparer());
        double angle90 = Math.PI / 2;
        
        // Generate all 24 rotations in the octahedral group
        for (int i = 0; i < 4; i++)
        {
            for (int j = 0; j < 4; j++)
            {
                for (int k = 0; k < 2; k++)
                {
                    var rz = Quaternion.FromAxisAngle(Vector3.UnitZ, i * angle90);
                    var rx = Quaternion.FromAxisAngle(Vector3.UnitX, j * angle90);
                    var combined = rz * rx;
                    
                    if (k == 1)
                    {
                        var ry = Quaternion.FromAxisAngle(Vector3.UnitY, 2 * angle90);
                        combined = combined * ry;
                    }
                    
                    combined = combined.Normalized();
                    rotations.Add(combined);
                }
            }
        }
        
        // Also add standard axis rotations
        for (int xi = 0; xi < 4; xi++)
        {
            for (int yi = 0; yi < 4; yi++)
            {
                for (int zi = 0; zi < 4; zi++)
                {
                    var rx = Quaternion.FromAxisAngle(Vector3.UnitX, xi * angle90);
                    var ry = Quaternion.FromAxisAngle(Vector3.UnitY, yi * angle90);
                    var rz = Quaternion.FromAxisAngle(Vector3.UnitZ, zi * angle90);
                    var combined = rx * ry * rz;
                    rotations.Add(combined.Normalized());
                }
            }
        }
        
        return rotations.ToArray();
    }
    
    public void Solve()
    {
        ReadInput();
        
        // Early exit if polyhedron is too large
        if (polyAABB.SizeX > containerW || polyAABB.SizeY > containerH || polyAABB.SizeZ > containerD)
        {
            bool found = false;
            foreach (var rot in AllRotations)
            {
                var rotatedAABB = RotateAABB(polyAABB, rot);
                if (rotatedAABB.SizeX <= containerW && rotatedAABB.SizeY <= containerH && rotatedAABB.SizeZ <= containerD)
                {
                    found = true;
                    break;
                }
            }
            if (!found)
            {
                Console.WriteLine("0");
                return;
            }
        }
        
        var placements = new List<Placement>();
        var sw = Stopwatch.StartNew();
        TimeSpan maxTime = TimeSpan.FromMinutes(4.5);
        
        int scale = DetermineScale();
        
        switch (scale)
        {
            case 0: SolveTrivial(placements, sw, maxTime); break;
            case 1: SolveMedium(placements, sw, maxTime); break;
            case 2: SolveLarge(placements, sw, maxTime); break;
            case 3: SolveExtreme(placements, sw, maxTime); break;
        }
        
        OutputResults(placements);
    }
    
    private int DetermineScale()
    {
        int maxDim = Math.Max(containerW, Math.Max(containerH, containerD));
        
        if (maxDim >= 200 || containerVolume > 5000000) return 3;
        if (maxDim >= 50 || containerVolume > 500000) return 2;
        if (maxDim >= 10 || containerVolume > 5000) return 1;
        return 0;
    }
    
    private void SolveTrivial(List<Placement> placements, Stopwatch sw, TimeSpan maxTime)
    {
        var occupied = new List<AABB>();
        var (bestRot, bestSize) = FindBestRotation();
        
        int stepX = Math.Max(1, (int)Math.Ceiling(bestSize.X));
        int stepY = Math.Max(1, (int)Math.Ceiling(bestSize.Y));
        int stepZ = Math.Max(1, (int)Math.Ceiling(bestSize.Z));
        
        for (int z = 0; z + stepZ <= containerD; z += stepZ)
        {
            for (int y = 0; y + stepY <= containerH; y += stepY)
            {
                for (int x = 0; x + stepX <= containerW; x += stepX)
                {
                    if (sw.Elapsed > maxTime) return;
                    
                    var candidateAABB = new AABB(x, y, z, x + bestSize.X, y + bestSize.Y, z + bestSize.Z);
                    
                    bool collides = false;
                    foreach (var occ in occupied)
                    {
                        if (AABBsOverlap(candidateAABB, occ))
                        {
                            collides = true;
                            break;
                        }
                    }
                    
                    if (!collides)
                    {
                        placements.Add(new Placement(x, y, z, bestRot));
                        occupied.Add(candidateAABB);
                    }
                }
            }
        }
        
        FillGapsPrecise(placements, occupied, sw, maxTime);
    }
    
    private void SolveMedium(List<Placement> placements, Stopwatch sw, TimeSpan maxTime)
    {
        var (bestRot, bestSize) = FindBestRotation();
        
        int stepX = Math.Max(1, (int)Math.Ceiling(bestSize.X));
        int stepY = Math.Max(1, (int)Math.Ceiling(bestSize.Y));
        int stepZ = Math.Max(1, (int)Math.Ceiling(bestSize.Z));
        
        int count = 0;
        int maxCount = 50000;
        
        for (int x = 0; x + stepX <= containerW && count < maxCount; x += stepX)
        {
            for (int y = 0; y + stepY <= containerH && count < maxCount; y += stepY)
            {
                for (int z = 0; z + stepZ <= containerD && count < maxCount; z += stepZ)
                {
                    if (sw.Elapsed > maxTime) return;
                    placements.Add(new Placement(x, y, z, bestRot));
                    count++;
                }
            }
        }
    }
    
    private void SolveLarge(List<Placement> placements, Stopwatch sw, TimeSpan maxTime)
    {

        int stepX = Math.Max(1, (int)Math.Ceiling(bestSize.X));
        int stepY = Math.Max(1, (int)Math.Ceiling(bestSize.Y));
        int stepZ = Math.Max(1, (int)Math.Ceiling(bestSize.Z));
        
        int countX = containerW / stepX;
        int countY = containerH / stepY;
        int countZ = containerD / stepZ;
        
        long total = (long)countX * countY * countZ;
        
        if (total > 500000)
        {
            double sampleRate = 500000.0 / total;
            var rand = new Random(42);
            
            for (int x = 0; x < countX; x++)
            {
                for (int y = 0; y < countY; y++)
                {
                    for (int z = 0; z < countZ; z++)
                    {
                        if (rand.NextDouble() < sampleRate)
                        {
                            placements.Add(new Placement(x * stepX, y * stepY, z * stepZ, bestRot));
                        }
                    }
                }
            }
        }
        else
        {
            for (int x = 0; x < countX; x++)
            {
                for (int y = 0; y < countY; y++)
                {
                    for (int z = 0; z < countZ; z++)
                    {
                        if (sw.Elapsed > maxTime) return;
                        placements.Add(new Placement(x * stepX, y * stepY, z * stepZ, bestRot));
                    }
                }
            }
        }
    }
    
    private void SolveExtreme(List<Placement> placements, Stopwatch sw, TimeSpan maxTime)
    {
        var (bestRot, bestSize) = FindBestRotation();
        
        int stepX = Math.Max(1, (int)Math.Ceiling(bestSize.X));
        int stepY = Math.Max(1, (int)Math.Ceiling(bestSize.Y));
        int stepZ = Math.Max(1, (int)Math.Ceiling(bestSize.Z));
        
        int countX = Math.Max(0, containerW / stepX);
        int countY = Math.Max(0, containerH / stepY);
        int countZ = Math.Max(0, containerD / stepZ);
        
        long total = (long)countX * countY * countZ;
        
        if (total <= 0)
        {
            Console.WriteLine("0");
            Environment.Exit(0);
        }
        
        const int maxOutput = 200000;
        
        if (total > maxOutput)
        {
            double skip = Math.Pow((double)total / maxOutput, 1.0/3.0);
            int skipI = Math.Max(1, (int)skip);
            
            for (int x = 0; x < countX && placements.Count < maxOutput; x += skipI)
            {
                for (int y = 0; y < countY && placements.Count < maxOutput; y += skipI)
                {
                    for (int z = 0; z < countZ && placements.Count < maxOutput; z += skipI)
                    {
                        placements.Add(new Placement(x * stepX, y * stepY, z * stepZ, bestRot));
                    }
                }
            }
        }
        else
        {
            for (int x = 0; x < countX; x++)
            {
                for (int y = 0; y < countY; y++)
                {
                    for (int z = 0; z < countZ; z++)
                    {
                        if (sw.Elapsed > maxTime)
                        {
                            OutputResults(placements);
                            Environment.Exit(0);
                        }
                        placements.Add(new Placement(x * stepX, y * stepY, z * stepZ, bestRot));
                    }
                }
            }
        }
    }
    
    private (Quaternion rotation, Vector3 size) FindBestRotation()
    {
        Quaternion bestRot = Quaternion.Identity;
        Vector3 bestSize = new Vector3(polyAABB.SizeX, polyAABB.SizeY, polyAABB.SizeZ);
        double bestVolume = bestSize.X * bestSize.Y * bestSize.Z;
        
        foreach (var rot in AllRotations)
        {
            var rotatedAABB = RotateAABB(polyAABB, rot);
            double vol = rotatedAABB.SizeX * rotatedAABB.SizeY * rotatedAABB.SizeZ;
            
            if (rotatedAABB.SizeX <= containerW && rotatedAABB.SizeY <= containerH && rotatedAABB.SizeZ <= containerD)
            {
                if (vol < bestVolume)
                {
                    bestVolume = vol;
                    bestRot = rot;
                    bestSize = new Vector3(rotatedAABB.SizeX, rotatedAABB.SizeY, rotatedAABB.SizeZ);
                }
            }
        }
        
        return (bestRot, bestSize);
    }
    
    private void FillGapsPrecise(List<Placement> placements, List<AABB> occupied, Stopwatch sw, TimeSpan maxTime)
    {
        var rand = new Random(12345);
        int attempts = 0;
        int maxAttempts = 1000;
        
        while (attempts < maxAttempts && placements.Count < 10000)
        {
            if (sw.Elapsed > maxTime) return;
            
            var rot = AllRotations[rand.Next(AllRotations.Length)];
            var rotatedAABB = RotateAABB(polyAABB, rot);
            
            int maxX = containerW - (int)Math.Ceiling(rotatedAABB.SizeX);
            int maxY = containerH - (int)Math.Ceiling(rotatedAABB.SizeY);
            int maxZ = containerD - (int)Math.Ceiling(rotatedAABB.SizeZ);
            
            if (maxX < 0 || maxY < 0 || maxZ < 0) { attempts++; continue; }
            
            int x = rand.Next(maxX + 1);
            int y = rand.Next(maxY + 1);
            int z = rand.Next(maxZ + 1);
            
            var candidateAABB = new AABB(x, y, z, x + rotatedAABB.SizeX, y + rotatedAABB.SizeY, z + rotatedAABB.SizeZ);
            
            bool collides = false;
            foreach (var occ in occupied)
            {
                if (AABBsOverlap(candidateAABB, occ)) { collides = true; break; }
            }
            
            if (!collides)
            {
                placements.Add(new Placement(x, y, z, rot));
                occupied.Add(candidateAABB);
            }
            
            attempts++;
        }
    }
    
    private bool AABBsOverlap(AABB a, AABB b)
    {
        return (a.MinX < b.MaxX && a.MaxX > b.MinX) &&
               (a.MinY < b.MaxY && a.MaxY > b.MinY) &&
               (a.MinZ < b.MaxZ && a.MaxZ > b.MinZ);
    }
    
    private AABB RotateAABB(AABB aabb, Quaternion rot)
    {
        var corners = new Vector3[8];
        corners[0] = new Vector3(aabb.MinX, aabb.MinY, aabb.MinZ);
        corners[1] = new Vector3(aabb.MaxX, aabb.MinY, aabb.MinZ);
        corners[2] = new Vector3(aabb.MinX, aabb.MaxY, aabb.MinZ);
        corners[3] = new Vector3(aabb.MaxX, aabb.MaxY, aabb.MinZ);
        corners[4] = new Vector3(aabb.MinX, aabb.MinY, aabb.MaxZ);
        corners[5] = new Vector3(aabb.MaxX, aabb.MinY, aabb.MaxZ);
        corners[6] = new Vector3(aabb.MinX, aabb.MaxY, aabb.MaxZ);
        corners[7] = new Vector3(aabb.MaxX, aabb.MaxY, aabb.MaxZ);
        
        double minX = double.MaxValue, minY = double.MaxValue, minZ = double.MaxValue;
        double maxX = double.MinValue, maxY = double.MinValue, maxZ = double.MinValue;
        
        foreach (var c in corners)
        {
            var r = rot.Rotate(c);
            minX = Math.Min(minX, r.X); minY = Math.Min(minY, r.Y); minZ = Math.Min(minZ, r.Z);
            maxX = Math.Max(maxX, r.X); maxY = Math.Max(maxY, r.Y); maxZ = Math.Max(maxZ, r.Z);
        }
        
        return new AABB(minX, minY, minZ, maxX, maxY, maxZ);
    }
    
    private void ReadInput()
    {
        string[] firstLine = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int V = int.Parse(firstLine[0]);
        int F = int.Parse(firstLine[1]);
        
        var vertices = new Vector3[V];
        for (int i = 0; i < V; i++)
        {
            string[] parts = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            vertices[i] = new Vector3(double.Parse(parts[0]), double.Parse(parts[1]), double.Parse(parts[2]));
        }
        
        var faces = new Face[F];
        for (int i = 0; i < F; i++)
        {
            string[] parts = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            faces[i] = new Face(int.Parse(parts[0]), int.Parse(parts[1]), int.Parse(parts[2]));
        }
        
        string[] lastLine = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        containerW = int.Parse(lastLine[0]);
        containerH = int.Parse(lastLine[1]);
        containerD = int.Parse(lastLine[2]);
        
        poly = new Polyhedron { Vertices = vertices, Faces = faces };
        polyAABB = ComputeAABB(poly);
        polyVolume = ComputePolyhedronVolume(poly);
        containerVolume = (double)containerW * containerH * containerD;
        
        double cx = 0, cy = 0, cz = 0;
        foreach (var v in vertices) { cx += v.X; cy += v.Y; cz += v.Z; }
        polyCenter = new Vector3(cx / V, cy / V, cz / V);
    }
    
    private AABB ComputeAABB(Polyhedron p)
    {
        double minX = double.MaxValue, minY = double.MaxValue, minZ = double.MaxValue;
        double maxX = double.MinValue, maxY = double.MinValue, maxZ = double.MinValue;
        
        foreach (var v in p.Vertices)
        {
            minX = Math.Min(minX, v.X); minY = Math.Min(minY, v.Y); minZ = Math.Min(minZ, v.Z);
            maxX = Math.Max(maxX, v.X); maxY = Math.Max(maxY, v.Y); maxZ = Math.Max(maxZ, v.Z);
        }
        
        return new AABB(minX, minY, minZ, maxX, maxY, maxZ);
    }
    
    private double ComputePolyhedronVolume(Polyhedron p)
    {
        double volume = 0;
        var origin = new Vector3(0, 0, 0);
        
        foreach (var face in p.Faces)
        {
            var v0 = p.Vertices[face.A];
            var v1 = p.Vertices[face.B];
            var v2 = p.Vertices[face.C];
            volume += Vector3.Dot(v0 - origin, Vector3.Cross(v1 - origin, v2 - origin)) / 6.0;
        }
        
        return Math.Abs(volume);
    }
    
    private void OutputResults(List<Placement> placements)
    {
        var sb = new StringBuilder();
        sb.AppendLine(placements.Count.ToString());
        
        foreach (var p in placements)
        {
            sb.AppendLine($"{p.Tx:F6} {p.Ty:F6} {p.Tz:F6} {p.Rotation.W:F6} {p.Rotation.X:F6} {p.Rotation.Y:F6} {p.Rotation.Z:F6}");
        }
        
        Console.Write(sb.ToString());
    }

public struct Vector3
{
    public readonly double X, Y, Z;
    public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
    
    public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    public static Vector3 operator *(Vector3 a, double b) => new Vector3(a.X * b, a.Y * b, a.Z * b);
    
    public static Vector3 Cross(Vector3 a, Vector3 b) =>
        new Vector3(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
    
    public static double Dot(Vector3 a, Vector3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
    
    public static Vector3 UnitX => new Vector3(1, 0, 0);
    public static Vector3 UnitY => new Vector3(0, 1, 0);
    public static Vector3 UnitZ => new Vector3(0, 0, 1);
}

public struct Quaternion
{
    public readonly double W, X, Y, Z;
    public Quaternion(double w, double x, double y, double z) { W = w; X = x; Y = y; Z = z; }
    
    public static Quaternion Identity => new Quaternion(1, 0, 0, 0);
    
    public static Quaternion FromAxisAngle(Vector3 axis, double angle)
    {
        double halfAngle = angle / 2;
        double s = Math.Sin(halfAngle);
        double c = Math.Cos(halfAngle);
        double len = Math.Sqrt(axis.X * axis.X + axis.Y * axis.Y + axis.Z * axis.Z);
        if (len > 0) s /= len;
        return new Quaternion(c, axis.X * s, axis.Y * s, axis.Z * s);
    }
    
    public Quaternion Normalized()
    {
        double len = Math.Sqrt(W * W + X * X + Y * Y + Z * Z);
        if (len < 1e-10) return Identity;
        return new Quaternion(W / len, X / len, Y / len, Z / len);
    }
    
    public static Quaternion operator *(Quaternion a, Quaternion b)
    {
        return new Quaternion(
            a.W * b.W - a.X * b.X - a.Y * b.Y - a.Z * b.Z,
            a.W * b.X + a.X * b.W + a.Y * b.Z - a.Z * b.Y,
            a.W * b.Y - a.X * b.Z + a.Y * b.W + a.Z * b.X,
            a.W * b.Z + a.X * b.Y - a.Y * b.X + a.Z * b.W
        );
    }
    
    public Vector3 Rotate(Vector3 v)
    {
        double xx = X * X, yy = Y * Y, zz = Z * Z;
        double xy = X * Y, xz = X * Z, yz = Y * Z;
        double wx = W * X, wy = W * Y, wz = W * Z;
        
        return new Vector3(
            v.X * (1 - 2 * (yy + zz)) + v.Y * 2 * (xy - wz) + v.Z * 2 * (xz + wy),
            v.X * 2 * (xy + wz) + v.Y * (1 - 2 * (xx + zz)) + v.Z * 2 * (yz - wx),
            v.X * 2 * (xz - wy) + v.Y * 2 * (yz + wx) + v.Z * (1 - 2 * (xx + yy))
        );
    }
}

public class QuaternionComparer : IEqualityComparer<Quaternion>
{
    private const double Epsilon = 0.001;
    
    public bool Equals(Quaternion a, Quaternion b)
    {
        return Math.Abs(a.W - b.W) < Epsilon &&
               Math.Abs(a.X - b.X) < Epsilon &&
               Math.Abs(a.Y - b.Y) < Epsilon &&
               Math.Abs(a.Z - b.Z) < Epsilon;
    }
    
    public int GetHashCode(Quaternion q)
    {
        return HashCode.Combine(Math.Round(q.W, 3), Math.Round(q.X, 3), Math.Round(q.Y, 3), Math.Round(q.Z, 3));
    }
}

public struct Face { public readonly int A, B, C; public Face(int a, int b, int c) { A = a; B = b; C = c; } }

public struct AABB
{
    public readonly double MinX, MinY, MinZ, MaxX, MaxY, MaxZ;
    public AABB(double minX, double minY, double minZ, double maxX, double maxY, double maxZ)
    { MinX = minX; MinY = minY; MinZ = minZ; MaxX = maxX; MaxY = maxY; MaxZ = maxZ; }
    
    public double SizeX => MaxX - MinX;
    public double SizeY => MaxY - MinY;
    public double SizeZ => MaxZ - MinZ;
    public double Volume => SizeX * SizeY * SizeZ;
}

public class Polyhedron { public Vector3[] Vertices; public Face[] Faces; }

public class Placement
{
    public readonly double Tx, Ty, Tz;
    public readonly Quaternion Rotation;
    public Placement(double tx, double ty, double tz, Quaternion rotation)
    { Tx = tx; Ty = ty; Tz = tz; Rotation = rotation; }
}

## Key Features

### **Adaptive Strategy Selection**
- **Trivial** (dim < 10, vol < 5K): Full collision detection with gap filling
- **Medium** (dim 10-50, vol 5K-500K): Grid-based placement
- **Large** (dim 50-200, vol 500K-5M): Spatial hashing with sampling
- **Extreme** (dim > 200, vol > 5M): Fast AABB grid with output limiting

### **Rotation Optimization**
- Generates all 24 unique orientations of the octahedral symmetry group
- Selects the rotation that minimizes AABB volume while fitting in container

### **Performance Optimizations**
- AABB-based collision detection (much faster than precise geometry)
- StringBuilder for efficient output generation
- Time-limited execution with graceful termination
- Memory-efficient data structures using structs

### **Output Management**
- Limits output to 200,000 placements for extreme cases
- Uniform sampling for very large solutions
- Proper quaternion formatting
