```spirv
; SPIR-V Assembly - Geometry Shader for thick lines with arrowheads
; Vulkan 1.0

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

OpDecorate %ubo Block
OpMemberDecorate %ubo 0 ColMajor
OpMemberDecorate %ubo 0 MatrixStride 16
OpMemberDecorate %ubo 0 Offset 0
OpMemberDecorate %ubo 1 ColMajor
OpMemberDecorate %ubo 1 MatrixStride 16
OpMemberDecorate %ubo 1 Offset 64
OpMemberDecorate %ubo 2 ColMajor
OpMemberDecorate %ubo 2 MatrixStride 16
OpMemberDecorate %ubo 2 Offset 128
OpMemberDecorate %ubo 3 Offset 192
OpMemberDecorate %ubo 4 Offset 204
OpMemberDecorate %ubo 5 Offset 216
OpMemberDecorate %ubo 6 Offset 228
OpMemberDecorate %ubo 7 Offset 232
OpDecorate %ubo_buffer Binding 0
OpDecorate %ubo_buffer DescriptorSet 0

; Types
%void = OpTypeVoid
%void_fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%int_6 = OpConstant %int 6
%int_7 = OpConstant %int 7

%uint = OpTypeInt 32 0
%uint_2 = OpConstant %uint 2

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75

; gl_PerVertex input type (array of 2 for InputLines)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; gl_PerVertex output type
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; User-defined inputs (arrays of 2)
%arr_v3_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_2

; UBO type
%ubo = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v3float %float %float
%ptr_ubo = OpTypePointer Uniform %ubo

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_out = OpVariable %ptr_out_pv Output
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%outColor = OpVariable %ptr_out_v3 Output
%ubo_buffer = OpVariable %ptr_ubo Uniform

%main = OpFunction %void None %void_fn
%entry = OpLabel

; Load p0, p1 from gl_in
%p0_pos = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_pos = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p0_v4 = OpLoad %v4float %p0_pos
%p1_v4 = OpLoad %v4float %p1_pos

; Extract worldPos (clip space positions)
%p0_wp_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p1_wp_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_1
%p0 = OpLoad %v3float %p0_wp_ptr
%p1 = OpLoad %v3float %p1_wp_ptr

; Load colors
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

; Compute direction: dir = p1 - p0
%dir = OpFSub %v3float %p1 %p0

; Extract dir.x and dir.y for perpendicular
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1

; Compute perpendicular: perp = normalize(vec2(-dir.y, dir.x))
%neg_dir_y = OpFNegate %float %dir_y
%perp_v2 = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp_len_sq = OpDot %float %perp_v2 %perp_v2
%perp_len = OpSqrt %float %perp_len_sq
%perp_len_safe = OpFMax %float %perp_len %float_0_0001
%inv_len = OpFDiv %float %float_1 %perp_len_safe
%perp_norm = OpVectorTimesScalar %v2float %perp_v2 %inv_len

; Scale perpendicular by thickness
%perp_015_v2 = OpVectorTimesScalar %v2float %perp_norm %float_0_015
%perp_05_v2 = OpVectorTimesScalar %v2float %perp_norm %float_0_05

; Extract perpendicular components as floats
%perp_015_x = OpCompositeExtract %float %perp_015_v2 0
%perp_015_y = OpCompositeExtract %float %perp_015_v2 1
%perp_05_x = OpCompositeExtract %float %perp_05_v2 0
%perp_05_y = OpCompositeExtract %float %perp_05_v2 1

; Compute mid = p0 + 0.75 * (p1 - p0) = p0 + 0.75 * dir
%dir_scaled = OpVectorTimesScalar %v3float %dir %float_0_75
%mid = OpFAdd %v3float %p0 %dir_scaled

; Extract p0 components
%p0_x = OpCompositeExtract %float %p0 0
%p0_y = OpCompositeExtract %float %p0 1
%p0_z = OpCompositeExtract %float %p0 2

; Extract mid components
%mid_x = OpCompositeExtract %float %mid 0
%mid_y = OpCompositeExtract %float %mid 1
%mid_z = OpCompositeExtract %float %mid 2

; Extract p1 components
%p1_x = OpCompositeExtract %float %p1 0
%p1_y = OpCompositeExtract %float %p1 1
%p1_z = OpCompositeExtract %float %p1 2

; ============ SHAFT (4 vertices) ============
; Vertex 1: p0 + perp*0.015
%shaft_v1_x = OpFAdd %float %p0_x %perp_015_x
%shaft_v1_y = OpFAdd %float %p0_y %perp_015_y
%shaft_v1 = OpCompositeConstruct %v3float %shaft_v1_x %shaft_v1_y %p0_z
%shaft_v1_v4 = OpCompositeConstruct %v4float %shaft_v1_x %shaft_v1_y %p0_z %float_1

; Vertex 2: p0 - perp*0.015
%shaft_v2_x = OpFSub %float %p0_x %perp_015_x
%shaft_v2_y = OpFSub %float %p0_y %perp_015_y
%shaft_v2 = OpCompositeConstruct %v3float %shaft_v2_x %shaft_v2_y %p0_z
%shaft_v2_v4 = OpCompositeConstruct %v4float %shaft_v2_x %shaft_v2_y %p0_z %float_1

; Vertex 3: mid + perp*0.015
%shaft_v3_x = OpFAdd %float %mid_x %perp_015_x
%shaft_v3_y = OpFAdd %float %mid_y %perp_015_y
%shaft_v3 = OpCompositeConstruct %v3float %shaft_v3_x %shaft_v3_y %mid_z
%shaft_v3_v4 = OpCompositeConstruct %v4float %shaft_v3_x %shaft_v3_y %mid_z %float_1

; Vertex 4: mid - perp*0.015
%shaft_v4_x = OpFSub %float %mid_x %perp_015_x
%shaft_v4_y = OpFSub %float %mid_y %perp_015_y
%shaft_v4 = OpCompositeConstruct %v3float %shaft_v4_x %shaft_v4_y %mid_z
%shaft_v4_v4 = OpCompositeConstruct %v4float %shaft_v4_x %shaft_v4_y %mid_z %float_1

; Emit shaft vertices (colored c0)
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Emit vertex 1
OpStore %out_pos_ptr %shaft_v1_v4
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 2
OpStore %out_pos_ptr %shaft_v2_v4
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 3
OpStore %out_pos_ptr %shaft_v3_v4
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 4
OpStore %out_pos_ptr %shaft_v4_v4
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

; ============ ARROWHEAD (3 vertices) ============
; Vertex 1: mid + perp*0.05
%arrow_v1_x = OpFAdd %float %mid_x %perp_05_x
%arrow_v1_y = OpFAdd %float %mid_y %perp_05_y
%arrow_v1_v4 = OpCompositeConstruct %v4float %arrow_v1_x %arrow_v1_y %mid_z %float_1

; Vertex 2: mid - perp*0.05
%arrow_v2_x = OpFSub %float %mid_x %perp_05_x
%arrow_v2_y = OpFSub %float %mid_y %perp_05_y
%arrow_v2_v4 = OpCompositeConstruct %v4float %arrow_v2_x %arrow_v2_y %mid_z %float_1

; Vertex 3: p1 (tip)
%p1_v4 = OpCompositeConstruct %v4float %p1_x %p1_y %p1_z %float_1

; Emit arrowhead vertices (colored c1)
; Emit vertex 1
OpStore %out_pos_ptr %arrow_v1_v4
OpStore %outColor %c1
OpEmitVertex

; Emit vertex 2
OpStore %out_pos_ptr %arrow_v2_v4
OpStore %outColor %c1
OpEmitVertex

; Emit vertex 3 (tip)
OpStore %out_pos_ptr %p1_v4
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
