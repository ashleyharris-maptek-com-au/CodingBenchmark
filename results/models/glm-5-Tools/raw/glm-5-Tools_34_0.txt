Here's the complete, runnable Python code for solving the Capacitated Vehicle Routing Problem (CVRP):

```python
#!/usr/bin/env python3
"""
Capacitated Vehicle Routing Problem (CVRP) Solver

Uses multiple heuristics:
- Clarke-Wright Savings Algorithm
- Nearest Neighbor Heuristic  
- Random Insertion with Cheapest Insertion
- 2-opt intra-route improvement
- Relocate and Exchange inter-route improvements
"""

import sys
import math
import random
from typing import List, Tuple, Set, Optional, Dict


class CVRPSolver:
    """Solver for the Capacitated Vehicle Routing Problem."""
    
    def __init__(self):
        self.depot: Tuple[float, float] = (0.0, 0.0)
        self.customers: List[Tuple[float, float, float]] = []
        self.n: int = 0
        self.k: int = 0
        self.capacity: float = 0
        self.distance_cache: Dict[Tuple, float] = {}
    
    def read_input(self) -> bool:
        """Read problem instance from stdin."""
        data = sys.stdin.read().strip().split()
        if not data:
            return False
        
        idx = 0
        self.n = int(data[idx]); idx += 1
        self.k = int(data[idx]); idx += 1
        self.capacity = float(data[idx]); idx += 1
        
        depot_x = float(data[idx]); idx += 1
        depot_y = float(data[idx]); idx += 1
        self.depot = (depot_x, depot_y)
        
        self.customers = []
        for _ in range(self.n):
            x = float(data[idx]); idx += 1
            y = float(data[idx]); idx += 1
            demand = float(data[idx]); idx += 1
            self.customers.append((x, y, demand))
        
        return True
    
    def dist(self, p1: Tuple[float, float], p2: Tuple[float, float]) -> float:
        """Calculate Euclidean distance with caching."""
        if p1 == p2:
            return 0.0
        key = (p1, p2)
        if key not in self.distance_cache:
            self.distance_cache[key] = math.sqrt(
                (p1[0] - p2[0])**2 + (p1[1] - p2[1])**2
            )
        return self.distance_cache[key]
    
    def customer_pos(self, idx: int) -> Tuple[float, float]:
        return (self.customers[idx][0], self.customers[idx][1])
    
    def customer_demand(self, idx: int) -> float:
        return self.customers[idx][2]
    
    def route_distance(self, route: List[int]) -> float:
        if not route:
            return 0.0
        total = self.dist(self.depot, self.customer_pos(route[0]))
        for i in range(len(route) - 1):
            total += self.dist(self.customer_pos(route[i]), self.customer_pos(route[i + 1]))
        total += self.dist(self.customer_pos(route[-1]), self.depot)
        return total
    
    def route_demand(self, route: List[int]) -> float:
        return sum(self.customer_demand(c) for c in route)
    
    def total_distance(self, routes: List[List[int]]) -> float:
        return sum(self.route_distance(r) for r in routes)
    
    def savings_clarke_wright(self) -> List[List[int]]:
        """Clarke-Wright savings algorithm."""
        savings = []
        for i in range(self.n):
            for j in range(i + 1, self.n):
                s = (self.dist(self.depot, self.customer_pos(i)) +
                     self.dist(self.depot, self.customer_pos(j)) -
                     self.dist(self.customer_pos(i), self.customer_pos(j)))
                savings.append((s, i, j))
        
        savings.sort(reverse=True, key=lambda x: x[0])
        
        routes: List[Optional[List[int]]] = [[i] for i in range(self.n)]
        customer_to_route: Dict[int, int] = {i: i for i in range(self.n)}
        
        for _, i, j in savings:
            route_i_idx = customer_to_route[i]
            route_j_idx = customer_to_route[j]
            
            if route_i_idx == route_j_idx:
                continue
            
            route_i = routes[route_i_idx]
            route_j = routes[route_j_idx]
            
            if route_i is None or route_j is None:
                continue
            
            i_at_start = route_i[0] == i
            i_at_end = route_i[-1] == i
            j_at_start = route_j[0] == j
            j_at_end = route_j[-1] == j
            
            new_route = None
            if i_at_end and j_at_start:
                new_route = route_i + route_j
            elif i_at_start and j_at_end:
                new_route = route_j + route_i
            elif i_at_end and j_at_end:
                new_route = route_i + list(reversed(route_j))
            elif i_at_start and j_at_start:
                new_route = list(reversed(route_i)) + route_j
            
            if new_route is None:
                continue
            
            if self.route_demand(new_route) <= self.capacity:
                routes[route_i_idx] = None
                routes[route_j_idx] = None
                routes.append(new_route)
                new_idx = len(routes) - 1
                for c in new_route:
                    customer_to_route[c] = new_idx
        
        return [r for r in routes if r is not None]
    
    def nearest_neighbor(self) -> List[List[int]]:
        """Nearest neighbor construction heuristic."""
        unvisited: Set[int] = set(range(self.n))
        routes: List[List[int]] = []
        
        while unvisited:
            route: List[int] = []
            current_load = 0.0
            current_pos = self.depot
            
            while True:
                best = None
                best_dist = float('inf')
                
                for c in unvisited:
                    if current_load + self.customer_demand(c) <= self.capacity:
                        d = self.dist(current_pos, self.customer_pos(c))
                        if d < best_dist:
                            best_dist = d
                            best = c
                
                if best is None:
                    break
                
                route.append(best)
                current_load += self.customer_demand(best)
                current_pos = self.customer_pos(best)
                unvisited.remove(best)
            
            if route:
                routes.append(route)
        
        return routes
    
    def random_insertion(self) -> List[List[int]]:
        """Random order insertion with cheapest insertion cost."""
        customers = list(range(self.n))
        random.shuffle(customers)
        
        routes: List[List[int]] = []
        
        for c in customers:
            demand_c = self.customer_demand(c)
            pos_c = self.customer_pos(c)
            
            best_route_idx = None
            best_pos = None
            best_increase = float('inf')
            
            for r_idx, route in enumerate(routes):
                if self.route_demand(route) + demand_c > self.capacity:
                    continue
                
                for pos in range(len(route) + 1):
                    increase = self._insertion_cost(route, pos, c)
                    if increase < best_increase:
                        best_increase = increase
                        best_route_idx = r_idx
                        best_pos = pos
            
            if best_route_idx is not None and best_pos is not None:
                routes[best_route_idx].insert(best_pos, c)
            else:
                routes.append([c])
        
        return routes
    
    def _insertion_cost(self, route: List[int], pos: int, customer: int) -> float:
        pos_c = self.customer_pos(customer)
        
        if not route:
            return 2 * self.dist(self.depot, pos_c)
        
        if pos == 0:
            old_dist = self.dist(self.depot, self.customer_pos(route[0]))
            new_dist = self.dist(self.depot, pos_c) + self.dist(pos_c, self.customer_pos(route[0]))
            return new_dist - old_dist
        
        if pos == len(route):
            old_dist = self.dist(self.customer_pos(route[-1]), self.depot)
            new_dist = self.dist(self.customer_pos(route[-1]), pos_c) + self.dist(pos_c, self.depot)
            return new_dist - old_dist
        
        prev_pos = self.customer_pos(route[pos - 1])
        next_pos = self.customer_pos(route[pos])
        old_dist = self.dist(prev_pos, next_pos)
        new_dist = self.dist(prev_pos, pos_c) + self.dist(pos_c, next_pos)
        return new_dist - old_dist
    
    def two_opt_route(self, route: List[int]) -> List[int]:
        """2-opt improvement for a single route."""
        if len(route) < 2:
            return route[:]
        
        improved = True
        best_route = route[:]
        
        while improved:
            improved = False
            for i in range(len(best_route) - 1):
                for j in range(i + 2, len(best_route)):
                    new_route = (best_route[:i + 1] + 
                                list(reversed(best_route[i + 1:j + 1])) + 
                                best_route[j + 1:])
                    if self.route_distance(new_route) < self.route_distance(best_route):
                        best_route = new_route
                        improved = True
                        break
                if improved:
                    break
        
        return best_route
    
    def two_opt_all(self, routes: List[List[int]]) -> List[List[int]]:
        return [self.two_opt_route(r) for r in routes]
    
    def relocate(self, routes: List[List[int]]) -> List[List[int]]:
        """Move a customer from one route to another."""
        routes = [r[:] for r in routes]
        improved = True
        best_total = self.total_distance(routes)
        
        while improved:
            improved = False
            for i in range(len(routes)):
                for pos_i in range(len(routes[i])):
                    customer = routes[i][pos_i]
                    demand_c = self.customer_demand(customer)
                    
                    for j in range(len(routes)):
                        if i == j:
                            continue
                        
                        if self.route_demand(routes[j]) + demand_c > self.capacity:
                            continue
                        
                        for pos_j in range(len(routes[j]) + 1):
                            new_routes = [r[:] for r in routes]
                            new_routes[i].pop(pos_i)
                            new_routes[j].insert(pos_j, customer)
                            
                            if not new_routes[i]:
                                new_routes.pop(i)
                            
                            new_total = self.total_distance(new_routes)
                            if new_total < best_total - 1e-9:
                                routes = new_routes
                                best_total = new_total
                                improved = True
                                break
                        
                        if improved:
                            break
                    if improved:
                        break
                if improved:
                    break
        
        return routes
    
    def exchange(self, routes: List[List[int]]) -> List[List[int]]:
        """Swap two customers between routes."""
        routes = [r[:] for r in routes]
        improved = True
        best_total = self.total_distance(routes)
        
        while improved:
            improved = False
            for i in range(len(routes)):
                for j in range(i + 1, len(routes)):
                    for pos_i in range(len(routes[i])):
                        for pos_j in range(len(routes[j])):
                            c_i = routes[i][pos_i]
                            c_j = routes[j][pos_j]
                            
                            new_demand_i = (self.route_demand(routes[i]) -
                                           self.customer_demand(c_i) +
                                           self.customer_demand(c_j))
                            new_demand_j = (self.route_demand(routes[j]) -
                                           self.customer_demand(c_j) +
                                           self.customer_demand(c_i))
                            
                            if new_demand_i > self.capacity or new_demand_j > self.capacity:
                                continue
                            
                            new_routes = [r[:] for r in routes]
                            new_routes[i][pos_i] = c_j
                            new_routes[j][pos_j] = c_i
                            
                            new_total = self.total_distance(new_routes)
                            if new_total < best_total - 1e-9:
                                routes = new_routes
                                best_total = new_total
                                improved = True
                                break
                        
                        if improved:
                            break
                    if improved:
                        break
                if improved:
                    break
        
        return routes
    
    def improve_routes(self, routes: List[List[int]]) -> List[List[int]]:
        routes = self.two_opt_all(routes)
        routes = self.relocate(routes)
        routes = self.exchange(routes)
        routes = self.two_opt_all(routes)
        return routes
    
    def solve(self) -> List[List[int]]:
        """Solve CVRP using multiple heuristics."""
        best_routes = None
        best_dist = float('inf')
        
        # Clarke-Wright
        routes = self.savings_clarke_wright()
        routes = self.improve_routes(routes)
        dist = self.total_distance(routes)
        if dist < best_dist:
            best_dist = dist
            best_routes = routes
        
        # Nearest neighbor
        routes = self.nearest_neighbor()
        routes = self.improve_routes(routes)
        dist = self.total_distance(routes)
        if dist < best_dist:
            best_dist = dist
            best_routes = routes
        
        # Random starts
        random.seed(42)
        for _ in range(5):
            routes = self.random_insertion()
            routes = self.improve_routes(routes)
            dist = self.total_distance(routes)
            if dist < best_dist:
                best_dist = dist
                best_routes = routes
        
        return best_routes if best_routes is not None else []


def main():
    solver = CVRPSolver()
    
    if not solver.read_input():
        print("0.00")
        print("0")
        return
    
    routes = solver.solve()
    
    total_dist = solver.total_distance(routes)
    print(f"{total_dist:.2f}")
    print(len(routes))
    for route in routes:
        print(' '.join(map(str, route)))


if __name__ == "__main__":
    main()
```

**Key Features:**

1. **Multiple Construction Heuristics:**
   - **Clarke-Wright Savings**: Merges routes based on distance savings
   - **Nearest Neighbor**: Greedy closest-customer selection
   - **Cheapest Insertion**: Random order with optimal insertion positions

2. **Local Search Improvements:**
   - **2-opt**: Eliminates route crossings by reversing segments
   - **Relocate**: Moves customers between routes
   - **Exchange**: Swaps customers between routes

3. **Efficiency:**
   - Distance caching for repeated calculations
   - Multiple starting points to avoid local optima
   - First-improvement strategy for faster iterations

4. **Output Format:**
   - Total distance (2 decimal places)
   - Number of routes
   - Each route as space-separated customer indices (0-based)
