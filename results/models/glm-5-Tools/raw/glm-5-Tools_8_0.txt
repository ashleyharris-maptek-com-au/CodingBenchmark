Here's the complete Rust solution:

```rust
use std::cmp::Reverse;
use std::collections::BinaryHeap;
use std::io::{self, BufRead, BufWriter, Write};

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut stdout = BufWriter::new(stdout.lock());
    
    let mut lines = stdin.lock().lines();
    
    // Parse dimensions
    let first_line = lines.next().unwrap().unwrap();
    let mut dims = first_line.split_whitespace();
    let height: usize = dims.next().unwrap().parse().unwrap();
    let width: usize = dims.next().unwrap().parse().unwrap();
    
    // Read maze into a flat vector (row-major order)
    // This is more cache-friendly than Vec<Vec<char>>
    let mut maze: Vec<u8> = Vec::with_capacity(height * width);
    let mut start: usize = 0;
    let mut end: usize = 0;
    let mut start_found = false;
    let mut end_found = false;
    
    for y in 0..height {
        let line = lines.next().unwrap().unwrap();
        let row_bytes = line.as_bytes();
        
        for (x, &byte) in row_bytes.iter().enumerate() {
            let idx = y * width + x;
            if x < width {
                match byte {
                    b'A' => {
                        start = idx;
                        start_found = true;
                        maze.push(b' '); // Start is walkable
                    }
                    b'B' => {
                        end = idx;
                        end_found = true;
                        maze.push(b' '); // End is walkable
                    }
                    other => maze.push(other),
                }
            }
        }
        
        // Pad with walls if line is shorter than width
        while maze.len() < (y + 1) * width {
            maze.push(b'#');
        }
    }
    
    // Handle case where start or end not found
    if !start_found || !end_found {
        writeln!(stdout, "0").unwrap();
        return;
    }
    
    let total_cells = height * width;
    
    // A* algorithm with Manhattan distance heuristic
    // f(n) = g(n) + h(n), where g is cost from start, h is heuristic to end
    
    let start_x = start % width;
    let start_y = start / width;
    let end_x = end % width;
    let end_y = end / width;
    
    // Manhattan distance heuristic
    fn manhattan(x1: usize, y1: usize, x2: usize, y2: usize) -> u32 {
        (if x1 > x2 { x1 - x2 } else { x2 - x1 }) as u32 + 
        (if y1 > y2 { y1 - y2 } else { y2 - y1 }) as u32
    }
    
    // Initialize arrays
    // dist: g-score (cost from start), initialized to max
    let mut dist: Vec<u32> = vec![u32::MAX; total_cells];
    // parent: stores the index of the parent cell for path reconstruction
    let mut parent: Vec<usize> = vec![usize::MAX; total_cells];
    // visited: tracks which cells have been processed
    let mut visited: Vec<bool> = vec![false; total_cells];
    
    // Priority queue: (f_score, g_score, index)
    // Using Reverse to make BinaryHeap a min-heap
    // Tie-breaker: prefer lower g_score (closer to goal in case of equal f)
    let mut pq: BinaryHeap<Reverse<(u32, u32, usize)>> = BinaryHeap::new();
    
    dist[start] = 0;
    let h_start = manhattan(start_x, start_y, end_x, end_y);
    pq.push(Reverse((h_start, 0, start)));
    
    // Direction offsets: right, down, left, up
    let dx: [isize; 4] = [1, 0, -1, 0];
    let dy: [isize; 4] = [0, 1, 0, -1];
    
    while let Some(Reverse((_, g, current))) = pq.pop() {
        if visited[current] {
            continue;
        }
        
        visited[current] = true;
        
        // Found the goal
        if current == end {
            // Reconstruct path
            let mut path: Vec<usize> = Vec::new();
            let mut node = end;
            
            while node != usize::MAX {
                path.push(node);
                node = parent[node];
            }
            
            path.reverse();
            
            // Output
            writeln!(stdout, "{}", path.len()).unwrap();
            for &idx in &path {
                let x = idx % width;
                let y = idx / width;
                writeln!(stdout, "{} {}", x, y).unwrap();
            }
            return;
        }
        
        let cur_x = current % width;
        let cur_y = current / width;
        
        // Explore neighbors
        for dir in 0..4 {
            let nx = cur_x as isize + dx[dir];
            let ny = cur_y as isize + dy[dir];
            
            // Bounds check
            if nx < 0 || nx >= width as isize || ny < 0 || ny >= height as isize {
                continue;
            }
            
            let nx = nx as usize;
            let ny = ny as usize;
            let neighbor = ny * width + nx;
            
            // Skip if already visited or is a wall
            if visited[neighbor] || maze[neighbor] == b'#' {
                continue;
            }
            
            let new_g = g + 1;
            
            if new_g < dist[neighbor] {
                dist[neighbor] = new_g;
                parent[neighbor] = current;
                
                let h = manhattan(nx, ny, end_x, end_y);
                let f = new_g + h;
                
                pq.push(Reverse((f, new_g, neighbor)));
            }
        }
    }
    
    // No path found
    writeln!(stdout, "0").unwrap();
}
```

## Key Design Decisions:

1. **Flat Vector Storage**: The maze is stored in a single `Vec<u8>` in row-major order, which is more cache-friendly and reduces memory overhead compared to `Vec<Vec<char>>`.

2. **A* Algorithm**: Uses Manhattan distance as the heuristic - guaranteed to find the shortest path while being more efficient than BFS for mazes with clear directional paths.

3. **Memory Layout**:
   - `maze`: 1 byte per cell
   - `dist`: 4 bytes per cell (u32)
   - `parent`: 8 bytes per cell (usize)
   - `visited`: 1 byte per cell
   
   Total: ~14 bytes per cell → ~35GB for 50k×50k maze (well within 128GB RAM)

4. **Priority Queue**: `BinaryHeap<Reverse<...>>` gives us min-heap behavior, ordered by `f = g + h`.

5. **Optimizations**:
   - Buffered I/O for both input and output
   - Early termination when goal is reached
   - Skip already-visited nodes
   - Inline bounds checking

This solution efficiently handles both tiny mazes (low overhead) and massive mazes (memory-efficient data structures).
