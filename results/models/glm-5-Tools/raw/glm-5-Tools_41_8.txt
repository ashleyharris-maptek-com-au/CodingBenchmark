; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%inWorldPos = OpVariable %ptr_Input_vec3 Input
%inNormal = OpVariable %ptr_Input_vec3 Input
%inUV = OpVariable %ptr_Input_vec2 Input
%inTangent = OpVariable %ptr_Input_vec3 Input
%inColor = OpVariable %ptr_Input_vec3 Input
%outColor = OpVariable %ptr_Output_vec4 Output
%ubo = OpVariable %ptr_Uniform_UBO Uniform
%void_fn = OpTypeFunction %void
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_25 = OpConstant %float 0.25
%float_0_5 = OpConstant %float 0.5
%float_0_75 = OpConstant %float 0.75
%float_0_2 = OpConstant %float 0.2
%float_0_4 = OpConstant %float 0.4
%float_0_7 = OpConstant %float 0.7
%main = OpFunction %void None %void_fn
%entry = OpLabel
%N_raw = OpLoad %vec3 %inNormal
%N = OpExtInst %vec3 %1 Normalize %N_raw
%lightPos_ptr = OpAccessChain %ptr_Uniform_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%worldPos = OpLoad %vec3 %inWorldPos
%lightPos_vec3 = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%toLight = OpFSub %vec3 %lightPos_vec3 %worldPos
%L = OpExtInst %vec3 %1 Normalize %toLight
%dot_NL = OpDot %float %N %L
%d_max = OpExtInst %float %1 FMax %dot_NL %float_0
%cond_075 = OpFOrdGreaterThan %bool %d_max %float_0_75
%band_075 = OpSelect %float %cond_075 %float_1 %float_0
%cond_05 = OpFOrdGreaterThan %bool %d_max %float_0_5
%band_05 = OpSelect %float %cond_05 %float_0_7 %float_0
%cond_025 = OpFOrdGreaterThan %bool %d_max %float_0_25
%band_025 = OpSelect %float %cond_025 %float_0_4 %float_0_2
%not_075 = OpLogicalNot %bool %cond_075
%and_05 = OpLogicalAnd %bool %not_075 %cond_05
%not_05 = OpLogicalNot %bool %cond_05
%and_025 = OpLogicalAnd %bool %not_05 %cond_025
%not_025 = OpLogicalNot %bool %cond_025
%band_final = OpSelect %float %cond_075 %float_1 %float_0
%band_final2 = OpSelect %float %and_05 %float_0_7 %band_final
%band_final3 = OpSelect %float %and_025 %float_0_4 %band_final2
%band = OpSelect %float %not_025 %float_0_2 %band_final3
%vertexColor = OpLoad %vec3 %inColor
%color = OpVectorTimesScalar %vec3 %vertexColor %band
%color4 = OpCompositeConstruct %vec4 %color %float_1
OpStore %outColor %color4
OpReturn
OpFunctionEnd
