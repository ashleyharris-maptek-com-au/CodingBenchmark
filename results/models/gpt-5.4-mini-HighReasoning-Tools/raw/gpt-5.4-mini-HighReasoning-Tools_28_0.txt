use std::io::{self, Read, Write};

struct Scanner {
    input: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> io::Result<Self> {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input)?;
        Ok(Self { input, idx: 0 })
    }

    fn next_usize(&mut self) -> usize {
        while self.idx < self.input.len() && self.input[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        let mut value = 0usize;
        while self.idx < self.input.len() && !self.input[self.idx].is_ascii_whitespace() {
            value = value * 10 + (self.input[self.idx] - b'0') as usize;
            self.idx += 1;
        }
        value
    }
}

fn select_vertex(
    v: usize,
    selected: &mut [u8],
    degree: &mut [u32],
    active_edge: &mut [u8],
    offsets: &[usize],
    adj: &[u32],
    edges_u: &[u32],
    edges_v: &[u32],
    stack: &mut Vec<usize>,
    order: &mut Vec<usize>,
) {
    if selected[v] != 0 {
        return;
    }

    selected[v] = 1;
    order.push(v);
    degree[v] = 0;

    let start = offsets[v];
    let end = offsets[v + 1];
    for idx in start..end {
        let eid = adj[idx] as usize;
        if active_edge[eid] == 0 {
            continue;
        }
        active_edge[eid] = 0;

        let a = edges_u[eid] as usize;
        let b = edges_v[eid] as usize;
        let other = if a == v { b } else { a };
        if degree[other] > 0 {
            degree[other] -= 1;
            if degree[other] == 1 {
                stack.push(other);
            }
        }
    }
}

fn main() -> io::Result<()> {
    let mut scanner = Scanner::new()?;
    let n = scanner.next_usize();
    let m = scanner.next_usize();

    let mut edges_u = Vec::with_capacity(m);
    let mut edges_v = Vec::with_capacity(m);
    let mut degree = vec![0u32; n];
    let mut loop_forced = vec![0u8; n];

    for _ in 0..m {
        let u = scanner.next_usize();
        let v = scanner.next_usize();
        if u == v {
            loop_forced[u] = 1;
        } else {
            edges_u.push(u as u32);
            edges_v.push(v as u32);
            degree[u] += 1;
            degree[v] += 1;
        }
    }

    let edge_count = edges_u.len();

    let mut offsets = vec![0usize; n + 1];
    for i in 0..n {
        offsets[i + 1] = offsets[i] + degree[i] as usize;
    }

    let mut adj = vec![0u32; edge_count * 2];
    let mut fill = offsets.clone();
    for eid in 0..edge_count {
        let u = edges_u[eid] as usize;
        let v = edges_v[eid] as usize;
        adj[fill[u]] = eid as u32;
        fill[u] += 1;
        adj[fill[v]] = eid as u32;
        fill[v] += 1;
    }

    let mut selected = vec![0u8; n];
    let mut active_edge = vec![1u8; edge_count];
    let mut stack = Vec::new();
    let mut order = Vec::new();

    for v in 0..n {
        if loop_forced[v] != 0 {
            select_vertex(
                v,
                &mut selected,
                &mut degree,
                &mut active_edge,
                &offsets,
                &adj,
                &edges_u,
                &edges_v,
                &mut stack,
                &mut order,
            );
        }
    }

    while let Some(v) = stack.pop() {
        if selected[v] != 0 || degree[v] != 1 {
            continue;
        }

        let mut neighbor = None;
        for idx in offsets[v]..offsets[v + 1] {
            let eid = adj[idx] as usize;
            if active_edge[eid] == 0 {
                continue;
            }
            let a = edges_u[eid] as usize;
            let b = edges_v[eid] as usize;
            neighbor = Some(if a == v { b } else { a });
            break;
        }

        if let Some(w) = neighbor {
            select_vertex(
                w,
                &mut selected,
                &mut degree,
                &mut active_edge,
                &offsets,
                &adj,
                &edges_u,
                &edges_v,
                &mut stack,
                &mut order,
            );
        }
    }

    for eid in 0..edge_count {
        if active_edge[eid] == 0 {
            continue;
        }
        let u = edges_u[eid] as usize;
        let v = edges_v[eid] as usize;
        if selected[u] == 0 && selected[v] == 0 {
            selected[u] = 1;
            selected[v] = 1;
            order.push(u);
            order.push(v);
        }
    }

    let mut candidate = selected.clone();
    for v in 0..n {
        if loop_forced[v] != 0 {
            candidate[v] = 0;
        }
    }
    for eid in 0..edge_count {
        let u = edges_u[eid] as usize;
        let v = edges_v[eid] as usize;
        if selected[u] != 0 && selected[v] == 0 {
            candidate[u] = 0;
        }
        if selected[v] != 0 && selected[u] == 0 {
            candidate[v] = 0;
        }
    }

    let mut removed_candidate = vec![0u8; n];
    for &v in order.iter().rev() {
        if candidate[v] == 0 || selected[v] == 0 {
            continue;
        }

        let mut removable = true;
        for idx in offsets[v]..offsets[v + 1] {
            let eid = adj[idx] as usize;
            let a = edges_u[eid] as usize;
            let b = edges_v[eid] as usize;
            let other = if a == v { b } else { a };
            if candidate[other] != 0 && removed_candidate[other] != 0 {
                removable = false;
                break;
            }
        }

        if removable {
            removed_candidate[v] = 1;
            selected[v] = 0;
        }
    }

    for eid in 0..edge_count {
        let u = edges_u[eid] as usize;
        let v = edges_v[eid] as usize;
        if selected[u] == 0 && selected[v] == 0 {
            selected[u] = 1;
        }
    }

    for v in 0..n {
        if loop_forced[v] != 0 {
            selected[v] = 1;
        }
    }

    let mut count = 0usize;
    for &flag in &selected {
        if flag != 0 {
            count += 1;
        }
    }

    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());

    writeln!(out, "{}", count)?;
    let mut first = true;
    for v in 0..n {
        if selected[v] != 0 {
            if !first {
                write!(out, " ")?;
            }
            first = false;
            write!(out, "{}", v)?;
        }
    }
    writeln!(out)?;
    out.flush()?;
    Ok(())
}
