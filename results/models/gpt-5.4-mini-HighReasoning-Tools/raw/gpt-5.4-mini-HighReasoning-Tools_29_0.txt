using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;

public static class Program
{
    static int N;
    static int M;
    static int[] Head = Array.Empty<int>();
    static int[] To = Array.Empty<int>();
    static int[] DegOrig = Array.Empty<int>();
    static int[] DegWork = Array.Empty<int>();
    static int[] Order = Array.Empty<int>();
    static int[] Pos = Array.Empty<int>();
    static int[] Core = Array.Empty<int>();
    static int[] Mark = Array.Empty<int>();
    static int MarkToken = 1;
    static int[] Path = Array.Empty<int>();
    static int[] Best = Array.Empty<int>();
    static int BestSize = 0;
    static long DeadlineTicks;
    static int MaxCore = 0;

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        N = fs.NextInt();
        M = fs.NextInt();

        if (N <= 0)
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }

        int[] eu = new int[M];
        int[] ev = new int[M];
        int valid = 0;
        for (int i = 0; i < M; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if ((uint)u >= (uint)N || (uint)v >= (uint)N || u == v)
            {
                continue;
            }
            eu[valid] = u;
            ev[valid] = v;
            valid++;
        }
        M = valid;

        BuildCsr(eu, ev);
        SortAdjacencyRanges();
        ComputeDegeneracy();

        Mark = new int[N];
        Path = new int[N];
        Best = new int[N];

        DeadlineTicks = Stopwatch.GetTimestamp() + Stopwatch.Frequency * 280L;

        // A few greedy starts give a strong lower bound before exact search.
        int greedySeeds = Math.Min(N, 32);
        for (int i = 0; i < greedySeeds && !TimeUp(); i++)
        {
            int seed = Order[N - 1 - i];
            if (DegOrig[seed] + 1 <= BestSize)
            {
                continue;
            }
            GreedyFromSeed(seed);
        }

        bool exactAll = (N <= 2000) || (MaxCore <= 80);
        int exactLimit = (N <= 2000) ? N : 80;
        int heuristicSeedLimit = exactAll ? N : Math.Min(N, 128);

        for (int i = 0; i < heuristicSeedLimit && !TimeUp(); i++)
        {
            int seed = Order[N - 1 - i];
            if (DegOrig[seed] + 1 <= BestSize)
            {
                continue;
            }

            int candCount = BuildLaterNeighbors(seed, exactLimit, null, 0);
            if (candCount == 0)
            {
                if (BestSize < 1)
                {
                    BestSize = 1;
                    Best[0] = seed;
                }
                continue;
            }

            if (candCount + 1 > BestSize && candCount <= exactLimit)
            {
                int[] cand = new int[candCount];
                BuildLaterNeighbors(seed, exactLimit, cand, candCount);
                Path[0] = seed;
                ExactExpand(1, cand, candCount);
            }
            else
            {
                GreedyFromSeed(seed);
            }
        }

        Array.Sort(Best, 0, BestSize);
        Console.WriteLine(BestSize);
        if (BestSize > 0)
        {
            for (int i = 0; i < BestSize; i++)
            {
                if (i > 0) Console.Write(' ');
                Console.Write(Best[i]);
            }
        }
        Console.WriteLine();
    }

    static void BuildCsr(int[] eu, int[] ev)
    {
        Head = new int[N + 1];
        for (int i = 0; i < M; i++)
        {
            Head[eu[i] + 1]++;
            Head[ev[i] + 1]++;
        }
        for (int i = 1; i <= N; i++)
        {
            Head[i] += Head[i - 1];
        }

        To = new int[Head[N]];
        int[] ptr = new int[N];
        Array.Copy(Head, ptr, N);
        for (int i = 0; i < M; i++)
        {
            int u = eu[i];
            int v = ev[i];
            To[ptr[u]++] = v;
            To[ptr[v]++] = u;
        }

        DegOrig = new int[N];
        DegWork = new int[N];
        for (int i = 0; i < N; i++)
        {
            int d = Head[i + 1] - Head[i];
            DegOrig[i] = d;
            DegWork[i] = d;
        }
    }

    static void SortAdjacencyRanges()
    {
        for (int i = 0; i < N; i++)
        {
            int start = Head[i];
            int len = Head[i + 1] - start;
            if (len > 1)
            {
                Array.Sort(To, start, len);
            }
        }
    }

    static void ComputeDegeneracy()
    {
        Order = new int[N];
        Pos = new int[N];
        Core = new int[N];
        bool[] removed = new bool[N];
        var pq = new PriorityQueue<int, int>();
        for (int v = 0; v < N; v++)
        {
            pq.Enqueue(v, DegWork[v]);
        }

        int idx = 0;
        MaxCore = 0;
        while (idx < N && pq.Count > 0)
        {
            int v;
            int d;
            do
            {
                if (!pq.TryDequeue(out v, out d))
                {
                    return;
                }
            } while (removed[v] || d != DegWork[v]);

            removed[v] = true;
            Order[idx] = v;
            Pos[v] = idx;
            Core[v] = d;
            if (d > MaxCore) MaxCore = d;
            idx++;

            for (int p = Head[v]; p < Head[v + 1]; p++)
            {
                int u = To[p];
                if (removed[u]) continue;
                DegWork[u]--;
                pq.Enqueue(u, DegWork[u]);
            }
        }
    }

    static int BuildLaterNeighbors(int seed, int limit, int[]? dest, int destLen)
    {
        int count = 0;
        int posSeed = Pos[seed];
        for (int p = Head[seed]; p < Head[seed + 1]; p++)
        {
            int u = To[p];
            if (Pos[u] <= posSeed) continue;
            if (DegOrig[u] + 1 <= BestSize) continue;
            if (dest != null)
            {
                dest[count] = u;
            }
            count++;
            if (dest == null && count > limit)
            {
                // Early stop when only counting is requested.
                return limit + 1;
            }
            if (dest != null && count >= destLen)
            {
                return count;
            }
        }
        return count;
    }

    static void GreedyFromSeed(int seed)
    {
        if (TimeUp()) return;

        int posSeed = Pos[seed];
        int candCount = 0;
        int[] cand = new int[Math.Max(4, DegOrig[seed])];
        for (int p = Head[seed]; p < Head[seed + 1]; p++)
        {
            int u = To[p];
            if (u == seed) continue;
            if (Pos[u] <= posSeed) continue;
            if (DegOrig[u] + 1 <= BestSize) continue;
            if (candCount == cand.Length)
            {
                Array.Resize(ref cand, cand.Length * 2);
            }
            cand[candCount++] = u;
        }

        Path[0] = seed;
        int depth = 1;

        int token = ++MarkToken;
        if (MarkToken == int.MaxValue - 1) MarkToken = 1;
        for (int i = 0; i < candCount; i++) Mark[cand[i]] = token;

        while (candCount > 0 && !TimeUp())
        {
            int bestV = -1;
            int bestScore = -1;
            int bestDeg = -1;

            if (candCount > 1024)
            {
                for (int i = 0; i < candCount; i++)
                {
                    int v = cand[i];
                    int deg = DegOrig[v];
                    if (deg > bestDeg)
                    {
                        bestDeg = deg;
                        bestV = v;
                    }
                }
            }
            else
            {
                for (int i = 0; i < candCount; i++)
                {
                    int v = cand[i];
                    int score = 0;
                    for (int p = Head[v]; p < Head[v + 1]; p++)
                    {
                        if (Mark[To[p]] == token) score++;
                    }
                    int deg = DegOrig[v];
                    if (score > bestScore || (score == bestScore && deg > bestDeg))
                    {
                        bestScore = score;
                        bestDeg = deg;
                        bestV = v;
                    }
                }
            }

            if (bestV < 0) break;

            Path[depth++] = bestV;
            if (depth > BestSize)
            {
                BestSize = depth;
                Array.Copy(Path, Best, depth);
            }

            int newToken = ++MarkToken;
            if (MarkToken == int.MaxValue - 1) MarkToken = 1;
            for (int p = Head[bestV]; p < Head[bestV + 1]; p++) Mark[To[p]] = newToken;

            int newCount = 0;
            for (int i = 0; i < candCount; i++)
            {
                int u = cand[i];
                if (u != bestV && Mark[u] == newToken)
                {
                    cand[newCount++] = u;
                }
            }
            candCount = newCount;
            token = newToken;
        }

        if (depth > BestSize)
        {
            BestSize = depth;
            Array.Copy(Path, Best, depth);
        }
    }

    static void ExactExpand(int depth, int[] cand, int candCount)
    {
        if (TimeUp()) return;
        if (depth + candCount <= BestSize) return;
        if (candCount == 0)
        {
            if (depth > BestSize)
            {
                BestSize = depth;
                Array.Copy(Path, Best, depth);
            }
            return;
        }

        int[] order = new int[candCount];
        int[] colors = new int[candCount];
        ColorSort(cand, candCount, order, colors);

        int[] next = new int[candCount];
        for (int i = candCount - 1; i >= 0; i--)
        {
            if (depth + colors[i] <= BestSize) return;

            int v = order[i];
            Path[depth] = v;

            int nextCount = 0;
            for (int j = 0; j < i; j++)
            {
                int u = order[j];
                if (IsAdjacent(v, u))
                {
                    next[nextCount++] = u;
                }
            }

            ExactExpand(depth + 1, next, nextCount);
            if (TimeUp()) return;
        }
    }

    static void ColorSort(int[] cand, int count, int[] order, int[] colors)
    {
        int colorCount = 0;
        int[][] classes = new int[count][];
        int[] classSize = new int[count];

        for (int i = 0; i < count; i++)
        {
            int v = cand[i];
            int assigned = -1;
            for (int c = 0; c < colorCount; c++)
            {
                bool ok = true;
                int[] cls = classes[c];
                int sz = classSize[c];
                for (int j = 0; j < sz; j++)
                {
                    if (IsAdjacent(v, cls[j]))
                    {
                        ok = false;
                        break;
                    }
                }
                if (ok)
                {
                    cls[sz] = v;
                    classSize[c] = sz + 1;
                    assigned = c;
                    break;
                }
            }

            if (assigned < 0)
            {
                classes[colorCount] = new int[count];
                classes[colorCount][0] = v;
                classSize[colorCount] = 1;
                assigned = colorCount;
                colorCount++;
            }
        }

        int idx = 0;
        for (int c = 0; c < colorCount; c++)
        {
            int[] cls = classes[c];
            int sz = classSize[c];
            for (int j = 0; j < sz; j++)
            {
                order[idx] = cls[j];
                colors[idx] = c + 1;
                idx++;
            }
        }
    }

    static bool IsAdjacent(int u, int v)
    {
        int l = Head[u];
        int r = Head[u + 1] - 1;
        while (l <= r)
        {
            int mid = (l + r) >> 1;
            int x = To[mid];
            if (x == v) return true;
            if (x < v) l = mid + 1;
            else r = mid - 1;
        }
        return false;
    }

    static bool TimeUp()
    {
        return Stopwatch.GetTimestamp() >= DeadlineTicks;
    }

    sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return 0;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32 && c != -1)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }
    }
}
