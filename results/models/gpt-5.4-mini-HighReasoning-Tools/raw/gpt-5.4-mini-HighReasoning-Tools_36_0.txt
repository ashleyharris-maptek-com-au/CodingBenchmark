#include <algorithm>
#include <cstdint>
#include <iostream>
#include <queue>
#include <vector>
#include <numeric>
#include <functional>
#include <intrin.h>

using namespace std;

static inline int popcount64(uint64_t x) {
#if defined(_MSC_VER)
    return (int)__popcnt64(x);
#else
    return __builtin_popcountll(x);
#endif
}

static inline bool testBit(const vector<uint64_t>& bs, int v) {
    return (bs[v >> 6] >> (v & 63)) & 1ULL;
}

static inline void setBit(vector<uint64_t>& bs, int v) {
    bs[v >> 6] |= 1ULL << (v & 63);
}

static inline void clearBit(vector<uint64_t>& bs, int v) {
    bs[v >> 6] &= ~(1ULL << (v & 63));
}

static inline int countBits(const vector<uint64_t>& bs) {
    int s = 0;
    for (uint64_t x : bs) s += popcount64(x);
    return s;
}

struct BipartiteResult {
    vector<int> vertices;
};

static BipartiteResult solveBipartiteComponent(
    const vector<vector<int>>& g,
    const vector<int>& globalOfLocal
) {
    int n = (int)g.size();
    vector<int> color(n, -1);
    bool ok = true;
    for (int s = 0; s < n && ok; ++s) {
        if (color[s] != -1) continue;
        queue<int> q;
        color[s] = 0;
        q.push(s);
        while (!q.empty() && ok) {
            int u = q.front();
            q.pop();
            for (int v : g[u]) {
                if (color[v] == -1) {
                    color[v] = color[u] ^ 1;
                    q.push(v);
                } else if (color[v] == color[u]) {
                    ok = false;
                    break;
                }
            }
        }
    }
    if (!ok) return {};

    vector<int> leftId(n, -1), rightId(n, -1);
    vector<int> leftVerts, rightVerts;
    for (int i = 0; i < n; ++i) {
        if (color[i] == 0) {
            leftId[i] = (int)leftVerts.size();
            leftVerts.push_back(i);
        } else {
            rightId[i] = (int)rightVerts.size();
            rightVerts.push_back(i);
        }
    }

    int L = (int)leftVerts.size();
    int R = (int)rightVerts.size();
    vector<vector<int>> adjL(L);
    for (int u = 0; u < n; ++u) {
        if (color[u] != 0) continue;
        int lu = leftId[u];
        for (int v : g[u]) {
            if (color[v] == 1) adjL[lu].push_back(rightId[v]);
        }
    }

    vector<int> pairU(L, -1), pairV(R, -1), dist(L);
    auto bfs = [&]() -> bool {
        queue<int> q;
        bool found = false;
        for (int u = 0; u < L; ++u) {
            if (pairU[u] == -1) {
                dist[u] = 0;
                q.push(u);
            } else {
                dist[u] = -1;
            }
        }
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (int v : adjL[u]) {
                int u2 = pairV[v];
                if (u2 == -1) {
                    found = true;
                } else if (dist[u2] == -1) {
                    dist[u2] = dist[u] + 1;
                    q.push(u2);
                }
            }
        }
        return found;
    };
    function<bool(int)> dfs = [&](int u) -> bool {
        for (int v : adjL[u]) {
            int u2 = pairV[v];
            if (u2 == -1 || (dist[u2] == dist[u] + 1 && dfs(u2))) {
                pairU[u] = v;
                pairV[v] = u;
                return true;
            }
        }
        dist[u] = -1;
        return false;
    };
    while (bfs()) {
        for (int u = 0; u < L; ++u) {
            if (pairU[u] == -1) dfs(u);
        }
    }

    vector<char> visU(L, 0), visV(R, 0);
    queue<pair<int, int>> q;
    for (int u = 0; u < L; ++u) {
        if (pairU[u] == -1) {
            visU[u] = 1;
            q.push({0, u});
        }
    }
    while (!q.empty()) {
        auto [side, x] = q.front();
        q.pop();
        if (side == 0) {
            int u = x;
            for (int v : adjL[u]) {
                if (pairU[u] == v) continue;
                if (!visV[v]) {
                    visV[v] = 1;
                    q.push({1, v});
                }
            }
        } else {
            int v = x;
            int u = pairV[v];
            if (u != -1 && !visU[u]) {
                visU[u] = 1;
                q.push({0, u});
            }
        }
    }

    vector<int> result;
    for (int u = 0; u < L; ++u) {
        if (visU[u]) result.push_back(globalOfLocal[leftVerts[u]]);
    }
    for (int v = 0; v < R; ++v) {
        if (!visV[v]) result.push_back(globalOfLocal[rightVerts[v]]);
    }
    sort(result.begin(), result.end());
    return {result};
}

struct ExactMISComponent {
    int n = 0;
    int W = 0;
    vector<vector<uint64_t>> adj;
    vector<int> orderDesc;
    vector<int> orderAsc;

    struct ColorInfo {
        int colors = 0;
        int branchVertex = -1;
    };

    vector<int> greedyIndependent(const vector<int>& order) const {
        vector<uint64_t> forbidden(W, 0);
        vector<int> res;
        for (int v : order) {
            if (testBit(forbidden, v)) continue;
            res.push_back(v);
            for (int i = 0; i < W; ++i) forbidden[i] |= adj[v][i];
        }
        return res;
    }

    ColorInfo greedyColor(const vector<uint64_t>& cand) const {
        vector<uint64_t> remain = cand;
        int remainCount = countBits(remain);
        int colors = 0;
        int branchVertex = -1;

        while (remainCount > 0) {
            ++colors;
            vector<uint64_t> blocked(W, 0);
            int firstInColor = -1;
            for (int v : orderDesc) {
                if (!testBit(remain, v)) continue;
                if (testBit(blocked, v)) continue;
                if (firstInColor == -1) firstInColor = v;
                clearBit(remain, v);
                --remainCount;
                for (int i = 0; i < W; ++i) blocked[i] |= adj[v][i];
            }
            branchVertex = firstInColor;
        }
        return {colors, branchVertex};
    }

    void dfs(vector<uint64_t> cand, vector<int>& cur, vector<int>& best) const {
        int candCount = countBits(cand);
        if (cur.size() + (size_t)candCount <= best.size()) return;
        if (candCount == 0) {
            if (cur.size() > best.size()) best = cur;
            return;
        }

        ColorInfo info = greedyColor(cand);
        if (cur.size() + (size_t)info.colors <= best.size()) return;

        int v = info.branchVertex;
        if (v < 0) return;

        vector<uint64_t> take = cand;
        for (int i = 0; i < W; ++i) take[i] &= ~adj[v][i];
        cur.push_back(v);
        dfs(take, cur, best);
        cur.pop_back();

        clearBit(cand, v);
        dfs(cand, cur, best);
    }

    vector<int> solve() const {
        if (n == 0) return {};

        vector<int> best = greedyIndependent(orderAsc);
        vector<int> alt = greedyIndependent(orderDesc);
        if (alt.size() > best.size()) best = alt;

        vector<uint64_t> cand(W, ~0ULL);
        if (W > 0 && (n & 63)) {
            cand[W - 1] = (1ULL << (n & 63)) - 1ULL;
        }

        vector<int> cur;
        dfs(cand, cur, best);
        sort(best.begin(), best.end());
        return best;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<vector<int>> g(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) continue;
        g[u].push_back(v);
        g[v].push_back(u);
    }

    vector<char> seen(n, 0);
    vector<int> answer;

    for (int s = 0; s < n; ++s) {
        if (seen[s]) continue;
        queue<int> q;
        vector<int> comp;
        seen[s] = 1;
        q.push(s);
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            comp.push_back(u);
            for (int v : g[u]) {
                if (!seen[v]) {
                    seen[v] = 1;
                    q.push(v);
                }
            }
        }

        int cn = (int)comp.size();
        vector<int> localOfGlobal(n, -1);
        for (int i = 0; i < cn; ++i) localOfGlobal[comp[i]] = i;

        vector<vector<int>> cg(cn);
        for (int i = 0; i < cn; ++i) {
            int u = comp[i];
            for (int v : g[u]) {
                int j = localOfGlobal[v];
                if (j != -1) cg[i].push_back(j);
            }
        }

        vector<int> color(cn, -1);
        bool bip = true;
        for (int i = 0; i < cn && bip; ++i) {
            if (color[i] != -1) continue;
            queue<int> qq;
            color[i] = 0;
            qq.push(i);
            while (!qq.empty() && bip) {
                int u = qq.front();
                qq.pop();
                for (int v : cg[u]) {
                    if (color[v] == -1) {
                        color[v] = color[u] ^ 1;
                        qq.push(v);
                    } else if (color[v] == color[u]) {
                        bip = false;
                        break;
                    }
                }
            }
        }

        if (bip) {
            BipartiteResult br = solveBipartiteComponent(cg, comp);
            answer.insert(answer.end(), br.vertices.begin(), br.vertices.end());
            continue;
        }

        ExactMISComponent solver;
        solver.n = cn;
        solver.W = (cn + 63) >> 6;
        solver.adj.assign(cn, vector<uint64_t>(solver.W, 0));
        vector<int> deg(cn, 0);
        for (int i = 0; i < cn; ++i) {
            for (int j : cg[i]) {
                if (i == j) continue;
                setBit(solver.adj[i], j);
                ++deg[i];
            }
            setBit(solver.adj[i], i);
        }

        solver.orderDesc.resize(cn);
        iota(solver.orderDesc.begin(), solver.orderDesc.end(), 0);
        sort(solver.orderDesc.begin(), solver.orderDesc.end(), [&](int a, int b) {
            if (deg[a] != deg[b]) return deg[a] > deg[b];
            return a < b;
        });
        solver.orderAsc = solver.orderDesc;
        reverse(solver.orderAsc.begin(), solver.orderAsc.end());

        vector<int> localAns = solver.solve();
        for (int v : localAns) answer.push_back(comp[v]);
    }

    sort(answer.begin(), answer.end());
    cout << answer.size() << '\n';
    for (size_t i = 0; i < answer.size(); ++i) {
        if (i) cout << ' ';
        cout << answer[i];
    }
    cout << '\n';
    return 0;
}
