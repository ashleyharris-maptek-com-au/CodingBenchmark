// Adaptive Euclidean TSP heuristic / exact solver.
// - Small N: exact brute force
// - Medium N: kd-tree nearest neighbor
// - Large N: Morton-order traversal

#include <algorithm>
#include <cmath>
#include <cstdint>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <limits>
#include <numeric>
#include <utility>
#include <vector>

using std::int32_t;
using std::int64_t;
using std::uint32_t;
using std::uint64_t;

struct Point {
    double x;
    double y;
};

static inline double dist2(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static inline double dist2_xy(const Point& a, double x, double y) {
    double dx = a.x - x;
    double dy = a.y - y;
    return dx * dx + dy * dy;
}

class FastScanner {
public:
    FastScanner() : in(stdin) {}

    bool readInt(int& out) {
        int c = skipSpaces();
        if (c == EOF) return false;
        int sign = 1;
        if (c == '-') {
            sign = -1;
            c = getChar();
        }
        long long val = 0;
        while (c >= '0' && c <= '9') {
            val = val * 10 + (c - '0');
            c = getChar();
        }
        out = static_cast<int>(val * sign);
        return true;
    }

    bool readDouble(double& out) {
        int c = skipSpaces();
        if (c == EOF) return false;
        int sign = 1;
        if (c == '-') {
            sign = -1;
            c = getChar();
        } else if (c == '+') {
            c = getChar();
        }

        long double integerPart = 0.0L;
        while (c >= '0' && c <= '9') {
            integerPart = integerPart * 10.0L + (c - '0');
            c = getChar();
        }

        long double fracPart = 0.0L;
        long double scale = 1.0L;
        if (c == '.') {
            c = getChar();
            while (c >= '0' && c <= '9') {
                fracPart = fracPart * 10.0L + (c - '0');
                scale *= 10.0L;
                c = getChar();
            }
        }

        long double val = integerPart + (scale > 1.0L ? fracPart / scale : 0.0L);
        if (c == 'e' || c == 'E') {
            c = getChar();
            int expSign = 1;
            if (c == '-') {
                expSign = -1;
                c = getChar();
            } else if (c == '+') {
                c = getChar();
            }
            int expVal = 0;
            while (c >= '0' && c <= '9') {
                expVal = expVal * 10 + (c - '0');
                c = getChar();
            }
            int exp = expSign * expVal;
            val *= pow10i(exp);
        }

        out = static_cast<double>(val * sign);
        return true;
    }

private:
    static constexpr size_t BUFSIZE = 1 << 16;
    FILE* in;
    char buf[BUFSIZE];
    size_t pos = 0;
    size_t len = 0;

    int getChar() {
        if (pos >= len) {
            len = std::fread(buf, 1, BUFSIZE, in);
            pos = 0;
            if (len == 0) return EOF;
        }
        return static_cast<unsigned char>(buf[pos++]);
    }

    int skipSpaces() {
        int c;
        do {
            c = getChar();
        } while (c != EOF && c <= ' ');
        return c;
    }

    static long double pow10i(int exp) {
        static const long double posPow[] = {
            1e0L, 1e1L, 1e2L, 1e3L, 1e4L, 1e5L, 1e6L, 1e7L, 1e8L, 1e9L,
            1e10L, 1e11L, 1e12L, 1e13L, 1e14L, 1e15L, 1e16L, 1e17L, 1e18L
        };
        static const long double negPow[] = {
            1e0L, 1e-1L, 1e-2L, 1e-3L, 1e-4L, 1e-5L, 1e-6L, 1e-7L, 1e-8L, 1e-9L,
            1e-10L, 1e-11L, 1e-12L, 1e-13L, 1e-14L, 1e-15L, 1e-16L, 1e-17L, 1e-18L
        };
        if (exp >= 0) {
            long double result = 1.0L;
            while (exp >= 18) {
                result *= 1e18L;
                exp -= 18;
            }
            if (exp > 0) result *= posPow[exp];
            return result;
        } else {
            exp = -exp;
            long double result = 1.0L;
            while (exp >= 18) {
                result *= 1e-18L;
                exp -= 18;
            }
            if (exp > 0) result *= negPow[exp];
            return result;
        }
    }
};

class FastOutput {
public:
    FastOutput() : out(stdout) {}
    ~FastOutput() { flush(); }

    void writeInt(int x) {
        if (idx + 16 >= BUFSIZE) flush();
        if (x == 0) {
            buf[idx++] = '0';
            return;
        }
        if (x < 0) {
            buf[idx++] = '-';
            x = -x;
        }
        char tmp[16];
        int n = 0;
        while (x > 0) {
            tmp[n++] = char('0' + (x % 10));
            x /= 10;
        }
        while (n--) buf[idx++] = tmp[n];
    }

    void writeChar(char c) {
        if (idx >= BUFSIZE) flush();
        buf[idx++] = c;
    }

    void writeSpace() { writeChar(' '); }
    void writeNewline() { writeChar('\n'); }

    void flush() {
        if (idx) {
            std::fwrite(buf, 1, idx, out);
            idx = 0;
        }
    }

private:
    static constexpr size_t BUFSIZE = 1 << 16;
    FILE* out;
    char buf[BUFSIZE];
    size_t idx = 0;
};

static inline uint64_t part1by1(uint32_t n) {
    uint64_t x = n & 0x001fffffU;
    x = (x | (x << 16)) & 0x0000FFFF0000FFFFULL;
    x = (x | (x << 8)) & 0x00FF00FF00FF00FFULL;
    x = (x | (x << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    x = (x | (x << 2)) & 0x3333333333333333ULL;
    x = (x | (x << 1)) & 0x5555555555555555ULL;
    return x;
}

static inline uint64_t mortonCode(double x, double y, double minx, double miny, double sx, double sy) {
    constexpr double SCALE = 2097151.0; // 21 bits
    double nx = sx > 0.0 ? (x - minx) / sx : 0.0;
    double ny = sy > 0.0 ? (y - miny) / sy : 0.0;
    if (nx < 0.0) nx = 0.0;
    if (nx > 1.0) nx = 1.0;
    if (ny < 0.0) ny = 0.0;
    if (ny > 1.0) ny = 1.0;
    uint32_t ix = static_cast<uint32_t>(nx * SCALE + 0.5);
    uint32_t iy = static_cast<uint32_t>(ny * SCALE + 0.5);
    return (part1by1(ix) << 1) | part1by1(iy);
}

static std::vector<int> solveExact(const std::vector<Point>& pts) {
    int n = static_cast<int>(pts.size());
    if (n == 1) return {0};

    std::vector<int> perm;
    perm.reserve(n - 1);
    for (int i = 1; i < n; ++i) perm.push_back(i);

    std::vector<int> bestRoute(n);
    bestRoute[0] = 0;
    double bestLen = std::numeric_limits<double>::infinity();

    do {
        double len = 0.0;
        int prev = 0;
        for (int v : perm) {
            len += std::sqrt(dist2(pts[prev], pts[v]));
            prev = v;
        }
        len += std::sqrt(dist2(pts[prev], pts[0]));
        if (len < bestLen) {
            bestLen = len;
            for (int i = 0; i < n - 1; ++i) bestRoute[i + 1] = perm[i];
        }
    } while (std::next_permutation(perm.begin(), perm.end()));

    return bestRoute;
}

struct KDNode {
    int idx = -1;
    int left = -1;
    int right = -1;
    uint8_t axis = 0;
};

class KDSolver {
public:
    explicit KDSolver(const std::vector<Point>& points) : pts(points) {
        ids.resize(static_cast<int>(pts.size()));
        std::iota(ids.begin(), ids.end(), 0);
        tree.reserve(pts.size());
        root = build(0, static_cast<int>(ids.size()), 0);
    }

    std::vector<int> solve() {
        int n = static_cast<int>(pts.size());
        std::vector<uint8_t> used(n, 0);
        std::vector<int> route;
        route.reserve(n);
        route.push_back(0);
        used[0] = 1;

        int cur = 0;
        for (int step = 1; step < n; ++step) {
            int best = -1;
            double bestD2 = std::numeric_limits<double>::infinity();
            nearest(root, pts[cur].x, pts[cur].y, used, best, bestD2);
            if (best < 0) {
                for (int i = 0; i < n; ++i) {
                    if (!used[i]) {
                        best = i;
                        break;
                    }
                }
            }
            route.push_back(best);
            used[best] = 1;
            cur = best;
        }
        return route;
    }

private:
    const std::vector<Point>& pts;
    std::vector<int> ids;
    std::vector<KDNode> tree;
    int root = -1;

    int build(int l, int r, int depth) {
        if (l >= r) return -1;
        int axis = depth & 1;
        int m = (l + r) >> 1;
        auto comp = [&](int a, int b) {
            if (axis == 0) {
                if (pts[a].x != pts[b].x) return pts[a].x < pts[b].x;
                return pts[a].y < pts[b].y;
            } else {
                if (pts[a].y != pts[b].y) return pts[a].y < pts[b].y;
                return pts[a].x < pts[b].x;
            }
        };
        std::nth_element(ids.begin() + l, ids.begin() + m, ids.begin() + r, comp);
        int node = static_cast<int>(tree.size());
        tree.push_back(KDNode{ids[m], -1, -1, static_cast<uint8_t>(axis)});
        tree[node].left = build(l, m, depth + 1);
        tree[node].right = build(m + 1, r, depth + 1);
        return node;
    }

    void nearest(int node, double tx, double ty, const std::vector<uint8_t>& used, int& bestIdx, double& bestD2) const {
        if (node < 0) return;
        const KDNode& nd = tree[node];
        const Point& p = pts[nd.idx];
        double dx = tx - p.x;
        double dy = ty - p.y;
        double d2 = dx * dx + dy * dy;
        if (!used[nd.idx] && d2 < bestD2) {
            bestD2 = d2;
            bestIdx = nd.idx;
        }

        double diff = nd.axis == 0 ? dx : dy;
        int first = diff < 0.0 ? nd.left : nd.right;
        int second = diff < 0.0 ? nd.right : nd.left;

        if (first >= 0) nearest(first, tx, ty, used, bestIdx, bestD2);
        if (second >= 0 && diff * diff < bestD2) nearest(second, tx, ty, used, bestIdx, bestD2);
    }
};

static std::vector<int> solveMorton(const std::vector<Point>& pts) {
    int n = static_cast<int>(pts.size());
    if (n == 1) return {0};

    double minx = pts[0].x, maxx = pts[0].x;
    double miny = pts[0].y, maxy = pts[0].y;
    for (const auto& p : pts) {
        minx = std::min(minx, p.x);
        maxx = std::max(maxx, p.x);
        miny = std::min(miny, p.y);
        maxy = std::max(maxy, p.y);
    }
    double sx = maxx - minx;
    double sy = maxy - miny;
    if (sx == 0.0) sx = 1.0;
    if (sy == 0.0) sy = 1.0;

    struct Item {
        uint64_t code;
        int idx;
    };
    std::vector<Item> items;
    items.reserve(n);
    for (int i = 0; i < n; ++i) {
        items.push_back(Item{mortonCode(pts[i].x, pts[i].y, minx, miny, sx, sy), i});
    }
    std::sort(items.begin(), items.end(), [](const Item& a, const Item& b) {
        if (a.code != b.code) return a.code < b.code;
        return a.idx < b.idx;
    });

    int pos0 = 0;
    while (pos0 < n && items[pos0].idx != 0) ++pos0;

    std::vector<int> route;
    route.reserve(n);
    route.push_back(0);
    for (int i = 1; i < n; ++i) {
        route.push_back(items[(pos0 + i) % n].idx);
    }
    return route;
}

int main() {
    FastScanner fs;
    FastOutput fo;

    int n;
    if (!fs.readInt(n)) return 0;
    std::vector<Point> pts(n);
    for (int i = 0; i < n; ++i) {
        fs.readDouble(pts[i].x);
        fs.readDouble(pts[i].y);
    }

    std::vector<int> route;
    if (n <= 11) {
        route = solveExact(pts);
    } else if (n <= 5000) {
        KDSolver solver(pts);
        route = solver.solve();
    } else {
        route = solveMorton(pts);
    }

    for (int i = 0; i < n; ++i) {
        if (i) fo.writeSpace();
        fo.writeInt(route[i]);
    }
    fo.writeNewline();
    return 0;
}
