use std::io::{self, Read};

#[derive(Clone, Copy, Default)]
struct Vec2 {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy, Default)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec2 {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }
}

impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Self { x, y, z }
    }
}

fn dot3(a: Vec3, b: Vec3) -> f64 {
    a.x * b.x + a.y * b.y + a.z * b.z
}

fn cross3(a: Vec3, b: Vec3) -> Vec3 {
    Vec3::new(
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x,
    )
}

fn norm3(v: Vec3) -> f64 {
    dot3(v, v).sqrt()
}

fn normalize3(v: Vec3) -> Vec3 {
    let n = norm3(v);
    Vec3::new(v.x / n, v.y / n, v.z / n)
}

fn add3(a: Vec3, b: Vec3) -> Vec3 {
    Vec3::new(a.x + b.x, a.y + b.y, a.z + b.z)
}

fn sub3(a: Vec3, b: Vec3) -> Vec3 {
    Vec3::new(a.x - b.x, a.y - b.y, a.z - b.z)
}

fn mul3(v: Vec3, k: f64) -> Vec3 {
    Vec3::new(v.x * k, v.y * k, v.z * k)
}

fn shadow_project(v: Vec3, sun: Vec3) -> Vec2 {
    let t = v.z / sun.z;
    Vec2::new(v.x - t * sun.x, v.y - t * sun.y)
}

fn quat_from_matrix(m: [[f64; 3]; 3]) -> (f64, f64, f64, f64) {
    let trace = m[0][0] + m[1][1] + m[2][2];
    let (w, x, y, z);

    if trace > 0.0 {
        let s = (trace + 1.0).sqrt() * 2.0;
        w = 0.25 * s;
        x = (m[2][1] - m[1][2]) / s;
        y = (m[0][2] - m[2][0]) / s;
        z = (m[1][0] - m[0][1]) / s;
    } else if m[0][0] > m[1][1] && m[0][0] > m[2][2] {
        let s = (1.0 + m[0][0] - m[1][1] - m[2][2]).sqrt() * 2.0;
        w = (m[2][1] - m[1][2]) / s;
        x = 0.25 * s;
        y = (m[0][1] + m[1][0]) / s;
        z = (m[0][2] + m[2][0]) / s;
    } else if m[1][1] > m[2][2] {
        let s = (1.0 + m[1][1] - m[0][0] - m[2][2]).sqrt() * 2.0;
        w = (m[0][2] - m[2][0]) / s;
        x = (m[0][1] + m[1][0]) / s;
        y = 0.25 * s;
        z = (m[1][2] + m[2][1]) / s;
    } else {
        let s = (1.0 + m[2][2] - m[0][0] - m[1][1]).sqrt() * 2.0;
        w = (m[1][0] - m[0][1]) / s;
        x = (m[0][2] + m[2][0]) / s;
        y = (m[1][2] + m[2][1]) / s;
        z = 0.25 * s;
    }

    let n = (w * w + x * x + y * y + z * z).sqrt();
    (w / n, x / n, y / n, z / n)
}

fn basis_coeffs(p: Vec2, b1: Vec2, b2: Vec2, det: f64) -> (f64, f64) {
    let a = (p.x * b2.y - p.y * b2.x) / det;
    let b = (-p.x * b1.y + p.y * b1.x) / det;
    (a, b)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = it.next().unwrap().parse().unwrap();
    let mut poly = Vec::with_capacity(n);
    for _ in 0..n {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        poly.push(Vec2::new(x, y));
    }

    let sx: f64 = it.next().unwrap().parse().unwrap();
    let sy: f64 = it.next().unwrap().parse().unwrap();
    let sz: f64 = it.next().unwrap().parse().unwrap();

    let sun = normalize3(Vec3::new(sx, sy, sz));

    let temp = if sun.z.abs() < 0.9 {
        Vec3::new(0.0, 0.0, 1.0)
    } else {
        Vec3::new(1.0, 0.0, 0.0)
    };

    let u = normalize3(sub3(temp, mul3(sun, dot3(temp, sun))));
    let v = cross3(sun, u);

    let b1 = shadow_project(u, sun);
    let b2 = shadow_project(v, sun);
    let det = b1.x * b2.y - b1.y * b2.x;
    let quat = quat_from_matrix([
        [u.x, v.x, sun.x],
        [u.y, v.y, sun.y],
        [u.z, v.z, sun.z],
    ]);

    let mut min_p = f64::INFINITY;
    let mut max_p = f64::NEG_INFINITY;
    let mut min_q = f64::INFINITY;
    let mut max_q = f64::NEG_INFINITY;

    for p in &poly {
        let (a, b) = basis_coeffs(*p, b1, b2, det);
        let pp = a + b;
        let qq = a - b;
        min_p = min_p.min(pp);
        max_p = max_p.max(pp);
        min_q = min_q.min(qq);
        max_q = max_q.max(qq);
    }

    let scale = 10.0_f64;
    let one_limit = 2.0 * scale;
    let eps = 1e-9;

    let mut centers: Vec<(f64, f64)> = Vec::new();

    if max_p - min_p <= one_limit + eps && max_q - min_q <= one_limit + eps {
        centers.push(((min_p + max_p) * 0.5, (min_q + max_q) * 0.5));
    } else {
        let cell = 19.8_f64;
        let p_start = ((min_p - eps) / cell).floor() as i64;
        let p_end = ((max_p + eps) / cell).floor() as i64;
        let q_start = ((min_q - eps) / cell).floor() as i64;
        let q_end = ((max_q + eps) / cell).floor() as i64;

        for ip in p_start..=p_end {
            let pc = (ip as f64 + 0.5) * cell;
            for iq in q_start..=q_end {
                let qc = (iq as f64 + 0.5) * cell;
                centers.push((pc, qc));
            }
        }
    }

    let gap = 30.0_f64;
    let mut out = String::new();
    out.push_str(&format!("{}\n", centers.len()));

    for (i, (pc, qc)) in centers.iter().copied().enumerate() {
        let a = (pc + qc) * 0.5;
        let b = (pc - qc) * 0.5;
        let ground = Vec2::new(a * b1.x + b * b2.x, a * b1.y + b * b2.y);

        let h = i as f64 * gap;
        let center = add3(Vec3::new(ground.x, ground.y, 0.0), mul3(sun, h));

        out.push_str(&format!(
            "{:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10}\n",
            center.x, center.y, center.z, quat.0, quat.1, quat.2, quat.3, scale
        ));
    }

    print!("{out}");
}
