; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %worldPos "worldPos"
OpName %outColor "outColor"
OpDecorate %worldPos Location 0
OpDecorate %outColor Location 0

%glsl450 = OpExtInstImport "GLSL.std.450"

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float

%float_0 = OpConstant %float 0.0
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%bottom = OpConstantComposite %v3float %float_0_1 %float_0_4 %float_0_1
%top = OpConstantComposite %v3float %float_1 %float_1 %float_1

%worldPos = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output

%main = OpFunction %void None %func
%entry = OpLabel
%worldPosVal = OpLoad %v3float %worldPos
%y = OpCompositeExtract %float %worldPosVal 1
%scaled = OpFMul %float %y %float_0_5
%biased = OpFAdd %float %scaled %float_0_5
%t = OpExtInst %float %glsl450 FClamp %biased %float_0 %float_1
%oneMinusT = OpFSub %float %float_1 %t
%bottomScaled = OpVectorTimesScalar %v3float %bottom %oneMinusT
%topScaled = OpVectorTimesScalar %v3float %top %t
%rgb = OpFAdd %v3float %bottomScaled %topScaled
%color = OpCompositeConstruct %v4float %rgb %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
