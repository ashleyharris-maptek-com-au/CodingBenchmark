declare const require: any;
declare const process: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number; priority: number };

const grid: string[] = data.grid as string[];
const robotsInput: Robot[] = data.robots as Robot[];
const jobs: Job[] = data.jobs as Job[];

const height = grid.length;
const width = height > 0 ? grid[0].length : 0;
const totalCells = height * width;
const passable = new Uint8Array(totalCells);

for (let y = 0; y < height; y++) {
  const row = grid[y];
  for (let x = 0; x < width; x++) {
    passable[y * width + x] = row.charCodeAt(x) === 35 ? 0 : 1;
  }
}

const robots: Robot[] = robotsInput.slice().sort((a, b) => a.id - b.id);
const robotCount = robots.length;
const jobCount = jobs.length;
const INF = 1e15;

function bfsFrom(startIndex: number): Int32Array {
  const dist = new Int32Array(totalCells);
  dist.fill(-1);
  const queue = new Int32Array(totalCells);
  let head = 0;
  let tail = 0;
  dist[startIndex] = 0;
  queue[tail++] = startIndex;

  while (head < tail) {
    const v = queue[head++];
    const nextDist = dist[v] + 1;
    const y = (v / width) | 0;
    const x = v - y * width;

    if (x > 0) {
      const u = v - 1;
      if (passable[u] && dist[u] === -1) {
        dist[u] = nextDist;
        queue[tail++] = u;
      }
    }
    if (x + 1 < width) {
      const u = v + 1;
      if (passable[u] && dist[u] === -1) {
        dist[u] = nextDist;
        queue[tail++] = u;
      }
    }
    if (y > 0) {
      const u = v - width;
      if (passable[u] && dist[u] === -1) {
        dist[u] = nextDist;
        queue[tail++] = u;
      }
    }
    if (y + 1 < height) {
      const u = v + width;
      if (passable[u] && dist[u] === -1) {
        dist[u] = nextDist;
        queue[tail++] = u;
      }
    }
  }

  return dist;
}

const robotJobDist: number[][] = Array.from({ length: robotCount }, () => new Array<number>(jobCount));
const robotAngleBucket: Uint8Array[] = Array.from({ length: robotCount }, () => new Uint8Array(jobCount));
const jobBestDist = new Array<number>(jobCount).fill(INF);

for (let r = 0; r < robotCount; r++) {
  const robot = robots[r];
  const startIndex = robot.y * width + robot.x;
  const dist = bfsFrom(startIndex);
  const perJob = robotJobDist[r];
  const angleBucket = robotAngleBucket[r];

  for (let j = 0; j < jobCount; j++) {
    const job = jobs[j];
    const idx = job.y * width + job.x;
    const d = dist[idx];
    const value = d >= 0 ? d : INF;
    perJob[j] = value;
    if (value < jobBestDist[j]) jobBestDist[j] = value;

    const dx = job.x - robot.x;
    const dy = job.y - robot.y;
    let bucket = 0;
    if (value < INF) {
      const ang = Math.atan2(dy, dx);
      bucket = Math.floor(((ang + Math.PI) * 64) / (2 * Math.PI));
      if (bucket < 0) bucket = 0;
      else if (bucket > 63) bucket = 63;
    }
    angleBucket[j] = bucket;
  }
}

const jobOrder = Array.from({ length: jobCount }, (_, i) => i);
jobOrder.sort((a, b) => {
  const pa = jobs[a].priority;
  const pb = jobs[b].priority;
  if (pa !== pb) return pb - pa;
  const da = jobBestDist[a];
  const db = jobBestDist[b];
  if (da !== db) return da - db;
  return a - b;
});

const assignmentsByRobot: number[][] = Array.from({ length: robotCount }, () => []);
const assignedCounts = new Int32Array(robotCount);
const balanceWeight = 1;

for (const jobIndex of jobOrder) {
  let bestRobot = 0;
  let bestScore = Infinity;
  let bestRawDist = INF;

  for (let r = 0; r < robotCount; r++) {
    const rawDist = robotJobDist[r][jobIndex];
    const score = rawDist + assignedCounts[r] * balanceWeight;
    if (
      score < bestScore ||
      (score === bestScore && rawDist < bestRawDist) ||
      (score === bestScore && rawDist === bestRawDist && robots[r].id < robots[bestRobot].id)
    ) {
      bestScore = score;
      bestRawDist = rawDist;
      bestRobot = r;
    }
  }

  assignmentsByRobot[bestRobot].push(jobIndex);
  assignedCounts[bestRobot]++;
}

const outputAssignments: number[][] = Array.from({ length: robots.length > 0 ? robots[robots.length - 1].id + 1 : 0 }, () => []);

for (let r = 0; r < robotCount; r++) {
  const robot = robots[r];
  const list = assignmentsByRobot[r];
  const distRow = robotJobDist[r];
  const bucketRow = robotAngleBucket[r];

  list.sort((a, b) => {
    const da = distRow[a];
    const db = distRow[b];
    if (da !== db) return da - db;
    const ba = bucketRow[a];
    const bb = bucketRow[b];
    if (ba !== bb) return ba - bb;
    const pa = jobs[a].priority;
    const pb = jobs[b].priority;
    if (pa !== pb) return pb - pa;
    return jobs[a].id - jobs[b].id;
  });

  outputAssignments[robot.id] = list.map((idx) => jobs[idx].id);
}

process.stdout.write(JSON.stringify({ assignments: outputAssignments }));
