OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

%void = OpTypeVoid
%funct = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_04 = OpConstant %float 0.04
%float_0_005 = OpConstant %float 0.005
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_2
%arr_v3float_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
%ptr_in_arr_v3float = OpTypePointer Input %arr_v3float_2
%ptr_out_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v3float = OpTypePointer Output %v3float
%ptr_in_v4float = OpTypePointer Input %v4float
%ptr_out_v4float = OpTypePointer Output %v4float
%glsl450 = OpExtInstImport "GLSL.std.450"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%gl_in = OpVariable %ptr_in_arr_gl_PerVertex_in Input
%worldPos = OpVariable %ptr_in_arr_v3float Input
%normal = OpVariable %ptr_in_arr_v3float Input
%color = OpVariable %ptr_in_arr_v3float Input
%gl_out = OpVariable %ptr_out_gl_PerVertex_out Output
%outColor = OpVariable %ptr_out_v3float Output

%main = OpFunction %void None %funct
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4float %gl_in %uint_0 %uint_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4float %gl_in %uint_1 %uint_0
%p1 = OpLoad %v4float %p1_ptr

%c0_ptr = OpAccessChain %ptr_in_v3float %color %uint_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3float %color %uint_1
%c1 = OpLoad %v3float %c1_ptr

%p0_x = OpCompositeExtract %float %p0 0
%p0_y = OpCompositeExtract %float %p0 1
%p0_z = OpCompositeExtract %float %p0 2
%p0_w = OpCompositeExtract %float %p0 3
%p1_x = OpCompositeExtract %float %p1 0
%p1_y = OpCompositeExtract %float %p1 1
%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3

%p0_xy = OpCompositeConstruct %v2float %p0_x %p0_y
%p1_xy = OpCompositeConstruct %v2float %p1_x %p1_y
%delta = OpFSub %v2float %p1_xy %p0_xy
%dir = OpExtInst %v2float %glsl450 Normalize %delta
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%perp_x = OpFNegate %float %dir_y
%perp = OpCompositeConstruct %v2float %perp_x %dir_x

%start_offset = OpVectorTimesScalar %v2float %perp %float_0_04
%end_offset = OpVectorTimesScalar %v2float %perp %float_0_005

%p0_plus_xy = OpFAdd %v2float %p0_xy %start_offset
%p0_minus_xy = OpFSub %v2float %p0_xy %start_offset
%p1_plus_xy = OpFAdd %v2float %p1_xy %end_offset
%p1_minus_xy = OpFSub %v2float %p1_xy %end_offset

%p0_plus = OpCompositeConstruct %v4float %p0_plus_xy %p0_z %p0_w
%p0_minus = OpCompositeConstruct %v4float %p0_minus_xy %p0_z %p0_w
%p1_plus = OpCompositeConstruct %v4float %p1_plus_xy %p1_z %p1_w
%p1_minus = OpCompositeConstruct %v4float %p1_minus_xy %p1_z %p1_w

%pos_ptr = OpAccessChain %ptr_out_v4float %gl_out %uint_0
OpStore %pos_ptr %p0_plus
OpStore %outColor %c0
OpEmitVertex

OpStore %pos_ptr %p0_minus
OpStore %outColor %c0
OpEmitVertex

OpStore %pos_ptr %p1_plus
OpStore %outColor %c1
OpEmitVertex

OpStore %pos_ptr %p1_minus
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
