declare const require: any;
declare const process: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Interval = [number, number];

const width: number = data.width | 0;
const height: number = data.height | 0;
const grid: string[] = Array.isArray(data.grid) ? data.grid : [];
const ignitionsInput: number[][] = Array.isArray(data.ignitions) ? data.ignitions : [];
const townsInput: Array<{ id: any; x: number; y: number; value: number }> = Array.isArray(data.towns) ? data.towns : [];
const cutBudget: number = Math.max(0, data.cutBudget | 0);

const n = width * height;
const root = n;

function idx(x: number, y: number): number {
  return y * width + x;
}

function inBounds(x: number, y: number): boolean {
  return x >= 0 && y >= 0 && x < width && y < height;
}

const walkable = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < width; x++) {
    if (row.charCodeAt(x) === 46) {
      walkable[idx(x, y)] = 1;
    }
  }
}

const townValue = new Float64Array(n);
const townCells: number[] = [];
for (const t of townsInput) {
  const x = t.x | 0;
  const y = t.y | 0;
  if (!inBounds(x, y)) continue;
  const u = idx(x, y);
  townValue[u] += Number(t.value) || 0;
}
for (let u = 0; u < n; u++) {
  if (townValue[u] > 0) townCells.push(u);
}

const ignitionFlag = new Uint8Array(n);
const ignitionList: number[] = [];
for (const p of ignitionsInput) {
  if (!Array.isArray(p) || p.length < 2) continue;
  const x = p[0] | 0;
  const y = p[1] | 0;
  if (!inBounds(x, y)) continue;
  const u = idx(x, y);
  if (!walkable[u] || ignitionFlag[u]) continue;
  ignitionFlag[u] = 1;
  ignitionList.push(u);
}

function neighborCount(u: number): number {
  if (u === root) return ignitionList.length;
  let count = 0;
  const y = (u / width) | 0;
  const x = u - y * width;
  if (y > 0 && walkable[u - width]) count++;
  if (y + 1 < height && walkable[u + width]) count++;
  if (x > 0 && walkable[u - 1]) count++;
  if (x + 1 < width && walkable[u + 1]) count++;
  if (ignitionFlag[u]) count++;
  return count;
}

function neighborAt(u: number, i: number): number {
  if (u === root) return ignitionList[i];
  let k = 0;
  const y = (u / width) | 0;
  const x = u - y * width;
  if (y > 0 && walkable[u - width]) {
    if (k === i) return u - width;
    k++;
  }
  if (y + 1 < height && walkable[u + width]) {
    if (k === i) return u + width;
    k++;
  }
  if (x > 0 && walkable[u - 1]) {
    if (k === i) return u - 1;
    k++;
  }
  if (x + 1 < width && walkable[u + 1]) {
    if (k === i) return u + 1;
    k++;
  }
  if (ignitionFlag[u]) {
    if (k === i) return root;
    k++;
  }
  return -1;
}

// Multi-source BFS over the burnable component to build a heuristic tree.
const parentBFS = new Int32Array(n);
parentBFS.fill(-1);
const distBFS = new Int32Array(n);
distBFS.fill(-1);
const bfsOrder: number[] = [];
if (ignitionList.length > 0) {
  const queue = new Int32Array(n);
  let qh = 0;
  let qt = 0;
  for (const s of ignitionList) {
    if (distBFS[s] !== -1) continue;
    distBFS[s] = 0;
    queue[qt++] = s;
  }
  while (qh < qt) {
    const u = queue[qh++];
    bfsOrder.push(u);
    const y = (u / width) | 0;
    const x = u - y * width;
    const nd = distBFS[u] + 1;
    if (y > 0) {
      const v = u - width;
      if (walkable[v] && distBFS[v] === -1) {
        distBFS[v] = nd;
        parentBFS[v] = u;
        queue[qt++] = v;
      }
    }
    if (y + 1 < height) {
      const v = u + width;
      if (walkable[v] && distBFS[v] === -1) {
        distBFS[v] = nd;
        parentBFS[v] = u;
        queue[qt++] = v;
      }
    }
    if (x > 0) {
      const v = u - 1;
      if (walkable[v] && distBFS[v] === -1) {
        distBFS[v] = nd;
        parentBFS[v] = u;
        queue[qt++] = v;
      }
    }
    if (x + 1 < width) {
      const v = u + 1;
      if (walkable[v] && distBFS[v] === -1) {
        distBFS[v] = nd;
        parentBFS[v] = u;
        queue[qt++] = v;
      }
    }
  }
}

if (ignitionList.length === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

// Iterative DFS from a virtual super-source to compute low-links and subtree sums.
const totalNodes = n + 1;
const disc = new Int32Array(totalNodes);
const low = new Int32Array(totalNodes);
const dfsParent = new Int32Array(totalNodes);
dfsParent.fill(-1);
const tin = new Int32Array(totalNodes);
const tout = new Int32Array(totalNodes);
const treeHead = new Int32Array(totalNodes);
treeHead.fill(-1);
const treeTo = new Int32Array(n);
const treeNext = new Int32Array(n);
let treeEdgeCount = 0;

function addTreeChild(p: number, c: number): void {
  treeTo[treeEdgeCount] = c;
  treeNext[treeEdgeCount] = treeHead[p];
  treeHead[p] = treeEdgeCount;
  treeEdgeCount++;
}

const dfsStackNode = new Int32Array(totalNodes);
const dfsStackPos = new Int32Array(totalNodes);
const finishOrder: number[] = [];
let time = 0;

disc[root] = low[root] = ++time;
tin[root] = time;
dfsStackNode[0] = root;
dfsStackPos[0] = 0;
let stackTop = 0;

while (stackTop >= 0) {
  const u = dfsStackNode[stackTop];
  let pos = dfsStackPos[stackTop];
  const deg = neighborCount(u);
  if (pos < deg) {
    const v = neighborAt(u, pos);
    dfsStackPos[stackTop] = pos + 1;
    if (v < 0) continue;
    if (disc[v] === 0) {
      dfsParent[v] = u;
      addTreeChild(u, v);
      disc[v] = low[v] = ++time;
      tin[v] = time;
      stackTop++;
      dfsStackNode[stackTop] = v;
      dfsStackPos[stackTop] = 0;
    } else if (v !== dfsParent[u]) {
      if (disc[v] < low[u]) low[u] = disc[v];
    }
  } else {
    tout[u] = time;
    finishOrder.push(u);
    stackTop--;
    const p = dfsParent[u];
    if (p !== -1) {
      if (low[u] < low[p]) low[p] = low[u];
    }
  }
}

// Subtree town value in the DFS tree.
const subtreeTown = new Float64Array(totalNodes);
for (let u = 0; u < totalNodes; u++) subtreeTown[u] = 0;
for (let u = 0; u < n; u++) subtreeTown[u] = townValue[u];
for (let i = 0; i < finishOrder.length; i++) {
  const u = finishOrder[i];
  const p = dfsParent[u];
  if (p !== -1) subtreeTown[p] += subtreeTown[u];
}

// Segment tree over DFS order, storing remaining town value at each visited node.
const visitedCount = time;
const base = new Float64Array(visitedCount + 1);
for (let u = 0; u < totalNodes; u++) {
  if (disc[u] !== 0) base[tin[u]] = u === root ? 0 : townValue[u];
}

class SegTree {
  n: number;
  sum: Float64Array;
  lazy: Uint8Array;
  constructor(arr: Float64Array) {
    this.n = arr.length - 1;
    this.sum = new Float64Array(this.n * 4 + 5);
    this.lazy = new Uint8Array(this.n * 4 + 5);
    this.build(1, 1, this.n, arr);
  }
  build(node: number, l: number, r: number, arr: Float64Array): void {
    if (l === r) {
      this.sum[node] = arr[l];
      return;
    }
    const m = (l + r) >> 1;
    this.build(node << 1, l, m, arr);
    this.build((node << 1) | 1, m + 1, r, arr);
    this.sum[node] = this.sum[node << 1] + this.sum[(node << 1) | 1];
  }
  rangeZero(ql: number, qr: number): void {
    if (ql > qr) return;
    this._rangeZero(1, 1, this.n, ql, qr);
  }
  _rangeZero(node: number, l: number, r: number, ql: number, qr: number): void {
    if (ql <= l && r <= qr) {
      this.sum[node] = 0;
      this.lazy[node] = 1;
      return;
    }
    if (this.lazy[node]) return;
    const m = (l + r) >> 1;
    if (ql <= m) this._rangeZero(node << 1, l, m, ql, qr);
    if (qr > m) this._rangeZero((node << 1) | 1, m + 1, r, ql, qr);
    this.sum[node] = this.sum[node << 1] + this.sum[(node << 1) | 1];
  }
  query(ql: number, qr: number): number {
    if (ql > qr) return 0;
    return this._query(1, 1, this.n, ql, qr);
  }
  _query(node: number, l: number, r: number, ql: number, qr: number): number {
    if (ql <= l && r <= qr) return this.sum[node];
    if (this.lazy[node]) return 0;
    const m = (l + r) >> 1;
    let ans = 0;
    if (ql <= m) ans += this._query(node << 1, l, m, ql, qr);
    if (qr > m) ans += this._query((node << 1) | 1, m + 1, r, ql, qr);
    return ans;
  }
  pointQuery(pos: number): number {
    return this.query(pos, pos);
  }
}

const seg = new SegTree(base);

const candNode: number[] = [];
const candGain: number[] = [];
const candStart: number[] = [];
const candCount: number[] = [];
const candL: number[] = [];
const candR: number[] = [];

for (let v = 0; v < n; v++) {
  if (disc[v] === 0) continue;
  if (v === root) continue;
  let start = candL.length;
  let count = 0;
  let gain = 0;
  if (townValue[v] > 0) {
    candL.push(tin[v]);
    candR.push(tin[v]);
    gain += seg.query(tin[v], tin[v]);
    count++;
  }
  for (let e = treeHead[v]; e !== -1; e = treeNext[e]) {
    const c = treeTo[e];
    if (low[c] >= disc[v] && subtreeTown[c] > 0) {
      candL.push(tin[c]);
      candR.push(tout[c]);
      gain += seg.query(tin[c], tout[c]);
      count++;
    }
  }
  if (count > 0 && gain > 0) {
    candNode.push(v);
    candGain.push(gain);
    candStart.push(start);
    candCount.push(count);
  } else {
    candL.length = start;
    candR.length = start;
  }
}

class MaxHeap {
  a: number[] = [];
  push(id: number): void {
    const arr = this.a;
    arr.push(id);
    let i = arr.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (candGain[arr[p]] >= candGain[arr[i]]) break;
      const tmp = arr[p];
      arr[p] = arr[i];
      arr[i] = tmp;
      i = p;
    }
  }
  pop(): number | undefined {
    const arr = this.a;
    if (arr.length === 0) return undefined;
    const top = arr[0];
    const last = arr.pop()!;
    if (arr.length > 0) {
      arr[0] = last;
      let i = 0;
      while (true) {
        let best = i;
        const l = i * 2 + 1;
        const r = l + 1;
        if (l < arr.length && candGain[arr[l]] > candGain[arr[best]]) best = l;
        if (r < arr.length && candGain[arr[r]] > candGain[arr[best]]) best = r;
        if (best === i) break;
        const tmp = arr[i];
        arr[i] = arr[best];
        arr[best] = tmp;
        i = best;
      }
    }
    return top;
  }
  get size(): number {
    return this.a.length;
  }
}

const heap = new MaxHeap();
for (let i = 0; i < candNode.length; i++) heap.push(i);

const selectedNode = new Uint8Array(totalNodes);
const firebreaks: number[][] = [];

while (firebreaks.length < cutBudget && heap.size > 0) {
  const id = heap.pop()!;
  const v = candNode[id];
  if (selectedNode[v]) continue;
  let current = 0;
  const start = candStart[id];
  const count = candCount[id];
  for (let k = 0; k < count; k++) {
    current += seg.query(candL[start + k], candR[start + k]);
  }
  if (current + 1e-9 < candGain[id] || current > candGain[id] + 1e-9) {
    candGain[id] = current;
    if (current > 0) heap.push(id);
    continue;
  }
  if (current <= 0) break;
  selectedNode[v] = 1;
  firebreaks.push([v % width, (v / width) | 0]);
  for (let k = 0; k < count; k++) {
    seg.rangeZero(candL[start + k], candR[start + k]);
  }
}

if (firebreaks.length >= cutBudget) {
  process.stdout.write(JSON.stringify({ firebreaks }));
  process.exit(0);
}

// Heuristic fallback: build a scoring field from remaining towns on a BFS tree.
const remainingValue = new Float64Array(n);
let remainingTotal = 0;
for (const u of townCells) {
  if (distBFS[u] < 0) continue;
  const val = seg.pointQuery(tin[u]);
  if (val > 0) {
    remainingValue[u] = val;
    remainingTotal += val;
  }
}

if (remainingTotal > 0) {
  const bfsSub = new Float64Array(n);
  for (let i = 0; i < bfsOrder.length; i++) {
    const u = bfsOrder[i];
    bfsSub[u] = remainingValue[u];
  }
  for (let i = bfsOrder.length - 1; i >= 0; i--) {
    const u = bfsOrder[i];
    const p = parentBFS[u];
    if (p !== -1) bfsSub[p] += bfsSub[u];
  }

  const pathScore = new Float64Array(n);
  const remainingCells: number[] = [];
  for (const u of townCells) {
    const val = remainingValue[u];
    if (val > 0) remainingCells.push(u);
  }
  remainingCells.sort((a, b) => remainingValue[b] - remainingValue[a]);
  let takeCount = 0;
  let takenValue = 0;
  const takeLimit = Math.min(400, remainingCells.length);
  const takeTarget = remainingTotal * 0.7;
  for (let i = 0; i < remainingCells.length && takeCount < takeLimit; i++) {
    const u = remainingCells[i];
    const val = remainingValue[u];
    takenValue += val;
    takeCount++;
    let cur = u;
    while (cur !== -1) {
      pathScore[cur] += val;
      cur = parentBFS[cur];
    }
    if (takenValue >= takeTarget) break;
  }

  const degree = new Int32Array(n);
  const waterAdj = new Int32Array(n);
  for (let y = 0; y < height; y++) {
    for (let x = 0; x < width; x++) {
      const u = idx(x, y);
      if (!walkable[u]) continue;
      let deg = 0;
      let wat = 0;
      if (y > 0) {
        if (walkable[u - width]) deg++; else wat++;
      } else wat++;
      if (y + 1 < height) {
        if (walkable[u + width]) deg++; else wat++;
      } else wat++;
      if (x > 0) {
        if (walkable[u - 1]) deg++; else wat++;
      } else wat++;
      if (x + 1 < width) {
        if (walkable[u + 1]) deg++; else wat++;
      } else wat++;
      degree[u] = deg;
      waterAdj[u] = wat;
    }
  }

  const fallback: Array<[number, number]> = [];
  for (let u = 0; u < n; u++) {
    if (!walkable[u] || selectedNode[u] || distBFS[u] < 0) continue;
    const baseScore = pathScore[u] + bfsSub[u];
    if (baseScore <= 0) continue;
    const score = baseScore * (1 + 0.25 * waterAdj[u]) * (1 + 0.2 / (1 + distBFS[u])) / (1 + degree[u]);
    fallback.push([u, score]);
  }
  fallback.sort((a, b) => b[1] - a[1]);

  for (let i = 0; i < fallback.length && firebreaks.length < cutBudget; i++) {
    const u = fallback[i][0];
    if (selectedNode[u]) continue;
    if (fallback[i][1] <= 0) break;
    selectedNode[u] = 1;
    firebreaks.push([u % width, (u / width) | 0]);
  }
}

process.stdout.write(JSON.stringify({ firebreaks }));
