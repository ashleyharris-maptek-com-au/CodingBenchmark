; SPIR-V
; Version: 1.0
; Generator: Codex
OpCapability Shader
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v4float = OpTypePointer Output %v4float

%float_0 = OpConstant %float 0.0
%float_0_1 = OpConstant %float 0.1
%float_0_6 = OpConstant %float 0.6
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1.0
%float_64 = OpConstant %float 64.0
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4

%ubo = OpVariable %ptr_Uniform_UBO Uniform
%worldPos = OpVariable %ptr_Input_v3float Input
%normal = OpVariable %ptr_Input_v3float Input
%outColor = OpVariable %ptr_Output_v4float Output

%main = OpFunction %void None %func
%label = OpLabel

%worldPosVal = OpLoad %v3float %worldPos
%normalVal = OpLoad %v3float %normal
%normalN = OpExtInst %v3float %glsl450 Normalize %normalVal

%lightPosPtr = OpAccessChain %ptr_Uniform_v4float %ubo %uint_3
%lightPos4 = OpLoad %v4float %lightPosPtr
%lightPos3 = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2

%cameraPosPtr = OpAccessChain %ptr_Uniform_v4float %ubo %uint_4
%cameraPos4 = OpLoad %v4float %cameraPosPtr
%cameraPos3 = OpVectorShuffle %v3float %cameraPos4 %cameraPos4 0 1 2

%Lraw = OpFSub %v3float %lightPos3 %worldPosVal
%L = OpExtInst %v3float %glsl450 Normalize %Lraw
%Vraw = OpFSub %v3float %cameraPos3 %worldPosVal
%V = OpExtInst %v3float %glsl450 Normalize %Vraw

%NdotL = OpDot %float %normalN %L
%diffuse = OpExtInst %float %glsl450 FMax %NdotL %float_0

%LplusV = OpFAdd %v3float %L %V
%H = OpExtInst %v3float %glsl450 Normalize %LplusV
%NdotH = OpDot %float %normalN %H
%specBase = OpExtInst %float %glsl450 FMax %NdotH %float_0
%specular = OpExtInst %float %glsl450 Pow %specBase %float_64

%diffuseTerm = OpFMul %float %float_0_6 %diffuse
%specularTerm = OpFMul %float %float_0_8 %specular
%lightTerm = OpFAdd %float %float_0_1 %diffuseTerm
%intensity = OpFAdd %float %lightTerm %specularTerm

%outColorVal = OpCompositeConstruct %v4float %intensity %intensity %intensity %float_1
OpStore %outColor %outColorVal
OpReturn
OpFunctionEnd
