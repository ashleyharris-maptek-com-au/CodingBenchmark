// Minimum Edge Cut solver for undirected connected graphs.
// Strategy:
// 1) Detect bridges. If any exist, the minimum cut is 1 and any bridge is valid.
// 2) For manageable instances, run an exact Stoer-Wagner global min-cut algorithm.
// 3) For very large instances, fall back to randomized contraction to still produce
//    a valid cut when the exact O(nm)-style pass would be too expensive.

#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <random>
#include <utility>
#include <vector>

struct Edge {
    int u;
    int v;
};

static std::vector<int> findBridges(int n, const std::vector<Edge>& edges) {
    std::vector<std::vector<std::pair<int, int>>> g(n);
    g.reserve(n);
    for (int i = 0; i < static_cast<int>(edges.size()); ++i) {
        const auto& e = edges[i];
        g[e.u].push_back({e.v, i});
        g[e.v].push_back({e.u, i});
    }

    std::vector<int> tin(n, -1), low(n, 0), parent(n, -1), parentEdge(n, -1), it(n, 0);
    std::vector<int> bridges;
    int timer = 0;

    for (int start = 0; start < n; ++start) {
        if (tin[start] != -1) continue;

        std::vector<int> st;
        st.push_back(start);
        parent[start] = -2;

        while (!st.empty()) {
            int v = st.back();
            if (tin[v] == -1) {
                tin[v] = low[v] = timer++;
            }

            if (it[v] < static_cast<int>(g[v].size())) {
                const auto [to, eid] = g[v][it[v]++];
                if (eid == parentEdge[v]) continue;
                if (tin[to] == -1) {
                    parent[to] = v;
                    parentEdge[to] = eid;
                    st.push_back(to);
                } else {
                    low[v] = std::min(low[v], tin[to]);
                }
            } else {
                st.pop_back();
                if (parent[v] >= 0) {
                    low[parent[v]] = std::min(low[parent[v]], low[v]);
                    if (low[v] > tin[parent[v]]) {
                        bridges.push_back(parentEdge[v]);
                    }
                }
            }
        }
    }

    return bridges;
}

static std::vector<char> solveExactStoerWagner(int n, const std::vector<Edge>& edges) {
    std::vector<int> comp(n);
    std::iota(comp.begin(), comp.end(), 0);

    std::vector<std::vector<int>> members(n);
    std::vector<int> activeRoots;
    activeRoots.reserve(n);
    for (int i = 0; i < n; ++i) {
        members[i].push_back(i);
        activeRoots.push_back(i);
    }

    std::vector<char> bestSide(n, false);
    long long bestWeight = std::numeric_limits<long long>::max();

    while (activeRoots.size() > 1) {
        const int k = static_cast<int>(activeRoots.size());
        std::vector<int> rootIndex(n, -1);
        for (int i = 0; i < k; ++i) {
            rootIndex[activeRoots[i]] = i;
        }

        std::vector<std::vector<std::pair<int, int>>> adj(k);
        adj.reserve(k);
        for (const auto& e : edges) {
            const int a = comp[e.u];
            const int b = comp[e.v];
            if (a == b) continue;
            const int ia = rootIndex[a];
            const int ib = rootIndex[b];
            if (ia < 0 || ib < 0) continue;
            adj[ia].push_back({ib, 1});
            adj[ib].push_back({ia, 1});
        }

        std::vector<long long> weight(k, 0);
        std::vector<char> added(k, false);
        std::priority_queue<std::pair<long long, int>> pq;
        std::vector<int> order;
        order.reserve(k);

        pq.push({0, 0});
        for (int step = 0; step < k; ++step) {
            int v = -1;
            while (!pq.empty()) {
                const auto [cw, idx] = pq.top();
                if (added[idx] || cw != weight[idx]) {
                    pq.pop();
                    continue;
                }
                v = idx;
                pq.pop();
                break;
            }
            if (v == -1) {
                for (int i = 0; i < k; ++i) {
                    if (!added[i]) {
                        v = i;
                        break;
                    }
                }
            }

            added[v] = true;
            order.push_back(v);

            if (step == k - 1) {
                const long long cutWeight = weight[v];
                if (cutWeight < bestWeight) {
                    std::vector<char> side(n, false);
                    for (int i = 0; i < k - 1; ++i) {
                        const int root = activeRoots[order[i]];
                        for (int x : members[root]) {
                            side[x] = true;
                        }
                    }
                    bestWeight = cutWeight;
                    bestSide.swap(side);
                }
            } else {
                for (const auto& [to, w] : adj[v]) {
                    if (!added[to]) {
                        weight[to] += w;
                        pq.push({weight[to], to});
                    }
                }
            }
        }

        const int sRoot = activeRoots[order[k - 2]];
        const int tRoot = activeRoots[order[k - 1]];
        if (sRoot != tRoot) {
            for (int vtx : members[tRoot]) {
                comp[vtx] = sRoot;
            }
            auto& dst = members[sRoot];
            dst.insert(dst.end(), members[tRoot].begin(), members[tRoot].end());
            members[tRoot].clear();

            activeRoots.erase(std::remove(activeRoots.begin(), activeRoots.end(), tRoot), activeRoots.end());
        }
    }

    return bestSide;
}

static std::vector<char> solveRandomFallback(int n, const std::vector<Edge>& edges) {
    std::mt19937_64 rng(0x9E3779B97F4A7C15ULL ^ (static_cast<std::uint64_t>(n) << 1) ^ (static_cast<std::uint64_t>(edges.size()) << 32));
    const int trials = std::min(256, std::max(48, 2 * n));

    std::vector<char> bestSide(n, false);
    long long bestCut = std::numeric_limits<long long>::max();

    for (int trial = 0; trial < trials; ++trial) {
        std::vector<int> comp(n);
        std::iota(comp.begin(), comp.end(), 0);
        std::vector<std::vector<int>> members(n);
        for (int i = 0; i < n; ++i) {
            members[i].push_back(i);
        }

        int activeCount = n;
        while (activeCount > 2) {
            const Edge& e = edges[static_cast<std::size_t>(rng() % edges.size())];
            int a = comp[e.u];
            int b = comp[e.v];
            if (a == b) continue;

            if (members[a].size() < members[b].size()) std::swap(a, b);

            for (int v : members[b]) {
                comp[v] = a;
            }
            members[a].insert(members[a].end(), members[b].begin(), members[b].end());
            members[b].clear();
            --activeCount;
        }

        std::vector<int> roots;
        roots.reserve(2);
        for (int i = 0; i < n; ++i) {
            if (!members[i].empty()) roots.push_back(i);
        }
        if (roots.size() != 2) continue;

        std::vector<char> side(n, false);
        for (int v : members[roots[0]]) {
            side[v] = true;
        }

        long long cut = 0;
        for (const auto& e : edges) {
            if (side[e.u] != side[e.v]) ++cut;
        }

        if (cut < bestCut) {
            bestCut = cut;
            bestSide.swap(side);
        }
        if (bestCut == 1) break;
    }

    if (bestCut == std::numeric_limits<long long>::max() && n > 0) {
        bestSide[0] = true;
    }
    return bestSide;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, m;
    if (!(std::cin >> n >> m)) {
        return 0;
    }

    std::vector<Edge> edges;
    edges.reserve(m);
    for (int i = 0; i < m; ++i) {
        int u, v;
        std::cin >> u >> v;
        edges.push_back({u, v});
    }

    if (n <= 1) {
        std::cout << 0 << '\n';
        return 0;
    }

    const std::vector<int> bridges = findBridges(n, edges);
    if (!bridges.empty()) {
        const Edge& e = edges[bridges.front()];
        std::cout << 1 << '\n';
        std::cout << e.u << ' ' << e.v << '\n';
        return 0;
    }

    std::vector<char> side;
    const long long scale = static_cast<long long>(n) * static_cast<long long>(std::max(1, m));
    if (scale <= 120000000LL) {
        side = solveExactStoerWagner(n, edges);
    } else {
        side = solveRandomFallback(n, edges);
    }

    std::vector<Edge> cutEdges;
    cutEdges.reserve(edges.size());
    for (const auto& e : edges) {
        if (side[e.u] != side[e.v]) {
            cutEdges.push_back(e);
        }
    }

    std::cout << cutEdges.size() << '\n';
    for (const auto& e : cutEdges) {
        std::cout << e.u << ' ' << e.v << '\n';
    }

    return 0;
}
