using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private struct Vec3
    {
        public double X;
        public double Y;
        public double Z;

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);

        public double Length()
        {
            return Math.Sqrt(X * X + Y * Y + Z * Z);
        }

        public static double Dot(in Vec3 a, in Vec3 b)
        {
            return a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        }

        public static Vec3 Cross(in Vec3 a, in Vec3 b)
        {
            return new Vec3(
                a.Y * b.Z - a.Z * b.Y,
                a.Z * b.X - a.X * b.Z,
                a.X * b.Y - a.Y * b.X);
        }
    }

    private struct Orbit
    {
        public Vec3 R;
        public Vec3 V;
        public Vec3 HUnit;
        public Vec3 EVec;
        public double A;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public bool TryReadDouble(out double value)
        {
            value = 0;
            byte c;
            do
            {
                c = Read();
                if (c == 0) return false;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            double integer = 0;
            while (c > 32 && c != '.' && c != 0)
            {
                integer = integer * 10 + (c - '0');
                c = Read();
            }

            double frac = 0;
            double div = 1;
            if (c == '.')
            {
                while (true)
                {
                    c = Read();
                    if (c <= 32 || c == 0) break;
                    frac = frac * 10 + (c - '0');
                    div *= 10;
                }
            }

            value = sign * (integer + frac / div);
            return true;
        }

        public int NextInt()
        {
            double v;
            if (!TryReadDouble(out v)) return 0;
            return (int)v;
        }
    }

    private static int N;
    private static double Mu;
    private static Orbit StartOrbit;
    private static Orbit[] Stations = Array.Empty<Orbit>();
    private static double[] StartCosts = Array.Empty<double>();
    private static double[,] Costs = new double[0, 0];
    private static double[] MinOut = Array.Empty<double>();
    private static int[] BestExact = Array.Empty<int>();
    private static double BestExactCost;

    public static void Main()
    {
        var fs = new FastScanner();
        N = fs.NextInt();
        if (!fs.TryReadDouble(out Mu))
        {
            return;
        }

        StartOrbit = ReadOrbit(fs, Mu);
        Stations = new Orbit[N];
        for (int i = 0; i < N; i++)
        {
            Stations[i] = ReadOrbit(fs, Mu);
        }

        StartCosts = new double[N];
        Costs = new double[N, N];
        MinOut = new double[N];
        for (int i = 0; i < N; i++)
        {
            StartCosts[i] = Cost(StartOrbit, Stations[i]);
        }

        for (int i = 0; i < N; i++)
        {
            double best = double.PositiveInfinity;
            for (int j = 0; j < N; j++)
            {
                if (i == j) continue;
                double c = Cost(Stations[i], Stations[j]);
                Costs[i, j] = c;
                if (c < best) best = c;
            }
            MinOut[i] = best < double.PositiveInfinity ? best : 0.0;
        }

        int[] answer = N <= 9 ? SolveExact() : SolveHeuristic();
        var sb = new StringBuilder();
        for (int i = 0; i < answer.Length; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(answer[i]);
        }
        Console.WriteLine(sb.ToString());
    }

    private static Orbit ReadOrbit(FastScanner fs, double mu)
    {
        double x = NextDouble(fs);
        double y = NextDouble(fs);
        double z = NextDouble(fs);
        double vx = NextDouble(fs);
        double vy = NextDouble(fs);
        double vz = NextDouble(fs);
        return BuildOrbit(new Vec3(x, y, z), new Vec3(vx, vy, vz), mu);
    }

    private static double NextDouble(FastScanner fs)
    {
        if (!fs.TryReadDouble(out double value))
        {
            return 0.0;
        }
        return value;
    }

    private static Orbit BuildOrbit(Vec3 r, Vec3 v, double mu)
    {
        double rMag = r.Length();
        double v2 = Vec3.Dot(v, v);
        Vec3 h = Vec3.Cross(r, v);
        double hMag = h.Length();
        Vec3 hUnit = hMag > 1e-12 ? h / hMag : new Vec3();
        Vec3 eVec = Vec3.Cross(v, h) / mu - r / rMag;

        double invA = 2.0 / rMag - v2 / mu;
        double a;
        if (Math.Abs(invA) < 1e-12)
        {
            a = invA >= 0 ? 1e12 : -1e12;
        }
        else
        {
            a = 1.0 / invA;
        }

        return new Orbit
        {
            R = r,
            V = v,
            HUnit = hUnit,
            EVec = eVec,
            A = a
        };
    }

    private static double Cost(Orbit a, Orbit b)
    {
        double dr = (a.R - b.R).Length() / 7000.0;
        double dv = (a.V - b.V).Length() / 7.5;
        double dh = 1.0 - Clamp(Vec3.Dot(a.HUnit, b.HUnit), -1.0, 1.0);
        double de = (a.EVec - b.EVec).Length();
        double aa = Math.Abs(a.A);
        double ab = Math.Abs(b.A);
        if (aa > 1e11) aa = 1e11;
        if (ab > 1e11) ab = 1e11;
        double da = Math.Abs(Math.Log(1.0 + aa / 7000.0) - Math.Log(1.0 + ab / 7000.0));

        return 0.28 * dr + 0.34 * dv + 1.60 * dh + 0.90 * de + 0.12 * da;
    }

    private static double Clamp(double x, double lo, double hi)
    {
        if (x < lo) return lo;
        if (x > hi) return hi;
        return x;
    }

    private static int[] SolveExact()
    {
        int[] current = new int[N];
        bool[] used = new bool[N];
        BestExact = new int[N];
        BestExactCost = double.PositiveInfinity;

        void Dfs(int depth, int last, double cost)
        {
            if (cost >= BestExactCost) return;
            if (depth == N)
            {
                BestExactCost = cost;
                Array.Copy(current, BestExact, N);
                return;
            }

            for (int i = 0; i < N; i++)
            {
                if (used[i]) continue;
                used[i] = true;
                current[depth] = i;
                double nextCost = cost;
                if (depth == 0)
                {
                    nextCost += StartCosts[i];
                }
                else
                {
                    nextCost += Costs[last, i];
                }
                Dfs(depth + 1, i, nextCost);
                used[i] = false;
            }
        }

        Dfs(0, -1, 0.0);
        return BestExact;
    }

    private static int[] SolveHeuristic()
    {
        int[] startOrder = new int[N];
        for (int i = 0; i < N; i++) startOrder[i] = i;
        Array.Sort(startOrder, (a, b) => StartCosts[a].CompareTo(StartCosts[b]));

        var seeds = new List<int>();
        int nearCount = Math.Min(8, N);
        for (int i = 0; i < nearCount; i++) seeds.Add(startOrder[i]);

        int[] probePositions = { N / 2, Math.Max(0, N - 1), Math.Max(0, N - 2), Math.Max(0, N / 3), Math.Max(0, (2 * N) / 3) };
        foreach (int pos in probePositions)
        {
            int idx = startOrder[pos];
            if (!seeds.Contains(idx)) seeds.Add(idx);
            if (seeds.Count >= Math.Min(N, 12)) break;
        }

        if (seeds.Count == 0) seeds.Add(0);

        int[] best = null;
        double bestCost = double.PositiveInfinity;
        for (int s = 0; s < seeds.Count; s++)
        {
            int seed = seeds[s];
            int[] route = BuildByInsertion(seed, 9001 + seed * 7919);
            ImproveTwoOpt(route);
            double cost = RouteCost(route);
            if (cost < bestCost)
            {
                bestCost = cost;
                best = route;
            }
        }

        return best ?? Array.Empty<int>();
    }

    private static int[] BuildByInsertion(int seed, int rngSeed)
    {
        var rng = new Random(rngSeed);
        bool[] used = new bool[N];
        int[] route = new int[N];
        int len = 0;

        route[len++] = seed;
        used[seed] = true;

        while (len < N)
        {
            double bestScore = double.PositiveInfinity;
            double bestDelta = double.PositiveInfinity;
            int bestNode = -1;
            int bestPos = -1;

            for (int u = 0; u < N; u++)
            {
                if (used[u]) continue;
                FindBestInsertion(route, len, u, out double delta, out int pos);
                double score = delta + 0.08 * MinOut[u];
                if (score < bestScore - 1e-12 || (Math.Abs(score - bestScore) <= 1e-12 && delta < bestDelta))
                {
                    bestScore = score;
                    bestDelta = delta;
                    bestNode = u;
                    bestPos = pos;
                }
                else if (Math.Abs(score - bestScore) <= 1e-12 && rng.Next(2) == 0)
                {
                    bestScore = score;
                    bestDelta = delta;
                    bestNode = u;
                    bestPos = pos;
                }
            }

            for (int i = len; i > bestPos; i--)
            {
                route[i] = route[i - 1];
            }
            route[bestPos] = bestNode;
            used[bestNode] = true;
            len++;
        }

        return route;
    }

    private static void FindBestInsertion(int[] route, int len, int node, out double bestDelta, out int bestPos)
    {
        bestDelta = double.PositiveInfinity;
        bestPos = 0;

        if (len == 1)
        {
            double before = StartCosts[node] + Costs[node, route[0]] - StartCosts[route[0]];
            double after = Costs[route[0], node];
            if (before <= after)
            {
                bestDelta = before;
                bestPos = 0;
            }
            else
            {
                bestDelta = after;
                bestPos = 1;
            }
            return;
        }

        double first = StartCosts[node] + Costs[node, route[0]] - StartCosts[route[0]];
        bestDelta = first;
        bestPos = 0;

        for (int i = 0; i < len - 1; i++)
        {
            double delta = Costs[route[i], node] + Costs[node, route[i + 1]] - Costs[route[i], route[i + 1]];
            if (delta < bestDelta)
            {
                bestDelta = delta;
                bestPos = i + 1;
            }
        }

        double last = Costs[route[len - 1], node];
        if (last < bestDelta)
        {
            bestDelta = last;
            bestPos = len;
        }
    }

    private static void ImproveTwoOpt(int[] route)
    {
        int n = route.Length;
        if (n < 3) return;

        for (int pass = 0; pass < 12; pass++)
        {
            bool improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                int a = i == 0 ? -1 : route[i - 1];
                int b = route[i];
                for (int j = i + 1; j < n; j++)
                {
                    int c = route[j];
                    int d = j + 1 < n ? route[j + 1] : -1;

                    double delta;
                    if (j == n - 1)
                    {
                        delta = (i == 0 ? StartCosts[c] : Costs[a, c]) - (i == 0 ? StartCosts[b] : Costs[a, b]);
                    }
                    else
                    {
                        double old1 = i == 0 ? StartCosts[b] : Costs[a, b];
                        double old2 = Costs[c, d];
                        double new1 = i == 0 ? StartCosts[c] : Costs[a, c];
                        double new2 = Costs[b, d];
                        delta = new1 + new2 - old1 - old2;
                    }

                    if (delta < -1e-10)
                    {
                        Array.Reverse(route, i, j - i + 1);
                        improved = true;
                        goto NextPass;
                    }
                }
            }
        NextPass:
            if (!improved) break;
        }
    }

    private static double RouteCost(int[] route)
    {
        if (route.Length == 0) return 0.0;
        double cost = StartCosts[route[0]];
        for (int i = 0; i + 1 < route.Length; i++)
        {
            cost += Costs[route[i], route[i + 1]];
        }
        return cost;
    }
}
