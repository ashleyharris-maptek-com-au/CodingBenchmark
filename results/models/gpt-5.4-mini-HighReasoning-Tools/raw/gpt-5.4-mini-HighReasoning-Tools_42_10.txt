OpCapability Shader
OpCapability Geometry
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

OpSource GLSL 450

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%functy = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4

%float_0 = OpConstant %float 0.0
%float_0_02 = OpConstant %float 0.02
%float_0_25 = OpConstant %float 0.25
%float_0_5 = OpConstant %float 0.5
%float_0_75 = OpConstant %float 0.75
%float_1 = OpConstant %float 1.0

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %functy
%entry = OpLabel
%gl_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr
%p3 = OpLoad %v4float %p3_ptr

%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2
%p3_xyz = OpVectorShuffle %v3float %p3 %p3 0 1 2

%start_vec = OpFSub %v3float %p2_xyz %p0_xyz
%end_vec = OpFSub %v3float %p3_xyz %p1_xyz
%start_tan = OpExtInst %v3float %glsl450 Normalize %start_vec
%end_tan = OpExtInst %v3float %glsl450 Normalize %end_vec

%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c1 = OpLoad %v3float %c1_ptr
%c2 = OpLoad %v3float %c2_ptr

%pos0 = OpVectorTimesScalar %v4float %p1 %float_1
%dir0 = OpVectorTimesScalar %v3float %start_tan %float_1
%dir0_n = OpExtInst %v3float %glsl450 Normalize %dir0
%dir0_xy = OpVectorShuffle %v2float %dir0_n %dir0_n 0 1
%dir0_x = OpCompositeExtract %float %dir0_xy 0
%dir0_y = OpCompositeExtract %float %dir0_xy 1
%dir0_neg_y = OpFNegate %float %dir0_y
%perp0_raw = OpCompositeConstruct %v2float %dir0_neg_y %dir0_x
%perp0_n = OpExtInst %v2float %glsl450 Normalize %perp0_raw
%perp0 = OpFMul %v2float %perp0_n %float_0_02
%perp0_x = OpCompositeExtract %float %perp0 0
%perp0_y = OpCompositeExtract %float %perp0 1
%perp0_v4 = OpCompositeConstruct %v4float %perp0_x %perp0_y %float_0 %float_0
%pos0_plus = OpFAdd %v4float %pos0 %perp0_v4
%pos0_minus = OpFSub %v4float %pos0 %perp0_v4
%col0 = OpVectorTimesScalar %v3float %c1 %float_1
OpStore %gl_out_pos %pos0_plus
OpStore %outColor %col0
OpEmitVertex
OpStore %gl_out_pos %pos0_minus
OpStore %outColor %col0
OpEmitVertex

%pos1a = OpVectorTimesScalar %v4float %p1 %float_0_75
%pos1b = OpVectorTimesScalar %v4float %p2 %float_0_25
%pos1 = OpFAdd %v4float %pos1a %pos1b
%dir1a = OpVectorTimesScalar %v3float %start_tan %float_0_75
%dir1b = OpVectorTimesScalar %v3float %end_tan %float_0_25
%dir1 = OpFAdd %v3float %dir1a %dir1b
%dir1_n = OpExtInst %v3float %glsl450 Normalize %dir1
%dir1_xy = OpVectorShuffle %v2float %dir1_n %dir1_n 0 1
%dir1_x = OpCompositeExtract %float %dir1_xy 0
%dir1_y = OpCompositeExtract %float %dir1_xy 1
%dir1_neg_y = OpFNegate %float %dir1_y
%perp1_raw = OpCompositeConstruct %v2float %dir1_neg_y %dir1_x
%perp1_n = OpExtInst %v2float %glsl450 Normalize %perp1_raw
%perp1 = OpFMul %v2float %perp1_n %float_0_02
%perp1_x = OpCompositeExtract %float %perp1 0
%perp1_y = OpCompositeExtract %float %perp1 1
%perp1_v4 = OpCompositeConstruct %v4float %perp1_x %perp1_y %float_0 %float_0
%pos1_plus = OpFAdd %v4float %pos1 %perp1_v4
%pos1_minus = OpFSub %v4float %pos1 %perp1_v4
%col1a = OpVectorTimesScalar %v3float %c1 %float_0_75
%col1b = OpVectorTimesScalar %v3float %c2 %float_0_25
%col1 = OpFAdd %v3float %col1a %col1b
OpStore %gl_out_pos %pos1_plus
OpStore %outColor %col1
OpEmitVertex
OpStore %gl_out_pos %pos1_minus
OpStore %outColor %col1
OpEmitVertex

%pos2a = OpVectorTimesScalar %v4float %p1 %float_0_5
%pos2b = OpVectorTimesScalar %v4float %p2 %float_0_5
%pos2 = OpFAdd %v4float %pos2a %pos2b
%dir2a = OpVectorTimesScalar %v3float %start_tan %float_0_5
%dir2b = OpVectorTimesScalar %v3float %end_tan %float_0_5
%dir2 = OpFAdd %v3float %dir2a %dir2b
%dir2_n = OpExtInst %v3float %glsl450 Normalize %dir2
%dir2_xy = OpVectorShuffle %v2float %dir2_n %dir2_n 0 1
%dir2_x = OpCompositeExtract %float %dir2_xy 0
%dir2_y = OpCompositeExtract %float %dir2_xy 1
%dir2_neg_y = OpFNegate %float %dir2_y
%perp2_raw = OpCompositeConstruct %v2float %dir2_neg_y %dir2_x
%perp2_n = OpExtInst %v2float %glsl450 Normalize %perp2_raw
%perp2 = OpFMul %v2float %perp2_n %float_0_02
%perp2_x = OpCompositeExtract %float %perp2 0
%perp2_y = OpCompositeExtract %float %perp2 1
%perp2_v4 = OpCompositeConstruct %v4float %perp2_x %perp2_y %float_0 %float_0
%pos2_plus = OpFAdd %v4float %pos2 %perp2_v4
%pos2_minus = OpFSub %v4float %pos2 %perp2_v4
%col2a = OpVectorTimesScalar %v3float %c1 %float_0_5
%col2b = OpVectorTimesScalar %v3float %c2 %float_0_5
%col2 = OpFAdd %v3float %col2a %col2b
OpStore %gl_out_pos %pos2_plus
OpStore %outColor %col2
OpEmitVertex
OpStore %gl_out_pos %pos2_minus
OpStore %outColor %col2
OpEmitVertex

%pos3a = OpVectorTimesScalar %v4float %p1 %float_0_25
%pos3b = OpVectorTimesScalar %v4float %p2 %float_0_75
%pos3 = OpFAdd %v4float %pos3a %pos3b
%dir3a = OpVectorTimesScalar %v3float %start_tan %float_0_25
%dir3b = OpVectorTimesScalar %v3float %end_tan %float_0_75
%dir3 = OpFAdd %v3float %dir3a %dir3b
%dir3_n = OpExtInst %v3float %glsl450 Normalize %dir3
%dir3_xy = OpVectorShuffle %v2float %dir3_n %dir3_n 0 1
%dir3_x = OpCompositeExtract %float %dir3_xy 0
%dir3_y = OpCompositeExtract %float %dir3_xy 1
%dir3_neg_y = OpFNegate %float %dir3_y
%perp3_raw = OpCompositeConstruct %v2float %dir3_neg_y %dir3_x

OpStore %gl_out_pos %pos3_plus
OpStore %outColor %col3
OpEmitVertex
OpStore %gl_out_pos %pos3_minus
OpStore %outColor %col3
OpEmitVertex

%pos4 = OpVectorTimesScalar %v4float %p2 %float_1
%dir4 = OpVectorTimesScalar %v3float %end_tan %float_1
%dir4_n = OpExtInst %v3float %glsl450 Normalize %dir4
%dir4_xy = OpVectorShuffle %v2float %dir4_n %dir4_n 0 1
%dir4_x = OpCompositeExtract %float %dir4_xy 0
%dir4_y = OpCompositeExtract %float %dir4_xy 1
%dir4_neg_y = OpFNegate %float %dir4_y

%perp4_n = OpExtInst %v2float %glsl450 Normalize %perp4_raw

OpEmitVertex

OpEndPrimitive
