using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    private const double Eps = 1e-9;
    private static readonly CultureInfo Ci = CultureInfo.InvariantCulture;

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int vCount = fs.NextInt();
        int fCount = fs.NextInt();
        if (vCount < 0 || fCount < 0)
        {
            return;
        }

        var vertices = new Vec3[vCount];
        for (int i = 0; i < vCount; i++)
        {
            vertices[i] = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
        }

        var faces = new Face[fCount];
        int minFaceIndex = int.MaxValue;
        int maxFaceIndex = int.MinValue;
        for (int i = 0; i < fCount; i++)
        {
            int a = fs.NextInt();
            int b = fs.NextInt();
            int c = fs.NextInt();
            faces[i] = new Face(a, b, c);
            if (a < minFaceIndex) minFaceIndex = a;
            if (b < minFaceIndex) minFaceIndex = b;
            if (c < minFaceIndex) minFaceIndex = c;
            if (a > maxFaceIndex) maxFaceIndex = a;
            if (b > maxFaceIndex) maxFaceIndex = b;
            if (c > maxFaceIndex) maxFaceIndex = c;
        }

        double w = fs.NextDouble();
        double h = fs.NextDouble();
        double d = fs.NextDouble();

        int indexOffset = (minFaceIndex >= 1 && maxFaceIndex <= vCount) ? 1 : 0;
        int budget = ChooseBudget(vCount, fCount, w, h, d);

        var candidates = BuildCandidates(vertices, faces, indexOffset, budget, vCount + fCount);
        if (candidates.Count == 0)
        {
            Console.WriteLine(0);
            return;
        }

        Candidate best = default;
        best.Count = -1;
        best.Volume = double.MaxValue;

        for (int i = 0; i < candidates.Count; i++)
        {
            Candidate cand = Evaluate(vertices, candidates[i], w, h, d);
            if (cand.Count > best.Count || (cand.Count == best.Count && cand.Volume < best.Volume))
            {
                best = cand;
                best.Orientation = candidates[i];
            }
        }

        var output = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        if (best.Count <= 0)
        {
            output.WriteLine(0);
            output.Flush();
            return;
        }

        output.WriteLine(best.Count);
        for (int z = 0; z < best.Nz; z++)
        {
            double tzBase = z * best.Dz - best.MinZ;
            for (int y = 0; y < best.Ny; y++)
            {
                double tyBase = y * best.Dy - best.MinY;
                for (int x = 0; x < best.Nx; x++)
                {
                    double tx = x * best.Dx - best.MinX;
                    double ty = tyBase;
                    double tz = tzBase;
                    WritePlacement(output, tx, ty, tz, best.Orientation.Q);
                }
            }
        }
        output.Flush();
    }

    private static int ChooseBudget(int vCount, int fCount, double w, double h, double d)
    {
        int complexity = vCount + fCount;
        double volume = Math.Abs(w * h * d);

        if (complexity <= 40 && volume <= 50_000.0) return 1536;
        if (complexity <= 120 && volume <= 500_000.0) return 768;
        if (complexity <= 400 && volume <= 5_000_000.0) return 256;
        if (complexity <= 1500) return 96;
        return 24;
    }

    private static List<Orientation> BuildCandidates(Vec3[] vertices, Face[] faces, int indexOffset, int budget, int complexity)
    {
        var list = new List<Orientation>(Math.Max(24, Math.Min(budget, 4096)));
        var seen = new HashSet<string>(StringComparer.Ordinal);

        void TryAdd(Mat3 m)
        {
            string key = m.Key();
            if (seen.Add(key))
            {
                list.Add(new Orientation(Quat.FromMatrix(m), m));
            }
        }

        AddCubeSymmetries(TryAdd);
        if (list.Count >= budget)
        {
            return list;
        }

        if (complexity <= 800)
        {
            var normals = CollectFaceNormals(vertices, faces, indexOffset, Math.Min(96, faces.Length));
            var axes = new[]
            {
                new Vec3(1, 0, 0), new Vec3(-1, 0, 0),
                new Vec3(0, 1, 0), new Vec3(0, -1, 0),
                new Vec3(0, 0, 1), new Vec3(0, 0, -1)
            };
            double[] twists = { 0.0, Math.PI * 0.5, Math.PI, Math.PI * 1.5 };

            foreach (var n in normals)
            {
                foreach (var axis in axes)
                {
                    Quat align = Quat.FromToRotation(n, axis);
                    Mat3 alignM = Mat3.FromQuaternion(align);
                    foreach (double twist in twists)
                    {
                        Mat3 twistM = Mat3.FromQuaternion(Quat.FromAxisAngle(axis, twist));
                        TryAdd(Mat3.Multiply(twistM, alignM));
                        if (list.Count >= budget)
                        {
                            return list;
                        }
                    }
                }
            }
        }

        int sampleBudget = Math.Max(0, budget - list.Count);
        int sampleCount = Math.Max(sampleBudget * 3, sampleBudget + 24);
        for (int i = 0; i < sampleCount && list.Count < budget; i++)
        {
            Quat q = SampleUniformQuaternion(i + 1);
            TryAdd(Mat3.FromQuaternion(q));
        }

        return list;
    }

    private static void AddCubeSymmetries(Action<Mat3> add)
    {
        Vec3[] dirs =
        {
            new Vec3(1, 0, 0), new Vec3(-1, 0, 0),
            new Vec3(0, 1, 0), new Vec3(0, -1, 0),
            new Vec3(0, 0, 1), new Vec3(0, 0, -1)
        };

        foreach (var forward in dirs)
        {
            foreach (var up in dirs)
            {
                if (Math.Abs(Vec3.Dot(forward, up)) > 0.5)
                {
                    continue;
                }

                Vec3 right = Vec3.Cross(up, forward);
                if (right.LengthSquared() < Eps)
                {
                    continue;
                }

                right = Vec3.Normalize(right);
                Vec3 u = Vec3.Normalize(up);
                Vec3 f = Vec3.Normalize(forward);

                var m = new Mat3(
                    right.X, u.X, f.X,
                    right.Y, u.Y, f.Y,
                    right.Z, u.Z, f.Z);
                add(m);
            }
        }
    }

    private static List<Vec3> CollectFaceNormals(Vec3[] vertices, Face[] faces, int indexOffset, int maxFaces)
    {
        var result = new List<Vec3>();
        var seen = new HashSet<string>(StringComparer.Ordinal);

        int limit = Math.Min(faces.Length, maxFaces);
        for (int i = 0; i < limit; i++)
        {
            int a = faces[i].A - indexOffset;
            int b = faces[i].B - indexOffset;
            int c = faces[i].C - indexOffset;
            if (a < 0 || b < 0 || c < 0 || a >= vertices.Length || b >= vertices.Length || c >= vertices.Length)
            {
                continue;
            }

            Vec3 ab = vertices[b] - vertices[a];
            Vec3 ac = vertices[c] - vertices[a];
            Vec3 n = Vec3.Cross(ab, ac);
            double len = n.Length();
            if (len <= Eps)
            {
                continue;
            }

            n = n / len;
            string key = NormalKey(n);
            if (seen.Add(key))
            {
                result.Add(n);
            }
        }

        return result;
    }

    private static string NormalKey(Vec3 n)
    {
        return string.Format(Ci, "{0:F4},{1:F4},{2:F4}", Math.Round(n.X, 4), Math.Round(n.Y, 4), Math.Round(n.Z, 4));
    }

    private static Quat SampleUniformQuaternion(int index)
    {
        double u1 = Halton(index, 2);
        double u2 = Halton(index, 3);
        double u3 = Halton(index, 5);

        double sqrt1 = Math.Sqrt(1.0 - u1);
        double sqrt2 = Math.Sqrt(u1);
        double twoPiU2 = 2.0 * Math.PI * u2;
        double twoPiU3 = 2.0 * Math.PI * u3;

        // Uniform random quaternion in (w, x, y, z) order.
        double x = sqrt1 * Math.Sin(twoPiU2);
        double y = sqrt1 * Math.Cos(twoPiU2);
        double z = sqrt2 * Math.Sin(twoPiU3);
        double w = sqrt2 * Math.Cos(twoPiU3);
        return Quat.Normalize(new Quat(w, x, y, z));
    }

    private static double Halton(int index, int b)
    {
        double f = 1.0;
        double r = 0.0;
        while (index > 0)
        {
            f /= b;
            r += f * (index % b);
            index /= b;
        }
        return r;
    }

    private static Candidate Evaluate(Vec3[] vertices, Orientation orientation, double w, double h, double d)
    {
        double minX = double.PositiveInfinity;
        double minY = double.PositiveInfinity;
        double minZ = double.PositiveInfinity;
        double maxX = double.NegativeInfinity;
        double maxY = double.NegativeInfinity;
        double maxZ = double.NegativeInfinity;

        Mat3 m = orientation.M;
        for (int i = 0; i < vertices.Length; i++)
        {
            Vec3 v = vertices[i];
            double x = m.M11 * v.X + m.M12 * v.Y + m.M13 * v.Z;
            double y = m.M21 * v.X + m.M22 * v.Y + m.M23 * v.Z;
            double z = m.M31 * v.X + m.M32 * v.Y + m.M33 * v.Z;

            if (x < minX) minX = x;
            if (y < minY) minY = y;
            if (z < minZ) minZ = z;
            if (x > maxX) maxX = x;
            if (y > maxY) maxY = y;
            if (z > maxZ) maxZ = z;
        }

        double dx = maxX - minX;
        double dy = maxY - minY;
        double dz = maxZ - minZ;

        int nx = dx <= Eps ? 0 : Math.Max(0, (int)Math.Floor((w + Eps) / dx));
        int ny = dy <= Eps ? 0 : Math.Max(0, (int)Math.Floor((h + Eps) / dy));
        int nz = dz <= Eps ? 0 : Math.Max(0, (int)Math.Floor((d + Eps) / dz));

        long count = (long)nx * ny * nz;
        double volume = dx * dy * dz;
        return new Candidate
        {
            Orientation = orientation,
            MinX = minX,
            MinY = minY,
            MinZ = minZ,
            Dx = dx,
            Dy = dy,
            Dz = dz,
            Nx = nx,
            Ny = ny,
            Nz = nz,
            Count = count,
            Volume = volume
        };
    }

    private static void WritePlacement(StreamWriter output, double tx, double ty, double tz, Quat q)
    {
        output.Write(tx.ToString("G17", Ci));
        output.Write(' ');
        output.Write(ty.ToString("G17", Ci));
        output.Write(' ');
        output.Write(tz.ToString("G17", Ci));
        output.Write(' ');
        output.Write(q.W.ToString("G17", Ci));
        output.Write(' ');
        output.Write(q.X.ToString("G17", Ci));
        output.Write(' ');
        output.Write(q.Y.ToString("G17", Ci));
        output.Write(' ');
        output.WriteLine(q.Z.ToString("G17", Ci));
    }

    private struct Face
    {
        public int A;
        public int B;
        public int C;

        public Face(int a, int b, int c)
        {
            A = a;
            B = b;
            C = c;
        }
    }

    private struct Candidate
    {
        public Orientation Orientation;
        public double MinX;
        public double MinY;
        public double MinZ;
        public double Dx;
        public double Dy;
        public double Dz;
        public int Nx;
        public int Ny;
        public int Nz;
        public long Count;
        public double Volume;
    }

    private readonly struct Orientation
    {
        public readonly Quat Q;
        public readonly Mat3 M;

        public Orientation(Quat q, Mat3 m)
        {
            Q = q;
            M = m;
        }
    }

    private readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public double LengthSquared()
        {
            return X * X + Y * Y + Z * Z;
        }

        public double Length()
        {
            return Math.Sqrt(LengthSquared());
        }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);

        public static double Dot(Vec3 a, Vec3 b)
        {
            return a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        }

        public static Vec3 Cross(Vec3 a, Vec3 b)
        {
            return new Vec3(
                a.Y * b.Z - a.Z * b.Y,
                a.Z * b.X - a.X * b.Z,
                a.X * b.Y - a.Y * b.X);
        }

        public static Vec3 Normalize(Vec3 v)
        {
            double len = v.Length();
            if (len <= Eps)
            {
                return new Vec3(0, 0, 0);
            }
            return v / len;
        }
    }

    private readonly struct Mat3
    {
        public readonly double M11;
        public readonly double M12;
        public readonly double M13;
        public readonly double M21;
        public readonly double M22;
        public readonly double M23;
        public readonly double M31;
        public readonly double M32;
        public readonly double M33;

        public Mat3(double m11, double m12, double m13, double m21, double m22, double m23, double m31, double m32, double m33)
        {
            M11 = m11; M12 = m12; M13 = m13;
            M21 = m21; M22 = m22; M23 = m23;
            M31 = m31; M32 = m32; M33 = m33;
        }

        public string Key()
        {
            return string.Format(
                Ci,
                "{0:F6},{1:F6},{2:F6},{3:F6},{4:F6},{5:F6},{6:F6},{7:F6},{8:F6}",
                Math.Round(M11, 6),
                Math.Round(M12, 6),
                Math.Round(M13, 6),
                Math.Round(M21, 6),
                Math.Round(M22, 6),
                Math.Round(M23, 6),
                Math.Round(M31, 6),
                Math.Round(M32, 6),
                Math.Round(M33, 6));
        }

        public static Mat3 Multiply(Mat3 a, Mat3 b)
        {
            return new Mat3(
                a.M11 * b.M11 + a.M12 * b.M21 + a.M13 * b.M31,
                a.M11 * b.M12 + a.M12 * b.M22 + a.M13 * b.M32,
                a.M11 * b.M13 + a.M12 * b.M23 + a.M13 * b.M33,

                a.M21 * b.M11 + a.M22 * b.M21 + a.M23 * b.M31,
                a.M21 * b.M12 + a.M22 * b.M22 + a.M23 * b.M32,
                a.M21 * b.M13 + a.M22 * b.M23 + a.M23 * b.M33,

                a.M31 * b.M11 + a.M32 * b.M21 + a.M33 * b.M31,
                a.M31 * b.M12 + a.M32 * b.M22 + a.M33 * b.M32,
                a.M31 * b.M13 + a.M32 * b.M23 + a.M33 * b.M33);
        }

        public static Mat3 FromQuaternion(Quat q)
        {
            q = Quat.Normalize(q);
            double w = q.W;
            double x = q.X;
            double y = q.Y;
            double z = q.Z;

            double xx = x * x;
            double yy = y * y;
            double zz = z * z;
            double xy = x * y;
            double xz = x * z;
            double yz = y * z;
            double wx = w * x;
            double wy = w * y;
            double wz = w * z;

            return new Mat3(
                1.0 - 2.0 * (yy + zz),
                2.0 * (xy - wz),
                2.0 * (xz + wy),

                2.0 * (xy + wz),
                1.0 - 2.0 * (xx + zz),
                2.0 * (yz - wx),

                2.0 * (xz - wy),
                2.0 * (yz + wx),
                1.0 - 2.0 * (xx + yy));
        }
    }

    private readonly struct Quat
    {
        public readonly double W;
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public Quat(double w, double x, double y, double z)
        {
            W = w;
            X = x;
            Y = y;
            Z = z;
        }

        public static Quat Normalize(Quat q)
        {
            double n = Math.Sqrt(q.W * q.W + q.X * q.X + q.Y * q.Y + q.Z * q.Z);
            if (n <= Eps)
            {
                return new Quat(1, 0, 0, 0);
            }
            return new Quat(q.W / n, q.X / n, q.Y / n, q.Z / n);
        }

        public static Quat FromAxisAngle(Vec3 axis, double angle)
        {
            axis = Vec3.Normalize(axis);
            double half = angle * 0.5;
            double s = Math.Sin(half);
            return Normalize(new Quat(Math.Cos(half), axis.X * s, axis.Y * s, axis.Z * s));
        }

        public static Quat FromToRotation(Vec3 from, Vec3 to)
        {
            from = Vec3.Normalize(from);
            to = Vec3.Normalize(to);

            double d = Vec3.Dot(from, to);
            if (d >= 1.0 - 1e-12)
            {
                return new Quat(1, 0, 0, 0);
            }

            if (d <= -1.0 + 1e-12)
            {
                Vec3 axis = Math.Abs(from.X) < 0.9 ? Vec3.Cross(from, new Vec3(1, 0, 0)) : Vec3.Cross(from, new Vec3(0, 1, 0));
                axis = Vec3.Normalize(axis);
                return FromAxisAngle(axis, Math.PI);
            }

            Vec3 c = Vec3.Cross(from, to);
            double s = Math.Sqrt((1.0 + d) * 2.0);
            double invS = 1.0 / s;
            return Normalize(new Quat(
                0.5 * s,
                c.X * invS,
                c.Y * invS,
                c.Z * invS));
        }

        public static Quat FromMatrix(Mat3 m)
        {
            double trace = m.M11 + m.M22 + m.M33;
            Quat q;

            if (trace > 0.0)
            {
                double s = Math.Sqrt(trace + 1.0) * 2.0;
                q = new Quat(
                    0.25 * s,
                    (m.M32 - m.M23) / s,
                    (m.M13 - m.M31) / s,
                    (m.M21 - m.M12) / s);
            }
            else if (m.M11 > m.M22 && m.M11 > m.M33)
            {
                double s = Math.Sqrt(1.0 + m.M11 - m.M22 - m.M33) * 2.0;
                q = new Quat(
                    (m.M32 - m.M23) / s,
                    0.25 * s,
                    (m.M12 + m.M21) / s,
                    (m.M13 + m.M31) / s);
            }
            else if (m.M22 > m.M33)
            {
                double s = Math.Sqrt(1.0 + m.M22 - m.M11 - m.M33) * 2.0;
                q = new Quat(
                    (m.M13 - m.M31) / s,
                    (m.M12 + m.M21) / s,
                    0.25 * s,
                    (m.M23 + m.M32) / s);
            }
            else
            {
                double s = Math.Sqrt(1.0 + m.M33 - m.M11 - m.M22) * 2.0;
                q = new Quat(
                    (m.M21 - m.M12) / s,
                    (m.M13 + m.M31) / s,
                    (m.M23 + m.M32) / s,
                    0.25 * s);
            }

            return Normalize(q);
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }

            return val * sign;
        }

        public double NextDouble()
        {
            return double.Parse(NextToken(), Ci);
        }

        private string NextToken()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            if (c < 0)
            {
                return string.Empty;
            }

            var sb = new StringBuilder();
            while (c > 32)
            {
                sb.Append((char)c);
                c = Read();
            }

            return sb.ToString();
        }
    }
