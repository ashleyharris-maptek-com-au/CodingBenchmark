use std::f64::consts::PI;
use std::io::{self, Read, Write};

struct FastInput {
    data: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data).unwrap();
        Self { data, idx: 0 }
    }

    fn next_usize(&mut self) -> usize {
        while self.idx < self.data.len() && self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        let mut val = 0usize;
        while self.idx < self.data.len() && !self.data[self.idx].is_ascii_whitespace() {
            val = val * 10 + (self.data[self.idx] - b'0') as usize;
            self.idx += 1;
        }
        val
    }
}

#[derive(Clone)]
struct Layout {
    points: Vec<(usize, f64, f64)>,
    min_x: f64,
    max_x: f64,
    min_y: f64,
    max_y: f64,
}

fn bfs_farthest(
    start: usize,
    cid: usize,
    comp_tag: &[usize],
    adj: &[Vec<usize>],
    mark: &mut [u32],
    dist: &mut [usize],
    parent: &mut [usize],
    stamp: &mut u32,
) -> usize {
    *stamp += 1;
    let cur_stamp = *stamp;
    let mut q = Vec::new();
    q.push(start);
    mark[start] = cur_stamp;
    dist[start] = 0;
    parent[start] = start;
    let mut head = 0usize;
    let mut far = start;

    while head < q.len() {
        let u = q[head];
        head += 1;
        far = u;
        let du = dist[u];
        for &v in &adj[u] {
            if comp_tag[v] != cid || mark[v] == cur_stamp {
                continue;
            }
            mark[v] = cur_stamp;
            dist[v] = du + 1;
            parent[v] = u;
            q.push(v);
        }
    }

    far
}

fn layout_degree_two(
    nodes: &[usize],
    cid: usize,
    comp_tag: &[usize],
    adj: &[Vec<usize>],
    deg: &[usize],
) -> Layout {
    let n = nodes.len();
    let mut start = nodes[0];
    let mut cycle = true;
    for &u in nodes {
        if deg[u] <= 1 {
            start = u;
            cycle = false;
            break;
        }
    }

    let mut order = Vec::with_capacity(n);
    let mut prev = usize::MAX;
    let mut cur = start;

    if cycle {
        for _ in 0..n {
            order.push(cur);
            let mut next = usize::MAX;
            for &v in &adj[cur] {
                if comp_tag[v] == cid && v != prev {
                    next = v;
                    break;
                }
            }
            prev = cur;
            cur = next;
        }
    } else {
        loop {
            order.push(cur);
            let mut next = usize::MAX;
            for &v in &adj[cur] {
                if comp_tag[v] == cid && v != prev {
                    next = v;
                    break;
                }
            }
            if next == usize::MAX {
                break;
            }
            prev = cur;
            cur = next;
        }
    }

    let mut points = Vec::with_capacity(n);
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;

    if cycle {
        let radius = (n as f64).max(6.0) / (2.0 * PI);
        for (i, &u) in order.iter().enumerate() {
            let ang = 2.0 * PI * (i as f64) / (n as f64);
            let x = radius * ang.cos();
            let y = radius * ang.sin();
            min_x = min_x.min(x);
            max_x = max_x.max(x);
            min_y = min_y.min(y);
            max_y = max_y.max(y);
            points.push((u, x, y));
        }
    } else {
        let offset = (order.len() as f64 - 1.0) * 0.5;
        for (i, &u) in order.iter().enumerate() {
            let x = i as f64 - offset;
            let y = 0.0;
            min_x = min_x.min(x);
            max_x = max_x.max(x);
            min_y = min_y.min(y);
            max_y = max_y.max(y);
            points.push((u, x, y));
        }
    }

    Layout {
        points,
        min_x,
        max_x,
        min_y,
        max_y,
    }
}

fn layout_circle(nodes: &[usize], _cid: usize, _comp_tag: &[usize], deg: &[usize]) -> Layout {
    let mut order: Vec<usize> = nodes.to_vec();
    order.sort_by(|&a, &b| deg[b].cmp(&deg[a]).then_with(|| a.cmp(&b)));

    let n = order.len();
    let radius = (n as f64).max(8.0) / (2.0 * PI);
    let mut points = Vec::with_capacity(n);
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;

    for (i, &u) in order.iter().enumerate() {
        let ang = 2.0 * PI * (i as f64) / (n as f64);
        let x = radius * ang.cos();
        let y = radius * ang.sin();
        min_x = min_x.min(x);
        max_x = max_x.max(x);
        min_y = min_y.min(y);
        max_y = max_y.max(y);
        points.push((u, x, y));
    }

    Layout {
        points,
        min_x,
        max_x,
        min_y,
        max_y,
    }
}

fn sort_level(
    level: &mut Vec<usize>,
    target_depth: usize,
    cid: usize,
    comp_tag: &[usize],
    adj: &[Vec<usize>],
    depth: &[usize],
    rank: &mut [usize],
) {
    let mut keyed: Vec<(f64, usize, usize)> = Vec::with_capacity(level.len());
    for &u in level.iter() {
        let mut sum = 0.0;
        let mut cnt = 0usize;
        for &v in &adj[u] {
            if comp_tag[v] == cid && depth[v] == target_depth {
                sum += rank[v] as f64;
                cnt += 1;
            }
        }
        let key = if cnt > 0 {
            sum / cnt as f64
        } else {
            rank[u] as f64
        };
        keyed.push((key, rank[u], u));
    }

    keyed.sort_by(|a, b| {
        a.0.total_cmp(&b.0)
            .then_with(|| a.1.cmp(&b.1))
            .then_with(|| a.2.cmp(&b.2))
    });

    for (i, &(_, _, u)) in keyed.iter().enumerate() {
        level[i] = u;
        rank[u] = i;
    }
}

fn layout_bfs_layers(
    nodes: &[usize],
    cid: usize,
    comp_tag: &[usize],
    adj: &[Vec<usize>],
    deg: &[usize],
    mark: &mut [u32],
    dist: &mut [usize],
    parent: &mut [usize],
    stamp: &mut u32,
    rank: &mut [usize],
) -> Layout {
    let start = nodes[0];
    let far_a = bfs_farthest(start, cid, comp_tag, adj, mark, dist, parent, stamp);
    let far_b = bfs_farthest(far_a, cid, comp_tag, adj, mark, dist, parent, stamp);

    let mut path = Vec::new();
    let mut cur = far_b;
    path.push(cur);
    while cur != far_a {
        cur = parent[cur];
        path.push(cur);
    }
    let root = path[path.len() / 2];

    bfs_farthest(root, cid, comp_tag, adj, mark, dist, parent, stamp);

    let mut max_depth = 0usize;
    for &u in nodes {
        max_depth = max_depth.max(dist[u]);
    }

    let mut levels: Vec<Vec<usize>> = vec![Vec::new(); max_depth + 1];
    for &u in nodes {
        levels[dist[u]].push(u);
    }

    for level in &mut levels {
        level.sort_by(|&a, &b| deg[b].cmp(&deg[a]).then_with(|| a.cmp(&b)));
    }
    for level in &levels {
        for (i, &u) in level.iter().enumerate() {
            rank[u] = i;
        }
    }

    let sweeps = if nodes.len() <= 32 {
        8
    } else if nodes.len() <= 200 {
        6
    } else {
        4
    };

    for _ in 0..sweeps {
        for d in 1..levels.len() {
            sort_level(
                &mut levels[d],
                d - 1,
                cid,
                comp_tag,
                adj,
                &dist,
                rank,
            );
        }
        if levels.len() > 1 {
            for d in (0..levels.len() - 1).rev() {
                sort_level(
                    &mut levels[d],
                    d + 1,
                    cid,
                    comp_tag,
                    adj,
                    &dist,
                    rank,
                );
            }
        }
    }

    let x_gap = 1.0;
    let y_gap = 2.0;
    let mut points = Vec::with_capacity(nodes.len());
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;

    for (depth, level) in levels.iter().enumerate() {
        let k = level.len();
        let offset = (k as f64 - 1.0) * 0.5;
        for (i, &u) in level.iter().enumerate() {
            let x = (i as f64 - offset) * x_gap;
            let y = depth as f64 * y_gap;
            min_x = min_x.min(x);
            max_x = max_x.max(x);
            min_y = min_y.min(y);
            max_y = max_y.max(y);
            points.push((u, x, y));
        }
    }

    Layout {
        points,
        min_x,
        max_x,
        min_y,
        max_y,
    }
}

fn main() {
    let mut input = FastInput::new();
    if input.data.is_empty() {
        return;
    }

    let n = input.next_usize();
    let m = input.next_usize();

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u = input.next_usize();
        let v = input.next_usize();
        if u < n && v < n && u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    let mut deg = vec![0usize; n];
    for i in 0..n {
        deg[i] = adj[i].len();
    }

    let mut comp_tag = vec![usize::MAX; n];
    let mut components: Vec<Vec<usize>> = Vec::new();
    for s in 0..n {
        if comp_tag[s] != usize::MAX {
            continue;
        }
        let cid = components.len();
        let mut stack = vec![s];
        comp_tag[s] = cid;
        let mut nodes = Vec::new();
        while let Some(u) = stack.pop() {
            nodes.push(u);
            for &v in &adj[u] {
                if comp_tag[v] == usize::MAX {
                    comp_tag[v] = cid;
                    stack.push(v);
                }
            }
        }
        components.push(nodes);
    }

    let mut mark = vec![0u32; n];
    let mut dist = vec![0usize; n];
    let mut parent = vec![0usize; n];
    let mut stamp = 0u32;
    let mut rank = vec![0usize; n];

    let mut layouts: Vec<Layout> = Vec::with_capacity(components.len());
    for (cid, nodes) in components.iter().enumerate() {
        let size = nodes.len();
        if size == 1 {
            let u = nodes[0];
            layouts.push(Layout {
                points: vec![(u, 0.0, 0.0)],
                min_x: 0.0,
                max_x: 0.0,
                min_y: 0.0,
                max_y: 0.0,
            });
            continue;
        }

        let edge_count = nodes.iter().map(|&u| adj[u].len()).sum::<usize>() / 2;
        let max_deg = nodes.iter().map(|&u| deg[u]).max().unwrap_or(0);

        if max_deg <= 2 {
            layouts.push(layout_degree_two(nodes, cid, &comp_tag, &adj, &deg));
        } else if size <= 300 && edge_count > size * 3 {
            layouts.push(layout_circle(nodes, cid, &comp_tag, &deg));
        } else {
            layouts.push(layout_bfs_layers(
                nodes,
                cid,
                &comp_tag,
                &adj,
                &deg,
                &mut mark,
                &mut dist,
                &mut parent,
                &mut stamp,
                &mut rank,
            ));
        }
    }

    layouts.sort_by(|a, b| {
        let aw = (a.max_x - a.min_x).max(1.0);
        let ah = (a.max_y - a.min_y).max(1.0);
        let bw = (b.max_x - b.min_x).max(1.0);
        let bh = (b.max_y - b.min_y).max(1.0);
        (bw * bh)
            .total_cmp(&(aw * ah))
            .then_with(|| b.points.len().cmp(&a.points.len()))
    });

    let mut out_x = vec![0.0f64; n];
    let mut out_y = vec![0.0f64; n];
    let target_width = ((n as f64).sqrt() * 10.0).max(120.0);
    let gap = 6.0;
    let mut cursor_x = 0.0;
    let mut cursor_y = 0.0;
    let mut row_h = 0.0;

    for layout in layouts {
        let w = (layout.max_x - layout.min_x).max(1.0);
        let h = (layout.max_y - layout.min_y).max(1.0);
        if cursor_x > 0.0 && cursor_x + w > target_width {
            cursor_x = 0.0;
            cursor_y += row_h + gap;
            row_h = 0.0;
        }
        for (u, x, y) in layout.points {
            out_x[u] = x - layout.min_x + cursor_x;
            out_y[u] = y - layout.min_y + cursor_y;
        }
        cursor_x += w + gap;
        row_h = row_h.max(h);
    }

    let mut stdout = io::BufWriter::new(io::stdout());
    for i in 0..n {
        writeln!(stdout, "{:.6} {:.6}", out_x[i], out_y[i]).unwrap();
    }
}
