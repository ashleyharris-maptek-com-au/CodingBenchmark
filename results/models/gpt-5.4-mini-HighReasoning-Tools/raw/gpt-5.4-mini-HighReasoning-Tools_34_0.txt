import math
import sys
from functools import lru_cache


def main():
    data = sys.stdin.buffer.read().split()
    if not data:
        return

    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    cap = int(next(it))

    depot_x = float(next(it))
    depot_y = float(next(it))

    xs = [depot_x]
    ys = [depot_y]
    demands = [0]
    angles = [0.0]
    depot_dist = [0.0]

    total_demand = 0
    for _ in range(n):
        x = float(next(it))
        y = float(next(it))
        d = int(next(it))
        xs.append(x)
        ys.append(y)
        demands.append(d)
        total_demand += d
        angles.append(math.atan2(y - depot_y, x - depot_x))
        depot_dist.append(math.hypot(x - depot_x, y - depot_y))

    if n == 0:
        sys.stdout.write("0.000000\n0\n")
        return

    @lru_cache(maxsize=None)
    def dist(i, j):
        if i > j:
            i, j = j, i
        return math.hypot(xs[i] - xs[j], ys[i] - ys[j])

    def route_length(route):
        if not route:
            return 0.0
        total = dist(0, route[0])
        for a, b in zip(route, route[1:]):
            total += dist(a, b)
        total += dist(route[-1], 0)
        return total

    def nn_order(nodes, start):
        remaining = set(nodes)
        order = [start]
        remaining.remove(start)
        cur = start
        while remaining:
            nxt = min(remaining, key=lambda j: dist(cur, j))
            order.append(nxt)
            remaining.remove(nxt)
            cur = nxt
        return order

    def best_oriented_route(route, allow_nn=False):
        if len(route) <= 2:
            return route[:]

        candidates = [route[:], list(reversed(route))]
        if allow_nn:
            start1 = min(route, key=lambda j: depot_dist[j])
            start2 = max(route, key=lambda j: depot_dist[j])
            seq1 = nn_order(route, start1)
            seq2 = nn_order(route, start2)
            candidates.extend([seq1, list(reversed(seq1)), seq2, list(reversed(seq2))])

        best = candidates[0]
        best_len = route_length(best)
        for cand in candidates[1:]:
            cand_len = route_length(cand)
            if cand_len < best_len:
                best = cand
                best_len = cand_len
        return best

    def greedy_sweep(order):
        routes = []
        cur = []
        cur_demand = 0
        for idx in order:
            d = demands[idx]
            if cur and cur_demand + d > cap:
                routes.append(cur)
                cur = [idx]
                cur_demand = d
            else:
                cur.append(idx)
                cur_demand += d
        if cur:
            routes.append(cur)
        return routes

    def best_fit_decreasing():
        items = sorted(range(1, n + 1), key=lambda j: (-demands[j], -depot_dist[j], j))
        bins = []  # [remaining_capacity, route]
        for idx in items:
            best_bin = -1
            best_left = None
            for bi, (rem, route) in enumerate(bins):
                if demands[idx] <= rem:
                    left = rem - demands[idx]
                    if best_left is None or left < best_left:
                        best_left = left
                        best_bin = bi
            if best_bin == -1:
                if len(bins) >= k:
                    return None
                bins.append([cap - demands[idx], [idx]])
            else:
                bins[best_bin][0] -= demands[idx]
                bins[best_bin][1].append(idx)
        return [route for _, route in bins]

    def seed_routes():
        remaining = set(range(1, n + 1))
        routes = []
        while remaining:
            seed = max(remaining, key=lambda j: (depot_dist[j], demands[j], -j))
            route = [seed]
            route_demand = demands[seed]
            remaining.remove(seed)
            cur = seed
            while True:
                best = None
                best_d = None
                for j in remaining:
                    if route_demand + demands[j] <= cap:
                        d = dist(cur, j)
                        if best_d is None or d < best_d:
                            best_d = d
                            best = j
                if best is None:
                    break
                route.append(best)
                route_demand += demands[best]
                remaining.remove(best)
                cur = best
            routes.append(route)
        return routes

    def score_candidate(routes, allow_nn=False):
        if not routes:
            return None
        seen = []
        seen_set = set()
        for route in routes:
            if not route:
                return None
            demand_sum = 0
            for idx in route:
                if idx < 1 or idx > n or idx in seen_set:
                    return None
                seen_set.add(idx)
                seen.append(idx)
                demand_sum += demands[idx]
            if demand_sum > cap:
                return None
        if len(seen_set) != n:
            return None
        if len(routes) > k:
            return None

        total = 0.0
        ordered = []
        for route in routes:
            seq = best_oriented_route(route, allow_nn=allow_nn)
            ordered.append(seq)
            total += route_length(seq)
        return total, ordered

    candidates = []

    sorted_by_angle = sorted(range(1, n + 1), key=lambda j: angles[j])
    if sorted_by_angle:
        starts = {0}
        starts.add(max(range(n), key=lambda i: depot_dist[sorted_by_angle[i]]))
        step = max(1, n // 12)
        for s in range(0, n, step):
            starts.add(s)
        for start in starts:
            order = sorted_by_angle[start:] + sorted_by_angle[:start]
            candidates.append((greedy_sweep(order), False))

    bf = best_fit_decreasing()
    if bf is not None:
        candidates.append((bf, True))

    sr = seed_routes()
    candidates.append((sr, True))

    best_total = None
    best_routes = None
    for routes, allow_nn in candidates:
        scored = score_candidate(routes, allow_nn=allow_nn)
        if scored is None:
            continue
        total, ordered = scored
        if best_total is None or total < best_total:
            best_total = total
            best_routes = ordered

    if best_routes is None:
        # Fallback: greedy sweep from the natural angular order.
        best_routes = [best_oriented_route(r, allow_nn=True) for r in greedy_sweep(sorted_by_angle)]
        best_total = sum(route_length(r) for r in best_routes)

    out = [f"{best_total:.6f}", str(len(best_routes))]
    out.extend(" ".join(str(idx - 1) for idx in route) for route in best_routes)
    sys.stdout.write("\n".join(out))


if __name__ == "__main__":
    main()
