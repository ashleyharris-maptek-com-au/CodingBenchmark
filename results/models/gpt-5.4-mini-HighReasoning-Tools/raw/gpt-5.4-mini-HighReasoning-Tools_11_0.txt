use std::f64::consts::PI;
use std::io::{self, Read};

const EPS: f64 = 1e-9;

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }
    fn add(self, o: Pt) -> Pt {
        Pt::new(self.x + o.x, self.y + o.y)
    }
    fn sub(self, o: Pt) -> Pt {
        Pt::new(self.x - o.x, self.y - o.y)
    }
    fn scale(self, k: f64) -> Pt {
        Pt::new(self.x * k, self.y * k)
    }
    fn dot(self, o: Pt) -> f64 {
        self.x * o.x + self.y * o.y
    }
    fn cross(self, o: Pt) -> f64 {
        self.x * o.y - self.y * o.x
    }
    fn norm2(self) -> f64 {
        self.x * self.x + self.y * self.y
    }
    fn dist2(self, o: Pt) -> f64 {
        self.sub(o).norm2()
    }
}

fn rotate(p: Pt, ang: f64) -> Pt {
    let (s, c) = ang.sin_cos();
    Pt::new(p.x * c - p.y * s, p.x * s + p.y * c)
}

fn polygon_area(poly: &[Pt]) -> f64 {
    let mut s = 0.0;
    for i in 0..poly.len() {
        let j = (i + 1) % poly.len();
        s += poly[i].cross(poly[j]);
    }
    0.5 * s
}

fn polygon_centroid(poly: &[Pt]) -> Pt {
    let a = polygon_area(poly);
    if a.abs() < EPS {
        let mut sx = 0.0;
        let mut sy = 0.0;
        for p in poly {
            sx += p.x;
            sy += p.y;
        }
        return Pt::new(sx / poly.len() as f64, sy / poly.len() as f64);
    }
    let mut cx = 0.0;
    let mut cy = 0.0;
    for i in 0..poly.len() {
        let j = (i + 1) % poly.len();
        let c = poly[i].cross(poly[j]);
        cx += (poly[i].x + poly[j].x) * c;
        cy += (poly[i].y + poly[j].y) * c;
    }
    Pt::new(cx / (6.0 * a), cy / (6.0 * a))
}

fn bbox(poly: &[Pt]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for p in poly {
        minx = minx.min(p.x);
        maxx = maxx.max(p.x);
        miny = miny.min(p.y);
        maxy = maxy.max(p.y);
    }
    (minx, maxx, miny, maxy)
}

fn signed_area_ccw(poly: &mut Vec<Pt>) {
    if polygon_area(poly) < 0.0 {
        poly.reverse();
    }
}

fn is_convex(poly: &[Pt]) -> bool {
    if poly.len() < 4 {
        return true;
    }
    let mut sign = 0i32;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let c = poly[(i + 2) % poly.len()];
        let v = (b.sub(a)).cross(c.sub(b));
        if v.abs() <= EPS {
            continue;
        }
        let s = if v > 0.0 { 1 } else { -1 };
        if sign == 0 {
            sign = s;
        } else if sign != s {
            return false;
        }
    }
    true
}

fn point_on_segment(p: Pt, a: Pt, b: Pt) -> bool {
    let ap = p.sub(a);
    let ab = b.sub(a);
    if ab.cross(ap).abs() > 1e-9 {
        return false;
    }
    ap.dot(p.sub(b)) <= 1e-9
}

fn orient(a: Pt, b: Pt, c: Pt) -> f64 {
    (b.sub(a)).cross(c.sub(a))
}

fn segments_proper_intersect(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);

    if o1.abs() <= EPS && point_on_segment(c, a, b) {
        return false;
    }
    if o2.abs() <= EPS && point_on_segment(d, a, b) {
        return false;
    }
    if o3.abs() <= EPS && point_on_segment(a, c, d) {
        return false;
    }
    if o4.abs() <= EPS && point_on_segment(b, c, d) {
        return false;
    }
    (o1 > 0.0) != (o2 > 0.0) && (o3 > 0.0) != (o4 > 0.0)
}

fn point_in_poly(poly: &[Pt], p: Pt) -> bool {
    let n = poly.len();
    let mut inside = false;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if point_on_segment(p, a, b) {
            return true;
        }
        let cond = (a.y > p.y) != (b.y > p.y);
        if cond {
            let x = a.x + (b.x - a.x) * (p.y - a.y) / (b.y - a.y);
            if x >= p.x - EPS {
                inside = !inside;
            }
        }
    }
    inside
}

fn point_in_convex(poly: &[Pt], p: Pt) -> bool {
    let n = poly.len();
    if n == 0 {
        return false;
    }
    if n == 1 {
        return poly[0].dist2(p) <= EPS * EPS;
    }
    if n == 2 {
        return point_on_segment(p, poly[0], poly[1]);
    }
    if orient(poly[0], poly[1], p) < -EPS || orient(poly[0], poly[n - 1], p) > EPS {
        return false;
    }
    let mut lo = 1usize;
    let mut hi = n - 1;
    while hi - lo > 1 {
        let mid = (lo + hi) / 2;
        if orient(poly[0], poly[mid], p) >= 0.0 {
            lo = mid;
        } else {
            hi = mid;
        }
    }
    orient(poly[lo], poly[lo + 1], p) >= -EPS
}

fn min_distance_to_edges(poly: &[Pt], p: Pt) -> f64 {
    let mut best = f64::INFINITY;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let ab = b.sub(a);
        let t = ((p.sub(a)).dot(ab) / ab.norm2()).clamp(0.0, 1.0);
        let q = a.add(ab.scale(t));
        best = best.min(p.dist2(q).sqrt());
    }
    best
}

fn polygon_samples(poly: &[Pt]) -> Vec<(Pt, f64)> {
    let mut out: Vec<(Pt, f64)> = Vec::new();
    let (minx, maxx, miny, maxy) = bbox(poly);
    let cx = (minx + maxx) * 0.5;
    let cy = (miny + maxy) * 0.5;
    let centroid = polygon_centroid(poly);
    let grid = 6usize;

    let push_unique = |p: Pt, score: f64, out: &mut Vec<(Pt, f64)>| {
        for (q, _) in out.iter() {
            if p.dist2(*q) < 1e-14 {
                return;
            }
        }
        out.push((p, score));
    };

    if point_in_poly(poly, centroid) {
        push_unique(centroid, min_distance_to_edges(poly, centroid), &mut out);
    }
    let center = Pt::new(cx, cy);
    if point_in_poly(poly, center) {
        push_unique(center, min_distance_to_edges(poly, center), &mut out);
    }
    for p in poly {
        if point_in_poly(poly, *p) {
            push_unique(*p, min_distance_to_edges(poly, *p), &mut out);
        }
    }
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let mid = Pt::new((a.x + b.x) * 0.5, (a.y + b.y) * 0.5);
        if point_in_poly(poly, mid) {
            push_unique(mid, min_distance_to_edges(poly, mid), &mut out);
        }
    }
    for iy in 0..=grid {
        let fy = if grid == 0 { 0.0 } else { iy as f64 / grid as f64 };
        let y = miny + (maxy - miny) * fy;
        for ix in 0..=grid {
            let fx = if grid == 0 { 0.0 } else { ix as f64 / grid as f64 };
            let x = minx + (maxx - minx) * fx;
            let p = Pt::new(x, y);
            if point_in_poly(poly, p) {
                push_unique(p, min_distance_to_edges(poly, p), &mut out);
            }
        }
    }
    out.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap());
    if out.len() > 32 {
        out.truncate(32);
    }
    out
}

fn rotate_polygon(poly: &[Pt], ang: f64) -> Vec<Pt> {
    poly.iter().map(|&p| rotate(p, ang)).collect()
}

fn longest_edge_angle(poly: &[Pt]) -> f64 {
    let mut best_len2 = -1.0;
    let mut best_ang = 0.0;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let d = b.sub(a);
        let l2 = d.norm2();
        if l2 > best_len2 {
            best_len2 = l2;
            best_ang = d.y.atan2(d.x);
        }
    }
    best_ang
}

fn transform(poly: &[Pt], shift: Pt) -> Vec<Pt> {
    poly.iter().map(|&p| p.add(shift)).collect()
}

fn fits(piece: &[Pt], stock: &[Pt], stock_convex: bool) -> bool {
    if stock_convex {
        for &p in piece {
            if !point_in_convex(stock, p) {
                return false;
            }
        }
        return true;
    }

    for &p in piece {
        if !point_in_poly(stock, p) {
            return false;
        }
    }
    for i in 0..piece.len() {
        let a1 = piece[i];
        let a2 = piece[(i + 1) % piece.len()];
        for j in 0..stock.len() {
            let b1 = stock[j];
            let b2 = stock[(j + 1) % stock.len()];
            if segments_proper_intersect(a1, a2, b1, b2) {
                return false;
            }
        }
    }
    true
}

fn candidate_rotations(piece: &[Pt]) -> Vec<f64> {
    let ang = longest_edge_angle(piece);
    let mut out = vec![0.0, PI * 0.5, -ang, PI * 0.5 - ang];
    let fallback = [PI * 0.25, PI * 0.75, PI * 1.25, PI * 1.75];
    out.extend_from_slice(&fallback);
    let mut uniq: Vec<f64> = Vec::new();
    'outer: for a in out {
        let mut aa = a;
        while aa < -PI {
            aa += 2.0 * PI;
        }
        while aa > PI {
            aa -= 2.0 * PI;
        }
        for &b in &uniq {
            let mut d = (aa - b).abs();
            while d > PI {
                d -= 2.0 * PI;
            }
            if d.abs() < 1e-8 {
                continue 'outer;
            }
        }
        uniq.push(aa);
    }
    uniq
}

fn piece_anchors(piece: &[Pt]) -> Vec<(Pt, f64)> {
    let centroid = polygon_centroid(piece);
    let (minx, maxx, miny, maxy) = bbox(piece);
    let center = Pt::new((minx + maxx) * 0.5, (miny + maxy) * 0.5);
    let first = piece[0];
    let mut longest_mid = first;
    let mut best_len2 = -1.0;
    for i in 0..piece.len() {
        let a = piece[i];
        let b = piece[(i + 1) % piece.len()];
        let d = b.sub(a);
        let l2 = d.norm2();
        if l2 > best_len2 {
            best_len2 = l2;
            longest_mid = Pt::new((a.x + b.x) * 0.5, (a.y + b.y) * 0.5);
        }
    }
    let refs = [centroid, center, first, longest_mid];
    let mut out = Vec::new();
    for r in refs {
        let mut rad: f64 = 0.0;
        for &p in piece {
            rad = rad.max(r.dist2(p).sqrt());
        }
        out.push((r, rad));
    }
    out
}

fn try_place(
    stock: &[Pt],
    stock_convex: bool,
    piece: &[Pt],
    sample_points: &[(Pt, f64)],
    rotations: &[f64],
) -> Option<(f64, f64, f64)> {
    let anchors = piece_anchors(piece);
    for &ang in rotations {
        let rotated: Vec<Pt> = rotate_polygon(piece, ang);
        let anchor_rotated: Vec<Pt> = anchors.iter().map(|(p, _)| rotate(*p, ang)).collect();

        for &(sample, clearance) in sample_points {
            for (idx, &(_, radius)) in anchors.iter().enumerate() {
                if clearance + 1e-9 < radius {
                    continue;
                }
                let shift = sample.sub(anchor_rotated[idx]);
                let placed = transform(&rotated, shift);
                if fits(&placed, stock, stock_convex) {
                    return Some((shift.x, shift.y, ang));
                }
            }
        }
    }
    None
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let sv: usize = it.next().unwrap_or("0").parse().unwrap_or(0);
    let np: usize = it.next().unwrap_or("0").parse().unwrap_or(0);

    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock.push(Pt::new(x, y));
    }
    signed_area_ccw(&mut stock);
    let stock_convex = is_convex(&stock);
    let stock_samples = polygon_samples(&stock);

    let mut pieces: Vec<Vec<Pt>> = Vec::with_capacity(np);
    for _ in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut poly = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            poly.push(Pt::new(x, y));
        }
        signed_area_ccw(&mut poly);
        pieces.push(poly);
    }

    let mut out = String::new();
    out.push_str(&format!("{}\n", np));

    for (i, piece) in pieces.iter().enumerate() {
        let mut samples = stock_samples.clone();
        if samples.is_empty() {
            samples.push((polygon_centroid(&stock), 0.0));
        }
        let rots1 = candidate_rotations(piece);
        let mut placed = try_place(&stock, stock_convex, piece, &samples, &rots1);

        if placed.is_none() {
            let (minx, maxx, miny, maxy) = bbox(&stock);
            let mut fallback_samples: Vec<(Pt, f64)> = Vec::new();
            let grid = 10usize;
            for iy in 0..=grid {
                let fy = iy as f64 / grid as f64;
                let y = miny + (maxy - miny) * fy;
                for ix in 0..=grid {
                    let fx = ix as f64 / grid as f64;
                    let x = minx + (maxx - minx) * fx;
                    let p = Pt::new(x, y);
                    if if stock_convex {
                        point_in_convex(&stock, p)
                    } else {
                        point_in_poly(&stock, p)
                    } {
                        fallback_samples.push((p, min_distance_to_edges(&stock, p)));
                    }
                }
            }
            fallback_samples.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap());
            if fallback_samples.len() > 40 {
                fallback_samples.truncate(40);
            }
            let mut rots2 = Vec::new();
            for k in 0..8 {
                rots2.push(k as f64 * PI / 4.0);
            }
            placed = try_place(&stock, stock_convex, piece, &fallback_samples, &rots2);
        }

        let (x, y, r) = placed.unwrap_or((0.0, 0.0, 0.0));
        out.push_str(&format!("{} {:.10} {:.10} {:.10}\n", i, x, y, r));
    }

    print!("{}", out);
}
