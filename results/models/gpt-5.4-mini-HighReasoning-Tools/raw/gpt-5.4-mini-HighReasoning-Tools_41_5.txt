OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %ubo %worldPos %normal %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

OpName %main "main"
OpName %ubo "ubo"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %outColor "outColor"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"

OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Function_v3float = OpTypePointer Function %v3float
%_ptr_Function_float = OpTypePointer Function %float

%c0 = OpConstant %float 0.0
%c0_2 = OpConstant %float 0.2
%c0_5 = OpConstant %float 0.5
%c0_8 = OpConstant %float 0.8
%c1 = OpConstant %float 1.0
%c32 = OpConstant %float 32.0
%i0 = OpConstant %int 0
%i1 = OpConstant %int 1
%i2 = OpConstant %int 2
%i3 = OpConstant %int 3
%i4 = OpConstant %int 4
%i5 = OpConstant %int 5

%glsl = OpExtInstImport "GLSL.std.450"

%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output

%main = OpFunction %void None %func
%label = OpLabel
%worldPosVal = OpLoad %v3float %worldPos
%normalVal = OpLoad %v3float %normal
%n = OpExtInst %v3float %glsl Normalize %normalVal

%lightPosPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %i3
%lightPos4 = OpLoad %v4float %lightPosPtr
%lightPos = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2

%cameraPosPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %i4
%cameraPos4 = OpLoad %v4float %cameraPosPtr
%cameraPos = OpVectorShuffle %v3float %cameraPos4 %cameraPos4 0 1 2

%toLight = OpFSub %v3float %lightPos %worldPosVal
%L = OpExtInst %v3float %glsl Normalize %toLight
%toCamera = OpFSub %v3float %cameraPos %worldPosVal
%V = OpExtInst %v3float %glsl Normalize %toCamera

%negL = OpFNegate %v3float %L
%R = OpExtInst %v3float %glsl Reflect %negL %n

%diffDot = OpExtInst %float %glsl Dot %n %L
%diffuse = OpExtInst %float %glsl FMax %diffDot %c0

%specDot = OpExtInst %float %glsl Dot %R %V
%specBase = OpExtInst %float %glsl FMax %specDot %c0
%specular = OpExtInst %float %glsl Pow %specBase %c32

%diffTerm = OpFMul %float %c0_5 %diffuse
%specTerm = OpFMul %float %c0_8 %specular
%intensity0 = OpFAdd %float %c0_2 %diffTerm
%intensity = OpFAdd %float %intensity0 %specTerm

%color = OpCompositeConstruct %v4float %intensity %intensity %intensity %c1
OpStore %outColor %color
OpReturn
OpFunctionEnd
