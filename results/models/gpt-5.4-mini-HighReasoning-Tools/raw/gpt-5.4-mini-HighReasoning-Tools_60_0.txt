const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input || '{}');

const tasksInput = Array.isArray(data.tasks) ? data.tasks : [];
const machineCountValue = Number(data.machineCount);
const machineCount = Math.max(1, Number.isFinite(machineCountValue) ? Math.floor(machineCountValue) : 1);

function topologicalFallback(tasks) {
  return tasks.map(t => t.id);
}

function buildGraph(tasks) {
  const n = tasks.length;
  const idToIndex = new Map();
  for (let i = 0; i < n; i++) idToIndex.set(tasks[i].id, i);

  const deps = new Array(n);
  const succ = new Array(n);
  const indeg = new Int32Array(n);
  const durations = new Float64Array(n);
  const deadlines = new Float64Array(n);
  const weights = new Float64Array(n);
  const outDeg = new Int32Array(n);

  for (let i = 0; i < n; i++) {
    const t = tasks[i];
    const dur = Number(t.duration);
    const ddl = Number(t.deadline);
    const w = Number(t.weight);
    durations[i] = Number.isFinite(dur) ? dur : 0;
    deadlines[i] = Number.isFinite(ddl) ? ddl : Number.POSITIVE_INFINITY;
    weights[i] = Number.isFinite(w) ? w : 1;
    const ds = Array.isArray(t.deps) ? t.deps : [];
    deps[i] = [];
    for (let j = 0; j < ds.length; j++) {
      const depIndex = idToIndex.get(ds[j]);
      if (depIndex !== undefined) deps[i].push(depIndex);
    }
    indeg[i] = deps[i].length;
  }

  for (let i = 0; i < n; i++) {
    for (const d of deps[i]) {
      outDeg[d]++;
    }
  }

  for (let i = 0; i < n; i++) succ[i] = new Int32Array(outDeg[i]);
  const fill = new Int32Array(n);
  for (let i = 0; i < n; i++) {
    for (const d of deps[i]) {
      succ[d][fill[d]++] = i;
    }
  }

  return { idToIndex, deps, succ, indeg, durations, deadlines, weights, outDeg };
}

function computeTail(durations, succ) {
  const n = durations.length;
  const indeg = new Int32Array(n);
  for (let i = 0; i < n; i++) {
    for (let k = 0; k < succ[i].length; k++) indeg[succ[i][k]]++;
  }
  const q = new Int32Array(n);
  let head = 0, tail = 0;
  for (let i = 0; i < n; i++) if (indeg[i] === 0) q[tail++] = i;

  const topo = new Int32Array(n);
  let topoLen = 0;
  while (head < tail) {
    const u = q[head++];
    topo[topoLen++] = u;
    const su = succ[u];
    for (let k = 0; k < su.length; k++) {
      const v = su[k];
      if (--indeg[v] === 0) q[tail++] = v;
    }
  }

  const tailLen = new Float64Array(n);
  for (let i = topoLen - 1; i >= 0; i--) {
    const u = topo[i];
    let best = 0;
    const su = succ[u];
    for (let k = 0; k < su.length; k++) {
      const v = su[k];
      if (tailLen[v] > best) best = tailLen[v];
    }
    tailLen[u] = durations[u] + best;
  }
  return tailLen;
}

class MinHeap {
  constructor(cmp) {
    this.a = [];
    this.cmp = cmp;
  }
  size() { return this.a.length; }
  peek() { return this.a[0]; }
  push(x) {
    const a = this.a;
    a.push(x);
    let i = a.length - 1;
    const cmp = this.cmp;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (cmp(a[i], a[p]) >= 0) break;
      [a[i], a[p]] = [a[p], a[i]];
      i = p;
    }
  }
  pop() {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop();
    if (a.length > 0) {
      a[0] = last;
      const cmp = this.cmp;
      let i = 0;
      while (true) {
        let l = i * 2 + 1;
        if (l >= a.length) break;
        let r = l + 1;
        let best = l;
        if (r < a.length && cmp(a[r], a[l]) < 0) best = r;
        if (cmp(a[best], a[i]) >= 0) break;
        [a[i], a[best]] = [a[best], a[i]];
        i = best;
      }
    }
    return top;
  }
}

function simulateOrder(graph, tailLen, strategy) {
  const { deps, succ, durations, deadlines, weights } = graph;
  const n = durations.length;
  const indeg = new Int32Array(n);
  for (let i = 0; i < n; i++) indeg[i] = deps[i].length;

  const readyCmp = strategy.readyCmpFactory({
    durations, deadlines, weights, tailLen, succ
  });
  const ready = new MinHeap(readyCmp);
  for (let i = 0; i < n; i++) if (indeg[i] === 0) ready.push(i);

  const running = new MinHeap((a, b) => a.finish - b.finish || a.task - b.task);
  let currentTime = 0;
  let available = machineCount;
  const order = [];
  const startTimes = new Float64Array(n);
  const finishTimes = new Float64Array(n);
  startTimes.fill(Number.NaN);
  finishTimes.fill(Number.NaN);

  while (order.length < n) {
    while (running.size() > 0 && running.peek().finish <= currentTime) {
      const done = running.pop();
      available++;
      const su = succ[done.task];
      for (let k = 0; k < su.length; k++) {
        const v = su[k];
        if (--indeg[v] === 0) ready.push(v);
      }
    }

    while (available > 0 && ready.size() > 0) {
      const task = ready.pop();
      order.push(task);
      const start = currentTime;
      const finish = start + durations[task];
      startTimes[task] = start;
      finishTimes[task] = finish;
      running.push({ finish, task });
      available--;
    }

    if (order.length >= n) break;
    if (running.size() === 0) {
      // Cycles or disconnected invalid input. Emit whatever we have.
      break;
    }
    currentTime = Math.max(currentTime, running.peek().finish);
  }

  // If invalid input or a cycle prevented completion, append remaining tasks.
  if (order.length < n) {
    const seen = new Uint8Array(n);
    for (const x of order) seen[x] = 1;
    for (let i = 0; i < n; i++) if (!seen[i]) order.push(i);
  }

  let weightedLate = 0;
  let makespan = 0;
  for (let i = 0; i < n; i++) {
    const c = finishTimes[i];
    if (Number.isFinite(c)) {
      if (c > makespan) makespan = c;
      const late = c - deadlines[i];
      if (late > 0) weightedLate += late * weights[i];
    }
  }

  const score = weightedLate + makespan;
  return { order, score };
}

function simulateOrderFromComparator(graph, tailLen, cmpFactory) {
  return simulateOrder(graph, tailLen, { readyCmpFactory: cmpFactory });
}

function makeComparatorFactory(type, machineCount) {
  const mFactor = 1 + Math.log2(Math.max(2, machineCount));
  return ({ durations, deadlines, weights, tailLen, succ }) => {
    const out = new Float64Array(durations.length);
    const fanout = new Float64Array(durations.length);
    for (let i = 0; i < durations.length; i++) fanout[i] = succ[i].length;
    for (let i = 0; i < durations.length; i++) {
      const d = durations[i];
      const w = weights[i];
      const t = tailLen[i];
      const dd = deadlines[i];
      const ratio = w / Math.max(1, d);
      let key;
      switch (type) {
        case 'edf':
          key = dd - 1.5 * mFactor * t - 0.05 * w - 0.01 * fanout[i];
          break;
        case 'critical':
          key = dd - 2.0 * mFactor * t - 0.1 * ratio - 0.02 * fanout[i];
          break;
        case 'density':
          key = -2.5 * ratio + 0.01 * dd - 0.2 * t + 0.05 * d;
          break;
        case 'hybrid':
          key = dd - 1.2 * t - 1.8 * ratio - 0.15 * fanout[i] + 0.03 * d;
          break;
        case 'fanout':
          key = dd - 1.0 * t - 0.35 * fanout[i] - 0.5 * ratio + 0.02 * d;
          break;
        default:
          key = dd - t;
      }
      out[i] = key;
    }
    return (a, b) => {
      const da = out[a], db = out[b];
      if (da !== db) return da - db;
      const wa = weights[a], wb = weights[b];
      if (wa !== wb) return wb - wa;
      const ta = tailLen[a], tb = tailLen[b];
      if (ta !== tb) return tb - ta;
      const la = durations[a], lb = durations[b];
      if (la !== lb) return la - lb;
      return a - b;
    };
  };
}

function solve(tasks) {
  const n = tasks.length;
  if (n === 0) return { order: [] };
  if (n === 1) return { order: [tasks[0].id] };

  const graph = buildGraph(tasks);
  const tailLen = computeTail(graph.durations, graph.succ);

  const strategies = ['edf', 'critical', 'density', 'hybrid', 'fanout'];
  let best = null;
  for (const type of strategies) {
    const candidate = simulateOrderFromComparator(graph, tailLen, makeComparatorFactory(type, machineCount));
    if (!best || candidate.score < best.score) best = candidate;
  }

  // Convert index order to original ids.
  const orderIds = best.order.map(i => tasks[i].id);
  return { order: orderIds };
}

const result = solve(tasksInput);
process.stdout.write(JSON.stringify(result));
