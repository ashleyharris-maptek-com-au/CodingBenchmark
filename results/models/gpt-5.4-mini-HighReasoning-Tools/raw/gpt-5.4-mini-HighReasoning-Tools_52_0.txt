import math


_S = {
    "speed_kmh": None,
    "yaw_rate_dps": 0.0,
    "lane_offset_m": 0.0,
    "curvature_1pm": 0.0,
    "front_gap_m": 200.0,
    "left_gap_m": 200.0,
    "right_gap_m": 200.0,
    "steer": 0.0,
    "throttle": 0.0,
    "brake": 0.0,
    "speed_int": 0.0,
    "avoid_side": 0,
    "avoid_until": -1.0,
    "sim_time_s": 0.0,
}


def _finite(x):
    return isinstance(x, (int, float)) and math.isfinite(float(x))


def _num(state, key, default=None):
    v = state.get(key, default)
    return float(v) if _finite(v) else default


def _flag(state, key, default=False):
    v = state.get(key, default)
    if isinstance(v, bool):
        return v
    if _finite(v):
        return abs(float(v)) > 0.5
    return default


def _clamp(x, lo, hi):
    return lo if x < lo else hi if x > hi else x


def _median(vals):
    vals = sorted(v for v in vals if _finite(v))
    if not vals:
        return None
    n = len(vals)
    m = n // 2
    if n % 2:
        return float(vals[m])
    return float((vals[m - 1] + vals[m]) / 2.0)


def _robust_group(vals, last=None, min_v=None, max_v=None):
    clean = []
    for v in vals:
        if not _finite(v):
            continue
        v = float(v)
        if min_v is not None and v < min_v:
            continue
        if max_v is not None and v > max_v:
            continue
        clean.append(v)
    if not clean:
        return last
    if len(clean) == 1:
        return clean[0]
    m = _median(clean)
    if m is None:
        return last
    if last is not None and _finite(last):
        last = float(last)
        close = [v for v in clean if abs(v - last) <= max(6.0, 0.25 * max(abs(last), 1.0))]
        if close:
            return sum(close) / len(close)
    spread = sorted(abs(v - m) for v in clean)
    mad = spread[len(spread) // 2] if spread else 0.0
    keep = [v for v in clean if abs(v - m) <= max(4.0, 2.5 * mad)]
    if keep:
        return sum(keep) / len(keep)
    return m


def _extract_group(state, specs):
    out = []
    for key, valid_key, lo, hi in specs:
        v = _num(state, key, None)
        if v is None:
            continue
        if lo is not None and v < lo:
            continue
        if hi is not None and v > hi:
            continue
        if valid_key is not None and not _flag(state, valid_key, True):
            # Weakly trust "invalid" values only if they look plausible.
            out.append(v)
        else:
            out.append(v)
    return out


def _mean(vals):
    vals = [float(v) for v in vals if _finite(v)]
    return sum(vals) / len(vals) if vals else None


def _best_bool(votes):
    trues = sum(1 for v in votes if v is True)
    falses = sum(1 for v in votes if v is False)
    return trues > falses


def _sensor_median(state, keys, valid_keys=None, lo=None, hi=None, last=None):
    vals = []
    for i, key in enumerate(keys):
        v = _num(state, key, None)
        if v is None:
            continue
        if lo is not None and v < lo:
            continue
        if hi is not None and v > hi:
            continue
        if valid_keys and i < len(valid_keys):
            if not _flag(state, valid_keys[i], True):
                vals.append(v)
            else:
                vals.append(v)
        else:
            vals.append(v)
    return _robust_group(vals, last=last, min_v=lo, max_v=hi)


def _fuse_speed(state):
    wheel = _sensor_median(
        state,
        [
            "wheel_speed_fl_kmh",
            "wheel_speed_fr_kmh",
            "wheel_speed_rl_kmh",
            "wheel_speed_rr_kmh",
        ],
        [
            "wheel_speed_fl_valid",
            "wheel_speed_fr_valid",
            "wheel_speed_rl_valid",
            "wheel_speed_rr_valid",
        ],
        lo=0.0,
        hi=350.0,
        last=_S["speed_kmh"],
    )
    gps = _sensor_median(
        state,
        ["gps_1_speed_kmh", "gps_2_speed_kmh"],
        ["gps_1_valid", "gps_2_valid"],
        lo=0.0,
        hi=350.0,
        last=_S["speed_kmh"],
    )
    veh = _mean([_num(state, "vehicle_speed_1_kmh"), _num(state, "vehicle_speed_2_kmh")])
    candidates = [v for v in [wheel, gps, veh] if _finite(v)]
    if not candidates:
        return _S["speed_kmh"] if _S["speed_kmh"] is not None else 0.0
    if len(candidates) == 1:
        return float(candidates[0])
    if _S["speed_kmh"] is not None:
        last = float(_S["speed_kmh"])
        close = [v for v in candidates if abs(v - last) <= max(10.0, 0.28 * max(last, 1.0))]
        if close:
            return sum(close) / len(close)
    # Prefer the tighter cluster when one source is clearly disagreeing.
    best = min(candidates, key=lambda v: sum(abs(v - u) for u in candidates))
    return float(best)


def _fuse_yaw_rate(state):
    return _sensor_median(
        state,
        ["gyro_1_yaw_rate_dps", "gyro_2_yaw_rate_dps", "gyro_3_yaw_rate_dps"],
        ["gyro_1_valid", "gyro_2_valid", "gyro_3_valid"],
        lo=-720.0,
        hi=720.0,
        last=_S["yaw_rate_dps"],
    )


def _vision_lane_offset(state):
    vals = []
    for i in (1, 2, 3):
        off = _num(state, f"vision_{i}_lane_offset_m", None)
        if _finite(off) and abs(off) < 20.0:
            vals.append(float(off))
        left = _num(state, f"vision_{i}_left_line_dist_m", None)
        right = _num(state, f"vision_{i}_right_line_dist_m", None)
        if _finite(left) and _finite(right) and left >= 0.0 and right >= 0.0:
            # Positive means the car is to the right of lane center.
            vals.append((float(left) - float(right)) / 2.0)
    return _robust_group(vals, last=_S["lane_offset_m"], min_v=-10.0, max_v=10.0) or 0.0


def _fuse_curvature(state):
    vals = []
    for i in (1, 2, 3):
        curv = _num(state, f"vision_{i}_road_curvature_1pm", None)
        if _finite(curv) and abs(curv) < 0.5:
            vals.append(float(curv))
    return _robust_group(vals, last=_S["curvature_1pm"], min_v=-1.0, max_v=1.0) or 0.0


def _fuse_speed_limit(state):
    vals = []
    for i in (1, 2, 3):
        if _flag(state, f"vision_{i}_speed_limit_valid", False):
            v = _num(state, f"vision_{i}_speed_limit_kmh", None)
            if _finite(v) and 20.0 <= v <= 200.0:
                vals.append(float(v))
    if vals:
        return _median(vals)
    return None


def _line_type_is_solid(v):
    if isinstance(v, str):
        return "solid" in v.lower()
    if _finite(v):
        # Best effort for encoded enums: 1 tends to be solid in many datasets.
        return int(round(float(v))) == 1
    return False


def _front_and_side_gaps(state):
    front_angles = [+5, +15, +30, -5, -15, -30]
    left_angles = [+60, +90]
    right_angles = [-60, -90]
    front = []
    left = []
    right = []
    for ang in front_angles:
        key = f"lidar_{ang:+04d}_deg_range_m"
        if _flag(state, f"lidar_{ang:+04d}_deg_valid", False):
            v = _num(state, key, None)
            if _finite(v) and 0.0 <= v <= 400.0:
                front.append(float(v))
    for ang in left_angles:
        key = f"lidar_{ang:+04d}_deg_range_m"
        if _flag(state, f"lidar_{ang:+04d}_deg_valid", False):
            v = _num(state, key, None)
            if _finite(v) and 0.0 <= v <= 400.0:
                left.append(float(v))
    for ang in right_angles:
        key = f"lidar_{ang:+04d}_deg_range_m"
        if _flag(state, f"lidar_{ang:+04d}_deg_valid", False):
            v = _num(state, key, None)
            if _finite(v) and 0.0 <= v <= 400.0:
                right.append(float(v))
    front_gap = min(front) if front else _S["front_gap_m"]
    left_gap = min(left) if left else _S["left_gap_m"]
    right_gap = min(right) if right else _S["right_gap_m"]
    return front_gap, left_gap, right_gap


def _obstacle_from_vision(state):
    detected = []
    dists = []
    rel_speeds = []
    for i in (1, 2, 3):
        if _flag(state, f"vision_{i}_obstacle_valid", False):
            det = state.get(f"vision_{i}_obstacle_detected", False)
            if isinstance(det, bool):
                detected.append(det)
            elif _finite(det):
                detected.append(abs(float(det)) > 0.5)
            dist = _num(state, f"vision_{i}_obstacle_dist_m", None)
            if _finite(dist) and 0.0 <= dist <= 500.0:
                dists.append(float(dist))
            rel = _num(state, f"vision_{i}_obstacle_rel_speed_kmh", None)
            if _finite(rel) and abs(rel) <= 400.0:
                rel_speeds.append(float(rel))
    return (_best_bool(detected) if detected else False, _median(dists), _median(rel_speeds))


def _traction_limit_kmh(mu, rain, temp_c):
    mu = _clamp(mu if _finite(mu) else 0.8, 0.05, 1.2)
    speed = 20.0 + 100.0 * (mu ** 0.8)
    if mu < 0.35:
        speed = min(speed, 55.0)
    if mu < 0.2:
        speed = min(speed, 40.0)
    if rain:
        speed *= 0.88
    if _finite(temp_c):
        if temp_c < 0.0:
            speed *= 0.80
        elif temp_c < 3.0:
            speed *= 0.90
    return _clamp(speed, 15.0, 140.0)


def autopilot_step(state: dict, dt: float) -> dict:
    if not isinstance(state, dict):
        state = {}

    now = _num(state, "sim_time_s", _S["sim_time_s"])
    if _finite(now):
        _S["sim_time_s"] = float(now)
    step_dt = _num(state, "dt", dt)
    if not _finite(step_dt) or step_dt <= 0.0:
        step_dt = dt if _finite(dt) and dt > 0.0 else 0.05
    step_dt = _clamp(float(step_dt), 0.02, 0.2)

    # Primary fused state.
    speed_kmh = _fuse_speed(state)
    yaw_rate_dps = _fuse_yaw_rate(state)
    lane_offset_m = _vision_lane_offset(state)
    curvature_1pm = _fuse_curvature(state)
    speed_limit_kmh = _fuse_speed_limit(state)
    front_gap_m, left_gap_m, right_gap_m = _front_and_side_gaps(state)
    obstacle_seen, obstacle_dist_m, obstacle_rel_speed = _obstacle_from_vision(state)

    if obstacle_dist_m is None:
        obstacle_dist_m = front_gap_m
    if not _finite(obstacle_dist_m):
        obstacle_dist_m = front_gap_m
    obstacle_dist_m = float(obstacle_dist_m)

    rain = _flag(state, "rain_sensor", False)
    mu = _num(state, "estimated_mu", 0.8)
    temp_c = _num(state, "temperature_c", None)
    traction_cap = _traction_limit_kmh(mu if _finite(mu) else 0.8, rain, temp_c)

    current_lane = _num(state, "current_lane", 1)
    target_lane = _num(state, "target_lane", current_lane if _finite(current_lane) else 1)
    if not _finite(current_lane):
        current_lane = 1.0
    if not _finite(target_lane):
        target_lane = current_lane
    lane_width = 3.7
    lane_bias = (float(target_lane) - float(current_lane)) * lane_width

    # Obstacle avoidance. Prefer the more open side, but keep the movement modest.
    avoid_side = 0
    avoid_until = _S["avoid_until"]
    if obstacle_seen or obstacle_dist_m < 35.0 or front_gap_m < 25.0:
        left_open = left_gap_m > 10.0
        right_open = right_gap_m > 10.0
        if left_open and right_open:
            if left_gap_m > right_gap_m + 2.0:
                avoid_side = -1
            elif right_gap_m > left_gap_m + 2.0:
                avoid_side = 1
        elif left_open:
            avoid_side = -1
        elif right_open:
            avoid_side = 1
        if avoid_side != 0:
            _S["avoid_side"] = avoid_side
            _S["avoid_until"] = _S["sim_time_s"] + 2.5
    if _S["sim_time_s"] <= avoid_until and _S["avoid_side"] != 0:
        avoid_side = _S["avoid_side"]
    else:
        _S["avoid_side"] = 0
        _S["avoid_until"] = -1.0

    if avoid_side < 0:
        lane_bias -= lane_width
    elif avoid_side > 0:
        lane_bias += lane_width
    lane_bias = _clamp(lane_bias, -2.0 * lane_width, 2.0 * lane_width)

    # Vision lines can block an unsafe lane change if they look solid.
    solid_left = False
    solid_right = False
    for i in (1, 2, 3):
        if _flag(state, f"vision_{i}_left_line_valid", False):
            solid_left = solid_left or _line_type_is_solid(state.get(f"vision_{i}_left_line_type"))
        if _flag(state, f"vision_{i}_right_line_valid", False):
            solid_right = solid_right or _line_type_is_solid(state.get(f"vision_{i}_right_line_type"))
    if lane_bias < 0.0 and solid_left and not (avoid_side < 0):
        lane_bias = 0.0
    if lane_bias > 0.0 and solid_right and not (avoid_side > 0):
        lane_bias = 0.0

    # Lane keeping uses offset, yaw rate, and a small curvature term.
    lane_error = float(lane_offset_m) - float(lane_bias)
    lane_error = _clamp(lane_error, -6.0, 6.0)
    speed_ms = speed_kmh / 3.6 if _finite(speed_kmh) else 0.0
    mu_clamped = _clamp(mu if _finite(mu) else 0.8, 0.05, 1.2)
    yaw_term = _clamp(yaw_rate_dps / 55.0, -1.0, 1.0)
    steer_kp = 0.28 + 0.10 * (1.0 - min(mu_clamped, 1.0))
    steer_kd = 0.18 + 0.10 * (1.0 - min(mu_clamped, 1.0))
    steer = -(steer_kp * lane_error) - (steer_kd * yaw_term) - (0.7 * curvature_1pm)

    # Stabilize on slippery pavement.
    lat_acc = None
    for i in (1, 2, 3):
        for prefix in ("accel_1", "accel_2", "accel_3"):
            pass
    lat_vals = [
        _num(state, "accel_1_lateral_ms2", None) if _flag(state, "accel_1_valid", False) else None,
        _num(state, "accel_2_lateral_ms2", None) if _flag(state, "accel_2_valid", False) else None,
        _num(state, "accel_3_lateral_ms2", None) if _flag(state, "accel_3_valid", False) else None,
    ]
    lat_vals = [v for v in lat_vals if _finite(v)]
    if lat_vals:
        lat_acc = _median(lat_vals)
    if _finite(lat_acc):
        steer -= _clamp(float(lat_acc) / 8.0, -0.35, 0.35)

    steer_limit = 1.0
    if mu_clamped < 0.35:
        steer_limit = 0.45
    elif mu_clamped < 0.6:
        steer_limit = 0.70
    if speed_kmh > 110.0:
        steer_limit *= 0.90
    if obstacle_dist_m < 20.0:
        steer_limit *= 0.80
    steer = _clamp(steer, -steer_limit, steer_limit)

    # Speed planning.
    target_speed = _num(state, "target_speed_kmh", None)
    desired_speed = float(speed_kmh) if not _finite(target_speed) else float(target_speed)
    if _finite(speed_limit_kmh):
        desired_speed = min(desired_speed, float(speed_limit_kmh))
    desired_speed = min(desired_speed, traction_cap)

    # Curvature speed cap: keep lateral acceleration reasonable.
    curv = abs(curvature_1pm)
    if curv > 1e-4:
        max_lat = max(1.5, 4.5 * mu_clamped)
        curve_cap_ms = math.sqrt(max(0.0, max_lat / max(curv, 1e-4)))
        desired_speed = min(desired_speed, curve_cap_ms * 3.6)

    # Braking envelope for obstacles ahead.
    obstacle_ahead = obstacle_seen or front_gap_m < 50.0 or obstacle_dist_m < 50.0
    if obstacle_ahead:
        gap = max(0.0, min(front_gap_m, obstacle_dist_m))
        safe_decel = max(1.8, 2.8 * mu_clamped + 0.8)
        if gap <= 4.0:
            desired_speed = 0.0
        else:
            safe_speed_ms = math.sqrt(max(0.0, 2.0 * safe_decel * max(0.0, gap - 2.0)))
            desired_speed = min(desired_speed, safe_speed_ms * 3.6)
        if obstacle_rel_speed is not None and _finite(obstacle_rel_speed):
            desired_speed = min(desired_speed, speed_kmh - 0.15 * float(obstacle_rel_speed))

    # On very low mu, bias hard toward safe slow travel.
    if mu_clamped < 0.2:
        desired_speed = min(desired_speed, 35.0)
    elif mu_clamped < 0.35:
        desired_speed = min(desired_speed, 55.0)

    desired_speed = _clamp(desired_speed, 0.0, max(5.0, traction_cap))

    # Persistent speed control with anti-windup.
    speed_error = desired_speed - float(speed_kmh)
    if abs(speed_error) < 1.5:
        _S["speed_int"] *= 0.90
    else:
        _S["speed_int"] += speed_error * step_dt
    _S["speed_int"] = _clamp(_S["speed_int"], -250.0, 250.0)

    speed_deriv = 0.0
    if _S["speed_kmh"] is not None and step_dt > 0.0:
        speed_deriv = (float(speed_kmh) - float(_S["speed_kmh"])) / step_dt

    kp = 0.018
    ki = 0.0035
    kd = 0.010
    accel_cmd = kp * speed_error + ki * _S["speed_int"] - kd * speed_deriv
    accel_cmd = _clamp(accel_cmd, -4.0, 3.0)

    throttle = 0.0
    brake = 0.0
    if accel_cmd >= 0.05:
        throttle = _clamp(accel_cmd / 2.5, 0.0, 1.0)
    elif accel_cmd <= -0.05:
        brake = _clamp((-accel_cmd) / 3.5, 0.0, 1.0)

    # Emergency overrides.
    if obstacle_ahead and desired_speed <= 5.0:
        brake = max(brake, 0.65)
        throttle = 0.0
    if front_gap_m < 8.0 or obstacle_dist_m < 8.0:
        brake = 1.0
        throttle = 0.0
    if mu_clamped < 0.25 and speed_error < -2.0:
        brake = max(brake, 0.35)
        throttle = 0.0

    # Gentle command smoothing for flicker resistance.
    steer = 0.55 * _S["steer"] + 0.45 * steer
    throttle = 0.50 * _S["throttle"] + 0.50 * throttle
    brake = 0.50 * _S["brake"] + 0.50 * brake

    if obstacle_ahead:
        close_gap = min(front_gap_m, obstacle_dist_m)
        if close_gap < 30.0:
            brake = max(brake, _clamp((30.0 - close_gap) / 20.0, 0.25, 1.0))

    # Never command power and brake together at high magnitude.
    if brake > 0.08:
        throttle *= max(0.0, 1.0 - 1.5 * brake)
    if throttle > 0.05 and brake > 0.05:
        if speed_error >= 0.0:
            brake *= 0.2
        else:
            throttle *= 0.2

    steer = _clamp(steer, -1.0, 1.0)
    throttle = _clamp(throttle, 0.0, 1.0)
    brake = _clamp(brake, 0.0, 1.0)

    # Update persistent state.
    _S["speed_kmh"] = float(speed_kmh) if _finite(speed_kmh) else _S["speed_kmh"]
    _S["yaw_rate_dps"] = float(yaw_rate_dps) if _finite(yaw_rate_dps) else _S["yaw_rate_dps"]
    _S["lane_offset_m"] = float(lane_offset_m) if _finite(lane_offset_m) else _S["lane_offset_m"]
    _S["curvature_1pm"] = float(curvature_1pm) if _finite(curvature_1pm) else _S["curvature_1pm"]
    _S["front_gap_m"] = float(front_gap_m) if _finite(front_gap_m) else _S["front_gap_m"]
    _S["left_gap_m"] = float(left_gap_m) if _finite(left_gap_m) else _S["left_gap_m"]
    _S["right_gap_m"] = float(right_gap_m) if _finite(right_gap_m) else _S["right_gap_m"]
    _S["steer"] = float(steer)
    _S["throttle"] = float(throttle)
    _S["brake"] = float(brake)

    return {
        "steering": float(steer),
        "throttle": float(throttle),
        "brake": float(brake),
    }
