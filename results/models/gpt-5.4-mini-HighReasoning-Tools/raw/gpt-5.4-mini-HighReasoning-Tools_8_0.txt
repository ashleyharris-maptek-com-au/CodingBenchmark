use std::io::{self, Read, Write};

fn parse_usize(bytes: &[u8], idx: &mut usize) -> usize {
    while *idx < bytes.len() && bytes[*idx].is_ascii_whitespace() {
        *idx += 1;
    }
    let mut value = 0usize;
    while *idx < bytes.len() && bytes[*idx].is_ascii_digit() {
        value = value * 10 + (bytes[*idx] - b'0') as usize;
        *idx += 1;
    }
    value
}

fn main() {
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();

    let mut p = 0usize;
    let h = parse_usize(&input, &mut p);
    let w = parse_usize(&input, &mut p);

    while p < input.len() && input[p] != b'\n' {
        p += 1;
    }
    if p < input.len() && input[p] == b'\n' {
        p += 1;
    }

    let total = match h.checked_mul(w) {
        Some(v) => v,
        None => {
            let mut out = io::BufWriter::new(io::stdout());
            writeln!(out, "0").unwrap();
            return;
        }
    };

    let mut grid = vec![b'#'; total];
    let mut parent = vec![0u8; total];
    let mut start = None::<usize>;
    let mut goal = None::<usize>;

    for r in 0..h {
        if p > input.len() {
            break;
        }
        let row_end = if let Some(pos) = input[p..].iter().position(|&b| b == b'\n') {
            p + pos
        } else {
            input.len()
        };
        let mut row = &input[p..row_end];
        if row.last() == Some(&b'\r') {
            row = &row[..row.len().saturating_sub(1)];
        }

        let base = r * w;
        let len = row.len().min(w);
        for c in 0..len {
            let ch = row[c];
            let idx = base + c;
            grid[idx] = ch;
            if ch == b'A' {
                start = Some(idx);
            } else if ch == b'B' {
                goal = Some(idx);
            }
        }

        p = if row_end < input.len() { row_end + 1 } else { row_end };
    }

    let (start, goal) = match (start, goal) {
        (Some(s), Some(g)) => (s, g),
        _ => {
            let mut out = io::BufWriter::new(io::stdout());
            writeln!(out, "0").unwrap();
            return;
        }
    };

    let mut stack = Vec::new();
    stack.push(start);
    grid[start] = b'.';
    parent[start] = 255;

    let mut found = false;

    while let Some(idx) = stack.pop() {
        if idx == goal {
            found = true;
            break;
        }

        let r = idx / w;
        let c = idx - r * w;

        if r > 0 {
            let n = idx - w;
            if grid[n] != b'#' && grid[n] != b'.' {
                grid[n] = b'.';
                parent[n] = 2; // parent is below current
                stack.push(n);
            }
        }
        if r + 1 < h {
            let n = idx + w;
            if grid[n] != b'#' && grid[n] != b'.' {
                grid[n] = b'.';
                parent[n] = 1; // parent is above current
                stack.push(n);
            }
        }
        if c > 0 {
            let n = idx - 1;
            if grid[n] != b'#' && grid[n] != b'.' {
                grid[n] = b'.';
                parent[n] = 4; // parent is right of current
                stack.push(n);
            }
        }
        if c + 1 < w {
            let n = idx + 1;
            if grid[n] != b'#' && grid[n] != b'.' {
                grid[n] = b'.';
                parent[n] = 3; // parent is left of current
                stack.push(n);
            }
        }
    }

    let mut out = io::BufWriter::new(io::stdout());
    if !found {
        writeln!(out, "0").unwrap();
        return;
    }

    let mut path = Vec::new();
    let mut cur = goal;
    loop {
        path.push(cur);
        if cur == start {
            break;
        }
        cur = match parent[cur] {
            1 => cur - w,
            2 => cur + w,
            3 => cur - 1,
            4 => cur + 1,
            _ => break,
        };
    }

    path.reverse();
    writeln!(out, "{}", path.len()).unwrap();
    for idx in path {
        writeln!(out, "{} {}", idx / w, idx % w).unwrap();
    }
}
