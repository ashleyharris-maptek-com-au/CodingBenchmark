use std::collections::BinaryHeap;
use std::io::{self, Read, Write};

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, idx: 0 }
    }

    fn next_usize(&mut self) -> usize {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        let mut value = 0usize;
        while self.idx < self.buf.len() {
            let b = self.buf[self.idx];
            if !b.is_ascii_digit() {
                break;
            }
            value = value * 10 + (b - b'0') as usize;
            self.idx += 1;
        }
        value
    }
}

fn cover_vertex(
    x: usize,
    adj: &[Vec<usize>],
    uncovered: &mut [bool],
    score: &mut [usize],
    deg: &[usize],
    heap: &mut BinaryHeap<(usize, usize, usize)>,
    remaining: &mut usize,
) {
    if !uncovered[x] {
        return;
    }
    uncovered[x] = false;
    *remaining -= 1;

    for &y in &adj[x] {
        if uncovered[y] {
            score[y] -= 1;
            heap.push((score[y], deg[y], y));
        }
    }
}

fn main() {
    let mut sc = Scanner::new();
    if sc.buf.is_empty() {
        return;
    }

    let n = sc.next_usize();
    let m = sc.next_usize();

    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let u = sc.next_usize();
        let v = sc.next_usize();
        if u != v {
            let (a, b) = if u < v { (u, v) } else { (v, u) };
            edges.push((a, b));
        }
    }

    edges.sort_unstable();
    edges.dedup();

    let mut deg = vec![0usize; n];
    for &(u, v) in &edges {
        deg[u] += 1;
        deg[v] += 1;
    }

    let mut adj: Vec<Vec<usize>> = deg.iter().map(|&d| Vec::with_capacity(d)).collect();
    for &(u, v) in &edges {
        adj[u].push(v);
        adj[v].push(u);
    }

    if n == 0 {
        let mut out = io::BufWriter::new(io::stdout());
        writeln!(out, "0").unwrap();
        writeln!(out).unwrap();
        return;
    }

    let mut uncovered = vec![true; n];
    let mut selected = vec![false; n];
    let mut score: Vec<usize> = deg.iter().map(|&d| d + 1).collect();
    let mut heap = BinaryHeap::with_capacity(n);
    for v in 0..n {
        heap.push((score[v], deg[v], v));
    }

    let mut remaining = n;
    while remaining > 0 {
        let v = loop {
            let (stored_score, stored_deg, v) = heap.pop().expect("heap unexpectedly empty");
            if !uncovered[v] {
                continue;
            }
            if score[v] != stored_score || deg[v] != stored_deg {
                heap.push((score[v], deg[v], v));
                continue;
            }
            break v;
        };

        selected[v] = true;
        cover_vertex(v, &adj, &mut uncovered, &mut score, &deg, &mut heap, &mut remaining);
        for &u in &adj[v] {
            cover_vertex(u, &adj, &mut uncovered, &mut score, &deg, &mut heap, &mut remaining);
        }
    }

    let mut dom_count = vec![0usize; n];
    for v in 0..n {
        if selected[v] {
            dom_count[v] += 1;
            for &u in &adj[v] {
                dom_count[u] += 1;
            }
        }
    }

    loop {
        let mut changed = false;
        for v in (0..n).rev() {
            if !selected[v] {
                continue;
            }
            if dom_count[v] <= 1 {
                continue;
            }
            let mut removable = true;
            for &u in &adj[v] {
                if dom_count[u] <= 1 {
                    removable = false;
                    break;
                }
            }
            if removable {
                selected[v] = false;
                dom_count[v] -= 1;
                for &u in &adj[v] {
                    dom_count[u] -= 1;
                }
                changed = true;
            }
        }
        if !changed {
            break;
        }
    }

    let mut ans = Vec::new();
    for v in 0..n {
        if selected[v] {
            ans.push(v);
        }
    }

    let mut out = io::BufWriter::new(io::stdout());
    writeln!(out, "{}", ans.len()).unwrap();
    if !ans.is_empty() {
        for (i, v) in ans.iter().enumerate() {
            if i > 0 {
                write!(out, " ").unwrap();
            }
            write!(out, "{}", v).unwrap();
        }
    }
    writeln!(out).unwrap();
}
