using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public string? Next()
        {
            int b;
            do
            {
                b = ReadByte();
                if (b < 0) return null;
            } while (b <= 32);

            var sb = new StringBuilder();
            do
            {
                sb.Append((char)b);
                b = ReadByte();
            } while (b > 32);

            return sb.ToString();
        }

        public int NextInt()
        {
            return int.Parse(Next()!);
        }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }

            return _buffer[_ptr++];
        }
    }

    private readonly struct Orientation
    {
        public readonly int Width;
        public readonly int Height;
        public readonly int[] RowMasks;

        public Orientation(int width, int height, int[] rowMasks)
        {
            Width = width;
            Height = height;
            RowMasks = rowMasks;
        }
    }

    private readonly struct Move
    {
        public readonly int Rotation;
        public readonly int Column;
        public readonly int LandingY;
        public readonly int LinesCleared;
        public readonly int MaxHeight;
        public readonly int Holes;
        public readonly int Bumpiness;
        public readonly int AggregateHeight;
        public readonly double Score;

        public Move(int rotation, int column, int landingY, int linesCleared, int maxHeight, int holes, int bumpiness, int aggregateHeight, double score)
        {
            Rotation = rotation;
            Column = column;
            LandingY = landingY;
            LinesCleared = linesCleared;
            MaxHeight = maxHeight;
            Holes = holes;
            Bumpiness = bumpiness;
            AggregateHeight = aggregateHeight;
            Score = score;
        }
    }

    private static int Width;
    private static int Height;
    private static ulong FullMask;
    private static ulong[] Board = Array.Empty<ulong>();
    private static Orientation[][] PieceOrientations = Array.Empty<Orientation[]>();

    public static void Main()
    {
        var scanner = new FastScanner();
        string? first = scanner.Next();
        if (first == null) return;

        Width = int.Parse(first);
        Height = scanner.NextInt();
        _ = scanner.NextInt(); // num_pieces is not required for online play.

        if (Width < 1 || Width > 64)
        {
            throw new NotSupportedException("This implementation supports board widths from 1 to 64.");
        }

        FullMask = Width == 64 ? ulong.MaxValue : ((1UL << Width) - 1UL);
        Board = new ulong[Height];
        PieceOrientations = BuildAllPieces();

        var output = new StringBuilder();
        var writer = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };

        while (true)
        {
            string? pieceToken = scanner.Next();
            if (pieceToken == null) break;

            char piece = char.ToUpperInvariant(pieceToken[0]);
            if (!TryGetPieceIndex(piece, out int pieceIndex))
            {
                break;
            }

            var chosen = ChooseBestMove(pieceIndex);
            output.Clear();
            output.Append(chosen.Rotation);
            output.Append(' ');
            output.Append(chosen.Column);
            writer.WriteLine(output.ToString());
            writer.Flush();

            string? result = scanner.Next();
            if (result == null) break;
            if (result.Equals("gameover", StringComparison.OrdinalIgnoreCase))
            {
                break;
            }

            ApplyMove(pieceIndex, chosen.Rotation, chosen.Column, chosen.LandingY);
        }
    }

    private static bool TryGetPieceIndex(char piece, out int index)
    {
        switch (piece)
        {
            case 'I': index = 0; return true;
            case 'O': index = 1; return true;
            case 'T': index = 2; return true;
            case 'S': index = 3; return true;
            case 'Z': index = 4; return true;
            case 'J': index = 5; return true;
            case 'L': index = 6; return true;
            default: index = -1; return false;
        }
    }

    private static Orientation[][] BuildAllPieces()
    {
        return new[]
        {
            BuildPiece(new[] { (0, 1), (1, 1), (2, 1), (3, 1) }), // I
            BuildPiece(new[] { (0, 0), (1, 0), (0, 1), (1, 1) }), // O
            BuildPiece(new[] { (0, 0), (1, 0), (2, 0), (1, 1) }), // T
            BuildPiece(new[] { (1, 0), (2, 0), (0, 1), (1, 1) }), // S
            BuildPiece(new[] { (0, 0), (1, 0), (1, 1), (2, 1) }), // Z
            BuildPiece(new[] { (0, 0), (0, 1), (0, 2), (1, 2) }), // J
            BuildPiece(new[] { (1, 0), (1, 1), (1, 2), (0, 2) })  // L
        };
    }

    private static Orientation[] BuildPiece((int x, int y)[] baseCells)
    {
        var result = new Orientation[4];
        var current = new List<(int x, int y)>(baseCells);

        for (int rot = 0; rot < 4; rot++)
        {
            int minX = int.MaxValue;
            int minY = int.MaxValue;
            int maxX = int.MinValue;
            int maxY = int.MinValue;
            for (int i = 0; i < current.Count; i++)
            {
                var c = current[i];
                if (c.x < minX) minX = c.x;
                if (c.y < minY) minY = c.y;
                if (c.x > maxX) maxX = c.x;
                if (c.y > maxY) maxY = c.y;
            }

            int width = maxX - minX + 1;
            int height = maxY - minY + 1;
            int[] masks = new int[height];
            for (int i = 0; i < current.Count; i++)
            {
                var c = current[i];
                int x = c.x - minX;
                int y = c.y - minY;
                masks[y] |= 1 << x;
            }

            result[rot] = new Orientation(width, height, masks);

            var next = new List<(int x, int y)>(current.Count);
            for (int i = 0; i < current.Count; i++)
            {
                var c = current[i];
                next.Add((3 - c.y, c.x));
            }
            current = next;
        }

        return result;
    }

    private static Move ChooseBestMove(int pieceIndex)
    {
        var orientations = PieceOrientations[pieceIndex];
        Move best = default;
        bool found = false;

        int occupiedCells = 0;
        for (int y = 0; y < Height; y++)
        {
            occupiedCells += PopCount(Board[y]);
        }

        AnalyzeBoard(Board, out int[] currentHeights, out int currentMaxHeight, out int currentHoles, out int currentBumpiness, out int currentAggregateHeight);
        double fillRatio = occupiedCells / (double)(Width * Height);
        double danger = Math.Max(currentMaxHeight / (double)Height, currentHoles / (double)(Width + Height));
        double aggressive = ComputeAggression(fillRatio, currentMaxHeight, currentHoles);

        for (int rot = 0; rot < 4; rot++)
        {
            var o = orientations[rot];
            if (o.Width > Width) continue;

            for (int col = 0; col <= Width - o.Width; col++)
            {
                int landingY = FindLandingY(Board, o, col);
                if (landingY < 0) continue;

                var sim = SimulateMove(Board, o, col, landingY);
                if (sim.linesCleared < 0) continue;

                double score = ScorePosition(
                    sim.linesCleared,
                    sim.maxHeight,
                    sim.holes,
                    sim.bumpiness,
                    sim.aggregateHeight,
                    fillRatio,
                    danger,
                    aggressive,
                    currentMaxHeight,
                    currentHoles,
                    currentBumpiness,
                    currentAggregateHeight);

                var candidate = new Move(rot, col, landingY, sim.linesCleared, sim.maxHeight, sim.holes, sim.bumpiness, sim.aggregateHeight, score);

                if (!found || IsBetter(candidate, best, aggressive))
                {
                    best = candidate;
                    found = true;
                }
            }
        }

        if (!found)
        {
            // Fallback: choose the first legal move, or the least bad top-out move.
            for (int rot = 0; rot < 4 && !found; rot++)
            {
                var o = orientations[rot];
                if (o.Width > Width) continue;
                for (int col = 0; col <= Width - o.Width; col++)
                {
                    int landingY = FindLandingY(Board, o, col);
                    if (landingY >= 0)
                    {
                        best = new Move(rot, col, landingY, 0, 0, 0, 0, 0, double.NegativeInfinity);
                        found = true;
                        break;
                    }
                }
            }
        }

        return best;
    }

    private static bool IsBetter(Move a, Move b, double aggression)
    {
        if (a.Score > b.Score + 1e-9) return true;
        if (a.Score < b.Score - 1e-9) return false;

        if (a.LinesCleared != b.LinesCleared)
        {
            return a.LinesCleared > b.LinesCleared;
        }

        if (aggression > 0.5)
        {
            if (a.MaxHeight != b.MaxHeight) return a.MaxHeight < b.MaxHeight;
            if (a.Holes != b.Holes) return a.Holes < b.Holes;
        }
        else
        {
            if (a.Holes != b.Holes) return a.Holes < b.Holes;
            if (a.Bumpiness != b.Bumpiness) return a.Bumpiness < b.Bumpiness;
        }

        if (a.AggregateHeight != b.AggregateHeight) return a.AggregateHeight < b.AggregateHeight;
        if (a.Column != b.Column) return a.Column < b.Column;
        return a.Rotation < b.Rotation;
    }

    private static double ComputeAggression(double fillRatio, int currentMaxHeight, int currentHoles)
    {
        double phase = fillRatio * 0.8 + currentMaxHeight / (double)Height * 0.9 + currentHoles / (double)Math.Max(1, Width);
        if (phase < 0) phase = 0;
        if (phase > 1) phase = 1;
        return phase;
    }

    private static double ScorePosition(
        int linesCleared,
        int maxHeight,
        int holes,
        int bumpiness,
        int aggregateHeight,
        double fillRatio,
        double danger,
        double aggression,
        int currentMaxHeight,
        int currentHoles,
        int currentBumpiness,
        int currentAggregateHeight)
    {
        // Phase 1: early / stable board. Prefer a flat surface and low hole creation.
        double early =
            linesCleared * 3.4
            - aggregateHeight * 0.42
            - holes * 2.65
            - bumpiness * 0.18
            - maxHeight * 0.25;

        // Phase 2: danger mode. Collapse height aggressively and value line clears more.
        double dangerMode =
            linesCleared * 12.0
            - aggregateHeight * 0.55
            - holes * 6.5
            - bumpiness * 0.12
            - maxHeight * 1.55;

        // Blend the two based on actual board state and current complexity.
        double blend = Clamp01(0.55 * aggression + 0.45 * danger);
        double score = early * (1.0 - blend) + dangerMode * blend;

        // Prefer moves that reduce the current risk profile.
        score += Math.Max(0, currentMaxHeight - maxHeight) * 0.35;
        score += Math.Max(0, currentHoles - holes) * 1.1;
        score += Math.Max(0, currentBumpiness - bumpiness) * 0.07;

        // Mild reward for using more of the board without making it ragged.
        score += (1.0 - fillRatio) * 0.05;

        return score;
    }

    private static double Clamp01(double v)
    {
        if (v < 0) return 0;
        if (v > 1) return 1;
        return v;
    }

    private static int FindLandingY(ulong[] board, Orientation o, int x)
    {
        int maxY = Height - o.Height;
        if (maxY < 0) return -1;

        int y = 0;
        while (y <= maxY && !Collides(board, o, x, y))
        {
            y++;
        }

        y--;
        return y >= 0 ? y : -1;
    }

    private static bool Collides(ulong[] board, Orientation o, int x, int y)
    {
        for (int r = 0; r < o.Height; r++)
        {
            ulong mask = (ulong)o.RowMasks[r] << x;
            if ((board[y + r] & mask) != 0)
            {
                return true;
            }
        }

        return false;
    }

    private static (int linesCleared, int maxHeight, int holes, int bumpiness, int aggregateHeight) SimulateMove(ulong[] board, Orientation o, int x, int y)
    {
        ulong[] next = (ulong[])board.Clone();
        for (int r = 0; r < o.Height; r++)
        {
            next[y + r] |= (ulong)o.RowMasks[r] << x;
        }

        int write = 0;
        int cleared = 0;
        for (int read = 0; read < Height; read++)
        {
            if (next[read] != FullMask)
            {
                next[write++] = next[read];
            }
            else
            {
                cleared++;
            }
        }
        while (write < Height)
        {
            next[write++] = 0;
        }

        AnalyzeBoard(next, out _, out int maxHeight, out int holes, out int bumpiness, out int aggregateHeight);
        return (cleared, maxHeight, holes, bumpiness, aggregateHeight);
    }

    private static void ApplyMove(int pieceIndex, int rotation, int column, int landingY)
    {
        var o = PieceOrientations[pieceIndex][rotation];
        for (int r = 0; r < o.Height; r++)
        {
            Board[landingY + r] |= (ulong)o.RowMasks[r] << column;
        }

        int write = 0;
        for (int read = 0; read < Height; read++)
        {
            if (Board[read] != FullMask)
            {
                Board[write++] = Board[read];
            }
            else
            {
                // Cleared rows are dropped by not copying them forward.
            }
        }
        while (write < Height)
        {
            Board[write++] = 0;
        }
    }

    private static void AnalyzeBoard(ulong[] board, out int[] heights, out int maxHeight, out int holes, out int bumpiness, out int aggregateHeight)
    {
        heights = new int[Width];
        holes = 0;
        aggregateHeight = 0;
        maxHeight = 0;

        for (int x = 0; x < Width; x++)
        {
            int h = 0;
            bool seenBlock = false;
            for (int y = Height - 1; y >= 0; y--)
            {
                bool occupied = ((board[y] >> x) & 1UL) != 0;
                if (occupied)
                {
                    if (!seenBlock)
                    {
                        h = y + 1;
                        seenBlock = true;
                    }
                }
                else if (seenBlock)
                {
                    holes++;
                }
            }
            heights[x] = h;
            aggregateHeight += h;
            if (h > maxHeight) maxHeight = h;
        }

        bumpiness = 0;
        for (int x = 0; x + 1 < Width; x++)
        {
            bumpiness += Math.Abs(heights[x] - heights[x + 1]);
        }
    }

    private static int PopCount(ulong value)
    {
        int count = 0;
        while (value != 0)
        {
            value &= value - 1;
            count++;
        }
        return count;
    }
}
