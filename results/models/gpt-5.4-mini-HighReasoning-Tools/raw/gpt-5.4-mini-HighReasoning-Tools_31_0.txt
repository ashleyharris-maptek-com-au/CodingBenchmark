use std::cmp::Reverse;
use std::convert::TryFrom;
use std::collections::HashMap;
use std::hash::{BuildHasherDefault, Hasher};
use std::io::{self, Read};

const BITSET_LIMIT: usize = 16_000_000;

#[derive(Clone, Copy)]
struct Item {
    value: i128,
    index: usize,
}

#[derive(Clone, Copy)]
struct Node {
    prev: usize,
    item_idx: usize,
}

#[derive(Default)]
struct FastHasher(u64);

impl Hasher for FastHasher {
    fn finish(&self) -> u64 {
        self.0
    }

    fn write(&mut self, bytes: &[u8]) {
        let mut hash = if self.0 == 0 {
            0xcbf2_9ce4_8422_2325u64
        } else {
            self.0
        };
        for &b in bytes {
            hash ^= u64::from(b);
            hash = hash.wrapping_mul(0x1000_0000_01b3);
            hash ^= hash >> 32;
        }
        self.0 = hash;
    }
}

type FastMap<K, V> = HashMap<K, V, BuildHasherDefault<FastHasher>>;

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> io::Result<Self> {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf)?;
        Ok(Self { buf, idx: 0 })
    }

    fn next_i128(&mut self) -> Option<i128> {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let mut sign = 1i128;
        if self.buf[self.idx] == b'-' {
            sign = -1;
            self.idx += 1;
        } else if self.buf[self.idx] == b'+' {
            self.idx += 1;
        }
        let mut value = 0i128;
        let mut found = false;
        while self.idx < self.buf.len() {
            let c = self.buf[self.idx];
            if !c.is_ascii_digit() {
                break;
            }
            found = true;
            value = value
                .saturating_mul(10)
                .saturating_add(i128::from(c - b'0'));
            self.idx += 1;
        }
        if found {
            Some(value * sign)
        } else {
            None
        }
    }
}

fn abs_i128(x: i128) -> i128 {
    x.checked_abs().unwrap_or(i128::MAX)
}

fn gcd_i128(mut a: i128, mut b: i128) -> i128 {
    a = abs_i128(a);
    b = abs_i128(b);
    while b != 0 {
        let r = a % b;
        a = b;
        b = r;
    }
    a
}

fn can_reach(sum: i128, rem_min: i128, rem_max: i128, target: i128) -> bool {
    target >= sum + rem_min && target <= sum + rem_max
}

fn reconstruct(mut node_idx: usize, arena: &[Node]) -> Vec<usize> {
    let mut result = Vec::new();
    while let Some(node) = arena.get(node_idx) {
        if node.prev == usize::MAX {
            break;
        }
        result.push(node.item_idx);
        node_idx = node.prev;
    }
    result.reverse();
    result
}

fn write_result(indices: &[usize]) {
    let mut out = String::new();
    out.push_str("YES\n");
    out.push('[');
    for (i, idx) in indices.iter().enumerate() {
        if i > 0 {
            out.push(' ');
        }
        out.push_str(&idx.to_string());
    }
    out.push_str("]\n");
    print!("{out}");
}

fn write_no() {
    print!("NO\n");
}

fn mask_for_len(bits: usize) -> u64 {
    let rem = bits % 64;
    if rem == 0 {
        !0u64
    } else {
        (1u64 << rem) - 1
    }
}

fn shift_left_into(src: &[u64], dst: &mut [u64], shift: usize, bits: usize) {
    dst.fill(0);
    if shift >= bits {
        return;
    }
    let words = src.len();
    let word_shift = shift / 64;
    let bit_shift = shift % 64;
    for i in 0..words {
        let w = src[i];
        if w == 0 {
            continue;
        }
        let j = i + word_shift;
        if j < words {
            dst[j] |= w << bit_shift;
        }
        if bit_shift != 0 && j + 1 < words {
            dst[j + 1] |= w >> (64 - bit_shift);
        }
    }
    if !bits.is_power_of_two() || bits % 64 != 0 {
        dst[words - 1] &= mask_for_len(bits);
    }
}

fn shift_right_into(src: &[u64], dst: &mut [u64], shift: usize, bits: usize) {
    dst.fill(0);
    if shift >= bits {
        return;
    }
    let words = src.len();
    let word_shift = shift / 64;
    let bit_shift = shift % 64;
    for i in word_shift..words {
        let w = src[i];
        if w == 0 {
            continue;
        }
        let j = i - word_shift;
        dst[j] |= w >> bit_shift;
        if bit_shift != 0 && j > 0 {
            dst[j - 1] |= w << (64 - bit_shift);
        }
    }
    if bits % 64 != 0 {
        dst[words - 1] &= mask_for_len(bits);
    }
}

fn bitset_positive_solver(items: &[Item], target: i128) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }
    let target_usize = usize::try_from(target).ok()?;
    if target_usize > BITSET_LIMIT {
        return None;
    }
    let bits = target_usize + 1;
    let words = bits.div_ceil(64);
    let mut reachable = vec![0u64; words];
    reachable[0] = 1;
    if bits % 64 != 0 {
        reachable[words - 1] &= mask_for_len(bits);
    }
    let mut shifted = vec![0u64; words];
    let mut parent_prev = vec![u32::MAX; bits];
    let mut parent_pick = vec![u32::MAX; bits];

    for item in items {
        let value = usize::try_from(item.value).ok()?;
        if value == 0 || value > target_usize {
            continue;
        }
        shift_left_into(&reachable, &mut shifted, value, bits);
        let mut new_bits = vec![0u64; words];
        for i in 0..words {
            new_bits[i] = shifted[i] & !reachable[i];
        }
        for (word_idx, mut word) in new_bits.iter().copied().enumerate() {
            while word != 0 {
                let bit = word.trailing_zeros() as usize;
                let idx = word_idx * 64 + bit;
                if idx < bits && parent_prev[idx] == u32::MAX {
                    let prev_idx = idx - value;
                    parent_prev[idx] = prev_idx as u32;
                    parent_pick[idx] = item.index as u32;
                }
                word &= word - 1;
            }
        }
        for i in 0..words {
            reachable[i] |= shifted[i];
        }
        if (reachable[target_usize / 64] >> (target_usize % 64)) & 1 != 0 {
            let mut cur = target_usize;
            let mut indices = Vec::new();
            while cur != 0 {
                let prev = parent_prev[cur];
                let pick = parent_pick[cur];
                if prev == u32::MAX || pick == u32::MAX {
                    return None;
                }
                indices.push(pick as usize);
                cur = prev as usize;
            }
            indices.reverse();
            return Some(indices);
        }
    }
    None
}

fn bitset_range_solver(items: &[Item], target: i128, min_sum: i128, max_sum: i128) -> Option<Vec<usize>> {
    if target < min_sum || target > max_sum {
        return None;
    }
    let bits_i128 = max_sum - min_sum + 1;
    let bits = usize::try_from(bits_i128).ok()?;
    if bits == 0 || bits > BITSET_LIMIT {
        return None;
    }
    let target_idx = usize::try_from(target - min_sum).ok()?;
    let offset_idx = usize::try_from(-min_sum).ok()?;
    let words = bits.div_ceil(64);
    let mut reachable = vec![0u64; words];
    reachable[offset_idx / 64] |= 1u64 << (offset_idx % 64);
    if bits % 64 != 0 {
        reachable[words - 1] &= mask_for_len(bits);
    }
    let mut shifted = vec![0u64; words];
    let mut parent_prev = vec![u32::MAX; bits];
    let mut parent_pick = vec![u32::MAX; bits];

    for item in items {
        let value = item.value;
        if value == 0 {
            continue;
        }
        let abs = usize::try_from(abs_i128(value)).ok()?;
        if value > 0 {
            shift_left_into(&reachable, &mut shifted, abs, bits);
        } else {
            shift_right_into(&reachable, &mut shifted, abs, bits);
        }
        let mut new_bits = vec![0u64; words];
        for i in 0..words {
            new_bits[i] = shifted[i] & !reachable[i];
        }
        for (word_idx, mut word) in new_bits.iter().copied().enumerate() {
            while word != 0 {
                let bit = word.trailing_zeros() as usize;
                let idx = word_idx * 64 + bit;
                if idx < bits && parent_prev[idx] == u32::MAX {
                    let cur_sum = min_sum + idx as i128;
                    let prev_sum = cur_sum - value;
                    let prev_idx = usize::try_from(prev_sum - min_sum).ok()?;
                    parent_prev[idx] = prev_idx as u32;
                    parent_pick[idx] = item.index as u32;
                }
                word &= word - 1;
            }
        }
        for i in 0..words {
            reachable[i] |= shifted[i];
        }
        if (reachable[target_idx / 64] >> (target_idx % 64)) & 1 != 0 {
            let mut cur = target_idx;
            let mut indices = Vec::new();
            while cur != offset_idx {
                let prev = parent_prev[cur];
                let pick = parent_pick[cur];
                if prev == u32::MAX || pick == u32::MAX {
                    return None;
                }
                indices.push(pick as usize);
                cur = prev as usize;
            }
            indices.reverse();
            return Some(indices);
        }
    }
    if (reachable[target_idx / 64] >> (target_idx % 64)) & 1 != 0 {
        let mut cur = target_idx;
        let mut indices = Vec::new();
        while cur != offset_idx {
            let prev = parent_prev[cur];
            let pick = parent_pick[cur];
            if prev == u32::MAX || pick == u32::MAX {
                return None;
            }
            indices.push(pick as usize);
            cur = prev as usize;
        }
        indices.reverse();
        return Some(indices);
    }
    None
}

fn sparse_solver(mut items: Vec<Item>, target: i128) -> Option<Vec<usize>> {
    items.sort_by_key(|it| Reverse(abs_i128(it.value)));
    let n = items.len();
    let mut suffix_min = vec![0i128; n + 1];
    let mut suffix_max = vec![0i128; n + 1];
    for i in (0..n).rev() {
        let v = items[i].value;
        suffix_min[i] = suffix_min[i + 1] + if v < 0 { v } else { 0 };
        suffix_max[i] = suffix_max[i + 1] + if v > 0 { v } else { 0 };
    }
    if !can_reach(0, suffix_min[0], suffix_max[0], target) {
        return None;
    }

    let mut arena = Vec::with_capacity(n + 1);
    arena.push(Node {
        prev: usize::MAX,
        item_idx: usize::MAX,
    });
    let mut current: FastMap<i128, usize> = HashMap::with_hasher(BuildHasherDefault::<FastHasher>::default());
    current.insert(0, 0);

    for i in 0..n {
        let rem_min = suffix_min[i + 1];
        let rem_max = suffix_max[i + 1];
        let item = items[i];
        let mut next: FastMap<i128, usize> =
            HashMap::with_capacity_and_hasher(current.len().saturating_mul(2).saturating_add(1), BuildHasherDefault::<FastHasher>::default());
        for (&sum, &node_idx) in &current {
            if can_reach(sum, rem_min, rem_max, target) {
                next.entry(sum).or_insert(node_idx);
            }
            let new_sum = sum + item.value;
            if can_reach(new_sum, rem_min, rem_max, target) {
                let new_node = arena.len();
                arena.push(Node {
                    prev: node_idx,
                    item_idx: item.index,
                });
                next.entry(new_sum).or_insert(new_node);
            }
        }
        if let Some(&idx) = next.get(&target) {
            return Some(reconstruct(idx, &arena));
        }
        if next.is_empty() {
            return None;
        }
        current = next;
    }
    current.get(&target).copied().map(|idx| reconstruct(idx, &arena))
}

fn mitm_solver(items: &[Item], target: i128) -> Option<Vec<usize>> {
    let n = items.len();
    let mid = n / 2;
    let left = &items[..mid];
    let right = &items[mid..];

    let mut left_sums: Vec<(i128, u32)> = vec![(0, 0)];
    for (i, item) in left.iter().enumerate() {
        let bit = 1u32 << i;
        let cur_len = left_sums.len();
        left_sums.reserve(cur_len);
        for j in 0..cur_len {
            let (sum, mask) = left_sums[j];
            left_sums.push((sum + item.value, mask | bit));
        }
    }

    let mut right_sums: Vec<(i128, u32)> = vec![(0, 0)];
    for (i, item) in right.iter().enumerate() {
        let bit = 1u32 << i;
        let cur_len = right_sums.len();
        right_sums.reserve(cur_len);
        for j in 0..cur_len {
            let (sum, mask) = right_sums[j];
            right_sums.push((sum + item.value, mask | bit));
        }
    }

    left_sums.sort_unstable_by_key(|x| x.0);
    right_sums.sort_unstable_by_key(|x| x.0);

    let mut i = 0usize;
    let mut j = right_sums.len().checked_sub(1)?;
    while i < left_sums.len() {
        let sum = left_sums[i].0 + right_sums[j].0;
        if sum == target {
            let mut result = Vec::new();
            let lmask = left_sums[i].1;
            for b in 0..left.len() {
                if (lmask >> b) & 1 != 0 {
                    result.push(left[b].index);
                }
            }
            let rmask = right_sums[j].1;
            for b in 0..right.len() {
                if (rmask >> b) & 1 != 0 {
                    result.push(right[b].index);
                }
            }
            result.sort_unstable();
            return Some(result);
        } else if sum < target {
            i += 1;
        } else if j == 0 {
            break;
        } else {
            j -= 1;
        }
    }
    None
}

fn solve(items: Vec<Item>, target: i128) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }
    if items.is_empty() {
        return None;
    }

    let mut gcd = 0i128;
    for item in &items {
        gcd = gcd_i128(gcd, item.value);
    }
    if gcd != 0 && target % gcd != 0 {
        return None;
    }

    if items.len() <= 46 {
        if let Some(ans) = mitm_solver(&items, target) {
            return Some(ans);
        }
    }

    let all_nonnegative = items.iter().all(|x| x.value >= 0);
    let all_nonpositive = items.iter().all(|x| x.value <= 0);

    if all_nonnegative {
        if target < 0 {
            return None;
        }
        let mut filtered = Vec::with_capacity(items.len());
        let mut sum = 0i128;
        for item in items {
            if item.value == 0 {
                continue;
            }
            if item.value == target {
                return Some(vec![item.index]);
            }
            if item.value < target {
                sum += item.value;
                filtered.push(item);
            }
        }
        if sum < target {
            return None;
        }
        if let Ok(t) = usize::try_from(target) {
            if t <= BITSET_LIMIT {
                if let Some(ans) = bitset_positive_solver(&filtered, target) {
                    return Some(ans);
                }
            }
        }
        return sparse_solver(filtered, target);
    }

    if all_nonpositive {
        if target > 0 {
            return None;
        }
        let mut transformed = Vec::with_capacity(items.len());
        let mut sum = 0i128;
        let neg_target = -target;
        for item in items {
            if item.value == 0 {
                continue;
            }
            let v = -item.value;
            if v == neg_target {
                return Some(vec![item.index]);
            }
            if v < neg_target {
                sum += v;
                transformed.push(Item {
                    value: v,
                    index: item.index,
                });
            }
        }
        if sum < neg_target {
            return None;
        }
        if let Ok(t) = usize::try_from(neg_target) {
            if t <= BITSET_LIMIT {
                if let Some(ans) = bitset_positive_solver(&transformed, neg_target) {
                    return Some(ans.into_iter().collect());
                }
            }
        }
        let mut result = sparse_solver(transformed, neg_target)?;
        return Some(result.drain(..).collect());
    }

    let mut min_sum = 0i128;
    let mut max_sum = 0i128;
    for item in &items {
        if item.value < 0 {
            min_sum += item.value;
        } else if item.value > 0 {
            max_sum += item.value;
        }
    }
    if target < min_sum || target > max_sum {
        return None;
    }
    let range = max_sum - min_sum + 1;
    if let Ok(range_usize) = usize::try_from(range) {
        if range_usize <= BITSET_LIMIT {
            if let Some(ans) = bitset_range_solver(&items, target, min_sum, max_sum) {
                return Some(ans);
            }
        }
    }
    sparse_solver(items, target)
}

fn main() {
    let mut scanner = match Scanner::new() {
        Ok(s) => s,
        Err(_) => {
            write_no();
            return;
        }
    };
    let n = match scanner.next_i128() {
        Some(v) if v >= 0 => v as usize,
        _ => {
            write_no();
            return;
        }
    };
    let target = match scanner.next_i128() {
        Some(v) => v,
        None => {
            write_no();
            return;
        }
    };

    let mut items = Vec::with_capacity(n);
    for idx in 0..n {
        match scanner.next_i128() {
            Some(v) => items.push(Item { value: v, index: idx }),
            None => break,
        }
    }

    if target == 0 {
        write_result(&[]);
        return;
    }

    if let Some(ans) = items.iter().find(|it| it.value == target) {
        write_result(&[ans.index]);
        return;
    }

    match solve(items, target) {
        Some(mut ans) => {
            ans.sort_unstable();
            write_result(&ans);
        }
        None => write_no(),
    }
}
