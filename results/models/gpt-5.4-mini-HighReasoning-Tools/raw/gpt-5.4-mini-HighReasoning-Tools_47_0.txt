@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write>  result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>              params   : vec4<u32>;

fn rotl(x : u32, s : u32) -> u32 {
    return (x << s) | (x >> (32u - s));
}

fn quarter_round(a : ptr<function, u32>,
                 b : ptr<function, u32>,
                 c : ptr<function, u32>,
                 d : ptr<function, u32>) {
    *a = *a + *b;
    *d = rotl(*d ^ *a, 16u);
    *c = *c + *d;
    *b = rotl(*b ^ *c, 12u);
    *a = *a + *b;
    *d = rotl(*d ^ *a, 8u);
    *c = *c + *d;
    *b = rotl(*b ^ *c, 7u);
}

fn ctz32(x : u32) -> u32 {
    if (x == 0u) {
        return 32u;
    }

    var n : u32 = 0u;
    var v : u32 = x;

    if ((v & 0x0000FFFFu) == 0u) {
        n = n + 16u;
        v = v >> 16u;
    }
    if ((v & 0x000000FFu) == 0u) {
        n = n + 8u;
        v = v >> 8u;
    }
    if ((v & 0x0000000Fu) == 0u) {
        n = n + 4u;
        v = v >> 4u;
    }
    if ((v & 0x00000003u) == 0u) {
        n = n + 2u;
        v = v >> 2u;
    }
    if ((v & 0x00000001u) == 0u) {
        n = n + 1u;
    }

    return n;
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    if (gid.x >= params.z) {
        return;
    }

    let nonce : u32 = params.y + gid.x;

    var state : array<u32, 16>;
    for (var i : u32 = 0u; i < 16u; i = i + 1u) {
        state[i] = baseData[i];
    }

    for (var i : u32 = 0u; i < 16u; i = i + 1u) {
        state[i] = state[i] ^ (nonce + i);
    }

    // Round 0
    quarter_round(&state[0], &state[1], &state[2], &state[3]);
    quarter_round(&state[4], &state[5], &state[6], &state[7]);
    quarter_round(&state[8], &state[9], &state[10], &state[11]);
    quarter_round(&state[12], &state[13], &state[14], &state[15]);

    // Round 1
    quarter_round(&state[0], &state[1], &state[2], &state[3]);
    quarter_round(&state[4], &state[5], &state[6], &state[7]);
    quarter_round(&state[8], &state[9], &state[10], &state[11]);
    quarter_round(&state[12], &state[13], &state[14], &state[15]);

    // Round 2
    quarter_round(&state[0], &state[1], &state[2], &state[3]);
    quarter_round(&state[4], &state[5], &state[6], &state[7]);
    quarter_round(&state[8], &state[9], &state[10], &state[11]);
    quarter_round(&state[12], &state[13], &state[14], &state[15]);

    // Round 3
    quarter_round(&state[0], &state[1], &state[2], &state[3]);
    quarter_round(&state[4], &state[5], &state[6], &state[7]);
    quarter_round(&state[8], &state[9], &state[10], &state[11]);
    quarter_round(&state[12], &state[13], &state[14], &state[15]);

    // Round 4
    quarter_round(&state[0], &state[1], &state[2], &state[3]);
    quarter_round(&state[4], &state[5], &state[6], &state[7]);
    quarter_round(&state[8], &state[9], &state[10], &state[11]);
    quarter_round(&state[12], &state[13], &state[14], &state[15]);

    // Round 5
    quarter_round(&state[0], &state[1], &state[2], &state[3]);
    quarter_round(&state[4], &state[5], &state[6], &state[7]);
    quarter_round(&state[8], &state[9], &state[10], &state[11]);
    quarter_round(&state[12], &state[13], &state[14], &state[15]);

    // Round 6
    quarter_round(&state[0], &state[1], &state[2], &state[3]);
    quarter_round(&state[4], &state[5], &state[6], &state[7]);
    quarter_round(&state[8], &state[9], &state[10], &state[11]);
    quarter_round(&state[12], &state[13], &state[14], &state[15]);

    // Round 7
    quarter_round(&state[0], &state[1], &state[2], &state[3]);
    quarter_round(&state[4], &state[5], &state[6], &state[7]);
    quarter_round(&state[8], &state[9], &state[10], &state[11]);
    quarter_round(&state[12], &state[13], &state[14], &state[15]);

    let h0 : u32 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
    let h1 : u32 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];

    var trailing_zeros : u32;
    if (h0 != 0u) {
        trailing_zeros = ctz32(h0);
    } else {
        trailing_zeros = 32u + ctz32(h1);
    }

    if (trailing_zeros >= params.x) {
        result[0] = 1u;
        result[1] = nonce;
        result[2] = h0;
        result[3] = h1;
    }
}
