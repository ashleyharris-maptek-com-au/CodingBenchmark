OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450

%void = OpTypeVoid
%functype = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_r = OpConstant %float 0.04
%float_0_5 = OpConstant %float 0.5
%float_n0_5 = OpConstant %float -0.5
%float_0_8660254 = OpConstant %float 0.8660254
%float_n0_8660254 = OpConstant %float -0.8660254
%float_n1 = OpConstant %float -1
%float_0_8 = OpConstant %float 0.8
%dark_factor = OpConstantComposite %v3float %float_0_8 %float_0_8 %float_0_8

%gl_PerVertex_in = OpTypeStruct %v4float
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_arr_pv_in = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_arr_pv_in Input

%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_arr_v3_in = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_arr_v3_in Input
OpDecorate %worldPos Location 0
%normal = OpVariable %ptr_arr_v3_in Input
OpDecorate %normal Location 1
%color = OpVariable %ptr_arr_v3_in Input
OpDecorate %color Location 2

%ptr_v3_in = OpTypePointer Input %v3float
%ptr_v4_out = OpTypePointer Output %v4float
%ptr_v3_out = OpTypePointer Output %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
%ptr_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_gl_PerVertex_out Output
%outColor = OpVariable %ptr_v3_out Output
OpDecorate %outColor Location 0

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

%main = OpFunction %void None %functype
%entry = OpLabel

%worldPos_ptr = OpAccessChain %ptr_v3_in %worldPos %int_0
%worldPos_val = OpLoad %v3float %worldPos_ptr
%color_ptr = OpAccessChain %ptr_v3_in %color %int_0
%color_val = OpLoad %v3float %color_ptr

%pos_x = OpCompositeExtract %float %worldPos_val 0
%pos_y = OpCompositeExtract %float %worldPos_val 1
%pos_z = OpCompositeExtract %float %worldPos_val 2
%center_pos = OpCompositeConstruct %v4float %pos_x %pos_y %pos_z %float_1
%dark_color = OpFMul %v3float %color_val %dark_factor
%center_out_ptr = OpAccessChain %ptr_v4_out %gl_out %int_0

%e0_dx = OpFMul %float %float_r %float_1
%e0_dy = OpFMul %float %float_r %float_0
%e0_x = OpFAdd %float %pos_x %e0_dx
%e0_y = OpFAdd %float %pos_y %e0_dy
%e0_pos = OpCompositeConstruct %v4float %e0_x %e0_y %pos_z %float_1

%e1_dx = OpFMul %float %float_r %float_0_5
%e1_dy = OpFMul %float %float_r %float_0_8660254
%e1_x = OpFAdd %float %pos_x %e1_dx
%e1_y = OpFAdd %float %pos_y %e1_dy
%e1_pos = OpCompositeConstruct %v4float %e1_x %e1_y %pos_z %float_1

%e2_dx = OpFMul %float %float_r %float_n0_5
%e2_dy = OpFMul %float %float_r %float_0_8660254
%e2_x = OpFAdd %float %pos_x %e2_dx
%e2_y = OpFAdd %float %pos_y %e2_dy
%e2_pos = OpCompositeConstruct %v4float %e2_x %e2_y %pos_z %float_1

%e3_dx = OpFMul %float %float_r %float_n1
%e3_dy = OpFMul %float %float_r %float_0
%e3_x = OpFAdd %float %pos_x %e3_dx
%e3_y = OpFAdd %float %pos_y %e3_dy
%e3_pos = OpCompositeConstruct %v4float %e3_x %e3_y %pos_z %float_1

%e4_dx = OpFMul %float %float_r %float_n0_5
%e4_dy = OpFMul %float %float_r %float_n0_8660254
%e4_x = OpFAdd %float %pos_x %e4_dx
%e4_y = OpFAdd %float %pos_y %e4_dy
%e4_pos = OpCompositeConstruct %v4float %e4_x %e4_y %pos_z %float_1

%e5_dx = OpFMul %float %float_r %float_0_5
%e5_dy = OpFMul %float %float_r %float_n0_8660254
%e5_x = OpFAdd %float %pos_x %e5_dx
%e5_y = OpFAdd %float %pos_y %e5_dy
%e5_pos = OpCompositeConstruct %v4float %e5_x %e5_y %pos_z %float_1

OpStore %center_out_ptr %center_pos
OpStore %outColor %color_val
OpEmitVertex
OpStore %center_out_ptr %e0_pos
OpStore %outColor %dark_color
OpEmitVertex
OpStore %center_out_ptr %e1_pos
OpStore %outColor %dark_color
OpEmitVertex
OpEndPrimitive

OpStore %outColor %dark_color

OpStore %outColor %dark_color
OpEmitVertex
OpStore %center_out_ptr %e5_pos
OpStore %outColor %dark_color
OpEmitVertex
OpEndPrimitive

OpStore %outColor %color_val
OpEmitVertex
OpStore %center_out_ptr %e5_pos
OpStore %outColor %dark_color
OpEmitVertex
OpStore %center_out_ptr %e0_pos
OpStore %outColor %dark_color
OpEmitVertex
OpEndPrimitive
