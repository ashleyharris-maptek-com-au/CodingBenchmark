OpCapability Shader
OpCapability Geometry

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor %uboVar
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0

OpDecorate %color Location 2
OpDecorate %outColor Location 0

OpDecorate %ubo Block
OpMemberDecorate %ubo 0 Offset 0
OpMemberDecorate %ubo 1 Offset 64
OpMemberDecorate %ubo 2 Offset 128
OpMemberDecorate %ubo 3 Offset 192
OpMemberDecorate %ubo 4 Offset 208
OpMemberDecorate %ubo 5 Offset 224
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

%void = OpTypeVoid
%funct = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4

%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3

%float_033333334 = OpConstant %float 0.333333343
%float_07 = OpConstant %float 0.699999988

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_3
%_ptr_Input_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
%_ptr_Input_v4float = OpTypePointer Input %v4float
%gl_in = OpVariable %_ptr_Input_arr_gl_PerVertex_in Input

%arr_v3float = OpTypeArray %v3float %uint_3
%_ptr_Input_arr_v3float = OpTypePointer Input %arr_v3float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%worldPos = OpVariable %_ptr_Input_arr_v3float Input
%normal = OpVariable %_ptr_Input_arr_v3float Input
%color = OpVariable %_ptr_Input_arr_v3float Input

%gl_PerVertex_out = OpTypeStruct %v4float
%_ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%_ptr_Output_v4float = OpTypePointer Output %v4float
%gl_out = OpVariable %_ptr_Output_gl_PerVertex_out Output

%_ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %_ptr_Output_v3float Output

%ubo = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%_ptr_Uniform_ubo = OpTypePointer Uniform %ubo
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%uboVar = OpVariable %_ptr_Uniform_ubo Uniform

%main = OpFunction %void None %funct
%entry = OpLabel

%p0_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr

%wp0_ptr = OpAccessChain %_ptr_Input_v3float %worldPos %int_0
%n0_ptr = OpAccessChain %_ptr_Input_v3float %normal %int_0
%c0_ptr = OpAccessChain %_ptr_Input_v3float %color %int_0
%wp0 = OpLoad %v3float %wp0_ptr
%n0 = OpLoad %v3float %n0_ptr
%c0 = OpLoad %v3float %c0_ptr

%ubo_params_ptr = OpAccessChain %_ptr_Uniform_v4float %uboVar %int_5
%ubo_params = OpLoad %v4float %ubo_params_ptr

%sum01 = OpFAdd %v4float %p0 %p1
%sum012 = OpFAdd %v4float %sum01 %p2
%centroid = OpVectorTimesScalar %v4float %sum012 %float_033333334

%d0 = OpFSub %v4float %p0 %centroid
%d1 = OpFSub %v4float %p1 %centroid
%d2 = OpFSub %v4float %p2 %centroid

%s0 = OpVectorTimesScalar %v4float %d0 %float_07
%s1 = OpVectorTimesScalar %v4float %d1 %float_07
%s2 = OpVectorTimesScalar %v4float %d2 %float_07

%n0p = OpFAdd %v4float %centroid %s0
%n1p = OpFAdd %v4float %centroid %s1
%n2p = OpFAdd %v4float %centroid %s2

%outpos0 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %outpos0 %n0p
OpStore %outColor %c0
OpEmitVertex

%c1_ptr = OpAccessChain %_ptr_Input_v3float %color %int_1
%c1 = OpLoad %v3float %c1_ptr
%outpos1 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %outpos1 %n1p
OpStore %outColor %c1
OpEmitVertex

%c2_ptr = OpAccessChain %_ptr_Input_v3float %color %int_2
%c2 = OpLoad %v3float %c2_ptr
%outpos2 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %outpos2 %n2p
OpStore %outColor %c2

OpEndPrimitive
