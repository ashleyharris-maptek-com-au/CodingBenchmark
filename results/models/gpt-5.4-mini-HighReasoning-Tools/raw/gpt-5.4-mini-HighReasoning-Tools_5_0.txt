use std::io::{self, Read};
use std::time::Instant;

fn main() {
    std::thread::Builder::new()
        .stack_size(64 * 1024 * 1024)
        .spawn(|| solve())
        .unwrap()
        .join()
        .unwrap();
}

fn solve() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let w: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let h: usize = it.next().unwrap().parse().unwrap();
    let obstacle_count: usize = it.next().unwrap().parse().unwrap();
    let require_cycle: bool = it.next().unwrap().parse::<i32>().unwrap() != 0;

    let n = w * h;
    let mut blocked = vec![false; n];
    for _ in 0..obstacle_count {
        let x: usize = it.next().unwrap().parse().unwrap();
        let y: usize = it.next().unwrap().parse().unwrap();
        if x < w && y < h {
            blocked[y * w + x] = true;
        }
    }

    let start = 0usize;
    if blocked[start] {
        println!("0");
        return;
    }

    let mut free_cells = Vec::new();
    for idx in 0..n {
        if !blocked[idx] {
            free_cells.push(idx);
        }
    }
    let total = free_cells.len();

    if total == 0 {
        println!("0");
        return;
    }

    if !connected_from_start(w, h, &blocked, start, total) {
        println!("0");
        return;
    }

    if obstacle_count == 0 {
        let ans = if require_cycle {
            open_rectangle_cycle(w, h)
        } else {
            Some(open_rectangle_path(w, h))
        };
        if let Some(path) = ans {
            emit_path(w, path);
        } else {
            println!("0");
        }
        return;
    }

    if total == 1 {
        if require_cycle {
            println!("0");
        } else {
            emit_path(w, vec![start]);
        }
        return;
    }

    let mut neighbors = vec![Vec::<usize>::new(); n];
    for y in 0..h {
        for x in 0..w {
            let idx = y * w + x;
            if blocked[idx] {
                continue;
            }
            if x > 0 {
                let j = idx - 1;
                if !blocked[j] {
                    neighbors[idx].push(j);
                }
            }
            if x + 1 < w {
                let j = idx + 1;
                if !blocked[j] {
                    neighbors[idx].push(j);
                }
            }
            if y > 0 {
                let j = idx - w;
                if !blocked[j] {
                    neighbors[idx].push(j);
                }
            }
            if y + 1 < h {
                let j = idx + w;
                if !blocked[j] {
                    neighbors[idx].push(j);
                }
            }
        }
    }

    let deadline = Instant::now() + std::time::Duration::from_secs(280);
    let mut visited = vec![false; n];
    visited[start] = true;
    let mut path = vec![start];
    let mut solver = Searcher {
        n,
        total,
        blocked,
        neighbors,
        start,
        require_cycle,
        deadline,
    };

    let found = solver.search(start, &mut visited, &mut path);
    if found {
        emit_path(w, path);
    } else {
        println!("0");
    }
}

fn emit_path(w: usize, path: Vec<usize>) {
    println!("{}", path.len());
    for idx in path {
        let x = idx % w;
        let y = idx / w;
        println!("{} {}", x, y);
    }
}

fn connected_from_start(
    w: usize,
    h: usize,
    blocked: &[bool],
    start: usize,
    total: usize,
) -> bool {
    let mut seen = vec![false; w * h];
    let mut q = std::collections::VecDeque::new();
    q.push_back(start);
    seen[start] = true;
    let mut cnt = 0usize;
    while let Some(v) = q.pop_front() {
        cnt += 1;
        let x = v % w;
        let y = v / w;
        let push = |nx: usize, ny: usize, q: &mut std::collections::VecDeque<usize>, seen: &mut [bool]| {
            let ni = ny * w + nx;
            if !blocked[ni] && !seen[ni] {
                seen[ni] = true;
                q.push_back(ni);
            }
        };
        if x > 0 {
            push(x - 1, y, &mut q, &mut seen);
        }
        if x + 1 < w {
            push(x + 1, y, &mut q, &mut seen);
        }
        if y > 0 {
            push(x, y - 1, &mut q, &mut seen);
        }
        if y + 1 < h {
            push(x, y + 1, &mut q, &mut seen);
        }
    }
    cnt == total
}

fn open_rectangle_path(w: usize, h: usize) -> Vec<usize> {
    let mut path = Vec::with_capacity(w * h);
    for y in 0..h {
        if y % 2 == 0 {
            for x in 0..w {
                path.push(y * w + x);
            }
        } else {
            for x in (0..w).rev() {
                path.push(y * w + x);
            }
        }
    }
    path
}

fn open_rectangle_cycle(w: usize, h: usize) -> Option<Vec<usize>> {
    if w == 1 || h == 1 {
        return None;
    }
    if (w * h) % 2 == 1 {
        return None;
    }
    let mut path = Vec::with_capacity(w * h);
    if h % 2 == 0 {
        for y in 0..h {
            path.push(y * w);
        }
        for y in (0..h).rev() {
            if (h - 1 - y) % 2 == 0 {
                for x in 1..w {
                    path.push(y * w + x);
                }
            } else {
                for x in (1..w).rev() {
                    path.push(y * w + x);
                }
            }
        }
        Some(path)
    } else {
        for x in 0..w {
            path.push(x);
        }
        for x in (0..w).rev() {
            if (w - 1 - x) % 2 == 0 {
                for y in 1..h {
                    path.push(y * w + x);
                }
            } else {
                for y in (1..h).rev() {
                    path.push(y * w + x);
                }
            }
        }
        Some(path)
    }
}

struct Searcher {
    n: usize,
    total: usize,
    blocked: Vec<bool>,
    neighbors: Vec<Vec<usize>>,
    start: usize,
    require_cycle: bool,
    deadline: Instant,
}

impl Searcher {
    fn search(&mut self, cur: usize, visited: &mut [bool], path: &mut Vec<usize>) -> bool {
        if Instant::now() >= self.deadline {
            return false;
        }
        if path.len() == self.total {
            if !self.require_cycle {
                return true;
            }
            return self.neighbors[cur].iter().any(|&n| n == self.start);
        }

        let mut cand = Vec::new();
        for &n in &self.neighbors[cur] {
            if !visited[n] {
                cand.push(n);
            }
        }
        cand.sort_by_key(|&n| self.score(n, visited));

        for next in cand {
            if !self.plausible(next, visited) {
                continue;
            }
            visited[next] = true;
            path.push(next);
            if self.search(next, visited, path) {
                return true;
            }
            path.pop();
            visited[next] = false;
        }
        false
    }

    fn score(&self, cell: usize, visited: &[bool]) -> usize {
        let mut s = 0usize;
        for &n in &self.neighbors[cell] {
            if !visited[n] {
                s += 1;
            }
        }
        s
    }

    fn plausible(&self, cur: usize, visited: &[bool]) -> bool {
        if Instant::now() >= self.deadline {
            return false;
        }

        let visited_count = visited.iter().filter(|&&v| v).count();
        if visited_count + 1 > self.total {
            return false;
        }
        let remaining = self.total - visited_count - 1;
        if remaining == 0 {
            if !self.require_cycle {
                return true;
            }
            return self.neighbors[cur].iter().any(|&n| n == self.start);
        }

        let mut seen = vec![false; self.n];
        let mut q = std::collections::VecDeque::new();
        q.push_back(cur);
        seen[cur] = true;
        let mut reach = 0usize;
        while let Some(v) = q.pop_front() {
            reach += 1;
            for &n in &self.neighbors[v] {
                if !visited[n] && !seen[n] {
                    seen[n] = true;
                    q.push_back(n);
                }
            }
        }
        if reach != remaining + 1 {
            return false;
        }

        let mut deg1 = 0usize;
        for idx in 0..self.n {
            if visited[idx] || self.blocked[idx] {
                continue;
            }
            let mut d = 0usize;
            for &n in &self.neighbors[idx] {
                if n == cur || !visited[n] {
                    d += 1;
                }
            }
            if d == 0 {
                return false;
            }
            if d == 1 {
                deg1 += 1;
                if deg1 > 2 {
                    return false;
                }
            }
        }

        let mut curdeg = 0usize;
        for &n in &self.neighbors[cur] {
            if !visited[n] {
                curdeg += 1;
            }
        }
        if curdeg == 0 {
            return false;
        }

        true
    }
}
