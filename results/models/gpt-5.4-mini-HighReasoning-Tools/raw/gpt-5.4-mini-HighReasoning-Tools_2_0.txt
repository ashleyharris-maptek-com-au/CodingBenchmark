using System;
using System.IO;
using System.Text;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int m = fs.NextInt();

        int halfEdgeCapacity = m * 4;
        int[] head = new int[n];
        Array.Fill(head, -1);
        int[] to = new int[halfEdgeCapacity];
        int[] next = new int[halfEdgeCapacity];
        bool[] usedCopy = new bool[m * 2];
        int halfEdgeCount = 0;

        void AddUndirectedCopy(int u, int v)
        {
            int e = halfEdgeCount;
            to[e] = v;
            next[e] = head[u];
            head[u] = e;
            to[e + 1] = u;
            next[e + 1] = head[v];
            head[v] = e + 1;
            halfEdgeCount += 2;
        }

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            _ = fs.NextInt();

            AddUndirectedCopy(u, v);
            AddUndirectedCopy(u, v);
        }

        int edgeCopies = m * 2;
        int circuitSize = edgeCopies + 1;
        int[] stack = new int[circuitSize];
        int[] circuit = new int[circuitSize];
        int[] iter = new int[n];
        Array.Copy(head, iter, n);

        int sp = 0;
        int cp = 0;
        stack[sp++] = 0;

        while (sp > 0)
        {
            int v = stack[sp - 1];
            int e = iter[v];

            while (e != -1 && usedCopy[e >> 1])
            {
                e = next[e];
            }

            iter[v] = e;

            if (e == -1)
            {
                circuit[cp++] = v;
                sp--;
            }
            else
            {
                usedCopy[e >> 1] = true;
                stack[sp++] = to[e];
                iter[v] = next[e];
            }
        }

        var sb = new StringBuilder();
        sb.Append(circuitSize);
        for (int i = cp - 1; i >= 0; i--)
        {
            sb.Append(' ');
            sb.Append(circuit[i]);
        }

        Console.Write(sb.ToString());
    }
}
