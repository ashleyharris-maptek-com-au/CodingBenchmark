import sys
from functools import lru_cache


def main():
    data = sys.stdin.read().splitlines()
    if not data:
        return

    first = data[0].split()
    if len(first) < 2:
        sys.stdout.write("NO SOLUTION\n")
        return

    U = int(first[0])
    S = int(first[1])

    lines = data[1:1 + S]
    if len(lines) < S:
        lines.extend([""] * (S - len(lines)))

    row_masks = []
    for line in lines:
        mask = 0
        for tok in line.split():
            e = int(tok)
            if 0 <= e < U:
                mask |= 1 << e
        row_masks.append(mask)

    if U == 0:
        sys.stdout.write("SOLUTION\n\n")
        return

    all_mask = (1 << U) - 1
    col_rows = [[] for _ in range(U)]
    for r, mask in enumerate(row_masks):
        if mask == 0:
            continue
        m = mask
        while m:
            b = m & -m
            c = b.bit_length() - 1
            col_rows[c].append(r)
            m ^= b

    for c in range(U):
        if not col_rows[c]:
            sys.stdout.write("NO SOLUTION\n")
            return

    sys.setrecursionlimit(1_000_000)

    @lru_cache(maxsize=None)
    def search(used):
        if used == all_mask:
            return ()

        remaining = all_mask ^ used
        best_candidates = None
        best_count = None

        m = remaining
        while m:
            b = m & -m
            c = b.bit_length() - 1
            m ^= b

            candidates = []
            for r in col_rows[c]:
                if row_masks[r] & used == 0:
                    candidates.append(r)

            cnt = len(candidates)
            if cnt == 0:
                return None
            if best_count is None or cnt < best_count:
                best_count = cnt
                best_candidates = candidates

        for r in best_candidates:
            res = search(used | row_masks[r])
            if res is not None:
                return (r,) + res
        return None

    sol = search(0)
    if sol is None:
        sys.stdout.write("NO SOLUTION\n")
        return

    sol = sorted(sol)
    sys.stdout.write("SOLUTION\n")
    sys.stdout.write(" ".join(map(str, sol)) + "\n")


if __name__ == "__main__":
    main()
