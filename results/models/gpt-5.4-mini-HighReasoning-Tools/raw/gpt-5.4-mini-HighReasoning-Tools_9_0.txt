using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    private struct Cut
    {
        public long Length;
        public int Index;
    }

    private sealed class CutComparer : IComparer<Cut>
    {
        public int Compare(Cut x, Cut y)
        {
            int cmp = y.Length.CompareTo(x.Length);
            if (cmp != 0) return cmp;
            return x.Index.CompareTo(y.Index);
        }
    }

    private struct BinKey : IComparable<BinKey>
    {
        public long Residual;
        public int BinId;

        public BinKey(long residual, int binId)
        {
            Residual = residual;
            BinId = binId;
        }

        public int CompareTo(BinKey other)
        {
            int cmp = Residual.CompareTo(other.Residual);
            if (cmp != 0) return cmp;
            return BinId.CompareTo(other.BinId);
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public long NextLong()
        {
            int c;
            do
            {
                c = ReadByte();
            } while (c <= 32 && c >= 0);

            long sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            long value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = ReadByte();
            }

            return value * sign;
        }

        public int NextInt() => (int)NextLong();
    }

    private sealed class FastOutput : IDisposable
    {
        private readonly Stream _stream = Console.OpenStandardOutput();
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _ptr;

        public void WriteChar(char c)
        {
            if (_ptr >= _buffer.Length) Flush();
            _buffer[_ptr++] = (byte)c;
        }

        public void WriteInt(int value)
        {
            WriteLong(value);
        }

        public void WriteLong(long value)
        {
            if (value == 0)
            {
                WriteChar('0');
                return;
            }

            if (value < 0)
            {
                WriteChar('-');
                value = -value;
            }

            Span<char> tmp = stackalloc char[20];
            int len = 0;
            while (value > 0)
            {
                tmp[len++] = (char)('0' + (value % 10));
                value /= 10;
            }

            for (int i = len - 1; i >= 0; i--)
            {
                WriteChar(tmp[i]);
            }
        }

        public void WriteSpace() => WriteChar(' ');
        public void WriteLine() => WriteChar('\n');

        public void Flush()
        {
            if (_ptr > 0)
            {
                _stream.Write(_buffer, 0, _ptr);
                _ptr = 0;
            }
        }

        public void Dispose()
        {
            Flush();
            _stream.Dispose();
        }
    }

    private static uint _rng = 2463534242u;
    private static int NextPriority()
    {
        _rng ^= _rng << 13;
        _rng ^= _rng >> 17;
        _rng ^= _rng << 5;
        return (int)_rng;
    }

    private static long[] _nodeKey = Array.Empty<long>();
    private static int[] _nodeLeft = Array.Empty<int>();
    private static int[] _nodeRight = Array.Empty<int>();
    private static int[] _nodePriority = Array.Empty<int>();
    private static int[] _nodeHeadBin = Array.Empty<int>();
    private static int[] _nodeTailBin = Array.Empty<int>();
    private static int[] _nodeCount = Array.Empty<int>();
    private static int _nodePool;
    private static int _root;
    private static Dictionary<long, int> _keyToNode = new Dictionary<long, int>();

    private static int[] _binNode = Array.Empty<int>();
    private static int[] _binPrev = Array.Empty<int>();
    private static int[] _binNext = Array.Empty<int>();

    private static int[] _cutHead = Array.Empty<int>();
    private static int[] _cutNext = Array.Empty<int>();

    private static int _binCount;

    private static int CreateNode(long key)
    {
        int node = ++_nodePool;
        _nodeKey[node] = key;
        _nodeLeft[node] = 0;
        _nodeRight[node] = 0;
        _nodePriority[node] = NextPriority();
        _nodeHeadBin[node] = 0;
        _nodeTailBin[node] = 0;
        _nodeCount[node] = 0;
        _keyToNode[key] = node;
        _root = Insert(_root, node);
        return node;
    }

    private static int RotateLeft(int node)
    {
        int r = _nodeRight[node];
        _nodeRight[node] = _nodeLeft[r];
        _nodeLeft[r] = node;
        return r;
    }

    private static int RotateRight(int node)
    {
        int l = _nodeLeft[node];
        _nodeLeft[node] = _nodeRight[l];
        _nodeRight[l] = node;
        return l;
    }

    private static int Insert(int root, int node)
    {
        if (root == 0) return node;
        if (_nodeKey[node] < _nodeKey[root])
        {
            _nodeLeft[root] = Insert(_nodeLeft[root], node);
            if (_nodePriority[_nodeLeft[root]] < _nodePriority[root])
            {
                root = RotateRight(root);
            }
        }
        else
        {
            _nodeRight[root] = Insert(_nodeRight[root], node);
            if (_nodePriority[_nodeRight[root]] < _nodePriority[root])
            {
                root = RotateLeft(root);
            }
        }
        return root;
    }

    private static int Merge(int left, int right)
    {
        if (left == 0) return right;
        if (right == 0) return left;

        if (_nodePriority[left] < _nodePriority[right])
        {
            _nodeRight[left] = Merge(_nodeRight[left], right);
            return left;
        }
        else
        {
            _nodeLeft[right] = Merge(left, _nodeLeft[right]);
            return right;
        }
    }

    private static int Erase(int root, long key)
    {
        if (root == 0) return 0;

        if (key < _nodeKey[root])
        {
            _nodeLeft[root] = Erase(_nodeLeft[root], key);
            return root;
        }
        if (key > _nodeKey[root])
        {
            _nodeRight[root] = Erase(_nodeRight[root], key);
            return root;
        }

        return Merge(_nodeLeft[root], _nodeRight[root]);
    }

    private static int FindLowerBound(long key)
    {
        int cur = _root;
        int best = 0;
        while (cur != 0)
        {
            if (_nodeKey[cur] >= key)
            {
                best = cur;
                cur = _nodeLeft[cur];
            }
            else
            {
                cur = _nodeRight[cur];
            }
        }
        return best;
    }

    private static void AddBinToNode(int node, int binId)
    {
        int oldHead = _nodeHeadBin[node];
        _binPrev[binId] = 0;
        _binNext[binId] = oldHead;
        if (oldHead != 0)
        {
            _binPrev[oldHead] = binId;
        }
        else
        {
            _nodeTailBin[node] = binId;
        }
        _nodeHeadBin[node] = binId;
        _nodeCount[node]++;
        _binNode[binId] = node;
    }

    private static void RemoveBinFromNode(int binId)
    {
        int node = _binNode[binId];
        int prev = _binPrev[binId];
        int next = _binNext[binId];

        if (prev != 0)
        {
            _binNext[prev] = next;
        }
        else
        {
            _nodeHeadBin[node] = next;
        }

        if (next != 0)
        {
            _binPrev[next] = prev;
        }
        else
        {
            _nodeTailBin[node] = prev;
        }

        _binPrev[binId] = 0;
        _binNext[binId] = 0;
        _binNode[binId] = 0;
        _nodeCount[node]--;

        if (_nodeCount[node] == 0)
        {
            long key = _nodeKey[node];
            _root = Erase(_root, key);
            _keyToNode.Remove(key);
        }
    }

    private static int GetOrCreateNode(long key)
    {
        if (_keyToNode.TryGetValue(key, out int node))
        {
            return node;
        }

        return CreateNode(key);
    }

    public static void Main()
    {
        var input = new FastScanner(Console.OpenStandardInput());
        int n = input.NextInt();
        long stockLength = input.NextLong();

        if (n <= 0)
        {
            using var output = new FastOutput();
            output.WriteInt(0);
            output.WriteLine();
            output.Flush();
            return;
        }

        Cut[] cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = new Cut
            {
                Length = input.NextLong(),
                Index = i
            };
        }

        Array.Sort(cuts, new CutComparer());

        _nodeKey = new long[n + 1];
        _nodeLeft = new int[n + 1];
        _nodeRight = new int[n + 1];
        _nodePriority = new int[n + 1];
        _nodeHeadBin = new int[n + 1];
        _nodeTailBin = new int[n + 1];
        _nodeCount = new int[n + 1];
        _binNode = new int[n + 1];
        _binPrev = new int[n + 1];
        _binNext = new int[n + 1];
        _cutHead = new int[n];
        _cutNext = new int[n];
        Array.Fill(_cutHead, -1);

        using var outputWriter = new FastOutput();

        for (int i = 0; i < n; i++)
        {
            long len = cuts[i].Length;
            int cutIndex = cuts[i].Index;

            if (len > stockLength)
            {
                // Input promises this does not happen, but keep the solver robust.
                continue;
            }

            int node = FindLowerBound(len);
            int binId;

            if (node == 0)
            {
                binId = _binCount++;
                _binNode[binId] = 0;
                _binPrev[binId] = 0;
                _binNext[binId] = 0;
                _cutHead[binId] = -1;

                long residual = stockLength - len;
                int targetNode = GetOrCreateNode(residual);
                AddBinToNode(targetNode, binId);
            }
            else
            {
                binId = _nodeHeadBin[node];
                long residual = _nodeKey[node];

                if (len > 0)
                {
                    RemoveBinFromNode(binId);
                    long newResidual = residual - len;
                    int targetNode = GetOrCreateNode(newResidual);
                    AddBinToNode(targetNode, binId);
                }
            }

            _cutNext[cutIndex] = _cutHead[binId];
            _cutHead[binId] = cutIndex;
        }

        outputWriter.WriteInt(_binCount);
        outputWriter.WriteLine();

        for (int binId = 0; binId < _binCount; binId++)
        {
            int cur = _cutHead[binId];
            bool first = true;
            while (cur != -1)
            {
                if (!first)
                {
                    outputWriter.WriteSpace();
                }
                outputWriter.WriteInt(cur);
                first = false;
                cur = _cutNext[cur];
            }
            outputWriter.WriteLine();
        }

        outputWriter.Flush();
    }
}
