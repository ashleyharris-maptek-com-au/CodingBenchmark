OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3
%float_008 = OpConstant %float 0.08

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_input_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
%ptr_input_v4float = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_input_arr_gl_PerVertex_in Input

%arr_v3float = OpTypeArray %v3float %uint_3
%ptr_input_arr_v3float = OpTypePointer Input %arr_v3float
%ptr_input_v3float = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_input_arr_v3float Input
%normal = OpVariable %ptr_input_arr_v3float Input
%color = OpVariable %ptr_input_arr_v3float Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_output_v4float = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_output_gl_PerVertex_out Output

%ptr_output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_output_v3float Output

%main = OpFunction %void None %func
%label = OpLabel

%p0_ptr = OpAccessChain %ptr_input_v4float %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_input_v4float %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_input_v4float %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr

%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz
%cross = OpExtInst %v3float %glsl Cross %e1 %e2
%N = OpExtInst %v3float %glsl Normalize %cross
%offset = OpVectorTimesScalar %v3float %N %float_008

%p0_off = OpFAdd %v3float %p0_xyz %offset
%p1_off = OpFAdd %v3float %p1_xyz %offset
%p2_off = OpFAdd %v3float %p2_xyz %offset

%p0_x = OpCompositeExtract %float %p0_off 0
%p0_y = OpCompositeExtract %float %p0_off 1
%p0_z = OpCompositeExtract %float %p0_off 2
%p0_w = OpCompositeExtract %float %p0 3

%p1_x = OpCompositeExtract %float %p1_off 0
%p1_y = OpCompositeExtract %float %p1_off 1
%p1_z = OpCompositeExtract %float %p1_off 2
%p1_w = OpCompositeExtract %float %p1 3

%p2_x = OpCompositeExtract %float %p2_off 0
%p2_y = OpCompositeExtract %float %p2_off 1
%p2_z = OpCompositeExtract %float %p2_off 2
%p2_w = OpCompositeExtract %float %p2 3

%p0_out = OpCompositeConstruct %v4float %p0_x %p0_y %p0_z %p0_w
%p1_out = OpCompositeConstruct %v4float %p1_x %p1_y %p1_z %p1_w
%p2_out = OpCompositeConstruct %v4float %p2_x %p2_y %p2_z %p2_w

%c0_ptr = OpAccessChain %ptr_input_v3float %color %int_0
%c1_ptr = OpAccessChain %ptr_input_v3float %color %int_1
%c2_ptr = OpAccessChain %ptr_input_v3float %color %int_2
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr
%c2 = OpLoad %v3float %c2_ptr

%gl_out_pos_ptr0 = OpAccessChain %ptr_output_v4float %gl_out %int_0
OpStore %gl_out_pos_ptr0 %p0_out
OpStore %outColor %c0
OpEmitVertex

%gl_out_pos_ptr1 = OpAccessChain %ptr_output_v4float %gl_out %int_0
OpStore %gl_out_pos_ptr1 %p1_out
OpStore %outColor %c1
OpEmitVertex

%gl_out_pos_ptr2 = OpAccessChain %ptr_output_v4float %gl_out %int_0
OpStore %gl_out_pos_ptr2 %p2_out
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
