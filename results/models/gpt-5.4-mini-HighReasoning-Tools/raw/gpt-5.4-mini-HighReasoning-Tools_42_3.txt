OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_out 0 "gl_Position"

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0
%float_thickness = OpConstant %float 0.03

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_Input_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in Input
%ptr_Input_v4float = OpTypePointer Input %v4float

%arr_v3float = OpTypeArray %v3float %uint_2
%ptr_Input_arr_v3float = OpTypePointer Input %arr_v3float
%worldPos = OpVariable %ptr_Input_arr_v3float Input
%normal = OpVariable %ptr_Input_arr_v3float Input
%color = OpVariable %ptr_Input_arr_v3float Input
%ptr_Input_v3float = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%ptr_Output_v4float = OpTypePointer Output %v4float

%ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3float Output

%main = OpFunction %void None %func
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

%c0_ptr = OpAccessChain %ptr_Input_v3float %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_Input_v3float %color %int_1
%c1 = OpLoad %v3float %c1_ptr

%p1x = OpCompositeExtract %float %p1 0
%p0x = OpCompositeExtract %float %p0 0
%dx = OpFSub %float %p1x %p0x
%p1y = OpCompositeExtract %float %p1 1
%p0y = OpCompositeExtract %float %p0 1
%dy = OpFSub %float %p1y %p0y

%negdy = OpFNegate %float %dy
%dir2 = OpCompositeConstruct %v2float %dx %dy
%perp_base = OpCompositeConstruct %v2float %negdy %dx
%perp_norm = OpExtInst %v2float %glsl Normalize %perp_base
%perp = OpVectorTimesScalar %v2float %perp_norm %float_thickness

%perp_x = OpCompositeExtract %float %perp 0
%perp_y = OpCompositeExtract %float %perp 1
%perp4 = OpCompositeConstruct %v4float %perp_x %perp_y %float_0 %float_0

%p0_plus = OpFAdd %v4float %p0 %perp4
%p0_minus = OpFSub %v4float %p0 %perp4
%p1_plus = OpFAdd %v4float %p1 %perp4
%p1_minus = OpFSub %v4float %p1 %perp4

%out_pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr %p0_plus
OpStore %outColor %c0
OpEmitVertex

OpStore %out_pos_ptr %p0_minus
OpStore %outColor %c0
OpEmitVertex

OpStore %out_pos_ptr %p1_plus
OpStore %outColor %c1
OpEmitVertex

OpStore %out_pos_ptr %p1_minus
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
