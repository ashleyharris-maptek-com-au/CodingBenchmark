const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = Array.isArray(data.probability) ? data.probability : [];
const drones = Array.isArray(data.drones) ? data.drones : [];
const maxSteps = Math.max(0, data.maxStepsPerDrone | 0);

const h = grid.length;
const w = h > 0 && Array.isArray(grid[0]) ? grid[0].length : 0;
const n = w * h;

const blocked = new Uint8Array(n);
const value = new Float64Array(n);

for (let y = 0; y < h; y++) {
  const row = grid[y] || [];
  for (let x = 0; x < w; x++) {
    const idx = y * w + x;
    const v = Number(row[x]);
    if (v === -1) {
      blocked[idx] = 1;
    } else if (v > 0) {
      value[idx] = v;
    }
  }
}

function idxOf(x, y) {
  return y * w + x;
}

function xOf(idx) {
  return idx % w;
}

function yOf(idx) {
  return (idx / w) | 0;
}

function inBounds(x, y) {
  return x >= 0 && y >= 0 && x < w && y < h;
}

function manhattan(a, b) {
  return Math.abs(xOf(a) - xOf(b)) + Math.abs(yOf(a) - yOf(b));
}

// Prefix sum of positive values for local density scoring.
const ps = new Float64Array((w + 1) * (h + 1));
for (let y = 0; y < h; y++) {
  let rowSum = 0;
  for (let x = 0; x < w; x++) {
    const idx = y * w + x;
    rowSum += value[idx];
    ps[(y + 1) * (w + 1) + (x + 1)] = ps[y * (w + 1) + (x + 1)] + rowSum;
  }
}

function rectSum(x1, y1, x2, y2) {
  if (x1 > x2 || y1 > y2) return 0;
  const A = ps[y1 * (w + 1) + x1];
  const B = ps[y1 * (w + 1) + (x2 + 1)];
  const C = ps[(y2 + 1) * (w + 1) + x1];
  const D = ps[(y2 + 1) * (w + 1) + (x2 + 1)];
  return D - B - C + A;
}

const density = new Float64Array(n);
const radius = 2;
for (let y = 0; y < h; y++) {
  const y1 = Math.max(0, y - radius);
  const y2 = Math.min(h - 1, y + radius);
  for (let x = 0; x < w; x++) {
    const x1 = Math.max(0, x - radius);
    const x2 = Math.min(w - 1, x + radius);
    density[y * w + x] = rectSum(x1, y1, x2, y2);
  }
}

const starts = drones.map((d, i) => {
  const x = d.x | 0;
  const y = d.y | 0;
  return {
    i,
    x,
    y,
    idx: idxOf(x, y),
    seedScore: inBounds(x, y) ? density[idxOf(x, y)] + value[idxOf(x, y)] * 5 : -1e18,
  };
});

// Multi-source BFS partitions reachable cells by nearest drone start.
const owner = new Int32Array(n);
owner.fill(-1);
const startDist = new Int32Array(n);
startDist.fill(-1);

const order = starts.slice().sort((a, b) => b.seedScore - a.seedScore);
const q = new Int32Array(Math.max(1, n));
let qh = 0;
let qt = 0;
for (const s of order) {
  if (!inBounds(s.x, s.y)) continue;
  if (blocked[s.idx]) continue;
  if (owner[s.idx] !== -1) continue;
  owner[s.idx] = s.i;
  startDist[s.idx] = 0;
  q[qt++] = s.idx;
}

const dx4 = [1, -1, 0, 0];
const dy4 = [0, 0, 1, -1];
while (qh < qt) {
  const cur = q[qh++];
  const curOwner = owner[cur];
  const cx = xOf(cur);
  const cy = yOf(cur);
  const nd = startDist[cur] + 1;
  for (let k = 0; k < 4; k++) {
    const nx = cx + dx4[k];
    const ny = cy + dy4[k];
    if (!inBounds(nx, ny)) continue;
    const ni = idxOf(nx, ny);
    if (blocked[ni] || owner[ni] !== -1) continue;
    owner[ni] = curOwner;
    startDist[ni] = nd;
    q[qt++] = ni;
  }
}

const claimed = new Uint8Array(n);
const localSeen = new Int32Array(n);
let localTag = 1;

const baseScore = new Float64Array(n);
const positiveCells = [];
for (let i = 0; i < n; i++) {
  if (value[i] > 0) {
    baseScore[i] = value[i] * 5 + density[i];
    positiveCells.push(i);
  }
}

const regionValue = new Float64Array(drones.length);
const regionCandidates = Array.from({ length: drones.length }, () => []);
for (const idx of positiveCells) {
  const o = owner[idx];
  if (o >= 0) {
    regionValue[o] += value[idx];
    regionCandidates[o].push(idx);
  }
}

for (let i = 0; i < regionCandidates.length; i++) {
  regionCandidates[i].sort((a, b) => baseScore[b] - baseScore[a]);
}

const planOrder = drones.map((_, i) => i).sort((a, b) => regionValue[b] - regionValue[a]);
const paths = Array.from({ length: drones.length }, () => []);

// A* search with a small binary heap.
const seen = new Int32Array(n);
const gScore = new Int32Array(n);
const parent = new Int32Array(n);
let searchTag = 1;

function pushHeap(heap, idx, g, f) {
  heap.idx.push(idx);
  heap.g.push(g);
  heap.f.push(f);
  let i = heap.idx.length - 1;
  while (i > 0) {
    const p = (i - 1) >> 1;
    if (heap.f[p] <= heap.f[i]) break;
    [heap.idx[p], heap.idx[i]] = [heap.idx[i], heap.idx[p]];
    [heap.g[p], heap.g[i]] = [heap.g[i], heap.g[p]];
    [heap.f[p], heap.f[i]] = [heap.f[i], heap.f[p]];
    i = p;
  }
}

function popHeap(heap) {
  const last = heap.idx.length - 1;
  const outIdx = heap.idx[0];
  const outG = heap.g[0];
  heap.idx[0] = heap.idx[last];
  heap.g[0] = heap.g[last];
  heap.f[0] = heap.f[last];
  heap.idx.pop();
  heap.g.pop();
  heap.f.pop();
  let i = 0;
  while (true) {
    const l = i * 2 + 1;
    const r = l + 1;
    let m = i;
    if (l < heap.idx.length && heap.f[l] < heap.f[m]) m = l;
    if (r < heap.idx.length && heap.f[r] < heap.f[m]) m = r;
    if (m === i) break;
    [heap.idx[m], heap.idx[i]] = [heap.idx[i], heap.idx[m]];
    [heap.g[m], heap.g[i]] = [heap.g[i], heap.g[m]];
    [heap.f[m], heap.f[i]] = [heap.f[i], heap.f[m]];
    i = m;
  }
  return { idx: outIdx, g: outG };
}

function reconstructPath(start, goal) {
  const rev = [];
  let cur = goal;
  while (cur !== -1) {
    rev.push(cur);
    if (cur === start) break;
    cur = parent[cur];
  }
  rev.reverse();
  return rev;
}

function findPath(start, goal, limit) {
  if (start === goal) return [start];
  searchTag++;
  if (searchTag === 0x7fffffff) {
    seen.fill(0);
    searchTag = 1;
  }
  const heap = { idx: [], g: [], f: [] };
  const sx = xOf(start);
  const sy = yOf(start);
  const gx = xOf(goal);
  const gy = yOf(goal);
  seen[start] = searchTag;
  gScore[start] = 0;
  parent[start] = -1;
  pushHeap(heap, start, 0, Math.abs(sx - gx) + Math.abs(sy - gy));

  while (heap.idx.length > 0) {
    const { idx, g } = popHeap(heap);
    if (seen[idx] !== searchTag || g !== gScore[idx]) continue;
    if (g > limit) continue;
    if (idx === goal) return reconstructPath(start, goal);
    const hx = xOf(idx);
    const hy = yOf(idx);
    const rem = limit - g;
    const man = Math.abs(hx - gx) + Math.abs(hy - gy);
    if (man > rem) continue;
    for (let k = 0; k < 4; k++) {
      const nx = hx + dx4[k];
      const ny = hy + dy4[k];
      if (!inBounds(nx, ny)) continue;
      const ni = idxOf(nx, ny);
      if (blocked[ni]) continue;
      const ng = g + 1;
      if (ng > limit) continue;
      if (seen[ni] !== searchTag || ng < gScore[ni]) {
        seen[ni] = searchTag;
        gScore[ni] = ng;
        parent[ni] = idx;
        const hCost = Math.abs(nx - gx) + Math.abs(ny - gy);
        pushHeap(heap, ni, ng, ng + hCost);
      }
    }
  }
  return null;
}

function bestApproxTargetScore(cellIdx, candidates, limit) {
  let best = 0;
  const cap = Math.min(limit, candidates.length);
  for (let i = 0; i < cap; i++) {
    const t = candidates[i];
    if (claimed[t] || localSeen[t] === localTag) continue;
    const d = manhattan(cellIdx, t);
    const s = baseScore[t] / (d + 1);
    if (s > best) best = s;
  }
  return best;
}

function chooseTarget(current, remaining, candidates) {
  const cap = Math.min(500, candidates.length);
  const picks = [];
  const pickScores = [];
  for (let i = 0; i < cap; i++) {
    const t = candidates[i];
    if (claimed[t] || localSeen[t] === localTag) continue;
    const d = manhattan(current, t);
    if (d > remaining) continue;
    const score = baseScore[t] / (d + 1);
    let pos = picks.length;
    while (pos > 0 && pickScores[pos - 1] < score) pos--;
    if (pos < 12) {
      picks.splice(pos, 0, t);
      pickScores.splice(pos, 0, score);
      if (picks.length > 12) {
        picks.pop();
        pickScores.pop();
      }
    }
  }
  for (const t of picks) {
    const path = findPath(current, t, remaining);
    if (path && path.length > 1 && path.length - 1 <= remaining) {
      return path;
    }
  }
  return null;
}

function chooseLocalMove(current, prev, droneOwner) {
  const cx = xOf(current);
  const cy = yOf(current);
  let best = current;
  let bestScore = -1e30;
  for (let k = 0; k < 5; k++) {
    const nx = k < 4 ? cx + dx4[k] : cx;
    const ny = k < 4 ? cy + dy4[k] : cy;
    if (!inBounds(nx, ny)) continue;
    const ni = idxOf(nx, ny);
    if (blocked[ni]) continue;
    let score = density[ni] * 0.03;
    if (value[ni] > 0 && !claimed[ni]) score += value[ni] * 5;
    else if (value[ni] > 0) score += value[ni] * 1.2;
    if (claimed[ni]) score -= 0.25;
    if (localSeen[ni] === localTag) score -= 1.4;
    else score += 0.5;
    if (ni === prev) score -= 1.2;
    if (ni === current) score -= 0.15;
    if (owner[ni] !== droneOwner && owner[ni] !== -1) score -= 0.08;
    score += bestApproxTargetScore(ni, regionCandidates[droneOwner], 24) * 0.25;
    if (score > bestScore) {
      bestScore = score;
      best = ni;
    }
  }
  return best;
}

for (const droneIdx of planOrder) {
  const d = drones[droneIdx] || { x: 0, y: 0 };
  let current = inBounds(d.x | 0, d.y | 0) ? idxOf(d.x | 0, d.y | 0) : 0;
  const path = [[xOf(current), yOf(current)]];

  localTag++;
  if (localTag === 0x7fffffff) {
    localSeen.fill(0);
    localTag = 1;
  }

  let prev = -1;
  localSeen[current] = localTag;
  claimed[current] = 1;

  let remaining = maxSteps;
  while (remaining > 0) {
    const targetPath = chooseTarget(current, remaining, regionCandidates[droneIdx]);
    if (targetPath) {
      let used = false;
      for (let i = 1; i < targetPath.length && remaining > 0; i++) {
        const next = targetPath[i];
        prev = current;
        current = next;
        remaining--;
        path.push([xOf(current), yOf(current)]);
        localSeen[current] = localTag;
        claimed[current] = 1;
        used = true;
      }
      if (used) continue;
    }

    const next = chooseLocalMove(current, prev, droneIdx);
    prev = current;
    current = next;
    remaining--;
    path.push([xOf(current), yOf(current)]);
    localSeen[current] = localTag;
    claimed[current] = 1;
  }

  paths[droneIdx] = path;
}

// Drones with no assigned region or no movement still need a full-length path.
for (let i = 0; i < drones.length; i++) {
  if (!paths[i] || paths[i].length === 0) {
    const d = drones[i] || { x: 0, y: 0 };
    const sx = d.x | 0;
    const sy = d.y | 0;
    const p = [];
    for (let s = 0; s <= maxSteps; s++) p.push([sx, sy]);
    paths[i] = p;
  } else if (paths[i].length < maxSteps + 1) {
    const last = paths[i][paths[i].length - 1];
    while (paths[i].length < maxSteps + 1) paths[i].push([last[0], last[1]]);
  }
}

process.stdout.write(JSON.stringify({ paths }));
