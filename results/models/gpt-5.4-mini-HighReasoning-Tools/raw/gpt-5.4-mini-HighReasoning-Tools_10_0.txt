// 2D Cutting Stock / Rectangle Bin Packing heuristic solver.
// Fast global shelf packing with best-fit queries over shelf heights and board heights.

#include <algorithm>
#include <iostream>
#include <limits>
#include <set>
#include <utility>
#include <vector>

using namespace std;

struct Rect {
    int w = 0;
    int h = 0;
    int id = 0;
};

struct Shelf {
    int board_id = -1;
    int y = 0;
    int height_idx = -1;
    int height = 0;
    int next_x = 0;
    int rem_w = 0;
};

struct Placement {
    int board = 0;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct SegTree {
    int n = 0;
    vector<int> t;

    explicit SegTree(int n_ = 0) { init(n_); }

    void init(int n_) {
        n = n_;
        t.assign(4 * max(1, n), 0);
    }

    void update(int idx, int val) { update(1, 0, n - 1, idx, val); }

    int first_at_least(int from, int need) const {
        if (n == 0) return -1;
        return first_at_least(1, 0, n - 1, from, need);
    }

private:
    void update(int node, int l, int r, int idx, int val) {
        if (l == r) {
            t[node] = val;
            return;
        }
        int mid = (l + r) >> 1;
        if (idx <= mid) update(node << 1, l, mid, idx, val);
        else update(node << 1 | 1, mid + 1, r, idx, val);
        t[node] = max(t[node << 1], t[node << 1 | 1]);
    }

    int first_at_least(int node, int l, int r, int from, int need) const {
        if (r < from || t[node] < need) return -1;
        if (l == r) return l;
        int mid = (l + r) >> 1;
        int left = first_at_least(node << 1, l, mid, from, need);
        if (left != -1) return left;
        return first_at_least(node << 1 | 1, mid + 1, r, from, need);
    }
};

struct Candidate {
    bool valid = false;
    bool existing = false;
    int shelf_idx = -1;
    int board_id = -1;
    int rotated = 0;
    int height = 0;
    int rem_after = 0;
    int width = 0;
};

static bool better_candidate(const Candidate& a, const Candidate& b) {
    if (!a.valid) return false;
    if (!b.valid) return true;
    if (a.existing != b.existing) return a.existing; // existing shelf beats opening a new one
    if (a.height != b.height) return a.height < b.height;
    if (a.existing) {
        if (a.rem_after != b.rem_after) return a.rem_after < b.rem_after;
    } else {
        if (a.rem_after != b.rem_after) return a.rem_after > b.rem_after;
    }
    if (a.width != b.width) return a.width < b.width;
    if (a.board_id != b.board_id) return a.board_id < b.board_id;
    return a.rotated < b.rotated;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, BW, BH;
    if (!(cin >> N >> BW >> BH)) return 0;

    vector<Rect> rects(N);
    vector<int> heights;
    heights.reserve(2 * N);

    for (int i = 0; i < N; ++i) {
        cin >> rects[i].w >> rects[i].h;
        rects[i].id = i;
        heights.push_back(rects[i].w);
        heights.push_back(rects[i].h);
    }

    sort(heights.begin(), heights.end());
    heights.erase(unique(heights.begin(), heights.end()), heights.end());

    auto height_index = [&](int h) {
        return int(lower_bound(heights.begin(), heights.end(), h) - heights.begin());
    };

    vector<Rect> order = rects;
    sort(order.begin(), order.end(), [](const Rect& a, const Rect& b) {
        int am = max(a.w, a.h);
        int bm = max(b.w, b.h);
        if (am != bm) return am > bm;
        long long aa = 1LL * a.w * a.h;
        long long ab = 1LL * b.w * b.h;
        if (aa != ab) return aa > ab;
        int an = min(a.w, a.h);
        int bn = min(b.w, b.h);
        if (an != bn) return an > bn;
        return a.id < b.id;
    });

    int M = (int)heights.size();
    vector<multiset<pair<int, int>>> bucket_sets(M);
    vector<int> bucket_max(M, 0);
    SegTree seg(M);

    vector<Shelf> shelves;
    shelves.reserve(N);

    vector<int> board_rem_h;
    set<pair<int, int>> board_set; // {remaining height, board id}

    vector<Placement> out(N);

    auto update_bucket = [&](int idx) {
        int mx = bucket_sets[idx].empty() ? 0 : bucket_sets[idx].rbegin()->first;
        bucket_max[idx] = mx;
        seg.update(idx, mx);
    };

    auto place_on_shelf = [&](int shelf_idx, int rect_w, int rect_h, int rot_flag, int orig_id) {
        Shelf& s = shelves[shelf_idx];
        Placement p;
        p.board = s.board_id;
        p.x = s.next_x;
        p.y = s.y;
        p.rotated = rot_flag;
        out[orig_id] = p;

        auto& ms = bucket_sets[s.height_idx];
        auto it = ms.find({s.rem_w, shelf_idx});
        if (it != ms.end()) ms.erase(it);
        s.next_x += rect_w;
        s.rem_w -= rect_w;
        ms.insert({s.rem_w, shelf_idx});
        update_bucket(s.height_idx);
    };

    auto open_new_shelf = [&](int board_id, int shelf_h, int shelf_w) {
        int y = BH - board_rem_h[board_id] - shelf_h;
        Shelf s;
        s.board_id = board_id;
        s.y = y;
        s.height = shelf_h;
        s.height_idx = height_index(shelf_h);
        s.next_x = shelf_w;
        s.rem_w = BW - shelf_w;
        int idx = (int)shelves.size();
        shelves.push_back(s);
        bucket_sets[s.height_idx].insert({s.rem_w, idx});
        update_bucket(s.height_idx);
        return idx;
    };

    auto open_board = [&]() {
        int id = (int)board_rem_h.size();
        board_rem_h.push_back(BH);
        board_set.insert({BH, id});
        return id;
    };

    for (const Rect& r : order) {
        Candidate best;

        for (int rot = 0; rot < 2; ++rot) {
            int w = rot ? r.h : r.w;
            int h = rot ? r.w : r.h;
            if (w > BW || h > BH) continue;

            int hi = height_index(h);
            int bucket_idx = seg.first_at_least(hi, w);
            if (bucket_idx != -1) {
                const auto& ms = bucket_sets[bucket_idx];
                auto sit = ms.lower_bound({w, -1});
                if (sit != ms.end()) {
                    int shelf_idx = sit->second;
                    const Shelf& s = shelves[shelf_idx];
                    Candidate cand;
                    cand.valid = true;
                    cand.existing = true;
                    cand.shelf_idx = shelf_idx;
                    cand.board_id = s.board_id;
                    cand.rotated = rot;
                    cand.height = s.height;
                    cand.rem_after = s.rem_w - w;
                    cand.width = w;
                    if (better_candidate(cand, best)) best = cand;
                }
            }

            auto it = board_set.lower_bound({h, -1});
            int board_id = -1;
            if (it != board_set.end()) {
                board_id = it->second;
            }

            if (board_id != -1 && board_rem_h[board_id] >= h) {
                Candidate cand;
                cand.valid = true;
                cand.existing = false;
                cand.board_id = board_id;
                cand.rotated = rot;
                cand.height = h;
                cand.rem_after = BW - w;
                cand.width = w;
                if (better_candidate(cand, best)) best = cand;
            }
        }

        if (!best.valid) {
            // Fallback: create a fresh board and place the rectangle in the only orientation that fits the board.
            int rot = 0;
            int w = r.w, h = r.h;
            if (!(w <= BW && h <= BH) && r.h <= BW && r.w <= BH) {
                rot = 1;
                w = r.h;
                h = r.w;
            }
            int board_id = open_board();
            Candidate cand;
            cand.valid = true;
            cand.existing = false;
            cand.board_id = board_id;
            cand.rotated = rot;
            cand.height = h;
            cand.rem_after = BW - w;
            cand.width = w;
            best = cand;
        }

        if (best.existing) {
            place_on_shelf(best.shelf_idx, best.rotated ? r.h : r.w, best.rotated ? r.w : r.h, best.rotated, r.id);
        } else {
            int w = best.rotated ? r.h : r.w;
            int h = best.rotated ? r.w : r.h;

            int board_id = best.board_id;
            if (board_id == -1) {
                board_id = open_board();
            }
            auto it = board_set.find({board_rem_h[board_id], board_id});
            if (it != board_set.end()) board_set.erase(it);
            board_rem_h[board_id] -= h;
            board_set.insert({board_rem_h[board_id], board_id});

            int shelf_idx = open_new_shelf(board_id, h, w);
            const Shelf& s = shelves[shelf_idx];
            out[r.id] = {s.board_id, 0, s.y, best.rotated};
        }
    }

    cout << board_rem_h.size() << '\n';
    for (int i = 0; i < N; ++i) {
        const Placement& p = out[i];
        cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << '\n';
    }

    return 0;
}
