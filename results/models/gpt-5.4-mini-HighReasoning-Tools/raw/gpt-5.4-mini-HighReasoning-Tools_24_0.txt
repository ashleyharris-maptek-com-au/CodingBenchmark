// Lunar Lander 3D controller
// C++20, MSVC-friendly, real-time stdin/stdout loop.

#include <algorithm>
#include <atomic>
#include <cmath>
#include <cstdint>
#include <fstream>
#include <iomanip>
#include <iostream>
#include <limits>
#include <sstream>
#include <string>
#include <thread>

using std::clamp;
using std::getline;
using std::ifstream;
using std::istringstream;
using std::max;
using std::min;
using std::string;
using std::thread;

struct Vec3 {
    double x = 0.0;
    double y = 0.0;
    double z = 0.0;
};

static Vec3 operator+(const Vec3& a, const Vec3& b) {
    return {a.x + b.x, a.y + b.y, a.z + b.z};
}

static Vec3 operator-(const Vec3& a, const Vec3& b) {
    return {a.x - b.x, a.y - b.y, a.z - b.z};
}

static Vec3 operator*(const Vec3& a, double s) {
    return {a.x * s, a.y * s, a.z * s};
}

static Vec3 operator/(const Vec3& a, double s) {
    return {a.x / s, a.y / s, a.z / s};
}

static double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static double norm2(const Vec3& a) {
    return dot(a, a);
}

static double norm(const Vec3& a) {
    return std::sqrt(norm2(a));
}

static Vec3 normalize(const Vec3& a) {
    double n = norm(a);
    if (n <= 1e-12) return {0.0, 0.0, 1.0};
    return a / n;
}

static double wrap_pi(double a) {
    const double pi = 3.14159265358979323846;
    while (a > pi) a -= 2.0 * pi;
    while (a < -pi) a += 2.0 * pi;
    return a;
}

static void trim_trailing_cr(string& s) {
    if (!s.empty() && s.back() == '\r') {
        s.pop_back();
    }
}

static void angles_from_thrust_dir(const Vec3& d, double& pitch, double& yaw) {
    Vec3 n = normalize(d);
    double horiz = std::hypot(n.x, n.y);
    pitch = std::atan2(horiz, std::max(1e-12, n.z));
    if (horiz < 1e-9) {
        yaw = 0.0;
    } else {
        yaw = std::atan2(-n.y, -n.x);
    }
}

struct State {
    double x = 0.0, y = 0.0, z = 0.0;
    double vx = 0.0, vy = 0.0, vz = 0.0;
    double pitch = 0.0, yaw = 0.0, roll = 0.0;
    double pitch_rate = 0.0, yaw_rate = 0.0, roll_rate = 0.0;
    double fuel = 0.0;
};

struct TerrainSummary {
    std::atomic<double> target_ground{0.0};
    std::atomic<double> local_max_ground{0.0};
    std::atomic<bool> ready{false};
};

static void sample_local_terrain(
    const string& path,
    int width,
    int depth,
    [[maybe_unused]] int height,
    double target_x,
    double target_y,
    TerrainSummary* summary) {
    ifstream file(path, std::ios::binary);
    if (!file) {
        summary->ready.store(true, std::memory_order_relaxed);
        return;
    }

    int tx = clamp(static_cast<int>(std::lround(target_x)), 0, width - 1);
    int ty = clamp(static_cast<int>(std::lround(target_y)), 0, depth - 1);
    int radius = min({64, max(8, width / 20), max(8, depth / 20)});

    double local_max = 0.0;
    double target_ground = 0.0;
    bool target_seen = false;

    for (int x = max(0, tx - radius); x <= min(width - 1, tx + radius); ++x) {
        for (int y = max(0, ty - radius); y <= min(depth - 1, ty + radius); ++y) {
            const std::uint64_t cell_index = static_cast<std::uint64_t>(x) * static_cast<std::uint64_t>(depth) + static_cast<std::uint64_t>(y);
            const std::uint64_t byte_offset = cell_index * 6ULL;
            file.seekg(static_cast<std::streamoff>(byte_offset), std::ios::beg);
            if (!file) continue;
            unsigned char buf[6] = {0, 0, 0, 0, 0, 0};
            file.read(reinterpret_cast<char*>(buf), 6);
            if (!file) continue;
            std::uint16_t a = static_cast<std::uint16_t>(buf[0]) | (static_cast<std::uint16_t>(buf[1]) << 8);
            std::uint16_t c = static_cast<std::uint16_t>(buf[4]) | (static_cast<std::uint16_t>(buf[5]) << 8);

            local_max = max(local_max, static_cast<double>(max(a, c)));
            if (x == tx && y == ty) {
                target_ground = static_cast<double>(a);
                target_seen = true;
            }
        }
    }

    if (!target_seen) {
        target_ground = 0.0;
    }

    summary->target_ground.store(target_ground, std::memory_order_relaxed);
    summary->local_max_ground.store(local_max, std::memory_order_relaxed);
    summary->ready.store(true, std::memory_order_relaxed);
}

static bool parse_state_line(const string& line, State& s) {
    istringstream iss(line);
    return static_cast<bool>(
        iss >> s.x >> s.y >> s.z
            >> s.vx >> s.vy >> s.vz
            >> s.pitch >> s.yaw >> s.roll
            >> s.pitch_rate >> s.yaw_rate >> s.roll_rate
            >> s.fuel);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int width = 0, depth = 0, height = 0;
    double gravity = 0.0;
    double max_thrust = 0.0;
    double fuel = 0.0;
    double start_x = 0.0, start_y = 0.0, start_z = 0.0;
    double target_x = 0.0, target_y = 0.0, target_z = 0.0;

    if (!(std::cin >> width >> depth >> height >> gravity >> max_thrust >> fuel)) {
        return 0;
    }
    if (!(std::cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z)) {
        return 0;
    }
    static_cast<void>(start_x);
    static_cast<void>(start_y);
    static_cast<void>(start_z);
    static_cast<void>(fuel);

    std::cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');

    string terrain_path;
    if (!getline(std::cin, terrain_path)) {
        return 0;
    }
    trim_trailing_cr(terrain_path);
    if (terrain_path.empty()) {
        if (!getline(std::cin, terrain_path)) {
            return 0;
        }
        trim_trailing_cr(terrain_path);
    }

    TerrainSummary terrain;
    thread terrain_thread(sample_local_terrain, terrain_path, width, depth, height, target_x, target_y, &terrain);

    string line;
    bool have_pending_line = false;
    string pending_line;
    if (getline(std::cin, pending_line)) {
        have_pending_line = true;
    }

    std::cout.setf(std::ios::fixed);
    std::cout << std::setprecision(6);

    while (true) {
        if (have_pending_line) {
            line = pending_line;
            have_pending_line = false;
        } else {
            if (!getline(std::cin, line)) {
                break;
            }
        }

        trim_trailing_cr(line);

        if (line.empty() || line == "STATE") {
            continue;
        }

        State s;
        if (!parse_state_line(line, s)) {
            continue;
        }

        const double landing_radius = max(5.0, min(width, depth) * 0.02);
        const double dx = target_x - s.x;
        const double dy = target_y - s.y;
        const double dist2d = std::hypot(dx, dy);
        const double speed3d = std::sqrt(s.vx * s.vx + s.vy * s.vy + s.vz * s.vz);

        double terrain_ground = target_z;
        double terrain_ceiling = target_z + 60.0;
        if (terrain.ready.load(std::memory_order_relaxed)) {
            terrain_ground = terrain.target_ground.load(std::memory_order_relaxed);
            terrain_ceiling = max(terrain.target_ground.load(std::memory_order_relaxed) + 50.0,
                                  terrain.local_max_ground.load(std::memory_order_relaxed) + 35.0);
        }

        // Glide-slope style altitude planning:
        // high when far away, then steadily lower as we close on the pad.
        const double desired_z_far = target_z + clamp(2.0 + 0.30 * dist2d, 2.0, max(60.0, terrain_ceiling - target_z));
        double desired_z = desired_z_far;
        if (dist2d < 8.0 * landing_radius) {
            desired_z = max(target_z + 1.5, target_z + 0.15 * dist2d);
        }
        if (dist2d < 2.5 * landing_radius) {
            desired_z = max(target_z + 0.8, target_z + 0.08 * dist2d);
        }
        desired_z = max(desired_z, terrain_ground + 1.0);

        // Guidance gains tuned for gentle landing under 0.1 s ticks.
        const double kpx = 0.020;
        const double kdx = 0.80;
        const double kpy = 0.020;
        const double kdy = 0.80;
        const double kpz = 0.060;
        const double kdz = 0.95;

        const double max_lateral_acc = max(1.5, min(max_thrust * 0.80, 8.0));

        double ax = kpx * dx - kdx * s.vx;
        double ay = kpy * dy - kdy * s.vy;
        ax = clamp(ax, -max_lateral_acc, max_lateral_acc);
        ay = clamp(ay, -max_lateral_acc, max_lateral_acc);

        // Reduce lateral aggression near touchdown.
        const double near_factor = clamp(dist2d / (landing_radius * 6.0 + 1e-9), 0.25, 1.0);
        ax *= near_factor;
        ay *= near_factor;

        double az = kpz * (desired_z - s.z) - kdz * s.vz;
        // Thrust cannot point "down", so convert desired net vertical accel
        // into required upward thrust accel above gravity.
        double thrust_z = az + gravity;
        if (thrust_z < 0.0) thrust_z = 0.0;

        Vec3 thrust_acc{ax, ay, thrust_z};
        double requested_mag = norm(thrust_acc);

        if (requested_mag > max_thrust && requested_mag > 1e-12) {
            thrust_acc = thrust_acc * (max_thrust / requested_mag);
            requested_mag = max_thrust;
        }

        double thrust = (max_thrust > 1e-12) ? (requested_mag / max_thrust) : 0.0;
        thrust = clamp(thrust, 0.0, 1.0);

        Vec3 desired_dir = normalize(thrust_acc);
        double desired_pitch = 0.0;
        double desired_yaw = 0.0;
        angles_from_thrust_dir(desired_dir, desired_pitch, desired_yaw);

        // If almost vertical, keep yaw steady to avoid noise.
        if (std::hypot(desired_dir.x, desired_dir.y) < 1e-6) {
            desired_yaw = s.yaw;
        }

        double pitch_err = desired_pitch - s.pitch;
        double yaw_err = wrap_pi(desired_yaw - s.yaw);

        double pitch_cmd = 2.2 * pitch_err - 0.65 * s.pitch_rate;
        double yaw_cmd = 2.2 * yaw_err - 0.65 * s.yaw_rate;

        pitch_cmd = clamp(pitch_cmd, -1.0, 1.0);
        yaw_cmd = clamp(yaw_cmd, -1.0, 1.0);

        // Roll is cosmetic in the problem statement; keep it near zero.
        double roll_cmd = clamp(-1.5 * s.roll - 0.6 * s.roll_rate, -1.0, 1.0);

        // If we are already close and slow, prefer a more vertical, gentle finish.
        if (dist2d < 2.0 * landing_radius && speed3d < 6.0) {
            thrust = (max_thrust > 1e-12)
                       ? clamp((gravity + 0.4 * (target_z - s.z) - 0.8 * s.vz) / max_thrust, 0.0, 1.0)
                       : 0.0;
            desired_yaw = s.yaw;
            pitch_cmd = clamp(-1.1 * s.pitch - 0.45 * s.pitch_rate, -1.0, 1.0);
            yaw_cmd = clamp(-1.1 * wrap_pi(s.yaw - desired_yaw) - 0.45 * s.yaw_rate, -1.0, 1.0);
            roll_cmd = clamp(-1.5 * s.roll - 0.6 * s.roll_rate, -1.0, 1.0);
        }

        std::cout << thrust << ' ' << pitch_cmd << ' ' << yaw_cmd << ' ' << roll_cmd << '\n' << std::flush;
    }

    if (terrain_thread.joinable()) {
        terrain_thread.join();
    }

    return 0;
}
