#include <vector>

using namespace std;

struct StringData {
    string s;
    vector<uint64_t> pref;
};

static inline void trim_cr(string& s) {
    if (!s.empty() && s.back() == '\r') s.pop_back();
}

struct SuffixAutomaton {
    struct Edge {
        int to;
        int next;
        unsigned char ch;
    };

    struct State {
        int link = -1;
        int len = 0;
        int head = -1;
        int first_pos = -1;
    };

    vector<State> st;
    vector<Edge> edges;
    int last = 0;

    explicit SuffixAutomaton(size_t reserve_states = 0, size_t reserve_edges = 0) {
        st.reserve(reserve_states);
        edges.reserve(reserve_edges);
        st.push_back(State{});
    }

    int find_edge(int v, unsigned char c) const {
        for (int e = st[v].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) return edges[e].to;
        }
        return -1;
    }

    void add_edge(int v, unsigned char c, int to) {
        edges.push_back(Edge{to, st[v].head, c});
        st[v].head = (int)edges.size() - 1;
    }

    void replace_edge(int v, unsigned char c, int to) {
        for (int e = st[v].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) {
                edges[e].to = to;
                return;
            }
        }
    }

    int new_state(int len) {
        st.push_back(State{});
        st.back().len = len;
        return (int)st.size() - 1;
    }

    void extend(unsigned char c, int pos) {
        int cur = new_state(st[last].len + 1);
        st[cur].first_pos = pos;

        int p = last;
        while (p != -1 && find_edge(p, c) == -1) {
            add_edge(p, c, cur);
            p = st[p].link;
        }

        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = find_edge(p, c);
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = new_state(st[p].len + 1);
                st[clone].head = st[q].head;
                st[clone].link = st[q].link;
                st[clone].first_pos = st[q].first_pos;

                while (p != -1 && find_edge(p, c) == q) {
                    replace_edge(p, c, clone);
                    p = st[p].link;
                }

                st[q].link = st[cur].link = clone;
            }
        }

        last = cur;
    }
};

static string solve_sam(const vector<StringData>& strs, size_t base_idx) {
    const string& base = strs[base_idx].s;
    if (base.empty()) return "";

    SuffixAutomaton sam(base.size() * 2 + 1, base.size() * 4 + 4);
    for (int i = 0; i < (int)base.size(); ++i) {
        sam.extend((unsigned char)base[i], i);
    }

    int states = (int)sam.st.size();
    vector<int> order(states);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        if (sam.st[a].len != sam.st[b].len) return sam.st[a].len < sam.st[b].len;
        return a < b;
    });

    vector<int> common(states);
    for (int i = 0; i < states; ++i) common[i] = sam.st[i].len;
    vector<int> mx(states, 0);

    for (size_t idx = 0; idx < strs.size(); ++idx) {
        if (idx == base_idx) continue;
        fill(mx.begin(), mx.end(), 0);

        int v = 0;
        int l = 0;
        const string& s = strs[idx].s;
        for (unsigned char ch : s) {
            while (v != 0 && sam.find_edge(v, ch) == -1) {
                v = sam.st[v].link;
                l = sam.st[v].len;
            }
            int to = sam.find_edge(v, ch);
            if (to != -1) {
                v = to;
                ++l;
            } else {
                v = 0;
                l = 0;
            }
            if (l > mx[v]) mx[v] = l;
        }

        for (int i = states - 1; i > 0; --i) {
            int v2 = order[i];
            int p = sam.st[v2].link;
            if (p >= 0 && mx[v2] > 0) {
                int val = mx[v2];
                if (val > sam.st[p].len) val = sam.st[p].len;
                if (val > mx[p]) mx[p] = val;
            }
        }

        for (int i = 0; i < states; ++i) {
            if (mx[i] < common[i]) common[i] = mx[i];
        }
    }

    int best_state = 0;
    int best_len = 0;
    for (int i = 0; i < states; ++i) {
        if (common[i] > best_len) {
            best_len = common[i];
            best_state = i;
        }
    }

    if (best_len <= 0) return "";
    int end_pos = sam.st[best_state].first_pos;
    int start = end_pos - best_len + 1;
    if (start < 0) return "";
    return base.substr((size_t)start, (size_t)best_len);
}

static uint64_t substring_hash(const vector<uint64_t>& pref, const vector<uint64_t>& powv, size_t l, size_t r) {
    return pref[r] - pref[l] * powv[r - l];
}

static bool exact_exists_all(const vector<StringData>& strs, const string& cand) {
    for (const auto& s : strs) {
        if (s.s.find(cand) == string::npos) return false;
    }
    return true;
}

static string locate_candidate_in_base(const string& base, const vector<uint64_t>& pref, const vector<uint64_t>& powv, uint64_t h, size_t L) {
    if (base.size() < L) return "";
    for (size_t i = 0; i + L <= base.size(); ++i) {
        if (substring_hash(pref, powv, i, i + L) == h) {
            return base.substr(i, L);
        }
    }
    return "";
}

static string solve_hash(const vector<StringData>& data, const vector<size_t>& order, const vector<uint64_t>& powv) {
    if (order.empty()) return "";
    size_t base_idx = order[0];
    const string& base = data[base_idx].s;
    if (base.empty()) return "";

    auto check = [&](size_t L) -> string {
        if (base.size() < L) return "";

        unordered_set<uint64_t> cur;
        cur.max_load_factor(0.7f);
        cur.reserve(base.size() - L + 1);
        const auto& base_pref = data[base_idx].pref;
        for (size_t i = 0; i + L <= base.size(); ++i) {
            cur.insert(substring_hash(base_pref, powv, i, i + L));
        }
        if (cur.empty()) return "";

        for (size_t ord = 1; ord < order.size() && !cur.empty(); ++ord) {
            size_t idx = order[ord];
            const string& s = data[idx].s;
            if (s.size() < L) return "";

            unordered_set<uint64_t> nxt;
            nxt.max_load_factor(0.7f);
            size_t reserve_cap = min(cur.size(), s.size() - L + 1);
            nxt.reserve(reserve_cap);

            const auto& pref = data[idx].pref;
            for (size_t i = 0; i + L <= s.size(); ++i) {
                uint64_t h = substring_hash(pref, powv, i, i + L);
                if (cur.find(h) != cur.end()) {
                    nxt.insert(h);
                }
            }
            cur.swap(nxt);
        }

        if (cur.empty()) return "";

        for (uint64_t h : cur) {
            for (size_t i = 0; i + L <= base.size(); ++i) {
                if (substring_hash(data[base_idx].pref, powv, i, i + L) == h) {
                    string cand = base.substr(i, L);
                    if (exact_exists_all(data, cand)) return cand;
                }
            }
        }
        return "";
    };

    size_t lo = 1, hi = base.size();
    string best;
    while (lo <= hi) {
        size_t mid = (lo + hi) / 2;
        string cand = check(mid);
        if (!cand.empty()) {
            best = std::move(cand);
            lo = mid + 1;
        } else {
            hi = mid - 1;
        }
    }
    return best;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    string dummy;
    getline(cin, dummy);

    vector<StringData> data;
    data.resize(N);
    size_t total_len = 0;
    size_t min_len = (size_t)-1;
    size_t min_idx = 0;

    for (int i = 0; i < N; ++i) {
        getline(cin, data[i].s);
        trim_cr(data[i].s);
        total_len += data[i].s.size();
        if (data[i].s.size() < min_len) {
            min_len = data[i].s.size();
            min_idx = (size_t)i;
        }
    }

    if (N == 0 || min_len == 0) {
        cout << '\n';
        return 0;
    }

    if (N == 1) {
        cout << data[0].s << '\n';
        return 0;
    }

    // Adaptive strategy:
    // - For moderate inputs, use an exact suffix automaton over the shortest string.
    // - For larger inputs, use rolling-hash binary search with candidate filtering.
    bool use_sam = (min_len <= 200000 && total_len <= 3000000 && N <= 40);

    if (use_sam) {
        string ans = solve_sam(data, min_idx);
        cout << ans << '\n';
        return 0;
    }

    vector<size_t> order(N);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](size_t a, size_t b) {
        if (data[a].s.size() != data[b].s.size()) return data[a].s.size() < data[b].s.size();
        return a < b;
    });

    size_t max_len = 0;
    for (const auto& x : data) max_len = max(max_len, x.s.size());

    uint64_t seed = (uint64_t)chrono::high_resolution_clock::now().time_since_epoch().count();
    uint64_t base = (seed | 1ULL);
    if (base < 257) base += 1000003ULL;

    vector<uint64_t> powv(max_len + 1, 0);
    powv[0] = 1;
    for (size_t i = 1; i <= max_len; ++i) {
        powv[i] = powv[i - 1] * base;
    }

    for (auto& x : data) {
        x.pref.assign(x.s.size() + 1, 0);
        for (size_t i = 0; i < x.s.size(); ++i) {
            x.pref[i + 1] = x.pref[i] * base + (uint64_t)(unsigned char)x.s[i] + 1ULL;
        }
    }

    string ans = solve_hash(data, order, powv);
    cout << ans << '\n';
    return 0;
}
