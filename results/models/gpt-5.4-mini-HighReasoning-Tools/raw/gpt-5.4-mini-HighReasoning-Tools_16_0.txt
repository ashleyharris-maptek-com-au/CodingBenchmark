// Feasible job-shop scheduler.
// Strategy:
// - Small instances: buffer all jobs, then schedule/output in a second pass.
// - Large instances: stream each job, schedule immediately, and emit output line by line.
//
// The schedule is a simple greedy earliest-start construction:
// for each task, start it at max(job_ready, machine_ready[m]).
// This always produces a valid non-preemptive schedule because tasks are processed
// in job order and each task only depends on the previous task in the same job.

#include <algorithm>
#include <cstdint>
#include <iostream>
#include <string>
#include <vector>

struct Task {
    int machine;
    long long duration;
};

struct FastOutput {
    static constexpr std::size_t kFlushLimit = 1u << 20;
    std::string buffer;

    void push(char c) {
        buffer.push_back(c);
        if (buffer.size() >= kFlushLimit) {
            flush();
        }
    }

    void writeLL(long long value) {
        if (value == 0) {
            push('0');
            return;
        }
        if (value < 0) {
            push('-');
            value = -value;
        }
        char tmp[32];
        int len = 0;
        while (value > 0) {
            tmp[len++] = static_cast<char>('0' + (value % 10));
            value /= 10;
        }
        while (len--) {
            push(tmp[len]);
        }
    }

    void writeInt(int value) {
        writeLL(static_cast<long long>(value));
    }

    void flush() {
        if (!buffer.empty()) {
            std::cout.write(buffer.data(), static_cast<std::streamsize>(buffer.size()));
            buffer.clear();
        }
    }
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int J = 0, M = 0;
    if (!(std::cin >> J >> M)) {
        return 0;
    }

    // Adaptive choice: small instances are buffered, larger ones are streamed.
    // The schedule itself is the same in both cases.
    const bool buffered_mode = (J <= 32 && M <= 32);

    FastOutput out;
    std::vector<long long> machine_ready(static_cast<std::size_t>(M), 0LL);

    if (buffered_mode) {
        std::vector<std::vector<Task>> jobs;
        jobs.resize(static_cast<std::size_t>(J));

        for (int j = 0; j < J; ++j) {
            int T = 0;
            std::cin >> T;
            auto& tasks = jobs[static_cast<std::size_t>(j)];
            tasks.reserve(static_cast<std::size_t>(T));
            for (int i = 0; i < T; ++i) {
                int machine = 0;
                long long duration = 0;
                std::cin >> machine >> duration;
                tasks.push_back(Task{machine, duration});
            }
        }

        for (int j = 0; j < J; ++j) {
            long long job_ready = 0;
            const auto& tasks = jobs[static_cast<std::size_t>(j)];
            for (const Task& task : tasks) {
                long long start = std::max(job_ready, machine_ready[static_cast<std::size_t>(task.machine)]);
                long long finish = start + task.duration;
                job_ready = finish;
                machine_ready[static_cast<std::size_t>(task.machine)] = finish;

                out.writeLL(start);
                out.push(' ');
                out.writeInt(task.machine);
                out.push(' ');
            }
            out.push('\n');
        }
    } else {
        for (int j = 0; j < J; ++j) {
            int T = 0;
            std::cin >> T;
            long long job_ready = 0;
            for (int i = 0; i < T; ++i) {
                int machine = 0;
                long long duration = 0;
                std::cin >> machine >> duration;

                long long start = std::max(job_ready, machine_ready[static_cast<std::size_t>(machine)]);
                long long finish = start + duration;
                job_ready = finish;
                machine_ready[static_cast<std::size_t>(machine)] = finish;

                out.writeLL(start);
                out.push(' ');
                out.writeInt(machine);
                out.push(' ');
            }
            out.push('\n');
        }
    }

    out.flush();
    return 0;
}
