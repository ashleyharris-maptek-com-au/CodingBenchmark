import sys
import time
import random
from array import array

try:
    import numpy as _np
except Exception:
    _np = None


def int_stream():
    data = sys.stdin.buffer
    num = 0
    sign = 1
    in_num = False
    while True:
        chunk = data.read(1 << 20)
        if not chunk:
            break
        for b in chunk:
            if 48 <= b <= 57:
                num = num * 10 + (b - 48)
                in_num = True
            elif b == 45:
                sign = -1
            else:
                if in_num:
                    yield sign * num
                    num = 0
                    sign = 1
                    in_num = False
                else:
                    sign = 1
    if in_num:
        yield sign * num


def pack_lit(x):
    v = -x if x < 0 else x
    return (v << 1) | (1 if x < 0 else 0)


def lit_true(assign, lit_code):
    return assign[lit_code >> 1] ^ (lit_code & 1)


def build_initial_assignment(n, pos, neg):
    if _np is not None:
        pos_np = _np.frombuffer(pos, dtype=_np.uint32)
        neg_np = _np.frombuffer(neg, dtype=_np.uint32)
        # Majority sign with pure literals automatically handled.
        assign_np = (pos_np >= neg_np).astype(_np.uint8, copy=False)
        return bytearray(assign_np.tobytes())
    assign = bytearray(n + 1)
    for v in range(1, n + 1):
        assign[v] = 1 if pos[v] >= neg[v] else 0
    return assign


def evaluate_formula(clauses, assign, m):
    sat_count = bytearray(m)
    unsat = []
    if _np is not None:
        cl = _np.frombuffer(clauses, dtype=_np.uint32)
        a = _np.frombuffer(assign, dtype=_np.uint8)
        chunk = 1_000_000
        for start in range(0, m, chunk):
            end = min(m, start + chunk)
            sl = cl[3 * start : 3 * end].reshape(end - start, 3)
            sat = a[sl >> 1] ^ (sl & 1)
            cnts = sat.sum(axis=1).astype(_np.uint8, copy=False)
            sat_count[start:end] = cnts.tobytes()
            unsat.extend(start + int(i) for i in _np.flatnonzero(cnts == 0))
        return sat_count, unsat
    for c in range(m):
        base = 3 * c
        cnt = (
            (assign[clauses[base] >> 1] ^ (clauses[base] & 1))
            + (assign[clauses[base + 1] >> 1] ^ (clauses[base + 1] & 1))
            + (assign[clauses[base + 2] >> 1] ^ (clauses[base + 2] & 1))
        )
        sat_count[c] = cnt
        if cnt == 0:
            unsat.append(c)
    return sat_count, unsat


def build_occurrences(n, m, clauses, pos, neg):
    occ_start = array("I", [0]) * (n + 2)
    total = 0
    for v in range(1, n + 1):
        occ_start[v] = total
        total += pos[v] + neg[v]
    occ_start[n + 1] = total

    occ = array("I", [0]) * (3 * m)
    nxt = array("I", occ_start)
    for c in range(m):
        base = 3 * c
        for j in range(3):
            lit = clauses[base + j]
            v = lit >> 1
            sign = lit & 1
            idx = nxt[v]
            occ[idx] = (c << 1) | sign
            nxt[v] = idx + 1
    return occ_start, occ


def solve(n, m, clauses, pos, neg):
    assign = build_initial_assignment(n, pos, neg)
    sat_count, unsat = evaluate_formula(clauses, assign, m)
    if not unsat:
        return assign

    occ_start, occ = build_occurrences(n, m, clauses, pos, neg)
    rng = random.Random(0xC0DE5A7A)
    deadline = time.time() + 285.0

    # Track the true number of unsatisfied clauses, while allowing stale entries
    # to remain in the unsat stack for speed.
    unsat_count = len(unsat)

    def recompute():
        nonlocal sat_count, unsat, unsat_count
        sat_count, unsat = evaluate_formula(clauses, assign, m)
        unsat_count = len(unsat)

    def flip(v):
        nonlocal unsat_count
        old_val = assign[v]
        new_val = old_val ^ 1
        if old_val == new_val:
            return
        assign[v] = new_val
        a = occ_start[v]
        b = occ_start[v + 1]
        for i in range(a, b):
            entry = occ[i]
            c = entry >> 1
            sign = entry & 1
            old_sat = old_val ^ sign
            new_sat = new_val ^ sign
            if old_sat == new_sat:
                continue
            sc = sat_count[c]
            if new_sat:
                sat_count[c] = sc + 1
                if sc == 0:
                    unsat_count -= 1
            else:
                sat_count[c] = sc - 1
                if sc == 1:
                    unsat_count += 1
                    unsat.append(c)

    def clean_unsat_tail():
        while unsat and sat_count[unsat[-1]] != 0:
            unsat.pop()

    # Restart schedule: mostly keep the majority assignment, but perturb it if
    # the search stalls.
    for restart in range(8):
        if time.time() > deadline:
            break
        if restart > 0:
            # Randomly re-seed uncertain variables. Pure literals stay fixed by
            # the majority rule (pos/neg asymmetry).
            for v in range(1, n + 1):
                pv = pos[v]
                nv = neg[v]
                if pv == 0 or nv == 0:
                    continue
                if abs(int(pv) - int(nv)) <= 2:
                    assign[v] = 1 if rng.getrandbits(1) else 0
                elif rng.random() < 0.08:
                    assign[v] ^= 1
            recompute()
            if not unsat:
                return assign

        stall = 0
        best_seen = unsat_count

        while unsat_count > 0 and time.time() <= deadline:
            clean_unsat_tail()
            if not unsat:
                recompute()
                if unsat_count == 0:
                    return assign
                continue

            # Prefer a recently unsatisfied clause; that tends to improve locality.
            if len(unsat) > 8 and rng.random() < 0.25:
                idx = -1 - rng.randrange(8)
            else:
                idx = -1
            c = unsat[idx]
            if sat_count[c] != 0:
                unsat.pop(idx)
                continue

            base = 3 * c
            lits = (clauses[base], clauses[base + 1], clauses[base + 2])

            # With some probability, do a random walk step.
            if rng.random() < 0.30:
                lit = lits[rng.randrange(3)]
                flip(lit >> 1)
            else:
                best_v = None
                best_bc = 1 << 30
                for lit in lits:
                    v = lit >> 1
                    a = occ_start[v]
                    b = occ_start[v + 1]
                    val = assign[v]
                    bc = 0
                    # Break count: clauses currently satisfied only by this
                    # literal would be broken if the variable flips.
                    for i in range(a, b):
                        entry = occ[i]
                        c2 = entry >> 1
                        if sat_count[c2] != 1:
                            continue
                        if (val ^ (entry & 1)):
                            bc += 1
                            if bc >= best_bc:
                                break
                    if bc < best_bc:
                        best_bc = bc
                        best_v = v
                        if bc == 0:
                            break
                flip(best_v if best_v is not None else (lits[0] >> 1))

            if unsat_count < best_seen:
                best_seen = unsat_count
                stall = 0
            else:
                stall += 1

            # If progress stalls, trigger a fresh restart.
            if stall > 50000:
                break

        if unsat_count == 0:
            return assign

    # Last resort: perform a few cheap global recomputations with more random
    # perturbation, then return the best-so-far assignment.
    for _ in range(4):
        if time.time() > deadline:
            break
        for v in range(1, n + 1):
            pv = pos[v]
            nv = neg[v]
            if pv == 0 or nv == 0:
                continue
            if rng.random() < 0.5:
                assign[v] ^= 1
        recompute()
        if unsat_count == 0:
            return assign

    return assign


def main():
    it = int_stream()
    try:
        n = next(it)
        m = next(it)
    except StopIteration:
        return

    # Dense compact storage.
    clauses = array("I", [0]) * (3 * m)
    pos = array("I", [0]) * (n + 1)
    neg = array("I", [0]) * (n + 1)

    for i in range(3 * m):
        try:
            lit = next(it)
        except StopIteration:
            lit = 0
        code = pack_lit(lit)
        clauses[i] = code
        v = code >> 1
        if code & 1:
            neg[v] += 1
        else:
            pos[v] += 1

    assign = solve(n, m, clauses, pos, neg)

    out = sys.stdout.write
    out("SAT\n")
    if n == 0:
        out("\n")
        return

    chunk = []
    for v in range(1, n + 1):
        chunk.append("1" if assign[v] else "0")
        if len(chunk) >= 4096:
            out(" ".join(chunk))
            out(" ")
            chunk.clear()
    if chunk:
        out(" ".join(chunk))
    out("\n")


if __name__ == "__main__":
    main()
