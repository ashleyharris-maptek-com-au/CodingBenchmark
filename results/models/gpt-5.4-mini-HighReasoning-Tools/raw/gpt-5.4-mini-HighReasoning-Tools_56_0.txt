const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = input.trim() ? JSON.parse(input) : {};

const depot = Array.isArray(data.depot) ? data.depot : [0, 0];
const packages = Array.isArray(data.packages) ? data.packages.slice() : [];

if (packages.length === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const depotX = depot[0];
const depotY = depot[1];

for (let i = 0; i < packages.length; i++) {
  const p = packages[i];
  p._idx = i;
  p._dx = p.x - depotX;
  p._dy = p.y - depotY;
  p._dist = Math.hypot(p._dx, p._dy);
  p._angle = Math.atan2(p._dy, p._dx);
  if (p._angle < 0) p._angle += Math.PI * 2;
}

const byId = new Map();
for (const p of packages) byId.set(p.id, p);

function distPoints(a, b) {
  return Math.hypot(a.x - b.x, a.y - b.y);
}

function distIdx(i, j) {
  const a = packages[i];
  const b = packages[j];
  return Math.hypot(a.x - b.x, a.y - b.y);
}

function distDepotPoint(p) {
  return Math.hypot(p.x - depotX, p.y - depotY);
}

function cmpNum(a, b) {
  return a < b ? -1 : a > b ? 1 : 0;
}

function sortCopy(arr, cmp) {
  return arr.slice().sort(cmp);
}

function buildAngleOrder() {
  return sortCopy(packages, (a, b) => {
    const d = cmpNum(a._angle, b._angle);
    if (d) return d;
    const r = cmpNum(a._dist, b._dist);
    if (r) return r;
    return a._idx - b._idx;
  });
}

function buildXOrder() {
  return sortCopy(packages, (a, b) => {
    const d = cmpNum(a.x, b.x);
    if (d) return d;
    const y = cmpNum(a.y, b.y);
    if (y) return y;
    return a._idx - b._idx;
  });
}

function buildYOrder() {
  return sortCopy(packages, (a, b) => {
    const d = cmpNum(a.y, b.y);
    if (d) return d;
    const x = cmpNum(a.x, b.x);
    if (x) return x;
    return a._idx - b._idx;
  });
}

function buildPcaOrder() {
  let sx = 0;
  let sy = 0;
  for (const p of packages) {
    sx += p._dx;
    sy += p._dy;
  }
  const cx = sx / packages.length;
  const cy = sy / packages.length;

  let cxx = 0;
  let cxy = 0;
  let cyy = 0;
  for (const p of packages) {
    const x = p._dx - cx;
    const y = p._dy - cy;
    cxx += x * x;
    cxy += x * y;
    cyy += y * y;
  }

  const theta = 0.5 * Math.atan2(2 * cxy, cxx - cyy);
  const ux = Math.cos(theta);
  const uy = Math.sin(theta);
  const vx = -uy;
  const vy = ux;

  return sortCopy(packages, (a, b) => {
    const ap = a._dx * ux + a._dy * uy;
    const bp = b._dx * ux + b._dy * uy;
    const d = cmpNum(ap, bp);
    if (d) return d;
    const ao = a._dx * vx + a._dy * vy;
    const bo = b._dx * vx + b._dy * vy;
    const o = cmpNum(ao, bo);
    if (o) return o;
    return a._idx - b._idx;
  });
}

function buildNearestInsertion(startIdx) {
  const n = packages.length;
  const used = new Uint8Array(n);
  const order = [startIdx];
  used[startIdx] = 1;

  const nearest = new Float64Array(n);
  for (let i = 0; i < n; i++) nearest[i] = distIdx(i, startIdx);
  nearest[startIdx] = Infinity;

  for (let step = 1; step < n; step++) {
    let pick = -1;
    let best = Infinity;
    for (let i = 0; i < n; i++) {
      if (!used[i] && nearest[i] < best) {
        best = nearest[i];
        pick = i;
      }
    }

    let bestPos = 0;
    let bestDelta = Infinity;
    const picked = packages[pick];

    for (let pos = 0; pos <= order.length; pos++) {
      let delta;
      if (pos === 0) {
        const first = packages[order[0]];
        delta = picked._dist + distPoints(picked, first) - first._dist;
      } else if (pos === order.length) {
        const last = packages[order[order.length - 1]];
        delta = distPoints(last, picked) + picked._dist - last._dist;
      } else {
        const a = packages[order[pos - 1]];
        const b = packages[order[pos]];
        delta = distPoints(a, picked) + distPoints(picked, b) - distPoints(a, b);
      }
      if (delta < bestDelta) {
        bestDelta = delta;
        bestPos = pos;
      }
    }

    order.splice(bestPos, 0, pick);
    used[pick] = 1;

    for (let i = 0; i < n; i++) {
      if (!used[i]) {
        const d = distIdx(i, pick);
        if (d < nearest[i]) nearest[i] = d;
      }
    }
  }

  return order.map(i => packages[i]);
}

function routeCost(order) {
  let total = order[0]._dist;
  for (let i = 0; i + 1 < order.length; i++) total += distPoints(order[i], order[i + 1]);
  total += order[order.length - 1]._dist;
  return total;
}

function reverseRange(arr, l, r) {
  while (l < r) {
    const tmp = arr[l];
    arr[l] = arr[r];
    arr[r] = tmp;
    l++;
    r--;
  }
}

function improveTwoOpt(order, maxPasses, window) {
  const n = order.length;
  if (n < 4) return order;
  const limit = n <= 600 ? n - 1 : Math.min(window, n - 1);

  for (let pass = 0; pass < maxPasses; pass++) {
    let improved = false;
    for (let i = 0; i < n - 1; i++) {
      const a = i === 0 ? null : order[i - 1];
      const b = order[i];
      const maxK = Math.min(n - 1, i + limit);
      for (let k = i + 1; k <= maxK; k++) {
      const c = order[k];
      const d = k === n - 1 ? null : order[k + 1];

        const before = (i === 0 ? b._dist : distPoints(a, b)) +
                       (k === n - 1 ? c._dist : distPoints(c, d));
        const after = (i === 0 ? c._dist : distPoints(a, c)) +
                      (k === n - 1 ? b._dist : distPoints(b, d));

        if (after + 1e-12 < before) {
          reverseRange(order, i, k);
          improved = true;
        }
      }
    }
    if (!improved) break;
  }
  return order;
}

function canonicalize(order) {
  return order.map(p => p.id);
}

const startNearest = packages.reduce((best, p, idx) => {
  if (best < 0 || packages[idx]._dist < packages[best]._dist) return idx;
  return best;
}, -1);

const startFarthest = packages.reduce((best, p, idx) => {
  if (best < 0 || packages[idx]._dist > packages[best]._dist) return idx;
  return best;
}, -1);

const candidates = [
  improveTwoOpt(buildNearestInsertion(startNearest), 2, 64),
  improveTwoOpt(buildNearestInsertion(startFarthest), 2, 64),
  improveTwoOpt(buildAngleOrder(), 3, 80),
  improveTwoOpt(buildXOrder(), 3, 80),
  improveTwoOpt(buildYOrder(), 3, 80),
  improveTwoOpt(buildPcaOrder(), 3, 80),
];

let bestOrder = candidates[0];
let bestCost = routeCost(bestOrder);
for (let i = 1; i < candidates.length; i++) {
  const cost = routeCost(candidates[i]);
  if (cost < bestCost - 1e-9) {
    bestCost = cost;
    bestOrder = candidates[i];
  }
}

process.stdout.write(JSON.stringify({ routes: [canonicalize(bestOrder)] }));
