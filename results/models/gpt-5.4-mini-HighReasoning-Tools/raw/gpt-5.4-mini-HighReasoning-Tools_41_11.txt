; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %uv "uv"
OpName %outColor "outColor"
OpName %Ubo "Ubo"
OpName %ubo "ubo"
OpMemberName %Ubo 0 "model"
OpMemberName %Ubo 1 "view"
OpMemberName %Ubo 2 "projection"
OpMemberName %Ubo 3 "lightPos"
OpMemberName %Ubo 4 "cameraPos"
OpMemberName %Ubo 5 "params"
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %outColor Location 0
OpDecorate %Ubo Block
OpMemberDecorate %Ubo 0 ColMajor
OpMemberDecorate %Ubo 0 MatrixStride 16
OpMemberDecorate %Ubo 0 Offset 0
OpMemberDecorate %Ubo 1 ColMajor
OpMemberDecorate %Ubo 1 MatrixStride 16
OpMemberDecorate %Ubo 1 Offset 64
OpMemberDecorate %Ubo 2 ColMajor
OpMemberDecorate %Ubo 2 MatrixStride 16
OpMemberDecorate %Ubo 2 Offset 128
OpMemberDecorate %Ubo 3 Offset 192
OpMemberDecorate %Ubo 4 Offset 208
OpMemberDecorate %Ubo 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
%Ubo = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_Ubo = OpTypePointer Uniform %Ubo
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%_ptr_Uniform_v3float = OpTypePointer Uniform %v3float
%glsl = OpExtInstImport "GLSL.std.450"
%c0 = OpConstant %int 0
%c1 = OpConstant %int 1
%c2 = OpConstant %int 2
%c3 = OpConstant %int 3
%f0 = OpConstant %float 0.0
%f1 = OpConstant %float 1.0
%f03 = OpConstant %float 0.3
%f05 = OpConstant %float 0.5
%f06 = OpConstant %float 6.0
%f07 = OpConstant %float 0.7
%f08 = OpConstant %float 0.8
%f01 = OpConstant %float 0.1
%yellow = OpConstantComposite %v3float %f1 %f08 %f0
%blue = OpConstantComposite %v3float %f01 %f01 %f03
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%uv = OpVariable %_ptr_Input_v2float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_Ubo Uniform
%main = OpFunction %void None %func
%label = OpLabel
%worldPosVal = OpLoad %v3float %worldPos
%normalVal = OpLoad %v3float %normal
%uvVal = OpLoad %v2float %uv
%uvScaled = OpVectorTimesScalar %v2float %uvVal %f06
%cell = OpExtInst %v2float %glsl Floor %uvScaled
%local = OpFSub %v2float %uvScaled %cell
%center = OpCompositeConstruct %v2float %f05 %f05
%delta2 = OpFSub %v2float %local %center
%dist = OpExtInst %float %glsl Length %delta2
%isDot = OpFOrdLessThanEqual %bool %dist %f03
%baseColor = OpSelect %v3float %isDot %yellow %blue
%lightPosPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %c3
%lightPos4 = OpLoad %v4float %lightPosPtr
%lightPos3 = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2
%lightDirRaw = OpFSub %v3float %lightPos3 %worldPosVal
%lightDir = OpExtInst %v3float %glsl Normalize %lightDirRaw
%normalN = OpExtInst %v3float %glsl Normalize %normalVal
%diffuse0 = OpDot %float %normalN %lightDir
%diffuse = OpExtInst %float %glsl FMax %diffuse0 %f0
%shadeTerm = OpFMul %float %diffuse %f07
%shade = OpFAdd %float %f03 %shadeTerm
%shadedRgb = OpVectorTimesScalar %v3float %baseColor %shade
%outVec = OpCompositeConstruct %v4float %shadedRgb %f1
OpStore %outColor %outVec
OpReturn
OpFunctionEnd
