; SPIR-V
; Version: 1.0
; Bound: 64
; Schema: 0

OpCapability Geometry
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; -------- Types / Consts --------
%void = OpTypeVoid
%fn = OpTypeFunction %void
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3

%float_0_7 = OpConstant %float 0.7
%float_inv3 = OpConstant %float 0.333333343
%v3_s = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_0_7
%v3_inv3 = OpConstantComposite %v3float %float_inv3 %float_inv3 %float_inv3

; -------- Interfaces --------
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%inWorldPos = OpVariable %ptr_in_arr_v3 Input
%inNormal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

; -------- Main --------
%main = OpFunction %void None %fn
%entry = OpLabel

; Load gl_in positions
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr

; xyz
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

; centroid C = (p0 + p1 + p2) / 3
%sum01 = OpFAdd %v3float %p0_xyz %p1_xyz
%sum = OpFAdd %v3float %sum01 %p2_xyz
%C = OpFMul %v3float %sum %v3_inv3

; ---- Emit vertex 0 ----
%c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0 = OpLoad %v3float %c0_ptr
%d0 = OpFSub %v3float %p0_xyz %C
%sd0 = OpFMul %v3float %d0 %v3_s
%np0 = OpFAdd %v3float %C %sd0
%np0_x = OpCompositeExtract %float %np0 0
%np0_y = OpCompositeExtract %float %np0 1
%np0_z = OpCompositeExtract %float %np0 2
%p0_w = OpCompositeExtract %float %p0 3
%np0_v4 = OpCompositeConstruct %v4float %np0_x %np0_y %np0_z %p0_w
%outpos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outpos_ptr %np0_v4
OpStore %outColor %c0
OpEmitVertex

; ---- Emit vertex 1 ----
%c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1 = OpLoad %v3float %c1_ptr
%d1 = OpFSub %v3float %p1_xyz %C
%sd1 = OpFMul %v3float %d1 %v3_s
%np1 = OpFAdd %v3float %C %sd1
%np1_x = OpCompositeExtract %float %np1 0
%np1_y = OpCompositeExtract %float %np1 1
%np1_z = OpCompositeExtract %float %np1 2
%p1_w = OpCompositeExtract %float %p1 3
%np1_v4 = OpCompositeConstruct %v4float %np1_x %np1_y %np1_z %p1_w
OpStore %outpos_ptr %np1_v4
OpStore %outColor %c1
OpEmitVertex

; ---- Emit vertex 2 ----
%c2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
%c2 = OpLoad %v3float %c2_ptr
%d2 = OpFSub %v3float %p2_xyz %C
%sd2 = OpFMul %v3float %d2 %v3_s
%np2 = OpFAdd %v3float %C %sd2
%np2_x = OpCompositeExtract %float %np2 0
%np2_y = OpCompositeExtract %float %np2 1
%np2_z = OpCompositeExtract %float %np2 2
%p2_w = OpCompositeExtract %float %p2 3
%np2_v4 = OpCompositeConstruct %v4float %np2_x %np2_y %np2_z %p2_w
OpStore %outpos_ptr %np2_v4
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
