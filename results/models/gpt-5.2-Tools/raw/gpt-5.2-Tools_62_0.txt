// Note: Saved as .txt for the benchmark harness; this is plain Node.js code (TypeScript-compatible).
// Avoid TS-only syntax so it runs directly under Node.
const fs = require("fs");

function idx(x, y, w) {
  return y * w + x;
}

function inBounds(x, y, w, h) {
  return x >= 0 && y >= 0 && x < w && y < h;
}

function bfsDistancesFromSources(
  w,
  h,
  forest,
  sources
) {
  const n = w * h;
  const dist = new Int32Array(n);
  dist.fill(-1);
  const q = new Int32Array(n);
  let qh = 0,
    qt = 0;
  for (let i = 0; i < sources.length; i++) {
    const s = sources[i];
    dist[s] = 0;
    q[qt++] = s;
  }
  while (qh < qt) {
    const v = q[qh++];
    const d = dist[v] + 1;
    const x = v % w;
    const y = (v / w) | 0;
    // 4-neighbors
    let nx = x - 1,
      ny = y;
    if (nx >= 0) {
      const ni = v - 1;
      if (forest[ni] === 1 && dist[ni] < 0) {
        dist[ni] = d;
        q[qt++] = ni;
      }
    }
    nx = x + 1;
    if (nx < w) {
      const ni = v + 1;
      if (forest[ni] === 1 && dist[ni] < 0) {
        dist[ni] = d;
        q[qt++] = ni;
      }
    }
    ny = y - 1;
    if (ny >= 0) {
      const ni = v - w;
      if (forest[ni] === 1 && dist[ni] < 0) {
        dist[ni] = d;
        q[qt++] = ni;
      }
    }
    ny = y + 1;
    if (ny < h) {
      const ni = v + w;
      if (forest[ni] === 1 && dist[ni] < 0) {
        dist[ni] = d;
        q[qt++] = ni;
      }
    }
  }
  return dist;
}

// Fire reachability with some cuts applied: multi-source BFS on remaining forest
function reachableWithCuts(
  w,
  h,
  forest,
  ignitionIdx,
  cut
) {
  const n = w * h;
  const vis = new Uint8Array(n);
  const q = new Int32Array(n);
  let qh = 0,
    qt = 0;
  for (let i = 0; i < ignitionIdx.length; i++) {
    const s = ignitionIdx[i];
    if (forest[s] === 1 && cut[s] === 0 && vis[s] === 0) {
      vis[s] = 1;
      q[qt++] = s;
    }
  }
  while (qh < qt) {
    const v = q[qh++];
    const x = v % w;
    const y = (v / w) | 0;
    // neighbors
    if (x > 0) {
      const ni = v - 1;
      if (forest[ni] === 1 && cut[ni] === 0 && vis[ni] === 0) {
        vis[ni] = 1;
        q[qt++] = ni;
      }
    }
    if (x + 1 < w) {
      const ni = v + 1;
      if (forest[ni] === 1 && cut[ni] === 0 && vis[ni] === 0) {
        vis[ni] = 1;
        q[qt++] = ni;
      }
    }
    if (y > 0) {
      const ni = v - w;
      if (forest[ni] === 1 && cut[ni] === 0 && vis[ni] === 0) {
        vis[ni] = 1;
        q[qt++] = ni;
      }
    }
    if (y + 1 < h) {
      const ni = v + w;
      if (forest[ni] === 1 && cut[ni] === 0 && vis[ni] === 0) {
        vis[ni] = 1;
        q[qt++] = ni;
      }
    }
  }
  return vis;
}

// Reconstruct a shortest-path corridor from `target` back to any source using `distFromIgn`
// Returns list of indices along the path excluding target itself? We'll include intermediate cells only.
function reconstructPathToSource(
  w,
  h,
  forest,
  distFromIgn,
  target,
  maxLen
) {
  const path = [];
  let cur = target;
  let d = distFromIgn[cur];
  if (d < 0) return path;
  // Walk down distances until 0; cap length to keep runtime bounded on huge maps
  let steps = 0;
  while (d > 0 && steps < maxLen) {
    const x = cur % w;
    const y = (cur / w) | 0;
    let next = -1;
    // choose a predecessor with dist d-1 (any)
    if (x > 0) {
      const ni = cur - 1;
      if (forest[ni] === 1 && distFromIgn[ni] === d - 1) next = ni;
    }
    if (next < 0 && x + 1 < w) {
      const ni = cur + 1;
      if (forest[ni] === 1 && distFromIgn[ni] === d - 1) next = ni;
    }
    if (next < 0 && y > 0) {
      const ni = cur - w;
      if (forest[ni] === 1 && distFromIgn[ni] === d - 1) next = ni;
    }
    if (next < 0 && y + 1 < h) {
      const ni = cur + w;
      if (forest[ni] === 1 && distFromIgn[ni] === d - 1) next = ni;
    }
    if (next < 0) break;
    // cur -> next; record cur as on-path (excluding ignition cell, include town cell handled elsewhere)
    path.push(cur);
    cur = next;
    d--;
    steps++;
  }
  return path;
}

function pickFirebreaks(data) {
  const w = data.width | 0;
  const h = data.height | 0;
  const n = w * h;
  const forest = new Uint8Array(n);
  for (let y = 0; y < h; y++) {
    const row = data.grid[y] || "";
    for (let x = 0; x < w; x++) {
      const c = row.charCodeAt(x);
      // '.' = 46, '~' = 126
      forest[idx(x, y, w)] = c === 46 ? 1 : 0;
    }
  }

  const ignitionIdx = [];
  for (let i = 0; i < data.ignitions.length; i++) {
    const [x, y] = data.ignitions[i];
    if (inBounds(x, y, w, h)) ignitionIdx.push(idx(x, y, w));
  }

  let budget = Math.max(0, data.cutBudget | 0);
  if (budget === 0 || ignitionIdx.length === 0) return [];

  // Precompute distances from ignitions on the original forest (no cuts). Used to bias toward likely corridors.
  const distFromIgn = bfsDistancesFromSources(w, h, forest, ignitionIdx);

  // Consider only towns that are on forest cells and initially reachable.
  const towns = [];
  for (let i = 0; i < data.towns.length; i++) {
    const t = data.towns[i];
    const x = t.x | 0;
    const y = t.y | 0;
    if (!inBounds(x, y, w, h)) continue;
    const ti = idx(x, y, w);
    if (forest[ti] !== 1) continue;
    const d = distFromIgn[ti];
    if (d < 0) continue; // never burns anyway
    const v = +t.value;
    if (!(v > 0)) continue;
    towns.push({ x, y, v, i: ti, d });
  }
  if (towns.length === 0) return [];

  // Prioritize towns by value discounted by distance (closer towns matter more).
  towns.sort((a, b) => {
    const sa = a.v / (1 + a.d);
    const sb = b.v / (1 + b.d);
    return sb - sa;
  });

  // Work with a limited number of top towns for scalability.
  const MAX_TOWNS = 300;
  if (towns.length > MAX_TOWNS) towns.length = MAX_TOWNS;

  // Cut set
  const cut = new Uint8Array(n);
  const firebreaks = [];

  // Iteratively: simulate reachability with current cuts, then try to block paths to high-value towns.
  // Keep iterations bounded.
  const MAX_ITERS = Math.min(25, budget);
  for (let iter = 0; iter < MAX_ITERS && budget > 0; iter++) {
    const reach = reachableWithCuts(w, h, forest, ignitionIdx, cut);

    // Build candidate scores from a set of corridor cells toward currently endangered towns.
    const score = new Float64Array(n);
    const seenTown = new Uint8Array(towns.length);

    // Only sample limited path length per town to keep cost low.
    const PATH_CAP = 800;
    for (let ti = 0; ti < towns.length; ti++) {
      const t = towns[ti];
      if (reach[t.i] === 0) continue; // already safe
      seenTown[ti] = 1;
      const weight = t.v / (1 + t.d);
      const path = reconstructPathToSource(w, h, forest, distFromIgn, t.i, PATH_CAP);
      for (let k = 0; k < path.length; k++) {
        const p = path[k];
        if (cut[p] === 0) score[p] += weight;
      }
    }

    // Pick best-scoring cell to cut that is forest, not already cut, not an ignition, and preferably closer to ignitions.
    let best = -1;
    let bestScore = 0;
    for (let i = 0; i < n; i++) {
      if (score[i] <= bestScore) continue;
      if (forest[i] !== 1 || cut[i] !== 0) continue;
      if (distFromIgn[i] < 0) continue;
      // avoid cutting ignition cells directly (though allowed, it may be ineffective if multiple ignitions)
      // Use reach[i] to focus on frontier; cutting unreachable cells doesn't help.
      if (reach[i] === 0) continue;
      best = i;
      bestScore = score[i];
    }

    if (best < 0 || bestScore <= 0) break;
    cut[best] = 1;
    budget--;
    firebreaks.push([best % w, (best / w) | 0]);

    // If budget remains, opportunistically cut adjacent cells around the chosen corridor cell when they are also high score.
    // This helps form a small barrier without global optimization.
    if (budget > 0) {
      const bx = best % w;
      const by = (best / w) | 0;
      const neigh = [];
      if (bx > 0) neigh.push(best - 1);
      if (bx + 1 < w) neigh.push(best + 1);
      if (by > 0) neigh.push(best - w);
      if (by + 1 < h) neigh.push(best + w);
      // pick up to 2 neighbors by score
      neigh.sort((a, b) => score[b] - score[a]);
      for (let k = 0; k < neigh.length && budget > 0 && k < 2; k++) {
        const ni = neigh[k];
        if (score[ni] <= bestScore * 0.25) break;
        if (forest[ni] === 1 && cut[ni] === 0) {
          cut[ni] = 1;
          budget--;
          firebreaks.push([ni % w, (ni / w) | 0]);
        }
      }
    }
  }

  // Ensure we don't exceed original budget even with neighbor adds.
  if (firebreaks.length > (data.cutBudget | 0)) firebreaks.length = data.cutBudget | 0;
  return firebreaks;
}

function main() {
  let input = fs.readFileSync(0, "utf8");
  // Strip UTF-8 BOM if present
  if (input.length > 0 && input.charCodeAt(0) === 0xfeff) input = input.slice(1);
  const data = JSON.parse(input);
  const firebreaks = pickFirebreaks(data);
  process.stdout.write(JSON.stringify({ firebreaks }));
}

main();
