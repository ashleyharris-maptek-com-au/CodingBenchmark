```cpp
#include <bits/stdc++.h>
using namespace std;

// Solves: maximize sum c[i]*x[i] subject to x[u]+x[v] <= 1 for each edge (u,v), x in {0,1}.
// This is Maximum Weight Independent Set on a general graph; solved exactly with branch-and-bound.

struct Item {
    int v;
    long long w;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;
    vector<long long> c(n);
    for (int i = 0; i < n; i++) cin >> c[i];

    vector<vector<int>> adj(n);
    adj.assign(n, {});
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue; // redundant
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    // Work on a filtered graph: any vertex with w<=0 is never beneficial to include (constraints only reduce options).
    vector<int> old_to_new(n, -1), new_to_old;
    for (int i = 0; i < n; i++) {
        if (c[i] > 0) {
            old_to_new[i] = (int)new_to_old.size();
            new_to_old.push_back(i);
        }
    }

    int N = (int)new_to_old.size();
    vector<long long> w(N);
    for (int i = 0; i < N; i++) w[i] = c[new_to_old[i]];

    // Build adjacency bitmasks for N<=64 fast path, else use vector<uint64_t>.
    const int WORD = 64;
    int words = (N + WORD - 1) / WORD;
    vector<vector<uint64_t>> nbh(N, vector<uint64_t>(words, 0));

    for (int i = 0; i < N; i++) {
        int oi = new_to_old[i];
        for (int oj : adj[oi]) {
            int j = (oj >= 0 && oj < n) ? old_to_new[oj] : -1;
            if (j < 0) continue;
            nbh[i][j / WORD] |= (uint64_t(1) << (j % WORD));
        }
    }

    auto popcount_vec = [&](const vector<uint64_t>& a) -> int {
        int s = 0;
        for (uint64_t x : a) s += __builtin_popcountll(x);
        return s;
    };

    auto andnot_into = [&](vector<uint64_t>& dst, const vector<uint64_t>& a, const vector<uint64_t>& b) {
        // dst = a & ~b
        for (int k = 0; k < words; k++) dst[k] = a[k] & ~b[k];
    };

    auto any_nonzero = [&](const vector<uint64_t>& a) -> bool {
        for (uint64_t x : a) if (x) return true;
        return false;
    };

    auto clear_bit = [&](vector<uint64_t>& a, int idx) {
        a[idx / WORD] &= ~(uint64_t(1) << (idx % WORD));
    };

    auto test_bit = [&](const vector<uint64_t>& a, int idx) -> bool {
        return (a[idx / WORD] >> (idx % WORD)) & 1ULL;
    };

    auto first_bit = [&](const vector<uint64_t>& a) -> int {
        for (int k = 0; k < words; k++) {
            if (a[k]) {
                return k * WORD + __builtin_ctzll(a[k]);
            }
        }
        return -1;
    };

    auto intersect_count = [&](const vector<uint64_t>& a, const vector<uint64_t>& b) -> int {
        int s = 0;
        for (int k = 0; k < words; k++) s += __builtin_popcountll(a[k] & b[k]);
        return s;
    };

    // Initial candidate set P = all vertices in filtered graph.
    vector<uint64_t> P(words, 0);
    for (int i = 0; i < N; i++) P[i / WORD] |= (uint64_t(1) << (i % WORD));

    vector<int> best_choice(N, 0), cur_choice(N, 0);
    long long best = 0;

    // Upper bound: greedy fractional coloring / clique cover approximation via degeneracy ordering:
    // We'll use a fast bound: sum of top weights in each independent "bucket" built by greedy coloring on complement
    // is complex; instead use a simpler admissible bound:
    // - Get list of remaining vertices sorted by weight, and run greedy matching of conflicts? That's not a bound.
    // We use a valid bound from greedy graph coloring on the subgraph (Tomita-style):
    // Assign vertices to color classes where each class is an independent set in the subgraph, so at most one from each class
    // can be chosen? Actually in MWIS, within an independent set you can take many, so this is not a bound.
    // Instead, color the *graph* (not complement): each color class is an independent set, and an independent set can take all
    // vertices in a color class, so sum of all weights is an upper bound but too weak.
    // We'll use a different valid bound: solve LP relaxation for this special case via "fractional packing" is hard.
    // Practical BnB: bound = current + sum(weights of remaining), which is valid. We'll strengthen with pruning on low weights
    // via ordering and include/exclude branching.

    // Precompute weights sum for quick bound updates during recursion.
    vector<long long> wsum_prefix; // not used; we'll compute remaining sum via list.

    // For branching heuristic: pick vertex with maximum weight * degree among remaining.
    function<void(const vector<uint64_t>&, long long, long long)> dfs;
    dfs = [&](const vector<uint64_t>& Pset, long long cur, long long sumRemaining) {
        if (cur + sumRemaining <= best) return;
        if (!any_nonzero(Pset)) {
            if (cur > best) {
                best = cur;
                best_choice = cur_choice;
            }
            return;
        }

        // Select branching vertex.
        int pick = -1;
        long long bestScore = -1;
        for (int i = 0; i < N; i++) {
            if (!test_bit(Pset, i)) continue;
            int deg = intersect_count(Pset, nbh[i]);
            long long score = w[i] * 10LL + deg; // weight-primary
            if (score > bestScore) {
                bestScore = score;
                pick = i;
            }
        }
        if (pick < 0) return;

        // Branch 1: include pick.
        {
            vector<uint64_t> P2(words, 0);
            // P2 = Pset & ~N[pick] & ~{pick}
            vector<uint64_t> nmask = nbh[pick];
            // also remove pick itself
            nmask[pick / WORD] |= (uint64_t(1) << (pick % WORD));
            andnot_into(P2, Pset, nmask);

            long long removedSum = 0;
            // compute sum weights removed when going from Pset to P2
            for (int i = 0; i < N; i++) {
                bool in1 = test_bit(Pset, i);
                bool in2 = test_bit(P2, i);
                if (in1 && !in2) removedSum += w[i];
            }

            cur_choice[pick] = 1;
            dfs(P2, cur + w[pick], sumRemaining - removedSum);
            cur_choice[pick] = 0;
        }

        // Branch 2: exclude pick.
        {
            vector<uint64_t> P2 = Pset;
            clear_bit(P2, pick);
            dfs(P2, cur, sumRemaining - w[pick]);
        }
    };

    long long totalRemaining = 0;
    for (int i = 0; i < N; i++) totalRemaining += w[i];

    dfs(P, 0, totalRemaining);

    // Build full solution for original n variables.
    vector<int> x(n, 0);
    long long obj = 0;
    for (int i = 0; i < N; i++) {
        if (best_choice[i]) {
            int oi = new_to_old[i];
            x[oi] = 1;
            obj += c[oi];
        }
    }

    // Safety: ensure feasibility (should be by construction). If any negative weights were filtered, they're 0.
    for (int u = 0; u < n; u++) {
        if (!x[u]) continue;
        for (int v : adj[u]) {
            if (v >= 0 && v < n && x[v]) {
                // If something went wrong, fall back to trivial feasible all-zero.
                fill(x.begin(), x.end(), 0);
                obj = 0;
                u = n;
                break;
            }
        }
    }

    cout << obj << "\n";
    for (int i = 0; i < n; i++) {
        if (i) cout << ' ';
        cout << x[i];
    }
    cout << "\n";
    return 0;
}
```
