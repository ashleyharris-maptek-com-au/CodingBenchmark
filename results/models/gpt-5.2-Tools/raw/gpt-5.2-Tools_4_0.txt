static constexpr double PI = 3.1415926535897932384626433832795;

struct Vec3 {
  double x, y, z;
  Vec3() : x(0), y(0), z(0) {}
  Vec3(double X, double Y, double Z) : x(X), y(Y), z(Z) {}
  Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
  Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
  Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
  Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
};

static inline double dot(const Vec3& a, const Vec3& b) { return a.x * b.x + a.y * b.y + a.z * b.z; }
static inline Vec3 cross(const Vec3& a, const Vec3& b) {
  return {a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}
static inline double norm2(const Vec3& v) { return dot(v, v); }
static inline double norm(const Vec3& v) { return std::sqrt(norm2(v)); }
static inline Vec3 minV(const Vec3& a, const Vec3& b) {
  return {std::min(a.x, b.x), std::min(a.y, b.y), std::min(a.z, b.z)};
}
static inline Vec3 maxV(const Vec3& a, const Vec3& b) {
  return {std::max(a.x, b.x), std::max(a.y, b.y), std::max(a.z, b.z)};
}

struct Plane {
  Vec3 n;   // outward normal
  double d; // dot(n, x) <= d is inside
};

struct Face {
  int a, b, c;
  Plane p;
};

static std::vector<Face> convexHull3D(const std::vector<Vec3>& pts) {
  const int n = (int)pts.size();
  std::vector<Face> faces;
  if (n < 4) return faces;

  int i0 = 0, i1 = -1, i2 = -1, i3 = -1;
  for (int i = 1; i < n; i++) {
    if (norm2(pts[i] - pts[i0]) > 0) {
      i1 = i;
      break;
    }
  }
  if (i1 < 0) return faces;
  for (int i = 0; i < n; i++) {
    if (i == i0 || i == i1) continue;
    Vec3 a = pts[i1] - pts[i0];
    Vec3 b = pts[i] - pts[i0];
    if (norm2(cross(a, b)) > 0) {
      i2 = i;
      break;
    }
  }
  if (i2 < 0) return faces;
  for (int i = 0; i < n; i++) {
    if (i == i0 || i == i1 || i == i2) continue;
    Vec3 a = pts[i1] - pts[i0];
    Vec3 b = pts[i2] - pts[i0];
    Vec3 c = pts[i] - pts[i0];
    double vol6 = dot(cross(a, b), c);
    if (std::fabs(vol6) > 0) {
      i3 = i;
      break;
    }
  }
  if (i3 < 0) return faces;

  Vec3 inner = (pts[i0] + pts[i1] + pts[i2] + pts[i3]) / 4.0;

  auto makeFace = [&](int a, int b, int c) -> Face {
    Vec3 nrm = cross(pts[b] - pts[a], pts[c] - pts[a]);
    double len = norm(nrm);
    if (len == 0) return {a, b, c, {{0, 0, 0}, 0}};
    nrm = nrm / len;
    double dd = dot(nrm, pts[a]);
    if (dot(nrm, inner) > dd) {
      std::swap(b, c);
      nrm = nrm * -1.0;
      dd = dot(nrm, pts[a]);
    }
    return {a, b, c, {nrm, dd}};
  };

  faces.push_back(makeFace(i0, i1, i2));
  faces.push_back(makeFace(i0, i3, i1));
  faces.push_back(makeFace(i0, i2, i3));
  faces.push_back(makeFace(i1, i3, i2));

  auto visible = [&](const Face& f, int pi) -> bool {
    const Vec3& p = pts[pi];
    double sc = std::max({std::fabs(f.p.d), std::fabs(p.x), std::fabs(p.y), std::fabs(p.z), 1.0});
    return dot(f.p.n, p) > f.p.d + 1e-10 * sc;
  };

  for (int pi = 0; pi < n; pi++) {
    if (pi == i0 || pi == i1 || pi == i2 || pi == i3) continue;
    std::vector<int> vis;
    vis.reserve(faces.size());
    for (int fi = 0; fi < (int)faces.size(); fi++) {
      if (visible(faces[fi], pi)) vis.push_back(fi);
    }
    if (vis.empty()) continue;

    struct EdgeKey {
      int u, v;
      bool operator==(const EdgeKey& o) const { return u == o.u && v == o.v; }
    };
    struct EdgeKeyHash {
      size_t operator()(const EdgeKey& e) const noexcept {
        return (size_t)e.u * 1315423911u ^ (size_t)e.v * 2654435761u;
      }
    };
    std::unordered_map<EdgeKey, int, EdgeKeyHash> edgeCount;
    edgeCount.reserve(vis.size() * 6);
    auto addEdge = [&](int u, int v) {
      EdgeKey k{u, v};
      edgeCount[k] += 1;
      EdgeKey kr{v, u};
      auto it = edgeCount.find(kr);
      if (it != edgeCount.end()) {
        if (--(it->second) <= 0) edgeCount.erase(it);
        if (--(edgeCount[k]) <= 0) edgeCount.erase(k);
      }
    };

    std::vector<char> isVis(faces.size(), 0);
    for (int idx : vis) isVis[idx] = 1;
    for (int idx : vis) {
      const Face& f = faces[idx];
      addEdge(f.a, f.b);
      addEdge(f.b, f.c);
      addEdge(f.c, f.a);
    }

    std::vector<Face> newFaces;
    newFaces.reserve(faces.size() + edgeCount.size());
    for (int fi = 0; fi < (int)faces.size(); fi++) {
      if (!isVis[fi]) newFaces.push_back(faces[fi]);
    }

    inner = (inner * 0.75) + (pts[pi] * 0.25);
    for (auto& kv : edgeCount) {
      const EdgeKey& e = kv.first;
      newFaces.push_back(makeFace(e.u, e.v, pi));
    }
    faces.swap(newFaces);
  }

  return faces;
}

static bool pointInsideConvex(const std::vector<Plane>& planes, const Vec3& p, double tol) {
  for (const auto& pl : planes) {
    if (dot(pl.n, p) > pl.d + tol) return false;
  }
  return true;
}

static std::array<std::array<double, 3>, 3> eulerZYX(double rx, double ry, double rz) {
  double cx = std::cos(rx), sx = std::sin(rx);
  double cy = std::cos(ry), sy = std::sin(ry);
  double cz = std::cos(rz), sz = std::sin(rz);
  return {{
      {cz * cy, cz * sy * sx - sz * cx, cz * sy * cx + sz * sx},
      {sz * cy, sz * sy * sx + cz * cx, sz * sy * cx - cz * sx},
      {-sy, cy * sx, cy * cx},
  }};
}

static Vec3 mul(const std::array<std::array<double, 3>, 3>& R, const Vec3& v) {
  return {R[0][0] * v.x + R[0][1] * v.y + R[0][2] * v.z,
          R[1][0] * v.x + R[1][1] * v.y + R[1][2] * v.z,
          R[2][0] * v.x + R[2][1] * v.y + R[2][2] * v.z};
}

struct TetraPlacement {
  Vec3 c;
  double rx, ry, rz;
  Vec3 aabbMin, aabbMax;
};

static std::array<Vec3, 4> canonicalTetra(double e) {
  std::array<Vec3, 4> v = {Vec3(1, 1, 1), Vec3(1, -1, -1), Vec3(-1, 1, -1), Vec3(-1, -1, 1)};
  double scale = e / (2.0 * std::sqrt(2.0));
  for (auto& p : v) p = p * scale;
  return v;
}

static void computeAABB(const std::array<Vec3, 4>& verts, Vec3& mn, Vec3& mx) {
  mn = verts[0];
  mx = verts[0];
  for (int i = 1; i < 4; i++) {
    mn = minV(mn, verts[i]);
    mx = maxV(mx, verts[i]);
  }
}

static bool satOverlapTetra(const std::array<Vec3, 4>& A, const std::array<Vec3, 4>& B) {
  auto proj = [](const std::array<Vec3, 4>& T, const Vec3& ax, double& mn, double& mx) {
    mn = mx = dot(T[0], ax);
    for (int i = 1; i < 4; i++) {
      double v = dot(T[i], ax);
      mn = std::min(mn, v);
      mx = std::max(mx, v);
    }
  };
  auto sepAxis = [&](const Vec3& ax) -> bool {
    double l2 = norm2(ax);
    if (l2 < 1e-18) return false;
    Vec3 a = ax / std::sqrt(l2);
    double a0, a1, b0, b1;
    proj(A, a, a0, a1);
    proj(B, a, b0, b1);
    double eps = 1e-12 * (1.0 + std::max({std::fabs(a0), std::fabs(a1), std::fabs(b0), std::fabs(b1)}));
    return (a1 < b0 - eps) || (b1 < a0 - eps);
  };

  static const int F[4][3] = {{0, 1, 2}, {0, 3, 1}, {0, 2, 3}, {1, 3, 2}};
  for (int fi = 0; fi < 4; fi++) {
    Vec3 nA = cross(A[F[fi][1]] - A[F[fi][0]], A[F[fi][2]] - A[F[fi][0]]);
    if (sepAxis(nA)) return false;
    Vec3 nB = cross(B[F[fi][1]] - B[F[fi][0]], B[F[fi][2]] - B[F[fi][0]]);
    if (sepAxis(nB)) return false;
  }

  static const int E[6][2] = {{0, 1}, {0, 2}, {0, 3}, {1, 2}, {1, 3}, {2, 3}};
  for (int ea = 0; ea < 6; ea++) {
    Vec3 a = A[E[ea][1]] - A[E[ea][0]];
    for (int eb = 0; eb < 6; eb++) {
      Vec3 b = B[E[eb][1]] - B[E[eb][0]];
      Vec3 ax = cross(a, b);
      if (sepAxis(ax)) return false;
    }
  }
  return true;
}

struct CellKey {
  int64_t x, y, z;
  bool operator==(const CellKey& o) const { return x == o.x && y == o.y && z == o.z; }
};
struct CellKeyHash {
  size_t operator()(const CellKey& k) const noexcept {
    size_t h = (size_t)k.x;
    h ^= (size_t)k.y + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
    h ^= (size_t)k.z + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
    return h;
  }
};
static CellKey cellOf(const Vec3& p, double invCell) {
  return {(int64_t)std::floor(p.x * invCell), (int64_t)std::floor(p.y * invCell),
          (int64_t)std::floor(p.z * invCell)};
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int V;
  double e;
  if (!(std::cin >> V >> e)) return 0;
  std::vector<Vec3> pts;
  pts.reserve(std::max(0, V));
  for (int i = 0; i < V; i++) {
    double x, y, z;
    std::cin >> x >> y >> z;
    pts.emplace_back(x, y, z);
  }
  if (pts.size() < 4 || e <= 0) {
    std::cout << 0 << "\n";
    return 0;
  }

  Vec3 mn = pts[0], mx = pts[0];
  for (auto& p : pts) {
    mn = minV(mn, p);
    mx = maxV(mx, p);
  }
  Vec3 box = mx - mn;
  double diag = norm(box);

  std::vector<Face> faces = convexHull3D(pts);
  if (faces.empty()) {
    std::cout << 0 << "\n";
    return 0;
  }
  std::vector<Plane> planes;
  planes.reserve(faces.size());
  for (auto& f : faces) planes.push_back(f.p);

  struct Ori {
    double rx, ry, rz;
  };
  std::vector<Ori> oris;
  oris.push_back({0, 0, 0});
  oris.push_back({PI / 2, 0, 0});
  oris.push_back({0, PI / 2, 0});
  oris.push_back({0, 0, PI / 2});
  oris.push_back({PI / 2, PI / 2, 0});
  oris.push_back({PI / 3, PI / 5, PI / 7});

  auto base = canonicalTetra(e);
  double R = e * std::sqrt(3.0 / 8.0);
  double safety = 1e-9 * (1.0 + diag);

  double step = e * 0.85;
  double volumeAABB = std::max(0.0, box.x) * std::max(0.0, box.y) * std::max(0.0, box.z);
  if (volumeAABB > 1e6 * e * e * e) step = e * 1.2;
  if (volumeAABB > 1e10 * e * e * e) step = e * 2.5;
  if (volumeAABB > 1e14 * e * e * e) step = e * 5.0;

  double cell = std::max(step, 2.0 * R);
  double invCell = 1.0 / cell;

  auto buildTetra = [&](const Vec3& c, const Ori& o, std::array<Vec3, 4>& out, Vec3& aabbMin,
                        Vec3& aabbMax) {
    auto Rm = eulerZYX(o.rx, o.ry, o.rz);
    for (int i = 0; i < 4; i++) out[i] = mul(Rm, base[i]) + c;
    computeAABB(out, aabbMin, aabbMax);
  };

  auto tetraInside = [&](const std::array<Vec3, 4>& tv) -> bool {
    for (int i = 0; i < 4; i++) {
      if (!pointInsideConvex(planes, tv[i], 1e-10 * (1.0 + diag))) return false;
    }
    return true;
  };

  std::vector<TetraPlacement> placed;
  placed.reserve(1024);
  std::unordered_map<CellKey, std::vector<int>, CellKeyHash> grid;
  grid.reserve(4096);

  auto aabbOverlap = [](const Vec3& a0, const Vec3& a1, const Vec3& b0, const Vec3& b1) -> bool {
    if (a1.x < b0.x || b1.x < a0.x) return false;
    if (a1.y < b0.y || b1.y < a0.y) return false;
    if (a1.z < b0.z || b1.z < a0.z) return false;
    return true;
  };

  auto collides = [&](const Vec3& c, const std::array<Vec3, 4>& tv, const Vec3& aabbMin,
                      const Vec3& aabbMax) -> bool {
    CellKey ck = cellOf(c, invCell);
    for (int dx = -1; dx <= 1; dx++) {
      for (int dy = -1; dy <= 1; dy++) {
        for (int dz = -1; dz <= 1; dz++) {
          CellKey nk{ck.x + dx, ck.y + dy, ck.z + dz};
          auto it = grid.find(nk);
          if (it == grid.end()) continue;
          for (int idx : it->second) {
            const auto& p = placed[idx];
            Vec3 dc = p.c - c;
            double d2 = norm2(dc);
            double thr = (2.0 * R);
            if (d2 >= (thr * thr) - safety) continue;
            if (!aabbOverlap(aabbMin, aabbMax, p.aabbMin, p.aabbMax)) continue;
            std::array<Vec3, 4> other;
            Vec3 omin, omax;
            buildTetra(p.c, {p.rx, p.ry, p.rz}, other, omin, omax);
            if (satOverlapTetra(tv, other)) return true;
          }
        }
      }
    }
    return false;
  };

  auto insertPlacement = [&](const TetraPlacement& tp) {
    int idx = (int)placed.size();
    placed.push_back(tp);
    CellKey ck = cellOf(tp.c, invCell);
    grid[ck].push_back(idx);
  };

  int64_t nx = (int64_t)std::floor((box.x) / step);
  int64_t ny = (int64_t)std::floor((box.y) / step);
  int64_t nz = (int64_t)std::floor((box.z) / step);
  if (nx < 1) nx = 1;
  if (ny < 1) ny = 1;
  if (nz < 1) nz = 1;
  long double totalCells = (long double)(nx + 1) * (long double)(ny + 1) * (long double)(nz + 1);
  const long double MAX_EVAL = 2.5e6L;
  long double stride = 1.0L;
  if (totalCells > MAX_EVAL) stride = std::cbrt((double)(totalCells / MAX_EVAL));
  int s = (int)std::max<long double>(1, std::floor(stride));

  std::mt19937_64 rng(1234567);
  std::uniform_real_distribution<double> jit(-0.15 * step, 0.15 * step);

  std::array<Vec3, 4> tv;
  Vec3 tavMin, tavMax;

  for (int64_t iz = 0; iz <= nz; iz += s) {
    for (int64_t iy = 0; iy <= ny; iy += s) {
      for (int64_t ix = 0; ix <= nx; ix += s) {
        Vec3 c(mn.x + ix * step + jit(rng), mn.y + iy * step + jit(rng), mn.z + iz * step + jit(rng));
        if (!pointInsideConvex(planes, c, 1e-9 * (1.0 + diag))) continue;
        for (const auto& o : oris) {
          buildTetra(c, o, tv, tavMin, tavMax);
          if (!tetraInside(tv)) continue;
          if (collides(c, tv, tavMin, tavMax)) continue;
          insertPlacement({c, o.rx, o.ry, o.rz, tavMin, tavMax});
          break;
        }
      }
    }
  }

  std::cout.setf(std::ios::fixed);
  std::cout.precision(10);
  std::cout << placed.size() << "\n";
  for (auto& p : placed) {
    std::cout << p.c.x << " " << p.c.y << " " << p.c.z << " " << p.rx << " " << p.ry << " " << p.rz
              << "\n";
  }
  return 0;
}
