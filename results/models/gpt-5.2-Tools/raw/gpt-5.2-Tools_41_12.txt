%ptr_Input_v3float = OpTypePointer Input %v3float

%ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

%in_normal = OpVariable %ptr_Input_v3float Input
%outColor = OpVariable %ptr_Output_v4float Output
%ubo = OpVariable %ptr_Uniform_UBO Uniform

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%float_3 = OpConstant %float 3
%float_0_5 = OpConstant %float 0.5
%float_0_3 = OpConstant %float 0.3

%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_0_15 = OpConstant %float 0.15
%float_0_05 = OpConstant %float 0.05

%c_sky = OpConstantComposite %v3float %float_0_4 %float_0_6 %float_1
%c_ground = OpConstantComposite %v3float %float_0_3 %float_0_15 %float_0_05

%c_lightPos_index = OpConstant %int 3

%main = OpFunction %void None %fn_void
%entry = OpLabel

; n = normalize(normal)
%n_in = OpLoad %v3float %in_normal
%n = OpExtInst %v3float %glsl Normalize %n_in

; hemisphere factor: f = n.y*0.5 + 0.5
%ny = OpCompositeExtract %float %n 1
%ny_half = OpFMul %float %ny %float_0_5
%f = OpFAdd %float %ny_half %float_0_5

; hemi = mix(ground, sky, f) = ground*(1-f) + sky*f
%one_minus_f = OpFSub %float %float_1 %f
%ground_term = OpVectorTimesScalar %v3float %c_ground %one_minus_f
%sky_term = OpVectorTimesScalar %v3float %c_sky %f
%hemi = OpFAdd %v3float %ground_term %sky_term

; light direction: l = normalize(lightPos.xyz)
%lightPos_ptr = OpAccessChain %ptr_Uniform_v4float %ubo %c_lightPos_index
%lightPos4 = OpLoad %v4float %lightPos_ptr
%lightPos3 = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2
%l = OpExtInst %v3float %glsl Normalize %lightPos3

; d = max(dot(n,l), 0)
%ndotl = OpDot %float %n %l
%d = OpExtInst %float %glsl FMax %ndotl %float_0
%diff = OpFMul %float %d %float_0_3
%diff3 = OpCompositeConstruct %v3float %diff %diff %diff

; rgb = hemi + diff3
%rgb = OpFAdd %v3float %hemi %diff3
%rgb4a = OpCompositeConstruct %v4float %rgb %float_1

OpStore %outColor %rgb4a
OpReturn
