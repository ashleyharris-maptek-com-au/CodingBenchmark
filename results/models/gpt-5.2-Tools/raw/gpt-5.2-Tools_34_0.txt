import sys
import math


def dist(a, b):
    return math.hypot(a[0] - b[0], a[1] - b[1])


def route_cost(route, pts, depot):
    if not route:
        return 0.0
    c = dist(depot, pts[route[0]])
    for i in range(len(route) - 1):
        c += dist(pts[route[i]], pts[route[i + 1]])
    c += dist(pts[route[-1]], depot)
    return c


def two_opt(route, pts, depot, max_iters=2000):
    n = len(route)
    if n < 4:
        return route
    best = route
    iters = 0
    improved = True
    while improved and iters < max_iters:
        improved = False
        for i in range(n - 2):
            a = depot if i == 0 else pts[best[i - 1]]
            b = pts[best[i]]
            for k in range(i + 1, n - 1):
                c = pts[best[k]]
                d = depot if k == n - 1 else pts[best[k + 1]]
                old = dist(a, b) + dist(c, d)
                new = dist(a, c) + dist(b, d)
                if new + 1e-12 < old:
                    best = best[:i] + list(reversed(best[i : k + 1])) + best[k + 1 :]
                    improved = True
                    iters += 1
                    if iters >= max_iters:
                        break
            if iters >= max_iters:
                break
        n = len(best)
    return best


def build_routes_savings(n, k, cap, depot, pts, dem):
    # Clarke–Wright savings heuristic (parallel).
    routes = [[i] for i in range(n)]
    loads = [dem[i] for i in range(n)]
    where = list(range(n))

    d0 = [dist(depot, pts[i]) for i in range(n)]
    savings = []
    for i in range(n):
        for j in range(i + 1, n):
            s = d0[i] + d0[j] - dist(pts[i], pts[j])
            savings.append((s, i, j))
    savings.sort(reverse=True)

    for _, i, j in savings:
        ri = where[i]
        rj = where[j]
        if ri == rj:
            continue
        r1 = routes[ri]
        r2 = routes[rj]
        if not r1 or not r2:
            continue
        i_is_start = r1[0] == i
        i_is_end = r1[-1] == i
        j_is_start = r2[0] == j
        j_is_end = r2[-1] == j
        if not (i_is_start or i_is_end):
            continue
        if not (j_is_start or j_is_end):
            continue
        if loads[ri] + loads[rj] > cap:
            continue

        if i_is_end and j_is_start:
            merged = r1 + r2
        elif i_is_start and j_is_end:
            merged = r2 + r1
        elif i_is_start and j_is_start:
            merged = list(reversed(r1)) + r2
        else:  # i_is_end and j_is_end
            merged = r1 + list(reversed(r2))

        routes[ri] = merged
        loads[ri] += loads[rj]
        routes[rj] = []
        loads[rj] = 0
        for c in merged:
            where[c] = ri

    routes = [r for r in routes if r]

    # If more than k routes remain, greedily merge whole routes if feasible.
    if len(routes) > k:
        rloads = [sum(dem[c] for c in r) for r in routes]
        order = sorted(range(len(routes)), key=lambda x: rloads[x], reverse=True)
        new_routes = []
        new_loads = []
        for idx in order:
            r = routes[idx]
            ld = rloads[idx]
            placed = False
            for t in range(len(new_routes)):
                if new_loads[t] + ld > cap:
                    continue
                base = new_routes[t]
                base_cost = route_cost(base, pts, depot)
                best = None
                best_delta = None
                cands = [
                    base + r,
                    r + base,
                    base + list(reversed(r)),
                    list(reversed(r)) + base,
                ]
                for cand in cands:
                    dlt = route_cost(cand, pts, depot) - base_cost
                    if best_delta is None or dlt < best_delta:
                        best_delta = dlt
                        best = cand
                new_routes[t] = best
                new_loads[t] += ld
                placed = True
                break
            if not placed:
                new_routes.append(r[:])
                new_loads.append(ld)
        routes = new_routes

    return routes


def relocate_improve(routes, pts, dem, cap, depot, max_passes=4):
    loads = [sum(dem[c] for c in r) for r in routes]
    for _ in range(max_passes):
        moved = False
        for ri in range(len(routes)):
            r = routes[ri]
            if len(r) <= 1:
                continue
            base_cost = route_cost(r, pts, depot)
            for pos in range(len(r)):
                cust = r[pos]
                r_removed = r[:pos] + r[pos + 1 :]
                cost_removed = route_cost(r_removed, pts, depot)
                for rj in range(len(routes)):
                    if rj == ri:
                        continue
                    if loads[rj] + dem[cust] > cap:
                        continue
                    r2 = routes[rj]
                    base2 = route_cost(r2, pts, depot)
                    best_cand2 = None
                    best_delta = 0.0
                    for ins in range(len(r2) + 1):
                        cand2 = r2[:ins] + [cust] + r2[ins:]
                        delta = (cost_removed - base_cost) + (route_cost(cand2, pts, depot) - base2)
                        if delta < best_delta - 1e-12:
                            best_delta = delta
                            best_cand2 = (rj, cand2)
                    if best_cand2 is not None:
                        rj2, cand2 = best_cand2
                        routes[ri] = r_removed
                        routes[rj2] = cand2
                        loads[ri] -= dem[cust]
                        loads[rj2] += dem[cust]
                        moved = True
                        break
                if moved:
                    break
            if moved:
                break
        if not moved:
            break
    return [r for r in routes if r]


def main():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    cap = int(next(it))
    depot = (float(next(it)), float(next(it)))
    pts = []
    dem = []
    for _ in range(n):
        x = float(next(it))
        y = float(next(it))
        d = int(next(it))
        pts.append((x, y))
        dem.append(d)

    routes = build_routes_savings(n, k, cap, depot, pts, dem)
    routes = [two_opt(r, pts, depot) for r in routes]
    routes = relocate_improve(routes, pts, dem, cap, depot)
    routes = [two_opt(r, pts, depot) for r in routes]

    # Validate coverage; if invalid, fall back to greedy nearest-neighbor packing.
    seen = [0] * n
    for r in routes:
        for c in r:
            if 0 <= c < n:
                seen[c] += 1
    if any(v != 1 for v in seen):
        unvisited = set(range(n))
        routes = []
        while unvisited and len(routes) < k:
            load = 0
            route = []
            cur = depot
            while True:
                best = None
                best_d = None
                for c in unvisited:
                    if load + dem[c] > cap:
                        continue
                    dcur = dist(cur, pts[c])
                    if best is None or dcur < best_d:
                        best = c
                        best_d = dcur
                if best is None:
                    break
                route.append(best)
                unvisited.remove(best)
                load += dem[best]
                cur = pts[best]
            routes.append(route)
        while unvisited:
            routes.append([unvisited.pop()])
        routes = [two_opt(r, pts, depot) for r in routes]

    # Enforce capacity by splitting routes if needed (keeps validity).
    final_routes = []
    for r in routes:
        if not r:
            continue
        load = 0
        cur = []
        for c in r:
            if load + dem[c] <= cap:
                cur.append(c)
                load += dem[c]
            else:
                final_routes.append(cur)
                cur = [c]
                load = dem[c]
        if cur:
            final_routes.append(cur)
    routes = [r for r in final_routes if r]

    total = sum(route_cost(r, pts, depot) for r in routes)
    out = [f"{total:.6f}", str(len(routes))]
    out.extend(" ".join(map(str, r)) for r in routes)
    sys.stdout.write("\n".join(out))


if __name__ == "__main__":
    main()
