using System;
using System.Buffers;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Numerics;

// Maximum Clique: exact on small/medium graphs; heuristic on large graphs.
// Output is always a valid clique; may be suboptimal in heuristic mode.

internal static class Program
{
    private const int ExactVertexLimit = 6000;
    private const int WordBits = 64;

    public static void Main()
    {
        var swTotal = Stopwatch.StartNew();
        var reader = new FastScanner(Console.OpenStandardInput());
        int n = reader.NextInt();
        int m = reader.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        int[] deg = new int[n];
        int[] eu = new int[m];
        int[] ev = new int[m];
        int readEdges = 0;
        for (int i = 0; i < m; i++)
        {
            int u = reader.NextInt();
            int v = reader.NextInt();
            if ((uint)u >= (uint)n || (uint)v >= (uint)n || u == v) continue;
            eu[readEdges] = u;
            ev[readEdges] = v;
            deg[u]++;
            deg[v]++;
            readEdges++;
        }
        m = readEdges;

        int[][] adj = BuildAdjacency(n, m, eu, ev, deg);

        if (m == 0)
        {
            Console.WriteLine(1);
            Console.WriteLine(0);
            return;
        }

        long totalBudgetMs = 5L * 60_000L - 500L;

        int[] best = Array.Empty<int>();
        if (n <= ExactVertexLimit)
        {
            long exactBudget = Math.Min(totalBudgetMs * 2 / 3, totalBudgetMs - 1000);
            if (exactBudget < 1500) exactBudget = Math.Min(1500, totalBudgetMs);
            var exact = new ExactBitsetClique(adj, n, exactBudget);
            exact.Solve();
            best = exact.BestClique;
        }

        long remaining = totalBudgetMs - swTotal.ElapsedMilliseconds;
        if (remaining < 200) remaining = 200;
        var heuristic = new HeuristicClique(adj, n, remaining);
        heuristic.Seed(best);
        heuristic.Run();
        if (heuristic.BestClique.Length > best.Length) best = heuristic.BestClique;

        if (!IsClique(best, adj))
            best = MakeCliqueGreedy(best, adj);

        Console.WriteLine(best.Length);
        if (best.Length == 0) Console.WriteLine();
        else
        {
            Array.Sort(best);
            Console.WriteLine(string.Join(' ', best));
        }
    }

    private static int[][] BuildAdjacency(int n, int m, int[] eu, int[] ev, int[] deg)
    {
        int[][] adj = new int[n][];
        int[] idx = new int[n];
        for (int i = 0; i < n; i++) adj[i] = new int[deg[i]];
        for (int i = 0; i < m; i++)
        {
            int u = eu[i], v = ev[i];
            adj[u][idx[u]++] = v;
            adj[v][idx[v]++] = u;
        }
        for (int i = 0; i < n; i++) Array.Sort(adj[i]);
        return adj;
    }

    private static bool IsClique(int[] vertices, int[][] adj)
    {
        for (int i = 0; i < vertices.Length; i++)
        {
            int u = vertices[i];
            var au = adj[u];
            for (int j = i + 1; j < vertices.Length; j++)
                if (Array.BinarySearch(au, vertices[j]) < 0) return false;
        }
        return true;
    }

    private static int[] MakeCliqueGreedy(int[] set, int[][] adj)
    {
        var chosen = new List<int>(set.Length);
        foreach (var v in set)
        {
            bool ok = true;
            var av = adj[v];
            for (int i = 0; i < chosen.Count; i++)
                if (Array.BinarySearch(av, chosen[i]) < 0) { ok = false; break; }
            if (ok) chosen.Add(v);
        }
        return chosen.ToArray();
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buf;
        private int _len, _ptr;
        public FastScanner(Stream stream, int bufSize = 1 << 16) { _stream = stream; _buf = new byte[bufSize]; }
        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buf[_ptr++];
        }
        public int NextInt()
        {
            byte c;
            do { c = Read(); } while (c <= 32 && c != 0);
            int sign = 1;
            if (c == (byte)'-') { sign = -1; c = Read(); }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - (byte)'0');
                c = Read();
            }
            return val * sign;
        }
    }

    private sealed class ExactBitsetClique
    {
        private readonly int[][] _adj;
        private readonly int _n;
        private readonly int _words;
        private readonly ulong[][] _adjBits;
        private readonly Stopwatch _sw = Stopwatch.StartNew();
        private readonly long _budgetMs;

        public int[] BestClique { get; private set; } = Array.Empty<int>();
        private int _best;

        public ExactBitsetClique(int[][] adj, int n, long budgetMs)
        {
            _adj = adj;
            _n = n;
            _budgetMs = budgetMs;
            _words = (_n + WordBits - 1) / WordBits;
            _adjBits = new ulong[_n][];
            for (int i = 0; i < _n; i++)
            {
                var bits = new ulong[_words];
                foreach (var v in _adj[i]) bits[v >> 6] |= 1UL << (v & 63);
                _adjBits[i] = bits;
            }
        }

        public void Solve()
        {
            var h = new HeuristicClique(_adj, _n, Math.Max(200, _budgetMs / 10));
            h.Run();
            BestClique = h.BestClique;
            _best = BestClique.Length;

            int[] order = DegeneracyOrder(_adj);
            var cand = new ulong[_words];
            for (int i = 0; i < _n; i++) SetBit(cand, order[i]);

            var current = new int[_n];
            Expand(cand, current, 0);
        }

        private void Expand(ulong[] cand, int[] current, int depth)
        {
            if (TimedOut()) return;
            int candCount = PopCount(cand);
            if (depth + candCount <= _best) return;
            if (candCount == 0)
            {
                if (depth > _best)
                {
                    _best = depth;
                    BestClique = current.AsSpan(0, depth).ToArray();
                }
                return;
            }

            var verts = ArrayPool<int>.Shared.Rent(candCount);
            int vcount = ExtractVertices(cand, verts);

            int colorBound = GreedyColorBound(verts, vcount);
            if (depth + colorBound <= _best)
            {
                ArrayPool<int>.Shared.Return(verts);
                return;
            }

            int pivot = ChoosePivot(verts, vcount, cand);
            var branch = new ulong[_words];
            Copy(cand, branch);
            if (pivot >= 0) AndNot(branch, _adjBits[pivot]);

            int bcnt = ExtractVertices(branch, verts);
            Array.Sort(verts, 0, bcnt, new DegDescComparer(_adj));
            for (int i = 0; i < bcnt; i++)
            {
                if (TimedOut()) break;
                int v = verts[i];
                var newCand = new ulong[_words];
                Intersect(cand, _adjBits[v], newCand);
                current[depth] = v;
                Expand(newCand, current, depth + 1);
                ClearBit(cand, v);
                if (depth + PopCount(cand) <= _best) break;
            }

            ArrayPool<int>.Shared.Return(verts);
        }

        private int GreedyColorBound(int[] verts, int count)
        {
            Array.Sort(verts, 0, count, new DegDescComparer(_adj));
            var remaining = new ulong[_words];
            for (int i = 0; i < count; i++) SetBit(remaining, verts[i]);
            int colors = 0;
            int colored = 0;
            while (colored < count)
            {
                colors++;
                var avail = new ulong[_words];
                Copy(remaining, avail);
                while (!IsZero(avail))
                {
                    int v = FirstBit(avail);
                    ClearBit(avail, v);
                    ClearBit(remaining, v);
                    colored++;
                    AndNot(avail, _adjBits[v]);
                }
            }
            return colors;
        }

        private int ChoosePivot(int[] verts, int count, ulong[] cand)
        {
            int bestV = -1, best = -1;
            for (int i = 0; i < count; i++)
            {
                int v = verts[i];
                int c = IntersectionCount(cand, _adjBits[v]);
                if (c > best) { best = c; bestV = v; }
            }
            return bestV;
        }

        private bool TimedOut() => _sw.ElapsedMilliseconds >= _budgetMs;

        private static int[] DegeneracyOrder(int[][] adj)
        {
            int n = adj.Length;
            int[] degree = new int[n];
            int maxDeg = 0;
            for (int i = 0; i < n; i++) { degree[i] = adj[i].Length; if (degree[i] > maxDeg) maxDeg = degree[i]; }
            int[] bin = new int[maxDeg + 1];
            for (int i = 0; i < n; i++) bin[degree[i]]++;
            int start = 0;
            for (int d = 0; d <= maxDeg; d++) { int num = bin[d]; bin[d] = start; start += num; }
            int[] pos = new int[n];
            int[] vert = new int[n];
            for (int v = 0; v < n; v++)
            {
                pos[v] = bin[degree[v]];
                vert[pos[v]] = v;
                bin[degree[v]]++;
            }
            for (int d = maxDeg; d >= 1; d--) bin[d] = bin[d - 1];
            bin[0] = 0;
            int[] order = new int[n];
            bool[] removed = new bool[n];
            for (int i = 0; i < n; i++)
            {
                int v = vert[i];
                order[i] = v;
                removed[v] = true;
                foreach (var u in adj[v])
                {
                    if (removed[u]) continue;
                    int du = degree[u];
                    int pu = pos[u];
                    int pw = bin[du];
                    int w = vert[pw];
                    if (u != w)
                    {
                        vert[pu] = w; pos[w] = pu;
                        vert[pw] = u; pos[u] = pw;
                    }
                    bin[du]++;
                    degree[u]--;
                }
            }
            Array.Reverse(order);
            return order;
        }

        private static void Copy(ulong[] src, ulong[] dst) => Array.Copy(src, dst, src.Length);
        private static void Intersect(ulong[] a, ulong[] b, ulong[] dst) { for (int i = 0; i < a.Length; i++) dst[i] = a[i] & b[i]; }
        private static void AndNot(ulong[] a, ulong[] b) { for (int i = 0; i < a.Length; i++) a[i] &= ~b[i]; }
        private static void SetBit(ulong[] bits, int v) => bits[v >> 6] |= 1UL << (v & 63);
        private static void ClearBit(ulong[] bits, int v) => bits[v >> 6] &= ~(1UL << (v & 63));
        private static bool IsZero(ulong[] bits) { for (int i = 0; i < bits.Length; i++) if (bits[i] != 0) return false; return true; }
        private static int PopCount(ulong[] bits) { int c = 0; for (int i = 0; i < bits.Length; i++) c += BitOperations.PopCount(bits[i]); return c; }
        private static int FirstBit(ulong[] bits)
        {
            for (int i = 0; i < bits.Length; i++)
            {
                ulong w = bits[i];
                if (w != 0) return (i << 6) + BitOperations.TrailingZeroCount(w);
            }
            return -1;
        }
        private static int ExtractVertices(ulong[] bits, int[] output)
        {
            int idx = 0;
            for (int w = 0; w < bits.Length; w++)
            {
                ulong x = bits[w];
                while (x != 0)
                {
                    int t = BitOperations.TrailingZeroCount(x);
                    output[idx++] = (w << 6) + t;
                    x &= x - 1;
                }
            }
            return idx;
        }
        private static int IntersectionCount(ulong[] a, ulong[] b)
        {
            int c = 0;
            for (int i = 0; i < a.Length; i++) c += BitOperations.PopCount(a[i] & b[i]);
            return c;
        }

        private readonly struct DegDescComparer : IComparer<int>
        {
            private readonly int[][] _adj;
            public DegDescComparer(int[][] adj) { _adj = adj; }
            public int Compare(int x, int y)
            {
                int c = _adj[y].Length.CompareTo(_adj[x].Length);
                return c != 0 ? c : x.CompareTo(y);
            }
        }
    }

    private sealed class HeuristicClique
    {
        private readonly int[][] _adj;
        private readonly int _n;
        private readonly Stopwatch _sw = Stopwatch.StartNew();
        private readonly long _budgetMs;
        private readonly Random _rng = new Random(1);

        public int[] BestClique { get; private set; } = Array.Empty<int>();

        public HeuristicClique(int[][] adj, int n, long budgetMs)
        {
            _adj = adj;
            _n = n;
            _budgetMs = budgetMs;
        }

        public void Seed(int[] clique)
        {
            if (clique != null && clique.Length > BestClique.Length && IsClique(clique))
                BestClique = (int[])clique.Clone();
        }

        public void Run()
        {
            ImproveFromStarts(Math.Min(_n, 256), pickHighDegree: true);
            int rounds = 0;
            while (!TimedOut())
            {
                rounds++;
                ImproveFromStarts(64, pickHighDegree: false);
                if (rounds % 4 == 0) PerturbAndImprove();
            }
        }

        private void ImproveFromStarts(int count, bool pickHighDegree)
        {
            int[] starts = pickHighDegree ? TopKByDegree(count) : RandomVertices(count);
            for (int i = 0; i < starts.Length && !TimedOut(); i++)
            {
                var clique = GreedyGrowFrom(starts[i]);
                clique = LocalImprove(clique);
                if (clique.Length > BestClique.Length) BestClique = clique;
            }
        }

        private int[] GreedyGrowFrom(int start)
        {
            var clique = new List<int>(32) { start };
            int[] cand = (int[])_adj[start].Clone();
            Array.Sort(cand, (a, b) => _adj[b].Length.CompareTo(_adj[a].Length));
            int candLen = cand.Length;
            int idx = 0;
            while (idx < candLen && !TimedOut())
            {
                int v = cand[idx++];
                if (CanAdd(v, clique))
                {
                    clique.Add(v);
                    candLen = IntersectInPlace(cand, candLen, _adj[v]);
                    if (idx > candLen) idx = candLen;
                }
            }
            return clique.ToArray();
        }

        private int[] LocalImprove(int[] clique)
        {
            var current = clique.ToList();
            bool changed = true;
            int loops = 0;
            while (changed && !TimedOut() && loops++ < 50)
            {
                changed = false;
                var ext = CandidatesAdjacentToAll(current);
                if (ext.Count > 0)
                {
                    ext.Sort((a, b) => _adj[b].Length.CompareTo(_adj[a].Length));
                    current.Add(ext[0]);
                    changed = true;
                    continue;
                }

                int tries = Math.Min(10, current.Count);
                for (int t = 0; t < tries && !TimedOut(); t++)
                {
                    int remIdx = _rng.Next(current.Count);
                    int rem = current[remIdx];
                    current.RemoveAt(remIdx);
                    var ext2 = CandidatesAdjacentToAll(current);
                    if (ext2.Count > 0)
                    {
                        ext2.Sort((a, b) => _adj[b].Length.CompareTo(_adj[a].Length));
                        current.Add(ext2[0]);
                        changed = true;
                        break;
                    }
                    current.Insert(remIdx, rem);
                }
            }
            return current.ToArray();
        }

        private void PerturbAndImprove()
        {
            if (BestClique.Length == 0) return;
            var baseClique = BestClique.ToList();
            int remove = Math.Min(3, baseClique.Count / 3 + 1);
            for (int i = 0; i < remove && baseClique.Count > 0; i++)
                baseClique.RemoveAt(_rng.Next(baseClique.Count));
            var ext = CandidatesAdjacentToAll(baseClique);
            ext.Sort((a, b) => _adj[b].Length.CompareTo(_adj[a].Length));
            for (int i = 0; i < ext.Count && !TimedOut(); i++)
                if (CanAdd(ext[i], baseClique)) baseClique.Add(ext[i]);
            var improved = LocalImprove(baseClique.ToArray());
            if (improved.Length > BestClique.Length) BestClique = improved;
        }

        private List<int> CandidatesAdjacentToAll(List<int> clique)
        {
            if (clique.Count == 0) return new List<int>();
            var cand = new List<int>(_adj[clique[0]]);
            cand.Sort();
            for (int i = 1; i < clique.Count; i++)
            {
                cand = IntersectSorted(cand, _adj[clique[i]]);
                if (cand.Count == 0) break;
            }
            if (cand.Count > 0)
            {
                var inClique = new HashSet<int>(clique);
                cand.RemoveAll(inClique.Contains);
            }
            return cand;
        }

        private bool CanAdd(int v, List<int> clique)
        {
            var av = _adj[v];
            for (int i = 0; i < clique.Count; i++)
                if (Array.BinarySearch(av, clique[i]) < 0) return false;
            return true;
        }

        private int[] TopKByDegree(int k)
        {
            k = Math.Min(k, _n);
            int[] idx = new int[_n];
            for (int i = 0; i < _n; i++) idx[i] = i;
            Array.Sort(idx, (a, b) => _adj[b].Length.CompareTo(_adj[a].Length));
            if (k == _n) return idx;
            var res = new int[k];
            Array.Copy(idx, res, k);
            return res;
        }

        private int[] RandomVertices(int k)
        {
            k = Math.Min(k, _n);
            var res = new int[k];
            for (int i = 0; i < k; i++) res[i] = _rng.Next(_n);
            return res;
        }

        private int IntersectInPlace(int[] a, int aLen, int[] bSorted)
        {
            Array.Sort(a, 0, aLen);
            int i = 0, j = 0, w = 0;
            while (i < aLen && j < bSorted.Length)
            {
                int x = a[i], y = bSorted[j];
                if (x == y) { a[w++] = x; i++; j++; }
                else if (x < y) i++;
                else j++;
            }
            return w;
        }

        private static List<int> IntersectSorted(List<int> aSorted, int[] bSorted)
        {
            var res = new List<int>(Math.Min(aSorted.Count, bSorted.Length));
            int i = 0, j = 0;
            while (i < aSorted.Count && j < bSorted.Length)
            {
                int x = aSorted[i], y = bSorted[j];
                if (x == y) { res.Add(x); i++; j++; }
                else if (x < y) i++;
                else j++;
            }
            return res;
        }

        private bool IsClique(int[] vertices)
        {
            for (int i = 0; i < vertices.Length; i++)
            {
                var au = _adj[vertices[i]];
                for (int j = i + 1; j < vertices.Length; j++)
                    if (Array.BinarySearch(au, vertices[j]) < 0) return false;
            }
            return true;
        }

        private bool TimedOut() => _sw.ElapsedMilliseconds >= _budgetMs;
    }
}
