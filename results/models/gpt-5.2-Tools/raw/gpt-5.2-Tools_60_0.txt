const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input.length ? JSON.parse(input) : { machineCount: 1, tasks: [] };

const machineCount = Math.max(1, data.machineCount | 0);
const tasks = Array.isArray(data.tasks) ? data.tasks : [];

// Build task maps and dependency graph
const n = tasks.length;
const idToIdx = new Map();
for (let i = 0; i < n; i++) idToIdx.set(tasks[i].id, i);

const duration = new Array(n);
const deadline = new Array(n);
const weight = new Array(n);
const indeg = new Array(n).fill(0);
const succ = Array.from({ length: n }, () => []);

for (let i = 0; i < n; i++) {
  const t = tasks[i] || {};
  const d = +t.duration;
  duration[i] = Number.isFinite(d) && d > 0 ? d : 0;
  const dl = +t.deadline;
  deadline[i] = Number.isFinite(dl) ? dl : Infinity;
  const w = +t.weight;
  weight[i] = Number.isFinite(w) ? w : 1;
}

for (let i = 0; i < n; i++) {
  const deps = Array.isArray(tasks[i].deps) ? tasks[i].deps : [];
  for (let k = 0; k < deps.length; k++) {
    const depId = deps[k];
    const u = idToIdx.get(depId);
    if (u === undefined) continue; // ignore unknown deps
    succ[u].push(i);
    indeg[i]++;
  }
}

// Kahn topological order (cycle-safe fallback)
function topoOrder() {
  const q = [];
  const deg = indeg.slice();
  for (let i = 0; i < n; i++) if (deg[i] === 0) q.push(i);
  // Simple queue with index
  const order = [];
  for (let qi = 0; qi < q.length; qi++) {
    const u = q[qi];
    order.push(u);
    const s = succ[u];
    for (let j = 0; j < s.length; j++) {
      const v = s[j];
      if (--deg[v] === 0) q.push(v);
    }
  }
  if (order.length !== n) {
    // Graph has cycles or missing nodes due to weird deps; append remaining deterministically
    const inOrder = new Array(n).fill(false);
    for (let i = 0; i < order.length; i++) inOrder[order[i]] = true;
    for (let i = 0; i < n; i++) if (!inOrder[i]) order.push(i);
  }
  return order;
}

const topo = topoOrder();

// Compute critical path "bottom levels" (longest path from node to an exit, incl. own duration)
const bottom = new Array(n).fill(0);
for (let oi = topo.length - 1; oi >= 0; oi--) {
  const u = topo[oi];
  let best = 0;
  const s = succ[u];
  for (let j = 0; j < s.length; j++) {
    const v = s[j];
    if (bottom[v] > best) best = bottom[v];
  }
  bottom[u] = duration[u] + best;
}

// Ready set maintained as a binary heap (max-heap by score)
class MaxHeap {
  constructor(cmp) {
    this.a = [];
    this.cmp = cmp;
  }
  get size() {
    return this.a.length;
  }
  push(x) {
    const a = this.a;
    a.push(x);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.cmp(a[i], a[p]) <= 0) break;
      const tmp = a[i];
      a[i] = a[p];
      a[p] = tmp;
      i = p;
    }
  }
  pop() {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop();
    if (a.length) {
      a[0] = last;
      this._down(0);
    }
    return top;
  }
  _down(i) {
    const a = this.a;
    const n = a.length;
    while (true) {
      let l = i * 2 + 1;
      if (l >= n) break;
      let r = l + 1;
      let best = l;
      if (r < n && this.cmp(a[r], a[l]) > 0) best = r;
      if (this.cmp(a[best], a[i]) <= 0) break;
      const tmp = a[i];
      a[i] = a[best];
      a[best] = tmp;
      i = best;
    }
  }
}

// Machine availability min-heap
class MinHeap {
  constructor(cmp) {
    this.a = [];
    this.cmp = cmp;
  }
  get size() {
    return this.a.length;
  }
  push(x) {
    const a = this.a;
    a.push(x);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.cmp(a[i], a[p]) >= 0) break;
      const tmp = a[i];
      a[i] = a[p];
      a[p] = tmp;
      i = p;
    }
  }
  pop() {
    const a = this.a;
    if (!a.length) return undefined;
    const top = a[0];
    const last = a.pop();
    if (a.length) {
      a[0] = last;
      this._down(0);
    }
    return top;
  }
  _down(i) {
    const a = this.a;
    const n = a.length;
    while (true) {
      let l = i * 2 + 1;
      if (l >= n) break;
      let r = l + 1;
      let best = l;
      if (r < n && this.cmp(a[r], a[l]) < 0) best = r;
      if (this.cmp(a[best], a[i]) >= 0) break;
      const tmp = a[i];
      a[i] = a[best];
      a[best] = tmp;
      i = best;
    }
  }
}

// Priority rule: larger bottom-level first, then earlier deadline, then higher weight, then longer duration, then stable by id string
function readyScore(i) {
  // For max-heap comparator we compute components; avoid huge combined numbers (Infinity ok)
  return 0; // unused; we compare via cmp directly
}

function cmpReady(i, j) {
  const bi = bottom[i], bj = bottom[j];
  if (bi !== bj) return bi > bj ? 1 : -1;
  const di = deadline[i], dj = deadline[j];
  if (di !== dj) return di < dj ? 1 : -1;
  const wi = weight[i], wj = weight[j];
  if (wi !== wj) return wi > wj ? 1 : -1;
  const pi = duration[i], pj = duration[j];
  if (pi !== pj) return pi > pj ? 1 : -1;
  const idi = String(tasks[i].id), idj = String(tasks[j].id);
  if (idi === idj) return 0;
  return idi < idj ? 1 : -1; // reverse for max-heap (smaller id last)
}

// Event-driven list scheduling:
// - Maintain ready tasks (all deps done) in a max-heap by priority.
// - Maintain machines by next-available time in min-heap.
// - When a machine is available, schedule the best ready task at max(available, depsReadyTime).
const depReadyTime = new Array(n).fill(0); // max finish time over deps
const remaining = indeg.slice();
const ready = new MaxHeap((a, b) => cmpReady(a, b));
for (let i = 0; i < n; i++) if (remaining[i] === 0) ready.push(i);

const machines = new MinHeap((a, b) => {
  // a,b: [time, machineIndex]
  if (a[0] !== b[0]) return a[0] - b[0];
  return a[1] - b[1];
});
for (let m = 0; m < machineCount; m++) machines.push([0, m]);

const schedule = [];
const finishTime = new Array(n).fill(0);

let scheduledCount = 0;

while (scheduledCount < n) {
  if (ready.size === 0) {
    // Deadlock (cycle). Output a topo-ish order as fallback.
    const orderIds = topo.map((idx) => tasks[idx].id);
    process.stdout.write(JSON.stringify({ order: orderIds }));
    process.exit(0);
  }

  const [availTime, machine] = machines.pop();
  const taskIdx = ready.pop();

  const start = Math.max(availTime, depReadyTime[taskIdx]);
  const fin = start + duration[taskIdx];

  schedule.push({ id: tasks[taskIdx].id, start: Math.trunc(start), machine });
  finishTime[taskIdx] = fin;
  scheduledCount++;

  const s = succ[taskIdx];
  for (let j = 0; j < s.length; j++) {
    const v = s[j];
    if (fin > depReadyTime[v]) depReadyTime[v] = fin;
    if (--remaining[v] === 0) ready.push(v);
  }

  machines.push([fin, machine]);
}

process.stdout.write(JSON.stringify({ schedule }));
