using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public static class Program
{
    private struct Box
    {
        public int Index;
        public int A, B, C;
        public int Volume;
        public int MinDim;
    }

    private struct OrientedBox
    {
        public int Index;
        public int W, H, D;
        public int Volume;
    }

    private struct Placement
    {
        public int Index;
        public int X, Y, Z;
        public int W, H, D;
    }

    private struct Space
    {
        public int X, Y, Z;
        public int W, H, D;
        public int Volume => W * H * D;
    }

    private sealed class SpaceComparer : IComparer<Space>
    {
        public int Compare(Space a, Space b)
        {
            int va = a.W * a.H * a.D;
            int vb = b.W * b.H * b.D;
            int c = vb.CompareTo(va);
            if (c != 0) return c;
            c = a.Z.CompareTo(b.Z);
            if (c != 0) return c;
            c = a.Y.CompareTo(b.Y);
            if (c != 0) return c;
            c = a.X.CompareTo(b.X);
            if (c != 0) return c;
            c = b.W.CompareTo(a.W);
            if (c != 0) return c;
            c = b.H.CompareTo(a.H);
            if (c != 0) return c;
            return b.D.CompareTo(a.D);
        }
    }

    private static readonly int[][] Perms = new int[][]
    {
        new []{0,1,2},
        new []{0,2,1},
        new []{1,0,2},
        new []{1,2,0},
        new []{2,0,1},
        new []{2,1,0},
    };

    private static void AddSpace(List<Space> spaces, Space s)
    {
        if (s.W <= 0 || s.H <= 0 || s.D <= 0) return;
        spaces.Add(s);
    }

    private static bool Intersects(Placement a, Placement b)
    {
        if (a.X + a.W <= b.X || b.X + b.W <= a.X) return false;
        if (a.Y + a.H <= b.Y || b.Y + b.H <= a.Y) return false;
        if (a.Z + a.D <= b.Z || b.Z + b.D <= a.Z) return false;
        return true;
    }

    private static void FilterDominatedSpaces(List<Space> spaces)
    {
        // O(m^2) pruning; only used when m is small.
        for (int i = 0; i < spaces.Count; i++)
        {
            var a = spaces[i];
            for (int j = spaces.Count - 1; j > i; j--)
            {
                var b = spaces[j];
                if (a.X <= b.X && a.Y <= b.Y && a.Z <= b.Z &&
                    a.X + a.W >= b.X + b.W &&
                    a.Y + a.H >= b.Y + b.H &&
                    a.Z + a.D >= b.Z + b.D)
                {
                    spaces.RemoveAt(j);
                }
                else if (b.X <= a.X && b.Y <= a.Y && b.Z <= a.Z &&
                         b.X + b.W >= a.X + a.W &&
                         b.Y + b.H >= a.Y + a.H &&
                         b.Z + b.D >= a.Z + a.D)
                {
                    spaces[i] = b;
                    a = b;
                    spaces.RemoveAt(j);
                }
            }
        }
    }

    private static OrientedBox ChooseBestOrientation(Box box, in Space sp)
    {
        int[] dims = { box.A, box.B, box.C };
        OrientedBox best = default;
        best.Volume = -1;
        int bestScore = int.MinValue;
        for (int pi = 0; pi < 6; pi++)
        {
            int w = dims[Perms[pi][0]];
            int h = dims[Perms[pi][1]];
            int d = dims[Perms[pi][2]];
            if (w > sp.W || h > sp.H || d > sp.D) continue;
            // Prefer tight fits: maximize used volume, then minimize leftover faces.
            int used = w * h * d;
            int slack = (sp.W - w) + (sp.H - h) + (sp.D - d);
            int faceSlack = Math.Abs(sp.W - w) + Math.Abs(sp.H - h) + Math.Abs(sp.D - d);
            int score = used * 8 - slack * 3 - faceSlack;
            if (score > bestScore)
            {
                bestScore = score;
                best.Index = box.Index;
                best.W = w;
                best.H = h;
                best.D = d;
                best.Volume = used;
            }
        }
        return best;
    }

    private static void SplitSpaceGuillotine(in Space sp, in Placement pl, List<Space> outSpaces)
    {
        // Split into up to 3 disjoint spaces around the placed box:
        // right, above, front, each aligned to the placed box's footprint to reduce fragmentation.
        // Right space
        AddSpace(outSpaces, new Space
        {
            X = sp.X + pl.W,
            Y = sp.Y,
            Z = sp.Z,
            W = sp.W - pl.W,
            H = sp.H,
            D = sp.D
        });
        // Above space (on Y)
        AddSpace(outSpaces, new Space
        {
            X = sp.X,
            Y = sp.Y + pl.H,
            Z = sp.Z,
            W = pl.W,
            H = sp.H - pl.H,
            D = sp.D
        });
        // Front space (on Z)
        AddSpace(outSpaces, new Space
        {
            X = sp.X,
            Y = sp.Y,
            Z = sp.Z + pl.D,
            W = pl.W,
            H = pl.H,
            D = sp.D - pl.D
        });
    }

    private static int ParseInt(string s) => int.Parse(s, NumberStyles.Integer, CultureInfo.InvariantCulture);

    public static void Main()
    {
        var input = Console.In;
        string? header = input.ReadLine();
        if (string.IsNullOrWhiteSpace(header))
        {
            Console.WriteLine(0);
            return;
        }

        var hparts = header.Split((char[]?)null, StringSplitOptions.RemoveEmptyEntries);
        int n = ParseInt(hparts[0]);
        int W = ParseInt(hparts[1]);
        int H = ParseInt(hparts[2]);
        int D = ParseInt(hparts[3]);

        var boxes = new Box[n];
        long totalBoxVol = 0;
        int maxDim = Math.Max(W, Math.Max(H, D));
        for (int i = 0; i < n; i++)
        {
            string? line = input.ReadLine();
            while (line != null && line.Length == 0) line = input.ReadLine();
            if (line == null) { n = i; Array.Resize(ref boxes, n); break; }
            var p = line.Split((char[]?)null, StringSplitOptions.RemoveEmptyEntries);
            int a = ParseInt(p[0]);
            int b = ParseInt(p[1]);
            int c = ParseInt(p[2]);
            int vol = a * b * c;
            int minDim = Math.Min(a, Math.Min(b, c));
            boxes[i] = new Box { Index = i, A = a, B = b, C = c, Volume = vol, MinDim = minDim };
            totalBoxVol += vol;
        }

        long containerVol = (long)W * H * D;
        bool hugeN = n >= 5000;
        bool hugeContainer = containerVol >= 5_000_000L;
        bool tiny = n <= 200 && containerVol <= 200_000L;

        // Filter impossible boxes quickly.
        var candidates = new List<Box>(n);
        for (int i = 0; i < n; i++)
        {
            var b = boxes[i];
            int[] dims = { b.A, b.B, b.C };
            bool fits = false;
            for (int pi = 0; pi < 6 && !fits; pi++)
            {
                int w = dims[Perms[pi][0]];
                int h = dims[Perms[pi][1]];
                int d = dims[Perms[pi][2]];
                if (w <= W && h <= H && d <= D) fits = true;
            }
            if (fits) candidates.Add(b);
        }

        // Sort heuristic:
        // - For large instances, sort by volume desc then max side desc.
        // - For small instances, also consider "cubishness" to improve packing density.
        if (hugeN || hugeContainer)
        {
            candidates.Sort((x, y) =>
            {
                int c = y.Volume.CompareTo(x.Volume);
                if (c != 0) return c;
                int xMax = Math.Max(x.A, Math.Max(x.B, x.C));
                int yMax = Math.Max(y.A, Math.Max(y.B, y.C));
                c = yMax.CompareTo(xMax);
                if (c != 0) return c;
                return x.Index.CompareTo(y.Index);
            });
        }
        else
        {
            candidates.Sort((x, y) =>
            {
                int c = y.Volume.CompareTo(x.Volume);
                if (c != 0) return c;
                int xMax = Math.Max(x.A, Math.Max(x.B, x.C));
                int yMax = Math.Max(y.A, Math.Max(y.B, y.C));
                c = yMax.CompareTo(xMax);
                if (c != 0) return c;
                int xMin = x.MinDim;
                int yMin = y.MinDim;
                c = yMin.CompareTo(xMin);
                if (c != 0) return c;
                return x.Index.CompareTo(y.Index);
            });
        }

        // Free-space management:
        // Use a list kept roughly ordered by volume; for large cases, cap list size.
        var spaces = new List<Space>(1024) { new Space { X = 0, Y = 0, Z = 0, W = W, H = H, D = D } };
        var spaceCmp = new SpaceComparer();

        int spaceCap = hugeN ? 4000 : (hugeContainer ? 8000 : 20000);
        int pruneThreshold = hugeN ? 2500 : 8000;

        var placements = new List<Placement>(Math.Min(n, 100000));
        bool doExactOverlapCheck = !hugeN && !hugeContainer; // for safety in smaller cases

        // Simple greedy placement:
        // For each box (descending), try best space among top-K by volume.
        int topK = hugeN ? 24 : (hugeContainer ? 48 : 128);
        var newSpaces = new List<Space>(8);

        for (int bi = 0; bi < candidates.Count; bi++)
        {
            var box = candidates[bi];
            if (spaces.Count == 0) break;

            // Ensure spaces are roughly ordered by volume descending using partial sort.
            // For performance, do a full sort occasionally when list is large.
            if ((bi & 255) == 0)
                spaces.Sort(spaceCmp);

            int limit = Math.Min(topK, spaces.Count);
            int bestSi = -1;
            OrientedBox bestOb = default;
            int bestScore = int.MinValue;

            for (int si = 0; si < limit; si++)
            {
                var sp = spaces[si];
                if (sp.W <= 0 || sp.H <= 0 || sp.D <= 0) continue;
                if (box.MinDim > Math.Max(sp.W, Math.Max(sp.H, sp.D))) continue;
                var ob = ChooseBestOrientation(box, sp);
                if (ob.Volume <= 0) continue;
                // Prefer higher fill ratio and placing earlier (lower z,y,x).
                int slack = (sp.W - ob.W) + (sp.H - ob.H) + (sp.D - ob.D);
                int score = ob.Volume * 16 - slack * 4 - sp.Z * 2 - sp.Y - sp.X;
                if (score > bestScore)
                {
                    bestScore = score;
                    bestSi = si;
                    bestOb = ob;
                }
            }

            if (bestSi < 0) continue;

            var chosenSpace = spaces[bestSi];
            var pl = new Placement
            {
                Index = bestOb.Index,
                X = chosenSpace.X,
                Y = chosenSpace.Y,
                Z = chosenSpace.Z,
                W = bestOb.W,
                H = bestOb.H,
                D = bestOb.D
            };

            if (doExactOverlapCheck)
            {
                bool ok = true;
                for (int pi = placements.Count - 1; pi >= 0; pi--)
                {
                    if (Intersects(placements[pi], pl)) { ok = false; break; }
                }
                if (!ok) continue;
            }

            placements.Add(pl);

            // Remove chosen space and split.
            spaces.RemoveAt(bestSi);
            newSpaces.Clear();
            SplitSpaceGuillotine(in chosenSpace, in pl, newSpaces);
            for (int i = 0; i < newSpaces.Count; i++) spaces.Add(newSpaces[i]);

            // Optional pruning / capping for scalability.
            if (spaces.Count > spaceCap)
            {
                spaces.Sort(spaceCmp);
                if (spaces.Count > spaceCap)
                    spaces.RemoveRange(spaceCap, spaces.Count - spaceCap);
            }

            if (tiny && spaces.Count > pruneThreshold && (bi & 31) == 0)
            {
                FilterDominatedSpaces(spaces);
            }
        }

        // Output
        var writer = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };
        writer.WriteLine(placements.Count.ToString(CultureInfo.InvariantCulture));
        for (int i = 0; i < placements.Count; i++)
        {
            var p = placements[i];
            writer.Write(p.Index);
            writer.Write(' ');
            writer.Write(p.X);
            writer.Write(' ');
            writer.Write(p.Y);
            writer.Write(' ');
            writer.WriteLine(p.Z);
        }
        writer.Flush();
    }
}
