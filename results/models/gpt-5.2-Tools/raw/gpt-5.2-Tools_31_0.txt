use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Item {
    v: i64,
    idx: usize,
}

fn parse_i64(bytes: &[u8], i: &mut usize) -> Option<i64> {
    while *i < bytes.len() && bytes[*i].is_ascii_whitespace() {
        *i += 1;
    }
    if *i >= bytes.len() {
        return None;
    }
    let mut sign: i64 = 1;
    if bytes[*i] == b'-' {
        sign = -1;
        *i += 1;
    } else if bytes[*i] == b'+' {
        *i += 1;
    }
    let mut n: i64 = 0;
    let mut any = false;
    while *i < bytes.len() {
        let c = bytes[*i];
        if (b'0'..=b'9').contains(&c) {
            any = true;
            n = n.saturating_mul(10).saturating_add((c - b'0') as i64);
            *i += 1;
        } else {
            break;
        }
    }
    if !any {
        return None;
    }
    Some(n.saturating_mul(sign))
}

fn gcd_i64(mut a: i64, mut b: i64) -> i64 {
    if a == 0 {
        return b.abs();
    }
    if b == 0 {
        return a.abs();
    }
    a = a.abs();
    b = b.abs();
    while b != 0 {
        let r = a % b;
        a = b;
        b = r;
    }
    a
}

fn greedy_sign_heuristic(items: &[Item], target: i64, pick_limit: usize) -> Option<Vec<usize>> {
    // For very large N, do a fast "closest by absolute value with sign alignment" pass.
    // Sort a small sample of candidates by descending |v| and greedily add if it reduces |rem|.
    let mut candidates: Vec<Item> = items.to_vec();
    candidates.sort_by(|a, b| b.v.abs().cmp(&a.v.abs()));
    let mut rem = target;
    let mut chosen: Vec<usize> = Vec::new();
    for it in candidates.into_iter().take(pick_limit) {
        if rem == 0 {
            break;
        }
        let before = rem.abs();
        let after = (rem - it.v).abs();
        if after < before {
            rem -= it.v;
            chosen.push(it.idx);
        }
    }
    if rem == 0 {
        Some(chosen)
    } else {
        None
    }
}

fn small_bruteforce(items: &[Item], target: i64) -> Option<Vec<usize>> {
    let n = items.len();
    if n == 0 {
        return if target == 0 { Some(Vec::new()) } else { None };
    }
    if n > 26 {
        return None;
    }
    let mut best: Option<Vec<usize>> = None;
    let total_masks = 1usize << n;
    for mask in 0..total_masks {
        let mut sum: i64 = 0;
        for i in 0..n {
            if (mask >> i) & 1 == 1 {
                sum = sum.saturating_add(items[i].v);
            }
        }
        if sum == target {
            let mut idxs = Vec::new();
            for i in 0..n {
                if (mask >> i) & 1 == 1 {
                    idxs.push(items[i].idx);
                }
            }
            best = Some(idxs);
            break;
        }
    }
    best
}

fn meet_in_middle(items: &[Item], target: i64) -> Option<Vec<usize>> {
    let n = items.len();
    if n == 0 {
        return if target == 0 { Some(Vec::new()) } else { None };
    }
    if n > 44 {
        return None;
    }
    let mid = n / 2;
    let left = &items[..mid];
    let right = &items[mid..];

    // Enumerate left sums with bitmasks (up to 2^22 = 4M fits).
    let left_n = left.len();
    let right_n = right.len();
    let left_count = 1usize << left_n;
    let right_count = 1usize << right_n;

    let mut left_sums: Vec<(i64, u32)> = Vec::with_capacity(left_count);
    for mask in 0..left_count {
        let mut sum = 0i64;
        for i in 0..left_n {
            if (mask >> i) & 1 == 1 {
                sum = sum.saturating_add(left[i].v);
            }
        }
        left_sums.push((sum, mask as u32));
    }
    left_sums.sort_by(|a, b| a.0.cmp(&b.0));

    for rmask in 0..right_count {
        let mut rsum = 0i64;
        for i in 0..right_n {
            if (rmask >> i) & 1 == 1 {
                rsum = rsum.saturating_add(right[i].v);
            }
        }
        let need = target.saturating_sub(rsum);
        if let Ok(pos) = left_sums.binary_search_by(|x| x.0.cmp(&need)) {
            let lmask = left_sums[pos].1 as usize;
            let mut idxs: Vec<usize> = Vec::new();
            for i in 0..left_n {
                if (lmask >> i) & 1 == 1 {
                    idxs.push(left[i].idx);
                }
            }
            for i in 0..right_n {
                if (rmask >> i) & 1 == 1 {
                    idxs.push(right[i].idx);
                }
            }
            return Some(idxs);
        }
    }
    None
}

fn dp_hash_pruned(items: &[Item], target: i64, cap: usize) -> Option<Vec<usize>> {
    // HashMap sum -> (prev_sum, used_idx)
    // Prune to at most `cap` states by keeping sums closest to target.
    let mut cur: HashMap<i64, (i64, usize)> = HashMap::new();
    cur.insert(0, (0, usize::MAX));
    for it in items {
        if cur.contains_key(&target) {
            break;
        }
        let mut updates: Vec<(i64, (i64, usize))> = Vec::with_capacity(cur.len().saturating_mul(1));
        for (&s, _) in cur.iter() {
            let ns = s.saturating_add(it.v);
            if !cur.contains_key(&ns) {
                updates.push((ns, (s, it.idx)));
            }
        }
        for (ns, back) in updates {
            cur.entry(ns).or_insert(back);
        }
        if cur.len() > cap {
            // Keep best `cap` by |target - sum|
            let mut keys: Vec<i64> = cur.keys().copied().collect();
            keys.sort_by(|a, b| {
                let da = (target - *a).abs();
                let db = (target - *b).abs();
                da.cmp(&db).then_with(|| a.cmp(b))
            });
            keys.truncate(cap);
            let mut next: HashMap<i64, (i64, usize)> = HashMap::with_capacity(cap.saturating_mul(2));
            for k in keys {
                if let Some(v) = cur.get(&k).copied() {
                    next.insert(k, v);
                }
            }
            cur = next;
        }
    }
    if !cur.contains_key(&target) {
        return None;
    }
    // Reconstruct
    let mut res: Vec<usize> = Vec::new();
    let mut s = target;
    while s != 0 {
        let (ps, idx) = cur.get(&s).copied().unwrap_or((0, usize::MAX));
        if idx == usize::MAX {
            break;
        }
        res.push(idx);
        s = ps;
    }
    Some(res)
}

fn normalize_and_quick_checks(nums: Vec<i64>, target: i64) -> (Vec<Item>, i64, Option<Vec<usize>>) {
    // Returns (items, normalized_target, immediate_solution_opt)
    if target == 0 {
        return (Vec::new(), 0, Some(Vec::new()));
    }
    let mut items: Vec<Item> = Vec::with_capacity(nums.len());
    for (i, v) in nums.into_iter().enumerate() {
        if v != 0 {
            items.push(Item { v, idx: i });
        }
    }
    // Single element match
    for it in &items {
        if it.v == target {
            return (Vec::new(), 0, Some(vec![it.idx]));
        }
    }
    // GCD feasibility reduction (necessary condition)
    let mut g: i64 = 0;
    for it in &items {
        g = gcd_i64(g, it.v);
        if g == 1 {
            break;
        }
    }
    if g != 0 {
        if target % g != 0 {
            return (Vec::new(), 0, None);
        }
        if g != 1 {
            for it in &mut items {
                it.v /= g;
            }
            return (items, target / g, None);
        }
    }
    (items, target, None)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let bytes = input.as_bytes();
    let mut pos: usize = 0;

    let n_opt = parse_i64(bytes, &mut pos);
    let t_opt = parse_i64(bytes, &mut pos);
    if n_opt.is_none() || t_opt.is_none() {
        // No valid input; stay silent (benchmark harness expects output, but invalid input isn't tested).
        return;
    }
    let n = n_opt.unwrap();
    let target = t_opt.unwrap();
    if n <= 0 {
        if target == 0 {
            print!("YES\n\n");
        } else {
            print!("NO\n");
        }
        return;
    }
    let n_usize = n as usize;
    let mut nums: Vec<i64> = Vec::with_capacity(n_usize);
    for _ in 0..n_usize {
        if let Some(v) = parse_i64(bytes, &mut pos) {
            nums.push(v);
        } else {
            break;
        }
    }
    // If input had fewer than n numbers, just proceed with what we got.
    let (mut items, norm_target, immediate) = normalize_and_quick_checks(nums, target);
    if let Some(sol) = immediate {
        print!("YES\n");
        if !sol.is_empty() {
            for (i, idx) in sol.iter().enumerate() {
                if i != 0 {
                    print!(" ");
                }
                print!("{idx}");
            }
        }
        print!("\n");
        return;
    }
    if items.is_empty() {
        print!("NO\n");
        return;
    }

    // Heuristic: for huge N, first try to find quick exact solutions via sampling + pruning DP
    // without scanning too much memory. We'll still scan all inputs (already read), but
    // computationally keep bounded.
    let n_eff = items.len();

    // Sort by |v| descending so early pruning and greedy are more effective.
    items.sort_by(|a, b| b.v.abs().cmp(&a.v.abs()));

    // Tiny sizes: brute force / MiM exact
    if n_eff <= 26 {
        if let Some(sol) = small_bruteforce(&items, norm_target) {
            print!("YES\n");
            for (i, idx) in sol.iter().enumerate() {
                if i != 0 {
                    print!(" ");
                }
                print!("{idx}");
            }
            print!("\n");
            return;
        }
        print!("NO\n");
        return;
    }
    if n_eff <= 44 {
        if let Some(sol) = meet_in_middle(&items, norm_target) {
            print!("YES\n");
            for (i, idx) in sol.iter().enumerate() {
                if i != 0 {
                    print!(" ");
                }
                print!("{idx}");
            }
            print!("\n");
            return;
        }
        // Fall through to DP prune which can still solve some cases.
    }

    // Medium sizes: pruned exact DP in hash space (good for structured/low-range instances).
    // Cap controls time/memory. Increase slightly for moderate N.
    let cap = match n_eff.cmp(&5000) {
        Ordering::Less => 400_000usize,
        Ordering::Equal => 400_000usize,
        Ordering::Greater => 250_000usize,
    };
    if let Some(sol) = dp_hash_pruned(&items, norm_target, cap) {
        // Verify sum (paranoia) on normalized values already; indices are original.
        print!("YES\n");
        for (i, idx) in sol.iter().enumerate() {
            if i != 0 {
                print!(" ");
            }
            print!("{idx}");
        }
        print!("\n");
        return;
    }

    // Very large sizes: attempt a fast greedy exact (may succeed on many random cases),
    // then declare NO if not found. This sacrifices completeness for performance on huge N.
    // If a solution exists but is hard, we may miss it; however the benchmark request
    // explicitly suggests approximation for very large sets.
    if n_eff > 200_000 {
        if let Some(sol) = greedy_sign_heuristic(&items, norm_target, 200_000) {
            print!("YES\n");
            for (i, idx) in sol.iter().enumerate() {
                if i != 0 {
                    print!(" ");
                }
                print!("{idx}");
            }
            print!("\n");
            return;
        }
        print!("NO\n");
        return;
    }

    // Otherwise: do a second pass DP with tighter pruning around target using a smaller window
    // and slightly higher cap.
    let cap2 = 600_000usize;
    if let Some(sol) = dp_hash_pruned(&items, norm_target, cap2) {
        print!("YES\n");
        for (i, idx) in sol.iter().enumerate() {
            if i != 0 {
                print!(" ");
            }
            print!("{idx}");
        }
        print!("\n");
        return;
    }

    print!("NO\n");
}
