import sys


def solve_exact_cover(universe_size, sets):
    # Map each element -> list of set indices that contain it
    elem_to_sets = [[] for _ in range(universe_size)]
    for i, s in enumerate(sets):
        for e in s:
            if 0 <= e < universe_size:
                elem_to_sets[e].append(i)

    # If any universe element has no covering set, impossible.
    for e in range(universe_size):
        if not elem_to_sets[e]:
            return None

    # Dancing-links-style backtracking with explicit cover counts.
    covered = [False] * universe_size
    used = [False] * len(sets)
    solution = []

    def pick_uncovered_element_with_fewest_options():
        best_e = -1
        best_cnt = 10**18
        for e in range(universe_size):
            if covered[e]:
                continue
            cnt = 0
            for si in elem_to_sets[e]:
                if used[si]:
                    continue
                # valid if it doesn't intersect already covered elements
                ok = True
                for x in sets[si]:
                    if covered[x]:
                        ok = False
                        break
                if ok:
                    cnt += 1
                    if cnt >= best_cnt:
                        break
            if cnt < best_cnt:
                best_cnt = cnt
                best_e = e
                if best_cnt <= 1:
                    break
        return best_e, best_cnt

    def backtrack(remaining):
        if remaining == 0:
            return True

        e, cnt = pick_uncovered_element_with_fewest_options()
        if e < 0 or cnt == 0:
            return False

        # Try each set covering e that is consistent
        for si in elem_to_sets[e]:
            if used[si]:
                continue
            # Check consistency: no already-covered element in set
            for x in sets[si]:
                if covered[x]:
                    break
            else:
                # choose si
                used[si] = True
                solution.append(si)
                newly = []
                for x in sets[si]:
                    if not covered[x]:
                        covered[x] = True
                        newly.append(x)
                if backtrack(remaining - len(newly)):
                    return True
                # undo
                for x in newly:
                    covered[x] = False
                solution.pop()
                used[si] = False

        return False

    if backtrack(universe_size):
        return solution
    return None


def main():
    data = sys.stdin.read().strip().splitlines()
    if not data:
        return
    first = data[0].strip().split()
    if len(first) < 2:
        print("NO SOLUTION")
        return
    try:
        U = int(first[0])
        S = int(first[1])
    except Exception:
        print("NO SOLUTION")
        return

    sets = []
    for i in range(1, 1 + S):
        if i >= len(data):
            sets.append([])
            continue
        line = data[i].strip()
        if not line:
            sets.append([])
            continue
        try:
            elems = [int(x) for x in line.split()]
        except Exception:
            elems = []
        # Remove duplicates; exact cover sets are typically treated as sets
        # but keep deterministic order for speed.
        seen = set()
        uniq = []
        for e in elems:
            if 0 <= e < U and e not in seen:
                seen.add(e)
                uniq.append(e)
        sets.append(uniq)

    sol = solve_exact_cover(U, sets)
    if sol is None:
        sys.stdout.write("NO SOLUTION\n")
    else:
        sys.stdout.write("SOLUTION\n")
        sys.stdout.write(" ".join(str(i) for i in sol) + "\n")


if __name__ == "__main__":
    main()
