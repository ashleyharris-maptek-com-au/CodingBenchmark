use std::cmp::min;
use std::io::{self, Read, Write};

#[derive(Clone, Copy, Default)]
struct Vec3 {
    x: f32,
    y: f32,
    z: f32,
}

impl Vec3 {
    #[inline]
    fn add(self, o: Vec3) -> Vec3 {
        Vec3 {
            x: self.x + o.x,
            y: self.y + o.y,
            z: self.z + o.z,
        }
    }
    #[inline]
    fn sub(self, o: Vec3) -> Vec3 {
        Vec3 {
            x: self.x - o.x,
            y: self.y - o.y,
            z: self.z - o.z,
        }
    }
    #[inline]
    fn mul(self, s: f32) -> Vec3 {
        Vec3 {
            x: self.x * s,
            y: self.y * s,
            z: self.z * s,
        }
    }
    #[inline]
    fn norm2(self) -> f32 {
        self.x * self.x + self.y * self.y + self.z * self.z
    }
}

#[derive(Clone)]
struct Rng64 {
    state: u64,
}

impl Rng64 {
    fn new(seed: u64) -> Self {
        let s = if seed == 0 { 0x9E3779B97F4A7C15 } else { seed };
        Self { state: s }
    }
    #[inline]
    fn next_u64(&mut self) -> u64 {
        self.state = self.state.wrapping_add(0x9E3779B97F4A7C15);
        let mut z = self.state;
        z = (z ^ (z >> 30)).wrapping_mul(0xBF58476D1CE4E5B9);
        z = (z ^ (z >> 27)).wrapping_mul(0x94D049BB133111EB);
        z ^ (z >> 31)
    }
    #[inline]
    fn next_f32(&mut self) -> f32 {
        let x = self.next_u64() >> 40;
        (x as f32) * (1.0 / ((1u32 << 24) as f32))
    }
    #[inline]
    fn gen_range_usize(&mut self, n: usize) -> usize {
        if n <= 1 {
            return 0;
        }
        (self.next_u64() % (n as u64)) as usize
    }
}

fn parse_header_from_ascii_prefix(buf: &[u8]) -> Option<(u64, u32, usize)> {
    let mut i = 0usize;
    while i < buf.len()
        && (buf[i] == b' ' || buf[i] == b'\n' || buf[i] == b'\r' || buf[i] == b'\t')
    {
        i += 1;
    }
    if i >= buf.len() {
        return None;
    }
    let start1 = i;
    while i < buf.len() && buf[i].is_ascii_digit() {
        i += 1;
    }
    if i == start1 {
        return None;
    }
    let n1 = std::str::from_utf8(&buf[start1..i]).ok()?.parse::<u64>().ok()?;
    while i < buf.len()
        && (buf[i] == b' ' || buf[i] == b'\n' || buf[i] == b'\r' || buf[i] == b'\t')
    {
        i += 1;
    }
    if i >= buf.len() {
        return None;
    }
    let start2 = i;
    while i < buf.len() && buf[i].is_ascii_digit() {
        i += 1;
    }
    if i == start2 {
        return None;
    }
    let k = std::str::from_utf8(&buf[start2..i]).ok()?.parse::<u32>().ok()?;
    while i < buf.len()
        && (buf[i] == b' ' || buf[i] == b'\n' || buf[i] == b'\r' || buf[i] == b'\t')
    {
        i += 1;
        if i > 0 && (buf[i - 1] == b'\n' || buf[i - 1] == b'\r') {
            while i < buf.len()
                && (buf[i] == b' ' || buf[i] == b'\n' || buf[i] == b'\r' || buf[i] == b'\t')
            {
                i += 1;
            }
            break;
        }
    }
    Some((n1, k, i))
}

fn read_f32_le(bytes: &[u8]) -> f32 {
    let mut arr = [0u8; 4];
    arr.copy_from_slice(&bytes[..4]);
    f32::from_le_bytes(arr)
}

fn write_u32_le(out: &mut Vec<u8>, v: u32) {
    out.extend_from_slice(&v.to_le_bytes());
}

fn assign_nearest(p: Vec3, centroids: &[Vec3]) -> u32 {
    let mut best = 0u32;
    let mut best_d = f32::INFINITY;
    for (i, &c) in centroids.iter().enumerate() {
        let d = p.sub(c).norm2();
        if d < best_d {
            best_d = d;
            best = i as u32;
        }
    }
    best
}

fn kmeanspp_init(sample: &[Vec3], k: usize, rng: &mut Rng64) -> Vec<Vec3> {
    let n = sample.len();
    if k == 0 {
        return Vec::new();
    }
    let mut centroids = Vec::with_capacity(k);
    centroids.push(sample[rng.gen_range_usize(n)]);
    let mut dist2 = vec![0f32; n];
    for _ in 1..k {
        let mut sum = 0f64;
        for i in 0..n {
            let p = sample[i];
            let mut best = f32::INFINITY;
            for &c in &centroids {
                let d = p.sub(c).norm2();
                if d < best {
                    best = d;
                }
            }
            dist2[i] = best.max(0.0);
            sum += dist2[i] as f64;
        }
        if !sum.is_finite() || sum <= 0.0 {
            centroids.push(sample[rng.gen_range_usize(n)]);
            continue;
        }
        let mut r = rng.next_f32() as f64 * sum;
        let mut idx = 0usize;
        for i in 0..n {
            r -= dist2[i] as f64;
            if r <= 0.0 {
                idx = i;
                break;
            }
        }
        centroids.push(sample[idx]);
    }
    centroids
}

fn lloyd_kmeans(points: &[Vec3], k: usize, max_iters: usize) -> Vec<u32> {
    let n = points.len();
    if k == 0 {
        return vec![0u32; n];
    }
    let mut rng = Rng64::new(0xD1B54A32D192ED03u64 ^ (n as u64).wrapping_mul(0x9E3779B97F4A7C15));
    let mut centroids = kmeanspp_init(points, k, &mut rng);
    let mut assign = vec![0u32; n];

    for _ in 0..max_iters {
        let mut sums = vec![Vec3::default(); k];
        let mut counts = vec![0u32; k];
        let mut changed = 0u64;

        for i in 0..n {
            let p = points[i];
            let a = assign_nearest(p, &centroids) as usize;
            if assign[i] as usize != a {
                changed += 1;
                assign[i] = a as u32;
            }
            sums[a] = sums[a].add(p);
            counts[a] += 1;
        }

        let mut drift = 0f64;
        for j in 0..k {
            if counts[j] == 0 {
                let idx = rng.gen_range_usize(n);
                let newc = points[idx];
                drift += centroids[j].sub(newc).norm2() as f64;
                centroids[j] = newc;
            } else {
                let inv = 1.0f32 / (counts[j] as f32);
                let newc = sums[j].mul(inv);
                drift += centroids[j].sub(newc).norm2() as f64;
                centroids[j] = newc;
            }
        }

        if changed == 0 || drift < 1e-6 {
            break;
        }
    }

    if k <= n {
        let mut seen = vec![false; k];
        for &a in &assign {
            seen[a as usize] = true;
        }
        let mut missing: Vec<usize> = (0..k).filter(|&i| !seen[i]).collect();
        if !missing.is_empty() {
            let mut order: Vec<(f32, usize)> = (0..n)
                .map(|i| (points[i].sub(centroids[assign[i] as usize]).norm2(), i))
                .collect();
            order.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap_or(std::cmp::Ordering::Equal));
            let mut oi = 0usize;
            for m in missing.drain(..) {
                while oi < order.len() && (assign[order[oi].1] as usize) == m {
                    oi += 1;
                }
                if oi >= order.len() {
                    break;
                }
                assign[order[oi].1] = m as u32;
                oi += 1;
            }
        }
    }

    assign
}

fn main() {
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();

    let (num_points_u64, num_clusters_u32, offset) =
        parse_header_from_ascii_prefix(&input).expect("failed to parse header");

    let n = num_points_u64 as usize;
    let k = num_clusters_u32 as usize;
    if k == 0 || n == 0 {
        return;
    }

    let available = input.len().saturating_sub(offset);
    let max_points = available / 12;
    let n_eff = min(n, max_points);

    let in_memory = (n_eff as u64) <= 10_000_000 && k <= 2048;

    let mut out = Vec::<u8>::with_capacity(n_eff * 4);

    if in_memory {
        let mut points = Vec::with_capacity(n_eff);
        let mut pos = offset;
        for _ in 0..n_eff {
            let x = read_f32_le(&input[pos..pos + 4]);
            let y = read_f32_le(&input[pos + 4..pos + 8]);
            let z = read_f32_le(&input[pos + 8..pos + 12]);
            points.push(Vec3 { x, y, z });
            pos += 12;
        }
        let assign = lloyd_kmeans(&points, k, 20);
        for a in assign {
            write_u32_le(&mut out, a as u32);
        }
        io::stdout().write_all(&out).unwrap();
        return;
    }

    let mut rng =
        Rng64::new(0xA0761D6478BD642Fu64 ^ (n_eff as u64).wrapping_mul(0x9E3779B97F4A7C15));

    let sample_target = {
        let base = (k as u64).saturating_mul(32).clamp(4096, 2_000_000);
        min(base as usize, n_eff.max(1))
    };

    let mut sample: Vec<Vec3> = Vec::with_capacity(sample_target);
    let mut pos = offset;
    for i in 0..n_eff {
        let x = read_f32_le(&input[pos..pos + 4]);
        let y = read_f32_le(&input[pos + 4..pos + 8]);
        let z = read_f32_le(&input[pos + 8..pos + 12]);
        let p = Vec3 { x, y, z };
        pos += 12;

        if sample.len() < sample_target {
            sample.push(p);
        } else {
            let j = (rng.next_u64() % ((i + 1) as u64)) as usize;
            if j < sample_target {
                sample[j] = p;
            }
        }
    }

    if k > n_eff {
        for i in 0..n_eff {
            write_u32_le(&mut out, (i as u32) % (k as u32));
        }
        io::stdout().write_all(&out).unwrap();
        return;
    }

    let mut centroids = kmeanspp_init(&sample, k, &mut rng);
    let mut counts = vec![1u64; k];

    let epochs = if n_eff as u64 > 1_000_000_000 {
        1
    } else if n_eff as u64 > 100_000_000 {
        2
    } else {
        3
    };

    let mut batch = vec![Vec3::default(); 0];
    let batch_size = min(131072usize, n_eff.max(1));
    batch.reserve(batch_size);

    for _epoch in 0..epochs {
        let mut pos = offset;
        let mut i = 0usize;
        while i < n_eff {
            batch.clear();
            let take = min(batch_size, n_eff - i);
            for _ in 0..take {
                let x = read_f32_le(&input[pos..pos + 4]);
                let y = read_f32_le(&input[pos + 4..pos + 8]);
                let z = read_f32_le(&input[pos + 8..pos + 12]);
                batch.push(Vec3 { x, y, z });
                pos += 12;
            }
            for &p in &batch {
                let a = assign_nearest(p, &centroids) as usize;
                counts[a] = counts[a].saturating_add(1);
                let lr = 1.0f32 / (counts[a] as f32);
                let c = centroids[a];
                centroids[a] = Vec3 {
                    x: c.x + (p.x - c.x) * lr,
                    y: c.y + (p.y - c.y) * lr,
                    z: c.z + (p.z - c.z) * lr,
                };
            }
            i += take;
        }

        if !sample.is_empty() {
            for j in 0..k {
                if counts[j] < 10 {
                    centroids[j] = sample[rng.gen_range_usize(sample.len())];
                    counts[j] = 10;
                }
            }
        }
    }

    let mut pos = offset;
    for _ in 0..n_eff {
        let x = read_f32_le(&input[pos..pos + 4]);
        let y = read_f32_le(&input[pos + 4..pos + 8]);
        let z = read_f32_le(&input[pos + 8..pos + 12]);
        pos += 12;
        let a = assign_nearest(Vec3 { x, y, z }, &centroids);
        write_u32_le(&mut out, a);
    }

    if k <= n_eff {
        let mut seen = vec![false; k];
        for i in 0..n_eff {
            let idx = i * 4;
            let mut arr = [0u8; 4];
            arr.copy_from_slice(&out[idx..idx + 4]);
            let a = u32::from_le_bytes(arr) as usize;
            if a < k {
                seen[a] = true;
            }
        }
        let mut missing: Vec<usize> = (0..k).filter(|&i| !seen[i]).collect();
        for (mi, m) in missing.drain(..).enumerate() {
            let idx = mi * 4;
            out[idx..idx + 4].copy_from_slice(&(m as u32).to_le_bytes());
        }
    }

    io::stdout().write_all(&out).unwrap();
}
