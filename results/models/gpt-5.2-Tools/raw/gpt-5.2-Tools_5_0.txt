use std::collections::{HashSet, VecDeque};
use std::io::{self, Read};

#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
struct P {
    x: i32,
    y: i32,
}

fn main() {
    // Read all input
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    if s.trim().is_empty() {
        return;
    }
    let mut it = s.split_whitespace();
    let w: i32 = it.next().unwrap().parse().unwrap();
    let h: i32 = it.next().unwrap().parse().unwrap();
    let obs_n: usize = it.next().unwrap().parse().unwrap();
    let require_cycle: bool = it.next().unwrap().parse::<i32>().unwrap() != 0;

    if w <= 0 || h <= 0 {
        println!("0");
        return;
    }

    let mut blocked: HashSet<P> = HashSet::with_capacity(obs_n * 2 + 8);
    for _ in 0..obs_n {
        let x: i32 = it.next().unwrap().parse().unwrap();
        let y: i32 = it.next().unwrap().parse().unwrap();
        blocked.insert(P { x, y });
    }

    // Start must be open.
    let start = P { x: 0, y: 0 };
    if start.x < 0 || start.x >= w || start.y < 0 || start.y >= h || blocked.contains(&start) {
        println!("0");
        return;
    }

    // Gather all open cells and index them.
    let mut cells: Vec<P> = Vec::new();
    cells.reserve((w as i64 * h as i64).min(50_000) as usize);
    for y in 0..h {
        for x in 0..w {
            let p = P { x, y };
            if !blocked.contains(&p) {
                cells.push(p);
            }
        }
    }
    let n = cells.len();
    if n == 0 {
        println!("0");
        return;
    }

    // Helper: map position to index using a compact vector if feasible, else hash.
    // For up to ~10 million we can store a vec<i32>. Here grids are at most "ludicrous",
    // but benchmark mentions 2500 cells; we still make it robust.
    let wh = (w as i64) * (h as i64);
    let use_dense = wh > 0 && wh <= 20_000_000;
    let mut dense_idx: Vec<i32> = Vec::new();
    let mut idx_map: std::collections::HashMap<P, usize> = std::collections::HashMap::new();
    if use_dense {
        dense_idx = vec![-1; wh as usize];
        for (i, &p) in cells.iter().enumerate() {
            dense_idx[(p.y as i64 * w as i64 + p.x as i64) as usize] = i as i32;
        }
    } else {
        idx_map.reserve(n * 2 + 8);
        for (i, &p) in cells.iter().enumerate() {
            idx_map.insert(p, i);
        }
    }
    let mut get_idx = |p: P| -> Option<usize> {
        if p.x < 0 || p.x >= w || p.y < 0 || p.y >= h {
            return None;
        }
        if use_dense {
            let v = dense_idx[(p.y as i64 * w as i64 + p.x as i64) as usize];
            if v >= 0 {
                Some(v as usize)
            } else {
                None
            }
        } else {
            idx_map.get(&p).copied()
        }
    };

    // Build adjacency lists.
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for (i, &p) in cells.iter().enumerate() {
        let neigh = [
            P { x: p.x + 1, y: p.y },
            P { x: p.x - 1, y: p.y },
            P { x: p.x, y: p.y + 1 },
            P { x: p.x, y: p.y - 1 },
        ];
        for &q in &neigh {
            if let Some(j) = get_idx(q) {
                adj[i].push(j);
            }
        }
    }

    let s_idx = get_idx(start).unwrap();

    // Quick connectivity check: all open cells must be reachable from start.
    if !is_connected(n, &adj, s_idx) {
        println!("0");
        return;
    }

    // Basic degree feasibility:
    // For a Hamiltonian path from fixed start:
    // - start must have degree >=1 unless n==1
    // - if require_cycle: all vertices must have degree >=2 (except n==1 handled separately)
    if n == 1 {
        if require_cycle {
            // Single vertex cycle is impossible under adjacency rule.
            println!("0");
        } else {
            println!("1");
            println!("0 0");
        }
        return;
    }
    if adj[s_idx].is_empty() {
        println!("0");
        return;
    }
    if require_cycle && adj.iter().any(|v| v.len() < 2) {
        println!("0");
        return;
    }

    // Grid-bipartite parity necessary conditions.
    if !parity_feasible(w, h, &blocked, require_cycle) {
        println!("0");
        return;
    }

    // Choose solver strategy:
    // - Small N (<= 22): exact DP (Held-Karp style) on graph.
    // - Moderate N (<= 2500): DFS backtracking with strong heuristics (Warnsdorff + pruning).
    // - Larger: constructive serpentine for obstacle-free rectangles; else fail fast with 0.
    //   (General Hamiltonian path with obstacles is NP-complete; no polynomial solver exists.)
    let path_idx_opt = if n <= 22 {
        exact_dp_path(n, &adj, s_idx, require_cycle, &cells, &mut get_idx)
    } else if n <= 2500 {
        heuristic_dfs(n, &adj, s_idx, require_cycle)
    } else {
        // Attempt constructive for empty obstacles only.
        if blocked.is_empty() {
            constructive_snake(w, h, require_cycle, &mut get_idx, n, &cells)
        } else {
            None
        }
    };

    let Some(path_idx) = path_idx_opt else {
        println!("0");
        return;
    };

    // Validate and output.
    if !validate(&path_idx, &adj, s_idx, require_cycle) {
        println!("0");
        return;
    }
    println!("{}", path_idx.len());
    for &i in &path_idx {
        let p = cells[i];
        println!("{} {}", p.x, p.y);
    }
}

fn is_connected(n: usize, adj: &[Vec<usize>], s: usize) -> bool {
    let mut vis = vec![false; n];
    let mut q = VecDeque::new();
    vis[s] = true;
    q.push_back(s);
    let mut seen = 1usize;
    while let Some(v) = q.pop_front() {
        for &u in &adj[v] {
            if !vis[u] {
                vis[u] = true;
                seen += 1;
                q.push_back(u);
            }
        }
    }
    seen == n
}

fn parity_feasible(w: i32, h: i32, blocked: &HashSet<P>, require_cycle: bool) -> bool {
    // Bipartite grid: black/white by (x+y)%2.
    // For Hamiltonian path visiting all open cells:
    // - If cycle required, counts of colors must be equal.
    // - Else, counts must differ by at most 1, and start color must be the larger when odd.
    let mut black = 0i32;
    let mut white = 0i32;
    for y in 0..h {
        for x in 0..w {
            let p = P { x, y };
            if blocked.contains(&p) {
                continue;
            }
            if ((x + y) & 1) == 0 {
                black += 1;
            } else {
                white += 1;
            }
        }
    }
    if require_cycle {
        black == white
    } else {
        let diff = (black - white).abs();
        if diff > 1 {
            return false;
        }
        if (black + white) & 1 == 1 {
            // odd: endpoints are same color; since start fixed at (0,0) (black), black must be larger.
            black >= white
        } else {
            true
        }
    }
}

fn validate(path: &[usize], adj: &[Vec<usize>], s_idx: usize, require_cycle: bool) -> bool {
    if path.is_empty() || path[0] != s_idx {
        return false;
    }
    let n = adj.len();
    if path.len() != n {
        return false;
    }
    let mut seen = vec![false; n];
    seen[path[0]] = true;
    for win in path.windows(2) {
        let a = win[0];
        let b = win[1];
        if seen[b] {
            return false;
        }
        if !adj[a].iter().any(|&x| x == b) {
            return false;
        }
        seen[b] = true;
    }
    if require_cycle {
        let last = *path.last().unwrap();
        adj[last].iter().any(|&x| x == s_idx)
    } else {
        true
    }
}

fn exact_dp_path(
    n: usize,
    adj: &[Vec<usize>],
    s_idx: usize,
    require_cycle: bool,
    _cells: &[P],
    _get_idx: &mut impl FnMut(P) -> Option<usize>,
) -> Option<Vec<usize>> {
    // Bitmask DP: dp[mask][v] = prev vertex u (or sentinel) reaching v with visited mask.
    // Fix start included.
    let start_bit = 1u64 << s_idx;
    let full = (1u64 << n) - 1;
    let mut dp: Vec<Vec<i16>> = vec![vec![-1; n]; (1u64 << n) as usize];
    dp[start_bit as usize][s_idx] = -2; // start marker
    for mask in 0u64..=full {
        if (mask & start_bit) == 0 {
            continue;
        }
        for v in 0..n {
            if dp[mask as usize][v] < -1 {
                // start marker ok
            } else if dp[mask as usize][v] == -1 {
                continue;
            }
            if (mask & (1u64 << v)) == 0 {
                continue;
            }
            for &u in &adj[v] {
                if (mask & (1u64 << u)) != 0 {
                    continue;
                }
                let nmask = mask | (1u64 << u);
                if dp[nmask as usize][u] == -1 {
                    dp[nmask as usize][u] = v as i16;
                }
            }
        }
    }

    // Reconstruct any end vertex satisfying cycle condition if needed.
    let mut end: Option<usize> = None;
    for v in 0..n {
        if dp[full as usize][v] != -1 {
            if require_cycle {
                if adj[v].iter().any(|&x| x == s_idx) {
                    end = Some(v);
                    break;
                }
            } else {
                end = Some(v);
                break;
            }
        }
    }
    let end = end?;
    let mut path = Vec::with_capacity(n);
    let mut mask = full;
    let mut v = end;
    loop {
        path.push(v);
        if v == s_idx && mask == start_bit {
            break;
        }
        let prev = dp[mask as usize][v];
        if prev == -1 {
            return None;
        }
        let pv = prev as i32;
        if pv < 0 {
            // start marker
            break;
        }
        mask &= !(1u64 << v);
        v = pv as usize;
    }
    path.reverse();
    if path.len() == n && path[0] == s_idx {
        Some(path)
    } else {
        None
    }
}

fn heuristic_dfs(n: usize, adj: &[Vec<usize>], s_idx: usize, require_cycle: bool) -> Option<Vec<usize>> {
    // Backtracking with:
    // - Warnsdorff heuristic (choose next with smallest unvisited degree)
    // - Forward-checking for forced moves (degree-1 vertices in remaining graph)
    // - Simple articulation-ish prune via connectivity on remaining graph occasionally
    let mut path: Vec<usize> = Vec::with_capacity(n);
    let mut used = vec![false; n];
    used[s_idx] = true;
    path.push(s_idx);

    // Precompute neighbor order arrays; will be re-sorted dynamically.
    let mut stack: Vec<(usize, Vec<usize>, usize)> = Vec::new();
    // (current vertex, candidate list, next index)

    let mut cur = s_idx;
    loop {
        if path.len() == n {
            if !require_cycle || adj[*path.last().unwrap()].iter().any(|&x| x == s_idx) {
                return Some(path);
            }
            // backtrack
        }

        // Generate candidates for current node.
        let mut cand: Vec<usize> = adj[cur].iter().copied().filter(|&u| !used[u]).collect();
        if !cand.is_empty() {
            // Sort by number of onward moves (Warnsdorff), plus tie-break by static degree.
            cand.sort_by_key(|&u| {
                let onward = adj[u].iter().filter(|&&v| !used[v]).count();
                (onward, adj[u].len())
            });
        }

        stack.push((cur, cand, 0));

        let mut advanced = false;
        while let Some((_v, cands, idx)) = stack.last_mut() {
            if *idx >= cands.len() {
                // Exhausted: pop and undo v unless it's start already undone later.
                stack.pop();
                if path.len() == 1 {
                    return None;
                }
                let last = path.pop().unwrap();
                used[last] = false;
                cur = *path.last().unwrap();
                advanced = true;
                break;
            }
            let u = cands[*idx];
            *idx += 1;
            if used[u] {
                continue;
            }

            // Pruning: if u would be dead-end too early (no unvisited neighbors) and not finishing.
            let u_onward = adj[u].iter().filter(|&&z| !used[z] && z != cur).count();
            let remaining_after = n - (path.len() + 1);
            if remaining_after > 0 && u_onward == 0 {
                continue;
            }

            // Optional: forced-degree prune (endpoints): count of vertices with 1 available degree too high.
            if !degree1_prune(n, adj, &used, require_cycle, s_idx, Some(u)) {
                continue;
            }

            // Occasionally check connectivity of remaining graph (every ~64 steps).
            if (path.len() & 63) == 0 {
                if !remaining_connected(n, adj, &used, u) {
                    continue;
                }
            }

            // Take move.
            used[u] = true;
            path.push(u);
            cur = u;
            advanced = true;
            break;
        }
        if !advanced {
            // No stack to backtrack
            return None;
        }
    }
}

fn degree1_prune(
    n: usize,
    adj: &[Vec<usize>],
    used: &[bool],
    require_cycle: bool,
    s_idx: usize,
    next: Option<usize>,
) -> bool {
    // Compute how many vertices (unvisited + potential next) have only 0/1 exits remaining.
    // For Hamiltonian path (non-cycle), at most 2 vertices can have remaining degree 1 in the final subgraph
    // (the endpoints). Since start fixed, it will be one endpoint; the other endpoint must exist.
    // This heuristic is conservative-ish.
    let mut used2 = used.to_vec();
    if let Some(u) = next {
        used2[u] = true; // assume we step into u now (so it becomes visited)
    }
    let mut deg1 = 0usize;
    let mut deg0 = 0usize;
    for v in 0..n {
        if used2[v] {
            continue;
        }
        let d = adj[v].iter().filter(|&&u| !used2[u]).count();
        if d == 0 {
            deg0 += 1;
            if deg0 > 0 {
                return false;
            }
        } else if d == 1 {
            deg1 += 1;
            if require_cycle {
                // in a cycle, no vertex can end up degree 1 in remaining graph
                if deg1 > 0 {
                    return false;
                }
            } else {
                // allow a small number of forced endpoints; start already one endpoint in the final path
                // (unless internal), but with fixed start, we allow up to 2.
                if deg1 > 2 {
                    return false;
                }
            }
        }
    }
    // Additionally, if require_cycle, ensure start will have 2 neighbors in final; hard to know here.
    // Mild prune: if start unvisited neighbors currently <2 and not yet finished, fail.
    if require_cycle {
        let start_un = adj[s_idx].iter().filter(|&&u| !used2[u]).count();
        if start_un == 0 {
            return false;
        }
    }
    true
}

fn remaining_connected(n: usize, adj: &[Vec<usize>], used: &[bool], current: usize) -> bool {
    // Check that all unvisited nodes are in one component when including current as entry.
    let mut start = None;
    for v in 0..n {
        if !used[v] {
            start = Some(v);
            break;
        }
    }
    let Some(s) = start else {
        return true;
    };
    // If current is visited, we still can traverse through visited? In actual continuation, visited nodes
    // can't be reused, so connectivity should be among unvisited with edges between unvisited plus current.
    let mut vis = vec![false; n];
    let mut q = VecDeque::new();
    vis[s] = true;
    q.push_back(s);
    let mut cnt = 1usize;
    while let Some(v) = q.pop_front() {
        for &u in &adj[v] {
            if used[u] && u != current {
                continue;
            }
            if used[v] && v != current {
                continue;
            }
            if !vis[u] && (!used[u] || u == current) {
                vis[u] = true;
                cnt += 1;
                q.push_back(u);
            }
        }
    }
    let total_un = used.iter().filter(|&&b| !b).count();
    cnt == total_un
}

fn constructive_snake(
    w: i32,
    h: i32,
    require_cycle: bool,
    get_idx: &mut impl FnMut(P) -> Option<usize>,
    n: usize,
    _cells: &[P],
) -> Option<Vec<usize>> {
    // Standard snake order for empty rectangles provides a Hamiltonian path from (0,0).
    // Cycle exists only when both dimensions even (for empty grid).
    if require_cycle {
        if (w % 2 != 0) || (h % 2 != 0) {
            return None;
        }
    }
    let mut order: Vec<P> = Vec::with_capacity(n);
    for y in 0..h {
        if (y & 1) == 0 {
            for x in 0..w {
                order.push(P { x, y });
            }
        } else {
            for x in (0..w).rev() {
                order.push(P { x, y });
            }
        }
    }
    let mut path: Vec<usize> = Vec::with_capacity(n);
    for p in order {
        path.push(get_idx(p)?);
    }
    Some(path)
}
