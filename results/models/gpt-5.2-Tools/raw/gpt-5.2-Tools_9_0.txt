using System;
using System.Buffers;
using System.Collections.Generic;
using System.IO;
using System.Runtime.CompilerServices;

// 1D Cutting Stock / Bin Packing (capacity = stock_length)
// Fast heuristic: sort cuts descending, then Best-Fit Decreasing using a Treap keyed by remaining space.
// Output: indices assigned to each stock.
public static class Program
{
    public static void Main()
    {
        var fs = Console.OpenStandardInput();
        using var br = new FastReader(fs);

        int n = br.ReadInt();
        int stockLen = br.ReadInt();
        if (n <= 0)
        {
            Console.Write("0");
            return;
        }

        var cuts = new int[n];
        for (int i = 0; i < n; i++) cuts[i] = br.ReadInt();

        // Validate quickly: if any cut > stockLen, impossible per constraints.
        for (int i = 0; i < n; i++)
        {
            if (cuts[i] > stockLen)
            {
                // Still must output something; emit each cut alone (will violate), but constraints say won't happen.
                // Safer: throw.
                Console.Error.WriteLine("Invalid input: cut exceeds stock length.");
                return;
            }
        }

        // Order cuts by length descending, keep original indices.
        var order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        Array.Sort(order, (a, b) => cuts[b].CompareTo(cuts[a]));

        // Stocks: list of assigned cut indices.
        var stocksCuts = new List<int[]>(capacity: Math.Min(n, 1024));
        var stocksCounts = new List<int>(capacity: Math.Min(n, 1024));

        // Treap of open stocks, keyed by remaining capacity.
        var treap = new Treap();

        for (int oi = 0; oi < n; oi++)
        {
            int idx = order[oi];
            int len = cuts[idx];

            // Find stock with smallest remaining >= len (best fit).
            int nodeId = treap.LowerBound(len);
            if (nodeId == 0)
            {
                // Open new stock
                int stockId = stocksCuts.Count;
                stocksCuts.Add(new int[4]);
                stocksCounts.Add(0);
                int rem = stockLen - len;
                AddCutToStock(stocksCuts, stocksCounts, stockId, idx);
                treap.Insert(rem, stockId);
            }
            else
            {
                // Use that stock
                var (rem, stockId) = treap.GetKeyVal(nodeId);
                treap.Erase(nodeId);
                int newRem = rem - len;
                AddCutToStock(stocksCuts, stocksCounts, stockId, idx);
                treap.Insert(newRem, stockId);
            }
        }

        // Output
        using var sw = new StreamWriter(Console.OpenStandardOutput(), new System.Text.UTF8Encoding(false), 1 << 20);
        sw.WriteLine(stocksCuts.Count);
        for (int s = 0; s < stocksCuts.Count; s++)
        {
            int cnt = stocksCounts[s];
            var arr = stocksCuts[s];
            for (int j = 0; j < cnt; j++)
            {
                if (j != 0) sw.Write(' ');
                sw.Write(arr[j]);
            }
            sw.WriteLine();
        }
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static void AddCutToStock(List<int[]> stocksCuts, List<int> stocksCounts, int stockId, int cutIndex)
    {
        int cnt = stocksCounts[stockId];
        var arr = stocksCuts[stockId];
        if ((uint)cnt >= (uint)arr.Length)
        {
            var newArr = new int[arr.Length << 1];
            Array.Copy(arr, newArr, arr.Length);
            arr = newArr;
            stocksCuts[stockId] = arr;
        }
        arr[cnt] = cutIndex;
        stocksCounts[stockId] = cnt + 1;
    }

    private sealed class FastReader : IDisposable
    {
        private readonly Stream _s;
        private byte[] _buf;
        private int _len, _pos;

        public FastReader(Stream s, int bufSize = 1 << 20)
        {
            _s = s;
            _buf = ArrayPool<byte>.Shared.Rent(bufSize);
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int Read()
        {
            if (_pos >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_pos++];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int ReadInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return 0;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int x = 0;
            while (c > 32 && c != -1)
            {
                x = x * 10 + (c - '0');
                c = Read();
            }
            return x * sign;
        }

        public void Dispose()
        {
            var b = _buf;
            _buf = Array.Empty<byte>();
            if (b.Length != 0) ArrayPool<byte>.Shared.Return(b);
        }
    }

    // Treap keyed by remaining capacity, with stockId as tie-breaker to keep keys unique.
    // Supports: lower bound by remaining>=need (i.e. key >= need) with key = (rem, stockId).
    private sealed class Treap
    {
        private struct Node
        {
            public int Rem;
            public int StockId;
            public uint Pri;
            public int L, R;
        }

        private Node[] _nodes = new Node[1 << 20];
        private int _cnt;
        private int _root;
        private uint _rng = 2463534242u;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private uint NextU()
        {
            // xorshift32
            uint x = _rng;
            x ^= x << 13;
            x ^= x >> 17;
            x ^= x << 5;
            _rng = x;
            return x;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private static bool Less(int remA, int idA, int remB, int idB)
        {
            return remA < remB || (remA == remB && idA < idB);
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int NewNode(int rem, int stockId)
        {
            int id = ++_cnt;
            if (id >= _nodes.Length)
            {
                Array.Resize(ref _nodes, _nodes.Length << 1);
            }
            _nodes[id].Rem = rem;
            _nodes[id].StockId = stockId;
            _nodes[id].Pri = NextU();
            _nodes[id].L = 0;
            _nodes[id].R = 0;
            return id;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void Insert(int rem, int stockId)
        {
            // If rem is negative (shouldn't), still store; lower bound logic will avoid it for positive needs.
            int node = NewNode(rem, stockId);
            InsertNode(ref _root, node);
        }

        private void InsertNode(ref int root, int node)
        {
            if (root == 0)
            {
                root = node;
                return;
            }
            ref var r = ref _nodes[root];
            ref var n = ref _nodes[node];
            if (Less(n.Rem, n.StockId, r.Rem, r.StockId))
            {
                InsertNode(ref r.L, node);
                if (_nodes[r.L].Pri < r.Pri) RotateRight(ref root);
            }
            else
            {
                InsertNode(ref r.R, node);
                if (_nodes[r.R].Pri < r.Pri) RotateLeft(ref root);
            }
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private void RotateRight(ref int root)
        {
            int x = root;
            int y = _nodes[x].L;
            _nodes[x].L = _nodes[y].R;
            _nodes[y].R = x;
            root = y;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private void RotateLeft(ref int root)
        {
            int x = root;
            int y = _nodes[x].R;
            _nodes[x].R = _nodes[y].L;
            _nodes[y].L = x;
            root = y;
        }

        // Returns node id with smallest (rem,stockId) such that rem >= need.
        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int LowerBound(int need)
        {
            int cur = _root;
            int best = 0;
            while (cur != 0)
            {
                var node = _nodes[cur];
                if (node.Rem >= need)
                {
                    best = cur;
                    cur = node.L;
                }
                else
                {
                    cur = node.R;
                }
            }
            return best;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public (int Rem, int StockId) GetKeyVal(int nodeId)
        {
            return (_nodes[nodeId].Rem, _nodes[nodeId].StockId);
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void Erase(int nodeId)
        {
            var node = _nodes[nodeId];
            EraseByKey(ref _root, node.Rem, node.StockId);
        }

        private void EraseByKey(ref int root, int rem, int stockId)
        {
            if (root == 0) return;
            ref var r = ref _nodes[root];
            if (r.Rem == rem && r.StockId == stockId)
            {
                // Merge children by priority
                root = Merge(r.L, r.R);
                return;
            }
            if (Less(rem, stockId, r.Rem, r.StockId))
            {
                EraseByKey(ref r.L, rem, stockId);
            }
            else
            {
                EraseByKey(ref r.R, rem, stockId);
            }
        }

        private int Merge(int a, int b)
        {
            if (a == 0) return b;
            if (b == 0) return a;
            if (_nodes[a].Pri < _nodes[b].Pri)
            {
                _nodes[a].R = Merge(_nodes[a].R, b);
                return a;
            }
            else
            {
                _nodes[b].L = Merge(a, _nodes[b].L);
                return b;
            }
        }
    }
}
