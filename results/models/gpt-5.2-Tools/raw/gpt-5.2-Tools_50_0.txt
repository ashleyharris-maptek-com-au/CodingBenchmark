#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

float safeInv(float x) { return (x > 0.0) ? (1.0 / x) : 0.0; }

vec3 limitSpeed(vec3 v, float maxSpeed) {
    float s2 = dot(v, v);
    float m2 = maxSpeed * maxSpeed;
    if (s2 > m2 && s2 > 0.0) {
        v *= maxSpeed * inversesqrt(s2);
    }
    return v;
}

void main() {
    uint id = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (id >= N) return;

    uint idx = id * 2u;
    vec3 pos = inData[idx].xyz;
    vec3 vel = inData[idx + 1u].xyz;

    float dt = radii.x;
    float sepR = radii.y;
    float alignR = radii.z;
    float cohR = radii.w;

    float sepW = weights.x;
    float alignW = weights.y;
    float cohW = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    vec3 sepAcc = vec3(0.0);
    vec3 alignSum = vec3(0.0);
    vec3 cohSum = vec3(0.0);

    uint sepCount = 0u;
    uint alignCount = 0u;
    uint cohCount = 0u;

    float sepR2 = sepR * sepR;
    float alignR2 = alignR * alignR;
    float cohR2 = cohR * cohR;

    for (uint j = 0u; j < N; ++j) {
        if (j == id) continue;
        uint jidx = j * 2u;
        vec3 p2 = inData[jidx].xyz;
        vec3 v2 = inData[jidx + 1u].xyz;

        vec3 d = p2 - pos;
        float dist2 = dot(d, d);
        if (dist2 == 0.0) continue;

        if (dist2 < sepR2) {
            // steer away, stronger when closer
            sepAcc -= d * inversesqrt(dist2);
            sepCount++;
        }
        if (dist2 < alignR2) {
            alignSum += v2;
            alignCount++;
        }
        if (dist2 < cohR2) {
            cohSum += p2;
            cohCount++;
        }
    }

    vec3 accel = vec3(0.0);

    if (sepCount > 0u) {
        accel += (sepAcc * safeInv(float(sepCount))) * sepW;
    }

    if (alignCount > 0u) {
        vec3 avgVel = alignSum * safeInv(float(alignCount));
        accel += (avgVel - vel) * alignW;
    }

    if (cohCount > 0u) {
        vec3 center = cohSum * safeInv(float(cohCount));
        accel += (center - pos) * cohW;
    }

    // Boundary force: when beyond boundSize*0.4, push back toward center.
    // Use boundForce and boundSize*0.5 scaling.
    float thresh = boundSize * 0.4;
    float halfSize = boundSize * 0.5;

    if (pos.x > thresh) accel.x -= boundForce * ((pos.x - thresh) / halfSize);
    else if (pos.x < -thresh) accel.x += boundForce * ((-thresh - pos.x) / halfSize);

    if (pos.y > thresh) accel.y -= boundForce * ((pos.y - thresh) / halfSize);
    else if (pos.y < -thresh) accel.y += boundForce * ((-thresh - pos.y) / halfSize);

    if (pos.z > thresh) accel.z -= boundForce * ((pos.z - thresh) / halfSize);
    else if (pos.z < -thresh) accel.z += boundForce * ((-thresh - pos.z) / halfSize);

    vel += accel * dt;
    vel = limitSpeed(vel, maxSpeed);
    pos += vel * dt;

    outData[idx] = vec4(pos, 0.0);
    outData[idx + 1u] = vec4(vel, 0.0);
}
