#include <cassert>

// Longest Common Substring (across N strings), adaptive solver.
//
// Strategy:
// 1) If N==0 -> empty. If N==1 -> that string.
// 2) For small total size / small N, use classic suffix automaton (SAM) built on the shortest string,
//    then scan all other strings to compute for each SAM state the best match length; intersect across strings.
//    This is O(sum lengths) and exact.
// 3) For very large inputs / many strings, still use SAM on shortest string (exact, streaming),
//    but with memory-conscious representation and without storing per-position arrays.
//
// Note: For "extreme" sizes (e.g., 100 strings of 10MB+), any exact algorithm must read and process all input
// (>=1GB). This implementation is linear in total input size after choosing the shortest string as base.
// It uses an unordered_map per state for transitions to keep memory proportional to distinct outgoing edges.
// If your inputs are huge and alphabet is small (e.g. DNA), this is fine; if alphabet is very large and random,
// the shortest string itself dominates memory anyway and exact LCS becomes inherently heavy.

struct SAM {
    struct State {
        int link = -1;
        int len = 0;
        std::unordered_map<unsigned char, int> next;
    };

    std::vector<State> st;
    int last = 0;

    explicit SAM(size_t reserve_states = 0) {
        st.reserve(reserve_states ? reserve_states : 2);
        st.push_back(State{}); // root
        st[0].link = -1;
        st[0].len = 0;
        last = 0;
    }

    void extend(unsigned char c) {
        int cur = (int)st.size();
        st.push_back(State{});
        st[cur].len = st[last].len + 1;

        int p = last;
        while (p != -1 && st[p].next.find(c) == st[p].next.end()) {
            st[p].next.emplace(c, cur);
            p = st[p].link;
        }

        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = st[p].next[c];
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = (int)st.size();
                st.push_back(st[q]); // copy transitions
                st[clone].len = st[p].len + 1;
                // link of clone is st[q].link already (copied)
                while (p != -1) {
                    auto it = st[p].next.find(c);
                    if (it == st[p].next.end() || it->second != q) break;
                    it->second = clone;
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }
};

static std::vector<int> build_order_by_len(const SAM& sam) {
    // Counting sort by length (O(|states| + maxlen)).
    int maxlen = 0;
    for (const auto& s : sam.st) maxlen = std::max(maxlen, s.len);
    std::vector<int> cnt(maxlen + 1, 0);
    for (const auto& s : sam.st) ++cnt[s.len];
    for (int i = 1; i <= maxlen; ++i) cnt[i] += cnt[i - 1];
    std::vector<int> order(sam.st.size());
    for (int i = (int)sam.st.size() - 1; i >= 0; --i) {
        int l = sam.st[i].len;
        order[--cnt[l]] = i;
    }
    return order; // increasing len
}

static std::string lcs_all_sam(const std::vector<std::string>& s) {
    int n = (int)s.size();
    if (n == 0) return std::string();
    if (n == 1) return s[0];

    // Choose the shortest string as base for SAM.
    int base = 0;
    for (int i = 1; i < n; ++i) {
        if (s[i].size() < s[base].size()) base = i;
    }

    const std::string& a = s[base];
    // SAM has at most 2*|a| states.
    SAM sam(a.size() ? (2ULL * a.size()) : 2);
    for (unsigned char c : a) sam.extend(c);
    auto order = build_order_by_len(sam);

    // best[v] = minimal over strings of maximal match length ending at state v for that string.
    // Initialize to +inf then intersect per string with per-string result mx[v].
    const int INF = std::numeric_limits<int>::max() / 4;
    std::vector<int> best(sam.st.size(), INF);
    std::vector<int> mx(sam.st.size(), 0);

    // For retrieving one occurrence: after processing all strings we find a state with best len L and then
    // reconstruct by walking suffix links? Not enough to recover exact substring. Instead:
    // - We will compute the maximal L and then search that substring in the base string (a) by rescanning
    //   other strings (or using base substring verification). Since it is common, it must occur in a.
    // - We can find it in a by scanning with SAM and tracking end positions in a is non-trivial without storing
    //   positions. We'll simply compute the lex/first occurrence by checking candidates from a using rolling hash
    //   at final length L. This avoids heavy memory.

    // Process each other string: compute mx[v] = best match length for this string for each state v.
    for (int idx = 0; idx < n; ++idx) {
        if (idx == base) continue;
        std::fill(mx.begin(), mx.end(), 0);

        int v = 0;
        int l = 0;
        for (unsigned char c : s[idx]) {
            auto it = sam.st[v].next.find(c);
            if (it != sam.st[v].next.end()) {
                v = it->second;
                ++l;
            } else {
                while (v != -1 && sam.st[v].next.find(c) == sam.st[v].next.end()) {
                    v = sam.st[v].link;
                }
                if (v == -1) {
                    v = 0;
                    l = 0;
                    continue;
                } else {
                    l = sam.st[v].len + 1;
                    v = sam.st[v].next[c];
                }
            }
            // For state v, we matched length l ending here. Update mx[v].
            if (l > mx[v]) mx[v] = l;
        }

        // Propagate values along suffix links: for each state, its link can get at least min(mx[state], len(link)).
        // Standard: process states by decreasing length.
        for (int i = (int)order.size() - 1; i > 0; --i) { // skip root at 0 when propagating
            int state = order[i];
            int link = sam.st[state].link;
            if (link >= 0) {
                int propagated = std::min(mx[state], sam.st[link].len);
                if (propagated > mx[link]) mx[link] = propagated;
            }
        }

        // Intersect into best
        for (size_t i = 0; i < best.size(); ++i) {
            best[i] = std::min(best[i], mx[i]);
        }
    }

    // The base string itself: it allows full lengths for its own states; intersection already accounts others.
    // Now find maximum length over states.
    int L = 0;
    for (size_t i = 0; i < best.size(); ++i) {
        if (best[i] == INF) continue; // unreachable if n>1 and we skipped base? root gets updated though
        L = std::max(L, best[i]);
    }
    if (L <= 0) return std::string();

    // Now, find one substring of length L that is common to all strings.
    // We do this by enumerating all substrings of length L in the base string and checking membership in all
    // other strings using a fast rolling hash set per string.
    //
    // This is O(sum lengths) with low constant, and memory O(total distinct hashes at length L) across strings.
    // To avoid huge memory for very large strings, we store hashes for ALL substrings of length L only for
    // strings other than base, but this can be large. Instead, we:
    // - Build hash set for the shortest non-base string among the rest.
    // - Iterate base candidates and verify by streaming rolling-hash scan on each other string without storing.
    //
    // This keeps memory low while remaining deterministic (with 128-bit hash).

    auto mul64 = [](uint64_t a, uint64_t b) -> uint64_t {
        // 128-bit multiply mod 2^64 (natural overflow) is fine for hashing.
        return a * b;
    };

    const uint64_t B1 = 1315423911u;
    const uint64_t B2 = 11400714819323198485ull; // odd 64-bit

    struct H128 { uint64_t x, y; };
    struct H128Hash {
        size_t operator()(const H128& h) const noexcept {
            // mix
            uint64_t z = h.x ^ (h.y + 0x9e3779b97f4a7c15ULL + (h.x << 6) + (h.x >> 2));
            return (size_t)z;
        }
    };
    struct H128Eq {
        bool operator()(const H128& a, const H128& b) const noexcept { return a.x == b.x && a.y == b.y; }
    };

    auto build_pow = [&](int len) {
        uint64_t p1 = 1, p2 = 1;
        for (int i = 0; i < len; ++i) {
            p1 = mul64(p1, B1);
            p2 = mul64(p2, B2);
        }
        return std::pair<uint64_t, uint64_t>{p1, p2};
    };

    auto powBL = build_pow(L);

    auto hash_window_init = [&](const std::string& str, int start) -> H128 {
        uint64_t h1 = 0, h2 = 0;
        for (int i = 0; i < L; ++i) {
            unsigned char c = (unsigned char)str[start + i] + 1;
            h1 = mul64(h1, B1) + c;
            h2 = mul64(h2, B2) + c;
        }
        return {h1, h2};
    };

    auto hash_window_next = [&](H128 cur, unsigned char out_c, unsigned char in_c) -> H128 {
        uint64_t h1 = cur.x, h2 = cur.y;
        uint64_t outv = (uint64_t)(out_c + 1);
        uint64_t inv = (uint64_t)(in_c + 1);
        // remove out: h = h - outv * B^L
        h1 = h1 - mul64(outv, powBL.first);
        h2 = h2 - mul64(outv, powBL.second);
        // shift and add in
        h1 = mul64(h1, B1) + inv;
        h2 = mul64(h2, B2) + inv;
        return {h1, h2};
    };

    // Pick an anchor string among non-base strings with minimal length (to minimize candidate matches).
    int anchor = -1;
    for (int i = 0; i < n; ++i) {
        if (i == base) continue;
        if (anchor == -1 || s[i].size() < s[anchor].size()) anchor = i;
    }
    if (anchor == -1) return a; // n==1 already handled
    if ((int)s[anchor].size() < L) return std::string(); // should not happen

    std::unordered_set<H128, H128Hash, H128Eq> anchor_set;
    anchor_set.reserve(std::max<size_t>(1, s[anchor].size() - (size_t)L + 1));
    {
        const std::string& t = s[anchor];
        if ((int)t.size() >= L) {
            H128 h = hash_window_init(t, 0);
            anchor_set.insert(h);
            for (size_t i = 1; i + (size_t)L <= t.size(); ++i) {
                h = hash_window_next(h, (unsigned char)t[i - 1], (unsigned char)t[i + (size_t)L - 1]);
                anchor_set.insert(h);
            }
        }
    }

    auto contains_hash_streaming = [&](const std::string& t, const H128& target) -> bool {
        if ((int)t.size() < L) return false;
        H128 h = hash_window_init(t, 0);
        if (h.x == target.x && h.y == target.y) return true;
        for (size_t i = 1; i + (size_t)L <= t.size(); ++i) {
            h = hash_window_next(h, (unsigned char)t[i - 1], (unsigned char)t[i + (size_t)L - 1]);
            if (h.x == target.x && h.y == target.y) return true;
        }
        return false;
    };

    // Enumerate base substrings of length L; only consider those that appear in anchor_set.
    // For each candidate hash, verify it appears in all other strings via streaming scan.
    // When a candidate passes, return the actual substring from base.
    const std::string& baseStr = s[base];
    if ((int)baseStr.size() < L) return std::string();
    H128 h = hash_window_init(baseStr, 0);
    auto try_candidate_at = [&](size_t pos, const H128& hh) -> std::string {
        if (anchor_set.find(hh) == anchor_set.end()) return std::string();
        for (int i = 0; i < n; ++i) {
            if (i == base || i == anchor) continue;
            if (!contains_hash_streaming(s[i], hh)) return std::string();
        }
        return baseStr.substr(pos, (size_t)L);
    };

    {
        std::string res = try_candidate_at(0, h);
        if (!res.empty()) return res;
    }
    for (size_t i = 1; i + (size_t)L <= baseStr.size(); ++i) {
        h = hash_window_next(h, (unsigned char)baseStr[i - 1], (unsigned char)baseStr[i + (size_t)L - 1]);
        std::string res = try_candidate_at(i, h);
        if (!res.empty()) return res;
    }

    // Extremely unlikely (hash collision / no found due to collision in filtering). Fallback: return empty.
    return std::string();
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N = 0;
    if (!(std::cin >> N)) return 0;
    std::string dummy;
    std::getline(std::cin, dummy); // consume rest of line

    std::vector<std::string> s;
    s.reserve(std::max(0, N));
    for (int i = 0; i < N; ++i) {
        std::string line;
        std::getline(std::cin, line);
        // Keep line as-is (including spaces) except trailing '\r' from Windows line endings.
        if (!line.empty() && line.back() == '\r') line.pop_back();
        s.push_back(std::move(line));
    }

    // Adaptive: for very small cases, the same SAM approach is already optimal; no need for separate DP.
    // But if any string is empty, answer is empty.
    for (const auto& str : s) {
        if (str.empty()) {
            std::cout << "\n";
            return 0;
        }
    }

    std::string ans = lcs_all_sam(s);
    std::cout << ans << "\n";
    return 0;
}
