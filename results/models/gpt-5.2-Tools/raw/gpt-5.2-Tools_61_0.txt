/**
 * Vehicle Routing (multi-route) heuristic solver.
 *
 * Reads one JSON doc from stdin with:
 * - depot: [x,y]
 * - truckCount: max routes
 * - packages: {id,x,y}[]
 *
 * Outputs: {"routes": [[id,...], ...]}
 * Each package appears exactly once.
 *
 * Approach:
 * 1) Build initial routes by deterministic greedy insertion with farthest-seed starts.
 * 2) Improve with 2-opt within routes.
 * 3) Improve with cross-route relocations and swaps under time budget.
 *
 * No external deps.
 */

declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Pkg = { id: any; x: number; y: number };

const depotArr: [number, number] = [Number(data.depot[0]), Number(data.depot[1])];
const truckCount: number = Math.max(1, Number(data.truckCount) | 0);
const packages: Pkg[] = (data.packages as any[]).map(p => ({ id: p.id, x: Number(p.x), y: Number(p.y) }));

const n = packages.length;

function nowMs(): number {
  // node hrtime for stable monotonic time
  const t = process.hrtime.bigint();
  return Number(t / 1000000n);
}

// Deterministic PRNG for tie-breaking (xorshift32)
let rngState = 0x9e3779b9 ^ (n * 2654435761);
function rndU32(): number {
  rngState |= 0;
  rngState ^= (rngState << 13);
  rngState ^= (rngState >>> 17);
  rngState ^= (rngState << 5);
  return rngState >>> 0;
}

// Build coordinate arrays
const xs = new Float64Array(n + 1);
const ys = new Float64Array(n + 1);
xs[0] = depotArr[0];
ys[0] = depotArr[1];
for (let i = 0; i < n; i++) {
  xs[i + 1] = packages[i].x;
  ys[i + 1] = packages[i].y;
}

function dist(a: number, b: number): number {
  const dx = xs[a] - xs[b];
  const dy = ys[a] - ys[b];
  return Math.hypot(dx, dy);
}

// Precompute depot distances for seeding and heuristics
const depotDist = new Float64Array(n);
for (let i = 0; i < n; i++) depotDist[i] = Math.hypot(xs[i + 1] - xs[0], ys[i + 1] - ys[0]);

// Choose seeds: farthest from depot, then farthest from existing seeds (k-center-ish)
const routeCount = Math.min(truckCount, Math.max(1, n));
const usedSeed = new Uint8Array(n);
const seeds: number[] = [];
if (n > 0) {
  // first: farthest from depot
  let best = 0;
  for (let i = 1; i < n; i++) if (depotDist[i] > depotDist[best]) best = i;
  seeds.push(best);
  usedSeed[best] = 1;
  while (seeds.length < routeCount) {
    let cand = -1;
    let candScore = -1;
    for (let i = 0; i < n; i++) {
      if (usedSeed[i]) continue;
      // distance to closest seed
      let minD = Infinity;
      const pi = i + 1;
      for (let s = 0; s < seeds.length; s++) {
        const sj = seeds[s] + 1;
        const d = Math.hypot(xs[pi] - xs[sj], ys[pi] - ys[sj]);
        if (d < minD) minD = d;
      }
      const score = minD;
      if (score > candScore) {
        candScore = score;
        cand = i;
      }
    }
    if (cand < 0) break;
    seeds.push(cand);
    usedSeed[cand] = 1;
  }
}

// Route representation: array of node indices (1..n) (package indices +1)
// distance caches
class Route {
  nodes: number[];
  len: number;
  constructor(seedNode: number) {
    this.nodes = seedNode ? [seedNode] : [];
    this.len = this.computeLen();
  }
  computeLen(): number {
    const m = this.nodes.length;
    if (m === 0) return 0;
    let total = dist(0, this.nodes[0]);
    for (let i = 0; i < m - 1; i++) total += dist(this.nodes[i], this.nodes[i + 1]);
    total += dist(this.nodes[m - 1], 0);
    return total;
  }
}

const assigned = new Uint8Array(n);
const routes: Route[] = [];
for (let r = 0; r < routeCount; r++) {
  const seedIdx = seeds[r] ?? -1;
  if (seedIdx >= 0) {
    assigned[seedIdx] = 1;
    routes.push(new Route(seedIdx + 1));
  } else {
    routes.push(new Route(0));
  }
}

// If fewer seeds than routes, fill empty routes with random unassigned
if (n > 0) {
  for (let r = 0; r < routes.length; r++) {
    if (routes[r].nodes.length) continue;
    let pick = -1;
    // pick farthest unassigned
    let bestD = -1;
    for (let i = 0; i < n; i++) {
      if (assigned[i]) continue;
      if (depotDist[i] > bestD) {
        bestD = depotDist[i];
        pick = i;
      }
    }
    if (pick >= 0) {
      assigned[pick] = 1;
      routes[r].nodes.push(pick + 1);
      routes[r].len = routes[r].computeLen();
    }
  }
}

// Gather unassigned nodes
const unassigned: number[] = [];
for (let i = 0; i < n; i++) if (!assigned[i]) unassigned.push(i + 1);

// Greedy best insertion: for each unassigned node, insert into best route position with min delta
function bestInsertion(route: Route, node: number): { pos: number; delta: number } {
  const nodes = route.nodes;
  const m = nodes.length;
  if (m === 0) {
    // depot -> node -> depot
    return { pos: 0, delta: 2 * dist(0, node) };
  }
  let bestPos = 0;
  let bestDelta = Infinity;
  // insert before first
  {
    const delta = dist(0, node) + dist(node, nodes[0]) - dist(0, nodes[0]);
    bestDelta = delta;
    bestPos = 0;
  }
  // between
  for (let i = 0; i < m - 1; i++) {
    const a = nodes[i];
    const b = nodes[i + 1];
    const delta = dist(a, node) + dist(node, b) - dist(a, b);
    if (delta < bestDelta - 1e-12) {
      bestDelta = delta;
      bestPos = i + 1;
    } else if (Math.abs(delta - bestDelta) <= 1e-12) {
      // deterministic tie-break
      if ((rndU32() & 1) === 0) {
        bestDelta = delta;
        bestPos = i + 1;
      }
    }
  }
  // after last
  {
    const last = nodes[m - 1];
    const delta = dist(last, node) + dist(node, 0) - dist(last, 0);
    if (delta < bestDelta - 1e-12) {
      bestDelta = delta;
      bestPos = m;
    } else if (Math.abs(delta - bestDelta) <= 1e-12) {
      if ((rndU32() & 1) === 0) {
        bestDelta = delta;
        bestPos = m;
      }
    }
  }
  return { pos: bestPos, delta: bestDelta };
}

// Insert nodes in order of decreasing depot distance (farthest first tends to help)
unassigned.sort((a, b) => depotDist[b - 1] - depotDist[a - 1] || (a - b));
for (let ui = 0; ui < unassigned.length; ui++) {
  const node = unassigned[ui];
  let bestR = 0;
  let bestPos = 0;
  let bestDelta = Infinity;
  for (let r = 0; r < routes.length; r++) {
    const ins = bestInsertion(routes[r], node);
    if (ins.delta < bestDelta - 1e-12) {
      bestDelta = ins.delta;
      bestR = r;
      bestPos = ins.pos;
    } else if (Math.abs(ins.delta - bestDelta) <= 1e-12) {
      // choose route with smaller current length; else deterministic
      const lr = routes[r].len;
      const lb = routes[bestR].len;
      if (lr < lb - 1e-9 || ((lr === lb) && ((rndU32() & 1) === 0))) {
        bestDelta = ins.delta;
        bestR = r;
        bestPos = ins.pos;
      }
    }
  }
  const rt = routes[bestR];
  rt.nodes.splice(bestPos, 0, node);
  rt.len += bestDelta;
}

// 2-opt within a route (first improvement)
function twoOpt(route: Route, maxIters: number): boolean {
  const nodes = route.nodes;
  const m = nodes.length;
  if (m < 4) return false;
  let improved = false;
  let iter = 0;
  while (iter < maxIters) {
    iter++;
    let found = false;
    // edges (i-1,i) and (k,k+1)
    for (let i = 0; i < m - 2; i++) {
      const a = (i === 0) ? 0 : nodes[i - 1];
      const b = nodes[i];
      for (let k = i + 1; k < m - 1; k++) {
        const c = nodes[k];
        const d = nodes[k + 1];
        const before = dist(a, b) + dist(c, d);
        const after = dist(a, c) + dist(b, d);
        if (after + 1e-12 < before) {
          // reverse segment [i..k]
          for (let l = i, r = k; l < r; l++, r--) {
            const tmp = nodes[l];
            nodes[l] = nodes[r];
            nodes[r] = tmp;
          }
          route.len += (after - before);
          improved = true;
          found = true;
          break;
        }
      }
      if (found) break;
    }
    if (!found) break;
  }
  return improved;
}

// Compute delta for removing node at pos from route
function removalDelta(nodes: number[], pos: number): number {
  const m = nodes.length;
  const prev = (pos === 0) ? 0 : nodes[pos - 1];
  const cur = nodes[pos];
  const next = (pos === m - 1) ? 0 : nodes[pos + 1];
  // length decreases by (prev-cur + cur-next - prev-next)
  return -(dist(prev, cur) + dist(cur, next) - dist(prev, next));
}

function insertionDelta(nodes: number[], insertPos: number, node: number): number {
  const m = nodes.length;
  if (m === 0) return 2 * dist(0, node);
  const prev = (insertPos === 0) ? 0 : nodes[insertPos - 1];
  const next = (insertPos === m) ? 0 : nodes[insertPos];
  return dist(prev, node) + dist(node, next) - dist(prev, next);
}

const start = nowMs();
// budget heuristics: scale with n but cap
const timeBudget = Math.max(50, Math.min(1800, 40 + Math.floor(Math.sqrt(n) * 35)));

// Initial 2-opt pass
for (let r = 0; r < routes.length; r++) twoOpt(routes[r], 50);

// Cross-route improvements under budget
function crossRouteSearch(): void {
  const endBy = start + timeBudget;
  const R = routes.length;
  if (n <= 1) return;

  // Precompute candidate lists: nearest neighbors for each node (limited)
  const K = n <= 300 ? 18 : (n <= 1200 ? 12 : 8);
  const neigh: number[][] = new Array(n + 1);
  // Build using sampling grid-ish: for large n, approximate by random sample for speed
  if (n <= 1800) {
    for (let i = 1; i <= n; i++) {
      const dists: { j: number; d: number }[] = [];
      for (let j = 1; j <= n; j++) {
        if (i === j) continue;
        const dx = xs[i] - xs[j];
        const dy = ys[i] - ys[j];
        dists.push({ j, d: dx * dx + dy * dy });
      }
      dists.sort((a, b) => a.d - b.d || (a.j - b.j));
      const arr: number[] = [];
      for (let t = 0; t < Math.min(K, dists.length); t++) arr.push(dists[t].j);
      neigh[i] = arr;
    }
  } else {
    // Large: use random sampling to get a few close-ish candidates
    for (let i = 1; i <= n; i++) {
      const arr: { j: number; d: number }[] = [];
      const samples = Math.min(n - 1, K * 40);
      for (let t = 0; t < samples; t++) {
        let j = (rndU32() % n) + 1;
        if (j === i) { t--; continue; }
        const dx = xs[i] - xs[j];
        const dy = ys[i] - ys[j];
        const d = dx * dx + dy * dy;
        arr.push({ j, d });
      }
      arr.sort((a, b) => a.d - b.d || (a.j - b.j));
      const uniq: number[] = [];
      const seen = new Uint8Array(n + 1);
      for (let t = 0; t < arr.length && uniq.length < K; t++) {
        const j = arr[t].j;
        if (!seen[j]) { seen[j] = 1; uniq.push(j); }
      }
      neigh[i] = uniq;
    }
  }

  // Map node->(routeIndex,pos)
  const routeOf = new Int32Array(n + 1);
  const posOf = new Int32Array(n + 1);
  function rebuildMaps() {
    for (let r = 0; r < R; r++) {
      const nodes = routes[r].nodes;
      for (let i = 0; i < nodes.length; i++) {
        const v = nodes[i];
        routeOf[v] = r;
        posOf[v] = i;
      }
    }
  }
  rebuildMaps();

  let improved = true;
  while (improved && nowMs() < endBy) {
    improved = false;

    // Attempt relocations guided by neighbor pairs
    for (let v = 1; v <= n && nowMs() < endBy; v++) {
      const rFrom = routeOf[v];
      const fromNodes = routes[rFrom].nodes;
      const pFrom = posOf[v];
      if (fromNodes.length <= 1) continue;
      const removeGain = -removalDelta(fromNodes, pFrom); // positive means improvement when removing
      // Try move near one of its neighbors
      const candidates = neigh[v];
      for (let ci = 0; ci < candidates.length && nowMs() < endBy; ci++) {
        const u = candidates[ci];
        const rTo = routeOf[u];
        if (rTo === rFrom) continue;
        const toNodes = routes[rTo].nodes;
        const pu = posOf[u];
        // try insert before or after u
        for (let side = 0; side < 2; side++) {
          const insPos = side === 0 ? pu : pu + 1;
          const insCost = insertionDelta(toNodes, insPos, v);
          const net = insCost - removeGain;
          if (net < -1e-9) {
            // apply move
            fromNodes.splice(pFrom, 1);
            // adjust if moving within same route (not possible here)
            toNodes.splice(insPos, 0, v);
            routes[rFrom].len += removalDelta(fromNodes, Math.min(pFrom, fromNodes.length));
            // recompute accurate lengths for touched routes (safer due to splice index drift)
            routes[rFrom].len = routes[rFrom].computeLen();
            routes[rTo].len = routes[rTo].computeLen();
            rebuildMaps();
            // quick 2-opt on affected routes
            twoOpt(routes[rFrom], 10);
            twoOpt(routes[rTo], 10);
            rebuildMaps();
            improved = true;
            break;
          }
        }
        if (improved) break;
      }
    }

    // Attempt swaps between routes for a limited number of tries
    const tries = Math.min(6000, 200 + n * 6);
    for (let t = 0; t < tries && nowMs() < endBy; t++) {
      const a = (rndU32() % n) + 1;
      const b = (rndU32() % n) + 1;
      if (a === b) continue;
      const ra = routeOf[a];
      const rb = routeOf[b];
      if (ra === rb) continue;
      const na = routes[ra].nodes;
      const nb = routes[rb].nodes;
      const pa = posOf[a];
      const pb = posOf[b];

      // compute delta if swap a and b
      function localDelta(nodes: number[], pos: number, newNode: number): number {
        const m = nodes.length;
        const prev = (pos === 0) ? 0 : nodes[pos - 1];
        const cur = nodes[pos];
        const next = (pos === m - 1) ? 0 : nodes[pos + 1];
        const before = dist(prev, cur) + dist(cur, next);
        const after = dist(prev, newNode) + dist(newNode, next);
        return after - before;
      }

      const deltaA = localDelta(na, pa, b);
      const deltaB = localDelta(nb, pb, a);
      const net = deltaA + deltaB;
      if (net < -1e-9) {
        na[pa] = b;
        nb[pb] = a;
        routes[ra].len += deltaA;
        routes[rb].len += deltaB;
        // cleanup with small 2-opt
        twoOpt(routes[ra], 6);
        twoOpt(routes[rb], 6);
        rebuildMaps();
        improved = true;
      }
    }
  }
}

crossRouteSearch();

// Final 2-opt polish (very small)
for (let r = 0; r < routes.length; r++) twoOpt(routes[r], 20);

// Output routes as package IDs
const outRoutes: any[] = [];
for (let r = 0; r < routes.length; r++) {
  const ids: any[] = [];
  for (let i = 0; i < routes[r].nodes.length; i++) {
    ids.push(packages[routes[r].nodes[i] - 1].id);
  }
  if (ids.length) outRoutes.push(ids);
}

// Edge case: no packages
const output = { routes: outRoutes };
process.stdout.write(JSON.stringify(output));
