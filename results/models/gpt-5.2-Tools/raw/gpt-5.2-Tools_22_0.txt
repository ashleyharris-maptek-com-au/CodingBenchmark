using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Runtime.CompilerServices;

public static class Program
{
    private enum Cell : byte { Empty = 0, Filled = 1 }

    private sealed class Board
    {
        public readonly int W;
        public readonly int H;
        public readonly Cell[,] Grid; // [y,x], y=0 bottom
        public readonly int[] RowCount;
        public readonly int[] ColHeight;
        public int MaxHeight;

        public Board(int w, int h)
        {
            W = w;
            H = h;
            Grid = new Cell[h, w];
            RowCount = new int[h];
            ColHeight = new int[w];
            MaxHeight = 0;
        }

        public void RecomputeHeights()
        {
            Array.Clear(ColHeight, 0, ColHeight.Length);
            Array.Clear(RowCount, 0, RowCount.Length);
            MaxHeight = 0;
            for (int y = 0; y < H; y++)
            {
                int cnt = 0;
                for (int x = 0; x < W; x++) if (Grid[y, x] != Cell.Empty) cnt++;
                RowCount[y] = cnt;
                if (cnt > 0) MaxHeight = Math.Max(MaxHeight, y + 1);
            }
            for (int x = 0; x < W; x++)
            {
                int y = H - 1;
                while (y >= 0 && Grid[y, x] == Cell.Empty) y--;
                ColHeight[x] = y + 1;
            }
        }

        public Board Clone()
        {
            var b = new Board(W, H);
            Buffer.BlockCopy(Grid, 0, b.Grid, 0, W * H);
            Array.Copy(RowCount, b.RowCount, H);
            Array.Copy(ColHeight, b.ColHeight, W);
            b.MaxHeight = MaxHeight;
            return b;
        }

        public int ClearFullLines()
        {
            int cleared = 0;
            int writeY = 0;
            for (int y = 0; y < H; y++)
            {
                if (RowCount[y] == W) { cleared++; continue; }
                if (writeY != y)
                {
                    for (int x = 0; x < W; x++) Grid[writeY, x] = Grid[y, x];
                    RowCount[writeY] = RowCount[y];
                }
                writeY++;
            }
            for (int y = writeY; y < H; y++)
            {
                if (RowCount[y] != 0)
                {
                    for (int x = 0; x < W; x++) Grid[y, x] = Cell.Empty;
                    RowCount[y] = 0;
                }
            }
            if (cleared > 0) RecomputeHeights();
            return cleared;
        }
    }

    private readonly struct Placement
    {
        public readonly int Rot;
        public readonly int X;
        public readonly double Score;
        public Placement(int rot, int x, double score) { Rot = rot; X = x; Score = score; }
    }

    private sealed class Piece
    {
        public readonly (sbyte x, sbyte y)[][] Blocks = new (sbyte, sbyte)[4][];
        public readonly sbyte[] Width = new sbyte[4];
        public readonly sbyte[] Height = new sbyte[4];
        public readonly bool[] ValidRot = new bool[4];
        public Piece((sbyte x, sbyte y)[][] rots)
        {
            for (int r = 0; r < 4; r++)
            {
                var b = rots[r];
                Blocks[r] = b;
                if (b == null || b.Length == 0) { ValidRot[r] = false; continue; }
                sbyte maxX = 0, maxY = 0;
                for (int i = 0; i < b.Length; i++)
                {
                    if (b[i].x > maxX) maxX = b[i].x;
                    if (b[i].y > maxY) maxY = b[i].y;
                }
                Width[r] = (sbyte)(maxX + 1);
                Height[r] = (sbyte)(maxY + 1);
                ValidRot[r] = true;
            }
        }
    }

    private static readonly Dictionary<char, Piece> Pieces = BuildPieces();

    private static Dictionary<char, Piece> BuildPieces()
    {
        static (sbyte, sbyte)[] Rot(params (int x, int y)[] pts)
        {
            var arr = new (sbyte, sbyte)[pts.Length];
            for (int i = 0; i < pts.Length; i++) arr[i] = ((sbyte)pts[i].x, (sbyte)pts[i].y);
            return arr;
        }

        var dict = new Dictionary<char, Piece>(8);
        dict['I'] = new Piece(new[] { Rot((0, 0), (1, 0), (2, 0), (3, 0)), Rot((0, 0), (0, 1), (0, 2), (0, 3)), Rot((0, 0), (1, 0), (2, 0), (3, 0)), Rot((0, 0), (0, 1), (0, 2), (0, 3)) });
        dict['O'] = new Piece(new[] { Rot((0, 0), (1, 0), (0, 1), (1, 1)), Rot((0, 0), (1, 0), (0, 1), (1, 1)), Rot((0, 0), (1, 0), (0, 1), (1, 1)), Rot((0, 0), (1, 0), (0, 1), (1, 1)) });
        dict['T'] = new Piece(new[] { Rot((0, 0), (1, 0), (2, 0), (1, 1)), Rot((0, 0), (0, 1), (0, 2), (1, 1)), Rot((0, 1), (1, 1), (2, 1), (1, 0)), Rot((1, 0), (1, 1), (1, 2), (0, 1)) });
        dict['S'] = new Piece(new[] { Rot((1, 0), (2, 0), (0, 1), (1, 1)), Rot((0, 0), (0, 1), (1, 1), (1, 2)), Rot((1, 0), (2, 0), (0, 1), (1, 1)), Rot((0, 0), (0, 1), (1, 1), (1, 2)) });
        dict['Z'] = new Piece(new[] { Rot((0, 0), (1, 0), (1, 1), (2, 1)), Rot((1, 0), (0, 1), (1, 1), (0, 2)), Rot((0, 0), (1, 0), (1, 1), (2, 1)), Rot((1, 0), (0, 1), (1, 1), (0, 2)) });
        dict['J'] = new Piece(new[] { Rot((0, 1), (0, 0), (1, 0), (2, 0)), Rot((0, 0), (1, 0), (0, 1), (0, 2)), Rot((0, 0), (1, 0), (2, 0), (2, 1)), Rot((1, 0), (1, 1), (1, 2), (0, 2)) });
        dict['L'] = new Piece(new[] { Rot((2, 1), (0, 0), (1, 0), (2, 0)), Rot((0, 0), (0, 1), (0, 2), (1, 2)), Rot((0, 1), (0, 0), (1, 0), (2, 0)), Rot((0, 0), (1, 0), (1, 1), (1, 2)) });
        return dict;
    }

    private static readonly double[] Uniform7 = { 1d / 7, 1d / 7, 1d / 7, 1d / 7, 1d / 7, 1d / 7, 1d / 7 };
    private static readonly Piece[] NextPieces7 = { Pieces['I'], Pieces['O'], Pieces['T'], Pieces['S'], Pieces['Z'], Pieces['J'], Pieces['L'] };

    public static int Main()
    {
        var input = new FastScanner();
        int width = input.NextInt();
        int height = input.NextInt();
        int numPieces = input.NextInt();

        var board = new Board(width, height);
        int piecesSeen = 0;

        for (;;)
        {
            string? tok = input.NextTokenOrNull();
            if (tok == null) break;
            if (tok.Length == 0) continue;
            char pt = tok[0];
            if (!Pieces.TryGetValue(pt, out var piece)) continue;

            piecesSeen++;
            double pressure = (double)board.MaxHeight / height;
            bool riskyMode = pressure > 0.70;
            bool deepMode = pressure < 0.35 && piecesSeen < 2000;

            Placement best = ChooseBestPlacement(board, piece, riskyMode, deepMode);
            Console.WriteLine($"{best.Rot} {best.X}");
            Console.Out.Flush();

            string? res = input.NextTokenOrNull();
            if (res == null) break;
            if (res.StartsWith("gameover", StringComparison.OrdinalIgnoreCase)) break;

            _ = TryPlace(board, piece, best.Rot, best.X, out _);

            if (numPieces > 0 && piecesSeen >= numPieces) break;
        }
        return 0;
    }

    private static Placement ChooseBestPlacement(Board board, Piece piece, bool riskyMode, bool deepMode)
    {
        double bestScore = double.NegativeInfinity;
        int bestRot = 0, bestX = 0;

        for (int rot = 0; rot < 4; rot++)
        {
            if (!piece.ValidRot[rot]) continue;
            int pw = piece.Width[rot];
            for (int x = 0; x <= board.W - pw; x++)
            {
                if (!TrySimulate(board, piece, rot, x, out var after, out int cleared)) continue;
                double score = Evaluate(after, cleared, riskyMode);

                if (deepMode)
                {
                    double ev = 0;
                    for (int i = 0; i < 7; i++)
                    {
                        var p2 = NextPieces7[i];
                        var pl2 = ChooseBestPlacementNoLookahead(after, p2, riskyMode);
                        ev += Uniform7[i] * pl2.Score;
                    }
                    score = score * 0.7 + ev * 0.3;
                }

                if (score > bestScore)
                {
                    bestScore = score;
                    bestRot = rot;
                    bestX = x;
                }
            }
        }

        return new Placement(bestRot, bestX, bestScore);
    }

    private static Placement ChooseBestPlacementNoLookahead(Board board, Piece piece, bool riskyMode)
    {
        double bestScore = double.NegativeInfinity;
        int bestRot = 0, bestX = 0;
        for (int rot = 0; rot < 4; rot++)
        {
            if (!piece.ValidRot[rot]) continue;
            int pw = piece.Width[rot];
            for (int x = 0; x <= board.W - pw; x++)
            {
                if (!TrySimulate(board, piece, rot, x, out var after, out int cleared)) continue;
                double score = Evaluate(after, cleared, riskyMode);
                if (score > bestScore) { bestScore = score; bestRot = rot; bestX = x; }
            }
        }
        return new Placement(bestRot, bestX, bestScore);
    }

    private static bool TrySimulate(Board board, Piece piece, int rot, int x, out Board after, out int cleared)
    {
        after = board.Clone();
        if (!TryPlace(after, piece, rot, x, out cleared)) return false;
        return true;
    }

    private static bool TryPlace(Board board, Piece piece, int rot, int x, out int cleared)
    {
        cleared = 0;
        if (!piece.ValidRot[rot]) return false;
        if (!TryFindDropY(board, piece, rot, x, out int y)) return false;
        var blocks = piece.Blocks[rot];
        for (int i = 0; i < blocks.Length; i++)
        {
            int bx = x + blocks[i].x;
            int by = y + blocks[i].y;
            if (by < 0 || by >= board.H || bx < 0 || bx >= board.W) return false;
            if (board.Grid[by, bx] != Cell.Empty) return false;
        }
        for (int i = 0; i < blocks.Length; i++)
        {
            int bx = x + blocks[i].x;
            int by = y + blocks[i].y;
            board.Grid[by, bx] = Cell.Filled;
            board.RowCount[by]++;
            if (by + 1 > board.ColHeight[bx]) board.ColHeight[bx] = by + 1;
            if (by + 1 > board.MaxHeight) board.MaxHeight = by + 1;
        }
        cleared = board.ClearFullLines();
        return true;
    }

    private static bool TryFindDropY(Board board, Piece piece, int rot, int x, out int dropY)
    {
        dropY = 0;
        var blocks = piece.Blocks[rot];
        int ph = piece.Height[rot];

        int y = Math.Min(board.H - ph, Math.Max(0, board.MaxHeight));
        while (y > 0 && !Collides(board, blocks, x, y - 1)) y--;
        if (Collides(board, blocks, x, y)) return false;
        dropY = y;
        return true;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static bool Collides(Board board, (sbyte x, sbyte y)[] blocks, int ox, int oy)
    {
        for (int i = 0; i < blocks.Length; i++)
        {
            int x = ox + blocks[i].x;
            int y = oy + blocks[i].y;
            if ((uint)x >= (uint)board.W) return true;
            if ((uint)y >= (uint)board.H) return true;
            if (board.Grid[y, x] != Cell.Empty) return true;
        }
        return false;
    }

    private static double Evaluate(Board board, int cleared, bool riskyMode)
    {
        int w = board.W, h = board.H;
        int aggHeight = 0;
        int bump = 0;
        int holes = 0;
        int wells = 0;

        for (int x = 0; x < w; x++)
        {
            int colH = board.ColHeight[x];
            aggHeight += colH;
            for (int y = 0; y < colH; y++) if (board.Grid[y, x] == Cell.Empty) holes++;
        }
        for (int x = 0; x < w - 1; x++) bump += Math.Abs(board.ColHeight[x] - board.ColHeight[x + 1]);
        for (int x = 0; x < w; x++)
        {
            int left = x == 0 ? int.MaxValue : board.ColHeight[x - 1];
            int right = x == w - 1 ? int.MaxValue : board.ColHeight[x + 1];
            int d = Math.Min(left, right) - board.ColHeight[x];
            if (d > 0) wells += d * (d + 1) / 2;
        }

        double danger = (double)board.MaxHeight / h;
        double topPenalty = riskyMode ? (danger * danger * danger) : (danger * danger);
        double wf = 10.0 / Math.Max(10, w);

        double score =
            + cleared * 1.6
            - aggHeight * (0.045 * wf)
            - board.MaxHeight * (0.11 * wf)
            - holes * 0.36
            - bump * (0.055 * wf)
            - wells * 0.02
            - topPenalty * 6.5;

        if (cleared >= 2) score += 0.6 * (cleared - 1);
        if (cleared == 4) score += 2.2;
        return score;
    }

    private sealed class FastScanner
    {
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len;
        private int _pos;
        private readonly System.IO.Stream _s = Console.OpenStandardInput();

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int ReadByte()
        {
            if (_pos >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_pos++];
        }

        public string? NextTokenOrNull()
        {
            int c;
            do { c = ReadByte(); if (c == -1) return null; } while (c <= 32);
            var sb = new System.Text.StringBuilder(8);
            while (c > 32)
            {
                sb.Append((char)c);
                c = ReadByte();
                if (c == -1) break;
            }
            return sb.ToString();
        }

        public int NextInt()
        {
            int c;
            do { c = ReadByte(); if (c == -1) return 0; } while (c <= 32);
            int sign = 1;
            if (c == '-') { sign = -1; c = ReadByte(); }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
                if (c == -1) break;
            }
            return val * sign;
        }
    }
}
