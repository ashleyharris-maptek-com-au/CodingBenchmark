use std::cmp::Ordering;
use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone)]
struct XorShift64 {
    state: u64,
}
impl XorShift64 {
    fn new(seed: u64) -> Self {
        let mut s = seed;
        if s == 0 {
            s = 0x9e3779b97f4a7c15;
        }
        // scramble a bit
        s ^= s << 7;
        s ^= s >> 9;
        s ^= s << 8;
        Self { state: s }
    }
    #[inline]
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 7;
        x ^= x >> 9;
        x ^= x << 8;
        self.state = x;
        x
    }
    #[inline]
    fn next_usize(&mut self, m: usize) -> usize {
        if m <= 1 {
            return 0;
        }
        (self.next_u64() as usize) % m
    }
    #[inline]
    fn next_f64(&mut self) -> f64 {
        // [0,1)
        let x = self.next_u64() >> 11;
        (x as f64) * (1.0 / ((1u64 << 53) as f64))
    }
}

fn build_greedy_solution(n: usize, adj: &Vec<Vec<usize>>, order: Option<&[usize]>) -> Vec<u8> {
    let mut in_set = vec![0u8; n];
    let mut dominated = vec![0u8; n];
    let mut undom = n;

    let mut deg = vec![0usize; n];
    for v in 0..n {
        deg[v] = adj[v].len();
    }

    // Helper: add vertex to DS.
    let mut add_vertex = |v: usize,
                          in_set: &mut Vec<u8>,
                          dominated: &mut Vec<u8>,
                          undom: &mut usize| {
        if in_set[v] == 1 {
            return;
        }
        in_set[v] = 1;
        if dominated[v] == 0 {
            dominated[v] = 1;
            *undom -= 1;
        }
        for &u in &adj[v] {
            if dominated[u] == 0 {
                dominated[u] = 1;
                *undom -= 1;
            }
        }
    };

    // 1) Always include isolated vertices
    for v in 0..n {
        if deg[v] == 0 {
            add_vertex(v, &mut in_set, &mut dominated, &mut undom);
        }
    }

    // 2) Greedy: pick vertex covering most currently-undominated vertices
    // To avoid O(n^2) in dense graphs, we do repeated scanning with occasional early exits.
    // Use provided order (randomized permutation) to break ties.
    while undom > 0 {
        let mut best_v = 0usize;
        let mut best_gain = 0usize;

        if let Some(ord) = order {
            for &v in ord {
                if in_set[v] == 1 {
                    continue;
                }
                let mut gain = 0usize;
                if dominated[v] == 0 {
                    gain += 1;
                }
                for &u in &adj[v] {
                    if dominated[u] == 0 {
                        gain += 1;
                    }
                }
                if gain > best_gain {
                    best_gain = gain;
                    best_v = v;
                    if best_gain == undom {
                        break;
                    }
                }
            }
        } else {
            for v in 0..n {
                if in_set[v] == 1 {
                    continue;
                }
                let mut gain = 0usize;
                if dominated[v] == 0 {
                    gain += 1;
                }
                for &u in &adj[v] {
                    if dominated[u] == 0 {
                        gain += 1;
                    }
                }
                if gain > best_gain {
                    best_gain = gain;
                    best_v = v;
                    if best_gain == undom {
                        break;
                    }
                }
            }
        }

        if best_gain == 0 {
            // Shouldn't happen, but to be safe: pick any undominated vertex and add itself
            let mut v = 0;
            while v < n && dominated[v] == 1 {
                v += 1;
            }
            if v == n {
                break;
            }
            add_vertex(v, &mut in_set, &mut dominated, &mut undom);
        } else {
            add_vertex(best_v, &mut in_set, &mut dominated, &mut undom);
        }
    }

    in_set
}

fn repair_to_dominating(
    n: usize,
    adj: &Vec<Vec<usize>>,
    in_set: &mut Vec<u8>,
    dominated: &mut Vec<u8>,
    dominated_by: &mut Vec<i32>,
    ds_list: &mut Vec<usize>,
) {
    // Ensure consistency of dominated/dominated_by/ds_list with in_set (assumed mostly consistent).
    // If any vertex is undominated, add a good vertex (best gain) to cover it.
    loop {
        let mut undom_v: Option<usize> = None;
        for v in 0..n {
            if dominated[v] == 0 {
                undom_v = Some(v);
                break;
            }
        }
        let Some(v0) = undom_v else { break };

        // Candidate: choose among v0 and its neighbors the vertex with max gain on undominated.
        let mut best_v = v0;
        let mut best_gain = 0usize;

        let mut eval = |cand: usize, best_v: &mut usize, best_gain: &mut usize| {
            if in_set[cand] == 1 {
                return;
            }
            let mut gain = 0usize;
            if dominated[cand] == 0 {
                gain += 1;
            }
            for &u in &adj[cand] {
                if dominated[u] == 0 {
                    gain += 1;
                }
            }
            if gain > *best_gain {
                *best_gain = gain;
                *best_v = cand;
            }
        };

        eval(v0, &mut best_v, &mut best_gain);
        for &u in &adj[v0] {
            eval(u, &mut best_v, &mut best_gain);
        }
        if best_gain == 0 {
            best_v = v0;
        }

        // Add best_v
        in_set[best_v] = 1;
        ds_list.push(best_v);
        // update dominated and dominated_by
        let mark = |x: usize,
                    dominator: usize,
                    dominated: &mut Vec<u8>,
                    dominated_by: &mut Vec<i32>| {
            if dominated[x] == 0 {
                dominated[x] = 1;
                dominated_by[x] = dominator as i32;
            }
        };
        mark(best_v, best_v, dominated, dominated_by);
        for &u in &adj[best_v] {
            mark(u, best_v, dominated, dominated_by);
        }
    }
}

fn compute_state(
    n: usize,
    adj: &Vec<Vec<usize>>,
    in_set: &Vec<u8>,
) -> (Vec<u8>, Vec<i32>, Vec<usize>) {
    let mut dominated = vec![0u8; n];
    let mut dominated_by = vec![-1i32; n];
    let mut ds_list = Vec::new();
    for v in 0..n {
        if in_set[v] == 1 {
            ds_list.push(v);
        }
    }
    // Initialize domination by assigning each dominated vertex to some dominator (first found).
    for &d in &ds_list {
        if dominated[d] == 0 {
            dominated[d] = 1;
            dominated_by[d] = d as i32;
        }
        for &u in &adj[d] {
            if dominated[u] == 0 {
                dominated[u] = 1;
                dominated_by[u] = d as i32;
            }
        }
    }
    (dominated, dominated_by, ds_list)
}

fn local_search(
    n: usize,
    adj: &Vec<Vec<usize>>,
    mut in_set: Vec<u8>,
    time_limit: Duration,
    seed: u64,
) -> Vec<u8> {
    let start = Instant::now();
    let mut rng = XorShift64::new(seed);

    let (mut dominated, mut dominated_by, mut ds_list) = compute_state(n, adj, &in_set);
    repair_to_dominating(
        n,
        adj,
        &mut in_set,
        &mut dominated,
        &mut dominated_by,
        &mut ds_list,
    );

    // Prepare list of vertices with degree 0 already fixed: keep them always.
    let mut fixed = vec![0u8; n];
    for v in 0..n {
        if adj[v].is_empty() {
            fixed[v] = 1;
        }
    }

    // Maintain positions of ds_list for O(1) removal swaps.
    let mut pos = vec![usize::MAX; n];
    for (i, &v) in ds_list.iter().enumerate() {
        pos[v] = i;
    }

    // Helper: remove a dominator vertex d (assumes in_set[d]==1 and not fixed)
    let mut remove_vertex = |d: usize,
                             in_set: &mut Vec<u8>,
                             dominated: &mut Vec<u8>,
                             dominated_by: &mut Vec<i32>,
                             ds_list: &mut Vec<usize>,
                             pos: &mut Vec<usize>| {
        in_set[d] = 0;
        // remove from ds_list swap-remove
        let idx = pos[d];
        let last = *ds_list.last().unwrap();
        ds_list[idx] = last;
        pos[last] = idx;
        ds_list.pop();
        pos[d] = usize::MAX;

        // vertices that were dominated by d may become undominated; we will recompute for those vertices only.
        // Collect affected vertices: d and neighbors.
        let mut affected = Vec::with_capacity(adj[d].len() + 1);
        affected.push(d);
        for &u in &adj[d] {
            affected.push(u);
        }
        // For each affected vertex x, if it was assigned to d, try to reassign to any remaining dominator.
        for x in affected {
            if dominated_by[x] == d as i32 {
                let mut new_dom: i32 = -1;
                if in_set[x] == 1 {
                    new_dom = x as i32;
                } else {
                    for &u in &adj[x] {
                        if in_set[u] == 1 {
                            new_dom = u as i32;
                            break;
                        }
                    }
                }
                if new_dom == -1 {
                    dominated[x] = 0;
                    dominated_by[x] = -1;
                } else {
                    dominated[x] = 1;
                    dominated_by[x] = new_dom;
                }
            }
        }
    };

    // Helper: add a vertex to DS and update domination assignment for newly dominated vertices
    let mut add_vertex = |v: usize,
                          in_set: &mut Vec<u8>,
                          dominated: &mut Vec<u8>,
                          dominated_by: &mut Vec<i32>,
                          ds_list: &mut Vec<usize>,
                          pos: &mut Vec<usize>| {
        if in_set[v] == 1 {
            return;
        }
        in_set[v] = 1;
        pos[v] = ds_list.len();
        ds_list.push(v);
        // for v and neighbors, if undominated then assign
        let mut mark = |x: usize,
                        dominator: usize,
                        dominated: &mut Vec<u8>,
                        dominated_by: &mut Vec<i32>| {
            if dominated[x] == 0 {
                dominated[x] = 1;
                dominated_by[x] = dominator as i32;
            }
        };
        mark(v, v, dominated, dominated_by);
        for &u in &adj[v] {
            mark(u, v, dominated, dominated_by);
        }
    };

    // Greedy redundant removal pass (deterministic):
    // Remove vertices that are not needed, in order of increasing degree (or random tiebreak).
    let mut vertices = (0..n).collect::<Vec<_>>();
    vertices.sort_by(|&a, &b| adj[a].len().cmp(&adj[b].len()));

    let mut is_dominated_if_removed = |d: usize,
                                      in_set: &Vec<u8>,
                                      adj: &Vec<Vec<usize>>,
                                      dominated: &Vec<u8>,
                                      dominated_by: &Vec<i32>| {
        // Check if removing d would undominate anyone in N[d] that is currently assigned to d
        // and has no alternate dominator in remaining set.
        if in_set[d] == 0 {
            return true;
        }
        // For each x in N[d] with dominated_by[x]==d, see if there is an alternate dominator.
        let mut check_x = |x: usize| -> bool {
            if dominated[x] == 0 || dominated_by[x] != d as i32 {
                return true;
            }
            // alternate could be x itself if in_set[x]==1 and x!=d (since d will be removed)
            if x != d && in_set[x] == 1 {
                return true;
            }
            for &u in &adj[x] {
                if u != d && in_set[u] == 1 {
                    return true;
                }
            }
            false
        };
        if !check_x(d) {
            return false;
        }
        for &u in &adj[d] {
            if !check_x(u) {
                return false;
            }
        }
        true
    };

    for &d in &vertices {
        if start.elapsed() >= time_limit {
            break;
        }
        if fixed[d] == 1 {
            continue;
        }
        if in_set[d] == 1 && is_dominated_if_removed(d, &in_set, adj, &dominated, &dominated_by) {
            remove_vertex(
                d,
                &mut in_set,
                &mut dominated,
                &mut dominated_by,
                &mut ds_list,
                &mut pos,
            );
        }
    }

    // Stochastic 1-for-2 / 1-for-3 improvements: try remove one vertex, then repair by adding few best.
    // Keep best solution found.
    let mut best = in_set.clone();
    let mut best_size = best.iter().map(|&x| x as usize).sum::<usize>();

    let max_add = 3usize;
    while start.elapsed() < time_limit {
        if ds_list.is_empty() {
            break;
        }
        // pick a random non-fixed dominator
        let mut tries = 0;
        let mut d: usize = ds_list[rng.next_usize(ds_list.len())];
        loop {
            let idx = rng.next_usize(ds_list.len());
            let cand = ds_list[idx];
            if fixed[cand] == 0 {
                d = cand;
                break;
            }
            tries += 1;
            if tries > 20 {
                // no non-fixed found quickly
                break;
            }
        }
        if tries > 20 {
            break;
        }

        // Snapshot small diff: we will perform tentative move and revert if not improved.
        // To revert efficiently, just clone vectors occasionally (acceptable under time limit).
        let in_set_before = in_set.clone();
        let dominated_before = dominated.clone();
        let dominated_by_before = dominated_by.clone();
        let ds_list_before = ds_list.clone();
        let pos_before = pos.clone();

        remove_vertex(
            d,
            &mut in_set,
            &mut dominated,
            &mut dominated_by,
            &mut ds_list,
            &mut pos,
        );

        // If already dominating after removal, great.
        // Else, repair by adding up to max_add vertices greedily focusing on undominated.
        let mut added = 0usize;
        while added < max_add {
            let mut undom_v = None;
            for v in 0..n {
                if dominated[v] == 0 {
                    undom_v = Some(v);
                    break;
                }
            }
            let Some(v0) = undom_v else { break };

            // Choose best candidate among v0 and its neighbors; with occasional random exploration.
            let mut candidates = Vec::with_capacity(adj[v0].len() + 1);
            candidates.push(v0);
            candidates.extend_from_slice(&adj[v0]);

            let mut best_v = candidates[0];
            let mut best_gain = 0usize;
            for &cand in &candidates {
                if in_set[cand] == 1 {
                    continue;
                }
                let mut gain = 0usize;
                if dominated[cand] == 0 {
                    gain += 1;
                }
                for &u in &adj[cand] {
                    if dominated[u] == 0 {
                        gain += 1;
                    }
                }
                if gain > best_gain {
                    best_gain = gain;
                    best_v = cand;
                }
            }
            // random accept sometimes
            if candidates.len() > 1 && rng.next_f64() < 0.10 {
                best_v = candidates[rng.next_usize(candidates.len())];
            }
            add_vertex(
                best_v,
                &mut in_set,
                &mut dominated,
                &mut dominated_by,
                &mut ds_list,
                &mut pos,
            );
            added += 1;
        }

        // If not fully dominating, revert.
        let mut ok = true;
        for v in 0..n {
            if dominated[v] == 0 {
                ok = false;
                break;
            }
        }
        if !ok {
            in_set = in_set_before;
            dominated = dominated_before;
            dominated_by = dominated_by_before;
            ds_list = ds_list_before;
            pos = pos_before;
            continue;
        }

        // Try extra redundant removals around recently changed area (simple global sweep, limited).
        // With probability to save time.
        if rng.next_f64() < 0.5 {
            // shuffle few candidates from current ds_list
            let samples = 32usize.min(ds_list.len());
            for _ in 0..samples {
                if start.elapsed() >= time_limit {
                    break;
                }
                let cand = ds_list[rng.next_usize(ds_list.len())];
                if fixed[cand] == 1 {
                    continue;
                }
                if in_set[cand] == 1
                    && is_dominated_if_removed(cand, &in_set, adj, &dominated, &dominated_by)
                {
                    remove_vertex(
                        cand,
                        &mut in_set,
                        &mut dominated,
                        &mut dominated_by,
                        &mut ds_list,
                        &mut pos,
                    );
                }
            }
        }

        let cur_size = in_set.iter().map(|&x| x as usize).sum::<usize>();
        if cur_size < best_size {
            best_size = cur_size;
            best = in_set.clone();
        } else {
            // If not improved, revert with some probability (simulated annealing-lite)
            let p_keep = 0.05;
            if rng.next_f64() > p_keep {
                in_set = in_set_before;
                dominated = dominated_before;
                dominated_by = dominated_by_before;
                ds_list = ds_list_before;
                pos = pos_before;
            }
        }
    }

    best
}

fn verify_dominating(n: usize, adj: &Vec<Vec<usize>>, in_set: &Vec<u8>) -> bool {
    let mut dom = vec![0u8; n];
    for v in 0..n {
        if in_set[v] == 1 {
            dom[v] = 1;
            for &u in &adj[v] {
                dom[u] = 1;
            }
        }
    }
    dom.into_iter().all(|x| x == 1)
}

fn main() {
    // Read all input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }
    let mut it = input.split_whitespace();
    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };
    let m: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let Some(a) = it.next() else { break };
        let Some(b) = it.next() else { break };
        let u: usize = a.parse().unwrap();
        let v: usize = b.parse().unwrap();
        if u == v || u >= n || v >= n {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }
    for v in 0..n {
        adj[v].sort_unstable();
        adj[v].dedup();
    }

    let start = Instant::now();
    let total_limit = Duration::from_secs(295); // keep a bit under 5 min

    // Build initial greedy solution (with degree-biased order)
    let mut vertices = (0..n).collect::<Vec<_>>();
    vertices.sort_by(|&a, &b| {
        let da = adj[a].len();
        let db = adj[b].len();
        match db.cmp(&da) {
            Ordering::Equal => a.cmp(&b),
            other => other,
        }
    });
    let mut best = build_greedy_solution(n, &adj, Some(&vertices));
    let mut best_size = best.iter().map(|&x| x as usize).sum::<usize>();

    // Multi-start: randomized orders + local search.
    // Allocate most time to local search; do a few greedy restarts.
    let mut rng = XorShift64::new(
        (n as u64).wrapping_mul(0x9e3779b97f4a7c15)
            ^ (m as u64).wrapping_mul(0xbf58476d1ce4e5b9)
            ^ 0x94d049bb133111eb,
    );

    let restarts = if n <= 2000 { 6 } else if n <= 20000 { 4 } else { 2 };
    for r in 0..restarts {
        if start.elapsed() >= total_limit {
            break;
        }
        // Create a randomized permutation with bias: start from degree-sorted then do random swaps.
        let mut ord = vertices.clone();
        let swaps = (n / 10).max(20).min(20000);
        for _ in 0..swaps {
            let i = rng.next_usize(n);
            let j = rng.next_usize(n);
            ord.swap(i, j);
        }
        let sol = build_greedy_solution(n, &adj, Some(&ord));
        let size = sol.iter().map(|&x| x as usize).sum::<usize>();
        if size < best_size {
            best_size = size;
            best = sol;
        }
        // quick local search from this start (increasing budget)
        let remaining = total_limit.saturating_sub(start.elapsed());
        if remaining <= Duration::from_millis(50) {
            break;
        }
        let slice = match r {
            0 => Duration::from_secs(20),
            1 => Duration::from_secs(30),
            _ => Duration::from_secs(40),
        };
        let t = if slice < remaining { slice } else { remaining };
        let improved = local_search(n, &adj, best.clone(), t, rng.next_u64());
        let size2 = improved.iter().map(|&x| x as usize).sum::<usize>();
        if size2 < best_size && verify_dominating(n, &adj, &improved) {
            best_size = size2;
            best = improved;
        }
    }

    // Final local search using remaining time from best.
    let remaining = total_limit.saturating_sub(start.elapsed());
    if remaining > Duration::from_secs(1) {
        let improved = local_search(n, &adj, best.clone(), remaining, rng.next_u64());
        let size2 = improved.iter().map(|&x| x as usize).sum::<usize>();
        if size2 < best_size && verify_dominating(n, &adj, &improved) {
            best_size = size2;
            best = improved;
        }
    }

    // Safety: if somehow invalid, fall back to greedy.
    if !verify_dominating(n, &adj, &best) {
        best = build_greedy_solution(n, &adj, Some(&vertices));
        best_size = best.iter().map(|&x| x as usize).sum::<usize>();
        if !verify_dominating(n, &adj, &best) {
            // ultimate fallback: all vertices
            best = vec![1u8; n];
            best_size = n;
        }
    }

    let mut out = String::new();
    out.push_str(&format!("{}\n", best_size));
    let mut first = true;
    for v in 0..n {
        if best[v] == 1 {
            if !first {
                out.push(' ');
            }
            first = false;
            out.push_str(&v.to_string());
        }
    }
    out.push('\n');
    print!("{out}");
}
