#include <bits/stdc++.h>
using namespace std;

// Job-Shop Scheduling (JSSP) constructive solver with adaptive strategies.
//
// Produces a feasible schedule for any input size.
// - Small instances: simple list scheduling with earliest-start insertion, job-order priority.
// - Large instances: fast list scheduling with O(1) machine availability (no insertion), plus a few
//   lightweight improvement passes when feasible.
//
// Output format:
// For each job: "start0 machine0 start1 machine1 ...", where tasks appear in job order.

struct TaskSpec {
  int machine;
  int dur;
};

struct MachineInterval {
  int64_t s, e;
};

static inline int64_t i64max(int64_t a, int64_t b) { return a > b ? a : b; }

// Insert-interval scheduling on a single machine: maintain sorted, non-overlapping intervals,
// and find earliest gap >= release time with length dur. O(k) where k = #intervals on machine.
struct MachineCalendar {
  vector<MachineInterval> iv;

  int64_t place_earliest(int64_t release, int dur) {
    int64_t t = release;
    // Find first interval that could conflict.
    // Linear scan is fine for tiny instances; for larger, we won't use calendar insertion.
    for (size_t i = 0; i < iv.size(); ++i) {
      if (t + dur <= iv[i].s) break;          // fits before interval i
      if (t >= iv[i].e) continue;             // after interval i
      t = iv[i].e;                            // push after interval i
    }
    // Insert [t, t+dur) in sorted position.
    MachineInterval ni{t, t + dur};
    auto it = lower_bound(iv.begin(), iv.end(), ni.s,
                          [](const MachineInterval& a, int64_t s) { return a.s < s; });
    iv.insert(it, ni);
    return t;
  }
};

struct Solver {
  int J = 0, M = 0;
  vector<vector<TaskSpec>> jobs;
  vector<vector<int64_t>> start; // [j][t]

  // Heuristic parameters (adaptive).
  bool use_calendar = false;
  int improve_passes = 0;

  void read() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> J >> M)) return;
    jobs.assign(J, {});
    start.assign(J, {});
    for (int j = 0; j < J; ++j) {
      int T;
      cin >> T;
      jobs[j].resize(T);
      start[j].assign(T, 0);
      for (int t = 0; t < T; ++t) {
        int m, d;
        cin >> m >> d;
        jobs[j][t] = {m, d};
      }
    }
    decide_strategy();
  }

  void decide_strategy() {
    int64_t totalTasks = 0;
    int maxT = 0;
    for (int j = 0; j < J; ++j) {
      totalTasks += (int)jobs[j].size();
      maxT = max<int>(maxT, (int)jobs[j].size());
    }

    // "Calendar insertion" has higher overhead but can reduce idle time on small instances.
    // For extreme sizes it is too slow.
    use_calendar = (totalTasks <= 5000 && M <= 50 && maxT <= 500);

    // Improvement passes: only when moderate size.
    if (totalTasks <= 20000) improve_passes = 2;
    else if (totalTasks <= 80000) improve_passes = 1;
    else improve_passes = 0;
  }

  // Build an initial priority order of jobs. Use simple static features:
  // - total processing time (descending)
  // - then longest chain length (descending)
  vector<int> initial_job_order() {
    vector<int64_t> sumP(J, 0), maxP(J, 0);
    for (int j = 0; j < J; ++j) {
      for (auto &ts : jobs[j]) {
        sumP[j] += ts.dur;
        maxP[j] = max<int64_t>(maxP[j], ts.dur);
      }
    }
    vector<int> ord(J);
    iota(ord.begin(), ord.end(), 0);
    stable_sort(ord.begin(), ord.end(), [&](int a, int b) {
      if (sumP[a] != sumP[b]) return sumP[a] > sumP[b];
      if (maxP[a] != maxP[b]) return maxP[a] > maxP[b];
      return a < b;
    });
    return ord;
  }

  // Core list scheduling:
  // For each job in some order, schedule its operations sequentially, on each machine at earliest feasible time.
  // Two variants:
  // - Calendar variant: allows insertion into machine gaps.
  // - Fast variant: uses only machine "ready time" (no gaps), O(totalTasks).
  void construct_schedule(const vector<int>& jobOrder, bool calendar) {
    // reset start times
    for (int j = 0; j < J; ++j) fill(start[j].begin(), start[j].end(), 0);

    if (calendar) {
      vector<MachineCalendar> cal(M);
      for (int jid : jobOrder) {
        int64_t jobReady = 0;
        for (int t = 0; t < (int)jobs[jid].size(); ++t) {
          int m = jobs[jid][t].machine;
          int d = jobs[jid][t].dur;
          int64_t st = cal[m].place_earliest(jobReady, d);
          start[jid][t] = st;
          jobReady = st + d;
        }
      }
    } else {
      vector<int64_t> machReady(M, 0);
      for (int jid : jobOrder) {
        int64_t jobReady = 0;
        for (int t = 0; t < (int)jobs[jid].size(); ++t) {
          int m = jobs[jid][t].machine;
          int d = jobs[jid][t].dur;
          int64_t st = i64max(jobReady, machReady[m]);
          start[jid][t] = st;
          int64_t en = st + d;
          jobReady = en;
          machReady[m] = en;
        }
      }
    }
  }

  int64_t makespan() const {
    int64_t ms = 0;
    for (int j = 0; j < J; ++j) {
      int64_t t = 0;
      for (int k = 0; k < (int)jobs[j].size(); ++k) {
        t = start[j][k] + jobs[j][k].dur;
      }
      ms = max(ms, t);
    }
    return ms;
  }

  // Validate feasibility (for safety). O(totalTasks log totalTasks) due to machine event sorting.
  bool validate() const {
    // job precedence
    for (int j = 0; j < J; ++j) {
      for (int t = 0; t + 1 < (int)jobs[j].size(); ++t) {
        int64_t endt = start[j][t] + jobs[j][t].dur;
        if (start[j][t + 1] < endt) return false;
        if (start[j][t] < 0) return false;
      }
      if (!jobs[j].empty() && start[j][0] < 0) return false;
    }
    // machine non-overlap
    vector<vector<pair<int64_t,int64_t>>> onM(M);
    for (int j = 0; j < J; ++j) {
      for (int t = 0; t < (int)jobs[j].size(); ++t) {
        int m = jobs[j][t].machine;
        int64_t s = start[j][t];
        int64_t e = s + jobs[j][t].dur;
        onM[m].push_back({s, e});
      }
    }
    for (int m = 0; m < M; ++m) {
      auto v = onM[m];
      sort(v.begin(), v.end());
      for (size_t i = 1; i < v.size(); ++i) {
        if (v[i].first < v[i - 1].second) return false;
      }
    }
    return true;
  }

  // Lightweight improvement:
  // Try a few alternative job orders by sorting jobs on different static keys and pick best.
  // This keeps runtime low even for large instances.
  vector<int> try_improve_orders(const vector<int>& base) {
    vector<int> best = base;
    int64_t bestMs;
    construct_schedule(best, use_calendar);
    bestMs = makespan();

    // Precompute features
    vector<int64_t> sumP(J, 0), maxP(J, 0), firstDur(J, 0);
    for (int j = 0; j < J; ++j) {
      if (!jobs[j].empty()) firstDur[j] = jobs[j][0].dur;
      for (auto &ts : jobs[j]) {
        sumP[j] += ts.dur;
        maxP[j] = max<int64_t>(maxP[j], ts.dur);
      }
    }

    auto eval = [&](vector<int> ord) {
      construct_schedule(ord, use_calendar);
      int64_t ms = makespan();
      if (ms < bestMs) {
        bestMs = ms;
        best = std::move(ord);
      }
    };

    // Variant 1: shortest processing time first (SPT)
    {
      vector<int> ord(J);
      iota(ord.begin(), ord.end(), 0);
      stable_sort(ord.begin(), ord.end(), [&](int a, int b) {
        if (sumP[a] != sumP[b]) return sumP[a] < sumP[b];
        return a < b;
      });
      eval(std::move(ord));
    }
    // Variant 2: largest first op first (to quickly occupy bottlenecks)
    {
      vector<int> ord(J);
      iota(ord.begin(), ord.end(), 0);
      stable_sort(ord.begin(), ord.end(), [&](int a, int b) {
        if (firstDur[a] != firstDur[b]) return firstDur[a] > firstDur[b];
        if (sumP[a] != sumP[b]) return sumP[a] > sumP[b];
        return a < b;
      });
      eval(std::move(ord));
    }
    // Variant 3: max op duration first
    {
      vector<int> ord(J);
      iota(ord.begin(), ord.end(), 0);
      stable_sort(ord.begin(), ord.end(), [&](int a, int b) {
        if (maxP[a] != maxP[b]) return maxP[a] > maxP[b];
        if (sumP[a] != sumP[b]) return sumP[a] > sumP[b];
        return a < b;
      });
      eval(std::move(ord));
    }

    // Keep the best schedule already constructed by last eval? Reconstruct later in solve().
    return best;
  }

  void solve() {
    if (J == 0) return;

    // Base order: LPT-like (descending total)
    vector<int> ord = initial_job_order();

    if (improve_passes > 0) {
      // Improvement here is just choosing better static order.
      for (int p = 0; p < improve_passes; ++p) {
        ord = try_improve_orders(ord);
      }
    }

    construct_schedule(ord, use_calendar);

    // Safety: if calendar mode produced an invalid schedule due to a bug, fall back to fast variant.
    if (!validate()) {
      use_calendar = false;
      construct_schedule(ord, false);
      // If still invalid (shouldn't happen), enforce a trivial always-feasible serial schedule.
      if (!validate()) {
        vector<int64_t> machReady(M, 0);
        for (int j = 0; j < J; ++j) {
          int64_t jobReady = 0;
          for (int t = 0; t < (int)jobs[j].size(); ++t) {
            int m = jobs[j][t].machine;
            int d = jobs[j][t].dur;
            int64_t st = max(jobReady, machReady[m]);
            start[j][t] = st;
            int64_t en = st + d;
            jobReady = en;
            machReady[m] = en;
          }
        }
      }
    }
  }

  void write() const {
    // For each job, print T pairs: start_time machine.
    // Use 64-bit times in case of large durations.
    for (int j = 0; j < J; ++j) {
      for (int t = 0; t < (int)jobs[j].size(); ++t) {
        if (t) cout << ' ';
        cout << start[j][t] << ' ' << jobs[j][t].machine;
      }
      cout << '\n';
    }
  }
};

int main() {
  Solver s;
  s.read();
  s.solve();
  s.write();
  return 0;
}
