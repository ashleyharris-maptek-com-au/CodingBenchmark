const fs = require('fs');

// Read input
const input = fs.readFileSync(0, 'utf8').trim();
const data = input ? JSON.parse(input) : {};

const width = data.width | 0;
const height = data.height | 0;
const gridLines = Array.isArray(data.grid) ? data.grid : [];
const ignitions = Array.isArray(data.ignitions) ? data.ignitions : [];
const towns = Array.isArray(data.towns) ? data.towns : [];
let cutBudget = data.cutBudget | 0;
if (!Number.isFinite(cutBudget) || cutBudget < 0) cutBudget = 0;

// Edge cases
if (width <= 0 || height <= 0 || gridLines.length !== height) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

// Helpers for indexing
const N = width * height;
function idx(x, y) { return y * width + x; }
function inBounds(x, y) { return x >= 0 && y >= 0 && x < width && y < height; }

// Parse base grid
// burnable: '.'; blocked: '~' (or anything else treated blocked)
const burnable = new Uint8Array(N);
for (let y = 0; y < height; y++) {
  const row = gridLines[y] || '';
  for (let x = 0; x < width; x++) {
    burnable[idx(x, y)] = row.charCodeAt(x) === 46 ? 1 : 0; // '.'
  }
}

// Multi-source BFS from ignitions to compute baseline fire arrival time
const INF = 0x3fffffff;
const time0 = new Int32Array(N);
for (let i = 0; i < N; i++) time0[i] = INF;

// Queue for BFS (ring buffer)
const q = new Int32Array(N);
let qh = 0, qt = 0;
for (let k = 0; k < ignitions.length; k++) {
  const p = ignitions[k];
  if (!p || p.length < 2) continue;
  const x = p[0] | 0, y = p[1] | 0;
  if (!inBounds(x, y)) continue;
  const id = idx(x, y);
  if (!burnable[id]) continue;
  if (time0[id] !== 0) {
    time0[id] = 0;
    q[qt++] = id;
  }
}

const dx = [1, -1, 0, 0];
const dy = [0, 0, 1, -1];

while (qh < qt) {
  const v = q[qh++];
  const t = time0[v] + 1;
  const x = v % width;
  const y = (v / width) | 0;
  for (let d = 0; d < 4; d++) {
    const nx = x + dx[d], ny = y + dy[d];
    if (!inBounds(nx, ny)) continue;
    const ni = idx(nx, ny);
    if (!burnable[ni]) continue;
    if (t < time0[ni]) {
      time0[ni] = t;
      q[qt++] = ni;
    }
  }
}

// Compute baseline burned town value and organize towns by their cell
let baselineBurnedValue = 0;
// Map from cell index to total town value at that cell (could be multiple)
const townValueAt = new Map();
for (let i = 0; i < towns.length; i++) {
  const tw = towns[i];
  if (!tw) continue;
  const x = tw.x | 0, y = tw.y | 0;
  const value = +tw.value;
  if (!Number.isFinite(value) || value <= 0) continue;
  if (!inBounds(x, y)) continue;
  const id = idx(x, y);
  const prev = townValueAt.get(id) || 0;
  townValueAt.set(id, prev + value);
  if (time0[id] !== INF) baselineBurnedValue += value;
}

if (cutBudget === 0 || baselineBurnedValue <= 0 || qt === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

// We'll choose cuts greedily via repeated "barrier BFS":
// pick a target cell (candidate to cut) that disconnects high-value burned area.
// Approach:
// 1) Keep a set of cut cells (blocked additionally).
// 2) Recompute reachability (not full times) from ignitions under current cuts.
// 3) Identify boundary between reachable and unreachable among burned region, and score cut candidates
//    on the reachable side that have many unreachable neighbors leading to town value.
//
// This aims to create a firebreak "front" that saves value behind it.

const cut = new Uint8Array(N); // 1 if cut

// Work arrays reused per iteration
const reachable = new Uint8Array(N);
const parent = new Int32Array(N); // BFS tree parent (for path to boundary cell)
const dist = new Int32Array(N);
const bfsQueue = new Int32Array(N);

function computeReachable() {
  reachable.fill(0);
  for (let i = 0; i < N; i++) { parent[i] = -1; dist[i] = INF; }
  let h = 0, t = 0;
  // start from ignitions
  for (let k = 0; k < ignitions.length; k++) {
    const p = ignitions[k];
    if (!p || p.length < 2) continue;
    const x = p[0] | 0, y = p[1] | 0;
    if (!inBounds(x, y)) continue;
    const id = idx(x, y);
    if (!burnable[id] || cut[id]) continue;
    if (!reachable[id]) {
      reachable[id] = 1;
      dist[id] = 0;
      bfsQueue[t++] = id;
    }
  }
  while (h < t) {
    const v = bfsQueue[h++];
    const x = v % width;
    const y = (v / width) | 0;
    const nd = dist[v] + 1;
    for (let d = 0; d < 4; d++) {
      const nx = x + dx[d], ny = y + dy[d];
      if (!inBounds(nx, ny)) continue;
      const ni = idx(nx, ny);
      if (!burnable[ni] || cut[ni] || reachable[ni]) continue;
      reachable[ni] = 1;
      parent[ni] = v;
      dist[ni] = nd;
      bfsQueue[t++] = ni;
    }
  }
  return t; // visited count
}

// Compute "saved value behind cell" by flooding the unreachable area adjacent to it.
// We approximate by computing connected components of currently unreachable cells and their town value.
// Then each candidate cut on reachable side bordering an unreachable component gets that component value as potential.

const compId = new Int32Array(N);
const compValue = []; // value per component
const compSize = [];
const compQueue = new Int32Array(N);

function computeUnreachableComponents() {
  for (let i = 0; i < N; i++) compId[i] = -1;
  compValue.length = 0;
  compSize.length = 0;
  let compCount = 0;

  for (let start = 0; start < N; start++) {
    if (!burnable[start] || cut[start]) continue;
    if (reachable[start]) continue;
    if (compId[start] !== -1) continue;

    let h = 0, t = 0;
    compQueue[t++] = start;
    compId[start] = compCount;
    let valueSum = townValueAt.get(start) || 0;
    let size = 0;
    while (h < t) {
      const v = compQueue[h++];
      size++;
      const x = v % width;
      const y = (v / width) | 0;
      for (let d = 0; d < 4; d++) {
        const nx = x + dx[d], ny = y + dy[d];
        if (!inBounds(nx, ny)) continue;
        const ni = idx(nx, ny);
        if (!burnable[ni] || cut[ni] || reachable[ni]) continue;
        if (compId[ni] !== -1) continue;
        compId[ni] = compCount;
        compQueue[t++] = ni;
        const tv = townValueAt.get(ni);
        if (tv) valueSum += tv;
      }
    }
    compValue.push(valueSum);
    compSize.push(size);
    compCount++;
  }
  return compCount;
}

// Candidate scoring:
// For each reachable cell (burnable, not cut), look at 4-neighbors that are unreachable burnable (not cut).
// Sum unique component values adjacent; prefer smaller distance from ignition and smaller cuts "leak".
// Also consider that cutting a cell already reachable might block multiple paths, but this heuristic is local.

function pickBestCutCell() {
  let bestCell = -1;
  let bestScore = 0;

  // We'll scan only cells that are reachable AND not already cut.
  // To reduce time on huge grids, sample by scanning a subset when needed.
  // If N is big, scanning all each iteration may be too slow. We'll do:
  // - full scan for N <= 200k
  // - otherwise, scan a stride sample increasing coverage over iterations.
  const fullScan = N <= 200000;
  const stride = fullScan ? 1 : 7;
  const offset = (Math.random() * stride) | 0; // pseudo-random-ish without RNG seed

  // Temp small set for up to 4 comps; avoid allocating Set
  const seenA = -1, seenB = -1, seenC = -1, seenD = -1;

  for (let i = offset; i < N; i += stride) {
    if (!reachable[i] || cut[i] || !burnable[i]) continue;
    // Optional: don't cut on a town cell unless necessary
    if (townValueAt.has(i)) continue;

    const x = i % width;
    const y = (i / width) | 0;
    let s = 0;
    // unique adjacent comps
    let c0 = -2, c1 = -2, c2 = -2, c3 = -2;
    for (let d = 0; d < 4; d++) {
      const nx = x + dx[d], ny = y + dy[d];
      if (!inBounds(nx, ny)) continue;
      const ni = idx(nx, ny);
      if (!burnable[ni] || cut[ni] || reachable[ni]) continue;
      const cid = compId[ni];
      if (cid < 0) continue;
      if (cid === c0 || cid === c1 || cid === c2 || cid === c3) continue;
      if (c0 === -2) c0 = cid;
      else if (c1 === -2) c1 = cid;
      else if (c2 === -2) c2 = cid;
      else c3 = cid;
      s += compValue[cid] || 0;
    }
    if (s <= 0) continue;

    // Penalize cutting very far from ignition (less likely to help early) and high-degree cell.
    // Prefer cuts that are closer (lower dist) so the fire is stopped earlier.
    const distPenalty = 1 + (dist[i] <= 0 ? 0 : dist[i] * 0.15);
    // Cutting a cell with many reachable neighbors may be bypassed; but also may be chokepoint.
    let reachNei = 0;
    for (let d = 0; d < 4; d++) {
      const nx = x + dx[d], ny = y + dy[d];
      if (!inBounds(nx, ny)) continue;
      const ni = idx(nx, ny);
      if (burnable[ni] && !cut[ni] && reachable[ni]) reachNei++;
    }
    const bypassPenalty = 1 + (reachNei - 1) * 0.1;

    const score = s / (distPenalty * bypassPenalty);
    if (score > bestScore) {
      bestScore = score;
      bestCell = i;
    }
  }
  return bestCell;
}

// If the heuristic can't find a useful local cut, fall back to cutting a frontier cell near the highest-value burned town.
function fallbackCutNearHighValueTown() {
  let bestTown = -1;
  let bestVal = -1;
  for (const [cell, val] of townValueAt.entries()) {
    if (val <= 0) continue;
    if (!reachable[cell]) continue; // currently would burn
    if (val > bestVal) { bestVal = val; bestTown = cell; }
  }
  if (bestTown < 0) return -1;

  // Walk back in BFS tree to find a cell to cut (avoid cutting the town itself)
  let v = bestTown;
  for (let steps = 0; steps < 3 && parent[v] !== -1; steps++) v = parent[v];
  // If still at town, move one more
  if (v === bestTown && parent[v] !== -1) v = parent[v];
  if (v < 0 || cut[v] || !reachable[v] || !burnable[v] || townValueAt.has(v)) return -1;
  return v;
}

// Main greedy loop
const firebreaks = [];

// Bound iterations to budget; also cap to avoid worst-case O(budget*N) on huge N
let iter = 0;
const maxIter = cutBudget;
while (iter < maxIter) {
  iter++;
  computeReachable();
  computeUnreachableComponents();

  let cell = pickBestCutCell();
  if (cell < 0) cell = fallbackCutNearHighValueTown();
  if (cell < 0) break;

  cut[cell] = 1;
  firebreaks.push([cell % width, (cell / width) | 0]);
}

process.stdout.write(JSON.stringify({ firebreaks }));
