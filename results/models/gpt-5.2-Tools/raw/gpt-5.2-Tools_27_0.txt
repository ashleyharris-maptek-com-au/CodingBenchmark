// Graph Coloring heuristic solver (exact-k coloring if possible).
// Strategies:
// - Fast greedy for huge graphs.
// - DSATUR-style greedy for medium graphs.
// - Conflict-repair via min-conflicts local search with random restarts when needed.
//
// Notes:
// - For extremely hard instances, no polynomial heuristic can guarantee success.
// - This implementation is engineered to be memory-conscious and scalable to millions of vertices.

struct Graph {
    int n = 0;
    int m = 0;
    std::vector<int> head; // adjacency list in CSR-like forward-star
    std::vector<int> to;
    std::vector<int> next;
    std::vector<int> deg;

    explicit Graph(int n_, int m_) : n(n_), m(m_), head(n_, -1), to(2 * m_), next(2 * m_), deg(n_, 0) {}

    void add_edge(int idx, int u, int v) {
        // idx is edge index in [0, 2*m)
        to[idx] = v;
        next[idx] = head[u];
        head[u] = idx;
        deg[u]++;
    }
};

static inline uint64_t splitmix64(uint64_t &x) {
    x += 0x9e3779b97f4a7c15ull;
    uint64_t z = x;
    z = (z ^ (z >> 30)) * 0xbf58476d1ce4e5b9ull;
    z = (z ^ (z >> 27)) * 0x94d049bb133111ebull;
    return z ^ (z >> 31);
}

struct RNG {
    uint64_t state;
    explicit RNG(uint64_t seed) : state(seed) {}
    uint32_t next_u32() {
        uint64_t x = splitmix64(state);
        return static_cast<uint32_t>(x);
    }
    int next_int(int lo, int hi) {
        // inclusive lo, exclusive hi
        uint32_t r = next_u32();
        return lo + (int)(r % (uint32_t)(hi - lo));
    }
    double next_double() {
        uint64_t x = splitmix64(state);
        // [0,1)
        return (x >> 11) * (1.0 / 9007199254740992.0);
    }
};

static inline bool is_power_of_two(size_t x) { return x && ((x & (x - 1)) == 0); }

static std::vector<int> greedy_largest_first(const Graph &g, int k) {
    int n = g.n;
    std::vector<int> order(n);
    std::iota(order.begin(), order.end(), 0);
    std::stable_sort(order.begin(), order.end(), [&](int a, int b) { return g.deg[a] > g.deg[b]; });

    std::vector<int> color(n, -1);
    std::vector<int> used(k, -1);
    for (int u : order) {
        int stamp = u + 1;
        for (int e = g.head[u]; e != -1; e = g.next[e]) {
            int v = g.to[e];
            int c = color[v];
            if (c >= 0 && c < k) used[c] = stamp;
        }
        int c = 0;
        for (; c < k; ++c) {
            if (used[c] != stamp) break;
        }
        if (c == k) {
            // leave uncolored; repaired later
            color[u] = -1;
        } else {
            color[u] = c;
        }
    }
    return color;
}

// DSATUR-like greedy coloring using k colors (may fail to color some vertices).
static std::vector<int> dsatur_greedy(const Graph &g, int k) {
    int n = g.n;
    std::vector<int> color(n, -1);
    std::vector<int> sat(n, 0);
    std::vector<int> deg = g.deg;

    // For medium n, we can store per-vertex bitsets of seen colors if k<=64; else use hash-stamp list.
    bool smallK = (k <= 64);
    std::vector<uint64_t> seenMask;
    std::vector<int> seenStamp;
    std::vector<std::vector<int>> seenList;
    if (smallK) {
        seenMask.assign(n, 0);
    } else {
        // store seen colors as list + stamp for membership tests
        seenStamp.assign(k, -1);
        seenList.assign(n, {});
        // reserve small-ish; saturation rarely huge
    }

    auto update_sat = [&](int v, int c) {
        if (smallK) {
            uint64_t before = seenMask[v];
            uint64_t bit = 1ull << (uint64_t)c;
            uint64_t after = before | bit;
            if (after != before) {
                seenMask[v] = after;
                sat[v] = (int)__popcnt64(after);
            }
        } else {
            // check membership via stamp array (global for check; O(1) per neighbor update needs per-vertex stamp -> too heavy)
            // Instead, store list and linear check; acceptable for medium sizes and moderate saturation.
            auto &lst = seenList[v];
            for (int x : lst) if (x == c) return;
            lst.push_back(c);
            sat[v] = (int)lst.size();
        }
    };

    // pick start vertex with max degree
    int start = 0;
    for (int i = 1; i < n; ++i) if (deg[i] > deg[start]) start = i;
    color[start] = 0;
    for (int e = g.head[start]; e != -1; e = g.next[e]) update_sat(g.to[e], 0);

    std::vector<int> used(k, -1);
    int colored = 1;
    for (int iter = 1; iter < n; ++iter) {
        // select uncolored vertex with highest saturation, break ties by degree
        int best = -1;
        int bestSat = -1, bestDeg = -1;
        for (int v = 0; v < n; ++v) {
            if (color[v] != -1) continue;
            int s = sat[v];
            int d = deg[v];
            if (s > bestSat || (s == bestSat && d > bestDeg)) {
                best = v;
                bestSat = s;
                bestDeg = d;
            }
        }
        if (best == -1) break;

        int stamp = best + 1;
        for (int e = g.head[best]; e != -1; e = g.next[e]) {
            int nb = g.to[e];
            int c = color[nb];
            if (c >= 0) used[c] = stamp;
        }
        int chosen = -1;
        // choose smallest available color
        for (int c = 0; c < k; ++c) {
            if (used[c] != stamp) {
                chosen = c;
                break;
            }
        }
        if (chosen == -1) {
            // fail to color within k
            color[best] = -1;
        } else {
            color[best] = chosen;
            colored++;
            for (int e = g.head[best]; e != -1; e = g.next[e]) {
                int nb = g.to[e];
                if (color[nb] == -1) update_sat(nb, chosen);
            }
        }
    }
    return color;
}

static int count_conflicts_vertex(const Graph &g, const std::vector<int> &color, int u) {
    int cu = color[u];
    if (cu < 0) return 0;
    int c = 0;
    for (int e = g.head[u]; e != -1; e = g.next[e]) {
        int v = g.to[e];
        if (color[v] == cu) c++;
    }
    return c;
}

// Local search: Min-conflicts with incremental neighbor-color counts.
// Returns true if finds conflict-free coloring. Mutates color.
static bool min_conflicts(const Graph &g, int k, std::vector<int> &color, RNG &rng, int maxSteps) {
    int n = g.n;
    // Build counts[u*k + c] might be too big for large n. Use per-vertex sparse counts by scanning neighbors when needed.
    // Hybrid: for k<=64 and medium graphs, build dense counts; else use on-demand eval.
    bool dense = (k <= 64 && (int64_t)n * (int64_t)k <= (int64_t)80'000'000); // ~80M ints limit
    std::vector<int> conflictCount(n, 0);
    int64_t totalConflicts = 0;

    std::vector<int> counts; // dense counts if enabled
    if (dense) counts.assign((size_t)n * (size_t)k, 0);

    auto rebuild = [&]() {
        std::fill(conflictCount.begin(), conflictCount.end(), 0);
        totalConflicts = 0;
        if (dense) std::fill(counts.begin(), counts.end(), 0);
        for (int u = 0; u < n; ++u) {
            int cu = color[u];
            if (cu < 0) continue;
            for (int e = g.head[u]; e != -1; e = g.next[e]) {
                int v = g.to[e];
                int cv = color[v];
                if (cv < 0) continue;
                if (dense) counts[(size_t)u * (size_t)k + (size_t)cv]++; // neighbor v contributes its color to u's counts
                if (u < v && cu == cv) {
                    conflictCount[u]++;
                    conflictCount[v]++;
                    totalConflicts++;
                }
            }
        }
    };

    auto conflicts_if_color = [&](int u, int c) -> int {
        if (dense) return counts[(size_t)u * (size_t)k + (size_t)c];
        int cnt = 0;
        for (int e = g.head[u]; e != -1; e = g.next[e]) {
            int v = g.to[e];
            if (color[v] == c) cnt++;
        }
        return cnt;
    };

    auto apply_color_change = [&](int u, int newC) {
        int oldC = color[u];
        if (oldC == newC) return;
        // update totalConflicts and per-vertex conflictCount for u and neighbors
        // We update by scanning neighbors; acceptable since we only do for conflicted vertices.
        for (int e = g.head[u]; e != -1; e = g.next[e]) {
            int v = g.to[e];
            int cv = color[v];
            if (cv < 0) continue;
            if (cv == oldC) {
                // was conflict edge, remove
                conflictCount[u]--;
                conflictCount[v]--;
                totalConflicts--;
            }
            if (cv == newC) {
                // becomes conflict edge, add
                conflictCount[u]++;
                conflictCount[v]++;
                totalConflicts++;
            }
            if (dense) {
                // u's contribution changes in v's neighbor-color counts
                counts[(size_t)v * (size_t)k + (size_t)oldC]--;
                counts[(size_t)v * (size_t)k + (size_t)newC]++;
            }
        }
        if (dense) {
            // rebuild u's counts row (since its neighbors unchanged, counts row doesn't depend on u's own color)
            // Actually counts[u,*] depends on neighbors' colors only, so no need.
        }
        color[u] = newC;
    };

    // Initialize any uncolored vertices randomly
    for (int i = 0; i < n; ++i) {
        if (color[i] < 0) color[i] = rng.next_int(0, k);
        else if (color[i] >= k) color[i] = color[i] % k;
    }

    rebuild();
    if (totalConflicts == 0) return true;

    std::vector<int> conflicted;
    conflicted.reserve(1024);
    for (int step = 0; step < maxSteps; ++step) {
        if (totalConflicts == 0) return true;

        // pick a random conflicted vertex (build list periodically or by sampling)
        int u = -1;
        // Try a few random samples to avoid full scan
        for (int tries = 0; tries < 32; ++tries) {
            int cand = rng.next_int(0, n);
            if (conflictCount[cand] > 0) {
                u = cand;
                break;
            }
        }
        if (u == -1) {
            // fallback full scan
            for (int i = 0; i < n; ++i) if (conflictCount[i] > 0) { u = i; break; }
            if (u == -1) return true;
        }

        int bestC = color[u];
        int bestConf = conflicts_if_color(u, bestC);
        // explore all colors when k small; otherwise sample a subset
        if (k <= 64) {
            for (int c = 0; c < k; ++c) {
                int cf = conflicts_if_color(u, c);
                if (cf < bestConf) {
                    bestConf = cf;
                    bestC = c;
                    if (bestConf == 0) break;
                }
            }
        } else {
            // sample colors
            int samples = 32;
            for (int s = 0; s < samples; ++s) {
                int c = rng.next_int(0, k);
                int cf = conflicts_if_color(u, c);
                if (cf < bestConf) {
                    bestConf = cf;
                    bestC = c;
                    if (bestConf == 0) break;
                }
            }
        }

        // random walk occasionally to escape local minima
        if (bestC == color[u] && (step % 17 == 0)) {
            bestC = rng.next_int(0, k);
        }
        apply_color_change(u, bestC);

        if ((step & 1023) == 0) {
            // occasional light check: if conflict counts drift (shouldn't), could rebuild; keep cheap by not doing.
        }
    }
    return totalConflicts == 0;
}

static bool verify_coloring(const Graph &g, const std::vector<int> &color, int k) {
    int n = g.n;
    for (int i = 0; i < n; ++i) {
        if (color[i] < 0 || color[i] >= k) return false;
    }
    for (int u = 0; u < n; ++u) {
        int cu = color[u];
        for (int e = g.head[u]; e != -1; e = g.next[e]) {
            int v = g.to[e];
            if (u < v && cu == color[v]) return false;
        }
    }
    return true;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, m, k;
    if (!(std::cin >> n >> m >> k)) return 0;
    if (n <= 0) return 0;
    if (k <= 0) {
        // impossible; still output something
        for (int i = 0; i < n; ++i) {
            if (i) std::cout << ' ';
            std::cout << 0;
        }
        std::cout << "\n";
        return 0;
    }

    Graph g(n, m);
    for (int i = 0; i < m; ++i) {
        int u, v;
        std::cin >> u >> v;
        if (u < 0 || v < 0 || u >= n || v >= n || u == v) continue;
        int idx = 2 * i;
        g.add_edge(idx, u, v);
        g.add_edge(idx + 1, v, u);
    }

    // Seed RNG with some entropy from sizes.
    uint64_t seed = 0x123456789abcdef0ull ^ (uint64_t)n * 0x9e3779b97f4a7c15ull ^ (uint64_t)m;
    RNG rng(seed);

    // Adaptive strategy selection.
    // Huge graphs: start with largest-first greedy (fast, streaming-friendly), then repair.
    // Medium graphs: dsatur then repair.
    // Small graphs: dsatur + more repair steps and restarts.
    std::vector<int> color;
    bool huge = (n >= 500000);
    bool medium = (n >= 20000);

    if (huge) {
        color = greedy_largest_first(g, k);
    } else {
        color = dsatur_greedy(g, k);
        // If dsatur leaves many uncolored, randomize those
        int uncolored = 0;
        for (int i = 0; i < n; ++i) if (color[i] < 0) uncolored++;
        if (uncolored > n / 50) {
            // fall back to greedy largest-first which tends to color more vertices quickly
            color = greedy_largest_first(g, k);
        }
    }

    // Repair with min-conflicts local search, with random restarts.
    // Step budget scales with edges and vertices but capped to keep runtime bounded.
    int64_t scale = (int64_t)n + (int64_t)m / 2;
    int baseSteps = 0;
    if (huge) baseSteps = 2000000;
    else if (medium) baseSteps = 6000000;
    else baseSteps = 20000000;
    // Adjust a bit with graph size; keep within int.
    int maxSteps = baseSteps;
    if (scale < 5'000'000) maxSteps = std::min<int>(maxSteps, 25'000'000);
    if (scale > 50'000'000) maxSteps = std::max<int>(maxSteps, 1'500'000);

    bool ok = verify_coloring(g, color, k);
    if (!ok) {
        int restarts = huge ? 4 : (medium ? 8 : 24);
        for (int r = 0; r < restarts; ++r) {
            // randomize some vertices to diversify
            for (int i = 0; i < n; ++i) {
                if (r == 0) {
                    if (color[i] < 0) color[i] = rng.next_int(0, k);
                } else {
                    // with small probability randomize
                    if ((rng.next_u32() & 1023u) == 0u) color[i] = rng.next_int(0, k);
                    if (color[i] < 0) color[i] = rng.next_int(0, k);
                }
            }
            int steps = maxSteps / (r + 1);
            if (steps < 250000) steps = 250000;
            if (min_conflicts(g, k, color, rng, steps)) {
                ok = true;
                break;
            }
        }
    }

    // If still not ok, do a final pass: enforce valid range and output best effort.
    // (Benchmark expects valid coloring; on hard cases this may fail, but we avoid UB.)
    for (int i = 0; i < n; ++i) {
        if (color[i] < 0) color[i] = 0;
        if (color[i] >= k) color[i] %= k;
    }

    // Output colors
    for (int i = 0; i < n; ++i) {
        if (i) std::cout << ' ';
        std::cout << color[i];
    }
    std::cout << "\n";
    return 0;
}
