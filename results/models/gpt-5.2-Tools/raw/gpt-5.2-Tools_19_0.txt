using System;
using System.Buffers;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    // Heuristic mining solver:
    // - For huge instances: output 0 operations (safe, fast, non-negative profit).
    // - For small/medium: greedily extract valuable ore (valuePerTonne>0) that is currently exposed,
    //   moving it along a simple Manhattan path to the processing plant (0,0). We do not dump waste.
    // This is intentionally conservative to avoid negative profit from junk ore or blocked dumping.

    private sealed class FastInput
    {
        private readonly Stream _s;
        private readonly byte[] _buf;
        private int _len, _pos;

        public FastInput(Stream s, int bufSize = 1 << 20)
        {
            _s = s;
            _buf = new byte[bufSize];
        }

        private int ReadByteInternal()
        {
            if (_pos >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_pos++];
        }

        public string? ReadToken()
        {
            var sb = new StringBuilder(32);
            int c;
            do
            {
                c = ReadByteInternal();
                if (c < 0) return sb.Length == 0 ? null : sb.ToString();
            } while (c == ' ' || c == '\n' || c == '\r' || c == '\t');
            while (c >= 0 && c != ' ' && c != '\n' && c != '\r' && c != '\t')
            {
                sb.Append((char)c);
                c = ReadByteInternal();
            }
            return sb.ToString();
        }

        public int ReadInt()
        {
            string t = ReadToken();
            if (t == null) throw new EndOfStreamException();
            return int.Parse(t, CultureInfo.InvariantCulture);
        }

        public double ReadDouble()
        {
            string t = ReadToken();
            if (t == null) throw new EndOfStreamException();
            return double.Parse(t, CultureInfo.InvariantCulture);
        }

        public bool TryReadRawByte(out byte b)
        {
            int c = ReadByteInternal();
            if (c < 0)
            {
                b = 0;
                return false;
            }
            b = (byte)c;
            return true;
        }
    }

    private sealed class Op
    {
        public int DigX, DigY, DigZ;
        public List<(int x, int y)> Steps = new List<(int x, int y)>(16);
        public int DestX, DestY;
    }

    public static void Main()
    {
        var input = new FastInput(Console.OpenStandardInput());
        int xCount = input.ReadInt();
        int yCount = input.ReadInt();
        int zCount = input.ReadInt();
        _ = input.ReadDouble(); // XBlockSize unused for heuristic
        _ = input.ReadDouble(); // YBlockSize
        _ = input.ReadDouble(); // ZBlockSize

        // Decide strategy based on scale / memory feasibility.
        // Storing full grid for 5000^3 is impossible, so we must short-circuit.
        // We use a conservative threshold based on voxel count.
        long total = (long)xCount * yCount * zCount;
        const long MaxStoreVoxels = 60_000_000; // ~60M bytes raw + overhead; keeps memory reasonable.

            if (total <= 0 || total > MaxStoreVoxels)
            {
            // Consume input bytes quickly so judge doesn't hang if it expects full read.
            // But for extremely large, input itself is enormous; streaming consume is still correct.
            // We just read and discard.
            byte tmp;
            while (input.TryReadRawByte(out tmp)) { }
            Console.WriteLine("0");
            return;
        }

        // Read full world into a flat byte array (raw).
        // Indexing order matches input loops: x -> y -> z.
        int yz = yCount * zCount;
            byte[] raw = ArrayPool<byte>.Shared.Rent((int)total);
        try
        {
            int i = 0;
            for (; i < (int)total; i++)
            {
                if (!input.TryReadRawByte(out raw[i]))
                {
                    raw[i] = 0;
                }
            }

            // Build "topZ" per (x,y): highest z with any voxel (mass>0 or value>0 counts as material).
            // Also, we treat mass==0 as empty (even if value>0 doesn't exist in encoding).
            int xy = xCount * yCount;
            int[] topZ = new int[xy];
            Array.Fill(topZ, -1);

            for (int x = 0; x < xCount; x++)
            {
                int xBase = x * yz;
                for (int y = 0; y < yCount; y++)
                {
                    int baseIdx = xBase + y * zCount;
                    int top = -1;
                    for (int z = zCount - 1; z >= 0; z--)
                    {
                        byte d = raw[baseIdx + z];
                        byte mass = (byte)(d & 0x0F);
                        if (mass != 0)
                        {
                            top = z;
                            break;
                        }
                    }
                    topZ[x * yCount + y] = top;
                }
            }

            // Collect exposed valuable voxels.
            // Exposed if at top of its column OR on a side where neighbor column top is below its z.
            // We'll attempt to extract highest-value first, then nearest to (0,0).
            var candidates = new List<(int x, int y, int z, byte mass, byte value)>();
            for (int x = 0; x < xCount; x++)
            {
                int xBase = x * yz;
                for (int y = 0; y < yCount; y++)
                {
                    int baseIdx = xBase + y * zCount;
                    int top = topZ[x * yCount + y];
                    if (top < 0) continue;

                    // Scan down a bit from the top to find valuable exposed layers.
                    // Limiting scan improves speed and avoids digging deep (which tends to be non-exposed).
                    int scanDepth = Math.Min(32, top + 1);
                    for (int dz = 0; dz < scanDepth; dz++)
                    {
                        int z = top - dz;
                        byte d = raw[baseIdx + z];
                        byte mass = (byte)(d & 0x0F);
                        if (mass == 0) continue;
                        byte value = (byte)(d >> 4);
                        if (value == 0) continue; // junk ore/dirt not worth taking to plant in this heuristic

                        if (IsExposed(x, y, z, xCount, yCount, topZ))
                            candidates.Add((x, y, z, mass, value));
                    }
                }
            }

            candidates.Sort((a, b) =>
            {
                int c = b.value.CompareTo(a.value);
                if (c != 0) return c;
                long da = (long)a.x * a.x + (long)a.y * a.y;
                long db = (long)b.x * b.x + (long)b.y * b.y;
                c = da.CompareTo(db);
                if (c != 0) return c;
                return b.z.CompareTo(a.z);
            });

            // Limit operations to keep output reasonable and runtime safe.
            int opLimit = Math.Min(5000, candidates.Count);

            var ops = new List<Op>(opLimit);

            // Simulate removing extracted voxels by updating topZ and raw at (x,y,z)=0.
            // We don't simulate collapses; rules say collapse occurs when disconnected, but our
            // extraction is from exposed surfaces near the top, so it's unlikely to create floating
            // islands. Conservative.
            for (int idx = 0; idx < candidates.Count && ops.Count < opLimit; idx++)
            {
                var c = candidates[idx];
                // Ensure still present and still exposed (may have changed due to earlier digs).
                int col = c.x * yCount + c.y;
                int baseIdx = c.x * yz + c.y * zCount;
                byte dcur = raw[baseIdx + c.z];
                if ((dcur & 0x0F) == 0) continue;
                byte vcur = (byte)(dcur >> 4);
                if (vcur == 0) continue;
                if (!IsExposed(c.x, c.y, c.z, xCount, yCount, topZ)) continue;

                // Plan a Manhattan path in XY to (0,0) with slope constraint relative to column tops.
                // We'll move along X then Y, checking slope between consecutive columns (top heights).
                // Downhill is fine; uphill limited to +1 per step.
                if (!TryPlanPathToPlant(c.x, c.y, topZ, xCount, yCount, out var steps))
                    continue;

                // Record operation: N digX digY digZ [steps...] destX destY
                var op = new Op
                {
                    DigX = c.x,
                    DigY = c.y,
                    DigZ = c.z,
                    DestX = 0,
                    DestY = 0
                };
                op.Steps.AddRange(steps);
                ops.Add(op);

                // Remove voxel (delivered to plant => disappears).
                raw[baseIdx + c.z] = 0;
                // Update topZ if we removed at/near top.
                if (topZ[col] == c.z)
                {
                    int newTop = -1;
                    for (int z = c.z - 1; z >= 0; z--)
                    {
                        byte dd = raw[baseIdx + z];
                        if ((dd & 0x0F) != 0)
                        {
                            newTop = z;
                            break;
                        }
                    }
                    topZ[col] = newTop;
                }
            }

            // Emit output
            var sbOut = new StringBuilder(ops.Count * 64);
            sbOut.Append(ops.Count).Append('\n');
            for (int iop = 0; iop < ops.Count; iop++)
            {
                var op = ops[iop];
                sbOut.Append(op.Steps.Count)
                    .Append(' ').Append(op.DigX)
                    .Append(' ').Append(op.DigY)
                    .Append(' ').Append(op.DigZ);
                for (int iStep = 0; iStep < op.Steps.Count; iStep++)
                {
                    sbOut.Append(' ').Append(op.Steps[iStep].x)
                         .Append(' ').Append(op.Steps[iStep].y);
                }
                sbOut.Append(' ').Append(op.DestX).Append(' ').Append(op.DestY).Append('\n');
            }
            Console.Write(sbOut.ToString());
        }
        finally
        {
            ArrayPool<byte>.Shared.Return(raw);
        }
    }

    private static bool IsExposed(int x, int y, int z, int xCount, int yCount, int[] topZ)
    {
        int hereTop = topZ[x * yCount + y];
        if (hereTop == z) return true; // exposed from above

        // exposed from a side if neighbor column's top is below z
        if (x > 0 && topZ[(x - 1) * yCount + y] < z) return true;
        if (x + 1 < xCount && topZ[(x + 1) * yCount + y] < z) return true;
        if (y > 0 && topZ[x * yCount + (y - 1)] < z) return true;
        if (y + 1 < yCount && topZ[x * yCount + (y + 1)] < z) return true;
        return false;
    }

    private static bool TryPlanPathToPlant(int startX, int startY, int[] topZ, int xCount, int yCount, out List<(int x, int y)> steps)
    {
        // Simple greedy path: at each step, reduce |x| then |y| (toward 0), only if slope constraint holds.
        // If blocked, try alternate axis. If still blocked, fail.
        steps = new List<(int x, int y)>(startX + startY + 4);
        int x = startX, y = startY;
        while (x != 0 || y != 0)
        {
            bool moved = false;
            if (x != 0)
            {
                int nx = x + (x > 0 ? -1 : 1);
                if (IsStepSlopeOk(x, y, nx, y, topZ, yCount))
                {
                    x = nx;
                    steps.Add((x, y));
                    moved = true;
                }
            }
            if (!moved && y != 0)
            {
                int ny = y + (y > 0 ? -1 : 1);
                if (IsStepSlopeOk(x, y, x, ny, topZ, yCount))
                {
                    y = ny;
                    steps.Add((x, y));
                    moved = true;
                }
            }
            if (!moved && x != 0)
            {
                int nx = x + (x > 0 ? -1 : 1);
                if (IsStepSlopeOk(x, y, nx, y, topZ, yCount))
                {
                    x = nx;
                    steps.Add((x, y));
                    moved = true;
                }
            }
            if (!moved) return false;
            if (x < 0 || x >= xCount || y < 0 || y >= yCount) return false;
            if (steps.Count > (xCount + yCount) * 2) return false;
        }
        return true;
    }

    private static bool IsStepSlopeOk(int x1, int y1, int x2, int y2, int[] topZ, int yCount)
    {
        int h1 = topZ[x1 * yCount + y1];
        int h2 = topZ[x2 * yCount + y2];
        // empty columns are treated as -1; moving into emptiness is downhill/free and ok.
        int dh = h2 - h1;
        return dh <= 1;
    }
}
