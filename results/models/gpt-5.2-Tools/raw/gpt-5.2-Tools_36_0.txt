#include <bits/stdc++.h>
using namespace std;

/*
Exact Maximum Independent Set solver.

Strategy:
- For n <= 64: branch-and-bound on 64-bit masks with greedy coloring upper bound (Tomita-style).
- For n > 64: iterative peeling for deg<=1, split remaining core into connected components,
  and solve each component exactly with branch-and-bound using dynamic bitset adjacency
  and greedy coloring bound.

Still exponential in the worst case (MIS is NP-hard), but exact and fast on many instances.
*/

static inline uint64_t popcnt64(uint64_t x) { return (uint64_t)__popcnt64(x); }

// ---------- Small n <= 64 solver ----------
struct MIS64 {
  int n;
  vector<uint64_t> adj;
  vector<int> best;
  int bestSize = 0;

  int colorBound(vector<int>& verts, vector<int>& colors, uint64_t candMask) {
    verts.clear();
    colors.clear();
    uint64_t cm = candMask;
    while (cm) {
      int v = __builtin_ctzll(cm);
      verts.push_back(v);
      cm &= cm - 1;
    }
    sort(verts.begin(), verts.end(), [&](int a, int b) {
      uint64_t da = popcnt64(adj[a] & candMask);
      uint64_t db = popcnt64(adj[b] & candMask);
      return da > db;
    });

    vector<uint64_t> colorSets;
    for (int v : verts) {
      bool placed = false;
      for (size_t i = 0; i < colorSets.size(); i++) {
        if ((adj[v] & colorSets[i]) == 0) {
          colorSets[i] |= (1ULL << v);
          placed = true;
          colors.push_back((int)i + 1);
          break;
        }
      }
      if (!placed) {
        colorSets.push_back(1ULL << v);
        colors.push_back((int)colorSets.size());
      }
    }
    return (int)colorSets.size();
  }

  void expand(uint64_t candMask, vector<int>& cur) {
    if (!candMask) {
      if ((int)cur.size() > bestSize) {
        bestSize = (int)cur.size();
        best = cur;
      }
      return;
    }

    vector<int> order, col;
    colorBound(order, col, candMask);
    for (int idx = (int)order.size() - 1; idx >= 0; --idx) {
      int v = order[idx];
      int bound = col[idx];
      if ((int)cur.size() + bound <= bestSize) return;
      uint64_t vbit = 1ULL << v;
      if ((candMask & vbit) == 0) continue;

      cur.push_back(v);
      expand(candMask & ~adj[v] & ~vbit, cur);
      cur.pop_back();

      candMask &= ~vbit;
      if (!candMask) break;
    }
  }

  vector<int> solve(int n_, const vector<pair<int,int>>& edges) {
    n = n_;
    adj.assign(n, 0);
    for (auto [u,v] : edges) {
      if (u==v) continue;
      adj[u] |= 1ULL<<v;
      adj[v] |= 1ULL<<u;
    }
    uint64_t all = (n==64)?~0ULL:((1ULL<<n)-1ULL);
    vector<int> cur;
    expand(all, cur);
    return best;
  }
};

// ---------- Large n solver ----------
struct BitsetGraph {
  int n = 0;
  int blocks = 0;
  vector<vector<uint64_t>> adj;
  vector<int> deg;

  BitsetGraph() = default;
  explicit BitsetGraph(int n_) { init(n_); }

  void init(int n_) {
    n = n_;
    blocks = (n + 63) / 64;
    adj.assign(n, vector<uint64_t>(blocks, 0));
    deg.assign(n, 0);
  }

  void add_edge(int u, int v) {
    if (u==v) return;
    int bu = v >> 6, bv = u >> 6;
    uint64_t mu = 1ULL << (v & 63);
    uint64_t mv = 1ULL << (u & 63);
    if ((adj[u][bu] & mu) == 0) { adj[u][bu] |= mu; deg[u]++; }
    if ((adj[v][bv] & mv) == 0) { adj[v][bv] |= mv; deg[v]++; }
  }
};

static inline bool bit_test(const vector<uint64_t>& bs, int v) {
  return (bs[v >> 6] >> (v & 63)) & 1ULL;
}
static inline void bit_set(vector<uint64_t>& bs, int v) {
  bs[v >> 6] |= 1ULL << (v & 63);
}
static inline void bit_reset(vector<uint64_t>& bs, int v) {
  bs[v >> 6] &= ~(1ULL << (v & 63));
}
static inline int bit_popcount(const vector<uint64_t>& bs) {
  int c = 0;
  for (uint64_t w : bs) c += (int)__popcnt64(w);
  return c;
}
static inline void bit_to_list(const vector<uint64_t>& bs, vector<int>& out) {
  out.clear();
  for (int bi = 0; bi < (int)bs.size(); bi++) {
    uint64_t w = bs[bi];
    while (w) {
      uint64_t lsb = w & -w;
      int bit = __builtin_ctzll(w);
      out.push_back((bi<<6) + bit);
      w ^= lsb;
    }
  }
}

struct MIS_BnB {
  BitsetGraph* g = nullptr;
  int blocks = 0;
  vector<int> best;
  int bestSize = 0;

  int greedy_color(const vector<uint64_t>& cand, vector<int>& order, vector<int>& colors) {
    vector<int> verts;
    bit_to_list(cand, verts);
    sort(verts.begin(), verts.end(), [&](int a, int b) { return g->deg[a] > g->deg[b]; });

    vector<vector<uint64_t>> colorSets;
    order.clear(); colors.clear();
    order.reserve(verts.size()); colors.reserve(verts.size());

    for (int v : verts) {
      bool placed = false;
      for (size_t c = 0; c < colorSets.size(); c++) {
        bool conflict = false;
        for (int bi = 0; bi < blocks; bi++) {
          if (g->adj[v][bi] & colorSets[c][bi]) { conflict = true; break; }
        }
        if (!conflict) {
          bit_set(colorSets[c], v);
          placed = true;
          order.push_back(v);
          colors.push_back((int)c + 1);
          break;
        }
      }
      if (!placed) {
        colorSets.push_back(vector<uint64_t>(blocks, 0));
        bit_set(colorSets.back(), v);
        order.push_back(v);
        colors.push_back((int)colorSets.size());
      }
    }
    return (int)colorSets.size();
  }

  void expand(vector<uint64_t>& cand, vector<int>& cur) {
    int candCount = bit_popcount(cand);
    if (candCount == 0) {
      if ((int)cur.size() > bestSize) { bestSize = (int)cur.size(); best = cur; }
      return;
    }
    if ((int)cur.size() + candCount <= bestSize) return;

    vector<int> order, colors;
    greedy_color(cand, order, colors);
    for (int idx = (int)order.size() - 1; idx >= 0; --idx) {
      if ((int)cur.size() + colors[idx] <= bestSize) return;
      int v = order[idx];
      if (!bit_test(cand, v)) continue;

      cur.push_back(v);
      vector<uint64_t> newCand = cand;
      bit_reset(newCand, v);
      for (int bi = 0; bi < blocks; bi++) newCand[bi] &= ~g->adj[v][bi];
      expand(newCand, cur);
      cur.pop_back();

      bit_reset(cand, v);
      if (bit_popcount(cand) == 0) break;
    }
  }

  vector<int> solve(BitsetGraph& graph, const vector<int>& verticesSubset) {
    int origN = graph.n;
    if ((int)verticesSubset.size() == origN) {
      g = &graph; blocks = graph.blocks;
      vector<uint64_t> cand(blocks, 0);
      for (int v = 0; v < origN; v++) bit_set(cand, v);
      vector<int> cur;
      best.clear(); bestSize = 0;
      expand(cand, cur);
      return best;
    }

    int subN = (int)verticesSubset.size();
    vector<int> inv(subN);
    vector<int> map(origN, -1);
    for (int i = 0; i < subN; i++) { map[verticesSubset[i]] = i; inv[i] = verticesSubset[i]; }

    BitsetGraph sub(subN);
    for (int i = 0; i < subN; i++) {
      int v = verticesSubset[i];
      for (int bi = 0; bi < graph.blocks; bi++) {
        uint64_t w = graph.adj[v][bi];
        while (w) {
          uint64_t lsb = w & -w;
          int bit = __builtin_ctzll(w);
          int u = (bi<<6) + bit;
          w ^= lsb;
          if (u > v && map[u] != -1) sub.add_edge(i, map[u]);
        }
      }
    }
    g = &sub; blocks = sub.blocks;
    vector<uint64_t> cand(blocks, 0);
    for (int v = 0; v < subN; v++) bit_set(cand, v);
    vector<int> cur;
    best.clear(); bestSize = 0;
    expand(cand, cur);
    for (int& x : best) x = inv[x];
    return best;
  }
};

static void build_components(const vector<vector<int>>& adjList, const vector<int>& verts, vector<vector<int>>& comps) {
  vector<char> inSet(adjList.size(), 0), vis(adjList.size(), 0);
  for (int v : verts) inSet[v] = 1;
  comps.clear();
  for (int s : verts) {
    if (vis[s]) continue;
    vector<int> comp;
    deque<int> dq;
    dq.push_back(s);
    vis[s] = 1;
    while (!dq.empty()) {
      int v = dq.front(); dq.pop_front();
      comp.push_back(v);
      for (int u : adjList[v]) {
        if (!inSet[u] || vis[u]) continue;
        vis[u] = 1;
        dq.push_back(u);
      }
    }
    comps.push_back(move(comp));
  }
}

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n, m;
  if (!(cin >> n >> m)) return 0;
  vector<pair<int,int>> edges;
  edges.reserve(m);
  vector<vector<int>> adjList(n);
  for (int i = 0; i < m; i++) {
    int u, v;
    cin >> u >> v;
    if (u < 0 || v < 0 || u >= n || v >= n) continue;
    if (u == v) continue;
    edges.push_back({u, v});
    adjList[u].push_back(v);
    adjList[v].push_back(u);
  }

  vector<int> answer;

  if (n <= 64) {
    MIS64 solver;
    answer = solver.solve(n, edges);
  } else {
    BitsetGraph g(n);
    for (auto [u,v] : edges) g.add_edge(u, v);

    vector<int> alive(n, 1);
    vector<int> deg = g.deg;
    deque<int> q;
    for (int i = 0; i < n; i++) if (deg[i] <= 1) q.push_back(i);

    vector<int> forcedIn;

    while (!q.empty()) {
      int v = q.front(); q.pop_front();
      if (!alive[v]) continue;
      if (deg[v] == 0) {
        alive[v] = 0;
        forcedIn.push_back(v);
      } else if (deg[v] == 1) {
        int u = -1;
        for (int nb : adjList[v]) if (alive[nb]) { u = nb; break; }
        if (u == -1) { deg[v] = 0; q.push_back(v); continue; }
        alive[v] = 0;
        forcedIn.push_back(v);
        if (alive[u]) {
          alive[u] = 0;
          for (int w : adjList[u]) if (alive[w]) { deg[w]--; if (deg[w] <= 1) q.push_back(w); }
        }
      }
      for (int w : adjList[v]) if (alive[w]) { deg[w]--; if (deg[w] <= 1) q.push_back(w); }
    }

    vector<int> core;
    core.reserve(n);
    for (int i = 0; i < n; i++) if (alive[i]) core.push_back(i);

    if (core.empty()) {
      answer = forcedIn;
    } else {
      vector<vector<int>> comps;
      build_components(adjList, core, comps);

      MIS_BnB solver;
      answer = forcedIn;
      for (auto& comp : comps) {
        vector<int> part = solver.solve(g, comp);
        answer.insert(answer.end(), part.begin(), part.end());
      }
    }
  }

  sort(answer.begin(), answer.end());
  answer.erase(unique(answer.begin(), answer.end()), answer.end());

  vector<char> chosen(n, 0);
  for (int v : answer) if (0 <= v && v < n) chosen[v] = 1;
  vector<int> finalAns;
  for (int v : answer) {
    bool ok = true;
    for (int u : adjList[v]) if (chosen[u]) { ok = false; break; }
    if (ok) finalAns.push_back(v);
    else chosen[v] = 0;
  }

  cout << finalAns.size() << "\n";
  for (size_t i = 0; i < finalAns.size(); i++) {
    if (i) cout << ' ';
    cout << finalAns[i];
  }
  cout << "\n";
  return 0;
}
