const fs = require('fs');

function dist(ax, ay, bx, by) {
  const dx = ax - bx;
  const dy = ay - by;
  return Math.hypot(dx, dy);
}

function mulberry32(seed) {
  let t = seed >>> 0;
  return function () {
    t += 0x6D2B79F5;
    let r = Math.imul(t ^ (t >>> 15), 1 | t);
    r ^= r + Math.imul(r ^ (r >>> 7), 61 | r);
    return ((r ^ (r >>> 14)) >>> 0) / 4294967296;
  };
}

function routeCost(routeIdxs, depot, packages) {
  let total = 0;
  let px = depot[0],
    py = depot[1];
  for (let i = 0; i < routeIdxs.length; i++) {
    const p = packages[routeIdxs[i]];
    total += dist(px, py, p.x, p.y);
    px = p.x;
    py = p.y;
  }
  total += dist(px, py, depot[0], depot[1]);
  return total;
}

function twoOptInPlace(routeIdxs, depot, packages, maxPasses) {
  const n = routeIdxs.length;
  if (n < 4) return;
  let passes = 0;
  while (passes++ < maxPasses) {
    let improved = false;
    let prevX = depot[0],
      prevY = depot[1];
    const xs = new Array(n);
    const ys = new Array(n);
    for (let i = 0; i < n; i++) {
      const p = packages[routeIdxs[i]];
      xs[i] = p.x;
      ys[i] = p.y;
    }

    for (let i = 0; i < n - 2; i++) {
      const ax = i === 0 ? depot[0] : xs[i - 1];
      const ay = i === 0 ? depot[1] : ys[i - 1];
      const bx = xs[i];
      const by = ys[i];
      for (let k = i + 1; k < n - 1; k++) {
        const cx = xs[k];
        const cy = ys[k];
        const dx = xs[k + 1];
        const dy = ys[k + 1];
        const before = dist(ax, ay, bx, by) + dist(cx, cy, dx, dy);
        const after = dist(ax, ay, cx, cy) + dist(bx, by, dx, dy);
        if (after + 1e-12 < before) {
          // reverse segment [i..k]
          for (let l = i, r = k; l < r; l++, r--) {
            const tmp = routeIdxs[l];
            routeIdxs[l] = routeIdxs[r];
            routeIdxs[r] = tmp;
          }
          // update cached coords for this route to keep loop correct
          for (let t = i; t <= k; t++) {
            const p = packages[routeIdxs[t]];
            xs[t] = p.x;
            ys[t] = p.y;
          }
          improved = true;
        }
      }
      prevX = bx;
      prevY = by;
    }
    if (!improved) break;
  }
}

function buildInitialRoutes(depot, truckCount, packages) {
  const n = packages.length;
  const routes = Array.from({ length: truckCount }, () => []);
  if (n === 0) return routes;

  const dx = new Array(n);
  const dy = new Array(n);
  const ang = new Array(n);
  const ddep = new Array(n);
  const cx = depot[0],
    cy = depot[1];
  for (let i = 0; i < n; i++) {
    const p = packages[i];
    const vx = p.x - cx;
    const vy = p.y - cy;
    dx[i] = vx;
    dy[i] = vy;
    ang[i] = Math.atan2(vy, vx);
    ddep[i] = Math.hypot(vx, vy);
  }

  // Deterministic pseudo-random phase shift based on input geometry
  let seed = 2166136261 >>> 0;
  for (let i = 0; i < n; i++) {
    seed ^= (packages[i].id * 2654435761) >>> 0;
    seed = Math.imul(seed, 16777619) >>> 0;
    seed ^= (Math.trunc(packages[i].x) | 0) >>> 0;
    seed = Math.imul(seed, 16777619) >>> 0;
    seed ^= (Math.trunc(packages[i].y) | 0) >>> 0;
    seed = Math.imul(seed, 16777619) >>> 0;
  }
  const rnd = mulberry32(seed);
  const phase = (rnd() - 0.5) * Math.PI * 0.25; // small stable shift

  const idxs = Array.from({ length: n }, (_, i) => i);
  idxs.sort((a, b) => {
    const aa = ang[a] + phase;
    const bb = ang[b] + phase;
    if (aa !== bb) return aa - bb;
    return ddep[a] - ddep[b];
  });

  // Split into contiguous wedges; aim balanced route distance proxy
  // by using farthest-first boundaries.
  const target = Math.ceil(n / Math.max(1, truckCount));
  let r = 0;
  for (let i = 0; i < n; i++) {
    routes[r].push(idxs[i]);
    if (routes[r].length >= target && r < truckCount - 1) r++;
  }

  // Ensure we don't output empty trailing routes unless required by distribution.
  return routes.filter((rt) => rt.length > 0);
}

function nearestNeighborOrder(routeIdxs, depot, packages) {
  const m = routeIdxs.length;
  if (m <= 2) return routeIdxs;
  const unvisited = routeIdxs.slice();
  const ordered = [];
  let cx = depot[0],
    cy = depot[1];
  while (unvisited.length) {
    let bestPos = 0;
    let bestD = Infinity;
    for (let i = 0; i < unvisited.length; i++) {
      const p = packages[unvisited[i]];
      const d = dist(cx, cy, p.x, p.y);
      if (d < bestD) {
        bestD = d;
        bestPos = i;
      }
    }
    const chosen = unvisited[bestPos];
    ordered.push(chosen);
    const p = packages[chosen];
    cx = p.x;
    cy = p.y;
    unvisited[bestPos] = unvisited[unvisited.length - 1];
    unvisited.pop();
  }
  return ordered;
}

function relocateImprove(routes, depot, packages, timeLimitMs) {
  const start = Date.now();
  if (routes.length <= 1) return;

  // Precompute pairwise distances to depot and between packages for faster deltas for large N
  // (Sparse approach: only compute on demand via cache)
  const pkgCount = packages.length;
  const cache = new Map();
  function key(a, b) {
    return a < b ? a * 1000003 + b : b * 1000003 + a;
  }
  function pd(a, b) {
    if (a === -1 && b === -1) return 0;
    if (a === -1) {
      const p = packages[b];
      return dist(depot[0], depot[1], p.x, p.y);
    }
    if (b === -1) {
      const p = packages[a];
      return dist(p.x, p.y, depot[0], depot[1]);
    }
    if (a === b) return 0;
    const k = key(a, b);
    const v = cache.get(k);
    if (v !== undefined) return v;
    const pa = packages[a];
    const pb = packages[b];
    const d = dist(pa.x, pa.y, pb.x, pb.y);
    cache.set(k, d);
    return d;
  }

  function routeDeltaRemove(route, pos) {
    const a = pos === 0 ? -1 : route[pos - 1];
    const b = route[pos];
    const c = pos === route.length - 1 ? -1 : route[pos + 1];
    return -pd(a, b) - pd(b, c) + pd(a, c);
  }

  function routeDeltaInsert(route, insertPos, b) {
    const a = insertPos === 0 ? -1 : route[insertPos - 1];
    const c = insertPos === route.length ? -1 : route[insertPos];
    return -pd(a, c) + pd(a, b) + pd(b, c);
  }

  // Greedy relocations between routes.
  while (Date.now() - start < timeLimitMs) {
    const nRoutes = routes.length;
    if (nRoutes <= 1) break;
    let bestGain = 0;
    let bestMove = null;
    for (let r1 = 0; r1 < nRoutes; r1++) {
      const route1 = routes[r1];
      for (let i = 0; i < route1.length; i++) {
        const pkgIdx = route1[i];
        const deltaRemove = routeDeltaRemove(route1, i);
        for (let r2 = 0; r2 < nRoutes; r2++) {
          if (r1 === r2) continue;
          const route2 = routes[r2];
          // Try a few promising insert positions near nearest neighbor in route2
          let candidatePositions = [0, route2.length];
          // Pick closest node in route2 to pkgIdx
          if (route2.length) {
            let bestJ = 0;
            let bestD = Infinity;
            const p = packages[pkgIdx];
            for (let j = 0; j < route2.length; j++) {
              const q = packages[route2[j]];
              const d = dist(p.x, p.y, q.x, q.y);
              if (d < bestD) {
                bestD = d;
                bestJ = j;
              }
            }
            candidatePositions.push(bestJ);
            candidatePositions.push(bestJ + 1);
          }
          // Dedup and clamp
          const seen = new Set();
          for (let t = 0; t < candidatePositions.length; t++) {
            let pos = candidatePositions[t];
            if (pos < 0) pos = 0;
            if (pos > route2.length) pos = route2.length;
            if (seen.has(pos)) continue;
            seen.add(pos);
            const deltaIns = routeDeltaInsert(route2, pos, pkgIdx);
            const gain = -(deltaRemove + deltaIns); // positive if improves total
            if (gain > bestGain + 1e-12) {
              bestGain = gain;
              bestMove = { r1, i, r2, pos };
            }
          }
        }
      }
    }
    if (!bestMove) break;
    const moved = routes[bestMove.r1].splice(bestMove.i, 1)[0];
    routes[bestMove.r2].splice(bestMove.pos, 0, moved);
    // keep route count stable; avoid empty routes by removing them and adjusting indices
    for (let r = routes.length - 1; r >= 0; r--) {
      if (routes[r].length === 0) routes.splice(r, 1);
    }
  }
}

function solve(data) {
  const depot = data.depot;
  const truckCount = Math.max(1, data.truckCount | 0);
  const packages = Array.isArray(data.packages) ? data.packages : [];

  const n = packages.length;
  if (n === 0) return { routes: [] };

  // Work with package indices internally, output ids.
  let routes = buildInitialRoutes(depot, truckCount, packages);

  // If we created fewer routes than allowed, that's fine; but ensure not exceeding truckCount.
  if (routes.length > truckCount) routes = routes.slice(0, truckCount);

  // Order each route using nearest neighbor, then 2-opt.
  for (let r = 0; r < routes.length; r++) {
    routes[r] = nearestNeighborOrder(routes[r], depot, packages);
    // passes tuned: more for smaller routes, fewer for huge
    const passes = routes[r].length <= 80 ? 6 : routes[r].length <= 250 ? 3 : 1;
    twoOptInPlace(routes[r], depot, packages, passes);
  }

  // If we have fewer routes than truckCount, we can split largest routes to use capacity.
  while (routes.length < truckCount) {
    let bestR = -1;
    let bestLen = 0;
    for (let r = 0; r < routes.length; r++) {
      if (routes[r].length > bestLen) {
        bestLen = routes[r].length;
        bestR = r;
      }
    }
    if (bestR === -1 || bestLen < 2) break;
    const route = routes[bestR];
    const half = Math.floor(route.length / 2);
    const a = route.slice(0, half);
    const b = route.slice(half);
    routes[bestR] = a;
    routes.push(b);
    routes[bestR] = nearestNeighborOrder(routes[bestR], depot, packages);
    routes[routes.length - 1] = nearestNeighborOrder(routes[routes.length - 1], depot, packages);
  }

  // Light inter-route improvement with a time cap based on instance size.
  const timeLimitMs = n <= 400 ? 60 : n <= 1500 ? 110 : 160;
  relocateImprove(routes, depot, packages, timeLimitMs);

  // Final 2-opt polish on each route with tiny budget.
  for (let r = 0; r < routes.length; r++) {
    const passes = routes[r].length <= 120 ? 2 : 1;
    twoOptInPlace(routes[r], depot, packages, passes);
  }

  // Map to package IDs.
  const outRoutes = routes
    .filter((rt) => rt.length > 0)
    .map((rt) => rt.map((idx) => packages[idx].id));
  return { routes: outRoutes };
}

const input = fs.readFileSync(0, 'utf8').trim();
const data = input ? JSON.parse(input) : {};
const result = solve(data);
process.stdout.write(JSON.stringify(result));
