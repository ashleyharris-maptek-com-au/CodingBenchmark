use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Vec2 {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy, Debug)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec2 {
    fn sub(self, o: Vec2) -> Vec2 {
        Vec2 {
            x: self.x - o.x,
            y: self.y - o.y,
        }
    }
    fn cross(self, o: Vec2) -> f64 {
        self.x * o.y - self.y * o.x
    }
}

impl Vec3 {
    fn dot(self, o: Vec3) -> f64 {
        self.x * o.x + self.y * o.y + self.z * o.z
    }
    fn len(self) -> f64 {
        self.dot(self).sqrt()
    }
    fn norm(self) -> Vec3 {
        let l = self.len();
        if l == 0.0 {
            Vec3 { x: 0.0, y: 0.0, z: 0.0 }
        } else {
            Vec3 {
                x: self.x / l,
                y: self.y / l,
                z: self.z / l,
            }
        }
    }
}

fn polygon_area(pts: &[Vec2]) -> f64 {
    let n = pts.len();
    if n < 3 {
        return 0.0;
    }
    let mut a = 0.0;
    for i in 0..n {
        let j = (i + 1) % n;
        a += pts[i].cross(pts[j]);
    }
    0.5 * a
}

fn polygon_centroid(pts: &[Vec2]) -> Vec2 {
    let n = pts.len();
    if n == 0 {
        return Vec2 { x: 0.0, y: 0.0 };
    }
    let a2 = polygon_area(pts) * 2.0;
    if a2.abs() < 1e-18 {
        // Fallback: average
        let mut sx = 0.0;
        let mut sy = 0.0;
        for p in pts {
            sx += p.x;
            sy += p.y;
        }
        return Vec2 {
            x: sx / (n as f64),
            y: sy / (n as f64),
        };
    }
    let mut cx = 0.0;
    let mut cy = 0.0;
    for i in 0..n {
        let j = (i + 1) % n;
        let cross = pts[i].cross(pts[j]);
        cx += (pts[i].x + pts[j].x) * cross;
        cy += (pts[i].y + pts[j].y) * cross;
    }
    let inv = 1.0 / (3.0 * a2);
    Vec2 {
        x: cx * inv,
        y: cy * inv,
    }
}

fn aabb(pts: &[Vec2]) -> (Vec2, Vec2) {
    let mut mn = Vec2 { x: f64::INFINITY, y: f64::INFINITY };
    let mut mx = Vec2 { x: f64::NEG_INFINITY, y: f64::NEG_INFINITY };
    for p in pts {
        if p.x < mn.x {
            mn.x = p.x;
        }
        if p.y < mn.y {
            mn.y = p.y;
        }
        if p.x > mx.x {
            mx.x = p.x;
        }
        if p.y > mx.y {
            mx.y = p.y;
        }
    }
    (mn, mx)
}

fn principal_axis_angle(pts: &[Vec2], c: Vec2) -> f64 {
    // 2D PCA-ish: compute covariance, take angle of largest eigenvector.
    let mut sxx = 0.0;
    let mut sxy = 0.0;
    let mut syy = 0.0;
    let mut cnt = 0.0;
    for p in pts {
        let dx = p.x - c.x;
        let dy = p.y - c.y;
        sxx += dx * dx;
        sxy += dx * dy;
        syy += dy * dy;
        cnt += 1.0;
    }
    if cnt == 0.0 {
        return 0.0;
    }
    sxx /= cnt;
    sxy /= cnt;
    syy /= cnt;
    // Angle for eigenvector of max eigenvalue:
    // tan(2theta) = 2sxy / (sxx - syy)
    let denom = sxx - syy;
    if denom.abs() < 1e-18 && sxy.abs() < 1e-18 {
        0.0
    } else {
        0.5 * (2.0 * sxy).atan2(denom)
    }
}

fn quat_from_z_rotation(theta: f64) -> (f64, f64, f64, f64) {
    let half = 0.5 * theta;
    (half.cos(), 0.0, 0.0, half.sin())
}

fn main() {
    // Read all stdin
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let nv_opt = it.next();
    if nv_opt.is_none() {
        return;
    }
    let nv: usize = nv_opt.unwrap().parse().unwrap();
    let mut pts: Vec<Vec2> = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        pts.push(Vec2 { x, y });
    }
    let sx: f64 = it.next().unwrap().parse().unwrap();
    let sy: f64 = it.next().unwrap().parse().unwrap();
    let sz: f64 = it.next().unwrap().parse().unwrap();
    let sun = Vec3 { x: sx, y: sy, z: sz }.norm();
    // If sun is (nearly) parallel to ground, any finite-height object yields extremely long shadow;
    // in that case we still emit a small set near the polygon.
    let sun_z = sun.z.abs().max(1e-9);

    // Compute polygon stats
    let area = polygon_area(&pts).abs();
    let centroid = polygon_centroid(&pts);
    let (mn, mx) = aabb(&pts);
    let w = (mx.x - mn.x).abs();
    let h = (mx.y - mn.y).abs();
    let diag = (w * w + h * h).sqrt().max(1e-9);

    // Adaptive strategy:
    // - Small / thin polygons: 1 tetra placed above centroid, oriented to principal axis.
    // - Larger polygons: tile a rotated grid of tetra shadows (no 3D intersections ensured by z-stacking).
    //
    // NOTE: This is a pragmatic baseline. It prioritizes producing valid, warning-free output and
    // has adaptive behavior for varying scales.

    let theta = principal_axis_angle(&pts, centroid);
    let (qw, qx, qy, qz) = quat_from_z_rotation(theta);

    // Estimate how "many" we need. Use area relative to what one max-scale tetra can cover.
    // We approximate a tetra's 2D shadow footprint as proportional to scale^2.
    // Use a conservative coefficient to avoid under-covering.
    let max_scale = 10.0_f64;
    let ref_cover = 60.0 * max_scale * max_scale; // heuristic coverage area per tetra
    let mut target = if area <= 1e-12 {
        1usize
    } else {
        ((area / ref_cover).ceil() as usize).max(1)
    };

    // For very small bounding boxes, keep it to 1.
    if diag < 1e-3 {
        target = 1;
    }

    // Cap for extreme complexity (keeps runtime stable).
    // The benchmark wants "minimize number", but a safe cap prevents runaway output.
    if target > 5000 {
        target = 5000;
    }

    // Choose scale per tetra: bigger when fewer, smaller when many; clamp to [0.5, 10].
    let mut scale = (area / (ref_cover * (target as f64))).sqrt() * max_scale;
    if !scale.is_finite() {
        scale = max_scale;
    }
    if scale < 0.5 {
        scale = 0.5;
    }
    if scale > max_scale {
        scale = max_scale;
    }

    // If we only output one tetra, place it safely above polygon so its shadow is long enough.
    if target == 1 {
        // Height selection: higher gives longer shadow if sun_z is small.
        let base_height = (0.25 * diag).max(0.5);
        let height = (base_height / sun_z).min(200.0).max(1.0);
        let px = centroid.x;
        let py = centroid.y;
        let pz = height;
        println!("1");
        println!(
            "{:.9} {:.9} {:.9} {:.9} {:.9} {:.9} {:.9} {:.9}",
            px, py, pz, qw, qx, qy, qz, scale
        );
        return;
    }

    // Multi-tetra tiling:
    // Build a rotated grid that covers the AABB expanded by margin, place tetra centers on z-stacked layers
    // to avoid any 3D intersections (strictly increasing z with enough separation).
    //
    // Shadow coverage is not computed exactly; we spread points over polygon AABB in rotated frame,
    // using more points when area is larger.
    let margin = 0.1 * diag + 1e-6;
    let (mn2, mx2) = (Vec2 { x: mn.x - margin, y: mn.y - margin }, Vec2 { x: mx.x + margin, y: mx.y + margin });
    let w2 = (mx2.x - mn2.x).max(1e-9);
    let h2 = (mx2.y - mn2.y).max(1e-9);

    // Decide grid dimensions near-square to hit target.
    let aspect = if h2 > 0.0 { w2 / h2 } else { 1.0 };
    let mut gx = ((target as f64) * aspect).sqrt().ceil() as usize;
    if gx == 0 {
        gx = 1;
    }
    let mut gy = ((target as f64) as f64 / (gx as f64)).ceil() as usize;
    if gy == 0 {
        gy = 1;
    }
    while gx * gy < target {
        gy += 1;
    }

    // Center grid on centroid
    let step_x = w2 / (gx as f64);
    let step_y = h2 / (gy as f64);
    let step = step_x.min(step_y).max(1e-6);

    // z stacking: ensure separation > 2*scale*(tetra radius approx 1.3) to avoid intersections.
    let sep = (3.0 * scale).max(0.1);
    let base_height = (0.5 * diag).max(1.0);
    let layer_height = (base_height / sun_z).min(400.0).max(2.0);

    // Precompute rotation for grid (theta) so we align with polygon axis.
    let ct = theta.cos();
    let st = theta.sin();
    let rot = |v: Vec2| -> Vec2 {
        Vec2 {
            x: ct * v.x - st * v.y,
            y: st * v.x + ct * v.y,
        }
    };
    let inv_rot = |v: Vec2| -> Vec2 {
        Vec2 {
            x: ct * v.x + st * v.y,
            y: -st * v.x + ct * v.y,
        }
    };

    // Work in rotated frame about centroid.
    let mn_r = rot(mn2.sub(centroid));
    let mx_r = rot(mx2.sub(centroid));
    let minx = mn_r.x.min(mx_r.x);
    let maxx = mn_r.x.max(mx_r.x);
    let miny = mn_r.y.min(mx_r.y);
    let maxy = mn_r.y.max(mx_r.y);
    let spanx = (maxx - minx).max(1e-9);
    let spany = (maxy - miny).max(1e-9);

    // Determine count and output.
    let count = gx * gy;
    println!("{}", count);

    // Emit tetrahedrons, layer by layer in scanline order with z increasing.
    // Use slight in-plane jitter to reduce degenerate alignments.
    let jitter = 0.1 * step;
    let mut idx = 0usize;
    for iy in 0..gy {
        for ix in 0..gx {
            let fx = (ix as f64 + 0.5) / (gx as f64);
            let fy = (iy as f64 + 0.5) / (gy as f64);
            let mut x_r = minx + fx * spanx;
            let mut y_r = miny + fy * spany;
            // deterministic jitter
            let t = (idx as f64) * 12.9898;
            let jx = (t.sin()) * jitter;
            let jy = (t.cos()) * jitter;
            x_r += jx;
            y_r += jy;
            let p_local = inv_rot(Vec2 { x: x_r, y: y_r });
            let px = centroid.x + p_local.x;
            let py = centroid.y + p_local.y;
            let pz = layer_height + (idx as f64) * sep;
            println!(
                "{:.9} {:.9} {:.9} {:.9} {:.9} {:.9} {:.9} {:.9}",
                px, py, pz, qw, qx, qy, qz, scale
            );
            idx += 1;
        }
    }
}
