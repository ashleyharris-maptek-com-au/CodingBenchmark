using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

public static class Program
{
    sealed class FastScanner
    {
        private readonly Stream _s;
        private readonly byte[] _buf = new byte[1 << 20];
        private int _len, _ptr;
        public FastScanner(Stream s) { _s = s; }
        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }
        public int NextInt()
        {
            int c;
            do { c = ReadByte(); if (c == -1) return int.MinValue; } while (c <= ' ');
            int sign = 1;
            if (c == '-') { sign = -1; c = ReadByte(); }
            int x = 0;
            while (c > ' ')
            {
                x = x * 10 + (c - '0');
                c = ReadByte();
            }
            return x * sign;
        }
    }

    struct AdjEdge
    {
        public int To;
        public int Id;
        public AdjEdge(int to, int id) { To = to; Id = id; }
    }

    static int[] BuildEulerTour(int n, List<AdjEdge>[] g, int m, int start)
    {
        var used = new bool[m];
        var it = new int[n];
        var stack = new int[Math.Max(4, m + 5)];
        var path = new int[m + 1];
        int sp = 0;
        int pp = 0;
        stack[sp++] = start;

        while (sp > 0)
        {
            int v = stack[sp - 1];
            var gv = g[v];
            int i = it[v];
            while (i < gv.Count && used[gv[i].Id]) i++;
            it[v] = i;
            if (i == gv.Count)
            {
                path[pp++] = v;
                sp--;
            }
            else
            {
                var e = gv[i];
                used[e.Id] = true;
                it[v] = i + 1;
                stack[sp++] = e.To;
            }
        }

        Array.Reverse(path, 0, pp);
        if (pp != m + 1)
        {
            Array.Resize(ref path, pp);
            return path;
        }
        return path;
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        if (n == int.MinValue) return;
        int mInput = fs.NextInt();

        var g = new List<AdjEdge>[n];
        for (int i = 0; i < n; i++) g[i] = new List<AdjEdge>();

        int[] deg = new int[n];

        int m = 0;
        for (int i = 0; i < mInput; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            _ = fs.NextInt();
            g[u].Add(new AdjEdge(v, m));
            g[v].Add(new AdjEdge(u, m));
            deg[u]++; deg[v]++;
            m++;
        }

        // BFS tree from 0
        var parent = new int[n];
        Array.Fill(parent, -1);
        parent[0] = 0;
        var q = new int[n];
        int qh = 0, qt = 0;
        q[qt++] = 0;
        while (qh < qt)
        {
            int v = q[qh++];
            foreach (var e in g[v])
            {
                int to = e.To;
                if (parent[to] != -1) continue;
                parent[to] = v;
                q[qt++] = to;
            }
        }

        for (int i = 0; i < n; i++)
        {
            if (parent[i] == -1 && deg[i] > 0)
            {
                Console.WriteLine("0 0");
                return;
            }
        }

        // Duplicate tree edges on paths from odd vertices to root
        for (int i = 0; i < n; i++)
        {
            if ((deg[i] & 1) == 0 || i == 0) continue;
            int v = i;
            while (v != 0)
            {
                int p = parent[v];
                g[v].Add(new AdjEdge(p, m));
                g[p].Add(new AdjEdge(v, m));
                deg[v]++; deg[p]++;
                m++;
                v = p;
            }
        }

        int[] tour = BuildEulerTour(n, g, m, 0);

        var sb = new StringBuilder(tour.Length * 2);
        for (int i = 0; i < tour.Length; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(tour[i]);
        }
        Console.Write(sb.ToString());
    }
}
