using System;
using System.Diagnostics;
using System.IO;
using System.Text;

// Quadratic Assignment Problem heuristic solver.
// Strategy:
// - For very small N (<= 11): exact search with branch-and-bound.
// - For larger N: build an initial permutation using a cheap greedy score,
//   then improve with 2-opt style swaps using delta evaluation (O(N) per swap),
//   in a multi-start timed loop.
public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;
        public FastScanner(Stream stream) { _stream = stream; }
        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }
        public int NextInt()
        {
            int c;
            do { c = ReadByte(); if (c == -1) return int.MinValue; } while (c <= 32);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }
            int x = 0;
            while (c > 32 && c != -1)
            {
                x = x * 10 + (c - '0');
                c = ReadByte();
            }
            return x * sign;
        }
    }

    private sealed class XorShift64
    {
        private ulong _x;
        public XorShift64(ulong seed) { _x = seed != 0 ? seed : 88172645463325252UL; }
        public ulong NextU64()
        {
            ulong x = _x;
            x ^= x << 7;
            x ^= x >> 9;
            _x = x;
            return x;
        }
        public int NextInt(int n) => (int)(NextU64() % (uint)n);
        public double NextDouble() => (NextU64() >> 11) * (1.0 / (1UL << 53));
    }

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n = scanner.NextInt();
        if (n == int.MinValue || n <= 0)
            return;

        int[][] flow = new int[n][];
        for (int i = 0; i < n; i++)
        {
            var row = new int[n];
            for (int j = 0; j < n; j++) row[j] = scanner.NextInt();
            flow[i] = row;
        }
        int[][] dist = new int[n][];
        for (int i = 0; i < n; i++)
        {
            var row = new int[n];
            for (int j = 0; j < n; j++) row[j] = scanner.NextInt();
            dist[i] = row;
        }

        int[] bestPerm;
        long bestCost;

        if (n <= 11)
        {
            SolveExactSmall(n, flow, dist, out bestPerm, out bestCost);
        }
        else
        {
            SolveHeuristic(n, flow, dist, out bestPerm, out bestCost);
        }

        var sb = new StringBuilder();
        sb.AppendLine(bestCost.ToString());
        for (int i = 0; i < n; i++)
        {
            if (i != 0) sb.Append(' ');
            sb.Append(bestPerm[i]);
        }
        sb.AppendLine();
        Console.Write(sb.ToString());
    }

    private static void SolveExactSmall(int n, int[][] flow, int[][] dist, out int[] bestPerm, out long bestCost)
    {
        // Simple branch-and-bound:
        // Assign facilities in order 0..n-1. Keep current cost of assigned pairs.
        // Bound: current cost + optimistic 0 for remaining pairs (weak but ok for n<=11).
        // Still fast for n<=11.
        // Use locals for capture by local function; assign to out params at end.
        int[] bestPermLocal = new int[n];
        long bestCostLocal = long.MaxValue;

        int[] perm = new int[n];
        bool[] usedLoc = new bool[n];

        // Precompute dist row sums etc is not necessary for small n.
        void Dfs(int k, long costSoFar)
        {
            if (costSoFar >= bestCostLocal) return;
            if (k == n)
            {
                bestCostLocal = costSoFar;
                Array.Copy(perm, bestPermLocal, n);
                return;
            }
            for (int loc = 0; loc < n; loc++)
            {
                if (usedLoc[loc]) continue;
                long add = 0;
                // incremental cost for pairs (i,k) and (k,i) with already assigned i<k
                for (int i = 0; i < k; i++)
                {
                    int li = perm[i];
                    add += (long)flow[k][i] * dist[loc][li];
                    add += (long)flow[i][k] * dist[li][loc];
                }
                usedLoc[loc] = true;
                perm[k] = loc;
                Dfs(k + 1, costSoFar + add);
                usedLoc[loc] = false;
            }
        }
        Dfs(0, 0);
        bestPerm = bestPermLocal;
        bestCost = bestCostLocal;
    }

    private static void SolveHeuristic(int n, int[][] flow, int[][] dist, out int[] bestPerm, out long bestCost)
    {
        var sw = Stopwatch.StartNew();
        // Keep some safety margin under 5 minutes.
        double timeLimitSec = 295.0;
        // For very large n, keep improvement work limited.
        int maxNoImproveRounds = n >= 1500 ? 2 : 6;

        // Precompute simple weights to build good initial solutions:
        // facilityWeight[i] = sum_j (flow[i][j] + flow[j][i])
        // locationWeight[l]  = sum_m (dist[l][m] + dist[m][l])
        long[] fW = new long[n];
        long[] lW = new long[n];
        for (int i = 0; i < n; i++)
        {
            long s = 0;
            var fi = flow[i];
            for (int j = 0; j < n; j++) s += fi[j] + (long)flow[j][i];
            fW[i] = s;
        }
        for (int i = 0; i < n; i++)
        {
            long s = 0;
            var di = dist[i];
            for (int j = 0; j < n; j++) s += di[j] + (long)dist[j][i];
            lW[i] = s;
        }

        int[] facilitiesByW = ArgsortDesc(fW);
        int[] locationsByW = ArgsortAsc(lW);

        // Baseline greedy: map high-flow facilities to low-distance locations.
        int[] basePerm = new int[n];
        for (int k = 0; k < n; k++)
            basePerm[facilitiesByW[k]] = locationsByW[k];

        bestPerm = (int[])basePerm.Clone();
        bestCost = ComputeCost(bestPerm, flow, dist);

        // Multi-start:
        // - start from greedy
        // - plus a few randomized perturbations of greedy ordering
        // - local search by swap (delta O(n))
        ulong seed = (ulong)DateTime.UtcNow.Ticks ^ (ulong)n * 0x9E3779B97F4A7C15UL;
        var rng = new XorShift64(seed);

        int[] perm = (int[])bestPerm.Clone();
        long permCost = bestCost;

        int roundsNoImprove = 0;
        int starts = 0;

        // Adaptive neighborhood sampling sizes
        int candidateIters = n <= 500 ? 25_000 : n <= 1200 ? 35_000 : 45_000;

        while (sw.Elapsed.TotalSeconds < timeLimitSec)
        {
            if (starts == 0)
            {
                Array.Copy(basePerm, perm, n);
                permCost = bestCost;
            }
            else
            {
                // Perturbation: start from greedy but scramble blocks, then do a few random swaps.
                Array.Copy(basePerm, perm, n);
                BlockShufflePermutationInPlace(perm, rng, Math.Max(4, n / 40));
                int kswaps = n <= 600 ? 50 : n <= 1200 ? 80 : 120;
                for (int t = 0; t < kswaps; t++)
                {
                    int a = rng.NextInt(n);
                    int b = rng.NextInt(n);
                    if (a == b) continue;
                    int tmp = perm[a]; perm[a] = perm[b]; perm[b] = tmp;
                }
                permCost = ComputeCost(perm, flow, dist);
            }

            // Local search: best-improving (stochastic sampled) swaps
            bool improvedThisStart = ImproveBySwapsTimed(perm, ref permCost, flow, dist, rng, sw, timeLimitSec, candidateIters);

            if (permCost < bestCost)
            {
                bestCost = permCost;
                bestPerm = (int[])perm.Clone();
                roundsNoImprove = 0;
            }
            else
            {
                if (!improvedThisStart) roundsNoImprove++;
                else roundsNoImprove++;
            }

            starts++;
            if (roundsNoImprove >= maxNoImproveRounds && sw.Elapsed.TotalSeconds > 1.0)
            {
                // Reduce work if not improving; for large n, stop restarting too much.
                candidateIters = Math.Max(candidateIters / 2, 8000);
                roundsNoImprove = 0;
            }
        }
    }

    private static int[] ArgsortDesc(long[] a)
    {
        int n = a.Length;
        int[] idx = new int[n];
        for (int i = 0; i < n; i++) idx[i] = i;
        Array.Sort(idx, (i, j) =>
        {
            long di = a[i], dj = a[j];
            if (di == dj) return i.CompareTo(j);
            return dj.CompareTo(di);
        });
        return idx;
    }

    private static int[] ArgsortAsc(long[] a)
    {
        int n = a.Length;
        int[] idx = new int[n];
        for (int i = 0; i < n; i++) idx[i] = i;
        Array.Sort(idx, (i, j) =>
        {
            long di = a[i], dj = a[j];
            if (di == dj) return i.CompareTo(j);
            return di.CompareTo(dj);
        });
        return idx;
    }

    private static void BlockShufflePermutationInPlace(int[] perm, XorShift64 rng, int blockSize)
    {
        int n = perm.Length;
        if (blockSize <= 1) return;
        int blocks = (n + blockSize - 1) / blockSize;
        // Fisher-Yates shuffle blocks by swapping segments.
        for (int b = blocks - 1; b > 0; b--)
        {
            int b2 = rng.NextInt(b + 1);
            if (b2 == b) continue;
            int start1 = b * blockSize;
            int start2 = b2 * blockSize;
            int len1 = Math.Min(blockSize, n - start1);
            int len2 = Math.Min(blockSize, n - start2);
            int len = Math.Min(len1, len2);
            for (int k = 0; k < len; k++)
            {
                int i1 = start1 + k;
                int i2 = start2 + k;
                int tmp = perm[i1];
                perm[i1] = perm[i2];
                perm[i2] = tmp;
            }
        }
    }

    private static long ComputeCost(int[] perm, int[][] flow, int[][] dist)
    {
        int n = perm.Length;
        long cost = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = perm[i];
            var fi = flow[i];
            for (int j = 0; j < n; j++)
            {
                cost += (long)fi[j] * dist[pi][perm[j]];
            }
        }
        return cost;
    }

    // Delta cost of swapping facilities a and b (i.e., swap perm[a], perm[b]) in O(n).
    private static long SwapDelta(int[] perm, int a, int b, int[][] flow, int[][] dist)
    {
        if (a == b) return 0;
        int n = perm.Length;
        int la = perm[a];
        int lb = perm[b];
        long delta = 0;

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;
            int lk = perm[k];

            // Terms affected:
            // flow[a][k]*dist[la][lk] + flow[k][a]*dist[lk][la]
            // flow[b][k]*dist[lb][lk] + flow[k][b]*dist[lk][lb]
            // after swap la<->lb
            delta -= (long)flow[a][k] * dist[la][lk];
            delta -= (long)flow[k][a] * dist[lk][la];
            delta -= (long)flow[b][k] * dist[lb][lk];
            delta -= (long)flow[k][b] * dist[lk][lb];

            delta += (long)flow[a][k] * dist[lb][lk];
            delta += (long)flow[k][a] * dist[lk][lb];
            delta += (long)flow[b][k] * dist[la][lk];
            delta += (long)flow[k][b] * dist[lk][la];
        }

        // Also pairs (a,b) and (b,a) and self pairs (a,a),(b,b) if any
        delta -= (long)flow[a][a] * dist[la][la];
        delta -= (long)flow[b][b] * dist[lb][lb];
        delta -= (long)flow[a][b] * dist[la][lb];
        delta -= (long)flow[b][a] * dist[lb][la];

        delta += (long)flow[a][a] * dist[lb][lb];
        delta += (long)flow[b][b] * dist[la][la];
        delta += (long)flow[a][b] * dist[lb][la];
        delta += (long)flow[b][a] * dist[la][lb];

        return delta;
    }

    private static bool ImproveBySwapsTimed(
        int[] perm,
        ref long cost,
        int[][] flow,
        int[][] dist,
        XorShift64 rng,
        Stopwatch sw,
        double timeLimitSec,
        int candidateIters)
    {
        int n = perm.Length;
        bool improved = false;

        // A few passes of sampled best-improving swaps.
        // For speed, we sample random pairs and apply the best negative delta found per pass.
        int passes = n <= 600 ? 80 : n <= 1200 ? 60 : 45;

        for (int pass = 0; pass < passes; pass++)
        {
            if (sw.Elapsed.TotalSeconds >= timeLimitSec) break;

            long bestDelta = 0;
            int bestA = -1, bestB = -1;

            int iters = candidateIters;
            // Slightly reduce later passes
            if (pass > 10) iters = (int)(iters * 0.75);
            if (pass > 25) iters = (int)(iters * 0.6);
            if (iters < 3000) iters = 3000;

            for (int t = 0; t < iters; t++)
            {
                int a = rng.NextInt(n);
                int b = rng.NextInt(n);
                if (a == b) continue;
                if (a > b) { int tmp = a; a = b; b = tmp; }
                long d = SwapDelta(perm, a, b, flow, dist);
                if (d < bestDelta)
                {
                    bestDelta = d;
                    bestA = a; bestB = b;
                    // Early accept if very good improvement
                    if (bestDelta < -1000000 && n > 800) break;
                }
            }

            if (bestDelta < 0)
            {
                int tmp = perm[bestA];
                perm[bestA] = perm[bestB];
                perm[bestB] = tmp;
                cost += bestDelta;
                improved = true;
            }
            else
            {
                // No improvement found in this pass; stop early
                break;
            }
        }

        return improved;
    }
}
