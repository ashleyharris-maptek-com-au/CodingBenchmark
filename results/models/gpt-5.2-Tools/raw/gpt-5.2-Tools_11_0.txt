use std::cmp::Ordering;
use std::collections::VecDeque;
use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug, Default)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    #[inline]
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }
    #[inline]
    fn add(self, o: Pt) -> Pt {
        Pt::new(self.x + o.x, self.y + o.y)
    }
    #[inline]
    fn sub(self, o: Pt) -> Pt {
        Pt::new(self.x - o.x, self.y - o.y)
    }
    #[inline]
    fn dot(self, o: Pt) -> f64 {
        self.x * o.x + self.y * o.y
    }
    #[inline]
    fn cross(self, o: Pt) -> f64 {
        self.x * o.y - self.y * o.x
    }
    #[inline]
    fn rot(self, ang: f64) -> Pt {
        let (s, c) = ang.sin_cos();
        Pt::new(self.x * c - self.y * s, self.x * s + self.y * c)
    }
}

#[derive(Clone)]
struct Poly {
    v: Vec<Pt>,
    area: f64,
    bbox_min: Pt,
    bbox_max: Pt,
}

fn poly_area(v: &[Pt]) -> f64 {
    let n = v.len();
    if n < 3 {
        return 0.0;
    }
    let mut s = 0.0;
    for i in 0..n {
        let a = v[i];
        let b = v[(i + 1) % n];
        s += a.cross(b);
    }
    0.5 * s
}

fn poly_bbox(v: &[Pt]) -> (Pt, Pt) {
    let mut mn = Pt::new(f64::INFINITY, f64::INFINITY);
    let mut mx = Pt::new(f64::NEG_INFINITY, f64::NEG_INFINITY);
    for &p in v {
        mn.x = mn.x.min(p.x);
        mn.y = mn.y.min(p.y);
        mx.x = mx.x.max(p.x);
        mx.y = mx.y.max(p.y);
    }
    (mn, mx)
}

fn normalize_ccw(mut v: Vec<Pt>) -> Vec<Pt> {
    if poly_area(&v) < 0.0 {
        v.reverse();
    }
    v
}

fn make_poly(v: Vec<Pt>) -> Poly {
    let v = normalize_ccw(v);
    let a = poly_area(&v).abs();
    let (mn, mx) = poly_bbox(&v);
    Poly {
        v,
        area: a,
        bbox_min: mn,
        bbox_max: mx,
    }
}

// ---- Geometry predicates ----
const EPS: f64 = 1e-9;

#[inline]
fn orient(a: Pt, b: Pt, c: Pt) -> f64 {
    (b.sub(a)).cross(c.sub(a))
}

#[inline]
fn on_segment(a: Pt, b: Pt, p: Pt) -> bool {
    let minx = a.x.min(b.x) - 1e-12;
    let maxx = a.x.max(b.x) + 1e-12;
    let miny = a.y.min(b.y) - 1e-12;
    let maxy = a.y.max(b.y) + 1e-12;
    p.x >= minx && p.x <= maxx && p.y >= miny && p.y <= maxy && orient(a, b, p).abs() <= 1e-9
}

fn seg_intersect(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);

    if (o1 > EPS && o2 < -EPS) || (o1 < -EPS && o2 > EPS) {
        if (o3 > EPS && o4 < -EPS) || (o3 < -EPS && o4 > EPS) {
            return true;
        }
    }

    if o1.abs() <= EPS && on_segment(a, b, c) {
        return true;
    }
    if o2.abs() <= EPS && on_segment(a, b, d) {
        return true;
    }
    if o3.abs() <= EPS && on_segment(c, d, a) {
        return true;
    }
    if o4.abs() <= EPS && on_segment(c, d, b) {
        return true;
    }
    false
}

fn point_in_poly(p: Pt, poly: &[Pt]) -> bool {
    // Winding / ray casting (works for non-convex). Treat boundary as inside.
    let n = poly.len();
    let mut inside = false;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if on_segment(a, b, p) {
            return true;
        }
        let cond1 = (a.y > p.y) != (b.y > p.y);
        if cond1 {
            let xint = (b.x - a.x) * (p.y - a.y) / (b.y - a.y) + a.x;
            if xint > p.x {
                inside = !inside;
            }
        }
    }
    inside
}

fn poly_edges(poly: &[Pt]) -> impl Iterator<Item = (Pt, Pt)> + '_ {
    poly.iter()
        .enumerate()
        .map(move |(i, &a)| (a, poly[(i + 1) % poly.len()]))
}

fn polys_overlap(a: &[Pt], b: &[Pt]) -> bool {
    // Edge intersections
    for (a1, a2) in poly_edges(a) {
        for (b1, b2) in poly_edges(b) {
            if seg_intersect(a1, a2, b1, b2) {
                return true;
            }
        }
    }
    // Containment
    if !a.is_empty() && point_in_poly(a[0], b) {
        return true;
    }
    if !b.is_empty() && point_in_poly(b[0], a) {
        return true;
    }
    false
}

fn poly_inside_stock(piece: &[Pt], stock: &[Pt]) -> bool {
    // Conservative: all vertices inside and no edge crosses outside
    for &p in piece {
        if !point_in_poly(p, stock) {
            return false;
        }
    }
    // If piece is inside and no edges intersect stock boundary improperly, accept.
    // Intersections might still be "touching" at boundary; that's ok.
    for (p1, p2) in poly_edges(piece) {
        for (s1, s2) in poly_edges(stock) {
            if seg_intersect(p1, p2, s1, s2) {
                // Allow touching at endpoints/collinear; since point_in_poly includes boundary,
                // we only reject if there is a proper crossing.
                let o1 = orient(p1, p2, s1);
                let o2 = orient(p1, p2, s2);
                let o3 = orient(s1, s2, p1);
                let o4 = orient(s1, s2, p2);
                let proper = ((o1 > EPS && o2 < -EPS) || (o1 < -EPS && o2 > EPS))
                    && ((o3 > EPS && o4 < -EPS) || (o3 < -EPS && o4 > EPS));
                if proper {
                    return false;
                }
            }
        }
    }
    true
}

// ---- Placement / heuristics ----

#[derive(Clone)]
struct PieceInfo {
    idx: usize,
    poly: Poly,
    // representative sizes for fast checks
    w0: f64,
    h0: f64,
    // candidate rotations
    rots: Vec<f64>,
}

#[derive(Clone)]
struct Placed {
    piece_idx: usize,
    stock_idx: usize,
    x: f64,
    y: f64,
    rot: f64,
    bbox_min: Pt,
    bbox_max: Pt,
    poly_world: Vec<Pt>,
}

#[derive(Clone)]
struct StockBin {
    placed: Vec<Placed>,
    // frontier points to try (bottom-left-ish)
    front: VecDeque<Pt>,
    // quick bbox extents for pruning
    bbox_min: Pt,
    bbox_max: Pt,
}

fn transformed_vertices(poly: &[Pt], rot: f64, t: Pt) -> Vec<Pt> {
    poly.iter().map(|&p| p.rot(rot).add(t)).collect()
}

fn bbox_of_transformed(poly: &[Pt], rot: f64, t: Pt) -> (Pt, Pt) {
    let mut mn = Pt::new(f64::INFINITY, f64::INFINITY);
    let mut mx = Pt::new(f64::NEG_INFINITY, f64::NEG_INFINITY);
    for &p in poly {
        let q = p.rot(rot).add(t);
        mn.x = mn.x.min(q.x);
        mn.y = mn.y.min(q.y);
        mx.x = mx.x.max(q.x);
        mx.y = mx.y.max(q.y);
    }
    (mn, mx)
}

fn aabb_overlap(a_min: Pt, a_max: Pt, b_min: Pt, b_max: Pt) -> bool {
    !(a_max.x < b_min.x - 1e-12
        || a_min.x > b_max.x + 1e-12
        || a_max.y < b_min.y - 1e-12
        || a_min.y > b_max.y + 1e-12)
}

fn stock_bbox(stock: &[Pt]) -> (Pt, Pt) {
    poly_bbox(stock)
}

fn choose_rotations(piece: &Poly, np: usize) -> Vec<f64> {
    // Adaptive: fewer rotations when NP large.
    // Always include 0; then 90deg; then 45deg; then finer if small.
    let mut rots = Vec::new();
    rots.push(0.0);
    if np <= 2000 {
        rots.push(PI * 0.5);
    }
    if np <= 800 {
        rots.push(PI * 0.25);
        rots.push(PI * 0.75);
    }
    if np <= 250 {
        // 15-degree steps
        for k in 1..12 {
            rots.push((k as f64) * PI / 12.0);
        }
    }
    // Dedup and normalize [-pi, pi)
    rots.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    rots.dedup_by(|a, b| (*a - *b).abs() < 1e-12);
    // filter NaNs (shouldn't)
    rots.retain(|r| r.is_finite());
    // If piece nearly square, 90deg is redundant; keep anyway but dedup will not remove.
    let _ = piece;
    rots
}

fn init_stock_bin(stock_bbox: (Pt, Pt)) -> StockBin {
    let (mn, mx) = stock_bbox;
    let mut front = VecDeque::new();
    // Start from stock bbox min corner; works well for rectangles and many shapes.
    front.push_back(Pt::new(mn.x, mn.y));
    StockBin {
        placed: Vec::new(),
        front,
        bbox_min: mn,
        bbox_max: mx,
    }
}

fn try_place_in_bin(
    piece: &PieceInfo,
    stock_poly: &[Pt],
    bin: &mut StockBin,
    stock_idx: usize,
    max_front: usize,
    max_trials: usize,
) -> Option<Placed> {
    // Iterate over a limited number of frontier points (FIFO) and rotations.
    let mut trials = 0usize;
    let mut best: Option<Placed> = None;

    let front_len = bin.front.len().min(max_front);
    for fi in 0..front_len {
        let base = bin.front[fi];
        for &rot in &piece.rots {
            if trials >= max_trials {
                break;
            }
            trials += 1;

            // Align piece bbox min to frontier point to get bottom-left packing.
            // For speed, compute rotated bbox mins from transformed vertices around origin.
            // We'll shift so that bbox_min becomes base.
            let (r_mn, r_mx) = bbox_of_transformed(&piece.poly.v, rot, Pt::new(0.0, 0.0));
            let t = Pt::new(base.x - r_mn.x, base.y - r_mn.y);
            let (mn, mx) = (r_mn.add(t), r_mx.add(t));

            // Quick reject vs stock bbox
            if mn.x < bin.bbox_min.x - 1e-9
                || mn.y < bin.bbox_min.y - 1e-9
                || mx.x > bin.bbox_max.x + 1e-9
                || mx.y > bin.bbox_max.y + 1e-9
            {
                continue;
            }

            // Quick AABB reject vs placed AABBs
            let mut aabb_ok = true;
            for p in &bin.placed {
                if aabb_overlap(mn, mx, p.bbox_min, p.bbox_max) {
                    aabb_ok = false;
                    break;
                }
            }
            if !aabb_ok {
                continue;
            }

            let world = transformed_vertices(&piece.poly.v, rot, t);
            if !poly_inside_stock(&world, stock_poly) {
                continue;
            }

            // Exact overlap check only when AABB says no overlap, so skip (already aabb_ok).
            // But AABB can be false-negative? No; we required no overlap with any AABB, so polygons can't overlap.
            // Still, because of EPS and boundary touch, do a cautious check with nearby boxes? Not needed.
            let placed = Placed {
                piece_idx: piece.idx,
                stock_idx,
                x: t.x,
                y: t.y,
                rot,
                bbox_min: mn,
                bbox_max: mx,
                poly_world: world,
            };

            // Choose best by minimal max(y), then minimal max(x) (bottom-left).
            let score_y = placed.bbox_max.y;
            let score_x = placed.bbox_max.x;
            let is_better = match &best {
                None => true,
                Some(b) => {
                    if score_y < b.bbox_max.y - 1e-12 {
                        true
                    } else if (score_y - b.bbox_max.y).abs() <= 1e-12 {
                        score_x < b.bbox_max.x - 1e-12
                    } else {
                        false
                    }
                }
            };
            if is_better {
                best = Some(placed);
            }
        }
        if trials >= max_trials {
            break;
        }
    }

    if let Some(pl) = best.clone() {
        // Before accepting, ensure no polygon overlaps (in case AABB missed due to float issues)
        for q in &bin.placed {
            if aabb_overlap(pl.bbox_min, pl.bbox_max, q.bbox_min, q.bbox_max) {
                if polys_overlap(&pl.poly_world, &q.poly_world) {
                    return None;
                }
            }
        }
        // Commit: add piece and add new frontier points: right and top of bbox.
        bin.placed.push(best.clone().unwrap());
        let right = Pt::new(pl.bbox_max.x, pl.bbox_min.y);
        let top = Pt::new(pl.bbox_min.x, pl.bbox_max.y);
        bin.front.push_back(right);
        bin.front.push_back(top);
        // Keep frontier size bounded to avoid blowup.
        while bin.front.len() > 4096 {
            bin.front.pop_front();
        }
        return best;
    }
    None
}

fn parse_input() -> (Vec<Pt>, Vec<Poly>) {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_whitespace();
    let sv: usize = it.next().unwrap().parse().unwrap();
    let np: usize = it.next().unwrap().parse().unwrap();
    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock.push(Pt::new(x, y));
    }
    let mut pieces = Vec::with_capacity(np);
    for _ in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut v = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            v.push(Pt::new(x, y));
        }
        pieces.push(make_poly(v));
    }
    (normalize_ccw(stock), pieces)
}

fn main() {
    let (stock_poly, piece_polys) = parse_input();
    let np = piece_polys.len();

    // Precompute stock bbox for initial frontier.
    let sb = stock_bbox(&stock_poly);

    // Prepare pieces with adaptive rotations and sort by decreasing area (FFD-like).
    let mut pieces: Vec<PieceInfo> = piece_polys
        .into_iter()
        .enumerate()
        .map(|(i, p)| {
            let w0 = p.bbox_max.x - p.bbox_min.x;
            let h0 = p.bbox_max.y - p.bbox_min.y;
            let rots = choose_rotations(&p, np);
            PieceInfo {
                idx: i,
                poly: p,
                w0,
                h0,
                rots,
            }
        })
        .collect();

    pieces.sort_by(|a, b| b.poly.area.partial_cmp(&a.poly.area).unwrap_or(Ordering::Equal));

    // Adaptive limits: big NP => fewer checks.
    let (max_front, max_trials) = if np > 4000 {
        (16usize, 128usize)
    } else if np > 1500 {
        (24usize, 256usize)
    } else if np > 400 {
        (48usize, 768usize)
    } else {
        (128usize, 4096usize)
    };

    let mut bins: Vec<StockBin> = Vec::new();
    let mut placements: Vec<Option<(usize, f64, f64, f64)>> = vec![None; np];

    for piece in pieces {
        let mut placed_opt: Option<Placed> = None;

        // Try existing bins first (First-Fit).
        for (bi, bin) in bins.iter_mut().enumerate() {
            if let Some(pl) = try_place_in_bin(&piece, &stock_poly, bin, bi, max_front, max_trials)
            {
                placed_opt = Some(pl);
                break;
            }
        }

        // If not placed, open a new bin and try there.
        if placed_opt.is_none() {
            let mut bin = init_stock_bin(sb);
            let bi = bins.len();
            if let Some(pl) = try_place_in_bin(&piece, &stock_poly, &mut bin, bi, max_front, max_trials)
            {
                placed_opt = Some(pl);
                bins.push(bin);
            } else {
                // Fallback: try centering at stock bbox min (still likely impossible if piece too big)
                // Place at stock bbox min with rot 0; may be invalid but we must output something.
                let t = Pt::new(sb.0.x, sb.0.y);
                placements[piece.idx] = Some((0, t.x, t.y, 0.0));
                continue;
            }
        }

        if let Some(pl) = placed_opt {
            placements[pl.piece_idx] = Some((pl.stock_idx, pl.x, pl.y, pl.rot));
        }
    }

    // Output
    let num_stocks = bins.len().max(1);
    let mut out = String::new();
    out.push_str(&format!("{}\n", num_stocks));
    for i in 0..np {
        if let Some((si, x, y, r)) = placements[i] {
            out.push_str(&format!("{} {:.10} {:.10} {:.10}\n", si, x, y, r));
        } else {
            out.push_str("0 0 0 0\n");
        }
    }
    print!("{}", out);
}
