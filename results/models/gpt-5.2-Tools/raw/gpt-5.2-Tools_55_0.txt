using namespace std;

static inline int clampi(int v, int lo, int hi) { return v < lo ? lo : (v > hi ? hi : v); }

enum Piece : int8_t {
    EMPTY = 0,
    WP = 1, WN = 2, WB = 3, WR = 4, WQ = 5, WK = 6,
    BP = -1, BN = -2, BB = -3, BR = -4, BQ = -5, BK = -6
};

static inline bool isWhite(Piece p) { return p > 0; }
static inline bool isBlack(Piece p) { return p < 0; }
static inline int absPiece(Piece p) { return p < 0 ? -p : p; }

static inline int fileOf(int sq) { return sq & 7; }
static inline int rankOf(int sq) { return sq >> 3; }
static inline bool onBoard(int sq) { return sq >= 0 && sq < 64; }
static inline string sqToStr(int sq) {
    char f = char('a' + fileOf(sq));
    char r = char('1' + rankOf(sq));
    return string() + f + r;
}
static inline int strToSq(const string& s) {
    if (s.size() < 2) return -1;
    int f = s[0] - 'a';
    int r = s[1] - '1';
    if (f < 0 || f > 7 || r < 0 || r > 7) return -1;
    return r * 8 + f;
}

struct Move {
    int from = -1;
    int to = -1;
    Piece promo = EMPTY;       // promotion piece type with color sign of mover
    bool isCapture = false;
    bool isEnPassant = false;
    bool isCastle = false;
};

struct Position {
    array<Piece, 64> b{};
    bool whiteToMove = true;
    bool wk = false, wq = false, bk = false, bq = false;
    int epSq = -1;            // en passant target square (the square behind the pawn that moved 2)
    int halfmove = 0;
    int fullmove = 1;
};

static Piece charToPiece(char c) {
    switch (c) {
    case 'P': return WP; case 'N': return WN; case 'B': return WB; case 'R': return WR; case 'Q': return WQ; case 'K': return WK;
    case 'p': return BP; case 'n': return BN; case 'b': return BB; case 'r': return BR; case 'q': return BQ; case 'k': return BK;
    default: return EMPTY;
    }
}

static bool parseFEN(const string& fen, Position& pos) {
    pos = Position{};
    pos.b.fill(EMPTY);
    istringstream iss(fen);
    string board, stm, castling, ep;
    if (!(iss >> board >> stm >> castling >> ep)) return false;
    iss >> pos.halfmove >> pos.fullmove;

    int sq = 56; // start at a8
    for (char c : board) {
        if (c == '/') { sq -= 16; continue; }
        if (isdigit((unsigned char)c)) {
            sq += (c - '0');
        } else {
            if (!onBoard(sq)) return false;
            pos.b[sq] = charToPiece(c);
            sq++;
        }
    }
    pos.whiteToMove = (stm == "w");
    pos.wk = pos.wq = pos.bk = pos.bq = false;
    if (castling.find('K') != string::npos) pos.wk = true;
    if (castling.find('Q') != string::npos) pos.wq = true;
    if (castling.find('k') != string::npos) pos.bk = true;
    if (castling.find('q') != string::npos) pos.bq = true;
    if (ep == "-") pos.epSq = -1;
    else pos.epSq = strToSq(ep);
    return true;
}

static int kingSquare(const Position& p, bool white) {
    Piece k = white ? WK : BK;
    for (int i = 0; i < 64; i++) if (p.b[i] == k) return i;
    return -1;
}

static bool isSquareAttacked(const Position& p, int sq, bool byWhite) {
    // Pawns
    int r = rankOf(sq), f = fileOf(sq);
    if (byWhite) {
        // white pawns attack up (towards rank increasing)
        if (r > 0) {
            if (f > 0 && p.b[sq - 9] == WP) return true;
            if (f < 7 && p.b[sq - 7] == WP) return true;
        }
    } else {
        if (r < 7) {
            if (f > 0 && p.b[sq + 7] == BP) return true;
            if (f < 7 && p.b[sq + 9] == BP) return true;
        }
    }
    // Knights
    static const int kD[8] = { -17,-15,-10,-6,6,10,15,17 };
    for (int d : kD) {
        int to = sq + d;
        if (!onBoard(to)) continue;
        int df = abs(fileOf(to) - f);
        int dr = abs(rankOf(to) - r);
        if (!((df == 1 && dr == 2) || (df == 2 && dr == 1))) continue;
        Piece pc = p.b[to];
        if (byWhite && pc == WN) return true;
        if (!byWhite && pc == BN) return true;
    }
    // King
    static const int kd[8] = { -9,-8,-7,-1,1,7,8,9 };
    for (int d : kd) {
        int to = sq + d;
        if (!onBoard(to)) continue;
        if (abs(fileOf(to) - f) > 1) continue;
        Piece pc = p.b[to];
        if (byWhite && pc == WK) return true;
        if (!byWhite && pc == BK) return true;
    }
    // Sliding pieces
    auto ray = [&](int step, Piece bishop, Piece rook, Piece queen) -> bool {
        int cur = sq;
        while (true) {
            int nxt = cur + step;
            if (!onBoard(nxt)) break;
            // file wrap check for horizontal/diagonal steps
            int df = abs(fileOf(nxt) - fileOf(cur));
            if (df > 1 && (step == -1 || step == 1 || step == -9 || step == 7 || step == -7 || step == 9)) break;
            cur = nxt;
            Piece pc = p.b[cur];
            if (pc == EMPTY) continue;
            if (byWhite) {
                if (pc == queen) return true;
                if ((step == -9 || step == -7 || step == 7 || step == 9) && pc == bishop) return true;
                if ((step == -8 || step == 8 || step == -1 || step == 1) && pc == rook) return true;
            } else {
                if (pc == Piece(-queen)) return true;
                if ((step == -9 || step == -7 || step == 7 || step == 9) && pc == Piece(-bishop)) return true;
                if ((step == -8 || step == 8 || step == -1 || step == 1) && pc == Piece(-rook)) return true;
            }
            break;
        }
        return false;
    };
    if (ray(-8, WB, WR, WQ)) return true;
    if (ray(8, WB, WR, WQ)) return true;
    if (ray(-1, WB, WR, WQ)) return true;
    if (ray(1, WB, WR, WQ)) return true;
    if (ray(-9, WB, WR, WQ)) return true;
    if (ray(-7, WB, WR, WQ)) return true;
    if (ray(7, WB, WR, WQ)) return true;
    if (ray(9, WB, WR, WQ)) return true;
    return false;
}

static bool inCheck(const Position& p, bool whiteKing) {
    int ks = kingSquare(p, whiteKing);
    if (ks < 0) return false;
    return isSquareAttacked(p, ks, !whiteKing);
}

struct Undo {
    Piece captured = EMPTY;
    int epSqPrev = -1;
    bool wk=false,wq=false,bk=false,bq=false;
    int halfmovePrev=0, fullmovePrev=1;
};

static void doMove(Position& p, const Move& m, Undo& u) {
    u.captured = p.b[m.to];
    u.epSqPrev = p.epSq;
    u.wk = p.wk; u.wq = p.wq; u.bk = p.bk; u.bq = p.bq;
    u.halfmovePrev = p.halfmove; u.fullmovePrev = p.fullmove;

    Piece moving = p.b[m.from];
    p.epSq = -1;

    // halfmove clock
    if (absPiece(moving) == 1 || m.isCapture) p.halfmove = 0;
    else p.halfmove++;

    // move piece
    p.b[m.to] = moving;
    p.b[m.from] = EMPTY;

    // en passant capture
    if (m.isEnPassant) {
        int capSq = m.to + (isWhite(moving) ? -8 : 8);
        u.captured = p.b[capSq];
        p.b[capSq] = EMPTY;
    }
    // promotion
    if (m.promo != EMPTY) {
        p.b[m.to] = m.promo;
    }
    // castling rook move
    if (m.isCastle) {
        if (m.to == strToSq("g1")) { // white O-O
            p.b[strToSq("f1")] = WR;
            p.b[strToSq("h1")] = EMPTY;
        } else if (m.to == strToSq("c1")) { // white O-O-O
            p.b[strToSq("d1")] = WR;
            p.b[strToSq("a1")] = EMPTY;
        } else if (m.to == strToSq("g8")) { // black O-O
            p.b[strToSq("f8")] = BR;
            p.b[strToSq("h8")] = EMPTY;
        } else if (m.to == strToSq("c8")) { // black O-O-O
            p.b[strToSq("d8")] = BR;
            p.b[strToSq("a8")] = EMPTY;
        }
    }
    // update castling rights
    if (moving == WK) { p.wk = p.wq = false; }
    if (moving == BK) { p.bk = p.bq = false; }
    if (m.from == strToSq("h1") || m.to == strToSq("h1")) p.wk = false;
    if (m.from == strToSq("a1") || m.to == strToSq("a1")) p.wq = false;
    if (m.from == strToSq("h8") || m.to == strToSq("h8")) p.bk = false;
    if (m.from == strToSq("a8") || m.to == strToSq("a8")) p.bq = false;

    // set en passant square if double pawn push
    if (absPiece(moving) == 1) {
        int dr = rankOf(m.to) - rankOf(m.from);
        if (dr == 2) p.epSq = m.from + 8;
        else if (dr == -2) p.epSq = m.from - 8;
    }

    // side to move / fullmove
    p.whiteToMove = !p.whiteToMove;
    if (p.whiteToMove) p.fullmove++;
}

static void undoMove(Position& p, const Move& m, const Undo& u) {
    p.whiteToMove = !p.whiteToMove;
    p.epSq = u.epSqPrev;
    p.wk = u.wk; p.wq = u.wq; p.bk = u.bk; p.bq = u.bq;
    p.halfmove = u.halfmovePrev; p.fullmove = u.fullmovePrev;

    Piece moving = p.b[m.to];
    // undo castle rook move first (because moving king is at to)
    if (m.isCastle) {
        if (m.to == strToSq("g1")) { p.b[strToSq("h1")] = WR; p.b[strToSq("f1")] = EMPTY; }
        else if (m.to == strToSq("c1")) { p.b[strToSq("a1")] = WR; p.b[strToSq("d1")] = EMPTY; }
        else if (m.to == strToSq("g8")) { p.b[strToSq("h8")] = BR; p.b[strToSq("f8")] = EMPTY; }
        else if (m.to == strToSq("c8")) { p.b[strToSq("a8")] = BR; p.b[strToSq("d8")] = EMPTY; }
    }
    // undo promotion
    if (m.promo != EMPTY) {
        moving = isWhite(moving) ? WP : BP;
    }
    p.b[m.from] = moving;
    p.b[m.to] = u.captured;
    if (m.isEnPassant) {
        int capSq = m.to + (isWhite(moving) ? -8 : 8);
        p.b[m.to] = EMPTY;
        p.b[capSq] = u.captured;
    }
}

static void addMove(vector<Move>& moves, int from, int to, bool cap, Piece promo = EMPTY, bool ep = false, bool castle = false) {
    Move m;
    m.from = from; m.to = to; m.isCapture = cap; m.promo = promo; m.isEnPassant = ep; m.isCastle = castle;
    moves.push_back(m);
}

static void genPseudoMoves(const Position& p, vector<Move>& moves) {
    moves.clear();
    bool white = p.whiteToMove;
    for (int sq = 0; sq < 64; sq++) {
        Piece pc = p.b[sq];
        if (pc == EMPTY) continue;
        if (white && !isWhite(pc)) continue;
        if (!white && !isBlack(pc)) continue;
        int f = fileOf(sq), r = rankOf(sq);
        int ap = absPiece(pc);
        if (ap == 1) {
            int dir = white ? 8 : -8;
            int startRank = white ? 1 : 6;
            int promRank = white ? 6 : 1;
            int next = sq + dir;
            if (onBoard(next) && p.b[next] == EMPTY) {
                if (r == promRank) {
                    Piece base = white ? WQ : BQ;
                    addMove(moves, sq, next, false, base);
                    addMove(moves, sq, next, false, white ? WR : BR);
                    addMove(moves, sq, next, false, white ? WB : BB);
                    addMove(moves, sq, next, false, white ? WN : BN);
                } else {
                    addMove(moves, sq, next, false);
                    if (r == startRank) {
                        int two = sq + 2 * dir;
                        if (onBoard(two) && p.b[two] == EMPTY) addMove(moves, sq, two, false);
                    }
                }
            }
            // captures
            for (int df : {-1, 1}) {
                int tf = f + df;
                if (tf < 0 || tf > 7) continue;
                int to = sq + dir + df;
                if (!onBoard(to)) continue;
                Piece target = p.b[to];
                if (target != EMPTY && (white ? isBlack(target) : isWhite(target))) {
                    if (r == promRank) {
                        addMove(moves, sq, to, true, white ? WQ : BQ);
                        addMove(moves, sq, to, true, white ? WR : BR);
                        addMove(moves, sq, to, true, white ? WB : BB);
                        addMove(moves, sq, to, true, white ? WN : BN);
                    } else {
                        addMove(moves, sq, to, true);
                    }
                }
                // en passant
                if (p.epSq == to) {
                    // ensure there is an adjacent enemy pawn to capture
                    int capSq = to + (white ? -8 : 8);
                    if (onBoard(capSq) && p.b[capSq] == (white ? BP : WP)) {
                        addMove(moves, sq, to, true, EMPTY, true);
                    }
                }
            }
        } else if (ap == 2) {
            static const int kD[8] = { -17,-15,-10,-6,6,10,15,17 };
            for (int d : kD) {
                int to = sq + d;
                if (!onBoard(to)) continue;
                int df = abs(fileOf(to) - f);
                int dr = abs(rankOf(to) - r);
                if (!((df == 1 && dr == 2) || (df == 2 && dr == 1))) continue;
                Piece t = p.b[to];
                if (t == EMPTY) addMove(moves, sq, to, false);
                else if (white ? isBlack(t) : isWhite(t)) addMove(moves, sq, to, true);
            }
        } else if (ap == 3 || ap == 4 || ap == 5) {
            vector<int> dirs;
            if (ap == 3 || ap == 5) { dirs.insert(dirs.end(), {-9,-7,7,9}); }
            if (ap == 4 || ap == 5) { dirs.insert(dirs.end(), {-8,8,-1,1}); }
            for (int step : dirs) {
                int cur = sq;
                while (true) {
                    int to = cur + step;
                    if (!onBoard(to)) break;
                    int df = abs(fileOf(to) - fileOf(cur));
                    if (df > 1 && (step == -1 || step == 1 || step == -9 || step == 7 || step == -7 || step == 9)) break;
                    cur = to;
                    Piece t = p.b[cur];
                    if (t == EMPTY) {
                        addMove(moves, sq, cur, false);
                    } else {
                        if (white ? isBlack(t) : isWhite(t)) addMove(moves, sq, cur, true);
                        break;
                    }
                }
            }
        } else if (ap == 6) {
            static const int kd[8] = { -9,-8,-7,-1,1,7,8,9 };
            for (int d : kd) {
                int to = sq + d;
                if (!onBoard(to)) continue;
                if (abs(fileOf(to) - f) > 1) continue;
                Piece t = p.b[to];
                if (t == EMPTY) addMove(moves, sq, to, false);
                else if (white ? isBlack(t) : isWhite(t)) addMove(moves, sq, to, true);
            }
            // castling
            if (white) {
                if (sq == strToSq("e1") && !inCheck(p, true)) {
                    if (p.wk && p.b[strToSq("f1")] == EMPTY && p.b[strToSq("g1")] == EMPTY) {
                        if (!isSquareAttacked(p, strToSq("f1"), false) && !isSquareAttacked(p, strToSq("g1"), false)) {
                            addMove(moves, sq, strToSq("g1"), false, EMPTY, false, true);
                        }
                    }
                    if (p.wq && p.b[strToSq("d1")] == EMPTY && p.b[strToSq("c1")] == EMPTY && p.b[strToSq("b1")] == EMPTY) {
                        if (!isSquareAttacked(p, strToSq("d1"), false) && !isSquareAttacked(p, strToSq("c1"), false)) {
                            addMove(moves, sq, strToSq("c1"), false, EMPTY, false, true);
                        }
                    }
                }
            } else {
                if (sq == strToSq("e8") && !inCheck(p, false)) {
                    if (p.bk && p.b[strToSq("f8")] == EMPTY && p.b[strToSq("g8")] == EMPTY) {
                        if (!isSquareAttacked(p, strToSq("f8"), true) && !isSquareAttacked(p, strToSq("g8"), true)) {
                            addMove(moves, sq, strToSq("g8"), false, EMPTY, false, true);
                        }
                    }
                    if (p.bq && p.b[strToSq("d8")] == EMPTY && p.b[strToSq("c8")] == EMPTY && p.b[strToSq("b8")] == EMPTY) {
                        if (!isSquareAttacked(p, strToSq("d8"), true) && !isSquareAttacked(p, strToSq("c8"), true)) {
                            addMove(moves, sq, strToSq("c8"), false, EMPTY, false, true);
                        }
                    }
                }
            }
        }
    }
}

static void genLegalMoves(Position& p, vector<Move>& out) {
    vector<Move> pseudo;
    genPseudoMoves(p, pseudo);
    out.clear();
    out.reserve(pseudo.size());
    bool movingWhite = p.whiteToMove;
    for (const auto& m : pseudo) {
        Undo u;
        doMove(p, m, u);
        bool ok = !inCheck(p, movingWhite);
        undoMove(p, m, u);
        if (ok) out.push_back(m);
    }
}

// Simple evaluation (centipawns, positive for white)
static int pieceValue(int ap) {
    switch (ap) {
    case 1: return 100;
    case 2: return 320;
    case 3: return 330;
    case 4: return 500;
    case 5: return 900;
    case 6: return 0;
    default: return 0;
    }
}

static inline int pstFor(Piece pc, int sq) {
    bool w = isWhite(pc);
    int idx = w ? sq : (56 ^ (sq & 7)) + (7 - rankOf(sq)) * 8; // mirror to use white tables
    int ap = absPiece(pc);
    switch (ap) {
    case 1: return PST_PAWN[idx];
    case 2: return PST_KNIGHT[idx];
    case 3: return PST_BISHOP[idx];
    case 4: return PST_ROOK[idx];
    case 5: return PST_QUEEN[idx];
    case 6: return PST_KING_MG[idx];
    default: return 0;
    }
}

static int evaluate(const Position& p) {
    int score = 0;
    int material = 0;
    for (int sq = 0; sq < 64; sq++) {
        Piece pc = p.b[sq];
        if (pc == EMPTY) continue;
        int ap = absPiece(pc);
        int val = pieceValue(ap);
        material += val;
        int ps = pstFor(pc, sq);
        if (isWhite(pc)) score += val + ps;
        else score -= val + ps;
    }
    // small mobility bonus
    Position tmp = p;
    vector<Move> moves;
    genLegalMoves(tmp, moves);
    int mob = (int)moves.size();
    score += (p.whiteToMove ? mob : -mob);
    // tempo
    score += (p.whiteToMove ? 10 : -10);
    return score;
}

static int mvvLvaScore(Piece attacker, Piece victim) {
    if (victim == EMPTY) return 0;
    int v = pieceValue(absPiece(victim));
    int a = pieceValue(absPiece(attacker));
    return 10 * v - a;
}

static void orderMoves(const Position& p, vector<Move>& moves) {
    stable_sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) {
        Piece pa = p.b[a.from];
        Piece pb = p.b[b.from];
        int sa = 0, sb = 0;
        if (a.isCapture) sa += 100000 + mvvLvaScore(pa, a.isEnPassant ? (isWhite(pa) ? BP : WP) : p.b[a.to]);
        if (b.isCapture) sb += 100000 + mvvLvaScore(pb, b.isEnPassant ? (isWhite(pb) ? BP : WP) : p.b[b.to]);
        if (a.promo != EMPTY) sa += 90000 + pieceValue(absPiece(a.promo));
        if (b.promo != EMPTY) sb += 90000 + pieceValue(absPiece(b.promo));
        // prefer centralization (cheap heuristic)
        auto cent = [](int sq) {
            int f = fileOf(sq), r = rankOf(sq);
            return 7 - (abs(f - 3) + abs(r - 3));
        };
        sa += cent(a.to);
        sb += cent(b.to);
        return sa > sb;
    });
}

struct SearchCtx {
    chrono::steady_clock::time_point deadline;
    atomic<bool> stop{false};
    int nodes = 0;
    int maxNodes = 0; // 0 means no limit
};

static inline bool timeUp(const SearchCtx& ctx) {
    return ctx.stop.load(memory_order_relaxed) || chrono::steady_clock::now() >= ctx.deadline;
}

static int quiescence(Position& p, SearchCtx& ctx, int alpha, int beta) {
    if (timeUp(ctx)) return evaluate(p);
    ctx.nodes++;
    int stand = evaluate(p);
    if (stand >= beta) return beta;
    if (stand > alpha) alpha = stand;

    vector<Move> moves;
    genPseudoMoves(p, moves);
    // only captures/promotions
    vector<Move> caps;
    caps.reserve(moves.size());
    for (auto& m : moves) if (m.isCapture || m.promo != EMPTY) caps.push_back(m);
    orderMoves(p, caps);
    bool moverWhite = p.whiteToMove;
    for (auto& m : caps) {
        Undo u;
        doMove(p, m, u);
        if (inCheck(p, moverWhite)) { undoMove(p, m, u); continue; }
        int score = -quiescence(p, ctx, -beta, -alpha);
        undoMove(p, m, u);
        if (timeUp(ctx)) break;
        if (score >= beta) return beta;
        if (score > alpha) alpha = score;
    }
    return alpha;
}

static int negamax(Position& p, SearchCtx& ctx, int depth, int alpha, int beta) {
    if (timeUp(ctx)) return evaluate(p);
    ctx.nodes++;
    if (depth <= 0) return quiescence(p, ctx, alpha, beta);

    vector<Move> moves;
    genLegalMoves(p, moves);
    if (moves.empty()) {
        bool side = p.whiteToMove;
        if (inCheck(p, side)) return -100000 + (3 - depth); // mate distance
        return 0; // stalemate
    }
    orderMoves(p, moves);
    int best = -10000000;
    for (auto& m : moves) {
        Undo u;
        doMove(p, m, u);
        int score = -negamax(p, ctx, depth - 1, -beta, -alpha);
        undoMove(p, m, u);
        if (timeUp(ctx)) break;
        if (score > best) best = score;
        if (score > alpha) alpha = score;
        if (alpha >= beta) break;
    }
    return best;
}

static bool isLegalMove(Position& p, const Move& m) {
    Undo u;
    bool moverWhite = p.whiteToMove;
    doMove(p, m, u);
    bool ok = !inCheck(p, moverWhite);
    undoMove(p, m, u);
    return ok;
}

static string moveToNotation(Position& p, const Move& m) {
    // Castling
    if (m.isCastle) {
        if (m.to == strToSq("g1") || m.to == strToSq("g8")) return "O-O";
        return "O-O-O";
    }
    Piece moving = p.b[m.from];
    char pieceChar;
    int ap = absPiece(moving);
    if (ap == 1) pieceChar = 'p';
    else if (ap == 2) pieceChar = 'N';
    else if (ap == 3) pieceChar = 'B';
    else if (ap == 4) pieceChar = 'R';
    else if (ap == 5) pieceChar = 'Q';
    else pieceChar = 'K';
    string s;
    s += pieceChar;
    s += sqToStr(m.from);
    if (m.isCapture) s += "x";
    s += sqToStr(m.to);
    if (m.promo != EMPTY) {
        s += "/";
        int ap2 = absPiece(m.promo);
        char pc2 = 'Q';
        if (ap2 == 2) pc2 = 'N';
        else if (ap2 == 3) pc2 = 'B';
        else if (ap2 == 4) pc2 = 'R';
        else if (ap2 == 5) pc2 = 'Q';
        s += pc2;
    }
    if (m.isEnPassant) s += "ep";

    // Check / mate markers: make move, test
    Undo u;
    bool moverWhite = p.whiteToMove;
    doMove(p, m, u);
    vector<Move> replies;
    genLegalMoves(p, replies);
    bool givesCheck = inCheck(p, !moverWhite);
    bool mate = givesCheck && replies.empty();
    undoMove(p, m, u);
    if (mate) s += "#";
    else if (givesCheck) s += "+";
    return s;
}

static Move pickFallback(Position& p) {
    vector<Move> moves;
    genLegalMoves(p, moves);
    if (moves.empty()) return Move{};
    orderMoves(p, moves);
    return moves[0];
}

static Move searchBestMove(Position root, double timeoutSeconds) {
    SearchCtx ctx;
    auto start = chrono::steady_clock::now();
    // leave some margin
    auto ms = (int)(timeoutSeconds * 1000.0);
    ms = max(ms, 1);
    int margin = clampi(ms / 30, 20, 150); // 3.3% or min 20ms
    ctx.deadline = start + chrono::milliseconds(max(1, ms - margin));

    Position p = root;
    vector<Move> moves;
    genLegalMoves(p, moves);
    if (moves.empty()) return Move{};
    orderMoves(p, moves);

    Move bestMove = moves[0];
    int bestScore = -10000000;
    int depth = 1;

    // simple aspiration window
    int lastScore = 0;
    while (!timeUp(ctx)) {
        int alpha = -10000000, beta = 10000000;
        if (depth >= 3) { alpha = lastScore - 50; beta = lastScore + 50; }

        Move localBest = bestMove;
        int localBestScore = -10000000;

        // root search
        for (size_t i = 0; i < moves.size(); i++) {
            if (timeUp(ctx)) break;
            Move m = moves[i];
            Undo u;
            doMove(p, m, u);
            int score = -negamax(p, ctx, depth - 1, -beta, -alpha);
            undoMove(p, m, u);
            if (timeUp(ctx)) break;
            if (score > localBestScore) {
                localBestScore = score;
                localBest = m;
            }
            if (score > alpha) alpha = score;
            if (alpha >= beta) {
                // aspiration fail high, widen and restart depth once
                if (depth >= 3 && beta != 10000000) {
                    alpha = -10000000; beta = 10000000;
                    localBestScore = -10000000;
                    localBest = bestMove;
                    i = (size_t)-1;
                    continue;
                }
                break;
            }
        }

        if (!timeUp(ctx)) {
            bestMove = localBest;
            bestScore = localBestScore;
            lastScore = bestScore;
            // move ordering: put best move first
            auto it = find_if(moves.begin(), moves.end(), [&](const Move& m) {
                return m.from == bestMove.from && m.to == bestMove.to && m.promo == bestMove.promo && m.isCastle == bestMove.isCastle && m.isEnPassant == bestMove.isEnPassant;
            });
            if (it != moves.end()) {
                Move tmpm = *it;
                moves.erase(it);
                moves.insert(moves.begin(), tmpm);
            }
            depth++;
            if (depth > 12) depth = 12; // cap
        } else {
            break;
        }
    }
    (void)bestScore;
    return bestMove;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    string line1;
    if (!std::getline(cin, line1)) return 0;
    while (!line1.empty() && (line1.back() == '\r' || line1.back() == '\n')) line1.pop_back();
    double timeout = 5.0;
    try { timeout = stod(line1); } catch (...) { timeout = 5.0; }
    string fen;
    if (!std::getline(cin, fen)) return 0;
    if (!fen.empty() && fen.back() == '\r') fen.pop_back();

    Position p;
    if (!parseFEN(fen, p)) {
        cout << "O-O\n";
        return 0;
    }

    Move best = searchBestMove(p, timeout);
    // Ensure legal fallback
    if (best.from < 0) {
        cout << "O-O\n";
        return 0;
    }
    // print notation
    cout << moveToNotation(p, best) << "\n";
    return 0;
}
