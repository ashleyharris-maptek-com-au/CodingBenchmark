const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input.length ? JSON.parse(input) : {};

const grid = data.grid || [];
const H = grid.length;
const W = H ? grid[0].length : 0;

function inBounds(x, y) {
  return x >= 0 && x < W && y >= 0 && y < H;
}
function isOpen(x, y) {
  return inBounds(x, y) && grid[y][x] !== '#';
}
function cellId(x, y) {
  return y * W + x;
}

const robots = (data.robots || []).slice();
const jobs = (data.jobs || []).slice();
const R = robots.length;
const J = jobs.length;

// If robot ids are not 0..R-1, output still expects "array index is robot id".
// We'll build an array sized to maxId+1 and place robots by id.
let maxRobotId = -1;
for (const r of robots) if (typeof r.id === 'number') maxRobotId = Math.max(maxRobotId, r.id);
const outSize = Math.max(R - 1, maxRobotId) + 1;

// Create robot list aligned by id; missing ids become null and get empty assignment arrays.
const robotsById = new Array(outSize).fill(null);
for (const r of robots) {
  if (r && typeof r.id === 'number' && r.id >= 0 && r.id < outSize) robotsById[r.id] = r;
}

// Group jobs by priority (higher priority first). Tie-break by id for determinism.
jobs.sort((a, b) => {
  const pa = (a && typeof a.priority === 'number') ? a.priority : 0;
  const pb = (b && typeof b.priority === 'number') ? b.priority : 0;
  if (pa !== pb) return pb - pa;
  const ia = (a && typeof a.id === 'number') ? a.id : String(a.id);
  const ib = (b && typeof b.id === 'number') ? b.id : String(b.id);
  if (ia < ib) return -1;
  if (ia > ib) return 1;
  return 0;
});

// Map job index by location for BFS queries
const jobLocs = new Array(J);
for (let i = 0; i < J; i++) {
  const j = jobs[i];
  jobLocs[i] = { x: j.x | 0, y: j.y | 0 };
}

// Precompute component ids so unreachable jobs can be handled deterministically.
const compId = new Int32Array(W * H);
compId.fill(-1);
let compCount = 0;
const qx = new Int32Array(W * H);
const qy = new Int32Array(W * H);
const dirs = [1, 0, -1, 0, 0, 1, 0, -1];

for (let y = 0; y < H; y++) {
  for (let x = 0; x < W; x++) {
    if (!isOpen(x, y)) continue;
    const id = cellId(x, y);
    if (compId[id] !== -1) continue;
    const cid = compCount++;
    let head = 0, tail = 0;
    qx[tail] = x; qy[tail] = y; tail++;
    compId[id] = cid;
    while (head < tail) {
      const cx = qx[head], cy = qy[head]; head++;
      for (let k = 0; k < 4; k++) {
        const nx = cx + dirs[k * 2];
        const ny = cy + dirs[k * 2 + 1];
        if (!isOpen(nx, ny)) continue;
        const nid = cellId(nx, ny);
        if (compId[nid] !== -1) continue;
        compId[nid] = cid;
        qx[tail] = nx; qy[tail] = ny; tail++;
      }
    }
  }
}

function getComp(x, y) {
  if (!isOpen(x, y)) return -1;
  return compId[cellId(x, y)];
}

// Multi-target BFS from a robot position; returns distance array for jobs in same component.
// Implemented with typed arrays and stamping to avoid reallocations.
const seen = new Int32Array(W * H);
let stamp = 1;
const dist = new Int32Array(W * H);

function distancesToJobsFrom(x0, y0, jobIndices, maxExpandDist) {
  // jobIndices: array of job indices to consider (sorted order already)
  // Returns Map jobIndex->dist (only reachable and <= maxExpandDist if provided)
  if (!isOpen(x0, y0) || W === 0 || H === 0) return new Map();
  const baseComp = getComp(x0, y0);
  if (baseComp === -1) return new Map();

  const remaining = new Int32Array(jobIndices.length);
  let remCount = 0;
  for (let i = 0; i < jobIndices.length; i++) {
    const ji = jobIndices[i];
    const loc = jobLocs[ji];
    if (getComp(loc.x, loc.y) === baseComp) remaining[remCount++] = ji;
  }
  if (remCount === 0) return new Map();

  if (stamp === 0x7fffffff) {
    seen.fill(0);
    stamp = 1;
  } else {
    stamp++;
  }

  let head = 0, tail = 0;
  qx[tail] = x0; qy[tail] = y0; tail++;
  const startId = cellId(x0, y0);
  seen[startId] = stamp;
  dist[startId] = 0;

  // For fast check whether a cell is a remaining target, build a map from cellId -> list of job indices.
  // But for scaling, avoid full-grid maps; use a temporary hash map over remaining jobs.
  const targetMap = new Map();
  for (let i = 0; i < remCount; i++) {
    const ji = remaining[i];
    const loc = jobLocs[ji];
    const tid = cellId(loc.x, loc.y);
    let arr = targetMap.get(tid);
    if (!arr) { arr = []; targetMap.set(tid, arr); }
    arr.push(ji);
  }

  const result = new Map();
  let targetsLeft = remCount;
  while (head < tail) {
    const cx = qx[head], cy = qy[head]; head++;
    const cid = cellId(cx, cy);
    const cd = dist[cid];
    if (maxExpandDist != null && cd > maxExpandDist) continue;

    const arr = targetMap.get(cid);
    if (arr) {
      for (const ji of arr) {
        if (!result.has(ji)) {
          result.set(ji, cd);
          targetsLeft--;
        }
      }
      targetMap.delete(cid);
      if (targetsLeft <= 0) break;
    }

    const nd = cd + 1;
    for (let k = 0; k < 4; k++) {
      const nx = cx + dirs[k * 2];
      const ny = cy + dirs[k * 2 + 1];
      if (!isOpen(nx, ny)) continue;
      const nid = cellId(nx, ny);
      if (seen[nid] === stamp) continue;
      seen[nid] = stamp;
      dist[nid] = nd;
      qx[tail] = nx; qy[tail] = ny; tail++;
    }
  }
  return result;
}

// Deterministic greedy assignment:
// Process jobs in (priority desc, id asc) order, assign each to the robot that can reach it with minimum
// (currentPos -> job) shortest-path distance. Tie-breakers: smaller distance, then smaller robot id.
// After assignment, robot position updates to job location (so travel accumulates).
const assignments = new Array(outSize);
for (let i = 0; i < outSize; i++) assignments[i] = [];

// Track robot current positions; null robots get ignored in distance calculations.
const robotPos = new Array(outSize);
for (let rid = 0; rid < outSize; rid++) {
  const r = robotsById[rid];
  if (r && typeof r.x === 'number' && typeof r.y === 'number') robotPos[rid] = { x: r.x | 0, y: r.y | 0 };
  else robotPos[rid] = null;
}

// Pre-split jobs into indices (0..J-1 in sorted order)
const jobIndicesAll = new Array(J);
for (let i = 0; i < J; i++) jobIndicesAll[i] = i;

// For scaling, do BFS per robot per batch of jobs rather than per job.
// We'll process jobs in chunks, recomputing per-robot distances to remaining jobs each chunk.
const unassigned = new Array(J).fill(true);
let remainingJobs = J;

function nextUnassignedIndices(limit) {
  const arr = [];
  for (let i = 0; i < J && arr.length < limit; i++) if (unassigned[i]) arr.push(i);
  return arr;
}

// Choose chunk size to balance BFS target-set overhead; grows with job count.
let chunkSize = 256;
if (J > 5000) chunkSize = 512;
if (J > 20000) chunkSize = 1024;

while (remainingJobs > 0) {
  const chunk = nextUnassignedIndices(chunkSize);
  if (chunk.length === 0) break;

  // For each robot, compute distances to chunk jobs in one BFS.
  const robotDists = new Array(outSize);
  for (let rid = 0; rid < outSize; rid++) {
    const pos = robotPos[rid];
    if (!pos) { robotDists[rid] = null; continue; }
    robotDists[rid] = distancesToJobsFrom(pos.x, pos.y, chunk, null);
  }

  // Assign jobs in chunk order (already sorted globally); per job choose best robot.
  for (const ji of chunk) {
    if (!unassigned[ji]) continue;
    const job = jobs[ji];
    let bestRid = -1;
    let bestDist = 0x7fffffff;
    for (let rid = 0; rid < outSize; rid++) {
      const m = robotDists[rid];
      if (!m) continue;
      const d = m.get(ji);
      if (d == null) continue;
      if (d < bestDist || (d === bestDist && rid < bestRid)) {
        bestDist = d;
        bestRid = rid;
      }
    }

    if (bestRid === -1) {
      // Unreachable from all robots: assign deterministically to smallest existing robot id (or 0).
      // This keeps "every job exactly once" even on disconnected grids.
      for (let rid = 0; rid < outSize; rid++) {
        if (robotPos[rid]) { bestRid = rid; break; }
      }
      if (bestRid === -1) bestRid = 0;
    }

    assignments[bestRid].push(job.id);
    unassigned[ji] = false;
    remainingJobs--;
    // Update robot position to the job location to reflect travel for subsequent assignments.
    if (robotPos[bestRid]) {
      robotPos[bestRid] = { x: job.x | 0, y: job.y | 0 };
    } else {
      robotPos[bestRid] = { x: job.x | 0, y: job.y | 0 };
    }
  }
}

process.stdout.write(JSON.stringify({ assignments }));
