#include <cstdint>

// N-D Snake solver / player.
// Strategy (dimension-agnostic):
// 1) Prefer shortest safe path to nearest food using A* (Manhattan heuristic).
// 2) Validate safety by ensuring after taking the path, the head can still reach the tail
//    in the resulting state (tail moves unless food eaten) via BFS.
// 3) If no safe food path, follow the tail (A* to current tail cell) to keep moving.
// 4) If that fails, take any safe move; otherwise output a dummy move.
//
// Notes:
// - Works for D up to 20 and beyond as long as total cells is manageable. For huge spaces,
//   search is bounded by iteration limits and falls back to greedy safe moves.
// - Input provides only head position and length, not body coordinates. We therefore assume
//   the initial snake occupies only the head cell (a common benchmark simplification) and
//   grow the body from movement history. This keeps the player valid during simulation.

using std::int64_t;

struct VecHash {
  size_t operator()(const std::vector<int>& v) const noexcept {
    // FNV-1a-ish
    size_t h = 1469598103934665603ull;
    for (int x : v) {
      h ^= (size_t)(uint32_t)(x + 0x9e3779b9);
      h *= 1099511628211ull;
    }
    return h;
  }
};

static inline int64_t manhattan(const std::vector<int>& a, const std::vector<int>& b) {
  int64_t s = 0;
  for (size_t i = 0; i < a.size(); ++i) s += std::llabs((int64_t)a[i] - (int64_t)b[i]);
  return s;
}

static inline bool in_bounds(const std::vector<int>& p, const std::vector<int>& bounds) {
  for (size_t i = 0; i < p.size(); ++i) {
    if (p[i] < 0 || p[i] >= bounds[i]) return false;
  }
  return true;
}

struct NDGame {
  int D = 0;
  int max_turns = 0;
  std::vector<int> bounds;

  // snake body: front = head, back = tail
  std::deque<std::vector<int>> snake;
  std::unordered_set<std::vector<int>, VecHash> snake_set;

  std::vector<std::vector<int>> foods;
  std::unordered_set<std::vector<int>, VecHash> food_set;

  std::unordered_set<std::vector<int>, VecHash> obstacles;

  std::mt19937_64 rng{123456789};
};

struct NodeInfo {
  int64_t g = 0;
  int64_t f = 0;
  std::vector<int> parent;
  bool has_parent = false;
};

// BFS reachability with dynamic tail allowance:
// If allow_tail == true, the target tail cell is considered free (because it moves away).
static bool bfs_reachable(const std::vector<int>& start,
                          const std::vector<int>& goal,
                          const std::vector<int>& bounds,
                          const std::unordered_set<std::vector<int>, VecHash>& blocked,
                          const std::vector<int>* allow_cell) {
  if (start == goal) return true;
  std::deque<std::vector<int>> q;
  std::unordered_set<std::vector<int>, VecHash> vis;
  q.push_back(start);
  vis.insert(start);
  const int D = (int)bounds.size();
  while (!q.empty()) {
    auto cur = std::move(q.front());
    q.pop_front();
    for (int axis = 0; axis < D; ++axis) {
      for (int dir : {-1, +1}) {
        auto nxt = cur;
        nxt[axis] += dir;
        if (!in_bounds(nxt, bounds)) continue;
        if (vis.find(nxt) != vis.end()) continue;
        if (blocked.find(nxt) != blocked.end()) {
          if (!allow_cell || nxt != *allow_cell) continue;
        }
        if (nxt == goal) return true;
        vis.insert(nxt);
        q.push_back(nxt);
      }
    }
    // Mild cap to avoid runaway in extreme spaces
    if (vis.size() > 200000) return false;
  }
  return false;
}

// A* pathfinder from start to goal on grid with obstacles and snake occupancy.
// Returns path as sequence of positions INCLUDING start and goal, or empty if not found.
static std::vector<std::vector<int>> astar_path(const std::vector<int>& start,
                                                const std::vector<int>& goal,
                                                const std::vector<int>& bounds,
                                                const std::unordered_set<std::vector<int>, VecHash>& blocked,
                                                const std::vector<int>* allow_cell,
                                                size_t expand_limit) {
  if (start == goal) return {start};

  struct PQItem {
    int64_t f;
    int64_t g;
    std::vector<int> pos;
  };
  struct Cmp {
    bool operator()(const PQItem& a, const PQItem& b) const { return a.f > b.f; }
  };

  std::priority_queue<PQItem, std::vector<PQItem>, Cmp> pq;
  std::unordered_map<std::vector<int>, NodeInfo, VecHash> best;

  NodeInfo si;
  si.g = 0;
  si.f = manhattan(start, goal);
  best[start] = si;
  pq.push({si.f, 0, start});

  const int D = (int)bounds.size();
  size_t expands = 0;

  while (!pq.empty()) {
    auto cur = pq.top();
    pq.pop();

    auto it = best.find(cur.pos);
    if (it == best.end()) continue;
    if (cur.g != it->second.g) continue; // stale

    if (cur.pos == goal) {
      // reconstruct
      std::vector<std::vector<int>> path;
      std::vector<int> p = goal;
      while (true) {
        path.push_back(p);
        auto jt = best.find(p);
        if (jt == best.end() || !jt->second.has_parent) break;
        p = jt->second.parent;
      }
      std::reverse(path.begin(), path.end());
      return path;
    }

    if (++expands > expand_limit) break;

    for (int axis = 0; axis < D; ++axis) {
      for (int dir : {-1, +1}) {
        auto nxt = cur.pos;
        nxt[axis] += dir;
        if (!in_bounds(nxt, bounds)) continue;
        if (blocked.find(nxt) != blocked.end()) {
          if (!allow_cell || nxt != *allow_cell) continue;
        }
        int64_t ng = cur.g + 1;
        auto kt = best.find(nxt);
        if (kt == best.end() || ng < kt->second.g) {
          NodeInfo ni;
          ni.g = ng;
          ni.f = ng + manhattan(nxt, goal);
          ni.parent = cur.pos;
          ni.has_parent = true;
          best[nxt] = std::move(ni);
          pq.push({best[nxt].f, ng, nxt});
        }
      }
    }
  }
  return {};
}

static std::pair<int,int> diff_to_move(const std::vector<int>& a, const std::vector<int>& b) {
  // returns (axis, dir) such that b = a + dir*e_axis
  for (size_t i = 0; i < a.size(); ++i) {
    int d = b[i] - a[i];
    if (d == 1) return {(int)i, +1};
    if (d == -1) return {(int)i, -1};
  }
  return {0, +1};
}

static std::vector<std::pair<int,int>> moves_from_path(const std::vector<std::vector<int>>& path) {
  std::vector<std::pair<int,int>> moves;
  for (size_t i = 1; i < path.size(); ++i) moves.push_back(diff_to_move(path[i-1], path[i]));
  return moves;
}

static bool is_cell_blocked(const NDGame& g, const std::vector<int>& cell, const std::vector<int>* allow_tail) {
  if (!in_bounds(cell, g.bounds)) return true;
  if (g.obstacles.find(cell) != g.obstacles.end()) return true;
  if (g.snake_set.find(cell) != g.snake_set.end()) {
    if (!allow_tail || cell != *allow_tail) return true;
  }
  return false;
}

static bool apply_move(NDGame& g, int axis, int dir) {
  auto head = g.snake.front();
  head[axis] += dir;
  if (!in_bounds(head, g.bounds)) return false;
  if (g.obstacles.find(head) != g.obstacles.end()) return false;

  bool eats = (g.food_set.find(head) != g.food_set.end());
  // collision with body is allowed only if stepping into current tail and tail will move away (no eat)
  const auto& tail = g.snake.back();
  bool stepping_into_tail = (head == tail);
  if (g.snake_set.find(head) != g.snake_set.end()) {
    if (!(stepping_into_tail && !eats)) return false;
  }

  g.snake.push_front(head);
  g.snake_set.insert(head);
  if (eats) {
    g.food_set.erase(head);
    // keep tail (grow)
  } else {
    auto old_tail = g.snake.back();
    g.snake.pop_back();
    g.snake_set.erase(old_tail);
  }
  return true;
}

// Safety check for a candidate path: simulate it on a copy, then ensure head can reach tail.
static bool path_is_safe(const NDGame& g, const std::vector<std::vector<int>>& path) {
  if (path.size() < 2) return false;
  NDGame sim = g;
  for (size_t i = 1; i < path.size(); ++i) {
    auto mv = diff_to_move(path[i-1], path[i]);
    if (!apply_move(sim, mv.first, mv.second)) return false;
  }
  auto head = sim.snake.front();
  auto tail = sim.snake.back();
  // Allow tail cell as free in reachability (it can move on subsequent step).
  std::unordered_set<std::vector<int>, VecHash> blocked = sim.obstacles;
  // add snake occupancy
  for (const auto& c : sim.snake) blocked.insert(c);
  return bfs_reachable(head, tail, sim.bounds, blocked, &tail);
}

static std::vector<std::pair<int,int>> plan_to_food(const NDGame& g) {
  if (g.food_set.empty()) return {};
  // Choose nearest food by Manhattan distance.
  std::vector<int> head = g.snake.front();
  std::vector<int> best_food;
  int64_t best_d = std::numeric_limits<int64_t>::max();
  for (const auto& f : g.food_set) {
    int64_t d = manhattan(head, f);
    if (d < best_d) {
      best_d = d;
      best_food = f;
    }
  }
  if (best_food.empty()) return {};

  // Build blocked set. Allow stepping into current tail because it moves if not eating.
  const auto& tail = g.snake.back();
  std::unordered_set<std::vector<int>, VecHash> blocked = g.obstacles;
  for (const auto& c : g.snake) blocked.insert(c);

  // Expansion limit scales with snake length and dimensionality; keep bounded.
  size_t limit = 20000;
  limit += (size_t)g.D * 2000;
  limit += (size_t)g.snake.size() * 200;
  // In very small grids, allow more.
  if (g.D <= 4) limit = std::max<size_t>(limit, 60000);

  auto path = astar_path(head, best_food, g.bounds, blocked, &tail, limit);
  if (path.empty()) return {};
  if (!path_is_safe(g, path)) return {};
  return moves_from_path(path);
}

static std::vector<std::pair<int,int>> plan_follow_tail(const NDGame& g) {
  if (g.snake.empty()) return {};
  std::vector<int> head = g.snake.front();
  std::vector<int> tail = g.snake.back();
  const auto& allow = tail;

  std::unordered_set<std::vector<int>, VecHash> blocked = g.obstacles;
  for (const auto& c : g.snake) blocked.insert(c);

  size_t limit = 15000 + (size_t)g.D * 1500 + (size_t)g.snake.size() * 100;
  if (g.D <= 4) limit = std::max<size_t>(limit, 50000);

  auto path = astar_path(head, tail, g.bounds, blocked, &allow, limit);
  if (path.size() < 2) return {};
  // following tail is usually safe; still quick-check first step legality
  auto mv = diff_to_move(path[0], path[1]);
  return {mv};
}

static std::pair<int,int> pick_any_safe_move(const NDGame& g) {
  std::vector<int> head = g.snake.front();
  const auto& tail = g.snake.back();
  // Randomize axes to avoid loops.
  std::vector<int> axes(g.D);
  for (int i = 0; i < g.D; ++i) axes[i] = i;
  // copy rng since g is const
  std::mt19937_64 rng(0xC0FFEE);
  std::shuffle(axes.begin(), axes.end(), rng);
  for (int axis : axes) {
    for (int dir : {+1, -1}) {
      auto nxt = head;
      nxt[axis] += dir;
      bool eats = (g.food_set.find(nxt) != g.food_set.end());
      // allow stepping into tail only if not eating
      const std::vector<int>* allow = (!eats ? &tail : nullptr);
      if (!is_cell_blocked(g, nxt, allow)) {
        return {axis, dir};
      }
    }
  }
  return {0, +1};
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  NDGame g;
  if (!(std::cin >> g.D >> g.max_turns)) return 0;
  g.bounds.resize(g.D);
  for (int i = 0; i < g.D; ++i) std::cin >> g.bounds[i];

  int snake_len = 0;
  std::vector<int> head(g.D);
  std::cin >> snake_len;
  for (int i = 0; i < g.D; ++i) std::cin >> head[i];

  // Initialize snake as just the head; length is tracked via growth when eating.
  // This avoids illegal assumptions about unknown initial body layout.
  g.snake.clear();
  g.snake.push_back(head);
  g.snake_set.clear();
  g.snake_set.insert(head);

  int F = 0;
  std::cin >> F;
  g.foods.resize(F, std::vector<int>(g.D));
  for (int i = 0; i < F; ++i) {
    for (int d = 0; d < g.D; ++d) std::cin >> g.foods[i][d];
    g.food_set.insert(g.foods[i]);
  }

  int O = 0;
  std::cin >> O;
  for (int i = 0; i < O; ++i) {
    std::vector<int> ob(g.D);
    for (int d = 0; d < g.D; ++d) std::cin >> ob[d];
    g.obstacles.insert(ob);
  }

  // Plan queue of moves; replan as needed.
  std::deque<std::pair<int,int>> plan;

  for (int turn = 0; turn < g.max_turns; ++turn) {
    if (g.food_set.empty()) break;

    if (plan.empty()) {
      auto to_food = plan_to_food(g);
      if (!to_food.empty()) {
        for (auto mv : to_food) plan.push_back(mv);
      } else {
        auto to_tail = plan_follow_tail(g);
        for (auto mv : to_tail) plan.push_back(mv);
      }
    }

    std::pair<int,int> mv;
    if (!plan.empty()) {
      mv = plan.front();
      plan.pop_front();
    } else {
      mv = pick_any_safe_move(g);
    }

    // Ensure the chosen move is immediately legal; otherwise fallback.
    NDGame tmp = g;
    if (!apply_move(tmp, mv.first, mv.second)) {
      mv = pick_any_safe_move(g);
      // if still illegal, output something and exit
      NDGame tmp2 = g;
      if (!apply_move(tmp2, mv.first, mv.second)) {
        std::cout << 0 << " " << 1 << "\n";
        std::cout.flush();
        return 0;
      }
      plan.clear();
    } else {
      // apply to real game state
      apply_move(g, mv.first, mv.second);
    }

    std::cout << mv.first << " " << mv.second << "\n";
    std::cout.flush();
  }

  return 0;
}
