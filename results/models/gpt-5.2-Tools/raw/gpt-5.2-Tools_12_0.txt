using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

public static class Program
{
    // Scalable 3D bin packing for repeated identical polyhedra.
    // Uses fast AABB-based packing with an adaptive orientation search budget.

    private struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
    }

    private struct Quat
    {
        public double W, X, Y, Z;
        public Quat(double w, double x, double y, double z) { W = w; X = x; Y = y; Z = z; }

        public Quat Normalized()
        {
            double n = Math.Sqrt(W * W + X * X + Y * Y + Z * Z);
            if (n <= 0) return new Quat(1, 0, 0, 0);
            double inv = 1.0 / n;
            return new Quat(W * inv, X * inv, Y * inv, Z * inv);
        }

        public static Quat FromAxisAngle(Vec3 axis, double radians)
        {
            double len = Math.Sqrt(axis.X * axis.X + axis.Y * axis.Y + axis.Z * axis.Z);
            if (len <= 0) return new Quat(1, 0, 0, 0);
            double inv = 1.0 / len;
            double ax = axis.X * inv, ay = axis.Y * inv, az = axis.Z * inv;
            double half = 0.5 * radians;
            double s = Math.Sin(half);
            return new Quat(Math.Cos(half), ax * s, ay * s, az * s).Normalized();
        }

        public Vec3 Rotate(Vec3 v)
        {
            double qw = W, qx = X, qy = Y, qz = Z;
            double vx = v.X, vy = v.Y, vz = v.Z;

            double tx = 2.0 * (qy * vz - qz * vy);
            double ty = 2.0 * (qz * vx - qx * vz);
            double tz = 2.0 * (qx * vy - qy * vx);

            double vpx = vx + qw * tx + (qy * tz - qz * ty);
            double vpy = vy + qw * ty + (qz * tx - qx * tz);
            double vpz = vz + qw * tz + (qx * ty - qy * tx);
            return new Vec3(vpx, vpy, vpz);
        }
    }

    private struct Aabb
    {
        public Vec3 Min;
        public Vec3 Max;
        public Vec3 Size => new Vec3(Max.X - Min.X, Max.Y - Min.Y, Max.Z - Min.Z);
    }

    private sealed class FastScanner
    {
        private readonly Stream _s;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _pos;
        public FastScanner(Stream s) { _s = s; }
        private int ReadByte()
        {
            if (_pos >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_pos++];
        }
        public string Next()
        {
            int c;
            do { c = ReadByte(); if (c == -1) return ""; } while (c <= 32);
            var sb = new System.Text.StringBuilder();
            while (c > 32)
            {
                sb.Append((char)c);
                c = ReadByte();
                if (c == -1) break;
            }
            return sb.ToString();
        }
        public int NextInt()
        {
            string t = Next();
            if (t.Length == 0) return 0;
            return int.Parse(t, CultureInfo.InvariantCulture);
        }
        public double NextDouble()
        {
            string t = Next();
            if (t.Length == 0) return 0;
            return double.Parse(t, CultureInfo.InvariantCulture);
        }
    }

    public static void Main()
    {
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;
        CultureInfo.CurrentUICulture = CultureInfo.InvariantCulture;

        var fs = new FastScanner(Console.OpenStandardInput());
        int V = fs.NextInt();
        int F = fs.NextInt();

        var verts = new Vec3[V];
        for (int i = 0; i < V; i++)
            verts[i] = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
        for (int i = 0; i < F; i++) { _ = fs.NextInt(); _ = fs.NextInt(); _ = fs.NextInt(); }

        double W = fs.NextDouble(), H = fs.NextDouble(), D = fs.NextDouble();

        long volClass = (long)Math.Round(W) * (long)Math.Round(H) * (long)Math.Round(D);
        bool extreme = W >= 200 || H >= 200 || D >= 200 || volClass >= 200L * 200L * 200L;
        bool large = !extreme && (W >= 50 || H >= 50 || D >= 50);
        bool medium = !extreme && !large && (W >= 10 || H >= 10 || D >= 10);

        int budget = extreme ? 6 : large ? 18 : medium ? 36 : 60;
        if (V > 2000) budget = Math.Min(budget, 6);
        else if (V > 500) budget = Math.Min(budget, 18);
        else if (V > 200) budget = Math.Min(budget, 36);

        var candidates = GenerateCandidateOrientations(budget);
        double safety = extreme ? 1e-6 : large ? 1e-7 : 1e-8;

        int bestIdx = 0;
        long bestCount = 0;
        Vec3 bestStep = new Vec3(0, 0, 0);
        Vec3 bestOffset = new Vec3(0, 0, 0);

        for (int i = 0; i < candidates.Count; i++)
        {
            Quat q = candidates[i];
            Aabb aabb = ComputeAabb(verts, q);
            Vec3 size = aabb.Size;
            if (size.X <= 0 || size.Y <= 0 || size.Z <= 0) continue;

            double sx = size.X * (1.0 + safety) + safety;
            double sy = size.Y * (1.0 + safety) + safety;
            double sz = size.Z * (1.0 + safety) + safety;
            if (sx > W || sy > H || sz > D) continue;

            long nx = (long)Math.Floor((W + 1e-12) / sx);
            long ny = (long)Math.Floor((H + 1e-12) / sy);
            long nz = (long)Math.Floor((D + 1e-12) / sz);
            if (nx <= 0 || ny <= 0 || nz <= 0) continue;

            long count = nx * ny * nz;
            if (count > bestCount)
            {
                bestCount = count;
                bestIdx = i;
                bestStep = new Vec3(sx, sy, sz);
                bestOffset = new Vec3(-aabb.Min.X, -aabb.Min.Y, -aabb.Min.Z);
            }
        }

        if (bestCount <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        Quat bestQ = candidates[bestIdx].Normalized();
        long bestNx = (long)Math.Floor((W + 1e-12) / bestStep.X);
        long bestNy = (long)Math.Floor((H + 1e-12) / bestStep.Y);
        long bestNz = (long)Math.Floor((D + 1e-12) / bestStep.Z);
        long total = bestNx * bestNy * bestNz;

        var sw = new StreamWriter(Console.OpenStandardOutput()) { NewLine = "\n" };
        sw.WriteLine(total.ToString(CultureInfo.InvariantCulture));

        for (long iz = 0; iz < bestNz; iz++)
        {
            double tzBase = iz * bestStep.Z + bestOffset.Z;
            for (long iy = 0; iy < bestNy; iy++)
            {
                double tyBase = iy * bestStep.Y + bestOffset.Y;
                for (long ix = 0; ix < bestNx; ix++)
                {
                    double tx = ix * bestStep.X + bestOffset.X;
                    double ty = tyBase;
                    double tz = tzBase;
                    sw.Write(tx.ToString("R", CultureInfo.InvariantCulture)); sw.Write(' ');
                    sw.Write(ty.ToString("R", CultureInfo.InvariantCulture)); sw.Write(' ');
                    sw.Write(tz.ToString("R", CultureInfo.InvariantCulture)); sw.Write(' ');
                    sw.Write(bestQ.W.ToString("R", CultureInfo.InvariantCulture)); sw.Write(' ');
                    sw.Write(bestQ.X.ToString("R", CultureInfo.InvariantCulture)); sw.Write(' ');
                    sw.Write(bestQ.Y.ToString("R", CultureInfo.InvariantCulture)); sw.Write(' ');
                    sw.WriteLine(bestQ.Z.ToString("R", CultureInfo.InvariantCulture));
                }
            }
        }

        sw.Flush();
    }

    private static Aabb ComputeAabb(Vec3[] verts, Quat q)
    {
        double minX = double.PositiveInfinity, minY = double.PositiveInfinity, minZ = double.PositiveInfinity;
        double maxX = double.NegativeInfinity, maxY = double.NegativeInfinity, maxZ = double.NegativeInfinity;
        for (int i = 0; i < verts.Length; i++)
        {
            Vec3 r = q.Rotate(verts[i]);
            if (r.X < minX) minX = r.X;
            if (r.Y < minY) minY = r.Y;
            if (r.Z < minZ) minZ = r.Z;
            if (r.X > maxX) maxX = r.X;
            if (r.Y > maxY) maxY = r.Y;
            if (r.Z > maxZ) maxZ = r.Z;
        }
        return new Aabb { Min = new Vec3(minX, minY, minZ), Max = new Vec3(maxX, maxY, maxZ) };
    }

    private static List<Quat> GenerateCandidateOrientations(int budget)
    {
        var list = new List<Quat>(budget + 8) { new Quat(1, 0, 0, 0) };
        double pi = Math.PI;

        var axes = new[] { new Vec3(1, 0, 0), new Vec3(0, 1, 0), new Vec3(0, 0, 1) };
        var angles = new[] { pi / 2, pi, 3 * pi / 2 };
        foreach (var axis in axes)
            foreach (var ang in angles)
                list.Add(Quat.FromAxisAngle(axis, ang));

        if (budget > 6)
        {
            list.Add(Quat.FromAxisAngle(new Vec3(1, 0, 0), pi / 4));
            list.Add(Quat.FromAxisAngle(new Vec3(0, 1, 0), pi / 4));
            list.Add(Quat.FromAxisAngle(new Vec3(0, 0, 1), pi / 4));
            list.Add(Quat.FromAxisAngle(new Vec3(1, 1, 0), pi / 4));
            list.Add(Quat.FromAxisAngle(new Vec3(1, 0, 1), pi / 4));
            list.Add(Quat.FromAxisAngle(new Vec3(0, 1, 1), pi / 4));
        }

        if (budget > list.Count)
        {
            int need = budget - list.Count;
            var rng = new XorShift64(0xC0D34203664AUL);
            for (int i = 0; i < need; i++)
            {
                double u1 = rng.NextDouble();
                double u2 = rng.NextDouble();
                double u3 = rng.NextDouble();
                double sqrt1MinusU1 = Math.Sqrt(1.0 - u1);
                double sqrtU1 = Math.Sqrt(u1);
                double theta1 = 2.0 * Math.PI * u2;
                double theta2 = 2.0 * Math.PI * u3;
                double x = sqrt1MinusU1 * Math.Sin(theta1);
                double y = sqrt1MinusU1 * Math.Cos(theta1);
                double z = sqrtU1 * Math.Sin(theta2);
                double w = sqrtU1 * Math.Cos(theta2);
                list.Add(new Quat(w, x, y, z).Normalized());
            }
        }

        var unique = new List<Quat>(list.Count);
        const double eps = 1e-9;
        foreach (var q in list)
        {
            Quat n = q.Normalized();
            bool dup = false;
            for (int i = 0; i < unique.Count; i++)
            {
                var u = unique[i];
                double d1 = Math.Abs(n.W - u.W) + Math.Abs(n.X - u.X) + Math.Abs(n.Y - u.Y) + Math.Abs(n.Z - u.Z);
                double d2 = Math.Abs(n.W + u.W) + Math.Abs(n.X + u.X) + Math.Abs(n.Y + u.Y) + Math.Abs(n.Z + u.Z);
                if (Math.Min(d1, d2) < eps) { dup = true; break; }
            }
            if (!dup) unique.Add(n);
            if (unique.Count >= budget) break;
        }
        return unique;
    }

    private struct XorShift64
    {
        private ulong _state;
        public XorShift64(ulong seed) { _state = seed != 0 ? seed : 0x9E3779B97F4A7C15UL; }
        private ulong NextU64()
        {
            ulong x = _state;
            x ^= x << 7;
            x ^= x >> 9;
            _state = x;
            return x;
        }
        public double NextDouble()
        {
            ulong u = NextU64();
            ulong mant = u >> 11;
            return mant * (1.0 / (1UL << 53));
        }
    }
}
