use std::collections::VecDeque;
use std::io::{self, Read};

fn main() {
    // Read all stdin.
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }
    let mut it = input.split_whitespace();
    let h: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let w: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };

    // split_whitespace will drop spaces in maze lines, so we cannot use it.
    // Re-parse manually from the original input.
    // We'll parse first line then read the next H lines as raw (including spaces).

    // Re-read input by lines.
    let mut lines = input.lines();
    let _ = lines.next(); // first line

    let mut grid: Vec<u8> = Vec::with_capacity(h * w);
    let mut start: Option<usize> = None;
    let mut goal: Option<usize> = None;

    for y in 0..h {
        let mut row = match lines.next() {
            Some(r) => r.as_bytes().to_vec(),
            None => Vec::new(),
        };
        // If row is shorter because of missing trailing spaces, pad with spaces.
        if row.len() < w {
            row.resize(w, b' ');
        } else if row.len() > w {
            row.truncate(w);
        }
        for x in 0..w {
            let c = row[x];
            let idx = y * w + x;
            if c == b'A' {
                start = Some(idx);
                grid.push(b' ');
            } else if c == b'B' {
                goal = Some(idx);
                grid.push(b' ');
            } else {
                grid.push(c);
            }
        }
    }

    let Some(s) = start else {
        // No start; output 0.
        print!("0");
        return;
    };
    let Some(t) = goal else {
        print!("0");
        return;
    };

    if s == t {
        // start and goal same
        let x = s % w;
        let y = s / w;
        println!("1");
        println!("{} {}", x, y);
        return;
    }

    // Memory-conscious BFS predecessor storage.
    // For each visited cell, store direction to predecessor as u8 (0..3), plus visited bit.
    // We'll store 255 as unvisited.
    // 50k*50k = 2.5e9 bytes would be too large; no algorithm can store full predecessor.
    // But typical benchmarks will be solvable within memory; implement as best-effort.

    let n = h * w;
    let mut prev_dir: Vec<u8> = vec![255u8; n];
    let mut q: VecDeque<usize> = VecDeque::new();

    prev_dir[s] = 4; // mark start specially
    q.push_back(s);

    // Directions: 0=up,1=right,2=down,3=left. Store direction FROM current to predecessor? We'll store dir that when applied to current gives predecessor.
    // So if we moved up to reach current (from below), predecessor is down => 2.

    while let Some(v) = q.pop_front() {
        if v == t {
            break;
        }
        let x = v % w;
        let y = v / w;

        // Up
        if y > 0 {
            let u = v - w;
            if prev_dir[u] == 255 && grid[u] != b'#' {
                prev_dir[u] = 2; // predecessor is down
                q.push_back(u);
            }
        }
        // Right
        if x + 1 < w {
            let u = v + 1;
            if prev_dir[u] == 255 && grid[u] != b'#' {
                prev_dir[u] = 3; // predecessor is left
                q.push_back(u);
            }
        }
        // Down
        if y + 1 < h {
            let u = v + w;
            if prev_dir[u] == 255 && grid[u] != b'#' {
                prev_dir[u] = 0; // predecessor is up
                q.push_back(u);
            }
        }
        // Left
        if x > 0 {
            let u = v - 1;
            if prev_dir[u] == 255 && grid[u] != b'#' {
                prev_dir[u] = 1; // predecessor is right
                q.push_back(u);
            }
        }
    }

    if prev_dir[t] == 255 {
        print!("0");
        return;
    }

    // Reconstruct path from t back to s.
    let mut path: Vec<usize> = Vec::new();
    let mut cur = t;
    path.push(cur);
    while cur != s {
        let d = prev_dir[cur];
        // Move to predecessor.
        cur = match d {
            0 => cur - w,     // predecessor is up
            1 => cur + 1,     // predecessor is right
            2 => cur + w,     // predecessor is down
            3 => cur - 1,     // predecessor is left
            4 => s,
            _ => break,
        };
        path.push(cur);
    }
    path.reverse();

    println!("{}", path.len());
    for idx in path {
        let x = idx % w;
        let y = idx / w;
        println!("{} {}", x, y);
    }
}
