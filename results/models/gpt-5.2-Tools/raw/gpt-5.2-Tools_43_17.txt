struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shade = 0.4 + 0.6 * diffuse;

    float2 uv = input.uv;

    // Brick grid: 8 x 16 tiles over UV [0,1]
    float2 grid = uv * float2(8.0, 16.0);

    // Stagger: half-brick offset on every other row
    float row = floor(grid.y);
    float stagger = (fmod(row, 2.0) >= 1.0) ? 0.5 : 0.0;
    grid.x += stagger;

    float2 cell = frac(grid);

    // Mortar thickness in cell space
    const float mortarX = 0.05;
    const float mortarY = 0.10;

    float mx = step(cell.x, mortarX) + step(1.0 - mortarX, cell.x);
    float my = step(cell.y, mortarY) + step(1.0 - mortarY, cell.y);
    float mortarMask = saturate(max(mx, my));

    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor  = float3(0.7, 0.2, 0.1);
    float3 baseColor = lerp(brickColor, mortarColor, mortarMask);

    return float4(baseColor * shade, 1.0);
}
