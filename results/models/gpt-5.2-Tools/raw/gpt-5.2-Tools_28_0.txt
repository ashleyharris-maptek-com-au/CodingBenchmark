use std::collections::BinaryHeap;
use std::io::{self, Read};

// Fast input parser for large graphs.
struct FastInput {
    buf: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self {
            buf: input.into_bytes(),
            idx: 0,
        }
    }

    #[inline]
    fn skip_whitespace(&mut self) {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
    }

    #[inline]
    fn next_usize(&mut self) -> Option<usize> {
        self.skip_whitespace();
        if self.idx >= self.buf.len() {
            return None;
        }
        let mut val: usize = 0;
        while self.idx < self.buf.len() && !self.buf[self.idx].is_ascii_whitespace() {
            val = val * 10 + (self.buf[self.idx] - b'0') as usize;
            self.idx += 1;
        }
        Some(val)
    }
}

fn main() {
    let mut input = FastInput::new();
    let n = match input.next_usize() {
        Some(v) => v,
        None => {
            // Empty input: output empty cover.
            println!("0\n");
            return;
        }
    };
    let m = match input.next_usize() {
        Some(v) => v,
        None => {
            println!("0\n");
            return;
        }
    };

    // adjacency as flat vector-of-vectors to allow iteration and deletions via covered flags.
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    let mut deg: Vec<usize> = vec![0; n];

    // Read edges; store both directions.
    for _ in 0..m {
        let u = match input.next_usize() {
            Some(v) => v,
            None => break,
        };
        let v = match input.next_usize() {
            Some(v) => v,
            None => break,
        };
        if u >= n || v >= n || u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
        deg[u] += 1;
        deg[v] += 1;
    }

    // Upper bound: take all vertices incident to some edge (or all vertices). We'll build cover incrementally.
    let mut in_cover = vec![false; n];
    let mut covered = vec![false; n]; // vertex removed from residual graph (in cover)

    // Priority queue for selecting highest residual degree vertices.
    let mut heap: BinaryHeap<(usize, usize)> = BinaryHeap::new(); // (deg, v)
    for v in 0..n {
        if deg[v] > 0 {
            heap.push((deg[v], v));
        }
    }

    // Residual number of uncovered edges is not tracked exactly; we keep removing until no vertex has residual degree.
    let mut residual_deg = deg.clone();

    // Greedy high-degree selection with local forcing:
    // - Pop vertex with current highest residual degree.
    // - If stale, skip.
    // - Add it to cover, remove it, decrease degrees of its neighbors.
    //
    // This yields a valid vertex cover (eventually all edges incident to remaining vertices vanish).
    while let Some((d, v)) = heap.pop() {
        if covered[v] {
            continue;
        }
        if residual_deg[v] != d {
            continue;
        }
        if d == 0 {
            break;
        }
        // Add v to cover.
        in_cover[v] = true;
        covered[v] = true;
        // Remove incident edges: for each neighbor u not removed, decrement its residual degree.
        for &u in &adj[v] {
            if covered[u] {
                continue;
            }
            // Decrease u's residual degree by 1 for the edge (u,v).
            if residual_deg[u] > 0 {
                residual_deg[u] -= 1;
                heap.push((residual_deg[u], u));
            }
        }
        residual_deg[v] = 0;
    }

    // At this point, the remaining residual graph might still contain edges between two non-covered vertices
    // if their degrees became stale due to duplicates or skipped edges; verify and fix using a maximal matching on remaining edges.
    // Build a 2-approx cover for any uncovered edges by taking both endpoints of a maximal matching.
    let mut matched = vec![false; n];
    for u in 0..n {
        if in_cover[u] {
            continue;
        }
        for &v in &adj[u] {
            if u == v || in_cover[v] {
                continue;
            }
            if !matched[u] && !matched[v] {
                matched[u] = true;
                matched[v] = true;
                in_cover[u] = true;
                in_cover[v] = true;
                break;
            }
        }
    }

    // Optional size reduction: simple 1-opt local improvement for a bounded budget.
    // Try to remove a vertex if all its incident edges are covered by other endpoints.
    // For huge graphs, keep it linear-ish: attempt only on low-degree vertices in cover.
    //
    // Build cover counts per vertex to check removability quickly: edge (u,v) covered if in_cover[u] || in_cover[v].
    // We'll use a neighbor-check for each candidate; this is O(sum deg of tested vertices).
    let mut cover_list: Vec<usize> = (0..n).filter(|&v| in_cover[v]).collect();
    // Sort by degree ascending to try removing small-degree vertices first.
    cover_list.sort_unstable_by_key(|&v| deg[v]);
    // Budget: up to 200_000 candidates or all if smaller.
    let budget = std::cmp::min(200_000usize, cover_list.len());
    for &v in cover_list.iter().take(budget) {
        if !in_cover[v] {
            continue;
        }
        // Quick lower bound: if v has any neighbor u not in cover, then edge (v,u) relies on v -> cannot remove.
        let mut removable = true;
        for &u in &adj[v] {
            if u == v {
                continue;
            }
            if !in_cover[u] {
                removable = false;
                break;
            }
        }
        if removable {
            in_cover[v] = false;
        }
    }

    let result: Vec<usize> = (0..n).filter(|&v| in_cover[v]).collect();
    println!("{}", result.len());
    if !result.is_empty() {
        // Fast output: build string.
        let mut out = String::new();
        for (i, v) in result.iter().enumerate() {
            if i > 0 {
                out.push(' ');
            }
            out.push_str(&v.to_string());
        }
        println!("{out}");
    } else {
        println!();
    }
}
