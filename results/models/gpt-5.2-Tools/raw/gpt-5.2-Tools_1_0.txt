// Adaptive TSP heuristic solver:
// - N <= 11: exact Held–Karp DP (O(N^2 2^N))
// - N <= 2000: greedy nearest-neighbor + 2-opt (restricted) + optional random restarts
// - N > 2000: space-filling curve (Hilbert-like via Morton/Z-order) ordering with coarse grid + local fixes
//
// Output: permutation of 0..N-1 starting at 0 (no return edge output).

namespace fastio {
    static const size_t BUFSZ = 1 << 20;
    static char buf[BUFSZ];
    static size_t bi = 0, bn = 0;

    inline bool refill() {
        bn = std::fread(buf, 1, BUFSZ, stdin);
        bi = 0;
        return bn != 0;
    }

    inline bool skip_blanks() {
        while (true) {
            if (bi >= bn) {
                if (!refill()) return false;
            }
            if (!std::isspace(static_cast<unsigned char>(buf[bi]))) return true;
            ++bi;
        }
    }

    template <class T>
    inline bool read_int(T &out) {
        if (!skip_blanks()) return false;
        bool neg = false;
        if (buf[bi] == '-') { neg = true; ++bi; }
        T val = 0;
        while (true) {
            if (bi >= bn) {
                if (!refill()) break;
            }
            char c = buf[bi];
            if (c < '0' || c > '9') break;
            val = val * 10 + (c - '0');
            ++bi;
        }
        out = neg ? -val : val;
        return true;
    }

    inline bool read_double(double &out) {
        if (!skip_blanks()) return false;
        // parse simple float: [sign] digits [.] digits [e[sign]digits]
        bool neg = false;
        if (buf[bi] == '-') { neg = true; ++bi; }
        double val = 0.0;
        while (true) {
            if (bi >= bn) {
                if (!refill()) break;
            }
            char c = buf[bi];
            if (c < '0' || c > '9') break;
            val = val * 10.0 + (c - '0');
            ++bi;
        }
        double frac = 0.0;
        double base = 1.0;
        if (bi < bn || refill()) {
            if (buf[bi] == '.') {
                ++bi;
                while (true) {
                    if (bi >= bn) {
                        if (!refill()) break;
                    }
                    char c = buf[bi];
                    if (c < '0' || c > '9') break;
                    frac = frac * 10.0 + (c - '0');
                    base *= 10.0;
                    ++bi;
                }
            }
        }
        val += frac / base;
        int exp10 = 0;
        if (bi < bn || refill()) {
            char c = buf[bi];
            if (c == 'e' || c == 'E') {
                ++bi;
                bool eneg = false;
                if (bi < bn || refill()) {
                    if (buf[bi] == '+') ++bi;
                    else if (buf[bi] == '-') { eneg = true; ++bi; }
                }
                int e = 0;
                while (true) {
                    if (bi >= bn) {
                        if (!refill()) break;
                    }
                    char d = buf[bi];
                    if (d < '0' || d > '9') break;
                    e = e * 10 + (d - '0');
                    ++bi;
                }
                exp10 = eneg ? -e : e;
            }
        }
        if (exp10 != 0) val *= std::pow(10.0, (double)exp10);
        out = neg ? -val : val;
        return true;
    }
}

struct Pt {
    float x, y;
};

static inline float dist2(const Pt &a, const Pt &b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return dx * dx + dy * dy;
}

// ---------------- Exact solver for very small N (Held–Karp DP) ----------------
static std::vector<int> solve_exact_held_karp(const std::vector<Pt> &p) {
    const int n = (int)p.size();
    // We must start at 0; produce a Hamiltonian path order starting at 0.
    // We'll solve TSP cycle and then output the tour starting at 0.
    const int FULL = 1 << n;
    const float INF = std::numeric_limits<float>::infinity();
    std::vector<float> dp(FULL * n, INF);
    std::vector<int16_t> parent(FULL * n, -1);
    auto idx = [n](int mask, int last) { return mask * n + last; };
    dp[idx(1, 0)] = 0.0f;
    for (int mask = 1; mask < FULL; ++mask) {
        if (!(mask & 1)) continue; // must include start
        for (int last = 0; last < n; ++last) {
            float cur = dp[idx(mask, last)];
            if (!(cur < INF)) continue;
            int rem = (~mask) & (FULL - 1);
            while (rem) {
                int nxt = __builtin_ctz(rem);
                rem &= rem - 1;
                int nmask = mask | (1 << nxt);
                float nd = cur + std::sqrt(dist2(p[last], p[nxt]));
                float &ref = dp[idx(nmask, nxt)];
                if (nd < ref) {
                    ref = nd;
                    parent[idx(nmask, nxt)] = (int16_t)last;
                }
            }
        }
    }
    int all = FULL - 1;
    // close cycle by returning to 0; pick best last.
    float best = INF;
    int best_last = 0;
    for (int last = 1; last < n; ++last) {
        float val = dp[idx(all, last)] + std::sqrt(dist2(p[last], p[0]));
        if (val < best) {
            best = val;
            best_last = last;
        }
    }
    // reconstruct order in reverse: ... -> best_last with all mask
    std::vector<int> rev;
    int mask = all;
    int last = best_last;
    while (last != -1) {
        rev.push_back(last);
        int pl = parent[idx(mask, last)];
        mask ^= (1 << last);
        last = pl;
    }
    // rev currently ends with 0
    std::reverse(rev.begin(), rev.end());
    // ensure starts at 0
    if (rev.empty() || rev[0] != 0) {
        // fallback: trivial order
        rev.resize(n);
        std::iota(rev.begin(), rev.end(), 0);
    }
    // For cycle tour, rev contains all nodes starting at 0.
    return rev;
}

// ---------------- Utility: rotate tour to start at 0 ----------------
static inline void rotate_to_start0(std::vector<int> &tour) {
    auto it = std::find(tour.begin(), tour.end(), 0);
    if (it != tour.end() && it != tour.begin()) std::rotate(tour.begin(), it, tour.end());
    if (tour.empty()) tour.push_back(0);
    if (tour[0] != 0) {
        // ensure 0 first by force (shouldn't happen)
        tour.erase(std::remove(tour.begin(), tour.end(), 0), tour.end());
        tour.insert(tour.begin(), 0);
    }
}

// ---------------- Greedy nearest neighbor (O(N^2) naive) ----------------
// For up to ~2000 we can do O(N^2) comfortably.
static std::vector<int> solve_nn_small(const std::vector<Pt> &p, int start, std::mt19937 &rng) {
    int n = (int)p.size();
    std::vector<int> tour;
    tour.reserve(n);
    std::vector<char> used(n, 0);
    int cur = start;
    used[cur] = 1;
    tour.push_back(cur);
    for (int step = 1; step < n; ++step) {
        int best = -1;
        float bestd = std::numeric_limits<float>::infinity();
        // occasional random candidate sampling for speed (still deterministic-ish)
        if (n > 800) {
            // sample K candidates first, then full scan if none (shouldn't happen)
            const int K = 64;
            for (int k = 0; k < K; ++k) {
                int j = (int)(rng() % (uint32_t)n);
                if (used[j]) continue;
                float d = dist2(p[cur], p[j]);
                if (d < bestd) { bestd = d; best = j; }
            }
            if (best != -1) {
                used[best] = 1;
                tour.push_back(best);
                cur = best;
                continue;
            }
        }
        for (int j = 0; j < n; ++j) {
            if (used[j]) continue;
            float d = dist2(p[cur], p[j]);
            if (d < bestd) { bestd = d; best = j; }
        }
        used[best] = 1;
        tour.push_back(best);
        cur = best;
    }
    return tour;
}

static inline float tour_len(const std::vector<Pt> &p, const std::vector<int> &tour) {
    const int n = (int)tour.size();
    double sum = 0.0;
    for (int i = 0; i + 1 < n; ++i) sum += std::sqrt(dist2(p[tour[i]], p[tour[i + 1]]));
    sum += std::sqrt(dist2(p[tour.back()], p[tour[0]]));
    return (float)sum;
}

// Restricted 2-opt: try improving swaps within a window to keep O(N * W)
static void two_opt_window(const std::vector<Pt> &p, std::vector<int> &tour, int window) {
    int n = (int)tour.size();
    if (n < 4) return;
    // position map for quick reverse updates is expensive; we just do in-place segment reverse.
    bool improved = true;
    int iter = 0;
    while (improved && iter < 6) { // cap iterations
        improved = false;
        ++iter;
        for (int i = 0; i < n; ++i) {
            int a = tour[i];
            int b = tour[(i + 1) % n];
            int jmax = std::min(n - 1, i + window);
            for (int j = i + 2; j <= jmax; ++j) {
                if (j == i || (j + 1) % n == i) continue;
                int c = tour[j];
                int d = tour[(j + 1) % n];
                float before = std::sqrt(dist2(p[a], p[b])) + std::sqrt(dist2(p[c], p[d]));
                float after  = std::sqrt(dist2(p[a], p[c])) + std::sqrt(dist2(p[b], p[d]));
                if (after + 1e-6f < before) {
                    // reverse segment (i+1..j)
                    int l = (i + 1);
                    int r = j;
                    while (l < r) std::swap(tour[l++], tour[r--]);
                    improved = true;
                }
            }
        }
    }
}

// ---------------- Large-scale ordering via Morton (Z-order) keys ----------------
// We avoid O(N log N) with huge memory? Sorting is still required to output a permutation.
// For truly enormous N, any solution must at least touch all inputs and output N ints; sorting is acceptable up to ~1e7.
// Beyond that, we do bucketed counting on a coarse grid to reduce sort cost.

static inline uint32_t part1by1_32(uint32_t x) {
    x &= 0x0000ffffu;
    x = (x | (x << 8)) & 0x00FF00FFu;
    x = (x | (x << 4)) & 0x0F0F0F0Fu;
    x = (x | (x << 2)) & 0x33333333u;
    x = (x | (x << 1)) & 0x55555555u;
    return x;
}
static inline uint32_t morton2D_16(uint32_t x, uint32_t y) {
    return (part1by1_32(y) << 1) | part1by1_32(x);
}

struct KeyIdx {
    uint32_t key;
    int idx;
};

static std::vector<int> solve_morton_sort(const std::vector<Pt> &p) {
    int n = (int)p.size();
    float minx = p[0].x, maxx = p[0].x, miny = p[0].y, maxy = p[0].y;
    for (int i = 1; i < n; ++i) {
        minx = std::min(minx, p[i].x);
        maxx = std::max(maxx, p[i].x);
        miny = std::min(miny, p[i].y);
        maxy = std::max(maxy, p[i].y);
    }
    float sx = (maxx > minx) ? (65535.0f / (maxx - minx)) : 0.0f;
    float sy = (maxy > miny) ? (65535.0f / (maxy - miny)) : 0.0f;

    std::vector<KeyIdx> v;
    v.reserve(n);
    for (int i = 0; i < n; ++i) {
        uint32_t xi = (sx == 0.0f) ? 0u : (uint32_t)std::min(65535.0f, std::max(0.0f, (p[i].x - minx) * sx));
        uint32_t yi = (sy == 0.0f) ? 0u : (uint32_t)std::min(65535.0f, std::max(0.0f, (p[i].y - miny) * sy));
        uint32_t key = morton2D_16(xi, yi);
        v.push_back({key, i});
    }
    std::sort(v.begin(), v.end(), [](const KeyIdx &a, const KeyIdx &b) {
        if (a.key != b.key) return a.key < b.key;
        return a.idx < b.idx;
    });
    std::vector<int> tour;
    tour.reserve(n);
    for (auto &e : v) tour.push_back(e.idx);
    return tour;
}

// Bucketed grid traversal for very large N to avoid heavy comparator cost:
// Quantize to GRID x GRID, count sort buckets, then within each bucket do local Morton sort.
static std::vector<int> solve_bucket_grid(const std::vector<Pt> &p, int grid_pow2) {
    int n = (int)p.size();
    int G = 1 << grid_pow2; // e.g., 1024 for pow2=10
    int64_t B = (int64_t)G * (int64_t)G;
    float minx = p[0].x, maxx = p[0].x, miny = p[0].y, maxy = p[0].y;
    for (int i = 1; i < n; ++i) {
        minx = std::min(minx, p[i].x);
        maxx = std::max(maxx, p[i].x);
        miny = std::min(miny, p[i].y);
        maxy = std::max(maxy, p[i].y);
    }
    float sx = (maxx > minx) ? ((float)(G - 1) / (maxx - minx)) : 0.0f;
    float sy = (maxy > miny) ? ((float)(G - 1) / (maxy - miny)) : 0.0f;

    std::vector<uint32_t> bucket(n);
    std::vector<uint32_t> counts((size_t)B, 0);
    for (int i = 0; i < n; ++i) {
        uint32_t xi = (sx == 0.0f) ? 0u : (uint32_t)std::min((float)(G - 1), std::max(0.0f, (p[i].x - minx) * sx));
        uint32_t yi = (sy == 0.0f) ? 0u : (uint32_t)std::min((float)(G - 1), std::max(0.0f, (p[i].y - miny) * sy));
        uint32_t b = yi * (uint32_t)G + xi;
        bucket[i] = b;
        counts[b]++;
    }
    // prefix sums
    std::vector<uint32_t> offset((size_t)B + 1, 0);
    for (int64_t i = 0; i < B; ++i) offset[i + 1] = offset[i] + counts[i];
    std::vector<int> perm(n);
    std::vector<uint32_t> cur = offset; // working cursor
    for (int i = 0; i < n; ++i) {
        uint32_t b = bucket[i];
        perm[cur[b]++] = i;
    }
    // For each bucket, apply local morton ordering on 16-bit quantization inside bucket.
    // Use small temporary buffer for keys.
    std::vector<int> tour;
    tour.reserve(n);
    std::vector<KeyIdx> tmp;
    for (int64_t b = 0; b < B; ++b) {
        uint32_t l = offset[b], r = offset[b + 1];
        uint32_t m = r - l;
        if (m == 0) continue;
        if (m == 1) { tour.push_back(perm[l]); continue; }
        tmp.clear();
        tmp.reserve(m);
        // compute local bounds for tighter quantization
        float bminx = p[perm[l]].x, bmaxx = bminx, bminy = p[perm[l]].y, bmaxy = bminy;
        for (uint32_t k = l + 1; k < r; ++k) {
            const Pt &q = p[perm[k]];
            bminx = std::min(bminx, q.x); bmaxx = std::max(bmaxx, q.x);
            bminy = std::min(bminy, q.y); bmaxy = std::max(bmaxy, q.y);
        }
        float lsx = (bmaxx > bminx) ? (65535.0f / (bmaxx - bminx)) : 0.0f;
        float lsy = (bmaxy > bminy) ? (65535.0f / (bmaxy - bminy)) : 0.0f;
        for (uint32_t k = l; k < r; ++k) {
            int i = perm[k];
            uint32_t xi = (lsx == 0.0f) ? 0u : (uint32_t)std::min(65535.0f, std::max(0.0f, (p[i].x - bminx) * lsx));
            uint32_t yi = (lsy == 0.0f) ? 0u : (uint32_t)std::min(65535.0f, std::max(0.0f, (p[i].y - bminy) * lsy));
            tmp.push_back({morton2D_16(xi, yi), i});
        }
        std::sort(tmp.begin(), tmp.end(), [](const KeyIdx &a, const KeyIdx &b) {
            if (a.key != b.key) return a.key < b.key;
            return a.idx < b.idx;
        });
        for (auto &e : tmp) tour.push_back(e.idx);
    }
    return tour;
}

// Ensure permutation validity and start at 0.
static void fix_and_output(std::vector<int> tour, int n) {
    if ((int)tour.size() != n) {
        tour.resize(n);
        std::iota(tour.begin(), tour.end(), 0);
    }
    // Remove duplicates / missing using visited mark if needed (shouldn't happen).
    std::vector<char> seen(n, 0);
    std::vector<int> fixed;
    fixed.reserve(n);
    for (int v : tour) {
        if (v < 0 || v >= n) continue;
        if (seen[v]) continue;
        seen[v] = 1;
        fixed.push_back(v);
    }
    for (int i = 0; i < n; ++i) if (!seen[i]) fixed.push_back(i);
    rotate_to_start0(fixed);
    // output
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    for (int i = 0; i < n; ++i) {
        if (i) std::cout << ' ';
        std::cout << fixed[i];
    }
    std::cout << "\n";
}

int main() {
    int n = 0;
    if (!fastio::read_int(n)) return 0;
    if (n <= 0) return 0;

    std::vector<Pt> p;
    p.resize(n);
    for (int i = 0; i < n; ++i) {
        double x, y;
        if (!fastio::read_double(x)) x = 0.0;
        if (!fastio::read_double(y)) y = 0.0;
        p[i].x = (float)x;
        p[i].y = (float)y;
    }

    // Seed from fixed value for determinism across runs.
    std::mt19937 rng(123456789u);

    std::vector<int> tour;
    if (n <= 11) {
        tour = solve_exact_held_karp(p);
        fix_and_output(std::move(tour), n);
        return 0;
    }

    if (n <= 2000) {
        // Multiple starts improve NN quality cheaply.
        int restarts = (n <= 400) ? 12 : 6;
        float bestLen = std::numeric_limits<float>::infinity();
        std::vector<int> bestTour;
        for (int r = 0; r < restarts; ++r) {
            int start = (r == 0) ? 0 : (int)(rng() % (uint32_t)n);
            auto t = solve_nn_small(p, start, rng);
            rotate_to_start0(t);
            // local improvement
            int window = (n <= 400) ? 120 : 60;
            two_opt_window(p, t, window);
            rotate_to_start0(t);
            float L = tour_len(p, t);
            if (L < bestLen) {
                bestLen = L;
                bestTour = std::move(t);
            }
        }
        fix_and_output(std::move(bestTour), n);
        return 0;
    }

    if (n <= 10000000) {
        // Morton sort works well and is simple.
        // For >= 2e6, bucketed grid reduces comparator overhead and improves cache behavior.
        if (n >= 2000000) {
            // Choose grid size based on N: keep average bucket size modest.
            // pow2=10 => 1024^2 ~1M buckets.
            int grid_pow2 = 10;
            if (n >= 8000000) grid_pow2 = 11; // 2048^2 ~4M buckets, counts array ~16MB
            tour = solve_bucket_grid(p, grid_pow2);
        } else {
            tour = solve_morton_sort(p);
        }
        // very light local cleanup for mid-large sizes
        if (n <= 200000) {
            rotate_to_start0(tour);
            two_opt_window(p, tour, 24);
        }
        fix_and_output(std::move(tour), n);
        return 0;
    }

    // Epic: output-only dominated. Avoid global sort comparator as much as possible.
    // Use coarse bucketing with modest grid; within buckets, emit in input order (no sort) to stay linear.
    // This is a very fast approximation that still preserves spatial locality.
    {
        float minx = p[0].x, maxx = p[0].x, miny = p[0].y, maxy = p[0].y;
        for (int i = 1; i < n; ++i) {
            minx = std::min(minx, p[i].x);
            maxx = std::max(maxx, p[i].x);
            miny = std::min(miny, p[i].y);
            maxy = std::max(maxy, p[i].y);
        }
        const int G = 1024; // fixed coarse grid
        const int64_t B = (int64_t)G * (int64_t)G;
        float sx = (maxx > minx) ? ((float)(G - 1) / (maxx - minx)) : 0.0f;
        float sy = (maxy > miny) ? ((float)(G - 1) / (maxy - miny)) : 0.0f;
        std::vector<uint32_t> counts((size_t)B, 0);
        std::vector<uint32_t> bucket(n);
        for (int i = 0; i < n; ++i) {
            uint32_t xi = (sx == 0.0f) ? 0u : (uint32_t)std::min((float)(G - 1), std::max(0.0f, (p[i].x - minx) * sx));
            uint32_t yi = (sy == 0.0f) ? 0u : (uint32_t)std::min((float)(G - 1), std::max(0.0f, (p[i].y - miny) * sy));
            uint32_t b = yi * (uint32_t)G + xi;
            bucket[i] = b;
            counts[b]++;
        }
        std::vector<uint32_t> offset((size_t)B + 1, 0);
        for (int64_t i = 0; i < B; ++i) offset[i + 1] = offset[i] + counts[i];
        std::vector<int> perm(n);
        std::vector<uint32_t> cur = offset;
        for (int i = 0; i < n; ++i) perm[cur[bucket[i]]++] = i;
        // Traverse buckets in Z-order (Morton) over coarse grid to improve locality.
        // We'll generate bucket order by sorting bucket ids by morton key (G up to 1024 => 10 bits each).
        struct BKey { uint32_t key; uint32_t b; };
        std::vector<BKey> border;
        border.reserve((size_t)B);
        for (uint32_t by = 0; by < (uint32_t)G; ++by) {
            for (uint32_t bx = 0; bx < (uint32_t)G; ++bx) {
                uint32_t b = by * (uint32_t)G + bx;
                if (counts[b] == 0) continue;
                uint32_t key = morton2D_16(bx << 6, by << 6); // upscale to 16-bit-ish
                border.push_back({key, b});
            }
        }
        std::sort(border.begin(), border.end(), [](const BKey &a, const BKey &b) {
            if (a.key != b.key) return a.key < b.key;
            return a.b < b.b;
        });
        tour.clear();
        tour.reserve(n);
        for (auto &bk : border) {
            uint32_t b = bk.b;
            uint32_t l = offset[b], r = offset[b + 1];
            for (uint32_t k = l; k < r; ++k) tour.push_back(perm[k]);
        }
        fix_and_output(std::move(tour), n);
        return 0;
    }
}
