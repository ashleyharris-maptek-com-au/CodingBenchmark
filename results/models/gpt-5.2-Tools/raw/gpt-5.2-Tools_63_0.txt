// Warehouse Robot Job Assignment Benchmark
// Node.js TypeScript source (compile with `tsc` or run via a TS runner in the harness).
//
// Reads JSON from stdin and prints:
//   {"assignments":[[jobId,...], ...]}
// where array index corresponds to robot id (dense 0..R-1) when possible; otherwise a deterministic id-sorted order.
//
// Approach (deterministic, scalable):
// - Precompute shortest grid-walking distances (4-neighborhood) from each robot to every job via BFS on open cells.
// - Assign jobs greedily by descending priority (tie: job id), choosing the closest robot (tie: robot id order).

declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Robot = { id: any; x: number; y: number };
type Job = { id: any; x: number; y: number; priority: number };

const grid: string[] = data.grid;
const robots: Robot[] = data.robots;
const jobs: Job[] = data.jobs;

const H = grid.length | 0;
const W = H ? (grid[0].length | 0) : 0;
const N = W * H;

function toKeyId(v: any): string {
  return typeof v === 'string' ? v : String(v);
}

function tryNumId(v: any): number | null {
  if (typeof v === 'number' && Number.isFinite(v)) return v;
  if (typeof v === 'string' && v.trim() !== '') {
    const n = Number(v);
    if (Number.isFinite(n)) return n;
  }
  return null;
}

function cmpIds(a: any, b: any): number {
  const an = tryNumId(a);
  const bn = tryNumId(b);
  if (an !== null && bn !== null) return an - bn;
  const as = toKeyId(a);
  const bs = toKeyId(b);
  return as < bs ? -1 : as > bs ? 1 : 0;
}

// Open floor mask
const open = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = grid[y];
  for (let x = 0; x < W; x++) {
    open[y * W + x] = row.charCodeAt(x) === 46 /* '.' */ ? 1 : 0;
  }
}

function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < W && y >= 0 && y < H;
}
function idx(x: number, y: number): number {
  return y * W + x;
}
function cellOk(x: number, y: number): boolean {
  return inBounds(x, y) && open[idx(x, y)] === 1;
}

// Output robot ordering:
// - If ids are a permutation of 0..R-1 => assignments index = id.
// - Else => deterministic order by cmpIds.
const R = robots.length | 0;
let idsAreDense = true;
const seenDense = new Uint8Array(R);
for (let i = 0; i < R; i++) {
  const rn = tryNumId(robots[i].id);
  if (rn === null || (rn | 0) !== rn || rn < 0 || rn >= R || seenDense[rn]) {
    idsAreDense = false;
    break;
  }
  seenDense[rn] = 1;
}

let robotsOrdered: Robot[];
if (idsAreDense) {
  robotsOrdered = new Array(R);
  for (let i = 0; i < R; i++) robotsOrdered[tryNumId(robots[i].id)!] = robots[i];
} else {
  robotsOrdered = robots.slice().sort((a, b) => cmpIds(a.id, b.id));
}

// Jobs order: descending priority, tie by id
const jobsOrdered: Job[] = jobs
  .slice()
  .sort((a, b) => (b.priority - a.priority) || cmpIds(a.id, b.id));
const J = jobs.length | 0;

// Precompute cell indices for jobs and robots
const jobCells = new Int32Array(J);
for (let j = 0; j < J; j++) {
  const jb = jobs[j];
  jobCells[j] = cellOk(jb.x, jb.y) ? idx(jb.x, jb.y) : -1;
}
const robotCells = new Int32Array(R);
for (let i = 0; i < R; i++) {
  const rb = robotsOrdered[i];
  robotCells[i] = cellOk(rb.x, rb.y) ? idx(rb.x, rb.y) : -1;
}

// BFS core with early exit to a fixed set of targets.
const INF = 0x3fffffff;
const dist = new Int32Array(N);
const q = new Int32Array(N);
const targetPos = new Int32Array(N);
targetPos.fill(-1);
let stamp = 1;
const targetStamp = new Int32Array(N);

function bfsToTargets(sourceIndex: number, targetsIdx: Int32Array, out: Int32Array): void {
  out.fill(INF);
  if (sourceIndex < 0 || sourceIndex >= N || open[sourceIndex] === 0) return;

  const s = stamp++;
  if (stamp === 0x7fffffff) {
    stamp = 1;
    targetStamp.fill(0);
  }
  for (let i = 0; i < targetsIdx.length; i++) {
    const ti = targetsIdx[i];
    if (ti >= 0 && ti < N) {
      targetStamp[ti] = s;
      targetPos[ti] = i;
    }
  }

  dist.fill(-1);
  let head = 0, tail = 0;
  q[tail++] = sourceIndex;
  dist[sourceIndex] = 0;

  let remaining = targetsIdx.length;
  if (targetStamp[sourceIndex] === s) {
    const p = targetPos[sourceIndex];
    if (p >= 0 && out[p] === INF) {
      out[p] = 0;
      remaining--;
    }
  }

  while (head < tail && remaining > 0) {
    const v = q[head++];
    const dv = (dist[v] + 1) | 0;
    const x = v % W;
    const y = (v / W) | 0;

    // left
    if (x > 0) {
      const n = v - 1;
      if (open[n] && dist[n] === -1) {
        dist[n] = dv;
        if (targetStamp[n] === s) {
          const p = targetPos[n];
          if (p >= 0 && out[p] === INF) {
            out[p] = dv;
            if (--remaining === 0) break;
          }
        }
        q[tail++] = n;
      }
    }
    // right
    if (x + 1 < W) {
      const n = v + 1;
      if (open[n] && dist[n] === -1) {
        dist[n] = dv;
        if (targetStamp[n] === s) {
          const p = targetPos[n];
          if (p >= 0 && out[p] === INF) {
            out[p] = dv;
            if (--remaining === 0) break;
          }
        }
        q[tail++] = n;
      }
    }
    // up
    if (y > 0) {
      const n = v - W;
      if (open[n] && dist[n] === -1) {
        dist[n] = dv;
        if (targetStamp[n] === s) {
          const p = targetPos[n];
          if (p >= 0 && out[p] === INF) {
            out[p] = dv;
            if (--remaining === 0) break;
          }
        }
        q[tail++] = n;
      }
    }
    // down
    if (y + 1 < H) {
      const n = v + W;
      if (open[n] && dist[n] === -1) {
        dist[n] = dv;
        if (targetStamp[n] === s) {
          const p = targetPos[n];
          if (p >= 0 && out[p] === INF) {
            out[p] = dv;
            if (--remaining === 0) break;
          }
        }
        q[tail++] = n;
      }
    }
  }

  for (let i = 0; i < targetsIdx.length; i++) {
    const ti = targetsIdx[i];
    if (ti >= 0 && ti < N) targetPos[ti] = -1;
  }
}

// Distances robot->job (R x J), row-major
const dRJ = new Int32Array(R * J);
for (let k = 0; k < dRJ.length; k++) dRJ[k] = INF;

const tmpOut = new Int32Array(J);
for (let i = 0; i < R; i++) {
  const rc = robotCells[i];
  if (rc < 0) continue;
  bfsToTargets(rc, jobCells, tmpOut);
  const base = i * J;
  for (let j = 0; j < J; j++) dRJ[base + j] = tmpOut[j];
}

// Map job id -> original index for distance lookup
const jobIndexById = new Map<string, number>();
for (let j = 0; j < J; j++) jobIndexById.set(toKeyId(jobs[j].id), j);

// Greedy assignment by priority
const assignments: any[][] = Array.from({ length: R }, () => []);
for (const job of jobsOrdered) {
  const jpos = jobIndexById.get(toKeyId(job.id));
  let bestRobot = 0;
  let bestDist = INF;
  if (jpos !== undefined) {
    for (let i = 0; i < R; i++) {
      const d = dRJ[i * J + jpos];
      if (d < bestDist) {
        bestDist = d;
        bestRobot = i;
      }
    }
  }
  assignments[bestRobot].push(job.id);
}

process.stdout.write(JSON.stringify({ assignments }));
