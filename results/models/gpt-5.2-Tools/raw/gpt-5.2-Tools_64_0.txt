// Node.js TypeScript solution for "Plan grid paths for search drones"
// Reads JSON from stdin, outputs {"paths": [...]}

declare const require: any;
declare const process: any;
const fs = require("fs");
let input: string = fs.readFileSync(0, "utf8");
// Handle UTF-8 BOM if present
if (input.length && input.charCodeAt(0) === 0xfeff) input = input.slice(1);
const data = JSON.parse(input);

type Cell = [number, number]; // [x,y]

interface Drone {
  id: any;
  x: number;
  y: number;
}

const grid: number[][] = data.probability;
const drones: Drone[] = data.drones;
const maxStepsPerDrone: number = data.maxStepsPerDrone | 0;

const H = grid.length;
const W = H > 0 ? grid[0].length : 0;

function inBounds(x: number, y: number): boolean {
  return x >= 0 && y >= 0 && x < W && y < H;
}
function isFree(x: number, y: number): boolean {
  return inBounds(x, y) && grid[y][x] !== -1;
}

const DIRS: Array<[number, number]> = [
  [1, 0],
  [-1, 0],
  [0, 1],
  [0, -1],
  [0, 0], // stay
];

function key(x: number, y: number): number {
  return y * W + x;
}

// Precompute positive "value" per cell (0 for obstacles/zero/negative non-obstacle values).
const value = new Float64Array(W * H);
for (let y = 0; y < H; y++) {
  const row = grid[y];
  for (let x = 0; x < W; x++) {
    const v = row[x];
    if (v !== -1 && v > 0) value[key(x, y)] = v;
  }
}

// Multi-source BFS to compute distance-to-nearest-obstacle (for mild preference of open areas).
function computeClearance(): Uint16Array {
  const INF = 0xffff;
  const dist = new Uint16Array(W * H);
  for (let i = 0; i < dist.length; i++) dist[i] = INF;
  const qx = new Int32Array(W * H);
  const qy = new Int32Array(W * H);
  let qh = 0,
    qt = 0;
  // Seed with obstacles and border-out-of-bounds as implicit obstacles by seeding border free cells with 1?
  // Here: set obstacle cells dist=0, BFS through free cells.
  for (let y = 0; y < H; y++) {
    for (let x = 0; x < W; x++) {
      if (grid[y][x] === -1) {
        const k = key(x, y);
        dist[k] = 0;
        qx[qt] = x;
        qy[qt] = y;
        qt++;
      }
    }
  }
  // If no obstacles, just leave INF -> treated as large.
  while (qh < qt) {
    const x = qx[qh];
    const y = qy[qh];
    qh++;
    const base = dist[key(x, y)];
    const nd = (base + 1) & 0xffff;
    for (let i = 0; i < 4; i++) {
      const nx = x + DIRS[i][0];
      const ny = y + DIRS[i][1];
      if (!inBounds(nx, ny)) continue;
      if (grid[ny][nx] === -1) continue;
      const nk = key(nx, ny);
      if (dist[nk] > nd) {
        dist[nk] = nd;
        qx[qt] = nx;
        qy[qt] = ny;
        qt++;
      }
    }
  }
  return dist;
}

const clearance = computeClearance();

// Global covered set: only counts first time a positive-value cell is visited by any drone.
const covered = new Uint8Array(W * H);

function manhattan(a: Cell, b: Cell): number {
  return Math.abs(a[0] - b[0]) + Math.abs(a[1] - b[1]);
}

// Pick a target cell for a drone based on best "gain"/distance ratio.
// We sample from top-valued cells not yet covered to avoid scanning full grid repeatedly.
const allPosCells: number[] = [];
for (let y = 0; y < H; y++) {
  for (let x = 0; x < W; x++) {
    const k = key(x, y);
    if (value[k] > 0) allPosCells.push(k);
  }
}
// Sort descending by value once.
allPosCells.sort((a, b) => value[b] - value[a]);

function cellFromKey(k: number): Cell {
  const y = Math.floor(k / W);
  const x = k - y * W;
  return [x, y];
}

function chooseTarget(pos: Cell, remaining: number, sampleN: number): number {
  // Return key of chosen target, or -1 if none.
  let bestK = -1;
  let bestScore = -1e30;
  // Dynamic sample: take some of highest value cells, plus some random-ish spread from rest.
  const n = allPosCells.length;
  if (n === 0) return -1;
  const topN = Math.min(sampleN, n);
  for (let i = 0; i < topN; i++) {
    const k = allPosCells[i];
    if (covered[k]) continue;
    const [tx, ty] = cellFromKey(k);
    // Cheap distance estimate: Manhattan; obstacles not considered, but ok for target picking.
    const d = Math.max(1, manhattan(pos, [tx, ty]));
    // If too far for remaining, de-prioritize but still consider (maybe partial progress helps).
    const feasible = d <= remaining ? 1.0 : 0.4;
    const clr = clearance[k] === 0xffff ? 64 : clearance[k];
    const openness = 1 + Math.min(8, clr) * 0.05;
    const s = (value[k] * openness * feasible) / d;
    if (s > bestScore) {
      bestScore = s;
      bestK = k;
    }
  }

  // If still none (all covered in sample), scan further until we find an uncovered cell (bounded).
  if (bestK === -1) {
    const limit = Math.min(n, topN + 2000);
    for (let i = topN; i < limit; i++) {
      const k = allPosCells[i];
      if (!covered[k]) return k;
    }
  }
  return bestK;
}

// A* pathfinding (4-neighborhood) between two points on grid.
function astar(start: Cell, goal: Cell, maxExpand: number): Cell[] | null {
  const [sx, sy] = start;
  const [gx, gy] = goal;
  if (!isFree(sx, sy) || !isFree(gx, gy)) return null;
  if (sx === gx && sy === gy) return [start];

  const N = W * H;
  const gScore = new Int32Array(N);
  const fScore = new Int32Array(N);
  const came = new Int32Array(N);
  const openFlag = new Uint8Array(N);
  const closed = new Uint8Array(N);

  for (let i = 0; i < N; i++) {
    gScore[i] = 0x3fffffff;
    fScore[i] = 0x3fffffff;
    came[i] = -1;
  }

  const startK = key(sx, sy);
  const goalK = key(gx, gy);
  gScore[startK] = 0;
  fScore[startK] = manhattan(start, goal);

  // Binary heap of keys with fScore ordering.
  const heap: number[] = [];
  function heapPush(k: number) {
    heap.push(k);
    let i = heap.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (fScore[heap[p]] <= fScore[heap[i]]) break;
      const tmp = heap[p];
      heap[p] = heap[i];
      heap[i] = tmp;
      i = p;
    }
  }
  function heapPop(): number | undefined {
    const n = heap.length;
    if (n === 0) return undefined;
    const top = heap[0];
    const last = heap.pop()!;
    if (n > 1) {
      heap[0] = last;
      let i = 0;
      while (true) {
        const l = i * 2 + 1;
        const r = l + 1;
        if (l >= heap.length) break;
        let m = l;
        if (r < heap.length && fScore[heap[r]] < fScore[heap[l]]) m = r;
        if (fScore[heap[i]] <= fScore[heap[m]]) break;
        const tmp = heap[i];
        heap[i] = heap[m];
        heap[m] = tmp;
        i = m;
      }
    }
    return top;
  }

  heapPush(startK);
  openFlag[startK] = 1;

  let expanded = 0;
  while (heap.length) {
    if (expanded++ > maxExpand) return null;
    const curK = heapPop()!;
    if (closed[curK]) continue;
    openFlag[curK] = 0;
    if (curK === goalK) {
      // Reconstruct
      const pathKeys: number[] = [];
      let k = curK;
      while (k !== -1) {
        pathKeys.push(k);
        k = came[k];
      }
      pathKeys.reverse();
      const out: Cell[] = new Array(pathKeys.length);
      for (let i = 0; i < pathKeys.length; i++) out[i] = cellFromKey(pathKeys[i]);
      return out;
    }
    closed[curK] = 1;
    const cy = Math.floor(curK / W);
    const cx = curK - cy * W;
    const baseG = gScore[curK] + 1;
    for (let i = 0; i < 4; i++) {
      const nx = cx + DIRS[i][0];
      const ny = cy + DIRS[i][1];
      if (!isFree(nx, ny)) continue;
      const nk = key(nx, ny);
      if (closed[nk]) continue;
      if (baseG < gScore[nk]) {
        came[nk] = curK;
        gScore[nk] = baseG;
        fScore[nk] = baseG + manhattan([nx, ny], goal);
        if (!openFlag[nk]) {
          openFlag[nk] = 1;
          heapPush(nk);
        } else {
          // push duplicate; lazy removal handled by closed[]/gScore checks
          heapPush(nk);
        }
      }
    }
  }
  return null;
}

function markCovered(path: Cell[]) {
  for (const [x, y] of path) {
    const k = key(x, y);
    if (value[k] > 0) covered[k] = 1;
  }
}

// Simple local greedy walker to collect nearby uncovered value when no good far target.
function localGreedyWalk(start: Cell, steps: number): Cell[] {
  let x = start[0],
    y = start[1];
  const out: Cell[] = [[x, y]];
  for (let t = 0; t < steps; t++) {
    let bestNx = x,
      bestNy = y;
    let best = -1e18;
    for (const [dx, dy] of DIRS) {
      const nx = x + dx;
      const ny = y + dy;
      if (!isFree(nx, ny)) continue;
      const k = key(nx, ny);
      const gain = covered[k] ? 0 : value[k];
      const clr = clearance[k] === 0xffff ? 64 : clearance[k];
      const openness = 1 + Math.min(6, clr) * 0.03;
      const s = gain * 5 + openness; // prioritize new value strongly
      if (s > best) {
        best = s;
        bestNx = nx;
        bestNy = ny;
      }
    }
    x = bestNx;
    y = bestNy;
    out.push([x, y]);
  }
  return out;
}

// Main planning:
// For each drone, build a path of length <= maxStepsPerDrone+1 (includes start).
// We do sequential planning with shared covered[].
const paths: Cell[][] = [];

// Budget for A* expansions per call based on grid size.
const baseExpand = Math.max(2000, Math.min(60000, (W * H) | 0));

for (let di = 0; di < drones.length; di++) {
  const d = drones[di];
  let cur: Cell = [d.x | 0, d.y | 0];
  if (!isFree(cur[0], cur[1])) {
    // If starting on obstacle/out of bounds, emit a degenerate stay path at nearest valid cell (fallback to 0,0).
    // But the spec likely guarantees valid starts; keep safe.
    let fx = 0,
      fy = 0;
    outer: for (let y = 0; y < H; y++) {
      for (let x = 0; x < W; x++) {
        if (isFree(x, y)) {
          fx = x;
          fy = y;
          break outer;
        }
      }
    }
    cur = [fx, fy];
  }

  const path: Cell[] = [cur];
  // Count "moves" as transitions; maxStepsPerDrone moves means path length maxStepsPerDrone+1.
  let remaining = maxStepsPerDrone;

  // First, mark starting cell covered if valuable.
  markCovered([cur]);

  while (remaining > 0) {
    // If plenty of uncovered high value remains, target it.
    const targetK = chooseTarget(cur, remaining, 800);
    if (targetK === -1) {
      // Nothing valuable left: just stay.
      for (let i = 0; i < remaining; i++) path.push(cur);
      remaining = 0;
      break;
    }
    const target = cellFromKey(targetK);
    // Attempt A* to target; cap expansions.
    const p = astar(cur, target, baseExpand);
    if (!p || p.length <= 1) {
      // Can't find or already there; do local greedy.
      const local = localGreedyWalk(cur, remaining);
      // local includes start; append next cells
      for (let i = 1; i < local.length; i++) path.push(local[i]);
      markCovered(local);
      remaining = 0;
      break;
    }

    // p includes start; we can only take up to remaining moves.
    const stepsToTake = Math.min(remaining, p.length - 1);
    // Append first stepsToTake cells from p (excluding first which is cur)
    for (let i = 1; i <= stepsToTake; i++) path.push(p[i]);
    const takenSegment = p.slice(0, stepsToTake + 1);
    markCovered(takenSegment);
    cur = path[path.length - 1];
    remaining -= stepsToTake;

    // If we couldn't reach target within remaining, continue loop (target may change).
    // If reached target and still time, we will pick another target next.
  }

  paths.push(path);
}

process.stdout.write(JSON.stringify({ paths }));
