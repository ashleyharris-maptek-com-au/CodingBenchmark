use std::cmp::Ordering;
use std::collections::VecDeque;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Edge {
    to: usize,
    next: i32,
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }
    let mut it = input.split_whitespace();
    let n: usize = match it.next() {
        Some(s) => s.parse().unwrap(),
        None => return,
    };
    let m: usize = match it.next() {
        Some(s) => s.parse().unwrap(),
        None => 0,
    };

    let mut head = vec![-1i32; n];
    let mut edges: Vec<Edge> = Vec::with_capacity(m.saturating_mul(2));
    let mut deg = vec![0u32; n];

    let mut add_edge = |u: usize, v: usize, head: &mut [i32], edges: &mut Vec<Edge>, deg: &mut [u32]| {
        edges.push(Edge { to: v, next: head[u] });
        head[u] = (edges.len() - 1) as i32;
        deg[u] += 1;
    };

    for _ in 0..m {
        let u: usize = match it.next() {
            Some(s) => s.parse().unwrap(),
            None => break,
        };
        let v: usize = match it.next() {
            Some(s) => s.parse().unwrap(),
            None => break,
        };
        if u >= n || v >= n {
            continue;
        }
        if u == v {
            deg[u] = deg[u].saturating_add(2);
            continue;
        }
        add_edge(u, v, &mut head, &mut edges, &mut deg);
        add_edge(v, u, &mut head, &mut edges, &mut deg);
    }

    if n == 0 {
        return;
    }
    if n == 1 {
        println!("0 0");
        return;
    }

    // Extreme scale: very fast degree-sorted circle/spiral
    if n >= 200_000 {
        let mut order: Vec<usize> = (0..n).collect();
        order.sort_unstable_by(|&a, &b| deg[b].cmp(&deg[a]).then_with(|| a.cmp(&b)));
        let mut posx = vec![0f64; n];
        let mut posy = vec![0f64; n];
        let two_pi = std::f64::consts::PI * 2.0;
        for (i, &v) in order.iter().enumerate() {
            let t = (i as f64) / (n as f64);
            let ang = two_pi * t;
            let r = 1.0 + ((deg[v] as f64 + 1.0).ln() * 0.02) + ((hash_u64(v as u64) as f64) * 1e-12);
            posx[v] = r * ang.cos();
            posy[v] = r * ang.sin();
        }
        for i in 0..n {
            println!("{:.6} {:.6}", posx[i], posy[i]);
        }
        return;
    }

    // Connected components
    let mut comp_id = vec![usize::MAX; n];
    let mut comps: Vec<Vec<usize>> = Vec::new();
    let mut q = VecDeque::new();

    for s in 0..n {
        if comp_id[s] != usize::MAX {
            continue;
        }
        let cid = comps.len();
        comp_id[s] = cid;
        q.clear();
        q.push_back(s);
        let mut nodes = Vec::new();
        nodes.push(s);
        while let Some(u) = q.pop_front() {
            let mut e = head[u];
            while e != -1 {
                let ei = e as usize;
                let v = edges[ei].to;
                if comp_id[v] == usize::MAX {
                    comp_id[v] = cid;
                    q.push_back(v);
                    nodes.push(v);
                }
                e = edges[ei].next;
            }
        }
        comps.push(nodes);
    }

    let mut x = vec![0f64; n];
    let mut y = vec![0f64; n];

    let comp_count = comps.len();
    let grid_cols = ((comp_count as f64).sqrt().ceil() as usize).max(1);

    for (ci, nodes) in comps.iter().enumerate() {
        if nodes.len() == 1 {
            let v = nodes[0];
            x[v] = (ci % grid_cols) as f64 * 5.0;
            y[v] = (ci / grid_cols) as f64 * 5.0;
            continue;
        }

        // root: max degree
        let mut root = nodes[0];
        let mut best = deg[root];
        for &v in nodes.iter() {
            if deg[v] > best {
                best = deg[v];
                root = v;
            }
        }

        // local index mapping
        let mut idx = vec![usize::MAX; n];
        for (i, &v) in nodes.iter().enumerate() {
            idx[v] = i;
        }

        // BFS layers
        let mut dist = vec![-1i32; nodes.len()];
        let mut q2 = VecDeque::new();
        dist[idx[root]] = 0;
        q2.push_back(root);
        while let Some(u) = q2.pop_front() {
            let du = dist[idx[u]];
            let mut e = head[u];
            while e != -1 {
                let ei = e as usize;
                let v = edges[ei].to;
                let iv = idx[v];
                if iv != usize::MAX && dist[iv] == -1 {
                    dist[iv] = du + 1;
                    q2.push_back(v);
                }
                e = edges[ei].next;
            }
        }

        let mut maxd = 0i32;
        for i in 0..nodes.len() {
            if dist[i] > maxd {
                maxd = dist[i];
            }
        }
        if maxd < 0 {
            maxd = 0;
        }
        let layers = (maxd as usize) + 1;
        let mut buckets: Vec<Vec<usize>> = vec![Vec::new(); layers];
        for &v in nodes.iter() {
            let d = dist[idx[v]];
            let li = if d < 0 { 0 } else { d as usize };
            buckets[li].push(v);
        }

        buckets[0].sort_unstable_by(|&a, &b| deg[b].cmp(&deg[a]).then_with(|| a.cmp(&b)));

        let mut pos_in_layer = vec![0f64; n];
        for (j, &v) in buckets[0].iter().enumerate() {
            pos_in_layer[v] = j as f64;
        }

        for li in 1..layers {
            // mark prev layer
            let mut prev_mark = vec![false; n];
            for &v in buckets[li - 1].iter() {
                prev_mark[v] = true;
            }
            let mut keyed: Vec<(f64, u64, usize)> = Vec::with_capacity(buckets[li].len());
            for &v in buckets[li].iter() {
                let mut sum = 0.0;
                let mut cnt = 0.0;
                let mut e = head[v];
                while e != -1 {
                    let ei = e as usize;
                    let u = edges[ei].to;
                    if prev_mark[u] {
                        sum += pos_in_layer[u];
                        cnt += 1.0;
                    }
                    e = edges[ei].next;
                }
                let bc = if cnt > 0.0 { sum / cnt } else { pos_in_layer[v] };
                keyed.push((bc, hash_u64(v as u64), v));
            }
            keyed.sort_unstable_by(|a, b| {
                let c = a.0.partial_cmp(&b.0).unwrap_or(Ordering::Equal);
                if c != Ordering::Equal {
                    return c;
                }
                a.1.cmp(&b.1)
            });
            buckets[li].clear();
            for (_, _, v) in keyed {
                pos_in_layer[v] = buckets[li].len() as f64;
                buckets[li].push(v);
            }
        }

        // initial coords
        for (li, layer) in buckets.iter().enumerate() {
            let len = layer.len() as f64;
            let y0 = -(len - 1.0) * 0.5;
            for (j, &v) in layer.iter().enumerate() {
                x[v] = li as f64;
                y[v] = y0 + j as f64;
            }
        }

        // Component-local relaxation (linear-ish); skip for larger comps
        let comp_n = nodes.len();
        if comp_n <= 20_000 {
            let steps = if comp_n <= 200 { 80 } else if comp_n <= 2000 { 50 } else { 20 };
            let k_attr = 0.06;
            let k_rep = 0.02;
            let mut dx = vec![0f64; comp_n];
            let mut dy = vec![0f64; comp_n];

            for _ in 0..steps {
                dx.fill(0.0);
                dy.fill(0.0);

                // grid for near-field repulsion
                let cell = 1.5;
                let mut minx = f64::INFINITY;
                let mut miny = f64::INFINITY;
                let mut maxx = f64::NEG_INFINITY;
                let mut maxy = f64::NEG_INFINITY;
                for &v in nodes.iter() {
                    minx = minx.min(x[v]);
                    miny = miny.min(y[v]);
                    maxx = maxx.max(x[v]);
                    maxy = maxy.max(y[v]);
                }
                let gx0 = (minx / cell).floor() as i32;
                let gy0 = (miny / cell).floor() as i32;
                let gw = ((maxx / cell).floor() as i32 - gx0 + 1).max(1) as usize;
                let gh = ((maxy / cell).floor() as i32 - gy0 + 1).max(1) as usize;

                if gw.saturating_mul(gh) <= 2_000_000 {
                    let mut grid: Vec<Vec<usize>> = vec![Vec::new(); gw * gh];
                    for (i, &v) in nodes.iter().enumerate() {
                        let cx = (x[v] / cell).floor() as i32 - gx0;
                        let cy = (y[v] / cell).floor() as i32 - gy0;
                        let ix = cx.clamp(0, gw as i32 - 1) as usize;
                        let iy = cy.clamp(0, gh as i32 - 1) as usize;
                        grid[iy * gw + ix].push(i);
                    }

                    for iy in 0..gh {
                        for ix in 0..gw {
                            let base = iy * gw + ix;
                            for ny in iy.saturating_sub(1)..=(iy + 1).min(gh - 1) {
                                for nx in ix.saturating_sub(1)..=(ix + 1).min(gw - 1) {
                                    let other = ny * gw + nx;
                                    if other < base {
                                        continue;
                                    }
                                    let a = &grid[base];
                                    let b = &grid[other];
                                    for &ia in a.iter() {
                                        let va = nodes[ia];
                                        let xa = x[va];
                                        let ya = y[va];
                                        for &ib in b.iter() {
                                            if other == base && ib <= ia {
                                                continue;
                                            }
                                            let vb = nodes[ib];
                                            let mut rx = xa - x[vb];
                                            let mut ry = ya - y[vb];
                                            let d2 = rx * rx + ry * ry + 1e-6;
                                            let inv = k_rep / d2;
                                            rx *= inv;
                                            ry *= inv;
                                            dx[ia] += rx;
                                            dy[ia] += ry;
                                            dx[ib] -= rx;
                                            dy[ib] -= ry;
                                        }
                                    }
                                }
                            }
                        }
                    }
                }

                // attraction (avoid double count by v<u)
                for (i, &v) in nodes.iter().enumerate() {
                    let mut e = head[v];
                    while e != -1 {
                        let ei = e as usize;
                        let u = edges[ei].to;
                        if u <= v || comp_id[u] != ci {
                            e = edges[ei].next;
                            continue;
                        }
                        let j = idx[u];
                        if j == usize::MAX {
                            e = edges[ei].next;
                            continue;
                        }
                        let mut ax = x[u] - x[v];
                        let mut ay = y[u] - y[v];
                        let dist = (ax * ax + ay * ay + 1e-6).sqrt();
                        let f = k_attr * (dist - 1.0);
                        ax = ax / dist * f;
                        ay = ay / dist * f;
                        dx[i] += ax;
                        dy[i] += ay;
                        dx[j] -= ax;
                        dy[j] -= ay;
                        e = edges[ei].next;
                    }
                }

                let step = 0.6;
                for (i, &v) in nodes.iter().enumerate() {
                    x[v] += dx[i] * step;
                    y[v] += dy[i] * step;
                }
            }
        }

        // pack into plane
        let col = ci % grid_cols;
        let row = ci / grid_cols;
        let scale = (nodes.len() as f64).sqrt().max(1.0);
        let offx = col as f64 * (scale + 5.0);
        let offy = row as f64 * (scale + 5.0);

        let mut cx = 0.0;
        let mut cy = 0.0;
        for &v in nodes.iter() {
            cx += x[v];
            cy += y[v];
        }
        let inv = 1.0 / (nodes.len() as f64);
        cx *= inv;
        cy *= inv;
        for &v in nodes.iter() {
            x[v] = (x[v] - cx) + offx;
            y[v] = (y[v] - cy) + offy;
        }
    }

    for i in 0..n {
        println!("{:.6} {:.6}", x[i], y[i]);
    }
}

#[inline]
fn hash_u64(mut x: u64) -> u64 {
    x = x.wrapping_add(0x9e3779b97f4a7c15);
    x = (x ^ (x >> 30)).wrapping_mul(0xbf58476d1ce4e5b9);
    x = (x ^ (x >> 27)).wrapping_mul(0x94d049bb133111eb);
    x ^ (x >> 31)
}
