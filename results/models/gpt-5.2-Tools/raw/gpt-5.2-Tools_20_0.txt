use std::cmp::Ordering;
use std::collections::HashMap;
use std::f64;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    #[inline]
    fn add(self, o: Vec3) -> Vec3 {
        Vec3 {
            x: self.x + o.x,
            y: self.y + o.y,
            z: self.z + o.z,
        }
    }
    #[inline]
    fn mul(self, s: f64) -> Vec3 {
        Vec3 {
            x: self.x * s,
            y: self.y * s,
            z: self.z * s,
        }
    }
    #[inline]
    fn dot(self, o: Vec3) -> f64 {
        self.x * o.x + self.y * o.y + self.z * o.z
    }
    #[inline]
    fn norm2(self) -> f64 {
        self.dot(self)
    }
    #[inline]
    fn dist2(self, o: Vec3) -> f64 {
        (self.x - o.x) * (self.x - o.x) + (self.y - o.y) * (self.y - o.y) + (self.z - o.z) * (self.z - o.z)
    }
}

#[derive(Debug)]
struct Hole {
    id: String,
    start: Vec3,
    dir: Vec3,
    len: f64,
    // samples: depth + props
    depths: Vec<f64>,
    props: Vec<Vec<f64>>, // props[p][i]
}

#[derive(Clone, Copy)]
struct CenterStats {
    median: f64,
    mad: f64, // median absolute deviation
    iqr: f64, // q3-q1
}

#[derive(Clone)]
struct Suspect {
    hole_id: String,
    sample_idx: usize,
    prop_name: String,
    confidence: f64,
}

// Fast whitespace tokenizer for stdin
struct Tok {
    s: Vec<u8>,
    i: usize,
}
impl Tok {
    fn new(input: String) -> Self {
        Self {
            s: input.into_bytes(),
            i: 0,
        }
    }
    #[inline]
    fn skip_ws(&mut self) {
        while self.i < self.s.len() && self.s[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
    }
    fn next(&mut self) -> Option<String> {
        self.skip_ws();
        if self.i >= self.s.len() {
            return None;
        }
        let start = self.i;
        while self.i < self.s.len() && !self.s[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
        Some(String::from_utf8_lossy(&self.s[start..self.i]).to_string())
    }
    fn next_usize(&mut self) -> Option<usize> {
        self.next().and_then(|t| t.parse::<usize>().ok())
    }
    fn next_f64(&mut self) -> Option<f64> {
        self.next().and_then(|t| t.parse::<f64>().ok())
    }
}

#[inline]
fn is_finite(x: f64) -> bool {
    x.is_finite()
}

fn normalize_dir(dir: Vec3) -> Vec3 {
    let n2 = dir.norm2();
    if !n2.is_finite() || n2 <= 0.0 {
        return Vec3 { x: 0.0, y: 0.0, z: 1.0 };
    }
    let inv = 1.0 / n2.sqrt();
    dir.mul(inv)
}

#[inline]
fn sample_pos(h: &Hole, depth: f64) -> Vec3 {
    h.start.add(h.dir.mul(depth))
}

fn median_sorted(v: &[f64]) -> f64 {
    if v.is_empty() {
        return f64::NAN;
    }
    let n = v.len();
    if n % 2 == 1 {
        v[n / 2]
    } else {
        0.5 * (v[n / 2 - 1] + v[n / 2])
    }
}

fn quantile_sorted(v: &[f64], q: f64) -> f64 {
    if v.is_empty() {
        return f64::NAN;
    }
    if v.len() == 1 {
        return v[0];
    }
    let q = q.clamp(0.0, 1.0);
    let x = (v.len() - 1) as f64 * q;
    let i = x.floor() as usize;
    let j = (i + 1).min(v.len() - 1);
    let t = x - i as f64;
    v[i] * (1.0 - t) + v[j] * t
}

fn robust_stats(mut data: Vec<f64>) -> Option<CenterStats> {
    data.retain(|&x| is_finite(x));
    if data.len() < 5 {
        return None;
    }
    data.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let med = median_sorted(&data);
    // MAD
    let mut devs: Vec<f64> = data.iter().map(|&x| (x - med).abs()).collect();
    devs.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let mad = median_sorted(&devs);
    // IQR
    let q1 = quantile_sorted(&data, 0.25);
    let q3 = quantile_sorted(&data, 0.75);
    let iqr = (q3 - q1).abs();
    Some(CenterStats {
        median: med,
        mad,
        iqr,
    })
}

#[inline]
fn robust_z(x: f64, st: CenterStats) -> f64 {
    // 0.6745 scales MAD to sigma for normal
    let scale = if st.mad.is_finite() && st.mad > 0.0 {
        st.mad
    } else if st.iqr.is_finite() && st.iqr > 0.0 {
        // IQR to sigma: IQR ~= 1.349 sigma
        st.iqr / 1.349
    } else {
        0.0
    };
    if scale <= 0.0 || !scale.is_finite() {
        0.0
    } else {
        0.6745 * (x - st.median) / scale
    }
}

#[inline]
fn confidence_from_scores(local_score: f64, global_score: f64) -> f64 {
    // Convert scores to a bounded confidence in [0,1).
    // Prefer local score; allow global to boost when local is weak.
    let a = local_score.max(0.0);
    let b = global_score.max(0.0);
    let s = a + 0.5 * b;
    1.0 - (-0.35 * s).exp()
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut tok = Tok::new(input);

    let num_holes = match tok.next_usize() {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let num_props = match tok.next_usize() {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut prop_names: Vec<String> = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        match tok.next() {
            Some(p) => prop_names.push(p),
            None => prop_names.push("prop".to_string()),
        }
    }

    let mut holes: Vec<Hole> = Vec::with_capacity(num_holes);
    // For global stats
    let mut global_vals: Vec<Vec<f64>> = (0..num_props).map(|_| Vec::new()).collect();

    for _ in 0..num_holes {
        let hole_id = match tok.next() {
            Some(v) => v,
            None => break,
        };
        let start_x = match tok.next_f64() {
            Some(v) => v,
            None => break,
        };
        let start_y = match tok.next_f64() {
            Some(v) => v,
            None => break,
        };
        let start_z = match tok.next_f64() {
            Some(v) => v,
            None => break,
        };
        let dir_x = tok.next_f64().unwrap_or(0.0);
        let dir_y = tok.next_f64().unwrap_or(0.0);
        let dir_z = tok.next_f64().unwrap_or(1.0);
        let len = tok.next_f64().unwrap_or(0.0);
        let num_samples = tok.next_usize().unwrap_or(0);

        let start = Vec3 {
            x: start_x,
            y: start_y,
            z: start_z,
        };
        let dir = normalize_dir(Vec3 {
            x: dir_x,
            y: dir_y,
            z: dir_z,
        });

        let mut depths = Vec::with_capacity(num_samples);
        let mut props: Vec<Vec<f64>> = (0..num_props).map(|_| Vec::with_capacity(num_samples)).collect();

        // Handle missing / malformed lines by reading what we can.
        for _ in 0..num_samples {
            let depth = match tok.next_f64() {
                Some(v) => v,
                None => break,
            };
            depths.push(depth);
            for p in 0..num_props {
                let v = tok.next_f64().unwrap_or(f64::NAN);
                props[p].push(v);
                if is_finite(v) {
                    global_vals[p].push(v);
                }
            }
        }

        holes.push(Hole {
            id: hole_id,
            start,
            dir,
            len,
            depths,
            props,
        });
    }

    if holes.is_empty() || num_props == 0 {
        println!("0");
        return;
    }

    // Determine scale and strategy
    let mut total_samples: usize = 0;
    for h in &holes {
        total_samples = total_samples.saturating_add(h.depths.len());
    }

    // Compute global robust stats per property
    let mut global_stats: Vec<Option<CenterStats>> = Vec::with_capacity(num_props);
    for p in 0..num_props {
        global_stats.push(robust_stats(std::mem::take(&mut global_vals[p])));
    }

    // Build point list for spatial indexing (one per sample)
    // Also retain mapping back to hole/sample index
    struct Pt {
        x: f64,
        y: f64,
        z: f64,
        hole_idx: usize,
        sample_idx: usize,
    }
    let mut points: Vec<Pt> = Vec::with_capacity(total_samples);
    for (hi, h) in holes.iter().enumerate() {
        for (si, &d) in h.depths.iter().enumerate() {
            if !d.is_finite() {
                continue;
            }
            let pos = sample_pos(h, d);
            if pos.x.is_finite() && pos.y.is_finite() && pos.z.is_finite() {
                points.push(Pt {
                    x: pos.x,
                    y: pos.y,
                    z: pos.z,
                    hole_idx: hi,
                    sample_idx: si,
                });
            }
        }
    }

    if points.is_empty() {
        println!("0");
        return;
    }

    // Adaptive parameters
    let large = points.len() >= 200_000;
    let medium = points.len() >= 40_000;
    let k_neighbors: usize = if large { 12 } else if medium { 16 } else { 24 };
    let max_suspects: usize = if large { 20_000 } else if medium { 50_000 } else { 200_000 };
    // Thresholds
    let local_z_thresh: f64 = if large { 4.5 } else if medium { 4.0 } else { 3.5 };
    let global_z_gate: f64 = if large { 7.0 } else if medium { 6.5 } else { 6.0 };

    // Build uniform grid index for approximate neighbor lookup
    // Choose cell size based on estimated spacing: use bounding box diag / cbrt(N) * factor
    let mut minx = points[0].x;
    let mut maxx = points[0].x;
    let mut miny = points[0].y;
    let mut maxy = points[0].y;
    let mut minz = points[0].z;
    let mut maxz = points[0].z;
    for p in &points {
        minx = minx.min(p.x);
        maxx = maxx.max(p.x);
        miny = miny.min(p.y);
        maxy = maxy.max(p.y);
        minz = minz.min(p.z);
        maxz = maxz.max(p.z);
    }
    let dx = (maxx - minx).abs().max(1e-9);
    let dy = (maxy - miny).abs().max(1e-9);
    let dz = (maxz - minz).abs().max(1e-9);
    let diag = (dx * dx + dy * dy + dz * dz).sqrt();
    let n = points.len() as f64;
    let base = (diag / n.cbrt()).max(1e-6);
    let cell_size = if large { base * 1.4 } else if medium { base * 1.2 } else { base * 1.0 };
    let inv_cs = 1.0 / cell_size;

    #[inline]
    fn cell_coord(v: f64, minv: f64, inv_cs: f64) -> i32 {
        ((v - minv) * inv_cs).floor() as i32
    }
    #[inline]
    fn key(ix: i32, iy: i32, iz: i32) -> u64 {
        // pack signed 21-bit coords into u64 (enough for typical extents)
        let mask: i64 = (1i64 << 21) - 1;
        let x = (ix as i64) & mask;
        let y = (iy as i64) & mask;
        let z = (iz as i64) & mask;
        ((x as u64) << 42) | ((y as u64) << 21) | (z as u64)
    }

    let mut grid: HashMap<u64, Vec<usize>> = HashMap::with_capacity(points.len() * 2);
    for (pi, p) in points.iter().enumerate() {
        let ix = cell_coord(p.x, minx, inv_cs);
        let iy = cell_coord(p.y, miny, inv_cs);
        let iz = cell_coord(p.z, minz, inv_cs);
        grid.entry(key(ix, iy, iz)).or_default().push(pi);
    }

    // Optional per-hole/per-property stats for very small data to catch internal inconsistencies
    let compute_hole_stats = points.len() <= 60_000 && num_props <= 64;
    let mut hole_stats: Option<Vec<Vec<Option<CenterStats>>>> = None;
    if compute_hole_stats {
        let mut hs: Vec<Vec<Option<CenterStats>>> = Vec::with_capacity(holes.len());
        for h in &holes {
            let mut per_prop: Vec<Option<CenterStats>> = Vec::with_capacity(num_props);
            for p in 0..num_props {
                per_prop.push(robust_stats(h.props[p].clone()));
            }
            hs.push(per_prop);
        }
        hole_stats = Some(hs);
    }

    // Main detection
    let mut suspects: Vec<Suspect> = Vec::new();
    suspects.reserve((points.len() / 50).min(max_suspects));

    // Pre-alloc neighbor candidate list to reduce allocs
    let mut cand: Vec<(f64, usize)> = Vec::with_capacity(512);
    let mut neigh_vals: Vec<f64> = Vec::with_capacity(64);

    for (pi, p) in points.iter().enumerate() {
        // Collect candidates from nearby grid cells, expanding radius until enough points
        cand.clear();
        let ix = cell_coord(p.x, minx, inv_cs);
        let iy = cell_coord(p.y, miny, inv_cs);
        let iz = cell_coord(p.z, minz, inv_cs);

        let mut radius: i32 = 1;
        let mut limit = if large { 2 } else if medium { 3 } else { 4 };
        while radius <= limit && cand.len() < k_neighbors * 5 {
            for dx in -radius..=radius {
                for dy in -radius..=radius {
                    for dz in -radius..=radius {
                        let k = key(ix + dx, iy + dy, iz + dz);
                        if let Some(list) = grid.get(&k) {
                            for &q in list {
                                if q == pi {
                                    continue;
                                }
                                let pp = &points[q];
                                let d2 = (p.x - pp.x) * (p.x - pp.x)
                                    + (p.y - pp.y) * (p.y - pp.y)
                                    + (p.z - pp.z) * (p.z - pp.z);
                                if d2.is_finite() {
                                    cand.push((d2, q));
                                }
                            }
                        }
                    }
                }
            }
            radius += 1;
            if radius > limit && cand.len() < k_neighbors {
                // relax expansion a bit for sparse data
                limit = (limit + 1).min(8);
            }
        }

        if cand.is_empty() {
            continue;
        }
        cand.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(Ordering::Equal));

        // neighbor indices (up to k)
        let take_k = cand.len().min(k_neighbors);
        let hole_idx = p.hole_idx;
        let sample_idx = p.sample_idx;

        // Basic sanity: duplicate depth or reversed interval inside hole can cause weirdness; we still just compute.
        for prop_idx in 0..num_props {
            let x = holes[hole_idx].props[prop_idx].get(sample_idx).copied().unwrap_or(f64::NAN);
            if !is_finite(x) {
                continue;
            }

            // Gate using global stats if available: only spend work on values that are somewhat extreme,
            // or when dataset is small.
            let mut global_z_abs = 0.0;
            if let Some(gs) = global_stats[prop_idx] {
                global_z_abs = robust_z(x, gs).abs();
            }
            if points.len() > 10_000 && global_z_abs < 2.5 {
                // Likely normal; skip local check for speed on big datasets.
                continue;
            }

            // Gather neighbor values for the same property
            neigh_vals.clear();
            for j in 0..take_k {
                let q = cand[j].1;
                let qq = &points[q];
                let v = holes[qq.hole_idx].props[prop_idx]
                    .get(qq.sample_idx)
                    .copied()
                    .unwrap_or(f64::NAN);
                if is_finite(v) {
                    neigh_vals.push(v);
                }
            }
            if neigh_vals.len() < 6 {
                // Not enough spatial neighbors with valid data; fall back to hole-level stats if present.
                if let (Some(hs), true) = (hole_stats.as_ref(), compute_hole_stats) {
                    if let Some(st) = hs[hole_idx][prop_idx] {
                        let z = robust_z(x, st).abs();
                        if z >= (local_z_thresh + 0.5) && global_z_abs >= 2.0 {
                            let conf = confidence_from_scores(z - local_z_thresh, (global_z_abs - global_z_gate).max(0.0));
                            if conf > 0.01 {
                                suspects.push(Suspect {
                                    hole_id: holes[hole_idx].id.clone(),
                                    sample_idx,
                                    prop_name: prop_names[prop_idx].clone(),
                                    confidence: conf,
                                });
                            }
                        }
                    }
                }
                continue;
            }

            // Compute robust local stats from neighbors
            neigh_vals.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
            let loc_med = median_sorted(&neigh_vals);
            let mut devs: Vec<f64> = neigh_vals.iter().map(|&v| (v - loc_med).abs()).collect();
            devs.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
            let loc_mad = median_sorted(&devs);
            let q1 = quantile_sorted(&neigh_vals, 0.25);
            let q3 = quantile_sorted(&neigh_vals, 0.75);
            let loc_iqr = (q3 - q1).abs();
            let st = CenterStats {
                median: loc_med,
                mad: loc_mad,
                iqr: loc_iqr,
            };
            let local_z_abs = robust_z(x, st).abs();

            // Also consider simple neighbor spread and distance-weight for additional confidence
            // but keep cheap: compare to IQR-based fences.
            let scale = if loc_iqr > 0.0 { loc_iqr } else { loc_mad };
            let fence = if scale > 0.0 { 3.0 * scale } else { 0.0 };
            let far_from_med = (x - loc_med).abs();

            // Score: primarily local robust z; global only for "obvious" gross errors.
            let local_score = (local_z_abs - local_z_thresh).max(0.0);
            let global_score = (global_z_abs - global_z_gate).max(0.0);
            let mut conf = confidence_from_scores(local_score, global_score);

            // Extra nudge if it's far beyond a robust fence
            if fence > 0.0 && far_from_med > fence {
                conf = (conf + 0.15).min(0.999_999);
            }

            // If local neighbors are very tight (small MAD/IQR) and the point deviates, bump.
            if (loc_mad > 0.0 && loc_mad < 1e-8) || (loc_iqr > 0.0 && loc_iqr < 1e-8) {
                if far_from_med > 0.0 {
                    conf = (conf + 0.1).min(0.999_999);
                }
            }

            // Filter: require local anomaly or extremely global anomaly
            let pass = local_z_abs >= local_z_thresh || global_z_abs >= (global_z_gate + 1.5);
            if pass && conf > 0.02 {
                suspects.push(Suspect {
                    hole_id: holes[hole_idx].id.clone(),
                    sample_idx,
                    prop_name: prop_names[prop_idx].clone(),
                    confidence: conf,
                });
                if suspects.len() >= max_suspects {
                    break;
                }
            }
        }

        if suspects.len() >= max_suspects {
            break;
        }
    }

    // Sort by confidence desc; stable-ish tie-breakers for determinism
    suspects.sort_by(|a, b| {
        b.confidence
            .partial_cmp(&a.confidence)
            .unwrap_or(Ordering::Equal)
            .then_with(|| a.hole_id.cmp(&b.hole_id))
            .then_with(|| a.sample_idx.cmp(&b.sample_idx))
            .then_with(|| a.prop_name.cmp(&b.prop_name))
    });

    // De-duplicate exact entries (can occur with multiple paths through fallbacks)
    let mut out: Vec<Suspect> = Vec::with_capacity(suspects.len());
    let mut last_key: Option<(String, usize, String)> = None;
    for s in suspects {
        let k = (s.hole_id.clone(), s.sample_idx, s.prop_name.clone());
        if last_key.as_ref() == Some(&k) {
            continue;
        }
        last_key = Some(k);
        out.push(s);
    }

    println!("{}", out.len());
    for s in out {
        // sample_idx is 0-based as read order; that's explicit in output spec
        // confidence: 0..1
        println!(
            "{} {} {} {:.6}",
            s.hole_id,
            s.sample_idx,
            s.prop_name,
            s.confidence
        );
    }
}
