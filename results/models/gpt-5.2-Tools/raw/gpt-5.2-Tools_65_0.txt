// Mission-control scheduler benchmark
// Node.js TypeScript program: reads one JSON on stdin, outputs JSON schedule or order.

declare const require: any;
const fs = require('fs');

type TaskInput = {
  id: string | number;
  duration: number;
  deps?: Array<string | number>;
  deadline?: number | null;
  weight?: number | null;
};

type Input = {
  machineCount: number;
  tasks: TaskInput[];
};

type ScheduleItem = { id: string | number; start: number; machine: number };

function asNumber(x: any, dflt: number): number {
  const n = Number(x);
  return Number.isFinite(n) ? n : dflt;
}

function stableStringId(id: string | number): string {
  return typeof id === 'string' ? id : String(id);
}

class MinHeap<T> {
  private a: T[] = [];
  constructor(private key: (v: T) => number) {}
  size(): number {
    return this.a.length;
  }
  push(v: T): void {
    const a = this.a;
    a.push(v);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.key(a[p]) <= this.key(a[i])) break;
      const t = a[p];
      a[p] = a[i];
      a[i] = t;
      i = p;
    }
  }
  pop(): T | undefined {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop()!;
    if (a.length > 0) {
      a[0] = last;
      let i = 0;
      while (true) {
        const l = i * 2 + 1;
        const r = l + 1;
        let s = i;
        if (l < a.length && this.key(a[l]) < this.key(a[s])) s = l;
        if (r < a.length && this.key(a[r]) < this.key(a[s])) s = r;
        if (s === i) break;
        const t = a[i];
        a[i] = a[s];
        a[s] = t;
        i = s;
      }
    }
    return top;
  }
}

function buildGraph(tasks: TaskInput[]) {
  const n = tasks.length;
  const idToIndex = new Map<string, number>();
  for (let i = 0; i < n; i++) idToIndex.set(stableStringId(tasks[i].id), i);

  const succ: number[][] = Array.from({ length: n }, () => []);
  const indeg0 = new Int32Array(n);

  for (let i = 0; i < n; i++) {
    const deps = tasks[i].deps || [];
    for (const d of deps) {
      const j = idToIndex.get(stableStringId(d));
      if (j === undefined) continue;
      succ[j].push(i);
      indeg0[i]++;
    }
  }
  return { succ, indeg0 };
}

function makePriorityFn(tasks: TaskInput[], succ: number[][]) {
  const n = tasks.length;
  const dur = new Float64Array(n);
  const ddl = new Float64Array(n);
  const w = new Float64Array(n);
  for (let i = 0; i < n; i++) {
    dur[i] = Math.max(0, asNumber(tasks[i].duration, 0));
    const d = tasks[i].deadline;
    ddl[i] = d === undefined || d === null ? Number.POSITIVE_INFINITY : asNumber(d, Number.POSITIVE_INFINITY);
    const ww = tasks[i].weight;
    w[i] = ww === undefined || ww === null ? 1 : Math.max(0, asNumber(ww, 1));
  }

  // Downstream criticality (approx longest-path) via reverse topo on acyclic portion.
  const { indeg0 } = buildGraph(tasks);
  const indeg = new Int32Array(indeg0);
  const q: number[] = [];
  for (let i = 0; i < n; i++) if (indeg[i] === 0) q.push(i);
  const topo: number[] = [];
  for (let h = 0; h < q.length; h++) {
    const u = q[h];
    topo.push(u);
    for (const v of succ[u]) {
      indeg[v]--;
      if (indeg[v] === 0) q.push(v);
    }
  }
  if (topo.length < n) {
    const seen = new Uint8Array(n);
    for (const x of topo) seen[x] = 1;
    for (let i = 0; i < n; i++) if (!seen[i]) topo.push(i);
  }
  const down = new Float64Array(n);
  for (let k = topo.length - 1; k >= 0; k--) {
    const u = topo[k];
    let best = 0;
    for (const v of succ[u]) if (down[v] > best) best = down[v];
    down[u] = dur[u] + best;
  }

  const outdeg = new Int32Array(n);
  for (let i = 0; i < n; i++) outdeg[i] = succ[i].length;

  return (i: number, currentTime: number): number => {
    const d = ddl[i];
    const slack = d === Number.POSITIVE_INFINITY ? Number.POSITIVE_INFINITY : d - (currentTime + dur[i]);
    const urgency = slack === Number.POSITIVE_INFINITY ? 0 : -slack;
    return w[i] * 10000 + urgency * 5 + down[i] * 2 + outdeg[i] * 0.1 + dur[i] * 0.01;
  };
}

function scheduleList(tasks: TaskInput[], machineCount: number): { schedule: ScheduleItem[] } | null {
  const n = tasks.length;
  if (n === 0) return { schedule: [] };
  const m = Math.max(1, Math.floor(asNumber(machineCount, 1)));

  const { succ, indeg0 } = buildGraph(tasks);
  const indeg = new Int32Array(indeg0);

  const dur = new Float64Array(n);
  for (let i = 0; i < n; i++) dur[i] = Math.max(0, asNumber(tasks[i].duration, 0));

  const priorityFn = makePriorityFn(tasks, succ);
  let currentTime = 0;
  const ready = new MinHeap<{ i: number; k: number }>((x) => x.k);
  for (let i = 0; i < n; i++) if (indeg[i] === 0) ready.push({ i, k: -priorityFn(i, 0) });

  const machineFreeAt = new Float64Array(m);
  for (let i = 0; i < m; i++) machineFreeAt[i] = 0;
  const running = new MinHeap<{ end: number; machine: number; i: number }>((x) => x.end);

  const started = new Uint8Array(n);
  const scheduled: ScheduleItem[] = [];
  let doneCount = 0;

  function freeMachines(atTime: number): number[] {
    const free: number[] = [];
    for (let mi = 0; mi < m; mi++) if (machineFreeAt[mi] <= atTime + 1e-9) free.push(mi);
    return free;
  }

  while (doneCount < n) {
    const freeNow = freeMachines(currentTime);
    if (freeNow.length > 0 && ready.size() > 0) {
      freeNow.sort((a, b) => a - b);
      let idx = 0;
      while (idx < freeNow.length && ready.size() > 0) {
        const mi = freeNow[idx++];
        const it = ready.pop()!;
        const i = it.i;
        if (started[i]) continue;
        started[i] = 1;
        const start = currentTime;
        const end = start + dur[i];
        machineFreeAt[mi] = end;
        running.push({ end, machine: mi, i });
        scheduled.push({ id: tasks[i].id, start: Math.trunc(start), machine: mi });
      }
      continue;
    }

    const next = running.pop();
    if (!next) return null; // cycle/deadlock
    currentTime = next.end;
    doneCount++;
    for (const v of succ[next.i]) {
      indeg[v]--;
      if (indeg[v] === 0) ready.push({ i: v, k: -priorityFn(v, currentTime) });
    }
  }

  return { schedule: scheduled };
}

function topoOrder(tasks: TaskInput[]): Array<string | number> {
  const n = tasks.length;
  const { succ, indeg0 } = buildGraph(tasks);
  const indeg = new Int32Array(indeg0);
  const priorityFn = makePriorityFn(tasks, succ);
  const ready = new MinHeap<{ i: number; k: number }>((x) => x.k);
  for (let i = 0; i < n; i++) if (indeg[i] === 0) ready.push({ i, k: -priorityFn(i, 0) });
  const order: Array<string | number> = [];
  while (ready.size() > 0) {
    const u = ready.pop()!.i;
    order.push(tasks[u].id);
    for (const v of succ[u]) {
      indeg[v]--;
      if (indeg[v] === 0) ready.push({ i: v, k: -priorityFn(v, 0) });
    }
  }
  if (order.length < n) {
    const seen = new Set(order.map(stableStringId));
    for (const t of tasks) if (!seen.has(stableStringId(t.id))) order.push(t.id);
  }
  return order;
}

function main() {
  const input: string = fs.readFileSync(0, 'utf8');
  const data: Input = JSON.parse(input);
  const tasks = Array.isArray((data as any).tasks) ? (data as any).tasks : [];
  const machineCount = asNumber((data as any).machineCount, 1);

  const sched = scheduleList(tasks, machineCount);
  if (sched) process.stdout.write(JSON.stringify(sched));
  else process.stdout.write(JSON.stringify({ order: topoOrder(tasks) }));
}

main();
