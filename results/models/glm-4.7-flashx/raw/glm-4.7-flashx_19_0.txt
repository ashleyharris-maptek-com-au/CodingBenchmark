        // We run the input through a "FastPrepass" to count total voxels.
        // This allows us to decide if we use a fast heuristic approach or the slow A* solver.

        {
            throw new Exception("Invalid input: No header found.");
        }

        {
            throw new Exception("Invalid header: expected 6 integers");
        }

        int yc = int.Parse(parts[1]);
        int zc = int.Parse(parts[2]);
        double xb = double.Parse(parts[3]);
        double yb = double.Parse(parts[4]);
        double zb = double.Parse(parts[5]);

    private static void SolveHeuristic(int xc, int yc, int zc, double xb, double yb, double zb, StreamReader reader)
    {
        // 1. Fast Count of Voxels.
        // If voxels are sparse, we can use a sparse map.
        // If dense, 2D array of values (0..15).
        
        // Read all raw bytes.
        byte[] buffer = new byte[reader.BaseStream.Length];
        reader.BaseStream.Read(buffer, 0, buffer.Length);
        int idx = 0;
        
        var map = new List<(int x, int y, int z, int value)>();
        
        for (int x = 0; x < xc; x++)
        {
            for (int y = 0; y < yc; y++)
            {
                for (int z = 0; z < zc; z++)
                {
                    int d = buffer[idx++];
                    int mass = d % 16;
                    int value = d >> 4;
                    if (mass > 0)
                    {
                        map.Add((x, y, z, value));
                    }
                }
            }
        }
        
        // 2. Target Area Approximation.
        // Processing plant is (0,0,0).
        // We want to bring things closer to (0,0) (Z doesn't matter for the plant except it's there).
        // And for Dirt, we want to dump it.
        // Since we can dump anywhere, we just need to find the "closest" valid dump spot to the Ore?
        // Or just Dump to (0,0) if it's ore?
        // For Ore: Distance is just X/Y distance to 0.
        // For Dirt: If we dump it at (0,0), we pay $1000/ton.
        // If we dump it at (1,0,0) (e.g.), cost is $1/voxel.
        
        // Let's create a "Clearance Map".
        // Start from Plant.
        // Grow a frontier: Cells that are reachable.
        // When we hit a voxel:
        //   If Value > 0: Keep it.
        //   If Value == 0: Dump it immediately (to a neighbor or on top) to clear the path.
        //   Actually, the prompt implies the "Main" route is the path we take.
        //   We just walk the route.
        
        // Sort by distance to (0,0) to prioritize that area.
        map.Sort((a, b) => 
        {
            int da = Math.Abs(a.x) + Math.Abs(a.y);
            int db = Math.Abs(b.x) + Math.Abs(b.y);
            return da.CompareTo(db);
        });
        
        List<Movement> plan = new List<Movement>();
        
        // We simulate the path.
        // We start at (0,0,zc) (Surface).
        int sx = 0, sy = 0, sz = zc - 1;
        
        // We use a simple stack for backtracking? No, too slow for 500k items.
        // We use a greedy strategy: Move towards (0,0,0) in X/Y. If stuck, try next best.
        // To maximize profit, we prioritize moving valuable ore.
        // Sorting ensures we process ore closest to the plant first.
        
        // Queue of Voxels to visit.
        var queue = new Queue<(int x, int y, int z, int value, int px, int py, int pz)>();
        
        // Add everything to queue
        foreach (var v in map)
        {
            // We want to dig it. To dig it, we need to reach its location.
            // We can only reach it if we dig a path to it.
            // For this heuristic, we assume we walk along X/Y lines.
            // We will add "trips".
            
            // Strategy:
            // For each voxel, we plan a trip: (CurrentPos) -> (Voxel) -> (Destination).
            // Destination:
            //   If Value > 0: (0,0,0).
            //   If Value == 0: Find the *lowest* reachable Z neighbor (or just one if available) to dump.
            
            // To know "reachable", we need a Z-map.
            // But we can't afford a full 3D map for large worlds?
            // 5000x5000x5000 is huge. 125 Billion voxels. Impossible.
            // "Size could be anything... handle extremes". 
            // This implies sparse data or clever logic.
            
            // We have a sparse list of voxels.
            // We need to find neighbors efficiently.
            // We can use a dictionary for neighbors.
            
            // We need to solve the "Dirt Dumping" part.
            // Dumping dirt just creates a new obstacle.
            // If we have dirt at (x,y,z), and we move to (nx, ny, nz) to dump it,
            // we create a block there.
            // This might block future paths.
            // Solution: Don't dump in front of you. Dump behind or to the side.
            
            // Robust Heuristic:
            // 1. Sort map by X then Y (Grid walk).
            // 2. Process each cell (x,y,z) in order.
            // 3. Calculate path from current pos to (x,y,z).
            // 4. Add path to plan.
            // 5. If (x,y,z) is Ore:
            //    Add path from (x,y,z) to (0,0,0).
            //    Delete (x,y,z).
            //    Move current pos to (0,0,0).
            //    Process plant (value, fuel cost, profit).
            //    Reset current pos to (0,0,0).
            // 6. If (x,y,z) is Dirt:
            //    Find a neighbor (nx, ny, nz) with lowest Z that is reachable?
            //    If none, dump on top (x, y, z+1).
            
            queue.Enqueue((v.x, v.y, v.z, v.value, sx, sy, sz));
        }
        
        while (queue.Count > 0)
        {
            var (vx, vy, vz, val, px, py, pz) = queue.Dequeue();
            
            // Calculate Path to (vx, vy, vz)
            // Use simple A* here, since the segment is short.
            var path = AStarSimple(px, py, pz, vx, vy, vz);
            
            if (path == null) continue; // Unreachable
            
            // Add path to plan
            for (int i = 1; i < path.Count; i++)
            {
                var n = path[i];
                plan.Add(new Movement(n.x, n.y, n.z, 0, 0, 0)); // Dig at target
                // Actually, AStarSimple moves to the target. We dig it.
                // The "Move" is implicit in the step.
                // But the step needs to be explicit for the "Move" command?
                // "digX digY digZ stepX1 stepY1 stepX2 stepY2 ... destX destY"
                // It seems we move step-by-step.
                // Let's interpret path nodes as "Dig locations".
                // If path has 5 nodes, we move from 1 to 2 to 3 to 4 to 5.
                // 1. Dig 2 (from 1).
                // 2. Dig 3 (from 2).
                // ...
                
                // My AStarSimple returns list of positions. 
                // Let's refine the loop.
            }
            
            // Correct Path Application:
            // We are at P_prev (px,py,pz). Target is V (vx,vy,vz).
            // Move P_prev -> V.
            // Add Move command for the segment P_prev -> V.
            // Then "Dig" at V.
            
            // Update current position to V.
            // Apply the voxel logic.
        }
        
        // The logic above is a bit tangled. Let's rewrite the "Heuristic" block to be cleaner.
        // Strategy: Sort cells by X, then Y.
        // We iterate.
        // We keep a Current Rover Position.
        // We need to move Rover to Cell.
        // Then execute Cell logic.
        
        var sortedCells = map.ToArray(); // Already sorted roughly by dist to 0
        // Re-sort strictly by X then Y for grid walking
        Array.Sort(sortedCells, (a, b) => 
        {
            if (a.x != b.x) return a.x.CompareTo(b.x);
            return a.y.CompareTo(b.y);
        });
        
        // We need to handle "Z" logic.
        // If we are at (x, y, z) and need to go to (nx, ny, nz).
        // We move in XY plane (Flat).
        // We might need to dig down/up along the way?
        // "Trace path... ensure path doesn't climb uphill more than 1 Z per X/Y".
        // This constraint applies to the path we traverse.
        
        // Let's iterate cells.
        int curX = 0, curY = 0, curZ = zc - 1; // Surface
        
        foreach (var v in sortedCells)
        {
            if (v.value == 0) continue; // Skip dirt (handled separately or later)
            // v.value > 0 is Ore.
            
            // We need to get there.
            // Since sorted by X/Y, we can just move from cur to v.
            // We need to solve the path between (curX, curY, curZ) and (v.x, v.y, v.z).
            // We will solve this sub-path with A*.
            
            // Get path.
            // But we need to respect Z constraints.
            // The path should be:
            // 1. Move (cur) to (v) in XY plane (flat). Keep Z constant?
            // If we keep Z constant, we are fine (Flat).
            // But we might be on a slope?
            // "Ground is stable... surface starts at zc-1".
            // Actually, the input defines the world.
            // Let's assume the Rover starts on the Surface.
            // It moves Flat X/Y.
            
            // Wait, if we move Flat X/Y, we might cross non-surface?
            // No, we only dig at the target.
            // But we need to move over dirt/air.
            // We can walk over dirt.
            
            // So, Move (curX, curY, curZ) -> (v.x, v.y, v.z) is a Flat move.
            // Dist = |v.x-curX| + |v.y-curY|.
            // We can split this into individual steps.
            
            // However, if we are high up (Z) and need to go to a low Z?
            // We can't move Downhill for free? "Downhill is free".
            // So moving Z doesn't cost fuel?
            // "Downhill is free". Moving uphill costs.
            // If we move Flat X/Y and Z is constant, it's 0 fuel.
            // If Z changes, it's a slope.
            
            // Let's use a local A* for this single jump.
            // Start: (curX, curY, curZ).
            // Target: (v.x, v.y, v.z).
            
            // The A* needs to consider "Air" (distance 1) and "Dirt" (distance 1, costs 0 fuel).
            // It just needs to find the path.
            
            // Since it's a single jump, A* is instant.
            var jumpPath = AStarSimple(curX, curY, curZ, v.x, v.y, v.z);
            
            if (jumpPath == null) continue; // Unreachable (shouldn't happen in sparse grid walk)
            
            // Execute Jump.
            // Move along jumpPath.
            for (int i = 0; i < jumpPath.Count - 1; i++)
            {
                var curr = jumpPath[i];
                var next = jumpPath[i+1];
                // Add movement command
                // Command format: N digX digY digZ stepX1 stepY1 stepX2 stepY2 ... destX destY
                // This seems to imply a path planning command.
                // Let's assume the output needs to be a list of operations.
                
                // Operation: Move from Current to Next.
                // Add current as source?
                // Format: "M digX digY digZ stepX1 stepY1 ... destX destY"
                // This looks like: Dig at digX.. Move step1.. step2.. Move to dest..
                
                // Let's just output the steps.
                plan.Add(new Movement(next.x, next.y, next.z, curr.x - next.x, curr.y - next.y, curr.z - next.z));
            }
            
            // Now we are at (v.x, v.y, v.z).
            // It's Ore.
            // Move to Plant (0,0,0).
            var plantPath = AStarSimple(v.x, v.y, v.z, 0, 0, 0);
            
            // Add plant path steps.
            for (int i = 0; i < plantPath.Count - 1; i++)
            {
                var curr = plantPath[i];
                var next = plantPath[i+1];
                plan.Add(new Movement(next.x, next.y, next.z, curr.x - next.x, curr.y - next.y, curr.z - next.z));
            }
            
            // Process Ore.
            // We dig at (0,0,0).
            // We pay: CarryCost * Distance + Profit.
            // Distance from v to plant.
            int dist = Math.Abs(v.x) + Math.Abs(v.y);
            int fuel = dist * CarryCostFlat;
            int value = v.value;
            int profit = value * value; // Base profit logic?
            // "Pays $1000 per ton per unit of value" -> Value * Mass * 1000.
            // mass = 1 (min).
            int tonProfit = value * 1000;
            
            // Accumulate stats?
            // We need to output "M" (Number of operations) and the ops.
            // We don't need to print fuel stats to stdout, just the operations.
            
            // Remove Ore from world.
            v.value = -1; // Mark as removed
            
            // Update Cur Pos to Plant.
            curX = 0; curY = 0; curZ = 0; // Plant is at 0,0,0.
        }
        
        // Now handle Dirt.
        // Dirt doesn't need to go to plant (costly).
        // We want to dump it to "clear space" or "nearest spot".
        // Since we are walking the grid X/Y sorted, we encounter dirt cells.
        // If we encounter a dirt cell, we must dig it.
        // But we shouldn't dig it if we don't need to (it blocks future X/Y lines? No, X/Y sorted means we process in bands).
        // Actually, sorting by X/Y works well for dirt too.
        // We dig dirt cells and dump them.
        // Dumping: Find a neighbor with lowest Z (that exists).
        
        foreach (var v in map)
        {
            if (v.value != 0) continue;
            
            // Dig dirt at v.
            // Move to v.
            // Dump it.
            
            // Move to v.
            var jumpPath = AStarSimple(curX, curY, curZ, v.x, v.y, v.z);
            for (int i = 0; i < jumpPath.Count - 1; i++)
            {
                var curr = jumpPath[i];
                var next = jumpPath[i+1];
                plan.Add(new Movement(next.x, next.y, next.z, curr.x - next.x, curr.y - next.y, curr.z - next.z));

            foreach (var (nx, ny, nz) in GetNeighbors(v.x, v.y, v.z))
            {
                // If neighbor exists (value > 0 or value == 0)
                // Wait, we can only dump on "Ground".
                // We can check neighbors.
                // We need to check if (nx, ny, nz) is a valid voxel (mass > 0).
                // Or is it Air? (value 0, mass 0).
                // If it's Air, we can't dump (must be ground).
                // So we look for mass > 0.
                
                var nVal = GetNeighborValue(v, nx, ny, nz);
                if (nVal > 0)
                {
                    // Move to neighbor.
                    // Move from v to (nx, ny, nz).
                    plan.Add(new Movement(nx, ny, nz, v.x - nx, v.y - ny, v.z - nz));
                    
                    // Add dirt to (nx, ny, nz).
                    // Increase value by 1? No, dirt is value 0.
                    // Mass is 1. Value 0.
                    // It stays 0.
                    // We just update the list to reflect the new position? 
                    // No, the list contains the source voxels.
                    // We need to update the "map" conceptually.
                    
                    // Mark v as gone.
                    v.value = -1;
                    
                    // Mark (nx, ny, nz) as "changed".
                    // We'll handle this by ensuring we don't process (nx, ny, nz) again if it was in the list.
                    // But list is fixed now.
                    // Alternative: Don't modify the list. Just move the dirt.
                    // This is fine.
                    
                    dumped = true;
                    break;
                }
            }
            
            if (!dumped)
            {
                // Dump on top.
                // Move up.
                plan.Add(new Movement(v.x, v.y, v.z + 1, 0, 0, 1));
                // Remove v.
                v.value = -1;
            }
            
            // Update Cur Pos.
            curX = v.x; curY = v.y; curZ = v.z; // Assuming we moved somewhere.
        }
        
        // Final moves: Back to Plant?
        // "Goal is to get valuable ore...".
        // Usually implies you stop when done.
        // But maybe you should return to 0,0?
        // If we end at 0,0, it's cleaner.
        // Let's check distance.
        if (curX != 0 || curY != 0)
        {
             var finalPath = AStarSimple(curX, curY, curZ, 0, 0, 0);
             for (int i = 0; i < finalPath.Count - 1; i++)
             {
                var curr = finalPath[i];
                var next = finalPath[i+1];
                plan.Add(new Movement(next.x, next.y, next.z, curr.x - next.x, curr.y - next.y, curr.z - next.z));
             }
        }
        
        OutputPlan(plan);
    }

    // --- A* Implementation ---
    // Returns List of nodes from Start to End.
    // Nodes include Start? Yes.
    // Includes End? Yes.
    // Neighbors: 4-connected (X, Y).
    // Constraint: Max slope 1 Z per X/Y step.
    // This means for every step in X or Y, DeltaZ <= 1 (in absolute value? or just max 1?).
    // "Climb uphill slope - 1 Z voxel per X/Y voxel". So DeltaZ <= 1.
    // "Downhill" implies negative DeltaZ.
    // Cost: Flat = 1, Up = 5, Down = 0.
    
    private static List<(int x, int y, int z)> AStarSimple(int sx, int sy, int sz, int ex, int ey, int ez)
    {
        // Check if same
        if (sx == ex && sy == ey && sz == ez) return new List<(int x, int y, int z)> { (sx, sy, sz) };

        var openSet = new SortedList<long, (int x, int y, int z, int f, int g, int px, int py, int pz, int cost)>(); // Priority Queue
        var cameFrom = new Dictionary<(int, int, int), (int, int, int, int)>(); // Key: Current, Val: Parent
        
        // Node: (x,y,z)
        // Initial Cost 0.
        long h = Math.Abs(ex - sx) + Math.Abs(ey - sy); // Manhattan + 0 Z h. (Z cost is exact).
        long f = 0 + h;
        
        openSet.Add(f, (sx, sy, sz, f, 0, -1, -1, -1, 0));
        
        var explored = new HashSet<(int, int, int)>();
        explored.Add((sx, sy, sz));
        
        while (openSet.Count > 0)
        {
            var key = openSet.Keys[0];
            var (x, y, z, f, g, px, py, pz, cost) = openSet.Values[key];
            openSet.RemoveAt(0);
            
            if (x == ex && y == ey && z == ez)
            {
                // Reconstruct path
                var path = new List<(int x, int y, int z)>();
                var curr = (x, y, z);
                path.Add(curr);
                while (cameFrom.ContainsKey(curr))
                {
                    var prev = cameFrom[curr];
                    path.Add((prev.Item1, prev.Item2, prev.Item3));
                    curr = (prev.Item1, prev.Item2, prev.Item3);
                }
                path.Reverse();
                return path;
            }
            
            foreach (var (nx, ny, nz) in GetNeighbors(x, y, z))
            {
                if (explored.Contains((nx, ny, nz))) continue;
                
                // Cost to move to (nx, ny, nz) from (x, y, z)
                int moveCost = GetMoveCost(x, y, z, nx, ny, nz);
                int newG = g + moveCost;
                
                int newH = Math.Abs(ex - nx) + Math.Abs(ey - ny);
                int newF = newG + newH;
                
                if (!openSet.ContainsKey(newF) || newG < openSet[newF].g) // Standard A* check
                {
                    openSet.Add(newF, (nx, ny, nz, newF, newG, x, y, z, moveCost));
                    cameFrom[(nx, ny, nz)] = (x, y, z);
                }
            }
        }
        
        return null; // No path
    }
    
    private static int GetMoveCost(int x1, int y1, int z1, int x2, int y2, int z2)
    {
        int dx = x2 - x1;
        int dy = y2 - y1;
        int dz = z2 - z1;
        
        if (dx == 0 && dy == 0) return 0; // Stationary?
        
        int steps = Math.Abs(dx) + Math.Abs(dy);
        if (steps == 0) return 0;
        
        int zDiff = Math.Abs(dz);
        if (zDiff > steps) return int.MaxValue; // Constraint violation (max 1 slope)
        
        int cost = steps * CarryCostFlat;
        
        // Add slope costs
        // "Climb uphill slope...". "Downhill is free".
        // If z2 > z1 (Uphill), cost is increased.
        if (dz > 0) cost += dz * CarryCostSlope;
        
        return cost;
    }

    private static IEnumerable<(int x, int y, int z)> GetNeighbors(int x, int y, int z)
    {
        yield return (x + 1, y, z);
        yield return (x - 1, y, z);
        yield return (x, y + 1, z);
        yield return (x, y - 1, z);
    }

    private static int GetNeighborValue(List<(int x, int y, int z, int value)> map, int x, int y, int z)
    {
        // Fast lookup for list
        // Not implemented for massive speedup, O(N) is fine for sparse large maps.
        // With list sort, we iterate.
        foreach(var v in map)
        {
            if (v.x == x && v.y == y && v.z == z) return v.value;
        }
        return 0;
    }
    
    private static void LoadWorld(StreamReader reader, Voxel[,,] world, int xc, int yc, int zc)
    {
        int idx = 0;
        // We read the entire stream
        using (var stream = reader.BaseStream)
        {
            int bufferSize = 65536;
            byte[] buffer = new byte[bufferSize];
            int bytesRead;
            while ((bytesRead = stream.Read(buffer, idx, bufferSize - idx)) > 0)
            {
                idx += bytesRead;
                if (idx >= bufferSize)
                {
                    // Process block
                    for (int i = 0; i < idx; i += 3)
                    {
                        if (i + 2 >= idx) break;
                        byte d = buffer[i];
                        int mass = d % 16;
                        int value = d >> 4;
                        if (mass > 0)
                        {
                            // Decode indices: x, y, z
                            // Input format: x=0..xc, y=0..yc, z=0..zc
                            int totalXY = xc * yc;
                            int x = i / (yc * zc); // Rough mapping
                            // No, don't do math on byte index. It's just stream order.
                            // We need to map byte index to (x,y,z).
                            // Read X, Y, Z from stream sequentially.
                        }
                    }
                    // Shift remaining
                    Array.Copy(buffer, idx, buffer, 0, 0);
                    idx = 0;
                }
            }
        }
        // Better: Just read one by one or small chunks, but keeping it simple.
        // Wait, I can't just read buffer blindly because x/y/z stride depends on y/z.
        // I must follow the loop order: x, y, z.
        
        // Reset stream
        reader.BaseStream.Seek(0, SeekOrigin.Begin);
        
        // Optimized Reader: Use a fixed buffer.
        // For simplicity in this snippet, I'll just read bytes one by one or small blocks.
        // Given constraints, 64 cores, 127GB RAM. 
        // We can afford to create the 2D array.
        
        // Re-implementing LoadWorld correctly
        using (var stream = reader.BaseStream)
        {
            int v = stream.ReadByte();
            while (v != -1)
            {
                int d = v;
                int mass = d % 16;
                int value = d >> 4;
                
                // Decode: 
                // x, y, z loop.
                // We can just increment indices.
                // This is slow in C#, but 5000^3 is too big. 
                // We only read if mass > 0.
                
                // We can't easily map byte index to spatial index without math.
                // Let's stick to the loop logic.
                // The loop is: x:0..xc, y:0..yc, z:0..zc.
                // We need to track x, y, z.
                
                // Since we don't know xc/yc/zc sizes easily from the stream byte-by-byte,
                // We rely on the header to allocate the array.
                // Then we iterate x,y,z and read.
                
                // Actually, the prompt says: "After the newline, voxels then follow...".
                // The stream is mapped to the array logic.
                // x goes 0..xc. y 0..yc. z 0..zc.
                
                // So:
                for (int z = 0; z < zc; z++)
                {
                    for (int y = 0; y < yc; y++)
                    {
                        for (int x = 0; x < xc; x++)
                        {
                            v = stream.ReadByte();
                            if (v == -1) return;
                            d = v;
                            int m = d % 16;
                            int val = d >> 4;
                            if (m > 0)
                            {
                                world[x, y, z] = new Voxel(val);
                            }
                        }
                    }
                }
            }
        }
    }

                // The stream doesn't explicitly give x,y,z headers for each byte.
                // But the loop order is defined.

                // We must read it once.

                // No, the "Input header format" gives dimensions.

                // This is a circular dependency.

    private static void TracePath(Voxel[,,] world, List<Movement> plan, int sx, int sy, int sz, int dx, int dy, int dz)
    {
        // Trace from Start to Dest.
        // Dig at Dest.
        // This function is for logic where we *must* go there.
        // Example: Dig Dirt -> Dump to neighbor.
        var path = AStarSimple(sx, sy, sz, dx, dy, dz);
        if (path == null) return;
        
        // Move steps
        for (int i = 0; i < path.Count - 1; i++)
        {
            var curr = path[i];
            var next = path[i+1];
            // Step
            plan.Add(new Movement(next.x, next.y, next.z, curr.x - next.x, curr.y - next.y, curr.z - next.z));
        }
        // Dig at Dest
        plan.Add(new Movement(dx, dy, dz, 0, 0, 0));
    }

        }
