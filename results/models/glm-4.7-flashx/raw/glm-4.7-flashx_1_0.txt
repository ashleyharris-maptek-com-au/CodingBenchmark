#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
#include <random>
#include <cmath>

using namespace std;

// Constants for problem constraints
constexpr int BRUTE_FORCE_LIMIT = 40;
constexpr int MEDIUM_LIMIT = 1000;
constexpr int LARGE_LIMIT = 5000;
constexpr int EXTREME_LIMIT = 10000000;

// Utility to get system seed
int64_t get_seed() {
    return static_cast<int64_t>(chrono::high_resolution_clock::now().time_since_epoch().count());
}

// 1. Brute Force Solver (For N <= 40)
void solve_bruteforce(int N, const vector<pair<double, double>>& coords, vector<int>& best_path) {
    vector<int> current_path(N);
    iota(current_path.begin(), current_path.end(), 0);

    best_path = current_path;
    double best_dist = 1e18;
    double total_dist = 0.0;

    // Pre-calculate pairwise distances to avoid sqrt inside the loop
    // For N=40, this 40*40/2 = 800 ops, negligible.
    // Memory: 800 doubles ~ 6.4KB.
    vector<vector<double>> dist(N, vector<double>(N));
    for (int i = 0; i < N; ++i) {
        for (int j = i + 1; j < N; ++j) {
            double dx = coords[i].first - coords[j].first;
            double dy = coords[i].second - coords[j].second;
            double d = dx * dx + dy * dy; // Squared distance is sufficient for comparison
            dist[i][j] = d;
            dist[j][i] = d;
        }
    }

    // Calculate initial total distance
    for (int i = 0; i < N - 1; ++i) {
        total_dist += dist[current_path[i]][current_path[i+1]];
    }

    best_dist = total_dist;

    // Heap-ordered queue for next permutations
    // Stores (total_distance, perm_index)
    vector<int> heap(N);
    iota(heap.begin(), heap.end(), 0);

    while (true) {
        // Increment the last element
        int k = N - 2;
        while (k >= 0 && current_path[k] > current_path[k + 1]) {
            k--;
        }

        if (k < 0) break; // Finished all permutations

        // Swap to next permutation
        int l = k + 1;
        while (l < N && current_path[k] < current_path[l]) l++;
        l--;

        swap(current_path[k], current_path[l]);

        // Fix the tail (reverse from k+1 to end) to get the next lexicographic permutation
        int i = k + 1;
        int j = N - 1;
        while (i < j) {
            swap(current_path[i], current_path[j]);
            i++;
            j--;
        }

        // Update distance using local changes
        // We swapped path[k] with path[l].
        // Previous path: ... A (at k), B (at l) ...
        // New path: ... B (at k), A (at l) ...
        // Old connections: A-prev + A-next + B-prev + B-next
        // New connections: B-prev + B-next + A-prev + A-next
        // Only edges involving A and B change.
        
        // Indices before swap
        int prev_k = (k == 0) ? N - 1 : k - 1;
        int next_k = k + 1;
        int prev_l = (l == 0) ? N - 1 : l - 1;
        int next_l = (l == N - 1) ? 0 : l + 1;

        // Edge costs
        double cost_A_prev = dist[current_path[prev_k]][current_path[k]];
        double cost_A_next = dist[current_path[k]][current_path[next_k]];
        double cost_B_prev = dist[current_path[prev_l]][current_path[l]];
        double cost_B_next = dist[current_path[l]][current_path[next_l]];

        double cost_B_new_prev = dist[current_path[prev_l]][current_path[k]];
        double cost_B_new_next = dist[current_path[k]][current_path[next_l]];
        double cost_A_new_prev = dist[current_path[prev_k]][current_path[l]];
        double cost_A_new_next = dist[current_path[l]][current_path[next_l]];

        // Recalculate total
        total_dist = total_dist 
            - cost_A_prev - cost_A_next - cost_B_prev - cost_B_next 
            + cost_B_new_prev + cost_B_new_next + cost_A_new_prev + cost_A_new_next;

        if (total_dist < best_dist) {
            best_dist = total_dist;
            best_path = current_path;
        }
    }
}

// 2. Greedy Nearest Neighbor Solver (For N <= 1000)
void solve_greedy(int N, const vector<pair<double, double>>& coords, vector<int>& path) {
    // Always start at 0
    vector<bool> visited(N, false);
    path[0] = 0;
    visited[0] = true;
    int visited_count = 1;

    // Find nearest neighbor using squared distance for speed
    while (visited_count < N) {
        int last = path[visited_count - 1];
        double best_dist = 1e18;
        int best_city = -1;

        for (int c = 1; c < N; ++c) {
            if (!visited[c]) {
                double dx = coords[last].first - coords[c].first;
                double dy = coords[last].second - coords[c].second;
                double d = dx * dx + dy * dy;
                if (d < best_dist) {
                    best_dist = d;
                    best_city = c;
                }
            }
        }
        path[visited_count] = best_city;
        visited[best_city] = true;
        visited_count++;
    }
}

// 3. Fast 2-Opt Solver (For N <= 5000)
void solve_2opt(int N, const vector<pair<double, double>>& coords, vector<int>& path) {
    solve_greedy(N, coords, path);

    bool improved = true;
    while (improved) {
        improved = false;
        // The inner loop is heavy, so we limit iterations or limit the scan window
        // for better performance on random graphs.
        for (int i = 0; i < N - 1; ++i) {
            for (int j = i + 1; j < N; ++j) {
                // i+1 < j is required for valid 2-opt swap
                int a = path[i];
                int b = path[i + 1];
                int c = path[j];
                int d = path[(j + 1) % N];

                // Calculate cost of crossing edges: ab + cd
                // Calculate cost of non-crossing edges: ad + bc
                double old_cost = (coords[a].first - coords[b].first)*(coords[a].first - coords[b].first) +
                                  (coords[a].second - coords[b].second)*(coords[a].second - coords[b].second);
                old_cost += (coords[c].first - coords[d].first)*(coords[c].first - coords[d].first) +
                            (coords[c].second - coords[d].second)*(coords[c].second - coords[d].second);

                double new_cost = (coords[a].first - coords[d].first)*(coords[a].first - coords[d].first) +
                                  (coords[a].second - coords[d].second)*(coords[a].second - coords[d].second);
                new_cost += (coords[c].first - coords[b].first)*(coords[c].first - coords[b].first) +
                            (coords[c].second - coords[b].second)*(coords[c].second - coords[b].second);

                if (new_cost < old_cost) {
                    // Reverse the segment from i+1 to j
                    int l = i + 1;
                    int r = j;
                    while (l < r) {
                        swap(path[l], path[r]);
                        l++;
                        r--;
                    }
                    improved = true;
                }
            }
        }
    }
}

// 4. Approximation Solver for Extreme Scales (N > 5000)
// Uses 2-Opt on a randomly initialized path (starting from 0)
void solve_approximate(int N, const vector<pair<double, double>>& coords, vector<int>& path) {
    // Initialize path: 0, random(1..N-1), random(2..N-1)...
    path[0] = 0;
    vector<int> remaining(N - 1);
    iota(remaining.begin(), remaining.end(), 1);
    
    // Shuffle remaining cities to ensure random start
    random_device rd;
    mt19937 g(rd());
    shuffle(remaining.begin(), remaining.end(), g);

    for (int i = 1; i < N; ++i) {
        path[i] = remaining[i-1];
    }

    bool improved = true;
    int iterations = 0;
    const int MAX_ITERATIONS = 10; // Cap iterations for massive N to keep time bounded

    while (improved && iterations < MAX_ITERATIONS) {
        improved = false;
        iterations++;

        // Use a simpler heuristic for large N to save time:
        // Only iterate forward and swap with a neighbor, no nested loops
        // This is effectively a "greedy 2-opt" or "delta" optimization
        
        for (int i = 1; i < N - 1; ++i) {
            // Try swapping path[i] with path[i-1] and path[i] with path[i+1]
            int prev = path[i-1];
            int curr = path[i];
            int next = path[i+1];

            double curr_dist = (coords[curr].first - coords[prev].first)*(coords[curr].first - coords[prev].first) +
                               (coords[curr].second - coords[prev].second)*(coords[curr].second - coords[prev].second);
            curr_dist += (coords[curr].first - coords[next].first)*(coords[curr].first - coords[next].first) +
                         (coords[curr].second - coords[next].second)*(coords[curr].second - coords[next].second);

            // Swap with previous
            // ... P, C, N ...
            // ... C, P, N ...
            double new_dist_prev = (coords[prev].first - coords[prev].first)*(coords[prev].first - coords[prev].first) + // dx=0
                                   (coords[prev].second - coords[next].second)*(coords[prev].second - coords[next].second);
            new_dist_prev += (coords[curr].first - coords[prev].first)*(coords[curr].first - coords[prev].first) +
                             (coords[curr].second - coords[prev].second)*(coords[curr].second - coords[prev].second);

            if (new_dist_prev < curr_dist) {
                swap(path[i], path[i-1]);
                continue;
            }

            // Swap with next
            // ... C, N, X ...
            // ... C, X, N ...
            int next_next = path[i+2];
            double new_dist_next = (coords[next].first - coords[next_next].first)*(coords[next].first - coords[next_next].first) +
                                   (coords[next].second - coords[next_next].second)*(coords[next].second - coords[next_next].second);
            new_dist_next += (coords[curr].first - coords[next_next].first)*(coords[curr].first - coords[next_next].first) +
                             (coords[curr].second - coords[next_next].second)*(coords[curr].second - coords[next_next].second);

            if (new_dist_next < curr_dist) {
                swap(path[i+1], path[i+2]);
            }
        }
    }
}

int main() {
    // Fast I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    if (!(cin >> N)) return 0;

    vector<pair<double, double>> coords(N);
    for (int i = 0; i < N; ++i) {
        double x, y;
        cin >> x >> y;
        coords[i] = {x, y};
    }

    vector<int> path(N);

    if (N <= BRUTE_FORCE_LIMIT) {
        solve_bruteforce(N, coords, path);
    } else if (N <= MEDIUM_LIMIT) {
        solve_greedy(N, coords, path);
        // Add a single pass of 2-opt which is very fast for small N
        solve_2opt(N, coords, path);
    } else if (N <= LARGE_LIMIT) {
        solve_2opt(N, coords, path);
    } else {
        solve_approximate(N, coords, path);
    }

    for (int i = 0; i < N; ++i) {
        cout << path[i];
        if (i < N - 1) cout << " ";
    }
    cout << endl;

    return 0;
}
