// Board Representation: 64-bit integer for squares.
// Bit 0: A8, Bit 7: H8, Bit 8: A7, ..., Bit 63: A1.
// Pockets (50-63) store captured pieces to handle En Passant.
using BitBoard = uint64_t;

// Piece Constants
enum PieceType {
    NONE, PAWN, KNIGHT, BISHOP, ROOK, QUEEN, KING
};

enum Color {
    WHITE, BLACK
};

// Color constants for bit shifting (64-bit friendly)
constexpr Color Color::WHITE = 0;
constexpr Color Color::BLACK = 1;
constexpr Color WHITE = Color::WHITE;
constexpr Color BLACK = Color::BLACK;

constexpr int TOTAL_SQUARES = 64;

// Piece Constants for Pockets
constexpr uint8_t W_PAWN = 1;
constexpr uint8_t W_KNIGHT = 2;
constexpr uint8_t W_BISHOP = 3;
constexpr uint8_t W_ROOK = 4;
constexpr uint8_t W_QUEEN = 5;
constexpr uint8_t W_KING = 6;
constexpr uint8_t B_PAWN = 9;
constexpr uint8_t B_KNIGHT = 10;
constexpr uint8_t B_BISHOP = 11;
constexpr uint8_t B_ROOK = 12;
constexpr uint8_t B_QUEEN = 13;
constexpr uint8_t B_KING = 14;

class Position {
public:
    // Bitboards for white and black pieces (includes captured pockets)
    BitBoard pieces[2] = { 0ULL, 0ULL };
    // Bitboards for pieces that haven't been captured (to identify legal moves of pieces on board)
    BitBoard pieces_on_board[2] = { 0ULL, 0ULL };
    
    // Occupancy bitboard (pieces on board)
    BitBoard occupancy = 0ULL;
    
    // Pockets for captured pieces (White captures black pieces, Black captures white pieces)
    uint64_t pockets[2] = { 0ULL, 0ULL };

    Color to_move = WHITE;
    int ply = 0;
    
    // Castling rights (3 bits per side: King moved, Rook moved (a1/h1), Rook captured)
    // Format: [WHITE][BLACK]
    uint8_t castling[2] = { 0, 0 }; 
    // En passant target square. -1 if none.
    int en_passant_target = -1; 

    void reset() {
        pieces[WHITE] = 0ULL; pieces_on_board[WHITE] = 0ULL;
        pieces[BLACK] = 0ULL; pieces_on_board[BLACK] = 0ULL;
        occupancy = 0ULL;
        pockets[WHITE] = 0ULL; pockets[BLACK] = 0ULL;
        to_move = WHITE;
        ply = 0;
        castling[WHITE] = 0; castling[BLACK] = 0;
        en_passant_target = -1;
    }

    void set_piece(Color c, PieceType p, int sq) {
        BitBoard b = 1ULL << sq;
        pieces[c] |= b;
        pieces_on_board[c] |= b;
        occupancy |= b;
        if (p == PAWN) pockets[c] |= 1ULL << (p + (sq % 64)); // Pawn ID is 1/9
    }

    void remove_piece(Color c, int sq) {
        BitBoard b = 1ULL << sq;
        pieces[c] ^= b;
        pieces_on_board[c] ^= b;
        occupancy ^= b;
    }

    // Get piece at square
    PieceType get_piece(int sq) const {
        if (sq < 0 || sq >= TOTAL_SQUARES) return NONE;
        if (pieces_on_board[WHITE] & (1ULL << sq)) {
            // If piece is on board, return based on standard piece bits
            if (pieces[WHITE] & (1ULL << sq)) return PAWN;
            if (pieces[WHITE] >> 50) return KING; // Stored in pockets
            if (pieces[WHITE] >> 55) return QUEEN;
            if (pieces[WHITE] >> 58) return ROOK;
            if (pieces[WHITE] >> 60) return BISHOP;
            return KNIGHT;
        }
        if (pieces_on_board[BLACK] & (1ULL << sq)) {
            if (pieces[BLACK] & (1ULL << sq)) return PAWN;
            if (pieces[BLACK] >> 50) return KING;
            if (pieces[BLACK] >> 55) return QUEEN;
            if (pieces[BLACK] >> 58) return ROOK;
            if (pieces[BLACK] >> 60) return BISHOP;
            return KNIGHT;
        }
        return NONE;
    }
};

// Simple PSQT tables (Shifted values for C++20 consteval would be ideal, but plain arrays work)
const int PSQT[][2][7] = {
    { // White
        {0, 0, 0, 0, 0, 0, 0}, // Pawn
        {-60, -42, -33, -31, -28, -25, -60}, // Knight
        {-64, -32, -12, -2,  8,  25, -64}, // Bishop
        {-78, -20,  5,  10, 20,  30, -78}, // Rook
        {-120, -70, -40, -20,  0,  20, -120}, // Queen
        {-200, -150, -100, -50, 0, 50, -200}, // King
    },
    { // Black
        {0, 0, 0, 0, 0, 0, 0},
        {60, 42, 33, 31, 28, 25, 60},
        {64, 32, 12, 2,  -8, -25, 64},
        {78, 20, -5, -10, -20, -30, 78},
        {120, 70, 40, 20,  0,  20, 120},
        {200, 150, 100, 50, 0, 50, 200},
    }
};

// Brute Force Evaluation
int evaluate(const Position& pos) {
    int score = 0;
    // White (Positive) vs Black (Negative)
    
    // Material
    // Masks for pieces in pieces[WHITE] and pieces[BLACK]
    // Standard pieces: 0-5 bits
    // Pocketed pieces: 6 bits shifted (King 50, Queen 55, Rook 58, Bishop 60, Knight 62)
    
    // White Material
    score += ((pos.pieces[WHITE] >> 50) ? 1000 : 0);
    score += ((pos.pieces[WHITE] >> 55) ? 900 : 0);
    score += ((pos.pieces[WHITE] >> 58) ? 500 : 0);
    score += ((pos.pieces[WHITE] >> 60) ? 330 : 0);
    score += ((pos.pieces[WHITE] >> 62) ? 320 : 0);
    
    // Black Material
    score -= ((pos.pieces[BLACK] >> 50) ? 1000 : 0);
    score -= ((pos.pieces[BLACK] >> 55) ? 900 : 0);
    score -= ((pos.pieces[BLACK] >> 58) ? 500 : 0);
    score -= ((pos.pieces[BLACK] >> 60) ? 330 : 0);
    score -= ((pos.pieces[BLACK] >> 62) ? 320 : 0);

    // Positional
    // We iterate all squares, but for speed in this simple evaluator, 
    // we can iterate bits on board. However, calculating from Bitboards directly is harder for position logic.
    // Let's iterate standard squares 0-63.
    
    for (int sq = 0; sq < TOTAL_SQUARES; ++sq) {
        PieceType p = pos.get_piece(sq);
        if (p != NONE) {
            Color c = (pos.pieces_on_board[WHITE] & (1ULL << sq)) ? WHITE : BLACK;
            score += PSQT[c][p][sq];
        }
    }

    return score;
}

// Shift definitions for 64-bit moves
constexpr BitBoard ShiftNorth(BitBoard b) { return b << 8; }
constexpr BitBoard ShiftSouth(BitBoard b) { return b >> 8; }
constexpr BitBoard ShiftEast(BitBoard b)  { return (b & 0xFEFEFEFEFEFEFEFEULL) << 1; }
constexpr BitBoard ShiftWest(BitBoard b)  { return (b & 0x7F7F7F7F7F7F7F7FULL) >> 1; }
constexpr BitBoard ShiftNorthEast(BitBoard b) { return ShiftEast(ShiftNorth(b)); }
constexpr BitBoard ShiftNorthWest(BitBoard b) { return ShiftWest(ShiftNorth(b)); }
constexpr BitBoard ShiftSouthEast(BitBoard b) { return ShiftEast(ShiftSouth(b)); }
constexpr BitBoard ShiftSouthWest(BitBoard b) { return ShiftWest(ShiftSouth(b)); }

constexpr BitBoard PawnAttacks(Color c) { return (c == WHITE) ? ShiftNorthWest(ShiftNorth(0ULL)) | ShiftNorthEast(ShiftNorth(0ULL)) : ShiftSouthWest(ShiftSouth(0ULL)) | ShiftSouthEast(ShiftSouth(0ULL)); }
constexpr BitBoard KnightMoves { 
    ShiftNorthWest(ShiftNorth(ShiftNorth(0ULL))) | ShiftNorth(ShiftNorth(ShiftNorthWest(0ULL))) | ShiftNorth(ShiftNorth(ShiftNorthEast(0ULL))) | ShiftNorthEast(ShiftNorth(ShiftNorthEast(0ULL))) |
    ShiftSouthWest(ShiftSouth(ShiftSouth(0ULL))) | ShiftSouth(ShiftSouth(ShiftSouthWest(0ULL))) | ShiftSouth(ShiftSouth(ShiftSouthEast(0ULL))) | ShiftSouthEast(ShiftSouth(ShiftSouthEast(0ULL))) 
};
constexpr BitBoard KingMoves { 
    ShiftNorth(0ULL) | ShiftNorthEast(0ULL) | ShiftEast(0ULL) | ShiftSouthEast(0ULL) | ShiftSouth(0ULL) | ShiftSouthWest(0ULL) | ShiftWest(0ULL) | ShiftNorthWest(0ULL) 
};

// Simple bitboards for sliding pieces (pre-computed)
BitBoard RookAttacks[64][64];
BitBoard BishopAttacks[64][64];
BitBoard QueenAttacks[64][64];

void init_attacks() {
    for (int from = 0; from < 64; ++from) {
        for (int to = 0; to < 64; ++to) {
            int df = from - to;
            if (df == 0) { RookAttacks[from][to] = 0; BishopAttacks[from][to] = 0; continue; }
            int dr = abs(df % 8);
            int dc = abs(df / 8);
            
            bool straight = (dr == 0);
            bool diag = (dr == dc);
            
            if (straight) {
                BitBoard b = 1ULL << to;
                if (df > 0) b = b << from; // Destination is north/east/south/west
                else b = b >> (from - to); 
                // Manual slide check is safer to avoid UB or logic errors
                // Actually, let's rely on the BitBoard shift macros, but for attack tables:
                RookAttacks[from][to] = 1ULL << to;
            } else if (diag) {
                BishopAttacks[from][to] = 1ULL << to;
            }
        }
    }
    
    // Precompute Queen = Rook | Bishop
    for (int from = 0; from < 64; ++from) {
        for (int to = 0; to < 64; ++to) {
            QueenAttacks[from][to] = RookAttacks[from][to] | BishopAttacks[from][to];
        }
    }
}

// Generate pseudo-legal moves
// Note: This generates all moves. Filtering for checks is expensive. 
// For a timeout of 5s on 64 cores, we can do full enumeration and verification, 
// or just use a Zobrist hash to prune.
// Given the constraints and lack of a hash table in the snippet, we will use a simple alpha-beta with quiescence.
// To make it fast enough for 5s, we rely on pruning and simple Move Ordering.

std::vector<Move> generate_moves(const Position& pos) {
    std::vector<Move> moves;
    BitBoard occupied = pos.occupancy;
    
    // Iterate over all pieces for White
    for (int from = 0; from < 64; ++from) {
        if (pos.get_piece(from) == NONE) continue;
        
        BitBoard piece_bb = 1ULL << from;
        Color color = (pos.pieces_on_board[WHITE] & piece_bb) ? WHITE : BLACK;
        PieceType p = pos.get_piece(from);
        
        // Generate moves for this piece
        // Pseudo moves (potential)
        BitBoard potential = 0ULL;
        
        if (p == PAWN) {
            BitBoard forward = (color == WHITE) ? ShiftNorth(piece_bb) : ShiftSouth(piece_bb);
            // Blocked by occupancy
            potential = forward & ~occupied;
            
            // Attack squares (Pawn pushes)
            BitBoard attacks = (color == WHITE) ? (ShiftNorthWest(piece_bb) | ShiftNorthEast(piece_bb)) : (ShiftSouthWest(piece_bb) | ShiftSouthEast(piece_bb));
            attacks &= occupied; // Only captures hit occupied
            
            // En Passant check
            if (pos.en_passant_target != -1) {
                // If pawn is on rank 3 or 6
                int ep_rank = (color == WHITE) ? 6 : 1;
                int ep_idx = pos.en_passant_target;
                if (abs(from - ep_idx) == 9 || abs(from - ep_idx) == 7) {
                    attacks |= (1ULL << ep_idx);
                }
            }
            
            // Promotion
            BitBoard promo_forward = forward & ~occupied;
            // Promotion rank (7 for white, 0 for black)
            int promo_rank = (color == WHITE) ? 7 : 0;
            
            if (promo_forward) {
                // Check if promotion happens on this push
                if (forward >> (64 - 9) || forward >> 8) { // Recheck simple rank
                    if ((color == WHITE && from >= 56) || (color == BLACK && from < 8)) {
                        // Promotion rank
                        // Push to target
                        for (int to = 0; to < 64; ++to) {
                            if ((promo_forward >> to) & 1) {
                                moves.push_back({from, to, NONE, false, true, QUEEN});
                            }
                        }
                    }
                }
                // Non-promotion pushes
                if (!(color == WHITE && from >= 56) && !(color == BLACK && from < 8)) {
                     for (int to = 0; to < 64; ++to) {
                        if ((potential >> to) & 1) {
                            moves.push_back({from, to, NONE, false, false, NONE});
                        }
                    }
                }
            } else {
                // No promotion push
                for (int to = 0; to < 64; ++to) {
                    if ((attacks >> to) & 1) {
                        moves.push_back({from, to, NONE, true, false, NONE});
                    }
                }
            }
        } 
        else if (p == KNIGHT) {
            // Magic shift or simple table lookup (no magic needed for knights)
            // Knight moves are 8 squares
            // Use a precomputed mask
            if ((KnightMoves >> from) & 1) { // This logic is inverted, let's do shifts
                // Manual loop
                potential = (KnightMoves >> (from + 6)) & 0x0E1; // Corners
                potential |= (KnightMoves >> (from + 1)) & 0x000E; // Left edge
                potential |= (KnightMoves >> (from - 1)) & 0x7800; // Right edge
                potential |= (KnightMoves >> (from - 7)) & 0x0080; // Bottom left
                potential |= (KnightMoves >> (from - 9)) & 0x0001; // Top left
                potential |= (KnightMoves >> (from + 7)) & 0x0002; // Top right
                potential |= (KnightMoves >> (from + 9)) & 0x0400; // Bottom right
            } else {
                // Hacky initialization, just assume valid
                 potential = (KnightMoves >> (from + 6)) & 0x0E1; 
            }

        } 
        else if (p == BISHOP) {
            // Sliding attacks
            potential = 0ULL;
            for (int d = 0; d < 8; d += 2) {
                BitBoard b = piece_bb;
                b >>= (d == 0 ? 7 : d == 1 ? 9 : d == 2 ? -7 : d == 3 ? -9 : 0);
                while (b && !(b & 0x0101010101010101ULL)) {
                    potential |= b;
                    if (b & 0x0101010101010101ULL) break; // Hit edge
                    b >>= (d == 0 ? 7 : d == 1 ? 9 : d == 2 ? -7 : d == 3 ? -9 : 0);
                }
            }
        }
        else if (p == ROOK) {
            // Sliding attacks
            potential = 0ULL;
            for (int d = 0; d < 8; d += 2) {
                BitBoard b = piece_bb;
                b >>= (d == 0 ? 8 : d == 1 ? 1 : d == 2 ? -8 : d == 3 ? -1 : 0);
                while (b && !(b & 0x0101010101010101ULL)) {
                    potential |= b;
                    if (b & 0x0101010101010101ULL) break;
                    b >>= (d == 0 ? 8 : d == 1 ? 1 : d == 2 ? -8 : d == 3 ? -1 : 0);
                }
            }
        }
        else if (p == QUEEN) {
            // Rook + Bishop
            potential = 0ULL;
            for (int d = 0; d < 8; d += 2) {
                BitBoard b = piece_bb;
                b >>= (d == 0 ? 8 : d == 1 ? 1 : d == 2 ? -8 : d == 3 ? -1 : 0);
                while (b && !(b & 0x0101010101010101ULL)) {
                    potential |= b;
                    if (b & 0x0101010101010101ULL) break;
                    b >>= (d == 0 ? 8 : d == 1 ? 1 : d == 2 ? -8 : d == 3 ? -1 : 0);
                }
            }
            for (int d = 0; d < 8; d += 2) {
                BitBoard b = piece_bb;
                b >>= (d == 0 ? 7 : d == 1 ? 9 : d == 2 ? -7 : d == 3 ? -9 : 0);
                while (b && !(b & 0x0101010101010101ULL)) {
                    potential |= b;
                    if (b & 0x0101010101010101ULL) break;
                    b >>= (d == 0 ? 7 : d == 1 ? 9 : d == 2 ? -7 : d == 3 ? -9 : 0);
                }
            }
        }
        else if (p == KING) {
            potential = KingMoves >> from; // Simple shift mask
        }

        // Apply moves
        if (potential) {
            for (int to = 0; to < 64; ++to) {
                if ((potential >> to) & 1) {
                    PieceType cap = pos.get_piece(to);
                    bool is_cap = (cap != NONE);
                    
                    // Filter castling
                    if (p == KING) {
                        if (abs(from - to) > 1) {
                            // Castling: from to to
                            int diff = to - from;
                            if (diff == 2) { // Kingside
                                if (pos.castling[pos.to_move] & 1) { // King side right
                                    // Check squares
                                    int f1 = (from + 1); int f2 = (from + 2);
                                    if (!pos.get_piece(f1) && !pos.get_piece(f2) && !(pos.occupancy & (1ULL << f1))) {
                                        // Also check if squares are attacked
                                        // Simplified: In real engine, you'd check move validity
                                        // For this code, we allow it if pseudo-legal.
                                        moves.push_back({from, to, NONE, false, false, NONE});
                                    }
                                }
                            } else if (diff == -2) { // Queenside
                                if (pos.castling[pos.to_move] & 2) { // Queen side right
                                    int f1 = (from - 1); int f2 = (from - 2); int f3 = (from - 3);
                                    if (!pos.get_piece(f1) && !pos.get_piece(f2) && !pos.get_piece(f3) && !(pos.occupancy & (1ULL << f1))) {
                                        moves.push_back({from, to, NONE, false, false, NONE});
                                    }
                                }
                            }
                        }
                    }

                    // Ep check
                    bool ep = false;
                    if (p == PAWN && to == pos.en_passant_target) {
                        ep = true;
                        is_cap = true;
                        cap = PAWN; // Capture the invisible pawn
                    }

                    moves.push_back({from, to, cap, is_cap || ep, false, NONE});
                }
            }
        }
    }
    return moves;
}

bool is_square_attacked(int sq, const Position& pos, Color attacker_color) {
    // Optimization: Reverse check attacks
    // Pawn attacks
    BitBoard pawn_att = (attacker_color == WHITE) ? ShiftNorthWest(1ULL << sq) | ShiftNorthEast(1ULL << sq) : 
                                          ShiftSouthWest(1ULL << sq) | ShiftSouthEast(1ULL << sq);
    if (pawn_att & pos.pieces[attacker_color]) return true;

    // Knight attacks
    BitBoard knight_att = KnightMoves >> sq; 
    // Check mask manually
    BitBoard kn_mask = 0ULL;
    for(int i=0; i<64; ++i) if(((KnightMoves >> i) >> sq)&1) kn_mask |= (1ULL<<i);
    if (kn_mask & pos.pieces[attacker_color]) return true;

    // King attacks
    BitBoard king_att = KingMoves >> sq;
    if (king_att & pos.pieces[attacker_color]) return true;

    // Sliding
    Color opponent = (attacker_color == WHITE) ? BLACK : WHITE;
    
    // Bishop
    BitBoard b_att = 0;
    int d = 0;
    BitBoard b = (1ULL << sq);
    // North West
    b >>= 7; while(b && !(b & 0x0101010101010101ULL)) { b_att |= b; b >>= 7; }
    // North East
    b = (1ULL << sq); b >>= 9; while(b && !(b & 0x0101010101010101ULL)) { b_att |= b; b >>= 9; }
    // South West
    b = (1ULL << sq); b <<= 7; while(b && !(b & 0x0101010101010101ULL)) { b_att |= b; b <<= 7; }
    // South East
    b = (1ULL << sq); b <<= 9; while(b && !(b & 0x0101010101010101ULL)) { b_att |= b; b <<= 9; }
    if (b_att & pos.pieces[opponent]) return true;

    // Rook
    BitBoard r_att = 0;
    b = (1ULL << sq);
    // North
    b >>= 8; while(b && !(b & 0x0101010101010101ULL)) { r_att |= b; b >>= 8; }
    // South
    b = (1ULL << sq); b <<= 8; while(b && !(b & 0x0101010101010101ULL)) { r_att |= b; b <<= 8; }
    // West
    b = (1ULL << sq); b >>= 1; while(b && !(b & 0xFEFEFEFEFEFEFEFEULL)) { r_att |= b; b >>= 1; }
    // East
    b = (1ULL << sq); b <<= 1; while(b && !(b & 0x7F7F7F7F7F7F7F7FULL)) { r_att |= b; b <<= 1; }
    if (r_att & pos.pieces[opponent]) return true;
    
    // Queen = Rook | Bishop
    if (b_att & pos.pieces[opponent]) return true;
    if (r_att & pos.pieces[opponent]) return true;

    return false;
}

bool is_check(const Position& pos) {
    int king_sq = -1;
    // Find king
    for (int i = 0; i < 64; ++i) {
        if (pos.get_piece(i) == KING && (pos.pieces_on_board[WHITE] & (1ULL << i))) {
            king_sq = i;
            break;
        }
    }
    if (king_sq == -1) return true; // No king? (bug) or passed pawn? (bug)
    
    return is_square_attacked(king_sq, pos, pos.to_move);
}

// Move Validation (Removes moves that leave king in check)
std::vector<Move> filter_legal_moves(const Position& pos, const std::vector<Move>& pseudo_moves) {
    std::vector<Move> legal_moves;
    bool in_check = is_check(pos);
    
    for (const auto& m : pseudo_moves) {
        // Make move
        Position temp = pos;
        temp.remove_piece(temp.to_move, m.from);
        
        PieceType captured = m.captured;
        if (m.is_capture) {
            // Remove captured piece from board
            temp.remove_piece(temp.to_move == WHITE ? BLACK : WHITE, m.to);
            if (m.to == temp.en_passant_target) {
                // Remove the pawn at from square
                temp.remove_piece(temp.to_move == WHITE ? BLACK : WHITE, m.from + (m.from % 8 < 7 ? 1 : -1)); 
                // Note: ep logic above handled this, but here we need to remove the OTHER pawn
                int ep_sq = (m.from + m.to) / 2; // The en passant pawn is at the midpoint
                temp.remove_piece(temp.to_move == WHITE ? BLACK : WHITE, ep_sq);
            }
        }
        
        temp.set_piece(temp.to_move, m.is_promotion ? m.promotion : temp.get_piece(m.from), m.to);
        
        // Update Flags
        if (m.from == 60 && temp.to_move == WHITE) temp.castling[WHITE] &= 6; // King moved
        if (m.to == 63 && temp.to_move == WHITE) temp.castling[WHITE] &= 5; // Rook moved
        if (m.to == 56 && temp.to_move == WHITE) temp.castling[WHITE] &= 13; // Rook captured
        if (m.from == 4 && temp.to_move == WHITE) temp.castling[WHITE] &= 11; // Rook moved
        
        if (m.from == 4 && temp.to_move == BLACK) temp.castling[BLACK] &= 6;
        if (m.to == 7 && temp.to_move == BLACK) temp.castling[BLACK] &= 5;
        if (m.to == 0 && temp.to_move == BLACK) temp.castling[BLACK] &= 13;
        if (m.from == 60 && temp.to_move == BLACK) temp.castling[BLACK] &= 11;

        if (temp.is_check()) {
            continue; // Illegal move
        }
        
        legal_moves.push_back(m);
    }
    return legal_moves;
}

// Renotation from internal move to UCI string
std::string move_to_uci(const Move& m, const Position& pos) {
    PieceType p = pos.get_piece(m.from);
    char p_char = PIECE_TO_CHAR(p);
    
    // Source square (e.g. e2)
    // Board 0-63. Convert to algebraic
    int r = 8 - (m.from / 8);
    char c = 'a' + (m.from % 8);
    std::string sq_src = std::string(1, c) + std::to_string(r);
    
    // Dest square
    int r2 = 8 - (m.to / 8);
    char c2 = 'a' + (m.to % 8);
    std::string sq_dst = std::string(1, c2) + std::to_string(r2);
    
    std::string res = p_char + sq_src + sq_dst;
    
    if (m.is_capture) res += 'x';
    res += sq_dst;
    
    if (m.is_promotion) res += "=" + std::string(1, PIECE_TO_CHAR(m.promotion));
    
    // Not implemented: O-O, O-O-O, ep, check/mate
    return res;
}

// --- Search ---

int search(Position& pos, int depth, int alpha, int beta, bool is_root) {
    if (depth == 0) {
        return evaluate(pos);
    }
    
    std::vector<Move> moves = generate_moves(pos);
    moves = filter_legal_moves(pos, moves);
    
    if (moves.empty()) {
        if (is_check(pos)) return -100000 + pos.ply; // Mate
        return 0; // Stalemate
    }
    
    // Simple ordering: Captures first
    std::sort(moves.begin(), moves.end(), [](const Move& a, const Move& b) {
        return a.is_capture > b.is_capture;
    });
    
    int best_score = -99999;
    for (const auto& m : moves) {
        Position next = pos;
        
        // Move execution
        next.remove_piece(next.to_move, m.from);
        if (m.is_capture) {
            next.remove_piece(next.to_move == WHITE ? BLACK : WHITE, m.to);
            if (m.to == next.en_passant_target) {
                int ep_sq = (m.from + m.to) / 2;
                next.remove_piece(next.to_move == WHITE ? BLACK : WHITE, ep_sq);
            }
        }
        next.set_piece(next.to_move, m.is_promotion ? m.promotion : next.get_piece(m.from), m.to);
        
        // Castling Rights
        if (m.from == 60 && next.to_move == WHITE) next.castling[WHITE] &= 6; 
        if (m.to == 63 && next.to_move == WHITE) next.castling[WHITE] &= 5; 
        if (m.to == 56 && next.to_move == WHITE) next.castling[WHITE] &= 13; 
        if (m.from == 4 && next.to_move == WHITE) next.castling[WHITE] &= 11; 
        
        if (m.from == 4 && next.to_move == BLACK) next.castling[BLACK] &= 6;
        if (m.to == 7 && next.to_move == BLACK) next.castling[BLACK] &= 5;
        if (m.to == 0 && next.to_move == BLACK) next.castling[BLACK] &= 13;
        if (m.from == 60 && next.to_move == BLACK) next.castling[BLACK] &= 11;

        int score = -search(next, depth - 1, -beta, -alpha, false);
        
        if (score > best_score) {
            best_score = score;
        }
        if (score > alpha) {
            alpha = score;
        }
        if (alpha >= beta) {
            break;
        }
    }
    
    return best_score;
}

int search(Position& pos, int depth) {
    return search(pos, depth, -99999, 99999, true);
}

// --- Main ---

void fen_to_pos(const std::string& fen, Position& pos) {
    pos.reset();
    std::istringstream iss(fen);
    std::string board_str, turn, cast_str, ep_str, hm_str, fstr;
    
    iss >> board_str >> turn >> cast_str >> ep_str >> hm_str >> fstr;
    
    // Board
    int rank = 8;
    int file = 0;
    for (char c : board_str) {
        if (c == '/') {
            rank--;
            file = 0;
        } else if (isdigit(c)) {
            file += (c - '0');
        } else {
            Color c_col = isupper(c) ? WHITE : BLACK;
            PieceType p = PIECE_FROM_CHAR(tolower(c));
            pos.set_piece(c_col, p, (rank * 8) + file);
            file++;
        }
    }
    
    pos.to_move = (turn == "w") ? WHITE : BLACK;
    
    // Castling
    for (char c : cast_str) {
        if (c == 'K') pos.castling[WHITE] |= 4;
        if (c == 'Q') pos.castling[WHITE] |= 2;
        if (c == 'k') pos.castling[BLACK] |= 1;
        if (c == 'q') pos.castling[BLACK] |= 8;
    }
    
    // En Passant
    if (ep_str != "-") {
        int ep_sq = (8 - (ep_str[1] - '0')) * 8 + (ep_str[0] - 'a');
        pos.en_passant_target = ep_sq;
    }
}

    // For robustness, let's do a small search iteration that tracks the best move.
    
    // Re-run search to get best move correctly
    best_score = -99999;
    std::vector<Move> moves = generate_moves(pos);
    moves = filter_legal_moves(pos, moves);
    
    // Simple ordering: Captures first, then quiet
    std::sort(moves.begin(), moves.end(), [](const Move& a, const Move& b) {
        if (a.is_capture != b.is_capture) return a.is_capture > b.is_capture;
        return a.to > b.to; 
    });
    
    auto search_start = std::chrono::high_resolution_clock::now();
    for (const auto& m : moves) {
        Position next = pos;

        next.remove_piece(next.to_move, m.from);
