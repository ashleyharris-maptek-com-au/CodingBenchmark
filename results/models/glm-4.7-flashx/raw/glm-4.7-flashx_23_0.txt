```rust
use std::io::{self, Read, Write, BufRead};
use std::time::{Duration, Instant};

// Configuration
const TARGET_THRESHOLD: f64 = 2.0; // m/s
const TARGET_LANDING_ZONE: f64 = 5.0; // meters radius
const GRAVITY: f64 = 3.71; // m/s^2 (Moon)
const MAX_THRUST: f64 = 20.0; // m/s^2
const FPS: f64 = 20.0;
const DT: f64 = 1.0 / FPS; // 0.05 seconds
const Y_DOWN: bool = true; // Y increases downward in this sim

#[derive(Clone, Copy, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

#[derive(Clone, Copy, Debug)]
struct MapData {
    width: usize,
    height: usize,
    target_x: f64,
    target_y: f64,
    // Store map as a 1D flat vector of chars for fast access without allocations
    // using `Option` to allow lazy initialization
    terrain: Option<Vec<u8>>,
}

struct Controller {
    state: State,
    map: MapData,
    last_update: Option<Instant>,
    // State for the output buffer
    output_buffer: Vec<u8>,
    last_output_time: Instant,
}

impl Controller {
    fn new(map: MapData, state: State) -> Self {
        Self {
            state,
            map,
            last_update: None,
            output_buffer: Vec::with_capacity(128),
            last_output_time: Instant::now(),
        }
    }

    fn update(&mut self, input: &[u8]) -> io::Result<()> {
        let now = Instant::now();
        let input_str = unsafe { std::str::from_utf8_unchecked(input) };
        
        // Parse state (whitespace separated)
        let parts: Vec<f64> = input_str
            .split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();

        if parts.len() != 6 {
            return Err(io::Error::new(io::ErrorKind::InvalidData, "Invalid state format"));
        }

        self.state.x = parts[0];
        self.state.y = parts[1];
        self.state.vx = parts[2];
        self.state.vy = parts[3];
        self.state.angle = parts[4];
        self.state.angular_vel = parts[5];
        self.state.fuel = parts[6];
        
        self.last_update = Some(now);
        Ok(())
    }

    fn physics_step(&mut self) {
        let dt = DT;
        let state = &mut self.state;
        let g = GRAVITY;

        // Angular physics
        state.angular_vel += 0.0 * dt; // Simple P controller
        state.angle += state.angular_vel * dt;
        // Clamp angle to [0, PI] (pointing down) to prevent spinning indefinitely
        state.angle = state.angle.max(0.0).min(std::f64::consts::PI);

        // Apply thrust
        // Check collision with ground first to prevent tunneling (optional but safer)
        // Here we trust the map bounds for safety.

        let mut fx = 0.0;
        let mut fy = 0.0;
        let thrust_level = 0.0; // Controlled by user, but kept 0 initially

        if thrust_level > 0.0 && state.fuel > 0.0 {
            // Thrust vector
            let thrust_vec_x = thrust_level * MAX_THRUST * state.angle.sin();
            let thrust_vec_y = thrust_level * MAX_THRUST * state.angle.cos();

            // Apply to velocity
            state.vx += thrust_vec_x * dt;
            state.vy += thrust_vec_y * dt;
            
            // Apply to fuel
            state.fuel -= thrust_level * dt;
        }

        // Apply Gravity
        state.vy += g * dt;

        // Apply velocity to position
        state.x += state.vx * dt;
        state.y += state.vy * dt;
    }

    fn execute_tick(&mut self) -> io::Result<()> {
        let now = Instant::now();
        let elapsed = now.duration_since(self.last_output_time);
        
        if elapsed < Duration::from_millis((1000.0 / FPS) as u64) {
            return Ok(());
        }
        
        self.last_output_time = now;

        // 1. Determine Control Output
        let (thrust, turn) = self.calculate_control();

        // 2. Update Physics
        self.physics_step();

        // 3. Check Termination
        if self.check_crash() || self.check_landed() {
            // Print final state and exit cleanly
            writeln!(self.output_buffer, "{:.2} {:.2}", thrust, turn)?;
        } else {
            // Print current instruction
            writeln!(self.output_buffer, "{:.2} {:.2}", thrust, turn)?;
        }

        // 4. Write to stdout
        // If we have accumulated data, flush it.
        // We do not flush every time if we want to batch writes for performance,
        // but given the 100ms stall rule, writing to stdout is safer to keep the buffer sane.
        io::stdout().write_all(&self.output_buffer)?;
        self.output_buffer.clear();
        io::stdout().flush()?;

        Ok(())
    }

    fn check_crash(&self) -> bool {
        // Out of bounds
        let (w, h) = (self.map.width, self.map.height);
        if self.state.x < 0.0 || self.state.x >= w as f64 || self.state.y < 0.0 || self.state.y >= h as f64 {
            return true;
        }

        // Ground collision
        // Check the underlying terrain cell. 
        // Note: State Y is 0..h. If terrain[y][x] is '#', it is an obstacle.
        let terrain = self.map.terrain.as_ref().expect("Map not loaded");
        let idx = self.state.y as usize * w + self.state.x as usize;
        
        // Simple bounding box check (lander is approx 1x1)
        // Check if the cell under the center is an obstacle
        if terrain[idx] == 1 {
            // Check bounds of lander
            let x_min = (self.state.x - 0.5).floor() as usize;
            let x_max = (self.state.x + 0.5).ceil() as usize;
            let y_min = (self.state.y - 0.5).floor() as usize;
            let y_max = (self.state.y + 0.5).ceil() as usize;

            for y in y_min..=y_max {
                for x in x_min..=x_max {
                    if y < h && x < w {
                        if terrain[y * w + x] == 1 {
                            return true;
                        }
                    }
                }
            }
        }
        false
    }

    fn check_landed(&self) -> bool {
        // Within target bounds
        let dx = self.state.x - self.map.target_x;
        let dy = self.state.y - self.map.target_y;
        let dist_sq = dx * dx + dy * dy;

        if dist_sq > TARGET_LANDING_ZONE * TARGET_LANDING_ZONE {
            return false;
        }

        // Velocity checks
        let vel = (self.state.vx * self.state.vx + self.state.vy * self.state.vy).sqrt();
        let hard_land = vel > TARGET_THRESHOLD;
        let forward_thrust = self.state.vx.abs() > 1.0; // Hard limit on forward speed
        
        // Angle check: Should be near PI (facing down)
        let angle_diff = (self.state.angle - std::f64::consts::PI).abs();
        let steep_angle = angle_diff > 0.2;

        hard_land || forward_thrust || steep_angle
    }

    fn calculate_control(&self) -> (f64, f64) {
        // Simple control law
        let (x, y, vx, vy, angle) = (self.state.x, self.state.y, self.state.vx, self.state.vy, self.state.angle);
        let target = (self.map.target_x, self.map.target_y);

        // Turn towards target
        let dx = target.0 - x;
        let dy = target.1 - y;
        
        // We want to be at (x, y) at (target.0, target.1)
        // If we are above target, we want positive vy (down). 
        // If we are below, we want negative vy (up).
        // The angle for the lander is 0 (up) to PI (down).
        
        // Calculate error for angle: We want the lander pointing roughly at the target.
        // Since Y increases downwards, we calculate angle in screen coordinates
        // Math.atan2(y, x) -> standard.
        // But our lander angle 0 is "up" (negative Y in screen space logic, but input says 0 is up).
        // Actually, input says: "0 is facing up". "y increases downward".
        // So angle 0 points towards -y (up).
        
        // Let's calculate the ideal angle to face the target.
        // vector to target: (dx, dy). 
        // We want the nose to point to the target. 
        // The nose angle is 0 (up).
        // In screen coords, Up is -90 deg (270 deg) or 3PI/2.
        // If we use atan2(y, x), angle 0 is East. Angle PI/2 is South (Down).
        // The input says 0 is Up. So input angle = atan2(y, x) + PI/2.
        // Therefore, atan2(y, x) = input angle - PI/2.
        
        let ideal_angle = (dy.atan2(dx) + std::f64::consts::PI / 2.0).rem_euclid(std::f64::consts::PI);
        
        // Error in angle (0 to PI)
        let angle_error = (angle - ideal_angle).abs();
        let turn_rate = if angle_error < 0.1 { 0.0 } else { (angle - ideal_angle).signum() * 2.0 }; // Clamp turn rate

        // Thrust control
        // Vertical velocity damping
        // Target vy = 0.
        // If we are above target (y < ty), we need positive vy (gravity helps).
        // If we are below target (y > ty), we need negative vy.
        
        let dist_y = y - target.1;
        let target_vy = if dist_y > 0.0 { -3.0 } else { 2.0 }; // Proportional-ish
        
        // Lerp thrust to achieve target vy, capped at max thrust
        let thrust = if vy < target_vy { 1.0 } else { 0.0 };

        (thrust, turn_rate)
    }
}

fn main() -> io::Result<()> {
    let mut stdin = io::stdin();
    let mut buffer = Vec::new();
    
    // 1. Read Initial Map
    // Scan until we find the map rows line
    let mut map_data = String::new();
    stdin.read_line(&mut map_data)?;
    
    let init: Vec<f64> = map_data.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    
    let width = init[0] as usize;
    let height = init[1] as usize;
    let gravity = init[2]; // Not used, fixed
    let max_thrust = init[3]; // Not used, fixed
    let fuel = init[4]; // Not used, infinite for this iteration
    
    let start_x = init[5];
    let start_y = init[6];
    let target_x = init[7];
    let target_y = init[8];

    // 2. Read Map Data
    let mut terrain = vec![0u8; width * height];
    let mut rows_read = 0;
    
    // Optimize map reading
    let mut reader = stdin.lock();
    while rows_read < height {
        let mut line = String::new();
        if reader.read_line(&mut line).is_err() { break; }
        
        if line.trim().is_empty() { continue; }
        
        // Remove newline
        if let Some('\n') = line.chars().last() {
            line.pop();
        }

        let line_bytes = line.as_bytes();
        if line_bytes.len() != width {
            // If line is shorter/longer, pad/truncate
            let len = line_bytes.len().min(width);
            for i in 0..len {
                terrain[rows_read * width + i] = if line_bytes[i] == b'#' { 1 } else { 0 };
            }
            for i in len..width {
                terrain[rows_read * width + i] = 1; // Default obstacle
            }
        } else {
            // Copy line bytes
            terrain[rows_read * width .. rows_read * width + width].copy_from_slice(line_bytes);
        }
        
        rows_read += 1;
    }

    // 3. Initialize Lander
    // Start facing up (0 rad) and stationary
    let lander = State {
        x: start_x,
        y: start_y,
        vx: 0.0,
        vy: 0.0,
        angle: 0.0,
        angular_vel: 0.0,
        fuel: fuel,
    };

    // 4. Run
    let mut controller = Controller::new(MapData {
        width,
        height,
        target_x,
        target_y,
        terrain: Some(terrain),
    }, lander);

    let mut input_buffer = [0u8; 256];

    loop {
        // Check if stdin has data. 
        // Blocking read is safest to avoid missing updates, but needs to be fast.
        // Since we need to process physics at 20Hz, we try to read a frame every tick.
        
        match stdin.lock().read(&mut input_buffer) {
            Ok(0) => break, // EOF
            Ok(n) => {
                if let Err(e) = controller.update(&input_buffer[..n]) {
                    eprintln!("Parse Error: {}", e);
                    break;
                }
            }
            Err(_) => break,
        }
        
        if let Err(e) = controller.execute_tick() {
            eprintln!("Execution Error: {}", e);
            break;
        }
    }
    
    Ok(())
}
```
