#include <cstdio>
#include <cmath>
#include <vector>
#include <algorithm>
#include <stack>
#include <unordered_map>
#include <random>

static const double EPS = 1e-5;

struct V3 {
    double x, y, z;
    V3(double x = 0, double y = 0, double z = 0) : x(x), y(y), z(z) {}
    V3 operator+(V3 o) const { return {x + o.x, y + o.y, z + o.z}; }
    V3 operator-(V3 o) const { return {x - o.x, y - o.y, z - o.z}; }
    V3 operator*(double t) const { return {x * t, y * t, z * t}; }
    double dot(V3 o) const { return x * o.x + y * o.y + z * o.z; }
    V3 cross(V3 o) const { return {y*o.z - z*o.y, z*o.x - x*o.z, x*o.y - y*o.x}; }
    double len() const { return std::sqrt(dot(*this)); }
    V3 unit() const { double l = len(); return l > 1e-15 ? *this * (1.0/l) : V3(); }
};

struct Plane {
    V3 n;
    double w = 0;
    static Plane from(V3 a, V3 b, V3 c) {
        V3 nn = (b - a).cross(c - a).unit();
        return {nn, nn.dot(a)};
    }
    bool ok() const { return n.dot(n) > 0.5; }
    void flip() { n = n * -1; w = -w; }
};

struct Poly {
    std::vector<V3> v;
    Plane pl;
    Poly() = default;
    Poly(std::vector<V3> vv) : v(std::move(vv)) {
        if (v.size() >= 3) pl = Plane::from(v[0], v[1], v[2]);
    }
    void flip() { std::reverse(v.begin(), v.end()); pl.flip(); }
};

enum { CO = 0, FR = 1, BK = 2, SP = 3 };

static void splitPoly(const Plane& pl, const Poly& pg,
    std::vector<Poly>& cf, std::vector<Poly>& cb,
    std::vector<Poly>& fr, std::vector<Poly>& bk)
{
    int n = (int)pg.v.size();
    int ty = 0;
    std::vector<int> ts(n);
    for (int i = 0; i < n; i++) {
        double d = pl.n.dot(pg.v[i]) - pl.w;
        ts[i] = d < -EPS ? BK : d > EPS ? FR : CO;
        ty |= ts[i];
    }
    if (ty == 0) { (pl.n.dot(pg.pl.n) > 0 ? cf : cb).push_back(pg); return; }
    if (ty == FR) { fr.push_back(pg); return; }
    if (ty == BK) { bk.push_back(pg); return; }
    std::vector<V3> fv, bv;
    for (int i = 0; i < n; i++) {
        int j = (i + 1) % n;
        if (ts[i] != BK) fv.push_back(pg.v[i]);
        if (ts[i] != FR) bv.push_back(pg.v[i]);
        if ((ts[i] | ts[j]) == SP) {
            V3 d = pg.v[j] - pg.v[i];
            double t = (pl.w - pl.n.dot(pg.v[i])) / pl.n.dot(d);
            V3 m = pg.v[i] + d * t;
            fv.push_back(m);
            bv.push_back(m);
        }
    }
    if (fv.size() >= 3) fr.emplace_back(std::move(fv));
    if (bv.size() >= 3) bk.emplace_back(std::move(bv));
}

struct Node {
    Plane pl;
    Node* fr = nullptr;
    Node* bk = nullptr;
    std::vector<Poly> ps;
    ~Node() { delete fr; delete bk; }

    void build(std::vector<Poly> input) {
        struct Task { Node* n; std::vector<Poly> p; };
        std::stack<Task> stk;
        stk.push({this, std::move(input)});
        while (!stk.empty()) {
            Task task = std::move(stk.top()); stk.pop();
            if (task.p.empty()) continue;
            Node* nd = task.n;
            if (!nd->pl.ok()) nd->pl = task.p[0].pl;
            std::vector<Poly> f, b;
            for (auto& p : task.p)
                splitPoly(nd->pl, p, nd->ps, nd->ps, f, b);
            if (!f.empty()) { nd->fr = new Node(); stk.push({nd->fr, std::move(f)}); }
            if (!b.empty()) { nd->bk = new Node(); stk.push({nd->bk, std::move(b)}); }
        }
    }

    std::vector<Poly> clip(std::vector<Poly> input) const {
        struct Task { const Node* n; std::vector<Poly> p; };
        std::vector<Poly> res;
        std::stack<Task> stk;
        stk.push({this, std::move(input)});
        while (!stk.empty()) {
            Task task = std::move(stk.top()); stk.pop();
            const Node* nd = task.n;
            if (!nd->pl.ok()) {
                for (auto& p : task.p) res.push_back(std::move(p));
                continue;
            }
            std::vector<Poly> f, b, c1, c2;
            for (auto& p : task.p)
                splitPoly(nd->pl, p, c1, c2, f, b);
            for (auto& p : c1) f.push_back(std::move(p));
            for (auto& p : c2) b.push_back(std::move(p));
            if (nd->fr && !f.empty()) stk.push({nd->fr, std::move(f)});
            else for (auto& p : f) res.push_back(std::move(p));
            if (nd->bk && !b.empty()) stk.push({nd->bk, std::move(b)});
        }
        return res;
    }

    void clipTo(const Node& o) {
        std::stack<Node*> stk;
        stk.push(this);
        while (!stk.empty()) {
            Node* nd = stk.top(); stk.pop();
            nd->ps = o.clip(std::move(nd->ps));
            if (nd->fr) stk.push(nd->fr);
            if (nd->bk) stk.push(nd->bk);
        }
    }

    void invert() {
        std::stack<Node*> stk;
        stk.push(this);
        while (!stk.empty()) {
            Node* nd = stk.top(); stk.pop();
            for (auto& p : nd->ps) p.flip();
            nd->pl.flip();
            std::swap(nd->fr, nd->bk);
            if (nd->fr) stk.push(nd->fr);
            if (nd->bk) stk.push(nd->bk);
        }
    }

    void collect(std::vector<Poly>& out) const {
        std::stack<const Node*> stk;
        stk.push(this);
        while (!stk.empty()) {
            const Node* nd = stk.top(); stk.pop();
            out.insert(out.end(), nd->ps.begin(), nd->ps.end());
            if (nd->fr) stk.push(nd->fr);
            if (nd->bk) stk.push(nd->bk);
        }
    }
};

int main() {
    int VA, FA, VB, FB;
    std::scanf("%d%d%d%d", &VA, &FA, &VB, &FB);

    std::vector<V3> va(VA), vb(VB);
    for (int i = 0; i < VA; i++) std::scanf("%lf%lf%lf", &va[i].x, &va[i].y, &va[i].z);
    std::vector<Poly> pa(FA);
    for (int i = 0; i < FA; i++) {
        int a, b, c; std::scanf("%d%d%d", &a, &b, &c);
        pa[i] = Poly({va[a], va[b], va[c]});
    }
    for (int i = 0; i < VB; i++) std::scanf("%lf%lf%lf", &vb[i].x, &vb[i].y, &vb[i].z);
    std::vector<Poly> pb(FB);
    for (int i = 0; i < FB; i++) {
        int a, b, c; std::scanf("%d%d%d", &a, &b, &c);
        pb[i] = Poly({vb[a], vb[b], vb[c]});
    }

    std::mt19937 rng(12345);
    std::shuffle(pa.begin(), pa.end(), rng);
    std::shuffle(pb.begin(), pb.end(), rng);

    Node A, B;
    A.build(std::move(pa));
    B.build(std::move(pb));

    A.clipTo(B);
    B.clipTo(A);
    B.invert();
    B.clipTo(A);
    B.invert();

    std::vector<Poly> res;
    A.collect(res);
    B.collect(res);

    struct VKey {
        long long a, b, c;
        bool operator==(const VKey& o) const { return a == o.a && b == o.b && c == o.c; }
    };
    struct VHash {
        size_t operator()(const VKey& k) const {
            return (size_t)k.a * 73856093ULL ^ (size_t)k.b * 19349663ULL ^ (size_t)k.c * 83492791ULL;
        }
    };

    const double G = 1e-9;
    std::unordered_map<VKey, int, VHash> vmap;
    std::vector<V3> ov;
    std::vector<int> ot;

    auto vid = [&](const V3& v) -> int {
        VKey k{(long long)std::round(v.x / G), (long long)std::round(v.y / G), (long long)std::round(v.z / G)};
        auto it = vmap.find(k);
        if (it != vmap.end()) return it->second;
        int id = (int)ov.size();
        ov.push_back(v);
        vmap[k] = id;
        return id;
    };

    for (auto& p : res) {
        int n = (int)p.v.size();
        if (n < 3) continue;
        int v0 = vid(p.v[0]);
        for (int i = 1; i + 1 < n; i++) {
            int v1 = vid(p.v[i]), v2 = vid(p.v[i + 1]);
            if (v0 != v1 && v1 != v2 && v0 != v2) {
                ot.push_back(v0);
                ot.push_back(v1);
                ot.push_back(v2);
            }
        }
    }

    std::printf("%d %d\n", (int)ov.size(), (int)ot.size() / 3);
    for (auto& v : ov) std::printf("%.15g %.15g %.15g\n", v.x, v.y, v.z);
    for (int i = 0; i < (int)ot.size(); i += 3)
        std::printf("%d %d %d\n", ot[i], ot[i + 1], ot[i + 2]);

    return 0;
}
