using System;
using System.Collections.Generic;
using System.IO;
using System.Numerics;
using System.Text;

class Solution
{
    static byte[] buf = new byte[1 << 16];
    static int bp = 0, bl = 0;
    static Stream si;

    static int RB()
    {
        if (bp >= bl) { bl = si.Read(buf, 0, buf.Length); bp = 0; if (bl == 0) return -1; }
        return buf[bp++];
    }

    static int RI()
    {
        int c = RB();
        while (c <= ' ' && c >= 0) c = RB();
        bool n = c == '-'; if (n) c = RB();
        int v = 0;
        while (c >= '0') { v = v * 10 + c - '0'; c = RB(); }
        return n ? -v : v;
    }

    static long RL()
    {
        int c = RB();
        while (c <= ' ' && c >= 0) c = RB();
        bool n = c == '-'; if (n) c = RB();
        long v = 0;
        while (c >= '0') { v = v * 10 + c - '0'; c = RB(); }
        return n ? -v : v;
    }

    static void Main()
    {
        si = Console.OpenStandardInput();
        int N = RI(), M = RI();

        if (M == 0) { Console.Write("0\n"); return; }

        var adj = new List<(int to, long w, int idx)>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<(int, long, int)>();
        int[] eu = new int[M], ev = new int[M];
        long[] ew = new long[M];

        for (int i = 0; i < M; i++)
        {
            eu[i] = RI(); ev[i] = RI(); ew[i] = RL();
            adj[eu[i]].Add((ev[i], ew[i], i));
            adj[ev[i]].Add((eu[i], ew[i], i));
        }

        int[] deg = new int[N];
        for (int i = 0; i < M; i++) { deg[eu[i]]++; deg[ev[i]]++; }
        var odd = new List<int>();
        for (int i = 0; i < N; i++) if (deg[i] % 2 == 1) odd.Add(i);

        var extra = new List<(int u, int v)>();

        if (odd.Count > 0 && odd.Count <= 20)
        {
            int K = odd.Count;
            long[][] dist = new long[K][];
            int[][] pred = new int[K][];

            for (int i = 0; i < K; i++)
            {
                dist[i] = new long[N];
                pred[i] = new int[N];
                Array.Fill(dist[i], long.MaxValue);
                Array.Fill(pred[i], -1);
                dist[i][odd[i]] = 0;
                var pq = new PriorityQueue<int, long>();
                pq.Enqueue(odd[i], 0);
                while (pq.Count > 0)
                {
                    pq.TryDequeue(out int u, out long d);
                    if (d > dist[i][u]) continue;
                    foreach (var (v, w, _) in adj[u])
                    {
                        long nd = d + w;
                        if (nd < dist[i][v])
                        {
                            dist[i][v] = nd;
                            pred[i][v] = u;
                            pq.Enqueue(v, nd);
                        }
                    }
                }
            }

            int full = (1 << K) - 1;
            long[] dp = new long[1 << K];
            int[] dpJ = new int[1 << K];
            Array.Fill(dp, long.MaxValue);
            dp[0] = 0;

            for (int mask = 0; mask < (1 << K); mask++)
            {
                if (dp[mask] == long.MaxValue) continue;
                int rem = ~mask & full;
                if (rem == 0) continue;
                int a = BitOperations.TrailingZeroCount(rem);
                int rest = rem & ~(1 << a);
                while (rest != 0)
                {
                    int b = BitOperations.TrailingZeroCount(rest);
                    long c = dist[a][odd[b]];
                    if (c < long.MaxValue)
                    {
                        int nm = mask | (1 << a) | (1 << b);
                        if (dp[mask] + c < dp[nm])
                        {
                            dp[nm] = dp[mask] + c;
                            dpJ[nm] = b;
                        }
                    }
                    rest &= rest - 1;
                }
            }

            int cur = full;
            while (cur != 0)
            {
                int a = BitOperations.TrailingZeroCount(cur);
                int b = dpJ[cur];
                int node = odd[b];
                while (node != odd[a])
                {
                    int p = pred[a][node];
                    extra.Add((p, node));
                    node = p;
                }
                cur ^= (1 << a) | (1 << b);
            }
        }
        else if (odd.Count > 0)
        {
            var unmatched = new HashSet<int>(odd);
            long[] dists = new long[N];
            int[] preds = new int[N];
            int[] ver = new int[N];
            int cv = 0;

            while (unmatched.Count > 0)
            {
                cv++;
                int src = -1;
                foreach (int v in unmatched) { src = v; break; }
                unmatched.Remove(src);

                dists[src] = 0; ver[src] = cv;
                var pq = new PriorityQueue<int, long>();
                pq.Enqueue(src, 0);
                int target = -1;

                while (pq.Count > 0)
                {
                    pq.TryDequeue(out int u, out long d);
                    if (d > dists[u]) continue;
                    if (u != src && unmatched.Contains(u)) { target = u; break; }
                    foreach (var (v, w, _) in adj[u])
                    {
                        long nd = d + w;
                        if (ver[v] != cv || nd < dists[v])
                        {
                            dists[v] = nd; preds[v] = u; ver[v] = cv;
                            pq.Enqueue(v, nd);
                        }
                    }
                }

                if (target != -1)
                {
                    unmatched.Remove(target);
                    int node = target;
                    while (node != src)
                    {
                        int p = preds[node];
                        extra.Add((p, node));
                        node = p;
                    }
                }
            }
        }

        int T = M + extra.Count;
        var adjE = new List<int>[N];
        for (int i = 0; i < N; i++) adjE[i] = new List<int>();
        int[] eU = new int[T], eV = new int[T];
        bool[] used = new bool[T];

        for (int i = 0; i < M; i++)
        {
            eU[i] = eu[i]; eV[i] = ev[i];
            adjE[eu[i]].Add(i); adjE[ev[i]].Add(i);
        }
        for (int i = 0; i < extra.Count; i++)
        {
            int idx = M + i;
            eU[idx] = extra[i].u; eV[idx] = extra[i].v;
            adjE[extra[i].u].Add(idx); adjE[extra[i].v].Add(idx);
        }

        int[] ptr = new int[N];
        var circuit = new List<int>(T + 1);
        var stack = new Stack<int>();
        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            bool found = false;
            while (ptr[u] < adjE[u].Count)
            {
                int ei = adjE[u][ptr[u]++];
                if (used[ei]) continue;
                used[ei] = true;
                int v = eU[ei] == u ? eV[ei] : eU[ei];
                stack.Push(v);
                found = true;
                break;
            }
            if (!found) { stack.Pop(); circuit.Add(u); }
        }

        var sw = new StreamWriter(Console.OpenStandardOutput(), Encoding.ASCII, 1 << 16);
        for (int i = 0; i < circuit.Count; i++)
        {
            if (i > 0) sw.Write(' ');
            sw.Write(circuit[i]);
        }
        sw.Write('\n');
        sw.Flush();
    }
}
