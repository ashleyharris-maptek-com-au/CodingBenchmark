using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    static int[] seg;
    static int sz;

    static void SegUpdate(int pos, int delta)
    {
        int i = pos + sz;
        seg[i] += delta;
        for (i >>= 1; i >= 1; i >>= 1)
            seg[i] = Math.Max(seg[2 * i], seg[2 * i + 1]);
    }

    static int SegCeiling(int node, int nL, int nR, int lo)
    {
        if (nR < lo || seg[node] == 0) return -1;
        if (nL == nR) return nL;
        int mid = (nL + nR) >> 1;
        int r = SegCeiling(2 * node, nL, mid, lo);
        return r != -1 ? r : SegCeiling(2 * node + 1, mid + 1, nR, lo);
    }

    static void SolveSegTree(int n, int stockLen, int[] cuts, int[] order)
    {
        sz = 1;
        while (sz <= stockLen) sz <<= 1;
        seg = new int[2 * sz];

        var capBins = new Dictionary<int, List<int>>();
        int numBins = 0;
        var binItems = new List<List<int>>();

        for (int i = 0; i < n; i++)
        {
            int ci = order[i];
            int cs = cuts[ci];
            int bestCap = SegCeiling(1, 0, sz - 1, cs);
            int binIdx, oldCap;

            if (bestCap >= 0 && bestCap <= stockLen)
            {
                var list = capBins[bestCap];
                binIdx = list[list.Count - 1];
                list.RemoveAt(list.Count - 1);
                if (list.Count == 0) capBins.Remove(bestCap);
                SegUpdate(bestCap, -1);
                oldCap = bestCap;
            }
            else
            {
                binIdx = numBins++;
                binItems.Add(new List<int>());
                oldCap = stockLen;
            }

            binItems[binIdx].Add(ci);
            int newCap = oldCap - cs;
            if (newCap > 0)
            {
                if (!capBins.TryGetValue(newCap, out var list2))
                {
                    list2 = new List<int>();
                    capBins[newCap] = list2;
                }
                list2.Add(binIdx);
                SegUpdate(newCap, 1);
            }
        }

        WriteOutput(numBins, binItems);
    }

    static void SolveSortedSet(int n, int stockLen, int[] cuts, int[] order)
    {
        var openBins = new SortedSet<(int cap, int id)>();
        int numBins = 0;
        var binItems = new List<List<int>>();

        for (int i = 0; i < n; i++)
        {
            int ci = order[i];
            int cs = cuts[ci];
            int binIdx, oldCap;

            if (openBins.Count > 0 && openBins.Max.cap >= cs)
            {
                var best = openBins.GetViewBetween((cs, 0), (stockLen, int.MaxValue)).Min;
                openBins.Remove(best);
                binIdx = best.id;
                oldCap = best.cap;
            }
            else
            {
                binIdx = numBins++;
                binItems.Add(new List<int>());
                oldCap = stockLen;
            }

            binItems[binIdx].Add(ci);
            int newCap = oldCap - cs;
            if (newCap > 0)
            {
                openBins.Add((newCap, binIdx));
            }
        }

        WriteOutput(numBins, binItems);
    }

    static void WriteOutput(int numBins, List<List<int>> binItems)
    {
        var sb = new StringBuilder();
        sb.Append(numBins);
        sb.Append('\n');
        for (int i = 0; i < numBins; i++)
        {
            var items = binItems[i];
            for (int j = 0; j < items.Count; j++)
            {
                if (j > 0) sb.Append(' ');
                sb.Append(items[j]);
            }
            sb.Append('\n');
        }
        using var w = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        w.Write(sb);
    }

    static void Main()
    {
        var ms = new MemoryStream();
        Console.OpenStandardInput().CopyTo(ms);
        byte[] buf = ms.GetBuffer();
        int bufLen = (int)ms.Length;
        int p = 0;

        int ReadInt()
        {
            while (p < bufLen && (buf[p] < (byte)'0' || buf[p] > (byte)'9')) p++;
            int v = 0;
            while (p < bufLen && buf[p] >= (byte)'0' && buf[p] <= (byte)'9')
                v = v * 10 + (buf[p++] - '0');
            return v;
        }

        int n = ReadInt();
        int stockLen = ReadInt();
        int[] cuts = new int[n];
        for (int i = 0; i < n; i++) cuts[i] = ReadInt();

        if (n == 0)
        {
            Console.WriteLine("0");
            return;
        }

        int[] negCuts = new int[n];
        int[] order = new int[n];
        for (int i = 0; i < n; i++) { negCuts[i] = -cuts[i]; order[i] = i; }
        Array.Sort(negCuts, order);

        if (stockLen <= 20_000_000)
            SolveSegTree(n, stockLen, cuts, order);
        else
            SolveSortedSet(n, stockLen, cuts, order);
    }
}
