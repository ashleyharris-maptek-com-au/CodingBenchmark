using System;
using System.Collections.Generic;
using System.Linq;

class MinesweeperSolver
{
    static int H, W;
    static char[,] board;
    static int[] dx = { -1, -1, -1, 0, 0, 1, 1, 1 };
    static int[] dy = { -1, 0, 1, -1, 1, -1, 0, 1 };

    static void Main()
    {
        var firstLine = Console.ReadLine().Split();
        H = int.Parse(firstLine[0]);
        W = int.Parse(firstLine[1]);
        board = new char[H, W];

        for (int r = 0; r < H; r++)
        {
            string line = Console.ReadLine();
            for (int c = 0; c < W; c++)
                board[r, c] = c < line.Length ? line[c] : ' ';
        }

        var safe = new HashSet<long>();
        var mines = new HashSet<long>();

        bool changed = true;
        while (changed)
        {
            changed = false;
            for (int r = 0; r < H; r++)
            {
                for (int c = 0; c < W; c++)
                {
                    if (board[r, c] < '0' || board[r, c] > '8') continue;
                    int num = board[r, c] - '0';
                    var unknowns = new List<(int r, int c)>();
                    int mineCount = 0;

                    for (int d = 0; d < 8; d++)
                    {
                        int nr = r + dx[d], nc = c + dy[d];
                        if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;
                        if (board[nr, nc] == '*') mineCount++;
                        else if (board[nr, nc] == ' ') unknowns.Add((nr, nc));
                    }

                    if (unknowns.Count == 0) continue;

                    if (mineCount == num)
                    {
                        foreach (var (ur, uc) in unknowns)
                        {
                            long key = (long)ur * W + uc;
                            if (safe.Add(key))
                            {
                                board[ur, uc] = 'S';
                                changed = true;
                            }
                        }
                    }
                    else if (mineCount + unknowns.Count == num)
                    {
                        foreach (var (ur, uc) in unknowns)
                        {
                            board[ur, uc] = '*';
                            changed = true;
                        }
                    }
                }
            }
        }

        // Collect border unknowns (unknowns adjacent to numbered cells) for CSP
        var borderUnknowns = new HashSet<long>();
        var constraints = new List<(List<int> cells, int minesNeeded)>();
        var unknownIndex = new Dictionary<long, int>();

        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (board[r, c] < '0' || board[r, c] > '8') continue;
                int num = board[r, c] - '0';
                var unknowns = new List<long>();
                int mineCount = 0;

                for (int d = 0; d < 8; d++)
                {
                    int nr = r + dx[d], nc = c + dy[d];
                    if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;
                    if (board[nr, nc] == '*') mineCount++;
                    else if (board[nr, nc] == ' ') unknowns.Add((long)nr * W + nc);
                }

                if (unknowns.Count == 0) continue;
                int remaining = num - mineCount;
                if (remaining < 0 || remaining > unknowns.Count) continue;

                foreach (var u in unknowns)
                {
                    borderUnknowns.Add(u);
                    if (!unknownIndex.ContainsKey(u))
                        unknownIndex[u] = unknownIndex.Count;
                }

                constraints.Add((unknowns.Select(u => unknownIndex[u]).ToList(), remaining));
            }
        }

        if (borderUnknowns.Count > 0)
        {
            int n = unknownIndex.Count;
            var idxToKey = new long[n];
            foreach (var kv in unknownIndex)
                idxToKey[kv.Value] = kv.Key;

            // Find connected components
            var adj = new List<int>[n];
            for (int i = 0; i < n; i++) adj[i] = new List<int>();

            foreach (var (cells, _) in constraints)
            {
                for (int i = 0; i < cells.Count; i++)
                    for (int j = i + 1; j < cells.Count; j++)
                    {
                        adj[cells[i]].Add(cells[j]);
                        adj[cells[j]].Add(cells[i]);
                    }
            }

            var visited = new int[n];
            Array.Fill(visited, -1);
            var components = new List<List<int>>();

            for (int i = 0; i < n; i++)
            {
                if (visited[i] >= 0) continue;
                int compId = components.Count;
                var comp = new List<int>();
                var queue = new Queue<int>();
                queue.Enqueue(i);
                visited[i] = compId;
                while (queue.Count > 0)
                {
                    int cur = queue.Dequeue();
                    comp.Add(cur);
                    foreach (int nb in adj[cur])
                    {
                        if (visited[nb] < 0)
                        {
                            visited[nb] = compId;
                            queue.Enqueue(nb);
                        }
                    }
                }
                components.Add(comp);
            }

            foreach (var comp in components)
            {
                var compSet = new HashSet<int>(comp);
                var localConstraints = new List<(List<int> cells, int minesNeeded)>();
                foreach (var (cells, mn) in constraints)
                {
                    var filtered = cells.Where(c => compSet.Contains(c)).ToList();
                    if (filtered.Count > 0)
                    {
                        int adjustedMines = mn;
                        var removed = cells.Where(c => !compSet.Contains(c)).ToList();
                        if (removed.Count > 0) continue;
                        localConstraints.Add((filtered, adjustedMines));
                    }
                }

                if (comp.Count <= 25)
                {
                    SolveByEnumeration(comp, localConstraints, idxToKey, safe);
                }
                else
                {
                    SolveWithSubsetConstraints(comp, localConstraints, idxToKey, safe);
                }
            }
        }

        // Restore 'S' marked cells to ' ' for output check, collect results
        var result = new List<(int c, int r)>();
        foreach (long key in safe)
        {
            int r = (int)(key / W);
            int c = (int)(key % W);
            result.Add((c, r));
        }

        Console.WriteLine(result.Count);
        foreach (var (c, r) in result)
            Console.WriteLine($"{c} {r}");
    }

    static void SolveByEnumeration(List<int> comp, List<(List<int> cells, int minesNeeded)> constraints,
        long[] idxToKey, HashSet<long> safe)
    {
        int cn = comp.Count;
        var localMap = new Dictionary<int, int>();
        for (int i = 0; i < cn; i++)
            localMap[comp[i]] = i;

        var remapped = new List<(List<int> cells, int minesNeeded)>();
        foreach (var (cells, mn) in constraints)
            remapped.Add((cells.Select(c => localMap[c]).ToList(), mn));

        var canBeMine = new bool[cn];
        var canBeSafe = new bool[cn];

        long total = 1L << cn;
        for (long mask = 0; mask < total; mask++)
        {
            bool valid = true;
            foreach (var (cells, mn) in remapped)
            {
                int cnt = 0;
                foreach (int idx in cells)
                    if ((mask & (1L << idx)) != 0) cnt++;
                if (cnt != mn) { valid = false; break; }
            }
            if (!valid) continue;

            for (int i = 0; i < cn; i++)
            {
                if ((mask & (1L << i)) != 0) canBeMine[i] = true;
                else canBeSafe[i] = true;
            }
        }

        for (int i = 0; i < cn; i++)
        {
            if (canBeSafe[i] && !canBeMine[i])
                safe.Add(idxToKey[comp[i]]);
        }
    }

    static void SolveWithSubsetConstraints(List<int> comp, List<(List<int> cells, int minesNeeded)> allConstraints,
        long[] idxToKey, HashSet<long> safe)
    {
        var compSet = new HashSet<int>(comp);
        var localConstraints = new List<(HashSet<int> cells, int minesNeeded)>();
        foreach (var (cells, mn) in allConstraints)
        {
            var filtered = new HashSet<int>(cells.Where(c => compSet.Contains(c)));
            if (filtered.Count > 0)
                localConstraints.Add((filtered, mn));
        }

        bool progress = true;
        int iterations = 0;
        while (progress && iterations < 1000)
        {
            progress = false;
            iterations++;

            for (int i = localConstraints.Count - 1; i >= 0; i--)
            {
                var (cells, mn) = localConstraints[i];
                if (cells.Count == 0) continue;

                if (mn == 0)
                {
                    foreach (int idx in cells)
                        safe.Add(idxToKey[idx]);
                    var toRemove = new HashSet<int>(cells);
                    foreach (var (oc, _) in localConstraints)
                        oc.ExceptWith(toRemove);
                    cells.Clear();
                    localConstraints[i] = (cells, 0);
                    progress = true;
                }
                else if (mn == cells.Count)
                {
                    var mineCells = new HashSet<int>(cells);
                    for (int j = 0; j < localConstraints.Count; j++)
                    {
                        if (j == i) continue;
                        var (oc, om) = localConstraints[j];
                        int removed = 0;
                        foreach (int idx in mineCells)
                            if (oc.Remove(idx)) removed++;
                        if (removed > 0)
                        {
                            localConstraints[j] = (oc, om - removed);
                            progress = true;
                        }
                    }
                    cells.Clear();
                    localConstraints[i] = (cells, 0);
                    progress = true;
                }
            }

            localConstraints.RemoveAll(x => x.cells.Count == 0);

            int beforeCount = localConstraints.Count;
            for (int i = 0; i < localConstraints.Count && i < 500; i++)
            {
                var (ca, ma) = localConstraints[i];
                if (ca.Count == 0) continue;
                for (int j = 0; j < localConstraints.Count && j < 500; j++)
                {
                    if (i == j) continue;
                    var (cb, mb) = localConstraints[j];
                    if (cb.Count == 0 || ca.Count >= cb.Count) continue;

                    if (ca.IsSubsetOf(cb))
                    {
                        var diff = new HashSet<int>(cb);
                        diff.ExceptWith(ca);
                        int diffMines = mb - ma;

                        if (diffMines >= 0 && diffMines <= diff.Count)
                        {
                            bool isNew = true;
                            foreach (var (ec, em) in localConstraints)
                            {
                                if (em == diffMines && ec.Count == diff.Count && ec.SetEquals(diff))
                                { isNew = false; break; }
                            }
                            if (isNew)
                            {
                                localConstraints.Add((diff, diffMines));
                                progress = true;
                            }
                        }
                    }
                }
            }

            localConstraints.RemoveAll(x => x.cells.Count == 0);

            // Deduplicate
            var seen = new List<(HashSet<int> cells, int mn)>();
            var deduped = new List<(HashSet<int> cells, int minesNeeded)>();
            foreach (var (cells, mn) in localConstraints)
            {
                bool dup = false;
                foreach (var (sc, sm) in seen)
                {
                    if (sm == mn && sc.Count == cells.Count && sc.SetEquals(cells))
                    { dup = true; break; }
                }
                if (!dup)
                {
                    seen.Add((cells, mn));
                    deduped.Add((cells, mn));
                }
            }
            localConstraints = deduped;
        }
    }
}
