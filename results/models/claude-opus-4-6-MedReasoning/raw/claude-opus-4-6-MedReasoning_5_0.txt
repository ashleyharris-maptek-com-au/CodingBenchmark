use std::io::{self, Read, Write, BufWriter};
use std::collections::VecDeque;
use std::time::Instant;

const DX: [i32; 4] = [0, 1, 0, -1];
const DY: [i32; 4] = [1, 0, -1, 0];

fn main() {
    std::thread::Builder::new()
        .stack_size(64 * 1024 * 1024)
        .spawn(run)
        .unwrap()
        .join()
        .unwrap();
}

fn run() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_ascii_whitespace();
    let w: usize = it.next().unwrap().parse().unwrap();
    let h: usize = it.next().unwrap().parse().unwrap();
    let obs: usize = it.next().unwrap().parse().unwrap();
    let need_cycle: bool = it.next().unwrap() != "0";
    let n = w * h;
    let mut blocked = vec![false; n];
    for _ in 0..obs {
        let x: usize = it.next().unwrap().parse().unwrap();
        let y: usize = it.next().unwrap().parse().unwrap();
        blocked[x * h + y] = true;
    }
    let out = io::stdout();
    let mut out = BufWriter::new(out.lock());

    if n == 0 || blocked[0] {
        writeln!(out, "0").unwrap();
        return;
    }
    let total = n - obs;
    if total == 1 {
        writeln!(out, "1").unwrap();
        writeln!(out, "0 0").unwrap();
        return;
    }

    let (mut ev, mut od) = (0usize, 0usize);
    for i in 0..n {
        if !blocked[i] {
            if ((i / h) + (i % h)) % 2 == 0 { ev += 1; } else { od += 1; }
        }
    }
    if ev < od || ev > od + 1 || (need_cycle && ev != od) {
        writeln!(out, "0").unwrap();
        return;
    }

    {
        let mut vis = vec![false; n];
        let mut q = VecDeque::new();
        vis[0] = true;
        q.push_back(0usize);
        let mut c = 1usize;
        while let Some(ci) = q.pop_front() {
            for d in 0..4 {
                let nx = (ci / h) as i32 + DX[d];
                let ny = (ci % h) as i32 + DY[d];
                if nx >= 0 && nx < w as i32 && ny >= 0 && ny < h as i32 {
                    let ni = nx as usize * h + ny as usize;
                    if !blocked[ni] && !vis[ni] {
                        vis[ni] = true;
                        q.push_back(ni);
                        c += 1;
                    }
                }
            }
        }
        if c != total {
            writeln!(out, "0").unwrap();
            return;
        }
    }

    let mut deg = vec![0u8; n];
    for i in 0..n {
        if blocked[i] { continue; }
        for d in 0..4 {
            let nx = (i / h) as i32 + DX[d];
            let ny = (i % h) as i32 + DY[d];
            if nx >= 0 && nx < w as i32 && ny >= 0 && ny < h as i32 && !blocked[nx as usize * h + ny as usize] {
                deg[i] += 1;
            }
        }
    }

    let t0 = Instant::now();
    let mut s = State {
        w: w as i32, h: h as i32, hu: h,
        blocked, deg,
        vis: vec![false; n],
        path: vec![0u32; total],
        plen: 0, target: total, need_cycle,
        ccs: vec![0u32; n], ccg: 0,
        stk: Vec::with_capacity(total),
        t0, timed_out: false, nc: 0,
    };

    s.vis[0] = true;
    s.path[0] = 0;
    s.plen = 1;
    for d in 0..4 {
        let (nx, ny) = (DX[d], DY[d]);
        if nx >= 0 && nx < s.w && ny >= 0 && ny < s.h {
            let ni = nx as usize * h + ny as usize;
            if !s.blocked[ni] { s.deg[ni] -= 1; }
        }
    }

    if s.bt() {
        emit(&s, &mut out);
        return;
    }

    if s.timed_out {
        let mut rng = 0xdeadbeef_cafe_u64;
        loop {
            if t0.elapsed().as_secs() > 290 { break; }
            for _ in 0..10000 {
                if s.greedy(&mut rng) {
                    emit(&s, &mut out);
                    return;
                }
            }
        }
    }

    writeln!(out, "0").unwrap();
}

fn emit(s: &State, out: &mut BufWriter<io::StdoutLock>) {
    writeln!(out, "{}", s.plen).unwrap();
    for i in 0..s.plen {
        let p = s.path[i] as usize;
        writeln!(out, "{} {}", p / s.hu, p % s.hu).unwrap();
    }
}

struct State {
    w: i32, h: i32, hu: usize,
    blocked: Vec<bool>,
    deg: Vec<u8>,
    vis: Vec<bool>,
    path: Vec<u32>,
    plen: usize,
    target: usize,
    need_cycle: bool,
    ccs: Vec<u32>,
    ccg: u32,
    stk: Vec<usize>,
    t0: Instant,
    timed_out: bool,
    nc: u64,
}

impl State {
    fn bt(&mut self) -> bool {
        if self.plen == self.target {
            if self.need_cycle {
                let p = self.path[self.plen - 1] as usize;
                let (x, y) = (p / self.hu, p % self.hu);
                return (x == 0 && y == 1) || (x == 1 && y == 0);
            }
            return true;
        }

        self.nc += 1;
        if self.nc & 0xFFFF == 0 && self.t0.elapsed().as_secs() > 150 {
            self.timed_out = true;
            return false;
        }

        let cp = self.path[self.plen - 1] as usize;
        let (cx, cy) = ((cp / self.hu) as i32, (cp % self.hu) as i32);
        let rem = self.target - self.plen;

        let mut nb = [(0u8, 0u32); 4];
        let mut nbc = 0;
        for d in 0..4 {
            let (nx, ny) = (cx + DX[d], cy + DY[d]);
            if nx >= 0 && nx < self.w && ny >= 0 && ny < self.h {
                let ni = nx as usize * self.hu + ny as usize;
                if !self.blocked[ni] && !self.vis[ni] {
                    nb[nbc] = (self.deg[ni], ni as u32);
                    nbc += 1;
                }
            }
        }
        if nbc == 0 { return false; }
        nb[..nbc].sort_unstable();

        for i in 0..nbc {
            if self.timed_out { return false; }
            let ni = nb[i].1 as usize;
            let (nx, ny) = ((ni / self.hu) as i32, (ni % self.hu) as i32);

            self.vis[ni] = true;
            self.path[self.plen] = ni as u32;
            self.plen += 1;

            let mut upd = [0usize; 4];
            let mut uc = 0;
            let mut ok = true;
            let mut zc = 0u32;

            for d in 0..4 {
                let (nnx, nny) = (nx + DX[d], ny + DY[d]);
                if nnx >= 0 && nnx < self.w && nny >= 0 && nny < self.h {
                    let nni = nnx as usize * self.hu + nny as usize;
                    if !self.blocked[nni] && !self.vis[nni] {
                        self.deg[nni] -= 1;
                        upd[uc] = nni;
                        uc += 1;
                        if self.deg[nni] == 0 { zc += 1; }
                    }
                }
            }

            if uc == 0 && rem > 1 { ok = false; }
            if ok && (zc > 1 || (zc == 1 && rem > 2)) { ok = false; }

            if ok && self.need_cycle && rem > 2 {
                let mut has = false;
                for d in 0..4 {
                    let (sx, sy) = (DX[d], DY[d]);
                    if sx >= 0 && sx < self.w && sy >= 0 && sy < self.h {
                        let si = sx as usize * self.hu + sy as usize;
                        if !self.blocked[si] && !self.vis[si] { has = true; break; }
                    }
                }
                if !has { ok = false; }
            }

            if ok && rem > 1 && uc > 0 {
                ok = self.connected(upd[0], rem - 1);
            }

            if ok && self.bt() { return true; }

            self.plen -= 1;
            self.vis[ni] = false;
            for j in 0..uc { self.deg[upd[j]] += 1; }
        }
        false
    }

    fn connected(&mut self, start: usize, exp: usize) -> bool {
        if exp <= 1 { return true; }
        self.ccg += 1;
        let g = self.ccg;
        self.stk.clear();
        self.ccs[start] = g;
        self.stk.push(start);
        let mut c = 1usize;
        while let Some(ci) = self.stk.pop() {
            if c == exp { return true; }
            let (cx, cy) = ((ci / self.hu) as i32, (ci % self.hu) as i32);
            for d in 0..4 {
                let (nx, ny) = (cx + DX[d], cy + DY[d]);
                if nx >= 0 && nx < self.w && ny >= 0 && ny < self.h {
                    let ni = nx as usize * self.hu + ny as usize;
                    if !self.blocked[ni] && !self.vis[ni] && self.ccs[ni] != g {
                        self.ccs[ni] = g;
                        self.stk.push(ni);
                        c += 1;
                    }
                }
            }
        }
        c == exp
    }

    fn greedy(&mut self, rng: &mut u64) -> bool {
        self.vis.fill(false);
        self.vis[0] = true;
        self.path[0] = 0;
        self.plen = 1;

        while self.plen < self.target {
            let cp = self.path[self.plen - 1] as usize;
            let (cx, cy) = ((cp / self.hu) as i32, (cp % self.hu) as i32);

            let mut nb = [(0u8, 0u32); 4];
            let mut nbc = 0;
            for d in 0..4 {
                let (nx, ny) = (cx + DX[d], cy + DY[d]);
                if nx >= 0 && nx < self.w && ny >= 0 && ny < self.h {
                    let ni = nx as usize * self.hu + ny as usize;
                    if !self.blocked[ni] && !self.vis[ni] {
                        let mut dg = 0u8;
                        for d2 in 0..4 {
                            let (nnx, nny) = (nx + DX[d2], ny + DY[d2]);
                            if nnx >= 0 && nnx < self.w && nny >= 0 && nny < self.h {
                                let nni = nnx as usize * self.hu + nny as usize;
                                if !self.blocked[nni] && !self.vis[nni] { dg += 1; }
                            }
                        }
                        nb[nbc] = (dg, ni as u32);
                        nbc += 1;
                    }
                }
            }
            if nbc == 0 { return false; }

            let md = nb[..nbc].iter().map(|x| x.0).min().unwrap();
            let mut mi = [0usize; 4];
            let mut mc = 0;
            for j in 0..nbc {
                if nb[j].0 == md { mi[mc] = j; mc += 1; }
            }

            let ch = if mc == 1 { mi[0] } else { mi[xor64(rng) as usize % mc] };
            let ni = nb[ch].1 as usize;
            self.vis[ni] = true;
            self.path[self.plen] = ni as u32;
            self.plen += 1;
        }

        if self.need_cycle {
            let p = self.path[self.plen - 1] as usize;
            let (x, y) = (p / self.hu, p % self.hu);
            (x == 0 && y == 1) || (x == 1 && y == 0)
        } else {
            true
        }
    }
}

fn xor64(s: &mut u64) -> u64 {
    let mut x = *s;
    x ^= x << 13;
    x ^= x >> 7;
    x ^= x << 17;
    *s = x;
    x
}
