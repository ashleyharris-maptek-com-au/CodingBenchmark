#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <unordered_set>
#include <unordered_map>
#include <numeric>
#include <cstring>

using namespace std;

static const uint64_t MOD1 = 1000000007ULL;
static const uint64_t MOD2 = 998244353ULL;
static const uint64_t BASE1 = 131ULL;
static const uint64_t BASE2 = 137ULL;

struct DualHash {
    uint64_t h1, h2;
    bool operator==(const DualHash& o) const { return h1 == o.h1 && h2 == o.h2; }
};

struct DualHashHasher {
    size_t operator()(const DualHash& d) const {
        return (size_t)(d.h1 * 1000000009ULL + d.h2);
    }
};

int checkLengthProper(const vector<string>& strings, int len, string& result) {
    if (len == 0) { result = ""; return 0; }
    int n = (int)strings.size();

    uint64_t pow1 = 1, pow2 = 1;
    for (int i = 0; i < len; i++) {
        pow1 = pow1 * BASE1 % MOD1;
        pow2 = pow2 * BASE2 % MOD2;
    }

    unordered_map<DualHash, int, DualHashHasher> currentSet;
    {
        const string& s = strings[0];
        int sLen = (int)s.size();
        if (sLen < len) return -1;
        uint64_t h1 = 0, h2 = 0;
        for (int i = 0; i < len; i++) {
            h1 = (h1 * BASE1 + (unsigned char)s[i]) % MOD1;
            h2 = (h2 * BASE2 + (unsigned char)s[i]) % MOD2;
        }
        currentSet.insert({{h1, h2}, 0});
        for (int i = len; i < sLen; i++) {
            h1 = (h1 * BASE1 % MOD1 + (unsigned char)s[i] + MOD1 - (uint64_t)(unsigned char)s[i - len] * pow1 % MOD1) % MOD1;
            h2 = (h2 * BASE2 % MOD2 + (unsigned char)s[i] + MOD2 - (uint64_t)(unsigned char)s[i - len] * pow2 % MOD2) % MOD2;
            if (currentSet.find({h1, h2}) == currentSet.end())
                currentSet[{h1, h2}] = i - len + 1;
        }
    }

    for (int si = 1; si < n && !currentSet.empty(); si++) {
        const string& s = strings[si];
        int sLen = (int)s.size();
        if (sLen < len) return -1;

        unordered_set<DualHash, DualHashHasher> found;
        uint64_t h1 = 0, h2 = 0;
        for (int i = 0; i < len; i++) {
            h1 = (h1 * BASE1 + (unsigned char)s[i]) % MOD1;
            h2 = (h2 * BASE2 + (unsigned char)s[i]) % MOD2;
        }
        if (currentSet.count({h1, h2})) found.insert({h1, h2});
        for (int i = len; i < sLen; i++) {
            h1 = (h1 * BASE1 % MOD1 + (unsigned char)s[i] + MOD1 - (uint64_t)(unsigned char)s[i - len] * pow1 % MOD1) % MOD1;
            h2 = (h2 * BASE2 % MOD2 + (unsigned char)s[i] + MOD2 - (uint64_t)(unsigned char)s[i - len] * pow2 % MOD2) % MOD2;
            DualHash dh = {h1, h2};
            if (currentSet.count(dh)) found.insert(dh);
        }

        unordered_map<DualHash, int, DualHashHasher> newSet;
        for (auto& dh : found) {
            newSet[dh] = currentSet[dh];
        }
        currentSet = move(newSet);
    }

    if (currentSet.empty()) return -1;
    int pos = currentSet.begin()->second;
    result = strings[0].substr(pos, len);

    for (int si = 1; si < n; si++) {
        if (strings[si].find(result) == string::npos) {
            for (auto& [dh, p] : currentSet) {
                string candidate = strings[0].substr(p, len);
                bool ok = true;
                for (int sj = 1; sj < n; sj++) {
                    if (strings[sj].find(candidate) == string::npos) {
                        ok = false;
                        break;
                    }
                }
                if (ok) {
                    result = candidate;
                    return 1;
                }
            }
            return -1;
        }
    }
    return 1;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    cin.ignore();

    vector<string> strings(n);
    for (int i = 0; i < n; i++) {
        getline(cin, strings[i]);
    }

    if (n == 0) {
        cout << "\n";
        return 0;
    }
    if (n == 1) {
        cout << strings[0] << "\n";
        return 0;
    }

    for (auto& s : strings) {
        if (s.empty()) {
            cout << "\n";
            return 0;
        }
    }

    // Sort strings by length so strings[0] is shortest
    sort(strings.begin(), strings.end(), [](const string& a, const string& b) {
        return a.size() < b.size();
    });

    int minLen = (int)strings[0].size();
    int lo = 1, hi = minLen;
    string best = "";

    while (lo <= hi) {
        int mid = (lo + hi) / 2;
        string res;
        int r = checkLengthProper(strings, mid, res);
        if (r >= 0) {
            best = res;
            lo = mid + 1;
        } else {
            hi = mid - 1;
        }
    }
    cout << best << "\n";
    return 0;
}
