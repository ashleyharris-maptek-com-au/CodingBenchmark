use std::io::{self, Read, Write, BufWriter};
use std::collections::VecDeque;

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let bytes = input.as_bytes();
    let mut pos = 0usize;

    // Parse H
    while pos < bytes.len() && bytes[pos].is_ascii_whitespace() { pos += 1; }
    let mut h: usize = 0;
    while pos < bytes.len() && bytes[pos].is_ascii_digit() {
        h = h * 10 + (bytes[pos] - b'0') as usize;
        pos += 1;
    }
    // Parse W
    while pos < bytes.len() && (bytes[pos] == b' ' || bytes[pos] == b'\t') { pos += 1; }
    let mut w: usize = 0;
    while pos < bytes.len() && bytes[pos].is_ascii_digit() {
        w = w * 10 + (bytes[pos] - b'0') as usize;
        pos += 1;
    }
    // Skip newline after W
    if pos < bytes.len() && bytes[pos] == b'\r' { pos += 1; }
    if pos < bytes.len() && bytes[pos] == b'\n' { pos += 1; }

    let total = h * w;
    let mut maze = vec![0u8; total];
    let mut start = 0usize;
    let mut end = 0usize;

    for r in 0..h {
        let mut col = 0;
        while col < w && pos < bytes.len() {
            let b = bytes[pos];
            if b == b'\r' || b == b'\n' {
                while col < w {
                    maze[r * w + col] = b'#';
                    col += 1;
                }
                break;
            }
            let idx = r * w + col;
            maze[idx] = b;
            if b == b'A' { start = idx; }
            if b == b'B' { end = idx; }
            col += 1;
            pos += 1;
        }
        while pos < bytes.len() && bytes[pos] != b'\n' { pos += 1; }
        if pos < bytes.len() && bytes[pos] == b'\n' { pos += 1; }
    }

    // BFS with direction-encoded visited array (1 byte per cell)
    const UNVISITED: u8 = 0;
    const FROM_UP: u8 = 1;
    const FROM_DOWN: u8 = 2;
    const FROM_LEFT: u8 = 3;
    const FROM_RIGHT: u8 = 4;
    const START_MARK: u8 = 5;

    let mut visited = vec![UNVISITED; total];
    visited[start] = START_MARK;

    let mut queue = VecDeque::new();
    queue.push_back(start);

    let found = 'bfs: {
        if start == end { break 'bfs true; }
        while let Some(cur) = queue.pop_front() {
            let r = cur / w;
            let c = cur % w;

            macro_rules! try_move {
                ($next:expr, $dir:expr) => {
                    let next = $next;
                    if visited[next] == UNVISITED && maze[next] != b'#' {
                        visited[next] = $dir;
                        if next == end { break 'bfs true; }
                        queue.push_back(next);
                    }
                };
            }

            if r > 0     { try_move!(cur - w, FROM_DOWN); }
            if r + 1 < h { try_move!(cur + w, FROM_UP); }
            if c > 0     { try_move!(cur - 1, FROM_RIGHT); }
            if c + 1 < w { try_move!(cur + 1, FROM_LEFT); }
        }
        false
    };

    if !found {
        writeln!(out, "0").unwrap();
        return;
    }

    // Trace back path
    let mut path = Vec::new();
    let mut cur = end;
    loop {
        path.push(cur);
        if cur == start { break; }
        cur = match visited[cur] {
            FROM_UP => cur - w,
            FROM_DOWN => cur + w,
            FROM_LEFT => cur - 1,
            FROM_RIGHT => cur + 1,
            _ => break,
        };
    }
    path.reverse();

    writeln!(out, "{}", path.len()).unwrap();
    for &p in &path {
        writeln!(out, "{} {}", p / w, p % w).unwrap();
    }
}
