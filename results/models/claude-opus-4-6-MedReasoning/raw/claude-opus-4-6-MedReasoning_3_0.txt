use std::io::{self, BufRead, Write, BufWriter};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();
    let first_line = lines.next().unwrap().unwrap();
    let mut iter = first_line.split_whitespace();
    let n: usize = iter.next().unwrap().parse().unwrap();
    let m: usize = iter.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![vec![]; n];
    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);

    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut it = line.split_whitespace();
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v || u >= n || v >= n {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
        edges.push((u, v));
    }

    if n == 0 {
        return;
    }
    if n == 1 {
        writeln!(out, "0.0 0.0").unwrap();
        return;
    }

    let mut pos = initial_layout(n, &adj);

    let edge_count = edges.len();

    if n <= 50 && edge_count <= 200 {
        force_directed(&mut pos, &edges, &adj, n, 2000, 5.0);
    } else if n <= 500 {
        force_directed(&mut pos, &edges, &adj, n, 800, 10.0);
    } else if n <= 5000 {
        force_directed_large(&mut pos, &edges, &adj, n, 300, 20.0);
    } else if n <= 50000 {
        force_directed_large(&mut pos, &edges, &adj, n, 100, 50.0);
    } else {
        force_directed_large(&mut pos, &edges, &adj, n, 40, 100.0);
    }

    for i in 0..n {
        writeln!(out, "{:.6} {:.6}", pos[i].0, pos[i].1).unwrap();
    }
}

fn initial_layout(n: usize, adj: &[Vec<usize>]) -> Vec<(f64, f64)> {
    let mut pos = vec![(0.0f64, 0.0f64); n];
    let mut visited = vec![false; n];
    let mut components: Vec<Vec<usize>> = Vec::new();

    for start in 0..n {
        if visited[start] {
            continue;
        }
        let mut comp = Vec::new();
        let mut queue = VecDeque::new();
        queue.push_back(start);
        visited[start] = true;
        while let Some(u) = queue.pop_front() {
            comp.push(u);
            for &v in &adj[u] {
                if !visited[v] {
                    visited[v] = true;
                    queue.push_back(v);
                }
            }
        }
        components.push(comp);
    }

    let mut offset_x = 0.0f64;

    for comp in &components {
        let cn = comp.len();
        if cn == 1 {
            pos[comp[0]] = (offset_x, 0.0);
            offset_x += 3.0;
            continue;
        }

        let best_start = *comp.iter().max_by_key(|&&u| adj[u].len()).unwrap();
        let mut dist = vec![usize::MAX; n];
        dist[best_start] = 0;

        let mut queue = VecDeque::new();
        queue.push_back(best_start);
        while let Some(u) = queue.pop_front() {
            let du = dist[u];
            for &v in &adj[u] {
                if dist[v] == usize::MAX {
                    dist[v] = du + 1;
                    queue.push_back(v);
                }
            }
        }

        let scale = (cn as f64).sqrt();
        for (i, &u) in comp.iter().enumerate() {
            let d = dist[u] as f64;
            let angle = d * 2.399 + (i as f64) * 0.618;
            let r = d * scale * 0.5 + 0.1;
            pos[u] = (offset_x + r * angle.cos(), r * angle.sin());
        }
        offset_x += scale * 4.0 + 5.0;
    }

    pos
}

fn force_directed(
    pos: &mut [(f64, f64)],
    edges: &[(usize, usize)],
    _adj: &[Vec<usize>],
    n: usize,
    iterations: usize,
    ideal_len: f64,
) {
    let k = ideal_len;
    let k2 = k * k;
    let mut temp = k * (n as f64).sqrt() * 0.1;
    let cooling = temp / (iterations as f64 + 1.0);

    let mut disp = vec![(0.0f64, 0.0f64); n];

    for _ in 0..iterations {
        for i in 0..n {
            disp[i] = (0.0, 0.0);
        }
        for i in 0..n {
            for j in (i + 1)..n {
                let dx = pos[i].0 - pos[j].0;
                let dy = pos[i].1 - pos[j].1;
                let dist2 = dx * dx + dy * dy;
                let dist = dist2.sqrt().max(0.01);
                let force = k2 / dist;
                let fx = dx / dist * force;
                let fy = dy / dist * force;
                disp[i].0 += fx;
                disp[i].1 += fy;
                disp[j].0 -= fx;
                disp[j].1 -= fy;
            }
        }

        for &(u, v) in edges {
            let dx = pos[u].0 - pos[v].0;
            let dy = pos[u].1 - pos[v].1;
            let dist = (dx * dx + dy * dy).sqrt().max(0.01);
            let force = dist / k;
            let fx = dx / dist * force;
            let fy = dy / dist * force;
            disp[u].0 -= fx;
            disp[u].1 -= fy;
            disp[v].0 += fx;
            disp[v].1 += fy;
        }

        for i in 0..n {
            let dx = disp[i].0;
            let dy = disp[i].1;
            let len = (dx * dx + dy * dy).sqrt().max(0.001);
            let scale = temp.min(len) / len;
            pos[i].0 += dx * scale;
            pos[i].1 += dy * scale;
        }

        temp -= cooling;
        if temp < 0.01 {
            break;
        }
    }
}

fn force_directed_large(
    pos: &mut [(f64, f64)],
    edges: &[(usize, usize)],
    _adj: &[Vec<usize>],
    n: usize,
    iterations: usize,
    ideal_len: f64,
) {
    let k = ideal_len;
    let k2 = k * k;
    let mut temp = k * (n as f64).sqrt() * 0.05;
    let cooling = temp / (iterations as f64 + 1.0);
    let cutoff = k * 4.0;
    let cutoff2 = cutoff * cutoff;

    let mut disp = vec![(0.0f64, 0.0f64); n];

    for _iter in 0..iterations {
        for i in 0..n {
            disp[i] = (0.0, 0.0);
        }

        let cell_size = cutoff;
        let mut min_x = f64::MAX;
        let mut min_y = f64::MAX;
        let mut max_x = f64::MIN;
        let mut max_y = f64::MIN;
        for i in 0..n {
            min_x = min_x.min(pos[i].0);
            min_y = min_y.min(pos[i].1);
            max_x = max_x.max(pos[i].0);
            max_y = max_y.max(pos[i].1);
        }
        min_x -= 1.0;
        min_y -= 1.0;
        max_x += 1.0;
        max_y += 1.0;

        let cols = ((max_x - min_x) / cell_size).ceil() as usize + 1;
        let rows = ((max_y - min_y) / cell_size).ceil() as usize + 1;
        let grid_size = cols.saturating_mul(rows);

        if grid_size < 4_000_000 {
            let mut grid: Vec<Vec<usize>> = vec![vec![]; grid_size];

            let mut cell_of = vec![0usize; n];
            for i in 0..n {
                let cx = ((pos[i].0 - min_x) / cell_size) as usize;
                let cy = ((pos[i].1 - min_y) / cell_size) as usize;
                let cell = cy * cols + cx;
                if cell < grid_size {
                    cell_of[i] = cell;
                    grid[cell].push(i);
                }
            }

            for i in 0..n {
                let cx = ((pos[i].0 - min_x) / cell_size) as usize;
                let cy = ((pos[i].1 - min_y) / cell_size) as usize;

                let cx_min = cx.saturating_sub(1);
                let cx_max = (cx + 1).min(cols - 1);
                let cy_min = cy.saturating_sub(1);
                let cy_max = (cy + 1).min(rows - 1);

                for gy in cy_min..=cy_max {
                    for gx in cx_min..=cx_max {
                        let cell = gy * cols + gx;
                        if cell >= grid_size {
                            continue;
                        }
                        for &j in &grid[cell] {
                            if j <= i {
                                continue;
                            }
                            let dx = pos[i].0 - pos[j].0;
                            let dy = pos[i].1 - pos[j].1;
                            let dist2 = dx * dx + dy * dy;
                            if dist2 > cutoff2 {
                                continue;
                            }
                            let dist = dist2.sqrt().max(0.01);
                            let force = k2 / dist;
                            let fx = dx / dist * force;
                            let fy = dy / dist * force;
                            disp[i].0 += fx;
                            disp[i].1 += fy;
                            disp[j].0 -= fx;
                            disp[j].1 -= fy;
                        }
                    }
                }
            }
        } else {
            let sample_size = 200.min(n);
            let step = if sample_size > 0 { n / sample_size } else { 1 };
            for i in 0..n {
                for s in 0..sample_size {
                    let j = (s * step + (i * 7 + 13) % step.max(1)) % n;
                    if j == i { continue; }
                    let dx = pos[i].0 - pos[j].0;
                    let dy = pos[i].1 - pos[j].1;
                    let dist2 = dx * dx + dy * dy;
                    if dist2 > cutoff2 * 4.0 { continue; }
                    let dist = dist2.sqrt().max(0.01);
                    let force = k2 / dist * (n as f64 / sample_size as f64);
                    let fx = dx / dist * force;
                    let fy = dy / dist * force;
                    disp[i].0 += fx;
                    disp[i].1 += fy;
                }
            }
        }

        for &(u, v) in edges {
            let dx = pos[u].0 - pos[v].0;
            let dy = pos[u].1 - pos[v].1;
            let dist = (dx * dx + dy * dy).sqrt().max(0.01);
            let force = dist / k;
            let fx = dx / dist * force;
            let fy = dy / dist * force;
            disp[u].0 -= fx;
            disp[u].1 -= fy;
            disp[v].0 += fx;
            disp[v].1 += fy;
        }

        let mut cx = 0.0f64;
        let mut cy = 0.0f64;
        for i in 0..n {
            cx += pos[i].0;
            cy += pos[i].1;
        }
        cx /= n as f64;
        cy /= n as f64;
        let gravity = 0.1;
        for i in 0..n {
            disp[i].0 -= (pos[i].0 - cx) * gravity;
            disp[i].1 -= (pos[i].1 - cy) * gravity;
        }

        for i in 0..n {
            let dx = disp[i].0;
            let dy = disp[i].1;
            let len = (dx * dx + dy * dy).sqrt().max(0.001);
            let scale = temp.min(len) / len;
            pos[i].0 += dx * scale;
            pos[i].1 += dy * scale;
        }

        temp -= cooling;
        if temp < 0.01 {
            break;
        }
    }
}
