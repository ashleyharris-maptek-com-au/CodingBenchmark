#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
#include <chrono>
#include <climits>

struct Rect {
    int w, h, idx;
};

struct Placement {
    int board, x, y, rotated;
};

struct FreeRect {
    int x, y, w, h;
};

struct MaxRectsBin {
    int W, H;
    std::vector<FreeRect> freeRects;

    MaxRectsBin(int W, int H) : W(W), H(H) {
        freeRects.push_back({0, 0, W, H});
    }

    bool insert(int w, int h, int& outX, int& outY, int& outRot, int heuristic) {
        int bestScore1 = INT_MAX, bestScore2 = INT_MAX;
        int bestX = 0, bestY = 0, bestRot = 0;
        bool found = false;

        for (int rot = 0; rot < 2; rot++) {
            int rw = rot ? h : w;
            int rh = rot ? w : h;
            if (rw > W || rh > H) continue;

            for (int i = 0; i < (int)freeRects.size(); i++) {
                auto& f = freeRects[i];
                if (rw <= f.w && rh <= f.h) {
                    int s1, s2;
                    switch (heuristic) {
                    case 0:
                        s1 = std::min(f.w - rw, f.h - rh);
                        s2 = std::max(f.w - rw, f.h - rh);
                        break;
                    case 1:
                        s1 = std::max(f.w - rw, f.h - rh);
                        s2 = std::min(f.w - rw, f.h - rh);
                        break;
                    case 2:
                        s1 = f.w * f.h - rw * rh;
                        s2 = std::min(f.w - rw, f.h - rh);
                        break;
                    case 3:
                        s1 = f.y + rh;
                        s2 = f.x;
                        break;
                    default:
                        s1 = std::min(f.w - rw, f.h - rh);
                        s2 = std::max(f.w - rw, f.h - rh);
                    }
                    if (s1 < bestScore1 || (s1 == bestScore1 && s2 < bestScore2)) {
                        bestScore1 = s1;
                        bestScore2 = s2;
                        bestX = f.x;
                        bestY = f.y;
                        bestRot = rot;
                        found = true;
                    }
                }
            }
        }

        if (!found) return false;

        outX = bestX;
        outY = bestY;
        outRot = bestRot;

        int rw = bestRot ? h : w;
        int rh = bestRot ? w : h;

        splitFreeRects(bestX, bestY, rw, rh);
        pruneFreeRects();
        return true;
    }

    void splitFreeRects(int px, int py, int pw, int ph) {
        int sz = (int)freeRects.size();
        for (int i = sz - 1; i >= 0; i--) {
            auto& f = freeRects[i];
            if (px >= f.x + f.w || px + pw <= f.x || py >= f.y + f.h || py + ph <= f.y)
                continue;

            FreeRect orig = f;
            freeRects[i] = freeRects.back();
            freeRects.pop_back();

            if (px > orig.x)
                freeRects.push_back({orig.x, orig.y, px - orig.x, orig.h});
            if (px + pw < orig.x + orig.w)
                freeRects.push_back({px + pw, orig.y, orig.x + orig.w - px - pw, orig.h});
            if (py > orig.y)
                freeRects.push_back({orig.x, orig.y, orig.w, py - orig.y});
            if (py + ph < orig.y + orig.h)
                freeRects.push_back({orig.x, py + ph, orig.w, orig.y + orig.h - py - ph});
        }
    }

    void pruneFreeRects() {
        for (int i = 0; i < (int)freeRects.size(); i++) {
            for (int j = i + 1; j < (int)freeRects.size(); ) {
                if (contains(freeRects[j], freeRects[i])) {
                    freeRects[i] = freeRects.back();
                    freeRects.pop_back();
                    if (i < (int)freeRects.size()) {
                        j = i + 1;
                    }
                    else break;
                    continue;
                }
                if (contains(freeRects[i], freeRects[j])) {
                    freeRects[j] = freeRects.back();
                    freeRects.pop_back();
                    continue;
                }
                j++;
            }
        }
    }

    static bool contains(const FreeRect& a, const FreeRect& b) {
        return a.x <= b.x && a.y <= b.y &&
               a.x + a.w >= b.x + b.w && a.y + a.h >= b.y + b.h;
    }
};

int solveWithHeuristic(int N, int W, int H, const std::vector<Rect>& order, int heuristic, std::vector<Placement>& placements) {
    std::vector<MaxRectsBin> bins;
    placements.resize(N);

    for (auto& r : order) {
        bool placed = false;
        for (int b = 0; b < (int)bins.size(); b++) {
            int ox, oy, orot;
            if (bins[b].insert(r.w, r.h, ox, oy, orot, heuristic)) {
                placements[r.idx] = {b, ox, oy, orot};
                placed = true;
                break;
            }
        }
        if (!placed) {
            bins.emplace_back(W, H);
            int ox, oy, orot;
            if (bins.back().insert(r.w, r.h, ox, oy, orot, heuristic)) {
                placements[r.idx] = {(int)bins.size() - 1, ox, oy, orot};
            }
        }
    }
    return (int)bins.size();
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(nullptr);

    auto startTime = std::chrono::steady_clock::now();

    int N, W, H;
    std::cin >> N >> W >> H;

    std::vector<Rect> rects(N);
    for (int i = 0; i < N; i++) {
        std::cin >> rects[i].w >> rects[i].h;
        rects[i].idx = i;
    }

    auto sortByArea = [](const Rect& a, const Rect& b) {
        long long aa = (long long)a.w * a.h, ab = (long long)b.w * b.h;
        if (aa != ab) return aa > ab;
        return std::max(a.w, a.h) > std::max(b.w, b.h);
    };

    auto sortByHeight = [](const Rect& a, const Rect& b) {
        int ha = std::max(a.w, a.h), hb = std::max(b.w, b.h);
        if (ha != hb) return ha > hb;
        return (long long)a.w * a.h > (long long)b.w * b.h;
    };

    auto sortByPerimeter = [](const Rect& a, const Rect& b) {
        int pa = a.w + a.h, pb = b.w + b.h;
        if (pa != pb) return pa > pb;
        return (long long)a.w * a.h > (long long)b.w * b.h;
    };

    auto sortByWidth = [](const Rect& a, const Rect& b) {
        int wa = std::min(a.w, a.h), wb = std::min(b.w, b.h);
        if (wa != wb) return wa > wb;
        return (long long)a.w * a.h > (long long)b.w * b.h;
    };

    int bestBoards = INT_MAX;
    std::vector<Placement> bestPlacements;

    int numHeuristics = 4;
    int numSorts = 4;
    double timeLimit = 25.0;

    if (N > 10000) {
        numHeuristics = 2;
        numSorts = 2;
    } else if (N > 5000) {
        numHeuristics = 3;
        numSorts = 3;
    }

    for (int s = 0; s < numSorts; s++) {
        auto now = std::chrono::steady_clock::now();
        double elapsed = std::chrono::duration<double>(now - startTime).count();
        if (elapsed > timeLimit) break;

        std::vector<Rect> sorted = rects;
        switch (s) {
        case 0: std::sort(sorted.begin(), sorted.end(), sortByArea); break;
        case 1: std::sort(sorted.begin(), sorted.end(), sortByHeight); break;
        case 2: std::sort(sorted.begin(), sorted.end(), sortByPerimeter); break;
        case 3: std::sort(sorted.begin(), sorted.end(), sortByWidth); break;
        }

        for (int h = 0; h < numHeuristics; h++) {
            now = std::chrono::steady_clock::now();
            elapsed = std::chrono::duration<double>(now - startTime).count();
            if (elapsed > timeLimit) break;

            std::vector<Placement> pl;
            int boards = solveWithHeuristic(N, W, H, sorted, h, pl);
            if (boards < bestBoards) {
                bestBoards = boards;
                bestPlacements = std::move(pl);
            }
        }
    }

    std::cout << bestBoards << "\n";
    for (int i = 0; i < N; i++) {
        std::cout << bestPlacements[i].board << " "
                  << bestPlacements[i].x << " "
                  << bestPlacements[i].y << " "
                  << bestPlacements[i].rotated << "\n";
    }

    return 0;
}
