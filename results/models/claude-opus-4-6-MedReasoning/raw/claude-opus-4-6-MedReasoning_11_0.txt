use std::io::{self, Read, Write, BufWriter};
use std::time::Instant;

type P = (f64, f64);

fn poly_area(p: &[P]) -> f64 {
    let n = p.len();
    let mut a = 0.0;
    for i in 0..n {
        let j = (i + 1) % n;
        a += p[i].0 * p[j].1 - p[j].0 * p[i].1;
    }
    a.abs() * 0.5
}

fn bb(p: &[P]) -> [f64; 4] {
    let mut r = [f64::MAX, f64::MAX, f64::MIN, f64::MIN];
    for &(x, y) in p {
        r[0] = r[0].min(x);
        r[1] = r[1].min(y);
        r[2] = r[2].max(x);
        r[3] = r[3].max(y);
    }
    r
}

fn rot_pts(p: &[P], a: f64) -> Vec<P> {
    let (c, s) = (a.cos(), a.sin());
    p.iter().map(|&(x, y)| (x * c - y * s, x * s + y * c)).collect()
}

fn pip(px: f64, py: f64, poly: &[P]) -> bool {
    let n = poly.len();
    let mut ins = false;
    let mut j = n - 1;
    for i in 0..n {
        let (xi, yi) = poly[i];
        let (xj, yj) = poly[j];
        if ((yi > py) != (yj > py)) && (px < (xj - xi) * (py - yi) / (yj - yi) + xi) {
            ins = !ins;
        }
        j = i;
    }
    ins
}

fn seg_x(a1: P, a2: P, b1: P, b2: P) -> bool {
    let cr = |o: P, a: P, b: P| (a.0 - o.0) * (b.1 - o.1) - (a.1 - o.1) * (b.0 - o.0);
    let d1 = cr(b1, b2, a1);
    let d2 = cr(b1, b2, a2);
    let d3 = cr(a1, a2, b1);
    let d4 = cr(a1, a2, b2);
    ((d1 > 0.0 && d2 < 0.0) || (d1 < 0.0 && d2 > 0.0))
        && ((d3 > 0.0 && d4 < 0.0) || (d3 < 0.0 && d4 > 0.0))
}

fn is_convex(p: &[P]) -> bool {
    let n = p.len();
    if n < 3 {
        return true;
    }
    let mut s = 0i32;
    for i in 0..n {
        let c = (p[(i + 1) % n].0 - p[i].0) * (p[(i + 2) % n].1 - p[(i + 1) % n].1)
            - (p[(i + 1) % n].1 - p[i].1) * (p[(i + 2) % n].0 - p[(i + 1) % n].0);
        if c > 1e-10 {
            if s < 0 {
                return false;
            }
            s = 1;
        } else if c < -1e-10 {
            if s > 0 {
                return false;
            }
            s = -1;
        }
    }
    true
}

fn fits(piece: &[P], stock: &[P], cv: bool) -> bool {
    if cv {
        let b = bb(piece);
        if pip(b[0], b[1], stock)
            && pip(b[2], b[1], stock)
            && pip(b[0], b[3], stock)
            && pip(b[2], b[3], stock)
        {
            return true;
        }
    }
    for &(x, y) in piece {
        if !pip(x, y, stock) {
            return false;
        }
    }
    if !cv {
        let (np, ns) = (piece.len(), stock.len());
        for i in 0..np {
            for j in 0..ns {
                if seg_x(
                    piece[i],
                    piece[(i + 1) % np],
                    stock[j],
                    stock[(j + 1) % ns],
                ) {
                    return false;
                }
            }
        }
    }
    true
}

fn main() {
    let t0 = Instant::now();
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let out = io::stdout();
    let mut out = BufWriter::new(out.lock());

    let mut it = input.split_ascii_whitespace();
    let sv: usize = it.next().unwrap().parse().unwrap();
    let np: usize = it.next().unwrap().parse().unwrap();
    let stock: Vec<P> = (0..sv)
        .map(|_| {
            (
                it.next().unwrap().parse::<f64>().unwrap(),
                it.next().unwrap().parse::<f64>().unwrap(),
            )
        })
        .collect();
    let pieces: Vec<Vec<P>> = (0..np)
        .map(|_| {
            let pv: usize = it.next().unwrap().parse().unwrap();
            (0..pv)
                .map(|_| {
                    (
                        it.next().unwrap().parse::<f64>().unwrap(),
                        it.next().unwrap().parse::<f64>().unwrap(),
                    )
                })
                .collect()
        })
        .collect();

    if np == 0 {
        writeln!(out, "0").unwrap();
        return;
    }

    let sb = bb(&stock);
    let sw = sb[2] - sb[0];
    let sh = sb[3] - sb[1];

    if sw < 1e-12 || sh < 1e-12 {
        writeln!(out, "{}", np).unwrap();
        for i in 0..np {
            writeln!(out, "{} 0.0 0.0 0.0", i).unwrap();
        }
        return;
    }

    let cv = is_convex(&stock);
    let nr = if np <= 10 {
        72
    } else if np <= 50 {
        36
    } else if np <= 200 {
        24
    } else if np <= 1000 {
        12
    } else if np <= 3000 {
        6
    } else {
        4
    };
    let rots: Vec<f64> = (0..nr)
        .map(|i| std::f64::consts::TAU * i as f64 / nr as f64)
        .collect();
    let gres: usize = if np <= 200 {
        100
    } else if np <= 2000 {
        80
    } else {
        60
    };
    let cw = sw / gres as f64;
    let ch = sh / gres as f64;
    let eps = sw.max(sh) * 1e-7;

    struct PD {
        idx: usize,
        cx: f64,
        cy: f64,
        centered: Vec<P>,
        area: f64,
    }
    let mut pd: Vec<PD> = pieces
        .iter()
        .enumerate()
        .map(|(i, v)| {
            let n = v.len() as f64;
            let cx = v.iter().map(|p| p.0).sum::<f64>() / n;
            let cy = v.iter().map(|p| p.1).sum::<f64>() / n;
            PD {
                idx: i,
                cx,
                cy,
                centered: v.iter().map(|&(x, y)| (x - cx, y - cy)).collect(),
                area: poly_area(v),
            }
        })
        .collect();
    pd.sort_by(|a, b| b.area.partial_cmp(&a.area).unwrap());

    let gsz = gres * gres;
    let mut grids: Vec<Vec<bool>> = Vec::new();
    let mut results = vec![(0usize, 0.0f64, 0.0f64, 0.0f64); np];
    let max_try: usize = if np <= 200 {
        30
    } else if np <= 1000 {
        10
    } else {
        5
    };

    for p in &pd {
        if t0.elapsed().as_secs() > 240 {
            let si = grids.len();
            grids.push(vec![false; gsz]);
            let scx = (sb[0] + sb[2]) * 0.5;
            let scy = (sb[1] + sb[3]) * 0.5;
            results[p.idx] = (si, scx - p.cx, scy - p.cy, 0.0);
            continue;
        }

        let mut rds: Vec<(f64, Vec<P>, [f64; 4], usize, usize)> = Vec::new();
        for &r in &rots {
            let v = rot_pts(&p.centered, r);
            let b = bb(&v);
            let pw = b[2] - b[0];
            let ph = b[3] - b[1];
            let gw = ((pw + 2.0 * eps) / cw).ceil() as usize;
            let gh = ((ph + 2.0 * eps) / ch).ceil() as usize;
            if gw <= gres && gh <= gres {
                rds.push((r, v, b, gw, gh));
            }
        }
        rds.sort_by_key(|r| r.3 * r.4);

        if rds.is_empty() {
            let si = grids.len();
            grids.push(vec![false; gsz]);
            results[p.idx] = (si, 0.0, 0.0, 0.0);
            continue;
        }

        let mut placed = false;
        let ns = grids.len();
        let nt = ns.min(max_try);

        'search: for off in 0..nt {
            let si = ns - 1 - off;
            for &(r, ref verts, ref rb, gw, gh) in &rds {
                let max_gy = gres - gh;
                let max_gx = gres - gw;
                for gy in 0..=max_gy {
                    for gx in 0..=max_gx {
                        let mut free = true;
                        'chk: for dy in 0..gh {
                            let row = (gy + dy) * gres;
                            for dx in 0..gw {
                                if grids[si][row + gx + dx] {
                                    free = false;
                                    break 'chk;
                                }
                            }
                        }
                        if !free {
                            continue;
                        }
                        let tx = sb[0] + gx as f64 * cw + eps - rb[0];
                        let ty = sb[1] + gy as f64 * ch + eps - rb[1];
                        let pv: Vec<P> =
                            verts.iter().map(|&(x, y)| (x + tx, y + ty)).collect();
                        if fits(&pv, &stock, cv) {
                            for dy in 0..gh {
                                let row = (gy + dy) * gres;
                                for dx in 0..gw {
                                    grids[si][row + gx + dx] = true;
                                }
                            }
                            let (c, s) = (r.cos(), r.sin());
                            results[p.idx] = (
                                si,
                                tx - (p.cx * c - p.cy * s),
                                ty - (p.cx * s + p.cy * c),
                                r,
                            );
                            placed = true;
                            break 'search;
                        }
                    }
                }
            }
        }

        if !placed {
            let si = grids.len();
            grids.push(vec![false; gsz]);
            'newstock: for &(r, ref verts, ref rb, gw, gh) in &rds {
                let max_gy = gres - gh;
                let max_gx = gres - gw;
                for gy in 0..=max_gy {
                    for gx in 0..=max_gx {
                        let mut free = true;
                        'chk2: for dy in 0..gh {
                            let row = (gy + dy) * gres;
                            for dx in 0..gw {
                                if grids[si][row + gx + dx] {
                                    free = false;
                                    break 'chk2;
                                }
                            }
                        }
                        if !free {
                            continue;
                        }
                        let tx = sb[0] + gx as f64 * cw + eps - rb[0];
                        let ty = sb[1] + gy as f64 * ch + eps - rb[1];
                        let pv: Vec<P> =
                            verts.iter().map(|&(x, y)| (x + tx, y + ty)).collect();
                        if fits(&pv, &stock, cv) {
                            for dy in 0..gh {
                                let row = (gy + dy) * gres;
                                for dx in 0..gw {
                                    grids[si][row + gx + dx] = true;
                                }
                            }
                            let (c, s) = (r.cos(), r.sin());
                            results[p.idx] = (
                                si,
                                tx - (p.cx * c - p.cy * s),
                                ty - (p.cx * s + p.cy * c),
                                r,
                            );
                            placed = true;
                            break 'newstock;
                        }
                    }
                }
            }
            if !placed {
                results[p.idx] = (si, 0.0, 0.0, 0.0);
            }
        }
    }

    let num_stocks = grids.len().max(1);
    writeln!(out, "{}", num_stocks).unwrap();
    for &(si, tx, ty, r) in &results {
        writeln!(out, "{} {:.10} {:.10} {:.10}", si, tx, ty, r).unwrap();
    }
}
