using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

class Program
{
    struct OE
    {
        public double a, e, inc, raan, argp;
        public double rMag, vMag;
    }

    static double Dot(double[] a, double[] b) => a[0]*b[0]+a[1]*b[1]+a[2]*b[2];
    static double Mag(double[] a) => Math.Sqrt(Dot(a,a));
    static double[] Cross(double[] a, double[] b) => new double[]{
        a[1]*b[2]-a[2]*b[1], a[2]*b[0]-a[0]*b[2], a[0]*b[1]-a[1]*b[0]};

    static double Clamp(double v, double lo, double hi) => v<lo?lo:v>hi?hi:v;

    static OE ToOE(double[] s, double mu)
    {
        double[] r={s[0],s[1],s[2]}, v={s[3],s[4],s[5]};
        double rm=Mag(r), vm=Mag(v);
        double[] h=Cross(r,v);
        double hm=Mag(h);
        double rdv=Dot(r,v);
        double[] ev=new double[3];
        for(int j=0;j<3;j++) ev[j]=((vm*vm-mu/rm)*r[j]-rdv*v[j])/mu;
        double em=Mag(ev);
        double energy=vm*vm/2-mu/rm;
        double a=(Math.Abs(em-1)<1e-10)?1e12:-mu/(2*energy);
        double inc=Math.Acos(Clamp(h[2]/hm,-1,1));
        double[] n=Cross(new double[]{0,0,1},h);
        double nm=Mag(n);
        double raan=0;
        if(nm>1e-10){raan=Math.Acos(Clamp(n[0]/nm,-1,1));if(n[1]<0)raan=2*Math.PI-raan;}
        double argp=0;
        if(nm>1e-10&&em>1e-10){argp=Math.Acos(Clamp(Dot(n,ev)/(nm*em),-1,1));if(ev[2]<0)argp=2*Math.PI-argp;}
        return new OE{a=a,e=em,inc=inc,raan=raan,argp=argp,rMag=rm,vMag=vm};
    }

    static double TransferCost(OE f, OE t, double mu)
    {
        double a1=Math.Abs(f.a), a2=Math.Abs(t.a);
        if(a1<100||a2<100||double.IsInfinity(a1)||double.IsInfinity(a2))
            return Math.Abs(f.vMag-t.vMag)+0.5;
        double v1c=Math.Sqrt(mu/a1), v2c=Math.Sqrt(mu/a2);
        double hdv=0;
        if(Math.Abs(a1-a2)>1)
        {
            double at=(a1+a2)/2;
            double vd=Math.Sqrt(Math.Max(0,mu*(2/a1-1/at)));
            double va=Math.Sqrt(Math.Max(0,mu*(2/a2-1/at)));
            hdv=Math.Abs(vd-v1c)+Math.Abs(v2c-va);
        }
        double cosdt=Math.Cos(f.inc)*Math.Cos(t.inc)+Math.Sin(f.inc)*Math.Sin(t.inc)*Math.Cos(t.raan-f.raan);
        double dt=Math.Acos(Clamp(cosdt,-1,1));
        double vp=Math.Min(v1c,v2c);
        double pdv=2*vp*Math.Sin(dt/2);
        double edv=Math.Abs(f.e-t.e)*Math.Max(v1c,v2c)*0.5;
        double dadv=Math.Abs(f.argp-t.argp);
        if(dadv>Math.PI) dadv=2*Math.PI-dadv;
        double adv=dadv/(2*Math.PI)*Math.Min(v1c,v2c)*Math.Min(f.e+t.e,1)*0.3;
        return Math.Sqrt(hdv*hdv+pdv*pdv)+edv+adv;
    }

    static double TourCost(double[,] c, int[] tour)
    {
        double s=c[0,tour[0]];
        for(int i=0;i<tour.Length-1;i++) s+=c[tour[i],tour[i+1]];
        return s;
    }

    static int[] BruteForce(double[,] c, int N)
    {
        int[] best=null; double bestC=double.MaxValue;
        int[] p=Enumerable.Range(1,N).ToArray();
        Permute(p,0,c,ref best,ref bestC);
        return best;
    }

    static void Permute(int[] p,int k,double[,] c,ref int[] best,ref double bestC)
    {
        if(k==p.Length)
        {
            double cost=TourCost(c,p);
            if(cost<bestC){bestC=cost;best=(int[])p.Clone();}
            return;
        }
        for(int i=k;i<p.Length;i++)
        {
            int t=p[k];p[k]=p[i];p[i]=t;
            double partial=c[0,p[0]];
            for(int j=0;j<k;j++) partial+=c[p[j],p[j+1]];
            if(partial<bestC) Permute(p,k+1,c,ref best,ref bestC);
            t=p[k];p[k]=p[i];p[i]=t;
        }
    }

    static int[] NNTour(double[,] c, int N, int firstNode)
    {
        bool[] vis=new bool[N+1]; vis[0]=true;
        int[] tour=new int[N];
        int cur;
        if(firstNode>0)
        {
            tour[0]=firstNode; vis[firstNode]=true; cur=firstNode;
            for(int step=1;step<N;step++)
            {
                double bd=double.MaxValue; int bj=-1;
                for(int j=1;j<=N;j++) if(!vis[j]&&c[cur,j]<bd){bd=c[cur,j];bj=j;}
                tour[step]=bj; vis[bj]=true; cur=bj;
            }
        }
        else
        {
            cur=0;
            for(int step=0;step<N;step++)
            {
                double bd=double.MaxValue; int bj=-1;
                for(int j=1;j<=N;j++) if(!vis[j]&&c[cur,j]<bd){bd=c[cur,j];bj=j;}
                tour[step]=bj; vis[bj]=true; cur=bj;
            }
        }
        return tour;
    }

    static void TwoOpt(double[,] c, int[] tour)
    {
        int N=tour.Length;
        bool improved=true;
        while(improved)
        {
            improved=false;
            for(int i=0;i<N-1;i++)
            {
                for(int j=i+1;j<N;j++)
                {
                    int pi=(i==0)?0:tour[i-1];
                    double oldC=c[pi,tour[i]]+(j<N-1?c[tour[j],tour[j+1]]:0);
                    double newC=c[pi,tour[j]]+(j<N-1?c[tour[i],tour[j+1]]:0);
                    if(newC<oldC-1e-12)
                    {
                        Array.Reverse(tour,i,j-i+1);
                        improved=true;
                    }
                }
            }
        }
    }

    static void OrOpt(double[,] c, int[] tourArr, ref int[] bestTour, ref double bestCost)
    {
        int N=tourArr.Length;
        var tour=new List<int>(tourArr);
        bool imp=true;
        while(imp)
        {
            imp=false;
            for(int segLen=1;segLen<=3;segLen++)
            {
                for(int i=0;i<tour.Count-segLen+1;i++)
                {
                    int prev_i=(i==0)?0:tour[i-1];
                    int after_i=(i+segLen<tour.Count)?tour[i+segLen]:-1;
                    double removeSave=c[prev_i,tour[i]];
                    if(after_i>=0) removeSave+=c[tour[i+segLen-1],after_i];
                    double removeNew=(after_i>=0)?c[prev_i,after_i]:0;
                    double removeDelta=removeSave-removeNew;

                    for(int j=0;j<tour.Count;j++)
                    {
                        if(j>=i-1&&j<=i+segLen) continue;
                        int pj=(j==0)?0:tour[j-1];
                        double insertOld=c[pj,tour[j]];
                        double insertNew=c[pj,tour[i]]+c[tour[i+segLen-1],tour[j]];
                        double insertDelta=insertNew-insertOld;

                        if(insertDelta<removeDelta-1e-12)
                        {
                            var seg=tour.GetRange(i,segLen);
                            tour.RemoveRange(i,segLen);
                            int insertPos=(j<i)?j:j-segLen;
                            if(insertPos<0) insertPos=0;
                            if(insertPos>tour.Count) insertPos=tour.Count;
                            tour.InsertRange(insertPos,seg);
                            imp=true;
                            break;
                        }
                    }
                    if(imp) break;
                }
                if(imp) break;
            }
        }
        double tc=TourCost(c,tour.ToArray());
        if(tc<bestCost){bestCost=tc;bestTour=tour.ToArray();}
    }

    static int[] SolveTSP(double[,] c, int N)
    {
        if(N<=10) return BruteForce(c,N);

        int[] bestTour=null; double bestCost=double.MaxValue;
        int tries=Math.Min(N,30);

        for(int s=0;s<tries;s++)
        {
            int first=(s==0)?0:(s);
            if(first>N) break;
            int[] tour=NNTour(c,N,first);
            TwoOpt(c,tour);
            double tc=TourCost(c,tour);
            if(tc<bestCost){bestCost=tc;bestTour=(int[])tour.Clone();}
        }

        OrOpt(c,bestTour,ref bestTour,ref bestCost);
        TwoOpt(c,bestTour);
        double fc=TourCost(c,bestTour);
        if(fc<bestCost){bestCost=fc;}

        return bestTour;
    }

    static void Main(string[] args)
    {
        CultureInfo.DefaultThreadCurrentCulture=CultureInfo.InvariantCulture;
        string[] fl=Console.ReadLine().Trim().Split(new[]{' ','\t'},StringSplitOptions.RemoveEmptyEntries);
        int N=int.Parse(fl[0]);
        double mu=double.Parse(fl[1],CultureInfo.InvariantCulture);

        double[] startState=Console.ReadLine().Trim().Split(new[]{' ','\t'},StringSplitOptions.RemoveEmptyEntries)
            .Select(p=>double.Parse(p,CultureInfo.InvariantCulture)).ToArray();
        double[][] stations=new double[N][];
        for(int i=0;i<N;i++)
            stations[i]=Console.ReadLine().Trim().Split(new[]{' ','\t'},StringSplitOptions.RemoveEmptyEntries)
                .Select(p=>double.Parse(p,CultureInfo.InvariantCulture)).ToArray();

        OE startOE=ToOE(startState,mu);
        OE[] sOE=new OE[N];
        for(int i=0;i<N;i++) sOE[i]=ToOE(stations[i],mu);

        double[,] cost=new double[N+1,N+1];
        for(int i=0;i<=N;i++)
        {
            OE oi=(i==0)?startOE:sOE[i-1];
            for(int j=i+1;j<=N;j++)
            {
                OE oj=(j==0)?startOE:sOE[j-1];
                double d=TransferCost(oi,oj,mu);
                cost[i,j]=d; cost[j,i]=d;
            }
        }

        int[] order=SolveTSP(cost,N);
        Console.WriteLine(string.Join(" ",order.Select(x=>(x-1).ToString())));
    }
}
