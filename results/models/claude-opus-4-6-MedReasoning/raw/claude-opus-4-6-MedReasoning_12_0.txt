using System;
using System.Globalization;
using System.IO;

class Program
{
    static int V;
    static double[] px, py, pz;

    static void AABB(double qw, double qx, double qy, double qz,
        out double mnX, out double mnY, out double mnZ,
        out double dx, out double dy, out double dz)
    {
        double minX = double.MaxValue, minY = double.MaxValue, minZ = double.MaxValue;
        double maxX = double.MinValue, maxY = double.MinValue, maxZ = double.MinValue;
        for (int i = 0; i < V; i++)
        {
            double x = px[i], y = py[i], z = pz[i];
            double tx = 2 * (qy * z - qz * y);
            double ty = 2 * (qz * x - qx * z);
            double tz = 2 * (qx * y - qy * x);
            double rx = x + qw * tx + qy * tz - qz * ty;
            double ry = y + qw * ty + qz * tx - qx * tz;
            double rz = z + qw * tz + qx * ty - qy * tx;
            if (rx < minX) minX = rx; if (rx > maxX) maxX = rx;
            if (ry < minY) minY = ry; if (ry > maxY) maxY = ry;
            if (rz < minZ) minZ = rz; if (rz > maxZ) maxZ = rz;
        }
        mnX = minX; mnY = minY; mnZ = minZ;
        dx = maxX - minX; dy = maxY - minY; dz = maxZ - minZ;
    }

    static long GridN(double dim, double piece)
    {
        if (piece < 1e-12 || piece > dim + 1e-9) return 0;
        long n = (long)(dim / piece + 1e-9);
        if (n * piece > dim + 1e-9) n--;
        return Math.Max(0, n);
    }

    static long Grid(double dx, double dy, double dz, double cW, double cH, double cD)
    {
        long nx = GridN(cW, dx), ny = GridN(cH, dy), nz = GridN(cD, dz);
        return nx * ny * nz;
    }

    static void Euler(double a, double b, double c,
        out double w, out double x, out double y, out double z)
    {
        double c1 = Math.Cos(a * 0.5), s1 = Math.Sin(a * 0.5);
        double c2 = Math.Cos(b * 0.5), s2 = Math.Sin(b * 0.5);
        double c3 = Math.Cos(c * 0.5), s3 = Math.Sin(c * 0.5);
        w = c3 * c2 * c1 + s3 * s2 * s1;
        x = s3 * c2 * c1 - c3 * s2 * s1;
        y = c3 * s2 * c1 + s3 * c2 * s1;
        z = c3 * c2 * s1 - s3 * s2 * c1;
    }

    static void QMul(double aw, double ax, double ay, double az,
                     double bw, double bx, double by, double bz,
                     out double w, out double x, out double y, out double z)
    {
        w = aw * bw - ax * bx - ay * by - az * bz;
        x = aw * bx + ax * bw + ay * bz - az * by;
        y = aw * by - ax * bz + ay * bw + az * bx;
        z = aw * bz + ax * by - ay * bx + az * bw;
    }

    static string[] Tok(string s) => s.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);

    static void Main()
    {
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;

        var t = Tok(Console.ReadLine());
        V = int.Parse(t[0]); int F = int.Parse(t[1]);
        px = new double[V]; py = new double[V]; pz = new double[V];
        for (int i = 0; i < V; i++)
        {
            t = Tok(Console.ReadLine());
            px[i] = double.Parse(t[0]); py[i] = double.Parse(t[1]); pz[i] = double.Parse(t[2]);
        }
        for (int i = 0; i < F; i++) Console.ReadLine();
        t = Tok(Console.ReadLine());
        double W = double.Parse(t[0]), H = double.Parse(t[1]), D = double.Parse(t[2]);

        const double DEG = Math.PI / 180.0;
        const int STEP = 15;

        int nC = 0;
        for (int a = 0; a < 360; a += STEP)
            for (int b = 0; b <= 180; b += STEP)
                for (int c = 0; c < 360; c += STEP)
                    nC++;

        int extra = 1;
        double[] cQw = new double[nC + extra], cQx = new double[nC + extra], cQy = new double[nC + extra], cQz = new double[nC + extra];
        double[] cmX = new double[nC + extra], cmY = new double[nC + extra], cmZ = new double[nC + extra];
        double[] cdX = new double[nC + extra], cdY = new double[nC + extra], cdZ = new double[nC + extra];

        int ci = 0;
        for (int a = 0; a < 360; a += STEP)
            for (int b = 0; b <= 180; b += STEP)
                for (int c = 0; c < 360; c += STEP)
                {
                    Euler(a * DEG, b * DEG, c * DEG, out cQw[ci], out cQx[ci], out cQy[ci], out cQz[ci]);
                    AABB(cQw[ci], cQx[ci], cQy[ci], cQz[ci], out cmX[ci], out cmY[ci], out cmZ[ci], out cdX[ci], out cdY[ci], out cdZ[ci]);
                    ci++;
                }

        long bestCnt = 0;
        double bQw = 1, bQx = 0, bQy = 0, bQz = 0;
        double bmX2 = 0, bmY2 = 0, bmZ2 = 0, bdX2 = 1, bdY2 = 1, bdZ2 = 1;

        for (int i = 0; i < nC; i++)
        {
            long cnt = Grid(cdX[i], cdY[i], cdZ[i], W, H, D);
            if (cnt > bestCnt)
            {
                bestCnt = cnt;
                bQw = cQw[i]; bQx = cQx[i]; bQy = cQy[i]; bQz = cQz[i];
                bmX2 = cmX[i]; bmY2 = cmY[i]; bmZ2 = cmZ[i];
                bdX2 = cdX[i]; bdY2 = cdY[i]; bdZ2 = cdZ[i];
            }
        }

        const int TOP = 30;
        int[] topI = new int[TOP];
        long[] topCV = new long[TOP];
        for (int i = 0; i < TOP; i++) { topI[i] = -1; topCV[i] = -1; }

        for (int i = 0; i < nC; i++)
        {
            long cnt = Grid(cdX[i], cdY[i], cdZ[i], W, H, D);
            int worst = 0;
            for (int j = 1; j < TOP; j++) if (topCV[j] < topCV[worst]) worst = j;
            if (cnt > topCV[worst]) { topI[worst] = i; topCV[worst] = cnt; }
        }

        void Try(double qw, double qx, double qy, double qz)
        {
            AABB(qw, qx, qy, qz, out double mx, out double my, out double mz, out double dx, out double dy, out double dz);
            long cnt = Grid(dx, dy, dz, W, H, D);
            if (cnt > bestCnt)
            {
                bestCnt = cnt;
                bQw = qw; bQx = qx; bQy = qy; bQz = qz;
                bmX2 = mx; bmY2 = my; bmZ2 = mz;
                bdX2 = dx; bdY2 = dy; bdZ2 = dz;
            }
        }

        int refineCount = V > 5000 ? 10 : TOP;
        for (int k = 0; k < refineCount; k++)
        {
            if (topI[k] < 0) continue;
            int ti = topI[k];
            for (int da = -14; da <= 14; da += 2)
                for (int db = -14; db <= 14; db += 2)
                    for (int dc = -14; dc <= 14; dc += 2)
                    {
                        if (da == 0 && db == 0 && dc == 0) continue;
                        Euler(da * DEG, db * DEG, dc * DEG, out double dw, out double dx2, out double dy2, out double dz2);
                        QMul(dw, dx2, dy2, dz2, cQw[ti], cQx[ti], cQy[ti], cQz[ti], out double nw, out double nx2, out double ny2, out double nz2);
                        Try(nw, nx2, ny2, nz2);
                    }
        }

        {
            double baseW = bQw, baseX = bQx, baseY = bQy, baseZ = bQz;
            int fineSteps = V > 5000 ? 7 : 10;
            for (int da = -fineSteps; da <= fineSteps; da++)
                for (int db = -fineSteps; db <= fineSteps; db++)
                    for (int dc = -fineSteps; dc <= fineSteps; dc++)
                    {
                        if (da == 0 && db == 0 && dc == 0) continue;
                        double ar = da * 0.3 * DEG, br = db * 0.3 * DEG, cr = dc * 0.3 * DEG;
                        Euler(ar, br, cr, out double dw, out double dx2, out double dy2, out double dz2);
                        QMul(dw, dx2, dy2, dz2, baseW, baseX, baseY, baseZ, out double nw, out double nx2, out double ny2, out double nz2);
                        Try(nw, nx2, ny2, nz2);
                    }
        }

        if (bestCnt == 0)
        {
            Console.WriteLine(0);
            return;
        }

        cQw[nC] = bQw; cQx[nC] = bQx; cQy[nC] = bQy; cQz[nC] = bQz;
        cmX[nC] = bmX2; cmY[nC] = bmY2; cmZ[nC] = bmZ2;
        cdX[nC] = bdX2; cdY[nC] = bdY2; cdZ[nC] = bdZ2;
        int nCands = nC + 1;

        long mainNx = GridN(W, bdX2), mainNy = GridN(H, bdY2), mainNz = GridN(D, bdZ2);
        double usedW = mainNx * bdX2, usedH = mainNy * bdY2, usedD = mainNz * bdZ2;

        double[] rW = { W - usedW, usedW, usedW };
        double[] rH = { H, H - usedH, usedH };
        double[] rD = { D, D, D - usedD };
        double[] rOX = { usedW, 0, 0 };
        double[] rOY = { 0, usedH, 0 };
        double[] rOZ = { 0, 0, usedD };

        int[] sI = { -1, -1, -1 };
        long[] sNx = new long[3], sNy = new long[3], sNz = new long[3];
        long stripTot = 0;

        for (int s = 0; s < 3; s++)
        {
            if (rW[s] < 1e-9 || rH[s] < 1e-9 || rD[s] < 1e-9) continue;
            long best = 0;
            for (int i = 0; i < nCands; i++)
            {
                long cnt = Grid(cdX[i], cdY[i], cdZ[i], rW[s], rH[s], rD[s]);
                if (cnt > best)
                {
                    best = cnt;
                    sI[s] = i;
                    sNx[s] = GridN(rW[s], cdX[i]);
                    sNy[s] = GridN(rH[s], cdY[i]);
                    sNz[s] = GridN(rD[s], cdZ[i]);
                }
            }
            stripTot += best;
        }

        long total = mainNx * mainNy * mainNz + stripTot;

        double len = Math.Sqrt(bQw * bQw + bQx * bQx + bQy * bQy + bQz * bQz);
        bQw /= len; bQx /= len; bQy /= len; bQz /= len;

        var writer = new StreamWriter(Console.OpenStandardOutput(), new System.Text.UTF8Encoding(false), 1 << 16);
        writer.WriteLine(total);

        for (long ix = 0; ix < mainNx; ix++)
            for (long iy = 0; iy < mainNy; iy++)
                for (long iz = 0; iz < mainNz; iz++)
                {
                    writer.Write(R(ix * bdX2 - bmX2)); writer.Write(' ');
                    writer.Write(R(iy * bdY2 - bmY2)); writer.Write(' ');
                    writer.Write(R(iz * bdZ2 - bmZ2)); writer.Write(' ');
                    writer.Write(R(bQw)); writer.Write(' ');
                    writer.Write(R(bQx)); writer.Write(' ');
                    writer.Write(R(bQy)); writer.Write(' ');
                    writer.WriteLine(R(bQz));
                }

        for (int s = 0; s < 3; s++)
        {
            if (sI[s] < 0) continue;
            int si = sI[s];
            double sqw = cQw[si], sqx = cQx[si], sqy = cQy[si], sqz = cQz[si];
            double slen = Math.Sqrt(sqw * sqw + sqx * sqx + sqy * sqy + sqz * sqz);
            sqw /= slen; sqx /= slen; sqy /= slen; sqz /= slen;

            for (long ix = 0; ix < sNx[s]; ix++)
                for (long iy = 0; iy < sNy[s]; iy++)
                    for (long iz = 0; iz < sNz[s]; iz++)
                    {
                        writer.Write(R(rOX[s] + ix * cdX[si] - cmX[si])); writer.Write(' ');
                        writer.Write(R(rOY[s] + iy * cdY[si] - cmY[si])); writer.Write(' ');
                        writer.Write(R(rOZ[s] + iz * cdZ[si] - cmZ[si])); writer.Write(' ');
                        writer.Write(R(sqw)); writer.Write(' ');
                        writer.Write(R(sqx)); writer.Write(' ');
                        writer.Write(R(sqy)); writer.Write(' ');
                        writer.WriteLine(R(sqz));
                    }
        }

        writer.Flush();
    }

    static string R(double v) => v.ToString("R", CultureInfo.InvariantCulture);
}
