#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <array>
#include <chrono>

#ifndef M_PI
#define M_PI 3.14159265358979323846
#endif

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
    Vec3 operator+(const Vec3& o) const { return {x+o.x, y+o.y, z+o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x-o.x, y-o.y, z-o.z}; }
    Vec3 operator*(double s) const { return {x*s, y*s, z*s}; }
    double dot(const Vec3& o) const { return x*o.x + y*o.y + z*o.z; }
    Vec3 cross(const Vec3& o) const {
        return {y*o.z - z*o.y, z*o.x - x*o.z, x*o.y - y*o.x};
    }
    double norm() const { return std::sqrt(x*x + y*y + z*z); }
    Vec3 normalized() const { double n = norm(); if(n<1e-15) return {0,0,0}; return {x/n, y/n, z/n}; }
};

struct Plane {
    Vec3 normal;
    double d;
    double distTo(const Vec3& p) const { return normal.dot(p) + d; }
};

static std::vector<Vec3> containerVerts;
static std::vector<Plane> containerPlanes;

struct Mat3 {
    double m[3][3];
    static Mat3 rotEuler(double rx, double ry, double rz) {
        double cx = cos(rx), sx = sin(rx);
        double cy = cos(ry), sy = sin(ry);
        double cz = cos(rz), sz = sin(rz);
        Mat3 r;
        r.m[0][0] = cy*cz; r.m[0][1] = sx*sy*cz - cx*sz; r.m[0][2] = cx*sy*cz + sx*sz;
        r.m[1][0] = cy*sz; r.m[1][1] = sx*sy*sz + cx*cz; r.m[1][2] = cx*sy*sz - sx*cz;
        r.m[2][0] = -sy;   r.m[2][1] = sx*cy;             r.m[2][2] = cx*cy;
        return r;
    }
    Vec3 apply(const Vec3& v) const {
        return {m[0][0]*v.x + m[0][1]*v.y + m[0][2]*v.z,
                m[1][0]*v.x + m[1][1]*v.y + m[1][2]*v.z,
                m[2][0]*v.x + m[2][1]*v.y + m[2][2]*v.z};
    }
};

void buildConvexHull() {
    int n = (int)containerVerts.size();
    if (n < 4) return;
    containerPlanes.clear();
    for (int i = 0; i < n; i++) {
        for (int j = i+1; j < n; j++) {
            for (int k = j+1; k < n; k++) {
                Vec3 ab = containerVerts[j] - containerVerts[i];
                Vec3 ac = containerVerts[k] - containerVerts[i];
                Vec3 normal = ab.cross(ac);
                if (normal.norm() < 1e-12) continue;
                normal = normal.normalized();
                double d = -normal.dot(containerVerts[i]);
                bool allNeg = true, allPos = true;
                for (int l = 0; l < n; l++) {
                    double dist = normal.dot(containerVerts[l]) + d;
                    if (dist > 1e-9) allNeg = false;
                    if (dist < -1e-9) allPos = false;
                }
                if (allNeg) {
                    containerPlanes.push_back({normal, d});
                } else if (allPos) {
                    containerPlanes.push_back({normal * (-1.0), -d});
                }
            }
        }
    }
    std::vector<Plane> unique;
    for (auto& p : containerPlanes) {
        bool dup = false;
        for (auto& u : unique) {
            if (std::abs(p.normal.dot(u.normal) - 1.0) < 1e-6 && std::abs(p.d - u.d) < 1e-6) {
                dup = true; break;
            }
        }
        if (!dup) unique.push_back(p);
    }
    containerPlanes = unique;
}

bool isInsideContainer(const Vec3& p) {
    for (auto& pl : containerPlanes) {
        if (pl.distTo(p) > 1e-9) return false;
    }
    return true;
}

std::array<Vec3,4> getTetraVerts(const Vec3& center, double rx, double ry, double rz, double e) {
    double a = e / (2.0 * std::sqrt(2.0));
    std::array<Vec3,4> base = {{
        {a, a, a}, {a, -a, -a}, {-a, a, -a}, {-a, -a, a}
    }};
    Mat3 rot = Mat3::rotEuler(rx, ry, rz);
    std::array<Vec3,4> result;
    for (int i = 0; i < 4; i++) {
        result[i] = center + rot.apply(base[i]);
    }
    return result;
}

bool tetraInsideContainer(const std::array<Vec3,4>& verts) {
    for (auto& v : verts) {
        if (!isInsideContainer(v)) return false;
    }
    return true;
}

struct PlacedTetra {
    Vec3 center;
    double rx, ry, rz;
};

int main() {
    auto startTime = std::chrono::steady_clock::now();
    auto elapsed = [&]() {
        return std::chrono::duration<double>(std::chrono::steady_clock::now() - startTime).count();
    };

    int V;
    double edgeLen;
    std::cin >> V >> edgeLen;

    containerVerts.resize(V);
    for (int i = 0; i < V; i++) {
        std::cin >> containerVerts[i].x >> containerVerts[i].y >> containerVerts[i].z;
    }

    buildConvexHull();

    if (containerPlanes.empty()) {
        std::cout << 0 << std::endl;
        return 0;
    }

    double bx0 = 1e18, by0 = 1e18, bz0 = 1e18;
    double bx1 = -1e18, by1 = -1e18, bz1 = -1e18;
    for (auto& v : containerVerts) {
        bx0 = std::min(bx0, v.x); by0 = std::min(by0, v.y); bz0 = std::min(bz0, v.z);
        bx1 = std::max(bx1, v.x); by1 = std::max(by1, v.y); bz1 = std::max(bz1, v.z);
    }

    double e = edgeLen;
    double a = e / (2.0 * std::sqrt(2.0));
    double step = 2.0 * a;

    double rx_up = 0, ry_up = 0, rz_up = 0;
    double rx_down = M_PI/2.0, ry_down = 0, rz_down = 0;

    int nx = (int)std::ceil((bx1 - bx0) / step) + 1;
    int ny = (int)std::ceil((by1 - by0) / step) + 1;
    int nz = (int)std::ceil((bz1 - bz0) / step) + 1;

    std::vector<PlacedTetra> placed;

    // Try multiple grid offsets to maximize packing
    std::vector<PlacedTetra> bestPlaced;

    double offsets[5][3] = {
        {0, 0, 0},
        {step*0.25, step*0.25, step*0.25},
        {step*0.5, 0, 0},
        {0, step*0.5, 0},
        {0, 0, step*0.5}
    };

    for (int oi = 0; oi < 5 && elapsed() < 20.0; oi++) {
        placed.clear();
        double ox = offsets[oi][0];
        double oy = offsets[oi][1];
        double oz = offsets[oi][2];

        for (int iz = -1; iz <= nz && elapsed() < 24.0; iz++) {
            for (int iy = -1; iy <= ny; iy++) {
                for (int ix = -1; ix <= nx; ix++) {
                    double cx = bx0 + ix * step + ox;
                    double cy = by0 + iy * step + oy;
                    double cz = bz0 + iz * step + oz;
                    Vec3 center(cx, cy, cz);

                    int parity = ((ix % 2 + 2) % 2 + (iy % 2 + 2) % 2 + (iz % 2 + 2) % 2) % 2;
                    double rx = parity == 0 ? rx_up : rx_down;
                    double ry = parity == 0 ? ry_up : ry_down;
                    double rz = parity == 0 ? rz_up : rz_down;

                    auto verts = getTetraVerts(center, rx, ry, rz, e);
                    if (tetraInsideContainer(verts)) {
                        placed.push_back({center, rx, ry, rz});
                    }
                }
            }
        }

        if (placed.size() > bestPlaced.size()) {
            bestPlaced = placed;
        }
    }

    std::cout << bestPlaced.size() << "\n";
    for (auto& t : bestPlaced) {
        std::cout << t.center.x << " " << t.center.y << " " << t.center.z << " "
                  << t.rx << " " << t.ry << " " << t.rz << "\n";
    }

    return 0;
}
