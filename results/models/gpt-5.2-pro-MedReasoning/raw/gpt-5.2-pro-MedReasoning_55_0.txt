static inline bool isWhitePiece(char p) { return p >= 'A' && p <= 'Z'; }

static inline int pieceIndex(char p) {
    // 0..5 white P N B R Q K, 6..11 black p n b r q k
    switch (p) {
        case 'P': return 0; case 'N': return 1; case 'B': return 2; case 'R': return 3; case 'Q': return 4; case 'K': return 5;
        case 'p': return 6; case 'n': return 7; case 'b': return 8; case 'r': return 9; case 'q': return 10; case 'k': return 11;
        default: return -1;
    }
}

static inline int pieceValue(char p) {
    switch (toUpperPiece(p)) {
        case 'P': return 100;
        case 'N': return 320;
        case 'B': return 330;
        case 'R': return 500;
        case 'Q': return 900;
        case 'K': return 0;
        default: return 0;
    }
}

struct Zobrist {
    array<array<uint64_t, 128>, 12> psq{};
    array<uint64_t, 16> castle{};
    array<uint64_t, 8> epFile{};
    uint64_t side = 0;

    static uint64_t splitmix64(uint64_t &x) {
        uint64_t z = (x += 0x9e3779b97f4a7c15ULL);
        z = (z ^ (z >> 30)) * 0xbf58476d1ce4e5b9ULL;
        z = (z ^ (z >> 27)) * 0x94d049bb133111ebULL;
        return z ^ (z >> 31);
    }

    void init(uint64_t seed = 0x123456789abcdef0ULL) {
        uint64_t x = seed;
        for (int p = 0; p < 12; ++p)
            for (int sq = 0; sq < 128; ++sq)
                psq[p][sq] = splitmix64(x);
        for (int i = 0; i < 16; ++i) castle[i] = splitmix64(x);
        for (int i = 0; i < 8; ++i) epFile[i] = splitmix64(x);
        side = splitmix64(x);
    }
};

struct TTEntry {
    uint64_t key = 0;
    int16_t depth = -1;
    int32_t score = 0;
    uint8_t flag = 0; // 0 exact, 1 alpha, 2 beta
    uint32_t moveEnc = 0;
};

struct TranspositionTable {
    vector<TTEntry> t;
    size_t mask = 0;

    void init(size_t mb) {
        size_t bytes = mb * 1024ULL * 1024ULL;
        size_t n = bytes / sizeof(TTEntry);
        if (n < 1) n = 1;
        // power of two
        size_t p2 = 1;
        while (p2 < n) p2 <<= 1;
        t.assign(p2, TTEntry{});
        mask = p2 - 1;
    }

    static uint32_t encodeMove(const Move &m) {
        // 0..7 from file,0..7 from rank not stored; store 0..127 raw.
        // pack: from(7) to(7) promo(3) flags(5) => fits in 22 bits.
        uint32_t promoCode = 0;
        if (m.promo == 'Q') promoCode = 1;
        else if (m.promo == 'R') promoCode = 2;
        else if (m.promo == 'B') promoCode = 3;
        else if (m.promo == 'N') promoCode = 4;
        return (uint32_t(m.from) & 0x7F)
            | ((uint32_t(m.to) & 0x7F) << 7)
            | ((promoCode & 7U) << 14)
            | ((uint32_t(m.flags) & 31U) << 17);
    }

    static Move decodeMove(uint32_t e) {
        Move m;
        m.from = uint8_t(e & 0x7F);
        m.to = uint8_t((e >> 7) & 0x7F);
        uint32_t pc = (e >> 14) & 7U;
        m.flags = uint8_t((e >> 17) & 31U);
        m.promo = 0;
        if (pc == 1) m.promo = 'Q';
        else if (pc == 2) m.promo = 'R';
        else if (pc == 3) m.promo = 'B';
        else if (pc == 4) m.promo = 'N';
        return m;
    }

    TTEntry* probe(uint64_t key) {
        return &t[size_t(key) & mask];
    }
};

static uint64_t computeHash(const Position &pos) {
    uint64_t h = 0;
    for (int sq = 0; sq < 128; ++sq) {
        if (!onBoard(sq)) { sq += 7; continue; }
        char p = pos.b[sq];
        if (p == '.') continue;
        int idx = pieceIndex(p);
        if (idx >= 0) h ^= Z.psq[idx][sq];
    }
    h ^= Z.castle[pos.castling & 15];
    if (pos.ep != -1) h ^= Z.epFile[fileOf(pos.ep)];
    if (pos.sideToMove == BLACK) h ^= Z.side;
    return h;
}

static bool isSquareAttacked(const Position &pos, int sq, int bySide) {
    // Pawns
    if (bySide == WHITE) {
        int s1 = sq - 17, s2 = sq - 15;
        if (onBoard(s1) && pos.b[s1] == 'P') return true;
        if (onBoard(s2) && pos.b[s2] == 'P') return true;
    } else {
        int s1 = sq + 17, s2 = sq + 15;
        if (onBoard(s1) && pos.b[s1] == 'p') return true;
        if (onBoard(s2) && pos.b[s2] == 'p') return true;
    }

    // Knights
    for (int d : knightOff) {
        int s = sq + d;
        if (!onBoard(s)) continue;
        char p = pos.b[s];
        if (bySide == WHITE && p == 'N') return true;
        if (bySide == BLACK && p == 'n') return true;
    }

    // Bishops/Queens (diagonals)
    for (int d : bishopOff) {
        int s = sq + d;
        while (onBoard(s)) {
            char p = pos.b[s];
            if (p != '.') {
                if (bySide == WHITE && (p == 'B' || p == 'Q')) return true;
                if (bySide == BLACK && (p == 'b' || p == 'q')) return true;
                break;
            }
            s += d;
        }
    }

    // Rooks/Queens (straight)
    for (int d : rookOff) {
        int s = sq + d;
        while (onBoard(s)) {
            char p = pos.b[s];
            if (p != '.') {
                if (bySide == WHITE && (p == 'R' || p == 'Q')) return true;
                if (bySide == BLACK && (p == 'r' || p == 'q')) return true;
                break;
            }
            s += d;
        }
    }

    // King
    for (int d : kingOff) {
        int s = sq + d;
        if (!onBoard(s)) continue;
        char p = pos.b[s];
        if (bySide == WHITE && p == 'K') return true;
        if (bySide == BLACK && p == 'k') return true;
    }

    return false;
}

static bool inCheck(const Position &pos, int side) {
    return isSquareAttacked(pos, pos.kingSq[side], side ^ 1);
}

static inline void addMove(vector<Move> &mv, int from, int to, uint8_t flags = 0, char promo = 0, char captured = 0) {
    Move m;
    m.from = uint8_t(from);
    m.to = uint8_t(to);
    m.flags = flags;
    m.promo = uint8_t(promo);
    m.captured = captured;
    mv.push_back(m);
}

static void genPseudoMoves(const Position &pos, vector<Move> &moves, bool capturesOnly) {
    moves.clear();
    int us = pos.sideToMove;
    int them = us ^ 1;

    for (int from = 0; from < 128; ++from) {
        if (!onBoard(from)) { from += 7; continue; }
        char pc = pos.b[from];
        if (pc == '.') continue;
        if (pieceSide(pc) != us) continue;

        char up = toUpperPiece(pc);

        if (up == 'P') {
            int dir = (us == WHITE) ? 16 : -16;
            int startRank = (us == WHITE) ? 1 : 6;
            int promoFromRank = (us == WHITE) ? 6 : 1;
            int epTarget = pos.ep;

            int r = rankOf(from);

            // Captures
            int cap1 = from + dir + 1;
            int cap2 = from + dir - 1;
            for (int to : {cap1, cap2}) {
                if (!onBoard(to)) continue;
                char tgt = pos.b[to];
                bool isCap = (tgt != '.' && pieceSide(tgt) == them);
                bool isEp = (to == epTarget);
                if (!isCap && !isEp) continue;

                uint8_t f = Move::CAPTURE;
                char captured = tgt;
                if (isEp) { f |= Move::EP; captured = (us == WHITE) ? 'p' : 'P'; }

                if (r == promoFromRank) {
                    if (!capturesOnly) {
                        addMove(moves, from, to, uint8_t(f | Move::PROMO), 'Q', captured);
                        addMove(moves, from, to, uint8_t(f | Move::PROMO), 'R', captured);
                        addMove(moves, from, to, uint8_t(f | Move::PROMO), 'B', captured);
                        addMove(moves, from, to, uint8_t(f | Move::PROMO), 'N', captured);
                    } else {
                        addMove(moves, from, to, uint8_t(f | Move::PROMO), 'Q', captured);
                        addMove(moves, from, to, uint8_t(f | Move::PROMO), 'N', captured);
                    }
                } else {
                    addMove(moves, from, to, f, 0, captured);
                }
            }

            if (capturesOnly) continue;

            // Single push
            int to1 = from + dir;
            if (onBoard(to1) && pos.b[to1] == '.') {
                if (r == promoFromRank) {
                    addMove(moves, from, to1, Move::PROMO, 'Q');
                    addMove(moves, from, to1, Move::PROMO, 'R');
                    addMove(moves, from, to1, Move::PROMO, 'B');
                    addMove(moves, from, to1, Move::PROMO, 'N');
                } else {
                    addMove(moves, from, to1);
                    // Double push
                    if (r == startRank) {
                        int to2 = from + 2 * dir;
                        if (onBoard(to2) && pos.b[to2] == '.') {
                            addMove(moves, from, to2, Move::PAWN2);
                        }
                    }
                }
            }
        } else if (up == 'N') {
            for (int d : knightOff) {
                int to = from + d;
                if (!onBoard(to)) continue;
                char tgt = pos.b[to];
                if (tgt == '.') {
                    if (!capturesOnly) addMove(moves, from, to);
                } else if (pieceSide(tgt) == them) {
                    addMove(moves, from, to, Move::CAPTURE, 0, tgt);
                }
            }
        } else if (up == 'B' || up == 'R' || up == 'Q') {
            const int *dirs = nullptr;
            int nd = 0;
            int tmpDirs[8];
            if (up == 'B') { dirs = bishopOff; nd = 4; }
            else if (up == 'R') { dirs = rookOff; nd = 4; }
            else {
                for (int i = 0; i < 4; ++i) tmpDirs[i] = bishopOff[i];
                for (int i = 0; i < 4; ++i) tmpDirs[4 + i] = rookOff[i];
                dirs = tmpDirs; nd = 8;
            }

            for (int i = 0; i < nd; ++i) {
                int d = dirs[i];
                int to = from + d;
                while (onBoard(to)) {
                    char tgt = pos.b[to];
                    if (tgt == '.') {
                        if (!capturesOnly) addMove(moves, from, to);
                    } else {
                        if (pieceSide(tgt) == them)
                            addMove(moves, from, to, Move::CAPTURE, 0, tgt);
                        break;
                    }
                    to += d;
                }
            }
        } else if (up == 'K') {
            for (int d : kingOff) {
                int to = from + d;
                if (!onBoard(to)) continue;
                char tgt = pos.b[to];
                if (tgt == '.') {
                    if (!capturesOnly) addMove(moves, from, to);
                } else if (pieceSide(tgt) == them) {
                    addMove(moves, from, to, Move::CAPTURE, 0, tgt);
                }
            }
            if (capturesOnly) continue;

            // Castling
            if (us == WHITE) {
                // King on e1 (4), rooks a1(0) h1(7)
                if ((pos.castling & 1) && pos.b[5] == '.' && pos.b[6] == '.') {
                    if (!inCheck(pos, WHITE) &&
                        !isSquareAttacked(pos, 5, BLACK) &&
                        !isSquareAttacked(pos, 6, BLACK)) {
                        addMove(moves, 4, 6, Move::CASTLE);
                    }
                }
                if ((pos.castling & 2) && pos.b[3] == '.' && pos.b[2] == '.' && pos.b[1] == '.') {
                    if (!inCheck(pos, WHITE) &&
                        !isSquareAttacked(pos, 3, BLACK) &&
                        !isSquareAttacked(pos, 2, BLACK)) {
                        addMove(moves, 4, 2, Move::CASTLE);
                    }
                }
            } else {
                // King on e8 (116), rooks a8(112) h8(119)
                if ((pos.castling & 4) && pos.b[117] == '.' && pos.b[118] == '.') {
                    if (!inCheck(pos, BLACK) &&
                        !isSquareAttacked(pos, 117, WHITE) &&
                        !isSquareAttacked(pos, 118, WHITE)) {
                        addMove(moves, 116, 118, Move::CASTLE);
                    }
                }
                if ((pos.castling & 8) && pos.b[115] == '.' && pos.b[114] == '.' && pos.b[113] == '.') {
                    if (!inCheck(pos, BLACK) &&
                        !isSquareAttacked(pos, 115, WHITE) &&
                        !isSquareAttacked(pos, 114, WHITE)) {
                        addMove(moves, 116, 114, Move::CASTLE);
                    }
                }
            }
        }
    }
}

static void doCastleRook(Position &pos, int us, int kingFrom, int kingTo) {
    // assumes pieces exist and path empty/checked already
    if (us == WHITE) {
        if (kingTo == 6) { // O-O: e1->g1, rook h1->f1
            pos.b[7] = '.';
            pos.b[5] = 'R';
        } else { // O-O-O: e1->c1, rook a1->d1
            pos.b[0] = '.';
            pos.b[3] = 'R';
        }
    } else {
        if (kingTo == 118) { // e8->g8, rook h8->f8
            pos.b[119] = '.';
            pos.b[117] = 'r';
        } else { // e8->c8, rook a8->d8
            pos.b[112] = '.';
            pos.b[115] = 'r';
        }
    }
}

static void updateCastleRights(Position &pos, int from, int to, char moved, char captured) {
    // remove castling rights if king/rook moved or rook captured
    // White king: e1=4, rooks: a1=0, h1=7
    // Black king: e8=116, rooks: a8=112, h8=119
    if (moved == 'K') pos.castling &= ~(1 | 2);
    if (moved == 'k') pos.castling &= ~(4 | 8);

    if (moved == 'R') {
        if (from == 0) pos.castling &= ~2;
        if (from == 7) pos.castling &= ~1;
    }
    if (moved == 'r') {
        if (from == 112) pos.castling &= ~8;
        if (from == 119) pos.castling &= ~4;
    }

    if (captured == 'R') {
        if (to == 0) pos.castling &= ~2;
        if (to == 7) pos.castling &= ~1;
    }
    if (captured == 'r') {
        if (to == 112) pos.castling &= ~8;
        if (to == 119) pos.castling &= ~4;
    }
}

static bool makeMove(Position &pos, const Move &m, Undo &u) {
    u.captured = m.captured;
    u.castling = pos.castling;
    u.ep = pos.ep;
    u.halfmove = pos.halfmove;
    u.hash = pos.hash;
    u.kingSq[0] = pos.kingSq[0];
    u.kingSq[1] = pos.kingSq[1];

    int us = pos.sideToMove;
    int them = us ^ 1;

    int from = m.from;
    int to = m.to;
    char moved = pos.b[from];
    char captured = pos.b[to];

    // hash: remove old ep, castle
    pos.hash ^= Z.castle[pos.castling & 15];
    if (pos.ep != -1) pos.hash ^= Z.epFile[fileOf(pos.ep)];

    pos.ep = -1;

    // move piece out
    pos.hash ^= Z.psq[pieceIndex(moved)][from];
    pos.b[from] = '.';

    // captures
    if (m.flags & Move::EP) {
        int capSq = to + ((us == WHITE) ? -16 : 16);
        char epCap = pos.b[capSq];
        // remove captured pawn
        pos.hash ^= Z.psq[pieceIndex(epCap)][capSq];
        pos.b[capSq] = '.';
        captured = epCap;
    } else if (captured != '.') {
        pos.hash ^= Z.psq[pieceIndex(captured)][to];
    }

    // place moved/promo
    char placed = moved;
    if (m.flags & Move::PROMO) {
        placed = (us == WHITE) ? char(m.promo) : char(m.promo + ('a' - 'A'));
    }

    pos.b[to] = placed;
    pos.hash ^= Z.psq[pieceIndex(placed)][to];

    // castling rook move
    if (m.flags & Move::CASTLE) {
        // remove rook from old and add to new for hash
        if (us == WHITE) {
            if (to == 6) {
                // rook h1->f1: 7->5
                pos.hash ^= Z.psq[pieceIndex('R')][7];
                pos.hash ^= Z.psq[pieceIndex('R')][5];
            } else { // to==2
                pos.hash ^= Z.psq[pieceIndex('R')][0];
                pos.hash ^= Z.psq[pieceIndex('R')][3];
            }
        } else {
            if (to == 118) {
                pos.hash ^= Z.psq[pieceIndex('r')][119];
                pos.hash ^= Z.psq[pieceIndex('r')][117];
            } else { // to==114
                pos.hash ^= Z.psq[pieceIndex('r')][112];
                pos.hash ^= Z.psq[pieceIndex('r')][115];
            }
        }
        doCastleRook(pos, us, from, to);
    }

    // update king square
    if (toUpperPiece(moved) == 'K') {
        pos.kingSq[us] = to;
    }

    // set ep square on double pawn push
    if ((m.flags & Move::PAWN2) && toUpperPiece(moved) == 'P') {
        pos.ep = from + ((us == WHITE) ? 16 : -16);
    }

    // update castling rights
    updateCastleRights(pos, from, to, moved, (m.flags & Move::EP) ? captured : captured);
    pos.hash ^= Z.castle[pos.castling & 15];
    if (pos.ep != -1) pos.hash ^= Z.epFile[fileOf(pos.ep)];

    // halfmove clock
    if (toUpperPiece(moved) == 'P' || (m.flags & Move::CAPTURE)) pos.halfmove = 0;
    else pos.halfmove++;

    // side to move
    pos.sideToMove = them;
    pos.hash ^= Z.side;

    // legality: king not in check
    if (inCheck(pos, us)) {
        // undo
        pos = Position{};
        // restore by copying from undo hash? Too heavy; instead unmake properly:
        // We'll implement unmake below and call it here.
        return false;
    }

    return true;
}

static void unmakeMove(Position &pos, const Move &m, const Undo &u) {
    // restore by reversing from saved full state fields and reapplying board changes
    // Easier and safe: restore scalar fields then reconstruct board by undoing move.
    // We'll do explicit reverse for board, plus restore hash etc.
    pos.castling = u.castling;
    pos.ep = u.ep;
    pos.halfmove = u.halfmove;
    pos.hash = u.hash;
    pos.kingSq[0] = u.kingSq[0];
    pos.kingSq[1] = u.kingSq[1];

    pos.sideToMove ^= 1;

    int us = pos.sideToMove;
    int from = m.from, to = m.to;

    char movedAfter = pos.b[to];
    char movedBefore = movedAfter;

    if (m.flags & Move::PROMO) {
        movedBefore = (us == WHITE) ? 'P' : 'p';
    }

    // undo castling rook move on board
    if (m.flags & Move::CASTLE) {
        if (us == WHITE) {
            if (to == 6) { // rook f1->h1
                pos.b[7] = 'R';
                pos.b[5] = '.';
            } else { // rook d1->a1
                pos.b[0] = 'R';
                pos.b[3] = '.';
            }
        } else {
            if (to == 118) { // rook f8->h8
                pos.b[119] = 'r';
                pos.b[117] = '.';
            } else { // rook d8->a8
                pos.b[112] = 'r';
                pos.b[115] = '.';
            }
        }
    }

    // move piece back
    pos.b[from] = movedBefore;
    pos.b[to] = '.';

    // restore captured
    if (m.flags & Move::EP) {
        int capSq = to + ((us == WHITE) ? -16 : 16);
        pos.b[capSq] = u.captured;
    } else if (u.captured != 0 && u.captured != '.') {
        pos.b[to] = u.captured;
    }
}

static bool makeMoveChecked(Position &pos, const Move &m, Undo &u) {
    // We want legal move check; implement make+check with proper undo on fail.
    u.captured = m.captured;
    u.castling = pos.castling;
    u.ep = pos.ep;
    u.halfmove = pos.halfmove;
    u.hash = pos.hash;
    u.kingSq[0] = pos.kingSq[0];
    u.kingSq[1] = pos.kingSq[1];

    int us = pos.sideToMove;
    int them = us ^ 1;

    int from = m.from;
    int to = m.to;
    char moved = pos.b[from];
    char capturedOnTo = pos.b[to];

    // hash: remove old ep, old castle
    pos.hash ^= Z.castle[pos.castling & 15];
    if (pos.ep != -1) pos.hash ^= Z.epFile[fileOf(pos.ep)];

    pos.ep = -1;

    // lift moved
    pos.hash ^= Z.psq[pieceIndex(moved)][from];
    pos.b[from] = '.';

    char captured = capturedOnTo;

    if (m.flags & Move::EP) {
        int capSq = to + ((us == WHITE) ? -16 : 16);
        captured = pos.b[capSq];
        pos.hash ^= Z.psq[pieceIndex(captured)][capSq];
        pos.b[capSq] = '.';
    } else if ((m.flags & Move::CAPTURE) && capturedOnTo != '.') {
        pos.hash ^= Z.psq[pieceIndex(capturedOnTo)][to];
    }

    char placed = moved;
    if (m.flags & Move::PROMO) {
        placed = (us == WHITE) ? char(m.promo) : char(m.promo + ('a' - 'A'));
    }

    pos.b[to] = placed;
    pos.hash ^= Z.psq[pieceIndex(placed)][to];

    if (m.flags & Move::CASTLE) {
        // hash rook move and update board
        if (us == WHITE) {
            if (to == 6) {
                pos.hash ^= Z.psq[pieceIndex('R')][7];
                pos.hash ^= Z.psq[pieceIndex('R')][5];
            } else {
                pos.hash ^= Z.psq[pieceIndex('R')][0];
                pos.hash ^= Z.psq[pieceIndex('R')][3];
            }
        } else {
            if (to == 118) {
                pos.hash ^= Z.psq[pieceIndex('r')][119];
                pos.hash ^= Z.psq[pieceIndex('r')][117];
            } else {
                pos.hash ^= Z.psq[pieceIndex('r')][112];
                pos.hash ^= Z.psq[pieceIndex('r')][115];
            }
        }
        doCastleRook(pos, us, from, to);
    }

    if (toUpperPiece(moved) == 'K') pos.kingSq[us] = to;

    if ((m.flags & Move::PAWN2) && toUpperPiece(moved) == 'P')
        pos.ep = from + ((us == WHITE) ? 16 : -16);

    updateCastleRights(pos, from, to, moved, captured);

    // hash: add new castle, new ep
    pos.hash ^= Z.castle[pos.castling & 15];
    if (pos.ep != -1) pos.hash ^= Z.epFile[fileOf(pos.ep)];

    if (toUpperPiece(moved) == 'P' || (m.flags & Move::CAPTURE)) pos.halfmove = 0;
    else pos.halfmove++;

    pos.sideToMove = them;
    pos.hash ^= Z.side;

    if (inCheck(pos, us)) {
        unmakeMove(pos, m, u);
        return false;
    }
    return true;
}

static void genLegalMoves(Position &pos, vector<Move> &legal, bool capturesOnly) {
    vector<Move> pseudo;
    genPseudoMoves(pos, pseudo, capturesOnly);
    legal.clear();
    legal.reserve(pseudo.size());

    for (auto &m : pseudo) {
        Undo u;
        if (makeMoveChecked(pos, m, u)) {
            unmakeMove(pos, m, u);
            // Fill captured for notation from undo info (pseudo may not have ep capture set properly)
            Move mm = m;
            mm.captured = u.captured;
            legal.push_back(mm);
        }
    }
}

static int eval(const Position &pos) {
    // Positive means good for side to move (we'll convert in search by perspective)
    int scoreW = 0, scoreB = 0;

    int bishopsW = 0, bishopsB = 0;

    for (int sq = 0; sq < 128; ++sq) {
        if (!onBoard(sq)) { sq += 7; continue; }
        char p = pos.b[sq];
        if (p == '.') continue;
        int v = pieceValue(p);
        int i64 = sq64(sq);
        int pst = 0;
        char up = toUpperPiece(p);
        switch (up) {
            case 'P': pst = PST_P[i64]; break;
            case 'N': pst = PST_N[i64]; break;
            case 'B': pst = PST_B[i64]; break;
            case 'R': pst = PST_R[i64]; break;
            case 'Q': pst = PST_Q[i64]; break;
            case 'K': pst = PST_K[i64]; break;
        }

        if (isWhitePiece(p)) {
            scoreW += v + pst;
            if (up == 'B') bishopsW++;
        } else {
            // mirror PST for black
            int mpst = 0;
            switch (up) {
                case 'P': mpst = PST_P[mirror64(i64)]; break;
                case 'N': mpst = PST_N[mirror64(i64)]; break;
                case 'B': mpst = PST_B[mirror64(i64)]; break;
                case 'R': mpst = PST_R[mirror64(i64)]; break;
                case 'Q': mpst = PST_Q[mirror64(i64)]; break;
                case 'K': mpst = PST_K[mirror64(i64)]; break;
            }
            scoreB += v + mpst;
            if (up == 'B') bishopsB++;
        }
    }

    if (bishopsW >= 2) scoreW += 20;
    if (bishopsB >= 2) scoreB += 20;

    int s = scoreW - scoreB;
    return (pos.sideToMove == WHITE) ? s : -s;
}

static inline int clampScoreForTT(int score, int ply) {
    // convert mate scores to be stored relative to root (common technique)
    if (score > MATE - 1000) return score + ply;
    if (score < -MATE + 1000) return score - ply;
    return score;
}
static inline int unclampScoreFromTT(int score, int ply) {
    if (score > MATE - 1000) return score - ply;
    if (score < -MATE + 1000) return score + ply;
    return score;
}

static int qsearch(Position &pos, int alpha, int beta, int ply);

static int search(Position &pos, int depth, int alpha, int beta, int ply,
                  array<array<Move, 2>, MAX_PLY> &killers,
                  array<array<int, 128>, 12> &history);

static int qsearch(Position &pos, int alpha, int beta, int ply) {
    if ((nodesSearched++ & 2047ULL) == 0 && TC.outOfTime()) return eval(pos);

    int stand = eval(pos);
    if (stand >= beta) return beta;
    if (stand > alpha) alpha = stand;

    vector<Move> moves;
    genLegalMoves(pos, moves, true);

    // order captures: MVV-LVA-ish
    auto scoreCap = [&](const Move &m) {
        int capv = (m.flags & Move::CAPTURE) ? pieceValue(m.captured) : 0;
        char moved = pos.b[m.from];
        int mv = pieceValue(moved);
        int promoBonus = (m.flags & Move::PROMO) ? 800 : 0;
        return capv * 10 - mv + promoBonus;
    };
    sort(moves.begin(), moves.end(), [&](const Move &a, const Move &b){ return scoreCap(a) > scoreCap(b); });

    for (auto &m : moves) {
        Undo u;
        if (!makeMoveChecked(pos, m, u)) continue;
        int score = -qsearch(pos, -beta, -alpha, ply + 1);
        unmakeMove(pos, m, u);
        if (TC.outOfTime()) return alpha;

        if (score >= beta) return beta;
        if (score > alpha) alpha = score;
    }

    return alpha;
}

static int search(Position &pos, int depth, int alpha, int beta, int ply,
                  array<array<Move, 2>, MAX_PLY> &killers,
                  array<array<int, 128>, 12> &history) {
    if ((nodesSearched++ & 2047ULL) == 0 && TC.outOfTime()) return eval(pos);

    bool inChk = inCheck(pos, pos.sideToMove);
    if (depth <= 0) return qsearch(pos, alpha, beta, ply);

    // TT probe
    TTEntry *e = TT.probe(pos.hash);
    Move ttMove{};
    if (e->key == pos.hash && e->depth >= depth) {
        int s = unclampScoreFromTT(e->score, ply);
        if (e->flag == 0) return s;
        if (e->flag == 1 && s <= alpha) return s;
        if (e->flag == 2 && s >= beta) return s;
        ttMove = TranspositionTable::decodeMove(e->moveEnc);
    } else if (e->key == pos.hash) {
        ttMove = TranspositionTable::decodeMove(e->moveEnc);
    }

    vector<Move> moves;
    genLegalMoves(pos, moves, false);

    if (moves.empty()) {
        if (inChk) return -MATE + ply;
        return 0; // stalemate
    }

    // Move ordering: TT move, captures, killers, history
    auto sameMove = [](const Move &a, const Move &b) {
        return a.from == b.from && a.to == b.to && a.promo == b.promo && a.flags == b.flags;
    };

    auto moveScore = [&](const Move &m) -> int {
        int sc = 0;
        if (ttMove.to || ttMove.from) {
            if (m.from == ttMove.from && m.to == ttMove.to && m.promo == ttMove.promo && ((m.flags & 31) == (ttMove.flags & 31)))
                sc += 1'000'000;
        }
        if (m.flags & Move::CAPTURE) {
            sc += 100'000 + 10 * pieceValue(m.captured) - pieceValue(pos.b[m.from]);
        } else {
            if (ply < MAX_PLY) {
                if (sameMove(m, killers[ply][0])) sc += 90'000;
                else if (sameMove(m, killers[ply][1])) sc += 80'000;
            }
            char p = pos.b[m.from];
            int idx = pieceIndex(p);
            if (idx >= 0) sc += history[idx][m.to];
        }
        if (m.flags & Move::PROMO) sc += 50'000 + pieceValue((char)m.promo);
        return sc;
    };

    stable_sort(moves.begin(), moves.end(), [&](const Move &a, const Move &b) {
        return moveScore(a) > moveScore(b);
    });

    int bestScore = -INF;
    Move bestMove{};
    int alphaOrig = alpha;

    for (size_t i = 0; i < moves.size(); ++i) {
        Move m = moves[i];
        Undo u;
        if (!makeMoveChecked(pos, m, u)) continue;

        int score;
        // Simple LMR-like: reduce late quiet moves slightly
        bool isQuiet = (m.flags & Move::CAPTURE) == 0 && (m.flags & Move::PROMO) == 0;
        int newDepth = depth - 1;
        int reduction = 0;
        if (isQuiet && !inChk && depth >= 3 && i >= 4) reduction = 1;
        if (reduction) {
            score = -search(pos, newDepth - reduction, -alpha - 1, -alpha, ply + 1, killers, history);
            if (score > alpha && score < beta) {
                score = -search(pos, newDepth, -beta, -alpha, ply + 1, killers, history);
            }
        } else {
            score = -search(pos, newDepth, -beta, -alpha, ply + 1, killers, history);
        }

        unmakeMove(pos, m, u);

        if (TC.outOfTime()) break;

        if (score > bestScore) {
            bestScore = score;
            bestMove = m;
        }
        if (score > alpha) {
            alpha = score;
            // update history for quiet moves
            if (isQuiet) {
                int idx = pieceIndex(pos.b[m.from]); // note: board restored, ok
                if (idx >= 0) history[idx][m.to] += depth * depth;
                if (ply < MAX_PLY) {
                    if (!sameMove(m, killers[ply][0])) {
                        killers[ply][1] = killers[ply][0];
                        killers[ply][0] = m;
                    }
                }
            }
        }
        if (alpha >= beta) {
            // TT store as beta cutoff
            break;
        }
    }

    // store TT
    TTEntry *w = TT.probe(pos.hash);
    uint8_t flag = 0;
    int storeScore = bestScore;
    if (bestScore <= alphaOrig) flag = 1; // alpha
    else if (bestScore >= beta) flag = 2; // beta
    else flag = 0;

    // replace if deeper
    if (w->key != pos.hash || w->depth <= depth) {
        w->key = pos.hash;
        w->depth = (int16_t)depth;
        w->flag = flag;
        w->score = clampScoreForTT(storeScore, ply);
        w->moveEnc = TranspositionTable::encodeMove(bestMove);
    }

    return bestScore;
}

static bool hasAnyLegalMove(Position &pos) {
    vector<Move> pseudo;
    genPseudoMoves(pos, pseudo, false);
    for (auto &m : pseudo) {
        Undo u;
        if (makeMoveChecked(pos, m, u)) {
            unmakeMove(pos, m, u);
            return true;
        }
    }
    return false;
}

static string moveToNotation(Position &pos, const Move &m) {
    // Castling
    if (m.flags & Move::CASTLE) {
        string s = (fileOf(m.to) == 6) ? "O-O" : "O-O-O";

        Undo u;
        if (!makeMoveChecked(pos, m, u)) return s;
        bool chk = inCheck(pos, pos.sideToMove); // after move, sideToMove is opponent
        bool mate = false;
        if (chk && !hasAnyLegalMove(pos)) mate = true;
        unmakeMove(pos, m, u);

        if (mate) s += "#";
        else if (chk) s += "+";
        return s;
    }

    char moved = pos.b[m.from];
    char up = toUpperPiece(moved);
    char pieceLetter = (up == 'P') ? 'p' : up; // 'N','B','R','Q','K'

    string s;
    s += pieceLetter;
    s += sqToStr(m.from);
    if (m.flags & Move::CAPTURE) s += "x";
    s += sqToStr(m.to);

    if (m.flags & Move::EP) s += "ep";
    if (m.flags & Move::PROMO) {
        s += "/";
        s += char(m.promo); // already uppercase
    }

    Undo u;
    if (makeMoveChecked(pos, m, u)) {
        bool chk = inCheck(pos, pos.sideToMove);
        bool mate = false;
        if (chk && !hasAnyLegalMove(pos)) mate = true;
        unmakeMove(pos, m, u);

        if (mate) s += "#";
        else if (chk) s += "+";
    }

    return s;
}

static int parseSquare(const string &s) {
    if (s.size() != 2) return -1;
    char f = s[0], r = s[1];
    if (f < 'a' || f > 'h') return -1;
    if (r < '1' || r > '8') return -1;
    int file = f - 'a';
    int rank = r - '1';
    return rank * 16 + file;
}

static bool parseFEN(const string &fen, Position &pos) {
    pos = Position{};
    for (int i = 0; i < 128; ++i) pos.b[i] = onBoard(i) ? '.' : 0;

    istringstream iss(fen);
    string board, stm, cast, ep;
    if (!(iss >> board >> stm >> cast >> ep)) return false;
    iss >> pos.halfmove >> pos.fullmove;

    int r = 7, f = 0;
    for (char c : board) {
        if (c == '/') { r--; f = 0; continue; }
        if (c >= '1' && c <= '8') {
            f += (c - '0');
            continue;
        }
        if (f < 0 || f > 7 || r < 0 || r > 7) return false;
        int sq = r * 16 + f;
        pos.b[sq] = c;
        if (c == 'K') pos.kingSq[WHITE] = sq;
        if (c == 'k') pos.kingSq[BLACK] = sq;
        f++;
    }

    pos.sideToMove = (stm == "w") ? WHITE : BLACK;

    pos.castling = 0;
    if (cast.find('K') != string::npos) pos.castling |= 1;
    if (cast.find('Q') != string::npos) pos.castling |= 2;
    if (cast.find('k') != string::npos) pos.castling |= 4;
    if (cast.find('q') != string::npos) pos.castling |= 8;

    pos.ep = (ep == "-") ? -1 : parseSquare(ep);

    pos.hash = computeHash(pos);
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    double timeoutSec = 5.0;
    if (!(cin >> timeoutSec)) return 0;
    string fen;
    {
        string dummy;
        getline(cin, dummy); // consume endline
        getline(cin, fen);
    }

    Z.init();
    TT.init(64); // 64 MB TT

    Position pos;
    if (!parseFEN(fen, pos)) {
        cout << "0000\n";
        return 0;
    }

    auto start = chrono::steady_clock::now();
    // keep a small safety margin
    TC.deadline = start + chrono::milliseconds((int)(timeoutSec * 1000.0 * 0.97));

    vector<Move> rootMoves;
    genLegalMoves(pos, rootMoves, false);
    if (rootMoves.empty()) {
        cout << "0000\n";
        return 0;
    }

    // Iterative deepening
    Move best = rootMoves[0];
    int bestScore = -INF;

    array<array<Move, 2>, MAX_PLY> killers{};
    array<array<int, 128>, 12> history{};
    for (auto &h : history) h.fill(0);

    for (int depth = 1; depth <= 64; ++depth) {
        if (TC.outOfTime()) break;
        nodesSearched = 0;

        int alpha = -INF, beta = INF;
        int localBestScore = -INF;
        Move localBest = best;

        // Re-generate legal root moves (for positions where legality might depend on subtle state; safe)
        genLegalMoves(pos, rootMoves, false);

        // Root ordering: TT move first by probing
        Move ttRoot{};
        TTEntry *e = TT.probe(pos.hash);
        if (e->key == pos.hash) ttRoot = TranspositionTable::decodeMove(e->moveEnc);

        auto rootScore = [&](const Move &m)->int{
            int sc = 0;
            if (ttRoot.to || ttRoot.from) {
                if (m.from == ttRoot.from && m.to == ttRoot.to && m.promo == ttRoot.promo && ((m.flags & 31) == (ttRoot.flags & 31)))
                    sc += 1'000'000;
            }
            if (m.flags & Move::CAPTURE) sc += 100'000 + 10 * pieceValue(m.captured) - pieceValue(pos.b[m.from]);
            if (m.flags & Move::PROMO) sc += 50'000 + pieceValue((char)m.promo);
            return sc;
        };
        stable_sort(rootMoves.begin(), rootMoves.end(), [&](const Move &a, const Move &b){
            return rootScore(a) > rootScore(b);
        });

        for (auto &m : rootMoves) {
            Undo u;
            if (!makeMoveChecked(pos, m, u)) continue;
            int score = -search(pos, depth - 1, -beta, -alpha, 1, killers, history);
            unmakeMove(pos, m, u);

            if (TC.outOfTime()) break;

            if (score > localBestScore) {
                localBestScore = score;
                localBest = m;
            }
            if (score > alpha) alpha = score;
        }

        if (!TC.outOfTime() && localBestScore > -INF/2) {
            best = localBest;
            bestScore = localBestScore;
        } else {
            break; // couldn't finish this depth reliably
        }
    }

    cout << moveToNotation(pos, best) << "\n";
    return 0;
}
