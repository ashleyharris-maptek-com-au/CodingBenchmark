```csharp
using System;
using System.Diagnostics;
using System.Text;

public class Program
{
    // -------- Fast scanner --------
    private sealed class FastScanner
    {
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = Console.OpenStandardInput().Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return int.MinValue;
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int x = 0;
            while (c > ' ')
            {
                x = x * 10 + (c - '0');
                c = ReadByte();
            }
            return x * sign;
        }
    }

    // -------- QAP helpers --------

    private static long ComputeCost(int[] flow, int[] dist, int[] p, int n)
    {
        long cost = 0;
        for (int i = 0; i < n; i++)
        {
            int li = p[i];
            int liOff = li * n;
            int fiOff = i * n;
            for (int j = 0; j < n; j++)
            {
                cost += (long)flow[fiOff + j] * dist[liOff + p[j]];
            }
        }
        return cost;
    }

    // Delta for swapping facilities a and b (i.e., swap p[a] and p[b]) in O(n).
    // Cost is sum_{i,j} F[i,j] * D[p[i], p[j]] over all ordered pairs (includes diagonals).
    private static long SwapDelta(int a, int b, int[] p, int[] pRowOff, int[] flow, int[] dist, int n)
    {
        if (a == b) return 0;

        int la = p[a];
        int lb = p[b];
        int laOff = pRowOff[a]; // la * n
        int lbOff = pRowOff[b]; // lb * n

        long delta = 0;

        int aOff = a * n;
        int bOff = b * n;

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;

            int lk = p[k];
            int lkOff = pRowOff[k]; // lk * n

            long f1 = (long)flow[aOff + k] - flow[bOff + k];
            long d1 = (long)dist[lbOff + lk] - dist[laOff + lk];
            delta += f1 * d1;

            long f2 = (long)flow[k * n + a] - flow[k * n + b];
            long d2 = (long)dist[lkOff + lb] - dist[lkOff + la];
            delta += f2 * d2;
        }

        // Diagonal/self terms
        long faa = flow[aOff + a];
        long fbb = flow[bOff + b];
        long dSelf = (long)dist[lbOff + lb] - dist[laOff + la];
        delta += (faa - fbb) * dSelf;

        // Cross terms (a,b) and (b,a)
        long fab = flow[aOff + b];
        long fba = flow[bOff + a];
        long dCross = (long)dist[lbOff + la] - dist[laOff + lb];
        delta += (fab - fba) * dCross;

        return delta;
    }

    private static void ApplySwap(int a, int b, int[] p, int[] pRowOff, int n)
    {
        int tmp = p[a];
        p[a] = p[b];
        p[b] = tmp;

        int tmpOff = pRowOff[a];
        pRowOff[a] = pRowOff[b];
        pRowOff[b] = tmpOff;
    }

    private static int[] BuildInitialPermutation(int n, long[] flowWeight, long[] distWeight)
    {
        int[] fac = new int[n];
        int[] loc = new int[n];
        for (int i = 0; i < n; i++) { fac[i] = i; loc[i] = i; }

        // Facilities: descending total flow; Locations: ascending total distance
        Array.Sort(fac, (i, j) =>
        {
            long di = flowWeight[i], dj = flowWeight[j];
            if (di == dj) return i.CompareTo(j);
            return dj.CompareTo(di);
        });

        Array.Sort(loc, (i, j) =>
        {
            long di = distWeight[i], dj = distWeight[j];
            if (di == dj) return i.CompareTo(j);
            return di.CompareTo(dj);
        });

        int[] p = new int[n];
        for (int r = 0; r < n; r++)
            p[fac[r]] = loc[r];

        return p;
    }

    public static void Main()
    {
        var fs = new FastScanner();

        int n = fs.NextInt();
        if (n <= 0)
            return;

        int nn = n * n;
        int[] flow = new int[nn];
        int[] dist = new int[nn];

        // Weights for a simple constructive heuristic:
        // flowWeight[i] = sum_j F[i,j] + sum_j F[j,i]
        // distWeight[l] = sum_m D[l,m] + sum_m D[m,l]
        long[] flowRow = new long[n];
        long[] flowCol = new long[n];

        for (int i = 0; i < n; i++)
        {
            int iOff = i * n;
            for (int j = 0; j < n; j++)
            {
                int v = fs.NextInt();
                flow[iOff + j] = v;
                flowRow[i] += v;
                flowCol[j] += v;
            }
        }

        long[] distRow = new long[n];
        long[] distCol = new long[n];

        for (int i = 0; i < n; i++)
        {
            int iOff = i * n;
            for (int j = 0; j < n; j++)
            {
                int v = fs.NextInt();
                dist[iOff + j] = v;
                distRow[i] += v;
                distCol[j] += v;
            }
        }

        long[] flowWeight = new long[n];
        long[] distWeight = new long[n];
        for (int i = 0; i < n; i++)
        {
            flowWeight[i] = flowRow[i] + flowCol[i];
            distWeight[i] = distRow[i] + distCol[i];
        }

        // Initial assignment (sorting heuristic)
        int[] p = BuildInitialPermutation(n, flowWeight, distWeight);
        int[] pRowOff = new int[n];
        for (int i = 0; i < n; i++) pRowOff[i] = p[i] * n;

        long curCost = ComputeCost(flow, dist, p, n);

        int[] bestP = new int[n];
        Array.Copy(p, bestP, n);
        long bestCost = curCost;

        // -------- Local search (random improving swaps with occasional kicks) --------
        var sw = Stopwatch.StartNew();

        // Target runtime: up to ~5 minutes as requested (keep a small safety margin).
        double timeLimitSeconds = 290.0;

        var rnd = new Random(1);

        long noImprove = 0;
        long kickAfter = Math.Max(50000, (long)n * 200); // scale with n
        int kickSwaps = (n >= 1000) ? 10 : 20;

        while (sw.Elapsed.TotalSeconds < timeLimitSeconds)
        {
            int a = rnd.Next(n);
            int b = rnd.Next(n - 1);
            if (b >= a) b++;

            long delta = SwapDelta(a, b, p, pRowOff, flow, dist, n);

            if (delta < 0)
            {
                ApplySwap(a, b, p, pRowOff, n);
                curCost += delta;
                noImprove = 0;

                if (curCost < bestCost)
                {
                    bestCost = curCost;
                    Array.Copy(p, bestP, n);
                }
            }
            else
            {
                noImprove++;

                if (noImprove >= kickAfter)
                {
                    // Kick: do a few random swaps (even if worse) to escape local minima
                    for (int t = 0; t < kickSwaps; t++)
                    {
                        int x = rnd.Next(n);
                        int y = rnd.Next(n - 1);
                        if (y >= x) y++;

                        long d = SwapDelta(x, y, p, pRowOff, flow, dist, n);
                        ApplySwap(x, y, p, pRowOff, n);
                        curCost += d;
                    }

                    // If we drifted too far from best, reset to best.
                    // (Keeps search from wandering excessively.)
                    if (curCost > bestCost + Math.Abs(bestCost) / 20 + 1) // ~5% band
                    {
                        Array.Copy(bestP, p, n);
                        for (int i = 0; i < n; i++) pRowOff[i] = p[i] * n;
                        curCost = bestCost;
                    }

                    noImprove = 0;
                }
            }
        }

        // -------- Output --------
        var sb = new StringBuilder();
        sb.AppendLine(bestCost.ToString());

        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(bestP[i]);
        }
        sb.AppendLine();

        Console.Write(sb.ToString());
    }
}
```
