using System;
using System.IO;
using System.Globalization;
using System.Collections.Generic;

public static class Program
{
    // Practical safeguard: printing tens/hundreds of millions of placements will time out in any judge.
    // This cap keeps runtime bounded even for "ludicrous" container/shape combinations.
    private const long MaxPlacementsToPrint = 3_000_000;

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        int V = fs.NextInt();
        int F = fs.NextInt();

        var verts = new Vec3d[V];
        for (int i = 0; i < V; i++)
            verts[i] = new Vec3d(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());

        // Faces are not needed for conservative AABB-based packing, but must be consumed.
        for (int i = 0; i < F; i++)
        {
            fs.NextInt(); fs.NextInt(); fs.NextInt();
        }

        double W = fs.NextDouble();
        double H = fs.NextDouble();
        double D = fs.NextDouble();

        double maxDim = Math.Max(W, Math.Max(H, D));

        // Strategy selection (adaptive):
        // - Always try all 24 axis-permutation right-handed rotations (fast, robust).
        // - For smaller problems, sample additional random orientations to possibly get tighter AABBs.
        var orientations = new List<QuatD>(256);
        orientations.AddRange(Generate24RightHandedAxisRotations());

        int randomSamples = 0;
        if (V <= 5000)
        {
            if (maxDim <= 30) randomSamples = 600;       // trivial/small: explore more
            else if (maxDim <= 80) randomSamples = 200;  // medium: some exploration
            else randomSamples = 0;                      // large/extreme: keep it deterministic+fast
        }

        if (randomSamples > 0)
        {
            var rng = new SplitMix64(123456789);
            for (int i = 0; i < randomSamples; i++)
                orientations.Add(RandomUniformQuaternion(ref rng));
        }

        // Evaluate best grid packing based on AABB of rotated polyhedron (conservative, always valid).
        Best best = default;
        best.Total = -1;

        for (int oi = 0; oi < orientations.Count; oi++)
        {
            var q = orientations[oi];
            ComputeAabbRotated(verts, q, out Vec3d min, out Vec3d max);

            double dx = max.X - min.X;
            double dy = max.Y - min.Y;
            double dz = max.Z - min.Z;

            const double eps = 1e-12;
            if (dx <= eps || dy <= eps || dz <= eps) continue;
            if (dx > W + 1e-9 || dy > H + 1e-9 || dz > D + 1e-9) continue;

            long nx = (long)Math.Floor((W + 1e-12) / dx);
            long ny = (long)Math.Floor((H + 1e-12) / dy);
            long nz = (long)Math.Floor((D + 1e-12) / dz);
            if (nx <= 0 || ny <= 0 || nz <= 0) continue;

            // Avoid overflow
            long total;
            try
            {
                checked { total = nx * ny * nz; }
            }
            catch
            {
                total = long.MaxValue / 4;
            }

            // Primary: maximize count.
            // Secondary: choose tighter AABB volume (often helps for future constraints).
            double vol = dx * dy * dz;

            if (total > best.Total || (total == best.Total && vol < best.AabbVol))
            {
                best = new Best
                {
                    Q = q,
                    Min = min,
                    Dx = dx, Dy = dy, Dz = dz,
                    Nx = nx, Ny = ny, Nz = nz,
                    Total = total,
                    AabbVol = vol
                };
            }
        }

        if (best.Total <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        long toPrint = best.Total;
        if (toPrint > MaxPlacementsToPrint) toPrint = MaxPlacementsToPrint;

        var inv = CultureInfo.InvariantCulture;
        using var sw = new StreamWriter(Console.OpenStandardOutput(), System.Text.Encoding.ASCII, bufferSize: 1 << 20);

        sw.WriteLine(toPrint.ToString(inv));

        // Stream placements directly (no memory blow-ups).
        long printed = 0;
        for (long iz = 0; iz < best.Nz && printed < toPrint; iz++)
        {
            double baseZ = iz * best.Dz;
            for (long iy = 0; iy < best.Ny && printed < toPrint; iy++)
            {
                double baseY = iy * best.Dy;
                for (long ix = 0; ix < best.Nx && printed < toPrint; ix++)
                {
                    double baseX = ix * best.Dx;

                    // Place the rotated polyhedron so its rotated AABB min corner sits at (baseX, baseY, baseZ).
                    // If rotated vertex is v', and AABB min is min, then translation t = base - min gives:
                    // v'+t in [base, base+dim] ⊆ container.
                    double tx = baseX - best.Min.X;
                    double ty = baseY - best.Min.Y;
                    double tz = baseZ - best.Min.Z;

                    // Output: tx ty tz qw qx qy qz
                    sw.Write(tx.ToString("G17", inv)); sw.Write(' ');
                    sw.Write(ty.ToString("G17", inv)); sw.Write(' ');
                    sw.Write(tz.ToString("G17", inv)); sw.Write(' ');
                    sw.Write(best.Q.W.ToString("G17", inv)); sw.Write(' ');
                    sw.Write(best.Q.X.ToString("G17", inv)); sw.Write(' ');
                    sw.Write(best.Q.Y.ToString("G17", inv)); sw.Write(' ');
                    sw.WriteLine(best.Q.Z.ToString("G17", inv));

                    printed++;
                }
            }
        }
    }

    private struct Best
    {
        public QuatD Q;
        public Vec3d Min;
        public double Dx, Dy, Dz;
        public long Nx, Ny, Nz;
        public long Total;
        public double AabbVol;
    }

    private static void ComputeAabbRotated(Vec3d[] verts, QuatD q, out Vec3d min, out Vec3d max)
    {
        double minX = double.PositiveInfinity, minY = double.PositiveInfinity, minZ = double.PositiveInfinity;
        double maxX = double.NegativeInfinity, maxY = double.NegativeInfinity, maxZ = double.NegativeInfinity;

        for (int i = 0; i < verts.Length; i++)
        {
            Vec3d r = Rotate(verts[i], q);
            if (r.X < minX) minX = r.X;
            if (r.Y < minY) minY = r.Y;
            if (r.Z < minZ) minZ = r.Z;
            if (r.X > maxX) maxX = r.X;
            if (r.Y > maxY) maxY = r.Y;
            if (r.Z > maxZ) maxZ = r.Z;
        }

        min = new Vec3d(minX, minY, minZ);
        max = new Vec3d(maxX, maxY, maxZ);
    }

    // Quaternion-vector rotation: v' = v + w*(2*cross(qv, v)) + cross(qv, 2*cross(qv, v))
    private static Vec3d Rotate(in Vec3d v, in QuatD q)
    {
        double qx = q.X, qy = q.Y, qz = q.Z, qw = q.W;

        // t = 2 * cross(qv, v)
        double tx = 2.0 * (qy * v.Z - qz * v.Y);
        double ty = 2.0 * (qz * v.X - qx * v.Z);
        double tz = 2.0 * (qx * v.Y - qy * v.X);

        // v' = v + qw*t + cross(qv, t)
        double cx = qy * tz - qz * ty;
        double cy = qz * tx - qx * tz;
        double cz = qx * ty - qy * tx;

        return new Vec3d(
            v.X + qw * tx + cx,
            v.Y + qw * ty + cy,
            v.Z + qw * tz + cz
        );
    }

    // 24 right-handed rotations that map axes to axes (cube symmetry group).
    private static List<QuatD> Generate24RightHandedAxisRotations()
    {
        var result = new List<QuatD>(24);

        int[][] perms = new[]
        {
            new[]{0,1,2}, new[]{0,2,1},
            new[]{1,0,2}, new[]{1,2,0},
            new[]{2,0,1}, new[]{2,1,0}
        };

        int Parity(int[] p)
        {
            // +1 for even, -1 for odd
            int inv = 0;
            for (int i = 0; i < 3; i++)
                for (int j = i + 1; j < 3; j++)
                    if (p[i] > p[j]) inv++;
            return (inv % 2 == 0) ? +1 : -1;
        }

        int[] signs = new[] { -1, +1 };

        foreach (var p in perms)
        {
            int permParity = Parity(p);
            foreach (int sx in signs)
            foreach (int sy in signs)
            foreach (int sz in signs)
            {
                int det = sx * sy * sz * permParity;
                if (det != +1) continue; // right-handed only

                // Rotation matrix R such that:
                // v' = [ sx * v[p0], sy * v[p1], sz * v[p2] ]
                double m00 = 0, m01 = 0, m02 = 0;
                double m10 = 0, m11 = 0, m12 = 0;
                double m20 = 0, m21 = 0, m22 = 0;

                // Row 0 uses p[0]
                if (p[0] == 0) m00 = sx; else if (p[0] == 1) m01 = sx; else m02 = sx;
                if (p[1] == 0) m10 = sy; else if (p[1] == 1) m11 = sy; else m12 = sy;
                if (p[2] == 0) m20 = sz; else if (p[2] == 1) m21 = sz; else m22 = sz;

                var q = QuaternionFromRotationMatrix(
                    m00, m01, m02,
                    m10, m11, m12,
                    m20, m21, m22);

                // Normalize (numerical safety)
                q = q.Normalized();
                // Ensure unique-ish (avoid duplicates from numeric drift)
                result.Add(q);
            }
        }

        // Deduplicate very close quaternions (q and -q are same rotation)
        var unique = new List<QuatD>(24);
        foreach (var q in result)
        {
            bool dup = false;
            foreach (var u in unique)
            {
                if (QuatD.SameRotation(q, u, 1e-12))
                {
                    dup = true;
                    break;
                }
            }
            if (!dup) unique.Add(q);
        }

        return unique;
    }

    private static QuatD QuaternionFromRotationMatrix(
        double m00, double m01, double m02,
        double m10, double m11, double m12,
        double m20, double m21, double m22)
    {
        // Standard conversion
        double trace = m00 + m11 + m22;
        double w, x, y, z;

        if (trace > 0.0)
        {
            double s = Math.Sqrt(trace + 1.0) * 2.0; // s = 4w
            w = 0.25 * s;
            x = (m21 - m12) / s;
            y = (m02 - m20) / s;
            z = (m10 - m01) / s;
        }
        else if (m00 > m11 && m00 > m22)
        {
            double s = Math.Sqrt(1.0 + m00 - m11 - m22) * 2.0; // s = 4x
            w = (m21 - m12) / s;
            x = 0.25 * s;
            y = (m01 + m10) / s;
            z = (m02 + m20) / s;
        }
        else if (m11 > m22)
        {
            double s = Math.Sqrt(1.0 + m11 - m00 - m22) * 2.0; // s = 4y
            w = (m02 - m20) / s;
            x = (m01 + m10) / s;
            y = 0.25 * s;
            z = (m12 + m21) / s;
        }
        else
        {
            double s = Math.Sqrt(1.0 + m22 - m00 - m11) * 2.0; // s = 4z
            w = (m10 - m01) / s;
            x = (m02 + m20) / s;
            y = (m12 + m21) / s;
            z = 0.25 * s;
        }

        return new QuatD(w, x, y, z);
    }

    private static QuatD RandomUniformQuaternion(ref SplitMix64 rng)
    {
        // Uniform random rotation (Shoemake)
        double u1 = rng.NextDouble01();
        double u2 = rng.NextDouble01();
        double u3 = rng.NextDouble01();

        double a = Math.Sqrt(1.0 - u1);
        double b = Math.Sqrt(u1);
        double t1 = 2.0 * Math.PI * u2;
        double t2 = 2.0 * Math.PI * u3;

        double x = a * Math.Sin(t1);
        double y = a * Math.Cos(t1);
        double z = b * Math.Sin(t2);
        double w = b * Math.Cos(t2);

        return new QuatD(w, x, y, z).Normalized();
    }

    // ---------- Math structs ----------

    private readonly struct Vec3d
    {
        public readonly double X, Y, Z;
        public Vec3d(double x, double y, double z) { X = x; Y = y; Z = z; }
    }

    private readonly struct QuatD
    {
        public readonly double W, X, Y, Z;
        public QuatD(double w, double x, double y, double z) { W = w; X = x; Y = y; Z = z; }

        public QuatD Normalized()
        {
            double n = Math.Sqrt(W * W + X * X + Y * Y + Z * Z);
            if (n <= 0) return new QuatD(1, 0, 0, 0);
            double inv = 1.0 / n;
            double w = W * inv, x = X * inv, y = Y * inv, z = Z * inv;

            // Canonicalize sign (q and -q represent the same rotation).
            if (w < 0) { w = -w; x = -x; y = -y; z = -z; }
            return new QuatD(w, x, y, z);
        }

        public static bool SameRotation(in QuatD a, in QuatD b, double eps)
        {
            // Compare both a~b and a~-b
            double dw = a.W - b.W, dx = a.X - b.X, dy = a.Y - b.Y, dz = a.Z - b.Z;
            double d1 = dw * dw + dx * dx + dy * dy + dz * dz;

            double dw2 = a.W + b.W, dx2 = a.X + b.X, dy2 = a.Y + b.Y, dz2 = a.Z + b.Z;
            double d2 = dw2 * dw2 + dx2 * dx2 + dy2 * dy2 + dz2 * dz2;

            return d1 <= eps * eps || d2 <= eps * eps;
        }
    }

    // ---------- Deterministic RNG ----------

    private struct SplitMix64
    {
        private ulong x;
        public SplitMix64(ulong seed) { x = seed; }

        public ulong NextU64()
        {
            ulong z = (x += 0x9E3779B97F4A7C15UL);
            z = (z ^ (z >> 30)) * 0xBF58476D1CE4E5B9UL;
            z = (z ^ (z >> 27)) * 0x94D049BB133111EBUL;
            return z ^ (z >> 31);
        }

        public double NextDouble01()
        {
            // 53-bit mantissa
            ulong r = NextU64();
            return ((r >> 11) * (1.0 / (1UL << 53)));
        }
    }

    // ---------- Fast input ----------

    private sealed class FastScanner
    {
        private readonly Stream _s;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream s) { _s = s; }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }

        private void SkipBlanks()
        {
            int c;
            while ((c = Read()) != -1)
            {
                if (c > 32) { _ptr--; return; }
            }
        }

        public int NextInt()
        {
            SkipBlanks();
            int c = Read();
            int sign = 1;
            if (c == '-') { sign = -1; c = Read(); }
            int val = 0;
            while (c > 32 && c != -1)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }

        public double NextDouble()
        {
            SkipBlanks();
            int c = Read();
            int sign = 1;
            if (c == '-') { sign = -1; c = Read(); }

            double val = 0;
            while (c >= '0' && c <= '9')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }

            if (c == '.')
            {
                double mul = 1.0;
                c = Read();
                while (c >= '0' && c <= '9')
                {
                    mul *= 0.1;
                    val += (c - '0') * mul;
                    c = Read();
                }
            }

            if (c == 'e' || c == 'E')
            {
                c = Read();
                int esign = 1;
                if (c == '-') { esign = -1; c = Read(); }
                else if (c == '+') { c = Read(); }

                int exp = 0;
                while (c >= '0' && c <= '9')
                {
                    exp = exp * 10 + (c - '0');
                    c = Read();
                }
                val *= Math.Pow(10.0, esign * exp);
            }

            return val * sign;
        }
    }
}
