```rust
use std::io::{self, Read, Write};

struct Scanner {
    buf: Vec<u8>,
    i: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, i: 0 }
    }

    fn next_usize(&mut self) -> usize {
        while self.i < self.buf.len() && self.buf[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
        let mut x: usize = 0;
        while self.i < self.buf.len() && !self.buf[self.i].is_ascii_whitespace() {
            x = x * 10 + (self.buf[self.i] - b'0') as usize;
            self.i += 1;
        }
        x
    }
}

fn main() {
    let mut sc = Scanner::new();
    let n = sc.next_usize();
    let m = sc.next_usize();

    // Greedy maximal matching (streaming):
    // - If both endpoints are currently unmatched, match them.
    // - Output all matched vertices. This is a valid vertex cover and is a 2-approximation.
    let mut in_cover = vec![0u8; n];

    for _ in 0..m {
        let u = sc.next_usize();
        let v = sc.next_usize();

        if u == v {
            // Self-loop must be covered by u.
            in_cover[u] = 1;
        } else if in_cover[u] == 0 && in_cover[v] == 0 {
            in_cover[u] = 1;
            in_cover[v] = 1;
        }
    }

    let mut count: usize = 0;
    for &b in &in_cover {
        count += b as usize;
    }

    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());

    writeln!(out, "{count}").unwrap();
    if count > 0 {
        let mut first = true;
        for (i, &b) in in_cover.iter().enumerate() {
            if b != 0 {
                if !first {
                    write!(out, " ").unwrap();
                }
                first = false;
                write!(out, "{i}").unwrap();
            }
        }
    }
    writeln!(out).unwrap();
}
```
