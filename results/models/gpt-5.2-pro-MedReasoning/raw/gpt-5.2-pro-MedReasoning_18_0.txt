// Minimum Edge Cut (Global Min-Cut) solver for undirected unweighted graphs.
// Adaptive strategy:
//  1) Stream input; keep edges either in RAM (if feasible) or in a temporary binary file.
//  2) Quick win: build a small 2-forest certificate and check for a bridge => min-cut = 1.
//  3) Otherwise, build a (k-forest) sparse certificate with k = minDegree + 1 (edge-disjoint forests).
//     This preserves all cuts of size < k, hence preserves the true global min cut.
//  4) Run exact Stoer–Wagner on the sparse certificate (in memory).
//  5) Re-scan original edges to count/print the cut edges.
//
// Notes:
//  - Uses only standard C/C++ + MSVC-supported headers; no non-standard headers.
//  - For extremely large graphs, the algorithm relies on sparsification to become tractable.
//  - If the minimum degree is huge, any exact algorithm can become infeasible in time; this implementation
//    still remains streaming-safe (won't require the full edge list in RAM).

#include <cstdint>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <vector>
#include <algorithm>
#include <unordered_map>
#include <queue>
#include <limits>
#include <iostream>

struct FastScanner {
    static constexpr size_t BUFSZ = 1 << 20;
    unsigned char buf[BUFSZ];
    size_t idx = 0, size = 0;

    inline unsigned char readByte() {
        if (idx >= size) {
            size = std::fread(buf, 1, BUFSZ, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    template <class T>
    bool readUnsigned(T &out) {
        out = 0;
        unsigned char c;
        do {
            c = readByte();
            if (!c) return false;
        } while (c <= ' ');

        // assumes non-negative input
        while (c > ' ') {
            out = out * 10 + (c - '0');
            c = readByte();
            if (!c) break;
        }
        return true;
    }
};

struct FastOutput {
    static constexpr size_t BUFSZ = 1 << 20;
    char buf[BUFSZ];
    size_t idx = 0;

    ~FastOutput() { flush(); }

    inline void flush() {
        if (idx) {
            std::fwrite(buf, 1, idx, stdout);
            idx = 0;
        }
    }

    inline void pushChar(char c) {
        if (idx >= BUFSZ) flush();
        buf[idx++] = c;
    }

    inline void writeUInt64(uint64_t x) {
        char s[32];
        int n = 0;
        do {
            s[n++] = char('0' + (x % 10));
            x /= 10;
        } while (x);
        while (n--) pushChar(s[n]);
    }

    inline void writeUInt32(uint32_t x) { writeUInt64(x); }

    inline void writeStr(const char* s) {
        while (*s) pushChar(*s++);
    }
};

struct Edge {
    uint32_t u, v;
};

struct DSU {
    std::vector<uint32_t> p;
    std::vector<uint32_t> sz;
    uint32_t comps;

    explicit DSU(uint32_t n) : p(n), sz(n, 1), comps(n) {
        for (uint32_t i = 0; i < n; ++i) p[i] = i;
    }
    uint32_t find(uint32_t a) {
        while (p[a] != a) {
            p[a] = p[p[a]];
            a = p[a];
        }
        return a;
    }
    bool unite(uint32_t a, uint32_t b) {
        a = find(a); b = find(b);
        if (a == b) return false;
        if (sz[a] < sz[b]) std::swap(a, b);
        p[b] = a;
        sz[a] += sz[b];
        --comps;
        return true;
    }
};

struct EdgeStore {
    bool inMemory = false;
    std::vector<Edge> mem;
    std::FILE* file = nullptr;
    uint64_t M = 0;

    ~EdgeStore() {
        if (file) std::fclose(file);
    }
};

static bool findAnyBridge(
    uint32_t N,
    const std::vector<Edge>& edges,
    Edge &bridgeOut
) {
    // Iterative Tarjan bridge-finding.
    std::vector<std::vector<std::pair<uint32_t, uint32_t>>> adj(N);
    adj.shrink_to_fit();
    adj.reserve(N);

    for (uint32_t id = 0; id < (uint32_t)edges.size(); ++id) {
        auto [u, v] = edges[id];
        if (u == v) continue;
        adj[u].push_back({v, id});
        adj[v].push_back({u, id});
    }

    std::vector<int> tin(N, -1), low(N, 0);
    int timer = 0;

    struct Frame {
        uint32_t v;
        uint32_t parent;
        uint32_t parentEdge;
        uint32_t it;
    };

    std::vector<Frame> st;
    st.reserve(N);

    for (uint32_t root = 0; root < N; ++root) {
        if (tin[root] != -1) continue;

        st.push_back(Frame{root, root, std::numeric_limits<uint32_t>::max(), 0});
        tin[root] = low[root] = timer++;

        while (!st.empty()) {
            Frame &f = st.back();
            uint32_t v = f.v;

            if (f.it == adj[v].size()) {
                // finish vertex
                if (v != f.parent) {
                    // update parent low
                    uint32_t p = f.parent;
                    low[p] = std::min(low[p], low[v]);

                    // bridge test
                    if (low[v] > tin[p]) {
                        bridgeOut = edges[f.parentEdge];
                        return true;
                    }
                }
                st.pop_back();
                continue;
            }

            auto [to, eid] = adj[v][f.it++];
            if (eid == f.parentEdge) continue;

            if (tin[to] != -1) {
                // back edge
                low[v] = std::min(low[v], tin[to]);
            } else {
                // tree edge
                st.push_back(Frame{to, v, eid, 0});
                tin[to] = low[to] = timer++;
            }
        }
    }
    return false;
}

static std::vector<Edge> buildKForestCertificate(
    uint32_t N,
    uint64_t M,
    const EdgeStore& store,
    uint64_t k
) {
    // Builds union of k edge-disjoint spanning forests by repeated "maximal forest" extraction.
    // Uses skip list of removed edge IDs (sorted) to avoid rewriting the edge list.
    std::vector<uint64_t> removed; // sorted edge IDs already selected in earlier passes
    removed.reserve((size_t)std::min<uint64_t>(M, k * (uint64_t)std::max<uint32_t>(1, N - 1)));

    std::vector<Edge> cert;
    cert.reserve((size_t)std::min<uint64_t>(k * (uint64_t)std::max<uint32_t>(1, N - 1), M));

    std::vector<uint64_t> newRemoved;
    newRemoved.reserve(N ? (N - 1) : 0);

    auto mergeRemoved = [&](const std::vector<uint64_t>& a, const std::vector<uint64_t>& b) {
        std::vector<uint64_t> merged;
        merged.reserve(a.size() + b.size());
        std::merge(a.begin(), a.end(), b.begin(), b.end(), std::back_inserter(merged));
        return merged;
    };

    for (uint64_t pass = 0; pass < k; ++pass) {
        DSU dsu(N);
        newRemoved.clear();

        uint64_t skipIdx = 0;

        if (store.inMemory) {
            for (uint64_t id = 0; id < M; ++id) {
                if (skipIdx < removed.size() && removed[skipIdx] == id) { ++skipIdx; continue; }
                const Edge &e = store.mem[(size_t)id];
                if (e.u == e.v) continue;
                if (dsu.unite(e.u, e.v)) {
                    newRemoved.push_back(id);
                    cert.push_back(e);
                    if (dsu.comps == 1) break; // spanning tree found
                }
            }
        } else {
            std::rewind(store.file);
            static constexpr size_t BUFE = 1 << 20;
            std::vector<Edge> buf;
            buf.resize(BUFE);

            uint64_t id = 0;
            while (id < M) {
                size_t want = (size_t)std::min<uint64_t>(BUFE, M - id);
                size_t got = std::fread(buf.data(), sizeof(Edge), want, store.file);
                if (got == 0) break;

                for (size_t i = 0; i < got; ++i, ++id) {
                    while (skipIdx < removed.size() && removed[skipIdx] < id) ++skipIdx;
                    if (skipIdx < removed.size() && removed[skipIdx] == id) { ++skipIdx; continue; }
                    const Edge &e = buf[i];
                    if (e.u == e.v) continue;
                    if (dsu.unite(e.u, e.v)) {
                        newRemoved.push_back(id);
                        cert.push_back(e);
                        if (dsu.comps == 1) goto forest_done;
                    }
                }
            }
        forest_done:
            (void)0;
        }

        if (newRemoved.empty()) break; // no more edges can be added in further forests
        // newRemoved is naturally sorted by increasing edge ID.
        removed = mergeRemoved(removed, newRemoved);
        if (removed.size() == (size_t)M) break;
    }

    return cert;
}

struct MinCutResult {
    uint64_t weight = 0;
    std::vector<uint32_t> side; // 1 if in S, 0 if in T
};

static MinCutResult stoerWagnerMinCut(
    uint32_t N,
    const std::vector<Edge>& edges
) {
    // Exact global min cut for undirected graphs, supports parallel edges via weights.
    // Uses adjacency hash maps and priority queue selection.

    MinCutResult res;
    if (N == 0) {
        res.weight = 0;
        return res;
    }
    if (N == 1) {
        res.weight = 0;
        res.side.assign(1, 0);
        return res;
    }

    // Build weighted adjacency
    std::vector<uint32_t> deg(N, 0);
    for (auto &e : edges) {
        if (e.u == e.v) continue;
        ++deg[e.u];
        ++deg[e.v];
    }

    std::vector<std::unordered_map<uint32_t, uint64_t>> g(N);
    for (uint32_t i = 0; i < N; ++i) {
        // reserve with slack; unordered_map overhead is high but helps avoid rehash
        g[i].reserve((size_t)deg[i] * 2 + 1);
    }

    for (auto &e : edges) {
        uint32_t u = e.u, v = e.v;
        if (u == v) continue;
        g[u][v] += 1;
        g[v][u] += 1;
    }

    // Active supernodes
    std::vector<uint32_t> active(N), pos(N);
    std::vector<unsigned char> alive(N, 1);
    for (uint32_t i = 0; i < N; ++i) { active[i] = i; pos[i] = i; }

    // Linked-list membership for each supernode (original vertices)
    std::vector<uint32_t> head(N), tail(N), nxt(N, std::numeric_limits<uint32_t>::max());
    for (uint32_t i = 0; i < N; ++i) head[i] = tail[i] = i;

    std::vector<uint64_t> w(N, 0);
    std::vector<uint32_t> stampW(N, 0), stampAdded(N, 0);
    uint32_t curStamp = 1;

    auto isAdded = [&](uint32_t v) -> bool { return stampAdded[v] == curStamp; };
    auto setAdded = [&](uint32_t v) { stampAdded[v] = curStamp; };
    auto getW = [&](uint32_t v) -> uint64_t { return (stampW[v] == curStamp) ? w[v] : 0ULL; };
    auto addW = [&](uint32_t v, uint64_t delta) {
        if (stampW[v] != curStamp) { stampW[v] = curStamp; w[v] = 0; }
        w[v] += delta;
    };

    uint64_t best = std::numeric_limits<uint64_t>::max();
    std::vector<uint32_t> bestMembers;

    auto copyMembers = [&](uint32_t super) {
        bestMembers.clear();
        uint32_t x = head[super];
        while (x != std::numeric_limits<uint32_t>::max()) {
            bestMembers.push_back(x);
            x = nxt[x];
        }
    };

    auto contract = [&](uint32_t t, uint32_t s) {
        // Merge t into s; update membership list
        if (head[t] != std::numeric_limits<uint32_t>::max()) {
            nxt[tail[s]] = head[t];
            tail[s] = tail[t];
            head[t] = tail[t] = std::numeric_limits<uint32_t>::max();
        }

        // Update adjacency: move all edges of t into s
        for (auto &kv : g[t]) {
            uint32_t u = kv.first;
            uint64_t wt = kv.second;
            if (!alive[u] || u == s) continue;

            g[s][u] += wt;
            g[u][s] += wt;
            g[u].erase(t);
        }
        g[s].erase(t);
        g[t].clear();

        // Remove t from active
        alive[t] = 0;
        uint32_t idx = pos[t];
        uint32_t last = active.back();
        active[idx] = last;
        pos[last] = idx;
        active.pop_back();
    };

    while (active.size() > 1) {
        // bump stamp; handle overflow by full reset
        ++curStamp;
        if (curStamp == 0) {
            curStamp = 1;
            std::fill(stampW.begin(), stampW.end(), 0);
            std::fill(stampAdded.begin(), stampAdded.end(), 0);
        }

        uint32_t start = active[0];
        setAdded(start);

        using Item = std::pair<uint64_t, uint32_t>; // (weight, vertex)
        std::priority_queue<Item> pq;

        // initialize weights from start
        for (auto &kv : g[start]) {
            uint32_t u = kv.first;
            if (!alive[u] || isAdded(u)) continue;
            addW(u, kv.second);
            pq.push({getW(u), u});
        }

        uint32_t prev = start;

        for (size_t iter = 1; iter < active.size(); ++iter) {
            uint32_t v = std::numeric_limits<uint32_t>::max();
            uint64_t vw = 0;

            while (!pq.empty()) {
                auto [cw, cv] = pq.top();
                pq.pop();
                if (!alive[cv] || isAdded(cv)) continue;
                uint64_t realW = getW(cv);
                if (realW != cw) continue;
                v = cv;
                vw = realW;
                break;
            }

            // If pq got empty due to stale entries, fall back to linear search (rare).
            if (v == std::numeric_limits<uint32_t>::max()) {
                uint64_t bestW = 0;
                uint32_t bestV = std::numeric_limits<uint32_t>::max();
                for (uint32_t u : active) {
                    if (isAdded(u)) continue;
                    uint64_t wu = getW(u);
                    if (bestV == std::numeric_limits<uint32_t>::max() || wu > bestW) {
                        bestW = wu;
                        bestV = u;
                    }
                }
                v = bestV;
                vw = bestW;
            }

            if (iter + 1 == active.size()) {
                // v is the last added (t), prev is second last (s)
                if (vw < best) {
                    best = vw;
                    copyMembers(v);
                }
                // contract v into prev
                contract(v, prev);
                break;
            } else {
                setAdded(v);
                // update weights
                for (auto &kv : g[v]) {
                    uint32_t u = kv.first;
                    if (!alive[u] || isAdded(u)) continue;
                    addW(u, kv.second);
                    pq.push({getW(u), u});
                }
                prev = v;
            }
        }
    }

    // Build side vector of original vertices from bestMembers
    res.weight = (best == std::numeric_limits<uint64_t>::max()) ? 0 : best;
    res.side.assign(N, 0);
    for (uint32_t x : bestMembers) res.side[x] = 1;
    return res;
}

static uint64_t countCutEdges(
    const EdgeStore& store,
    uint32_t N,
    const std::vector<uint32_t>& side
) {
    (void)N;
    uint64_t cnt = 0;
    if (store.inMemory) {
        for (const auto &e : store.mem) {
            if (e.u == e.v) continue;
            if ((side[e.u] ^ side[e.v]) != 0) ++cnt;
        }
    } else {
        std::rewind(store.file);
        static constexpr size_t BUFE = 1 << 20;
        std::vector<Edge> buf(BUFE);

        uint64_t readTotal = 0;
        while (readTotal < store.M) {
            size_t want = (size_t)std::min<uint64_t>(BUFE, store.M - readTotal);
            size_t got = std::fread(buf.data(), sizeof(Edge), want, store.file);
            if (got == 0) break;
            for (size_t i = 0; i < got; ++i) {
                const auto &e = buf[i];
                if (e.u == e.v) continue;
                if ((side[e.u] ^ side[e.v]) != 0) ++cnt;
            }
            readTotal += got;
        }
    }
    return cnt;
}

static void emitCutEdges(
    const EdgeStore& store,
    const std::vector<uint32_t>& side,
    FastOutput &out
) {
    if (store.inMemory) {
        for (const auto &e : store.mem) {
            if (e.u == e.v) continue;
            if ((side[e.u] ^ side[e.v]) != 0) {
                out.writeUInt32(e.u);
                out.pushChar(' ');
                out.writeUInt32(e.v);
                out.pushChar('\n');
            }
        }
    } else {
        std::rewind(store.file);
        static constexpr size_t BUFE = 1 << 20;
        std::vector<Edge> buf(BUFE);

        uint64_t readTotal = 0;
        while (readTotal < store.M) {
            size_t want = (size_t)std::min<uint64_t>(BUFE, store.M - readTotal);
            size_t got = std::fread(buf.data(), sizeof(Edge), want, store.file);
            if (got == 0) break;
            for (size_t i = 0; i < got; ++i) {
                const auto &e = buf[i];
                if (e.u == e.v) continue;
                if ((side[e.u] ^ side[e.v]) != 0) {
                    out.writeUInt32(e.u);
                    out.pushChar(' ');
                    out.writeUInt32(e.v);
                    out.pushChar('\n');
                }
            }
            readTotal += got;
        }
    }
}

int main() {
    FastScanner fs;
    FastOutput out;

    uint64_t N64 = 0, M64 = 0;
    if (!fs.readUnsigned(N64) || !fs.readUnsigned(M64)) return 0;

    uint32_t N = (N64 > std::numeric_limits<uint32_t>::max())
        ? std::numeric_limits<uint32_t>::max()
        : (uint32_t)N64;
    uint64_t M = M64;

    if (N64 != (uint64_t)N) {
        // N too large to address with 32-bit vertex IDs; cannot proceed meaningfully.
        // Output a valid "cut" of size 0 (graph assumed connected in statement, so this would fail tests,
        // but prevents UB/crash).
        out.writeStr("0\n");
        out.flush();
        return 0;
    }

    if (N <= 1) {
        out.writeStr("0\n");
        out.flush();
        return 0;
    }
    if (M == 0) {
        out.writeStr("0\n");
        out.flush();
        return 0;
    }

    // Decide storage mode: try in-memory up to a soft threshold; otherwise spool to tmpfile.
    EdgeStore store;
    store.M = M;

    // Degree array (needed for min-degree upper bound).
    std::vector<uint32_t> deg;
    try {
        deg.assign(N, 0);
    } catch (...) {
        out.writeStr("0\n");
        out.flush();
        return 0;
    }

    bool canTryMemory = (M <= 80'000'000ULL); // ~640MB for Edge vector; conservative default
    if (canTryMemory) {
        try {
            store.mem.reserve((size_t)M);
            store.inMemory = true;
        } catch (...) {
            store.inMemory = false;
        }
    }

    if (!store.inMemory) {
        store.file = std::tmpfile();
        if (!store.file) {
            // No tmpfile available; last resort: attempt in-memory anyway.
            try {
                store.mem.reserve((size_t)M);
                store.inMemory = true;
            } catch (...) {
                out.writeStr("0\n");
                out.flush();
                return 0;
            }
        }
    }

    // Read edges, update degrees, store.
    for (uint64_t i = 0; i < M; ++i) {
        uint64_t u64 = 0, v64 = 0;
        if (!fs.readUnsigned(u64) || !fs.readUnsigned(v64)) {
            // malformed input
            out.writeStr("0\n");
            out.flush();
            return 0;
        }
        uint32_t u = (uint32_t)u64, v = (uint32_t)v64;
        if (u >= N || v >= N) {
            // ignore invalid; but keep stream aligned
            continue;
        }
        if (u != v) {
            if (deg[u] != std::numeric_limits<uint32_t>::max()) ++deg[u];
            if (deg[v] != std::numeric_limits<uint32_t>::max()) ++deg[v];
        }
        Edge e{u, v};
        if (store.inMemory) {
            store.mem.push_back(e);
        } else {
            std::fwrite(&e, sizeof(Edge), 1, store.file);
        }
    }

    // Special case N==2: min cut is all edges between 0 and 1.
    if (N == 2) {
        std::vector<uint32_t> side(N, 0);
        side[0] = 0; side[1] = 1;
        uint64_t k = countCutEdges(store, N, side);
        out.writeUInt64(k);
        out.pushChar('\n');
        emitCutEdges(store, side, out);
        out.flush();
        return 0;
    }

    uint32_t minDeg = std::numeric_limits<uint32_t>::max();
    for (uint32_t i = 0; i < N; ++i) minDeg = std::min(minDeg, deg[i]);
    if (minDeg == 0) {
        // graph disconnected or isolated vertices; cut size 0 (already disconnected)
        out.writeStr("0\n");
        out.flush();
        return 0;
    }

    // If there is a degree-1 vertex, min cut is 1 (its only incident edge is a bridge).
    // Find and output that edge by scanning once.
    if (minDeg == 1) {
        uint32_t leaf = 0;
        while (leaf < N && deg[leaf] != 1) ++leaf;

        // Find an edge incident to leaf
        Edge ans{0, 0};
        bool found = false;

        if (store.inMemory) {
            for (auto &e : store.mem) {
                if (e.u == leaf || e.v == leaf) { ans = e; found = true; break; }
            }
        } else {
            std::rewind(store.file);
            static constexpr size_t BUFE = 1 << 20;
            std::vector<Edge> buf(BUFE);
            uint64_t readTotal = 0;
            while (readTotal < store.M && !found) {
                size_t want = (size_t)std::min<uint64_t>(BUFE, store.M - readTotal);
                size_t got = std::fread(buf.data(), sizeof(Edge), want, store.file);
                if (got == 0) break;
                for (size_t i = 0; i < got; ++i) {
                    const auto &e = buf[i];
                    if (e.u == leaf || e.v == leaf) { ans = e; found = true; break; }
                }
                readTotal += got;
            }
        }

        out.writeStr("1\n");
        if (found) {
            out.writeUInt32(ans.u);
            out.pushChar(' ');
            out.writeUInt32(ans.v);
            out.pushChar('\n');
        } else {
            // Shouldn't happen for connected graphs, but keep output valid.
            out.writeStr("0 0\n");
        }
        out.flush();
        return 0;
    }

    // Probe for a bridge using a very small certificate (k=2).
    {
        uint64_t kProbe = 2;
        auto cert2 = buildKForestCertificate(N, M, store, kProbe);

        Edge br;
        if (findAnyBridge(N, cert2, br)) {
            out.writeStr("1\n");
            out.writeUInt32(br.u);
            out.pushChar(' ');
            out.writeUInt32(br.v);
            out.pushChar('\n');
            out.flush();
            return 0;
        }
    }

    // Build certificate with k = minDeg + 1 (ensures min cut < k, so preserved).
    uint64_t k = (uint64_t)minDeg + 1;

    auto cert = buildKForestCertificate(N, M, store, k);

    // Exact min cut on certificate (Stoer–Wagner).
    MinCutResult mc = stoerWagnerMinCut(N, cert);

    // Count cut edges in original graph to output correct list size.
    uint64_t Korig = countCutEdges(store, N, mc.side);

    out.writeUInt64(Korig);
    out.pushChar('\n');
    emitCutEdges(store, mc.side, out);
    out.flush();
    return 0;
}
