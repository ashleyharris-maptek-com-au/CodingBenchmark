#include <cstdio>
#include <cstdint>
#include <cstring>
#include <vector>
#include <algorithm>
#include <cmath>
#include <limits>

struct FastScanner {
    static constexpr size_t BUFSIZE = 1 << 20;
    unsigned char buf[BUFSIZE];
    size_t idx = 0, size = 0;

    inline unsigned char readByte() {
        if (idx >= size) {
            size = std::fread(buf, 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    template <class T>
    bool readInt(T &out) {
        unsigned char c;
        do {
            c = readByte();
            if (!c) return false;
        } while (c <= ' ');

        bool neg = false;
        if (c == '-') {
            neg = true;
            c = readByte();
        }

        T val = 0;
        while (c >= '0' && c <= '9') {
            val = val * 10 + (c - '0');
            c = readByte();
        }
        out = neg ? -val : val;
        return true;
    }

    bool readFloat(float &out) {
        unsigned char c;
        do {
            c = readByte();
            if (!c) return false;
        } while (c <= ' ');

        bool neg = false;
        if (c == '-') {
            neg = true;
            c = readByte();
        }

        double val = 0.0;
        while (c >= '0' && c <= '9') {
            val = val * 10.0 + double(c - '0');
            c = readByte();
        }
        if (c == '.') {
            c = readByte();
            double base = 0.1;
            while (c >= '0' && c <= '9') {
                val += double(c - '0') * base;
                base *= 0.1;
                c = readByte();
            }
        }
        if (c == 'e' || c == 'E') {
            c = readByte();
            bool eneg = false;
            if (c == '-') { eneg = true; c = readByte(); }
            else if (c == '+') { c = readByte(); }
            int exp = 0;
            while (c >= '0' && c <= '9') {
                exp = exp * 10 + (c - '0');
                c = readByte();
            }
            double factor = std::pow(10.0, eneg ? -exp : exp);
            val *= factor;
        }

        if (neg) val = -val;
        out = (float)val;
        return true;
    }
};

struct FastOutput {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t pos = 0;

    ~FastOutput() { flush(); }

    inline void flush() {
        if (pos) {
            std::fwrite(buf, 1, pos, stdout);
            pos = 0;
        }
    }

    inline void pushChar(char c) {
        if (pos >= BUFSIZE) flush();
        buf[pos++] = c;
    }

    inline void writeUInt(uint32_t x) {
        char tmp[16];
        int n = 0;
        do {
            tmp[n++] = char('0' + (x % 10));
            x /= 10;
        } while (x);
        while (n--) pushChar(tmp[n]);
    }

    inline void writeIntWithSep(uint32_t x, bool isLast) {
        writeUInt(x);
        pushChar(isLast ? '\n' : ' ');
    }
};

struct Pt {
    float x, y;
};

static inline double dist2(const Pt& a, const Pt& b) {
    double dx = (double)a.x - (double)b.x;
    double dy = (double)a.y - (double)b.y;
    return dx*dx + dy*dy;
}

static inline double dist(const Pt& a, const Pt& b) {
    return std::sqrt(dist2(a, b));
}

// Split bits (Morton)
static inline uint64_t splitBy1(uint32_t a) {
    uint64_t x = (uint64_t)a;
    x = (x | (x << 16)) & 0x0000FFFF0000FFFFULL;
    x = (x | (x << 8))  & 0x00FF00FF00FF00FFULL;
    x = (x | (x << 4))  & 0x0F0F0F0F0F0F0F0FULL;
    x = (x | (x << 2))  & 0x3333333333333333ULL;
    x = (x | (x << 1))  & 0x5555555555555555ULL;
    return x;
}

static inline uint64_t morton2D(uint32_t x, uint32_t y) {
    return splitBy1(x) | (splitBy1(y) << 1);
}

struct KeyIdx {
    uint64_t key;
    uint32_t idx;
};

// Radix sort (LSD, 16-bit digits) for KeyIdx by key
static void radixSortKeyIdx(std::vector<KeyIdx>& a) {
    const size_t n = a.size();
    std::vector<KeyIdx> tmp(n);
    static constexpr size_t RAD = 1u << 16;
    std::vector<size_t> cnt(RAD);

    for (int pass = 0; pass < 4; ++pass) {
        std::fill(cnt.begin(), cnt.end(), 0);
        const int shift = pass * 16;
        for (size_t i = 0; i < n; ++i) {
            uint32_t b = (uint32_t)((a[i].key >> shift) & 0xFFFFULL);
            cnt[b]++;
        }
        size_t sum = 0;
        for (size_t i = 0; i < RAD; ++i) {
            size_t c = cnt[i];
            cnt[i] = sum;
            sum += c;
        }
        for (size_t i = 0; i < n; ++i) {
            uint32_t b = (uint32_t)((a[i].key >> shift) & 0xFFFFULL);
            tmp[cnt[b]++] = a[i];
        }
        a.swap(tmp);
    }
}

static std::vector<uint32_t> solveHeldKarp(const std::vector<Pt>& pts) {
    const int n = (int)pts.size();
    const int FULL = (1 << n);

    std::vector<std::vector<double>> d(n, std::vector<double>(n, 0.0));
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < n; ++j)
            d[i][j] = dist(pts[i], pts[j]);

    const double INF = std::numeric_limits<double>::infinity();
    std::vector<std::vector<double>> dp(FULL, std::vector<double>(n, INF));
    std::vector<std::vector<int>> parent(FULL, std::vector<int>(n, -1));

    dp[1][0] = 0.0;

    for (int mask = 1; mask < FULL; ++mask) {
        for (int u = 0; u < n; ++u) if (dp[mask][u] < INF) {
            for (int v = 0; v < n; ++v) if (!(mask & (1 << v))) {
                int nmask = mask | (1 << v);
                double cand = dp[mask][u] + d[u][v];
                if (cand < dp[nmask][v]) {
                    dp[nmask][v] = cand;
                    parent[nmask][v] = u;
                }
            }
        }
    }

    // Close the cycle back to 0
    int bestEnd = 0;
    double best = INF;
    for (int u = 0; u < n; ++u) {
        double cand = dp[FULL - 1][u] + d[u][0];
        if (cand < best) {
            best = cand;
            bestEnd = u;
        }
    }

    std::vector<uint32_t> rev;
    rev.reserve(n);
    int mask = FULL - 1;
    int cur = bestEnd;
    while (cur != -1) {
        rev.push_back((uint32_t)cur);
        int p = parent[mask][cur];
        mask ^= (1 << cur);
        cur = p;
    }
    std::reverse(rev.begin(), rev.end());

    // Ensure it starts at 0 (it should, but keep robust)
    auto it = std::find(rev.begin(), rev.end(), 0u);
    if (it != rev.begin() && it != rev.end()) {
        std::rotate(rev.begin(), it, rev.end());
    }
    return rev;
}

static std::vector<uint32_t> solveNearestNeighbor2Opt(const std::vector<Pt>& pts) {
    const int n = (int)pts.size();
    std::vector<uint32_t> tour(n);
    std::vector<unsigned char> used(n, 0);

    uint32_t cur = 0;
    tour[0] = 0;
    used[0] = 1;

    for (int step = 1; step < n; ++step) {
        uint32_t best = UINT32_MAX;
        double bestD = std::numeric_limits<double>::infinity();
        const Pt& pc = pts[cur];
        for (uint32_t j = 0; j < (uint32_t)n; ++j) {
            if (!used[j]) {
                double d = dist2(pc, pts[j]);
                if (d < bestD) {
                    bestD = d;
                    best = j;
                }
            }
        }
        cur = best;
        used[cur] = 1;
        tour[step] = cur;
    }

    // 2-opt in a limited window to keep it fast
    auto D = [&](uint32_t a, uint32_t b) -> double { return dist(pts[a], pts[b]); };

    int K = (n <= 5000) ? 200 : (n <= 200000 ? 64 : 24);
    int passes = (n <= 50000) ? 3 : 2;

    for (int pass = 0; pass < passes; ++pass) {
        bool improved = false;
        for (int i = 0; i < n; ++i) {
            int i2 = (i + 1) % n;
            uint32_t a = tour[i];
            uint32_t b = tour[i2];
            int jmax = std::min(n - 1, i + K);
            for (int j = i + 2; j <= jmax; ++j) {
                int j2 = (j + 1) % n;
                if (i == 0 && j == n - 1) continue; // avoid breaking the cycle edge at start/end
                uint32_t c = tour[j];
                uint32_t d = tour[j2];

                double before = D(a, b) + D(c, d);
                double after  = D(a, c) + D(b, d);
                if (after + 1e-12 < before) {
                    std::reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }

    // Rotate so city 0 is first
    auto it = std::find(tour.begin(), tour.end(), 0u);
    if (it != tour.begin() && it != tour.end())
        std::rotate(tour.begin(), it, tour.end());
    return tour;
}

static std::vector<uint32_t> solveMorton(const std::vector<Pt>& pts, bool doLocalImprove) {
    const uint32_t n = (uint32_t)pts.size();
    if (n == 0) return {};

    float minX = pts[0].x, maxX = pts[0].x;
    float minY = pts[0].y, maxY = pts[0].y;
    for (uint32_t i = 1; i < n; ++i) {
        minX = std::min(minX, pts[i].x);
        maxX = std::max(maxX, pts[i].x);
        minY = std::min(minY, pts[i].y);
        maxY = std::max(maxY, pts[i].y);
    }

    constexpr int BITS = 21;
    const uint32_t MAXQ = (1u << BITS) - 1u;

    double sx = (maxX > minX) ? (double)MAXQ / (double)(maxX - minX) : 0.0;
    double sy = (maxY > minY) ? (double)MAXQ / (double)(maxY - minY) : 0.0;

    std::vector<KeyIdx> a(n);
    for (uint32_t i = 0; i < n; ++i) {
        uint32_t qx = (sx == 0.0) ? 0u : (uint32_t)std::llround(((double)pts[i].x - (double)minX) * sx);
        uint32_t qy = (sy == 0.0) ? 0u : (uint32_t)std::llround(((double)pts[i].y - (double)minY) * sy);
        if (qx > MAXQ) qx = MAXQ;
        if (qy > MAXQ) qy = MAXQ;
        a[i] = KeyIdx{morton2D(qx, qy), i};
    }

    if (n <= 2000000u) {
        std::sort(a.begin(), a.end(), [](const KeyIdx& p, const KeyIdx& q) {
            return p.key < q.key;
        });
    } else {
        radixSortKeyIdx(a);
    }

    std::vector<uint32_t> tour(n);
    for (uint32_t i = 0; i < n; ++i) tour[i] = a[i].idx;

    // Rotate so 0 is first
    auto it0 = std::find(tour.begin(), tour.end(), 0u);
    if (it0 != tour.begin() && it0 != tour.end())
        std::rotate(tour.begin(), it0, tour.end());

    if (doLocalImprove && n >= 4) {
        // Very light local 2-opt with tiny window (keeps O(n))
        auto D = [&](uint32_t u, uint32_t v) -> double { return dist(pts[u], pts[v]); };
        int K = (n <= 200000u) ? 16 : 8;
        for (uint32_t i = 0; i + 3 < n; ++i) {
            uint32_t a0 = tour[i];
            uint32_t b0 = tour[i + 1];
            uint32_t bestj = UINT32_MAX;
            double bestGain = 0.0;

            uint32_t jmax = std::min<uint32_t>(n - 2, i + (uint32_t)K);
            for (uint32_t j = i + 2; j <= jmax; ++j) {
                uint32_t c0 = tour[j];
                uint32_t d0 = tour[j + 1];
                double before = D(a0, b0) + D(c0, d0);
                double after  = D(a0, c0) + D(b0, d0);
                double gain = before - after;
                if (gain > bestGain) {
                    bestGain = gain;
                    bestj = j;
                }
            }
            if (bestj != UINT32_MAX && bestGain > 1e-12) {
                std::reverse(tour.begin() + (i + 1), tour.begin() + (bestj + 1));
            }
        }

        // Ensure 0 is still first
        it0 = std::find(tour.begin(), tour.end(), 0u);
        if (it0 != tour.begin() && it0 != tour.end())
            std::rotate(tour.begin(), it0, tour.end());
    }

    return tour;
}

int main() {
    FastScanner fs;
    FastOutput fo;

    uint32_t N_u32 = 0;
    if (!fs.readInt(N_u32)) return 0;
    const uint64_t N = N_u32;

    // Always read all coordinates in any non-stream mode.
    // For extremely large N, fallback to streaming output in input order (O(1) memory).
    // (Note: for truly enormous N, I/O dominates anyway.)
    const uint64_t STREAM_THRESHOLD = 20000000ULL; // 20 million cities

    if (N == 0) {
        // No cities: nothing to output
        return 0;
    }

    if (N > STREAM_THRESHOLD) {
        // Streaming mode: output 0..N-1 as we read coordinates, ignore coords.
        // Valid permutation, starts at 0, fastest possible.
        float x, y;
        for (uint64_t i = 0; i < N; ++i) {
            fs.readFloat(x);
            fs.readFloat(y);
            fo.writeIntWithSep((uint32_t)i, i + 1 == N);
        }
        return 0;
    }

    // Read all points
    std::vector<Pt> pts;
    pts.resize((size_t)N);
    for (uint32_t i = 0; i < (uint32_t)N; ++i) {
        fs.readFloat(pts[i].x);
        fs.readFloat(pts[i].y);
    }

    std::vector<uint32_t> tour;

    if (N <= 12) {
        tour = solveHeldKarp(pts); // exact
    } else if (N <= 2000) {
        tour = solveNearestNeighbor2Opt(pts);
    } else if (N <= 2000000) {
        bool localImprove = (N <= 200000); // improve more for smaller of these
        tour = solveMorton(pts, localImprove);
    } else {
        tour = solveMorton(pts, false);
    }

    // Output
    // Must be N integers, a permutation 0..N-1, starting with 0.
    for (uint32_t i = 0; i < (uint32_t)N; ++i) {
        fo.writeIntWithSep(tour[i], i + 1 == (uint32_t)N);
    }
    return 0;
}
