// Vehicle Routing (no capacity) heuristic: angular sweep clustering + limited 2-opt per route.
// Reads one JSON from stdin, writes one JSON to stdout. No external dependencies.

declare const require: any;
const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8");
const data = JSON.parse(input);

type Depot = [number, number];
type PkgIn = { id: any; x: number; y: number };

const depot: Depot = Array.isArray(data.depot) ? data.depot : [0, 0];
const depotX = +depot[0];
const depotY = +depot[1];

const packages: PkgIn[] = Array.isArray(data.packages) ? data.packages : [];
const n = packages.length;

let truckCount: number = Number.isFinite(data.truckCount) ? Math.floor(data.truckCount) : 1;
if (!Number.isFinite(truckCount)) truckCount = 1;
if (truckCount < 1) truckCount = 1;

if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const routeCount = Math.min(truckCount, n);

type Pkg = { id: any; x: number; y: number; ang: number; r2: number; idx: number };

const enriched: Pkg[] = new Array(n);
for (let i = 0; i < n; i++) {
  const p = packages[i];
  const dx = p.x - depotX;
  const dy = p.y - depotY;
  let ang = Math.atan2(dy, dx);
  if (ang < 0) ang += Math.PI * 2;
  const r2 = dx * dx + dy * dy;
  enriched[i] = { id: p.id, x: +p.x, y: +p.y, ang, r2, idx: i };
}

// Deterministic ordering: angle, radius, then stable tie-breaker by index then id stringified.
enriched.sort((a, b) => {
  const da = a.ang - b.ang;
  if (da !== 0) return da;
  const dr = a.r2 - b.r2;
  if (dr !== 0) return dr;
  if (a.idx !== b.idx) return a.idx - b.idx;
  const sa = String(a.id);
  const sb = String(b.id);
  return sa < sb ? -1 : sa > sb ? 1 : 0;
});

// Rotate so the cut is at the largest angular gap (common sweep improvement).
let bestCut = 0;
let bestGap = -1;
for (let i = 0; i < n; i++) {
  const a1 = enriched[i].ang;
  const a2 = enriched[(i + 1) % n].ang;
  const gap = (i + 1 < n) ? (a2 - a1) : (a2 + Math.PI * 2 - a1);
  if (gap > bestGap + 1e-15) {
    bestGap = gap;
    bestCut = (i + 1) % n;
  }
}
if (bestCut !== 0) {
  const rotated = enriched.slice(bestCut).concat(enriched.slice(0, bestCut));
  for (let i = 0; i < n; i++) enriched[i] = rotated[i];
}

function distXY(ax: number, ay: number, bx: number, by: number): number {
  return Math.hypot(ax - bx, ay - by);
}

function twoOptLimited(route: Pkg[], window: number, passes: number): void {
  const m = route.length;
  if (m < 4) return;

  // In-place reverse of segment [i, j]
  const reverseSegment = (i: number, j: number) => {
    while (i < j) {
      const tmp = route[i];
      route[i] = route[j];
      route[j] = tmp;
      i++;
      j--;
    }
  };

  for (let pass = 0; pass < passes; pass++) {
    let improved = false;

    for (let i = 0; i < m - 2; i++) {
      const ax = (i === 0) ? depotX : route[i - 1].x;
      const ay = (i === 0) ? depotY : route[i - 1].y;
      const bx = route[i].x;
      const by = route[i].y;

      const jMax = Math.min(m - 1, i + window);
      for (let j = i + 1; j <= jMax; j++) {
        if (j === i) continue;
        if (j === i + 1) continue; // adjacent swap doesn't help for 2-opt here

        const cx = route[j].x;
        const cy = route[j].y;
        const dx = (j === m - 1) ? depotX : route[j + 1].x;
        const dy = (j === m - 1) ? depotY : route[j + 1].y;

        const before = distXY(ax, ay, bx, by) + distXY(cx, cy, dx, dy);
        const after = distXY(ax, ay, cx, cy) + distXY(bx, by, dx, dy);

        if (after + 1e-12 < before) {
          reverseSegment(i, j);
          improved = true;
          // Continue scanning from current i to allow cascading improvements deterministically.
        }
      }
    }

    if (!improved) break;
  }
}

// Split into contiguous segments (equal by count) to form routes.
const routes: Pkg[][] = [];
const base = Math.floor(n / routeCount);
const rem = n % routeCount;

let pos = 0;
for (let r = 0; r < routeCount; r++) {
  const size = base + (r < rem ? 1 : 0);
  const seg = enriched.slice(pos, pos + size);
  pos += size;

  // Improve each segment with limited 2-opt.
  // Tune parameters to scale: small routes get more optimization.
  const m = seg.length;
  if (m >= 4) {
    const window = m <= 120 ? 120 : (m <= 800 ? 64 : 32);
    const passes = m <= 120 ? 4 : (m <= 800 ? 3 : 2);
    twoOptLimited(seg, window, passes);
  }

  routes.push(seg);
}

const out = {
  routes: routes.map((rt) => rt.map((p) => p.id)),
};

process.stdout.write(JSON.stringify(out));
