#include <iostream>
#include <string>
#include <vector>
#include <queue>
#include <algorithm>
#include <cstdint>
#include <limits>

using std::string;
using std::vector;

static inline void strip_cr(string& s) {
    if (!s.empty() && s.back() == '\r') s.pop_back();
}

// O(n*m) DP for 2 strings (only used for small sizes).
static string lcs_2_dp(const string& a, const string& b) {
    const int n = (int)a.size(), m = (int)b.size();
    vector<int> prev(m + 1, 0), cur(m + 1, 0);
    int bestLen = 0, bestEnd = -1; // end position in a (0-based)
    for (int i = 1; i <= n; ++i) {
        // cur[0] = 0; // already 0 from initialization below
        for (int j = 1; j <= m; ++j) {
            if (a[i - 1] == b[j - 1]) {
                cur[j] = prev[j - 1] + 1;
                if (cur[j] > bestLen) {
                    bestLen = cur[j];
                    bestEnd = i - 1;
                }
            } else {
                cur[j] = 0;
            }
        }
        std::swap(prev, cur);
        std::fill(cur.begin(), cur.end(), 0);
    }
    if (bestLen <= 0) return string();
    return a.substr(bestEnd - bestLen + 1, bestLen);
}

// Suffix Automaton with memory-lean adjacency as global edge arrays + per-state head (linked list).
struct SAM {
    // State arrays
    vector<int32_t> link;
    vector<int32_t> len;
    vector<int32_t> head;
    vector<int32_t> firstpos;

    // Edge arrays (linked list per state)
    vector<int32_t> to;
    vector<int32_t> next;
    vector<uint8_t> ch;

    int32_t last = 0;

    SAM() = default;

    explicit SAM(size_t reserveStates, size_t reserveEdges) {
        link.reserve(reserveStates);
        len.reserve(reserveStates);
        head.reserve(reserveStates);
        firstpos.reserve(reserveStates);
        to.reserve(reserveEdges);
        next.reserve(reserveEdges);
        ch.reserve(reserveEdges);
        new_state(); // root = 0
        link[0] = -1;
        len[0] = 0;
        head[0] = -1;
        firstpos[0] = -1;
        last = 0;
    }

    int32_t new_state() {
        int32_t id = (int32_t)len.size();
        len.push_back(0);
        link.push_back(-1);
        head.push_back(-1);
        firstpos.push_back(-1);
        return id;
    }

    inline void add_edge(int32_t v, uint8_t c, int32_t u) {
        int32_t e = (int32_t)to.size();
        to.push_back(u);
        ch.push_back(c);
        next.push_back(head[v]);
        head[v] = e;
    }

    inline int32_t get_next(int32_t v, uint8_t c) const {
        for (int32_t e = head[v]; e != -1; e = next[e]) {
            if (ch[e] == c) return to[e];
        }
        return -1;
    }

    inline void set_next(int32_t v, uint8_t c, int32_t u) {
        for (int32_t e = head[v]; e != -1; e = next[e]) {
            if (ch[e] == c) {
                to[e] = u;
                return;
            }
        }
        add_edge(v, c, u);
    }

    inline void copy_edges(int32_t src, int32_t dst) {
        // Copy adjacency list of src into dst (order doesn't matter).
        for (int32_t e = head[src]; e != -1; e = next[e]) {
            add_edge(dst, ch[e], to[e]);
        }
    }

    void extend(uint8_t c, int32_t pos) {
        int32_t cur = new_state();
        len[cur] = len[last] + 1;
        firstpos[cur] = pos;

        int32_t p = last;
        while (p != -1 && get_next(p, c) == -1) {
            add_edge(p, c, cur);
            p = link[p];
        }

        if (p == -1) {
            link[cur] = 0;
        } else {
            int32_t q = get_next(p, c);
            if (len[p] + 1 == len[q]) {
                link[cur] = q;
            } else {
                int32_t clone = new_state();
                len[clone] = len[p] + 1;
                link[clone] = link[q];
                firstpos[clone] = firstpos[q];
                copy_edges(q, clone);

                while (p != -1) {
                    int32_t nxt = get_next(p, c);
                    if (nxt != q) break;
                    set_next(p, c, clone);
                    p = link[p];
                }

                link[q] = link[cur] = clone;
            }
        }

        last = cur;
    }
};

// Multi-string LCS using SAM built from the shortest string.
// Sparse per-string processing: only touches states that actually get non-zero matches.
static string lcs_multi_sam(const vector<string>& strs) {
    const int N = (int)strs.size();
    if (N == 0) return string();
    if (N == 1) return strs[0];

    // If any string is empty => longest common substring is empty.
    for (const auto& s : strs) if (s.empty()) return string();

    // Build SAM from shortest string (assumed at index 0 by caller).
    const string& base = strs[0];
    const int32_t baseLen = (int32_t)base.size();

    // Heuristic reserves: states <= 2*|S|, edges often around ~2-3*|S| in practice.
    size_t reserveStates = (size_t)baseLen * 2 + 5;
    size_t reserveEdges  = (size_t)baseLen * 3 + 5;

    SAM sam(reserveStates, reserveEdges);
    for (int32_t i = 0; i < baseLen; ++i) {
        sam.extend((uint8_t)base[(size_t)i], i);
    }

    const int32_t S = (int32_t)sam.len.size();

    // Per-state accumulators across strings:
    // count[v] = number of non-base strings where state v had a non-zero match.
    // minv[v]  = minimum across those strings of the maximum match length for that state.
    vector<uint16_t> count((size_t)S, 0);
    vector<uint16_t> lastSeen((size_t)S, 0); // string index (1..N-1), 0 means never
    vector<int32_t> minv((size_t)S, 0);

    // Per-string sparse arrays:
    vector<int32_t> tmp((size_t)S, 0);
    vector<int32_t> nxtBucket((size_t)S, -1);
    vector<int32_t> touched;
    touched.reserve(1 << 20);

    // Buckets by state length for decreasing-length propagation, sparsely reset by stamp.
    vector<int32_t> bucketHead((size_t)baseLen + 1, -1);
    vector<uint32_t> bucketStamp((size_t)baseLen + 1, 0);
    uint32_t stamp = 1;

    auto bucket_touch = [&](int32_t v, std::priority_queue<int32_t>& pq) {
        int32_t l = sam.len[v];
        if (l <= 0) return; // don't bucket root or invalid
        if (bucketStamp[(size_t)l] != stamp) {
            bucketStamp[(size_t)l] = stamp;
            bucketHead[(size_t)l] = -1;
            pq.push(l);
        }
        nxtBucket[(size_t)v] = bucketHead[(size_t)l];
        bucketHead[(size_t)l] = v;
    };

    auto touch_state = [&](int32_t v, std::priority_queue<int32_t>& pq) {
        if (v == 0) return;
        if (tmp[(size_t)v] == 0) {
            touched.push_back(v);
            bucket_touch(v, pq);
        }
    };

    // Process each other string.
    for (int si = 1; si < N; ++si) {
        const string& t = strs[si];

        // Stamp management (handle wrap very conservatively).
        ++stamp;
        if (stamp == 0) {
            std::fill(bucketStamp.begin(), bucketStamp.end(), 0);
            stamp = 1;
        }

        std::priority_queue<int32_t> pq;
        touched.clear();

        int32_t v = 0;
        int32_t l = 0;

        // Traverse t through SAM; record best match length ending at each visited state.
        for (unsigned char cc : t) {
            uint8_t c = (uint8_t)cc;

            int32_t nxt = sam.get_next(v, c);
            while (v != 0 && nxt == -1) {
                v = sam.link[v];
                l = std::min<int32_t>(l, sam.len[v]);
                nxt = sam.get_next(v, c);
            }
            if (nxt != -1) {
                v = nxt;
                ++l;
            } else {
                // v == 0 and no transition
                v = 0;
                l = 0;
            }

            if (v != 0) {
                if (l > tmp[(size_t)v]) {
                    touch_state(v, pq);
                    tmp[(size_t)v] = l;
                }
            }
        }

        // Propagate matches from longer states to suffix links (decreasing length).
        while (!pq.empty()) {
            int32_t curLen = pq.top();
            pq.pop();

            if (curLen <= 0) continue;
            if (bucketStamp[(size_t)curLen] != stamp) continue;

            int32_t x = bucketHead[(size_t)curLen];
            bucketHead[(size_t)curLen] = -1; // consumed

            while (x != -1) {
                int32_t nextX = nxtBucket[(size_t)x];

                int32_t p = sam.link[x];
                if (p > 0) {
                    int32_t val = tmp[(size_t)x];
                    if (val > sam.len[p]) val = sam.len[p];
                    if (val > tmp[(size_t)p]) {
                        touch_state(p, pq);
                        tmp[(size_t)p] = val;
                    }
                }

                x = nextX;
            }
        }

        // Apply this string's contributions to count/minv, and reset tmp for touched states.
        // We only want to count each state once per string.
        const uint16_t mark = (uint16_t)si;
        for (int32_t st : touched) {
            int32_t tv = tmp[(size_t)st];
            if (tv > 0) {
                if (lastSeen[(size_t)st] != mark) {
                    lastSeen[(size_t)st] = mark;
                    uint16_t c = (uint16_t)(count[(size_t)st] + 1);
                    count[(size_t)st] = c;
                    if (c == 1) minv[(size_t)st] = tv;
                    else minv[(size_t)st] = std::min(minv[(size_t)st], tv);
                } else {
                    // Shouldn't happen with our touch logic, but safe.
                    minv[(size_t)st] = (minv[(size_t)st] == 0) ? tv : std::min(minv[(size_t)st], tv);
                }
            }
            tmp[(size_t)st] = 0;
        }
    }

    // Pick best state that appeared in all N-1 non-base strings.
    int32_t bestLen = 0;
    int32_t bestState = -1;
    const uint16_t need = (uint16_t)(N - 1);
    for (int32_t st = 1; st < S; ++st) {
        if (count[(size_t)st] == need) {
            int32_t cand = minv[(size_t)st];
            if (cand > bestLen) {
                bestLen = cand;
                bestState = st;
            }
        }
    }

    if (bestLen <= 0 || bestState < 0) return string();

    int32_t endPos = sam.firstpos[bestState];
    int32_t start = endPos - bestLen + 1;
    if (start < 0) start = 0;
    return base.substr((size_t)start, (size_t)bestLen);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N;
    if (!(std::cin >> N)) return 0;
    string dummy;
    std::getline(std::cin, dummy); // consume rest of line

    vector<string> strs;
    strs.reserve(std::max(0, N));
    for (int i = 0; i < N; ++i) {
        string s;
        std::getline(std::cin, s);
        strip_cr(s);
        strs.push_back(std::move(s));
    }

    if (N <= 0) {
        std::cout << "\n";
        return 0;
    }
    if (N == 1) {
        std::cout << strs[0] << "\n";
        return 0;
    }

    // Choose strategy adaptively.
    // 1) For N==2 and small product, use O(n*m) DP.
    // 2) Otherwise, reorder to put the shortest string first, and use multi-string SAM.
    if (N == 2) {
        const size_t n0 = strs[0].size(), n1 = strs[1].size();
        const uint64_t prod = (uint64_t)n0 * (uint64_t)n1;
        // Threshold tuned to stay comfortably fast/memory-light.
        const uint64_t DP_THRESHOLD = 50ULL * 1000ULL * 1000ULL; // 50 million
        const size_t MAX_SIDE = 5000;

        if (n0 == 0 || n1 == 0) {
            std::cout << "\n";
            return 0;
        }

        if (n0 <= MAX_SIDE && n1 <= MAX_SIDE && prod <= DP_THRESHOLD) {
            std::cout << lcs_2_dp(strs[0], strs[1]) << "\n";
            return 0;
        }
        // Fallthrough to SAM (faster for larger sizes).
    }

    // Put shortest string at index 0 for SAM.
    int idxMin = 0;
    for (int i = 1; i < N; ++i) {
        if (strs[i].size() < strs[idxMin].size()) idxMin = i;
    }
    if (idxMin != 0) std::swap(strs[0], strs[idxMin]);

    // Early exit if shortest is empty.
    if (strs[0].empty()) {
        std::cout << "\n";
        return 0;
    }

    std::cout << lcs_multi_sam(strs) << "\n";
    return 0;
}
