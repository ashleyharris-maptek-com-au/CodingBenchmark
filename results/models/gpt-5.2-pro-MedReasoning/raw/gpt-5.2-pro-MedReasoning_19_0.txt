```csharp
using System;
using System.IO;
using System.Text;
using System.Globalization;

public static class Program
{
    // Fast, buffered reader that can read both tokens (ASCII) and raw bytes (0..255).
    private sealed class FastInput
    {
        private readonly Stream _s;
        private readonly byte[] _buf;
        private int _len, _pos;

        private int _pushback = -1; // single-byte pushback (0..255), or -1

        public FastInput(Stream s, int bufferSize = 1 << 20)
        {
            _s = s;
            _buf = new byte[bufferSize];
        }

        private int ReadByteInternal()
        {
            if (_pos >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_pos++];
        }

        public int ReadByteRaw()
        {
            if (_pushback >= 0)
            {
                int b = _pushback;
                _pushback = -1;
                return b;
            }
            return ReadByteInternal();
        }

        public void PushbackByte(int b)
        {
            _pushback = b;
        }

        public string ReadToken()
        {
            int c;
            do
            {
                c = ReadByteInternal();
                if (c < 0) return "";
            } while (c <= 32);

            var sb = new StringBuilder(32);
            while (c > 32)
            {
                sb.Append((char)c);
                c = ReadByteInternal();
                if (c < 0) break;
            }
            return sb.ToString();
        }
    }

    private enum ActionType : byte
    {
        Plant = 0, // dump at (0,0) => removed; if junk causes penalty (modeled in profit)
        DumpAdjacent = 1 // dump at chosen adjacent column => raises its height by 1
    }

    public static void Main()
    {
        var input = new FastInput(Console.OpenStandardInput());

        // Header: XVoxelCount YVoxelCount ZVoxelCount XBlockSize YBlockSize ZBlockSize
        // We only need the counts for this solver; block sizes are parsed then ignored.
        int xCount = int.Parse(input.ReadToken(), CultureInfo.InvariantCulture);
        int yCount = int.Parse(input.ReadToken(), CultureInfo.InvariantCulture);
        int zCount = int.Parse(input.ReadToken(), CultureInfo.InvariantCulture);
        _ = double.Parse(input.ReadToken(), CultureInfo.InvariantCulture);
        _ = double.Parse(input.ReadToken(), CultureInfo.InvariantCulture);
        _ = double.Parse(input.ReadToken(), CultureInfo.InvariantCulture);

        // Consume exactly the newline after the header, without skipping arbitrary whitespace
        // (voxel bytes may be any values including whitespace).
        int b0 = input.ReadByteRaw();
        if (b0 == '\r')
        {
            int b1 = input.ReadByteRaw();
            if (b1 != '\n' && b1 >= 0) input.PushbackByte(b1);
        }
        else if (b0 != '\n' && b0 >= 0)
        {
            // Header didn't end with '\n' as expected; treat this as first voxel byte.
            input.PushbackByte(b0);
        }

        // Strategy selection (adaptive):
        // - If we have an adjacent column available, we can dump junk there to avoid the huge plant damage.
        // - Use DP when DP table is reasonably sized; otherwise greedy.
        // Regardless of world scale, we only read the minimum needed bytes (origin column + 1 adjacent column).
        bool hasAdjacent = (xCount > 1) || (yCount > 1);
        int dumpX = 0, dumpY = 0;
        bool dumpIsNextColumnInStream;
        if (!hasAdjacent)
        {
            dumpIsNextColumnInStream = false;
        }
        else if (yCount > 1)
        {
            // Next column in the stream after (0,0) is (0,1).
            dumpX = 0; dumpY = 1;
            dumpIsNextColumnInStream = true;
        }
        else
        {
            // yCount == 1 and xCount > 1 -> next column in the stream after (0,0) is (1,0).
            dumpX = 1; dumpY = 0;
            dumpIsNextColumnInStream = true;
        }

        // Read origin column bytes (x=0,y=0,z=0..zCount-1)
        var origin = new byte[zCount];
        for (int z = 0; z < zCount; z++)
        {
            int bb = input.ReadByteRaw();
            if (bb < 0) bb = 0;
            origin[z] = (byte)bb;
        }

        // Read adjacent dump column (if present and immediately next in stream)
        int dumpTopZ = -1;
        if (hasAdjacent && dumpIsNextColumnInStream)
        {
            for (int z = 0; z < zCount; z++)
            {
                int bb = input.ReadByteRaw();
                if (bb < 0) bb = 0;
                int mass = bb & 15;
                if (mass > 0) dumpTopZ = z;
            }
        }
        else
        {
            // No adjacent column available for dumping.
            dumpTopZ = zCount - 1; // effectively "no capacity"
        }

        // Build list of diggable solids in origin column from top-down (skip mass==0 air).
        // Dig order is fixed: always highest remaining solid (exposed from above).
        int solidCount = 0;
        for (int z = zCount - 1; z >= 0; z--)
        {
            int d = origin[z];
            if ((d & 15) > 0) solidCount++;
        }

        if (solidCount == 0)
        {
            Console.WriteLine("0");
            return;
        }

        var zList = new int[solidCount];
        var massList = new int[solidCount];
        var valueList = new int[solidCount];

        int idx = 0;
        for (int z = zCount - 1; z >= 0; z--)
        {
            int d = origin[z];
            int mass = d & 15;
            if (mass <= 0) continue;
            int value = (d >> 4) & 15;

            zList[idx] = z;
            massList[idx] = mass;
            valueList[idx] = value;
            idx++;
        }

        // Basic per-voxel plant profit (no fuel because N=0).
        static long PlantProfit(int mass, int value)
        {
            if (value == 0) return -1000L * mass;      // junk damages plant
            return 1000L * mass * value;              // ore pays
        }

        // If we dump junk adjacent: no payment, no plant damage; only fuel for the single X/Y step.
        // Fuel: downhill free, flat $1/ton, uphill $5/ton. Must satisfy slope constraint |dz| <= 1.
        static bool AdjacentDumpFeasibleAndFuel(int digZ, int dumpTopZ, int zCount, out int fuelPerTon)
        {
            fuelPerTon = 0;

            // Need capacity to place one more voxel at dumpTopZ+1.
            if (dumpTopZ >= zCount - 1) return false;

            int dz = dumpTopZ - digZ;
            if (dz < -1 || dz > 1) return false;

            if (dz == 1) fuelPerTon = 5;       // uphill
            else if (dz == 0) fuelPerTon = 1;  // flat
            else fuelPerTon = 0;               // downhill
            return true;
        }

        // Decide strategy: DP if we can dump adjacent and table size is reasonable.
        // DP state = how many junk voxels have been dumped adjacent so far (which raises dumpTopZ).
        bool canUseAdjacentDump = hasAdjacent && dumpTopZ < zCount - 1;
        int initialDumpTopZ = dumpTopZ;

        // Upper bound of how many times we can dump adjacent.
        int capacity = canUseAdjacentDump ? (zCount - 1 - initialDumpTopZ) : 0;

        // Count junk voxels in origin (value==0) to tighten DP state range.
        int junkCount = 0;
        for (int i = 0; i < solidCount; i++) if (valueList[i] == 0) junkCount++;
        int maxDumps = Math.Min(capacity, junkCount);

        // DP decision table size ~ solidCount*(maxDumps+1).
        long dpCells = (long)solidCount * (maxDumps + 1L);

        // Adaptive thresholds:
        // - If dpCells is modest, do DP; else do greedy.
        bool useDP = canUseAdjacentDump && dpCells <= 40_000_000; // ~40MB for decision bytes, OK

        ActionType[] actions;
        int bestLen; // number of solids to dig (prefix length)
        if (useDP)
        {
            const long NEG = long.MinValue / 4;

            int states = maxDumps + 1;
            var dp = new long[states];
            var ndp = new long[states];
            for (int t = 0; t < states; t++) dp[t] = NEG;
            dp[0] = 0;

            // decision[i, t] : after processing i (0-based), being in dumpCount=t,
            // 0 => came from Plant at same t, 1 => came from Dump at t-1, 2 => unreachable.
            var decision = new byte[solidCount * states];
            for (int i = 0; i < decision.Length; i++) decision[i] = 2;

            long bestProfit = 0;
            int bestI = -1;
            int bestT = 0;

            for (int i = 0; i < solidCount; i++)
            {
                for (int t = 0; t < states; t++) ndp[t] = NEG;

                int digZ = zList[i];
                int mass = massList[i];
                int val = valueList[i];

                for (int t = 0; t < states; t++)
                {
                    long cur = dp[t];
                    if (cur <= NEG / 2) continue;

                    // Option A: dump at plant
                    long candPlant = cur + PlantProfit(mass, val);
                    if (candPlant > ndp[t])
                    {
                        ndp[t] = candPlant;
                        decision[i * states + t] = 0;
                    }

                    // Option B: if junk, dump adjacent (one step)
                    if (val == 0 && t < maxDumps)
                    {
                        int currentDumpTop = initialDumpTopZ + t;
                        if (AdjacentDumpFeasibleAndFuel(digZ, currentDumpTop, zCount, out int fuelPerTon))
                        {
                            long candDump = cur - (long)fuelPerTon * mass;
                            if (candDump > ndp[t + 1])
                            {
                                ndp[t + 1] = candDump;
                                decision[i * states + (t + 1)] = 1;
                            }
                        }
                    }
                }

                // swap dp
                var tmp = dp; dp = ndp; ndp = tmp;

                // record best stopping point at this prefix length
                for (int t = 0; t < states; t++)
                {
                    long p = dp[t];
                    if (p > bestProfit)
                    {
                        bestProfit = p;
                        bestI = i;
                        bestT = t;
                    }
                }
            }

            if (bestI < 0)
            {
                Console.WriteLine("0");
                return;
            }

            bestLen = bestI + 1;
            actions = new ActionType[bestLen];

            int tcur = bestT;
            for (int i = bestI; i >= 0; i--)
            {
                byte dec = decision[i * (maxDumps + 1) + tcur];
                if (dec == 1)
                {
                    actions[i] = ActionType.DumpAdjacent;
                    tcur--;
                }
                else
                {
                    actions[i] = ActionType.Plant;
                }
            }
        }
        else
        {
            // Greedy fallback:
            // - Always dump valuable at plant.
            // - For junk, dump adjacent if feasible and better than plant damage.
            // - Stop at best profit prefix.
            long profit = 0, bestProfit = 0;
            bestLen = 0;

            actions = new ActionType[solidCount];
            int curDumpTop = initialDumpTopZ;

            for (int i = 0; i < solidCount; i++)
            {
                int digZ = zList[i];
                int mass = massList[i];
                int val = valueList[i];

                long bestStepProfit;
                ActionType act;

                // Plant option always available
                long plant = PlantProfit(mass, val);
                bestStepProfit = plant;
                act = ActionType.Plant;

                // Adjacent dump option only for junk, if feasible
                if (canUseAdjacentDump && val == 0 && curDumpTop < zCount - 1)
                {
                    if (AdjacentDumpFeasibleAndFuel(digZ, curDumpTop, zCount, out int fuelPerTon))
                    {
                        long dumpProfit = -(long)fuelPerTon * mass;
                        if (dumpProfit > bestStepProfit)
                        {
                            bestStepProfit = dumpProfit;
                            act = ActionType.DumpAdjacent;
                        }
                    }
                }

                actions[i] = act;
                profit += bestStepProfit;

                // Apply dump height change if we chose adjacent dump
                if (act == ActionType.DumpAdjacent) curDumpTop++;

                if (profit > bestProfit)
                {
                    bestProfit = profit;
                    bestLen = i + 1;
                }
            }

            if (bestLen == 0)
            {
                Console.WriteLine("0");
                return;
            }

            // Trim actions to bestLen
            if (bestLen < actions.Length)
            {
                var trimmed = new ActionType[bestLen];
                Array.Copy(actions, trimmed, bestLen);
                actions = trimmed;
            }
        }

        // Emit operations:
        // For each mined voxel (in fixed top-down order), output:
        //   N digX digY digZ (N step pairs...) destX destY
        // We mine only from (0,0).
        var sbOut = new StringBuilder(bestLen * 32);
        sbOut.Append(bestLen).Append('\n');

        for (int i = 0; i < bestLen; i++)
        {
            int digZ = zList[i];
            if (actions[i] == ActionType.Plant)
            {
                // N=0, dump at plant (0,0)
                sbOut.Append("0 0 0 ").Append(digZ).Append(" 0 0\n");
            }
            else
            {
                // N=1, step to (dumpX,dumpY), dest = (dumpX,dumpY)
                sbOut.Append("1 0 0 ").Append(digZ).Append(' ')
                     .Append(dumpX).Append(' ').Append(dumpY).Append(' ')
                     .Append(dumpX).Append(' ').Append(dumpY).Append('\n');
            }
        }

        Console.Write(sbOut.ToString());
    }
}
```
