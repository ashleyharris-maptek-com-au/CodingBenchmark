```spirv
OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450

%ext = OpExtInstImport "GLSL.std.450"

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_in 0 Offset 0

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 Offset 0

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn = OpTypeFunction %void

%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%uint_2 = OpConstant %uint 2

%f0 = OpConstant %float 0
%f075 = OpConstant %float 0.75
%f0015 = OpConstant %float 0.015
%f005 = OpConstant %float 0.05

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %fn
%entry = OpLabel

; Load p0 = gl_in[0].gl_Position, p1 = gl_in[1].gl_Position
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

; Load colors c0, c1 from input color[]
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %c1_ptr

; dir4 = p1 - p0
%dir4 = OpFSub %v4float %p1 %p0

; dir2 = dir4.xy
%dirx = OpCompositeExtract %float %dir4 0
%diry = OpCompositeExtract %float %dir4 1
%dir2 = OpCompositeConstruct %v2float %dirx %diry

; perp2 = normalize( vec2(-dir.y, dir.x) )
%neg_diry = OpFNegate %float %diry
%perp2 = OpCompositeConstruct %v2float %neg_diry %dirx
%perp2n = OpExtInst %v2float %ext Normalize %perp2

; offset2_shaft = perp2n * 0.015
%offset2_shaft = OpVectorTimesScalar %v2float %perp2n %f0015
%offsx_s = OpCompositeExtract %float %offset2_shaft 0
%offsy_s = OpCompositeExtract %float %offset2_shaft 1
%offset4_shaft = OpCompositeConstruct %v4float %offsx_s %offsy_s %f0 %f0

; offset2_arrow = perp2n * 0.05
%offset2_arrow = OpVectorTimesScalar %v2float %perp2n %f005
%offsx_a = OpCompositeExtract %float %offset2_arrow 0
%offsy_a = OpCompositeExtract %float %offset2_arrow 1
%offset4_arrow = OpCompositeConstruct %v4float %offsx_a %offsy_a %f0 %f0

; mid = p0 + 0.75 * (p1 - p0)
%dir4_scaled = OpVectorTimesScalar %v4float %dir4 %f075
%mid = OpFAdd %v4float %p0 %dir4_scaled

; Pointers for outputs
%pos_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; ---- Shaft strip (4 verts): p0±off, mid±off; color = c0 ----
%p0_plus = OpFAdd %v4float %p0 %offset4_shaft
OpStore %pos_out_ptr %p0_plus
OpStore %outColor %c0
OpEmitVertex

%p0_minus = OpFSub %v4float %p0 %offset4_shaft
OpStore %pos_out_ptr %p0_minus
OpStore %outColor %c0
OpEmitVertex

%mid_plus = OpFAdd %v4float %mid %offset4_shaft
OpStore %pos_out_ptr %mid_plus
OpStore %outColor %c0
OpEmitVertex

%mid_minus = OpFSub %v4float %mid %offset4_shaft
OpStore %pos_out_ptr %mid_minus
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

; ---- Arrowhead strip (3 verts): mid±offArrow, p1; color = c1 ----
%amid_plus = OpFAdd %v4float %mid %offset4_arrow
OpStore %pos_out_ptr %amid_plus
OpStore %outColor %c1
OpEmitVertex

%amid_minus = OpFSub %v4float %mid %offset4_arrow
OpStore %pos_out_ptr %amid_minus
OpStore %outColor %c1
OpEmitVertex

OpStore %pos_out_ptr %p1
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
