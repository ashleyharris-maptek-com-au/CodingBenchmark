```rust
use std::cmp::min;
use std::io::{self, Read};

#[derive(Clone)]
struct Graph {
    n: usize,
    edges: Vec<(usize, usize)>,
    adj: Vec<Vec<usize>>,
    deg: Vec<usize>,
}

fn compute_cut(edges: &[(usize, usize)], in_a: &[bool]) -> i32 {
    let mut cut = 0i32;
    for &(u, v) in edges {
        if in_a[u] ^ in_a[v] {
            cut += 1;
        }
    }
    cut
}

struct Solver {
    g: Graph,
    half: usize,

    order: Vec<usize>,
    pos_in_order: Vec<usize>,

    // Current partial assignment state
    in_a: Vec<bool>,
    cnt_to_a: Vec<i16>, // for each vertex: #neighbors already assigned to A
    cnt_to_b: Vec<i16>, // for each vertex: #neighbors already assigned to B

    // Best solution found
    best_cut: i32,
    best_in_a: Vec<bool>,
}

impl Solver {
    fn new(g: Graph) -> Self {
        let n = g.n;
        let half = n / 2;

        // Vertex ordering: 0 fixed first (symmetry breaking), rest by decreasing degree.
        let mut verts: Vec<usize> = (0..n).collect();
        verts.sort_by_key(|&v| std::cmp::Reverse(g.deg[v]));
        // Ensure 0 is first.
        verts.retain(|&v| v != 0);
        let mut order = Vec::with_capacity(n);
        order.push(0);
        order.extend(verts);

        let mut pos_in_order = vec![0usize; n];
        for (i, &v) in order.iter().enumerate() {
            pos_in_order[v] = i;
        }

        Self {
            g,
            half,
            order,
            pos_in_order,
            in_a: vec![false; n],
            cnt_to_a: vec![0; n],
            cnt_to_b: vec![0; n],
            best_cut: i32::MAX,
            best_in_a: vec![false; n],
        }
    }

    fn greedy_initial_upper_bound(&mut self) {
        let n = self.g.n;

        let mut in_a = vec![false; n];
        let mut cnt_to_a = vec![0i16; n];
        let mut cnt_to_b = vec![0i16; n];

        // Fix vertex 0 in A.
        in_a[0] = true;
        let mut count_a = 1usize;
        let mut cut = 0i32;

        // Apply updates from placing 0 in A.
        let idx0 = 0usize;
        let v0 = self.order[idx0];
        debug_assert_eq!(v0, 0);
        for &w in &self.g.adj[v0] {
            if self.pos_in_order[w] > idx0 {
                cnt_to_a[w] += 1;
            }
        }

        for idx in 1..n {
            let v = self.order[idx];
            let remaining = n - idx; // including v
            let need_a = self.half.saturating_sub(count_a);

            // Forced assignments to satisfy exact size.
            let must_put_in_a = need_a == remaining;
            let must_put_in_b = count_a == self.half;

            let cost_if_a = cnt_to_b[v] as i32; // edges to assigned B
            let cost_if_b = cnt_to_a[v] as i32; // edges to assigned A

            let choose_a = if must_put_in_a {
                true
            } else if must_put_in_b {
                false
            } else {
                cost_if_a <= cost_if_b
            };

            if choose_a {
                in_a[v] = true;
                cut += cost_if_a;
                count_a += 1;
                for &w in &self.g.adj[v] {
                    if self.pos_in_order[w] > idx {
                        cnt_to_a[w] += 1;
                    }
                }
            } else {
                in_a[v] = false;
                cut += cost_if_b;
                for &w in &self.g.adj[v] {
                    if self.pos_in_order[w] > idx {
                        cnt_to_b[w] += 1;
                    }
                }
            }
        }

        if count_a == self.half {
            let exact_cut = compute_cut(&self.g.edges, &in_a);
            self.best_cut = exact_cut;
            self.best_in_a = in_a;
        }
    }

    // Lower bound on additional cut coming from edges between already-assigned vertices and unassigned vertices,
    // respecting the exact size constraint on A for the unassigned vertices.
    fn lower_bound(&self, idx: usize, count_a: usize, cut_so_far: i32) -> i32 {
        let n = self.g.n;
        let u = n - idx;
        let mut need_a = self.half as i32 - count_a as i32;
        if need_a < 0 || need_a as usize > u {
            return i32::MAX / 4;
        }

        // Base: assign all unassigned to B => cost is sum(cnt_to_a[v])
        // Switching v from B to A changes cost by delta = cnt_to_b[v] - cnt_to_a[v].
        let mut base = 0i32;

        // delta range is within [-n, n] (safe upper bound).
        let range = n as i32;
        let mut buckets = vec![0i32; (2 * range + 1) as usize];

        for j in idx..n {
            let v = self.order[j];
            let a = self.cnt_to_a[v] as i32;
            let b = self.cnt_to_b[v] as i32;
            base += a;
            let delta = b - a;
            buckets[(delta + range) as usize] += 1;
        }

        let mut add = 0i32;
        // pick smallest need_a deltas
        for d in -range..=range {
            if need_a == 0 {
                break;
            }
            let c = buckets[(d + range) as usize];
            if c == 0 {
                continue;
            }
            let take = min(need_a, c);
            add += d * take;
            need_a -= take;
        }

        cut_so_far + base + add
    }

    fn dfs(&mut self, idx: usize, count_a: usize, cut_so_far: i32) {
        let n = self.g.n;
        if idx == n {
            if count_a == self.half {
                if cut_so_far < self.best_cut {
                    self.best_cut = cut_so_far;
                    self.best_in_a.clone_from(&self.in_a);
                }
            }
            return;
        }

        // Feasibility pruning by size
        let remaining = n - idx;
        if count_a > self.half || count_a + remaining < self.half {
            return;
        }

        // Lower bound pruning
        let lb = self.lower_bound(idx, count_a, cut_so_far);
        if lb >= self.best_cut {
            return;
        }

        let v = self.order[idx];

        // Forced assignment if needed to reach exact size
        let need_a = self.half - count_a;
        let must_put_in_a = need_a == remaining;
        let must_put_in_b = count_a == self.half;

        let cost_if_a = self.cnt_to_b[v] as i32;
        let cost_if_b = self.cnt_to_a[v] as i32;

        // Try the more promising branch first (smaller immediate cost), unless forced.
        let mut branches: [(bool, i32); 2] = [(true, cost_if_a), (false, cost_if_b)];
        if branches[0].1 > branches[1].1 {
            branches.swap(0, 1);
        }

        for &(choose_a, add_cost) in &branches {
            if must_put_in_a && !choose_a {
                continue;
            }
            if must_put_in_b && choose_a {
                continue;
            }
            if choose_a && count_a + 1 > self.half {
                continue;
            }

            // Apply assignment of v
            self.in_a[v] = choose_a;
            let mut changes: Vec<usize> = Vec::new(); // store affected neighbor vertices

            if choose_a {
                // v in A: update cnt_to_a on its unassigned neighbors
                for &w in &self.g.adj[v] {
                    if self.pos_in_order[w] > idx {
                        self.cnt_to_a[w] += 1;
                        changes.push(w);
                    }
                }
                self.dfs(idx + 1, count_a + 1, cut_so_far + add_cost);
                // rollback
                for &w in &changes {
                    self.cnt_to_a[w] -= 1;
                }
            } else {
                // v in B: update cnt_to_b on its unassigned neighbors
                for &w in &self.g.adj[v] {
                    if self.pos_in_order[w] > idx {
                        self.cnt_to_b[w] += 1;
                        changes.push(w);
                    }
                }
                self.dfs(idx + 1, count_a, cut_so_far + add_cost);
                // rollback
                for &w in &changes {
                    self.cnt_to_b[w] -= 1;
                }
            }
        }
    }

    fn solve(&mut self) {
        // Start with a quick upper bound to help pruning.
        self.greedy_initial_upper_bound();
        if self.best_cut == i32::MAX {
            self.best_cut = i32::MAX / 4;
        }

        // Fix vertex 0 in A (symmetry breaking).
        let n = self.g.n;
        self.in_a.fill(false);
        self.cnt_to_a.fill(0);
        self.cnt_to_b.fill(0);

        self.in_a[0] = true;

        // Apply updates from placing 0 in A.
        let idx0 = 0usize;
        let v0 = self.order[idx0];
        debug_assert_eq!(v0, 0);
        for &w in &self.g.adj[v0] {
            if self.pos_in_order[w] > idx0 {
                self.cnt_to_a[w] += 1;
            }
        }

        // Recurse from idx=1 with count_a=1
        self.dfs(1, 1, 0);

        // Ensure reported cut matches the best partition (safety).
        self.best_cut = compute_cut(&self.g.edges, &self.best_in_a);

        // Ensure exactly half in A; if something went wrong, fallback to greedy partition.
        let count_a = self.best_in_a.iter().filter(|&&x| x).count();
        if count_a != self.half {
            // Fallback: deterministic partition first half (still must be minimum in valid tests)
            // but keep it safe format-wise.
            let mut in_a = vec![false; n];
            for v in 0..self.half {
                in_a[v] = true;
            }
            self.best_in_a = in_a;
            self.best_cut = compute_cut(&self.g.edges, &self.best_in_a);
        }
    }
}

fn main() {
    // Read input
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut edges = Vec::with_capacity(m);
    let mut adj = vec![Vec::<usize>::new(); n];
    let mut deg = vec![0usize; n];

    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        edges.push((u, v));
        adj[u].push(v);
        adj[v].push(u);
        deg[u] += 1;
        deg[v] += 1;
    }

    let g = Graph { n, edges, adj, deg };
    let mut solver = Solver::new(g);
    solver.solve();

    // Output
    let half = solver.half;
    let mut part_a: Vec<usize> = (0..n).filter(|&v| solver.best_in_a[v]).collect();
    part_a.sort_unstable();

    // As required: exactly n/2 vertices in partition A.
    // If due to fallback/edge-case it isn't, force format correctness.
    if part_a.len() != half {
        part_a.clear();
        for v in 0..half {
            part_a.push(v);
        }
    }

    println!("{}", solver.best_cut);
    for (i, v) in part_a.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
```
