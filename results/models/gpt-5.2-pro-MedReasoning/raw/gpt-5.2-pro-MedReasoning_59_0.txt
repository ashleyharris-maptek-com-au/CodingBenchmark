```javascript
'use strict';

// Read input
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.probability;
const drones = data.drones || [];
const maxStepsPerDrone = Math.max(0, data.maxStepsPerDrone | 0);

const H = grid.length | 0;
const W = H > 0 ? (grid[0].length | 0) : 0;
const N = (W * H) | 0;

function idxOf(x, y) { return (y * W + x) | 0; }
function xOf(idx) { return (idx % W) | 0; }
function yOf(idx) { return ((idx / W) | 0); }

// Flatten grid
const passable = new Uint8Array(N);
const value = new Float64Array(N);

for (let y = 0; y < H; y++) {
  const row = grid[y];
  for (let x = 0; x < W; x++) {
    const v = row[x];
    const idx = (y * W + x) | 0;
    if (v !== -1) {
      passable[idx] = 1;
      value[idx] = +v;
    }
  }
}

// Global tracking
const covered = new Uint8Array(N);     // only meaningful when value[idx] > 0
const visitCount = new Int32Array(N);  // visit counts by any drone

// Per-drone state
const D = drones.length | 0;
const curIdx = new Int32Array(D);
const planTarget = new Int32Array(D);
const planPos = new Int32Array(D);
const planLen = new Int32Array(D);
const planBuf = Array.from({ length: D }, () => new Int32Array(0)); // resized on demand

// Output storage: per-drone coordinates as Int32Array [x0,y0, x1,y1, ...]
const coordBuf = new Array(D);
const coordWritePos = new Int32Array(D); // counts ints written (2 per position)
const coordSize = 2 * (maxStepsPerDrone + 1);
for (let i = 0; i < D; i++) {
  coordBuf[i] = new Int32Array(coordSize);
  coordWritePos[i] = 0;
}

// Initialize drones
for (let i = 0; i < D; i++) {
  const dx = drones[i].x | 0;
  const dy = drones[i].y | 0;
  let idx = idxOf(dx, dy);

  // Assume valid starts; if not, clamp/stay in bounds and on passable if possible
  if (dx < 0 || dx >= W || dy < 0 || dy >= H || passable[idx] === 0) {
    // Fallback: find nearest passable (bounded BFS). If none, default to (0,0).
    let found = -1;
    if (N > 0) {
      const q = new Int32Array(Math.min(N, 16384));
      const seen = new Uint8Array(N);
      let head = 0, tail = 0;
      // try clamped start
      const cx = Math.min(W - 1, Math.max(0, dx));
      const cy = Math.min(H - 1, Math.max(0, dy));
      const start = idxOf(cx, cy);
      q[tail++] = start;
      seen[start] = 1;
      while (head < tail) {
        const v = q[head++];
        if (passable[v]) { found = v; break; }
        const x = v % W;
        if (x > 0) {
          const n = v - 1;
          if (!seen[n]) { seen[n] = 1; if (tail < q.length) q[tail++] = n; }
        }
        if (x + 1 < W) {
          const n = v + 1;
          if (!seen[n]) { seen[n] = 1; if (tail < q.length) q[tail++] = n; }
        }
        if (v >= W) {
          const n = v - W;
          if (!seen[n]) { seen[n] = 1; if (tail < q.length) q[tail++] = n; }
        }
        if (v + W < N) {
          const n = v + W;
          if (!seen[n]) { seen[n] = 1; if (tail < q.length) q[tail++] = n; }
        }
        if (tail >= q.length) break;
      }
    }
    idx = found >= 0 ? found : 0;
  }

  curIdx[i] = idx;
  planTarget[i] = -1;
  planPos[i] = 0;
  planLen[i] = 0;

  // record start
  const x = xOf(idx), y = yOf(idx);
  coordBuf[i][0] = x;
  coordBuf[i][1] = y;
  coordWritePos[i] = 2;

  visitCount[idx] += 1;
  if (value[idx] > 0 && covered[idx] === 0) covered[idx] = 1;
}

// Planner helpers
const MAX_L = 60; // bounded BFS radius
const queue = new Int32Array(1 + 2 * MAX_L * (MAX_L + 1) + 16); // safe for diamond BFS
const seenStamp = new Int32Array(N);
const distArr = new Int16Array(N);
const parentArr = new Int32Array(N);
let stamp = 0;

function ensurePlanCapacity(droneIndex, neededLen) {
  let buf = planBuf[droneIndex];
  if (buf.length < neededLen) {
    // grow to next power-ish
    let newLen = buf.length ? buf.length : 16;
    while (newLen < neededLen) newLen <<= 1;
    planBuf[droneIndex] = new Int32Array(newLen);
  }
}

function boundedBestPathToUncoveredValue(start, maxDepth) {
  if (N === 0) return 0;

  stamp = (stamp + 1) | 0;
  if (stamp === 0x7fffffff) {
    seenStamp.fill(0);
    stamp = 1;
  }

  let head = 0, tail = 0;
  queue[tail++] = start;
  seenStamp[start] = stamp;
  distArr[start] = 0;
  parentArr[start] = -1;

  let best = -1;
  let bestScore = -1;

  while (head < tail) {
    const v = queue[head++];
    const d = distArr[v] | 0;

    const vv = value[v];
    if (vv > 0 && covered[v] === 0) {
      // prefer high value, closer distance, and low visitation
      const score = (vv / (d + 1)) / (1 + (visitCount[v] | 0));
      if (score > bestScore) {
        bestScore = score;
        best = v;
      }
    }

    if (d >= maxDepth) continue;

    const x = v % W;

    // left
    if (x > 0) {
      const n = (v - 1) | 0;
      if (passable[n] && seenStamp[n] !== stamp) {
        seenStamp[n] = stamp;
        distArr[n] = (d + 1) | 0;
        parentArr[n] = v;
        queue[tail++] = n;
      }
    }
    // right
    if (x + 1 < W) {
      const n = (v + 1) | 0;
      if (passable[n] && seenStamp[n] !== stamp) {
        seenStamp[n] = stamp;
        distArr[n] = (d + 1) | 0;
        parentArr[n] = v;
        queue[tail++] = n;
      }
    }
    // up
    if (v >= W) {
      const n = (v - W) | 0;
      if (passable[n] && seenStamp[n] !== stamp) {
        seenStamp[n] = stamp;
        distArr[n] = (d + 1) | 0;
        parentArr[n] = v;
        queue[tail++] = n;
      }
    }
    // down
    if (v + W < N) {
      const n = (v + W) | 0;
      if (passable[n] && seenStamp[n] !== stamp) {
        seenStamp[n] = stamp;
        distArr[n] = (d + 1) | 0;
        parentArr[n] = v;
        queue[tail++] = n;
      }
    }

    // queue capacity should be enough for maxDepth <= MAX_L; if it overflows, stop expanding
    if (tail >= queue.length) break;
  }

  return best;
}

function writePlanFromParents(droneIndex, start, target) {
  if (target < 0 || target === start) {
    planLen[droneIndex] = 0;
    planPos[droneIndex] = 0;
    planTarget[droneIndex] = -1;
    return;
  }

  // Reconstruct backwards, then reverse into planBuf
  let len = 0;
  let cur = target;
  while (cur !== start && cur >= 0) {
    len++;
    cur = parentArr[cur];
  }
  if (cur !== start) {
    planLen[droneIndex] = 0;
    planPos[droneIndex] = 0;
    planTarget[droneIndex] = -1;
    return;
  }

  ensurePlanCapacity(droneIndex, len);
  const buf = planBuf[droneIndex];

  cur = target;
  for (let i = len - 1; i >= 0; i--) {
    buf[i] = cur;
    cur = parentArr[cur];
  }

  planLen[droneIndex] = len | 0;
  planPos[droneIndex] = 0;
  planTarget[droneIndex] = target | 0;
}

function chooseNeighborGreedy(idx) {
  // picks among stay + 4-neighbors: minimize visitCount, tie-break by higher value
  let best = idx;
  let bestVisits = visitCount[idx] | 0;
  let bestVal = value[idx];

  const x = idx % W;

  function consider(n) {
    if (!passable[n]) return;
    const vc = visitCount[n] | 0;
    const vv = value[n];
    if (vc < bestVisits || (vc === bestVisits && vv > bestVal)) {
      best = n;
      bestVisits = vc;
      bestVal = vv;
    }
  }

  if (x > 0) consider((idx - 1) | 0);
  if (x + 1 < W) consider((idx + 1) | 0);
  if (idx >= W) consider((idx - W) | 0);
  if (idx + W < N) consider((idx + W) | 0);

  return best;
}

// Main simulation
for (let step = 0; step < maxStepsPerDrone; step++) {
  const rem = (maxStepsPerDrone - step) | 0;
  const L = Math.min(MAX_L, rem) | 0;

  for (let i = 0; i < D; i++) {
    let cur = curIdx[i] | 0;

    // If current plan's target already covered, discard and replan
    if (planLen[i] > 0) {
      const tgt = planTarget[i] | 0;
      if (tgt >= 0 && value[tgt] > 0 && covered[tgt] === 1) {
        planLen[i] = 0;
        planPos[i] = 0;
        planTarget[i] = -1;
      }
    }

    // Replan if no plan
    if (planPos[i] >= planLen[i]) {
      planLen[i] = 0;
      planPos[i] = 0;
      planTarget[i] = -1;

      const target = boundedBestPathToUncoveredValue(cur, L);
      if (target >= 0 && target !== cur && value[target] > 0 && covered[target] === 0) {
        writePlanFromParents(i, cur, target);
      }
    }

    let next = cur;

    if (planPos[i] < planLen[i]) {
      next = planBuf[i][planPos[i]++] | 0;
      // safety: ensure next is passable; otherwise cancel plan and fallback
      if (!passable[next]) {
        planLen[i] = 0;
        planPos[i] = 0;
        planTarget[i] = -1;
        next = chooseNeighborGreedy(cur);
      }
    } else {
      next = chooseNeighborGreedy(cur); // includes possible stay
    }

    // Apply move
    curIdx[i] = next;

    // Record output coordinate
    const wp = coordWritePos[i] | 0;
    coordBuf[i][wp] = xOf(next);
    coordBuf[i][wp + 1] = yOf(next);
    coordWritePos[i] = (wp + 2) | 0;

    // Update global stats
    visitCount[next] += 1;
    if (value[next] > 0 && covered[next] === 0) covered[next] = 1;
  }
}

// Serialize output efficiently
let parts = [];
let partsLen = 0;
function pushOut(s) {
  parts.push(s);
  partsLen += s.length;
  if (partsLen >= (1 << 20)) {
    process.stdout.write(parts.join(''));
    parts = [];
    partsLen = 0;
  }
}

pushOut('{"paths":[');
for (let i = 0; i < D; i++) {
  if (i) pushOut(',');
  pushOut('[');
  const coords = coordBuf[i];
  const positions = maxStepsPerDrone + 1;
  for (let s = 0; s < positions; s++) {
    if (s) pushOut(',');
    const x = coords[(s << 1)] | 0;
    const y = coords[(s << 1) + 1] | 0;
    pushOut('[' + x + ',' + y + ']');
  }
  pushOut(']');
}
pushOut(']}');

if (parts.length) process.stdout.write(parts.join(''));
```
