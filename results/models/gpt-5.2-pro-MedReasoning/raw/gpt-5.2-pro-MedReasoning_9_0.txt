using System;
using System.IO;
using System.Runtime.CompilerServices;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        int n = fs.NextInt();
        int stockLen = fs.NextInt();

        if (n <= 0)
        {
            var fo0 = new FastOutput(Console.OpenStandardOutput());
            fo0.WriteInt(0);
            fo0.WriteNewLine();
            fo0.Flush();
            return;
        }

        // Pack (length, originalIndex) into a sortable 64-bit key:
        // high 32 bits = length (uint), low 32 bits = index (uint)
        // Sort ascending, then iterate from the end for decreasing length.
        ulong[] keys = new ulong[n];
        for (int i = 0; i < n; i++)
        {
            int len = fs.NextInt();
            // Constraint says len <= stockLen; assume valid.
            keys[i] = ((ulong)(uint)len << 32) | (uint)i;
        }

        RadixSortUInt64(keys);

        // Linked list of cuts per bin (stock piece): headCut[bin] -> cut -> cut...
        // nextCut[cutIdx] links cuts within a bin.
        int[] headCut = new int[n];      // max bins <= n
        int[] nextCut = new int[n];

        // For treap buckets of bins that share same remaining capacity:
        int[] nextBinSameRem = new int[n]; // binId -> next binId in that remainder bucket

        var treap = new RemTreap(maxNodes: n, nextBinSameRem);

        int binCount = 0;

        for (int t = n - 1; t >= 0; t--)
        {
            ulong k = keys[t];
            int len = (int)(k >> 32);
            int cutIdx = (int)(k & 0xFFFFFFFFu);

            if (!treap.TryPopLowerBound(len, out int binId, out int remBefore))
            {
                // New stock piece
                binId = binCount++;
                headCut[binId] = -1;

                nextCut[cutIdx] = headCut[binId];
                headCut[binId] = cutIdx;

                int newRem = stockLen - len;
                if (newRem > 0) treap.Insert(newRem, binId);
            }
            else
            {
                // Reuse an existing stock piece
                nextCut[cutIdx] = headCut[binId];
                headCut[binId] = cutIdx;

                int newRem = remBefore - len;
                if (newRem > 0) treap.Insert(newRem, binId);
            }
        }

        var fo = new FastOutput(Console.OpenStandardOutput());
        fo.WriteInt(binCount);
        fo.WriteNewLine();

        for (int b = 0; b < binCount; b++)
        {
            int cur = headCut[b];
            if (cur < 0)
            {
                fo.WriteNewLine();
                continue;
            }

            fo.WriteInt(cur);
            cur = nextCut[cur];
            while (cur >= 0)
            {
                fo.WriteSpace();
                fo.WriteInt(cur);
                cur = nextCut[cur];
            }
            fo.WriteNewLine();
        }

        fo.Flush();
    }

    // -------- Radix sort (LSD) for ulong[] (4 passes of 16 bits) --------
    private static void RadixSortUInt64(ulong[] a)
    {
        int n = a.Length;
        if (n <= 1) return;

        ulong[] tmp = new ulong[n];
        const int B = 1 << 16; // 65536
        int[] count = new int[B];

        ulong[] src = a;
        ulong[] dst = tmp;

        for (int pass = 0; pass < 4; pass++)
        {
            Array.Clear(count, 0, B);
            int shift = pass * 16;

            for (int i = 0; i < n; i++)
            {
                int bucket = (int)((src[i] >> shift) & 0xFFFF);
                count[bucket]++;
            }

            int sum = 0;
            for (int i = 0; i < B; i++)
            {
                int c = count[i];
                count[i] = sum;
                sum += c;
            }

            for (int i = 0; i < n; i++)
            {
                ulong v = src[i];
                int bucket = (int)((v >> shift) & 0xFFFF);
                dst[count[bucket]++] = v;
            }

            // swap src/dst
            var t = src; src = dst; dst = t;
        }

        // After 4 passes, src == a (even number of swaps), so already in place.
        // If you change number of passes, you may need a copy-back here.
    }

    // -------- Treap keyed by remaining capacity; each node holds a stack of binIds --------
    private sealed class RemTreap
    {
        private readonly int[] _key;
        private readonly int[] _prio;
        private readonly int[] _left;
        private readonly int[] _right;
        private readonly int[] _headBin; // head of binId stack for this remaining key

        private readonly int[] _nextBinSameRem; // external array: binId -> next binId in stack
        private int _root;
        private int _nodes;

        private uint _rng = 2463534242u;

        public RemTreap(int maxNodes, int[] nextBinSameRem)
        {
            // Node index 0 is "null"
            _key = new int[maxNodes + 1];
            _prio = new int[maxNodes + 1];
            _left = new int[maxNodes + 1];
            _right = new int[maxNodes + 1];
            _headBin = new int[maxNodes + 1];

            _nextBinSameRem = nextBinSameRem;
            _root = 0;
            _nodes = 0;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int NextPrio()
        {
            // xorshift32
            uint x = _rng;
            x ^= x << 13;
            x ^= x >> 17;
            x ^= x << 5;
            _rng = x;
            return (int)x;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int NewNode(int k, int binId)
        {
            int id = ++_nodes;
            _key[id] = k;
            _prio[id] = NextPrio();
            _left[id] = 0;
            _right[id] = 0;
            _headBin[id] = binId;
            _nextBinSameRem[binId] = -1;
            return id;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private static bool HigherPrio(int aPrio, int bPrio) => (uint)aPrio > (uint)bPrio;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int RotateRight(int p)
        {
            int q = _left[p];
            _left[p] = _right[q];
            _right[q] = p;
            return q;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int RotateLeft(int p)
        {
            int q = _right[p];
            _right[p] = _left[q];
            _left[q] = p;
            return q;
        }

        public void Insert(int k, int binId)
        {
            _root = Insert(_root, k, binId);
        }

        private int Insert(int node, int k, int binId)
        {
            if (node == 0) return NewNode(k, binId);

            int nk = _key[node];
            if (k == nk)
            {
                _nextBinSameRem[binId] = _headBin[node];
                _headBin[node] = binId;
                return node;
            }

            if (k < nk)
            {
                int l = Insert(_left[node], k, binId);
                _left[node] = l;
                if (l != 0 && HigherPrio(_prio[l], _prio[node])) node = RotateRight(node);
            }
            else
            {
                int r = Insert(_right[node], k, binId);
                _right[node] = r;
                if (r != 0 && HigherPrio(_prio[r], _prio[node])) node = RotateLeft(node);
            }

            return node;
        }

        public bool TryPopLowerBound(int need, out int binId, out int remKey)
        {
            int r = _root;
            bool ok = TryPopLowerBound(ref r, need, out binId, out remKey);
            _root = r;
            return ok;
        }

        private bool TryPopLowerBound(ref int node, int need, out int binId, out int remKey)
        {
            if (node == 0)
            {
                binId = -1;
                remKey = 0;
                return false;
            }

            int nk = _key[node];

            if (nk < need)
            {
                int r = _right[node];
                bool found = TryPopLowerBound(ref r, need, out binId, out remKey);
                _right[node] = r;
                return found;
            }
            else
            {
                int l = _left[node];
                if (TryPopLowerBound(ref l, need, out binId, out remKey))
                {
                    _left[node] = l;
                    return true;
                }

                // Use this node
                remKey = nk;
                binId = _headBin[node];
                _headBin[node] = _nextBinSameRem[binId]; // could become -1

                if (_headBin[node] == -1)
                {
                    node = Merge(_left[node], _right[node]);
                }
                return true;
            }
        }

        private int Merge(int a, int b)
        {
            if (a == 0) return b;
            if (b == 0) return a;

            if (HigherPrio(_prio[a], _prio[b]))
            {
                _right[a] = Merge(_right[a], b);
                return a;
            }
            else
            {
                _left[b] = Merge(a, _left[b]);
                return b;
            }
        }
    }

    // -------- Fast input --------
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buf;
        private int _len, _ptr;

        public FastScanner(Stream stream, int bufferSize = 1 << 16)
        {
            _stream = stream;
            _buf = new byte[bufferSize];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c < 0) return 0;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int x = 0;
            while (c > 32 && c >= 0)
            {
                x = x * 10 + (c - '0');
                c = ReadByte();
            }
            return x * sign;
        }
    }

    // -------- Fast output --------
    private sealed class FastOutput
    {
        private readonly Stream _stream;
        private readonly byte[] _buf;
        private int _ptr;

        public FastOutput(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buf = new byte[bufferSize];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private void WriteByte(byte b)
        {
            if (_ptr >= _buf.Length) Flush();
            _buf[_ptr++] = b;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteSpace() => WriteByte((byte)' ');

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteNewLine() => WriteByte((byte)'\n');

        public void WriteInt(int v)
        {
            if (v == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            if (v < 0)
            {
                WriteByte((byte)'-');
                v = -v;
            }

            Span<byte> tmp = stackalloc byte[11];
            int p = 0;
            while (v > 0)
            {
                int q = v / 10;
                tmp[p++] = (byte)('0' + (v - q * 10));
                v = q;
            }
            for (int i = p - 1; i >= 0; i--) WriteByte(tmp[i]);
        }

        public void Flush()
        {
            if (_ptr > 0)
            {
                _stream.Write(_buf, 0, _ptr);
                _ptr = 0;
            }
        }
    }
}
