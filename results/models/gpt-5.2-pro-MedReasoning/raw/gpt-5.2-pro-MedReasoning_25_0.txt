// Asteroid Kinetic-Impact Intercept Solver (heuristic, direct-transfer + optional midcourse correction)
// - Reads initial barycentric states for massive bodies (incl. "Earth" and "Asteroid").
// - Builds an N-body ephemeris using a KDK (kick-drift-kick) integrator over [0, warning_days].
// - Searches for a feasible direct Lambert transfer (Sun-centered patched conic) from Earth to Asteroid.
// - Validates and refines with one optional midcourse correction using finite-difference targeting in the full N-body field.
// - Estimates post-impact deflection (momentum transfer) and selects the best candidate.
// Notes:
// - The DE421 file path is read but not used here (the simulation host uses it internally).
// - This solver does not implement multi-flyby optimization; it attempts robust direct intercepts.

using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public static class Program
{
    const double G = 6.67430e-11;
    const double DaySec = 86400.0;

    public struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator *(double s, Vec3 a) => a * s;
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);

        public double Dot(in Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public Vec3 Cross(in Vec3 b) => new Vec3(
            Y * b.Z - Z * b.Y,
            Z * b.X - X * b.Z,
            X * b.Y - Y * b.X
        );

        public double Norm2() => X * X + Y * Y + Z * Z;
        public double Norm() => Math.Sqrt(Norm2());
        public Vec3 Normalized()
        {
            var n = Norm();
            return n > 0 ? this / n : new Vec3(0, 0, 0);
        }

        public override string ToString() => $"{X} {Y} {Z}";
    }

    public sealed class Body
    {
        public string Name = "";
        public double Mass;
        public Vec3 R;
        public Vec3 V;
    }

    public sealed class Ephemeris
    {
        public readonly int Steps;
        public readonly int N;
        public readonly double DtDays;
        public readonly double DtSec;
        public readonly Vec3[][] R; // [step][body]
        public readonly Vec3[][] V; // [step][body]
        public readonly double[] Masses;
        public readonly string[] Names;

        public Ephemeris(int steps, int n, double dtDays, double[] masses, string[] names)
        {
            Steps = steps;
            N = n;
            DtDays = dtDays;
            DtSec = dtDays * DaySec;
            Masses = masses;
            Names = names;
            R = new Vec3[steps + 1][];
            V = new Vec3[steps + 1][];
            for (int i = 0; i <= steps; i++)
            {
                R[i] = new Vec3[n];
                V[i] = new Vec3[n];
            }
        }

        public void GetState(int bodyIndex, double tDays, out Vec3 r, out Vec3 v)
        {
            if (tDays <= 0)
            {
                r = R[0][bodyIndex];
                v = V[0][bodyIndex];
                return;
            }
            double tMax = Steps * DtDays;
            if (tDays >= tMax)
            {
                r = R[Steps][bodyIndex];
                v = V[Steps][bodyIndex];
                return;
            }

            double f = tDays / DtDays;
            int i = (int)Math.Floor(f);
            if (i < 0) i = 0;
            if (i >= Steps) i = Steps - 1;
            double u = f - i;

            Vec3 r0 = R[i][bodyIndex];
            Vec3 r1 = R[i + 1][bodyIndex];
            Vec3 v0 = V[i][bodyIndex];
            Vec3 v1 = V[i + 1][bodyIndex];

            double h = DtSec;
            // Cubic Hermite interpolation in time
            double u2 = u * u;
            double u3 = u2 * u;

            double h00 = 2 * u3 - 3 * u2 + 1;
            double h10 = u3 - 2 * u2 + u;
            double h01 = -2 * u3 + 3 * u2;
            double h11 = u3 - u2;

            r = (h00 * r0) + (h10 * (h * v0)) + (h01 * r1) + (h11 * (h * v1));

            // Derivative of Hermite curve (velocity)
            double dh00 = (6 * u2 - 6 * u) / h;
            double dh10 = (3 * u2 - 4 * u + 1) / h;
            double dh01 = (-6 * u2 + 6 * u) / h;
            double dh11 = (3 * u2 - 2 * u) / h;

            v = (dh00 * r0) + (dh10 * (h * v0)) + (dh01 * r1) + (dh11 * (h * v1));
        }
    }

    static (double C, double S) Stumpff(double z)
    {
        if (Math.Abs(z) < 1e-12) return (0.5, 1.0 / 6.0);
        if (z > 0)
        {
            double sz = Math.Sqrt(z);
            double c = (1.0 - Math.Cos(sz)) / z;
            double s = (sz - Math.Sin(sz)) / (sz * sz * sz);
            return (c, s);
        }
        else
        {
            double sz = Math.Sqrt(-z);
            double c = (Math.Cosh(sz) - 1.0) / (-z);
            double s = (Math.Sinh(sz) - sz) / (sz * sz * sz);
            return (c, s);
        }
    }

    // Returns false if no 0-rev solution found with simple bracketing.
    static bool LambertUniversal(in Vec3 r1, in Vec3 r2, double dtSec, double mu, out Vec3 v1, out Vec3 v2)
    {
        v1 = default;
        v2 = default;

        double r1n = r1.Norm();
        double r2n = r2.Norm();
        if (r1n < 1 || r2n < 1 || dtSec <= 0 || mu <= 0) return false;

        double cosD = r1.Dot(r2) / (r1n * r2n);
        cosD = Math.Max(-1.0, Math.Min(1.0, cosD));

        Vec3 cr = r1.Cross(r2);
        double sinD = cr.Norm() / (r1n * r2n);
        if (sinD < 1e-12) return false; // near-collinear degeneracy

        // Short-way (0..pi) transfer:
        double A = sinD * Math.Sqrt(r1n * r2n / (1.0 - cosD));
        if (Math.Abs(A) < 1e-12) return false;

        double SqrtMu = Math.Sqrt(mu);

        double DtFromZ(double z)
        {
            var (C, S) = Stumpff(z);
            if (C <= 0) return double.NaN;
            double sqrtC = Math.Sqrt(C);
            double y = r1n + r2n + A * (z * S - 1.0) / sqrtC;
            if (y < 0) return double.NaN;
            double x = Math.Sqrt(y / C);
            return (x * x * x * S + A * Math.Sqrt(y)) / SqrtMu;
        }

        double F(double z)
        {
            double d = DtFromZ(z);
            if (double.IsNaN(d)) return double.NaN;
            return d - dtSec;
        }

        // Bracket a root for F(z)=0 by scanning.
        double zMin = -100.0;
        double zMax = 100.0;
        double step = 0.5;

        double zLo = double.NaN, zHi = double.NaN;
        double fPrev = double.NaN;
        double zPrev = zMin;

        for (double z = zMin; z <= zMax; z += step)
        {
            double f = F(z);
            if (double.IsNaN(f))
            {
                zPrev = z;
                fPrev = double.NaN;
                continue;
            }

            if (!double.IsNaN(fPrev) && fPrev * f <= 0)
            {
                zLo = zPrev;
                zHi = z;
                break;
            }

            zPrev = z;
            fPrev = f;
        }

        if (double.IsNaN(zLo))
        {
            // Try a smaller scan range for very short transfers
            zMin = -10; zMax = 10; step = 0.1;
            fPrev = double.NaN;
            zPrev = zMin;

            for (double z = zMin; z <= zMax; z += step)
            {
                double f = F(z);
                if (double.IsNaN(f))
                {
                    zPrev = z;
                    fPrev = double.NaN;
                    continue;
                }

                if (!double.IsNaN(fPrev) && fPrev * f <= 0)
                {
                    zLo = zPrev;
                    zHi = z;
                    break;
                }

                zPrev = z;
                fPrev = f;
            }
        }

        if (double.IsNaN(zLo)) return false;

        // Bisection
        double flo = F(zLo);
        double fhi = F(zHi);
        if (double.IsNaN(flo) || double.IsNaN(fhi)) return false;

        for (int it = 0; it < 80; it++)
        {
            double zm = 0.5 * (zLo + zHi);
            double fm = F(zm);
            if (double.IsNaN(fm))
            {
                // move slightly
                zLo = zm;
                continue;
            }
            if (Math.Abs(fm) < 1e-8) { zLo = zHi = zm; break; }
            if (flo * fm <= 0)
            {
                zHi = zm;
                fhi = fm;
            }
            else
            {
                zLo = zm;
                flo = fm;
            }
        }

        double zStar = 0.5 * (zLo + zHi);
        var (Cstar, Sstar) = Stumpff(zStar);
        if (Cstar <= 0) return false;

        double yStar = r1n + r2n + A * (zStar * Sstar - 1.0) / Math.Sqrt(Cstar);
        if (yStar < 0) return false;

        double fLam = 1.0 - yStar / r1n;
        double gLam = A * Math.Sqrt(yStar / mu);
        double gDot = 1.0 - yStar / r2n;

        if (Math.Abs(gLam) < 1e-12) return false;

        v1 = (r2 - fLam * r1) / gLam;
        v2 = (gDot * r2 - r1) / gLam;
        return true;
    }

    static void ComputePairwiseAccelerations(Vec3[] r, double[] masses, Vec3[] aOut)
    {
        int n = r.Length;
        for (int i = 0; i < n; i++) aOut[i] = new Vec3(0, 0, 0);

        for (int i = 0; i < n; i++)
        {
            for (int j = i + 1; j < n; j++)
            {
                Vec3 d = r[j] - r[i];
                double dist2 = d.Norm2() + 1e-6; // tiny softening
                double invDist = 1.0 / Math.Sqrt(dist2);
                double invDist3 = invDist * invDist * invDist;

                // a_i += G*mj*d/r^3 ; a_j -= G*mi*d/r^3
                Vec3 aij = (G * invDist3) * d;
                aOut[i] = aOut[i] + aij * masses[j];
                aOut[j] = aOut[j] - aij * masses[i];
            }
        }
    }

    static Ephemeris BuildEphemeris(List<Body> bodies, double warningDays, double dtDays)
    {
        int n = bodies.Count;
        int steps = (int)Math.Ceiling(warningDays / dtDays);
        var masses = bodies.Select(b => b.Mass).ToArray();
        var names = bodies.Select(b => b.Name).ToArray();

        var eph = new Ephemeris(steps, n, dtDays, masses, names);

        Vec3[] r = new Vec3[n];
        Vec3[] v = new Vec3[n];
        for (int i = 0; i < n; i++) { r[i] = bodies[i].R; v[i] = bodies[i].V; }

        Vec3[] a = new Vec3[n];
        double h = eph.DtSec;

        // store step 0
        for (int i = 0; i < n; i++) { eph.R[0][i] = r[i]; eph.V[0][i] = v[i]; }

        for (int s = 0; s < steps; s++)
        {
            // kick
            ComputePairwiseAccelerations(r, masses, a);
            for (int i = 0; i < n; i++) v[i] = v[i] + a[i] * (0.5 * h);

            // drift
            for (int i = 0; i < n; i++) r[i] = r[i] + v[i] * h;

            // kick
            ComputePairwiseAccelerations(r, masses, a);
            for (int i = 0; i < n; i++) v[i] = v[i] + a[i] * (0.5 * h);

            for (int i = 0; i < n; i++) { eph.R[s + 1][i] = r[i]; eph.V[s + 1][i] = v[i]; }
        }

        return eph;
    }

    static Vec3 AccelFromEphemeris(Ephemeris eph, double tDays, in Vec3 rSc, int skipBody = -1)
    {
        Vec3 a = new Vec3(0, 0, 0);
        for (int i = 0; i < eph.N; i++)
        {
            if (i == skipBody) continue;
            eph.GetState(i, tDays, out Vec3 rb, out _);
            Vec3 d = rb - rSc;
            double dist2 = d.Norm2() + 1e-6;
            double inv = 1.0 / Math.Sqrt(dist2);
            double inv3 = inv * inv * inv;
            a = a + d * (G * eph.Masses[i] * inv3);
        }
        return a;
    }

    static void PropagateSpacecraft(
        Ephemeris eph,
        double t0Days,
        in Vec3 r0,
        in Vec3 v0,
        double tEndDays,
        out Vec3 rEnd,
        out Vec3 vEnd,
        double? tBurnDays,
        in Vec3 dvBurn)
    {
        // KDK with variable final step to land exactly on target times; burn applied instantly at tBurnDays.
        Vec3 r = r0;
        Vec3 v = v0;
        double t = t0Days;

        double dtNom = eph.DtDays;
        double dtNomSec = eph.DtSec;

        bool burnApplied = !tBurnDays.HasValue;

        while (t < tEndDays - 1e-12)
        {
            double nextT = Math.Min(t + dtNom, tEndDays);

            // If burn time lies inside this step, split so we land exactly on burn time first.
            if (!burnApplied && tBurnDays!.Value > t + 1e-12 && tBurnDays.Value < nextT - 1e-12)
            {
                nextT = tBurnDays.Value;
            }

            double hDays = nextT - t;
            double h = hDays * DaySec;

            Vec3 a0 = AccelFromEphemeris(eph, t, r);
            v = v + a0 * (0.5 * h);
            r = r + v * h;
            t = nextT;
            Vec3 a1 = AccelFromEphemeris(eph, t, r);
            v = v + a1 * (0.5 * h);

            if (!burnApplied && Math.Abs(t - tBurnDays!.Value) <= 1e-9)
            {
                v = v + dvBurn;
                burnApplied = true;
            }
        }

        rEnd = r;
        vEnd = v;
    }

    static bool Solve3x3(double[,] A, double[] b, out double[] x)
    {
        // Gaussian elimination with partial pivoting.
        x = new double[3];
        double[,] M = new double[3, 4];
        for (int i = 0; i < 3; i++)
        {
            for (int j = 0; j < 3; j++) M[i, j] = A[i, j];
            M[i, 3] = b[i];
        }

        for (int col = 0; col < 3; col++)
        {
            int piv = col;
            double best = Math.Abs(M[col, col]);
            for (int r = col + 1; r < 3; r++)
            {
                double val = Math.Abs(M[r, col]);
                if (val > best) { best = val; piv = r; }
            }
            if (best < 1e-18) return false;
            if (piv != col)
            {
                for (int c = col; c < 4; c++)
                {
                    double tmp = M[col, c];
                    M[col, c] = M[piv, c];
                    M[piv, c] = tmp;
                }
            }

            double diag = M[col, col];
            for (int c = col; c < 4; c++) M[col, c] /= diag;

            for (int r = 0; r < 3; r++)
            {
                if (r == col) continue;
                double f = M[r, col];
                for (int c = col; c < 4; c++) M[r, c] -= f * M[col, c];
            }
        }

        x[0] = M[0, 3];
        x[1] = M[1, 3];
        x[2] = M[2, 3];
        return true;
    }

    static void Main()
    {
        CultureInfo inv = CultureInfo.InvariantCulture;
        string input = Console.In.ReadToEnd();
        if (string.IsNullOrWhiteSpace(input))
        {
            // Minimal valid output
            Console.WriteLine("2");
            Console.WriteLine("0 0 0 0");
            Console.WriteLine("0 0 0 0");
            return;
        }

        string[] tok = input.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        int p = 0;

        int numBodies = int.Parse(tok[p++], inv);
        double warningDays = double.Parse(tok[p++], inv);
        double impactorMass = double.Parse(tok[p++], inv);
        double dvBudget = double.Parse(tok[p++], inv);
        double asteroidMassInput = double.Parse(tok[p++], inv);
        double t0Unix = double.Parse(tok[p++], inv); // not used except reference

        var bodies = new List<Body>(numBodies);
        for (int i = 0; i < numBodies; i++)
        {
            var b = new Body();
            b.Name = tok[p++];
            b.Mass = double.Parse(tok[p++], inv);
            double x = double.Parse(tok[p++], inv);
            double y = double.Parse(tok[p++], inv);
            double z = double.Parse(tok[p++], inv);
            double vx = double.Parse(tok[p++], inv);
            double vy = double.Parse(tok[p++], inv);
            double vz = double.Parse(tok[p++], inv);
            b.R = new Vec3(x, y, z);
            b.V = new Vec3(vx, vy, vz);
            bodies.Add(b);
        }

        string de421Path = (p < tok.Length) ? tok[p++] : "";

        int idxEarth = bodies.FindIndex(b => b.Name == "Earth");
        int idxAst = bodies.FindIndex(b => b.Name == "Asteroid");
        int idxSun = bodies.FindIndex(b => b.Name == "Sun");

        if (idxEarth < 0 || idxAst < 0 || idxSun < 0)
        {
            // Fallback: print no-op burns
            Console.WriteLine("2");
            Console.WriteLine("0 0 0 0");
            Console.WriteLine($"{warningDays.ToString(inv)} 0 0 0");
            return;
        }

        // Prefer provided asteroid mass parameter if body mass looks inconsistent/zero.
        if (bodies[idxAst].Mass <= 0 && asteroidMassInput > 0) bodies[idxAst].Mass = asteroidMassInput;

        // Build N-body ephemeris over the warning horizon.
        double dtDays = Math.Clamp(warningDays / 20000.0, 0.25, 2.0); // 6h .. 2d
        var eph = BuildEphemeris(bodies, warningDays, dtDays);

        // Find time of closest approach Earth-Asteroid (as proxy for impact time).
        double tClosest = 0;
        double dClosest = double.PositiveInfinity;
        for (int s = 0; s <= eph.Steps; s++)
        {
            double t = s * eph.DtDays;
            Vec3 rE = eph.R[s][idxEarth];
            Vec3 rA = eph.R[s][idxAst];
            double d = (rA - rE).Norm();
            if (d < dClosest)
            {
                dClosest = d;
                tClosest = t;
            }
        }
        // Ensure intercept happens before that.
        double latestIntercept = Math.Max(0.0, Math.Min(warningDays, tClosest - 5.0));

        // Sun mu for Lambert (patched conic).
        double muSun = G * bodies[idxSun].Mass;

        // Generate coarse candidate set using Lambert on heliocentric vectors.
        int launchSamples = 25;
        int tofSamples = 32;
        double minTofDays = 10.0;

        var candidates = new List<Candidate>(launchSamples * tofSamples);

        double launchMax = Math.Min(warningDays, latestIntercept);
        if (launchMax < 0.1) launchMax = Math.Min(warningDays, 0.1);

        for (int i = 0; i < launchSamples; i++)
        {
            double tL = (launchSamples == 1) ? 0.0 : (launchMax * i / (launchSamples - 1));

            // TOF upper bound for this launch
            double maxTof = latestIntercept - tL;
            if (maxTof < minTofDays) continue;

            // Log-ish spacing for TOF: covers short and long transfers
            for (int k = 0; k < tofSamples; k++)
            {
                double u = (tofSamples == 1) ? 0.0 : (k / (double)(tofSamples - 1));
                // Blend between linear and exponential
                double tof = minTofDays * Math.Pow(maxTof / minTofDays, u);
                double tI = tL + tof;
                if (tI > latestIntercept) continue;

                eph.GetState(idxSun, tL, out Vec3 rS_L, out Vec3 vS_L);
                eph.GetState(idxSun, tI, out Vec3 rS_I, out Vec3 vS_I);
                eph.GetState(idxEarth, tL, out Vec3 rE_L, out Vec3 vE_L);
                eph.GetState(idxAst, tI, out Vec3 rA_I, out Vec3 vA_I);

                Vec3 r1 = rE_L - rS_L; // heliocentric
                Vec3 r2 = rA_I - rS_I; // heliocentric (approx)

                double dtSec = (tI - tL) * DaySec;
                if (!LambertUniversal(r1, r2, dtSec, muSun, out Vec3 v1_helio, out _))
                    continue;

                Vec3 vE_helio = vE_L - vS_L;
                Vec3 dvLaunchHelio = v1_helio - vE_helio;
                double dvMag = dvLaunchHelio.Norm();

                if (dvMag > dvBudget * 1.05) continue; // early prune

                // Convert to barycentric dv (spacecraft starts with Earth's barycentric velocity).
                Vec3 vScBary = v1_helio + vS_L;
                Vec3 dvLaunchBary = vScBary - vE_L;

                candidates.Add(new Candidate
                {
                    tLaunch = tL,
                    tIntercept = tI,
                    dvLaunchBary = dvLaunchBary,
                    dvLaunchMag = dvLaunchBary.Norm()
                });
            }
        }

        candidates.Sort((a, b) => a.dvLaunchMag.CompareTo(b.dvLaunchMag));
        int keep = Math.Min(24, candidates.Count);
        candidates = candidates.Take(keep).ToList();

        // Evaluate candidates with full N-body propagation + one midcourse correction targeting.
        const double interceptTol = 1_000_000.0; // 1000 km
        const double safeMiss = 7_371_000.0;     // Earth radius + 1000 km (approx), meters

        Solution best = null;

        foreach (var c in candidates)
        {
            // Initial spacecraft state: at Earth's center at launch, with Earth's barycentric velocity.
            eph.GetState(idxEarth, c.tLaunch, out Vec3 rE_L, out Vec3 vE_L);
            Vec3 r0 = rE_L;
            Vec3 v0 = vE_L + c.dvLaunchBary;

            // First propagate without correction
            PropagateSpacecraft(eph, c.tLaunch, r0, v0, c.tIntercept, out Vec3 rSc_I_nom, out Vec3 vSc_I_nom, null, new Vec3(0, 0, 0));
            eph.GetState(idxAst, c.tIntercept, out Vec3 rA_I, out Vec3 vA_I);

            double missNom = (rSc_I_nom - rA_I).Norm();

            Vec3 dvCorr = new Vec3(0, 0, 0);
            bool hasCorr = false;
            double tCorr = 0.5 * (c.tLaunch + c.tIntercept);

            Vec3 rSc_I = rSc_I_nom;
            Vec3 vSc_I = vSc_I_nom;

            double dvTotal = c.dvLaunchBary.Norm();

            if (missNom > interceptTol)
            {
                // One targeting correction at midcourse using finite-difference sensitivity of final position to dv at tCorr.
                // First propagate to tCorr:
                PropagateSpacecraft(eph, c.tLaunch, r0, v0, tCorr, out Vec3 rSc_C, out Vec3 vSc_C, null, new Vec3(0, 0, 0));

                // Nominal from tCorr to intercept:
                PropagateSpacecraft(eph, tCorr, rSc_C, vSc_C, c.tIntercept, out Vec3 rNom, out Vec3 vNom, null, new Vec3(0, 0, 0));

                Vec3 dr = rA_I - rNom;

                double eps = 1.0; // 1 m/s impulse basis
                Vec3[] cols = new Vec3[3];

                for (int ax = 0; ax < 3; ax++)
                {
                    Vec3 dv = ax == 0 ? new Vec3(eps, 0, 0) : ax == 1 ? new Vec3(0, eps, 0) : new Vec3(0, 0, eps);
                    PropagateSpacecraft(eph, tCorr, rSc_C, vSc_C + dv, c.tIntercept, out Vec3 rPert, out _, null, new Vec3(0, 0, 0));
                    cols[ax] = (rPert - rNom) / eps; // meters per (m/s)
                }

                double[,] A = new double[3, 3]
                {
                    { cols[0].X, cols[1].X, cols[2].X },
                    { cols[0].Y, cols[1].Y, cols[2].Y },
                    { cols[0].Z, cols[1].Z, cols[2].Z },
                };
                double[] b = new double[3] { dr.X, dr.Y, dr.Z };

                if (!Solve3x3(A, b, out double[] sol))
                    continue;

                dvCorr = new Vec3(sol[0], sol[1], sol[2]);
                double corrMag = dvCorr.Norm();
                if (!double.IsFinite(corrMag)) continue;

                if (dvTotal + corrMag > dvBudget) continue;

                // Validate corrected trajectory: propagate with burn at tCorr.
                PropagateSpacecraft(eph, c.tLaunch, r0, v0, c.tIntercept, out rSc_I, out vSc_I, tCorr, dvCorr);
                double miss = (rSc_I - rA_I).Norm();
                if (miss > interceptTol) continue;

                dvTotal += corrMag;
                hasCorr = corrMag > 1e-6;
            }
            else
            {
                // already within tolerance
                rSc_I = rSc_I_nom;
                vSc_I = vSc_I_nom;
            }

            // Estimate asteroid deflection from kinetic impact, then propagate asteroid to end and compute min Earth distance.
            // Perfectly inelastic momentum transfer (approx):
            Vec3 dvAst = (impactorMass / Math.Max(1.0, bodies[idxAst].Mass)) * (vSc_I - vA_I);
            Vec3 rAst0 = rA_I;
            Vec3 vAst0 = vA_I + dvAst;

            // Propagate asteroid from intercept to warningDays under gravity of other bodies (from ephemeris).
            Vec3 rAst = rAst0;
            Vec3 vAst = vAst0;
            double t = c.tIntercept;

            double minDist = double.PositiveInfinity;
            double dtStepDays = eph.DtDays;

            while (t < warningDays - 1e-12)
            {
                // update min distance at current time
                eph.GetState(idxEarth, t, out Vec3 rE, out _);
                double d = (rAst - rE).Norm();
                if (d < minDist) minDist = d;

                double nextT = Math.Min(t + dtStepDays, warningDays);
                double hDays = nextT - t;
                double h = hDays * DaySec;

                Vec3 a0 = AccelFromEphemeris(eph, t, rAst, skipBody: idxAst);
                vAst = vAst + a0 * (0.5 * h);
                rAst = rAst + vAst * h;
                t = nextT;
                Vec3 a1 = AccelFromEphemeris(eph, t, rAst, skipBody: idxAst);
                vAst = vAst + a1 * (0.5 * h);
            }

            // final sample
            eph.GetState(idxEarth, warningDays, out Vec3 rEend, out _);
            minDist = Math.Min(minDist, (rAst - rEend).Norm());

            var soln = new Solution
            {
                tLaunch = c.tLaunch,
                tCorr = tCorr,
                tImpact = c.tIntercept,
                dvLaunch = c.dvLaunchBary,
                dvCorr = dvCorr,
                hasCorr = hasCorr,
                dvTotal = dvTotal,
                minEarthDistAfterImpact = minDist,
                missAtIntercept = (rSc_I - rA_I).Norm()
            };

            if (best == null ||
                soln.minEarthDistAfterImpact > best.minEarthDistAfterImpact ||
                (Math.Abs(soln.minEarthDistAfterImpact - best.minEarthDistAfterImpact) < 1e3 && soln.dvTotal < best.dvTotal))
            {
                best = soln;

                // Stop early if we appear to have a safe miss.
                if (best.minEarthDistAfterImpact > safeMiss) break;
            }
        }

        // If no candidate found, output no-op within constraints.
        if (best == null)
        {
            Console.WriteLine("2");
            Console.WriteLine("0 0 0 0");
            Console.WriteLine($"{Math.Min(warningDays, Math.Max(0.0, latestIntercept)).ToString(inv)} 0 0 0");
            return;
        }

        // Output burns:
        // Always include launch and impact lines; include correction if used.
        // Burn vectors are barycentric delta-v (m/s).
        var lines = new List<string>();

        lines.Add($"{best.tLaunch.ToString(inv)} {best.dvLaunch.X.ToString(inv)} {best.dvLaunch.Y.ToString(inv)} {best.dvLaunch.Z.ToString(inv)}");
        if (best.hasCorr)
        {
            lines.Add($"{best.tCorr.ToString(inv)} {best.dvCorr.X.ToString(inv)} {best.dvCorr.Y.ToString(inv)} {best.dvCorr.Z.ToString(inv)}");
        }
        // Impact "burn" line (0); simulation will treat intercept event as kinetic impact.
        lines.Add($"{best.tImpact.ToString(inv)} 0 0 0");

        Console.WriteLine(lines.Count.ToString(inv));
        foreach (var ln in lines) Console.WriteLine(ln);
    }
}
