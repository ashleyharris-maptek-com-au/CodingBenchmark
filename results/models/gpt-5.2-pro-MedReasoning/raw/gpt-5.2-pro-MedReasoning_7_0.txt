// CSG Union (triangle meshes) - scalable BVH-based boolean with triangle splitting along intersection chords.
// Assumptions (as stated): clean, manifold, watertight, no self-intersections.
// Notes:
//  - Handles disjoint / containment quickly.
//  - For intersecting meshes: finds triangle-triangle intersection segments, splits only affected triangles,
//    classifies fragments via point-in-solid ray casting with BVH, outputs union boundary.
//  - Coplanar overlaps are treated as "no intersection" (assumes general position / no coplanar face overlap).
//
// MSVC /std:c++20 compatible. No non-standard headers.

static constexpr double INF_D = std::numeric_limits<double>::infinity();

struct FastScanner {
    static constexpr size_t BUFSZ = 1u << 20;
    unsigned char buf[BUFSZ];
    size_t idx = 0, size = 0;

    inline unsigned char read() {
        if (idx >= size) {
            size = std::fread(buf, 1, BUFSZ, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    template <class T>
    bool readInt(T &out) {
        unsigned char c;
        do {
            c = read();
            if (!c) return false;
        } while (c <= ' ');

        bool neg = false;
        if (c == '-') { neg = true; c = read(); }
        T val = 0;
        while (c > ' ') {
            if (c < '0' || c > '9') return false;
            val = (T)(val * 10 + (c - '0'));
            c = read();
        }
        out = neg ? -val : val;
        return true;
    }

    bool readDouble(double &out) {
        unsigned char c;
        do {
            c = read();
            if (!c) return false;
        } while (c <= ' ');

        bool neg = false;
        if (c == '-') { neg = true; c = read(); }

        double val = 0.0;
        while (c > ' ' && c != '.' && c != 'e' && c != 'E') {
            if (c < '0' || c > '9') return false;
            val = val * 10.0 + (c - '0');
            c = read();
        }
        if (c == '.') {
            c = read();
            double base = 0.1;
            while (c > ' ' && c != 'e' && c != 'E') {
                if (c < '0' || c > '9') return false;
                val += (c - '0') * base;
                base *= 0.1;
                c = read();
            }
        }
        if (c == 'e' || c == 'E') {
            c = read();
            bool eneg = false;
            if (c == '-') { eneg = true; c = read(); }
            else if (c == '+') { c = read(); }
            int exp = 0;
            while (c > ' ') {
                if (c < '0' || c > '9') return false;
                exp = exp * 10 + (c - '0');
                c = read();
            }
            double pow10 = std::pow(10.0, eneg ? -exp : exp);
            val *= pow10;
        }

        out = neg ? -val : val;
        return true;
    }
};

struct FastOutput {
    static constexpr size_t BUFSZ = 1u << 20;
    char buf[BUFSZ];
    size_t idx = 0;

    ~FastOutput() { flush(); }

    inline void flush() {
        if (idx) {
            std::fwrite(buf, 1, idx, stdout);
            idx = 0;
        }
    }

    inline void putChar(char c) {
        if (idx >= BUFSZ) flush();
        buf[idx++] = c;
    }

    inline void writeStr(const char* s) {
        while (*s) putChar(*s++);
    }

    void writeInt64(int64_t x) {
        if (x == 0) { putChar('0'); return; }
        if (x < 0) { putChar('-'); x = -x; }
        char tmp[32];
        int n = 0;
        while (x) { tmp[n++] = char('0' + (x % 10)); x /= 10; }
        while (n--) putChar(tmp[n]);
    }

    void writeUInt64(uint64_t x) {
        if (x == 0) { putChar('0'); return; }
        char tmp[32];
        int n = 0;
        while (x) { tmp[n++] = char('0' + (x % 10)); x /= 10; }
        while (n--) putChar(tmp[n]);
    }

    void writeDouble(double v) {
        // Print with enough precision for robust reconstruction.
        // Using snprintf is slower but acceptable; output dominates for huge meshes anyway.
        char tmp[64];
        int n = std::snprintf(tmp, sizeof(tmp), "%.17g", v);
        for (int i = 0; i < n; i++) putChar(tmp[i]);
    }
};

struct Vec3 {
    double x=0,y=0,z=0;
    Vec3() = default;
    Vec3(double X,double Y,double Z):x(X),y(Y),z(Z){}
};
static inline Vec3 operator+(const Vec3&a,const Vec3&b){return {a.x+b.x,a.y+b.y,a.z+b.z};}
static inline Vec3 operator-(const Vec3&a,const Vec3&b){return {a.x-b.x,a.y-b.y,a.z-b.z};}
static inline Vec3 operator*(const Vec3&a,double s){return {a.x*s,a.y*s,a.z*s};}
static inline Vec3 operator/(const Vec3&a,double s){return {a.x/s,a.y/s,a.z/s};}
static inline double dot(const Vec3&a,const Vec3&b){return a.x*b.x+a.y*b.y+a.z*b.z;}
static inline Vec3 cross(const Vec3&a,const Vec3&b){
    return {a.y*b.z-a.z*b.y, a.z*b.x-a.x*b.z, a.x*b.y-a.y*b.x};
}
static inline double norm2(const Vec3&a){return dot(a,a);}
static inline double norm(const Vec3&a){return std::sqrt(norm2(a));}
static inline Vec3 normalize(const Vec3&a){
    double n = norm(a);
    if (n == 0) return {0,0,0};
    return a / n;
}

struct AABB {
    Vec3 mn{ INF_D, INF_D, INF_D };
    Vec3 mx{ -INF_D, -INF_D, -INF_D };
};
static inline void expand(AABB& b, const Vec3& p){
    b.mn.x = std::min(b.mn.x, p.x);
    b.mn.y = std::min(b.mn.y, p.y);
    b.mn.z = std::min(b.mn.z, p.z);
    b.mx.x = std::max(b.mx.x, p.x);
    b.mx.y = std::max(b.mx.y, p.y);
    b.mx.z = std::max(b.mx.z, p.z);
}
static inline void expand(AABB& b, const AABB& a){
    expand(b, a.mn);
    expand(b, a.mx);
}
static inline bool overlap(const AABB&a,const AABB&b){
    return (a.mn.x <= b.mx.x && a.mx.x >= b.mn.x) &&
           (a.mn.y <= b.mx.y && a.mx.y >= b.mn.y) &&
           (a.mn.z <= b.mx.z && a.mx.z >= b.mn.z);
}
static inline bool contains(const AABB&a,const Vec3&p){
    return p.x>=a.mn.x && p.x<=a.mx.x && p.y>=a.mn.y && p.y<=a.mx.y && p.z>=a.mn.z && p.z<=a.mx.z;
}

struct Tri { int a,b,c; };

struct BVHNode {
    AABB box;
    int left = -1, right = -1;
    int start = 0, count = 0; // leaf if count>0
};

struct Mesh {
    // vertices are in global shared vertex array
    vector<Tri> tris;
    vector<AABB> triBox;
    vector<Vec3> triCentroid;
    AABB box;
};

static inline AABB triAABB(const Vec3& p0, const Vec3& p1, const Vec3& p2){
    AABB b;
    expand(b,p0); expand(b,p1); expand(b,p2);
    return b;
}

struct BVH {
    const vector<Vec3>* verts = nullptr;
    const vector<Tri>* tris = nullptr;
    vector<int> indices;
    vector<BVHNode> nodes;
    int leafSize = 8;

    int buildNode(int start, int count, const vector<Vec3>& centroids) {
        BVHNode node;
        node.start = start;
        node.count = count;

        // bbox for this node
        AABB bb;
        for (int i = 0; i < count; i++) {
            int ti = indices[start + i];
            const Tri& t = (*tris)[ti];
            expand(bb, (*verts)[t.a]);
            expand(bb, (*verts)[t.b]);
            expand(bb, (*verts)[t.c]);
        }
        node.box = bb;

        int myIndex = (int)nodes.size();
        nodes.push_back(node);

        if (count <= leafSize) return myIndex;

        // centroid bbox
        AABB cb;
        for (int i = 0; i < count; i++) {
            expand(cb, centroids[ indices[start + i] ]);
        }
        Vec3 ext = cb.mx - cb.mn;
        int axis = 0;
        if (ext.y > ext.x && ext.y >= ext.z) axis = 1;
        else if (ext.z > ext.x && ext.z >= ext.y) axis = 2;

        int mid = start + count/2;
        auto key = [&](int triIdx)->double{
            const Vec3& c = centroids[triIdx];
            return axis==0?c.x:(axis==1?c.y:c.z);
        };
        std::nth_element(indices.begin()+start, indices.begin()+mid, indices.begin()+start+count,
            [&](int lhs,int rhs){ return key(lhs) < key(rhs); });

        int left = buildNode(start, mid-start, centroids);
        int right = buildNode(mid, start+count-mid, centroids);

        nodes[myIndex].left = left;
        nodes[myIndex].right = right;
        nodes[myIndex].start = 0;
        nodes[myIndex].count = 0;
        // bbox already set
        return myIndex;
    }

    int build(const vector<Vec3>& V, const vector<Tri>& T, const vector<Vec3>& centroids) {
        verts = &V;
        tris = &T;
        indices.resize((int)T.size());
        std::iota(indices.begin(), indices.end(), 0);
        nodes.clear();
        nodes.reserve((int)T.size()*2);
        if (T.empty()) return -1;
        return buildNode(0, (int)T.size(), centroids);
    }
};

static inline bool rayAABB(const Vec3& ro, const Vec3& rd, const AABB& b, double& tminOut, double& tmaxOut) {
    double tmin = 0.0;
    double tmax = INF_D;

    auto slab = [&](double roC, double rdC, double mn, double mx)->bool{
        if (std::abs(rdC) < 1e-30) {
            return (roC >= mn && roC <= mx);
        }
        double inv = 1.0 / rdC;
        double t0 = (mn - roC) * inv;
        double t1 = (mx - roC) * inv;
        if (t0 > t1) std::swap(t0,t1);
        tmin = std::max(tmin, t0);
        tmax = std::min(tmax, t1);
        return tmin <= tmax;
    };

    if (!slab(ro.x, rd.x, b.mn.x, b.mx.x)) return false;
    if (!slab(ro.y, rd.y, b.mn.y, b.mx.y)) return false;
    if (!slab(ro.z, rd.z, b.mn.z, b.mx.z)) return false;

    tminOut = tmin;
    tmaxOut = tmax;
    return true;
}

static inline bool rayTriMT(const Vec3& ro, const Vec3& rd,
                           const Vec3& v0, const Vec3& v1, const Vec3& v2,
                           double eps, double& tHit) {
    // Möller–Trumbore
    Vec3 e1 = v1 - v0;
    Vec3 e2 = v2 - v0;
    Vec3 pvec = cross(rd, e2);
    double det = dot(e1, pvec);
    if (std::abs(det) < eps) return false;
    double invDet = 1.0 / det;
    Vec3 tvec = ro - v0;
    double u = dot(tvec, pvec) * invDet;
    if (u <= eps || u >= 1.0 - eps) return false; // avoid edge hits
    Vec3 qvec = cross(tvec, e1);
    double v = dot(rd, qvec) * invDet;
    if (v <= eps || u + v >= 1.0 - eps) return false; // avoid edge hits
    double t = dot(e2, qvec) * invDet;
    if (t <= eps) return false;
    tHit = t;
    return true;
}

static inline bool pointInMesh(const Vec3& p, const Mesh& M,
                               const vector<Vec3>& V,
                               const BVH& bvh, int root,
                               double epsRay) {
    if (root < 0) return false;
    if (!contains(M.box, p)) return false;

    // Deterministic "jitter" to avoid degeneracies
    Vec3 rd = normalize(Vec3{1.0, 0.327, 0.123});
    Vec3 ro = p;
    ro.x += std::sin(p.y*12.9898 + p.z*78.233) * (epsRay*0.25);
    ro.y += std::sin(p.z*37.719 + p.x*11.123) * (epsRay*0.25);
    ro.z += std::sin(p.x*93.989 + p.y*67.345) * (epsRay*0.25);

    int hits = 0;
    // stack traversal
    int stack[64];
    int sp = 0;
    stack[sp++] = root;

    while (sp) {
        int ni = stack[--sp];
        const BVHNode& n = bvh.nodes[ni];
        double tmin, tmax;
        if (!rayAABB(ro, rd, n.box, tmin, tmax)) continue;

        if (n.count > 0) {
            for (int i = 0; i < n.count; i++) {
                int ti = bvh.indices[n.start + i];
                const Tri& t = M.tris[ti];
                double th;
                if (rayTriMT(ro, rd, V[t.a], V[t.b], V[t.c], epsRay, th)) hits++;
            }
        } else {
            if (n.left >= 0) stack[sp++] = n.left;
            if (n.right >= 0) stack[sp++] = n.right;
        }
    }
    return (hits & 1) != 0;
}

struct QKey {
    int64_t x=0,y=0,z=0;
    bool operator==(const QKey& o) const { return x==o.x && y==o.y && z==o.z; }
    bool operator<(const QKey& o) const {
        if (x!=o.x) return x<o.x;
        if (y!=o.y) return y<o.y;
        return z<o.z;
    }
};

struct QHash {
    size_t operator()(const QKey& k) const noexcept {
        // splitmix64-ish combine
        auto mix = [](uint64_t v)->uint64_t {
            v += 0x9e3779b97f4a7c15ULL;
            v = (v ^ (v >> 30)) * 0xbf58476d1ce4e5b9ULL;
            v = (v ^ (v >> 27)) * 0x94d049bb133111ebULL;
            return v ^ (v >> 31);
        };
        uint64_t h1 = mix((uint64_t)k.x);
        uint64_t h2 = mix((uint64_t)k.y);
        uint64_t h3 = mix((uint64_t)k.z);
        return (size_t)(h1 ^ (h2<<1) ^ (h3<<7));
    }
};

static inline int64_t qround(double v, double invTol) {
    // llround can be slow; still fine.
    long double x = (long double)v * (long double)invTol;
    long double r = std::floor(x + (x >= 0 ? 0.5L : -0.5L));
    if (r > (long double)std::numeric_limits<int64_t>::max()) return std::numeric_limits<int64_t>::max();
    if (r < (long double)std::numeric_limits<int64_t>::min()) return std::numeric_limits<int64_t>::min();
    return (int64_t)r;
}

static inline QKey quantize(const Vec3& p, double invTol) {
    return { qround(p.x, invTol), qround(p.y, invTol), qround(p.z, invTol) };
}

static inline bool close3(const Vec3& a, const Vec3& b, double tol) {
    return std::abs(a.x-b.x) <= tol && std::abs(a.y-b.y) <= tol && std::abs(a.z-b.z) <= tol;
}

struct VertexManager {
    vector<Vec3>& verts; // global output vertex list; first originalUniqueCount are welded input verts
    const vector<std::pair<QKey,int>>& sortedOrig; // sorted by QKey for original welded vertices only
    int originalUniqueCount = 0;

    double tol = 0;
    double invTol = 1.0;

    std::unordered_map<QKey, vector<int>, QHash> newMap; // QKey -> indices (for newly created vertices)

    int findInOrig(const Vec3& p) const {
        QKey q = quantize(p, invTol);
        auto lo = std::lower_bound(sortedOrig.begin(), sortedOrig.end(), std::make_pair(q, -1),
            [](auto const& a, auto const& b){ return a.first < b.first; });
        for (auto it = lo; it != sortedOrig.end() && !(q < it->first) && !(it->first < q); ++it) {
            int idx = it->second;
            if (close3(verts[idx], p, tol)) return idx;
        }
        return -1;
    }

    int getOrCreate(const Vec3& p) {
        QKey q = quantize(p, invTol);
        auto it = newMap.find(q);
        if (it != newMap.end()) {
            for (int idx : it->second) {
                if (close3(verts[idx], p, tol)) return idx;
            }
        }
        int o = findInOrig(p);
        if (o >= 0) return o;
        int idx = (int)verts.size();
        verts.push_back(p);
        newMap[q].push_back(idx);
        return idx;
    }

    int getExistingOrCreate(const Vec3& p) { return getOrCreate(p); }
};

// ---------- triangle intersection -> segment endpoints ----------

struct Plane {
    Vec3 n;
    double d; // n·x + d = 0
};

static inline Plane planeFromTri(const Vec3& a, const Vec3& b, const Vec3& c) {
    Vec3 n = cross(b-a, c-a);
    double len = norm(n);
    if (len == 0) return {Vec3{0,0,0}, 0.0};
    n = n / len;
    double d = -dot(n, a);
    return {n,d};
}

static inline double signedDist(const Plane& pl, const Vec3& p) {
    return dot(pl.n, p) + pl.d;
}

static inline bool barycentricInTri(const Vec3& p, const Vec3& a, const Vec3& b, const Vec3& c, double eps) {
    // compute barycentric using dot products
    Vec3 v0 = b - a, v1 = c - a, v2 = p - a;
    double d00 = dot(v0,v0);
    double d01 = dot(v0,v1);
    double d11 = dot(v1,v1);
    double d20 = dot(v2,v0);
    double d21 = dot(v2,v1);
    double denom = d00*d11 - d01*d01;
    if (std::abs(denom) < 1e-30) return false;
    double v = (d11*d20 - d01*d21) / denom;
    double w = (d00*d21 - d01*d20) / denom;
    double u = 1.0 - v - w;
    return u >= -eps && v >= -eps && w >= -eps && (u+v+w) <= 1.0 + 3*eps;
}

static inline void addUniquePoint(vector<Vec3>& pts, const Vec3& p, double tol) {
    for (const auto& q : pts) {
        if (norm2(q-p) <= tol*tol) return;
    }
    pts.push_back(p);
}

static inline bool triTriIntersectionSegment(const Vec3& a0, const Vec3& a1, const Vec3& a2,
                                            const Vec3& b0, const Vec3& b1, const Vec3& b2,
                                            double eps, Vec3& outP, Vec3& outQ) {
    Plane pb = planeFromTri(b0,b1,b2);
    Plane pa = planeFromTri(a0,a1,a2);
    // nearly coplanar? ignore
    double align = std::abs(dot(pa.n, pb.n));
    if (align > 1.0 - 1e-12) {
        // coplanar or near-coplanar; under given assumptions we ignore
        return false;
    }

    vector<Vec3> ip;
    ip.reserve(6);

    auto edgePlane = [&](const Vec3& p0, const Vec3& p1, const Plane& pl,
                         const Vec3& t0, const Vec3& t1, const Vec3& t2) {
        double d0 = signedDist(pl, p0);
        double d1 = signedDist(pl, p1);

        if (std::abs(d0) <= eps && barycentricInTri(p0, t0,t1,t2, eps*10)) addUniquePoint(ip, p0, eps*10);
        if (d0 * d1 < -eps*eps) {
            double t = d0 / (d0 - d1);
            if (t >= -1e-12 && t <= 1.0+1e-12) {
                Vec3 p = p0 + (p1 - p0) * t;
                if (barycentricInTri(p, t0,t1,t2, eps*10)) addUniquePoint(ip, p, eps*10);
            }
        }
    };

    // A edges with B plane; point must be in B tri
    edgePlane(a0,a1,pb, b0,b1,b2);
    edgePlane(a1,a2,pb, b0,b1,b2);
    edgePlane(a2,a0,pb, b0,b1,b2);

    // B edges with A plane; point must be in A tri
    edgePlane(b0,b1,pa, a0,a1,a2);
    edgePlane(b1,b2,pa, a0,a1,a2);
    edgePlane(b2,b0,pa, a0,a1,a2);

    if ((int)ip.size() < 2) return false;

    // choose farthest pair as segment endpoints
    double best = -1.0;
    int bi=0,bj=1;
    for (int i=0;i<(int)ip.size();i++){
        for (int j=i+1;j<(int)ip.size();j++){
            double d = norm2(ip[i]-ip[j]);
            if (d > best) { best = d; bi=i; bj=j; }
        }
    }
    if (best <= eps*eps) return false; // point intersection only
    outP = ip[bi];
    outQ = ip[bj];
    return true;
}

struct CutPoint {
    Vec3 p;
    int edge = 0;     // 0: v0-v1, 1: v1-v2, 2: v2-v0
    double t = 0.0;   // along that edge
    int g = -1;       // global vertex index (after vertex creation)
};

struct TriCut {
    vector<CutPoint> pts;
    vector<std::pair<int,int>> segs;
};

static inline void barycentricWeights(const Vec3& p, const Vec3& a, const Vec3& b, const Vec3& c,
                                      double& w0, double& w1, double& w2) {
    // from barycentric of v0=b-a,v1=c-a
    Vec3 v0 = b - a, v1 = c - a, v2 = p - a;
    double d00 = dot(v0,v0);
    double d01 = dot(v0,v1);
    double d11 = dot(v1,v1);
    double d20 = dot(v2,v0);
    double d21 = dot(v2,v1);
    double denom = d00*d11 - d01*d01;
    if (std::abs(denom) < 1e-30) { w0=1; w1=w2=0; return; }
    double v = (d11*d20 - d01*d21) / denom;
    double w = (d00*d21 - d01*d20) / denom;
    w0 = 1.0 - v - w;
    w1 = v;
    w2 = w;
}

static inline void classifyEdgeT(const Vec3& p, const Vec3& v0, const Vec3& v1, const Vec3& v2,
                                 int& edge, double& t) {
    // Choose edge by barycentric weight near zero (opposite vertex).
    double w0,w1,w2;
    barycentricWeights(p, v0,v1,v2, w0,w1,w2);
    double aw0 = std::abs(w0), aw1 = std::abs(w1), aw2 = std::abs(w2);
    // opposite vertex 2 => edge 0 (v0-v1)
    // opposite vertex 0 => edge 1 (v1-v2)
    // opposite vertex 1 => edge 2 (v2-v0)
    if (aw2 <= aw0 && aw2 <= aw1) edge = 0;
    else if (aw0 <= aw1 && aw0 <= aw2) edge = 1;
    else edge = 2;

    Vec3 a = (edge==0? v0 : (edge==1? v1 : v2));
    Vec3 b = (edge==0? v1 : (edge==1? v2 : v0));
    Vec3 ab = b-a;
    double den = dot(ab,ab);
    if (den <= 1e-30) { t = 0.0; return; }
    t = dot(p-a, ab) / den;
    if (t < 0.0) t = 0.0;
    if (t > 1.0) t = 1.0;
}

static inline int addCutPoint(TriCut& tc, const Vec3& p,
                              const Vec3& v0, const Vec3& v1, const Vec3& v2,
                              double tol) {
    for (int i=0;i<(int)tc.pts.size();i++){
        if (norm2(tc.pts[i].p - p) <= tol*tol) return i;
    }
    CutPoint cp;
    cp.p = p;
    classifyEdgeT(p, v0,v1,v2, cp.edge, cp.t);
    tc.pts.push_back(cp);
    return (int)tc.pts.size()-1;
}

// ---------- BVH-BVH overlap -> leaf pairs ----------

static void collectLeafPairs(const BVH& A, int aNode, const BVH& B, int bNode,
                             vector<std::pair<int,int>>& outPairs) {
    const BVHNode& na = A.nodes[aNode];
    const BVHNode& nb = B.nodes[bNode];
    if (!overlap(na.box, nb.box)) return;
    bool aLeaf = na.count > 0;
    bool bLeaf = nb.count > 0;
    if (aLeaf && bLeaf) {
        outPairs.emplace_back(aNode, bNode);
        return;
    }
    if (aLeaf && !bLeaf) {
        collectLeafPairs(A, aNode, B, nb.left, outPairs);
        collectLeafPairs(A, aNode, B, nb.right, outPairs);
    } else if (!aLeaf && bLeaf) {
        collectLeafPairs(A, na.left, B, bNode, outPairs);
        collectLeafPairs(A, na.right, B, bNode, outPairs);
    } else {
        collectLeafPairs(A, na.left, B, nb.left, outPairs);
        collectLeafPairs(A, na.left, B, nb.right, outPairs);
        collectLeafPairs(A, na.right, B, nb.left, outPairs);
        collectLeafPairs(A, na.right, B, nb.right, outPairs);
    }
}

// ---------- polygon splitting inside (convex) triangle boundary with non-crossing chords ----------

static inline bool containsVertex(const vector<int>& poly, int v) {
    for (int x : poly) if (x == v) return true;
    return false;
}

static bool splitPolygonByChord(vector<int>& poly, int u, int v, vector<int>& out2) {
    // Split poly along chord (u,v). poly is cyclic list (no repeated start/end).
    if (u==v) return false;
    int n = (int)poly.size();
    int iu=-1, iv=-1;
    for (int i=0;i<n;i++){
        if (poly[i]==u) iu=i;
        if (poly[i]==v) iv=i;
    }
    if (iu<0 || iv<0) return false;
    if (iu==iv) return false;
    // adjacent => no split
    if ((iu+1)%n == iv || (iv+1)%n == iu) return false;

    if (iu > iv) std::swap(iu, iv), std::swap(u, v);

    vector<int> p1, p2;
    p1.reserve(n+1); p2.reserve(n+1);
    // iu..iv
    for (int i=iu; i<=iv; i++) p1.push_back(poly[i]);
    // iv..end + 0..iu
    for (int i=iv; i<n; i++) p2.push_back(poly[i]);
    for (int i=0; i<=iu; i++) p2.push_back(poly[i]);

    // both p1 and p2 include u and v
    if ((int)p1.size() < 3 || (int)p2.size() < 3) return false;

    poly.swap(p1);
    out2.swap(p2);
    return true;
}

// ---------- main union pipeline ----------

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    FastScanner fs;

    int VA, FA, VB, FB;
    if (!fs.readInt(VA)) return 0;
    fs.readInt(FA);
    fs.readInt(VB);
    fs.readInt(FB);

    vector<Vec3> inVerts;
    inVerts.reserve((size_t)VA + (size_t)VB);

    AABB sceneBox;
    for (int i=0;i<VA;i++){
        double x,y,z;
        fs.readDouble(x); fs.readDouble(y); fs.readDouble(z);
        Vec3 p{x,y,z};
        inVerts.push_back(p);
        expand(sceneBox, p);
    }
    vector<Tri> facesA;
    facesA.reserve(FA);
    for (int i=0;i<FA;i++){
        int a,b,c;
        fs.readInt(a); fs.readInt(b); fs.readInt(c);
        facesA.push_back({a,b,c});
    }

    int offsetB = VA;
    for (int i=0;i<VB;i++){
        double x,y,z;
        fs.readDouble(x); fs.readDouble(y); fs.readDouble(z);
        Vec3 p{x,y,z};
        inVerts.push_back(p);
        expand(sceneBox, p);
    }
    vector<Tri> facesB;
    facesB.reserve(FB);
    for (int i=0;i<FB;i++){
        int a,b,c;
        fs.readInt(a); fs.readInt(b); fs.readInt(c);
        facesB.push_back({a+offsetB, b+offsetB, c+offsetB});
    }

    Vec3 ext = sceneBox.mx - sceneBox.mn;
    double scale = std::max({ext.x, ext.y, ext.z, 1.0});
    double tol = scale * 1e-9 + 1e-12;
    double invTol = 1.0 / tol;
    double epsRay = tol * 1e-7 + 1e-12; // for MT / ray
    double epsOffset = tol * 50.0;

    // Weld input vertices (A and B together).
    struct Item { QKey q; int idx; };
    vector<Item> items;
    items.reserve(inVerts.size());
    for (int i=0;i<(int)inVerts.size();i++){
        items.push_back({ quantize(inVerts[i], invTol), i });
    }
    std::sort(items.begin(), items.end(), [&](const Item& a, const Item& b){
        if (a.q < b.q) return true;
        if (b.q < a.q) return false;
        // stable tie-breaker
        return a.idx < b.idx;
    });

    vector<Vec3> verts; // welded (and later appended with cut vertices)
    verts.reserve(inVerts.size());

    vector<int> oldToNew(inVerts.size(), -1);
    vector<std::pair<QKey,int>> sortedOrig; // built after welding
    sortedOrig.reserve(inVerts.size());

    for (size_t i=0;i<items.size();) {
        size_t j = i+1;
        while (j < items.size() && !(items[i].q < items[j].q) && !(items[j].q < items[i].q)) j++;
        // within same q, cluster by actual closeness (rare to need >1)
        int rep = (int)verts.size();
        verts.push_back(inVerts[items[i].idx]);
        oldToNew[items[i].idx] = rep;

        for (size_t k=i+1;k<j;k++){
            const Vec3& p = inVerts[items[k].idx];
            if (close3(p, verts[rep], tol)) {
                oldToNew[items[k].idx] = rep;
            } else {
                int ni = (int)verts.size();
                verts.push_back(p);
                oldToNew[items[k].idx] = ni;
            }
        }
        i = j;
    }

    int originalUniqueCount = (int)verts.size();

    // Build sortedOrig for original welded vertices (for later getOrCreate binary search).
    vector<std::pair<QKey,int>> tmpPairs;
    tmpPairs.reserve(originalUniqueCount);
    for (int i=0;i<originalUniqueCount;i++){
        tmpPairs.push_back({ quantize(verts[i], invTol), i });
    }
    std::sort(tmpPairs.begin(), tmpPairs.end(), [](auto const& a, auto const& b){
        if (a.first < b.first) return true;
        if (b.first < a.first) return false;
        return a.second < b.second;
    });
    sortedOrig.swap(tmpPairs);

    // Remap faces
    Mesh A, B;
    A.tris.reserve(facesA.size());
    B.tris.reserve(facesB.size());

    for (const auto& f : facesA) A.tris.push_back({ oldToNew[f.a], oldToNew[f.b], oldToNew[f.c] });
    for (const auto& f : facesB) B.tris.push_back({ oldToNew[f.a], oldToNew[f.b], oldToNew[f.c] });

    auto finalizeMesh = [&](Mesh& M){
        M.triBox.resize(M.tris.size());
        M.triCentroid.resize(M.tris.size());
        M.box = AABB{};
        for (size_t i=0;i<M.tris.size();i++){
            const Tri& t = M.tris[i];
            const Vec3& p0 = verts[t.a];
            const Vec3& p1 = verts[t.b];
            const Vec3& p2 = verts[t.c];
            M.triBox[i] = triAABB(p0,p1,p2);
            M.triCentroid[i] = (p0+p1+p2) / 3.0;
            expand(M.box, M.triBox[i]);
        }
    };
    finalizeMesh(A);
    finalizeMesh(B);

    // Disjoint AABB -> union is just both meshes.
    if (!overlap(A.box, B.box)) {
        FastOutput fo;
        fo.writeInt64((int64_t)verts.size()); fo.putChar(' ');
        fo.writeInt64((int64_t)(A.tris.size() + B.tris.size())); fo.putChar('\n');
        for (const auto& p : verts) {
            fo.writeDouble(p.x); fo.putChar(' ');
            fo.writeDouble(p.y); fo.putChar(' ');
            fo.writeDouble(p.z); fo.putChar('\n');
        }
        for (const auto& t : A.tris) {
            fo.writeInt64(t.a); fo.putChar(' ');
            fo.writeInt64(t.b); fo.putChar(' ');
            fo.writeInt64(t.c); fo.putChar('\n');
        }
        for (const auto& t : B.tris) {
            fo.writeInt64(t.a); fo.putChar(' ');
            fo.writeInt64(t.b); fo.putChar(' ');
            fo.writeInt64(t.c); fo.putChar('\n');
        }
        fo.flush();
        return 0;
    }

    // Build BVHs for intersection detection and point-in-mesh tests.
    BVH bvhA, bvhB;
    int rootA = bvhA.build(verts, A.tris, A.triCentroid);
    int rootB = bvhB.build(verts, B.tris, B.triCentroid);

    // Collect overlapping leaf pairs.
    vector<std::pair<int,int>> leafPairs;
    leafPairs.reserve(1024);
    collectLeafPairs(bvhA, rootA, bvhB, rootB, leafPairs);

    // Cuts storage with striped locks
    std::unordered_map<int, TriCut> cutsA, cutsB;
    cutsA.reserve(1024);
    cutsB.reserve(1024);

    static constexpr int LOCKS = 4096;
    vector<std::mutex> locksA(LOCKS), locksB(LOCKS);

    std::atomic<int> foundIntersections{0};

    auto workerIntersect = [&](int tid, int T){
        for (size_t k = (size_t)tid; k < leafPairs.size(); k += (size_t)T) {
            int na = leafPairs[k].first;
            int nb = leafPairs[k].second;
            const BVHNode& aN = bvhA.nodes[na];
            const BVHNode& bN = bvhB.nodes[nb];

            for (int ia=0; ia<aN.count; ia++){
                int ta = bvhA.indices[aN.start + ia];
                const Tri& triA = A.tris[ta];
                Vec3 a0 = verts[triA.a], a1 = verts[triA.b], a2 = verts[triA.c];
                AABB boxa = A.triBox[ta];

                for (int ib=0; ib<bN.count; ib++){
                    int tb = bvhB.indices[bN.start + ib];
                    if (!overlap(boxa, B.triBox[tb])) continue;

                    const Tri& triB = B.tris[tb];
                    Vec3 b0 = verts[triB.a], b1 = verts[triB.b], b2 = verts[triB.c];

                    Vec3 p,q;
                    if (!triTriIntersectionSegment(a0,a1,a2, b0,b1,b2, tol*10.0, p,q)) continue;
                    foundIntersections.fetch_add(1, std::memory_order_relaxed);

                    // Add to cutsA[ta]
                    {
                        std::mutex& m = locksA[ta & (LOCKS-1)];
                        std::lock_guard<std::mutex> lg(m);
                        TriCut& c = cutsA[ta];
                        int ip = addCutPoint(c, p, a0,a1,a2, tol*10.0);
                        int iq = addCutPoint(c, q, a0,a1,a2, tol*10.0);
                        if (ip != iq) c.segs.emplace_back(ip,iq);
                    }
                    // Add to cutsB[tb]
                    {
                        std::mutex& m = locksB[tb & (LOCKS-1)];
                        std::lock_guard<std::mutex> lg(m);
                        TriCut& c = cutsB[tb];
                        int ip = addCutPoint(c, p, b0,b1,b2, tol*10.0);
                        int iq = addCutPoint(c, q, b0,b1,b2, tol*10.0);
                        if (ip != iq) c.segs.emplace_back(ip,iq);
                    }
                }
            }
        }
    };

    int numThreads = (int)std::max(1u, std::thread::hardware_concurrency());
    numThreads = std::min(numThreads, 12); // environment indicates 12 cores

    {
        vector<std::thread> threads;
        threads.reserve(numThreads);
        for (int t=0;t<numThreads;t++) threads.emplace_back(workerIntersect, t, numThreads);
        for (auto& th : threads) th.join();
    }

    // If no intersections, do quick containment / disjoint check and output without per-triangle classification.
    if (foundIntersections.load(std::memory_order_relaxed) == 0) {
        // Pick representative points
        Vec3 pA = verts[A.tris.empty()?0:A.tris[0].a];
        Vec3 pB = verts[B.tris.empty()?0:B.tris[0].a];
        bool A_in_B = pointInMesh(pA, B, verts, bvhB, rootB, epsRay);
        bool B_in_A = pointInMesh(pB, A, verts, bvhA, rootA, epsRay);

        FastOutput fo;
        if (A_in_B && !B_in_A) {
            fo.writeInt64((int64_t)verts.size()); fo.putChar(' ');
            fo.writeInt64((int64_t)B.tris.size()); fo.putChar('\n');
            for (const auto& p : verts) {
                fo.writeDouble(p.x); fo.putChar(' ');
                fo.writeDouble(p.y); fo.putChar(' ');
                fo.writeDouble(p.z); fo.putChar('\n');
            }
            for (const auto& t : B.tris) {
                fo.writeInt64(t.a); fo.putChar(' ');
                fo.writeInt64(t.b); fo.putChar(' ');
                fo.writeInt64(t.c); fo.putChar('\n');
            }
        } else if (B_in_A && !A_in_B) {
            fo.writeInt64((int64_t)verts.size()); fo.putChar(' ');
            fo.writeInt64((int64_t)A.tris.size()); fo.putChar('\n');
            for (const auto& p : verts) {
                fo.writeDouble(p.x); fo.putChar(' ');
                fo.writeDouble(p.y); fo.putChar(' ');
                fo.writeDouble(p.z); fo.putChar('\n');
            }
            for (const auto& t : A.tris) {
                fo.writeInt64(t.a); fo.putChar(' ');
                fo.writeInt64(t.b); fo.putChar(' ');
                fo.writeInt64(t.c); fo.putChar('\n');
            }
        } else {
            // disjoint solids (bbox overlapped but actually separate) => both
            fo.writeInt64((int64_t)verts.size()); fo.putChar(' ');
            fo.writeInt64((int64_t)(A.tris.size() + B.tris.size())); fo.putChar('\n');
            for (const auto& p : verts) {
                fo.writeDouble(p.x); fo.putChar(' ');
                fo.writeDouble(p.y); fo.putChar(' ');
                fo.writeDouble(p.z); fo.putChar('\n');
            }
            for (const auto& t : A.tris) {
                fo.writeInt64(t.a); fo.putChar(' ');
                fo.writeInt64(t.b); fo.putChar(' ');
                fo.writeInt64(t.c); fo.putChar('\n');
            }
            for (const auto& t : B.tris) {
                fo.writeInt64(t.a); fo.putChar(' ');
                fo.writeInt64(t.b); fo.putChar(' ');
                fo.writeInt64(t.c); fo.putChar('\n');
            }
        }
        fo.flush();
        return 0;
    }

    // Create global vertices for all cut points (shared across both meshes) with global dedup.
    VertexManager vm{ verts, sortedOrig, originalUniqueCount, tol, invTol, {} };
    vm.newMap.reserve((cutsA.size()+cutsB.size())*4);

    auto createCutVertices = [&](std::unordered_map<int, TriCut>& cuts){
        for (auto& kv : cuts) {
            TriCut& tc = kv.second;
            for (auto& cp : tc.pts) {
                cp.g = vm.getExistingOrCreate(cp.p);
            }
        }
    };
    createCutVertices(cutsA);
    createCutVertices(cutsB);

    // Output faces (global vertex indices)
    struct OutTri { int a,b,c; };
    vector<OutTri> outFaces;
    outFaces.reserve(A.tris.size() + B.tris.size());

    auto processMesh = [&](const Mesh& M, const Mesh& Other,
                           const BVH& otherBVH, int otherRoot,
                           const std::unordered_map<int, TriCut>& cuts,
                           bool isA,
                           vector<OutTri>& outLocal) {
        outLocal.clear();
        // heuristic reserve (over-reserve a bit)
        outLocal.reserve(M.tris.size() / numThreads + 1024);

        for (int i=0;i<(int)M.tris.size();i++){
            const Tri& t = M.tris[i];
            const Vec3& p0 = verts[t.a];
            const Vec3& p1 = verts[t.b];
            const Vec3& p2 = verts[t.c];
            Vec3 n0 = normalize(cross(p1-p0, p2-p0));

            auto it = cuts.find(i);
            if (it == cuts.end()) {
                // quick bbox disjoint with other bbox => certainly outside
                if (!overlap(M.triBox[i], Other.box)) {
                    outLocal.push_back({t.a,t.b,t.c});
                    continue;
                }
                Vec3 c = (p0+p1+p2) / 3.0;
                Vec3 q = c + n0 * epsOffset;
                bool inside = pointInMesh(q, Other, verts, otherBVH, otherRoot, epsRay);
                if (!inside) outLocal.push_back({t.a,t.b,t.c});
                continue;
            }

            // Split triangle by intersection chords
            const TriCut& tc = it->second;

            // collect boundary cycle with inserted points on each edge
            vector<int> boundary;
            boundary.reserve(3 + (int)tc.pts.size());

            auto gatherEdge = [&](int edge, int vStart, int vEnd, bool skipLast){
                vector<std::pair<double,int>> pts;
                pts.reserve(8);
                pts.emplace_back(0.0, vStart);
                pts.emplace_back(1.0, vEnd);
                for (const auto& cp : tc.pts) {
                    if (cp.edge == edge) pts.emplace_back(cp.t, cp.g);
                }
                std::sort(pts.begin(), pts.end(), [](auto const& a, auto const& b){
                    if (a.first != b.first) return a.first < b.first;
                    return a.second < b.second;
                });
                // dedupe by vertex id (keep first)
                vector<std::pair<double,int>> uniq;
                uniq.reserve(pts.size());
                for (auto &pr : pts) {
                    if (uniq.empty() || uniq.back().second != pr.second) uniq.push_back(pr);
                }
                // append, skipping last endpoint to avoid duplication at corners
                int lim = (int)uniq.size() - (skipLast ? 1 : 0);
                for (int i=0;i<lim;i++) boundary.push_back(uniq[i].second);
            };

            // edges: 0: a-b, 1: b-c, 2: c-a
            gatherEdge(0, t.a, t.b, true);
            gatherEdge(1, t.b, t.c, true);
            gatherEdge(2, t.c, t.a, true);

            // If boundary is degenerate, fall back
            if ((int)boundary.size() < 3) {
                Vec3 c = (p0+p1+p2) / 3.0;
                Vec3 q = c + n0 * epsOffset;
                bool inside = pointInMesh(q, Other, verts, otherBVH, otherRoot, epsRay);
                if (!inside) outLocal.push_back({t.a,t.b,t.c});
                continue;
            }

            // prepare chords
            vector<std::pair<int,int>> chords;
            chords.reserve(tc.segs.size());
            for (auto [uId, vId] : tc.segs) {
                int u = tc.pts[uId].g;
                int v = tc.pts[vId].g;
                if (u==v) continue;
                chords.emplace_back(u,v);
            }
            // dedupe chords (small)
            std::sort(chords.begin(), chords.end());
            chords.erase(std::unique(chords.begin(), chords.end()), chords.end());

            // Split polygon by chords
            vector<vector<int>> polys;
            polys.push_back(boundary);

            for (auto [u,v] : chords) {
                for (size_t pi=0; pi<polys.size(); pi++){
                    auto& poly = polys[pi];
                    if (!containsVertex(poly, u) || !containsVertex(poly, v)) continue;
                    vector<int> otherPoly;
                    if (splitPolygonByChord(poly, u, v, otherPoly)) {
                        polys.push_back(std::move(otherPoly));
                    }
                    break; // chord placed
                }
            }

            // Triangulate each convex subpolygon and classify
            for (auto& poly : polys) {
                if ((int)poly.size() < 3) continue;
                int v0 = poly[0];
                for (int k=1; k+1 < (int)poly.size(); k++){
                    int v1i = poly[k];
                    int v2i = poly[k+1];
                    const Vec3& q0 = verts[v0];
                    const Vec3& q1 = verts[v1i];
                    const Vec3& q2 = verts[v2i];
                    Vec3 c = (q0+q1+q2) / 3.0;
                    Vec3 q = c + n0 * epsOffset; // use original triangle normal
                    bool inside = pointInMesh(q, Other, verts, otherBVH, otherRoot, epsRay);
                    if (!inside) outLocal.push_back({v0, v1i, v2i});
                }
            }
        }
    };

    // Parallel classification/splitting
    vector<vector<OutTri>> locals(numThreads);

    auto runParallel = [&](const Mesh& M, const Mesh& Other,
                           const BVH& otherBVH, int otherRoot,
                           const std::unordered_map<int, TriCut>& cuts,
                           vector<OutTri>& outAll) {
        std::atomic<int> next{0};
        int N = (int)M.tris.size();

        auto worker = [&](int tid){
            vector<OutTri>& outLocal = locals[tid];
            outLocal.clear();
            outLocal.reserve((size_t)N/numThreads + 1024);

            while (true) {
                int i = next.fetch_add(1);
                if (i >= N) break;

                const Tri& t = M.tris[i];
                const Vec3& p0 = verts[t.a];
                const Vec3& p1 = verts[t.b];
                const Vec3& p2 = verts[t.c];
                Vec3 n0 = normalize(cross(p1-p0, p2-p0));

                auto it = cuts.find(i);
                if (it == cuts.end()) {
                    if (!overlap(M.triBox[i], Other.box)) {
                        outLocal.push_back({t.a,t.b,t.c});
                        continue;
                    }
                    Vec3 c = (p0+p1+p2)/3.0;
                    Vec3 q = c + n0 * epsOffset;
                    bool inside = pointInMesh(q, Other, verts, otherBVH, otherRoot, epsRay);
                    if (!inside) outLocal.push_back({t.a,t.b,t.c});
                    continue;
                }

                const TriCut& tc = it->second;

                vector<int> boundary;
                boundary.reserve(3 + (int)tc.pts.size());

                auto gatherEdge = [&](int edge, int vStart, int vEnd){
                    vector<std::pair<double,int>> pts;
                    pts.reserve(8);
                    pts.emplace_back(0.0, vStart);
                    pts.emplace_back(1.0, vEnd);
                    for (const auto& cp : tc.pts) if (cp.edge == edge) pts.emplace_back(cp.t, cp.g);
                    std::sort(pts.begin(), pts.end(), [](auto const& a, auto const& b){
                        if (a.first != b.first) return a.first < b.first;
                        return a.second < b.second;
                    });
                    vector<std::pair<double,int>> uniq;
                    uniq.reserve(pts.size());
                    for (auto &pr : pts) if (uniq.empty() || uniq.back().second != pr.second) uniq.push_back(pr);
                    // skip last vertex to avoid duplicating corners across edges
                    int lim = (int)uniq.size() - 1;
                    for (int j=0;j<lim;j++) boundary.push_back(uniq[j].second);
                };

                gatherEdge(0, t.a, t.b);
                gatherEdge(1, t.b, t.c);
                gatherEdge(2, t.c, t.a);

                if ((int)boundary.size() < 3) {
                    Vec3 c = (p0+p1+p2) / 3.0;
                    Vec3 q = c + n0 * epsOffset;
                    bool inside = pointInMesh(q, Other, verts, otherBVH, otherRoot, epsRay);
                    if (!inside) outLocal.push_back({t.a,t.b,t.c});
                    continue;
                }

                vector<std::pair<int,int>> chords;
                chords.reserve(tc.segs.size());
                for (auto [uId, vId] : tc.segs) {
                    int u = tc.pts[uId].g;
                    int v = tc.pts[vId].g;
                    if (u==v) continue;
                    if (u > v) std::swap(u,v);
                    chords.emplace_back(u,v);
                }
                std::sort(chords.begin(), chords.end());
                chords.erase(std::unique(chords.begin(), chords.end()), chords.end());

                vector<vector<int>> polys;
                polys.push_back(boundary);

                for (auto [u,v] : chords) {
                    for (size_t pi=0; pi<polys.size(); pi++){
                        auto& poly = polys[pi];
                        if (!containsVertex(poly, u) || !containsVertex(poly, v)) continue;
                        vector<int> otherPoly;
                        if (splitPolygonByChord(poly, u, v, otherPoly)) polys.push_back(std::move(otherPoly));
                        break;
                    }
                }

                for (auto& poly : polys) {
                    if ((int)poly.size() < 3) continue;
                    int v0 = poly[0];
                    for (int k=1; k+1 < (int)poly.size(); k++){
                        int v1i = poly[k];
                        int v2i = poly[k+1];
                        Vec3 c = (verts[v0] + verts[v1i] + verts[v2i]) / 3.0;
                        Vec3 q = c + n0 * epsOffset;
                        bool inside = pointInMesh(q, Other, verts, otherBVH, otherRoot, epsRay);
                        if (!inside) outLocal.push_back({v0,v1i,v2i});
                    }
                }
            }
        };

        vector<std::thread> threads;
        threads.reserve(numThreads);
        for (int t=0;t<numThreads;t++) threads.emplace_back(worker, t);
        for (auto& th : threads) th.join();

        // merge
        size_t total = 0;
        for (int t=0;t<numThreads;t++) total += locals[t].size();
        outAll.reserve(outAll.size() + total);
        for (int t=0;t<numThreads;t++) {
            outAll.insert(outAll.end(), locals[t].begin(), locals[t].end());
        }
    };

    // A parts outside B
    runParallel(A, B, bvhB, rootB, cutsA, outFaces);
    // B parts outside A
    runParallel(B, A, bvhA, rootA, cutsB, outFaces);

    // Emit output.
    FastOutput fo;
    fo.writeInt64((int64_t)verts.size()); fo.putChar(' ');
    fo.writeInt64((int64_t)outFaces.size()); fo.putChar('\n');

    for (const auto& p : verts) {
        fo.writeDouble(p.x); fo.putChar(' ');
        fo.writeDouble(p.y); fo.putChar(' ');
        fo.writeDouble(p.z); fo.putChar('\n');
    }
    for (const auto& t : outFaces) {
        fo.writeInt64(t.a); fo.putChar(' ');
        fo.writeInt64(t.b); fo.putChar(' ');
        fo.writeInt64(t.c); fo.putChar('\n');
    }
    fo.flush();
    return 0;
}
