```javascript
'use strict';

const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input ? JSON.parse(input) : {};

const depot = data.depot || [0, 0];
const depotX = +depot[0] || 0;
const depotY = +depot[1] || 0;

let K = (data.truckCount | 0);
if (K <= 0) K = 1;

const packages = Array.isArray(data.packages) ? data.packages : [];
const n = packages.length;

function distXY(x1, y1, x2, y2) {
  const dx = x1 - x2;
  const dy = y1 - y2;
  return Math.sqrt(dx * dx + dy * dy);
}

if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const ids = new Array(n);
const xs = new Float64Array(n);
const ys = new Float64Array(n);

let minX = Infinity, minY = Infinity, maxX = -Infinity, maxY = -Infinity;
for (let i = 0; i < n; i++) {
  const p = packages[i];
  ids[i] = p.id;
  const x = +p.x || 0;
  const y = +p.y || 0;
  xs[i] = x;
  ys[i] = y;
  if (x < minX) minX = x;
  if (x > maxX) maxX = x;
  if (y < minY) minY = y;
  if (y > maxY) maxY = y;
}

function makeIndexArray() {
  const idx = new Array(n);
  for (let i = 0; i < n; i++) idx[i] = i;
  return idx;
}

function buildOrderAngle() {
  const idx = makeIndexArray();
  const angles = new Float64Array(n);
  const r2s = new Float64Array(n);
  for (let i = 0; i < n; i++) {
    const dx = xs[i] - depotX;
    const dy = ys[i] - depotY;
    angles[i] = Math.atan2(dy, dx);
    r2s[i] = dx * dx + dy * dy;
  }
  idx.sort((a, b) => {
    const da = angles[a] - angles[b];
    if (da !== 0) return da < 0 ? -1 : 1;
    const dr = r2s[a] - r2s[b];
    if (dr !== 0) return dr < 0 ? -1 : 1;
    return a - b;
  });
  return idx;
}

function part1By1_16(n) {
  n &= 0xffff;
  n = (n | (n << 8)) & 0x00ff00ff;
  n = (n | (n << 4)) & 0x0f0f0f0f;
  n = (n | (n << 2)) & 0x33333333;
  n = (n | (n << 1)) & 0x55555555;
  return n >>> 0;
}

function buildOrderMorton() {
  const idx = makeIndexArray();

  const dx = (maxX - minX) || 1;
  const dy = (maxY - minY) || 1;

  const key = new Uint32Array(n);
  for (let i = 0; i < n; i++) {
    let xi = ((xs[i] - minX) / dx) * 65535;
    let yi = ((ys[i] - minY) / dy) * 65535;
    xi = xi < 0 ? 0 : xi > 65535 ? 65535 : xi | 0;
    yi = yi < 0 ? 0 : yi > 65535 ? 65535 : yi | 0;
    const mx = part1By1_16(xi);
    const my = part1By1_16(yi);
    key[i] = (mx | (my << 1)) >>> 0;
  }

  idx.sort((a, b) => {
    const ka = key[a], kb = key[b];
    if (ka !== kb) return ka < kb ? -1 : 1;
    const xa = xs[a], xb = xs[b];
    if (xa !== xb) return xa < xb ? -1 : 1;
    const ya = ys[a], yb = ys[b];
    if (ya !== yb) return ya < yb ? -1 : 1;
    return a - b;
  });

  return idx;
}

function rotateToBestDepotBreak(cycleOrder) {
  if (n <= 1) return cycleOrder.slice();

  let bestPos = 0;
  let bestDelta = Infinity;

  for (let i = 0; i < n; i++) {
    const a = cycleOrder[i];
    const b = cycleOrder[(i + 1) % n];

    const delta =
      distXY(xs[a], ys[a], depotX, depotY) +
      distXY(xs[b], ys[b], depotX, depotY) -
      distXY(xs[a], ys[a], xs[b], ys[b]);

    if (delta < bestDelta) {
      bestDelta = delta;
      bestPos = i;
    }
  }

  const start = (bestPos + 1) % n;
  const path = new Array(n);
  for (let t = 0; t < n; t++) path[t] = cycleOrder[(start + t) % n];
  return path;
}

function reverseSegment(arr, i, k) {
  while (i < k) {
    const tmp = arr[i];
    arr[i] = arr[k];
    arr[k] = tmp;
    i++;
    k--;
  }
}

function apply2OptPath(path, W, passes) {
  if (n < 4 || passes <= 0 || W <= 1) return;

  for (let pass = 0; pass < passes; pass++) {
    let improved = false;

    outer:
    for (let i = 0; i <= n - 3; i++) {
      const bi = path[i];
      const ax = (i === 0) ? depotX : xs[path[i - 1]];
      const ay = (i === 0) ? depotY : ys[path[i - 1]];
      const bx = xs[bi], by = ys[bi];

      const kMax = Math.min(n - 1, i + W);
      for (let k = i + 2; k <= kMax; k++) {
        const ck = path[k];
        const cx = xs[ck], cy = ys[ck];
        const dx = (k === n - 1) ? depotX : xs[path[k + 1]];
        const dy = (k === n - 1) ? depotY : ys[path[k + 1]];

        const oldCost = distXY(ax, ay, bx, by) + distXY(cx, cy, dx, dy);
        const newCost = distXY(ax, ay, cx, cy) + distXY(bx, by, dx, dy);

        if (newCost + 1e-12 < oldCost) {
          reverseSegment(path, i, k);
          improved = true;
          // Restart near the modified region for determinism + speed
          i = Math.max(-1, i - 2);
          continue outer;
        }
      }
    }

    if (!improved) break;
  }
}

function computeBasePathCost(path) {
  if (n === 0) return 0;
  let total = distXY(depotX, depotY, xs[path[0]], ys[path[0]]);
  for (let i = 0; i < n - 1; i++) {
    const a = path[i], b = path[i + 1];
    total += distXY(xs[a], ys[a], xs[b], ys[b]);
  }
  total += distXY(xs[path[n - 1]], ys[path[n - 1]], depotX, depotY);
  return total;
}

function buildRoutesAndCost(path, maxRoutes) {
  if (n === 0) return { routes: [], cost: 0 };

  const baseCost = computeBasePathCost(path);
  if (maxRoutes <= 1 || n === 1) {
    return { routes: [path.map(i => ids[i])], cost: baseCost };
  }

  const cutCandidates = [];
  for (let i = 0; i < n - 1; i++) {
    const a = path[i], b = path[i + 1];
    const delta =
      distXY(xs[a], ys[a], depotX, depotY) +
      distXY(depotX, depotY, xs[b], ys[b]) -
      distXY(xs[a], ys[a], xs[b], ys[b]);
    if (delta < 0) cutCandidates.push([delta, i]);
  }

  cutCandidates.sort((p, q) => {
    if (p[0] !== q[0]) return p[0] < q[0] ? -1 : 1;
    return p[1] - q[1];
  });

  const wantCuts = Math.min(maxRoutes - 1, cutCandidates.length);
  const cutAt = new Uint8Array(n - 1);
  let extra = 0;
  for (let t = 0; t < wantCuts; t++) {
    const [delta, pos] = cutCandidates[t];
    cutAt[pos] = 1;
    extra += delta;
  }

  const routes = [];
  let current = [];
  for (let i = 0; i < n; i++) {
    current.push(ids[path[i]]);
    if (i === n - 1 || cutAt[i]) {
      routes.push(current);
      current = [];
    }
  }

  return { routes, cost: baseCost + extra };
}

// Candidate selection based on size
const candidates = [];
if (n > 150000) {
  candidates.push('morton');
} else {
  candidates.push('angle', 'morton');
  if (n <= 30000) candidates.push('angle'); // cheap extra deterministic duplicate not needed; keep list simple
}

// Deduplicate
const candSet = new Set(candidates);
const candList = Array.from(candSet);

let best = null;

for (const c of candList) {
  let order;
  if (c === 'morton') order = buildOrderMorton();
  else order = buildOrderAngle();

  const path = rotateToBestDepotBreak(order);

  let passes = 0, W = 0;
  if (n <= 80000) { passes = 2; W = 30; }
  else if (n <= 150000) { passes = 1; W = 20; }
  else { passes = 0; W = 0; }

  apply2OptPath(path, W, passes);

  const sol = buildRoutesAndCost(path, Math.min(K, n));

  if (
    best === null ||
    sol.cost < best.cost - 1e-9 ||
    (Math.abs(sol.cost - best.cost) <= 1e-9 && sol.routes.length < best.routes.length)
  ) {
    best = sol;
  }
}

process.stdout.write(JSON.stringify({ routes: best ? best.routes : [] }));
```
