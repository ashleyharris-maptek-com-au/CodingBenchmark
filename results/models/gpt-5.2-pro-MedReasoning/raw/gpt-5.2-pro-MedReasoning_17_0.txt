using System;
using System.Collections.Generic;
using System.Text;

public class Program
{
    // --------- Fast Scanner ----------
    sealed class FastScanner
    {
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;
        private readonly System.IO.Stream _in = Console.OpenStandardInput();

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _in.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do c = Read(); while (c <= 32);

            int sgn = 1;
            if (c == (byte)'-')
            {
                sgn = -1;
                c = Read();
            }

            int x = 0;
            while (c > 32)
            {
                x = x * 10 + (c - (byte)'0');
                c = Read();
            }
            return x * sgn;
        }
    }

    // --------- Data Types ----------
    readonly struct Orient
    {
        public readonly int W, H, D;
        public Orient(int w, int h, int d) { W = w; H = h; D = d; }
    }

    sealed class Box
    {
        public int Index;
        public int W, H, D;
        public long Vol;
        public Orient[] Orients = Array.Empty<Orient>();
    }

    struct Space
    {
        public int X, Y, Z, W, H, D;
        public long Vol;
        public Space(int x, int y, int z, int w, int h, int d)
        {
            X = x; Y = y; Z = z; W = w; H = h; D = d;
            Vol = (long)w * h * d;
        }
    }

    readonly struct Placement
    {
        public readonly int BoxIndex;
        public readonly int X, Y, Z;
        public Placement(int boxIndex, int x, int y, int z)
        {
            BoxIndex = boxIndex; X = x; Y = y; Z = z;
        }
    }

    enum Strategy
    {
        SmallFirst,
        BigFirst,
        MinDimFirst
    }

    // --------- Min-Heap for spaces (by low coordinates, then smaller volume) ----------
    sealed class SpaceHeap
    {
        private readonly List<int> _heap = new List<int>(1024);
        private readonly List<Space> _spaces;
        private readonly List<byte> _alive; // 1 alive, 0 dead

        public SpaceHeap(List<Space> spaces, List<byte> alive)
        {
            _spaces = spaces;
            _alive = alive;
        }

        private int Cmp(int aId, int bId)
        {
            // dead ones will be filtered on pop; comparator assumes ids are valid.
            Space a = _spaces[aId];
            Space b = _spaces[bId];

            int c = a.Z.CompareTo(b.Z); if (c != 0) return c;
            c = a.Y.CompareTo(b.Y); if (c != 0) return c;
            c = a.X.CompareTo(b.X); if (c != 0) return c;
            c = a.Vol.CompareTo(b.Vol); if (c != 0) return c;
            return aId.CompareTo(bId);
        }

        public int Count => _heap.Count;

        public void Push(int id)
        {
            _heap.Add(id);
            SiftUp(_heap.Count - 1);
        }

        public int PopAlive()
        {
            while (_heap.Count > 0)
            {
                int id = PopRaw();
                if (_alive[id] != 0) return id;
            }
            return -1;
        }

        private int PopRaw()
        {
            int res = _heap[0];
            int last = _heap[^1];
            _heap.RemoveAt(_heap.Count - 1);
            if (_heap.Count > 0)
            {
                _heap[0] = last;
                SiftDown(0);
            }
            return res;
        }

        private void SiftUp(int i)
        {
            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (Cmp(_heap[i], _heap[p]) >= 0) break;
                (_heap[i], _heap[p]) = (_heap[p], _heap[i]);
                i = p;
            }
        }

        private void SiftDown(int i)
        {
            int n = _heap.Count;
            while (true)
            {
                int l = i * 2 + 1;
                if (l >= n) break;
                int r = l + 1;
                int best = (r < n && Cmp(_heap[r], _heap[l]) < 0) ? r : l;
                if (Cmp(_heap[best], _heap[i]) >= 0) break;
                (_heap[i], _heap[best]) = (_heap[best], _heap[i]);
                i = best;
            }
        }
    }

    static Orient[] BuildUniqueOrients(int w, int h, int d)
    {
        // 6 permutations with manual de-dup (boxes can have equal sides)
        Orient[] tmp = new Orient[]
        {
            new Orient(w,h,d),
            new Orient(w,d,h),
            new Orient(h,w,d),
            new Orient(h,d,w),
            new Orient(d,w,h),
            new Orient(d,h,w),
        };

        Span<Orient> uniq = stackalloc Orient[6];
        int cnt = 0;
        for (int i = 0; i < 6; i++)
        {
            bool ok = true;
            for (int j = 0; j < cnt; j++)
            {
                if (tmp[i].W == uniq[j].W && tmp[i].H == uniq[j].H && tmp[i].D == uniq[j].D)
                {
                    ok = false;
                    break;
                }
            }
            if (ok) uniq[cnt++] = tmp[i];
        }

        Orient[] res = new Orient[cnt];
        for (int i = 0; i < cnt; i++) res[i] = uniq[i];
        return res;
    }

    static void Sort3(ref int a, ref int b, ref int c)
    {
        if (a > b) (a, b) = (b, a);
        if (b > c) (b, c) = (c, b);
        if (a > b) (a, b) = (b, a);
    }

    static bool FitsContainerAnyOrient(Box box, int W, int H, int D)
    {
        // Fast sufficient/necessary check using sorted dimensions:
        int a = box.W, b = box.H, c = box.D;
        int x = W, y = H, z = D;
        Sort3(ref a, ref b, ref c);
        Sort3(ref x, ref y, ref z);
        return a <= x && b <= y && c <= z;
    }

    static List<Placement> Pack(List<Box> inputBoxes, int contW, int contH, int contD, Strategy strategy)
    {
        // Prepare boxes (copy references) + sort by strategy
        var boxes = new List<Box>(inputBoxes);

        boxes.Sort((p, q) =>
        {
            switch (strategy)
            {
                case Strategy.BigFirst:
                    {
                        int c = q.Vol.CompareTo(p.Vol);
                        if (c != 0) return c;
                        // tiebreak: larger max side first
                        int pm = Math.Max(p.W, Math.Max(p.H, p.D));
                        int qm = Math.Max(q.W, Math.Max(q.H, q.D));
                        c = qm.CompareTo(pm);
                        if (c != 0) return c;
                        return p.Index.CompareTo(q.Index);
                    }
                case Strategy.MinDimFirst:
                    {
                        int pmin = Math.Min(p.W, Math.Min(p.H, p.D));
                        int qmin = Math.Min(q.W, Math.Min(q.H, q.D));
                        int c = pmin.CompareTo(qmin);
                        if (c != 0) return c;
                        c = p.Vol.CompareTo(q.Vol);
                        if (c != 0) return c;
                        return p.Index.CompareTo(q.Index);
                    }
                default: // SmallFirst
                    {
                        int c = p.Vol.CompareTo(q.Vol);
                        if (c != 0) return c;
                        // tiebreak: smaller max side first
                        int pm = Math.Max(p.W, Math.Max(p.H, p.D));
                        int qm = Math.Max(q.W, Math.Max(q.H, q.D));
                        c = pm.CompareTo(qm);
                        if (c != 0) return c;
                        return p.Index.CompareTo(q.Index);
                    }
            }
        });

        // Spaces storage with lazy-deletion
        var spaces = new List<Space>(1 + 2 * boxes.Count);
        var alive = new List<byte>(1 + 2 * boxes.Count);
        spaces.Add(new Space(0, 0, 0, contW, contH, contD));
        alive.Add(1);

        int aliveCount = 1;
        var heap = new SpaceHeap(spaces, alive);
        heap.Push(0);

        var placements = new List<Placement>(boxes.Count);

        // Rounds help boxes that only become placeable after creating shifted-origin spaces.
        int n = boxes.Count;
        int maxRounds = (n <= 200 ? 12 : (n <= 2000 ? 6 : 2));

        // K: how many candidate spaces to consider per attempt (performance knob).
        int K = (n <= 200 ? 4096 : (n <= 2000 ? 512 : 96));

        // Remaining list for multi-round attempts
        var remaining = new List<Box>(boxes);

        // Temporary arrays reused
        var candSpaceIds = new List<int>(K);

        for (int round = 0; round < maxRounds && remaining.Count > 0; round++)
        {
            bool progress = false;
            var nextRemaining = new List<Box>(remaining.Count);

            bool allowFallbackFullScan = (n <= 2000) || (round == maxRounds - 1);

            foreach (var box in remaining)
            {
                if (aliveCount == 0)
                {
                    nextRemaining.Add(box);
                    continue;
                }

                if (TryPlaceBox(box, spaces, alive, ref aliveCount, heap, placements, K, allowFallbackFullScan, candSpaceIds))
                {
                    progress = true;
                }
                else
                {
                    nextRemaining.Add(box);
                }
            }

            remaining = nextRemaining;
            if (!progress) break;
        }

        return placements;
    }

    static bool TryPlaceBox(
        Box box,
        List<Space> spaces,
        List<byte> alive,
        ref int aliveCount,
        SpaceHeap heap,
        List<Placement> placements,
        int K,
        bool allowFallbackFullScan,
        List<int> candSpaceIds)
    {
        candSpaceIds.Clear();

        // Gather up to K alive candidate spaces from heap (then push back unchosen)
        int pulled = 0;
        while (pulled < K)
        {
            int sid = heap.PopAlive();
            if (sid < 0) break;
            candSpaceIds.Add(sid);
            pulled++;
        }

        int bestSid = -1;
        int bestW = 0, bestH = 0, bestD = 0;
        long bestScore = long.MaxValue;

        // Evaluate candidates
        for (int i = 0; i < candSpaceIds.Count; i++)
        {
            int sid = candSpaceIds[i];
            if (alive[sid] == 0) continue;
            Space s = spaces[sid];

            // Quick reject: if even smallest sorted dims don't fit, skip.
            // (But testing all orientations is cheap; we still do it.)
            for (int oi = 0; oi < box.Orients.Length; oi++)
            {
                Orient o = box.Orients[oi];
                if (o.W > s.W || o.H > s.H || o.D > s.D) continue;

                int dx = s.W - o.W;
                int dy = s.H - o.H;
                int dz = s.D - o.D;
                long leftover = s.Vol - box.Vol;
                int maxRem = dx; if (dy > maxRem) maxRem = dy; if (dz > maxRem) maxRem = dz;
                int sumRem = dx + dy + dz;

                // Primary: minimize leftover (tight fit)
                // Secondary: minimize max remainder, then sum remainder
                // Tertiary: earlier coordinates
                long score =
                    (leftover << 20) +
                    ((long)maxRem << 10) +
                    (long)sumRem;

                // Coordinate tiebreak (small weight)
                score = (score << 20)
                        + ((long)s.Z << 12)
                        + ((long)s.Y << 6)
                        + (long)s.X;

                if (score < bestScore)
                {
                    bestScore = score;
                    bestSid = sid;
                    bestW = o.W; bestH = o.H; bestD = o.D;
                }
            }
        }

        if (bestSid < 0 && allowFallbackFullScan)
        {
            // Full scan fallback can find a space not in the top-K by (z,y,x).
            for (int sid = 0; sid < spaces.Count; sid++)
            {
                if (alive[sid] == 0) continue;
                Space s = spaces[sid];

                for (int oi = 0; oi < box.Orients.Length; oi++)
                {
                    Orient o = box.Orients[oi];
                    if (o.W > s.W || o.H > s.H || o.D > s.D) continue;

                    int dx = s.W - o.W;
                    int dy = s.H - o.H;
                    int dz = s.D - o.D;
                    long leftover = s.Vol - box.Vol;
                    int maxRem = dx; if (dy > maxRem) maxRem = dy; if (dz > maxRem) maxRem = dz;
                    int sumRem = dx + dy + dz;

                    long score =
                        (leftover << 20) +
                        ((long)maxRem << 10) +
                        (long)sumRem;

                    score = (score << 20)
                            + ((long)s.Z << 12)
                            + ((long)s.Y << 6)
                            + (long)s.X;

                    if (score < bestScore)
                    {
                        bestScore = score;
                        bestSid = sid;
                        bestW = o.W; bestH = o.H; bestD = o.D;
                    }
                }
            }
        }

        // Push back candidates not chosen (or all if no choice)
        for (int i = 0; i < candSpaceIds.Count; i++)
        {
            int sid = candSpaceIds[i];
            if (sid == bestSid) continue; // keep it out (will be killed if we place; if no placement, bestSid=-1)
            if (alive[sid] != 0) heap.Push(sid);
        }
        if (bestSid < 0)
        {
            // No placement; push back all candidates (we skipped only bestSid==-1, so none skipped)
            return false;
        }

        // Place box at the origin of chosen space; kill that space; split into 3 disjoint child spaces
        Space chosen = spaces[bestSid];
        placements.Add(new Placement(box.Index, chosen.X, chosen.Y, chosen.Z));

        alive[bestSid] = 0;
        aliveCount--;

        // Right (X+)
        int rw = chosen.W - bestW;
        if (rw > 0)
        {
            int id = spaces.Count;
            spaces.Add(new Space(chosen.X + bestW, chosen.Y, chosen.Z, rw, chosen.H, chosen.D));
            alive.Add(1);
            aliveCount++;
            heap.Push(id);
        }

        // Top (Y+), limited to placed box footprint in X (bestW), full depth
        int th = chosen.H - bestH;
        if (th > 0)
        {
            int id = spaces.Count;
            spaces.Add(new Space(chosen.X, chosen.Y + bestH, chosen.Z, bestW, th, chosen.D));
            alive.Add(1);
            aliveCount++;
            heap.Push(id);
        }

        // Front (Z+), limited to placed footprint in X and Y
        int fd = chosen.D - bestD;
        if (fd > 0)
        {
            int id = spaces.Count;
            spaces.Add(new Space(chosen.X, chosen.Y, chosen.Z + bestD, bestW, bestH, fd));
            alive.Add(1);
            aliveCount++;
            heap.Push(id);
        }

        return true;
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int N = fs.NextInt();
        int W = fs.NextInt();
        int H = fs.NextInt();
        int D = fs.NextInt();

        var boxes = new List<Box>(N);
        for (int i = 0; i < N; i++)
        {
            int w = fs.NextInt();
            int h = fs.NextInt();
            int d = fs.NextInt();

            var b = new Box
            {
                Index = i,
                W = w,
                H = h,
                D = d,
                Vol = (long)w * h * d
            };
            b.Orients = BuildUniqueOrients(w, h, d);

            // Filter boxes that can never fit in container in any orientation (fast pre-check)
            if (FitsContainerAnyOrient(b, W, H, D))
                boxes.Add(b);
        }

        if (boxes.Count == 0)
        {
            Console.WriteLine(0);
            return;
        }

        // Adaptive: run multiple strategies, pick best by count, then by utilized volume
        var results = new List<(List<Placement> placements, long usedVol)>(3);

        void Run(Strategy s)
        {
            var pl = Pack(boxes, W, H, D, s);
            long used = 0;
            // placements store original indices; compute used volume by looking up dims from input boxes list is costly.
            // But we can approximate by summing from filtered list via a dictionary; better: build volume lookup once.
            results.Add((pl, 0)); // fill later
        }

        // Build volume lookup for all original indices (only for filtered + cheap)
        var volByIndex = new long[N];
        foreach (var b in boxes) volByIndex[b.Index] = b.Vol;

        // Decide which strategies to run based on scale
        // (Large N: 2 strategies; small N: 3 strategies)
        if (boxes.Count <= 800)
        {
            Run(Strategy.SmallFirst);
            Run(Strategy.MinDimFirst);
            Run(Strategy.BigFirst);
        }
        else
        {
            Run(Strategy.SmallFirst);
            Run(Strategy.MinDimFirst);
        }

        // Compute used volume and pick best
        int bestIdx = 0;
        int bestCount = -1;
        long bestUsed = -1;

        for (int i = 0; i < results.Count; i++)
        {
            var pl = results[i].placements;
            long used = 0;
            for (int j = 0; j < pl.Count; j++) used += volByIndex[pl[j].BoxIndex];
            results[i] = (pl, used);

            if (pl.Count > bestCount || (pl.Count == bestCount && used > bestUsed))
            {
                bestIdx = i;
                bestCount = pl.Count;
                bestUsed = used;
            }
        }

        var best = results[bestIdx].placements;

        var sb = new StringBuilder(best.Count * 24 + 32);
        sb.Append(best.Count).Append('\n');
        for (int i = 0; i < best.Count; i++)
        {
            var p = best[i];
            sb.Append(p.BoxIndex).Append(' ')
              .Append(p.X).Append(' ')
              .Append(p.Y).Append(' ')
              .Append(p.Z).Append('\n');
        }
        Console.Write(sb.ToString());
    }
}
