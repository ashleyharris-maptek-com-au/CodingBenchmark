using System;
using System.Diagnostics;
using System.Globalization;
using System.IO;

public static class Program
{
    // ---------- Fast scanner ----------
    private sealed class FastScanner
    {
        private readonly Stream _s;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream s) { _s = s; }

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buf[_ptr++];
        }

        public string Next()
        {
            byte c;
            do { c = Read(); if (c == 0) return ""; } while (c <= 32);
            int start = _ptr - 1;
            while (true)
            {
                c = Read();
                if (c == 0 || c <= 32) break;
            }
            int end = (c == 0) ? _ptr : _ptr - 1;
            return System.Text.Encoding.ASCII.GetString(_buf, start, end - start);
        }

        public int NextInt() => int.Parse(Next(), CultureInfo.InvariantCulture);
        public double NextDouble() => double.Parse(Next(), CultureInfo.InvariantCulture);
    }

    // ---------- Math / orbit features ----------
    private struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);

        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public Vec3 Cross(Vec3 b) => new Vec3(Y * b.Z - Z * b.Y, Z * b.X - X * b.Z, X * b.Y - Y * b.X);
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public Vec3 Unit()
        {
            double n = Norm();
            return n > 0 ? this / n : new Vec3(0, 0, 0);
        }
    }

    private sealed class OrbitFeat
    {
        public Vec3 R, V;
        public Vec3 NormalUnit;  // orbital plane normal (unit)
        public Vec3 RUnit;       // position direction at epoch
        public double A;         // semimajor axis (km) (fallback to r for degenerate)
        public double E;         // eccentricity magnitude
        public double P;         // semi-latus rectum (km) (fallback to r)
        public double H;         // specific angular momentum magnitude
        public double Rmag;      // |r|
        public double Vmag;      // |v|
    }

    private static OrbitFeat ComputeFeat(Vec3 r, Vec3 v, double mu)
    {
        var f = new OrbitFeat();
        f.R = r; f.V = v;
        f.Rmag = r.Norm();
        f.Vmag = v.Norm();
        f.RUnit = f.Rmag > 0 ? r / f.Rmag : new Vec3(1, 0, 0);

        Vec3 hVec = r.Cross(v);
        f.H = hVec.Norm();
        f.NormalUnit = f.H > 0 ? hVec / f.H : new Vec3(0, 0, 1);

        // Eccentricity vector: e = (v x h)/mu - r/|r|
        Vec3 eVec = (v.Cross(hVec) * (1.0 / mu)) - (f.Rmag > 0 ? (r / f.Rmag) : new Vec3(1, 0, 0));
        f.E = eVec.Norm();

        // Semi-major axis from specific orbital energy
        double energy = 0.5 * f.Vmag * f.Vmag - mu / Math.Max(1e-9, f.Rmag);
        if (Math.Abs(energy) > 1e-12)
        {
            double a = -mu / (2.0 * energy);
            // If hyperbolic (a < 0), we still keep magnitude as a scale to avoid NaNs.
            f.A = (double.IsFinite(a) && Math.Abs(a) > 1e-9) ? a : f.Rmag;
        }
        else
        {
            f.A = f.Rmag;
        }

        // Semi-latus rectum p = h^2/mu (valid for conics); clamp to positive scale.
        double p = (f.H > 0) ? (f.H * f.H / mu) : f.Rmag;
        f.P = (double.IsFinite(p) && p > 1e-6) ? p : Math.Max(1.0, f.Rmag);

        return f;
    }

    private static double Clamp(double x, double a, double b) => x < a ? a : (x > b ? b : x);

    // Symmetric "delta-V-like" proxy between two orbits at epoch.
    // This is NOT a full rendezvous solver (Lambert + phasing), but it is stable, fast, and scales well for routing.
    private static double TransferCost(OrbitFeat o1, OrbitFeat o2, double mu)
    {
        // Plane angle
        double cosPlane = Clamp(o1.NormalUnit.Dot(o2.NormalUnit), -1.0, 1.0);
        double planeAngle = Math.Acos(cosPlane);

        // Characteristic radius/speed scale
        double rChar = 0.5 * (o1.P + o2.P);
        if (!(rChar > 1e-6) || !double.IsFinite(rChar)) rChar = 0.5 * (Math.Max(1.0, o1.Rmag) + Math.Max(1.0, o2.Rmag));
        if (!(rChar > 1e-6)) rChar = 7000.0;

        double vChar = Math.Sqrt(mu / rChar);

        // Approx plane-change DV
        double dvPlane = 2.0 * vChar * Math.Sin(0.5 * planeAngle);

        // In-plane "shape/size" proxy (normalized)
        double a1 = o1.A, a2 = o2.A;
        double aAvg = 0.5 * (Math.Abs(a1) + Math.Abs(a2)) + 1.0;
        double aDiffN = Math.Abs(a1 - a2) / aAvg;
        double eDiff = Math.Abs(o1.E - o2.E);

        double dvInPlane = vChar * (0.75 * aDiffN + 1.35 * eDiff);

        // Small epoch phase penalty (because waiting is allowed in reality); keep it weak.
        double cosPhase = Clamp(o1.RUnit.Dot(o2.RUnit), -1.0, 1.0);
        double phase = Math.Acos(cosPhase);
        double dvPhase = 0.05 * vChar * phase;

        double dv = dvPlane + dvInPlane + dvPhase;

        // Always positive, avoid zeros that can create ties.
        if (!double.IsFinite(dv) || dv < 1e-9) dv = 1e-9;
        return dv;
    }

    // ---------- Exact DP for small N: shortest Hamiltonian path starting from "start orbit" ----------
    private static int[] SolveExactDP(int n, double[] costStart, double[,] cost)
    {
        int states = 1 << n;
        int stride = n;
        double[] dp = new double[states * stride];
        int[] parent = new int[states * stride];

        const double INF = 1e300;
        for (int i = 0; i < dp.Length; i++) { dp[i] = INF; parent[i] = -1; }

        for (int i = 0; i < n; i++)
        {
            int mask = 1 << i;
            dp[mask * stride + i] = costStart[i];
            parent[mask * stride + i] = -2; // start marker
        }

        for (int mask = 1; mask < states; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                double cur = dp[mask * stride + last];
                if (cur >= INF) continue;
                int remaining = (~mask) & (states - 1);
                while (remaining != 0)
                {
                    int lsb = remaining & -remaining;
                    int j = BitOperations.TrailingZeroCount((uint)lsb);
                    remaining -= lsb;

                    int nmask = mask | (1 << j);
                    double cand = cur + cost[last, j];
                    int idx = nmask * stride + j;
                    if (cand < dp[idx])
                    {
                        dp[idx] = cand;
                        parent[idx] = last;
                    }
                }
            }
        }

        int full = states - 1;
        double best = INF;
        int bestLast = 0;
        for (int last = 0; last < n; last++)
        {
            double v = dp[full * stride + last];
            if (v < best) { best = v; bestLast = last; }
        }

        int[] tour = new int[n];
        int curMask = full;
        int curLast = bestLast;
        for (int pos = n - 1; pos >= 0; pos--)
        {
            tour[pos] = curLast;
            int idx = curMask * stride + curLast;
            int p = parent[idx];
            curMask ^= (1 << curLast);
            curLast = p;
            if (curLast == -2 && pos != 0) curLast = 0; // should not happen; safety
        }
        return tour;
    }

    // ---------- Heuristics for medium/large/extreme N ----------
    private struct XorShift64
    {
        private ulong _x;
        public XorShift64(ulong seed) { _x = seed != 0 ? seed : 88172645463393265UL; }
        public ulong NextU64()
        {
            ulong x = _x;
            x ^= x << 7;
            x ^= x >> 9;
            _x = x;
            return x;
        }
        public int NextInt(int bound) => (int)(NextU64() % (uint)bound);
        public double NextDouble() => (NextU64() >> 11) * (1.0 / (1UL << 53));
    }

    private static double TourCost(int[] tour, double[] costStart, double[,] cost)
    {
        int n = tour.Length;
        if (n == 0) return 0.0;
        double sum = costStart[tour[0]];
        for (int i = 0; i + 1 < n; i++) sum += cost[tour[i], tour[i + 1]];
        return sum;
    }

    private static int[] BuildRandomizedCheapestInsertion(int n, double[] costStart, double[,] cost, ref XorShift64 rng)
    {
        if (n == 0) return Array.Empty<int>();
        if (n == 1) return new[] { 0 };

        bool[] used = new bool[n];

        // Choose first node among a few best by start-cost to stabilize early path.
        int sample = Math.Min(8, n);
        int[] bestFirst = new int[sample];
        double[] bestVal = new double[sample];
        for (int i = 0; i < sample; i++) { bestVal[i] = 1e300; bestFirst[i] = -1; }
        for (int i = 0; i < n; i++)
        {
            double v = costStart[i];
            // insert into top list
            for (int k = 0; k < sample; k++)
            {
                if (v < bestVal[k])
                {
                    for (int t = sample - 1; t > k; t--) { bestVal[t] = bestVal[t - 1]; bestFirst[t] = bestFirst[t - 1]; }
                    bestVal[k] = v; bestFirst[k] = i;
                    break;
                }
            }
        }
        int first = bestFirst[rng.NextInt(sample)];
        if (first < 0) first = 0;

        used[first] = true;
        int[] tour = new int[n];
        int tourLen = 1;
        tour[0] = first;

        // Choose second as nearest to first (break ties randomly)
        int second = -1;
        double best = 1e300;
        for (int j = 0; j < n; j++)
        {
            if (used[j]) continue;
            double c = cost[first, j];
            if (c < best - 1e-12 || (Math.Abs(c - best) <= 1e-12 && rng.NextInt(2) == 0))
            {
                best = c;
                second = j;
            }
        }
        if (second < 0)
        {
            // fallback
            for (int j = 0; j < n; j++) if (!used[j]) { second = j; break; }
        }
        used[second] = true;
        tour[1] = second;
        tourLen = 2;

        // Insert remaining nodes: randomized among top few cheapest insertions each step.
        int R = (n <= 30) ? 5 : 3; // top-R candidate insertions to randomize
        while (tourLen < n)
        {
            // Track top R (node, position) by delta increase
            double[] bestD = new double[R];
            int[] bestNode = new int[R];
            int[] bestPos = new int[R];
            for (int i = 0; i < R; i++) { bestD[i] = 1e300; bestNode[i] = -1; bestPos[i] = -1; }

            for (int u = 0; u < n; u++)
            {
                if (used[u]) continue;

                // Positions: 0..tourLen (insert before index pos)
                // Delta for insert at beginning:
                double deltaBegin = costStart[u] + cost[u, tour[0]] - costStart[tour[0]];
                Consider(deltaBegin, u, 0);

                // Delta for insert at end:
                double deltaEnd = cost[tour[tourLen - 1], u];
                Consider(deltaEnd, u, tourLen);

                // Delta for insert between i and i+1:
                for (int i = 0; i + 1 < tourLen; i++)
                {
                    int a = tour[i], b = tour[i + 1];
                    double delta = cost[a, u] + cost[u, b] - cost[a, b];
                    Consider(delta, u, i + 1);
                }
            }

            int pick = 0;
            // choose among available entries (non -1)
            int avail = 0;
            for (int i = 0; i < R; i++) if (bestNode[i] >= 0) avail++;
            if (avail > 1)
            {
                // Bias toward better but still random
                double r = rng.NextDouble();
                if (r < 0.65) pick = 0;
                else if (r < 0.88) pick = Math.Min(1, avail - 1);
                else pick = Math.Min(2, avail - 1);
            }

            int node = bestNode[pick];
            int posIns = bestPos[pick];
            if (node < 0)
            {
                // fallback: add any unused at end
                for (int u = 0; u < n; u++) if (!used[u]) { node = u; posIns = tourLen; break; }
            }

            // Insert
            for (int i = tourLen; i > posIns; i--) tour[i] = tour[i - 1];
            tour[posIns] = node;
            used[node] = true;
            tourLen++;

            void Consider(double d, int nodeCand, int posCand)
            {
                for (int k = 0; k < R; k++)
                {
                    if (d < bestD[k])
                    {
                        for (int t = R - 1; t > k; t--)
                        {
                            bestD[t] = bestD[t - 1];
                            bestNode[t] = bestNode[t - 1];
                            bestPos[t] = bestPos[t - 1];
                        }
                        bestD[k] = d;
                        bestNode[k] = nodeCand;
                        bestPos[k] = posCand;
                        return;
                    }
                }
            }
        }

        return tour;
    }

    private static void ReverseSegment(int[] a, int l, int r)
    {
        while (l < r)
        {
            int t = a[l];
            a[l] = a[r];
            a[r] = t;
            l++; r--;
        }
    }

    private static void TwoOptImprovePath(int[] tour, double[] costStart, double[,] cost, Stopwatch sw, long timeLimitTicks)
    {
        int n = tour.Length;
        if (n < 3) return;

        double EdgeCost(int from, int to)
        {
            if (to < 0) return 0.0;      // end marker
            if (from < 0) return costStart[to]; // from "start orbit"
            return cost[from, to];
        }

        while (true)
        {
            if (sw.ElapsedTicks > timeLimitTicks) return;
            bool improved = false;

            for (int i = 0; i < n - 1; i++)
            {
                int prev = (i == 0) ? -1 : tour[i - 1];
                int a = tour[i];

                for (int k = i + 1; k < n; k++)
                {
                    if (sw.ElapsedTicks > timeLimitTicks) return;

                    int b = tour[k];
                    int next = (k == n - 1) ? -1 : tour[k + 1];

                    // Old edges: prev->a, b->next
                    // New edges after reversing [i..k]: prev->b, a->next
                    double oldCost = EdgeCost(prev, a) + EdgeCost(b, next);
                    double newCost = EdgeCost(prev, b) + EdgeCost(a, next);

                    if (newCost + 1e-12 < oldCost)
                    {
                        ReverseSegment(tour, i, k);
                        improved = true;
                        break;
                    }
                }

                if (improved) break;
            }

            if (!improved) return;
        }
    }

    private static void OrOpt1ImprovePath(int[] tour, double[] costStart, double[,] cost, Stopwatch sw, long timeLimitTicks)
    {
        int n = tour.Length;
        if (n < 4) return;

        double EdgeCost(int from, int to)
        {
            if (to < 0) return 0.0;
            if (from < 0) return costStart[to];
            return cost[from, to];
        }

        while (true)
        {
            if (sw.ElapsedTicks > timeLimitTicks) return;
            bool improved = false;

            for (int i = 0; i < n; i++)
            {
                if (sw.ElapsedTicks > timeLimitTicks) return;
                int node = tour[i];
                int prevI = (i == 0) ? -1 : tour[i - 1];
                int nextI = (i == n - 1) ? -1 : tour[i + 1];

                double removeDelta = -EdgeCost(prevI, node) - EdgeCost(node, nextI) + EdgeCost(prevI, nextI);

                // Try inserting node at position j (before index j), skipping its current neighborhood
                for (int j = 0; j <= n; j++)
                {
                    if (j == i || j == i + 1) continue; // no-op positions
                    int prevJ = (j == 0) ? -1 : tour[j - 1];
                    int nextJ = (j == n) ? -1 : tour[j];

                    // If moving forward, after removal indices shift; handle by using current array logic with skip rules above.

                    double insertDelta = -EdgeCost(prevJ, nextJ) + EdgeCost(prevJ, node) + EdgeCost(node, nextJ);
                    double totalDelta = removeDelta + insertDelta;

                    if (totalDelta + 1e-12 < 0.0)
                    {
                        // Apply move: remove at i, insert before j (after removal).
                        int[] tmp = new int[n];
                        int idx = 0;
                        for (int t = 0; t < n; t++) if (t != i) tmp[idx++] = tour[t];

                        int insertPos = j;
                        if (j > i) insertPos = j - 1;

                        for (int t = n - 1; t > insertPos; t--) tmp[t] = tmp[t - 1];
                        tmp[insertPos] = node;

                        Array.Copy(tmp, tour, n);
                        improved = true;
                        break;
                    }
                }
                if (improved) break;
            }

            if (!improved) return;
        }
    }

    private static int[] SolveHeuristic(int n, double[] costStart, double[,] cost, int timeLimitMs)
    {
        var sw = Stopwatch.StartNew();
        long timeLimitTicks = sw.ElapsedTicks + (long)(timeLimitMs * (Stopwatch.Frequency / 1000.0));

        // Restart count tuned for stability/scale.
        int restarts =
            n <= 20 ? 40 :
            n <= 35 ? 25 :
            n <= 60 ? 15 : 10;

        // Ensure at least a couple restarts if we have time.
        if (timeLimitMs < 2000) restarts = Math.Min(restarts, 8);

        int[] bestTour = null!;
        double bestCost = 1e300;

        // Seed from time + n for variability, but deterministic-enough per run.
        ulong seed = (ulong)DateTime.UtcNow.Ticks ^ (ulong)(n * 0x9E3779B97F4A7C15UL);
        var rng = new XorShift64(seed);

        for (int r = 0; r < restarts; r++)
        {
            if (sw.ElapsedTicks > timeLimitTicks) break;

            var localRng = new XorShift64(rng.NextU64());
            int[] tour = BuildRandomizedCheapestInsertion(n, costStart, cost, ref localRng);

            // Local improvement
            TwoOptImprovePath(tour, costStart, cost, sw, timeLimitTicks);
            OrOpt1ImprovePath(tour, costStart, cost, sw, timeLimitTicks);
            TwoOptImprovePath(tour, costStart, cost, sw, timeLimitTicks);

            double c = TourCost(tour, costStart, cost);
            if (c < bestCost)
            {
                bestCost = c;
                bestTour = (int[])tour.Clone();
            }
        }

        // If nothing due to extremely tiny time limit, fall back to simple greedy.
        if (bestTour == null)
        {
            bestTour = new int[n];
            bool[] used = new bool[n];
            int cur = 0;
            // start at minimal costStart
            double best = 1e300;
            for (int i = 0; i < n; i++)
            {
                if (costStart[i] < best) { best = costStart[i]; cur = i; }
            }
            bestTour[0] = cur;
            used[cur] = true;
            for (int pos = 1; pos < n; pos++)
            {
                int next = -1;
                double b = 1e300;
                for (int j = 0; j < n; j++)
                {
                    if (used[j]) continue;
                    double cc = cost[cur, j];
                    if (cc < b) { b = cc; next = j; }
                }
                if (next < 0) for (int j = 0; j < n; j++) if (!used[j]) { next = j; break; }
                bestTour[pos] = next;
                used[next] = true;
                cur = next;
            }
        }

        return bestTour;
    }

    // Polyfill for trailing zero count without requiring newer intrinsics packages.
    private static class BitOperations
    {
        public static int TrailingZeroCount(uint v)
        {
            if (v == 0) return 32;
            int c = 0;
            while ((v & 1) == 0) { v >>= 1; c++; }
            return c;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        string t0 = fs.Next();
        if (t0.Length == 0) return;

        int N = int.Parse(t0, CultureInfo.InvariantCulture);
        double mu = fs.NextDouble();

        Vec3 r0 = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
        Vec3 v0 = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());

        OrbitFeat start = ComputeFeat(r0, v0, mu);

        OrbitFeat[] st = new OrbitFeat[N];
        for (int i = 0; i < N; i++)
        {
            Vec3 r = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
            Vec3 v = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
            st[i] = ComputeFeat(r, v, mu);
        }

        // Build costs
        double[] costStart = new double[N];
        for (int i = 0; i < N; i++) costStart[i] = TransferCost(start, st[i], mu);

        double[,] cost = new double[N, N];
        for (int i = 0; i < N; i++)
        {
            cost[i, i] = 0.0;
            for (int j = i + 1; j < N; j++)
            {
                double c = TransferCost(st[i], st[j], mu);
                cost[i, j] = c;
                cost[j, i] = c;
            }
        }

        int[] tour;

        // Exact for small N, heuristic otherwise.
        // (Exact DP grows as O(N^2 2^N).)
        if (N <= 16)
        {
            tour = SolveExactDP(N, costStart, cost);
        }
        else
        {
            // Time budget: scales gently; stays comfortably under the 5-minute limit even for N=100.
            int timeLimitMs = 1500 + 2 * N * N;        // N=100 => ~21.5s
            timeLimitMs = Math.Min(timeLimitMs, 120_000); // cap 2 minutes
            tour = SolveHeuristic(N, costStart, cost, timeLimitMs);
        }

        // Output visitation order (station indices 0..N-1)
        using var w = new StreamWriter(Console.OpenStandardOutput());
        for (int i = 0; i < tour.Length; i++)
        {
            if (i != 0) w.Write(' ');
            w.Write(tour[i]);
        }
        w.WriteLine();
    }
}
