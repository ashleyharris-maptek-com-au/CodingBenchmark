use std::cmp::Reverse;
use std::collections::HashMap;
use std::io::{self, Read, Write};

const SCALE_KEY: i64 = 1_000_000;

#[derive(Clone)]
struct Graph {
    n: usize,
    offsets: Vec<usize>, // len n+1
    adj: Vec<u32>,       // neighbors in CSR
    deg: Vec<u32>,       // degree (excluding self-loops)
}

struct FastInput {
    data: Vec<u8>,
    idx: usize,
}
impl FastInput {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data).unwrap();
        Self { data, idx: 0 }
    }
    fn skip_ws(&mut self) {
        while self.idx < self.data.len() && self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
    }
    fn next_usize(&mut self) -> Option<usize> {
        self.skip_ws();
        if self.idx >= self.data.len() {
            return None;
        }
        let mut x: usize = 0;
        while self.idx < self.data.len() {
            let c = self.data[self.idx];
            if !c.is_ascii_digit() {
                break;
            }
            x = x * 10 + (c - b'0') as usize;
            self.idx += 1;
        }
        Some(x)
    }
}

fn read_graph() -> Graph {
    let mut fi = FastInput::new();
    let n = fi.next_usize().unwrap_or(0);
    let m = fi.next_usize().unwrap_or(0);

    let mut edges: Vec<(u32, u32)> = Vec::with_capacity(m);
    let mut deg = vec![0u32; n];

    for _ in 0..m {
        let u = match fi.next_usize() {
            Some(v) => v,
            None => break,
        };
        let v = match fi.next_usize() {
            Some(v) => v,
            None => break,
        };
        if u >= n || v >= n {
            continue; // ignore invalid endpoints
        }
        if u == v {
            continue; // ignore self-loop for layout purposes
        }
        let uu = u as u32;
        let vv = v as u32;
        edges.push((uu, vv));
        deg[u] = deg[u].saturating_add(1);
        deg[v] = deg[v].saturating_add(1);
    }

    let mut offsets = vec![0usize; n + 1];
    for i in 0..n {
        offsets[i + 1] = offsets[i] + deg[i] as usize;
    }
    let mut adj = vec![0u32; offsets[n]];
    let mut cur = offsets[..n].to_vec();

    for &(uu, vv) in &edges {
        let u = uu as usize;
        let v = vv as usize;

        adj[cur[u]] = vv;
        cur[u] += 1;
        adj[cur[v]] = uu;
        cur[v] += 1;
    }

    Graph { n, offsets, adj, deg }
}

#[inline]
fn neighbors<'a>(g: &'a Graph, u: usize) -> &'a [u32] {
    &g.adj[g.offsets[u]..g.offsets[u + 1]]
}

fn splitmix64(mut x: u64) -> u64 {
    x = x.wrapping_add(0x9E3779B97F4A7C15);
    let mut z = x;
    z = (z ^ (z >> 30)).wrapping_mul(0xBF58476D1CE4E5B9);
    z = (z ^ (z >> 27)).wrapping_mul(0x94D049BB133111EB);
    z ^ (z >> 31)
}

fn component_edges_undirected(g: &Graph, nodes: &[usize]) -> usize {
    let mut sum: u64 = 0;
    for &u in nodes {
        sum += g.deg[u] as u64;
    }
    (sum / 2) as usize
}

/// Rooted tree layout (planar, crossing-free for trees) using postorder assignment.
/// Produces y = depth, x = leaf-order midpoint of subtree.
fn layout_tree_component(
    g: &Graph,
    cid: i32,
    nodes: &[usize],
    comp_id: &[i32],
    root: usize,
    parent: &mut [i32],
    depth: &mut [i32],
    left: &mut [i32],
    right: &mut [i32],
    x: &mut [f32],
    y: &mut [f32],
) {
    // Initialize arrays for nodes
    for &u in nodes {
        parent[u] = -1;
        depth[u] = 0;
        left[u] = -1;
        right[u] = -1;
    }

    // Build parent, depth via stack DFS (tree => no cross edges besides parent)
    let mut stack = Vec::with_capacity(nodes.len());
    parent[root] = root as i32;
    depth[root] = 0;
    stack.push(root);

    while let Some(u) = stack.pop() {
        let pu = parent[u] as usize;
        let du = depth[u];
        for &vv in neighbors(g, u) {
            let v = vv as usize;
            if comp_id[v] != cid {
                continue;
            }
            if v == pu {
                continue;
            }
            if parent[v] != -1 {
                continue;
            }
            parent[v] = u as i32;
            depth[v] = du + 1;
            stack.push(v);
        }
    }

    // Postorder traversal using explicit enter/exit states
    let mut st: Vec<(usize, u8)> = Vec::with_capacity(nodes.len() * 2);
    st.push((root, 0));

    let mut cur_leaf_x: i32 = 0;
    while let Some((u, state)) = st.pop() {
        if state == 0 {
            st.push((u, 1));
            // push children
            let pu = parent[u] as usize;
            // reverse order for deterministic-ish output
            for &vv in neighbors(g, u).iter().rev() {
                let v = vv as usize;
                if comp_id[v] != cid {
                    continue;
                }
                if v == pu {
                    continue;
                }
                if parent[v] == u as i32 {
                    st.push((v, 0));
                }
            }
        } else {
            // exit: compute left/right and x[u]
            let mut has_child = false;
            let mut l = i32::MAX;
            let mut r = i32::MIN;

            let pu = parent[u] as usize;
            for &vv in neighbors(g, u) {
                let v = vv as usize;
                if comp_id[v] != cid {
                    continue;
                }
                if v == pu {
                    continue;
                }
                if parent[v] == u as i32 {
                    has_child = true;
                    l = l.min(left[v]);
                    r = r.max(right[v]);
                }
            }

            if !has_child {
                left[u] = cur_leaf_x;
                right[u] = cur_leaf_x;
                x[u] = cur_leaf_x as f32;
                cur_leaf_x += 1;
            } else {
                left[u] = l;
                right[u] = r;
                x[u] = (l + r) as f32 * 0.5;
            }
            y[u] = depth[u] as f32;
        }
    }

    // small deterministic jitter to reduce exact overlaps in degenerate cases
    for &u in nodes {
        let h = splitmix64(u as u64);
        let j = ((h & 0xFFFF) as f32) / 65535.0 - 0.5;
        x[u] += j * 0.01;
    }
}

fn bary_key(g: &Graph, u: usize, neighbor_level: i32, level: &[i32], pos: &[i32]) -> i64 {
    let mut sum: i64 = 0;
    let mut cnt: i64 = 0;
    for &vv in neighbors(g, u) {
        let v = vv as usize;
        if level[v] == neighbor_level {
            sum += pos[v] as i64;
            cnt += 1;
        }
    }
    if cnt == 0 {
        (pos[u] as i64) * SCALE_KEY
    } else {
        (sum * SCALE_KEY) / cnt
    }
}

/// Layered BFS + barycentric reordering (Sugiyama-style heuristic).
fn layout_layered_component(
    g: &Graph,
    nodes: &[usize],
    root: usize,
    level: &mut [i32],
    pos: &mut [i32],
    x: &mut [f32],
    y: &mut [f32],
) {
    // reset levels for component
    for &u in nodes {
        level[u] = -1;
    }

    // BFS for levels
    let mut q: Vec<usize> = Vec::with_capacity(nodes.len());
    let mut head = 0usize;
    level[root] = 0;
    q.push(root);

    while head < q.len() {
        let u = q[head];
        head += 1;
        let lu = level[u];
        for &vv in neighbors(g, u) {
            let v = vv as usize;
            if level[v] == -1 {
                level[v] = lu + 1;
                q.push(v);
            }
        }
    }

    // In rare cases (shouldn't for connected component), set remaining to 0
    for &u in nodes {
        if level[u] < 0 {
            level[u] = 0;
        }
    }

    let mut max_level = 0i32;
    for &u in nodes {
        max_level = max_level.max(level[u]);
    }
    let lcnt = (max_level as usize) + 1;

    // Counting sort nodes by level into `order`
    let mut counts = vec![0usize; lcnt];
    for &u in nodes {
        counts[level[u] as usize] += 1;
    }
    let mut level_offsets = vec![0usize; lcnt + 1];
    for i in 0..lcnt {
        level_offsets[i + 1] = level_offsets[i] + counts[i];
    }
    let mut cursor = level_offsets[..lcnt].to_vec();
    let mut order = vec![0usize; nodes.len()];
    for &u in nodes {
        let li = level[u] as usize;
        order[cursor[li]] = u;
        cursor[li] += 1;
    }

    // Initial per-level ordering by degree (helps a bit for small/medium graphs)
    for li in 0..lcnt {
        let (s, e) = (level_offsets[li], level_offsets[li + 1]);
        order[s..e].sort_unstable_by_key(|&u| Reverse(g.deg[u]));
    }

    // helper: update pos for a level slice
    let mut update_pos_level = |li: usize, ord: &[usize], pos: &mut [i32], level_offsets: &[usize]| {
        let (s, e) = (level_offsets[li], level_offsets[li + 1]);
        for (i, &u) in ord[s..e].iter().enumerate() {
            pos[u] = i as i32;
        }
    };

    // Initialize pos for all
    for li in 0..lcnt {
        update_pos_level(li, &order, pos, &level_offsets);
    }

    let n = nodes.len();
    let iters = if n <= 50 {
        24
    } else if n <= 500 {
        12
    } else if n <= 5_000 {
        6
    } else if n <= 20_000 {
        4
    } else {
        2
    };

    let mut max_layer_size = 0usize;
    for &c in &counts {
        max_layer_size = max_layer_size.max(c);
    }
    let mut pairs: Vec<(i64, usize)> = Vec::with_capacity(max_layer_size);

    let mut reorder_level = |li: usize, neighbor_li: usize,
                             g: &Graph,
                             order: &mut [usize],
                             level: &[i32],
                             pos: &mut [i32],
                             level_offsets: &[usize],
                             pairs: &mut Vec<(i64, usize)>| {
        let (s, e) = (level_offsets[li], level_offsets[li + 1]);
        if e - s <= 1 {
            if e - s == 1 {
                pos[order[s]] = 0;
            }
            return;
        }

        pairs.clear();
        pairs.extend(order[s..e].iter().map(|&u| {
            let k = bary_key(g, u, neighbor_li as i32, level, pos);
            (k, u)
        }));

        pairs.sort_unstable_by(|a, b| a.0.cmp(&b.0).then_with(|| a.1.cmp(&b.1)));

        for (i, &(_, u)) in pairs.iter().enumerate() {
            order[s + i] = u;
            pos[u] = i as i32;
        }
    };

    for _ in 0..iters {
        // forward sweep: align level i to i-1
        for li in 1..lcnt {
            reorder_level(li, li - 1, g, &mut order, level, pos, &level_offsets, &mut pairs);
        }
        // backward sweep: align level i to i+1
        if lcnt >= 2 {
            for li in (0..=(lcnt - 2)).rev() {
                reorder_level(li, li + 1, g, &mut order, level, pos, &level_offsets, &mut pairs);
            }
        }
    }

    // Assign coordinates
    let x_spacing: f32 = 1.0;
    let y_spacing: f32 = 2.0;
    for li in 0..lcnt {
        let (s, e) = (level_offsets[li], level_offsets[li + 1]);
        let len = (e - s) as f32;
        let center = (len - 1.0) * 0.5;
        for (i, &u) in order[s..e].iter().enumerate() {
            let mut xx = (i as f32 - center) * x_spacing;
            let yy = li as f32 * y_spacing;

            // tiny deterministic jitter to reduce exact overlaps
            let h = splitmix64(u as u64);
            let j = ((h & 0xFFFF) as f32) / 65535.0 - 0.5;
            xx += j * 0.01;

            x[u] = xx;
            y[u] = yy;
        }
    }
}

/// Optional light 1D-ish refinement (only for small components) to spread nodes horizontally.
/// This is intentionally conservative to keep runtime predictable.
fn refine_small_component(
    g: &Graph,
    cid: i32,
    nodes: &[usize],
    comp_id: &[i32],
    x: &mut [f32],
    y: &mut [f32],
    disp_x: &mut [f32],
) {
    let n = nodes.len();
    if n <= 2 {
        return;
    }
    let m = component_edges_undirected(g, nodes);
    if n > 2500 || m > 200_000 {
        return;
    }

    // bounding box
    let mut minx = f32::INFINITY;
    let mut maxx = f32::NEG_INFINITY;
    let mut miny = f32::INFINITY;
    let mut maxy = f32::NEG_INFINITY;
    for &u in nodes {
        minx = minx.min(x[u]);
        maxx = maxx.max(x[u]);
        miny = miny.min(y[u]);
        maxy = maxy.max(y[u]);
    }
    let w = (maxx - minx).max(1.0);
    let h = (maxy - miny).max(1.0);

    let area = (w * h).max((n as f32) * 4.0);
    let k = (area / (n as f32)).sqrt().max(0.5);

    let iters = if n <= 200 { 40 } else if n <= 1000 { 25 } else { 15 };
    let mut temp = k * 2.0;

    // grid buckets for repulsion
    let cell = (k * 1.5).max(0.5);

    for _ in 0..iters {
        for &u in nodes {
            disp_x[u] = 0.0;
        }

        // build buckets
        let mut buckets: HashMap<i64, Vec<usize>> = HashMap::with_capacity(n * 2);
        for &u in nodes {
            let cx = (x[u] / cell).floor() as i32;
            let cy = (y[u] / cell).floor() as i32;
            let key = ((cx as i64) << 32) ^ (cy as u32 as i64);
            buckets.entry(key).or_default().push(u);
        }

        // repulsion (approx): check 3x3 neighborhood buckets
        for &u in nodes {
            let cx = (x[u] / cell).floor() as i32;
            let cy = (y[u] / cell).floor() as i32;
            let mut fx = 0.0f32;

            for dx_cell in -1..=1 {
                for dy_cell in -1..=1 {
                    let nx = cx + dx_cell;
                    let ny = cy + dy_cell;
                    let key = ((nx as i64) << 32) ^ (ny as u32 as i64);
                    if let Some(vecu) = buckets.get(&key) {
                        for &v in vecu {
                            if v == u {
                                continue;
                            }
                            let dx = x[u] - x[v];
                            let dy = y[u] - y[v];
                            let dist2 = dx * dx + dy * dy + 1e-6;
                            // FR repulsion ~ k^2 / dist
                            fx += dx * (k * k / dist2);
                        }
                    }
                }
            }
            disp_x[u] += fx;
        }

        // attraction along edges (horizontal component only; keeps layering stable)
        for &u in nodes {
            for &vv in neighbors(g, u) {
                let v = vv as usize;
                if comp_id[v] != cid {
                    continue;
                }
                if u < v {
                    let dx = x[u] - x[v];
                    let dy = y[u] - y[v];
                    let dist = (dx * dx + dy * dy + 1e-6).sqrt();
                    // FR attraction ~ dist^2 / k
                    let f = (dist * dist) / k;
                    let fx = dx / dist * f;
                    disp_x[u] -= fx;
                    disp_x[v] += fx;
                }
            }
        }

        // apply displacements with temperature
        for &u in nodes {
            let dx = disp_x[u];
            let adx = dx.abs();
            if adx > 0.0 {
                let step = adx.min(temp);
                x[u] += dx / adx * step;
            }
        }

        temp *= 0.85;
        if temp < 0.01 {
            break;
        }
    }
}

fn main() {
    let g = read_graph();
    let n = g.n;

    let mut x = vec![0f32; n];
    let mut y = vec![0f32; n];

    let mut visited = vec![false; n];
    let mut comp_id = vec![-1i32; n];

    // Globals reused across components
    let mut level = vec![0i32; n];
    let mut pos = vec![0i32; n];

    // For tree layout
    let mut parent = vec![-1i32; n];
    let mut depth = vec![0i32; n];
    let mut left = vec![-1i32; n];
    let mut right = vec![-1i32; n];

    // For refinement
    let mut disp_x = vec![0f32; n];

    // Pack components left-to-right
    let mut pack_x: f32 = 0.0;
    let margin: f32 = 10.0;

    let mut cid: i32 = 0;
    let mut queue: Vec<usize> = Vec::new();

    for s in 0..n {
        if visited[s] {
            continue;
        }
        // BFS for component discovery
        queue.clear();
        let mut head = 0usize;
        visited[s] = true;
        comp_id[s] = cid;
        queue.push(s);

        while head < queue.len() {
            let u = queue[head];
            head += 1;
            for &vv in neighbors(&g, u) {
                let v = vv as usize;
                if !visited[v] {
                    visited[v] = true;
                    comp_id[v] = cid;
                    queue.push(v);
                }
            }
        }

        let nodes: &[usize] = &queue;

        // Pick a root: max degree within component
        let mut root = nodes[0];
        let mut best_deg = g.deg[root];
        for &u in nodes {
            let du = g.deg[u];
            if du > best_deg {
                best_deg = du;
                root = u;
            }
        }

        // Layout
        if nodes.len() == 1 {
            let u = nodes[0];
            x[u] = pack_x;
            y[u] = 0.0;
        } else {
            let m_comp = component_edges_undirected(&g, nodes);
            if m_comp + 1 == nodes.len() {
                layout_tree_component(
                    &g, cid, nodes, &comp_id, root, &mut parent, &mut depth, &mut left, &mut right,
                    &mut x, &mut y,
                );
            } else {
                layout_layered_component(&g, nodes, root, &mut level, &mut pos, &mut x, &mut y);
            }

            // Optional refinement for small components (kept conservative)
            refine_small_component(&g, cid, nodes, &comp_id, &mut x, &mut y, &mut disp_x);

            // Pack component: shift to start at current pack_x, and y>=0
            let mut minx = f32::INFINITY;
            let mut maxx = f32::NEG_INFINITY;
            let mut miny = f32::INFINITY;
            let mut maxy = f32::NEG_INFINITY;
            for &u in nodes {
                minx = minx.min(x[u]);
                maxx = maxx.max(x[u]);
                miny = miny.min(y[u]);
                maxy = maxy.max(y[u]);
            }
            let shift_x = pack_x - minx;
            let shift_y = -miny;

            for &u in nodes {
                x[u] += shift_x;
                y[u] += shift_y;
            }

            let width = (maxx - minx).max(1.0);
            let _height = (maxy - miny).max(1.0);
            pack_x += width + margin;
        }

        cid += 1;
    }

    // Output
    let mut out = io::BufWriter::new(io::stdout());
    for i in 0..n {
        // f64 formatting not required; f32 is fine for layout
        writeln!(&mut out, "{:.6} {:.6}", x[i] as f64, y[i] as f64).unwrap();
    }
}
