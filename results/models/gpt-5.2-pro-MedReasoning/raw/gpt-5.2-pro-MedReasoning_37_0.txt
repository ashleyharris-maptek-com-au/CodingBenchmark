    // ---------------- Fast scanner ----------------
    private sealed class FastScanner
    {
        private readonly Stream _s;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream s) { _s = s; }

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do { c = Read(); if (c == 0) return int.MinValue; } while (c <= 32);
            int sign = 1;
            if (c == (byte)'-') { sign = -1; c = Read(); }
            int x = 0;
            while (c > 32)
            {
                x = x * 10 + (c - (byte)'0');
                c = Read();
            }
            return x * sign;
        }
    }

    // ---------------- Full-graph SCC (Kosaraju, iterative) ----------------
    private static int[] ComputeSccKosaraju(
        int n,
        int[] outStart, int[] outTo,
        int[] inStart, int[] inTo,
        out int compCount)
    {
        var vis = new bool[n];
        var order = new int[n];
        int ordPos = 0;

        // iterative DFS: stack of (v, nextEdgeIndex)
        var stV = new int[n];
        var stE = new int[n];
        int sp;

        for (int v = 0; v < n; v++)
        {
            if (vis[v]) continue;
            sp = 0;
            stV[sp] = v;
            stE[sp] = outStart[v];
            vis[v] = true;

            while (sp >= 0)
            {
                int x = stV[sp];
                int ei = stE[sp];
                int end = outStart[x + 1];

                if (ei < end)
                {
                    int u = outTo[ei];
                    stE[sp] = ei + 1;
                    if (!vis[u])
                    {
                        vis[u] = true;
                        sp++;
                        stV[sp] = u;
                        stE[sp] = outStart[u];
                    }
                }
                else
                {
                    order[ordPos++] = x;
                    sp--;
                }
            }
        }

        var comp = new int[n];
        Array.Fill(comp, -1);
        compCount = 0;

        // 2nd pass on reversed graph
        var stack = new int[n];
        for (int i = n - 1; i >= 0; i--)
        {
            int v = order[i];
            if (comp[v] != -1) continue;

            int top = 0;
            stack[top++] = v;
            comp[v] = compCount;

            while (top > 0)
            {
                int x = stack[--top];
                for (int ei = inStart[x]; ei < inStart[x + 1]; ei++)
                {
                    int u = inTo[ei];
                    if (comp[u] == -1)
                    {
                        comp[u] = compCount;
                        stack[top++] = u;
                    }
                }
            }

            compCount++;
        }

        return comp;
    }

    // ---------------- Small SCC exact solver (<= 64 vertices, bitmask) ----------------
    private readonly struct SmallRes
    {
        public readonly int Size;
        public readonly ulong RemovedMask;
        public SmallRes(int size, ulong rm) { Size = size; RemovedMask = rm; }
    }

    private sealed class SmallExactFvsSolver
    {
        private readonly int _n; // <= 64
        private readonly ulong[] _adj;
        private readonly ulong[] _radj;

        private readonly Dictionary<ulong, SmallRes> _memo = new Dictionary<ulong, SmallRes>(1 << 16);

        // temp arrays reused for FindCycle
        private readonly byte[] _color;
        private readonly int[] _parent;
        private readonly List<int> _cycle = new List<int>(64);

        public SmallExactFvsSolver(ulong[] adj, ulong[] radj)
        {
            _n = adj.Length;
            _adj = adj;
            _radj = radj;
            _color = new byte[_n];
            _parent = new int[_n];
        }

        private static int PopCount(ulong x) => (int)BitOperations.PopCount(x);

        private ulong Peel(ulong mask)
        {
            bool changed;
            do
            {
                changed = false;
                ulong m = mask;
                while (m != 0)
                {
                    int v = BitOperations.TrailingZeroCount(m);
                    ulong vb = 1UL << v;
                    m &= m - 1;

                    if (((_adj[v] & mask) == 0) || ((_radj[v] & mask) == 0))
                    {
                        mask &= ~vb;
                        changed = true;
                    }
                }
            } while (changed);

            return mask;
        }

        private bool IsTrivialAcyclicSingleton(ulong compMask)
        {
            if ((compMask & (compMask - 1)) != 0) return false; // size > 1
            int v = BitOperations.TrailingZeroCount(compMask);
            return (_adj[v] & compMask & (1UL << v)) == 0;
        }

        private void Dfs1(int v, ulong mask, ref ulong vis, int[] order, ref int pos)
        {
            vis |= 1UL << v;
            ulong neigh = _adj[v] & mask;
            while (neigh != 0)
            {
                int u = BitOperations.TrailingZeroCount(neigh);
                neigh &= neigh - 1;
                if (((vis >> u) & 1UL) == 0) Dfs1(u, mask, ref vis, order, ref pos);
            }
            order[pos++] = v;
        }

        private void Dfs2(int v, ulong mask, ref ulong vis, ref ulong comp)
        {
            vis |= 1UL << v;
            comp |= 1UL << v;
            ulong neigh = _radj[v] & mask;
            while (neigh != 0)
            {
                int u = BitOperations.TrailingZeroCount(neigh);
                neigh &= neigh - 1;
                if (((vis >> u) & 1UL) == 0) Dfs2(u, mask, ref vis, ref comp);
            }
        }

        private List<ulong> GetSccs(ulong mask)
        {
            // Kosaraju with bitmasks (n <= 64)
            ulong vis = 0;
            int[] order = new int[_n];
            int pos = 0;

            ulong m = mask;
            while (m != 0)
            {
                int v = BitOperations.TrailingZeroCount(m);
                m &= m - 1;
                if (((vis >> v) & 1UL) == 0) Dfs1(v, mask, ref vis, order, ref pos);
            }

            var comps = new List<ulong>();
            ulong vis2 = 0;
            for (int i = pos - 1; i >= 0; i--)
            {
                int v = order[i];
                if (((mask >> v) & 1UL) == 0) continue;
                if (((vis2 >> v) & 1UL) != 0) continue;

                ulong comp = 0;
                Dfs2(v, mask, ref vis2, ref comp);
                comps.Add(comp);
            }
            return comps;
        }

        private bool DfsFindCycle(int v, ulong mask)
        {
            _color[v] = 1;
            ulong neigh = _adj[v] & mask;
            while (neigh != 0)
            {
                int u = BitOperations.TrailingZeroCount(neigh);
                neigh &= neigh - 1;

                if (_color[u] == 0)
                {
                    _parent[u] = v;
                    if (DfsFindCycle(u, mask)) return true;
                }
                else if (_color[u] == 1)
                {
                    _cycle.Clear();
                    _cycle.Add(u);
                    int x = v;
                    while (x != u)
                    {
                        _cycle.Add(x);
                        x = _parent[x];
                    }
                    return true;
                }
            }
            _color[v] = 2;
            return false;
        }

        private List<int> FindAnyCycle(ulong mask)
        {
            Array.Fill(_color, (byte)0);
            Array.Fill(_parent, -1);

            ulong m = mask;
            while (m != 0)
            {
                int v = BitOperations.TrailingZeroCount(m);
                m &= m - 1;
                if (_color[v] == 0)
                {
                    _parent[v] = -1;
                    if (DfsFindCycle(v, mask))
                        return _cycle;
                }
            }
            _cycle.Clear();
            return _cycle;
        }

        private SmallRes GreedyUpperBound(ulong mask)
        {
            ulong cur = mask;
            ulong removed = 0;

            while (true)
            {
                cur = Peel(cur);
                if (cur == 0) break;

                var cyc = FindAnyCycle(cur);
                // pick vertex with max (in+out) within cur
                int bestV = cyc[0];
                int bestDeg = -1;
                for (int i = 0; i < cyc.Count; i++)
                {
                    int v = cyc[i];
                    int deg = PopCount(_adj[v] & cur) + PopCount(_radj[v] & cur);
                    if (deg > bestDeg) { bestDeg = deg; bestV = v; }
                }

                removed |= 1UL << bestV;
                cur &= ~(1UL << bestV);
            }

            return new SmallRes(PopCount(removed), removed);
        }

        private int GreedyDisjointCyclePackingLowerBound(ulong mask, int stopAt)
        {
            // Greedily find vertex-disjoint cycles; must remove >=1 vertex per such cycle.
            ulong cur = mask;
            int count = 0;
            while (count < stopAt)
            {
                cur = Peel(cur);
                if (cur == 0) break;
                var cyc = FindAnyCycle(cur);
                count++;
                for (int i = 0; i < cyc.Count; i++)
                    cur &= ~(1UL << cyc[i]);
            }
            return count;
        }

        public SmallRes Solve(ulong mask)
        {
            mask = Peel(mask);
            if (mask == 0) return new SmallRes(0, 0);

            if (_memo.TryGetValue(mask, out var cached))
                return cached;

            // Decompose into SCCs (after peeling, can split a lot)
            var comps = GetSccs(mask);
            if (comps.Count > 1)
            {
                int totalSize = 0;
                ulong totalRemoved = 0;
                foreach (var comp in comps)
                {
                    if (comp == 0) continue;
                    if (IsTrivialAcyclicSingleton(comp)) continue;

                    var r = Solve(comp);
                    totalSize += r.Size;
                    totalRemoved |= r.RemovedMask;
                }
                var res = new SmallRes(totalSize, totalRemoved);
                _memo[mask] = res;
                return res;
            }

            // Single SCC with a cycle (or singleton self-loop)
            SmallRes best = GreedyUpperBound(mask);
            if (best.Size == 0)
            {
                _memo[mask] = best;
                return best;
            }

            int lb = GreedyDisjointCyclePackingLowerBound(mask, best.Size);
            if (lb >= best.Size)
            {
                // Can't beat best; best is optimal (for this bound), accept.
                _memo[mask] = best;
                return best;
            }

            var cycle = FindAnyCycle(mask);
            // branch on removing one vertex from cycle
            cycle.Sort((a, b) =>
            {
                int da = PopCount(_adj[a] & mask) + PopCount(_radj[a] & mask);
                int db = PopCount(_adj[b] & mask) + PopCount(_radj[b] & mask);
                return db.CompareTo(da);
            });

            for (int i = 0; i < cycle.Count; i++)
            {
                int v = cycle[i];
                if (best.Size == 1) break; // can't do better
                ulong nm = mask & ~(1UL << v);

                // quick bound: need at least 1 vertex to hit current cycle
                // plus disjoint packing bound in remainder
                int lbChild = 1 + GreedyDisjointCyclePackingLowerBound(nm, best.Size - 1);
                if (lbChild >= best.Size) continue;

                var child = Solve(nm);
                int candSize = child.Size + 1;
                if (candSize < best.Size)
                {
                    best = new SmallRes(candSize, child.RemovedMask | (1UL << v));
                }
            }

            _memo[mask] = best;
            return best;
        }
    }

    // ---------------- Large SCC exact solver (iterative deepening, adjacency lists) ----------------
    private enum ChangeType : byte { Marker = 0, InDeg = 1, OutDeg = 2 }
    private readonly struct Change
    {
        public readonly int V;
        public readonly ChangeType T;
        public Change(int v, ChangeType t) { V = v; T = t; }
    }

    private sealed class LargeExactFvsSolver
    {
        private readonly int _n;
        private readonly int[][] _out;
        private readonly int[][] _in;

        private readonly bool[] _alive;
        private readonly int[] _indeg;
        private readonly int[] _outdeg;

        private readonly int[] _indeg0;
        private readonly int[] _outdeg0;

        private int _aliveCount;

        private readonly Stack<Change> _changes = new Stack<Change>(1 << 16);
        private readonly List<int> _chosen = new List<int>();

        // for cycle finding
        private readonly byte[] _color;
        private readonly int[] _parent;
        private readonly List<int> _touched = new List<int>(1024);

        public LargeExactFvsSolver(int[][] outAdj, int[][] inAdj, int[] indeg0, int[] outdeg0)
        {
            _out = outAdj;
            _in = inAdj;
            _n = outAdj.Length;

            _alive = new bool[_n];
            _indeg = new int[_n];
            _outdeg = new int[_n];

            _indeg0 = indeg0;
            _outdeg0 = outdeg0;

            _color = new byte[_n];
            _parent = new int[_n];
        }

        private void ResetState()
        {
            Array.Fill(_alive, true);
            Array.Copy(_indeg0, _indeg, _n);
            Array.Copy(_outdeg0, _outdeg, _n);
            _aliveCount = _n;
            _changes.Clear();
            _chosen.Clear();
        }

        private void RemoveVertex(int v)
        {
            if (!_alive[v]) return;
            _alive[v] = false;
            _aliveCount--;
            _changes.Push(new Change(v, ChangeType.Marker));

            // maintain counts for ALL vertices (alive or not), so restoration stays consistent
            var o = _out[v];
            for (int i = 0; i < o.Length; i++)
            {
                int u = o[i];
                _indeg[u]--;
                _changes.Push(new Change(u, ChangeType.InDeg));
            }

            var inn = _in[v];
            for (int i = 0; i < inn.Length; i++)
            {
                int u = inn[i];
                _outdeg[u]--;
                _changes.Push(new Change(u, ChangeType.OutDeg));
            }
        }

        private void UndoTo(int snap)
        {
            while (_changes.Count > snap)
            {
                var c = _changes.Pop();
                switch (c.T)
                {
                    case ChangeType.InDeg: _indeg[c.V]++; break;
                    case ChangeType.OutDeg: _outdeg[c.V]++; break;
                    case ChangeType.Marker:
                        _alive[c.V] = true;
                        _aliveCount++;
                        break;
                }
            }
        }

        private void Peel()
        {
            // Local stack (no need to restore contents on undo)
            var st = new Stack<int>();

            for (int v = 0; v < _n; v++)
                if (_alive[v] && (_indeg[v] == 0 || _outdeg[v] == 0))
                    st.Push(v);

            while (st.Count > 0)
            {
                int v = st.Pop();
                if (!_alive[v]) continue;
                if (_indeg[v] != 0 && _outdeg[v] != 0) continue;

                RemoveVertex(v);

                var o = _out[v];
                for (int i = 0; i < o.Length; i++)
                {
                    int u = o[i];
                    if (_alive[u] && (_indeg[u] == 0 || _outdeg[u] == 0)) st.Push(u);
                }
                var inn = _in[v];
                for (int i = 0; i < inn.Length; i++)
                {
                    int u = inn[i];
                    if (_alive[u] && (_indeg[u] == 0 || _outdeg[u] == 0)) st.Push(u);
                }
            }
        }

        private List<int> FindAnyCycle()
        {
            _touched.Clear();
            Array.Fill(_color, (byte)0);
            Array.Fill(_parent, -1);

            // iterative DFS stack arrays (to reduce allocations)
            var stV = new int[_n];
            var stI = new int[_n];
            int sp;

            for (int start = 0; start < _n; start++)
            {
                if (!_alive[start] || _color[start] != 0) continue;

                sp = 0;
                stV[sp] = start;
                stI[sp] = 0;
                _color[start] = 1;
                _touched.Add(start);
                _parent[start] = -1;

                while (sp >= 0)
                {
                    int v = stV[sp];
                    int idx = stI[sp];

                    var o = _out[v];
                    if (idx >= o.Length)
                    {
                        _color[v] = 2;
                        sp--;
                        continue;
                    }

                    int u = o[idx];
                    stI[sp] = idx + 1;

                    if (!_alive[u]) continue;

                    if (_color[u] == 0)
                    {
                        _parent[u] = v;
                        _color[u] = 1;
                        _touched.Add(u);
                        sp++;
                        stV[sp] = u;
                        stI[sp] = 0;
                    }
                    else if (_color[u] == 1)
                    {
                        // found back-edge v -> u, build cycle
                        var cyc = new List<int>();
                        cyc.Add(u);
                        int x = v;
                        while (x != u)
                        {
                            cyc.Add(x);
                            x = _parent[x];
                        }
                        return cyc;
                    }
                }
            }

            return new List<int>(); // no cycle
        }

        // Greedy upper bound for IDDFS (not required for correctness, but helps performance)
        private List<int> GreedyUpperBound()
        {
            var alive = new bool[_n];
            Array.Fill(alive, true);

            var indeg = (int[])_indeg0.Clone();
            var outdeg = (int[])_outdeg0.Clone();
            int aliveCount = _n;

            void RemoveTmp(int v)
            {
                if (!alive[v]) return;
                alive[v] = false;
                aliveCount--;

                var o = _out[v];
                for (int i = 0; i < o.Length; i++) indeg[o[i]]--;
                var inn = _in[v];
                for (int i = 0; i < inn.Length; i++) outdeg[inn[i]]--;
            }

            void PeelTmp()
            {
                var st = new Stack<int>();
                for (int v = 0; v < _n; v++)
                    if (alive[v] && (indeg[v] == 0 || outdeg[v] == 0))
                        st.Push(v);

                while (st.Count > 0)
                {
                    int v = st.Pop();
                    if (!alive[v]) continue;
                    if (indeg[v] != 0 && outdeg[v] != 0) continue;

                    RemoveTmp(v);

                    var o = _out[v];
                    for (int i = 0; i < o.Length; i++)
                    {
                        int u = o[i];
                        if (alive[u] && (indeg[u] == 0 || outdeg[u] == 0)) st.Push(u);
                    }
                    var inn = _in[v];
                    for (int i = 0; i < inn.Length; i++)
                    {
                        int u = inn[i];
                        if (alive[u] && (indeg[u] == 0 || outdeg[u] == 0)) st.Push(u);
                    }
                }
            }

            List<int> FindCycleTmp()
            {
                var color = new byte[_n];
                var parent = new int[_n];
                Array.Fill(parent, -1);

                var stV = new int[_n];
                var stI = new int[_n];

                for (int start = 0; start < _n; start++)
                {
                    if (!alive[start] || color[start] != 0) continue;

                    int sp = 0;
                    stV[sp] = start;
                    stI[sp] = 0;
                    color[start] = 1;

                    while (sp >= 0)
                    {
                        int v = stV[sp];
                        int idx = stI[sp];

                        var o = _out[v];
                        if (idx >= o.Length)
                        {
                            color[v] = 2;
                            sp--;
                            continue;
                        }

                        int u = o[idx];
                        stI[sp] = idx + 1;
                        if (!alive[u]) continue;

                        if (color[u] == 0)
                        {
                            parent[u] = v;
                            color[u] = 1;
                            sp++;
                            stV[sp] = u;
                            stI[sp] = 0;
                        }
                        else if (color[u] == 1)
                        {
                            var cyc = new List<int>();
                            cyc.Add(u);
                            int x = v;
                            while (x != u)
                            {
                                cyc.Add(x);
                                x = parent[x];
                            }
                            return cyc;
                        }
                    }
                }

                return new List<int>();
            }

            var removed = new List<int>();
            while (true)
            {
                PeelTmp();
                if (aliveCount == 0) break;

                var cyc = FindCycleTmp();
                if (cyc.Count == 0) break; // acyclic

                int bestV = cyc[0];
                int bestDeg = -1;
                for (int i = 0; i < cyc.Count; i++)
                {
                    int v = cyc[i];
                    int deg = indeg[v] + outdeg[v];
                    if (deg > bestDeg) { bestDeg = deg; bestV = v; }
                }

                removed.Add(bestV);
                RemoveTmp(bestV);
            }

            return removed;
        }

        private int GreedyDisjointCyclePackingLowerBound()
        {
            var alive = new bool[_n];
            Array.Fill(alive, true);

            var indeg = (int[])_indeg0.Clone();
            var outdeg = (int[])_outdeg0.Clone();
            int aliveCount = _n;

            void RemoveTmp(int v)
            {
                if (!alive[v]) return;
                alive[v] = false;
                aliveCount--;

                var o = _out[v];
                for (int i = 0; i < o.Length; i++) indeg[o[i]]--;
                var inn = _in[v];
                for (int i = 0; i < inn.Length; i++) outdeg[inn[i]]--;
            }

            void PeelTmp()
            {
                var st = new Stack<int>();
                for (int v = 0; v < _n; v++)
                    if (alive[v] && (indeg[v] == 0 || outdeg[v] == 0))
                        st.Push(v);

                while (st.Count > 0)
                {
                    int v = st.Pop();
                    if (!alive[v]) continue;
                    if (indeg[v] != 0 && outdeg[v] != 0) continue;

                    RemoveTmp(v);

                    var o = _out[v];
                    for (int i = 0; i < o.Length; i++)
                    {
                        int u = o[i];
                        if (alive[u] && (indeg[u] == 0 || outdeg[u] == 0)) st.Push(u);
                    }
                    var inn = _in[v];
                    for (int i = 0; i < inn.Length; i++)
                    {
                        int u = inn[i];
                        if (alive[u] && (indeg[u] == 0 || outdeg[u] == 0)) st.Push(u);
                    }
                }
            }

            List<int> FindCycleTmp()
            {
                var color = new byte[_n];
                var parent = new int[_n];
                Array.Fill(parent, -1);

                var stV = new int[_n];
                var stI = new int[_n];

                for (int start = 0; start < _n; start++)
                {
                    if (!alive[start] || color[start] != 0) continue;

                    int sp = 0;
                    stV[sp] = start;
                    stI[sp] = 0;
                    color[start] = 1;

                    while (sp >= 0)
                    {
                        int v = stV[sp];
                        int idx = stI[sp];

                        var o = _out[v];
                        if (idx >= o.Length)
                        {
                            color[v] = 2;
                            sp--;
                            continue;
                        }

                        int u = o[idx];
                        stI[sp] = idx + 1;
                        if (!alive[u]) continue;

                        if (color[u] == 0)
                        {
                            parent[u] = v;
                            color[u] = 1;
                            sp++;
                            stV[sp] = u;
                            stI[sp] = 0;
                        }
                        else if (color[u] == 1)
                        {
                            var cyc = new List<int>();
                            cyc.Add(u);
                            int x = v;
                            while (x != u)
                            {
                                cyc.Add(x);
                                x = parent[x];
                            }
                            return cyc;
                        }
                    }
                }

                return new List<int>();
            }

            int lb = 0;
            while (true)
            {
                PeelTmp();
                if (aliveCount == 0) break;

                var cyc = FindCycleTmp();
                if (cyc.Count == 0) break;
                lb++;

                // remove entire cycle to keep cycles vertex-disjoint
                for (int i = 0; i < cyc.Count; i++) RemoveTmp(cyc[i]);
            }

            return lb;
        }

        private bool Search(int kRemaining)
        {
            int snap = _changes.Count;
            int chosenSnap = _chosen.Count;

            Peel();
            if (_aliveCount == 0) return true;

            if (kRemaining == 0)
            {
                UndoTo(snap);
                if (_chosen.Count > chosenSnap) _chosen.RemoveRange(chosenSnap, _chosen.Count - chosenSnap);
                return false;
            }

            var cyc = FindAnyCycle();
            if (cyc.Count == 0)
            {
                // should not happen after peeling, but safe
                return true;
            }

            cyc.Sort((a, b) => (_indeg[b] + _outdeg[b]).CompareTo(_indeg[a] + _outdeg[a]));

            for (int i = 0; i < cyc.Count; i++)
            {
                int v = cyc[i];
                if (!_alive[v]) continue;

                int snap2 = _changes.Count;
                int chosenSnap2 = _chosen.Count;

                _chosen.Add(v);
                RemoveVertex(v);

                if (Search(kRemaining - 1)) return true;

                UndoTo(snap2);
                if (_chosen.Count > chosenSnap2) _chosen.RemoveRange(chosenSnap2, _chosen.Count - chosenSnap2);
            }

            UndoTo(snap);
            if (_chosen.Count > chosenSnap) _chosen.RemoveRange(chosenSnap, _chosen.Count - chosenSnap);
            return false;
        }

        public List<int> SolveExactMinimum()
        {
            // Upper and lower bounds for iterative deepening
            var greedy = GreedyUpperBound();
            int ub = greedy.Count;
            int lb = GreedyDisjointCyclePackingLowerBound();
            if (ub == 0) return greedy; // acyclic

            for (int k = lb; k <= ub; k++)
            {
                ResetState();
                if (Search(k))
                    return new List<int>(_chosen);
            }

            // Should never happen (k=ub should be feasible), but keep a safe fallback:
            return greedy;
        }
    }

    // ---------------- Main ----------------
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        if (n == int.MinValue)
        {
            Console.WriteLine("0\n");
            return;
        }
        int m = fs.NextInt();

        if (n <= 0)
        {
            Console.WriteLine("0\n");
            return;
        }

        int[] U = new int[m];
        int[] V = new int[m];

        var outDeg = new int[n];
        var inDeg = new int[n];

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            U[i] = u; V[i] = v;
            outDeg[u]++;
            inDeg[v]++;
        }

        // Build CSR for out and in
        int[] outStart = new int[n + 1];
        int[] inStart = new int[n + 1];
        for (int i = 0; i < n; i++)
        {
            outStart[i + 1] = outStart[i] + outDeg[i];
            inStart[i + 1] = inStart[i] + inDeg[i];
        }

        int[] outTo = new int[m];
        int[] inTo = new int[m];
        int[] curOut = (int[])outStart.Clone();
        int[] curIn = (int[])inStart.Clone();

        for (int i = 0; i < m; i++)
        {
            int u = U[i], v = V[i];
            outTo[curOut[u]++] = v;
            inTo[curIn[v]++] = u;
        }

        // SCC decomposition
        int compCount;
        int[] comp = ComputeSccKosaraju(n, outStart, outTo, inStart, inTo, out compCount);

        // Build vertex lists per component
        int[] compSize = new int[compCount];
        for (int v = 0; v < n; v++) compSize[comp[v]]++;

        int[] compStart = new int[compCount + 1];
        for (int c = 0; c < compCount; c++) compStart[c + 1] = compStart[c] + compSize[c];

        int[] compFill = (int[])compStart.Clone();
        int[] compVerts = new int[n];
        for (int v = 0; v < n; v++)
        {
            int c = comp[v];
            compVerts[compFill[c]++] = v;
        }

        // Helpers for building local graphs
        int[] mark = new int[n];
        int[] localIndex = new int[n];
        int token = 1;

        var answer = new List<int>();

        for (int c = 0; c < compCount; c++)
        {
            int sz = compSize[c];
            if (sz == 0) continue;

            int start = compStart[c];
            // Check if this SCC is cyclic
            bool cyclic;
            if (sz == 1)
            {
                int v = compVerts[start];
                cyclic = false;
                for (int ei = outStart[v]; ei < outStart[v + 1]; ei++)
                {
                    if (outTo[ei] == v) { cyclic = true; break; }
                }
            }
            else cyclic = true;

            if (!cyclic) continue;

            if (sz <= 64)
            {
                // Build bitmask graph for SCC
                if (token == int.MaxValue) { Array.Clear(mark); token = 1; } // extremely unlikely
                int t = token++;
                for (int i = 0; i < sz; i++)
                {
                    int gv = compVerts[start + i];
                    mark[gv] = t;
                    localIndex[gv] = i;
                }

                var adj = new ulong[sz];
                var radj = new ulong[sz];

                for (int i = 0; i < sz; i++)
                {
                    int gv = compVerts[start + i];
                    for (int ei = outStart[gv]; ei < outStart[gv + 1]; ei++)
                    {
                        int gto = outTo[ei];
                        if (mark[gto] == t)
                        {
                            int j = localIndex[gto];
                            adj[i] |= 1UL << j;
                            radj[j] |= 1UL << i;
                        }
                    }
                }

                ulong fullMask = (sz == 64) ? ulong.MaxValue : ((1UL << sz) - 1UL);
                var solver = new SmallExactFvsSolver(adj, radj);
                var res = solver.Solve(fullMask);

                ulong rm = res.RemovedMask;
                while (rm != 0)
                {
                    int li = BitOperations.TrailingZeroCount(rm);
                    rm &= rm - 1;
                    answer.Add(compVerts[start + li]);
                }
            }
            else
            {
                // Build adjacency lists (exact via iterative deepening)
                if (token == int.MaxValue) { Array.Clear(mark); token = 1; }
                int t = token++;
                for (int i = 0; i < sz; i++)
                {
                    int gv = compVerts[start + i];
                    mark[gv] = t;
                    localIndex[gv] = i;
                }

                int[] outCnt = new int[sz];
                int[] inCnt = new int[sz];

                // count edges within SCC
                for (int i = 0; i < sz; i++)
                {
                    int gv = compVerts[start + i];
                    for (int ei = outStart[gv]; ei < outStart[gv + 1]; ei++)
                    {
                        int gto = outTo[ei];
                        if (mark[gto] == t)
                        {
                            int j = localIndex[gto];
                            outCnt[i]++;
                            inCnt[j]++;
                        }
                    }
                }

                var outAdj = new int[sz][];
                var inAdj = new int[sz][];
                for (int i = 0; i < sz; i++)
                {
                    outAdj[i] = new int[outCnt[i]];
                    inAdj[i] = new int[inCnt[i]];
                    outCnt[i] = 0; // reuse as fill pointers
                    inCnt[i] = 0;
                }

                // fill
                for (int i = 0; i < sz; i++)
                {
                    int gv = compVerts[start + i];
                    for (int ei = outStart[gv]; ei < outStart[gv + 1]; ei++)
                    {
                        int gto = outTo[ei];
                        if (mark[gto] == t)
                        {
                            int j = localIndex[gto];
                            outAdj[i][outCnt[i]++] = j;
                            inAdj[j][inCnt[j]++] = i;
                        }
                    }
                }

                // initial degrees inside SCC
                var indeg0 = new int[sz];
                var outdeg0 = new int[sz];
                for (int i = 0; i < sz; i++)
                {
                    outdeg0[i] = outAdj[i].Length;
                    indeg0[i] = inAdj[i].Length;
                }

                var solver = new LargeExactFvsSolver(outAdj, inAdj, indeg0, outdeg0);
                var removedLocal = solver.SolveExactMinimum();

                for (int i = 0; i < removedLocal.Count; i++)
                    answer.Add(compVerts[start + removedLocal[i]]);
            }
        }

        // Output
        using var sw = new StreamWriter(Console.OpenStandardOutput());
        sw.WriteLine(answer.Count);
        if (answer.Count == 0) sw.WriteLine();
        else sw.WriteLine(string.Join(' ', answer));
    }
