```spirv
; SPIR-V
; Version: 1.0
; Generator: custom
; Bound: 80
OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_in 0 Offset 0

OpDecorate %arr_pv_in ArrayStride 16

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 Offset 0

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void

%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_h = OpConstant %float 0.04

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%arr_v3_in = OpTypeArray %v3float %uint_1

%ptr_Input_arr_pv_in = OpTypePointer Input %arr_pv_in
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_Input_v3float = OpTypePointer Input %v3float

%ptr_Output_pv_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_Input_arr_pv_in Input
%worldPos = OpVariable %ptr_Input_arr_v3 Input
%normal = OpVariable %ptr_Input_arr_v3 Input
%color = OpVariable %ptr_Input_arr_v3 Input

%gl_out = OpVariable %ptr_Output_pv_out Output
%outColor = OpVariable %ptr_Output_v3float Output

%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load input position: gl_in[0].gl_Position
%in_pos_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%in_pos = OpLoad %v4float %in_pos_ptr
%x = OpCompositeExtract %float %in_pos 0
%y = OpCompositeExtract %float %in_pos 1

; Load input color: color[0]
%in_col_ptr = OpAccessChain %ptr_Input_v3float %color %int_0
%in_col = OpLoad %v3float %in_col_ptr

; Precompute x +/- h, y +/- h
%xmh = OpFSub %float %x %float_h
%xph = OpFAdd %float %x %float_h
%ymh = OpFSub %float %y %float_h
%yph = OpFAdd %float %y %float_h

; Output position pointer: gl_out.gl_Position
%out_pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0

; Vertex 0: (x-h, y-h)
%p0 = OpCompositeConstruct %v4float %xmh %ymh %float_0 %float_1
OpStore %out_pos_ptr %p0
OpStore %outColor %in_col
OpEmitVertex

; Vertex 1: (x+h, y-h)
%p1 = OpCompositeConstruct %v4float %xph %ymh %float_0 %float_1
OpStore %out_pos_ptr %p1
OpStore %outColor %in_col
OpEmitVertex

; Vertex 2: (x-h, y+h)
%p2 = OpCompositeConstruct %v4float %xmh %yph %float_0 %float_1
OpStore %out_pos_ptr %p2
OpStore %outColor %in_col
OpEmitVertex

; Vertex 3: (x+h, y+h)
%p3 = OpCompositeConstruct %v4float %xph %yph %float_0 %float_1
OpStore %out_pos_ptr %p3
OpStore %outColor %in_col
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
