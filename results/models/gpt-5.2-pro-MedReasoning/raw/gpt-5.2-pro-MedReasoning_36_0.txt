using namespace std;

struct HopcroftKarp {
    int L = 0, R = 0;
    vector<vector<int>> g;      // L -> [0..R)
    vector<int> dist;
    vector<int> matchL, matchR; // matchL[u] in [0..R) or -1, matchR[v] in [0..L) or -1

    HopcroftKarp() = default;
    HopcroftKarp(int l, int r) { init(l, r); }

    void init(int l, int r) {
        L = l; R = r;
        g.assign(L, {});
        dist.assign(L, 0);
        matchL.assign(L, -1);
        matchR.assign(R, -1);
    }

    void addEdge(int uL, int vR) { g[uL].push_back(vR); }

    bool bfs() {
        queue<int> q;
        for (int u = 0; u < L; ++u) {
            if (matchL[u] == -1) {
                dist[u] = 0;
                q.push(u);
            } else {
                dist[u] = -1;
            }
        }
        bool foundFreeOnRight = false;
        while (!q.empty()) {
            int u = q.front(); q.pop();
            for (int v : g[u]) {
                int u2 = matchR[v];
                if (u2 == -1) {
                    foundFreeOnRight = true;
                } else if (dist[u2] == -1) {
                    dist[u2] = dist[u] + 1;
                    q.push(u2);
                }
            }
        }
        return foundFreeOnRight;
    }

    bool dfs(int u) {
        for (int v : g[u]) {
            int u2 = matchR[v];
            if (u2 == -1 || (dist[u2] == dist[u] + 1 && dfs(u2))) {
                matchL[u] = v;
                matchR[v] = u;
                return true;
            }
        }
        dist[u] = -1;
        return false;
    }

    int maxMatching() {
        int matching = 0;
        while (bfs()) {
            for (int u = 0; u < L; ++u)
                if (matchL[u] == -1 && dfs(u))
                    ++matching;
        }
        return matching;
    }

    // After maxMatching(), compute min vertex cover via Konig:
    // Z = vertices reachable from free left vertices by alternating paths.
    // MVC = (L \ Z_L) union (R intersect Z_R)
    // MIS = V \ MVC.
    void minVertexCover(vector<char>& inCoverL, vector<char>& inCoverR) {
        vector<char> visL(L, 0), visR(R, 0);
        queue<int> q;
        for (int u = 0; u < L; ++u) {
            if (matchL[u] == -1) {
                visL[u] = 1;
                q.push(u);
            }
        }
        while (!q.empty()) {
            int u = q.front(); q.pop();
            // from left, traverse UNMATCHED edges to right
            for (int v : g[u]) {
                if (matchL[u] == v) continue; // matched edge, not allowed in this direction
                if (!visR[v]) {
                    visR[v] = 1;
                    int u2 = matchR[v];
                    if (u2 != -1 && !visL[u2]) {
                        visL[u2] = 1;
                        q.push(u2);
                    }
                }
            }
        }
        inCoverL.assign(L, 0);
        inCoverR.assign(R, 0);
        for (int u = 0; u < L; ++u) if (!visL[u]) inCoverL[u] = 1;
        for (int v = 0; v < R; ++v) if (visR[v])  inCoverR[v] = 1;
    }
};

struct MaxCliqueTomita {
    int n = 0;
    int W = 0;
    vector<vector<uint64_t>> adj; // adjacency bitsets in the graph for which we seek maximum clique
    vector<int> best;

    explicit MaxCliqueTomita(vector<vector<uint64_t>> a) : n((int)a.size()), adj(std::move(a)) {
        W = (n + 63) / 64;
        best.clear();
    }

    inline bool isEdge(int u, int v) const {
        return (adj[u][v >> 6] >> (v & 63)) & 1ULL;
    }

    inline void setBit(vector<uint64_t>& bs, int v) const {
        bs[v >> 6] |= (1ULL << (v & 63));
    }

    inline bool intersects(int v, const vector<uint64_t>& bs) const {
        const auto& av = adj[v];
        for (int i = 0; i < W; ++i) {
            if (av[i] & bs[i]) return true;
        }
        return false;
    }

    // Greedy coloring for an upper bound.
    // Produces an ordering "order" and color numbers "colors" (nondecreasing),
    // such that for prefix [0..i], clique size <= colors[i].
    void colorSort(const vector<int>& cand, vector<int>& order, vector<int>& colors) const {
        order.clear();
        colors.clear();
        order.reserve(cand.size());
        colors.reserve(cand.size());

        vector<int> remaining = cand;
        vector<uint64_t> used(W);

        int color = 0;
        while (!remaining.empty()) {
            ++color;
            fill(used.begin(), used.end(), 0ULL);

            vector<int> next;
            next.reserve(remaining.size());

            for (int v : remaining) {
                if (!intersects(v, used)) {
                    setBit(used, v);
                    order.push_back(v);
                    colors.push_back(color);
                } else {
                    next.push_back(v);
                }
            }
            remaining.swap(next);
        }
    }

    void expand(vector<int>& clique, const vector<int>& cand) {
        if (cand.empty()) {
            if (clique.size() > best.size()) best = clique;
            return;
        }

        // Quick optimistic bound:
        if (clique.size() + cand.size() <= best.size()) return;

        vector<int> order, colors;
        colorSort(cand, order, colors);

        for (int idx = (int)order.size() - 1; idx >= 0; --idx) {
            if (clique.size() + (size_t)colors[idx] <= best.size()) return;

            int v = order[idx];

            // new candidates are those earlier in order that connect to v
            vector<int> newCand;
            newCand.reserve(idx);
            for (int j = 0; j < idx; ++j) {
                int u = order[j];
                if (isEdge(v, u)) newCand.push_back(u);
            }

            clique.push_back(v);
            if (newCand.empty()) {
                if (clique.size() > best.size()) best = clique;
            } else {
                expand(clique, newCand);
            }
            clique.pop_back();
        }
    }

    vector<int> maximumClique(vector<int> cand) {
        // Heuristic: sort candidates by descending degree to help early pruning
        vector<int> deg(n, 0);
        for (int v = 0; v < n; ++v) {
            int d = 0;
            for (int i = 0; i < W; ++i) d += (int)std::popcount(adj[v][i]);
            deg[v] = d;
        }
        sort(cand.begin(), cand.end(), [&](int a, int b) {
            if (deg[a] != deg[b]) return deg[a] > deg[b];
            return a < b;
        });

        best.clear();
        vector<int> clique;
        clique.reserve(n);
        expand(clique, cand);
        return best;
    }
};

static vector<int> solveTreeMIS(const vector<vector<int>>& ladj, const vector<int>& localToGlobal) {
    int n = (int)ladj.size();
    vector<int> parent(n, -1), order;
    order.reserve(n);

    // iterative DFS to get parent and order
    vector<int> st;
    st.push_back(0);
    parent[0] = 0;
    while (!st.empty()) {
        int v = st.back(); st.pop_back();
        order.push_back(v);
        for (int to : ladj[v]) if (parent[to] == -1) {
            parent[to] = v;
            st.push_back(to);
        }
    }

    vector<int> post = order;
    reverse(post.begin(), post.end());

    vector<int> dp0(n, 0), dp1(n, 0);
    for (int v : post) {
        dp1[v] = 1;
        dp0[v] = 0;
        for (int to : ladj[v]) if (to != parent[v]) {
            dp1[v] += dp0[to];
            dp0[v] += max(dp0[to], dp1[to]);
        }
    }

    vector<char> take(n, 0);
    // reconstruct
    vector<int> stack2 = {0};
    take[0] = (dp1[0] > dp0[0]);
    while (!stack2.empty()) {
        int v = stack2.back(); stack2.pop_back();
        for (int to : ladj[v]) if (to != parent[v]) {
            if (take[v]) take[to] = 0;
            else take[to] = (dp1[to] > dp0[to]);
            stack2.push_back(to);
        }
    }

    vector<int> res;
    for (int i = 0; i < n; ++i) if (take[i]) res.push_back(localToGlobal[i]);
    return res;
}

static vector<int> solveComponentMIS_Exact(const vector<int>& compVerts,
                                          const vector<vector<int>>& g,
                                          const vector<char>& hasLoop) {
    // If any self-loop exists, that vertex cannot be in an independent set (safe choice).
    vector<int> verts;
    verts.reserve(compVerts.size());
    for (int v : compVerts) if (!hasLoop[v]) verts.push_back(v);
    if (verts.empty()) return {};

    int s = (int)verts.size();
    if (s == 1) return {verts[0]};

    vector<int> localIndex;
    // map global -> local only for this component (use temporary array sized by n would be too large here?),
    // but n can be big; still, a vector<int>(n,-1) is acceptable memory-wise for typical constraints.
    // We'll build one lazily in main and pass; but here keep simple by using unordered_map-like technique:
    // (We cannot efficiently map without a global array; do it here using a global-sized static is not available.)
    // So we do: create array of pairs then sort and binary-search.
    // However, to keep it fast, we accept O(s log s) mapping using sorting.
    vector<pair<int,int>> pairs;
    pairs.reserve(s);
    for (int i = 0; i < s; ++i) pairs.push_back({verts[i], i});
    sort(pairs.begin(), pairs.end());
    auto getLocal = [&](int gv) -> int {
        auto it = lower_bound(pairs.begin(), pairs.end(), pair<int,int>{gv, -1});
        if (it != pairs.end() && it->first == gv) return it->second;
        return -1;
    };

    vector<vector<int>> ladj(s);
    long long edgeCount2 = 0;
    for (int i = 0; i < s; ++i) {
        int u = verts[i];
        for (int v : g[u]) {
            int j = getLocal(v);
            if (j != -1) ladj[i].push_back(j);
        }
        sort(ladj[i].begin(), ladj[i].end());
        ladj[i].erase(unique(ladj[i].begin(), ladj[i].end()), ladj[i].end());
        edgeCount2 += (long long)ladj[i].size();
    }
    long long m = edgeCount2 / 2;

    // Tree case (connected component in original graph => tree iff m == s-1)
    if (m == s - 1) {
        return solveTreeMIS(ladj, verts);
    }

    // Check bipartite and if so solve exactly via Konig (MIS = V - MVC)
    vector<int> color(s, -1);
    bool bip = true;
    queue<int> q;
    color[0] = 0;
    q.push(0);
    while (!q.empty() && bip) {
        int v = q.front(); q.pop();
        for (int to : ladj[v]) {
            if (color[to] == -1) {
                color[to] = color[v] ^ 1;
                q.push(to);
            } else if (color[to] == color[v]) {
                bip = false;
                break;
            }
        }
    }
    if (bip) {
        // Build partitions and Hopcroft-Karp
        vector<int> mapL(s, -1), mapR(s, -1), invL, invR;
        for (int i = 0; i < s; ++i) {
            if (color[i] == 0) {
                mapL[i] = (int)invL.size();
                invL.push_back(i);
            } else {
                mapR[i] = (int)invR.size();
                invR.push_back(i);
            }
        }
        HopcroftKarp hk((int)invL.size(), (int)invR.size());
        for (int i = 0; i < s; ++i) if (color[i] == 0) {
            int uL = mapL[i];
            for (int to : ladj[i]) {
                if (color[to] == 1) hk.addEdge(uL, mapR[to]);
            }
        }
        hk.maxMatching();
        vector<char> inCoverL, inCoverR;
        hk.minVertexCover(inCoverL, inCoverR);

        // MIS = V \ MVC
        vector<int> res;
        res.reserve(s);
        for (int uL = 0; uL < hk.L; ++uL) {
            if (!inCoverL[uL]) res.push_back(verts[invL[uL]]);
        }
        for (int vR = 0; vR < hk.R; ++vR) {
            if (!inCoverR[vR]) res.push_back(verts[invR[vR]]);
        }
        return res;
    }

    // General case: solve MIS by maximum clique in complement graph on this component:
    // MIS(G) == omega(complement(G))
    int W = (s + 63) / 64;
    vector<uint64_t> full(W, ~0ULL);
    if (s % 64) full[W - 1] = (1ULL << (s % 64)) - 1ULL;

    vector<vector<uint64_t>> origAdjBS(s, vector<uint64_t>(W, 0ULL));
    for (int i = 0; i < s; ++i) {
        for (int to : ladj[i]) {
            origAdjBS[i][to >> 6] |= (1ULL << (to & 63));
        }
    }

    vector<vector<uint64_t>> compAdjBS(s, vector<uint64_t>(W, 0ULL));
    for (int i = 0; i < s; ++i) {
        for (int w = 0; w < W; ++w) {
            compAdjBS[i][w] = full[w] & ~origAdjBS[i][w];
        }
        // remove self-loop in complement representation
        compAdjBS[i][i >> 6] &= ~(1ULL << (i & 63));
    }

    MaxCliqueTomita mc(std::move(compAdjBS));
    vector<int> cand(s);
    iota(cand.begin(), cand.end(), 0);
    vector<int> cliqueLocal = mc.maximumClique(std::move(cand));

    vector<int> res;
    res.reserve(cliqueLocal.size());
    for (int li : cliqueLocal) res.push_back(verts[li]);
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    int m;
    if (!(cin >> n >> m)) {
        return 0;
    }

    vector<vector<int>> g(n);
    vector<char> hasLoop(n, 0);

    g.reserve(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) {
            hasLoop[u] = 1;
            continue;
        }
        g[u].push_back(v);
        g[v].push_back(u);
    }
    for (int i = 0; i < n; ++i) {
        sort(g[i].begin(), g[i].end());
        g[i].erase(unique(g[i].begin(), g[i].end()), g[i].end());
    }

    // Split into connected components in the ORIGINAL graph.
    vector<int> compId(n, -1);
    vector<vector<int>> comps;
    comps.reserve(n);

    int cid = 0;
    vector<int> q;
    q.reserve(n);
    for (int i = 0; i < n; ++i) {
        if (compId[i] != -1) continue;
        compId[i] = cid;
        q.clear();
        q.push_back(i);
        for (size_t qi = 0; qi < q.size(); ++qi) {
            int v = q[qi];
            for (int to : g[v]) {
                if (compId[to] == -1) {
                    compId[to] = cid;
                    q.push_back(to);
                }
            }
        }
        comps.push_back(q);
        ++cid;
    }

    vector<int> answer;
    answer.reserve(n);

    for (const auto& compVerts : comps) {
        // Single isolated vertex component is trivially handled (unless it has a self-loop)
        if (compVerts.size() == 1) {
            int v = compVerts[0];
            if (!hasLoop[v]) answer.push_back(v);
            continue;
        }
        vector<int> part = solveComponentMIS_Exact(compVerts, g, hasLoop);
        answer.insert(answer.end(), part.begin(), part.end());
    }

    // Output
    cout << answer.size() << "\n";
    for (size_t i = 0; i < answer.size(); ++i) {
        if (i) cout << ' ';
        cout << answer[i];
    }
    cout << "\n";
    return 0;
}
