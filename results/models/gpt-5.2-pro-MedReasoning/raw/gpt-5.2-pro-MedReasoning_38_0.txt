import sys

def solve_exact_cover(U, sets_list):
    S = len(sets_list)

    # row_cols[r] = set of columns (elements) covered by row r
    row_cols = [set(s) for s in sets_list]

    # columns[c] = set of rows that cover column c
    columns = {c: set() for c in range(U)}
    for r in range(S):
        for c in row_cols[r]:
            if 0 <= c < U:
                columns[c].add(r)
            else:
                # element outside universe => invalid row for this universe
                return None

    # Quick infeasibility: any column with no covering row
    for c in range(U):
        if not columns[c]:
            return None

    sys.setrecursionlimit(10**7)
    solution = []

    def cover(col):
        col_rows = columns.pop(col)  # remove column
        # remove all rows in this column from the other columns they touch
        for r in col_rows:
            for j in row_cols[r]:
                if j in columns:
                    columns[j].discard(r)
        return col_rows

    def uncover(col, col_rows):
        columns[col] = col_rows
        for r in col_rows:
            for j in row_cols[r]:
                if j in columns and j != col:
                    columns[j].add(r)

    def search():
        if not columns:
            return True

        # Choose column with minimum remaining rows (tie by column id)
        c = min(columns.keys(), key=lambda x: (len(columns[x]), x))
        if not columns[c]:
            return False

        # Deterministic row iteration
        for r in sorted(columns[c]):
            solution.append(r)

            covered = []
            # Cover all columns in this row (deterministic order)
            for j in sorted(row_cols[r]):
                if j in columns:
                    covered.append((j, cover(j)))

            if search():
                return True

            # Backtrack
            for j, col_rows in reversed(covered):
                uncover(j, col_rows)
            solution.pop()

        return False

    if search():
        return solution
    return None


def main():
    data = sys.stdin.read().splitlines()
    if not data:
        return

    header = data[0].strip().split()
    if len(header) < 2:
        print("NO SOLUTION")
        return

    U = int(header[0])
    S = int(header[1])

    rest = data[1:]

    # Drop trailing empty lines
    while rest and rest[-1].strip() == "":
        rest.pop()

    # Input may optionally include an explicit universe line after "U S"
    start = 0
    if len(rest) == S + 1:
        start = 1

    sets_list = []
    for i in range(S):
        line = rest[start + i] if start + i < len(rest) else ""
        line = line.strip()
        if line == "":
            sets_list.append([])
        else:
            sets_list.append(list(map(int, line.split())))

    sol = solve_exact_cover(U, sets_list)

    if sol is None:
        sys.stdout.write("NO SOLUTION\n")
    else:
        sol_sorted = sorted(sol)
        sys.stdout.write("SOLUTION\n")
        sys.stdout.write((" ".join(map(str, sol_sorted))) + "\n")


if __name__ == "__main__":
    main()
