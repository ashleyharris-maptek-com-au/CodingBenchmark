use std::collections::{HashSet, VecDeque};
use std::io::{self, Read, Write};
use std::time::{Duration, Instant};

#[derive(Clone)]
struct Frame {
    v: usize,
    cands: Vec<usize>,
    next: usize,
    changed: [usize; 4],
    changed_len: usize,
}

impl Frame {
    fn new(v: usize, cands: Vec<usize>, changed: [usize; 4], changed_len: usize) -> Self {
        Self {
            v,
            cands,
            next: 0,
            changed,
            changed_len,
        }
    }
}

#[derive(Clone)]
struct XorShift64 {
    s: u64,
}
impl XorShift64 {
    fn new(seed: u64) -> Self {
        let s = if seed == 0 { 0x9e3779b97f4a7c15 } else { seed };
        Self { s }
    }
    fn next_u64(&mut self) -> u64 {
        let mut x = self.s;
        x ^= x << 7;
        x ^= x >> 9;
        x ^= x << 8;
        self.s = x;
        x
    }
    fn next_u32(&mut self) -> u32 {
        (self.next_u64() >> 32) as u32
    }
}

fn manhattan_adj(ax: usize, ay: usize, bx: usize, by: usize) -> bool {
    let dx = if ax > bx { ax - bx } else { bx - ax };
    let dy = if ay > by { ay - by } else { by - ay };
    dx + dy == 1
}

fn parity_necessary_check(
    xs: &[usize],
    ys: &[usize],
    start: usize,
    require_cycle: bool,
) -> bool {
    let n = xs.len();
    let mut c0 = 0usize;
    let mut c1 = 0usize;
    for i in 0..n {
        if ((xs[i] + ys[i]) & 1) == 0 {
            c0 += 1;
        } else {
            c1 += 1;
        }
    }
    if require_cycle {
        // In bipartite graph, Hamiltonian cycle implies equal partitions (and n even).
        return n % 2 == 0 && c0 == c1;
    }
    // Hamiltonian path: partition size difference must be <=1.
    if c0 + c1 != n {
        return false;
    }
    if c0 > c1 + 1 || c1 > c0 + 1 {
        return false;
    }
    // Additionally, since start is fixed:
    // If n even => end color != start color => partitions equal.
    // If n odd  => end color == start color => start's color partition larger by 1.
    let start_color = (xs[start] + ys[start]) & 1;
    if n % 2 == 0 {
        c0 == c1
    } else {
        if start_color == 0 { c0 == c1 + 1 } else { c1 == c0 + 1 }
    }
}

fn connected_check(neigh: &[Vec<usize>], start: usize) -> bool {
    let n = neigh.len();
    let mut seen = vec![false; n];
    let mut q = VecDeque::new();
    seen[start] = true;
    q.push_back(start);
    let mut cnt = 1usize;
    while let Some(v) = q.pop_front() {
        for &u in &neigh[v] {
            if !seen[u] {
                seen[u] = true;
                cnt += 1;
                q.push_back(u);
            }
        }
    }
    cnt == n
}

fn degree_necessary_check(neigh: &[Vec<usize>], start: usize, require_cycle: bool) -> bool {
    let n = neigh.len();
    if n == 0 {
        return false;
    }
    if n == 1 {
        // Single node path exists; cycle requires adjacency to itself (not allowed).
        return !require_cycle;
    }
    for i in 0..n {
        let d = neigh[i].len();
        if d == 0 {
            return false;
        }
        if require_cycle && d < 2 {
            return false;
        }
    }
    // Start must have at least 1 neighbor (already covered by d==0).
    let _ = start;
    true
}

fn verify_solution(
    path: &[usize],
    xs: &[usize],
    ys: &[usize],
    neigh: &[Vec<usize>],
    start: usize,
    require_cycle: bool,
) -> bool {
    let n = xs.len();
    if path.len() != n {
        return false;
    }
    if path[0] != start {
        return false;
    }
    let mut seen = vec![false; n];
    for &v in path {
        if v >= n || seen[v] {
            return false;
        }
        seen[v] = true;
    }
    for w in path.windows(2) {
        let a = w[0];
        let b = w[1];
        if !neigh[a].iter().any(|&u| u == b) {
            return false;
        }
    }
    if require_cycle {
        let last = *path.last().unwrap();
        if !neigh[last].iter().any(|&u| u == start) {
            return false;
        }
    }
    true
}

fn solve_small_bitmask(
    neigh: &[Vec<usize>],
    start: usize,
    require_cycle: bool,
) -> Option<Vec<usize>> {
    let n = neigh.len();
    if n > 64 {
        return None;
    }
    let start_mask = 1u64 << start;
    let mut path = Vec::with_capacity(n);
    path.push(start);

    // Precompute adjacency to start for quick cycle end check.
    let mut adj_start = vec![false; n];
    for &u in &neigh[start] {
        adj_start[u] = true;
    }

    fn dfs(
        neigh: &[Vec<usize>],
        adj_start: &[bool],
        require_cycle: bool,
        n: usize,
        cur: usize,
        visited: u64,
        path: &mut Vec<usize>,
    ) -> bool {
        if path.len() == n {
            return !require_cycle || adj_start[cur];
        }
        // Candidates: unvisited neighbors.
        let mut cands = Vec::new();
        for &u in &neigh[cur] {
            if (visited >> u) & 1 == 0 {
                cands.push(u);
            }
        }
        // Warnsdorff: sort by onward degree into unvisited set.
        cands.sort_by_key(|&u| {
            let mut cnt = 0u32;
            for &v in &neigh[u] {
                if (visited >> v) & 1 == 0 {
                    cnt += 1;
                }
            }
            // If require_cycle and this would be the last move, ensure adjacency to start.
            if require_cycle && path.len() + 1 == n && !adj_start[u] {
                cnt + 1000
            } else {
                cnt
            }
        });

        for u in cands {
            // Simple dead-end prune: if not last step, don't go into a node that has no other exits unless it will be the final end.
            if path.len() + 1 < n {
                let mut onward = 0usize;
                for &v in &neigh[u] {
                    if ((visited >> v) & 1) == 0 && v != cur {
                        onward += 1;
                    }
                }
                if onward == 0 && path.len() + 1 != n {
                    // only acceptable if u becomes final (handled by len check)
                    continue;
                }
            }

            path.push(u);
            if dfs(neigh, adj_start, require_cycle, n, u, visited | (1u64 << u), path) {
                return true;
            }
            path.pop();
        }
        false
    }

    if dfs(neigh, &adj_start, require_cycle, n, start, start_mask, &mut path) {
        Some(path)
    } else {
        None
    }
}

fn build_candidates(
    v: usize,
    neigh: &[Vec<usize>],
    visited: &[bool],
    unvis_deg: &[u8],
    rng: &mut XorShift64,
    require_cycle: bool,
    start: usize,
    step: usize,
    n: usize,
) -> Vec<usize> {
    let mut cands: Vec<usize> = neigh[v].iter().copied().filter(|&u| !visited[u]).collect();

    // If this is the last move in cycle mode, next must be adjacent to start (so the final node is adjacent to start).
    if require_cycle && step + 1 == n {
        cands.retain(|&u| neigh[u].iter().any(|&x| x == start));
    }

    // Sort by (unvisited degree, random tiebreak).
    cands.sort_by_key(|&u| {
        let d = unvis_deg[u] as u32;
        let r = rng.next_u32();
        (d, r)
    });

    cands
}

fn solve_backtracking_iterative(
    neigh: &[Vec<usize>],
    start: usize,
    require_cycle: bool,
    time_limit: Duration,
    expansion_limit_opt: Option<u64>,
    seed: u64,
) -> Option<Vec<usize>> {
    let n = neigh.len();
    let t0 = Instant::now();
    let mut rng = XorShift64::new(seed);

    let mut visited = vec![false; n];
    let mut unvis_deg: Vec<u8> = neigh.iter().map(|v| v.len() as u8).collect();

    // dynamic count: number of *unvisited* vertices with unvis_deg == 0
    let mut zero_unvis = unvis_deg.iter().filter(|&&d| d == 0).count() as i32;

    let mut visited_count = 0usize;

    let mut path: Vec<usize> = Vec::with_capacity(n);
    let mut stack: Vec<Frame> = Vec::with_capacity(n);

    let mut expansions: u64 = 0;

    let mut visit = |v: usize,
                     visited: &mut [bool],
                     unvis_deg: &mut [u8],
                     zero_unvis: &mut i32,
                     visited_count: &mut usize,
                     neigh: &[Vec<usize>]|
     -> ([usize; 4], usize) {
        // Remove v from zero_unvis if it is currently unvisited and has degree 0.
        if !visited[v] && unvis_deg[v] == 0 {
            *zero_unvis -= 1;
        }
        visited[v] = true;
        *visited_count += 1;

        let mut changed = [usize::MAX; 4];
        let mut len = 0usize;
        for &u in &neigh[v] {
            if !visited[u] {
                let d = unvis_deg[u];
                // d must be >=1 because v was unvisited before visiting, hence contributed.
                if d == 1 {
                    *zero_unvis += 1; // will become 0
                }
                unvis_deg[u] = d - 1;
                changed[len] = u;
                len += 1;
            }
        }
        (changed, len)
    };

    let mut unvisit = |v: usize,
                       visited: &mut [bool],
                       unvis_deg: &mut [u8],
                       zero_unvis: &mut i32,
                       visited_count: &mut usize,
                       changed: &[usize; 4],
                       changed_len: usize| {
        // Undo neighbor decrements first.
        for i in 0..changed_len {
            let u = changed[i];
            let d = unvis_deg[u];
            if d == 0 {
                *zero_unvis -= 1; // 0 -> 1
            }
            unvis_deg[u] = d + 1;
        }
        visited[v] = false;
        *visited_count -= 1;
        if unvis_deg[v] == 0 {
            *zero_unvis += 1; // v becomes unvisited and has 0
        }
    };

    // Start visit
    path.push(start);
    let (ch0, ch0len) = visit(
        start,
        &mut visited,
        &mut unvis_deg,
        &mut zero_unvis,
        &mut visited_count,
        neigh,
    );
    let c0 = build_candidates(
        start,
        neigh,
        &visited,
        &unvis_deg,
        &mut rng,
        require_cycle,
        start,
        1,
        n,
    );
    stack.push(Frame::new(start, c0, ch0, ch0len));

    // Helper: prune checks after visiting current end.
    let mut prune_ok = |cur: usize,
                        visited: &[bool],
                        unvis_deg: &[u8],
                        zero_unvis: i32,
                        visited_count: usize|
     -> bool {
        if visited_count == n {
            return true;
        }
        if require_cycle {
            // No dead-end unvisited nodes allowed in a cycle.
            if zero_unvis > 0 {
                return false;
            }
            // Must keep at least one unvisited neighbor of start for closure (unless we're done).
            if neigh[start].iter().all(|&u| visited[u]) {
                return false;
            }
            return true;
        }

        // Path case: at most one unvisited dead-end (it will have to be the final endpoint),
        // and if it exists before the last step, it must be adjacent to the current endpoint (so we can take it next).
        if zero_unvis > 1 {
            return false;
        }
        if zero_unvis == 1 {
            // Find the dead-end node (rare; scan O(n)).
            let mut dead = None;
            for i in 0..n {
                if !visited[i] && unvis_deg[i] == 0 {
                    dead = Some(i);
                    break;
                }
            }
            if let Some(d) = dead {
                // If there is more than one unvisited node remaining after taking next,
                // the only way to include d is to go there immediately from cur.
                let remaining_unvisited = n - visited_count;
                if remaining_unvisited > 1 {
                    if !neigh[cur].iter().any(|&u| u == d) {
                        return false;
                    }
                } else {
                    // remaining_unvisited == 1 => d is the last node; it must be adjacent to cur to finish.
                    if !neigh[cur].iter().any(|&u| u == d) {
                        return false;
                    }
                }
            }
        }
        true
    };

    // Periodic connectivity check on the remaining subgraph (current + unvisited).
    let connectivity_ok = |cur: usize, visited: &[bool], neigh: &[Vec<usize>]| -> bool {
        let n = neigh.len();
        let mut seen = vec![false; n];
        let mut q = VecDeque::new();
        seen[cur] = true;
        q.push_back(cur);
        while let Some(v) = q.pop_front() {
            for &u in &neigh[v] {
                if !seen[u] && (!visited[u] || u == cur) {
                    // allow transition into unvisited nodes; cur is visited but included.
                    if !visited[u] || u == cur {
                        seen[u] = true;
                        q.push_back(u);
                    }
                }
            }
            // also allow moves from visited nodes only if it's cur; we never enqueue other visited nodes.
        }
        for i in 0..n {
            if !visited[i] && !seen[i] {
                return false;
            }
        }
        true
    };

    let mut loop_iters: u64 = 0;
    while !stack.is_empty() {
        loop_iters += 1;
        if (loop_iters & 0x3fff) == 0 {
            if t0.elapsed() > time_limit {
                return None;
            }
        }
        if let Some(limit) = expansion_limit_opt {
            if expansions > limit {
                return None;
            }
        }

        if path.len() == n {
            let last = *path.last().unwrap();
            if !require_cycle || neigh[last].iter().any(|&u| u == start) {
                return Some(path);
            }
            // else: must backtrack
        }

        let depth = path.len();
        let cur = *path.last().unwrap();
        let top = stack.last_mut().unwrap();

        if top.next >= top.cands.len() || depth == n {
            // backtrack from current node (except if it's start and we've exhausted everything)
            if path.len() == 1 {
                break;
            }
            let v = path.pop().unwrap();
            let fr = stack.pop().unwrap();
            unvisit(
                v,
                &mut visited,
                &mut unvis_deg,
                &mut zero_unvis,
                &mut visited_count,
                &fr.changed,
                fr.changed_len,
            );
            continue;
        }

        let nxt = top.cands[top.next];
        top.next += 1;
        if visited[nxt] {
            continue;
        }

        // Expand
        expansions += 1;

        path.push(nxt);
        let (ch, chlen) = visit(
            nxt,
            &mut visited,
            &mut unvis_deg,
            &mut zero_unvis,
            &mut visited_count,
            neigh,
        );

        // Prune checks
        if !prune_ok(nxt, &visited, &unvis_deg, zero_unvis, visited_count) {
            let v = path.pop().unwrap();
            unvisit(
                v,
                &mut visited,
                &mut unvis_deg,
                &mut zero_unvis,
                &mut visited_count,
                &ch,
                chlen,
            );
            continue;
        }

        // Connectivity check occasionally (and when we are near the end).
        if (visited_count & 63) == 0 || (n - visited_count) < 64 {
            if !connectivity_ok(nxt, &visited, neigh) {
                let v = path.pop().unwrap();
                unvisit(
                    v,
                    &mut visited,
                    &mut unvis_deg,
                    &mut zero_unvis,
                    &mut visited_count,
                    &ch,
                    chlen,
                );
                continue;
            }
        }

        let cands = build_candidates(
            nxt,
            neigh,
            &visited,
            &unvis_deg,
            &mut rng,
            require_cycle,
            start,
            path.len(),
            n,
        );
        stack.push(Frame::new(nxt, cands, ch, chlen));
    }

    None
}

// Obstacle-free output constructors (streaming; no need to allocate the whole path).
fn output_snake_path<W: Write>(mut out: W, w: usize, h: usize) -> io::Result<()> {
    let n = w.checked_mul(h).unwrap_or(0);
    writeln!(out, "{n}")?;
    for y in 0..h {
        if y % 2 == 0 {
            for x in 0..w {
                writeln!(out, "{x} {y}")?;
            }
        } else {
            for x in (0..w).rev() {
                writeln!(out, "{x} {y}")?;
            }
        }
    }
    Ok(())
}

// For require_cycle on empty grid:
// If width is even (and h>1), output a Hamiltonian path ending at (1,0) adjacent to (0,0).
// If height is even (and w>1), use the transposed version (ends at (0,1)).
fn output_cycle_like_path_even_width<W: Write>(
    mut out: W,
    w: usize,
    h: usize,
) -> io::Result<()> {
    let n = w.checked_mul(h).unwrap_or(0);
    writeln!(out, "{n}")?;

    // Column 0 downward (includes (0,0))
    for y in 0..h {
        writeln!(out, "0 {y}")?;
    }

    // Columns 1..w-1, alternating direction but only rows 1..h-1 (leave top row for the end).
    for x in 1..w {
        if x % 2 == 1 {
            for y in (1..h).rev() {
                writeln!(out, "{x} {y}")?;
            }
        } else {
            for y in 1..h {
                writeln!(out, "{x} {y}")?;
            }
        }
    }

    // Finally traverse the top row from right to left, excluding (0,0).
    for x in (1..w).rev() {
        writeln!(out, "{x} 0")?;
    }
    Ok(())
}

fn main() {
    // Read all stdin as integers
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let w: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let h: usize = it.next().unwrap().parse().unwrap();
    let obstacle_count: usize = it.next().unwrap().parse().unwrap();
    let require_cycle: bool = it.next().unwrap().parse::<u32>().unwrap() != 0;

    // Fast-path: obstacle-free constructions (scales to huge sizes; output dominates runtime).
    if obstacle_count == 0 {
        let n = w.checked_mul(h).unwrap_or(0);

        if w == 0 || h == 0 || n == 0 {
            println!("0");
            return;
        }
        if require_cycle {
            // Cycle requires at least 2x2 and one even dimension (grid graph is bipartite).
            if n == 1 || w == 1 || h == 1 {
                println!("0");
                return;
            }
            let mut out = io::BufWriter::new(io::stdout());
            if w % 2 == 0 {
                output_cycle_like_path_even_width(&mut out, w, h).unwrap();
                return;
            } else if h % 2 == 0 {
                // Transpose: generate on (h,w) then swap coords in output.
                // We cannot reuse the streaming writer directly without buffering, so implement transposed streaming.
                // Use even-width constructor on transposed (width = h, height = w) and print swapped.
                writeln!(out, "{n}").unwrap();

                // Column 0 downward in transposed => (tx=0,ty=y) => original (x=ty,y=tx) => (x=y,y=0)
                for ty in 0..w {
                    let x = ty;
                    let y = 0usize;
                    writeln!(out, "{x} {y}").unwrap();
                }
                // Columns 1..h-1 in transposed => tx in 1..h
                for tx in 1..h {
                    if tx % 2 == 1 {
                        for ty in (1..w).rev() {
                            let x = ty;
                            let y = tx;
                            writeln!(out, "{x} {y}").unwrap();
                        }
                    } else {
                        for ty in 1..w {
                            let x = ty;
                            let y = tx;
                            writeln!(out, "{x} {y}").unwrap();
                        }
                    }
                }
                // Top row in transposed from right to left excluding (0,0) => (tx,0) => original (0,tx)
                for tx in (1..h).rev() {
                    let x = 0usize;
                    let y = tx;
                    writeln!(out, "{x} {y}").unwrap();
                }
                return;
            } else {
                println!("0");
                return;
            }
        } else {
            // Hamiltonian path always exists on a full rectangle; output row-snake.
            let out = io::BufWriter::new(io::stdout());
            output_snake_path(out, w, h).unwrap();
            return;
        }
    }

    // For obstacles: we build the explicit graph.
    // (Hamiltonian path in general grid-with-holes is NP-complete; this solver is exact for small cases,
    // and uses backtracking with strong pruning for larger ones.)
    let size = match w.checked_mul(h) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    if size == 0 {
        println!("0");
        return;
    }

    // Avoid pathological allocations on extremely large grids with obstacles.
    // (Output size would also be enormous in such cases.)
    if size > 2_000_000 {
        println!("0");
        return;
    }

    let mut blocked = vec![false; size];
    let mut obstacles = HashSet::with_capacity(obstacle_count * 2 + 1);
    for _ in 0..obstacle_count {
        if let (Some(xs), Some(ys)) = (it.next(), it.next()) {
            let x: usize = xs.parse().unwrap();
            let y: usize = ys.parse().unwrap();
            if x < w && y < h {
                let idx = y * w + x;
                if !blocked[idx] {
                    blocked[idx] = true;
                    obstacles.insert(idx);
                }
            }
        } else {
            break;
        }
    }

    // Start must be free.
    if blocked[0] {
        println!("0");
        return;
    }

    // Build compact id mapping.
    let mut grid_to_id = vec![usize::MAX; size];
    let mut xs = Vec::new();
    let mut ys = Vec::new();
    for y in 0..h {
        for x in 0..w {
            let idx = y * w + x;
            if !blocked[idx] {
                let id = xs.len();
                grid_to_id[idx] = id;
                xs.push(x);
                ys.push(y);
            }
        }
    }
    let n = xs.len();
    if n == 0 {
        println!("0");
        return;
    }
    let start = grid_to_id[0];
    if start == usize::MAX {
        println!("0");
        return;
    }

    // Build adjacency list.
    let mut neigh: Vec<Vec<usize>> = vec![Vec::with_capacity(4); n];
    for id in 0..n {
        let x = xs[id];
        let y = ys[id];
        let mut add = |nx: isize, ny: isize| {
            if nx >= 0 && ny >= 0 {
                let nxu = nx as usize;
                let nyu = ny as usize;
                if nxu < w && nyu < h {
                    let g = nyu * w + nxu;
                    let nid = grid_to_id[g];
                    if nid != usize::MAX {
                        neigh[id].push(nid);
                    }
                }
            }
        };
        add(x as isize - 1, y as isize);
        add(x as isize + 1, y as isize);
        add(x as isize, y as isize - 1);
        add(x as isize, y as isize + 1);
    }

    // Necessary checks.
    if !degree_necessary_check(&neigh, start, require_cycle) {
        println!("0");
        return;
    }
    if !connected_check(&neigh, start) {
        println!("0");
        return;
    }
    if !parity_necessary_check(&xs, &ys, start, require_cycle) {
        println!("0");
        return;
    }

    // Solve.
    let time_limit = Duration::from_secs(295);

    let sol = if n <= 64 {
        solve_small_bitmask(&neigh, start, require_cycle)
    } else {
        // For larger instances, use iterative backtracking with pruning.
        // For very large n, cap expansions per attempt and restart with different tie-breaking.
        let base_seed = 0x1234_5678_9abc_def0u64 ^ (n as u64).wrapping_mul(0x9e3779b97f4a7c15);
        let t0 = Instant::now();

        let mut best: Option<Vec<usize>> = None;

        let attempts = if n <= 400 { 1 } else { 32 };
        for a in 0..attempts {
            let elapsed = t0.elapsed();
            if elapsed >= time_limit {
                break;
            }
            let remaining = time_limit - elapsed;

            let expansion_limit = if n <= 400 {
                None
            } else {
                Some(2_000_000u64)
            };

            let seed = base_seed ^ (a as u64).wrapping_mul(0x9e3779b97f4a7c15);
            let cur = solve_backtracking_iterative(
                &neigh,
                start,
                require_cycle,
                remaining,
                expansion_limit,
                seed,
            );
            if cur.is_some() {
                best = cur;
                break;
            }
        }
        best
    };

    if let Some(path) = sol {
        if !verify_solution(&path, &xs, &ys, &neigh, start, require_cycle) {
            // Safety: never output an invalid path.
            println!("0");
            return;
        }
        let mut out = io::BufWriter::new(io::stdout());
        writeln!(out, "{}", path.len()).unwrap();
        for &id in &path {
            writeln!(out, "{} {}", xs[id], ys[id]).unwrap();
        }
    } else {
        println!("0");
    }
}
