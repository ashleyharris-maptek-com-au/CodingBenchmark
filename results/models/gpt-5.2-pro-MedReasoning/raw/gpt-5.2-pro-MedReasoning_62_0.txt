```typescript
declare const require: any;
const fs = require("fs");

const input: string = fs.readFileSync(0, "utf8");
const data = JSON.parse(input);

type Town = { id: string | number; x: number; y: number; value: number };

const width: number = data.width | 0;
const height: number = data.height | 0;
const grid: string[] = data.grid;
const ignitionsIn: number[][] = data.ignitions || [];
const townsIn: Town[] = data.towns || [];
const cutBudget: number = Math.max(0, data.cutBudget | 0);

const N = width * height;

function idx(x: number, y: number): number {
  return y * width + x;
}
function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < width && y >= 0 && y < height;
}
function xOf(i: number): number {
  return i % width;
}
function yOf(i: number): number {
  return (i / width) | 0;
}

// Burnable mask: '.' is forest, '~' is blocked. Treat anything not '.' as blocked.
const burnable = new Uint8Array(N);
for (let y = 0; y < height; y++) {
  const row = grid[y] || "";
  for (let x = 0; x < width; x++) {
    const c = row.charCodeAt(x);
    burnable[idx(x, y)] = c === 46 /* '.' */ ? 1 : 0;
  }
}

// Masks for safety constraints
const townCellMask = new Uint8Array(N);
for (const t of townsIn) {
  if (!inBounds(t.x | 0, t.y | 0)) continue;
  townCellMask[idx(t.x | 0, t.y | 0)] = 1;
}
const ignitionMask = new Uint8Array(N);
const ignitions: number[] = [];
for (const p of ignitionsIn) {
  const x = p[0] | 0,
    y = p[1] | 0;
  if (!inBounds(x, y)) continue;
  const i = idx(x, y);
  ignitionMask[i] = 1;
  ignitions.push(i);
}

// BFS with cuts
function computeDistances(cutMask: Uint8Array): { dist: Int32Array; order: Int32Array; orderLen: number } {
  const dist = new Int32Array(N);
  dist.fill(-1);

  const queue = new Int32Array(N);
  const order = new Int32Array(N);
  let qh = 0,
    qt = 0,
    ol = 0;

  for (let k = 0; k < ignitions.length; k++) {
    const s = ignitions[k];
    if (!burnable[s]) continue;
    if (cutMask[s]) continue;
    if (dist[s] !== -1) continue;
    dist[s] = 0;
    queue[qt++] = s;
    order[ol++] = s;
  }

  while (qh < qt) {
    const v = queue[qh++];
    const dnext = dist[v] + 1;
    const x = v % width;

    // up
    if (v >= width) {
      const u = v - width;
      if (dist[u] === -1 && burnable[u] && !cutMask[u]) {
        dist[u] = dnext;
        queue[qt++] = u;
        order[ol++] = u;
      }
    }
    // down
    if (v + width < N) {
      const u = v + width;
      if (dist[u] === -1 && burnable[u] && !cutMask[u]) {
        dist[u] = dnext;
        queue[qt++] = u;
        order[ol++] = u;
      }
    }
    // left
    if (x !== 0) {
      const u = v - 1;
      if (dist[u] === -1 && burnable[u] && !cutMask[u]) {
        dist[u] = dnext;
        queue[qt++] = u;
        order[ol++] = u;
      }
    }
    // right
    if (x !== width - 1) {
      const u = v + 1;
      if (dist[u] === -1 && burnable[u] && !cutMask[u]) {
        dist[u] = dnext;
        queue[qt++] = u;
        order[ol++] = u;
      }
    }
  }

  return { dist, order, orderLen: ol };
}

// Baseline (no cuts): determine which towns are "otherwise burned"
const emptyCuts = new Uint8Array(N);
const base = computeDistances(emptyCuts);
const baselineThreatTowns: { cell: number; value: number }[] = [];
for (const t of townsIn) {
  const x = t.x | 0,
    y = t.y | 0;
  if (!inBounds(x, y)) continue;
  const c = idx(x, y);
  if (!burnable[c]) continue;
  if (base.dist[c] >= 0) baselineThreatTowns.push({ cell: c, value: +t.value });
}

if (cutBudget <= 0 || ignitions.length === 0 || baselineThreatTowns.length === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

// Greedy iterations with re-simulation
const cutMask = new Uint8Array(N);
const firebreaks: number[] = [];

// Degree arrays (computed for reached cells each iteration)
const parentsCount = new Uint8Array(N);
const outDeg = new Uint8Array(N);

// Value array for back-propagation (only meaningful on reached cells)
const val = new Float64Array(N);

// Min-heap for top candidates
class MinHeap {
  scores: number[] = [];
  idxs: number[] = [];
  size(): number {
    return this.scores.length;
  }
  peekScore(): number {
    return this.scores[0];
  }
  push(score: number, id: number): void {
    let i = this.scores.length;
    this.scores.push(score);
    this.idxs.push(id);
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.scores[p] <= score) break;
      this.scores[i] = this.scores[p];
      this.idxs[i] = this.idxs[p];
      i = p;
    }
    this.scores[i] = score;
    this.idxs[i] = id;
  }
  replaceTop(score: number, id: number): void {
    const n = this.scores.length;
    let i = 0;
    while (true) {
      let l = i * 2 + 1;
      if (l >= n) break;
      let r = l + 1;
      let c = r < n && this.scores[r] < this.scores[l] ? r : l;
      if (this.scores[c] >= score) break;
      // If child is smaller, we still need min-heap, so bubble-down top:
      // But since we are replacing with a larger score, we bubble it down by swapping with smaller child.
      this.scores[i] = this.scores[c];
      this.idxs[i] = this.idxs[c];
      i = c;
    }
    this.scores[i] = score;
    this.idxs[i] = id;
    // Now fix upwards if needed (rare): actually not needed for replaceTop in min-heap with larger key.
  }
  pop(): { score: number; id: number } | null {
    const n = this.scores.length;
    if (n === 0) return null;
    const res = { score: this.scores[0], id: this.idxs[0] };
    const score = this.scores[n - 1];
    const id = this.idxs[n - 1];
    this.scores.pop();
    this.idxs.pop();
    if (n - 1 === 0) return res;

    // heapify down
    let i = 0;
    const nn = n - 1;
    while (true) {
      let l = i * 2 + 1;
      if (l >= nn) break;
      let r = l + 1;
      let c = r < nn && this.scores[r] < this.scores[l] ? r : l;
      if (this.scores[c] >= score) break;
      this.scores[i] = this.scores[c];
      this.idxs[i] = this.idxs[c];
      i = c;
    }
    this.scores[i] = score;
    this.idxs[i] = id;
    return res;
  }
}

function computeDegrees(dist: Int32Array, order: Int32Array, orderLen: number): void {
  for (let oi = 0; oi < orderLen; oi++) {
    const v = order[oi];
    const d = dist[v];
    const x = v % width;

    let pc = 0;
    let od = 0;

    // up
    if (v >= width) {
      const u = v - width;
      const du = dist[u];
      if (du === d - 1) pc++;
      else if (du === d + 1) od++;
    }
    // down
    if (v + width < N) {
      const u = v + width;
      const du = dist[u];
      if (du === d - 1) pc++;
      else if (du === d + 1) od++;
    }
    // left
    if (x !== 0) {
      const u = v - 1;
      const du = dist[u];
      if (du === d - 1) pc++;
      else if (du === d + 1) od++;
    }
    // right
    if (x !== width - 1) {
      const u = v + 1;
      const du = dist[u];
      if (du === d - 1) pc++;
      else if (du === d + 1) od++;
    }

    parentsCount[v] = pc;
    outDeg[v] = od;
  }
}

function backpropValue(dist: Int32Array, order: Int32Array, orderLen: number): void {
  for (let oi = orderLen - 1; oi >= 0; oi--) {
    const v = order[oi];
    const d = dist[v];
    if (d <= 0) continue;
    const vval = val[v];
    if (vval <= 0) continue;

    const pc = parentsCount[v];
    if (pc <= 0) continue;
    const share = vval / pc;

    const x = v % width;
    // up parent
    if (v >= width) {
      const u = v - width;
      if (dist[u] === d - 1) val[u] += share;
    }
    // down parent
    if (v + width < N) {
      const u = v + width;
      if (dist[u] === d - 1) val[u] += share;
    }
    // left parent
    if (x !== 0) {
      const u = v - 1;
      if (dist[u] === d - 1) val[u] += share;
    }
    // right parent
    if (x !== width - 1) {
      const u = v + 1;
      if (dist[u] === d - 1) val[u] += share;
    }
  }
}

function addCutsFromScores(
  dist: Int32Array,
  order: Int32Array,
  orderLen: number,
  toAdd: number
): number {
  if (toAdd <= 0) return 0;

  const heapLimit = Math.min(orderLen, Math.max(5000, Math.min(200000, toAdd * 12)));
  const heap = new MinHeap();

  for (let oi = 0; oi < orderLen; oi++) {
    const v = order[oi];
    if (cutMask[v]) continue;
    if (!burnable[v]) continue;
    if (townCellMask[v]) continue;
    if (ignitionMask[v]) continue;

    const vv = val[v];
    if (vv <= 0) continue;

    const d = dist[v];
    if (d < 0) continue;

    const od = outDeg[v] + 1;
    const pc = parentsCount[v];

    // Heuristic score:
    // - prefer high threatened-value flow
    // - prefer more downstream branching (higher out-degree)
    // - mildly prefer earlier cuts (smaller d)
    // - mildly penalize high number of parents (many alternative shortest paths)
    const distFactor = Math.pow(d + 1, 0.22);
    const parentPenalty = 1 + 0.55 * Math.max(0, pc - 1);
    const score = (vv * od) / (distFactor * parentPenalty);

    if (heap.size() < heapLimit) {
      heap.push(score, v);
    } else if (score > heap.peekScore()) {
      heap.replaceTop(score, v);
    }
  }

  const items: { score: number; id: number }[] = [];
  while (heap.size() > 0) {
    const it = heap.pop();
    if (it) items.push(it);
  }
  items.sort((a, b) => b.score - a.score);

  let added = 0;
  for (let i = 0; i < items.length && added < toAdd; i++) {
    const v = items[i].id;
    if (cutMask[v]) continue;
    if (!burnable[v]) continue;
    if (townCellMask[v]) continue;
    if (ignitionMask[v]) continue;
    // Only meaningful if the cell is currently reachable by fire
    if (dist[v] < 0) continue;

    cutMask[v] = 1;
    firebreaks.push(v);
    added++;
  }
  return added;
}

let remaining = cutBudget;
const maxIters = 3;

for (let iter = 0; iter < maxIters && remaining > 0; iter++) {
  const sim = computeDistances(cutMask);
  const dist = sim.dist;

  // Clear values on reached cells only
  for (let oi = 0; oi < sim.orderLen; oi++) {
    val[sim.order[oi]] = 0;
  }

  // Seed with towns that would otherwise burn, and still burn under current cuts
  let threatenedValue = 0;
  for (let i = 0; i < baselineThreatTowns.length; i++) {
    const c = baselineThreatTowns[i].cell;
    if (dist[c] >= 0) {
      const v = baselineThreatTowns[i].value;
      val[c] += v;
      threatenedValue += v;
    }
  }
  if (threatenedValue <= 0) break;

  computeDegrees(dist, sim.order, sim.orderLen);
  backpropValue(dist, sim.order, sim.orderLen);

  const itersLeft = maxIters - iter;
  const toAdd = Math.min(remaining, Math.max(1, Math.ceil(remaining / itersLeft)));
  const added = addCutsFromScores(dist, sim.order, sim.orderLen, toAdd);

  remaining -= added;
  if (added === 0) break;
}

// Output coordinates
const out: number[][] = new Array(firebreaks.length);
for (let i = 0; i < firebreaks.length; i++) {
  const v = firebreaks[i];
  out[i] = [xOf(v), yOf(v)];
}

process.stdout.write(JSON.stringify({ firebreaks: out }));
```
