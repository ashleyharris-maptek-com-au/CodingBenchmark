```spirv
; SPIR-V
; Version: 1.0
; Generator: AI; 0
; Bound: 200
; Schema: 0
OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_in 0 Offset 0

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 Offset 0

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%void_fn = OpTypeFunction %void

%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%uint_4 = OpConstant %uint 4

%float_thickness = OpConstant %float 0.025

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4float = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float = OpTypePointer Output %v4float

%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3float = OpTypePointer Input %v3float
%ptr_out_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3float Output

%main = OpFunction %void None %void_fn
%entry = OpLabel

; Dummy reads to ensure listed interfaces are used
%wp0_ptr = OpAccessChain %ptr_in_v3float %worldPos %int_0
%wp0 = OpLoad %v3float %wp0_ptr
%nm0_ptr = OpAccessChain %ptr_in_v3float %normal %int_0
%nm0 = OpLoad %v3float %nm0_ptr

; Load gl_in positions: p0(prev), p1(start), p2(end), p3(next)
%p0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p3_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_ptr

%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p3_xy = OpVectorShuffle %v2float %p3 %p3 0 1

%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3
%p2_z = OpCompositeExtract %float %p2 2
%p2_w = OpCompositeExtract %float %p2 3

; t0 = normalize(p2.xy - p0.xy)
%delta0 = OpFSub %v2float %p2_xy %p0_xy
%t0 = OpExtInst %v2float %glsl Normalize %delta0

; t1 = normalize(p3.xy - p1.xy)
%delta1 = OpFSub %v2float %p3_xy %p1_xy
%t1 = OpExtInst %v2float %glsl Normalize %delta1

; perp0 = vec2(-t0.y, t0.x)
%t0_x = OpCompositeExtract %float %t0 0
%t0_y = OpCompositeExtract %float %t0 1
%neg_t0_y = OpFNegate %float %t0_y
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x

; perp1 = vec2(-t1.y, t1.x)
%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1
%neg_t1_y = OpFNegate %float %t1_y
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x

%off0 = OpVectorTimesScalar %v2float %perp0 %float_thickness
%off1 = OpVectorTimesScalar %v2float %perp1 %float_thickness

%v0_xy = OpFAdd %v2float %p1_xy %off0
%v1_xy = OpFSub %v2float %p1_xy %off0
%v2_xy = OpFAdd %v2float %p2_xy %off1
%v3_xy = OpFSub %v2float %p2_xy %off1

; Load colors for start/end
%col1_ptr = OpAccessChain %ptr_in_v3float %color %int_1
%col1 = OpLoad %v3float %col1_ptr
%col2_ptr = OpAccessChain %ptr_in_v3float %color %int_2
%col2 = OpLoad %v3float %col2_ptr

%gl_out_pos_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0

; Emit vertex 0: start + perp0*thickness, color[1]
%v0_x = OpCompositeExtract %float %v0_xy 0
%v0_y = OpCompositeExtract %float %v0_xy 1
%pos0 = OpCompositeConstruct %v4float %v0_x %v0_y %p1_z %p1_w
OpStore %gl_out_pos_ptr %pos0
OpStore %outColor %col1
OpEmitVertex

; Emit vertex 1: start - perp0*thickness, color[1]
%v1_x = OpCompositeExtract %float %v1_xy 0
%v1_y = OpCompositeExtract %float %v1_xy 1
%pos1 = OpCompositeConstruct %v4float %v1_x %v1_y %p1_z %p1_w
OpStore %gl_out_pos_ptr %pos1
OpStore %outColor %col1
OpEmitVertex

; Emit vertex 2: end + perp1*thickness, color[2]
%v2_x = OpCompositeExtract %float %v2_xy 0
%v2_y = OpCompositeExtract %float %v2_xy 1
%pos2 = OpCompositeConstruct %v4float %v2_x %v2_y %p2_z %p2_w
OpStore %gl_out_pos_ptr %pos2
OpStore %outColor %col2
OpEmitVertex

; Emit vertex 3: end - perp1*thickness, color[2]
%v3_x = OpCompositeExtract %float %v3_xy 0
%v3_y = OpCompositeExtract %float %v3_xy 1
%pos3 = OpCompositeConstruct %v4float %v3_x %v3_y %p2_z %p2_w
OpStore %gl_out_pos_ptr %pos3
OpStore %outColor %col2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
