```glsl
#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main()
{
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against out-of-range invocations
    if (i >= N) {
        return;
    }

    float dt   = physics.x;
    float eps  = physics.y;      // softening
    float G    = physics.z;

    uint baseI = i * 2u;

    vec4 posMass_i = inData[baseI + 0u];
    vec3 pos_i     = posMass_i.xyz;
    float mass_i   = posMass_i.w;

    vec3 vel_i     = inData[baseI + 1u].xyz;

    // Accumulate acceleration from all other bodies
    vec3 acc = vec3(0.0);
    float eps2 = eps * eps;

    for (uint j = 0u; j < N; ++j) {
        if (j == i) continue; // Skip self-interaction

        uint baseJ = j * 2u;
        vec4 posMass_j = inData[baseJ + 0u];

        vec3 r = posMass_j.xyz - pos_i;  // vector from i to j
        float mj = posMass_j.w;

        float r2 = dot(r, r) + eps2;

        // invDist^3 = 1 / (r^2)^(3/2)
        float invDist = inversesqrt(r2);
        float invDist3 = invDist * invDist * invDist;

        acc += (G * mj) * r * invDist3;
    }

    // Semi-implicit Euler (a.k.a. Euler/leapfrog-style update used in many N-body demos):
    //   v_{t+dt} = v_t + a_t * dt
    //   x_{t+dt} = x_t + v_{t+dt} * dt
    vec3 vel_new = vel_i + acc * dt;
    vec3 pos_new = pos_i + vel_new * dt;

    // Write updated state (preserve mass and velocity.w = 0)
    outData[baseI + 0u] = vec4(pos_new, mass_i);
    outData[baseI + 1u] = vec4(vel_new, 0.0);
}
```
