#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <unordered_map>
#include <cstdint>
#include <limits>

using namespace std;

static constexpr long long INF = (std::numeric_limits<long long>::max() / 4);

struct AdjEdge {
    int to;
    long long w;
};

struct Edge {
    int u, v;
    long long w;
};

struct DSU {
    vector<int> p, r;
    vector<int> termCnt; // number of terminals in the component (for constrained Kruskal)

    DSU() = default;
    DSU(int n, const vector<char>& isTerm) { init(n, isTerm); }

    void init(int n, const vector<char>& isTerm) {
        p.resize(n);
        r.assign(n, 0);
        termCnt.assign(n, 0);
        for (int i = 0; i < n; ++i) {
            p[i] = i;
            termCnt[i] = isTerm[i] ? 1 : 0;
        }
    }

    int find(int a) {
        while (p[a] != a) {
            p[a] = p[p[a]];
            a = p[a];
        }
        return a;
    }

    // returns true if merged
    bool unite(int a, int b, int& terminalComponents) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) std::swap(a, b);
        if (r[a] == r[b]) r[a]++;

        // If both components already contain at least one terminal, merging reduces terminal-components by 1.
        if (termCnt[a] > 0 && termCnt[b] > 0) terminalComponents--;

        p[b] = a;
        termCnt[a] += termCnt[b];
        return true;
    }
};

static inline uint64_t edgeKey(int a, int b) {
    if (a > b) std::swap(a, b);
    return (uint64_t(uint32_t(a)) << 32) | uint32_t(b);
}

static void addUndirEdge(unordered_map<uint64_t, long long>& mp, int a, int b, long long w) {
    if (a == b) return;
    uint64_t k = edgeKey(a, b);
    auto it = mp.find(k);
    if (it == mp.end()) mp.emplace(k, w);
    else it->second = min(it->second, w);
}

static vector<long long> dijkstra(
    const vector<vector<AdjEdge>>& g,
    int src,
    vector<int>* parentV,              // optional
    vector<long long>* parentW,        // optional
    int target = -1                    // optional early-stop
) {
    int n = (int)g.size();
    vector<long long> dist(n, INF);
    dist[src] = 0;

    if (parentV) parentV->assign(n, -1);
    if (parentW) parentW->assign(n, 0);

    using P = pair<long long, int>;
    priority_queue<P, vector<P>, greater<P>> pq;
    pq.push({0, src});

    while (!pq.empty()) {
        auto [d, v] = pq.top();
        pq.pop();
        if (d != dist[v]) continue;
        if (target >= 0 && v == target) break;

        for (const auto& e : g[v]) {
            long long nd = d + e.w;
            if (nd < dist[e.to]) {
                dist[e.to] = nd;
                if (parentV) (*parentV)[e.to] = v;
                if (parentW) (*parentW)[e.to] = e.w;
                pq.push({nd, e.to});
            }
        }
    }
    return dist;
}

// Kruskal variant: picks a minimum-weight acyclic set of edges that makes all terminals connected.
// Skips edges between two components that both currently contain 0 terminals (they can't help connect terminals).
static bool constrainedKruskalConnectTerminals(
    int n,
    const vector<int>& terminals,
    const vector<Edge>& edges,
    vector<Edge>& chosen,
    long long& totalW
) {
    vector<char> isTerm(n, 0);
    for (int t : terminals) isTerm[t] = 1;

    // Count initial terminal components (unique terminals)
    int terminalComponents = 0;
    {
        vector<char> seen(n, 0);
        for (int t : terminals) if (!seen[t]) { seen[t] = 1; terminalComponents++; }
    }
    if (terminalComponents <= 1) {
        chosen.clear();
        totalW = 0;
        return true;
    }

    DSU dsu(n, isTerm);

    vector<Edge> es = edges;
    sort(es.begin(), es.end(), [](const Edge& a, const Edge& b) {
        if (a.w != b.w) return a.w < b.w;
        if (a.u != b.u) return a.u < b.u;
        return a.v < b.v;
    });

    chosen.clear();
    totalW = 0;

    for (const auto& e : es) {
        if (terminalComponents == 1) break;
        int ru = dsu.find(e.u);
        int rv = dsu.find(e.v);
        if (ru == rv) continue;
        if (dsu.termCnt[ru] == 0 && dsu.termCnt[rv] == 0) continue; // useless
        if (dsu.unite(ru, rv, terminalComponents)) {
            chosen.push_back(e);
            totalW += e.w;
        }
    }
    return terminalComponents == 1;
}

// Heuristic: compute MST of the original graph, then prune non-terminal leaves.
// Produces a connected, acyclic subgraph containing all terminals (if possible).
static bool mstPruneHeuristic(
    int n,
    const vector<Edge>& edges,
    const vector<int>& terminals,
    vector<Edge>& outEdges,
    long long& outW
) {
    vector<char> isTerm(n, 0);
    for (int t : terminals) isTerm[t] = 1;

    // Build MST (Kruskal over all vertices)
    vector<Edge> es = edges;
    sort(es.begin(), es.end(), [](const Edge& a, const Edge& b) {
        if (a.w != b.w) return a.w < b.w;
        if (a.u != b.u) return a.u < b.u;
        return a.v < b.v;
    });

    // plain DSU
    struct DSU0 {
        vector<int> p, r;
        DSU0(int n): p(n), r(n,0) { for(int i=0;i<n;i++) p[i]=i; }
        int find(int a){ while(p[a]!=a){ p[a]=p[p[a]]; a=p[a]; } return a; }
        bool unite(int a,int b){
            a=find(a); b=find(b);
            if(a==b) return false;
            if(r[a]<r[b]) swap(a,b);
            if(r[a]==r[b]) r[a]++;
            p[b]=a;
            return true;
        }
    } dsu(n);

    vector<Edge> mst;
    mst.reserve(n - 1);
    for (auto &e : es) {
        if ((int)mst.size() == n - 1) break;
        if (dsu.unite(e.u, e.v)) mst.push_back(e);
    }
    if (mst.empty()) return false;

    // Check if all terminals are in same component in the MST forest
    int root0 = dsu.find(terminals[0]);
    for (int t : terminals) {
        if (dsu.find(t) != root0) return false; // disconnected graph w.r.t terminals
    }

    // Build adjacency of the MST
    vector<vector<pair<int,int>>> adj(n); // (neighbor, edgeIndex)
    adj.assign(n, {});
    for (int i = 0; i < (int)mst.size(); ++i) {
        adj[mst[i].u].push_back({mst[i].v, i});
        adj[mst[i].v].push_back({mst[i].u, i});
    }

    vector<int> deg(n, 0);
    for (int i = 0; i < n; ++i) deg[i] = (int)adj[i].size();

    vector<char> removed(mst.size(), 0);
    queue<int> q;
    for (int v = 0; v < n; ++v) {
        if (!isTerm[v] && deg[v] <= 1) q.push(v);
    }

    auto removeEdge = [&](int edgeIdx) {
        if (removed[edgeIdx]) return;
        removed[edgeIdx] = 1;
        int a = mst[edgeIdx].u;
        int b = mst[edgeIdx].v;
        deg[a]--;
        deg[b]--;
        if (!isTerm[a] && deg[a] == 1) q.push(a);
        if (!isTerm[b] && deg[b] == 1) q.push(b);
    };

    while (!q.empty()) {
        int v = q.front();
        q.pop();
        if (isTerm[v]) continue;
        if (deg[v] != 1) continue;

        // find the unique remaining incident edge
        int edgeIdx = -1;
        for (auto [to, ei] : adj[v]) {
            if (!removed[ei]) { edgeIdx = ei; break; }
        }
        if (edgeIdx >= 0) removeEdge(edgeIdx);
    }

    outEdges.clear();
    outW = 0;
    for (int i = 0; i < (int)mst.size(); ++i) {
        if (!removed[i]) {
            outEdges.push_back(mst[i]);
            outW += mst[i].w;
        }
    }
    // outEdges form a tree (subtree of MST), hence connected & acyclic, and contains all terminals.
    return true;
}

// Heuristic: metric-closure MST on terminals (2-approx idea), expand paths, then constrained Kruskal.
static bool metricClosureHeuristic(
    const vector<vector<AdjEdge>>& g,
    const vector<Edge>& inputEdges, // not used directly, but kept for signature symmetry
    const vector<int>& terminals,
    vector<Edge>& outEdges,
    long long& outW
) {
    (void)inputEdges;
    int n = (int)g.size();
    int K = (int)terminals.size();
    if (K <= 1) { outEdges.clear(); outW = 0; return true; }

    // Pairwise terminal distances
    vector<vector<long long>> distK(K, vector<long long>(K, INF));
    for (int i = 0; i < K; ++i) {
        auto dist = dijkstra(g, terminals[i], nullptr, nullptr, -1);
        for (int j = 0; j < K; ++j) distK[i][j] = dist[terminals[j]];
        distK[i][i] = 0;
    }

    // Prim MST on complete terminal graph
    vector<long long> key(K, INF);
    vector<int> parent(K, -1);
    vector<char> inMST(K, 0);
    key[0] = 0;

    for (int it = 0; it < K; ++it) {
        int u = -1;
        long long best = INF;
        for (int i = 0; i < K; ++i) if (!inMST[i] && key[i] < best) { best = key[i]; u = i; }
        if (u < 0 || best >= INF/2) return false; // unreachable terminals
        inMST[u] = 1;
        for (int v = 0; v < K; ++v) {
            if (!inMST[v] && distK[u][v] < key[v]) {
                key[v] = distK[u][v];
                parent[v] = u;
            }
        }
    }

    // Expand MST edges back to original graph paths
    unordered_map<uint64_t, long long> used;
    used.reserve((size_t)K * 64);

    for (int v = 1; v < K; ++v) {
        int u = parent[v];
        if (u < 0) return false;

        int src = terminals[v];
        int dst = terminals[u];

        vector<int> parV;
        vector<long long> parW;
        auto dist = dijkstra(g, src, &parV, &parW, dst);
        if (dist[dst] >= INF/2) return false;

        int cur = dst;
        while (cur != src) {
            int p = parV[cur];
            long long w = parW[cur];
            if (p < 0) return false;
            addUndirEdge(used, p, cur, w);
            cur = p;
        }
    }

    vector<Edge> subEdges;
    subEdges.reserve(used.size());
    for (auto &kv : used) {
        uint64_t k = kv.first;
        int a = int(k >> 32);
        int b = int(uint32_t(k));
        subEdges.push_back({a, b, kv.second});
    }

    return constrainedKruskalConnectTerminals(n, terminals, subEdges, outEdges, outW);
}

// Exact (Dreyfus-Wagner style DP) for small K and small enough (2^K * n) state space.
// Returns a set of edges (tree) via reconstruction + constrained Kruskal on collected edges.
static bool exactSteinerDP(
    const vector<vector<AdjEdge>>& g,
    const vector<int>& terminals,
    vector<Edge>& outEdges,
    long long& outW
) {
    int n = (int)g.size();
    int K = (int)terminals.size();
    if (K <= 1) { outEdges.clear(); outW = 0; return true; }
    if (K > 20) return false;

    uint64_t fullMask = (K == 64 ? ~0ULL : ((1ULL << K) - 1ULL));
    uint64_t states = (fullMask + 1ULL) * (uint64_t)n;

    // Precompute shortest paths from each terminal to all nodes (for base DP + reconstruction).
    vector<long long> distT((uint64_t)K * n, INF);
    vector<int> parTV((uint64_t)K * n, -1);
    vector<long long> parTW((uint64_t)K * n, 0);

    for (int i = 0; i < K; ++i) {
        vector<int> parV;
        vector<long long> parW;
        auto dist = dijkstra(g, terminals[i], &parV, &parW, -1);
        for (int v = 0; v < n; ++v) {
            distT[(uint64_t)i * n + v] = dist[v];
            parTV[(uint64_t)i * n + v] = parV[v];
            parTW[(uint64_t)i * n + v] = parW[v];
        }
    }

    // DP arrays over (mask, v)
    vector<long long> dp(states, INF);
    vector<int> dpParV(states, -1);          // >=0: edge-step parent; -1: not an edge-step
    vector<long long> dpParW(states, 0);     // weight for edge-step
    vector<uint32_t> dpSplit(states, 0);     // if nonzero: combine at v with submask=dpSplit
    vector<int16_t> dpBase(states, -1);      // if base: terminal index [0..K)

    auto idx = [&](uint32_t mask, int v) -> uint64_t {
        return (uint64_t)mask * (uint64_t)n + (uint64_t)v;
    };

    // init singletons
    for (int i = 0; i < K; ++i) {
        uint32_t mask = (1u << i);
        for (int v = 0; v < n; ++v) {
            uint64_t id = idx(mask, v);
            dp[id] = distT[(uint64_t)i * n + v];
            dpParV[id] = -1;
            dpSplit[id] = 0;
            dpBase[id] = (int16_t)i;
        }
    }

    // DP over masks
    for (uint32_t mask = 1; mask <= (uint32_t)fullMask; ++mask) {
        if ((mask & (mask - 1u)) == 0u) continue; // singleton already set

        // Combine submasks at the same root vertex v:
        // dp[mask][v] = min_{sub proper} dp[sub][v] + dp[mask-sub][v]
        uint64_t baseOff = (uint64_t)mask * (uint64_t)n;
        // dp[mask][*] currently INF; we will update by combine then run Dijkstra-relax.
        for (uint32_t sub = (mask - 1u) & mask; sub; sub = (sub - 1u) & mask) {
            uint32_t other = mask ^ sub;
            if (sub > other) continue; // avoid duplicate splits
            uint64_t offA = (uint64_t)sub * (uint64_t)n;
            uint64_t offB = (uint64_t)other * (uint64_t)n;

            for (int v = 0; v < n; ++v) {
                long long a = dp[offA + (uint64_t)v];
                long long b = dp[offB + (uint64_t)v];
                if (a >= INF/2 || b >= INF/2) continue;
                long long cand = a + b;
                uint64_t id = baseOff + (uint64_t)v;
                if (cand < dp[id]) {
                    dp[id] = cand;
                    dpParV[id] = -1;
                    dpParW[id] = 0;
                    dpSplit[id] = sub;
                    dpBase[id] = -1;
                }
            }
        }

        // Multi-source Dijkstra to allow moving the root v along edges:
        // dp[mask][to] <= dp[mask][v] + w(v,to)
        using P = pair<long long, int>;
        priority_queue<P, vector<P>, greater<P>> pq;
        for (int v = 0; v < n; ++v) {
            long long d0 = dp[baseOff + (uint64_t)v];
            if (d0 < INF/2) pq.push({d0, v});
        }

        while (!pq.empty()) {
            auto [d, v] = pq.top();
            pq.pop();
            uint64_t idv = baseOff + (uint64_t)v;
            if (d != dp[idv]) continue;
            for (const auto& e : g[v]) {
                int to = e.to;
                long long nd = d + e.w;
                uint64_t idto = baseOff + (uint64_t)to;
                if (nd < dp[idto]) {
                    dp[idto] = nd;
                    dpParV[idto] = v;
                    dpParW[idto] = e.w;
                    dpSplit[idto] = 0;
                    dpBase[idto] = -1;
                    pq.push({nd, to});
                }
            }
        }
    }

    // Choose best root v for fullMask
    uint32_t all = (uint32_t)fullMask;
    uint64_t offAll = (uint64_t)all * (uint64_t)n;
    long long best = INF;
    int bestV = -1;
    for (int v = 0; v < n; ++v) {
        long long d = dp[offAll + (uint64_t)v];
        if (d < best) { best = d; bestV = v; }
    }
    if (bestV < 0 || best >= INF/2) return false;

    // Reconstruct by following dpParV / dpSplit / dpBase.
    unordered_map<uint64_t, long long> used;
    used.reserve(4096);

    auto addEdgePath = [&](int a, int b, long long w) {
        addUndirEdge(used, a, b, w);
    };

    // recursion with explicit stack (avoid deep recursion)
    struct Frame { uint32_t mask; int v; };
    vector<Frame> st;
    st.push_back({all, bestV});

    while (!st.empty()) {
        auto [mask, v] = st.back();
        st.pop_back();

        uint64_t id = idx(mask, v);

        if (dpParV[id] >= 0) {
            int p = dpParV[id];
            long long w = dpParW[id];
            addEdgePath(p, v, w);
            st.push_back({mask, p});
            continue;
        }
        if (dpSplit[id] != 0) {
            uint32_t sub = dpSplit[id];
            st.push_back({sub, v});
            st.push_back({mask ^ sub, v});
            continue;
        }
        int tIdx = (int)dpBase[id];
        if (tIdx < 0) {
            // Should not happen if DP is consistent
            return false;
        }
        int src = terminals[tIdx];
        int cur = v;
        while (cur != src) {
            int p = parTV[(uint64_t)tIdx * n + (uint64_t)cur];
            long long w = parTW[(uint64_t)tIdx * n + (uint64_t)cur];
            if (p < 0) return false;
            addEdgePath(p, cur, w);
            cur = p;
        }
    }

    vector<Edge> subEdges;
    subEdges.reserve(used.size());
    for (auto &kv : used) {
        uint64_t k = kv.first;
        int a = int(k >> 32);
        int b = int(uint32_t(k));
        subEdges.push_back({a, b, kv.second});
    }

    return constrainedKruskalConnectTerminals(n, terminals, subEdges, outEdges, outW);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, K;
    if (!(cin >> n >> m >> K)) return 0;

    vector<vector<AdjEdge>> g(n);
    vector<Edge> edges;
    edges.reserve(m);

    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        g[u].push_back({v, w});
        g[v].push_back({u, w});
        edges.push_back({u, v, w});
    }

    vector<int> terminals(K);
    for (int i = 0; i < K; ++i) cin >> terminals[i];

    // Handle trivial cases
    if (K <= 1) {
        cout << 0 << " " << 0 << "\n";
        return 0;
    }

    // Decide exact vs heuristic based on state size and memory considerations.
    bool solved = false;
    vector<Edge> answer;
    long long totalW = 0;

    bool tryExact = false;
    if (K <= 20) {
        uint64_t fullMask = (1ULL << K) - 1ULL;
        uint64_t states = (fullMask + 1ULL) * (uint64_t)n;

        // Memory heuristic: DP holds several arrays per state.
        // Keep state count modest to avoid OOM on typical judge limits.
        const uint64_t STATE_LIMIT = 4'000'000ULL;
        if (states <= STATE_LIMIT) tryExact = true;
    }

    if (tryExact) {
        solved = exactSteinerDP(g, terminals, answer, totalW);
    }

    if (!solved) {
        // If terminal set is not too big, try metric closure; else do MST+prune.
        if (K <= 200) {
            solved = metricClosureHeuristic(g, edges, terminals, answer, totalW);
        }
        if (!solved) {
            solved = mstPruneHeuristic(n, edges, terminals, answer, totalW);
        }
    }

    if (!solved) {
        // If graph is disconnected w.r.t terminals (or other failure).
        cout << 0 << " " << 0 << "\n";
        return 0;
    }

    cout << totalW << " " << answer.size() << "\n";
    for (const auto& e : answer) {
        cout << e.u << " " << e.v << "\n";
    }
    return 0;
}
