#!/usr/bin/env python3
import sys
import math
import random
from typing import List, Optional, Tuple

def read_input() -> Tuple[int, int, int, Tuple[float, float], List[Tuple[float, float]], List[int]]:
    data = sys.stdin.read().strip().split()
    if not data:
        raise SystemExit("Empty input")

    it = iter(data)
    N = int(next(it))
    K = int(next(it))
    CAP = int(next(it))

    depot = (float(next(it)), float(next(it)))
    coords = []
    demands = []
    for _ in range(N):
        x = float(next(it)); y = float(next(it)); d = int(next(it))
        coords.append((x, y))
        demands.append(d)
    return N, K, CAP, depot, coords, demands


class Dist:
    """Distance provider with optional full matrix precompute for small/medium N."""
    def __init__(self, depot, coords):
        self.N = len(coords)
        self.depot_id = self.N
        self.coords = coords + [depot]  # depot as last index
        self.mat = None

        # Precompute if feasible.
        # N=2000 -> ~4M doubles -> ~32MB; OK. Larger can be too big.
        if self.N <= 2000:
            n = self.N + 1
            mat = [[0.0] * n for _ in range(n)]
            for i in range(n):
                xi, yi = self.coords[i]
                row = mat[i]
                for j in range(i + 1, n):
                    xj, yj = self.coords[j]
                    d = math.hypot(xi - xj, yi - yj)
                    row[j] = d
                    mat[j][i] = d
            self.mat = mat

    def d(self, a: int, b: int) -> float:
        if self.mat is not None:
            return self.mat[a][b]
        xa, ya = self.coords[a]
        xb, yb = self.coords[b]
        return math.hypot(xa - xb, ya - yb)


def route_distance(route: List[int], dist: Dist) -> float:
    dep = dist.depot_id
    if not route:
        return 0.0
    total = dist.d(dep, route[0])
    for i in range(len(route) - 1):
        total += dist.d(route[i], route[i + 1])
    total += dist.d(route[-1], dep)
    return total


def nearest_neighbor_order(nodes: List[int], dist: Dist) -> List[int]:
    """Build a visit order by nearest-neighbor starting from depot."""
    if len(nodes) <= 1:
        return nodes[:]
    remaining = set(nodes)
    dep = dist.depot_id
    curr = dep
    out = []
    while remaining:
        # find closest to curr
        best = None
        bestd = float("inf")
        for v in remaining:
            dv = dist.d(curr, v)
            if dv < bestd:
                bestd = dv
                best = v
        out.append(best)
        remaining.remove(best)
        curr = best
    return out


def two_opt(route: List[int], dist: Dist, max_passes: int = 3) -> List[int]:
    """Simple 2-opt (first improvement), capped passes; skips for large routes."""
    n = len(route)
    if n < 4:
        return route
    if n > 300:
        return route  # avoid O(n^2) on very large routes

    dep = dist.depot_id
    r = route[:]

    def node_at(i):
        if i == -1:
            return dep
        if i == n:
            return dep
        return r[i]

    for _ in range(max_passes):
        improved = False
        # edges are (i-1,i) and (k,k+1), reverse i..k
        for i in range(1, n - 1):
            a = node_at(i - 1)
            b = node_at(i)
            dab = dist.d(a, b)
            for k in range(i + 1, n):
                c = node_at(k)
                d = node_at(k + 1)
                # current: (a-b) + (c-d); new: (a-c) + (b-d)
                delta = (dist.d(a, c) + dist.d(b, d)) - (dab + dist.d(c, d))
                if delta < -1e-12:
                    r[i:k + 1] = reversed(r[i:k + 1])
                    improved = True
                    break
            if improved:
                break
        if not improved:
            break
    return r


def best_fit_partition(order: List[int], K: int, CAP: int, demands: List[int]) -> Optional[List[List[int]]]:
    """Best-fit decreasing packing into <=K bins by demand."""
    routes: List[List[int]] = []
    rem: List[int] = []

    for c in order:
        dem = demands[c]
        best_idx = -1
        best_rem_after = None
        for i in range(len(routes)):
            if rem[i] >= dem:
                after = rem[i] - dem
                if best_rem_after is None or after < best_rem_after:
                    best_rem_after = after
                    best_idx = i

        if best_idx >= 0:
            routes[best_idx].append(c)
            rem[best_idx] -= dem
        else:
            if len(routes) >= K:
                return None
            routes.append([c])
            rem.append(CAP - dem)
            if rem[-1] < 0:
                return None
    return routes


def construct_initial_solution(N: int, K: int, CAP: int, demands: List[int], dist: Dist) -> List[List[int]]:
    customers = list(range(N))

    # Multi-start packing: reorder by demand desc with shuffled ties.
    best_routes = None
    best_score = float("inf")

    # quick score: sum depot->node->depot for packed nodes, + intra NN approximation later
    def quick_score(routes_raw: List[List[int]]) -> float:
        dep = dist.depot_id
        s = 0.0
        for rr in routes_raw:
            if not rr:
                continue
            for v in rr:
                s += 2.0 * dist.d(dep, v)
        # slight penalty for more routes
        s += 1000.0 * len(routes_raw)
        return s

    base = sorted(customers, key=lambda c: (-demands[c], c))
    rng = random.Random(0)

    for attempt in range(80):
        # shuffle within equal-demand blocks
        order = base[:]
        i = 0
        while i < N:
            j = i + 1
            while j < N and demands[order[j]] == demands[order[i]]:
                j += 1
            if j - i > 1:
                block = order[i:j]
                rng.shuffle(block)
                order[i:j] = block
            i = j
        # small random perturbation
        if N > 2:
            for _ in range(min(10, N // 10 + 1)):
                a = rng.randrange(N)
                b = rng.randrange(N)
                order[a], order[b] = order[b], order[a]

        routes_raw = best_fit_partition(order, K, CAP, demands)
        if routes_raw is None:
            rng.seed(attempt + 1)
            continue

        sc = quick_score(routes_raw)
        if sc < best_score:
            best_score = sc
            best_routes = routes_raw

        rng.seed(attempt + 1)

    if best_routes is None:
        # final deterministic try
        routes_raw = best_fit_partition(base, K, CAP, demands)
        if routes_raw is None:
            raise SystemExit("No feasible packing found with given K and CAPACITY (may be infeasible).")
        best_routes = routes_raw

    # Build tours in each route
    routes: List[List[int]] = []
    for rr in best_routes:
        seq = nearest_neighbor_order(rr, dist)
        seq = two_opt(seq, dist)
        routes.append(seq)

    return routes


def improve_by_relocate(routes: List[List[int]], K: int, CAP: int, demands: List[int], dist: Dist, passes: int = 3) -> List[List[int]]:
    """Inter-route relocate moves for distance reduction while keeping capacity."""
    # capacities
    loads = [sum(demands[c] for c in r) for r in routes]
    rem = [CAP - l for l in loads]

    dep = dist.depot_id

    def removal_saving(route: List[int], idx: int) -> float:
        """Positive saving (distance decreases) if removing this node."""
        node = route[idx]
        prev = dep if idx == 0 else route[idx - 1]
        nxt = dep if idx == len(route) - 1 else route[idx + 1]
        return dist.d(prev, node) + dist.d(node, nxt) - dist.d(prev, nxt)

    def best_insertion_increase(route: List[int], node: int) -> Tuple[float, int]:
        """Return (min increase, position to insert). position in [0..len(route)]."""
        m = len(route)
        if m == 0:
            return 2.0 * dist.d(dep, node), 0
        best_inc = float("inf")
        best_pos = 0
        # check all positions for moderate routes, otherwise sample a few
        positions = range(m + 1) if m <= 200 else (
            [0, m] + [random.randrange(m + 1) for _ in range(20)]
        )
        for pos in positions:
            prev = dep if pos == 0 else route[pos - 1]
            nxt = dep if pos == m else route[pos]
            inc = dist.d(prev, node) + dist.d(node, nxt) - dist.d(prev, nxt)
            if inc < best_inc:
                best_inc = inc
                best_pos = pos
        return best_inc, best_pos

    for _ in range(passes):
        improved = False
        for i in range(len(routes)):
            if not routes[i]:
                continue
            a = routes[i]
            # iterate over a copy of nodes to keep indices manageable after edits
            idx = 0
            while idx < len(a):
                node = a[idx]
                dem = demands[node]
                save = removal_saving(a, idx)

                best_delta = 0.0
                best_j = None
                best_pos = None

                for j in range(len(routes)):
                    if j == i:
                        continue
                    if rem[j] < dem:
                        continue
                    inc, pos = best_insertion_increase(routes[j], node)
                    delta = -save + inc  # new_total - old_total
                    if delta < best_delta - 1e-12:
                        best_delta = delta
                        best_j = j
                        best_pos = pos

                if best_j is not None:
                    # apply move i -> best_j
                    a.pop(idx)
                    rem[i] += dem
                    loads[i] -= dem

                    routes[best_j].insert(best_pos, node)
                    rem[best_j] -= dem
                    loads[best_j] += dem

                    improved = True
                    # do not increment idx (new node at idx needs checking)
                else:
                    idx += 1

        # clean empty routes (still <=K)
        routes = [r for r in routes if r]
        loads = [sum(demands[c] for c in r) for r in routes]
        rem = [CAP - l for l in loads]

        if not improved:
            break

    # light intra-route 2-opt again
    for t in range(len(routes)):
        routes[t] = two_opt(routes[t], dist, max_passes=2)
    return routes


def main():
    N, K, CAP, depot, coords, demands = read_input()
    dist = Dist(depot, coords)

    routes = construct_initial_solution(N, K, CAP, demands, dist)
    routes = improve_by_relocate(routes, K, CAP, demands, dist, passes=3)

    # Validate coverage + capacity
    seen = [0] * N
    for r in routes:
        load = sum(demands[c] for c in r)
        if load > CAP:
            raise SystemExit("Internal error: capacity violated")
        for c in r:
            if not (0 <= c < N):
                raise SystemExit("Internal error: invalid customer index")
            seen[c] += 1
    if any(v != 1 for v in seen):
        # As a last-resort repair (should rarely happen): put missing customers into new routes if possible.
        missing = [i for i, v in enumerate(seen) if v == 0]
        if missing:
            # try insert missing into existing routes
            loads = [sum(demands[c] for c in r) for r in routes]
            for c in missing:
                placed = False
                for i in range(len(routes)):
                    if loads[i] + demands[c] <= CAP:
                        routes[i].append(c)
                        loads[i] += demands[c]
                        placed = True
                        break
                if not placed:
                    if len(routes) < K and demands[c] <= CAP:
                        routes.append([c])
                        loads.append(demands[c])
                        placed = True
                if not placed:
                    raise SystemExit("Could not repair to a valid solution (likely infeasible).")
            # re-opt routes a bit
            for i in range(len(routes)):
                routes[i] = nearest_neighbor_order(routes[i], dist)
                routes[i] = two_opt(routes[i], dist)

        # check duplicates
        counts = [0] * N
        for r in routes:
            for c in r:
                counts[c] += 1
        if any(v != 1 for v in counts):
            raise SystemExit("Internal error: customers not covered exactly once")

    total = sum(route_distance(r, dist) for r in routes)

    out_lines = []
    out_lines.append(f"{total:.6f}")
    out_lines.append(str(len(routes)))
    for r in routes:
        out_lines.append(" ".join(map(str, r)))
    sys.stdout.write("\n".join(out_lines) + "\n")


if __name__ == "__main__":
    main()
