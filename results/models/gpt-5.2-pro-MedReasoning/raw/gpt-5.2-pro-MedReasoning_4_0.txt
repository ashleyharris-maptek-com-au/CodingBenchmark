#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <unordered_map>
#include <chrono>
#include <iomanip>
#include <cstdint>
#include <limits>

using std::vector;

struct Vec3 {
    double x=0, y=0, z=0;
    Vec3() = default;
    Vec3(double X,double Y,double Z):x(X),y(Y),z(Z){}
    Vec3 operator+(const Vec3& o) const { return {x+o.x,y+o.y,z+o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x-o.x,y-o.y,z-o.z}; }
    Vec3 operator*(double s) const { return {x*s,y*s,z*s}; }
    Vec3 operator/(double s) const { return {x/s,y/s,z/s}; }
};

static inline double dot(const Vec3& a, const Vec3& b){ return a.x*b.x + a.y*b.y + a.z*b.z; }
static inline Vec3 cross(const Vec3& a, const Vec3& b){
    return {a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x};
}
static inline double norm2(const Vec3& a){ return dot(a,a); }
static inline double norm(const Vec3& a){ return std::sqrt(norm2(a)); }

struct Plane {
    Vec3 n; // unit normal, outward
    double d; // n·p + d = 0
    double dist(const Vec3& p) const { return dot(n,p) + d; }
};

struct Face {
    int a=-1,b=-1,c=-1;
    Plane pl;
    vector<int> outside; // points assigned to this face (indices)
    bool removed=false;
};

static inline uint64_t edgeKeyUndir(int u, int v){
    uint32_t a = (uint32_t)std::min(u,v);
    uint32_t b = (uint32_t)std::max(u,v);
    return (uint64_t(a) << 32) | uint64_t(b);
}

static Plane makePlaneOriented(const Vec3& A, const Vec3& B, const Vec3& C, const Vec3& interior)
{
    Vec3 n = cross(B-A, C-A);
    double len = norm(n);
    if (len == 0) return {{0,0,0}, 0};
    n = n / len;
    double d = -dot(n, A);
    // Ensure interior is on non-positive side (inside): n·interior + d <= 0
    if (dot(n, interior) + d > 0) {
        n = n * -1.0;
        d = -d;
    }
    return {n, d};
}

static Face makeFace(int a,int b,int c,const Vec3& interior,const vector<Vec3>& pts){
    Face f;
    f.a=a; f.b=b; f.c=c;
    f.pl = makePlaneOriented(pts[a], pts[b], pts[c], interior);
    return f;
}

static double distPointLine(const Vec3& p, const Vec3& a, const Vec3& b){
    Vec3 ab = b-a;
    double ab2 = norm2(ab);
    if (ab2 == 0) return norm(p-a);
    return norm(cross(p-a, ab)) / std::sqrt(ab2);
}

static double distPointPlaneAbs(const Vec3& p, const Vec3& a, const Vec3& b, const Vec3& c){
    Vec3 n = cross(b-a, c-a);
    double len = norm(n);
    if (len == 0) return 0.0;
    return std::abs(dot(n, p-a)) / len;
}

// Quickhull (double, epsilon-based). Assumes non-degenerate 3D point set for a meaningful hull.
static bool buildConvexHullPlanesQuickhull(const vector<Vec3>& pts, vector<Plane>& outPlanes)
{
    outPlanes.clear();
    const int n = (int)pts.size();
    if (n < 4) return false;

    Vec3 bbMin = pts[0], bbMax = pts[0];
    for (auto &p: pts){
        bbMin.x = std::min(bbMin.x, p.x); bbMin.y = std::min(bbMin.y, p.y); bbMin.z = std::min(bbMin.z, p.z);
        bbMax.x = std::max(bbMax.x, p.x); bbMax.y = std::max(bbMax.y, p.y); bbMax.z = std::max(bbMax.z, p.z);
    }
    double diag = norm(bbMax - bbMin);
    double eps = 1e-12 * std::max(1.0, diag);

    // 1) initial edge by extremes in x
    int i0 = 0, i1 = 0;
    for (int i=1;i<n;i++){
        if (pts[i].x < pts[i0].x) i0=i;
        if (pts[i].x > pts[i1].x) i1=i;
    }
    if (i0 == i1) return false;

    // 2) farthest from line
    int i2 = -1;
    double best = -1;
    for (int i=0;i<n;i++){
        if (i==i0 || i==i1) continue;
        double d = distPointLine(pts[i], pts[i0], pts[i1]);
        if (d > best){ best=d; i2=i; }
    }
    if (i2 < 0 || best <= eps) return false; // collinear

    // 3) farthest from plane
    int i3 = -1;
    best = -1;
    for (int i=0;i<n;i++){
        if (i==i0 || i==i1 || i==i2) continue;
        double d = distPointPlaneAbs(pts[i], pts[i0], pts[i1], pts[i2]);
        if (d > best){ best=d; i3=i; }
    }
    if (i3 < 0 || best <= eps) return false; // coplanar

    Vec3 interior = (pts[i0] + pts[i1] + pts[i2] + pts[i3]) / 4.0;

    vector<Face> faces;
    faces.reserve(2*n);

    // Initial tetra faces (makeFace orients by interior point)
    faces.push_back(makeFace(i0,i1,i2, interior, pts));
    faces.push_back(makeFace(i0,i3,i1, interior, pts));
    faces.push_back(makeFace(i0,i2,i3, interior, pts));
    faces.push_back(makeFace(i1,i3,i2, interior, pts));

    auto isInit = [&](int idx){
        return idx==i0 || idx==i1 || idx==i2 || idx==i3;
    };

    // Assign points to faces
    for (int i=0;i<n;i++){
        if (isInit(i)) continue;
        int bestF = -1;
        double bestDist = eps;
        for (int fi=0; fi<(int)faces.size(); fi++){
            double d = faces[fi].pl.dist(pts[i]);
            if (d > bestDist){
                bestDist = d;
                bestF = fi;
            }
        }
        if (bestF != -1) faces[bestF].outside.push_back(i);
    }

    // Main loop
    for (int iter=0; iter<200000; iter++){
        // Find a face with outside points and the farthest point
        int seedFace = -1;
        int farPt = -1;
        double farDist = eps;

        for (int fi=0; fi<(int)faces.size(); fi++){
            if (faces[fi].removed || faces[fi].outside.empty()) continue;
            // Find farthest among assigned points
            int bestIdx = -1;
            double bestD = eps;
            for (int pi : faces[fi].outside){
                double d = faces[fi].pl.dist(pts[pi]);
                if (d > bestD){
                    bestD = d;
                    bestIdx = pi;
                }
            }
            if (bestIdx != -1 && bestD > farDist){
                farDist = bestD;
                farPt = bestIdx;
                seedFace = fi;
            }
        }

        if (seedFace == -1) break; // done

        // Build edge map for current non-removed faces
        std::unordered_map<uint64_t, vector<int>> edgeMap;
        edgeMap.reserve((size_t)faces.size()*3);

        for (int fi=0; fi<(int)faces.size(); fi++){
            if (faces[fi].removed) continue;
            const Face &f = faces[fi];
            edgeMap[edgeKeyUndir(f.a,f.b)].push_back(fi);
            edgeMap[edgeKeyUndir(f.b,f.c)].push_back(fi);
            edgeMap[edgeKeyUndir(f.c,f.a)].push_back(fi);
        }

        auto neighborAcross = [&](int fi, int u, int v)->int{
            auto it = edgeMap.find(edgeKeyUndir(u,v));
            if (it == edgeMap.end()) return -1;
            const auto& vec = it->second;
            if (vec.size() == 1) return -1;
            if (vec[0] == fi) return vec[1];
            if (vec[1] == fi) return vec[0];
            return -1;
        };

        // Visible faces BFS
        vector<char> vis(faces.size(), 0);
        vector<int> stack;
        stack.push_back(seedFace);
        vis[seedFace] = 1;

        for (size_t si=0; si<stack.size(); si++){
            int fi = stack[si];
            const Face& f = faces[fi];
            // If seedFace is visible, it is; for neighbors check visibility
            // Explore neighbors across each edge
            int ea[3][2] = {{f.a,f.b},{f.b,f.c},{f.c,f.a}};
            for (int e=0;e<3;e++){
                int u=ea[e][0], v=ea[e][1];
                int nb = neighborAcross(fi, u, v);
                if (nb < 0) continue;
                if (faces[nb].removed) continue;
                if (vis[nb]) continue;
                if (faces[nb].pl.dist(pts[farPt]) > eps){
                    vis[nb] = 1;
                    stack.push_back(nb);
                }
            }
        }

        // Collect horizon edges and points to reassign, remove visible faces
        vector<std::pair<int,int>> horizon;
        horizon.reserve(stack.size()*2);
        vector<int> reassign;
        reassign.reserve(1024);

        for (int fi=0; fi<(int)faces.size(); fi++){
            if (!vis[fi]) continue;
            if (faces[fi].removed) continue;

            // collect outside points
            for (int pi : faces[fi].outside) reassign.push_back(pi);

            const Face& f = faces[fi];
            int ed[3][2] = {{f.a,f.b},{f.b,f.c},{f.c,f.a}};
            for (int e=0;e<3;e++){
                int u=ed[e][0], v=ed[e][1];
                int nb = neighborAcross(fi,u,v);
                if (nb < 0 || !vis[nb]) {
                    // Edge on horizon, keep directed as in this visible face
                    horizon.push_back({u,v});
                }
            }
            faces[fi].removed = true;
            faces[fi].outside.clear();
        }

        if (horizon.size() < 3) break;

        // Order horizon edges into a cycle by following u->v links
        vector<char> used(horizon.size(), 0);
        std::unordered_map<int, vector<int>> from;
        from.reserve(horizon.size()*2);
        for (int i=0;i<(int)horizon.size();i++){
            from[horizon[i].first].push_back(i);
        }

        vector<std::pair<int,int>> ordered;
        ordered.reserve(horizon.size());

        // Start with first edge, then follow
        int startU = horizon[0].first;
        int curU = horizon[0].first;
        int curV = horizon[0].second;
        used[0] = 1;
        ordered.push_back(horizon[0]);

        for (size_t step=1; step<horizon.size(); step++){
            auto it = from.find(curV);
            int nextIdx = -1;
            if (it != from.end()){
                auto &lst = it->second;
                while (!lst.empty()){
                    int cand = lst.back();
                    lst.pop_back();
                    if (!used[cand]){
                        nextIdx = cand;
                        break;
                    }
                }
            }
            if (nextIdx == -1) break;
            used[nextIdx] = 1;
            ordered.push_back(horizon[nextIdx]);
            curU = horizon[nextIdx].first;
            curV = horizon[nextIdx].second;
            if (curV == startU) {
                // cycle closed; still may have unused due to numeric issues; ignore extras
                break;
            }
        }

        if (ordered.size() < 3) {
            // Fallback: use unsorted horizon list
            ordered = horizon;
        }

        // Create new faces around farPt
        int baseNew = (int)faces.size();
        vector<int> newFaceIdx;
        newFaceIdx.reserve(ordered.size());

        for (auto [u,v] : ordered){
            Face nf = makeFace(u,v,farPt, interior, pts);
            faces.push_back(std::move(nf));
            newFaceIdx.push_back((int)faces.size()-1);
        }

        // Reassign points from removed faces to new faces
        // (Only those points can lie outside new hull)
        for (int pi : reassign){
            if (pi == farPt) continue;
            int bestF = -1;
            double bestD = eps;
            for (int fi : newFaceIdx){
                double d = faces[fi].pl.dist(pts[pi]);
                if (d > bestD){
                    bestD = d;
                    bestF = fi;
                }
            }
            if (bestF != -1) faces[bestF].outside.push_back(pi);
        }

        // continue
    }

    // Extract planes from remaining faces
    vector<Plane> planes;
    planes.reserve(faces.size());
    for (auto &f: faces){
        if (f.removed) continue;
        // Validate plane normal
        double ln = norm(f.pl.n);
        if (ln == 0) continue;
        Plane pl = f.pl;
        pl.n = pl.n / ln;
        pl.d = pl.d / ln;
        planes.push_back(pl);
    }
    if (planes.size() < 4) return false;

    outPlanes = std::move(planes);
    return true;
}

// Euler rotation: Rz(az)*Ry(ay)*Rx(ax)
static inline Vec3 rotateEuler(const Vec3& p, double rx, double ry, double rz){
    double cx=std::cos(rx), sx=std::sin(rx);
    double cy=std::cos(ry), sy=std::sin(ry);
    double cz=std::cos(rz), sz=std::sin(rz);

    // Rx
    Vec3 q1{p.x, cx*p.y - sx*p.z, sx*p.y + cx*p.z};
    // Ry
    Vec3 q2{cy*q1.x + sy*q1.z, q1.y, -sy*q1.x + cy*q1.z};
    // Rz
    Vec3 q3{cz*q2.x - sz*q2.y, sz*q2.x + cz*q2.y, q2.z};
    return q3;
}

struct Placement {
    Vec3 c;
    double rx, ry, rz;
};

static inline bool pointInsideConvex(const Vec3& p, const vector<Plane>& planes, double eps){
    for (auto &pl : planes){
        if (pl.dist(p) > eps) return false;
    }
    return true;
}

static inline bool tetraFitsAt(
    const Vec3& center,
    double rx, double ry, double rz,
    const vector<Vec3>& localVerts,
    const vector<Plane>& planes,
    double R, double eps)
{
    // Fast necessary rejection: if center is farther than R outside any plane, tetra cannot fit.
    // Fast acceptance: if center is at least R inside all planes, circumsphere fits => tetra fits.
    bool sphereGuaranteedInside = true;
    for (const auto& pl : planes){
        double d = pl.dist(center);
        if (d > R + eps) return false;
        if (d > -R) sphereGuaranteedInside = false;
    }
    if (sphereGuaranteedInside) return true;

    // Exact vertex test
    for (const auto& lv : localVerts){
        Vec3 gv = center + rotateEuler(lv, rx, ry, rz);
        if (!pointInsideConvex(gv, planes, eps)) return false;
    }
    return true;
}

int main(){
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int V;
    double e;
    if (!(std::cin >> V >> e)) return 0;
    vector<Vec3> pts;
    pts.reserve(std::max(0, V));
    for (int i=0;i<V;i++){
        double x,y,z;
        std::cin >> x >> y >> z;
        pts.emplace_back(x,y,z);
    }
    if (V < 4 || e <= 0){
        std::cout << 0 << "\n";
        return 0;
    }

    // Build convex hull planes (container assumed to be convex hull of given vertices).
    vector<Plane> planes;
    if (!buildConvexHullPlanesQuickhull(pts, planes)) {
        std::cout << 0 << "\n";
        return 0;
    }

    // Bounding box
    Vec3 bbMin = pts[0], bbMax = pts[0];
    for (auto &p: pts){
        bbMin.x = std::min(bbMin.x, p.x); bbMin.y = std::min(bbMin.y, p.y); bbMin.z = std::min(bbMin.z, p.z);
        bbMax.x = std::max(bbMax.x, p.x); bbMax.y = std::max(bbMax.y, p.y); bbMax.z = std::max(bbMax.z, p.z);
    }
    double diag = norm(bbMax - bbMin);
    double eps = 1e-10 * std::max(1.0, diag);

    // Tetra geometry (centered at origin)
    // Base vertices: (±1,±1,±1) with an even number of minus signs -> regular tetra
    // Scaled so edge length is e.
    const double scale = e / (2.0 * std::sqrt(2.0));
    vector<Vec3> localVerts = {
        Vec3( 1, 1, 1) * scale,
        Vec3( 1,-1,-1) * scale,
        Vec3(-1, 1,-1) * scale,
        Vec3(-1,-1, 1) * scale
    };

    // Circumradius
    const double R = e * std::sqrt(3.0/8.0);

    // Sphere packing spacing (FCC). Nearest neighbor distance = 2R.
    const double sphereDist = 2.0 * R * 1.000001; // tiny separation margin
    double cell = sphereDist * std::sqrt(2.0);     // FCC conventional cell edge

    // Estimate candidates from bounding box
    auto estimateCandidates = [&](double cellEdge)->long double{
        long double nx = std::floor((bbMax.x - bbMin.x) / cellEdge) + 4;
        long double ny = std::floor((bbMax.y - bbMin.y) / cellEdge) + 4;
        long double nz = std::floor((bbMax.z - bbMin.z) / cellEdge) + 4;
        if (nx < 0) nx = 0; if (ny < 0) ny = 0; if (nz < 0) nz = 0;
        return 4.0L * nx * ny * nz;
    };

    const long double TARGET_CAND = 2'000'000.0L;  // keep runtime bounded
    long double cand0 = estimateCandidates(cell);
    if (cand0 > TARGET_CAND) {
        long double factor = std::cbrt((double)(cand0 / TARGET_CAND));
        cell *= (double)factor;
    }

    // Multiple origin shifts (choose best) only when candidate count is manageable
    vector<Vec3> shifts;
    shifts.push_back({0,0,0});
    if (estimateCandidates(cell) <= 400000.0L) {
        shifts.push_back({cell*0.5, 0, 0});
        shifts.push_back({0, cell*0.5, 0});
        shifts.push_back({0, 0, cell*0.5});
    }

    // A small orientation set (symmetry makes many equivalent, but helps near boundaries)
    const double PI = 3.1415926535897932384626433832795;
    vector<std::array<double,3>> orientations = {
        {0,0,0},
        {0, PI*0.5, 0},
        {PI*0.5, 0, 0},
        {0, 0, PI*0.5},
        {PI*0.5, PI*0.5, 0},
        {PI*0.5, 0, PI*0.5}
    };
    if (estimateCandidates(cell) > 1'000'000.0L) {
        // For huge instances, reduce per-candidate work.
        orientations = { {0,0,0} };
    }

    const auto tStart = std::chrono::steady_clock::now();
    auto timeExceeded = [&]()->bool{
        using namespace std::chrono;
        double s = duration<double>(steady_clock::now() - tStart).count();
        return s > 29.0;
    };

    const int MAXK = 300000; // prevents runaway output/memory in ludicrous containers

    auto runPass = [&](const Vec3& shift)->vector<Placement>{
        vector<Placement> res;
        res.reserve(4096);

        Vec3 origin = bbMin - Vec3(cell,cell,cell) + shift;
        Vec3 extent = (bbMax - bbMin) + Vec3(2*cell,2*cell,2*cell);

        auto safeFloor = [](double x)->int64_t{
            double f = std::floor(x);
            if (f < (double)std::numeric_limits<int64_t>::min()) return std::numeric_limits<int64_t>::min();
            if (f > (double)std::numeric_limits<int64_t>::max()) return std::numeric_limits<int64_t>::max();
            return (int64_t)f;
        };

        int64_t nx = safeFloor(extent.x / cell) + 1;
        int64_t ny = safeFloor(extent.y / cell) + 1;
        int64_t nz = safeFloor(extent.z / cell) + 1;
        if (nx < 0) nx = 0; if (ny < 0) ny = 0; if (nz < 0) nz = 0;

        Vec3 offsets[4] = {
            {0,0,0},
            {0, cell*0.5, cell*0.5},
            {cell*0.5, 0, cell*0.5},
            {cell*0.5, cell*0.5, 0}
        };

        for (int64_t i=0;i<nx;i++){
            if (timeExceeded() || (int)res.size() >= MAXK) break;
            double x0 = origin.x + (double)i * cell;
            for (int64_t j=0;j<ny;j++){
                if (timeExceeded() || (int)res.size() >= MAXK) break;
                double y0 = origin.y + (double)j * cell;
                for (int64_t k=0;k<nz;k++){
                    if (timeExceeded() || (int)res.size() >= MAXK) break;
                    double z0 = origin.z + (double)k * cell;

                    Vec3 base{x0,y0,z0};
                    for (int b=0;b<4;b++){
                        Vec3 c = base + offsets[b];

                        // quick reject: if center isn't even inside expanded region, skip
                        // (No need for bbox check; plane checks handle it)
                        bool placed = false;
                        for (auto &ori : orientations){
                            double rx = ori[0], ry = ori[1], rz = ori[2];
                            if (tetraFitsAt(c, rx, ry, rz, localVerts, planes, R, eps)) {
                                res.push_back({c, rx, ry, rz});
                                placed = true;
                                break;
                            }
                        }
                        (void)placed;
                        if ((int)res.size() >= MAXK) break;
                    }
                }
            }
        }
        return res;
    };

    vector<Placement> best;
    best.reserve(4096);
    for (const auto& sh : shifts){
        if (timeExceeded()) break;
        vector<Placement> cur = runPass(sh);
        if (cur.size() > best.size()) best = std::move(cur);
        if ((int)best.size() >= MAXK) break;
    }

    std::cout << best.size() << "\n";
    std::cout << std::setprecision(10) << std::fixed;
    for (auto &p : best){
        std::cout << p.c.x << " " << p.c.y << " " << p.c.z << " "
                  << p.rx << " " << p.ry << " " << p.rz << "\n";
    }
    return 0;
}
