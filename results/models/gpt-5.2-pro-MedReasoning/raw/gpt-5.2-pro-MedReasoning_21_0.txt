#include <iostream>
#include <vector>
#include <array>
#include <deque>
#include <unordered_set>
#include <unordered_map>
#include <queue>
#include <limits>
#include <cstdint>
#include <cmath>
#include <algorithm>

using namespace std;

static int gD = 0;
static const int MAXD = 25;

struct Pos {
    array<int, MAXD> x{};

    bool operator==(const Pos& other) const noexcept {
        for (int i = 0; i < gD; ++i) if (x[i] != other.x[i]) return false;
        return true;
    }
};

static inline uint64_t splitmix64(uint64_t v) {
    v += 0x9e3779b97f4a7c15ull;
    v = (v ^ (v >> 30)) * 0xbf58476d1ce4e5b9ull;
    v = (v ^ (v >> 27)) * 0x94d049bb133111ebull;
    return v ^ (v >> 31);
}

struct PosHash {
    size_t operator()(Pos const& p) const noexcept {
        uint64_t h = 0x123456789abcdef0ull;
        for (int i = 0; i < gD; ++i) {
            uint64_t v = (uint64_t)(uint32_t)(p.x[i] + 1000003); // shift to reduce negatives if any
            h = splitmix64(h ^ splitmix64(v + 0x9e3779b97f4a7c15ull * (uint64_t)(i + 1)));
        }
        return (size_t)h;
    }
};

static inline long long manhattan(const Pos& a, const Pos& b) {
    long long s = 0;
    for (int i = 0; i < gD; ++i) s += llabs((long long)a.x[i] - (long long)b.x[i]);
    return s;
}

struct Move { int axis = 0; int dir = 1; };

struct Game {
    vector<int> bounds;
    long long maxTurns = 0;

    deque<Pos> body; // front=head, back=tail
    unordered_set<Pos, PosHash> occupied;

    unordered_set<Pos, PosHash> food;
    unordered_set<Pos, PosHash> obstacles;

    Pos head() const { return body.front(); }
    Pos tail() const { return body.back(); }

    bool inBounds(const Pos& p) const {
        for (int i = 0; i < gD; ++i) {
            if (p.x[i] < 0 || p.x[i] >= bounds[i]) return false;
        }
        return true;
    }

    // Safety check for *next immediate move* from current state.
    bool isSafeNext(const Pos& np) const {
        if (!inBounds(np)) return false;
        if (obstacles.find(np) != obstacles.end()) return false;

        bool grows = (food.find(np) != food.end());
        if (occupied.find(np) != occupied.end()) {
            // allow stepping onto the tail only if we are not growing (tail will move away)
            if (!grows && np == tail()) return true;
            return false;
        }
        return true;
    }

    // Count safe neighbors if head were at 'p' (approx, uses current occupied/tail rule with p as next).
    int safeNeighborCount(const Pos& p) const {
        int cnt = 0;
        for (int ax = 0; ax < gD; ++ax) {
            for (int dir : {-1, +1}) {
                Pos q = p;
                q.x[ax] += dir;
                // approximate: check as a normal "next" move from current state (good enough as heuristic)
                if (isSafeNext(q)) ++cnt;
            }
        }
        return cnt;
    }

    Pos applyMove(const Move& m, bool& ok, bool& ateFood) {
        ok = true; ateFood = false;
        Pos nh = head();
        nh.x[m.axis] += m.dir;

        if (!inBounds(nh) || obstacles.find(nh) != obstacles.end()) {
            ok = false; return nh;
        }

        bool grows = (food.find(nh) != food.end());
        // collision check with current occupied, allowing moving into tail if not growing
        if (occupied.find(nh) != occupied.end()) {
            if (!( !grows && nh == tail() )) {
                ok = false; return nh;
            }
        }

        // commit move
        body.push_front(nh);
        occupied.insert(nh);

        if (grows) {
            ateFood = true;
            food.erase(nh);
        } else {
            Pos t = body.back();
            body.pop_back();
            occupied.erase(t);
        }
        return nh;
    }

    Pos pickNearestFood() const {
        // assumes food not empty
        Pos h = head();
        auto it = food.begin();
        Pos best = *it;
        long long bestD = manhattan(h, best);
        ++it;
        for (; it != food.end(); ++it) {
            long long d = manhattan(h, *it);
            if (d < bestD) { bestD = d; best = *it; }
        }
        return best;
    }

    Move greedyToward(const Pos& target, bool& found) const {
        found = false;
        Pos h = head();
        long long curD = manhattan(h, target);

        Move bestMove;
        long long bestDist = (long long)4e18;
        int bestDeg = -1;

        // try moves that strictly reduce manhattan first
        for (int ax = 0; ax < gD; ++ax) {
            int delta = target.x[ax] - h.x[ax];
            if (delta == 0) continue;
            int dir = (delta > 0 ? +1 : -1);
            Pos np = h;
            np.x[ax] += dir;
            if (!isSafeNext(np)) continue;

            long long nd = manhattan(np, target);
            if (nd > curD) continue; // prefer not to increase distance
            int deg = safeNeighborCount(np);

            if (!found || nd < bestDist || (nd == bestDist && deg > bestDeg)) {
                found = true;
                bestDist = nd;
                bestDeg = deg;
                bestMove = {ax, dir};
            }
        }
        if (found) return bestMove;

        // otherwise, allow any safe move as fallback (still greedy-ish: minimize distance)
        for (int ax = 0; ax < gD; ++ax) {
            for (int dir : {-1, +1}) {
                Pos np = h;
                np.x[ax] += dir;
                if (!isSafeNext(np)) continue;
                long long nd = manhattan(np, target);
                int deg = safeNeighborCount(np);
                if (!found || nd < bestDist || (nd == bestDist && deg > bestDeg)) {
                    found = true;
                    bestDist = nd;
                    bestDeg = deg;
                    bestMove = {ax, dir};
                }
            }
        }
        return bestMove;
    }

    // Local A* over positions only (not full snake state), limited expansions.
    // Goal: reach target OR reach any state closer than start.
    bool localAStar(const Pos& target, Move& outMove,
                    int depthLimit = 10, int nodeLimit = 120000) const
    {
        const Pos start = head();
        const long long startDist = manhattan(start, target);

        struct Node {
            Pos p;
            int g;
            long long f;
            int firstAxis;
            int firstDir;
        };
        struct Cmp {
            bool operator()(Node const& a, Node const& b) const { return a.f > b.f; }
        };

        priority_queue<Node, vector<Node>, Cmp> pq;
        unordered_map<Pos, int, PosHash> bestG;
        bestG.reserve((size_t)nodeLimit * 2);

        auto pushNode = [&](const Pos& p, int g, int firstAxis, int firstDir) {
            long long h = manhattan(p, target);
            Node n{p, g, (long long)g + h, firstAxis, firstDir};
            pq.push(n);
        };

        // seed with immediate moves (so we can return first move without parent maps)
        for (int ax = 0; ax < gD; ++ax) {
            for (int dir : {-1, +1}) {
                Pos np = start;
                np.x[ax] += dir;
                if (!isSafeNext(np)) continue;
                pushNode(np, 1, ax, dir);
                bestG[np] = 1;
            }
        }

        int expansions = 0;
        while (!pq.empty() && expansions < nodeLimit) {
            Node cur = pq.top(); pq.pop();
            ++expansions;

            auto it = bestG.find(cur.p);
            if (it != bestG.end() && cur.g != it->second) continue;

            long long d = manhattan(cur.p, target);
            if (cur.p == target || d < startDist) {
                outMove = {cur.firstAxis, cur.firstDir};
                return true;
            }
            if (cur.g >= depthLimit) continue;

            // Expand
            for (int ax = 0; ax < gD; ++ax) {
                for (int dir : {-1, +1}) {
                    Pos np = cur.p;
                    np.x[ax] += dir;
                    // safety relative to current state (conservative)
                    if (!inBounds(np)) continue;
                    if (obstacles.find(np) != obstacles.end()) continue;

                    // Occupied check: conservative as well; allow stepping into current tail if not food
                    if (occupied.find(np) != occupied.end()) {
                        if (!(np == tail() && food.find(np) == food.end())) continue;
                    }

                    int ng = cur.g + 1;
                    auto jt = bestG.find(np);
                    if (jt != bestG.end() && jt->second <= ng) continue;
                    bestG[np] = ng;

                    pushNode(np, ng, cur.firstAxis, cur.firstDir);
                }
            }
        }
        return false;
    }

    Move survivalMove(const Pos& target, const Pos& prevHead, bool hasPrev) const {
        Pos h = head();
        bool found = false;
        Move best{0, +1};
        long long bestKey = (long long)-4e18;

        for (int ax = 0; ax < gD; ++ax) {
            for (int dir : {-1, +1}) {
                Pos np = h;
                np.x[ax] += dir;
                if (!isSafeNext(np)) continue;

                int deg = safeNeighborCount(np);
                long long distToTarget = manhattan(np, target);

                // discourage immediate backtracking if alternatives exist
                long long backPenalty = 0;
                if (hasPrev && np == prevHead) backPenalty = 1000;

                // prefer more options, and also prefer reducing distance to target (small bias)
                long long key = (long long)deg * 10000 - (long long)distToTarget * 10 - backPenalty;

                if (!found || key > bestKey) {
                    found = true;
                    bestKey = key;
                    best = {ax, dir};
                }
            }
        }

        if (found) return best;

        // No safe moves: output something (game will end anyway)
        return {0, +1};
    }
};

static bool tryBuildBodyLine(const vector<int>& bounds,
                            const unordered_set<Pos, PosHash>& obstacles,
                            const Pos& head, int L,
                            int axis, int dir,
                            deque<Pos>& outBody,
                            unordered_set<Pos, PosHash>& outOcc)
{
    outBody.clear();
    outOcc.clear();
    outBody.push_back(head);
    outOcc.insert(head);

    Pos cur = head;
    for (int k = 1; k < L; ++k) {
        cur.x[axis] -= dir; // extend "behind" the head
        // bounds
        if (cur.x[axis] < 0 || cur.x[axis] >= bounds[axis]) return false;
        // obstacle
        if (obstacles.find(cur) != obstacles.end()) return false;
        // uniqueness
        if (outOcc.find(cur) != outOcc.end()) return false;
        outBody.push_back(cur);
        outOcc.insert(cur);
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Game G;

    int D;
    long long max_turns;
    if (!(cin >> D >> max_turns)) return 0;
    gD = D;
    G.maxTurns = max_turns;

    G.bounds.resize(D);
    for (int i = 0; i < D; ++i) cin >> G.bounds[i];

    int snakeLen;
    Pos head{};
    cin >> snakeLen;
    for (int i = 0; i < D; ++i) cin >> head.x[i];

    int F;
    cin >> F;
    G.food.reserve((size_t)F * 2 + 8);
    for (int i = 0; i < F; ++i) {
        Pos p{};
        for (int j = 0; j < D; ++j) cin >> p.x[j];
        G.food.insert(p);
    }

    int O;
    cin >> O;
    G.obstacles.reserve((size_t)O * 2 + 8);
    for (int i = 0; i < O; ++i) {
        Pos p{};
        for (int j = 0; j < D; ++j) cin >> p.x[j];
        G.obstacles.insert(p);
    }

    // Build an initial body guess (best effort). If it fails, fall back to length 1.
    {
        deque<Pos> candidateBody;
        unordered_set<Pos, PosHash> candidateOcc;
        bool ok = false;

        if (snakeLen <= 1) {
            ok = true;
            candidateBody = deque<Pos>{head};
            candidateOcc.insert(head);
        } else {
            // Try all axis/dir
            for (int ax = 0; ax < D && !ok; ++ax) {
                for (int dir : {-1, +1}) {
                    if (tryBuildBodyLine(G.bounds, G.obstacles, head, snakeLen, ax, dir, candidateBody, candidateOcc)) {
                        ok = true;
                        break;
                    }
                }
            }
        }

        if (!ok) {
            candidateBody = deque<Pos>{head};
            candidateOcc.clear();
            candidateOcc.insert(head);
        }

        G.body = std::move(candidateBody);
        G.occupied = std::move(candidateOcc);
    }

    vector<Move> outputs;
    outputs.reserve((size_t)min<long long>(G.maxTurns, 200000));

    Pos prevHead{};
    bool hasPrev = false;

    for (long long turn = 0; turn < G.maxTurns; ++turn) {
        if (G.food.empty()) break;

        Pos target = G.pickNearestFood();

        Move mv;
        bool got = false;

        // 1) Greedy move
        mv = G.greedyToward(target, got);

        // 2) If greedy failed to produce anything safe (should be rare), A*
        if (!got) {
            if (G.localAStar(target, mv, /*depthLimit=*/10, /*nodeLimit=*/120000)) {
                got = true;
            }
        }

        // 3) Survival fallback
        if (!got) {
            mv = G.survivalMove(target, prevHead, hasPrev);
        }

        outputs.push_back(mv);

        // Simulate move; if we die, stop producing further outputs
        bool ok = false, ate = false;
        prevHead = G.head();
        hasPrev = true;
        G.applyMove(mv, ok, ate);
        if (!ok) break;
    }

    // Print all planned moves
    for (const auto& m : outputs) {
        cout << m.axis << " " << m.dir << "\n";
    }
    return 0;
}
