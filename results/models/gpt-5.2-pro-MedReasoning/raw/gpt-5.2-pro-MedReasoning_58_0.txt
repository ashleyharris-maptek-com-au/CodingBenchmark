```javascript
'use strict';

const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input.length ? JSON.parse(input) : {};

const grid = data.grid || [];
const robotsIn = data.robots || [];
const jobsIn = data.jobs || [];

const H = grid.length;
const W = H ? grid[0].length : 0;
const N = W * H;

function idxOf(x, y) {
  return y * W + x;
}

// Open cell map
const open = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = grid[y];
  for (let x = 0; x < W; x++) {
    open[idxOf(x, y)] = (row.charCodeAt(x) === 35 /* # */) ? 0 : 1;
  }
}

// Robots: sort by id for deterministic tie-breaking
const robots = robotsIn.slice().sort((a, b) => {
  if (a.id !== b.id) return a.id < b.id ? -1 : 1;
  return 0;
});
const R = robots.length;

const robotIds = new Array(R);
const robotPos = new Int32Array(R);
let maxRobotId = -1;

for (let i = 0; i < R; i++) {
  const r = robots[i];
  robotIds[i] = r.id;
  if (typeof r.id === 'number' && r.id > maxRobotId) maxRobotId = r.id;
  robotPos[i] = idxOf(r.x | 0, r.y | 0);
}

// Jobs: sort by priority desc, then id asc for determinism
const jobs = jobsIn.slice().sort((a, b) => {
  const ap = a.priority ?? 0;
  const bp = b.priority ?? 0;
  if (ap !== bp) return bp - ap;
  const aid = a.id, bid = b.id;
  if (typeof aid === 'number' && typeof bid === 'number') return aid - bid;
  const as = String(aid), bs = String(bid);
  if (as < bs) return -1;
  if (as > bs) return 1;
  return 0;
});

// Output assignments indexed by robot id
const assignments = Array.from({ length: Math.max(0, maxRobotId + 1) }, () => []);
function pushAssignment(robotInternalIdx, jobId) {
  const rid = robotIds[robotInternalIdx];
  if (typeof rid === 'number' && rid >= 0) assignments[rid].push(jobId);
}

// BFS workspace (reused)
const visited = new Uint32Array(N);
const dist = new Int32Array(N);
const queue = new Int32Array(N);
let stamp = 1;

function bumpStamp() {
  stamp++;
  if (stamp === 0xFFFFFFFF) {
    visited.fill(0);
    stamp = 1;
  }
}

/**
 * Find nearest robot (by BFS grid distance) to a job cell.
 * Tie-break: smaller robot id.
 * Stops BFS after fully processing the first distance layer where any robot is found.
 */
function findNearestRobot(jobIdx, robotPosMap) {
  bumpStamp();

  let head = 0, tail = 0;
  queue[tail++] = jobIdx;
  visited[jobIdx] = stamp;
  dist[jobIdx] = 0;

  let bestDist = -1;
  let bestRobot = -1;

  while (head < tail) {
    const v = queue[head++];
    const dv = dist[v];

    if (bestDist !== -1 && dv > bestDist) break;

    const arr = robotPosMap.get(v);
    if (arr !== undefined) {
      if (bestDist === -1) {
        bestDist = dv;
        // choose min robot id among those at this cell
        let br = arr[0];
        for (let k = 1; k < arr.length; k++) {
          const ri = arr[k];
          if (robotIds[ri] < robotIds[br]) br = ri;
        }
        bestRobot = br;
      } else if (dv === bestDist) {
        for (let k = 0; k < arr.length; k++) {
          const ri = arr[k];
          if (robotIds[ri] < robotIds[bestRobot]) bestRobot = ri;
        }
      }
    }

    // No need to expand beyond bestDist
    if (bestDist !== -1 && dv >= bestDist) continue;

    // Expand neighbors
    const x = v % W;

    // left
    if (x !== 0) {
      const nb = v - 1;
      if (open[nb] && visited[nb] !== stamp) {
        visited[nb] = stamp;
        dist[nb] = dv + 1;
        queue[tail++] = nb;
      }
    }
    // right
    if (x !== W - 1) {
      const nb = v + 1;
      if (open[nb] && visited[nb] !== stamp) {
        visited[nb] = stamp;
        dist[nb] = dv + 1;
        queue[tail++] = nb;
      }
    }
    // up
    if (v >= W) {
      const nb = v - W;
      if (open[nb] && visited[nb] !== stamp) {
        visited[nb] = stamp;
        dist[nb] = dv + 1;
        queue[tail++] = nb;
      }
    }
    // down
    if (v < N - W) {
      const nb = v + W;
      if (open[nb] && visited[nb] !== stamp) {
        visited[nb] = stamp;
        dist[nb] = dv + 1;
        queue[tail++] = nb;
      }
    }
  }

  if (bestRobot === -1) {
    // If somehow unreachable/disconnected, fall back to smallest id robot
    return 0;
  }
  return bestRobot;
}

// Main greedy assignment
for (let j = 0; j < jobs.length; j++) {
  const job = jobs[j];
  const jobIdx = idxOf(job.x | 0, job.y | 0);

  // Build current robot position map: cellIdx -> [robotInternalIdx...]
  const posMap = new Map();
  for (let i = 0; i < R; i++) {
    const p = robotPos[i];
    const existing = posMap.get(p);
    if (existing === undefined) posMap.set(p, [i]);
    else existing.push(i);
  }

  const chosen = findNearestRobot(jobIdx, posMap);
  pushAssignment(chosen, job.id);
  robotPos[chosen] = jobIdx;
}

process.stdout.write(JSON.stringify({ assignments }));
```
