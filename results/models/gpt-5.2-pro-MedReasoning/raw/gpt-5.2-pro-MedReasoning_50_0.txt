```glsl
#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(std430, set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(std430, set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(std140, set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main()
{
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) return;

    // Load this boid
    vec3 pos = inData[i * 2u + 0u].xyz;
    vec3 vel = inData[i * 2u + 1u].xyz;

    float dt         = radii.x;
    float sepR       = radii.y;
    float alignR     = radii.z;
    float cohR       = radii.w;

    float sepW       = weights.x;
    float alignW     = weights.y;
    float cohW       = weights.z;
    float maxSpeed   = weights.w;

    float boundSize  = bounds.x;
    float boundForce = bounds.y;

    float sepR2   = sepR   * sepR;
    float alignR2 = alignR * alignR;
    float cohR2   = cohR   * cohR;

    // Accumulators
    vec3 sepSum   = vec3(0.0);
    vec3 alignSum = vec3(0.0);
    vec3 cohSum   = vec3(0.0);

    uint sepCount   = 0u;
    uint alignCount = 0u;
    uint cohCount   = 0u;

    // Naive O(N^2) neighbor search
    for (uint j = 0u; j < N; ++j)
    {
        if (j == i) continue;

        vec3 npos = inData[j * 2u + 0u].xyz;
        vec3 nvel = inData[j * 2u + 1u].xyz;

        vec3 d = npos - pos;
        float dist2 = dot(d, d);
        if (dist2 <= 0.0) continue; // skip zero-distance neighbors

        // Separation (steer away)
        if (dist2 < sepR2)
        {
            // Push away stronger when closer: (pos - npos) / dist^2
            float invDist = inversesqrt(dist2);
            vec3 away = (-d) * (invDist * invDist); // (-d)/dist^2
            sepSum += away;
            sepCount++;
        }

        // Alignment (match velocity)
        if (dist2 < alignR2)
        {
            alignSum += nvel;
            alignCount++;
        }

        // Cohesion (move toward center of neighbors)
        if (dist2 < cohR2)
        {
            cohSum += npos;
            cohCount++;
        }
    }

    // Average contributions (avoid division by zero)
    vec3 sepForce = vec3(0.0);
    if (sepCount > 0u)
        sepForce = (sepSum / float(sepCount)) * sepW;

    vec3 alignForce = vec3(0.0);
    if (alignCount > 0u)
    {
        vec3 avgVel = alignSum / float(alignCount);
        alignForce = (avgVel - vel) * alignW;
    }

    vec3 cohForce = vec3(0.0);
    if (cohCount > 0u)
    {
        vec3 avgPos = cohSum / float(cohCount);
        cohForce = (avgPos - pos) * cohW;
    }

    vec3 accel = sepForce + alignForce + cohForce;

    // Boundary force: when a component exceeds boundSize*0.4, push toward center.
    float inner = boundSize * 0.4;
    float scale = boundSize * 0.5;

    if (pos.x > inner)  accel.x += (-boundForce) * ((pos.x - inner) / scale);
    if (pos.x < -inner) accel.x += ( boundForce) * ((-inner - pos.x) / scale);

    if (pos.y > inner)  accel.y += (-boundForce) * ((pos.y - inner) / scale);
    if (pos.y < -inner) accel.y += ( boundForce) * ((-inner - pos.y) / scale);

    if (pos.z > inner)  accel.z += (-boundForce) * ((pos.z - inner) / scale);
    if (pos.z < -inner) accel.z += ( boundForce) * ((-inner - pos.z) / scale);

    // Integrate
    vel += accel * dt;

    // Clamp to max speed
    float spd2 = dot(vel, vel);
    float maxSpd2 = maxSpeed * maxSpeed;
    if (spd2 > maxSpd2 && spd2 > 0.0)
        vel *= (maxSpeed * inversesqrt(spd2));

    pos += vel * dt;

    // Store
    outData[i * 2u + 0u] = vec4(pos, 0.0);
    outData[i * 2u + 1u] = vec4(vel, 0.0);
}
```
