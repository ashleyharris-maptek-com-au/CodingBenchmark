```spirv
; SPIR-V
; Version: 1.0
; Bound: 200
; Schema: 0
OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal   Location 1
OpDecorate %inColor    Location 2
OpDecorate %outColor   Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_6 = OpConstant %uint 6
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%arr_v3_in = OpTypeArray %v3float %uint_6

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%inWorldPos = OpVariable %ptr_in_arr_v3 Input
%inNormal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%float_0 = OpConstant %float 0
%float_0_015 = OpConstant %float 0.015
%float_0_3 = OpConstant %float 0.3
%float_1 = OpConstant %float 1

%v3_white = OpConstantComposite %v3float %float_1 %float_1 %float_1

%fn_main = OpTypeFunction %void
%fn_emitEdge = OpTypeFunction %void %v4float %v4float

%emitEdge = OpFunction %void None %fn_emitEdge
%p0 = OpFunctionParameter %v4float
%p1 = OpFunctionParameter %v4float
%emitEdge_entry = OpLabel

%outPosPtr = OpAccessChain %ptr_out_v4 %gl_out %int_0

%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1

%p0_w = OpCompositeExtract %float %p0 3
%p1_w = OpCompositeExtract %float %p1 3

%p0_w2 = OpCompositeConstruct %v2float %p0_w %p0_w
%p1_w2 = OpCompositeConstruct %v2float %p1_w %p1_w

%ndc0 = OpFDiv %v2float %p0_xy %p0_w2
%ndc1 = OpFDiv %v2float %p1_xy %p1_w2
%d = OpFSub %v2float %ndc1 %ndc0

%dir = OpExtInst %v2float %glsl Normalize %d
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp = OpCompositeConstruct %v2float %neg_dir_y %dir_x

%off_ndc = OpVectorTimesScalar %v2float %perp %float_0_015
%off0_xy = OpFMul %v2float %off_ndc %p0_w2
%off1_xy = OpFMul %v2float %off_ndc %p1_w2

%off0_x = OpCompositeExtract %float %off0_xy 0
%off0_y = OpCompositeExtract %float %off0_xy 1
%off1_x = OpCompositeExtract %float %off1_xy 0
%off1_y = OpCompositeExtract %float %off1_xy 1

%off0 = OpCompositeConstruct %v4float %off0_x %off0_y %float_0 %float_0
%off1 = OpCompositeConstruct %v4float %off1_x %off1_y %float_0 %float_0

%p0_plus = OpFAdd %v4float %p0 %off0
%p0_minus = OpFSub %v4float %p0 %off0
%p1_plus = OpFAdd %v4float %p1 %off1
%p1_minus = OpFSub %v4float %p1 %off1

OpStore %outColor %v3_white
OpStore %outPosPtr %p0_plus
OpEmitVertex

OpStore %outColor %v3_white
OpStore %outPosPtr %p0_minus
OpEmitVertex

OpStore %outColor %v3_white
OpStore %outPosPtr %p1_plus
OpEmitVertex

OpStore %outColor %v3_white
OpStore %outPosPtr %p1_minus
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd

%main = OpFunction %void None %fn_main
%main_entry = OpLabel

%mainOutPosPtr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; ---- Emit filled triangle (indices 0,2,4) with dimmed colors ----
%pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%col0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%col0 = OpLoad %v3float %col0_ptr
%col0_dim = OpVectorTimesScalar %v3float %col0 %float_0_3
OpStore %mainOutPosPtr %pos0
OpStore %outColor %col0_dim
OpEmitVertex

%pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%col2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
%col2 = OpLoad %v3float %col2_ptr
%col2_dim = OpVectorTimesScalar %v3float %col2 %float_0_3
OpStore %mainOutPosPtr %pos2
OpStore %outColor %col2_dim
OpEmitVertex

%pos4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %pos4_ptr
%col4_ptr = OpAccessChain %ptr_in_v3 %inColor %int_4
%col4 = OpLoad %v3float %col4_ptr
%col4_dim = OpVectorTimesScalar %v3float %col4 %float_0_3
OpStore %mainOutPosPtr %pos4
OpStore %outColor %col4_dim
OpEmitVertex

OpEndPrimitive

; ---- Emit thick quads for all 3 triangle edges (0-2, 2-4, 4-0) ----
OpFunctionCall %void %emitEdge %pos0 %pos2
OpFunctionCall %void %emitEdge %pos2 %pos4
OpFunctionCall %void %emitEdge %pos4 %pos0

OpReturn
OpFunctionEnd
```
