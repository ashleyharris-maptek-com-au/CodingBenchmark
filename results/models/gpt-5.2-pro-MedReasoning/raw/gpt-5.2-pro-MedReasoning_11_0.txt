use std::cmp::Ordering;
use std::f64::consts::PI;
use std::io::{self, Read};

const EPS: f64 = 1e-9;

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    #[inline]
    fn sub(self, o: Point) -> Point {
        Point {
            x: self.x - o.x,
            y: self.y - o.y,
        }
    }
    #[inline]
    fn add(self, o: Point) -> Point {
        Point {
            x: self.x + o.x,
            y: self.y + o.y,
        }
    }
}

#[inline]
fn cross(a: Point, b: Point) -> f64 {
    a.x * b.y - a.y * b.x
}

#[inline]
fn dot(a: Point, b: Point) -> f64 {
    a.x * b.x + a.y * b.y
}

#[inline]
fn rotate_point(p: Point, sin_a: f64, cos_a: f64) -> Point {
    Point {
        x: p.x * cos_a - p.y * sin_a,
        y: p.x * sin_a + p.y * cos_a,
    }
}

#[derive(Clone, Copy, Debug)]
struct Rect {
    x: f64,
    y: f64,
    w: f64,
    h: f64,
}

impl Rect {
    #[inline]
    fn right(&self) -> f64 {
        self.x + self.w
    }
    #[inline]
    fn top(&self) -> f64 {
        self.y + self.h
    }
    #[inline]
    fn area(&self) -> f64 {
        self.w * self.h
    }
    #[inline]
    fn contains(&self, o: &Rect) -> bool {
        o.x + EPS >= self.x
            && o.y + EPS >= self.y
            && o.right() <= self.right() + EPS
            && o.top() <= self.top() + EPS
    }
    #[inline]
    fn intersects(&self, o: &Rect) -> bool {
        self.x < o.right() - EPS && self.right() > o.x + EPS && self.y < o.top() - EPS && self.top() > o.y + EPS
    }
}

fn polygon_signed_area(poly: &[Point]) -> f64 {
    let n = poly.len();
    if n < 3 {
        return 0.0;
    }
    let mut a = 0.0;
    for i in 0..n {
        let p = poly[i];
        let q = poly[(i + 1) % n];
        a += p.x * q.y - q.x * p.y;
    }
    0.5 * a
}

fn polygon_bbox(poly: &[Point]) -> Rect {
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for &p in poly {
        minx = minx.min(p.x);
        miny = miny.min(p.y);
        maxx = maxx.max(p.x);
        maxy = maxy.max(p.y);
    }
    Rect {
        x: minx,
        y: miny,
        w: (maxx - minx).max(0.0),
        h: (maxy - miny).max(0.0),
    }
}

fn is_convex_ccw(poly: &[Point]) -> bool {
    let n = poly.len();
    if n < 4 {
        return true;
    }
    let mut sign = 0i32;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        let c = poly[(i + 2) % n];
        let z = cross(b.sub(a), c.sub(b));
        if z.abs() <= EPS {
            continue;
        }
        let s = if z > 0.0 { 1 } else { -1 };
        if sign == 0 {
            sign = s;
        } else if sign != s {
            return false;
        }
    }
    true
}

#[inline]
fn point_on_segment(p: Point, a: Point, b: Point) -> bool {
    let ab = b.sub(a);
    let ap = p.sub(a);
    if cross(ab, ap).abs() > 1e-8 {
        return false;
    }
    let d = dot(ap, ab);
    if d < -1e-8 {
        return false;
    }
    let ab2 = dot(ab, ab);
    if d > ab2 + 1e-8 {
        return false;
    }
    true
}

/// Inclusive point-in-polygon (ray casting) with boundary treated as inside.
fn point_in_polygon_inclusive(p: Point, poly: &[Point]) -> bool {
    let n = poly.len();
    if n < 3 {
        return false;
    }
    // Boundary check
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if point_on_segment(p, a, b) {
            return true;
        }
    }

    let mut inside = false;
    let mut j = n - 1;
    for i in 0..n {
        let pi = poly[i];
        let pj = poly[j];
        // Check if edge crosses horizontal ray to +inf
        let intersect = ((pi.y > p.y) != (pj.y > p.y))
            && (p.x < (pj.x - pi.x) * (p.y - pi.y) / (pj.y - pi.y + 0.0) + pi.x);
        if intersect {
            inside = !inside;
        }
        j = i;
    }
    inside
}

/// Inclusive point-in-convex-polygon (CCW) using half-plane tests.
fn point_in_convex_inclusive(p: Point, poly_ccw: &[Point]) -> bool {
    let n = poly_ccw.len();
    for i in 0..n {
        let a = poly_ccw[i];
        let b = poly_ccw[(i + 1) % n];
        let z = cross(b.sub(a), p.sub(a));
        if z < -1e-9 {
            return false;
        }
    }
    true
}

/// Proper segment intersection (strict crossing). Touching at endpoints or collinear overlaps are NOT "proper".
fn segments_intersect_proper(a1: Point, a2: Point, b1: Point, b2: Point) -> bool {
    let a = a2.sub(a1);
    let b = b2.sub(b1);
    let c1 = cross(a, b1.sub(a1));
    let c2 = cross(a, b2.sub(a1));
    let c3 = cross(b, a1.sub(b1));
    let c4 = cross(b, a2.sub(b1));

    // Strict opposite signs on both pairs => proper intersection
    (c1 > EPS && c2 < -EPS || c1 < -EPS && c2 > EPS) && (c3 > EPS && c4 < -EPS || c3 < -EPS && c4 > EPS)
}

struct EdgeGrid {
    dim: usize,
    minx: f64,
    miny: f64,
    cell_w: f64,
    cell_h: f64,
    cells: Vec<Vec<usize>>, // indices into edges
    seen: Vec<u32>,
    stamp: u32,
}

impl EdgeGrid {
    fn new(stock_poly: &[Point], edges: &[(Point, Point)], dim: usize) -> Self {
        let bb = polygon_bbox(stock_poly);
        let mut cell_w = bb.w / dim as f64;
        let mut cell_h = bb.h / dim as f64;
        if cell_w.abs() < 1e-12 {
            cell_w = 1.0;
        }
        if cell_h.abs() < 1e-12 {
            cell_h = 1.0;
        }
        let mut cells = vec![Vec::<usize>::new(); dim * dim];

        let mut grid = EdgeGrid {
            dim,
            minx: bb.x,
            miny: bb.y,
            cell_w,
            cell_h,
            cells,
            seen: vec![0u32; edges.len()],
            stamp: 1,
        };

        for (idx, &(a, b)) in edges.iter().enumerate() {
            let minx = a.x.min(b.x);
            let maxx = a.x.max(b.x);
            let miny = a.y.min(b.y);
            let maxy = a.y.max(b.y);

            let (ix0, ix1, iy0, iy1) = grid.cell_range(minx, maxx, miny, maxy);
            for iy in iy0..=iy1 {
                for ix in ix0..=ix1 {
                    grid.cells[iy * dim + ix].push(idx);
                }
            }
        }

        grid
    }

    #[inline]
    fn clamp_i(&self, v: i32) -> usize {
        if v < 0 {
            0
        } else if v as usize >= self.dim {
            self.dim - 1
        } else {
            v as usize
        }
    }

    #[inline]
    fn cell_range(&self, minx: f64, maxx: f64, miny: f64, maxy: f64) -> (usize, usize, usize, usize) {
        let ix0 = ((minx - self.minx) / self.cell_w).floor() as i32;
        let ix1 = ((maxx - self.minx) / self.cell_w).floor() as i32;
        let iy0 = ((miny - self.miny) / self.cell_h).floor() as i32;
        let iy1 = ((maxy - self.miny) / self.cell_h).floor() as i32;
        (
            self.clamp_i(ix0),
            self.clamp_i(ix1),
            self.clamp_i(iy0),
            self.clamp_i(iy1),
        )
    }

    /// Collect unique edge indices potentially intersecting a query AABB.
    fn query_edges_unique(&mut self, q: &Rect, out: &mut Vec<usize>) {
        out.clear();
        self.stamp = self.stamp.wrapping_add(1);
        if self.stamp == 0 {
            self.stamp = 1;
            self.seen.fill(0);
        }
        let (ix0, ix1, iy0, iy1) = self.cell_range(q.x, q.right(), q.y, q.top());
        for iy in iy0..=iy1 {
            for ix in ix0..=ix1 {
                let cell = &self.cells[iy * self.dim + ix];
                for &ei in cell {
                    if self.seen[ei] != self.stamp {
                        self.seen[ei] = self.stamp;
                        out.push(ei);
                    }
                }
            }
        }
    }
}

struct StockGeom {
    poly: Vec<Point>, // CCW
    edges: Vec<(Point, Point)>,
    bbox: Rect,
    convex: bool,
    grid: Option<EdgeGrid>, // only used if non-convex
}

impl StockGeom {
    fn new(mut poly: Vec<Point>) -> Self {
        // Ensure CCW
        if polygon_signed_area(&poly) < 0.0 {
            poly.reverse();
        }
        let n = poly.len();
        let mut edges = Vec::with_capacity(n);
        for i in 0..n {
            edges.push((poly[i], poly[(i + 1) % n]));
        }
        let bbox = polygon_bbox(&poly);
        let convex = is_convex_ccw(&poly);

        let grid = if convex {
            None
        } else {
            // Fixed 64x64 grid: fast and stable memory.
            Some(EdgeGrid::new(&poly, &edges, 64))
        };

        StockGeom {
            poly,
            edges,
            bbox,
            convex,
            grid,
        }
    }

    fn contains_transformed_polygon(&mut self, pts_rot: &[Point], tx: f64, ty: f64) -> bool {
        // Quick bbox check vs stock bbox (should already hold, but cheap).
        // We can't get exact bbox without scanning, so skip here.

        // Vertex inside test
        if self.convex {
            for &p in pts_rot {
                let q = Point { x: p.x + tx, y: p.y + ty };
                if !point_in_convex_inclusive(q, &self.poly) {
                    return false;
                }
            }
            true
        } else {
            // Non-convex:
            for &p in pts_rot {
                let q = Point { x: p.x + tx, y: p.y + ty };
                if !point_in_polygon_inclusive(q, &self.poly) {
                    return false;
                }
            }
            // Edge crossing test against stock boundary
            let grid = self.grid.as_mut().unwrap();
            let mut candidates = Vec::<usize>::new();

            let m = pts_rot.len();
            for i in 0..m {
                let a0 = pts_rot[i];
                let b0 = pts_rot[(i + 1) % m];
                let a = Point { x: a0.x + tx, y: a0.y + ty };
                let b = Point { x: b0.x + tx, y: b0.y + ty };
                let minx = a.x.min(b.x);
                let maxx = a.x.max(b.x);
                let miny = a.y.min(b.y);
                let maxy = a.y.max(b.y);
                let qbb = Rect { x: minx, y: miny, w: (maxx - minx).max(0.0), h: (maxy - miny).max(0.0) };

                grid.query_edges_unique(&qbb, &mut candidates);

                for &ei in &candidates {
                    let (s1, s2) = self.edges[ei];
                    // Proper crossing means piece edge leaves/enters: invalid.
                    if segments_intersect_proper(a, b, s1, s2) {
                        return false;
                    }
                }
            }
            true
        }
    }
}

#[derive(Clone)]
struct Variant {
    angle: f64,
    sin_a: f64,
    cos_a: f64,
    pts_rot: Vec<Point>,
    // rotated bbox
    minx: f64,
    miny: f64,
    w: f64,
    h: f64,
}

fn build_variant(poly: &[Point], angle: f64) -> Variant {
    let (sin_a, cos_a) = angle.sin_cos();
    let mut pts_rot = Vec::with_capacity(poly.len());
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for &p in poly {
        let r = rotate_point(p, sin_a, cos_a);
        minx = minx.min(r.x);
        miny = miny.min(r.y);
        maxx = maxx.max(r.x);
        maxy = maxy.max(r.y);
        pts_rot.push(r);
    }
    Variant {
        angle,
        sin_a,
        cos_a,
        pts_rot,
        minx,
        miny,
        w: (maxx - minx).max(0.0),
        h: (maxy - miny).max(0.0),
    }
}

fn best_extra_angle_if_needed(poly: &[Point], stock_bbox: Rect) -> Option<f64> {
    // If none of the preset variants fit inside stock bbox, try 32 sampled angles and pick one that fits bbox.
    let mut best: Option<(f64, f64)> = None; // (area, angle)
    for t in 0..32 {
        let angle = (t as f64) * (PI / 32.0);
        let (sin_a, cos_a) = angle.sin_cos();
        let mut minx = f64::INFINITY;
        let mut miny = f64::INFINITY;
        let mut maxx = f64::NEG_INFINITY;
        let mut maxy = f64::NEG_INFINITY;
        for &p in poly {
            let r = rotate_point(p, sin_a, cos_a);
            minx = minx.min(r.x);
            miny = miny.min(r.y);
            maxx = maxx.max(r.x);
            maxy = maxy.max(r.y);
        }
        let w = (maxx - minx).max(0.0);
        let h = (maxy - miny).max(0.0);
        if w <= stock_bbox.w + 1e-9 && h <= stock_bbox.h + 1e-9 {
            let area = w * h;
            match best {
                None => best = Some((area, angle)),
                Some((ba, _)) if area < ba => best = Some((area, angle)),
                _ => {}
            }
        }
    }
    best.map(|(_, a)| a)
}

struct XorShift64 {
    state: u64,
}
impl XorShift64 {
    fn new(seed: u64) -> Self {
        Self { state: if seed == 0 { 0x9E3779B97F4A7C15 } else { seed } }
    }
    #[inline]
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }
    #[inline]
    fn next_f64(&mut self) -> f64 {
        // [0,1)
        let v = self.next_u64() >> 11;
        (v as f64) * (1.0 / ((1u64 << 53) as f64))
    }
}

enum BinMode {
    MaxRects,
    Shelf,
}

struct MaxRectsBin {
    container: Rect,
    free: Vec<Rect>,
}

impl MaxRectsBin {
    fn new(container: Rect) -> Self {
        Self { container, free: vec![container] }
    }

    fn find_positions(&self, w: f64, h: f64, out: &mut Vec<(f64, f64, f64)>) {
        // (x,y,score) where score smaller is better.
        out.clear();
        for fr in &self.free {
            if w <= fr.w + EPS && h <= fr.h + EPS {
                let positions = [
                    (fr.x, fr.y),
                    (fr.right() - w, fr.y),
                    (fr.x, fr.top() - h),
                    (fr.right() - w, fr.top() - h),
                ];
                for &(x, y) in &positions {
                    if x + w <= fr.right() + 1e-9 && y + h <= fr.top() + 1e-9 && x + 1e-9 >= fr.x && y + 1e-9 >= fr.y {
                        let leftover_w = (fr.w - w).abs();
                        let leftover_h = (fr.h - h).abs();
                        let score = leftover_w.min(leftover_h) + 1e-6 * (leftover_w.max(leftover_h));
                        out.push((x, y, score));
                    }
                }
            }
        }
        out.sort_by(|a, b| a.2.partial_cmp(&b.2).unwrap_or(Ordering::Equal));
        if out.len() > 16 {
            out.truncate(16);
        }
    }

    fn place_rect(&mut self, r: Rect) {
        let mut new_free = Vec::<Rect>::with_capacity(self.free.len() + 8);
        for fr in &self.free {
            if !fr.intersects(&r) {
                new_free.push(*fr);
                continue;
            }

            // Split fr around r (guillotine-like).
            let fr_right = fr.right();
            let fr_top = fr.top();
            let r_right = r.right();
            let r_top = r.top();

            // Left
            if r.x > fr.x + EPS {
                let w = r.x - fr.x;
                new_free.push(Rect { x: fr.x, y: fr.y, w, h: fr.h });
            }
            // Right
            if r_right < fr_right - EPS {
                let x = r_right;
                let w = fr_right - r_right;
                new_free.push(Rect { x, y: fr.y, w, h: fr.h });
            }
            // Bottom
            if r.y > fr.y + EPS {
                let h = r.y - fr.y;
                new_free.push(Rect { x: fr.x, y: fr.y, w: fr.w, h });
            }
            // Top
            if r_top < fr_top - EPS {
                let y = r_top;
                let h = fr_top - r_top;
                new_free.push(Rect { x: fr.x, y, w: fr.w, h });
            }
        }

        // Prune: remove invalid / tiny, then remove contained rects.
        new_free.retain(|rr| rr.w > 1e-10 && rr.h > 1e-10);

        // Containment pruning O(m^2) but m is kept bounded.
        let mut i = 0;
        while i < new_free.len() {
            let mut contained = false;
            for j in 0..new_free.len() {
                if i != j && new_free[j].contains(&new_free[i]) {
                    contained = true;
                    break;
                }
            }
            if contained {
                new_free.swap_remove(i);
            } else {
                i += 1;
            }
        }

        // Keep free list bounded (important for NP=5000).
        if new_free.len() > 4096 {
            new_free.sort_by(|a, b| b.area().partial_cmp(&a.area()).unwrap_or(Ordering::Equal));
            new_free.truncate(4096);
        }

        self.free = new_free;
    }
}

struct ShelfBin {
    container: Rect,
    x: f64,
    y: f64,
    shelf_h: f64,
}

impl ShelfBin {
    fn new(container: Rect) -> Self {
        Self { container, x: container.x, y: container.y, shelf_h: 0.0 }
    }

    fn candidates(&self, w: f64, h: f64, out: &mut Vec<(f64, f64)>) {
        out.clear();
        // current position
        out.push((self.x, self.y));
        // try a few forward offsets within row (leave gap; still safe).
        let remaining = self.container.right() - self.x;
        if remaining >= w + 1e-9 {
            let step = (w * 0.25).max(1e-6);
            let mut k = 1;
            while k <= 3 {
                let nx = self.x + (k as f64) * step;
                if nx + w <= self.container.right() + 1e-9 {
                    out.push((nx, self.y));
                }
                k += 1;
            }
        }
    }

    fn try_advance_and_place(&mut self, x: f64, w: f64, h: f64) -> Option<Rect> {
        // If x is on current shelf line, we can place and set cursor after it.
        let y = self.y;
        if x < self.container.x - 1e-9 || x + w > self.container.right() + 1e-9 {
            return None;
        }
        if y + h > self.container.top() + 1e-9 {
            return None;
        }
        // Ensure it doesn't overlap the "already used segment": we only allow placing at or after current x
        if x + 1e-9 < self.x {
            return None;
        }
        // Place
        let rect = Rect { x, y, w, h };
        self.x = x + w;
        self.shelf_h = self.shelf_h.max(h);
        Some(rect)
    }

    fn new_shelf(&mut self) -> bool {
        self.x = self.container.x;
        self.y += self.shelf_h;
        self.shelf_h = 0.0;
        self.y < self.container.top() - 1e-9
    }
}

struct StockBin {
    mode: BinMode,
    maxrects: Option<MaxRectsBin>,
    shelf: Option<ShelfBin>,
}

impl StockBin {
    fn new(mode: BinMode, container: Rect) -> Self {
        match mode {
            BinMode::MaxRects => StockBin { mode, maxrects: Some(MaxRectsBin::new(container)), shelf: None },
            BinMode::Shelf => StockBin { mode, maxrects: None, shelf: Some(ShelfBin::new(container)) },
        }
    }
}

struct Piece {
    idx: usize,
    poly: Vec<Point>,
    area_abs: f64,
    variants: Vec<Variant>,
}

fn choose_rotations(np: usize, total_vertices: usize) -> Vec<f64> {
    if np > 1500 || total_vertices > 1_000_000 {
        vec![0.0, PI * 0.5]
    } else if np > 500 || total_vertices > 200_000 {
        vec![0.0, PI * 0.5, PI, PI * 1.5]
    } else {
        // 8-way
        (0..8).map(|k| (k as f64) * (PI / 4.0)).collect()
    }
}

fn try_place_piece_in_bin(
    stock_geom: &mut StockGeom,
    bin: &mut StockBin,
    variants: &[Variant],
) -> Option<(Placement, Rect)> {
    let container = stock_geom.bbox;

    match bin.mode {
        BinMode::MaxRects => {
            let mr = bin.maxrects.as_ref().unwrap();
            let mut pos_candidates: Vec<(f64, f64, f64)> = Vec::new();

            // Try each variant; choose first feasible (already sorted by piece order outside).
            for v in variants {
                if v.w > container.w + 1e-9 || v.h > container.h + 1e-9 {
                    continue;
                }
                bin.maxrects.as_ref().unwrap().find_positions(v.w, v.h, &mut pos_candidates);
                for &(x, y, _score) in &pos_candidates {
                    // Translate so rotated bbox min corner lands at (x,y).
                    let tx = x - v.minx;
                    let ty = y - v.miny;

                    if stock_geom.contains_transformed_polygon(&v.pts_rot, tx, ty) {
                        let placed = Rect { x, y, w: v.w, h: v.h };
                        return Some((
                            Placement {
                                stock: 0, // filled by caller
                                tx,
                                ty,
                                angle: v.angle,
                            },
                            placed,
                        ));
                    }
                }
            }
            None
        }
        BinMode::Shelf => {
            let shelf = bin.shelf.as_ref().unwrap();
            let mut cands: Vec<(f64, f64)> = Vec::new();

            for v in variants {
                if v.w > container.w + 1e-9 || v.h > container.h + 1e-9 {
                    continue;
                }
                shelf.candidates(v.w, v.h, &mut cands);
                for &(x, y) in &cands {
                    let tx = x - v.minx;
                    let ty = y - v.miny;
                    if stock_geom.contains_transformed_polygon(&v.pts_rot, tx, ty) {
                        let placed = Rect { x, y, w: v.w, h: v.h };
                        return Some((
                            Placement {
                                stock: 0, // filled by caller
                                tx,
                                ty,
                                angle: v.angle,
                            },
                            placed,
                        ));
                    }
                }
            }
            None
        }
    }
}

fn commit_place(bin: &mut StockBin, placed: Rect, x_for_shelf: f64, w_for_shelf: f64, h_for_shelf: f64) {
    match bin.mode {
        BinMode::MaxRects => {
            bin.maxrects.as_mut().unwrap().place_rect(placed);
        }
        BinMode::Shelf => {
            // The shelf cursor has to be advanced according to the chosen x.
            let shelf = bin.shelf.as_mut().unwrap();
            let _ = shelf.try_advance_and_place(x_for_shelf, w_for_shelf, h_for_shelf);
        }
    }
}

fn fallback_single_stock_search(
    stock_geom: &mut StockGeom,
    piece_poly: &[Point],
    rng: &mut XorShift64,
) -> Option<Placement> {
    // Randomized search: open stock is empty so only containment matters.
    let bbox = stock_geom.bbox;

    // Precompute a simple "centroid" (average of vertices) for translation target.
    let mut cx = 0.0;
    let mut cy = 0.0;
    for &p in piece_poly {
        cx += p.x;
        cy += p.y;
    }
    let inv = 1.0 / (piece_poly.len().max(1) as f64);
    cx *= inv;
    cy *= inv;

    // Try angles + positions.
    let attempts = 2000usize;
    for _ in 0..attempts {
        let angle = rng.next_f64() * 2.0 * PI;
        let (sin_a, cos_a) = angle.sin_cos();

        // Build rotated points on the fly (no allocation reuse: keep it simple but bounded attempts).
        let mut pts_rot: Vec<Point> = Vec::with_capacity(piece_poly.len());
        let mut minx = f64::INFINITY;
        let mut miny = f64::INFINITY;
        let mut maxx = f64::NEG_INFINITY;
        let mut maxy = f64::NEG_INFINITY;
        for &p in piece_poly {
            let r = rotate_point(p, sin_a, cos_a);
            minx = minx.min(r.x);
            miny = miny.min(r.y);
            maxx = maxx.max(r.x);
            maxy = maxy.max(r.y);
            pts_rot.push(r);
        }
        let w = (maxx - minx).max(0.0);
        let h = (maxy - miny).max(0.0);
        if w > bbox.w + 1e-9 || h > bbox.h + 1e-9 {
            continue;
        }

        // Random target point in bbox; translate so (rotated average) lands there.
        let px = bbox.x + rng.next_f64() * bbox.w;
        let py = bbox.y + rng.next_f64() * bbox.h;

        let c_rot = rotate_point(Point { x: cx, y: cy }, sin_a, cos_a);
        let tx = px - c_rot.x;
        let ty = py - c_rot.y;

        // Quick bbox-in-bbox
        let bb2 = Rect { x: minx + tx, y: miny + ty, w, h };
        if !bbox.contains(&bb2) {
            continue;
        }

        if stock_geom.contains_transformed_polygon(&pts_rot, tx, ty) {
            return Some(Placement { stock: 0, tx, ty, angle });
        }
    }
    None
}

/* ---------------- Input ---------------- */

struct Scanner {
    data: Vec<u8>,
    i: usize,
}
impl Scanner {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data).unwrap();
        Self { data, i: 0 }
    }
    #[inline]
    fn skip_ws(&mut self) {
        while self.i < self.data.len() && self.data[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
    }
    fn next_usize(&mut self) -> usize {
        self.skip_ws();
        let mut s = 0usize;
        while self.i < self.data.len() && !self.data[self.i].is_ascii_whitespace() {
            s = s * 10 + (self.data[self.i] - b'0') as usize;
            self.i += 1;
        }
        s
    }
    fn next_f64(&mut self) -> f64 {
        self.skip_ws();
        let start = self.i;
        while self.i < self.data.len() && !self.data[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
        // Safety: parse from slice
        std::str::from_utf8(&self.data[start..self.i]).unwrap().parse::<f64>().unwrap()
    }
}

fn main() {
    let mut sc = Scanner::new();

    let sv = sc.next_usize();
    let np = sc.next_usize();

    let mut stock_poly = Vec::<Point>::with_capacity(sv);
    for _ in 0..sv {
        let x = sc.next_f64();
        let y = sc.next_f64();
        stock_poly.push(Point { x, y });
    }

    let mut stock_geom = StockGeom::new(stock_poly);

    let mut pieces_raw: Vec<Vec<Point>> = Vec::with_capacity(np);
    let mut total_vertices: usize = 0;
    for _ in 0..np {
        let pv = sc.next_usize();
        total_vertices += pv;
        let mut poly = Vec::<Point>::with_capacity(pv);
        for _ in 0..pv {
            let x = sc.next_f64();
            let y = sc.next_f64();
            poly.push(Point { x, y });
        }
        pieces_raw.push(poly);
    }

    let rot_angles = choose_rotations(np, total_vertices);

    // Build pieces with variants
    let mut pieces: Vec<Piece> = Vec::with_capacity(np);
    for (idx, poly) in pieces_raw.into_iter().enumerate() {
        let area_abs = polygon_signed_area(&poly).abs();
        let mut variants: Vec<Variant> = rot_angles.iter().map(|&a| build_variant(&poly, a)).collect();

        // If none fit stock bbox, try to add one extra sampled angle.
        let fits_any = variants.iter().any(|v| v.w <= stock_geom.bbox.w + 1e-9 && v.h <= stock_geom.bbox.h + 1e-9);
        if !fits_any {
            if let Some(extra) = best_extra_angle_if_needed(&poly, stock_geom.bbox) {
                // Avoid duplicates
                let dup = variants.iter().any(|v| (v.angle - extra).abs() < 1e-12);
                if !dup {
                    variants.push(build_variant(&poly, extra));
                }
            }
        }

        // Sort variants by bbox area ascending (try tighter first)
        variants.sort_by(|a, b| (a.w * a.h).partial_cmp(&(b.w * b.h)).unwrap_or(Ordering::Equal));

        pieces.push(Piece {
            idx,
            poly,
            area_abs,
            variants,
        });
    }

    // Place larger first
    pieces.sort_by(|a, b| b.area_abs.partial_cmp(&a.area_abs).unwrap_or(Ordering::Equal));

    // Adapt bin mode to scale
    let mode = if np > 1500 || total_vertices > 1_000_000 {
        BinMode::Shelf
    } else {
        BinMode::MaxRects
    };

    let mut bins: Vec<StockBin> = Vec::new();
    bins.push(StockBin::new(mode, stock_geom.bbox));

    let mut placements: Vec<Option<Placement>> = vec![None; np];

    let mut rng = XorShift64::new(0xC0FFEE_u64 ^ (np as u64).wrapping_mul(0x9E3779B97F4A7C15));

    for piece in &pieces {
        let mut placed_any = false;

        // Try existing bins
        for (bi, bin) in bins.iter_mut().enumerate() {
            if let Some((mut pl, placed_rect)) = try_place_piece_in_bin(&mut stock_geom, bin, &piece.variants) {
                pl.stock = bi;
                // Commit placement into bin
                match bin.mode {
                    BinMode::MaxRects => {
                        commit_place(bin, placed_rect, placed_rect.x, placed_rect.w, placed_rect.h);
                    }
                    BinMode::Shelf => {
                        commit_place(bin, placed_rect, placed_rect.x, placed_rect.w, placed_rect.h);
                    }
                }
                placements[piece.idx] = Some(pl);
                placed_any = true;
                break;
            } else if let BinMode::Shelf = bin.mode {
                // For shelf, if we couldn't place at current shelf, try new shelf once and retry quickly.
                let shelf = bin.shelf.as_mut().unwrap();
                if shelf.new_shelf() {
                    if let Some((mut pl, placed_rect)) = try_place_piece_in_bin(&mut stock_geom, bin, &piece.variants) {
                        pl.stock = bi;
                        commit_place(bin, placed_rect, placed_rect.x, placed_rect.w, placed_rect.h);
                        placements[piece.idx] = Some(pl);
                        placed_any = true;
                        break;
                    }
                }
            }
        }

        if placed_any {
            continue;
        }

        // Open new bin and try again
        let new_index = bins.len();
        bins.push(StockBin::new(mode, stock_geom.bbox));
        {
            let bin = bins.last_mut().unwrap();
            if let Some((mut pl, placed_rect)) = try_place_piece_in_bin(&mut stock_geom, bin, &piece.variants) {
                pl.stock = new_index;
                commit_place(bin, placed_rect, placed_rect.x, placed_rect.w, placed_rect.h);
                placements[piece.idx] = Some(pl);
                continue;
            }
        }

        // Fallback: put this piece alone into its own new stock and do randomized containment search.
        let solo_index = bins.len();
        bins.push(StockBin::new(mode, stock_geom.bbox));
        if let Some(mut pl) = fallback_single_stock_search(&mut stock_geom, &piece.poly, &mut rng) {
            pl.stock = solo_index;
            placements[piece.idx] = Some(pl);
        } else {
            // Last resort (should be rare): output a trivial placement (may be invalid if instance is pathological).
            placements[piece.idx] = Some(Placement {
                stock: solo_index,
                tx: 0.0,
                ty: 0.0,
                angle: 0.0,
            });
        }
    }

    // Count used stocks: we conservatively report all bins created (they are valid indices).
    // (If you want to compress unused bins, you'd need to track if any piece used a bin.)
    let mut used = vec![false; bins.len()];
    for p in &placements {
        if let Some(pl) = p {
            used[pl.stock] = true;
        }
    }
    // Build remap to compress indices
    let mut remap = vec![usize::MAX; bins.len()];
    let mut num_used = 0usize;
    for i in 0..bins.len() {
        if used[i] {
            remap[i] = num_used;
            num_used += 1;
        }
    }

    println!("{}", num_used);
    for i in 0..np {
        if let Some(pl) = placements[i] {
            let si = remap[pl.stock];
            let si = if si == usize::MAX { 0 } else { si };
            println!("{si} {:.10} {:.10} {:.10}", pl.tx, pl.ty, pl.angle);
        } else {
            println!("0 0 0 0");
        }
    }
}
