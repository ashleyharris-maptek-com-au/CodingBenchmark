; SPIR-V
; Version: 1.0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4v4 %mat4v4 %mat4v4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%functype = OpTypeFunction %void

%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%float_0 = OpConstant %float 0
%float_0_2 = OpConstant %float 0.2
%float_0_3 = OpConstant %float 0.3
%float_0_7 = OpConstant %float 0.7
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1
%float_1_5 = OpConstant %float 1.5
%float_3 = OpConstant %float 3

%baseColorConst = OpConstantComposite %v3float %float_0_8 %float_0_2 %float_0_2
%fogColorConst = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_0_8

%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%inWorldPos = OpVariable %_ptr_Input_v3float Input
%inNormal = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output

%main = OpFunction %void None %functype
%entry = OpLabel
%worldPos = OpLoad %v3float %inWorldPos
%normal = OpLoad %v3float %inNormal
%nrm = OpExtInst %v3float %1 Normalize %normal

%lightPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%light4 = OpLoad %v4float %lightPtr
%lightPos = OpVectorShuffle %v3float %light4 %light4 0 1 2

%Lvec = OpFSub %v3float %lightPos %worldPos
%Ldir = OpExtInst %v3float %1 Normalize %Lvec
%ndotl = OpDot %float %nrm %Ldir
%diffuse = OpExtInst %float %1 FMax %ndotl %float_0
%diffScaled = OpFMul %float %float_0_7 %diffuse
%lambert = OpFAdd %float %float_0_3 %diffScaled
%baseLit = OpVectorTimesScalar %v3float %baseColorConst %lambert

%camPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_4
%cam4 = OpLoad %v4float %camPtr
%cameraPos = OpVectorShuffle %v3float %cam4 %cam4 0 1 2

%toCam = OpFSub %v3float %cameraPos %worldPos
%dist = OpExtInst %float %1 Length %toCam
%fogNumer = OpFSub %float %dist %float_1_5
%fogUnclamped = OpFDiv %float %fogNumer %float_3
%fog = OpExtInst %float %1 FClamp %fogUnclamped %float_0 %float_1

%oneMinusFog = OpFSub %float %float_1 %fog
%basePart = OpVectorTimesScalar %v3float %baseLit %oneMinusFog
%fogPart = OpVectorTimesScalar %v3float %fogColorConst %fog
%finalRgb = OpFAdd %v3float %basePart %fogPart

%r = OpCompositeExtract %float %finalRgb 0
%g = OpCompositeExtract %float %finalRgb 1
%b = OpCompositeExtract %float %finalRgb 2
%finalColor = OpCompositeConstruct %v4float %r %g %b %float_1
OpStore %outColor %finalColor
OpReturn
OpFunctionEnd
