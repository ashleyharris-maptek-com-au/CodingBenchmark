```cpp
#include <bits/stdc++.h>
using namespace std;

// ------------------------------------------------------------
// Job-Shop Scheduling: scalable constructive heuristic solver.
// Produces a feasible schedule for any size (tiny to very large).
//
// Output format per job line: (start_time machine) repeated for all tasks.
// ------------------------------------------------------------

using i64 = long long;

// Global mode for priority rules (set per run)
static int g_mode = 0;

// Fast RNG (splitmix64)
struct SplitMix64 {
    uint64_t x;
    explicit SplitMix64(uint64_t seed) : x(seed) {}
    uint64_t nextU64() {
        uint64_t z = (x += 0x9e3779b97f4a7c15ULL);
        z = (z ^ (z >> 30)) * 0xbf58476d1ce4e5b9ULL;
        z = (z ^ (z >> 27)) * 0x94d049bb133111ebULL;
        return z ^ (z >> 31);
    }
    uint32_t nextU32() { return (uint32_t)nextU64(); }
};

struct Entry {
    int job = -1;
    int opInJob = -1;
    int idx = -1;        // flat index of operation
    int machine = -1;
    int dur = 0;
    i64 release = 0;     // time when job is ready for this operation
    i64 rem = 0;         // remaining processing time in job from this op (including it)
    uint32_t rnd = 0;    // random tie-break
};

// futurePQ: smallest release first
struct FutureCmp {
    bool operator()(Entry const& a, Entry const& b) const {
        if (a.release != b.release) return a.release > b.release; // min-heap by release
        if (a.rnd != b.rnd) return a.rnd > b.rnd;
        if (a.job != b.job) return a.job > b.job;
        return a.opInJob > b.opInJob;
    }
};

// readyPQ: heuristic ordering among operations that can start when machine is free
// priority_queue top() returns "best" (according to comparator rules below)
struct ReadyCmp {
    bool operator()(Entry const& a, Entry const& b) const {
        // Return true if a is WORSE than b (so b comes on top).
        // Modes:
        // 0: MWKR then SPT     (more remaining work first, then shorter processing time)
        // 1: SPT then MWKR
        // 2: MWKR then random then SPT
        // 3: random then MWKR then SPT
        if (g_mode == 1) {
            if (a.dur != b.dur) return a.dur > b.dur;          // shorter dur better
            if (a.rem != b.rem) return a.rem < b.rem;          // larger rem better
        } else if (g_mode == 2) {
            if (a.rem != b.rem) return a.rem < b.rem;
            if (a.rnd != b.rnd) return a.rnd > b.rnd;
            if (a.dur != b.dur) return a.dur > b.dur;
        } else if (g_mode == 3) {
            if (a.rnd != b.rnd) return a.rnd > b.rnd;
            if (a.rem != b.rem) return a.rem < b.rem;
            if (a.dur != b.dur) return a.dur > b.dur;
        } else { // g_mode == 0 default
            if (a.rem != b.rem) return a.rem < b.rem;
            if (a.dur != b.dur) return a.dur > b.dur;
        }

        // Stable ties:
        if (a.release != b.release) return a.release > b.release;
        if (a.job != b.job) return a.job > b.job;
        return a.opInJob > b.opInJob;
    }
};

struct MachineState {
    i64 readyTime = 0;
    uint64_t ver = 0;
    std::priority_queue<Entry, vector<Entry>, ReadyCmp> readyPQ;
    std::priority_queue<Entry, vector<Entry>, FutureCmp> futurePQ;
};

struct MachineKey {
    i64 est = 0;         // estimated next start time for this machine
    i64 rem = 0;
    int dur = 0;
    uint32_t rnd = 0;
    int machine = -1;
    uint64_t ver = 0;
};

// globalPQ: choose machine with smallest next-start time; tie by operation heuristic
struct MachineKeyCmp {
    bool operator()(MachineKey const& a, MachineKey const& b) const {
        // Return true if a is WORSE than b (so b on top).
        if (a.est != b.est) return a.est > b.est;

        if (g_mode == 1) {
            if (a.dur != b.dur) return a.dur > b.dur;
            if (a.rem != b.rem) return a.rem < b.rem;
        } else if (g_mode == 2) {
            if (a.rem != b.rem) return a.rem < b.rem;
            if (a.rnd != b.rnd) return a.rnd > b.rnd;
            if (a.dur != b.dur) return a.dur > b.dur;
        } else if (g_mode == 3) {
            if (a.rnd != b.rnd) return a.rnd > b.rnd;
            if (a.rem != b.rem) return a.rem < b.rem;
            if (a.dur != b.dur) return a.dur > b.dur;
        } else {
            if (a.rem != b.rem) return a.rem < b.rem;
            if (a.dur != b.dur) return a.dur > b.dur;
        }

        return a.machine > b.machine;
    }
};

struct Instance {
    int J = 0, M = 0;
    vector<int> jobLen;
    vector<int> jobOff;
    int totalOps = 0;

    vector<int> opMachine;   // size totalOps
    vector<int> opDur;       // size totalOps
    vector<i64> opRem;       // size totalOps, remaining work suffix within each job
};

struct Result {
    i64 makespan = (i64)4e18;
    vector<i64> start; // size totalOps
};

// Promote operations whose release <= machine readyTime from futurePQ into readyPQ
static inline void promote(MachineState &ms) {
    while (!ms.futurePQ.empty() && ms.futurePQ.top().release <= ms.readyTime) {
        ms.readyPQ.push(ms.futurePQ.top());
        ms.futurePQ.pop();
    }
}

static inline bool computeMachineKey(MachineState &ms, int m, MachineKey &outKey) {
    promote(ms);
    if (!ms.readyPQ.empty()) {
        const Entry &e = ms.readyPQ.top();
        outKey.est = ms.readyTime;
        outKey.rem = e.rem;
        outKey.dur = e.dur;
        outKey.rnd = e.rnd;
        outKey.machine = m;
        outKey.ver = ms.ver;
        return true;
    }
    if (!ms.futurePQ.empty()) {
        const Entry &e = ms.futurePQ.top();
        // No ready ops; machine will idle until the earliest release.
        outKey.est = e.release; // e.release > ms.readyTime due to promote()
        outKey.rem = e.rem;
        outKey.dur = e.dur;
        outKey.rnd = e.rnd;
        outKey.machine = m;
        outKey.ver = ms.ver;
        return true;
    }
    return false;
}

static Result solveOneRun(const Instance &ins, int mode, uint64_t seed) {
    g_mode = mode;
    SplitMix64 rng(seed);

    Result res;
    res.start.assign(ins.totalOps, -1);

    vector<MachineState> machines(ins.M);
    vector<int> nextOpInJob(ins.J, 0);
    vector<i64> jobEnd(ins.J, 0);

    // Global PQ over machines
    std::priority_queue<MachineKey, vector<MachineKey>, MachineKeyCmp> globalPQ;

    auto touch = [&](int m) {
        machines[m].ver++;
        MachineKey key;
        if (computeMachineKey(machines[m], m, key)) globalPQ.push(key);
    };

    auto addOpToMachine = [&](int job, int opInJob, int idx, int m, int dur, i64 release, i64 rem) {
        Entry e;
        e.job = job;
        e.opInJob = opInJob;
        e.idx = idx;
        e.machine = m;
        e.dur = dur;
        e.release = release;
        e.rem = rem;
        // Randomization only in some modes; still cheap.
        e.rnd = (mode >= 2 ? rng.nextU32() : 0u);

        MachineState &ms = machines[m];
        if (release <= ms.readyTime) ms.readyPQ.push(e);
        else ms.futurePQ.push(e);

        touch(m);
    };

    // Initialize: first operation of each job is released at t=0.
    for (int j = 0; j < ins.J; ++j) {
        if (ins.jobLen[j] == 0) continue;
        int op = 0;
        int idx = ins.jobOff[j] + op;
        addOpToMachine(j, op, idx, ins.opMachine[idx], ins.opDur[idx], 0, ins.opRem[idx]);
    }

    int scheduled = 0;
    while (scheduled < ins.totalOps) {
        if (globalPQ.empty()) {
            // Should not happen for valid instances; but guard anyway.
            // (Would indicate unreachable ops, which shouldn't exist in JSSP.)
            break;
        }

        MachineKey mk = globalPQ.top();
        globalPQ.pop();

        int m = mk.machine;
        MachineState &ms = machines[m];

        if (mk.ver != ms.ver) continue; // stale

        // Recompute key to be safe (promote may change top); if changed, push updated and continue.
        MachineKey cur;
        if (!computeMachineKey(ms, m, cur)) continue;
        if (cur.ver != mk.ver || cur.est != mk.est || cur.rem != mk.rem || cur.dur != mk.dur || cur.rnd != mk.rnd) {
            globalPQ.push(cur);
            continue;
        }

        promote(ms);

        Entry e;
        i64 startTime = 0;
        if (!ms.readyPQ.empty()) {
            e = ms.readyPQ.top();
            ms.readyPQ.pop();
            startTime = ms.readyTime; // machine is free at readyTime, op already released
        } else {
            // Must wait for earliest release
            e = ms.futurePQ.top();
            ms.futurePQ.pop();
            if (ms.readyTime < e.release) ms.readyTime = e.release;
            startTime = ms.readyTime;
        }

        // Schedule it
        res.start[e.idx] = startTime;
        i64 endTime = startTime + (i64)e.dur;

        ms.readyTime = endTime;
        touch(m); // machine state changed (time advanced, may promote future -> ready inside computeKey)

        // Advance job
        jobEnd[e.job] = endTime;
        nextOpInJob[e.job]++;

        // Add next operation of this job (if any)
        int nxt = nextOpInJob[e.job];
        if (nxt < ins.jobLen[e.job]) {
            int idx2 = ins.jobOff[e.job] + nxt;
            int m2 = ins.opMachine[idx2];
            int d2 = ins.opDur[idx2];
            addOpToMachine(e.job, nxt, idx2, m2, d2, endTime, ins.opRem[idx2]);
        }

        scheduled++;
    }

    // Makespan = max machine readyTime
    i64 makespan = 0;
    for (int m = 0; m < ins.M; ++m) makespan = max(makespan, machines[m].readyTime);
    res.makespan = makespan;
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Instance ins;
    if (!(cin >> ins.J >> ins.M)) return 0;

    ins.jobLen.assign(ins.J, 0);
    ins.jobOff.assign(ins.J, 0);

    // First pass: read jobs, store temporarily per-job then flatten.
    vector<vector<pair<int,int>>> jobs(ins.J);
    ins.totalOps = 0;

    for (int j = 0; j < ins.J; ++j) {
        int T;
        cin >> T;
        ins.jobLen[j] = T;
        jobs[j].reserve(max(0, T));
        for (int t = 0; t < T; ++t) {
            int m, d;
            cin >> m >> d;
            jobs[j].push_back({m, d});
        }
        ins.totalOps += T;
    }

    ins.opMachine.resize(ins.totalOps);
    ins.opDur.resize(ins.totalOps);
    ins.opRem.resize(ins.totalOps);

    int ptr = 0;
    for (int j = 0; j < ins.J; ++j) {
        ins.jobOff[j] = ptr;
        for (int t = 0; t < ins.jobLen[j]; ++t) {
            ins.opMachine[ptr] = jobs[j][t].first;
            ins.opDur[ptr] = jobs[j][t].second;
            ptr++;
        }
    }

    // Precompute remaining processing time suffix per job
    for (int j = 0; j < ins.J; ++j) {
        i64 suf = 0;
        for (int t = ins.jobLen[j] - 1; t >= 0; --t) {
            int idx = ins.jobOff[j] + t;
            suf += (i64)ins.opDur[idx];
            ins.opRem[idx] = suf;
        }
    }

    // Lower bound (simple): max over job sums and machine sums
    i64 lbJob = 0;
    for (int j = 0; j < ins.J; ++j) {
        i64 sum = 0;
        for (int t = 0; t < ins.jobLen[j]; ++t) sum += ins.opDur[ins.jobOff[j] + t];
        lbJob = max(lbJob, sum);
    }
    vector<i64> machineSum(ins.M, 0);
    for (int idx = 0; idx < ins.totalOps; ++idx) machineSum[ins.opMachine[idx]] += ins.opDur[idx];
    i64 lbMach = 0;
    for (int m = 0; m < ins.M; ++m) lbMach = max(lbMach, machineSum[m]);
    i64 lowerBound = max(lbJob, lbMach);

    // Adaptive number of runs based on size
    int N = ins.totalOps;
    int runs = 1;
    if (N <= 80) runs = 5000;
    else if (N <= 200) runs = 2000;
    else if (N <= 1000) runs = 300;
    else if (N <= 5000) runs = 60;
    else if (N <= 20000) runs = 12;
    else runs = 3;

    // Strategy selection (adaptive):
    // - Small: many randomized runs over different tie rules
    // - Large: a few runs (deterministic MWKR/SPT + a randomized variant)
    Result best;
    best.makespan = (i64)4e18;

    uint64_t baseSeed = 0xC0FFEE123456789ULL;

    for (int r = 0; r < runs; ++r) {
        int mode;
        if (N <= 1000) mode = (r % 4);                 // explore all modes
        else mode = (r % 2 == 0 ? 0 : 2);              // MWKR/SPT and MWKR+random

        uint64_t seed = baseSeed + (uint64_t)r * 0x9E3779B97F4A7C15ULL;

        Result cur = solveOneRun(ins, mode, seed);
        if (cur.makespan < best.makespan) {
            best = std::move(cur);
            if (best.makespan <= lowerBound) break; // cannot beat the lower bound
        }
    }

    // Output: for each job, print T pairs: start_time machine
    // (Machine is fixed by the instance; start_time from best schedule.)
    for (int j = 0; j < ins.J; ++j) {
        int T = ins.jobLen[j];
        for (int t = 0; t < T; ++t) {
            int idx = ins.jobOff[j] + t;
            if (t) cout << ' ';
            cout << best.start[idx] << ' ' << ins.opMachine[idx];
        }
        cout << '\n';
    }
    return 0;
}
```
