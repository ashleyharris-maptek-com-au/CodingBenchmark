import math

# ------------------------ Module-level persistent state ------------------------

_EST = {
    "init": False,
    "pos": [0.0, 0.0, 0.0],   # m, LVLH
    "vel": [0.0, 0.0, 0.0],   # m/s, LVLH
    "last_meas_pos": None,
    "last_meas_vel": None,
    "last_meas_t": None,
    "t": 0.0,
}

# ----------------------------- Small math helpers -----------------------------

def _clip(x, lo, hi):
    return lo if x < lo else hi if x > hi else x

def _norm3(v):
    return math.sqrt(v[0]*v[0] + v[1]*v[1] + v[2]*v[2])

def _dot3(a, b):
    return a[0]*b[0] + a[1]*b[1] + a[2]*b[2]

def _add3(a, b):
    return [a[0]+b[0], a[1]+b[1], a[2]+b[2]]

def _sub3(a, b):
    return [a[0]-b[0], a[1]-b[1], a[2]-b[2]]

def _mul3(s, v):
    return [s*v[0], s*v[1], s*v[2]]

def _is_valid_num(x):
    return x is not None and isinstance(x, (int, float)) and math.isfinite(x)

def _get(state, key, default=None):
    return state[key] if key in state else default

def _median(vals):
    vals = [v for v in vals if _is_valid_num(v)]
    if not vals:
        return None
    vals.sort()
    n = len(vals)
    mid = n // 2
    return vals[mid] if n % 2 else 0.5 * (vals[mid - 1] + vals[mid])

def _robust_vec3(candidates):
    """
    candidates: list of tuples (x, x_valid, y, y_valid, z, z_valid)
    Returns (vec, valid)
    Uses median per axis across valid candidates.
    """
    xs, ys, zs = [], [], []
    for x, xv, y, yv, z, zv in candidates:
        if xv and _is_valid_num(x): xs.append(x)
        if yv and _is_valid_num(y): ys.append(y)
        if zv and _is_valid_num(z): zs.append(z)
    mx, my, mz = _median(xs), _median(ys), _median(zs)
    if mx is None or my is None or mz is None:
        return ([0.0, 0.0, 0.0], False)
    return ([mx, my, mz], True)

def _robust_scalar(vals):
    """vals: list of tuples (value, valid)"""
    good = [v for v, ok in vals if ok and _is_valid_num(v)]
    if not good:
        return (0.0, False)
    good.sort()
    return (good[len(good)//2], True)

# --------------------------- Sensor selection / fusion -------------------------

def _pick_rel_meas(state):
    """
    Returns:
      pos_meas (vec3), pos_ok
      vel_meas (vec3), vel_ok
      range_meas (float), range_ok
    Preference: lidar > radar > docking camera (bearing+range) > rangefinders (range only)
    """
    # Lidar
    lidar_pos, lidar_pos_ok = _robust_vec3([(
        _get(state, "lidar_rel_pos_x_m"), bool(_get(state, "lidar_rel_x_valid")),
        _get(state, "lidar_rel_pos_y_m"), bool(_get(state, "lidar_rel_y_valid")),
        _get(state, "lidar_rel_pos_z_m"), bool(_get(state, "lidar_rel_z_valid")),
    )])
    lidar_vel, lidar_vel_ok = _robust_vec3([(
        _get(state, "lidar_rel_vel_x_ms"), bool(_get(state, "lidar_relv_x_valid")),
        _get(state, "lidar_rel_vel_y_ms"), bool(_get(state, "lidar_relv_y_valid")),
        _get(state, "lidar_rel_vel_z_ms"), bool(_get(state, "lidar_relv_z_valid")),
    )])

    # Radar
    radar_pos, radar_pos_ok = _robust_vec3([(
        _get(state, "radar_rel_pos_x_m"), bool(_get(state, "radar_rel_x_valid")),
        _get(state, "radar_rel_pos_y_m"), bool(_get(state, "radar_rel_y_valid")),
        _get(state, "radar_rel_pos_z_m"), bool(_get(state, "radar_rel_z_valid")),
    )])
    radar_vel, radar_vel_ok = _robust_vec3([(
        _get(state, "radar_rel_vel_x_ms"), bool(_get(state, "radar_relv_x_valid")),
        _get(state, "radar_rel_vel_y_ms"), bool(_get(state, "radar_relv_y_valid")),
        _get(state, "radar_rel_vel_z_ms"), bool(_get(state, "radar_relv_z_valid")),
    )])

    # Docking camera (bearing + range). Assumed to be in LVLH axes.
    cam_active = bool(_get(state, "dock_camera_active"))
    cam_range = _get(state, "dock_camera_range_m")
    bx, by, bz = _get(state, "dock_camera_bearing_x"), _get(state, "dock_camera_bearing_y"), _get(state, "dock_camera_bearing_z")
    bxv = bool(_get(state, "dock_camera_bearing_x_valid"))
    byv = bool(_get(state, "dock_camera_bearing_y_valid"))
    bzv = bool(_get(state, "dock_camera_bearing_z_valid"))
    cam_ok = cam_active and bxv and byv and bzv and _is_valid_num(bx) and _is_valid_num(by) and _is_valid_num(bz) and _is_valid_num(cam_range) and cam_range > 0.0
    if cam_ok:
        bnorm = math.sqrt(bx*bx + by*by + bz*bz)
        if bnorm > 1e-9:
            cam_pos = [cam_range * bx / bnorm, cam_range * by / bnorm, cam_range * bz / bnorm]
            cam_pos_ok = True
        else:
            cam_pos, cam_pos_ok = [0.0, 0.0, 0.0], False
    else:
        cam_pos, cam_pos_ok = [0.0, 0.0, 0.0], False

    # Rangefinders (scalar only)
    r1, r1v = _get(state, "range_1_m"), bool(_get(state, "range_1_valid"))
    r2, r2v = _get(state, "range_2_m"), bool(_get(state, "range_2_valid"))
    r3, r3v = _get(state, "range_3_m"), bool(_get(state, "range_3_valid"))
    rf_range, rf_ok = _robust_scalar([(r1, r1v), (r2, r2v), (r3, r3v)])

    # Choose best pos/vel sources with graceful degradation
    if lidar_pos_ok:
        pos_meas, pos_ok = lidar_pos, True
    elif radar_pos_ok:
        pos_meas, pos_ok = radar_pos, True
    elif cam_pos_ok:
        pos_meas, pos_ok = cam_pos, True
    else:
        pos_meas, pos_ok = [0.0, 0.0, 0.0], False

    if lidar_vel_ok:
        vel_meas, vel_ok = lidar_vel, True
    elif radar_vel_ok:
        vel_meas, vel_ok = radar_vel, True
    else:
        vel_meas, vel_ok = [0.0, 0.0, 0.0], False

    # Range (prefer from chosen pos; else camera; else rangefinders)
    if pos_ok:
        range_meas, range_ok = _norm3(pos_meas), True
    elif cam_ok:
        range_meas, range_ok = float(cam_range), True
    elif rf_ok:
        range_meas, range_ok = float(rf_range), True
    else:
        range_meas, range_ok = 0.0, False

    return pos_meas, pos_ok, vel_meas, vel_ok, range_meas, range_ok

def _robust_accel(state):
    ax, ax_ok = _robust_scalar([
        (_get(state, "accel_1_x_ms2"), bool(_get(state, "accel_1_x_valid"))),
        (_get(state, "accel_2_x_ms2"), bool(_get(state, "accel_2_x_valid"))),
        (_get(state, "accel_3_x_ms2"), bool(_get(state, "accel_3_x_valid"))),
    ])
    ay, ay_ok = _robust_scalar([
        (_get(state, "accel_1_y_ms2"), bool(_get(state, "accel_1_y_valid"))),
        (_get(state, "accel_2_y_ms2"), bool(_get(state, "accel_2_y_valid"))),
        (_get(state, "accel_3_y_ms2"), bool(_get(state, "accel_3_y_valid"))),
    ])
    az, az_ok = _robust_scalar([
        (_get(state, "accel_1_z_ms2"), bool(_get(state, "accel_1_z_valid"))),
        (_get(state, "accel_2_z_ms2"), bool(_get(state, "accel_2_z_valid"))),
        (_get(state, "accel_3_z_ms2"), bool(_get(state, "accel_3_z_valid"))),
    ])
    ok = ax_ok and ay_ok and az_ok
    return ([ax, ay, az], ok)

def _robust_gyro_rates(state):
    # deg/s -> rad/s, median over gyros, per-axis independent
    pr = _median([
        _get(state, "gyro_1_pitch_dps") if bool(_get(state, "gyro_1_pitch_valid")) else None,
        _get(state, "gyro_2_pitch_dps") if bool(_get(state, "gyro_2_pitch_valid")) else None,
        _get(state, "gyro_3_pitch_dps") if bool(_get(state, "gyro_3_pitch_valid")) else None,
    ])
    rr = _median([
        _get(state, "gyro_1_roll_dps") if bool(_get(state, "gyro_1_roll_valid")) else None,
        _get(state, "gyro_2_roll_dps") if bool(_get(state, "gyro_2_roll_valid")) else None,
        _get(state, "gyro_3_roll_dps") if bool(_get(state, "gyro_3_roll_valid")) else None,
    ])
    yr = _median([
        _get(state, "gyro_1_yaw_dps") if bool(_get(state, "gyro_1_yaw_valid")) else None,
        _get(state, "gyro_2_yaw_dps") if bool(_get(state, "gyro_2_yaw_valid")) else None,
        _get(state, "gyro_3_yaw_dps") if bool(_get(state, "gyro_3_yaw_valid")) else None,
    ])
    if pr is None or rr is None or yr is None:
        return ([0.0, 0.0, 0.0], False)
    d2r = math.pi / 180.0
    return ([rr * d2r, pr * d2r, yr * d2r], True)  # roll, pitch, yaw order

def _mean_motion_n(state):
    # Prefer ground station a & speed: for near-circular orbit, n ≈ v/a
    a = _get(state, "ground_semi_major_axis_m")
    av = bool(_get(state, "ground_semi_major_axis_m_valid"))
    v = _get(state, "ground_speed_ms")
    vv = bool(_get(state, "ground_speed_ms_valid"))
    if av and vv and _is_valid_num(a) and _is_valid_num(v) and a > 1.0:
        n = abs(v) / a
        return _clip(n, 5e-4, 2.5e-3)

    # Else use altitude if available
    alt_km = _get(state, "ground_altitude_km")
    altv = bool(_get(state, "ground_altitude_km_valid"))
    if altv and _is_valid_num(alt_km):
        mu = 3.986004418e14
        Re = 6371e3
        a2 = Re + alt_km * 1000.0
        if a2 > Re:
            n = math.sqrt(mu / (a2**3))
            return _clip(n, 5e-4, 2.5e-3)

    # Default LEO ~ 90 min
    return 0.0011

# ------------------------------ Main autopilot -------------------------------

def autopilot_step(state: dict, dt: float) -> dict:
    global _EST
    dt = float(dt) if _is_valid_num(dt) and dt > 1e-4 else 0.1
    _EST["t"] += dt

    # --- Measurements ---
    pos_meas, pos_ok, vel_meas, vel_ok, range_meas, range_ok = _pick_rel_meas(state)
    accel_meas, accel_ok = _robust_accel(state)

    # --- Initialize estimator on first good measurement ---
    if (not _EST["init"]) and pos_ok:
        _EST["pos"] = pos_meas[:]
        _EST["vel"] = vel_meas[:] if vel_ok else [0.0, 0.0, 0.0]
        _EST["init"] = True
        _EST["last_meas_pos"] = pos_meas[:]
        _EST["last_meas_vel"] = vel_meas[:] if vel_ok else None
        _EST["last_meas_t"] = _EST["t"]

    # --- Estimator: alpha-beta with accel-aided propagation ---
    pos = _EST["pos"][:]
    vel = _EST["vel"][:]

    # Predict
    if accel_ok:
        vel = _add3(vel, _mul3(dt, accel_meas))
    pos = _add3(pos, _mul3(dt, vel))

    # Update gains (more smoothing when sensors flicker/noisy)
    # Slightly more aggressive when close.
    r_pred = _norm3(pos)
    if r_pred < 10.0:
        alpha, beta = 0.75, 0.55
    elif r_pred < 200.0:
        alpha, beta = 0.60, 0.40
    else:
        alpha, beta = 0.45, 0.25

    if pos_ok:
        pos = _add3(pos, _mul3(alpha, _sub3(pos_meas, pos)))

        # If no vel measurement, synthesize from consecutive position measurements
        if not vel_ok and _EST["last_meas_pos"] is not None and _EST["last_meas_t"] is not None:
            dtt = max(1e-3, _EST["t"] - _EST["last_meas_t"])
            v_synth = _mul3(1.0 / dtt, _sub3(pos_meas, _EST["last_meas_pos"]))
            vel = _add3(vel, _mul3(0.20, _sub3(v_synth, vel)))

        _EST["last_meas_pos"] = pos_meas[:]
        _EST["last_meas_t"] = _EST["t"]

    if vel_ok:
        vel = _add3(vel, _mul3(beta, _sub3(vel_meas, vel)))
        _EST["last_meas_vel"] = vel_meas[:]

    # If only scalar range available, adjust magnitude along current estimated line-of-sight
    if (not pos_ok) and range_ok:
        mag = _norm3(pos)
        if mag > 1e-3:
            pos = _mul3(range_meas / mag, pos)

    _EST["pos"], _EST["vel"] = pos[:], vel[:]

    # If still not initialized, output safe "do nothing but stabilize attitude"
    if not _EST["init"]:
        rates, rok = _robust_gyro_rates(state)
        k_att = 0.8
        max_rate = 0.035
        att_roll = _clip((-k_att * rates[0]) if rok else 0.0, -max_rate, max_rate)
        att_pitch = _clip((-k_att * rates[1]) if rok else 0.0, -max_rate, max_rate)
        att_yaw = _clip((-k_att * rates[2]) if rok else 0.0, -max_rate, max_rate)
        return {
            "thrust_x": 0.0, "thrust_y": 0.0, "thrust_z": 0.0,
            "att_rate_roll": att_roll, "att_rate_pitch": att_pitch, "att_rate_yaw": att_yaw
        }

    # --- Control ---
    x, y, z = pos
    vx, vy, vz = vel
    rng = _norm3(pos)
    n = _mean_motion_n(state)

    # Mass estimate for mapping accel->thrust command (rough but stable)
    fuel_kg = _get(state, "fuel_remaining_kg")
    if not _is_valid_num(fuel_kg) or fuel_kg < 0.0:
        fuel_kg = 300.0
    dry_mass = 900.0
    m = max(600.0, dry_mass + fuel_kg)

    Fmax = 220.0  # N per axis (assumed)
    amax = Fmax / m  # m/s^2 when thrust command = 1.0

    fuel_pct = _get(state, "fuel_remaining_pct")
    if not _is_valid_num(fuel_pct):
        fuel_pct = 50.0
    fuel_frac = _clip(fuel_pct / 100.0, 0.0, 1.0)

    # Thrust authority scaling under low fuel (prioritize stability, avoid waste)
    if fuel_kg < 60.0 or fuel_frac < 0.12:
        auth = 0.35
    elif fuel_kg < 120.0 or fuel_frac < 0.25:
        auth = 0.55
    else:
        auth = 1.0

    # Mode thresholds
    FAR = rng > 200.0
    MED = (rng <= 200.0 and rng > 50.0)
    CLOSE = (rng <= 50.0 and rng > 10.0)
    FINAL = rng <= 10.0

    # Desired acceleration in LVLH
    ax_cmd = ay_cmd = az_cmd = 0.0

    # Closing speed limiter (radial toward origin)
    if rng > 1e-3:
        r_hat = [x / rng, y / rng, z / rng]
        closing = -_dot3(vel, r_hat)  # m/s, positive means getting closer
    else:
        r_hat = [0.0, 0.0, 0.0]
        closing = 0.0

    # Speed caps by range
    if rng > 200.0:
        vmax_close = 0.8
    elif rng > 50.0:
        vmax_close = 0.3
    elif rng > 10.0:
        vmax_close = 0.18
    elif rng > 2.0:
        vmax_close = 0.10
    else:
        vmax_close = 0.05

    # --- Guidance + control (key counter-intuitive orbital phasing handled here) ---
    # The critical insight is applied in FAR mode: to close along-track gap when behind (y<0),
    # command *retrograde* (vy more negative), which produces positive along-track drift over time.
    if FAR:
        # Phasing time: long and gentle, proportional to along-track gap
        T = _clip(abs(y) / 0.6, 350.0, 2500.0)

        # CW far-field approximation for long-time drift dominated by y ≈ -3 t Δv_y
        vy_ref = y / (3.0 * T)  # y<0 => vy_ref<0 (retrograde), closes the gap over time

        # Keep x,z bounded but avoid aggressive radial "chasing" (which causes oscillations)
        # Use soft spring-damper on x,z; stronger damping on vx,vz.
        ax_cmd = (-0.010 * x) + (-0.18 * vx) + (0.02 * n * vy)  # slight coupling compensation
        az_cmd = (-0.012 * z) + (-0.22 * vz)

        # Along-track: velocity tracking (dominant); minimal direct position term
        ay_cmd = 0.35 * (vy_ref - vy)

        # If very large cross-track, spend a bit more to remove it
        if abs(z) > 150.0:
            az_cmd += -0.010 * z

    elif MED:
        # Blend from orbital-phasing logic to more "direct" closing as we enter linear regime
        # while maintaining a V-bar-ish (x≈0, z≈0) approach.
        blend = _clip((200.0 - rng) / 150.0, 0.0, 1.0)  # 0 at 200m, 1 at 50m

        T_phase = _clip(abs(y) / 0.8, 180.0, 1200.0)
        vy_phase = y / (3.0 * T_phase)  # still counter-intuitive component

        # Direct closing component (linear-ish) with a gentle cap
        T_dir = _clip(rng / 0.35, 90.0, 500.0)
        vy_dir = _clip((-y) / T_dir, -0.25, 0.25)

        vy_ref = (1.0 - blend) * vy_phase + blend * vy_dir

        # Bias toward keeping x,z small (V-bar)
        ax_cmd = (-0.030 * x) + (-0.35 * vx) + (0.05 * n * (vy_ref - vy))
        az_cmd = (-0.035 * z) + (-0.40 * vz)

        ay_cmd = 0.70 * (vy_ref - vy)

        # If we're ahead (y>0), avoid "diving in": keep approach very slow until y settles
        if y > 50.0:
            ay_cmd *= 0.6

    elif CLOSE:
        # Linear regime: drive toward the origin with a time-constant controller.
        tau = 22.0
        v_des = _mul3(-1.0 / tau, pos)

        # Cap commanded approach speed magnitude
        vmag = _norm3(v_des)
        if vmag > 1e-6:
            vcap = 0.18
            if vmag > vcap:
                v_des = _mul3(vcap / vmag, v_des)

        # Velocity tracking -> acceleration
        tau_v = 4.0
        a_des = _mul3(1.0 / tau_v, _sub3(v_des, vel))
        ax_cmd, ay_cmd, az_cmd = a_des[0], a_des[1], a_des[2]

        # Keep x,z tighter than y to avoid scraping
        ax_cmd += (-0.020 * x)
        az_cmd += (-0.020 * z)

    else:  # FINAL (<=10m)
        # Very gentle: essentially null relative velocity and creep in.
        tau = 18.0
        v_des = _mul3(-1.0 / tau, pos)

        # Tight cap on approach speed
        vmag = _norm3(v_des)
        if vmag > 1e-6:
            vcap = 0.08 if rng > 2.0 else 0.04
            if vmag > vcap:
                v_des = _mul3(vcap / vmag, v_des)

        tau_v = 2.8
        a_des = _mul3(1.0 / tau_v, _sub3(v_des, vel))
        ax_cmd, ay_cmd, az_cmd = a_des[0], a_des[1], a_des[2]

        # At very close range, strongly damp all axes
        if rng < 2.5:
            ax_cmd += -0.20 * vx
            ay_cmd += -0.20 * vy
            az_cmd += -0.20 * vz

    # --- Closing-speed safety brake (prevents scenario "closing too fast") ---
    if rng > 1e-3 and closing > vmax_close:
        # Accelerate opposite velocity along line of sight
        # (i.e., reduce closing rate). Use proportional brake.
        k_brake = 0.9
        a_brake = k_brake * (closing - vmax_close)
        ax_cmd += a_brake * r_hat[0]
        ay_cmd += a_brake * r_hat[1]
        az_cmd += a_brake * r_hat[2]

    # --- Map desired acceleration to normalized thrust commands ---
    # Limit total accel gently to avoid waste / oscillation
    a_vec = [ax_cmd, ay_cmd, az_cmd]
    amag = _norm3(a_vec)
    # Axis-wise limits (z corrections are usually cheap; keep y moderate in close)
    a_axis_lim = amax * auth
    if amag > 1e-9:
        # Soft global limit: allow up to ~1.2 axis-limits combined
        a_global_lim = 1.2 * a_axis_lim
        if amag > a_global_lim:
            a_vec = _mul3(a_global_lim / amag, a_vec)

    # Final axis clamp
    ax_cmd = _clip(a_vec[0], -a_axis_lim, a_axis_lim)
    ay_cmd = _clip(a_vec[1], -a_axis_lim, a_axis_lim)
    az_cmd = _clip(a_vec[2], -a_axis_lim, a_axis_lim)

    thrust_x = _clip(ax_cmd / amax, -1.0, 1.0) if amax > 1e-9 else 0.0
    thrust_y = _clip(ay_cmd / amax, -1.0, 1.0) if amax > 1e-9 else 0.0
    thrust_z = _clip(az_cmd / amax, -1.0, 1.0) if amax > 1e-9 else 0.0

    # --- Attitude rate damping (simple, robust) ---
    rates, rok = _robust_gyro_rates(state)
    k_att = 0.9
    max_rate = 0.035
    att_roll  = _clip((-k_att * rates[0]) if rok else 0.0, -max_rate, max_rate)
    att_pitch = _clip((-k_att * rates[1]) if rok else 0.0, -max_rate, max_rate)
    att_yaw   = _clip((-k_att * rates[2]) if rok else 0.0, -max_rate, max_rate)

    return {
        "thrust_x": float(thrust_x),
        "thrust_y": float(thrust_y),
        "thrust_z": float(thrust_z),
        "att_rate_roll": float(att_roll),
        "att_rate_pitch": float(att_pitch),
        "att_rate_yaw": float(att_yaw),
    }
