use std::cmp::Ordering;
use std::f64;
use std::io::{self, Read, Write};

#[derive(Clone, Copy, Debug, Default)]
struct Vec2 {
    x: f64,
    y: f64,
}
impl Vec2 {
    fn sub(self, o: Vec2) -> Vec2 {
        Vec2 {
            x: self.x - o.x,
            y: self.y - o.y,
        }
    }
    fn len(self) -> f64 {
        (self.x * self.x + self.y * self.y).sqrt()
    }
    fn cross(self, o: Vec2) -> f64 {
        self.x * o.y - self.y * o.x
    }
}

#[derive(Clone, Copy, Debug, Default)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}
impl Vec3 {
    fn dot(self, o: Vec3) -> f64 {
        self.x * o.x + self.y * o.y + self.z * o.z
    }
    fn cross(self, o: Vec3) -> Vec3 {
        Vec3 {
            x: self.y * o.z - self.z * o.y,
            y: self.z * o.x - self.x * o.z,
            z: self.x * o.y - self.y * o.x,
        }
    }
    fn mul(self, s: f64) -> Vec3 {
        Vec3 {
            x: self.x * s,
            y: self.y * s,
            z: self.z * s,
        }
    }
    fn add(self, o: Vec3) -> Vec3 {
        Vec3 {
            x: self.x + o.x,
            y: self.y + o.y,
            z: self.z + o.z,
        }
    }
    fn norm(self) -> f64 {
        (self.x * self.x + self.y * self.y + self.z * self.z).sqrt()
    }
}

#[derive(Clone, Copy, Debug)]
struct Mat3 {
    // row-major 3x3
    m: [[f64; 3]; 3],
}
impl Mat3 {
    fn mul_vec3(&self, v: Vec3) -> Vec3 {
        Vec3 {
            x: self.m[0][0] * v.x + self.m[0][1] * v.y + self.m[0][2] * v.z,
            y: self.m[1][0] * v.x + self.m[1][1] * v.y + self.m[1][2] * v.z,
            z: self.m[2][0] * v.x + self.m[2][1] * v.y + self.m[2][2] * v.z,
        }
    }
    fn to_quat(&self) -> Quat {
        // Standard matrix->quaternion conversion
        let m = &self.m;
        let tr = m[0][0] + m[1][1] + m[2][2];
        let (qw, qx, qy, qz) = if tr > 0.0 {
            let s = (tr + 1.0).sqrt() * 2.0;
            (
                0.25 * s,
                (m[2][1] - m[1][2]) / s,
                (m[0][2] - m[2][0]) / s,
                (m[1][0] - m[0][1]) / s,
            )
        } else if m[0][0] > m[1][1] && m[0][0] > m[2][2] {
            let s = (1.0 + m[0][0] - m[1][1] - m[2][2]).sqrt() * 2.0;
            (
                (m[2][1] - m[1][2]) / s,
                0.25 * s,
                (m[0][1] + m[1][0]) / s,
                (m[0][2] + m[2][0]) / s,
            )
        } else if m[1][1] > m[2][2] {
            let s = (1.0 + m[1][1] - m[0][0] - m[2][2]).sqrt() * 2.0;
            (
                (m[0][2] - m[2][0]) / s,
                (m[0][1] + m[1][0]) / s,
                0.25 * s,
                (m[1][2] + m[2][1]) / s,
            )
        } else {
            let s = (1.0 + m[2][2] - m[0][0] - m[1][1]).sqrt() * 2.0;
            (
                (m[1][0] - m[0][1]) / s,
                (m[0][2] + m[2][0]) / s,
                (m[1][2] + m[2][1]) / s,
                0.25 * s,
            )
        };
        let mut q = Quat {
            w: qw,
            x: qx,
            y: qy,
            z: qz,
        };
        q.normalize();
        q
    }
}

#[derive(Clone, Copy, Debug)]
struct Quat {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}
impl Quat {
    fn normalize(&mut self) {
        let n = (self.w * self.w + self.x * self.x + self.y * self.y + self.z * self.z).sqrt();
        if n > 0.0 {
            self.w /= n;
            self.x /= n;
            self.y /= n;
            self.z /= n;
        } else {
            self.w = 1.0;
            self.x = 0.0;
            self.y = 0.0;
            self.z = 0.0;
        }
    }
}

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}
impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self {
            buf: input.into_bytes(),
            idx: 0,
        }
    }
    fn next_f64(&mut self) -> f64 {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        let start = self.idx;
        while self.idx < self.buf.len() && !self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        let s = std::str::from_utf8(&self.buf[start..self.idx]).unwrap();
        s.parse::<f64>().unwrap()
    }
    fn next_usize(&mut self) -> usize {
        self.next_f64() as usize
    }
}

fn convex_hull(mut pts: Vec<Vec2>) -> Vec<Vec2> {
    // Monotonic chain; tolerates duplicate points.
    pts.sort_by(|a, b| {
        let ox = a.x.partial_cmp(&b.x).unwrap_or(Ordering::Equal);
        if ox != Ordering::Equal {
            return ox;
        }
        a.y.partial_cmp(&b.y).unwrap_or(Ordering::Equal)
    });

    if pts.len() <= 1 {
        return pts;
    }

    let mut lower: Vec<Vec2> = Vec::with_capacity(pts.len());
    for p in &pts {
        while lower.len() >= 2 {
            let a = lower[lower.len() - 2];
            let b = lower[lower.len() - 1];
            let ab = b.sub(a);
            let ap = p.sub(a);
            if ab.cross(ap) <= 1e-12 {
                lower.pop();
            } else {
                break;
            }
        }
        lower.push(*p);
    }

    let mut upper: Vec<Vec2> = Vec::with_capacity(pts.len());
    for p in pts.iter().rev() {
        while upper.len() >= 2 {
            let a = upper[upper.len() - 2];
            let b = upper[upper.len() - 1];
            let ab = b.sub(a);
            let ap = p.sub(a);
            if ab.cross(ap) <= 1e-12 {
                upper.pop();
            } else {
                break;
            }
        }
        upper.push(*p);
    }

    lower.pop();
    upper.pop();
    lower.extend(upper);
    lower
}

fn inradius_about_origin(hull: &[Vec2]) -> f64 {
    if hull.len() < 2 {
        return 0.0;
    }
    if hull.len() == 2 {
        // Degenerate line; radius is 0 in 2D.
        return 0.0;
    }
    let mut r = f64::INFINITY;
    for i in 0..hull.len() {
        let a = hull[i];
        let b = hull[(i + 1) % hull.len()];
        let ab = b.sub(a);
        let denom = ab.len();
        if denom <= 1e-15 {
            continue;
        }
        // distance from origin to supporting line through segment a->b
        // |cross(ab, -a)| / |ab| == |cross(ab, a)| / |ab|
        let dist = (ab.cross(a)).abs() / denom;
        if dist < r {
            r = dist;
        }
    }
    if r.is_finite() { r } else { 0.0 }
}

fn generate_24_rotations() -> Vec<Mat3> {
    // Generate all proper rotations mapping axes to axes (24 elements).
    let dirs: [Vec3; 6] = [
        Vec3 { x: 1.0, y: 0.0, z: 0.0 },
        Vec3 { x: -1.0, y: 0.0, z: 0.0 },
        Vec3 { x: 0.0, y: 1.0, z: 0.0 },
        Vec3 { x: 0.0, y: -1.0, z: 0.0 },
        Vec3 { x: 0.0, y: 0.0, z: 1.0 },
        Vec3 { x: 0.0, y: 0.0, z: -1.0 },
    ];
    let mut mats: Vec<Mat3> = Vec::with_capacity(24);

    for &x in &dirs {
        for &y in &dirs {
            if x.dot(y).abs() > 0.0 {
                continue;
            }
            let z = x.cross(y);
            // Ensure right-handed and axis-aligned z
            let is_axis_aligned = (z.x.abs() + z.y.abs() + z.z.abs() - 1.0).abs() < 1e-12;
            if !is_axis_aligned {
                continue;
            }
            // Build matrix with columns x,y,z (image of local basis)
            let mat = Mat3 {
                m: [
                    [x.x, y.x, z.x],
                    [x.y, y.y, z.y],
                    [x.z, y.z, z.z],
                ],
            };

            // De-duplicate (should be small, so O(n) ok)
            let mut dup = false;
            for mm in &mats {
                let mut same = true;
                for r in 0..3 {
                    for c in 0..3 {
                        if (mm.m[r][c] - mat.m[r][c]).abs() > 1e-12 {
                            same = false;
                        }
                    }
                }
                if same {
                    dup = true;
                    break;
                }
            }
            if !dup {
                mats.push(mat);
            }
        }
    }
    mats
}

fn main() {
    let mut sc = Scanner::new();

    let nv = sc.next_usize();
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = -f64::INFINITY;
    let mut maxy = -f64::INFINITY;

    for _ in 0..nv {
        let x = sc.next_f64();
        let y = sc.next_f64();
        if x < minx {
            minx = x;
        }
        if y < miny {
            miny = y;
        }
        if x > maxx {
            maxx = x;
        }
        if y > maxy {
            maxy = y;
        }
    }

    let mut sx = sc.next_f64();
    let mut sy = sc.next_f64();
    let mut sz = sc.next_f64();

    // Stabilize pathological sz ~ 0 cases for numeric safety.
    // (Still uses the given direction essentially; only avoids division by 0.)
    if sz.abs() < 1e-12 {
        // Keep sign "toward ground" if possible, otherwise force a small negative.
        sz = if sz < 0.0 { -1e-12 } else { -1e-12 };
    }

    // Standard reference tetra vertices (centered at origin)
    let base: [Vec3; 4] = [
        Vec3 { x: 1.0, y: 0.0, z: -0.707 },
        Vec3 { x: -1.0, y: 0.0, z: -0.707 },
        Vec3 { x: 0.0, y: 1.0, z: 0.707 },
        Vec3 { x: 0.0, y: -1.0, z: 0.707 },
    ];

    // Adaptive orientation selection: try 24 axis rotations and pick one maximizing
    // a guaranteed inscribed disk radius (about the projected center).
    let rotations = generate_24_rotations();
    let mut best_mat = rotations[0];
    let mut best_r1 = 0.0;

    let sx_over_sz = sx / sz;
    let sy_over_sz = sy / sz;

    for mat in rotations {
        let mut pts2: Vec<Vec2> = Vec::with_capacity(4);
        for &v in &base {
            let vr = mat.mul_vec3(v);
            // Parallel projection along sun direction onto z=0 plane:
            // (x',y') = (x - z*sx/sz, y - z*sy/sz)
            pts2.push(Vec2 {
                x: vr.x - vr.z * sx_over_sz,
                y: vr.y - vr.z * sy_over_sz,
            });
        }
        let hull = convex_hull(pts2);
        let r1 = inradius_about_origin(&hull);
        if r1 > best_r1 {
            best_r1 = r1;
            best_mat = mat;
        }
    }

    // Always use maximum allowed scale to reduce count.
    let scale: f64 = 10.0;

    // Compute guaranteed disk radius at this scale.
    let mut pts2_best: Vec<Vec2> = Vec::with_capacity(4);
    for &v in &base {
        let vr = best_mat.mul_vec3(v.mul(scale));
        pts2_best.push(Vec2 {
            x: vr.x - vr.z * sx_over_sz,
            y: vr.y - vr.z * sy_over_sz,
        });
    }
    let hull_best = convex_hull(pts2_best);
    let mut r = inradius_about_origin(&hull_best);
    if !r.is_finite() || r < 1e-9 {
        // Fallback: if projection is degenerate, choose a tiny radius (will force a dense grid).
        r = 1e-6;
    }

    // Physical (3D) bounding sphere radius for non-intersection reasoning.
    let mut r_phys = 0.0;
    let mut hz = 0.0;
    for &v in &base {
        let vr = best_mat.mul_vec3(v.mul(scale));
        r_phys = r_phys.max(vr.norm());
        hz = hz.max(vr.z.abs());
    }

    // Covering grid spacing: for a square grid, to cover the plane with disks radius r:
    // step <= sqrt(2)*r. Use a safety factor.
    let step = r * 1.35;

    // If the polygon's AABB fits into one guaranteed disk, emit exactly one tetra.
    let dx = maxx - minx;
    let dy = maxy - miny;
    let half_diag2 = (0.5 * dx) * (0.5 * dx) + (0.5 * dy) * (0.5 * dy);
    let one_ok = half_diag2 <= r * r;

    let quat = best_mat.to_quat();

    let mut out = io::BufWriter::new(io::stdout());

    if one_ok || (dx.abs() < 1e-12 && dy.abs() < 1e-12) {
        let cx2 = (minx + maxx) * 0.5;
        let cy2 = (miny + maxy) * 0.5;

        // Put it at z=0 for exact shadow center (cx2,cy2). (No intersection issues with single.)
        let px = cx2;
        let py = cy2;
        let pz = 0.0;

        writeln!(&mut out, "1").unwrap();
        writeln!(
            &mut out,
            "{:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10}",
            px, py, pz, quat.w, quat.x, quat.y, quat.z, scale
        )
        .unwrap();
        return;
    }

    // Bounding-box cover (guaranteed): cover the AABB expanded by r, so the union of disks covers AABB.
    let start_x = minx - r;
    let start_y = miny - r;
    let end_x = maxx + r;
    let end_y = maxy + r;

    let nx = (((end_x - start_x) / step).ceil() as isize + 1).max(1) as usize;
    let ny = (((end_y - start_y) / step).ceil() as isize + 1).max(1) as usize;

    // Adaptive non-intersection:
    // - If step is big enough, we can put all tetrahedra at the same height (cz=0).
    // - Otherwise, "color" the grid into m*m layers so that within each layer centers are far apart;
    //   separate layers in z by > 2*hz to guarantee no intersection.
    let need_sep = step < (2.0 * r_phys + 0.5);
    let m = if need_sep {
        (((2.0 * r_phys + 0.5) / step).ceil() as isize).max(1) as usize
    } else {
        1
    };
    let layers = m * m;

    let z_step = if layers > 1 {
        // Disjoint z-intervals across layers => no 3D intersection across layers.
        2.0 * hz + 1.0
    } else {
        0.0
    };

    // Count and output
    let count = nx.saturating_mul(ny);
    writeln!(&mut out, "{count}").unwrap();

    // Precompute shift ratios for keeping the same shadow center when moving in z:
    // For center (px,py,pz), shadow center on z=0 is (px - pz*sx/sz, py - pz*sy/sz).
    // To keep shadow center at (u,v): px = u + pz*sx/sz, py = v + pz*sy/sz.
    let kx = sx_over_sz;
    let ky = sy_over_sz;

    // Stream output (no large in-memory accumulation).
    for iy in 0..ny {
        let v = start_y + (iy as f64) * step;
        for ix in 0..nx {
            let u = start_x + (ix as f64) * step;

            let layer = if layers > 1 {
                (ix % m) + (iy % m) * m
            } else {
                0
            };
            let pz = (layer as f64) * z_step;
            let px = u + pz * kx;
            let py = v + pz * ky;

            writeln!(
                &mut out,
                "{:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10}",
                px, py, pz, quat.w, quat.x, quat.y, quat.z, scale
            )
            .unwrap();
        }
    }
}
