use std::cmp::Ordering;
use std::collections::HashMap;
use std::hash::{BuildHasherDefault, Hasher};
use std::io::{self, Read};

/// ------------------------- Fast input -------------------------
struct Scanner {
    buf: Vec<u8>,
    i: usize,
}
impl Scanner {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, i: 0 }
    }
    fn skip_ws(&mut self) {
        while self.i < self.buf.len() && self.buf[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
    }
    fn next_bytes<'a>(&'a mut self) -> Option<&'a [u8]> {
        self.skip_ws();
        if self.i >= self.buf.len() {
            return None;
        }
        let start = self.i;
        while self.i < self.buf.len() && !self.buf[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
        Some(&self.buf[start..self.i])
    }
}

fn eq_ascii_case_insensitive(a: &[u8], b: &[u8]) -> bool {
    if a.len() != b.len() {
        return false;
    }
    a.iter()
        .zip(b.iter())
        .all(|(&x, &y)| x.to_ascii_lowercase() == y.to_ascii_lowercase())
}

fn parse_i64(tok: &[u8]) -> Option<i64> {
    if tok.is_empty() {
        return None;
    }
    let mut i = 0;
    let mut sign = 1i64;
    if tok[0] == b'+' {
        i += 1;
    } else if tok[0] == b'-' {
        sign = -1;
        i += 1;
    }
    if i >= tok.len() {
        return None;
    }
    let mut v: i64 = 0;
    let mut any = false;
    while i < tok.len() {
        let c = tok[i];
        if !(b'0'..=b'9').contains(&c) {
            return None;
        }
        any = true;
        v = v.saturating_mul(10).saturating_add((c - b'0') as i64);
        i += 1;
    }
    if !any {
        None
    } else {
        Some(sign * v)
    }
}

// Fast-ish float parser: supports [-+]?(digits)?(.digits)?([eE][-+]?digits)?
fn parse_f32_opt(tok: &[u8]) -> Option<f32> {
    if tok.is_empty() {
        return None;
    }
    // common missing markers
    if tok[0] == b'*'
        || tok[0] == b'?'
        || eq_ascii_case_insensitive(tok, b"na")
        || eq_ascii_case_insensitive(tok, b"nan")
        || eq_ascii_case_insensitive(tok, b"null")
    {
        return None;
    }

    let mut i = 0usize;
    let mut sign = 1.0f64;
    if tok[i] == b'+' {
        i += 1;
    } else if tok[i] == b'-' {
        sign = -1.0;
        i += 1;
    }
    if i >= tok.len() {
        return None;
    }

    let mut int_part: u64 = 0;
    let mut int_any = false;
    while i < tok.len() && (b'0'..=b'9').contains(&tok[i]) {
        int_any = true;
        int_part = int_part.saturating_mul(10).saturating_add((tok[i] - b'0') as u64);
        i += 1;
    }

    let mut frac_part: u64 = 0;
    let mut frac_div: f64 = 1.0;
    if i < tok.len() && tok[i] == b'.' {
        i += 1;
        while i < tok.len() && (b'0'..=b'9').contains(&tok[i]) {
            frac_part = frac_part.saturating_mul(10).saturating_add((tok[i] - b'0') as u64);
            frac_div *= 10.0;
            i += 1;
        }
    }

    if !int_any && frac_div == 1.0 {
        return None; // no digits at all
    }

    let mut exp: i32 = 0;
    if i < tok.len() && (tok[i] == b'e' || tok[i] == b'E') {
        i += 1;
        if i >= tok.len() {
            return None;
        }
        let mut exp_sign: i32 = 1;
        if tok[i] == b'+' {
            i += 1;
        } else if tok[i] == b'-' {
            exp_sign = -1;
            i += 1;
        }
        if i >= tok.len() {
            return None;
        }
        let mut exp_any = false;
        let mut e: i32 = 0;
        while i < tok.len() && (b'0'..=b'9').contains(&tok[i]) {
            exp_any = true;
            e = e.saturating_mul(10).saturating_add((tok[i] - b'0') as i32);
            i += 1;
        }
        if !exp_any {
            return None;
        }
        exp = exp_sign * e;
    }

    if i != tok.len() {
        return None; // trailing junk
    }

    let mut v = (int_part as f64) + (frac_part as f64) / frac_div;
    v *= sign;
    if exp != 0 {
        v *= 10f64.powi(exp);
    }
    if !v.is_finite() {
        return None;
    }
    Some(v as f32)
}

/// ------------------------- Data structures -------------------------
#[derive(Clone, Copy)]
struct SampleMeta {
    hole: u32,
    sample_idx: u32,
    x: f32,
    y: f32,
    z: f32,
}

#[derive(Clone)]
struct Suspect {
    conf: f32,
    hole: u32,
    sample_idx: u32,
    prop: u16,
    // (optional) could include value, but output format doesn't request it
}

impl Suspect {
    fn cmp_conf_desc(a: &Suspect, b: &Suspect) -> Ordering {
        b.conf.total_cmp(&a.conf)
    }
}

/// ------------------------- Fast HashMap for u64 keys -------------------------
/// HashMap default SipHash can be a big bottleneck at large scale.
/// This hasher is NOT cryptographically secure; it's for performance in trusted input.
#[derive(Default)]
struct IdentityHasher {
    state: u64,
}
impl Hasher for IdentityHasher {
    fn write(&mut self, bytes: &[u8]) {
        // FNV-1a-ish mixing for non-u64 writes (rare here).
        let mut h = 14695981039346656037u64;
        for &b in bytes {
            h ^= b as u64;
            h = h.wrapping_mul(1099511628211u64);
        }
        self.state = self.state ^ h;
    }
    fn write_u64(&mut self, i: u64) {
        self.state = i;
    }
    fn finish(&self) -> u64 {
        self.state
    }
}
type FastBuildHasher = BuildHasherDefault<IdentityHasher>;

fn cell_key(ix: i32, iy: i32, iz: i32) -> u64 {
    // Mix 3 signed ints into one u64 (two's complement cast is fine).
    // Constants are from splitmix64-ish multipliers.
    let x = (ix as u32) as u64;
    let y = (iy as u32) as u64;
    let z = (iz as u32) as u64;
    x.wrapping_mul(0x9E3779B97F4A7C15)
        ^ y.wrapping_mul(0xC2B2AE3D27D4EB4F)
        ^ z.wrapping_mul(0x165667B19E3779F9)
}

/// ------------------------- Robust local stats -------------------------
fn median_sorted(v: &[f32]) -> f32 {
    let n = v.len();
    if n == 0 {
        return f32::NAN;
    }
    if (n & 1) == 1 {
        v[n / 2]
    } else {
        (v[n / 2 - 1] + v[n / 2]) * 0.5
    }
}

fn mad_from_sorted(v_sorted: &[f32], med: f32, tmp: &mut Vec<f32>) -> f32 {
    tmp.clear();
    tmp.extend(v_sorted.iter().map(|&x| (x - med).abs()));
    tmp.sort_unstable_by(|a, b| a.total_cmp(b));
    median_sorted(tmp)
}

/// ------------------------- Neighbor search -------------------------
enum NeighborMode {
    BruteForce,
    Grid,
}

struct NeighborIndex {
    mode: NeighborMode,
    // for grid mode:
    origin_x: f32,
    origin_y: f32,
    origin_z: f32,
    cell_size: f32,
    grid: HashMap<u64, Vec<u32>, FastBuildHasher>,
}

impl NeighborIndex {
    fn build_grid(samples: &[SampleMeta], cell_size: f32, origin: (f32, f32, f32)) -> Self {
        let mut grid: HashMap<u64, Vec<u32>, FastBuildHasher> =
            HashMap::with_capacity_and_hasher(samples.len() * 2, Default::default());

        let (ox, oy, oz) = origin;
        let inv = 1.0f32 / cell_size.max(1e-6);

        for (idx, s) in samples.iter().enumerate() {
            let mut ix = ((s.x - ox) * inv).floor() as i32;
            let mut iy = ((s.y - oy) * inv).floor() as i32;
            let mut iz = ((s.z - oz) * inv).floor() as i32;
            // clamp small negatives due to float noise
            if ix < 0 {
                ix = 0
            }
            if iy < 0 {
                iy = 0
            }
            if iz < 0 {
                iz = 0
            }
            let k = cell_key(ix, iy, iz);
            grid.entry(k).or_insert_with(Vec::new).push(idx as u32);
        }

        Self {
            mode: NeighborMode::Grid,
            origin_x: ox,
            origin_y: oy,
            origin_z: oz,
            cell_size,
            grid,
        }
    }

    fn brute() -> Self {
        Self {
            mode: NeighborMode::BruteForce,
            origin_x: 0.0,
            origin_y: 0.0,
            origin_z: 0.0,
            cell_size: 1.0,
            grid: HashMap::with_hasher(Default::default()),
        }
    }

    fn find_neighbors(
        &self,
        samples: &[SampleMeta],
        i: usize,
        radius2: f32,
        min_neighbors: usize,
        max_neighbors: usize,
        max_ring: i32,
        out: &mut Vec<u32>,
    ) {
        out.clear();
        match self.mode {
            NeighborMode::BruteForce => {
                let si = samples[i];
                for (j, sj) in samples.iter().enumerate() {
                    if j == i {
                        continue;
                    }
                    let dx = sj.x - si.x;
                    let dy = sj.y - si.y;
                    let dz = sj.z - si.z;
                    let d2 = dx * dx + dy * dy + dz * dz;
                    if d2 <= radius2 {
                        out.push(j as u32);
                        if out.len() >= max_neighbors {
                            break;
                        }
                    }
                }
            }
            NeighborMode::Grid => {
                let si = samples[i];
                let inv = 1.0f32 / self.cell_size.max(1e-6);

                let mut ix = ((si.x - self.origin_x) * inv).floor() as i32;
                let mut iy = ((si.y - self.origin_y) * inv).floor() as i32;
                let mut iz = ((si.z - self.origin_z) * inv).floor() as i32;
                if ix < 0 {
                    ix = 0
                }
                if iy < 0 {
                    iy = 0
                }
                if iz < 0 {
                    iz = 0
                }

                // Expand cube ring-by-ring until we have enough neighbors or hit max_ring.
                for ring in 1..=max_ring {
                    for dx in -ring..=ring {
                        for dy in -ring..=ring {
                            for dz in -ring..=ring {
                                // Only the outer shell of this ring (optional); but full cube is simpler.
                                let k = cell_key(ix + dx, iy + dy, iz + dz);
                                if let Some(v) = self.grid.get(&k) {
                                    for &j_u32 in v {
                                        let j = j_u32 as usize;
                                        if j == i {
                                            continue;
                                        }
                                        let sj = samples[j];
                                        let dx = sj.x - si.x;
                                        let dy = sj.y - si.y;
                                        let dz = sj.z - si.z;
                                        let d2 = dx * dx + dy * dy + dz * dz;
                                        if d2 <= radius2 {
                                            out.push(j_u32);
                                            if out.len() >= max_neighbors {
                                                return;
                                            }
                                        }
                                    }
                                }
                            }
                        }
                    }
                    if out.len() >= min_neighbors {
                        break;
                    }
                }
            }
        }
    }
}

/// ------------------------- Main -------------------------
fn main() {
    let mut sc = Scanner::new();

    let nh = match sc.next_bytes().and_then(parse_i64) {
        Some(v) if v > 0 => v as usize,
        _ => {
            // No valid header.
            println!("0");
            return;
        }
    };
    let np = match sc.next_bytes().and_then(parse_i64) {
        Some(v) if v > 0 => v as usize,
        _ => {
            println!("0");
            return;
        }
    };

    let mut prop_names: Vec<String> = Vec::with_capacity(np);
    for _ in 0..np {
        if let Some(t) = sc.next_bytes() {
            prop_names.push(String::from_utf8_lossy(t).to_string());
        } else {
            prop_names.push("prop".to_string());
        }
    }

    let mut hole_ids: Vec<String> = Vec::with_capacity(nh);
    let mut samples: Vec<SampleMeta> = Vec::new();
    let mut values: Vec<f32> = Vec::new();

    // Global stats per property (Welford)
    let mut cnt: Vec<u64> = vec![0; np];
    let mut mean: Vec<f64> = vec![0.0; np];
    let mut m2: Vec<f64> = vec![0.0; np];

    // Bounding box of sample positions
    let mut minx = f32::INFINITY;
    let mut miny = f32::INFINITY;
    let mut minz = f32::INFINITY;
    let mut maxx = f32::NEG_INFINITY;
    let mut maxy = f32::NEG_INFINITY;
    let mut maxz = f32::NEG_INFINITY;

    for hole_idx in 0..nh {
        let hole_id = match sc.next_bytes() {
            Some(t) => String::from_utf8_lossy(t).to_string(),
            None => break,
        };

        let sx = match sc.next_bytes().and_then(parse_f32_opt) {
            Some(v) => v,
            None => 0.0,
        };
        let sy = match sc.next_bytes().and_then(parse_f32_opt) {
            Some(v) => v,
            None => 0.0,
        };
        let sz = match sc.next_bytes().and_then(parse_f32_opt) {
            Some(v) => v,
            None => 0.0,
        };

        let dx = match sc.next_bytes().and_then(parse_f32_opt) {
            Some(v) => v,
            None => 0.0,
        };
        let dy = match sc.next_bytes().and_then(parse_f32_opt) {
            Some(v) => v,
            None => 0.0,
        };
        let dz = match sc.next_bytes().and_then(parse_f32_opt) {
            Some(v) => v,
            None => 1.0,
        };

        let _length = match sc.next_bytes().and_then(parse_f32_opt) {
            Some(v) => v,
            None => 0.0,
        };

        let nsamp = match sc.next_bytes().and_then(parse_i64) {
            Some(v) if v >= 0 => v as usize,
            _ => 0,
        };

        let norm = (dx * dx + dy * dy + dz * dz).sqrt();
        let (ux, uy, uz) = if norm > 0.0 {
            (dx / norm, dy / norm, dz / norm)
        } else {
            (0.0, 0.0, 1.0)
        };

        hole_ids.push(hole_id);

        for sidx in 0..nsamp {
            let depth = match sc.next_bytes().and_then(parse_f32_opt) {
                Some(v) => v,
                None => {
                    // If a sample line is missing/corrupt, stop consuming this hole's samples.
                    break;
                }
            };

            let px = sx + ux * depth;
            let py = sy + uy * depth;
            let pz = sz + uz * depth;

            minx = minx.min(px);
            miny = miny.min(py);
            minz = minz.min(pz);
            maxx = maxx.max(px);
            maxy = maxy.max(py);
            maxz = maxz.max(pz);

            let hole_u32 = hole_idx as u32;
            samples.push(SampleMeta {
                hole: hole_u32,
                sample_idx: sidx as u32,
                x: px,
                y: py,
                z: pz,
            });

            // property values
            for p in 0..np {
                let v = match sc.next_bytes() {
                    Some(t) => parse_f32_opt(t).unwrap_or(f32::NAN),
                    None => f32::NAN,
                };
                values.push(v);

                if v.is_finite() {
                    let x = v as f64;
                    cnt[p] += 1;
                    let c = cnt[p] as f64;
                    let delta = x - mean[p];
                    mean[p] += delta / c;
                    let delta2 = x - mean[p];
                    m2[p] += delta * delta2;
                }
            }
        }
    }

    let n = samples.len();
    if n == 0 || np == 0 {
        println!("0");
        return;
    }

    // Global std dev (fallback/filters)
    let mut gstd: Vec<f64> = vec![0.0; np];
    for p in 0..np {
        if cnt[p] >= 2 {
            gstd[p] = (m2[p] / ((cnt[p] - 1) as f64)).sqrt();
        } else {
            gstd[p] = 0.0;
        }
    }

    // Adaptive strategy based on dataset scale
    let volume = ((maxx - minx) as f64).max(0.0)
        * ((maxy - miny) as f64).max(0.0)
        * ((maxz - minz) as f64).max(0.0);

    let spacing = if volume > 0.0 {
        (volume / (n as f64)).cbrt()
    } else {
        // degenerate bbox: assume some spacing
        1.0
    };

    let target_neighbors = if n < 20_000 { 30.0 } else { 20.0 };
    let c = (target_neighbors / 4.1887902047863905f64).cbrt(); // inverse sphere factor
    let mut radius = (spacing * c).max(spacing * 1.0).min(spacing * 6.0);
    if n < 1_000 {
        radius *= 2.0;
    }
    radius = radius.max(0.25); // avoid extremely tiny radii

    let radius2 = (radius as f32) * (radius as f32);

    let (prefilter_gz, threshold_z, min_neighbors, max_neighbors, max_ring, mode) = if n <= 8_000 {
        // Small: brute force is acceptable and simpler.
        (0.0f32, 5.0f32, 3usize, 256usize, 1i32, NeighborMode::BruteForce)
    } else if n <= 50_000 {
        (2.0f32, 5.5f32, 5usize, 192usize, 3i32, NeighborMode::Grid)
    } else {
        (3.0f32, 6.0f32, 6usize, 96usize, 3i32, NeighborMode::Grid)
    };

    let index = match mode {
        NeighborMode::BruteForce => NeighborIndex::brute(),
        NeighborMode::Grid => NeighborIndex::build_grid(
            &samples,
            radius as f32,
            (minx, miny, minz), // origin
        ),
    };

    let max_out: usize = if n < 200_000 { 50_000 } else { 200_000 };

    let mut suspects: Vec<Suspect> = Vec::new();
    let mut neighbors: Vec<u32> = Vec::with_capacity(max_neighbors);
    let mut candidate_props: Vec<(usize, f32)> = Vec::with_capacity(8);
    let mut neigh_vals: Vec<f32> = Vec::with_capacity(max_neighbors);
    let mut tmp_abs: Vec<f32> = Vec::with_capacity(max_neighbors);

    // Main detection loop
    for i in 0..n {
        candidate_props.clear();

        let base = i * np;
        // Step 1: quick global z-score filter to avoid expensive neighbor stats for normal values.
        for p in 0..np {
            let v = values[base + p];
            if !v.is_finite() {
                continue;
            }
            let sd = gstd[p];
            if sd <= 0.0 {
                continue;
            }
            let gz = (((v as f64) - mean[p]) / sd).abs() as f32;
            if gz >= prefilter_gz {
                candidate_props.push((p, gz));
            }
        }

        if candidate_props.is_empty() {
            continue;
        }

        // Step 2: neighbor query once per sample (reused across candidate properties).
        index.find_neighbors(
            &samples,
            i,
            radius2,
            min_neighbors,
            max_neighbors,
            max_ring,
            &mut neighbors,
        );

        // If we don't have local context, only report extreme global anomalies.
        if neighbors.len() < min_neighbors {
            for &(p, gz) in &candidate_props {
                if gz >= (threshold_z + 6.0) {
                    let conf = gz * 0.9;
                    suspects.push(Suspect {
                        conf,
                        hole: samples[i].hole,
                        sample_idx: samples[i].sample_idx,
                        prop: p as u16,
                    });
                }
            }
        } else {
            // Step 3: robust local test per property using neighbor median + MAD.
            for &(p, gz) in &candidate_props {
                let v0 = values[base + p];
                if !v0.is_finite() {
                    continue;
                }

                neigh_vals.clear();
                for &j_u32 in &neighbors {
                    let j = j_u32 as usize;
                    let vj = values[j * np + p];
                    if vj.is_finite() {
                        neigh_vals.push(vj);
                    }
                    if neigh_vals.len() >= max_neighbors {
                        break;
                    }
                }

                if neigh_vals.len() < min_neighbors {
                    // fallback: if global is still huge, accept.
                    if gz >= (threshold_z + 6.0) {
                        let conf = gz * 0.85;
                        suspects.push(Suspect {
                            conf,
                            hole: samples[i].hole,
                            sample_idx: samples[i].sample_idx,
                            prop: p as u16,
                        });
                    }
                    continue;
                }

                neigh_vals.sort_unstable_by(|a, b| a.total_cmp(b));
                let med = median_sorted(&neigh_vals);
                if !med.is_finite() {
                    continue;
                }

                let mad = mad_from_sorted(&neigh_vals, med, &mut tmp_abs);
                let mut sigma = 1.4826f32 * mad;
                if !(sigma > 1e-12) {
                    // very flat neighborhood: use a tiny scale to avoid division by ~0
                    sigma = 1e-6;
                }

                let z = ((v0 - med).abs()) / sigma;
                if z.is_finite() && z >= threshold_z {
                    // Confidence: mainly local robust z-score, boosted by neighbor count + global z.
                    let nn = neigh_vals.len() as f32;
                    let conf = z * (1.0 + nn.ln()) * (1.0 + 0.05 * gz);
                    suspects.push(Suspect {
                        conf,
                        hole: samples[i].hole,
                        sample_idx: samples[i].sample_idx,
                        prop: p as u16,
                    });
                }
            }
        }

        // Periodic pruning to cap memory while keeping the most confident outliers.
        if suspects.len() > max_out * 2 {
            let kth = max_out;
            // Keep top `max_out` by confidence.
            suspects.select_nth_unstable_by(kth, Suspect::cmp_conf_desc);
            suspects.truncate(max_out);
        }
    }

    // Final sort for output
    suspects.sort_unstable_by(Suspect::cmp_conf_desc);

    // Output
    let mut out = String::new();
    out.push_str(&format!("{}\n", suspects.len()));
    for s in &suspects {
        let hole = hole_ids
            .get(s.hole as usize)
            .map(|x| x.as_str())
            .unwrap_or("UNKNOWN");
        let prop = prop_names
            .get(s.prop as usize)
            .map(|x| x.as_str())
            .unwrap_or("prop");
        out.push_str(&format!(
            "{} {} {} {:.3}\n",
            hole, s.sample_idx, prop, s.conf
        ));
    }
    print!("{out}");
}
