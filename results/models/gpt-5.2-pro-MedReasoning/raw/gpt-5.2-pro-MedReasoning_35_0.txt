```rust
use std::cmp::Reverse;
use std::collections::BinaryHeap;
use std::io::{self, Read};

struct FastInput {
    data: Vec<u8>,
    idx: usize,
}
impl FastInput {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data).unwrap();
        Self { data, idx: 0 }
    }
    fn skip_whitespace(&mut self) {
        while self.idx < self.data.len() && self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
    }
    fn next_usize(&mut self) -> Option<usize> {
        self.skip_whitespace();
        if self.idx >= self.data.len() {
            return None;
        }
        let mut val: usize = 0;
        while self.idx < self.data.len() && !self.data[self.idx].is_ascii_whitespace() {
            val = val * 10 + (self.data[self.idx] - b'0') as usize;
            self.idx += 1;
        }
        Some(val)
    }
}

fn build_dom_count(adj: &[Vec<usize>], selected: &[bool]) -> Vec<u32> {
    let n = adj.len();
    let mut dom = vec![0u32; n];
    for v in 0..n {
        if !selected[v] {
            continue;
        }
        dom[v] += 1;
        for &u in &adj[v] {
            dom[u] += 1;
        }
    }
    dom
}

fn can_remove_vertex(adj: &[Vec<usize>], v: usize, dom: &[u32]) -> bool {
    // After removing v, every vertex in N[v] (including v) must still have dom_count >= 1,
    // i.e., currently >= 2.
    if dom[v] < 2 {
        return false;
    }
    for &u in &adj[v] {
        if dom[u] < 2 {
            return false;
        }
    }
    true
}

fn validate_and_fix(adj: &[Vec<usize>], sol: &mut Vec<usize>) {
    let n = adj.len();
    let mut in_sol = vec![false; n];
    for &v in sol.iter() {
        if v < n {
            in_sol[v] = true;
        }
    }

    let mut dominated = vec![false; n];
    for &v in sol.iter() {
        dominated[v] = true;
        for &u in &adj[v] {
            dominated[u] = true;
        }
    }

    for u in 0..n {
        if !dominated[u] {
            // Add the undominated vertex itself (always valid).
            if !in_sol[u] {
                in_sol[u] = true;
                sol.push(u);
            }
            dominated[u] = true;
            for &w in &adj[u] {
                dominated[w] = true;
            }
        }
    }
}

fn main() {
    let mut inp = FastInput::new();
    let n = match inp.next_usize() {
        Some(x) => x,
        None => {
            // Empty input
            println!("0\n");
            return;
        }
    };
    let m = inp.next_usize().unwrap_or(0);

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u = match inp.next_usize() {
            Some(x) => x,
            None => break,
        };
        let v = match inp.next_usize() {
            Some(x) => x,
            None => break,
        };
        if u >= n || v >= n || u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    // Deduplicate adjacency lists (important for correctness of gain updates).
    for v in 0..n {
        adj[v].sort_unstable();
        adj[v].dedup();
    }

    if n == 0 {
        println!("0\n");
        return;
    }

    // Greedy construction with dynamic gain updates.
    // gain[v] = number of currently undominated vertices in closed neighborhood N[v] ∪ {v}.
    let mut gain: Vec<i32> = vec![0; n];
    for v in 0..n {
        gain[v] = (adj[v].len() as i32) + 1;
    }

    let mut dominated = vec![false; n];
    let mut selected = vec![false; n];
    let mut undom_remaining = n;

    let mut heap: BinaryHeap<(i32, usize)> = BinaryHeap::new();
    for v in 0..n {
        heap.push((gain[v], v));
    }

    let mut sol: Vec<usize> = Vec::new();
    let mut newly_dominated: Vec<usize> = Vec::new();

    while undom_remaining > 0 {
        let mut chosen: Option<usize> = None;

        while let Some((g, v)) = heap.pop() {
            if selected[v] {
                continue;
            }
            if g != gain[v] {
                continue; // stale
            }
            if g <= 0 {
                continue; // cannot help
            }
            chosen = Some(v);
            break;
        }

        let v = if let Some(v) = chosen {
            v
        } else {
            // Fallback (should be rare): pick any undominated vertex.
            let mut any = None;
            for u in 0..n {
                if !dominated[u] {
                    any = Some(u);
                    break;
                }
            }
            any.unwrap_or(0)
        };

        // Select v
        selected[v] = true;
        sol.push(v);

        newly_dominated.clear();

        // Mark vertices in closed neighborhood of v as dominated (first time only).
        if !dominated[v] {
            dominated[v] = true;
            undom_remaining -= 1;
            newly_dominated.push(v);
        }
        for &u in &adj[v] {
            if !dominated[u] {
                dominated[u] = true;
                undom_remaining -= 1;
                newly_dominated.push(u);
            }
        }

        // For each vertex that just became dominated, it reduces the gain of all vertices
        // in its closed neighborhood by 1 (since that vertex is no longer undominated).
        for &u in &newly_dominated {
            // w = u itself
            gain[u] -= 1;
            heap.push((gain[u], u));
            // w in neighbors of u
            for &w in &adj[u] {
                gain[w] -= 1;
                heap.push((gain[w], w));
            }
        }
    }

    // Pruning: remove redundant vertices if domination remains.
    // Do a couple of passes.
    for _pass in 0..2 {
        let mut dom = build_dom_count(&adj, &selected);

        let mut order: Vec<usize> = sol.iter().copied().collect();
        order.sort_unstable_by_key(|&v| adj[v].len()); // keep higher-degree vertices more often

        for v in order {
            if !selected[v] {
                continue;
            }
            if !can_remove_vertex(&adj, v, &dom) {
                continue;
            }

            // Remove v
            selected[v] = false;
            dom[v] -= 1;
            for &u in &adj[v] {
                dom[u] -= 1;
            }
        }

        sol.clear();
        for v in 0..n {
            if selected[v] {
                sol.push(v);
            }
        }
    }

    // Final safety: ensure validity (adds vertices if something went wrong).
    validate_and_fix(&adj, &mut sol);

    // Output
    println!("{}", sol.len());
    if sol.is_empty() {
        println!();
    } else {
        let mut out = String::new();
        for (i, v) in sol.iter().enumerate() {
            if i > 0 {
                out.push(' ');
            }
            out.push_str(&v.to_string());
        }
        println!("{}", out);
    }
}
```
