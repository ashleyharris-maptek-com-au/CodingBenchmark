#include <bits/stdc++.h>
using namespace std;

struct DynBitset {
    int n = 0;
    int words = 0;
    uint64_t lastMask = 0;
    vector<uint64_t> a;

    DynBitset() = default;
    explicit DynBitset(int n_) { init(n_); }

    void init(int n_) {
        n = n_;
        words = (n + 63) / 64;
        a.assign(words, 0ULL);
        int r = n % 64;
        lastMask = (r == 0) ? ~0ULL : ((1ULL << r) - 1ULL);
    }

    void setAll() {
        for (int i = 0; i < words; i++) a[i] = ~0ULL;
        if (words) a.back() &= lastMask;
    }

    inline void set(int i) { a[i >> 6] |= (1ULL << (i & 63)); }
    inline void reset(int i) { a[i >> 6] &= ~(1ULL << (i & 63)); }
    inline bool test(int i) const { return (a[i >> 6] >> (i & 63)) & 1ULL; }

    bool any() const {
        for (uint64_t w : a) if (w) return true;
        return false;
    }

    int find_first() const {
        for (int wi = 0; wi < words; wi++) {
            uint64_t w = a[wi];
            if (w) return (wi << 6) + __builtin_ctzll(w);
        }
        return -1;
    }

    void andWith(const DynBitset& o) {
        for (int i = 0; i < words; i++) a[i] &= o.a[i];
    }

    // this &= ~o
    void andNot(const DynBitset& o) {
        for (int i = 0; i < words; i++) a[i] &= ~o.a[i];
        if (words) a.back() &= lastMask;
    }
};

struct MaxWeightCliqueDyn {
    int n = 0;
    vector<long long> w;
    vector<DynBitset> adj;     // complement graph adjacency
    vector<int> order;         // vertices sorted by weight desc
    long long bestW = 0;
    vector<int> bestClique;
    vector<int> curClique;

    long long upperBound(const DynBitset& P) {
        // Greedy coloring bound (weighted): sum over color classes of max weight in class.
        DynBitset U = P;
        long long sum = 0;
        while (U.any()) {
            DynBitset avail = U;
            long long bestInColor = 0;
            while (avail.any()) {
                int v = avail.find_first();
                avail.reset(v);
                bestInColor = max(bestInColor, w[v]);
                U.reset(v);
                // same color must be an independent set in complement => remove neighbors
                avail.andNot(adj[v]);
            }
            sum += bestInColor;
        }
        return sum;
    }

    int selectVertex(const DynBitset& P) const {
        for (int v : order) if (P.test(v)) return v;
        // fallback
        return P.find_first();
    }

    void expand(DynBitset P, long long curW) {
        while (P.any()) {
            if (curW + upperBound(P) <= bestW) return;

            int v = selectVertex(P);

            DynBitset P2 = P;
            P2.andWith(adj[v]); // only vertices adjacent to v can remain in clique
            curClique.push_back(v);
            expand(P2, curW + w[v]);
            curClique.pop_back();

            P.reset(v); // exclude v
        }
        if (curW > bestW) {
            bestW = curW;
            bestClique = curClique;
        }
    }

    vector<int> solve(int n_, const vector<long long>& w_, const vector<DynBitset>& adj_) {
        n = n_;
        w = w_;
        adj = adj_;
        order.resize(n);
        iota(order.begin(), order.end(), 0);
        sort(order.begin(), order.end(), [&](int a, int b) {
            if (w[a] != w[b]) return w[a] > w[b];
            return a < b;
        });

        bestW = 0;
        bestClique.clear();
        curClique.clear();

        DynBitset P(n);
        P.setAll();
        expand(P, 0);

        return bestClique;
    }
};

struct MaxWeightClique64 {
    int n = 0;
    vector<long long> w;
    vector<uint64_t> adj; // complement graph adjacency masks (bit i set => neighbor i)
    vector<int> order;
    long long bestW = 0;
    uint64_t bestMask = 0;
    uint64_t curMask = 0;

    long long upperBound(uint64_t P) {
        long long sum = 0;
        uint64_t U = P;
        while (U) {
            uint64_t avail = U;
            long long bestInColor = 0;
            while (avail) {
                int v = __builtin_ctzll(avail);
                avail &= (avail - 1);
                bestInColor = max(bestInColor, w[v]);
                U &= ~(1ULL << v);
                avail &= ~adj[v];
            }
            sum += bestInColor;
        }
        return sum;
    }

    int selectVertex(uint64_t P) const {
        for (int v : order) if (P & (1ULL << v)) return v;
        return __builtin_ctzll(P);
    }

    void expand(uint64_t P, long long curW) {
        while (P) {
            if (curW + upperBound(P) <= bestW) return;

            int v = selectVertex(P);

            uint64_t P2 = P & adj[v];
            curMask |= (1ULL << v);
            expand(P2, curW + w[v]);
            curMask &= ~(1ULL << v);

            P &= ~(1ULL << v);
        }
        if (curW > bestW) {
            bestW = curW;
            bestMask = curMask;
        }
    }

    vector<int> solve(int n_, const vector<long long>& w_, const vector<uint64_t>& adj_) {
        n = n_;
        w = w_;
        adj = adj_;

        order.resize(n);
        iota(order.begin(), order.end(), 0);
        sort(order.begin(), order.end(), [&](int a, int b) {
            if (w[a] != w[b]) return w[a] > w[b];
            return a < b;
        });

        bestW = 0;
        bestMask = 0;
        curMask = 0;

        uint64_t P = (n == 64) ? ~0ULL : ((1ULL << n) - 1ULL);
        expand(P, 0);

        vector<int> clique;
        uint64_t m = bestMask;
        while (m) {
            int v = __builtin_ctzll(m);
            m &= (m - 1);
            clique.push_back(v);
        }
        return clique;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<long long> c(n);
    for (int i = 0; i < n; i++) cin >> c[i];

    vector<vector<int>> g(n);
    g.reserve(n);
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue;
        g[u].push_back(v);
        g[v].push_back(u);
    }

    vector<int> x(n, 0);

    // Only vertices with positive weight can ever help (since x_i can be 0).
    vector<char> active(n, 0);
    for (int i = 0; i < n; i++) active[i] = (c[i] > 0);

    // Connected components in the conflict graph induced by active vertices.
    vector<char> vis(n, 0);
    vector<int> stack;
    stack.reserve(n);

    for (int s = 0; s < n; s++) {
        if (!active[s] || vis[s]) continue;

        vector<int> comp;
        stack.clear();
        stack.push_back(s);
        vis[s] = 1;

        while (!stack.empty()) {
            int u = stack.back();
            stack.pop_back();
            comp.push_back(u);
            for (int v : g[u]) {
                if (active[v] && !vis[v]) {
                    vis[v] = 1;
                    stack.push_back(v);
                }
            }
        }

        if (comp.size() == 1) {
            x[comp[0]] = 1;
            continue;
        }

        // Build local index map for this component.
        int sz = (int)comp.size();
        vector<int> idx(n, -1);
        for (int i = 0; i < sz; i++) idx[comp[i]] = i;

        // Solve MWIS on this component by max weight clique in the complement graph.
        // Complement adjacency: edge exists iff there is NO conflict edge.
        if (sz <= 64) {
            vector<long long> w(sz);
            for (int i = 0; i < sz; i++) w[i] = c[comp[i]];

            uint64_t allMask = (sz == 64) ? ~0ULL : ((1ULL << sz) - 1ULL);
            vector<uint64_t> adj(sz, 0ULL);
            for (int i = 0; i < sz; i++) {
                adj[i] = allMask & ~(1ULL << i); // all except self
            }
            // Remove conflict edges from complement adjacency
            for (int iu = 0; iu < sz; iu++) {
                int u = comp[iu];
                for (int v : g[u]) {
                    int iv = (v >= 0 && v < n) ? idx[v] : -1;
                    if (iv != -1) {
                        adj[iu] &= ~(1ULL << iv);
                    }
                }
            }

            MaxWeightClique64 solver;
            vector<int> cliqueLocal = solver.solve(sz, w, adj);
            for (int li : cliqueLocal) x[comp[li]] = 1;
        } else {
            vector<long long> w(sz);
            for (int i = 0; i < sz; i++) w[i] = c[comp[i]];

            vector<DynBitset> adj(sz, DynBitset(sz));
            for (int i = 0; i < sz; i++) {
                adj[i].setAll();
                adj[i].reset(i);
            }
            // Remove conflict edges from complement adjacency
            for (int iu = 0; iu < sz; iu++) {
                int u = comp[iu];
                for (int v : g[u]) {
                    int iv = (v >= 0 && v < n) ? idx[v] : -1;
                    if (iv != -1) {
                        adj[iu].reset(iv);
                    }
                }
            }

            MaxWeightCliqueDyn solver;
            vector<int> cliqueLocal = solver.solve(sz, w, adj);
            for (int li : cliqueLocal) x[comp[li]] = 1;
        }
    }

    long long obj = 0;
    for (int i = 0; i < n; i++) if (x[i]) obj += c[i];

    cout << obj << "\n";
    for (int i = 0; i < n; i++) {
        if (i) cout << ' ';
        cout << x[i];
    }
    cout << "\n";
    return 0;
}
