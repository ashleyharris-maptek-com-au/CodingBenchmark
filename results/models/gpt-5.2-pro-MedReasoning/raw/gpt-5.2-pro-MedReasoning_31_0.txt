use std::collections::HashMap;
use std::io::{self, Read};

struct FastInput {
    data: Vec<u8>,
    pos: usize,
}
impl FastInput {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data).unwrap();
        Self { data, pos: 0 }
    }

    fn skip_ws(&mut self) {
        while self.pos < self.data.len() && self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
    }

    fn next_i64(&mut self) -> Option<i64> {
        self.skip_ws();
        if self.pos >= self.data.len() {
            return None;
        }
        let mut sign = 1i64;
        if self.data[self.pos] == b'-' {
            sign = -1;
            self.pos += 1;
        }
        let mut x: i64 = 0;
        while self.pos < self.data.len() {
            let c = self.data[self.pos];
            if !c.is_ascii_digit() {
                break;
            }
            x = x.saturating_mul(10).saturating_add((c - b'0') as i64);
            self.pos += 1;
        }
        Some(x * sign)
    }

    fn next_usize(&mut self) -> Option<usize> {
        self.next_i64().and_then(|v| {
            if v < 0 {
                None
            } else {
                Some(v as usize)
            }
        })
    }
}

fn gcd_u64(mut a: u64, mut b: u64) -> u64 {
    while b != 0 {
        let r = a % b;
        a = b;
        b = r;
    }
    a
}

fn last_word_mask(last_bit_inclusive: usize) -> u64 {
    let bits = (last_bit_inclusive % 64) + 1;
    if bits == 64 {
        u64::MAX
    } else {
        (1u64 << bits) - 1
    }
}

fn bit_test(bits: &[u64], idx: usize) -> bool {
    let w = idx >> 6;
    let b = idx & 63;
    ((bits[w] >> b) & 1) != 0
}

fn bit_set(bits: &mut [u64], idx: usize) {
    let w = idx >> 6;
    let b = idx & 63;
    bits[w] |= 1u64 << b;
}

/// Non-negative subset sum solver (values >= 0, target >= 0), returns original indices.
fn subset_sum_nonneg(values: &[i64], target: i64) -> Option<Vec<usize>> {
    if target < 0 {
        return None;
    }
    let target_u = target as usize;
    if target_u == 0 {
        return Some(Vec::new());
    }

    // Fast single-element check + build filtered items (0 < v <= target).
    let mut vals: Vec<usize> = Vec::new();
    let mut orig_idx: Vec<usize> = Vec::new();
    vals.reserve(values.len().min(1_000_000));
    orig_idx.reserve(values.len().min(1_000_000));

    // Quick 2-sum precheck when target is not too large.
    // Also gives us direct reconstruction (two indices).
    const TWOSUM_LIMIT: usize = 5_000_000;
    let mut first_idx: Vec<usize> = Vec::new();
    let mut second_idx: Vec<usize> = Vec::new();
    if target_u <= TWOSUM_LIMIT {
        first_idx = vec![usize::MAX; target_u + 1];
        second_idx = vec![usize::MAX; target_u + 1];
    }

    // gcd precheck for non-negative values relevant to target
    let mut g: u64 = 0;
    let mut sum_all: u128 = 0;

    for (i, &x) in values.iter().enumerate() {
        if x == target {
            return Some(vec![i]);
        }
        if x > 0 && x <= target {
            let xu = x as usize;
            vals.push(xu);
            orig_idx.push(i);

            g = if g == 0 { xu as u64 } else { gcd_u64(g, xu as u64) };

            if !first_idx.is_empty() {
                if first_idx[xu] == usize::MAX {
                    first_idx[xu] = i;
                } else if second_idx[xu] == usize::MAX {
                    second_idx[xu] = i;
                }
            }
        }
        if x > 0 {
            sum_all = sum_all.saturating_add(x as u128);
        }
    }

    if sum_all < target_u as u128 {
        return None;
    }

    if g != 0 && (target_u as u64) % g != 0 {
        return None;
    }

    if !first_idx.is_empty() {
        if first_idx[target_u] != usize::MAX {
            return Some(vec![first_idx[target_u]]);
        }
        for v in 1..=target_u {
            let w = target_u - v;
            if first_idx[v] != usize::MAX && first_idx[w] != usize::MAX {
                if v != w {
                    return Some(vec![first_idx[v], first_idx[w]]);
                } else if second_idx[v] != usize::MAX {
                    return Some(vec![first_idx[v], second_idx[v]]);
                }
            }
        }
    }

    if vals.is_empty() {
        return None;
    }

    let words = (target_u >> 6) + 1;
    let mask_last = last_word_mask(target_u);

    // If target isn't huge, track parents during the bitset DP (exact reconstruction in one pass).
    const PARENT_LIMIT: usize = 10_000_000;

    if target_u <= PARENT_LIMIT {
        let mut dp = vec![0u64; words];
        bit_set(&mut dp, 0);

        let mut parent_item = vec![-1i32; target_u + 1];
        let mut parent_prev = vec![0u32; target_u + 1];
        parent_item[0] = -2;

        for (j, &v) in vals.iter().enumerate() {
            let shift = v;
            let word_shift = shift >> 6;
            let bit_shift = shift & 63;

            // shift-left OR in-place, descending to preserve sources
            for i in (word_shift..words).rev() {
                let src = i - word_shift;
                let mut shifted = dp[src] << bit_shift;
                if bit_shift != 0 && src > 0 {
                    shifted |= dp[src - 1] >> (64 - bit_shift);
                }
                if i == words - 1 {
                    shifted &= mask_last;
                }

                let new_bits = shifted & !dp[i];
                if new_bits != 0 {
                    let base = i << 6;
                    let mut w = new_bits;
                    while w != 0 {
                        let tz = w.trailing_zeros() as usize;
                        let sum = base + tz;
                        if sum <= target_u {
                            parent_item[sum] = j as i32;
                            parent_prev[sum] = (sum - shift) as u32;
                        }
                        w &= w - 1;
                    }
                }
                dp[i] |= shifted;
            }

            if bit_test(&dp, target_u) {
                break;
            }
        }

        if !bit_test(&dp, target_u) {
            return None;
        }

        // Backtrack
        let mut need = target_u;
        let mut chosen: Vec<usize> = Vec::new();
        while need != 0 {
            let pj = parent_item[need];
            if pj < 0 {
                // Should not happen if reachable; defensive.
                return None;
            }
            let jj = pj as usize;
            chosen.push(orig_idx[jj]);
            need = parent_prev[need] as usize;
        }
        chosen.sort_unstable();
        return Some(chosen);
    }

    // Large target: store periodic checkpoints; reconstruct chunk-by-chunk using a HashMap of parents.
    // This is still exact, but may be slow/memory-heavy for worst cases.
    let mut dp = vec![0u64; words];
    bit_set(&mut dp, 0);

    let dp_bytes = words.saturating_mul(8);
    let mem_cap: usize = 768usize * 1024 * 1024; // ~768MB for checkpoints
    let mut max_ckpt = if dp_bytes == 0 { 2 } else { mem_cap / dp_bytes };
    if max_ckpt < 2 {
        max_ckpt = 2;
    }
    if max_ckpt > 1024 {
        max_ckpt = 1024;
    }
    let m = vals.len();
    let chunk_size = (m + max_ckpt - 1) / max_ckpt;

    let mut checkpoints: Vec<Vec<u64>> = Vec::new();
    checkpoints.push(dp.clone());

    for (j, &v) in vals.iter().enumerate() {
        let shift = v;
        let word_shift = shift >> 6;
        let bit_shift = shift & 63;

        for i in (word_shift..words).rev() {
            let src = i - word_shift;
            let mut shifted = dp[src] << bit_shift;
            if bit_shift != 0 && src > 0 {
                shifted |= dp[src - 1] >> (64 - bit_shift);
            }
            if i == words - 1 {
                shifted &= mask_last;
            }
            dp[i] |= shifted;
        }

        if (j + 1) % chunk_size == 0 {
            checkpoints.push(dp.clone());
        }
        if bit_test(&dp, target_u) {
            break;
        }
    }
    if checkpoints.last().map_or(true, |last| last.as_ptr() != dp.as_ptr()) {
        if checkpoints.last().map_or(true, |c| c.len() != dp.len()) || checkpoints.last().unwrap() != &dp {
            // Ensure final checkpoint exists.
            if checkpoints.last().map_or(true, |c| !std::ptr::eq(c.as_ptr(), dp.as_ptr())) {
                // no-op; just avoid warnings about ptr::eq usage in the condition
            }
        }
    }
    if checkpoints.last().map_or(true, |c| !bit_test(c, target_u)) && bit_test(&dp, target_u) {
        checkpoints.push(dp.clone());
    }
    if !bit_test(&dp, target_u) {
        return None;
    }

    let mut need = target_u;
    let mut chosen: Vec<usize> = Vec::new();

    let chunks = checkpoints.len() - 1;
    for chunk in (0..chunks).rev() {
        let start = chunk * chunk_size;
        let end = (start + chunk_size).min(m);
        let dp_before = &checkpoints[chunk];

        if bit_test(dp_before, need) {
            continue;
        }

        // Recompute DP for this chunk, starting from dp_before, recording first parent for each newly reached sum.
        let mut cur = dp_before.clone();
        let mut parents: HashMap<usize, (usize, usize)> = HashMap::new();

        for j in start..end {
            let shift = vals[j];
            let word_shift = shift >> 6;
            let bit_shift = shift & 63;

            for i in (word_shift..words).rev() {
                let src = i - word_shift;
                let mut shifted = cur[src] << bit_shift;
                if bit_shift != 0 && src > 0 {
                    shifted |= cur[src - 1] >> (64 - bit_shift);
                }
                if i == words - 1 {
                    shifted &= mask_last;
                }

                let new_bits = shifted & !cur[i];
                if new_bits != 0 {
                    let base = i << 6;
                    let mut w = new_bits;
                    while w != 0 {
                        let tz = w.trailing_zeros() as usize;
                        let sum = base + tz;
                        if sum <= target_u {
                            // sum was not reachable before, so parent is unique here.
                            parents.insert(sum, (sum - shift, j));
                        }
                        w &= w - 1;
                    }
                }

                cur[i] |= shifted;
            }

            if bit_test(&cur, need) {
                break;
            }
        }

        if !bit_test(&cur, need) {
            // Defensive: should not happen if checkpoints are correct.
            return None;
        }

        // Backtrack within this chunk until we reach a sum that was reachable before the chunk.
        while !bit_test(dp_before, need) {
            let (prev, j) = match parents.get(&need) {
                Some(v) => *v,
                None => return None,
            };
            chosen.push(orig_idx[j]);
            need = prev;
        }

        if need == 0 {
            break;
        }
    }

    if need != 0 {
        // Defensive: expected to reach 0.
        return None;
    }

    chosen.sort_unstable();
    Some(chosen)
}

/// Meet-in-the-middle exact for mixed signs when n is small.
fn subset_sum_mitm(values: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = values.len();
    let mid = n / 2;

    let left: Vec<(i64, usize)> = (0..mid).map(|i| (values[i], i)).collect();
    let right: Vec<(i64, usize)> = (mid..n).map(|i| (values[i], i)).collect();

    fn gen_sums(arr: &[(i64, usize)]) -> Vec<(i64, u64)> {
        let m = arr.len();
        let mut out = Vec::with_capacity(1usize << m);
        out.push((0, 0));
        for (i, &(v, _idx)) in arr.iter().enumerate() {
            let bit = 1u64 << i;
            let cur_len = out.len();
            for j in 0..cur_len {
                let (s, mask) = out[j];
                out.push((s + v, mask | bit));
            }
        }
        out
    }

    let mut sums_l = gen_sums(&left);
    sums_l.sort_by(|a, b| a.0.cmp(&b.0));

    let sums_r = gen_sums(&right);

    // For binary search on sums_l
    let mut l_sums_only = Vec::with_capacity(sums_l.len());
    for (s, _) in sums_l.iter() {
        l_sums_only.push(*s);
    }

    for (sr, mr) in sums_r {
        let need = target - sr;
        if let Ok(pos) = l_sums_only.binary_search(&need) {
            let ml = sums_l[pos].1;
            let mut chosen = Vec::new();

            for i in 0..left.len() {
                if ((ml >> i) & 1) != 0 {
                    chosen.push(left[i].1);
                }
            }
            for i in 0..right.len() {
                if ((mr >> i) & 1) != 0 {
                    chosen.push(right[i].1);
                }
            }
            chosen.sort_unstable();
            return Some(chosen);
        }
    }

    None
}

/// Mixed-sign exact solver using offset bitset when the reachable sum range is small enough.
fn subset_sum_mixed_range(values: &[i64], target: i64) -> Option<Vec<usize>> {
    // Compute min/max achievable sums by taking all negatives / all positives.
    let mut min_sum: i128 = 0;
    let mut max_sum: i128 = 0;
    for &x in values {
        if x < 0 {
            min_sum += x as i128;
        } else {
            max_sum += x as i128;
        }
    }
    let t = target as i128;
    if t < min_sum || t > max_sum {
        return None;
    }

    let range_i128 = max_sum - min_sum;
    if range_i128 < 0 {
        return None;
    }
    if range_i128 > (usize::MAX as i128) {
        return None;
    }
    let range = range_i128 as usize;

    const RANGE_LIMIT: usize = 10_000_000;
    if range > RANGE_LIMIT {
        return None;
    }

    let offset_i128 = -min_sum;
    if offset_i128 < 0 || offset_i128 > (usize::MAX as i128) {
        return None;
    }
    let offset = offset_i128 as usize;

    let target_idx_i128 = t + offset_i128;
    if target_idx_i128 < 0 || target_idx_i128 > (range as i128) {
        return None;
    }
    let target_idx = target_idx_i128 as usize;

    let words = (range >> 6) + 1;
    let mask_last = last_word_mask(range);

    let mut dp = vec![0u64; words];
    bit_set(&mut dp, offset);

    let mut parent_item = vec![-1i32; range + 1];
    let mut parent_prev = vec![0u32; range + 1];
    parent_item[offset] = -2;

    for (j, &val) in values.iter().enumerate() {
        if val == 0 {
            continue;
        }
        if val > 0 {
            let shift = val as usize;
            let word_shift = shift >> 6;
            let bit_shift = shift & 63;

            for i in (word_shift..words).rev() {
                let src = i - word_shift;
                let mut shifted = dp[src] << bit_shift;
                if bit_shift != 0 && src > 0 {
                    shifted |= dp[src - 1] >> (64 - bit_shift);
                }
                if i == words - 1 {
                    shifted &= mask_last;
                }
                let new_bits = shifted & !dp[i];
                if new_bits != 0 {
                    let base = i << 6;
                    let mut w = new_bits;
                    while w != 0 {
                        let tz = w.trailing_zeros() as usize;
                        let idx = base + tz;
                        if idx <= range {
                            parent_item[idx] = j as i32;
                            parent_prev[idx] = (idx - shift) as u32;
                        }
                        w &= w - 1;
                    }
                }
                dp[i] |= shifted;
            }
        } else {
            let shift = (-val) as usize;
            let word_shift = shift >> 6;
            let bit_shift = shift & 63;

            // shift-right OR in-place, ascending to preserve sources
            let max_i = words - word_shift;
            for i in 0..max_i {
                let src = i + word_shift;
                let mut shifted = dp[src] >> bit_shift;
                if bit_shift != 0 && src + 1 < words {
                    shifted |= dp[src + 1] << (64 - bit_shift);
                }
                // mask is only necessary on the last word; still safe to keep dp masked after update.
                let new_bits = shifted & !dp[i];
                if new_bits != 0 {
                    let base = i << 6;
                    let mut w = new_bits;
                    while w != 0 {
                        let tz = w.trailing_zeros() as usize;
                        let idx = base + tz;
                        if idx <= range {
                            parent_item[idx] = j as i32;
                            parent_prev[idx] = (idx + shift) as u32;
                        }
                        w &= w - 1;
                    }
                }
                dp[i] |= shifted;
            }
            dp[words - 1] &= mask_last;
        }

        if bit_test(&dp, target_idx) {
            break;
        }
    }

    if !bit_test(&dp, target_idx) {
        return None;
    }

    // Backtrack
    let mut idx = target_idx;
    let mut chosen: Vec<usize> = Vec::new();
    while idx != offset {
        let pj = parent_item[idx];
        if pj < 0 {
            return None;
        }
        chosen.push(pj as usize);
        idx = parent_prev[idx] as usize;
    }
    chosen.sort_unstable();
    Some(chosen)
}

/// Mixed-sign fallback using HashMap DP (exact, but only feasible for modest state sizes).
fn subset_sum_hash_dp(values: &[i64], target: i64) -> Option<Vec<usize>> {
    const MAX_STATES: usize = 5_000_000;

    // sum -> (prev_sum, index_used)
    let mut parent: HashMap<i64, (i64, usize)> = HashMap::new();
    parent.insert(0, (i64::MIN, usize::MAX));

    for (i, &v) in values.iter().enumerate() {
        let keys: Vec<i64> = parent.keys().copied().collect();
        for s in keys {
            let ns = s.saturating_add(v);
            if !parent.contains_key(&ns) {
                parent.insert(ns, (s, i));
                if ns == target {
                    break;
                }
            }
        }
        if parent.len() > MAX_STATES {
            return None;
        }
        if parent.contains_key(&target) {
            break;
        }
    }

    if !parent.contains_key(&target) {
        return None;
    }

    let mut chosen: Vec<usize> = Vec::new();
    let mut cur = target;
    while cur != 0 {
        let (prev, idx) = match parent.get(&cur) {
            Some(v) => *v,
            None => return None,
        };
        if idx == usize::MAX {
            return None;
        }
        chosen.push(idx);
        cur = prev;
    }
    chosen.sort_unstable();
    Some(chosen)
}

fn solve(values: &[i64], target: i64) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }
    if values.is_empty() {
        return None;
    }

    let all_nonneg = values.iter().all(|&x| x >= 0);
    let all_nonpos = values.iter().all(|&x| x <= 0);

    if all_nonneg {
        if target < 0 {
            return None;
        }
        return subset_sum_nonneg(values, target);
    }
    if all_nonpos {
        if target > 0 {
            return None;
        }
        // Negate to reduce to non-negative case.
        let mut neg = Vec::with_capacity(values.len());
        for &x in values {
            neg.push(-x);
        }
        return subset_sum_nonneg(&neg, -target);
    }

    // Mixed signs:
    // 1) exact MiTM for small n
    if values.len() <= 44 {
        if let Some(ans) = subset_sum_mitm(values, target) {
            return Some(ans);
        }
    }

    // 2) exact offset bitset if range is small
    if let Some(ans) = subset_sum_mixed_range(values, target) {
        return Some(ans);
    }

    // 3) exact hash DP if state space stays small
    subset_sum_hash_dp(values, target)
}

fn main() {
    let mut input = FastInput::new();
    let n = match input.next_usize() {
        Some(v) => v,
        None => return,
    };
    let target = match input.next_i64() {
        Some(v) => v,
        None => return,
    };

    let mut values = Vec::with_capacity(n);
    for _ in 0..n {
        if let Some(x) = input.next_i64() {
            values.push(x);
        } else {
            break;
        }
    }

    let ans = solve(&values, target);

    let mut out = String::new();
    match ans {
        None => {
            out.push_str("NO\n");
        }
        Some(mut idxs) => {
            out.push_str("YES\n");
            idxs.sort_unstable();
            if idxs.is_empty() {
                out.push('\n');
            } else {
                for (k, idx) in idxs.iter().enumerate() {
                    if k != 0 {
                        out.push(' ');
                    }
                    out.push_str(&idx.to_string());
                }
                out.push('\n');
            }
        }
    }
    print!("{out}");
}
