OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

%void = OpTypeVoid
%fn = OpTypeFunction %void

%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptrUniformUBO = OpTypePointer Uniform %UBO
%uboVar = OpVariable %ptrUniformUBO Uniform

%ptrInputV3 = OpTypePointer Input %v3float
%ptrInputV2 = OpTypePointer Input %v2float
%ptrOutputV4 = OpTypePointer Output %v4float

%inWorldPos = OpVariable %ptrInputV3 Input
%inNormal = OpVariable %ptrInputV3 Input
%inUV = OpVariable %ptrInputV2 Input
%outColor = OpVariable %ptrOutputV4 Output

%ptrUniformV4 = OpTypePointer Uniform %v4float

%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3

%f0 = OpConstant %float 0.0
%f1 = OpConstant %float 1.0
%f0_4 = OpConstant %float 0.4
%f0_6 = OpConstant %float 0.6
%f0_5 = OpConstant %float 0.5
%f2 = OpConstant %float 2.0
%f8 = OpConstant %float 8.0
%f16 = OpConstant %float 16.0
%f0_05 = OpConstant %float 0.05
%f0_1 = OpConstant %float 0.1

%f0_7 = OpConstant %float 0.7
%f0_2 = OpConstant %float 0.2

%brickColor = OpConstantComposite %v3float %f0_7 %f0_2 %f0_1
%mortarColor = OpConstantComposite %v3float %f0_7 %f0_7 %f0_7

%main = OpFunction %void None %fn
%entry = OpLabel

%uv = OpLoad %v2float %inUV
%u = OpCompositeExtract %float %uv 0
%v = OpCompositeExtract %float %uv 1

%uScaled = OpFMul %float %u %f8
%vScaled = OpFMul %float %v %f16

%row = OpExtInst %float %1 Floor %vScaled
%rowHalfTmp = OpFMul %float %row %f0_5
%rowHalf = OpExtInst %float %1 Floor %rowHalfTmp
%evenPart = OpFMul %float %rowHalf %f2
%parity = OpFSub %float %row %evenPart
%offset = OpFMul %float %parity %f0_5
%xScaled = OpFAdd %float %uScaled %offset

%fx = OpExtInst %float %1 Fract %xScaled
%fy = OpExtInst %float %1 Fract %vScaled

%condX1 = OpFOrdLessThan %bool %fx %f0_05
%oneMinusMx = OpFSub %float %f1 %f0_05
%condX2 = OpFOrdGreaterThan %bool %fx %oneMinusMx
%inX = OpLogicalOr %bool %condX1 %condX2

%condY1 = OpFOrdLessThan %bool %fy %f0_1
%oneMinusMy = OpFSub %float %f1 %f0_1
%condY2 = OpFOrdGreaterThan %bool %fy %oneMinusMy
%inY = OpLogicalOr %bool %condY1 %condY2

%isMortar = OpLogicalOr %bool %inX %inY
%baseColor = OpSelect %v3float %isMortar %mortarColor %brickColor

%nIn = OpLoad %v3float %inNormal
%n = OpExtInst %v3float %1 Normalize %nIn

%pos = OpLoad %v3float %inWorldPos
%lightPosPtr = OpAccessChain %ptrUniformV4 %uboVar %int_3
%lightPos4 = OpLoad %v4float %lightPosPtr
%lightPos3 = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2

%Lraw = OpFSub %v3float %lightPos3 %pos
%L = OpExtInst %v3float %1 Normalize %Lraw

%ndotl = OpDot %float %n %L
%diffuse = OpExtInst %float %1 FMax %ndotl %f0
%scaledDiffuse = OpFMul %float %f0_6 %diffuse
%shade = OpFAdd %float %f0_4 %scaledDiffuse

%lit = OpVectorTimesScalar %v3float %baseColor %shade

%r = OpCompositeExtract %float %lit 0
%g = OpCompositeExtract %float %lit 1
%b = OpCompositeExtract %float %lit 2
%outV4 = OpCompositeConstruct %v4float %r %g %b %f1
OpStore %outColor %outV4

OpReturn
OpFunctionEnd
