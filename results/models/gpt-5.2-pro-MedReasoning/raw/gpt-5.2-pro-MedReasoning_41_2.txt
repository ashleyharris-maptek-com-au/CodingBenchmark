; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inUV %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %inUV Location 2
OpDecorate %outColor Location 0

OpName %main "main"
OpName %inUV "uv"
OpName %outColor "outColor"

%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v4float = OpTypeVector %float 4
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%functype = OpTypeFunction %void

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1

%inUV = OpVariable %_ptr_Input_v2float Input
%outColor = OpVariable %_ptr_Output_v4float Output

%main = OpFunction %void None %functype
%entry = OpLabel
%uv = OpLoad %v2float %inUV
%u = OpCompositeExtract %float %uv 0
%v = OpCompositeExtract %float %uv 1
%rgba = OpCompositeConstruct %v4float %u %v %float_0 %float_1
OpStore %outColor %rgba
OpReturn
OpFunctionEnd
