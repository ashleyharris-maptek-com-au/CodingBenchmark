```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 80
OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_in "gl_in"
OpName %color_in "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_in 0 Offset 0

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 Offset 0

OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %arr_v3_in ArrayStride 16

OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

%void = OpTypeVoid
%void_fn = OpTypeFunction %void

%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%f_one_third = OpConstant %float 0.333333343
%f_shrink = OpConstant %float 0.699999988

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%arr_v3_in = OpTypeArray %v3float %uint_3

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%color_in = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %void_fn
%entry = OpLabel

; p0, p1, p2 from gl_in[].gl_Position
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr

; centroid C = (p0 + p1 + p2) / 3
%sum01 = OpFAdd %v4float %p0 %p1
%sum = OpFAdd %v4float %sum01 %p2
%C = OpVectorTimesScalar %v4float %sum %f_one_third

; out position pointer (gl_out.gl_Position)
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; ---- emit vertex 0 ----
%diff0 = OpFSub %v4float %p0 %C
%sdiff0 = OpVectorTimesScalar %v4float %diff0 %f_shrink
%new0 = OpFAdd %v4float %C %sdiff0
OpStore %out_pos_ptr %new0

%c0_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
%c0 = OpLoad %v3float %c0_ptr
OpStore %outColor %c0
OpEmitVertex

; ---- emit vertex 1 ----
%diff1 = OpFSub %v4float %p1 %C
%sdiff1 = OpVectorTimesScalar %v4float %diff1 %f_shrink
%new1 = OpFAdd %v4float %C %sdiff1
OpStore %out_pos_ptr %new1

%c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
%c1 = OpLoad %v3float %c1_ptr
OpStore %outColor %c1
OpEmitVertex

; ---- emit vertex 2 ----
%diff2 = OpFSub %v4float %p2 %C
%sdiff2 = OpVectorTimesScalar %v4float %diff2 %f_shrink
%new2 = OpFAdd %v4float %C %sdiff2
OpStore %out_pos_ptr %new2

%c2_ptr = OpAccessChain %ptr_in_v3 %color_in %int_2
%c2 = OpLoad %v3float %c2_ptr
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
