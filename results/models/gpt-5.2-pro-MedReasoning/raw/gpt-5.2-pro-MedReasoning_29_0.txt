using System;
using System.Buffers;
using System.Collections.Concurrent;
using System.Collections.Generic;
using System.Diagnostics;
using System.Numerics;
using System.Threading;
using System.Threading.Tasks;

public static class Program
{
    // Tuneable knobs
    const int GlobalBitsetNLimit = 30000;     // build full bitset adjacency and solve directly
    const int LocalBitsetLimit = 6000;        // per-vertex induced subproblem max size for exact bitset solver
    const int SeedGreedyTries = 48;           // heuristic seeds for initial lower bound
    const int TimeLimitSeconds = 295;         // safety margin under 5 minutes

    public static void Main()
    {
        var sw = Stopwatch.StartNew();
        var fs = new FastScanner(Console.OpenStandardInput());

        int n = fs.NextInt();
        int m = fs.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }
        if (m <= 0)
        {
            Console.WriteLine(1);
            Console.WriteLine(0);
            return;
        }

        // Read edges once (needed to build CSR or global bitset).
        int[] eu = new int[m];
        int[] ev = new int[m];
        int[] deg = new int[n];

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if ((uint)u >= (uint)n || (uint)v >= (uint)n || u == v)
            {
                // Ignore malformed/self-loop edges (shouldn't appear in valid tests).
                // Still store something safe to keep indexing consistent.
                eu[i] = 0; ev[i] = 0;
                continue;
            }
            eu[i] = u;
            ev[i] = v;
            deg[u]++;
            deg[v]++;
        }

        // If graph is small enough, solve with a single global bitset adjacency (exact).
        if (n <= GlobalBitsetNLimit && sw.Elapsed.TotalSeconds < TimeLimitSeconds)
        {
            SolveWithGlobalBitset(n, m, eu, ev, sw);
            return;
        }

        // Otherwise, build CSR adjacency lists (sorted), compute core/degeneracy order, then
        // run a degeneracy-based exact search with per-vertex induced bitset subproblems.
        SolveWithCSRAndDegeneracy(n, m, eu, ev, deg, sw);
    }

    static void SolveWithGlobalBitset(int n, int m, int[] eu, int[] ev, Stopwatch sw)
    {
        int words = (n + 63) >> 6;
        ulong[] adj = new ulong[n * words];

        for (int i = 0; i < m; i++)
        {
            int u = eu[i], v = ev[i];
            if (u == v) continue;
            if ((uint)u >= (uint)n || (uint)v >= (uint)n) continue;

            int uw = u * words + (v >> 6);
            int vw = v * words + (u >> 6);
            ulong um = 1UL << (v & 63);
            ulong vm = 1UL << (u & 63);
            adj[uw] |= um;
            adj[vw] |= vm;
        }

        // Heuristic lower bound: greedy on bitsets
        int[] seedClique = new int[n];
        int seedSize = GreedyCliqueBitset(adj, n, words, seedClique);

        int[] bestClique = new int[n];
        Array.Copy(seedClique, bestClique, seedSize);

        // Exact solve (iterative Tomita-style with coloring bound).
        var solver = new BitsetMaxCliqueSolver(adj, n, words);
        int bestSize = solver.Solve(seedSize, bestClique, sw, TimeLimitSeconds);

        // Output
        Console.WriteLine(bestSize);
        for (int i = 0; i < bestSize; i++)
        {
            if (i != 0) Console.Write(' ');
            Console.Write(bestClique[i]);
        }
        Console.WriteLine();
    }

    static void SolveWithCSRAndDegeneracy(int n, int m, int[] eu, int[] ev, int[] deg, Stopwatch sw)
    {
        // Build CSR adjacency: offsets + edges (size 2m)
        int[] off = new int[n + 1];
        for (int i = 0; i < n; i++) off[i + 1] = off[i] + deg[i];
        int[] cur = (int[])off.Clone();
        int[] adj = new int[off[n]];

        for (int i = 0; i < m; i++)
        {
            int u = eu[i], v = ev[i];
            if ((uint)u >= (uint)n || (uint)v >= (uint)n || u == v) continue;
            adj[cur[u]++] = v;
            adj[cur[v]++] = u;
        }

        // Free edge arrays if possible
        eu = null!;
        ev = null!;
        GC.Collect(0, GCCollectionMode.Forced, blocking: false, compacting: false);

        // Sort each adjacency segment for binary search / intersections
        for (int v = 0; v < n; v++)
        {
            int start = off[v];
            int len = off[v + 1] - start;
            if (len > 1) Array.Sort(adj, start, len);
        }

        // Core decomposition (degeneracy ordering)
        int[] core = new int[n];
        int[] order = new int[n];
        int maxCore = CoreDecomposition(n, off, adj, deg, core, order);
        int upperBound = maxCore + 1;

        // Quick heuristic to get a decent initial clique (lower bound)
        int[] bestClique = new int[Math.Min(upperBound, 1024)]; // will grow if needed
        int bestSize = 1;
        bestClique[0] = 0;

        // Seeds: max degree vertex, max core vertex, plus some pseudo-random vertices.
        int vMaxDeg = 0, vMaxCore = 0;
        for (int i = 0; i < n; i++)
        {
            if (deg[i] > deg[vMaxDeg]) vMaxDeg = i;
            if (core[i] > core[vMaxCore]) vMaxCore = i;
        }

        var seedVertices = new List<int>(SeedGreedyTries + 4) { vMaxDeg, vMaxCore };
        // deterministic xorshift-ish generator
        ulong x = 0x9E3779B97F4A7C15UL;
        for (int i = 0; i < SeedGreedyTries; i++)
        {
            x ^= x << 7; x ^= x >> 9; x ^= x << 8;
            int v = (int)(x % (ulong)n);
            seedVertices.Add(v);
        }

        object bestLock = new object();

        Parallel.ForEach(seedVertices,
            new ParallelOptions { MaxDegreeOfParallelism = Environment.ProcessorCount },
            () => new GreedyWorkBuffers(),
            (seed, loopState, local) =>
            {
                if (sw.Elapsed.TotalSeconds >= TimeLimitSeconds) { loopState.Stop(); return local; }
                int[] clique = local.Clique;
                int size = GreedyCliqueCSR(seed, off, adj, deg, clique, local);
                lock (bestLock)
                {
                    if (size > bestSize)
                    {
                        bestSize = size;
                        if (bestClique.Length < bestSize) bestClique = new int[Math.Max(bestSize, bestClique.Length * 2)];
                        Array.Copy(clique, bestClique, bestSize);
                    }
                }
                return local;
            },
            _ => { });

        if (bestSize >= upperBound)
        {
            Console.WriteLine(bestSize);
            for (int i = 0; i < bestSize; i++)
            {
                if (i != 0) Console.Write(' ');
                Console.Write(bestClique[i]);
            }
            Console.WriteLine();
            return;
        }

        // Degeneracy rank for later-neighbor filtering
        int[] rank = new int[n];
        for (int i = 0; i < n; i++) rank[order[i]] = i;

        // Cancellation when optimum reached or time exceeded
        using var cts = new CancellationTokenSource();
        var token = cts.Token;

        // Parallel exact search over vertices in degeneracy order.
        // For each vertex v, solve maximum clique inside N+(v) (later neighbors),
        // then add v.
        int globalBestSize = bestSize;
        int[] globalBestClique = bestClique; // may be resized under lock

        int chunk = 1024;
        var rangePartitioner = Partitioner.Create(0, n, chunk);

        Parallel.ForEach(rangePartitioner,
            new ParallelOptions { CancellationToken = token, MaxDegreeOfParallelism = Environment.ProcessorCount },
            () => new LocalSearchContext(),
            (range, loopState, ctx) =>
            {
                for (int idx = range.Item1; idx < range.Item2; idx++)
                {
                    if (sw.Elapsed.TotalSeconds >= TimeLimitSeconds)
                    {
                        cts.Cancel();
                        break;
                    }

                    int v = order[idx];
                    int curBest = Volatile.Read(ref globalBestSize);
                    if (curBest >= upperBound)
                    {
                        cts.Cancel();
                        break;
                    }

                    // Upper bound for any clique containing v via core number
                    if (core[v] + 1 <= curBest) continue;

                    // Collect later neighbors (rank larger). Also safe to filter by core>=curBest
                    // because we only care about cliques that can beat curBest.
                    int start = off[v], end = off[v + 1];
                    int degv = end - start;
                    if (degv <= 0) continue;

                    int needCore = curBest; // any vertex in a clique of size > curBest has core >= curBest
                    int candCount = 0;
                    int[] cand = ctx.RentInt(degv);
                    int rv = idx; // rank[v] == idx since v = order[idx]
                    for (int p = start; p < end; p++)
                    {
                        int u = adj[p];
                        if (rank[u] > rv && core[u] >= needCore)
                            cand[candCount++] = u;
                    }

                    if (1 + candCount <= curBest) { ctx.ReturnInt(cand); continue; }
                    if (candCount > LocalBitsetLimit)
                    {
                        // Too large for exact local bitset; attempt a quick greedy improvement only.
                        // (If the graph is very dense with huge degeneracy at this scale, exact is infeasible.)
                        int[] tmpClique = ctx.TempClique;
                        int got = GreedyCliqueFromCandidates(v, cand, candCount, off, adj, deg, tmpClique, ctx.GreedyBuf);
                        if (got > curBest)
                        {
                            lock (bestLock)
                            {
                                if (got > globalBestSize)
                                {
                                    globalBestSize = got;
                                    if (globalBestClique.Length < got) globalBestClique = new int[Math.Max(got, globalBestClique.Length * 2)];
                                    Array.Copy(tmpClique, globalBestClique, got);
                                }
                            }
                        }
                        ctx.ReturnInt(cand);
                        continue;
                    }

                    // Exact solve induced subgraph on cand using local bitset graph
                    Array.Sort(cand, 0, candCount); // stable for deterministic mapping

                    ctx.Map.Build(cand, candCount);
                    int t = candCount;
                    int words = (t + 63) >> 6;
                    int flatLen = t * words;

                    ulong[] localAdj = ctx.RentUlong(flatLen);
                    Array.Clear(localAdj, 0, flatLen);

                    for (int i = 0; i < t; i++)
                    {
                        int u = cand[i];
                        int su = off[u], euu = off[u + 1];
                        int rowBase = i * words;

                        for (int p = su; p < euu; p++)
                        {
                            int w = adj[p];
                            int j = ctx.Map.Get(w);
                            if (j >= 0)
                            {
                                localAdj[rowBase + (j >> 6)] |= 1UL << (j & 63);
                            }
                        }
                    }

                    // Symmetrize (in case some edges were missed due to duplicates/ordering)
                    for (int i = 0; i < t; i++)
                    {
                        int rowBase = i * words;
                        for (int wi = 0; wi < words; wi++)
                        {
                            ulong bits = localAdj[rowBase + wi];
                            while (bits != 0)
                            {
                                int b = BitOperations.TrailingZeroCount(bits);
                                int j = (wi << 6) + b;
                                if (j < t)
                                {
                                    localAdj[j * words + (i >> 6)] |= 1UL << (i & 63);
                                }
                                bits &= bits - 1;
                            }
                        }
                    }

                    // Lower bound for local solver is (curBest - 1) because we add v after.
                    int lower = curBest - 1;
                    if (lower < 0) lower = 0;

                    var solver = ctx.Solver;
                    solver.Reset(localAdj, t, words);

                    int[] bestLocal = ctx.BestLocal;
                    int bestLocalSize = solver.Solve(lower, bestLocal, sw, TimeLimitSeconds);

                    if (bestLocalSize + 1 > curBest)
                    {
                        // Build the global clique: v + mapped vertices
                        int newSize = bestLocalSize + 1;
                        int[] candidateClique = ctx.TempClique;
                        candidateClique[0] = v;
                        for (int i = 0; i < bestLocalSize; i++)
                            candidateClique[i + 1] = cand[bestLocal[i]];

                        lock (bestLock)
                        {
                            if (newSize > globalBestSize)
                            {
                                globalBestSize = newSize;
                                if (globalBestClique.Length < newSize)
                                    globalBestClique = new int[Math.Max(newSize, globalBestClique.Length * 2)];
                                Array.Copy(candidateClique, globalBestClique, newSize);

                                if (globalBestSize >= upperBound) cts.Cancel();
                            }
                        }
                    }

                    ctx.ReturnUlong(localAdj);
                    ctx.ReturnInt(cand);
                }
                return ctx;
            },
            _ => { });

        // Output best found (exact if it finished; heuristic if time canceled)
        Console.WriteLine(globalBestSize);
        for (int i = 0; i < globalBestSize; i++)
        {
            if (i != 0) Console.Write(' ');
            Console.Write(globalBestClique[i]);
        }
        Console.WriteLine();
    }

    // -------------------- CSR Greedy --------------------

    sealed class GreedyWorkBuffers
    {
        public int[] Clique = new int[4096];
        public int[] BufA = new int[0];
        public int[] BufB = new int[0];
    }

    static int GreedyCliqueCSR(int seed, int[] off, int[] adj, int[] deg, int[] cliqueOut, GreedyWorkBuffers buf)
    {
        int size = 0;
        cliqueOut[size++] = seed;

        int[] cand = RentInt(ref buf.BufA, off[seed + 1] - off[seed]);
        int candCount = off[seed + 1] - off[seed];
        Array.Copy(adj, off[seed], cand, 0, candCount);

        // Iteratively pick a candidate with highest degree, intersect candidates with its neighbors
        int[] next = RentInt(ref buf.BufB, candCount);

        while (candCount > 0)
        {
            // pick best candidate by degree
            int bestIdx = 0;
            int bestV = cand[0];
            int bestD = deg[bestV];
            for (int i = 1; i < candCount; i++)
            {
                int v = cand[i];
                int d = deg[v];
                if (d > bestD)
                {
                    bestD = d;
                    bestV = v;
                    bestIdx = i;
                }
            }

            if (size >= cliqueOut.Length) Array.Resize(ref cliqueOut, cliqueOut.Length * 2);
            cliqueOut[size++] = bestV;

            // Intersect cand with N(bestV)
            int aCount = candCount;
            // (We can keep cand unsorted; intersection wants sorted lists -> but adjacency is sorted,
            // so we sort cand once per iteration if needed. To keep it cheap, sort every time.
            // For typical sparse graphs, candCount is small.)
            Array.Sort(cand, 0, aCount);

            int nbStart = off[bestV], nbEnd = off[bestV + 1];
            int iA = 0, iB = nbStart;
            int outCount = 0;
            while (iA < aCount && iB < nbEnd)
            {
                int va = cand[iA];
                int vb = adj[iB];
                if (va == vb)
                {
                    if (va != bestV) next[outCount++] = va;
                    iA++; iB++;
                }
                else if (va < vb) iA++;
                else iB++;
            }

            // swap cand/next
            var tmp = cand; cand = next; next = tmp;
            candCount = outCount;
        }

        return size;
    }

    static int GreedyCliqueFromCandidates(int baseV, int[] cand, int candCount, int[] off, int[] adj, int[] deg, int[] cliqueOut, GreedyWorkBuffers buf)
    {
        int size = 0;
        cliqueOut[size++] = baseV;
        if (candCount <= 0) return size;

        int[] cur = RentInt(ref buf.BufA, candCount);
        Array.Copy(cand, 0, cur, 0, candCount);
        int curCount = candCount;

        int[] next = RentInt(ref buf.BufB, candCount);

        while (curCount > 0)
        {
            int bestIdx = 0;
            int bestV = cur[0];
            int bestD = deg[bestV];
            for (int i = 1; i < curCount; i++)
            {
                int v = cur[i];
                int d = deg[v];
                if (d > bestD)
                {
                    bestD = d;
                    bestV = v;
                    bestIdx = i;
                }
            }

            cliqueOut[size++] = bestV;

            Array.Sort(cur, 0, curCount);
            int nbStart = off[bestV], nbEnd = off[bestV + 1];

            int iA = 0, iB = nbStart, outCount = 0;
            while (iA < curCount && iB < nbEnd)
            {
                int va = cur[iA];
                int vb = adj[iB];
                if (va == vb)
                {
                    if (va != bestV) next[outCount++] = va;
                    iA++; iB++;
                }
                else if (va < vb) iA++;
                else iB++;
            }

            var tmp = cur; cur = next; next = tmp;
            curCount = outCount;
        }
        return size;
    }

    static int[] RentInt(ref int[] arr, int needed)
    {
        if (arr.Length < needed) arr = new int[needed];
        return arr;
    }

    // -------------------- Core Decomposition (degeneracy) --------------------

    static int CoreDecomposition(int n, int[] off, int[] adj, int[] degInput, int[] core, int[] order)
    {
        int[] deg = new int[n];
        int maxDeg = 0;
        for (int i = 0; i < n; i++)
        {
            int d = degInput[i];
            deg[i] = d;
            if (d > maxDeg) maxDeg = d;
        }

        int[] bin = new int[maxDeg + 1];
        for (int i = 0; i < n; i++) bin[deg[i]]++;

        int start = 0;
        for (int d = 0; d <= maxDeg; d++)
        {
            int num = bin[d];
            bin[d] = start;
            start += num;
        }

        int[] vert = new int[n];
        int[] pos = new int[n];
        for (int v = 0; v < n; v++)
        {
            pos[v] = bin[deg[v]];
            vert[pos[v]] = v;
            bin[deg[v]]++;
        }

        for (int d = maxDeg; d >= 1; d--) bin[d] = bin[d - 1];
        bin[0] = 0;

        int maxCore = 0;

        for (int i = 0; i < n; i++)
        {
            int v = vert[i];
            order[i] = v;

            int dv = deg[v];
            core[v] = dv;
            if (dv > maxCore) maxCore = dv;

            int sv = off[v], ev = off[v + 1];
            for (int p = sv; p < ev; p++)
            {
                int u = adj[p];
                if (deg[u] > dv)
                {
                    int du = deg[u];
                    int pu = pos[u];
                    int pw = bin[du];
                    int w = vert[pw];

                    if (u != w)
                    {
                        vert[pu] = w; pos[w] = pu;
                        vert[pw] = u; pos[u] = pw;
                    }
                    bin[du]++;
                    deg[u]--;
                }
            }
        }

        return maxCore;
    }

    // -------------------- Bitset greedy for lower bound (global bitset case) --------------------

    static int GreedyCliqueBitset(ulong[] adj, int n, int words, int[] cliqueOut)
    {
        // Pick starting vertex with maximum degree
        int bestV = 0;
        int bestD = -1;
        for (int v = 0; v < n; v++)
        {
            int d = 0;
            int baseRow = v * words;
            for (int i = 0; i < words; i++) d += BitOperations.PopCount(adj[baseRow + i]);
            if (d > bestD) { bestD = d; bestV = v; }
        }

        int size = 0;
        cliqueOut[size++] = bestV;

        ulong[] cand = ArrayPool<ulong>.Shared.Rent(words);
        ulong[] next = ArrayPool<ulong>.Shared.Rent(words);
        try
        {
            // cand = N(bestV)
            Buffer.BlockCopy(adj, bestV * words * sizeof(ulong), cand, 0, words * sizeof(ulong));

            while (!IsEmpty(cand, words))
            {
                // pick a vertex from cand (LSB)
                int v = GetAndClearLSB(cand, words);
                cliqueOut[size++] = v;

                // next = cand & N(v)
                int row = v * words;
                for (int i = 0; i < words; i++) next[i] = cand[i] & adj[row + i];

                // swap
                var tmp = cand; cand = next; next = tmp;
            }
        }
        finally
        {
            ArrayPool<ulong>.Shared.Return(cand, clearArray: true);
            ArrayPool<ulong>.Shared.Return(next, clearArray: true);
        }
        return size;
    }

    static bool IsEmpty(ulong[] bs, int words)
    {
        for (int i = 0; i < words; i++) if (bs[i] != 0) return false;
        return true;
    }

    static int GetAndClearLSB(ulong[] bs, int words)
    {
        for (int wi = 0; wi < words; wi++)
        {
            ulong x = bs[wi];
            if (x != 0)
            {
                int b = BitOperations.TrailingZeroCount(x);
                bs[wi] = x & (x - 1);
                return (wi << 6) + b;
            }
        }
        return -1;
    }

    // -------------------- Local search context for CSR exact search --------------------

    sealed class LocalSearchContext
    {
        public readonly LocalIntMap Map = new LocalIntMap();
        public readonly BitsetMaxCliqueSolver Solver = new BitsetMaxCliqueSolver();
        public readonly GreedyWorkBuffers GreedyBuf = new GreedyWorkBuffers();

        private readonly ArrayPool<int> _ip = ArrayPool<int>.Shared;
        private readonly ArrayPool<ulong> _up = ArrayPool<ulong>.Shared;

        public int[] TempClique = new int[Math.Max(1024, LocalBitsetLimit + 1)];
        public int[] BestLocal = new int[Math.Max(1024, LocalBitsetLimit)];

        public int[] RentInt(int len) => _ip.Rent(len);
        public void ReturnInt(int[] arr) => _ip.Return(arr, clearArray: false);

        public ulong[] RentUlong(int len) => _up.Rent(len);
        public void ReturnUlong(ulong[] arr) => _up.Return(arr, clearArray: false);
    }

    sealed class LocalIntMap
    {
        private int[] _keys = Array.Empty<int>();
        private int[] _vals = Array.Empty<int>();
        private int _mask;

        public void Build(int[] verts, int count)
        {
            int cap = 1;
            while (cap < (count * 2)) cap <<= 1;
            if (_keys.Length < cap)
            {
                _keys = new int[cap];
                _vals = new int[cap];
            }
            Array.Fill(_keys, -1, 0, cap);
            _mask = cap - 1;

            for (int i = 0; i < count; i++)
                Put(verts[i], i);
        }

        private void Put(int k, int v)
        {
            int idx = Mix(k) & _mask;
            while (_keys[idx] != -1) idx = (idx + 1) & _mask;
            _keys[idx] = k;
            _vals[idx] = v;
        }

        public int Get(int k)
        {
            int idx = Mix(k) & _mask;
            while (true)
            {
                int kk = _keys[idx];
                if (kk == k) return _vals[idx];
                if (kk == -1) return -1;
                idx = (idx + 1) & _mask;
            }
        }

        static int Mix(int x)
        {
            unchecked
            {
                uint z = (uint)x;
                z ^= z >> 16;
                z *= 0x7feb352dU;
                z ^= z >> 15;
                z *= 0x846ca68bU;
                z ^= z >> 16;
                return (int)z;
            }
        }
    }

    // -------------------- Bitset Maximum Clique Solver (iterative Tomita-style) --------------------

    sealed class BitsetMaxCliqueSolver
    {
        private ulong[] _adj = Array.Empty<ulong>(); // flat adjacency: row-major [v*words + w]
        private int _n;
        private int _words;

        public BitsetMaxCliqueSolver() { }
        public BitsetMaxCliqueSolver(ulong[] adj, int n, int words) { Reset(adj, n, words); }

        public void Reset(ulong[] adj, int n, int words)
        {
            _adj = adj;
            _n = n;
            _words = words;
        }

        // Returns bestSize (>= lowerBound), fills bestCliqueOut[0..bestSize-1] with vertex indices (local).
        public int Solve(int lowerBound, int[] bestCliqueOut, Stopwatch sw, int timeLimitSeconds)
        {
            int bestSize = Math.Max(0, lowerBound);

            // Current clique along the DFS path
            int[] clique = ArrayPool<int>.Shared.Rent(_n);
            int cliqueDepth = 0;

            // Root P = all vertices
            ulong[] rootP = ArrayPool<ulong>.Shared.Rent(_words);
            Array.Clear(rootP, 0, _words);
            for (int v = 0; v < _n; v++)
                rootP[v >> 6] |= 1UL << (v & 63);

            // If P itself is a clique, finish immediately (prevents deep recursion on complete graphs)
            int rootCount = PopCount(rootP);
            if (rootCount + 0 > bestSize && rootCount > 1024 && IsClique(rootP))
            {
                bestSize = rootCount;
                int outIdx = 0;
                for (int v = 0; v < _n; v++)
                {
                    if (((rootP[v >> 6] >> (v & 63)) & 1UL) != 0)
                        bestCliqueOut[outIdx++] = v;
                }
                ArrayPool<ulong>.Shared.Return(rootP, clearArray: true);
                ArrayPool<int>.Shared.Return(clique, clearArray: false);
                return bestSize;
            }

            var stack = new List<Frame>(64);
            PushFrame(stack, rootP, 0, ref bestSize);

            while (stack.Count > 0)
            {
                if (sw.Elapsed.TotalSeconds >= timeLimitSeconds)
                    break;

                Frame f = stack[stack.Count - 1];

                if (f.Idx < 0)
                {
                    // pop
                    stack.RemoveAt(stack.Count - 1);
                    ArrayPool<ulong>.Shared.Return(f.P, clearArray: true);
                    ArrayPool<int>.Shared.Return(f.Order, clearArray: false);
                    ArrayPool<int>.Shared.Return(f.Bounds, clearArray: false);
                    cliqueDepth = f.Depth; // restore clique depth to frame's depth
                    continue;
                }

                // Prune by coloring bound
                if (f.Depth + f.Bounds[f.Idx] <= bestSize)
                {
                    // pop whole frame
                    stack.RemoveAt(stack.Count - 1);
                    ArrayPool<ulong>.Shared.Return(f.P, clearArray: true);
                    ArrayPool<int>.Shared.Return(f.Order, clearArray: false);
                    ArrayPool<int>.Shared.Return(f.Bounds, clearArray: false);
                    cliqueDepth = f.Depth;
                    continue;
                }

                // Choose next vertex to branch on
                int v = f.Order[f.Idx];
                f.Idx--;
                // Update stored frame
                stack[stack.Count - 1] = f;

                // Include v in clique
                clique[f.Depth] = v;
                cliqueDepth = f.Depth + 1;

                // Compute child candidate set: Pchild = P & N(v), using current frame's P
                ClearBit(f.P, v);
                ulong[] childP = ArrayPool<ulong>.Shared.Rent(_words);
                int row = v * _words;
                for (int i = 0; i < _words; i++) childP[i] = f.P[i] & _adj[row + i];

                int childCount = PopCount(childP);

                if (childCount == 0)
                {
                    // Found a maximal clique of size cliqueDepth
                    if (cliqueDepth > bestSize)
                    {
                        bestSize = cliqueDepth;
                        for (int i = 0; i < bestSize; i++) bestCliqueOut[i] = clique[i];
                    }
                    ArrayPool<ulong>.Shared.Return(childP, clearArray: true);
                    continue;
                }

                // Upper bound by size
                if (cliqueDepth + childCount <= bestSize)
                {
                    ArrayPool<ulong>.Shared.Return(childP, clearArray: true);
                    continue;
                }

                // Complete-subgraph shortcut to avoid huge depths on dense graphs
                if (childCount > 1024 && IsClique(childP))
                {
                    int total = cliqueDepth + childCount;
                    if (total > bestSize)
                    {
                        bestSize = total;
                        int outPos = 0;
                        for (int i = 0; i < cliqueDepth; i++) bestCliqueOut[outPos++] = clique[i];
                        // append child vertices
                        for (int u = 0; u < _n; u++)
                        {
                            if (((childP[u >> 6] >> (u & 63)) & 1UL) != 0)
                                bestCliqueOut[outPos++] = u;
                        }
                    }
                    ArrayPool<ulong>.Shared.Return(childP, clearArray: true);
                    continue;
                }

                // Descend
                PushFrame(stack, childP, cliqueDepth, ref bestSize);
            }

            // Clean up remaining frames
            for (int i = stack.Count - 1; i >= 0; i--)
            {
                var f = stack[i];
                ArrayPool<ulong>.Shared.Return(f.P, clearArray: true);
                ArrayPool<int>.Shared.Return(f.Order, clearArray: false);
                ArrayPool<int>.Shared.Return(f.Bounds, clearArray: false);
            }

            ArrayPool<int>.Shared.Return(clique, clearArray: false);
            return bestSize;
        }

        private struct Frame
        {
            public ulong[] P;
            public int Depth;
            public int[] Order;
            public int[] Bounds;
            public int Idx;
        }

        private void PushFrame(List<Frame> stack, ulong[] P, int depth, ref int bestSize)
        {
            // ColorSort(P) -> order, bounds. bounds[i] = color number (1..k) for order[i].
            int pCount = PopCount(P);

            int[] order = ArrayPool<int>.Shared.Rent(pCount);
            int[] bounds = ArrayPool<int>.Shared.Rent(pCount);

            ColorSort(P, order, bounds, pCount);

            // Small optimization: if depth + bounds[last] <= bestSize, prune immediately
            int last = pCount - 1;
            if (last >= 0 && depth + bounds[last] <= bestSize)
            {
                ArrayPool<ulong>.Shared.Return(P, clearArray: true);
                ArrayPool<int>.Shared.Return(order, clearArray: false);
                ArrayPool<int>.Shared.Return(bounds, clearArray: false);
                return;
            }

            stack.Add(new Frame { P = P, Depth = depth, Order = order, Bounds = bounds, Idx = pCount - 1 });
        }

        private int PopCount(ulong[] bs)
        {
            int c = 0;
            for (int i = 0; i < _words; i++) c += BitOperations.PopCount(bs[i]);
            return c;
        }

        private void ClearBit(ulong[] bs, int v)
        {
            int wi = v >> 6;
            bs[wi] &= ~(1UL << (v & 63));
        }

        private bool IsClique(ulong[] set)
        {
            // Check if induced subgraph on 'set' is complete:
            // for each v in set: (set \ {v}) subset of N(v)
            ulong[] tmp = ArrayPool<ulong>.Shared.Rent(_words);
            try
            {
                Buffer.BlockCopy(set, 0, tmp, 0, _words * sizeof(ulong));

                while (true)
                {
                    int v = FindAny(tmp);
                    if (v < 0) return true;
                    ClearBit(tmp, v);

                    int row = v * _words;
                    for (int i = 0; i < _words; i++)
                    {
                        ulong missing = tmp[i] & ~_adj[row + i];
                        if (missing != 0) return false;
                    }
                }
            }
            finally
            {
                ArrayPool<ulong>.Shared.Return(tmp, clearArray: true);
            }
        }

        private int FindAny(ulong[] bs)
        {
            for (int wi = 0; wi < _words; wi++)
            {
                ulong x = bs[wi];
                if (x != 0)
                {
                    int b = BitOperations.TrailingZeroCount(x);
                    return (wi << 6) + b;
                }
            }
            return -1;
        }

        private void ColorSort(ulong[] P, int[] order, int[] bounds, int pCount)
        {
            // Greedy coloring by repeatedly extracting maximal independent sets.
            // Writes vertices in nondecreasing color classes:
            // color 1 vertices first, then color 2, etc.
            ulong[] remaining = ArrayPool<ulong>.Shared.Rent(_words);
            ulong[] Q = ArrayPool<ulong>.Shared.Rent(_words);
            try
            {
                Buffer.BlockCopy(P, 0, remaining, 0, _words * sizeof(ulong));
                int outPos = 0;
                int color = 0;

                while (!IsEmptyLocal(remaining))
                {
                    color++;
                    Buffer.BlockCopy(remaining, 0, Q, 0, _words * sizeof(ulong));

                    while (true)
                    {
                        int v = GetAndClearAny(Q);
                        if (v < 0) break;

                        order[outPos] = v;
                        bounds[outPos] = color;
                        outPos++;

                        // remove v from remaining
                        ClearBit(remaining, v);

                        // Q = Q \ N(v)
                        int row = v * _words;
                        for (int i = 0; i < _words; i++)
                            Q[i] &= ~_adj[row + i];
                    }
                }

                // outPos should equal pCount, but duplicates/oddities could reduce it; keep it safe.
                // If smaller, shrink by shifting remaining entries (rare; mostly due to malformed input).
                // We won't attempt to fix further; solver remains safe but might be weaker.
                // (In valid graphs, outPos == pCount.)
            }
            finally
            {
                ArrayPool<ulong>.Shared.Return(remaining, clearArray: true);
                ArrayPool<ulong>.Shared.Return(Q, clearArray: true);
            }

            bool IsEmptyLocal(ulong[] bs)
            {
                for (int i = 0; i < _words; i++) if (bs[i] != 0) return false;
                return true;
            }

            int GetAndClearAny(ulong[] bs)
            {
                for (int wi = 0; wi < _words; wi++)
                {
                    ulong x = bs[wi];
                    if (x != 0)
                    {
                        int b = BitOperations.TrailingZeroCount(x);
                        bs[wi] = x & (x - 1);
                        return (wi << 6) + b;
                    }
                }
                return -1;
            }
        }
    }

    // -------------------- Fast Scanner --------------------

    sealed class FastScanner
    {
        private readonly System.IO.Stream _s;
        private readonly byte[] _buf;
        private int _len, _ptr;

        public FastScanner(System.IO.Stream s, int bufSize = 1 << 20)
        {
            _s = s;
            _buf = new byte[bufSize];
        }

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do { c = Read(); } while (c <= 32 && c != 0);

            int sign = 1;
            if (c == (byte)'-')
            {
                sign = -1;
                c = Read();
            }

            int x = 0;
            while (c > 32)
            {
                x = x * 10 + (c - (byte)'0');
                c = Read();
            }
            return x * sign;
        }
    }
