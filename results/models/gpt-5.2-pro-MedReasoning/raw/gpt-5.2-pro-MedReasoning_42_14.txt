```spirv
; SPIR-V
; Version: 1.0
; Generator: hand-written
; Bound: 200
OpCapability Shader
OpCapability Geometry
%ext = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor %ubo
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpName %ubo "ubo"

; -------- Types / Constants --------
%void = OpTypeVoid
%fn_void = OpTypeFunction %void

%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%uint_6 = OpConstant %uint 6

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_0_33333334 = OpConstant %float 0.33333334

%L_un = OpConstantComposite %v3float %float_1 %float_1 %float_1

; -------- Interface blocks --------
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
OpDecorate %arr_pv_in ArrayStride 16

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_6
OpDecorate %arr_v3_in ArrayStride 16

%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float

%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal   = OpVariable %ptr_in_arr_v3 Input
%color    = OpVariable %ptr_in_arr_v3 Input

OpDecorate %worldPos Location 0
OpDecorate %normal   Location 1
OpDecorate %color    Location 2

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; -------- UBO (set=0,binding=0) --------
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%ptr_ubo = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_ubo Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; -------- Main --------
%main = OpFunction %void None %fn_void
%entry = OpLabel

; p0 = worldPos[0], p2 = worldPos[2], p4 = worldPos[4]
%wp0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p0 = OpLoad %v3float %wp0_ptr
%wp2_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_2
%p2 = OpLoad %v3float %wp2_ptr
%wp4_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_4
%p4 = OpLoad %v3float %wp4_ptr

%a = OpFSub %v3float %p2 %p0
%b = OpFSub %v3float %p4 %p0

; cross(a,b) manually
%ax = OpCompositeExtract %float %a 0
%ay = OpCompositeExtract %float %a 1
%az = OpCompositeExtract %float %a 2
%bx = OpCompositeExtract %float %b 0
%by = OpCompositeExtract %float %b 1
%bz = OpCompositeExtract %float %b 2

%ay_bz = OpFMul %float %ay %bz
%az_by = OpFMul %float %az %by
%cx = OpFSub %float %ay_bz %az_by

%az_bx = OpFMul %float %az %bx
%ax_bz = OpFMul %float %ax %bz
%cy = OpFSub %float %az_bx %ax_bz

%ax_by = OpFMul %float %ax %by
%ay_bx = OpFMul %float %ay %bx
%cz = OpFSub %float %ax_by %ay_bx

%cross = OpCompositeConstruct %v3float %cx %cy %cz
%N = OpExtInst %v3float %ext Normalize %cross

%L = OpExtInst %v3float %ext Normalize %L_un
%ndotl = OpDot %float %N %L
%diffuse = OpExtInst %float %ext FMax %ndotl %float_0

; base = (color0 + color2 + color4) / 3
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %c2_ptr
%c4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
%c4 = OpLoad %v3float %c4_ptr

%sum02 = OpFAdd %v3float %c0 %c2
%sum024 = OpFAdd %v3float %sum02 %c4
%base = OpVectorTimesScalar %v3float %sum024 %float_0_33333334

; final = base * (0.2 + 0.8 * diffuse)
%diff_08 = OpFMul %float %float_0_8 %diffuse
%ambient_plus = OpFAdd %float %float_0_2 %diff_08
%finalColor = OpVectorTimesScalar %v3float %base %ambient_plus

; Emit vertex 0
%inpos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%inpos0 = OpLoad %v4float %inpos0_ptr
%outpos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outpos_ptr %inpos0
OpStore %outColor %finalColor
OpEmitVertex

; Emit vertex 2
%inpos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%inpos2 = OpLoad %v4float %inpos2_ptr
OpStore %outpos_ptr %inpos2
OpStore %outColor %finalColor
OpEmitVertex

; Emit vertex 4
%inpos4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%inpos4 = OpLoad %v4float %inpos4_ptr
OpStore %outpos_ptr %inpos4
OpStore %outColor %finalColor
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
