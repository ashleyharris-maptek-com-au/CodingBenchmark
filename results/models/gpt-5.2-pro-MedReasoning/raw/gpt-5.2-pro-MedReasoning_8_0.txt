use std::io::{self, BufRead, Write};

#[derive(Clone)]
struct Bitset {
    data: Vec<u64>,
}
impl Bitset {
    fn new(n_bits: usize) -> Self {
        let n_words = (n_bits + 63) / 64;
        Self { data: vec![0u64; n_words] }
    }
    #[inline]
    fn get_u32(&self, i: u32) -> bool {
        let i = i as usize;
        let w = i >> 6;
        let b = i & 63;
        ((self.data[w] >> b) & 1) != 0
    }
    #[inline]
    fn set_u32(&mut self, i: u32) {
        let i = i as usize;
        let w = i >> 6;
        let b = i & 63;
        self.data[w] |= 1u64 << b;
    }
}

#[derive(Clone)]
struct Packed2 {
    // 2 bits per entry, packed into u64
    data: Vec<u64>,
}
impl Packed2 {
    fn new(n_entries: usize) -> Self {
        let n_bits = n_entries.checked_mul(2).unwrap();
        let n_words = (n_bits + 63) / 64;
        Self { data: vec![0u64; n_words] }
    }
    #[inline]
    fn set_u32(&mut self, i: u32, val: u8) {
        debug_assert!(val < 4);
        let bit = (i as usize) * 2;
        let w = bit >> 6;
        let s = bit & 63;

        if s <= 62 {
            let mask = 0b11u64 << s;
            self.data[w] = (self.data[w] & !mask) | ((val as u64) << s);
        } else {
            // s == 63, crosses word boundary
            // bit at 63 in word w, and bit at 0 in word w+1
            let v0 = (val as u64) & 1;
            let v1 = ((val as u64) >> 1) & 1;
            self.data[w] = (self.data[w] & !(1u64 << 63)) | (v0 << 63);
            self.data[w + 1] = (self.data[w + 1] & !1u64) | v1;
        }
    }
    #[inline]
    fn get_u32(&self, i: u32) -> u8 {
        let bit = (i as usize) * 2;
        let w = bit >> 6;
        let s = bit & 63;

        if s <= 62 {
            ((self.data[w] >> s) & 0b11) as u8
        } else {
            // s == 63
            let b0 = (self.data[w] >> 63) & 1;
            let b1 = self.data[w + 1] & 1;
            (b0 | (b1 << 1)) as u8
        }
    }
}

#[derive(Copy, Clone)]
struct Node {
    idx: u32, // y*w + x
    x: u32,   // cached x to avoid div/mod in BFS
}

// Directions encoded as: 0=Up, 1=Right, 2=Down, 3=Left
#[inline]
fn step_idx(idx: u32, dir: u8, w: u32) -> u32 {
    match dir {
        0 => idx - w,
        1 => idx + 1,
        2 => idx + w,
        3 => idx - 1,
        _ => unreachable!(),
    }
}

#[inline]
fn expand_frontier(
    frontier: &[Node],
    next: &mut Vec<Node>,
    w: u32,
    n_cells: u32,
    walls: &Bitset,
    visited_this: &mut Bitset,
    parent_this: &mut Packed2,
    visited_other: &Bitset,
) -> Option<u32> {
    next.clear();

    for &node in frontier {
        let idx = node.idx;

        // If the other side already visited this cell, we met.
        if visited_other.get_u32(idx) {
            return Some(idx);
        }

        // Up neighbor
        if idx >= w {
            let nidx = idx - w;
            if !walls.get_u32(nidx) && !visited_this.get_u32(nidx) {
                visited_this.set_u32(nidx);
                parent_this.set_u32(nidx, 2); // from nidx, go Down to return to idx
                if visited_other.get_u32(nidx) {
                    return Some(nidx);
                }
                next.push(Node { idx: nidx, x: node.x });
            }
        }
        // Down neighbor
        {
            let down = idx as u64 + w as u64;
            if down < n_cells as u64 {
                let nidx = down as u32;
                if !walls.get_u32(nidx) && !visited_this.get_u32(nidx) {
                    visited_this.set_u32(nidx);
                    parent_this.set_u32(nidx, 0); // from nidx, go Up to return to idx
                    if visited_other.get_u32(nidx) {
                        return Some(nidx);
                    }
                    next.push(Node { idx: nidx, x: node.x });
                }
            }
        }
        // Left neighbor
        if node.x > 0 {
            let nidx = idx - 1;
            if !walls.get_u32(nidx) && !visited_this.get_u32(nidx) {
                visited_this.set_u32(nidx);
                parent_this.set_u32(nidx, 1); // from nidx, go Right to return to idx
                if visited_other.get_u32(nidx) {
                    return Some(nidx);
                }
                next.push(Node { idx: nidx, x: node.x - 1 });
            }
        }
        // Right neighbor
        if node.x + 1 < w {
            let nidx = idx + 1;
            if !walls.get_u32(nidx) && !visited_this.get_u32(nidx) {
                visited_this.set_u32(nidx);
                parent_this.set_u32(nidx, 3); // from nidx, go Left to return to idx
                if visited_other.get_u32(nidx) {
                    return Some(nidx);
                }
                next.push(Node { idx: nidx, x: node.x + 1 });
            }
        }
    }

    None
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut r = stdin.lock();

    // Read first line: H W
    let mut first = String::new();
    if r.read_line(&mut first)? == 0 {
        return Ok(());
    }
    let mut it = first.split_whitespace();
    let h: u32 = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return Ok(()),
    };
    let w: u32 = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return Ok(()),
    };

    let n64 = (h as u64) * (w as u64);
    if n64 == 0 || n64 > (u32::MAX as u64) {
        // Either empty grid or too large for this implementation's u32 indexing.
        // (Given constraints, this shouldn't happen.)
        let mut out = io::BufWriter::new(io::stdout().lock());
        writeln!(out, "0")?;
        return Ok(());
    }
    let n_cells: u32 = n64 as u32;

    // Build wall bitset while reading rows; find A and B.
    let mut walls = Bitset::new(n_cells as usize);
    let mut start: Option<(u32, u32, u32)> = None; // (idx, x, y)
    let mut goal: Option<(u32, u32, u32)> = None;

    let mut line = Vec::<u8>::with_capacity((w as usize) + 2);

    for y in 0..h {
        line.clear();
        // Read a full row (including spaces). Uses '\n' as delimiter.
        let _ = r.read_until(b'\n', &mut line)?;
        if line.last() == Some(&b'\n') {
            line.pop();
            if line.last() == Some(&b'\r') {
                line.pop();
            }
        }

        // If the last line might not end with \n, read_until still returns it as-is.
        if line.len() != w as usize {
            // Input is malformed (or missing trailing spaces). Best-effort: pad short lines.
            if line.len() < w as usize {
                line.resize(w as usize, b' ');
            } else {
                line.truncate(w as usize);
            }
        }

        let base = (y as u64) * (w as u64);
        for x in 0..w {
            let c = line[x as usize];
            let idx = (base + x as u64) as u32;
            match c {
                b'#' => walls.set_u32(idx),
                b'A' => start = Some((idx, x, y)),
                b'B' => goal = Some((idx, x, y)),
                _ => {}
            }
        }
    }

    let (s_idx, s_x, _s_y) = match start {
        Some(v) => v,
        None => {
            let mut out = io::BufWriter::new(io::stdout().lock());
            writeln!(out, "0")?;
            return Ok(());
        }
    };
    let (t_idx, t_x, _t_y) = match goal {
        Some(v) => v,
        None => {
            let mut out = io::BufWriter::new(io::stdout().lock());
            writeln!(out, "0")?;
            return Ok(());
        }
    };

    if s_idx == t_idx {
        let mut out = io::BufWriter::new(io::stdout().lock());
        writeln!(out, "1")?;
        // output x y
        let y = s_idx / w;
        let x = s_idx - y * w;
        writeln!(out, "{} {}", x, y)?;
        return Ok(());
    }

    // Bidirectional BFS with:
    // - walls: 1 bit/cell
    // - visited per side: 1 bit/cell
    // - parent direction per side: 2 bits/cell (only meaningful when visited)
    let mut visited_s = Bitset::new(n_cells as usize);
    let mut visited_t = Bitset::new(n_cells as usize);
    let mut parent_s = Packed2::new(n_cells as usize);
    let mut parent_t = Packed2::new(n_cells as usize);

    let mut front_s: Vec<Node> = Vec::new();
    let mut front_t: Vec<Node> = Vec::new();
    let mut next_s: Vec<Node> = Vec::new();
    let mut next_t: Vec<Node> = Vec::new();

    visited_s.set_u32(s_idx);
    visited_t.set_u32(t_idx);
    front_s.push(Node { idx: s_idx, x: s_x });
    front_t.push(Node { idx: t_idx, x: t_x });

    let mut meet: Option<u32> = None;

    while !front_s.is_empty() && !front_t.is_empty() {
        if front_s.len() <= front_t.len() {
            if let Some(m) = expand_frontier(
                &front_s,
                &mut next_s,
                w,
                n_cells,
                &walls,
                &mut visited_s,
                &mut parent_s,
                &visited_t,
            ) {
                meet = Some(m);
                break;
            }
            front_s.clear();
            std::mem::swap(&mut front_s, &mut next_s);
        } else {
            if let Some(m) = expand_frontier(
                &front_t,
                &mut next_t,
                w,
                n_cells,
                &walls,
                &mut visited_t,
                &mut parent_t,
                &visited_s,
            ) {
                meet = Some(m);
                break;
            }
            front_t.clear();
            std::mem::swap(&mut front_t, &mut next_t);
        }
    }

    let meet = match meet {
        Some(m) => m,
        None => {
            let mut out = io::BufWriter::new(io::stdout().lock());
            writeln!(out, "0")?;
            return Ok(());
        }
    };

    // Reconstruct:
    // start -> ... -> meet  (using parent_s, walking backwards to start)
    // meet  -> ... -> goal  (using parent_t, walking backwards to goal)
    //
    // parent_* stores the direction to move from a cell to its parent in the BFS tree.
    let mut left_rev: Vec<u32> = Vec::new(); // meet -> ... -> start
    {
        let mut cur = meet;
        left_rev.push(cur);
        while cur != s_idx {
            let dir = parent_s.get_u32(cur);
            cur = step_idx(cur, dir, w);
            left_rev.push(cur);
        }
    }
    left_rev.reverse(); // start -> ... -> meet

    let mut right: Vec<u32> = Vec::new(); // (excluding meet) ... -> goal
    {
        let mut cur = meet;
        while cur != t_idx {
            let dir = parent_t.get_u32(cur);
            cur = step_idx(cur, dir, w);
            right.push(cur);
        }
    }

    let mut path: Vec<u32> = left_rev;
    path.extend(right);

    // Output
    let mut out = io::BufWriter::new(io::stdout().lock());
    writeln!(out, "{}", path.len())?;
    for idx in path {
        let y = idx / w;
        let x = idx - y * w;
        writeln!(out, "{} {}", x, y)?;
    }

    Ok(())
}
