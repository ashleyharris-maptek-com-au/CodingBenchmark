#include <iostream>
#include <vector>
#include <algorithm>
#include <map>
#include <cstdint>
#include <limits>

using std::cin;
using std::cout;
using std::vector;
using std::multimap;

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0; // 0 = as given, 1 = swapped
};

struct InputRect {
    int id;
    int w, h;
};

static inline bool canFitEither(int bw, int bh, int w, int h) {
    return (w <= bw && h <= bh) || (h <= bw && w <= bh);
}

// ------------------------
// Small/Medium: MaxRects
// ------------------------
namespace maxrects {

struct Rect {
    int x, y, w, h;
};

static inline bool intersects(const Rect& a, const Rect& b) {
    return !(a.x >= b.x + b.w || a.x + a.w <= b.x || a.y >= b.y + b.h || a.y + a.h <= b.y);
}

static inline bool contains(const Rect& outer, const Rect& inner) {
    return inner.x >= outer.x && inner.y >= outer.y
        && inner.x + inner.w <= outer.x + outer.w
        && inner.y + inner.h <= outer.y + outer.h;
}

struct Candidate {
    bool ok = false;
    int x = 0, y = 0;
    int w = 0, h = 0;
    int rotated = 0;
    int score1 = 0; // short side fit
    int score2 = 0; // long side fit
};

struct Bin {
    int W, H;
    vector<Rect> freeRects;

    explicit Bin(int W_, int H_) : W(W_), H(H_) {
        freeRects.push_back(Rect{0, 0, W, H});
    }

    Candidate findBest(int iw, int ih, int origW, int origH) const {
        Candidate best;
        best.ok = false;
        best.score1 = std::numeric_limits<int>::max();
        best.score2 = std::numeric_limits<int>::max();

        auto consider = [&](int w, int h) {
            for (const auto& fr : freeRects) {
                if (w <= fr.w && h <= fr.h) {
                    int leftoverH = fr.w - w;
                    int leftoverV = fr.h - h;
                    int shortFit = std::min(leftoverH, leftoverV);
                    int longFit  = std::max(leftoverH, leftoverV);

                    if (shortFit < best.score1 || (shortFit == best.score1 && longFit < best.score2)) {
                        best.ok = true;
                        best.x = fr.x;
                        best.y = fr.y;
                        best.w = w;
                        best.h = h;
                        best.score1 = shortFit;
                        best.score2 = longFit;

                        if (origW == origH) best.rotated = 0;
                        else best.rotated = (w == origW && h == origH) ? 0 : 1;
                    }
                }
            }
        };

        consider(iw, ih);
        if (iw != ih) consider(ih, iw);

        return best;
    }

    void placeRect(const Rect& used) {
        // Split all free rects that intersect.
        vector<Rect> newFree;
        newFree.reserve(freeRects.size() + 4);

        for (const auto& fr : freeRects) {
            if (!intersects(used, fr)) {
                newFree.push_back(fr);
                continue;
            }

            // Left
            if (used.x > fr.x) {
                newFree.push_back(Rect{fr.x, fr.y, used.x - fr.x, fr.h});
            }
            // Right
            if (used.x + used.w < fr.x + fr.w) {
                newFree.push_back(Rect{used.x + used.w, fr.y, (fr.x + fr.w) - (used.x + used.w), fr.h});
            }
            // Bottom
            if (used.y > fr.y) {
                newFree.push_back(Rect{fr.x, fr.y, fr.w, used.y - fr.y});
            }
            // Top
            if (used.y + used.h < fr.y + fr.h) {
                newFree.push_back(Rect{fr.x, used.y + used.h, fr.w, (fr.y + fr.h) - (used.y + used.h)});
            }
        }

        freeRects.swap(newFree);

        // Prune contained rectangles (O(F^2), OK for small/medium).
        for (size_t i = 0; i < freeRects.size(); ++i) {
            for (size_t j = i + 1; j < freeRects.size();) {
                if (contains(freeRects[i], freeRects[j])) {
                    freeRects.erase(freeRects.begin() + static_cast<std::ptrdiff_t>(j));
                } else if (contains(freeRects[j], freeRects[i])) {
                    freeRects.erase(freeRects.begin() + static_cast<std::ptrdiff_t>(i));
                    --i;
                    break;
                } else {
                    ++j;
                }
            }
        }
    }
};

static vector<Placement> solve(int BW, int BH, const vector<InputRect>& rects) {
    struct Item {
        int id;
        int w, h;
        int64_t area;
        int mx;
    };
    vector<Item> items;
    items.reserve(rects.size());
    for (auto& r : rects) {
        items.push_back(Item{r.id, r.w, r.h, int64_t(r.w) * r.h, std::max(r.w, r.h)});
    }
    std::sort(items.begin(), items.end(), [](const Item& a, const Item& b) {
        if (a.area != b.area) return a.area > b.area;
        if (a.mx != b.mx) return a.mx > b.mx;
        if (a.w != b.w) return a.w > b.w;
        return a.h > b.h;
    });

    vector<Placement> out(rects.size());
    vector<Bin> bins;
    bins.reserve(128);

    for (const auto& it : items) {
        Candidate best;
        int bestBin = -1;

        for (int bi = 0; bi < (int)bins.size(); ++bi) {
            Candidate c = bins[bi].findBest(it.w, it.h, it.w, it.h);
            if (!c.ok) continue;

            if (!best.ok || c.score1 < best.score1 || (c.score1 == best.score1 && c.score2 < best.score2)) {
                best = c;
                bestBin = bi;
            }
        }

        if (!best.ok) {
            bins.emplace_back(BW, BH);
            bestBin = (int)bins.size() - 1;
            best = bins.back().findBest(it.w, it.h, it.w, it.h);
            // Assume instances are feasible.
        }

        out[it.id] = Placement{bestBin, best.x, best.y, best.rotated};
        bins[bestBin].placeRect(Rect{best.x, best.y, best.w, best.h});
    }

    return out;
}

} // namespace maxrects

// ------------------------
// Large: Shelf Best-Fit
// ------------------------
namespace shelfpack {

struct Shelf {
    int bin = -1;
    int y = 0;
    int height = 0;
    int usedW = 0;
    multimap<int, int>::iterator it; // key = remaining width, value = shelf index
};

struct Bin {
    int usedH = 0;
    multimap<int, int>::iterator it; // key = remaining height, value = bin index
};

struct Orientation {
    int w = 0, h = 0;
    int rotated = 0;
};

struct Solver {
    int BW, BH;
    vector<Bin> bins;
    vector<Shelf> shelves;
    multimap<int, int> shelvesByRemW;
    multimap<int, int> binsByRemH;

    explicit Solver(int BW_, int BH_) : BW(BW_), BH(BH_) {
        bins.reserve(1024);
        shelves.reserve(8192);
    }

    int createBin() {
        int idx = (int)bins.size();
        bins.push_back(Bin{0});
        int remH = BH;
        bins[idx].it = binsByRemH.insert({remH, idx});
        return idx;
    }

    void updateBinRemH(int bi, int newRemH) {
        binsByRemH.erase(bins[bi].it);
        if (newRemH > 0) bins[bi].it = binsByRemH.insert({newRemH, bi});
        else bins[bi].it = binsByRemH.end();
    }

    void updateShelfRemW(int si, int newRemW) {
        shelvesByRemW.erase(shelves[si].it);
        shelves[si].it = shelvesByRemW.insert({newRemW, si});
    }

    // Find first shelf (best-fit by remaining width) that can fit (w,h).
    // Returns shelf index, or -1.
    int findShelf(int w, int h, int& outRemW) {
        auto it = shelvesByRemW.lower_bound(w);
        for (; it != shelvesByRemW.end(); ++it) {
            int si = it->second;
            const Shelf& sh = shelves[si];
            if (sh.height >= h) {
                outRemW = it->first;
                return si;
            }
        }
        return -1;
    }

    // Choose best shelf and orientation among two options, by minimal leftover width.
    bool tryPlaceInExistingShelves(const Orientation& a, const Orientation& b,
                                   int& chosenShelf, Orientation& chosenOri) {
        int remA = 0, remB = 0;
        int sa = -1, sb = -1;

        if (a.w <= BW && a.h <= BH) sa = findShelf(a.w, a.h, remA);
        if (b.w <= BW && b.h <= BH) sb = findShelf(b.w, b.h, remB);

        if (sa < 0 && sb < 0) return false;

        auto leftover = [](int rem, int w) { return rem - w; };

        if (sa >= 0 && sb >= 0) {
            int la = leftover(remA, a.w);
            int lb = leftover(remB, b.w);
            if (la < lb) {
                chosenShelf = sa; chosenOri = a; return true;
            }
            if (lb < la) {
                chosenShelf = sb; chosenOri = b; return true;
            }
            // Tie: choose lower y, then lower board index (stability)
            const Shelf& sha = shelves[sa];
            const Shelf& shb = shelves[sb];
            if (sha.y != shb.y) {
                chosenShelf = (sha.y < shb.y) ? sa : sb;
                chosenOri = (sha.y < shb.y) ? a : b;
                return true;
            }
            chosenShelf = (sha.bin < shb.bin) ? sa : sb;
            chosenOri = (sha.bin < shb.bin) ? a : b;
            return true;
        }

        if (sa >= 0) { chosenShelf = sa; chosenOri = a; return true; }
        chosenShelf = sb; chosenOri = b; return true;
    }

    int pickBinForNewShelf(int shelfH) {
        auto it = binsByRemH.lower_bound(shelfH);
        if (it == binsByRemH.end()) return -1;
        return it->second; // best-fit: smallest remaining height that works
    }

    int createShelfInBin(int bi, int shelfH, int firstW) {
        // assumes shelfH <= remaining height and firstW <= BW
        int si = (int)shelves.size();
        Shelf sh;
        sh.bin = bi;
        sh.y = bins[bi].usedH;
        sh.height = shelfH;
        sh.usedW = firstW;

        int remW = BW - firstW;
        sh.it = shelvesByRemW.insert({remW, si});

        shelves.push_back(sh);

        bins[bi].usedH += shelfH;
        int newRemH = BH - bins[bi].usedH;
        updateBinRemH(bi, newRemH);

        return si;
    }

    vector<Placement> solve(const vector<InputRect>& rects) {
        struct Item {
            int id;
            int w, h;
            int longSide, shortSide;
        };

        vector<Item> items;
        items.reserve(rects.size());
        for (auto& r : rects) {
            int L = std::max(r.w, r.h);
            int S = std::min(r.w, r.h);
            items.push_back(Item{r.id, r.w, r.h, L, S});
        }

        // Sort by long side descending, then short side descending.
        std::sort(items.begin(), items.end(), [](const Item& a, const Item& b) {
            if (a.longSide != b.longSide) return a.longSide > b.longSide;
            if (a.shortSide != b.shortSide) return a.shortSide > b.shortSide;
            if (a.w != b.w) return a.w > b.w;
            return a.h > b.h;
        });

        vector<Placement> out(rects.size());

        // Start with one bin.
        createBin();

        for (const auto& it : items) {
            // Two orientations relative to original (it.w, it.h)
            Orientation o1{it.shortSide, it.longSide, 0}; // "tall shelf" orientation
            Orientation o2{it.longSide, it.shortSide, 0}; // rotated

            // Determine rotated flags relative to original dimensions:
            auto setRot = [&](Orientation& o) {
                if (it.w == it.h) { o.rotated = 0; return; }
                o.rotated = (o.w == it.w && o.h == it.h) ? 0 : 1;
            };
            setRot(o1);
            setRot(o2);

            // 1) Try existing shelves (best-fit by remaining width).
            int chosenShelf = -1;
            Orientation chosenOri;
            if (tryPlaceInExistingShelves(o1, o2, chosenShelf, chosenOri)) {
                Shelf& sh = shelves[chosenShelf];
                int px = sh.usedW;
                int py = sh.y;

                sh.usedW += chosenOri.w;
                int newRemW = BW - sh.usedW;
                updateShelfRemW(chosenShelf, newRemW);

                out[it.id] = Placement{sh.bin, px, py, chosenOri.rotated};
                continue;
            }

            // 2) Need a new shelf in some bin:
            // Prefer o1 (tall shelf), but if no bin can fit its height, try o2 (short shelf).
            int bi = -1;
            Orientation shelfOri;

            if (o1.w <= BW && o1.h <= BH) {
                bi = pickBinForNewShelf(o1.h);
                if (bi >= 0) shelfOri = o1;
            }
            if (bi < 0 && o2.w <= BW && o2.h <= BH) {
                bi = pickBinForNewShelf(o2.h);
                if (bi >= 0) shelfOri = o2;
            }

            if (bi < 0) {
                // Open a new bin and try again (prefer o1, else o2).
                bi = createBin();
                if (o1.w <= BW && o1.h <= BH) shelfOri = o1;
                else shelfOri = o2; // assume feasible instances
            }

            int shelfH = shelfOri.h;
            int firstW = shelfOri.w;

            // Create shelf and place at (0, shelfY).
            int shelfIdx = createShelfInBin(bi, shelfH, firstW);
            Shelf& sh = shelves[shelfIdx];

            out[it.id] = Placement{bi, 0, sh.y, shelfOri.rotated};
        }

        return out;
    }
};

} // namespace shelfpack

int main() {
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    int BW, BH;
    if (!(cin >> N >> BW >> BH)) return 0;

    vector<InputRect> rects;
    rects.reserve(N);
    for (int i = 0; i < N; ++i) {
        int w, h;
        cin >> w >> h;
        rects.push_back(InputRect{i, w, h});
    }

    // Assume feasible instances; if not feasible, we still must output something.
    // (Typical cutting stock tasks only provide feasible rectangles.)
    for (const auto& r : rects) {
        if (!canFitEither(BW, BH, r.w, r.h)) {
            // Output a clearly invalid but well-formed result (best effort).
            cout << 0 << "\n";
            for (int i = 0; i < N; ++i) cout << "0 0 0 0\n";
            return 0;
        }
    }

    vector<Placement> ans;
    int numBoards = 0;

    // Adaptive strategy:
    // - For smaller instances: MaxRects (higher quality).
    // - For huge instances: Shelf best-fit (very fast, scalable).
    if (N <= 2500) {
        ans = maxrects::solve(BW, BH, rects);
        numBoards = 0;
        for (const auto& p : ans) numBoards = std::max(numBoards, p.board + 1);
    } else {
        shelfpack::Solver solver(BW, BH);
        ans = solver.solve(rects);
        numBoards = 0;
        for (const auto& p : ans) numBoards = std::max(numBoards, p.board + 1);
    }

    cout << numBoards << "\n";
    for (int i = 0; i < N; ++i) {
        const auto& p = ans[i];
        cout << p.board << " " << p.x << " " << p.y << " " << p.rotated << "\n";
    }
    return 0;
}
