#!/usr/bin/env python3
import sys


def int_stream():
    """
    Fast streaming int parser over stdin.buffer.
    Yields signed integers separated by arbitrary whitespace/newlines.
    """
    data = sys.stdin.buffer
    buf = bytearray(1 << 20)
    mv = memoryview(buf)

    num = 0
    sign = 1
    in_num = False

    while True:
        n = data.readinto(buf)
        if not n:
            break
        for b in mv[:n]:
            if 48 <= b <= 57:  # '0'..'9'
                num = num * 10 + (b - 48)
                in_num = True
            elif b == 45:  # '-'
                # only meaningful before digits; input is well-formed
                sign = -1
            else:
                if in_num:
                    yield sign * num
                    num = 0
                    sign = 1
                    in_num = False
                else:
                    sign = 1

    if in_num:
        yield sign * num


def main():
    it = int_stream()
    nxt = it.__next__

    try:
        nvars = nxt()
        nclauses = nxt()
    except StopIteration:
        return

    # assignment: 0=False, 1=True, 2=unassigned
    assign = bytearray(nvars + 1)
    assign[:] = b"\x02" * (nvars + 1)

    # "locked" variables: avoid flipping these in rare conflict repairs
    locked = bytearray(nvars + 1)

    a_arr = assign
    l_arr = locked

    for _ in range(nclauses):
        a = nxt()
        b = nxt()
        c = nxt()

        # literal a
        if a < 0:
            va = -a
            na = 0
        else:
            va = a
            na = 1
        vala = a_arr[va]
        ua = (vala == 2)
        ta = (not ua) and (vala == na)

        # literal b
        if b < 0:
            vb = -b
            nb = 0
        else:
            vb = b
            nb = 1
        valb = a_arr[vb]
        ub = (valb == 2)
        tb = (not ub) and (valb == nb)

        # literal c
        if c < 0:
            vc = -c
            nc = 0
        else:
            vc = c
            nc = 1
        valc = a_arr[vc]
        uc = (valc == 2)
        tc = (not uc) and (valc == nc)

        if ta or tb or tc:
            # If all assigned and exactly one is true, lock that variable.
            if (not ua) and (not ub) and (not uc):
                cnt = (1 if ta else 0) + (1 if tb else 0) + (1 if tc else 0)
                if cnt == 1:
                    if ta:
                        l_arr[va] = 1
                    elif tb:
                        l_arr[vb] = 1
                    else:
                        l_arr[vc] = 1
            continue

        # Not currently satisfied: try assigning an unassigned var to satisfy.
        if ua:
            a_arr[va] = na
            # If the other two are assigned and false, this var is critical -> lock it.
            if (not ub) and (not tb) and (not uc) and (not tc):
                l_arr[va] = 1
            continue
        if ub:
            a_arr[vb] = nb
            if (not ua) and (not ta) and (not uc) and (not tc):
                l_arr[vb] = 1
            continue
        if uc:
            a_arr[vc] = nc
            if (not ua) and (not ta) and (not ub) and (not tb):
                l_arr[vc] = 1
            continue

        # Conflict: all three assigned but all false under current assignment.
        # Heuristic repair: flip a non-locked variable to satisfy this clause.
        if not l_arr[va]:
            a_arr[va] = na
        elif not l_arr[vb]:
            a_arr[vb] = nb
        elif not l_arr[vc]:
            a_arr[vc] = nc
        else:
            # Last resort: flip anyway.
            a_arr[va] = na

    out = sys.stdout.buffer
    out.write(b"SAT\n")

    # Stream assignment as "0/1" space-separated, newline at end.
    # Unassigned -> set to True (1).
    one_sp = b"1 "
    zero_sp = b"0 "
    buf = bytearray()
    flush_limit = 1 << 20  # ~1MB

    for i in range(1, nvars + 1):
        v = a_arr[i]
        if v == 2:
            v = 1
        buf += (one_sp if v else zero_sp)
        if len(buf) >= flush_limit:
            out.write(buf)
            buf.clear()

    if nvars > 0:
        if not buf:
            # unlikely, but handle edge-case
            out.write(b"\n")
        else:
            buf[-1] = 10  # replace trailing space with '\n'
            out.write(buf)
    else:
        out.write(b"\n")


if __name__ == "__main__":
    main()
