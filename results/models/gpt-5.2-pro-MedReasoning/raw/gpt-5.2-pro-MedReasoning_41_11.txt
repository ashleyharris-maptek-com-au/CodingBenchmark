; SPIR-V
; Version: 1.0
; Bound: 100
; Schema: 0
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %in_worldPos %in_normal %in_uv %ubo
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%int = OpTypeInt 32 1

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4

%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float

%func = OpTypeFunction %void

%float_0 = OpConstant %float 0.0
%float_0_1 = OpConstant %float 0.1
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_7 = OpConstant %float 0.7
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1.0
%float_6 = OpConstant %float 6.0

%v2_half = OpConstantComposite %v2float %float_0_5 %float_0_5
%v3_yellow = OpConstantComposite %v3float %float_1 %float_0_8 %float_0
%v3_bg = OpConstantComposite %v3float %float_0_1 %float_0_1 %float_0_3

%int_3 = OpConstant %int 3

%in_worldPos = OpVariable %_ptr_Input_v3float Input
%in_normal = OpVariable %_ptr_Input_v3float Input
%in_uv = OpVariable %_ptr_Input_v2float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%main = OpFunction %void None %func
%entry = OpLabel

%uv = OpLoad %v2float %in_uv
%uv6 = OpVectorTimesScalar %v2float %uv %float_6
%cell = OpExtInst %v2float %glsl Fract %uv6
%delta = OpFSub %v2float %cell %v2_half
%dist = OpExtInst %float %glsl Length %delta
%isDot = OpFOrdLessThan %bool %dist %float_0_3
%base = OpSelect %v3float %isDot %v3_yellow %v3_bg

%N_in = OpLoad %v3float %in_normal
%N = OpExtInst %v3float %glsl Normalize %N_in
%pos = OpLoad %v3float %in_worldPos

%lightPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%light4 = OpLoad %v4float %lightPtr
%light3 = OpVectorShuffle %v3float %light4 %light4 0 1 2
%Lvec = OpFSub %v3float %light3 %pos
%L = OpExtInst %v3float %glsl Normalize %Lvec
%dotNL = OpDot %float %N %L
%diff = OpExtInst %float %glsl FMax %dotNL %float_0
%scaled = OpFMul %float %diff %float_0_7
%shade = OpFAdd %float %scaled %float_0_3

%rgb = OpVectorTimesScalar %v3float %base %shade
%r = OpCompositeExtract %float %rgb 0
%g = OpCompositeExtract %float %rgb 1
%b = OpCompositeExtract %float %rgb 2
%out = OpCompositeConstruct %v4float %r %g %b %float_1
OpStore %outColor %out

OpReturn
OpFunctionEnd
