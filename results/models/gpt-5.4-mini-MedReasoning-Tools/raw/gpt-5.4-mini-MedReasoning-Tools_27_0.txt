#include <algorithm>
#include <chrono>
#include <cstdint>
#include <iostream>
#include <limits>
#include <bit>
#include <numeric>
#include <queue>
#include <stack>
#include <unordered_map>
#include <utility>
#include <vector>

using namespace std;

struct Component {
    vector<int> verts;
    long long edges = 0;
};

struct ExactSolver {
    int k;
    int n;
    const vector<vector<int>>* adj;
    vector<int> color;
    vector<int> degree;
    uint64_t fullMask = 0;
    size_t nodesVisited = 0;
    chrono::steady_clock::time_point start;
    double timeLimitSec = 1.5;

    bool timedOut() const {
        return chrono::duration<double>(chrono::steady_clock::now() - start).count() > timeLimitSec;
    }

    uint64_t usedMask(int v) const {
        uint64_t mask = 0;
        for (int u : (*adj)[v]) {
            int c = color[u];
            if (c >= 0) {
                mask |= (1ULL << c);
            }
        }
        return mask;
    }

    int selectVertex() const {
        int best = -1;
        int bestSat = -1;
        int bestDeg = -1;
        for (int i = 0; i < n; ++i) {
            if (color[i] != -1) continue;
            uint64_t mask = usedMask(i);
            int sat = (int)std::popcount(mask);
            int deg = degree[i];
            if (sat > bestSat || (sat == bestSat && deg > bestDeg)) {
                bestSat = sat;
                bestDeg = deg;
                best = i;
            }
        }
        return best;
    }

    bool dfs(int colored) {
        if (colored == n) return true;
        if ((++nodesVisited & 1023ULL) == 0 && timedOut()) return false;

        int v = selectVertex();
        if (v < 0) return false;
        uint64_t used = usedMask(v);
        uint64_t avail = fullMask & ~used;
        if (!avail) return false;

        while (avail) {
            int c = (int)std::countr_zero(avail);
            avail &= (avail - 1);
            color[v] = c;
            if (dfs(colored + 1)) return true;
            color[v] = -1;
        }
        return false;
    }
};

struct Solver {
    int n = 0, m = 0, k = 0;
    vector<int> U, V;
    vector<int> off;
    vector<int> adj;
    vector<int> compId;
    vector<int> color;
    vector<Component> comps;
    vector<int> repairSeen;
    int repairStamp = 0;
    uint64_t fullMask = 0;
    chrono::steady_clock::time_point start;
    double timeLimitSec = 280.0;

    bool timedOut() const {
        return chrono::duration<double>(chrono::steady_clock::now() - start).count() > timeLimitSec;
    }

    void buildCSR() {
        off.assign(n + 1, 0);
        for (int i = 0; i < m; ++i) {
            ++off[U[i] + 1];
            ++off[V[i] + 1];
        }
        for (int i = 1; i <= n; ++i) off[i] += off[i - 1];
        adj.assign(2 * m, -1);
        vector<int> cur = off;
        for (int i = 0; i < m; ++i) {
            int a = U[i], b = V[i];
            adj[cur[a]++] = b;
            adj[cur[b]++] = a;
        }
    }

    void findComponents() {
        compId.assign(n, -1);
        comps.clear();
        vector<int> q;
        q.reserve(n);
        for (int s = 0; s < n; ++s) {
            if (compId[s] != -1) continue;
            int cid = (int)comps.size();
            comps.push_back(Component{});
            auto& comp = comps.back();
            q.clear();
            q.push_back(s);
            compId[s] = cid;
            for (size_t qi = 0; qi < q.size(); ++qi) {
                int v = q[qi];
                comp.verts.push_back(v);
                comp.edges += off[v + 1] - off[v];
                for (int ei = off[v]; ei < off[v + 1]; ++ei) {
                    int u = adj[ei];
                    if (compId[u] == -1) {
                        compId[u] = cid;
                        q.push_back(u);
                    }
                }
            }
            comp.edges /= 2;
        }
    }

    uint64_t usedMask(int v, int cid) const {
        uint64_t mask = 0;
        for (int ei = off[v]; ei < off[v + 1]; ++ei) {
            int u = adj[ei];
            if (compId[u] != cid) continue;
            int c = color[u];
            if (c >= 0) mask |= (1ULL << c);
        }
        return mask;
    }

    bool directAssign(int v, int cid) {
        uint64_t used = usedMask(v, cid);
        uint64_t avail = fullMask & ~used;
        if (!avail) return false;
        int c = (int)std::countr_zero(avail);
        color[v] = c;
        return true;
    }

    void rollback(vector<pair<int,int>>& trail, size_t checkpoint) {
        while (trail.size() > checkpoint) {
            auto [v, oldColor] = trail.back();
            trail.pop_back();
            color[v] = oldColor;
        }
    }

    bool recolorAway(int v, int banned, int cid, int depth, vector<pair<int,int>>& trail) {
        if (repairSeen[v] == repairStamp) return false;
        repairSeen[v] = repairStamp;

        uint64_t used = usedMask(v, cid);
        for (int c = 0; c < k; ++c) {
            if (c == banned) continue;
            if ((used & (1ULL << c)) == 0) {
                trail.push_back({v, color[v]});
                color[v] = c;
                return true;
            }
        }
        if (depth == 0) return false;

        for (int ei = off[v]; ei < off[v + 1]; ++ei) {
            int u = adj[ei];
            if (compId[u] != cid) continue;
            int cu = color[u];
            if (cu < 0) continue;
            if (repairSeen[u] == repairStamp) continue;
            size_t cp = trail.size();
            if (recolorAway(u, cu, cid, depth - 1, trail)) {
                used = usedMask(v, cid);
                for (int c = 0; c < k; ++c) {
                    if (c == banned) continue;
                    if ((used & (1ULL << c)) == 0) {
                        trail.push_back({v, color[v]});
                        color[v] = c;
                        return true;
                    }
                }
            }
            rollback(trail, cp);
        }
        return false;
    }

    bool assignWithRepair(int v, int cid, int depth) {
        uint64_t used = usedMask(v, cid);
        int witness[64];
        for (int i = 0; i < 64; ++i) witness[i] = -1;
        for (int ei = off[v]; ei < off[v + 1]; ++ei) {
            int u = adj[ei];
            if (compId[u] != cid) continue;
            int c = color[u];
            if (c >= 0 && c < 64 && witness[c] == -1) witness[c] = u;
        }
        if (used != fullMask) {
            for (int c = 0; c < k; ++c) {
                if ((used & (1ULL << c)) == 0) {
                    color[v] = c;
                    return true;
                }
            }
        }
        if (depth <= 0) return false;

        vector<pair<int,int>> trail;
        for (int c = 0; c < k; ++c) {
            int u = witness[c];
            if (u == -1) continue;
            size_t cp = trail.size();
            ++repairStamp;
            if (recolorAway(u, c, cid, depth - 1, trail)) {
                used = usedMask(v, cid);
                for (int cc = 0; cc < k; ++cc) {
                    if ((used & (1ULL << cc)) == 0) {
                        trail.push_back({v, -1});
                        color[v] = cc;
                        return true;
                    }
                }
            }
            rollback(trail, cp);
        }
        return false;
    }

    vector<int> buildSmallestLastOrder(const Component& comp) {
        int sz = (int)comp.verts.size();
        unordered_map<int,int> idx;
        idx.reserve(sz * 2 + 1);
        for (int i = 0; i < sz; ++i) idx[comp.verts[i]] = i;

        vector<int> deg(sz, 0);
        for (int i = 0; i < sz; ++i) {
            int v = comp.verts[i];
            for (int ei = off[v]; ei < off[v + 1]; ++ei) {
                int u = adj[ei];
                if (compId[u] == compId[v]) ++deg[i];
            }
        }

        using Node = pair<int,int>;
        priority_queue<Node, vector<Node>, greater<Node>> pq;
        for (int i = 0; i < sz; ++i) pq.push({deg[i], i});

        vector<char> removed(sz, 0);
        vector<int> order;
        order.reserve(sz);
        while (!pq.empty()) {
            auto [d, i] = pq.top();
            pq.pop();
            if (removed[i] || d != deg[i]) continue;
            removed[i] = 1;
            order.push_back(comp.verts[i]);
            int v = comp.verts[i];
            for (int ei = off[v]; ei < off[v + 1]; ++ei) {
                int u = adj[ei];
                if (compId[u] != compId[v]) continue;
                auto it = idx.find(u);
                if (it == idx.end()) continue;
                int j = it->second;
                if (removed[j]) continue;
                --deg[j];
                pq.push({deg[j], j});
            }
        }
        reverse(order.begin(), order.end());
        return order;
    }

    vector<int> buildLargestDegreeOrder(const Component& comp) {
        vector<int> order = comp.verts;
        sort(order.begin(), order.end(), [&](int a, int b) {
            int da = off[a + 1] - off[a];
            int db = off[b + 1] - off[b];
            if (da != db) return da > db;
            return a < b;
        });
        return order;
    }

    bool greedyColorOrder(const vector<int>& order, int cid, int repairDepth) {
        for (int v : order) {
            if (color[v] != -1) continue;
            if (!directAssign(v, cid)) {
                if (!assignWithRepair(v, cid, repairDepth)) return false;
            }
        }
        return true;
    }

    bool solveExactSmall(const Component& comp) {
        int sz = (int)comp.verts.size();
        vector<vector<int>> ladj(sz);
        unordered_map<int,int> pos;
        pos.reserve(sz * 2 + 1);
        for (int i = 0; i < sz; ++i) pos[comp.verts[i]] = i;
        for (int i = 0; i < sz; ++i) {
            int v = comp.verts[i];
            for (int ei = off[v]; ei < off[v + 1]; ++ei) {
                int u = adj[ei];
                if (compId[u] != compId[v]) continue;
                auto it = pos.find(u);
                if (it != pos.end()) ladj[i].push_back(it->second);
            }
        }

        ExactSolver ex;
        ex.k = k;
        ex.n = sz;
        ex.adj = &ladj;
        ex.color.assign(sz, -1);
        ex.degree.assign(sz, 0);
        for (int i = 0; i < sz; ++i) ex.degree[i] = (int)ladj[i].size();
        ex.fullMask = fullMask;
        ex.start = start;
        ex.timeLimitSec = 1.0;
        if (!ex.dfs(0)) return false;
        for (int i = 0; i < sz; ++i) color[comp.verts[i]] = ex.color[i];
        return true;
    }

    bool solveComponent(const Component& comp) {
        if (timedOut()) return false;
        int sz = (int)comp.verts.size();
        int cid = compId[comp.verts[0]];

        if (sz <= 80) {
            for (int v : comp.verts) color[v] = -1;
            if (solveExactSmall(comp)) return true;
            for (int v : comp.verts) color[v] = -1;
        }

        double avgDeg = sz ? (2.0 * (double)comp.edges / (double)sz) : 0.0;
        vector<int> order1, order2, order3;

        if (avgDeg > 12.0 || sz > 50000) {
            // Dense or very large components tend to benefit from saturation-first ordering.
            order1 = comp.verts;
            order2 = buildSmallestLastOrder(comp);
            order3 = buildLargestDegreeOrder(comp);
        } else {
            order1 = buildSmallestLastOrder(comp);
            order2 = buildLargestDegreeOrder(comp);
            order3 = comp.verts;
        }

        auto reset = [&]() {
            for (int v : comp.verts) color[v] = -1;
        };

        reset();
        if (greedyColorOrder(order1, cid, sz <= 20000 ? 2 : 1)) return true;
        reset();
        if (order2 != order1 && greedyColorOrder(order2, cid, sz <= 20000 ? 2 : 1)) return true;
        reset();
        if (order3 != order1 && order3 != order2 && greedyColorOrder(order3, cid, sz <= 20000 ? 2 : 1)) return true;

        // A bounded DSATUR pass as final heuristic.
        reset();
        vector<uint64_t> satMask(n, 0);
        struct Node {
            int sat;
            int deg;
            int v;
        };
        struct Cmp {
            bool operator()(const Node& a, const Node& b) const {
                if (a.sat != b.sat) return a.sat < b.sat;
                if (a.deg != b.deg) return a.deg < b.deg;
                return a.v > b.v;
            }
        };
        priority_queue<Node, vector<Node>, Cmp> pq;
        for (int v : comp.verts) pq.push({0, off[v + 1] - off[v], v});
        int colored = 0;
        vector<int> seenLocal(n, 0);
        int token = 1;
        while (colored < sz) {
            if (timedOut()) break;
            Node cur{};
            bool found = false;
            while (!pq.empty()) {
                cur = pq.top();
                pq.pop();
                if (color[cur.v] != -1) continue;
                int actualSat = (int)std::popcount(satMask[cur.v]);
                if (actualSat != cur.sat) {
                    pq.push({actualSat, cur.deg, cur.v});
                    continue;
                }
                found = true;
                break;
            }
            if (!found) break;
            int v = cur.v;

            uint64_t used = usedMask(v, cid);
            if (used == fullMask) {
                if (!assignWithRepair(v, cid, sz <= 20000 ? 2 : 1)) break;
            } else {
                int c = -1;
                for (int cc = 0; cc < k; ++cc) {
                    if ((used & (1ULL << cc)) == 0) {
                        c = cc;
                        break;
                    }
                }
                if (c < 0) {
                    if (!assignWithRepair(v, cid, sz <= 20000 ? 2 : 1)) break;
                } else {
                    color[v] = c;
                }
            }
            ++colored;
            for (int ei = off[v]; ei < off[v + 1]; ++ei) {
                int u = adj[ei];
                if (compId[u] != cid || color[u] != -1) continue;
                uint64_t old = satMask[u];
                uint64_t bit = 1ULL << color[v];
                if ((old & bit) == 0) {
                    satMask[u] = old | bit;
                    pq.push({(int)std::popcount(satMask[u]), off[u + 1] - off[u], u});
                }
            }
        }
        if (colored == sz) return true;

        return false;
    }

    bool validate() const {
        for (int i = 0; i < m; ++i) {
            if (color[U[i]] == color[V[i]]) return false;
        }
        return true;
    }

    void run() {
        ios::sync_with_stdio(false);
        cin.tie(nullptr);

        cin >> n >> m >> k;
        U.resize(m);
        V.resize(m);
        for (int i = 0; i < m; ++i) {
            cin >> U[i] >> V[i];
        }

        if (k <= 0) {
            // Degenerate input; emit a harmless fallback.
            for (int i = 0; i < n; ++i) {
                if (i) cout << ' ';
                cout << 0;
            }
            cout << '\n';
            return;
        }

        fullMask = (k >= 64) ? ~0ULL : ((1ULL << k) - 1ULL);
        color.assign(n, -1);
        repairSeen.assign(n, 0);

        buildCSR();
        findComponents();
        sort(comps.begin(), comps.end(), [](const Component& a, const Component& b) {
            return a.verts.size() > b.verts.size();
        });

        for (const auto& comp : comps) {
            if (!solveComponent(comp)) {
                // Last resort: simple greedy in input order.
                for (int v : comp.verts) color[v] = -1;
                for (int v : comp.verts) {
                    uint64_t used = usedMask(v, compId[v]);
                    uint64_t avail = fullMask & ~used;
                    if (!avail) {
                        color[v] = 0;
                    } else {
                        color[v] = (int)std::countr_zero(avail);
                    }
                }
            }
        }

        if (!validate()) {
            // As a safety net, try a global smallest-last pass on all vertices.
            fill(color.begin(), color.end(), -1);
            Component all;
            all.verts.resize(n);
            iota(all.verts.begin(), all.verts.end(), 0);
            all.edges = m;
            compId.assign(n, 0);
            if (!solveComponent(all)) {
                // If even this fails, emit a fallback coloring.
                for (int i = 0; i < n; ++i) color[i] = i % k;
            }
        }

        for (int i = 0; i < n; ++i) {
            if (i) cout << ' ';
            cout << color[i];
        }
        cout << '\n';
    }
};

int main() {
    Solver solver;
    solver.start = chrono::steady_clock::now();
    solver.run();
    return 0;
}
