using System;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var input = new FastInput(Console.OpenStandardInput());

        int xCount = int.Parse(input.ReadToken(), CultureInfo.InvariantCulture);
        int yCount = int.Parse(input.ReadToken(), CultureInfo.InvariantCulture);
        int zCount = int.Parse(input.ReadToken(), CultureInfo.InvariantCulture);

        // Block sizes are part of the header but do not affect the chosen voxel heuristic.
        input.ReadToken();
        input.ReadToken();
        input.ReadLastToken();

        long totalVoxels = (long)xCount * yCount * zCount;
        bool exactMode = totalVoxels <= 5_000_000L;

        // In exact mode we examine every solid segment in every column.
        // In large mode we cap the work to the area most likely to be profitable.
        int largeRadiusLimit = 256;
        int largeMaxSolidPerColumn = 8;

        byte[] column = new byte[zCount];
        var output = new StringBuilder();
        long opCount = 0;

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    int b = input.ReadRawByte();
                    if (b < 0)
                    {
                        throw new EndOfStreamException("Unexpected end of voxel data.");
                    }
                    column[z] = (byte)b;
                }

                if (!exactMode && x + y > largeRadiusLimit)
                {
                    continue;
                }

                if (exactMode)
                {
                    ProcessColumnExact(x, y, zCount, column, output, ref opCount);
                }
                else
                {
                    ProcessColumnFast(x, y, zCount, column, largeMaxSolidPerColumn, output, ref opCount);
                }
            }
        }

        using var writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        writer.WriteLine(opCount);
        writer.Write(output.ToString());
    }

    private static void ProcessColumnExact(int x, int y, int zCount, byte[] column, StringBuilder output, ref long opCount)
    {
        int z = zCount - 1;
        while (z >= 0)
        {
            while (z >= 0 && column[z] == 0)
            {
                z--;
            }

            if (z < 0)
            {
                break;
            }

            int segmentTop = z;
            while (z >= 0 && column[z] != 0)
            {
                z--;
            }

            int segmentBottom = z + 1;
            int segmentLen = segmentTop - segmentBottom + 1;

            long bestProfit = 0;
            long runningProfit = 0;
            int bestPrefixLen = 0;

            for (int i = 0; i < segmentLen; i++)
            {
                int idx = segmentTop - i;
                byte d = column[idx];
                int mass = d & 15;
                if (mass == 0)
                {
                    continue;
                }

                int value = d >> 4;
                long voxelProfit = (long)mass * (1000L * value - (x + y));
                runningProfit += voxelProfit;
                if (runningProfit > bestProfit)
                {
                    bestProfit = runningProfit;
                    bestPrefixLen = i + 1;
                }
            }

            if (bestProfit > 0 && bestPrefixLen > 0)
            {
                for (int i = 0; i < bestPrefixLen; i++)
                {
                    int idx = segmentTop - i;
                    byte d = column[idx];
                    int mass = d & 15;
                    if (mass == 0)
                    {
                        continue;
                    }

                    AppendOperation(output, x, y, idx);
                    opCount++;
                }
            }
        }
    }

    private static void ProcessColumnFast(int x, int y, int zCount, byte[] column, int maxSolidToInspect, StringBuilder output, ref long opCount)
    {
        int z = zCount - 1;
        int solidsSeen = 0;

        while (z >= 0 && solidsSeen < maxSolidToInspect)
        {
            while (z >= 0 && column[z] == 0 && solidsSeen < maxSolidToInspect)
            {
                z--;
            }

            if (z < 0 || solidsSeen >= maxSolidToInspect)
            {
                break;
            }

            int segmentTop = z;
            while (z >= 0 && column[z] != 0 && solidsSeen < maxSolidToInspect)
            {
                z--;
                solidsSeen++;
            }

            int segmentBottom = z + 1;
            int segmentLen = segmentTop - segmentBottom + 1;

            long bestProfit = 0;
            long runningProfit = 0;
            int bestPrefixLen = 0;

            for (int i = 0; i < segmentLen; i++)
            {
                int idx = segmentTop - i;
                byte d = column[idx];
                int mass = d & 15;
                if (mass == 0)
                {
                    continue;
                }

                int value = d >> 4;
                long voxelProfit = (long)mass * (1000L * value - (x + y));
                runningProfit += voxelProfit;
                if (runningProfit > bestProfit)
                {
                    bestProfit = runningProfit;
                    bestPrefixLen = i + 1;
                }
            }

            if (bestProfit > 0 && bestPrefixLen > 0)
            {
                for (int i = 0; i < bestPrefixLen; i++)
                {
                    int idx = segmentTop - i;
                    byte d = column[idx];
                    int mass = d & 15;
                    if (mass == 0)
                    {
                        continue;
                    }

                    AppendOperation(output, x, y, idx);
                    opCount++;
                }
            }
        }
    }

    private static void AppendOperation(StringBuilder output, int digX, int digY, int digZ)
    {
        int dist = digX + digY;
        int intermediateCount = dist == 0 ? 0 : dist - 1;

        output.Append(intermediateCount);
        output.Append(' ');
        output.Append(digX);
        output.Append(' ');
        output.Append(digY);
        output.Append(' ');
        output.Append(digZ);

        int x = digX;
        int y = digY;

        while (x != 0)
        {
            x += x > 0 ? -1 : 1;
            if (x != 0 || y != 0)
            {
                output.Append(' ');
                output.Append(x);
                output.Append(' ');
                output.Append(y);
            }
        }

        while (y != 0)
        {
            y += y > 0 ? -1 : 1;
            if (x != 0 || y != 0)
            {
                output.Append(' ');
                output.Append(x);
                output.Append(' ');
                output.Append(y);
            }
        }

        output.Append(' ');
        output.Append(0);
        output.Append(' ');
        output.Append(0);
        output.AppendLine();
    }
}

internal sealed class FastInput
{
    private readonly Stream _stream;
    private readonly byte[] _buffer;
    private int _length;
    private int _position;
    private int _pushback = -1;

    public FastInput(Stream stream)
    {
        _stream = stream;
        _buffer = new byte[1 << 20];
    }

    public string ReadToken()
    {
        int c;
        do
        {
            c = Read();
        } while (c >= 0 && c <= 32);

        if (c < 0)
        {
            return string.Empty;
        }

        var sb = new StringBuilder();
        while (c >= 0 && c > 32)
        {
            sb.Append((char)c);
            c = Read();
        }

        return sb.ToString();
    }

    public string ReadLastToken()
    {
        int c;
        do
        {
            c = Read();
        } while (c >= 0 && c <= 32);

        if (c < 0)
        {
            return string.Empty;
        }

        var sb = new StringBuilder();
        while (c >= 0 && c != '\r' && c != '\n')
        {
            sb.Append((char)c);
            c = Read();
        }

        if (c == '\r')
        {
            int next = Read();
            if (next != '\n' && next >= 0)
            {
                _pushback = next;
            }
        }

        return sb.ToString();
    }

    public int ReadRawByte()
    {
        return Read();
    }

    private int Read()
    {
        if (_pushback >= 0)
        {
            int c = _pushback;
            _pushback = -1;
            return c;
        }

        if (_position >= _length)
        {
            _length = _stream.Read(_buffer, 0, _buffer.Length);
            _position = 0;
            if (_length <= 0)
            {
                return -1;
            }
        }

        return _buffer[_position++];
    }
}
