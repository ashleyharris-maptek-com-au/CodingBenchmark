use std::io::{self, BufRead, BufReader, BufWriter, Read, Write};

#[derive(Clone, Copy, Default)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

#[derive(Clone, Copy, Default)]
struct Center {
    x: f64,
    y: f64,
    z: f64,
}

impl Center {
    fn from_point(p: Point) -> Self {
        Self {
            x: p.x as f64,
            y: p.y as f64,
            z: p.z as f64,
        }
    }

    fn add_point(&mut self, p: Point) {
        self.x += p.x as f64;
        self.y += p.y as f64;
        self.z += p.z as f64;
    }

    fn scale(self, s: f64) -> Self {
        Self {
            x: self.x * s,
            y: self.y * s,
            z: self.z * s,
        }
    }
}

fn is_ws(b: u8) -> bool {
    matches!(b, b' ' | b'\n' | b'\r' | b'\t' | 0x0b | 0x0c)
}

fn read_usize_token<R: BufRead>(reader: &mut R) -> io::Result<usize> {
    let mut value: usize = 0;
    let mut started = false;

    loop {
        let buf = reader.fill_buf()?;
        if buf.is_empty() {
            return if started {
                Ok(value)
            } else {
                Err(io::Error::new(io::ErrorKind::UnexpectedEof, "missing token"))
            };
        }

        let mut consumed = 0usize;
        while consumed < buf.len() {
            let b = buf[consumed];
            if !started {
                if is_ws(b) {
                    consumed += 1;
                    continue;
                }
                started = true;
            }

            if is_ws(b) {
                consumed += 1;
                reader.consume(consumed);
                return Ok(value);
            }

            if (b'0'..=b'9').contains(&b) {
                value = value.saturating_mul(10).saturating_add((b - b'0') as usize);
            }
            consumed += 1;
        }
        reader.consume(consumed);
    }
}

fn read_point<R: Read>(reader: &mut R) -> io::Result<Point> {
    let mut buf = [0u8; 12];
    reader.read_exact(&mut buf)?;
    Ok(Point {
        x: f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]),
        y: f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]),
        z: f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]),
    })
}

fn coord_point(p: &Point, axis: usize) -> f32 {
    match axis {
        0 => p.x,
        1 => p.y,
        _ => p.z,
    }
}

fn coord_center(c: &Center, axis: usize) -> f64 {
    match axis {
        0 => c.x,
        1 => c.y,
        _ => c.z,
    }
}

fn dist2_point_center(p: &Point, c: &Center) -> f64 {
    let dx = p.x as f64 - c.x;
    let dy = p.y as f64 - c.y;
    let dz = p.z as f64 - c.z;
    dx * dx + dy * dy + dz * dz
}

fn dist2_center_center(a: Center, b: Center) -> f64 {
    let dx = a.x - b.x;
    let dy = a.y - b.y;
    let dz = a.z - b.z;
    dx * dx + dy * dy + dz * dz
}

fn point_hash(p: &Point) -> u64 {
    let mut x = p.x.to_bits() as u64;
    let mut y = p.y.to_bits() as u64;
    let mut z = p.z.to_bits() as u64;
    x ^= x >> 33;
    x = x.wrapping_mul(0xff51afd7ed558ccd);
    x ^= x >> 33;
    y ^= y >> 33;
    y = y.wrapping_mul(0xc4ceb9fe1a85ec53);
    y ^= y >> 33;
    z ^= z >> 33;
    z = z.wrapping_mul(0x9e3779b185ebca87);
    z ^= z >> 33;
    x ^ y.rotate_left(17) ^ z.rotate_left(31)
}

fn choose_left(p: &Point, axis: usize, split: f32) -> bool {
    let v = coord_point(p, axis);
    if v < split {
        true
    } else if v > split {
        false
    } else {
        (point_hash(p) & 1) == 0
    }
}

fn best_axis_points(points: &[Point], idxs: &[usize]) -> usize {
    let mut mean = [0.0f64; 3];
    for &i in idxs {
        mean[0] += points[i].x as f64;
        mean[1] += points[i].y as f64;
        mean[2] += points[i].z as f64;
    }
    let inv = 1.0 / idxs.len() as f64;
    mean[0] *= inv;
    mean[1] *= inv;
    mean[2] *= inv;

    let mut var = [0.0f64; 3];
    for &i in idxs {
        let dx = points[i].x as f64 - mean[0];
        let dy = points[i].y as f64 - mean[1];
        let dz = points[i].z as f64 - mean[2];
        var[0] += dx * dx;
        var[1] += dy * dy;
        var[2] += dz * dz;
    }

    let mut axis = 0usize;
    if var[1] > var[axis] {
        axis = 1;
    }
    if var[2] > var[axis] {
        axis = 2;
    }
    axis
}

fn best_axis_centers(centers: &[Center], idxs: &[usize]) -> usize {
    let mut mean = [0.0f64; 3];
    for &i in idxs {
        mean[0] += centers[i].x;
        mean[1] += centers[i].y;
        mean[2] += centers[i].z;
    }
    let inv = 1.0 / idxs.len() as f64;
    mean[0] *= inv;
    mean[1] *= inv;
    mean[2] *= inv;

    let mut var = [0.0f64; 3];
    for &i in idxs {
        let dx = centers[i].x - mean[0];
        let dy = centers[i].y - mean[1];
        let dz = centers[i].z - mean[2];
        var[0] += dx * dx;
        var[1] += dy * dy;
        var[2] += dz * dz;
    }

    let mut axis = 0usize;
    if var[1] > var[axis] {
        axis = 1;
    }
    if var[2] > var[axis] {
        axis = 2;
    }
    axis
}

#[derive(Clone)]
enum PartNode {
    Leaf { cluster: u32 },
    Split {
        axis: u8,
        split: f32,
        left: usize,
        right: usize,
    },
}

struct PartTree {
    nodes: Vec<PartNode>,
    root: usize,
    leaf_count: usize,
}

fn build_partition_rec(
    points: &[Point],
    idxs: &mut [usize],
    target: usize,
    nodes: &mut Vec<PartNode>,
    next_leaf: &mut u32,
) -> usize {
    if idxs.is_empty() {
        let id = *next_leaf;
        *next_leaf += 1;
        nodes.push(PartNode::Leaf { cluster: id });
        return nodes.len() - 1;
    }

    if target <= 1 || idxs.len() <= 1 {
        let id = *next_leaf;
        *next_leaf += 1;
        nodes.push(PartNode::Leaf { cluster: id });
        return nodes.len() - 1;
    }

    let len = idxs.len();
    let axis = best_axis_points(points, idxs);
    idxs.sort_unstable_by(|&a, &b| {
        coord_point(&points[a], axis).total_cmp(&coord_point(&points[b], axis))
    });

    let mid = len / 2;
    let split = if mid > 0 && mid < len {
        let a = coord_point(&points[idxs[mid - 1]], axis) as f64;
        let b = coord_point(&points[idxs[mid]], axis) as f64;
        ((a + b) * 0.5) as f32
    } else {
        coord_point(&points[idxs[0]], axis)
    };

    let left_target = ((target * mid) / len).max(1).min(target - 1);
    let right_target = target - left_target;
    let (left_idxs, right_idxs) = idxs.split_at_mut(mid);

    let left = build_partition_rec(points, left_idxs, left_target, nodes, next_leaf);
    let right = build_partition_rec(points, right_idxs, right_target, nodes, next_leaf);

    nodes.push(PartNode::Split {
        axis: axis as u8,
        split,
        left,
        right,
    });
    nodes.len() - 1
}

fn build_partition_tree(points: &[Point], target_leaves: usize) -> PartTree {
    if points.is_empty() {
        return PartTree {
            nodes: Vec::new(),
            root: 0,
            leaf_count: 0,
        };
    }

    let mut idxs: Vec<usize> = (0..points.len()).collect();
    let target = target_leaves.max(1).min(points.len());
    let mut nodes = Vec::with_capacity(target * 2 + 1);
    let mut next_leaf = 0u32;
    let root = build_partition_rec(points, &mut idxs, target, &mut nodes, &mut next_leaf);
    PartTree {
        nodes,
        root,
        leaf_count: next_leaf as usize,
    }
}

fn partition_assign(tree: &PartTree, p: &Point) -> u32 {
    if tree.nodes.is_empty() {
        return 0;
    }
    let mut idx = tree.root;
    loop {
        match &tree.nodes[idx] {
            PartNode::Leaf { cluster } => return *cluster,
            PartNode::Split {
                axis,
                split,
                left,
                right,
            } => {
                idx = if choose_left(p, *axis as usize, *split) {
                    *left
                } else {
                    *right
                };
            }
        }
    }
}

#[derive(Clone, Copy)]
struct KdNode {
    pivot: usize,
    axis: u8,
    split: f32,
    left: Option<usize>,
    right: Option<usize>,
}

struct KdTree {
    nodes: Vec<KdNode>,
    root: Option<usize>,
    centers: Vec<Center>,
}

fn build_kd_rec(centers: &[Center], idxs: &mut [usize], nodes: &mut Vec<KdNode>) -> usize {
    if idxs.len() == 1 {
        let pivot = idxs[0];
        nodes.push(KdNode {
            pivot,
            axis: 0,
            split: centers[pivot].x as f32,
            left: None,
            right: None,
        });
        return nodes.len() - 1;
    }

    let axis = best_axis_centers(centers, idxs);
    idxs.sort_unstable_by(|&a, &b| {
        coord_center(&centers[a], axis).total_cmp(&coord_center(&centers[b], axis))
    });

    let len = idxs.len();
    let mid = len / 2;
    let pivot = idxs[mid];
    let split = coord_center(&centers[pivot], axis) as f32;

    let (left_idxs, rest) = idxs.split_at_mut(mid);
    let (_, right_idxs) = rest.split_first_mut().unwrap();

    let left = if left_idxs.is_empty() {
        None
    } else {
        Some(build_kd_rec(centers, left_idxs, nodes))
    };
    let right = if right_idxs.is_empty() {
        None
    } else {
        Some(build_kd_rec(centers, right_idxs, nodes))
    };

    nodes.push(KdNode {
        pivot,
        axis: axis as u8,
        split,
        left,
        right,
    });
    nodes.len() - 1
}

impl KdTree {
    fn build(centers: &[Center]) -> Self {
        if centers.is_empty() {
            return Self {
                nodes: Vec::new(),
                root: None,
                centers: Vec::new(),
            };
        }

        let mut idxs: Vec<usize> = (0..centers.len()).collect();
        let mut nodes = Vec::with_capacity(centers.len() * 2 + 1);
        let root = Some(build_kd_rec(centers, &mut idxs, &mut nodes));
        Self {
            nodes,
            root,
            centers: centers.to_vec(),
        }
    }

    fn nearest(&self, p: &Point) -> (usize, f64) {
        if self.nodes.is_empty() {
            return (0, f64::INFINITY);
        }

        let mut best_idx = 0usize;
        let mut best_dist = f64::INFINITY;
        self.nearest_rec(self.root.unwrap(), p, &mut best_idx, &mut best_dist);
        (best_idx, best_dist)
    }

    fn nearest_rec(&self, node_idx: usize, p: &Point, best_idx: &mut usize, best_dist: &mut f64) {
        let node = &self.nodes[node_idx];
        let c = self.centers[node.pivot];
        let d = dist2_point_center(p, &c);
        if d < *best_dist {
            *best_dist = d;
            *best_idx = node.pivot;
        }

        let diff = match node.axis {
            0 => p.x as f64 - node.split as f64,
            1 => p.y as f64 - node.split as f64,
            _ => p.z as f64 - node.split as f64,
        };

        let (near, far) = if diff <= 0.0 {
            (node.left, node.right)
        } else {
            (node.right, node.left)
        };

        if let Some(child) = near {
            self.nearest_rec(child, p, best_idx, best_dist);
        }
        if diff * diff < *best_dist {
            if let Some(child) = far {
                self.nearest_rec(child, p, best_idx, best_dist);
            }
        }
    }
}

fn sample_evenly(points: &[Point], cap: usize) -> Vec<Point> {
    if points.len() <= cap {
        return points.to_vec();
    }

    let mut sample = Vec::with_capacity(cap);
    let n = points.len();
    for i in 0..cap {
        let idx = i * n / cap;
        sample.push(points[idx]);
    }
    sample
}

fn init_centers_from_sample(sample: &[Point], k: usize) -> Vec<Center> {
    if k == 0 {
        return Vec::new();
    }
    if sample.is_empty() {
        return vec![Center::default(); k];
    }

    let tree = build_partition_tree(sample, k.min(sample.len()));
    let leaf_count = tree.leaf_count.max(1);
    let mut sums = vec![Center::default(); leaf_count];
    let mut counts = vec![0usize; leaf_count];

    for p in sample {
        let id = partition_assign(&tree, p) as usize;
        if id < leaf_count {
            sums[id].add_point(*p);
            counts[id] += 1;
        }
    }

    let mut centers = vec![Center::default(); k];
    for i in 0..leaf_count {
        if counts[i] > 0 {
            centers[i] = sums[i].scale(1.0 / counts[i] as f64);
        } else {
            centers[i] = Center::from_point(sample[i % sample.len()]);
        }
    }
    for i in leaf_count..k {
        centers[i] = centers[i % leaf_count];
    }
    centers
}

fn refine_kmeans(points: &[Point], mut centers: Vec<Center>) -> Vec<u32> {
    let n = points.len();
    let k = centers.len();
    let mut labels = vec![0u32; n];
    if k == 0 || n == 0 {
        return labels;
    }

    let mut prev = vec![u32::MAX; n];
    let max_iters = if n <= 1_000_000 { 12 } else { 6 };

    for _ in 0..max_iters {
        let tree = KdTree::build(&centers);
        let mut sums = vec![Center::default(); k];
        let mut counts = vec![0usize; k];
        let mut farthest_idx = 0usize;
        let mut farthest_dist = -1.0f64;
        let mut changed = false;

        for (i, p) in points.iter().enumerate() {
            let (label, dist) = tree.nearest(p);
            labels[i] = label as u32;
            if labels[i] != prev[i] {
                changed = true;
                prev[i] = labels[i];
            }
            counts[label] += 1;
            sums[label].add_point(*p);
            if dist > farthest_dist {
                farthest_dist = dist;
                farthest_idx = i;
            }
        }

        let fallback = points[farthest_idx];
        let mut max_move = 0.0f64;
        for i in 0..k {
            let new_center = if counts[i] > 0 {
                sums[i].scale(1.0 / counts[i] as f64)
            } else {
                Center::from_point(fallback)
            };
            max_move = max_move.max(dist2_center_center(centers[i], new_center));
            centers[i] = new_center;
        }

        if !changed || max_move < 1e-12 {
            break;
        }
    }

    let tree = KdTree::build(&centers);
    for (i, p) in points.iter().enumerate() {
        labels[i] = tree.nearest(p).0 as u32;
    }
    labels
}

fn read_all_points<R: Read>(reader: &mut R, n: usize) -> io::Result<Vec<Point>> {
    let mut points = Vec::with_capacity(n);
    for _ in 0..n {
        points.push(read_point(reader)?);
    }
    Ok(points)
}

fn process_streaming<R: Read, W: Write>(
    reader: &mut R,
    writer: &mut W,
    n: usize,
    k: usize,
) -> io::Result<()> {
    if n == 0 || k == 0 {
        return Ok(());
    }

    let sample_cap = n.min((k.saturating_mul(16)).max(4096).min(100_000));
    let mut sample = Vec::with_capacity(sample_cap);
    for _ in 0..sample_cap {
        sample.push(read_point(reader)?);
    }

    let tree = build_partition_tree(&sample, k.min(sample.len()).max(1));
    for p in &sample {
        let label = partition_assign(&tree, p);
        writer.write_all(&label.to_le_bytes())?;
    }

    for _ in sample_cap..n {
        let p = read_point(reader)?;
        let label = partition_assign(&tree, &p);
        writer.write_all(&label.to_le_bytes())?;
    }

    Ok(())
}

fn process_full<R: Read, W: Write>(
    reader: &mut R,
    writer: &mut W,
    n: usize,
    k: usize,
) -> io::Result<()> {
    if n == 0 || k == 0 {
        return Ok(());
    }

    let points = read_all_points(reader, n)?;
    let sample = sample_evenly(&points, (k.saturating_mul(16)).max(4096).min(100_000));
    let centers = init_centers_from_sample(&sample, k);
    let labels = refine_kmeans(&points, centers);

    for label in labels {
        writer.write_all(&label.to_le_bytes())?;
    }

    Ok(())
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut reader = BufReader::new(stdin.lock());
    let mut writer = BufWriter::new(stdout.lock());

    let n = read_usize_token(&mut reader)?;
    let k = read_usize_token(&mut reader)?;

    const FULL_MODE_LIMIT: usize = 5_000_000;
    if n <= FULL_MODE_LIMIT {
        process_full(&mut reader, &mut writer, n, k)?;
    } else {
        process_streaming(&mut reader, &mut writer, n, k)?;
    }

    writer.flush()?;
    Ok(())
}
