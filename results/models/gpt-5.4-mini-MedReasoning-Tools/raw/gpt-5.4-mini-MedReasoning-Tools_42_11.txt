OpCapability Shader
OpCapability Geometry
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%bool = OpTypeBool

%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_33333334 = OpConstant %float 0.33333334
%float_0_66666669 = OpConstant %float 0.66666669
%float_0_01 = OpConstant %float 0.01

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

%arr_v3 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%ptr_in_v3 = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %arr_v3 ArrayStride 16
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1
OpSource GLSL 450

%main = OpFunction %void None %func
%label = OpLabel
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

%ptr_prev_pos = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%prev_pos = OpLoad %v4float %ptr_prev_pos
%ptr_start_pos = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%start_pos = OpLoad %v4float %ptr_start_pos
%ptr_end_pos = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%end_pos = OpLoad %v4float %ptr_end_pos
%ptr_next_pos = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%next_pos = OpLoad %v4float %ptr_next_pos

%ptr_c1 = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %ptr_c1
%ptr_c2 = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %ptr_c2

%prev_xy = OpVectorShuffle %v2float %prev_pos %prev_pos 0 1
%start_xy = OpVectorShuffle %v2float %start_pos %start_pos 0 1
%end_xy = OpVectorShuffle %v2float %end_pos %end_pos 0 1
%next_xy = OpVectorShuffle %v2float %next_pos %next_pos 0 1

%prev_w = OpCompositeExtract %float %prev_pos 3
%start_w = OpCompositeExtract %float %start_pos 3
%end_w = OpCompositeExtract %float %end_pos 3
%next_w = OpCompositeExtract %float %next_pos 3

%prev_wv = OpCompositeConstruct %v2float %prev_w %prev_w
%start_wv = OpCompositeConstruct %v2float %start_w %start_w
%end_wv = OpCompositeConstruct %v2float %end_w %end_w
%next_wv = OpCompositeConstruct %v2float %next_w %next_w

%prev_ndc = OpFDiv %v2float %prev_xy %prev_wv
%start_ndc = OpFDiv %v2float %start_xy %start_wv
%end_ndc = OpFDiv %v2float %end_xy %end_wv
%next_ndc = OpFDiv %v2float %next_xy %next_wv

%d0 = OpFSub %v2float %start_ndc %prev_ndc
%d1 = OpFSub %v2float %end_ndc %start_ndc
%d2 = OpFSub %v2float %next_ndc %end_ndc
%dir0 = OpExtInst %v2float %glsl450 Normalize %d0
%dir1 = OpExtInst %v2float %glsl450 Normalize %d1
%dir2 = OpExtInst %v2float %glsl450 Normalize %d2

%sumStart = OpFAdd %v2float %dir0 %dir1
%sumEnd = OpFAdd %v2float %dir1 %dir2
%tanStart = OpExtInst %v2float %glsl450 Normalize %sumStart
%tanEnd = OpExtInst %v2float %glsl450 Normalize %sumEnd

%tanStartY = OpCompositeExtract %float %tanStart 1
%tanStartX = OpCompositeExtract %float %tanStart 0
%negStartY = OpFNegate %float %tanStartY
%perpStart = OpCompositeConstruct %v2float %negStartY %tanStartX

%tanEndY = OpCompositeExtract %float %tanEnd 1
%tanEndX = OpCompositeExtract %float %tanEnd 0
%negEndY = OpFNegate %float %tanEndY
%perpEnd = OpCompositeConstruct %v2float %negEndY %tanEndX

%deltaPos = OpFSub %v4float %end_pos %start_pos
%deltaColor = OpFSub %v3float %c2 %c1
%deltaPerp = OpFSub %v2float %perpEnd %perpStart

%start_off = OpVectorTimesScalar %v2float %perpStart %float_0_01
%start_off_clip = OpVectorTimesScalar %v2float %start_off %start_w
%start_minus_xy = OpFSub %v2float %start_xy %start_off_clip
%start_plus_xy = OpFAdd %v2float %start_xy %start_off_clip
%start_z = OpCompositeExtract %float %start_pos 2
%start_minus_pos = OpCompositeConstruct %v4float %start_minus_xy %start_z %start_w
OpStore %out_pos_ptr %start_minus_pos
OpStore %outColor %c1
OpEmitVertex
%start_plus_pos = OpCompositeConstruct %v4float %start_plus_xy %start_z %start_w
OpStore %out_pos_ptr %start_plus_pos
OpStore %outColor %c1
OpEmitVertex

%t1_pos = OpVectorTimesScalar %v4float %deltaPos %float_0_33333334
%base1_pos = OpFAdd %v4float %start_pos %t1_pos
%t1_color = OpVectorTimesScalar %v3float %deltaColor %float_0_33333334
%base1_color = OpFAdd %v3float %c1 %t1_color
%t1_perp = OpVectorTimesScalar %v2float %deltaPerp %float_0_33333334
%base1_perp_mix = OpFAdd %v2float %perpStart %t1_perp
%base1_perp = OpExtInst %v2float %glsl450 Normalize %base1_perp_mix
%base1_off = OpVectorTimesScalar %v2float %base1_perp %float_0_01
%base1_w = OpCompositeExtract %float %base1_pos 3
%base1_off_clip = OpVectorTimesScalar %v2float %base1_off %base1_w
%base1_xy = OpVectorShuffle %v2float %base1_pos %base1_pos 0 1
%base1_minus_xy = OpFSub %v2float %base1_xy %base1_off_clip
%base1_plus_xy = OpFAdd %v2float %base1_xy %base1_off_clip
%base1_z = OpCompositeExtract %float %base1_pos 2
%base1_minus_pos = OpCompositeConstruct %v4float %base1_minus_xy %base1_z %base1_w
OpStore %out_pos_ptr %base1_minus_pos
OpStore %outColor %base1_color
OpEmitVertex
%base1_plus_pos = OpCompositeConstruct %v4float %base1_plus_xy %base1_z %base1_w
OpStore %out_pos_ptr %base1_plus_pos
OpStore %outColor %base1_color
OpEmitVertex
OpEndPrimitive

%t2_pos = OpVectorTimesScalar %v4float %deltaPos %float_0_66666669
%base2_pos = OpFAdd %v4float %start_pos %t2_pos
%t2_color = OpVectorTimesScalar %v3float %deltaColor %float_0_66666669
%base2_color = OpFAdd %v3float %c1 %t2_color
%t2_perp = OpVectorTimesScalar %v2float %deltaPerp %float_0_66666669
%base2_perp_mix = OpFAdd %v2float %perpStart %t2_perp
%base2_perp = OpExtInst %v2float %glsl450 Normalize %base2_perp_mix
%base2_off = OpVectorTimesScalar %v2float %base2_perp %float_0_01
%base2_w = OpCompositeExtract %float %base2_pos 3
%base2_off_clip = OpVectorTimesScalar %v2float %base2_off %base2_w
%base2_xy = OpVectorShuffle %v2float %base2_pos %base2_pos 0 1
%base2_minus_xy = OpFSub %v2float %base2_xy %base2_off_clip
%base2_plus_xy = OpFAdd %v2float %base2_xy %base2_off_clip
%base2_z = OpCompositeExtract %float %base2_pos 2
%base2_minus_pos = OpCompositeConstruct %v4float %base2_minus_xy %base2_z %base2_w
OpStore %out_pos_ptr %base2_minus_pos
OpStore %outColor %base2_color
OpEmitVertex
%base2_plus_pos = OpCompositeConstruct %v4float %base2_plus_xy %base2_z %base2_w
OpStore %out_pos_ptr %base2_plus_pos
OpStore %outColor %base2_color
OpEmitVertex

%end_off = OpVectorTimesScalar %v2float %perpEnd %float_0_01
%end_off_clip = OpVectorTimesScalar %v2float %end_off %end_w
%end_minus_xy = OpFSub %v2float %end_xy %end_off_clip
%end_plus_xy = OpFAdd %v2float %end_xy %end_off_clip
%end_z = OpCompositeExtract %float %end_pos 2
%end_minus_pos = OpCompositeConstruct %v4float %end_minus_xy %end_z %end_w
OpStore %out_pos_ptr %end_minus_pos
OpStore %outColor %c2
OpEmitVertex
%end_plus_pos = OpCompositeConstruct %v4float %end_plus_xy %end_z %end_w
OpStore %out_pos_ptr %end_plus_pos
OpStore %outColor %c2
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
