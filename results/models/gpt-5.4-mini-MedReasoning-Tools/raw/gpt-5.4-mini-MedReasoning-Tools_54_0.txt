from __future__ import annotations

import math
from statistics import median


_STATE = {
    "last_ias": None,
    "last_radio_alt": None,
    "last_time": None,
    "flaps": 0.0,
}


def _clip(x, lo, hi):
    return max(lo, min(hi, x))


def _wrap180(deg):
    if deg is None:
        return 0.0
    return ((deg + 180.0) % 360.0) - 180.0


def _num(x, default=0.0):
    try:
        if x is None:
            return default
        if isinstance(x, bool):
            return 1.0 if x else 0.0
        return float(x)
    except Exception:
        return default


def _bool(x):
    return bool(x) if x is not None else False


def _vote(values):
    vals = [v for v in values if v is not None and not (isinstance(v, float) and math.isnan(v))]
    if not vals:
        return None
    try:
        return float(median(vals))
    except Exception:
        return float(sum(vals) / len(vals))


def _sensor_value(state, base_name, suffix, count=3):
    vals = []
    for i in range(1, count + 1):
        v = state.get(f"{base_name}_{i}{suffix}")
        valid = state.get(f"{base_name}_{i}_valid")
        if valid is None or bool(valid):
            vals.append(_num(v, None))
    if not vals:
        vals = [_num(state.get(f"{base_name}_{i}{suffix}"), None) for i in range(1, count + 1)]
    return _vote(vals)


def _pick_engine_throttle(base, error, running, fire):
    if fire or not running:
        return 0.0
    return _clip(base + 0.02 * error, 0.0, 1.0)


def _approach_flaps(ias, radio_alt, engines_ok):
    # Conservative, speed-driven flap schedule that still allows a clean glide when an engine is lost.
    if radio_alt is not None and radio_alt < 20:
        return 40.0
    if radio_alt is not None and radio_alt < 60:
        return 30.0
    if radio_alt is not None and radio_alt < 200:
        return 30.0 if ias is None or ias <= 150 else 20.0
    if radio_alt is not None and radio_alt < 600:
        if ias is not None and ias < 155:
            return 20.0
        return 10.0
    if ias is not None:
        if ias < 145:
            return 20.0
        if ias < 165:
            return 10.0
    return 5.0 if engines_ok else 10.0


def autopilot_step(state: dict, dt: float) -> dict:
    global _STATE

    # Redundant sensor fusion.
    loc = _sensor_value(state, "localizer", "_dots")
    gs = _sensor_value(state, "glideslope", "_dots")
    ias = _sensor_value(state, "ias", "_kt")
    mach = _sensor_value(state, "mach", "")
    aoa = _sensor_value(state, "aoa", "_deg")
    baro_alt = _sensor_value(state, "baro_alt", "_ft")
    radio_alt = _num(state.get("radio_alt_ft"), None)
    if radio_alt is None:
        radio_alt = _num(state.get("radio_alt"), None)
    gps_alt = _vote([_num(state.get(f"gps_{i}_alt_ft"), None) for i in range(1, 4)])
    pitch = _sensor_value(state, "irs", "_pitch_deg")
    roll = _sensor_value(state, "irs", "_roll_deg")
    track = _sensor_value(state, "irs", "_track_deg")
    heading = _sensor_value(state, "irs", "_heading_deg")
    vs = _sensor_value(state, "irs", "_vs_fpm")
    groundspeed = _num(state.get("groundspeed_kt"), None)
    if groundspeed is None:
        groundspeed = _vote([_num(state.get(f"gps_{i}_gs_kt"), None) for i in range(1, 4)])
    target_ias = _num(state.get("target_ias_kt"), 137.0)
    target_vref = _num(state.get("target_vref_kt"), target_ias)
    target_loc = _num(state.get("target_loc_dots"), 0.0)
    target_gs = _num(state.get("target_gs_dots"), 0.0)
    target_hdg = _num(state.get("target_hdg_deg"), heading if heading is not None else 0.0)
    target_vs = _num(state.get("target_vs_fpm"), -700.0)
    below_dh = _bool(state.get("below_decision_height"))
    on_ground = _bool(state.get("on_ground"))
    wow = _bool(state.get("weight_on_wheels"))

    engine_1_running = _bool(state.get("engine_1_running"))
    engine_2_running = _bool(state.get("engine_2_running"))
    engine_1_fire = _bool(state.get("engine_1_fire"))
    engine_2_fire = _bool(state.get("engine_2_fire"))
    engine_1_thrust = _num(state.get("engine_1_thrust_pct"), 0.0)
    engine_2_thrust = _num(state.get("engine_2_thrust_pct"), 0.0)
    engine_1_failed = engine_1_fire or not engine_1_running or engine_1_thrust < 2.0
    engine_2_failed = engine_2_fire or not engine_2_running or engine_2_thrust < 2.0
    engines_ok = not engine_1_failed and not engine_2_failed
    single_engine = engine_1_failed ^ engine_2_failed
    both_engines_failed = engine_1_failed and engine_2_failed

    if radio_alt is None and baro_alt is not None and gps_alt is not None:
        radio_alt = max(0.0, baro_alt - gps_alt)
    if radio_alt is None:
        radio_alt = 9999.0

    if ias is None:
        ias = target_ias
    if pitch is None:
        pitch = 0.0
    if roll is None:
        roll = 0.0
    if vs is None:
        vs = 0.0
    if track is None:
        track = heading if heading is not None else target_hdg
    if heading is None:
        heading = target_hdg
    if loc is None:
        loc = 0.0
    if gs is None:
        gs = 0.0
    if aoa is None:
        aoa = 6.0

    heading_err = _wrap180(heading - target_hdg)
    track_err = _wrap180(track - target_hdg)
    speed_err = target_vref - ias

    if _STATE["last_ias"] is None:
        _STATE["last_ias"] = ias
    if _STATE["last_radio_alt"] is None:
        _STATE["last_radio_alt"] = radio_alt

    ias_drop = _STATE["last_ias"] - ias
    sink_rate = -vs
    severe_windshear = radio_alt < 600 and ias_drop > 10.0 and sink_rate > 700.0
    unstable_lateral = abs(loc - target_loc) > 2.2 or abs(roll) > 25.0
    unstable_vertical = abs(gs - target_gs) > 1.6 or sink_rate > 2200.0 or vs > 900.0
    speed_unstable = ias > target_vref + 35.0 or ias < max(70.0, target_vref - 25.0)
    goaround = False

    if not on_ground and not wow:
        if radio_alt > 1000.0:
            if unstable_lateral or unstable_vertical or speed_unstable or severe_windshear:
                goaround = True
        elif radio_alt > 200.0:
            if below_dh and (unstable_lateral or unstable_vertical or speed_unstable):
                goaround = True
            elif severe_windshear and ias < target_vref + 5.0:
                goaround = True
        elif radio_alt > 50.0:
            if severe_windshear or sink_rate > 3000.0:
                goaround = True

    _STATE["last_ias"] = ias
    _STATE["last_radio_alt"] = radio_alt

    # Default outputs.
    out = {
        "elevator": 0.0,
        "aileron": 0.0,
        "rudder": 0.0,
        "throttle_1": 0.0,
        "throttle_2": 0.0,
        "elevator_trim": 0.0,
        "flaps": _STATE["flaps"],
        "gear": 1.0 if radio_alt < 2500.0 else 0.0,
        "speedbrake": 0.0,
        "spoiler": 0.0,
        "autobrake": 0.0,
        "reverse_1": 0.0,
        "reverse_2": 0.0,
    }

    if goaround and not wow:
        # Go-around: TOGA, positive pitch, clean up speedbrake, keep the airplane coordinated.
        out.update(
            {
                "elevator": 0.65,
                "aileron": _clip(-0.25 * loc - 0.04 * roll, -1.0, 1.0),
                "rudder": _clip(-0.06 * track_err - 0.02 * roll, -1.0, 1.0),
                "throttle_1": 1.0,
                "throttle_2": 1.0,
                "elevator_trim": 0.15,
                "flaps": max(_STATE["flaps"], 15.0),
                "gear": 1.0,
                "speedbrake": 0.0,
                "spoiler": 0.0,
                "autobrake": 0.0,
                "reverse_1": 0.0,
                "reverse_2": 0.0,
            }
        )
        _STATE["flaps"] = float(out["flaps"])
        return out

    # Rollout and landing protection.
    if wow or on_ground or radio_alt < 3.0:
        out["spoiler"] = 1.0
        out["speedbrake"] = 1.0
        out["gear"] = 1.0
        out["flaps"] = max(_STATE["flaps"], 30.0)
        out["autobrake"] = 0.8 if ias > 60.0 or groundspeed > 60.0 else 0.5
        if groundspeed > 30.0:
            out["reverse_1"] = 1.0 if not engine_1_failed else 0.0
            out["reverse_2"] = 1.0 if not engine_2_failed else 0.0
        out["elevator"] = _clip(0.15 + 0.002 * (70.0 - groundspeed), -0.2, 0.3)
        out["aileron"] = _clip(-0.15 * loc - 0.03 * roll, -0.4, 0.4)
        out["rudder"] = _clip(-0.12 * track_err - 0.02 * roll, -0.6, 0.6)
        out["throttle_1"] = 0.0
        out["throttle_2"] = 0.0
        _STATE["flaps"] = float(out["flaps"])
        return out

    # Approach/flare control.
    if radio_alt < 50.0:
        # Flare: arrest sink, protect touchdown rate, and bring thrust to idle.
        flare_gain = 0.012 * max(0.0, 50.0 - radio_alt)
        target_flare_vs = -max(80.0, radio_alt * 12.0)
        elev = 0.18 + flare_gain + 0.0008 * (target_flare_vs - vs) + 0.01 * speed_err
        throttle_base = 0.12 if both_engines_failed else 0.05
        out["elevator"] = _clip(elev, -0.05, 0.65)
        out["aileron"] = _clip(-0.30 * (loc - target_loc) - 0.03 * roll, -1.0, 1.0)
        out["rudder"] = _clip(-0.07 * track_err - 0.02 * roll, -1.0, 1.0)
        out["throttle_1"] = 0.0 if engine_1_failed else _clip(throttle_base, 0.0, 0.2)
        out["throttle_2"] = 0.0 if engine_2_failed else _clip(throttle_base, 0.0, 0.2)
        out["elevator_trim"] = _clip(0.05 + 0.002 * speed_err, -0.2, 0.2)
        out["flaps"] = 40.0 if ias <= target_vref + 15.0 or both_engines_failed else max(_STATE["flaps"], 30.0)
        out["gear"] = 1.0
        out["speedbrake"] = 0.0
        out["spoiler"] = 0.0
        out["autobrake"] = 0.0
        _STATE["flaps"] = float(out["flaps"])
        return out

    # Stable approach, with speed and path coupled to touchdown planning.
    flap_target = _approach_flaps(ias, radio_alt, engines_ok)
    if single_engine:
        flap_target = max(flap_target, 20.0)
    if both_engines_failed:
        flap_target = max(flap_target, 10.0 if radio_alt > 500.0 else 20.0)

    # Smooth flap changes to avoid abrupt asymmetry or sink.
    current_flaps = _STATE["flaps"]
    if flap_target > current_flaps:
        current_flaps = min(flap_target, current_flaps + 10.0)
    else:
        current_flaps = max(flap_target, current_flaps - 10.0)
    _STATE["flaps"] = current_flaps

    # Keep the airplane slightly fast on single-engine or contaminated-runway-like conditions.
    approach_speed_bias = 0.0
    if single_engine:
        approach_speed_bias = 8.0
    elif both_engines_failed:
        approach_speed_bias = 5.0

    cmd_ias = target_vref + approach_speed_bias
    speed_err = cmd_ias - ias

    # Longitudinal control: glideslope + vertical speed + speed protection.
    gs_err = target_gs - gs
    vs_cmd = target_vs
    if radio_alt > 1000.0:
        vs_cmd = -700.0
    elif radio_alt > 200.0:
        vs_cmd = -550.0
    elif radio_alt > 50.0:
        vs_cmd = -250.0
    elev = 0.55 * gs_err + 0.00045 * (vs_cmd - vs) + 0.02 * speed_err - 0.01 * pitch
    elev += 0.03 * max(0.0, aoa - 10.0)
    if single_engine:
        elev += 0.02
    if both_engines_failed:
        elev += 0.03
    out["elevator"] = _clip(elev, -0.6, 0.85)
    out["elevator_trim"] = _clip(0.05 + 0.0015 * speed_err - 0.0002 * vs, -0.5, 0.5)

    # Lateral control: localizer, roll damping, and mild crab/track correction.
    ail = -0.45 * (loc - target_loc) - 0.04 * roll - 0.005 * track_err
    rud = -0.10 * track_err - 0.03 * roll - 0.02 * (loc - target_loc)
    if single_engine:
        failed_left = engine_1_failed and not engine_2_failed
        rud += 0.08 if failed_left else -0.08
        ail += 0.03 if failed_left else -0.03
    if abs(heading_err) > 20.0:
        rud += _clip(-0.01 * heading_err, -0.4, 0.4)
    out["aileron"] = _clip(ail, -1.0, 1.0)
    out["rudder"] = _clip(rud, -1.0, 1.0)

    # Thrust management.
    throttle_base = 0.34 + 0.02 * speed_err + 0.00015 * max(0.0, -vs)
    throttle_base = _clip(throttle_base, 0.0, 1.0)
    if radio_alt < 120.0:
        throttle_base = min(throttle_base, 0.22)
    if radio_alt < 50.0:
        throttle_base = min(throttle_base, 0.12)
    if both_engines_failed:
        throttle_base = 0.0

    out["throttle_1"] = _pick_engine_throttle(throttle_base, speed_err, engine_1_running, engine_1_fire)
    out["throttle_2"] = _pick_engine_throttle(throttle_base, speed_err, engine_2_running, engine_2_fire)

    # If one engine is dead, keep the live engine a touch more energetic to preserve approach speed.
    if single_engine:
        if engine_1_failed and not engine_2_failed:
            out["throttle_2"] = _clip(out["throttle_2"] + 0.08, 0.0, 1.0)
            out["throttle_1"] = 0.0
        elif engine_2_failed and not engine_1_failed:
            out["throttle_1"] = _clip(out["throttle_1"] + 0.08, 0.0, 1.0)
            out["throttle_2"] = 0.0

    # Configuration.
    out["gear"] = 1.0 if radio_alt < 2500.0 or ias < target_vref + 40.0 else 0.0
    out["speedbrake"] = 0.0
    out["spoiler"] = 0.0
    out["autobrake"] = 0.0
    out["flaps"] = current_flaps

    # Slightly stronger touchdown plan on wet/icy or heavy arrivals can be approximated from speed margin.
    if radio_alt < 200.0:
        if ias > target_vref + 10.0:
            out["autobrake"] = 0.4
        elif ias < target_vref + 5.0:
            out["autobrake"] = 0.6

    return out
