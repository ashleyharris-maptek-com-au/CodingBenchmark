use std::cmp::Reverse;
use std::io::{self, Read};

fn cut_of(mask: u128, adj: &[u128], n: usize) -> i64 {
    let all_mask = if n == 128 { u128::MAX } else { (1u128 << n) - 1 };
    let comp = all_mask ^ mask;
    let mut cut = 0i64;
    let mut bits = mask;
    while bits != 0 {
        let v = bits.trailing_zeros() as usize;
        bits &= bits - 1;
        cut += (adj[v] & comp).count_ones() as i64;
    }
    cut
}

fn greedy_solution(n: usize, adj: &[u128], deg: &[u32]) -> (i64, u128) {
    let half = n / 2;
    let mut side = vec![0u8; n]; // 1 = A, 2 = B
    let mut order: Vec<usize> = (1..n).collect();
    order.sort_by_key(|&v| Reverse(deg[v]));

    side[0] = 1;
    let mut a_count = 1usize;
    let mut assigned = 1usize;

    for &v in &order {
        let remaining = n - assigned;
        let need_a = half - a_count;

        let mut cost_a = 0i64;
        let mut cost_b = 0i64;
        for u in 0..n {
            if side[u] != 0 && ((adj[v] >> u) & 1) != 0 {
                if side[u] == 1 {
                    cost_b += 1;
                } else {
                    cost_a += 1;
                }
            }
        }

        if need_a == remaining {
            side[v] = 1;
            a_count += 1;
        } else if need_a == 0 {
            side[v] = 2;
        } else if cost_a <= cost_b {
            side[v] = 1;
            a_count += 1;
        } else {
            side[v] = 2;
        }
        assigned += 1;
    }

    let mut mask = 0u128;
    for i in 0..n {
        if side[i] == 1 {
            mask |= 1u128 << i;
        }
    }
    let cut = cut_of(mask, adj, n);
    (cut, mask)
}

fn recurse_with_assignment(
    side_a: bool,
    v: usize,
    bit: u128,
    n: usize,
    half: usize,
    adj: &[u128],
    deg: &[u32],
    unassigned_after: u128,
    count_a: usize,
    cur_cut: i64,
    cur_mask: u128,
    to_a: &mut [u32],
    to_b: &mut [u32],
    best_cut: &mut i64,
    best_mask: &mut u128,
) {
    let mut changed = Vec::new();
    let mut bits = adj[v] & unassigned_after;
    while bits != 0 {
        let u = bits.trailing_zeros() as usize;
        bits &= bits - 1;
        if side_a {
            to_a[u] += 1;
        } else {
            to_b[u] += 1;
        }
        changed.push(u);
    }

    let added = if side_a {
        to_b[v] as i64
    } else {
        to_a[v] as i64
    };
    let new_cut = cur_cut + added;
    let new_mask = if side_a { cur_mask | bit } else { cur_mask };
    let new_count_a = if side_a { count_a + 1 } else { count_a };

    dfs(
        n,
        half,
        adj,
        deg,
        unassigned_after,
        new_count_a,
        new_cut,
        new_mask,
        to_a,
        to_b,
        best_cut,
        best_mask,
    );

    for u in changed {
        if side_a {
            to_a[u] -= 1;
        } else {
            to_b[u] -= 1;
        }
    }
}

fn dfs(
    n: usize,
    half: usize,
    adj: &[u128],
    deg: &[u32],
    unassigned: u128,
    count_a: usize,
    cur_cut: i64,
    cur_mask: u128,
    to_a: &mut [u32],
    to_b: &mut [u32],
    best_cut: &mut i64,
    best_mask: &mut u128,
) {
    let remaining = unassigned.count_ones() as usize;
    if count_a > half || count_a + remaining < half {
        return;
    }
    if remaining == 0 {
        if cur_cut < *best_cut {
            *best_cut = cur_cut;
            *best_mask = cur_mask;
        }
        return;
    }

    let need_a = half - count_a;

    let mut base = 0i64;
    let mut diffs = Vec::with_capacity(remaining);
    let mut pivot = None;
    let mut pivot_score = i64::MIN;
    let mut bits = unassigned;
    while bits != 0 {
        let v = bits.trailing_zeros() as usize;
        bits &= bits - 1;
        base += to_a[v] as i64;
        diffs.push(to_b[v] as i64 - to_a[v] as i64);
        let score = ((to_a[v] + to_b[v]) as i64) * 1_000 + deg[v] as i64;
        if score > pivot_score {
            pivot_score = score;
            pivot = Some(v);
        }
    }

    diffs.sort_unstable();
    let mut lb = cur_cut + base;
    for i in 0..need_a {
        lb += diffs[i];
    }
    if lb >= *best_cut {
        return;
    }

    let v = pivot.unwrap();
    let bit = 1u128 << v;
    let unassigned_after = unassigned & !bit;

    let cost_a = to_b[v] as i64;
    let cost_b = to_a[v] as i64;
    let try_a_first = cost_a <= cost_b;

    if need_a == remaining {
        recurse_with_assignment(
            true,
            v,
            bit,
            n,
            half,
            adj,
            deg,
            unassigned_after,
            count_a,
            cur_cut,
            cur_mask,
            to_a,
            to_b,
            best_cut,
            best_mask,
        );
    } else if need_a == 0 {
        recurse_with_assignment(
            false,
            v,
            bit,
            n,
            half,
            adj,
            deg,
            unassigned_after,
            count_a,
            cur_cut,
            cur_mask,
            to_a,
            to_b,
            best_cut,
            best_mask,
        );
    } else if try_a_first {
        recurse_with_assignment(
            true,
            v,
            bit,
            n,
            half,
            adj,
            deg,
            unassigned_after,
            count_a,
            cur_cut,
            cur_mask,
            to_a,
            to_b,
            best_cut,
            best_mask,
        );
        if cur_cut < *best_cut {
            recurse_with_assignment(
                false,
                v,
                bit,
                n,
                half,
                adj,
                deg,
                unassigned_after,
                count_a,
                cur_cut,
                cur_mask,
                to_a,
                to_b,
                best_cut,
                best_mask,
            );
        }
    } else {
        recurse_with_assignment(
            false,
            v,
            bit,
            n,
            half,
            adj,
            deg,
            unassigned_after,
            count_a,
            cur_cut,
            cur_mask,
            to_a,
            to_b,
            best_cut,
            best_mask,
        );
        if cur_cut < *best_cut {
            recurse_with_assignment(
                true,
                v,
                bit,
                n,
                half,
                adj,
                deg,
                unassigned_after,
                count_a,
                cur_cut,
                cur_mask,
                to_a,
                to_b,
                best_cut,
                best_mask,
            );
        }
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj = vec![0u128; n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            continue;
        }
        adj[u] |= 1u128 << v;
        adj[v] |= 1u128 << u;
    }

    let deg: Vec<u32> = adj.iter().map(|&x| x.count_ones()).collect();
    let half = n / 2;

    let (mut best_cut, mut best_mask) = greedy_solution(n, &adj, &deg);

    let mut to_a = vec![0u32; n];
    let mut to_b = vec![0u32; n];
    let mut unassigned = if n == 128 { u128::MAX } else { (1u128 << n) - 1 };
    let fixed = 1u128; // vertex 0 in A to break symmetry
    unassigned &= !fixed;

    let mut bits = adj[0] & unassigned;
    while bits != 0 {
        let v = bits.trailing_zeros() as usize;
        bits &= bits - 1;
        to_a[v] += 1;
    }

    dfs(
        n,
        half,
        &adj,
        &deg,
        unassigned,
        1,
        0,
        fixed,
        &mut to_a,
        &mut to_b,
        &mut best_cut,
        &mut best_mask,
    );

    let mut out = String::new();
    out.push_str(&format!("{}\n", best_cut));

    let mut first = true;
    for v in 0..n {
        if ((best_mask >> v) & 1) != 0 {
            if !first {
                out.push(' ');
            }
            first = false;
            out.push_str(&v.to_string());
        }
    }
    out.push('\n');
    print!("{}", out);
}
