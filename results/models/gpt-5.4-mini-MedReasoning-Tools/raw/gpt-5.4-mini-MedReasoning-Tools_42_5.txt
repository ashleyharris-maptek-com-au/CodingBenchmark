OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1
OpDecorate %arr_v3 ArrayStride 16
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%void = OpTypeVoid
%funct = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_015 = OpConstant %float 0.015
%float_05 = OpConstant %float 0.05
%float_075 = OpConstant %float 0.75
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_pv_arr = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_pv_arr Input
%gl_PerVertex_out = OpTypeStruct %v4float
%arr_pv_out = OpTypeArray %gl_PerVertex_out %uint_2
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_pv Output
%arr_v3 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v3 = OpTypePointer Output %v3float
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %funct
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p1_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_1
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%p0 = OpLoad %v3float %p0_ptr
%p1 = OpLoad %v3float %p1_ptr
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

%dir = OpFSub %v3float %p1 %p0
%dirx = OpCompositeExtract %float %dir 0
%diry = OpCompositeExtract %float %dir 1
%negdiry = OpFNegate %float %diry
%perp_base = OpCompositeConstruct %v2float %negdiry %dirx
%perp = OpExtInst %v2float %glsl Normalize %perp_base

%delta = OpFSub %v3float %p1 %p0
%mid_delta = OpVectorTimesScalar %v3float %delta %float_075
%mid = OpFAdd %v3float %p0 %mid_delta

%shaft_off_xy = OpVectorTimesScalar %v2float %perp %float_015
%shaft_off_x = OpCompositeExtract %float %shaft_off_xy 0
%shaft_off_y = OpCompositeExtract %float %shaft_off_xy 1
%shaft_off = OpCompositeConstruct %v3float %shaft_off_x %shaft_off_y %float_0

%shaft_p0_plus = OpFAdd %v3float %p0 %shaft_off
%shaft_p0_minus = OpFSub %v3float %p0 %shaft_off
%shaft_mid_plus = OpFAdd %v3float %mid %shaft_off
%shaft_mid_minus = OpFSub %v3float %mid %shaft_off

%tip0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%tip0_x = OpCompositeExtract %float %shaft_p0_plus 0
%tip0_y = OpCompositeExtract %float %shaft_p0_plus 1
%tip0_z = OpCompositeExtract %float %shaft_p0_plus 2
%tip0_v4 = OpCompositeConstruct %v4float %tip0_x %tip0_y %tip0_z %float_1
OpStore %tip0 %tip0_v4
OpStore %outColor %c0
OpEmitVertex

%tip1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%tip1_x = OpCompositeExtract %float %shaft_p0_minus 0
%tip1_y = OpCompositeExtract %float %shaft_p0_minus 1
%tip1_z = OpCompositeExtract %float %shaft_p0_minus 2
%tip1_v4 = OpCompositeConstruct %v4float %tip1_x %tip1_y %tip1_z %float_1
OpStore %tip1 %tip1_v4
OpStore %outColor %c0
OpEmitVertex

%tip2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%tip2_x = OpCompositeExtract %float %shaft_mid_plus 0
%tip2_y = OpCompositeExtract %float %shaft_mid_plus 1
%tip2_z = OpCompositeExtract %float %shaft_mid_plus 2
%tip2_v4 = OpCompositeConstruct %v4float %tip2_x %tip2_y %tip2_z %float_1
OpStore %tip2 %tip2_v4
OpStore %outColor %c0
OpEmitVertex

%tip3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%tip3_x = OpCompositeExtract %float %shaft_mid_minus 0
%tip3_y = OpCompositeExtract %float %shaft_mid_minus 1
%tip3_z = OpCompositeExtract %float %shaft_mid_minus 2
%tip3_v4 = OpCompositeConstruct %v4float %tip3_x %tip3_y %tip3_z %float_1
OpStore %tip3 %tip3_v4
OpStore %outColor %c0
OpEmitVertex
OpEndPrimitive

%arrow_off_xy = OpVectorTimesScalar %v2float %perp %float_05
%arrow_off_x = OpCompositeExtract %float %arrow_off_xy 0
%arrow_off_y = OpCompositeExtract %float %arrow_off_xy 1
%arrow_off = OpCompositeConstruct %v3float %arrow_off_x %arrow_off_y %float_0

%arrow_plus = OpFAdd %v3float %mid %arrow_off
%arrow_minus = OpFSub %v3float %mid %arrow_off

%a0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%a0_x = OpCompositeExtract %float %arrow_plus 0
%a0_y = OpCompositeExtract %float %arrow_plus 1
%a0_z = OpCompositeExtract %float %arrow_plus 2
%a0_v4 = OpCompositeConstruct %v4float %a0_x %a0_y %a0_z %float_1
OpStore %a0 %a0_v4
OpStore %outColor %c1
OpEmitVertex

%a1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%a1_x = OpCompositeExtract %float %arrow_minus 0
%a1_y = OpCompositeExtract %float %arrow_minus 1
%a1_z = OpCompositeExtract %float %arrow_minus 2
%a1_v4 = OpCompositeConstruct %v4float %a1_x %a1_y %a1_z %float_1
OpStore %a1 %a1_v4
OpStore %outColor %c1
OpEmitVertex

%a2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%a2_x = OpCompositeExtract %float %p1 0
%a2_y = OpCompositeExtract %float %p1 1
%a2_z = OpCompositeExtract %float %p1 2
%a2_v4 = OpCompositeConstruct %v4float %a2_x %a2_y %a2_z %float_1
OpStore %a2 %a2_v4
OpStore %outColor %c1
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
