using System;
using System.Collections.Generic;
using System.IO;
using System.Numerics;
using System.Text;

public class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        var adj = new List<int>[n];
        var radj = new List<int>[n];
        var selfLoop = new bool[n];
        for (int i = 0; i < n; i++)
        {
            adj[i] = new List<int>();
            radj[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            adj[u].Add(v);
            radj[v].Add(u);
            if (u == v) selfLoop[u] = true;
        }

        var components = GetSccs(n, adj, radj);
        var answer = new List<int>();

        foreach (var comp in components)
        {
            if (comp.Count == 1)
            {
                int v = comp[0];
                if (selfLoop[v])
                    answer.Add(v);
                continue;
            }

            var solver = new ComponentSolver(comp, adj, radj);
            answer.AddRange(solver.Solve());
        }

        answer.Sort();
        var sb = new StringBuilder();
        sb.AppendLine(answer.Count.ToString());
        if (answer.Count > 0)
            sb.AppendLine(string.Join(" ", answer));
        else
            sb.AppendLine();
        Console.Write(sb.ToString());
    }

    private static List<List<int>> GetSccs(int n, List<int>[] adj, List<int>[] radj)
    {
        var visited = new bool[n];
        var order = new List<int>(n);
        var stack = new Stack<(int v, int idx)>();

        for (int s = 0; s < n; s++)
        {
            if (visited[s]) continue;
            stack.Push((s, 0));
            visited[s] = true;
            while (stack.Count > 0)
            {
                var (v, idx) = stack.Pop();
                var list = adj[v];
                while (idx < list.Count && visited[list[idx]]) idx++;
                if (idx < list.Count)
                {
                    stack.Push((v, idx + 1));
                    int to = list[idx];
                    if (!visited[to])
                    {
                        visited[to] = true;
                        stack.Push((to, 0));
                    }
                }
                else
                {
                    order.Add(v);
                }
            }
        }

        Array.Fill(visited, false);
        var comps = new List<List<int>>();
        stack.Clear();

        for (int oi = order.Count - 1; oi >= 0; oi--)
        {
            int s = order[oi];
            if (visited[s]) continue;
            var comp = new List<int>();
            stack.Push((s, 0));
            visited[s] = true;
            while (stack.Count > 0)
            {
                var (v, idx) = stack.Pop();
                var list = radj[v];
                while (idx < list.Count && visited[list[idx]]) idx++;
                if (idx < list.Count)
                {
                    stack.Push((v, idx + 1));
                    int to = list[idx];
                    if (!visited[to])
                    {
                        visited[to] = true;
                        stack.Push((to, 0));
                    }
                }
                else
                {
                    comp.Add(v);
                }
            }
            comps.Add(comp);
        }

        return comps;
    }

    private sealed class ComponentSolver
    {
        private readonly int _n;
        private readonly int[] _global;
        private readonly int[][] _out;
        private readonly int[][] _in;
        private readonly ulong[] _outMask;
        private readonly ulong[] _inMask;
        private readonly bool[] _selfLoop;
        private readonly bool _useMask;
        private readonly Dictionary<ulong, ulong> _memoMask;

        public ComponentSolver(List<int> vertices, List<int>[] adj, List<int>[] radj)
        {
            _n = vertices.Count;
            _global = vertices.ToArray();
            var localIndex = new Dictionary<int, int>(_n);
            for (int i = 0; i < _n; i++) localIndex[_global[i]] = i;

            var outLists = new List<int>[_n];
            var inLists = new List<int>[_n];
            for (int i = 0; i < _n; i++)
            {
                outLists[i] = new List<int>();
                inLists[i] = new List<int>();
            }

            for (int i = 0; i < _n; i++)
            {
                int gv = _global[i];
                foreach (int toG in adj[gv])
                {
                    if (localIndex.TryGetValue(toG, out int to))
                    {
                        outLists[i].Add(to);
                        inLists[to].Add(i);
                    }
                }
            }

            _out = new int[_n][];
            _in = new int[_n][];
            for (int i = 0; i < _n; i++)
            {
                _out[i] = outLists[i].ToArray();
                _in[i] = inLists[i].ToArray();
            }

            _selfLoop = new bool[_n];
            if (_n <= 63)
            {
                _outMask = new ulong[_n];
                _inMask = new ulong[_n];
                for (int i = 0; i < _n; i++)
                {
                    ulong om = 0, im = 0;
                    foreach (int to in _out[i]) om |= 1UL << to;
                    foreach (int from in _in[i]) im |= 1UL << from;
                    _outMask[i] = om;
                    _inMask[i] = im;
                    _selfLoop[i] = ((om >> i) & 1UL) != 0;
                }
                _useMask = true;
                _memoMask = new Dictionary<ulong, ulong>();
            }
            else
            {
                for (int i = 0; i < _n; i++)
                {
                    foreach (int to in _out[i])
                        if (to == i) _selfLoop[i] = true;
                }
                _useMask = false;
                _memoMask = null;
            }
        }

        public List<int> Solve()
        {
            if (_n == 0) return new List<int>();
            if (_useMask)
            {
                ulong full = _n == 64 ? ulong.MaxValue : ((1UL << _n) - 1UL);
                ulong sol = SolveMask(full);
                return MaskToGlobalList(sol);
            }
            else
            {
                var active = new bool[_n];
                for (int i = 0; i < _n; i++) active[i] = true;
                var solLocal = SolveArray(active);
                var res = new List<int>(solLocal.Count);
                foreach (int i in solLocal) res.Add(_global[i]);
                return res;
            }
        }

        private List<int> MaskToGlobalList(ulong mask)
        {
            var res = new List<int>();
            while (mask != 0)
            {
                int v = BitOperations.TrailingZeroCount(mask);
                mask &= mask - 1;
                res.Add(_global[v]);
            }
            return res;
        }

        private ulong SolveMask(ulong mask)
        {
            if (mask == 0) return 0;
            if (_memoMask.TryGetValue(mask, out var cached)) return cached;

            ulong original = mask;
            ulong forced = 0;

            while (true)
            {
                bool changed = false;
                ulong temp = mask;
                while (temp != 0)
                {
                    int v = BitOperations.TrailingZeroCount(temp);
                    temp &= temp - 1;

                    if (((mask >> v) & 1UL) == 0) continue;

                    if (_selfLoop[v])
                    {
                        forced |= 1UL << v;
                        mask &= ~(1UL << v);
                        changed = true;
                        continue;
                    }

                    int indeg = BitOperations.PopCount(_inMask[v] & mask);
                    if (indeg == 0)
                    {
                        mask &= ~(1UL << v);
                        changed = true;
                        continue;
                    }

                    int outdeg = BitOperations.PopCount(_outMask[v] & mask);
                    if (outdeg == 0)
                    {
                        mask &= ~(1UL << v);
                        changed = true;
                    }
                }
                if (!changed) break;
            }

            if (mask == 0)
            {
                _memoMask[original] = forced;
                _memoMask[mask] = forced;
                return forced;
            }

            var sccs = GetSccMasks(mask);
            if (sccs.Count > 1)
            {
                ulong sol = forced;
                foreach (ulong comp in sccs)
                    sol |= SolveMask(comp);
                _memoMask[original] = sol;
                _memoMask[mask] = sol;
                return sol;
            }

            int count = BitOperations.PopCount(mask);
            if (count <= 14)
            {
                ulong brute = BruteForceMask(mask);
                ulong sol = forced | brute;
                _memoMask[original] = sol;
                _memoMask[mask] = sol;
                return sol;
            }

            ulong best = GreedyUpperBoundMask(mask);
            int bestCount = BitOperations.PopCount(best) + BitOperations.PopCount(forced);
            int lowerBound = BitOperations.PopCount(forced) + 1;
            if (bestCount == lowerBound)
            {
                ulong sol = forced | best;
                _memoMask[original] = sol;
                _memoMask[mask] = sol;
                return sol;
            }

            ulong cycle = FindCycleMask(mask);
            if (cycle == 0)
            {
                ulong sol = forced;
                _memoMask[original] = sol;
                _memoMask[mask] = sol;
                return sol;
            }

            var cycleVerts = new List<int>();
            ulong ctemp = cycle;
            while (ctemp != 0)
            {
                int v = BitOperations.TrailingZeroCount(ctemp);
                ctemp &= ctemp - 1;
                cycleVerts.Add(v);
            }
            cycleVerts.Sort((a, b) =>
            {
                int da = BitOperations.PopCount(_inMask[a] & mask) + BitOperations.PopCount(_outMask[a] & mask);
                int db = BitOperations.PopCount(_inMask[b] & mask) + BitOperations.PopCount(_outMask[b] & mask);
                int cmp = db.CompareTo(da);
                if (cmp != 0) return cmp;
                return a.CompareTo(b);
            });

            foreach (int v in cycleVerts)
            {
                ulong child = SolveMask(mask & ~(1UL << v));
                ulong cand = child | (1UL << v);
                int candCount = BitOperations.PopCount(cand) + BitOperations.PopCount(forced);
                if (candCount < bestCount)
                {
                    best = cand;
                    bestCount = candCount;
                }
                if (bestCount == lowerBound) break;
            }

            ulong result = forced | best;
            _memoMask[original] = result;
            _memoMask[mask] = result;
            return result;
        }

        private ulong BruteForceMask(ulong mask)
        {
            var verts = new List<int>();
            ulong t = mask;
            while (t != 0)
            {
                int v = BitOperations.TrailingZeroCount(t);
                t &= t - 1;
                verts.Add(v);
            }

            int k = verts.Count;
            int bestSize = int.MaxValue;
            ulong best = 0;

            for (int remove = 0; remove <= k; remove++)
            {
                if (remove >= bestSize) break;
                if (!ChooseAndTestMask(verts, 0, remove, 0UL, mask, ref best, ref bestSize))
                    continue;
            }
            return best;
        }

        private bool ChooseAndTestMask(List<int> verts, int idx, int need, ulong chosen, ulong mask, ref ulong best, ref int bestSize)
        {
            if (need == 0)
            {
                ulong remaining = mask & ~chosen;
                if (IsAcyclicMask(remaining))
                {
                    int size = BitOperations.PopCount(chosen);
                    if (size < bestSize)
                    {
                        bestSize = size;
                        best = chosen;
                    }
                    return true;
                }
                return false;
            }
            if (idx >= verts.Count) return false;
            if (verts.Count - idx < need) return false;

            bool found = false;
            found |= ChooseAndTestMask(verts, idx + 1, need - 1, chosen | (1UL << verts[idx]), mask, ref best, ref bestSize);
            found |= ChooseAndTestMask(verts, idx + 1, need, chosen, mask, ref best, ref bestSize);
            return found;
        }

        private bool IsAcyclicMask(ulong mask)
        {
            var comps = GetSccMasks(mask);
            foreach (ulong c in comps)
            {
                if (BitOperations.PopCount(c) > 1) return false;
                int v = BitOperations.TrailingZeroCount(c);
                if (((_outMask[v] >> v) & 1UL) != 0) return false;
            }
            return true;
        }

        private ulong GreedyUpperBoundMask(ulong mask)
        {
            ulong chosen = 0;
            ulong active = mask;

            while (true)
            {
                active = RemoveTrivialMask(active);
                if (active == 0) break;

                ulong cycle = FindCycleMask(active);
                if (cycle == 0) break;

                int bestV = -1;
                int bestScore = int.MinValue;
                ulong t = cycle;
                while (t != 0)
                {
                    int v = BitOperations.TrailingZeroCount(t);
                    t &= t - 1;
                    int score = BitOperations.PopCount(_inMask[v] & active) + BitOperations.PopCount(_outMask[v] & active);
                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestV = v;
                    }
                }

                chosen |= 1UL << bestV;
                active &= ~(1UL << bestV);
            }

            return chosen;
        }

        private ulong RemoveTrivialMask(ulong mask)
        {
            while (true)
            {
                bool changed = false;
                ulong temp = mask;
                while (temp != 0)
                {
                    int v = BitOperations.TrailingZeroCount(temp);
                    temp &= temp - 1;
                    if (((mask >> v) & 1UL) == 0) continue;
                    if (_selfLoop[v])
                    {
                        mask &= ~(1UL << v);
                        changed = true;
                        continue;
                    }
                    int indeg = BitOperations.PopCount(_inMask[v] & mask);
                    int outdeg = BitOperations.PopCount(_outMask[v] & mask);
                    if (indeg == 0 || outdeg == 0)
                    {
                        mask &= ~(1UL << v);
                        changed = true;
                    }
                }
                if (!changed) break;
            }
            return mask;
        }

        private ulong FindCycleMask(ulong mask)
        {
            // Prefer 2-cycles when they exist.
            ulong temp = mask;
            while (temp != 0)
            {
                int u = BitOperations.TrailingZeroCount(temp);
                temp &= temp - 1;
                ulong outs = _outMask[u] & mask;
                while (outs != 0)
                {
                    int v = BitOperations.TrailingZeroCount(outs);
                    outs &= outs - 1;
                    if (u != v && (((_outMask[v] >> u) & 1UL) != 0))
                        return (1UL << u) | (1UL << v);
                }
            }

            int[] color = new int[_n];
            int[] parent = new int[_n];
            Array.Fill(parent, -1);
            var stack = new Stack<(int v, int idx)>();

            temp = mask;
            while (temp != 0)
            {
                int start = BitOperations.TrailingZeroCount(temp);
                temp &= temp - 1;
                if (color[start] != 0) continue;

                color[start] = 1;
                stack.Push((start, 0));
                while (stack.Count > 0)
                {
                    var top = stack.Pop();
                    int v = top.v;
                    int idx = top.idx;
                    var list = _out[v];
                    while (idx < list.Length && ((mask >> list[idx]) & 1UL) == 0) idx++;

                    if (idx >= list.Length)
                    {
                        color[v] = 2;
                        continue;
                    }

                    stack.Push((v, idx + 1));
                    int to = list[idx];
                    if (color[to] == 0)
                    {
                        parent[to] = v;
                        color[to] = 1;
                        stack.Push((to, 0));
                    }
                    else if (color[to] == 1)
                    {
                        ulong cyc = 1UL << to;
                        int x = v;
                        while (x != to && x != -1)
                        {
                            cyc |= 1UL << x;
                            x = parent[x];
                        }
                        if (x == to) return cyc;
                    }
                }
            }

            return 0;
        }

        private List<ulong> GetSccMasks(ulong mask)
        {
            var visited = new bool[_n];
            var order = new List<int>();
            var stack = new Stack<(int v, int idx)>();

            ulong temp = mask;
            while (temp != 0)
            {
                int s = BitOperations.TrailingZeroCount(temp);
                temp &= temp - 1;
                if (visited[s]) continue;
                visited[s] = true;
                stack.Push((s, 0));
                while (stack.Count > 0)
                {
                    var (v, idx) = stack.Pop();
                    var list = _out[v];
                    while (idx < list.Length && ((mask >> list[idx]) & 1UL) == 0) idx++;
                    if (idx < list.Length)
                    {
                        stack.Push((v, idx + 1));
                        int to = list[idx];
                        if (!visited[to])
                        {
                            visited[to] = true;
                            stack.Push((to, 0));
                        }
                    }
                    else
                    {
                        order.Add(v);
                    }
                }
            }

            Array.Fill(visited, false);
            var comps = new List<ulong>();
            stack.Clear();

            for (int oi = order.Count - 1; oi >= 0; oi--)
            {
                int s = order[oi];
                if (visited[s]) continue;
                ulong comp = 0;
                visited[s] = true;
                stack.Push((s, 0));
                while (stack.Count > 0)
                {
                    var (v, idx) = stack.Pop();
                    var list = _in[v];
                    while (idx < list.Length && ((mask >> list[idx]) & 1UL) == 0) idx++;
                    if (idx < list.Length)
                    {
                        stack.Push((v, idx + 1));
                        int to = list[idx];
                        if (!visited[to])
                        {
                            visited[to] = true;
                            stack.Push((to, 0));
                        }
                    }
                    else
                    {
                        comp |= 1UL << v;
                    }
                }
                comps.Add(comp);
            }

            return comps;
        }

        private List<int> SolveArray(bool[] active)
        {
            var forced = new List<int>();
            ReduceArray(active, forced);
            if (IsEmpty(active)) return forced;

            var comps = GetSccComponents(active);
            if (comps.Count > 1)
            {
                var sol = new List<int>(forced);
                foreach (var comp in comps)
                {
                    var sub = new bool[_n];
                    foreach (int v in comp) sub[v] = true;
                    sol.AddRange(SolveArray(sub));
                }
                return sol;
            }

            int activeCount = CountActive(active);
            if (activeCount <= 14)
            {
                var brute = BruteForceArray(active);
                forced.AddRange(brute);
                return forced;
            }

            var best = GreedyUpperBoundArray(active);
            int bestCount = forced.Count + best.Count;
            if (bestCount == forced.Count + 1)
            {
                forced.AddRange(best);
                return forced;
            }

            var cycle = FindCycleArray(active);
            if (cycle.Count == 0)
            {
                return forced;
            }

            cycle.Sort((a, b) =>
            {
                int da = DegreeScore(a, active);
                int db = DegreeScore(b, active);
                int cmp = db.CompareTo(da);
                if (cmp != 0) return cmp;
                return a.CompareTo(b);
            });

            foreach (int v in cycle)
            {
                var child = CloneActive(active);
                child[v] = false;
                var cand = SolveArray(child);
                cand.Add(v);
                int candCount = forced.Count + cand.Count;
                if (candCount < bestCount)
                {
                    best = cand;
                    bestCount = candCount;
                }
                if (bestCount == forced.Count + 1)
                    break;
            }

            forced.AddRange(best);
            return forced;
        }

        private void ReduceArray(bool[] active, List<int> forced)
        {
            while (true)
            {
                bool changed = false;
                for (int i = 0; i < _n; i++)
                {
                    if (!active[i]) continue;
                    if (_selfLoop[i])
                    {
                        active[i] = false;
                        forced.Add(i);
                        changed = true;
                        continue;
                    }
                    int indeg = 0;
                    foreach (int from in _in[i]) if (active[from]) indeg++;
                    if (indeg == 0)
                    {
                        active[i] = false;
                        changed = true;
                        continue;
                    }
                    int outdeg = 0;
                    foreach (int to in _out[i]) if (active[to]) outdeg++;
                    if (outdeg == 0)
                    {
                        active[i] = false;
                        changed = true;
                    }
                }
                if (!changed) break;
            }
        }

        private List<List<int>> GetSccComponents(bool[] active)
        {
            var visited = new bool[_n];
            var order = new List<int>();
            var stack = new Stack<(int v, int idx)>();

            for (int s = 0; s < _n; s++)
            {
                if (!active[s] || visited[s]) continue;
                visited[s] = true;
                stack.Push((s, 0));
                while (stack.Count > 0)
                {
                    var (v, idx) = stack.Pop();
                    var list = _out[v];
                    while (idx < list.Length && (!active[list[idx]] || visited[list[idx]])) idx++;
                    if (idx < list.Length)
                    {
                        stack.Push((v, idx + 1));
                        int to = list[idx];
                        if (!visited[to])
                        {
                            visited[to] = true;
                            stack.Push((to, 0));
                        }
                    }
                    else
                    {
                        order.Add(v);
                    }
                }
            }

            Array.Fill(visited, false);
            var comps = new List<List<int>>();
            stack.Clear();

            for (int oi = order.Count - 1; oi >= 0; oi--)
            {
                int s = order[oi];
                if (!active[s] || visited[s]) continue;
                var comp = new List<int>();
                visited[s] = true;
                stack.Push((s, 0));
                while (stack.Count > 0)
                {
                    var (v, idx) = stack.Pop();
                    var list = _in[v];
                    while (idx < list.Length && (!active[list[idx]] || visited[list[idx]])) idx++;
                    if (idx < list.Length)
                    {
                        stack.Push((v, idx + 1));
                        int to = list[idx];
                        if (!visited[to])
                        {
                            visited[to] = true;
                            stack.Push((to, 0));
                        }
                    }
                    else
                    {
                        comp.Add(v);
                    }
                }
                comps.Add(comp);
            }

            return comps;
        }

        private List<int> GreedyUpperBoundArray(bool[] active)
        {
            var chosen = new List<int>();
            var work = CloneActive(active);
            ReduceArray(work, new List<int>());

            while (true)
            {
                var cycle = FindCycleArray(work);
                if (cycle.Count == 0) break;
                int bestV = cycle[0];
                int bestScore = int.MinValue;
                foreach (int v in cycle)
                {
                    int score = DegreeScore(v, work);
                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestV = v;
                    }
                }
                chosen.Add(bestV);
                work[bestV] = false;
                ReduceArray(work, new List<int>());
            }

            return chosen;
        }

        private List<int> BruteForceArray(bool[] active)
        {
            var verts = new List<int>();
            for (int i = 0; i < _n; i++) if (active[i]) verts.Add(i);
            int k = verts.Count;
            var best = new List<int>();
            int bestSize = int.MaxValue;

            for (int remove = 0; remove <= k; remove++)
            {
                if (remove >= bestSize) break;
                ChooseAndTestArray(verts, 0, remove, new List<int>(), active, ref best, ref bestSize);
            }

            return best;
        }

        private bool ChooseAndTestArray(List<int> verts, int idx, int need, List<int> chosen, bool[] active, ref List<int> best, ref int bestSize)
        {
            if (need == 0)
            {
                var work = CloneActive(active);
                foreach (int v in chosen) work[v] = false;
                if (IsAcyclicArray(work))
                {
                    if (chosen.Count < bestSize)
                    {
                        bestSize = chosen.Count;
                        best = new List<int>(chosen);
                    }
                    return true;
                }
                return false;
            }
            if (idx >= verts.Count) return false;
            if (verts.Count - idx < need) return false;

            bool found = false;
            chosen.Add(verts[idx]);
            found |= ChooseAndTestArray(verts, idx + 1, need - 1, chosen, active, ref best, ref bestSize);
            chosen.RemoveAt(chosen.Count - 1);
            found |= ChooseAndTestArray(verts, idx + 1, need, chosen, active, ref best, ref bestSize);
            return found;
        }

        private bool IsAcyclicArray(bool[] active)
        {
            var comps = GetSccComponents(active);
            foreach (var comp in comps)
            {
                if (comp.Count > 1) return false;
                int v = comp[0];
                foreach (int to in _out[v])
                    if (to == v && active[to]) return false;
            }
            return true;
        }

        private List<int> FindCycleArray(bool[] active)
        {
            // Prefer 2-cycles.
            for (int u = 0; u < _n; u++)
            {
                if (!active[u]) continue;
                foreach (int v in _out[u])
                {
                    if (!active[v] || u == v) continue;
                    foreach (int back in _out[v])
                    {
                        if (back == u)
                            return new List<int> { u, v };
                    }
                }
            }

            var color = new int[_n];
            var parent = new int[_n];
            Array.Fill(parent, -1);
            var stack = new Stack<(int v, int idx)>();

            for (int s = 0; s < _n; s++)
            {
                if (!active[s] || color[s] != 0) continue;
                color[s] = 1;
                stack.Push((s, 0));
                while (stack.Count > 0)
                {
                    var (v, idx) = stack.Pop();
                    var list = _out[v];
                    while (idx < list.Length && !active[list[idx]]) idx++;
                    if (idx >= list.Length)
                    {
                        color[v] = 2;
                        continue;
                    }
                    stack.Push((v, idx + 1));
                    int to = list[idx];
                    if (color[to] == 0)
                    {
                        parent[to] = v;
                        color[to] = 1;
                        stack.Push((to, 0));
                    }
                    else if (color[to] == 1)
                    {
                        var cyc = new List<int> { to };
                        int x = v;
                        while (x != to && x != -1)
                        {
                            cyc.Add(x);
                            x = parent[x];
                        }
                        if (x == to) return cyc;
                    }
                }
            }

            return new List<int>();
        }

        private int DegreeScore(int v, bool[] active)
        {
            int score = 0;
            foreach (int to in _out[v]) if (active[to]) score++;
            foreach (int from in _in[v]) if (active[from]) score++;
            return score;
        }

        private static bool[] CloneActive(bool[] active)
        {
            var copy = new bool[active.Length];
            Array.Copy(active, copy, active.Length);
            return copy;
        }

        private bool IsEmpty(bool[] active)
        {
            for (int i = 0; i < active.Length; i++)
                if (active[i]) return false;
            return true;
        }

        private int CountActive(bool[] active)
        {
            int c = 0;
            for (int i = 0; i < active.Length; i++)
                if (active[i]) c++;
            return c;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do c = Read(); while (c <= 32 && c != 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }
