#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

vec3 boundaryForce(vec3 pos)
{
    float threshold = bounds.x * 0.4;
    float scale = max(bounds.x * 0.5, 1e-6);
    vec3 force = vec3(0.0);

    if (pos.x > threshold) force.x -= bounds.y * ((pos.x - threshold) / scale);
    if (pos.x < -threshold) force.x += bounds.y * (((-threshold) - pos.x) / scale);

    if (pos.y > threshold) force.y -= bounds.y * ((pos.y - threshold) / scale);
    if (pos.y < -threshold) force.y += bounds.y * (((-threshold) - pos.y) / scale);

    if (pos.z > threshold) force.z -= bounds.y * ((pos.z - threshold) / scale);
    if (pos.z < -threshold) force.z += bounds.y * (((-threshold) - pos.z) / scale);

    return force;
}

void main()
{
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) {
        return;
    }

    uint base = i * 2u;
    vec3 pos = inData[base].xyz;
    vec3 vel = inData[base + 1u].xyz;

    vec3 sepSum = vec3(0.0);
    vec3 alignSum = vec3(0.0);
    vec3 cohSum = vec3(0.0);
    uint sepCount = 0u;
    uint alignCount = 0u;
    uint cohCount = 0u;

    for (uint j = 0u; j < N; ++j) {
        if (j == i) {
            continue;
        }

        uint jBase = j * 2u;
        vec3 otherPos = inData[jBase].xyz;
        vec3 otherVel = inData[jBase + 1u].xyz;
        vec3 delta = otherPos - pos;
        float dist2 = dot(delta, delta);
        if (dist2 <= 0.0) {
            continue;
        }

        float dist = sqrt(dist2);

        if (dist < radii.y) {
            sepSum += (-delta / dist);
            sepCount++;
        }
        if (dist < radii.z) {
            alignSum += otherVel;
            alignCount++;
        }
        if (dist < radii.w) {
            cohSum += otherPos;
            cohCount++;
        }
    }

    vec3 accel = vec3(0.0);

    if (sepCount > 0u) {
        accel += (sepSum / float(sepCount)) * weights.x;
    }

    if (alignCount > 0u) {
        vec3 avgVel = alignSum / float(alignCount);
        accel += (avgVel - vel) * weights.y;
    }

    if (cohCount > 0u) {
        vec3 center = cohSum / float(cohCount);
        accel += (center - pos) * weights.z;
    }

    accel += boundaryForce(pos);

    vel += accel * radii.x;

    float speed = length(vel);
    if (speed > weights.w && speed > 0.0) {
        vel = (vel / speed) * weights.w;
    }

    pos += vel * radii.x;

    outData[base] = vec4(pos, 0.0);
    outData[base + 1u] = vec4(vel, 0.0);
}
