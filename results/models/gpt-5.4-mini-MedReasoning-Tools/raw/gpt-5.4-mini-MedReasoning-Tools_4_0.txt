// 3D packing solver for regular tetrahedra inside a convex polyhedron.
// C++20, standard library only.

#include <algorithm>
#include <array>
#include <cmath>
#include <cstdint>
#include <iomanip>
#include <iostream>
#include <limits>
#include <unordered_map>
#include <utility>
#include <vector>

using namespace std;

struct Vec3 {
    double x = 0, y = 0, z = 0;

    Vec3() = default;
    Vec3(double xx, double yy, double zz) : x(xx), y(yy), z(zz) {}

    Vec3 operator + (const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator - (const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator * (double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator / (double s) const { return {x / s, y / s, z / s}; }
};

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

static inline double norm(const Vec3& a) {
    return sqrt(dot(a, a));
}

struct Face {
    int a = -1, b = -1, c = -1;
    Vec3 n;
    double d = 0;
    double len = 0;
    bool removed = false;
};

static Face makeFace(int a, int b, int c, const vector<Vec3>& p, const Vec3& interior) {
    Face f;
    f.a = a; f.b = b; f.c = c;
    f.n = cross(p[b] - p[a], p[c] - p[a]);
    f.len = norm(f.n);
    if (f.len == 0.0) return f;
    f.d = -dot(f.n, p[a]);
    if (dot(f.n, interior) + f.d > 0.0) {
        swap(f.b, f.c);
        f.n = cross(p[f.b] - p[f.a], p[f.c] - p[f.a]);
        f.len = norm(f.n);
        if (f.len != 0.0) f.d = -dot(f.n, p[f.a]);
    }
    return f;
}

static inline uint64_t edgeKey(int u, int v) {
    if (u > v) swap(u, v);
    return (uint64_t(uint32_t(u)) << 32) ^ uint64_t(uint32_t(v));
}

static vector<Vec3> dedupPoints(vector<Vec3> pts) {
    sort(pts.begin(), pts.end(), [](const Vec3& a, const Vec3& b) {
        if (a.x != b.x) return a.x < b.x;
        if (a.y != b.y) return a.y < b.y;
        return a.z < b.z;
    });
    pts.erase(unique(pts.begin(), pts.end(), [](const Vec3& a, const Vec3& b) {
        return a.x == b.x && a.y == b.y && a.z == b.z;
    }), pts.end());
    return pts;
}

static vector<Face> buildConvexHull(const vector<Vec3>& pts) {
    const int n = (int)pts.size();
    vector<Face> faces;
    if (n < 4) return faces;

    int i0 = 0;
    int i1 = 0;
    double best = -1.0;
    for (int i = 1; i < n; ++i) {
        double d2 = dot(pts[i] - pts[i0], pts[i] - pts[i0]);
        if (d2 > best) {
            best = d2;
            i1 = i;
        }
    }
    if (best <= 0.0) return faces;

    int i2 = -1;
    double bestArea2 = 0.0;
    for (int i = 0; i < n; ++i) {
        if (i == i0 || i == i1) continue;
        double area2 = dot(cross(pts[i1] - pts[i0], pts[i] - pts[i0]),
                           cross(pts[i1] - pts[i0], pts[i] - pts[i0]));
        if (area2 > bestArea2) {
            bestArea2 = area2;
            i2 = i;
        }
    }
    if (i2 < 0 || bestArea2 <= 0.0) return faces;

    int i3 = -1;
    double bestVol = 0.0;
    Vec3 baseN = cross(pts[i1] - pts[i0], pts[i2] - pts[i0]);
    for (int i = 0; i < n; ++i) {
        if (i == i0 || i == i1 || i == i2) continue;
        double vol6 = fabs(dot(baseN, pts[i] - pts[i0]));
        if (vol6 > bestVol) {
            bestVol = vol6;
            i3 = i;
        }
    }
    if (i3 < 0 || bestVol <= 0.0) return faces;

    Vec3 interior = (pts[i0] + pts[i1] + pts[i2] + pts[i3]) * 0.25;
    faces.reserve(2 * n + 16);

    faces.push_back(makeFace(i0, i1, i2, pts, interior));
    faces.push_back(makeFace(i0, i3, i1, pts, interior));
    faces.push_back(makeFace(i0, i2, i3, pts, interior));
    faces.push_back(makeFace(i1, i3, i2, pts, interior));

    vector<char> isSeed(n, 0);
    isSeed[i0] = isSeed[i1] = isSeed[i2] = isSeed[i3] = 1;

    Vec3 bbMin = pts[0], bbMax = pts[0];
    for (const auto& p : pts) {
        bbMin.x = min(bbMin.x, p.x);
        bbMin.y = min(bbMin.y, p.y);
        bbMin.z = min(bbMin.z, p.z);
        bbMax.x = max(bbMax.x, p.x);
        bbMax.y = max(bbMax.y, p.y);
        bbMax.z = max(bbMax.z, p.z);
    }
    double diag = norm(bbMax - bbMin);
    double visTol = max(1e-12, 1e-10 * diag);

    for (int idx = 0; idx < n; ++idx) {
        if (isSeed[idx]) continue;

        vector<int> visible;
        visible.reserve(16);
        unordered_map<uint64_t, int> edgeCount;
        edgeCount.reserve(64);

        for (int fi = 0; fi < (int)faces.size(); ++fi) {
            Face& f = faces[fi];
            if (f.removed || f.len == 0.0) continue;
            double s = dot(f.n, pts[idx]) + f.d;
            if (s > visTol * f.len) {
                visible.push_back(fi);
                ++edgeCount[edgeKey(f.a, f.b)];
                ++edgeCount[edgeKey(f.b, f.c)];
                ++edgeCount[edgeKey(f.c, f.a)];
            }
        }

        if (visible.empty()) continue;

        for (int fi : visible) faces[fi].removed = true;

        for (const auto& kv : edgeCount) {
            if (kv.second != 1) continue;
            int u = int(kv.first >> 32);
            int v = int(kv.first & 0xffffffffu);
            Face nf = makeFace(u, v, idx, pts, interior);
            if (nf.len > 0.0) faces.push_back(nf);
        }
    }

    vector<Face> out;
    out.reserve(faces.size());
    for (const auto& f : faces) {
        if (!f.removed && f.len > 0.0) out.push_back(f);
    }
    return out;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double edgeLen;
    if (!(cin >> V >> edgeLen)) return 0;

    vector<Vec3> pts;
    pts.reserve(max(0, V));
    for (int i = 0; i < V; ++i) {
        double x, y, z;
        cin >> x >> y >> z;
        pts.push_back({x, y, z});
    }

    pts = dedupPoints(move(pts));
    if (pts.size() < 4 || edgeLen <= 0.0) {
        cout << 0 << '\n';
        return 0;
    }

    vector<Face> hull = buildConvexHull(pts);
    if (hull.empty()) {
        cout << 0 << '\n';
        return 0;
    }

    Vec3 bbMin = pts[0], bbMax = pts[0];
    for (const auto& p : pts) {
        bbMin.x = min(bbMin.x, p.x);
        bbMin.y = min(bbMin.y, p.y);
        bbMin.z = min(bbMin.z, p.z);
        bbMax.x = max(bbMax.x, p.x);
        bbMax.y = max(bbMax.y, p.y);
        bbMax.z = max(bbMax.z, p.z);
    }
    double diag = norm(bbMax - bbMin);

    const double R = edgeLen * sqrt(3.0 / 8.0);
    const double baseStep = 2.0 * sqrt(2.0) * R * 1.02; // FCC lattice, with a safety margin.
    const double boxVol = max(0.0, (bbMax.x - bbMin.x) * (bbMax.y - bbMin.y) * (bbMax.z - bbMin.z));
    const double estCount = (baseStep > 0.0) ? boxVol * 4.0 / (baseStep * baseStep * baseStep) : 0.0;

    const double faceCount = (double)max<size_t>(size_t(1), hull.size());
    const double targetCandidates = max(1000.0, min(60000.0, 60000000.0 / faceCount));
    double step = baseStep;
    if (estCount > targetCandidates && estCount > 0.0) {
        step *= cbrt(estCount / targetCandidates);
    }

    const double placeTol = max(1e-9 * diag, 1e-12);
    const double requiredClearance = R + placeTol;

    Vec3 origin = bbMin - Vec3(step * 0.5, step * 0.5, step * 0.5);
    int ix0 = (int)floor((bbMin.x - origin.x) / step) - 2;
    int iy0 = (int)floor((bbMin.y - origin.y) / step) - 2;
    int iz0 = (int)floor((bbMin.z - origin.z) / step) - 2;
    int ix1 = (int)ceil((bbMax.x - origin.x) / step) + 2;
    int iy1 = (int)ceil((bbMax.y - origin.y) / step) + 2;
    int iz1 = (int)ceil((bbMax.z - origin.z) / step) + 2;

    array<Vec3, 4> offsets = {
        Vec3{0.0, 0.0, 0.0},
        Vec3{0.0, 0.5, 0.5},
        Vec3{0.5, 0.0, 0.5},
        Vec3{0.5, 0.5, 0.0}
    };

    vector<Vec3> centers;
    centers.reserve(60000);

    for (const auto& off : offsets) {
        for (int ix = ix0; ix <= ix1; ++ix) {
            double x = origin.x + (ix + off.x) * step;
            for (int iy = iy0; iy <= iy1; ++iy) {
                double y = origin.y + (iy + off.y) * step;
                for (int iz = iz0; iz <= iz1; ++iz) {
                    if (centers.size() >= 60000) break;
                    double z = origin.z + (iz + off.z) * step;
                    Vec3 c{x, y, z};

                    bool ok = true;
                    for (const auto& f : hull) {
                        double s = dot(f.n, c) + f.d;
                        if (s > -requiredClearance * f.len) {
                            ok = false;
                            break;
                        }
                    }
                    if (ok) centers.push_back(c);
                }
                if (centers.size() >= 60000) break;
            }
            if (centers.size() >= 60000) break;
        }
        if (centers.size() >= 60000) break;
    }

    cout.setf(std::ios::fixed);
    cout << setprecision(10);
    cout << centers.size() << '\n';
    for (const auto& c : centers) {
        cout << c.x << ' ' << c.y << ' ' << c.z << " 0 0 0\n";
    }
    return 0;
}
