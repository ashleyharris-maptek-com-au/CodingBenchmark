#if defined(_MSC_VER)
#include <intrin.h>
#endif

using namespace std;

using int64 = long long;

static inline int ctz64(uint64_t x) {
#if defined(_MSC_VER)
    unsigned long idx = 0;
    _BitScanForward64(&idx, x);
    return static_cast<int>(idx);
#else
    return __builtin_ctzll(x);
#endif
}

static inline int popcount64(uint64_t x) {
#if defined(_MSC_VER)
    return static_cast<int>(__popcnt64(x));
#else
    return __builtin_popcountll(x);
#endif
}

struct Bitset {
    int n = 0;
    vector<uint64_t> b;

    Bitset() = default;
    explicit Bitset(int n_) : n(n_), b((n_ + 63) / 64, 0) {}

    bool empty() const {
        for (uint64_t x : b) if (x) return false;
        return true;
    }

    void set(int i) {
        b[i >> 6] |= 1ULL << (i & 63);
    }

    void reset(int i) {
        b[i >> 6] &= ~(1ULL << (i & 63));
    }

    bool test(int i) const {
        return (b[i >> 6] >> (i & 63)) & 1ULL;
    }

    void clear_all() {
        fill(b.begin(), b.end(), 0);
    }

    void and_with(const Bitset& o) {
        for (size_t i = 0; i < b.size(); ++i) b[i] &= o.b[i];
    }

    void andnot_with(const Bitset& o) {
        for (size_t i = 0; i < b.size(); ++i) b[i] &= ~o.b[i];
    }

    vector<int> to_vec() const {
        vector<int> v;
        for (size_t bi = 0; bi < b.size(); ++bi) {
            uint64_t x = b[bi];
            while (x) {
                int t = ctz64(x);
                v.push_back(int(bi * 64 + t));
                x &= x - 1;
            }
        }
        return v;
    }
};

struct Solver {
    int n = 0;
    vector<int64> w;
    vector<Bitset> adj;
    vector<int> order;
    int64 best = 0;
    vector<int> best_choice;

    bool has_neighbor_in_set(int v, const Bitset& P) const {
        for (size_t i = 0; i < P.b.size(); ++i) {
            if (adj[v].b[i] & P.b[i]) return true;
        }
        return false;
    }

    int degree_in_set(int v, const Bitset& P) const {
        int d = 0;
        for (size_t i = 0; i < P.b.size(); ++i) {
            d += popcount64(adj[v].b[i] & P.b[i]);
        }
        return d;
    }

    int select_branch_vertex(const Bitset& P) const {
        vector<int> verts = P.to_vec();
        int best_v = verts[0];
        int best_deg = -1;
        int64 best_w = -1;
        for (int v : verts) {
            int d = degree_in_set(v, P);
            if (d > best_deg || (d == best_deg && w[v] > best_w)) {
                best_deg = d;
                best_w = w[v];
                best_v = v;
            }
        }
        return best_v;
    }

    int64 clique_cover_bound(const Bitset& P) const {
        Bitset rem = P;
        int64 ub = 0;
        while (!rem.empty()) {
            int seed = -1;
            for (int v : order) {
                if (rem.test(v)) {
                    seed = v;
                    break;
                }
            }
            if (seed < 0) break;
            ub += w[seed];

            Bitset cand = rem;
            cand.and_with(adj[seed]);
            rem.reset(seed);

            for (int v : order) {
                if (v == seed) continue;
                if (!cand.test(v)) continue;
                rem.reset(v);
                cand.and_with(adj[v]);
            }
        }
        return ub;
    }

    void update_best(int64 val, const vector<int>& chosen) {
        if (val > best) {
            best = val;
            best_choice = chosen;
        }
    }

    void dfs(Bitset P, int64 cur, vector<int>& chosen) {
        size_t start_size = chosen.size();
        int64 forced_value = 0;

        while (true) {
            bool changed = false;
            vector<int> verts = P.to_vec();
            for (int v : verts) {
                if (!P.test(v)) continue;
                if (!has_neighbor_in_set(v, P)) {
                    P.reset(v);
                    chosen.push_back(v);
                    forced_value += w[v];
                    changed = true;
                }
            }
            if (!changed) break;
        }

        if (P.empty()) {
            update_best(cur + forced_value, chosen);
            chosen.resize(start_size);
            return;
        }

        int64 ub = cur + forced_value + clique_cover_bound(P);
        if (ub <= best) {
            chosen.resize(start_size);
            return;
        }

        int v = select_branch_vertex(P);

        Bitset include_set = P;
        include_set.reset(v);
        include_set.andnot_with(adj[v]);
        chosen.push_back(v);
        dfs(include_set, cur + forced_value + w[v], chosen);
        chosen.pop_back();

        Bitset exclude_set = P;
        exclude_set.reset(v);
        dfs(exclude_set, cur + forced_value, chosen);

        chosen.resize(start_size);
    }

    vector<int> greedy_solution_by_order(const vector<int>& ord) const {
        Bitset rem(n);
        for (int i = 0; i < n; ++i) rem.set(i);
        vector<int> chosen;
        for (int v : ord) {
            if (!rem.test(v)) continue;
            chosen.push_back(v);
            rem.reset(v);
            rem.andnot_with(adj[v]);
        }
        return chosen;
    }

    pair<int64, vector<int>> solve_component() {
        vector<int> deg(n, 0);
        for (int i = 0; i < n; ++i) {
            deg[i] = degree_in_set(i, all_set());
        }

        vector<int> ord1(n), ord2(n);
        iota(ord1.begin(), ord1.end(), 0);
        iota(ord2.begin(), ord2.end(), 0);

        sort(ord1.begin(), ord1.end(), [&](int a, int b) {
            if (w[a] != w[b]) return w[a] > w[b];
            if (deg[a] != deg[b]) return deg[a] < deg[b];
            return a < b;
        });
        sort(ord2.begin(), ord2.end(), [&](int a, int b) {
            int64 sa = w[a] * 1000LL / (deg[a] + 1);
            int64 sb = w[b] * 1000LL / (deg[b] + 1);
            if (sa != sb) return sa > sb;
            if (w[a] != w[b]) return w[a] > w[b];
            return a < b;
        });

        vector<int> best_greedy = greedy_solution_by_order(ord1);
        int64 best_val = 0;
        for (int v : best_greedy) best_val += w[v];

        vector<int> best_greedy2 = greedy_solution_by_order(ord2);
        int64 best_val2 = 0;
        for (int v : best_greedy2) best_val2 += w[v];
        if (best_val2 > best_val) {
            best_val = best_val2;
            best_greedy.swap(best_greedy2);
        }

        best = best_val;
        best_choice = best_greedy;
        order = ord1;

        Bitset P(n);
        for (int i = 0; i < n; ++i) P.set(i);
        vector<int> chosen;
        dfs(P, 0, chosen);

        return {best, best_choice};
    }

    Bitset all_set() const {
        Bitset s(n);
        for (int i = 0; i < n; ++i) s.set(i);
        return s;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;
    vector<int64> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    vector<pair<int, int>> edges;
    edges.reserve(m);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (0 <= u && u < n && 0 <= v && v < n) {
            edges.push_back({u, v});
        }
    }

    vector<int> active_id(n, -1);
    vector<int> active_vertices;
    for (int i = 0; i < n; ++i) {
        if (c[i] > 0) {
            active_id[i] = (int)active_vertices.size();
            active_vertices.push_back(i);
        }
    }

    int k = (int)active_vertices.size();
    vector<vector<int>> g(k);
    vector<Bitset> adj(k, Bitset(k));
    for (auto [u0, v0] : edges) {
        int u = active_id[u0];
        int v = active_id[v0];
        if (u < 0 || v < 0 || u == v) continue;
        if (!adj[u].test(v)) {
            adj[u].set(v);
            adj[v].set(u);
            g[u].push_back(v);
            g[v].push_back(u);
        }
    }

    vector<int> comp_id(k, -1);
    vector<vector<int>> comps;
    for (int i = 0; i < k; ++i) {
        if (comp_id[i] != -1) continue;
        queue<int> q;
        q.push(i);
        comp_id[i] = (int)comps.size();
        comps.push_back({});
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            comps.back().push_back(u);
            for (int v : g[u]) {
                if (comp_id[v] == -1) {
                    comp_id[v] = comp_id[i];
                    q.push(v);
                }
            }
        }
    }

    vector<int> answer(n, 0);
    int64 total = 0;

    for (const auto& comp : comps) {
        if (comp.empty()) continue;
        if (comp.size() == 1) {
            int local = comp[0];
            int global = active_vertices[local];
            answer[global] = 1;
            total += c[global];
            continue;
        }

        unordered_map<int, int> loc;
        loc.reserve(comp.size() * 2 + 1);
        for (int i = 0; i < (int)comp.size(); ++i) loc[comp[i]] = i;

        Solver solver;
        solver.n = (int)comp.size();
        solver.w.assign(solver.n, 0);
        solver.adj.assign(solver.n, Bitset(solver.n));

        for (int i = 0; i < solver.n; ++i) {
            int u = comp[i];
            int global_u = active_vertices[u];
            solver.w[i] = c[global_u];
        }

        for (int i = 0; i < solver.n; ++i) {
            int u = comp[i];
            for (int v_global : g[u]) {
                auto it = loc.find(v_global);
                if (it == loc.end()) continue;
                int j = it->second;
                if (i == j) continue;
                solver.adj[i].set(j);
            }
        }

        auto [best_val, chosen] = solver.solve_component();
        total += best_val;
        for (int local : chosen) {
            int global_active = comp[local];
            int global = active_vertices[global_active];
            answer[global] = 1;
        }
    }

    cout << total << '\n';
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << answer[i];
    }
    cout << '\n';
    return 0;
}
