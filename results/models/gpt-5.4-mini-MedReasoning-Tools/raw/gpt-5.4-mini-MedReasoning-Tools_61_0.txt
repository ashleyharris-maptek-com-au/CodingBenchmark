declare const require: any;
declare const process: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type PackageItem = {
  id: any;
  x: number;
  y: number;
  depDist: number;
};

type CandidateResult = {
  cost: number;
  routes: any[][];
};

function dist(ax: number, ay: number, bx: number, by: number): number {
  return Math.hypot(ax - bx, ay - by);
}

function buildRoutesFromOrder(
  order: PackageItem[],
  depotX: number,
  depotY: number,
  truckCount: number
): CandidateResult {
  const n = order.length;
  if (n === 0) {
    return { cost: 0, routes: [] };
  }

  if (n === 1) {
    const only = order[0];
    return {
      cost: 2 * only.depDist,
      routes: [[only.id]],
    };
  }

  const deltas: { idx: number; delta: number }[] = [];
  let cost = dist(depotX, depotY, order[0].x, order[0].y);
  for (let i = 0; i < n - 1; i++) {
    const a = order[i];
    const b = order[i + 1];
    const edge = dist(a.x, a.y, b.x, b.y);
    cost += edge;
    deltas.push({
      idx: i,
      delta: a.depDist + b.depDist - edge,
    });
  }
  cost += dist(order[n - 1].x, order[n - 1].y, depotX, depotY);

  deltas.sort((p, q) => p.delta - q.delta);

  const maxCuts = Math.min(Math.max(0, truckCount - 1), n - 1);
  const chosen: number[] = [];
  for (let i = 0; i < deltas.length && chosen.length < maxCuts; i++) {
    if (deltas[i].delta < 0) {
      chosen.push(deltas[i].idx);
      cost += deltas[i].delta;
    } else {
      break;
    }
  }

  chosen.sort((a, b) => a - b);
  const routes: any[][] = [];
  let start = 0;
  for (const cut of chosen) {
    const route: any[] = [];
    for (let i = start; i <= cut; i++) {
      route.push(order[i].id);
    }
    routes.push(route);
    start = cut + 1;
  }
  const lastRoute: any[] = [];
  for (let i = start; i < n; i++) {
    lastRoute.push(order[i].id);
  }
  routes.push(lastRoute);

  return { cost, routes };
}

function angleOrder(
  packages: PackageItem[],
  centerX: number,
  centerY: number,
  depotX: number,
  depotY: number,
  startMode: 'largestGap' | 'closestDepot'
): PackageItem[] {
  const arr = packages.map((p) => {
    const dx = p.x - centerX;
    const dy = p.y - centerY;
    return {
      item: p,
      angle: Math.atan2(dy, dx),
      depDist: p.depDist,
      x: p.x,
      y: p.y,
    };
  });

  arr.sort((a, b) => {
    if (a.angle !== b.angle) return a.angle - b.angle;
    if (a.depDist !== b.depDist) return a.depDist - b.depDist;
    if (a.x !== b.x) return a.x - b.x;
    if (a.y !== b.y) return a.y - b.y;
    const as = String(a.item.id);
    const bs = String(b.item.id);
    if (as < bs) return -1;
    if (as > bs) return 1;
    return 0;
  });

  const n = arr.length;
  if (n <= 1) {
    return arr.map((p) => p.item);
  }

  let startIndex = 0;
  if (startMode === 'largestGap') {
    let bestGap = -1;
    for (let i = 0; i < n; i++) {
      const cur = arr[i].angle;
      const next = arr[(i + 1) % n].angle;
      const gap = i === n - 1 ? arr[0].angle + Math.PI * 2 - cur : next - cur;
      if (gap > bestGap) {
        bestGap = gap;
        startIndex = (i + 1) % n;
      }
    }
  } else {
    let best = 0;
    let bestDist = arr[0].depDist;
    for (let i = 1; i < n; i++) {
      if (arr[i].depDist < bestDist) {
        bestDist = arr[i].depDist;
        best = i;
      }
    }
    startIndex = best;
  }

  const order: PackageItem[] = new Array(n);
  for (let i = 0; i < n; i++) {
    order[i] = arr[(startIndex + i) % n].item;
  }
  return order;
}

function xOrder(packages: PackageItem[], tieByDep: boolean): PackageItem[] {
  const arr = packages.slice();
  arr.sort((a, b) => {
    if (a.x !== b.x) return a.x - b.x;
    if (tieByDep && a.depDist !== b.depDist) return a.depDist - b.depDist;
    if (a.y !== b.y) return a.y - b.y;
    const as = String(a.id);
    const bs = String(b.id);
    if (as < bs) return -1;
    if (as > bs) return 1;
    return 0;
  });
  return arr;
}

function yOrder(packages: PackageItem[], tieByDep: boolean): PackageItem[] {
  const arr = packages.slice();
  arr.sort((a, b) => {
    if (a.y !== b.y) return a.y - b.y;
    if (tieByDep && a.depDist !== b.depDist) return a.depDist - b.depDist;
    if (a.x !== b.x) return a.x - b.x;
    const as = String(a.id);
    const bs = String(b.id);
    if (as < bs) return -1;
    if (as > bs) return 1;
    return 0;
  });
  return arr;
}

function mortonOrder(packages: PackageItem[]): PackageItem[] {
  const arr = packages.slice();
  let minX = arr[0].x;
  let maxX = arr[0].x;
  let minY = arr[0].y;
  let maxY = arr[0].y;
  for (const p of arr) {
    if (p.x < minX) minX = p.x;
    if (p.x > maxX) maxX = p.x;
    if (p.y < minY) minY = p.y;
    if (p.y > maxY) maxY = p.y;
  }

  const scaleX = maxX > minX ? 65535 / (maxX - minX) : 0;
  const scaleY = maxY > minY ? 65535 / (maxY - minY) : 0;

  function part1By1(n: number): number {
    n &= 0xffff;
    n = (n | (n << 8)) & 0x00ff00ff;
    n = (n | (n << 4)) & 0x0f0f0f0f;
    n = (n | (n << 2)) & 0x33333333;
    n = (n | (n << 1)) & 0x55555555;
    return n;
  }

  const keyed = arr.map((p) => {
    const ix = scaleX === 0 ? 0 : Math.max(0, Math.min(65535, Math.floor((p.x - minX) * scaleX)));
    const iy = scaleY === 0 ? 0 : Math.max(0, Math.min(65535, Math.floor((p.y - minY) * scaleY)));
    const code = (part1By1(ix) | (part1By1(iy) << 1)) >>> 0;
    return { p, code };
  });

  keyed.sort((a, b) => {
    if (a.code !== b.code) return a.code - b.code;
    if (a.p.depDist !== b.p.depDist) return a.p.depDist - b.p.depDist;
    if (a.p.x !== b.p.x) return a.p.x - b.p.x;
    if (a.p.y !== b.p.y) return a.p.y - b.p.y;
    const as = String(a.p.id);
    const bs = String(b.p.id);
    if (as < bs) return -1;
    if (as > bs) return 1;
    return 0;
  });

  return keyed.map((k) => k.p);
}

function evaluateCandidates(
  packages: PackageItem[],
  depotX: number,
  depotY: number,
  truckCount: number
): CandidateResult {
  const n = packages.length;
  if (n === 0) return { cost: 0, routes: [] };

  let sumX = 0;
  let sumY = 0;
  for (const p of packages) {
    sumX += p.x;
    sumY += p.y;
  }
  const centroidX = sumX / n;
  const centroidY = sumY / n;

  const candidates: PackageItem[][] = [];
  candidates.push(angleOrder(packages, depotX, depotY, depotX, depotY, 'largestGap'));
  candidates.push(angleOrder(packages, depotX, depotY, depotX, depotY, 'closestDepot'));
  candidates.push(angleOrder(packages, centroidX, centroidY, depotX, depotY, 'largestGap'));
  candidates.push(angleOrder(packages, centroidX, centroidY, depotX, depotY, 'closestDepot'));
  candidates.push(xOrder(packages, true));
  candidates.push(yOrder(packages, true));
  candidates.push(mortonOrder(packages));

  let best: CandidateResult | null = null;
  for (const order of candidates) {
    const candidate = buildRoutesFromOrder(order, depotX, depotY, truckCount);
    if (!best || candidate.cost < best.cost) {
      best = candidate;
    }
  }

  return best as CandidateResult;
}

const depot = data.depot as [number, number];
const truckCountRaw = Number(data.truckCount);
const truckCount = Number.isFinite(truckCountRaw) ? Math.max(1, Math.floor(truckCountRaw)) : 1;
const rawPackages = Array.isArray(data.packages) ? data.packages : [];

const packages: PackageItem[] = rawPackages.map((p: any) => {
  const item: PackageItem = {
    id: p.id,
    x: Number(p.x),
    y: Number(p.y),
    depDist: 0,
  };
  item.depDist = dist(depot[0], depot[1], item.x, item.y);
  return item;
});

const result = evaluateCandidates(packages, depot[0], depot[1], truckCount);
process.stdout.write(JSON.stringify({ routes: result.routes }));
