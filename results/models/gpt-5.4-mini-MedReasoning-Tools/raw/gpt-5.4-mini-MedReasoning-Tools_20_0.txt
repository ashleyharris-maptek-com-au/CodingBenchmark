use std::cmp::Ordering;
use std::io::{self, Read, Write};

#[derive(Clone)]
struct Sample {
    orig_idx: usize,
    depth: f64,
    values: Vec<f64>,
}

struct Hole {
    id: String,
    sample_indices: Vec<usize>,
}

#[derive(Clone, Copy)]
struct PropertyStats {
    median: f64,
    scale: f64,
    valid: bool,
}

#[derive(Clone)]
struct Suspect {
    hole_id: String,
    sample_idx: usize,
    property_idx: usize,
    confidence: f64,
}

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> io::Result<Self> {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf)?;
        Ok(Self { buf, idx: 0 })
    }

    fn next_token(&mut self) -> Option<&[u8]> {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let start = self.idx;
        while self.idx < self.buf.len() && !self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        Some(&self.buf[start..self.idx])
    }

    fn next_string(&mut self) -> String {
        self.next_token()
            .map(|t| String::from_utf8_lossy(t).into_owned())
            .unwrap_or_default()
    }

    fn next_usize(&mut self) -> usize {
        self.next_token()
            .and_then(|t| std::str::from_utf8(t).ok())
            .and_then(|s| s.parse::<usize>().ok())
            .unwrap_or(0)
    }

    fn next_f64(&mut self) -> f64 {
        let tok = match self.next_token() {
            Some(t) => t,
            None => return f64::NAN,
        };
        let s = match std::str::from_utf8(tok) {
            Ok(v) => v,
            Err(_) => return f64::NAN,
        };
        if s.eq_ignore_ascii_case("na")
            || s.eq_ignore_ascii_case("nan")
            || s.eq_ignore_ascii_case("null")
            || s.eq_ignore_ascii_case("none")
            || s == "."
            || s.is_empty()
        {
            return f64::NAN;
        }
        match s.parse::<f64>() {
            Ok(v) if v.is_finite() => v,
            _ => f64::NAN,
        }
    }
}

fn median_sorted(data: &[f64]) -> f64 {
    let n = data.len();
    if n == 0 {
        return f64::NAN;
    }
    if n % 2 == 1 {
        data[n / 2]
    } else {
        (data[n / 2 - 1] + data[n / 2]) * 0.5
    }
}

fn percentile_sorted(data: &[f64], p: f64) -> f64 {
    let n = data.len();
    if n == 0 {
        return f64::NAN;
    }
    if n == 1 {
        return data[0];
    }
    let pos = p.clamp(0.0, 1.0) * (n as f64 - 1.0);
    let lo = pos.floor() as usize;
    let hi = pos.ceil() as usize;
    if lo == hi {
        data[lo]
    } else {
        let w = pos - lo as f64;
        data[lo] * (1.0 - w) + data[hi] * w
    }
}

fn robust_stats(mut vals: Vec<f64>) -> PropertyStats {
    if vals.is_empty() {
        return PropertyStats {
            median: 0.0,
            scale: 1.0,
            valid: false,
        };
    }
    vals.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let median = median_sorted(&vals);
    let mut devs: Vec<f64> = vals.iter().map(|v| (v - median).abs()).collect();
    devs.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let mad = median_sorted(&devs);
    let q1 = percentile_sorted(&vals, 0.25);
    let q3 = percentile_sorted(&vals, 0.75);

    let mut scale = mad * 1.4826;
    if !scale.is_finite() || scale < 1e-12 {
        scale = (q3 - q1).abs() / 1.349;
    }
    if !scale.is_finite() || scale < 1e-12 {
        scale = median.abs() * 0.02 + 1e-6;
    }
    PropertyStats {
        median,
        scale,
        valid: true,
    }
}

fn local_score(v: f64, neigh: &[f64], global_scale: f64) -> f64 {
    if neigh.is_empty() {
        return 0.0;
    }
    if neigh.len() == 1 {
        return (v - neigh[0]).abs() / global_scale.max(1e-12);
    }
    let mut tmp = neigh.to_vec();
    tmp.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let med = median_sorted(&tmp);
    let mut devs: Vec<f64> = neigh.iter().map(|x| (x - med).abs()).collect();
    devs.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let mad = median_sorted(&devs);
    let mut scale = mad * 1.4826;
    if !scale.is_finite() || scale < global_scale * 0.25 {
        scale = global_scale;
    }
    if !scale.is_finite() || scale < 1e-12 {
        scale = 1e-12;
    }
    (v - med).abs() / scale
}

fn main() -> io::Result<()> {
    let mut scanner = Scanner::new()?;

    let num_holes = scanner.next_usize();
    let num_properties = scanner.next_usize();
    let mut property_names = Vec::with_capacity(num_properties);
    for _ in 0..num_properties {
        property_names.push(scanner.next_string());
    }

    let mut holes: Vec<Hole> = Vec::with_capacity(num_holes);
    let mut samples: Vec<Sample> = Vec::new();

    for _hole_idx in 0..num_holes {
        let hole_id = scanner.next_string();
        let _start_x = scanner.next_f64();
        let _start_y = scanner.next_f64();
        let _start_z = scanner.next_f64();
        let _dir_x = scanner.next_f64();
        let _dir_y = scanner.next_f64();
        let _dir_z = scanner.next_f64();
        let _length = scanner.next_f64();
        let num_samples = scanner.next_usize();

        let mut sample_indices = Vec::with_capacity(num_samples);
        for sample_idx in 0..num_samples {
            let depth = scanner.next_f64();
            let mut values = Vec::with_capacity(num_properties);
            for _ in 0..num_properties {
                values.push(scanner.next_f64());
            }
            samples.push(Sample { orig_idx: sample_idx, depth, values });
            sample_indices.push(samples.len() - 1);
        }

        holes.push(Hole {
            id: hole_id,
            sample_indices,
        });
    }

    let total_samples = samples.len();
    let local_radius = if total_samples < 20_000 { 3 } else { 2 };
    let threshold = if total_samples < 2_000 {
        4.0
    } else if total_samples < 50_000 {
        4.75
    } else {
        5.25
    };

    let mut per_property_values: Vec<Vec<f64>> = vec![Vec::new(); num_properties];
    for sample in &samples {
        for (p, &v) in sample.values.iter().enumerate() {
            if v.is_finite() {
                per_property_values[p].push(v);
            }
        }
    }

    let stats: Vec<PropertyStats> = per_property_values
        .into_iter()
        .map(robust_stats)
        .collect();

    let mut suspects: Vec<Suspect> = Vec::new();

    for hole in &holes {
        let mut ordered = hole.sample_indices.clone();
        ordered.sort_by(|&a, &b| {
            samples[a]
                .depth
                .partial_cmp(&samples[b].depth)
                .unwrap_or(Ordering::Equal)
                .then_with(|| samples[a].orig_idx.cmp(&samples[b].orig_idx))
        });

        for prop_idx in 0..num_properties {
            let stat = stats[prop_idx];
            if !stat.valid {
                continue;
            }

            let mut seq: Vec<(usize, f64)> = Vec::with_capacity(ordered.len());
            for &sidx in &ordered {
                let v = samples[sidx].values[prop_idx];
                if v.is_finite() {
                    seq.push((sidx, v));
                }
            }
            if seq.len() < 2 {
                continue;
            }

            for i in 0..seq.len() {
                let (sidx, v) = seq[i];
                let gscore = (v - stat.median).abs() / stat.scale.max(1e-12);

                let mut neigh: Vec<f64> = Vec::with_capacity((local_radius * 2) as usize);
                let left_start = i.saturating_sub(local_radius);
                for j in left_start..i {
                    neigh.push(seq[j].1);
                }
                let right_end = (i + local_radius + 1).min(seq.len());
                for j in (i + 1)..right_end {
                    neigh.push(seq[j].1);
                }

                let lscore = local_score(v, &neigh, stat.scale);
                let score = gscore.max(lscore);

                let strong_local_spike =
                    neigh.len() >= 2 && lscore >= threshold - 0.5 && gscore >= 2.5;
                let strong_global_outlier = gscore >= threshold + 0.5;
                let obvious = score >= threshold || strong_local_spike || strong_global_outlier;

                if obvious {
                    suspects.push(Suspect {
                        hole_id: hole.id.clone(),
                        sample_idx: samples[sidx].orig_idx,
                        property_idx: prop_idx,
                        confidence: score,
                    });
                }
            }
        }
    }

    suspects.sort_by(|a, b| {
        b.confidence
            .partial_cmp(&a.confidence)
            .unwrap_or(Ordering::Equal)
            .then_with(|| a.hole_id.cmp(&b.hole_id))
            .then_with(|| a.sample_idx.cmp(&b.sample_idx))
            .then_with(|| a.property_idx.cmp(&b.property_idx))
    });

    let mut out = String::new();
    out.push_str(&format!("{}\n", suspects.len()));
    for s in suspects {
        out.push_str(&format!(
            "{} {} {} {:.6}\n",
            s.hole_id, s.sample_idx, property_names[s.property_idx], s.confidence
        ));
    }

    let mut stdout = io::BufWriter::new(io::stdout());
    stdout.write_all(out.as_bytes())?;
    Ok(())
}
