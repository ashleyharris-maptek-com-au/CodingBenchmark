const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = Array.isArray(data.grid) ? data.grid : [];
const robots = Array.isArray(data.robots) ? data.robots : [];
const jobs = Array.isArray(data.jobs) ? data.jobs : [];

const H = grid.length;
const W = H > 0 ? grid[0].length : 0;
const N = H * W;

const open = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < W; x++) {
    open[y * W + x] = row.charCodeAt(x) !== 35 ? 1 : 0;
  }
}

function cellIndex(x, y) {
  return y * W + x;
}

function cellX(idx) {
  return idx % W;
}

function cellY(idx) {
  return (idx / W) | 0;
}

// Connected components of open cells.
const comp = new Int32Array(N);
comp.fill(-1);
const bfsQueue = new Int32Array(N);
let compCount = 0;

for (let start = 0; start < N; start++) {
  if (!open[start] || comp[start] !== -1) continue;
  let head = 0;
  let tail = 0;
  bfsQueue[tail++] = start;
  comp[start] = compCount;
  while (head < tail) {
    const v = bfsQueue[head++];
    const x = cellX(v);
    const y = cellY(v);

    if (x > 0) {
      const n = v - 1;
      if (open[n] && comp[n] === -1) {
        comp[n] = compCount;
        bfsQueue[tail++] = n;
      }
    }
    if (x + 1 < W) {
      const n = v + 1;
      if (open[n] && comp[n] === -1) {
        comp[n] = compCount;
        bfsQueue[tail++] = n;
      }
    }
    if (y > 0) {
      const n = v - W;
      if (open[n] && comp[n] === -1) {
        comp[n] = compCount;
        bfsQueue[tail++] = n;
      }
    }
    if (y + 1 < H) {
      const n = v + W;
      if (open[n] && comp[n] === -1) {
        comp[n] = compCount;
        bfsQueue[tail++] = n;
      }
    }
  }
  compCount++;
}

const maxRobotId = robots.reduce((m, r) => Math.max(m, r && Number.isFinite(r.id) ? r.id : -1), -1);
const assignmentsById = Array.from({ length: Math.max(0, maxRobotId + 1) }, () => []);
const robotById = new Array(Math.max(0, maxRobotId + 1));
for (const r of robots) {
  if (!r || !Number.isFinite(r.id)) continue;
  robotById[r.id] = r;
}

const robotsByComp = new Map();
for (const r of robots) {
  if (!r || !Number.isFinite(r.id)) continue;
  const idx = cellIndex(r.x, r.y);
  const cid = idx >= 0 && idx < N ? comp[idx] : -1;
  if (!robotsByComp.has(cid)) robotsByComp.set(cid, []);
  robotsByComp.get(cid).push(r);
}

const jobsByComp = new Map();
for (const j of jobs) {
  if (!j) continue;
  const idx = cellIndex(j.x, j.y);
  const cid = idx >= 0 && idx < N ? comp[idx] : -1;
  if (!jobsByComp.has(cid)) jobsByComp.set(cid, []);
  jobsByComp.get(cid).push(j);
}

// Mark each job to the nearest robot in the same connected region.
const owner = new Int32Array(N);
owner.fill(-1);
const assignQueue = new Int32Array(N);

for (const [cid, compJobs] of jobsByComp.entries()) {
  if (cid < 0) continue;
  const compRobots = robotsByComp.get(cid);
  if (!compRobots || compRobots.length === 0) continue;

  compRobots.sort((a, b) => a.id - b.id);

  let head = 0;
  let tail = 0;
  const touched = [];

  for (const r of compRobots) {
    const idx = cellIndex(r.x, r.y);
    if (idx < 0 || idx >= N || comp[idx] !== cid) continue;
    if (owner[idx] !== -1) continue;
    owner[idx] = r.id;
    assignQueue[tail++] = idx;
    touched.push(idx);
  }

  while (head < tail) {
    const v = assignQueue[head++];
    const o = owner[v];
    const x = cellX(v);
    const y = cellY(v);

    if (x > 0) {
      const n = v - 1;
      if (open[n] && comp[n] === cid && owner[n] === -1) {
        owner[n] = o;
        assignQueue[tail++] = n;
        touched.push(n);
      }
    }
    if (x + 1 < W) {
      const n = v + 1;
      if (open[n] && comp[n] === cid && owner[n] === -1) {
        owner[n] = o;
        assignQueue[tail++] = n;
        touched.push(n);
      }
    }
    if (y > 0) {
      const n = v - W;
      if (open[n] && comp[n] === cid && owner[n] === -1) {
        owner[n] = o;
        assignQueue[tail++] = n;
        touched.push(n);
      }
    }
    if (y + 1 < H) {
      const n = v + W;
      if (open[n] && comp[n] === cid && owner[n] === -1) {
        owner[n] = o;
        assignQueue[tail++] = n;
        touched.push(n);
      }
    }
  }

  for (const j of compJobs) {
    const idx = cellIndex(j.x, j.y);
    const rid = idx >= 0 && idx < N ? owner[idx] : -1;
    if (rid >= 0 && rid < assignmentsById.length) {
      assignmentsById[rid].push(j);
    }
  }

  for (const idx of touched) owner[idx] = -1;
}

// Any jobs in components without robots fall back to the nearest robot by Manhattan distance.
if (robots.length > 0) {
  const fallbackRobots = robots.slice().sort((a, b) => a.id - b.id);
  for (const [cid, compJobs] of jobsByComp.entries()) {
    if (cid >= 0 && robotsByComp.has(cid)) continue;
    for (const j of compJobs) {
      let bestId = fallbackRobots[0].id;
      let bestD = Math.abs(j.x - fallbackRobots[0].x) + Math.abs(j.y - fallbackRobots[0].y);
      for (let i = 1; i < fallbackRobots.length; i++) {
        const r = fallbackRobots[i];
        const d = Math.abs(j.x - r.x) + Math.abs(j.y - r.y);
        if (d < bestD || (d === bestD && r.id < bestId)) {
          bestD = d;
          bestId = r.id;
        }
      }
      if (bestId >= 0 && bestId < assignmentsById.length) {
        assignmentsById[bestId].push(j);
      }
    }
  }
}

function orderJobs(robot, list) {
  if (!list || list.length <= 1) return list ? list.map(j => j.id) : [];

  const n = list.length;
  if (n <= 1200) {
    const used = new Uint8Array(n);
    const order = new Array(n);
    let cx = robot.x;
    let cy = robot.y;

    for (let k = 0; k < n; k++) {
      let best = -1;
      let bestD = Infinity;
      let bestP = -Infinity;
      let bestId = Infinity;
      for (let i = 0; i < n; i++) {
        if (used[i]) continue;
        const j = list[i];
        const pr = j.priority ?? 0;
        const d = Math.abs(j.x - cx) + Math.abs(j.y - cy);
        if (d < bestD || (d === bestD && (pr > bestP || (pr === bestP && j.id < bestId)))) {
          best = i;
          bestD = d;
          bestP = pr;
          bestId = j.id;
        }
      }
      used[best] = 1;
      const chosen = list[best];
      order[k] = chosen.id;
      cx = chosen.x;
      cy = chosen.y;
    }
    return order;
  }

  let minX = robot.x, maxX = robot.x, minY = robot.y, maxY = robot.y;
  for (const j of list) {
    if (j.x < minX) minX = j.x;
    if (j.x > maxX) maxX = j.x;
    if (j.y < minY) minY = j.y;
    if (j.y > maxY) maxY = j.y;
  }

  const spanX = maxX > minX ? (65535 / (maxX - minX)) : 0;
  const spanY = maxY > minY ? (65535 / (maxY - minY)) : 0;

  function part1by1(n) {
    n &= 0xffff;
    n = (n | (n << 8)) & 0x00FF00FF;
    n = (n | (n << 4)) & 0x0F0F0F0F;
    n = (n | (n << 2)) & 0x33333333;
    n = (n | (n << 1)) & 0x55555555;
    return n;
  }

  function morton(x, y) {
    return (part1by1(x) | (part1by1(y) << 1)) >>> 0;
  }

  const keyed = new Array(n);
  for (let i = 0; i < n; i++) {
    const j = list[i];
    const dx = Math.abs(j.x - robot.x);
    const dy = Math.abs(j.y - robot.y);
    const dist = dx + dy;
    const nx = spanX === 0 ? 0 : Math.max(0, Math.min(65535, Math.floor((j.x - minX) * spanX)));
    const ny = spanY === 0 ? 0 : Math.max(0, Math.min(65535, Math.floor((j.y - minY) * spanY)));
    keyed[i] = { id: j.id, dist, key: morton(nx, ny), pr: j.priority ?? 0 };
  }
  keyed.sort((a, b) => a.dist - b.dist || a.key - b.key || b.pr - a.pr || a.id - b.id);
  return keyed.map(v => v.id);
}

const assignments = Array.from({ length: Math.max(0, maxRobotId + 1) }, (_, id) => {
  const robot = robotById[id];
  if (!robot) return [];
  return orderJobs(robot, assignmentsById[id]);
});

process.stdout.write(JSON.stringify({ assignments }));
