using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    private struct Box
    {
        public int A;
        public int B;
        public int C;
        public int Index;
    }

    private struct Space
    {
        public int X;
        public int Y;
        public int Z;
        public int W;
        public int H;
        public int D;

        public Space(int x, int y, int z, int w, int h, int d)
        {
            X = x;
            Y = y;
            Z = z;
            W = w;
            H = h;
            D = d;
        }
    }

    private struct Placement
    {
        public int Index;
        public int X;
        public int Y;
        public int Z;

        public Placement(int index, int x, int y, int z)
        {
            Index = index;
            X = x;
            Y = y;
            Z = z;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _position;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private byte Read()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length <= 0)
                {
                    return 0;
                }
            }

            return _buffer[_position++];
        }

        public int ReadInt()
        {
            byte c;
            do
            {
                c = Read();
            }
            while (c <= 32 && c != 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }
    }

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());

        int n = scanner.ReadInt();
        int containerW = scanner.ReadInt();
        int containerH = scanner.ReadInt();
        int containerD = scanner.ReadInt();

        Sort3Desc(containerW, containerH, containerD, out int cw0, out int cw1, out int cw2);

        var boxes = new List<Box>(n);
        for (int i = 0; i < n; i++)
        {
            int a = scanner.ReadInt();
            int b = scanner.ReadInt();
            int c = scanner.ReadInt();

            Sort3Desc(a, b, c, out int x, out int y, out int z);
            if (x <= cw0 && y <= cw1 && z <= cw2)
            {
                boxes.Add(new Box
                {
                    A = x,
                    B = y,
                    C = z,
                    Index = i
                });
            }
        }

        boxes.Sort((left, right) =>
        {
            if (left.A != right.A) return right.A.CompareTo(left.A);
            if (left.B != right.B) return right.B.CompareTo(left.B);
            if (left.C != right.C) return right.C.CompareTo(left.C);
            return left.Index.CompareTo(right.Index);
        });

        var spaces = new List<Space>(Math.Max(4, boxes.Count * 3 + 1))
        {
            new Space(0, 0, 0, containerW, containerH, containerD)
        };

        var placements = new List<Placement>(boxes.Count);

        for (int bi = 0; bi < boxes.Count; bi++)
        {
            Box box = boxes[bi];
            int bestSpaceIndex = -1;
            int bestBw = 0;
            int bestBh = 0;
            int bestBd = 0;
            long bestScore = long.MaxValue;
            long bestSpaceVolume = long.MaxValue;

            for (int si = 0; si < spaces.Count; si++)
            {
                Space s = spaces[si];
                Sort3DescWithAxes(s.W, s.H, s.D,
                    out int sw0, out int sa0,
                    out int sw1, out int sa1,
                    out int sw2, out int sa2);

                if (box.A > sw0 || box.B > sw1 || box.C > sw2)
                {
                    continue;
                }

                int bw = 0;
                int bh = 0;
                int bd = 0;

                if (sa0 == 0) bw = box.A;
                else if (sa1 == 0) bw = box.B;
                else bw = box.C;

                if (sa0 == 1) bh = box.A;
                else if (sa1 == 1) bh = box.B;
                else bh = box.C;

                if (sa0 == 2) bd = box.A;
                else if (sa1 == 2) bd = box.B;
                else bd = box.C;

                long rx = s.W - bw;
                long ry = s.H - bh;
                long rz = s.D - bd;

                long maxSlack = rx;
                if (ry > maxSlack) maxSlack = ry;
                if (rz > maxSlack) maxSlack = rz;

                long score = maxSlack * 1_000_000_000L + rx * rx + ry * ry + rz * rz;
                long spaceVolume = (long)s.W * s.H * s.D;

                if (score < bestScore || (score == bestScore && spaceVolume < bestSpaceVolume))
                {
                    bestScore = score;
                    bestSpaceVolume = spaceVolume;
                    bestSpaceIndex = si;
                    bestBw = bw;
                    bestBh = bh;
                    bestBd = bd;
                }
            }

            if (bestSpaceIndex < 0)
            {
                continue;
            }

            Space chosen = spaces[bestSpaceIndex];
            placements.Add(new Placement(box.Index, chosen.X, chosen.Y, chosen.Z));

            int last = spaces.Count - 1;
            spaces[bestSpaceIndex] = spaces[last];
            spaces.RemoveAt(last);

            int rw = chosen.W - bestBw;
            int rh = chosen.H - bestBh;
            int rd = chosen.D - bestBd;

            if (rw > 0)
            {
                spaces.Add(new Space(chosen.X + bestBw, chosen.Y, chosen.Z, rw, chosen.H, chosen.D));
            }

            if (rh > 0)
            {
                spaces.Add(new Space(chosen.X, chosen.Y + bestBh, chosen.Z, bestBw, rh, chosen.D));
            }

            if (rd > 0)
            {
                spaces.Add(new Space(chosen.X, chosen.Y, chosen.Z + bestBd, bestBw, bestBh, rd));
            }
        }

        var writer = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };
        writer.WriteLine(placements.Count);
        for (int i = 0; i < placements.Count; i++)
        {
            Placement p = placements[i];
            writer.WriteLine($"{p.Index} {p.X} {p.Y} {p.Z}");
        }
        writer.Flush();
    }

    private static void Sort3Desc(int a, int b, int c, out int x, out int y, out int z)
    {
        if (a < b)
        {
            int t = a;
            a = b;
            b = t;
        }

        if (a < c)
        {
            int t = a;
            a = c;
            c = t;
        }

        if (b < c)
        {
            int t = b;
            b = c;
            c = t;
        }

        x = a;
        y = b;
        z = c;
    }

    private static void Sort3DescWithAxes(
        int a, int b, int c,
        out int v0, out int i0,
        out int v1, out int i1,
        out int v2, out int i2)
    {
        v0 = a; i0 = 0;
        v1 = b; i1 = 1;
        v2 = c; i2 = 2;

        if (v0 < v1)
        {
            Swap(ref v0, ref v1);
            Swap(ref i0, ref i1);
        }

        if (v0 < v2)
        {
            Swap(ref v0, ref v2);
            Swap(ref i0, ref i2);
        }

        if (v1 < v2)
        {
            Swap(ref v1, ref v2);
            Swap(ref i1, ref i2);
        }
    }

    private static void Swap(ref int a, ref int b)
    {
        int t = a;
        a = b;
        b = t;
    }
}
