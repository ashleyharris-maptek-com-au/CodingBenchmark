OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor %ubo
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %ubo "ubo"
OpName %UBO "UBO"
OpName %gl_PerVertex_in "gl_PerVertex"
OpName %gl_PerVertex_out "gl_PerVertex"

OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "proj"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpDecorate %outColor Location 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%fn = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%float_1 = OpConstant %float 1
%float_h = OpConstant %float 0.05

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_arr_pv_in = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_arr_pv_in Input

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%arr_v3_1 = OpTypeArray %v3float %uint_1
%ptr_arr_v3_in = OpTypePointer Input %arr_v3_1
%worldPos = OpVariable %ptr_arr_v3_in Input
%normal = OpVariable %ptr_arr_v3_in Input
%color = OpVariable %ptr_arr_v3_in Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_gl_pervertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_gl_pervertex_out Output
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_ubo Uniform

%main = OpFunction %void None %fn
%entry = OpLabel

%worldPos_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%worldPos_val = OpLoad %v3float %worldPos_ptr
%color_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%color_val = OpLoad %v3float %color_ptr

%x = OpCompositeExtract %float %worldPos_val 0
%y = OpCompositeExtract %float %worldPos_val 1
%z = OpCompositeExtract %float %worldPos_val 2
%y_top = OpFAdd %float %y %float_h
%x_right = OpFAdd %float %x %float_h
%y_bottom = OpFSub %float %y %float_h
%x_left = OpFSub %float %x %float_h

%out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

%top_pos = OpCompositeConstruct %v4float %x %y_top %z %float_1
OpStore %out_pos %top_pos
OpStore %outColor %color_val
OpEmitVertex

%right_pos = OpCompositeConstruct %v4float %x_right %y %z %float_1
OpStore %out_pos %right_pos
OpStore %outColor %color_val
OpEmitVertex

%left_pos = OpCompositeConstruct %v4float %x_left %y %z %float_1
OpStore %out_pos %left_pos
OpStore %outColor %color_val
OpEmitVertex

%bottom_pos = OpCompositeConstruct %v4float %x %y_bottom %z %float_1
OpStore %out_pos %bottom_pos
OpStore %outColor %color_val
OpEmitVertex

OpEndPrimitive
OpFunctionEnd
