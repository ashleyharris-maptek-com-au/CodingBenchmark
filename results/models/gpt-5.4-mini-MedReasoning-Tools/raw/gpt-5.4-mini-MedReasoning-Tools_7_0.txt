// CSG Union via BSP trees
// The implementation follows the classic CSG boolean approach:
// 1. Build BSP trees from both triangle meshes.
// 2. Clip polygons against the opposite tree.
// 3. Merge the surviving polygons.
// 4. Triangulate the convex output polygons and deduplicate vertices.

#include <algorithm>
#include <array>
#include <cmath>
#include <cstdint>
#include <cstdlib>
#include <iomanip>
#include <iostream>
#include <limits>
#include <memory>
#include <optional>
#include <unordered_map>
#include <utility>
#include <vector>

using namespace std;

static double gEps = 1e-9;

struct Vec3 {
    double x = 0.0, y = 0.0, z = 0.0;

    Vec3() = default;
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}

    Vec3 operator + (const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator - (const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator * (double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator / (double s) const { return {x / s, y / s, z / s}; }
    Vec3& operator += (const Vec3& o) { x += o.x; y += o.y; z += o.z; return *this; }
};

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

static inline double norm(const Vec3& a) {
    return sqrt(dot(a, a));
}

static inline Vec3 normalized(const Vec3& a) {
    double n = norm(a);
    if (n <= 0.0) return {0, 0, 0};
    return a / n;
}

static inline bool nearZero(double v) {
    return fabs(v) <= gEps;
}

struct Plane {
    Vec3 n;
    double w = 0.0;

    Plane() = default;
    Plane(const Vec3& a, const Vec3& b, const Vec3& c) {
        n = normalized(cross(b - a, c - a));
        w = dot(n, a);
    }

    void flip() {
        n = n * -1.0;
        w = -w;
    }
};

struct Polygon {
    vector<Vec3> vertices;
    Plane plane;

    void computePlane() {
        if (vertices.size() >= 3) {
            plane = Plane(vertices[0], vertices[1], vertices[2]);
        }
    }

    void flip() {
        reverse(vertices.begin(), vertices.end());
        plane.flip();
    }
};

static Vec3 lerp(const Vec3& a, const Vec3& b, double t) {
    return a + (b - a) * t;
}

static Polygon cleanPolygon(const Polygon& in) {
    Polygon out;
    const auto& v = in.vertices;
    if (v.size() < 3) return out;

    out.vertices.reserve(v.size());
    for (const auto& p : v) {
        if (out.vertices.empty()) {
            out.vertices.push_back(p);
        } else {
            const Vec3& q = out.vertices.back();
            if (fabs(p.x - q.x) > gEps || fabs(p.y - q.y) > gEps || fabs(p.z - q.z) > gEps) {
                out.vertices.push_back(p);
            }
        }
    }
    if (out.vertices.size() >= 2) {
        const Vec3& a = out.vertices.front();
        const Vec3& b = out.vertices.back();
        if (fabs(a.x - b.x) <= gEps && fabs(a.y - b.y) <= gEps && fabs(a.z - b.z) <= gEps) {
            out.vertices.pop_back();
        }
    }

    if (out.vertices.size() < 3) return Polygon{};

    // Remove near-collinear vertices on the boundary to keep polygons simple.
    bool changed = true;
    while (changed && out.vertices.size() >= 3) {
        changed = false;
        vector<Vec3> tmp;
        size_t n = out.vertices.size();
        tmp.reserve(n);
        for (size_t i = 0; i < n; ++i) {
            const Vec3& prev = out.vertices[(i + n - 1) % n];
            const Vec3& cur  = out.vertices[i];
            const Vec3& next = out.vertices[(i + 1) % n];
            Vec3 a = cur - prev;
            Vec3 b = next - cur;
            Vec3 cr = cross(a, b);
            if (norm(cr) <= gEps * (norm(a) + norm(b) + 1.0)) {
                changed = true;
                continue;
            }
            tmp.push_back(cur);
        }
        if (tmp.size() < 3) return Polygon{};
        out.vertices.swap(tmp);
    }

    out.computePlane();
    return out;
}

static void splitPolygonByPlane(
    const Plane& plane,
    const Polygon& poly,
    vector<Polygon>& coplanarFront,
    vector<Polygon>& coplanarBack,
    vector<Polygon>& front,
    vector<Polygon>& back)
{
    enum Type { COPLANAR = 0, FRONT = 1, BACK = 2, SPANNING = 3 };

    const auto& verts = poly.vertices;
    size_t n = verts.size();
    if (n < 3) return;

    vector<int> types(n);
    int polygonType = 0;
    for (size_t i = 0; i < n; ++i) {
        double t = dot(plane.n, verts[i]) - plane.w;
        int type = (t < -gEps) ? BACK : (t > gEps ? FRONT : COPLANAR);
        types[i] = type;
        polygonType |= type;
    }

    switch (polygonType) {
        case COPLANAR:
            if (dot(plane.n, poly.plane.n) >= 0.0) coplanarFront.push_back(poly);
            else coplanarBack.push_back(poly);
            return;
        case FRONT:
            front.push_back(poly);
            return;
        case BACK:
            back.push_back(poly);
            return;
        default:
            break;
    }

    vector<Vec3> f, b;
    f.reserve(n + 2);
    b.reserve(n + 2);
    for (size_t i = 0; i < n; ++i) {
        size_t j = (i + 1) % n;
        const Vec3& vi = verts[i];
        const Vec3& vj = verts[j];
        int ti = types[i];
        int tj = types[j];

        if (ti != BACK) f.push_back(vi);
        if (ti != FRONT) b.push_back(vi);

        if ((ti | tj) == SPANNING) {
            double denom = dot(plane.n, vj - vi);
            if (fabs(denom) > gEps) {
                double t = (plane.w - dot(plane.n, vi)) / denom;
                Vec3 v = lerp(vi, vj, t);
                f.push_back(v);
                b.push_back(v);
            }
        }
    }

    if (f.size() >= 3) {
        Polygon p; p.vertices = move(f); p.computePlane();
        p = cleanPolygon(p);
        if (p.vertices.size() >= 3) front.push_back(move(p));
    }
    if (b.size() >= 3) {
        Polygon p; p.vertices = move(b); p.computePlane();
        p = cleanPolygon(p);
        if (p.vertices.size() >= 3) back.push_back(move(p));
    }
}

struct Node {
    optional<Plane> plane;
    vector<Polygon> polygons;
    unique_ptr<Node> front;
    unique_ptr<Node> back;

    Node() = default;
    explicit Node(const vector<Polygon>& polys) { build(polys); }

    void build(const vector<Polygon>& polys) {
        if (polys.empty()) return;
        if (!plane.has_value()) {
            plane = polys.front().plane;
        }

        vector<Polygon> frontList, backList;
        for (const auto& p : polys) {
            splitPolygonByPlane(*plane, p, polygons, polygons, frontList, backList);
        }

        if (!frontList.empty()) {
            if (!front) front = make_unique<Node>();
            front->build(frontList);
        }
        if (!backList.empty()) {
            if (!back) back = make_unique<Node>();
            back->build(backList);
        }
    }

    vector<Polygon> allPolygons() const {
        vector<Polygon> res = polygons;
        if (front) {
            auto v = front->allPolygons();
            res.insert(res.end(), v.begin(), v.end());
        }
        if (back) {
            auto v = back->allPolygons();
            res.insert(res.end(), v.begin(), v.end());
        }
        return res;
    }

    vector<Polygon> clipPolygons(const vector<Polygon>& polys) const {
        if (!plane.has_value()) return polys;

        vector<Polygon> frontList, backList;
        for (const auto& p : polys) {
            splitPolygonByPlane(*plane, p, frontList, backList, frontList, backList);
        }

        if (front) frontList = front->clipPolygons(frontList);
        if (back) backList = back->clipPolygons(backList);
        else backList.clear();

        frontList.insert(frontList.end(), backList.begin(), backList.end());
        return frontList;
    }

    void clipTo(const Node& bsp) {
        polygons = bsp.clipPolygons(polygons);
        if (front) front->clipTo(bsp);
        if (back) back->clipTo(bsp);
    }

    void invert() {
        for (auto& p : polygons) p.flip();
        if (plane.has_value()) plane->flip();
        if (front) front->invert();
        if (back) back->invert();
        swap(front, back);
    }
};

struct Mesh {
    vector<Vec3> vertices;
    vector<array<int, 3>> faces;
    Vec3 bmin{numeric_limits<double>::infinity(), numeric_limits<double>::infinity(), numeric_limits<double>::infinity()};
    Vec3 bmax{-numeric_limits<double>::infinity(), -numeric_limits<double>::infinity(), -numeric_limits<double>::infinity()};

    void updateBounds(const Vec3& p) {
        bmin.x = min(bmin.x, p.x); bmin.y = min(bmin.y, p.y); bmin.z = min(bmin.z, p.z);
        bmax.x = max(bmax.x, p.x); bmax.y = max(bmax.y, p.y); bmax.z = max(bmax.z, p.z);
    }

    double volume() const {
        double vol = 0.0;
        for (const auto& f : faces) {
            const Vec3& a = vertices[f[0]];
            const Vec3& b = vertices[f[1]];
            const Vec3& c = vertices[f[2]];
            vol += dot(a, cross(b, c));
        }
        return vol / 6.0;
    }

    void normalizeWinding() {
        if (volume() < 0.0) {
            for (auto& f : faces) swap(f[1], f[2]);
        }
    }
};

static vector<Polygon> meshToPolygons(const Mesh& m) {
    vector<Polygon> polys;
    polys.reserve(m.faces.size());
    for (const auto& f : m.faces) {
        Polygon p;
        p.vertices = {m.vertices[f[0]], m.vertices[f[1]], m.vertices[f[2]]};
        p.computePlane();
        p = cleanPolygon(p);
        if (p.vertices.size() >= 3) polys.push_back(move(p));
    }
    return polys;
}

static bool bboxDisjoint(const Mesh& a, const Mesh& b) {
    if (a.bmax.x < b.bmin.x - gEps || b.bmax.x < a.bmin.x - gEps) return true;
    if (a.bmax.y < b.bmin.y - gEps || b.bmax.y < a.bmin.y - gEps) return true;
    if (a.bmax.z < b.bmin.z - gEps || b.bmax.z < a.bmin.z - gEps) return true;
    return false;
}

struct QuantKey {
    long long x, y, z;
    bool operator==(const QuantKey& o) const noexcept {
        return x == o.x && y == o.y && z == o.z;
    }
};

struct QuantHash {
    size_t operator()(const QuantKey& k) const noexcept {
        size_t h = std::hash<long long>{}(k.x);
        h ^= std::hash<long long>{}(k.y) + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        h ^= std::hash<long long>{}(k.z) + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        return h;
    }
};

static QuantKey makeKey(const Vec3& p, double cell) {
    return {
        (long long) llround(p.x / cell),
        (long long) llround(p.y / cell),
        (long long) llround(p.z / cell)
    };
}

static int addVertex(
    const Vec3& p,
    vector<Vec3>& outVerts,
    unordered_map<QuantKey, vector<int>, QuantHash>& buckets,
    double cell)
{
    QuantKey key = makeKey(p, cell);
    auto& bucket = buckets[key];
    for (int idx : bucket) {
        const Vec3& q = outVerts[idx];
        if (fabs(p.x - q.x) <= cell && fabs(p.y - q.y) <= cell && fabs(p.z - q.z) <= cell) {
            return idx;
        }
    }
    int idx = (int)outVerts.size();
    outVerts.push_back(p);
    bucket.push_back(idx);
    return idx;
}

static void appendPolygonAsTriangles(
    const Polygon& poly,
    vector<Vec3>& outVerts,
    vector<array<int, 3>>& outFaces,
    unordered_map<QuantKey, vector<int>, QuantHash>& buckets,
    double cell)
{
    if (poly.vertices.size() < 3) return;

    vector<int> ids;
    ids.reserve(poly.vertices.size());
    for (const auto& p : poly.vertices) {
        ids.push_back(addVertex(p, outVerts, buckets, cell));
    }

    // Remove accidental duplicate consecutive indices.
    vector<int> cleanIds;
    cleanIds.reserve(ids.size());
    for (int id : ids) {
        if (cleanIds.empty() || cleanIds.back() != id) cleanIds.push_back(id);
    }
    if (cleanIds.size() >= 2 && cleanIds.front() == cleanIds.back()) cleanIds.pop_back();
    if (cleanIds.size() < 3) return;

    // Fan triangulate convex polygons.
    for (size_t i = 1; i + 1 < cleanIds.size(); ++i) {
        int a = cleanIds[0], b = cleanIds[i], c = cleanIds[i + 1];
        const Vec3& pa = outVerts[a];
        const Vec3& pb = outVerts[b];
        const Vec3& pc = outVerts[c];
        Vec3 cr = cross(pb - pa, pc - pa);
        if (norm(cr) <= cell * 1e-3) continue;
        outFaces.push_back({a, b, c});
    }
}

static void outputMesh(
    const vector<Vec3>& verts,
    const vector<array<int, 3>>& faces)
{
    cout << verts.size() << ' ' << faces.size() << '\n';
    cout.setf(std::ios::fixed);
    cout << setprecision(10);
    for (const auto& v : verts) {
        cout << v.x << ' ' << v.y << ' ' << v.z << '\n';
    }
    for (const auto& f : faces) {
        cout << f[0] << ' ' << f[1] << ' ' << f[2] << '\n';
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(cin >> VA >> FA >> VB >> FB)) return 0;

    Mesh A, B;
    A.vertices.resize(VA);
    A.faces.resize(FA);
    B.vertices.resize(VB);
    B.faces.resize(FB);

    for (int i = 0; i < VA; ++i) {
        cin >> A.vertices[i].x >> A.vertices[i].y >> A.vertices[i].z;
        A.updateBounds(A.vertices[i]);
    }
    for (int i = 0; i < FA; ++i) {
        cin >> A.faces[i][0] >> A.faces[i][1] >> A.faces[i][2];
    }
    for (int i = 0; i < VB; ++i) {
        cin >> B.vertices[i].x >> B.vertices[i].y >> B.vertices[i].z;
        B.updateBounds(B.vertices[i]);
    }
    for (int i = 0; i < FB; ++i) {
        cin >> B.faces[i][0] >> B.faces[i][1] >> B.faces[i][2];
    }

    // Set a geometry-scale epsilon.
    Vec3 globalMin{
        min(A.bmin.x, B.bmin.x),
        min(A.bmin.y, B.bmin.y),
        min(A.bmin.z, B.bmin.z)
    };
    Vec3 globalMax{
        max(A.bmax.x, B.bmax.x),
        max(A.bmax.y, B.bmax.y),
        max(A.bmax.z, B.bmax.z)
    };
    double span = max({globalMax.x - globalMin.x, globalMax.y - globalMin.y, globalMax.z - globalMin.z, 1.0});
    gEps = 1e-9 * span;

    A.normalizeWinding();
    B.normalizeWinding();

    if (bboxDisjoint(A, B)) {
        // Fast path: disjoint solids. Concatenate the original meshes.
        vector<Vec3> outVerts = A.vertices;
        vector<array<int, 3>> outFaces = A.faces;
        int offset = (int)outVerts.size();
        outVerts.insert(outVerts.end(), B.vertices.begin(), B.vertices.end());
        for (auto f : B.faces) {
            outFaces.push_back({f[0] + offset, f[1] + offset, f[2] + offset});
        }
        outputMesh(outVerts, outFaces);
        return 0;
    }

    vector<Polygon> polysA = meshToPolygons(A);
    vector<Polygon> polysB = meshToPolygons(B);

    Node a(polysA);
    Node b(polysB);

    // Classic BSP union.
    a.clipTo(b);
    b.clipTo(a);
    b.invert();
    b.clipTo(a);
    b.invert();
    a.build(b.allPolygons());

    vector<Polygon> resultPolys = a.allPolygons();

    vector<Vec3> outVerts;
    vector<array<int, 3>> outFaces;
    unordered_map<QuantKey, vector<int>, QuantHash> buckets;
    buckets.reserve(resultPolys.size() * 2 + 16);

    double dedupCell = max(gEps * 10.0, 1e-12 * span);
    for (const auto& p : resultPolys) {
        Polygon q = cleanPolygon(p);
        if (q.vertices.size() < 3) continue;
        appendPolygonAsTriangles(q, outVerts, outFaces, buckets, dedupCell);
    }

    outputMesh(outVerts, outFaces);
    return 0;
}
