; SPIR-V
; Version: 1.0
; Bound: 100
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %uv %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %uv Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v4float = OpTypeVector %float 4
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Function_v2float = OpTypePointer Function %v2float
%ptr_Function_int = OpTypePointer Function %int

%uv = OpVariable %ptr_Input_v2float Input
%outColor = OpVariable %ptr_Output_v4float Output

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0
%float_3 = OpConstant %float 3.0
%float_4 = OpConstant %float 4.0
%float_half = OpConstant %float 0.5
%float_n1 = OpConstant %float -1.0
%float_n2 = OpConstant %float -2.0
%float_20 = OpConstant %float 20.0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_20 = OpConstant %int 20

%main = OpFunction %void None %fn_void
%entry = OpLabel
%c = OpVariable %ptr_Function_v2float Function
%z = OpVariable %ptr_Function_v2float Function
%iter = OpVariable %ptr_Function_int Function

%uv_val = OpLoad %v2float %uv
%uv_x = OpCompositeExtract %float %uv_val 0
%uv_y = OpCompositeExtract %float %uv_val 1
%scaled_x = OpFMul %float %uv_x %float_3
%cx = OpFAdd %float %scaled_x %float_n2
%scaled_y = OpFMul %float %uv_y %float_2
%cy = OpFAdd %float %scaled_y %float_n1
%c_val = OpCompositeConstruct %v2float %cx %cy
OpStore %c %c_val
%z0 = OpCompositeConstruct %v2float %float_0 %float_0
OpStore %z %z0
OpStore %iter %int_0
OpBranch %loop

%loop = OpLabel
OpLoopMerge %merge %continue None
OpBranch %check

%check = OpLabel
%iter_val = OpLoad %int %iter
%cond_iter = OpSLessThan %bool %iter_val %int_20
%z_val = OpLoad %v2float %z
%z_x = OpCompositeExtract %float %z_val 0
%z_y = OpCompositeExtract %float %z_val 1
%zx2 = OpFMul %float %z_x %z_x
%zy2 = OpFMul %float %z_y %z_y
%len2 = OpFAdd %float %zx2 %zy2
%cond_escape = OpFOrdGreaterThan %bool %len2 %float_4
%not_escape = OpLogicalNot %bool %cond_escape
%cond_run = OpLogicalAnd %bool %cond_iter %not_escape
OpSelectionMerge %check_merge None
OpBranchConditional %cond_run %body %check_merge

%body = OpLabel
%z_prev = OpLoad %v2float %z
%c_prev = OpLoad %v2float %c
%zx = OpCompositeExtract %float %z_prev 0
%zy = OpCompositeExtract %float %z_prev 1
%cx2 = OpCompositeExtract %float %c_prev 0
%cy2 = OpCompositeExtract %float %c_prev 1
%zx_sq = OpFMul %float %zx %zx
%zy_sq = OpFMul %float %zy %zy
%xy = OpFMul %float %zx %zy
%new_x_base = OpFSub %float %zx_sq %zy_sq
%new_x = OpFAdd %float %new_x_base %cx2
%two_xy = OpFMul %float %float_2 %xy
%new_y = OpFAdd %float %two_xy %cy2
%new_z = OpCompositeConstruct %v2float %new_x %new_y
OpStore %z %new_z
OpBranch %continue

%check_merge = OpLabel
OpBranch %merge

%continue = OpLabel
%iter_cur = OpLoad %int %iter
%iter_next = OpIAdd %int %iter_cur %int_1
OpStore %iter %iter_next
OpBranch %loop

%merge = OpLabel
%iter_end = OpLoad %int %iter
%iter_f = OpConvertSToF %float %iter_end
%t = OpFDiv %float %iter_f %float_20
%half_t = OpFMul %float %t %float_half
%one_minus_t = OpFSub %float %float_1 %t
%color = OpCompositeConstruct %v4float %t %half_t %one_minus_t %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
