OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%funct = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%uint_6 = OpConstant %uint 6
%float_0 = OpConstant %float 0
%float_0_2 = OpConstant %float 0.2
%float_0_33333334 = OpConstant %float 0.33333334
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%arr_v3 = OpTypeArray %v3float %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %funct
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p2_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_2
%p4_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_4
%p0 = OpLoad %v3float %p0_ptr
%p2 = OpLoad %v3float %p2_ptr
%p4 = OpLoad %v3float %p4_ptr

%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
%c0 = OpLoad %v3float %c0_ptr
%c2 = OpLoad %v3float %c2_ptr
%c4 = OpLoad %v3float %c4_ptr

%e1 = OpFSub %v3float %p2 %p0
%e2 = OpFSub %v3float %p4 %p0

%e1x = OpCompositeExtract %float %e1 0
%e1y = OpCompositeExtract %float %e1 1
%e1z = OpCompositeExtract %float %e1 2
%e2x = OpCompositeExtract %float %e2 0
%e2y = OpCompositeExtract %float %e2 1
%e2z = OpCompositeExtract %float %e2 2

%cx1 = OpFMul %float %e1y %e2z
%cx2 = OpFMul %float %e1z %e2y
%cx = OpFSub %float %cx1 %cx2
%cy1 = OpFMul %float %e1z %e2x
%cy2 = OpFMul %float %e1x %e2z
%cy = OpFSub %float %cy1 %cy2
%cz1 = OpFMul %float %e1x %e2y
%cz2 = OpFMul %float %e1y %e2x
%cz = OpFSub %float %cz1 %cz2
%cross = OpCompositeConstruct %v3float %cx %cy %cz
%normalVec = OpExtInst %v3float %glsl Normalize %cross

%lightConst = OpCompositeConstruct %v3float %float_1 %float_1 %float_1
%lightDir = OpExtInst %v3float %glsl Normalize %lightConst

%ndot = OpDot %float %normalVec %lightDir
%diffuse = OpExtInst %float %glsl FMax %ndot %float_0

%sum0 = OpFAdd %v3float %c0 %c2
%sum1 = OpFAdd %v3float %sum0 %c4
%base = OpFMul %v3float %sum1 %float_0_33333334

%diffScaled = OpFMul %float %diffuse %float_0_8
%factor = OpFAdd %float %float_0_2 %diffScaled
%finalColor = OpVectorTimesScalar %v3float %base %factor

%pos0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
%pos0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0val = OpLoad %v4float %pos0
OpStore %pos0_ptr %pos0val
OpStore %outColor %finalColor
OpEmitVertex

%pos1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
%pos1 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos1val = OpLoad %v4float %pos1
OpStore %pos1_ptr %pos1val
OpStore %outColor %finalColor
OpEmitVertex

%pos2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
%pos2 = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos2val = OpLoad %v4float %pos2
OpStore %pos2_ptr %pos2val
OpStore %outColor %finalColor
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
