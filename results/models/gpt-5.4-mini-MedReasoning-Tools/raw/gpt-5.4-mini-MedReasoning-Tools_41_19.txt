; SPIR-V
; Version: 1.0
; Generator: Codex
; Bound: 0

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %color %ubo
OpExecutionMode %main OriginUpperLeft

OpSource GLSL 450

OpName %main "main"
OpName %outColor "outColor"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %ubo "ubo"
OpName %UBO "UBO"

OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 4
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%func_void = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v4float = OpTypePointer Output %v4float

%float_0 = OpConstant %float 0
%float_0_05 = OpConstant %float 0.05
%float_0_1 = OpConstant %float 0.1
%float_1 = OpConstant %float 1
%float_32 = OpConstant %float 32
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%v3float_0 = OpConstantComposite %v3float %float_0 %float_0 %float_0
%v3float_1 = OpConstantComposite %v3float %float_1 %float_1 %float_1
%ambient = OpConstantComposite %v3float %float_0_1 %float_0_05 %float_0_05

%GLSLstd450 = OpExtInstImport "GLSL.std.450"

%outColor = OpVariable %ptr_Output_v4float Output
%worldPos = OpVariable %ptr_Input_v3float Input
%normal = OpVariable %ptr_Input_v3float Input
%color = OpVariable %ptr_Input_v3float Input
%ubo = OpVariable %ptr_Uniform_UBO Uniform

%main = OpFunction %void None %func_void
%entry = OpLabel

%worldPosVal = OpLoad %v3float %worldPos
%normalVal = OpLoad %v3float %normal
%colorVal = OpLoad %v3float %color

%N = OpExtInst %v3float %GLSLstd450 Normalize %normalVal

%lightPosPtr = OpAccessChain %ptr_Uniform_v4float %ubo %int_3
%lightPos4 = OpLoad %v4float %lightPosPtr
%lightPos3 = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2
%Lvec = OpFSub %v3float %lightPos3 %worldPosVal
%L = OpExtInst %v3float %GLSLstd450 Normalize %Lvec

%cameraPosPtr = OpAccessChain %ptr_Uniform_v4float %ubo %int_4
%cameraPos4 = OpLoad %v4float %cameraPosPtr
%cameraPos3 = OpVectorShuffle %v3float %cameraPos4 %cameraPos4 0 1 2
%Vvec = OpFSub %v3float %cameraPos3 %worldPosVal
%V = OpExtInst %v3float %GLSLstd450 Normalize %Vvec

%ndotl = OpDot %float %N %L
%diffuseFactor = OpExtInst %float %GLSLstd450 FMax %ndotl %float_0
%diffuse = OpVectorTimesScalar %v3float %colorVal %diffuseFactor

%negL = OpFNegate %v3float %L
%R = OpExtInst %v3float %GLSLstd450 Reflect %negL %N
%rdotv = OpDot %float %R %V
%specBase = OpExtInst %float %GLSLstd450 FMax %rdotv %float_0
%specFactor = OpExtInst %float %GLSLstd450 Pow %specBase %float_32
%specular = OpVectorTimesScalar %v3float %v3float_1 %specFactor

%lit = OpFAdd %v3float %ambient %diffuse
%lit2 = OpFAdd %v3float %lit %specular
%clamped = OpExtInst %v3float %GLSLstd450 FClamp %lit2 %v3float_0 %v3float_1
%outVec = OpCompositeConstruct %v4float %clamped %float_1
OpStore %outColor %outVec
OpReturn
OpFunctionEnd
