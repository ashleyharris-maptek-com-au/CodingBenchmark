import math
from statistics import median


_S = {
    "inited": False,
    "prev_t": None,
    "prev": {},
    "trim_bias": 0.0,
    "elevator_i": 0.0,
    "aileron_i": 0.0,
    "rudder_i": 0.0,
    "speed_i": 0.0,
    "hdg_i": 0.0,
    "alt_i": 0.0,
    "flt": {},
    "last_controls": {
        "elevator": 0.0,
        "aileron": 0.0,
        "rudder": 0.0,
        "throttle_1": 0.55,
        "throttle_2": 0.55,
        "elevator_trim": 0.0,
        "flaps": 0,
        "speedbrake": 0.0,
        "gear": 1,
    },
}


def _clamp(x, lo, hi):
    return lo if x < lo else hi if x > hi else x


def _finite(x, default=0.0):
    try:
        x = float(x)
    except Exception:
        return default
    return x if math.isfinite(x) else default


def _wrap180(deg):
    return (deg + 180.0) % 360.0 - 180.0


def _wrap360(deg):
    return deg % 360.0


def _angle_mean(values, weights=None):
    xs = []
    ys = []
    if weights is None:
        weights = [1.0] * len(values)
    for a, w in zip(values, weights):
        if a is None:
            continue
        r = math.radians(_wrap360(a))
        xs.append(math.cos(r) * w)
        ys.append(math.sin(r) * w)
    if not xs:
        return None
    ang = math.degrees(math.atan2(sum(ys), sum(xs)))
    return _wrap360(ang)


def _robust_scalar(values, valids=None, prior=None, step_limit=None):
    items = []
    for i, v in enumerate(values):
        if v is None:
            continue
        fv = _finite(v, None)
        if fv is None:
            continue
        ok = True if valids is None else bool(valids[i])
        items.append((fv, ok))
    if not items:
        return prior if prior is not None else 0.0
    vals = [v for v, _ in items]
    center = median(vals)
    if prior is None:
        prior = center
    if step_limit is None:
        step_limit = max(1.0, abs(center - prior))
    scored = []
    for v, ok in items:
        score = abs(v - center) + 0.35 * abs(v - prior) / step_limit
        if not ok:
            score += 0.2 * step_limit
        scored.append((score, v))
    scored.sort(key=lambda t: t[0])
    return scored[0][1]


def _fuse_angle(values, valids=None, prior=None):
    items = []
    for i, v in enumerate(values):
        if v is None:
            continue
        fv = _finite(v, None)
        if fv is None:
            continue
        ok = True if valids is None else bool(valids[i])
        items.append((fv, ok))
    if not items:
        return prior
    if prior is None:
        prior = items[0][0]
    pr = math.radians(_wrap360(prior))
    px, py = math.cos(pr), math.sin(pr)
    scored = []
    for v, ok in items:
        a = math.radians(_wrap360(v))
        x, y = math.cos(a), math.sin(a)
        d = math.atan2(y - py, x - px)
        score = abs(d)
        if not ok:
            score += 0.35
        scored.append((score, v))
    scored.sort(key=lambda t: t[0])
    return scored[0][1]


def _smooth(name, value, alpha=0.25):
    prev = _S["flt"].get(name)
    if prev is None:
        out = value
    else:
        out = prev + alpha * (value - prev)
    _S["flt"][name] = out
    return out


def _sample(state, key, default=0.0):
    return _finite(state.get(key, default), default)


def _sensor_triplet(state, base, suffix, default=0.0):
    return [_sample(state, f"{base}_{i}_{suffix}", default) for i in (1, 2, 3)]


def _valid_triplet(state, base):
    return [bool(state.get(f"{base}_{i}_valid", True)) for i in (1, 2, 3)]


def _baro_altitude(state):
    vals = _sensor_triplet(state, "baro_alt", "ft", 0.0)
    valids = _valid_triplet(state, "baro_alt")
    prior = _S["prev"].get("alt_ft")
    return _robust_scalar(vals, valids, prior=prior, step_limit=250.0)


def _gps_altitude(state):
    vals = _sensor_triplet(state, "gps", "alt_ft", 0.0)
    valids = _valid_triplet(state, "gps")
    prior = _S["prev"].get("alt_ft")
    return _robust_scalar(vals, valids, prior=prior, step_limit=400.0)


def _radar_altitude(state):
    vals = _sensor_triplet(state, "radar_alt", "ft", 0.0)
    valids = _valid_triplet(state, "radar_alt")
    return _robust_scalar(vals, valids, prior=_S["prev"].get("radar_alt_ft", 0.0), step_limit=150.0)


def _ias(state):
    vals = _sensor_triplet(state, "ias", "kt", 0.0)
    valids = _valid_triplet(state, "ias")
    prior = _S["prev"].get("ias_kt")
    return _robust_scalar(vals, valids, prior=prior, step_limit=30.0)


def _mach(state):
    return _robust_scalar(_sensor_triplet(state, "mach", "", 0.0), None, prior=_S["prev"].get("mach", 0.0), step_limit=0.05)


def _aoa(state):
    vals = _sensor_triplet(state, "aoa", "deg", 0.0)
    valids = _valid_triplet(state, "aoa")
    prior = _S["prev"].get("aoa_deg")
    return _robust_scalar(vals, valids, prior=prior, step_limit=4.0)


def _irs_best(state, field, suffix, step_limit):
    vals = _sensor_triplet(state, f"irs_{field}", suffix, 0.0)
    valids = _valid_triplet(state, f"irs_{field}")
    return _robust_scalar(vals, valids, prior=_S["prev"].get(f"irs_{field}_{suffix}"), step_limit=step_limit)


def _engine_health(state, idx):
    run = bool(state.get(f"engine_{idx}_running", False))
    fire = bool(state.get(f"engine_{idx}_fire", False))
    n1 = _sample(state, f"engine_{idx}_n1_pct", 0.0)
    n2 = _sample(state, f"engine_{idx}_n2_pct", 0.0)
    thrust = _sample(state, f"engine_{idx}_thrust_pct", 0.0)
    egt = _sample(state, f"engine_{idx}_egt_c", 0.0)
    score = 1.0
    if fire:
        score -= 1.0
    if not run and n1 < 15.0 and n2 < 15.0 and thrust < 5.0:
        score -= 0.9
    if n1 < 20.0 and thrust < 5.0:
        score -= 0.5
    if egt > 980.0 and thrust < 20.0:
        score -= 0.15
    return _clamp(score, 0.0, 1.0)


def _heading_source(state):
    candidates = []
    names = [
        ("magnetic_heading_deg", 1.0),
        ("true_heading_deg", 0.8),
        ("track_deg", 0.8),
        ("irs_1_heading_deg", 0.7),
        ("irs_2_heading_deg", 0.7),
        ("irs_3_heading_deg", 0.7),
        ("gps_1_track_deg", 0.5),
        ("gps_2_track_deg", 0.5),
        ("gps_3_track_deg", 0.5),
    ]
    for key, w in names:
        if key in state:
            v = _finite(state.get(key), None)
            if v is not None:
                candidates.append((v, w))
    if not candidates:
        return _S["prev"].get("hdg_deg", 0.0)
    hdg = _angle_mean([v for v, _ in candidates], [w for _, w in candidates])
    return hdg if hdg is not None else _S["prev"].get("hdg_deg", 0.0)


def _vertical_speed(state, alt_ft, dt):
    irs_vs = _robust_scalar(
        [_sample(state, "irs_1_vs_fpm", 0.0), _sample(state, "irs_2_vs_fpm", 0.0), _sample(state, "irs_3_vs_fpm", 0.0)],
        _valid_triplet(state, "irs"),
        prior=_S["prev"].get("vs_fpm"),
        step_limit=4000.0,
    )
    prev_alt = _S["prev"].get("alt_ft")
    if prev_alt is None or dt <= 1e-4:
        deriv = irs_vs
    else:
        deriv = (alt_ft - prev_alt) / dt * 60.0
    if irs_vs is None:
        vs = deriv
    else:
        vs = 0.7 * irs_vs + 0.3 * deriv
    return _smooth("vs_fpm", vs, 0.25)


def _roll_pitch(state):
    pitch = _irs_best(state, 1, "pitch_deg", 20.0)
    if pitch is None:
        pitch = _irs_best(state, 2, "pitch_deg", 20.0)
    if pitch is None:
        pitch = _irs_best(state, 3, "pitch_deg", 20.0)
    roll = _irs_best(state, 1, "roll_deg", 20.0)
    if roll is None:
        roll = _irs_best(state, 2, "roll_deg", 20.0)
    if roll is None:
        roll = _irs_best(state, 3, "roll_deg", 20.0)
    return _smooth("pitch_deg", pitch or 0.0, 0.35), _smooth("roll_deg", roll or 0.0, 0.35)


def _attitude_rates(state):
    pr = _robust_scalar(
        [_sample(state, "gyro_1_pitch_rate_dps", 0.0), _sample(state, "gyro_2_pitch_rate_dps", 0.0), _sample(state, "gyro_3_pitch_rate_dps", 0.0)],
        _valid_triplet(state, "gyro"),
        prior=_S["prev"].get("pitch_rate_dps"),
        step_limit=20.0,
    )
    rr = _robust_scalar(
        [_sample(state, "gyro_1_roll_rate_dps", 0.0), _sample(state, "gyro_2_roll_rate_dps", 0.0), _sample(state, "gyro_3_roll_rate_dps", 0.0)],
        _valid_triplet(state, "gyro"),
        prior=_S["prev"].get("roll_rate_dps"),
        step_limit=20.0,
    )
    yr = _robust_scalar(
        [_sample(state, "gyro_1_yaw_rate_dps", 0.0), _sample(state, "gyro_2_yaw_rate_dps", 0.0), _sample(state, "gyro_3_yaw_rate_dps", 0.0)],
        _valid_triplet(state, "gyro"),
        prior=_S["prev"].get("yaw_rate_dps"),
        step_limit=20.0,
    )
    return pr, rr, yr


def _weights_on_wheels(state):
    return sum(
        1 for k in ("weight_on_wheels_left", "weight_on_wheels_nose", "weight_on_wheels_right") if bool(state.get(k, False))
    )


def _target_heading_cmd(target_hdg, current_hdg, current_roll, ias_kt):
    err = _wrap180(target_hdg - current_hdg)
    bank_limit = 25.0
    if ias_kt < 160.0:
        bank_limit = 15.0
    if abs(current_roll) > 35.0:
        bank_limit = min(bank_limit, 12.0)
    bank_cmd = _clamp(err * 0.35, -bank_limit, bank_limit)
    return bank_cmd, err


def _target_vertical_speed(target_alt, alt_ft, ias_kt, vs_fpm, aoa_deg, flaps, gear_down):
    err = target_alt - alt_ft
    climb_limit = 2500.0
    descend_limit = 2500.0
    if gear_down or flaps > 5:
        climb_limit = 1800.0
        descend_limit = 1800.0
    if alt_ft < 3000.0:
        climb_limit = min(climb_limit, 1200.0)
        descend_limit = min(descend_limit, 1200.0)
    if abs(err) < 150.0:
        return _clamp(-vs_fpm * 0.35 + err * 1.5, -700.0, 700.0)
    if abs(err) < 1000.0:
        return _clamp(err * 1.1 - vs_fpm * 0.15, -1200.0, 1200.0)
    desired = _clamp(err * 1.8, -descend_limit, climb_limit)
    if ias_kt < 160.0 or aoa_deg > 11.0:
        desired = min(desired, 300.0)
    return desired


def _flight_phase(alt_ft, target_alt, ias_kt, vs_fpm, gear_down, wow):
    if wow >= 2 and alt_ft < 50.0:
        return "ground"
    if wow >= 1 and ias_kt < 80.0:
        return "roll"
    if target_alt < 1000.0 or (alt_ft < 3000.0 and vs_fpm < -300.0):
        return "approach"
    if alt_ft < 8000.0 and abs(target_alt - alt_ft) < 5000.0:
        return "terminal"
    if abs(target_alt - alt_ft) > 1500.0:
        return "climb_descent"
    return "cruise"


def autopilot_step(state: dict, dt: float) -> dict:
    dt = _finite(state.get("dt", dt), dt if dt and dt > 0 else 0.05)
    dt = max(0.01, dt)

    # Primary fused state
    alt_baro = _baro_altitude(state)
    alt_gps = _gps_altitude(state)
    alt_radar = _radar_altitude(state)
    ias = _ias(state)
    mach = _mach(state)
    aoa = _aoa(state)
    hdg = _heading_source(state)
    pitch, roll = _roll_pitch(state)
    pitch_rate, roll_rate, yaw_rate = _attitude_rates(state)
    vs = _vertical_speed(state, alt_baro, dt)

    # Altitude fusion: baro is primary, GPS sanity-check, radar used only low.
    low_alt = alt_radar is not None and alt_radar < 2500.0
    if alt_gps is None:
        alt_ft = alt_baro
    else:
        alt_ft = 0.75 * alt_baro + 0.25 * alt_gps
    if low_alt and alt_radar is not None:
        alt_ft = 0.6 * alt_ft + 0.4 * max(0.0, alt_radar)
    alt_ft = _smooth("alt_ft", alt_ft, 0.22)

    target_alt = _sample(state, "target_alt_ft", alt_ft)
    target_hdg = _wrap360(_sample(state, "target_hdg_deg", hdg))
    target_ias = _sample(state, "target_ias_kt", ias)
    target_vs = _sample(state, "target_vs_fpm", 0.0)

    ap_engaged = bool(state.get("ap_engaged", True))
    at_engaged = bool(state.get("at_engaged", True))
    wow = _weights_on_wheels(state)
    gear_down = bool(state.get("gear_handle_down", True))
    phase = _flight_phase(alt_ft, target_alt, ias, vs, gear_down, wow)

    # Engine health and asymmetry.
    h1 = _engine_health(state, 1)
    h2 = _engine_health(state, 2)
    t1_raw = _sample(state, "throttle_1_pos", _S["last_controls"]["throttle_1"])
    t2_raw = _sample(state, "throttle_2_pos", _S["last_controls"]["throttle_2"])
    throttle_mean = 0.5 * (t1_raw + t2_raw)
    engine1_good = h1 > 0.5
    engine2_good = h2 > 0.5

    # Speed protection and mode selection.
    stall_guard = 120.0
    if gear_down or phase in ("approach", "ground"):
        stall_guard = 135.0
    if alt_ft < 1000.0:
        stall_guard = 145.0
    overspeed_guard = 330.0 if mach < 0.72 else 300.0
    if mach > 0.79:
        overspeed_guard = min(overspeed_guard, 285.0)

    speed_err = target_ias - ias
    alt_err = target_alt - alt_ft
    if target_vs != 0.0 and abs(alt_err) > 400.0:
        vs_cmd = 0.5 * _target_vertical_speed(target_alt, alt_ft, ias, vs, aoa, 0, gear_down) + 0.5 * target_vs
    else:
        vs_cmd = _target_vertical_speed(target_alt, alt_ft, ias, vs, aoa, 0, gear_down)

    # Energy management: slow if fast, descend if fast, climb if slow.
    if ias < stall_guard:
        vs_cmd = min(vs_cmd, 0.0)
        speed_brake = 0.0
    else:
        speed_brake = 0.0
    if ias > overspeed_guard:
        vs_cmd = min(vs_cmd, -700.0)

    # Heading / lateral.
    bank_cmd, hdg_err = _target_heading_cmd(target_hdg, hdg, roll, ias)
    if phase == "ground":
        bank_cmd = _clamp(hdg_err * 0.15, -8.0, 8.0)

    # Pitch command and elevator.
    pitch_cmd = _clamp((vs_cmd - vs) * 0.0035 + speed_err * 0.015 - (ias - target_ias) * 0.002, -12.0, 15.0)
    if ias < stall_guard + 10.0:
        pitch_cmd = min(pitch_cmd, 4.0)
    if ias > overspeed_guard - 10.0:
        pitch_cmd = min(pitch_cmd, -2.0)
    if aoa > 12.0:
        pitch_cmd = min(pitch_cmd, -4.0)

    pitch_err = pitch_cmd - pitch
    _S["elevator_i"] = _clamp(_S["elevator_i"] + pitch_err * dt * 0.05, -0.4, 0.4)
    elevator = _clamp(0.08 * pitch_err - 0.02 * pitch_rate + _S["elevator_i"] - _S["trim_bias"], -1.0, 1.0)

    # Slow trim adaptation to unload elevator.
    trim_d = _clamp(0.004 * pitch_err + 0.001 * speed_err - 0.002 * elevator, -0.03, 0.03)
    _S["trim_bias"] = _clamp(_S["trim_bias"] + trim_d * dt * 2.0, -1.0, 1.0)
    elevator_trim = _clamp(_S["trim_bias"], -1.0, 1.0)

    roll_err = bank_cmd - roll
    _S["aileron_i"] = _clamp(_S["aileron_i"] + roll_err * dt * 0.04, -0.25, 0.25)
    aileron = _clamp(0.05 * roll_err - 0.015 * roll_rate + _S["aileron_i"], -1.0, 1.0)

    # Rudder from yaw damping, coordinated turn, and asymmetric thrust.
    thrust_imbalance = _sample(state, "engine_1_thrust_pct", 0.0) - _sample(state, "engine_2_thrust_pct", 0.0)
    desired_rudder = (-0.06 * yaw_rate) + 0.015 * roll + 0.003 * thrust_imbalance + 0.004 * bank_cmd + 0.002 * _wrap180(hdg_err)
    if engine1_good and not engine2_good:
        desired_rudder += 0.08 * max(0.0, _sample(state, "engine_1_thrust_pct", 0.0))
    elif engine2_good and not engine1_good:
        desired_rudder -= 0.08 * max(0.0, _sample(state, "engine_2_thrust_pct", 0.0))
    _S["rudder_i"] = _clamp(_S["rudder_i"] + desired_rudder * dt * 0.5, -0.3, 0.3)
    rudder = _clamp(desired_rudder + _S["rudder_i"], -1.0, 1.0)

    # Thrust management.
    base_thrust = throttle_mean
    if not at_engaged:
        base_thrust = throttle_mean
    else:
        if phase == "ground":
            base_thrust = 0.30 if ias < 60.0 else 0.40
        elif phase == "roll":
            base_thrust = 0.95
        elif phase == "approach":
            base_thrust = 0.35 + 0.0012 * max(0.0, target_alt - alt_ft) + 0.0035 * speed_err
        elif phase == "terminal":
            base_thrust = 0.55 + 0.002 * speed_err + 0.00015 * max(0.0, alt_err)
        elif phase == "climb_descent":
            base_thrust = 0.55 + 0.0025 * speed_err + 0.00018 * alt_err
        else:
            base_thrust = 0.48 + 0.0028 * speed_err + 0.00012 * alt_err

    if ias < stall_guard + 8.0 or aoa > 11.0:
        base_thrust = max(base_thrust, 0.85)
    if ias > overspeed_guard + 8.0:
        base_thrust = min(base_thrust, 0.15)
    if mach > 0.81:
        base_thrust = min(base_thrust, 0.18)
    if abs(alt_err) < 200.0 and abs(vs) < 250.0:
        base_thrust = min(base_thrust, 0.55)
    base_thrust = _clamp(base_thrust, 0.0, 1.0)

    # Engine-out handling and fire protection.
    if bool(state.get("engine_1_fire", False)):
        t1 = 0.0
        t2 = _clamp(base_thrust + 0.12, 0.0, 1.0)
    elif bool(state.get("engine_2_fire", False)):
        t1 = _clamp(base_thrust + 0.12, 0.0, 1.0)
        t2 = 0.0
    elif engine1_good and not engine2_good:
        t1 = _clamp(base_thrust + 0.10, 0.0, 1.0)
        t2 = 0.0
    elif engine2_good and not engine1_good:
        t1 = 0.0
        t2 = _clamp(base_thrust + 0.10, 0.0, 1.0)
    else:
        asym = _clamp(0.10 * rudder + 0.05 * roll / 30.0, -0.12, 0.12)
        t1 = _clamp(base_thrust - asym, 0.0, 1.0)
        t2 = _clamp(base_thrust + asym, 0.0, 1.0)

    # High-altitude / high-speed protection.
    speedbrake = 0.0
    if ias > overspeed_guard + 15.0 or mach > 0.82:
        speedbrake = _clamp((ias - overspeed_guard) / 60.0 + (mach - 0.82) * 2.0, 0.0, 1.0)
        pitch_cmd = min(pitch_cmd, -3.0)
    elif phase in ("approach", "terminal") and ias > target_ias + 20.0:
        speedbrake = _clamp((ias - target_ias - 20.0) / 70.0, 0.0, 0.6)

    # Flaps and gear logic.
    if phase == "ground":
        flaps = 0
        gear = 1
    elif phase == "roll":
        flaps = 1
        gear = 1
    elif phase == "approach":
        gear = 1
        if ias < 150.0:
            flaps = 40
        elif ias < 180.0:
            flaps = 30
        elif ias < 210.0:
            flaps = 20
        else:
            flaps = 10
    elif phase == "terminal":
        gear = 1 if alt_ft < 2500.0 else 0
        if ias < 140.0:
            flaps = 30
        elif ias < 180.0:
            flaps = 20
        elif ias < 220.0:
            flaps = 10
        else:
            flaps = 0
    else:
        gear = 0 if alt_ft > 1200.0 and ias > 130.0 else 1
        flaps = 0

    if alt_ft > 1000.0 and ias > 220.0:
        flaps = min(flaps, 5)
    if alt_ft > 10000.0:
        flaps = 0
    if ias > 240.0:
        flaps = 0
    if ias < 120.0 and alt_ft < 5000.0 and target_alt < 5000.0:
        flaps = max(flaps, 10)

    # Final protections for stalls / overspeed.
    if aoa > 13.5 or ias < stall_guard - 10.0:
        elevator = min(elevator, -0.15)
        speedbrake = 0.0
        if alt_ft > 80.0:
            flaps = max(flaps, 10)
    if ias > overspeed_guard + 25.0 or mach > 0.825:
        elevator = max(elevator, -0.35)
        speedbrake = max(speedbrake, 0.4)

    # If AP is off, keep the last trimmed attitude and avoid abrupt output.
    if not ap_engaged:
        elevator = 0.0
        aileron = 0.0
        rudder = 0.0
        t1 = t1_raw
        t2 = t2_raw
        speedbrake = 0.0
        flaps = int(state.get("flaps_pos_left_deg", 0.0))
        gear = int(bool(state.get("gear_handle_down", True)))

    controls = {
        "elevator": _clamp(elevator, -1.0, 1.0),
        "aileron": _clamp(aileron, -1.0, 1.0),
        "rudder": _clamp(rudder, -1.0, 1.0),
        "throttle_1": _clamp(t1, 0.0, 1.0),
        "throttle_2": _clamp(t2, 0.0, 1.0),
        "elevator_trim": _clamp(elevator_trim, -1.0, 1.0),
        "flaps": int(_clamp(flaps, 0, 40)),
        "speedbrake": _clamp(speedbrake, 0.0, 1.0),
        "gear": int(_clamp(gear, 0, 1)),
    }

    # Persist useful state for the next call.
    _S["prev"] = {
        "alt_ft": alt_ft,
        "radar_alt_ft": alt_radar,
        "ias_kt": ias,
        "mach": mach,
        "aoa_deg": aoa,
        "hdg_deg": hdg,
        "pitch_deg": pitch,
        "roll_deg": roll,
        "vs_fpm": vs,
        "pitch_rate_dps": pitch_rate,
        "roll_rate_dps": roll_rate,
