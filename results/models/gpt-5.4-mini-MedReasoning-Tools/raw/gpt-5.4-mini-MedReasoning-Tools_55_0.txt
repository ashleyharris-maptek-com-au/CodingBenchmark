static inline int file_of(int sq) { return sq & 7; }
static inline int rank_of(int sq) { return sq >> 3; }
static inline bool on_board(int sq) { return sq >= 0 && sq < 64; }
static inline int mirror_sq(int sq) { return sq ^ 56; }
static inline int color_index(int side) { return side == 1 ? 0 : 1; }
static inline int abs_piece(int p) { return p < 0 ? -p : p; }

static inline int piece_index(int piece) {
    if (piece > 0) return piece - 1;
    return 6 + (-piece - 1);
}

static inline int piece_value(int piece) {
    switch (abs_piece(piece)) {
        case PAWN: return 100;
        case KNIGHT: return 320;
        case BISHOP: return 330;
        case ROOK: return 500;
        case QUEEN: return 900;
        case KING: return 0;
        default: return 0;
    }
}

static inline int sq_from_string(const string& s) {
    return (s[0] - 'a') + (s[1] - '1') * 8;
}

static inline string square_to_string(int sq) {
    string s(2, ' ');
    s[0] = char('a' + file_of(sq));
    s[1] = char('1' + rank_of(sq));
    return s;
}

static void init_zobrist() {
    for (int p = 0; p < 12; ++p) {
        for (int sq = 0; sq < 64; ++sq) {
            ZPiece[p][sq] = rng();
        }
    }
    for (int i = 0; i < 16; ++i) ZCastle[i] = rng();
    for (int i = 0; i < 8; ++i) ZEp[i] = rng();
    ZSide = rng();
}

static uint64_t compute_hash(const Position& pos) {
    uint64_t h = 0;
    for (int sq = 0; sq < 64; ++sq) {
        int p = pos.board[sq];
        if (p != EMPTY) h ^= ZPiece[piece_index(p)][sq];
    }
    h ^= ZCastle[pos.castling];
    if (pos.ep != -1) h ^= ZEp[file_of(pos.ep)];
    if (pos.side == -1) h ^= ZSide;
    return h;
}

static bool is_square_attacked(const Position& pos, int sq, int bySide) {
    if (bySide == 1) {
        int r = rank_of(sq), f = file_of(sq);
        if (r > 0) {
            if (f > 0 && pos.board[sq - 9] == PAWN) return true;
            if (f < 7 && pos.board[sq - 7] == PAWN) return true;
        }
    } else {
        int r = rank_of(sq), f = file_of(sq);
        if (r < 7) {
            if (f > 0 && pos.board[sq + 7] == -PAWN) return true;
            if (f < 7 && pos.board[sq + 9] == -PAWN) return true;
        }
    }

    static const int kOffsets[8] = {15, 17, 10, 6, -15, -17, -10, -6};
    for (int d : kOffsets) {
        int t = sq + d;
        if (!on_board(t)) continue;
        int df = abs(file_of(t) - file_of(sq));
        int dr = abs(rank_of(t) - rank_of(sq));
        if (df + dr != 3 || df == 0 || dr == 0) continue;
        int p = pos.board[t];
        if (p == bySide * KNIGHT) return true;
    }

    static const int kingOffsets[8] = {1, -1, 8, -8, 9, -9, 7, -7};
    for (int d : kingOffsets) {
        int t = sq + d;
        if (!on_board(t)) continue;
        if (abs(file_of(t) - file_of(sq)) > 1 || abs(rank_of(t) - rank_of(sq)) > 1) continue;
        int p = pos.board[t];
        if (p == bySide * KING) return true;
    }

    static const int bishopDirs[4] = {9, 7, -9, -7};
    for (int d : bishopDirs) {
        int t = sq;
        while (true) {
            int n = t + d;
            if (!on_board(n)) break;
            if (abs(file_of(n) - file_of(t)) != 1 || abs(rank_of(n) - rank_of(t)) != 1) break;
            t = n;
            int p = pos.board[t];
            if (p != EMPTY) {
                if (p == bySide * BISHOP || p == bySide * QUEEN) return true;
                break;
            }
        }
    }

    static const int rookDirs[4] = {1, -1, 8, -8};
    for (int d : rookDirs) {
        int t = sq;
        while (true) {
            int n = t + d;
            if (!on_board(n)) break;
            if ((d == 1 || d == -1) && rank_of(n) != rank_of(t)) break;
            t = n;
            int p = pos.board[t];
            if (p != EMPTY) {
                if (p == bySide * ROOK || p == bySide * QUEEN) return true;
                break;
            }
        }
    }

    return false;
}

static int evaluate(const Position& pos) {
    int score = 0;
    int whiteBishops = 0;
    int blackBishops = 0;

    for (int sq = 0; sq < 64; ++sq) {
        int p = pos.board[sq];
        if (p == EMPTY) continue;
        int side = p > 0 ? 1 : -1;
        int absP = abs_piece(p);
        int psq = 0;
        switch (absP) {
            case PAWN: psq = PST_PAWN[side == 1 ? sq : mirror_sq(sq)]; break;
            case KNIGHT: psq = PST_KNIGHT[side == 1 ? sq : mirror_sq(sq)]; break;
            case BISHOP: psq = PST_BISHOP[side == 1 ? sq : mirror_sq(sq)]; break;
            case ROOK: psq = PST_ROOK[side == 1 ? sq : mirror_sq(sq)]; break;
            case QUEEN: psq = PST_QUEEN[side == 1 ? sq : mirror_sq(sq)]; break;
            case KING: psq = PST_KING[side == 1 ? sq : mirror_sq(sq)]; break;
        }
        int v = piece_value(p) + psq;
        score += side * v;
        if (absP == BISHOP) {
            if (side == 1) ++whiteBishops;
            else ++blackBishops;
        }
    }

    if (whiteBishops >= 2) score += 25;
    if (blackBishops >= 2) score -= 25;

    return score;
}

static inline bool same_move(const Move& a, const Move& b) {
    return a.from == b.from && a.to == b.to && a.promo == b.promo && a.flags == b.flags;
}

static bool make_move(Position& pos, const Move& m, Undo& u) {
    u.captured = EMPTY;
    u.castling = pos.castling;
    u.ep = pos.ep;
    u.halfmove = pos.halfmove;
    u.fullmove = pos.fullmove;
    u.kingSqW = pos.kingSqW;
    u.kingSqB = pos.kingSqB;
    u.movedPiece = pos.board[m.from];
    u.hash = pos.hash;

    int us = pos.side;
    int them = -us;
    int moving = pos.board[m.from];
    int capturedSq = m.to;

    if (moving == EMPTY) return false;

    pos.ep = -1;
    if (m.flags & FLAG_DOUBLE) {
        pos.ep = m.from + (us == 1 ? 8 : -8);
    }

    if (m.flags & FLAG_EP) {
        capturedSq = m.to + (us == 1 ? -8 : 8);
        u.captured = pos.board[capturedSq];
        pos.board[capturedSq] = EMPTY;
    } else {
        u.captured = pos.board[m.to];
    }

    pos.board[m.from] = EMPTY;

    int placed = moving;
    if (m.flags & FLAG_PROMO) placed = us * m.promo;

    if (m.flags & FLAG_CASTLE) {
        if (us == 1) {
            if (m.to == 6) {
                pos.board[5] = pos.board[7];
                pos.board[7] = EMPTY;
            } else {
                pos.board[3] = pos.board[0];
                pos.board[0] = EMPTY;
            }
        } else {
            if (m.to == 62) {
                pos.board[61] = pos.board[63];
                pos.board[63] = EMPTY;
            } else {
                pos.board[59] = pos.board[56];
                pos.board[56] = EMPTY;
            }
        }
    }

    pos.board[m.to] = placed;

    if (abs_piece(moving) == KING) {
        if (us == 1) pos.kingSqW = m.to;
        else pos.kingSqB = m.to;
    }

    int newCastling = pos.castling;
    if (moving == KING) {
        if (us == 1) newCastling &= ~(1 | 2);
        else newCastling &= ~(4 | 8);
    } else if (moving == -KING) {
        newCastling &= ~(4 | 8);
    }

    if (moving == ROOK) {
        if (m.from == 0) newCastling &= ~2;
        else if (m.from == 7) newCastling &= ~1;
    } else if (moving == -ROOK) {
        if (m.from == 56) newCastling &= ~8;
        else if (m.from == 63) newCastling &= ~4;
    }

    if (u.captured == ROOK) {
        if (m.to == 0) newCastling &= ~2;
        else if (m.to == 7) newCastling &= ~1;
    } else if (u.captured == -ROOK) {
        if (m.to == 56) newCastling &= ~8;
        else if (m.to == 63) newCastling &= ~4;
    }

    pos.castling = newCastling;
    pos.side = them;
    if (us == -1) ++pos.fullmove;

    if (abs_piece(moving) == PAWN || u.captured != EMPTY) pos.halfmove = 0;
    else ++pos.halfmove;

    pos.hash = compute_hash(pos);

    if (is_square_attacked(pos, us == 1 ? pos.kingSqW : pos.kingSqB, pos.side)) {
        return false;
    }
    return true;
}

static void unmake_move(Position& pos, const Move& m, const Undo& u) {
    int us = -pos.side;
    pos.side = us;
    pos.castling = u.castling;
    pos.ep = u.ep;
    pos.halfmove = u.halfmove;
    pos.fullmove = u.fullmove;
    pos.kingSqW = u.kingSqW;
    pos.kingSqB = u.kingSqB;
    pos.hash = u.hash;

    int moving = u.movedPiece;

    if (m.flags & FLAG_CASTLE) {
        if (us == 1) {
            if (m.to == 6) {
                pos.board[7] = pos.board[5];
                pos.board[5] = EMPTY;
            } else {
                pos.board[0] = pos.board[3];
                pos.board[3] = EMPTY;
            }
        } else {
            if (m.to == 62) {
                pos.board[63] = pos.board[61];
                pos.board[61] = EMPTY;
            } else {
                pos.board[56] = pos.board[59];
                pos.board[59] = EMPTY;
            }
        }
    }

    pos.board[m.from] = moving;
    if (m.flags & FLAG_EP) {
        pos.board[m.to] = EMPTY;
        pos.board[m.to + (us == 1 ? -8 : 8)] = u.captured;
    } else {
        pos.board[m.to] = u.captured;
    }
}

static void add_move(vector<Move>& moves, int from, int to, int flags, int promo = 0) {
    Move m;
    m.from = static_cast<uint8_t>(from);
    m.to = static_cast<uint8_t>(to);
    m.flags = static_cast<uint8_t>(flags);
    m.promo = static_cast<uint8_t>(promo);
    moves.push_back(m);
}

static void generate_moves(const Position& pos, vector<Move>& moves, bool capturesOnly = false) {
    moves.clear();
    int us = pos.side;
    int them = -us;

    for (int sq = 0; sq < 64; ++sq) {
        int p = pos.board[sq];
        if (p == EMPTY || (p > 0) != (us > 0)) continue;
        int absP = abs_piece(p);

        if (absP == PAWN) {
            int dir = us == 1 ? 8 : -8;
            int startRank = us == 1 ? 1 : 6;
            int promoRank = us == 1 ? 6 : 1;
            int one = sq + dir;
            if (!capturesOnly && on_board(one) && pos.board[one] == EMPTY) {
                if (rank_of(sq) == promoRank) {
                    add_move(moves, sq, one, FLAG_PROMO, KNIGHT);
                    add_move(moves, sq, one, FLAG_PROMO, BISHOP);
                    add_move(moves, sq, one, FLAG_PROMO, ROOK);
                    add_move(moves, sq, one, FLAG_PROMO, QUEEN);
                } else {
                    add_move(moves, sq, one, 0);
                    if (rank_of(sq) == startRank) {
                        int two = sq + dir * 2;
                        if (on_board(two) && pos.board[two] == EMPTY) {
                            add_move(moves, sq, two, FLAG_DOUBLE);
                        }
                    }
                }
            }
            int caps[2] = {sq + dir + 1, sq + dir - 1};
            for (int to : caps) {
                if (!on_board(to)) continue;
                if (abs(file_of(to) - file_of(sq)) != 1) continue;
                if (to == pos.ep) {
                    add_move(moves, sq, to, FLAG_EP | FLAG_CAPTURE);
                } else {
                    int target = pos.board[to];
                    if (target != EMPTY && (target > 0) != (us > 0)) {
                        if (rank_of(sq) == promoRank) {
                            add_move(moves, sq, to, FLAG_CAPTURE | FLAG_PROMO, KNIGHT);
                            add_move(moves, sq, to, FLAG_CAPTURE | FLAG_PROMO, BISHOP);
                            add_move(moves, sq, to, FLAG_CAPTURE | FLAG_PROMO, ROOK);
                            add_move(moves, sq, to, FLAG_CAPTURE | FLAG_PROMO, QUEEN);
                        } else {
                            add_move(moves, sq, to, FLAG_CAPTURE);
                        }
                    }
                }
            }
        } else if (absP == KNIGHT) {
            static const int offs[8] = {15, 17, 10, 6, -15, -17, -10, -6};
            for (int d : offs) {
                int to = sq + d;
                if (!on_board(to)) continue;
                int df = abs(file_of(to) - file_of(sq));
                int dr = abs(rank_of(to) - rank_of(sq));
                if (df + dr != 3 || df == 0 || dr == 0) continue;
                int target = pos.board[to];
                if (target == EMPTY) {
                    if (!capturesOnly) add_move(moves, sq, to, 0);
                } else if ((target > 0) != (us > 0)) {
                    add_move(moves, sq, to, FLAG_CAPTURE);
                }
            }
        } else if (absP == BISHOP || absP == ROOK || absP == QUEEN) {
            static const int bishopDirs[4] = {9, 7, -9, -7};
            static const int rookDirs[4] = {1, -1, 8, -8};
            const int* dirs = nullptr;
            int dirCount = 0;
            if (absP == BISHOP) {
                dirs = bishopDirs;
                dirCount = 4;
            } else if (absP == ROOK) {
                dirs = rookDirs;
                dirCount = 4;
            } else {
                static const int queenDirs[8] = {9, 7, -9, -7, 1, -1, 8, -8};
                dirs = queenDirs;
                dirCount = 8;
            }
            for (int i = 0; i < dirCount; ++i) {
                int d = dirs[i];
                int to = sq;
                while (true) {
                    int n = to + d;
                    if (!on_board(n)) break;
                    if (abs(file_of(n) - file_of(to)) > 1) break;
                    if ((d == 1 || d == -1) && rank_of(n) != rank_of(to)) break;
                    if ((d == 9 || d == -9 || d == 7 || d == -7) && abs(rank_of(n) - rank_of(to)) != 1) break;
                    to = n;
                    int target = pos.board[to];
                    if (target == EMPTY) {
                        if (!capturesOnly) add_move(moves, sq, to, 0);
                    } else {
                        if ((target > 0) != (us > 0)) add_move(moves, sq, to, FLAG_CAPTURE);
                        break;
                    }
                }
            }
        } else if (absP == KING) {
            static const int offs[8] = {1, -1, 8, -8, 9, -9, 7, -7};
            for (int d : offs) {
                int to = sq + d;
                if (!on_board(to)) continue;
                if (abs(file_of(to) - file_of(sq)) > 1 || abs(rank_of(to) - rank_of(sq)) > 1) continue;
                int target = pos.board[to];
                if (target == EMPTY) {
                    if (!capturesOnly) add_move(moves, sq, to, 0);
                } else if ((target > 0) != (us > 0)) {
                    add_move(moves, sq, to, FLAG_CAPTURE);
                }
            }

            if (!capturesOnly) {
                if (us == 1 && sq == 4) {
                    if ((pos.castling & 1) && pos.board[5] == EMPTY && pos.board[6] == EMPTY &&
                        !is_square_attacked(pos, 4, -us) && !is_square_attacked(pos, 5, -us) && !is_square_attacked(pos, 6, -us)) {
                        add_move(moves, 4, 6, FLAG_CASTLE);
                    }
                    if ((pos.castling & 2) && pos.board[3] == EMPTY && pos.board[2] == EMPTY && pos.board[1] == EMPTY &&
                        !is_square_attacked(pos, 4, -us) && !is_square_attacked(pos, 3, -us) && !is_square_attacked(pos, 2, -us)) {
                        add_move(moves, 4, 2, FLAG_CASTLE);
                    }
                } else if (us == -1 && sq == 60) {
                    if ((pos.castling & 4) && pos.board[61] == EMPTY && pos.board[62] == EMPTY &&
                        !is_square_attacked(pos, 60, -us) && !is_square_attacked(pos, 61, -us) && !is_square_attacked(pos, 62, -us)) {
                        add_move(moves, 60, 62, FLAG_CASTLE);
                    }
                    if ((pos.castling & 8) && pos.board[59] == EMPTY && pos.board[58] == EMPTY && pos.board[57] == EMPTY &&
                        !is_square_attacked(pos, 60, -us) && !is_square_attacked(pos, 59, -us) && !is_square_attacked(pos, 58, -us)) {
                        add_move(moves, 60, 58, FLAG_CASTLE);
                    }
                }
            }
        }
    }
}

static inline int store_score(int score, int ply) {
    if (score > MATE - 1000) return score + ply;
    if (score < -MATE + 1000) return score - ply;
    return score;
}

static inline int load_score(int score, int ply) {
    if (score > MATE - 1000) return score - ply;
    if (score < -MATE + 1000) return score + ply;
    return score;
}

static int qsearch(Position& pos, int alpha, int beta, int ply) {
    if (stopSearch) return 0;
    if ((nodeCount & 2047ULL) == 0ULL) {
        if (chrono::steady_clock::now() >= deadline) {
            stopSearch = true;
            return 0;
        }
    }
    ++nodeCount;

    int stand = evaluate(pos) * pos.side;
    if (stand >= beta) return beta;
    if (stand > alpha) alpha = stand;

    vector<Move> moves;
    generate_moves(pos, moves, true);
    sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) {
        auto scoreMove = [&](const Move& m) {
            if (m.flags & FLAG_PROMO) return 50000 + piece_value(m.promo);
            int victim = 0;
            if (m.flags & FLAG_EP) victim = 100;
            else victim = piece_value(pos.board[m.to]);
            int attacker = piece_value(pos.board[m.from]);
            return 10000 + victim * 10 - attacker;
        };
        return scoreMove(a) > scoreMove(b);
    });

    for (const Move& m : moves) {
        Undo u;
        if (!make_move(pos, m, u)) {
            unmake_move(pos, m, u);
            continue;
        }
        int score = -qsearch(pos, -beta, -alpha, ply + 1);
        unmake_move(pos, m, u);
        if (stopSearch) return 0;
        if (score >= beta) return beta;
        if (score > alpha) alpha = score;
    }

    return alpha;
}

static int search(Position& pos, int depth, int alpha, int beta, int ply) {
    if (stopSearch) return 0;
    if ((nodeCount & 2047ULL) == 0ULL) {
        if (chrono::steady_clock::now() >= deadline) {
            stopSearch = true;
            return 0;
        }
    }
    ++nodeCount;

    if (pos.halfmove >= 100) return 0;

    Move ttMove{};
    const TTEntry& te = TT[pos.hash & (TT_SIZE - 1)];
    if (te.key == pos.hash) {
        ttMove = te.best;
        int ttScore = load_score(te.score, ply);
        if (te.depth >= depth) {
            if (te.flag == 0) return ttScore;
            if (te.flag == 1 && ttScore <= alpha) return alpha;
            if (te.flag == 2 && ttScore >= beta) return beta;
        }
    }

    bool inCheck = is_square_attacked(pos, pos.side == 1 ? pos.kingSqW : pos.kingSqB, -pos.side);
    if (inCheck) ++depth;

    if (depth <= 0) return qsearch(pos, alpha, beta, ply);

    vector<Move> moves;
    generate_moves(pos, moves, false);
    if (moves.empty()) {
        if (inCheck) return -MATE + ply;
        return 0;
    }

    struct ScoredMove { Move m; int score; };
    vector<ScoredMove> ordered;
    ordered.reserve(moves.size());
    int sideIdx = color_index(pos.side);

    for (const Move& m : moves) {
        int score = 0;
        if (same_move(m, ttMove)) {
            score = 2000000;
        } else if (m.flags & FLAG_CAPTURE) {
            int victim = 0;
            if (m.flags & FLAG_EP) victim = 100;
            else victim = piece_value(pos.board[m.to]);
            int attacker = piece_value(pos.board[m.from]);
            score = 1000000 + victim * 10 - attacker;
            if (m.flags & FLAG_PROMO) score += 5000 + piece_value(m.promo);
        } else {
            score = historyTable[sideIdx][m.from][m.to];
            if (same_move(m, killers[ply][0])) score += 900000;
            else if (same_move(m, killers[ply][1])) score += 800000;
            if (m.flags & FLAG_PROMO) score += 5000 + piece_value(m.promo);
        }
        ordered.push_back({m, score});
    }
    sort(ordered.begin(), ordered.end(), [](const ScoredMove& a, const ScoredMove& b) {
        return a.score > b.score;
    });

    int bestScore = -INF;
    Move bestMove{};
    int origAlpha = alpha;
    bool foundLegal = false;

    for (const auto& sm : ordered) {
        Undo u;
        if (!make_move(pos, sm.m, u)) {
            unmake_move(pos, sm.m, u);
            continue;
        }
        foundLegal = true;
        int score = -search(pos, depth - 1, -beta, -alpha, ply + 1);
        unmake_move(pos, sm.m, u);
        if (stopSearch) return 0;

        if (score > bestScore) {
            bestScore = score;
            bestMove = sm.m;
        }
        if (score > alpha) {
            alpha = score;
        }
        if (alpha >= beta) {
            if (!(sm.m.flags & FLAG_CAPTURE) && !(sm.m.flags & FLAG_PROMO)) {
                Move& k0 = killers[ply][0];
                if (!same_move(k0, sm.m)) {
                    killers[ply][1] = k0;
                    k0 = sm.m;
                }
                historyTable[sideIdx][sm.m.from][sm.m.to] += depth * depth;
            }
            break;
        }
    }

    if (!foundLegal) {
        if (inCheck) return -MATE + ply;
        return 0;
    }

    TTEntry& e = TT[pos.hash & (TT_SIZE - 1)];
    e.key = pos.hash;
    e.depth = static_cast<int16_t>(depth);
    e.best = bestMove;
    e.score = static_cast<int16_t>(store_score(bestScore, ply));
    if (bestScore <= origAlpha) e.flag = 1;
    else if (bestScore >= beta) e.flag = 2;
    else e.flag = 0;

    return bestScore;
}

static string format_move(const Position& pos, const Move& m) {
    string s;
    if (m.flags & FLAG_CASTLE) {
        s = (m.to > m.from) ? "O-O" : "O-O-O";
    } else {
        char pc = 'p';
        switch (abs_piece(pos.board[m.from])) {
            case PAWN: pc = 'p'; break;
            case KNIGHT: pc = 'N'; break;
            case BISHOP: pc = 'B'; break;
            case ROOK: pc = 'R'; break;
            case QUEEN: pc = 'Q'; break;
            case KING: pc = 'K'; break;
        }
        s.push_back(pc);
        s += square_to_string(m.from);
        s += (m.flags & FLAG_CAPTURE) ? "x" : "";
        s += square_to_string(m.to);
        if (m.flags & FLAG_PROMO) {
            s.push_back('/');
            char promo = 'Q';
            switch (m.promo) {
                case KNIGHT: promo = 'N'; break;
                case BISHOP: promo = 'B'; break;
                case ROOK: promo = 'R'; break;
                case QUEEN: promo = 'Q'; break;
            }
            s.push_back(promo);
        }
        if (m.flags & FLAG_EP) {
            s += "ep";
        }
    }

    Position tmp = pos;
    Undo u;
    if (make_move(tmp, m, u)) {
        vector<Move> replies;
        generate_moves(tmp, replies, false);
        bool mate = replies.empty() && is_square_attacked(tmp, tmp.side == 1 ? tmp.kingSqW : tmp.kingSqB, -tmp.side);
        if (mate) s.push_back('#');
        else if (is_square_attacked(tmp, tmp.side == 1 ? tmp.kingSqW : tmp.kingSqB, -tmp.side)) s.push_back('+');
        unmake_move(tmp, m, u);
    }
    return s;
}

static Position parse_fen(const string& fen) {
    Position pos;
    pos.board.fill(EMPTY);
    stringstream ss(fen);
    string placement, side, castling, ep;
    ss >> placement >> side >> castling >> ep >> pos.halfmove >> pos.fullmove;

    int sq = 56;
    for (char c : placement) {
        if (c == '/') {
            sq -= 16;
        } else if (c >= '1' && c <= '8') {
            sq += c - '0';
        } else {
            int p = EMPTY;
            switch (c) {
                case 'P': p = PAWN; break;
                case 'N': p = KNIGHT; break;
                case 'B': p = BISHOP; break;
                case 'R': p = ROOK; break;
                case 'Q': p = QUEEN; break;
                case 'K': p = KING; break;
                case 'p': p = -PAWN; break;
                case 'n': p = -KNIGHT; break;
                case 'b': p = -BISHOP; break;
                case 'r': p = -ROOK; break;
                case 'q': p = -QUEEN; break;
                case 'k': p = -KING; break;
            }
            pos.board[sq] = p;
            if (p == KING) pos.kingSqW = sq;
            if (p == -KING) pos.kingSqB = sq;
            ++sq;
        }
    }

    pos.side = (side == "w") ? 1 : -1;
    pos.castling = 0;
    if (castling.find('K') != string::npos) pos.castling |= 1;
    if (castling.find('Q') != string::npos) pos.castling |= 2;
    if (castling.find('k') != string::npos) pos.castling |= 4;
    if (castling.find('q') != string::npos) pos.castling |= 8;
    pos.ep = (ep == "-") ? -1 : sq_from_string(ep);
    pos.hash = compute_hash(pos);
    return pos;
}

static Move search_best_move(Position pos, double timeoutSeconds) {
    deadline = chrono::steady_clock::now() + chrono::milliseconds((long long)(timeoutSeconds * 1000.0) - 50);
    stopSearch = false;
    nodeCount = 0;

    vector<Move> rootMoves;
    generate_moves(pos, rootMoves, false);
    vector<Move> legalRoots;
    legalRoots.reserve(rootMoves.size());
    for (const Move& m : rootMoves) {
        Undo u;
        if (make_move(pos, m, u)) {
            legalRoots.push_back(m);
            unmake_move(pos, m, u);
        } else {
            unmake_move(pos, m, u);
        }
    }

    if (legalRoots.empty()) return Move{};

    Move best = legalRoots[0];
    int bestScore = -INF;

    for (int depth = 1; !stopSearch; ++depth) {
        Move localBest = best;
        int localBestScore = -INF;
        int alpha = -INF;
        int beta = INF;

        vector<pair<Move, int>> ordered;
        ordered.reserve(legalRoots.size());
        for (const Move& m : legalRoots) {
            int score = 0;
            if (same_move(m, best)) score = 2000000;
            else if (m.flags & FLAG_CAPTURE) {
                int victim = (m.flags & FLAG_EP) ? 100 : piece_value(pos.board[m.to]);
                int attacker = piece_value(pos.board[m.from]);
                score = 1000000 + victim * 10 - attacker;
            } else {
                score = historyTable[color_index(pos.side)][m.from][m.to];
            }
            ordered.push_back({m, score});
        }
        sort(ordered.begin(), ordered.end(), [](const auto& a, const auto& b) { return a.second > b.second; });

        for (const auto& om : ordered) {
            Undo u;
            if (!make_move(pos, om.first, u)) {
                unmake_move(pos, om.first, u);
                continue;
            }
            int score = -search(pos, depth - 1, -beta, -alpha, 1);
            unmake_move(pos, om.first, u);
            if (stopSearch) break;
            if (score > localBestScore) {
                localBestScore = score;
                localBest = om.first;
            }
            if (score > alpha) alpha = score;
        }

        if (!stopSearch) {
            best = localBest;
            bestScore = localBestScore;
        } else {
            break;
        }
    }

    return best;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    init_zobrist();
    TT.resize(TT_SIZE);

    double timeoutSeconds = 5.0;
    string fen;
    if (!(cin >> timeoutSeconds)) return 0;
    getline(cin, fen);
    if (fen.empty()) getline(cin, fen);
    if (fen.size() && fen[0] == ' ') fen.erase(fen.begin());

    Position pos = parse_fen(fen);
    Move best = search_best_move(pos, timeoutSeconds);
    string out = format_move(pos, best);
    cout << out << '\n';
    return 0;
}
