// N-dimensional snake path planner
#include <algorithm>
#include <climits>

using namespace std;

struct Pos {
    vector<int> c;

    bool operator==(const Pos& other) const noexcept {
        return c == other.c;
    }
};

struct PosHash {
    size_t operator()(const Pos& p) const noexcept {
        uint64_t h = 1469598103934665603ull;
        for (int x : p.c) {
            uint64_t y = static_cast<uint64_t>(static_cast<int64_t>(x)) + 0x9e3779b97f4a7c15ull;
            h ^= y;
            h *= 1099511628211ull;
        }
        return static_cast<size_t>(h);
    }
};

struct Move {
    int axis = 0;
    int dir = 1;
};

struct NodeState {
    Pos parent;
    int axis = -1;
    int dir = 0;
    long long g = LLONG_MAX;
    bool hasParent = false;
};

static long long manhattan(const Pos& a, const Pos& b) {
    long long d = 0;
    for (size_t i = 0; i < a.c.size(); ++i) {
        d += std::llabs(static_cast<long long>(a.c[i]) - static_cast<long long>(b.c[i]));
    }
    return d;
}

static bool inBounds(const Pos& p, const vector<int>& bounds) {
    for (size_t i = 0; i < p.c.size(); ++i) {
        if (p.c[i] < 0 || p.c[i] >= bounds[i]) {
            return false;
        }
    }
    return true;
}

static vector<Move> reconstructPath(
    const Pos& start,
    const Pos& goal,
    const unordered_map<Pos, NodeState, PosHash>& best
) {
    vector<Move> rev;
    Pos cur = goal;
    while (!(cur == start)) {
        auto it = best.find(cur);
        if (it == best.end() || !it->second.hasParent) {
            return {};
        }
        rev.push_back(Move{it->second.axis, it->second.dir});
        cur = it->second.parent;
    }
    reverse(rev.begin(), rev.end());
    return rev;
}

static vector<vector<int>> candidateOrders(const Pos& cur, const Pos& goal, const vector<int>& bounds) {
    const int D = static_cast<int>(cur.c.size());
    vector<int> axes(D);
    iota(axes.begin(), axes.end(), 0);

    vector<vector<int>> orders;
    auto addOrder = [&](vector<int> ord) {
        for (const auto& existing : orders) {
            if (existing == ord) {
                return;
            }
        }
        orders.push_back(std::move(ord));
    };

    {
        vector<int> ord = axes;
        sort(ord.begin(), ord.end(), [&](int a, int b) {
            int da = std::abs(goal.c[a] - cur.c[a]);
            int db = std::abs(goal.c[b] - cur.c[b]);
            if (da != db) return da > db;
            return a < b;
        });
        addOrder(std::move(ord));
    }
    {
        vector<int> ord = axes;
        sort(ord.begin(), ord.end(), [&](int a, int b) {
            int da = std::abs(goal.c[a] - cur.c[a]);
            int db = std::abs(goal.c[b] - cur.c[b]);
            if (da != db) return da < db;
            return a < b;
        });
        addOrder(std::move(ord));
    }
    {
        vector<int> ord = axes;
        sort(ord.begin(), ord.end());
        addOrder(std::move(ord));
    }
    {
        vector<int> ord = axes;
        sort(ord.begin(), ord.end(), greater<int>());
        addOrder(std::move(ord));
    }
    {
        vector<int> ord = axes;
        sort(ord.begin(), ord.end(), [&](int a, int b) {
            int sa = min(cur.c[a], bounds[a] - 1 - cur.c[a]);
            int sb = min(cur.c[b], bounds[b] - 1 - cur.c[b]);
            if (sa != sb) return sa < sb;
            return a < b;
        });
        addOrder(std::move(ord));
    }

    return orders;
}

static vector<Move> tryStraightPath(
    const Pos& start,
    const Pos& goal,
    const vector<int>& order,
    const unordered_set<Pos, PosHash>& blocked,
    const vector<int>& bounds
) {
    Pos cur = start;
    vector<Move> out;
    out.reserve(128);

    for (int axis : order) {
        while (cur.c[axis] != goal.c[axis]) {
            int dir = (goal.c[axis] > cur.c[axis]) ? 1 : -1;
            Pos nxt = cur;
            nxt.c[axis] += dir;
            if (!inBounds(nxt, bounds) || blocked.find(nxt) != blocked.end()) {
                return {};
            }
            out.push_back(Move{axis, dir});
            cur = std::move(nxt);
        }
    }

    if (cur == goal) return out;
    return {};
}

struct AStarNode {
    long long f = 0;
    long long g = 0;
    Pos pos;
};

struct AStarCmp {
    bool operator()(const AStarNode& a, const AStarNode& b) const {
        if (a.f != b.f) return a.f > b.f;
        return a.g < b.g;
    }
};

static vector<Move> aStarPath(
    const Pos& start,
    const Pos& goal,
    const unordered_set<Pos, PosHash>& blocked,
    const vector<int>& bounds,
    int expansionCap
) {
    if (start == goal) return {};

    unordered_map<Pos, NodeState, PosHash> best;
    priority_queue<AStarNode, vector<AStarNode>, AStarCmp> pq;

    best[start].g = 0;
    pq.push(AStarNode{manhattan(start, goal), 0, start});

    const int D = static_cast<int>(start.c.size());
    int expanded = 0;

    while (!pq.empty() && expanded < expansionCap) {
        AStarNode cur = pq.top();
        pq.pop();

        auto it = best.find(cur.pos);
        if (it == best.end() || it->second.g != cur.g) {
            continue;
        }
        if (cur.pos == goal) {
            return reconstructPath(start, goal, best);
        }

        ++expanded;

        struct Cand {
            Move mv;
            Pos nxt;
            long long score;
        };

        vector<Cand> cands;
        cands.reserve(static_cast<size_t>(D) * 2);

        for (int axis = 0; axis < D; ++axis) {
            int delta = goal.c[axis] - cur.pos.c[axis];

            auto addMove = [&](int dir) {
                Pos nxt = cur.pos;
                nxt.c[axis] += dir;
                if (!inBounds(nxt, bounds)) return;
                if (blocked.find(nxt) != blocked.end()) return;
                long long h = manhattan(nxt, goal);
                cands.push_back(Cand{Move{axis, dir}, std::move(nxt), h});
            };

            if (delta > 0) {
                addMove(+1);
                addMove(-1);
            } else if (delta < 0) {
                addMove(-1);
                addMove(+1);
            } else {
                addMove(+1);
                addMove(-1);
            }
        }

        sort(cands.begin(), cands.end(), [](const Cand& a, const Cand& b) {
            if (a.score != b.score) return a.score < b.score;
            if (a.mv.axis != b.mv.axis) return a.mv.axis < b.mv.axis;
            return a.mv.dir < b.mv.dir;
        });

        for (const Cand& cand : cands) {
            long long g2 = cur.g + 1;
            auto jt = best.find(cand.nxt);
            if (jt != best.end() && jt->second.g <= g2) {
                continue;
            }
            NodeState& ns = best[cand.nxt];
            ns.parent = cur.pos;
            ns.axis = cand.mv.axis;
            ns.dir = cand.mv.dir;
            ns.g = g2;
            ns.hasParent = true;
            pq.push(AStarNode{g2 + cand.score, g2, cand.nxt});
        }
    }

    return {};
}

static vector<Move> greedyFallback(
    const Pos& start,
    const Pos& goal,
    const unordered_set<Pos, PosHash>& blocked,
    const vector<int>& bounds,
    int stepCap
) {
    if (start == goal) return {};

    const int D = static_cast<int>(start.c.size());
    Pos cur = start;
    unordered_set<Pos, PosHash> seen;
    seen.insert(cur);
    vector<Move> out;
    out.reserve(stepCap > 0 ? stepCap : 0);

    for (int step = 0; step < stepCap && !(cur == goal); ++step) {
        struct Cand {
            Move mv;
            Pos nxt;
            long long score = 0;
        };

        vector<Cand> cands;
        cands.reserve(static_cast<size_t>(D) * 2);

        auto gather = [&](bool forbidRevisit) {
            for (int axis = 0; axis < D; ++axis) {
                int delta = goal.c[axis] - cur.c[axis];
                for (int pass = 0; pass < 2; ++pass) {
                    int dir = 0;
                    if (delta > 0) {
                        dir = (pass == 0) ? 1 : -1;
                    } else if (delta < 0) {
                        dir = (pass == 0) ? -1 : 1;
                    } else {
                        dir = (pass == 0) ? 1 : -1;
                    }

                    Pos nxt = cur;
                    nxt.c[axis] += dir;
                    if (!inBounds(nxt, bounds)) continue;
                    if (blocked.find(nxt) != blocked.end()) continue;
                    if (forbidRevisit && seen.find(nxt) != seen.end()) continue;

                    long long nd = manhattan(nxt, goal);
                    long long score = nd * 1000;
                    if (delta != 0 && dir == ((delta > 0) ? 1 : -1)) score -= 30;
                    if (delta == 0) score += 20;
                    score += axis;
                    cands.push_back(Cand{Move{axis, dir}, std::move(nxt), score});
                }
            }
        };

        gather(true);
        if (cands.empty()) {
            gather(false);
        }

        if (cands.empty()) {
            break;
        }

        sort(cands.begin(), cands.end(), [](const Cand& a, const Cand& b) {
            if (a.score != b.score) return a.score < b.score;
            if (a.mv.axis != b.mv.axis) return a.mv.axis < b.mv.axis;
            return a.mv.dir < b.mv.dir;
        });

        const Cand chosen = cands.front();
        cur = chosen.nxt;
        seen.insert(cur);
        out.push_back(chosen.mv);
    }

    if (cur == goal) return out;
    return {};
}

static vector<Move> planSegment(
    const Pos& start,
    const Pos& goal,
    const unordered_set<Pos, PosHash>& blocked,
    const vector<int>& bounds
) {
    if (start == goal) return {};

    vector<vector<int>> orders = candidateOrders(start, goal, bounds);
    for (const auto& ord : orders) {
        vector<Move> path = tryStraightPath(start, goal, ord, blocked, bounds);
        if (!path.empty()) {
            return path;
        }
    }

    int cap = 120000 + 5000 * static_cast<int>(bounds.size());
    vector<Move> path = aStarPath(start, goal, blocked, bounds, cap);
    if (!path.empty()) {
        return path;
    }

    int base = static_cast<int>(manhattan(start, goal));
    path = greedyFallback(start, goal, blocked, bounds, base + 4 * static_cast<int>(bounds.size()) + 32);
    return path;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int D = 0;
    long long maxTurns = 0;
    if (!(cin >> D >> maxTurns)) {
        return 0;
    }

    vector<int> bounds(D);
    for (int i = 0; i < D; ++i) {
        cin >> bounds[i];
    }

    int snakeLength = 0;
    Pos start;
    start.c.resize(D);
    cin >> snakeLength;
    for (int i = 0; i < D; ++i) {
        cin >> start.c[i];
    }
    (void)snakeLength;

    int F = 0;
    cin >> F;
    vector<Pos> foods(F);
    for (int i = 0; i < F; ++i) {
        foods[i].c.resize(D);
        for (int j = 0; j < D; ++j) {
            cin >> foods[i].c[j];
        }
    }

    int O = 0;
    cin >> O;
    unordered_set<Pos, PosHash> blocked;
    blocked.reserve(static_cast<size_t>(O) * 2 + 8);
    for (int i = 0; i < O; ++i) {
        Pos p;
        p.c.resize(D);
        for (int j = 0; j < D; ++j) {
            cin >> p.c[j];
        }
        blocked.insert(std::move(p));
    }

    vector<char> eaten(F, 0);
    Pos cur = start;
    long long remaining = maxTurns;

    auto anyRemainingFood = [&]() -> bool {
        for (int i = 0; i < F; ++i) {
            if (!eaten[i]) return true;
        }
        return false;
    };

    while (remaining > 0 && anyRemainingFood()) {
        int bestIdx = -1;
        vector<Move> bestPath;
        size_t bestLen = numeric_limits<size_t>::max();

        for (int i = 0; i < F; ++i) {
            if (eaten[i]) continue;
            if (foods[i] == cur) {
                bestIdx = i;
                bestPath.clear();
                bestLen = 0;
                break;
            }

            vector<Move> path = planSegment(cur, foods[i], blocked, bounds);
            if (!path.empty() && path.size() < bestLen) {
                bestLen = path.size();
                bestIdx = i;
                bestPath = std::move(path);
            }
        }

        if (bestIdx == -1) {
            long long bestDist = LLONG_MAX;
            for (int i = 0; i < F; ++i) {
                if (eaten[i]) continue;
                long long d = manhattan(cur, foods[i]);
                if (d < bestDist) {
                    bestDist = d;
                    bestIdx = i;
                }
            }
            if (bestIdx == -1) break;
            bestPath = greedyFallback(cur, foods[bestIdx], blocked, bounds,
                                      static_cast<int>(bestDist) + 4 * D + 32);
            if (bestPath.empty() && !(cur == foods[bestIdx])) {
                break;
            }
        }

        if (bestPath.empty()) {
            eaten[bestIdx] = 1;
            continue;
        }

        if (static_cast<long long>(bestPath.size()) > remaining) {
            for (long long i = 0; i < remaining; ++i) {
                const Move& mv = bestPath[static_cast<size_t>(i)];
                cout << mv.axis << ' ' << mv.dir << '\n';
            }
            return 0;
        }

        for (const Move& mv : bestPath) {
            cout << mv.axis << ' ' << mv.dir << '\n';
        }
        remaining -= static_cast<long long>(bestPath.size());

        if (foods[bestIdx] == cur) {
            eaten[bestIdx] = 1;
        } else {
            cur = foods[bestIdx];
            eaten[bestIdx] = 1;
        }
    }

    return 0;
}
