declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type RawTask = {
  id: any;
  duration: number;
  deps?: any[];
  deadline?: number;
  weight?: number;
};

type ScheduleEntry = {
  id: any;
  start: number;
  machine: number;
};

class Heap<T> {
  private a: T[] = [];
  private readonly cmp: (x: T, y: T) => number;
  constructor(cmp: (x: T, y: T) => number) {
    this.cmp = cmp;
  }
  size(): number {
    return this.a.length;
  }
  peek(): T | undefined {
    return this.a[0];
  }
  push(v: T): void {
    const a = this.a;
    a.push(v);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.cmp(a[i], a[p]) >= 0) break;
      [a[i], a[p]] = [a[p], a[i]];
      i = p;
    }
  }
  pop(): T | undefined {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop()!;
    if (a.length > 0) {
      a[0] = last;
      let i = 0;
      for (;;) {
        let left = i * 2 + 1;
        if (left >= a.length) break;
        let right = left + 1;
        let best = left;
        if (right < a.length && this.cmp(a[right], a[left]) < 0) best = right;
        if (this.cmp(a[best], a[i]) >= 0) break;
        [a[i], a[best]] = [a[best], a[i]];
        i = best;
      }
    }
    return top;
  }
}

type Task = {
  id: any;
  duration: number;
  deadline: number;
  weight: number;
  deps: number[];
  succ: number[];
  indegree: number;
  release: number;
  tail: number;
};

const rawTasks: RawTask[] = Array.isArray(data.tasks) ? data.tasks : [];
const machineCount: number = Math.max(0, data.machineCount | 0);
const n = rawTasks.length;

if (machineCount === 0 || n === 0) {
  process.stdout.write(JSON.stringify({ schedule: [] }));
  process.exit(0);
}

const tasks: Task[] = new Array(n);
const idToIndex = new Map<any, number>();
for (let i = 0; i < n; i++) {
  const rt = rawTasks[i];
  idToIndex.set(rt.id, i);
  tasks[i] = {
    id: rt.id,
    duration: Number(rt.duration) || 0,
    deadline: Number(rt.deadline) || 0,
    weight: Number(rt.weight) || 0,
    deps: [],
    succ: [],
    indegree: 0,
    release: 0,
    tail: 0,
  };
}

for (let i = 0; i < n; i++) {
  const deps = Array.isArray(rawTasks[i].deps) ? rawTasks[i].deps : [];
  const mapped: number[] = [];
  for (const depId of deps) {
    const j = idToIndex.get(depId);
    if (j === undefined) continue;
    mapped.push(j);
    tasks[j].succ.push(i);
    tasks[i].indegree++;
  }
  tasks[i].deps = mapped;
}

// Critical-path tail lengths help preserve makespan when deadlines are similar.
const indegForTopo = tasks.map(t => t.indegree);
const topo: number[] = [];
const q: number[] = [];
for (let i = 0; i < n; i++) {
  if (indegForTopo[i] === 0) q.push(i);
}
for (let head = 0; head < q.length; head++) {
  const u = q[head];
  topo.push(u);
  for (const v of tasks[u].succ) {
    indegForTopo[v]--;
    if (indegForTopo[v] === 0) q.push(v);
  }
}
for (let k = topo.length - 1; k >= 0; k--) {
  const u = topo[k];
  let best = 0;
  for (const v of tasks[u].succ) {
    if (tasks[v].tail > best) best = tasks[v].tail;
  }
  tasks[u].tail = tasks[u].duration + best;
}
for (let i = 0; i < n; i++) {
  if (tasks[i].tail === 0) tasks[i].tail = tasks[i].duration;
}

function taskPriority(a: number, b: number): number {
  const ta = tasks[a];
  const tb = tasks[b];
  const ea = ta.deadline - ta.duration;
  const eb = tb.deadline - tb.duration;
  if (ea !== eb) return ea - eb;
  if (ta.tail !== tb.tail) return tb.tail - ta.tail;
  if (ta.weight !== tb.weight) return tb.weight - ta.weight;
  if (ta.duration !== tb.duration) return ta.duration - tb.duration;
  return a - b;
}

const waiting = new Heap<number>((a, b) => {
  const ra = tasks[a].release;
  const rb = tasks[b].release;
  if (ra !== rb) return ra - rb;
  return taskPriority(a, b);
});
const ready = new Heap<number>(taskPriority);
const freeMachines = new Heap<number>((a, b) => a - b);
const running = new Heap<{ finish: number; machine: number; task: number }>((a, b) => {
  if (a.finish !== b.finish) return a.finish - b.finish;
  if (a.machine !== b.machine) return a.machine - b.machine;
  return a.task - b.task;
});

for (let i = 0; i < machineCount; i++) freeMachines.push(i);
for (let i = 0; i < n; i++) {
  if (tasks[i].indegree === 0) waiting.push(i);
}

const schedule: ScheduleEntry[] = [];
let completed = 0;
let time = 0;
const INF = Number.POSITIVE_INFINITY;

while (completed < n) {
  while (running.size() > 0) {
    const ev = running.peek()!;
    if (ev.finish > time) break;
    running.pop();
    freeMachines.push(ev.machine);
    completed++;
    const finishTime = ev.finish;
    const u = ev.task;
    for (const v of tasks[u].succ) {
      if (tasks[v].release < finishTime) tasks[v].release = finishTime;
      tasks[v].indegree--;
      if (tasks[v].indegree === 0) waiting.push(v);
    }
  }

  while (waiting.size() > 0) {
    const u = waiting.peek()!;
    if (tasks[u].release > time) break;
    waiting.pop();
    ready.push(u);
  }

  let assigned = false;
  while (freeMachines.size() > 0 && ready.size() > 0) {
    const machine = freeMachines.pop()!;
    const u = ready.pop()!;
    const start = Math.max(time, tasks[u].release);
    const finish = start + tasks[u].duration;
    schedule.push({ id: tasks[u].id, start, machine });
    running.push({ finish, machine, task: u });
    assigned = true;
  }

  if (completed >= n) break;
  if (assigned) continue;

  let nextTime = INF;
  if (running.size() > 0) {
    const ev = running.peek()!;
    if (ev.finish < nextTime) nextTime = ev.finish;
  }
  if (waiting.size() > 0) {
    const u = waiting.peek()!;
    if (tasks[u].release < nextTime) nextTime = tasks[u].release;
  }

  if (!Number.isFinite(nextTime)) break;
  if (nextTime > time) {
    time = nextTime;
  } else {
    time++;
  }
}

process.stdout.write(JSON.stringify({ schedule }));
