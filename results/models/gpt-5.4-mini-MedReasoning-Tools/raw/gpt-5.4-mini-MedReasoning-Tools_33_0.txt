using System;
using System.IO;
using System.Text;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private byte ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do
            {
                c = ReadByte();
            } while (c <= 32 && c != 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = ReadByte();
            }
            return value * sign;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        if (n <= 0)
        {
            return;
        }

        int nn = n * n;
        long[] flow = new long[nn];
        long[] dist = new long[nn];

        for (int i = 0; i < nn; i++)
        {
            flow[i] = fs.NextInt();
        }
        for (int i = 0; i < nn; i++)
        {
            dist[i] = fs.NextInt();
        }

        long[] facScore = new long[n];
        long[] locScore = new long[n];

        for (int i = 0; i < n; i++)
        {
            int row = i * n;
            long s = 0;
            long t = 0;
            for (int j = 0; j < n; j++)
            {
                s += flow[row + j] + flow[j * n + i];
                t += dist[row + j] + dist[j * n + i];
            }
            facScore[i] = s;
            locScore[i] = t;
        }

        int[] facOrder = new int[n];
        int[] locOrder = new int[n];
        for (int i = 0; i < n; i++)
        {
            facOrder[i] = i;
            locOrder[i] = i;
        }

        Array.Sort(facOrder, (x, y) =>
        {
            int cmp = facScore[y].CompareTo(facScore[x]);
            return cmp != 0 ? cmp : x.CompareTo(y);
        });
        Array.Sort(locOrder, (x, y) =>
        {
            int cmp = locScore[x].CompareTo(locScore[y]);
            return cmp != 0 ? cmp : x.CompareTo(y);
        });

        int[] perm = new int[n];
        int[] locToFac = new int[n];
        for (int i = 0; i < n; i++)
        {
            int f = facOrder[i];
            int l = locOrder[i];
            perm[f] = l;
            locToFac[l] = f;
        }

        long cost = ComputeCost(n, flow, dist, perm);

        int top = Math.Min(n, 100);
        for (int pass = 0; pass < 2; pass++)
        {
            bool improved = false;
            for (int ai = 0; ai < top; ai++)
            {
                int a = facOrder[ai];
                int la = perm[a];
                for (int bi = ai + 1; bi < top; bi++)
                {
                    int b = facOrder[bi];
                    int lb = perm[b];
                    long delta = SwapDelta(n, a, b, la, lb, perm, flow, dist);
                    if (delta < 0)
                    {
                        perm[a] = lb;
                        perm[b] = la;
                        locToFac[la] = b;
                        locToFac[lb] = a;
                        cost += delta;
                        la = perm[a];
                        improved = true;
                    }
                }
            }
            if (!improved) break;
        }

        var sb = new StringBuilder();
        sb.AppendLine(cost.ToString());
        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(perm[i]);
        }
        sb.AppendLine();
        Console.Write(sb.ToString());
    }

    private static long ComputeCost(int n, long[] flow, long[] dist, int[] perm)
    {
        long total = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = perm[i];
            int rowF = i * n;
            int rowD = pi * n;
            for (int j = 0; j < n; j++)
            {
                total += flow[rowF + j] * dist[rowD + perm[j]];
            }
        }
        return total;
    }

    private static long SwapDelta(int n, int a, int b, int la, int lb, int[] perm, long[] flow, long[] dist)
    {
        long delta = 0;
        int rowA = a * n;
        int rowB = b * n;

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;

            int pk = perm[k];
            int rowK = k * n;

            long d1 = dist[lb * n + pk] - dist[la * n + pk];
            long d2 = dist[pk * n + lb] - dist[pk * n + la];

            delta += (flow[rowA + k] - flow[rowB + k]) * d1;
            delta += (flow[rowK + a] - flow[rowK + b]) * d2;
        }

        delta += flow[rowA + a] * (dist[lb * n + lb] - dist[la * n + la]);
        delta += flow[rowB + b] * (dist[la * n + la] - dist[lb * n + lb]);
        delta += flow[rowA + b] * (dist[lb * n + la] - dist[la * n + lb]);
        delta += flow[rowB + a] * (dist[la * n + lb] - dist[lb * n + la]);

        return delta;
    }
}
