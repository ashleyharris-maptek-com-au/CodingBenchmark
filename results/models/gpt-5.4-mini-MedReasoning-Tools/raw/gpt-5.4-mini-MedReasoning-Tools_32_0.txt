#include <algorithm>
#include <functional>
#include <iostream>
#include <limits>
#include <queue>
#include <utility>
#include <vector>

using namespace std;

using ll = long long;
static const ll INF = numeric_limits<ll>::max() / 4;

struct OrigEdge {
    int u = 0, v = 0;
    ll w = 0;
};

struct AdjEdge {
    int to = 0;
    ll w = 0;
    int id = 0;
};

struct DijkstraResult {
    vector<int> parent_v;
    vector<int> parent_e;
    vector<ll> dist_terms;
};

struct DSU {
    vector<int> p, r;
    explicit DSU(int n = 0) { init(n); }
    void init(int n) {
        p.resize(n);
        r.assign(n, 0);
        for (int i = 0; i < n; ++i) p[i] = i;
    }
    int find(int x) {
        while (p[x] != x) {
            p[x] = p[p[x]];
            x = p[x];
        }
        return x;
    }
    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) ++r[a];
        return true;
    }
};

DijkstraResult run_dijkstra(
    int src,
    const vector<vector<AdjEdge>>& g,
    const vector<int>& terminals
) {
    const int n = (int)g.size();
    vector<ll> dist(n, INF);
    vector<int> parent_v(n, -1), parent_e(n, -1);
    priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;

    dist[src] = 0;
    pq.push({0, src});

    while (!pq.empty()) {
        auto [d, v] = pq.top();
        pq.pop();
        if (d != dist[v]) continue;
        for (const auto& e : g[v]) {
            ll nd = d + e.w;
            if (nd < dist[e.to]) {
                dist[e.to] = nd;
                parent_v[e.to] = v;
                parent_e[e.to] = e.id;
                pq.push({nd, e.to});
            }
        }
    }

    DijkstraResult res;
    res.parent_v = move(parent_v);
    res.parent_e = move(parent_e);
    res.dist_terms.resize(terminals.size(), INF);
    for (size_t i = 0; i < terminals.size(); ++i) {
        res.dist_terms[i] = dist[terminals[i]];
    }
    return res;
}

vector<int> collect_path_edges(int src, int dst, const DijkstraResult& res) {
    vector<int> path;
    int cur = dst;
    while (cur != src) {
        int pe = res.parent_e[cur];
        int pv = res.parent_v[cur];
        if (pv < 0 || pe < 0) {
            path.clear();
            return path;
        }
        path.push_back(pe);
        cur = pv;
    }
    reverse(path.begin(), path.end());
    return path;
}

struct BuiltTree {
    bool valid = false;
    ll weight = INF;
    vector<int> edge_ids;
};

BuiltTree build_tree_from_union_edges(
    int n,
    const vector<OrigEdge>& edges,
    const vector<int>& terminals,
    const vector<int>& union_edge_ids
) {
    BuiltTree out;
    if (terminals.empty()) {
        out.valid = true;
        out.weight = 0;
        return out;
    }
    if (terminals.size() == 1) {
        out.valid = true;
        out.weight = 0;
        return out;
    }

    vector<int> ord = union_edge_ids;
    sort(ord.begin(), ord.end(), [&](int a, int b) {
        if (edges[a].w != edges[b].w) return edges[a].w < edges[b].w;
        return a < b;
    });

    DSU dsu(n);
    vector<int> tree_edge_ids;
    tree_edge_ids.reserve(ord.size());
    for (int id : ord) {
        const auto& e = edges[id];
        if (dsu.unite(e.u, e.v)) tree_edge_ids.push_back(id);
    }

    int root_terminal = terminals[0];
    int root_comp = dsu.find(root_terminal);
    for (int t : terminals) {
        if (dsu.find(t) != root_comp) return out;
    }

    vector<vector<int>> incident(n);
    vector<int> degree(n, 0);
    for (int idx = 0; idx < (int)tree_edge_ids.size(); ++idx) {
        int id = tree_edge_ids[idx];
        incident[edges[id].u].push_back(idx);
        incident[edges[id].v].push_back(idx);
        ++degree[edges[id].u];
        ++degree[edges[id].v];
    }

    vector<char> active(tree_edge_ids.size(), 1);
    vector<char> is_terminal(n, 0);
    for (int t : terminals) is_terminal[t] = 1;

    queue<int> q;
    for (int v = 0; v < n; ++v) {
        if (!is_terminal[v] && degree[v] == 1) q.push(v);
    }

    while (!q.empty()) {
        int v = q.front();
        q.pop();
        if (is_terminal[v] || degree[v] != 1) continue;
        int idx = -1;
        for (int cand : incident[v]) {
            if (active[cand]) {
                idx = cand;
                break;
            }
        }
        if (idx < 0) continue;
        active[idx] = 0;
        int id = tree_edge_ids[idx];
        int a = edges[id].u;
        int b = edges[id].v;
        int u = (a == v) ? b : a;
        --degree[v];
        --degree[u];
        if (!is_terminal[u] && degree[u] == 1) q.push(u);
    }

    vector<int> final_edges;
    ll total = 0;
    final_edges.reserve(tree_edge_ids.size());
    for (int idx = 0; idx < (int)tree_edge_ids.size(); ++idx) {
        if (active[idx]) {
            int id = tree_edge_ids[idx];
            final_edges.push_back(id);
            total += edges[id].w;
        }
    }

    DSU check(n);
    for (int id : final_edges) check.unite(edges[id].u, edges[id].v);
    int check_root = check.find(terminals[0]);
    for (int t : terminals) {
        if (check.find(t) != check_root) return out;
    }

    out.valid = true;
    out.weight = total;
    out.edge_ids = move(final_edges);
    return out;
}

void add_path_edges(
    int src_idx,
    int dst_idx,
    const vector<int>& terminals,
    const DijkstraResult& res,
    vector<char>& seen,
    vector<int>& union_edge_ids
) {
    int src = terminals[src_idx];
    int dst = terminals[dst_idx];
    vector<int> path = collect_path_edges(src, dst, res);
    for (int id : path) {
        if (!seen[id]) {
            seen[id] = 1;
            union_edge_ids.push_back(id);
        }
    }
}

BuiltTree build_root_path_candidate(
    int n,
    const vector<OrigEdge>& edges,
    const vector<int>& terminals,
    int root_idx,
    const DijkstraResult& res
) {
    vector<char> seen(edges.size(), 0);
    vector<int> union_edge_ids;
    for (int i = 0; i < (int)terminals.size(); ++i) {
        if (i == root_idx) continue;
        add_path_edges(root_idx, i, terminals, res, seen, union_edge_ids);
    }
    return build_tree_from_union_edges(n, edges, terminals, union_edge_ids);
}

BuiltTree build_metric_mst_candidate(
    int n,
    const vector<OrigEdge>& edges,
    const vector<int>& terminals,
    const vector<DijkstraResult>& results,
    const vector<vector<ll>>& dist_mat
) {
    const int k = (int)terminals.size();
    if (k <= 1) {
        BuiltTree trivial;
        trivial.valid = true;
        trivial.weight = 0;
        return trivial;
    }

    vector<ll> best(k, INF);
    vector<int> parent(k, -1);
    vector<char> used(k, 0);
    best[0] = 0;
    vector<pair<int, int>> mst_pairs;
    mst_pairs.reserve(k - 1);

    for (int it = 0; it < k; ++it) {
        int v = -1;
        ll bv = INF;
        for (int i = 0; i < k; ++i) {
            if (!used[i] && best[i] < bv) {
                bv = best[i];
                v = i;
            }
        }
        if (v < 0 || best[v] >= INF / 2) return BuiltTree{};
        used[v] = 1;
        if (parent[v] != -1) mst_pairs.push_back({parent[v], v});

        for (int u = 0; u < k; ++u) {
            if (!used[u] && dist_mat[v][u] < best[u]) {
                best[u] = dist_mat[v][u];
                parent[u] = v;
            }
        }
    }

    vector<char> seen(edges.size(), 0);
    vector<int> union_edge_ids;
    for (auto [a, b] : mst_pairs) {
        add_path_edges(a, b, terminals, results[a], seen, union_edge_ids);
    }
    return build_tree_from_union_edges(n, edges, terminals, union_edge_ids);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<OrigEdge> edges(m);
    vector<vector<AdjEdge>> g(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        ll w;
        cin >> u >> v >> w;
        edges[i] = {u, v, w};
        g[u].push_back({v, w, i});
        g[v].push_back({u, w, i});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) cin >> terminals[i];
    sort(terminals.begin(), terminals.end());
    terminals.erase(unique(terminals.begin(), terminals.end()), terminals.end());
    k = (int)terminals.size();

    if (k <= 1) {
        cout << 0 << ' ' << 0 << '\n';
        return 0;
    }

    const int SMALL_K = 30;

    vector<DijkstraResult> results;
    vector<vector<ll>> dist_mat;
    results.reserve((k <= SMALL_K) ? k : 2);

    auto res0 = run_dijkstra(terminals[0], g, terminals);
    results.push_back(res0);

    int far_idx = 0;
    ll far_dist = -1;
    for (int i = 0; i < k; ++i) {
        if (i == 0) continue;
        if (res0.dist_terms[i] < INF / 2 && res0.dist_terms[i] > far_dist) {
            far_dist = res0.dist_terms[i];
            far_idx = i;
        }
    }
    if (far_idx != 0) {
        results.push_back(run_dijkstra(terminals[far_idx], g, terminals));
    }

    BuiltTree best;
    best.weight = INF;

    auto consider = [&](const BuiltTree& cand) {
        if (cand.valid && cand.weight < best.weight) best = cand;
    };

    consider(build_root_path_candidate(n, edges, terminals, 0, results[0]));
    if ((int)results.size() > 1) {
        consider(build_root_path_candidate(n, edges, terminals, far_idx, results[1]));
    }

    if (k <= SMALL_K) {
        results.resize(k);
        dist_mat.assign(k, vector<ll>(k, INF));
        for (int i = 1; i < k; ++i) {
            results[i] = run_dijkstra(terminals[i], g, terminals);
        }
        for (int i = 0; i < k; ++i) {
            for (int j = 0; j < k; ++j) {
                dist_mat[i][j] = results[i].dist_terms[j];
            }
        }

        vector<ll> sum_dist(k, INF);
        int best_root = 0;
        ll best_sum = INF;
        for (int i = 0; i < k; ++i) {
            ll s = 0;
            bool ok = true;
            for (int j = 0; j < k; ++j) {
                if (dist_mat[i][j] >= INF / 2) {
                    ok = false;
                    break;
                }
                s += dist_mat[i][j];
            }
            sum_dist[i] = ok ? s : INF;
            if (ok && s < best_sum) {
                best_sum = s;
                best_root = i;
            }
        }

        if (best_sum < INF / 2) {
            consider(build_root_path_candidate(n, edges, terminals, best_root, results[best_root]));
            consider(build_metric_mst_candidate(n, edges, terminals, results, dist_mat));
        }
    }

    if (!best.valid) {
        // Fallback: use the shortest-path tree from the first terminal.
        vector<int> fallback_edges;
        for (int v = 0; v < n; ++v) {
            if (results[0].parent_v[v] != -1 && results[0].parent_e[v] != -1) {
                fallback_edges.push_back(results[0].parent_e[v]);
            }
        }
        BuiltTree fb = build_tree_from_union_edges(n, edges, terminals, fallback_edges);
        if (fb.valid) best = move(fb);
    }

    if (!best.valid) {
        // Last resort: output no edges.
        cout << 0 << ' ' << 0 << '\n';
        return 0;
    }

    cout << best.weight << ' ' << best.edge_ids.size() << '\n';
    for (int id : best.edge_ids) {
        cout << edges[id].u << ' ' << edges[id].v << '\n';
    }
    return 0;
}
