declare const require: any;
declare const process: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Drone = { id: number; x: number; y: number };

const prob: number[][] = Array.isArray(data.probability) ? data.probability : [];
const drones: Drone[] = Array.isArray(data.drones) ? data.drones : [];
const maxSteps: number = Math.max(0, Math.floor(Number(data.maxStepsPerDrone) || 0));

const h = prob.length;
const w = h > 0 ? prob[0].length : 0;
const n = h * w;

function idx(x: number, y: number): number {
  return y * w + x;
}

function xOf(i: number): number {
  return i % w;
}

function yOf(i: number): number {
  return Math.floor(i / w);
}

function inBounds(x: number, y: number): boolean {
  return x >= 0 && y >= 0 && x < w && y < h;
}

const passable = new Uint8Array(n);
const cellValue = new Float64Array(n);
for (let y = 0; y < h; y++) {
  const row = prob[y] || [];
  for (let x = 0; x < w; x++) {
    const v = Number(row[x]);
    const i = idx(x, y);
    if (v !== -1) {
      passable[i] = 1;
      cellValue[i] = v > 0 ? v : 0;
    }
  }
}

const owner = new Int32Array(n);
const parent = new Int32Array(n);
const dist = new Int32Array(n);
owner.fill(-1);
parent.fill(-1);
dist.fill(-1);

const q = new Int32Array(Math.max(1, n));
let qHead = 0;
let qTail = 0;

const roots: number[] = [];
for (let i = 0; i < drones.length; i++) {
  const d = drones[i];
  const x = Math.floor(Number(d.x));
  const y = Math.floor(Number(d.y));
  roots.push(inBounds(x, y) ? idx(x, y) : -1);
}

for (let i = 0; i < roots.length; i++) {
  const r = roots[i];
  if (r >= 0 && passable[r] && owner[r] === -1) {
    owner[r] = i;
    parent[r] = -1;
    dist[r] = 0;
    q[qTail++] = r;
  }
}

const dirs = [1, 0, -1, 0, 1];
while (qHead < qTail) {
  const cur = q[qHead++];
  const cx = xOf(cur);
  const cy = yOf(cur);
  const nextOwner = owner[cur];
  const nd = dist[cur] + 1;

  for (let k = 0; k < 4; k++) {
    const nx = cx + dirs[k];
    const ny = cy + dirs[k + 1];
    if (!inBounds(nx, ny)) continue;
    const ni = idx(nx, ny);
    if (!passable[ni] || owner[ni] !== -1) continue;
    owner[ni] = nextOwner;
    parent[ni] = cur;
    dist[ni] = nd;
    q[qTail++] = ni;
  }
}

const firstChild = new Int32Array(n);
const nextSibling = new Int32Array(n);
firstChild.fill(-1);
nextSibling.fill(-1);

for (let i = 0; i < n; i++) {
  const p = parent[i];
  if (p >= 0) {
    nextSibling[i] = firstChild[p];
    firstChild[p] = i;
  }
}

const subtreeReward = new Float64Array(n);
const subtreeSize = new Int32Array(n);
const orderedChildren: (number[] | undefined)[] = new Array(n);

function buildPathForRoot(root: number): number[] {
  const preorder: number[] = [];
  const stack: number[] = [root];
  while (stack.length > 0) {
    const node = stack.pop()!;
    preorder.push(node);
    for (let child = firstChild[node]; child !== -1; child = nextSibling[child]) {
      stack.push(child);
    }
  }

  for (let i = preorder.length - 1; i >= 0; i--) {
    const node = preorder[i];
    let reward = cellValue[node];
    let size = 1;
    for (let child = firstChild[node]; child !== -1; child = nextSibling[child]) {
      reward += subtreeReward[child];
      size += subtreeSize[child];
    }
    subtreeReward[node] = reward;
    subtreeSize[node] = size;
  }

  for (const node of preorder) {
    const children: number[] = [];
    for (let child = firstChild[node]; child !== -1; child = nextSibling[child]) {
      if (subtreeReward[child] > 0) children.push(child);
    }
    children.sort((a, b) => {
      const pa = subtreeReward[a] / (subtreeSize[a] + 1);
      const pb = subtreeReward[b] / (subtreeSize[b] + 1);
      if (pb !== pa) return pb - pa;
      if (subtreeReward[b] !== subtreeReward[a]) return subtreeReward[b] - subtreeReward[a];
      if (subtreeSize[a] !== subtreeSize[b]) return subtreeSize[a] - subtreeSize[b];
      return a - b;
    });
    orderedChildren[node] = children;
  }

  const pathNodes: number[] = [root];
  let remaining = maxSteps;
  type Frame = { node: number; children: number[]; idx: number };
  const dfsStack: Frame[] = [{ node: root, children: orderedChildren[root] || [], idx: 0 }];

  while (dfsStack.length > 0 && remaining > 0) {
    const top = dfsStack[dfsStack.length - 1];
    if (top.idx < top.children.length) {
      const child = top.children[top.idx++];
      pathNodes.push(child);
      remaining--;
      dfsStack.push({ node: child, children: orderedChildren[child] || [], idx: 0 });
    } else {
      dfsStack.pop();
      if (dfsStack.length > 0 && remaining > 0) {
        const parentNode = dfsStack[dfsStack.length - 1].node;
        pathNodes.push(parentNode);
        remaining--;
      }
    }
  }

  while (pathNodes.length < maxSteps + 1) {
    pathNodes.push(pathNodes[pathNodes.length - 1]);
  }

  return pathNodes;
}

const result: number[][][] = [];
for (let i = 0; i < drones.length; i++) {
  const root = roots[i];
  if (root < 0 || root >= n || !passable[root] || owner[root] !== i) {
    const x = Math.floor(Number(drones[i].x));
    const y = Math.floor(Number(drones[i].y));
    const path: number[][] = [];
    for (let s = 0; s <= maxSteps; s++) path.push([x, y]);
    result.push(path);
    continue;
  }

  const nodes = buildPathForRoot(root);
  const path: number[][] = new Array(nodes.length);
  for (let j = 0; j < nodes.length; j++) {
    const p = nodes[j];
    path[j] = [xOf(p), yOf(p)];
  }
  result.push(path);
}

process.stdout.write(JSON.stringify({ paths: result }));
