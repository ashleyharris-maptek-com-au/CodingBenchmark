using System;
using System.Globalization;
using System.IO;
using System.Linq;

public static class Program
{
    const double G = 6.67430e-11;
    const double SecondsPerDay = 86400.0;

    public static void Main()
    {
        var input = Console.In.ReadToEnd();
        if (string.IsNullOrWhiteSpace(input))
        {
            Console.WriteLine("0");
            return;
        }

        var tokens = input.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        int p = 0;
        int numBodies = int.Parse(tokens[p++], CultureInfo.InvariantCulture);
        double warningDays = double.Parse(tokens[p++], CultureInfo.InvariantCulture);
        double impactorMass = double.Parse(tokens[p++], CultureInfo.InvariantCulture);
        double deltaVBudget = double.Parse(tokens[p++], CultureInfo.InvariantCulture);
        double asteroidMass = double.Parse(tokens[p++], CultureInfo.InvariantCulture);
        double t0UnixUtc = double.Parse(tokens[p++], CultureInfo.InvariantCulture);

        var bodies = new Body[numBodies];
        int earthIndex = -1;
        int asteroidIndex = -1;

        for (int i = 0; i < numBodies; i++)
        {
            string name = tokens[p++];
            double mass = double.Parse(tokens[p++], CultureInfo.InvariantCulture);
            Vec3 pos = new Vec3(
                double.Parse(tokens[p++], CultureInfo.InvariantCulture),
                double.Parse(tokens[p++], CultureInfo.InvariantCulture),
                double.Parse(tokens[p++], CultureInfo.InvariantCulture));
            Vec3 vel = new Vec3(
                double.Parse(tokens[p++], CultureInfo.InvariantCulture),
                double.Parse(tokens[p++], CultureInfo.InvariantCulture),
                double.Parse(tokens[p++], CultureInfo.InvariantCulture));

            bodies[i] = new Body(name, mass, pos, vel);
            if (name == "Earth") earthIndex = i;
            if (name == "Asteroid") asteroidIndex = i;
        }

        // The final token is the DE421 path. We do not need it for the fallback planner.
        string de421Path = p < tokens.Length ? tokens[p] : string.Empty;

        if (earthIndex < 0 || asteroidIndex < 0)
        {
            Console.WriteLine("0");
            return;
        }

        double horizonDays = Math.Max(0.1, warningDays);
        double desiredStep = Clamp(horizonDays / 4000.0, 0.25, 0.5);
        int steps = Math.Max(1, (int)Math.Ceiling(horizonDays / desiredStep));
        double dtDays = horizonDays / steps;
        double dtSeconds = dtDays * SecondsPerDay;

        var times = new double[steps + 1];
        var posHist = new Vec3[numBodies][];
        var velHist = new Vec3[numBodies][];
        for (int i = 0; i < numBodies; i++)
        {
            posHist[i] = new Vec3[steps + 1];
            velHist[i] = new Vec3[steps + 1];
        }

        var currentPos = new Vec3[numBodies];
        var currentVel = new Vec3[numBodies];
        for (int i = 0; i < numBodies; i++)
        {
            currentPos[i] = bodies[i].Pos;
            currentVel[i] = bodies[i].Vel;
            posHist[i][0] = currentPos[i];
            velHist[i][0] = currentVel[i];
        }

        for (int s = 0; s < steps; s++)
        {
            times[s] = s * dtDays;
            double h = (s == steps - 1) ? (horizonDays - s * dtDays) * SecondsPerDay : dtSeconds;
            if (h <= 0) h = dtSeconds;

            var acc0 = ComputeAccelerations(currentPos, bodies);
            for (int i = 0; i < numBodies; i++) currentVel[i] += acc0[i] * (0.5 * h);
            for (int i = 0; i < numBodies; i++) currentPos[i] += currentVel[i] * h;
            var acc1 = ComputeAccelerations(currentPos, bodies);
            for (int i = 0; i < numBodies; i++) currentVel[i] += acc1[i] * (0.5 * h);

            for (int i = 0; i < numBodies; i++)
            {
                posHist[i][s + 1] = currentPos[i];
                velHist[i][s + 1] = currentVel[i];
            }
        }
        times[steps] = horizonDays;

        var earthHist = new BodyHistory(bodies[earthIndex].Mass, posHist[earthIndex], velHist[earthIndex], dtDays);
        var asteroidHist = new BodyHistory(bodies[asteroidIndex].Mass, posHist[asteroidIndex], velHist[asteroidIndex], dtDays);
        var allHist = new BodyHistory[numBodies];
        for (int i = 0; i < numBodies; i++) allHist[i] = new BodyHistory(bodies[i].Mass, posHist[i], velHist[i], dtDays);

        // Search a modest grid of launch/impact windows and solve a shooting problem for each.
        Candidate bestFeasible = default;
        Candidate bestOverall = default;
        bool haveFeasible = false;
        bool haveOverall = false;

        double[] impactFractions = { 0.20, 0.35, 0.50, 0.70, 0.90, 1.00 };
        double[] launchFractions = { 0.00, 0.25, 0.50, 0.75 };

        foreach (double impactFrac in impactFractions)
        {
            double impactTime = Math.Min(horizonDays, Math.Max(0.1, warningDays * impactFrac));
            if (impactTime <= 0) continue;

            foreach (double launchFrac in launchFractions)
            {
                double launchTime = Math.Max(0.0, Math.Min(impactTime - 0.1, impactTime * launchFrac));
                if (launchTime < 0 || launchTime >= impactTime) continue;

                if (!TrySolveLaunch(allHist, earthHist, asteroidHist, launchTime, impactTime, deltaVBudget, out Vec3 dv, out double finalErr))
                    continue;

                double dvMag = dv.Norm();
                var cand = new Candidate
                {
                    LaunchTime = launchTime,
                    ImpactTime = impactTime,
                    LaunchDv = dv,
                    DvMag = dvMag,
                    ErrorNorm = finalErr,
                    Score = finalErr + Math.Max(0.0, dvMag - deltaVBudget) * 1e8 + dvMag * 1e-3
                };

                if (!haveOverall || cand.Score < bestOverall.Score)
                {
                    bestOverall = cand;
                    haveOverall = true;
                }

                if (dvMag <= deltaVBudget + 1e-6 && (!haveFeasible || cand.Score < bestFeasible.Score))
                {
                    bestFeasible = cand;
                    haveFeasible = true;
                }
            }
        }

        Candidate chosen = haveFeasible ? bestFeasible : bestOverall;
        if (!haveOverall)
        {
            // Fallback: launch immediately with a simple Earth-to-asteroid pointing vector.
            Vec3 e0 = InterpolateState(earthHist, 0.0).Pos;
            Vec3 a0 = InterpolateState(asteroidHist, Math.Min(horizonDays, warningDays)).Pos;
            Vec3 guess = a0 - e0;
            if (guess.Norm() > 0) guess = guess.Normalized() * Math.Min(deltaVBudget, 1000.0);
            chosen = new Candidate
            {
                LaunchTime = 0.0,
                ImpactTime = Math.Min(horizonDays, warningDays),
                LaunchDv = guess,
                DvMag = guess.Norm(),
                ErrorNorm = 0.0,
                Score = 0.0
            };
        }

        // Output a minimal plan: launch burn and a zero-delta-v terminal impact event.
        // The simulation should interpret the final line as the planned collision epoch.
        Console.WriteLine("2");
        WriteBurn(chosen.LaunchTime, chosen.LaunchDv);
        WriteBurn(chosen.ImpactTime, Vec3.Zero);
    }

    static bool TrySolveLaunch(BodyHistory[] allHist, BodyHistory earthHist, BodyHistory asteroidHist,
        double launchTime, double impactTime, double deltaVBudget, out Vec3 launchDv, out double finalError)
    {
        Vec3 earthPos = InterpolateState(earthHist, launchTime).Pos;
        Vec3 earthVel = InterpolateState(earthHist, launchTime).Vel;
        Vec3 targetPos = InterpolateState(asteroidHist, impactTime).Pos;
        double dt = (impactTime - launchTime) * SecondsPerDay;
        if (dt <= 0)
        {
            launchDv = Vec3.Zero;
            finalError = double.PositiveInfinity;
            return false;
        }

        // First guess ignores gravity.
        Vec3 dv = (targetPos - earthPos) / dt - earthVel;
        double guessMag = dv.Norm();
        if (guessMag > deltaVBudget * 3.0 && guessMag > 0)
        {
            dv = dv * (deltaVBudget * 3.0 / guessMag);
        }

        finalError = double.PositiveInfinity;
        for (int iter = 0; iter < 4; iter++)
        {
            State baseState = Propagate(allHist, earthPos, earthVel + dv, launchTime, impactTime);
            Vec3 err = baseState.Pos - targetPos;
            finalError = err.Norm();
            if (finalError < 1_000_000.0)
            {
                launchDv = dv;
                return true;
            }

            double eps = Math.Max(1.0, Math.Min(25.0, Math.Max(1.0, dv.Norm() * 0.001)));
            Vec3[] cols = new Vec3[3];
            for (int axis = 0; axis < 3; axis++)
            {
                Vec3 perturb = axis == 0 ? new Vec3(eps, 0, 0) : axis == 1 ? new Vec3(0, eps, 0) : new Vec3(0, 0, eps);
                State perturbed = Propagate(allHist, earthPos, earthVel + dv + perturb, launchTime, impactTime);
                cols[axis] = (perturbed.Pos - baseState.Pos) / eps;
            }

            if (!Solve3x3(cols[0], cols[1], cols[2], err, out Vec3 delta))
                break;

            dv -= delta;
            if (delta.Norm() < 0.5)
                break;
        }

        State finalState = Propagate(allHist, earthPos, earthVel + dv, launchTime, impactTime);
        finalError = (finalState.Pos - targetPos).Norm();
        launchDv = dv;
        return true;
    }

    static State Propagate(BodyHistory[] allHist, Vec3 pos, Vec3 vel, double startTimeDays, double endTimeDays)
    {
        double totalDays = endTimeDays - startTimeDays;
        if (totalDays <= 0)
            return new State(pos, vel);

        double stepDays = Math.Max(0.25, Math.Min(1.0, totalDays / 200.0));
        int steps = Math.Max(1, (int)Math.Ceiling(totalDays / stepDays));
        double hDays = totalDays / steps;
        double h = hDays * SecondsPerDay;

        double t = startTimeDays;
        for (int i = 0; i < steps; i++)
        {
            Vec3 k1p, k1v, k2p, k2v, k3p, k3v, k4p, k4v;

            k1p = vel;
            k1v = AccelerationAt(allHist, t, pos);

            Vec3 p2 = pos + k1p * (0.5 * h);
            Vec3 v2 = vel + k1v * (0.5 * h);
            k2p = v2;
            k2v = AccelerationAt(allHist, t + 0.5 * hDays, p2);

            Vec3 p3 = pos + k2p * (0.5 * h);
            Vec3 v3 = vel + k2v * (0.5 * h);
            k3p = v3;
            k3v = AccelerationAt(allHist, t + 0.5 * hDays, p3);

            Vec3 p4 = pos + k3p * h;
            Vec3 v4 = vel + k3v * h;
            k4p = v4;
            k4v = AccelerationAt(allHist, t + hDays, p4);

            pos += (k1p + 2.0 * k2p + 2.0 * k3p + k4p) * (h / 6.0);
            vel += (k1v + 2.0 * k2v + 2.0 * k3v + k4v) * (h / 6.0);
            t += hDays;
        }

        return new State(pos, vel);
    }

    static Vec3 AccelerationAt(BodyHistory[] allHist, double tDays, Vec3 pos)
    {
        Vec3 acc = Vec3.Zero;
        for (int i = 0; i < allHist.Length; i++)
        {
            Vec3 bp = InterpolatePosition(allHist[i].Positions, tDays, allHist[i].StepDays);
            Vec3 r = bp - pos;
            double dist2 = r.LengthSquared();
            double dist = Math.Sqrt(dist2 + 1e-12);
            double invR3 = 1.0 / (dist2 * dist + 1e-30);
            acc += r * (G * allHist[i].Mass * invR3);
        }
        return acc;
    }

    static Vec3 InterpolatePosition(Vec3[] history, double tDays, double stepDays)
    {
        if (tDays <= 0) return history[0];
        int lastIndex = history.Length - 1;
        double maxTime = lastIndex * stepDays;
        if (tDays >= maxTime) return history[lastIndex];
        double x = tDays / stepDays;
        int i = (int)Math.Floor(x);
        if (i >= lastIndex) return history[lastIndex];
        double frac = x - i;
        return history[i] * (1.0 - frac) + history[i + 1] * frac;
    }

    static State InterpolateState(BodyHistory hist, double tDays)
    {
        int last = hist.Positions.Length - 1;
        if (tDays <= 0) return new State(hist.Positions[0], hist.Velocities[0]);
        double maxTime = last * hist.StepDays;
        if (tDays >= maxTime) return new State(hist.Positions[last], hist.Velocities[last]);
        double x = tDays / hist.StepDays;
        int i = (int)Math.Floor(x);
        if (i >= last) return new State(hist.Positions[last], hist.Velocities[last]);
        double frac = x - i;
        Vec3 p = hist.Positions[i] * (1.0 - frac) + hist.Positions[i + 1] * frac;
        Vec3 v = hist.Velocities[i] * (1.0 - frac) + hist.Velocities[i + 1] * frac;
        return new State(p, v);
    }

    static Vec3[] ComputeAccelerations(Vec3[] positions, Body[] bodies)
    {
        int n = bodies.Length;
        var acc = new Vec3[n];
        for (int i = 0; i < n; i++)
        {
            Vec3 ai = Vec3.Zero;
            Vec3 pi = positions[i];
            for (int j = 0; j < n; j++)
            {
                if (i == j) continue;
                Vec3 r = positions[j] - pi;
                double dist2 = r.LengthSquared() + 1e-12;
                double dist = Math.Sqrt(dist2);
                double invR3 = 1.0 / (dist2 * dist + 1e-30);
                ai += r * (G * bodies[j].Mass * invR3);
            }
            acc[i] = ai;
        }
        return acc;
    }

    static bool Solve3x3(Vec3 c0, Vec3 c1, Vec3 c2, Vec3 b, out Vec3 x)
    {
        // Solve [c0 c1 c2] x = b by Gaussian elimination.
        double[,] m = new double[3, 4];
        m[0, 0] = c0.X; m[1, 0] = c0.Y; m[2, 0] = c0.Z;
        m[0, 1] = c1.X; m[1, 1] = c1.Y; m[2, 1] = c1.Z;
        m[0, 2] = c2.X; m[1, 2] = c2.Y; m[2, 2] = c2.Z;
        m[0, 3] = b.X;  m[1, 3] = b.Y;  m[2, 3] = b.Z;

        for (int col = 0; col < 3; col++)
        {
            int pivot = col;
            double best = Math.Abs(m[col, col]);
            for (int row = col + 1; row < 3; row++)
            {
                double val = Math.Abs(m[row, col]);
                if (val > best)
                {
                    best = val;
                    pivot = row;
                }
            }
            if (best < 1e-18)
            {
                x = Vec3.Zero;
                return false;
            }
            if (pivot != col)
            {
                for (int k = col; k < 4; k++)
                {
                    double tmp = m[col, k];
                    m[col, k] = m[pivot, k];
                    m[pivot, k] = tmp;
                }
            }

            double div = m[col, col];
            for (int k = col; k < 4; k++) m[col, k] /= div;

            for (int row = 0; row < 3; row++)
            {
                if (row == col) continue;
                double factor = m[row, col];
                for (int k = col; k < 4; k++) m[row, k] -= factor * m[col, k];
            }
        }

        x = new Vec3(m[0, 3], m[1, 3], m[2, 3]);
        return true;
    }

    static void WriteBurn(double timeDays, Vec3 dv)
    {
        Console.WriteLine(
            $"{Fmt(timeDays)} {Fmt(dv.X)} {Fmt(dv.Y)} {Fmt(dv.Z)}");
    }

    static string Fmt(double x) => x.ToString("0.###############E+0", CultureInfo.InvariantCulture);

    static double Clamp(double x, double lo, double hi) => x < lo ? lo : (x > hi ? hi : x);

    readonly struct Body
    {
        public readonly string Name;
        public readonly double Mass;
        public readonly Vec3 Pos;
        public readonly Vec3 Vel;

        public Body(string name, double mass, Vec3 pos, Vec3 vel)
        {
            Name = name;
            Mass = mass;
            Pos = pos;
            Vel = vel;
        }
    }

    readonly struct BodyHistory
    {
        public readonly double Mass;
        public readonly Vec3[] Positions;
        public readonly Vec3[] Velocities;
        public readonly double StepDays;

        public BodyHistory(double mass, Vec3[] positions, Vec3[] velocities, double stepDays)
        {
            Mass = mass;
            Positions = positions;
            Velocities = velocities;
            StepDays = stepDays;
        }
    }

    readonly struct State
    {
        public readonly Vec3 Pos;
        public readonly Vec3 Vel;
        public State(Vec3 pos, Vec3 vel)
        {
            Pos = pos;
            Vel = vel;
        }
    }

    struct Candidate
    {
        public double LaunchTime;
        public double ImpactTime;
        public Vec3 LaunchDv;
        public double DvMag;
        public double ErrorNorm;
        public double Score;
    }

    struct Vec3
    {
        public double X;
        public double Y;
        public double Z;

        public static Vec3 Zero => new Vec3(0, 0, 0);

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public double LengthSquared() => X * X + Y * Y + Z * Z;
        public double Norm() => Math.Sqrt(LengthSquared());
        public Vec3 Normalized()
        {
            double n = Norm();
            return n > 0 ? this / n : Zero;
        }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator *(double s, Vec3 a) => a * s;
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);
    }
}
