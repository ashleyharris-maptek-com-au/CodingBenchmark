using System;
using System.Collections.Generic;
using System.IO;
using System.Numerics;
using System.Text;

public class Program
{
    private sealed class Constraint
    {
        public int[] Vars = Array.Empty<int>();
        public int Rem;
    }

    private struct ExactConstraint
    {
        public ulong Mask;
        public int Rem;
    }

    private readonly struct StateKey : IEquatable<StateKey>
    {
        public readonly ulong Mine;
        public readonly ulong Safe;

        public StateKey(ulong mine, ulong safe)
        {
            Mine = mine;
            Safe = safe;
        }

        public bool Equals(StateKey other) => Mine == other.Mine && Safe == other.Safe;
        public override bool Equals(object? obj) => obj is StateKey other && Equals(other);
        public override int GetHashCode() => HashCode.Combine(Mine, Safe);
    }

    private static int H;
    private static int W;
    private static string[] Grid = Array.Empty<string>();

    private static readonly Dictionary<long, int> PosToVar = new();
    private static readonly List<int> VarRows = new();
    private static readonly List<int> VarCols = new();

    private static Constraint[] Constraints = Array.Empty<Constraint>();
    private static List<int>[] VarToConstraints = Array.Empty<List<int>>();
    private static int[] ConsRem = Array.Empty<int>();
    private static int[] ConsUnknown = Array.Empty<int>();

    private static byte[] VarState = Array.Empty<byte>(); // 0 unknown, 1 safe, 2 mine
    private static bool[] SafeOut = Array.Empty<bool>();
    private static int SafeCount;

    private static Queue<int> Queue = new();
    private static bool[] InQueue = Array.Empty<bool>();

    private const int ExactThreshold = 22;

    public static void Main()
    {
        var reader = new StreamReader(Console.OpenStandardInput(), Encoding.ASCII, false, 1 << 20);
        string? first = reader.ReadLine();
        if (first == null)
        {
            return;
        }

        string[] hw = first.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        H = int.Parse(hw[0]);
        W = int.Parse(hw[1]);

        Grid = new string[H];
        for (int r = 0; r < H; r++)
        {
            string? line = reader.ReadLine();
            if (line == null)
            {
                line = string.Empty;
            }
            if (line.Length < W)
            {
                line = line.PadRight(W, ' ');
            }
            Grid[r] = line;
        }

        BuildConstraints();
        Solve();
        WriteAnswer();
    }

    private static void BuildConstraints()
    {
        PosToVar.Clear();
        VarRows.Clear();
        VarCols.Clear();

        List<int[]> constraintVars = new();
        List<int> constraintRem = new();

        for (int r = 0; r < H; r++)
        {
            string row = Grid[r];
            for (int c = 0; c < W; c++)
            {
                char ch = row[c];
                if (ch < '0' || ch > '8')
                {
                    continue;
                }

                int clue = ch - '0';
                int flagged = 0;
                int[] tmp = new int[8];
                int count = 0;

                for (int dr = -1; dr <= 1; dr++)
                {
                    int nr = r + dr;
                    if ((uint)nr >= (uint)H)
                    {
                        continue;
                    }

                    for (int dc = -1; dc <= 1; dc++)
                    {
                        if (dr == 0 && dc == 0)
                        {
                            continue;
                        }

                        int nc = c + dc;
                        if ((uint)nc >= (uint)W)
                        {
                            continue;
                        }

                        char n = Grid[nr][nc];
                        if (n == '*')
                        {
                            flagged++;
                        }
                        else if (n == ' ')
                        {
                            long key = (((long)nr) << 32) ^ (uint)nc;
                            if (!PosToVar.TryGetValue(key, out int id))
                            {
                                id = VarRows.Count;
                                PosToVar[key] = id;
                                VarRows.Add(nr);
                                VarCols.Add(nc);
                            }
                            tmp[count++] = id;
                        }
                    }
                }

                int rem = clue - flagged;
                if (count > 0)
                {
                    int[] vars = new int[count];
                    Array.Copy(tmp, vars, count);
                    constraintVars.Add(vars);
                    constraintRem.Add(rem);
                }
            }
        }

        int varCount = VarRows.Count;
        int consCount = constraintVars.Count;

        Constraints = new Constraint[consCount];
        ConsRem = new int[consCount];
        ConsUnknown = new int[consCount];
        VarState = new byte[varCount];
        SafeOut = new bool[varCount];
        VarToConstraints = new List<int>[varCount];
        InQueue = new bool[consCount];
        SafeCount = 0;

        for (int i = 0; i < varCount; i++)
        {
            VarToConstraints[i] = new List<int>(4);
        }

        for (int i = 0; i < consCount; i++)
        {
            Constraints[i] = new Constraint
            {
                Vars = constraintVars[i],
                Rem = constraintRem[i]
            };
            ConsRem[i] = constraintRem[i];
            ConsUnknown[i] = constraintVars[i].Length;
            foreach (int v in constraintVars[i])
            {
                VarToConstraints[v].Add(i);
            }
        }
    }

    private static void Solve()
    {
        while (true)
        {
            bool changed = false;
            changed |= PropagateSimple();
            changed |= SolveExactComponents();
            if (!changed)
            {
                break;
            }
        }
    }

    private static bool PropagateSimple()
    {
        bool changed = false;

        Queue.Clear();
        Array.Fill(InQueue, false);
        for (int i = 0; i < Constraints.Length; i++)
        {
            if (ConsUnknown[i] > 0)
            {
                Queue.Enqueue(i);
                InQueue[i] = true;
            }
        }

        while (Queue.Count > 0)
        {
            int ci = Queue.Dequeue();
            InQueue[ci] = false;

            int unk = ConsUnknown[ci];
            if (unk == 0)
            {
                continue;
            }

            int rem = ConsRem[ci];
            if (rem < 0 || rem > unk)
            {
                continue;
            }

            if (rem != 0 && rem != unk)
            {
                continue;
            }

            byte assignState = rem == 0 ? (byte)1 : (byte)2;
            int[] vars = Constraints[ci].Vars;
            for (int i = 0; i < vars.Length; i++)
            {
                if (AssignVar(vars[i], assignState))
                {
                    changed = true;
                }
            }
        }

        return changed;
    }

    private static bool AssignVar(int varId, byte state)
    {
        if (VarState[varId] != 0)
        {
            return false;
        }

        VarState[varId] = state;
        if (state == 1)
        {
            SafeOut[varId] = true;
            SafeCount++;
        }

        List<int> incident = VarToConstraints[varId];
        for (int i = 0; i < incident.Count; i++)
        {
            int ci = incident[i];
            ConsUnknown[ci]--;
            if (state == 2)
            {
                ConsRem[ci]--;
            }

            if (!InQueue[ci])
            {
                Queue.Enqueue(ci);
                InQueue[ci] = true;
            }
        }

        return true;
    }

    private sealed class ExactComponent
    {
        public int[] GlobalVarIds = Array.Empty<int>();
        public ExactConstraint[] Constraints = Array.Empty<ExactConstraint>();
        public ulong AllMask;
        public int[] Degrees = Array.Empty<int>();
    }

    private static bool SolveExactComponents()
    {
        int varCount = VarRows.Count;
        if (varCount == 0)
        {
            return false;
        }

        bool[] visitedVar = new bool[varCount];
        bool[] visitedCons = new bool[Constraints.Length];
        Queue<int> q = new();
        bool changed = false;

        for (int start = 0; start < varCount; start++)
        {
            if (VarState[start] != 0 || visitedVar[start] || VarToConstraints[start].Count == 0)
            {
                continue;
            }

            List<int> compVars = new();
            List<int> compCons = new();

            visitedVar[start] = true;
            q.Enqueue(start);

            while (q.Count > 0)
            {
                int v = q.Dequeue();
                compVars.Add(v);

                List<int> incident = VarToConstraints[v];
                for (int i = 0; i < incident.Count; i++)
                {
                    int ci = incident[i];
                    if (ConsUnknown[ci] == 0 || visitedCons[ci])
                    {
                        continue;
                    }

                    visitedCons[ci] = true;
                    compCons.Add(ci);

                    int[] vars = Constraints[ci].Vars;
                    for (int j = 0; j < vars.Length; j++)
                    {
                        int u = vars[j];
                        if (VarState[u] == 0 && !visitedVar[u])
                        {
                            visitedVar[u] = true;
                            q.Enqueue(u);
                        }
                    }
                }
            }

            if (compVars.Count <= ExactThreshold && compCons.Count > 0)
            {
                ExactComponent comp = BuildExactComponent(compVars, compCons);
                if (comp.GlobalVarIds.Length > 0 && comp.Constraints.Length > 0)
                {
                    changed |= ApplyExactDeductions(comp);
                }
            }
        }

        return changed;
    }

    private static ExactComponent BuildExactComponent(List<int> compVars, List<int> compCons)
    {
        var localIndex = new Dictionary<int, int>(compVars.Count);
        for (int i = 0; i < compVars.Count; i++)
        {
            localIndex[compVars[i]] = i;
        }

        List<ExactConstraint> exactCons = new(compCons.Count);
        int[] degrees = new int[compVars.Count];

        for (int k = 0; k < compCons.Count; k++)
        {
            int ci = compCons[k];
            int[] vars = Constraints[ci].Vars;
            ulong mask = 0;
            for (int i = 0; i < vars.Length; i++)
            {
                int v = vars[i];
                if (VarState[v] != 0)
                {
                    continue;
                }
                if (!localIndex.TryGetValue(v, out int bit))
                {
                    continue;
                }
                mask |= 1UL << bit;
            }

            if (mask == 0)
            {
                continue;
            }

            ExactConstraint ec = new ExactConstraint
            {
                Mask = mask,
                Rem = ConsRem[ci]
            };
            exactCons.Add(ec);

            ulong t = mask;
            while (t != 0)
            {
                int bit = BitOperations.TrailingZeroCount(t);
                degrees[bit]++;
                t &= t - 1;
            }
        }

        ulong allMask = compVars.Count == 64 ? ulong.MaxValue : ((1UL << compVars.Count) - 1UL);
        ExactComponent comp = new ExactComponent
        {
            GlobalVarIds = compVars.ToArray(),
            Constraints = exactCons.ToArray(),
            AllMask = allMask,
            Degrees = degrees
        };
        return comp;
    }

    private static bool ApplyExactDeductions(ExactComponent comp)
    {
        bool changed = false;
        var memo = new Dictionary<StateKey, bool>();
        ulong safeMask = 0;
        ulong mineMask = 0;

        for (int i = 0; i < comp.GlobalVarIds.Length; i++)
        {
            int g = comp.GlobalVarIds[i];
            if (VarState[g] != 0)
            {
                continue;
            }

            ulong bit = 1UL << i;
            bool minePossible = ExistsSolution(comp, bit, 0, memo);
            if (!minePossible)
            {
                safeMask |= bit;
                continue;
            }

            bool safePossible = ExistsSolution(comp, 0, bit, memo);
            if (!safePossible)
            {
                mineMask |= bit;
            }
        }

        for (int i = 0; i < comp.GlobalVarIds.Length; i++)
        {
            int g = comp.GlobalVarIds[i];
            ulong bit = 1UL << i;
            if ((safeMask & bit) != 0)
            {
                if (AssignVar(g, 1))
                {
                    changed = true;
                }
            }
            else if ((mineMask & bit) != 0)
            {
                if (AssignVar(g, 2))
                {
                    changed = true;
                }
            }
        }

        return changed;
    }

    private static bool ExistsSolution(ExactComponent comp, ulong mineMask, ulong safeMask, Dictionary<StateKey, bool> memo)
    {
        if ((mineMask & safeMask) != 0)
        {
            return false;
        }

        StateKey key = new StateKey(mineMask, safeMask);
        if (memo.TryGetValue(key, out bool cached))
        {
            return cached;
        }

        bool result = ExistsSolutionCore(comp, mineMask, safeMask, memo);
        memo[key] = result;
        return result;
    }

    private static bool ExistsSolutionCore(ExactComponent comp, ulong mineMask, ulong safeMask, Dictionary<StateKey, bool> memo)
    {
        while (true)
        {
            bool changed = false;
            ulong assigned = mineMask | safeMask;

            for (int i = 0; i < comp.Constraints.Length; i++)
            {
                ExactConstraint c = comp.Constraints[i];
                ulong relevantAssigned = c.Mask & assigned;
                int assignedCount = BitOperations.PopCount(relevantAssigned);
                int mineCount = BitOperations.PopCount(c.Mask & mineMask);
                ulong unassignedMask = c.Mask & ~assigned;
                int unassignedCount = BitOperations.PopCount(unassignedMask);

                if (mineCount > c.Rem || mineCount + unassignedCount < c.Rem)
                {
                    return false;
                }

                if (unassignedCount == 0)
                {
                    continue;
                }

                if (mineCount == c.Rem)
                {
                    ulong newSafe = unassignedMask;
                    if (newSafe != 0)
                    {
                        safeMask |= newSafe;
                        changed = true;
                    }
                }
                else if (mineCount + unassignedCount == c.Rem)
                {
                    ulong newMine = unassignedMask;
                    if (newMine != 0)
                    {
                        mineMask |= newMine;
                        changed = true;
                    }
                }
            }

            if (!changed)
            {
                break;
            }
        }

        ulong allAssigned = mineMask | safeMask;
        if (allAssigned == comp.AllMask)
        {
            return true;
        }

        ulong unassignedAll = comp.AllMask & ~allAssigned;
        if (unassignedAll == 0)
        {
            return true;
        }

        int bestBit = -1;
        int bestSize = int.MaxValue;
        for (int i = 0; i < comp.Constraints.Length; i++)
        {
            ulong ua = comp.Constraints[i].Mask & unassignedAll;
            if (ua == 0)
            {
                continue;
            }

            int size = BitOperations.PopCount(ua);
            if (size < bestSize)
            {
                bestSize = size;
                bestBit = BitOperations.TrailingZeroCount(ua);
                if (size == 1)
                {
                    break;
                }
            }
        }

        if (bestBit < 0)
        {
            bestBit = BitOperations.TrailingZeroCount(unassignedAll);
        }

        ulong bit = 1UL << bestBit;
        if (ExistsSolution(comp, mineMask | bit, safeMask, memo))
        {
            return true;
        }

        return ExistsSolution(comp, mineMask, safeMask | bit, memo);
    }

    private static void WriteAnswer()
    {
        var output = new StreamWriter(Console.OpenStandardOutput(), Encoding.ASCII, 1 << 20);
        output.WriteLine(SafeCount);
        for (int i = 0; i < VarRows.Count; i++)
        {
            if (SafeOut[i])
            {
                output.Write(VarCols[i]);
                output.Write(' ');
                output.Write(VarRows[i]);
                output.Write('\n');
            }
        }
        output.Flush();
    }
}
