OpCapability Shader
OpCapability Geometry
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%funct = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%uint_4 = OpConstant %uint 4
%float_0025 = OpConstant %float 0.025

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_input_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_input_arr_gl_PerVertex_in Input

%arr_v3float = OpTypeArray %v3float %uint_4
%ptr_input_arr_v3float = OpTypePointer Input %arr_v3float
%worldPos = OpVariable %ptr_input_arr_v3float Input
%normal = OpVariable %ptr_input_arr_v3float Input
%color = OpVariable %ptr_input_arr_v3float Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_output_gl_PerVertex_out Output

%ptr_input_v4float = OpTypePointer Input %v4float
%ptr_input_v3float = OpTypePointer Input %v3float
%ptr_output_v4float = OpTypePointer Output %v4float
%ptr_output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_output_v3float Output

%main = OpFunction %void None %funct
%label = OpLabel

%pos0_ptr = OpAccessChain %ptr_input_v4float %gl_in %int_0 %int_0
%pos1_ptr = OpAccessChain %ptr_input_v4float %gl_in %int_1 %int_0
%pos2_ptr = OpAccessChain %ptr_input_v4float %gl_in %int_2 %int_0
%pos3_ptr = OpAccessChain %ptr_input_v4float %gl_in %int_3 %int_0

%pos0 = OpLoad %v4float %pos0_ptr
%pos1 = OpLoad %v4float %pos1_ptr
%pos2 = OpLoad %v4float %pos2_ptr
%pos3 = OpLoad %v4float %pos3_ptr

%start_color_ptr = OpAccessChain %ptr_input_v3float %color %int_1
%end_color_ptr = OpAccessChain %ptr_input_v3float %color %int_2
%start_color = OpLoad %v3float %start_color_ptr
%end_color = OpLoad %v3float %end_color_ptr

%pos0_xy = OpVectorShuffle %v2float %pos0 %pos0 0 1
%pos1_xy = OpVectorShuffle %v2float %pos1 %pos1 0 1
%pos2_xy = OpVectorShuffle %v2float %pos2 %pos2 0 1
%pos3_xy = OpVectorShuffle %v2float %pos3 %pos3 0 1

%t0_delta = OpFSub %v2float %pos2_xy %pos0_xy
%t1_delta = OpFSub %v2float %pos3_xy %pos1_xy

%t0 = OpExtInst %v2float %glsl450 Normalize %t0_delta
%t1 = OpExtInst %v2float %glsl450 Normalize %t1_delta

%t0_x = OpCompositeExtract %float %t0 0
%t0_y = OpCompositeExtract %float %t0 1
%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1

%neg_t0_y = OpFNegate %float %t0_y
%neg_t1_y = OpFNegate %float %t1_y

%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x

%off0 = OpVectorTimesScalar %v2float %perp0 %float_0025
%off1 = OpVectorTimesScalar %v2float %perp1 %float_0025

%start_plus = OpFAdd %v2float %pos1_xy %off0
%start_minus = OpFSub %v2float %pos1_xy %off0
%end_plus = OpFAdd %v2float %pos2_xy %off1
%end_minus = OpFSub %v2float %pos2_xy %off1

%start_plus_x = OpCompositeExtract %float %start_plus 0
%start_plus_y = OpCompositeExtract %float %start_plus 1
%start_minus_x = OpCompositeExtract %float %start_minus 0
%start_minus_y = OpCompositeExtract %float %start_minus 1
%end_plus_x = OpCompositeExtract %float %end_plus 0
%end_plus_y = OpCompositeExtract %float %end_plus 1
%end_minus_x = OpCompositeExtract %float %end_minus 0
%end_minus_y = OpCompositeExtract %float %end_minus 1

%start_z = OpCompositeExtract %float %pos1 2
%start_w = OpCompositeExtract %float %pos1 3
%end_z = OpCompositeExtract %float %pos2 2
%end_w = OpCompositeExtract %float %pos2 3

%v0 = OpCompositeConstruct %v4float %start_plus_x %start_plus_y %start_z %start_w
%v1 = OpCompositeConstruct %v4float %start_minus_x %start_minus_y %start_z %start_w
%v2 = OpCompositeConstruct %v4float %end_plus_x %end_plus_y %end_z %end_w
%v3 = OpCompositeConstruct %v4float %end_minus_x %end_minus_y %end_z %end_w

%gl_out_pos_ptr0 = OpAccessChain %ptr_output_v4float %gl_out %int_0
OpStore %gl_out_pos_ptr0 %v0
OpStore %outColor %start_color
OpEmitVertex

%gl_out_pos_ptr1 = OpAccessChain %ptr_output_v4float %gl_out %int_0
OpStore %gl_out_pos_ptr1 %v1
OpStore %outColor %start_color
OpEmitVertex

%gl_out_pos_ptr2 = OpAccessChain %ptr_output_v4float %gl_out %int_0
OpStore %gl_out_pos_ptr2 %v2
OpStore %outColor %end_color
OpEmitVertex

%gl_out_pos_ptr3 = OpAccessChain %ptr_output_v4float %gl_out %int_0
OpStore %gl_out_pos_ptr3 %v3
OpStore %outColor %end_color
OpEmitVertex

OpEndPrimitive
OpFunctionEnd
