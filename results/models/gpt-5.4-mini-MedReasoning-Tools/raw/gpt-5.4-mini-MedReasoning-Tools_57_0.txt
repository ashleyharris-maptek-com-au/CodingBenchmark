const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
if (!input) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const data = JSON.parse(input);
const W = data.width | 0;
const H = data.height | 0;
const N = W * H;
const grid = data.grid || [];
const towns = data.towns || [];
const ignitionsInput = data.ignitions || [];
const budget = Math.max(0, data.cutBudget | 0);

const PASS = 46; // '.'

const passable = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < W; x++) {
    passable[y * W + x] = row.charCodeAt(x) === PASS ? 1 : 0;
  }
}

const townValueByIdx = new Float64Array(N);
const ignitionByIdx = new Uint8Array(N);
let totalTownValue = 0;
for (const t of towns) {
  const x = t.x | 0;
  const y = t.y | 0;
  if (x < 0 || x >= W || y < 0 || y >= H) continue;
  const idx = y * W + x;
  const v = Number(t.value) || 0;
  if (v > 0 && passable[idx]) {
    townValueByIdx[idx] += v;
    totalTownValue += v;
  }
}
for (const ig of ignitionsInput) {
  const x = ig[0] | 0;
  const y = ig[1] | 0;
  if (x < 0 || x >= W || y < 0 || y >= H) continue;
  ignitionByIdx[y * W + x] = 1;
}

if (budget === 0 || totalTownValue <= 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

// Multi-source BFS to find the fire-reachable region and build a parent tree.
const reachable = new Uint8Array(N);
const dist = new Int32Array(N);
dist.fill(-1);
const parentIdx = new Int32Array(N);
parentIdx.fill(-1);
const visitOrder = [];
const queue = new Int32Array(N);
let qh = 0;
let qt = 0;

for (let idx = 0; idx < N; idx++) {
  if (ignitionByIdx[idx] && passable[idx] && !reachable[idx]) {
    reachable[idx] = 1;
    dist[idx] = 0;
    queue[qt++] = idx;
    visitOrder.push(idx);
  }
}

while (qh < qt) {
  const idx = queue[qh++];
  const x = idx % W;
  const y = (idx / W) | 0;
  const nd = dist[idx] + 1;

  if (x > 0) {
    const ni = idx - 1;
    if (passable[ni] && !reachable[ni]) {
      reachable[ni] = 1;
      dist[ni] = nd;
      parentIdx[ni] = idx;
      queue[qt++] = ni;
      visitOrder.push(ni);
    }
  }
  if (x + 1 < W) {
    const ni = idx + 1;
    if (passable[ni] && !reachable[ni]) {
      reachable[ni] = 1;
      dist[ni] = nd;
      parentIdx[ni] = idx;
      queue[qt++] = ni;
      visitOrder.push(ni);
    }
  }
  if (y > 0) {
    const ni = idx - W;
    if (passable[ni] && !reachable[ni]) {
      reachable[ni] = 1;
      dist[ni] = nd;
      parentIdx[ni] = idx;
      queue[qt++] = ni;
      visitOrder.push(ni);
    }
  }
  if (y + 1 < H) {
    const ni = idx + W;
    if (passable[ni] && !reachable[ni]) {
      reachable[ni] = 1;
      dist[ni] = nd;
      parentIdx[ni] = idx;
      queue[qt++] = ni;
      visitOrder.push(ni);
    }
  }
}

// Map reachable cells to compact ids.
const cellId = new Int32Array(N);
cellId.fill(-1);
const idToIdx = [];
for (let i = 0; i < visitOrder.length; i++) {
  const idx = visitOrder[i];
  const id = idToIdx.length;
  cellId[idx] = id;
  idToIdx.push(idx);
}
const R = idToIdx.length;

if (R === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const idxToId = new Int32Array(N);
idxToId.fill(-1);
for (let id = 0; id < R; id++) {
  idxToId[idToIdx[id]] = id;
}

const isIgnition = new Uint8Array(R);
const townValue = new Float64Array(R);
const parent = new Int32Array(R);
parent.fill(-1);
const depth = new Int32Array(R);
const deg = new Uint8Array(R);
const nbr = new Int32Array(R * 4);
nbr.fill(-1);

for (let id = 0; id < R; id++) {
  const idx = idToIdx[id];
  townValue[id] = townValueByIdx[idx];
  isIgnition[id] = ignitionByIdx[idx];
  depth[id] = dist[idx];
  const pidx = parentIdx[idx];
  if (pidx !== -1) parent[id] = idxToId[pidx];
}

// Build reachable adjacency (at most 4 neighbors).
for (let id = 0; id < R; id++) {
  const idx = idToIdx[id];
  const x = idx % W;
  const y = (idx / W) | 0;
  let k = 0;
  if (x > 0) {
    const nid = idxToId[idx - 1];
    if (nid !== -1) nbr[id * 4 + k++] = nid;
  }
  if (x + 1 < W) {
    const nid = idxToId[idx + 1];
    if (nid !== -1) nbr[id * 4 + k++] = nid;
  }
  if (y > 0) {
    const nid = idxToId[idx - W];
    if (nid !== -1) nbr[id * 4 + k++] = nid;
  }
  if (y + 1 < H) {
    const nid = idxToId[idx + W];
    if (nid !== -1) nbr[id * 4 + k++] = nid;
  }
  deg[id] = k;
}

// BFS tree children.
const firstChild = new Int32Array(R);
firstChild.fill(-1);
const nextSibling = new Int32Array(R);
nextSibling.fill(-1);
for (let id = 0; id < R; id++) {
  const p = parent[id];
  if (p !== -1) {
    nextSibling[id] = firstChild[p];
    firstChild[p] = id;
  }
}

// Tree subtree town totals on the BFS forest.
const subTown = new Float64Array(R);
subTown.set(townValue);
for (let i = visitOrder.length - 1; i >= 0; i--) {
  const id = idxToId[visitOrder[i]];
  const p = parent[id];
  if (p !== -1) subTown[p] += subTown[id];
}

// Euler tour for subtree coverage.
const tin = new Int32Array(R);
const tout = new Int32Array(R);
let timer = 0;
const stackNode = new Int32Array(R + 5);
const stackChild = new Int32Array(R + 5);
const stackState = new Uint8Array(R + 5);
let sp = 0;
for (let id = 0; id < R; id++) {
  if (parent[id] !== -1) continue;
  stackNode[sp] = id;
  stackChild[sp] = firstChild[id];
  stackState[sp] = 0;
  sp++;
  while (sp > 0) {
    const top = sp - 1;
    const u = stackNode[top];
    if (stackState[top] === 0) {
      tin[u] = timer++;
      stackState[top] = 1;
    } else {
      const v = stackChild[top];
      if (v !== -1) {
        stackChild[top] = nextSibling[v];
        stackNode[sp] = v;
        stackChild[sp] = firstChild[v];
        stackState[sp] = 0;
        sp++;
      } else {
        tout[u] = timer - 1;
        sp--;
      }
    }
  }
}

function fenwickAdd(bit, idx, delta) {
  for (let i = idx; i < bit.length; i += i & -i) bit[i] += delta;
}

function fenwickSum(bit, idx) {
  let s = 0;
  for (let i = idx; i > 0; i -= i & -i) s += bit[i];
  return s;
}

function cuttableCell(id) {
  return !isIgnition[id] && townValue[id] <= 0;
}

// Tarjan articulation analysis on the reachable graph.
const savedArt = new Float64Array(R);

function computeArticulationSavings() {
  const ROOT = R;
  const totalNodes = R + 1;
  const disc = new Int32Array(totalNodes);
  const low = new Int32Array(totalNodes);
  const dfsParent = new Int32Array(totalNodes);
  dfsParent.fill(-1);
  const subIgn = new Uint8Array(totalNodes);
  const subTown2 = new Float64Array(totalNodes);
  const rootChildren = [];
  for (let id = 0; id < R; id++) {
    if (isIgnition[id]) rootChildren.push(id);
  }
  subIgn[ROOT] = 1;
  const stNode = new Int32Array(totalNodes * 2 + 5);
  const stNext = new Int32Array(totalNodes * 2 + 5);
  let top = 0;
  stNode[top] = ROOT;
  stNext[top] = 0;
  top++;
  let time = 1;
  while (top > 0) {
    const frame = top - 1;
    const u = stNode[frame];
    if (disc[u] === 0) {
      disc[u] = low[u] = time++;
      if (u === ROOT) {
        subIgn[u] = 1;
      } else {
        subIgn[u] = isIgnition[u];
        subTown2[u] = townValue[u];
      }
    }

    let progressed = false;
    if (u === ROOT) {
      let i = stNext[frame];
      while (i < rootChildren.length) {
        const v = rootChildren[i++];
        stNext[frame] = i;
        if (disc[v] === 0) {
          dfsParent[v] = ROOT;
          stNode[top] = v;
          stNext[top] = 0;
          top++;
          progressed = true;
          break;
        } else {
          if (v !== dfsParent[u]) low[u] = Math.min(low[u], disc[v]);
        }
      }
    } else {
      let i = stNext[frame];
      while (i < deg[u]) {
        const v = nbr[u * 4 + i++];
        stNext[frame] = i;
        if (v < 0) continue;
        if (disc[v] === 0) {
          dfsParent[v] = u;
          stNode[top] = v;
          stNext[top] = 0;
          top++;
          progressed = true;
          break;
        } else if (v !== dfsParent[u]) {
          low[u] = Math.min(low[u], disc[v]);
        }
      }
    }
    if (progressed) continue;

    // Post-order processing.
    top--;
    const p = dfsParent[u];
    if (p !== -1) {
      if (low[u] >= disc[p] && !subIgn[u] && p !== ROOT) {
        savedArt[p] += subTown2[u];
      }
      low[p] = Math.min(low[p], low[u]);
      subTown2[p] += subTown2[u];
      subIgn[p] = subIgn[p] || subIgn[u];
    }
  }
}

computeArticulationSavings();

const baseScore = new Float64Array(R);
for (let id = 0; id < R; id++) {
  if (!cuttableCell(id)) {
    baseScore[id] = -1;
    continue;
  }
  // Articulation savings are the most reliable signal; the BFS subtree adds a
  // softer preference for branches with many towns.
  baseScore[id] = savedArt[id] * 2 + subTown[id];
}

function evaluateCutSet(cutSet) {
  const blocked = new Uint8Array(R);
  for (let i = 0; i < cutSet.length; i++) blocked[cutSet[i]] = 1;
  const seen = new Uint8Array(R);
  const q = new Int32Array(R);
  let h = 0;
  let t = 0;
  for (let i = 0; i < R; i++) {
    if (parent[i] === -1 && isIgnition[i] && !blocked[i] && !seen[i]) {
      seen[i] = 1;
      q[t++] = i;
    }
  }
  while (h < t) {
    const u = q[h++];
    for (let j = 0; j < deg[u]; j++) {
      const v = nbr[u * 4 + j];
      if (v < 0 || blocked[v] || seen[v]) continue;
      seen[v] = 1;
      q[t++] = v;
    }
  }
  let burned = 0;
  for (let i = 0; i < R; i++) {
    if (seen[i]) burned += townValue[i];
  }
  return totalTownValue - burned;
}

function selectByScore(scoreArr) {
  const ids = [];
  for (let id = 0; id < R; id++) {
    if (cuttableCell(id)) ids.push(id);
  }
  ids.sort((a, b) => {
    const da = scoreArr[a];
    const db = scoreArr[b];
    if (db !== da) return db - da;
    if (depth[a] !== depth[b]) return depth[a] - depth[b];
    return a - b;
  });

  const selected = [];
  const chosen = new Uint8Array(R);
  const bit = new Int32Array(R + 3);

  for (let i = 0; i < ids.length && selected.length < budget; i++) {
    const id = ids[i];
    if (scoreArr[id] <= 0) continue;
    if (chosen[id]) continue;
    if (fenwickSum(bit, tin[id] + 1) > 0) continue;
    chosen[id] = 1;
    selected.push(id);
    fenwickAdd(bit, tin[id] + 1, 1);
    fenwickAdd(bit, tout[id] + 2, -1);
  }
  for (let i = 0; i < ids.length && selected.length < budget; i++) {
    const id = ids[i];
    if (chosen[id]) continue;
    chosen[id] = 1;
    selected.push(id);
  }
  return selected;
}

function idsToCoords(ids) {
  const out = [];
  for (let i = 0; i < ids.length; i++) {
    const idx = idToIdx[ids[i]];
    out.push([idx % W, (idx / W) | 0]);
  }
  return out;
}

let finalSelected = selectByScore(baseScore);
let finalSaved = evaluateCutSet(finalSelected);

// For smaller reachable regions, sample a few exact min-cut solutions and use
// them to boost cells that repeatedly appear in strong separators.
const FLOW_LIMIT = 12000;
if (R <= FLOW_LIMIT) {
  const bonus = new Float64Array(R);
  const approx = Math.max(1, Math.floor(totalTownValue / Math.max(1, budget)));
  const lambdaSamples = [];
  const seenL = new Set();
  const rawSamples = [
    1,
    Math.max(1, Math.floor(approx / 4)),
    Math.max(1, Math.floor(approx / 2)),
    approx,
    Math.max(1, approx * 2),
    Math.max(1, approx * 4),
    Math.max(1, Math.floor(totalTownValue)),
  ];
  for (const v of rawSamples) {
    if (v <= 0) continue;
    const k = Math.min(v, Math.floor(totalTownValue) + 1);
    if (!seenL.has(k)) {
      seenL.add(k);
      lambdaSamples.push(k);
    }
  }
  lambdaSamples.sort((a, b) => a - b);

  class PushRelabel {
    constructor(n) {
      this.n = n;
      this.head = new Int32Array(n);
      this.head.fill(-1);
      this.to = [];
      this.cap = [];
      this.next = [];
      this.cur = new Int32Array(n);
    }
    addEdge(u, v, c) {
      const e1 = this.to.length;
      this.to.push(v);
      this.cap.push(c);
      this.next.push(this.head[u]);
      this.head[u] = e1;
      const e2 = this.to.length;
      this.to.push(u);
      this.cap.push(0);
      this.next.push(this.head[v]);
      this.head[v] = e2;
    }
    maxFlow(s, t) {
      const n = this.n;
      const height = new Int32Array(n);
      const excess = new Float64Array(n);
      const inQ = new Uint8Array(n);
      const q = [];
      let qh = 0;

      const enqueue = (v) => {
        if (v !== s && v !== t && !inQ[v] && excess[v] > 0) {
          inQ[v] = 1;
          q.push(v);
        }
      };

      height[s] = n;
      for (let e = this.head[s]; e !== -1; e = this.next[e]) {
        const v = this.to[e];
        const c = this.cap[e];
        if (c > 0) {
          this.cap[e] = 0;
          this.cap[e ^ 1] += c;
          excess[v] += c;
          excess[s] -= c;
          enqueue(v);
        }
      }

      for (let i = 0; i < n; i++) this.cur[i] = this.head[i];

      while (qh < q.length) {
        const u = q[qh++];
        inQ[u] = 0;
        if (excess[u] <= 0) continue;
        while (excess[u] > 0) {
          let e = this.cur[u];
          if (e === -1) {
            let minH = 1e9;
            for (let ee = this.head[u]; ee !== -1; ee = this.next[ee]) {
              if (this.cap[ee] > 0) minH = Math.min(minH, height[this.to[ee]]);
            }
            if (minH === 1e9) {
              height[u] = n + 1;
              break;
            }
            height[u] = minH + 1;
            this.cur[u] = this.head[u];
            continue;
          }
          const v = this.to[e];
          if (this.cap[e] > 0 && height[u] === height[v] + 1) {
            const send = Math.min(excess[u], this.cap[e]);
            this.cap[e] -= send;
            this.cap[e ^ 1] += send;
            excess[u] -= send;
            excess[v] += send;
            enqueue(v);
          } else {
            this.cur[u] = this.next[e];
          }
        }
      }

      const seen = new Uint8Array(n);
      const stack = new Int32Array(n);
      let sp2 = 0;
      stack[sp2++] = s;
      seen[s] = 1;
      while (sp2 > 0) {
        const u = stack[--sp2];
        for (let e = this.head[u]; e !== -1; e = this.next[e]) {
          const v = this.to[e];
          if (this.cap[e] > 0 && !seen[v]) {
            seen[v] = 1;
            stack[sp2++] = v;
          }
        }
      }
      return seen;
    }
  }

  function exactCutSetForLambda(lambda) {
    const nodeCount = R * 2 + 2;
    const S = R * 2;
    const T = R * 2 + 1;
    const g = new PushRelabel(nodeCount);
    const INF = totalTownValue + budget + 1e6 + 1;

    for (let id = 0; id < R; id++) {
      const inn = id * 2;
      const out = inn + 1;
      const cutCap = cuttableCell(id) ? lambda : INF;
      g.addEdge(inn, out, cutCap);
      if (isIgnition[id]) g.addEdge(S, inn, INF);
      if (townValue[id] > 0) g.addEdge(out, T, townValue[id]);
    }

    for (let id = 0; id < R; id++) {
      const out = id * 2 + 1;
      for (let j = 0; j < deg[id]; j++) {
        const v = nbr[id * 4 + j];
        if (v < 0 || v <= id) continue;
        g.addEdge(out, v * 2, INF);
        g.addEdge(v * 2 + 1, id * 2, INF);
      }
    }

    const seen = g.maxFlow(S, T);
    const cuts = [];
    for (let id = 0; id < R; id++) {
      if (!cuttableCell(id)) continue;
      const inn = id * 2;
      const out = inn + 1;
      if (seen[inn] && !seen[out]) cuts.push(id);
    }
    return cuts;
  }

  for (const lambda of lambdaSamples) {
    const cuts = exactCutSetForLambda(lambda);
    if (cuts.length === 0) continue;
    const saved = evaluateCutSet(cuts);
    if (saved <= 0) continue;
    const sorted = cuts.slice().sort((a, b) => {
      if (baseScore[b] !== baseScore[a]) return baseScore[b] - baseScore[a];
      if (depth[a] !== depth[b]) return depth[a] - depth[b];
      return a - b;
    });
    const k = Math.min(sorted.length, Math.max(20, budget * 3));
    const share = saved / k;
    for (let i = 0; i < k; i++) bonus[sorted[i]] += share;
  }

  const boostedScore = new Float64Array(R);
  for (let id = 0; id < R; id++) boostedScore[id] = baseScore[id] + bonus[id];

  const boostedSelected = selectByScore(boostedScore);
  const boostedSaved = evaluateCutSet(boostedSelected);
  if (boostedSaved > finalSaved) {
    finalSaved = boostedSaved;
    finalSelected = boostedSelected;
  }
}

if (finalSaved <= 0) {
  finalSelected = [];
}

const firebreaks = idsToCoords(finalSelected);
process.stdout.write(JSON.stringify({ firebreaks }));
