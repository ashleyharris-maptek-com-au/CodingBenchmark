OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_1 = OpConstant %uint 1

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_h = OpConstant %float 0.04

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_arr_pv_in = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_arr_pv_in Input
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_arr_v3_in = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_arr_v3_in Input
%normal = OpVariable %ptr_arr_v3_in Input
%color = OpVariable %ptr_arr_v3_in Input
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func
%entry = OpLabel

%in_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%in_pos = OpLoad %v4float %in_pos_ptr
%x = OpCompositeExtract %float %in_pos 0
%y = OpCompositeExtract %float %in_pos 1

%color_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c = OpLoad %v3float %color_ptr

%xmh = OpFSub %float %x %float_h
%ymh = OpFSub %float %y %float_h
%xph = OpFAdd %float %x %float_h
%yph = OpFAdd %float %y %float_h

%p0 = OpCompositeConstruct %v4float %xmh %ymh %float_0 %float_1
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr %p0
OpStore %outColor %c
OpEmitVertex

%p1 = OpCompositeConstruct %v4float %xph %ymh %float_0 %float_1
OpStore %out_pos_ptr %p1
OpStore %outColor %c
OpEmitVertex

%p2 = OpCompositeConstruct %v4float %xmh %yph %float_0 %float_1
OpStore %out_pos_ptr %p2
OpStore %outColor %c
OpEmitVertex

%p3 = OpCompositeConstruct %v4float %xph %yph %float_0 %float_1
OpStore %out_pos_ptr %p3
OpStore %outColor %c
OpEmitVertex

OpEndPrimitive
OpFunctionEnd
