use std::cmp::Ordering;
use std::collections::{hash_map::Entry, HashMap};
use std::fmt::Write as FmtWrite;
use std::io::{self, Read};
use std::thread;

#[derive(Clone, Copy)]
struct Item {
    value: i64,
    index: usize,
}

struct State {
    sum: i128,
    parent: usize,
    via_item: usize,
}

fn main() {
    let values = read_numbers();
    if values.len() < 2 {
        println!("NO");
        return;
    }

    let declared_raw = values[0];
    if declared_raw < 0 {
        println!("NO");
        return;
    }
    let declared = declared_raw as usize;

    let mut target = values[1];
    if target == 0 {
        println!("YES");
        println!("[]");
        return;
    }

    let available = values.len().saturating_sub(2);
    let take = declared.min(available);
    if take == 0 {
        println!("NO");
        return;
    }

    let mut items = Vec::with_capacity(take);
    for i in 0..take {
        items.push(Item {
            value: values[2 + i],
            index: i,
        });
    }

    for item in &items {
        if item.value == target {
            output_yes(vec![item.index]);
            return;
        }
    }

    let total_sum: i128 = items.iter().map(|it| it.value as i128).sum();
    if total_sum == target as i128 {
        let indices: Vec<usize> = items.iter().map(|it| it.index).collect();
        output_yes(indices);
        return;
    }

    let mut g: i128 = 0;
    for item in &items {
        g = gcd_i128(g, item.value as i128);
    }

    if g == 0 {
        println!("NO");
        return;
    }

    if g > 1 {
        if (target as i128) % g != 0 {
            println!("NO");
            return;
        }
        target = (target as i128 / g) as i64;
        let g_i64 = g as i64;
        for item in &mut items {
            item.value /= g_i64;
        }
    }

    items.retain(|item| item.value != 0);
    if items.is_empty() {
        println!("NO");
        return;
    }

    for item in &items {
        if item.value == target {
            output_yes(vec![item.index]);
            return;
        }
    }

    let scaled_sum: i128 = items.iter().map(|it| it.value as i128).sum();
    if scaled_sum == target as i128 {
        let indices: Vec<usize> = items.iter().map(|it| it.index).collect();
        output_yes(indices);
        return;
    }

    if items.len() <= 26 {
        if let Some(solution) = brute_force(&items, target) {
            output_yes(solution);
            return;
        }
        println!("NO");
        return;
    }

    if items.len() <= 46 {
        if let Some(solution) = meet_in_middle(&items, target) {
            output_yes(solution);
            return;
        }
    }

    let ordered_items = reorder_items(&items, target);
    if let Some(solution) = sparse_dp(&ordered_items, target) {
        output_yes(solution);
        return;
    }

    println!("NO");
}

fn read_numbers() -> Vec<i64> {
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();
    let mut nums = Vec::new();
    let mut idx = 0;
    while idx < input.len() {
        while idx < input.len() && input[idx].is_ascii_whitespace() {
            idx += 1;
        }
        if idx >= input.len() {
            break;
        }
        let mut sign: i128 = 1;
        if input[idx] == b'-' {
            sign = -1;
            idx += 1;
        } else if input[idx] == b'+' {
            idx += 1;
        }
        let mut value: i128 = 0;
        while idx < input.len() && input[idx].is_ascii_digit() {
            value = value * 10 + (input[idx] - b'0') as i128;
            idx += 1;
        }
        nums.push((value * sign) as i64);
    }
    nums
}

fn gcd_i128(mut a: i128, mut b: i128) -> i128 {
    if a == 0 {
        return b.abs();
    }
    if b == 0 {
        return a.abs();
    }
    a = a.abs();
    b = b.abs();
    while b != 0 {
        let tmp = a % b;
        a = b;
        b = tmp;
    }
    a
}

fn output_yes(mut indices: Vec<usize>) {
    indices.sort_unstable();
    println!("YES");
    let mut line = String::new();
    line.push('[');
    for (i, idx) in indices.iter().enumerate() {
        if i > 0 {
            line.push(' ');
        }
        let _ = write!(&mut line, "{}", idx);
    }
    line.push(']');
    println!("{line}");
}

fn brute_force(items: &[Item], target: i64) -> Option<Vec<usize>> {
    let n = items.len();
    if n == 0 {
        return None;
    }
    let total = 1usize << n;
    let target128 = target as i128;
    for mask in 0..total {
        let mut sum = 0i128;
        for (i, item) in items.iter().enumerate() {
            if (mask >> i) & 1 == 1 {
                sum += item.value as i128;
            }
        }
        if sum == target128 {
            return Some(mask_to_indices(items, mask as u64));
        }
    }
    None
}

fn enumerate_half(items: &[Item]) -> Vec<(i64, u64)> {
    let len = items.len();
    let total = 1usize << len;
    let mut result = vec![(0i64, 0u64); total.max(1)];
    for mask in 1..total {
        let prev_mask = mask & (mask - 1);
        let bit = (mask ^ prev_mask).trailing_zeros() as usize;
        let sum = result[prev_mask].0 + items[bit].value;
        result[mask] = (sum, mask as u64);
    }
    result
}

fn mask_to_indices(items: &[Item], mask: u64) -> Vec<usize> {
    let mut result = Vec::new();
    for (i, item) in items.iter().enumerate() {
        if (mask >> i) & 1 == 1 {
            result.push(item.index);
        }
    }
    result
}

fn meet_in_middle(items: &[Item], target: i64) -> Option<Vec<usize>> {
    if items.is_empty() {
        return None;
    }
    let mid = items.len() / 2;
    let (left, right) = items.split_at(mid);
    let (mut left_sums, right_sums) = thread::scope(|scope| {
        let left_handle = scope.spawn(|| enumerate_half(left));
        let right_handle = scope.spawn(|| enumerate_half(right));
        let left_res = left_handle.join().unwrap();
        let right_res = right_handle.join().unwrap();
        (left_res, right_res)
    });

    for &(sum, mask) in &left_sums {
        if sum == target {
            return Some(mask_to_indices(left, mask));
        }
    }
    for &(sum, mask) in &right_sums {
        if sum == target {
            return Some(mask_to_indices(right, mask));
        }
    }

    left_sums.sort_unstable_by(|a, b| a.0.cmp(&b.0));
    for &(sum_r, mask_r) in &right_sums {
        let needed = target - sum_r;
        if let Ok(idx) = left_sums.binary_search_by(|entry| entry.0.cmp(&needed)) {
            let left_mask = left_sums[idx].1;
            let mut indices = mask_to_indices(left, left_mask);
            let mut right_indices = mask_to_indices(right, mask_r);
            indices.append(&mut right_indices);
            return Some(indices);
        }
    }
    None
}

fn reorder_items(items: &[Item], target: i64) -> Vec<Item> {
    if items.is_empty() {
        return Vec::new();
    }
    let prefer_non_negative = target >= 0;
    let mut preferred = Vec::new();
    let mut others = Vec::new();
    for item in items.iter().copied() {
        let goes_preferred = if prefer_non_negative {
            item.value >= 0
        } else {
            item.value <= 0
        };
        if goes_preferred {
            preferred.push(item);
        } else {
            others.push(item);
        }
    }
    preferred.sort_unstable_by(|a, b| {
        let abs_a = (a.value as i128).abs();
        let abs_b = (b.value as i128).abs();
        match abs_a.cmp(&abs_b) {
            Ordering::Equal => a.index.cmp(&b.index),
            other => other,
        }
    });
    others.sort_unstable_by(|a, b| {
        let abs_a = (a.value as i128).abs();
        let abs_b = (b.value as i128).abs();
        match abs_a.cmp(&abs_b) {
            Ordering::Equal => a.index.cmp(&b.index),
            other => other,
        }
    });
    preferred.extend_from_slice(&others);
    preferred
}

fn sparse_dp(items: &[Item], target: i64) -> Option<Vec<usize>> {
    if items.is_empty() {
        return None;
    }
    let target128 = target as i128;
    let mut states = Vec::with_capacity(items.len().max(1));
    states.push(State {
        sum: 0,
        parent: 0,
        via_item: usize::MAX,
    });
    let initial_cap = items.len().saturating_mul(2).max(16);
    let mut sum_to_state: HashMap<i128, usize> = HashMap::with_capacity(initial_cap);
    sum_to_state.insert(0, 0);

    for (idx, item) in items.iter().enumerate() {
        let base_len = states.len();
        for state_idx in 0..base_len {
            let new_sum = states[state_idx].sum + item.value as i128;
            match sum_to_state.entry(new_sum) {
                Entry::Occupied(_) => continue,
                Entry::Vacant(entry) => {
                    let new_state_idx = states.len();
                    states.push(State {
                        sum: new_sum,
                        parent: state_idx,
                        via_item: idx,
                    });
                    entry.insert(new_state_idx);
                    if new_sum == target128 {
                        return Some(reconstruct(&states, new_state_idx, items));
                    }
                }
            }
        }
    }
    None
}

fn reconstruct(states: &[State], mut idx: usize, items: &[Item]) -> Vec<usize> {
    let mut result = Vec::new();
    while idx != 0 {
        let state = &states[idx];
        let item = &items[state.via_item];
        result.push(item.index);
        idx = state.parent;
    }
    result.reverse();
    result
}
