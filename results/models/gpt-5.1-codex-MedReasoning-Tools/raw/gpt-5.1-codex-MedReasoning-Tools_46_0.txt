```glsl
#version 450

layout(local_size_x = 256) in;

// Input buffers contain interleaved position/mass and velocity records.
layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    if (i >= N) {
        return;
    }

    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    uint baseIndex = i * 2u;
    vec4 posMass = inData[baseIndex];
    vec4 vel4 = inData[baseIndex + 1u];

    vec3 position = posMass.xyz;
    float mass = posMass.w;
    vec3 velocity = vel4.xyz;

    vec3 acceleration = vec3(0.0);

    for (uint j = 0u; j < N; ++j) {
        if (j == i) {
            continue;
        }

        vec4 other = inData[j * 2u];
        vec3 diff = other.xyz - position;
        float distSqr = dot(diff, diff) + softening * softening;
        float invDist = inversesqrt(distSqr);
        float invDist3 = invDist * invDist * invDist;

        float scale = G * other.w * invDist3;
        acceleration += diff * scale;
    }

    vec3 newVelocity = velocity + acceleration * dt;
    vec3 newPosition = position + newVelocity * dt;

    outData[baseIndex] = vec4(newPosition, mass);
    outData[baseIndex + 1u] = vec4(newVelocity, 0.0);
}
```
