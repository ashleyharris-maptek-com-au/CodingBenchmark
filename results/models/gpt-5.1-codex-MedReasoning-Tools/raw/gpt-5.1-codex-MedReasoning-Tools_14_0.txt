Below is a complete C# implementation that combines repeated deterministic (basic) reasoning with an adaptive constraint-satisfaction search. It scales its CSP search limits based on overall board complexity and current frontier size so that trivial boards can be exhaustively reasoned about, while extreme boards fall back to purely local deductions when necessary. The solver only reports cells that are provably safe.

```csharp
using System;
using System.Collections.Generic;
using System.Globalization;

class Program
{
    private static readonly int[] Dx = { -1, 0, 1, -1, 1, -1, 0, 1 };
    private static readonly int[] Dy = { -1, -1, -1, 0, 0, 1, 1, 1 };

    private static int H;
    private static int W;
    private static char[][] board = Array.Empty<char[]>();
    private static byte[][] cellState = Array.Empty<byte[]>();
    private static List<NumberCell> numberCells = new List<NumberCell>();
    private static List<Position> safeCells = new List<Position>();

    private const byte Unknown = 0;
    private const byte Safe = 1;
    private const byte Mine = 2;

    static void Main()
    {
        if (!ReadInput())
        {
            Console.WriteLine(0);
            return;
        }

        BoardComplexity complexity = DetermineComplexity((long)H * W);
        GetStrategyParameters(complexity, out int baseComponentLimit, out int baseSolutionLimit);

        bool progress;
        do
        {
            progress = false;
            while (RunBasicInference())
            {
                progress = true;
            }

            if (RunAdvancedInference(baseComponentLimit, baseSolutionLimit))
            {
                progress = true;
            }
        } while (progress);

        safeCells.Sort();
        Console.WriteLine(safeCells.Count);
        foreach (var pos in safeCells)
        {
            Console.WriteLine($"{pos.X} {pos.Y}");
        }
    }

    private static bool ReadInput()
    {
        string? header;
        do
        {
            header = Console.ReadLine();
            if (header == null)
            {
                return false;
            }
        } while (string.IsNullOrWhiteSpace(header));

        string[] parts = header.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 2)
        {
            return false;
        }

        if (!int.TryParse(parts[0], NumberStyles.Integer, CultureInfo.InvariantCulture, out H) ||
            !int.TryParse(parts[1], NumberStyles.Integer, CultureInfo.InvariantCulture, out W))
        {
            return false;
        }

        if (H < 0) H = 0;
        if (W < 0) W = 0;

        board = new char[H][];
        cellState = new byte[H][];
        numberCells = new List<NumberCell>();
        safeCells = new List<Position>();

        for (int y = 0; y < H; y++)
        {
            string? line = Console.ReadLine() ?? string.Empty;
            if (line.Length < W)
            {
                line = line.PadRight(W, ' ');
            }
            else if (line.Length > W)
            {
                line = line.Substring(0, W);
            }

            char[] row = line.ToCharArray();
            board[y] = row;
            cellState[y] = new byte[W];

            for (int x = 0; x < W; x++)
            {
                char c = row[x];
                if (c >= '0' && c <= '8')
                {
                    numberCells.Add(new NumberCell(x, y, c - '0'));
                }
                else if (c == '*')
                {
                    cellState[y][x] = Mine;
                }
            }
        }

        return true;
    }

    private static bool RunBasicInference()
    {
        bool changed = false;

        foreach (var number in numberCells)
        {
            Span<int> unknown = stackalloc int[8];
            int unknownCount = 0;
            int flagged = 0;

            for (int dir = 0; dir < 8; dir++)
            {
                int nx = number.X + Dx[dir];
                int ny = number.Y + Dy[dir];

                if (nx < 0 || ny < 0 || nx >= W || ny >= H)
                {
                    continue;
                }

                bool isMine = false;
                if (board[ny][nx] == '*')
                {
                    isMine = true;
                }
                else if (board[ny][nx] == ' ')
                {
                    byte state = cellState[ny][nx];
                    if (state == Mine)
                    {
                        isMine = true;
                    }
                    else if (state == Unknown)
                    {
                        unknown[unknownCount++] = ny * W + nx;
                    }
                }
                else if (cellState[ny][nx] == Mine)
                {
                    isMine = true;
                }

                if (isMine)
                {
                    flagged++;
                }
            }

            if (unknownCount == 0)
            {
                continue;
            }

            int required = number.Value - flagged;
            if (required < 0 || required > unknownCount)
            {
                continue;
            }

            if (required == 0)
            {
                for (int i = 0; i < unknownCount; i++)
                {
                    if (MarkSafe(unknown[i]))
                    {
                        changed = true;
                    }
                }
            }
            else if (required == unknownCount)
            {
                for (int i = 0; i < unknownCount; i++)
                {
                    if (MarkMine(unknown[i]))
                    {
                        changed = true;
                    }
                }
            }
        }

        return changed;
    }

    private static bool RunAdvancedInference(int baseComponentLimit, int baseSolutionLimit)
    {
        var constraints = new List<ConstraintGlobal>();
        var adjacency = new Dictionary<int, List<int>>();
        int frontierCap = ComputeFrontierCap(baseComponentLimit);

        foreach (var number in numberCells)
        {
            Span<int> unknown = stackalloc int[8];
            int unknownCount = 0;
            int flagged = 0;

            for (int dir = 0; dir < 8; dir++)
            {
                int nx = number.X + Dx[dir];
                int ny = number.Y + Dy[dir];

                if (nx < 0 || ny < 0 || nx >= W || ny >= H)
                {
                    continue;
                }

                bool isMine = false;
                if (board[ny][nx] == '*')
                {
                    isMine = true;
                }
                else if (board[ny][nx] == ' ')
                {
                    byte state = cellState[ny][nx];
                    if (state == Mine)
                    {
                        isMine = true;
                    }
                    else if (state == Unknown)
                    {
                        unknown[unknownCount++] = ny * W + nx;
                    }
                }
                else if (cellState[ny][nx] == Mine)
                {
                    isMine = true;
                }

                if (isMine)
                {
                    flagged++;
                }
            }

            if (unknownCount == 0)
            {
                continue;
            }

            int required = number.Value - flagged;
            if (required < 0 || required > unknownCount)
            {
                continue;
            }

            var vars = new List<int>(unknownCount);
            for (int i = 0; i < unknownCount; i++)
            {
                int idx = unknown[i];
                vars.Add(idx);
                if (!adjacency.ContainsKey(idx))
                {
                    adjacency[idx] = new List<int>();
                    if (adjacency.Count > frontierCap)
                    {
                        return false; // frontier too large for CSP this round
                    }
                }
            }

            constraints.Add(new ConstraintGlobal(vars, required));

            if (vars.Count > 1)
            {
                for (int i = 0; i < vars.Count; i++)
                {
                    int vi = vars[i];
                    var list = adjacency[vi];
                    for (int j = i + 1; j < vars.Count; j++)
                    {
                        int vj = vars[j];
                        list.Add(vj);
                        adjacency[vj].Add(vi);
                    }
                }
            }
        }

        if (constraints.Count == 0 || adjacency.Count == 0)
        {
            return false;
        }

        int frontierSize = adjacency.Count;
        int componentLimit = AdjustComponentLimit(baseComponentLimit, frontierSize);
        int solutionLimit = AdjustSolutionLimit(baseSolutionLimit, frontierSize);

        var components = new List<ComponentData>();
        var componentMap = new Dictionary<int, int>(frontierSize);

        foreach (var start in adjacency.Keys)
        {
            if (componentMap.ContainsKey(start))
            {
                continue;
            }

            var queue = new Queue<int>();
            queue.Enqueue(start);
            int compId = components.Count;
            var component = new ComponentData();
            componentMap[start] = compId;

            while (queue.Count > 0)
            {
                int current = queue.Dequeue();
                component.Variables.Add(current);

                if (!adjacency.TryGetValue(current, out var neighbors))
                {
                    continue;
                }

                for (int i = 0; i < neighbors.Count; i++)
                {
                    int neighbor = neighbors[i];
                    if (!componentMap.ContainsKey(neighbor))
                    {
                        componentMap[neighbor] = compId;
                        queue.Enqueue(neighbor);
                    }
                }
            }

            components.Add(component);
        }

        foreach (var constraint in constraints)
        {
            if (constraint.Variables.Count == 0)
            {
                continue;
            }

            int compId = componentMap[constraint.Variables[0]];
            components[compId].Constraints.Add(constraint);
        }

        bool changed = false;

        foreach (var component in components)
        {
            if (component.Variables.Count == 0 || component.Constraints.Count == 0)
            {
                continue;
            }

            if (component.Variables.Count > componentLimit)
            {
                continue;
            }

            int localLimit = solutionLimit + component.Variables.Count * 2048;
            if (localLimit < 1)
            {
                localLimit = 1;
            }

            var status = SolveComponent(component, localLimit);
            if (status == SolveStatus.Changed)
            {
                changed = true;
            }
        }

        return changed;
    }

    private static SolveStatus SolveComponent(ComponentData component, int solutionLimit)
    {
        int varCount = component.Variables.Count;
        int constraintCount = component.Constraints.Count;

        if (varCount == 0 || constraintCount == 0)
        {
            return SolveStatus.NoChange;
        }

        var localIndex = new Dictionary<int, int>(varCount);
        for (int i = 0; i < varCount; i++)
        {
            localIndex[component.Variables[i]] = i;
        }

        var localConstraints = new ConstraintLocal[constraintCount];
        for (int i = 0; i < constraintCount; i++)
        {
            var globalConstraint = component.Constraints[i];
            int len = globalConstraint.Variables.Count;
            var vars = new int[len];
            for (int j = 0; j < len; j++)
            {
                vars[j] = localIndex[globalConstraint.Variables[j]];
            }

            localConstraints[i] = new ConstraintLocal(vars, globalConstraint.Required);
        }

        var varConstraints = new List<int>[varCount];
        for (int i = 0; i < varCount; i++)
        {
            varConstraints[i] = new List<int>();
        }

        int[] remainingRequired = new int[constraintCount];
        int[] remainingUnassigned = new int[constraintCount];

        for (int i = 0; i < constraintCount; i++)
        {
            remainingRequired[i] = localConstraints[i].Required;
            remainingUnassigned[i] = localConstraints[i].Variables.Length;

            foreach (int var in localConstraints[i].Variables)
            {
                varConstraints[var].Add(i);
            }
        }

        int[] assignment = new int[varCount];
        int[] mineCounts = new int[varCount];
        for (int i = 0; i < varCount; i++)
        {
            assignment[i] = -1;
        }

        int[] varOrder = new int[varCount];
        for (int i = 0; i < varCount; i++)
        {
            varOrder[i] = i;
        }

        Array.Sort(varOrder, (a, b) =>
        {
            int diff = varConstraints[b].Count.CompareTo(varConstraints[a].Count);
            return diff != 0 ? diff : a.CompareTo(b);
        });

        var changeStack = new List<Change>();
        int solutionCounter = 0;
        bool aborted = false;

        void UndoChanges(int marker)
        {
            while (changeStack.Count > marker)
            {
                var change = changeStack[changeStack.Count - 1];
                changeStack.RemoveAt(changeStack.Count - 1);
                remainingRequired[change.ConstraintIndex] += change.RequiredDelta;
                remainingUnassigned[change.ConstraintIndex] += change.UnassignedDelta;
            }
        }

        bool ApplyValue(int var, int value, int marker)
        {
            var related = varConstraints[var];

            for (int i = 0; i < related.Count; i++)
            {
                int cIdx = related[i];
                remainingRequired[cIdx] -= value;
                remainingUnassigned[cIdx]--;
                changeStack.Add(new Change(cIdx, value, 1));

                if (remainingRequired[cIdx] < 0 || remainingRequired[cIdx] > remainingUnassigned[cIdx])
                {
                    UndoChanges(marker);
                    return false;
                }

                if (remainingUnassigned[cIdx] == 0 && remainingRequired[cIdx] != 0)
                {
                    UndoChanges(marker);
                    return false;
                }
            }

            assignment[var] = value;
            return true;
        }

        void ExploreValue(int var, int value, int depth)
        {
            if (aborted)
            {
                return;
            }

            int marker = changeStack.Count;
            if (!ApplyValue(var, value, marker))
            {
                return;
            }

            DepthFirstSearch(depth + 1);
            assignment[var] = -1;
            UndoChanges(marker);
        }

        void DepthFirstSearch(int depth)
        {
            if (aborted)
            {
                return;
            }

            if (depth == varCount)
            {
                for (int i = 0; i < constraintCount; i++)
                {
                    if (remainingRequired[i] != 0)
                    {
                        return;
                    }
                }

                solutionCounter++;
                if (solutionCounter >= solutionLimit)
                {
                    aborted = true;
                }

                for (int i = 0; i < varCount; i++)
                {
                    if (assignment[i] == 1)
                    {
                        mineCounts[i]++;
                    }
                }

                return;
            }

            int var = varOrder[depth];
            if (assignment[var] != -1)
            {
                DepthFirstSearch(depth + 1);
                return;
            }

            int bias = 0;
            foreach (int cIdx in varConstraints[var])
            {
                var constraint = localConstraints[cIdx];
                bias += constraint.Required * 2 - constraint.Variables.Length;
            }

            int firstValue = bias > 0 ? 1 : 0;
            ExploreValue(var, firstValue, depth);
            ExploreValue(var, 1 - firstValue, depth);
        }

        DepthFirstSearch(0);

        if (aborted || solutionCounter == 0)
        {
            return SolveStatus.NoChange;
        }

        bool changed = false;

        for (int i = 0; i < varCount; i++)
        {
            int globalIndex = component.Variables[i];

            if (mineCounts[i] == 0)
            {
                if (MarkSafe(globalIndex))
                {
                    changed = true;
                }
            }
            else if (mineCounts[i] == solutionCounter)
            {
                if (MarkMine(globalIndex))
                {
                    changed = true;
                }
            }
        }

        return changed ? SolveStatus.Changed : SolveStatus.NoChange;
    }

    private static bool MarkSafe(int linearIndex)
    {
        if (W == 0)
        {
            return false;
        }

        int y = linearIndex / W;
        int x = linearIndex % W;

        if (y < 0 || y >= H || x < 0 || x >= W)
        {
            return false;
        }

        if (cellState[y][x] != Unknown || board[y][x] != ' ')
        {
            return false;
        }

        cellState[y][x] = Safe;
        safeCells.Add(new Position(x, y));
        return true;
    }

    private static bool MarkMine(int linearIndex)
    {
        if (W == 0)
        {
            return false;
        }

        int y = linearIndex / W;
        int x = linearIndex % W;

        if (y < 0 || y >= H || x < 0 || x >= W)
        {
            return false;
        }

        if (cellState[y][x] == Mine)
        {
            return false;
        }

        cellState[y][x] = Mine;
        return true;
    }

    private static BoardComplexity DetermineComplexity(long area)
    {
        if (area <= 64) return BoardComplexity.Trivial;
        if (area <= 10_000) return BoardComplexity.Small;
        if (area <= 1_000_000) return BoardComplexity.Large;
        return BoardComplexity.Extreme;
    }

    private static void GetStrategyParameters(BoardComplexity complexity, out int componentLimit, out int solutionLimit)
    {
        switch (complexity)
        {
            case BoardComplexity.Trivial:
                componentLimit = 48;
                solutionLimit = 600_000;
                break;
            case BoardComplexity.Small:
                componentLimit = 32;
                solutionLimit = 300_000;
                break;
            case BoardComplexity.Large:
                componentLimit = 24;
                solutionLimit = 150_000;
                break;
            default:
                componentLimit = 18;
                solutionLimit = 80_000;
                break;
        }
    }

    private static int AdjustComponentLimit(int baseLimit, int frontierSize)
    {
        int limit = baseLimit;

        if (frontierSize <= 64) limit += 6;
        else if (frontierSize <= 256) limit += 4;
        else if (frontierSize <= 1024) limit += 2;
        else if (frontierSize >= 15_000) limit -= 2;
        else if (frontierSize >= 6_000) limit -= 1;

        return Clamp(limit, 12, 64);
    }

    private static int AdjustSolutionLimit(int baseLimit, int frontierSize)
    {
        int limit = baseLimit;

        if (frontierSize <= 64) limit += 60_000;
        else if (frontierSize <= 256) limit += 40_000;
        else if (frontierSize >= 6_000) limit = Math.Max(20_000, limit - 20_000);

        return limit;
    }

    private static int ComputeFrontierCap(int componentLimit)
    {
        long cap = (long)componentLimit * 1200L;
        if (cap < 4_000) cap = 4_000;
        if (cap > 200_000) cap = 200_000;
        return (int)cap;
    }

    private static int Clamp(int value, int min, int max)
    {
        if (value < min) return min;
        if (value > max) return max;
        return value;
    }

    private readonly struct NumberCell
    {
        public int X { get; }
        public int Y { get; }
        public int Value { get; }

        public NumberCell(int x, int y, int value)
        {
            X = x;
            Y = y;
            Value = value;
        }
    }

    private struct Position : IComparable<Position>
    {
        public int X;
        public int Y;

        public Position(int x, int y)
        {
            X = x;
            Y = y;
        }

        public int CompareTo(Position other)
        {
            int cmp = Y.CompareTo(other.Y);
            if (cmp != 0)
            {
                return cmp;
            }

            return X.CompareTo(other.X);
        }
    }

    private sealed class ConstraintGlobal
    {
        public List<int> Variables { get; }
        public int Required { get; }

        public ConstraintGlobal(List<int> variables, int required)
        {
            Variables = variables;
            Required = required;
        }
    }

    private sealed class ComponentData
    {
        public List<int> Variables { get; } = new List<int>();
        public List<ConstraintGlobal> Constraints { get; } = new List<ConstraintGlobal>();
    }

    private sealed class ConstraintLocal
    {
        public int[] Variables { get; }
        public int Required { get; }

        public ConstraintLocal(int[] variables, int required)
        {
            Variables = variables;
            Required = required;
        }
    }

    private struct Change
    {
        public int ConstraintIndex;
        public int RequiredDelta;
        public int UnassignedDelta;

        public Change(int constraintIndex, int requiredDelta, int unassignedDelta)
        {
            ConstraintIndex = constraintIndex;
            RequiredDelta = requiredDelta;
            UnassignedDelta = unassignedDelta;
        }
    }

    private enum SolveStatus
    {
        NoChange,
        Changed
    }

    private enum BoardComplexity
    {
        Trivial,
        Small,
        Large,
        Extreme
    }
}
```

This solver keeps iterating until no further deductions are possible, ensuring that every reported coordinate is guaranteed safe without any guessing, even on boards that scale into the tens of millions of cells.
