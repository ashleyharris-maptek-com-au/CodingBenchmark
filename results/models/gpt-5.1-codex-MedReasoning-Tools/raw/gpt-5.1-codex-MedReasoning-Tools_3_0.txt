use std::collections::VecDeque;
use std::fmt::Write;
use std::io::{self, Read};
use std::f64::consts::PI;

struct FastInput {
    data: Vec<u8>,
    pos: usize,
}
impl FastInput {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data).unwrap();
        Self { data, pos: 0 }
    }
    fn next_usize(&mut self) -> Option<usize> {
        self.skip_whitespace();
        if self.pos >= self.data.len() {
            return None;
        }
        let mut value = 0usize;
        while self.pos < self.data.len() {
            let byte = self.data[self.pos];
            if !byte.is_ascii_digit() {
                break;
            }
            value = value * 10 + (byte - b'0') as usize;
            self.pos += 1;
        }
        Some(value)
    }
    fn skip_whitespace(&mut self) {
        while self.pos < self.data.len() && self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
    }
}

fn main() {
    let mut input = FastInput::new();
    let n = match input.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m = input.next_usize().unwrap_or(0);

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u = match input.next_usize() {
            Some(v) => v,
            None => break,
        };
        let v = match input.next_usize() {
            Some(v) => v,
            None => break,
        };
        if u >= n || v >= n {
            continue;
        }
        adj[u].push(v);
        if u != v {
            adj[v].push(u);
        }
    }

    let components = build_components(&adj);
    let mut positions = vec![(0.0_f64, 0.0_f64); n];
    let mut mapping = vec![usize::MAX; n];

    let mut cursor_x = 0.0;
    let mut cursor_y = 0.0;
    let mut row_height = 0.0;
    let layout_limit = 2000.0_f64.max((n as f64).sqrt() * 120.0);
    let padding = 30.0;

    for comp in components.iter() {
        for (idx, &node) in comp.iter().enumerate() {
            mapping[node] = idx;
        }
        let local_positions = layout_component(comp, &adj, &mapping);

        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for &(x, y) in &local_positions {
            if x < min_x {
                min_x = x;
            }
            if x > max_x {
                max_x = x;
            }
            if y < min_y {
                min_y = y;
            }
            if y > max_y {
                max_y = y;
            }
        }

        if min_x == f64::INFINITY {
            min_x = 0.0;
            max_x = 0.0;
            min_y = 0.0;
            max_y = 0.0;
        }

        let mut width = max_x - min_x;
        if width < 1.0 {
            width = 1.0;
        }
        let mut height = max_y - min_y;
        if height < 1.0 {
            height = 1.0;
        }

        if cursor_x > 0.0 && cursor_x + width > layout_limit {
            cursor_x = 0.0;
            cursor_y += row_height + padding;
            row_height = 0.0;
        }

        let offset_x = cursor_x - min_x;
        let offset_y = cursor_y - min_y;

        for (idx, &node) in comp.iter().enumerate() {
            let (x, y) = local_positions[idx];
            positions[node] = (x + offset_x, y + offset_y);
        }

        cursor_x += width + padding;
        row_height = row_height.max(height);

        for &node in comp {
            mapping[node] = usize::MAX;
        }
    }

    let mut output = String::with_capacity(n * 32);
    for &(x, y) in &positions {
        let _ = writeln!(output, "{:.6} {:.6}", x, y);
    }
    print!("{output}");
}

fn build_components(adj: &[Vec<usize>]) -> Vec<Vec<usize>> {
    let n = adj.len();
    let mut visited = vec![false; n];
    let mut components = Vec::new();
    for start in 0..n {
        if visited[start] {
            continue;
        }
        let mut stack = vec![start];
        visited[start] = true;
        let mut comp = Vec::new();
        while let Some(node) = stack.pop() {
            comp.push(node);
            for &nbr in &adj[node] {
                if nbr < n && !visited[nbr] {
                    visited[nbr] = true;
                    stack.push(nbr);
                }
            }
        }
        components.push(comp);
    }
    components
}

fn layout_component(nodes: &[usize], adj: &[Vec<usize>], mapping: &[usize]) -> Vec<(f64, f64)> {
    match nodes.len() {
        0 => Vec::new(),
        1 => vec![(0.0, 0.0)],
        size if size <= 200 => fruchterman_layout(nodes, adj, mapping),
        size if size <= 5000 => {
            let (mut pos, depths) = radial_layout(nodes, adj, mapping);
            let iter_est = ((size as f64).ln() * 8.0).ceil() as usize;
            let smooth_iters = iter_est.clamp(6, 60);
            laplacian_smooth(nodes, adj, mapping, &mut pos, &depths, smooth_iters);
            pos
        }
        _ => spiral_layout(nodes, adj, mapping),
    }
}

fn fruchterman_layout(nodes: &[usize], adj: &[Vec<usize>], mapping: &[usize]) -> Vec<(f64, f64)> {
    let size = nodes.len();
    let mut positions = vec![(0.0, 0.0); size];
    if size == 0 {
        return positions;
    }
    if size == 1 {
        positions[0] = (0.0, 0.0);
        return positions;
    }

    let width = (size as f64).sqrt() * 60.0 + 20.0;
    let area = width * width;
    let k = (area / size as f64).sqrt();
    let angle_step = 2.0 * PI / size as f64;
    for (i, pos) in positions.iter_mut().enumerate() {
        let angle = i as f64 * angle_step;
        let radius = width / 3.0;
        *pos = (radius * angle.cos(), radius * angle.sin());
    }

    let max_iter = if size <= 50 { 200 } else { 120 };
    let mut disp = vec![(0.0, 0.0); size];

    for iter in 0..max_iter {
        for d in disp.iter_mut() {
            *d = (0.0, 0.0);
        }
        for i in 0..size {
            for j in (i + 1)..size {
                let dx = positions[i].0 - positions[j].0;
                let dy = positions[i].1 - positions[j].1;
                let dist = (dx * dx + dy * dy).sqrt().max(1e-9);
                let force = (k * k) / dist;
                let fx = dx / dist * force;
                let fy = dy / dist * force;
                disp[i].0 += fx;
                disp[i].1 += fy;
                disp[j].0 -= fx;
                disp[j].1 -= fy;
            }
        }
        for (i, &node) in nodes.iter().enumerate() {
            for &nbr in &adj[node] {
                let j = mapping[nbr];
                if j == usize::MAX || i <= j {
                    continue;
                }
                let dx = positions[i].0 - positions[j].0;
                let dy = positions[i].1 - positions[j].1;
                let dist = (dx * dx + dy * dy).sqrt().max(1e-9);
                let force = (dist * dist) / k;
                let fx = dx / dist * force;
                let fy = dy / dist * force;
                disp[i].0 -= fx;
                disp[i].1 -= fy;
                disp[j].0 += fx;
                disp[j].1 += fy;
            }
        }
        let temperature = width / (iter as f64 + 1.0);
        for i in 0..size {
            let dx = disp[i].0;
            let dy = disp[i].1;
            let disp_len = (dx * dx + dy * dy).sqrt();
            if disp_len > 0.0 {
                let limited = temperature.min(disp_len);
                positions[i].0 += dx / disp_len * limited;
                positions[i].1 += dy / disp_len * limited;
            }
            positions[i].0 = positions[i].0.clamp(-width, width);
            positions[i].1 = positions[i].1.clamp(-width, width);
        }
    }

    positions
}

fn radial_layout(
    nodes: &[usize],
    adj: &[Vec<usize>],
    mapping: &[usize],
) -> (Vec<(f64, f64)>, Vec<usize>) {
    let size = nodes.len();
    let mut positions = vec![(0.0, 0.0); size];
    let mut depths = vec![usize::MAX; size];
    if size == 0 {
        return (positions, depths);
    }

    let mut root_node = nodes[0];
    for &node in nodes.iter().skip(1) {
        if adj[node].len() > adj[root_node].len() {
            root_node = node;
        }
    }
    let root_local = mapping[root_node];
    let mut queue = VecDeque::new();
    queue.push_back(root_local);
    depths[root_local] = 0;
    let mut nodes_by_depth: Vec<Vec<usize>> = vec![vec![root_local]];

    while let Some(local_idx) = queue.pop_front() {
        let node = nodes[local_idx];
        let depth = depths[local_idx];
        for &nbr in &adj[node] {
            let nbr_local = mapping[nbr];
            if nbr_local == usize::MAX {
                continue;
            }
            if depths[nbr_local] == usize::MAX {
                let next_depth = depth + 1;
                if nodes_by_depth.len() <= next_depth {
                    nodes_by_depth.resize_with(next_depth + 1, Vec::new);
                }
                nodes_by_depth[next_depth].push(nbr_local);
                depths[nbr_local] = next_depth;
                queue.push_back(nbr_local);
            }
        }
    }

    for local_idx in 0..size {
        if depths[local_idx] == usize::MAX {
            let depth = nodes_by_depth.len();
            nodes_by_depth.resize_with(depth + 1, Vec::new);
            nodes_by_depth[depth].push(local_idx);
            depths[local_idx] = depth;
        }
    }

    let sum_deg: f64 = nodes.iter().map(|&node| adj[node].len() as f64).sum();
    let avg_deg = if size > 0 {
        sum_deg / size as f64
    } else {
        0.0
    };
    let base_step = 6.0 + avg_deg.sqrt() * 2.0;
    let golden_offset = 0.4;

    for depth in 0..nodes_by_depth.len() {
        let layer = &nodes_by_depth[depth];
        if layer.is_empty() {
            continue;
        }
        if depth == 0 {
            positions[layer[0]] = (0.0, 0.0);
            continue;
        }
        let radius = base_step * depth as f64;
        let count = layer.len();
        if count == 1 {
            let angle = depth as f64 * golden_offset;
            positions[layer[0]] = (radius * angle.cos(), radius * angle.sin());
            continue;
        }
        let angle_step = 2.0 * PI / count as f64;
        for (idx, &local_idx) in layer.iter().enumerate() {
            let jitter = ((local_idx * 73 % 100) as f64 / 100.0) - 0.5;
            let angle = idx as f64 * angle_step + depth as f64 * golden_offset + jitter * 0.1;
            let r = radius * (1.0 + jitter * 0.05);
            positions[local_idx] = (r * angle.cos(), r * angle.sin());
        }
    }

    (positions, depths)
}

fn laplacian_smooth(
    nodes: &[usize],
    adj: &[Vec<usize>],
    mapping: &[usize],
    positions: &mut Vec<(f64, f64)>,
    depths: &[usize],
    iterations: usize,
) {
    if iterations == 0 || nodes.is_empty() {
        return;
    }
    let mut buffer = positions.clone();
    for _ in 0..iterations {
        for (local_idx, &node) in nodes.iter().enumerate() {
            if depths.get(local_idx).map_or(false, |&d| d == 0) {
                buffer[local_idx] = positions[local_idx];
                continue;
            }
            let mut sum_x = 0.0;
            let mut sum_y = 0.0;
            let mut count = 0usize;
            for &nbr in &adj[node] {
                let nbr_local = mapping[nbr];
                if nbr_local == usize::MAX {
                    continue;
                }
                sum_x += positions[nbr_local].0;
                sum_y += positions[nbr_local].1;
                count += 1;
            }
            if count == 0 {
                buffer[local_idx] = positions[local_idx];
                continue;
            }
            let avg_x = sum_x / count as f64;
            let avg_y = sum_y / count as f64;
            let (x, y) = positions[local_idx];
            let alpha = 0.35;
            buffer[local_idx] = (
                x * (1.0 - alpha) + avg_x * alpha,
                y * (1.0 - alpha) + avg_y * alpha,
            );
        }
        std::mem::swap(positions, &mut buffer);
    }
}

fn spiral_layout(nodes: &[usize], adj: &[Vec<usize>], mapping: &[usize]) -> Vec<(f64, f64)> {
    let size = nodes.len();
    let mut positions = vec![(0.0, 0.0); size];
    if size == 0 {
        return positions;
    }

    let mut visited = vec![false; size];
    let mut order = Vec::with_capacity(size);
    let mut queue = VecDeque::new();

    let mut root_node = nodes[0];
    for &node in nodes.iter().skip(1) {
        if adj[node].len() > adj[root_node].len() {
            root_node = node;
        }
    }
    let root_local = mapping[root_node];
    queue.push_back(root_local);
    visited[root_local] = true;

    while let Some(local_idx) = queue.pop_front() {
        order.push(local_idx);
        let node = nodes[local_idx];
        for &nbr in &adj[node] {
            let nbr_local = mapping[nbr];
            if nbr_local == usize::MAX || visited[nbr_local] {
                continue;
            }
            visited[nbr_local] = true;
            queue.push_back(nbr_local);
        }
    }

    if order.len() < size {
        for idx in 0..size {
            if !visited[idx] {
                order.push(idx);
            }
        }
    }

    let sum_deg: f64 = nodes.iter().map(|&node| adj[node].len() as f64).sum();
    let avg_deg = if size > 0 {
        sum_deg / size as f64
    } else {
        0.0
    };
    let spacing = 2.0 + avg_deg.sqrt();
    let golden_angle = 2.39996322972865332;

    for (i, &local_idx) in order.iter().enumerate() {
        let radius = spacing * (i as f64).sqrt();
        let theta = golden_angle * i as f64;
        let jitter = ((local_idx * 53 % 97) as f64 / 97.0) - 0.5;
        positions[local_idx] = (
            radius * theta.cos() + jitter * 0.5,
            radius * theta.sin() + jitter * 0.5,
        );
    }

    positions
}
