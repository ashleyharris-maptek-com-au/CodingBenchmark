OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
%glsl = OpExtInstImport "GLSL.std.450"
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%uint_2 = OpConstant %uint 2
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%arr_v3float = OpTypeArray %v3float %uint_2
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Output_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Input_arr_v3float = OpTypePointer Input %arr_v3float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v3float = OpTypePointer Output %v3float
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_thickness = OpConstant %float 0.03
%float_eps = OpConstant %float 0.000001
%gl_in = OpVariable %ptr_Input_arr_pv Input
%worldPos = OpVariable %ptr_Input_arr_v3float Input
%normal = OpVariable %ptr_Input_arr_v3float Input
%color = OpVariable %ptr_Input_arr_v3float Input
%gl_out = OpVariable %ptr_Output_gl_PerVertex Output
%outColor = OpVariable %ptr_Output_v3float Output
%main = OpFunction %void None %func
%entry = OpLabel
%pos0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%pos1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %pos1_ptr
%dir = OpFSub %v4float %pos1 %pos0
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%dx2 = OpFMul %float %dir_x %dir_x
%dy2 = OpFMul %float %dir_y %dir_y
%lenSq = OpFAdd %float %dx2 %dy2
%lenSqSafe = OpExtInst %float %glsl FMax %lenSq %float_eps
%invLen = OpExtInst %float %glsl InverseSqrt %lenSqSafe
%neg_dir_y = OpFNegate %float %dir_y
%norm_x = OpFMul %float %neg_dir_y %invLen
%norm_y = OpFMul %float %dir_x %invLen
%perp_x = OpFMul %float %norm_x %float_thickness
%perp_y = OpFMul %float %norm_y %float_thickness
%perp_vec = OpCompositeConstruct %v4float %perp_x %perp_y %float_0 %float_0
%p0_plus = OpFAdd %v4float %pos0 %perp_vec
%p0_minus = OpFSub %v4float %pos0 %perp_vec
%p1_plus = OpFAdd %v4float %pos1 %perp_vec
%p1_minus = OpFSub %v4float %pos1 %perp_vec
%col0_ptr = OpAccessChain %ptr_Input_v3float %color %int_0
%col0 = OpLoad %v3float %col0_ptr
%col1_ptr = OpAccessChain %ptr_Input_v3float %color %int_1
%col1 = OpLoad %v3float %col1_ptr
%out_pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr %p0_plus
OpStore %outColor %col0
OpEmitVertex
OpStore %out_pos_ptr %p0_minus
OpStore %outColor %col0
OpEmitVertex
OpStore %out_pos_ptr %p1_plus
OpStore %outColor %col1
OpEmitVertex
OpStore %out_pos_ptr %p1_minus
OpStore %outColor %col1
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
