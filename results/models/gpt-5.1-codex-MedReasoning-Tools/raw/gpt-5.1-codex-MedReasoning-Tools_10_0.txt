```cpp
#include <algorithm>
#include <iostream>
#include <limits>
#include <numeric>
#include <utility>
#include <vector>

enum class BoardMode : unsigned char { Active, Parked, Closed };

class SkylineBoard {
public:
    struct Node { int x; int y; };
    struct Position { bool ok; int x; int y; };

    SkylineBoard() : width(0), height(0), totalArea(0), usedArea(0),
                     maxHeightUsed(0), mode(BoardMode::Closed),
                     activeSlot(-1), parkedSlot(-1) {}

    SkylineBoard(int w, int h) { init(w, h); }

    void init(int w, int h) {
        width = w;
        height = h;
        totalArea = 1LL * w * h;
        usedArea = 0;
        maxHeightUsed = 0;
        mode = BoardMode::Closed;
        activeSlot = -1;
        parkedSlot = -1;
        nodes.clear();
        nodes.reserve(128);
        nodes.push_back({0, 0});
        nodes.push_back({w, 0});
    }

    Position findPosition(int rectW, int rectH) const {
        Position best{false, 0, 0};
        if (rectW > width || rectH > height) {
            return best;
        }
        int bestY = std::numeric_limits<int>::max();
        int bestX = std::numeric_limits<int>::max();
        for (std::size_t i = 0; i + 1 < nodes.size(); ++i) {
            int xStart = nodes[i].x;
            if (xStart + rectW > width) break;
            int targetX = xStart + rectW;
            std::size_t idx = i;
            int currX = xStart;
            int maxY = 0;
            bool fits = true;
            while (currX < targetX) {
                if (idx >= nodes.size() - 1) { fits = false; break; }
                maxY = std::max(maxY, nodes[idx].y);
                int segEnd = nodes[idx + 1].x;
                int advanceLimit = std::min(segEnd, targetX);
                int advance = advanceLimit - currX;
                if (advance <= 0) { fits = false; break; }
                currX += advance;
                if (maxY + rectH > height) { fits = false; break; }
                if (currX >= segEnd) {
                    ++idx;
                }
            }
            if (!fits) continue;
            if (!best.ok || maxY < bestY || (maxY == bestY && xStart < bestX)) {
                best.ok = true;
                best.x = xStart;
                best.y = maxY;
                bestX = xStart;
                bestY = maxY;
                if (bestY == 0 && bestX == 0) break;
            }
        }
        return best;
    }

    void place(int x, int y, int rectW, int rectH) {
        int startIdx = ensureNode(x);
        int endIdx = ensureNode(x + rectW);
        nodes[startIdx].y = y + rectH;
        nodes.erase(nodes.begin() + startIdx + 1, nodes.begin() + endIdx);
        mergeNeighborsAround(startIdx);
        usedArea += 1LL * rectW * rectH;
        if (maxHeightUsed < y + rectH) {
            maxHeightUsed = y + rectH;
        }
    }

    bool canFit(int rectW, int rectH) const { return rectW <= width && rectH <= height; }
    long long freeArea() const { return totalArea - usedArea; }

    double fillScore() const {
        double heightRatio = height ? static_cast<double>(maxHeightUsed) / height : 1.0;
        double areaRatio = totalArea ? static_cast<double>(usedArea) / totalArea : 1.0;
        return 0.7 * heightRatio + 0.3 * areaRatio;
    }

    bool isPerfectlyFull() const { return usedArea == totalArea; }

    int width = 0;
    int height = 0;
    long long totalArea = 0;
    long long usedArea = 0;
    int maxHeightUsed = 0;
    BoardMode mode = BoardMode::Closed;
    int activeSlot = -1;
    int parkedSlot = -1;

private:
    std::vector<Node> nodes;

    int ensureNode(int x) {
        auto it = std::lower_bound(nodes.begin(), nodes.end(), x,
                                   [](const Node& node, int value) { return node.x < value; });
        if (it != nodes.end() && it->x == x) {
            return static_cast<int>(it - nodes.begin());
        }
        int y = 0;
        if (it != nodes.begin()) {
            y = (it - 1)->y;
        }
        it = nodes.insert(it, Node{x, y});
        return static_cast<int>(it - nodes.begin());
    }

    void mergeNeighborsAround(int idx) {
        if (idx > 0 && nodes[idx - 1].y == nodes[idx].y) {
            nodes.erase(nodes.begin() + idx);
            --idx;
        }
        if (idx + 1 < static_cast<int>(nodes.size()) && nodes[idx].y == nodes[idx + 1].y) {
            nodes.erase(nodes.begin() + idx + 1);
        }
    }
};

struct Item {
    int id = 0;
    int w = 0;
    int h = 0;
    long long area() const { return 1LL * w * h; }
};

struct PlacementOutput {
    int board = 0;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct CandidatePlacement {
    bool found = false;
    int boardIdx = -1;
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
    bool rotated = false;
};

size_t computeActiveLimit(size_t total, size_t processed) {
    const size_t INF = std::numeric_limits<size_t>::max();
    if (total <= 4000) {
        return INF;
    }
    double progress = total ? static_cast<double>(processed) / static_cast<double>(total) : 0.0;
    if (total <= 10000) {
        if (progress < 0.6) return 768;
        return INF;
    }
    if (total <= 20000) {
        if (progress < 0.45) return 640;
        if (progress < 0.8) return 1024;
        return INF;
    }
    if (total <= 40000) {
        if (progress < 0.35) return 512;
        if (progress < 0.7) return 768;
        return 1024;
    }
    if (progress < 0.35) return 256;
    if (progress < 0.7) return 384;
    return 640;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N, boardW, boardH;
    if (!(std::cin >> N >> boardW >> boardH)) {
        return 0;
    }

    std::vector<Item> items(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> items[i].w >> items[i].h;
        items[i].id = i;
    }

    std::vector<int> order(N);
    std::iota(order.begin(), order.end(), 0);
    std::sort(order.begin(), order.end(), [&](int lhs, int rhs) {
        const Item& a = items[lhs];
        const Item& b = items[rhs];
        int longestA = std::max(a.w, a.h);
        int longestB = std::max(b.w, b.h);
        if (longestA != longestB) return longestA > longestB;
        long long areaA = a.area();
        long long areaB = b.area();
        if (areaA != areaB) return areaA > areaB;
        int shortestA = std::min(a.w, a.h);
        int shortestB = std::min(b.w, b.h);
        if (shortestA != shortestB) return shortestA > shortestB;
        return a.id < b.id;
    });

    std::vector<PlacementOutput> placements(N);

    std::vector<SkylineBoard> boards;
    boards.reserve(std::max(1, N / 6 + 1));

    std::vector<int> activeBoards;
    std::vector<int> parkedBoards;
    activeBoards.reserve(512);
    parkedBoards.reserve(512);
    std::size_t parkedProbeCursor = 0;

    auto normalizeParkedCursor = [&]() {
        if (parkedBoards.empty()) {
            parkedProbeCursor = 0;
        } else if (parkedProbeCursor >= parkedBoards.size()) {
            parkedProbeCursor %= parkedBoards.size();
        }
    };

    auto detachActive = [&](int boardIdx) {
        SkylineBoard& b = boards[boardIdx];
        int pos = b.activeSlot;
        if (pos < 0) return;
        int lastIdx = activeBoards.back();
        activeBoards[pos] = lastIdx;
        boards[lastIdx].activeSlot = pos;
        activeBoards.pop_back();
        b.activeSlot = -1;
    };

    auto detachParked = [&](int boardIdx) {
        SkylineBoard& b = boards[boardIdx];
        int pos = b.parkedSlot;
        if (pos < 0) return;
        int lastIdx = parkedBoards.back();
        parkedBoards[pos] = lastIdx;
        boards[lastIdx].parkedSlot = pos;
        parkedBoards.pop_back();
        b.parkedSlot = -1;
        normalizeParkedCursor();
    };

    auto parkBoard = [&](int boardIdx) {
        SkylineBoard& b = boards[boardIdx];
        if (b.mode == BoardMode::Closed || b.mode == BoardMode::Parked) return;
        detachActive(boardIdx);
        b.mode = BoardMode::Parked;
        b.parkedSlot = static_cast<int>(parkedBoards.size());
        parkedBoards.push_back(boardIdx);
        normalizeParkedCursor();
    };

    auto activateBoard = [&](int boardIdx) {
        SkylineBoard& b = boards[boardIdx];
        if (b.mode == BoardMode::Closed || b.mode == BoardMode::Active) return;
        if (b.mode == BoardMode::Parked) {
            detachParked(boardIdx);
        }
        b.mode = BoardMode::Active;
        b.activeSlot = static_cast<int>(activeBoards.size());
        activeBoards.push_back(boardIdx);
    };

    auto closeBoard = [&](int boardIdx) {
        SkylineBoard& b = boards[boardIdx];
        if (b.mode == BoardMode::Active) {
            detachActive(boardIdx);
        } else if (b.mode == BoardMode::Parked) {
            detachParked(boardIdx);
        }
        b.mode = BoardMode::Closed;
    };

    const std::size_t NO_LIMIT = std::numeric_limits<std::size_t>::max();

    auto enforceLimit = [&](std::size_t limit) {
        if (limit == NO_LIMIT) return;
        while (activeBoards.size() > limit) {
            std::size_t removePos = 0;
            double worstScore = -1.0;
            for (std::size_t idx = 0; idx < activeBoards.size(); ++idx) {
                double score = boards[activeBoards[idx]].fillScore();
                if (score > worstScore) {
                    worstScore = score;
                    removePos = idx;
                }
            }
            int boardIdx = activeBoards[removePos];
            parkBoard(boardIdx);
        }
        while (activeBoards.size() < limit && !parkedBoards.empty()) {
            std::size_t bestPos = 0;
            double bestScore = std::numeric_limits<double>::max();
            for (std::size_t idx = 0; idx < parkedBoards.size(); ++idx) {
                double score = boards[parkedBoards[idx]].fillScore();
                if (score < bestScore) {
                    bestScore = score;
                    bestPos = idx;
                }
            }
            int boardIdx = parkedBoards[bestPos];
            activateBoard(boardIdx);
        }
    };

    auto considerCandidate = [&](int boardIdx, const SkylineBoard::Position& pos,
                                 int rw, int rh, bool rotated, CandidatePlacement& cand) {
        if (!pos.ok) return;
        if (!cand.found) {
            cand.found = true;
            cand.boardIdx = boardIdx;
            cand.x = pos.x;
            cand.y = pos.y;
            cand.w = rw;
            cand.h = rh;
            cand.rotated = rotated;
            return;
        }
        if (pos.y < cand.y) {
            cand.boardIdx = boardIdx;
            cand.x = pos.x;
            cand.y = pos.y;
            cand.w = rw;
            cand.h = rh;
            cand.rotated = rotated;
            return;
        }
        if (pos.y == cand.y) {
            int heightAfter = pos.y + rh;
            int bestHeightAfter = cand.y + cand.h;
            if (heightAfter < bestHeightAfter ||
                (heightAfter == bestHeightAfter &&
                 (pos.x < cand.x || (pos.x == cand.x && boardIdx < cand.boardIdx)))) {
                cand.boardIdx = boardIdx;
                cand.x = pos.x;
                cand.y = pos.y;
                cand.w = rw;
                cand.h = rh;
                cand.rotated = rotated;
            }
        }
    };

    auto tryBoard = [&](int boardIdx, const Item& item, CandidatePlacement& cand) {
        SkylineBoard& board = boards[boardIdx];
        if (board.freeArea() < item.area()) return;
        auto attempt = [&](int rw, int rh, bool rotated) {
            if (!board.canFit(rw, rh)) return;
            auto pos = board.findPosition(rw, rh);
            considerCandidate(boardIdx, pos, rw, rh, rotated, cand);
        };
        attempt(item.w, item.h, false);
        if (item.w != item.h) {
            attempt(item.h, item.w, true);
        }
    };

    auto probeParkedBoards = [&](const Item& item, CandidatePlacement& cand, std::size_t maxProbe) {
        if (parkedBoards.empty() || maxProbe == 0) return;
        normalizeParkedCursor();
        std::size_t cursor = parkedProbeCursor;
        std::size_t attempts = 0;
        std::size_t total = parkedBoards.size();
        if (maxProbe > total) maxProbe = total;
        while (attempts < maxProbe && !parkedBoards.empty()) {
            if (cursor >= parkedBoards.size()) cursor = 0;
            int boardIdx = parkedBoards[cursor];
            ++cursor;
            ++attempts;
            tryBoard(boardIdx, item, cand);
        }
        if (!parkedBoards.empty()) {
            parkedProbeCursor = cursor % parkedBoards.size();
        } else {
            parkedProbeCursor = 0;
        }
    };

    std::size_t totalRects = static_cast<std::size_t>(N);
    std::size_t activeLimit = computeActiveLimit(totalRects, 0);
    std::size_t processed = 0;

    for (int sortedPos = 0; sortedPos < N; ++sortedPos) {
        int idx = order[sortedPos];
        Item& item = items[idx];

        std::size_t targetLimit = computeActiveLimit(totalRects, processed);
        if (targetLimit != activeLimit) {
            activeLimit = targetLimit;
            enforceLimit(activeLimit);
        }

        CandidatePlacement best;

        for (int boardIdx : activeBoards) {
            tryBoard(boardIdx, item, best);
        }

        if (!best.found) {
            std::size_t probeBudget = (totalRects > 30000) ? 48u : 96u;
            probeParkedBoards(item, best, probeBudget);
        }

        if (!best.found) {
            int boardIdx = static_cast<int>(boards.size());
            boards.emplace_back(boardW, boardH);
            SkylineBoard& newBoard = boards.back();
            newBoard.mode = BoardMode::Active;
            newBoard.activeSlot = static_cast<int>(activeBoards.size());
            newBoard.parkedSlot = -1;
            activeBoards.push_back(boardIdx);
            tryBoard(boardIdx, item, best);
            if (!best.found) {
                // Input is assumed to be feasible; this is a final safeguard.
                best.found = true;
                best.boardIdx = boardIdx;
                best.x = 0;
                best.y = 0;
                int rw = std::min(item.w, boardW);
                int rh = std::min(item.h, boardH);
                bool rotate = false;
                if (item.w > boardW && item.h <= boardW && item.w <= boardH) {
                    std::swap(rw, rh);
                    rotate = true;
                }
                best.w = rw;
                best.h = rh;
                best.rotated = rotate;
            }
        }

        SkylineBoard& chosenBoard = boards[best.boardIdx];
        if (chosenBoard.mode == BoardMode::Parked) {
            activateBoard(best.boardIdx);
        }
        chosenBoard.place(best.x, best.y, best.w, best.h);
        placements[item.id] = {best.boardIdx, best.x, best.y, best.rotated ? 1 : 0};

        if (chosenBoard.isPerfectlyFull()) {
            closeBoard(best.boardIdx);
        }

        enforceLimit(activeLimit);
        ++processed;
    }

    std::cout << boards.size() << '\n';
    for (int i = 0; i < N; ++i) {
        const auto& p = placements[i];
        std::cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << '\n';
    }

    return 0;
}
```
