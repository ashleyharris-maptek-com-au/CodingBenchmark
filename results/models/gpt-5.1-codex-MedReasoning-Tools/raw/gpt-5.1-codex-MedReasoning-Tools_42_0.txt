OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %arr_vec3 ArrayStride 12
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void        = OpTypeVoid
%func        = OpTypeFunction %void
%float       = OpTypeFloat 32
%v3float     = OpTypeVector %float 3
%v4float     = OpTypeVector %float 4
%int         = OpTypeInt 32 1
%int_0       = OpConstant %int 0
%uint        = OpTypeInt 32 0
%uint_1      = OpConstant %uint 1
%float_0     = OpConstant %float 0
%float_1     = OpConstant %float 1
%float_0_04  = OpConstant %float 0.04

%gl_PerVertex_in  = OpTypeStruct %v4float
%arr_pv_in        = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in            = OpVariable %ptr_Input_arr_pv Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out           = OpVariable %ptr_Output_gl_pv Output
%ptr_Input_vec4   = OpTypePointer Input %v4float
%ptr_Output_vec4  = OpTypePointer Output %v4float

%arr_vec3         = OpTypeArray %v3float %uint_1
%ptr_Input_arr_v3 = OpTypePointer Input %arr_vec3
%worldPos         = OpVariable %ptr_Input_arr_v3 Input
%normal           = OpVariable %ptr_Input_arr_v3 Input
%color            = OpVariable %ptr_Input_arr_v3 Input
%ptr_Input_vec3   = OpTypePointer Input %v3float
%ptr_Output_vec3  = OpTypePointer Output %v3float
%outColor         = OpVariable %ptr_Output_vec3 Output

%main = OpFunction %void None %func
%entry = OpLabel
%pos_ptr = OpAccessChain %ptr_Input_vec4 %gl_in %int_0 %int_0
%center = OpLoad %v4float %pos_ptr
%cx = OpCompositeExtract %float %center 0
%cy = OpCompositeExtract %float %center 1
%col_ptr = OpAccessChain %ptr_Input_vec3 %color %int_0
%col = OpLoad %v3float %col_ptr
%xm = OpFSub %float %cx %float_0_04
%xp = OpFAdd %float %cx %float_0_04
%ym = OpFSub %float %cy %float_0_04
%yp = OpFAdd %float %cy %float_0_04
%out_pos_ptr = OpAccessChain %ptr_Output_vec4 %gl_out %int_0

%v1 = OpCompositeConstruct %v4float %xm %ym %float_0 %float_1
OpStore %out_pos_ptr %v1
OpStore %outColor %col
OpEmitVertex

%v2 = OpCompositeConstruct %v4float %xp %ym %float_0 %float_1
OpStore %out_pos_ptr %v2
OpStore %outColor %col
OpEmitVertex

%v3 = OpCompositeConstruct %v4float %xm %yp %float_0 %float_1
OpStore %out_pos_ptr %v3
OpStore %outColor %col
OpEmitVertex

%v4 = OpCompositeConstruct %v4float %xp %yp %float_0 %float_1
OpStore %out_pos_ptr %v4
OpStore %outColor %col
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
