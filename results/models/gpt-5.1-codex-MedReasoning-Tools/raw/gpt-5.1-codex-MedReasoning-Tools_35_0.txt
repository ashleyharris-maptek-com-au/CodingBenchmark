```rust
use std::io::{self, Read};
use std::time::{Duration, Instant};

struct Graph {
    adj: Vec<Vec<usize>>,
    edges: usize,
}

fn main() {
    let graph = read_graph();
    let n = graph.adj.len();

    if n == 0 {
        println!("0");
        println!();
        return;
    }

    let start = Instant::now();
    let time_limit = Duration::from_millis(295_000);

    let degrees: Vec<usize> = graph.adj.iter().map(|neigh| neigh.len()).collect();

    let mut order_high: Vec<usize> = (0..n).collect();
    order_high.sort_unstable_by(|&a, &b| degrees[b].cmp(&degrees[a]).then_with(|| a.cmp(&b)));

    let mut order_low: Vec<usize> = (0..n).collect();
    order_low.sort_unstable_by(|&a, &b| degrees[a].cmp(&degrees[b]).then_with(|| a.cmp(&b)));

    let mut order_rand: Vec<usize> = (0..n).collect();
    let seed = (n as u64)
        ^ ((graph.edges as u64).wrapping_mul(0x9E3779B97F4A7C15))
        ^ 0xBF58476D1CE4E5B9;
    shuffle(&mut order_rand, seed);

    let orders = vec![order_high, order_low, order_rand];

    let mut best_solution: Vec<usize> = (0..n).collect();
    let mut best_size = best_solution.len();

    for order in &orders {
        if start.elapsed() > time_limit {
            break;
        }
        let flags = construct(&graph, order);
        let sol = flags_to_vertices(&flags);
        if sol.len() < best_size {
            best_size = sol.len();
            best_solution = sol;
        }
    }

    println!("{}", best_size);
    if best_solution.is_empty() {
        println!();
    } else {
        for (idx, v) in best_solution.iter().enumerate() {
            if idx > 0 {
                print!(" ");
            }
            print!("{}", v);
        }
        println!();
    }
}

fn read_graph() -> Graph {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut iter = input.split_whitespace();

    let n = iter.next().and_then(|t| t.parse::<usize>().ok()).unwrap_or(0);
    let m = iter.next().and_then(|t| t.parse::<usize>().ok()).unwrap_or(0);

    let mut adj = vec![Vec::new(); n];

    for _ in 0..m {
        let u = match iter.next().and_then(|t| t.parse::<usize>().ok()) {
            Some(val) => val,
            None => break,
        };
        let v = match iter.next().and_then(|t| t.parse::<usize>().ok()) {
            Some(val) => val,
            None => break,
        };
        if u >= n || v >= n || u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    for neighbors in &mut adj {
        neighbors.sort_unstable();
        neighbors.dedup();
    }

    Graph { adj, edges: m }
}

fn construct(graph: &Graph, order: &[usize]) -> Vec<bool> {
    let n = graph.adj.len();
    let mut coverage = vec![0u32; n];
    let mut in_solution = vec![false; n];
    let mut undominated = n;

    for &v in order {
        if undominated == 0 {
            break;
        }
        if coverage[v] == 0 {
            let best = choose_best_candidate(v, &coverage, graph);
            let newly = add_vertex(best, &mut in_solution, &mut coverage, &graph.adj);
            undominated = undominated.saturating_sub(newly);
        }
    }

    while undominated > 0 {
        let mut target = None;
        for (idx, &cov) in coverage.iter().enumerate() {
            if cov == 0 {
                target = Some(idx);
                break;
            }
        }
        let v = match target {
            Some(val) => val,
            None => break,
        };
        let best = choose_best_candidate(v, &coverage, graph);
        let newly = add_vertex(best, &mut in_solution, &mut coverage, &graph.adj);
        undominated = undominated.saturating_sub(newly);
    }

    prune(&mut in_solution, &mut coverage, &graph.adj);

    in_solution
}

fn choose_best_candidate(v: usize, coverage: &[u32], graph: &Graph) -> usize {
    let adj = &graph.adj;
    let mut best = v;
    let mut best_gain = compute_gain(v, coverage, adj);
    for &nb in &adj[v] {
        let gain = compute_gain(nb, coverage, adj);
        if gain > best_gain || (gain == best_gain && better_candidate(nb, best, adj)) {
            best = nb;
            best_gain = gain;
        }
    }
    best
}

fn compute_gain(v: usize, coverage: &[u32], adj: &[Vec<usize>]) -> usize {
    let mut gain = if coverage[v] == 0 { 1 } else { 0 };
    for &nb in &adj[v] {
        if coverage[nb] == 0 {
            gain += 1;
        }
    }
    gain
}

fn better_candidate(a: usize, b: usize, adj: &[Vec<usize>]) -> bool {
    let da = adj[a].len();
    let db = adj[b].len();
    if da != db {
        da > db
    } else {
        a < b
    }
}

fn add_vertex(
    v: usize,
    in_solution: &mut [bool],
    coverage: &mut [u32],
    adj: &[Vec<usize>],
) -> usize {
    if in_solution[v] {
        return 0;
    }
    in_solution[v] = true;
    let mut newly = 0;
    if coverage[v] == 0 {
        newly += 1;
    }
    coverage[v] += 1;
    for &nb in &adj[v] {
        if coverage[nb] == 0 {
            newly += 1;
        }
        coverage[nb] += 1;
    }
    newly
}

fn prune(in_solution: &mut [bool], coverage: &mut [u32], adj: &[Vec<usize>]) {
    let n = in_solution.len();
    if n == 0 {
        return;
    }
    let mut changed = true;
    let mut candidates = Vec::new();
    let mut scratch = Vec::new();

    while changed {
        changed = false;
        candidates.clear();
        for v in 0..n {
            if in_solution[v] {
                candidates.push(v);
            }
        }
        if candidates.is_empty() {
            break;
        }
        candidates.sort_unstable_by(|&a, &b| adj[a].len().cmp(&adj[b].len()).then(a.cmp(&b)));
        for &node in &candidates {
            if !in_solution[node] {
                continue;
            }
            scratch.clear();
            scratch.push(node);
            in_solution[node] = false;
            coverage[node] -= 1;
            for &nb in &adj[node] {
                coverage[nb] -= 1;
                scratch.push(nb);
            }
            let mut valid = true;
            for &u in &scratch {
                if coverage[u] == 0 {
                    valid = false;
                    break;
                }
            }
            if !valid {
                let _ = add_vertex(node, in_solution, coverage, adj);
            } else {
                changed = true;
            }
        }
    }
}

fn flags_to_vertices(flags: &[bool]) -> Vec<usize> {
    let mut res = Vec::new();
    for (idx, &flag) in flags.iter().enumerate() {
        if flag {
            res.push(idx);
        }
    }
    res
}

fn shuffle(order: &mut [usize], mut state: u64) {
    if order.is_empty() {
        return;
    }
    for i in (1..order.len()).rev() {
        state = state.wrapping_mul(6364136223846793005).wrapping_add(1);
        let j = (state as usize) % (i + 1);
        order.swap(i, j);
    }
}
```
