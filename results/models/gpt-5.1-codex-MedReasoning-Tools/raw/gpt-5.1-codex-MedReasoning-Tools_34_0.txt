Here's a complete Python program that reads a CVRP instance from stdin, uses a Clarke–Wright savings heuristic combined with capacity-aware post-merging and route-level 2-opt refinement, and prints feasible routes that respect the capacity and “up to K vehicles” constraints.

```python
import sys
import math


def read_input():
    data = sys.stdin.read().strip().split()
    if not data:
        return None
    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    capacity = float(next(it))
    depot = (float(next(it)), float(next(it)))
    customers = []
    demands = []
    for _ in range(n):
        x = float(next(it))
        y = float(next(it))
        d = float(next(it))
        customers.append((x, y))
        demands.append(d)
    return n, k, capacity, depot, customers, demands


def build_distance_matrix(coords):
    size = len(coords)
    dist = [[0.0] * size for _ in range(size)]
    for i in range(size):
        xi, yi = coords[i]
        for j in range(i + 1, size):
            xj, yj = coords[j]
            d = math.hypot(xi - xj, yi - yj)
            dist[i][j] = dist[j][i] = d
    return dist


def route_distance(nodes, dist):
    if not nodes:
        return 0.0
    total = dist[0][nodes[0]]
    for a, b in zip(nodes, nodes[1:]):
        total += dist[a][b]
    total += dist[nodes[-1]][0]
    return total


def two_opt(nodes, dist):
    if len(nodes) < 3:
        return nodes
    improved = True
    while improved:
        improved = False
        for i in range(len(nodes) - 1):
            for j in range(i + 2, len(nodes)):
                if j - i == 1:
                    continue
                a_prev = 0 if i == 0 else nodes[i - 1]
                a = nodes[i]
                b = nodes[j]
                b_next = 0 if j == len(nodes) - 1 else nodes[j + 1]
                current = dist[a_prev][a] + dist[b][b_next]
                proposed = dist[a_prev][b] + dist[a][b_next]
                if proposed + 1e-9 < current:
                    nodes[i:j + 1] = reversed(nodes[i:j + 1])
                    improved = True
    return nodes


def main():
    parsed = read_input()
    if not parsed:
        return
    n, k, capacity, depot, customers, demands = parsed
    coords = [depot] + customers
    dist = build_distance_matrix(coords)

    if n == 0:
        print("0.0")
        print(0)
        return

    routes = []
    route_of = {}
    for idx in range(1, n + 1):
        demand = demands[idx - 1]
        route = {'nodes': [idx], 'load': demand}
        routes.append(route)
        route_of[idx] = route

    savings = []
    for i in range(1, n + 1):
        for j in range(i + 1, n + 1):
            s = dist[0][i] + dist[0][j] - dist[i][j]
            savings.append((s, i, j))
    savings.sort(reverse=True, key=lambda x: x[0])

    for _, i, j in savings:
        route_i = route_of[i]
        route_j = route_of[j]
        if route_i is route_j:
            continue
        if route_i['load'] + route_j['load'] > capacity + 1e-9:
            continue
        nodes_i = route_i['nodes']
        nodes_j = route_j['nodes']
        if not (nodes_i[0] == i or nodes_i[-1] == i):
            continue
        if not (nodes_j[0] == j or nodes_j[-1] == j):
            continue
        seq_i = nodes_i if nodes_i[-1] == i else list(reversed(nodes_i))
        seq_j = nodes_j if nodes_j[0] == j else list(reversed(nodes_j))
        new_nodes = seq_i + seq_j
        new_route = {'nodes': new_nodes, 'load': route_i['load'] + route_j['load']}
        routes.remove(route_i)
        routes.remove(route_j)
        routes.append(new_route)
        for node in new_nodes:
            route_of[node] = new_route

    while len(routes) > k:
        best_pair = None
        best_seq = None
        best_delta = float('inf')
        for idx_a in range(len(routes)):
            for idx_b in range(idx_a + 1, len(routes)):
                route_a = routes[idx_a]
                route_b = routes[idx_b]
                if route_a['load'] + route_b['load'] > capacity + 1e-9:
                    continue
                base = route_distance(route_a['nodes'], dist) + route_distance(route_b['nodes'], dist)
                for seq_a in (route_a['nodes'], list(reversed(route_a['nodes']))):
                    for seq_b in (route_b['nodes'], list(reversed(route_b['nodes']))):
                        new_seq = seq_a + seq_b
                        new_len = route_distance(new_seq, dist)
                        delta = new_len - base
                        if delta < best_delta:
                            best_delta = delta
                            best_pair = (route_a, route_b)
                            best_seq = new_seq
        if best_pair is None:
            break
        route_a, route_b = best_pair
        new_route = {'nodes': best_seq, 'load': route_a['load'] + route_b['load']}
        routes.remove(route_a)
        routes.remove(route_b)
        routes.append(new_route)
        for node in new_route['nodes']:
            route_of[node] = new_route

    for route in routes:
        route['nodes'] = two_opt(route['nodes'], dist)

    total_distance = sum(route_distance(route['nodes'], dist) for route in routes)

    print(f"{total_distance:.6f}")
    print(len(routes))
    for route in routes:
        customer_indices = [str(node - 1) for node in route['nodes']]
        print(" ".join(customer_indices))


if __name__ == "__main__":
    main()
```

**How it works (high level):**
1. Parses the instance from `stdin`.
2. Precomputes all Euclidean distances (including depot).
3. Initializes one single-customer route per customer.
4. Applies the Clarke–Wright savings heuristic to merge routes while respecting capacity.
5. If more than `K` routes remain, repeatedly merges the most distance-efficient pairs that keep loads feasible.
6. Refines each route individually with a depot-aware 2-opt local search.
7. Outputs the total distance, number of routes, and each route’s customer visit order (0-based indices).
