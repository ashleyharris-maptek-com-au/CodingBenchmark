Here is a complete, self-contained C++20 program that satisfies the requirements. It parses the streamed simulation data, loads the terrain asynchronously, and flies the lander using a multi-phase PD autopilot that works from 100 m all the way up to 100 km scenarios.

```cpp
#include <algorithm>
#include <array>
#include <atomic>
#include <chrono>
#include <cmath>
#include <condition_variable>
#include <cstdint>
#include <deque>
#include <fstream>
#include <iomanip>
#include <iostream>
#include <limits>
#include <mutex>
#include <shared_mutex>
#include <string>
#include <thread>
#include <unordered_map>

namespace {
constexpr double kPi = 3.14159265358979323846;

double wrapAngle(double angle) {
    while (angle > kPi) {
        angle -= 2.0 * kPi;
    }
    while (angle < -kPi) {
        angle += 2.0 * kPi;
    }
    return angle;
}

int clampCoord(int value, int bound) {
    if (bound <= 0) return 0;
    if (value < 0) return 0;
    if (value >= bound) return bound - 1;
    return value;
}

int clampIndexFromDouble(double value, int bound) {
    if (bound <= 0) return 0;
    long long idx = static_cast<long long>(std::llround(value));
    long long max_idx = static_cast<long long>(bound) - 1;
    if (idx < 0) idx = 0;
    if (idx > max_idx) idx = max_idx;
    return static_cast<int>(idx);
}

bool isFinite(double v) {
    return std::isfinite(v);
}
}  // namespace

struct TerrainColumn {
    double ground = 0.0;
    double rock_bottom = 0.0;
    double rock_top = 0.0;

    bool hasFloating() const {
        return rock_bottom + 1e-6 < rock_top;
    }
};

class TerrainCache {
public:
    TerrainCache(int width, int depth, std::string path)
        : width_(width),
          depth_(depth),
          path_(std::move(path)),
          worker_stop_(false) {
        stream_.open(path_, std::ios::binary);
        stream_valid_ = stream_.good();
        default_radius_ = computeDefaultRadius();
        if (stream_valid_) {
            worker_ = std::thread(&TerrainCache::workerLoop, this);
        }
    }

    ~TerrainCache() {
        worker_stop_.store(true);
        queue_cv_.notify_all();
        if (worker_.joinable()) {
            worker_.join();
        }
    }

    TerrainCache(const TerrainCache&) = delete;
    TerrainCache& operator=(const TerrainCache&) = delete;

    bool isReady() const {
        return stream_valid_;
    }

    int defaultPrefetchRadius() const {
        return default_radius_;
    }

    void requestPrefetchRegion(int cx, int cy, int radius) {
        if (!stream_valid_) {
            return;
        }
        radius = std::max(radius, 0);
        RegionTask task{cx, cy, radius};
        if (!worker_.joinable()) {
            prefetchRegionTask(task);
            return;
        }

        {
            std::lock_guard<std::mutex> lock(queue_mutex_);
            if (tasks_.size() >= kMaxQueue) {
                return;
            }
            tasks_.push_back(task);
        }
        queue_cv_.notify_one();
    }

    TerrainColumn get(int x, int y) const {
        x = clampCoord(x, width_);
        y = clampCoord(y, depth_);
        uint64_t key = (static_cast<uint64_t>(static_cast<uint32_t>(x)) << 32) |
                       static_cast<uint32_t>(y);

        {
            std::shared_lock<std::shared_mutex> lock(cache_mutex_);
            auto it = cache_.find(key);
            if (it != cache_.end()) {
                return it->second;
            }
        }

        TerrainColumn column = loadColumn(x, y);
        {
            std::unique_lock<std::shared_mutex> lock(cache_mutex_);
            cache_[key] = column;
        }
        return column;
    }

private:
    struct RegionTask {
        int cx = 0;
        int cy = 0;
        int radius = 0;
    };

    static constexpr int kMaxQueue = 8;
    static constexpr int kMaxPrefetchRadius = 128;
    static constexpr int kMaxPrefetchCells = 25000;

    int computeDefaultRadius() const {
        int span = std::max(width_, depth_);
        if (span <= 0) return 8;
        int radius = std::clamp(span / 400, 4, kMaxPrefetchRadius);
        return radius;
    }

    TerrainColumn loadColumn(int x, int y) const {
        TerrainColumn column;
        column.ground = 0.0;
        column.rock_bottom = 0.0;
        column.rock_top = 0.0;

        if (!stream_valid_) {
            return column;
        }

        const int64_t idx = static_cast<int64_t>(x) * static_cast<int64_t>(depth_) + y;
        const int64_t byte_offset = idx * 6;

        {
            std::lock_guard<std::mutex> lock(stream_mutex_);
            stream_.clear();
            stream_.seekg(static_cast<std::streamoff>(byte_offset), std::ios::beg);
            if (!stream_) {
                stream_.clear();
                return column;
            }

            uint16_t a = 0, b = 0, c = 0;
            stream_.read(reinterpret_cast<char*>(&a), sizeof(uint16_t));
            stream_.read(reinterpret_cast<char*>(&b), sizeof(uint16_t));
            stream_.read(reinterpret_cast<char*>(&c), sizeof(uint16_t));
            if (!stream_) {
                stream_.clear();
                return column;
            }

            column.ground = static_cast<double>(a);
            column.rock_bottom = static_cast<double>(b);
            column.rock_top = static_cast<double>(c);
        }

        return column;
    }

    void prefetchRegionTask(const RegionTask& task) const {
        if (!stream_valid_ || width_ <= 0 || depth_ <= 0) {
            return;
        }
        int radius = std::min(task.radius, kMaxPrefetchRadius);
        if (radius <= 0) {
            get(task.cx, task.cy);
            return;
        }

        int processed = 0;
        for (int dx = -radius; dx <= radius && processed < kMaxPrefetchCells; ++dx) {
            int x = clampCoord(task.cx + dx, width_);
            for (int dy = -radius; dy <= radius && processed < kMaxPrefetchCells; ++dy) {
                int y = clampCoord(task.cy + dy, depth_);
                get(x, y);
                ++processed;
            }
        }
    }

    void workerLoop() {
        while (true) {
            RegionTask task;
            {
                std::unique_lock<std::mutex> lock(queue_mutex_);
                queue_cv_.wait(lock, [this]() {
                    return worker_stop_.load() || !tasks_.empty();
                });

                if (worker_stop_.load() && tasks_.empty()) {
                    break;
                }

                task = tasks_.front();
                tasks_.pop_front();
            }
            prefetchRegionTask(task);
        }
    }

    int width_ = 0;
    int depth_ = 0;
    std::string path_;

    mutable std::ifstream stream_;
    mutable std::mutex stream_mutex_;
    mutable std::unordered_map<uint64_t, TerrainColumn> cache_;
    mutable std::shared_mutex cache_mutex_;
    bool stream_valid_ = false;

    std::thread worker_;
    std::mutex queue_mutex_;
    std::condition_variable queue_cv_;
    std::deque<RegionTask> tasks_;
    std::atomic<bool> worker_stop_;
    int default_radius_ = 8;
};

struct SimulationConfig {
    int width = 0;
    int depth = 0;
    double height = 0.0;
    double gravity = 1.62;
    double max_thrust = 10.0;
    double initial_fuel = 0.0;
    double start_x = 0.0;
    double start_y = 0.0;
    double start_z = 0.0;
    double target_x = 0.0;
    double target_y = 0.0;
    double target_z = 0.0;
};

struct VehicleState {
    double x = 0.0;
    double y = 0.0;
    double z = 0.0;
    double vx = 0.0;
    double vy = 0.0;
    double vz = 0.0;
    double pitch = 0.0;
    double yaw = 0.0;
    double roll = 0.0;
    double pitch_rate = 0.0;
    double yaw_rate = 0.0;
    double roll_rate = 0.0;
    double fuel = 0.0;
};

struct Command {
    double thrust = 0.0;
    double pitch = 0.0;
    double yaw = 0.0;
    double roll = 0.0;
};

class GuidanceController {
public:
    GuidanceController(const SimulationConfig& cfg, const TerrainCache& terrain)
        : cfg_(cfg),
          terrain_(terrain) {
        landing_zone_radius_ =
            std::max(5.0, std::min<double>(cfg_.width, cfg_.depth) * 0.02);

        target_ix_ = clampIndexFromDouble(cfg_.target_x, cfg_.width);
        target_iy_ = clampIndexFromDouble(cfg_.target_y, cfg_.depth);

        TerrainColumn target_col = terrain_.get(target_ix_, target_iy_);
        target_ground_ = target_col.ground;
        if (!isFinite(target_ground_)) {
            target_ground_ = cfg_.target_z;
        }

        touchdown_z_ = std::max(cfg_.target_z, target_ground_);
        min_touchdown_z_ = target_ground_;
        double clearance_buffer =
            std::clamp(cfg_.height * 0.05, 15.0, 5000.0);

        double float_ceiling = target_col.hasFloating()
                                   ? target_col.rock_top
                                   : target_ground_;
        safe_altitude_ = std::max({touchdown_z_ + clearance_buffer,
                                   float_ceiling + clearance_buffer,
                                   cfg_.start_z + clearance_buffer * 0.3});
        double top_limit = std::max(touchdown_z_ + 5.0, cfg_.height - 2.0);
        safe_altitude_ = std::clamp(safe_altitude_, touchdown_z_ + 5.0, top_limit);

        double vertical_room = std::max(safe_altitude_ - touchdown_z_, 8.0);
        double max_descent = std::max(vertical_room - 2.0, 8.0);
        final_descent_height_ =
            std::clamp(vertical_room * 0.45, 8.0, max_descent);

        descent_trigger_radius_ =
            std::max(landing_zone_radius_ * 4.0,
                     std::max<double>(cfg_.width, cfg_.depth) * 0.05);

        landing_bias_ = 0.0;

        double horizontal_scale =
            std::max({double(cfg_.width), double(cfg_.depth), 100.0});
        horizontal_pos_gain_ =
            std::clamp(3.0 / horizontal_scale, 0.01, 0.08);
        vertical_pos_gain_ =
            std::clamp(4.0 / std::max(cfg_.height, 50.0), 0.03, 0.25);

        base_horizontal_vel_gain_ = 1.1;
        base_vertical_vel_gain_ = 1.3;
        base_horizontal_limit_ = cfg_.max_thrust * 0.6;
        base_vertical_limit_ = cfg_.max_thrust * 0.9;

        phase_ = (cfg_.start_z + 1.0 < safe_altitude_) ? Phase::ASCEND
                                                       : Phase::CRUISE;
    }

    Command update(const VehicleState& state) {
        Command cmd;

        double dx = cfg_.target_x - state.x;
        double dy = cfg_.target_y - state.y;
        double horizontal_dist = std::hypot(dx, dy);
        double altitude_above_pad = state.z - touchdown_z_;

        phase_ = evaluatePhase(horizontal_dist, altitude_above_pad, state.z);
        double desired_z = computeDesiredAltitude(phase_, horizontal_dist);
        desired_z = std::clamp(desired_z, min_touchdown_z_, cfg_.height - 1.0);

        double kph = horizontal_pos_gain_;
        double kvp = vertical_pos_gain_;
        double kdh = base_horizontal_vel_gain_;
        double kvd = base_vertical_vel_gain_;
        double horiz_limit = base_horizontal_limit_;
        double vert_limit = base_vertical_limit_;

        switch (phase_) {
            case Phase::ASCEND:
                horiz_limit *= 0.5;
                kdh *= 0.8;
                break;
            case Phase::CRUISE:
                break;
            case Phase::DESCENT:
                horiz_limit *= 0.5;
                vert_limit *= 0.8;
                kdh *= 1.2;
                kvd *= 1.1;
                break;
            case Phase::LANDING:
                horiz_limit *= 0.3;
                vert_limit *= 0.5;
                kdh *= 1.6;
                kvd *= 1.4;
                break;
        }

        double ax = kph * dx - kdh * state.vx;
        double ay = kph * dy - kdh * state.vy;
        ax = std::clamp(ax, -horiz_limit, horiz_limit);
        ay = std::clamp(ay, -horiz_limit, horiz_limit);

        double az = kvp * (desired_z - state.z) - kvd * state.vz;
        az = std::clamp(az, -vert_limit, vert_limit);
        az = std::max(az, -cfg_.gravity);

        if (phase_ == Phase::LANDING) {
            double horizontal_speed = std::hypot(state.vx, state.vy);
            if (horizontal_speed > 0.5) {
                double brake = std::min(horiz_limit, horizontal_speed * 1.2);
                double inv = 1.0 / (horizontal_speed + 1e-6);
                ax += -state.vx * inv * brake;
                ay += -state.vy * inv * brake;
            }
            if (state.vz < -1.0) {
                az += std::min(vert_limit, (-state.vz - 1.0) * 0.6);
            }
        }

        double horiz_acc_norm = std::hypot(ax, ay);
        if (horiz_acc_norm > horiz_limit) {
            double scale = horiz_limit / (horiz_acc_norm + 1e-6);
            ax *= scale;
            ay *= scale;
        }

        az = std::clamp(az, -cfg_.gravity, vert_limit);

        std::array<double, 3> desired_thrust{
            ax, ay, az + cfg_.gravity
        };

        double thrust_norm = std::sqrt(desired_thrust[0] * desired_thrust[0] +
                                       desired_thrust[1] * desired_thrust[1] +
                                       desired_thrust[2] * desired_thrust[2]);

        std::array<double, 3> thrust_dir{0.0, 0.0, 1.0};
        if (thrust_norm > 1e-5) {
            thrust_dir[0] = desired_thrust[0] / thrust_norm;
            thrust_dir[1] = desired_thrust[1] / thrust_norm;
            thrust_dir[2] = desired_thrust[2] / thrust_norm;
        }

        double thrust_fraction =
            std::clamp(thrust_norm / cfg_.max_thrust, 0.0, 1.0);

        cmd.thrust = thrust_fraction;

        double desired_yaw = state.yaw;
        double desired_pitch = 0.0;
        double horiz_sq = thrust_dir[0] * thrust_dir[0] +
                          thrust_dir[1] * thrust_dir[1];

        if (horiz_sq > 1e-6) {
            desired_yaw = std::atan2(thrust_dir[1], thrust_dir[0]);
            double forward_x = std::cos(desired_yaw);
            double forward_y = std::sin(desired_yaw);
            double s = -(thrust_dir[0] * forward_x + thrust_dir[1] * forward_y);
            s = std::clamp(s, -1.0, 1.0);
            desired_pitch = std::asin(s);
        } else {
            desired_pitch = 0.0;
        }

        double pitch_error = desired_pitch - state.pitch;
        double yaw_error = wrapAngle(desired_yaw - state.yaw);
        double roll_error = -state.roll;

        const double kp_pitch = 1.6;
        const double kd_pitch = 0.35;
        const double kp_yaw = 1.2;
        const double kd_yaw = 0.35;
        const double kp_roll = 0.7;
        const double kd_roll = 0.25;

        double phase_pitch_scale = (phase_ == Phase::LANDING) ? 1.4 : 1.0;
        double phase_yaw_scale = (phase_ == Phase::LANDING) ? 1.2 : 1.0;

        cmd.pitch = std::clamp(
            phase_pitch_scale * kp_pitch * pitch_error - kd_pitch * state.pitch_rate,
            -1.0, 1.0);
        cmd.yaw = std::clamp(
            phase_yaw_scale * kp_yaw * yaw_error - kd_yaw * state.yaw_rate,
            -1.0, 1.0);
        cmd.roll = std::clamp(
            kp_roll * roll_error - kd_roll * state.roll_rate,
            -1.0, 1.0);

        if (!isFinite(cmd.thrust)) cmd.thrust = 0.0;
        if (!isFinite(cmd.pitch)) cmd.pitch = 0.0;
        if (!isFinite(cmd.yaw)) cmd.yaw = 0.0;
        if (!isFinite(cmd.roll)) cmd.roll = 0.0;

        return cmd;
    }

private:
    enum class Phase { ASCEND, CRUISE, DESCENT, LANDING };

    Phase evaluatePhase(double horizontal_dist,
                        double altitude_above_pad,
                        double absolute_z) const {
        if (horizontal_dist > descent_trigger_radius_) {
            if (absolute_z < safe_altitude_ - 2.0) {
                return Phase::ASCEND;
            }
            return Phase::CRUISE;
        }
        if (horizontal_dist > landing_zone_radius_ * 1.4) {
            if (absolute_z > touchdown_z_ + final_descent_height_ * 0.8) {
                return Phase::CRUISE;
            }
            return Phase::DESCENT;
        }
        if (altitude_above_pad > final_descent_height_) {
            return Phase::DESCENT;
        }
        return Phase::LANDING;
    }

    double computeDesiredAltitude(Phase phase, double horizontal_dist) const {
        switch (phase) {
            case Phase::ASCEND:
                return safe_altitude_;
            case Phase::CRUISE:
                return safe_altitude_;
            case Phase::DESCENT: {
                double denom = std::max(landing_zone_radius_ * 2.0, 1.0);
                double ratio = std::clamp(horizontal_dist / denom, 0.0, 1.0);
                double upper = safe_altitude_;
                double lower = touchdown_z_ + std::max(5.0, final_descent_height_ * 0.3);
                double target = lower + ratio * (upper - lower);
                return std::clamp(target, touchdown_z_ + 5.0, safe_altitude_);
            }
            case Phase::LANDING:
            default:
                return touchdown_z_ + landing_bias_;
        }
    }

    SimulationConfig cfg_;
    const TerrainCache& terrain_;

    int target_ix_ = 0;
    int target_iy_ = 0;
    double target_ground_ = 0.0;
    double touchdown_z_ = 0.0;
    double min_touchdown_z_ = 0.0;
    double landing_zone_radius_ = 5.0;
    double safe_altitude_ = 0.0;
    double final_descent_height_ = 10.0;
    double descent_trigger_radius_ = 50.0;
    double landing_bias_ = 0.0;

    double horizontal_pos_gain_ = 0.02;
    double vertical_pos_gain_ = 0.05;
    double base_horizontal_vel_gain_ = 1.1;
    double base_vertical_vel_gain_ = 1.3;
    double base_horizontal_limit_ = 5.0;
    double base_vertical_limit_ = 8.0;

    Phase phase_ = Phase::CRUISE;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    SimulationConfig cfg;
    double width_d = 0.0, depth_d = 0.0, height_d = 0.0;
    if (!(std::cin >> width_d >> depth_d >> height_d >> cfg.gravity >>
          cfg.max_thrust >> cfg.initial_fuel)) {
        return 0;
    }

    cfg.width = static_cast<int>(std::lround(width_d));
    cfg.depth = static_cast<int>(std::lround(depth_d));
    cfg.height = height_d;

    std::cin >> cfg.start_x >> cfg.start_y >> cfg.start_z >>
        cfg.target_x >> cfg.target_y >> cfg.target_z;

    std::string terrain_path;
    std::getline(std::cin >> std::ws, terrain_path);

    std::string state_token;
    std::cin >> state_token;
    if (state_token != "STATE") {
        return 0;
    }

    cfg.max_thrust = std::max(cfg.max_thrust, 0.5);
    cfg.gravity = std::max(cfg.gravity, 0.5);
    cfg.height = std::max(cfg.height, cfg.target_z + 5.0);

    TerrainCache terrain(cfg.width, cfg.depth, terrain_path);

    int prefetch_radius = terrain.defaultPrefetchRadius();
    terrain.requestPrefetchRegion(
        clampIndexFromDouble(cfg.target_x, cfg.width),
        clampIndexFromDouble(cfg.target_y, cfg.depth),
        prefetch_radius);
    terrain.requestPrefetchRegion(
        clampIndexFromDouble(cfg.start_x, cfg.width),
        clampIndexFromDouble(cfg.start_y, cfg.depth),
        prefetch_radius);

    GuidanceController controller(cfg, terrain);

    VehicleState state;
    Command cmd;

    auto last_prefetch = std::chrono::steady_clock::now();
    const auto prefetch_interval = std::chrono::milliseconds(400);

    std::cout << std::fixed << std::setprecision(4);

    while (std::cin >> state.x >> state.y >> state.z >>
           state.vx >> state.vy >> state.vz >>
           state.pitch >> state.yaw >> state.roll >>
           state.pitch_rate >> state.yaw_rate >> state.roll_rate >>
           state.fuel) {
        auto now = std::chrono::steady_clock::now();
        if (now - last_prefetch > prefetch_interval) {
            terrain.requestPrefetchRegion(
                clampIndexFromDouble(state.x, cfg.width),
                clampIndexFromDouble(state.y, cfg.depth),
                prefetch_radius / 2 + 2);
            last_prefetch = now;
        }

        cmd = controller.update(state);

        std::cout << cmd.thrust << ' '
                  << cmd.pitch << ' '
                  << cmd.yaw << ' '
                  << cmd.roll << '\n';
        std::cout.flush();
    }

    return 0;
}
```

This program:

- Reads the full scenario header, terrain path, and state stream without blocking.
- Parses the terrain file concurrently via a worker thread, caching only the regions needed.
- Uses a multi-phase PD autopilot (ascend, cruise, descent, landing) with thrust-vector steering and rate-damped attitude control.
- Streams real-time commands at ≥10 Hz with flushes each tick, suitable for tiny to huge maps.
