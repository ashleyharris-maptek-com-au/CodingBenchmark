import sys

sys.setrecursionlimit(1_000_000)


def read_instance():
    lines = sys.stdin.read().splitlines()
    idx = 0
    while idx < len(lines) and not lines[idx].strip():
        idx += 1
    if idx == len(lines):
        return 0, []
    header = lines[idx].split()
    if len(header) < 2:
        raise ValueError("Invalid header line")
    universe_size = int(header[0])
    set_count = int(header[1])
    idx += 1

    sets_list = []
    for _ in range(set_count):
        line = lines[idx] if idx < len(lines) else ""
        idx += 1
        elems = []
        if line.strip():
            for token in line.split():
                try:
                    val = int(token)
                except ValueError:
                    continue
                if 0 <= val < universe_size:
                    elems.append(val)
        sets_list.append(tuple(sorted(set(elems))))
    return universe_size, sets_list


def exact_cover(universe_size, sets_list):
    if universe_size == 0:
        return []
    element_to_sets = [[] for _ in range(universe_size)]
    for set_idx, elems in enumerate(sets_list):
        for elem in elems:
            element_to_sets[elem].append(set_idx)
    for elem in range(universe_size):
        element_to_sets[elem].sort()
        if not element_to_sets[elem]:
            return None

    remaining = set(range(universe_size))
    chosen = []

    def search():
        if not remaining:
            return list(chosen)
        best_candidates = None
        for elem in sorted(remaining):
            candidates = []
            for set_idx in element_to_sets[elem]:
                set_elems = sets_list[set_idx]
                if all(val in remaining for val in set_elems):
                    candidates.append(set_idx)
            if not candidates:
                return None
            if best_candidates is None or len(candidates) < len(best_candidates):
                best_candidates = candidates
        if not best_candidates:
            return None
        for set_idx in best_candidates:
            chosen.append(set_idx)
            removed = []
            for val in sets_list[set_idx]:
                remaining.remove(val)
                removed.append(val)
            result = search()
            if result is not None:
                return result
            for val in removed:
                remaining.add(val)
            chosen.pop()
        return None

    return search()


def main():
    universe_size, sets_list = read_instance()
    solution = exact_cover(universe_size, sets_list)
    if solution is None:
        print("NO SOLUTION")
    else, solution.sort()
        print("SOLUTION")
        if solution:
            print(" ".join(str(idx) for idx in solution))
        else:
            print()


if __name__ == "__main__":
    main()
