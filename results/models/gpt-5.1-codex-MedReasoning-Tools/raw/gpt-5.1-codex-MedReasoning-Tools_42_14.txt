OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4

%float_0_33333334 = OpConstant %float 0.333333343
%float_1 = OpConstant %float 1
%vec3_one = OpConstantComposite %v3float %float_1 %float_1 %float_1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_input_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_input_v4 = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_output_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_output_v4 = OpTypePointer Output %v4float

%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_input_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_input_arr_pv Input
%worldPos = OpVariable %ptr_input_arr_v3 Input
%normal = OpVariable %ptr_input_arr_v3 Input
%color = OpVariable %ptr_input_arr_v3 Input
%gl_out = OpVariable %ptr_output_pv Output
%outColor = OpVariable %ptr_output_v3 Output

%main = OpFunction %void None %func
%entry = OpLabel
%p0_ptr = OpAccessChain %ptr_input_v3 %worldPos %int_0
%p2_ptr = OpAccessChain %ptr_input_v3 %worldPos %int_2
%p4_ptr = OpAccessChain %ptr_input_v3 %worldPos %int_4
%p0 = OpLoad %v3float %p0_ptr
%p2 = OpLoad %v3float %p2_ptr
%p4 = OpLoad %v3float %p4_ptr
%v20 = OpFSub %v3float %p2 %p0
%v40 = OpFSub %v3float %p4 %p0
%a_x = OpCompositeExtract %float %v20 0
%a_y = OpCompositeExtract %float %v20 1
%a_z = OpCompositeExtract %float %v20 2
%b_x = OpCompositeExtract %float %v40 0
%b_y = OpCompositeExtract %float %v40 1
%b_z = OpCompositeExtract %float %v40 2
%ay_bz = OpFMul %float %a_y %b_z
%az_by = OpFMul %float %a_z %b_y
%cross_x = OpFSub %float %ay_bz %az_by
%az_bx = OpFMul %float %a_z %b_x
%ax_bz = OpFMul %float %a_x %b_z
%cross_y = OpFSub %float %az_bx %ax_bz
%ax_by = OpFMul %float %a_x %b_y
%ay_bx = OpFMul %float %a_y %b_x
%cross_z = OpFSub %float %ax_by %ay_bx
%cross = OpCompositeConstruct %v3float %cross_x %cross_y %cross_z
%N = OpExtInst %v3float %glsl Normalize %cross
%L = OpExtInst %v3float %glsl Normalize %vec3_one
%dot = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl FMax %dot %float_0
%c0_ptr = OpAccessChain %ptr_input_v3 %color %int_0
%c2_ptr = OpAccessChain %ptr_input_v3 %color %int_2
%c4_ptr = OpAccessChain %ptr_input_v3 %color %int_4
%c0 = OpLoad %v3float %c0_ptr
%c2 = OpLoad %v3float %c2_ptr
%c4 = OpLoad %v3float %c4_ptr
%sum02 = OpFAdd %v3float %c0 %c2
%sum = OpFAdd %v3float %sum02 %c4
%base = OpVectorTimesScalar %v3float %sum %float_0_33333334
%scaled_diff = OpFMul %float %float_0_8 %diffuse
%factor = OpFAdd %float %float_0_2 %scaled_diff
%finalColor = OpVectorTimesScalar %v3float %base %factor
%out_pos_ptr = OpAccessChain %ptr_output_v4 %gl_out %int_0

%g0_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_0 %int_0
%g0 = OpLoad %v4float %g0_ptr
OpStore %out_pos_ptr %g0

%g2 = OpLoad %v4float %g2_ptr
OpStore %out_pos_ptr %g2
OpStore %outColor %finalColor

%g4 = OpLoad %v4float %g4_ptr

OpEndPrimitive
OpReturn
