#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>

#include <vector>

class ComponentMIS {
public:
    explicit ComponentMIS(std::vector<std::vector<int>> adjacency)
        : n(static_cast<int>(adjacency.size())),
          words(n == 0 ? 0 : ((n + 63) >> 6)),
          graph(std::move(adjacency)),
          complement((n == 0) ? 0 : static_cast<size_t>(n) * static_cast<size_t>(words), ~0ULL),
          degOriginal(n, 0),
          degComplement(n, 0),
          maskBuffer(words, 0ULL) {
        if (n > 0 && words > 0) {
            buildComplement();
            currentClique.reserve(n);
            bestClique.reserve(n);
        }
    }

    std::vector<int> solve() {
        if (n == 0) {
            return {};
        }
        initDegrees();
        std::vector<int> initial = buildInitialSolution();
        bestClique = std::move(initial);
        currentClique.clear();

        if (static_cast<int>(bestClique.size()) == n) {
            return bestClique;
        }

        std::vector<int> candidates(n);
        std::iota(candidates.begin(), candidates.end(), 0);
        std::sort(candidates.begin(), candidates.end(), [&](int a, int b) {
            if (degComplement[a] != degComplement[b]) return degComplement[a] > degComplement[b];
            if (degOriginal[a] != degOriginal[b]) return degOriginal[a] < degOriginal[b];
            return a < b;
        });

        int maxDeg = 0;
        for (int deg : degComplement) {
            if (deg > maxDeg) maxDeg = deg;
        }
        if (static_cast<int>(bestClique.size()) >= maxDeg + 1) {
            return bestClique;
        }

        expand(candidates);
        return bestClique;
    }

private:
    int n;
    int words;
    std::vector<std::vector<int>> graph;
    std::vector<uint64_t> complement;
    std::vector<int> degOriginal;
    std::vector<int> degComplement;
    std::vector<int> bestClique;
    std::vector<int> currentClique;
    std::vector<uint64_t> maskBuffer;

    void buildComplement() {
        if (words == 0) return;
        if ((n & 63) != 0) {
            uint64_t tailMask = (uint64_t(1) << (n & 63)) - 1ULL;
            for (int v = 0; v < n; ++v) {
                complement[static_cast<size_t>(v) * words + (words - 1)] &= tailMask;
            }
        }
        for (int v = 0; v < n; ++v) {
            complement[static_cast<size_t>(v) * words + (v >> 6)] &= ~(uint64_t(1) << (v & 63));
        }
        for (int v = 0; v < n; ++v) {
            for (int u : graph[v]) {
                complement[static_cast<size_t>(v) * words + (u >> 6)] &= ~(uint64_t(1) << (u & 63));
            }
        }
    }

    void initDegrees() {
        for (int v = 0; v < n; ++v) {
            degOriginal[v] = static_cast<int>(graph[v].size());
            degComplement[v] = n - 1 - degOriginal[v];
        }
    }

    std::vector<int> runGreedy(const std::vector<int>& order) const {
        std::vector<char> removed(n, false);
        std::vector<int> result;
        result.reserve(order.size());
        for (int v : order) {
            if (removed[v]) continue;
            result.push_back(v);
            removed[v] = true;
            for (int nb : graph[v]) {
                removed[nb] = true;
            }
        }
        return result;
    }

    std::vector<int> buildInitialSolution() const {
        std::vector<int> order(n);
        std::iota(order.begin(), order.end(), 0);
        std::vector<int> best;

        auto consider = [&](const std::vector<int>& seq) {
            std::vector<int> candidate = runGreedy(seq);
            if (candidate.size() > best.size()) {
                best = std::move(candidate);
            }
        };

        consider(order);

        std::vector<int> degreeAsc = order;
        std::sort(degreeAsc.begin(), degreeAsc.end(), [&](int a, int b) {
            if (degOriginal[a] != degOriginal[b]) return degOriginal[a] < degOriginal[b];
            return a < b;
        });
        consider(degreeAsc);

        std::vector<int> degreeDesc = degreeAsc;
        std::reverse(degreeDesc.begin(), degreeDesc.end());
        consider(degreeDesc);

        std::vector<int> randomOrder = order;
        std::mt19937 rng(712367);
        std::shuffle(randomOrder.begin(), randomOrder.end(), rng);
        consider(randomOrder);

        return best;
    }

    void colorSort(const std::vector<int>& P, std::vector<int>& order, std::vector<int>& colors) {
        order.clear();
        colors.clear();
        if (P.empty()) return;
        std::vector<int> rest = P;
        std::vector<int> next;
        next.reserve(rest.size());
        int color = 0;
        while (!rest.empty()) {
            ++color;
            std::fill(maskBuffer.begin(), maskBuffer.end(), 0ULL);
            next.clear();
            for (int v : rest) {
                const uint64_t* row = complement.data() + static_cast<size_t>(v) * words;
                bool conflict = false;
                for (int w = 0; w < words; ++w) {
                    if (row[w] & maskBuffer[w]) {
                        conflict = true;
                        break;
                    }
                }
                if (!conflict) {
                    order.push_back(v);
                    colors.push_back(color);
                    int wordIdx = v >> 6;
                    maskBuffer[wordIdx] |= (uint64_t(1) << (v & 63));
                } else {
                    next.push_back(v);
                }
            }
            rest.swap(next);
        }
    }

    bool adjacent(int u, int v) const {
        const uint64_t* row = complement.data() + static_cast<size_t>(u) * words;
        return (row[v >> 6] >> (v & 63)) & 1ULL;
    }

    void expand(std::vector<int>& P) {
        if (currentClique.size() > bestClique.size()) {
            bestClique = currentClique;
        }
        if (P.empty()) {
            return;
        }
        if (currentClique.size() + P.size() <= bestClique.size()) {
            return;
        }

        std::vector<int> order;
        order.reserve(P.size());
        std::vector<int> colors;
        colors.reserve(P.size());
        colorSort(P, order, colors);

        std::vector<int> newP;
        newP.reserve(P.size());

        for (int idx = static_cast<int>(order.size()) - 1; idx >= 0; --idx) {
            int v = order[idx];
            if (currentClique.size() + static_cast<size_t>(colors[idx]) <= bestClique.size()) {
                return;
            }
            currentClique.push_back(v);
            newP.clear();
            for (int j = 0; j < idx; ++j) {
                int u = order[j];
                if (adjacent(v, u)) {
                    newP.push_back(u);
                }
            }
            if (newP.empty()) {
                if (currentClique.size() > bestClique.size()) {
                    bestClique = currentClique;
                }
            } else {
                expand(newP);
            }
            currentClique.pop_back();
        }
    }
};

std::vector<int> maximumIndependentSetBipartite(const std::vector<std::vector<int>>& adj,
                                                const std::vector<int>& color) {
    int n = static_cast<int>(adj.size());
    int leftCount = 0;
    for (int c : color) {
        if (c == 0) ++leftCount;
    }
    int rightCount = n - leftCount;

    std::vector<int> leftNodes;
    std::vector<int> rightNodes;
    leftNodes.reserve(leftCount);
    rightNodes.reserve(rightCount);
    std::vector<int> leftIndex(n, -1), rightIndex(n, -1);
    for (int v = 0; v < n; ++v) {
        if (color[v] == 0) {
            leftIndex[v] = static_cast<int>(leftNodes.size());
            leftNodes.push_back(v);
        } else {
            rightIndex[v] = static_cast<int>(rightNodes.size());
            rightNodes.push_back(v);
        }
    }

    std::vector<std::vector<int>> leftAdj(leftCount);
    for (int i = 0; i < leftCount; ++i) {
        int v = leftNodes[i];
        auto& lst = leftAdj[i];
        lst.reserve(adj[v].size());
        for (int nb : adj[v]) {
            if (color[nb] == 1) {
                lst.push_back(rightIndex[nb]);
            }
        }
    }

    std::vector<int> matchLeft(leftCount, -1);
    std::vector<int> matchRight(rightCount, -1);
    std::vector<int> dist(leftCount);
    const int INF = std::numeric_limits<int>::max();

    auto bfs = [&]() -> bool {
        std::queue<int> q;
        bool reachable = false;
        for (int i = 0; i < leftCount; ++i) {
            if (matchLeft[i] == -1) {
                dist[i] = 0;
                q.push(i);
            } else {
                dist[i] = INF;
            }
        }
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (int v : leftAdj[u]) {
                int mate = matchRight[v];
                if (mate != -1 && dist[mate] == INF) {
                    dist[mate] = dist[u] + 1;
                    q.push(mate);
                }
                if (mate == -1) {
                    reachable = true;
                }
            }
        }
        return reachable;
    };

    auto dfs = [&](auto&& self, int u) -> bool {
        for (int v : leftAdj[u]) {
            int mate = matchRight[v];
            if (mate == -1 || (dist[mate] == dist[u] + 1 && self(self, mate))) {
                matchLeft[u] = v;
                matchRight[v] = u;
                return true;
            }
        }
        dist[u] = INF;
        return false;
    };

    while (bfs()) {
        for (int i = 0; i < leftCount; ++i) {
            if (matchLeft[i] == -1) {
                dfs(dfs, i);
            }
        }
    }

    std::queue<int> q;
    std::vector<char> seenLeft(leftCount, false), seenRight(rightCount, false);
    for (int i = 0; i < leftCount; ++i) {
        if (matchLeft[i] == -1) {
            seenLeft[i] = true;
            q.push(i);
        }
    }
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        for (int v : leftAdj[u]) {
            if (matchLeft[u] != v && !seenRight[v]) {
                seenRight[v] = true;
                int mate = matchRight[v];
                if (mate != -1 && !seenLeft[mate]) {
                    seenLeft[mate] = true;
                    q.push(mate);
                }
            }
        }
    }

    std::vector<int> independent;
    independent.reserve(n);
    for (int i = 0; i < leftCount; ++i) {
        if (seenLeft[i]) {
            independent.push_back(leftNodes[i]);
        }
    }
    for (int j = 0; j < rightCount; ++j) {
        if (!seenRight[j]) {
            independent.push_back(rightNodes[j]);
        }
    }
    return independent;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n;
    long long m;
    if (!(std::cin >> n >> m)) {
        return 0;
    }

    if (n <= 0) {
        for (long long i = 0; i < m; ++i) {
            int u, v;
            if (!(std::cin >> u >> v)) break;
        }
        std::cout << 0 << "\n\n";
        return 0;
    }

    std::vector<std::vector<int>> globalAdj(n);
    for (long long i = 0; i < m; ++i) {
        int u, v;
        std::cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) continue;
        globalAdj[u].push_back(v);
        globalAdj[v].push_back(u);
    }

    for (int v = 0; v < n; ++v) {
        auto& nbr = globalAdj[v];
        std::sort(nbr.begin(), nbr.end());
        nbr.erase(std::unique(nbr.begin(), nbr.end()), nbr.end());
    }

    std::vector<char> visited(n, false);
    std::vector<int> stack;
    stack.reserve(n);
    std::vector<int> nodes;
    nodes.reserve(n);
    std::vector<int> globalToLocal(n, -1);
    std::vector<int> answer;
    answer.reserve(n);

    for (int start = 0; start < n; ++start) {
        if (visited[start]) continue;
        stack.clear();
        stack.push_back(start);
        visited[start] = true;
        nodes.clear();

        while (!stack.empty()) {
            int v = stack.back();
            stack.pop_back();
            nodes.push_back(v);
            for (int nb : globalAdj[v]) {
                if (!visited[nb]) {
                    visited[nb] = true;
                    stack.push_back(nb);
                }
            }
        }

        int compSize = static_cast<int>(nodes.size());
        if (compSize == 0) continue;

        for (int i = 0; i < compSize; ++i) {
            globalToLocal[nodes[i]] = i;
        }

        std::vector<std::vector<int>> localAdj(compSize);
        for (int i = 0; i < compSize; ++i) {
            int gv = nodes[i];
            auto& lst = localAdj[i];
            lst.reserve(globalAdj[gv].size());
            for (int nb : globalAdj[gv]) {
                int id = globalToLocal[nb];
                if (id != -1) {
                    lst.push_back(id);
                }
            }
        }

        std::vector<int> color(compSize, -1);
        bool isBip = true;
        std::vector<int> bfsQueue;
        bfsQueue.reserve(compSize);
        for (int i = 0; i < compSize && isBip; ++i) {
            if (color[i] != -1) continue;
            bfsQueue.clear();
            bfsQueue.push_back(i);
            color[i] = 0;
            for (size_t pos = 0; pos < bfsQueue.size() && isBip; ++pos) {
                int v = bfsQueue[pos];
                for (int nb : localAdj[v]) {
                    if (color[nb] == -1) {
                        color[nb] = color[v] ^ 1;
                        bfsQueue.push_back(nb);
                    } else if (color[nb] == color[v]) {
                        isBip = false;
                        break;
                    }
                }
            }
        }

        std::vector<int> localSolution;
        if (isBip) {
            localSolution = maximumIndependentSetBipartite(localAdj, color);
        } else {
            ComponentMIS solver(std::move(localAdj));
            localSolution = solver.solve();
        }

        for (int idx : localSolution) {
            answer.push_back(nodes[idx]);
        }

        for (int v : nodes) {
            globalToLocal[v] = -1;
        }
    }

    std::sort(answer.begin(), answer.end());
    std::cout << answer.size() << "\n";
    if (!answer.empty()) {
        for (size_t i = 0; i < answer.size(); ++i) {
            if (i) std::cout << ' ';
            std::cout << answer[i];
        }
    }
    std::cout << "\n";
    return 0;
}

This program:
- Reads the graph, deduplicates edges, and splits it into connected components.
- Uses Hopcroft–Karp and Konig’s theorem for bipartite components (polynomial-time exact MIS).
- Falls back to a branch-and-bound maximum clique search on the complement graph (with coloring bounds and heuristics) for other components.
- Outputs the exact maximum independent set with the required format.
