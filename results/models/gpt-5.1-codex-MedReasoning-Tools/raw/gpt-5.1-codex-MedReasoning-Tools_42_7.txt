OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
%glsl450 = OpExtInstImport "GLSL.std.450"
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpSource GLSL 450

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0 = OpConstant %float 0
%float_0_08 = OpConstant %float 0.0800000008
%float_0_33333334 = OpConstant %float 0.333333343

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_Input_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
%ptr_Input_v4float = OpTypePointer Input %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%arr_v3float = OpTypeArray %v3float %uint_3
%ptr_Input_arr_v3float = OpTypePointer Input %arr_v3float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v3float = OpTypePointer Output %v3float

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_gl_PerVertex_in ArrayStride 16
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %arr_v3float ArrayStride 16
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in Input
%worldPos = OpVariable %ptr_Input_arr_v3float Input
%normal = OpVariable %ptr_Input_arr_v3float Input
%color = OpVariable %ptr_Input_arr_v3float Input
%gl_out = OpVariable %ptr_Output_gl_PerVertex Output
%outColor = OpVariable %ptr_Output_v3float Output

%main = OpFunction %void None %func
%entry = OpLabel
%p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%sum01 = OpFAdd %v4float %p0 %p1
%sum012 = OpFAdd %v4float %sum01 %p2
%center = OpVectorTimesScalar %v4float %sum012 %float_0_33333334
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2
%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz
%cross = OpExtInst %v3float %glsl450 Cross %e1 %e2
%normalVec = OpExtInst %v3float %glsl450 Normalize %cross
%offset = OpVectorTimesScalar %v3float %normalVec %float_0_08
%off_x = OpCompositeExtract %float %offset 0
%off_y = OpCompositeExtract %float %offset 1
%off_z = OpCompositeExtract %float %offset 2
%offset4 = OpCompositeConstruct %v4float %off_x %off_y %off_z %float_0
%out_pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0

%p0_shift = OpFAdd %v4float %p0 %offset4
OpStore %out_pos_ptr %p0_shift
%color0_ptr = OpAccessChain %ptr_Input_v3float %color %int_0
%color0 = OpLoad %v3float %color0_ptr
OpStore %outColor %color0
OpEmitVertex

%p1_shift = OpFAdd %v4float %p1 %offset4
OpStore %out_pos_ptr %p1_shift
%color1_ptr = OpAccessChain %ptr_Input_v3float %color %int_1
%color1 = OpLoad %v3float %color1_ptr
OpStore %outColor %color1
OpEmitVertex

%p2_shift = OpFAdd %v4float %p2 %offset4
OpStore %out_pos_ptr %p2_shift
%color2_ptr = OpAccessChain %ptr_Input_v3float %color %int_2
%color2 = OpLoad %v3float %color2_ptr
OpStore %outColor %color2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
