Below is a complete C++20 program that reads the generalized N‑dimensional snake configuration, repeatedly plans safe moves (using A* toward the nearest food, tail-chasing fallback, and finally greedy heuristics), and prints one move per turn until all food is eaten or the move/turn budget is exhausted.

```cpp
#include <array>
#include <vector>
#include <deque>
#include <unordered_map>
#include <unordered_set>
#include <queue>
#include <iostream>
#include <limits>
#include <cstdint>
#include <algorithm>

using namespace std;

constexpr int MAX_D = 20;
int g_dim = 0;
array<int, MAX_D> g_bounds{};

struct Coord {
    array<int, MAX_D> v{};
};

struct CoordHash {
    size_t operator()(const Coord& c) const noexcept {
        size_t h = 0x9e3779b97f4a7c15ULL;
        for (int i = 0; i < g_dim; ++i) {
            size_t val = static_cast<size_t>(c.v[i]);
            h ^= val + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        }
        return h;
    }
};

struct CoordEq {
    bool operator()(const Coord& a, const Coord& b) const noexcept {
        for (int i = 0; i < g_dim; ++i) {
            if (a.v[i] != b.v[i]) return false;
        }
        return true;
    }
};

using CoordMap = unordered_map<Coord, int, CoordHash, CoordEq>;
using CoordSet = unordered_set<Coord, CoordHash, CoordEq>;
using CoordIndex = unordered_map<Coord, size_t, CoordHash, CoordEq>;

struct Move {
    int axis = 0;
    int dir = 1;
};

inline Coord readCoord(istream& in) {
    Coord c;
    for (int i = 0; i < g_dim; ++i) {
        in >> c.v[i];
    }
    return c;
}

inline bool inBounds(const Coord& c) {
    for (int i = 0; i < g_dim; ++i) {
        if (c.v[i] < 0 || c.v[i] >= g_bounds[i]) return false;
    }
    return true;
}

inline Coord stepCoord(const Coord& c, int axis, int dir) {
    Coord n = c;
    n.v[axis] += dir;
    return n;
}

inline int manhattan(const Coord& a, const Coord& b) {
    int dist = 0;
    for (int i = 0; i < g_dim; ++i) {
        dist += std::abs(a.v[i] - b.v[i]);
    }
    return dist;
}

inline int closestFoodDist(const Coord& c, const vector<Coord>& foods) {
    if (foods.empty()) return 0;
    int best = numeric_limits<int>::max();
    for (const auto& f : foods) {
        int d = 0;
        for (int i = 0; i < g_dim; ++i) {
            d += std::abs(c.v[i] - f.v[i]);
        }
        if (d < best) best = d;
    }
    if (best == numeric_limits<int>::max()) best = 0;
    return best;
}

inline int boundaryMargin(const Coord& c) {
    int sum = 0;
    for (int i = 0; i < g_dim; ++i) {
        int lower = c.v[i];
        int upper = g_bounds[i] - 1 - c.v[i];
        sum += std::min(lower, upper);
    }
    return sum;
}

struct NodeInfo {
    Coord parent{};
    int axis = -1;
    int dir = 0;
    bool hasParent = false;
    int firstAxis = -1;
    int firstDir = 0;
    bool hasFirst = false;
};

struct FrontierEntry {
    Coord coord{};
    int g = 0;
    int f = 0;
    int h = 0;
};

struct FrontierCmp {
    bool operator()(const FrontierEntry& lhs, const FrontierEntry& rhs) const noexcept {
        if (lhs.f != rhs.f) return lhs.f > rhs.f;
        return lhs.h > rhs.h;
    }
};

template <typename GoalTest, typename Heuristic>
bool planFirstStep(const Coord& start,
                   const GoalTest& isGoal,
                   const Heuristic& heuristic,
                   const CoordSet& obstacles,
                   const CoordMap& occupancy,
                   const Coord& tail,
                   bool tailCanBeUsed,
                   int nodeLimit,
                   Move& outMove) {
    if (isGoal(start)) return false;

    unordered_map<Coord, int, CoordHash, CoordEq> best;
    unordered_map<Coord, NodeInfo, CoordHash, CoordEq> info;
    priority_queue<FrontierEntry, vector<FrontierEntry>, FrontierCmp> pq;
    int startH = heuristic(start);
    best[start] = 0;
    NodeInfo startInfo;
    info[start] = startInfo;
    pq.push({start, 0, startH, startH});

    int processed = 0;
    while (!pq.empty()) {
        auto cur = pq.top();
        pq.pop();
        auto itBest = best.find(cur.coord);
        if (itBest == best.end() || cur.g != itBest->second) continue;

        if (isGoal(cur.coord) && cur.g > 0) {
            const NodeInfo& nodeInfo = info[cur.coord];
            if (!nodeInfo.hasFirst) continue;
            outMove.axis = nodeInfo.firstAxis;
            outMove.dir = nodeInfo.firstDir;
            return true;
        }

        if (processed++ > nodeLimit) break;

        const NodeInfo& curInfo = info[cur.coord];

        for (int axis = 0; axis < g_dim; ++axis) {
            for (int dir = -1; dir <= 1; dir += 2) {
                Coord nxt = cur.coord;
                nxt.v[axis] += dir;
                if (!inBounds(nxt)) continue;
                if (obstacles.find(nxt) != obstacles.end()) continue;
                auto occIt = occupancy.find(nxt);
                if (occIt != occupancy.end()) {
                    bool tailCell = (CoordEq{}(nxt, tail));
                    bool permitted = tailCell && tailCanBeUsed && occIt->second == 1;
                    if (!permitted) continue;
                }
                int newG = cur.g + 1;
                auto bestNext = best.find(nxt);
                if (bestNext != best.end() && newG >= bestNext->second) continue;
                best[nxt] = newG;

                NodeInfo nxtInfo;
                nxtInfo.parent = cur.coord;
                nxtInfo.axis = axis;
                nxtInfo.dir = dir;
                nxtInfo.hasParent = true;
                if (curInfo.hasFirst) {
                    nxtInfo.firstAxis = curInfo.firstAxis;
                    nxtInfo.firstDir = curInfo.firstDir;
                } else {
                    nxtInfo.firstAxis = axis;
                    nxtInfo.firstDir = dir;
                }
                nxtInfo.hasFirst = true;
                info[nxt] = nxtInfo;

                int h = heuristic(nxt);
                pq.push({nxt, newG, newG + h, h});
            }
        }
    }
    return false;
}

bool planToFoods(const Coord& head,
                 const CoordSet& obstacles,
                 const CoordMap& occupancy,
                 const Coord& tail,
                 bool tailCanBeUsed,
                 const vector<Coord>& foods,
                 const CoordIndex& foodIndex,
                 Move& outMove,
                 int nodeLimit) {
    if (foods.empty()) return false;
    auto goalTest = [&](const Coord& c) -> bool {
        return foodIndex.find(c) != foodIndex.end();
    };
    auto heuristic = [&](const Coord& c) -> int {
        return closestFoodDist(c, foods);
    };
    return planFirstStep(head, goalTest, heuristic, obstacles, occupancy, tail,
                         tailCanBeUsed, nodeLimit, outMove);
}

bool planToTail(const Coord& head,
                const CoordSet& obstacles,
                const CoordMap& occupancy,
                const Coord& tail,
                bool tailCanBeUsed,
                Move& outMove,
                int nodeLimit) {
    auto goalTest = [&](const Coord& c) -> bool {
        return CoordEq{}(c, tail);
    };
    auto heuristic = [&](const Coord& c) -> int {
        return manhattan(c, tail);
    };
    return planFirstStep(head, goalTest, heuristic, obstacles, occupancy, tail,
                         tailCanBeUsed, nodeLimit, outMove);
}

bool chooseGreedyMove(const deque<Coord>& body,
                      const CoordSet& obstacles,
                      const CoordMap& occupancy,
                      const vector<Coord>& foods,
                      const CoordIndex& foodIndex,
                      long long growthBudget,
                      Move& outMove) {
    if (body.empty()) return false;
    const Coord& head = body.front();
    const Coord& tail = body.back();
    bool found = false;
    int bestScore = numeric_limits<int>::min();
    Move bestMove;
    for (int axis = 0; axis < g_dim; ++axis) {
        for (int dir = -1; dir <= 1; dir += 2) {
            Coord nxt = stepCoord(head, axis, dir);
            if (!inBounds(nxt)) continue;
            if (obstacles.find(nxt) != obstacles.end()) continue;
            bool willEat = foodIndex.find(nxt) != foodIndex.end();
            bool tailMoves = (growthBudget == 0) && !willEat;
            auto occIt = occupancy.find(nxt);
            if (occIt != occupancy.end()) {
                bool tailCell = CoordEq{}(nxt, tail);
                bool allowed = tailCell && tailMoves && occIt->second == 1;
                if (!allowed) continue;
            }
            int score;
            if (!foods.empty()) {
                score = -closestFoodDist(nxt, foods);
            } else {
                score = boundaryMargin(nxt);
            }
            if (!found || score > bestScore) {
                found = true;
                bestScore = score;
                bestMove = {axis, dir};
            }
        }
    }
    if (found) {
        outMove = bestMove;
    }
    return found;
}

bool chooseEmergencyMove(const deque<Coord>& body, Move& outMove) {
    if (body.empty()) return false;
    const Coord& head = body.front();
    for (int axis = 0; axis < g_dim; ++axis) {
        for (int dir = -1; dir <= 1; dir += 2) {
            Coord nxt = stepCoord(head, axis, dir);
            if (!inBounds(nxt)) continue;
            outMove = {axis, dir};
            return true;
        }
    }
    if (g_dim > 0) {
        outMove = {0, (g_bounds[0] > 1 ? 1 : -1)};
        return true;
    }
    return false;
}

void removeFood(const Coord& c, vector<Coord>& foods, CoordIndex& idx) {
    auto it = idx.find(c);
    if (it == idx.end()) return;
    size_t pos = it->second;
    size_t last = foods.size() - 1;
    if (pos != last) {
        foods[pos] = foods[last];
        idx[foods[pos]] = pos;
    }
    foods.pop_back();
    idx.erase(it);
}

bool isSafeMove(const Move& move,
                const deque<Coord>& body,
                const CoordSet& obstacles,
                const CoordMap& occupancy,
                const CoordIndex& foodIndex,
                long long growthBudget) {
    if (body.empty()) return false;
    const Coord& head = body.front();
    Coord nxt = stepCoord(head, move.axis, move.dir);
    if (!inBounds(nxt)) return false;
    if (obstacles.find(nxt) != obstacles.end()) return false;
    bool willEat = foodIndex.find(nxt) != foodIndex.end();
    bool tailMoves = (growthBudget == 0) && !willEat;
    const Coord& tail = body.back();
    auto occIt = occupancy.find(nxt);
    if (occIt != occupancy.end()) {
        bool tailCell = CoordEq{}(nxt, tail);
        bool allowed = tailCell && tailMoves && occIt->second == 1;
        if (!allowed) return false;
    }
    return true;
}

void applyMove(const Move& move,
               deque<Coord>& body,
               CoordMap& occupancy,
               vector<Coord>& foods,
               CoordIndex& foodIndex,
               long long& growthBudget) {
    Coord newHead = stepCoord(body.front(), move.axis, move.dir);
    body.push_front(newHead);
    occupancy[newHead] += 1;

    bool ate = false;
    if (foodIndex.find(newHead) != foodIndex.end()) {
        ate = true;
        removeFood(newHead, foods, foodIndex);
        growthBudget += 1;
    }

    if (growthBudget > 0) {
        --growthBudget;
    } else {
        Coord tail = body.back();
        auto it = occupancy.find(tail);
        if (it != occupancy.end()) {
            it->second -= 1;
            if (it->second == 0) occupancy.erase(it);
        }
        body.pop_back();
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long maxTurns;
    if (!(cin >> g_dim >> maxTurns)) {
        return 0;
    }
    if (g_dim <= 0 || g_dim > MAX_D) {
        return 0;
    }
    for (int i = 0; i < g_dim; ++i) {
        cin >> g_bounds[i];
    }

    int snakeLength;
    cin >> snakeLength;
    Coord head = readCoord(cin);

    deque<Coord> body;
    body.clear();
    for (int i = 0; i < snakeLength; ++i) {
        body.push_back(head);
    }

    CoordMap occupancy;
    occupancy[head] = snakeLength;

    long long growthBudget = 0;

    int foodCount;
    cin >> foodCount;
    vector<Coord> foods;
    foods.reserve(foodCount);
    CoordIndex foodIndex;
    for (int i = 0; i < foodCount; ++i) {
        Coord f = readCoord(cin);
        foodIndex[f] = foods.size();
        foods.push_back(f);
    }

    int obstacleCount;
    cin >> obstacleCount;
    CoordSet obstacles;
    for (int i = 0; i < obstacleCount; ++i) {
        Coord o = readCoord(cin);
        obstacles.insert(o);
    }

    const int NODE_LIMIT = 600000;
    long long turns = 0;
    while (turns < maxTurns && !foods.empty()) {
        const Coord& currentHead = body.front();
        const Coord& currentTail = body.back();
        bool tailSingle = false;
        auto tailIt = occupancy.find(currentTail);
        if (tailIt != occupancy.end() && tailIt->second == 1) {
            tailSingle = true;
        }
        bool tailCanBeUsed = tailSingle && (growthBudget == 0);

        Move chosen;
        bool haveMove = false;

        haveMove = planToFoods(currentHead, obstacles, occupancy, currentTail,
                               tailCanBeUsed, foods, foodIndex, chosen, NODE_LIMIT);

        if (!haveMove && body.size() > 1) {
            haveMove = planToTail(currentHead, obstacles, occupancy, currentTail,
                                  tailCanBeUsed, chosen, NODE_LIMIT);
        }

        if (!haveMove) {
            haveMove = chooseGreedyMove(body, obstacles, occupancy, foods,
                                        foodIndex, growthBudget, chosen);
        }

        if (!haveMove || !isSafeMove(chosen, body, obstacles, occupancy, foodIndex, growthBudget)) {
            Move emergency;
            if (chooseEmergencyMove(body, emergency)) {
                cout << emergency.axis << ' ' << emergency.dir << '\n';
            }
            break;
        }

        applyMove(chosen, body, occupancy, foods, foodIndex, growthBudget);
        cout << chosen.axis << ' ' << chosen.dir << '\n';
        ++turns;
    }

    return 0;
}
```

This solver:

- Supports up to 20 dimensions with arbitrary per-axis bounds.
- Uses A* search each turn to head for the nearest reachable food, treating self/obstacles appropriately and allowing entry into the tail only when it will vacate.
- Falls back to tail-chasing and greedy heuristics if food paths are temporarily blocked.
- Maintains exact body occupancy counts so duplicated coordinates (during initial overlap or growth) are tracked safely.
- Outputs one move per simulated turn until either all food is consumed, the search cannot find any safe move, or the maximum turn budget is reached.
