OpCapability Geometry
OpMemoryModel Logical GLSL450
%glsl450 = OpExtInstImport "GLSL.std.450"
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %arr_v3float ArrayStride 12
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_thickness = OpConstant %float 0.025
%float_eps = OpConstant %float 0.000001
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_Input_v4float = OpTypePointer Input %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_glpv = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%arr_v3float = OpTypeArray %v3float %uint_4
%ptr_Input_arr_v3float = OpTypePointer Input %arr_v3float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v3float = OpTypePointer Output %v3float
%gl_in = OpVariable %ptr_Input_arr_pv Input
%in_worldPos = OpVariable %ptr_Input_arr_v3float Input
%in_normal = OpVariable %ptr_Input_arr_v3float Input
%in_color = OpVariable %ptr_Input_arr_v3float Input
%gl_out = OpVariable %ptr_Output_glpv Output
%outColor = OpVariable %ptr_Output_v3float Output
%main = OpFunction %void None %fn
%label = OpLabel
%ptr_prev = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%pos_prev = OpLoad %v4float %ptr_prev
%ptr_start = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%pos_start = OpLoad %v4float %ptr_start
%ptr_end = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%pos_end = OpLoad %v4float %ptr_end
%ptr_next = OpAccessChain %ptr_Input_v4float %gl_in %int_3 %int_0
%pos_next = OpLoad %v4float %ptr_next
%ptr_col_start = OpAccessChain %ptr_Input_v3float %in_color %int_1
%color_start = OpLoad %v3float %ptr_col_start
%ptr_col_end = OpAccessChain %ptr_Input_v3float %in_color %int_2
%color_end = OpLoad %v3float %ptr_col_end
%diff0 = OpFSub %v4float %pos_end %pos_prev
%diff0_xy = OpVectorShuffle %v2float %diff0 %diff0 0 1
%dot0 = OpDot %float %diff0_xy %diff0_xy
%len0 = OpExtInst %float %glsl450 Sqrt %dot0
%len0_safe = OpExtInst %float %glsl450 FMax %len0 %float_eps
%inv_len0 = OpFDiv %float %float_1 %len0_safe
%t0 = OpVectorTimesScalar %v2float %diff0_xy %inv_len0
%diff1 = OpFSub %v4float %pos_next %pos_start
%diff1_xy = OpVectorShuffle %v2float %diff1 %diff1 0 1
%dot1 = OpDot %float %diff1_xy %diff1_xy
%len1 = OpExtInst %float %glsl450 Sqrt %dot1
%len1_safe = OpExtInst %float %glsl450 FMax %len1 %float_eps
%inv_len1 = OpFDiv %float %float_1 %len1_safe
%t1 = OpVectorTimesScalar %v2float %diff1_xy %inv_len1
%t0_x = OpCompositeExtract %float %t0 0
%t0_y = OpCompositeExtract %float %t0 1
%neg_t0_y = OpFNegate %float %t0_y
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x
%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1
%neg_t1_y = OpFNegate %float %t1_y
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x
%offset0_2 = OpVectorTimesScalar %v2float %perp0 %float_thickness
%offset1_2 = OpVectorTimesScalar %v2float %perp1 %float_thickness
%offset0_x = OpCompositeExtract %float %offset0_2 0
%offset0_y = OpCompositeExtract %float %offset0_2 1
%offset0_4 = OpCompositeConstruct %v4float %offset0_x %offset0_y %float_0 %float_0
%offset1_x = OpCompositeExtract %float %offset1_2 0
%offset1_y = OpCompositeExtract %float %offset1_2 1
%offset1_4 = OpCompositeConstruct %v4float %offset1_x %offset1_y %float_0 %float_0
%pos_start_plus = OpFAdd %v4float %pos_start %offset0_4
%pos_start_minus = OpFSub %v4float %pos_start %offset0_4
%pos_end_plus = OpFAdd %v4float %pos_end %offset1_4
%pos_end_minus = OpFSub %v4float %pos_end %offset1_4
%out_pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr %pos_start_plus
OpStore %outColor %color_start
OpEmitVertex
OpStore %out_pos_ptr %pos_start_minus
OpStore %outColor %color_start
OpEmitVertex
OpStore %out_pos_ptr %pos_end_plus
OpStore %outColor %color_end
OpEmitVertex
OpStore %out_pos_ptr %pos_end_minus
OpStore %outColor %color_end
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
