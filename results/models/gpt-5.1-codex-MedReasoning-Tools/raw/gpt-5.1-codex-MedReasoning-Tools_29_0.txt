Below is a complete C# program that combines fast I/O, parallel seeding heuristics, degeneracy ordering, and a Tomita-style branch-and-bound with greedy coloring and time-guards. It reads the graph from stdin, writes the maximum clique (size and vertices) to stdout, and automatically falls back to the best heuristic clique if the exact search would exceed the time budget.

```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Runtime.CompilerServices;
using System.Threading;
using System.Threading.Tasks;

public class Program
{
    const long TIME_LIMIT_MILLIS = 295000;          // keep a cushion under 5 minutes
    const int HEURISTIC_NEIGHBOR_CAP = 12000;       // cap neighbor pool per seed to avoid huge copies
    static Graph graph = null!;
    static readonly Stopwatch stopwatch = new Stopwatch();
    static List<int> bestClique = new List<int>();
    static int bestSize = 0;
    static readonly object bestLock = new object();
    static int vertexCount;

    public static void Main()
    {
        var scanner = new FastScanner();
        if (!scanner.TryReadInt(out vertexCount))
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }

        if (!scanner.TryReadInt(out int edgeCount))
        {
            Console.WriteLine(vertexCount > 0 ? "1" : "0");
            Console.WriteLine(vertexCount > 0 ? "0" : "");
            return;
        }

        var adjacency = new List<int>[vertexCount];
        for (int i = 0; i < edgeCount; i++)
        {
            if (!scanner.TryReadInt(out int u) || !scanner.TryReadInt(out int v))
                break;

            if (u < 0 || v < 0 || u >= vertexCount || v >= vertexCount || u == v)
                continue;

            adjacency[u] ??= new List<int>();
            adjacency[v] ??= new List<int>();
            adjacency[u].Add(v);
            adjacency[v].Add(u);
        }

        graph = new Graph(vertexCount, adjacency);
        stopwatch.Start();
        InitializeBest();

        if (vertexCount > 0)
        {
            RunHeuristics();
            if (TimeExceeded() || GetBestSize() >= vertexCount)
            {
                OutputSolution();
                return;
            }

            var degeneracyData = BuildDegeneracyData();
            if (GetBestSize() >= degeneracyData.MaxCore + 1 || TimeExceeded())
            {
                OutputSolution();
                return;
            }

            RunExactSearch(degeneracyData);
        }

        OutputSolution();
    }

    static void InitializeBest()
    {
        if (vertexCount == 0)
        {
            bestClique = new List<int>();
            bestSize = 0;
            return;
        }

        int bestVertex = 0;
        int bestDegree = graph.Degree[0];
        for (int i = 1; i < vertexCount; i++)
        {
            if (graph.Degree[i] > bestDegree)
            {
                bestDegree = graph.Degree[i];
                bestVertex = i;
            }
        }
        bestClique = new List<int> { bestVertex };
        bestSize = 1;
    }

    static void RunHeuristics()
    {
        var order = BuildDegreeOrder();
        TryUpdate(BuildCliqueGreedy(order));
        if (TimeExceeded()) return;

        int seedCount = Math.Min(order.Length, 512);
        if (seedCount == 0) return;

        var options = new ParallelOptions { MaxDegreeOfParallelism = Math.Min(Environment.ProcessorCount, 12) };
        Parallel.For(0, seedCount, options, (i, state) =>
        {
            if (TimeExceeded())
            {
                state.Stop();
                return;
            }

            var clique = BuildCliqueFromSeed(order[i]);
            TryUpdate(clique);
        });
    }

    static int[] BuildDegreeOrder()
    {
        int[] order = new int[vertexCount];
        for (int i = 0; i < vertexCount; i++) order[i] = i;
        Array.Sort(order, (a, b) =>
        {
            int cmp = graph.Degree[b].CompareTo(graph.Degree[a]);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });
        return order;
    }

    static List<int> BuildCliqueGreedy(int[] order)
    {
        var clique = new List<int>();
        foreach (int v in order)
        {
            if (ConnectsToAll(v, clique))
            {
                clique.Add(v);
                if (clique.Count == vertexCount)
                    break;
            }
        }
        return clique;
    }

    static List<int> BuildCliqueFromSeed(int seed)
    {
        var clique = new List<int> { seed };
        var candidate = GetCandidatePool(seed);
        if (candidate.Count == 0) return clique;

        candidate.Sort((a, b) =>
        {
            int cmp = graph.Degree[b].CompareTo(graph.Degree[a]);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });

        while (candidate.Count > 0 && !TimeExceeded())
        {
            int idx = candidate.Count - 1;
            int v = candidate[idx];
            candidate.RemoveAt(idx);

            if (!ConnectsToAll(v, clique))
                continue;

            clique.Add(v);
            if (clique.Count == vertexCount)
                break;

            candidate = IntersectCandidates(candidate, v);
            if (candidate.Count > 1)
            {
                candidate.Sort((a, b) =>
                {
                    int cmp = graph.Degree[b].CompareTo(graph.Degree[a]);
                    return cmp != 0 ? cmp : a.CompareTo(b);
                });
            }
        }

        return clique;
    }

    static List<int> GetCandidatePool(int seed)
    {
        var neighbors = graph.Neighbors[seed];
        int length = neighbors.Length;
        if (length == 0) return new List<int>();
        if (length <= HEURISTIC_NEIGHBOR_CAP)
            return new List<int>(neighbors);

        int step = Math.Max(1, length / HEURISTIC_NEIGHBOR_CAP);
        var pool = new List<int>(HEURISTIC_NEIGHBOR_CAP);
        for (int i = 0; i < length && pool.Count < HEURISTIC_NEIGHBOR_CAP; i += step)
        {
            pool.Add(neighbors[i]);
        }
        // ensure some tail elements are included
        for (int i = length - 1; i >= 0 && pool.Count < HEURISTIC_NEIGHBOR_CAP; i--)
        {
            pool.Add(neighbors[i]);
        }
        return pool;
    }

    static List<int> IntersectCandidates(List<int> candidates, int vertex)
    {
        var result = new List<int>(candidates.Count);
        var neighbors = graph.Neighbors[vertex];
        foreach (int u in candidates)
        {
            if (ContainsSorted(neighbors, u))
                result.Add(u);
        }
        return result;
    }

    static void RunExactSearch(DegeneracyData data)
    {
        var order = data.Order;
        int n = order.Length;
        var rank = new int[n];
        for (int i = 0; i < n; i++) rank[order[i]] = i;

        var current = new List<int>();
        for (int i = 0; i < n && !TimeExceeded(); i++)
        {
            int v = order[i];
            if (graph.Degree[v] + 1 <= GetBestSize())
                continue;

            var candidates = new List<int>();
            var neighbors = graph.Neighbors[v];
            int threshold = GetBestSize() - 1;
            for (int j = 0; j < neighbors.Length; j++)
            {
                int u = neighbors[j];
                if (rank[u] > rank[v] && graph.Degree[u] >= threshold)
                    candidates.Add(u);
            }

            if (candidates.Count + 1 <= GetBestSize())
                continue;

            current.Clear();
            current.Add(v);
            Expand(candidates, current);
        }
    }

    static void Expand(List<int> candidates, List<int> current)
    {
        if (TimeExceeded())
            return;

        if (candidates.Count == 0)
        {
            TryUpdate(current);
            return;
        }

        if (current.Count + candidates.Count <= GetBestSize())
        {
            TryUpdate(current);
            return;
        }

        var colorData = Colorize(candidates);
        var order = colorData.Order;
        var bounds = colorData.Bounds;

        for (int i = order.Length - 1; i >= 0; i--)
        {
            if (TimeExceeded())
                return;

            if (current.Count + bounds[i] <= GetBestSize())
                return;

            int v = order[i];
            current.Add(v);

            var newCandidates = new List<int>(i);
            for (int j = 0; j < i; j++)
            {
                int u = order[j];
                if (AreAdjacent(v, u))
                    newCandidates.Add(u);
            }

            if (newCandidates.Count == 0)
            {
                TryUpdate(current);
            }
            else if (current.Count + newCandidates.Count <= GetBestSize())
            {
                TryUpdate(current);
            }
            else
            {
                Expand(newCandidates, current);
            }

            current.RemoveAt(current.Count - 1);
        }
    }

    static ColorData Colorize(List<int> candidates)
    {
        int m = candidates.Count;
        int[] order = new int[m];
        int[] bounds = new int[m];

        var temp = new List<int>(candidates);
        var next = new List<int>(temp.Count);
        var colorClass = new List<int>();
        int idx = 0;
        int color = 0;

        while (temp.Count > 0)
        {
            color++;
            colorClass.Clear();
            next.Clear();

            for (int i = temp.Count - 1; i >= 0; i--)
            {
                int v = temp[i];
                bool ok = true;
                for (int j = 0; j < colorClass.Count; j++)
                {
                    if (!AreAdjacent(v, colorClass[j]))
                    {
                        ok = false;
                        break;
                    }
                }

                if (ok)
                {
                    order[idx] = v;
                    bounds[idx] = color;
                    idx++;
                    colorClass.Add(v);
                }
                else
                {
                    next.Add(v);
                }
            }

            var swap = temp;
            temp = next;
            next = swap;
        }

        return new ColorData(order, bounds);
    }

    static DegeneracyData BuildDegeneracyData()
    {
        int n = vertexCount;
        var degree = new int[n];
        int maxDegree = 0;
        for (int i = 0; i < n; i++)
        {
            degree[i] = graph.Degree[i];
            if (degree[i] > maxDegree) maxDegree = degree[i];
        }

        int[] head = new int[maxDegree + 1];
        Array.Fill(head, -1);
        int[] next = new int[n];
        int[] prev = new int[n];

        for (int v = 0; v < n; v++)
        {
            int d = degree[v];
            next[v] = head[d];
            if (head[d] != -1) prev[head[d]] = v;
            prev[v] = -1;
            head[d] = v;
        }

        bool[] removed = new bool[n];
        int[] order = new int[n];
        int idx = 0;
        int remaining = n;
        int mindeg = 0;
        int maxCore = 0;

        while (remaining > 0)
        {
            while (mindeg <= maxDegree && head[mindeg] == -1) mindeg++;
            if (mindeg > maxDegree) break;

            int v = head[mindeg];
            RemoveFromBucket(v, degree[v], head, next, prev);
            removed[v] = true;
            order[idx++] = v;
            if (degree[v] > maxCore) maxCore = degree[v];
            remaining--;

            var neighbors = graph.Neighbors[v];
            for (int i = 0; i < neighbors.Length; i++)
            {
                int u = neighbors[i];
                if (removed[u]) continue;
                int du = degree[u];
                RemoveFromBucket(u, du, head, next, prev);
                degree[u] = du - 1;
                InsertIntoBucket(u, du - 1, head, next, prev);
                if (du - 1 < mindeg) mindeg = du - 1;
            }
        }

        if (idx < n)
        {
            for (int v = 0; v < n; v++)
                if (!removed[v]) order[idx++] = v;
        }

        return new DegeneracyData(order, maxCore);
    }

    static void RemoveFromBucket(int vertex, int degree, int[] head, int[] next, int[] prev)
    {
        int pv = prev[vertex];
        int nv = next[vertex];
        if (pv == -1)
            head[degree] = nv;
        else
            next[pv] = nv;
        if (nv != -1)
            prev[nv] = pv;
        prev[vertex] = -1;
        next[vertex] = -1;
    }

    static void InsertIntoBucket(int vertex, int degree, int[] head, int[] next, int[] prev)
    {
        if (degree < 0) degree = 0;
        int h = head[degree];
        if (h != -1) prev[h] = vertex;
        next[vertex] = h;
        prev[vertex] = -1;
        head[degree] = vertex;
    }

    static bool ConnectsToAll(int vertex, List<int> clique)
    {
        for (int i = 0; i < clique.Count; i++)
        {
            if (!AreAdjacent(vertex, clique[i]))
                return false;
        }
        return true;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    static bool AreAdjacent(int u, int v)
    {
        var neighborsU = graph.Neighbors[u];
        var neighborsV = graph.Neighbors[v];
        if (neighborsU.Length < neighborsV.Length)
            return ContainsSorted(neighborsU, v);
        return ContainsSorted(neighborsV, u);
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    static bool ContainsSorted(int[] array, int target)
    {
        int left = 0;
        int right = array.Length - 1;
        while (left <= right)
        {
            int mid = (left + right) >> 1;
            int value = array[mid];
            if (value == target) return true;
            if (value < target) left = mid + 1;
            else right = mid - 1;
        }
        return false;
    }

    static void TryUpdate(List<int> candidate)
    {
        if (candidate == null) return;
        int count = candidate.Count;
        if (count <= GetBestSize()) return;

        lock (bestLock)
        {
            if (count > bestSize)
            {
                bestSize = count;
                bestClique = new List<int>(candidate);
            }
        }
    }

    static int GetBestSize() => Volatile.Read(ref bestSize);

    static bool TimeExceeded() => stopwatch.ElapsedMilliseconds >= TIME_LIMIT_MILLIS;

    static void OutputSolution()
    {
        List<int> result;
        lock (bestLock)
        {
            result = new List<int>(bestClique);
        }
        result.Sort();
        Console.WriteLine(result.Count);
        if (result.Count > 0)
            Console.WriteLine(string.Join(" ", result));
        else
            Console.WriteLine();
    }

    sealed class Graph
    {
        public int VertexCount { get; }
        public int[][] Neighbors { get; }
        public int[] Degree { get; }

        public Graph(int vertexCount, List<int>[] adjacency)
        {
            VertexCount = vertexCount;
            Neighbors = new int[vertexCount][];
            Degree = new int[vertexCount];

            for (int i = 0; i < vertexCount; i++)
            {
                var list = adjacency[i];
                if (list == null || list.Count == 0)
                {
                    Neighbors[i] = Array.Empty<int>();
                    Degree[i] = 0;
                    continue;
                }

                list.Sort();
                int prev = int.MinValue;
                var dedup = new List<int>(list.Count);
                foreach (int v in list)
                {
                    if (v == i || v == prev) continue;
                    dedup.Add(v);
                    prev = v;
                }

                Neighbors[i] = dedup.ToArray();
                Degree[i] = Neighbors[i].Length;
            }
        }
    }

    sealed class FastScanner
    {
        readonly System.IO.Stream stream = Console.OpenStandardInput();
        readonly byte[] buffer = new byte[1 << 16];
        int len, ptr;

        byte Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return 0;
            }
            return buffer[ptr++];
        }

        public bool TryReadInt(out int value)
        {
            value = 0;
            int c;
            do
            {
                c = Read();
                if (c == 0) return false;
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int result = 0;
            while (c > ' ')
            {
                result = result * 10 + (c - '0');
                c = Read();
            }
            value = result * sign;
            return true;
        }
    }

    readonly struct DegeneracyData
    {
        public readonly int[] Order;
        public readonly int MaxCore;
        public DegeneracyData(int[] order, int maxCore)
        {
            Order = order;
            MaxCore = maxCore;
        }
    }

    readonly struct ColorData
    {
        public readonly int[] Order;
        public readonly int[] Bounds;
        public ColorData(int[] order, int[] bounds)
        {
            Order = order;
            Bounds = bounds;
        }
    }
}
```

**Key features**

- **Fast input** via a buffered scanner keeps startup overhead low.
- **Parallel seeding heuristics** (greedy and multi-start greedy with neighbor sampling) quickly find strong initial cliques and exploit all CPU cores.
- **Degeneracy ordering** plus a Tomita-style branch-and-bound with greedy coloring provides an exact search path with strong pruning; the search halts early only if time would exceed the budget, in which case the best verified clique so far is returned.
- **Bounds-aware pruning** (degree tests, coloring bounds, candidate-size limits) lets the same binary handle sparse and very dense graphs, from tens to millions of vertices, within the 5-minute requirement.
