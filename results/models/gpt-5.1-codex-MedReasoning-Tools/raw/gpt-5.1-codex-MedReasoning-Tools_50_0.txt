```glsl
#version 450
layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;   // counts.x = N
    vec4 radii;     // x=dt, y=sepRadius, z=alignRadius, w=cohRadius
    vec4 weights;   // x=sepWeight, y=alignWeight, z=cohWeight, w=maxSpeed
    vec4 bounds;    // x=boundSize, y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint boidCount = counts.x;
    if (i >= boidCount) {
        return;
    }

    uint baseIndex = i * 2u;
    vec3 pos = inData[baseIndex].xyz;
    vec3 vel = inData[baseIndex + 1u].xyz;

    float dt          = radii.x;
    float sepRadius   = radii.y;
    float alignRadius = radii.z;
    float cohRadius   = radii.w;

    float sepWeight   = weights.x;
    float alignWeight = weights.y;
    float cohWeight   = weights.z;
    float maxSpeed    = weights.w;

    float boundSize   = bounds.x;
    float boundForce  = bounds.y;

    float sepRadiusSq   = sepRadius * sepRadius;
    float alignRadiusSq = alignRadius * alignRadius;
    float cohRadiusSq   = cohRadius * cohRadius;

    vec3 sepAccum = vec3(0.0);
    vec3 alignAccum = vec3(0.0);
    vec3 cohAccum = vec3(0.0);
    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;

    for (uint j = 0u; j < boidCount; ++j) {
        if (j == i) {
            continue;
        }

        uint otherIndex = j * 2u;
        vec3 otherPos = inData[otherIndex].xyz;
        vec3 offset = otherPos - pos;
        float distSq = dot(offset, offset);
        if (distSq == 0.0) {
            continue;
        }

        if (distSq < sepRadiusSq) {
            float dist = sqrt(distSq);
            sepAccum += (pos - otherPos) / dist;
            ++sepCount;
        }

        if (distSq < alignRadiusSq) {
            vec3 otherVel = inData[otherIndex + 1u].xyz;
            alignAccum += otherVel;
            ++alignCount;
        }

        if (distSq < cohRadiusSq) {
            cohAccum += otherPos;
            ++cohCount;
        }
    }

    vec3 sepForce = (sepCount > 0) ? (sepAccum / float(sepCount)) : vec3(0.0);
    vec3 alignForce = vec3(0.0);
    if (alignCount > 0) {
        vec3 avgVel = alignAccum / float(alignCount);
        alignForce = avgVel - vel;
    }

    vec3 cohForce = vec3(0.0);
    if (cohCount > 0) {
        vec3 center = cohAccum / float(cohCount);
        cohForce = center - pos;
    }

    vec3 accel = sepForce * sepWeight + alignForce * alignWeight + cohForce * cohWeight;

    float innerLimit = boundSize * 0.4;
    float outerLimit = boundSize * 0.5;
    float limitRange = max(outerLimit - innerLimit, 0.0001);

    for (int axis = 0; axis < 3; ++axis) {
        float component = pos[axis];
        float absComp = abs(component);
        if (absComp > innerLimit) {
            float t = clamp((absComp - innerLimit) / limitRange, 0.0, 1.0);
            float direction = component > 0.0 ? -1.0 : 1.0;
            accel[axis] += direction * boundForce * t;
        }
    }

    vec3 newVel = vel + accel * dt;
    float speed = length(newVel);
    if (speed > maxSpeed) {
        newVel *= maxSpeed / speed;
    }

    vec3 newPos = pos + newVel * dt;

    outData[baseIndex] = vec4(newPos, 0.0);
    outData[baseIndex + 1u] = vec4(newVel, 0.0);
}
```
