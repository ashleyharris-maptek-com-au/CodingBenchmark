```rust
use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read};

#[derive(Clone)]
struct Sample {
    hole_idx: usize,
    sample_idx: usize,
    point: [f64; 3],
}

#[derive(Clone, Copy, Hash, Eq, PartialEq)]
struct CellKey(i64, i64, i64);

struct GridIndex {
    cell_size: f64,
    min: [f64; 3],
    max_radius: i32,
    cells: HashMap<CellKey, Vec<usize>>,
}

impl GridIndex {
    fn new(samples: &[Sample], cell_size: f64, min: [f64; 3], max_radius: i32) -> Self {
        let mut cells: HashMap<CellKey, Vec<usize>> = HashMap::with_capacity(samples.len().max(1));
        for (idx, sample) in samples.iter().enumerate() {
            let key = Self::cell_coords_from_point(&sample.point, &min, cell_size);
            cells.entry(key).or_default().push(idx);
        }
        Self {
            cell_size,
            min,
            max_radius,
            cells,
        }
    }

    fn cell_coords_from_point(point: &[f64; 3], min: &[f64; 3], cell_size: f64) -> CellKey {
        fn coord(value: f64, min: f64, size: f64) -> i64 {
            let raw = ((value - min) / size).floor();
            if !raw.is_finite() {
                0
            } else if raw > i64::MAX as f64 {
                i64::MAX
            } else if raw < i64::MIN as f64 {
                i64::MIN
            } else {
                raw as i64
            }
        }
        CellKey(
            coord(point[0], min[0], cell_size),
            coord(point[1], min[1], cell_size),
            coord(point[2], min[2], cell_size),
        )
    }

    fn gather(
        &self,
        samples: &[Sample],
        target_idx: usize,
        desired: usize,
        buffer: &mut Vec<(usize, f64)>,
    ) {
        buffer.clear();
        if samples.is_empty() {
            return;
        }
        let center = Self::cell_coords_from_point(&samples[target_idx].point, &self.min, self.cell_size);
        let target_point = samples[target_idx].point;
        for radius in 0..=self.max_radius {
            let radius_i64 = radius as i64;
            for dx in -radius_i64..=radius_i64 {
                for dy in -radius_i64..=radius_i64 {
                    for dz in -radius_i64..=radius_i64 {
                        if radius == 0 {
                            if dx != 0 || dy != 0 || dz != 0 {
                                continue;
                            }
                        } else {
                            let max_component = dx.abs().max(dy.abs()).max(dz.abs());
                            if max_component != radius_i64 {
                                continue;
                            }
                        }
                        let key = CellKey(center.0 + dx, center.1 + dy, center.2 + dz);
                        if let Some(indices) = self.cells.get(&key) {
                            for &idx in indices {
                                if idx == target_idx {
                                    continue;
                                }
                                let dist2 = distance_sq(&target_point, &samples[idx].point);
                                buffer.push((idx, dist2));
                            }
                        }
                    }
                }
            }
            if buffer.len() >= desired {
                break;
            }
        }
        buffer.sort_by(|a, b| {
            a.1.partial_cmp(&b.1)
                .unwrap_or(Ordering::Equal)
        });
        if buffer.len() > desired {
            buffer.truncate(desired);
        }
    }
}

#[derive(Clone)]
struct Suspect {
    hole_idx: usize,
    sample_idx: usize,
    property_idx: usize,
    confidence: f64,
}

fn parse_optional_f64(token: &str) => Option<f64> {
    match token {
        "NA" | "NaN" | "nan" | "NAN" | "NULL" | "null" | "-" => None,
        _ => token.parse::<f64>().ok(),
    }
}

fn read_f64_or(token: Option<&str>, default: f64) -> f64 {
    token
        .and_then(parse_optional_f64)
        .unwrap_or(default)
}

fn read_f64_or_nan(token: Option<&str>) -> f64 {
    token
        .and_then(parse_optional_f64)
        .unwrap_or(f64::NAN)
}

fn distance_sq(a: &[f64; 3], b: &[f64; 3]) -> f64 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

fn median_sorted(values: &[f64]) -> f64 {
    if values.is_empty() {
        return f64::NAN;
    }
    let mid = values.len() / 2;
    if values.len() % 2 == 1 {
        values[mid]
    } else {
        (values[mid - 1] + values[mid]) * 0.5
    }
}

fn gather_neighbors_brute(
    samples: &[Sample],
    target_idx: usize,
    desired: usize,
    buffer: &mut Vec<(usize, f64)>,
) {
    buffer.clear();
    if samples.is_empty() {
        return;
    }
    let point = samples[target_idx].point;
    for (idx, other) in samples.iter().enumerate() {
        if idx == target_idx {
            continue;
        }
        let d2 = distance_sq(&point, &other.point);
        buffer.push((idx, d2));
    }
    buffer.sort_by(|a, b| {
        a.1.partial_cmp(&b.1)
            .unwrap_or(Ordering::Equal)
    });
    if buffer.len() > desired {
        buffer.truncate(desired);
    }
}

fn sanitize_coord(v: f64) -> f64 {
    if v.is_finite() {
        v
    } else {
        0.0
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        println!("0");
        return;
    }
    let mut tokens = input.split_whitespace();
    let num_holes = tokens
        .next()
        .and_then(|t| t.parse::<usize>().ok())
        .unwrap_or(0);
    let num_properties = tokens
        .next()
        .and_then(|t| t.parse::<usize>().ok())
        .unwrap_or(0);

    if num_properties == 0 {
        println!("0");
        return;
    }

    let mut property_names = Vec::with_capacity(num_properties);
    for idx in 0..num_properties {
        let name = tokens
            .next()
            .map(|s| s.to_string())
            .unwrap_or_else(|| format!("property_{}", idx + 1));
        property_names.push(name);
    }

    let mut hole_ids: Vec<String> = Vec::with_capacity(num_holes.max(1));
    let mut samples: Vec<Sample> = Vec::new();
    let mut values: Vec<f64> = Vec::new();

    let mut min_pt = [f64::INFINITY; 3];
    let mut max_pt = [f64::NEG_INFINITY; 3];

    let mut depth_step_sum = 0.0;
    let mut depth_step_count: usize = 0;
    let mut total_length_sum = 0.0;

    for hole_idx in 0..num_holes {
        let hole_id = tokens
            .next()
            .map(|s| s.to_string())
            .unwrap_or_else(|| format!("hole{}", hole_idx + 1));
        hole_ids.push(hole_id);

        let start_x = read_f64_or(tokens.next(), 0.0);
        let start_y = read_f64_or(tokens.next(), 0.0);
        let start_z = read_f64_or(tokens.next(), 0.0);

        let dir_x = read_f64_or(tokens.next(), 0.0);
        let dir_y = read_f64_or(tokens.next(), 0.0);
        let dir_z = read_f64_or(tokens.next(), 1.0);

        let length = read_f64_or(tokens.next(), 0.0).max(0.0);
        total_length_sum += length;

        let num_samples_in_hole = tokens
            .next()
            .and_then(|t| t.parse::<usize>().ok())
            .unwrap_or(0);

        let dir_len = (dir_x * dir_x + dir_y * dir_y + dir_z * dir_z).sqrt();
        let dir_unit = if dir_len > 0.0 {
            [dir_x / dir_len, dir_y / dir_len, dir_z / dir_len]
        } else {
            [0.0, 0.0, 0.0]
        };

        let mut prev_depth: Option<f64> = None;

        for sample_idx in 0..num_samples_in_hole {
            let depth = read_f64_or(tokens.next(), 0.0).clamp(0.0, length.max(0.0));
            if let Some(prev) = prev_depth {
                if depth >= prev {
                    depth_step_sum += depth - prev;
                    depth_step_count += 1;
                }
            }
            prev_depth = Some(depth);

            let px = sanitize_coord(start_x + dir_unit[0] * depth);
            let py = sanitize_coord(start_y + dir_unit[1] * depth);
            let pz = sanitize_coord(start_z + dir_unit[2] * depth);

            for axis in 0..3 {
                min_pt[axis] = min_pt[axis].min([px, py, pz][axis]);
                max_pt[axis] = max_pt[axis].max([px, py, pz][axis]);
            }

            samples.push(Sample {
                hole_idx,
                sample_idx,
                point: [px, py, pz],
            });

            for _ in 0..num_properties {
                let value = read_f64_or_nan(tokens.next());
                values.push(value);
            }
        }
    }

    let total_samples = samples.len();
    if total_samples == 0 {
        println!("0");
        return;
    }

    let diag = if min_pt.iter().all(|v| v.is_finite()) && max_pt.iter().all(|v| v.is_finite()) {
        let dx = max_pt[0] - min_pt[0];
        let dy = max_pt[1] - min_pt[1];
        let dz = max_pt[2] - min_pt[2];
        (dx * dx + dy * dy + dz * dz).sqrt()
    } else {
        0.0
    };

    let mut typical_spacing = if depth_step_count > 0 {
        depth_step_sum / depth_step_count as f64
    } else {
        0.0
    };

    if typical_spacing <= 0.0 {
        if diag > 0.0 {
            typical_spacing = diag / (total_samples as f64).cbrt().max(1.0);
        } else if total_length_sum > 0.0 && total_samples > 0 {
            typical_spacing = total_length_sum / total_samples as f64;
        }
    }
    if typical_spacing <= 0.0 || !typical_spacing.is_finite() {
        typical_spacing = 1.0;
    }

    let mut cell_size = typical_spacing * 3.0;
    if cell_size <= 0.0 || !cell_size.is_finite() {
        cell_size = 1.0;
    }

    let desired_neighbors = if total_samples <= 5_000 {
        8
    } else if total_samples <= 50_000 {
        16
    } else {
        24
    };
    let neighbor_limit = desired_neighbors * 2;
    let min_neighbors = desired_neighbors / 2 + 1;

    let base_threshold = if total_samples <= 5_000 {
        2.5
    } else if total_samples <= 50_000 {
        3.0
    } else if total_samples <= 200_000 {
        3.4
    } else {
        3.8
    };

    let neighbor_strategy = if total_samples <= 5_000 {
        None
    } else {
        let max_radius = if total_samples <= 50_000 {
            3
        } else if total_samples <= 200_000 {
            4
        } else {
            5
        };
        Some(GridIndex::new(
            &samples,
            cell_size.max(1e-6),
            min_pt,
            max_radius,
        ))
    };

    let mut neighbor_buffer: Vec<(usize, f64)> = Vec::with_capacity(neighbor_limit * 2);
    let mut neighbor_values: Vec<f64> = Vec::with_capacity(neighbor_limit);
    let mut deviation_buffer: Vec<f64> = Vec::with_capacity(neighbor_limit);

    let mut suspects: Vec<Suspect> = Vec::new();

    for (sample_idx, sample) in samples.iter().enumerate() {
        if let Some(grid) = &neighbor_strategy {
            grid.gather(&samples, sample_idx, neighbor_limit, &mut neighbor_buffer);
        } else {
            gather_neighbors_brute(&samples, sample_idx, neighbor_limit, &mut neighbor_buffer);
        }

        if neighbor_buffer.len() < min_neighbors {
            continue;
        }
        let usable_neighbors = neighbor_buffer.len().min(neighbor_limit);

        for prop_idx in 0..num_properties {
            let sample_value = values[sample_idx * num_properties + prop_idx];
            if !sample_value.is_finite() {
                continue;
            }

            neighbor_values.clear();
            for &(n_idx, _) in neighbor_buffer.iter().take(usable_neighbors) {
                let v = values[n_idx * num_properties + prop_idx];
                if v.is_finite() {
                    neighbor_values.push(v);
                }
            }

            if neighbor_values.len() < min_neighbors {
                continue;
            }

            neighbor_values.sort_by(|a, b| {
                a.partial_cmp(b)
                    .unwrap_or(Ordering::Equal)
            });
            let median = median_sorted(&neighbor_values);

            deviation_buffer.clear();
            for &v in &neighbor_values {
                deviation_buffer.push((v - median).abs());
            }
            deviation_buffer.sort_by(|a, b| {
                a.partial_cmp(b)
                    .unwrap_or(Ordering::Equal)
            });
            let mad = median_sorted(&deviation_buffer);
            let tolerance = 1e-9.max(median.abs() * 1e-6);
            let scaled_mad = (mad * 1.4826).max(tolerance);
            let diff = sample_value - median;
            let abs_diff = diff.abs();
            if abs_diff <= tolerance * 4.0 {
                continue;
            }

            let neighbor_factor =
                neighbor_values.len() as f64 / (neighbor_values.len() as f64 + 2.0);
            let confidence = (abs_diff / scaled_mad) * neighbor_factor;
            if confidence.is_finite() && confidence >= base_threshold {
                suspects.push(Suspect {
                    hole_idx: sample.hole_idx,
                    sample_idx: sample.sample_idx,
                    property_idx: prop_idx,
                    confidence,
                });
            }
        }
    }

    suspects.sort_by(|a, b| {
        b.confidence
            .partial_cmp(&a.confidence)
            .unwrap_or(Ordering::Equal)
            .then_with(|| hole_ids[a.hole_idx].cmp(&hole_ids[b.hole_idx]))
            .then_with(|| a.sample_idx.cmp(&b.sample_idx))
            .then(a.property_idx.cmp(&b.property_idx))
    });

    println!("{}", suspects.len());
    for suspect in suspects {
        println!(
            "{} {} {} {:.4}",
            hole_ids[suspect.hole_idx],
            suspect.sample_idx,
            property_names[suspect.property_idx],
            suspect.confidence
        );
    }
}
```

**How it works (high level):**

1. **Parsing:** Reads the structured drillhole input, builds per-sample 3‑D coordinates from start point + downhole depth along the (normalized) hole direction, and stores all property values (handling missing entries as `NaN`).

2. **Adaptive neighbor search:**
   - For datasets ≤ 5,000 samples, it uses a brute-force distance scan (accurate but only feasible for smaller inputs).
   - Beyond that, it constructs a sparse 3‑D hash grid whose cell size scales with the observed sampling interval. Neighbor searches expand cell “shells” until enough nearby samples are gathered, which keeps lookups quasi-linear even for hundreds of thousands of samples.

3. **Outlier scoring:** For each property at each sample:
   - Collects nearby measurements, computes the local median and MAD (median absolute deviation) to stay robust against skewed distributions.
   - Converts the deviation of the current value from the neighborhood median into a pseudo‑z score, damped when few neighbors are available.
   - Flags entries whose confidence exceeds a scale-dependent threshold, ensuring only the most obvious typos or measurement errors surface.

4. **Output:** Emits the suspect list sorted by descending confidence, reporting hole ID, sample index, property name, and confidence score.
