use std::io::{self, Read};

fn ensure_fixed(part: &mut [u8], fixed: usize) {
    if part[fixed] == 0 {
        for val in part.iter_mut() {
            *val ^= 1;
        }
    }
}

fn compute_cut_u8(part: &[u8], adj: &[Vec<i64>]) -> i64 {
    let n = part.len();
    let mut cut = 0i64;
    for u in 0..n {
        for v in (u + 1)..n {
            let w = adj[u][v];
            if w != 0 && part[u] != part[v] {
                cut += w;
            }
        }
    }
    cut
}

fn compute_cut_i8(part: &[i8], adj: &[Vec<i64>]) -> i64 {
    let n = part.len();
    let mut cut = 0i64;
    for u in 0..n {
        for v in (u + 1)..n {
            let w = adj[u][v];
            if w != 0 && part[u] != part[v] {
                cut += w;
            }
        }
    }
    cut
}

fn kernighan_lin(part: &mut Vec<u8>, adj: &[Vec<i64>]) {
    let n = part.len();
    if n < 2 {
        return;
    }
    loop {
        let mut d = vec![0i64; n];
        for v in 0..n {
            let mut internal = 0i64;
            let mut external = 0i64;
            for u in 0..n {
                let w = adj[v][u];
                if w == 0 {
                    continue;
                }
                if part[u] == part[v] {
                    internal += w;
                } else {
                    external += w;
                }
            }
            d[v] = external - internal;
        }
        let mut used = vec![false; n];
        let mut gains = Vec::new();
        let mut pairs = Vec::new();
        for _ in 0..(n / 2) {
            let mut best_gain = i64::MIN;
            let mut best_pair = (usize::MAX, usize::MAX);
            let mut found = false;
            for a in 0..n {
                if part[a] != 1 || used[a] {
                    continue;
                }
                for b in 0..n {
                    if part[b] != 0 || used[b] {
                        continue;
                    }
                    let w_ab = adj[a][b];
                    let gain = d[a] + d[b] - 2 * w_ab;
                    if gain > best_gain {
                        best_gain = gain;
                        best_pair = (a, b);
                        found = true;
                    }
                }
            }
            if !found {
                break;
            }
            let (a, b) = best_pair;
            used[a] = true;
            used[b] = true;
            gains.push(best_gain);
            pairs.push((a, b));
            for k in 0..n {
                if used[k] {
                    continue;
                }
                let wka = adj[k][a];
                let wkb = adj[k][b];
                if part[k] == 1 {
                    d[k] += 2 * wka;
                    d[k] -= 2 * wkb;
                } else {
                    d[k] += 2 * wkb;
                    d[k] -= 2 * wka;
                }
            }
        }
        if gains.is_empty() {
            break;
        }
        let mut best_total = i64::MIN;
        let mut best_k = 0usize;
        let mut sum = 0i64;
        for (i, gain) in gains.iter().enumerate() {
            sum += *gain;
            if sum > best_total {
                best_total = sum;
                best_k = i + 1;
            }
        }
        if best_total > 0 {
            for i in 0..best_k {
                let (a, b) = pairs[i];
                part[a] = 0;
                part[b] = 1;
            }
        } else {
            break;
        }
    }
}

struct Solver {
    n: usize,
    target: usize,
    neighbors: Vec<Vec<(usize, i64)>>,
    order: Vec<usize>,
    assignment: Vec<i8>,
    best_assignment: Vec<i8>,
    best_cut: i64,
}

impl Solver {
    fn dfs(&mut self, idx: usize, current_cut: i64, count_a: usize, count_b: usize) {
        if current_cut >= self.best_cut || count_a > self.target || count_b > self.target {
            return;
        }
        if idx == self.order.len() {
            if count_a == self.target && count_b == self.target {
                self.best_cut = current_cut;
                self.best_assignment = self.assignment.clone();
            }
            return;
        }

        let remaining = self.order.len() - idx;
        let need_a = self.target - count_a;
        let need_b = self.target - count_b;
        if need_a > remaining || need_b > remaining {
            return;
        }

        let lower = self.lower_bound(idx, count_a, current_cut);
        if lower >= self.best_cut {
            return;
        }

        let v = self.order[idx];

        if need_a == remaining {
            self.assign_and_recurse(v, 1, idx, current_cut, count_a, count_b);
            return;
        }
        if need_b == remaining {
            self.assign_and_recurse(v, 0, idx, current_cut, count_a, count_b);
            return;
        }

        let mut options: Vec<(i64, i8)> = Vec::with_capacity(2);
        if count_a < self.target {
            options.push((self.delta(v, 1), 1));
        }
        if count_b < self.target {
            options.push((self.delta(v, 0), 0));
        }
        options.sort_by_key(|&(delta, _)| delta);

        for (delta, part) in options {
            let new_cut = current_cut + delta;
            if new_cut >= self.best_cut {
                continue;
            }
            self.assignment[v] = part;
            let (na, nb) = if part == 1 {
                (count_a + 1, count_b)
            } else {
                (count_a, count_b + 1)
            };
            self.dfs(idx + 1, new_cut, na, nb);
            self.assignment[v] = -1;
        }
    }

    fn assign_and_recurse(
        &mut self,
        v: usize,
        part: i8,
        idx: usize,
        current_cut: i64,
        count_a: usize,
        count_b: usize,
    ) {
        let delta = self.delta(v, part);
        let new_cut = current_cut + delta;
        if new_cut >= self.best_cut {
            return;
        }
        self.assignment[v] = part;
        let (na, nb) = if part == 1 {
            (count_a + 1, count_b)
        } else {
            (count_a, count_b + 1)
        };
        self.dfs(idx + 1, new_cut, na, nb);
        self.assignment[v] = -1;
    }

    fn delta(&self, v: usize, part: i8) -> i64 {
        let mut delta = 0i64;
        for &(nbr, weight) in &self.neighbors[v] {
            let assigned = self.assignment[nbr];
            if assigned == -1 {
                continue;
            }
            if assigned != part {
                delta += weight;
            }
        }
        delta
    }

    fn lower_bound(&self, idx: usize, count_a: usize, current_cut: i64) -> i64 {
        let remaining = self.order.len() - idx;
        if remaining == 0 {
            return current_cut;
        }
        let mut sum_cost_b = 0i64;
        let mut diffs = Vec::with_capacity(remaining);
        for t in idx..self.order.len() {
            let v = self.order[t];
            let mut to_a = 0i64;
            let mut to_b = 0i64;
            for &(nbr, weight) in &self.neighbors[v] {
                match self.assignment[nbr] {
                    1 => to_a += weight,
                    0 => to_b += weight,
                    _ => {}
                }
            }
            sum_cost_b += to_a;
            diffs.push(to_b - to_a);
        }
        diffs.sort_unstable();
        let need_a = self.target - count_a;
        if need_a > diffs.len() {
            return i64::MAX / 4;
        }
        let mut extra = sum_cost_b;
        for i in 0..need_a {
            extra += diffs[i];
        }
        current_cut + extra
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }
    let mut iter = input.split_whitespace();
    let n: usize = iter.next().unwrap().parse().unwrap();
    let m: usize = iter.next().unwrap().parse().unwrap();

    if n == 0 {
        println!("0");
        println!();
        return;
    }

    let mut adj = vec![vec![0i64; n]; n];
    for _ in 0..m {
        let u: usize = iter.next().unwrap().parse().unwrap();
        let v: usize = iter.next().unwrap().parse().unwrap();
        if u >= n || v >= n || u == v {
            continue;
        }
        adj[u][v] += 1;
        adj[v][u] += 1;
    }

    let mut neighbors = vec![Vec::<(usize, i64)>::new(); n];
    let mut degrees = vec![0i64; n];
    for u in 0..n {
        for v in 0..n {
            let w = adj[u][v];
            if w != 0 {
                neighbors[u].push((v, w));
                degrees[u] += w;
            }
        }
    }

    let mut order_by_degree: Vec<usize> = (0..n).collect();
    order_by_degree.sort_by(|&a, &b| degrees[b].cmp(&degrees[a]).then_with(|| a.cmp(&b)));
    let fixed = order_by_degree[0];
    let target = n / 2;

    let mut candidate_orders: Vec<Vec<usize>> = Vec::new();
    candidate_orders.push(order_by_degree.clone());
    candidate_orders.push((0..n).collect::<Vec<_>>());
    let mut reversed = order_by_degree.clone();
    reversed.reverse();
    candidate_orders.push(reversed);
    let mut parity = Vec::with_capacity(n);
    let mut even = 0;
    while even < n {
        parity.push(even);
        even += 2;
    }
    let mut odd = 1;
    while odd < n {
        parity.push(odd);
        odd += 2;
    }
    candidate_orders.push(parity);

    let mut best_cut = i64::MAX;
    let mut best_assignment = vec![0i8; n];

    for order in candidate_orders {
        let mut part = vec![0u8; n];
        for &v in order.iter().take(target) {
            part[v] = 1;
        }
        ensure_fixed(&mut part, fixed);
        kernighan_lin(&mut part, &adj);
        ensure_fixed(&mut part, fixed);
        let cut = compute_cut_u8(&part, &adj);
        if cut < best_cut {
            best_cut = cut;
            best_assignment = part.iter().map(|&x| x as i8).collect();
        }
    }

    if best_cut == i64::MAX {
        let mut part = vec![0u8; n];
        for &v in order_by_degree.iter().take(target) {
            part[v] = 1;
        }
        ensure_fixed(&mut part, fixed);
        best_cut = compute_cut_u8(&part, &adj);
        best_assignment = part.iter().map(|&x| x as i8).collect();
    }

    let mut assignment = vec![-1i8; n];
    assignment[fixed] = 1;

    let mut order = order_by_degree.clone();
    order.retain(|v| *v != fixed);

    let mut solver = Solver {
        n,
        target,
        neighbors,
        order,
        assignment,
        best_assignment,
        best_cut,
    };

    if solver.best_cut > 0 {
        solver.dfs(0, 0, 1, 0);
    }

    let Solver { best_assignment, .. } = solver;
    let final_cut = compute_cut_i8(&best_assignment, &adj);
    let mut part_a: Vec<usize> = best_assignment
        .iter()
        .enumerate()
        .filter_map(|(idx, &val)| if val == 1 { Some(idx) } else { None })
        .collect();
    part_a.sort_unstable();

    println!("{}", final_cut);
    if !part_a.is_empty() {
        for (i, v) in part_a.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", v);
        }
        println!();
    } else {
        println!();
    }
}
