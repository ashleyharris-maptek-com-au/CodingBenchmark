Below is a self-contained Rust program that tackles the described 2D polygon cutting stock problem. It uses heuristic, multi-resolution placement searches (grid, neighbor seeding, and randomized probing) plus adaptive angle sampling to keep runtime manageable while scaling up to thousands of pieces. Geometry checks (point-in-polygon, edge intersections, polygon overlap) are implemented from scratch to avoid external dependencies, and a lightweight spatial grid accelerates collision detection among placed pieces.

```rust
use std::cmp::Ordering;
use std::collections::HashMap;
use std::f64::consts::PI;
use std::io::{self, Read};
use std::ops::{Add, Sub};

const EPS: f64 = 1e-9;
const EDGE_EPS: f64 = 1e-9;
const MAX_GRID_DIV: usize = 120;
const MAX_ANGLE_LIMIT: usize = 96;

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).expect("read stdin");
    let mut it = input.split_whitespace();

    let stock_vertices_count: usize = it.next().unwrap().parse().unwrap();
    let piece_count: usize = it.next().unwrap().parse().unwrap();

    let mut stock_vertices = Vec::with_capacity(stock_vertices_count);
    for _ in 0..stock_vertices_count {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock_vertices.push(Vec2::new(x, y));
    }

    let mut pieces = Vec::with_capacity(piece_count);
    for idx in 0..piece_count {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut vertices = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            vertices.push(Vec2::new(x, y));
        }
        pieces.push(Piece::new(idx, vertices));
    }

    let mut solver = Solver::new(stock_vertices, pieces);
    let (num_stocks, placements) = solver.solve();

    println!("{}", num_stocks);
    for placement in placements {
        println!(
            "{} {:.8} {:.8} {:.8}",
            placement.stock_index, placement.translation.x, placement.translation.y, placement.rotation
        );
    }
}

#[derive(Clone, Copy, Debug, Default)]
struct Vec2 {
    x: f64,
    y: f64,
}
impl Vec2 {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }
    fn zero() -> Self {
        Self { x: 0.0, y: 0.0 }
    }
}
impl Add for Vec2 {
    type Output = Vec2;
    fn add(self, other: Vec2) -> Vec2 {
        Vec2::new(self.x + other.x, self.y + other.y)
    }
}
impl Sub for Vec2 {
    type Output = Vec2;
    fn sub(self, other: Vec2) -> Vec2 {
        Vec2::new(self.x - other.x, self.y - other.y)
    }
}

#[derive(Clone, Copy, Debug)]
struct Rect {
    min: Vec2,
    max: Vec2,
}
impl Rect {
    fn from_points(points: &[Vec2]) -> Self {
        let mut min_x = points[0].x;
        let mut max_x = points[0].x;
        let mut min_y = points[0].y;
        let mut max_y = points[0].y;
        for p in points.iter().skip(1) {
            if p.x < min_x {
                min_x = p.x;
            }
            if p.x > max_x {
                max_x = p.x;
            }
            if p.y < min_y {
                min_y = p.y;
            }
            if p.y > max_y {
                max_y = p.y;
            }
        }
        Rect {
            min: Vec2::new(min_x, min_y),
            max: Vec2::new(max_x, max_y),
        }
    }
    fn width(&self) -> f64 {
        self.max.x - self.min.x
    }
    fn height(&self) -> f64 {
        self.max.y - self.min.y
    }
    fn overlaps(&self, other: &Rect) -> bool {
        !(self.max.x + EDGE_EPS < other.min.x
            || self.min.x > other.max.x + EDGE_EPS
            || self.max.y + EDGE_EPS < other.min.y
            || self.min.y > other.max.y + EDGE_EPS)
    }
}

#[derive(Clone)]
struct Piece {
    id: usize,
    vertices: Vec<Vec2>,
    rel_vertices: Vec<Vec2>,
    centroid: Vec2,
    area: f64,
    base_angles: Vec<f64>,
}
impl Piece {
    fn new(id: usize, vertices: Vec<Vec2>) -> Self {
        let centroid = polygon_centroid(&vertices);
        let rel_vertices: Vec<Vec2> = vertices.iter().map(|v| *v - centroid).collect();
        let area = polygon_area(&vertices);
        let mut base_angles = Vec::new();
        if vertices.len() >= 2 {
            for i in 0..vertices.len() {
                let a = vertices[i];
                let b = vertices[(i + 1) % vertices.len()];
                let edge = b - a;
                if edge.x.abs() < EPS && edge.y.abs() < EPS {
                    continue;
                }
                push_unique_angle(&mut base_angles, edge.y.atan2(edge.x));
                push_unique_angle(&mut base_angles, edge.y.atan2(edge.x) + PI / 2.0);
            }
        }
        Self {
            id,
            vertices,
            rel_vertices,
            centroid,
            area,
            base_angles,
        }
    }
    fn angle_candidates(&self, limit: usize) -> Vec<f64> {
        let mut result = Vec::new();
        result.push(0.0);
        for &angle in &self.base_angles {
            push_unique_angle(&mut result, angle);
            if result.len() >= limit {
                return result;
            }
        }
        let mut extra = 0usize;
        while result.len() < limit {
            let denom = (limit - result.len()).max(1);
            let angle = 2.0 * PI * (extra as f64 / denom as f64);
            push_unique_angle(&mut result, angle);
            extra += 1;
            if extra > limit * 3 {
                break;
            }
        }
        result.truncate(limit);
        result
    }
}

fn push_unique_angle(vec: &mut Vec<f64>, angle: f64) {
    let mut a = angle;
    while a <= -PI {
        a += 2.0 * PI;
    }
    while a > PI {
        a -= 2.0 * PI;
    }
    if !vec.iter().any(|&existing| (existing - a).abs() < 1e-4) {
        vec.push(a);
    }
}

fn polygon_signed_area(vertices: &[Vec2]) -> f64 {
    let mut sum = 0.0;
    for i in 0..vertices.len() {
        let a = vertices[i];
        let b = vertices[(i + 1) % vertices.len()];
        sum += a.x * b.y - b.x * a.y;
    }
    sum * 0.5
}
fn polygon_area(vertices: &[Vec2]) -> f64 {
    polygon_signed_area(vertices).abs()
}
fn polygon_centroid(vertices: &[Vec2]) -> Vec2 {
    let signed_area = polygon_signed_area(vertices);
    if signed_area.abs() < EPS {
        if vertices.is_empty() {
            return Vec2::zero();
        }
        let mut sum = Vec2::zero();
        for v in vertices {
            sum = sum + *v;
        }
        return Vec2::new(sum.x / vertices.len() as f64, sum.y / vertices.len() as f64);
    }
    let mut cx = 0.0;
    let mut cy = 0.0;
    for i in 0..vertices.len() {
        let a = vertices[i];
        let b = vertices[(i + 1) % vertices.len()];
        let cross = a.x * b.y - b.x * a.y;
        cx += (a.x + b.x) * cross;
        cy += (a.y + b.y) * cross;
    }
    let area6 = 6.0 * signed_area;
    Vec2::new(cx / area6, cy / area6)
}

struct StockGeometry {
    polygon: Vec<Vec2>,
    edges: Vec<(Vec2, Vec2)>,
    bbox: Rect,
    centroid: Vec2,
    area: f64,
    cell_size: f64,
}

impl StockGeometry {
    fn new(vertices: Vec<Vec2>) -> Self {
        let bbox = Rect::from_points(&vertices);
        let centroid = polygon_centroid(&vertices);
        let area = polygon_area(&vertices);
        let cell_size = ((bbox.width().max(bbox.height())) / 25.0).max(1.0);
        let mut edges = Vec::with_capacity(vertices.len());
        for i in 0..vertices.len() {
            edges.push((vertices[i], vertices[(i + 1) % vertices.len()]));
        }
        Self {
            polygon: vertices,
            edges,
            bbox,
            centroid,
            area,
            cell_size,
        }
    }
    fn contains_polygon(&self, subject: &[Vec2]) -> bool {
        for &pt in subject {
            if !point_in_polygon(pt, &self.polygon) {
                return false;
            }
        }
        for i in 0..subject.len() {
            let a = subject[i];
            let b = subject[(i + 1) % subject.len()];
            for &(c, d) in &self.edges {
                if segments_intersect_strict(a, b, c, d) {
                    return false;
                }
            }
        }
        true
    }
}

struct Solver {
    stock_geom: StockGeometry,
    pieces: Vec<Piece>,
    settings: SearchSettings,
    rng: XorShift64,
}

impl Solver {
    fn new(stock_vertices: Vec<Vec2>, pieces: Vec<Piece>) -> Self {
        let stock_geom = StockGeometry::new(stock_vertices);
        let piece_count = pieces.len();
        let settings = SearchSettings::new(piece_count);
        let rng_seed = (piece_count as u64).wrapping_mul(0x9E3779B97F4A7C15);
        Self {
            stock_geom,
            pieces,
            settings,
            rng: XorShift64::new(rng_seed | 1),
        }
    }

    fn solve(&mut self) -> (usize, Vec<PlacementInfo>) {
        let np = self.pieces.len();
        let mut outputs = vec![PlacementInfo::default(); np];
        let mut stocks: Vec<StockState> = Vec::new();

        let mut order: Vec<usize> = (0..np).collect();
        order.sort_by(|&a, &b| {
            self.pieces[b]
                .area
                .partial_cmp(&self.pieces[a].area)
                .unwrap_or(Ordering::Equal)
        });

        for piece_idx in order {
            let mut placed: Option<PlacementInfo> = None;

            for stock_idx in 0..stocks.len() {
                if let Some(info) = self.place_on_stock(piece_idx, stock_idx, &mut stocks[stock_idx], &[SearchMode::Normal, SearchMode::Intensive]) {
                    placed = Some(info);
                    break;
                }
            }

            if placed.is_none() && !stocks.is_empty() {
                let mut candidates: Vec<usize> = (0..stocks.len()).collect();
                candidates.sort_by(|&a, &b| {
                    stocks[a]
                        .fill_ratio(self.stock_geom.area)
                        .partial_cmp(&stocks[b].fill_ratio(self.stock_geom.area))
                        .unwrap_or(Ordering::Equal)
                });
                for &stock_idx in candidates.iter().take(1) {
                    if let Some(info) = self.place_on_stock(piece_idx, stock_idx, &mut stocks[stock_idx], &[SearchMode::Exhaustive]) {
                        placed = Some(info);
                        break;
                    }
                }
            }

            if placed.is_none() {
                let mut new_stock = StockState::new(self.stock_geom.cell_size, self.stock_geom.bbox.min);
                let new_index = stocks.len();
                let mut info_opt = None;
                for mode in [SearchMode::Normal, SearchMode::Intensive, SearchMode::Exhaustive] {
                    if let Some(info) = self.try_place_on_stock(piece_idx, new_index, &mut new_stock, mode) {
                        info_opt = Some(info);
                        break;
                    }
                }
                if info_opt.is_none() {
                    info_opt = self.force_place(piece_idx, new_index, &mut new_stock);
                }
                if let Some(info) = info_opt {
                    placed = Some(info);
                    stocks.push(new_stock);
                } else {
                    // Ultimate fallback: keep placements consistent even if impossible.
                    // Place at stock centroid (may be invalid but prevents panic).
                    let default_translation = self.stock_geom.centroid;
                    let fallback = PlacementInfo {
                        stock_index: new_index,
                        translation: default_translation,
                        rotation: 0.0,
                    };
                    outputs[piece_idx] = fallback;
                    continue;
                }
            }

            if let Some(info) = placed {
                outputs[piece_idx] = info;
            }
        }

        (stocks.len(), outputs)
    }

    fn place_on_stock(
        &mut self,
        piece_idx: usize,
        stock_index: usize,
        stock_state: &mut StockState,
        modes: &[SearchMode],
    ) -> Option<PlacementInfo> {
        for &mode in modes {
            if let Some(info) = self.try_place_on_stock(piece_idx, stock_index, stock_state, mode) {
                return Some(info);
            }
        }
        None
    }

    fn try_place_on_stock(
        &mut self,
        piece_idx: usize,
        stock_index: usize,
        stock_state: &mut StockState,
        mode: SearchMode,
    ) -> Option<PlacementInfo> {
        let params = self.settings.params(mode);
        self.try_place_with_params(piece_idx, stock_index, stock_state, params)
    }

    fn force_place(
        &mut self,
        piece_idx: usize,
        stock_index: usize,
        stock_state: &mut StockState,
    ) -> Option<PlacementInfo> {
        let params = SearchParams {
            neighbor_seed_limit: 64,
            random_seed_count: 600,
            grid_divisions: MAX_GRID_DIV,
            angle_limit: MAX_ANGLE_LIMIT,
            random_rounds: 800,
        };
        self.try_place_with_params(piece_idx, stock_index, stock_state, params)
    }

    fn try_place_with_params(
        &mut self,
        piece_idx: usize,
        stock_index: usize,
        stock_state: &mut StockState,
        params: SearchParams,
    ) -> Option<PlacementInfo> {
        let piece = &self.pieces[piece_idx];
        let angle_candidates = piece.angle_candidates(params.angle_limit);
        for &angle in &angle_candidates {
            let rotated = rotate_vertices(&piece.rel_vertices, angle);
            if let Some(candidate) = self.place_with_angle(stock_state, &rotated, &params) {
                stock_state.insert(candidate.clone(), piece_idx, angle, piece.area);
                return Some(PlacementInfo {
                    stock_index,
                    translation: candidate.translation,
                    rotation: angle,
                });
            }
        }
        None
    }

    fn place_with_angle(
        &mut self,
        stock_state: &StockState,
        rotated: &[Vec2],
        params: &SearchParams,
    ) -> Option<CandidateData> {
        let rotated_bbox = Rect::from_points(rotated);
        let range = match self.translation_range(&rotated_bbox) {
            Some(r) => r,
            None => return None,
        };
        let mut attempt = |translation: Vec2, solver: &mut Solver| -> Option<CandidateData> {
            solver.evaluate_candidate(stock_state, rotated, translation)
        };

        let mut seeds = Vec::new();
        seeds.push(range.clamp(self.stock_geom.centroid));
        seeds.push(range.center());
        seeds.push(range.clamp(Vec2::new(range.start_x, range.start_y)));
        seeds.push(range.clamp(Vec2::new(range.end_x, range.end_y)));
        for placement in stock_state
            .placements
            .iter()
            .rev()
            .take(params.neighbor_seed_limit)
        {
            seeds.push(range.clamp(placement.translation));
        }
        for seed in seeds {
            if let Some(result) = attempt(seed, self) {
                return Some(result);
            }
        }

        let diff_x = (range.end_x - range.start_x).max(0.0);
        let diff_y = (range.end_y - range.start_y).max(0.0);
        let steps = params.grid_divisions.max(1);
        for xi in 0..=steps {
            let fx = if steps == 0 { 0.0 } else { xi as f64 / steps as f64 };
            let x = range.start_x + diff_x * fx;
            for yi in 0..=steps {
                let fy = if steps == 0 { 0.0 } else { yi as f64 / steps as f64 };
                let y = range.start_y + diff_y * fy;
                if let Some(result) = attempt(Vec2::new(x, y), self) {
                    return Some(result);
                }
            }
        }

        let random_total = params.random_seed_count.max(1);
        for _ in 0..random_total {
            let x = self.rng.random_range(range.start_x, range.end_x);
            let y = self.rng.random_range(range.start_y, range.end_y);
            if let Some(result) = attempt(Vec2::new(x, y), self) {
                return Some(result);
            }
        }

        if !stock_state.placements.is_empty() {
            for _ in 0..params.random_rounds {
                let idx = (self.rng.next_u64() as usize) % stock_state.placements.len();
                let base = stock_state.placements[idx].translation;
                let jitter_x = (self.rng.next_f64() - 0.5) * rotated_bbox.width().max(1.0);
                let jitter_y = (self.rng.next_f64() - 0.5) * rotated_bbox.height().max(1.0);
                let candidate = range.clamp(Vec2::new(base.x + jitter_x, base.y + jitter_y));
                if let Some(result) = attempt(candidate, self) {
                    return Some(result);
                }
            }
        }
        None
    }

    fn translation_range(&self, rotated_bbox: &Rect) -> Option<TranslationRange> {
        let start_x = self.stock_geom.bbox.min.x - rotated_bbox.min.x;
        let end_x = self.stock_geom.bbox.max.x - rotated_bbox.max.x;
        let start_y = self.stock_geom.bbox.min.y - rotated_bbox.min.y;
        let end_y = self.stock_geom.bbox.max.y - rotated_bbox.max.y;
        if start_x > end_x + 1e-6 || start_y > end_y + 1e-6 {
            return None;
        }
        Some(TranslationRange {
            start_x,
            end_x,
            start_y,
            end_y,
        })
    }

    fn evaluate_candidate(
        &mut self,
        stock_state: &StockState,
        rotated_rel: &[Vec2],
        translation: Vec2,
    ) -> Option<CandidateData> {
        let mut transformed = Vec::with_capacity(rotated_rel.len());
        for v in rotated_rel {
            transformed.push(Vec2::new(v.x + translation.x, v.y + translation.y));
        }
        if transformed.is_empty() {
            return None;
        }
        let bbox = Rect::from_points(&transformed);
        if !self.stock_geom.contains_polygon(&transformed) {
            return None;
        }
        let neighbors = stock_state.query_neighbors(&bbox);
        for idx in neighbors {
            let other = &stock_state.placements[idx];
            if !bbox.overlaps(&other.bbox) {
                continue;
            }
            if polygons_overlap(&transformed, &other.vertices) {
                return None;
            }
        }
        Some(CandidateData {
            translation,
            bbox,
            vertices: transformed,
        })
    }
}

fn rotate_vertices(vertices: &[Vec2], angle: f64) -> Vec<Vec2> {
    let (s, c) = angle.sin_cos();
    vertices
        .iter()
        .map(|v| Vec2::new(v.x * c - v.y * s, v.x * s + v.y * c))
        .collect()
}

#[derive(Clone)]
struct CandidateData {
    translation: Vec2,
    bbox: Rect,
    vertices: Vec<Vec2>,
}

#[derive(Clone, Copy, Debug)]
struct PlacementInfo {
    stock_index: usize,
    translation: Vec2,
    rotation: f64,
}
impl Default for PlacementInfo {
    fn default() -> Self {
        Self {
            stock_index: 0,
            translation: Vec2::zero(),
            rotation: 0.0,
        }
    }
}

#[derive(Hash, Eq, PartialEq, Clone, Copy)]
struct GridKey {
    x: i32,
    y: i32,
}

#[derive(Clone)]
struct PlacementRecord {
    piece_id: usize,
    translation: Vec2,
    rotation: f64,
    vertices: Vec<Vec2>,
    bbox: Rect,
}

struct StockState {
    placements: Vec<PlacementRecord>,
    grid: HashMap<GridKey, Vec<usize>>,
    cell_size: f64,
    grid_origin: Vec2,
    used_area: f64,
}
impl StockState {
    fn new(cell_size: f64, grid_origin: Vec2) -> Self {
        Self {
            placements: Vec::new(),
            grid: HashMap::new(),
            cell_size: cell_size.max(1e-6),
            grid_origin,
            used_area: 0.0,
        }
    }
    fn insert(&mut self, candidate: CandidateData, piece_id: usize, rotation: f64, piece_area: f64) {
        let idx = self.placements.len();
        self.placements.push(PlacementRecord {
            piece_id,
            translation: candidate.translation,
            rotation,
            vertices: candidate.vertices,
            bbox: candidate.bbox,
        });
        self.used_area += piece_area;
        let (min_cx, max_cx, min_cy, max_cy) = self.cell_bounds(&self.placements[idx].bbox);
        for cx in min_cx..=max_cx {
            for cy in min_cy..=max_cy {
                self.grid
                    .entry(GridKey { x: cx, y: cy })
                    .or_default()
                    .push(idx);
            }
        }
    }
    fn cell_bounds(&self, rect: &Rect) -> (i32, i32, i32, i32) {
        let min_cx = ((rect.min.x - EDGE_EPS - self.grid_origin.x) / self.cell_size).floor() as i32;
        let max_cx = ((rect.max.x + EDGE_EPS - self.grid_origin.x) / self.cell_size).floor() as i32;
        let min_cy = ((rect.min.y - EDGE_EPS - self.grid_origin.y) / self.cell_size).floor() as i32;
        let max_cy = ((rect.max.y + EDGE_EPS - self.grid_origin.y) / self.cell_size).floor() as i32;
        (
            min_cx,
            max_cx.max(min_cx),
            min_cy,
            max_cy.max(min_cy),
        )
    }
    fn query_neighbors(&self, rect: &Rect) -> Vec<usize> {
        if self.grid.is_empty() {
            return Vec::new();
        }
        let (min_cx, max_cx, min_cy, max_cy) = self.cell_bounds(rect);
        let mut result = Vec::new();
        for cx in min_cx..=max_cx {
            for cy in min_cy..=max_cy {
                if let Some(list) = self.grid.get(&GridKey { x: cx, y: cy }) {
                    result.extend_from_slice(list);
                }
            }
        }
        if result.len() > 1 {
            result.sort_unstable();
            result.dedup();
        }
        result
    }
    fn fill_ratio(&self, stock_area: f64) -> f64 {
        if stock_area <= EPS {
            1.0
        } else {
            self.used_area / stock_area
        }
    }
}

#[derive(Clone, Copy)]
struct TranslationRange {
    start_x: f64,
    end_x: f64,
    start_y: f64,
    end_y: f64,
}
impl TranslationRange {
    fn clamp(&self, value: Vec2) -> Vec2 {
        Vec2::new(
            clamp(value.x, self.start_x, self.end_x),
            clamp(value.y, self.start_y, self.end_y),
        )
    }
    fn center(&self) -> Vec2 {
        Vec2::new(
            0.5 * (self.start_x + self.end_x),
            0.5 * (self.start_y + self.end_y),
        )
    }
}

fn clamp(v: f64, min_v: f64, max_v: f64) -> f64 {
    if v < min_v {
        min_v
    } else if v > max_v {
        max_v
    } else {
        v
    }
}

fn point_in_polygon(point: Vec2, polygon: &[Vec2]) -> bool {
    let n = polygon.len();
    if n < 3 {
        return false;
    }
    for i in 0..n {
        let a = polygon[i];
        let b = polygon[(i + 1) % n];
        if point_on_segment(a, b, point) {
            return true;
        }
    }
    let mut inside = false;
    for i in 0..n {
        let a = polygon[i];
        let b = polygon[(i + 1) % n];
        let cond = (a.y > point.y) != (b.y > point.y);
        if cond {
            let t = (point.y - a.y) / (b.y - a.y);
            let x_cross = a.x + t * (b.x - a.x);
            if point.x < x_cross {
                inside = !inside;
            }
        }
    }
    inside
}

fn point_on_segment(a: Vec2, b: Vec2, p: Vec2) -> bool {
    orientation(a, b, p).abs() <= EDGE_EPS
        && p.x >= a.x.min(b.x) - EDGE_EPS
        && p.x <= a.x.max(b.x) + EDGE_EPS
        && p.y >= a.y.min(b.y) - EDGE_EPS
        && p.y <= a.y.max(b.y) + EDGE_EPS
}

fn orientation(a: Vec2, b: Vec2, c: Vec2) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn segments_intersect_strict(a: Vec2, b: Vec2, c: Vec2, d: Vec2) -> bool {
    let o1 = orientation(a, b, c);
    let o2 = orientation(a, b, d);
    let o3 = orientation(c, d, a);
    let o4 = orientation(c, d, b);
    (o1 * o2 < -EDGE_EPS) && (o3 * o4 < -EDGE_EPS)
}

fn segments_intersect_inclusive(a: Vec2, b: Vec2, c: Vec2, d: Vec2) -> bool {
    if segments_intersect_strict(a, b, c, d) {
        return true;
    }
    let o1 = orientation(a, b, c);
    let o2 = orientation(a, b, d);
    let o3 = orientation(c, d, a);
    let o4 = orientation(c, d, b);
    (o1.abs() <= EDGE_EPS && point_on_segment(a, b, c))
        || (o2.abs() <= EDGE_EPS && point_on_segment(a, b, d))
        || (o3.abs() <= EDGE_EPS && point_on_segment(c, d, a))
        || (o4.abs() <= EDGE_EPS && point_on_segment(c, d, b))
}

fn polygons_overlap(a: &[Vec2], b: &[Vec2]) -> bool {
    for i in 0..a.len() {
        for j in 0..b.len() {
            if segments_intersect_inclusive(a[i], a[(i + 1) % a.len()], b[j], b[(j + 1) % b.len()]) {
                return true;
            }
        }
    }
    point_in_polygon(a[0], b) || point_in_polygon(b[0], a)
}

struct SearchSettings {
    neighbor_seed_limit: usize,
    random_seed_count: usize,
    grid_divisions: usize,
    angle_limit: usize,
    random_rounds: usize,
}

impl SearchSettings {
    fn new(piece_count: usize) -> Self {
        let neighbor_seed_limit = if piece_count <= 200 { 12 } else if piece_count <= 1500 { 8 } else { 5 };
        let random_seed_count = if piece_count <= 200 { 24 } else if piece_count <= 1500 { 16 } else { 10 };
        let grid_divisions = if piece_count <= 200 { 28 } else if piece_count <= 1500 { 36 } else { 44 };
        let angle_limit = if piece_count <= 200 { 20 } else if piece_count <= 1500 { 14 } else { 10 };
        let random_rounds = if piece_count <= 200 { 60 } else if piece_count <= 1500 { 90 } else { 120 };
        Self {
            neighbor_seed_limit,
            random_seed_count,
            grid_divisions,
            angle_limit,
            random_rounds,
        }
    }
    fn params(&self, mode: SearchMode) -> SearchParams {
        match mode {
            SearchMode::Normal => SearchParams {
                neighbor_seed_limit: self.neighbor_seed_limit,
                random_seed_count: self.random_seed_count,
                grid_divisions: self.grid_divisions,
                angle_limit: self.angle_limit,
                random_rounds: self.random_rounds,
            },
            SearchMode::Intensive => SearchParams {
                neighbor_seed_limit: self.neighbor_seed_limit * 2 + 4,
                random_seed_count: self.random_seed_count * 2 + 8,
                grid_divisions: (self.grid_divisions * 3 / 2).min(MAX_GRID_DIV),
                angle_limit: (self.angle_limit + 6).min(MAX_ANGLE_LIMIT),
                random_rounds: self.random_rounds * 2 + 40,
            },
            SearchMode::Exhaustive => SearchParams {
                neighbor_seed_limit: self.neighbor_seed_limit * 3 + 12,
                random_seed_count: (self.random_seed_count * 5 + 50).min(800),
                grid_divisions: (self.grid_divisions * 3).min(MAX_GRID_DIV),
                angle_limit: (self.angle_limit + 24).min(MAX_ANGLE_LIMIT),
                random_rounds: self.random_rounds * 4 + 200,
            },
        }
    }
}

struct XorShift64 {
    state: u64,
}
impl XorShift64 {
    fn new(seed: u64) -> Self {
        Self { state: if seed == 0 { 1 } else { seed } }
    }
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 7;
        x ^= x >> 9;
        x ^= x << 8;
        self.state = x;
        x
    }
    fn next_f64(&mut self) -> f64 {
        const SCALE: f64 = 1.0 / ((1u64 << 53) as f64);
        ((self.next_u64() >> 11) as f64) * SCALE
    }
    fn random_range(&mut self, min: f64, max: f64) -> f64 {
        if (max - min).abs() < EPS {
            min
        } else {
            min + self.next_f64() * (max - min)
        }
    }
}
```
