#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <queue>
#include <random>
#include <utility>
#include <vector>

struct Operation {
    int machine = 0;
    long long duration = 0;
};

class JobShopSolver {
public:
    JobShopSolver(int jobsCount, int machinesCount, std::vector<std::vector<Operation>> jobData)
        : J(jobsCount), M(machinesCount), jobs(std::move(jobData)) {
        totalTasks = 0;
        totalDuration = 0;
        jobTotalWork.assign(J, 0);
        for (int j = 0; j < J; ++j) {
            for (const auto& op : jobs[j]) {
                jobTotalWork[j] += op.duration;
                totalDuration += op.duration;
                ++totalTasks;
            }
        }
        machineTotalWork.assign(M, 0);
        for (int j = 0; j < J; ++j) {
            for (const auto& op : jobs[j]) {
                if (op.machine >= 0 && op.machine < M) {
                    machineTotalWork[op.machine] += op.duration;
                }
            }
        }
    }

    std::vector<std::vector<long long>> solve() const {
        std::vector<std::vector<long long>> empty(J);
        if (totalTasks == 0) {
            for (int j = 0; j < J; ++j) {
                empty[j].assign(jobs[j].size(), 0LL);
            }
            return empty;
        }
        if (totalTasks <= tinyThreshold) {
            return solveExactTiny();
        }
        return solveHeuristic();
    }

    const std::vector<std::vector<Operation>>& jobDefinitions() const { return jobs; }

private:
    static constexpr int tinyThreshold = 18;
    static constexpr int smallThreshold = 500;
    static constexpr int mediumThreshold = 5000;
    static constexpr int largeThreshold = 20000;
    static constexpr uint64_t seedBase = 0x9e3779b97f4a7c15ULL;

    int J = 0;
    int M = 0;
    std::vector<std::vector<Operation>> jobs;
    long long totalTasks = 0;
    long long totalDuration = 0;
    std::vector<long long> jobTotalWork;
    std::vector<long long> machineTotalWork;

    struct CandidateData {
        long long earliestStart;
        long long duration;
        long long remainingWork;
        long long jobReady;
        long long machineReady;
        int jobIndex;
        int tasksLeft;
    };

    enum class RuleVariant {
        EarliestStart,
        LongestProcessing,
        ShortestProcessing,
        MostRemaining,
        LeastRemaining,
        MachineFavor,
        DeepJobFirst
    };

    struct RuleRun {
        RuleVariant rule;
        long long primaryJitter;
        int tieNoise;
        uint64_t seed;
    };

    struct ScheduleResult {
        std::vector<std::vector<long long>> start;
        long long makespan = std::numeric_limits<long long>::max();
    };

    struct PQEntry {
        long long primary;
        long long secondary;
        long long earliestSnapshot;
        int tertiary;
        int job;
        int version;
        long long seq;
    };

    struct PQCompare {
        bool operator()(const PQEntry& a, const PQEntry& b) const {
            if (a.primary != b.primary) return a.primary > b.primary;
            if (a.secondary != b.secondary) return a.secondary > b.secondary;
            if (a.earliestSnapshot != b.earliestSnapshot) return a.earliestSnapshot > b.earliestSnapshot;
            if (a.tertiary != b.tertiary) return a.tertiary > b.tertiary;
            if (a.job != b.job) return a.job > b.job;
            return a.seq > b.seq;
        }
    };

    static long long safeAdd(long long a, long long b) {
        if (b > 0 && a > std::numeric_limits<long long>::max() - b) {
            return std::numeric_limits<long long>::max();
        }
        if (b < 0 && a < std::numeric_limits<long long>::min() - b) {
            return std::numeric_limits<long long>::min();
        }
        return a + b;
    }

    std::pair<long long, long long> computeKey(RuleVariant rule, const CandidateData& c) const {
        switch (rule) {
            case RuleVariant::EarliestStart:
                return {c.earliestStart, static_cast<long long>(c.jobIndex)};
            case RuleVariant::LongestProcessing:
                return {c.earliestStart, -c.duration};
            case RuleVariant::ShortestProcessing:
                return {safeAdd(c.earliestStart, c.duration), c.duration};
            case RuleVariant::MostRemaining:
                return {c.earliestStart, -c.remainingWork};
            case RuleVariant::LeastRemaining:
                return {c.earliestStart, c.remainingWork};
            case RuleVariant::MachineFavor:
                return {c.machineReady, c.earliestStart};
            case RuleVariant::DeepJobFirst:
                return {c.earliestStart, -static_cast<long long>(c.tasksLeft)};
            default:
                return {c.earliestStart, static_cast<long long>(c.jobIndex)};
        }
    }

    ScheduleResult runHeuristic(const RuleRun& config) const {
        ScheduleResult res;
        res.start.resize(J);
        for (int j = 0; j < J; ++j) {
            res.start[j].assign(jobs[j].size(), 0LL);
        }
        if (totalTasks == 0) {
            res.makespan = 0;
            return res;
        }

        std::vector<int> jobNext(J, 0);
        std::vector<int> jobVersion(J, 0);
        std::vector<long long> jobReady(J, 0);
        std::vector<long long> jobRemaining = jobTotalWork;
        std::vector<long long> machineReady(M, 0);

        std::mt19937_64 rng(config.seed);
        long long seqCounter = 0;
        std::priority_queue<PQEntry, std::vector<PQEntry>, PQCompare> pq;

        auto pushJob = [&](int j) {
            if (jobNext[j] >= static_cast<int>(jobs[j].size())) return;
            const auto& op = jobs[j][jobNext[j]];
            int machine = op.machine;
            long long earliestStart = std::max(jobReady[j], machineReady[machine]);
            CandidateData data{earliestStart,
                               op.duration,
                               jobRemaining[j],
                               jobReady[j],
                               machineReady[machine],
                               j,
                               static_cast<int>(jobs[j].size() - jobNext[j])};
            auto key = computeKey(config.rule, data);
            long long primary = key.first;
            if (config.primaryJitter > 0) {
                long long jitter = static_cast<long long>(rng() % config.primaryJitter);
                primary = safeAdd(primary, jitter);
            }
            long long secondary = key.second;
            int tertiary = (config.tieNoise > 0) ? static_cast<int>(rng() % config.tieNoise) : j;
            pq.push(PQEntry{primary, secondary, earliestStart, tertiary, j, jobVersion[j], seqCounter++});
        };

        for (int j = 0; j < J; ++j) {
            pushJob(j);
        }

        size_t scheduled = 0;
        long long makespan = 0;
        while (scheduled < static_cast<size_t>(totalTasks)) {
            if (pq.empty()) {
                for (int j = 0; j < J; ++j) pushJob(j);
                if (pq.empty()) break;  // safety
            }
            auto entry = pq.top();
            pq.pop();
            int job = entry.job;
            if (entry.version != jobVersion[job]) continue;
            int idx = jobNext[job];
            if (idx >= static_cast<int>(jobs[job].size())) continue;
            const auto& op = jobs[job][idx];
            int machine = op.machine;
            long long start = std::max(jobReady[job], machineReady[machine]);
            long long finish = safeAdd(start, op.duration);
            res.start[job][idx] = start;
            jobReady[job] = finish;
            machineReady[machine] = finish;
            jobNext[job] = idx + 1;
            jobRemaining[job] -= op.duration;
            if (jobRemaining[job] < 0) jobRemaining[job] = 0;
            ++jobVersion[job];
            ++scheduled;
            if (finish > makespan) makespan = finish;
            if (jobNext[job] < static_cast<int>(jobs[job].size())) {
                pushJob(job);
            }
        }
        res.makespan = makespan;
        return res;
    }

    std::vector<std::vector<long long>> solveHeuristic() const {
        long long avgDuration = totalTasks ? std::max<long long>(1, totalDuration / totalTasks) : 1;
        auto plans = buildRunPlan(avgDuration);
        if (plans.empty()) {
            plans.push_back({RuleVariant::EarliestStart, 0, 0, seedBase});
        }
        ScheduleResult best;
        best.makespan = std::numeric_limits<long long>::max();
        for (const auto& cfg : plans) {
            auto res = runHeuristic(cfg);
            if (res.makespan < best.makespan) {
                best = std::move(res);
            }
        }
        if (best.makespan == std::numeric_limits<long long>::max()) {
            best = runHeuristic({RuleVariant::EarliestStart, 0, 0, seedBase});
        }
        return best.start;
    }

    std::vector<std::vector<long long>> solveExactTiny() const {
        std::vector<std::vector<long long>> current(J), best(J);
        for (int j = 0; j < J; ++j) {
            current[j].assign(jobs[j].size(), 0LL);
            best[j] = current[j];
        }
        std::vector<int> jobNext(J, 0);
        std::vector<long long> jobReady(J, 0);
        std::vector<long long> machineReady(M, 0);
        std::vector<long long> jobRemaining = jobTotalWork;
        std::vector<long long> machineRemaining = machineTotalWork;
        long long bestMakespan = std::numeric_limits<long long>::max();

        auto dfs = [&](auto&& self, int scheduled, long long currentMax) -> void {
            if (currentMax >= bestMakespan) return;
            if (scheduled == totalTasks) {
                bestMakespan = currentMax;
                best = current;
                return;
            }
            std::vector<std::pair<long long, int>> order;
            order.reserve(J);
            for (int j = 0; j < J; ++j) {
                if (jobNext[j] < static_cast<int>(jobs[j].size())) {
                    const auto& op = jobs[j][jobNext[j]];
                    long long earliest = std::max(jobReady[j], machineReady[op.machine]);
                    order.emplace_back(earliest, j);
                }
            }
            std::sort(order.begin(), order.end());
            for (const auto& item : order) {
                int job = item.second;
                const auto& op = jobs[job][jobNext[job]];
                int machine = op.machine;
                long long start = std::max(jobReady[job], machineReady[machine]);
                long long finish = safeAdd(start, op.duration);

                int prevJobNext = jobNext[job];
                long long prevJobReady = jobReady[job];
                long long prevMachineReady = machineReady[machine];
                long long prevJobRemaining = jobRemaining[job];
                long long prevMachineRemaining = machineRemaining[machine];

                jobNext[job]++;
                jobReady[job] = finish;
                machineReady[machine] = finish;
                jobRemaining[job] -= op.duration;
                machineRemaining[machine] -= op.duration;
                current[job][prevJobNext] = start;

                long long newMax = std::max(currentMax, finish);
                long long optimistic = newMax;
                for (int jj = 0; jj < J; ++jj) {
                    optimistic = std::max(optimistic, jobReady[jj] + jobRemaining[jj]);
                }
                for (int mm = 0; mm < M; ++mm) {
                    optimistic = std::max(optimistic, machineReady[mm] + machineRemaining[mm]);
                }

                if (optimistic < bestMakespan) {
                    self(self, scheduled + 1, newMax);
                }

                jobNext[job] = prevJobNext;
                jobReady[job] = prevJobReady;
                machineReady[machine] = prevMachineReady;
                jobRemaining[job] = prevJobRemaining;
                machineRemaining[machine] = prevMachineRemaining;
            }
        };

        dfs(dfs, 0, 0LL);
        return best;
    }

    std::vector<RuleRun> buildRunPlan(long long avgDuration) const {
        std::vector<RuleRun> runs;
        auto add = [&](RuleVariant rule, long long jitter, int noise) {
            long long adjustedJitter = std::max<long long>(0, jitter);
            int tieNoise = std::max(0, noise);
            uint64_t seed = seedBase + static_cast<uint64_t>(runs.size()) * 0x94d049bb133111ebULL;
            runs.push_back({rule, adjustedJitter, tieNoise, seed});
        };

        long long jitterSmall = std::max<long long>(1, avgDuration);
        long long jitterMedium = std::max<long long>(1, safeAdd(avgDuration, avgDuration));
        long long jitterLarge = std::max<long long>(1, safeAdd(jitterMedium, jitterMedium));
        int noiseHeavy = 1'000'003;

        if (totalTasks <= smallThreshold) {
            add(RuleVariant::EarliestStart, 0, 0);
            add(RuleVariant::ShortestProcessing, 0, 0);
            add(RuleVariant::LongestProcessing, 0, 0);
            add(RuleVariant::MostRemaining, 0, 0);
            add(RuleVariant::LeastRemaining, 0, 0);
            add(RuleVariant::MachineFavor, 0, 0);
            add(RuleVariant::DeepJobFirst, 0, 0);
            add(RuleVariant::EarliestStart, jitterSmall, noiseHeavy);
            add(RuleVariant::LongestProcessing, jitterSmall, noiseHeavy);
            add(RuleVariant::MachineFavor, jitterSmall, noiseHeavy);
        } else if (totalTasks <= mediumThreshold) {
            add(RuleVariant::EarliestStart, 0, 0);
            add(RuleVariant::LongestProcessing, 0, 0);
            add(RuleVariant::MostRemaining, 0, 0);
            add(RuleVariant::MachineFavor, 0, 0);
            add(RuleVariant::DeepJobFirst, 0, 0);
            add(RuleVariant::EarliestStart, jitterSmall, noiseHeavy);
            add(RuleVariant::LongestProcessing, jitterSmall, noiseHeavy);
            add(RuleVariant::MachineFavor, jitterMedium, noiseHeavy);
            add(RuleVariant::MostRemaining, jitterSmall, noiseHeavy);
        } else if (totalTasks <= largeThreshold) {
            add(RuleVariant::EarliestStart, 0, 0);
            add(RuleVariant::LongestProcessing, 0, 0);
            add(RuleVariant::MostRemaining, 0, 0);
            add(RuleVariant::MachineFavor, 0, 0);
            add(RuleVariant::EarliestStart, jitterMedium, noiseHeavy);
            add(RuleVariant::MachineFavor, jitterMedium, noiseHeavy);
            add(RuleVariant::MostRemaining, jitterMedium, noiseHeavy);
        } else {
            add(RuleVariant::EarliestStart, 0, 0);
            add(RuleVariant::MachineFavor, 0, 0);
            add(RuleVariant::MostRemaining, 0, 0);
            add(RuleVariant::EarliestStart, jitterLarge, noiseHeavy);
            add(RuleVariant::MachineFavor, jitterLarge, noiseHeavy);
        }
        return runs;
    }
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int J, M;
    if (!(std::cin >> J >> M)) {
        return 0;
    }

    std::vector<std::vector<Operation>> jobs(J);
    for (int j = 0; j < J; ++j) {
        int T;
        std::cin >> T;
        jobs[j].resize(T);
        for (int t = 0; t < T; ++t) {
            int machine;
            long long duration;
            std::cin >> machine >> duration;
            jobs[j][t] = {machine, duration};
        }
    }

    JobShopSolver solver(J, M, std::move(jobs));
    auto schedule = solver.solve();
    const auto& jobDefs = solver.jobDefinitions();

    for (int j = 0; j < J; ++j) {
        const auto& ops = jobDefs[j];
        const auto& starts = schedule[j];
        for (size_t t = 0; t < ops.size(); ++t) {
            if (t) std::cout << ' ';
            std::cout << starts[t] << ' ' << ops[t].machine;
        }
        if (j + 1 < J) std::cout << '\n';
    }
    return 0;
}
