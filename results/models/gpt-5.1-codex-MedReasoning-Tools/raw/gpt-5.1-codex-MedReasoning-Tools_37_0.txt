Here is a complete C# implementation that exactly solves the Directed Feedback Vertex Set problem by decomposing the graph into strongly connected components, aggressively reducing acyclic parts, and performing memoized branching only on the remaining cyclic cores:

```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Numerics;

class Program
{
    static void Main()
    {
        var scanner = new FastScanner();
        int n = scanner.NextInt();
        int m = scanner.NextInt();

        var adj = new List<int>[n];
        var rev = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            adj[i] = new List<int>();
            rev[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            adj[u].Add(v);
            rev[v].Add(u);
        }

        var components = StronglyConnectedComponents(adj, rev);
        var answer = new List<int>();

        foreach (var comp in components)
        {
            if (comp.Count == 1)
            {
                int v = comp[0];
                bool selfLoop = false;
                foreach (var to in adj[v])
                {
                    if (to == v)
                    {
                        selfLoop = true;
                        break;
                    }
                }
                if (selfLoop)
                {
                    answer.Add(v);
                }
            }
            else
            {
                var solver = new ComponentSolver(comp, adj);
                var part = solver.SolveAll();
                answer.AddRange(part);
            }
        }

        answer.Sort();
        Console.WriteLine(answer.Count);
        if (answer.Count > 0)
        {
            Console.WriteLine(string.Join(" ", answer));
        }
        else
        {
            Console.WriteLine();
        }
    }

    static List<List<int>> StronglyConnectedComponents(List<int>[] adj, List<int>[] rev)
    {
        int n = adj.Length;
        var order = new List<int>(n);
        var visited = new bool[n];
        var stack = new Stack<(int node, bool processed)>();

        for (int i = 0; i < n; i++)
        {
            if (visited[i]) continue;
            stack.Push((i, false));
            while (stack.Count > 0)
            {
                var (node, processed) = stack.Pop();
                if (processed)
                {
                    order.Add(node);
                    continue;
                }
                if (visited[node]) continue;
                visited[node] = true;
                stack.Push((node, true));
                var neighbors = adj[node];
                for (int idx = neighbors.Count - 1; idx >= 0; idx--)
                {
                    int to = neighbors[idx];
                    if (!visited[to])
                    {
                        stack.Push((to, false));
                    }
                }
            }
        }

        var compId = new int[n];
        Array.Fill(compId, -1);
        var components = new List<List<int>>();

        for (int i = order.Count - 1; i >= 0; i--)
        {
            int node = order[i];
            if (compId[node] != -1) continue;
            var comp = new List<int>();
            var stack2 = new Stack<int>();
            stack2.Push(node);
            compId[node] = components.Count;
            while (stack2.Count > 0)
            {
                int cur = stack2.Pop();
                comp.Add(cur);
                foreach (var prev in rev[cur])
                {
                    if (compId[prev] != -1) continue;
                    compId[prev] = components.Count;
                    stack2.Push(prev);
                }
            }
            components.Add(comp);
        }

        return components;
    }
}

class ComponentSolver
{
    private readonly int size;
    private readonly int[] global;
    private readonly int[][] adj;
    private readonly int[][] rev;
    private readonly bool[] selfLoop;
    private readonly Dictionary<BitsetKey, int[]> memo = new Dictionary<BitsetKey, int[]>();

    public ComponentSolver(List<int> nodes, List<int>[] globalAdj)
    {
        size = nodes.Count;
        global = nodes.ToArray();
        var map = new Dictionary<int, int>(size);
        for (int i = 0; i < size; i++)
        {
            map[global[i]] = i;
        }

        adj = new int[size][];
        var revTemp = new List<int>[size];
        for (int i = 0; i < size; i++)
        {
            revTemp[i] = new List<int>();
        }
        selfLoop = new bool[size];

        for (int i = 0; i < size; i++)
        {
            int g = global[i];
            var outgoing = globalAdj[g];
            var localList = new List<int>();
            foreach (var dest in outgoing)
            {
                if (map.TryGetValue(dest, out int idx))
                {
                    localList.Add(idx);
                    revTemp[idx].Add(i);
                    if (idx == i) selfLoop[i] = true;
                }
            }
            adj[i] = localList.ToArray();
        }

        rev = new int[size][];
        for (int i = 0; i < size; i++)
        {
            rev[i] = revTemp[i].ToArray();
        }
    }

    public List<int> SolveAll()
    {
        var mask = Mask.Full(size);
        return Solve(mask);
    }

    private List<int> Solve(Mask mask)
    {
        var key = new BitsetKey(mask);
        if (memo.TryGetValue(key, out var cached))
        {
            return new List<int>(cached);
        }

        if (mask.IsEmpty)
        {
            memo[key] = Array.Empty<int>();
            return new List<int>();
        }

        var working = mask.Clone();
        var forced = new List<int>();
        bool changed = Reduce(working, forced);
        if (changed)
        {
            var tail = Solve(working);
            if (forced.Count == 0)
            {
                return StoreResult(key, tail);
            }
            else
            {
                var total = new List<int>(forced);
                total.AddRange(tail);
                return StoreResult(key, total);
            }
        }

        var components = GetSccMasks(working);
        if (components.Count > 1)
        {
            var total = new List<int>();
            foreach (var comp in components)
            {
                var sub = Solve(comp);
                total.AddRange(sub);
            }
            return StoreResult(key, total);
        }

        var cycle = FindCycle(working);
        List<int> best = null;
        foreach (var vertex in cycle)
        {
            var next = working.Clone();
            next.RemoveInPlace(vertex);
            var sub = Solve(next);
            var candidate = new List<int> { global[vertex] };
            candidate.AddRange(sub);
            if (best == null || candidate.Count < best.Count)
            {
                best = candidate;
            }
        }

        return StoreResult(key, best ?? new List<int>());
    }

    private List<int> StoreResult(BitsetKey key, List<int> list)
    {
        int[] arr = list.Count == 0 ? Array.Empty<int>() : list.ToArray();
        memo[key] = arr;
        return new List<int>(arr);
    }

    private bool Reduce(Mask mask, List<int> forced)
    {
        if (mask.IsEmpty) return false;
        var active = mask.ToList();
        if (active.Count == 0) return false;

        var indeg = new int[size];
        var outdeg = new int[size];
        foreach (var v in active)
        {
            if (!mask.Contains(v)) continue;
            var neighbors = adj[v];
            for (int i = 0; i < neighbors.Length; i++)
            {
                int to = neighbors[i];
                if (!mask.Contains(to)) continue;
                outdeg[v]++;
                indeg[to]++;
            }
        }

        var queue = new Queue<int>();
        var inQueue = new bool[size];
        bool removedAny = false;

        void Enqueue(int v)
        {
            if (!inQueue[v])
            {
                queue.Enqueue(v);
                inQueue[v] = true;
            }
        }

        void RemoveVertex(int v)
        {
            if (!mask.Contains(v)) return;
            mask.RemoveInPlace(v);
            removedAny = true;
            foreach (var to in adj[v])
            {
                if (!mask.Contains(to)) continue;
                indeg[to]--;
                if (indeg[to] == 0) Enqueue(to);
            }
            foreach (var from in rev[v])
            {
                if (!mask.Contains(from)) continue;
                outdeg[from]--;
                if (outdeg[from] == 0) Enqueue(from);
            }
        }

        foreach (var v in active)
        {
            if (!mask.Contains(v)) continue;
            if (selfLoop[v])
            {
                forced.Add(global[v]);
                RemoveVertex(v);
            }
        }

        foreach (var v in mask.ActiveVertices())
        {
            if (indeg[v] == 0 || outdeg[v] == 0)
            {
                Enqueue(v);
            }
        }

        while (queue.Count > 0)
        {
            int v = queue.Dequeue();
            inQueue[v] = false;
            if (!mask.Contains(v)) continue;
            RemoveVertex(v);
        }

        return removedAny;
    }

    private List<Mask> GetSccMasks(Mask mask)
    {
        var components = new List<Mask>();
        if (mask.IsEmpty) return components;

        var order = new List<int>();
        var visited = new bool[size];
        foreach (var start in mask.ActiveVertices())
        {
            if (visited[start]) continue;
            var stack = new Stack<(int node, bool processed)>();
            stack.Push((start, false));
            while (stack.Count > 0)
            {
                var (node, processed) = stack.Pop();
                if (processed)
                {
                    order.Add(node);
                    continue;
                }
                if (visited[node]) continue;
                visited[node] = true;
                stack.Push((node, true));
                var neighbors = adj[node];
                for (int idx = neighbors.Length - 1; idx >= 0; idx--)
                {
                    int to = neighbors[idx];
                    if (!mask.Contains(to) || visited[to]) continue;
                    stack.Push((to, false));
                }
            }
        }

        var assigned = new bool[size];
        for (int i = order.Count - 1; i >= 0; i--)
        {
            int node = order[i];
            if (!mask.Contains(node) || assigned[node]) continue;
            var stack = new Stack<int>();
            var vertices = new List<int>();
            stack.Push(node);
            assigned[node] = true;
            while (stack.Count > 0)
            {
                int cur = stack.Pop();
                vertices.Add(cur);
                foreach (var prev in rev[cur])
                {
                    if (!mask.Contains(prev) || assigned[prev]) continue;
                    assigned[prev] = true;
                    stack.Push(prev);
                }
            }
            components.Add(Mask.FromVertices(vertices, size));
        }

        return components;
    }

    private List<int> FindCycle(Mask mask)
    {
        var visited = new bool[size];
        var onStack = new bool[size];
        var parent = new int[size];
        Array.Fill(parent, -1);

        foreach (var start in mask.ActiveVertices())
        {
            if (visited[start]) continue;
            var stack = new Stack<(int node, int nextIndex)>();
            stack.Push((start, 0));
            parent[start] = -1;

            while (stack.Count > 0)
            {
                var (node, idx) = stack.Pop();
                if (idx == 0 && !visited[node])
                {
                    visited[node] = true;
                    onStack[node] = true;
                }

                var neighbors = adj[node];
                bool advanced = false;
                for (int i = idx; i < neighbors.Length; i++)
                {
                    int to = neighbors[i];
                    if (!mask.Contains(to)) continue;
                    stack.Push((node, i + 1));
                    if (!visited[to])
                    {
                        parent[to] = node;
                        stack.Push((to, 0));
                    }
                    else if (onStack[to])
                    {
                        return BuildCycle(node, to, parent);
                    }
                    advanced = true;
                    break;
                }

                if (!advanced)
                {
                    onStack[node] = false;
                }
            }
        }

        throw new InvalidOperationException("Cycle expected but not found.");
    }

    private List<int> BuildCycle(int from, int to, int[] parent)
    {
        var cycle = new List<int>();
        cycle.Add(to);
        int cur = from;
        cycle.Add(cur);
        while (cur != to)
        {
            cur = parent[cur];
            if (cur == -1) throw new InvalidOperationException("Invalid parent chain.");
            cycle.Add(cur);
        }
        cycle.RemoveAt(cycle.Count - 1);
        return cycle;
    }
}

class Mask
{
    private readonly ulong[] data;
    public int Size { get; }

    public Mask(int size)
    {
        Size = size;
        data = new ulong[(size + 63) / 64];
    }

    private Mask(int size, ulong[] copyFrom)
    {
        Size = size;
        data = copyFrom;
    }

    public static Mask Full(int size)
    {
        var mask = new Mask(size);
        for (int i = 0; i < size; i++)
        {
            mask.SetBit(i);
        }
        return mask;
    }

    public static Mask FromVertices(List<int> vertices, int size)
    {
        var mask = new Mask(size);
        for (int i = 0; i < vertices.Count; i++)
        {
            mask.SetBit(vertices[i]);
        }
        return mask;
    }

    public Mask Clone()
    {
        var copy = new ulong[data.Length];
        Array.Copy(data, copy, data.Length);
        return new Mask(Size, copy);
    }

    public void SetBit(int idx)
    {
        data[idx >> 6] |= 1UL << (idx & 63);
    }

    public void RemoveInPlace(int idx)
    {
        data[idx >> 6] &= ~(1UL << (idx & 63));
    }

    public bool Contains(int idx)
    {
        return (data[idx >> 6] & (1UL << (idx & 63))) != 0;
    }

    public bool IsEmpty
    {
        get
        {
            for (int i = 0; i < data.Length; i++)
            {
                if (data[i] != 0) return false;
            }
            return true;
        }
    }

    public IEnumerable<int> ActiveVertices()
    {
        for (int block = 0; block < data.Length; block++)
        {
            ulong bits = data[block];
            while (bits != 0)
            {
                int offset = BitOperations.TrailingZeroCount(bits);
                int value = (block << 6) + offset;
                if (value < Size)
                {
                    yield return value;
                }
                bits &= bits - 1;
            }
        }
    }

    public List<int> ToList()
    {
        var list = new List<int>();
        foreach (var v in ActiveVertices())
        {
            list.Add(v);
        }
        return list;
    }

    public ulong[] Data => data;
}

readonly struct BitsetKey : IEquatable<BitsetKey>
{
    private readonly ulong[] data;

    public BitsetKey(Mask mask)
    {
        var source = mask.Data;
        data = new ulong[source.Length];
        Array.Copy(source, data, source.Length);
    }

    public bool Equals(BitsetKey other)
    {
        if (data.Length != other.data.Length) return false;
        for (int i = 0; i < data.Length; i++)
        {
            if (data[i] != other.data[i]) return false;
        }
        return true;
    }

    public override bool Equals(object obj) => obj is BitsetKey other && Equals(other);

    public override int GetHashCode()
    {
        unchecked
        {
            int hash = 17;
            for (int i = 0; i < data.Length; i++)
            {
                hash = hash * 31 + data[i].GetHashCode();
            }
            return hash;
        }
    }
}

class FastScanner
{
    private readonly Stream stream = Console.OpenStandardInput();
    private readonly byte[] buffer = new byte[1 << 16];
    private int ptr;
    private int len;

    private int Read()
    {
        if (ptr >= len)
        {
            len = stream.Read(buffer, 0, buffer.Length);
            ptr = 0;
            if (len <= 0) return -1;
        }
        return buffer[ptr++];
    }

    public int NextInt()
    {
        int c;
        do
        {
            c = Read();
        } while (c <= ' ' && c != -1);

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        int val = 0;
        while (c > ' ')
        {
            val = val * 10 + (c - '0');
            c = Read();
        }

        return val * sign;
    }
}
```

This solver exactly searches the solution space only where necessary, thanks to continual reductions (removing sinks/sources and forced self-loops), decomposition into smaller SCCs, and memoization keyed by the remaining vertex set.
