void initGlobals() {

int charToPiece(char c) {
    switch (c) {
        case 'P': return WPAWN;
        case 'N': return WKNIGHT;
        case 'B': return WBISHOP;
        case 'R': return WROOK;
        case 'Q': return WQUEEN;
        case 'K': return WKING;
        case 'p': return BPAWN;
        case 'n': return BKNIGHT;
        case 'b': return BBISHOP;
        case 'r': return BROOK;
        case 'q': return BQUEEN;
        case 'k': return BKING;
        default: return EMPTY;
    }
}

bool parseFEN(const string& fen, Board& board) {
    stringstream ss(fen);
    string boardStr, active, castling, ep;
    if (!(ss >> boardStr >> active >> castling >> ep)) return false;
    int halfmove = 0, fullmove = 1;
    ss >> halfmove >> fullmove;
    board = Board();
    int rank = 7;
    int file = 0;
    for (char c : boardStr) {
        if (c == '/') {
            rank--;
            file = 0;
            continue;
        }
        if (isdigit(static_cast<unsigned char>(c))) {
            file += c - '0';
            continue;
        }
        if (file >= 8 || rank < 0) return false;
        int sq = rank * 8 + file;
        int piece = charToPiece(c);
        board.squares[sq] = piece;
        if (piece == WKING) board.kingSquare[WHITE] = sq;
        if (piece == BKING) board.kingSquare[BLACK] = sq;
        file++;
    }
    board.whiteToMove = (active == "w");
    board.castlingRights = 0;
    if (castling.find('K') != string::npos) board.castlingRights |= CASTLE_WHITE_K;
    if (castling.find('Q') != string::npos) board.castlingRights |= CASTLE_WHITE_Q;
    if (castling.find('k') != string::npos) board.castlingRights |= CASTLE_BLACK_K;
    if (castling.find('q') != string::npos) board.castlingRights |= CASTLE_BLACK_Q;
    if (ep != "-" && ep.size() == 2) {
        int f = ep[0] - 'a';
        int r = ep[1] - '1';
        if (f >= 0 && f < 8 && r >= 0 && r < 8) {
            board.epSquare = r * 8 + f;
        }
    }
    board.halfmoveClock = halfmove;
    board.fullmoveNumber = fullmove;
    board.zobrist = 0;
    for (int sq = 0; sq < 64; ++sq) {
        int piece = board.squares[sq];
        if (!piece) continue;
        int idx = (piece <= WKING) ? piece - 1 : piece - 7;
        board.zobrist ^= zobristPieces[idx][sq];
    }
    if (!board.whiteToMove) board.zobrist ^= zobristSide;
    board.zobrist ^= zobristCastle[board.castlingRights];
    if (board.epSquare != -1) board.zobrist ^= zobristEP[fileOf[board.epSquare]];
    return board.kingSquare[WHITE] != -1 && board.kingSquare[BLACK] != -1;
}

inline bool shouldStop() {
    if (searchInfo.stop) return true;
    if ((searchInfo.nodes & 0x3FFF) == 0) {
        auto now = chrono::steady_clock::now();
        double elapsed = chrono::duration<double>(now - searchInfo.start).count();
        if (elapsed >= searchInfo.timeLimit) {
            searchInfo.stop = true;
            return true;
        }
    }
    return false;
}

bool squareAttacked(const Board& b, int sq, bool byWhite) {
    int r = rankOf[sq];
    int f = fileOf[sq];
    if (byWhite) {
        if (r > 0) {
            if (f > 0 && b.squares[sq - 9] == WPAWN) return true;
            if (f < 7 && b.squares[sq - 7] == WPAWN) return true;
        }
    } else {
        if (r < 7) {
            if (f > 0 && b.squares[sq + 7] == BPAWN) return true;
            if (f < 7 && b.squares[sq + 9] == BPAWN) return true;
        }
    }
    static const int knightDirs[8][2] = {
        {1,2},{2,1},{2,-1},{1,-2},{-1,-2},{-2,-1},{-2,1},{-1,2}
    };
    for (const auto& d : knightDirs) {
        int nf = f + d[0];
        int nr = r + d[1];
        if (nf < 0 || nf >= 8 || nr < 0 || nr >= 8) continue;
        int piece = b.squares[nr * 8 + nf];
        if (!piece) continue;
        if (byWhite ? piece == WKNIGHT : piece == BKNIGHT) return true;
    }
    static const int bishopDirs[4][2] = {{1,1},{1,-1},{-1,1},{-1,-1}};
    for (const auto& d : bishopDirs) {
        int nf = f + d[0];
        int nr = r + d[1];
        while (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
            int piece = b.squares[nr * 8 + nf];
            if (piece) {
                if (byWhite) {
                    if (piece == WBISHOP || piece == WQUEEN) return true;
                } else {
                    if (piece == BBISHOP || piece == BQUEEN) return true;
                }
                break;
            }
            nf += d[0];
            nr += d[1];
        }
    }
    static const int rookDirs[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};
    for (const auto& d : rookDirs) {
        int nf = f + d[0];
        int nr = r + d[1];
        while (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
            int piece = b.squares[nr * 8 + nf];
            if (piece) {
                if (byWhite) {
                    if (piece == WROOK || piece == WQUEEN) return true;
                } else {
                    if (piece == BROOK || piece == BQUEEN) return true;
                }
                break;
            }
            nf += d[0];
            nr += d[1];
        }
    }
    for (int dr = -1; dr <= 1; ++dr) {
        for (int df = -1; df <= 1; ++df) {
            if (!dr && !df) continue;
            int nf = f + df;
            int nr = r + dr;
            if (nf < 0 || nf >= 8 || nr < 0 || nr >= 8) continue;
            int piece = b.squares[nr * 8 + nf];
            if (byWhite ? piece == WKING : piece == BKING) return true;
        }
    }
    return false;
}

inline bool inCheck(const Board& b, bool whiteKing) {
    int sq = b.kingSquare[whiteKing ? WHITE : BLACK];
    return squareAttacked(b, sq, !whiteKing);
}

void updateCastling(Board& b, int from, int to, int movingPiece, int capturedPiece) {
    uint8_t rights = b.castlingRights;
    if (movingPiece == WKING) rights &= ~(CASTLE_WHITE_K | CASTLE_WHITE_Q);
    if (movingPiece == BKING) rights &= ~(CASTLE_BLACK_K | CASTLE_BLACK_Q);
    if (movingPiece == WROOK) {
        if (from == 0) rights &= ~CASTLE_WHITE_Q;
        if (from == 7) rights &= ~CASTLE_WHITE_K;
    }
    if (movingPiece == BROOK) {
        if (from == 56) rights &= ~CASTLE_BLACK_Q;
        if (from == 63) rights &= ~CASTLE_BLACK_K;
    }
    if (capturedPiece == WROOK) {
        if (to == 0) rights &= ~CASTLE_WHITE_Q;
        if (to == 7) rights &= ~CASTLE_WHITE_K;
    }
    if (capturedPiece == BROOK) {
        if (to == 56) rights &= ~CASTLE_BLACK_Q;
        if (to == 63) rights &= ~CASTLE_BLACK_K;
    }
    if (rights != b.castlingRights) {
        b.zobrist ^= zobristCastle[b.castlingRights];
        b.castlingRights = rights;
        b.zobrist ^= zobristCastle[b.castlingRights];
    }
}

bool makeMove(Board& b, const Move& move, State& st) {
    st.whiteToMove = b.whiteToMove;
    st.castlingRights = b.castlingRights;
    st.epSquare = b.epSquare;
    st.halfmoveClock = b.halfmoveClock;
    st.fullmoveNumber = b.fullmoveNumber;
    st.zobrist = b.zobrist;
    st.kingSquare[0] = b.kingSquare[0];
    st.kingSquare[1] = b.kingSquare[1];
    st.capturedPiece = 0;

    bool movingWhite = b.whiteToMove;

    if (b.epSquare != -1) b.zobrist ^= zobristEP[fileOf[b.epSquare]];
    b.epSquare = -1;

    int from = move.from;
    int to = move.to;
    int piece = move.piece;

    int captureSquare = to;
    if (move.flags & FLAG_ENPASSANT) {
        captureSquare = movingWhite ? (to - 8) : (to + 8);
    }

    int captured = 0;
    if (move.flags & FLAG_ENPASSANT) {
        captured = movingWhite ? BPAWN : WPAWN;
        b.zobrist ^= zobristPieces[(captured <= WKING) ? captured - 1 : captured - 7][captureSquare];
        b.squares[captureSquare] = EMPTY;
    } else {
        captured = b.squares[to];
        if (captured) {
            b.zobrist ^= zobristPieces[(captured <= WKING) ? captured - 1 : captured - 7][to];
        }
    }
    st.capturedPiece = captured;

    b.zobrist ^= zobristPieces[(piece <= WKING) ? piece - 1 : piece - 7][from];
    b.squares[from] = EMPTY;

    int placedPiece = move.promo ? move.promo : piece;
    b.squares[to] = placedPiece;
    b.zobrist ^= zobristPieces[(placedPiece <= WKING) ? placedPiece - 1 : placedPiece - 7][to];

    if (move.flags & FLAG_CASTLE) {
        if (to == 6) {
            b.squares[5] = WROOK;
            b.squares[7] = EMPTY;
            b.zobrist ^= zobristPieces[WROOK - 1][5];
            b.zobrist ^= zobristPieces[WROOK - 1][7];
        } else if (to == 2) {
            b.squares[3] = WROOK;
            b.squares[0] = EMPTY;
            b.zobrist ^= zobristPieces[WROOK - 1][3];
            b.zobrist ^= zobristPieces[WROOK - 1][0];
        } else if (to == 62) {
            b.squares[61] = BROOK;
            b.squares[63] = EMPTY;
            b.zobrist ^= zobristPieces[BROOK - 7][61];
            b.zobrist ^= zobristPieces[BROOK - 7][63];
        } else if (to == 58) {
            b.squares[59] = BROOK;
            b.squares[56] = EMPTY;
            b.zobrist ^= zobristPieces[BROOK - 7][59];
            b.zobrist ^= zobristPieces[BROOK - 7][56];
        }
    }

    if (piece == WKING) b.kingSquare[WHITE] = to;
    if (piece == BKING) b.kingSquare[BLACK] = to;

    updateCastling(b, from, to, piece, captured);

    if (piece == WPAWN || piece == BPAWN || captured) b.halfmoveClock = 0;
    else b.halfmoveClock++;

    if (move.flags & FLAG_DOUBLE) {
        int ep = movingWhite ? (to - 8) : (to + 8);
        b.epSquare = ep;
        b.zobrist ^= zobristEP[fileOf[ep]];
    }

    if (!movingWhite) b.fullmoveNumber++;

    if (inCheck(b, movingWhite)) {
        // undo via state
        b.squares[from] = piece;
        b.squares[to] = move.flags & FLAG_PROMOTION ? 0 : b.squares[to];
        // quick revert using stored state
        b.whiteToMove = st.whiteToMove;
        b.castlingRights = st.castlingRights;
        b.epSquare = st.epSquare;
        b.halfmoveClock = st.halfmoveClock;
        b.fullmoveNumber = st.fullmoveNumber;
        b.kingSquare[0] = st.kingSquare[0];
        b.kingSquare[1] = st.kingSquare[1];
        b.zobrist = st.zobrist;
        b.squares = st.whiteToMove ? b.squares : b.squares; // handled below
        // rebuild squares from state by full restore
        for (int sq = 0; sq < 64; ++sq) {
            // Nothing to do here because we restored earlier? to avoid complexity, we'll unmake via dedicated function.
        }
        return false;
    }

    b.whiteToMove = !b.whiteToMove;
    b.zobrist ^= zobristSide;
    return true;
}

void unmakeMove(Board& b, const Move& move, const State& st) {
    b.whiteToMove = st.whiteToMove;
    b.castlingRights = st.castlingRights;
    b.epSquare = st.epSquare;
    b.halfmoveClock = st.halfmoveClock;
    b.fullmoveNumber = st.fullmoveNumber;
    b.zobrist = st.zobrist;
    b.kingSquare[0] = st.kingSquare[0];
    b.kingSquare[1] = st.kingSquare[1];

    int from = move.from;
    int to = move.to;

    if (move.flags & FLAG_CASTLE) {
        if (to == 6) {
            b.squares[7] = WROOK;
            b.squares[5] = EMPTY;
        } else if (to == 2) {
            b.squares[0] = WROOK;
            b.squares[3] = EMPTY;
        } else if (to == 62) {
            b.squares[63] = BROOK;
            b.squares[61] = EMPTY;
        } else if (to == 58) {
            b.squares[56] = BROOK;
            b.squares[59] = EMPTY;
        }
    }

    b.squares[from] = move.piece;
    if (move.flags & FLAG_ENPASSANT) {
        b.squares[to] = EMPTY;
        int capSq = st.whiteToMove ? (to - 8) : (to + 8);
        b.squares[capSq] = st.capturedPiece;
    } else {
        b.squares[to] = st.capturedPiece;
    }
}
