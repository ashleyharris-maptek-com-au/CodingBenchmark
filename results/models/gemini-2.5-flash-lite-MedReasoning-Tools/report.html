#include <iostream>
#include <vector>
#include <cmath>
#include <limits>
#include <algorithm> // For std::reverse, std::find, std::rotate

// Structure to represent a city
struct Point {
    float x, y;
    int id;
};

// Function to calculate Euclidean distance between two points
// Using double for distance calculation to maintain precision
double dist(const Point& p1, const Point& p2) {
    double dx = p1.x - p2.x;
    double dy = p1.y - p2.y;
    return std::sqrt(dx * dx + dy * dy);
}

// Nearest Neighbor (NN) heuristic to find an initial tour
std::vector<int> nearestNeighbor(int N, const std::vector<Point>& cities) {
    std::vector<int> tour;
    tour.reserve(N);
    std::vector<bool> visited(N, false);

    // Start at city 0
    // Find the actual city 0 from the cities vector by ID
    int start_city_idx = -1;
    for (int i = 0; i < N; ++i) {
        if (cities[i].id == 0) {
            start_city_idx = i;
            break;
        }
    }
    if (start_city_idx == -1) {
        // Handle error: city 0 not found. This should not happen in a valid input.
        return {};
    }

    int current_city_idx = start_city_idx;
    tour.push_back(cities[current_city_idx].id);
    visited[current_city_idx] = true;

    for (int i = 1; i < N; ++i) {
        int next_city_idx = -1;
        double min_dist = std::numeric_limits<double>::max();

        for (int j = 0; j < N; ++j) {
            if (!visited[j]) {
                double d = dist(cities[current_city_idx], cities[j]);
                if (d < min_dist) {
                    min_dist = d;
                    next_city_idx = j;
                }
            }
        }
        
        // This should not happen if N > 0 and not all cities are visited
        if (next_city_idx == -1) {
            break; 
        }

        tour.push_back(cities[next_city_idx].id);
        visited[next_city_idx] = true;
        current_city_idx = next_city_idx;
    }

    return tour;
}

// 2-opt local search algorithm
std::vector<int> twoOpt(int N, std::vector<int> tour, const std::vector<Point>& cities, int max_iterations) {
    bool changed = true;
    int iterations = 0;

    // Create a mapping from city ID to its index in the `cities` vector
    // This is important because `cities` is indexed 0 to N-1, but city IDs might not be.
    // However, the problem statement implies 0-indexed cities, and I set cities[i].id = i,
    // so `cities[city_id_value]` should work. I'll keep this simpler for now.
    // If cities.id is not guaranteed to be cities index, this would need a lookup table.

    while (changed && iterations < max_iterations) {
        changed = false;
        // The first city (index 0) is fixed as city 0, so i starts from 1.
        // We also want to keep the segment ending at city 0 fixed, so k < N-1.
        // The tour is P_0, P_1, ..., P_{N-1}. We implicitly return from P_{N-1} to P_0.
        // Edges are (P_0,P_1), (P_1,P_2), ..., (P_{N-2},P_{N-1}), (P_{N-1},P_0).
        // To fix P_0, we can't change the edges involving P_0 directly.
        // The 2-opt swap reverses a segment. If we reverse (i, k),
        // the edges (tour[i-1], tour[i]) and (tour[k], tour[k+1]) are considered.
        // If i=0, then tour[i-1] is effectively tour[N-1].
        // To fix starting city 0, we perform swaps on internal edges.
        // Indices in tour vector: 0, 1, ..., N-1
        // Edges to consider swapping: (tour[i-1], tour[i]) and (tour[j-1], tour[j])
        // The 2-opt swap typically involves reversing the segment between i and j.

        // To preserve tour[0] (city 0) at the start:
        // We avoid swapping the edge (tour[N-1], tour[0]) or (tour[0], tour[1]).
        // This means 'i' should not be 0, and 'k' should not be N-1.
        for (int i = 1; i < N - 1; ++i) { // i-th city in the tour (first edge is (tour[i-1], tour[i]))
            for (int k = i + 1; k < N; ++k) { // k-th city in the tour (second edge is (tour[k-1], tour[k]))
                // Original edges are (tour[i-1], tour[i]) and (tour[k], tour[(k+1)%N])
                // New edges are (tour[i-1], tour[k]) and (tour[i], tour[(k+1)%N])

                // Get actual Point objects using the ID stored in tour
                const Point& p_i_minus_1 = cities[tour[i-1]];
                const Point& p_i = cities[tour[i]];
                const Point& p_k = cities[tour[k]];
                const Point& p_k_plus_1 = cities[tour[(k + 1) % N]]; // Handles wrap-around for the last segment

                double current_len = dist(p_i_minus_1, p_i) + dist(p_k, p_k_plus_1);
                double new_len = dist(p_i_minus_1, p_k) + dist(p_i, p_k_plus_1);
                
                if (new_len < current_len) {
                    std::reverse(tour.begin() + i, tour.begin() + k + 1);
                    changed = true;
                }
            }
        }
        iterations++;
    }
    return tour;
}

2.  **Odd-Degree Vertex Identification**: Determines all vertices with an odd degree.
3.  **All-Pairs Shortest Paths for Odd Vertices**: Employs Dijkstra's algorithm, run from each odd-degree vertex, to find shortest paths and their corresponding edge sequences to all other odd-degree vertices.
