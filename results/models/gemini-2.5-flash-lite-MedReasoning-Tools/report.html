// Structure to represent a city
struct Point {
    float x, y;
    int id;
};

// Function to calculate Euclidean distance between two points
// Using double for distance calculation to maintain precision
double dist(const Point& p1, const Point& p2) {
    double dx = p1.x - p2.x;
    double dy = p1.y - p2.y;
    return std::sqrt(dx * dx + dy * dy);
}

// Nearest Neighbor (NN) heuristic to find an initial tour
std::vector<int> nearestNeighbor(int N, const std::vector<Point>& cities) {
    std::vector<int> tour;
    tour.reserve(N);
    std::vector<bool> visited(N, false);

    // Start at city 0
    // Find the actual city 0 from the cities vector by ID
    int start_city_idx = -1;
    for (int i = 0; i < N; ++i) {
        if (cities[i].id == 0) {
            start_city_idx = i;
            break;
        }
    }
    if (start_city_idx == -1) {
        // Handle error: city 0 not found. This should not happen in a valid input.
        return {};
    }

    int current_city_idx = start_city_idx;
    tour.push_back(cities[current_city_idx].id);
    visited[current_city_idx] = true;

    for (int i = 1; i < N; ++i) {
        int next_city_idx = -1;
        double min_dist = std::numeric_limits<double>::max();

        for (int j = 0; j < N; ++j) {
            if (!visited[j]) {
                double d = dist(cities[current_city_idx], cities[j]);
                if (d < min_dist) {
                    min_dist = d;
                    next_city_idx = j;
                }
            }
        }
        
        // This should not happen if N > 0 and not all cities are visited
        if (next_city_idx == -1) {
            break; 
        }

        tour.push_back(cities[next_city_idx].id);
        visited[next_city_idx] = true;
        current_city_idx = next_city_idx;
    }

    return tour;
}

// 2-opt local search algorithm
std::vector<int> twoOpt(int N, std::vector<int> tour, const std::vector<Point>& cities, int max_iterations) {
    bool changed = true;
    int iterations = 0;

    // Create a mapping from city ID to its index in the `cities` vector
    // This is important because `cities` is indexed 0 to N-1, but city IDs might not be.
    // However, the problem statement implies 0-indexed cities, and I set cities[i].id = i,
    // so `cities[city_id_value]` should work. I'll keep this simpler for now.
    // If cities.id is not guaranteed to be cities index, this would need a lookup table.

    while (changed && iterations < max_iterations) {
        changed = false;
        // The first city (index 0) is fixed as city 0, so i starts from 1.
        // We also want to keep the segment ending at city 0 fixed, so k < N-1.
        // The tour is P_0, P_1, ..., P_{N-1}. We implicitly return from P_{N-1} to P_0.
        // Edges are (P_0,P_1), (P_1,P_2), ..., (P_{N-2},P_{N-1}), (P_{N-1},P_0).
        // To fix P_0, we can't change the edges involving P_0 directly.
        // The 2-opt swap reverses a segment. If we reverse (i, k),
        // the edges (tour[i-1], tour[i]) and (tour[k], tour[k+1]) are considered.
        // If i=0, then tour[i-1] is effectively tour[N-1].
        // To fix starting city 0, we perform swaps on internal edges.
        // Indices in tour vector: 0, 1, ..., N-1
        // Edges to consider swapping: (tour[i-1], tour[i]) and (tour[j-1], tour[j])
        // The 2-opt swap typically involves reversing the segment between i and j.

        // To preserve tour[0] (city 0) at the start:
        // We avoid swapping the edge (tour[N-1], tour[0]) or (tour[0], tour[1]).
        // This means 'i' should not be 0, and 'k' should not be N-1.
        for (int i = 1; i < N - 1; ++i) { // i-th city in the tour (first edge is (tour[i-1], tour[i]))
            for (int k = i + 1; k < N; ++k) { // k-th city in the tour (second edge is (tour[k-1], tour[k]))
                // Original edges are (tour[i-1], tour[i]) and (tour[k], tour[(k+1)%N])
                // New edges are (tour[i-1], tour[k]) and (tour[i], tour[(k+1)%N])

                // Get actual Point objects using the ID stored in tour
                const Point& p_i_minus_1 = cities[tour[i-1]];
                const Point& p_i = cities[tour[i]];
                const Point& p_k = cities[tour[k]];
                const Point& p_k_plus_1 = cities[tour[(k + 1) % N]]; // Handles wrap-around for the last segment

                double current_len = dist(p_i_minus_1, p_i) + dist(p_k, p_k_plus_1);
                double new_len = dist(p_i_minus_1, p_k) + dist(p_i, p_k_plus_1);
                
                if (new_len < current_len) {
                    std::reverse(tour.begin() + i, tour.begin() + k + 1);
                    changed = true;
                }
            }
        }
        iterations++;
    }
    return tour;
}

fn solve_maze(
    maze: &[Vec<char>],
    start: (usize, usize),
    end: (usize, usize),
    h: usize,
    w: usize,
) -> Option<Vec<(usize, usize)>> {
    let mut visited: Vec<Vec<bool>> = vec![vec![false; w]; h];
    // Stack for iterative DFS: (r, c, current_path_index_in_parent_path)
    // To reconstruct the path efficiently, instead of storing the whole path in the stack,
    // we can store the parent of each cell.
    let mut parent: Vec<Vec<Option<(usize, usize)>>> = vec![vec![None; w]; h];
    let mut stack: Vec<(usize, usize)> = Vec::new();

    stack.push(start);
    visited[start.0][start.1] = true;

    let dr: [isize; 4] = [-1, 1, 0, 0]; // Up, Down, Left, Right
    let dc: [isize; 4] = [0, 0, -1, 1]; // Up, Down, Left, Right

    while let Some(current) = stack.pop() {
        let (r, c) = current;

        if (r, c) == end {
            // Reconstruct path
            let mut path = Vec::new();
            let mut curr = Some(end);
            while let Some((cr, cc)) = curr {
                path.push((cr, cc));
                if (cr, cc) == start {
                    break;
                }
                curr = parent[cr][cc];
            }
            path.reverse();
            return Some(path);
        }

        for i in 0..4 {
            let next_r_isize = r as isize + dr[i];
            let next_c_isize = c as isize + dc[i];

            // Check if within bounds using isize to handle negative results correctly

                }

// Structure to hold information about each suffix
struct SuffixInfo {
    int original_index; // Original starting index of the suffix in the text
    int rank_pair[2];   // Pair of ranks for sorting (current and next segment)
};

// Comparison function for sorting SuffixInfo objects
bool compareSuffixInfo(const SuffixInfo& a, const SuffixInfo& b) {
    if (a.rank_pair[0] != b.rank_pair[0]) {
        return a.rank_pair[0] < b.rank_pair[0];
    }
    return a.rank_pair[1] < b.rank_pair[1];
}

// Builds the suffix array using the O(N log N) algorithm (counting sort based logic implicit with std::sort)
std::vector<int> buildSuffixArray(const std::string& txt) {
    int n = txt.length();
    std::vector<SuffixInfo> suffixes(n);

    // Initialize ranks for characters
    for (int i = 0; i < n; i++) {
        suffixes[i].original_index = i;
        suffixes[i].rank_pair[0] = txt[i]; // Use ASCII value directly
        suffixes[i].rank_pair[1] = (i + 1 < n) ? txt[i + 1] : -1;
    }

    std::sort(suffixes.begin(), suffixes.end(), compareSuffixInfo);

    std::vector<int> inv_sa(n); // Inverse suffix array: inv_sa[i] = rank of suffix starting at i

    for (int k = 4; k < 2 * n; k *= 2) {
        int rank_val = 0;
        suffixes[0].rank_pair[0] = rank_val;
        inv_sa[suffixes[0].original_index] = rank_val;

        for (int i = 1; i < n; i++) {
            if (suffixes[i].rank_pair[0] == suffixes[i - 1].rank_pair[0] &&
                suffixes[i].rank_pair[1] == suffixes[i - 1].rank_pair[1]) {
                suffixes[i].rank_pair[0] = rank_val;
            } else {
                suffixes[i].rank_pair[0] = ++rank_val;
            }
            inv_sa[suffixes[i].original_index] = suffixes[i].rank_pair[0];
        }

        for (int i = 0; i < n; i++) {
            suffixes[i].rank_pair[1] = (suffixes[i].original_index + k < n) ?
                                      inv_sa[suffixes[i].original_index + k] : -1;
        }

        // Using std::sort, for a truly O(N log N) solution, a custom radix sort for pairs would be used here.
        // std::sort makes this O(N log^2 N) in the worst case, but often performs well enough.
        std::sort(suffixes.begin(), suffixes.end(), compareSuffixInfo);
    }

    // Extract the suffix array from the sorted SuffixInfo objects
    std::vector<int> suffixArr(n);
    for (int i = 0; i < n; i++) {
        suffixArr[i] = suffixes[i].original_index;
    }
    return suffixArr;
}

// Builds the LCP array using Kasai's algorithm (O(N))
std::vector<int> buildLCPArray(const std::string& txt, const std::vector<int>& suffixArr) {
    int n = txt.length();
    std::vector<int> lcp(n, 0);
    std::vector<int> invSuff(n, 0); // To store inverse of suffix array: invSuff[i] = rank of suffix i

    for (int i = 0; i < n; i++) {
        invSuff[suffixArr[i]] = i;
    }

    int k = 0; // Stores the current LCP value
    for (int i = 0; i < n; i++) {
        if (invSuff[i] == n - 1) { // If it's the last suffix in sorted order, no next suffix to compare
            k = 0;
            continue;
        }

        // Get the suffix that appears just after the current suffix in the sorted suffix array
        int j = suffixArr[invSuff[i] + 1];

        // Compare characters until mismatch or end of string
        while (i + k < n && j + k < n && txt[i + k] == txt[j + k]) {
            k++;
        }

        lcp[invSuff[i]] = k; // LCP for current suffix (between suffixArr[invSuff[i]] and suffixArr[invSuff[i]+1])
        if (k > 0) {
            k--; // For next iteration, LCP value will be at least k-1
        }
    }
    return lcp;
}

// Sparse Table for Range Minimum Query (RMQ)
class SparseTable {
public:
    std::vector<std::vector<int>> st; // Sparse Table
    std::vector<int> log_table; // Precomputed log2 values
    int n;

    SparseTable(const std::vector<int>& arr) {
        n = arr.size();
        if (n == 0) return; // Handle empty array case

        log_table.resize(n + 1);
        log_table[1] = 0;
        for (int i = 2; i <= n; i++) {
            log_table[i] = log_table[i / 2] + 1;
        }

        int max_log = log_table[n];
        st.assign(n, std::vector<int>(max_log + 1));

        for (int i = 0; i < n; i++) {
            st[i][0] = arr[i];
        }

        for (int j = 1; j <= max_log; j++) {
            for (int i = 0; i + (1 << j) <= n; i++) {
                st[i][j] = std::min(st[i][j - 1], st[i + (1 << (j - 1))][j - 1]);
            }
        }
    }

    int query(int L, int R) { // Query minimum in range [L, R] inclusive
        if (L > R || L < 0 || R >= n || n == 0) return 0; // Handle invalid ranges, or empty table
        int j = log_table[R - L + 1];
        return std::min(st[L][j], st[R - (1 << j) + 1][j]);
    }
};

            Col = col;

        public bool IsMineFlagged => Value == '*';
        public bool IsNumbered => char.IsDigit(Value);

        public override bool Equals(object obj)

        {
            return HashCode.Combine(Row, Col);

            HashSet<Cell> inQueue = new HashSet<Cell>();

            // Initialize queue with all initially numbered cells
            for (int r = 0; r < _height; r++)
            {
                for (int c = 0; c < _width; c++)
                {
                    Cell cell = _grid[r, c];
                    if (cell.IsNumbered)
                    {
                        processingQueue.Enqueue(cell);
                        inQueue.Add(cell);
                    }
                }
            }

struct Point {
    x: f64,
    y: f64,
    z: f64,

// Represents a 3D vector
#[derive(Debug, Clone, Copy)]
struct Vector {
    x: f64,
    y: f64,
    z: f64,
}

impl Vector {
    // Normalizes the vector to a unit vector
    fn normalize(&self) -> Vector {
        let magnitude = (self.x * self.x + self.y * self.y + self.z * self.z).sqrt();
        if magnitude == 0.0 {
            // If the vector is zero, return it as is to avoid division by zero.
            // Or, ideally, handle this as an error or a special case in data validation.
            // For now, returning self is a reasonable default.
            *self
        } else {
            Vector {
                x: self.x / magnitude,
                y: self.y / magnitude,
                z: self.z / magnitude,
            }
        }
    }
}

// --- Grid-based spatial indexing ---

struct Grid {
    cell_size: f64,
    // HashMap: (ix, iy, iz) -> Vec<sample_index>
    // Stores indices to the global all_samples vector
    cells: HashMap<(i32, i32, i32), Vec<usize>>,
    min_coords: Point, // Store min coordinates to offset grid indices
}

impl Grid {
    fn new(samples: &[Sample], cell_size: f64) -> Grid {
        let mut min_x = f64::INFINITY;
        let mut min_y = f64::INFINITY;
        let mut min_z = f64::INFINITY;

        if samples.is_empty() {
            return Grid {
                cell_size,
                cells: HashMap::new(),
                min_coords: Point { x: 0.0, y: 0.0, z: 0.0 },
            };
        }

        for sample in samples {
            min_x = min_x.min(sample.position.x);
            min_y = min_y.min(sample.position.y);
            min_z = min_z.min(sample.position.z);
        }

        let min_coords = Point { x: min_x, y: min_y, z: min_z };
        let mut cells = HashMap::new();

        for (i, sample) in samples.iter().enumerate() {
            let ix = ((sample.position.x - min_coords.x) / cell_size) as i32;
            let iy = ((sample.position.y - min_coords.y) / cell_size) as i32;
            let iz = ((sample.position.z - min_coords.z) / cell_size) as i32;
            cells.entry((ix, iy, iz)).or_insert_with(Vec::new).push(i);
        }

        Grid {
            cell_size,
            cells,
            min_coords,
        }
    }

    // Get the grid cell coordinates for a given point
    fn get_cell_coords(&self, p: Point) -> (i32, i32, i32) {
        let ix = ((p.x - self.min_coords.x) / self.cell_size) as i32;
        let iy = ((p.y - self.min_coords.y) / self.cell_size) as i32;
        let iz = ((p.z - self.min_coords.z) / self.cell_size) as i32;
        (ix, iy, iz)
    }

    // Finds neighbor sample indices within a search_radius for a given sample_index
    fn find_neighbors(&self, sample_idx: usize, all_samples: &[Sample], search_radius: f64) -> Vec<usize> {
        let current_sample = &all_samples[sample_idx];
        let mut neighbors = Vec::new();

        let (current_ix, current_iy, current_iz) = self.get_cell_coords(current_sample.position);

        // Determine how many cells to check in each direction
        // The +1 ensures we check cells that are partially within the search radius
        let cells_to_check_radius = (search_radius / self.cell_size).ceil() as i32;

        for dx in -cells_to_check_radius..=cells_to_check_radius {
            for dy in -cells_to_check_radius..=cells_to_check_radius {
                for dz in -cells_to_check_radius..=cells_to_check_radius {
                    let cell_key = (current_ix + dx, current_iy + dy, current_iz + dz);
                    if let Some(indices) = self.cells.get(&cell_key) {
                        for &neighbor_idx in indices {
                            if neighbor_idx == sample_idx { continue; } // Don't consider self

                            let neighbor_sample = &all_samples[neighbor_idx];
                            let dist_sq = (current_sample.position.x - neighbor_sample.position.x).powi(2) +
                                          (current_sample.position.y - neighbor_sample.position.y).powi(2) +
                                          (current_sample.position.z - neighbor_sample.position.z).powi(2);

                            if dist_sq <= search_radius.powi(2) {
                                neighbors.push(neighbor_idx);
                            }
                        }
                    }
                }
            }
        }
        neighbors
    }
}

// --- Outlier detection logic ---

fn calculate_outlier_confidence(
    value: f64,
    neighbor_values: &[f64],
    min_neighbors_for_stats: usize,
) -> Option<f64> {
    if neighbor_values.len() < min_neighbors_for_stats {
        return None; // Not enough neighbors to make a confident assessment
    }

    let sum: f64 = neighbor_values.iter().sum();
    let mean = sum / (neighbor_values.len() as f64);

    let variance_sum: f64 = neighbor_values.iter().map(|&v| (v - mean).powi(2)).sum();
    let std_dev = (variance_sum / (neighbor_values.len() as f64)).sqrt();

    if std_dev == 0.0 {
        // All neighbor values are the same. If the sample value is different, it's an outlier.
        if (value - mean).abs() > f64::EPSILON { // Use EPSILON for float comparison
            return Some(f64::INFINITY); // Very high confidence outlier
        } else {
            return Some(0.0); // Not an outlier
        }
    }

    let z_score = (value - mean).abs() / std_dev;
    Some(z_score)
}

                hole_id: hole_id.clone(),
                sample_idx,
                depth,
                property_values,
                position,
            };

            all_samples.push(sample); // Add to flat list immediately

            id: hole_id,
            start_point,
            direction,
            length,
            samples: current_hole_samples,

    let confidence_threshold: f64;

        search_radius = 10.0;
        min_neighbors_for_stats = 3;
        confidence_threshold = 2.5; // Z-score of 2.5

        search_radius = 20.0;
        min_neighbors_for_stats = 5;
        confidence_threshold = 3.0; // Z-score of 3.0

        min_neighbors_for_stats = 7;
        confidence_threshold = 3.5; // Z-score of 3.5, stricter for large data

    // Build the spatial grid
    let grid = Grid::new(&all_samples, cell_size);

    let mut suspect_entries: Vec<SuspectEntry> = Vec::new();

        if neighbors_indices.len() < min_neighbors_for_stats {
            // Not enough neighbors, skip for now. Could be an edge case sample
            // or an isolated sample, which might not be an outlier in the traditional sense
            // but rather a data sparsity issue.
            continue;

        // For each property in the current sample, calculate its confidence
        for (prop_idx, &prop_value) in sample.property_values.iter().enumerate() {
            let neighbor_prop_values: Vec<f64> = neighbors_indices.iter()

                .collect();

            if let Some(confidence) = calculate_outlier_confidence(
                prop_value,
                &neighbor_prop_values,
                min_neighbors_for_stats,
            ) {
                if confidence >= confidence_threshold {
                    suspect_entries.push(SuspectEntry {
                        hole_id: sample.hole_id.clone(),
                        sample_idx: sample.sample_idx,
                        property_name: property_names[prop_idx].clone(),
                        confidence,
                    });
                }
            }
        }
    }

    // Sort suspect entries by confidence in descending order
    suspect_entries.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap_or(std::cmp::Ordering::Equal));

    let m = suspect_entries.len();
    println!("{}", m);
    for entry in suspect_entries {
        println!("{} {} {} {}", entry.hole_id, entry.sample_idx, entry.property_name, entry.confidence);
    }

    Ok(())
}
