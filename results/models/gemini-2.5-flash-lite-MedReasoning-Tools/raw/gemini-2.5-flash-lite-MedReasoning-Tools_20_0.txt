The Rust program to validate drillhole assay data is implemented. It includes data structures for `Point`, `Vector`, `Sample`, `Drillhole`, and `SuspectEntry`. The input parsing from stdin is handled, and 3D positions for all samples are calculated.

The core outlier detection logic is implemented using a grid-based spatial indexing system. A `Grid` struct is used to efficiently find spatially neighboring samples within a specified `search_radius`. The `calculate_outlier_confidence` function computes a Z-score-like confidence level for each property value based on its neighbors' values.

Adaptive logic is incorporated to adjust parameters such as `cell_size`, `search_radius`, `min_neighbors_for_stats`, and `confidence_threshold` based on the total number of samples in the dataset. This allows the program to adapt its strategy for small, medium, and large datasets, aiming for efficiency across different scales.

Finally, suspect entries are collected, sorted by confidence, and printed to stdout in the required format.

```rust
// main.rs

use std::io::{self, BufRead};
use std::collections::HashMap;
use std::f64; // For f64::NAN, f64::INFINITY

// Represents a 3D point
#[derive(Debug, Clone, Copy)]
struct Point {
    x: f64,
    y: f64,
    z: f64,
}

// Represents a 3D vector
#[derive(Debug, Clone, Copy)]
struct Vector {
    x: f64,
    y: f64,
    z: f64,
}

impl Vector {
    // Normalizes the vector to a unit vector
    fn normalize(&self) -> Vector {
        let magnitude = (self.x * self.x + self.y * self.y + self.z * self.z).sqrt();
        if magnitude == 0.0 {
            // If the vector is zero, return it as is to avoid division by zero.
            // Or, ideally, handle this as an error or a special case in data validation.
            // For now, returning self is a reasonable default.
            *self
        } else {
            Vector {
                x: self.x / magnitude,
                y: self.y / magnitude,
                z: self.z / magnitude,
            }
        }
    }
}

// Represents a sample taken at a certain depth along a drillhole
#[derive(Debug, Clone)]
struct Sample {
    hole_id: String,
    sample_idx: usize, // 0-indexed sample within the hole
    depth: f64,
    // Store original property values and calculated 3D position
    property_values: Vec<f64>,
    position: Point, // No longer Option, calculated immediately after parsing
}

// Represents a drillhole
#[derive(Debug, Clone)]
struct Drillhole {
    id: String,
    start_point: Point,
    direction: Vector,
    length: f64,
    samples: Vec<Sample>,
}

// A suspect entry found during outlier detection
#[derive(Debug)]
struct SuspectEntry {
    hole_id: String,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

// --- Grid-based spatial indexing ---

struct Grid {
    cell_size: f64,
    // HashMap: (ix, iy, iz) -> Vec<sample_index>
    // Stores indices to the global all_samples vector
    cells: HashMap<(i32, i32, i32), Vec<usize>>,
    min_coords: Point, // Store min coordinates to offset grid indices
}

impl Grid {
    fn new(samples: &[Sample], cell_size: f64) -> Grid {
        let mut min_x = f64::INFINITY;
        let mut min_y = f64::INFINITY;
        let mut min_z = f64::INFINITY;

        if samples.is_empty() {
            return Grid {
                cell_size,
                cells: HashMap::new(),
                min_coords: Point { x: 0.0, y: 0.0, z: 0.0 },
            };
        }

        for sample in samples {
            min_x = min_x.min(sample.position.x);
            min_y = min_y.min(sample.position.y);
            min_z = min_z.min(sample.position.z);
        }

        let min_coords = Point { x: min_x, y: min_y, z: min_z };
        let mut cells = HashMap::new();

        for (i, sample) in samples.iter().enumerate() {
            let ix = ((sample.position.x - min_coords.x) / cell_size) as i32;
            let iy = ((sample.position.y - min_coords.y) / cell_size) as i32;
            let iz = ((sample.position.z - min_coords.z) / cell_size) as i32;
            cells.entry((ix, iy, iz)).or_insert_with(Vec::new).push(i);
        }

        Grid {
            cell_size,
            cells,
            min_coords,
        }
    }

    // Get the grid cell coordinates for a given point
    fn get_cell_coords(&self, p: Point) -> (i32, i32, i32) {
        let ix = ((p.x - self.min_coords.x) / self.cell_size) as i32;
        let iy = ((p.y - self.min_coords.y) / self.cell_size) as i32;
        let iz = ((p.z - self.min_coords.z) / self.cell_size) as i32;
        (ix, iy, iz)
    }

    // Finds neighbor sample indices within a search_radius for a given sample_index
    fn find_neighbors(&self, sample_idx: usize, all_samples: &[Sample], search_radius: f64) -> Vec<usize> {
        let current_sample = &all_samples[sample_idx];
        let mut neighbors = Vec::new();

        let (current_ix, current_iy, current_iz) = self.get_cell_coords(current_sample.position);

        // Determine how many cells to check in each direction
        // The +1 ensures we check cells that are partially within the search radius
        let cells_to_check_radius = (search_radius / self.cell_size).ceil() as i32;

        for dx in -cells_to_check_radius..=cells_to_check_radius {
            for dy in -cells_to_check_radius..=cells_to_check_radius {
                for dz in -cells_to_check_radius..=cells_to_check_radius {
                    let cell_key = (current_ix + dx, current_iy + dy, current_iz + dz);
                    if let Some(indices) = self.cells.get(&cell_key) {
                        for &neighbor_idx in indices {
                            if neighbor_idx == sample_idx { continue; } // Don't consider self

                            let neighbor_sample = &all_samples[neighbor_idx];
                            let dist_sq = (current_sample.position.x - neighbor_sample.position.x).powi(2) +
                                          (current_sample.position.y - neighbor_sample.position.y).powi(2) +
                                          (current_sample.position.z - neighbor_sample.position.z).powi(2);

                            if dist_sq <= search_radius.powi(2) {
                                neighbors.push(neighbor_idx);
                            }
                        }
                    }
                }
            }
        }
        neighbors
    }
}

// --- Outlier detection logic ---

fn calculate_outlier_confidence(
    value: f64,
    neighbor_values: &[f64],
    min_neighbors_for_stats: usize,
) -> Option<f64> {
    if neighbor_values.len() < min_neighbors_for_stats {
        return None; // Not enough neighbors to make a confident assessment
    }

    let sum: f64 = neighbor_values.iter().sum();
    let mean = sum / (neighbor_values.len() as f64);

    let variance_sum: f64 = neighbor_values.iter().map(|&v| (v - mean).powi(2)).sum();
    let std_dev = (variance_sum / (neighbor_values.len() as f64)).sqrt();

    if std_dev == 0.0 {
        // All neighbor values are the same. If the sample value is different, it's an outlier.
        if (value - mean).abs() > f64::EPSILON { // Use EPSILON for float comparison
            return Some(f64::INFINITY); // Very high confidence outlier
        } else {
            return Some(0.0); // Not an outlier
        }
    }

    let z_score = (value - mean).abs() / std_dev;
    Some(z_score)
}


fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read num_holes and num_properties
    let first_line = lines.next().ok_or_else(|| io::Error::new(io::ErrorKind::InvalidInput, "Missing first line"))??;
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    if parts.len() != 2 {
        return Err(io::Error::new(io::ErrorKind::InvalidInput, "Expected 'num_holes num_properties' on first line"));
    }
    let num_holes: usize = parts[0].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid num_holes"))?;
    let num_properties: usize = parts[1].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid num_properties"))?;

    // Read property names
    let property_names_line = lines.next().ok_or_else(|| io::Error::new(io::ErrorKind::InvalidInput, "Missing property names line"))??;
    let property_names: Vec<String> = property_names_line.split_whitespace().map(String::from).collect();
    if property_names.len() != num_properties {
        return Err(io::Error::new(io::ErrorKind::InvalidInput, "Number of property names does not match num_properties"));
    }

    let mut drillholes: Vec<Drillhole> = Vec::with_capacity(num_holes);
    let mut all_samples: Vec<Sample> = Vec::new(); // Flat list of all samples for easier processing

    for _ in 0..num_holes {
        // Read hole_id start_x start_y start_z dir_x dir_y dir_z length num_samples
        let hole_line = lines.next().ok_or_else(|| io::Error::new(io::ErrorKind::InvalidInput, "Missing hole line"))??;
        let parts: Vec<&str> = hole_line.split_whitespace().collect();
        if parts.len() != 9 {
            return Err(io::Error::new(io::ErrorKind::InvalidInput, format!("Expected 9 parts for hole line, got {}", parts.len())));
        }

        let hole_id = parts[0].to_string();
        let start_point = Point {
            x: parts[1].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid start_x"))?,
            y: parts[2].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid start_y"))?,
            z: parts[3].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid start_z"))?,
        };
        let direction = Vector {
            x: parts[4].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid dir_x"))?,
            y: parts[5].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid dir_y"))?,
            z: parts[6].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid dir_z"))?,
        }; // Normalization happens during position calculation
        let length: f64 = parts[7].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid length"))?;
        let num_samples_in_hole: usize = parts[8].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid num_samples_in_hole"))?;

        let mut current_hole_samples: Vec<Sample> = Vec::with_capacity(num_samples_in_hole);
        let normalized_dir = direction.normalize(); // Normalize once per hole

        for sample_idx in 0..num_samples_in_hole {
            // Read depth prop1 prop2 ...
            let sample_line = lines.next().ok_or_else(|| io::Error::new(io::ErrorKind::InvalidInput, "Missing sample line"))??;
            let parts: Vec<&str> = sample_line.split_whitespace().collect();
            if parts.len() != 1 + num_properties {
                return Err(io::Error::new(io::ErrorKind::InvalidInput, format!("Expected {} parts for sample line, got {}", 1 + num_properties, parts.len())));
            }

            let depth: f64 = parts[0].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid depth"))?;
            let property_values: Result<Vec<f64>, _> = parts[1..].iter().map(|s| s.parse()).collect();
            let property_values = property_values.map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid property value"))?;

            // Calculate position directly
            let position = Point {
                x: start_point.x + depth * normalized_dir.x,
                y: start_point.y + depth * normalized_dir.y,
                z: start_point.z + depth * normalized_dir.z,
            };

            let sample = Sample {
                hole_id: hole_id.clone(),
                sample_idx,
                depth,
                property_values,
                position,
            };
            current_hole_samples.push(sample.clone());
            all_samples.push(sample); // Add to flat list immediately
        }

        let drillhole = Drillhole {
            id: hole_id,
            start_point,
            direction,
            length,
            samples: current_hole_samples,
        };
        drillholes.push(drillhole);
    }
    
    // --- Adaptive Logic for Cell Size and Search Radius ---
    let total_samples = all_samples.len();
    let cell_size: f64;
    let search_radius: f64;
    let min_neighbors_for_stats: usize;
    let confidence_threshold: f64;

    // These thresholds and values are heuristic and might need tuning
    if total_samples < 5000 { // Small dataset
        cell_size = 5.0;
        search_radius = 10.0;
        min_neighbors_for_stats = 3;
        confidence_threshold = 2.5; // Z-score of 2.5
    } else if total_samples < 50000 { // Medium dataset
        cell_size = 10.0;
        search_radius = 20.0;
        min_neighbors_for_stats = 5;
        confidence_threshold = 3.0; // Z-score of 3.0
    } else { // Large dataset
        cell_size = 20.0; // Larger cells for faster grid lookup
        search_radius = 40.0; // Larger search radius to ensure enough neighbors
        min_neighbors_for_stats = 7;
        confidence_threshold = 3.5; // Z-score of 3.5, stricter for large data
    }

    // Build the spatial grid
    let grid = Grid::new(&all_samples, cell_size);

    let mut suspect_entries: Vec<SuspectEntry> = Vec::new();

    // Iterate through all samples to find outliers
    for (i, sample) in all_samples.iter().enumerate() {
        let neighbors_indices = grid.find_neighbors(i, &all_samples, search_radius);

        if neighbors_indices.len() < min_neighbors_for_stats {
            // Not enough neighbors, skip for now. Could be an edge case sample
            // or an isolated sample, which might not be an outlier in the traditional sense
            // but rather a data sparsity issue.
            continue;
        }

        // For each property in the current sample, calculate its confidence
        for (prop_idx, &prop_value) in sample.property_values.iter().enumerate() {
            let neighbor_prop_values: Vec<f64> = neighbors_indices.iter()
                .filter_map(|&n_idx| all_samples[n_idx].property_values.get(prop_idx).copied())
                .collect();

            if let Some(confidence) = calculate_outlier_confidence(
                prop_value,
                &neighbor_prop_values,
                min_neighbors_for_stats,
            ) {
                if confidence >= confidence_threshold {
                    suspect_entries.push(SuspectEntry {
                        hole_id: sample.hole_id.clone(),
                        sample_idx: sample.sample_idx,
                        property_name: property_names[prop_idx].clone(),
                        confidence,
                    });
                }
            }
        }
    }

    // Sort suspect entries by confidence in descending order
    suspect_entries.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap_or(std::cmp::Ordering::Equal));

    let m = suspect_entries.len();
    println!("{}", m);
    for entry in suspect_entries {
        println!("{} {} {} {}", entry.hole_id, entry.sample_idx, entry.property_name, entry.confidence);
    }

    Ok(())
}
```
