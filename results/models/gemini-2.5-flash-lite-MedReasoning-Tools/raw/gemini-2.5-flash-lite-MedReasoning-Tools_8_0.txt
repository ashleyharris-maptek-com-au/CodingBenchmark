use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read H and W
    let first_line = lines.next().unwrap().unwrap();
    let dims: Vec<usize> = first_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let h = dims[0];
    let w = dims[1];

    // Read maze
    let mut maze: Vec<Vec<char>> = Vec::with_capacity(h);
    let mut start = (0, 0);
    let mut end = (0, 0);

    for r in 0..h {
        let line = lines.next().unwrap().unwrap();
        let row: Vec<char> = line.chars().collect();
        for c in 0..w {
            match row[c] {
                'A' => start = (r, c),
                'B' => end = (r, c),
                _ => {}
            }
        }
        maze.push(row);
    }

    // Solve maze using iterative DFS
    let path = solve_maze(&maze, start, end, h, w);

    // Print result
    match path {
        Some(p) => {
            println!("{}", p.len());
            for (r, c) in p {
                println!("{} {}", c, r); // Output format is x y, so column then row
            }
        }
        None => {
            println!("0");
        }
    }
}

fn solve_maze(
    maze: &[Vec<char>],
    start: (usize, usize),
    end: (usize, usize),
    h: usize,
    w: usize,
) -> Option<Vec<(usize, usize)>> {
    let mut visited: Vec<Vec<bool>> = vec![vec![false; w]; h];
    // Stack for iterative DFS: (r, c, current_path_index_in_parent_path)
    // To reconstruct the path efficiently, instead of storing the whole path in the stack,
    // we can store the parent of each cell.
    let mut parent: Vec<Vec<Option<(usize, usize)>>> = vec![vec![None; w]; h];
    let mut stack: Vec<(usize, usize)> = Vec::new();

    stack.push(start);
    visited[start.0][start.1] = true;

    let dr: [isize; 4] = [-1, 1, 0, 0]; // Up, Down, Left, Right
    let dc: [isize; 4] = [0, 0, -1, 1]; // Up, Down, Left, Right

    while let Some(current) = stack.pop() {
        let (r, c) = current;

        if (r, c) == end {
            // Reconstruct path
            let mut path = Vec::new();
            let mut curr = Some(end);
            while let Some((cr, cc)) = curr {
                path.push((cr, cc));
                if (cr, cc) == start {
                    break;
                }
                curr = parent[cr][cc];
            }
            path.reverse();
            return Some(path);
        }

        for i in 0..4 {
            let next_r_isize = r as isize + dr[i];
            let next_c_isize = c as isize + dc[i];

            // Check if within bounds using isize to handle negative results correctly
            if next_r_isize >= 0 && next_r_isize < h as isize &&
               next_c_isize >= 0 && next_c_isize < w as isize {

                let nr = next_r_isize as usize;
                let nc = next_c_isize as usize;

                // Check if not wall and not visited
                if maze[nr][nc] != '#' && !visited[nr][nc] {
                    visited[nr][nc] = true;
                    parent[nr][nc] = Some(current);
                    stack.push((nr, nc));
                }
            }
        }
    }

    None // No path found
}
