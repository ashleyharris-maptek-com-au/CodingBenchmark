#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <map>
#include <cmath> // For log2 in SparseTable

// Structure to hold information about each suffix
struct SuffixInfo {
    int original_index; // Original starting index of the suffix in the text
    int rank_pair[2];   // Pair of ranks for sorting (current and next segment)
};

// Comparison function for sorting SuffixInfo objects
bool compareSuffixInfo(const SuffixInfo& a, const SuffixInfo& b) {
    if (a.rank_pair[0] != b.rank_pair[0]) {
        return a.rank_pair[0] < b.rank_pair[0];
    }
    return a.rank_pair[1] < b.rank_pair[1];
}

// Builds the suffix array using the O(N log N) algorithm (counting sort based logic implicit with std::sort)
std::vector<int> buildSuffixArray(const std::string& txt) {
    int n = txt.length();
    std::vector<SuffixInfo> suffixes(n);

    // Initialize ranks for characters
    for (int i = 0; i < n; i++) {
        suffixes[i].original_index = i;
        suffixes[i].rank_pair[0] = txt[i]; // Use ASCII value directly
        suffixes[i].rank_pair[1] = (i + 1 < n) ? txt[i + 1] : -1;
    }

    std::sort(suffixes.begin(), suffixes.end(), compareSuffixInfo);

    std::vector<int> inv_sa(n); // Inverse suffix array: inv_sa[i] = rank of suffix starting at i

    for (int k = 4; k < 2 * n; k *= 2) {
        int rank_val = 0;
        suffixes[0].rank_pair[0] = rank_val;
        inv_sa[suffixes[0].original_index] = rank_val;

        for (int i = 1; i < n; i++) {
            if (suffixes[i].rank_pair[0] == suffixes[i - 1].rank_pair[0] &&
                suffixes[i].rank_pair[1] == suffixes[i - 1].rank_pair[1]) {
                suffixes[i].rank_pair[0] = rank_val;
            } else {
                suffixes[i].rank_pair[0] = ++rank_val;
            }
            inv_sa[suffixes[i].original_index] = suffixes[i].rank_pair[0];
        }

        for (int i = 0; i < n; i++) {
            suffixes[i].rank_pair[1] = (suffixes[i].original_index + k < n) ?
                                      inv_sa[suffixes[i].original_index + k] : -1;
        }

        // Using std::sort, for a truly O(N log N) solution, a custom radix sort for pairs would be used here.
        // std::sort makes this O(N log^2 N) in the worst case, but often performs well enough.
        std::sort(suffixes.begin(), suffixes.end(), compareSuffixInfo);
    }

    // Extract the suffix array from the sorted SuffixInfo objects
    std::vector<int> suffixArr(n);
    for (int i = 0; i < n; i++) {
        suffixArr[i] = suffixes[i].original_index;
    }
    return suffixArr;
}

// Builds the LCP array using Kasai's algorithm (O(N))
std::vector<int> buildLCPArray(const std::string& txt, const std::vector<int>& suffixArr) {
    int n = txt.length();
    std::vector<int> lcp(n, 0);
    std::vector<int> invSuff(n, 0); // To store inverse of suffix array: invSuff[i] = rank of suffix i

    for (int i = 0; i < n; i++) {
        invSuff[suffixArr[i]] = i;
    }

    int k = 0; // Stores the current LCP value
    for (int i = 0; i < n; i++) {
        if (invSuff[i] == n - 1) { // If it's the last suffix in sorted order, no next suffix to compare
            k = 0;
            continue;
        }

        // Get the suffix that appears just after the current suffix in the sorted suffix array
        int j = suffixArr[invSuff[i] + 1];

        // Compare characters until mismatch or end of string
        while (i + k < n && j + k < n && txt[i + k] == txt[j + k]) {
            k++;
        }

        lcp[invSuff[i]] = k; // LCP for current suffix (between suffixArr[invSuff[i]] and suffixArr[invSuff[i]+1])
        if (k > 0) {
            k--; // For next iteration, LCP value will be at least k-1
        }
    }
    return lcp;
}

// Sparse Table for Range Minimum Query (RMQ)
class SparseTable {
public:
    std::vector<std::vector<int>> st; // Sparse Table
    std::vector<int> log_table; // Precomputed log2 values
    int n;

    SparseTable(const std::vector<int>& arr) {
        n = arr.size();
        if (n == 0) return; // Handle empty array case

        log_table.resize(n + 1);
        log_table[1] = 0;
        for (int i = 2; i <= n; i++) {
            log_table[i] = log_table[i / 2] + 1;
        }

        int max_log = log_table[n];
        st.assign(n, std::vector<int>(max_log + 1));

        for (int i = 0; i < n; i++) {
            st[i][0] = arr[i];
        }

        for (int j = 1; j <= max_log; j++) {
            for (int i = 0; i + (1 << j) <= n; i++) {
                st[i][j] = std::min(st[i][j - 1], st[i + (1 << (j - 1))][j - 1]);
            }
        }
    }

    int query(int L, int R) { // Query minimum in range [L, R] inclusive
        if (L > R || L < 0 || R >= n || n == 0) return 0; // Handle invalid ranges, or empty table
        int j = log_table[R - L + 1];
        return std::min(st[L][j], st[R - (1 << j) + 1][j]);
    }
};

// Function to find the longest common substring among multiple strings
std::string longestCommonSubstring(const std::vector<std::string>& strings) {
    if (strings.empty()) {
        return "";
    }
    if (strings.size() == 1) {
        return strings[0];
    }

    std::string concatenated_str = "";
    std::vector<int> string_id_map; // Maps index in concatenated_str to original string ID

    int num_original_strings = strings.size();
    
    // Delimiter strategy: Use char values 1 to num_original_strings-1 as unique delimiters.
    // This assumes input strings do not contain these characters.
    // For N=100, we need 99 delimiters (char 1 to 99). This is generally safe if strings are printable ASCII.
    // This is a standard competitive programming assumption for this type of problem.
    int current_delimiter_val = 1;

    for (int i = 0; i < num_original_strings; ++i) {
        for (char c : strings[i]) {
            concatenated_str += c;
            string_id_map.push_back(i);
        }
        if (i < num_original_strings - 1) {
            concatenated_str += (char)current_delimiter_val++;
            string_id_map.push_back(-1); // Delimiter has no string ID
        }
    }

    // Edge case: if concatenated_str is empty (e.g., all input strings were empty)
    if (concatenated_str.empty()) {
        return "";
    }

    // Build Suffix Array and LCP Array
    std::vector<int> suffixArr = buildSuffixArray(concatenated_str);
    std::vector<int> lcpArr = buildLCPArray(concatenated_str, suffixArr);

    // Build Sparse Table for RMQ on LCP array
    SparseTable rmq_lcp(lcpArr);

    int max_lcs_len = 0;
    int lcs_start_idx = 0;

    std::vector<int> current_string_counts(num_original_strings, 0);
    int distinct_strings_in_window = 0;

    int left = 0;
    // Iterate 'right' pointer from 0 to n-1 over the suffix array
    for (int right = 0; right < concatenated_str.length(); ++right) {
        // Expand window to the right by including the suffix at suffixArr[right]
        int original_string_id_at_suffix_right = string_id_map[suffixArr[right]];
        if (original_string_id_at_suffix_right != -1) { // If it's not a delimiter
            if (current_string_counts[original_string_id_at_suffix_right] == 0) {
                distinct_strings_in_window++;
            }
            current_string_counts[original_string_id_at_suffix_right]++;
        }

        // If all strings are represented in the current window [left, right]
        while (distinct_strings_in_window == num_original_strings && left <= right) {
            // The candidate LCS length for this window is the minimum LCP
            // among suffixes in the range [left, right-1] of the LCP array.
            // lcpArr[i] stores the LCP between suffixArr[i] and suffixArr[i+1].
            // So, for suffixes from suffixArr[left] to suffixArr[right],
            // we are interested in lcpArr indices from 'left' to 'right-1'.
            int current_min_lcp = rmq_lcp.query(left, right - 1);
            
            if (current_min_lcp > max_lcs_len) {
                max_lcs_len = current_min_lcp;
                // The starting index of this LCS in the concatenated string.
                // We can pick any suffix in the current window [left, right]
                // that contributes to this common prefix. suffixArr[left] is a valid choice.
                lcs_start_idx = suffixArr[left]; 
            }

            // Shrink window from the left by removing suffixArr[left]
            int original_string_id_at_suffix_left = string_id_map[suffixArr[left]];
            if (original_string_id_at_suffix_left != -1) { // If it's not a delimiter
                current_string_counts[original_string_id_at_suffix_left]--;
                if (current_string_counts[original_string_id_at_suffix_left] == 0) {
                    distinct_strings_in_window--;
                }
            }
            left++;
        }
    }
    
    if (max_lcs_len == 0) {
        return ""; // No common substring found
    }
    return concatenated_str.substr(lcs_start_idx, max_lcs_len);
}

int main() {
    // Optimize C++ standard streams for faster input/output
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int N;
    std::cin >> N; // Read the number of strings
    std::vector<std::string> strings(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> strings[i]; // Read each string
    }

    // Call the function to find and print the longest common substring
    std::cout << longestCommonSubstring(strings) << std::endl;

    return 0;
}
