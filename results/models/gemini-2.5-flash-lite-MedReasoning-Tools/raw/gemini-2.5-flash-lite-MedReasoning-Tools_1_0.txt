```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <limits>
#include <algorithm> // For std::reverse, std::find, std::rotate

// Structure to represent a city
struct Point {
    float x, y;
    int id;
};

// Function to calculate Euclidean distance between two points
// Using double for distance calculation to maintain precision
double dist(const Point& p1, const Point& p2) {
    double dx = p1.x - p2.x;
    double dy = p1.y - p2.y;
    return std::sqrt(dx * dx + dy * dy);
}

// Nearest Neighbor (NN) heuristic to find an initial tour
std::vector<int> nearestNeighbor(int N, const std::vector<Point>& cities) {
    std::vector<int> tour;
    tour.reserve(N);
    std::vector<bool> visited(N, false);

    // Start at city 0
    // Find the actual city 0 from the cities vector by ID
    int start_city_idx = -1;
    for (int i = 0; i < N; ++i) {
        if (cities[i].id == 0) {
            start_city_idx = i;
            break;
        }
    }
    if (start_city_idx == -1) {
        // Handle error: city 0 not found. This should not happen in a valid input.
        return {};
    }

    int current_city_idx = start_city_idx;
    tour.push_back(cities[current_city_idx].id);
    visited[current_city_idx] = true;

    for (int i = 1; i < N; ++i) {
        int next_city_idx = -1;
        double min_dist = std::numeric_limits<double>::max();

        for (int j = 0; j < N; ++j) {
            if (!visited[j]) {
                double d = dist(cities[current_city_idx], cities[j]);
                if (d < min_dist) {
                    min_dist = d;
                    next_city_idx = j;
                }
            }
        }
        
        // This should not happen if N > 0 and not all cities are visited
        if (next_city_idx == -1) {
            break; 
        }

        tour.push_back(cities[next_city_idx].id);
        visited[next_city_idx] = true;
        current_city_idx = next_city_idx;
    }

    return tour;
}

// 2-opt local search algorithm
std::vector<int> twoOpt(int N, std::vector<int> tour, const std::vector<Point>& cities, int max_iterations) {
    bool changed = true;
    int iterations = 0;

    // Create a mapping from city ID to its index in the `cities` vector
    // This is important because `cities` is indexed 0 to N-1, but city IDs might not be.
    // However, the problem statement implies 0-indexed cities, and I set cities[i].id = i,
    // so `cities[city_id_value]` should work. I'll keep this simpler for now.
    // If cities.id is not guaranteed to be cities index, this would need a lookup table.

    while (changed && iterations < max_iterations) {
        changed = false;
        // The first city (index 0) is fixed as city 0, so i starts from 1.
        // We also want to keep the segment ending at city 0 fixed, so k < N-1.
        // The tour is P_0, P_1, ..., P_{N-1}. We implicitly return from P_{N-1} to P_0.
        // Edges are (P_0,P_1), (P_1,P_2), ..., (P_{N-2},P_{N-1}), (P_{N-1},P_0).
        // To fix P_0, we can't change the edges involving P_0 directly.
        // The 2-opt swap reverses a segment. If we reverse (i, k),
        // the edges (tour[i-1], tour[i]) and (tour[k], tour[k+1]) are considered.
        // If i=0, then tour[i-1] is effectively tour[N-1].
        // To fix starting city 0, we perform swaps on internal edges.
        // Indices in tour vector: 0, 1, ..., N-1
        // Edges to consider swapping: (tour[i-1], tour[i]) and (tour[j-1], tour[j])
        // The 2-opt swap typically involves reversing the segment between i and j.

        // To preserve tour[0] (city 0) at the start:
        // We avoid swapping the edge (tour[N-1], tour[0]) or (tour[0], tour[1]).
        // This means 'i' should not be 0, and 'k' should not be N-1.
        for (int i = 1; i < N - 1; ++i) { // i-th city in the tour (first edge is (tour[i-1], tour[i]))
            for (int k = i + 1; k < N; ++k) { // k-th city in the tour (second edge is (tour[k-1], tour[k]))
                // Original edges are (tour[i-1], tour[i]) and (tour[k], tour[(k+1)%N])
                // New edges are (tour[i-1], tour[k]) and (tour[i], tour[(k+1)%N])

                // Get actual Point objects using the ID stored in tour
                const Point& p_i_minus_1 = cities[tour[i-1]];
                const Point& p_i = cities[tour[i]];
                const Point& p_k = cities[tour[k]];
                const Point& p_k_plus_1 = cities[tour[(k + 1) % N]]; // Handles wrap-around for the last segment

                double current_len = dist(p_i_minus_1, p_i) + dist(p_k, p_k_plus_1);
                double new_len = dist(p_i_minus_1, p_k) + dist(p_i, p_k_plus_1);
                
                if (new_len < current_len) {
                    std::reverse(tour.begin() + i, tour.begin() + k + 1);
                    changed = true;
                }
            }
        }
        iterations++;
    }
    return tour;
}


int main() {
    // Fast I/O
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int N;
    std::cin >> N;

    std::vector<Point> cities(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> cities[i].x >> cities[i].y;
        cities[i].id = i; // Store original index as ID
    }

    std::vector<int> final_tour;

    // Adaptive strategy based on N
    if (N <= 1000) {
        // For smaller N, use Nearest Neighbor and then apply 2-opt for improvement
        std::vector<int> nn_tour = nearestNeighbor(N, cities);
        final_tour = twoOpt(N, nn_tour, cities, 10); // 10 iterations of 2-opt for better quality
    } else {
        // For larger N, rely solely on Nearest Neighbor due to performance constraints
        final_tour = nearestNeighbor(N, cities);
    }
    
    // Ensure the tour starts with city 0
    // The nearestNeighbor function is designed to start with city 0,
    // and 2-opt preserves the start city by not swapping the first segment (0, tour[1])
    // However, if the first city is not 0 due to some edge case, find it and rotate.
    // The previous 2-opt implementation was adjusted to handle indices from 1 to N-1
    // to keep city 0 at the start.
    if (!final_tour.empty() && final_tour[0] != 0) {
        auto it = std::find(final_tour.begin(), final_tour.end(), 0);
        if (it != final_tour.end()) {
            std::rotate(final_tour.begin(), final_tour.end(), it);
        }
    }

    // Output the tour
    for (int i = 0; i < N; ++i) {
        std::cout << final_tour[i] << (i == N - 1 ? "" : " ");
    }
    std::cout << std::endl;

    return 0;
}
```

**Summary of Approach:**

1.  **Nearest Neighbor (NN) Heuristic:** Implemented as the primary tour construction method. It is a greedy algorithm that builds a tour by repeatedly visiting the closest unvisited city. This provides an O(N^2) solution which is a good baseline for performance.
2.  **2-opt Local Search:** For smaller problem instances (`N <= 1000`), the initial tour generated by NN is further optimized using a 2-opt local search. This algorithm iteratively improves the tour by swapping two non-adjacent edges if it results in a shorter tour. This significantly improves the quality of the solution for smaller N without excessively impacting performance. The 2-opt implementation is designed to preserve city 0 as the starting point.
3.  **Adaptive Strategy:**
    *   **`N <= 1000`:** Uses NN + 10 iterations of 2-opt. This provides a balance of speed and solution quality for trivial to medium-sized problems.
    *   **`N > 1000`:** Uses only the NN heuristic. For larger `N`, the O(N^2) complexity of 2-opt becomes too computationally expensive within the time limit. The pure NN heuristic, despite its O(N^2) complexity, can realistically handle up to approximately 500,000 cities within the 300-second constraint on modern hardware (based on an estimated 10^9 operations per second, allowing ~3 * 10^11 operations in 300s, meaning N^2 ~ 3 * 10^11 => N ~ 5.4 * 10^5).
4.  **Input/Output:** Standard input is read efficiently using `std::ios_base::sync_with_stdio(false); std::cin.tie(NULL);`. The output is a space-separated list of city indices, starting with 0, as required.
5.  **Precision:** `double` is used for distance calculations to maintain accuracy, while city coordinates are read as `float` as per the input format.
6.  **Correctness:** The code ensures the tour always starts with city 0 and is a valid permutation of city indices 0 to N-1.

This approach balances the critical requirements of scalability, performance, and quality by dynamically adjusting the algorithmic complexity based on the input size.
