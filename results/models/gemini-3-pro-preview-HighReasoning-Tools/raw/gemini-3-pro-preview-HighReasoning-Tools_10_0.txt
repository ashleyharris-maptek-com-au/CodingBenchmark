#include <iostream>
#include <vector>
#include <algorithm>
#include <chrono>

using namespace std;

struct Rect {
    int x, y, w, h;
};

struct Bin {
    vector<Rect> F;
    int max_min_dim;
    int max_max_dim;
    
    Bin(int W, int H) {
        F.reserve(1024);
        F.push_back({0, 0, W, H});
        max_min_dim = min(W, H);
        max_max_dim = max(W, H);
    }
    
    void update_dims() {
        max_min_dim = 0;
        max_max_dim = 0;
        for (const auto& r : F) {
            int mn = r.w < r.h ? r.w : r.h;
            int mx = r.w > r.h ? r.w : r.h;
            if (mn > max_min_dim) max_min_dim = mn;
            if (mx > max_max_dim) max_max_dim = mx;
        }
    }
    
    inline bool can_fit_fast(int w, int h) const {
        return (w < h ? w : h) <= max_min_dim && (w > h ? w : h) <= max_max_dim;
    }
    
    int score(int w, int h, bool& rot, int& out_idx) const {
        int best = 1e9;
        rot = false;
        out_idx = -1;
        int sz = F.size();
        for (int i = 0; i < sz; ++i) {
            const auto& r = F[i];
            if (w <= r.w && h <= r.h) {
                int s = (r.w - w < r.h - h) ? (r.w - w) : (r.h - h);
                if (s < best) { best = s; rot = false; out_idx = i; }
            }
            if (h <= r.w && w <= r.h) {
                int s = (r.w - h < r.h - w) ? (r.w - h) : (r.h - w);
                if (s < best) { best = s; rot = true; out_idx = i; }
            }
        }
        return best;
    }
    
    bool place(int w, int h, int& out_x, int& out_y, bool& out_rot, int best_idx, bool rot) {
        out_rot = rot;
        out_x = F[best_idx].x;
        out_y = F[best_idx].y;
        
        int rw = rot ? h : w;
        int rh = rot ? w : h;
        
        Rect r = F[best_idx];
        
        F[best_idx] = F.back();
        F.pop_back();
        
        int w1 = r.w - rw;
        int h1 = r.h;
        int w2 = rw;
        int h2 = r.h - rh;
        
        int w3 = r.w;
        int h3 = r.h - rh;
        int w4 = r.w - rw;
        int h4 = rh;
        
        int area_vert = (w1 > 0 && h1 > 0 ? w1 * h1 : 0) > (w2 > 0 && h2 > 0 ? w2 * h2 : 0) ? 
                        (w1 > 0 && h1 > 0 ? w1 * h1 : 0) : (w2 > 0 && h2 > 0 ? w2 * h2 : 0);
        int area_horiz = (w3 > 0 && h3 > 0 ? w3 * h3 : 0) > (w4 > 0 && h4 > 0 ? w4 * h4 : 0) ? 
                         (w3 > 0 && h3 > 0 ? w3 * h3 : 0) : (w4 > 0 && h4 > 0 ? w4 * h4 : 0);
        
        if (area_vert > area_horiz) {
            if (w1 > 0 && h1 > 0) F.push_back({r.x + rw, r.y, w1, h1});
            if (w2 > 0 && h2 > 0) F.push_back({r.x, r.y + rh, w2, h2});
        } else {
            if (w3 > 0 && h3 > 0) F.push_back({r.x, r.y + rh, w3, h3});
            if (w4 > 0 && h4 > 0) F.push_back({r.x + rw, r.y, w4, h4});
        }
        
        update_dims();
        return true;
    }
};

struct Item {
    int id;
    int w, h;
    int b_idx, x, y, rot;
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int N, W, H;
    if (!(cin >> N >> W >> H)) return 0;
    
    vector<Item> items(N);
    for (int i = 0; i < N; ++i) {
        items[i].id = i;
        cin >> items[i].w >> items[i].h;
    }
    
    vector<Item*> sorted_items(N);
    for (int i = 0; i < N; ++i) sorted_items[i] = &items[i];
    
    sort(sorted_items.begin(), sorted_items.end(), [](const Item* a, const Item* b) {
        int max_a = a->w > a->h ? a->w : a->h;
        int max_b = b->w > b->h ? b->w : b->h;
        if (max_a != max_b) return max_a > max_b;
        
        int min_a = a->w < a->h ? a->w : a->h;
        int min_b = b->w < b->h ? b->w : b->h;
        if (min_a != min_b) return min_a > min_b;
        
        return a->id < b->id;
    });
    
    vector<Bin> bins;
    bins.reserve(N / 10 + 1);
    
    auto start_time = chrono::high_resolution_clock::now();
    bool speed_mode = false;
    
    for (int i = 0; i < N; ++i) {
        Item* item = sorted_items[i];
        int best_bin = -1;
        int best_score = 1e9;
        int best_rect_idx = -1;
        bool best_rot = false;
        
        if (i > 0 && i % 2000 == 0) {
            auto now = chrono::high_resolution_clock::now();
            chrono::duration<double> diff = now - start_time;
            if (diff.count() > 25.0) {
                speed_mode = true;
            }
        }
        
        int checked = 0;
        int bins_sz = bins.size();
        for (int b = 0; b < bins_sz; ++b) {
            if (!bins[b].can_fit_fast(item->w, item->h)) continue;
            
            bool rot;
            int idx;
            int s = bins[b].score(item->w, item->h, rot, idx);
            if (s < best_score) {
                best_score = s;
                best_bin = b;
                best_rect_idx = idx;
                best_rot = rot;
                if (best_score == 0) break; 
            }
            if (speed_mode) {
                checked++;
                if (checked > 50) break;
            }
        }
        
        if (best_bin == -1) {
            best_bin = bins.size();
            bins.push_back(Bin(W, H));
            bins[best_bin].score(item->w, item->h, best_rot, best_rect_idx);
        }
        
        bins[best_bin].place(item->w, item->h, item->x, item->y, item->rot, best_rect_idx, best_rot);
        item->b_idx = best_bin;
    }
    
    cout << bins.size() << "\n";
    for (int i = 0; i < N; ++i) {
        cout << items[i].b_idx << " " << items[i].x << " " << items[i].y << " " << items[i].rot << "\n";
    }
    
    return 0;
}
