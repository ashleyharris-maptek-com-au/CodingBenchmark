using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Text;
using System.Threading;
using System.Threading.Tasks;

namespace CuttingStock
{
    struct Cut
    {
        public int Length;
        public int OriginalIndex;
    }

    class Scanner
    {
        private Stream stream;
        private byte[] buffer = new byte[1024 * 1024];
        private int head = 0;
        private int tail = 0;

        public Scanner(Stream stream)
        {
            this.stream = stream;
        }

        public int NextInt()
        {
            int c = Read();
            while (c <= ' ')
            {
                if (c == -1) return -1;
                c = Read();
            }
            bool neg = false;
            if (c == '-')
            {
                neg = true;
                c = Read();
            }
            int res = 0;
            while (c > ' ')
            {
                res = res * 10 + (c - '0');
                c = Read();
            }
            return neg ? -res : res;
        }

        private int Read()
        {
            if (head >= tail)
            {
                head = 0;
                tail = stream.Read(buffer, 0, buffer.Length);
                if (tail == 0) return -1;
            }
            return buffer[head++];
        }
    }

    class SegmentTreeFFD
    {
        public int[] tree;
        public int size;
        public int C;

        public SegmentTreeFFD(int maxBins, int C)
        {
            this.C = C;
            size = 1;
            while (size < maxBins) size *= 2;
            tree = new int[2 * size];
        }

        public void Reset()
        {
            Array.Fill(tree, C);
        }

        public int Insert(int L, int maxAllowedBins)
        {
            if (tree[1] < L) return -1;

            int node = 1;
            while (node < size)
            {
                if (tree[2 * node] >= L)
                {
                    node = 2 * node;
                }
                else
                {
                    node = 2 * node + 1;
                }
            }

            int binIndex = node - size;
            if (binIndex >= maxAllowedBins) return -1;

            tree[node] -= L;
            int current = node / 2;
            while (current > 0)
            {
                tree[current] = tree[2 * current] > tree[2 * current + 1] ? tree[2 * current] : tree[2 * current + 1];
                current /= 2;
            }
            return binIndex;
        }
    }

    class Program
    {
        static object lockObj = new object();
        static int globalBestCount = int.MaxValue;
        static List<List<int>> bestSolution = null;
        static volatile bool stopThreads = false;

        static void Main(string[] args)
        {
            Stopwatch sw = Stopwatch.StartNew();

            using Stream stdin = Console.OpenStandardInput();
            Scanner scanner = new Scanner(stdin);

            int N = scanner.NextInt();
            if (N <= 0) return;
            int C = scanner.NextInt();

            Cut[] cuts = new Cut[N];
            for (int i = 0; i < N; i++)
            {
                cuts[i] = new Cut { Length = scanner.NextInt(), OriginalIndex = i };
            }

            Array.Sort(cuts, (a, b) => b.Length.CompareTo(a.Length));

            List<List<int>> exactBins = new List<List<int>>();
            bool[] paired = new bool[N];

            for (int i = 0; i < N; i++)
            {
                if (cuts[i].Length == C)
                {
                    exactBins.Add(new List<int> { cuts[i].OriginalIndex });
                    paired[i] = true;
                }
                else if (cuts[i].Length < C)
                {
                    break;
                }
            }

            int l = 0, r = N - 1;
            while (l < r)
            {
                if (paired[l]) { l++; continue; }
                if (paired[r]) { r--; continue; }
                long sum = (long)cuts[l].Length + cuts[r].Length;
                if (sum == C)
                {
                    exactBins.Add(new List<int> { cuts[l].OriginalIndex, cuts[r].OriginalIndex });
                    paired[l] = true;
                    paired[r] = true;
                    l++; r--;
                }
                else if (sum > C)
                {
                    l++;
                }
                else
                {
                    r--;
                }
            }

            List<Cut> remainingList = new List<Cut>(N);
            long sumRem = 0;
            for (int i = 0; i < N; i++)
            {
                if (!paired[i])
                {
                    remainingList.Add(cuts[i]);
                    sumRem += cuts[i].Length;
                }
            }

            Cut[] baseCuts = remainingList.ToArray();

            if (baseCuts.Length == 0)
            {
                PrintSolution(exactBins, new List<List<int>>());
                return;
            }

            int L1 = (int)((sumRem + C - 1) / C);

            int initialBest = InitialFFD(baseCuts, C, out int[] initialHead, out int[] initialNext);

            globalBestCount = initialBest;
            SaveSolution(initialHead, initialNext, baseCuts, initialBest);

            if (globalBestCount == L1)
            {
                PrintSolution(exactBins, bestSolution);
                return;
            }

            int numTasks = Environment.ProcessorCount;
            if (numTasks < 1) numTasks = 1;

            Thread[] threads = new Thread[numTasks];
            for (int t = 0; t < numTasks; t++)
            {
                int taskId = t;
                threads[t] = new Thread(() => Worker(baseCuts, taskId, L1, C, sw, initialBest));
                threads[t].Priority = ThreadPriority.Highest;
                threads[t].Start();
            }

            foreach (var thread in threads)
            {
                thread.Join();
            }

            PrintSolution(exactBins, bestSolution);
        }

        static int InitialFFD(Cut[] cuts, int C, out int[] head, out int[] next)
        {
            int N = cuts.Length;
            SegmentTreeFFD segTree = new SegmentTreeFFD(N, C);
            segTree.Reset();
            head = new int[N];
            Array.Fill(head, -1);
            next = new int[N];

            int active = 0;
            for (int i = 0; i < N; i++)
            {
                int bin = segTree.Insert(cuts[i].Length, N);
                if (bin == -1) throw new Exception("Initial FFD failed to pack items.");
                next[i] = head[bin];
                head[bin] = i;
                if (bin >= active) active = bin + 1;
            }
            return active;
        }

        static void Worker(Cut[] baseCuts, int taskId, int L1, int C, Stopwatch sw, int maxPossibleBins)
        {
            int N = baseCuts.Length;
            Cut[] localCuts = new Cut[N];
            Random rnd = new Random(taskId + 1337);

            SegmentTreeFFD segTree = new SegmentTreeFFD(maxPossibleBins, C);
            int[] head = new int[maxPossibleBins];
            int[] next = new int[N];
            long[] keys = new long[N];

            while (!stopThreads)
            {
                if (sw.Elapsed.TotalSeconds > 280)
                {
                    stopThreads = true;
                    break;
                }

                Array.Copy(baseCuts, localCuts, N);
                int threadStrategy = taskId % 4;

                if (threadStrategy == 0)
                {
                    int numSwaps = rnd.Next(1, N / 10 + 2);
                    for (int i = 0; i < numSwaps; i++)
                    {
                        int a = rnd.Next(N);
                        int b = a + rnd.Next(-5, 6);
                        if (b >= 0 && b < N)
                        {
                            var tmp = localCuts[a];
                            localCuts[a] = localCuts[b];
                            localCuts[b] = tmp;
                        }
                    }
                }
                else if (threadStrategy == 1)
                {
                    for (int i = 0; i < N; i++)
                    {
                        keys[i] = -((long)localCuts[i].Length * 1000 + rnd.Next(500));
                    }
                    Array.Sort(keys, localCuts);
                }
                else if (threadStrategy == 2)
                {
                    int blockSize = rnd.Next(2, 20);
                    for (int i = 0; i < N - blockSize; i += blockSize)
                    {
                        if (rnd.NextDouble() < 0.5)
                        {
                            Array.Reverse(localCuts, i, blockSize);
                        }
                    }
                }
                else
                {
                    for (int i = 0; i < N; i++)
                    {
                        keys[i] = -((long)localCuts[i].Length * 1000 + rnd.Next(2000));
                    }
                    Array.Sort(keys, localCuts);
                }

                int currentBest = globalBestCount;
                if (currentBest == L1)
                {
                    stopThreads = true;
                    break;
                }

                int allowedBins = currentBest - 1;
                if (allowedBins > maxPossibleBins) allowedBins = maxPossibleBins;
                if (allowedBins <= 0) break;

                segTree.Reset();
                Array.Fill(head, -1, 0, allowedBins);

                bool success = true;
                int active = 0;
                for (int i = 0; i < N; i++)
                {
                    int bin = segTree.Insert(localCuts[i].Length, allowedBins);
                    if (bin == -1)
                    {
                        success = false;
                        break;
                    }
                    next[i] = head[bin];
                    head[bin] = i;
                    if (bin >= active) active = bin + 1;
                }

                if (success && active < globalBestCount)
                {
                    lock (lockObj)
                    {
                        if (active < globalBestCount)
                        {
                            globalBestCount = active;
                            SaveSolution(head, next, localCuts, active);
                            if (active == L1)
                            {
                                stopThreads = true;
                            }
                        }
                    }
                }
            }
        }

        static void SaveSolution(int[] head, int[] next, Cut[] localCuts, int bins)
        {
            var newBest = new List<List<int>>(bins);
            for (int b = 0; b < bins; b++)
            {
                List<int> binItems = new List<int>();
                int curr = head[b];
                while (curr != -1)
                {
                    binItems.Add(localCuts[curr].OriginalIndex);
                    curr = next[curr];
                }
                newBest.Add(binItems);
            }
            bestSolution = newBest;
        }

        static void PrintSolution(List<List<int>> exactBins, List<List<int>> bestBins)
        {
            int totalStocks = exactBins.Count + (bestBins != null ? bestBins.Count : 0);
            Console.WriteLine(totalStocks);

            using StreamWriter writer = new StreamWriter(Console.OpenStandardOutput(), Encoding.ASCII, 1024 * 1024);

            void WriteBin(List<int> bin)
            {
                for (int i = 0; i < bin.Count; i++)
                {
                    writer.Write(bin[i]);
                    if (i < bin.Count - 1) writer.Write(' ');
                }
                writer.WriteLine();
            }

            foreach (var bin in exactBins)
            {
                WriteBin(bin);
            }
            if (bestBins != null)
            {
                foreach (var bin in bestBins)
                {
                    WriteBin(bin);
                }
            }
        }
    }
}
