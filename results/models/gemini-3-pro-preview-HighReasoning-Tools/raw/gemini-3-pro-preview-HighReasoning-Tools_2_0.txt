using System;
using System.Collections.Generic;
using System.Linq;

namespace ChinesePostman
{
    public class Edge
    {
        public int U { get; set; }
        public int V { get; set; }
        public int Weight { get; set; }
        
        public Edge(int u, int v, int weight)
        {
            U = u;
            V = v;
            Weight = weight;
        }
    }

    public class Graph
    {
        public int VerticesCount { get; private set; }
        public List<Edge> Edges { get; private set; }

        public Graph(int verticesCount)
        {
            VerticesCount = verticesCount;
            Edges = new List<Edge>();
        }

        public void AddEdge(int u, int v, int weight)
        {
            Edges.Add(new Edge(u, v, weight));
        }
    }

    public class ChinesePostmanSolver
    {
        private Graph _graph;

        public ChinesePostmanSolver(Graph graph)
        {
            _graph = graph;
        }

        public void Solve()
        {
            // 1. Calculate degrees of all vertices
            int[] degrees = new int[_graph.VerticesCount];
            int totalWeight = 0;

            foreach (var edge in _graph.Edges)
            {
                degrees[edge.U]++;
                degrees[edge.V]++;
                totalWeight += edge.Weight;
            }

            // 2. Find odd degree vertices
            List<int> oddVertices = new List<int>();
            for (int i = 0; i < _graph.VerticesCount; i++)
            {
                if (degrees[i] % 2 != 0)
                {
                    oddVertices.Add(i);
                }
            }

            if (oddVertices.Count == 0)
            {
                Console.WriteLine("The graph has an Eulerian Circuit.");
                Console.WriteLine($"Total Cost: {totalWeight}");
                PrintEulerianCircuit(_graph.Edges);
                return;
            }

            // 3. Find shortest paths between all odd vertices using Dijkstra
            var shortestPaths = ComputeShortestPaths(oddVertices);

            // 4. Find Minimum Weight Perfect Matching for the odd vertices
            var matching = FindMinimumWeightPerfectMatching(oddVertices, shortestPaths);

            // 5. Add matching edges to the graph (creating a multigraph)
            var eulerianEdges = new List<Edge>(_graph.Edges);
            int additionalWeight = 0;

            foreach (var match in matching)
            {
                int u = match.Item1;
                int v = match.Item2;
                int weight = shortestPaths[u][v];
                additionalWeight += weight;

                // We add a virtual edge directly connecting the nodes for simplicity.
                // In a complete implementation, you would add the actual edges of the shortest path.
                eulerianEdges.Add(new Edge(u, v, weight));
            }

            Console.WriteLine($"Number of odd vertices: {oddVertices.Count}");
            Console.WriteLine($"Original Total Weight: {totalWeight}");
            Console.WriteLine($"Additional Weight for matching: {additionalWeight}");
            Console.WriteLine($"Total Cost of Chinese Postman Tour: {totalWeight + additionalWeight}");

            // 6. Find and print the Eulerian circuit
            PrintEulerianCircuit(eulerianEdges);
        }

        private Dictionary<int, Dictionary<int, int>> ComputeShortestPaths(List<int> oddVertices)
        {
            var paths = new Dictionary<int, Dictionary<int, int>>();
            foreach (var startNode in oddVertices)
            {
                paths[startNode] = Dijkstra(startNode);
            }
            return paths;
        }

        private Dictionary<int, int> Dijkstra(int startNode)
        {
            var distances = new Dictionary<int, int>();
            var pq = new PriorityQueue<int, int>();

            for (int i = 0; i < _graph.VerticesCount; i++)
            {
                distances[i] = int.MaxValue;
            }

            distances[startNode] = 0;
            pq.Enqueue(startNode, 0);

            while (pq.Count > 0)
            {
                int current = pq.Dequeue();

                foreach (var edge in _graph.Edges)
                {
                    if (edge.U == current || edge.V == current)
                    {
                        int neighbor = (edge.U == current) ? edge.V : edge.U;
                        int newDist = distances[current] + edge.Weight;

                        if (newDist < distances[neighbor])
                        {
                            distances[neighbor] = newDist;
                            pq.Enqueue(neighbor, newDist);
                        }
                    }
                }
            }

            return distances;
        }

        private List<Tuple<int, int>> FindMinimumWeightPerfectMatching(List<int> oddVertices, Dictionary<int, Dictionary<int, int>> shortestPaths)
        {
            // Simple backtracking approach for finding minimum weight perfect matching.
            // This is efficient enough for small numbers of odd vertices, as a graph typically doesn't have many odd nodes.
            bool[] paired = new bool[_graph.VerticesCount];
            List<Tuple<int, int>> bestMatching = null;
            int minWeight = int.MaxValue;

            void Backtrack(List<Tuple<int, int>> currentMatching, int currentWeight)
            {
                if (currentWeight >= minWeight) return; // Pruning

                int firstUnpaired = -1;
                foreach (var v in oddVertices)
                {
                    if (!paired[v])
                    {
                        firstUnpaired = v;
                        break;
                    }
                }

                if (firstUnpaired == -1)
                {
                    minWeight = currentWeight;
                    bestMatching = new List<Tuple<int, int>>(currentMatching);
                    return;
                }

                paired[firstUnpaired] = true;

                foreach (var v in oddVertices)
                {
                    if (!paired[v] && v != firstUnpaired)
                    {
                        paired[v] = true;
                        currentMatching.Add(new Tuple<int, int>(firstUnpaired, v));
                        
                        Backtrack(currentMatching, currentWeight + shortestPaths[firstUnpaired][v]);
                        
                        currentMatching.RemoveAt(currentMatching.Count - 1);
                        paired[v] = false;
                    }
                }

                paired[firstUnpaired] = false;
            }

            Backtrack(new List<Tuple<int, int>>(), 0);

            return bestMatching ?? new List<Tuple<int, int>>();
        }

        private void PrintEulerianCircuit(List<Edge> edges)
        {
            // Hierholzer's Algorithm
            var adj = new Dictionary<int, List<Edge>>();
            for (int i = 0; i < _graph.VerticesCount; i++)
                adj[i] = new List<Edge>();

            var availableEdges = new List<Edge>(edges);
            
            foreach (var edge in availableEdges)
            {
                adj[edge.U].Add(edge);
                adj[edge.V].Add(edge);
            }

            int startVertex = 0;
            if (availableEdges.Count > 0)
                startVertex = availableEdges[0].U;

            var stack = new Stack<int>();
            var circuit = new List<int>();

            stack.Push(startVertex);
            
            while (stack.Count > 0)
            {
                int curr = stack.Peek();

                if (adj[curr].Count > 0)
                {
                    var edge = adj[curr].First();
                    int next = (edge.U == curr) ? edge.V : edge.U;
                    
                    stack.Push(next);
                    
                    // Remove the edge from adjacency list of both vertices
                    adj[curr].Remove(edge);
                    adj[next].Remove(edge);
                }
                else
                {
                    circuit.Add(curr);
                    stack.Pop();
                }
            }

            circuit.Reverse();
            Console.WriteLine("Eulerian Circuit:");
            Console.WriteLine(string.Join(" -> ", circuit));
        }
    }

    // A simple PriorityQueue implementation for Dijkstra
    public class PriorityQueue<TElement, TPriority> where TPriority : IComparable<TPriority>
    {
        private List<Tuple<TElement, TPriority>> _elements = new List<Tuple<TElement, TPriority>>();

        public int Count => _elements.Count;

        public void Enqueue(TElement element, TPriority priority)
        {
            _elements.Add(new Tuple<TElement, TPriority>(element, priority));
            _elements.Sort((x, y) => x.Item2.CompareTo(y.Item2)); 
        }

        public TElement Dequeue()
        {
            if (_elements.Count == 0) throw new InvalidOperationException("Queue is empty.");
            var bestItem = _elements[0];
            _elements.RemoveAt(0);
            return bestItem.Item1;
        }
    }
    
    class Program
    {
        static void Main(string[] args)
        {
            // Simple Graph Setup
            Graph g = new Graph(4);
            g.AddEdge(0, 1, 10);
            g.AddEdge(0, 2, 15);
            g.AddEdge(1, 2, 20);
            g.AddEdge(1, 3, 25);
            g.AddEdge(2, 3, 30);
            
            ChinesePostmanSolver solver = new ChinesePostmanSolver(g);
            solver.Solve();
        }
    }
}
