using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Edge {
    public int u, v;
    public long w;
}

class Scanner {
    private Stream stream;
    private byte[] buffer = new byte[1024 * 64];
    private int head;
    private int tail;

    public Scanner(Stream stream) {
        this.stream = stream;
    }

    public int NextInt() {
        int c = Read();
        while (c <= 32) {
            if (c == -1) return -1;
            c = Read();
        }
        bool negative = false;
        if (c == '-') {
            negative = true;
            c = Read();
        }
        int res = 0;
        while (c > 32) {
            res = res * 10 + (c - '0');
            c = Read();
        }
        return negative ? -res : res;
    }
    
    public long NextLong() {
        int c = Read();
        while (c <= 32) {
            if (c == -1) return -1;
            c = Read();
        }
        bool negative = false;
        if (c == '-') {
            negative = true;
            c = Read();
        }
        long res = 0;
        while (c > 32) {
            res = res * 10 + (c - '0');
            c = Read();
        }
        return negative ? -res : res;
    }

    private int Read() {
        if (head >= tail) {
            head = 0;
            tail = stream.Read(buffer, 0, buffer.Length);
            if (tail == 0) return -1;
        }
        return buffer[head++];
    }
}

class Program {
    static int N, M;
    static Edge[] edges;
    static List<int>[] adj;
    static int[] originalDegree;
    static List<Edge> duplicated_edges = new List<Edge>();
    
    static void Main(string[] args) {
        Scanner sc = new Scanner(Console.OpenStandardInput());
        N = sc.NextInt();
        if (N == -1) return;
        M = sc.NextInt();
        
        edges = new Edge[M];
        adj = new List<int>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<int>();
        originalDegree = new int[N];
        
        for (int i = 0; i < M; i++) {
            edges[i] = new Edge();
            edges[i].u = sc.NextInt();
            edges[i].v = sc.NextInt();
            edges[i].w = sc.NextLong();
            
            adj[edges[i].u].Add(i);
            adj[edges[i].v].Add(i);
            originalDegree[edges[i].u]++;
            originalDegree[edges[i].v]++;
        }
        
        List<int> odds = new List<int>();
        for (int i = 0; i < N; i++) {
            if (originalDegree[i] % 2 != 0) odds.Add(i);
        }
        
        int K = odds.Count;
        if (K > 0) {
            if (K <= 20) {
                SolveExactDP(odds);
            } else if ((long)K * M <= 200_000_000L) {
                SolveHeuristicMatching(odds);
            } else {
                SolveMSTHeuristic(odds);
            }
        }
        
        BuildAndPrintEulerianCircuit();
    }
    
    static long[,] ComputeDistances(List<int> odds) {
        int K = odds.Count;
        long[,] dist = new long[K, K];
        long[] d = new long[N];
        
        for (int i = 0; i < K; i++) {
            Array.Fill(d, long.MaxValue);
            d[odds[i]] = 0;
            var pq = new PriorityQueue<int, long>();
            pq.Enqueue(odds[i], 0);
            
            while (pq.Count > 0) {
                pq.TryDequeue(out int u, out long currentDist);
                if (currentDist > d[u]) continue;
                
                foreach (var edgeIndex in adj[u]) {
                    var e = edges[edgeIndex];
                    int v = (e.u == u) ? e.v : e.u;
                    long newDist = currentDist + e.w;
                    if (newDist < d[v]) {
                        d[v] = newDist;
                        pq.Enqueue(v, newDist);
                    }
                }
            }
            
            for (int j = 0; j < K; j++) {
                dist[i, j] = d[odds[j]];
            }
        }
        return dist;
    }
    
    static void AddPathEdges(int startNode, int endNode) {
        long[] d = new long[N];
        int[] parentEdge = new int[N];
        Array.Fill(d, long.MaxValue);
        Array.Fill(parentEdge, -1);
        
        d[startNode] = 0;
        var pq = new PriorityQueue<int, long>();
        pq.Enqueue(startNode, 0);
        
        while (pq.Count > 0) {
            pq.TryDequeue(out int u, out long currentDist);
            if (currentDist > d[u]) continue;
            if (u == endNode) break;
            
            foreach (var edgeIndex in adj[u]) {
                var e = edges[edgeIndex];
                int v = (e.u == u) ? e.v : e.u;
                long newDist = currentDist + e.w;
                if (newDist < d[v]) {
                    d[v] = newDist;
                    parentEdge[v] = edgeIndex;
                    pq.Enqueue(v, newDist);
                }
            }
        }
        
        int curr = endNode;
        while (curr != startNode) {
            int eIdx = parentEdge[curr];
            duplicated_edges.Add(edges[eIdx]);
            curr = (edges[eIdx].u == curr) ? edges[eIdx].v : edges[eIdx].u;
        }
    }
    
    static void SolveExactDP(List<int> odds) {
        int K = odds.Count;
        long[,] dist = ComputeDistances(odds);
        
        long[] dp = new long[1 << K];
        int[] parent = new int[1 << K];
        Array.Fill(dp, long.MaxValue / 2);
        dp[0] = 0;
        
        for (int mask = 0; mask < (1 << K); mask++) {
            if (dp[mask] >= long.MaxValue / 2) continue;
            
            int i = -1;
            for (int b = 0; b < K; b++) {
                if ((mask & (1 << b)) == 0) {
                    i = b;
                    break;
                }
            }
            if (i == -1) continue;
            
            for (int j = i + 1; j < K; j++) {
                if ((mask & (1 << j)) == 0) {
                    int nmask = mask | (1 << i) | (1 << j);
                    if (dp[mask] + dist[i, j] < dp[nmask]) {
                        dp[nmask] = dp[mask] + dist[i, j];
                        parent[nmask] = mask;
                    }
                }
            }
        }
        
        int currMask = (1 << K) - 1;
        while (currMask > 0) {
            int pMask = parent[currMask];
            int diff = currMask ^ pMask;
            int u = -1, v = -1;
            for (int b = 0; b < K; b++) {
                if ((diff & (1 << b)) != 0) {
                    if (u == -1) u = b;
                    else v = b;
                }
            }
            AddPathEdges(odds[u], odds[v]);
            currMask = pMask;
        }
    }
    
    static void SolveHeuristicMatching(List<int> odds) {
        int K = odds.Count;
        long[,] dist = ComputeDistances(odds);
        
        bool[] matched = new bool[K];
        int[] match = new int[K];
        Array.Fill(match, -1);
        
        List<(long w, int u, int v)> edgeList = new List<(long, int, int)>();
        for (int i = 0; i < K; i++) {
            for (int j = i + 1; j < K; j++) {
                edgeList.Add((dist[i, j], i, j));
            }
        }
        edgeList.Sort((a, b) => a.w.CompareTo(b.w));
        
        foreach (var e in edgeList) {
            if (!matched[e.u] && !matched[e.v]) {
                matched[e.u] = true;
                matched[e.v] = true;
                match[e.u] = e.v;
                match[e.v] = e.u;
            }
        }
        
        bool improved = true;
        while (improved) {
            improved = false;
            for (int i = 0; i < K; i++) {
                int u1 = i;
                int v1 = match[i];
                if (u1 > v1) continue;
                
                for (int j = i + 1; j < K; j++) {
                    int u2 = j;
                    int v2 = match[j];
                    if (u2 > v2) continue;
                    if (u1 == u2 || u1 == v2 || v1 == u2 || v1 == v2) continue;
                    
                    long currentCost = dist[u1, v1] + dist[u2, v2];
                    long swap1Cost = dist[u1, u2] + dist[v1, v2];
                    long swap2Cost = dist[u1, v2] + dist[v1, u2];
                    
                    if (swap1Cost < currentCost && swap1Cost <= swap2Cost) {
                        match[u1] = u2; match[u2] = u1;
                        match[v1] = v2; match[v2] = v1;
                        improved = true;
                        break;
                    } else if (swap2Cost < currentCost) {
                        match[u1] = v2; match[v2] = u1;
                        match[v1] = u2; match[u2] = v1;
                        improved = true;
                        break;
                    }
                }
                if (improved) break;
            }
        }
        
        bool[] done = new bool[K];
        for (int i = 0; i < K; i++) {
            if (!done[i]) {
                int j = match[i];
                AddPathEdges(odds[i], odds[j]);
                done[i] = true;
                done[j] = true;
            }
        }
    }
    
    static void SolveMSTHeuristic(List<int> odds) {
        bool[] inTree = new bool[N];
        long[] minEdge = new long[N];
        int[] parentEdge = new int[N];
        Array.Fill(minEdge, long.MaxValue);
        Array.Fill(parentEdge, -1);
        
        var pq = new PriorityQueue<int, long>();
        pq.Enqueue(0, 0);
        minEdge[0] = 0;
        
        List<int>[] mstAdj = new List<int>[N];
        for(int i = 0; i < N; i++) mstAdj[i] = new List<int>();
        
        while (pq.Count > 0) {
            pq.TryDequeue(out int u, out long d);
            if (inTree[u]) continue;
            inTree[u] = true;
            
            if (parentEdge[u] != -1) {
                int p = (edges[parentEdge[u]].u == u) ? edges[parentEdge[u]].v : edges[parentEdge[u]].u;
                mstAdj[u].Add(parentEdge[u]);
                mstAdj[p].Add(parentEdge[u]);
            }
            
            foreach (var edgeIndex in adj[u]) {
                var e = edges[edgeIndex];
                int v = (e.u == u) ? e.v : e.u;
                if (!inTree[v] && e.w < minEdge[v]) {
                    minEdge[v] = e.w;
                    parentEdge[v] = edgeIndex;
                    pq.Enqueue(v, e.w);
                }
            }
        }
        
        int[] order = new int[N];
        int orderIdx = 0;
        
        int[] parentNode = new int[N];
        int[] edgeFromParent = new int[N];
        Array.Fill(parentNode, -1);
        Array.Fill(edgeFromParent, -1);
        
        Queue<int> q = new Queue<int>();
        q.Enqueue(0);
        bool[] visited = new bool[N];
        visited[0] = true;
        
        while (q.Count > 0) {
            int u = q.Dequeue();
            order[orderIdx++] = u;
            
            foreach (var edgeIndex in mstAdj[u]) {
                if (edgeIndex == edgeFromParent[u]) continue;
                var e = edges[edgeIndex];
                int v = (e.u == u) ? e.v : e.u;
                if (!visited[v]) {
                    visited[v] = true;
                    parentNode[v] = u;
                    edgeFromParent[v] = edgeIndex;
                    q.Enqueue(v);
                }
            }
        }
        
        bool[] isOdd = new bool[N];
        for (int i = 0; i < N; i++) {
            isOdd[i] = (originalDegree[i] % 2 != 0);
        }
        
        for (int i = orderIdx - 1; i >= 0; i--) {
            int u = order[i];
            if (u != 0) {
                if (isOdd[u]) {
                    duplicated_edges.Add(edges[edgeFromParent[u]]);
                    isOdd[u] = !isOdd[u];
                    isOdd[parentNode[u]] = !isOdd[parentNode[u]];
                }
            }
        }
    }
    
    static void BuildAndPrintEulerianCircuit() {
        int totalEdges = M + duplicated_edges.Count;
        int[] head = new int[N];
        Array.Fill(head, -1);
        int[] to = new int[totalEdges * 2];
        int[] edgeId = new int[totalEdges * 2];
        int[] next = new int[totalEdges * 2];
        int edgeCnt = 0;
        
        void AddMultigraphEdge(int u, int v, int id) {
            to[edgeCnt] = v;
            edgeId[edgeCnt] = id;
            next[edgeCnt] = head[u];
            head[u] = edgeCnt++;
            
            to[edgeCnt] = u;
            edgeId[edgeCnt] = id;
            next[edgeCnt] = head[v];
            head[v] = edgeCnt++;
        }
        
        for (int i = 0; i < M; i++) {
            AddMultigraphEdge(edges[i].u, edges[i].v, i);
        }
        int curIdx = M;
        foreach (var e in duplicated_edges) {
            AddMultigraphEdge(e.u, e.v, curIdx);
            curIdx++;
        }
        
        bool[] usedEdge = new bool[totalEdges];
        int[] curHead = (int[])head.Clone();
        
        Stack<int> stack = new Stack<int>(totalEdges + 1);
        List<int> route = new List<int>(totalEdges + 1);
        
        stack.Push(0);
        while (stack.Count > 0) {
            int u = stack.Peek();
            bool found = false;
            while (curHead[u] != -1) {
                int eId = edgeId[curHead[u]];
                int v = to[curHead[u]];
                curHead[u] = next[curHead[u]];
                if (!usedEdge[eId]) {
                    usedEdge[eId] = true;
                    stack.Push(v);
                    found = true;
                    break;
                }
            }
            if (!found) {
                route.Add(u);
                stack.Pop();
            }
        }
        
        route.Reverse();
        
        using (StreamWriter writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 65536)) {
            for (int i = 0; i < route.Count; i++) {
                writer.Write(route[i]);
                if (i < route.Count - 1) writer.Write(" ");
            }
            writer.WriteLine();
        }
    }
}
