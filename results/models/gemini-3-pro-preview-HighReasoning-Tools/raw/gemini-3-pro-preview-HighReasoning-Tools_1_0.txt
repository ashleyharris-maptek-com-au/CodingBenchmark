using namespace std;

// Fast I/O
static const int BUF_SIZE = 1 << 24;
char in_buf[BUF_SIZE];
int in_pos = 0, in_len = 0;

inline char get_char() {
    if (in_pos == in_len) {
        in_len = (int)fread(in_buf, 1, BUF_SIZE, stdin);
        in_pos = 0;
        if (in_len == 0) return EOF;
    }
    return in_buf[in_pos++];
}

inline void skip_whitespace() {
    char c;
    while ((c = get_char()) != EOF && c <= ' ');
    if (c != EOF) in_pos--;
}

inline int read_int() {
    skip_whitespace();
    char c = get_char();
    if (c == EOF) return 0;
    int val = 0;
    while (c >= '0' && c <= '9') {
        val = val * 10 + (c - '0');
        c = get_char();
    }
    if (c != EOF) in_pos--;
    return val;
}

inline float read_float() {
    skip_whitespace();
    char c = get_char();
    if (c == EOF) return 0.0f;
    bool neg = false;
    if (c == '-') { neg = true; c = get_char(); }
    else if (c == '+') { c = get_char(); }
    
    double val = 0.0;
    while (c >= '0' && c <= '9') {
        val = val * 10.0 + (c - '0');
        c = get_char();
    }
    if (c == '.') {
        c = get_char();
        double frac = 1.0;
        while (c >= '0' && c <= '9') {
            frac *= 0.1;
            val += (c - '0') * frac;
            c = get_char();
        }
    }
    if (c == 'e' || c == 'E') {
        c = get_char();
        bool eneg = false;
        if (c == '-') { eneg = true; c = get_char(); }
        else if (c == '+') { c = get_char(); }
        int exp = 0;
        while (c >= '0' && c <= '9') {
            exp = exp * 10 + (c - '0');
            c = get_char();
        }
        if (eneg) {
            while(exp--) val *= 0.1;
        } else {
            while(exp--) val *= 10.0;
        }
    }
    if (c != EOF) in_pos--;
    return (float)(neg ? -val : val);
}

// Fast output
char out_buf[BUF_SIZE];
int out_pos = 0;

inline void flush_out() {
    if (out_pos > 0) {
        fwrite(out_buf, 1, out_pos, stdout);
        out_pos = 0;
    }
}

inline void write_int(int val) {
    if (val == 0) {
        if (out_pos + 2 > BUF_SIZE) flush_out();
        out_buf[out_pos++] = '0';
        out_buf[out_pos++] = ' ';
        return;
    }
    char temp[12];
    int t = 0;
    while (val > 0) {
        temp[t++] = '0' + (val % 10);
        val /= 10;
    }
    if (out_pos + t + 1 > BUF_SIZE) flush_out();
    while (t > 0) {
        out_buf[out_pos++] = temp[--t];
    }
    out_buf[out_pos++] = ' ';
}

// Hilbert curve mapping
uint64_t xy2d(uint32_t n, uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (uint32_t s = n / 2; s > 0; s /= 2) {
        uint32_t rx = (x & s) > 0;
        uint32_t ry = (y & s) > 0;
        d += (uint64_t)s * (uint64_t)s * ((3 * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) {
                x = n - 1 - x;
                y = n - 1 - y;
            }
            uint32_t t = x; x = y; y = t;
        }
    }
    return d;
}

struct City {
    int id;
    uint64_t h;
};

// Main Solver
int main() {
    // Optimizing basic IO streams if they are accidentally used instead of fast IO
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

#ifdef _WIN32
    _setmode(_fileno(stdin), _O_BINARY);
    _setmode(_fileno(stdout), _O_BINARY);
#endif

    int n = read_int();
    if (n == 0) return 0;
    
    vector<float> x(n), y(n);
    for (int i = 0; i < n; ++i) {
        x[i] = read_float();
        y[i] = read_float();
    }
    
    vector<int> tour;
    tour.reserve(n);
    
    if (n <= 30000) {
        int best_start = 0;
        float global_best_len = 1e30f;
        vector<int> global_best_tour;
        
        int start_nodes = (n <= 1000) ? n : 1; 
        
        for (int s = 0; s < start_nodes; ++s) {
            vector<int> current_tour;
            current_tour.reserve(n);
            vector<int> unvisited;
            unvisited.reserve(n - 1);
            for (int i = 0; i < n; ++i) {
                if (i != s) unvisited.push_back(i);
            }
            current_tour.push_back(s);
            int curr = s;
            for (int i = 1; i < n; ++i) {
                float best_d = 1e30f;
                int best_idx = -1;
                float cx = x[curr], cy = y[curr];
                int m = (int)unvisited.size();
                for (int j = 0; j < m; ++j) {
                    int v = unvisited[j];
                    float dx = cx - x[v];
                    float dy = cy - y[v];
                    float d = dx*dx + dy*dy;
                    if (d < best_d) {
                        best_d = d;
                        best_idx = j;
                    }
                }
                curr = unvisited[best_idx];
                current_tour.push_back(curr);
                unvisited[best_idx] = unvisited.back();
                unvisited.pop_back();
            }
            
            float len = 0;
            for (int i = 0; i < n - 1; ++i) {
                float dx = x[current_tour[i]] - x[current_tour[i+1]];
                float dy = y[current_tour[i]] - y[current_tour[i+1]];
                len += sqrt(dx*dx + dy*dy);
            }
            float dx = x[current_tour[n-1]] - x[current_tour[0]];
            float dy = y[current_tour[n-1]] - y[current_tour[0]];
            len += sqrt(dx*dx + dy*dy);
            
            if (len < global_best_len) {
                global_best_len = len;
                global_best_tour = current_tour;
            }
        }
        
        tour = global_best_tour;
        
        int window_size = n;
        bool improved = true;
        int max_iter = 10;
        while (improved && max_iter-- > 0) {
            improved = false;
            for (int i = 0; i < n - 2; ++i) {
                int max_j = min(n - 1, i + window_size);
                for (int j = i + 2; j < max_j; ++j) {
                    int t_i = tour[i];
                    int t_next = tour[i+1];
                    int t_j = tour[j];
                    int t_jnext = tour[j+1];
                    
                    float dx1 = x[t_i]-x[t_next], dy1 = y[t_i]-y[t_next];
                    float dx2 = x[t_j]-x[t_jnext], dy2 = y[t_j]-y[t_jnext];
                    float dx3 = x[t_i]-x[t_j], dy3 = y[t_i]-y[t_j];
                    float dx4 = x[t_next]-x[t_jnext], dy4 = y[t_next]-y[t_jnext];
                    
                    float d_old = sqrt(dx1*dx1 + dy1*dy1) + sqrt(dx2*dx2 + dy2*dy2);
                    float d_new = sqrt(dx3*dx3 + dy3*dy3) + sqrt(dx4*dx4 + dy4*dy4);
                    
                    if (d_new < d_old - 1e-5f) {
                        reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                        improved = true;
                    }
                }
            }
        }
        
        int zero_idx = 0;
        for (int i = 0; i < n; ++i) {
            if (tour[i] == 0) {
                zero_idx = i;
                break;
            }
        }
        if (zero_idx != 0) {
            vector<int> aligned_tour;
            aligned_tour.reserve(n);
            for (int i = zero_idx; i < n; ++i) aligned_tour.push_back(tour[i]);
            for (int i = 0; i < zero_idx; ++i) aligned_tour.push_back(tour[i]);
            tour = aligned_tour;
        }
        
    } else {
        float min_x = x[0], max_x = x[0];
        float min_y = y[0], max_y = y[0];
        for (int i = 1; i < n; ++i) {
            if (x[i] < min_x) min_x = x[i];
            if (x[i] > max_x) max_x = x[i];
            if (y[i] < min_y) min_y = y[i];
            if (y[i] > max_y) max_y = y[i];
        }
        float range_x = max_x - min_x;
        float range_y = max_y - min_y;
        float range = max(range_x, range_y);
        if (range == 0) range = 1;

        uint32_t HILBERT_ORDER = 21;
        uint32_t grid_size = 1u << HILBERT_ORDER;

        vector<City> cities(n);
        for (int i = 0; i < n; ++i) {
            cities[i].id = i;
            uint32_t gx = (uint32_t)((x[i] - min_x) / range * (grid_size - 1));
            uint32_t gy = (uint32_t)((y[i] - min_y) / range * (grid_size - 1));
            cities[i].h = xy2d(grid_size, gx, gy);
        }
        sort(cities.begin(), cities.end(), [](const City& a, const City& b){
            return a.h < b.h;
        });
        
        int zero_idx = 0;
        for (int i = 0; i < n; ++i) {
            if (cities[i].id == 0) {
                zero_idx = i;
                break;
            }
        }
        for (int i = zero_idx; i < n; ++i) tour.push_back(cities[i].id);
        for (int i = 0; i < zero_idx; ++i) tour.push_back(cities[i].id);
        
        int window_size = 50; 
        if (n <= 100000) window_size = 200;
        bool improved = true;
        int max_iter = (n <= 100000) ? 5 : 1; 
        while (improved && max_iter-- > 0) {
            improved = false;
            for (int i = 0; i < n - 2; ++i) {
                int max_j = min(n - 1, i + window_size);
                for (int j = i + 2; j < max_j; ++j) {
                    int t_i = tour[i];
                    int t_next = tour[i+1];
                    int t_j = tour[j];
                    int t_jnext = tour[j+1];
                    
                    float dx1 = x[t_i]-x[t_next], dy1 = y[t_i]-y[t_next];
                    float dx2 = x[t_j]-x[t_jnext], dy2 = y[t_j]-y[t_jnext];
                    float dx3 = x[t_i]-x[t_j], dy3 = y[t_i]-y[t_j];
                    float dx4 = x[t_next]-x[t_jnext], dy4 = y[t_next]-y[t_jnext];
                    
                    float d_old = sqrt(dx1*dx1 + dy1*dy1) + sqrt(dx2*dx2 + dy2*dy2);
                    float d_new = sqrt(dx3*dx3 + dy3*dy3) + sqrt(dx4*dx4 + dy4*dy4);
                    
                    if (d_new < d_old - 1e-5f) {
                        reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                        improved = true;
                    }
                }
            }
        }
    }
    
    for (int i = 0; i < n; ++i) {
        write_int(tour[i]);
    }
    
    if (out_pos + 1 > BUF_SIZE) flush_out();
    out_buf[out_pos++] = '\n';
    flush_out();
    
    return 0;
}
