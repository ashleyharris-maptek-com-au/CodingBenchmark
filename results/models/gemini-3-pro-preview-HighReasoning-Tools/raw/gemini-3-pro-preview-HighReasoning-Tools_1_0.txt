#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const int INF = 1e9; // Representing infinity

// Solves TSP using Dynamic Programming with Bitmasking (Held-Karp algorithm)
// Time Complexity: O(n^2 * 2^n)
// Space Complexity: O(n * 2^n)
int tsp(int mask, int pos, int n, const vector<vector<int>>& dist, vector<vector<int>>& dp) {
    // Base case: If all cities have been visited
    // mask == (1 << n) - 1 means all bits from 0 to n-1 are set to 1
    if (mask == (1 << n) - 1) {
        return dist[pos][0]; // Return cost to go back to the starting city (0)
    }

    // Return the precalculated result if available
    if (dp[mask][pos] != -1) {
        return dp[mask][pos];
    }

    int min_cost = INF;

    // Try visiting all unvisited cities
    for (int city = 0; city < n; ++city) {
        // If 'city' is not visited yet in the current mask
        if ((mask & (1 << city)) == 0) {
            // New cost = distance to 'city' + TSP from 'city' with updated mask
            int new_cost = dist[pos][city] + tsp(mask | (1 << city), city, n, dist, dp);
            min_cost = min(min_cost, new_cost);
        }
    }

    // Memoize and return the result
    return dp[mask][pos] = min_cost;
}

int main() {
    // Example distance matrix (graph representation)
    // 4 cities: 0, 1, 2, 3
    vector<vector<int>> dist = {
        {0, 10, 15, 20},
        {10, 0, 35, 25},
        {15, 35, 0, 30},
        {20, 25, 30, 0}
    };
    
    int n = dist.size();

    // dp table initialization: 2^n rows, n columns, initialized with -1
    // 1 << n calculates 2^n
    vector<vector<int>> dp(1 << n, vector<int>(n, -1));

    // Start TSP from city 0
    // Initial mask = 1 (since city 0 is visited), position = 0
    int min_path_cost = tsp(1, 0, n, dist, dp);

    cout << "Minimum Cost of Travelling Salesman Tour: " << min_path_cost << endl;
    // Expected output for this matrix: 80 (0 -> 1 -> 3 -> 2 -> 0 => 10 + 25 + 30 + 15 = 80)

    return 0;
}
