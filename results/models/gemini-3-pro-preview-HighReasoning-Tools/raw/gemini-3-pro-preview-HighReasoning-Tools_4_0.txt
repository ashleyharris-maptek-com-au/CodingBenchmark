#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <chrono>
#include <map>
#include <unordered_set>
#include <queue>
#include <tuple>
#include <iomanip>
#include <random>

using namespace std;

struct Point3 {
    double x, y, z;
};

struct Face {
    int a, b, c;
};

struct Plane {
    double nx, ny, nz, d;
    double d_even, d_odd;
};

struct Result {
    double cx, cy, cz;
    double rx, ry, rz;
};

Point3 cross_p(Point3 a, Point3 b) {
    return {a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}

Point3 sub_p(Point3 a, Point3 b) {
    return {a.x - b.x, a.y - b.y, a.z - b.z};
}

double dot_p(Point3 a, Point3 b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

double length_p(Point3 a) {
    return sqrt(dot_p(a, a));
}

struct TupleHash {
    size_t operator()(const tuple<int, int, int>& t) const {
        auto [x, y, z] = t;
        size_t h1 = std::hash<int>{}(x);
        size_t h2 = std::hash<int>{}(y);
        size_t h3 = std::hash<int>{}(z);
        return h1 ^ (h2 << 1) ^ (h3 << 2);
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    auto start_time = chrono::steady_clock::now();

    int V;
    double e;
    if (!(cin >> V >> e)) return 0;

    vector<Point3> pts(V);
    for (int i = 0; i < V; i++) {
        cin >> pts[i].x >> pts[i].y >> pts[i].z;
    }

    // Sort to remove duplicates efficiently
    sort(pts.begin(), pts.end(), [](const Point3& a, const Point3& b) {
        if (abs(a.x - b.x) > 1e-7) return a.x < b.x;
        if (abs(a.y - b.y) > 1e-7) return a.y < b.y;
        return a.z < b.z;
    });

    vector<Point3> unique_pts;
    for (auto p : pts) {
        if (unique_pts.empty() || length_p(sub_p(p, unique_pts.back())) > 1e-7) {
            unique_pts.push_back(p);
        }
    }
    pts = unique_pts;
    int n = pts.size();

    if (n < 4) {
        cout << 0 << "\n";
        return 0;
    }

    int p1 = -1;
    double max_dist = -1;
    for (int i = 1; i < n; i++) {
        double d = length_p(sub_p(pts[i], pts[0]));
        if (d > max_dist) { max_dist = d; p1 = i; }
    }
    if (p1 == -1) { cout << 0 << "\n"; return 0; }
    swap(pts[1], pts[p1]);

    int p2 = -1;
    double max_area = -1;
    for (int i = 2; i < n; i++) {
        double a = length_p(cross_p(sub_p(pts[i], pts[0]), sub_p(pts[1], pts[0])));
        if (a > max_area) { max_area = a; p2 = i; }
    }
    if (p2 == -1) { cout << 0 << "\n"; return 0; }
    swap(pts[2], pts[p2]);

    int p3 = -1;
    double max_vol = -1;
    Point3 n012 = cross_p(sub_p(pts[1], pts[0]), sub_p(pts[2], pts[0]));
    double len_n = length_p(n012);
    if (len_n > 1e-7) {
        n012.x /= len_n; n012.y /= len_n; n012.z /= len_n;
        for (int i = 3; i < n; i++) {
            double v = abs(dot_p(n012, sub_p(pts[i], pts[0])));
            if (v > max_vol) { max_vol = v; p3 = i; }
        }
    }
    if (max_vol < 1e-5) { cout << 0 << "\n"; return 0; }
    swap(pts[3], pts[p3]);

    mt19937 rng(42);
    shuffle(pts.begin() + 4, pts.end(), rng);

    vector<Face> faces = {{0, 1, 2}, {0, 2, 3}, {0, 3, 1}, {1, 3, 2}};
    auto get_norm = [&](int a, int b, int c) {
        Point3 nm = cross_p(sub_p(pts[b], pts[a]), sub_p(pts[c], pts[a]));
        double len = length_p(nm);
        return Point3{nm.x / len, nm.y / len, nm.z / len};
    };

    for (int i = 0; i < 4; i++) {
        Face& f = faces[i];
        Point3 nm = get_norm(f.a, f.b, f.c);
        double d = -dot_p(nm, pts[f.a]);
        int other = 6 - (f.a + f.b + f.c);
        if (dot_p(nm, pts[other]) + d > 0) {
            swap(f.b, f.c);
        }
    }

    for (int i = 4; i < n; i++) {
        vector<int> visible;
        vector<Point3> normals(faces.size());
        vector<double> ds(faces.size());
        for (size_t j = 0; j < faces.size(); j++) {
            normals[j] = get_norm(faces[j].a, faces[j].b, faces[j].c);
            ds[j] = -dot_p(normals[j], pts[faces[j].a]);
            if (dot_p(normals[j], pts[i]) + ds[j] > 1e-7) {
                visible.push_back(j);
            }
        }
        if (visible.empty()) continue;

        map<pair<int, int>, int> edge_counts;
        for (int idx : visible) {
            Face f = faces[idx];
            edge_counts[{f.a, f.b}]++;
            edge_counts[{f.b, f.c}]++;
            edge_counts[{f.c, f.a}]++;
        }

        vector<Face> next_faces;
        for (size_t j = 0; j < faces.size(); j++) {
            if (find(visible.begin(), visible.end(), j) == visible.end()) {
                next_faces.push_back(faces[j]);
            }
        }

        for (auto& kv : edge_counts) {
            if (kv.second == 1) {
                next_faces.push_back({kv.first.first, kv.first.second, i});
            }
        }
        faces = next_faces;
    }

    vector<Plane> planes;
    double scale = e / sqrt(8.0);
    vector<Point3> v_even = {
        {-scale, -scale, -scale},
        {scale, scale, -scale},
        {scale, -scale, scale},
        {-scale, scale, scale}
    };
    vector<Point3> v_odd = {
        {scale, scale, scale},
        {-scale, -scale, scale},
        {-scale, scale, -scale},
        {scale, -scale, -scale}
    };

    for (const auto& f : faces) {
        Point3 nm = get_norm(f.a, f.b, f.c);
        double d = -dot_p(nm, pts[f.a]);
        double max_even = -1e18;
        for (auto v : v_even) max_even = max(max_even, dot_p(nm, v));
        double max_odd = -1e18;
        for (auto v : v_odd) max_odd = max(max_odd, dot_p(nm, v));

        planes.push_back({nm.x, nm.y, nm.z, d, max_even, max_odd});
    }

    Point3 centroid = {0, 0, 0};
    for (auto p : pts) { centroid.x += p.x; centroid.y += p.y; centroid.z += p.z; }
    centroid.x /= n; centroid.y /= n; centroid.z /= n;

    vector<Point3> seeds;
    seeds.push_back(centroid);
    for (auto v : pts) {
        seeds.push_back({(centroid.x + v.x) / 2, (centroid.y + v.y) / 2, (centroid.z + v.z) / 2});
        seeds.push_back({(centroid.x * 3 + v.x) / 4, (centroid.y * 3 + v.y) / 4, (centroid.z * 3 + v.z) / 4});
    }

    double grid_len = e / sqrt(2.0);
    double R_sphere = e * sqrt(3.0 / 8.0);
    int best_k = -1;
    vector<Result> best_results;
    int attempts = 0;

    mt19937 offset_rng(12345);
    uniform_real_distribution<double> dist_offset(0.0, grid_len);

    while (true) {
        auto now = chrono::steady_clock::now();
        if (chrono::duration_cast<chrono::milliseconds>(now - start_time).count() > 28000) break; // 28s limit

        double dx = dist_offset(offset_rng);
        double dy = dist_offset(offset_rng);
        double dz = dist_offset(offset_rng);

        vector<Result> current_results;
        unordered_set<tuple<int, int, int>, TupleHash> visited;
        queue<tuple<int, int, int>> q;

        auto enqueue = [&](int X, int Y, int Z) {
            if (visited.find({X, Y, Z}) == visited.end()) {
                visited.insert({X, Y, Z});
                q.push({X, Y, Z});
            }
        };

        for (auto seed : seeds) {
            int X = floor((seed.x - dx) / grid_len - 0.5);
            int Y = floor((seed.y - dy) / grid_len - 0.5);
            int Z = floor((seed.z - dz) / grid_len - 0.5);
            enqueue(X, Y, Z);
            enqueue(X + 1, Y, Z); enqueue(X - 1, Y, Z);
            enqueue(X, Y + 1, Z); enqueue(X, Y - 1, Z);
            enqueue(X, Y, Z + 1); enqueue(X, Y, Z - 1);
        }

        while (!q.empty()) {
            auto [X, Y, Z] = q.front();
            q.pop();

            double cx = (X + 0.5) * grid_len + dx;
            double cy = (Y + 0.5) * grid_len + dy;
            double cz = (Z + 0.5) * grid_len + dz;

            bool explore = true;
            for (const auto& p : planes) {
                if (p.nx * cx + p.ny * cy + p.nz * cz + p.d > R_sphere + 1e-7) {
                    explore = false;
                    break;
                }
            }

            if (!explore) continue;

            bool even = (abs(X + Y + Z) % 2 == 0);
            bool valid_tet = true;
            for (const auto& p : planes) {
                double v = p.nx * cx + p.ny * cy + p.nz * cz + p.d;
                v += (even ? p.d_even : p.d_odd);
                if (v > 1e-7) { valid_tet = false; break; }
            }

            if (valid_tet) {
                if (even) {
                    current_results.push_back({cx, cy, cz, 2.25551552, -0.42053433, 2.03444393});
                } else {
                    current_results.push_back({cx, cy, cz, 0.88607713, 0.42053433, 2.03444393});
                }
            }

            enqueue(X + 1, Y, Z); enqueue(X - 1, Y, Z);
            enqueue(X, Y + 1, Z); enqueue(X, Y - 1, Z);
            enqueue(X, Y, Z + 1); enqueue(X, Y, Z - 1);
        }

        if ((int)current_results.size() > best_k) {
            best_k = current_results.size();
            best_results = current_results;
        }
        attempts++;
    }

    cout << max(0, best_k) << "\n";
    cout << fixed << setprecision(8);
    for (const auto& r : best_results) {
        cout << r.cx << " " << r.cy << " " << r.cz << " "
             << r.rx << " " << r.ry << " " << r.rz << "\n";
    }

    return 0;
}
