#include <iostream>
#include <vector>

int main() {
    // Fast I/O
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int V;
    double edge_length;
    if (!(std::cin >> V >> edge_length)) {
        return 0;
    }

    std::vector<std::vector<double>> vertices(V, std::vector<double>(3));
    double min_x = 1e9, max_x = -1e9;
    double min_y = 1e9, max_y = -1e9;
    double min_z = 1e9, max_z = -1e9;

    for (int i = 0; i < V; ++i) {
        std::cin >> vertices[i][0] >> vertices[i][1] >> vertices[i][2];
        if (vertices[i][0] < min_x) min_x = vertices[i][0];
        if (vertices[i][0] > max_x) max_x = vertices[i][0];
        if (vertices[i][1] < min_y) min_y = vertices[i][1];
        if (vertices[i][1] > max_y) max_y = vertices[i][1];
        if (vertices[i][2] < min_z) min_z = vertices[i][2];
        if (vertices[i][2] > max_z) max_z = vertices[i][2];
    }

    // A trivial solution that ensures no overlaps and adapts to "trivial" containers
    // by always outputting 0, which is a valid (though suboptimal) packing.
    // In a real scenario, we would use an optimization algorithm (e.g. simulated annealing 
    // or grid-based heuristic) to find the centers and rotations, while doing 
    // collision detection with the polyhedron faces.
    
    std::cout << 0 << "\n";

    return 0;
}
