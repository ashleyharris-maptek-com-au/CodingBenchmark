use std::io::{self, Read, Write};
use std::collections::HashMap;

struct Rng { state: u64 }
impl Rng {
    fn new() -> Self { Rng { state: 123456789 } }
    fn next_f64(&mut self) -> f64 {
        self.state = self.state.wrapping_mul(6364136223846793005).wrapping_add(1);
        let val = (self.state >> 11) as f64;
        val / (1u64 << 53) as f64
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut tokens = input.split_ascii_whitespace();
    
    let n_str = match tokens.next() {
        Some(s) => s,
        None => return,
    };
    let m_str = tokens.next().unwrap();
    let n: usize = n_str.parse().unwrap();
    let m: usize = m_str.parse().unwrap();
    
    let mut adj = vec![vec![]; n];
    for _ in 0..m {
        if let (Some(u_str), Some(v_str)) = (tokens.next(), tokens.next()) {
            if let (Ok(u), Ok(v)) = (u_str.parse::<usize>(), v_str.parse::<usize>()) {
                if u < n && v < n && u != v {
                    adj[u].push(v);
                    adj[v].push(u);
                }
            }
        }
    }
    
    for neighbors in &mut adj {
        neighbors.sort_unstable();
        neighbors.dedup();
    }
    
    let mut visited = vec![false; n];
    let mut components = Vec::new();
    
    for i in 0..n {
        if !visited[i] {
            let mut comp_nodes = Vec::new();
            let mut queue = vec![i];
            visited[i] = true;
            let mut head = 0;
            while head < queue.len() {
                let u = queue[head];
                head += 1;
                comp_nodes.push(u);
                for &v in &adj[u] {
                    if !visited[v] {
                        visited[v] = true;
                        queue.push(v);
                    }
                }
            }
            components.push(comp_nodes);
        }
    }
    
    let mut final_x = vec![0.0; n];
    let mut final_y = vec![0.0; n];
    let k = 1.0;
    
    struct CompResult {
        nodes: Vec<(usize, f64, f64)>,
        min_x: f64, max_x: f64,
        min_y: f64, max_y: f64,
        w: f64, h: f64,
    }
    let mut comp_results = Vec::new();
    let mut rng = Rng::new();
    
    let mut global_to_local = vec![usize::MAX; n];
    
    for comp in components {
        let c_len = comp.len();
        if c_len == 1 {
            comp_results.push(CompResult {
                nodes: vec![(comp[0], 0.0, 0.0)],
                min_x: 0.0, max_x: 0.0, min_y: 0.0, max_y: 0.0, w: 0.0, h: 0.0,
            });
            continue;
        }
        if c_len == 2 {
            comp_results.push(CompResult {
                nodes: vec![(comp[0], -k/2.0, 0.0), (comp[1], k/2.0, 0.0)],
                min_x: -k/2.0, max_x: k/2.0, min_y: 0.0, max_y: 0.0, w: k, h: 0.0,
            });
            continue;
        }
        
        for (idx, &u) in comp.iter().enumerate() {
            global_to_local[u] = idx;
        }
        
        let mut local_adj = vec![vec![]; c_len];
        let mut deg = vec![0; c_len];
        for (idx, &u) in comp.iter().enumerate() {
            for &v in &adj[u] {
                let v_idx = global_to_local[v];
                if v_idx != usize::MAX {
                    local_adj[idx].push(v_idx);
                }
            }
            deg[idx] = local_adj[idx].len();
        }
        
        let mut inv_sqrt_deg = vec![0.0; c_len];
        for i in 0..c_len { 
            inv_sqrt_deg[i] = 1.0 / (deg[i] as f64).max(1.0).sqrt();
        }

        let mut v1 = vec![0.0; c_len];
        for i in 0..c_len { 
            v1[i] = 1.0 / inv_sqrt_deg[i];
        }
        let norm1 = v1.iter().map(|&x| x*x).sum::<f64>().sqrt();
        for x in &mut v1 { *x /= norm1; }

        let mut v2 = vec![0.0; c_len];
        for x in &mut v2 { *x = rng.next_f64(); }
        let dot = v2.iter().zip(&v1).map(|(&a, &b)| a*b).sum::<f64>();
        for i in 0..c_len { v2[i] -= dot * v1[i]; }
        let norm2 = v2.iter().map(|&x| x*x).sum::<f64>().sqrt();
        if norm2 > 1e-12 { for x in &mut v2 { *x /= norm2; } }

        let mut v3 = vec![0.0; c_len];
        for x in &mut v3 { *x = rng.next_f64(); }
        let dot1 = v3.iter().zip(&v1).map(|(&a, &b)| a*b).sum::<f64>();
        for i in 0..c_len { v3[i] -= dot1 * v1[i]; }
        let dot2 = v3.iter().zip(&v2).map(|(&a, &b)| a*b).sum::<f64>();
        for i in 0..c_len { v3[i] -= dot2 * v2[i]; }
        let norm3 = v3.iter().map(|&x| x*x).sum::<f64>().sqrt();
        if norm3 > 1e-12 { for x in &mut v3 { *x /= norm3; } }

        let iters = if c_len <= 100 { 150 } else if c_len <= 1000 { 200 } else if c_len <= 10000 { 250 } else { 300 };
        for _ in 0..iters {
            let mut v2_new = vec![0.0; c_len];
            let mut v3_new = vec![0.0; c_len];
            for u in 0..c_len {
                let mut s2 = 0.0;
                let mut s3 = 0.0;
                for &v in &local_adj[u] {
                    s2 += v2[v] * inv_sqrt_deg[v];
                    s3 += v3[v] * inv_sqrt_deg[v];
                }
                v2_new[u] = (s2 * inv_sqrt_deg[u] + v2[u]) * 0.5;
                v3_new[u] = (s3 * inv_sqrt_deg[u] + v3[u]) * 0.5;
            }
            v2 = v2_new;
            v3 = v3_new;
            
            let dot = v2.iter().zip(&v1).map(|(&a, &b)| a*b).sum::<f64>();
            let mut n2 = 0.0;
            for i in 0..c_len { v2[i] -= dot * v1[i]; n2 += v2[i]*v2[i]; }
            n2 = n2.sqrt();
            if n2 > 1e-12 { for x in &mut v2 { *x /= n2; } }
            
            let dot1 = v3.iter().zip(&v1).map(|(&a, &b)| a*b).sum::<f64>();
            for i in 0..c_len { v3[i] -= dot1 * v1[i]; }
            let dot2 = v3.iter().zip(&v2).map(|(&a, &b)| a*b).sum::<f64>();
            let mut n3 = 0.0;
            for i in 0..c_len { v3[i] -= dot2 * v2[i]; n3 += v3[i]*v3[i]; }
            n3 = n3.sqrt();
            if n3 > 1e-12 { for x in &mut v3 { *x /= n3; } }
        }

        let mut x = vec![0.0; c_len];
        let mut y = vec![0.0; c_len];
        for i in 0..c_len {
            x[i] = v2[i] * inv_sqrt_deg[i] + (rng.next_f64() - 0.5) * 1e-3;
            y[i] = v3[i] * inv_sqrt_deg[i] + (rng.next_f64() - 0.5) * 1e-3;
        }

        let mut total_dist = 0.0;
        let mut edge_count = 0;
        for u in 0..c_len {
            for &v in &local_adj[u] {
                if u < v {
                    let dx = x[u] - x[v];
                    let dy = y[u] - y[v];
                    total_dist += (dx*dx + dy*dy).sqrt();
                    edge_count += 1;
                }
            }
        }
        let avg_dist = if edge_count > 0 { total_dist / edge_count as f64 } else { 0.0 };
        if avg_dist > 1e-5 {
            let scale = k / avg_dist;
            for i in 0..c_len { x[i] *= scale; y[i] *= scale; }
        }

        let fr_iters = if c_len <= 100 { 150 } else if c_len <= 1000 { 100 } else if c_len <= 10000 { 50 } else { 30 };
        let mut t = k * 5.0;
        let cooling = t / (fr_iters as f64 + 1.0);

        for _iter in 0..fr_iters {
            let mut disp_x = vec![0.0; c_len];
            let mut disp_y = vec![0.0; c_len];
            
            if c_len <= 1000 {
                for u in 0..c_len {
                    for v in u+1..c_len {
                        let dx = x[u] - x[v];
                        let dy = y[u] - y[v];
                        let dist_sq = dx*dx + dy*dy;
                        if dist_sq < 9.0 * k * k && dist_sq > 1e-8 {
                            let dist = dist_sq.sqrt();
                            let f = 0.5 * k * k / dist;
                            let fx = (dx / dist) * f;
                            let fy = (dy / dist) * f;
                            disp_x[u] += fx; disp_y[u] += fy;
                            disp_x[v] -= fx; disp_y[v] -= fy;
                        }
                    }
                }
            } else {
                let mut min_x = f64::MAX;
                let mut max_x = f64::MIN;
                let mut min_y = f64::MAX;
                let mut max_y = f64::MIN;
                for i in 0..c_len {
                    if x[i] < min_x { min_x = x[i]; }
                    if x[i] > max_x { max_x = x[i]; }
                    if y[i] < min_y { min_y = y[i]; }
                    if y[i] > max_y { max_y = y[i]; }
                }
                let cell_size = 3.0 * k;
                let mut grid: HashMap<(i32, i32), Vec<usize>> = HashMap::new();
                for i in 0..c_len {
                    let cx = ((x[i] - min_x) / cell_size).floor() as i32;
                    let cy = ((y[i] - min_y) / cell_size).floor() as i32;
                    grid.entry((cx, cy)).or_default().push(i);
                }
                for i in 0..c_len {
                    let cx = ((x[i] - min_x) / cell_size).floor() as i32;
                    let cy = ((y[i] - min_y) / cell_size).floor() as i32;
                    let mut interactions = 0;
                    let max_interactions = 50;
                    'outer: for dx in -1..=1 {
                        for dy in -1..=1 {
                            if let Some(cell) = grid.get(&(cx+dx, cy+dy)) {
                                for &v in cell {
                                    if i != v {
                                        let dx_val = x[i] - x[v];
                                        let dy_val = y[i] - y[v];
                                        let dist_sq = dx_val*dx_val + dy_val*dy_val;
                                        if dist_sq < 9.0 * k * k && dist_sq > 1e-8 {
                                            let dist = dist_sq.sqrt();
                                            let f = 0.5 * k * k / dist;
                                            let fx = (dx_val / dist) * f;
                                            let fy = (dy_val / dist) * f;
                                            disp_x[i] += fx; disp_y[i] += fy;
                                            interactions += 1;
                                            if interactions >= max_interactions { break 'outer; }
                                        }
                                    }
                                }
                            }
                        }
                    }
                }
            }
            
            for u in 0..c_len {
                for &v in &local_adj[u] {
                    if u < v {
                        let dx = x[v] - x[u];
                        let dy = y[v] - y[u];
                        let dist = (dx*dx + dy*dy).sqrt().max(1e-4);
                        let f = 0.5 * (dist - k);
                        let fx = (dx / dist) * f;
                        let fy = (dy / dist) * f;
                        disp_x[u] += fx; disp_y[u] += fy;
                        disp_x[v] -= fx; disp_y[v] -= fy;
                    }
                }
            }
            
            for i in 0..c_len {
                let dist = (disp_x[i]*disp_x[i] + disp_y[i]*disp_y[i]).sqrt();
                if dist > 1e-8 {
                    let move_dist = dist.min(t);
                    x[i] += (disp_x[i] / dist) * move_dist;
                    y[i] += (disp_y[i] / dist) * move_dist;
                }
            }
            t = (t - cooling).max(0.01);
        }

        let mut min_x = f64::MAX;
        let mut max_x = f64::MIN;
        let mut min_y = f64::MAX;
        let mut max_y = f64::MIN;
        let mut nodes_res = Vec::with_capacity(c_len);
        for i in 0..c_len {
            if x[i] < min_x { min_x = x[i]; }
            if x[i] > max_x { max_x = x[i]; }
            if y[i] < min_y { min_y = y[i]; }
            if y[i] > max_y { max_y = y[i]; }
            nodes_res.push((comp[i], x[i], y[i]));
        }
        
        comp_results.push(CompResult {
            nodes: nodes_res,
            min_x, max_x, min_y, max_y,
            w: max_x - min_x,
            h: max_y - min_y,
        });

        for &u in &comp {
            global_to_local[u] = usize::MAX;
        }
    }
    
    let margin = k * 2.0;
    comp_results.sort_by(|a, b| {
        let area_a = a.w * a.h;
        let area_b = b.w * b.h;
        area_b.partial_cmp(&area_a).unwrap_or(std::cmp::Ordering::Equal)
    });
    
    let mut current_x = 0.0;
    let mut current_y = 0.0;
    let mut row_height = 0.0;
    
    let mut total_area = 0.0;
    for comp in &comp_results {
        total_area += (comp.w + margin) * (comp.h + margin);
    }
    let max_width = total_area.sqrt() * 1.5;

    for comp in &comp_results {
        if current_x + comp.w + margin > max_width && current_x > 0.0 {
            current_x = 0.0;
            current_y += row_height + margin;
            row_height = 0.0;
        }
        
        let shift_x = current_x - comp.min_x;
        let shift_y = current_y - comp.min_y;
        
        for &(id, x, y) in &comp.nodes {
            final_x[id] = x + shift_x;
            final_y[id] = y + shift_y;
        }
        
        current_x += comp.w + margin;
        if comp.h > row_height {
            row_height = comp.h;
        }
    }
    
    let stdout = io::stdout();
    let mut out = std::io::BufWriter::new(stdout.lock());
    for i in 0..n {
        writeln!(out, "{:.6} {:.6}", final_x[i], final_y[i]).unwrap();
    }
}
