using System;
using System.Diagnostics;
using System.Globalization;

class OrbitalTSP
{
    struct Vector3
    {
        public double X, Y, Z;
        public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vector3 operator *(Vector3 a, double d) => new Vector3(a.X * d, a.Y * d, a.Z * d);
        public static Vector3 operator /(Vector3 a, double d) => new Vector3(a.X / d, a.Y / d, a.Z / d);
        public static double Dot(Vector3 a, Vector3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vector3 Cross(Vector3 a, Vector3 b) => new Vector3(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
        public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);
    }

    struct Orbit
    {
        public Vector3 h;
        public Vector3 e;
        public double a;
        public double Vc;
        public Vector3 h_hat;
    }

    static double mu;

    static Orbit GetOrbit(Vector3 r, Vector3 v)
    {
        Orbit orb = new Orbit();
        orb.h = Vector3.Cross(r, v);
        double h_len = orb.h.Length();
        orb.h_hat = orb.h / h_len;
        
        Vector3 v_x_h = Vector3.Cross(v, orb.h);
        double r_len = r.Length();
        orb.e = (v_x_h / mu) - (r / r_len);
        
        double e_len = orb.e.Length();
        double p = h_len * h_len / mu;
        orb.a = p / (1 - e_len * e_len);
        
        if (orb.a < 0) orb.a = Math.Abs(orb.a); // Hyperbolic safeguard
        
        orb.Vc = Math.Sqrt(mu / orb.a);
        return orb;
    }

    static double GetCost(Orbit o1, Orbit o2)
    {
        double cosTheta = Vector3.Dot(o1.h_hat, o2.h_hat);
        if (cosTheta > 1.0) cosTheta = 1.0;
        if (cosTheta < -1.0) cosTheta = -1.0;
        
        double dV_plane_sq = o1.Vc * o1.Vc + o2.Vc * o2.Vc - 2 * o1.Vc * o2.Vc * cosTheta;
        double dV_plane = dV_plane_sq > 0 ? Math.Sqrt(dV_plane_sq) : 0;
        
        double dV_ecc = ((o1.Vc + o2.Vc) / 2.0) * (o1.e - o2.e).Length();
        
        return dV_plane + dV_ecc;
    }

    static void Main()
    {
        string line1 = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line1)) return;
        var parts1 = line1.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(parts1[0]);
        mu = double.Parse(parts1[1], CultureInfo.InvariantCulture);

        string line2 = Console.ReadLine();
        var parts2 = line2.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        Vector3 startR = new Vector3(double.Parse(parts2[0], CultureInfo.InvariantCulture), double.Parse(parts2[1], CultureInfo.InvariantCulture), double.Parse(parts2[2], CultureInfo.InvariantCulture));
        Vector3 startV = new Vector3(double.Parse(parts2[3], CultureInfo.InvariantCulture), double.Parse(parts2[4], CultureInfo.InvariantCulture), double.Parse(parts2[5], CultureInfo.InvariantCulture));
        Orbit startOrbit = GetOrbit(startR, startV);

        Orbit[] stations = new Orbit[N];
        for (int i = 0; i < N; i++)
        {
            string line = Console.ReadLine();
            var parts = line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            Vector3 r = new Vector3(double.Parse(parts[0], CultureInfo.InvariantCulture), double.Parse(parts[1], CultureInfo.InvariantCulture), double.Parse(parts[2], CultureInfo.InvariantCulture));
            Vector3 v = new Vector3(double.Parse(parts[3], CultureInfo.InvariantCulture), double.Parse(parts[4], CultureInfo.InvariantCulture), double.Parse(parts[5], CultureInfo.InvariantCulture));
            stations[i] = GetOrbit(r, v);
        }

        double[,] dist = new double[N + 1, N + 1];
        for (int i = 0; i < N; i++)
        {
            dist[N, i] = GetCost(startOrbit, stations[i]);
            dist[i, N] = dist[N, i];
            for (int j = 0; j < N; j++)
            {
                dist[i, j] = GetCost(stations[i], stations[j]);
            }
        }

        int[] bestPath = new int[N];
        for (int i = 0; i < N; i++) bestPath[i] = i;

        if (N <= 10)
        {
            double bestCost = double.MaxValue;
            int[] currentPath = new int[N];
            for (int i = 0; i < N; i++) currentPath[i] = i;
            
            do
            {
                double cost = Evaluate(currentPath, dist, N);
                if (cost < bestCost)
                {
                    bestCost = cost;
                    Array.Copy(currentPath, bestPath, N);
                }
            } while (NextPermutation(currentPath));
        }
        else
        {
            Random rnd = new Random(42);
            double bestCost = Evaluate(bestPath, dist, N);
            
            Stopwatch sw = Stopwatch.StartNew();
            int[] currentPath = (int[])bestPath.Clone();
            double currentCost = bestCost;
            
            double T0 = 1000.0;
            double T = T0;
            double alpha = 0.9999;
            
            long maxTimeMs = N <= 20 ? 500 : (N <= 50 ? 1000 : 2500);
            
            while (sw.ElapsedMilliseconds < maxTimeMs)
            {
                int maxIters = 10000;
                for (int i = 0; i < maxIters; i++)
                {
                    int type = rnd.Next(3);
                    if (type == 0) // Swap
                    {
                        int a = rnd.Next(N);
                        int b = rnd.Next(N);
                        if (a == b) continue;
                        
                        int temp = currentPath[a]; currentPath[a] = currentPath[b]; currentPath[b] = temp;
                        double newCost = Evaluate(currentPath, dist, N);
                        
                        if (newCost < currentCost || rnd.NextDouble() < Math.Exp((currentCost - newCost) / T))
                        {
                            currentCost = newCost;
                            if (currentCost < bestCost)
                            {
                                bestCost = currentCost;
                                Array.Copy(currentPath, bestPath, N);
                            }
                        }
                        else
                        {
                            temp = currentPath[a]; currentPath[a] = currentPath[b]; currentPath[b] = temp;
                        }
                    }
                    else if (type == 1) // 2-opt reverse
                    {
                        int a = rnd.Next(N);
                        int b = rnd.Next(N);
                        if (a == b) continue;
                        if (a > b) { int temp = a; a = b; b = temp; }
                        
                        Array.Reverse(currentPath, a, b - a + 1);
                        double newCost = Evaluate(currentPath, dist, N);
                        
                        if (newCost < currentCost || rnd.NextDouble() < Math.Exp((currentCost - newCost) / T))
                        {
                            currentCost = newCost;
                            if (currentCost < bestCost)
                            {
                                bestCost = currentCost;
                                Array.Copy(currentPath, bestPath, N);
                            }
                        }
                        else
                        {
                            Array.Reverse(currentPath, a, b - a + 1);
                        }
                    }
                    else // Insert
                    {
                        int a = rnd.Next(N);
                        int b = rnd.Next(N);
                        if (a == b) continue;
                        
                        int val = currentPath[a];
                        if (a < b) {
                            for(int j=a; j<b; j++) currentPath[j] = currentPath[j+1];
                        } else {
                            for(int j=a; j>b; j--) currentPath[j] = currentPath[j-1];
                        }
                        currentPath[b] = val;
                        
                        double newCost = Evaluate(currentPath, dist, N);
                        
                        if (newCost < currentCost || rnd.NextDouble() < Math.Exp((currentCost - newCost) / T))
                        {
                            currentCost = newCost;
                            if (currentCost < bestCost)
                            {
                                bestCost = currentCost;
                                Array.Copy(currentPath, bestPath, N);
                            }
                        }
                        else
                        {
                            if (a < b) {
                                for(int j=b; j>a; j--) currentPath[j] = currentPath[j-1];
                            } else {
                                for(int j=b; j<a; j++) currentPath[j] = currentPath[j+1];
                            }
                            currentPath[a] = val;
                        }
                    }
                }
                T *= alpha;
                if (T < 1e-5) T = T0; 
            }
        }

        Console.WriteLine(string.Join(" ", bestPath));
    }

    static double Evaluate(int[] path, double[,] dist, int N)
    {
        double cost = dist[N, path[0]];
        for (int i = 0; i < N - 1; i++) cost += dist[path[i], path[i + 1]];
        return cost;
    }

    static bool NextPermutation(int[] array)
    {
        int i = array.Length - 2;
        while (i >= 0 && array[i] >= array[i + 1]) i--;
        if (i < 0) return false;

        int j = array.Length - 1;
        while (array[j] <= array[i]) j--;

        int temp = array[i];
        array[i] = array[j];
        array[j] = temp;

        Array.Reverse(array, i + 1, array.Length - (i + 1));
        return true;
    }
}
