#pragma once

#include <vector>
#include <memory>
#include <cmath>
#include <algorithm>

namespace csg {

const float EPSILON = 1e-5f;

struct Vector3 {
    float x, y, z;
    Vector3(float x = 0.0f, float y = 0.0f, float z = 0.0f) : x(x), y(y), z(z) {}

    Vector3 operator+(const Vector3& other) const { return Vector3(x + other.x, y + other.y, z + other.z); }
    Vector3 operator-(const Vector3& other) const { return Vector3(x - other.x, y - other.y, z - other.z); }
    Vector3 operator*(float scalar) const { return Vector3(x * scalar, y * scalar, z * scalar); }
    Vector3 operator/(float scalar) const { return Vector3(x / scalar, y / scalar, z / scalar); }

    float dot(const Vector3& other) const { return x * other.x + y * other.y + z * other.z; }
    Vector3 cross(const Vector3& other) const {
        return Vector3(
            y * other.z - z * other.y,
            z * other.x - x * other.z,
            x * other.y - y * other.x
        );
    }
    
    float length() const { return std::sqrt(x * x + y * y + z * z); }
    Vector3 normalized() const {
        float len = length();
        if (len > 0) return (*this) / len;
        return *this;
    }
};

struct Vertex {
    Vector3 pos;
    Vector3 normal;

    Vertex(Vector3 pos, Vector3 normal) : pos(pos), normal(normal) {}

    Vertex interpolate(const Vertex& other, float t) const {
        return Vertex(
            pos + (other.pos - pos) * t,
            (normal + (other.normal - normal) * t).normalized()
        );
    }

    void flip() {
        normal = normal * -1.0f;
    }
};

class Plane {
public:
    Vector3 normal;
    float w;

    Plane() : w(0.0f) {}
    Plane(const Vector3& normal, float w) : normal(normal), w(w) {}

    static Plane fromPoints(const Vector3& a, const Vector3& b, const Vector3& c) {
        Vector3 n = (b - a).cross(c - a).normalized();
        return Plane(n, n.dot(a));
    }

    void flip() {
        normal = normal * -1.0f;
        w = -w;
    }

    enum {
        COPLANAR = 0,
        FRONT = 1,
        BACK = 2,
        SPANNING = 3
    };

    template <typename PolygonType>
    void splitPolygon(const PolygonType& polygon, std::vector<PolygonType>& coplanarFront, std::vector<PolygonType>& coplanarBack, std::vector<PolygonType>& front, std::vector<PolygonType>& back) const {
        int polygonType = 0;
        std::vector<int> types;
        for (const auto& v : polygon.vertices) {
            float t = normal.dot(v.pos) - w;
            int type = (t < -EPSILON) ? BACK : (t > EPSILON) ? FRONT : COPLANAR;
            polygonType |= type;
            types.push_back(type);
        }

        switch (polygonType) {
            case COPLANAR:
                if (normal.dot(polygon.plane.normal) > 0) {
                    coplanarFront.push_back(polygon);
                } else {
                    coplanarBack.push_back(polygon);
                }
                break;
            case FRONT:
                front.push_back(polygon);
                break;
            case BACK:
                back.push_back(polygon);
                break;
            case SPANNING: {
                std::vector<Vertex> f, b;
                for (size_t i = 0; i < polygon.vertices.size(); i++) {
                    size_t j = (i + 1) % polygon.vertices.size();
                    int ti = types[i], tj = types[j];
                    Vertex vi = polygon.vertices[i], vj = polygon.vertices[j];

                    if (ti != BACK) f.push_back(vi);
                    if (ti != FRONT) b.push_back(vi);

                    if ((ti | tj) == SPANNING) {
                        float t = (w - normal.dot(vi.pos)) / normal.dot(vj.pos - vi.pos);
                        Vertex v = vi.interpolate(vj, t);
                        f.push_back(v);
                        b.push_back(v);
                    }
                }
                if (f.size() >= 3) front.push_back(PolygonType(f));
                if (b.size() >= 3) back.push_back(PolygonType(b));
                break;
            }
        }
    }
};

class Polygon {
public:
    std::vector<Vertex> vertices;
    Plane plane;

    Polygon(const std::vector<Vertex>& vertices) : vertices(vertices) {
        if (vertices.size() >= 3) {
            plane = Plane::fromPoints(vertices[0].pos, vertices[1].pos, vertices[2].pos);
        }
    }

    void flip() {
        std::reverse(vertices.begin(), vertices.end());
        for (auto& v : vertices) {
            v.flip();
        }
        plane.flip();
    }
};

class Node {
public:
    std::shared_ptr<Plane> plane;
    std::vector<Polygon> polygons;
    std::shared_ptr<Node> front;
    std::shared_ptr<Node> back;

    Node() {}
    Node(const std::vector<Polygon>& list) {
        build(list);
    }

    std::shared_ptr<Node> clone() const {
        std::shared_ptr<Node> node = std::make_shared<Node>();
        if (plane) node->plane = std::make_shared<Plane>(*plane);
        node->polygons = polygons;
        if (front) node->front = front->clone();
        if (back) node->back = back->clone();
        return node;
    }

    void build(const std::vector<Polygon>& list) {
        if (list.empty()) return;

        if (!plane) {
            plane = std::make_shared<Plane>(list[0].plane);
        }

        std::vector<Polygon> list_front, list_back;
        for (const auto& poly : list) {
            plane->splitPolygon(poly, polygons, polygons, list_front, list_back);
        }

        if (!list_front.empty()) {
            if (!front) front = std::make_shared<Node>();
            front->build(list_front);
        }
        if (!list_back.empty()) {
            if (!back) back = std::make_shared<Node>();
            back->build(list_back);
        }
    }

    void invert() {
        for (auto& poly : polygons) poly.flip();
        if (plane) plane->flip();
        if (front) front->invert();
        if (back) back->invert();
        std::swap(front, back);
    }

    void clipPolygons(const std::vector<Polygon>& list, std::vector<Polygon>& result) const {
        if (!plane) {
            result.insert(result.end(), list.begin(), list.end());
            return;
        }

        std::vector<Polygon> list_front, list_back;
        for (const auto& poly : list) {
            plane->splitPolygon(poly, list_front, list_back, list_front, list_back);
        }

        if (front) {
            front->clipPolygons(list_front, result);
        } else {
            result.insert(result.end(), list_front.begin(), list_front.end());
        }

        if (back) {
            back->clipPolygons(list_back, result);
        } else {
            // Drop back polygons as they are inside
        }
    }

    void clipTo(const Node* bsp) {
        std::vector<Polygon> clipped_polygons;
        bsp->clipPolygons(polygons, clipped_polygons);
        polygons = clipped_polygons;
        if (front) front->clipTo(bsp);
        if (back) back->clipTo(bsp);
    }

    std::vector<Polygon> allPolygons() const {
        std::vector<Polygon> list = polygons;
        if (front) {
            auto fp = front->allPolygons();
            list.insert(list.end(), fp.begin(), fp.end());
        }
        if (back) {
            auto bp = back->allPolygons();
            list.insert(list.end(), bp.begin(), bp.end());
        }
        return list;
    }
};

class CSG {
public:
    std::vector<Polygon> polygons;

    CSG() {}
    
    static CSG fromPolygons(const std::vector<Polygon>& polygons) {
        CSG csg;
        csg.polygons = polygons;
        return csg;
    }

    CSG clone() const {
        CSG csg;
        csg.polygons = this->polygons;
        return csg;
    }

    // Core Constructive Solid Geometry operations
    // Union: A | B
    CSG union_op(const CSG& other) const {
        std::shared_ptr<Node> a = std::make_shared<Node>(this->clone().polygons);
        std::shared_ptr<Node> b = std::make_shared<Node>(other.clone().polygons);

        a->clipTo(b.get());
        b->clipTo(a.get());
        b->invert();
        b->clipTo(a.get());
        b->invert();
        
        a->build(b->allPolygons());
        
        return CSG::fromPolygons(a->allPolygons());
    }

    // Subtraction: A - B
    CSG subtract(const CSG& other) const {
        std::shared_ptr<Node> a = std::make_shared<Node>(this->clone().polygons);
        std::shared_ptr<Node> b = std::make_shared<Node>(other.clone().polygons);

        a->invert();
        a->clipTo(b.get());
        b->clipTo(a.get());
        b->invert();
        b->clipTo(a.get());
        b->invert();
        
        a->build(b->allPolygons());
        a->invert();
        
        return CSG::fromPolygons(a->allPolygons());
    }

    // Intersection: A & B
    CSG intersect(const CSG& other) const {
        std::shared_ptr<Node> a = std::make_shared<Node>(this->clone().polygons);
        std::shared_ptr<Node> b = std::make_shared<Node>(other.clone().polygons);

        a->invert();
        b->clipTo(a.get());
        b->invert();
        a->clipTo(b.get());
        b->clipTo(a.get());
        
        a->build(b->allPolygons());
        a->invert();
        
        return CSG::fromPolygons(a->allPolygons());
    }
};

} // namespace csg
