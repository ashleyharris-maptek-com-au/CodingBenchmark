using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    private struct State
    {
        public double X, Y, Z, Vx, Vy, Vz;
    }

    private static int N;
    private static double Mu;
    private static State Start;
    private static State[] Stations = Array.Empty<State>();

    private static double[] Radius = Array.Empty<double>();
    private static double[] Speed = Array.Empty<double>();
    private static double[,] Cost = new double[0, 0];
    private static double[] StartCost = Array.Empty<double>();

    private static readonly Random Rng = new Random(1234567);

    public static void Main()
    {
        var first = ReadDoubles();
        if (first.Length < 2) return;

        N = (int)Math.Round(first[0]);
        Mu = first[1];

        Start = ReadState();
        Stations = new State[N];
        for (int i = 0; i < N; i++) Stations[i] = ReadState();

        Precompute();

        int[] best;
        if (N <= 2)
        {
            best = Enumerable.Range(0, N).ToArray();
        }
        else if (N <= 10)
        {
            best = SolveExactHeldKarp();
        }
        else if (N <= 20)
        {
            best = SolveLarge(iters: 25000, restarts: 12);
        }
        else
        {
            best = SolveLarge(iters: 60000, restarts: 20);
        }

        Console.WriteLine(string.Join(" ", best));
    }

    private static void Precompute()
    {
        Radius = new double[N];
        Speed = new double[N];
        for (int i = 0; i < N; i++)
        {
            Radius[i] = Norm(Stations[i].X, Stations[i].Y, Stations[i].Z);
            Speed[i] = Norm(Stations[i].Vx, Stations[i].Vy, Stations[i].Vz);
        }

        Cost = new double[N, N];
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                if (i == j) Cost[i, j] = 0.0;
                else Cost[i, j] = TransferCost(Stations[i], Stations[j], Radius[i], Radius[j], Speed[i], Speed[j]);
            }
        }

        double startR = Norm(Start.X, Start.Y, Start.Z);
        double startV = Norm(Start.Vx, Start.Vy, Start.Vz);
        StartCost = new double[N];
        for (int i = 0; i < N; i++)
            StartCost[i] = TransferCost(Start, Stations[i], startR, Radius[i], startV, Speed[i]);
    }

    // Approximate orbital-transfer metric.
    // Combines Hohmann-like radial change and plane-mismatch proxy from velocity vectors.
    private static double TransferCost(State a, State b, double ra, double rb, double va, double vb)
    {
        double r1 = Math.Max(ra, 1.0);
        double r2 = Math.Max(rb, 1.0);

        double vc1 = Math.Sqrt(Mu / r1);
        double vc2 = Math.Sqrt(Mu / r2);

        double aT = 0.5 * (r1 + r2);
        double vt1 = Math.Sqrt(Math.Max(0.0, Mu * (2.0 / r1 - 1.0 / aT)));
        double vt2 = Math.Sqrt(Math.Max(0.0, Mu * (2.0 / r2 - 1.0 / aT)));
        double dvH = Math.Abs(vt1 - vc1) + Math.Abs(vc2 - vt2);

        double dot = a.Vx * b.Vx + a.Vy * b.Vy + a.Vz * b.Vz;
        double mag = Math.Max(1e-9, va * vb);
        double c = Clamp(dot / mag, -1.0, 1.0);
        double angle = Math.Acos(c);
        double vRef = 0.5 * (vc1 + vc2);
        double dvPlane = 2.0 * vRef * Math.Sin(0.5 * angle);

        double dr = Norm(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        double scale = Math.Max(1000.0, 0.5 * (r1 + r2));
        double dvPos = 0.05 * (dr / scale) * vRef;

        return dvH + 0.6 * dvPlane + dvPos;
    }

    private static int[] SolveExactHeldKarp()
    {
        int size = 1 << N;
        double inf = 1e100;
        double[,] dp = new double[size, N];
        int[,] parent = new int[size, N];

        for (int s = 0; s < size; s++)
            for (int i = 0; i < N; i++)
            {
                dp[s, i] = inf;
                parent[s, i] = -1;
            }

        for (int i = 0; i < N; i++)
        {
            int m = 1 << i;
            dp[m, i] = StartCost[i];
        }

        for (int mask = 1; mask < size; mask++)
        {
            for (int last = 0; last < N; last++)
            {
                if ((mask & (1 << last)) == 0) continue;
                double cur = dp[mask, last];
                if (cur >= inf) continue;

                int rem = ((size - 1) ^ mask);
                while (rem != 0)
                {
                    int bit = rem & -rem;
                    int nxt = BitIndex(bit);
                    int nm = mask | bit;
                    double cand = cur + Cost[last, nxt];
                    if (cand < dp[nm, nxt])
                    {
                        dp[nm, nxt] = cand;
                        parent[nm, nxt] = last;
                    }
                    rem ^= bit;
                }
            }
        }

        int full = size - 1;
        double best = inf;
        int end = 0;
        for (int i = 0; i < N; i++)
        {
            if (dp[full, i] < best)
            {
                best = dp[full, i];
                end = i;
            }
        }

        int[] route = new int[N];
        int msk = full;
        int curNode = end;
        for (int k = N - 1; k >= 0; k--)
        {
            route[k] = curNode;
            int p = parent[msk, curNode];
            msk ^= (1 << curNode);
            curNode = p;
        }
        return route;
    }

    private static int[] SolveLarge(int iters, int restarts)
    {
        int[] globalBest = BuildGreedy();
        double globalCost = PathCost(globalBest);

        for (int r = 0; r < restarts; r++)
        {
            int seed = PickStartByRegret(r);
            int[] cur = BuildGreedy(seed);
            TwoOpt(cur, maxPasses: 3);

            int[] bestLocal = (int[])cur.Clone();
            double bestLocalCost = PathCost(bestLocal);
            double curCost = bestLocalCost;

            double t0 = Math.Max(1.0, bestLocalCost / Math.Max(1, N));
            double t = t0;
            const double alpha = 0.9994;

            for (int it = 0; it < iters; it++)
            {
                if ((it & 1) == 0)
                {
                    int i = 1 + Rng.Next(N - 1);
                    int j = 1 + Rng.Next(N - 1);
                    if (i > j) { int tmp = i; i = j; j = tmp; }
                    if (i == j) continue;

                    double delta = DeltaTwoOpt(cur, i, j);
                    if (Accept(delta, t))
                    {
                        Reverse(cur, i, j);
                        curCost += delta;
                    }
                }
                else
                {
                    int i = 1 + Rng.Next(N - 1);
                    int j = Rng.Next(N);
                    if (i == j || i == j + 1) continue;

                    double delta = DeltaRelocate(cur, i, j);
                    if (Accept(delta, t))
                    {
                        ApplyRelocate(cur, i, j);
                        curCost += delta;
                    }
                }

                if (curCost < bestLocalCost)
                {
                    bestLocalCost = curCost;
                    Array.Copy(cur, bestLocal, N);
                }

                t *= alpha;
                if (t < 1e-4) t = t0;
            }

            TwoOpt(bestLocal, maxPasses: 2);
            bestLocalCost = PathCost(bestLocal);
            if (bestLocalCost < globalCost)
            {
                globalCost = bestLocalCost;
                globalBest = bestLocal;
            }
        }

        return globalBest;
    }

    private static bool Accept(double delta, double t)
    {
        if (delta <= 0) return true;
        double p = Math.Exp(-delta / Math.Max(1e-9, t));
        return Rng.NextDouble() < p;
    }

    private static int[] BuildGreedy(int startNode = -1)
    {
        bool[] used = new bool[N];
        int[] route = new int[N];

        if (startNode < 0)
        {
            startNode = 0;
            double best = StartCost[0];
            for (int i = 1; i < N; i++)
            {
                if (StartCost[i] < best)
                {
                    best = StartCost[i];
                    startNode = i;
                }
            }
        }

        route[0] = startNode;
        used[startNode] = true;

        for (int pos = 1; pos < N; pos++)
        {
            int prev = route[pos - 1];
            int bestNode = -1;
            double bestScore = double.PositiveInfinity;
            for (int c = 0; c < N; c++)
            {
                if (used[c]) continue;
                double score = Cost[prev, c] + 0.2 * StartCost[c];
                if (score < bestScore)
                {
                    bestScore = score;
                    bestNode = c;
                }
            }
            route[pos] = bestNode;
            used[bestNode] = true;
        }

        return route;
    }

    private static int PickStartByRegret(int salt)
    {
        int k = Math.Min(10, N);
        int[] idx = Enumerable.Range(0, N).OrderBy(i => StartCost[i]).Take(k).ToArray();
        return idx[(salt + Rng.Next(k)) % k];
    }

    private static void TwoOpt(int[] route, int maxPasses)
    {
        for (int pass = 0; pass < maxPasses; pass++)
        {
            bool improved = false;
            for (int i = 1; i < N - 1; i++)
            {
                for (int j = i + 1; j < N; j++)
                {
                    double delta = DeltaTwoOpt(route, i, j);
                    if (delta < -1e-12)
                    {
                        Reverse(route, i, j);
                        improved = true;
                    }
                }
            }
            if (!improved) break;
        }
    }

    private static double DeltaTwoOpt(int[] r, int i, int j)
    {
        int a = r[i - 1];
        int b = r[i];
        int c = r[j];
        int d = (j + 1 < N) ? r[j + 1] : -1;

        double before = Edge(a, b) + (d >= 0 ? Edge(c, d) : 0.0);
        double after = Edge(a, c) + (d >= 0 ? Edge(b, d) : 0.0);
        return after - before;
    }

    private static double DeltaRelocate(int[] r, int from, int toAfter)
    {
        int node = r[from];
        int prev = r[from - 1];
        int next = (from + 1 < N) ? r[from + 1] : -1;

        double remove = Edge(prev, node) + (next >= 0 ? Edge(node, next) : 0.0) - (next >= 0 ? Edge(prev, next) : 0.0);

        int insLeft = r[toAfter];
        int insRight = (toAfter + 1 < N) ? r[toAfter + 1] : -1;

        if (toAfter == from || toAfter == from - 1) return 0.0;

        double add = Edge(insLeft, node) + (insRight >= 0 ? Edge(node, insRight) : 0.0) - (insRight >= 0 ? Edge(insLeft, insRight) : 0.0);

        return add - remove;
    }

    private static void ApplyRelocate(int[] r, int from, int toAfter)
    {
        int node = r[from];
        if (from < toAfter)
        {
            for (int i = from; i < toAfter; i++) r[i] = r[i + 1];
            r[toAfter] = node;
        }
        else
        {
            for (int i = from; i > toAfter + 1; i--) r[i] = r[i - 1];
            r[toAfter + 1] = node;
        }
    }

    private static void Reverse(int[] a, int l, int r)
    {
        while (l < r)
        {
            int t = a[l];
            a[l] = a[r];
            a[r] = t;
            l++;
            r--;
        }
    }

    private static double PathCost(int[] route)
    {
        if (route.Length == 0) return 0.0;
        double total = StartCost[route[0]];
        for (int i = 1; i < route.Length; i++)
            total += Cost[route[i - 1], route[i]];
        return total;
    }

    private static double Edge(int a, int b)
    {
        return Cost[a, b];
    }

    private static State ReadState()
    {
        var d = ReadDoubles();
        return new State { X = d[0], Y = d[1], Z = d[2], Vx = d[3], Vy = d[4], Vz = d[5] };
    }

    private static double[] ReadDoubles()
    {
        string? line;
        do { line = Console.ReadLine(); } while (line != null && line.Trim().Length == 0);
        if (line == null) return Array.Empty<double>();

        var parts = line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        var arr = new double[parts.Length];
        for (int i = 0; i < parts.Length; i++)
            arr[i] = double.Parse(parts[i], CultureInfo.InvariantCulture);
        return arr;
    }

    private static int BitIndex(int bit)
    {
        int idx = 0;
        while ((bit >>= 1) != 0) idx++;
        return idx;
    }

    private static double Norm(double x, double y, double z)
    {
        return Math.Sqrt(x * x + y * y + z * z);
    }

    private static double Clamp(double v, double lo, double hi)
    {
        if (v < lo) return lo;
        if (v > hi) return hi;
        return v;
    }
}
