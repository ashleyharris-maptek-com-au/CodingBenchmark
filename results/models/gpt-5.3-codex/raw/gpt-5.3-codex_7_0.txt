struct Vec3 {
    double x = 0.0, y = 0.0, z = 0.0;
    Vec3() = default;
    Vec3(double xx, double yy, double zz) : x(xx), y(yy), z(zz) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
};

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

struct Tri {
    int a = 0, b = 0, c = 0;
};

struct Mesh {
    std::vector<Vec3> v;
    std::vector<Tri> f;
};

struct AABB {
    Vec3 mn{ std::numeric_limits<double>::infinity(),
             std::numeric_limits<double>::infinity(),
             std::numeric_limits<double>::infinity() };
    Vec3 mx{ -std::numeric_limits<double>::infinity(),
             -std::numeric_limits<double>::infinity(),
             -std::numeric_limits<double>::infinity() };

    void grow(const Vec3& p) {
        mn.x = std::min(mn.x, p.x); mn.y = std::min(mn.y, p.y); mn.z = std::min(mn.z, p.z);
        mx.x = std::max(mx.x, p.x); mx.y = std::max(mx.y, p.y); mx.z = std::max(mx.z, p.z);
    }

    bool valid() const {
        return mn.x <= mx.x && mn.y <= mx.y && mn.z <= mx.z;
    }
};

static inline AABB triAABB(const Mesh& m, int fi) {
    const Tri& t = m.f[fi];
    AABB b;
    b.grow(m.v[t.a]);
    b.grow(m.v[t.b]);
    b.grow(m.v[t.c]);
    return b;
}

static inline bool overlapAABB(const AABB& a, const AABB& b, double eps = 0.0) {
    if (a.mx.x + eps < b.mn.x || b.mx.x + eps < a.mn.x) return false;
    if (a.mx.y + eps < b.mn.y || b.mx.y + eps < a.mn.y) return false;
    if (a.mx.z + eps < b.mn.z || b.mx.z + eps < a.mn.z) return false;
    return true;
}

static inline AABB meshAABB(const Mesh& m) {
    AABB b;
    for (const auto& p : m.v) b.grow(p);
    return b;
}

static bool rayTriIntersect(const Vec3& orig, const Vec3& dir, const Vec3& v0, const Vec3& v1, const Vec3& v2, double& tOut) {
    constexpr double EPS = 1e-12;
    Vec3 e1 = v1 - v0;
    Vec3 e2 = v2 - v0;
    Vec3 p = cross(dir, e2);
    double det = dot(e1, p);
    if (std::abs(det) < EPS) return false;
    double invDet = 1.0 / det;
    Vec3 s = orig - v0;
    double u = dot(s, p) * invDet;
    if (u < -EPS || u > 1.0 + EPS) return false;
    Vec3 q = cross(s, e1);
    double v = dot(dir, q) * invDet;
    if (v < -EPS || u + v > 1.0 + EPS) return false;
    double t = dot(e2, q) * invDet;
    if (t <= EPS) return false;
    tOut = t;
    return true;
}

struct UniformGrid {
    AABB box;
    int nx = 1, ny = 1, nz = 1;
    double sx = 1.0, sy = 1.0, sz = 1.0;
    std::unordered_map<std::uint64_t, std::vector<int>> cells;

    static std::uint64_t key(int x, int y, int z) {
        return (static_cast<std::uint64_t>(x) << 42) ^ (static_cast<std::uint64_t>(y) << 21) ^ static_cast<std::uint64_t>(z);
    }

    int clampi(int v, int lo, int hi) const {
        return (v < lo) ? lo : (v > hi ? hi : v);
    }

    void build(const Mesh& m, int targetCellsPerAxis = 64) {
        box = meshAABB(m);
        Vec3 d = box.mx - box.mn;
        nx = ny = nz = targetCellsPerAxis;
        sx = (d.x > 0.0) ? d.x / nx : 1.0;
        sy = (d.y > 0.0) ? d.y / ny : 1.0;
        sz = (d.z > 0.0) ? d.z / nz : 1.0;

        cells.reserve(static_cast<std::size_t>(m.f.size()) * 2);

        for (int i = 0; i < static_cast<int>(m.f.size()); ++i) {
            AABB tb = triAABB(m, i);
            int ix0 = clampi(static_cast<int>(std::floor((tb.mn.x - box.mn.x) / sx)), 0, nx - 1);
            int iy0 = clampi(static_cast<int>(std::floor((tb.mn.y - box.mn.y) / sy)), 0, ny - 1);
            int iz0 = clampi(static_cast<int>(std::floor((tb.mn.z - box.mn.z) / sz)), 0, nz - 1);
            int ix1 = clampi(static_cast<int>(std::floor((tb.mx.x - box.mn.x) / sx)), 0, nx - 1);
            int iy1 = clampi(static_cast<int>(std::floor((tb.mx.y - box.mn.y) / sy)), 0, ny - 1);
            int iz1 = clampi(static_cast<int>(std::floor((tb.mx.z - box.mn.z) / sz)), 0, nz - 1);

            for (int z = iz0; z <= iz1; ++z) {
                for (int y = iy0; y <= iy1; ++y) {
                    for (int x = ix0; x <= ix1; ++x) {
                        cells[key(x, y, z)].push_back(i);
                    }
                }
            }
        }
    }

    template <typename Fn>
    void queryRay(const Vec3& orig, const Vec3& dir, Fn&& fn) const {
        for (const auto& kv : cells) {
            for (int ti : kv.second) fn(ti);
        }
    }
};

static bool pointInMeshParity(const Mesh& m, const UniformGrid& grid, const Vec3& p) {
    const Vec3 dir(1.0, 0.000371, 0.000913);
    int hits = 0;

    std::vector<int> tested;
    tested.reserve(m.f.size());
    grid.queryRay(p, dir, [&](int ti) {
        tested.push_back(ti);
    });
    std::sort(tested.begin(), tested.end());
    tested.erase(std::unique(tested.begin(), tested.end()), tested.end());

    for (int ti : tested) {
        const Tri& t = m.f[ti];
        double tt = 0.0;
        if (rayTriIntersect(p, dir, m.v[t.a], m.v[t.b], m.v[t.c], tt)) {
            ++hits;
        }
    }
    return (hits % 2) == 1;
}

static Vec3 triCentroid(const Mesh& m, const Tri& t) {
    return (m.v[t.a] + m.v[t.b] + m.v[t.c]) * (1.0 / 3.0);
}

static Mesh adaptiveUnionApprox(const Mesh& A, const Mesh& B) {
    Mesh out;
    AABB aBox = meshAABB(A);
    AABB bBox = meshAABB(B);

    if (!overlapAABB(aBox, bBox, 0.0)) {
        out = A;
        int offset = static_cast<int>(out.v.size());
        out.v.insert(out.v.end(), B.v.begin(), B.v.end());
        for (const auto& t : B.f) out.f.push_back({t.a + offset, t.b + offset, t.c + offset});
        return out;
    }

    const std::size_t totalFaces = A.f.size() + B.f.size();
    int gridRes = 32;
    if (totalFaces > 1000000) gridRes = 128;
    else if (totalFaces > 200000) gridRes = 96;
    else if (totalFaces > 50000) gridRes = 64;

    UniformGrid gridA, gridB;
    gridA.build(A, gridRes);
    gridB.build(B, gridRes);

    std::vector<char> keepA(A.f.size(), 1), keepB(B.f.size(), 1);

    for (int i = 0; i < static_cast<int>(A.f.size()); ++i) {
        Vec3 c = triCentroid(A, A.f[i]);
        if (pointInMeshParity(B, gridB, c)) keepA[i] = 0;
    }
    for (int i = 0; i < static_cast<int>(B.f.size()); ++i) {
        Vec3 c = triCentroid(B, B.f[i]);
        if (pointInMeshParity(A, gridA, c)) keepB[i] = 0;
    }

    std::vector<int> remapA(A.v.size(), -1), remapB(B.v.size(), -1);

    for (int i = 0; i < static_cast<int>(A.f.size()); ++i) {
        if (!keepA[i]) continue;
        const Tri& t = A.f[i];
        int ids[3] = {t.a, t.b, t.c};
        int o[3];
        for (int k = 0; k < 3; ++k) {
            if (remapA[ids[k]] < 0) {
                remapA[ids[k]] = static_cast<int>(out.v.size());
                out.v.push_back(A.v[ids[k]]);
            }
            o[k] = remapA[ids[k]];
        }
        out.f.push_back({o[0], o[1], o[2]});
    }

    for (int i = 0; i < static_cast<int>(B.f.size()); ++i) {
        if (!keepB[i]) continue;
        const Tri& t = B.f[i];
        int ids[3] = {t.a, t.b, t.c};
        int o[3];
        for (int k = 0; k < 3; ++k) {
            if (remapB[ids[k]] < 0) {
                remapB[ids[k]] = static_cast<int>(out.v.size());
                out.v.push_back(B.v[ids[k]]);
            }
            o[k] = remapB[ids[k]];
        }
        out.f.push_back({o[0], o[1], o[2]});
    }

    if (out.f.empty()) {
        out = A;
    }

    return out;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) return 0;

    Mesh A, B;
    A.v.resize(VA);
    A.f.resize(FA);
    B.v.resize(VB);
    B.f.resize(FB);

    for (int i = 0; i < VA; ++i) std::cin >> A.v[i].x >> A.v[i].y >> A.v[i].z;
    for (int i = 0; i < FA; ++i) std::cin >> A.f[i].a >> A.f[i].b >> A.f[i].c;
    for (int i = 0; i < VB; ++i) std::cin >> B.v[i].x >> B.v[i].y >> B.v[i].z;
    for (int i = 0; i < FB; ++i) std::cin >> B.f[i].a >> B.f[i].b >> B.f[i].c;

    Mesh U = adaptiveUnionApprox(A, B);

    std::cout << U.v.size() << ' ' << U.f.size() << '\n';
    std::cout << std::setprecision(17);
    for (const auto& p : U.v) {
        std::cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    }
    for (const auto& t : U.f) {
        std::cout << t.a << ' ' << t.b << ' ' << t.c << '\n';
    }

    return 0;
}
