#include <algorithm>
#include <array>
#include <chrono>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <tuple>
#include <unordered_map>
#include <utility>
#include <vector>

using namespace std;

struct Rect {
    int w;
    int h;
    int id;
};

struct Placement {
    int board = -1;
    int x = -1;
    int y = -1;
    int rotated = 0;
};

struct Node {
    int x;
    int y;
    int w;
    int h;
};

struct BoardState {
    vector<Node> freeRects;
    int freeArea = 0;
};

static inline long long areaLL(int w, int h) {
    return 1LL * w * h;
}

static inline bool fits(int rw, int rh, const Node& n) {
    return rw <= n.w && rh <= n.h;
}

struct Candidate {
    bool ok = false;
    int nodeIdx = -1;
    int x = 0;
    int y = 0;
    int rw = 0;
    int rh = 0;
    int rotated = 0;
    int scoreA = numeric_limits<int>::max();
    int scoreB = numeric_limits<int>::max();
};

static inline void pruneFreeList(vector<Node>& freeRects) {
    // Remove contained rectangles. O(m^2), but m is typically modest due to splitting discipline.
    size_t m = freeRects.size();
    vector<char> dead(m, 0);
    for (size_t i = 0; i < m; ++i) {
        if (dead[i]) continue;
        const Node& a = freeRects[i];
        for (size_t j = i + 1; j < m; ++j) {
            if (dead[j]) continue;
            const Node& b = freeRects[j];
            bool aInB = a.x >= b.x && a.y >= b.y && a.x + a.w <= b.x + b.w && a.y + a.h <= b.y + b.h;
            bool bInA = b.x >= a.x && b.y >= a.y && b.x + b.w <= a.x + a.w && b.y + b.h <= a.y + a.h;
            if (aInB && (a.w * a.h <= b.w * b.h)) {
                dead[i] = 1;
                break;
            }
            if (bInA && (b.w * b.h <= a.w * a.h)) {
                dead[j] = 1;
            }
        }
    }
    vector<Node> keep;
    keep.reserve(m);
    for (size_t i = 0; i < m; ++i) if (!dead[i]) keep.push_back(freeRects[i]);
    freeRects.swap(keep);
}

static inline void splitNode(const Node& f, int px, int py, int pw, int ph, vector<Node>& out) {
    // Guillotine-like split into up to 4 regions around the placed rect.
    int fx2 = f.x + f.w;
    int fy2 = f.y + f.h;
    int px2 = px + pw;
    int py2 = py + ph;

    // Left
    if (px > f.x) out.push_back({f.x, f.y, px - f.x, f.h});
    // Right
    if (px2 < fx2) out.push_back({px2, f.y, fx2 - px2, f.h});
    // Bottom
    if (py > f.y) out.push_back({f.x, f.y, f.w, py - f.y});
    // Top
    if (py2 < fy2) out.push_back({f.x, py2, f.w, fy2 - py2});
}

static inline void placeInBoard(BoardState& b, const Candidate& c) {
    Node used = b.freeRects[c.nodeIdx];
    b.freeRects[c.nodeIdx] = b.freeRects.back();
    b.freeRects.pop_back();

    vector<Node> add;
    add.reserve(4);
    splitNode(used, c.x, c.y, c.rw, c.rh, add);
    for (const auto& n : add) {
        if (n.w > 0 && n.h > 0) b.freeRects.push_back(n);
    }

    b.freeArea -= c.rw * c.rh;
    pruneFreeList(b.freeRects);
}

static Candidate bestCandidateInBoard(const BoardState& b, int w, int h, bool allowRotate, bool highQuality) {
    Candidate best;

    for (int i = 0; i < (int)b.freeRects.size(); ++i) {
        const Node& n = b.freeRects[i];

        auto eval = [&](int rw, int rh, int rot) {
            if (!fits(rw, rh, n)) return;
            int dw = n.w - rw;
            int dh = n.h - rh;
            Candidate c;
            c.ok = true;
            c.nodeIdx = i;
            c.x = n.x;
            c.y = n.y;
            c.rw = rw;
            c.rh = rh;
            c.rotated = rot;

            if (highQuality) {
                // Best area fit first, then short side fit.
                c.scoreA = dw * dh;
                c.scoreB = min(dw, dh);
            } else {
                // Faster / stable: short side then long side.
                c.scoreA = min(dw, dh);
                c.scoreB = max(dw, dh);
            }

            if (!best.ok || c.scoreA < best.scoreA || (c.scoreA == best.scoreA && c.scoreB < best.scoreB)) {
                best = c;
            }
        };

        eval(w, h, 0);
        if (allowRotate && w != h) eval(h, w, 1);
    }

    return best;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, BW, BH;
    if (!(cin >> N >> BW >> BH)) {
        return 0;
    }

    vector<Rect> rects;
    rects.reserve(N);

    long long totalArea = 0;
    int maxSide = max(BW, BH);

    for (int i = 0; i < N; ++i) {
        int w, h;
        cin >> w >> h;
        rects.push_back({w, h, i});
        totalArea += areaLL(w, h);

        // Early impossible detection will be checked after reading.
    }

    vector<Placement> ans(N);

    // Filter impossible rectangles first.
    for (const auto& r : rects) {
        bool ok = (r.w <= BW && r.h <= BH) || (r.h <= BW && r.w <= BH);
        if (!ok) {
            // No feasible packing exists under constraints. Emit 0 boards and dummy placements.
            cout << 0 << '\n';
            for (int i = 0; i < N; ++i) {
                cout << 0 << ' ' << 0 << ' ' << 0 << ' ' << 0 << '\n';
            }
            return 0;
        }
    }

    // Adaptive mode selection.
    bool tiny = N <= 200;
    bool huge = N >= 12000;
    bool veryHuge = N >= 30000;

    bool highQuality = tiny;       // Spend extra effort for small instances.
    bool allowGlobalRepair = tiny; // Keep simple (disabled here), reserved for extension.
    (void)allowGlobalRepair;

    // Ordering heuristic: larger / more awkward rectangles first.
    sort(rects.begin(), rects.end(), [&](const Rect& a, const Rect& b) {
        long long aa = areaLL(a.w, a.h);
        long long bb = areaLL(b.w, b.h);
        if (aa != bb) return aa > bb;
        int aper = a.w + a.h;
        int bper = b.w + b.h;
        if (aper != bper) return aper > bper;
        int as = max(a.w, a.h);
        int bs = max(b.w, b.h);
        return as > bs;
    });

    vector<BoardState> boards;
    boards.reserve(max(8, (int)(totalArea / max(1LL, areaLL(BW, BH))) + 8));

    auto addBoard = [&]() {
        BoardState b;
        b.freeRects.push_back({0, 0, BW, BH});
        b.freeArea = BW * BH;
        boards.push_back(move(b));
        return (int)boards.size() - 1;
    };

    auto boardOrder = vector<int>();
    boardOrder.reserve(boards.capacity());

    for (int idx = 0; idx < N; ++idx) {
        const Rect& r = rects[idx];

        Candidate best;
        int bestBoard = -1;

        // Search policy:
        // - tiny/medium: check all boards
        // - huge: check top-K boards by free area proxy via incremental list
        if (!huge) {
            for (int bi = 0; bi < (int)boards.size(); ++bi) {
                Candidate c = bestCandidateInBoard(boards[bi], r.w, r.h, true, highQuality);
                if (c.ok) {
                    if (!best.ok || c.scoreA < best.scoreA || (c.scoreA == best.scoreA && c.scoreB < best.scoreB)) {
                        best = c;
                        bestBoard = bi;
                    }
                }
            }
        } else {
            // Build candidate board list quickly from last few and a few with high free area.
            boardOrder.clear();
            int B = (int)boards.size();
            int tail = min(B, veryHuge ? 24 : 40);
            for (int i = B - tail; i < B; ++i) if (i >= 0) boardOrder.push_back(i);

            // Add some best-free-area boards.
            if (B > 0) {
                vector<int> ids(B);
                iota(ids.begin(), ids.end(), 0);
                int k = min(B, veryHuge ? 24 : 40);
                nth_element(ids.begin(), ids.begin() + (k - 1), ids.end(), [&](int a, int b) {
                    return boards[a].freeArea > boards[b].freeArea;
                });
                ids.resize(k);
                for (int id : ids) boardOrder.push_back(id);
            }

            sort(boardOrder.begin(), boardOrder.end());
            boardOrder.erase(unique(boardOrder.begin(), boardOrder.end()), boardOrder.end());

            for (int bi : boardOrder) {
                Candidate c = bestCandidateInBoard(boards[bi], r.w, r.h, true, false);
                if (c.ok) {
                    if (!best.ok || c.scoreA < best.scoreA || (c.scoreA == best.scoreA && c.scoreB < best.scoreB)) {
                        best = c;
                        bestBoard = bi;
                    }
                }
            }
        }

        if (!best.ok) {
            int bi = addBoard();
            best = bestCandidateInBoard(boards[bi], r.w, r.h, true, false);
            bestBoard = bi;
            if (!best.ok) {
                // Should not happen due to pre-check, but keep safe fallback.
                cout << 0 << '\n';
                for (int i = 0; i < N; ++i) {
                    cout << 0 << ' ' << 0 << ' ' << 0 << ' ' << 0 << '\n';
                }
                return 0;
            }
        }

        placeInBoard(boards[bestBoard], best);

        ans[r.id].board = bestBoard;
        ans[r.id].x = best.x;
        ans[r.id].y = best.y;
        ans[r.id].rotated = best.rotated;
    }

    cout << boards.size() << '\n';
    for (int i = 0; i < N; ++i) {
        cout << ans[i].board << ' ' << ans[i].x << ' ' << ans[i].y << ' ' << ans[i].rotated << '\n';
    }

    return 0;
}
