using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public static class Program
{
    private struct Edge
    {
        public int To;
        public int W;
        public int Id;
        public Edge(int to, int w, int id) { To = to; W = w; Id = id; }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-') { sign = -1; c = Read(); }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }

    private sealed class MinHeap
    {
        private readonly List<(long d, int v)> _a = new();
        public int Count => _a.Count;

        public void Push(long d, int v)
        {
            _a.Add((d, v));
            int i = _a.Count - 1;
            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (_a[p].d <= _a[i].d) break;
                (_a[p], _a[i]) = (_a[i], _a[p]);
                i = p;
            }
        }

        public (long d, int v) Pop()
        {
            var root = _a[0];
            int last = _a.Count - 1;
            _a[0] = _a[last];
            _a.RemoveAt(last);
            int n = _a.Count;
            int i = 0;
            while (true)
            {
                int l = i * 2 + 1, r = l + 1, m = i;
                if (l < n && _a[l].d < _a[m].d) m = l;
                if (r < n && _a[r].d < _a[m].d) m = r;
                if (m == i) break;
                (_a[m], _a[i]) = (_a[i], _a[m]);
                i = m;
            }
            return root;
        }
    }

    private static void Dijkstra(int src, List<Edge>[] g, int[] activeMap, out long[] dist, out int[] parentV, out int[] parentE)
    {
        int n = g.Length;
        dist = new long[n];
        parentV = new int[n];
        parentE = new int[n];
        Array.Fill(dist, long.MaxValue / 4);
        Array.Fill(parentV, -1);
        Array.Fill(parentE, -1);

        var pq = new MinHeap();
        dist[src] = 0;
        pq.Push(0, src);

        while (pq.Count > 0)
        {
            var (d, v) = pq.Pop();
            if (d != dist[v]) continue;
            foreach (var e in g[v])
            {
                long nd = d + e.W;
                if (nd < dist[e.To])
                {
                    dist[e.To] = nd;
                    parentV[e.To] = v;
                    parentE[e.To] = e.Id;
                    pq.Push(nd, e.To);
                }
            }
        }
    }

    private static int[] BuildEulerTour(List<Edge>[] g, int edgeCount)
    {
        int n = g.Length;
        var used = new bool[edgeCount];
        var it = new int[n];
        var st = new Stack<int>();
        var path = new List<int>(edgeCount + 1);

        st.Push(0);
        while (st.Count > 0)
        {
            int v = st.Peek();
            while (it[v] < g[v].Count && used[g[v][it[v]].Id]) it[v]++;
            if (it[v] == g[v].Count)
            {
                path.Add(v);
                st.Pop();
            }
            else
            {
                var e = g[v][it[v]++];
                if (used[e.Id]) continue;
                used[e.Id] = true;
                st.Push(e.To);
            }
        }

        path.Reverse();
        return path.ToArray();
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int m = fs.NextInt();

        var g = new List<Edge>[n];
        for (int i = 0; i < n; i++) g[i] = new List<Edge>();

        int nextEdgeId = 0;
        int[] deg = new int[n];

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            int w = fs.NextInt();
            g[u].Add(new Edge(v, w, nextEdgeId));
            g[v].Add(new Edge(u, w, nextEdgeId));
            nextEdgeId++;
            deg[u]++;
            deg[v]++;
        }

        var odd = new List<int>();
        for (int i = 0; i < n; i++) if ((deg[i] & 1) != 0) odd.Add(i);

        int oddCount = odd.Count;
        // Exact minimum pairing only for small odd-set; greedy otherwise for scalability.
        int exactThreshold = 22;

        if (oddCount > 0)
        {
            // Build all-pairs distances between odd nodes via per-source Dijkstra.
            long[,] dOdd = new long[oddCount, oddCount];
            int[][] parentVAll = new int[oddCount][];
            int[][] parentEAll = new int[oddCount][];

            for (int i = 0; i < oddCount; i++)
            {
                Dijkstra(odd[i], g, null, out var dist, out var parentV, out var parentE);
                parentVAll[i] = parentV;
                parentEAll[i] = parentE;
                for (int j = 0; j < oddCount; j++) dOdd[i, j] = dist[odd[j]];
            }

            var pairs = new List<(int a, int b)>();

            if (oddCount <= exactThreshold)
            {
                int maxMask = 1 << oddCount;
                var dp = new long[maxMask];
                var choiceA = new int[maxMask];
                var choiceB = new int[maxMask];
                Array.Fill(dp, long.MaxValue / 4);
                dp[0] = 0;

                for (int mask = 1; mask < maxMask; mask++)
                {
                    if ((BitCount(mask) & 1) == 1) continue;
                    int i = 0;
                    while (((mask >> i) & 1) == 0) i++;
                    int m2 = mask ^ (1 << i);
                    for (int j = i + 1; j < oddCount; j++)
                    {
                        if (((m2 >> j) & 1) == 0) continue;
                        int prev = m2 ^ (1 << j);
                        long cand = dp[prev] + dOdd[i, j];
                        if (cand < dp[mask])
                        {
                            dp[mask] = cand;
                            choiceA[mask] = i;
                            choiceB[mask] = j;
                        }
                    }
                }

                int cur = maxMask - 1;
                while (cur != 0)
                {
                    int a = choiceA[cur];
                    int b = choiceB[cur];
                    pairs.Add((a, b));
                    cur ^= (1 << a);
                    cur ^= (1 << b);
                }
            }
            else
            {
                // Greedy nearest-neighbor matching (fast and robust at large scale).
                var rem = new HashSet<int>(Enumerable.Range(0, oddCount));
                while (rem.Count > 0)
                {
                    int a = rem.First();
                    rem.Remove(a);
                    long best = long.MaxValue;
                    int bestB = -1;
                    foreach (var b in rem)
                    {
                        long x = dOdd[a, b];
                        if (x < best)
                        {
                            best = x;
                            bestB = b;
                        }
                    }
                    rem.Remove(bestB);
                    pairs.Add((a, bestB));
                }
            }

            // Duplicate shortest paths for chosen pairings.
            foreach (var (aIdx, bIdx) in pairs)
            {
                int target = odd[bIdx];
                var parentV = parentVAll[aIdx];
                var cur = target;
                while (cur != odd[aIdx])
                {
                    int pv = parentV[cur];
                    if (pv < 0) break;
                    g[cur].Add(new Edge(pv, 1, nextEdgeId));
                    g[pv].Add(new Edge(cur, 1, nextEdgeId));
                    nextEdgeId++;
                    cur = pv;
                }
            }
        }

        int[] tour = BuildEulerTour(g, nextEdgeId);

        var sw = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };
        sw.Write(tour.Length);
        for (int i = 0; i < tour.Length; i++)
        {
            sw.Write(' ');
            sw.Write(tour[i]);
        }
        sw.WriteLine();
        sw.Flush();
    }

    private static int BitCount(int x)
    {
        int c = 0;
        while (x != 0) { x &= x - 1; c++; }
        return c;
    }
}
