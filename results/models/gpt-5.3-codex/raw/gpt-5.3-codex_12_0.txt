using System;
using System.Collections.Generic;
using System.Globalization;
using System.Numerics;

public static class Program
{
    private struct Triangle
    {
        public int A, B, C;
    }

    private struct Aabb
    {
        public double MinX, MinY, MinZ;
        public double MaxX, MaxY, MaxZ;

        public double SizeX => MaxX - MinX;
        public double SizeY => MaxY - MinY;
        public double SizeZ => MaxZ - MinZ;
        public double Volume => Math.Max(0.0, SizeX) * Math.Max(0.0, SizeY) * Math.Max(0.0, SizeZ);
    }

    private sealed class Orientation
    {
        public Quaternion Q;
        public Aabb Bounds;
    }

    private sealed class Placement
    {
        public double Tx, Ty, Tz;
        public Quaternion Q;
    }

    private sealed class SpatialHash
    {
        private readonly double _cell;
        private readonly Dictionary<long, List<int>> _map;

        public SpatialHash(double cell)
        {
            _cell = Math.Max(1e-6, cell);
            _map = new Dictionary<long, List<int>>(1 << 14);
        }

        private static long Key(int x, int y, int z)
        {
            unchecked
            {
                long h = 1469598103934665603L;
                h = (h ^ x) * 1099511628211L;
                h = (h ^ y) * 1099511628211L;
                h = (h ^ z) * 1099511628211L;
                return h;
            }
        }

        private int ToCell(double x)
        {
            return (int)Math.Floor(x / _cell);
        }

        public void Insert(int id, Aabb a)
        {
            int ix0 = ToCell(a.MinX), iy0 = ToCell(a.MinY), iz0 = ToCell(a.MinZ);
            int ix1 = ToCell(a.MaxX), iy1 = ToCell(a.MaxY), iz1 = ToCell(a.MaxZ);

            for (int ix = ix0; ix <= ix1; ix++)
            for (int iy = iy0; iy <= iy1; iy++)
            for (int iz = iz0; iz <= iz1; iz++)
            {
                long k = Key(ix, iy, iz);
                if (!_map.TryGetValue(k, out var list))
                {
                    list = new List<int>(4);
                    _map[k] = list;
                }
                list.Add(id);
            }
        }

        public void GatherCandidates(Aabb a, HashSet<int> outSet)
        {
            int ix0 = ToCell(a.MinX), iy0 = ToCell(a.MinY), iz0 = ToCell(a.MinZ);
            int ix1 = ToCell(a.MaxX), iy1 = ToCell(a.MaxY), iz1 = ToCell(a.MaxZ);

            for (int ix = ix0; ix <= ix1; ix++)
            for (int iy = iy0; iy <= iy1; iy++)
            for (int iz = iz0; iz <= iz1; iz++)
            {
                long k = Key(ix, iy, iz);
                if (_map.TryGetValue(k, out var list))
                {
                    for (int i = 0; i < list.Count; i++) outSet.Add(list[i]);
                }
            }
        }
    }

    private enum Scale
    {
        Trivial,
        Medium,
        Large,
        Extreme
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int v = fs.NextInt();
        int f = fs.NextInt();

        if (v <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var verts = new Vector3[v];
        for (int i = 0; i < v; i++)
        {
            float x = (float)fs.NextDouble();
            float y = (float)fs.NextDouble();
            float z = (float)fs.NextDouble();
            verts[i] = new Vector3(x, y, z);
        }

        var tris = new Triangle[Math.Max(0, f)];
        for (int i = 0; i < f; i++)
        {
            tris[i].A = fs.NextInt();
            tris[i].B = fs.NextInt();
            tris[i].C = fs.NextInt();
        }

        double W = fs.NextDouble();
        double H = fs.NextDouble();
        double D = fs.NextDouble();

        if (W <= 0 || H <= 0 || D <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        Scale scale = ClassifyScale(W, H, D, v, f);
        var orientations = BuildOrientations(verts, scale);
        if (orientations.Count == 0)
        {
            Console.WriteLine(0);
            return;
        }

        var chosen = ChooseBestOrientation(orientations, W, H, D);
        if (chosen == null)
        {
            Console.WriteLine(0);
            return;
        }

        var placements = PackByScale(chosen, W, H, D, scale, verts.Length);
        Emit(placements);
    }

    private static Scale ClassifyScale(double w, double h, double d, int v, int f)
    {
        double maxDim = Math.Max(w, Math.Max(h, d));
        long geomComplexity = (long)v + (long)f;

        if (maxDim <= 6 && geomComplexity <= 256) return Scale.Trivial;
        if (maxDim <= 25 && geomComplexity <= 2500) return Scale.Medium;
        if (maxDim <= 120 && geomComplexity <= 30000) return Scale.Large;
        return Scale.Extreme;
    }

    private static List<Orientation> BuildOrientations(Vector3[] verts, Scale scale)
    {
        var result = new List<Orientation>();
        var seen = new HashSet<string>(StringComparer.Ordinal);

        IEnumerable<Quaternion> qs = GenerateQuaternions(scale);
        foreach (var q in qs)
        {
            var b = ComputeBounds(verts, q);
            if (b.SizeX <= 1e-9 || b.SizeY <= 1e-9 || b.SizeZ <= 1e-9) continue;

            string key = QuantKey(b.SizeX, b.SizeY, b.SizeZ, 1e-4);
            if (seen.Add(key))
            {
                result.Add(new Orientation { Q = Quaternion.Normalize(q), Bounds = b });
            }
        }

        return result;
    }

    private static IEnumerable<Quaternion> GenerateQuaternions(Scale scale)
    {
        // Always include identity and axis-aligned 90-degree rotations.
        var baseRots = new List<Quaternion>
        {
            Quaternion.Identity,
            Quaternion.CreateFromAxisAngle(Vector3.UnitX, (float)(Math.PI * 0.5)),
            Quaternion.CreateFromAxisAngle(Vector3.UnitX, (float)Math.PI),
            Quaternion.CreateFromAxisAngle(Vector3.UnitX, (float)(Math.PI * 1.5)),
            Quaternion.CreateFromAxisAngle(Vector3.UnitY, (float)(Math.PI * 0.5)),
            Quaternion.CreateFromAxisAngle(Vector3.UnitY, (float)Math.PI),
            Quaternion.CreateFromAxisAngle(Vector3.UnitY, (float)(Math.PI * 1.5)),
            Quaternion.CreateFromAxisAngle(Vector3.UnitZ, (float)(Math.PI * 0.5)),
            Quaternion.CreateFromAxisAngle(Vector3.UnitZ, (float)Math.PI),
            Quaternion.CreateFromAxisAngle(Vector3.UnitZ, (float)(Math.PI * 1.5))
        };

        foreach (var q in baseRots) yield return Quaternion.Normalize(q);

        if (scale == Scale.Trivial || scale == Scale.Medium)
        {
            // Additional dense sampling for smaller instances.
            float[] ang =
            {
                0f,
                (float)(Math.PI / 6.0),
                (float)(Math.PI / 4.0),
                (float)(Math.PI / 3.0),
                (float)(Math.PI / 2.0)
            };

            for (int ix = 0; ix < ang.Length; ix++)
            for (int iy = 0; iy < ang.Length; iy++)
            for (int iz = 0; iz < ang.Length; iz++)
            {
                var qx = Quaternion.CreateFromAxisAngle(Vector3.UnitX, ang[ix]);
                var qy = Quaternion.CreateFromAxisAngle(Vector3.UnitY, ang[iy]);
                var qz = Quaternion.CreateFromAxisAngle(Vector3.UnitZ, ang[iz]);
                yield return Quaternion.Normalize(qz * qy * qx);
            }
        }
    }

    private static Orientation? ChooseBestOrientation(List<Orientation> orientations, double W, double H, double D)
    {
        Orientation? best = null;
        double bestCount = -1;
        double bestSlack = double.MaxValue;

        for (int i = 0; i < orientations.Count; i++)
        {
            var o = orientations[i];
            long nx = (long)Math.Floor(W / o.Bounds.SizeX);
            long ny = (long)Math.Floor(H / o.Bounds.SizeY);
            long nz = (long)Math.Floor(D / o.Bounds.SizeZ);
            if (nx <= 0 || ny <= 0 || nz <= 0) continue;

            double count = (double)nx * ny * nz;
            double slack = (W - nx * o.Bounds.SizeX) + (H - ny * o.Bounds.SizeY) + (D - nz * o.Bounds.SizeZ);

            if (count > bestCount || (Math.Abs(count - bestCount) < 0.5 && slack < bestSlack))
            {
                bestCount = count;
                bestSlack = slack;
                best = o;
            }
        }

        return best;
    }

    private static List<Placement> PackByScale(Orientation o, double W, double H, double D, Scale scale, int vertexCount)
    {
        return scale switch
        {
            Scale.Trivial => ExactGridPack(o, W, H, D, limit: 2_000_000),
            Scale.Medium => ExactGridPack(o, W, H, D, limit: 2_000_000),
            Scale.Large => HashedGreedyPack(o, W, H, D, maxCount: 1_500_000),
            _ => UltraFastGridPack(o, W, H, D, maxCount: 2_000_000)
        };
    }

    private static List<Placement> ExactGridPack(Orientation o, double W, double H, double D, int limit)
    {
        double sx = o.Bounds.SizeX, sy = o.Bounds.SizeY, sz = o.Bounds.SizeZ;
        long nx = (long)Math.Floor(W / sx);
        long ny = (long)Math.Floor(H / sy);
        long nz = (long)Math.Floor(D / sz);

        if (nx <= 0 || ny <= 0 || nz <= 0) return new List<Placement>(0);

        long total = nx * ny * nz;
        int cap = (int)Math.Min(total, limit);
        var list = new List<Placement>(cap);

        double ox = -o.Bounds.MinX;
        double oy = -o.Bounds.MinY;
        double oz = -o.Bounds.MinZ;

        for (long iz = 0; iz < nz && list.Count < cap; iz++)
        for (long iy = 0; iy < ny && list.Count < cap; iy++)
        for (long ix = 0; ix < nx && list.Count < cap; ix++)
        {
            list.Add(new Placement
            {
                Tx = ox + ix * sx,
                Ty = oy + iy * sy,
                Tz = oz + iz * sz,
                Q = o.Q
            });
        }

        return list;
    }

    private static List<Placement> HashedGreedyPack(Orientation o, double W, double H, double D, int maxCount)
    {
        double sx = o.Bounds.SizeX, sy = o.Bounds.SizeY, sz = o.Bounds.SizeZ;
        long nx = (long)Math.Floor(W / sx);
        long ny = (long)Math.Floor(H / sy);
        long nz = (long)Math.Floor(D / sz);
        if (nx <= 0 || ny <= 0 || nz <= 0) return new List<Placement>(0);

        double ox = -o.Bounds.MinX;
        double oy = -o.Bounds.MinY;
        double oz = -o.Bounds.MinZ;

        var placedAabbs = new List<Aabb>(Math.Min(maxCount, 200000));
        var placed = new List<Placement>(Math.Min(maxCount, 200000));
        var hash = new SpatialHash(Math.Max(sx, Math.Max(sy, sz)));
        var cand = new HashSet<int>();

        for (long iz = 0; iz < nz && placed.Count < maxCount; iz++)
        {
            for (long iy = 0; iy < ny && placed.Count < maxCount; iy++)
            {
                for (long ix = 0; ix < nx && placed.Count < maxCount; ix++)
                {
                    double tx = ox + ix * sx;
                    double ty = oy + iy * sy;
                    double tz = oz + iz * sz;

                    var box = new Aabb
                    {
                        MinX = tx + o.Bounds.MinX,
                        MinY = ty + o.Bounds.MinY,
                        MinZ = tz + o.Bounds.MinZ,
                        MaxX = tx + o.Bounds.MaxX,
                        MaxY = ty + o.Bounds.MaxY,
                        MaxZ = tz + o.Bounds.MaxZ
                    };

                    cand.Clear();
                    hash.GatherCandidates(box, cand);

                    bool overlap = false;
                    foreach (int id in cand)
                    {
                        if (Intersects(box, placedAabbs[id]))
                        {
                            overlap = true;
                            break;
                        }
                    }

                    if (!overlap)
                    {
                        int id = placedAabbs.Count;
                        placedAabbs.Add(box);
                        hash.Insert(id, box);
                        placed.Add(new Placement { Tx = tx, Ty = ty, Tz = tz, Q = o.Q });
                    }
                }
            }
        }

        return placed;
    }

    private static List<Placement> UltraFastGridPack(Orientation o, double W, double H, double D, int maxCount)
    {
        // Extreme-scale mode: deterministic O(N) streaming with bounded memory.
        double sx = o.Bounds.SizeX, sy = o.Bounds.SizeY, sz = o.Bounds.SizeZ;
        long nx = (long)Math.Floor(W / sx);
        long ny = (long)Math.Floor(H / sy);
        long nz = (long)Math.Floor(D / sz);

        if (nx <= 0 || ny <= 0 || nz <= 0) return new List<Placement>(0);

        long total = nx * ny * nz;
        int take = (int)Math.Min(total, (long)maxCount);
        var list = new List<Placement>(take);

        double ox = -o.Bounds.MinX;
        double oy = -o.Bounds.MinY;
        double oz = -o.Bounds.MinZ;

        for (long linear = 0; linear < total && list.Count < take; linear++)
        {
            long ix = linear % nx;
            long iy = (linear / nx) % ny;
            long iz = linear / (nx * ny);

            list.Add(new Placement
            {
                Tx = ox + ix * sx,
                Ty = oy + iy * sy,
                Tz = oz + iz * sz,
                Q = o.Q
            });
        }

        return list;
    }

    private static bool Intersects(Aabb a, Aabb b)
    {
        return a.MinX < b.MaxX && a.MaxX > b.MinX
            && a.MinY < b.MaxY && a.MaxY > b.MinY
            && a.MinZ < b.MaxZ && a.MaxZ > b.MinZ;
    }

    private static Aabb ComputeBounds(Vector3[] verts, Quaternion q)
    {
        Vector3 r0 = Vector3.Transform(verts[0], q);
        double minX = r0.X, minY = r0.Y, minZ = r0.Z;
        double maxX = r0.X, maxY = r0.Y, maxZ = r0.Z;

        for (int i = 1; i < verts.Length; i++)
        {
            Vector3 r = Vector3.Transform(verts[i], q);
            if (r.X < minX) minX = r.X;
            if (r.Y < minY) minY = r.Y;
            if (r.Z < minZ) minZ = r.Z;
            if (r.X > maxX) maxX = r.X;
            if (r.Y > maxY) maxY = r.Y;
            if (r.Z > maxZ) maxZ = r.Z;
        }

        return new Aabb { MinX = minX, MinY = minY, MinZ = minZ, MaxX = maxX, MaxY = maxY, MaxZ = maxZ };
    }

    private static string QuantKey(double x, double y, double z, double eps)
    {
        long qx = (long)Math.Round(x / eps);
        long qy = (long)Math.Round(y / eps);
        long qz = (long)Math.Round(z / eps);
        return qx.ToString(CultureInfo.InvariantCulture) + ":" +
               qy.ToString(CultureInfo.InvariantCulture) + ":" +
               qz.ToString(CultureInfo.InvariantCulture);
    }

    private static void Emit(List<Placement> placements)
    {
        var inv = CultureInfo.InvariantCulture;
        Console.WriteLine(placements.Count);
        for (int i = 0; i < placements.Count; i++)
        {
            var p = placements[i];
            Console.WriteLine(
                p.Tx.ToString("0.######", inv) + " " +
                p.Ty.ToString("0.######", inv) + " " +
                p.Tz.ToString("0.######", inv) + " " +
                p.Q.W.ToString("0.######", inv) + " " +
                p.Q.X.ToString("0.######", inv) + " " +
                p.Q.Y.ToString("0.######", inv) + " " +
                p.Q.Z.ToString("0.######", inv));
        }
    }

    private sealed class FastScanner
    {
        private readonly System.IO.Stream _stream;
        private readonly byte[] _buf;
        private int _len, _ptr;

        public FastScanner()
        {
            _stream = Console.OpenStandardInput();
            _buf = new byte[1 << 16];
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }

        private int SkipWhite()
        {
            int c;
            do c = Read(); while (c >= 0 && c <= 32);
            return c;
        }

        public int NextInt()
        {
            int c = SkipWhite();
            int sign = 1;
            if (c == '-') { sign = -1; c = Read(); }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }

        public double NextDouble()
        {
            int c = SkipWhite();
            int sign = 1;
            if (c == '-') { sign = -1; c = Read(); }

            double val = 0;
            while (c > 32 && c != '.' && c != 'e' && c != 'E')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }

            if (c == '.')
            {
                double mul = 0.1;
                c = Read();
                while (c > 32 && c != 'e' && c != 'E')
                {
                    val += (c - '0') * mul;
                    mul *= 0.1;
                    c = Read();
                }
            }

            if (c == 'e' || c == 'E')
            {
                int esign = 1;
                c = Read();
                if (c == '-') { esign = -1; c = Read(); }
                else if (c == '+') { c = Read(); }

                int exp = 0;
                while (c > 32)
                {
                    exp = exp * 10 + (c - '0');
                    c = Read();
                }

                val *= Math.Pow(10, esign * exp);
            }

            return sign * val;
        }
    }
}
