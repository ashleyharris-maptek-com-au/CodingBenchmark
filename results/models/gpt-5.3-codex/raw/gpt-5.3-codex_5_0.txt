use std::cmp::Reverse;
use std::collections::{BinaryHeap, VecDeque};
use std::io::{self, Read};

#[derive(Clone, Copy, Debug, PartialEq, Eq)]
struct Pt {
    x: i32,
    y: i32,
}

impl Pt {
    fn manhattan(self, o: Pt) -> i32 {
        (self.x - o.x).abs() + (self.y - o.y).abs()
    }
}

struct Grid {
    w: usize,
    h: usize,
    blocked: Vec<bool>,
    free_count: usize,
}

impl Grid {
    fn idx(&self, x: usize, y: usize) -> usize {
        y * self.w + x
    }
    fn in_bounds(&self, x: i32, y: i32) -> bool {
        x >= 0 && y >= 0 && (x as usize) < self.w && (y as usize) < self.h
    }
    fn is_free_xy(&self, x: usize, y: usize) -> bool {
        !self.blocked[self.idx(x, y)]
    }
    fn is_free_pt(&self, p: Pt) -> bool {
        self.in_bounds(p.x, p.y) && self.is_free_xy(p.x as usize, p.y as usize)
    }
    fn neighbors(&self, p: Pt) -> [Option<Pt>; 4] {
        let dirs = [(1, 0), (-1, 0), (0, 1), (0, -1)];
        let mut out = [None, None, None, None];
        for (i, (dx, dy)) in dirs.iter().enumerate() {
            let nx = p.x + dx;
            let ny = p.y + dy;
            if self.in_bounds(nx, ny) && self.is_free_xy(nx as usize, ny as usize) {
                out[i] = Some(Pt { x: nx, y: ny });
            }
        }
        out
    }
    fn degree_unvisited(&self, p: Pt, used: &[bool]) -> usize {
        self.neighbors(p)
            .iter()
            .flatten()
            .filter(|q| !used[self.idx(q.x as usize, q.y as usize)])
            .count()
    }
}

fn bipartite_counts(grid: &Grid) -> (usize, usize) {
    let mut a = 0usize;
    let mut b = 0usize;
    for y in 0..grid.h {
        for x in 0..grid.w {
            if grid.is_free_xy(x, y) {
                if (x + y) % 2 == 0 {
                    a += 1;
                } else {
                    b += 1;
                }
            }
        }
    }
    (a, b)
}

fn is_connected_free(grid: &Grid, start: Pt) -> bool {
    if !grid.is_free_pt(start) {
        return false;
    }
    let mut vis = vec![false; grid.w * grid.h];
    let mut q = VecDeque::new();
    let sidx = grid.idx(start.x as usize, start.y as usize);
    vis[sidx] = true;
    q.push_back(start);
    let mut cnt = 1usize;
    while let Some(p) = q.pop_front() {
        for n in grid.neighbors(p).iter().flatten() {
            let i = grid.idx(n.x as usize, n.y as usize);
            if !vis[i] {
                vis[i] = true;
                cnt += 1;
                q.push_back(*n);
            }
        }
    }
    cnt == grid.free_count
}

fn parity_prune(grid: &Grid, require_cycle: bool, start: Pt) -> bool {
    let (a, b) = bipartite_counts(grid);
    if require_cycle {
        a == b
    } else {
        let n = grid.free_count;
        if n == 0 {
            return false;
        }
        let start_color_even = ((start.x + start.y) & 1) == 0;
        if n % 2 == 0 {
            a == b
        } else if start_color_even {
            a == b + 1
        } else {
            b == a + 1
        }
    }
}

fn articulation_prune(grid: &Grid, used: &[bool], current: Pt, visited_count: usize) -> bool {
    let remaining = grid.free_count - visited_count;
    if remaining <= 1 {
        return true;
    }

    let n = grid.w * grid.h;
    let mut vis = vec![false; n];
    let mut start_opt = None;
    for npt in grid.neighbors(current).iter().flatten() {
        let i = grid.idx(npt.x as usize, npt.y as usize);
        if !used[i] {
            start_opt = Some(*npt);
            break;
        }
    }
    let start = if let Some(s) = start_opt { s } else { return false };

    let mut q = VecDeque::new();
    let si = grid.idx(start.x as usize, start.y as usize);
    vis[si] = true;
    q.push_back(start);
    let mut seen = 1usize;

    while let Some(p) = q.pop_front() {
        for nb in grid.neighbors(p).iter().flatten() {
            let i = grid.idx(nb.x as usize, nb.y as usize);
            if used[i] || vis[i] {
                continue;
            }
            vis[i] = true;
            seen += 1;
            q.push_back(*nb);
        }
    }

    seen == remaining
}

fn exact_dfs(grid: &Grid, require_cycle: bool) -> Option<Vec<Pt>> {
    let start = Pt { x: 0, y: 0 };
    if !grid.is_free_pt(start) || grid.free_count == 0 {
        return None;
    }
    if !is_connected_free(grid, start) || !parity_prune(grid, require_cycle, start) {
        return None;
    }

    let n = grid.w * grid.h;
    let mut used = vec![false; n];
    let mut path = Vec::with_capacity(grid.free_count);

    fn rec(
        grid: &Grid,
        require_cycle: bool,
        p: Pt,
        used: &mut [bool],
        path: &mut Vec<Pt>,
        visited_count: usize,
    ) -> bool {
        let pi = grid.idx(p.x as usize, p.y as usize);
        used[pi] = true;
        path.push(p);

        if visited_count == grid.free_count {
            if !require_cycle {
                return true;
            }
            let ok = grid
                .neighbors(p)
                .iter()
                .flatten()
                .any(|q| q.x == 0 && q.y == 0);
            if ok {
                return true;
            }
            used[pi] = false;
            path.pop();
            return false;
        }

        if !articulation_prune(grid, used, p, visited_count) {
            used[pi] = false;
            path.pop();
            return false;
        }

        let mut cand: Vec<Pt> = grid
            .neighbors(p)
            .iter()
            .flatten()
            .copied()
            .filter(|q| !used[grid.idx(q.x as usize, q.y as usize)])
            .collect();

        cand.sort_by_key(|q| {
            let deg = grid.degree_unvisited(*q, used);
            let dist0 = q.manhattan(Pt { x: 0, y: 0 }) as usize;
            (deg, Reverse(dist0))
        });

        for q in cand {
            if rec(grid, require_cycle, q, used, path, visited_count + 1) {
                return true;
            }
        }

        used[pi] = false;
        path.pop();
        false
    }

    if rec(grid, require_cycle, start, &mut used, &mut path, 1) {
        Some(path)
    } else {
        None
    }
}

fn serpentine_full_rect(w: usize, h: usize) -> Vec<Pt> {
    let mut p = Vec::with_capacity(w * h);
    for y in 0..h {
        if y % 2 == 0 {
            for x in 0..w {
                p.push(Pt {
                    x: x as i32,
                    y: y as i32,
                });
            }
        } else {
            for x in (0..w).rev() {
                p.push(Pt {
                    x: x as i32,
                    y: y as i32,
                });
            }
        }
    }
    p
}

fn greedy_path(grid: &Grid, require_cycle: bool) -> Option<Vec<Pt>> {
    let start = Pt { x: 0, y: 0 };
    if !grid.is_free_pt(start) || grid.free_count == 0 {
        return None;
    }
    if !is_connected_free(grid, start) || !parity_prune(grid, require_cycle, start) {
        return None;
    }

    let n = grid.w * grid.h;
    let mut used = vec![false; n];
    let mut path = Vec::with_capacity(grid.free_count);

    let mut cur = start;
    used[grid.idx(cur.x as usize, cur.y as usize)] = true;
    path.push(cur);

    for step in 1..grid.free_count {
        let mut cand: Vec<Pt> = grid
            .neighbors(cur)
            .iter()
            .flatten()
            .copied()
            .filter(|q| !used[grid.idx(q.x as usize, q.y as usize)])
            .collect();

        if cand.is_empty() {
            return None;
        }

        cand.sort_by_key(|q| {
            let mut tmp_used = used.clone();
            tmp_used[grid.idx(q.x as usize, q.y as usize)] = true;
            let deg = grid.degree_unvisited(*q, &tmp_used);
            let bridge_penalty = if articulation_prune(grid, &tmp_used, *q, step + 1) { 0 } else { 1000 };
            let dist0 = q.manhattan(Pt { x: 0, y: 0 }) as usize;
            (bridge_penalty, deg, Reverse(dist0))
        });

        let next = cand[0];
        used[grid.idx(next.x as usize, next.y as usize)] = true;
        path.push(next);
        cur = next;
    }

    if require_cycle {
        let ok = grid
            .neighbors(cur)
            .iter()
            .flatten()
            .any(|q| q.x == 0 && q.y == 0);
        if !ok {
            return None;
        }
    }

    Some(path)
}

fn validate(grid: &Grid, path: &[Pt], require_cycle: bool) -> bool {
    if path.len() != grid.free_count || path.is_empty() {
        return false;
    }
    if path[0].x != 0 || path[0].y != 0 {
        return false;
    }
    let mut seen = vec![false; grid.w * grid.h];
    for (i, p) in path.iter().enumerate() {
        if !grid.is_free_pt(*p) {
            return false;
        }
        let idx = grid.idx(p.x as usize, p.y as usize);
        if seen[idx] {
            return false;
        }
        seen[idx] = true;
        if i > 0 {
            if path[i - 1].manhattan(*p) != 1 {
                return false;
            }
        }
    }
    if require_cycle && path[path.len() - 1].manhattan(path[0]) != 1 {
        return false;
    }
    true
}

fn solve(grid: &Grid, require_cycle: bool) -> Option<Vec<Pt>> {
    let start = Pt { x: 0, y: 0 };
    if !grid.is_free_pt(start) || grid.free_count == 0 {
        return None;
    }

    if !is_connected_free(grid, start) || !parity_prune(grid, require_cycle, start) {
        return None;
    }

    if grid.blocked.iter().all(|&b| !b) && !require_cycle {
        let p = serpentine_full_rect(grid.w, grid.h);
        if validate(grid, &p, require_cycle) {
            return Some(p);
        }
    }

    let small_threshold = 90usize;
    if grid.free_count <= small_threshold {
        if let Some(p) = exact_dfs(grid, require_cycle) {
            if validate(grid, &p, require_cycle) {
                return Some(p);
            }
        }
    }

    if let Some(p) = greedy_path(grid, require_cycle) {
        if validate(grid, &p, require_cycle) {
            return Some(p);
        }
    }

    if grid.free_count <= 180 {
        if let Some(p) = exact_dfs(grid, require_cycle) {
            if validate(grid, &p, require_cycle) {
                return Some(p);
            }
        }
    }

    None
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        println!("0");
        return;
    }

    let mut it = input.split_whitespace();
    let w: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let h: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let k: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let require_cycle = match it.next().and_then(|s| s.parse::<u8>().ok()) {
        Some(v) => v != 0,
        None => {
            println!("0");
            return;
        }
    };

    if w == 0 || h == 0 {
        println!("0");
        return;
    }

    let mut blocked = vec![false; w * h];
    for _ in 0..k {
        let x: usize = if let Some(v) = it.next().and_then(|s| s.parse().ok()) {
            v
        } else {
            println!("0");
            return;
        };
        let y: usize = if let Some(v) = it.next().and_then(|s| s.parse().ok()) {
            v
        } else {
            println!("0");
            return;
        };
        if x >= w || y >= h {
            println!("0");
            return;
        }
        blocked[y * w + x] = true;
    }

    let free_count = blocked.iter().filter(|&&b| !b).count();
    let grid = Grid {
        w,
        h,
        blocked,
        free_count,
    };

    match solve(&grid, require_cycle) {
        Some(path) => {
            println!("{}", path.len());
            for p in path {
                println!("{} {}", p.x, p.y);
            }
        }
        None => println!("0"),
    }
}
