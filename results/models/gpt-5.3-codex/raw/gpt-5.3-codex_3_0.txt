use std::cmp::Ordering;
use std::collections::VecDeque;
use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone)]
struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        let s = if seed == 0 { 0x9E3779B97F4A7C15 } else { seed };
        Self { state: s }
    }

    #[inline]
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 7;
        x ^= x >> 9;
        x ^= x << 8;
        self.state = x;
        x
    }

    #[inline]
    fn next_f64(&mut self) -> f64 {
        // [0,1)
        let v = self.next_u64() >> 11;
        (v as f64) * (1.0 / ((1u64 << 53) as f64))
    }

    #[inline]
    fn gen_range_usize(&mut self, hi: usize) -> usize {
        if hi <= 1 {
            0
        } else {
            (self.next_u64() % hi as u64) as usize
        }
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }

    let mut it = input.split_whitespace();
    let n: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let m: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };

    let mut edges_raw: Vec<(usize, usize)> = Vec::with_capacity(m);
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];

    for _ in 0..m {
        let u = match it.next().and_then(|s| s.parse::<usize>().ok()) {
            Some(v) if v < n => v,
            _ => continue,
        };
        let v = match it.next().and_then(|s| s.parse::<usize>().ok()) {
            Some(vv) if vv < n => vv,
            _ => continue,
        };
        edges_raw.push((u, v));
        if u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    if n == 0 {
        return;
    }
    if n == 1 {
        println!("0 0");
        return;
    }

    let mut deg: Vec<usize> = vec![0; n];
    let mut undirected_edges = 0usize;
    for u in 0..n {
        deg[u] = adj[u].len();
        undirected_edges += deg[u];
    }
    undirected_edges /= 2;

    let density = if n > 1 {
        (2.0 * undirected_edges as f64) / ((n as f64) * ((n - 1) as f64))
    } else {
        0.0
    };

    let mut comp_id = vec![usize::MAX; n];
    let mut comps: Vec<Vec<usize>> = Vec::new();
    {
        let mut q = VecDeque::new();
        let mut cid = 0usize;
        for s in 0..n {
            if comp_id[s] != usize::MAX {
                continue;
            }
            comp_id[s] = cid;
            q.push_back(s);
            let mut nodes = Vec::new();
            while let Some(u) = q.pop_front() {
                nodes.push(u);
                for &v in &adj[u] {
                    if comp_id[v] == usize::MAX {
                        comp_id[v] = cid;
                        q.push_back(v);
                    }
                }
            }
            comps.push(nodes);
            cid += 1;
        }
    }

    let mut xs = vec![0.0f64; n];
    let mut ys = vec![0.0f64; n];

    // Adaptive strategy selection.
    if n <= 1200 {
        // Medium / moderately large: multilevel-like spectral seed + FR refinement.
        layout_spectral_seed(n, &adj, &deg, &mut xs, &mut ys);
        let its = if n <= 80 {
            500
        } else if n <= 300 {
            300
        } else if n <= 800 {
            180
        } else {
            100
        };
        fr_refine(n, &adj, &mut xs, &mut ys, its, 16);
        local_barycenter_smoothing(n, &adj, &mut xs, &mut ys, 2);
    } else if n <= 15_000 {
        // Large: component-aware radial + light refinement.
        layout_component_radial(n, &comps, &adj, &deg, &mut xs, &mut ys);
        let its = if n <= 5000 { 80 } else { 35 };
        fr_refine(n, &adj, &mut xs, &mut ys, its, 8);
    } else {
        // Extreme: ultra-fast deterministic circular-by-order with tiny smoothing.
        layout_extreme_fast(n, &adj, &deg, &mut xs, &mut ys, density);
        local_barycenter_smoothing(n, &adj, &mut xs, &mut ys, 1);
    }

    normalize_layout(&mut xs, &mut ys);

    for i in 0..n {
        println!("{:.6} {:.6}", xs[i], ys[i]);
    }
}

fn layout_spectral_seed(
    n: usize,
    adj: &[Vec<usize>],
    deg: &[usize],
    xs: &mut [f64],
    ys: &mut [f64],
) {
    let mut rng = XorShift64::new(0x1234_5678_9ABC_DEF0 ^ n as u64);
    let mut vx = vec![0.0; n];
    let mut vy = vec![0.0; n];

    for i in 0..n {
        vx[i] = rng.next_f64() * 2.0 - 1.0;
        vy[i] = rng.next_f64() * 2.0 - 1.0;
    }

    orthonormalize(&mut vx, None);
    orthonormalize(&mut vy, Some(&vx));

    let steps = if n <= 150 { 90 } else if n <= 600 { 55 } else { 35 };
    let alpha = 0.08;

    let mut tx = vec![0.0; n];
    let mut ty = vec![0.0; n];

    for _ in 0..steps {
        for i in 0..n {
            let mut sx = 0.0;
            let mut sy = 0.0;
            for &j in &adj[i] {
                sx += vx[j];
                sy += vy[j];
            }
            let d = deg[i] as f64;
            tx[i] = (1.0 - alpha) * vx[i] + alpha * (sx - d * vx[i]);
            ty[i] = (1.0 - alpha) * vy[i] + alpha * (sy - d * vy[i]);
        }
        vx.copy_from_slice(&tx);
        vy.copy_from_slice(&ty);
        orthonormalize(&mut vx, None);
        orthonormalize(&mut vy, Some(&vx));
    }

    for i in 0..n {
        xs[i] = vx[i];
        ys[i] = vy[i];
    }
}

fn orthonormalize(v: &mut [f64], against: Option<&[f64]>) {
    let n = v.len();
    let mean = v.iter().sum::<f64>() / n as f64;
    for x in v.iter_mut() {
        *x -= mean;
    }
    if let Some(a) = against {
        let mut dot = 0.0;
        let mut na = 0.0;
        for i in 0..n {
            dot += v[i] * a[i];
            na += a[i] * a[i];
        }
        if na > 1e-20 {
            let c = dot / na;
            for i in 0..n {
                v[i] -= c * a[i];
            }
        }
    }
    let mut norm = v.iter().map(|x| x * x).sum::<f64>().sqrt();
    if norm < 1e-20 {
        norm = 1.0;
    }
    for x in v.iter_mut() {
        *x /= norm;
    }
}

fn fr_refine(
    n: usize,
    adj: &[Vec<usize>],
    xs: &mut [f64],
    ys: &mut [f64],
    iters: usize,
    grid_bins: usize,
) {
    if n <= 2 || iters == 0 {
        return;
    }

    let area = 4.0;
    let k = (area / n as f64).sqrt().max(1e-4);
    let mut temp = 0.25;

    let mut dx = vec![0.0f64; n];
    let mut dy = vec![0.0f64; n];

    for _ in 0..iters {
        dx.fill(0.0);
        dy.fill(0.0);

        // Approximate repulsion using spatial bins.
        let bins = grid_bins.max(4);
        let mut cells: Vec<Vec<usize>> = vec![Vec::new(); bins * bins];

        for i in 0..n {
            let gx = (((xs[i] + 1.0) * 0.5) * bins as f64).floor() as isize;
            let gy = (((ys[i] + 1.0) * 0.5) * bins as f64).floor() as isize;
            let cx = gx.clamp(0, bins as isize - 1) as usize;
            let cy = gy.clamp(0, bins as isize - 1) as usize;
            cells[cy * bins + cx].push(i);
        }

        for cy in 0..bins {
            for cx in 0..bins {
                let cidx = cy * bins + cx;
                let here = &cells[cidx];

                // Intra-cell exact.
                for a in 0..here.len() {
                    let i = here[a];
                    for &j in &here[a + 1..] {
                        let mut rx = xs[i] - xs[j];
                        let mut ry = ys[i] - ys[j];
                        let dist2 = rx * rx + ry * ry + 1e-9;
                        let dist = dist2.sqrt();
                        let f = (k * k) / dist;
                        rx /= dist;
                        ry /= dist;
                        dx[i] += rx * f;
                        dy[i] += ry * f;
                        dx[j] -= rx * f;
                        dy[j] -= ry * f;
                    }
                }

                // Neighbor cells.
                for ny in cy.saturating_sub(1)..=(cy + 1).min(bins - 1) {
                    for nx in cx.saturating_sub(1)..=(cx + 1).min(bins - 1) {
                        let nidx = ny * bins + nx;
                        if nidx <= cidx {
                            continue;
                        }
                        let other = &cells[nidx];
                        for &i in here {
                            for &j in other {
                                let mut rx = xs[i] - xs[j];
                                let mut ry = ys[i] - ys[j];
                                let dist2 = rx * rx + ry * ry + 1e-9;
                                let dist = dist2.sqrt();
                                let f = (k * k) / dist;
                                rx /= dist;
                                ry /= dist;
                                dx[i] += rx * f;
                                dy[i] += ry * f;
                                dx[j] -= rx * f;
                                dy[j] -= ry * f;
                            }
                        }
                    }
                }
            }
        }

        // Attractive edges.
        for u in 0..n {
            for &v in &adj[u] {
                if u < v {
                    let mut ax = xs[u] - xs[v];
                    let mut ay = ys[u] - ys[v];
                    let dist2 = ax * ax + ay * ay + 1e-9;
                    let dist = dist2.sqrt();
                    let f = dist2 / k;
                    ax /= dist;
                    ay /= dist;
                    dx[u] -= ax * f;
                    dy[u] -= ay * f;
                    dx[v] += ax * f;
                    dy[v] += ay * f;
                }
            }
        }

        // Move with temperature cap.
        for i in 0..n {
            let d2 = dx[i] * dx[i] + dy[i] * dy[i];
            if d2 > 1e-20 {
                let d = d2.sqrt();
                let lim = temp.min(d);
                xs[i] += dx[i] / d * lim;
                ys[i] += dy[i] / d * lim;
            }
        }

        temp *= 0.96;
        if temp < 1e-4 {
            break;
        }

        // Keep centered and bounded.
        recenter(xs, ys);
        soft_clip(xs, ys, 2.5);
    }
}

fn local_barycenter_smoothing(n: usize, adj: &[Vec<usize>], xs: &mut [f64], ys: &mut [f64], rounds: usize) {
    if rounds == 0 {
        return;
    }
    let mut nx = vec![0.0; n];
    let mut ny = vec![0.0; n];

    for _ in 0..rounds {
        for i in 0..n {
            if adj[i].is_empty() {
                nx[i] = xs[i];
                ny[i] = ys[i];
            } else {
                let mut sx = 0.0;
                let mut sy = 0.0;
                for &j in &adj[i] {
                    sx += xs[j];
                    sy += ys[j];
                }
                let inv = 1.0 / adj[i].len() as f64;
                let bx = sx * inv;
                let by = sy * inv;
                nx[i] = 0.75 * xs[i] + 0.25 * bx;
                ny[i] = 0.75 * ys[i] + 0.25 * by;
            }
        }
        xs.copy_from_slice(&nx);
        ys.copy_from_slice(&ny);
    }
}

fn layout_component_radial(
    n: usize,
    comps: &[Vec<usize>],
    adj: &[Vec<usize>],
    deg: &[usize],
    xs: &mut [f64],
    ys: &mut [f64],
) {
    let mut comp_order: Vec<usize> = (0..comps.len()).collect();
    comp_order.sort_by(|&a, &b| comps[b].len().cmp(&comps[a].len()));

    let mut rng = XorShift64::new(0xCAFEBABE12345678 ^ n as u64);

    let mut ring = 0usize;
    let mut idx_on_ring = 0usize;
    let mut ring_cap = 6usize;

    for &cid in &comp_order {
        let nodes = &comps[cid];
        let csize = nodes.len();
        if csize == 0 {
            continue;
        }

        if idx_on_ring >= ring_cap {
            ring += 1;
            idx_on_ring = 0;
            ring_cap = 6 * (ring + 1);
        }

        let base_r = 4.0 + ring as f64 * 6.0;
        let ang = 2.0 * PI * (idx_on_ring as f64 / ring_cap as f64);
        idx_on_ring += 1;

        let cx = base_r * ang.cos();
        let cy = base_r * ang.sin();

        let mut order = nodes.clone();
        order.sort_by(|&u, &v| {
            deg[v]
                .cmp(&deg[u])
                .then_with(|| u.cmp(&v))
        });

        if csize == 1 {
            let u = order[0];
            xs[u] = cx;
            ys[u] = cy;
            continue;
        }

        // Place highest degree near center and others on local spiral.
        for (t, &u) in order.iter().enumerate() {
            let rr = 0.4 + 0.15 * (t as f64).sqrt();
            let aa = 2.399963229728653 * t as f64 + rng.next_f64() * 0.05;
            xs[u] = cx + rr * aa.cos();
            ys[u] = cy + rr * aa.sin();
        }

        // One cheap local relaxation pass inside component.
        for &u in nodes {
            if adj[u].is_empty() {
                continue;
            }
            let mut sx = 0.0;
            let mut sy = 0.0;
            let mut cnt = 0usize;
            for &v in &adj[u] {
                sx += xs[v];
                sy += ys[v];
                cnt += 1;
            }
            if cnt > 0 {
                let inv = 1.0 / cnt as f64;
                xs[u] = 0.7 * xs[u] + 0.3 * sx * inv;
                ys[u] = 0.7 * ys[u] + 0.3 * sy * inv;
            }
        }
    }
}

fn layout_extreme_fast(
    n: usize,
    adj: &[Vec<usize>],
    deg: &[usize],
    xs: &mut [f64],
    ys: &mut [f64],
    density: f64,
) {
    let mut nodes: Vec<usize> = (0..n).collect();

    // Degree-first + neighbor-degree tie break gives decent crossing reduction cheaply.
    let mut ndeg_sum = vec![0usize; n];
    for u in 0..n {
        let mut s = 0usize;
        for &v in &adj[u] {
            s += deg[v];
        }
        ndeg_sum[u] = s;
    }

    nodes.sort_by(|&a, &b| {
        deg[b]
            .cmp(&deg[a])
            .then_with(|| ndeg_sum[b].cmp(&ndeg_sum[a]))
            .then_with(|| a.cmp(&b))
    });

    // Alternate placement to spread hubs.
    let mut perm = vec![0usize; n];
    let mut l = 0usize;
    let mut r = n.saturating_sub(1);
    for (i, &u) in nodes.iter().enumerate() {
        if i % 2 == 0 {
            perm[l] = u;
            l += 1;
        } else {
            perm[r] = u;
            if r > 0 {
                r -= 1;
            }
        }
    }

    let radius = (n as f64).sqrt() * (if density > 0.05 { 1.0 } else { 1.6 });
    for (i, &u) in perm.iter().enumerate() {
        let a = 2.0 * PI * (i as f64 / n as f64);
        xs[u] = radius * a.cos();
        ys[u] = radius * a.sin();
    }
}

fn recenter(xs: &mut [f64], ys: &mut [f64]) {
    let n = xs.len();
    if n == 0 {
        return;
    }
    let mx = xs.iter().sum::<f64>() / n as f64;
    let my = ys.iter().sum::<f64>() / n as f64;
    for i in 0..n {
        xs[i] -= mx;
        ys[i] -= my;
    }
}

fn soft_clip(xs: &mut [f64], ys: &mut [f64], lim: f64) {
    for i in 0..xs.len() {
        let r2 = xs[i] * xs[i] + ys[i] * ys[i];
        let lim2 = lim * lim;
        if r2 > lim2 {
            let r = r2.sqrt();
            let s = lim / r;
            xs[i] *= s;
            ys[i] *= s;
        }
    }
}

fn normalize_layout(xs: &mut [f64], ys: &mut [f64]) {
    recenter(xs, ys);

    let mut max_abs = 0.0f64;
    for i in 0..xs.len() {
        max_abs = max_abs.max(xs[i].abs()).max(ys[i].abs());
    }
    if max_abs < 1e-12 {
        // fallback tiny circle
        let n = xs.len();
        for i in 0..n {
            let a = 2.0 * PI * (i as f64 / n as f64);
            xs[i] = a.cos();
            ys[i] = a.sin();
        }
        return;
    }

    let inv = 1.0 / max_abs;
    for i in 0..xs.len() {
        xs[i] *= inv;
        ys[i] *= inv;
    }

    // Deterministic orientation for stability.
    let mut idx: Vec<usize> = (0..xs.len()).collect();
    idx.sort_by(|&a, &b| {
        xs[a]
            .partial_cmp(&xs[b])
            .unwrap_or(Ordering::Equal)
            .then_with(|| ys[a].partial_cmp(&ys[b]).unwrap_or(Ordering::Equal))
    });
    if let Some(&left) = idx.first() {
        if ys[left] < 0.0 {
            for y in ys.iter_mut() {
                *y = -*y;
            }
        }
    }
}
