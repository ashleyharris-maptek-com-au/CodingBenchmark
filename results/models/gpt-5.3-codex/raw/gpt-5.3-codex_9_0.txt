using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private struct Cut
    {
        public int Length;
        public int Index;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public bool TryReadInt(out int value)
        {
            int c;
            do
            {
                c = Read();
                if (c == -1)
                {
                    value = 0;
                    return false;
                }
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int v = 0;
            while (c > 32)
            {
                v = v * 10 + (c - '0');
                c = Read();
            }

            value = v * sign;
            return true;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        if (!fs.TryReadInt(out int n)) return;
        if (!fs.TryReadInt(out int stockLength)) return;

        if (n <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            if (!fs.TryReadInt(out int len)) len = 0;

            if (len < 0) len = 0;
            if (len > stockLength)
            {
                // Per constraints this should not happen, but clamp defensively.
                len = stockLength;
            }

            cuts[i] = new Cut { Length = len, Index = i };
        }

        Array.Sort(cuts, (a, b) => b.Length.CompareTo(a.Length));

        var remainToBins = new SortedDictionary<int, Stack<int>>();
        var bins = new List<List<int>>(Math.Min(n, 1024));

        for (int i = 0; i < n; i++)
        {
            int len = cuts[i].Length;
            int need = len;

            int chosenRemain = -1;
            foreach (var kv in remainToBins)
            {
                if (kv.Key >= need)
                {
                    chosenRemain = kv.Key;
                    break;
                }
            }

            int binId;
            if (chosenRemain == -1)
            {
                binId = bins.Count;
                var list = new List<int>(4);
                list.Add(cuts[i].Index);
                bins.Add(list);

                int newRemain = stockLength - len;
                if (!remainToBins.TryGetValue(newRemain, out var st))
                {
                    st = new Stack<int>();
                    remainToBins[newRemain] = st;
                }
                st.Push(binId);
            }
            else
            {
                var stack = remainToBins[chosenRemain];
                binId = stack.Pop();
                if (stack.Count == 0) remainToBins.Remove(chosenRemain);

                bins[binId].Add(cuts[i].Index);

                int newRemain = chosenRemain - len;
                if (!remainToBins.TryGetValue(newRemain, out var st2))
                {
                    st2 = new Stack<int>();
                    remainToBins[newRemain] = st2;
                }
                st2.Push(binId);
            }
        }

        var sb = new StringBuilder(Math.Max(32, n * 4));
        sb.Append(bins.Count).Append('\n');

        for (int i = 0; i < bins.Count; i++)
        {
            var list = bins[i];
            if (list.Count == 0)
            {
                sb.Append('\n');
                continue;
            }

            sb.Append(list[0]);
            for (int j = 1; j < list.Count; j++)
            {
                sb.Append(' ').Append(list[j]);
            }
            sb.Append('\n');
        }

        Console.Write(sb.ToString());
    }
}
