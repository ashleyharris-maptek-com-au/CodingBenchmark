use std::cmp::Ordering;
use std::f64::consts::PI;
use std::io::{self, Read};

const EPS: f64 = 1e-9;

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    #[inline]
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }

    #[inline]
    fn add(self, o: Pt) -> Pt {
        Pt::new(self.x + o.x, self.y + o.y)
    }

    #[inline]
    fn sub(self, o: Pt) -> Pt {
        Pt::new(self.x - o.x, self.y - o.y)
    }

    #[inline]
    fn mul(self, k: f64) -> Pt {
        Pt::new(self.x * k, self.y * k)
    }

    #[inline]
    fn dot(self, o: Pt) -> f64 {
        self.x * o.x + self.y * o.y
    }

    #[inline]
    fn cross(self, o: Pt) -> f64 {
        self.x * o.y - self.y * o.x
    }
}

#[derive(Clone)]
struct Aabb {
    minx: f64,
    miny: f64,
    maxx: f64,
    maxy: f64,
}

impl Aabb {
    fn from_poly(poly: &[Pt]) -> Self {
        let mut minx = f64::INFINITY;
        let mut miny = f64::INFINITY;
        let mut maxx = f64::NEG_INFINITY;
        let mut maxy = f64::NEG_INFINITY;
        for &p in poly {
            minx = minx.min(p.x);
            miny = miny.min(p.y);
            maxx = maxx.max(p.x);
            maxy = maxy.max(p.y);
        }
        Self { minx, miny, maxx, maxy }
    }

    #[inline]
    fn intersects(&self, o: &Aabb) -> bool {
        !(self.maxx < o.minx + EPS || o.maxx < self.minx + EPS || self.maxy < o.miny + EPS || o.maxy < self.miny + EPS)
    }
}

#[derive(Clone)]
struct Piece {
    idx: usize,
    raw: Vec<Pt>,
    area: f64,
}

#[derive(Clone)]
struct RotShape {
    angle: f64,
    verts: Vec<Pt>,
    minx: f64,
    miny: f64,
    width: f64,
    height: f64,
}

#[derive(Clone)]
struct Placement {
    piece_idx: usize,
    stock_idx: usize,
    x: f64,
    y: f64,
    rot: f64,
}

#[derive(Clone)]
struct PlacedPoly {
    poly: Vec<Pt>,
    aabb: Aabb,
}

#[derive(Clone)]
struct StockState {
    placed: Vec<PlacedPoly>,
}

fn polygon_area2(poly: &[Pt]) -> f64 {
    let n = poly.len();
    let mut s = 0.0;
    for i in 0..n {
        let j = (i + 1) % n;
        s += poly[i].cross(poly[j]);
    }
    s
}

fn ensure_ccw(poly: &mut Vec<Pt>) {
    if polygon_area2(poly) < 0.0 {
        poly.reverse();
    }
}

fn rotate_poly(poly: &[Pt], ang: f64) -> Vec<Pt> {
    let c = ang.cos();
    let s = ang.sin();
    poly.iter()
        .map(|&p| Pt::new(p.x * c - p.y * s, p.x * s + p.y * c))
        .collect()
}

fn on_segment(a: Pt, b: Pt, p: Pt) -> bool {
    let ab = b.sub(a);
    let ap = p.sub(a);
    if ab.cross(ap).abs() > 1e-8 {
        return false;
    }
    let d = ap.dot(b.sub(p));
    d >= -1e-8
}

fn seg_intersects(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    let ab = b.sub(a);
    let ac = c.sub(a);
    let ad = d.sub(a);
    let cd = d.sub(c);
    let ca = a.sub(c);
    let cb = b.sub(c);

    let o1 = ab.cross(ac);
    let o2 = ab.cross(ad);
    let o3 = cd.cross(ca);
    let o4 = cd.cross(cb);

    if ((o1 > EPS && o2 < -EPS) || (o1 < -EPS && o2 > EPS)) && ((o3 > EPS && o4 < -EPS) || (o3 < -EPS && o4 > EPS)) {
        return true;
    }

    if o1.abs() <= EPS && on_segment(a, b, c) {
        return true;
    }
    if o2.abs() <= EPS && on_segment(a, b, d) {
        return true;
    }
    if o3.abs() <= EPS && on_segment(c, d, a) {
        return true;
    }
    if o4.abs() <= EPS && on_segment(c, d, b) {
        return true;
    }
    false
}

fn point_in_poly(poly: &[Pt], p: Pt) -> bool {
    let n = poly.len();
    let mut wn = 0i32;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if on_segment(a, b, p) {
            return true;
        }
        if a.y <= p.y {
            if b.y > p.y && (b.sub(a)).cross(p.sub(a)) > EPS {
                wn += 1;
            }
        } else if b.y <= p.y && (b.sub(a)).cross(p.sub(a)) < -EPS {
            wn -= 1;
        }
    }
    wn != 0
}

fn poly_inside_stock(stock: &[Pt], poly: &[Pt]) -> bool {
    poly.iter().all(|&p| point_in_poly(stock, p))
}

fn polys_overlap_strict(a: &[Pt], b: &[Pt]) -> bool {
    let na = a.len();
    let nb = b.len();
    for i in 0..na {
        let a1 = a[i];
        let a2 = a[(i + 1) % na];
        for j in 0..nb {
            let b1 = b[j];
            let b2 = b[(j + 1) % nb];
            if seg_intersects(a1, a2, b1, b2) {
                let v1 = (a1.sub(b1)).cross(b2.sub(b1));
                let v2 = (a2.sub(b1)).cross(b2.sub(b1));
                let v3 = (b1.sub(a1)).cross(a2.sub(a1));
                let v4 = (b2.sub(a1)).cross(a2.sub(a1));
                if v1.abs() <= EPS || v2.abs() <= EPS || v3.abs() <= EPS || v4.abs() <= EPS {
                    continue;
                }
                return true;
            }
        }
    }

    let ca = a[0];
    if point_in_poly(b, ca) {
        return true;
    }
    let cb = b[0];
    if point_in_poly(a, cb) {
        return true;
    }
    false
}

fn make_rotations(piece: &[Pt], count: usize) -> Vec<RotShape> {
    let mut rots = Vec::with_capacity(count);
    for k in 0..count {
        let ang = 2.0 * PI * (k as f64) / (count as f64);
        let r = rotate_poly(piece, ang);
        let aabb = Aabb::from_poly(&r);
        rots.push(RotShape {
            angle: ang,
            verts: r,
            minx: aabb.minx,
            miny: aabb.miny,
            width: aabb.maxx - aabb.minx,
            height: aabb.maxy - aabb.miny,
        });
    }
    rots
}

fn generate_candidate_points(stock: &[Pt], state: &StockState) -> Vec<Pt> {
    let mut pts = Vec::with_capacity(1 + stock.len() + state.placed.len() * 4);
    pts.push(Pt::new(0.0, 0.0));
    for &p in stock {
        pts.push(p);
    }
    for pp in &state.placed {
        for &v in &pp.poly {
            pts.push(v);
        }
    }
    pts.sort_by(|a, b| {
        let ox = a.x.partial_cmp(&b.x).unwrap_or(Ordering::Equal);
        if ox != Ordering::Equal {
            ox
        } else {
            a.y.partial_cmp(&b.y).unwrap_or(Ordering::Equal)
        }
    });
    pts.dedup_by(|a, b| (a.x - b.x).abs() < 1e-7 && (a.y - b.y).abs() < 1e-7);
    pts
}

fn try_place_on_stock(stock: &[Pt], state: &StockState, rots: &[RotShape], coarse: bool) -> Option<(f64, f64, f64, Vec<Pt>)> {
    let anchors = generate_candidate_points(stock, state);
    let mut best: Option<(f64, f64, f64, Vec<Pt>, f64)> = None;

    for r in rots {
        let mut local_anchors: Vec<Pt> = if coarse {
            anchors.iter().step_by(2).cloned().collect()
        } else {
            anchors.clone()
        };
        if local_anchors.is_empty() {
            local_anchors = anchors.clone();
        }

        for a in local_anchors {
            let tx = a.x - r.minx;
            let ty = a.y - r.miny;
            let moved: Vec<Pt> = r.verts.iter().map(|&p| p.add(Pt::new(tx, ty))).collect();
            if !poly_inside_stock(stock, &moved) {
                continue;
            }

            let maabb = Aabb::from_poly(&moved);
            let mut ok = true;
            for other in &state.placed {
                if !maabb.intersects(&other.aabb) {
                    continue;
                }
                if polys_overlap_strict(&moved, &other.poly) {
                    ok = false;
                    break;
                }
            }
            if !ok {
                continue;
            }

            let score = maabb.maxy * 1e6 + maabb.maxx;
            match &best {
                None => best = Some((tx, ty, r.angle, moved, score)),
                Some((_, _, _, _, bs)) => {
                    if score < *bs {
                        best = Some((tx, ty, r.angle, moved, score));
                    }
                }
            }
        }
    }

    best.map(|(x, y, ang, poly, _)| (x, y, ang, poly))
}

fn parse_input() -> (Vec<Pt>, Vec<Piece>) {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_whitespace();

    let sv: usize = it.next().unwrap_or("0").parse().unwrap_or(0);
    let np: usize = it.next().unwrap_or("0").parse().unwrap_or(0);

    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock.push(Pt::new(x, y));
    }
    ensure_ccw(&mut stock);

    let mut pieces = Vec::with_capacity(np);
    for idx in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut poly = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            poly.push(Pt::new(x, y));
        }
        ensure_ccw(&mut poly);
        let a = polygon_area2(&poly).abs() * 0.5;
        pieces.push(Piece { idx, raw: poly, area: a });
    }
    (stock, pieces)
}

fn main() {
    let (stock, mut pieces) = parse_input();
    let n = pieces.len();

    pieces.sort_by(|a, b| b.area.partial_cmp(&a.area).unwrap_or(Ordering::Equal));

    let rotation_count = if n <= 40 { 24 } else if n <= 200 { 16 } else if n <= 1000 { 10 } else { 6 };
    let coarse = n > 1200;

    let all_rots: Vec<Vec<RotShape>> = pieces
        .iter()
        .map(|p| make_rotations(&p.raw, rotation_count))
        .collect();

    let mut stocks: Vec<StockState> = Vec::new();
    let mut out: Vec<Placement> = vec![Placement { piece_idx: 0, stock_idx: 0, x: 0.0, y: 0.0, rot: 0.0 }; n];

    for (order_i, p) in pieces.iter().enumerate() {
        let rots = &all_rots[order_i];
        let mut placed = false;

        for si in 0..stocks.len() {
            if let Some((x, y, rot, poly)) = try_place_on_stock(&stock, &stocks[si], rots, coarse) {
                let aabb = Aabb::from_poly(&poly);
                stocks[si].placed.push(PlacedPoly { poly, aabb });
                out[p.idx] = Placement {
                    piece_idx: p.idx,
                    stock_idx: si,
                    x,
                    y,
                    rot,
                };
                placed = true;
                break;
            }
        }

        if !placed {
            let mut new_state = StockState { placed: Vec::new() };
            if let Some((x, y, rot, poly)) = try_place_on_stock(&stock, &new_state, rots, false) {
                let aabb = Aabb::from_poly(&poly);
                new_state.placed.push(PlacedPoly { poly, aabb });
                let si = stocks.len();
                stocks.push(new_state);
                out[p.idx] = Placement {
                    piece_idx: p.idx,
                    stock_idx: si,
                    x,
                    y,
                    rot,
                };
            } else {
                let si = stocks.len();
                stocks.push(new_state);
                out[p.idx] = Placement {
                    piece_idx: p.idx,
                    stock_idx: si,
                    x: 0.0,
                    y: 0.0,
                    rot: 0.0,
                };
            }
        }
    }

    println!("{}", stocks.len());
    for i in 0..n {
        let p = &out[i];
        let _ = p.piece_idx;
        println!("{} {:.9} {:.9} {:.9}", p.stock_idx, p.x, p.y, p.rot);
    }
}
