use std::collections::VecDeque;
use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        return;
    }
    let mut lines = input.lines();

    let first = match lines.next() {
        Some(s) => s,
        None => return,
    };

    let mut dims = first.split_whitespace();
    let h: usize = match dims.next().and_then(|x| x.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let w: usize = match dims.next().and_then(|x| x.parse().ok()) {
        Some(v) => v,
        None => return,
    };

    if h == 0 || w == 0 {
        println!("0");
        return;
    }

    let total = match h.checked_mul(w) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut grid = vec![b'#'; total];
    let mut start: Option<usize> = None;
    let mut goal: Option<usize> = None;

    for r in 0..h {
        let line = lines.next().unwrap_or("");
        let bytes = line.as_bytes();
        for c in 0..w {
            let ch = if c < bytes.len() { bytes[c] } else { b'#' };
            let idx = r * w + c;
            grid[idx] = ch;
            if ch == b'A' {
                start = Some(idx);
            } else if ch == b'B' {
                goal = Some(idx);
            }
        }
    }

    let s = match start {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let g = match goal {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    if s == g {
        let x = s % w;
        let y = s / w;
        println!("1");
        println!("{} {}", x, y);
        return;
    }

    let mut visited = vec![false; total];
    let mut parent = vec![usize::MAX; total];
    let mut q = VecDeque::new();

    visited[s] = true;
    q.push_back(s);

    while let Some(cur) = q.pop_front() {
        if cur == g {
            break;
        }

        let r = cur / w;
        let c = cur % w;

        if r > 0 {
            let ni = cur - w;
            if !visited[ni] && grid[ni] != b'#' {
                visited[ni] = true;
                parent[ni] = cur;
                q.push_back(ni);
            }
        }
        if r + 1 < h {
            let ni = cur + w;
            if !visited[ni] && grid[ni] != b'#' {
                visited[ni] = true;
                parent[ni] = cur;
                q.push_back(ni);
            }
        }
        if c > 0 {
            let ni = cur - 1;
            if !visited[ni] && grid[ni] != b'#' {
                visited[ni] = true;
                parent[ni] = cur;
                q.push_back(ni);
            }
        }
        if c + 1 < w {
            let ni = cur + 1;
            if !visited[ni] && grid[ni] != b'#' {
                visited[ni] = true;
                parent[ni] = cur;
                q.push_back(ni);
            }
        }
    }

    if !visited[g] {
        println!("0");
        return;
    }

    let mut path = Vec::new();
    let mut cur = g;
    loop {
        path.push(cur);
        if cur == s {
            break;
        }
        cur = parent[cur];
    }
    path.reverse();

    println!("{}", path.len());
    for &idx in &path {
        let x = idx % w;
        let y = idx / w;
        println!("{} {}", x, y);
    }
}
