#include <algorithm>
#include <cmath>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <unordered_set>
#include <utility>
#include <vector>

struct City {
    float x;
    float y;
    int id;
};

static inline float dist2(const City& a, const City& b) {
    const float dx = a.x - b.x;
    const float dy = a.y - b.y;
    return dx * dx + dy * dy;
}

std::vector<int> nearest_neighbor_small(const std::vector<City>& cities) {
    const int n = static_cast<int>(cities.size());
    std::vector<int> tour;
    tour.reserve(n);

    std::vector<unsigned char> used(n, 0);
    int cur = 0;
    used[cur] = 1;
    tour.push_back(cur);

    for (int step = 1; step < n; ++step) {
        float best = std::numeric_limits<float>::infinity();
        int best_j = -1;
        for (int j = 0; j < n; ++j) {
            if (used[j]) continue;
            const float d = dist2(cities[cur], cities[j]);
            if (d < best) {
                best = d;
                best_j = j;
            }
        }
        cur = best_j;
        used[cur] = 1;
        tour.push_back(cur);
    }

    return tour;
}

void two_opt_limited(std::vector<int>& tour, const std::vector<City>& cities, int max_passes = 2) {
    const int n = static_cast<int>(tour.size());
    if (n < 4) return;

    for (int pass = 0; pass < max_passes; ++pass) {
        bool improved = false;
        for (int i = 1; i + 2 < n; ++i) {
            const City& a = cities[tour[i - 1]];
            const City& b = cities[tour[i]];
            const int max_j = std::min(n - 1, i + 256);
            for (int j = i + 1; j < max_j; ++j) {
                const City& c = cities[tour[j]];
                const City& d = cities[tour[j + 1]];
                const float old_cost = dist2(a, b) + dist2(c, d);
                const float new_cost = dist2(a, c) + dist2(b, d);
                if (new_cost + 1e-6f < old_cost) {
                    std::reverse(tour.begin() + i, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }
}

std::vector<int> grid_snake_order(const std::vector<City>& cities, int gx, int gy) {
    const int n = static_cast<int>(cities.size());
    float min_x = cities[0].x, max_x = cities[0].x;
    float min_y = cities[0].y, max_y = cities[0].y;
    for (const auto& c : cities) {
        min_x = std::min(min_x, c.x);
        max_x = std::max(max_x, c.x);
        min_y = std::min(min_y, c.y);
        max_y = std::max(max_y, c.y);
    }

    const float dx = std::max(1e-12f, max_x - min_x);
    const float dy = std::max(1e-12f, max_y - min_y);

    const int cells = gx * gy;
    std::vector<std::vector<int>> buckets(static_cast<size_t>(cells));
    buckets.shrink_to_fit();

    for (int i = 0; i < n; ++i) {
        int cx = static_cast<int>((cities[i].x - min_x) / dx * gx);
        int cy = static_cast<int>((cities[i].y - min_y) / dy * gy);
        if (cx < 0) cx = 0;
        if (cy < 0) cy = 0;
        if (cx >= gx) cx = gx - 1;
        if (cy >= gy) cy = gy - 1;
        buckets[cy * gx + cx].push_back(i);
    }

    std::vector<int> order;
    order.reserve(n);

    for (int y = 0; y < gy; ++y) {
        if ((y & 1) == 0) {
            for (int x = 0; x < gx; ++x) {
                auto& b = buckets[y * gx + x];
                if ((x & 1) == 0) {
                    std::sort(b.begin(), b.end(), [&](int a, int bidx) {
                        return cities[a].x < cities[bidx].x;
                    });
                } else {
                    std::sort(b.begin(), b.end(), [&](int a, int bidx) {
                        return cities[a].x > cities[bidx].x;
                    });
                }
                for (int id : b) order.push_back(id);
            }
        } else {
            for (int x = gx - 1; x >= 0; --x) {
                auto& b = buckets[y * gx + x];
                if ((x & 1) == 0) {
                    std::sort(b.begin(), b.end(), [&](int a, int bidx) {
                        return cities[a].x > cities[bidx].x;
                    });
                } else {
                    std::sort(b.begin(), b.end(), [&](int a, int bidx) {
                        return cities[a].x < cities[bidx].x;
                    });
                }
                for (int id : b) order.push_back(id);
            }
        }
    }

    return order;
}

std::vector<int> space_filling_order(const std::vector<City>& cities) {
    const int n = static_cast<int>(cities.size());

    if (n <= 2000) {
        auto t = nearest_neighbor_small(cities);
        if (n <= 1200) two_opt_limited(t, cities, 2);
        return t;
    }

    if (n <= 200000) {
        int g = static_cast<int>(std::sqrt(static_cast<double>(n) / 8.0));
        g = std::max(8, std::min(g, 1024));
        return grid_snake_order(cities, g, g);
    }

    int gx = static_cast<int>(std::sqrt(static_cast<double>(n) / 16.0));
    gx = std::max(32, std::min(gx, 4096));
    int gy = gx;
    return grid_snake_order(cities, gx, gy);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n;
    if (!(std::cin >> n)) return 0;
    if (n <= 0) return 0;

    std::vector<City> cities(static_cast<size_t>(n));
    for (int i = 0; i < n; ++i) {
        std::cin >> cities[i].x >> cities[i].y;
        cities[i].id = i;
    }

    std::vector<int> order = space_filling_order(cities);

    // Ensure valid permutation and required start at city 0.
    std::vector<unsigned char> seen(static_cast<size_t>(n), 0);
    std::vector<int> cleaned;
    cleaned.reserve(n);

    auto push_if_valid = [&](int v) {
        if (v >= 0 && v < n && !seen[v]) {
            seen[v] = 1;
            cleaned.push_back(v);
        }
    };

    push_if_valid(0);
    for (int v : order) {
        if (v != 0) push_if_valid(v);
    }
    for (int i = 1; i < n; ++i) {
        if (!seen[i]) push_if_valid(i);
    }

    for (int i = 0; i < n; ++i) {
        if (i) std::cout << ' ';
        std::cout << cleaned[i];
    }
    std::cout << '\n';

    return 0;
}
