using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    private readonly struct Cell
    {
        public readonly int X;
        public readonly int Y;
        public Cell(int x, int y) { X = x; Y = y; }
    }

    private sealed class NumberInfo
    {
        public int Required;
        public int Flagged;
        public int UnknownCount;
        public int[] UnknownIds = Array.Empty<int>();
    }

    public static void Main()
    {
        var first = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(first))
        {
            Console.WriteLine(0);
            return;
        }

        var parts = first.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 2)
        {
            Console.WriteLine(0);
            return;
        }

        int h = int.Parse(parts[0]);
        int w = int.Parse(parts[1]);

        var board = new char[h][];
        for (int y = 0; y < h; y++)
        {
            string line = Console.ReadLine() ?? string.Empty;
            if (line.Length < w)
            {
                line = line.PadRight(w, ' ');
            }
            else if (line.Length > w)
            {
                line = line.Substring(0, w);
            }
            board[y] = line.ToCharArray();
        }

        var safe = Solve(board, h, w);

        using var writer = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };
        writer.WriteLine(safe.Count);
        for (int i = 0; i < safe.Count; i++)
        {
            writer.Write(safe[i].X);
            writer.Write(' ');
            writer.WriteLine(safe[i].Y);
        }
        writer.Flush();
    }

    private static List<Cell> Solve(char[][] board, int h, int w)
    {
        int unknownCount = 0;
        int numberCount = 0;

        for (int y = 0; y < h; y++)
        {
            var row = board[y];
            for (int x = 0; x < w; x++)
            {
                char c = row[x];
                if (c == ' ') unknownCount++;
                else if (c >= '0' && c <= '8') numberCount++;
            }
        }

        if (unknownCount == 0 || numberCount == 0)
            return new List<Cell>(0);

        double density = (double)unknownCount / (h * (double)w);

        // Adaptive strategy:
        // 1) Fast local propagation for all boards.
        // 2) For manageable frontiers, add exact CSP per connected component.
        // 3) For huge/complex frontiers, skip expensive enumeration.
        bool allowExact = (unknownCount <= 300000 && numberCount <= 400000 && density <= 0.85);

        return SolveAdaptive(board, h, w, allowExact);
    }

    private static List<Cell> SolveAdaptive(char[][] board, int h, int w, bool allowExact)
    {
        int n = h * w;

        var isUnknown = new bool[n];
        var isFlagged = new bool[n];
        var isNumber = new bool[n];
        var numberValue = new int[n];

        for (int y = 0; y < h; y++)
        {
            for (int x = 0; x < w; x++)
            {
                int id = y * w + x;
                char c = board[y][x];
                if (c == ' ')
                {
                    isUnknown[id] = true;
                }
                else if (c == '*')
                {
                    isFlagged[id] = true;
                }
                else if (c >= '0' && c <= '8')
                {
                    isNumber[id] = true;
                    numberValue[id] = c - '0';
                }
            }
        }

        var infos = new Dictionary<int, NumberInfo>(capacity: 1 << 12);
        var unknownToNumbers = new Dictionary<int, List<int>>(capacity: 1 << 12);

        BuildConstraints(board, h, w, isUnknown, isFlagged, isNumber, numberValue, infos, unknownToNumbers);

        var provenSafe = new HashSet<int>();
        var provenMine = new HashSet<int>();

        LocalPropagation(w, infos, unknownToNumbers, provenSafe, provenMine);

        if (allowExact)
        {
            ExactByComponents(w, infos, unknownToNumbers, provenSafe, provenMine);
        }

        var result = new List<Cell>(provenSafe.Count);
        foreach (int id in provenSafe)
        {
            if (!isUnknown[id]) continue;
            int y = id / w;
            int x = id - y * w;
            result.Add(new Cell(x, y));
        }

        result.Sort((a, b) =>
        {
            int cmp = a.Y.CompareTo(b.Y);
            return cmp != 0 ? cmp : a.X.CompareTo(b.X);
        });

        return result;
    }

    private static void BuildConstraints(
        char[][] board,
        int h,
        int w,
        bool[] isUnknown,
        bool[] isFlagged,
        bool[] isNumber,
        int[] numberValue,
        Dictionary<int, NumberInfo> infos,
        Dictionary<int, List<int>> unknownToNumbers)
    {
        for (int y = 0; y < h; y++)
        {
            for (int x = 0; x < w; x++)
            {
                int nid = y * w + x;
                if (!isNumber[nid]) continue;

                int required = numberValue[nid];
                int flagged = 0;

                var unknowns = new List<int>(8);

                for (int dy = -1; dy <= 1; dy++)
                {
                    int ny = y + dy;
                    if ((uint)ny >= (uint)h) continue;

                    for (int dx = -1; dx <= 1; dx++)
                    {
                        if (dx == 0 && dy == 0) continue;
                        int nx = x + dx;
                        if ((uint)nx >= (uint)w) continue;

                        int uid = ny * w + nx;
                        if (isFlagged[uid]) flagged++;
                        else if (isUnknown[uid]) unknowns.Add(uid);
                    }
                }

                int rem = required - flagged;
                if (rem < 0) rem = 0;

                if (unknowns.Count == 0) continue;

                var info = new NumberInfo
                {
                    Required = rem,
                    Flagged = 0,
                    UnknownCount = unknowns.Count,
                    UnknownIds = unknowns.ToArray()
                };

                infos[nid] = info;

                for (int i = 0; i < info.UnknownIds.Length; i++)
                {
                    int uid = info.UnknownIds[i];
                    if (!unknownToNumbers.TryGetValue(uid, out var lst))
                    {
                        lst = new List<int>(4);
                        unknownToNumbers[uid] = lst;
                    }
                    lst.Add(nid);
                }
            }
        }
    }

    private static void LocalPropagation(
        int w,
        Dictionary<int, NumberInfo> infos,
        Dictionary<int, List<int>> unknownToNumbers,
        HashSet<int> provenSafe,
        HashSet<int> provenMine)
    {
        var queue = new Queue<int>(infos.Keys);
        var inQueue = new HashSet<int>(infos.Keys);

        while (queue.Count > 0)
        {
            int nid = queue.Dequeue();
            inQueue.Remove(nid);

            if (!infos.TryGetValue(nid, out var info)) continue;

            int need = info.Required - info.Flagged;
            int unk = info.UnknownCount;

            if (unk <= 0) continue;
            if (need < 0) need = 0;
            if (need > unk) continue;

            bool markAllSafe = (need == 0);
            bool markAllMine = (need == unk);

            if (!markAllSafe && !markAllMine) continue;

            for (int i = 0; i < info.UnknownIds.Length; i++)
            {
                int uid = info.UnknownIds[i];

                if (markAllSafe)
                {
                    if (provenMine.Contains(uid) || provenSafe.Contains(uid)) continue;
                    provenSafe.Add(uid);
                    UpdateNeighbors(uid, false, infos, unknownToNumbers, queue, inQueue);
                }
                else
                {
                    if (provenSafe.Contains(uid) || provenMine.Contains(uid)) continue;
                    provenMine.Add(uid);
                    UpdateNeighbors(uid, true, infos, unknownToNumbers, queue, inQueue);
                }
            }
        }
    }

    private static void UpdateNeighbors(
        int uid,
        bool asMine,
        Dictionary<int, NumberInfo> infos,
        Dictionary<int, List<int>> unknownToNumbers,
        Queue<int> queue,
        HashSet<int> inQueue)
    {
        if (!unknownToNumbers.TryGetValue(uid, out var adjNums)) return;

        for (int i = 0; i < adjNums.Count; i++)
        {
            int nid = adjNums[i];
            if (!infos.TryGetValue(nid, out var info)) continue;

            if (info.UnknownCount > 0) info.UnknownCount--;
            if (asMine) info.Flagged++;

            if (inQueue.Add(nid)) queue.Enqueue(nid);
        }
    }

    private static void ExactByComponents(
        int w,
        Dictionary<int, NumberInfo> infos,
        Dictionary<int, List<int>> unknownToNumbers,
        HashSet<int> provenSafe,
        HashSet<int> provenMine)
    {
        var frontierUnknowns = new List<int>();
        foreach (var kv in unknownToNumbers)
        {
            int uid = kv.Key;
            if (provenSafe.Contains(uid) || provenMine.Contains(uid)) continue;
            if (kv.Value.Count > 0) frontierUnknowns.Add(uid);
        }

        var visitedU = new HashSet<int>();
        var visitedN = new HashSet<int>();

        var uQueue = new Queue<int>();
        var nQueue = new Queue<int>();

        foreach (int startU in frontierUnknowns)
        {
            if (!visitedU.Add(startU)) continue;

            var compU = new List<int>();
            var compN = new List<int>();

            uQueue.Enqueue(startU);
            compU.Add(startU);

            while (uQueue.Count > 0 || nQueue.Count > 0)
            {
                while (uQueue.Count > 0)
                {
                    int u = uQueue.Dequeue();
                    if (!unknownToNumbers.TryGetValue(u, out var nums)) continue;

                    for (int i = 0; i < nums.Count; i++)
                    {
                        int n = nums[i];
                        if (visitedN.Add(n))
                        {
                            compN.Add(n);
                            nQueue.Enqueue(n);
                        }
                    }
                }

                while (nQueue.Count > 0)
                {
                    int n = nQueue.Dequeue();
                    if (!infos.TryGetValue(n, out var info)) continue;

                    for (int i = 0; i < info.UnknownIds.Length; i++)
                    {
                        int u = info.UnknownIds[i];
                        if (provenSafe.Contains(u) || provenMine.Contains(u)) continue;
                        if (visitedU.Add(u))
                        {
                            compU.Add(u);
                            uQueue.Enqueue(u);
                        }
                    }
                }
            }

            if (compU.Count == 0 || compN.Count == 0) continue;

            // Keep exact solving bounded for extreme boards.
            if (compU.Count > 22 || compN.Count > 120) continue;

            SolveComponentExactly(compU, compN, infos, unknownToNumbers, provenSafe, provenMine);
        }
    }

    private static void SolveComponentExactly(
        List<int> compU,
        List<int> compN,
        Dictionary<int, NumberInfo> infos,
        Dictionary<int, List<int>> unknownToNumbers,
        HashSet<int> provenSafe,
        HashSet<int> provenMine)
    {
        int uCount = compU.Count;
        int nCount = compN.Count;

        var uIndex = new Dictionary<int, int>(uCount);
        for (int i = 0; i < uCount; i++) uIndex[compU[i]] = i;

        var constraints = new int[nCount][];
        var rhs = new int[nCount];

        for (int i = 0; i < nCount; i++)
        {
            int nid = compN[i];
            var info = infos[nid];

            rhs[i] = info.Required - info.Flagged;
            if (rhs[i] < 0) rhs[i] = 0;

            var list = new List<int>(info.UnknownIds.Length);
            for (int k = 0; k < info.UnknownIds.Length; k++)
            {
                int uid = info.UnknownIds[k];
                if (provenSafe.Contains(uid) || provenMine.Contains(uid)) continue;
                if (uIndex.TryGetValue(uid, out int idx)) list.Add(idx);
            }
            constraints[i] = list.ToArray();
        }

        var varToConstraints = new List<int>[uCount];
        for (int i = 0; i < uCount; i++) varToConstraints[i] = new List<int>(4);

        var remUnk = new int[nCount];
        var remNeed = new int[nCount];

        for (int ci = 0; ci < nCount; ci++)
        {
            remUnk[ci] = constraints[ci].Length;
            remNeed[ci] = rhs[ci];

            for (int j = 0; j < constraints[ci].Length; j++)
            {
                int vi = constraints[ci][j];
                varToConstraints[vi].Add(ci);
            }
        }

        var assigned = new sbyte[uCount];
        Array.Fill(assigned, (sbyte)-1);

        long totalSolutions = 0;
        var mineCountPerVar = new long[uCount];

        int[] order = BuildVariableOrder(uCount, varToConstraints);

        void Dfs(int depth)
        {
            if (depth == uCount)
            {
                for (int ci = 0; ci < nCount; ci++)
                {
                    if (remNeed[ci] != 0) return;
                }

                totalSolutions++;
                for (int i = 0; i < uCount; i++)
                {
                    if (assigned[i] == 1) mineCountPerVar[i]++;
                }
                return;
            }

            int varIdx = order[depth];
            if (assigned[varIdx] != -1)
            {
                Dfs(depth + 1);
                return;
            }

            // Try safe first to discover safe proofs quickly.
            for (int value = 0; value <= 1; value++)
            {
                if (!CanAssign(varIdx, value, varToConstraints, remUnk, remNeed)) continue;

                ApplyAssign(varIdx, value, varToConstraints, remUnk, remNeed);
                assigned[varIdx] = (sbyte)value;

                Dfs(depth + 1);

                assigned[varIdx] = -1;
                RevertAssign(varIdx, value, varToConstraints, remUnk, remNeed);
            }
        }

        Dfs(0);

        if (totalSolutions == 0) return;

        for (int i = 0; i < uCount; i++)
        {
            int uid = compU[i];
            long mines = mineCountPerVar[i];
            if (mines == 0)
            {
                provenSafe.Add(uid);
            }
            else if (mines == totalSolutions)
            {
                provenMine.Add(uid);
            }
        }
    }

    private static int[] BuildVariableOrder(int uCount, List<int>[] varToConstraints)
    {
        var vars = new int[uCount];
        for (int i = 0; i < uCount; i++) vars[i] = i;

        Array.Sort(vars, (a, b) =>
        {
            int da = varToConstraints[a].Count;
            int db = varToConstraints[b].Count;
            return db.CompareTo(da);
        });

        return vars;
    }

    private static bool CanAssign(int varIdx, int value, List<int>[] varToConstraints, int[] remUnk, int[] remNeed)
    {
        var refs = varToConstraints[varIdx];
        for (int i = 0; i < refs.Count; i++)
        {
            int ci = refs[i];
            int ru = remUnk[ci] - 1;
            int rn = remNeed[ci] - value;

            if (rn < 0) return false;
            if (rn > ru) return false;
        }
        return true;
    }

    private static void ApplyAssign(int varIdx, int value, List<int>[] varToConstraints, int[] remUnk, int[] remNeed)
    {
        var refs = varToConstraints[varIdx];
        for (int i = 0; i < refs.Count; i++)
        {
            int ci = refs[i];
            remUnk[ci]--;
            remNeed[ci] -= value;
        }
    }

    private static void RevertAssign(int varIdx, int value, List<int>[] varToConstraints, int[] remUnk, int[] remNeed)
    {
        var refs = varToConstraints[varIdx];
        for (int i = 0; i < refs.Count; i++)
        {
            int ci = refs[i];
            remUnk[ci]++;
            remNeed[ci] += value;
        }
    }
}
