#include <array>

#include <unordered_set>
#include <utility>
#include <vector>

using namespace std;

struct RH {
    static constexpr uint64_t BASE = 1315423911ULL;
    static constexpr uint64_t MIX  = 11400714819323198485ULL;
    vector<uint64_t> pref;
    vector<uint64_t> pw;

    explicit RH(const string& s) {
        const size_t n = s.size();
        pref.resize(n + 1, 0);
        pw.resize(n + 1, 1);
        for (size_t i = 0; i < n; ++i) {
            pref[i + 1] = pref[i] * BASE + static_cast<unsigned char>(s[i]) + 1U;
            pw[i + 1] = pw[i] * BASE;
        }
    }

    uint64_t get(size_t l, size_t r) const {
        return pref[r] - pref[l] * pw[r - l];
    }
};

static inline uint64_t fmix(uint64_t x) {
    x ^= x >> 33;
    x *= 0xff51afd7ed558ccdULL;
    x ^= x >> 33;
    x *= 0xc4ceb9fe1a85ec53ULL;
    x ^= x >> 33;
    return x;
}

class Solver {
public:
    explicit Solver(vector<string> in) : s(move(in)) {
        sort(s.begin(), s.end(), [](const string& a, const string& b) {
            return a.size() < b.size();
        });
        n = static_cast<int>(s.size());
        if (n > 0) {
            minLen = s[0].size();
        }
        rh.reserve(n);
        for (const auto& str : s) rh.emplace_back(str);
    }

    string run() {
        if (n == 0) return "";
        if (minLen == 0) return "";

        if (isTinyCase()) {
            return bruteForceSmall();
        }

        size_t lo = 0;
        size_t hi = minLen;
        size_t bestLen = 0;
        size_t bestPos = 0;

        while (lo <= hi) {
            size_t mid = lo + (hi - lo) / 2;
            size_t pos = 0;
            if (existsCommon(mid, pos)) {
                bestLen = mid;
                bestPos = pos;
                lo = mid + 1;
            } else {
                if (mid == 0) break;
                hi = mid - 1;
            }
        }

        if (bestLen == 0) return "";
        return s[0].substr(bestPos, bestLen);
    }

private:
    vector<string> s;
    vector<RH> rh;
    int n = 0;
    size_t minLen = 0;

    bool isTinyCase() const {
        if (n > 3) return false;
        size_t total = 0;
        for (const auto& x : s) total += x.size();
        return total <= 300;
    }

    string bruteForceSmall() const {
        const string& base = s[0];
        for (size_t len = base.size(); len >= 1; --len) {
            for (size_t i = 0; i + len <= base.size(); ++i) {
                string_view cand(base.data() + i, len);
                bool ok = true;
                for (int k = 1; k < n && ok; ++k) {
                    if (s[k].find(cand) == string::npos) ok = false;
                }
                if (ok) return string(cand);
            }
            if (len == 1) break;
        }
        return "";
    }

    bool existsCommon(size_t L, size_t& outPos) {
        if (L == 0) {
            outPos = 0;
            return true;
        }
        if (L > minLen) return false;

        // Adaptive strategy:
        // - Small/moderate windows: full set intersection for deterministic guarantee.
        // - Huge windows count: same deterministic set flow, but reserve aggressively.
        unordered_map<uint64_t, vector<size_t>> posInBase;
        const size_t cnt0 = s[0].size() - L + 1;
        posInBase.reserve(cnt0 * 2 + 1);

        for (size_t i = 0; i + L <= s[0].size(); ++i) {
            uint64_t h = fmix(rh[0].get(i, i + L));
            posInBase[h].push_back(i);
        }

        unordered_set<uint64_t> common;
        common.reserve(posInBase.size() * 2 + 1);
        for (const auto& kv : posInBase) common.insert(kv.first);

        for (int idx = 1; idx < n; ++idx) {
            unordered_set<uint64_t> cur;
            const size_t c = s[idx].size() >= L ? (s[idx].size() - L + 1) : 0;
            cur.reserve(c * 2 + 1);

            for (size_t i = 0; i + L <= s[idx].size(); ++i) {
                uint64_t h = fmix(rh[idx].get(i, i + L));
                if (common.find(h) != common.end()) cur.insert(h);
            }

            if (cur.empty()) return false;
            common.swap(cur);
        }

        for (const auto& h : common) {
            const auto it = posInBase.find(h);
            if (it == posInBase.end()) continue;
            for (size_t p : it->second) {
                string_view cand(s[0].data() + p, L);
                bool ok = true;
                for (int idx = 1; idx < n && ok; ++idx) {
                    if (s[idx].find(cand) == string::npos) ok = false;
                }
                if (ok) {
                    outPos = p;
                    return true;
                }
            }
        }
        return false;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) {
        cout << '\n';
        return 0;
    }
    string dummy;
    getline(cin, dummy);

    vector<string> arr;
    arr.reserve(max(0, N));
    for (int i = 0; i < N; ++i) {
        string line;
        getline(cin, line);
        arr.push_back(move(line));
    }

    Solver solver(move(arr));
    cout << solver.run() << '\n';
    return 0;
}
