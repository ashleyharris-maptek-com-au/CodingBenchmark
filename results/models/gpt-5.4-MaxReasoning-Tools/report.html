static inline double sq_distance(const Point& a, const Point& b) {
    const double dx = static_cast<double>(a.x) - static_cast<double>(b.x);
    const double dy = static_cast<double>(a.y) - static_cast<double>(b.y);
    return dx * dx + dy * dy;
}

static inline double distance_euclid(const Point& a, const Point& b) {
    return std::sqrt(sq_distance(a, b));
}

static inline std::uint64_t spread_bits(std::uint32_t value) {
    std::uint64_t x = value;
    x = (x | (x << 16)) & 0x0000FFFF0000FFFFULL;
    x = (x | (x << 8)) & 0x00FF00FF00FF00FFULL;
    x = (x | (x << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    x = (x | (x << 2)) & 0x3333333333333333ULL;
    x = (x | (x << 1)) & 0x5555555555555555ULL;
    return x;
}

static inline std::uint64_t morton_code(std::uint32_t x, std::uint32_t y) {
    return spread_bits(x) | (spread_bits(y) << 1);

class SuccessorDSU {
public:
    explicit SuccessorDSU(int n) : parent_(static_cast<std::size_t>(n) + 1u) {
        std::iota(parent_.begin(), parent_.end(), 0);
    }

    int next(int x) {
        if (x < 0) {
            x = 0;
        }
        const int n = static_cast<int>(parent_.size()) - 1;
        if (x > n) {
            return n;
        }
        return find(x);
    }

    void erase(int x) {
        const int n = static_cast<int>(parent_.size()) - 1;
        if (x < 0 || x >= n) {
            return;
        }
        parent_[static_cast<std::size_t>(x)] = find(x + 1);
    }

private:
    std::vector<int> parent_;

    int find(int x) {
        int root = x;
        while (parent_[static_cast<std::size_t>(root)] != root) {
            root = parent_[static_cast<std::size_t>(root)];
        }
        while (parent_[static_cast<std::size_t>(x)] != x) {
            const int next = parent_[static_cast<std::size_t>(x)];
            parent_[static_cast<std::size_t>(x)] = root;
            x = next;
        }
        return root;
    }
};

void radix_sort_keys(std::vector<KeyId>& data) {
    const std::size_t n = data.size();
    if (n <= 1) {
        return;
    }

    std::vector<KeyId> temp(n);
    std::vector<std::size_t> count(1u << 16, 0);

    for (int shift = 0; shift < 64; shift += 16) {
        std::fill(count.begin(), count.end(), 0);
        for (const KeyId& item : data) {
            ++count[(item.key >> shift) & 0xFFFFu];
        }

        std::size_t sum = 0;
        for (std::size_t i = 0; i < count.size(); ++i) {
            const std::size_t c = count[i];
            count[i] = sum;
            sum += c;
        }

        for (const KeyId& item : data) {
            temp[count[(item.key >> shift) & 0xFFFFu]++] = item;
        }
        data.swap(temp);
    }
}

std::vector<std::uint32_t> greedy_route_from_sorted(const std::vector<Point>& points, const std::vector<KeyId>& keys, int window) {

                return;

void improve_two_opt(std::vector<std::uint32_t>& route, const std::vector<Point>& points, int window, int max_passes) {

        bool improved = false;

            const std::size_t limit = std::min(n - 1, i + static_cast<std::size_t>(window));
            for (std::size_t j = i + 1; j <= limit; ++j) {
                const std::size_t next_j = (j + 1 == n) ? 0 : (j + 1);

                const std::uint32_t b = route[i];
                const std::uint32_t c = route[j];

                const double after = distance_euclid(points[a], points[c]) + distance_euclid(points[b], points[d]);

            break;
        }
    }
}

    x: f64,
    y: f64,
}

struct FastInput {
    data: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data).unwrap();
        Self { data, idx: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.data.len() && self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.data.len() {
            return None;
        }
        let mut value = 0usize;
        while self.idx < self.data.len() && !self.data[self.idx].is_ascii_whitespace() {
            value = value * 10 + (self.data[self.idx] - b'0') as usize;
            self.idx += 1;
        }
        Some(value)
    }
}

struct Graph {
    n: usize,
    offsets: Vec<usize>,
    adj: Vec<usize>,
    degree: Vec<usize>,
}

impl Graph {
    fn neighbors(&self, u: usize) -> &[usize] {
        &self.adj[self.offsets[u]..self.offsets[u + 1]]
    }
}

struct ComponentInfo {
    nodes: Vec<usize>,
    edge_count: usize,
    max_degree_node: usize,
}

struct PivotData {
    center_local: usize,
    center_dist: Vec<i32>,
    pivots: Vec<usize>,
    dists: Vec<Vec<i32>>,
    diameter_est: i32,
}

struct ComponentLayout {
    nodes: Vec<usize>,
    coords: Vec<Point>,
    minx: f64,
    maxx: f64,
    miny: f64,
    maxy: f64,
}

fn main() {
    let mut input = FastInput::new();
    let n = match input.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m = input.next_usize().unwrap_or(0);

    let mut degree = vec![0usize; n];
    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let u = input.next_usize().unwrap_or(0);
        let v = input.next_usize().unwrap_or(0);
        if u < n && v < n && u != v {
            degree[u] += 1;
            degree[v] += 1;
            edges.push((u, v));
        }
    }

    let mut offsets = vec![0usize; n + 1];
    for i in 0..n {
        offsets[i + 1] = offsets[i] + degree[i];
    }
    let mut cursor = offsets[..n].to_vec();
    let mut adj = vec![0usize; offsets[n]];
    for (u, v) in edges {
        adj[cursor[u]] = v;
        cursor[u] += 1;
        adj[cursor[v]] = u;
        cursor[v] += 1;
    }
    let graph = Graph {
        n,
        offsets,
        adj,
        degree,
    };

    let components = find_components(&graph);
    let mut loc = vec![usize::MAX; n];
    let mut layouts = Vec::with_capacity(components.len());
    for comp in &components {
        for (local, &node) in comp.nodes.iter().enumerate() {
            loc[node] = local;
        }

        let mut coords = layout_component(&graph, comp, &loc);
        normalize_component(&mut coords, (comp.nodes.len() as f64).sqrt() * 2.25 + 1.0);
        let (minx, maxx, miny, maxy) = bbox(&coords);
        layouts.push(ComponentLayout {
            nodes: comp.nodes.clone(),
            coords,
            minx,
            maxx,
            miny,
            maxy,
        });

        for &node in &comp.nodes {
            loc[node] = usize::MAX;
        }
    }

    let positions = pack_components(graph.n, layouts);
    let mut out = String::with_capacity(n.saturating_mul(32));
    for p in positions {
        let _ = writeln!(out, "{:.6} {:.6}", p.x, p.y);
    }
    print!("{out}");
}

fn find_components(graph: &Graph) -> Vec<ComponentInfo> {
    let mut visited = vec![false; graph.n];
    let mut comps = Vec::new();

    for start in 0..graph.n {
        if visited[start] {
            continue;
        }
        visited[start] = true;
        if graph.degree[start] == 0 {
            comps.push(ComponentInfo {
                nodes: vec![start],
                edge_count: 0,
                max_degree_node: start,
            });
            continue;
        }

        let mut queue = vec![start];
        let mut head = 0usize;
        let mut nodes = Vec::new();
        let mut degree_sum = 0usize;
        let mut max_degree_node = start;

        while head < queue.len() {
            let u = queue[head];
            head += 1;
            nodes.push(u);
            degree_sum += graph.degree[u];
            if graph.degree[u] > graph.degree[max_degree_node] {
                max_degree_node = u;
            }
            for &v in graph.neighbors(u) {
                if !visited[v] {
                    visited[v] = true;
                    queue.push(v);
                }
            }
        }

        comps.push(ComponentInfo {
            nodes,
            edge_count: degree_sum / 2,
            max_degree_node,
        });
    }

    comps
}

fn layout_component(graph: &Graph, comp: &ComponentInfo, loc: &[usize]) -> Vec<Point> {
    let n = comp.nodes.len();
    if n == 1 {
        return vec![Point { x: 0.0, y: 0.0 }];
    }
    if comp.edge_count == n - 1 {
        return tree_layout(graph, &comp.nodes, loc);
    }

    let pivots = choose_pivots(graph, comp, loc);
    let very_dense = comp.edge_count > 4 * n;
    let use_shell = (pivots.diameter_est <= 4 && n >= 24) || (very_dense && pivots.diameter_est <= 6);

    let mut coords = if use_shell {
        shell_layout(
            graph,
            &comp.nodes,
            pivots.center_local,
            &pivots.center_dist,
            pivots.dists.first(),
            pivots.dists.get(1),
        )
    } else {
        landmark_layout(&pivots)
    };

    ensure_non_degenerate(&mut coords, &comp.nodes);

    if n <= 2000 {
        let edges = collect_local_edges(graph, &comp.nodes, loc);
        let iterations = choose_iterations(n, comp.edge_count);
        refine_force(&mut coords, &edges, iterations);
    }

    coords
}

fn choose_pivots(graph: &Graph, comp: &ComponentInfo, loc: &[usize]) -> PivotData {
    let n = comp.nodes.len();
    let center_local = loc[comp.max_degree_node];
    let center_dist = bfs_local(graph, &comp.nodes, loc, center_local);
    let first = farthest_local(graph, &comp.nodes, &center_dist);

    let mut pivots = vec![first];
    let mut dists = vec![bfs_local(graph, &comp.nodes, loc, first)];
    let target = if n <= 8 {
        n
    } else if n <= 2000 {
        6
    } else if n <= 20000 {
        5
    } else {
        4
    };

    while pivots.len() < target {
        let mut best = usize::MAX;
        let mut best_score = -1i32;
        let mut best_degree = 0usize;
        for local in 0..n {
            if pivots.contains(&local) {
                continue;
            }
            let mut score = i32::MAX;
            for dist in &dists {
                score = score.min(dist[local]);
            }
            let deg = graph.degree[comp.nodes[local]];
            if score > best_score || (score == best_score && deg > best_degree) {
                best = local;
                best_score = score;
                best_degree = deg;
            }
        }
        if best == usize::MAX {
            break;
        }
        pivots.push(best);
        dists.push(bfs_local(graph, &comp.nodes, loc, best));
    }

    let diameter_est = dists
        .iter()
        .flat_map(|d| d.iter().copied())
        .max()
        .unwrap_or(0);

    PivotData {
        center_local,
        center_dist,
        pivots,
        dists,
        diameter_est,
    }
}

fn bfs_local(graph: &Graph, nodes: &[usize], loc: &[usize], src_local: usize) -> Vec<i32> {
    let n = nodes.len();
    let mut dist = vec![-1i32; n];
    let mut queue = Vec::with_capacity(n);
    dist[src_local] = 0;
    queue.push(src_local);
    let mut head = 0usize;
    while head < queue.len() {
        let u_local = queue[head];
        head += 1;
        let next_dist = dist[u_local] + 1;
        let u = nodes[u_local];
        for &v in graph.neighbors(u) {
            let v_local = loc[v];
            if v_local != usize::MAX && dist[v_local] < 0 {
                dist[v_local] = next_dist;
                queue.push(v_local);
            }
        }
    }
    dist
}

fn bfs_local_with_parent(
    graph: &Graph,
    nodes: &[usize],
    loc: &[usize],
    src_local: usize,
) -> (Vec<i32>, Vec<usize>) {
    let n = nodes.len();
    let mut dist = vec![-1i32; n];
    let mut parent = vec![usize::MAX; n];
    let mut queue = Vec::with_capacity(n);
    dist[src_local] = 0;
    parent[src_local] = src_local;
    queue.push(src_local);
    let mut head = 0usize;
    while head < queue.len() {
        let u_local = queue[head];
        head += 1;
        let next_dist = dist[u_local] + 1;
        let u = nodes[u_local];
        for &v in graph.neighbors(u) {
            let v_local = loc[v];
            if v_local != usize::MAX && dist[v_local] < 0 {
                dist[v_local] = next_dist;
                parent[v_local] = u_local;
                queue.push(v_local);
            }
        }
    }
    (dist, parent)
}

fn farthest_local(graph: &Graph, nodes: &[usize], dist: &[i32]) -> usize {
    let mut best = 0usize;
    for i in 1..dist.len() {
        let better = dist[i] > dist[best]
            || (dist[i] == dist[best] && graph.degree[nodes[i]] > graph.degree[nodes[best]]);
        if better {
            best = i;
        }
    }
    best
}

fn tree_layout(graph: &Graph, nodes: &[usize], loc: &[usize]) -> Vec<Point> {
    let n = nodes.len();
    if n == 2 {
        return vec![Point { x: -0.5, y: 0.0 }, Point { x: 0.5, y: 0.0 }];
    }

    let (d0, _) = bfs_local_with_parent(graph, nodes, loc, 0);
    let a = farthest_local(graph, nodes, &d0);
    let (da, parent_from_a) = bfs_local_with_parent(graph, nodes, loc, a);
    let b = farthest_local(graph, nodes, &da);

    let mut diameter_path = Vec::new();
    let mut cur = b;
    loop {
        diameter_path.push(cur);
        if cur == a {
            break;
        }
        cur = parent_from_a[cur];
    }
    let root = diameter_path[diameter_path.len() / 2];

    let mut parent = vec![usize::MAX; n];
    let mut depth = vec![0usize; n];
    let mut order = Vec::with_capacity(n);
    order.push(root);
    parent[root] = root;
    let mut head = 0usize;
    while head < order.len() {
        let u_local = order[head];
        head += 1;
        let u = nodes[u_local];
        for &v in graph.neighbors(u) {
            let v_local = loc[v];
            if v_local != usize::MAX && v_local != parent[u_local] {
                parent[v_local] = u_local;
                depth[v_local] = depth[u_local] + 1;
                order.push(v_local);
            }
        }
    }

    let mut children = vec![Vec::<usize>::new(); n];
    for v in 0..n {
        if v != root {
            children[parent[v]].push(v);
        }
    }

    let mut leaves = vec![0usize; n];
    for &u in order.iter().rev() {
        if children[u].is_empty() {
            leaves[u] = 1;
        } else {
            leaves[u] = children[u].iter().map(|&c| leaves[c]).sum();
        }
    }

    let mut pos = vec![Point::default(); n];
    let mut stack = vec![(root, 0.0f64, 2.0 * PI)];
    while let Some((u, start, end)) = stack.pop() {
        let angle = 0.5 * (start + end);
        let radius = depth[u] as f64;
        pos[u] = Point {
            x: radius * angle.cos(),
            y: radius * angle.sin(),
        };
        if children[u].is_empty() {
            continue;
        }
        let span = end - start;
        let total = leaves[u] as f64;
        let mut cur_start = start;
        for &child in children[u].iter().rev() {
            let frac = leaves[child] as f64 / total;
            let child_end = cur_start + span * frac;
            stack.push((child, cur_start, child_end));
            cur_start = child_end;
        }
    }

    center_positions(&mut pos);
    pos
}

fn shell_layout(
    graph: &Graph,
    nodes: &[usize],
    center_local: usize,
    center_dist: &[i32],
    d1: Option<&Vec<i32>>,
    d2: Option<&Vec<i32>>,
) -> Vec<Point> {
    let max_layer = center_dist.iter().copied().max().unwrap_or(0).max(0) as usize;
    let mut shells = vec![Vec::<usize>::new(); max_layer + 1];
    for local in 0..nodes.len() {
        shells[center_dist[local].max(0) as usize].push(local);
    }

    let mut pos = vec![Point::default(); nodes.len()];
    pos[center_local] = Point { x: 0.0, y: 0.0 };

    for (layer, shell) in shells.iter_mut().enumerate() {
        shell.sort_by(|&a, &b| {
            let ka = shell_key(graph, nodes, a, d1, d2);
            let kb = shell_key(graph, nodes, b, d1, d2);
            ka.cmp(&kb)
        });
        if layer == 0 {
            if let Some(&u) = shell.first() {
                pos[u] = Point { x: 0.0, y: 0.0 };
            }
            continue;
        }
        let count = shell.len();
        if count == 0 {
            continue;
        }
        let radius = layer as f64 * 1.8 + (count as f64).sqrt() * 0.25;
        let offset = if layer % 2 == 0 { 0.0 } else { PI / count as f64 };
        for (idx, &u) in shell.iter().enumerate() {
            let angle = 2.0 * PI * (idx as f64 / count as f64) + offset;
            pos[u] = Point {
                x: radius * angle.cos(),
                y: radius * angle.sin(),
            };
        }
    }

    pos
}

fn shell_key(
    graph: &Graph,
    nodes: &[usize],
    local: usize,
    d1: Option<&Vec<i32>>,
    d2: Option<&Vec<i32>>,
) -> (i32, i32, usize, usize) {
    let first = d1.map(|d| d[local]).unwrap_or(0);
    let second = d2.map(|d| d[local]).unwrap_or(0);
    let deg = graph.degree[nodes[local]];
    (first - second, second, usize::MAX - deg, nodes[local])
}

fn landmark_layout(pivots: &PivotData) -> Vec<Point> {
    if pivots.pivots.is_empty() {
        return Vec::new();
    }
    if pivots.pivots.len() == 1 {
        return circle_layout(pivots.center_dist.len());
    }

    let pivot_coords = pivot_coordinates(&pivots.pivots, &pivots.dists);
    let mut coords = embed_from_pivots(&pivots.pivots, &pivots.dists, &pivot_coords);
    center_positions(&mut coords);
    coords
}

fn pivot_coordinates(pivots: &[usize], dists: &[Vec<i32>]) -> Vec<Point> {
    let k = pivots.len();
    let mut dist_sq = vec![vec![0.0f64; k]; k];
    for i in 0..k {
        for j in 0..k {
            let d = dists[i][pivots[j]] as f64;
            dist_sq[i][j] = d * d;
        }
    }

    let mut row_mean = vec![0.0f64; k];
    let mut total_mean = 0.0f64;
    for i in 0..k {
        for j in 0..k {
            row_mean[i] += dist_sq[i][j];
            total_mean += dist_sq[i][j];
        }
        row_mean[i] /= k as f64;
    }
    let col_mean = row_mean.clone();
    total_mean /= (k * k) as f64;

    let mut b = vec![vec![0.0f64; k]; k];
    for i in 0..k {
        for j in 0..k {
            b[i][j] = -0.5 * (dist_sq[i][j] - row_mean[i] - col_mean[j] + total_mean);
        }
    }

    let (eigenvalues, eigenvectors) = jacobi_eigen(b);
    let mut coords = vec![Point::default(); k];
    let mut dims = 0usize;

    if !eigenvalues.is_empty() && eigenvalues[0] > 1e-9 {
        let scale = eigenvalues[0].sqrt();
        for i in 0..k {
            coords[i].x = eigenvectors[0][i] * scale;
        }
        dims += 1;
    }
    if eigenvalues.len() > 1 && eigenvalues[1] > 1e-9 {
        let scale = eigenvalues[1].sqrt();
        for i in 0..k {
            coords[i].y = eigenvectors[1][i] * scale;
        }
        dims += 1;
    }

    if dims == 0 {
        return circle_layout(k);
    }
    if dims == 1 {
        for (i, p) in coords.iter_mut().enumerate() {
            p.y = if i % 2 == 0 { -0.05 } else { 0.05 };
        }
    }

    coords
}

fn jacobi_eigen(mut a: Vec<Vec<f64>>) -> (Vec<f64>, Vec<Vec<f64>>) {
    let n = a.len();
    if n == 0 {
        return (Vec::new(), Vec::new());
    }

    let mut v = vec![vec![0.0f64; n]; n];
    for i in 0..n {
        v[i][i] = 1.0;
    }

    let max_iter = 50 * n * n;
    for _ in 0..max_iter {
        let mut p = 0usize;
        let mut q = 1usize.min(n - 1);
        let mut best = 0.0f64;
        for i in 0..n {
            for j in (i + 1)..n {
                let val = a[i][j].abs();
                if val > best {
                    best = val;
                    p = i;
                    q = j;
                }
            }
        }
        if best < 1e-12 {
            break;
        }

        let app = a[p][p];
        let aqq = a[q][q];
        let apq = a[p][q];
        let tau = (aqq - app) / (2.0 * apq);
        let t = if tau >= 0.0 {
            1.0 / (tau + (1.0 + tau * tau).sqrt())
        } else {
            -1.0 / (-tau + (1.0 + tau * tau).sqrt())
        };
        let c = 1.0 / (1.0 + t * t).sqrt();
        let s = t * c;

        for k in 0..n {
            if k != p && k != q {
                let akp = a[k][p];
                let akq = a[k][q];
                a[k][p] = c * akp - s * akq;
                a[p][k] = a[k][p];
                a[k][q] = s * akp + c * akq;
                a[q][k] = a[k][q];
            }
        }

        a[p][p] = c * c * app - 2.0 * s * c * apq + s * s * aqq;
        a[q][q] = s * s * app + 2.0 * s * c * apq + c * c * aqq;
        a[p][q] = 0.0;
        a[q][p] = 0.0;

        for row in &mut v {
            let vip = row[p];
            let viq = row[q];
            row[p] = c * vip - s * viq;
            row[q] = s * vip + c * viq;
        }
    }

    let mut order: Vec<usize> = (0..n).collect();
    order.sort_by(|&i, &j| a[j][j].partial_cmp(&a[i][i]).unwrap_or(Ordering::Equal));

    let eigenvalues = order.iter().map(|&i| a[i][i]).collect::<Vec<_>>();
    let eigenvectors = order
        .iter()
        .map(|&col| (0..n).map(|row| v[row][col]).collect::<Vec<_>>())
        .collect::<Vec<_>>();
    (eigenvalues, eigenvectors)
}

fn embed_from_pivots(pivots: &[usize], dists: &[Vec<i32>], pivot_coords: &[Point]) -> Vec<Point> {
    let n = dists[0].len();
    let k = pivots.len();
    let mut coords = vec![Point::default(); n];

    if k == 1 {
        return circle_layout(n);
    }

    let base = pivot_coords[0];
    for v in 0..n {
        let d0 = dists[0][v] as f64;
        let mut s_xx = 0.0f64;
        let mut s_xy = 0.0f64;
        let mut s_yy = 0.0f64;
        let mut t_x = 0.0f64;
        let mut t_y = 0.0f64;
        let mut rows = 0usize;

        for i in 1..k {
            let dx = pivot_coords[i].x - base.x;
            let dy = pivot_coords[i].y - base.y;
            let a = 2.0 * dx;
            let b = 2.0 * dy;
            if (a.abs() + b.abs()) < 1e-12 {
                continue;
            }
            let di = dists[i][v] as f64;
            let rhs = pivot_coords[i].x * pivot_coords[i].x
                + pivot_coords[i].y * pivot_coords[i].y
                - base.x * base.x
                - base.y * base.y
                - (di * di - d0 * d0);
            s_xx += a * a;
            s_xy += a * b;
            s_yy += b * b;
            t_x += a * rhs;
            t_y += b * rhs;
            rows += 1;
        }

        let det = s_xx * s_yy - s_xy * s_xy;
        if rows >= 2 && det.abs() > 1e-9 {
            coords[v].x = (t_x * s_yy - t_y * s_xy) / det;
            coords[v].y = (s_xx * t_y - s_xy * t_x) / det;
        } else {
            coords[v].x = if k >= 2 { d0 - dists[1][v] as f64 } else { v as f64 };
            coords[v].y = if k >= 3 { d0 - dists[2][v] as f64 } else { 0.0 };
        }
    }

    for (i, &pivot_local) in pivots.iter().enumerate() {
        coords[pivot_local] = pivot_coords[i];
    }
    coords
}

fn collect_local_edges(graph: &Graph, nodes: &[usize], loc: &[usize]) -> Vec<(usize, usize)> {
    let mut edges = Vec::new();
    for (u_local, &u) in nodes.iter().enumerate() {
        for &v in graph.neighbors(u) {
            let v_local = loc[v];
            if v_local != usize::MAX && u_local < v_local {
                edges.push((u_local, v_local));
            }
        }
    }
    edges
}

fn choose_iterations(n: usize, m: usize) -> usize {
    let mut iters = if n <= 40 {
        150
    } else if n <= 120 {
        110
    } else if n <= 300 {
        72
    } else if n <= 800 {
        42
    } else {
        24
    };
    if m > 12 * n {
        iters = iters.min(28);
    }
    if m > 40 * n {
        iters = iters.min(16);
    }
    iters
}

fn refine_force(pos: &mut [Point], edges: &[(usize, usize)], iterations: usize) {
    let n = pos.len();
    if n <= 1 || iterations == 0 {
        return;
    }

    normalize_component(pos, (n as f64).sqrt() * 2.0 + 1.0);
    let desired_len = 1.25f64;
    let repulsion = 2.2f64;
    let spring = 0.16f64;
    let gravity = 0.015f64;

    let sampled_edges_storage;
    let edge_view: &[(usize, usize)] = if edges.len() > 14 * n {
        let step = edges.len() / (14 * n) + 1;
        sampled_edges_storage = edges.iter().step_by(step).copied().collect::<Vec<_>>();
        &sampled_edges_storage
    } else {
        edges
    };

    let mut disp = vec![Point::default(); n];
    let mut temp = (n as f64).sqrt() * 0.30 + 0.8;

    for _ in 0..iterations {

        .sum::<f64>()
        .max(1.0);

    let mut order: Vec<usize> = (0..comps.len()).collect();
    order.sort_by(|&a, &b| {
        let sa = (comps[a].maxx - comps[a].minx).max(comps[a].maxy - comps[a].miny);
        let sb = (comps[b].maxx - comps[b].minx).max(comps[b].maxy - comps[b].miny);
        sb.partial_cmp(&sa).unwrap_or(Ordering::Equal)
    });

    let mut out = vec![Point::default(); total_nodes];

    let mut y = 0.0f64;

        let comp = &comps[idx];
        let width = comp.maxx - comp.minx;
        let height = comp.maxy - comp.miny;

        let box_h = height + gap;

            y += row_h;
            row_h = 0.0;

}

    if (i3 < 0 || bestPlaneDist <= eps) {
        return result;
    }

    const Vec3 innerPoint = (pts[i0] + pts[i1] + pts[i2] + pts[i3]) / 4.0;
    result.innerPoint = innerPoint;

    vector<Face> faces;
    faces.push_back(makeFace(i0, i1, i2, pts, innerPoint));
    faces.push_back(makeFace(i0, i3, i1, pts, innerPoint));
    faces.push_back(makeFace(i0, i2, i3, pts, innerPoint));
    faces.push_back(makeFace(i1, i3, i2, pts, innerPoint));

    vector<char> isSeed(n, 0);
    isSeed[i0] = isSeed[i1] = isSeed[i2] = isSeed[i3] = 1;
    for (int i = 0; i < n; ++i) {
        if (isSeed[i]) continue;
        int bestFace = -1;
        double bestDist = eps;
        for (int f = 0; f < static_cast<int>(faces.size()); ++f) {
            const double dist = signedDistance(faces[f], pts[i]);
            if (dist > bestDist) {
                bestDist = dist;
                bestFace = f;
            }
        }
        if (bestFace >= 0) {
            faces[bestFace].outside.push_back(i);
        }
    }

    vector<int> stamp(n, 0);
    int curStamp = 1;

    while (true) {
        int chosenFace = -1;
        int chosenPoint = -1;
        double farthest = eps;
        for (int f = 0; f < static_cast<int>(faces.size()); ++f) {
            if (!faces[f].alive) continue;
            for (int idx : faces[f].outside) {
                const double dist = signedDistance(faces[f], pts[idx]);
                if (dist > farthest) {
                    farthest = dist;
                    chosenFace = f;
                    chosenPoint = idx;
                }
            }
        }
        if (chosenFace < 0) {
            break;
        }

        vector<int> visible;
        for (int f = 0; f < static_cast<int>(faces.size()); ++f) {
            if (!faces[f].alive) continue;
            if (signedDistance(faces[f], pts[chosenPoint]) > eps) {
                visible.push_back(f);
            }
        }

        map<pair<int, int>, int> edgeMap;
        vector<int> candidatePoints;
        ++curStamp;
        if (curStamp == numeric_limits<int>::max()) {
            fill(stamp.begin(), stamp.end(), 0);
            curStamp = 1;
        }

        auto addEdge = [&](int u, int v) {
            auto it = edgeMap.find({v, u});
            if (it != edgeMap.end()) {
                edgeMap.erase(it);
            } else {
                edgeMap[{u, v}] = 1;
            }
        };

        for (int f : visible) {
            Face& face = faces[f];
            addEdge(face.a, face.b);
            addEdge(face.b, face.c);
            addEdge(face.c, face.a);
            for (int idx : face.outside) {
                if (idx == chosenPoint) continue;
                if (stamp[idx] != curStamp) {
                    stamp[idx] = curStamp;
                    candidatePoints.push_back(idx);
                }
            }
            face.outside.clear();
            face.alive = false;
        }

        vector<int> newFaces;
        newFaces.reserve(edgeMap.size());
        for (const auto& entry : edgeMap) {
            const int u = entry.first.first;
            const int v = entry.first.second;
            Face face = makeFace(u, v, chosenPoint, pts, innerPoint);
            if (norm2(face.n) == 0.0) {
                continue;
            }
            faces.push_back(move(face));
            newFaces.push_back(static_cast<int>(faces.size()) - 1);
        }

        for (int idx : candidatePoints) {
            int bestFace = -1;
            double bestDist = eps;
            for (int f : newFaces) {
                const double dist = signedDistance(faces[f], pts[idx]);
                if (dist > bestDist) {
                    bestDist = dist;
                    bestFace = f;
                }
            }
            if (bestFace >= 0) {
                faces[bestFace].outside.push_back(idx);
            }
        }
    }

    for (const Face& face : faces) {
        if (face.alive) {
            result.faces.push_back(face);
        }
    }
    result.ok = !result.faces.empty();
    return result;
}

struct Basis {
    Vec3 u;
    Vec3 v;
    Vec3 w;
    string name;
};

bool almostSameDirection(const Vec3& a, const Vec3& b, double eps) {
    const double na = norm(a);
    const double nb = norm(b);
    if (na == 0.0 || nb == 0.0) return false;
    const double c = fabs(dot(a, b) / (na * nb));
    return c > 1.0 - eps;
}

bool almostSameBasis(const Basis& a, const Basis& b) {
    return almostSameDirection(a.u, b.u, 1e-6)
        && almostSameDirection(a.v, b.v, 1e-6)
        && almostSameDirection(a.w, b.w, 1e-6);
}

bool orthonormalizeBasis(Vec3 u, Vec3 v, Basis& out, const string& name, double eps) {
    if (norm(u) <= eps || norm(v) <= eps) return false;
    u = normalize(u);
    v = v - u * dot(u, v);
    if (norm(v) <= eps) return false;
    v = normalize(v);
    Vec3 w = cross(u, v);
    if (norm(w) <= eps) return false;
    w = normalize(w);
    v = normalize(cross(w, u));
    out.u = u;
    out.v = v;
    out.w = w;
    out.name = name;
    return true;
}

struct Symmetric3 {
    double a[3][3] = {};
};

pair<array<double, 3>, Mat3> jacobiEigenDecomposition(const Symmetric3& input) {
    double a[3][3];
    for (int i = 0; i < 3; ++i) {
        for (int j = 0; j < 3; ++j) {
            a[i][j] = input.a[i][j];
        }
    }
    Mat3 v = Mat3::identity();

    for (int iter = 0; iter < 32; ++iter) {
        int p = 0;
        int q = 1;
        double maxOff = fabs(a[0][1]);
        if (fabs(a[0][2]) > maxOff) {
            p = 0; q = 2; maxOff = fabs(a[0][2]);
        }
        if (fabs(a[1][2]) > maxOff) {
            p = 1; q = 2; maxOff = fabs(a[1][2]);
        }
        if (maxOff < 1e-14) {
            break;
        }

        const double app = a[p][p];
        const double aqq = a[q][q];
        const double apq = a[p][q];
        const double tau = (aqq - app) / (2.0 * apq);
        const double t = ((tau >= 0.0) ? 1.0 : -1.0) / (fabs(tau) + sqrt(1.0 + tau * tau));
        const double c = 1.0 / sqrt(1.0 + t * t);
        const double s = t * c;

        for (int r = 0; r < 3; ++r) {
            if (r == p || r == q) continue;
            const double arp = a[r][p];
            const double arq = a[r][q];
            a[r][p] = a[p][r] = c * arp - s * arq;
            a[r][q] = a[q][r] = c * arq + s * arp;
        }
        a[p][p] = c * c * app - 2.0 * s * c * apq + s * s * aqq;
        a[q][q] = s * s * app + 2.0 * s * c * apq + c * c * aqq;
        a[p][q] = a[q][p] = 0.0;

        for (int r = 0; r < 3; ++r) {
            const double vrp = v.m[r][p];
            const double vrq = v.m[r][q];
            v.m[r][p] = c * vrp - s * vrq;
            v.m[r][q] = s * vrp + c * vrq;
        }
    }

    array<double, 3> evals = {a[0][0], a[1][1], a[2][2]};
    return {evals, v};
}

bool buildPcaBasis(const vector<Vec3>& pts, Basis& basis, double eps) {
    if (pts.size() < 3) return false;
    Vec3 mean(0.0, 0.0, 0.0);
    for (const Vec3& p : pts) {
        mean += p;
    }
    mean = mean / static_cast<double>(pts.size());

    Symmetric3 cov;
    for (const Vec3& p : pts) {
        const Vec3 d = p - mean;
        cov.a[0][0] += d.x * d.x;
        cov.a[0][1] += d.x * d.y;
        cov.a[0][2] += d.x * d.z;
        cov.a[1][1] += d.y * d.y;
        cov.a[1][2] += d.y * d.z;
        cov.a[2][2] += d.z * d.z;
    }
    cov.a[1][0] = cov.a[0][1];
    cov.a[2][0] = cov.a[0][2];
    cov.a[2][1] = cov.a[1][2];

    auto eig = jacobiEigenDecomposition(cov);
    array<int, 3> order = {0, 1, 2};
    sort(order.begin(), order.end(), [&](int a, int b) {
        return eig.first[a] > eig.first[b];
    });

    Vec3 u(eig.second.m[0][order[0]], eig.second.m[1][order[0]], eig.second.m[2][order[0]]);
    Vec3 v(eig.second.m[0][order[1]], eig.second.m[1][order[1]], eig.second.m[2][order[1]]);
    Basis tmp;
    if (!orthonormalizeBasis(u, v, tmp, "pca", eps)) {
        return false;
    }
    basis = tmp;
    return true;
}

bool buildEdgeBasis(const vector<Vec3>& pts, Basis& basis, double eps) {
    const int n = static_cast<int>(pts.size());
    if (n < 3) return false;

    int a = 0;
    int b = 0;
    double best = -1.0;
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            const double d2 = norm2(pts[i] - pts[j]);
            if (d2 > best) {
                best = d2;
                a = i;
                b = j;
            }
        }
    }
    if (best <= eps * eps) return false;
    const Vec3 u = pts[b] - pts[a];
    int c = -1;
    double bestLine = -1.0;
    for (int i = 0; i < n; ++i) {
        if (i == a || i == b) continue;
        const double dist = norm(cross(pts[i] - pts[a], u)) / norm(u);
        if (dist > bestLine) {
            bestLine = dist;
            c = i;
        }
    }
    if (c < 0 || bestLine <= eps) return false;

    Basis tmp;
    if (!orthonormalizeBasis(u, pts[c] - pts[a], tmp, "edge", eps)) {
        return false;
    }
    basis = tmp;
    return true;
}

double frac01(double x) {
    x -= floor(x);
    if (x < 0.0) x += 1.0;
    if (x >= 1.0) x -= 1.0;
    return x;
}

struct PlaneCoeff {
    double cu = 0.0;
    double cv = 0.0;
    double cw = 0.0;
    double d = 0.0;
    double maxA = 0.0;
    double maxB = 0.0;
};

struct BasisData {
    Basis basis;
    vector<PlaneCoeff> planes;
    array<double, 3> minCoord = {0.0, 0.0, 0.0};
    array<double, 3> maxCoord = {0.0, 0.0, 0.0};
    array<double, 3> anchorOffset = {0.0, 0.0, 0.0};
    long double estimatedBoxCells = 0.0L;
    Mat3 rotA;
    Mat3 rotB;
    array<double, 3> eulerA = {0.0, 0.0, 0.0};
    array<double, 3> eulerB = {0.0, 0.0, 0.0};
};

array<double, 3> matrixToEulerZYX(const Mat3& r) {
    array<double, 3> out = {0.0, 0.0, 0.0};
    const double sy = -r.m[2][0];
    if (fabs(sy) < 1.0 - 1e-12) {
        out[1] = asin(sy);
        out[0] = atan2(r.m[2][1], r.m[2][2]);
        out[2] = atan2(r.m[1][0], r.m[0][0]);
    } else {
        out[1] = (sy >= 0.0) ? (kPi / 2.0) : (-kPi / 2.0);
        out[0] = atan2(-r.m[0][1], r.m[1][1]);
        out[2] = 0.0;
    }
    return out;
}

Mat3 solveRotationBetweenTetraSets(const array<Vec3, 4>& src, const array<Vec3, 4>& dst) {
    Mat3 bestRot = Mat3::identity();
    double bestErr = numeric_limits<double>::infinity();
    array<int, 4> perm = {0, 1, 2, 3};

    const Vec3 s0 = src[0];
    const Vec3 su = normalize(src[1] - s0);
    const Vec3 sw = normalize(cross(src[1] - s0, src[2] - s0));
    const Vec3 sv = normalize(cross(sw, su));
    const Mat3 sourceBasis = matFromColumns(su, sv, sw);
    const Mat3 sourceBasisT = transpose(sourceBasis);

    do {
        if (signOfPermutation(perm) < 0) continue;
        const Vec3 t0 = dst[perm[0]];
        const Vec3 tu = normalize(dst[perm[1]] - t0);
        const Vec3 tw = normalize(cross(dst[perm[1]] - t0, dst[perm[2]] - t0));
        if (norm(tw) == 0.0) continue;
        const Vec3 tv = normalize(cross(tw, tu));
        const Mat3 targetBasis = matFromColumns(tu, tv, tw);
        Mat3 rot = multiply(targetBasis, sourceBasisT);
        if (det(rot) < 0.0) continue;
        double err = 0.0;
        for (int i = 0; i < 4; ++i) {
            const Vec3 mapped = multiply(rot, src[i]);
            err = max(err, norm(mapped - dst[perm[i]]));
        }
        if (err < bestErr) {
            bestErr = err;
            bestRot = rot;
        }
    } while (next_permutation(perm.begin(), perm.end()));

    return bestRot;
}

BasisData prepareBasisData(
    const Basis& basis,
    const vector<Vec3>& vertices,
    const vector<Face>& faces,
    double cellSize,
    const Mat3& localRotB
) {
    BasisData out;
    out.basis = basis;
    out.rotA = matFromColumns(basis.u, basis.v, basis.w);
    out.rotB = multiply(out.rotA, localRotB);
    out.eulerA = matrixToEulerZYX(out.rotA);
    out.eulerB = matrixToEulerZYX(out.rotB);

    const Vec3 centroid = accumulate(vertices.begin(), vertices.end(), Vec3(0.0, 0.0, 0.0))
        / static_cast<double>(vertices.size());

    out.minCoord = {numeric_limits<double>::infinity(),
                    numeric_limits<double>::infinity(),
                    numeric_limits<double>::infinity()};
    out.maxCoord = {-numeric_limits<double>::infinity(),
                    -numeric_limits<double>::infinity(),
                    -numeric_limits<double>::infinity()};

    for (const Vec3& p : vertices) {
        const double x = dot(p, basis.u) / cellSize;
        const double y = dot(p, basis.v) / cellSize;
        const double z = dot(p, basis.w) / cellSize;
        out.minCoord[0] = min(out.minCoord[0], x);
        out.minCoord[1] = min(out.minCoord[1], y);
        out.minCoord[2] = min(out.minCoord[2], z);
        out.maxCoord[0] = max(out.maxCoord[0], x);
        out.maxCoord[1] = max(out.maxCoord[1], y);
        out.maxCoord[2] = max(out.maxCoord[2], z);
    }

    const double cx = dot(centroid, basis.u) / cellSize;
    const double cy = dot(centroid, basis.v) / cellSize;
    const double cz = dot(centroid, basis.w) / cellSize;
    out.anchorOffset = {
        frac01(cx - 0.5),
        frac01(cy - 0.5),
        frac01(cz - 0.5)
    };

    const long double nx = ceil(out.maxCoord[0] - out.minCoord[0] + 3.0);
    const long double ny = ceil(out.maxCoord[1] - out.minCoord[1] + 3.0);
    const long double nz = ceil(out.maxCoord[2] - out.minCoord[2] + 3.0);
    out.estimatedBoxCells = max(1.0L, nx) * max(1.0L, ny) * max(1.0L, nz);

    out.planes.reserve(faces.size());
    for (const Face& face : faces) {
        PlaneCoeff pc;
        pc.cu = cellSize * dot(face.n, basis.u);
        pc.cv = cellSize * dot(face.n, basis.v);
        pc.cw = cellSize * dot(face.n, basis.w);
        pc.d = face.d;
        pc.maxA = max(max(0.0, pc.cu + pc.cv), max(pc.cu + pc.cw, pc.cv + pc.cw));
        pc.maxB = max(max(pc.cu, pc.cv), max(pc.cw, pc.cu + pc.cv + pc.cw));
        out.planes.push_back(pc);
    }
    return out;
}

struct IndexRange {
    long long minI = 0;
    long long maxI = -1;
};

IndexRange computeIndexRange(double minCoord, double maxCoord, double offset) {
    IndexRange r;
    r.minI = static_cast<long long>(floor(minCoord - offset - 1.0)) - 1LL;
    r.maxI = static_cast<long long>(ceil(maxCoord - offset)) + 1LL;
    return r;
}

struct Config {
    int basisIndex = -1;
    array<double, 3> offset = {0.0, 0.0, 0.0};
    long long count = 0;
};

long long countConfig(const BasisData& bd, const array<double, 3>& offset, double eps) {
    const IndexRange rx = computeIndexRange(bd.minCoord[0], bd.maxCoord[0], offset[0]);
    const IndexRange ry = computeIndexRange(bd.minCoord[1], bd.maxCoord[1], offset[1]);
    const IndexRange rz = computeIndexRange(bd.minCoord[2], bd.maxCoord[2], offset[2]);
    const int fCount = static_cast<int>(bd.planes.size());

    vector<double> baseI(fCount);
    vector<double> baseJ(fCount);
    vector<double> cur(fCount);

    for (int p = 0; p < fCount; ++p) {
        const PlaneCoeff& pc = bd.planes[p];
        baseI[p] = pc.d + offset[0] * pc.cu + offset[1] * pc.cv + offset[2] * pc.cw;
    }

    long long count = 0;
    for (long long i = rx.minI; i <= rx.maxI; ++i) {
        for (int p = 0; p < fCount; ++p) {
            baseJ[p] = baseI[p] + static_cast<double>(i) * bd.planes[p].cu;
        }
        for (long long j = ry.minI; j <= ry.maxI; ++j) {
            for (int p = 0; p < fCount; ++p) {
                cur[p] = baseJ[p] + static_cast<double>(j) * bd.planes[p].cv
                       + static_cast<double>(rz.minI) * bd.planes[p].cw;
            }
            for (long long k = rz.minI; k <= rz.maxI; ++k) {
                bool fitA = true;
                bool fitB = true;
                for (int p = 0; p < fCount; ++p) {
                    const double v = cur[p];
                    if (v + bd.planes[p].maxA > eps) fitA = false;
                    if (v + bd.planes[p].maxB > eps) fitB = false;
                    if (!fitA && !fitB) break;
                }
                if (fitA || fitB) {
                    ++count;
                }
                for (int p = 0; p < fCount; ++p) {
                    cur[p] += bd.planes[p].cw;
                }
            }
        }
    }
    return count;
}

    vector<double> baseJ(fCount);

                bool fitA = true;
                bool fitB = true;

                double worstB = -numeric_limits<double>::infinity();
                for (int p = 0; p < fCount; ++p) {
                    const double v = cur[p];
                    const double va = v + bd.planes[p].maxA;
                    const double vb = v + bd.planes[p].maxB;

                    if (vb > eps) fitB = false;
                    if (!fitA && !fitB) break;
                }
                if (fitA || fitB) {
                    const bool useA = fitA && (!fitB || worstA <= worstB);
                    const double cx = (static_cast<double>(i) + offset[0] + 0.5) * cellSize;
                    const double cy = (static_cast<double>(j) + offset[1] + 0.5) * cellSize;

                    cur[p] += bd.planes[p].cw;
                }
            }
        }
    }
}

vector<array<double, 3>> makeOffsets(const BasisData& bd, int divisions) {
    vector<array<double, 3>> offsets;
    if (divisions <= 1) {
        offsets.push_back(bd.anchorOffset);
        return offsets;
    }
    offsets.reserve(divisions * divisions * divisions);
    for (int ix = 0; ix < divisions; ++ix) {
        for (int iy = 0; iy < divisions; ++iy) {
            for (int iz = 0; iz < divisions; ++iz) {
                offsets.push_back({
                    frac01(bd.anchorOffset[0] + static_cast<double>(ix) / divisions),
                    frac01(bd.anchorOffset[1] + static_cast<double>(iy) / divisions),
                    frac01(bd.anchorOffset[2] + static_cast<double>(iz) / divisions)
                });
            }
        }
    }
    return offsets;
}

            double eccentricity = eVec.Length();

            double specificEnergy = 0.5 * speed * speed - mu / Math.Max(radius, 1e-12);
            double semiMajorAxis;
            if (specificEnergy < -1e-12)
            {
                semiMajorAxis = -mu / (2.0 * specificEnergy);
            }
            else
            {
                semiMajorAxis = radius;
            }

            semiMajorAxis = Math.Max(semiMajorAxis, EarthRadiusKm + 1.0);

            double periapsis = semiMajorAxis * Math.Max(1.0 - eccentricity, 0.05);
            double apoapsis;
            if (eccentricity < 1.0)
            {
                apoapsis = semiMajorAxis * (1.0 + eccentricity);
            }
            else
            {
                apoapsis = Math.Max(radius, periapsis);
            }

            periapsis = Math.Max(periapsis, EarthRadiusKm + 1.0);
            apoapsis = Math.Max(apoapsis, periapsis);

            double meanRadius = semiMajorAxis * (1.0 + 0.5 * eccentricity * eccentricity);
            meanRadius = Math.Max(meanRadius, periapsis);

            Vec3 pHat;
            if (eccentricity > 1e-9)
            {
                pHat = eVec / eccentricity;
            }
            else
            {
                Vec3 radialInPlane = r - hHat * Vec3.Dot(r, hHat);
                double radialInPlaneMag = radialInPlane.Length();
                if (radialInPlaneMag > 1e-12)
                {
                    pHat = radialInPlane / radialInPlaneMag;
                }
                else
                {
                    pHat = Vec3.OrthogonalUnit(hHat);
                }
            }

            Vec3 qHat = Vec3.Cross(hHat, pHat).NormalizedOrFallback(Vec3.OrthogonalUnit(pHat));
            double phase = Math.Atan2(Vec3.Dot(r, qHat), Vec3.Dot(r, pHat));

            return new OrbitData(
                r,
                v,
                hHat,
                eVec,
                pHat,
                qHat,
                radius,
                speed,
                semiMajorAxis,
                eccentricity,
                periapsis,
                apoapsis,
                meanRadius,
                phase);
        }
    }

    private sealed class Solver
    {
        private readonly double _mu;
        private readonly OrbitData _start;
        private readonly OrbitData[] _stations;
        private readonly int _n;
        private readonly double[] _startCost;
        private readonly double[] _cost;
        private readonly int[][] _neighbors;
        private readonly int[] _startOrder;

        public Solver(double mu, OrbitData start, OrbitData[] stations)
        {
            _mu = mu;
            _start = start;
            _stations = stations;
            _n = stations.Length;
            _startCost = new double[_n];
            _cost = new double[_n * _n];

            for (int i = 0; i < _n; i++)
            {
                _startCost[i] = TransferCost(_start, _stations[i]);
            }

            for (int i = 0; i < _n; i++)
            {
                _cost[(i * _n) + i] = 0.0;
                for (int j = i + 1; j < _n; j++)
                {
                    double c = TransferCost(_stations[i], _stations[j]);
                    _cost[(i * _n) + j] = c;
                    _cost[(j * _n) + i] = c;
                }
            }

            _neighbors = BuildNeighborLists();
            _startOrder = BuildStartOrder();
        }

        public int[] Solve()
        {
            if (_n <= 1)
            {
                return _n == 0 ? Array.Empty<int>() : new[] { 0 };
            }

            if (_n <= 20)
            {
                return SolveExactly();
            }

            return SolveHeuristically();
        }

        private int[] SolveExactly()
        {
            int stateCount = 1 << _n;
            int tableSize = stateCount * _n;
            double[] dp = new double[tableSize];
            short[] parent = new short[tableSize];
            Array.Fill(dp, Infinity);
            Array.Fill(parent, (short)-1);

            for (int i = 0; i < _n; i++)
            {
                dp[((1 << i) * _n) + i] = _startCost[i];
            }

            int fullMask = stateCount - 1;
            for (int mask = 1; mask <= fullMask; mask++)
            {
                int bits = mask;
                while (bits != 0)
                {
                    int last = BitOperations.TrailingZeroCount((uint)bits);
                    bits &= bits - 1;

                    int baseIndex = (mask * _n) + last;
                    double current = dp[baseIndex];
                    if (current >= Infinity * 0.5)
                    {
                        continue;
                    }

                    int remaining = fullMask ^ mask;
                    int rem = remaining;
                    while (rem != 0)
                    {
                        int next = BitOperations.TrailingZeroCount((uint)rem);
                        rem &= rem - 1;

                        int nextMask = mask | (1 << next);
                        int nextIndex = (nextMask * _n) + next;
                        double candidate = current + GetCost(last, next);
                        if (candidate < dp[nextIndex])
                        {
                            dp[nextIndex] = candidate;
                            parent[nextIndex] = (short)last;
                        }
                    }
                }
            }

            double bestCost = Infinity;
            int bestLast = 0;
            int finalRow = fullMask * _n;
            for (int last = 0; last < _n; last++)
            {
                double candidate = dp[finalRow + last];
                if (candidate < bestCost)
                {
                    bestCost = candidate;
                    bestLast = last;
                }
            }

            int[] route = new int[_n];
            int currentMask = fullMask;
            int cursor = bestLast;
            for (int pos = _n - 1; pos >= 0; pos--)
            {
                route[pos] = cursor;
                int idx = (currentMask * _n) + cursor;
                int previous = parent[idx];
                currentMask ^= 1 << cursor;
                cursor = previous;
                if (currentMask == 0)
                {
                    break;
                }
            }

            return route;
        }

        private int[] SolveHeuristically()
        {
            var random = new Random(1234567 + (_n * 97));

            int[] bestRoute = BuildCheapestInsertion();
            LocalSearch(bestRoute);
            double bestCost = RouteCost(bestRoute);

            TryCandidate(BuildFarthestInsertion(), ref bestRoute, ref bestCost);

            int greedyTrials = Math.Max(8, Math.Min(24, _n / 3));
            for (int t = 0; t < greedyTrials; t++)
            {
                int rclSize = 2 + (t % 4);
                TryCandidate(BuildGreedyRoute(random, rclSize), ref bestRoute, ref bestCost);
            }

            int perturbTrials = _n >= 80 ? 24 : (_n >= 50 ? 18 : 12);
            for (int t = 0; t < perturbTrials; t++)
            {
                int[] perturbed = (int[])bestRoute.Clone();
                Perturb(perturbed, random);
                LocalSearch(perturbed);
                double candidateCost = RouteCost(perturbed);
                if (candidateCost < bestCost)
                {
                    bestCost = candidateCost;
                    bestRoute = perturbed;
                }
            }

            return bestRoute;
        }

        private void TryCandidate(int[] route, ref int[] bestRoute, ref double bestCost)
        {
            LocalSearch(route);
            double candidateCost = RouteCost(route);
            if (candidateCost < bestCost)
            {
                bestCost = candidateCost;
                bestRoute = route;
            }
        }

        private int[] BuildCheapestInsertion()
        {
            int[] route = new int[_n];
            bool[] used = new bool[_n];
            int length = 0;

            int first = _startOrder[0];
            route[length++] = first;
            used[first] = true;

            if (_n == 1)
            {
                return route;
            }

            int second = -1;
            double bestSecond = Infinity;
            for (int rank = 1; rank < _startOrder.Length; rank++)
            {
                int candidate = _startOrder[rank];
                double score = Math.Min(_startCost[first] + GetCost(first, candidate), _startCost[candidate] + GetCost(candidate, first));
                if (score < bestSecond)
                {
                    bestSecond = score;
                    second = candidate;
                }
            }

            route[length++] = second;
            used[second] = true;

            while (length < _n)
            {
                double bestIncrease = Infinity;
                int bestNode = -1;
                int bestPosition = 0;

                for (int node = 0; node < _n; node++)
                {
                    if (used[node])
                    {
                        continue;
                    }

                    double frontIncrease = EdgeCost(-1, node) + EdgeCost(node, route[0]) - EdgeCost(-1, route[0]);
                    if (frontIncrease < bestIncrease)
                    {
                        bestIncrease = frontIncrease;
                        bestNode = node;
                        bestPosition = 0;
                    }

                    for (int i = 0; i < length - 1; i++)
                    {
                        double increase = GetCost(route[i], node) + GetCost(node, route[i + 1]) - GetCost(route[i], route[i + 1]);
                        if (increase < bestIncrease)
                        {
                            bestIncrease = increase;
                            bestNode = node;
                            bestPosition = i + 1;
                        }
                    }

                    double endIncrease = EdgeCost(route[length - 1], node);
                    if (endIncrease < bestIncrease)
                    {
                        bestIncrease = endIncrease;
                        bestNode = node;
                        bestPosition = length;
                    }
                }

                InsertAt(route, ref length, bestPosition, bestNode);
                used[bestNode] = true;
            }

            return route;
        }

        private int[] BuildFarthestInsertion()
        {
            int[] route = new int[_n];
            bool[] used = new bool[_n];
            int length = 0;

            int first = _startOrder[0];
            int second = _startOrder[Math.Min(_startOrder.Length - 1, Math.Max(1, _n / 4))];

            route[length++] = first;
            used[first] = true;

            if (!used[second])
            {
                route[length++] = second;
                used[second] = true;
            }

            if (length == 1)
            {
                for (int i = 0; i < _n; i++)
                {
                    if (!used[i])
                    {
                        route[length++] = i;
                        used[i] = true;
                        break;
                    }
                }
            }

            while (length < _n)
            {
                int farthestNode = -1;
                double farthestDistance = double.NegativeInfinity;

                for (int node = 0; node < _n; node++)
                {
                    if (used[node])
                    {
                        continue;
                    }

                    double nearest = _startCost[node];
                    for (int i = 0; i < length; i++)
                    {
                        nearest = Math.Min(nearest, GetCost(node, route[i]));
                    }

                    if (nearest > farthestDistance)
                    {
                        farthestDistance = nearest;
                        farthestNode = node;
                    }
                }

                int bestPosition = 0;
                double bestIncrease = Infinity;

                double frontIncrease = EdgeCost(-1, farthestNode) + EdgeCost(farthestNode, route[0]) - EdgeCost(-1, route[0]);
                if (frontIncrease < bestIncrease)
                {
                    bestIncrease = frontIncrease;
                    bestPosition = 0;
                }

                for (int i = 0; i < length - 1; i++)
                {
                    double increase = GetCost(route[i], farthestNode) + GetCost(farthestNode, route[i + 1]) - GetCost(route[i], route[i + 1]);
                    if (increase < bestIncrease)
                    {
                        bestIncrease = increase;
                        bestPosition = i + 1;
                    }
                }

                double endIncrease = EdgeCost(route[length - 1], farthestNode);
                if (endIncrease < bestIncrease)
                {
                    bestIncrease = endIncrease;
                    bestPosition = length;
                }

                InsertAt(route, ref length, bestPosition, farthestNode);
                used[farthestNode] = true;
            }

            return route;
        }

        private int[] BuildGreedyRoute(Random random, int rclSize)
        {
            int[] route = new int[_n];
            bool[] used = new bool[_n];

            int firstChoices = Math.Min(_n, Math.Max(2, rclSize + 1));
            int first = _startOrder[random.Next(firstChoices)];
            route[0] = first;
            used[first] = true;

            for (int pos = 1; pos < _n; pos++)
            {
                int current = route[pos - 1];
                int selected = -1;
                int seen = 0;
                int[] neighbors = _neighbors[current];

                for (int i = 0; i < neighbors.Length; i++)
                {
                    int candidate = neighbors[i];
                    if (used[candidate])
                    {
                        continue;
                    }

                    if (seen == 0)
                    {
                        selected = candidate;
                    }

                    if (seen < rclSize)
                    {
                        if (random.Next(seen + 1) == 0)
                        {
                            selected = candidate;
                        }
                    }
                    else
                    {
                        break;
                    }

                    seen++;
                }

                if (selected < 0)
                {
                    for (int i = 0; i < _n; i++)
                    {
                        if (!used[i])
                        {
                            selected = i;
                            break;
                        }
                    }
                }

                route[pos] = selected;
                used[selected] = true;
            }

            return route;
        }

        private void LocalSearch(int[] route)
        {
            int maxRounds = Math.Max(8, _n * 6);
            for (int round = 0; round < maxRounds; round++)
            {
                bool improved = false;

                if (RelocateOnePass(route))
                {
                    improved = true;
                }

                if (TwoOptPass(route))
                {
                    improved = true;
                }

                if (!improved)
                {
                    break;
                }
            }
        }

        private bool TwoOptPass(int[] route)
        {
            for (int i = 0; i < _n - 1; i++)
            {
                int prev = i == 0 ? -1 : route[i - 1];
                int first = route[i];
                double oldLeft = EdgeCost(prev, first);

                for (int k = i + 1; k < _n; k++)
                {
                    int last = route[k];
                    int next = k + 1 < _n ? route[k + 1] : -1;
                    double oldRight = EdgeCost(last, next);

                    double newLeft = EdgeCost(prev, last);
                    double newRight = EdgeCost(first, next);
                    double delta = (newLeft + newRight) - (oldLeft + oldRight);

                    if (delta < -1e-10)
                    {
                        Array.Reverse(route, i, k - i + 1);
                        return true;
                    }
                }
            }

            return false;
        }

        private bool RelocateOnePass(int[] route)
        {
            double baseline = RouteCost(route);
            int[] candidate = new int[_n];

            for (int i = 0; i < _n; i++)
            {
                for (int target = 0; target < _n; target++)
                {
                    if (target == i)
                    {
                        continue;
                    }

                    Array.Copy(route, candidate, _n);
                    MoveNode(candidate, i, target);
                    double improvedCost = RouteCost(candidate);
                    if (improvedCost < baseline - 1e-10)
                    {
                        Array.Copy(candidate, route, _n);
                        return true;
                    }
                }
            }

            return false;
        }

        private void Perturb(int[] route, Random random)
        {
            if (_n >= 12)
            {
                int a = 1 + random.Next(_n - 8);
                int b = a + 1 + random.Next(_n - a - 6);
                int c = b + 1 + random.Next(_n - b - 4);
                int d = c + 1 + random.Next(_n - c - 2);

                int[] scratch = new int[_n];
                int idx = 0;
                CopyBlock(route, 0, a, scratch, ref idx);
                CopyBlock(route, c, d, scratch, ref idx);
                CopyBlock(route, b, c, scratch, ref idx);
                CopyBlock(route, a, b, scratch, ref idx);
                CopyBlock(route, d, _n, scratch, ref idx);
                Array.Copy(scratch, route, _n);
            }

            int randomMoves = Math.Max(2, _n / 30);
            for (int m = 0; m < randomMoves; m++)
            {
                int from = random.Next(_n);
                int to = random.Next(_n);
                MoveNode(route, from, to);
            }

            if (_n >= 6)
            {
                int start = random.Next(_n - 3);
                int length = 2 + random.Next(Math.Min(6, _n - start));
                Array.Reverse(route, start, length);
            }
        }

        private static void CopyBlock(int[] source, int start, int endExclusive, int[] destination, ref int index)
        {
            for (int i = start; i < endExclusive; i++)
            {
                destination[index++] = source[i];
            }
        }

        private static void InsertAt(int[] route, ref int length, int position, int node)
        {
            for (int i = length; i > position; i--)
            {
                route[i] = route[i - 1];
            }

            route[position] = node;
            length++;
        }

        private static void MoveNode(int[] route, int from, int to)
        {
            if (from == to)
            {
                return;
            }

            int node = route[from];
            if (to < from)
            {
                Array.Copy(route, to, route, to + 1, from - to);
                route[to] = node;
            }
            else
            {
                Array.Copy(route, from + 1, route, from, to - from);
                route[to] = node;
            }
        }

        private double RouteCost(int[] route)
        {
            double total = _startCost[route[0]];
            for (int i = 0; i < route.Length - 1; i++)
            {
                total += GetCost(route[i], route[i + 1]);
            }

            return total;
        }

        private double TransferCost(OrbitData a, OrbitData b)
        {
            double meanRadiusA = Math.Max(a.MeanRadius, EarthRadiusKm + 1.0);
            double meanRadiusB = Math.Max(b.MeanRadius, EarthRadiusKm + 1.0);
            double averageRadius = 0.5 * (meanRadiusA + meanRadiusB);
            double baseSpeed = Math.Sqrt(_mu / Math.Max(averageRadius, EarthRadiusKm + 1.0));

            double planeAngle = SafeAcos(Vec3.Dot(a.HHat, b.HHat));
            double planeChange = baseSpeed * 2.0 * Math.Sin(0.5 * planeAngle);

            double radialChange = HohmannDeltaV(meanRadiusA, meanRadiusB, _mu);
            double overlapStart = Math.Max(a.Periapsis, b.Periapsis);
            double overlapEnd = Math.Min(a.Apoapsis, b.Apoapsis);
            if (overlapEnd > overlapStart)
            {
                double unionSpan = Math.Max(a.Apoapsis, b.Apoapsis) - Math.Min(a.Periapsis, b.Periapsis);
                if (unionSpan > 1.0)
                {
                    double overlapRatio = (overlapEnd - overlapStart) / unionSpan;
                    radialChange *= 1.0 - (0.45 * overlapRatio);
                }
            }

            double eccentricityChange = 0.55 * baseSpeed * (a.EVec - b.EVec).Length();

            double phaseGap = Math.Abs(WrapAngle(a.Phase - b.Phase));
            double similarity = Math.Exp(
                (-1.2 * planeAngle) -
                (1.3 * Math.Abs(Math.Log(meanRadiusA / meanRadiusB))) -
                (0.9 * Math.Abs(a.Eccentricity - b.Eccentricity)));
            double phaseChange = 0.01 * baseSpeed * similarity * phaseGap;

            double primary = Math.Sqrt((radialChange * radialChange) + (planeChange * planeChange));
            return Math.Max(1e-6, primary + eccentricityChange + phaseChange);
        }

        private static double HohmannDeltaV(double radiusA, double radiusB, double mu)
        {
            if (radiusA <= 0.0 || radiusB <= 0.0)
            {
                return 0.0;
            }

            if (Math.Abs(radiusA - radiusB) < 1e-9)
            {
                return 0.0;
            }

            double v1 = Math.Sqrt(mu / radiusA);
            double v2 = Math.Sqrt(mu / radiusB);
            double transfer1 = Math.Abs(v1 * (Math.Sqrt((2.0 * radiusB) / (radiusA + radiusB)) - 1.0));
            double transfer2 = Math.Abs(v2 * (1.0 - Math.Sqrt((2.0 * radiusA) / (radiusA + radiusB))));
            return transfer1 + transfer2;
        }

        private double EdgeCost(int from, int to)
        {
            if (to < 0)
            {
                return 0.0;
            }

            if (from < 0)
            {
                return _startCost[to];
            }

            return GetCost(from, to);
        }

        private double GetCost(int from, int to)
        {
            return _cost[(from * _n) + to];
        }

        private int[][] BuildNeighborLists()
        {
            int[][] result = new int[_n][];
            for (int i = 0; i < _n; i++)
            {
                int[] order = new int[_n - 1];
                int idx = 0;
                for (int j = 0; j < _n; j++)
                {
                    if (j != i)
                    {
                        order[idx++] = j;
                    }
                }

                Array.Sort(order, (a, b) => GetCost(i, a).CompareTo(GetCost(i, b)));
                result[i] = order;
            }

            return result;
        }

        private int[] BuildStartOrder()
        {
            int[] order = new int[_n];
            for (int i = 0; i < _n; i++)
            {
                order[i] = i;
            }

            Array.Sort(order, (a, b) => _startCost[a].CompareTo(_startCost[b]));
            return order;
        }

        private static double WrapAngle(double angle)
        {
            while (angle <= -Math.PI)
            {
                angle += 2.0 * Math.PI;
            }

            while (angle > Math.PI)
            {
                angle -= 2.0 * Math.PI;
            }

            return angle;
        }

        private static double SafeAcos(double x)
        {
            if (x <= -1.0)
            {
                return Math.PI;
            }

            if (x >= 1.0)
            {
                return 0.0;
            }

            return Math.Acos(x);
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _index;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        public int NextInt()
        {
            return int.Parse(NextToken(), CultureInfo.InvariantCulture);
        }

        public double NextDouble()
        {
            return double.Parse(NextToken(), CultureInfo.InvariantCulture);
        }

        private string NextToken()
        {
            int b;
            do
            {
                b = Read();
                if (b < 0)
                {
                    return string.Empty;
                }
            }
            while (b <= 32);

            char[] chars = new char[32];
            int count = 0;

            while (b > 32)
            {
                if (count == chars.Length)
                {
                    Array.Resize(ref chars, chars.Length * 2);
                }

                chars[count++] = (char)b;
                b = Read();
            }

            return new string(chars, 0, count);
        }

        private int Read()
        {
            if (_index >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _index = 0;
                if (_length <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_index++];
        }
    }
