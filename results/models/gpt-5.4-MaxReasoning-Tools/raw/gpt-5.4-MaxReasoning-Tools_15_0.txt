use std::cmp::Ordering;
use std::fmt::Write as _;
use std::io::{self, Read};

const EPS: f64 = 1.0e-9;
const NEWTON_EPS: f64 = 1.0e-10;
const MAX_SCALE: f64 = 10.0;
const MAX_EDGE: f64 = 2.0 * MAX_SCALE;
const TETRA_HEIGHT_FACTOR: f64 = 0.816_496_580_927_726; // sqrt(2/3)
const STACK_GAP: f64 = 1.0;
const MAX_DEPTH: u32 = 64;

const REF_VERTS: [Vec3; 4] = [
    Vec3::new(1.0, 0.0, -0.707_106_781_186_547_6),
    Vec3::new(-1.0, 0.0, -0.707_106_781_186_547_6),
    Vec3::new(0.0, 1.0, 0.707_106_781_186_547_6),
    Vec3::new(0.0, -1.0, 0.707_106_781_186_547_6),
];

#[derive(Clone, Copy, Debug)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    const fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }

    fn dot(self, other: Self) -> f64 {
        self.x * other.x + self.y * other.y
    }

    fn cross(self, other: Self) -> f64 {
        self.x * other.y - self.y * other.x
    }

    fn norm_sq(self) -> f64 {
        self.dot(self)
    }

    fn norm(self) -> f64 {
        self.norm_sq().sqrt()
    }
}

impl std::ops::Add for Vec2 {
    type Output = Self;

    fn add(self, rhs: Self) -> Self::Output {
        Self::new(self.x + rhs.x, self.y + rhs.y)
    }
}

impl std::ops::Sub for Vec2 {
    type Output = Self;

    fn sub(self, rhs: Self) -> Self::Output {
        Self::new(self.x - rhs.x, self.y - rhs.y)
    }
}

impl std::ops::Mul<f64> for Vec2 {
    type Output = Self;

    fn mul(self, rhs: f64) -> Self::Output {
        Self::new(self.x * rhs, self.y * rhs)
    }
}

impl std::ops::Div<f64> for Vec2 {
    type Output = Self;

    fn div(self, rhs: f64) -> Self::Output {
        Self::new(self.x / rhs, self.y / rhs)
    }
}

#[derive(Clone, Copy, Debug)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    const fn new(x: f64, y: f64, z: f64) -> Self {
        Self { x, y, z }
    }

    fn dot(self, other: Self) -> f64 {
        self.x * other.x + self.y * other.y + self.z * other.z
    }

    fn cross(self, other: Self) -> Self {
        Self::new(
            self.y * other.z - self.z * other.y,
            self.z * other.x - self.x * other.z,
            self.x * other.y - self.y * other.x,
        )
    }

    fn norm_sq(self) -> f64 {
        self.dot(self)
    }

    fn norm(self) -> f64 {
        self.norm_sq().sqrt()
    }

    fn normalized(self) -> Self {
        let len = self.norm();
        if len <= EPS {
            Self::new(0.0, 0.0, 0.0)
        } else {
            self / len
        }
    }
}

impl std::ops::Add for Vec3 {
    type Output = Self;

    fn add(self, rhs: Self) -> Self::Output {
        Self::new(self.x + rhs.x, self.y + rhs.y, self.z + rhs.z)
    }
}

impl std::ops::Sub for Vec3 {
    type Output = Self;

    fn sub(self, rhs: Self) -> Self::Output {
        Self::new(self.x - rhs.x, self.y - rhs.y, self.z - rhs.z)
    }
}

impl std::ops::Mul<f64> for Vec3 {
    type Output = Self;

    fn mul(self, rhs: f64) -> Self::Output {
        Self::new(self.x * rhs, self.y * rhs, self.z * rhs)
    }
}

impl std::ops::Div<f64> for Vec3 {
    type Output = Self;

    fn div(self, rhs: f64) -> Self::Output {
        Self::new(self.x / rhs, self.y / rhs, self.z / rhs)
    }
}

#[derive(Clone, Copy)]
struct Mat3 {
    m: [[f64; 3]; 3],
}

impl Mat3 {
    fn from_cols(c0: Vec3, c1: Vec3, c2: Vec3) -> Self {
        Self {
            m: [
                [c0.x, c1.x, c2.x],
                [c0.y, c1.y, c2.y],
                [c0.z, c1.z, c2.z],
            ],
        }
    }

    fn transpose(self) -> Self {
        let m = self.m;
        Self {
            m: [
                [m[0][0], m[1][0], m[2][0]],
                [m[0][1], m[1][1], m[2][1]],
                [m[0][2], m[1][2], m[2][2]],
            ],
        }
    }

    fn mul(self, rhs: Self) -> Self {
        let mut out = [[0.0; 3]; 3];
        for (r, row) in out.iter_mut().enumerate() {
            for (c, value) in row.iter_mut().enumerate() {
                *value = self.m[r][0] * rhs.m[0][c]
                    + self.m[r][1] * rhs.m[1][c]
                    + self.m[r][2] * rhs.m[2][c];
            }
        }
        Self { m: out }
    }
}

#[derive(Clone, Copy)]
struct LiftSolution {
    z1: f64,
    z2: f64,
    edge: f64,
}

#[derive(Clone, Copy)]
struct TriangleTask {
    verts: [Vec2; 3],
    depth: u32,
}

#[derive(Clone, Copy)]
struct Tetra {
    center: Vec3,
    quat: [f64; 4],
    scale: f64,
    min_dot: f64,
    max_dot: f64,
    min_z: f64,
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut tokens = input.split_whitespace();

    let nv = match tokens.next() {
        Some(value) => value.parse::<usize>().unwrap_or(0),
        None => {
            println!("0");
            return;
        }
    };

    let mut polygon = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x = match tokens.next() {
            Some(value) => value.parse::<f64>().unwrap_or(0.0),
            None => {
                println!("0");
                return;
            }
        };
        let y = match tokens.next() {
            Some(value) => value.parse::<f64>().unwrap_or(0.0),
            None => {
                println!("0");
                return;
            }
        };
        polygon.push(Vec2::new(x, y));
    }

    let mut sun = Vec3::new(
        tokens
            .next()
            .and_then(|value| value.parse::<f64>().ok())
            .unwrap_or(0.0),
        tokens
            .next()
            .and_then(|value| value.parse::<f64>().ok())
            .unwrap_or(0.0),
        tokens
            .next()
            .and_then(|value| value.parse::<f64>().ok())
            .unwrap_or(-1.0),
    );

    if polygon.len() < 3 {
        println!("0");
        return;
    }

    if sun.z > 0.0 {
        sun = sun * -1.0;
    }
    if sun.z.abs() < 1.0e-9 {
        sun.z = -1.0e-9;
    }

    let hull = convex_hull(&polygon);
    if hull.len() < 3 || polygon_area_abs(&hull) <= EPS {
        println!("0");
        return;
    }

    let shear = Vec2::new(sun.x / sun.z, sun.y / sun.z);
    let stack_dir = (sun * -1.0).normalized();

    let ref_basis = face_basis(REF_VERTS[0], REF_VERTS[1], REF_VERTS[2]).unwrap();
    let ref_basis_t = ref_basis.transpose();

    let mut work = Vec::with_capacity(hull.len());
    for i in 1..(hull.len() - 1) {
        work.push(TriangleTask {
            verts: [hull[0], hull[i], hull[i + 1]],
            depth: 0,
        });
    }

    let mut tetrahedra = Vec::new();

    while let Some(task) = work.pop() {
        if triangle_area_abs(task.verts) <= EPS {
            continue;
        }

        let solution = solve_equilateral_lift(task.verts, shear);
        let Some(lift) = solution else {
            if task.depth >= MAX_DEPTH {
                continue;
            }
            split_four(task.verts, task.depth + 1, &mut work);
            continue;
        };

        if lift.edge <= MAX_EDGE * (1.0 + 1.0e-8) {
            if let Some(tetra) = build_tetra(task.verts, lift, shear, stack_dir, ref_basis_t) {
                tetrahedra.push(tetra);
            }
            continue;
        }

        if task.depth >= MAX_DEPTH {
            continue;
        }

        let overshoot = lift.edge / MAX_EDGE;
        if overshoot > 2.5 {
            split_four(task.verts, task.depth + 1, &mut work);
        } else {
            split_longest_edge(task.verts, task.depth + 1, &mut work);
        }
    }

    if tetrahedra.is_empty() {
        println!("0");
        return;
    }

    tetrahedra.sort_by(|a, b| {
        a.min_dot
            .partial_cmp(&b.min_dot)
            .unwrap_or(Ordering::Equal)
    });

    let mut current_end = f64::NEG_INFINITY;
    let mut output = String::new();
    writeln!(&mut output, "{}", tetrahedra.len()).unwrap();

    for tetra in tetrahedra {
        let min_shift_for_ground = ((1.0 - tetra.min_z) / stack_dir.z).max(0.0);
        let min_shift_for_separation = if current_end.is_finite() {
            current_end + STACK_GAP - tetra.min_dot
        } else {
            0.0
        };
        let shift = min_shift_for_ground.max(min_shift_for_separation.max(0.0));
        current_end = tetra.max_dot + shift;
        let center = tetra.center + stack_dir * shift;
        writeln!(
            &mut output,
            "{:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10}",
            center.x,
            center.y,
            center.z,
            tetra.quat[0],
            tetra.quat[1],
            tetra.quat[2],
            tetra.quat[3],
            tetra.scale
        )
        .unwrap();
    }

    print!("{output}");
}

fn polygon_area_abs(poly: &[Vec2]) -> f64 {
    if poly.len() < 3 {
        return 0.0;
    }
    let mut area = 0.0;
    for i in 0..poly.len() {
        area += poly[i].cross(poly[(i + 1) % poly.len()]);
    }
    0.5 * area.abs()
}

fn triangle_area_abs(tri: [Vec2; 3]) -> f64 {
    ((tri[1] - tri[0]).cross(tri[2] - tri[0])).abs() * 0.5
}

fn convex_hull(points: &[Vec2]) -> Vec<Vec2> {
    let mut pts = points.to_vec();
    pts.sort_by(|a, b| {
        a.x.partial_cmp(&b.x)
            .unwrap_or(Ordering::Equal)
            .then_with(|| a.y.partial_cmp(&b.y).unwrap_or(Ordering::Equal))
    });

    pts.dedup_by(|a, b| (a.x - b.x).abs() <= EPS && (a.y - b.y).abs() <= EPS);
    if pts.len() <= 1 {
        return pts;
    }

    let mut lower: Vec<Vec2> = Vec::with_capacity(pts.len());
    for &p in &pts {
        while lower.len() >= 2 {
            let n = lower.len();
            let cross = (lower[n - 1] - lower[n - 2]).cross(p - lower[n - 1]);
            if cross > EPS {
                break;
            }
            lower.pop();
        }
        lower.push(p);
    }

    let mut upper: Vec<Vec2> = Vec::with_capacity(pts.len());
    for &p in pts.iter().rev() {
        while upper.len() >= 2 {
            let n = upper.len();
            let cross = (upper[n - 1] - upper[n - 2]).cross(p - upper[n - 1]);
            if cross > EPS {
                break;
            }
            upper.pop();
        }
        upper.push(p);
    }

    lower.pop();
    upper.pop();
    lower.extend(upper);
    lower
}

fn solve_equilateral_lift(tri: [Vec2; 3], shear: Vec2) -> Option<LiftSolution> {
    let a = tri[1] - tri[0];
    let b = tri[2] - tri[0];
    let c = tri[2] - tri[1];

    let aa = a.norm_sq();
    let bb = b.norm_sq();
    let cc = c.norm_sq();

    let alpha = a.dot(shear);
    let beta = b.dot(shear);
    let gamma = c.dot(shear);
    let k = 1.0 + shear.norm_sq();

    let tri_scale = aa.max(bb).max(cc).sqrt().max(1.0);
    let seed_scale = tri_scale * (1.0 + shear.norm());

    let mut seeds = vec![
        (0.0, 0.0),
        (tri_scale, 0.0),
        (0.0, tri_scale),
        (tri_scale, tri_scale),
        (-tri_scale, tri_scale),
        (tri_scale, -tri_scale),
        (-tri_scale, -tri_scale),
    ];

    for factor in [0.25, 0.5, 1.0, 2.0, 4.0, 8.0] {
        let s = seed_scale * factor;
        seeds.push((s, 0.0));
        seeds.push((0.0, s));
        seeds.push((s, s));
        seeds.push((-s, s));
        seeds.push((s, -s));
        seeds.push((-s, -s));
    }

    let mut best_err = f64::INFINITY;
    let mut best = None;

    for (mut u, mut v) in seeds {
        for _ in 0..64 {
            let (f1, f2, d1, d2, d3) = lift_system(aa, bb, cc, alpha, beta, gamma, k, u, v);
            let err = f1.abs().max(f2.abs());
            if err < best_err {
                best_err = err;
                best = Some(LiftSolution {
                    z1: u,
                    z2: v,
                    edge: ((d1 + d2 + d3) / 3.0).sqrt(),
                });
            }

            if err <= NEWTON_EPS {
                return Some(LiftSolution {
                    z1: u,
                    z2: v,
                    edge: ((d1 + d2 + d3) / 3.0).sqrt(),
                });
            }

            let j11 = 2.0 * alpha + 2.0 * k * u;
            let j12 = -2.0 * beta - 2.0 * k * v;
            let j21 = 2.0 * (alpha + gamma + k * v);
            let j22 = -2.0 * (gamma + k * (v - u));
            let det = j11 * j22 - j12 * j21;
            if det.abs() <= 1.0e-12 {
                break;
            }

            let du = (-f1 * j22 + j12 * f2) / det;
            let dv = (j21 * f1 - j11 * f2) / det;
            let mut step = 1.0;
            let mut improved = false;

            for _ in 0..12 {
                let nu = u + du * step;
                let nv = v + dv * step;
                let (nf1, nf2, _, _, _) =
                    lift_system(aa, bb, cc, alpha, beta, gamma, k, nu, nv);
                let next_err = nf1.abs().max(nf2.abs());
                if next_err < err {
                    u = nu;
                    v = nv;
                    improved = true;
                    break;
                }
                step *= 0.5;
            }

            if !improved {
                break;
            }
        }
    }

    if best_err <= 1.0e-7 {
        best
    } else {
        None
    }
}

fn lift_system(
    aa: f64,
    bb: f64,
    cc: f64,
    alpha: f64,
    beta: f64,
    gamma: f64,
    k: f64,
    u: f64,
    v: f64,
) -> (f64, f64, f64, f64, f64) {
    let d1 = aa + 2.0 * alpha * u + k * u * u;
    let d2 = bb + 2.0 * beta * v + k * v * v;
    let w = v - u;
    let d3 = cc + 2.0 * gamma * w + k * w * w;
    let f1 = d1 - d2;
    let f2 = d1 - d3;
    (f1, f2, d1, d2, d3)
}

fn split_four(tri: [Vec2; 3], depth: u32, work: &mut Vec<TriangleTask>) {
    let m01 = (tri[0] + tri[1]) * 0.5;
    let m12 = (tri[1] + tri[2]) * 0.5;
    let m20 = (tri[2] + tri[0]) * 0.5;
    work.push(TriangleTask {
        verts: [tri[0], m01, m20],
        depth,
    });
    work.push(TriangleTask {
        verts: [m01, tri[1], m12],
        depth,
    });
    work.push(TriangleTask {
        verts: [m20, m12, tri[2]],
        depth,
    });
    work.push(TriangleTask {
        verts: [m01, m12, m20],
        depth,
    });
}

fn split_longest_edge(tri: [Vec2; 3], depth: u32, work: &mut Vec<TriangleTask>) {
    let l01 = (tri[1] - tri[0]).norm_sq();
    let l12 = (tri[2] - tri[1]).norm_sq();
    let l20 = (tri[0] - tri[2]).norm_sq();

    if l01 >= l12 && l01 >= l20 {
        let mid = (tri[0] + tri[1]) * 0.5;
        work.push(TriangleTask {
            verts: [tri[0], mid, tri[2]],
            depth,
        });
        work.push(TriangleTask {
            verts: [mid, tri[1], tri[2]],
            depth,
        });
    } else if l12 >= l20 {
        let mid = (tri[1] + tri[2]) * 0.5;
        work.push(TriangleTask {
            verts: [tri[1], mid, tri[0]],
            depth,
        });
        work.push(TriangleTask {
            verts: [mid, tri[2], tri[0]],
            depth,
        });
    } else {
        let mid = (tri[2] + tri[0]) * 0.5;
        work.push(TriangleTask {
            verts: [tri[2], mid, tri[1]],
            depth,
        });
        work.push(TriangleTask {
            verts: [mid, tri[0], tri[1]],
            depth,
        });
    }
}

fn build_tetra(
    tri: [Vec2; 3],
    lift: LiftSolution,
    shear: Vec2,
    stack_dir: Vec3,
    ref_basis_t: Mat3,
) -> Option<Tetra> {
    let p0 = Vec3::new(tri[0].x, tri[0].y, 0.0);
    let p1 = Vec3::new(
        tri[1].x + shear.x * lift.z1,
        tri[1].y + shear.y * lift.z1,
        lift.z1,
    );
    let p2 = Vec3::new(
        tri[2].x + shear.x * lift.z2,
        tri[2].y + shear.y * lift.z2,
        lift.z2,
    );

    let side = ((p1 - p0).norm() + (p2 - p0).norm() + (p2 - p1).norm()) / 3.0;
    if !(side.is_finite() && side > EPS) {
        return None;
    }

    let face_basis_t = face_basis(p0, p1, p2)?;
    let centroid = (p0 + p1 + p2) / 3.0;
    let normal = column(face_basis_t, 2);
    let p3 = centroid - normal * (TETRA_HEIGHT_FACTOR * side);

    let center = (p0 + p1 + p2 + p3) / 4.0;
    let rotation = face_basis_t.mul(ref_basis_t);
    let quat = quat_from_rotation(rotation);
    let scale = side * 0.5;

    if !(scale.is_finite() && scale > 0.0 && scale <= MAX_SCALE * (1.0 + 1.0e-8)) {
        return None;
    }

    let raw_vertices = [p0, p1, p2, p3];
    let mut min_dot = f64::INFINITY;
    let mut max_dot = f64::NEG_INFINITY;
    let mut min_z = f64::INFINITY;
    for v in raw_vertices {
        let d = v.dot(stack_dir);
        min_dot = min_dot.min(d);
        max_dot = max_dot.max(d);
        min_z = min_z.min(v.z);
    }

    Some(Tetra {
        center,
        quat,
        scale,
        min_dot,
        max_dot,
        min_z,
    })
}

fn face_basis(a: Vec3, b: Vec3, c: Vec3) -> Option<Mat3> {
    let e1 = (b - a).normalized();
    if e1.norm_sq() <= EPS {
        return None;
    }
    let ac = c - a;
    let proj = e1 * ac.dot(e1);
    let e2 = (ac - proj).normalized();
    if e2.norm_sq() <= EPS {
        return None;
    }
    let e3 = e1.cross(e2).normalized();
    if e3.norm_sq() <= EPS {
        return None;
    }
    Some(Mat3::from_cols(e1, e2, e3))
}

fn column(m: Mat3, idx: usize) -> Vec3 {
    Vec3::new(m.m[0][idx], m.m[1][idx], m.m[2][idx])
}

fn quat_from_rotation(m: Mat3) -> [f64; 4] {
    let r = m.m;
    let (mut w, mut x, mut y, mut z);
    let trace = r[0][0] + r[1][1] + r[2][2];

    if trace > 0.0 {
        let s = (trace + 1.0).sqrt() * 2.0;
        w = 0.25 * s;
        x = (r[2][1] - r[1][2]) / s;
        y = (r[0][2] - r[2][0]) / s;
        z = (r[1][0] - r[0][1]) / s;
    } else if r[0][0] > r[1][1] && r[0][0] > r[2][2] {
        let s = (1.0 + r[0][0] - r[1][1] - r[2][2]).sqrt() * 2.0;
        w = (r[2][1] - r[1][2]) / s;
        x = 0.25 * s;
        y = (r[0][1] + r[1][0]) / s;
        z = (r[0][2] + r[2][0]) / s;
    } else if r[1][1] > r[2][2] {
        let s = (1.0 + r[1][1] - r[0][0] - r[2][2]).sqrt() * 2.0;
        w = (r[0][2] - r[2][0]) / s;
        x = (r[0][1] + r[1][0]) / s;
        y = 0.25 * s;
        z = (r[1][2] + r[2][1]) / s;
    } else {
        let s = (1.0 + r[2][2] - r[0][0] - r[1][1]).sqrt() * 2.0;
        w = (r[1][0] - r[0][1]) / s;
        x = (r[0][2] + r[2][0]) / s;
        y = (r[1][2] + r[2][1]) / s;
        z = 0.25 * s;
    }

    let len = (w * w + x * x + y * y + z * z).sqrt();
    if len > 0.0 {
        w /= len;
        x /= len;
        y /= len;
        z /= len;
    } else {
        w = 1.0;
        x = 0.0;
        y = 0.0;
        z = 0.0;
    }

    if w < 0.0 {
        [-w, -x, -y, -z]
    } else {
        [w, x, y, z]
    }
}
