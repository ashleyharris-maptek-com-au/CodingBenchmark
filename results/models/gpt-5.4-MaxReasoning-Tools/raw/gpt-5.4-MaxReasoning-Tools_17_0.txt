using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int containerW = fs.NextInt();
        int containerH = fs.NextInt();
        int containerD = fs.NextInt();

        var boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            int a = fs.NextInt();
            int b = fs.NextInt();
            int c = fs.NextInt();
            boxes[i] = new Box(i, a, b, c);
        }

        var solver = new Solver(boxes, containerW, containerH, containerD);
        var result = solver.Solve();

        using var writer = new StreamWriter(Console.OpenStandardOutput());
        writer.WriteLine(result.Count);
        for (int i = 0; i < result.Count; i++)
        {
            var p = result[i];
            writer.WriteLine($"{p.BoxIndex} {p.X} {p.Y} {p.Z}");
        }
    }
}

internal sealed class Solver
{
    private readonly Box[] _boxes;
    private readonly int _containerW;
    private readonly int _containerH;
    private readonly int _containerD;
    private readonly long _containerVolume;
    private readonly Stopwatch _clock;

    public Solver(Box[] boxes, int containerW, int containerH, int containerD)
    {
        _boxes = boxes;
        _containerW = containerW;
        _containerH = containerH;
        _containerD = containerD;
        _containerVolume = 1L * containerW * containerH * containerD;
        _clock = Stopwatch.StartNew();
    }

    public List<Placement> Solve()
    {
        var best = new List<Placement>();
        long bestVolume = 0;

        if (ShouldRunExact())
        {
            var exact = new ExactSolver(_boxes, _containerW, _containerH, _containerD, _clock, 1800);
            var exactResult = exact.Solve();
            if (IsBetter(exactResult, exact.PackedVolume, best, bestVolume))
            {
                best = exactResult;
                bestVolume = exact.PackedVolume;
            }
        }

        var permutations = new[]
        {
            new AxisPermutation(0, 1, 2),
            new AxisPermutation(0, 2, 1),
            new AxisPermutation(1, 2, 0),
        };

        var layerModes = new[]
        {
            LayerSelectionMode.Density,
            LayerSelectionMode.Count
        };

        foreach (var permutation in permutations)
        {
            foreach (var mode in layerModes)
            {
                if (_clock.ElapsedMilliseconds >= 26000)
                {
                    break;
                }

                var heuristic = RunLayeredHeuristic(permutation, mode);
                if (IsBetter(heuristic.Placements, heuristic.PackedVolume, best, bestVolume))
                {
                    best = heuristic.Placements;
                    bestVolume = heuristic.PackedVolume;
                }
            }
        }

        if (best.Count == 0)
        {
            best = RunSingleBoxFallback();
        }

        return best;
    }

    private bool ShouldRunExact()
    {
        if (_boxes.Length <= 12)
        {
            return true;
        }

        if (_boxes.Length <= 16 && _containerVolume <= 12000)
        {
            return true;
        }

        if (_boxes.Length <= 20 && _containerVolume <= 2500)
        {
            return true;
        }

        return false;
    }

    private static bool IsBetter(List<Placement> candidate, long candidateVolume, List<Placement> currentBest, long currentBestVolume)
    {
        if (candidate.Count != currentBest.Count)
        {
            return candidate.Count > currentBest.Count;
        }

        if (candidateVolume != currentBestVolume)
        {
            return candidateVolume > currentBestVolume;
        }

        return false;
    }

    private HeuristicResult RunLayeredHeuristic(AxisPermutation permutation, LayerSelectionMode mode)
    {
        var localContainer = permutation.ToLocalDimensions(_containerW, _containerH, _containerD);
        int localW = localContainer.A;
        int localH = localContainer.B;
        int localD = localContainer.C;

        var remaining = new bool[_boxes.Length];
        for (int i = 0; i < remaining.Length; i++)
        {
            remaining[i] = true;
        }

        var placements = new List<Placement>();
        long packedVolume = 0;
        int currentDepth = 0;

        while (currentDepth < localD)
        {
            if (_clock.ElapsedMilliseconds >= 28000)
            {
                break;
            }

            int depthLeft = localD - currentDepth;
            var candidateDepths = CollectCandidateDepths(depthLeft, remaining);
            if (candidateDepths.Count == 0)
            {
                break;
            }

            LayerPackResult bestLayer = null;
            for (int i = 0; i < candidateDepths.Count; i++)
            {
                var simulated = PackOneLayer(localW, localH, candidateDepths[i], remaining);
                if (simulated.Count == 0)
                {
                    continue;
                }

                if (bestLayer == null || CompareLayer(simulated, bestLayer, mode) > 0)
                {
                    bestLayer = simulated;
                }
            }

            if (bestLayer == null)
            {
                break;
            }

            for (int i = 0; i < bestLayer.Items.Count; i++)
            {
                var item = bestLayer.Items[i];
                remaining[item.BoxIndex] = false;
                packedVolume += _boxes[item.BoxIndex].Volume;

                var global = permutation.ToGlobal(item.X, item.Y, currentDepth);
                placements.Add(new Placement(item.BoxIndex, global.X, global.Y, global.Z));
            }

            currentDepth += bestLayer.Depth;
        }

        return new HeuristicResult(placements, packedVolume);
    }

    private List<int> CollectCandidateDepths(int depthLeft, bool[] remaining)
    {
        var freq = new Dictionary<int, int>();
        int smallest = int.MaxValue;

        for (int i = 0; i < _boxes.Length; i++)
        {
            if (!remaining[i])
            {
                continue;
            }

            var box = _boxes[i];
            if (box.A <= depthLeft)
            {
                AddDepthCandidate(freq, box.A);
                if (box.A < smallest) smallest = box.A;
            }

            if (box.B <= depthLeft)
            {
                AddDepthCandidate(freq, box.B);
                if (box.B < smallest) smallest = box.B;
            }

            if (box.C <= depthLeft)
            {
                AddDepthCandidate(freq, box.C);
                if (box.C < smallest) smallest = box.C;
            }
        }

        if (freq.Count == 0)
        {
            return new List<int>();
        }

        var distinct = new List<int>(freq.Keys);
        distinct.Sort();

        var result = new List<int>(12);
        for (int i = 0; i < distinct.Count && i < 6; i++)
        {
            result.Add(distinct[i]);
        }

        var ranked = new List<KeyValuePair<int, int>>(freq);
        ranked.Sort((x, y) =>
        {
            int cmp = y.Value.CompareTo(x.Value);
            if (cmp != 0) return cmp;
            return x.Key.CompareTo(y.Key);
        });

        for (int i = 0; i < ranked.Count && i < 4; i++)
        {
            TryAddUnique(result, ranked[i].Key);
        }

        TryAddUnique(result, smallest);
        if (freq.ContainsKey(depthLeft))
        {
            TryAddUnique(result, depthLeft);
        }

        result.Sort();
        return result;
    }

    private static void AddDepthCandidate(Dictionary<int, int> freq, int depth)
    {
        if (freq.TryGetValue(depth, out int count))
        {
            freq[depth] = count + 1;
        }
        else
        {
            freq[depth] = 1;
        }
    }

    private static void TryAddUnique(List<int> list, int value)
    {
        for (int i = 0; i < list.Count; i++)
        {
            if (list[i] == value)
            {
                return;
            }
        }

        list.Add(value);
    }

    private LayerPackResult PackOneLayer(int width, int height, int layerDepth, bool[] remaining)
    {
        var candidates = new List<LayerCandidate>();

        for (int i = 0; i < _boxes.Length; i++)
        {
            if (!remaining[i])
            {
                continue;
            }

            var orientations = CollectLayerOrientations(_boxes[i], width, height, layerDepth);
            if (orientations.Count == 0)
            {
                continue;
            }

            int minArea = int.MaxValue;
            int minMaxSide = int.MaxValue;
            for (int j = 0; j < orientations.Count; j++)
            {
                int area = orientations[j].W * orientations[j].H;
                if (area < minArea) minArea = area;
                int maxSide = orientations[j].W > orientations[j].H ? orientations[j].W : orientations[j].H;
                if (maxSide < minMaxSide) minMaxSide = maxSide;
            }

            candidates.Add(new LayerCandidate(i, orientations, minArea, minMaxSide, _boxes[i].Volume));
        }

        candidates.Sort((left, right) =>
        {
            int cmp = left.MinArea.CompareTo(right.MinArea);
            if (cmp != 0) return cmp;
            cmp = left.MinMaxSide.CompareTo(right.MinMaxSide);
            if (cmp != 0) return cmp;
            cmp = left.Volume.CompareTo(right.Volume);
            if (cmp != 0) return cmp;
            return left.BoxIndex.CompareTo(right.BoxIndex);
        });

        var skyline = new SkylineBin(width, height);
        var placements = new List<LayerPlacement>();
        int usedArea = 0;

        for (int i = 0; i < candidates.Count; i++)
        {
            int bestOri = -1;
            int bestX = 0;
            int bestY = 0;
            int bestTop = int.MaxValue;
            int bestWaste = int.MaxValue;

            for (int j = 0; j < candidates[i].Orientations.Count; j++)
            {
                var orientation = candidates[i].Orientations[j];
                if (!skyline.TryFindPosition(orientation.W, orientation.H, out int x, out int y))
                {
                    continue;
                }

                int top = y + orientation.H;
                int waste = (width - x - orientation.W) + (height - y - orientation.H);
                if (top < bestTop || (top == bestTop && (y < bestY || (y == bestY && (x < bestX || (x == bestX && waste < bestWaste))))))
                {
                    bestOri = j;
                    bestX = x;
                    bestY = y;
                    bestTop = top;
                    bestWaste = waste;
                }
            }

            if (bestOri < 0)
            {
                continue;
            }

            var chosen = candidates[i].Orientations[bestOri];
            skyline.Place(chosen.W, chosen.H, bestX, bestY);
            placements.Add(new LayerPlacement(candidates[i].BoxIndex, bestX, bestY));
            usedArea += chosen.W * chosen.H;
        }

        return new LayerPackResult(layerDepth, placements, usedArea);
    }

    private static List<Orientation> CollectLayerOrientations(Box box, int width, int height, int layerDepth)
    {
        var result = new List<Orientation>(3);
        for (int i = 0; i < box.Orientations.Length; i++)
        {
            var orientation = box.Orientations[i];
            if (orientation.D != layerDepth)
            {
                continue;
            }

            if (orientation.W <= width && orientation.H <= height)
            {
                bool duplicate = false;
                for (int j = 0; j < result.Count; j++)
                {
                    if (result[j].W == orientation.W && result[j].H == orientation.H)
                    {
                        duplicate = true;
                        break;
                    }
                }

                if (!duplicate)
                {
                    result.Add(orientation);
                }
            }
        }

        return result;
    }

    private static int CompareLayer(LayerPackResult left, LayerPackResult right, LayerSelectionMode mode)
    {
        if (mode == LayerSelectionMode.Density)
        {
            long lhs = 1L * left.Count * right.Depth;
            long rhs = 1L * right.Count * left.Depth;
            if (lhs != rhs)
            {
                return lhs.CompareTo(rhs);
            }
        }

        if (left.Count != right.Count)
        {
            return left.Count.CompareTo(right.Count);
        }

        long leftUsed = 1L * left.UsedArea * right.Depth;
        long rightUsed = 1L * right.UsedArea * left.Depth;
        if (leftUsed != rightUsed)
        {
            return leftUsed.CompareTo(rightUsed);
        }

        if (left.Depth != right.Depth)
        {
            return right.Depth.CompareTo(left.Depth);
        }

        return 0;
    }

    private List<Placement> RunSingleBoxFallback()
    {
        var result = new List<Placement>(1);
        for (int i = 0; i < _boxes.Length; i++)
        {
            if (FitsAnywhere(_boxes[i]))
            {
                result.Add(new Placement(_boxes[i].Index, 0, 0, 0));
                break;
            }
        }

        return result;
    }

    private bool FitsAnywhere(Box box)
    {
        for (int i = 0; i < box.Orientations.Length; i++)
        {
            var o = box.Orientations[i];
            if (o.W <= _containerW && o.H <= _containerH && o.D <= _containerD)
            {
                return true;
            }
        }

        return false;
    }
}

internal sealed class ExactSolver
{
    private readonly Box[] _boxes;
    private readonly int _w;
    private readonly int _h;
    private readonly int _d;
    private readonly int _volume;
    private readonly bool[] _occupied;
    private readonly bool[] _used;
    private readonly Stopwatch _clock;
    private readonly long _deadlineMs;
    private readonly int[] _boxOrder;
    private readonly List<Placement> _current;
    private List<Placement> _best;
    private long _bestVolume;

    public ExactSolver(Box[] boxes, int w, int h, int d, Stopwatch clock, long timeBudgetMs)
    {
        _boxes = boxes;
        _w = w;
        _h = h;
        _d = d;
        _volume = w * h * d;
        _occupied = new bool[_volume];
        _used = new bool[boxes.Length];
        _clock = clock;
        _deadlineMs = clock.ElapsedMilliseconds + timeBudgetMs;
        _boxOrder = BuildBoxOrder(boxes);
        _current = new List<Placement>();
        _best = new List<Placement>();
        _bestVolume = 0;
    }

    public long PackedVolume => _bestVolume;

    public List<Placement> Solve()
    {
        Search(0, _volume, 0);
        return _best;
    }

    private void Search(int scanStart, int freeCells, long packedVolume)
    {
        if (_clock.ElapsedMilliseconds >= _deadlineMs)
        {
            return;
        }

        if (_current.Count > _best.Count || (_current.Count == _best.Count && packedVolume > _bestVolume))
        {
            _best = new List<Placement>(_current);
            _bestVolume = packedVolume;
        }

        int optimistic = _current.Count + ComputeVolumeBound(freeCells);
        if (optimistic <= _best.Count)
        {
            return;
        }

        int cell = FindFirstFree(scanStart);
        if (cell < 0)
        {
            return;
        }

        Decode(cell, out int x, out int y, out int z);
        bool placedAny = false;

        for (int orderIndex = 0; orderIndex < _boxOrder.Length; orderIndex++)
        {
            int boxIndex = _boxOrder[orderIndex];
            if (_used[boxIndex])
            {
                continue;
            }

            var box = _boxes[boxIndex];
            if (box.Volume > freeCells)
            {
                continue;
            }

            for (int oi = 0; oi < box.Orientations.Length; oi++)
            {
                var o = box.Orientations[oi];
                if (x + o.W > _w || y + o.H > _h || z + o.D > _d)
                {
                    continue;
                }

                if (!CanPlace(x, y, z, o.W, o.H, o.D))
                {
                    continue;
                }

                placedAny = true;
                Fill(x, y, z, o.W, o.H, o.D, true);
                _used[boxIndex] = true;
                _current.Add(new Placement(boxIndex, x, y, z));

                Search(cell + 1, freeCells - box.Volume, packedVolume + box.Volume);

                _current.RemoveAt(_current.Count - 1);
                _used[boxIndex] = false;
                Fill(x, y, z, o.W, o.H, o.D, false);
            }
        }

        if (!placedAny)
        {
            _occupied[cell] = true;
            Search(cell + 1, freeCells - 1, packedVolume);
            _occupied[cell] = false;
        }
    }

    private int ComputeVolumeBound(int freeCells)
    {
        int packed = 0;
        int cells = 0;

        for (int i = 0; i < _boxOrder.Length; i++)
        {
            int boxIndex = _boxOrder[i];
            if (_used[boxIndex])
            {
                continue;
            }

            cells += _boxes[boxIndex].Volume;
            if (cells > freeCells)
            {
                break;
            }

            packed++;
        }

        return packed;
    }

    private static int[] BuildBoxOrder(Box[] boxes)
    {
        var order = new int[boxes.Length];
        for (int i = 0; i < order.Length; i++)
        {
            order[i] = i;
        }

        Array.Sort(order, (left, right) =>
        {
            int cmp = boxes[left].Volume.CompareTo(boxes[right].Volume);
            if (cmp != 0) return cmp;
            cmp = boxes[left].MaxSide.CompareTo(boxes[right].MaxSide);
            if (cmp != 0) return cmp;
            return boxes[left].Index.CompareTo(boxes[right].Index);
        });

        return order;
    }

    private int FindFirstFree(int start)
    {
        for (int i = start; i < _volume; i++)
        {
            if (!_occupied[i])
            {
                return i;
            }
        }

        return -1;
    }

    private bool CanPlace(int x, int y, int z, int bw, int bh, int bd)
    {
        for (int dz = 0; dz < bd; dz++)
        {
            int baseZ = z + dz;
            for (int dy = 0; dy < bh; dy++)
            {
                int baseIndex = ((baseZ * _h) + (y + dy)) * _w + x;
                for (int dx = 0; dx < bw; dx++)
                {
                    if (_occupied[baseIndex + dx])
                    {
                        return false;
                    }
                }
            }
        }

        return true;
    }

    private void Fill(int x, int y, int z, int bw, int bh, int bd, bool value)
    {
        for (int dz = 0; dz < bd; dz++)
        {
            int baseZ = z + dz;
            for (int dy = 0; dy < bh; dy++)
            {
                int baseIndex = ((baseZ * _h) + (y + dy)) * _w + x;
                for (int dx = 0; dx < bw; dx++)
                {
                    _occupied[baseIndex + dx] = value;
                }
            }
        }
    }

    private void Decode(int index, out int x, out int y, out int z)
    {
        x = index % _w;
        int tmp = index / _w;
        y = tmp % _h;
        z = tmp / _h;
    }
}

internal sealed class SkylineBin
{
    private readonly int _width;
    private readonly int _height;
    private readonly List<SkylineNode> _nodes;

    public SkylineBin(int width, int height)
    {
        _width = width;
        _height = height;
        _nodes = new List<SkylineNode> { new SkylineNode(0, 0, width) };
    }

    public bool TryFindPosition(int rectWidth, int rectHeight, out int bestX, out int bestY)
    {
        int bestTop = int.MaxValue;
        bestX = 0;
        bestY = 0;
        int bestIndex = -1;

        for (int i = 0; i < _nodes.Count; i++)
        {
            if (!Fits(i, rectWidth, rectHeight, out int y))
            {
                continue;
            }

            int top = y + rectHeight;
            if (top < bestTop || (top == bestTop && (_nodes[i].X < bestX || bestIndex < 0)))
            {
                bestTop = top;
                bestIndex = i;
                bestX = _nodes[i].X;
                bestY = y;
            }
        }

        return bestIndex >= 0;
    }

    public void Place(int rectWidth, int rectHeight, int x, int y)
    {
        int insertIndex = 0;
        while (insertIndex < _nodes.Count && _nodes[insertIndex].X < x)
        {
            insertIndex++;
        }

        _nodes.Insert(insertIndex, new SkylineNode(x, y + rectHeight, rectWidth));

        for (int i = insertIndex + 1; i < _nodes.Count; i++)
        {
            var prev = _nodes[i - 1];
            var node = _nodes[i];
            int shrink = prev.X + prev.Width - node.X;
            if (shrink <= 0)
            {
                break;
            }

            node = new SkylineNode(node.X + shrink, node.Y, node.Width - shrink);
            if (node.Width <= 0)
            {
                _nodes.RemoveAt(i);
                i--;
            }
            else
            {
                _nodes[i] = node;
                break;
            }
        }

        Merge();
    }

    private bool Fits(int index, int rectWidth, int rectHeight, out int y)
    {
        int x = _nodes[index].X;
        if (x + rectWidth > _width)
        {
            y = 0;
            return false;
        }

        y = _nodes[index].Y;
        int widthLeft = rectWidth;
        int i = index;

        while (widthLeft > 0)
        {
            if (i >= _nodes.Count)
            {
                return false;
            }

            y = Math.Max(y, _nodes[i].Y);
            if (y + rectHeight > _height)
            {
                return false;
            }

            widthLeft -= _nodes[i].Width;
            i++;
        }

        return true;
    }

    private void Merge()
    {
        for (int i = 0; i < _nodes.Count - 1; i++)
        {
            if (_nodes[i].Y != _nodes[i + 1].Y)
            {
                continue;
            }

            _nodes[i] = new SkylineNode(_nodes[i].X, _nodes[i].Y, _nodes[i].Width + _nodes[i + 1].Width);
            _nodes.RemoveAt(i + 1);
            i--;
        }
    }
}

internal readonly struct Box
{
    public readonly int Index;
    public readonly int A;
    public readonly int B;
    public readonly int C;
    public readonly int Volume;
    public readonly int MaxSide;
    public readonly Orientation[] Orientations;

    public Box(int index, int a, int b, int c)
    {
        Index = index;
        A = a;
        B = b;
        C = c;
        Volume = a * b * c;
        MaxSide = Math.Max(a, Math.Max(b, c));
        Orientations = BuildOrientations(a, b, c);
    }

    private static Orientation[] BuildOrientations(int a, int b, int c)
    {
        var all = new[]
        {
            new Orientation(a, b, c),
            new Orientation(a, c, b),
            new Orientation(b, a, c),
            new Orientation(b, c, a),
            new Orientation(c, a, b),
            new Orientation(c, b, a),
        };

        var unique = new List<Orientation>(6);
        for (int i = 0; i < all.Length; i++)
        {
            bool duplicate = false;
            for (int j = 0; j < unique.Count; j++)
            {
                if (unique[j].W == all[i].W && unique[j].H == all[i].H && unique[j].D == all[i].D)
                {
                    duplicate = true;
                    break;
                }
            }

            if (!duplicate)
            {
                unique.Add(all[i]);
            }
        }

        return unique.ToArray();
    }
}

internal readonly struct Orientation
{
    public readonly int W;
    public readonly int H;
    public readonly int D;

    public Orientation(int w, int h, int d)
    {
        W = w;
        H = h;
        D = d;
    }
}

internal readonly struct Placement
{
    public readonly int BoxIndex;
    public readonly int X;
    public readonly int Y;
    public readonly int Z;

    public Placement(int boxIndex, int x, int y, int z)
    {
        BoxIndex = boxIndex;
        X = x;
        Y = y;
        Z = z;
    }
}

internal readonly struct LayerPlacement
{
    public readonly int BoxIndex;
    public readonly int X;
    public readonly int Y;

    public LayerPlacement(int boxIndex, int x, int y)
    {
        BoxIndex = boxIndex;
        X = x;
        Y = y;
    }
}

internal sealed class LayerPackResult
{
    public readonly int Depth;
    public readonly List<LayerPlacement> Items;
    public readonly int UsedArea;

    public LayerPackResult(int depth, List<LayerPlacement> items, int usedArea)
    {
        Depth = depth;
        Items = items;
        UsedArea = usedArea;
    }

    public int Count => Items.Count;
}

internal readonly struct LayerCandidate
{
    public readonly int BoxIndex;
    public readonly List<Orientation> Orientations;
    public readonly int MinArea;
    public readonly int MinMaxSide;
    public readonly int Volume;

    public LayerCandidate(int boxIndex, List<Orientation> orientations, int minArea, int minMaxSide, int volume)
    {
        BoxIndex = boxIndex;
        Orientations = orientations;
        MinArea = minArea;
        MinMaxSide = minMaxSide;
        Volume = volume;
    }
}

{
    public readonly int X;
    public readonly int Y;
    public readonly int Width;

    public SkylineNode(int x, int y, int width)
    {
        X = x;
        Y = y;
        Width = width;
    }
}

internal readonly struct GlobalPoint
{
    public readonly int X;

internal readonly struct AxisPermutation
{
    private readonly int _xAxis;
    private readonly int _yAxis;
    private readonly int _zAxis;

    public AxisPermutation(int xAxis, int yAxis, int zAxis)
    {
        _xAxis = xAxis;
        _yAxis = yAxis;
        _zAxis = zAxis;
    }

    public Dim3 ToLocalDimensions(int w, int h, int d)
    {
        int[] dims = { w, h, d };
        return new Dim3(dims[_xAxis], dims[_yAxis], dims[_zAxis]);
    }

    public GlobalPoint ToGlobal(int x, int y, int z)
    {
        int[] coords = new int[3];
        coords[_xAxis] = x;
        coords[_yAxis] = y;
        coords[_zAxis] = z;
        return new GlobalPoint(coords[0], coords[1], coords[2]);
    }
}

internal sealed class FastScanner
{
    private readonly Stream _stream;
    private readonly byte[] _buffer;
    private int _length;
    private int _ptr;

    public FastScanner(Stream stream)
    {
        _stream = stream;
        _buffer = new byte[1 << 16];
    }

    public int NextInt()
    {
        int c;
        do
        {
            c = Read();
        } while (c <= 32 && c >= 0);

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        int value = 0;
        while (c > 32)
        {
            value = value * 10 + (c - '0');
            c = Read();
        }

        return value * sign;
    }

    private int Read()
    {
        if (_ptr >= _length)
        {
            _length = _stream.Read(_buffer, 0, _buffer.Length);
            _ptr = 0;
            if (_length <= 0)
            {
                return -1;
            }
        }

        return _buffer[_ptr++];
    }
}
