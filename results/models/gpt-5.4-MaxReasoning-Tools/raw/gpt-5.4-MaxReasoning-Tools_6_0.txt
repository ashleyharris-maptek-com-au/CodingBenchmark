using System;
using System.Diagnostics;
using System.Globalization;
using System.IO;
using System.Numerics;

public static class Program
{
    private const double EarthRadiusKm = 6371.0;
    private const double Infinity = 1e100;

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int stationCount = scanner.NextInt();
        double mu = scanner.NextDouble();

        if (stationCount <= 0)
        {
            Console.WriteLine();
            return;
        }

        OrbitData start = OrbitData.FromState(ReadState(scanner), mu);
        OrbitData[] stations = new OrbitData[stationCount];
        for (int i = 0; i < stationCount; i++)
        {
            stations[i] = OrbitData.FromState(ReadState(scanner), mu);
        }

        var solver = new Solver(mu, start, stations);
        int[] order = solver.Solve();
        Console.WriteLine(string.Join(" ", order));
    }

    private static OrbitState ReadState(FastScanner scanner)
    {
        return new OrbitState(
            new Vec3(scanner.NextDouble(), scanner.NextDouble(), scanner.NextDouble()),
            new Vec3(scanner.NextDouble(), scanner.NextDouble(), scanner.NextDouble()));
    }

    private readonly struct OrbitState
    {
        public OrbitState(Vec3 position, Vec3 velocity)
        {
            Position = position;
            Velocity = velocity;
        }

        public Vec3 Position { get; }
        public Vec3 Velocity { get; }
    }

    private readonly struct OrbitData
    {
        public OrbitData(
            Vec3 position,
            Vec3 velocity,
            Vec3 hHat,
            Vec3 eVec,
            Vec3 pHat,
            Vec3 qHat,
            double radius,
            double speed,
            double semiMajorAxis,
            double eccentricity,
            double periapsis,
            double apoapsis,
            double meanRadius,
            double phase)
        {
            Position = position;
            Velocity = velocity;
            HHat = hHat;
            EVec = eVec;
            PHat = pHat;
            QHat = qHat;
            Radius = radius;
            Speed = speed;
            SemiMajorAxis = semiMajorAxis;
            Eccentricity = eccentricity;
            Periapsis = periapsis;
            Apoapsis = apoapsis;
            MeanRadius = meanRadius;
            Phase = phase;
        }

        public Vec3 Position { get; }
        public Vec3 Velocity { get; }
        public Vec3 HHat { get; }
        public Vec3 EVec { get; }
        public Vec3 PHat { get; }
        public Vec3 QHat { get; }
        public double Radius { get; }
        public double Speed { get; }
        public double SemiMajorAxis { get; }
        public double Eccentricity { get; }
        public double Periapsis { get; }
        public double Apoapsis { get; }
        public double MeanRadius { get; }
        public double Phase { get; }

        public static OrbitData FromState(OrbitState state, double mu)
        {
            Vec3 r = state.Position;
            Vec3 v = state.Velocity;

            double radius = r.Length();
            double speed = v.Length();

            Vec3 h = Vec3.Cross(r, v);
            double hMag = Math.Max(h.Length(), 1e-12);
            Vec3 hHat = h / hMag;

            Vec3 eVec = Vec3.Cross(v, h) / mu - (r / Math.Max(radius, 1e-12));
            double eccentricity = eVec.Length();

            double specificEnergy = 0.5 * speed * speed - mu / Math.Max(radius, 1e-12);
            double semiMajorAxis;
            if (specificEnergy < -1e-12)
            {
                semiMajorAxis = -mu / (2.0 * specificEnergy);
            }
            else
            {
                semiMajorAxis = radius;
            }

            semiMajorAxis = Math.Max(semiMajorAxis, EarthRadiusKm + 1.0);

            double periapsis = semiMajorAxis * Math.Max(1.0 - eccentricity, 0.05);
            double apoapsis;
            if (eccentricity < 1.0)
            {
                apoapsis = semiMajorAxis * (1.0 + eccentricity);
            }
            else
            {
                apoapsis = Math.Max(radius, periapsis);
            }

            periapsis = Math.Max(periapsis, EarthRadiusKm + 1.0);
            apoapsis = Math.Max(apoapsis, periapsis);

            double meanRadius = semiMajorAxis * (1.0 + 0.5 * eccentricity * eccentricity);
            meanRadius = Math.Max(meanRadius, periapsis);

            Vec3 pHat;
            if (eccentricity > 1e-9)
            {
                pHat = eVec / eccentricity;
            }
            else
            {
                Vec3 radialInPlane = r - hHat * Vec3.Dot(r, hHat);
                double radialInPlaneMag = radialInPlane.Length();
                if (radialInPlaneMag > 1e-12)
                {
                    pHat = radialInPlane / radialInPlaneMag;
                }
                else
                {
                    pHat = Vec3.OrthogonalUnit(hHat);
                }
            }

            Vec3 qHat = Vec3.Cross(hHat, pHat).NormalizedOrFallback(Vec3.OrthogonalUnit(pHat));
            double phase = Math.Atan2(Vec3.Dot(r, qHat), Vec3.Dot(r, pHat));

            return new OrbitData(
                r,
                v,
                hHat,
                eVec,
                pHat,
                qHat,
                radius,
                speed,
                semiMajorAxis,
                eccentricity,
                periapsis,
                apoapsis,
                meanRadius,
                phase);
        }
    }

    private readonly struct Vec3
    {
        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public double X { get; }
        public double Y { get; }
        public double Z { get; }

        public double Length()
        {
            return Math.Sqrt((X * X) + (Y * Y) + (Z * Z));
        }

        public Vec3 NormalizedOrFallback(Vec3 fallback)
        {
            double len = Length();
            return len > 1e-12 ? this / len : fallback;
        }

        public static Vec3 OrthogonalUnit(Vec3 v)
        {
            Vec3 axis = Math.Abs(v.X) < 0.8 ? new Vec3(1.0, 0.0, 0.0) : new Vec3(0.0, 1.0, 0.0);
            return Cross(v, axis).NormalizedOrFallback(new Vec3(0.0, 0.0, 1.0));
        }

        public static double Dot(Vec3 a, Vec3 b)
        {
            return (a.X * b.X) + (a.Y * b.Y) + (a.Z * b.Z);
        }

        public static Vec3 Cross(Vec3 a, Vec3 b)
        {
            return new Vec3(
                (a.Y * b.Z) - (a.Z * b.Y),
                (a.Z * b.X) - (a.X * b.Z),
                (a.X * b.Y) - (a.Y * b.X));
        }

        public static Vec3 operator +(Vec3 a, Vec3 b)
        {
            return new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        }

        public static Vec3 operator -(Vec3 a, Vec3 b)
        {
            return new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        }

        public static Vec3 operator *(Vec3 a, double k)
        {
            return new Vec3(a.X * k, a.Y * k, a.Z * k);
        }

        public static Vec3 operator *(double k, Vec3 a)
        {
            return a * k;
        }

        public static Vec3 operator /(Vec3 a, double k)
        {
            return new Vec3(a.X / k, a.Y / k, a.Z / k);
        }
    }

    private sealed class Solver
    {
        private readonly double _mu;
        private readonly OrbitData _start;
        private readonly OrbitData[] _stations;
        private readonly int _n;
        private readonly double[] _startCost;
        private readonly double[] _cost;
        private readonly int[][] _neighbors;
        private readonly int[] _startOrder;

        public Solver(double mu, OrbitData start, OrbitData[] stations)
        {
            _mu = mu;
            _start = start;
            _stations = stations;
            _n = stations.Length;
            _startCost = new double[_n];
            _cost = new double[_n * _n];

            for (int i = 0; i < _n; i++)
            {
                _startCost[i] = TransferCost(_start, _stations[i]);
            }

            for (int i = 0; i < _n; i++)
            {
                _cost[(i * _n) + i] = 0.0;
                for (int j = i + 1; j < _n; j++)
                {
                    double c = TransferCost(_stations[i], _stations[j]);
                    _cost[(i * _n) + j] = c;
                    _cost[(j * _n) + i] = c;
                }
            }

            _neighbors = BuildNeighborLists();
            _startOrder = BuildStartOrder();
        }

        public int[] Solve()
        {
            if (_n <= 1)
            {
                return _n == 0 ? Array.Empty<int>() : new[] { 0 };
            }

            if (_n <= 20)
            {
                return SolveExactly();
            }

            return SolveHeuristically();
        }

        private int[] SolveExactly()
        {
            int stateCount = 1 << _n;
            int tableSize = stateCount * _n;
            double[] dp = new double[tableSize];
            short[] parent = new short[tableSize];
            Array.Fill(dp, Infinity);
            Array.Fill(parent, (short)-1);

            for (int i = 0; i < _n; i++)
            {
                dp[((1 << i) * _n) + i] = _startCost[i];
            }

            int fullMask = stateCount - 1;
            for (int mask = 1; mask <= fullMask; mask++)
            {
                int bits = mask;
                while (bits != 0)
                {
                    int last = BitOperations.TrailingZeroCount((uint)bits);
                    bits &= bits - 1;

                    int baseIndex = (mask * _n) + last;
                    double current = dp[baseIndex];
                    if (current >= Infinity * 0.5)
                    {
                        continue;
                    }

                    int remaining = fullMask ^ mask;
                    int rem = remaining;
                    while (rem != 0)
                    {
                        int next = BitOperations.TrailingZeroCount((uint)rem);
                        rem &= rem - 1;

                        int nextMask = mask | (1 << next);
                        int nextIndex = (nextMask * _n) + next;
                        double candidate = current + GetCost(last, next);
                        if (candidate < dp[nextIndex])
                        {
                            dp[nextIndex] = candidate;
                            parent[nextIndex] = (short)last;
                        }
                    }
                }
            }

            double bestCost = Infinity;
            int bestLast = 0;
            int finalRow = fullMask * _n;
            for (int last = 0; last < _n; last++)
            {
                double candidate = dp[finalRow + last];
                if (candidate < bestCost)
                {
                    bestCost = candidate;
                    bestLast = last;
                }
            }

            int[] route = new int[_n];
            int currentMask = fullMask;
            int cursor = bestLast;
            for (int pos = _n - 1; pos >= 0; pos--)
            {
                route[pos] = cursor;
                int idx = (currentMask * _n) + cursor;
                int previous = parent[idx];
                currentMask ^= 1 << cursor;
                cursor = previous;
                if (currentMask == 0)
                {
                    break;
                }
            }

            return route;
        }

        private int[] SolveHeuristically()
        {
            var random = new Random(1234567 + (_n * 97));

            int[] bestRoute = BuildCheapestInsertion();
            LocalSearch(bestRoute);
            double bestCost = RouteCost(bestRoute);

            TryCandidate(BuildFarthestInsertion(), ref bestRoute, ref bestCost);

            int greedyTrials = Math.Max(8, Math.Min(24, _n / 3));
            for (int t = 0; t < greedyTrials; t++)
            {
                int rclSize = 2 + (t % 4);
                TryCandidate(BuildGreedyRoute(random, rclSize), ref bestRoute, ref bestCost);
            }

            int perturbTrials = _n >= 80 ? 24 : (_n >= 50 ? 18 : 12);
            for (int t = 0; t < perturbTrials; t++)
            {
                int[] perturbed = (int[])bestRoute.Clone();
                Perturb(perturbed, random);
                LocalSearch(perturbed);
                double candidateCost = RouteCost(perturbed);
                if (candidateCost < bestCost)
                {
                    bestCost = candidateCost;
                    bestRoute = perturbed;
                }
            }

            return bestRoute;
        }

        private void TryCandidate(int[] route, ref int[] bestRoute, ref double bestCost)
        {
            LocalSearch(route);
            double candidateCost = RouteCost(route);
            if (candidateCost < bestCost)
            {
                bestCost = candidateCost;
                bestRoute = route;
            }
        }

        private int[] BuildCheapestInsertion()
        {
            int[] route = new int[_n];
            bool[] used = new bool[_n];
            int length = 0;

            int first = _startOrder[0];
            route[length++] = first;
            used[first] = true;

            if (_n == 1)
            {
                return route;
            }

            int second = -1;
            double bestSecond = Infinity;
            for (int rank = 1; rank < _startOrder.Length; rank++)
            {
                int candidate = _startOrder[rank];
                double score = Math.Min(_startCost[first] + GetCost(first, candidate), _startCost[candidate] + GetCost(candidate, first));
                if (score < bestSecond)
                {
                    bestSecond = score;
                    second = candidate;
                }
            }

            route[length++] = second;
            used[second] = true;

            while (length < _n)
            {
                double bestIncrease = Infinity;
                int bestNode = -1;
                int bestPosition = 0;

                for (int node = 0; node < _n; node++)
                {
                    if (used[node])
                    {
                        continue;
                    }

                    double frontIncrease = EdgeCost(-1, node) + EdgeCost(node, route[0]) - EdgeCost(-1, route[0]);
                    if (frontIncrease < bestIncrease)
                    {
                        bestIncrease = frontIncrease;
                        bestNode = node;
                        bestPosition = 0;
                    }

                    for (int i = 0; i < length - 1; i++)
                    {
                        double increase = GetCost(route[i], node) + GetCost(node, route[i + 1]) - GetCost(route[i], route[i + 1]);
                        if (increase < bestIncrease)
                        {
                            bestIncrease = increase;
                            bestNode = node;
                            bestPosition = i + 1;
                        }
                    }

                    double endIncrease = EdgeCost(route[length - 1], node);
                    if (endIncrease < bestIncrease)
                    {
                        bestIncrease = endIncrease;
                        bestNode = node;
                        bestPosition = length;
                    }
                }

                InsertAt(route, ref length, bestPosition, bestNode);
                used[bestNode] = true;
            }

            return route;
        }

        private int[] BuildFarthestInsertion()
        {
            int[] route = new int[_n];
            bool[] used = new bool[_n];
            int length = 0;

            int first = _startOrder[0];
            int second = _startOrder[Math.Min(_startOrder.Length - 1, Math.Max(1, _n / 4))];

            route[length++] = first;
            used[first] = true;

            if (!used[second])
            {
                route[length++] = second;
                used[second] = true;
            }

            if (length == 1)
            {
                for (int i = 0; i < _n; i++)
                {
                    if (!used[i])
                    {
                        route[length++] = i;
                        used[i] = true;
                        break;
                    }
                }
            }

            while (length < _n)
            {
                int farthestNode = -1;
                double farthestDistance = double.NegativeInfinity;

                for (int node = 0; node < _n; node++)
                {
                    if (used[node])
                    {
                        continue;
                    }

                    double nearest = _startCost[node];
                    for (int i = 0; i < length; i++)
                    {
                        nearest = Math.Min(nearest, GetCost(node, route[i]));
                    }

                    if (nearest > farthestDistance)
                    {
                        farthestDistance = nearest;
                        farthestNode = node;
                    }
                }

                int bestPosition = 0;
                double bestIncrease = Infinity;

                double frontIncrease = EdgeCost(-1, farthestNode) + EdgeCost(farthestNode, route[0]) - EdgeCost(-1, route[0]);
                if (frontIncrease < bestIncrease)
                {
                    bestIncrease = frontIncrease;
                    bestPosition = 0;
                }

                for (int i = 0; i < length - 1; i++)
                {
                    double increase = GetCost(route[i], farthestNode) + GetCost(farthestNode, route[i + 1]) - GetCost(route[i], route[i + 1]);
                    if (increase < bestIncrease)
                    {
                        bestIncrease = increase;
                        bestPosition = i + 1;
                    }
                }

                double endIncrease = EdgeCost(route[length - 1], farthestNode);
                if (endIncrease < bestIncrease)
                {
                    bestIncrease = endIncrease;
                    bestPosition = length;
                }

                InsertAt(route, ref length, bestPosition, farthestNode);
                used[farthestNode] = true;
            }

            return route;
        }

        private int[] BuildGreedyRoute(Random random, int rclSize)
        {
            int[] route = new int[_n];
            bool[] used = new bool[_n];

            int firstChoices = Math.Min(_n, Math.Max(2, rclSize + 1));
            int first = _startOrder[random.Next(firstChoices)];
            route[0] = first;
            used[first] = true;

            for (int pos = 1; pos < _n; pos++)
            {
                int current = route[pos - 1];
                int selected = -1;
                int seen = 0;
                int[] neighbors = _neighbors[current];

                for (int i = 0; i < neighbors.Length; i++)
                {
                    int candidate = neighbors[i];
                    if (used[candidate])
                    {
                        continue;
                    }

                    if (seen == 0)
                    {
                        selected = candidate;
                    }

                    if (seen < rclSize)
                    {
                        if (random.Next(seen + 1) == 0)
                        {
                            selected = candidate;
                        }
                    }
                    else
                    {
                        break;
                    }

                    seen++;
                }

                if (selected < 0)
                {
                    for (int i = 0; i < _n; i++)
                    {
                        if (!used[i])
                        {
                            selected = i;
                            break;
                        }
                    }
                }

                route[pos] = selected;
                used[selected] = true;
            }

            return route;
        }

        private void LocalSearch(int[] route)
        {
            int maxRounds = Math.Max(8, _n * 6);
            for (int round = 0; round < maxRounds; round++)
            {
                bool improved = false;

                if (RelocateOnePass(route))
                {
                    improved = true;
                }

                if (TwoOptPass(route))
                {
                    improved = true;
                }

                if (!improved)
                {
                    break;
                }
            }
        }

        private bool TwoOptPass(int[] route)
        {
            for (int i = 0; i < _n - 1; i++)
            {
                int prev = i == 0 ? -1 : route[i - 1];
                int first = route[i];
                double oldLeft = EdgeCost(prev, first);

                for (int k = i + 1; k < _n; k++)
                {
                    int last = route[k];
                    int next = k + 1 < _n ? route[k + 1] : -1;
                    double oldRight = EdgeCost(last, next);

                    double newLeft = EdgeCost(prev, last);
                    double newRight = EdgeCost(first, next);
                    double delta = (newLeft + newRight) - (oldLeft + oldRight);

                    if (delta < -1e-10)
                    {
                        Array.Reverse(route, i, k - i + 1);
                        return true;
                    }
                }
            }

            return false;
        }

        private bool RelocateOnePass(int[] route)
        {
            double baseline = RouteCost(route);
            int[] candidate = new int[_n];

            for (int i = 0; i < _n; i++)
            {
                for (int target = 0; target < _n; target++)
                {
                    if (target == i)
                    {
                        continue;
                    }

                    Array.Copy(route, candidate, _n);
                    MoveNode(candidate, i, target);
                    double improvedCost = RouteCost(candidate);
                    if (improvedCost < baseline - 1e-10)
                    {
                        Array.Copy(candidate, route, _n);
                        return true;
                    }
                }
            }

            return false;
        }

        private void Perturb(int[] route, Random random)
        {
            if (_n >= 12)
            {
                int a = 1 + random.Next(_n - 8);
                int b = a + 1 + random.Next(_n - a - 6);
                int c = b + 1 + random.Next(_n - b - 4);
                int d = c + 1 + random.Next(_n - c - 2);

                int[] scratch = new int[_n];
                int idx = 0;
                CopyBlock(route, 0, a, scratch, ref idx);
                CopyBlock(route, c, d, scratch, ref idx);
                CopyBlock(route, b, c, scratch, ref idx);
                CopyBlock(route, a, b, scratch, ref idx);
                CopyBlock(route, d, _n, scratch, ref idx);
                Array.Copy(scratch, route, _n);
            }

            int randomMoves = Math.Max(2, _n / 30);
            for (int m = 0; m < randomMoves; m++)
            {
                int from = random.Next(_n);
                int to = random.Next(_n);
                MoveNode(route, from, to);
            }

            if (_n >= 6)
            {
                int start = random.Next(_n - 3);
                int length = 2 + random.Next(Math.Min(6, _n - start));
                Array.Reverse(route, start, length);
            }
        }

        private static void CopyBlock(int[] source, int start, int endExclusive, int[] destination, ref int index)
        {
            for (int i = start; i < endExclusive; i++)
            {
                destination[index++] = source[i];
            }
        }

        private static void InsertAt(int[] route, ref int length, int position, int node)
        {
            for (int i = length; i > position; i--)
            {
                route[i] = route[i - 1];
            }

            route[position] = node;
            length++;
        }

        private static void MoveNode(int[] route, int from, int to)
        {
            if (from == to)
            {
                return;
            }

            int node = route[from];
            if (to < from)
            {
                Array.Copy(route, to, route, to + 1, from - to);
                route[to] = node;
            }
            else
            {
                Array.Copy(route, from + 1, route, from, to - from);
                route[to] = node;
            }
        }

        private double RouteCost(int[] route)
        {
            double total = _startCost[route[0]];
            for (int i = 0; i < route.Length - 1; i++)
            {
                total += GetCost(route[i], route[i + 1]);
            }

            return total;
        }

        private double TransferCost(OrbitData a, OrbitData b)
        {
            double meanRadiusA = Math.Max(a.MeanRadius, EarthRadiusKm + 1.0);
            double meanRadiusB = Math.Max(b.MeanRadius, EarthRadiusKm + 1.0);
            double averageRadius = 0.5 * (meanRadiusA + meanRadiusB);
            double baseSpeed = Math.Sqrt(_mu / Math.Max(averageRadius, EarthRadiusKm + 1.0));

            double planeAngle = SafeAcos(Vec3.Dot(a.HHat, b.HHat));
            double planeChange = baseSpeed * 2.0 * Math.Sin(0.5 * planeAngle);

            double radialChange = HohmannDeltaV(meanRadiusA, meanRadiusB, _mu);
            double overlapStart = Math.Max(a.Periapsis, b.Periapsis);
            double overlapEnd = Math.Min(a.Apoapsis, b.Apoapsis);
            if (overlapEnd > overlapStart)
            {
                double unionSpan = Math.Max(a.Apoapsis, b.Apoapsis) - Math.Min(a.Periapsis, b.Periapsis);
                if (unionSpan > 1.0)
                {
                    double overlapRatio = (overlapEnd - overlapStart) / unionSpan;
                    radialChange *= 1.0 - (0.45 * overlapRatio);
                }
            }

            double eccentricityChange = 0.55 * baseSpeed * (a.EVec - b.EVec).Length();

            double phaseGap = Math.Abs(WrapAngle(a.Phase - b.Phase));
            double similarity = Math.Exp(
                (-1.2 * planeAngle) -
                (1.3 * Math.Abs(Math.Log(meanRadiusA / meanRadiusB))) -
                (0.9 * Math.Abs(a.Eccentricity - b.Eccentricity)));
            double phaseChange = 0.01 * baseSpeed * similarity * phaseGap;

            double primary = Math.Sqrt((radialChange * radialChange) + (planeChange * planeChange));
            return Math.Max(1e-6, primary + eccentricityChange + phaseChange);
        }

        private static double HohmannDeltaV(double radiusA, double radiusB, double mu)
        {
            if (radiusA <= 0.0 || radiusB <= 0.0)
            {
                return 0.0;
            }

            if (Math.Abs(radiusA - radiusB) < 1e-9)
            {
                return 0.0;
            }

            double v1 = Math.Sqrt(mu / radiusA);
            double v2 = Math.Sqrt(mu / radiusB);
            double transfer1 = Math.Abs(v1 * (Math.Sqrt((2.0 * radiusB) / (radiusA + radiusB)) - 1.0));
            double transfer2 = Math.Abs(v2 * (1.0 - Math.Sqrt((2.0 * radiusA) / (radiusA + radiusB))));
            return transfer1 + transfer2;
        }

        private double EdgeCost(int from, int to)
        {
            if (to < 0)
            {
                return 0.0;
            }

            if (from < 0)
            {
                return _startCost[to];
            }

            return GetCost(from, to);
        }

        private double GetCost(int from, int to)
        {
            return _cost[(from * _n) + to];
        }

        private int[][] BuildNeighborLists()
        {
            int[][] result = new int[_n][];
            for (int i = 0; i < _n; i++)
            {
                int[] order = new int[_n - 1];
                int idx = 0;
                for (int j = 0; j < _n; j++)
                {
                    if (j != i)
                    {
                        order[idx++] = j;
                    }
                }

                Array.Sort(order, (a, b) => GetCost(i, a).CompareTo(GetCost(i, b)));
                result[i] = order;
            }

            return result;
        }

        private int[] BuildStartOrder()
        {
            int[] order = new int[_n];
            for (int i = 0; i < _n; i++)
            {
                order[i] = i;
            }

            Array.Sort(order, (a, b) => _startCost[a].CompareTo(_startCost[b]));
            return order;
        }

        private static double WrapAngle(double angle)
        {
            while (angle <= -Math.PI)
            {
                angle += 2.0 * Math.PI;
            }

            while (angle > Math.PI)
            {
                angle -= 2.0 * Math.PI;
            }

            return angle;
        }

        private static double SafeAcos(double x)
        {
            if (x <= -1.0)
            {
                return Math.PI;
            }

            if (x >= 1.0)
            {
                return 0.0;
            }

            return Math.Acos(x);
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _index;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        public int NextInt()
        {
            return int.Parse(NextToken(), CultureInfo.InvariantCulture);
        }

        public double NextDouble()
        {
            return double.Parse(NextToken(), CultureInfo.InvariantCulture);
        }

        private string NextToken()
        {
            int b;
            do
            {
                b = Read();
                if (b < 0)
                {
                    return string.Empty;
                }
            }
            while (b <= 32);

            char[] chars = new char[32];
            int count = 0;

            while (b > 32)
            {
                if (count == chars.Length)
                {
                    Array.Resize(ref chars, chars.Length * 2);
                }

                chars[count++] = (char)b;
                b = Read();
            }

            return new string(chars, 0, count);
        }

        private int Read()
        {
            if (_index >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _index = 0;
                if (_length <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_index++];
        }
    }
