use std::fs::{remove_file, File};
use std::io::{self, BufRead, BufReader, BufWriter, Read, Seek, SeekFrom, Write};

const DIR_UP: u8 = 0;
const DIR_RIGHT: u8 = 1;
const DIR_DOWN: u8 = 2;
const DIR_LEFT: u8 = 3;

const IN_MEMORY_PATH_LIMIT: usize = 8_000_000;
const TEMP_PATH_FILE: &str = "__maze_path_rev.bin";

struct BitSet {
    data: Vec<u64>,
}

impl BitSet {
    fn new(bits: usize) -> Self {
        Self {
            data: vec![0; bits.div_ceil(64)],
        }
    }

    fn get(&self, idx: usize) -> bool {
        ((self.data[idx >> 6] >> (idx & 63)) & 1) != 0
    }

    fn set(&mut self, idx: usize) {
        self.data[idx >> 6] |= 1u64 << (idx & 63);
    }
}

struct PackedDirs {
    data: Vec<u64>,
}

impl PackedDirs {
    fn new(cells: usize) -> Self {
        Self {
            data: vec![0; cells.div_ceil(32)],
        }
    }

    fn get(&self, idx: usize) -> u8 {
        let word = self.data[idx >> 5];
        let shift = ((idx & 31) << 1) as u32;
        ((word >> shift) & 0b11) as u8
    }

    fn set(&mut self, idx: usize, dir: u8) {
        let slot = idx >> 5;
        let shift = ((idx & 31) << 1) as u32;
        let mask = !(0b11u64 << shift);
        self.data[slot] = (self.data[slot] & mask) | ((dir as u64) << shift);
    }
}

#[inline]
fn opposite(dir: u8) -> u8 {
    (dir + 2) & 3
}

#[inline]
fn step_index(idx: usize, dir: u8, width: usize) -> usize {
    match dir {
        DIR_UP => idx - width,
        DIR_RIGHT => idx + 1,
        DIR_DOWN => idx + width,
        DIR_LEFT => idx - 1,
        _ => unreachable!(),
    }
}

#[inline]
fn ordered_dirs(row: usize, col: usize, goal_row: usize, goal_col: usize) -> [u8; 4] {
    let vertical = if goal_row >= row { DIR_DOWN } else { DIR_UP };
    let horizontal = if goal_col >= col {
        DIR_RIGHT
    } else {
        DIR_LEFT
    };

    if goal_row.abs_diff(row) >= goal_col.abs_diff(col) {
        [vertical, horizontal, opposite(horizontal), opposite(vertical)]
    } else {
        [horizontal, vertical, opposite(vertical), opposite(horizontal)]
    }
}

#[inline]
fn neighbor(
    idx: usize,
    row: usize,
    col: usize,
    width: usize,
    height: usize,
    dir: u8,
) -> Option<usize> {
    match dir {
        DIR_UP if row > 0 => Some(idx - width),
        DIR_RIGHT if col + 1 < width => Some(idx + 1),
        DIR_DOWN if row + 1 < height => Some(idx + width),
        DIR_LEFT if col > 0 => Some(idx - 1),
        _ => None,
    }
}

fn choose_next(
    idx: usize,
    width: usize,
    height: usize,
    goal_row: usize,
    goal_col: usize,
    passable: &BitSet,
    visited: &BitSet,
) -> Option<(usize, u8)> {
    let row = idx / width;
    let col = idx % width;

    for dir in ordered_dirs(row, col, goal_row, goal_col) {
        if let Some(next) = neighbor(idx, row, col, width, height, dir) {
            if passable.get(next) && !visited.get(next) {
                return Some((next, opposite(dir)));
            }
        }
    }

    None
}

fn walk_forced(
    mut parent_idx: usize,
    mut current: usize,
    goal: usize,
    width: usize,
    height: usize,
    goal_row: usize,
    goal_col: usize,
    passable: &BitSet,
    visited: &mut BitSet,
    parents: &mut PackedDirs,
) -> usize {
    loop {
        if current == goal {
            return current;
        }

        let row = current / width;
        let col = current % width;
        let mut candidate: Option<(usize, u8)> = None;
        let mut count = 0u8;

        for dir in ordered_dirs(row, col, goal_row, goal_col) {
            if let Some(next) = neighbor(current, row, col, width, height, dir) {
                if next == parent_idx {
                    continue;
                }
                if passable.get(next) && !visited.get(next) {
                    count += 1;
                    if count > 1 {
                        return current;
                    }
                    candidate = Some((next, opposite(dir)));
                }
            }
        }

        match candidate {
            Some((next, back_dir)) => {
                visited.set(next);
                parents.set(next, back_dir);
                parent_idx = current;
                current = next;
            }
            None => return current,
        }
    }
}

fn find_path(
    start: usize,
    goal: usize,
    width: usize,
    height: usize,
    passable: &BitSet,
    visited: &mut BitSet,
    parents: &mut PackedDirs,
) -> bool {
    let goal_row = goal / width;
    let goal_col = goal % width;
    let mut current = start;
    visited.set(start);

    loop {
        if current == goal {
            return true;
        }

        if let Some((next, back_dir)) =
            choose_next(current, width, height, goal_row, goal_col, passable, visited)
        {
            visited.set(next);
            parents.set(next, back_dir);
            current = walk_forced(
                current, next, goal, width, height, goal_row, goal_col, passable, visited, parents,
            );
        } else {
            if current == start {
                return false;
            }
            current = step_index(current, parents.get(current), width);
        }
    }
}

fn path_len(start: usize, goal: usize, width: usize, parents: &PackedDirs) -> usize {
    let mut len = 1usize;
    let mut current = goal;

    while current != start {
        current = step_index(current, parents.get(current), width);
        len += 1;
    }

    len
}

fn write_coord_line<W: Write>(out: &mut W, idx: usize, width: usize) -> io::Result<()> {
    let x = idx % width;
    let y = idx / width;
    writeln!(out, "{x} {y}")
}

fn print_path(start: usize, goal: usize, width: usize, parents: &PackedDirs) -> io::Result<()> {
    let len = path_len(start, goal, width, parents);
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "{len}")?;

    if len <= IN_MEMORY_PATH_LIMIT {
        let mut reversed = Vec::with_capacity(len);
        let mut current = goal;
        reversed.push(current as u32);

        while current != start {
            current = step_index(current, parents.get(current), width);
            reversed.push(current as u32);
        }

        for &idx in reversed.iter().rev() {
            write_coord_line(&mut out, idx as usize, width)?;
        }

        return out.flush();
    }

    {
        let file = File::create(TEMP_PATH_FILE)?;
        let mut writer = BufWriter::new(file);
        let mut current = goal;

        loop {
            writer.write_all(&(current as u32).to_le_bytes())?;
            if current == start {
                break;
            }
            current = step_index(current, parents.get(current), width);
        }

        writer.flush()?;
    }

    let temp_result = (|| -> io::Result<()> {
        let mut file = File::open(TEMP_PATH_FILE)?;
        let mut remaining = len;
        let chunk_elems = 1usize << 20;
        let mut buffer = vec![0u8; chunk_elems * 4];

        while remaining > 0 {
            let take = remaining.min(chunk_elems);
            let start_elem = remaining - take;
            let bytes = take * 4;
            file.seek(SeekFrom::Start((start_elem * 4) as u64))?;
            file.read_exact(&mut buffer[..bytes])?;

            let mut i = take;
            while i > 0 {
                i -= 1;
                let offset = i * 4;
                let idx = u32::from_le_bytes([
                    buffer[offset],
                    buffer[offset + 1],
                    buffer[offset + 2],
                    buffer[offset + 3],
                ]) as usize;
                write_coord_line(&mut out, idx, width)?;
            }

            remaining = start_elem;
        }

        out.flush()
    })();

    let _ = remove_file(TEMP_PATH_FILE);
    temp_result
}

fn print_no_path() -> io::Result<()> {
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "0")
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());

    let mut header = String::new();
    if reader.read_line(&mut header)? == 0 {
        return Ok(());
    }

    let mut dims = header.split_whitespace();
    let Some(h_text) = dims.next() else {
        return print_no_path();
    };
    let Some(w_text) = dims.next() else {
        return print_no_path();
    };

    let height: usize = match h_text.parse() {
        Ok(v) => v,
        Err(_) => return print_no_path(),
    };
    let width: usize = match w_text.parse() {
        Ok(v) => v,
        Err(_) => return print_no_path(),
    };

    if height == 0 || width == 0 {
        return print_no_path();
    }

    let cells = match height.checked_mul(width) {
        Some(v) => v,
        None => return print_no_path(),
    };

    if cells > u32::MAX as usize {
        return print_no_path();
    }

    let mut passable = BitSet::new(cells);
    let mut row_buf = Vec::with_capacity(width.saturating_add(2));
    let mut start = None;
    let mut goal = None;

    for row in 0..height {
        row_buf.clear();
        if reader.read_until(b'\n', &mut row_buf)? == 0 {
            return print_no_path();
        }

        if matches!(row_buf.last(), Some(b'\n')) {
            row_buf.pop();
        }
        if matches!(row_buf.last(), Some(b'\r')) {
            row_buf.pop();
        }

        if row_buf.len() < width {
            return print_no_path();
        }

        for col in 0..width {
            let idx = row * width + col;
            match row_buf[col] {
                b'#' => {}
                b'A' => {
                    passable.set(idx);
                    start = Some(idx);
                }
                b'B' => {
                    passable.set(idx);
                    goal = Some(idx);
                }
                _ => {
                    passable.set(idx);
                }
            }
        }
    }

    let (start, goal) = match (start, goal) {
        (Some(s), Some(g)) => (s, g),
        _ => return print_no_path(),
    };

    if start == goal {
        let stdout = io::stdout();
        let mut out = BufWriter::new(stdout.lock());
        writeln!(out, "1")?;
        return write_coord_line(&mut out, start, width);
    }

    let mut visited = BitSet::new(cells);
    let mut parents = PackedDirs::new(cells);

    if find_path(start, goal, width, height, &passable, &mut visited, &mut parents) {
        print_path(start, goal, width, &parents)
    } else {
        print_no_path()
    }
}
