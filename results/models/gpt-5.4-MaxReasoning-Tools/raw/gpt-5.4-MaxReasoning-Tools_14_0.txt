using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private const long MassiveBoardCellThreshold = 6_000_000;
    private const long MassiveUnknownThreshold = 1_500_000;
    private const long MassiveClueThreshold = 1_500_000;

    public static void Main()
    {
        var reader = new StreamReader(Console.OpenStandardInput(), Encoding.ASCII, false, 1 << 20);
        string header = reader.ReadLine();
        if (header == null)
        {
            return;
        }

        ParseHeader(header, out int height, out int width);

        var board = new string[height];
        long unknownCount = 0;
        long clueCount = 0;

        for (int y = 0; y < height; y++)
        {
            string line = reader.ReadLine() ?? string.Empty;
            if (line.Length < width)
            {
                line = line.PadRight(width, ' ');
            }
            else if (line.Length > width)
            {
                line = line.Substring(0, width);
            }

            board[y] = line;

            for (int x = 0; x < width; x++)
            {
                char ch = line[x];
                if (ch == ' ')
                {
                    unknownCount++;
                }
                else if (IsClue(ch))
                {
                    clueCount++;
                }
            }
        }

        var writer = new StreamWriter(Console.OpenStandardOutput(), Encoding.ASCII, 1 << 20);

        bool useMassiveMode =
            (long)height * width >= MassiveBoardCellThreshold ||
            unknownCount >= MassiveUnknownThreshold ||
            clueCount >= MassiveClueThreshold;

        if (!useMassiveMode)
        {
            try
            {
                var advanced = new AdvancedSolver(board, height, width);
                advanced.Solve();
                advanced.WriteOutput(writer);
                writer.Flush();
                return;
            }
            catch (TooLargeForAdvancedException)
            {
                useMassiveMode = true;
            }
        }

        if (useMassiveMode)
        {
            var safeCells = new MassiveSolver(board, height, width).Solve();
            writer.WriteLine(safeCells.Count);
            foreach (long key in safeCells)
            {
                writer.Write(UnpackX(key));
                writer.Write(' ');
                writer.WriteLine(UnpackY(key));
            }
        }

        writer.Flush();
    }

    private static void ParseHeader(string header, out int height, out int width)
    {
        int i = 0;
        while (i < header.Length && header[i] == ' ')
        {
            i++;
        }

        int value = 0;
        while (i < header.Length && header[i] >= '0' && header[i] <= '9')
        {
            value = value * 10 + (header[i] - '0');
            i++;
        }

        height = value;

        while (i < header.Length && header[i] == ' ')
        {
            i++;
        }

        value = 0;
        while (i < header.Length && header[i] >= '0' && header[i] <= '9')
        {
            value = value * 10 + (header[i] - '0');
            i++;
        }

        width = value;
    }

    private static bool IsClue(char ch)
    {
        return ch >= '0' && ch <= '8';
    }

    private static long Pack(int x, int y)
    {
        return ((long)y << 32) | (uint)x;
    }

    private static int UnpackX(long key)
    {
        return (int)(key & 0xFFFFFFFF);
    }

    private static int UnpackY(long key)
    {
        return (int)(key >> 32);
    }

    private sealed class TooLargeForAdvancedException : Exception
    {
    }

    private sealed class MassiveSolver
    {
        private readonly string[] board;
        private readonly int height;
        private readonly int width;
        private readonly HashSet<long> safe = new HashSet<long>();
        private readonly HashSet<long> mine = new HashSet<long>();
        private readonly HashSet<long> activeClues = new HashSet<long>();
        private readonly List<long> activeBuffer = new List<long>();

        public MassiveSolver(string[] board, int height, int width)
        {
            this.board = board;
            this.height = height;
            this.width = width;
        }

        public HashSet<long> Solve()
        {
            for (int y = 0; y < height; y++)
            {
                string row = board[y];
                for (int x = 0; x < width; x++)
                {
                    if (IsClue(row[x]))
                    {
                        EvaluateClue(x, y);
                    }
                }
            }

            while (activeClues.Count > 0)
            {
                activeBuffer.Clear();
                foreach (long key in activeClues)
                {
                    activeBuffer.Add(key);
                }

                activeClues.Clear();

                for (int i = 0; i < activeBuffer.Count; i++)
                {
                    long key = activeBuffer[i];
                    EvaluateClue(UnpackX(key), UnpackY(key));
                }
            }

            return safe;
        }

        private void EvaluateClue(int x, int y)
        {
            int clue = board[y][x] - '0';
            Span<long> unresolved = stackalloc long[8];
            int unresolvedCount = 0;
            int knownMines = 0;

            int minY = y > 0 ? y - 1 : 0;
            int maxY = y + 1 < height ? y + 1 : height - 1;
            int minX = x > 0 ? x - 1 : 0;
            int maxX = x + 1 < width ? x + 1 : width - 1;

            for (int ny = minY; ny <= maxY; ny++)
            {
                string row = board[ny];
                for (int nx = minX; nx <= maxX; nx++)
                {
                    if (nx == x && ny == y)
                    {
                        continue;
                    }

                    char ch = row[nx];
                    if (ch == '*')
                    {
                        knownMines++;
                        continue;
                    }

                    if (ch != ' ')
                    {
                        continue;
                    }

                    long key = Pack(nx, ny);
                    if (mine.Contains(key))
                    {
                        knownMines++;
                    }
                    else if (!safe.Contains(key))
                    {
                        unresolved[unresolvedCount++] = key;
                    }
                }
            }

            int remaining = clue - knownMines;
            if (remaining < 0 || remaining > unresolvedCount || unresolvedCount == 0)
            {
                return;
            }

            if (remaining == 0)
            {
                for (int i = 0; i < unresolvedCount; i++)
                {
                    AssignSafe(unresolved[i]);
                }
            }
            else if (remaining == unresolvedCount)
            {
                for (int i = 0; i < unresolvedCount; i++)
                {
                    AssignMine(unresolved[i]);
                }
            }
        }

        private void AssignSafe(long key)
        {
            if (safe.Add(key))
            {
                MarkAdjacentCluesActive(UnpackX(key), UnpackY(key));
            }
        }

        private void AssignMine(long key)
        {
            if (mine.Add(key))
            {
                MarkAdjacentCluesActive(UnpackX(key), UnpackY(key));
            }
        }

        private void MarkAdjacentCluesActive(int x, int y)
        {
            int minY = y > 0 ? y - 1 : 0;
            int maxY = y + 1 < height ? y + 1 : height - 1;
            int minX = x > 0 ? x - 1 : 0;
            int maxX = x + 1 < width ? x + 1 : width - 1;

            for (int ny = minY; ny <= maxY; ny++)
            {
                string row = board[ny];
                for (int nx = minX; nx <= maxX; nx++)
                {
                    if (IsClue(row[nx]))
                    {
                        activeClues.Add(Pack(nx, ny));
                    }
                }
            }
        }
    }

    private sealed class AdvancedSolver
    {
        private const int AdvancedVariableCap = 1_500_000;
        private const int AdvancedConstraintCap = 1_500_000;
        private const int SubsetVariableLimit = 160_000;
        private const int SubsetConstraintLimit = 80_000;
        private const int SubsetPairLimit = 500_000;
        private const int ExactVariableBudget = 2_000;
        private const int ExactComponentVariableLimit = 24;
        private const int ExactComponentConstraintLimit = 64;
        private const int ExactNodeBudget = 50_000;

        private readonly string[] board;
        private readonly int height;
        private readonly int width;

        private readonly List<int[]> constraints = new List<int[]>();
        private int[] remaining = Array.Empty<int>();
        private int[] unresolved = Array.Empty<int>();
        private int[] varX = Array.Empty<int>();
        private int[] varY = Array.Empty<int>();
        private int[] varAdjStart = Array.Empty<int>();
        private int[] varAdjItems = Array.Empty<int>();
        private byte[] status = Array.Empty<byte>();
        private bool[] queued = Array.Empty<bool>();
        private Queue<int> queue = new Queue<int>();

        public AdvancedSolver(string[] board, int height, int width)
        {
            this.board = board;
            this.height = height;
            this.width = width;
        }

        public void Solve()
        {
            Build();

            queued = new bool[constraints.Count];
            queue = new Queue<int>(constraints.Count);

            for (int c = 0; c < constraints.Count; c++)
            {
                Enqueue(c);
            }

            PropagateQueue();

            bool allowSubset = varX.Length <= SubsetVariableLimit && constraints.Count <= SubsetConstraintLimit;
            bool allowExact = varX.Length > 0;
            int exactBudget = ExactVariableBudget;

            for (int round = 0; round < 8; round++)
            {
                bool changed = false;

                if (allowSubset && RunSubsetReasoning())
                {
                    changed = true;
                    PropagateQueue();
                }

                if (allowExact && exactBudget > 0 && RunExactComponents(ref exactBudget))
                {
                    changed = true;
                    PropagateQueue();
                }

                if (!changed)
                {
                    break;
                }
            }
        }

        public void WriteOutput(StreamWriter writer)
        {
            int safeCount = 0;
            for (int i = 0; i < status.Length; i++)
            {
                if (status[i] == 1)
                {
                    safeCount++;
                }
            }

            writer.WriteLine(safeCount);
            for (int i = 0; i < status.Length; i++)
            {
                if (status[i] == 1)
                {
                    writer.Write(varX[i]);
                    writer.Write(' ');
                    writer.WriteLine(varY[i]);
                }
            }
        }

        private void Build()
        {
            var idByCell = new Dictionary<long, int>(1024);
            var xList = new List<int>(1024);
            var yList = new List<int>(1024);
            var degreeList = new List<int>(1024);
            var remainingList = new List<int>(1024);

            for (int y = 0; y < height; y++)
            {
                string row = board[y];
                for (int x = 0; x < width; x++)
                {
                    char ch = row[x];
                    if (!IsClue(ch))
                    {
                        continue;
                    }

                    int[] vars = new int[8];
                    int count = 0;
                    int fixedMines = 0;

                    int minY = y > 0 ? y - 1 : 0;
                    int maxY = y + 1 < height ? y + 1 : height - 1;
                    int minX = x > 0 ? x - 1 : 0;
                    int maxX = x + 1 < width ? x + 1 : width - 1;

                    for (int ny = minY; ny <= maxY; ny++)
                    {
                        string neighborRow = board[ny];
                        for (int nx = minX; nx <= maxX; nx++)
                        {
                            if (nx == x && ny == y)
                            {
                                continue;
                            }

                            char neighbor = neighborRow[nx];
                            if (neighbor == '*')
                            {
                                fixedMines++;
                                continue;
                            }

                            if (neighbor != ' ')
                            {
                                continue;
                            }

                            long key = Pack(nx, ny);
                            if (!idByCell.TryGetValue(key, out int id))
                            {
                                id = xList.Count;
                                idByCell.Add(key, id);
                                xList.Add(nx);
                                yList.Add(ny);
                                degreeList.Add(0);
                                if (xList.Count > AdvancedVariableCap)
                                {
                                    throw new TooLargeForAdvancedException();
                                }
                            }

                            vars[count++] = id;
                        }
                    }

                    if (count == 0)
                    {
                        continue;
                    }

                    int[] constraintVars = new int[count];
                    for (int i = 0; i < count; i++)
                    {
                        int id = vars[i];
                        constraintVars[i] = id;
                        degreeList[id]++;
                    }

                    constraints.Add(constraintVars);
                    remainingList.Add((ch - '0') - fixedMines);

                    if (constraints.Count > AdvancedConstraintCap)
                    {
                        throw new TooLargeForAdvancedException();
                    }
                }
            }

            varX = xList.ToArray();
            varY = yList.ToArray();
            status = new byte[varX.Length];
            remaining = remainingList.ToArray();
            unresolved = new int[constraints.Count];

            varAdjStart = new int[varX.Length + 1];
            for (int i = 0; i < degreeList.Count; i++)
            {
                varAdjStart[i + 1] = varAdjStart[i] + degreeList[i];
            }

            varAdjItems = new int[varAdjStart[varAdjStart.Length - 1]];
            int[] offsets = (int[])varAdjStart.Clone();

            for (int c = 0; c < constraints.Count; c++)
            {
                int[] vars = constraints[c];
                unresolved[c] = vars.Length;
                for (int i = 0; i < vars.Length; i++)
                {
                    int id = vars[i];
                    varAdjItems[offsets[id]++] = c;
                }
            }
        }

        private void Enqueue(int constraint)
        {
            if (!queued[constraint])
            {
                queued[constraint] = true;
                queue.Enqueue(constraint);
            }
        }

        private bool Assign(int variable, byte value)
        {
            byte current = status[variable];
            if (current != 0)
            {
                return false;
            }

            status[variable] = value;

            for (int p = varAdjStart[variable]; p < varAdjStart[variable + 1]; p++)
            {
                int constraint = varAdjItems[p];
                unresolved[constraint]--;
                if (value == 2)
                {
                    remaining[constraint]--;
                }

                Enqueue(constraint);
            }

            return true;
        }

        private bool PropagateQueue()
        {
            bool changed = false;

            while (queue.Count > 0)
            {
                int constraint = queue.Dequeue();
                queued[constraint] = false;

                int minesLeft = remaining[constraint];
                int varsLeft = unresolved[constraint];
                if (varsLeft == 0 || minesLeft < 0 || minesLeft > varsLeft)
                {
                    continue;
                }

                int[] vars = constraints[constraint];
                if (minesLeft == 0)
                {
                    for (int i = 0; i < vars.Length; i++)
                    {
                        if (status[vars[i]] == 0 && Assign(vars[i], 1))
                        {
                            changed = true;
                        }
                    }
                }
                else if (minesLeft == varsLeft)
                {
                    for (int i = 0; i < vars.Length; i++)
                    {
                        if (status[vars[i]] == 0 && Assign(vars[i], 2))
                        {
                            changed = true;
                        }
                    }
                }
            }

            return changed;
        }

        private bool RunSubsetReasoning()
        {
            var seenPairs = new HashSet<long>();
            bool changed = false;
            int pairCount = 0;

            for (int variable = 0; variable < status.Length; variable++)
            {
                if (status[variable] != 0)
                {
                    continue;
                }

                int[] localConstraints = new int[8];
                int localCount = 0;

                for (int p = varAdjStart[variable]; p < varAdjStart[variable + 1]; p++)
                {
                    int constraint = varAdjItems[p];
                    if (unresolved[constraint] > 0)
                    {
                        localConstraints[localCount++] = constraint;
                    }
                }

                for (int i = 0; i < localCount; i++)
                {
                    for (int j = i + 1; j < localCount; j++)
                    {
                        int a = localConstraints[i];
                        int b = localConstraints[j];
                        if (a > b)
                        {
                            int temp = a;
                            a = b;
                            b = temp;
                        }

                        long key = ((long)a << 32) | (uint)b;
                        if (!seenPairs.Add(key))
                        {
                            continue;
                        }

                        pairCount++;
                        if (pairCount > SubsetPairLimit)
                        {
                            return changed;
                        }

                        if (TryApplySubset(a, b))
                        {
                            changed = true;
                        }
                    }
                }
            }

            return changed;
        }

        private bool TryApplySubset(int first, int second)
        {
            Span<int> a = stackalloc int[8];
            Span<int> b = stackalloc int[8];
            int countA = GetUnresolvedVars(first, a);
            int countB = GetUnresolvedVars(second, b);

            if (countA == 0 || countB == 0)
            {
                return false;
            }

            if (countA <= countB && IsSubset(a, countA, b, countB))
            {
                return ApplyDifference(a, countA, remaining[first], b, countB, remaining[second]);
            }

            if (countB < countA && IsSubset(b, countB, a, countA))
            {
                return ApplyDifference(b, countB, remaining[second], a, countA, remaining[first]);
            }

            return false;
        }

        private int GetUnresolvedVars(int constraint, Span<int> buffer)
        {
            int count = 0;
            int[] vars = constraints[constraint];
            for (int i = 0; i < vars.Length; i++)
            {
                int variable = vars[i];
                if (status[variable] == 0)
                {
                    buffer[count++] = variable;
                }
            }

            return count;
        }

        private static bool IsSubset(Span<int> small, int smallCount, Span<int> big, int bigCount)
        {
            for (int i = 0; i < smallCount; i++)
            {
                bool found = false;
                int value = small[i];
                for (int j = 0; j < bigCount; j++)
                {
                    if (big[j] == value)
                    {
                        found = true;
                        break;
                    }
                }

                if (!found)
                {
                    return false;
                }
            }

            return true;
        }

        private static bool Contains(Span<int> values, int count, int target)
        {
            for (int i = 0; i < count; i++)
            {
                if (values[i] == target)
                {
                    return true;
                }
            }

            return false;
        }

        private bool ApplyDifference(Span<int> subset, int subsetCount, int subsetMines, Span<int> superset, int supersetCount, int supersetMines)
        {
            int diffCount = supersetCount - subsetCount;
            int diffMines = supersetMines - subsetMines;
            if (diffCount <= 0 || diffMines < 0 || diffMines > diffCount)
            {
                return false;
            }

            byte forced;
            if (diffMines == 0)
            {
                forced = 1;
            }
            else if (diffMines == diffCount)
            {
                forced = 2;
            }
            else
            {
                return false;
            }

            bool changed = false;
            for (int i = 0; i < supersetCount; i++)
            {
                int variable = superset[i];
                if (!Contains(subset, subsetCount, variable) && status[variable] == 0 && Assign(variable, forced))
                {
                    changed = true;
                }
            }

            return changed;
        }

        private bool RunExactComponents(ref int exactBudget)
        {
            bool changed = false;
            var visitedVar = new bool[status.Length];
            var visitedConstraint = new bool[constraints.Count];
            var frontier = new List<int>(64);

            for (int root = 0; root < status.Length && exactBudget > 0; root++)
            {
                if (status[root] != 0 || visitedVar[root])
                {
                    continue;
                }

                frontier.Clear();
                frontier.Add(root);
                visitedVar[root] = true;

                var componentVars = new List<int>(16);
                var componentConstraints = new List<int>(16);

                for (int index = 0; index < frontier.Count; index++)
                {
                    int variable = frontier[index];
                    componentVars.Add(variable);

                    for (int p = varAdjStart[variable]; p < varAdjStart[variable + 1]; p++)
                    {
                        int constraint = varAdjItems[p];
                        if (unresolved[constraint] == 0)
                        {
                            continue;
                        }

                        if (!visitedConstraint[constraint])
                        {
                            visitedConstraint[constraint] = true;
                            componentConstraints.Add(constraint);
                        }

                        int[] vars = constraints[constraint];
                        for (int i = 0; i < vars.Length; i++)
                        {
                            int other = vars[i];
                            if (status[other] == 0 && !visitedVar[other])
                            {
                                visitedVar[other] = true;
                                frontier.Add(other);
                            }
                        }
                    }
                }

                if (componentVars.Count == 0 ||
                    componentVars.Count > ExactComponentVariableLimit ||
                    componentConstraints.Count > ExactComponentConstraintLimit ||
                    componentVars.Count > exactBudget)
                {
                    continue;
                }

                exactBudget -= componentVars.Count;

                var localIndex = new Dictionary<int, int>(componentVars.Count);
                for (int i = 0; i < componentVars.Count; i++)
                {
                    localIndex[componentVars[i]] = i;
                }

                int[][] localConstraintVars = new int[componentConstraints.Count][];
                int[] localRemaining = new int[componentConstraints.Count];
                int[] localUnresolved = new int[componentConstraints.Count];
                int[] localDegree = new int[componentVars.Count];
                bool valid = true;

                for (int c = 0; c < componentConstraints.Count; c++)
                {
                    int globalConstraint = componentConstraints[c];
                    int[] temp = new int[8];
                    int count = 0;
                    int[] vars = constraints[globalConstraint];

                    for (int i = 0; i < vars.Length; i++)
                    {
                        int globalVar = vars[i];
                        if (status[globalVar] == 0)
                        {
                            temp[count++] = localIndex[globalVar];
                        }
                    }

                    if (remaining[globalConstraint] < 0 || remaining[globalConstraint] > count)
                    {
                        valid = false;
                        break;
                    }

                    int[] localVars = new int[count];
                    for (int i = 0; i < count; i++)
                    {
                        int localVar = temp[i];
                        localVars[i] = localVar;
                        localDegree[localVar]++;
                    }

                    localConstraintVars[c] = localVars;
                    localRemaining[c] = remaining[globalConstraint];
                    localUnresolved[c] = count;
                }

                if (!valid)
                {
                    continue;
                }

                var tempAdj = new List<int>[componentVars.Count];
                for (int i = 0; i < tempAdj.Length; i++)
                {
                    tempAdj[i] = new List<int>(4);
                }

                for (int c = 0; c < localConstraintVars.Length; c++)
                {
                    int[] vars = localConstraintVars[c];
                    for (int i = 0; i < vars.Length; i++)
                    {
                        tempAdj[vars[i]].Add(c);
                    }
                }

                int[][] localVarToConstraints = new int[componentVars.Count][];
                for (int i = 0; i < componentVars.Count; i++)
                {
                    localVarToConstraints[i] = tempAdj[i].ToArray();
                }

                var exact = new ExactComponentSolver(
                    localConstraintVars,
                    localRemaining,
                    localUnresolved,
                    localVarToConstraints,
                    localDegree,
                    ExactNodeBudget);

                ExactResult result = exact.Solve();
                if (!result.Completed || result.FixedStates == null)
                {
                    continue;
                }

                byte[] fixedStates = result.FixedStates;
                for (int i = 0; i < fixedStates.Length; i++)
                {
                    byte value = fixedStates[i];
                    if (value != 0 && Assign(componentVars[i], value))
                    {
                        changed = true;
                    }
                }
            }

            return changed;
        }
    }

    private readonly struct ExactResult
    {
        public ExactResult(bool completed, byte[] fixedStates)
        {
            Completed = completed;
            FixedStates = fixedStates;
        }

        public bool Completed { get; }
        public byte[] FixedStates { get; }
    }

    private sealed class ExactComponentSolver
    {
        private readonly int[][] constraintVars;
        private readonly int[] initialRemaining;
        private readonly int[] initialUnresolved;
        private readonly int[][] varToConstraints;
        private readonly int[] varDegree;
        private readonly int nodeBudget;

        private int nodes;
        private bool overBudget;
        private bool stopEarly;
        private bool hasSolution;
        private int fixedCandidateCount;
        private byte[] intersection;

        public ExactComponentSolver(
            int[][] constraintVars,
            int[] initialRemaining,
            int[] initialUnresolved,
            int[][] varToConstraints,
            int[] varDegree,
            int nodeBudget)
        {
            this.constraintVars = constraintVars;
            this.initialRemaining = initialRemaining;
            this.initialUnresolved = initialUnresolved;
            this.varToConstraints = varToConstraints;
            this.varDegree = varDegree;
            this.nodeBudget = nodeBudget;
        }

        public ExactResult Solve()
        {
            Search(
                new byte[varToConstraints.Length],
                (int[])initialRemaining.Clone(),
                (int[])initialUnresolved.Clone());

            if (!hasSolution || overBudget)
            {
                return new ExactResult(false, null);
            }

            return new ExactResult(true, intersection);
        }

        private void Search(byte[] assignment, int[] remaining, int[] unresolved)
        {
            if (overBudget || stopEarly)
            {
                return;
            }

            nodes++;
            if (nodes > nodeBudget)
            {
                overBudget = true;
                return;
            }

            if (!Propagate(assignment, remaining, unresolved))
            {
                return;
            }

            int branchVariable = SelectBranchVariable(assignment, unresolved);
            if (branchVariable < 0)
            {
                RecordSolution(assignment);
                return;
            }

            bool preferMine = PreferMine(branchVariable, remaining, unresolved);

            if (preferMine)
            {
                Branch(assignment, remaining, unresolved, branchVariable, 2);
                Branch(assignment, remaining, unresolved, branchVariable, 1);
            }
            else
            {
                Branch(assignment, remaining, unresolved, branchVariable, 1);
                Branch(assignment, remaining, unresolved, branchVariable, 2);
            }
        }

        private void Branch(byte[] assignment, int[] remaining, int[] unresolved, int variable, byte value)
        {
            if (overBudget || stopEarly)
            {
                return;
            }

            byte[] nextAssignment = (byte[])assignment.Clone();
            int[] nextRemaining = (int[])remaining.Clone();
            int[] nextUnresolved = (int[])unresolved.Clone();

            if (Apply(nextAssignment, nextRemaining, nextUnresolved, variable, value))
            {
                Search(nextAssignment, nextRemaining, nextUnresolved);
            }
        }

        private bool Apply(byte[] assignment, int[] remaining, int[] unresolved, int variable, byte value)
        {
            byte current = assignment[variable];
            if (current != 0)
            {
                return current == value;
            }

            assignment[variable] = value;
            int[] affectedConstraints = varToConstraints[variable];

            for (int i = 0; i < affectedConstraints.Length; i++)
            {
                int constraint = affectedConstraints[i];
                unresolved[constraint]--;
                if (value == 2)
                {
                    remaining[constraint]--;
                }

                if (remaining[constraint] < 0 || remaining[constraint] > unresolved[constraint])
                {
                    return false;
                }
            }

            return true;
        }

        private bool Propagate(byte[] assignment, int[] remaining, int[] unresolved)
        {
            bool changed;
            do
            {
                changed = false;

                for (int constraint = 0; constraint < constraintVars.Length; constraint++)
                {
                    int minesLeft = remaining[constraint];
                    int varsLeft = unresolved[constraint];
                    if (minesLeft < 0 || minesLeft > varsLeft)
                    {
                        return false;
                    }

                    if (varsLeft == 0)
                    {
                        continue;
                    }

                    if (minesLeft == 0 || minesLeft == varsLeft)
                    {
                        byte forced = minesLeft == 0 ? (byte)1 : (byte)2;
                        int[] vars = constraintVars[constraint];
                        for (int i = 0; i < vars.Length; i++)
                        {
                            int variable = vars[i];
                            if (assignment[variable] == 0)
                            {
                                if (!Apply(assignment, remaining, unresolved, variable, forced))
                                {
                                    return false;
                                }

                                changed = true;
                            }
                        }
                    }
                }
            }
            while (changed);

            return true;
        }

        private int SelectBranchVariable(byte[] assignment, int[] unresolved)
        {
            int bestConstraint = -1;
            int bestConstraintSize = int.MaxValue;

            for (int constraint = 0; constraint < constraintVars.Length; constraint++)
            {
                int varsLeft = unresolved[constraint];
                if (varsLeft > 1 && varsLeft < bestConstraintSize)
                {
                    bestConstraintSize = varsLeft;
                    bestConstraint = constraint;
                }
            }

            if (bestConstraint >= 0)
            {
                int bestVariable = -1;
                int bestDegree = -1;
                int[] vars = constraintVars[bestConstraint];
                for (int i = 0; i < vars.Length; i++)
                {
                    int variable = vars[i];
                    if (assignment[variable] == 0 && varDegree[variable] > bestDegree)
                    {
                        bestDegree = varDegree[variable];
                        bestVariable = variable;
                    }
                }

                if (bestVariable >= 0)
                {
                    return bestVariable;
                }
            }

            for (int variable = 0; variable < assignment.Length; variable++)
            {
                if (assignment[variable] == 0)
                {
                    return variable;
                }
            }

            return -1;
        }

        private bool PreferMine(int variable, int[] remaining, int[] unresolved)
        {
            int bestBias = int.MinValue;
            bool preferMine = false;
            int[] affectedConstraints = varToConstraints[variable];

            for (int i = 0; i < affectedConstraints.Length; i++)
            {
                int constraint = affectedConstraints[i];
                int varsLeft = unresolved[constraint];
                if (varsLeft <= 0)
                {
                    continue;
                }

                int minesLeft = remaining[constraint];
                int bias = Math.Abs((minesLeft << 1) - varsLeft);
                if (bias > bestBias)
                {
                    bestBias = bias;
                    preferMine = (minesLeft << 1) > varsLeft;
                }
            }

            return preferMine;
        }

        private void RecordSolution(byte[] assignment)
        {
            if (!hasSolution)
            {
                hasSolution = true;
                intersection = (byte[])assignment.Clone();
                fixedCandidateCount = 0;
                for (int i = 0; i < intersection.Length; i++)
                {
                    if (intersection[i] != 0)
                    {
                        fixedCandidateCount++;
                    }
                }

                return;
            }

            for (int i = 0; i < intersection.Length; i++)
            {
                if (intersection[i] != 0 && intersection[i] != assignment[i])
                {
                    intersection[i] = 0;
                    fixedCandidateCount--;
                }
            }

            if (fixedCandidateCount == 0)
            {
                stopEarly = true;
            }
        }
    }
