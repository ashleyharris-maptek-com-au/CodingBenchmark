#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>

#include <set>

#include <utility>
#include <vector>

struct InputRect {
    int w = 0;
    int h = 0;
    int index = 0;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct FreeRect {
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
    int board = 0;
};

struct TrialConfig {
    int orderMode = 0;
    int splitMode = 0;
};

class DimensionIndex {
public:
    explicit DimensionIndex(int limit)
        : limit_(limit), buckets_(limit + 1), seg_(4 * (limit + 1), 0) {}

    void insert(int primary, int secondary, int id) {
        buckets_[primary].insert({secondary, id});
        update(1, 1, limit_, primary, buckets_[primary].rbegin()->first);
    }

    void erase(int primary, int secondary, int id) {
        auto it = buckets_[primary].find({secondary, id});
        if (it != buckets_[primary].end()) {
            buckets_[primary].erase(it);
            const int nextMax = buckets_[primary].empty() ? 0 : buckets_[primary].rbegin()->first;
            update(1, 1, limit_, primary, nextMax);
        }
    }

    int firstFit(int minPrimary, int minSecondary) const {
        if (minPrimary > limit_) {
            return -1;
        }
        return firstFitRec(1, 1, limit_, minPrimary, minSecondary);
    }

    const std::multiset<std::pair<int, int>>& bucket(int primary) const {
        return buckets_[primary];
    }

private:
    int limit_ = 0;
    std::vector<std::multiset<std::pair<int, int>>> buckets_;
    std::vector<int> seg_;

    void update(int node, int left, int right, int pos, int value) {
        if (left == right) {
            seg_[node] = value;
            return;
        }
        const int mid = (left + right) / 2;
        if (pos <= mid) {
            update(node * 2, left, mid, pos, value);
        } else {
            update(node * 2 + 1, mid + 1, right, pos, value);
        }
        seg_[node] = std::max(seg_[node * 2], seg_[node * 2 + 1]);
    }

    int firstFitRec(int node, int left, int right, int queryLeft, int minSecondary) const {
        if (right < queryLeft || seg_[node] < minSecondary) {
            return -1;
        }
        if (left == right) {
            return left;
        }
        const int mid = (left + right) / 2;
        const int leftResult = firstFitRec(node * 2, left, mid, queryLeft, minSecondary);
        if (leftResult != -1) {
            return leftResult;
        }
        return firstFitRec(node * 2 + 1, mid + 1, right, queryLeft, minSecondary);
    }
};

class GuillotinePacker {
public:
    struct Result {
        bool success = false;
        int boards = 0;
        std::vector<Placement> placements;
    };

    GuillotinePacker(int boardW, int boardH, const std::vector<InputRect>& rects, int splitMode)
        : boardW_(boardW),
          boardH_(boardH),
          rects_(rects),
          splitMode_(splitMode),
          widthIndex_(boardW),
          heightIndex_(boardH),
          placements_(rects.size()) {
        freeRects_.reserve(rects.size() * 2 + 16);
    }

    Result pack(const std::vector<int>& order) {
        if (boardW_ <= 0 || boardH_ <= 0) {
            return {};
        }

        for (int rectId : order) {
            if (!placeRect(rects_[rectId])) {
                return {};
            }
        }

        return {true, boardCount_, placements_};
    }

private:
    struct Candidate {
        int freeId = -1;
        int rw = 0;
        int rh = 0;
        int rotated = 0;
        int shortFit = std::numeric_limits<int>::max();
        int longFit = std::numeric_limits<int>::max();
        long long waste = (std::numeric_limits<long long>::max)();

        bool valid() const {
            return freeId >= 0;
        }
    };

    int boardW_ = 0;
    int boardH_ = 0;
    const std::vector<InputRect>& rects_;
    int splitMode_ = 0;
    int boardCount_ = 0;
    DimensionIndex widthIndex_;
    DimensionIndex heightIndex_;
    std::vector<FreeRect> freeRects_;
    std::vector<Placement> placements_;

    static bool betterCandidate(const Candidate& lhs, const Candidate& rhs) {
        if (!lhs.valid()) {
            return false;
        }
        if (!rhs.valid()) {
            return true;
        }
        return std::tie(lhs.shortFit, lhs.longFit, lhs.waste, lhs.rotated, lhs.freeId) <
               std::tie(rhs.shortFit, rhs.longFit, rhs.waste, rhs.rotated, rhs.freeId);
    }

    Candidate buildCandidate(int freeId, int rw, int rh, int rotated) const {
        const FreeRect& fr = freeRects_[freeId];
        Candidate candidate;
        candidate.freeId = freeId;
        candidate.rw = rw;
        candidate.rh = rh;
        candidate.rotated = rotated;
        const int dw = fr.w - rw;
        const int dh = fr.h - rh;
        candidate.shortFit = std::min(dw, dh);
        candidate.longFit = std::max(dw, dh);
        candidate.waste = 1LL * fr.w * fr.h - 1LL * rw * rh;
        return candidate;
    }

    Candidate queryOrientation(int rw, int rh, int rotated) const {
        Candidate best;

        const int widthPos = widthIndex_.firstFit(rw, rh);
        if (widthPos != -1) {
            const auto& bucket = widthIndex_.bucket(widthPos);
            const auto it = bucket.lower_bound({rh, -1});
            if (it != bucket.end()) {
                const Candidate candidate = buildCandidate(it->second, rw, rh, rotated);
                if (betterCandidate(candidate, best)) {
                    best = candidate;
                }
            }
        }

        const int heightPos = heightIndex_.firstFit(rh, rw);
        if (heightPos != -1) {
            const auto& bucket = heightIndex_.bucket(heightPos);
            const auto it = bucket.lower_bound({rw, -1});
            if (it != bucket.end()) {
                const Candidate candidate = buildCandidate(it->second, rw, rh, rotated);
                if (betterCandidate(candidate, best)) {
                    best = candidate;
                }
            }
        }

        return best;
    }

    Candidate bestExistingCandidate(const InputRect& rect) const {
        Candidate best = queryOrientation(rect.w, rect.h, 0);
        if (rect.w != rect.h) {
            const Candidate rotated = queryOrientation(rect.h, rect.w, 1);
            if (betterCandidate(rotated, best)) {
                best = rotated;
            }
        }
        return best;
    }

    int addFreeRect(int x, int y, int w, int h, int board) {
        if (w <= 0 || h <= 0) {
            return -1;
        }
        const int id = static_cast<int>(freeRects_.size());
        freeRects_.push_back({x, y, w, h, board});
        widthIndex_.insert(w, h, id);
        heightIndex_.insert(h, w, id);
        return id;
    }

    int openBoard() {
        const int board = boardCount_++;
        return addFreeRect(0, 0, boardW_, boardH_, board);
    }

    void removeFreeRect(int id) {
        const FreeRect& fr = freeRects_[id];
        widthIndex_.erase(fr.w, fr.h, id);
        heightIndex_.erase(fr.h, fr.w, id);
    }

    void splitFreeRect(const FreeRect& fr, int rw, int rh) {
        const int dw = fr.w - rw;
        const int dh = fr.h - rh;

        const bool splitVerticalFirst = (splitMode_ == 0) ? (dw <= dh) : (dw > dh);
        if (splitVerticalFirst) {
            addFreeRect(fr.x + rw, fr.y, dw, fr.h, fr.board);
            addFreeRect(fr.x, fr.y + rh, rw, dh, fr.board);
        } else {
            addFreeRect(fr.x + rw, fr.y, dw, rh, fr.board);
            addFreeRect(fr.x, fr.y + rh, fr.w, dh, fr.board);
        }
    }

    bool placeRect(const InputRect& rect) {
        Candidate best = bestExistingCandidate(rect);

        if (!best.valid()) {
            const bool fitsNormal = rect.w <= boardW_ && rect.h <= boardH_;
            const bool fitsRotated = rect.h <= boardW_ && rect.w <= boardH_;
            if (!fitsNormal && !fitsRotated) {
                return false;
            }

            const int freeId = openBoard();
            if (fitsNormal) {
                best = buildCandidate(freeId, rect.w, rect.h, 0);
            }
            if (rect.w != rect.h && fitsRotated) {
                const Candidate rotated = buildCandidate(freeId, rect.h, rect.w, 1);
                if (betterCandidate(rotated, best)) {
                    best = rotated;
                }
            }
        }

        const FreeRect placedIn = freeRects_[best.freeId];
        removeFreeRect(best.freeId);
        placements_[rect.index] = {placedIn.board, placedIn.x, placedIn.y, best.rotated};
        splitFreeRect(placedIn, best.rw, best.rh);
        return true;
    }
};

static long long areaOf(const InputRect& rect) {
    return 1LL * rect.w * rect.h;
}

static int maxSideOf(const InputRect& rect) {
    return std::max(rect.w, rect.h);
}

static int minSideOf(const InputRect& rect) {
    return std::min(rect.w, rect.h);
}

static int perimeterOf(const InputRect& rect) {
    return rect.w + rect.h;
}

static int skewOf(const InputRect& rect) {
    return std::abs(rect.w - rect.h);
}

static std::vector<int> buildOrder(const std::vector<InputRect>& rects, int mode) {
    std::vector<int> order(rects.size());
    std::iota(order.begin(), order.end(), 0);

    auto better = [&](int lhsId, int rhsId) {
        const InputRect& lhs = rects[lhsId];
        const InputRect& rhs = rects[rhsId];

        switch (mode) {
        case 0:
            if (maxSideOf(lhs) != maxSideOf(rhs)) return maxSideOf(lhs) > maxSideOf(rhs);
            if (minSideOf(lhs) != minSideOf(rhs)) return minSideOf(lhs) > minSideOf(rhs);
            if (areaOf(lhs) != areaOf(rhs)) return areaOf(lhs) > areaOf(rhs);
            break;
        case 1:
            if (areaOf(lhs) != areaOf(rhs)) return areaOf(lhs) > areaOf(rhs);
            if (maxSideOf(lhs) != maxSideOf(rhs)) return maxSideOf(lhs) > maxSideOf(rhs);
            if (minSideOf(lhs) != minSideOf(rhs)) return minSideOf(lhs) > minSideOf(rhs);
            break;
        case 2:
            if (perimeterOf(lhs) != perimeterOf(rhs)) return perimeterOf(lhs) > perimeterOf(rhs);
            if (areaOf(lhs) != areaOf(rhs)) return areaOf(lhs) > areaOf(rhs);
            if (maxSideOf(lhs) != maxSideOf(rhs)) return maxSideOf(lhs) > maxSideOf(rhs);
            break;
        case 3:
            if (minSideOf(lhs) != minSideOf(rhs)) return minSideOf(lhs) > minSideOf(rhs);
            if (maxSideOf(lhs) != maxSideOf(rhs)) return maxSideOf(lhs) > maxSideOf(rhs);
            if (areaOf(lhs) != areaOf(rhs)) return areaOf(lhs) > areaOf(rhs);
            break;
        default:
            if (skewOf(lhs) != skewOf(rhs)) return skewOf(lhs) > skewOf(rhs);
            if (maxSideOf(lhs) != maxSideOf(rhs)) return maxSideOf(lhs) > maxSideOf(rhs);
            if (areaOf(lhs) != areaOf(rhs)) return areaOf(lhs) > areaOf(rhs);
            break;
        }

        return lhs.index < rhs.index;
    };

    std::sort(order.begin(), order.end(), better);
    return order;
}

static std::vector<TrialConfig> trialConfigsFor(int n) {
    if (n <= 200) {
        return {
            {0, 0}, {1, 0}, {2, 0}, {3, 0},
            {0, 1}, {1, 1}, {4, 0}, {4, 1}
        };
    }
    if (n <= 2000) {
        return {
            {0, 0}, {1, 0}, {2, 0}, {0, 1}, {3, 0}
        };
    }
    if (n <= 20000) {
        return {
            {0, 0}, {1, 0}, {2, 0}, {0, 1}
        };
    }
    return {
        {0, 0}, {1, 0}, {0, 1}
    };
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n = 0;
    int boardW = 0;
    int boardH = 0;
    if (!(std::cin >> n >> boardW >> boardH)) {
        return 0;
    }

    std::vector<InputRect> rects(n);
    long long totalArea = 0;
    for (int i = 0; i < n; ++i) {
        std::cin >> rects[i].w >> rects[i].h;
        rects[i].index = i;
        totalArea += areaOf(rects[i]);
    }

    const long long boardArea = 1LL * boardW * boardH;
    if (n == 0) {
        std::cout << 0 << '\n';
        return 0;
    }
    if (boardW <= 0 || boardH <= 0 || boardArea <= 0) {
        std::cout << -1 << '\n';
        return 0;
    }

    for (const auto& rect : rects) {
        const bool fitsNormal = rect.w <= boardW && rect.h <= boardH;
        const bool fitsRotated = rect.h <= boardW && rect.w <= boardH;
        if (!fitsNormal && !fitsRotated) {
            std::cout << -1 << '\n';
            return 0;
        }
    }

    const long long areaLowerBound = (totalArea + boardArea - 1) / boardArea;
    GuillotinePacker::Result bestResult;
    bool haveBest = false;

    const std::vector<TrialConfig> configs = trialConfigsFor(n);
    for (const TrialConfig& config : configs) {
        const std::vector<int> order = buildOrder(rects, config.orderMode);
        GuillotinePacker packer(boardW, boardH, rects, config.splitMode);
        GuillotinePacker::Result result = packer.pack(order);
        if (!result.success) {
            continue;
        }
        if (!haveBest || result.boards < bestResult.boards) {
            bestResult = std::move(result);
            haveBest = true;
            if (bestResult.boards == areaLowerBound) {
                break;
            }
        }
    }

    if (!haveBest) {
        std::cout << -1 << '\n';
        return 0;
    }

    std::cout << bestResult.boards << '\n';
    for (const Placement& placement : bestResult.placements) {
        std::cout << placement.board << ' '
                  << placement.x << ' '
                  << placement.y << ' '
                  << placement.rotated << '\n';
    }

    return 0;
}
