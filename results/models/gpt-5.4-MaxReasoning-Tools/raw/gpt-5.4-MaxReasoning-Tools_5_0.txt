use std::collections::HashSet;
use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone)]
struct GridGraph {
    width: usize,
    height: usize,
    require_cycle: bool,
    root: usize,
    id_at: Vec<usize>,
    positions: Vec<(usize, usize)>,
    adj: Vec<Vec<usize>>,
    color: Vec<u8>,
}

enum SearchStatus {
    Found,
    Failed,
    Timeout,
}

struct LargeSearchCtx {
    visited: Vec<bool>,
    path: Vec<usize>,
    nodes: u64,
    node_budget: u64,
    deadline: Instant,
}

impl GridGraph {
    fn from_input(input: &str) -> Option<Self> {
        let mut it = input.split_whitespace();
        let width: usize = it.next()?.parse().ok()?;
        let height: usize = it.next()?.parse().ok()?;
        let obstacle_count: usize = it.next()?.parse().ok()?;
        let require_cycle = it.next()?.parse::<u32>().ok()? != 0;
        if width == 0 || height == 0 {
            return None;
        }

        let mut blocked = vec![false; width * height];
        for _ in 0..obstacle_count {
            let x: usize = it.next()?.parse().ok()?;
            let y: usize = it.next()?.parse().ok()?;
            if x >= width || y >= height {
                return None;
            }
            blocked[y * width + x] = true;
        }
        if blocked[0] {
            return None;
        }

        let mut id_at = vec![usize::MAX; width * height];
        let mut positions = Vec::new();
        for y in 0..height {
            for x in 0..width {
                let idx = y * width + x;
                if !blocked[idx] {
                    id_at[idx] = positions.len();
                    positions.push((x, y));
                }
            }
        }
        if positions.is_empty() {
            return None;
        }

        let mut adj = vec![Vec::new(); positions.len()];
        let mut color = vec![0u8; positions.len()];
        for (id, &(x, y)) in positions.iter().enumerate() {
            color[id] = ((x + y) & 1) as u8;
            if x > 0 {
                let left = id_at[y * width + (x - 1)];
                if left != usize::MAX {
                    adj[id].push(left);
                }
            }
            if x + 1 < width {
                let right = id_at[y * width + (x + 1)];
                if right != usize::MAX {
                    adj[id].push(right);
                }
            }
            if y > 0 {
                let up = id_at[(y - 1) * width + x];
                if up != usize::MAX {
                    adj[id].push(up);
                }
            }
            if y + 1 < height {
                let down = id_at[(y + 1) * width + x];
                if down != usize::MAX {
                    adj[id].push(down);
                }
            }
        }

        Some(Self {
            width,
            height,
            require_cycle,
            root: id_at[0],
            id_at,
            positions,
            adj,
            color,
        })
    }

    fn n(&self) -> usize {
        self.positions.len()
    }

    fn coord_id(&self, x: usize, y: usize) -> Option<usize> {
        if x >= self.width || y >= self.height {
            return None;
        }
        let id = self.id_at[y * self.width + x];
        if id == usize::MAX {
            None
        } else {
            Some(id)
        }
    }

    fn is_empty_rectangle(&self) -> bool {
        self.n() == self.width * self.height
    }

    fn is_adjacent(&self, a: usize, b: usize) -> bool {
        self.adj[a].iter().any(|&v| v == b)
    }

    fn validate_path(&self, path: &[usize]) -> bool {
        if path.len() != self.n() || path.first().copied() != Some(self.root) {
            return false;
        }
        let mut seen = vec![false; self.n()];
        for (i, &v) in path.iter().enumerate() {
            if seen[v] {
                return false;
            }
            seen[v] = true;
            if i > 0 && !self.is_adjacent(path[i - 1], v) {
                return false;
            }
        }
        if self.require_cycle {
            if let Some(&last) = path.last() {
                self.is_adjacent(last, self.root)
            } else {
                false
            }
        } else {
            true
        }
    }

    fn connected_all(&self) -> bool {
        let mut stack = vec![self.root];
        let mut seen = vec![false; self.n()];
        seen[self.root] = true;
        let mut count = 1usize;
        while let Some(u) = stack.pop() {
            for &v in &self.adj[u] {
                if !seen[v] {
                    seen[v] = true;
                    count += 1;
                    stack.push(v);
                }
            }
        }
        count == self.n()
    }

    fn basic_feasible(&self) -> bool {
        let n = self.n();
        if n == 0 || !self.connected_all() {
            return false;
        }

        let mut black = 0usize;
        let mut white = 0usize;
        let mut degree_one_non_root = 0usize;
        for v in 0..n {
            if self.color[v] == 0 {
                black += 1;
            } else {
                white += 1;
            }
            let deg = self.adj[v].len();
            if n > 1 && deg == 0 {
                return false;
            }
            if v != self.root && deg == 1 {
                degree_one_non_root += 1;
            }
        }

        if self.require_cycle {
            if n < 4 || black != white {
                return false;
            }
            if self.adj.iter().any(|nbrs| nbrs.len() < 2) {
                return false;
            }
            true
        } else {
            if n % 2 == 0 {
                if black != white {
                    return false;
                }
            } else if self.color[self.root] == 0 {
                if black != white + 1 {
                    return false;
                }
            } else if white != black + 1 {
                return false;
            }

            degree_one_non_root <= 1
        }
    }

    fn solve(&self) -> Option<Vec<usize>> {
        if !self.basic_feasible() {
            return None;
        }
        if self.n() == 1 {
            return (!self.require_cycle).then_some(vec![self.root]);
        }

        if let Some(path) = self.solve_empty_rectangle() {
            return Some(path);
        }
        if let Some(path) = self.solve_degree_le_two() {
            return Some(path);
        }

        let deadline = Instant::now() + Duration::from_secs(295);

        if self.n() <= 64 {
            match self.solve_exact_small(deadline) {
                ExactOutcome::Found(path) => return Some(path),
                ExactOutcome::Impossible => return None,
                ExactOutcome::Timeout => {}
            }
        }

        if let Some(path) = self.solve_greedy(4) {
            return Some(path);
        }

        if let Some(path) = self.solve_guided_backtracking(deadline, 2_000_000) {
            return Some(path);
        }

        None
    }

    fn solve_empty_rectangle(&self) -> Option<Vec<usize>> {
        if !self.is_empty_rectangle() {
            return None;
        }
        if self.require_cycle {
            let coords = if self.height % 2 == 0 {
                self.rectangle_cycle_even_height(self.width, self.height)?
            } else if self.width % 2 == 0 {
                let transposed = self.rectangle_cycle_even_height(self.height, self.width)?;
                transposed.into_iter().map(|(x, y)| (y, x)).collect()
            } else {
                return None;
            };
            let path = self.coords_to_path(&coords)?;
            self.validate_path(&path).then_some(path)
        } else {
            let mut coords = Vec::with_capacity(self.n());
            for y in 0..self.height {
                if y % 2 == 0 {
                    for x in 0..self.width {
                        coords.push((x, y));
                    }
                } else {
                    for x in (0..self.width).rev() {
                        coords.push((x, y));
                    }
                }
            }
            let path = self.coords_to_path(&coords)?;
            self.validate_path(&path).then_some(path)
        }
    }

    fn rectangle_cycle_even_height(
        &self,
        width: usize,
        height: usize,
    ) -> Option<Vec<(usize, usize)>> {
        if width < 2 || height < 2 || height % 2 != 0 {
            return None;
        }
        let mut coords = Vec::with_capacity(width * height);
        for x in 0..width {
            coords.push((x, 0));
        }
        for y in 1..height {
            if y % 2 == 1 {
                for x in (1..width).rev() {
                    coords.push((x, y));
                }
            } else {
                for x in 1..width {
                    coords.push((x, y));
                }
            }
        }
        for y in (1..height).rev() {
            coords.push((0, y));
        }
        Some(coords)
    }

    fn coords_to_path(&self, coords: &[(usize, usize)]) -> Option<Vec<usize>> {
        let mut path = Vec::with_capacity(coords.len());
        for &(x, y) in coords {
            path.push(self.coord_id(x, y)?);
        }
        Some(path)
    }

    fn solve_degree_le_two(&self) -> Option<Vec<usize>> {
        if self.adj.iter().any(|nbrs| nbrs.len() > 2) {
            return None;
        }

        let degree_one: Vec<usize> = (0..self.n()).filter(|&v| self.adj[v].len() == 1).collect();
        if self.require_cycle {
            if degree_one.is_empty() && self.adj.iter().all(|nbrs| nbrs.len() == 2) {
                let path = self.traverse_cycle(self.adj[self.root][0])?;
                return self.validate_path(&path).then_some(path);
            }
            return None;
        }

        if self.n() == 1 {
            return Some(vec![self.root]);
        }
        if degree_one.len() == 2 {
            if !degree_one.contains(&self.root) {
                return None;
            }
            let path = self.traverse_path(self.root)?;
            return self.validate_path(&path).then_some(path);
        }
        if degree_one.is_empty() && self.adj.iter().all(|nbrs| nbrs.len() == 2) {
            let path = self.traverse_cycle(self.adj[self.root][0])?;
            return self.validate_path(&path).then_some(path);
        }
        None
    }

    fn traverse_path(&self, start: usize) -> Option<Vec<usize>> {
        let mut visited = vec![false; self.n()];
        let mut path = Vec::with_capacity(self.n());
        let mut prev = usize::MAX;
        let mut cur = start;
        loop {
            if visited[cur] {
                return None;
            }
            visited[cur] = true;
            path.push(cur);
            let nexts: Vec<usize> = self.adj[cur]
                .iter()
                .copied()
                .filter(|&v| v != prev && !visited[v])
                .collect();
            if nexts.is_empty() {
                break;
            }
            if nexts.len() != 1 {
                return None;
            }
            prev = cur;
            cur = nexts[0];
        }
        (path.len() == self.n()).then_some(path)
    }

    fn traverse_cycle(&self, first_neighbor: usize) -> Option<Vec<usize>> {
        let mut visited = vec![false; self.n()];
        let mut path = Vec::with_capacity(self.n());
        let mut prev = self.root;
        let mut cur = first_neighbor;
        visited[self.root] = true;
        path.push(self.root);
        while path.len() < self.n() {
            if visited[cur] {
                return None;
            }
            visited[cur] = true;
            path.push(cur);
            if path.len() == self.n() {
                break;
            }
            let nexts: Vec<usize> = self.adj[cur]
                .iter()
                .copied()
                .filter(|&v| v != prev && !visited[v])
                .collect();
            if nexts.len() != 1 {
                return None;
            }
            prev = cur;
            cur = nexts[0];
        }
        Some(path)
    }

    fn solve_exact_small(&self, deadline: Instant) -> ExactOutcome {
        if self.require_cycle {
            self.solve_exact_small_cycle(deadline)
        } else {
            self.solve_exact_small_path(deadline)
        }
    }

    fn solve_exact_small_path(&self, deadline: Instant) -> ExactOutcome {
        let mut dead = HashSet::new();
        let mut path = vec![self.root];
        let visited = 1u128 << self.root;
        let mut nodes = 0u64;
        match self.dfs_small(
            visited,
            self.root,
            None,
            &mut path,
            &mut dead,
            deadline,
            &mut nodes,
        ) {
            SearchStatus::Found => ExactOutcome::Found(path),
            SearchStatus::Failed => ExactOutcome::Impossible,
            SearchStatus::Timeout => ExactOutcome::Timeout,
        }
    }

    fn solve_exact_small_cycle(&self, deadline: Instant) -> ExactOutcome {
        let mut neighbors = self.adj[self.root].clone();
        neighbors.sort_by_key(|&v| self.small_candidate_score(1u128 << self.root, v, Some(self.root), 0));
        let mut timed_out = false;
        for first in neighbors {
            let mut dead = HashSet::new();
            let mut path = vec![self.root, first];
            let visited = (1u128 << self.root) | (1u128 << first);
            let mut nodes = 0u64;
            match self.dfs_small(
                visited,
                first,
                Some(self.root),
                &mut path,
                &mut dead,
                deadline,
                &mut nodes,
            ) {
                SearchStatus::Found => return ExactOutcome::Found(path),
                SearchStatus::Failed => {}
                SearchStatus::Timeout => timed_out = true,
            }
        }
        if timed_out {
            ExactOutcome::Timeout
        } else {
            ExactOutcome::Impossible
        }
    }

    fn dfs_small(
        &self,
        visited: u128,
        current: usize,
        target_fixed: Option<usize>,
        path: &mut Vec<usize>,
        dead: &mut HashSet<(u128, usize)>,
        deadline: Instant,
        nodes: &mut u64,
    ) -> SearchStatus {
        *nodes += 1;
        if (*nodes & 4095) == 0 && Instant::now() >= deadline {
            return SearchStatus::Timeout;
        }
        if path.len() == self.n() {
            return match target_fixed {
                None => SearchStatus::Found,
                Some(t) => {
                    if self.is_adjacent(current, t) {
                        SearchStatus::Found
                    } else {
                        SearchStatus::Failed
                    }
                }
            };
        }
        if !self.promising_small(visited, current, target_fixed) {
            return SearchStatus::Failed;
        }
        if dead.contains(&(visited, current)) {
            return SearchStatus::Failed;
        }

        let mut candidates: Vec<usize> = self.adj[current]
            .iter()
            .copied()
            .filter(|&v| (visited & (1u128 << v)) == 0)
            .collect();
        candidates.sort_by_key(|&v| self.small_candidate_score(visited, v, target_fixed, 0));

        for next in candidates {
            let next_visited = visited | (1u128 << next);
            path.push(next);
            match self.dfs_small(next_visited, next, target_fixed, path, dead, deadline, nodes) {
                SearchStatus::Found => return SearchStatus::Found,
                SearchStatus::Failed => {}
                SearchStatus::Timeout => {
                    path.pop();
                    return SearchStatus::Timeout;
                }
            }
            path.pop();
        }

        dead.insert((visited, current));
        SearchStatus::Failed
    }

    fn promising_small(&self, visited: u128, current: usize, target_fixed: Option<usize>) -> bool {
        let alive = self.alive_mask_small(visited, current, target_fixed);
        if !self.connected_mask(alive) {
            return false;
        }

        let without_current = alive & !(1u128 << current);
        if without_current != 0 && !self.connected_mask(without_current) {
            return false;
        }
        if let Some(t) = target_fixed {
            let without_target = alive & !(1u128 << t);
            if without_target != 0 && !self.connected_mask(without_target) {
                return false;
            }
        }

        let mut black = 0usize;
        let mut white = 0usize;
        let mut path_leaf_count = 0usize;
        let mut current_deg = 0usize;
        let target = target_fixed;
        for v in 0..self.n() {
            let bit = 1u128 << v;
            if (alive & bit) == 0 {
                continue;
            }
            if self.color[v] == 0 {
                black += 1;
            } else {
                white += 1;
            }
            let deg = self.alive_degree_small(alive, v);
            if v == current {
                current_deg = deg;
                continue;
            }
            if Some(v) == target {
                if deg == 0 {
                    return false;
                }
                continue;
            }
            if target.is_some() {
                if deg < 2 {
                    return false;
                }
            } else {
                if deg == 0 {
                    return false;
                }
                if deg == 1 {
                    path_leaf_count += 1;
                    if path_leaf_count > 1 {
                        return false;
                    }
                }
            }
        }
        if current_deg == 0 && alive.count_ones() > 1 {
            return false;
        }

        let alive_count = alive.count_ones() as usize;
        match target_fixed {
            None => {
                if alive_count % 2 == 0 {
                    black == white
                } else if self.color[current] == 0 {
                    black == white + 1
                } else {
                    white == black + 1
                }
            }
            Some(t) => {
                if alive_count % 2 == 0 {
                    self.color[current] != self.color[t] && black == white
                } else if self.color[current] != self.color[t] {
                    false
                } else if self.color[current] == 0 {
                    black == white + 1
                } else {
                    white == black + 1
                }
            }
        }
    }

    fn alive_mask_small(&self, visited: u128, current: usize, target_fixed: Option<usize>) -> u128 {
        let mut alive = 0u128;
        for v in 0..self.n() {
            let bit = 1u128 << v;
            if v == current || Some(v) == target_fixed || (visited & bit) == 0 {
                alive |= bit;
            }
        }
        alive
    }

    fn alive_degree_small(&self, alive: u128, v: usize) -> usize {
        self.adj[v]
            .iter()
            .filter(|&&to| (alive & (1u128 << to)) != 0)
            .count()
    }

    fn connected_mask(&self, alive: u128) -> bool {
        if alive == 0 {
            return true;
        }
        let start = alive.trailing_zeros() as usize;
        let mut seen = 0u128;
        let mut stack = vec![start];
        seen |= 1u128 << start;
        while let Some(u) = stack.pop() {
            for &v in &self.adj[u] {
                let bit = 1u128 << v;
                if (alive & bit) != 0 && (seen & bit) == 0 {
                    seen |= bit;
                    stack.push(v);
                }
            }
        }
        seen == alive
    }

    fn small_candidate_score(
        &self,
        visited: u128,
        next: usize,
        target_fixed: Option<usize>,
        mode: usize,
    ) -> (usize, usize, usize, usize, usize) {
        let mut alive = 0u128;
        for v in 0..self.n() {
            let bit = 1u128 << v;
            if v == next || Some(v) == target_fixed || ((visited & bit) == 0 && v != next) {
                alive |= bit;
            }
        }
        let mut leafs = 0usize;
        for v in 0..self.n() {
            let bit = 1u128 << v;
            if (alive & bit) == 0 || v == next || Some(v) == target_fixed {
                continue;
            }
            if self.alive_degree_small(alive, v) == 1 {
                leafs += 1;
            }
        }
        let next_deg = self.alive_degree_small(alive, next);
        let (x, y) = self.positions[next];
        let spatial = self.spatial_bias(x, y, mode);
        let target_bias = target_fixed
            .map(|t| if self.is_adjacent(next, t) { 0 } else { 1 })
            .unwrap_or(0);
        (leafs, next_deg, target_bias, self.adj[next].len(), spatial)
    }

    fn solve_greedy(&self, attempts: usize) -> Option<Vec<usize>> {
        if self.require_cycle {
            let exact_small = self.n() <= 64;
            let mut starts = self.adj[self.root].clone();
            let base_visited = vec![false; self.n()];
            starts.sort_by_key(|&v| {
                if exact_small {
                    self.small_candidate_score(1u128 << self.root, v, Some(self.root), 0)
                } else {
                    self.large_candidate_score_simple(&base_visited, v, Some(self.root), 0)
                }
            });
            for mode in 0..attempts {
                for &first in &starts {
                    let mut visited = vec![false; self.n()];
                    visited[self.root] = true;
                    visited[first] = true;
                    let mut path = vec![self.root, first];
                    if self.greedy_walk(&mut visited, &mut path, Some(self.root), mode)
                        && self.validate_path(&path)
                    {
                        return Some(path);
                    }
                }
            }
        } else {
            for mode in 0..attempts {
                let mut visited = vec![false; self.n()];
                visited[self.root] = true;
                let mut path = vec![self.root];
                if self.greedy_walk(&mut visited, &mut path, None, mode) && self.validate_path(&path) {
                    return Some(path);
                }
            }
        }
        None
    }

    fn greedy_walk(
        &self,
        visited: &mut [bool],
        path: &mut Vec<usize>,
        target_fixed: Option<usize>,
        mode: usize,
    ) -> bool {
        while path.len() < self.n() {
            let current = *path.last().unwrap();
            let mut candidates: Vec<usize> = self.adj[current]
                .iter()
                .copied()
                .filter(|&v| !visited[v])
                .collect();
            candidates.sort_by_key(|&v| self.large_candidate_score_with_mode(visited, v, target_fixed, mode));

            let mut chosen = None;
            for next in candidates {
                visited[next] = true;
                if self.promising_large(visited, next, target_fixed) {
                    chosen = Some(next);
                    break;
                }
                visited[next] = false;
            }
            let Some(next) = chosen else {
                return false;
            };
            path.push(next);
        }
        match target_fixed {
            None => true,
            Some(t) => self.is_adjacent(*path.last().unwrap(), t),
        }
    }

    fn solve_guided_backtracking(&self, deadline: Instant, node_budget: u64) -> Option<Vec<usize>> {
        if self.require_cycle {
            let mut starts = self.adj[self.root].clone();
            starts.sort_by_key(|&v| self.large_candidate_score_simple(&vec![false; self.n()], v, Some(self.root), 0));
            for first in starts {
                let mut ctx = LargeSearchCtx {
                    visited: vec![false; self.n()],
                    path: vec![self.root, first],
                    nodes: 0,
                    node_budget,
                    deadline,
                };
                ctx.visited[self.root] = true;
                ctx.visited[first] = true;
                match self.dfs_large(&mut ctx, first, Some(self.root)) {
                    SearchStatus::Found => return Some(ctx.path),
                    SearchStatus::Failed | SearchStatus::Timeout => {}
                }
            }
            None
        } else {
            let mut ctx = LargeSearchCtx {
                visited: vec![false; self.n()],
                path: vec![self.root],
                nodes: 0,
                node_budget,
                deadline,
            };
            ctx.visited[self.root] = true;
            match self.dfs_large(&mut ctx, self.root, None) {
                SearchStatus::Found => Some(ctx.path),
                SearchStatus::Failed | SearchStatus::Timeout => None,
            }
        }
    }

    fn dfs_large(
        &self,
        ctx: &mut LargeSearchCtx,
        current: usize,
        target_fixed: Option<usize>,
    ) -> SearchStatus {
        ctx.nodes += 1;
        if ctx.nodes >= ctx.node_budget {
            return SearchStatus::Timeout;
        }
        if (ctx.nodes & 4095) == 0 && Instant::now() >= ctx.deadline {
            return SearchStatus::Timeout;
        }
        if ctx.path.len() == self.n() {
            return match target_fixed {
                None => SearchStatus::Found,
                Some(t) => {
                    if self.is_adjacent(current, t) {
                        SearchStatus::Found
                    } else {
                        SearchStatus::Failed
                    }
                }
            };
        }
        if !self.promising_large(&ctx.visited, current, target_fixed) {
            return SearchStatus::Failed;
        }

        let mut candidates: Vec<usize> = self.adj[current]
            .iter()
            .copied()
            .filter(|&v| !ctx.visited[v])
            .collect();
        candidates.sort_by_key(|&v| self.large_candidate_score_with_mode(&ctx.visited, v, target_fixed, 0));

        for next in candidates {
            ctx.visited[next] = true;
            ctx.path.push(next);
            match self.dfs_large(ctx, next, target_fixed) {
                SearchStatus::Found => return SearchStatus::Found,
                SearchStatus::Failed => {}
                SearchStatus::Timeout => {
                    ctx.path.pop();
                    ctx.visited[next] = false;
                    return SearchStatus::Timeout;
                }
            }
            ctx.path.pop();
            ctx.visited[next] = false;
        }
        SearchStatus::Failed
    }

    fn promising_large(&self, visited: &[bool], current: usize, target_fixed: Option<usize>) -> bool {
        let alive_count = self.alive_count_large(visited, current, target_fixed);
        if alive_count == 0 {
            return true;
        }
        if !self.connected_large(visited, current, target_fixed, None) {
            return false;
        }
        if alive_count > 1 && !self.connected_large(visited, current, target_fixed, Some(current)) {
            return false;
        }
        if let Some(t) = target_fixed {
            if alive_count > 1 && !self.connected_large(visited, current, target_fixed, Some(t)) {
                return false;
            }
        }

        let mut black = 0usize;
        let mut white = 0usize;
        let mut leafs = 0usize;
        let mut current_deg = 0usize;
        for v in 0..self.n() {
            if !self.is_alive_large(visited, current, target_fixed, v) {
                continue;
            }
            if self.color[v] == 0 {
                black += 1;
            } else {
                white += 1;
            }
            let deg = self.alive_degree_large(visited, current, target_fixed, v);
            if v == current {
                current_deg = deg;
                continue;
            }
            if Some(v) == target_fixed {
                if deg == 0 {
                    return false;
                }
                continue;
            }
            if target_fixed.is_some() {
                if deg < 2 {
                    return false;
                }
            } else {
                if deg == 0 {
                    return false;
                }
                if deg == 1 {
                    leafs += 1;
                    if leafs > 1 {
                        return false;
                    }
                }
            }
        }
        if current_deg == 0 && alive_count > 1 {
            return false;
        }

        match target_fixed {
            None => {
                if alive_count % 2 == 0 {
                    black == white
                } else if self.color[current] == 0 {
                    black == white + 1
                } else {
                    white == black + 1
                }
            }
            Some(t) => {
                if alive_count % 2 == 0 {
                    self.color[current] != self.color[t] && black == white
                } else if self.color[current] != self.color[t] {
                    false
                } else if self.color[current] == 0 {
                    black == white + 1
                } else {
                    white == black + 1
                }
            }
        }
    }

    fn alive_count_large(&self, visited: &[bool], current: usize, target_fixed: Option<usize>) -> usize {
        (0..self.n())
            .filter(|&v| self.is_alive_large(visited, current, target_fixed, v))
            .count()
    }

    fn is_alive_large(
        &self,
        visited: &[bool],
        current: usize,
        target_fixed: Option<usize>,
        v: usize,
    ) -> bool {
        v == current || Some(v) == target_fixed || !visited[v]
    }

    fn alive_degree_large(
        &self,
        visited: &[bool],
        current: usize,
        target_fixed: Option<usize>,
        v: usize,
    ) -> usize {
        self.adj[v]
            .iter()
            .filter(|&&to| self.is_alive_large(visited, current, target_fixed, to))
            .count()
    }

    fn connected_large(
        &self,
        visited: &[bool],
        current: usize,
        target_fixed: Option<usize>,
        banned: Option<usize>,
    ) -> bool {
        let start = (0..self.n()).find(|&v| {
            Some(v) != banned && self.is_alive_large(visited, current, target_fixed, v)
        });
        let Some(start) = start else {
            return true;
        };

        let mut stack = vec![start];
        let mut seen = vec![false; self.n()];
        seen[start] = true;
        while let Some(u) = stack.pop() {
            for &v in &self.adj[u] {
                if Some(v) == banned || !self.is_alive_large(visited, current, target_fixed, v) || seen[v] {
                    continue;
                }
                seen[v] = true;
                stack.push(v);
            }
        }

        (0..self.n()).all(|v| {
            Some(v) == banned || !self.is_alive_large(visited, current, target_fixed, v) || seen[v]
        })
    }

    fn large_candidate_score_simple(
        &self,
        visited: &[bool],
        next: usize,
        target_fixed: Option<usize>,
        mode: usize,
    ) -> (usize, usize, usize, usize, usize) {
        self.large_candidate_score_with_mode(visited, next, target_fixed, mode)
    }

    fn large_candidate_score_with_mode(
        &self,
        visited: &[bool],
        next: usize,
        target_fixed: Option<usize>,
        mode: usize,
    ) -> (usize, usize, usize, usize, usize) {
        let mut leafs = 0usize;
        let mut next_deg = 0usize;
        for v in 0..self.n() {
            let alive = v == next || Some(v) == target_fixed || (!visited[v] && v != next);
            if !alive {
                continue;
            }
            let deg = self.adj[v]
                .iter()
                .filter(|&&to| to == next || Some(to) == target_fixed || (!visited[to] && to != next))
                .count();
            if v == next {
                next_deg = deg;
            } else if Some(v) != target_fixed && deg == 1 {
                leafs += 1;
            }
        }
        let (x, y) = self.positions[next];
        let spatial = self.spatial_bias(x, y, mode);
        let target_bias = target_fixed
            .map(|t| if self.is_adjacent(next, t) { 0 } else { 1 })
            .unwrap_or(0);
        (leafs, next_deg, target_bias, self.adj[next].len(), spatial)
    }

    fn spatial_bias(&self, x: usize, y: usize, mode: usize) -> usize {
        match mode % 4 {
            0 => x + y,
            1 => (self.width - 1 - x) + y,
            2 => x + (self.height - 1 - y),
            _ => (self.width - 1 - x) + (self.height - 1 - y),
        }
    }

    fn format_solution(&self, path: Option<Vec<usize>>) -> String {
        let Some(path) = path else {
            return "0\n".to_string();
        };
        let mut out = String::new();
        out.push_str(&format!("{}\n", path.len()));
        for id in path {
            let (x, y) = self.positions[id];
            out.push_str(&format!("{x} {y}\n"));
        }
        out
    }
}

enum ExactOutcome {
    Found(Vec<usize>),
    Impossible,
    Timeout,
}

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        print!("0\n");
        return;
    }
    let Some(graph) = GridGraph::from_input(&input) else {
        print!("0\n");
        return;
    };
    let answer = graph.solve();
    print!("{}", graph.format_solution(answer));
}
