using System;
using System.IO;
using System.Runtime.CompilerServices;

public static class Program
{
    public static void Main()
    {
        var scanner = new FastScanner();
        int n = scanner.NextInt();
        int stockLength = scanner.NextInt();

        if (n <= 0)
        {
            var emptyWriter = new FastWriter();
            emptyWriter.WriteInt(0);
            emptyWriter.WriteNewLine();
            emptyWriter.Flush();
            return;
        }

        int[] lengths = new int[n];
        int[] indices = new int[n];

        for (int i = 0; i < n; i++)
        {
            int length = scanner.NextInt();
            if (length > stockLength)
            {
                throw new InvalidOperationException("Cut length exceeds stock length.");
            }

            lengths[i] = length;
            indices[i] = i;
        }

        RadixSort(lengths, indices);

        int[] nextCut = new int[n];
        Array.Fill(nextCut, -1);

        int[] binHeads = new int[n];
        int[] binTails = new int[n];
        int binCount = 0;

        var openBins = new Treap();

        for (int i = n - 1; i >= 0; i--)
        {
            int cutLength = lengths[i];
            int cutIndex = indices[i];

            if (openBins.TryTakeLowerBound(cutLength, out int remainingBefore, out int binId))
            {
                nextCut[binTails[binId]] = cutIndex;
                binTails[binId] = cutIndex;

                int remainingAfter = remainingBefore - cutLength;
                if (remainingAfter > 0)
                {
                    openBins.Add(remainingAfter, binId);
                }
            }
            else
            {
                int newBinId = binCount++;
                binHeads[newBinId] = cutIndex;
                binTails[newBinId] = cutIndex;

                int remainingAfter = stockLength - cutLength;
                if (remainingAfter > 0)
                {
                    openBins.Add(remainingAfter, newBinId);
                }
            }
        }

        var writer = new FastWriter();
        writer.WriteInt(binCount);
        writer.WriteNewLine();

        for (int binId = 0; binId < binCount; binId++)
        {
            int current = binHeads[binId];
            writer.WriteInt(current);
            current = nextCut[current];

            while (current != -1)
            {
                writer.WriteSpace();
                writer.WriteInt(current);
                current = nextCut[current];
            }

            writer.WriteNewLine();
        }

        writer.Flush();
    }

    private static void RadixSort(int[] keys, int[] values)
    {
        int n = keys.Length;
        if (n <= 1)
        {
            return;
        }

        const int BucketCount = 1 << 16;
        const int Mask = BucketCount - 1;

        int[] counts = new int[BucketCount];
        int[] tempKeys = new int[n];
        int[] tempValues = new int[n];

        for (int i = 0; i < n; i++)
        {
            counts[keys[i] & Mask]++;
        }

        int sum = 0;
        for (int i = 0; i < BucketCount; i++)
        {
            int count = counts[i];
            counts[i] = sum;
            sum += count;
        }

        for (int i = 0; i < n; i++)
        {
            int key = keys[i];
            int bucket = key & Mask;
            int position = counts[bucket]++;
            tempKeys[position] = key;
            tempValues[position] = values[i];
        }

        Array.Clear(counts, 0, counts.Length);

        for (int i = 0; i < n; i++)
        {
            counts[(tempKeys[i] >> 16) & Mask]++;
        }

        sum = 0;
        for (int i = 0; i < BucketCount; i++)
        {
            int count = counts[i];
            counts[i] = sum;
            sum += count;
        }

        for (int i = 0; i < n; i++)
        {
            int key = tempKeys[i];
            int bucket = (key >> 16) & Mask;
            int position = counts[bucket]++;
            keys[position] = key;
            values[position] = tempValues[i];
        }
    }

    private sealed class Treap
    {
        private Node _root;
        private uint _rng = 2463534242u;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void Add(int remaining, int binId)
        {
            _root = Insert(_root, remaining, binId, NextPriority());
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public bool TryTakeLowerBound(int minimumRemaining, out int remaining, out int binId)
        {
            bool found;
            _root = TakeLowerBound(_root, minimumRemaining, out found, out remaining, out binId);
            return found;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private uint NextPriority()
        {
            uint x = _rng;
            x ^= x << 13;
            x ^= x >> 17;
            x ^= x << 5;
            _rng = x;
            return x;
        }

        private static Node Insert(Node node, int key, int binId, uint priority)
        {
            if (node == null)
            {
                return new Node(key, priority, binId);
            }

            if (key < node.Key)
            {
                node.Left = Insert(node.Left, key, binId, priority);
                if (node.Left.Priority > node.Priority)
                {
                    node = RotateRight(node);
                }
            }
            else if (key > node.Key)
            {
                node.Right = Insert(node.Right, key, binId, priority);
                if (node.Right.Priority > node.Priority)
                {
                    node = RotateLeft(node);
                }
            }
            else
            {
                node.Push(binId);
            }

            return node;
        }

        private static Node TakeLowerBound(Node node, int minimumRemaining, out bool found, out int remaining, out int binId)
        {
            if (node == null)
            {
                found = false;
                remaining = 0;
                binId = 0;
                return null;
            }

            if (node.Key < minimumRemaining)
            {
                node.Right = TakeLowerBound(node.Right, minimumRemaining, out found, out remaining, out binId);
                return node;
            }

            node.Left = TakeLowerBound(node.Left, minimumRemaining, out found, out remaining, out binId);
            if (found)
            {
                return node;
            }

            found = true;
            remaining = node.Key;
            binId = node.Pop();

            if (node.Count == 0)
            {
                return Merge(node.Left, node.Right);
            }

            return node;
        }

        private static Node Merge(Node left, Node right)
        {
            if (left == null)
            {
                return right;
            }

            if (right == null)
            {
                return left;
            }

            if (left.Priority > right.Priority)
            {
                left.Right = Merge(left.Right, right);
                return left;
            }

            right.Left = Merge(left, right.Left);
            return right;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private static Node RotateLeft(Node node)
        {
            Node right = node.Right;
            node.Right = right.Left;
            right.Left = node;
            return right;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private static Node RotateRight(Node node)
        {
            Node left = node.Left;
            node.Left = left.Right;
            left.Right = node;
            return left;
        }

        private sealed class Node
        {
            public readonly int Key;
            public readonly uint Priority;
            public Node Left;
            public Node Right;
            private int[] _binIds;
            public int Count;

            public Node(int key, uint priority, int binId)
            {
                Key = key;
                Priority = priority;
                _binIds = new int[4];
                _binIds[0] = binId;
                Count = 1;
            }

            [MethodImpl(MethodImplOptions.AggressiveInlining)]
            public void Push(int binId)
            {
                if (Count == _binIds.Length)
                {
                    Array.Resize(ref _binIds, Count << 1);
                }

                _binIds[Count++] = binId;
            }

            [MethodImpl(MethodImplOptions.AggressiveInlining)]
            public int Pop()
            {
                return _binIds[--Count];
            }
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _position;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int Read()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_position++];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            }
            while (c >= 0 && c != '-' && (uint)(c - '0') > 9);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = (value * 10) + (c - '0');
                c = Read();
            }

            return value * sign;
        }
    }

    private sealed class FastWriter
    {
        private readonly Stream _stream = Console.OpenStandardOutput();
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _position;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteSpace()
        {
            WriteByte((byte)' ');
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteNewLine()
        {
            WriteByte((byte)'\n');
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private void WriteByte(byte value)
        {
            if (_position == _buffer.Length)
            {
                Flush();
            }

            _buffer[_position++] = value;
        }

        public void WriteInt(int value)
        {
            if (value == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            Span<byte> digits = stackalloc byte[11];
            int length = 0;

            while (value > 0)
            {
                digits[length++] = (byte)('0' + (value % 10));
                value /= 10;
            }

            for (int i = length - 1; i >= 0; i--)
            {
                WriteByte(digits[i]);
            }
        }

        public void Flush()
        {
            if (_position > 0)
            {
                _stream.Write(_buffer, 0, _position);
                _position = 0;
            }

            _stream.Flush();
        }
    }
}
