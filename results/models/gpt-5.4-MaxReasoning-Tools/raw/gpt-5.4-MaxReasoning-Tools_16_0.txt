#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <tuple>
#include <utility>
#include <vector>

struct Task {
    int machine = 0;
    std::int64_t duration = 0;
};

struct Instance {
    int job_count = 0;
    int machine_count = 0;
    std::vector<std::vector<Task>> jobs;
    std::vector<std::vector<std::int64_t>> suffix_job_work;
    std::vector<std::int64_t> machine_total_work;
    std::size_t total_tasks = 0;
};

struct Schedule {
    std::vector<std::vector<std::int64_t>> start_times;
    std::int64_t makespan = std::numeric_limits<std::int64_t>::max();
};

static Instance read_instance() {
    Instance inst;
    std::cin >> inst.job_count >> inst.machine_count;
    inst.jobs.assign(inst.job_count, {});
    inst.suffix_job_work.assign(inst.job_count, {});
    inst.machine_total_work.assign(inst.machine_count, 0);

    for (int j = 0; j < inst.job_count; ++j) {
        int task_count = 0;
        std::cin >> task_count;
        inst.jobs[j].resize(task_count);
        for (int k = 0; k < task_count; ++k) {
            Task task;
            std::cin >> task.machine >> task.duration;
            inst.jobs[j][k] = task;
            if (task.machine >= 0 && task.machine < inst.machine_count) {
                inst.machine_total_work[task.machine] += task.duration;
            }
            ++inst.total_tasks;
        }
    }

    for (int j = 0; j < inst.job_count; ++j) {
        const int t = static_cast<int>(inst.jobs[j].size());
        inst.suffix_job_work[j].assign(t + 1, 0);
        for (int k = t - 1; k >= 0; --k) {
            inst.suffix_job_work[j][k] = inst.suffix_job_work[j][k + 1] + inst.jobs[j][k].duration;
        }
    }

    return inst;
}

static Schedule make_empty_schedule(const Instance& inst) {
    Schedule schedule;
    schedule.start_times.resize(inst.job_count);
    for (int j = 0; j < inst.job_count; ++j) {
        schedule.start_times[j].assign(inst.jobs[j].size(), 0);
    }
    schedule.makespan = 0;
    return schedule;
}

static bool is_better(const Schedule& lhs, const Schedule& rhs) {
    return lhs.makespan < rhs.makespan;
}

static bool is_valid(const Instance& inst, const Schedule& schedule) {
    if (static_cast<int>(schedule.start_times.size()) != inst.job_count) {
        return false;
    }

    std::vector<std::vector<std::pair<std::int64_t, std::int64_t>>> machine_intervals(inst.machine_count);

    for (int j = 0; j < inst.job_count; ++j) {
        if (schedule.start_times[j].size() != inst.jobs[j].size()) {
            return false;
        }
        std::int64_t prev_finish = 0;
        for (std::size_t k = 0; k < inst.jobs[j].size(); ++k) {
            const Task& task = inst.jobs[j][k];
            const std::int64_t start = schedule.start_times[j][k];
            if (start < prev_finish) {
                return false;
            }
            const std::int64_t finish = start + task.duration;
            prev_finish = finish;
            if (task.machine < 0 || task.machine >= inst.machine_count) {
                return false;
            }
            machine_intervals[task.machine].push_back({start, finish});
        }
    }

    for (int m = 0; m < inst.machine_count; ++m) {
        auto& intervals = machine_intervals[m];
        std::sort(intervals.begin(), intervals.end());
        for (std::size_t i = 1; i < intervals.size(); ++i) {
            if (intervals[i].first < intervals[i - 1].second) {
                return false;
            }
        }
    }
    return true;
}

struct CandidateKey {
    std::int64_t a = 0;
    std::int64_t b = 0;
    std::int64_t c = 0;
    std::int64_t d = 0;
    int e = 0;
};

static CandidateKey build_key(
    int rule,
    std::int64_t earliest_start,
    std::int64_t remaining_job_work,
    std::int64_t machine_remaining_work,
    std::int64_t duration,
    int job_id
) {
    CandidateKey key;
    switch (rule) {
        case 0:
            key = {earliest_start, -remaining_job_work, -duration, -machine_remaining_work, job_id};
            break;
        case 1:
            key = {earliest_start, -machine_remaining_work, -remaining_job_work, -duration, job_id};
            break;
        case 2:
            key = {-remaining_job_work, earliest_start, -duration, -machine_remaining_work, job_id};
            break;
        default:
            key = {earliest_start, duration, -remaining_job_work, -machine_remaining_work, job_id};
            break;
    }
    return key;
}

static bool key_less(const CandidateKey& lhs, const CandidateKey& rhs) {
    return std::tie(lhs.a, lhs.b, lhs.c, lhs.d, lhs.e) <
           std::tie(rhs.a, rhs.b, rhs.c, rhs.d, rhs.e);
}

static Schedule greedy_schedule(const Instance& inst, int rule) {
    Schedule schedule = make_empty_schedule(inst);
    std::vector<int> next_task(inst.job_count, 0);
    std::vector<std::int64_t> job_ready(inst.job_count, 0);
    std::vector<std::int64_t> machine_ready(inst.machine_count, 0);
    std::vector<std::int64_t> machine_remaining = inst.machine_total_work;

    for (std::size_t scheduled = 0; scheduled < inst.total_tasks; ++scheduled) {
        int best_job = -1;
        CandidateKey best_key;

        for (int j = 0; j < inst.job_count; ++j) {
            const int k = next_task[j];
            if (k >= static_cast<int>(inst.jobs[j].size())) {
                continue;
            }
            const Task& task = inst.jobs[j][k];
            const std::int64_t earliest_start = std::max(job_ready[j], machine_ready[task.machine]);
            const std::int64_t remaining_job_work = inst.suffix_job_work[j][k];
            const std::int64_t machine_remaining_work = machine_remaining[task.machine];
            const CandidateKey key = build_key(
                rule, earliest_start, remaining_job_work, machine_remaining_work, task.duration, j
            );
            if (best_job == -1 || key_less(key, best_key)) {
                best_job = j;
                best_key = key;
            }
        }

        const int k = next_task[best_job];
        const Task& task = inst.jobs[best_job][k];
        const std::int64_t start = std::max(job_ready[best_job], machine_ready[task.machine]);
        const std::int64_t finish = start + task.duration;

        schedule.start_times[best_job][k] = start;
        job_ready[best_job] = finish;
        machine_ready[task.machine] = finish;
        machine_remaining[task.machine] -= task.duration;
        ++next_task[best_job];
        if (finish > schedule.makespan) {
            schedule.makespan = finish;
        }
    }

    return schedule;
}

class ExactSolver {
public:
    ExactSolver(const Instance& instance, Schedule incumbent)
        : inst(instance),
          best(std::move(incumbent)),
          next_task(inst.job_count, 0),
          job_ready(inst.job_count, 0),
          machine_ready(inst.machine_count, 0),
          current(inst.job_count),
          remaining_job(inst.job_count, 0),
          remaining_machine(inst.machine_total_work) {
        for (int j = 0; j < inst.job_count; ++j) {
            current[j].assign(inst.jobs[j].size(), 0);
            remaining_job[j] = inst.suffix_job_work[j].empty() ? 0 : inst.suffix_job_work[j][0];
        }
    }

    Schedule solve() {
        dfs(0, 0);
        return best;
    }

private:
    const Instance& inst;
    Schedule best;
    std::vector<int> next_task;
    std::vector<std::int64_t> job_ready;
    std::vector<std::int64_t> machine_ready;
    std::vector<std::vector<std::int64_t>> current;
    std::vector<std::int64_t> remaining_job;
    std::vector<std::int64_t> remaining_machine;

    std::int64_t lower_bound(std::int64_t current_makespan) const {
        std::int64_t bound = current_makespan;
        for (int j = 0; j < inst.job_count; ++j) {
            bound = std::max(bound, job_ready[j] + remaining_job[j]);
        }
        for (int m = 0; m < inst.machine_count; ++m) {
            bound = std::max(bound, machine_ready[m] + remaining_machine[m]);
        }
        return bound;
    }

    void dfs(std::size_t scheduled_count, std::int64_t current_makespan) {
        if (scheduled_count == inst.total_tasks) {
            if (current_makespan < best.makespan) {
                best.start_times = current;
                best.makespan = current_makespan;
            }
            return;
        }

        if (lower_bound(current_makespan) >= best.makespan) {
            return;
        }

        std::vector<int> candidates;
        candidates.reserve(inst.job_count);
        for (int j = 0; j < inst.job_count; ++j) {
            if (next_task[j] < static_cast<int>(inst.jobs[j].size())) {
                candidates.push_back(j);
            }
        }

        std::sort(candidates.begin(), candidates.end(), [&](int lhs, int rhs) {
            const int kl = next_task[lhs];
            const int kr = next_task[rhs];
            const Task& tl = inst.jobs[lhs][kl];
            const Task& tr = inst.jobs[rhs][kr];
            const std::int64_t est_l = std::max(job_ready[lhs], machine_ready[tl.machine]);
            const std::int64_t est_r = std::max(job_ready[rhs], machine_ready[tr.machine]);
            const CandidateKey key_l = build_key(
                0, est_l, remaining_job[lhs], remaining_machine[tl.machine], tl.duration, lhs
            );
            const CandidateKey key_r = build_key(
                0, est_r, remaining_job[rhs], remaining_machine[tr.machine], tr.duration, rhs
            );
            return key_less(key_l, key_r);
        });

        for (int j : candidates) {
            const int k = next_task[j];
            const Task& task = inst.jobs[j][k];
            const int m = task.machine;
            const std::int64_t old_job_ready = job_ready[j];
            const std::int64_t old_machine_ready = machine_ready[m];
            const std::int64_t old_remaining_job = remaining_job[j];
            const std::int64_t old_remaining_machine = remaining_machine[m];

            const std::int64_t start = std::max(job_ready[j], machine_ready[m]);
            const std::int64_t finish = start + task.duration;

            current[j][k] = start;
            ++next_task[j];
            job_ready[j] = finish;
            machine_ready[m] = finish;
            remaining_job[j] -= task.duration;
            remaining_machine[m] -= task.duration;

            dfs(scheduled_count + 1, std::max(current_makespan, finish));

            remaining_machine[m] = old_remaining_machine;
            remaining_job[j] = old_remaining_job;
            machine_ready[m] = old_machine_ready;
            job_ready[j] = old_job_ready;
            --next_task[j];
        }
    }
};

static Schedule solve_instance(const Instance& inst) {
    if (inst.total_tasks == 0) {
        return make_empty_schedule(inst);
    }

    if (inst.job_count <= 1 || inst.machine_count <= 1) {
        return greedy_schedule(inst, 0);
    }

    Schedule best = greedy_schedule(inst, 0);

    if (inst.total_tasks <= 18 && inst.job_count <= 6 && inst.machine_count <= 6) {
        for (int rule = 1; rule < 4; ++rule) {
            Schedule candidate = greedy_schedule(inst, rule);
            if (is_better(candidate, best)) {
                best = std::move(candidate);
            }
        }
        ExactSolver solver(inst, best);
        return solver.solve();
    }

    int rule_count = 1;
    if (inst.total_tasks <= 10000) {
        rule_count = 4;
    } else if (inst.total_tasks <= 50000) {
        rule_count = 2;
    }

    for (int rule = 1; rule < rule_count; ++rule) {
        Schedule candidate = greedy_schedule(inst, rule);
        if (is_better(candidate, best)) {
            best = std::move(candidate);
        }
    }

    return best;
}

static void write_schedule(const Instance& inst, const Schedule& schedule) {
    for (int j = 0; j < inst.job_count; ++j) {
        for (std::size_t k = 0; k < inst.jobs[j].size(); ++k) {
            if (k > 0) {
                std::cout << ' ';
            }
            std::cout << schedule.start_times[j][k] << ' ' << inst.jobs[j][k].machine;
        }
        std::cout << '\n';
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    const Instance inst = read_instance();
    Schedule schedule = solve_instance(inst);

    if (!is_valid(inst, schedule)) {
        schedule = greedy_schedule(inst, 0);
    }

    write_schedule(inst, schedule);
    return 0;
}
