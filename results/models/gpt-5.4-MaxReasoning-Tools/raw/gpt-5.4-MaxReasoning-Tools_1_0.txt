#include <bit>
#include <cmath>
#include <cstdint>

struct FastInput {
    static constexpr std::size_t kBufferSize = 1u << 20;

    std::FILE* in = stdin;
    std::vector<char> buffer = std::vector<char>(kBufferSize);
    std::size_t index = 0;
    std::size_t size = 0;

    int peek() {
        if (index >= size) {
            size = std::fread(buffer.data(), 1, kBufferSize, in);
            index = 0;
            if (size == 0) {
                return EOF;
            }
        }
        return static_cast<unsigned char>(buffer[index]);
    }

    int get() {
        const int c = peek();
        if (c != EOF) {
            ++index;
        }
        return c;
    }

    void skip_spaces() {
        int c = peek();
        while (c != EOF && c <= ' ') {
            ++index;
            c = peek();
        }
    }

    bool read_u64(std::uint64_t& out) {
        skip_spaces();
        int c = peek();
        if (c == EOF) {
            return false;
        }
        std::uint64_t value = 0;
        while (c >= '0' && c <= '9') {
            value = value * 10 + static_cast<std::uint64_t>(c - '0');
            ++index;
            c = peek();
        }
        out = value;
        return true;
    }

    bool read_double(double& out) {
        skip_spaces();
        int c = peek();
        if (c == EOF) {
            return false;
        }

        int sign = 1;
        if (c == '-' || c == '+') {
            sign = (c == '-') ? -1 : 1;
            ++index;
        }

        double value = 0.0;
        c = peek();
        while (c >= '0' && c <= '9') {
            value = value * 10.0 + static_cast<double>(c - '0');
            ++index;
            c = peek();
        }

        if (c == '.') {
            ++index;
            double scale = 0.1;
            c = peek();
            while (c >= '0' && c <= '9') {
                value += static_cast<double>(c - '0') * scale;
                scale *= 0.1;
                ++index;
                c = peek();
            }
        }

        if (c == 'e' || c == 'E') {
            ++index;
            int exp_sign = 1;
            c = peek();
            if (c == '-' || c == '+') {
                exp_sign = (c == '-') ? -1 : 1;
                ++index;
            }

            int exponent = 0;
            c = peek();
            while (c >= '0' && c <= '9') {
                exponent = exponent * 10 + (c - '0');
                ++index;
                c = peek();
            }
            value *= std::pow(10.0, static_cast<double>(exp_sign * exponent));
        }

        out = sign * value;
        return true;
    }
};

struct FastOutput {
    static constexpr std::size_t kBufferSize = 1u << 20;

    std::vector<char> buffer = std::vector<char>(kBufferSize);
    std::size_t index = 0;

    ~FastOutput() {
        flush();
    }

    void flush() {
        if (index != 0) {
            std::fwrite(buffer.data(), 1, index, stdout);
            index = 0;
        }
    }

    void push(char c) {
        if (index == kBufferSize) {
            flush();
        }
        buffer[index++] = c;
    }

    void write_u64(std::uint64_t value) {
        char temp[32];
        int len = 0;
        do {
            temp[len++] = static_cast<char>('0' + (value % 10));
            value /= 10;
        } while (value != 0);
        while (len > 0) {
            push(temp[--len]);
        }
    }
};

struct Point {
    float x = 0.0f;
    float y = 0.0f;
};

struct Bounds {
    double min_x = std::numeric_limits<double>::infinity();
    double min_y = std::numeric_limits<double>::infinity();
    double max_x = -std::numeric_limits<double>::infinity();
    double max_y = -std::numeric_limits<double>::infinity();
};

struct KeyId {
    std::uint64_t key = 0;
    std::uint32_t id = 0;
};

static inline double sq_distance(const Point& a, const Point& b) {
    const double dx = static_cast<double>(a.x) - static_cast<double>(b.x);
    const double dy = static_cast<double>(a.y) - static_cast<double>(b.y);
    return dx * dx + dy * dy;
}

static inline double distance_euclid(const Point& a, const Point& b) {
    return std::sqrt(sq_distance(a, b));
}

static inline std::uint64_t spread_bits(std::uint32_t value) {
    std::uint64_t x = value;
    x = (x | (x << 16)) & 0x0000FFFF0000FFFFULL;
    x = (x | (x << 8)) & 0x00FF00FF00FF00FFULL;
    x = (x | (x << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    x = (x | (x << 2)) & 0x3333333333333333ULL;
    x = (x | (x << 1)) & 0x5555555555555555ULL;
    return x;
}

static inline std::uint64_t morton_code(std::uint32_t x, std::uint32_t y) {
    return spread_bits(x) | (spread_bits(y) << 1);
}

class SuccessorDSU {
public:
    explicit SuccessorDSU(int n) : parent_(static_cast<std::size_t>(n) + 1u) {
        std::iota(parent_.begin(), parent_.end(), 0);
    }

    int next(int x) {
        if (x < 0) {
            x = 0;
        }
        const int n = static_cast<int>(parent_.size()) - 1;
        if (x > n) {
            return n;
        }
        return find(x);
    }

    void erase(int x) {
        const int n = static_cast<int>(parent_.size()) - 1;
        if (x < 0 || x >= n) {
            return;
        }
        parent_[static_cast<std::size_t>(x)] = find(x + 1);
    }

private:
    std::vector<int> parent_;

    int find(int x) {
        int root = x;
        while (parent_[static_cast<std::size_t>(root)] != root) {
            root = parent_[static_cast<std::size_t>(root)];
        }
        while (parent_[static_cast<std::size_t>(x)] != x) {
            const int next = parent_[static_cast<std::size_t>(x)];
            parent_[static_cast<std::size_t>(x)] = root;
            x = next;
        }
        return root;
    }
};

std::vector<std::uint32_t> solve_exact(const std::vector<Point>& points) {
    const int n = static_cast<int>(points.size());
    if (n == 0) {
        return {};
    }
    if (n == 1) {
        return {0};
    }

    std::vector<double> d(static_cast<std::size_t>(n) * static_cast<std::size_t>(n), 0.0);
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            const double value = distance_euclid(points[static_cast<std::size_t>(i)], points[static_cast<std::size_t>(j)]);
            d[static_cast<std::size_t>(i) * static_cast<std::size_t>(n) + static_cast<std::size_t>(j)] = value;
            d[static_cast<std::size_t>(j) * static_cast<std::size_t>(n) + static_cast<std::size_t>(i)] = value;
        }
    }

    const int m = n - 1;
    const std::uint32_t full_mask = (1u << m);
    const double inf = std::numeric_limits<double>::infinity();

    std::vector<double> dp(static_cast<std::size_t>(full_mask) * static_cast<std::size_t>(m), inf);
    std::vector<std::int16_t> parent(static_cast<std::size_t>(full_mask) * static_cast<std::size_t>(m), -1);

    for (int i = 0; i < m; ++i) {
        dp[static_cast<std::size_t>(1u << i) * static_cast<std::size_t>(m) + static_cast<std::size_t>(i)] =
            d[static_cast<std::size_t>(i + 1)];
    }

    for (std::uint32_t mask = 1; mask < full_mask; ++mask) {
        for (int last = 0; last < m; ++last) {
            if ((mask & (1u << last)) == 0) {
                continue;
            }
            const std::size_t state = static_cast<std::size_t>(mask) * static_cast<std::size_t>(m) + static_cast<std::size_t>(last);
            const double cur = dp[state];
            if (!std::isfinite(cur)) {
                continue;
            }
            std::uint32_t remain = (full_mask - 1u) ^ mask;
            while (remain != 0) {
                const std::uint32_t bit = remain & (~remain + 1u);
                const int next = std::countr_zero(bit);
                const std::uint32_t next_mask = mask | bit;
                const double cand = cur + d[static_cast<std::size_t>(last + 1) * static_cast<std::size_t>(n) + static_cast<std::size_t>(next + 1)];
                const std::size_t next_state =
                    static_cast<std::size_t>(next_mask) * static_cast<std::size_t>(m) + static_cast<std::size_t>(next);
                if (cand < dp[next_state]) {
                    dp[next_state] = cand;
                    parent[next_state] = static_cast<std::int16_t>(last);
                }
                remain ^= bit;
            }
        }
    }

    const std::uint32_t all = full_mask - 1u;
    double best = inf;
    int best_last = 0;
    for (int last = 0; last < m; ++last) {
        const double cand =
            dp[static_cast<std::size_t>(all) * static_cast<std::size_t>(m) + static_cast<std::size_t>(last)] +
            d[static_cast<std::size_t>(last + 1) * static_cast<std::size_t>(n)];
        if (cand < best) {
            best = cand;
            best_last = last;
        }
    }

    std::vector<std::uint32_t> route(static_cast<std::size_t>(n), 0);
    std::uint32_t mask = all;
    int last = best_last;
    for (int pos = n - 1; pos >= 1; --pos) {
        route[static_cast<std::size_t>(pos)] = static_cast<std::uint32_t>(last + 1);
        const std::size_t state = static_cast<std::size_t>(mask) * static_cast<std::size_t>(m) + static_cast<std::size_t>(last);
        const int prev = parent[state];
        mask ^= (1u << last);
        last = prev;
    }
    route[0] = 0;
    return route;
}

std::vector<KeyId> build_keys(const std::vector<Point>& points, const Bounds& bounds) {
    const std::size_t n = points.size();
    std::vector<KeyId> keys(n);

    const double range_x = bounds.max_x - bounds.min_x;
    const double range_y = bounds.max_y - bounds.min_y;
    const double scale_x = (range_x > 0.0) ? (4294967295.0 / range_x) : 0.0;
    const double scale_y = (range_y > 0.0) ? (4294967295.0 / range_y) : 0.0;

    for (std::size_t i = 0; i < n; ++i) {
        double qx = 0.0;
        double qy = 0.0;
        if (scale_x > 0.0) {
            qx = (static_cast<double>(points[i].x) - bounds.min_x) * scale_x;
        }
        if (scale_y > 0.0) {
            qy = (static_cast<double>(points[i].y) - bounds.min_y) * scale_y;
        }
        if (qx < 0.0) {
            qx = 0.0;
        } else if (qx > 4294967295.0) {
            qx = 4294967295.0;
        }
        if (qy < 0.0) {
            qy = 0.0;
        } else if (qy > 4294967295.0) {
            qy = 4294967295.0;
        }

        const std::uint32_t ux = static_cast<std::uint32_t>(qx + 0.5);
        const std::uint32_t uy = static_cast<std::uint32_t>(qy + 0.5);
        keys[i].key = morton_code(ux, uy);
        keys[i].id = static_cast<std::uint32_t>(i);
    }
    return keys;
}

void radix_sort_keys(std::vector<KeyId>& data) {
    const std::size_t n = data.size();
    if (n <= 1) {
        return;
    }

    std::vector<KeyId> temp(n);
    std::vector<std::size_t> count(1u << 16, 0);

    for (int shift = 0; shift < 64; shift += 16) {
        std::fill(count.begin(), count.end(), 0);
        for (const KeyId& item : data) {
            ++count[(item.key >> shift) & 0xFFFFu];
        }

        std::size_t sum = 0;
        for (std::size_t i = 0; i < count.size(); ++i) {
            const std::size_t c = count[i];
            count[i] = sum;
            sum += c;
        }

        for (const KeyId& item : data) {
            temp[count[(item.key >> shift) & 0xFFFFu]++] = item;
        }
        data.swap(temp);
    }
}

std::vector<std::uint32_t> route_from_sorted_keys(const std::vector<KeyId>& keys) {
    const std::size_t n = keys.size();
    std::vector<std::uint32_t> route(n, 0);
    std::size_t pos0 = 0;
    while (pos0 < n && keys[pos0].id != 0u) {
        ++pos0;
    }
    for (std::size_t i = 1; i < n; ++i) {
        route[i] = keys[(pos0 + i) % n].id;
    }
    return route;
}

std::vector<std::uint32_t> greedy_route_from_sorted(const std::vector<Point>& points, const std::vector<KeyId>& keys, int window) {
    const int n = static_cast<int>(points.size());
    std::vector<std::uint32_t> sorted_ids(static_cast<std::size_t>(n), 0);
    std::vector<int> pos_of(static_cast<std::size_t>(n), 0);
    for (int i = 0; i < n; ++i) {
        sorted_ids[static_cast<std::size_t>(i)] = keys[static_cast<std::size_t>(i)].id;
        pos_of[sorted_ids[static_cast<std::size_t>(i)]] = i;
    }

    SuccessorDSU right(n);
    SuccessorDSU left(n);

    const int pos0 = pos_of[0];
    right.erase(pos0);
    left.erase(n - 1 - pos0);

    std::vector<std::uint32_t> route;
    route.reserve(static_cast<std::size_t>(n));
    route.push_back(0);

    std::uint32_t current = 0;
    for (int step = 1; step < n; ++step) {
        const int p = pos_of[current];
        int best_pos = -1;
        double best_dist = std::numeric_limits<double>::infinity();

        auto consider = [&](int candidate_pos) {
            if (candidate_pos < 0 || candidate_pos >= n) {
                return;
            }
            const std::uint32_t city = sorted_ids[static_cast<std::size_t>(candidate_pos)];
            const double cand = sq_distance(points[static_cast<std::size_t>(current)], points[static_cast<std::size_t>(city)]);
            if (cand < best_dist) {
                best_dist = cand;
                best_pos = candidate_pos;
            }
        };

        int rr = right.next(p);
        for (int seen = 0; seen < window && rr < n; ++seen) {
            consider(rr);
            rr = right.next(rr + 1);
        }

        int ll = left.next(n - 1 - p);
        for (int seen = 0; seen < window && ll < n; ++seen) {
            consider(n - 1 - ll);
            ll = left.next(ll + 1);
        }

        if (best_pos < 0) {
            rr = right.next(0);
            for (int seen = 0; seen < window && rr < n; ++seen) {
                consider(rr);
                rr = right.next(rr + 1);
            }
            ll = left.next(0);
            for (int seen = 0; seen < window && ll < n; ++seen) {
                consider(n - 1 - ll);
                ll = left.next(ll + 1);
            }
        }

        if (best_pos < 0) {
            best_pos = right.next(0);
        }

        current = sorted_ids[static_cast<std::size_t>(best_pos)];
        route.push_back(current);
        right.erase(best_pos);
        left.erase(n - 1 - best_pos);
    }

    return route;
}

void improve_two_opt(std::vector<std::uint32_t>& route, const std::vector<Point>& points, int window, int max_passes) {
    const std::size_t n = route.size();
    if (n < 4) {
        return;
    }

    for (int pass = 0; pass < max_passes; ++pass) {
        bool improved = false;
        for (std::size_t i = 1; i + 1 < n; ++i) {
            const std::size_t limit = std::min(n - 1, i + static_cast<std::size_t>(window));
            for (std::size_t j = i + 1; j <= limit; ++j) {
                const std::size_t next_j = (j + 1 == n) ? 0 : (j + 1);
                if (i == 1 && next_j == 0) {
                    continue;
                }

                const std::uint32_t a = route[i - 1];
                const std::uint32_t b = route[i];
                const std::uint32_t c = route[j];
                const std::uint32_t d = route[next_j];

                const double before = distance_euclid(points[a], points[b]) + distance_euclid(points[c], points[d]);
                const double after = distance_euclid(points[a], points[c]) + distance_euclid(points[b], points[d]);
                if (after + 1e-12 < before) {
                    std::reverse(route.begin() + static_cast<std::ptrdiff_t>(i), route.begin() + static_cast<std::ptrdiff_t>(j + 1));
                    improved = true;
                    break;
                }
            }
        }
        if (!improved) {
            break;
        }
    }
}

void write_route(const std::vector<std::uint32_t>& route, FastOutput& out) {
    for (std::size_t i = 0; i < route.size(); ++i) {
        out.write_u64(route[i]);
        out.push((i + 1 == route.size()) ? '\n' : ' ');
    }
}

void write_sorted_route_direct(const std::vector<KeyId>& keys, FastOutput& out) {
    const std::size_t n = keys.size();
    std::size_t pos0 = 0;
    while (pos0 < n && keys[pos0].id != 0u) {
        ++pos0;
    }
    out.write_u64(0);
    if (n == 1) {
        out.push('\n');
        return;
    }
    out.push(' ');
    for (std::size_t i = 1; i < n; ++i) {
        out.write_u64(keys[(pos0 + i) % n].id);
        out.push((i + 1 == n) ? '\n' : ' ');
    }
}

int main() {
    FastInput in;
    FastOutput out;

    std::uint64_t n64 = 0;
    if (!in.read_u64(n64)) {
        return 0;
    }

    if (n64 == 0) {
        out.push('\n');
        return 0;
    }

    constexpr std::uint64_t kExactLimit = 18;
    constexpr std::uint64_t kGreedyLimit = 300000;
    constexpr std::uint64_t kStoredLimit = 20000000;
    constexpr std::uint64_t kRadixLimit = 200000;

    if (n64 > kStoredLimit) {
        double x = 0.0;
        double y = 0.0;
        for (std::uint64_t i = 0; i < n64; ++i) {
            in.read_double(x);
            in.read_double(y);
        }
        for (std::uint64_t i = 0; i < n64; ++i) {
            out.write_u64(i);
            out.push((i + 1 == n64) ? '\n' : ' ');
        }
        return 0;
    }

    const std::size_t n = static_cast<std::size_t>(n64);
    std::vector<Point> points;
    points.reserve(n);
    Bounds bounds;

    for (std::size_t i = 0; i < n; ++i) {
        double x = 0.0;
        double y = 0.0;
        in.read_double(x);
        in.read_double(y);

        Point p;
        p.x = static_cast<float>(x);
        p.y = static_cast<float>(y);
        points.push_back(p);

        bounds.min_x = std::min(bounds.min_x, x);
        bounds.min_y = std::min(bounds.min_y, y);
        bounds.max_x = std::max(bounds.max_x, x);
        bounds.max_y = std::max(bounds.max_y, y);
    }

    if (n64 <= kExactLimit) {
        const std::vector<std::uint32_t> route = solve_exact(points);
        write_route(route, out);
        return 0;
    }

    std::vector<KeyId> keys = build_keys(points, bounds);
    if (n64 >= kRadixLimit) {
        radix_sort_keys(keys);
    } else {
        std::sort(keys.begin(), keys.end(), [](const KeyId& a, const KeyId& b) {
            if (a.key != b.key) {
                return a.key < b.key;
            }
            return a.id < b.id;
        });
    }

    if (n64 <= kGreedyLimit) {
        int window = 10;
        if (n64 <= 5000) {
            window = 28;
        } else if (n64 <= 50000) {
            window = 16;
        }

        std::vector<std::uint32_t> route = greedy_route_from_sorted(points, keys, window);
        if (n64 <= 2000) {
            improve_two_opt(route, points, 64, 6);
        } else if (n64 <= 10000) {
            improve_two_opt(route, points, 28, 3);
        } else if (n64 <= 50000) {
            improve_two_opt(route, points, 10, 1);
        }
        write_route(route, out);
    } else {
        write_sorted_route_direct(keys, out);
    }

    return 0;
}
