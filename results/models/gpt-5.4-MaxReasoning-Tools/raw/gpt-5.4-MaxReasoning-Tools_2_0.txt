using System;
using System.Collections.Generic;
using System.IO;
using System.Numerics;
using System.Text;

public static class Program
{
    private const long Inf = long.MaxValue / 4;
    private const int ExactOddLimit = 20;
    private const int GreedyOddLimit = 200;
    private const int GreedyNodeLimit = 5000;
    private const int GreedyEdgeLimit = 50000;

    private static int _n;
    private static int _m;
    private static int[] _u = Array.Empty<int>();
    private static int[] _v = Array.Empty<int>();
    private static long[] _w = Array.Empty<long>();
    private static int[] _degree = Array.Empty<int>();
    private static int[] _head = Array.Empty<int>();
    private static int[] _to = Array.Empty<int>();
    private static int[] _adjEdge = Array.Empty<int>();

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        _n = scanner.NextInt();
        _m = scanner.NextInt();

        if (_n <= 0)
        {
            return;
        }

        _u = new int[_m];
        _v = new int[_m];
        _w = new long[_m];
        _degree = new int[_n];

        for (int i = 0; i < _m; i++)
        {
            int a = scanner.NextInt();
            int b = scanner.NextInt();
            long c = scanner.NextLong();

            _u[i] = a;
            _v[i] = b;
            _w[i] = c;
            _degree[a]++;
            _degree[b]++;
        }

        var writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);

        if (_m == 0)
        {
            writer.WriteLine(0);
            writer.Flush();
            return;
        }

        BuildOriginalGraph();

        var odd = new List<int>();
        for (int i = 0; i < _n; i++)
        {
            if ((_degree[i] & 1) != 0)
            {
                odd.Add(i);
            }
        }

        List<int> duplicates = odd.Count == 0 ? new List<int>() : BuildAugmentation(odd);
        List<int> route = BuildEulerCircuit(duplicates);

        for (int i = 0; i < route.Count; i++)
        {
            if (i > 0)
            {
                writer.Write(' ');
            }

            writer.Write(route[i]);
        }

        writer.WriteLine();
        writer.Flush();
    }

    private static void BuildOriginalGraph()
    {
        _head = new int[_n + 1];
        for (int i = 0; i < _n; i++)
        {
            _head[i + 1] = _head[i] + _degree[i];
        }

        _to = new int[_head[_n]];
        _adjEdge = new int[_head[_n]];
        int[] cursor = (int[])_head.Clone();

        for (int i = 0; i < _m; i++)
        {
            int a = _u[i];
            int b = _v[i];

            int p = cursor[a]++;
            _to[p] = b;
            _adjEdge[p] = i;

            p = cursor[b]++;
            _to[p] = a;
            _adjEdge[p] = i;
        }
    }

    private static List<int> BuildAugmentation(List<int> odd)
    {
        if (odd.Count <= ExactOddLimit)
        {
            return BuildExactAugmentation(odd);
        }

        if (odd.Count <= GreedyOddLimit && _n <= GreedyNodeLimit && _m <= GreedyEdgeLimit)
        {
            return BuildGreedyAugmentation(odd);
        }

        return BuildTreeAugmentation();
    }

    private static List<int> BuildExactAugmentation(List<int> odd)
    {
        int k = odd.Count;
        long[,] pairDist = new long[k, k];
        int[][] parentBySource = new int[k][];

        for (int i = 0; i < k; i++)
        {
            DijkstraResult result = RunDijkstra(odd[i], true, false);
            parentBySource[i] = result.ParentEdge;

            for (int j = 0; j < k; j++)
            {
                pairDist[i, j] = result.Dist[odd[j]];
            }
        }

        int states = 1 << k;
        long[] dp = new long[states];
        int[] choice = new int[states];
        Array.Fill(dp, -1L);
        Array.Fill(choice, -1);

        long Solve(int mask)
        {
            if (mask == 0)
            {
                return 0;
            }

            if (dp[mask] >= 0)
            {
                return dp[mask];
            }

            int i = BitOperations.TrailingZeroCount((uint)mask);
            int rest = mask ^ (1 << i);
            long best = Inf;
            int bestJ = -1;

            for (int j = i + 1; j < k; j++)
            {
                if ((rest & (1 << j)) == 0)
                {
                    continue;
                }

                long d = pairDist[i, j];
                if (d >= Inf)
                {
                    continue;
                }

                long candidate = d + Solve(rest ^ (1 << j));
                if (candidate < best)
                {
                    best = candidate;
                    bestJ = j;
                }
            }

            choice[mask] = bestJ;
            dp[mask] = best;
            return best;
        }

        int fullMask = states - 1;
        Solve(fullMask);

        var duplicates = new List<int>(k * 4);
        int currentMask = fullMask;

        while (currentMask != 0)
        {
            int i = BitOperations.TrailingZeroCount((uint)currentMask);
            int j = choice[currentMask];
            if (j < 0)
            {
                break;
            }

            AddPathEdges(parentBySource[i], odd[i], odd[j], duplicates);
            currentMask ^= 1 << i;
            currentMask ^= 1 << j;
        }

        return duplicates;
    }

    private static List<int> BuildGreedyAugmentation(List<int> odd)
    {
        int k = odd.Count;
        long[,] pairDist = new long[k, k];
        int[][] parentBySource = new int[k][];

        for (int i = 0; i < k; i++)
        {
            DijkstraResult result = RunDijkstra(odd[i], true, false);
            parentBySource[i] = result.ParentEdge;

            for (int j = 0; j < k; j++)
            {
                pairDist[i, j] = result.Dist[odd[j]];
            }
        }

        var duplicates = new List<int>(k * 8);
        bool[] used = new bool[k];
        int remaining = k;

        while (remaining > 0)
        {
            long best = Inf;
            int bestI = -1;
            int bestJ = -1;

            for (int i = 0; i < k; i++)
            {
                if (used[i])
                {
                    continue;
                }

                for (int j = i + 1; j < k; j++)
                {
                    if (used[j])
                    {
                        continue;
                    }

                    long d = pairDist[i, j];
                    if (d < best)
                    {
                        best = d;
                        bestI = i;
                        bestJ = j;
                    }
                }
            }

            if (bestI < 0)
            {
                break;
            }

            used[bestI] = true;
            used[bestJ] = true;
            remaining -= 2;
            AddPathEdges(parentBySource[bestI], odd[bestI], odd[bestJ], duplicates);
        }

        return duplicates;
    }

    private static List<int> BuildTreeAugmentation()
    {
        DijkstraResult result = RunDijkstra(0, true, true);
        var duplicates = new List<int>(Math.Max(4, _n / 2));
        bool[] need = new bool[_n];

        for (int i = 0; i < _n; i++)
        {
            need[i] = (_degree[i] & 1) != 0;
        }

        List<int> order = result.Order;
        for (int i = order.Count - 1; i >= 0; i--)
        {
            int node = order[i];
            if (node == 0 || !need[node])
            {
                continue;
            }

            int edgeId = result.ParentEdge[node];
            if (edgeId < 0)
            {
                continue;
            }

            duplicates.Add(edgeId);
            int parent = OtherEndpoint(edgeId, node);
            need[parent] = !need[parent];
        }

        return duplicates;
    }

    private static void AddPathEdges(int[] parentEdge, int source, int target, List<int> duplicates)
    {
        int current = target;

        while (current != source)
        {
            int edgeId = parentEdge[current];
            if (edgeId < 0)
            {
                break;
            }

            duplicates.Add(edgeId);
            current = OtherEndpoint(edgeId, current);
        }
    }

    private static int OtherEndpoint(int edgeId, int node)
    {
        return _u[edgeId] == node ? _v[edgeId] : _u[edgeId];
    }

    private static List<int> BuildEulerCircuit(List<int> duplicates)
    {
        int extra = duplicates.Count;
        int totalEdges = _m + extra;
        int[] degree2 = new int[_n];
        Array.Copy(_degree, degree2, _n);

        for (int i = 0; i < extra; i++)
        {
            int edgeId = duplicates[i];
            degree2[_u[edgeId]]++;
            degree2[_v[edgeId]]++;
        }

        int[] head2 = new int[_n + 1];
        for (int i = 0; i < _n; i++)
        {
            head2[i + 1] = head2[i] + degree2[i];
        }

        int[] to2 = new int[head2[_n]];
        int[] occId2 = new int[head2[_n]];
        int[] cursor = (int[])head2.Clone();

        void AddOccurrence(int a, int b, int occurrenceId)
        {
            int p = cursor[a]++;
            to2[p] = b;
            occId2[p] = occurrenceId;

            p = cursor[b]++;
            to2[p] = a;
            occId2[p] = occurrenceId;
        }

        int occurrence = 0;
        for (int i = 0; i < _m; i++)
        {
            AddOccurrence(_u[i], _v[i], occurrence++);
        }

        for (int i = 0; i < extra; i++)
        {
            int edgeId = duplicates[i];
            AddOccurrence(_u[edgeId], _v[edgeId], occurrence++);
        }

        bool[] used = new bool[totalEdges];
        int[] ptr = new int[_n];
        Array.Copy(head2, ptr, _n);
        var stack = new Stack<int>(totalEdges + 1);
        var circuit = new List<int>(totalEdges + 1);
        stack.Push(0);

        while (stack.Count > 0)
        {
            int node = stack.Peek();
            int p = ptr[node];
            int end = head2[node + 1];

            while (p < end && used[occId2[p]])
            {
                p++;
            }

            ptr[node] = p;

            if (p == end)
            {
                circuit.Add(node);
                stack.Pop();
                continue;
            }

            int occ = occId2[p];
            used[occ] = true;
            ptr[node] = p + 1;
            stack.Push(to2[p]);
        }

        circuit.Reverse();
        return circuit;
    }

    private static DijkstraResult RunDijkstra(int source, bool needParents, bool recordOrder)
    {
        long[] dist = new long[_n];
        Array.Fill(dist, Inf);

        int[] parentEdge = needParents ? new int[_n] : Array.Empty<int>();
        if (needParents)
        {
            Array.Fill(parentEdge, -1);
        }

        List<int> order = recordOrder ? new List<int>(_n) : new List<int>(0);
        var pq = new PriorityQueue<int, long>();
        dist[source] = 0;
        pq.Enqueue(source, 0);

        while (pq.Count > 0)
        {
            pq.TryDequeue(out int node, out long currentDist);
            if (currentDist != dist[node])
            {
                continue;
            }

            if (recordOrder)
            {
                order.Add(node);
            }

            for (int p = _head[node]; p < _head[node + 1]; p++)
            {
                int next = _to[p];
                int edgeId = _adjEdge[p];
                long nd = currentDist + _w[edgeId];

                if (nd >= dist[next])
                {
                    continue;
                }

                dist[next] = nd;
                if (needParents)
                {
                    parentEdge[next] = edgeId;
                }

                pq.Enqueue(next, nd);
            }
        }

        return new DijkstraResult(dist, parentEdge, order);
    }

    private readonly struct DijkstraResult
    {
        public DijkstraResult(long[] dist, int[] parentEdge, List<int> order)
        {
            Dist = dist;
            ParentEdge = parentEdge;
            Order = order;
        }

        public long[] Dist { get; }
        public int[] ParentEdge { get; }
        public List<int> Order { get; }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _length;
        private int _index;

        public FastScanner(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 16];
        }

        public int NextInt()
        {
            int c = Read();
            while (c <= 32 && c >= 0)
            {
                c = Read();
            }

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        public long NextLong()
        {
            int c = Read();
            while (c <= 32 && c >= 0)
            {
                c = Read();
            }

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            long value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (_index >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _index = 0;
                if (_length == 0)
                {
                    return -1;
                }
            }

            return _buffer[_index++];
        }
    }
}
