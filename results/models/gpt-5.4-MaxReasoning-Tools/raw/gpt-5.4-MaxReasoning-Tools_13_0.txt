#include <algorithm>
#include <cstdint>
#include <iostream>
#include <numeric>
#include <string>
#include <utility>
#include <vector>

namespace {

struct SuffixAutomaton {
    struct State {
        int len = 0;
        int link = -1;
        int first_edge = -1;
        int first_pos = -1;
        int common = 0;
    };

    struct Edge {
        int to = 0;
        int next = -1;
        unsigned char ch = 0;
    };

    std::vector<State> states;
    std::vector<Edge> edges;
    std::vector<int> order;
    std::vector<int> best;
    int last = 0;

    explicit SuffixAutomaton(std::size_t text_len = 0) {
        states.reserve(text_len == 0 ? 1 : text_len * 2);
        edges.reserve(text_len == 0 ? 0 : text_len * 3);
        states.push_back(State{});
    }

    int find_transition(int state, unsigned char ch) const {
        for (int edge = states[state].first_edge; edge != -1; edge = edges[edge].next) {
            if (edges[edge].ch == ch) {
                return edges[edge].to;
            }
        }
        return -1;
    }

    int find_transition_edge(int state, unsigned char ch) {
        for (int edge = states[state].first_edge; edge != -1; edge = edges[edge].next) {
            if (edges[edge].ch == ch) {
                return edge;
            }
        }
        return -1;
    }

    void add_transition(int state, unsigned char ch, int to) {
        edges.push_back(Edge{to, states[state].first_edge, ch});
        states[state].first_edge = static_cast<int>(edges.size()) - 1;
    }

    void extend(unsigned char ch, int pos) {
        const int cur = static_cast<int>(states.size());
        states.push_back(State{states[last].len + 1, 0, -1, pos, 0});

        int p = last;
        while (p != -1 && find_transition(p, ch) == -1) {
            add_transition(p, ch, cur);
            p = states[p].link;
        }

        if (p == -1) {
            states[cur].link = 0;
        } else {
            const int q = find_transition(p, ch);
            if (states[p].len + 1 == states[q].len) {
                states[cur].link = q;
            } else {
                const int clone = static_cast<int>(states.size());
                states.push_back(states[q]);
                states[clone].len = states[p].len + 1;

                while (p != -1) {
                    const int edge_idx = find_transition_edge(p, ch);
                    if (edge_idx == -1 || edges[edge_idx].to != q) {
                        break;
                    }
                    edges[edge_idx].to = clone;
                    p = states[p].link;
                }

                states[q].link = clone;
                states[cur].link = clone;
            }
        }

        last = cur;
    }

    void build(const std::string& s) {
        for (int i = 0; i < static_cast<int>(s.size()); ++i) {
            extend(static_cast<unsigned char>(s[i]), i);
        }

        best.assign(states.size(), 0);
        for (auto& state : states) {
            state.common = state.len;
        }

        std::vector<int> count(s.size() + 1, 0);
        for (const auto& state : states) {
            ++count[state.len];
        }
        for (std::size_t i = 1; i < count.size(); ++i) {
            count[i] += count[i - 1];
        }

        order.assign(states.size(), 0);
        for (int i = static_cast<int>(states.size()) - 1; i >= 0; --i) {
            order[--count[states[i].len]] = i;
        }
    }

    void incorporate(const std::string& s) {
        std::fill(best.begin(), best.end(), 0);

        int state = 0;
        int match_len = 0;
        for (unsigned char ch : s) {
            int next = find_transition(state, ch);
            while (state != -1 && next == -1) {
                state = states[state].link;
                if (state == -1) {
                    match_len = 0;
                    state = 0;
                    break;
                }
                match_len = states[state].len;
                next = find_transition(state, ch);
            }

            if (next != -1) {
                state = next;
                ++match_len;
                if (match_len > best[state]) {
                    best[state] = match_len;
                }
            } else {
                state = 0;
                match_len = 0;
            }
        }

        for (int i = static_cast<int>(order.size()) - 1; i > 0; --i) {
            const int v = order[i];
            const int parent = states[v].link;
            if (parent >= 0) {
                const int propagated = std::min(best[v], states[parent].len);
                if (propagated > best[parent]) {
                    best[parent] = propagated;
                }
            }
        }

        for (std::size_t i = 0; i < states.size(); ++i) {
            if (best[i] < states[i].common) {
                states[i].common = best[i];
            }
        }
    }

    std::string answer_from(const std::string& base) const {
        int best_len = 0;
        int best_pos = -1;
        for (const auto& state : states) {
            if (state.common > best_len) {
                best_len = state.common;
                best_pos = state.first_pos;
            }
        }
        if (best_len <= 0) {
            return std::string();
        }
        return base.substr(best_pos - best_len + 1, best_len);
    }
};

void trim_trailing_cr(std::string& s) {
    if (!s.empty() && s.back() == '\r') {
        s.pop_back();
    }
}

std::string solve_tiny_bruteforce(const std::vector<std::string>& strings) {
    int base_index = 0;
    for (int i = 1; i < static_cast<int>(strings.size()); ++i) {
        if (strings[i].size() < strings[base_index].size()) {
            base_index = i;
        }
    }

    const std::string& base = strings[base_index];
    for (int len = static_cast<int>(base.size()); len >= 1; --len) {
        for (int start = 0; start + len <= static_cast<int>(base.size()); ++start) {
            const std::string candidate = base.substr(start, len);
            bool ok = true;
            for (int i = 0; i < static_cast<int>(strings.size()); ++i) {
                if (i == base_index) {
                    continue;
                }
                if (strings[i].find(candidate) == std::string::npos) {
                    ok = false;
                    break;
                }
            }
            if (ok) {
                return candidate;
            }
        }
    }
    return std::string();
}

std::string solve_two_string_dp(const std::string& a, const std::string& b) {
    const std::string* rows = &a;
    const std::string* cols = &b;
    bool answer_from_a = true;
    if (b.size() < a.size()) {
        rows = &b;
        cols = &a;
        answer_from_a = false;
    }

    std::vector<int> prev(cols->size() + 1, 0);
    std::vector<int> cur(cols->size() + 1, 0);

    int best_len = 0;
    int best_end = -1;
    for (std::size_t i = 1; i <= rows->size(); ++i) {
        for (std::size_t j = 1; j <= cols->size(); ++j) {
            if ((*rows)[i - 1] == (*cols)[j - 1]) {
                cur[j] = prev[j - 1] + 1;
                if (cur[j] > best_len) {
                    best_len = cur[j];
                    best_end = static_cast<int>(i) - 1;
                }
            } else {
                cur[j] = 0;
            }
        }
        std::swap(prev, cur);
        std::fill(cur.begin(), cur.end(), 0);
    }

    if (best_len == 0) {
        return std::string();
    }

    const std::string& src = answer_from_a ? a : b;
    return src.substr(best_end - best_len + 1, best_len);
}

std::string solve_with_suffix_automaton(std::vector<std::string> strings) {
    int base_index = 0;
    for (int i = 1; i < static_cast<int>(strings.size()); ++i) {
        if (strings[i].size() < strings[base_index].size()) {
            base_index = i;
        }
    }
    std::swap(strings[0], strings[base_index]);

    std::vector<int> order(strings.size() - 1);
    std::iota(order.begin(), order.end(), 1);
    std::sort(order.begin(), order.end(), [&](int lhs, int rhs) {
        return strings[lhs].size() < strings[rhs].size();
    });

    SuffixAutomaton sam(strings[0].size());
    sam.build(strings[0]);
    for (int idx : order) {
        sam.incorporate(strings[idx]);
        std::string().swap(strings[idx]);
    }
    return sam.answer_from(strings[0]);
}

std::string solve(std::vector<std::string> strings) {
    if (strings.empty()) {
        return std::string();
    }
    if (strings.size() == 1) {
        return strings[0];
    }

    std::size_t min_len = strings[0].size();
    std::uint64_t total_len = 0;
    for (const auto& s : strings) {
        min_len = std::min(min_len, s.size());
        total_len += static_cast<std::uint64_t>(s.size());
    }

    if (min_len == 0) {
        return std::string();
    }

    if (total_len <= 512 && min_len <= 64) {
        return solve_tiny_bruteforce(strings);
    }

    if (strings.size() == 2) {
        const std::uint64_t product =
            static_cast<std::uint64_t>(strings[0].size()) * static_cast<std::uint64_t>(strings[1].size());
        if (product <= 16'000'000ULL) {
            return solve_two_string_dp(strings[0], strings[1]);
        }
    }

    return solve_with_suffix_automaton(std::move(strings));
}

}  // namespace

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n = 0;
    if (!(std::cin >> n)) {
        return 0;
    }

    std::string line;
    std::getline(std::cin, line);

    std::vector<std::string> strings;
    strings.reserve(n);
    for (int i = 0; i < n; ++i) {
        std::getline(std::cin, line);
        trim_trailing_cr(line);
        strings.push_back(std::move(line));
    }

    std::cout << solve(std::move(strings)) << '\n';
    return 0;
}
