use std::f64::consts::PI;
use std::io::{self, BufRead, BufReader, Read, Write};
use std::sync::mpsc;
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Debug)]
struct Header {
    width: usize,
    height: usize,
    gravity: f64,
    max_thrust: f64,
    fuel: f64,
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
}

#[derive(Clone, Copy, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,

// Read the next whitespace-separated token from the BufRead stream without allocating huge buffers.
fn read_next_token<R: BufRead>(reader: &mut R) -> io::Result<Option<String>> {
    let mut token = Vec::<u8>::new();
    loop {
        let buf = reader.fill_buf()?;
        if buf.is_empty() {
            if token.is_empty() {
                return Ok(None);
            } else {
                // end of stream, return what we have
                let s = String::from_utf8(token).unwrap_or_default();
                return Ok(Some(s));
            }
        }
        let mut i = 0usize;
        // skip leading whitespace
        while i < buf.len() && buf[i].is_ascii_whitespace() {
            i += 1;
        }
        if i > 0 {
            reader.consume(i);
            continue;
        }
        // Now at first non-whitespace
        let mut j = 0usize;
        while j < buf.len() && !buf[j].is_ascii_whitespace() {
            j += 1;
        }
        // append [0..j]
        token.extend_from_slice(&buf[0..j]);
        reader.consume(j);
        // if token ended inside this buffer (next byte is whitespace or we've ended the buffer), return
        if j < buf.len() {
            let s = String::from_utf8(token).unwrap_or_default();
            return Ok(Some(s));
        }
        // else continue to read more chunks (token spans buffers)
    }
}

// Skip a single line (consume bytes up to and including the next '\n') using fill_buf to avoid allocating big strings
fn skip_line<R: BufRead>(reader: &mut R) -> io::Result<()> {
    loop {
        let buf = reader.fill_buf()?;
        if buf.is_empty() {
            return Ok(());
        }
        let mut found = false;
        for (i, &b) in buf.iter().enumerate() {
            if b == b'\n' {
                reader.consume(i + 1);
                found = true;
                break;
            }
        }
        if found {
            return Ok(());
        }
        // no newline in this chunk, consume all and continue
        let len = buf.len();
        reader.consume(len);
    }
}

fn reader_thread(header_tx: mpsc::Sender<Header>, state_tx: mpsc::Sender<State>) {

    // Read header tokens: width height gravity max_thrust fuel
    let mut got = Vec::new();

        match read_next_token(&mut reader) {
            Ok(Some(tok)) => got.push(tok),
            Ok(None) => {
                // EOF
                return;
            }
            Err(_) => return,
        }
    }
    let width: usize = got[0].parse().unwrap_or(0);
    let height: usize = got[1].parse().unwrap_or(0);
    let gravity: f64 = got[2].parse().unwrap_or(0.0);
    let max_thrust: f64 = got[3].parse().unwrap_or(0.0);
    let fuel: f64 = got[4].parse().unwrap_or(0.0);

    // Read start_x start_y target_x target_y (4 tokens)
    got.clear();
    while got.len() < 4 {
        match read_next_token(&mut reader) {
            Ok(Some(tok)) => got.push(tok),
            Ok(None) => return,
            Err(_) => return,
        }
    }
    let start_x: f64 = got[0].parse().unwrap_or(0.0);
    let start_y: f64 = got[1].parse().unwrap_or(0.0);
    let target_x: f64 = got[2].parse().unwrap_or(0.0);
    let target_y: f64 = got[3].parse().unwrap_or(0.0);

    // Skip the map rows without storing them to reduce memory usage.
    for _ in 0..height {
        if let Err(_) = skip_line(&mut reader) {
            // premature EOF; just return
            return;
        }
    }

    let header = Header {
        width,
        height,
        gravity,
        max_thrust,
        fuel,
        start_x,
        start_y,
        target_x,
        target_y,
    };

    // Send header to controller
    if header_tx.send(header).is_err() {
        return; // main thread not listening
    }

    // Now read continuous states as groups of 7 tokens
    let mut token_buf: Vec<String> = Vec::new();
    loop {
        match read_next_token(&mut reader) {
            Ok(Some(tok)) => {
                token_buf.push(tok);
                while token_buf.len() >= 7 {
                    // parse 7 tokens into a State
                    let sx: f64 = token_buf[0].parse().unwrap_or(0.0);
                    let sy: f64 = token_buf[1].parse().unwrap_or(0.0);
                    let svx: f64 = token_buf[2].parse().unwrap_or(0.0);
                    let svy: f64 = token_buf[3].parse().unwrap_or(0.0);
                    let sangle: f64 = token_buf[4].parse().unwrap_or(0.0);
                    let sang_vel: f64 = token_buf[5].parse().unwrap_or(0.0);
                    let sfuel: f64 = token_buf[6].parse().unwrap_or(0.0);
                    let st = State {
                        x: sx,
                        y: sy,
                        vx: svx,
                        vy: svy,
                        angle: sangle,
                        angular_vel: sang_vel,
                        fuel: sfuel,
                    };
                    // attempt to send; ignore errors (receiver could have been dropped)
                    let _ = state_tx.send(st);
                    // remove first 7 tokens
                    // use drain to avoid shifting costs too often; this should be OK because updates are frequent
                    token_buf.drain(0..7);
                }
            }
            Ok(None) => {
                // EOF
                return;
            }
            Err(_) => {
                // IO error
                return;
            }
        }
    }
}

fn normalize_angle(mut a: f64) -> f64 {
    while a <= -PI {
        a += 2.0 * PI;
    }
    while a > PI {
        a -= 2.0 * PI;
    }
    a
}

// Compute control outputs based on current state and header
fn compute_control(h: &Header, s: &State) -> (f64, f64) {
    // target position
    let dx = h.target_x - s.x;
    let dy = h.target_y - s.y; // positive if target below

    // altitude above target (positive when above target)
    let altitude = dy;

    // Estimate time horizon t to reach target vertically
    let mut t = 0.5_f64;
    if altitude <= 0.0 {
        t = 0.5;
    } else {
        if s.vy > 0.05 {
            // moving downward: estimate time to impact
            t = (altitude / s.vy).clamp(0.3, 6.0);
        } else {
            // not descending: estimate via free-fall or available thrust
            let g = h.gravity.max(1e-3);
            let t_est = (2.0 * altitude / g).sqrt();
            t = t_est.clamp(0.3, 6.0);
        }
    }

    // Compute desired net accelerations (including gravity in Y) to reach target in time t
    // Using constant-acceleration kinematic solve: a = 2*(pos_error - vel*t)/t^2
    let a_net_x = 2.0 * (dx - s.vx * t) / (t * t);
    let a_net_y = 2.0 * (dy - s.vy * t) / (t * t);

    // Convert net accel to thrust accel by removing gravity
    let a_thrust_x = a_net_x;
    let mut a_thrust_y = a_net_y - h.gravity;

    // Desired thrust magnitude
    let mut thrust_req = (a_thrust_x * a_thrust_x + a_thrust_y * a_thrust_y).sqrt();

    // If required thrust is larger than available, scale down the thrust vector
    let mut thrust_frac = (thrust_req / h.max_thrust).clamp(0.0, 1.0);
    if thrust_req > h.max_thrust && thrust_req > 0.0 {
        let scale = h.max_thrust / thrust_req;
        // scale thrust components so the orientation is consistent
        // (orientation computed from these scaled components)
        let ax = a_thrust_x * scale;
        a_thrust_y = a_thrust_y * scale;
        thrust_req = h.max_thrust;
        thrust_frac = 1.0;
        // compute desired angle from scaled components
        let desired_angle = (ax).atan2(-a_thrust_y);
        // turn PD
        let angle_error = normalize_angle(desired_angle - s.angle);
        let ang_kp = 2.0;
        let ang_kd = 0.5;
        let desired_ang_vel = (angle_error * ang_kp) - (s.angular_vel * ang_kd);
        let turn_cmd = desired_ang_vel.clamp(-1.0, 1.0);
        return (thrust_frac, turn_cmd);
    }

    // Compute desired angle from the (possibly unscaled) thrust components
    let desired_angle = if thrust_req > 0.0 {
        (a_thrust_x).atan2(-a_thrust_y)
    } else {
        0.0
    };

    // Small altitude heuristic: keep near-vertical attitude when very close
    let mut desired_angle_clamped = desired_angle;
    if altitude.abs() < 6.0 {
        // allow only small tilt near ground to prioritize vertical braking
        let max_tilt = 0.35; // radians (~20 deg)
        if desired_angle_clamped > max_tilt {
            desired_angle_clamped = max_tilt;
        } else if desired_angle_clamped < -max_tilt {
            desired_angle_clamped = -max_tilt;
        }
    }
    if altitude.abs() < 2.0 {
        // very close: force vertical attitude
        desired_angle_clamped = 0.0;
    }

    // Angle control: PD on angle error
    let angle_error = normalize_angle(desired_angle_clamped - s.angle);
    let ang_kp = 2.0; // proportional gain
    let ang_kd = 0.5; // derivative on angular velocity
    let desired_ang_vel = (angle_error * ang_kp) - (s.angular_vel * ang_kd);
    let turn_cmd = desired_ang_vel.clamp(-1.0, 1.0);

    // Thrust fraction already computed as thrust_req / max_thrust, but we can apply additional damping
    // For safety, if we're far above target allow gentle thrust; if we are close ensure thrust can slow us
    // Slight heuristic: when descending fast and near ground, cap thrust to 1.0 (already handled); when near and vertical,
    // ensure thrust is sufficient to keep vy under limit.

    // Another heuristic: if altitude is small and vy is small, reduce thrust to avoid bouncing
    if altitude <= 1.5 && s.vy.abs() < 1.0 {
        // small nudges only
        thrust_frac = thrust_frac.min(0.4);
    }

    (thrust_frac, turn_cmd)
}

fn main() {
    // Channel to receive header once
    let (header_tx, header_rx) = mpsc::channel::<Header>();
    // Channel to stream states
    let (state_tx, state_rx) = mpsc::channel::<State>();

    // Spawn reader thread to parse stdin
    let reader_handle = thread::spawn(move || {
        reader_thread(header_tx, state_tx);
    });

    // Wait for header
    let header = match header_rx.recv() {
        Ok(h) => h,
        Err(_) => {
            // reader died or closed
            return;
        }
    };

    // Control loop at >= 20Hz; use 25Hz (40ms) for margin
    let tick = Duration::from_millis(40);

    let stdout = io::stdout();
    let mut writer = stdout.lock();

    let mut latest_state: Option<State> = None;

    let mut next_instant = Instant::now();

    loop {
        // Drain state channel (non-blocking) to get the latest state
        while let Ok(st) = state_rx.try_recv() {
            latest_state = Some(st);
        }

        // Compute control based on latest known state
        let (thrust_cmd, turn_cmd) = if let Some(st) = latest_state {
            compute_control(&header, &st)
        } else {
            // No state yet: keep zero thrust and zero turn to be safe
            (0.0_f64, 0.0_f64)
        };

        // Output two numbers: thrust turn
        // Thrust: 0.0..1.0 ; turn: -1.0..1.0 (radians/sec)
        // Print with modest precision
        let out = format!("{:.4} {:.4}\n", thrust_cmd, turn_cmd);
        if writer.write_all(out.as_bytes()).is_err() {
            // if we can't write to stdout, just exit
            break;
        }
        if writer.flush().is_err() {
            break;
        }

        // Sleep until next tick but make sure we don't fall behind indefinitely
        next_instant += tick;
        let now = Instant::now();
        if next_instant > now {
            thread::sleep(next_instant - now);
        } else {
            // We're behind; skip sleeping and reset next_instant
            next_instant = now;
        }
    }

    // Try to join reader thread gracefully
    let _ = reader_handle.join();
}

    let target_x: f64 = got[2].parse().unwrap_or(0.0);
    let target_y: f64 = got[3].parse().unwrap_or(0.0);

    // Skip the map rows without storing them to reduce memory usage.
    for _ in 0..height {
        if let Err(_) = skip_line(&mut reader) {
            // premature EOF; just return
            return;
        }
    }

    let header = Header {
        width,
        height,
        gravity,
        max_thrust,
        fuel,
        start_x,
        start_y,
        target_x,
        target_y,
    };

    // Send header to controller
    if header_tx.send(header).is_err() {
        return; // main thread not listening
    }

    // Now read continuous states as groups of 7 tokens
    let mut token_buf: Vec<String> = Vec::new();
    loop {
        match read_next_token(&mut reader) {
            Ok(Some(tok)) => {
                token_buf.push(tok);
                while token_buf.len() >= 7 {
                    // parse 7 tokens into a State
                    let sx: f64 = token_buf[0].parse().unwrap_or(0.0);
                    let sy: f64 = token_buf[1].parse().unwrap_or(0.0);
                    let svx: f64 = token_buf[2].parse().unwrap_or(0.0);
                    let svy: f64 = token_buf[3].parse().unwrap_or(0.0);
                    let sangle: f64 = token_buf[4].parse().unwrap_or(0.0);
                    let sang_vel: f64 = token_buf[5].parse().unwrap_or(0.0);
                    let sfuel: f64 = token_buf[6].parse().unwrap_or(0.0);
                    let st = State {
                        x: sx,
                        y: sy,
                        vx: svx,
                        vy: svy,
                        angle: sangle,
                        angular_vel: sang_vel,
                        fuel: sfuel,
                    };
                    // attempt to send; ignore errors (receiver could have been dropped)
                    let _ = state_tx.send(st);
                    // remove first 7 tokens
                    // use drain to avoid shifting costs too often; this should be OK because updates are frequent
                    token_buf.drain(0..7);
                }
            }
            Ok(None) => {
                // EOF
                return;
            }
            Err(_) => {
                // IO error
                return;
            }
        }
    }
}

fn normalize_angle(mut a: f64) -> f64 {
    while a <= -PI {
        a += 2.0 * PI;
    }
    while a > PI {
        a -= 2.0 * PI;
    }
    a
}

// Compute control outputs based on current state and header
fn compute_control(h: &Header, s: &State) -> (f64, f64) {
    // target position
    let dx = h.target_x - s.x;
    let dy = h.target_y - s.y; // positive if target below

    // altitude above target (positive when above target)
    let altitude = dy;

    // Estimate time horizon t to reach target vertically
    let mut t = 0.5_f64;
    if altitude <= 0.0 {
        t = 0.5;
    } else {
        if s.vy > 0.05 {
            // moving downward: estimate time to impact
            t = (altitude / s.vy).clamp(0.3, 6.0);
        } else {
            // not descending: estimate via free-fall or available thrust
            let g = h.gravity.max(1e-3);
            let t_est = (2.0 * altitude / g).sqrt();
            t = t_est.clamp(0.3, 6.0);
        }
    }

    // Compute desired net accelerations (including gravity in Y) to reach target in time t
    // Using constant-acceleration kinematic solve: a = 2*(pos_error - vel*t)/t^2
    let a_net_x = 2.0 * (dx - s.vx * t) / (t * t);
    let a_net_y = 2.0 * (dy - s.vy * t) / (t * t);

    // Convert net accel to thrust accel by removing gravity
    let a_thrust_x = a_net_x;
    let mut a_thrust_y = a_net_y - h.gravity;

    // Desired thrust magnitude
    let mut thrust_req = (a_thrust_x * a_thrust_x + a_thrust_y * a_thrust_y).sqrt();

    // If required thrust is larger than available, scale down the thrust vector
    let mut thrust_frac = (thrust_req / h.max_thrust).clamp(0.0, 1.0);
    if thrust_req > h.max_thrust && thrust_req > 0.0 {
        let scale = h.max_thrust / thrust_req;
        // scale thrust components so the orientation is consistent
        // (orientation computed from these scaled components)
        let ax = a_thrust_x * scale;
        a_thrust_y = a_thrust_y * scale;
        thrust_req = h.max_thrust;
        thrust_frac = 1.0;
        // compute desired angle from scaled components
        let desired_angle = (ax).atan2(-a_thrust_y);
        // turn PD
        let angle_error = normalize_angle(desired_angle - s.angle);
        let ang_kp = 2.0;
        let ang_kd = 0.5;
        let desired_ang_vel = (angle_error * ang_kp) - (s.angular_vel * ang_kd);
        let turn_cmd = desired_ang_vel.clamp(-1.0, 1.0);
        return (thrust_frac, turn_cmd);
    }

    // Compute desired angle from the (possibly unscaled) thrust components
    let desired_angle = if thrust_req > 0.0 {
        (a_thrust_x).atan2(-a_thrust_y)
    } else {
        0.0
    };

    // Small altitude heuristic: keep near-vertical attitude when very close
    let mut desired_angle_clamped = desired_angle;
    if altitude.abs() < 6.0 {
        // allow only small tilt near ground to prioritize vertical braking
        let max_tilt = 0.35; // radians (~20 deg)
        if desired_angle_clamped > max_tilt {
            desired_angle_clamped = max_tilt;
        } else if desired_angle_clamped < -max_tilt {
            desired_angle_clamped = -max_tilt;
        }
    }
    if altitude.abs() < 2.0 {
        // very close: force vertical attitude
        desired_angle_clamped = 0.0;
    }

    // Angle control: PD on angle error
    let angle_error = normalize_angle(desired_angle_clamped - s.angle);
    let ang_kp = 2.0; // proportional gain
    let ang_kd = 0.5; // derivative on angular velocity
    let desired_ang_vel = (angle_error * ang_kp) - (s.angular_vel * ang_kd);
    let turn_cmd = desired_ang_vel.clamp(-1.0, 1.0);

    // Thrust fraction already computed as thrust_req / max_thrust, but we can apply additional damping
    // For safety, if we're far above target allow gentle thrust; if we are close ensure thrust can slow us
    // Slight heuristic: when descending fast and near ground, cap thrust to 1.0 (already handled); when near and vertical,
    // ensure thrust is sufficient to keep vy under limit.

    // Another heuristic: if altitude is small and vy is small, reduce thrust to avoid bouncing
    if altitude <= 1.5 && s.vy.abs() < 1.0 {
        // small nudges only
        thrust_frac = thrust_frac.min(0.4);
    }

    (thrust_frac, turn_cmd)
}

fn main() {
    // Channel to receive header once
    let (header_tx, header_rx) = mpsc::channel::<Header>();
    // Channel to stream states
    let (state_tx, state_rx) = mpsc::channel::<State>();

    // Spawn reader thread to parse stdin
    let reader_handle = thread::spawn(move || {
        reader_thread(header_tx, state_tx);
    });

    // Wait for header
    let header = match header_rx.recv() {
        Ok(h) => h,
        Err(_) => {
            // reader died or closed
            return;
        }
    };

    // Control loop at >= 20Hz; use 25Hz (40ms) for margin
    let tick = Duration::from_millis(40);

    let stdout = io::stdout();
    let mut writer = stdout.lock();

    let mut latest_state: Option<State> = None;

    let mut next_instant = Instant::now();

    loop {
        // Drain state channel (non-blocking) to get the latest state
        while let Ok(st) = state_rx.try_recv() {
            latest_state = Some(st);
        }

        // Compute control based on latest known state
        let (thrust_cmd, turn_cmd) = if let Some(st) = latest_state {
            compute_control(&header, &st)
        } else {
            // No state yet: keep zero thrust and zero turn to be safe
            (0.0_f64, 0.0_f64)
        };

        // Output two numbers: thrust turn
        // Thrust: 0.0..1.0 ; turn: -1.0..1.0 (radians/sec)
        // Print with modest precision
        let out = format!("{:.4} {:.4}\n", thrust_cmd, turn_cmd);
        if writer.write_all(out.as_bytes()).is_err() {
            // if we can't write to stdout, just exit
            break;
        }
        if writer.flush().is_err() {
            break;
        }

        // Sleep until next tick but make sure we don't fall behind indefinitely
        next_instant += tick;
        let now = Instant::now();
        if next_instant > now {
            thread::sleep(next_instant - now);
        } else {
            // We're behind; skip sleeping and reset next_instant
            next_instant = now;
        }
    }

    // Try to join reader thread gracefully
    let _ = reader_handle.join();
}
