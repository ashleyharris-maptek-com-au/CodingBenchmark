use std::io::{self, BufRead};
use std::f64::consts::PI;

#[derive(Debug, Clone)]
struct Point {
    x: f64,
    y: f64,
}

#[derive(Debug, Clone)]
struct Polygon {
    vertices: Vec<Point>,
}

#[derive(Debug, Clone)]
struct Piece {
    polygon: Polygon,
    index: usize,
}

#[derive(Debug, Clone)]
struct Placement {
    stock_index: usize,
    x: f64,
    y: f64,
    rotation: f64,
}

#[derive(Debug, Clone)]
struct Stock {
    polygon: Polygon,
    placements: Vec<(Piece, Placement)>,
}

impl Point {
    fn new(x: f64, y: f64) -> Self {
        Point { x, y }
    }

    fn rotate(&self, angle: f64) -> Point {
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        Point {
            x: self.x * cos_a - self.y * sin_a,
            y: self.x * sin_a + self.y * cos_a,
        }
    }

    fn translate(&self, dx: f64, dy: f64) -> Point {
        Point {
            x: self.x + dx,
            y: self.y + dy,
        }
    }

    fn distance_sq(&self, other: &Point) -> f64 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        dx * dx + dy * dy
    }
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        Polygon { vertices }
    }

    fn centroid(&self) -> Point {
        let n = self.vertices.len() as f64;
        let sum_x: f64 = self.vertices.iter().map(|p| p.x).sum();
        let sum_y: f64 = self.vertices.iter().map(|p| p.y).sum();
        Point::new(sum_x / n, sum_y / n)
    }

    fn translate_to_origin(&self) -> (Polygon, Point) {
        let c = self.centroid();
        let translated = Polygon::new(
            self.vertices.iter()
                .map(|p| Point::new(p.x - c.x, p.y - c.y))
                .collect()
        );
        (translated, c)
    }

    fn transform(&self, rotation: f64, translation: &Point) -> Polygon {
        Polygon::new(
            self.vertices.iter()
                .map(|p| p.rotate(rotation).translate(translation.x, translation.y))
                .collect()
        )
    }

    fn bounds(&self) -> (Point, Point) {
        let mut min_x = f64::INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for p in &self.vertices {
            min_x = min_x.min(p.x);
            min_y = min_y.min(p.y);
            max_x = max_x.max(p.x);
            max_y = max_y.max(p.y);
        }

        (Point::new(min_x, min_y), Point::new(max_x, max_y))
    }

    fn area(&self) -> f64 {
        let mut area = 0.0;
        let n = self.vertices.len();
        for i in 0..n {
            let j = (i + 1) % n;
            area += self.vertices[i].x * self.vertices[j].y;
            area -= self.vertices[j].x * self.vertices[i].y;
        }
        (area / 2.0).abs()
    }

    fn contains_point(&self, p: &Point) -> bool {
        let mut inside = false;
        let n = self.vertices.len();

        for i in 0..n {
            let j = (i + 1) % n;
            let vi = &self.vertices[i];
            let vj = &self.vertices[j];

            if ((vi.y > p.y) != (vj.y > p.y)) &&
               (p.x < (vj.x - vi.x) * (p.y - vi.y) / (vj.y - vi.y) + vi.x) {
                inside = !inside;
            }
        }
        inside
    }

    fn contains_polygon(&self, other: &Polygon) -> bool {
        for vertex in &other.vertices {
            if !self.contains_point(vertex) {
                return false;
            }
        }
        true
    }

    fn intersects(&self, other: &Polygon) -> bool {
        // Check if any edges intersect
        for i in 0..self.vertices.len() {
            let a1 = &self.vertices[i];
            let a2 = &self.vertices[(i + 1) % self.vertices.len()];

            for j in 0..other.vertices.len() {
                let b1 = &other.vertices[j];
                let b2 = &other.vertices[(j + 1) % other.vertices.len()];

                if segments_intersect(a1, a2, b1, b2) {
                    return true;
                }
            }
        }

        // Check if one polygon is inside the other
        if self.contains_point(&other.vertices[0]) || other.contains_point(&self.vertices[0]) {
            return true;
        }

        false
    }
}

fn segments_intersect(a1: &Point, a2: &Point, b1: &Point, b2: &Point) -> bool {
    let d1 = direction(b1, b2, a1);
    let d2 = direction(b1, b2, a2);
    let d3 = direction(a1, a2, b1);
    let d4 = direction(a1, a2, b2);

    if ((d1 > 0.0 && d2 < 0.0) || (d1 < 0.0 && d2 > 0.0)) &&
       ((d3 > 0.0 && d4 < 0.0) || (d3 < 0.0 && d4 > 0.0)) {
        return true;
    }

    if d1.abs() < 1e-10 && on_segment(b1, a1, b2) { return true; }
    if d2.abs() < 1e-10 && on_segment(b1, a2, b2) { return true; }
    if d3.abs() < 1e-10 && on_segment(a1, b1, a2) { return true; }
    if d4.abs() < 1e-10 && on_segment(a1, b2, a2) { return true; }

    false
}

fn direction(p1: &Point, p2: &Point, p3: &Point) -> f64 {
    (p3.x - p1.x) * (p2.y - p1.y) - (p2.x - p1.x) * (p3.y - p1.y)
}

fn on_segment(p1: &Point, p2: &Point, p3: &Point) -> bool {
    p2.x.min(p1.x.max(p3.x)) <= p2.x && p2.x <= p2.x.max(p1.x.min(p3.x)) &&
    p2.y.min(p1.y.max(p3.y)) <= p2.y && p2.y <= p2.y.max(p1.y.min(p3.y))
}

fn read_input() -> io::Result<(Polygon, Vec<Polygon>)> {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap()?;
    let nums: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let sv = nums[0];
    let np = nums[1];

    let mut stock_vertices = Vec::new();
    for _ in 0..sv {
        let line = lines.next().unwrap()?;
        let coords: Vec<f64> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        stock_vertices.push(Point::new(coords[0], coords[1]));
    }
    let stock = Polygon::new(stock_vertices);

    let mut pieces = Vec::new();
    for _ in 0..np {
        let line = lines.next().unwrap()?;
        let pv: usize = line.trim().parse().unwrap();

        let mut piece_vertices = Vec::new();
        for _ in 0..pv {
            let line = lines.next().unwrap()?;
            let coords: Vec<f64> = line.split_whitespace()
                .map(|s| s.parse().unwrap())
                .collect();
            piece_vertices.push(Point::new(coords[0], coords[1]));
        }
        pieces.push(Polygon::new(piece_vertices));
    }

    Ok((stock, pieces))
}

fn generate_rotation_angles(num_pieces: usize) -> Vec<f64> {
    if num_pieces < 50 {
        // Fine-grained rotation for small problems
        (0..24).map(|i| (i as f64) * PI / 12.0).collect()
    } else if num_pieces < 500 {
        // Medium granularity
        (0..8).map(|i| (i as f64) * PI / 4.0).collect()
    } else {
        // Coarse granularity for large problems
        vec![0.0, PI / 2.0, PI, 3.0 * PI / 2.0]
    }
}

fn try_place_piece(
    piece: &Polygon,
    stock: &Polygon,
    existing: &[(Polygon, Point)],
    rotation: f64,
    grid_size: f64,
) -> Option<(f64, f64)> {
    let (piece_centered, _) = piece.translate_to_origin();
    let rotated = piece_centered.transform(rotation, &Point::new(0.0, 0.0));

    let (stock_min, stock_max) = stock.bounds();
    let (piece_min, piece_max) = rotated.bounds();
    let piece_width = piece_max.x - piece_min.x;
    let piece_height = piece_max.y - piece_min.y;

    let start_x = stock_min.x - piece_min.x;
    let end_x = stock_max.x - piece_max.x;
    let start_y = stock_min.y - piece_min.y;
    let end_y = stock_max.y - piece_max.y;

    let mut candidates = Vec::new();

    // Grid-based placement attempts
    let x_steps = ((end_x - start_x) / grid_size).ceil() as usize + 1;
    let y_steps = ((end_y - start_y) / grid_size).ceil() as usize + 1;

    for xi in 0..x_steps.min(100) {
        for yi in 0..y_steps.min(100) {
            let tx = start_x + (xi as f64) * grid_size;
            let ty = start_y + (yi as f64) * grid_size;

            if tx < start_x || tx > end_x || ty < start_y || ty > end_y {
                continue;
            }

            let placed = rotated.transform(0.0, &Point::new(tx, ty));

            if !stock.contains_polygon(&placed) {
                continue;
            }

            let mut valid = true;
            for (existing_poly, _) in existing {
                if placed.intersects(existing_poly) {
                    valid = false;
                    break;
                }
            }

            if valid {
                candidates.push((tx, ty));
            }
        }
    }

    // Return bottom-left-most valid position
    candidates.sort_by(|a, b| {
        let score_a = a.1 * 1000.0 + a.0;
        let score_b = b.1 * 1000.0 + b.0;
        score_a.partial_cmp(&score_b).unwrap()
    });

    candidates.first().copied()
}

fn place_pieces(stock: &Polygon, pieces: Vec<Polygon>) -> Vec<Placement> {
    let num_pieces = pieces.len();
    let rotation_angles = generate_rotation_angles(num_pieces);

    // Adaptive grid size based on problem size
    let stock_area = stock.area();
    let grid_size = if num_pieces < 100 {
        (stock_area / 1000.0).sqrt()
    } else {
        (stock_area / 100.0).sqrt()
    };

    let mut stocks = vec![Stock {
        polygon: stock.clone(),
        placements: Vec::new(),
    }];

    let mut placements = vec![None; num_pieces];

    // Sort pieces by area (largest first) for better packing
    let mut indexed_pieces: Vec<_> = pieces.iter().enumerate().collect();
    indexed_pieces.sort_by(|a, b| b.1.area().partial_cmp(&a.1.area()).unwrap());

    for (original_idx, piece) in indexed_pieces {
        let mut placed = false;

        for stock_idx in 0..stocks.len() {
            let existing: Vec<_> = stocks[stock_idx].placements.iter()
                .map(|(p, pl)| {
                    let (centered, _) = p.polygon.translate_to_origin();
                    let transformed = centered.transform(pl.rotation, &Point::new(pl.x, pl.y));
                    (transformed, Point::new(pl.x, pl.y))
                })
                .collect();

            for &rotation in &rotation_angles {
                if let Some((x, y)) = try_place_piece(piece, &stock.polygon, &existing, rotation, grid_size) {
                    let placement = Placement {
                        stock_index: stock_idx,
                        x,
                        y,
                        rotation,
                    };

                    stocks[stock_idx].placements.push((
                        Piece { polygon: piece.clone(), index: original_idx },
                        placement.clone()
                    ));

                    placements[original_idx] = Some(placement);
                    placed = true;
                    break;
                }
            }

            if placed {
                break;
            }
        }

        if !placed {
            // Need a new stock
            stocks.push(Stock {
                polygon: stock.clone(),
                placements: Vec::new(),
            });

            let stock_idx = stocks.len() - 1;
            let rotation = 0.0;

            if let Some((x, y)) = try_place_piece(piece, &stock.polygon, &[], rotation, grid_size) {
                let placement = Placement {
                    stock_index: stock_idx,
                    x,
                    y,
                    rotation,
                };

                stocks[stock_idx].placements.push((
                    Piece { polygon: piece.clone(), index: original_idx },
                    placement.clone()
                ));

                placements[original_idx] = Some(placement);
            } else {
                // Fallback: place at stock centroid
                let center = stock.centroid();
                let placement = Placement {
                    stock_index: stock_idx,
                    x: center.x,
                    y: center.y,
                    rotation: 0.0,
                };

                stocks[stock_idx].placements.push((
                    Piece { polygon: piece.clone(), index: original_idx },
                    placement.clone()
                ));

                placements[original_idx] = Some(placement);
            }
        }
    }

    placements.into_iter().map(|p| p.unwrap()).collect()
}

fn main() -> io::Result<()> {
    let (stock, pieces) = read_input()?;
    let placements = place_pieces(&stock, pieces);

    let num_stocks = placements.iter().map(|p| p.stock_index).max().unwrap_or(0) + 1;
    println!("{}", num_stocks);

    for placement in placements {
        println!("{} {} {} {}",
            placement.stock_index,
            placement.x,
            placement.y,
            placement.rotation
        );
    }

    Ok(())
}
