use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

#[derive(Clone)]
struct GameInfo {
    width: f64,
    height: f64,
    gravity: f64,
    max_thrust: f64,
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Parse initial map
    let first_line = lines.next().unwrap().unwrap();
    let params: Vec<f64> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    let info = GameInfo {
        width: params[0],
        height: params[1],
        gravity: params[2],
        max_thrust: params[3],
        start_x: 0.0,
        start_y: 0.0,
        target_x: 0.0,
        target_y: 0.0,
    };

    let second_line = lines.next().unwrap().unwrap();
    let coords: Vec<f64> = second_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    let mut info = info;
    info.start_x = coords[0];
    info.start_y = coords[1];
    info.target_x = coords[2];
    info.target_y = coords[3];

    // Skip map rows
    for _ in 0..(info.height as usize) {
        if lines.next().is_none() {
            break;
        }
    }

    // Shared state
    let current_state = Arc::new(Mutex::new(State {
        x: info.start_x,
        y: info.start_y,
        vx: 0.0,
        vy: 0.0,
        angle: 0.0,
        angular_vel: 0.0,
        fuel: params[4],
    }));

    let state_for_reader = Arc::clone(&current_state);

    // Spawn reader thread
    thread::spawn(move || {
        for line in lines {
            if let Ok(line) = line {
                let values: Vec<f64> = line.split_whitespace()
                    .filter_map(|s| s.parse().ok())
                    .collect();

                if values.len() >= 7 {
                    let mut state = state_for_reader.lock().unwrap();
                    state.x = values[0];
                    state.y = values[1];
                    state.vx = values[2];
                    state.vy = values[3];
                    state.angle = values[4];
                    state.angular_vel = values[5];
                    state.fuel = values[6];
                }
            }
        }
    });

    // Main control loop - output at 50Hz
    let mut stdout = io::stdout();
    let output_interval = Duration::from_millis(20); // 50Hz

    loop {
        let start = Instant::now();

        let state = current_state.lock().unwrap().clone();
        drop(current_state.lock());

        // Calculate control commands
        let (thrust, turn) = calculate_control(&state, &info);

        // Output commands
        writeln!(stdout, "{:.6} {:.6}", thrust, turn).unwrap();
        stdout.flush().unwrap();

        // Sleep to maintain frequency
        let elapsed = start.elapsed();
        if elapsed < output_interval {
            thread::sleep(output_interval - elapsed);
        }
    }
}

fn calculate_control(state: &State, info: &GameInfo) -> (f64, f64) {
    if state.fuel <= 0.0 {
        return (0.0, 0.0);
    }

    // Calculate distance to target
    let dx = info.target_x - state.x;
    let dy = info.target_y - state.y;
    let dist = (dx * dx + dy * dy).sqrt();

    // Desired angle to target (0 = up, positive = clockwise)
    let target_angle = if dist > 5.0 {
        // Point toward target when far away
        let angle_to_target = dy.atan2(dx);
        // Convert to lander's coordinate system
        angle_to_target - std::f64::consts::FRAC_PI_2
    } else {
        // Point upward when close
        0.0
    };

    // Angle control with PID-like behavior
    let angle_error = target_angle - state.angle;
    let angle_error = normalize_angle(angle_error);

    // Turn control: proportional to angle error, derivative damping
    let kp_angle = 3.0;
    let kd_angle = 2.0;
    let mut turn = kp_angle * angle_error - kd_angle * state.angular_vel;
    turn = turn.clamp(-1.0, 1.0);

    // Thrust control
    let altitude = info.target_y - state.y;
    let velocity_magnitude = (state.vx * state.vx + state.vy * state.vy).sqrt();

    // Desired vertical velocity based on altitude
    let target_vy = if altitude > 0.0 {
        // Moving down toward target
        let safe_speed = (2.0 * altitude).sqrt().min(10.0);
        safe_speed
    } else {
        // Above target, slow down
        -2.0
    };

    // Vertical velocity error
    let vy_error = state.vy - target_vy;

    // Calculate thrust needed
    let cos_angle = state.angle.cos();

    // Base thrust to counteract gravity
    let gravity_compensation = if cos_angle.abs() > 0.1 {
        info.gravity / (info.max_thrust * cos_angle.abs())
    } else {
        1.0
    };

    // Add velocity correction
    let kp_vel = 0.3;
    let vel_correction = -kp_vel * vy_error / info.max_thrust;

    let mut thrust = gravity_compensation + vel_correction;

    // Aggressive landing phase
    if altitude.abs() < 10.0 && velocity_magnitude > 1.5 {
        thrust = 1.0;
    }

    // Fuel conservation when plenty of altitude
    if altitude > 100.0 && state.vy < 5.0 {
        thrust *= 0.5;
    }

    // Final approach: ensure soft landing
    if altitude.abs() < 5.0 {
        if state.vy > 1.0 {
            thrust = 1.0;
        } else if state.vy < -0.5 {
            thrust = 0.0;
        }
    }

    thrust = thrust.clamp(0.0, 1.0);

    (thrust, turn)
}

fn normalize_angle(angle: f64) -> f64 {
    let pi = std::f64::consts::PI;
    let mut a = angle % (2.0 * pi);
    if a > pi {
        a -= 2.0 * pi;
    } else if a < -pi {
        a += 2.0 * pi;
    }
    a
}
