using System;
using System.Collections.Generic;
using System.Linq;

public class OrbitalTSP
{
    const double MU = 398600.4418; // km³/s²
    const double EARTH_RADIUS = 6371.0; // km

    struct OrbitalState
    {
        public double[] Position; // x, y, z
        public double[] Velocity; // vx, vy, vz

        public OrbitalState(double x, double y, double z, double vx, double vy, double vz)
        {
            Position = new double[] { x, y, z };
            Velocity = new double[] { vx, vy, vz };
        }
    }

    static void Main()
    {
        // Read input
        string[] firstLine = Console.ReadLine().Split();
        int N = int.Parse(firstLine[0]);

        string[] startLine = Console.ReadLine().Split();
        OrbitalState start = new OrbitalState(
            double.Parse(startLine[0]), double.Parse(startLine[1]), double.Parse(startLine[2]),
            double.Parse(startLine[3]), double.Parse(startLine[4]), double.Parse(startLine[5])
        );

        List<OrbitalState> stations = new List<OrbitalState>();
        for (int i = 0; i < N; i++)
        {
            string[] line = Console.ReadLine().Split();
            stations.Add(new OrbitalState(
                double.Parse(line[0]), double.Parse(line[1]), double.Parse(line[2]),
                double.Parse(line[3]), double.Parse(line[4]), double.Parse(line[5])
            ));
        }

        // Solve TSP with appropriate algorithm based on N
        int[] solution;
        if (N <= 10)
        {
            solution = SolveSmall(start, stations);
        }
        else if (N <= 20)
        {
            solution = SolveMedium(start, stations);
        }
        else
        {
            solution = SolveLarge(start, stations);
        }

        // Output solution
        Console.WriteLine(string.Join(" ", solution));
    }

    // Brute force with branch and bound for small instances
    static int[] SolveSmall(OrbitalState start, List<OrbitalState> stations)
    {
        int N = stations.Count;
        double[,] distMatrix = BuildDistanceMatrix(start, stations);

        if (N <= 8)
        {
            return BranchAndBound(distMatrix, N);
        }
        else
        {
            return NearestNeighborWith2Opt(distMatrix, N);
        }
    }

    // Nearest neighbor + 2-opt for medium instances
    static int[] SolveMedium(OrbitalState start, List<OrbitalState> stations)
    {
        int N = stations.Count;
        double[,] distMatrix = BuildDistanceMatrix(start, stations);
        return NearestNeighborWith2Opt(distMatrix, N);
    }

    // Advanced heuristics for large instances
    static int[] SolveLarge(OrbitalState start, List<OrbitalState> stations)
    {
        int N = stations.Count;
        double[,] distMatrix = BuildDistanceMatrix(start, stations);

        // Use multiple strategies and pick best
        int[] best = NearestNeighborWith2Opt(distMatrix, N);
        double bestCost = CalculateTourCost(best, distMatrix);

        // Try multiple random starts
        Random rand = new Random(42);
        int iterations = Math.Min(50, 1000 / N);

        for (int iter = 0; iter < iterations; iter++)
        {
            int[] candidate = RandomTourWith2Opt(distMatrix, N, rand);
            double cost = CalculateTourCost(candidate, distMatrix);
            if (cost < bestCost)
            {
                best = candidate;
                bestCost = cost;
            }
        }

        return best;
    }

    static double[,] BuildDistanceMatrix(OrbitalState start, List<OrbitalState> stations)
    {
        int N = stations.Count;
        double[,] dist = new double[N + 1, N + 1];

        // Distance from start (index N) to each station
        for (int i = 0; i < N; i++)
        {
            dist[N, i] = EstimateDeltaV(start, stations[i]);
            dist[i, N] = dist[N, i];
        }

        // Distance between stations
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                if (i == j)
                    dist[i, j] = 0;
                else if (i < j)
                {
                    dist[i, j] = EstimateDeltaV(stations[i], stations[j]);
                    dist[j, i] = dist[i, j];
                }
            }
        }

        return dist;
    }

    static double EstimateDeltaV(OrbitalState from, OrbitalState to)
    {
        // Simple Euclidean distance in position-velocity space
        // Weighted to approximate delta-V requirements
        double posDiff = 0, velDiff = 0;

        for (int i = 0; i < 3; i++)
        {
            double dp = to.Position[i] - from.Position[i];
            double dv = to.Velocity[i] - from.Velocity[i];
            posDiff += dp * dp;
            velDiff += dv * dv;
        }

        // Delta-V is primarily driven by velocity differences
        // Position difference matters for phasing
        return Math.Sqrt(velDiff) + 0.01 * Math.Sqrt(posDiff);
    }

    static int[] NearestNeighborWith2Opt(double[,] dist, int N)
    {
        // Nearest neighbor starting from start position (index N)
        List<int> tour = new List<int>();
        bool[] visited = new bool[N];
        int current = N; // Start position

        for (int i = 0; i < N; i++)
        {
            int nearest = -1;
            double minDist = double.MaxValue;

            for (int j = 0; j < N; j++)
            {
                if (!visited[j] && dist[current, j] < minDist)
                {
                    minDist = dist[current, j];
                    nearest = j;
                }
            }

            tour.Add(nearest);
            visited[nearest] = true;
            current = nearest;
        }

        int[] result = tour.ToArray();
        TwoOpt(result, dist);
        return result;
    }

    static void TwoOpt(int[] tour, double[,] dist)
    {
        int N = tour.Length;
        bool improved = true;

        while (improved)
        {
            improved = false;

            for (int i = 0; i < N - 1; i++)
            {
                for (int j = i + 2; j < N; j++)
                {
                    double delta = Calculate2OptDelta(tour, dist, i, j);

                    if (delta < -1e-6)
                    {
                        Reverse(tour, i + 1, j);
                        improved = true;
                    }
                }
            }
        }
    }

    static double Calculate2OptDelta(int[] tour, double[,] dist, int i, int j)
    {
        int N = tour.Length;

        // Get indices
        int prev_i = (i == 0) ? tour.Length : tour[i - 1];
        int curr_i = tour[i];
        int curr_j = tour[j];
        int next_j = (j == N - 1) ? tour.Length : tour[j + 1];

        double oldCost = dist[prev_i, curr_i] + dist[curr_j, next_j];
        double newCost = dist[prev_i, curr_j] + dist[curr_i, next_j];

        return newCost - oldCost;
    }

    static void Reverse(int[] array, int start, int end)
    {
        while (start < end)
        {
            int temp = array[start];
            array[start] = array[end];
            array[end] = temp;
            start++;
            end--;
        }
    }

    static int[] RandomTourWith2Opt(double[,] dist, int N, Random rand)
    {
        List<int> stations = Enumerable.Range(0, N).ToList();

        // Shuffle
        for (int i = N - 1; i > 0; i--)
        {
            int j = rand.Next(i + 1);
            int temp = stations[i];
            stations[i] = stations[j];
            stations[j] = temp;
        }

        int[] tour = stations.ToArray();
        TwoOpt(tour, dist);
        return tour;
    }

    static double CalculateTourCost(int[] tour, double[,] dist)
    {
        int N = tour.Length;
        double cost = dist[N, tour[0]]; // Start to first station

        for (int i = 0; i < N - 1; i++)
        {
            cost += dist[tour[i], tour[i + 1]];
        }

        return cost;
    }

    static int[] BranchAndBound(double[,] dist, int N)
    {
        int[] bestTour = null;
        double bestCost = double.MaxValue;

        bool[] visited = new bool[N];
        int[] currentTour = new int[N];

        void Search(int depth, double currentCost)
        {
            if (depth == N)
            {
                if (currentCost < bestCost)
                {
                    bestCost = currentCost;
                    bestTour = (int[])currentTour.Clone();
                }
                return;
            }

            int prevStation = (depth == 0) ? N : currentTour[depth - 1];

            for (int i = 0; i < N; i++)
            {
                if (!visited[i])
                {
                    double newCost = currentCost + dist[prevStation, i];

                    // Bound: if current cost already exceeds best, prune
                    if (newCost < bestCost)
                    {
                        visited[i] = true;
                        currentTour[depth] = i;
                        Search(depth + 1, newCost);
                        visited[i] = false;
                    }
                }
            }
        }

        Search(0, 0);
        return bestTour ?? Enumerable.Range(0, N).ToArray();
    }
}
