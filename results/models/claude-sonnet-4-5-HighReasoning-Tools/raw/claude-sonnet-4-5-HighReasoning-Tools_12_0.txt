using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading.Tasks;

class Program
{
    static void Main()
    {
        var input = ReadInput();
        var solver = new BinPackingSolver(input);
        var result = solver.Solve();
        WriteOutput(result);
    }

    static Input ReadInput()
    {
        var line1 = Console.ReadLine().Split();
        int V = int.Parse(line1[0]);
        int F = int.Parse(line1[1]);

        var vertices = new Vector3[V];
        for (int i = 0; i < V; i++)
        {
            var parts = Console.ReadLine().Split();
            vertices[i] = new Vector3(
                double.Parse(parts[0]),
                double.Parse(parts[1]),
                double.Parse(parts[2])
            );
        }

        var faces = new Face[F];
        for (int i = 0; i < F; i++)
        {
            var parts = Console.ReadLine().Split();
            faces[i] = new Face(
                int.Parse(parts[0]),
                int.Parse(parts[1]),
                int.Parse(parts[2])
            );
        }

        var dims = Console.ReadLine().Split();
        double W = double.Parse(dims[0]);
        double H = double.Parse(dims[1]);
        double D = double.Parse(dims[2]);

        return new Input { Vertices = vertices, Faces = faces, ContainerSize = new Vector3(W, H, D) };
    }

    static void WriteOutput(List<Placement> placements)
    {
        Console.WriteLine(placements.Count);
        foreach (var p in placements)
        {
            Console.WriteLine($"{p.Translation.X} {p.Translation.Y} {p.Translation.Z} {p.Rotation.W} {p.Rotation.X} {p.Rotation.Y} {p.Rotation.Z}");
        }
    }
}

struct Vector3
{
    public double X, Y, Z;
    public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
    public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    public static Vector3 operator *(Vector3 a, double s) => new Vector3(a.X * s, a.Y * s, a.Z * s);
    public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);
}

struct Quaternion
{
    public double W, X, Y, Z;
    public Quaternion(double w, double x, double y, double z) { W = w; X = x; Y = y; Z = z; }

    public static Quaternion Identity => new Quaternion(1, 0, 0, 0);

    public Quaternion Conjugate() => new Quaternion(W, -X, -Y, -Z);

    public static Quaternion operator *(Quaternion q1, Quaternion q2)
    {
        return new Quaternion(
            q1.W * q2.W - q1.X * q2.X - q1.Y * q2.Y - q1.Z * q2.Z,
            q1.W * q2.X + q1.X * q2.W + q1.Y * q2.Z - q1.Z * q2.Y,
            q1.W * q2.Y - q1.X * q2.Z + q1.Y * q2.W + q1.Z * q2.X,
            q1.W * q2.Z + q1.X * q2.Y - q1.Y * q2.X + q1.Z * q2.W
        );
    }

    public Vector3 Rotate(Vector3 v)
    {
        Quaternion qv = new Quaternion(0, v.X, v.Y, v.Z);
        Quaternion result = this * qv * this.Conjugate();
        return new Vector3(result.X, result.Y, result.Z);
    }
}

struct Face
{
    public int A, B, C;
    public Face(int a, int b, int c) { A = a; B = b; C = c; }
}

struct AABB
{
    public Vector3 Min, Max;
    public AABB(Vector3 min, Vector3 max) { Min = min; Max = max; }

    public bool Intersects(AABB other)
    {
        return Min.X <= other.Max.X && Max.X >= other.Min.X &&
               Min.Y <= other.Max.Y && Max.Y >= other.Min.Y &&
               Min.Z <= other.Max.Z && Max.Z >= other.Min.Z;
    }

    public bool IsInside(Vector3 containerSize)
    {
        return Min.X >= 0 && Min.Y >= 0 && Min.Z >= 0 &&
               Max.X <= containerSize.X && Max.Y <= containerSize.Y && Max.Z <= containerSize.Z;
    }
}

class Placement
{
    public Vector3 Translation;
    public Quaternion Rotation;
    public AABB BoundingBox;
}

class Input
{
    public Vector3[] Vertices;
    public Face[] Faces;
    public Vector3 ContainerSize;
}

class BinPackingSolver
{
    private Input input;
    private List<Placement> placements = new List<Placement>();
    private Vector3 containerSize;
    private Quaternion[] candidateRotations;
    private double gridResolution;
    private int maxRotations;

    public BinPackingSolver(Input input)
    {
        this.input = input;
        this.containerSize = input.ContainerSize;
        DetermineStrategy();
    }

    void DetermineStrategy()
    {
        double volume = containerSize.X * containerSize.Y * containerSize.Z;
        double maxDim = Math.Max(containerSize.X, Math.Max(containerSize.Y, containerSize.Z));

        if (maxDim <= 10)
        {
            // Trivial/Small
            gridResolution = 0.5;
            maxRotations = 24;
            candidateRotations = GenerateRotations(24);
        }
        else if (maxDim <= 30)
        {
            // Medium
            gridResolution = 1.0;
            maxRotations = 8;
            candidateRotations = GenerateRotations(8);
        }
        else if (maxDim <= 100)
        {
            // Large
            gridResolution = 2.0;
            maxRotations = 4;
            candidateRotations = GenerateRotations(4);
        }
        else
        {
            // Extreme
            gridResolution = 5.0;
            maxRotations = 2;
            candidateRotations = GenerateRotations(2);
        }
    }

    Quaternion[] GenerateRotations(int count)
    {
        var rotations = new List<Quaternion> { Quaternion.Identity };

        if (count > 1)
        {
            // 90° rotations around each axis
            double s = Math.Sqrt(0.5);
            rotations.Add(new Quaternion(s, s, 0, 0)); // 90° X
            rotations.Add(new Quaternion(s, 0, s, 0)); // 90° Y
            rotations.Add(new Quaternion(s, 0, 0, s)); // 90° Z
        }

        if (count > 4)
        {
            // 180° rotations
            rotations.Add(new Quaternion(0, 1, 0, 0)); // 180° X
            rotations.Add(new Quaternion(0, 0, 1, 0)); // 180° Y
            rotations.Add(new Quaternion(0, 0, 0, 1)); // 180° Z
        }

        if (count > 7)
        {
            // 270° rotations
            double s = Math.Sqrt(0.5);
            rotations.Add(new Quaternion(s, -s, 0, 0)); // 270° X
        }

        if (count > 8)
        {
            // Additional diagonal rotations
            rotations.Add(new Quaternion(s, -0, s, 0)); // 270° Y
            rotations.Add(new Quaternion(s, 0, 0, -s)); // 270° Z

            // Combinations
            double h = 0.5;
            rotations.Add(new Quaternion(h, h, h, h));
            rotations.Add(new Quaternion(h, -h, h, h));
            rotations.Add(new Quaternion(h, h, -h, h));
            rotations.Add(new Quaternion(h, h, h, -h));
        }

        if (count > 15)
        {
            // More combinations for thorough coverage
            double h = 0.5;
            rotations.Add(new Quaternion(h, -h, -h, h));
            rotations.Add(new Quaternion(h, -h, h, -h));
            rotations.Add(new Quaternion(h, h, -h, -h));
            rotations.Add(new Quaternion(h, -h, -h, -h));
        }

        return rotations.Take(count).ToArray();
    }

    public List<Placement> Solve()
    {
        // Layer-by-layer greedy packing
        double z = 0;

        while (z < containerSize.Z)
        {
            double layerHeight = PackLayer(z);
            if (layerHeight <= 0) break;
            z += layerHeight;
        }

        return placements;
    }

    double PackLayer(double z)
    {
        double layerHeight = 0;
        double y = 0;

        while (y < containerSize.Y)
        {
            double x = 0;
            double rowHeight = 0;

            while (x < containerSize.X)
            {
                var placement = TryPlaceAt(new Vector3(x, y, z));
                if (placement != null)
                {
                    placements.Add(placement);
                    double width = placement.BoundingBox.Max.X - placement.BoundingBox.Min.X;
                    double height = placement.BoundingBox.Max.Y - placement.BoundingBox.Min.Y;
                    double depth = placement.BoundingBox.Max.Z - placement.BoundingBox.Min.Z;

                    x += width + gridResolution;
                    rowHeight = Math.Max(rowHeight, height);
                    layerHeight = Math.Max(layerHeight, depth);
                }
                else
                {
                    x += gridResolution;
                }
            }

            if (rowHeight <= 0) break;
            y += rowHeight + gridResolution;
        }

        return layerHeight;
    }

    Placement TryPlaceAt(Vector3 position)
    {
        foreach (var rotation in candidateRotations)
        {
            var placement = CreatePlacement(position, rotation);

            if (!placement.BoundingBox.IsInside(containerSize))
                continue;

            if (HasCollision(placement))
                continue;

            return placement;
        }

        return null;
    }

    Placement CreatePlacement(Vector3 translation, Quaternion rotation)
    {
        var transformed = input.Vertices.Select(v => rotation.Rotate(v) + translation).ToArray();

        var minX = transformed.Min(v => v.X);
        var minY = transformed.Min(v => v.Y);
        var minZ = transformed.Min(v => v.Z);
        var maxX = transformed.Max(v => v.X);
        var maxY = transformed.Max(v => v.Y);
        var maxZ = transformed.Max(v => v.Z);

        return new Placement
        {
            Translation = translation,
            Rotation = rotation,
            BoundingBox = new AABB(new Vector3(minX, minY, minZ), new Vector3(maxX, maxY, maxZ))
        };
    }

    bool HasCollision(Placement newPlacement)
    {
        // Quick AABB check first
        foreach (var existing in placements)
        {
            if (newPlacement.BoundingBox.Intersects(existing.BoundingBox))
            {
                // For more accurate collision, could do triangle-triangle tests
                // but AABB is sufficient for conservative packing
                return true;
            }
        }
        return false;
    }
}
