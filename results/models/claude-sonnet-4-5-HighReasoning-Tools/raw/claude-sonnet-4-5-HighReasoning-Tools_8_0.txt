use std::collections::{VecDeque, HashMap};
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read dimensions
    let first_line = lines.next().unwrap().unwrap();
    let dims: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let (h, w) = (dims[0], dims[1]);

    // Read maze and convert to bytes for O(1) access
    let mut maze = Vec::with_capacity(h);
    for _ in 0..h {
        let line = lines.next().unwrap().unwrap();
        maze.push(line.into_bytes());
    }

    // Find start and end positions
    let mut start = None;
    let mut end = None;

    for (y, row) in maze.iter().enumerate() {
        for (x, &ch) in row.iter().enumerate() {
            if ch == b'A' {
                start = Some((y, x));
            } else if ch == b'B' {
                end = Some((y, x));
            }
        }
    }

    if start.is_none() || end.is_none() {
        println!("0");
        return;
    }

    let start = start.unwrap();
    let end = end.unwrap();

    // Special case: start == end
    if start == end {
        println!("1");
        println!("{} {}", start.1, start.0);
        return;
    }

    // BFS to find shortest path
    let mut queue = VecDeque::new();
    let mut parent: HashMap<(usize, usize), Option<(usize, usize)>> = HashMap::new();

    queue.push_back(start);
    parent.insert(start, None);

    let directions = [(0, 1), (1, 0), (0, -1), (-1, 0)];

    while let Some((y, x)) = queue.pop_front() {
        if (y, x) == end {
            break;
        }

        for (dy, dx) in &directions {
            let ny = y as isize + dy;
            let nx = x as isize + dx;

            if ny < 0 || ny >= h as isize || nx < 0 || nx >= w as isize {
                continue;
            }

            let ny = ny as usize;
            let nx = nx as usize;

            if parent.contains_key(&(ny, nx)) {
                continue;
            }

            let cell = maze[ny][nx];
            if cell == b'#' {
                continue;
            }

            parent.insert((ny, nx), Some((y, x)));
            queue.push_back((ny, nx));
        }
    }

    // Reconstruct path
    if !parent.contains_key(&end) {
        println!("0");
        return;
    }

    let mut path = Vec::new();
    let mut current = end;
    path.push(current);

    while let Some(p) = parent.get(&current).unwrap() {
        current = *p;
        path.push(current);
    }

    path.reverse();

    // Output path: x (column) then y (row)
    println!("{}", path.len());
    for (y, x) in path {
        println!("{} {}", x, y);
    }
}
