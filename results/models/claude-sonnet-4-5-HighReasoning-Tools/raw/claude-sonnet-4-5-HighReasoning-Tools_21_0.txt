#include <iostream>
#include <vector>
#include <queue>
#include <unordered_set>
#include <unordered_map>
#include <algorithm>
#include <cmath>
#include <deque>

using namespace std;

// N-dimensional position
struct Pos {
    vector<int> coords;

    Pos() {}
    Pos(int D) : coords(D, 0) {}
    Pos(const vector<int>& c) : coords(c) {}

    bool operator==(const Pos& other) const {
        return coords == other.coords;
    }

    bool operator!=(const Pos& other) const {
        return coords != other.coords;
    }

    Pos move(int axis, int dir) const {
        Pos result = *this;
        result.coords[axis] += dir;
        return result;
    }

    int manhattanDist(const Pos& other) const {
        int dist = 0;
        for (size_t i = 0; i < coords.size(); i++) {
            dist += abs(coords[i] - other.coords[i]);
        }
        return dist;
    }
};

// Hash function for Pos
struct PosHash {
    size_t operator()(const Pos& p) const {
        size_t h = 0;
        for (size_t i = 0; i < p.coords.size(); i++) {
            h ^= hash<int>()(p.coords[i]) + 0x9e3779b9 + (h << 6) + (h >> 2);
        }
        return h;
    }
};

// Game state
struct GameState {
    int D;
    int max_turns;
    vector<int> bounds;
    deque<Pos> snake;
    vector<Pos> food;
    unordered_set<Pos, PosHash> obstacles;
    unordered_set<Pos, PosHash> foodSet;
    int turns_taken = 0;

    bool isValid(const Pos& p) const {
        for (int i = 0; i < D; i++) {
            if (p.coords[i] < 0 || p.coords[i] >= bounds[i]) {
                return false;
            }
        }
        return true;
    }

    bool isOccupied(const Pos& p) const {
        if (obstacles.count(p)) return true;
        for (const auto& seg : snake) {
            if (seg == p) return true;
        }
        return false;
    }

    void makeMove(int axis, int dir) {
        Pos newHead = snake.front().move(axis, dir);
        snake.push_front(newHead);

        bool ateFood = foodSet.count(newHead) > 0;
        if (ateFood) {
            foodSet.erase(newHead);
            food.erase(remove(food.begin(), food.end(), newHead), food.end());
        } else {
            snake.pop_back();
        }

        turns_taken++;
    }
};

// BFS to find path to target
pair<int, int> findPathToFood(const GameState& state, const Pos& target) {
    queue<pair<Pos, vector<pair<int, int>>>> q;
    unordered_set<Pos, PosHash> visited;

    q.push({state.snake.front(), {}});
    visited.insert(state.snake.front());

    // Create a set of snake body positions (excluding tail which will move)
    unordered_set<Pos, PosHash> snakeSet;
    for (size_t i = 0; i < state.snake.size() - 1; i++) {
        snakeSet.insert(state.snake[i]);
    }

    while (!q.empty()) {
        auto [current, path] = q.front();
        q.pop();

        if (current == target) {
            if (!path.empty()) {
                return path[0];
            }
            return {-1, 0};
        }

        // Try all possible moves
        for (int axis = 0; axis < state.D; axis++) {
            for (int dir = -1; dir <= 1; dir += 2) {
                Pos next = current.move(axis, dir);

                if (!state.isValid(next)) continue;
                if (state.obstacles.count(next)) continue;
                if (snakeSet.count(next)) continue;
                if (visited.count(next)) continue;

                visited.insert(next);
                vector<pair<int, int>> newPath = path;
                newPath.push_back({axis, dir});
                q.push({next, newPath});
            }
        }
    }

    return {-1, 0}; // No path found
}

// Find best move using multiple strategies
pair<int, int> findBestMove(GameState& state) {
    if (state.food.empty()) {
        return {-1, 0};
    }

    // Find nearest food
    Pos nearestFood = state.food[0];
    int minDist = state.snake.front().manhattanDist(nearestFood);
    for (const auto& f : state.food) {
        int dist = state.snake.front().manhattanDist(f);
        if (dist < minDist) {
            minDist = dist;
            nearestFood = f;
        }
    }

    // Try to find path to nearest food
    auto result = findPathToFood(state, nearestFood);
    if (result.first != -1) {
        return result;
    }

    // If no path to food, try any safe move
    vector<pair<int, int>> safeMoves;
    for (int axis = 0; axis < state.D; axis++) {
        for (int dir = -1; dir <= 1; dir += 2) {
            Pos next = state.snake.front().move(axis, dir);
            if (state.isValid(next) && !state.isOccupied(next)) {
                safeMoves.push_back({axis, dir});
            }
        }
    }

    if (!safeMoves.empty()) {
        // Pick move that keeps us away from walls and obstacles
        int bestScore = -1000000;
        pair<int, int> bestMove = safeMoves[0];

        for (const auto& move : safeMoves) {
            Pos next = state.snake.front().move(move.first, move.second);
            int score = 0;

            // Count available neighbors
            for (int axis = 0; axis < state.D; axis++) {
                for (int dir = -1; dir <= 1; dir += 2) {
                    Pos neighbor = next.move(axis, dir);
                    if (state.isValid(neighbor) && !state.obstacles.count(neighbor)) {
                        score++;
                    }
                }
            }

            // Prefer staying closer to food
            score -= next.manhattanDist(nearestFood);

            if (score > bestScore) {
                bestScore = score;
                bestMove = move;
            }
        }

        return bestMove;
    }

    // Desperate: just return any valid move direction
    for (int axis = 0; axis < state.D; axis++) {
        for (int dir = -1; dir <= 1; dir += 2) {
            Pos next = state.snake.front().move(axis, dir);
            if (state.isValid(next)) {
                return {axis, dir};
            }
        }
    }

    return {0, 1}; // Default fallback
}

int main() {
    GameState state;

    // Read input
    cin >> state.D >> state.max_turns;

    state.bounds.resize(state.D);
    for (int i = 0; i < state.D; i++) {
        cin >> state.bounds[i];
    }

    int snake_length;
    cin >> snake_length;

    Pos head(state.D);
    for (int i = 0; i < state.D; i++) {
        cin >> head.coords[i];
    }

    state.snake.push_back(head);

    int F;
    cin >> F;
    state.food.resize(F);
    for (int i = 0; i < F; i++) {
        state.food[i] = Pos(state.D);
        for (int j = 0; j < state.D; j++) {
            cin >> state.food[i].coords[j];
        }
        state.foodSet.insert(state.food[i]);
    }

    int O;
    cin >> O;
    for (int i = 0; i < O; i++) {
        Pos obs(state.D);
        for (int j = 0; j < state.D; j++) {
            cin >> obs.coords[j];
        }
        state.obstacles.insert(obs);
    }

    // Game loop
    while (state.turns_taken < state.max_turns && !state.food.empty()) {
        auto move = findBestMove(state);

        if (move.first == -1) {
            break; // No valid move
        }

        Pos nextPos = state.snake.front().move(move.first, move.second);
        if (!state.isValid(nextPos) || state.obstacles.count(nextPos)) {
            break; // Would hit wall or obstacle
        }

        // Check self-collision (excluding tail if not eating food)
        bool willEatFood = state.foodSet.count(nextPos) > 0;
        for (size_t i = 0; i < state.snake.size(); i++) {
            if (!willEatFood && i == state.snake.size() - 1) continue;
            if (state.snake[i] == nextPos) {
                goto game_over; // Self-collision
            }
        }

        cout << move.first << " " << move.second << endl;
        state.makeMove(move.first, move.second);
    }

    game_over:
    return 0;
}
