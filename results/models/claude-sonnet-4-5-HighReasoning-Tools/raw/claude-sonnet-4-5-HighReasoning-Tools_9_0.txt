using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        // Parse input
        var line1 = Console.ReadLine().Split();
        int n = int.Parse(line1[0]);
        int stockLength = int.Parse(line1[1]);

        var line2 = Console.ReadLine().Split();
        var indexedCuts = new (int len, int idx)[n];
        for (int i = 0; i < n; i++)
        {
            indexedCuts[i] = (int.Parse(line2[i]), i);
        }

        // Sort by length descending (First Fit Decreasing)
        Array.Sort(indexedCuts, (a, b) => b.len.CompareTo(a.len));

        // Assign cuts to stocks
        var stocks = new List<List<int>>();
        var capacities = new List<int>();

        foreach (var (len, idx) in indexedCuts)
        {
            // Find first stock with enough capacity
            int stockIdx = -1;
            for (int i = 0; i < capacities.Count; i++)
            {
                if (capacities[i] >= len)
                {
                    stockIdx = i;
                    break;
                }
            }

            if (stockIdx == -1)
            {
                // Create new stock
                stocks.Add(new List<int> { idx });
                capacities.Add(stockLength - len);
            }
            else
            {
                // Add to existing stock
                stocks[stockIdx].Add(idx);
                capacities[stockIdx] -= len;
            }
        }

        // Output result
        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            Console.WriteLine(string.Join(" ", stock));
        }
    }
}
