use std::io::{self, BufRead};
use std::collections::{HashSet, HashMap};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read first line
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let num_vertices = parts[0];
    let num_edges = parts[1];

    // Build adjacency list and edge list
    let mut adj: Vec<HashSet<usize>> = vec![HashSet::new(); num_vertices];
    let mut edges = Vec::with_capacity(num_edges);

    for _ in 0..num_edges {
        let line = lines.next().unwrap().unwrap();
        let edge: Vec<usize> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        let u = edge[0];
        let v = edge[1];
        edges.push((u, v));
        adj[u].insert(v);
        adj[v].insert(u);
    }

    // Greedy vertex cover: repeatedly pick vertex covering most uncovered edges
    let mut cover = HashSet::new();
    let mut uncovered_edges: HashSet<(usize, usize)> = edges.iter()
        .map(|&(u, v)| if u < v { (u, v) } else { (v, u) })
        .collect();

    while !uncovered_edges.is_empty() {
        // Count how many uncovered edges each vertex covers
        let mut vertex_coverage: HashMap<usize, usize> = HashMap::new();

        for &(u, v) in &uncovered_edges {
            *vertex_coverage.entry(u).or_insert(0) += 1;
            *vertex_coverage.entry(v).or_insert(0) += 1;
        }

        // Pick vertex covering the most uncovered edges
        let &best_vertex = vertex_coverage.iter()
            .max_by_key(|(_, &count)| count)
            .map(|(v, _)| v)
            .unwrap();

        cover.insert(best_vertex);

        // Remove all edges incident to this vertex
        uncovered_edges.retain(|&(u, v)| u != best_vertex && v != best_vertex);
    }

    // Output
    let mut cover_vec: Vec<usize> = cover.into_iter().collect();
    cover_vec.sort_unstable();

    println!("{}", cover_vec.len());
    for (i, v) in cover_vec.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
