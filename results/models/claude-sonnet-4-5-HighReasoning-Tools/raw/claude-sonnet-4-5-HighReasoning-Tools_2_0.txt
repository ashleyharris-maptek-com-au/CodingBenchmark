using System;
using System.Collections.Generic;
using System.Linq;

class ChinesePostman
{
    class Edge
    {
        public int To;
        public int Weight;
        public int Id;
        public Edge(int to, int weight, int id) { To = to; Weight = weight; Id = id; }
    }

    static void Main()
    {
        // Fast I/O
        var input = Console.In.ReadToEnd().Split(new[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        int n = int.Parse(input[idx++]);
        int m = int.Parse(input[idx++]);

        // Build adjacency list
        var graph = new List<Edge>[n];
        for (int i = 0; i < n; i++) graph[i] = new List<Edge>();

        int edgeIdCounter = 0;
        for (int i = 0; i < m; i++)
        {
            int u = int.Parse(input[idx++]);
            int v = int.Parse(input[idx++]);
            int w = int.Parse(input[idx++]);

            // Each undirected edge gets two directed edges with same ID
            graph[u].Add(new Edge(v, w, edgeIdCounter));
            graph[v].Add(new Edge(u, w, edgeIdCounter));
            edgeIdCounter++;
        }

        // Find vertices with odd degree
        var oddVertices = new List<int>();
        for (int i = 0; i < n; i++)
        {
            if (graph[i].Count % 2 == 1)
                oddVertices.Add(i);
        }

        // If there are odd vertices, need to add duplicate edges
        if (oddVertices.Count > 0)
        {
            // Compute shortest paths between all pairs of odd vertices
            var distances = new int[oddVertices.Count, oddVertices.Count];
            var paths = new List<int>[oddVertices.Count, oddVertices.Count];

            for (int i = 0; i < oddVertices.Count; i++)
            {
                var (dist, prev) = Dijkstra(graph, n, oddVertices[i]);
                for (int j = i + 1; j < oddVertices.Count; j++)
                {
                    distances[i, j] = distances[j, i] = dist[oddVertices[j]];
                    paths[i, j] = paths[j, i] = ReconstructPath(prev, oddVertices[i], oddVertices[j]);
                }
            }

            // Find minimum weight perfect matching on odd vertices
            var matching = MinWeightPerfectMatching(oddVertices.Count, distances);

            // Duplicate edges along matching paths
            foreach (var (i, j) in matching)
            {
                var path = paths[i, j];
                for (int k = 0; k < path.Count - 1; k++)
                {
                    int u = path[k];
                    int v = path[k + 1];

                    // Find the edge weight between u and v
                    int weight = graph[u].First(e => e.To == v).Weight;

                    // Add duplicate edges
                    graph[u].Add(new Edge(v, weight, edgeIdCounter));
                    graph[v].Add(new Edge(u, weight, edgeIdCounter));
                    edgeIdCounter++;
                }
            }
        }

        // Find Eulerian circuit starting from node 0
        var circuit = FindEulerianCircuit(graph, n, 0);

        // Output the circuit
        Console.WriteLine(string.Join(" ", circuit));
    }

    static (int[] dist, int[] prev) Dijkstra(List<Edge>[] graph, int n, int start)
    {
        var dist = new int[n];
        var prev = new int[n];
        Array.Fill(dist, int.MaxValue);
        Array.Fill(prev, -1);
        dist[start] = 0;

        var pq = new PriorityQueue<int, int>();
        pq.Enqueue(start, 0);

        while (pq.Count > 0)
        {
            int u = pq.Dequeue();

            foreach (var edge in graph[u])
            {
                int v = edge.To;
                int newDist = dist[u] + edge.Weight;
                if (newDist < dist[v])
                {
                    dist[v] = newDist;
                    prev[v] = u;
                    pq.Enqueue(v, newDist);
                }
            }
        }

        return (dist, prev);
    }

    static List<int> ReconstructPath(int[] prev, int start, int end)
    {
        var path = new List<int>();
        for (int at = end; at != -1; at = prev[at])
            path.Add(at);
        path.Reverse();
        return path;
    }

    static List<(int, int)> MinWeightPerfectMatching(int k, int[,] dist)
    {
        // DP with bitmask for small k (up to ~20)
        // dp[mask] = minimum cost to match vertices in mask
        int fullMask = (1 << k) - 1;
        var dp = new int[1 << k];
        var parent = new (int, int)[1 << k];
        Array.Fill(dp, int.MaxValue);
        dp[0] = 0;

        for (int mask = 0; mask < (1 << k); mask++)
        {
            if (dp[mask] == int.MaxValue) continue;

            // Find first unmatched vertex
            int first = -1;
            for (int i = 0; i < k; i++)
            {
                if ((mask & (1 << i)) == 0)
                {
                    first = i;
                    break;
                }
            }

            if (first == -1) continue;

            // Try pairing with all other unmatched vertices
            for (int second = first + 1; second < k; second++)
            {
                if ((mask & (1 << second)) == 0)
                {
                    int newMask = mask | (1 << first) | (1 << second);
                    int newCost = dp[mask] + dist[first, second];
                    if (newCost < dp[newMask])
                    {
                        dp[newMask] = newCost;
                        parent[newMask] = (first, second);
                    }
                }
            }
        }

        // Reconstruct matching
        var matching = new List<(int, int)>();
        int curMask = fullMask;
        while (curMask > 0)
        {
            var (i, j) = parent[curMask];
            matching.Add((i, j));
            curMask ^= (1 << i) | (1 << j);
        }

        return matching;
    }

    static List<int> FindEulerianCircuit(List<Edge>[] graph, int n, int start)
    {
        // Hierholzer's algorithm
        var usedEdges = new HashSet<int>();
        var stack = new Stack<int>();
        var circuit = new List<int>();

        // Track current position in adjacency list for each node
        var edgeIndex = new int[n];

        stack.Push(start);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            bool foundEdge = false;

            // Find an unused edge from u
            while (edgeIndex[u] < graph[u].Count)
            {
                var edge = graph[u][edgeIndex[u]];
                edgeIndex[u]++;

                if (!usedEdges.Contains(edge.Id))
                {
                    usedEdges.Add(edge.Id);
                    stack.Push(edge.To);
                    foundEdge = true;
                    break;
                }
            }

            if (!foundEdge)
            {
                circuit.Add(stack.Pop());
            }
        }

        circuit.Reverse();
        return circuit;
    }
}
