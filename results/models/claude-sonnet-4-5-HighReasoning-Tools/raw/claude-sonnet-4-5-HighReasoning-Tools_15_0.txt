use std::io::{self, BufRead};
use std::f64::consts::PI;

#[derive(Clone, Copy, Debug)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Vec3 { x, y, z }
    }

    fn dot(&self, other: &Vec3) -> f64 {
        self.x * other.x + self.y * other.y + self.z * other.z
    }

    fn cross(&self, other: &Vec3) -> Vec3 {
        Vec3::new(
            self.y * other.z - self.z * other.y,
            self.z * other.x - self.x * other.z,
            self.x * other.y - self.y * other.x,
        )
    }

    fn length(&self) -> f64 {
        (self.x * self.x + self.y * self.y + self.z * self.z).sqrt()
    }

    fn normalize(&self) -> Vec3 {
        let len = self.length();
        if len > 1e-10 {
            Vec3::new(self.x / len, self.y / len, self.z / len)
        } else {
            Vec3::new(0.0, 0.0, 1.0)
        }
    }

    fn add(&self, other: &Vec3) -> Vec3 {
        Vec3::new(self.x + other.x, self.y + other.y, self.z + other.z)
    }
}

#[derive(Clone, Copy, Debug)]
struct Quaternion {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

impl Quaternion {
    fn identity() -> Self {
        Quaternion { w: 1.0, x: 0.0, y: 0.0, z: 0.0 }
    }

    fn from_axis_angle(axis: Vec3, angle: f64) -> Self {
        let half_angle = angle / 2.0;
        let s = half_angle.sin();
        Quaternion {
            w: half_angle.cos(),
            x: axis.x * s,
            y: axis.y * s,
            z: axis.z * s,
        }
    }

    fn from_rotation_between(from: Vec3, to: Vec3) -> Self {
        let from_n = from.normalize();
        let to_n = to.normalize();
        let dot = from_n.dot(&to_n);

        if dot > 0.9999 {
            return Quaternion::identity();
        }

        if dot < -0.9999 {
            let axis = Vec3::new(1.0, 0.0, 0.0).cross(&from_n);
            let axis = if axis.length() < 1e-6 {
                Vec3::new(0.0, 1.0, 0.0).cross(&from_n)
            } else {
                axis
            };
            return Quaternion::from_axis_angle(axis.normalize(), PI);
        }

        let axis = from_n.cross(&to_n);
        let s = ((1.0 + dot) * 2.0).sqrt();
        let invs = 1.0 / s;

        Quaternion {
            w: s * 0.5,
            x: axis.x * invs,
            y: axis.y * invs,
            z: axis.z * invs,
        }
    }
}

struct Tetrahedron {
    position: Vec3,
    rotation: Quaternion,
    scale: f64,
}

fn compute_polygon_bounds(vertices: &[Vec3]) -> (Vec3, Vec3, Vec3) {
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;

    for v in vertices {
        min_x = min_x.min(v.x);
        max_x = max_x.max(v.x);
        min_y = min_y.min(v.y);
        max_y = max_y.max(v.y);
    }

    let centroid = Vec3::new(
        vertices.iter().map(|v| v.x).sum::<f64>() / vertices.len() as f64,
        vertices.iter().map(|v| v.y).sum::<f64>() / vertices.len() as f64,
        0.0,
    );

    let bbox_min = Vec3::new(min_x, min_y, 0.0);
    let bbox_max = Vec3::new(max_x, max_y, 0.0);

    (centroid, bbox_min, bbox_max)
}

fn compute_polygon_area(vertices: &[Vec3]) -> f64 {
    let mut area = 0.0;
    let n = vertices.len();
    for i in 0..n {
        let j = (i + 1) % n;
        area += vertices[i].x * vertices[j].y;
        area -= vertices[j].x * vertices[i].y;
    }
    area.abs() / 2.0
}

fn solve_shadow_covering(vertices: Vec<Vec3>, sun_dir: Vec3) -> Vec<Tetrahedron> {
    let (centroid, bbox_min, bbox_max) = compute_polygon_bounds(&vertices);

    let width = bbox_max.x - bbox_min.x;
    let height = bbox_max.y - bbox_min.y;
    let bbox_size = width.max(height);

    let area = compute_polygon_area(&vertices);
    let num_vertices = vertices.len();

    let sun_normalized = sun_dir.normalize();

    // Determine complexity
    let is_simple = num_vertices <= 6 && area < 100.0;
    let is_medium = num_vertices <= 20 && area < 1000.0;

    let mut tetrahedra = Vec::new();

    if is_simple {
        // Simple case: use 1-2 large tetrahedrons
        let scale = (bbox_size / 2.0).min(10.0).max(1.0);

        // Position above centroid
        let height = scale * 2.0;
        let position = centroid.add(&Vec3::new(0.0, 0.0, height));

        // Orient to face sun direction
        let up = Vec3::new(0.0, 0.0, 1.0);
        let rotation = if sun_normalized.z.abs() > 0.99 {
            Quaternion::identity()
        } else {
            let target_up = Vec3::new(-sun_normalized.x, -sun_normalized.y, -sun_normalized.z);
            Quaternion::from_rotation_between(up, target_up)
        };

        tetrahedra.push(Tetrahedron {
            position,
            rotation,
            scale,
        });

    } else if is_medium {
        // Medium complexity: use 2-4 tetrahedrons in a grid pattern
        let grid_size = if num_vertices <= 12 { 2 } else { 2 };
        let scale = (bbox_size / (grid_size as f64 * 1.5)).min(10.0).max(1.0);
        let height = scale * 2.5;

        let up = Vec3::new(0.0, 0.0, 1.0);
        let rotation = if sun_normalized.z.abs() > 0.99 {
            Quaternion::identity()
        } else {
            let target_up = Vec3::new(-sun_normalized.x, -sun_normalized.y, -sun_normalized.z);
            Quaternion::from_rotation_between(up, target_up)
        };

        for i in 0..grid_size {
            for j in 0..grid_size {
                let offset_x = (i as f64 - (grid_size as f64 - 1.0) / 2.0) * bbox_size / grid_size as f64;
                let offset_y = (j as f64 - (grid_size as f64 - 1.0) / 2.0) * bbox_size / grid_size as f64;

                let position = Vec3::new(
                    centroid.x + offset_x,
                    centroid.y + offset_y,
                    height,
                );

                tetrahedra.push(Tetrahedron {
                    position,
                    rotation,
                    scale,
                });
            }
        }

    } else {
        // Complex case: use a finer grid
        let grid_size = ((num_vertices as f64 / 5.0).sqrt().ceil() as usize).max(3).min(8);
        let scale = (bbox_size / (grid_size as f64 * 1.2)).min(10.0).max(0.5);
        let height = scale * 2.0;

        let up = Vec3::new(0.0, 0.0, 1.0);
        let rotation = if sun_normalized.z.abs() > 0.99 {
            Quaternion::identity()
        } else {
            let target_up = Vec3::new(-sun_normalized.x, -sun_normalized.y, -sun_normalized.z);
            Quaternion::from_rotation_between(up, target_up)
        };

        // Create overlapping coverage
        let step = bbox_size / grid_size as f64;
        let overlap = 0.3;

        for i in 0..grid_size {
            for j in 0..grid_size {
                let offset_x = bbox_min.x + (i as f64 + 0.5) * step;
                let offset_y = bbox_min.y + (j as f64 + 0.5) * step;

                let position = Vec3::new(offset_x, offset_y, height);

                tetrahedra.push(Tetrahedron {
                    position,
                    rotation,
                    scale: scale * (1.0 + overlap),
                });
            }
        }
    }

    tetrahedra
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read number of vertices
    let nv: usize = lines
        .next()
        .unwrap()
        .unwrap()
        .trim()
        .parse()
        .unwrap();

    // Read polygon vertices
    let mut vertices = Vec::with_capacity(nv);
    for _ in 0..nv {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<f64> = line
            .split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        vertices.push(Vec3::new(parts[0], parts[1], 0.0));
    }

    // Read sun direction
    let line = lines.next().unwrap().unwrap();
    let parts: Vec<f64> = line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let sun_dir = Vec3::new(parts[0], parts[1], parts[2]);

    // Solve the problem
    let tetrahedra = solve_shadow_covering(vertices, sun_dir);

    // Output results
    println!("{}", tetrahedra.len());
    for tet in &tetrahedra {
        println!(
            "{} {} {} {} {} {} {} {}",
            tet.position.x,
            tet.position.y,
            tet.position.z,
            tet.rotation.w,
            tet.rotation.x,
            tet.rotation.y,
            tet.rotation.z,
            tet.scale
        );
    }
}
