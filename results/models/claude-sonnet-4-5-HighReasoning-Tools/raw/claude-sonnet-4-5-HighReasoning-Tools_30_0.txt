import sys
from collections import defaultdict, deque

def solve_3sat():
    # Read input
    line = sys.stdin.readline().split()
    num_vars = int(line[0])
    num_clauses = int(line[1])

    clauses = []
    for _ in range(num_clauses):
        clause = list(map(int, sys.stdin.readline().split()))
        clauses.append(clause)

    # Initialize assignment (None = unassigned, True/False for assigned)
    assignment = [None] * (num_vars + 1)

    # Track which clauses are satisfied
    satisfied = [False] * num_clauses

    # Build occurrence lists
    pos_occ = defaultdict(set)
    neg_occ = defaultdict(set)

    for idx, clause in enumerate(clauses):
        for lit in clause:
            if lit > 0:
                pos_occ[lit].add(idx)
            else:
                neg_occ[-lit].add(idx)

    # Unit propagation with queue
    unit_queue = deque()

    # Find initial unit clauses and propagate
    def find_units():
        for idx, clause in enumerate(clauses):
            if satisfied[idx]:
                continue
            unassigned = []
            for lit in clause:
                var = abs(lit)
                if assignment[var] is None:
                    unassigned.append(lit)
                elif (lit > 0 and assignment[var]) or (lit < 0 and not assignment[var]):
                    satisfied[idx] = True
                    break
            if not satisfied[idx] and len(unassigned) == 1:
                unit_queue.append(unassigned[0])

    find_units()

    def propagate():
        while unit_queue:
            lit = unit_queue.popleft()
            var = abs(lit)

            if assignment[var] is not None:
                continue

            # Assign variable
            assignment[var] = (lit > 0)

            # Mark clauses with this literal as satisfied
            clauses_to_satisfy = pos_occ[var] if assignment[var] else neg_occ[var]
            for idx in clauses_to_satisfy:
                satisfied[idx] = True

            # Check clauses with opposite literal
            clauses_to_check = neg_occ[var] if assignment[var] else pos_occ[var]
            for idx in clauses_to_check:
                if satisfied[idx]:
                    continue

                unassigned = []
                for l in clauses[idx]:
                    v = abs(l)
                    if assignment[v] is None:
                        unassigned.append(l)
                    elif (l > 0 and assignment[v]) or (l < 0 and not assignment[v]):
                        satisfied[idx] = True
                        break

                if not satisfied[idx] and len(unassigned) == 1:
                    unit_queue.append(unassigned[0])

    propagate()

    # Greedy assignment for remaining variables
    for var in range(1, num_vars + 1):
        if assignment[var] is not None:
            continue

        # Count occurrences in unsatisfied clauses
        pos_count = sum(1 for idx in pos_occ[var] if not satisfied[idx])
        neg_count = sum(1 for idx in neg_occ[var] if not satisfied[idx])

        # Assign to satisfy more clauses (default to True if equal)
        assignment[var] = (pos_count >= neg_count)

        # Mark satisfied clauses
        if assignment[var]:
            for idx in pos_occ[var]:
                satisfied[idx] = True
        else:
            for idx in neg_occ[var]:
                satisfied[idx] = True

    # Output
    print("SAT")
    result = []
    for i in range(1, num_vars + 1):
        if assignment[i] is True or assignment[i] is None:
            result.append('1')
        else:
            result.append('0')
    print(' '.join(result))

if __name__ == "__main__":
    solve_3sat()
