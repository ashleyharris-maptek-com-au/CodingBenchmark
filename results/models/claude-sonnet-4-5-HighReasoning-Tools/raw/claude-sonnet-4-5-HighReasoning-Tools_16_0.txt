#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <cstring>

using namespace std;

struct Task {
    int job;
    int task_idx;
    int machine;
    int duration;
    int start_time;
};

struct Job {
    vector<Task> tasks;
    int next_task;
};

// Priority queue comparator for tasks
struct TaskComparator {
    bool operator()(const Task* a, const Task* b) const {
        // Shortest Processing Time first, then FIFO by job ID
        if (a->duration != b->duration) {
            return a->duration > b->duration;
        }
        return a->job > b->job;
    }
};

class JobShopScheduler {
private:
    int num_jobs;
    int num_machines;
    vector<Job> jobs;
    vector<int> machine_available_time;
    vector<int> job_available_time;

public:
    void read_input() {
        cin >> num_jobs >> num_machines;

        jobs.resize(num_jobs);
        machine_available_time.resize(num_machines, 0);
        job_available_time.resize(num_jobs, 0);

        for (int j = 0; j < num_jobs; j++) {
            int num_tasks;
            cin >> num_tasks;
            jobs[j].tasks.resize(num_tasks);
            jobs[j].next_task = 0;

            for (int t = 0; t < num_tasks; t++) {
                int machine, duration;
                cin >> machine >> duration;
                jobs[j].tasks[t] = {j, t, machine, duration, -1};
            }
        }
    }

    void solve_greedy_spt() {
        // Shortest Processing Time greedy heuristic
        priority_queue<Task*, vector<Task*>, TaskComparator> ready_queue;

        // Initialize with first task of each job
        for (int j = 0; j < num_jobs; j++) {
            if (jobs[j].next_task < (int)jobs[j].tasks.size()) {
                ready_queue.push(&jobs[j].tasks[jobs[j].next_task]);
            }
        }

        while (!ready_queue.empty()) {
            Task* task = ready_queue.top();
            ready_queue.pop();

            int job_id = task->job;
            int machine = task->machine;

            // Schedule this task
            int earliest_start = max(job_available_time[job_id],
                                    machine_available_time[machine]);
            task->start_time = earliest_start;

            // Update availability times
            int finish_time = earliest_start + task->duration;
            job_available_time[job_id] = finish_time;
            machine_available_time[machine] = finish_time;

            // Add next task from this job to ready queue
            jobs[job_id].next_task++;
            if (jobs[job_id].next_task < (int)jobs[job_id].tasks.size()) {
                ready_queue.push(&jobs[job_id].tasks[jobs[job_id].next_task]);
            }
        }
    }

    void solve_fifo() {
        // First-In-First-Out scheduling
        queue<Task*> ready_queue;

        // Initialize with first task of each job
        for (int j = 0; j < num_jobs; j++) {
            if (jobs[j].next_task < (int)jobs[j].tasks.size()) {
                ready_queue.push(&jobs[j].tasks[jobs[j].next_task]);
            }
        }

        while (!ready_queue.empty()) {
            Task* task = ready_queue.front();
            ready_queue.pop();

            int job_id = task->job;
            int machine = task->machine;

            // Schedule this task
            int earliest_start = max(job_available_time[job_id],
                                    machine_available_time[machine]);
            task->start_time = earliest_start;

            // Update availability times
            int finish_time = earliest_start + task->duration;
            job_available_time[job_id] = finish_time;
            machine_available_time[machine] = finish_time;

            // Add next task from this job to ready queue
            jobs[job_id].next_task++;
            if (jobs[job_id].next_task < (int)jobs[job_id].tasks.size()) {
                ready_queue.push(&jobs[job_id].tasks[jobs[job_id].next_task]);
            }
        }
    }

    int get_makespan() {
        int makespan = 0;
        for (const auto& job : jobs) {
            for (const auto& task : job.tasks) {
                makespan = max(makespan, task.start_time + task.duration);
            }
        }
        return makespan;
    }

    void reset() {
        fill(machine_available_time.begin(), machine_available_time.end(), 0);
        fill(job_available_time.begin(), job_available_time.end(), 0);
        for (auto& job : jobs) {
            job.next_task = 0;
            for (auto& task : job.tasks) {
                task.start_time = -1;
            }
        }
    }

    void solve() {
        int total_tasks = 0;
        for (const auto& job : jobs) {
            total_tasks += job.tasks.size();
        }

        // Adaptive strategy based on problem size
        if (total_tasks < 100) {
            // Small problem: try multiple heuristics and pick best
            solve_greedy_spt();
            int spt_makespan = get_makespan();

            // Save SPT solution
            vector<Job> spt_jobs = jobs;

            // Try FIFO
            reset();
            solve_fifo();
            int fifo_makespan = get_makespan();

            // Use better solution
            if (spt_makespan <= fifo_makespan) {
                jobs = spt_jobs;
            }
        } else {
            // Large problem: use fast SPT heuristic
            solve_greedy_spt();
        }
    }

    void output() {
        for (const auto& job : jobs) {
            bool first = true;
            for (const auto& task : job.tasks) {
                if (!first) cout << " ";
                cout << task.start_time << " " << task.machine;
                first = false;
            }
            cout << "\n";
        }
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    JobShopScheduler scheduler;
    scheduler.read_input();
    scheduler.solve();
    scheduler.output();

    return 0;
}
