using System;
using System.Collections.Generic;
using System.Linq;

class TetrisPlayer
{
    static void Main()
    {
        var input = Console.ReadLine().Split();
        int width = int.Parse(input[0]);
        int height = int.Parse(input[1]);
        int numPieces = int.Parse(input[2]);

        var game = new TetrisGame(width, height);

        for (int i = 0; i < numPieces; i++)
        {
            string pieceType = Console.ReadLine().Trim();
            var (rotation, column) = game.FindBestPlacement(pieceType);
            Console.WriteLine($"{rotation} {column}");
            Console.Out.Flush();

            string result = Console.ReadLine().Trim();
            game.ApplyPlacement(pieceType, rotation, column, result);
        }
    }
}

class TetrisGame
{
    private int width;
    private int height;
    private int[,] board;
    private Dictionary<string, List<int[,]>> pieces;

    public TetrisGame(int width, int height)
    {
        this.width = width;
        this.height = height;
        this.board = new int[height, width];
        InitializePieces();
    }

    private void InitializePieces()
    {
        pieces = new Dictionary<string, List<int[,]>>();

        // I piece
        pieces["I"] = new List<int[,]>
        {
            new int[,] { {1,1,1,1} },
            new int[,] { {1}, {1}, {1}, {1} }
        };

        // O piece
        pieces["O"] = new List<int[,]>
        {
            new int[,] { {1,1}, {1,1} }
        };

        // T piece
        pieces["T"] = new List<int[,]>
        {
            new int[,] { {0,1,0}, {1,1,1} },
            new int[,] { {1,0}, {1,1}, {1,0} },
            new int[,] { {1,1,1}, {0,1,0} },
            new int[,] { {0,1}, {1,1}, {0,1} }
        };

        // S piece
        pieces["S"] = new List<int[,]>
        {
            new int[,] { {0,1,1}, {1,1,0} },
            new int[,] { {1,0}, {1,1}, {0,1} }
        };

        // Z piece
        pieces["Z"] = new List<int[,]>
        {
            new int[,] { {1,1,0}, {0,1,1} },
            new int[,] { {0,1}, {1,1}, {1,0} }
        };

        // J piece
        pieces["J"] = new List<int[,]>
        {
            new int[,] { {1,0,0}, {1,1,1} },
            new int[,] { {1,1}, {1,0}, {1,0} },
            new int[,] { {1,1,1}, {0,0,1} },
            new int[,] { {0,1}, {0,1}, {1,1} }
        };

        // L piece
        pieces["L"] = new List<int[,]>
        {
            new int[,] { {0,0,1}, {1,1,1} },
            new int[,] { {1,0}, {1,0}, {1,1} },
            new int[,] { {1,1,1}, {1,0,0} },
            new int[,] { {1,1}, {0,1}, {0,1} }
        };
    }

    public (int rotation, int column) FindBestPlacement(string pieceType)
    {
        int bestRotation = 0;
        int bestColumn = 0;
        double bestScore = double.MinValue;

        var rotations = pieces[pieceType];

        for (int r = 0; r < rotations.Count; r++)
        {
            var piece = rotations[r];
            int pieceWidth = piece.GetLength(1);

            for (int col = 0; col <= width - pieceWidth; col++)
            {
                var testBoard = (int[,])board.Clone();
                if (TryPlacePiece(testBoard, piece, col, out int linesCleared))
                {
                    double score = EvaluateBoard(testBoard, linesCleared);
                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestRotation = r;
                        bestColumn = col;
                    }
                }
            }
        }

        return (bestRotation, bestColumn);
    }

    private bool TryPlacePiece(int[,] testBoard, int[,] piece, int column, out int linesCleared)
    {
        linesCleared = 0;
        int pieceHeight = piece.GetLength(0);
        int pieceWidth = piece.GetLength(1);

        // Find landing row
        int row = 0;
        for (row = 0; row < height; row++)
        {
            if (WouldCollide(testBoard, piece, row, column))
            {
                row--;
                break;
            }
        }
        if (row >= height) row = height - 1;

        // Check if placement is valid
        if (row < 0 || WouldCollide(testBoard, piece, row, column))
            return false;

        // Place piece
        for (int pr = 0; pr < pieceHeight; pr++)
        {
            for (int pc = 0; pc < pieceWidth; pc++)
            {
                if (piece[pr, pc] == 1)
                {
                    int boardRow = row + pr;
                    if (boardRow >= 0 && boardRow < height)
                        testBoard[boardRow, column + pc] = 1;
                }
            }
        }

        // Clear lines
        linesCleared = ClearLines(testBoard);
        return true;
    }

    private bool WouldCollide(int[,] testBoard, int[,] piece, int row, int column)
    {
        int pieceHeight = piece.GetLength(0);
        int pieceWidth = piece.GetLength(1);

        for (int pr = 0; pr < pieceHeight; pr++)
        {
            for (int pc = 0; pc < pieceWidth; pc++)
            {
                if (piece[pr, pc] == 1)
                {
                    int boardRow = row + pr;
                    int boardCol = column + pc;

                    if (boardRow >= height || boardCol >= width)
                        return true;

                    if (boardRow >= 0 && testBoard[boardRow, boardCol] == 1)
                        return true;
                }
            }
        }
        return false;
    }

    private int ClearLines(int[,] testBoard)
    {
        int cleared = 0;
        for (int r = height - 1; r >= 0; r--)
        {
            bool fullLine = true;
            for (int c = 0; c < width; c++)
            {
                if (testBoard[r, c] == 0)
                {
                    fullLine = false;
                    break;
                }
            }

            if (fullLine)
            {
                cleared++;
                // Shift down
                for (int rr = r; rr > 0; rr--)
                {
                    for (int c = 0; c < width; c++)
                    {
                        testBoard[rr, c] = testBoard[rr - 1, c];
                    }
                }
                // Clear top row
                for (int c = 0; c < width; c++)
                {
                    testBoard[0, c] = 0;
                }
                r++; // Check same row again
            }
        }
        return cleared;
    }

    private double EvaluateBoard(int[,] testBoard, int linesCleared)
    {
        double score = 0;

        // Reward line clears heavily
        score += linesCleared * linesCleared * 1000;

        // Calculate column heights
        int[] columnHeights = new int[width];
        int maxHeight = 0;
        for (int c = 0; c < width; c++)
        {
            for (int r = 0; r < height; r++)
            {
                if (testBoard[r, c] == 1)
                {
                    columnHeights[c] = height - r;
                    maxHeight = Math.Max(maxHeight, columnHeights[c]);
                    break;
                }
            }
        }

        // Penalize height
        score -= maxHeight * 10;

        // Penalize total height
        score -= columnHeights.Sum() * 2;

        // Penalize bumpiness (height differences between adjacent columns)
        int bumpiness = 0;
        for (int c = 0; c < width - 1; c++)
        {
            bumpiness += Math.Abs(columnHeights[c] - columnHeights[c + 1]);
        }
        score -= bumpiness * 5;

        // Count and penalize holes
        int holes = 0;
        for (int c = 0; c < width; c++)
        {
            bool blockFound = false;
            for (int r = 0; r < height; r++)
            {
                if (testBoard[r, c] == 1)
                {
                    blockFound = true;
                }
                else if (blockFound && testBoard[r, c] == 0)
                {
                    holes++;
                }
            }
        }
        score -= holes * 50;

        // Penalize wells (deep single-width gaps)
        for (int c = 0; c < width; c++)
        {
            int leftHeight = c > 0 ? columnHeights[c - 1] : 0;
            int rightHeight = c < width - 1 ? columnHeights[c + 1] : 0;
            int depth = Math.Min(leftHeight, rightHeight) - columnHeights[c];
            if (depth > 2)
            {
                score -= depth * 15;
            }
        }

        return score;
    }

    public void ApplyPlacement(string pieceType, int rotation, int column, string result)
    {
        var piece = pieces[pieceType][rotation];

        // Find landing row
        int row = 0;
        for (row = 0; row < height; row++)
        {
            if (WouldCollide(board, piece, row, column))
            {
                row--;
                break;
            }
        }
        if (row >= height) row = height - 1;

        // Place piece on actual board
        int pieceHeight = piece.GetLength(0);
        int pieceWidth = piece.GetLength(1);
        for (int pr = 0; pr < pieceHeight; pr++)
        {
            for (int pc = 0; pc < pieceWidth; pc++)
            {
                if (piece[pr, pc] == 1)
                {
                    int boardRow = row + pr;
                    if (boardRow >= 0 && boardRow < height)
                        board[boardRow, column + pc] = 1;
                }
            }
        }

        // Clear lines based on result
        if (result.StartsWith("cleared"))
        {
            ClearLines(board);
        }
    }
}
