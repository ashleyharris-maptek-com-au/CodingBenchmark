#include <iostream>
#include <string>
#include <vector>
#include <unordered_set>
#include <unordered_map>
#include <algorithm>
#include <climits>

using namespace std;

typedef unsigned long long ull;

// Dual hash to minimize collisions
struct HashPair {
    ull h1, h2;
    bool operator==(const HashPair& other) const {
        return h1 == other.h1 && h2 == other.h2;
    }
};

namespace std {
    template<>
    struct hash<HashPair> {
        size_t operator()(const HashPair& p) const {
            return hash<ull>()(p.h1) ^ (hash<ull>()(p.h2) << 1);
        }
    };
}

const ull BASE1 = 311;
const ull MOD1 = 1000000007;
const ull BASE2 = 337;
const ull MOD2 = 1000000009;

// Check if there exists a common substring of given length
bool hasCommonSubstringOfLength(const vector<string>& strings, int len, string& result) {
    if (len == 0) {
        result = "";
        return true;
    }

    int n = strings.size();
    if (n == 0) return false;

    const string& first = strings[0];
    if ((int)first.length() < len) return false;

    // Precompute powers for rolling hash
    ull pow1 = 1, pow2 = 1;
    for (int i = 0; i < len; i++) {
        pow1 = (pow1 * BASE1) % MOD1;
        pow2 = (pow2 * BASE2) % MOD2;
    }

    // Build hash set from first string with positions
    unordered_map<HashPair, int> hash_to_pos;

    ull h1 = 0, h2 = 0;
    for (int i = 0; i < len; i++) {
        h1 = (h1 * BASE1 + (ull)(unsigned char)first[i]) % MOD1;
        h2 = (h2 * BASE2 + (ull)(unsigned char)first[i]) % MOD2;
    }
    hash_to_pos[{h1, h2}] = 0;

    for (int i = len; i < (int)first.length(); i++) {
        h1 = (h1 + MOD1 - ((ull)(unsigned char)first[i - len] * pow1) % MOD1) % MOD1;
        h1 = (h1 * BASE1 + (ull)(unsigned char)first[i]) % MOD1;
        h2 = (h2 + MOD2 - ((ull)(unsigned char)first[i - len] * pow2) % MOD2) % MOD2;
        h2 = (h2 * BASE2 + (ull)(unsigned char)first[i]) % MOD2;
        if (hash_to_pos.find({h1, h2}) == hash_to_pos.end()) {
            hash_to_pos[{h1, h2}] = i - len + 1;
        }
    }

    // Intersect with remaining strings
    for (int idx = 1; idx < n; idx++) {
        const string& s = strings[idx];
        if ((int)s.length() < len) return false;

        unordered_set<HashPair> current_hashes;

        h1 = 0; h2 = 0;
        for (int i = 0; i < len; i++) {
            h1 = (h1 * BASE1 + (ull)(unsigned char)s[i]) % MOD1;
            h2 = (h2 * BASE2 + (ull)(unsigned char)s[i]) % MOD2;
        }
        if (hash_to_pos.count({h1, h2})) {
            current_hashes.insert({h1, h2});
        }

        for (int i = len; i < (int)s.length(); i++) {
            h1 = (h1 + MOD1 - ((ull)(unsigned char)s[i - len] * pow1) % MOD1) % MOD1;
            h1 = (h1 * BASE1 + (ull)(unsigned char)s[i]) % MOD1;
            h2 = (h2 + MOD2 - ((ull)(unsigned char)s[i - len] * pow2) % MOD2) % MOD2;
            h2 = (h2 * BASE2 + (ull)(unsigned char)s[i]) % MOD2;
            if (hash_to_pos.count({h1, h2})) {
                current_hashes.insert({h1, h2});
            }
        }

        // Keep only common hashes
        unordered_map<HashPair, int> new_map;
        for (const auto& hp : current_hashes) {
            new_map[hp] = hash_to_pos[hp];
        }
        hash_to_pos = new_map;

        if (hash_to_pos.empty()) return false;
    }

    // Extract actual substring from first string
    if (!hash_to_pos.empty()) {
        int pos = hash_to_pos.begin()->second;
        result = first.substr(pos, len);
        return true;
    }

    return false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    cin.ignore();

    vector<string> strings(n);
    for (int i = 0; i < n; i++) {
        getline(cin, strings[i]);
    }

    if (n == 0) {
        cout << "\n";
        return 0;
    }

    // Find minimum string length
    int min_len = INT_MAX;
    for (const auto& s : strings) {
        min_len = min(min_len, (int)s.length());
    }

    if (min_len == 0) {
        cout << "\n";
        return 0;
    }

    // Binary search on the length of longest common substring
    int left = 0, right = min_len;
    string best = "";

    while (left <= right) {
        int mid = left + (right - left) / 2;
        string result;

        if (hasCommonSubstringOfLength(strings, mid, result)) {
            best = result;
            left = mid + 1;  // Try longer
        } else {
            right = mid - 1;  // Try shorter
        }
    }

    cout << best << "\n";

    return 0;
}
