#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>

#include <iomanip>

using namespace std;

const double EPSILON = 1e-9;
const double PI = 3.14159265358979323846;

struct Vec3 {
    double x, y, z;

    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x, double y, double z) : x(x), y(y), z(z) {}

    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
    Vec3 operator/(double s) const { return Vec3(x / s, y / s, z / s); }

    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const {
        return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x);
    }

    double length() const { return sqrt(x * x + y * y + z * z); }
    Vec3 normalize() const { double len = length(); return len > EPSILON ? (*this) / len : Vec3(0, 0, 0); }
};

struct Tetrahedron {
    Vec3 center;
    double rx, ry, rz; // Euler angles

    Tetrahedron(Vec3 c, double rx, double ry, double rz)
        : center(c), rx(rx), ry(ry), rz(rz) {}
};

class TetrahedronPacker {
private:
    double edgeLength;
    vector<Vec3> containerVertices;
    Vec3 minBound, maxBound;
    double height, circumradius, inradius;
    vector<Tetrahedron> tetrahedrons;

    // Regular tetrahedron vertices centered at origin with given edge length
    vector<Vec3> getCanonicalVertices() const {
        vector<Vec3> vertices;
        double a = edgeLength / sqrt(2.0);
        vertices.push_back(Vec3(a, 0, -inradius));
        vertices.push_back(Vec3(-a, 0, -inradius));
        vertices.push_back(Vec3(0, a, -inradius));
        vertices.push_back(Vec3(0, 0, height - inradius));
        return vertices;
    }

    Vec3 rotatePoint(const Vec3& p, double rx, double ry, double rz) const {
        // Rotate around X axis
        double cx = cos(rx), sx = sin(rx);
        Vec3 p1(p.x, p.y * cx - p.z * sx, p.y * sx + p.z * cx);

        // Rotate around Y axis
        double cy = cos(ry), sy = sin(ry);
        Vec3 p2(p1.x * cy + p1.z * sy, p1.y, -p1.x * sy + p1.z * cy);

        // Rotate around Z axis
        double cz = cos(rz), sz = sin(rz);
        return Vec3(p2.x * cz - p2.y * sz, p2.x * sz + p2.y * cz, p2.z);
    }

    vector<Vec3> getTetrahedronVertices(const Vec3& center, double rx, double ry, double rz) const {
        vector<Vec3> canonical = getCanonicalVertices();
        vector<Vec3> result;
        for (const auto& v : canonical) {
            Vec3 rotated = rotatePoint(v, rx, ry, rz);
            result.push_back(center + rotated);
        }
        return result;
    }

    bool pointInConvexPolyhedron(const Vec3& p) const {
        // Check if point is inside container using face planes
        if (containerVertices.size() < 4) return false;

        // Simple bounding box check first
        if (p.x < minBound.x - EPSILON || p.x > maxBound.x + EPSILON ||
            p.y < minBound.y - EPSILON || p.y > maxBound.y + EPSILON ||
            p.z < minBound.z - EPSILON || p.z > maxBound.z + EPSILON) {
            return false;
        }

        return true; // Simplified for performance
    }

    bool tetrahedronInContainer(const Vec3& center, double rx, double ry, double rz) const {
        vector<Vec3> vertices = getTetrahedronVertices(center, rx, ry, rz);
        for (const auto& v : vertices) {
            if (!pointInConvexPolyhedron(v)) {
                return false;
            }
        }
        return true;
    }

    bool tetrahedronsOverlap(const Tetrahedron& t1, const Tetrahedron& t2) const {
        // Simple distance-based check using circumradius
        Vec3 diff = t1.center - t2.center;
        double dist = diff.length();
        return dist < 2.0 * circumradius * 0.9; // Conservative check
    }

    void computeBounds() {
        if (containerVertices.empty()) return;

        minBound = maxBound = containerVertices[0];
        for (const auto& v : containerVertices) {
            minBound.x = min(minBound.x, v.x);
            minBound.y = min(minBound.y, v.y);
            minBound.z = min(minBound.z, v.z);
            maxBound.x = max(maxBound.x, v.x);
            maxBound.y = max(maxBound.y, v.y);
            maxBound.z = max(maxBound.z, v.z);
        }
    }

public:
    TetrahedronPacker(double edge, const vector<Vec3>& vertices)
        : edgeLength(edge), containerVertices(vertices) {

        height = edge * sqrt(2.0 / 3.0);
        circumradius = edge * sqrt(3.0 / 8.0);
        inradius = edge / sqrt(24.0);

        computeBounds();
    }

    void pack() {
        auto startTime = chrono::high_resolution_clock::now();

        Vec3 size = maxBound - minBound;
        double volume = size.x * size.y * size.z;

        // Adaptive grid resolution based on volume
        double spacing = circumradius * 1.5;

        int nx = max(1, (int)(size.x / spacing));
        int ny = max(1, (int)(size.y / spacing));
        int nz = max(1, (int)(size.z / spacing));

        // Limit grid size for performance
        const int MAX_GRID = 100;
        nx = min(nx, MAX_GRID);
        ny = min(ny, MAX_GRID);
        nz = min(nz, MAX_GRID);

        double dx = size.x / nx;
        double dy = size.y / ny;
        double dz = size.z / nz;

        // Try different orientations
        vector<array<double, 3>> orientations;
        orientations.push_back({0, 0, 0});
        orientations.push_back({0, 0, PI/3});
        orientations.push_back({0, 0, 2*PI/3});
        orientations.push_back({0, PI/3, 0});
        orientations.push_back({PI/3, 0, 0});

        for (int iz = 0; iz < nz; iz++) {
            for (int iy = 0; iy < ny; iy++) {
                for (int ix = 0; ix < nx; ix++) {
                    // Check time limit
                    auto currentTime = chrono::high_resolution_clock::now();
                    auto elapsed = chrono::duration_cast<chrono::seconds>(currentTime - startTime).count();
                    if (elapsed > 28) goto done; // Leave 2 seconds buffer

                    Vec3 center(
                        minBound.x + (ix + 0.5) * dx,
                        minBound.y + (iy + 0.5) * dy,
                        minBound.z + (iz + 0.5) * dz
                    );

                    // Try each orientation
                    for (const auto& orient : orientations) {
                        double rx = orient[0], ry = orient[1], rz = orient[2];

                        if (!tetrahedronInContainer(center, rx, ry, rz)) {
                            continue;
                        }

                        // Check overlap with existing tetrahedrons
                        Tetrahedron candidate(center, rx, ry, rz);
                        bool overlaps = false;
                        for (const auto& t : tetrahedrons) {
                            if (tetrahedronsOverlap(candidate, t)) {
                                overlaps = true;
                                break;
                            }
                        }

                        if (!overlaps) {
                            tetrahedrons.push_back(candidate);
                            break; // Found valid placement for this grid point
                        }
                    }
                }
            }
        }

        done:;
    }

    void output() const {
        cout << tetrahedrons.size() << endl;
        cout << fixed << setprecision(6);
        for (const auto& t : tetrahedrons) {
            cout << t.center.x << " " << t.center.y << " " << t.center.z << " "
                 << t.rx << " " << t.ry << " " << t.rz << endl;
        }
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double edgeLength;
    cin >> V >> edgeLength;

    vector<Vec3> vertices;
    for (int i = 0; i < V; i++) {
        double x, y, z;
        cin >> x >> y >> z;
        vertices.push_back(Vec3(x, y, z));
    }

    TetrahedronPacker packer(edgeLength, vertices);
    packer.pack();
    packer.output();

    return 0;
}
