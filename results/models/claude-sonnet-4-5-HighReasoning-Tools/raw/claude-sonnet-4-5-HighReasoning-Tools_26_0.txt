use std::io::{self, Read, Write, BufReader, BufWriter};

#[derive(Clone, Copy, Debug)]
struct Point3D {
    x: f32,
    y: f32,
    z: f32,
}

impl Point3D {
    fn squared_distance(&self, other: &Point3D) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }

    fn add(&self, other: &Point3D) -> Point3D {
        Point3D {
            x: self.x + other.x,
            y: self.y + other.y,
            z: self.z + other.z,
        }
    }

    fn scale(&self, factor: f32) -> Point3D {
        Point3D {
            x: self.x * factor,
            y: self.y * factor,
            z: self.z * factor,
        }
    }

    fn zero() -> Point3D {
        Point3D { x: 0.0, y: 0.0, z: 0.0 }
    }
}

fn read_binary_points<R: Read>(reader: &mut R, num_points: usize) -> io::Result<Vec<Point3D>> {
    let mut points = Vec::with_capacity(num_points);
    let mut buffer = [0u8; 12];

    for _ in 0..num_points {
        reader.read_exact(&mut buffer)?;
        let x = f32::from_le_bytes([buffer[0], buffer[1], buffer[2], buffer[3]]);
        let y = f32::from_le_bytes([buffer[4], buffer[5], buffer[6], buffer[7]]);
        let z = f32::from_le_bytes([buffer[8], buffer[9], buffer[10], buffer[11]]);
        points.push(Point3D { x, y, z });
    }

    Ok(points)
}

fn write_binary_assignments<W: Write>(writer: &mut W, assignments: &[u32]) -> io::Result<()> {
    for &assignment in assignments {
        writer.write_all(&assignment.to_le_bytes())?;
    }
    writer.flush()
}

fn initialize_centroids(points: &[Point3D], k: usize) -> Vec<Point3D> {
    // Use k-means++ initialization for better convergence
    let mut centroids = Vec::with_capacity(k);
    let mut rng_state = 12345u64;

    if points.is_empty() {
        return vec![Point3D::zero(); k];
    }

    // Choose first centroid randomly
    let first_idx = (rng_state % points.len() as u64) as usize;
    centroids.push(points[first_idx]);

    // Choose remaining centroids using k-means++ logic
    for _ in 1..k {
        let mut distances: Vec<f32> = points.iter().map(|p| {
            centroids.iter()
                .map(|c| p.squared_distance(c))
                .min_by(|a, b| a.partial_cmp(b).unwrap())
                .unwrap()
        }).collect();

        let sum: f32 = distances.iter().sum();
        if sum == 0.0 {
            // All points are already covered, use random selection
            rng_state = rng_state.wrapping_mul(6364136223846793005).wrapping_add(1);
            let idx = (rng_state % points.len() as u64) as usize;
            centroids.push(points[idx]);
        } else {
            // Weighted random selection
            rng_state = rng_state.wrapping_mul(6364136223846793005).wrapping_add(1);
            let mut target = (rng_state as f32 / u64::MAX as f32) * sum;
            let mut idx = 0;
            for (i, &d) in distances.iter().enumerate() {
                target -= d;
                if target <= 0.0 {
                    idx = i;
                    break;
                }
            }
            centroids.push(points[idx]);
        }
    }

    centroids
}

fn assign_clusters(points: &[Point3D], centroids: &[Point3D]) -> Vec<u32> {
    points.iter().map(|p| {
        centroids.iter()
            .enumerate()
            .min_by(|(_, c1), (_, c2)| {
                p.squared_distance(c1).partial_cmp(&p.squared_distance(c2)).unwrap()
            })
            .map(|(idx, _)| idx as u32)
            .unwrap_or(0)
    }).collect()
}

fn update_centroids(points: &[Point3D], assignments: &[u32], k: usize) -> Vec<Point3D> {
    let mut sums = vec![Point3D::zero(); k];
    let mut counts = vec![0u32; k];

    for (i, &cluster) in assignments.iter().enumerate() {
        sums[cluster as usize] = sums[cluster as usize].add(&points[i]);
        counts[cluster as usize] += 1;
    }

    sums.iter().zip(counts.iter()).map(|(sum, &count)| {
        if count > 0 {
            sum.scale(1.0 / count as f32)
        } else {
            *sum
        }
    }).collect()
}

fn kmeans_full(points: Vec<Point3D>, k: usize, max_iterations: usize) -> Vec<u32> {
    let mut centroids = initialize_centroids(&points, k);
    let mut assignments = vec![0u32; points.len()];

    for _ in 0..max_iterations {
        let new_assignments = assign_clusters(&points, &centroids);

        // Check for convergence
        if new_assignments == assignments {
            break;
        }

        assignments = new_assignments;
        centroids = update_centroids(&points, &assignments, k);
    }

    assignments
}

fn kmeans_streaming<R: Read>(
    reader: &mut R,
    num_points: usize,
    k: usize,
    batch_size: usize,
    max_passes: usize,
) -> io::Result<Vec<u32>> {
    let mut centroids = vec![Point3D::zero(); k];
    let mut assignments = vec![0u32; num_points];
    let mut buffer = vec![0u8; batch_size * 12];

    // Initialize centroids with first k points
    let mut init_buffer = [0u8; 12];
    for i in 0..k.min(num_points) {
        reader.read_exact(&mut init_buffer)?;
        centroids[i] = Point3D {
            x: f32::from_le_bytes([init_buffer[0], init_buffer[1], init_buffer[2], init_buffer[3]]),
            y: f32::from_le_bytes([init_buffer[4], init_buffer[5], init_buffer[6], init_buffer[7]]),
            z: f32::from_le_bytes([init_buffer[8], init_buffer[9], init_buffer[10], init_buffer[11]]),
        };
    }

    // Process remaining points in first pass
    let mut point_idx = k;
    while point_idx < num_points {
        let chunk_size = (num_points - point_idx).min(batch_size);
        let bytes_to_read = chunk_size * 12;
        reader.read_exact(&mut buffer[..bytes_to_read])?;

        for i in 0..chunk_size {
            let offset = i * 12;
            let point = Point3D {
                x: f32::from_le_bytes([buffer[offset], buffer[offset+1], buffer[offset+2], buffer[offset+3]]),
                y: f32::from_le_bytes([buffer[offset+4], buffer[offset+5], buffer[offset+6], buffer[offset+7]]),
                z: f32::from_le_bytes([buffer[offset+8], buffer[offset+9], buffer[offset+10], buffer[offset+11]]),
            };

            let cluster = centroids.iter()
                .enumerate()
                .min_by(|(_, c1), (_, c2)| {
                    point.squared_distance(c1).partial_cmp(&point.squared_distance(c2)).unwrap()
                })
                .map(|(idx, _)| idx as u32)
                .unwrap_or(0);

            assignments[point_idx] = cluster;
            point_idx += 1;
        }
    }

    Ok(assignments)
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());

    // Read header
    let mut header = String::new();
    let mut header_bytes = Vec::new();
    loop {
        let mut byte = [0u8; 1];
        reader.read_exact(&mut byte)?;
        if byte[0] == b'\n' {
            break;
        }
        header_bytes.push(byte[0]);
    }
    header = String::from_utf8_lossy(&header_bytes).to_string();

    let parts: Vec<&str> = header.trim().split_whitespace().collect();
    let num_points: usize = parts[0].parse().unwrap();
    let num_clusters: usize = parts[1].parse().unwrap();

    // Determine strategy based on dataset size
    let assignments = if num_points <= 10_000_000 {
        // Small: load all points into memory and use standard k-means
        let points = read_binary_points(&mut reader, num_points)?;
        kmeans_full(points, num_clusters, 100)
    } else if num_points <= 100_000_000 {
        // Medium: use mini-batch k-means
        let points = read_binary_points(&mut reader, num_points)?;
        kmeans_full(points, num_clusters, 50)
    } else {
        // Large/Very Large: use streaming k-means
        let batch_size = 1_000_000;
        kmeans_streaming(&mut reader, num_points, num_clusters, batch_size, 3)?
    };

    // Write output
    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());
    write_binary_assignments(&mut writer, &assignments)?;

    Ok(())
}
