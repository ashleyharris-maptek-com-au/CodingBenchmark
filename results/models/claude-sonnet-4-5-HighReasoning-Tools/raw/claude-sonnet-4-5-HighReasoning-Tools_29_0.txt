using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading.Tasks;
using System.Diagnostics;

class MaximumClique
{
    static void Main()
    {
        var input = Console.ReadLine().Split();
        int n = int.Parse(input[0]);
        int m = int.Parse(input[1]);

        var adj = new List<int>[n];
        for (int i = 0; i < n; i++)
            adj[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            var edge = Console.ReadLine().Split();
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            adj[u].Add(v);
            adj[v].Add(u);
        }

        var solver = new CliqueSolver(adj, n);
        var result = solver.Solve();

        Console.WriteLine(result.Count);
        Console.WriteLine(string.Join(" ", result.OrderBy(x => x)));
    }
}

class CliqueSolver
{
    private List<int>[] adj;
    private int n;
    private List<int> bestClique;
    private object lockObj = new object();
    private Stopwatch sw;
    private const int TIME_LIMIT_MS = 280000; // 4:40 to leave buffer
    private const int EXACT_THRESHOLD = 2000;

    public CliqueSolver(List<int>[] adj, int n)
    {
        this.adj = adj;
        this.n = n;
        this.bestClique = new List<int>();
    }

    public List<int> Solve()
    {
        sw = Stopwatch.StartNew();

        // Strategy based on graph size
        if (n <= EXACT_THRESHOLD)
        {
            SolveExact();
        }
        else
        {
            // Large graph: use heuristics with parallel execution
            SolveHeuristic();
        }

        // Fallback if nothing found
        if (bestClique.Count == 0)
        {
            bestClique = new List<int> { 0 };
        }

        return bestClique;
    }

    private void SolveExact()
    {
        // Get degeneracy ordering for better branch-and-bound performance
        var ordering = GetDegeneracyOrdering();

        // Parallel execution across starting vertices
        int threads = Math.Min(12, Math.Max(1, n / 20));
        var tasks = new List<Task>();

        for (int t = 0; t < threads; t++)
        {
            int start = t;
            int step = threads;
            tasks.Add(Task.Run(() =>
            {
                for (int idx = start; idx < ordering.Count && sw.ElapsedMilliseconds < TIME_LIMIT_MS; idx += step)
                {
                    int v = ordering[idx];

                    // Build candidate set: neighbors that come after v in ordering
                    var candidates = new HashSet<int>();
                    foreach (int u in adj[v])
                    {
                        if (ordering.IndexOf(u) > idx)
                            candidates.Add(u);
                    }

                    var clique = new List<int> { v };
                    BronKerbosch(clique, candidates, new HashSet<int>());
                }
            }));
        }

        Task.WaitAll(tasks.ToArray());
    }

    private List<int> GetDegeneracyOrdering()
    {
        var degrees = new int[n];
        var active = new bool[n];

        for (int i = 0; i < n; i++)
        {
            degrees[i] = adj[i].Count;
            active[i] = true;
        }

        var ordering = new List<int>();

        for (int i = 0; i < n; i++)
        {
            int minDeg = int.MaxValue;
            int minV = -1;

            for (int v = 0; v < n; v++)
            {
                if (active[v] && degrees[v] < minDeg)
                {
                    minDeg = degrees[v];
                    minV = v;
                }
            }

            if (minV == -1) break;

            ordering.Add(minV);
            active[minV] = false;

            foreach (int u in adj[minV])
            {
                if (active[u])
                    degrees[u]--;
            }
        }

        ordering.Reverse();
        return ordering;
    }

    private void BronKerbosch(List<int> clique, HashSet<int> candidates, HashSet<int> excluded)
    {
        if (sw.ElapsedMilliseconds > TIME_LIMIT_MS)
            return;

        if (candidates.Count == 0 && excluded.Count == 0)
        {
            lock (lockObj)
            {
                if (clique.Count > bestClique.Count)
                {
                    bestClique = new List<int>(clique);
                }
            }
            return;
        }

        // Pruning: can't improve best solution
        int currentBest;
        lock (lockObj) { currentBest = bestClique.Count; }

        if (clique.Count + candidates.Count <= currentBest)
            return;

        // Pivot selection for efficiency
        int pivot = -1;
        int maxConn = -1;

        foreach (int v in candidates.Union(excluded))
        {
            int conn = 0;
            foreach (int u in adj[v])
            {
                if (candidates.Contains(u))
                    conn++;
            }

            if (conn > maxConn)
            {
                maxConn = conn;
                pivot = v;
            }
        }

        // Process vertices not connected to pivot
        var toProcess = new List<int>();
        foreach (int v in candidates)
        {
            if (pivot == -1 || !adj[pivot].Contains(v))
                toProcess.Add(v);
        }

        foreach (int v in toProcess)
        {
            if (sw.ElapsedMilliseconds > TIME_LIMIT_MS)
                return;

            var newClique = new List<int>(clique) { v };
            var newCandidates = new HashSet<int>();
            var newExcluded = new HashSet<int>();

            foreach (int u in adj[v])
            {
                if (candidates.Contains(u))
                    newCandidates.Add(u);
                if (excluded.Contains(u))
                    newExcluded.Add(u);
            }

            BronKerbosch(newClique, newCandidates, newExcluded);

            candidates.Remove(v);
            excluded.Add(v);
        }
    }

    private void SolveHeuristic()
    {
        // Multi-start greedy approach with parallel execution
        var vertices = Enumerable.Range(0, n)
            .OrderByDescending(v => adj[v].Count)
            .ToList();

        int iterations = Math.Min(200, n);
        var tasks = new List<Task>();

        for (int i = 0; i < iterations; i++)
        {
            int start = vertices[i % vertices.Count];
            tasks.Add(Task.Run(() => GreedyFromVertex(start)));

            // Process in batches
            if (tasks.Count >= 12 || i == iterations - 1)
            {
                Task.WaitAll(tasks.ToArray());
                tasks.Clear();
            }

            if (sw.ElapsedMilliseconds > TIME_LIMIT_MS)
                break;
        }

        // Try some random starts as well
        var rnd = new Random(42);
        for (int i = 0; i < Math.Min(100, n / 10) && sw.ElapsedMilliseconds < TIME_LIMIT_MS; i++)
        {
            int start = rnd.Next(n);
            GreedyFromVertex(start);
        }
    }

    private void GreedyFromVertex(int start)
    {
        var clique = new List<int> { start };
        var candidates = new HashSet<int>(adj[start]);

        while (candidates.Count > 0)
        {
            // Select vertex with maximum connections to current candidates
            int best = -1;
            int maxConn = -1;

            foreach (int v in candidates)
            {
                int conn = 0;
                foreach (int u in adj[v])
                {
                    if (candidates.Contains(u))
                        conn++;
                }

                if (conn > maxConn || (conn == maxConn && best >= 0 && adj[v].Count > adj[best].Count))
                {
                    maxConn = conn;
                    best = v;
                }
            }

            if (best < 0) break;

            clique.Add(best);

            // Update candidates: keep only neighbors of best
            var newCandidates = new HashSet<int>();
            foreach (int u in adj[best])
            {
                if (candidates.Contains(u))
                    newCandidates.Add(u);
            }
            candidates = newCandidates;
        }

        lock (lockObj)
        {
            if (clique.Count > bestClique.Count)
            {
                bestClique = new List<int>(clique);
            }
        }
    }
}
