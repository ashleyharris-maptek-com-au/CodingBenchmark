use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read graph
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let n = parts[0];
    let m = parts[1];

    let mut adj: Vec<HashSet<usize>> = vec![HashSet::new(); n];

    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let edge: Vec<usize> = line.split_whitespace()
                .filter_map(|s| s.parse().ok())
                .collect();
            if edge.len() >= 2 {
                let u = edge[0];
                let v = edge[1];
                if u < n && v < n && u != v {
                    adj[u].insert(v);
                    adj[v].insert(u);
                }
            }
        }
    }

    // Initialize positions
    let mut positions = initialize_positions(n);

    // Choose algorithm based on graph size
    if n <= 500 {
        // Small graphs: full force-directed layout
        fruchterman_reingold(&adj, &mut positions, 200);
    } else if n <= 5000 {
        // Medium graphs: fewer iterations
        fruchterman_reingold(&adj, &mut positions, 50);
    } else {
        // Large graphs: fast spring layout (edge forces only)
        fast_spring_layout(&adj, &mut positions, 30);
    }

    // Output
    println!("{}", n);
    for pos in positions {
        println!("{} {}", pos.0, pos.1);
    }
}

fn initialize_positions(n: usize) -> Vec<(f64, f64)> {
    let mut positions = Vec::with_capacity(n);
    let radius = (n as f64).sqrt() * 5.0;

    // Circular initialization
    for i in 0..n {
        let angle = 2.0 * std::f64::consts::PI * i as f64 / n as f64;
        let x = radius * angle.cos();
        let y = radius * angle.sin();
        positions.push((x, y));
    }

    positions
}

fn fruchterman_reingold(adj: &[HashSet<usize>], positions: &mut [(f64, f64)], iterations: usize) {
    let n = positions.len();
    let area = 10000.0;
    let k = (area / n as f64).sqrt().max(1.0);
    let mut temp = (n as f64).sqrt() * 2.0;
    let cooling = 0.95;

    for _ in 0..iterations {
        let mut disp = vec![(0.0, 0.0); n];

        // Repulsive forces between all pairs
        for i in 0..n {
            for j in (i + 1)..n {
                let dx = positions[i].0 - positions[j].0;
                let dy = positions[i].1 - positions[j].1;
                let dist_sq = dx * dx + dy * dy;
                let dist = dist_sq.sqrt().max(0.01);

                let force = k * k / dist;
                let fx = (dx / dist) * force;
                let fy = (dy / dist) * force;

                disp[i].0 += fx;
                disp[i].1 += fy;
                disp[j].0 -= fx;
                disp[j].1 -= fy;
            }
        }

        // Attractive forces for edges
        for i in 0..n {
            for &j in &adj[i] {
                if i < j {
                    let dx = positions[i].0 - positions[j].0;
                    let dy = positions[i].1 - positions[j].1;
                    let dist_sq = dx * dx + dy * dy;
                    let dist = dist_sq.sqrt().max(0.01);

                    let force = dist_sq / k;
                    let fx = (dx / dist) * force;
                    let fy = (dy / dist) * force;

                    disp[i].0 -= fx;
                    disp[i].1 -= fy;
                    disp[j].0 += fx;
                    disp[j].1 += fy;
                }
            }
        }

        // Apply displacements with temperature limiting
        for i in 0..n {
            let disp_len = (disp[i].0 * disp[i].0 + disp[i].1 * disp[i].1).sqrt();
            if disp_len > 0.001 {
                let limited = disp_len.min(temp);
                positions[i].0 += (disp[i].0 / disp_len) * limited;
                positions[i].1 += (disp[i].1 / disp_len) * limited;
            }
        }

        temp *= cooling;
    }
}

fn fast_spring_layout(adj: &[HashSet<usize>], positions: &mut [(f64, f64)], iterations: usize) {
    let n = positions.len();
    let target_len = 20.0;
    let spring_strength = 0.05;
    let damping = 0.85;

    for iter in 0..iterations {
        let mut forces = vec![(0.0, 0.0); n];

        // Spring forces from edges
        for i in 0..n {
            for &j in &adj[i] {
                let dx = positions[j].0 - positions[i].0;
                let dy = positions[j].1 - positions[i].1;
                let dist = (dx * dx + dy * dy).sqrt().max(0.01);

                // Spring force
                let displacement = dist - target_len;
                let force = displacement * spring_strength;

                forces[i].0 += (dx / dist) * force;
                forces[i].1 += (dy / dist) * force;
            }
        }

        // Light repulsion from connected neighbors
        if iter < iterations / 2 {
            for i in 0..n {
                for &j in &adj[i] {
                    let dx = positions[i].0 - positions[j].0;
                    let dy = positions[i].1 - positions[j].1;
                    let dist_sq = dx * dx + dy * dy;

                    if dist_sq < 100.0 && dist_sq > 0.01 {
                        let dist = dist_sq.sqrt();
                        let repulsion = 10.0 / dist_sq;
                        forces[i].0 += (dx / dist) * repulsion;
                        forces[i].1 += (dy / dist) * repulsion;
                    }
                }
            }
        }

        // Update positions
        for i in 0..n {
            positions[i].0 += forces[i].0 * damping;
            positions[i].1 += forces[i].1 * damping;
        }
    }
}
