#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdint>

struct Rectangle {
    int w, h;
    int original_index;
};

struct Placement {
    int board_index;
    int x, y;
    bool rotated;
};

struct FreeRect {
    int x, y, w, h;
};

class Board {
public:
    int width, height;
    std::vector<FreeRect> free_rects;

    Board(int w, int h) : width(w), height(h) {
        free_rects.push_back({0, 0, w, h});
    }

    bool try_place(int rect_w, int rect_h, int& out_x, int& out_y) {
        int best_idx = -1;
        int best_area_fit = INT32_MAX;
        int best_short_side_fit = INT32_MAX;

        // Find best fitting free rectangle using Best Short Side Fit
        for (size_t i = 0; i < free_rects.size(); i++) {
            const auto& fr = free_rects[i];
            if (fr.w >= rect_w && fr.h >= rect_h) {
                int leftover_w = fr.w - rect_w;
                int leftover_h = fr.h - rect_h;
                int short_side_fit = std::min(leftover_w, leftover_h);
                int area_fit = fr.w * fr.h - rect_w * rect_h;

                if (short_side_fit < best_short_side_fit ||
                    (short_side_fit == best_short_side_fit && area_fit < best_area_fit)) {
                    best_idx = i;
                    best_short_side_fit = short_side_fit;
                    best_area_fit = area_fit;
                    out_x = fr.x;
                    out_y = fr.y;
                }
            }
        }

        if (best_idx == -1) return false;

        // Split the free rectangle using guillotine method
        FreeRect placed = free_rects[best_idx];
        free_rects.erase(free_rects.begin() + best_idx);

        // Create new free rectangles from the split
        if (placed.w > rect_w) {
            free_rects.push_back({placed.x + rect_w, placed.y, placed.w - rect_w, rect_h});
        }
        if (placed.h > rect_h) {
            free_rects.push_back({placed.x, placed.y + rect_h, rect_w, placed.h - rect_h});
        }
        if (placed.w > rect_w && placed.h > rect_h) {
            // Bottom-right remainder
            free_rects.push_back({placed.x + rect_w, placed.y + rect_h,
                                 placed.w - rect_w, placed.h - rect_h});
        }

        // Merge overlapping free rectangles for efficiency (simplified)
        if (free_rects.size() > 100) {
            prune_free_rects();
        }

        return true;
    }

    void prune_free_rects() {
        // Remove free rectangles that are contained within others
        for (size_t i = 0; i < free_rects.size(); i++) {
            for (size_t j = i + 1; j < free_rects.size(); ) {
                if (is_contained(free_rects[j], free_rects[i])) {
                    free_rects.erase(free_rects.begin() + j);
                } else if (is_contained(free_rects[i], free_rects[j])) {
                    free_rects.erase(free_rects.begin() + i);
                    i--;
                    break;
                } else {
                    j++;
                }
            }
        }
    }

    bool is_contained(const FreeRect& a, const FreeRect& b) {
        return a.x >= b.x && a.y >= b.y &&
               a.x + a.w <= b.x + b.w && a.y + a.h <= b.y + b.h;
    }
};

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, board_w, board_h;
    std::cin >> n >> board_w >> board_h;

    std::vector<Rectangle> rects(n);
    for (int i = 0; i < n; i++) {
        std::cin >> rects[i].w >> rects[i].h;
        rects[i].original_index = i;
    }

    // Sort rectangles by area (largest first) for better packing
    std::vector<int> order(n);
    for (int i = 0; i < n; i++) order[i] = i;

    std::sort(order.begin(), order.end(), [&](int a, int b) {
        int area_a = rects[a].w * rects[a].h;
        int area_b = rects[b].w * rects[b].h;
        if (area_a != area_b) return area_a > area_b;
        return std::max(rects[a].w, rects[a].h) > std::max(rects[b].w, rects[b].h);
    });

    std::vector<Placement> placements(n);
    std::vector<Board> boards;

    // Place each rectangle
    for (int idx : order) {
        const auto& rect = rects[idx];
        bool placed = false;

        // Try to place on existing boards
        for (size_t b = 0; b < boards.size() && !placed; b++) {
            int x, y;

            // Try normal orientation
            if (boards[b].try_place(rect.w, rect.h, x, y)) {
                placements[idx] = {(int)b, x, y, false};
                placed = true;
            }
            // Try rotated orientation
            else if (boards[b].try_place(rect.h, rect.w, x, y)) {
                placements[idx] = {(int)b, x, y, true};
                placed = true;
            }
        }

        // Create new board if necessary
        if (!placed) {
            boards.emplace_back(board_w, board_h);
            int x, y;

            // Try normal orientation
            if (boards.back().try_place(rect.w, rect.h, x, y)) {
                placements[idx] = {(int)boards.size() - 1, x, y, false};
                placed = true;
            }
            // Try rotated orientation
            else if (boards.back().try_place(rect.h, rect.w, x, y)) {
                placements[idx] = {(int)boards.size() - 1, x, y, true};
                placed = true;
            }
        }

        if (!placed) {
            std::cerr << "Error: Could not place rectangle " << idx << std::endl;
            return 1;
        }
    }

    // Output results
    std::cout << boards.size() << "\n";
    for (int i = 0; i < n; i++) {
        const auto& p = placements[i];
        std::cout << p.board_index << " " << p.x << " " << p.y << " "
                  << (p.rotated ? 1 : 0) << "\n";
    }

    return 0;
}
