#include <iostream>
#include <vector>
#include <algorithm>
#include <map>
#include <set>
#include <queue>
#include <cstdint>

using namespace std;

const int INF = 1e9;

struct Graph {
    int n;
    vector<pair<int, int>> edges;
    vector<map<int, int>> adj;

    Graph(int nodes) : n(nodes), adj(nodes) {}

    void addEdge(int u, int v) {
        edges.push_back({u, v});
        adj[u][v]++;
        adj[v][u]++;
    }

    // Stoer-Wagner algorithm for minimum cut
    pair<int, set<int>> minCutStoerWagner() {
        // Capacity map between super-nodes
        vector<map<int, int>> capacity(n);
        for (int i = 0; i < n; i++) {
            capacity[i] = adj[i];
        }

        // Track which original nodes are in each super-node
        vector<set<int>> superNodes(n);
        for (int i = 0; i < n; i++) {
            superNodes[i].insert(i);
        }

        // Active super-nodes
        vector<int> active;
        for (int i = 0; i < n; i++) {
            active.push_back(i);
        }

        int minCut = INF;
        set<int> bestCutNodes;

        while (active.size() > 1) {
            int sz = active.size();
            vector<int> weight(sz, 0);
            vector<bool> inA(sz, false);
            int prev = -1, last = -1;

            // Maximum adjacency search: add vertices one by one
            for (int phase = 0; phase < sz; phase++) {
                int maxIdx = -1;
                int maxW = -1;

                // Find vertex with maximum weight to A
                for (int i = 0; i < sz; i++) {
                    if (!inA[i] && weight[i] > maxW) {
                        maxW = weight[i];
                        maxIdx = i;
                    }
                }

                inA[maxIdx] = true;
                prev = last;
                last = maxIdx;

                // Update weights for remaining vertices
                for (int i = 0; i < sz; i++) {
                    if (!inA[i]) {
                        weight[i] += capacity[active[maxIdx]][active[i]];
                    }
                }
            }

            // Cut-of-the-phase: weight[last] is the cut value
            if (weight[last] < minCut) {
                minCut = weight[last];
                bestCutNodes = superNodes[active[last]];
            }

            // Merge last into prev
            int s = active[prev];
            int t = active[last];

            // Merge capacities
            for (auto& [node, w] : capacity[t]) {
                capacity[s][node] += w;
                capacity[node][s] += w;
            }
            capacity[s].erase(s); // Remove self-loop

            // Merge super-nodes
            superNodes[s].insert(superNodes[t].begin(), superNodes[t].end());

            // Remove t from active list
            active.erase(active.begin() + last);
        }

        return {minCut, bestCutNodes};
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    Graph g(n);

    // Read edges
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        g.addEdge(u, v);
    }

    // Find minimum cut
    auto [cutSize, cutNodes] = g.minCutStoerWagner();

    // Identify edges in the cut
    vector<pair<int, int>> cutEdges;
    for (auto [u, v] : g.edges) {
        bool uInCut = cutNodes.count(u) > 0;
        bool vInCut = cutNodes.count(v) > 0;
        if (uInCut != vInCut) {
            cutEdges.push_back({u, v});
        }
    }

    // Output result
    cout << cutEdges.size() << "\n";
    for (auto [u, v] : cutEdges) {
        cout << u << " " << v << "\n";
    }

    return 0;
}
