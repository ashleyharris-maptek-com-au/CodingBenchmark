use std::io::{self, BufRead};
use std::collections::VecDeque;

#[derive(Clone)]
struct Grid {
    width: usize,
    height: usize,
    obstacles: Vec<Vec<bool>>,
    visited: Vec<Vec<bool>>,
    path: Vec<(usize, usize)>,
    total_cells: usize,
    require_cycle: bool,
}

impl Grid {
    fn new(width: usize, height: usize, obstacles: Vec<(usize, usize)>, require_cycle: bool) -> Self {
        let mut obs_grid = vec![vec![false; height]; width];
        for (x, y) in obstacles {
            if x < width && y < height {
                obs_grid[x][y] = true;
            }
        }

        let total_cells = width * height - obs_grid.iter().flatten().filter(|&&b| b).count();

        Grid {
            width,
            height,
            obstacles: obs_grid,
            visited: vec![vec![false; height]; width],
            path: Vec::with_capacity(total_cells),
            total_cells,
            require_cycle,
        }
    }

    fn is_valid(&self, x: usize, y: usize) -> bool {
        x < self.width && y < self.height && !self.obstacles[x][y] && !self.visited[x][y]
    }

    fn neighbors(&self, x: usize, y: usize) -> Vec<(usize, usize)> {
        let mut result = Vec::new();

        if x > 0 { result.push((x - 1, y)); }
        if x + 1 < self.width { result.push((x + 1, y)); }
        if y > 0 { result.push((x, y - 1)); }
        if y + 1 < self.height { result.push((x, y + 1)); }

        result
    }

    fn count_reachable(&self, start_x: usize, start_y: usize) -> usize {
        let mut visited = vec![vec![false; self.height]; self.width];
        let mut queue = VecDeque::new();
        queue.push_back((start_x, start_y));
        visited[start_x][start_y] = true;
        let mut count = 1;

        while let Some((x, y)) = queue.pop_front() {
            for (nx, ny) in self.neighbors(x, y) {
                if !self.obstacles[nx][ny] && !self.visited[nx][ny] && !visited[nx][ny] {
                    visited[nx][ny] = true;
                    queue.push_back((nx, ny));
                    count += 1;
                }
            }
        }

        count
    }

    fn would_disconnect(&self, x: usize, y: usize) -> bool {
        // Temporarily mark as visited to check connectivity
        let old_visited = self.visited[x][y];
        let mut grid_clone = self.clone();
        grid_clone.visited[x][y] = true;

        let unvisited_neighbors: Vec<_> = grid_clone.neighbors(x, y)
            .into_iter()
            .filter(|&(nx, ny)| !grid_clone.obstacles[nx][ny] && !grid_clone.visited[nx][ny])
            .collect();

        if unvisited_neighbors.is_empty() {
            return false;
        }

        let expected_count = self.total_cells - self.path.len() - 1;
        let reachable = grid_clone.count_reachable(unvisited_neighbors[0].0, unvisited_neighbors[0].1);

        reachable != expected_count
    }

    fn degree(&self, x: usize, y: usize) -> usize {
        self.neighbors(x, y)
            .into_iter()
            .filter(|&(nx, ny)| self.is_valid(nx, ny))
            .count()
    }

    fn solve(&mut self) -> bool {
        if self.obstacles[0][0] {
            return false;
        }

        self.visited[0][0] = true;
        self.path.push((0, 0));

        let result = self.backtrack(0, 0);

        if !result {
            self.visited[0][0] = false;
            self.path.clear();
        }

        result
    }

    fn backtrack(&mut self, x: usize, y: usize) -> bool {
        if self.path.len() == self.total_cells {
            if self.require_cycle {
                // Check if current position is adjacent to (0, 0)
                return self.neighbors(x, y).contains(&(0, 0));
            }
            return true;
        }

        // Get neighbors and sort by degree (Warnsdorff's heuristic)
        let mut next_moves: Vec<_> = self.neighbors(x, y)
            .into_iter()
            .filter(|&(nx, ny)| self.is_valid(nx, ny))
            .collect();

        // Sort by degree (fewest options first)
        next_moves.sort_by_key(|&(nx, ny)| self.degree(nx, ny));

        for (nx, ny) in next_moves {
            // Pruning: don't visit if it would disconnect the graph
            if self.path.len() + 1 < self.total_cells && self.would_disconnect(nx, ny) {
                continue;
            }

            self.visited[nx][ny] = true;
            self.path.push((nx, ny));

            if self.backtrack(nx, ny) {
                return true;
            }

            self.path.pop();
            self.visited[nx][ny] = false;
        }

        false
    }
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read first line
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    let width = parts[0];
    let height = parts[1];
    let obstacle_count = parts[2];
    let require_cycle = parts[3] == 1;

    // Read obstacles
    let mut obstacles = Vec::new();
    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<usize> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        obstacles.push((coords[0], coords[1]));
    }

    let mut grid = Grid::new(width, height, obstacles, require_cycle);

    if grid.solve() {
        println!("{}", grid.path.len());
        for (x, y) in &grid.path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}
