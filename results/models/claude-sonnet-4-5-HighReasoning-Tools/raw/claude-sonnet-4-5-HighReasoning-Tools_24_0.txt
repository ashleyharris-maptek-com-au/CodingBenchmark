#include <iostream>
#include <thread>
#include <mutex>
#include <atomic>
#include <cmath>
#include <chrono>
#include <string>
#include <algorithm>

using namespace std;

template<typename T>
T clamp(T v, T lo, T hi) {
    return max(lo, min(v, hi));
}

struct State {
    double x = 0, y = 0, z = 0;
    double vx = 0, vy = 0, vz = 0;
    double pitch = 0, yaw = 0, roll = 0;
    double pitch_rate = 0, yaw_rate = 0, roll_rate = 0;
    double fuel = 0;
    bool valid = false;
};

struct Params {
    int width = 0, depth = 0, height = 0;
    double gravity = 0, max_thrust = 0, fuel = 0;
    double target_x = 0, target_y = 0, target_z = 0;
    bool ready = false;
};

mutex mtx;
State state;
Params params;
atomic<bool> running(true);

void input_thread() {
    double sx, sy, sz;
    cin >> params.width >> params.depth >> params.height
        >> params.gravity >> params.max_thrust >> params.fuel
        >> sx >> sy >> sz
        >> params.target_x >> params.target_y >> params.target_z;
    cin.ignore();

    string terrain_file;
    getline(cin, terrain_file);

    string marker;
    cin >> marker;

    {
        lock_guard<mutex> lock(mtx);
        params.ready = true;
    }

    while (running && cin) {
        State s;
        if (cin >> s.x >> s.y >> s.z >> s.vx >> s.vy >> s.vz
                >> s.pitch >> s.yaw >> s.roll
                >> s.pitch_rate >> s.yaw_rate >> s.roll_rate >> s.fuel) {
            s.valid = true;
            lock_guard<mutex> lock(mtx);
            state = s;
        }
    }
}

void control_thread() {
    while (running) {
        Params p;
        State s;
        {
            lock_guard<mutex> lock(mtx);
            if (!params.ready || !state.valid) {
                this_thread::sleep_for(chrono::milliseconds(10));
                continue;
            }
            p = params;
            s = state;
        }

        double dx = p.target_x - s.x;
        double dy = p.target_y - s.y;
        double dz = p.target_z - s.z;
        double hdist = sqrt(dx*dx + dy*dy);

        double landing_radius = max(5.0, min(p.width, p.depth) * 0.02);

        double dvx = 0, dvy = 0, dvz = 0;

        if (hdist > landing_radius * 3) {
            double vmax = min(10.0, hdist * 0.2);
            dvx = vmax * dx / hdist;
            dvy = vmax * dy / hdist;

            if (dz > 20) dvz = 3.0;
            else if (dz < -20) dvz = -3.0;
            else dvz = 0.15 * dz;
        } else {
            dvx = clamp(0.3 * dx, -2.0, 2.0);
            dvy = clamp(0.3 * dy, -2.0, 2.0);
            dvz = (dz < -1) ? max(-1.2, 0.4 * dz) : -0.5;
        }

        double evx = dvx - s.vx;
        double evy = dvy - s.vy;
        double evz = dvz - s.vz;

        double ax = 1.5 * evx;
        double ay = 1.5 * evy;
        double az = 1.5 * evz + p.gravity;

        double a_mag = sqrt(ax*ax + ay*ay + az*az);
        double thrust = clamp(a_mag / p.max_thrust, 0.0, 1.0);

        double des_pitch = 0, des_yaw = 0;

        if (a_mag > 1.0 && thrust > 0.1) {
            double tilt_x = ax / a_mag;
            double tilt_y = ay / a_mag;
            double tilt_h = sqrt(tilt_x*tilt_x + tilt_y*tilt_y);

            if (tilt_h > 0.01) {
                des_yaw = atan2(-tilt_y, -tilt_x);
                des_pitch = clamp(asin(tilt_h), -0.6, 0.6);
            }
        }

        double yaw_err = des_yaw - s.yaw;
        while (yaw_err > M_PI) yaw_err -= 2 * M_PI;
        while (yaw_err < -M_PI) yaw_err += 2 * M_PI;

        double pitch_cmd = clamp(4.0 * (des_pitch - s.pitch) - 1.5 * s.pitch_rate, -1.0, 1.0);
        double yaw_cmd = clamp(4.0 * yaw_err - 1.5 * s.yaw_rate, -1.0, 1.0);
        double roll_cmd = clamp(-1.5 * s.roll_rate, -1.0, 1.0);

        cout << thrust << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n";
        cout.flush();

        this_thread::sleep_for(chrono::milliseconds(50));
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    thread t1(input_thread);
    thread t2(control_thread);

    t1.join();
    running = false;
    t2.join();

    return 0;
}
