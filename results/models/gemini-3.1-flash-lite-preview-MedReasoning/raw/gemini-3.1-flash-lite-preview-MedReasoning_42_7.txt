; SPIR-V Geometry Shader
; Explode spheres: Triangles input, TriangleStrip output
; Input Primitive: Triangles (3)
; Output Primitive: TriangleStrip, Max Vertices 3

OpCapability Geometry
OpCapability Shader
%ext = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 0
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_3 = OpConstant %float 3.0
%float_0_08 = OpConstant %float 0.08
%uint_3 = OpConstant %uint 3

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%inColor = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_pv_pos = OpTypePointer Output %v4float
%ptr_in_pv_pos = OpTypePointer Input %v4float

%main = OpFunction %void None %void
%label_main = OpLabel

; 1. Load Positions
%p0_ptr = OpAccessChain %ptr_in_pv_pos %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_pv_pos %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_pv_pos %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr

; 2. Face Normal
%v0 = OpVectorShuffle %v3float %p0 %p0 0 1 2
%v1 = OpVectorShuffle %v3float %p1 %p1 0 1 2
%v2 = OpVectorShuffle %v3float %p2 %p2 0 1 2

%e1 = OpFSub %v3float %v1 %v0
%e2 = OpFSub %v3float %v2 %v0
%normal = OpExtInst %v3float %ext Cross %e1 %e2
%n_norm = OpExtInst %v3float %ext Normalize %normal
%offset = OpVectorTimesScalar %v3float %n_norm %float_0_08
%offset_comp0 = OpCompositeExtract %float %offset 0
%offset_comp1 = OpCompositeExtract %float %offset 1
%offset_comp2 = OpCompositeExtract %float %offset 2
%float_0 = OpConstant %float 0.0
%offset_v4_val = OpCompositeConstruct %v4float %offset_comp0 %offset_comp1 %offset_comp2 %float_0
OpBranch %emit_0

%emit_0 = OpLabel
%c0 = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0v = OpLoad %v3float %c0
OpStore %outColor %c0v
%pos0 = OpFAdd %v4float %p0 %offset_v4_val
%out_pos_ptr = OpAccessChain %ptr_pv_pos %gl_out %int_0
OpStore %out_pos_ptr %pos0
OpEmitVertex
OpBranch %emit_1

%emit_1 = OpLabel
%c1 = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1v = OpLoad %v3float %c1
OpStore %outColor %c1v
%pos1 = OpFAdd %v4float %p1 %offset_v4_val
OpStore %out_pos_ptr %pos1
OpEmitVertex
OpBranch %emit_2

%emit_2 = OpLabel
%c2 = OpAccessChain %ptr_in_v3 %inColor %int_2
%c2v = OpLoad %v3float %c2
OpStore %outColor %c2v
%pos2 = OpFAdd %v4float %p2 %offset_v4_val
OpStore %out_pos_ptr %pos2
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
