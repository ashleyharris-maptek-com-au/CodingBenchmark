#version 430 core

// Local workgroup size
layout(local_size_x = 256) in;

// Particle structure
struct Particle {
    vec4 position; // xyz: position, w: mass
    vec4 velocity; // xyz: velocity, w: padding
};

// Shader Storage Buffer Object
layout(std430, binding = 0) buffer ParticleBuffer {
    Particle particles[];
};

// Uniforms
uniform float dt;         // Time step
uniform float G;          // Gravitational constant
uniform float softening;  // Softening factor to avoid singularity

void main() {
    // Get the global particle index
    uint i = gl_GlobalInvocationID.x;
    
    // Bounds check
    if (i >= particles.length()) return;

    // Load current particle data
    vec3 pos = particles[i].position.xyz;
    vec3 vel = particles[i].velocity.xyz;
    float mass = particles[i].position.w;

    vec3 force = vec3(0.0);

    // Calculate gravitational force from all other particles
    for (uint j = 0; j < particles.length(); j++) {
        if (i == j) continue;

        vec3 r = particles[j].position.xyz - pos;
        float distSq = dot(r, r) + softening * softening;
        float dist = sqrt(distSq);
        
        // Force magnitude: F = G * m1 * m2 / r^2
        // Direction vector: r / r
        // Force = (G * m1 * m2 / r^2) * (r / r) = (G * m1 * m2 / r^3) * r
        float f = (G * mass * particles[j].position.w) / (distSq * dist);
        force += f * r;
    }

    // Integrate: F = ma => a = F/m
    vec3 acc = force / mass;
    vel += acc * dt;
    pos += vel * dt;

    // Update particle data
    particles[i].position.xyz = pos;
    particles[i].velocity.xyz = vel;
}
