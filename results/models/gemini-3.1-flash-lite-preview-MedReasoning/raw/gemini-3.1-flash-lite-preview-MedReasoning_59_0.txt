const fs = require('fs');

function solve() {
  const input = fs.readFileSync(0, 'utf8');
  const { probability, drones, maxStepsPerDrone } = JSON.parse(input);

  const rows = probability.length;
  const cols = probability[0].length;

  const paths = drones.map(drone => {
    let currX = drone.x;
    let currY = drone.y;
    const path = [[currX, currY]];
    const visited = new Set([`${currX},${currY}`]);

    for (let step = 0; step < maxStepsPerDrone; step++) {
      let bestMove = { x: currX, y: currY, score: -1 };
      
      // Possible moves: Up, Down, Left, Right, Stay
      const moves = [
        { dx: 0, dy: 0 },
        { dx: 1, dy: 0 },
        { dx: -1, dy: 0 },
        { dx: 0, dy: 1 },
        { dx: 0, dy: -1 }
      ];

      for (const move of moves) {
        const nextX = currX + move.dx;
        const nextY = currY + move.dy;

        if (nextX >= 0 && nextX < rows && nextY >= 0 && nextY < cols && probability[nextX][nextY] !== -1) {
          const score = visited.has(`${nextX},${nextY}`) ? 0 : probability[nextX][nextY];
          if (score > bestMove.score) {
            bestMove = { x: nextX, y: nextY, score };
          }
        }
      }

      currX = bestMove.x;
      currY = bestMove.y;
      path.push([currX, currY]);
      visited.add(`${currX},${currY}`);
    }
    return path;
  });

  process.stdout.write(JSON.stringify({ paths }));
}

solve();
