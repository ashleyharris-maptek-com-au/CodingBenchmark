const fs = require('fs');

function solve() {
  const input = fs.readFileSync(0, 'utf8');
  const { machineCount, tasks } = JSON.parse(input);

  const adj = new Map();
  const inDegree = new Map();
  const taskMap = new Map();

  for (const task of tasks) {
    taskMap.set(task.id, task);
    if (!adj.has(task.id)) adj.set(task.id, []);
    if (!inDegree.has(task.id)) inDegree.set(task.id, 0);
  }

  for (const task of tasks) {
    for (const dep of task.deps) {
      if (!adj.has(dep)) adj.set(dep, []);
      adj.get(dep).push(task.id);
      inDegree.set(task.id, (inDegree.get(task.id) || 0) + 1);
    }
  }

  const queue = [];
  for (const [id, degree] of inDegree.entries()) {
    if (degree === 0) queue.push(id);
  }

  // Simple heuristic: schedule tasks with later deadlines or higher weights first
  queue.sort((a, b) => {
    const taskA = taskMap.get(a);
    const taskB = taskMap.get(b);
    return taskA.deadline - taskB.deadline || taskB.weight - taskA.weight;
  });

  const order = [];
  while (queue.length > 0) {
    const u = queue.shift();
    order.push(u);

    for (const v of adj.get(u)) {
      inDegree.set(v, inDegree.get(v) - 1);
      if (inDegree.get(v) === 0) {
        queue.push(v);
        // Resort to maintain priority
        queue.sort((a, b) => {
          const taskA = taskMap.get(a);
          const taskB = taskMap.get(b);
          return taskA.deadline - taskB.deadline || taskB.weight - taskA.weight;
        });
      }
    }
  }

  process.stdout.write(JSON.stringify({ order }));
}

solve();
