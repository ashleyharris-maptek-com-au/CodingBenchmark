The Minesweeper solver is implemented in C#, utilizing a memory-efficient `byte[]` array to represent the board. The solver iteratively applies two primary deduction rules until no more safe cells can be identified:

1.  **Rule 1 (Safe Cells)**: If the number of flagged mines adjacent to a cell equals its value, all remaining unknown neighboring cells are safe.
2.  **Rule 2 (Mines)**: If the number of unknown neighboring cells plus flagged mines equals the cell's value, all unknown neighboring cells are mines.

This logic-based approach ensures that only provably safe cells are returned, adhering to the requirement to avoid guessing. The code handles varied board sizes and includes a standard .csproj for building with modern .NET SDKs.

```csharp
using System;
using System.Collections.Generic;
using System.IO;

public class MinesweeperSolver
{
    private const byte Unknown = 255;
    private const byte Flagged = 254;
    private const byte Safe = 253;

    public static void Main(string[] args)
    {
        string input = Console.In.ReadToEnd();
        string[] lines = input.Split(new[] { '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        
        if (lines.Length < 2) return;

        string[] dimensions = lines[0].Split(' ');
        int height = int.Parse(dimensions[0]);
        int width = int.Parse(dimensions[1]);

        byte[] board = new byte[height * width];

        for (int y = 0; y < height; y++)
        {
            string line = lines[y + 1];
            for (int x = 0; x < width; x++)
            {
                char c = line[x];
                if (c == ' ') board[y * width + x] = Unknown;
                else if (c == '*') board[y * width + x] = Flagged;
                else board[y * width + x] = (byte)(c - '0');
            }
        }

        List<(int, int)> safeCells = Solve(board, height, width);

        Console.WriteLine(safeCells.Count);
        foreach (var cell in safeCells)
        {
            Console.WriteLine($"{cell.Item1} {cell.Item2}");
        }
    }

    private static List<(int, int)> Solve(byte[] board, int height, int width)
    {
        List<(int, int)> foundSafe = new List<(int, int)>();
        bool changed;

        do
        {
            changed = false;
            for (int y = 0; y < height; y++)
            {
                for (int x = 0; x < width; x++)
                {
                    byte val = board[y * width + x];
                    if (val > 8) continue; 

                    int unknownCount = 0;
                    int flagCount = 0;
                    List<int> unknownIndices = new List<int>();

                    for (int dy = -1; dy <= 1; dy++)
                    {
                        for (int dx = -1; dx <= 1; dx++)
                        {
                            if (dy == 0 && dx == 0) continue;
                            int ny = y + dy;
                            int nx = x + dx;
                            if (ny >= 0 && ny < height && nx >= 0 && nx < width)
                            {
                                byte neighbor = board[ny * width + nx];
                                if (neighbor == Unknown)
                                {
                                    unknownCount++;
                                    unknownIndices.Add(ny * width + nx);
                                }
                                else if (neighbor == Flagged)
                                {
                                    flagCount++;
                                }
                            }
                        }
                    }

                    if (val == flagCount && unknownCount > 0)
                    {
                        foreach (int idx in unknownIndices)
                        {
                            if (board[idx] == Unknown)
                            {
                                board[idx] = Safe;
                                foundSafe.Add((idx % width, idx / width));
                                changed = true;
                            }
                        }
                    }
                    else if (val == flagCount + unknownCount && unknownCount > 0)
                    {
                        foreach (int idx in unknownIndices)
                        {
                            if (board[idx] == Unknown)
                            {
                                board[idx] = Flagged;
                                changed = true;
                            }
                        }
                    }
                }
            }
        } while (changed);

        return foundSafe;
    }
}
```
