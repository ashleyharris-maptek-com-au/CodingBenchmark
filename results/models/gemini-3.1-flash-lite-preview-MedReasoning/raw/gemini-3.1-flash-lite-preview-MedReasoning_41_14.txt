OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 0
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4

%float_0 = OpConstant %float 0.0
%float_0_3 = OpConstant %float 0.3
%float_0_7 = OpConstant %float 0.7
%float_1_0 = OpConstant %float 1.0
%float_1_5 = OpConstant %float 1.5
%float_3_0 = OpConstant %float 3.0
%float_0_8 = OpConstant %float 0.8
%float_0_2 = OpConstant %float 0.2
%vec3_base = OpConstantComposite %vec3 %float_0_8 %float_0_2 %float_0_2
%vec3_fog = OpConstantComposite %vec3 %float_0_7 %float_0_7 %float_0_8

%main_type = OpTypeFunction %void
%ubo = OpVariable %ptr_uniform_UBO Uniform
%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output

%main = OpFunction %void None %main_type
%label = OpLabel

; Calculate Normal (normalized)
%n_raw = OpLoad %vec3 %normal
%n = OpExtInst %vec3 %glsl Normalize %n_raw
; Lighting (Lambert)
%p = OpLoad %vec3 %worldPos
%p_ptr = OpAccessChain %ptr_uniform_UBO %ubo %int_3 ; LightPos
%lp = OpLoad %vec4 %p_ptr
%light_dir = OpVectorShuffle %vec3 %lp %lp 0 1 2
%l = OpExtInst %vec3 %glsl Normalize %light_dir
%dot = OpDot %float %n %l
%diffuse = OpExtInst %float %glsl FClamp %dot %float_0 %float_1_0
%lighting = OpFMul %float %float_0_7 %diffuse
%intensity = OpFAdd %float %float_0_3 %lighting
%color_lit = OpVectorTimesScalar %vec3 %vec3_base %intensity

; Fog calculation
%c_ptr = OpAccessChain %ptr_uniform_UBO %ubo %int_4 ; CameraPos
%cp = OpLoad %vec4 %c_ptr
%camera_pos = OpVectorShuffle %vec3 %cp %cp 0 1 2
%dist_vec = OpFSub %vec3 %p %camera_pos
%dist = OpExtInst %float %glsl Length %dist_vec
%fog_in = OpFSub %float %dist %float_1_5
%fog_factor = OpFDiv %float %fog_in %float_3_0
%fog_factor_clamped = OpExtInst %glsl %float FClamp %fog_factor %float_0 %float_1_0
%color_foggy = OpExtInst %vec3 %glsl FMix %color_lit %vec3_fog %fog_factor_clamped

%final_color = OpCompositeConstruct %vec4 %color_foggy %float_1_0
OpStore %outColor %final_color
OpReturn
OpFunctionEnd
