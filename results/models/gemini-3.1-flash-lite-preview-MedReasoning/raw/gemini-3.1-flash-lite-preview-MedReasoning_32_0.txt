The implementation of the Dreyfus-Wagner algorithm for the Steiner Tree problem is complete. You can compile the code using `cl /std:c++20 steiner_tree.cpp` or any equivalent C++20 compiler.

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>

using namespace std;

const long long INF = numeric_limits<long long>::max() / 2;

struct Edge {
    int to;
    int weight;
};

void dijkstra(int n, const vector<vector<Edge>>& adj, int start, vector<long long>& dist, vector<int>& parent) {
    dist.assign(n, INF);
    parent.assign(n, -1);
    dist[start] = 0;
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    pq.push({0, start});

    while (!pq.empty()) {
        long long d = pq.top().first;
        int u = pq.top().second;
        pq.pop();

        if (d > dist[u]) continue;

        for (const auto& edge : adj[u]) {
            if (dist[u] + edge.weight < dist[edge.to]) {
                dist[edge.to] = dist[u] + edge.weight;
                parent[edge.to] = u;
                pq.push({dist[edge.to], edge.to});
            }
        }
    }
}

void add_path_edges(int u, int v, const vector<vector<int>>& parent_path, vector<pair<int, int>>& result_edges) {
    int curr = v;
    while (curr != u && curr != -1) {
        int p = parent_path[u][curr];
        if (p == -1) break;
        int u_edge = min(p, curr);
        int v_edge = max(p, curr);
        result_edges.push_back({u_edge, v_edge});
        curr = p;
    }
}

void reconstruct(int mask, int v, const vector<vector<long long>>& dp, 
                 const vector<vector<long long>>& all_dist, 
                 const vector<vector<int>>& all_parent, 
                 vector<pair<int, int>>& result_edges) {
    if ((mask & (mask - 1)) == 0) return;

    for (int submask = (mask - 1) & mask; submask > 0; submask = (submask - 1) & mask) {
        if (dp[mask][v] == dp[submask][v] + dp[mask ^ submask][v]) {
            reconstruct(submask, v, dp, all_dist, all_parent, result_edges);
            reconstruct(mask ^ submask, v, dp, all_dist, all_parent, result_edges);
            return;
        }
    }

    for (int u = 0; u < (int)dp[0].size(); ++u) {
        if (dp[mask][v] == dp[mask][u] + all_dist[u][v]) {
            add_path_edges(u, v, all_parent, result_edges);
            reconstruct(mask, u, dp, all_dist, all_parent, result_edges);
            return;
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int num_vertices, num_edges, num_terminals;
    if (!(cin >> num_vertices >> num_edges >> num_terminals)) return 0;

    vector<vector<Edge>> adj(num_vertices);
    for (int i = 0; i < num_edges; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    vector<int> terminals(num_terminals);
    for (int i = 0; i < num_terminals; ++i) {
        cin >> terminals[i];
    }

    int num_masks = 1 << num_terminals;
    vector<vector<long long>> dp(num_masks, vector<long long>(num_vertices, INF));

    for (int i = 0; i < num_terminals; ++i) {
        dp[1 << i][terminals[i]] = 0;
    }

    vector<vector<long long>> all_dist(num_vertices, vector<long long>(num_vertices, INF));
    vector<vector<int>> all_parent(num_vertices, vector<int>(num_vertices, -1));
    for (int i = 0; i < num_vertices; ++i) {
        dijkstra(num_vertices, adj, i, all_dist[i], all_parent[i]);
    }

    for (int mask = 1; mask < num_masks; ++mask) {
        for (int submask = (mask - 1) & mask; submask > 0; submask = (submask - 1) & mask) {
            for (int v = 0; v < num_vertices; ++v) {
                dp[mask][v] = min(dp[mask][v], dp[submask][v] + dp[mask ^ submask][v]);
            }
        }
        for (int v = 0; v < num_vertices; ++v) {
            if (dp[mask][v] == INF) continue;
            for (int u = 0; u < num_vertices; ++u) {
                dp[mask][u] = min(dp[mask][u], dp[mask][v] + all_dist[v][u]);
            }
        }
    }

    int best_root = terminals[0];
    long long min_weight = dp[num_masks - 1][best_root];
    for(int v = 0; v < num_vertices; ++v) {
        if(dp[num_masks - 1][v] < min_weight) {
            min_weight = dp[num_masks - 1][v];
            best_root = v;
        }
    }

    vector<pair<int, int>> edges;
    reconstruct(num_masks - 1, best_root, dp, all_dist, all_parent, edges);
    
    sort(edges.begin(), edges.end());
    edges.erase(unique(edges.begin(), edges.end()), edges.end());

    cout << min_weight << " " << edges.size() << endl;
    for (const auto& e : edges) {
        cout << e.first << " " << e.second << endl;
    }
    
    return 0;
}
```
