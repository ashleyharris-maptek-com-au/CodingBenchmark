#include <iostream>
#include <vector>
#include <algorithm>

struct Rectangle {
    int id;
    int w, h;
    int board_index;
    int x, y;
    bool rotated;
};

struct Board {
    int width, height;
    int current_y = 0;
    int current_x = 0;
    int shelf_height = 0;
};

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int N, board_w, board_h;
    if (!(std::cin >> N >> board_w >> board_h)) return 0;

    std::vector<Rectangle> rects(N);
    for (int i = 0; i < N; ++i) {
        rects[i].id = i;
        std::cin >> rects[i].w >> rects[i].h;
        rects[i].rotated = false;
    }

    // Sort by largest dimension descending
    std::vector<int> indices(N);
    for(int i=0; i<N; ++i) indices[i] = i;
    std::sort(indices.begin(), indices.end(), [&](int a, int b) {
        return std::max(rects[a].w, rects[a].h) > std::max(rects[b].w, rects[b].h);
    });

    std::vector<Board> boards;
    std::vector<Rectangle> result(N);

    for (int i : indices) {
        Rectangle& r = rects[i];
        bool placed = false;
        
        for (size_t b_idx = 0; b_idx < boards.size(); ++b_idx) {
            Board& b = boards[b_idx];
            
            // Try normal
            if (b.current_x + r.w <= b.width && b.current_y + r.h <= b.height) {
                r.board_index = b_idx;
                r.x = b.current_x;
                r.y = b.current_y;
                r.rotated = false;
                b.current_x += r.w;
                b.shelf_height = std::max(b.shelf_height, r.h);
                placed = true;
            } 
            // Try rotated
            else if (b.current_x + r.h <= b.width && b.current_y + r.w <= b.height) {
                r.board_index = b_idx;
                r.x = b.current_x;
                r.y = b.current_y;
                r.rotated = true;
                b.current_x += r.h;
                b.shelf_height = std::max(b.shelf_height, r.w);
                placed = true;
            }
            // Next shelf?
            else if (b.current_x + r.w > b.width && b.current_y + b.shelf_height + r.h <= b.height) {
                b.current_y += b.shelf_height;
                b.current_x = 0;
                b.shelf_height = 0;
                r.board_index = b_idx;
                r.x = b.current_x;
                r.y = b.current_y;
                r.rotated = false;
                b.current_x += r.w;
                b.shelf_height = r.h;
                placed = true;
            }
            
            if (placed) {
                result[i] = r;
                break;
            }
        }
        
        if (!placed) {
            Board new_b;
            new_b.width = board_w;
            new_b.height = board_h;
            r.board_index = (int)boards.size();
            r.x = 0;
            r.y = 0;
            r.rotated = false;
            new_b.current_x = r.w;
            new_b.shelf_height = r.h;
            boards.push_back(new_b);
            result[i] = r;
        }
    }

    std::cout << boards.size() << "\n";
    for (int i = 0; i < N; ++i) {
        std::cout << result[i].board_index + 1 << " " << result[i].x << " " << result[i].y << " " << (result[i].rotated ? 1 : 0) << "\n";
    }

    return 0;
}
