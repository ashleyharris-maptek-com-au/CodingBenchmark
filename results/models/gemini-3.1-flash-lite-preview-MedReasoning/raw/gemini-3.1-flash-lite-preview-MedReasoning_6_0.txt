using System;
using System.Collections.Generic;
using System.Linq;

public struct Vector3 {
    public double X, Y, Z;
    public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
    public double LengthSquared() => X * X + Y * Y + Z * Z;
    public double Length() => Math.Sqrt(LengthSquared());
    public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
}

public class Orbit {
    public int Id;
    public Vector3 Position;
    public Vector3 Velocity;
}

public class Solver {
    public static void Main() {
        string[] line1 = Console.ReadLine().Split(' ');
        int N = int.Parse(line1[0]);
        double mu = double.Parse(line1[1]);

        string[] startLine = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries);
        Orbit startOrbit = new Orbit {
            Id = -1,
            Position = new Vector3(double.Parse(startLine[0]), double.Parse(startLine[1]), double.Parse(startLine[2])),
            Velocity = new Vector3(double.Parse(startLine[3]), double.Parse(startLine[4]), double.Parse(startLine[5]))
        };

        List<Orbit> stations = new List<Orbit>();
        for (int i = 0; i < N; i++) {
            string[] stationLine = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries);
            stations.Add(new Orbit {
                Id = i,
                Position = new Vector3(double.Parse(stationLine[0]), double.Parse(stationLine[1]), double.Parse(stationLine[2])),
                Velocity = new Vector3(double.Parse(stationLine[3]), double.Parse(stationLine[4]), double.Parse(stationLine[5]))
            });
        }

        List<int> order = SolveTSP(startOrbit, stations);
        Console.WriteLine(string.Join(" ", order));
    }

    private static double CalculateDeltaV(Orbit from, Orbit to) {
        // A simple proxy for DeltaV is the magnitude of the difference in velocity vectors.
        return (to.Velocity - from.Velocity).Length();
    }

    private static List<int> SolveTSP(Orbit start, List<Orbit> stations) {
        List<int> order = new List<int>();
        HashSet<int> visited = new HashSet<int>();
        Orbit current = start;

        // Nearest Neighbor
        for (int i = 0; i < stations.Count; i++) {
            int nextStation = -1;
            double minDeltaV = double.MaxValue;

            foreach (var station in stations) {
                if (!visited.Contains(station.Id)) {
                    double dv = CalculateDeltaV(current, station);
                    if (dv < minDeltaV) {
                        minDeltaV = dv;
                        nextStation = station.Id;
                    }
                }
            }
            order.Add(nextStation);
            visited.Add(nextStation);
            current = stations.First(s => s.Id == nextStation);
        }
        return order;
    }
}
