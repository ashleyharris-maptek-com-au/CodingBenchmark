; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 1000

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %out_color
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %out_color Location 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%vec2 = OpTypeVector %float 2
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4
%func_void = OpTypeFunction %void

%ubo_var = OpVariable %ptr_uniform_UBO Uniform
%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%in_uv = OpVariable %ptr_input_vec2 Input
%out_color = OpVariable %ptr_output_vec4 Output

%main = OpFunction %void None %func_void
%label_main = OpLabel

%uv = OpLoad %vec2 %in_uv
%uv_x = OpCompositeExtract %float %uv 0
%uv_y = OpCompositeExtract %float %uv 1

%scale_x = OpConstant %float 8.0
%scale_y = OpConstant %float 16.0
%scaled_x = OpFMul %float %uv_x %scale_x
%scaled_y = OpFMul %float %uv_y %scale_y

; Staggered rows: offset every other row
%row = OpFloor %float %scaled_y
%is_odd = OpFMod %float %row %2.0
%offset = OpSelect %float %is_odd %0.5 %0.0
%final_x = OpFAdd %float %scaled_x %offset

%brick_x = OpFract %float %final_x
%brick_y = OpFract %float %scaled_y

%mortar_x = OpConstant %float 0.05
%mortar_y = OpConstant %float 0.1
%not_mortar_x = OpFOrdGreaterThan %bool %brick_x %mortar_x
%not_mortar_y = OpFOrdGreaterThan %bool %brick_y %mortar_y
%is_brick = OpLogicalAnd %bool %not_mortar_x %not_mortar_y

%brick_col = OpConstantComposite %vec4 0.7 0.2 0.1 1.0
%mortar_col = OpConstantComposite %vec4 0.7 0.7 0.7 1.0
%color = OpSelect %vec4 %is_brick %brick_col %mortar_col

; Lambertian Shading
%normal = OpLoad %vec3 %in_normal
%norm = OpExtInst %vec3 %glsl Normalize %normal
%pos = OpLoad %vec3 %in_worldPos
%ptr_light = OpAccessChain %ptr_input_vec3 %ubo_var %3
%light_pos = OpLoad %vec3 %ptr_light
%light_dir = OpFSub %vec3 %light_pos %pos
%light_norm = OpExtInst %vec3 %glsl Normalize %light_dir
%diffuse = OpExtInst %float %glsl FMax 0.0 %dot %norm %light_norm
%light_factor = OpFAdd %float %0.4 %FMul %0.6 %diffuse
%final_rgba = OpVectorTimesScalar %vec4 %color %light_factor
%final_rgba_alpha = OpCompositeInsert %vec4 %1.0 %final_rgba 3

OpStore %out_color %final_rgba_alpha
OpReturn
OpFunctionEnd
