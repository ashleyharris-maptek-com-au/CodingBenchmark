; SPIR-V Geometry Shader
; Input: InputPoints (1 vertex)
; Output: TriangleStrip (4 vertices, quad)

OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_color Location 0
OpDecorate %out_color Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_h = OpConstant %float 0.04
%neg_float_h = OpConstant %float -0.04

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_v4 = OpTypePointer Output %v4float
%ptr_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float

%f_ptr_in_arr = OpTypePointer Input (OpTypeArray %v3float %uint_1)
%f_ptr_out = OpTypePointer Output %v3float

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_out = OpVariable %ptr_out_pv Output
%in_color = OpVariable %f_ptr_in_arr Input
%out_color = OpVariable %f_ptr_out Output
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out

%fn_type = OpTypeFunction %void

%main = OpFunction %void None %fn_type
%lbl_entry = OpLabel

; Read position and color
%pos_ptr = OpAccessChain %ptr_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1
%z = OpCompositeExtract %float %pos 2
%w = OpCompositeExtract %float %pos 3

%col_ptr = OpAccessChain %f_ptr_out %in_color %int_0
%col = OpLoad %v3float %col_ptr

; Helper to emit vertex
%emit_v = OpFunction %void None %fn_type
%lbl_emit = OpLabel
; This is a conceptual representation of emitting vertices 
; For brevity and clarity in SPIR-V assembly:
OpStore %out_color %col
; Emit 4 vertices manually as requested
; 1: x-h, y-h
%pos1 = OpCompositeConstruct %v4float %neg_float_h %neg_float_h %z %w
%pos_out1 = OpAccessChain %ptr_v4 %gl_out %int_0
OpStore %pos_out1 %pos1
OpEmitVertex
; 2: x+h, y-h
%pos2 = OpCompositeConstruct %v4float %float_h %neg_float_h %z %w
%pos_out2 = OpAccessChain %ptr_v4 %gl_out %int_0
OpStore %pos_out2 %pos2
OpEmitVertex
; 3: x-h, y+h
%pos3 = OpCompositeConstruct %v4float %neg_float_h %float_h %z %w
%pos_out3 = OpAccessChain %ptr_v4 %gl_out %int_0
OpStore %pos_out3 %pos3
OpEmitVertex
; 4: x+h, y+h
%pos4 = OpCompositeConstruct %v4float %float_h %float_h %z %w
%pos_out4 = OpAccessChain %ptr_v4 %gl_out %int_0
OpStore %pos_out4 %pos4
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
