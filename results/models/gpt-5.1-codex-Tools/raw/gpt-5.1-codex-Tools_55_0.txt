void Board::loadFEN(const string& fen) {
    fill(begin(squares), end(squares), 0);
    istringstream iss(fen);
    string boardPart, sideStr, castleStr, epStr;
    int halfmove = 0, fullmove = 1;
    iss >> boardPart >> sideStr >> castleStr >> epStr >> halfmove >> fullmove;

    int idx = 0;
    for (char c : boardPart) {
        if (c == '/') continue;
        if (isdigit(static_cast<unsigned char>(c))) {
            idx += c - '0';
        } else {
            bool isWhite = isupper(static_cast<unsigned char>(c));
            char lc = static_cast<char>(tolower(static_cast<unsigned char>(c)));
            int type = 0;
            switch (lc) {
                case 'p': type = PT_PAWN; break;
                case 'n': type = PT_KNIGHT; break;
                case 'b': type = PT_BISHOP; break;
                case 'r': type = PT_ROOK; break;
                case 'q': type = PT_QUEEN; break;
                case 'k': type = PT_KING; break;
                default: type = 0; break;
            }
            squares[idx++] = isWhite ? type : -type;
        }
    }

    whiteToMove = (sideStr == "w");
    castlingRights = 0;
    if (castleStr.find('K') != string::npos) castlingRights |= WHITE_KING_SIDE;
    if (castleStr.find('Q') != string::npos) castlingRights |= WHITE_QUEEN_SIDE;
    if (castleStr.find('k') != string::npos) castlingRights |= BLACK_KING_SIDE;
    if (castleStr.find('q') != string::npos) castlingRights |= BLACK_QUEEN_SIDE;

    if (epStr == "-") {
        enPassantSquare = -1;
    } else if (epStr.size() == 2) {
        int file = epStr[0] - 'a';
        int rank = epStr[1] - '0';
        int boardRank = 8 - rank;
        enPassantSquare = boardRank * 8 + file;
    } else {
        enPassantSquare = -1;
    }

    halfmoveClock = halfmove;
    fullmoveNumber = fullmove;
}

int Board::kingSquare(bool white) const {
    int target = white ? PT_KING : -PT_KING;
    for (int i = 0; i < 64; ++i) {
        if (squares[i] == target) return i;
    }
    return -1;
}

bool Board::isSquareAttacked(int square, bool byWhite) const {
    int rank = square / 8;
    int file = square % 8;

    // Pawns
    if (byWhite) {
        if (rank + 1 <= 7 && file - 1 >= 0) {
            if (squares[(rank + 1) * 8 + (file - 1)] == PT_PAWN) return true;
        }
        if (rank + 1 <= 7 && file + 1 <= 7) {
            if (squares[(rank + 1) * 8 + (file + 1)] == PT_PAWN) return true;
        }
    } else {
        if (rank - 1 >= 0 && file - 1 >= 0) {
            if (squares[(rank - 1) * 8 + (file - 1)] == -PT_PAWN) return true;
        }
        if (rank - 1 >= 0 && file + 1 <= 7) {
            if (squares[(rank - 1) * 8 + (file + 1)] == -PT_PAWN) return true;
        }
    }

    // Knights
    const int knightFile[8] = {-1, 1, -2, 2, -2, 2, -1, 1};
    const int knightRank[8] = {-2, -2, -1, -1, 1, 1, 2, 2};
    for (int i = 0; i < 8; ++i) {
        int nf = file + knightFile[i];
        int nr = rank + knightRank[i];
        if (nf < 0 || nf > 7 || nr < 0 || nr > 7) continue;
        int piece = squares[nr * 8 + nf];
        if (!piece) continue;
        if (byWhite && piece == PT_KNIGHT) return true;
        if (!byWhite && piece == -PT_KNIGHT) return true;
    }

    // Sliding pieces
    const int bishopDirs[4][2] = {{1,-1},{-1,-1},{1,1},{-1,1}};
    for (auto& dir : bishopDirs) {
        int f = file + dir[0];
        int r = rank + dir[1];
        while (f >= 0 && f < 8 && r >= 0 && r < 8) {
            int piece = squares[r * 8 + f];
            if (piece != 0) {
                if (byWhite) {
                    if (piece == PT_BISHOP || piece == PT_QUEEN) return true;
                } else {
                    if (piece == -PT_BISHOP || piece == -PT_QUEEN) return true;
                }
                break;
            }
            f += dir[0];
            r += dir[1];
        }
    }

    const int rookDirs[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};
    for (auto& dir : rookDirs) {
        int f = file + dir[0];
        int r = rank + dir[1];
        while (f >= 0 && f < 8 && r >= 0 && r < 8) {
            int piece = squares[r * 8 + f];
            if (piece != 0) {
                if (byWhite) {
                    if (piece == PT_ROOK || piece == PT_QUEEN) return true;
                } else {
                    if (piece == -PT_ROOK || piece == -PT_QUEEN) return true;
                }
                break;
            }
            f += dir[0];
            r += dir[1];
        }
    }

    // King
    for (int df = -1; df <= 1; ++df) {
        for (int dr = -1; dr <= 1; ++dr) {
            if (df == 0 && dr == 0) continue;
            int nf = file + df;
            int nr = rank + dr;
            if (nf < 0 || nf > 7 || nr < 0 || nr > 7) continue;
            int piece = squares[nr * 8 + nf];
            if (byWhite && piece == PT_KING) return true;
            if (!byWhite && piece == -PT_KING) return true;
        }
    }
    return false;
}

bool Board::inCheck(bool white) const {
    int ks = kingSquare(white);
    if (ks == -1) return false;
    return isSquareAttacked(ks, !white);
}

void Board::generatePseudoMoves(vector<Move>& moves, bool capturesOnly) const {
    moves.clear();
    bool white = whiteToMove;

    auto pushMove = [&](int from, int to, int promotion, bool isCapture,
                        bool isEnPassant, bool isCastling, int captured) {
        if (capturesOnly && !isCapture && promotion == 0) return;
        Move mv;
        mv.from = from;
        mv.to = to;
        mv.promotion = promotion;
        mv.isCapture = isCapture;
        mv.isEnPassant = isEnPassant;
        mv.isCastling = isCastling;
        mv.capturedPiece = captured;
        moves.push_back(mv);
    };

    for (int sq = 0; sq < 64; ++sq) {
        int piece = squares[sq];
        if (piece == 0) continue;
        if (white && piece < 0) continue;
        if (!white && piece > 0) continue;

        int absPiece = std::abs(piece);
        int rank = sq / 8;
        int file = sq % 8;

        if (absPiece == PT_PAWN) {
            int forward = white ? -1 : 1;
            int startRank = white ? 6 : 1;
            int promotionRank = white ? 0 : 7;
            int nextRank = rank + forward;
            if (nextRank >= 0 && nextRank < 8) {
                int forwardSq = nextRank * 8 + file;
                if (squares[forwardSq] == 0) {
                    if (nextRank == promotionRank) {
                        int promos[4] = {PT_QUEEN, PT_ROOK, PT_BISHOP, PT_KNIGHT};
                        for (int pType : promos) {
                            pushMove(sq, forwardSq, pType, false, false, false, 0);
                        }
                    } else if (!capturesOnly) {
                        pushMove(sq, forwardSq, 0, false, false, false, 0);
                        if (rank == startRank) {
                            int doubleSq = (rank + 2 * forward) * 8 + file;
                            if (doubleSq >= 0 && doubleSq < 64 && squares[doubleSq] == 0) {
                                pushMove(sq, doubleSq, 0, false, false, false, 0);
                            }
                        }
                    }
                }
            }
            int captureRank = rank + forward;
            if (captureRank >= 0 && captureRank < 8) {
                for (int df : {-1, 1}) {
                    int nf = file + df;
                    if (nf < 0 || nf > 7) continue;
                    int target = captureRank * 8 + nf;
                    int targetPiece = squares[target];
                    if (targetPiece != 0 && ((white && targetPiece < 0) || (!white && targetPiece > 0))) {
                        int promo = (captureRank == promotionRank) ? PT_QUEEN : 0;
                        if (captureRank == promotionRank) {
                            int promos[4] = {PT_QUEEN, PT_ROOK, PT_BISHOP, PT_KNIGHT};
                            for (int pType : promos) {
                                pushMove(sq, target, pType, true, false, false, targetPiece);
                            }
                        } else {
                            pushMove(sq, target, 0, true, false, false, targetPiece);
                        }
                    } else if (target == enPassantSquare) {
                        int cap = white ? -PT_PAWN : PT_PAWN;
                        pushMove(sq, target, 0, true, true, false, cap);
                    }
                }
            }
            continue;
        }

        if (absPiece == PT_KNIGHT) {
            const int knightFile[8] = {-1, 1, -2, 2, -2, 2, -1, 1};
            const int knightRank[8] = {-2, -2, -1, -1, 1, 1, 2, 2};
            for (int i = 0; i < 8; ++i) {
                int nf = file + knightFile[i];
                int nr = rank + knightRank[i];
                if (nf < 0 || nf > 7 || nr < 0 || nr > 7) continue;
                int target = nr * 8 + nf;
                int occ = squares[target];
                if (occ == 0) {
                    pushMove(sq, target, 0, false, false, false, 0);
                } else if ((white && occ < 0) || (!white && occ > 0)) {
                    pushMove(sq, target, 0, true, false, false, occ);
                }
            }
            continue;
        }

        if (absPiece == PT_BISHOP || absPiece == PT_ROOK || absPiece == PT_QUEEN) {
            vector<pair<int,int>> dirs;
            if (absPiece == PT_BISHOP || absPiece == PT_QUEEN) {
                dirs.insert(dirs.end(), {{1,-1},{-1,-1},{1,1},{-1,1}});
            }
            if (absPiece == PT_ROOK || absPiece == PT_QUEEN) {
                dirs.insert(dirs.end(), {{1,0},{-1,0},{0,1},{0,-1}});
            }
            for (auto [df, dr] : dirs) {
                int nf = file + df;
                int nr = rank + dr;
                while (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
                    int target = nr * 8 + nf;
                    int occ = squares[target];
                    if (occ == 0) {
                        pushMove(sq, target, 0, false, false, false, 0);
                    } else {
                        if ((white && occ < 0) || (!white && occ > 0)) {
                            pushMove(sq, target, 0, true, false, false, occ);
                        }
                        break;
                    }
                    nf += df;
                    nr += dr;
                }
            }
            continue;
        }

        if (absPiece == PT_KING) {
            for (int df = -1; df <= 1; ++df) {
                for (int dr = -1; dr <= 1; ++dr) {
                    if (df == 0 && dr == 0) continue;
                    int nf = file + df;
                    int nr = rank + dr;
                    if (nf < 0 || nf > 7 || nr < 0 || nr > 7) continue;
                    int target = nr * 8 + nf;
                    int occ = squares[target];
                    if (occ == 0) {
                        pushMove(sq, target, 0, false, false, false, 0);
                    } else if ((white && occ < 0) || (!white && occ > 0)) {
                        pushMove(sq, target, 0, true, false, false, occ);
                    }
                }
            }
            if (!capturesOnly) {
                if (white && sq == 60) {
                    if ((castlingRights & WHITE_KING_SIDE) &&
                        squares[61] == 0 && squares[62] == 0 &&
                        !isSquareAttacked(60, false) &&
                        !isSquareAttacked(61, false) &&
                        !isSquareAttacked(62, false) &&
                        squares[63] == PT_ROOK) {
                        pushMove(60, 62, 0, false, false, true, 0);
                    }
                    if ((castlingRights & WHITE_QUEEN_SIDE) &&
                        squares[59] == 0 && squares[58] == 0 && squares[57] == 0 &&
                        !isSquareAttacked(60, false) &&
                        !isSquareAttacked(59, false) &&
                        !isSquareAttacked(58, false) &&
                        squares[56] == PT_ROOK) {
                        pushMove(60, 58, 0, false, false, true, 0);
                    }
                } else if (!white && sq == 4) {
                    if ((castlingRights & BLACK_KING_SIDE) &&
                        squares[5] == 0 && squares[6] == 0 &&
                        !isSquareAttacked(4, true) &&
                        !isSquareAttacked(5, true) &&
                        !isSquareAttacked(6, true) &&
                        squares[7] == -PT_ROOK) {
                        pushMove(4, 6, 0, false, false, true, 0);
                    }
                    if ((castlingRights & BLACK_QUEEN_SIDE) &&
                        squares[3] == 0 && squares[2] == 0 && squares[1] == 0 &&
                        !isSquareAttacked(4, true) &&
                        !isSquareAttacked(3, true) &&
                        !isSquareAttacked(2, true) &&
                        squares[0] == -PT_ROOK) {
                        pushMove(4, 2, 0, false, false, true, 0);
                    }
                }
            }
        }
    }
}

void Board::generateLegalMoves(vector<Move>& moves, bool capturesOnly) {
    static thread_local vector<Move> pseudo;
    pseudo.clear();
    generatePseudoMoves(pseudo, capturesOnly);
    moves.clear();
    moves.reserve(pseudo.size());
    for (const auto& mv : pseudo) {
        MoveState st;
        if (makeMove(mv, st)) {
            moves.push_back(mv);
            unmakeMove(mv, st);
        }
    }
}

bool Board::makeMove(const Move& move, MoveState& st) {
    st.whiteToMove = whiteToMove;
    st.castlingRights = castlingRights;
    st.enPassantSquare = enPassantSquare;
    st.halfmoveClock = halfmoveClock;
    st.fullmoveNumber = fullmoveNumber;
    st.capturedPiece = 0;

    bool white = whiteToMove;
    int movingPiece = squares[move.from];
    int captured = 0;

    if (move.isEnPassant) {
        int capSq = move.to + (white ? 8 : -8);
        captured = squares[capSq];
        squares[capSq] = 0;
    } else {
        captured = squares[move.to];
    }
    st.capturedPiece = captured;

    squares[move.to] = movingPiece;
    squares[move.from] = 0;

    if (move.promotion) {
        squares[move.to] = white ? move.promotion : -move.promotion;
    }

    if (move.isCastling) {
        if (move.to == move.from + 2) {
            int rookFrom = move.from + 3;
            int rookTo = move.from + 1;
            squares[rookTo] = squares[rookFrom];
            squares[rookFrom] = 0;
        } else if (move.to == move.from - 2) {
            int rookFrom = move.from - 4;
            int rookTo = move.from - 1;
            squares[rookTo] = squares[rookFrom];
            squares[rookFrom] = 0;
        }
    }

    if (abs(movingPiece) == PT_PAWN || captured != 0) {
        halfmoveClock = 0;
    } else {
        ++halfmoveClock;
    }

    if (!white) {
        ++fullmoveNumber;
    }

    enPassantSquare = -1;
    if (abs(movingPiece) == PT_PAWN && abs(move.to - move.from) == 16) {
        enPassantSquare = white ? move.from - 8 : move.from + 8;
    }

    if (abs(movingPiece) == PT_KING) {
        if (white) castlingRights &= ~(WHITE_KING_SIDE | WHITE_QUEEN_SIDE);
        else castlingRights &= ~(BLACK_KING_SIDE | BLACK_QUEEN_SIDE);
    }
    if (abs(movingPiece) == PT_ROOK) {
        if (move.from == 63) castlingRights &= ~WHITE_KING_SIDE;
        if (move.from == 56) castlingRights &= ~WHITE_QUEEN_SIDE;
        if (move.from == 7) castlingRights &= ~BLACK_KING_SIDE;
        if (move.from == 0) castlingRights &= ~BLACK_QUEEN_SIDE;
    }
    if (!move.isEnPassant && captured != 0 && abs(captured) == PT_ROOK) {
        if (move.to == 63) castlingRights &= ~WHITE_KING_SIDE;
        if (move.to == 56) castlingRights &= ~WHITE_QUEEN_SIDE;
        if (move.to == 7) castlingRights &= ~BLACK_KING_SIDE;
        if (move.to == 0) castlingRights &= ~BLACK_QUEEN_SIDE;
    }

    whiteToMove = !white;

    if (inCheck(!whiteToMove)) {
        unmakeMove(move, st);
        return false;
    }
    return true;
}

void Board::unmakeMove(const Move& move, const MoveState& st) {
    whiteToMove = st.whiteToMove;
    castlingRights = st.castlingRights;
    enPassantSquare = st.enPassantSquare;
    halfmoveClock = st.halfmoveClock;
    fullmoveNumber = st.fullmoveNumber;

    bool white = whiteToMove;
    if (move.isCastling) {
        if (move.to == move.from + 2) {
            int rookFrom = move.from + 3;
            int rookTo = move.from + 1;
            squares[rookFrom] = squares[rookTo];
            squares[rookTo] = 0;
        } else if (move.to == move.from - 2) {
            int rookFrom = move.from - 4;
            int rookTo = move.from - 1;
            squares[rookFrom] = squares[rookTo];
            squares[rookTo] = 0;
        }
    }

    int piece = squares[move.to];
    if (move.promotion) {
        piece = white ? PT_PAWN : -PT_PAWN;
    }
    squares[move.from] = piece;

    if (move.isEnPassant) {
        squares[move.to] = 0;
        int capSq = move.to + (white ? 8 : -8);
        squares[capSq] = st.capturedPiece;
    } else {
        squares[move.to] = st.capturedPiece;
    }
}

int Board::evaluate() const {
    int score = 0;
    int whiteMaterial = 0, blackMaterial = 0;
    int whiteBishops = 0, blackBishops = 0;
    int whiteKingSq = -1, blackKingSq = -1;

    for (int sq = 0; sq < 64; ++sq) {
        int piece = squares[sq];
        if (!piece) continue;
        int type = std::abs(piece);
        int val = PIECE_VALUE[type];
        if (piece > 0) {
            score += val;
            if (type != PT_KING) whiteMaterial += val;
            switch (type) {
                case PT_PAWN: score += PAWN_PST[sq]; break;
                case PT_KNIGHT: score += KNIGHT_PST[sq]; break;
                case PT_BISHOP: score += BISHOP_PST[sq]; ++whiteBishops; break;
                case PT_ROOK: score += ROOK_PST[sq]; break;
                case PT_QUEEN: score += QUEEN_PST[sq]; break;
                case PT_KING: whiteKingSq = sq; break;
            }
        } else {
            score -= val;
            if (type != PT_KING) blackMaterial += val;
            int mirrored = mirrorSquare(sq);
            switch (type) {
                case PT_PAWN: score -= PAWN_PST[mirrored]; break;
                case PT_KNIGHT: score -= KNIGHT_PST[mirrored]; break;
                case PT_BISHOP: score -= BISHOP_PST[mirrored]; ++blackBishops; break;
                case PT_ROOK: score -= ROOK_PST[mirrored]; break;
                case PT_QUEEN: score -= QUEEN_PST[mirrored]; break;
                case PT_KING: blackKingSq = sq; break;
            }
        }
    }

    bool endgame = (whiteMaterial <= ENDGAME_THRESHOLD && blackMaterial <= ENDGAME_THRESHOLD);
    if (whiteKingSq != -1) {
        score += endgame ? KING_END_PST[whiteKingSq] : KING_MID_PST[whiteKingSq];
    }
    if (blackKingSq != -1) {
        int mirrored = mirrorSquare(blackKingSq);
        score -= endgame ? KING_END_PST[mirrored] : KING_MID_PST[mirrored];
    }

    if (whiteBishops >= 2) score += 30;
    if (blackBishops >= 2) score -= 30;

    return score;
}

// Search globals
double g_timeLimitSeconds = 1.0;
chrono::steady_clock::time_point g_startTime;
bool g_timeUp = false;
int historyHeuristic[2][64][64];

inline bool timeExceeded() {
    if (g_timeUp) return true;
    double elapsed = chrono::duration<double>(chrono::steady_clock::now() - g_startTime).count();
    if (elapsed >= g_timeLimitSeconds) {
        g_timeUp = true;
        return true;
    }
    return false;
}

inline int evaluateForSide(const Board& board) {
    int eval = board.evaluate();
    return board.whiteToMove ? eval : -eval;
}

int negamax(Board& board, int depth, int alpha, int beta, int ply);
int quiescence(Board& board, int alpha, int beta, int ply);

int moveOrderingScore(const Board& board, const Move& move, int colorIndex) {
    int score = 0;
    if (move.isCapture) {
        int capturedVal = PIECE_VALUE[std::abs(move.capturedPiece)];
        int moverVal = PIECE_VALUE[std::abs(board.squares[move.from])];
        score = 100000 + 10 * capturedVal - moverVal;
    } else if (move.promotion) {
        score = 80000 + PIECE_VALUE[move.promotion];
    } else if (move.isCastling) {
        score = 50;
    } else {
        score = historyHeuristic[colorIndex][move.from][move.to];
    }
    return score;
}

int quiescence(Board& board, int alpha, int beta, int ply) {
    if (timeExceeded()) return evaluateForSide(board);
    bool inChk = board.inCheck(board.whiteToMove);
    if (ply > 128) return evaluateForSide(board);

    if (inChk) {
        vector<Move> moves;
        board.generateLegalMoves(moves);
        if (moves.empty()) {
            return -MATE_SCORE + ply;
        }
        int best = -INF_SCORE;
        int colorIndex = board.whiteToMove ? 0 : 1;
        sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) {
            return moveOrderingScore(board, a, colorIndex) > moveOrderingScore(board, b, colorIndex);
        });
        for (const auto& move : moves) {
            MoveState st;
            if (!board.makeMove(move, st)) continue;
            int score = -quiescence(board, -beta, -alpha, ply + 1);
            board.unmakeMove(move, st);
            if (timeExceeded()) return evaluateForSide(board);
            best = max(best, score);
            alpha = max(alpha, score);
            if (alpha >= beta) break;
        }
        return best;
    }

    int standPat = evaluateForSide(board);
    if (standPat >= beta) return standPat;
    if (standPat > alpha) alpha = standPat;

    vector<Move> moves;
    board.generateLegalMoves(moves, true);
    if (moves.empty()) return standPat;

    int colorIndex = board.whiteToMove ? 0 : 1;
    sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) {
        return moveOrderingScore(board, a, colorIndex) > moveOrderingScore(board, b, colorIndex);
    });

    for (const auto& move : moves) {
        MoveState st;
        if (!board.makeMove(move, st)) continue;
        int score = -quiescence(board, -beta, -alpha, ply + 1);
        board.unmakeMove(move, st);
        if (timeExceeded()) return evaluateForSide(board);
        if (score > alpha) {
            alpha = score;
            if (alpha >= beta) break;
        }
    }
    return alpha;
}

int negamax(Board& board, int depth, int alpha, int beta, int ply) {
    if (timeExceeded()) return evaluateForSide(board);
    bool inChk = board.inCheck(board.whiteToMove);
    if (inChk) depth++;

    if (depth <= 0) {
        return quiescence(board, alpha, beta, ply);
    }

    vector<Move> moves;
    board.generateLegalMoves(moves);
    if (moves.empty()) {
        if (inChk) return -MATE_SCORE + ply;
        return 0;
    }

    int colorIndex = board.whiteToMove ? 0 : 1;
    sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) {
        return moveOrderingScore(board, a, colorIndex) > moveOrderingScore(board, b, colorIndex);
    });

    int bestScore = -INF_SCORE;
    for (const auto& move : moves) {
        MoveState st;
        if (!board.makeMove(move, st)) continue;
        int score = -negamax(board, depth - 1, -beta, -alpha, ply + 1);
        board.unmakeMove(move, st);
        if (timeExceeded()) return evaluateForSide(board);

        if (score > bestScore) {
            bestScore = score;
        }
        if (score > alpha) {
            alpha = score;
            if (!move.isCapture && !move.promotion) {
                historyHeuristic[colorIndex][move.from][move.to] += depth * depth;
            }
        }
        if (alpha >= beta) {
            if (!move.isCapture && !move.promotion) {
                historyHeuristic[colorIndex][move.from][move.to] += depth * depth;
            }
            break;
        }
    }
    return bestScore;
}

string formatMove(const Board& original, const Move& move) {
    string result;
    if (move.isCastling) {
        result = (move.to == move.from + 2) ? "O-O" : "O-O-O";
    } else {
        int piece = original.squares[move.from];
        char pieceChar = PIECE_NOTATION[std::abs(piece)];
        if (std::abs(piece) == PT_PAWN) pieceChar = 'p';
        result += pieceChar;
        result += squareToString(move.from);
        if (move.isCapture) result += 'x';
        result += squareToString(move.to);
        if (move.isEnPassant) result += "ep";
        if (move.promotion) {
            result += '/';
            result += PIECE_NOTATION[move.promotion];
        }
    }

    Board copy = original;
    MoveState st;
    if (copy.makeMove(move, st)) {
        vector<Move> replies;
        copy.generateLegalMoves(replies);
        bool oppInCheck = copy.inCheck(copy.whiteToMove);
        if (oppInCheck) {
            if (replies.empty()) result += '#';
            else result += '+';
        }
        copy.unmakeMove(move, st);
    }
    return result;
}

string trim(const string& s) {
    size_t start = s.find_first_not_of(" \t\r\n");
    if (start == string::npos) return "";
    size_t end = s.find_last_not_of(" \t\r\n");
    return s.substr(start, end - start + 1);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    string timeoutLine;
    if (!getline(cin, timeoutLine)) {
        return 0;
    }
    string fenLine;
    if (!getline(cin, fenLine)) {
        return 0;
    }
    timeoutLine = trim(timeoutLine);
    fenLine = trim(fenLine);

    double timeoutSeconds = 5.0;
    {
        stringstream ss(timeoutLine);
        ss >> timeoutSeconds;
    }

    Board board;
    board.loadFEN(fenLine);

    memset(historyHeuristic, 0, sizeof(historyHeuristic));
    g_timeLimitSeconds = max(0.1, timeoutSeconds * 0.95);
    g_startTime = chrono::steady_clock::now();
    g_timeUp = false;

    vector<Move> legal;
    board.generateLegalMoves(legal);
    if (legal.empty()) {
        cout << "resign\n";
        return 0;
    }

    Move bestMove = legal[0];
    bool haveBest = false;
    int maxDepth = 64;

    for (int depth = 1; depth <= maxDepth; ++depth) {
        vector<Move> moves;
        board.generateLegalMoves(moves);
        if (moves.empty()) break;

        if (haveBest) {
            auto it = find(moves.begin(), moves.end(), bestMove);
            if (it != moves.end()) {
                swap(moves.front(), *it);
            }
        }

        int alpha = -INF_SCORE;
        int beta = INF_SCORE;
        Move currentBest = moves.front();
        bool aborted = false;

        for (const auto& move : moves) {
            MoveState st;
            if (!board.makeMove(move, st)) continue;
            int score = -negamax(board, depth - 1, -beta, -alpha, 1);
            board.unmakeMove(move, st);
            if (timeExceeded()) {
                aborted = true;
                break;
            }
            if (score > alpha) {
                alpha = score;
                currentBest = move;
            }
        }

        if (!aborted) {
            bestMove = currentBest;
            haveBest = true;
        } else {
            break;
        }

        if (timeExceeded()) break;
    }

    string output = formatMove(board, bestMove);
    cout << output << '\n';
    return 0;
}
