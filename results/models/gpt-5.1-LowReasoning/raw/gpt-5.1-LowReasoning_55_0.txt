#include <iostream>

using namespace std;

// Board representation: 0..63, a1 = 0, b1 = 1, ..., h8 = 63
// rank = sq / 8 (0 = rank 1), file = sq % 8 (0 = file a)

enum Color { WHITE = 0, BLACK = 1 };

struct Move {
    int from;
    int to;
    char piece;      // moving piece
    char capture;    // captured piece or 0
    char promo;      // promotion piece or 0
    bool isCastle;
    bool isEnPassant;
};

struct Position {
    char board[64];       // '.' for empty, 'P'..'K', 'p'..'k'
    Color sideToMove;
    bool castleWK, castleWQ, castleBK, castleBQ;
    int epSquare;         // -1 if none
};

double pieceValue(char p) {
    switch (tolower(p)) {
        case 'p': return 100;
        case 'n': return 320;
        case 'b': return 330;
        case 'r': return 500;
        case 'q': return 900;
        case 'k': return 20000;
        default: return 0;
    }
}

bool isWhite(char p) { return p >= 'A' && p <= 'Z'; }
bool isBlack(char p) { return p >= 'a' && p <= 'z'; }
bool isEmpty(char p) { return p == '.'; }

int rank_of(int sq) { return sq / 8; }
int file_of(int sq) { return sq % 8; }

bool inBounds(int r, int f) { return r >= 0 && r < 8 && f >= 0 && f < 8; }

int rf_to_sq(int r, int f) { return r * 8 + f; }

bool isSquareAttacked(const Position &pos, int sq, Color bySide);

void parseFEN(const string &fen, Position &pos) {
    for (int i = 0; i < 64; ++i) pos.board[i] = '.';
    pos.castleWK = pos.castleWQ = pos.castleBK = pos.castleBQ = false;
    pos.epSquare = -1;

    stringstream ss(fen);
    string boardPart, sidePart, castlePart, epPart;
    ss >> boardPart >> sidePart >> castlePart >> epPart;
    // ignore halfmove/fullmove

    int sq = 56; // a8
    for (char c : boardPart) {
        if (c == '/') {
            sq -= 16; // next rank down
        } else if (isdigit((unsigned char)c)) {
            sq += c - '0';
        } else {
            pos.board[sq] = c;
            sq++;
        }
    }

    pos.sideToMove = (sidePart == "w") ? WHITE : BLACK;

    if (castlePart.find('K') != string::npos) pos.castleWK = true;
    if (castlePart.find('Q') != string::npos) pos.castleWQ = true;
    if (castlePart.find('k') != string::npos) pos.castleBK = true;
    if (castlePart.find('q') != string::npos) pos.castleBQ = true;

    if (epPart != "-" && epPart.size() == 2) {
        int file = epPart[0] - 'a';
        int rank = epPart[1] - '1';
        pos.epSquare = rf_to_sq(rank, file);
    } else {
        pos.epSquare = -1;
    }
}

string squareToString(int sq) {
    char file = 'a' + file_of(sq);
    char rank = '1' + rank_of(sq);
    string s;
    s += file;
    s += rank;
    return s;
}

// Make/unmake move (simple, not using move stacks for speed; ok here)

struct Undo {
    char captured;
    bool castleWK, castleWQ, castleBK, castleBQ;
    int epSquare;
};

void makeMove(Position &pos, const Move &m, Undo &u) {
    u.captured = m.capture;
    u.castleWK = pos.castleWK;
    u.castleWQ = pos.castleWQ;
    u.castleBK = pos.castleBK;
    u.castleBQ = pos.castleBQ;
    u.epSquare = pos.epSquare;

    char piece = m.piece;
    pos.board[m.from] = '.';

    // En passant capture
    if (m.isEnPassant) {
        if (pos.sideToMove == WHITE) {
            int capSq = m.to - 8;
            pos.board[capSq] = '.';
        } else {
            int capSq = m.to + 8;
            pos.board[capSq] = '.';
        }
    }

    // Castling: move rook
    if (m.isCastle) {
        if (pos.sideToMove == WHITE) {
            if (m.to == rf_to_sq(0, 6)) { // O-O
                pos.board[rf_to_sq(0, 7)] = '.';
                pos.board[rf_to_sq(0, 5)] = 'R';
            } else if (m.to == rf_to_sq(0, 2)) { // O-O-O
                pos.board[rf_to_sq(0, 0)] = '.';
                pos.board[rf_to_sq(0, 3)] = 'R';
            }
        } else {
            if (m.to == rf_to_sq(7, 6)) { // O-O
                pos.board[rf_to_sq(7, 7)] = '.';
                pos.board[rf_to_sq(7, 5)] = 'r';
            } else if (m.to == rf_to_sq(7, 2)) { // O-O-O
                pos.board[rf_to_sq(7, 0)] = '.';
                pos.board[rf_to_sq(7, 3)] = 'r';
            }
        }
    }

    char placed = piece;
    if (m.promo) placed = m.promo;
    pos.board[m.to] = placed;

    // Update castling rights
    // If king or rooks move/captured
    if (piece == 'K') {
        pos.castleWK = pos.castleWQ = false;
    }
    if (piece == 'k') {
        pos.castleBK = pos.castleBQ = false;
    }
    if (m.from == rf_to_sq(0, 0) || m.to == rf_to_sq(0, 0)) pos.castleWQ = false;
    if (m.from == rf_to_sq(0, 7) || m.to == rf_to_sq(0, 7)) pos.castleWK = false;
    if (m.from == rf_to_sq(7, 0) || m.to == rf_to_sq(7, 0)) pos.castleBQ = false;
    if (m.from == rf_to_sq(7, 7) || m.to == rf_to_sq(7, 7)) pos.castleBK = false;

    // Update ep square
    pos.epSquare = -1;
    if (tolower(piece) == 'p') {
        int fromRank = rank_of(m.from);
        int toRank = rank_of(m.to);
        if (abs(toRank - fromRank) == 2) {
            // double pawn push
            int epRank = (fromRank + toRank) / 2;
            pos.epSquare = rf_to_sq(epRank, file_of(m.from));
        }
    }

    pos.sideToMove = (pos.sideToMove == WHITE ? BLACK : WHITE);
}

void unmakeMove(Position &pos, const Move &m, const Undo &u) {
    pos.sideToMove = (pos.sideToMove == WHITE ? BLACK : WHITE);
    pos.castleWK = u.castleWK;
    pos.castleWQ = u.castleWQ;
    pos.castleBK = u.castleBK;
    pos.castleBQ = u.castleBQ;
    pos.epSquare = u.epSquare;

    // Undo castling rook move
    if (m.isCastle) {
        if (pos.sideToMove == WHITE) {
            if (m.to == rf_to_sq(0, 6)) { // O-O
                pos.board[rf_to_sq(0, 7)] = 'R';
                pos.board[rf_to_sq(0, 5)] = '.';
            } else if (m.to == rf_to_sq(0, 2)) { // O-O-O
                pos.board[rf_to_sq(0, 0)] = 'R';
                pos.board[rf_to_sq(0, 3)] = '.';
            }
        } else {
            if (m.to == rf_to_sq(7, 6)) { // O-O
                pos.board[rf_to_sq(7, 7)] = 'r';
                pos.board[rf_to_sq(7, 5)] = '.';
            } else if (m.to == rf_to_sq(7, 2)) { // O-O-O
                pos.board[rf_to_sq(7, 0)] = 'r';
                pos.board[rf_to_sq(7, 3)] = '.';
            }
        }
    }

    // Undo en passant capture
    if (m.isEnPassant) {
        pos.board[m.to] = '.';
        if (pos.sideToMove == WHITE) {
            int capSq = m.to - 8;
            pos.board[capSq] = 'p';
        } else {
            int capSq = m.to + 8;
            pos.board[capSq] = 'P';
        }
    } else {
        // Normal move
        pos.board[m.to] = m.capture ? m.capture : '.';
    }

    char piece = m.piece;
    pos.board[m.from] = piece;
}

int findKing(const Position &pos, Color c) {
    char k = (c == WHITE) ? 'K' : 'k';
    for (int i = 0; i < 64; ++i) if (pos.board[i] == k) return i;
    return -1;
}

bool isSquareAttacked(const Position &pos, int sq, Color bySide) {
    // Pawns
    int r = rank_of(sq), f = file_of(sq);
    if (bySide == WHITE) {
        int ar = r - 1;
        if (ar >= 0) {
            if (f - 1 >= 0) {
                char p = pos.board[rf_to_sq(ar, f - 1)];
                if (p == 'P') return true;
            }
            if (f + 1 < 8) {
                char p = pos.board[rf_to_sq(ar, f + 1)];
                if (p == 'P') return true;
            }
        }
    } else {
        int ar = r + 1;
        if (ar < 8) {
            if (f - 1 >= 0) {
                char p = pos.board[rf_to_sq(ar, f - 1)];
                if (p == 'p') return true;
            }
            if (f + 1 < 8) {
                char p = pos.board[rf_to_sq(ar, f + 1)];
                if (p == 'p') return true;
            }
        }
    }

    // Knights
    static const int knightD[8][2] = {
        {1,2},{2,1},{-1,2},{-2,1},{1,-2},{2,-1},{-1,-2},{-2,-1}
    };
    for (auto &d : knightD) {
        int nr = r + d[0], nf = f + d[1];
        if (inBounds(nr, nf)) {
            char p = pos.board[rf_to_sq(nr, nf)];
            if (bySide == WHITE && p == 'N') return true;
            if (bySide == BLACK && p == 'n') return true;
        }
    }

    // Sliding: bishops/queens
    static const int bishopD[4][2] = { {1,1},{1,-1},{-1,1},{-1,-1} };
    for (auto &d : bishopD) {
        int nr = r + d[0], nf = f + d[1];
        while (inBounds(nr, nf)) {
            char p = pos.board[rf_to_sq(nr, nf)];
            if (!isEmpty(p)) {
                if (bySide == WHITE && (p == 'B' || p == 'Q')) return true;
                if (bySide == BLACK && (p == 'b' || p == 'q')) return true;
                break;
            }
            nr += d[0]; nf += d[1];
        }
    }

    // Sliding: rooks/queens
    static const int rookD[4][2] = { {1,0},{-1,0},{0,1},{0,-1} };
    for (auto &d : rookD) {
        int nr = r + d[0], nf = f + d[1];
        while (inBounds(nr, nf)) {
            char p = pos.board[rf_to_sq(nr, nf)];
            if (!isEmpty(p)) {
                if (bySide == WHITE && (p == 'R' || p == 'Q')) return true;
                if (bySide == BLACK && (p == 'r' || p == 'q')) return true;
                break;
            }
            nr += d[0]; nf += d[1];
        }
    }

    // Kings
    for (int dr = -1; dr <= 1; ++dr)
        for (int df = -1; df <= 1; ++df)
            if (dr || df) {
                int nr = r + dr, nf = f + df;
                if (inBounds(nr, nf)) {
                    char p = pos.board[rf_to_sq(nr, nf)];
                    if (bySide == WHITE && p == 'K') return true;
                    if (bySide == BLACK && p == 'k') return true;
                }
            }

    return false;
}

void addMove(const Position &pos, vector<Move> &moves, int from, int to, char promo = 0, bool isEp = false, bool isCastle = false) {
    Move m;
    m.from = from;
    m.to = to;
    m.piece = pos.board[from];
    m.capture = pos.board[to];
    m.promo = promo;
    m.isEnPassant = isEp;
    m.isCastle = isCastle;
    if (isEp) {
        m.capture = (pos.sideToMove == WHITE ? 'p' : 'P');
    }
    moves.push_back(m);
}

void generateMoves(const Position &pos, vector<Move> &moves) {
    moves.clear();
    Color stm = pos.sideToMove;

    for (int sq = 0; sq < 64; ++sq) {
        char p = pos.board[sq];
        if (p == '.') continue;
        if ((stm == WHITE && isBlack(p)) || (stm == BLACK && isWhite(p))) continue;

        int r = rank_of(sq), f = file_of(sq);
        if (tolower(p) == 'p') {
            int dir = (stm == WHITE) ? 1 : -1;
            int startRank = (stm == WHITE) ? 1 : 6;
            int promoRank = (stm == WHITE) ? 6 : 1;
            int nextR = r + dir;
            if (inBounds(nextR, f)) {
                int forwardSq = rf_to_sq(nextR, f);
                if (isEmpty(pos.board[forwardSq])) {
                    if (r == promoRank) {
                        // promotions
                        char promoPieces[4] = { (stm == WHITE ? 'Q' : 'q'),
                                                (stm == WHITE ? 'R' : 'r'),
                                                (stm == WHITE ? 'B' : 'b'),
                                                (stm == WHITE ? 'N' : 'n') };
                        for (char pr : promoPieces)
                            addMove(pos, moves, sq, forwardSq, pr);
                    } else {
                        addMove(pos, moves, sq, forwardSq);
                        // double push
                        if (r == startRank) {
                            int next2R = r + 2 * dir;
                            int forwardSq2 = rf_to_sq(next2R, f);
                            if (isEmpty(pos.board[forwardSq2]))
                                addMove(pos, moves, sq, forwardSq2);
                        }
                    }
                }
            }
            // captures
            for (int df = -1; df <= 1; df += 2) {
                int cf = f + df;
                int cr = r + dir;
                if (!inBounds(cr, cf)) continue;
                int tsq = rf_to_sq(cr, cf);
                char target = pos.board[tsq];
                bool enemy = (stm == WHITE) ? isBlack(target) : isWhite(target);
                bool isEp = (tsq == pos.epSquare);
                if (enemy || isEp) {
                    if (r == promoRank) {
                        char promoPieces[4] = { (stm == WHITE ? 'Q' : 'q'),
                                                (stm == WHITE ? 'R' : 'r'),
                                                (stm == WHITE ? 'B' : 'b'),
                                                (stm == WHITE ? 'N' : 'n') };
                        for (char pr : promoPieces)
                            addMove(pos, moves, sq, tsq, pr, isEp);
                    } else {
                        addMove(pos, moves, sq, tsq, 0, isEp);
                    }
                }
            }
        } else if (tolower(p) == 'n') {
            static const int kD[8][2] = {
                {1,2},{2,1},{-1,2},{-2,1},{1,-2},{2,-1},{-1,-2},{-2,-1}
            };
            for (auto &d : kD) {
                int nr = r + d[0], nf = f + d[1];
                if (!inBounds(nr, nf)) continue;
                int tsq = rf_to_sq(nr, nf);
                char t = pos.board[tsq];
                if (isEmpty(t) || (stm == WHITE ? isBlack(t) : isWhite(t)))
                    addMove(pos, moves, sq, tsq);
            }
        } else if (tolower(p) == 'b' || tolower(p) == 'r' || tolower(p) == 'q') {
            static const int bishopD[4][2] = { {1,1},{1,-1},{-1,1},{-1,-1} };
            static const int rookD[4][2]   = { {1,0},{-1,0},{0,1},{0,-1} };
            const int (*dirs)[2] = nullptr;
            int nd = 0;
            if (tolower(p) == 'b') { dirs = bishopD; nd = 4; }
            else if (tolower(p) == 'r') { dirs = rookD; nd = 4; }
            else { // queen
                // handle both bishop and rook directions
                for (auto &d : bishopD) {
                    int nr = r + d[0], nf = f + d[1];
                    while (inBounds(nr, nf)) {
                        int tsq = rf_to_sq(nr, nf);
                        char t = pos.board[tsq];
                        if (isEmpty(t)) {
                            addMove(pos, moves, sq, tsq);
                        } else {
                            if (stm == WHITE ? isBlack(t) : isWhite(t))
                                addMove(pos, moves, sq, tsq);
                            break;
                        }
                        nr += d[0]; nf += d[1];
                    }
                }
                for (auto &d : rookD) {
                    int nr = r + d[0], nf = f + d[1];
                    while (inBounds(nr, nf)) {
                        int tsq = rf_to_sq(nr, nf);
                        char t = pos.board[tsq];
                        if (isEmpty(t)) {
                            addMove(pos, moves, sq, tsq);
                        } else {
                            if (stm == WHITE ? isBlack(t) : isWhite(t))
                                addMove(pos, moves, sq, tsq);
                            break;
                        }
                        nr += d[0]; nf += d[1];
                    }
                }
                continue;
            }
            for (int i = 0; i < nd; ++i) {
                int dr = dirs[i][0], df = dirs[i][1];
                int nr = r + dr, nf = f + df;
                while (inBounds(nr, nf)) {
                    int tsq = rf_to_sq(nr, nf);
                    char t = pos.board[tsq];
                    if (isEmpty(t)) {
                        addMove(pos, moves, sq, tsq);
                    } else {
                        if (stm == WHITE ? isBlack(t) : isWhite(t))
                            addMove(pos, moves, sq, tsq);
                        break;
                    }
                    nr += dr; nf += df;
                }
            }
        } else if (tolower(p) == 'k') {
            for (int dr = -1; dr <= 1; ++dr)
                for (int df = -1; df <= 1; ++df) {
                    if (!dr && !df) continue;
                    int nr = r + dr, nf = f + df;
                    if (!inBounds(nr, nf)) continue;
                    int tsq = rf_to_sq(nr, nf);
                    char t = pos.board[tsq];
                    if (isEmpty(t) || (stm == WHITE ? isBlack(t) : isWhite(t)))
                        addMove(pos, moves, sq, tsq);
                }
            // Castling
            if (stm == WHITE && p == 'K' && r == 0 && f == 4) {
                if (pos.castleWK) {
                    if (isEmpty(pos.board[rf_to_sq(0,5)]) &&
                        isEmpty(pos.board[rf_to_sq(0,6)]) &&
                        !isSquareAttacked(pos, rf_to_sq(0,4), BLACK) &&
                        !isSquareAttacked(pos, rf_to_sq(0,5), BLACK) &&
                        !isSquareAttacked(pos, rf_to_sq(0,6), BLACK)) {
                        addMove(pos, moves, sq, rf_to_sq(0,6), 0, false, true);
                    }
                }
                if (pos.castleWQ) {
                    if (isEmpty(pos.board[rf_to_sq(0,3)]) &&
                        isEmpty(pos.board[rf_to_sq(0,2)]) &&
                        isEmpty(pos.board[rf_to_sq(0,1)]) &&
                        !isSquareAttacked(pos, rf_to_sq(0,4), BLACK) &&
                        !isSquareAttacked(pos, rf_to_sq(0,3), BLACK) &&
                        !isSquareAttacked(pos, rf_to_sq(0,2), BLACK)) {
                        addMove(pos, moves, sq, rf_to_sq(0,2), 0, false, true);
                    }
                }
            } else if (stm == BLACK && p == 'k' && r == 7 && f == 4) {
                if (pos.castleBK) {
                    if (isEmpty(pos.board[rf_to_sq(7,5)]) &&
                        isEmpty(pos.board[rf_to_sq(7,6)]) &&
                        !isSquareAttacked(pos, rf_to_sq(7,4), WHITE) &&
                        !isSquareAttacked(pos, rf_to_sq(7,5), WHITE) &&
                        !isSquareAttacked(pos, rf_to_sq(7,6), WHITE)) {
                        addMove(pos, moves, sq, rf_to_sq(7,6), 0, false, true);
                    }
                }
                if (pos.castleBQ) {
                    if (isEmpty(pos.board[rf_to_sq(7,3)]) &&
                        isEmpty(pos.board[rf_to_sq(7,2)]) &&
                        isEmpty(pos.board[rf_to_sq(7,1)]) &&
                        !isSquareAttacked(pos, rf_to_sq(7,4), WHITE) &&
                        !isSquareAttacked(pos, rf_to_sq(7,3), WHITE) &&
                        !isSquareAttacked(pos, rf_to_sq(7,2), WHITE)) {
                        addMove(pos, moves, sq, rf_to_sq(7,2), 0, false, true);
                    }
                }
            }
        }
    }

    // Filter out illegal moves (leaving own king in check)
    vector<Move> legal;
    legal.reserve(moves.size());
    for (const Move &m : moves) {
        Undo u;
        Position tmp = pos;
        makeMove(tmp, m, u);
        int kingSq = findKing(tmp, (Color)tmp.sideToMove ^ 1); // side that moved
        if (kingSq != -1 && !isSquareAttacked(tmp, kingSq, tmp.sideToMove)) {
            legal.push_back(m);
        }
    }
    moves.swap(legal);
}

double evaluate(const Position &pos) {
    double score = 0;
    for (int i = 0; i < 64; ++i) {
        char p = pos.board[i];
        if (p == '.') continue;
        double v = pieceValue(p);
        if (isWhite(p)) score += v;
        else score -= v;
    }
    return (pos.sideToMove == WHITE ? score : -score);
}

using Clock = chrono::steady_clock;
Clock::time_point timeLimit;
bool timeUp = false;

double negamax(Position &pos, int depth, double alpha, double beta) {
    if (timeUp) return evaluate(pos);
    if (depth == 0) return evaluate(pos);

    vector<Move> moves;
    generateMoves(pos, moves);

    if (moves.empty()) {
        // checkmate or stalemate
        int kingSq = findKing(pos, pos.sideToMove);
        if (kingSq != -1 && isSquareAttacked(pos, kingSq, (Color)(pos.sideToMove ^ 1))) {
            return -1000000.0 + (5 - depth); // mate
        } else {
            return 0.0; // stalemate
        }
    }

    // simple move ordering: captures first
    sort(moves.begin(), moves.end(), [](const Move &a, const Move &b) {
        return (a.capture != 0) > (b.capture != 0);
    });

    double best = -numeric_limits<double>::infinity();
    for (const Move &m : moves) {
        if (Clock::now() >= timeLimit) {
            timeUp = true;
            break;
        }
        Undo u;
        makeMove(pos, m, u);
        double score = -negamax(pos, depth - 1, -beta, -alpha);
        unmakeMove(pos, m, u);
        if (timeUp) break;
        if (score > best) best = score;
        if (score > alpha) {
            alpha = score;
            if (alpha >= beta) break;
        }
    }
    return best;
}

bool hasLegalMoves(const Position &pos) {
    vector<Move> moves;
    generateMoves(pos, moves);
    return !moves.empty();
}

bool isCheck(const Position &pos) {
    int kingSq = findKing(pos, pos.sideToMove);
    if (kingSq == -1) return false;
    return isSquareAttacked(pos, kingSq, (Color)(pos.sideToMove ^ 1));
}

string moveToAlgebraic(const Position &pos, const Move &m) {
    // Castling
    if (m.isCastle) {
        int toFile = file_of(m.to);
        if (toFile == 6) return "O-O";
        else return "O-O-O";
    }

    string s;
    // Piece letter
    char p = m.piece;
    if (tolower(p) == 'p') s += 'p';
    else s += toupper(p);

    // from square
    s += squareToString(m.from);

    // capture or move
    if (m.isEnPassant || m.capture) s += 'x';

    // to square
    s += squareToString(m.to);

    // en passant suffix
    if (m.isEnPassant) s += "ep";

    // promotion
    if (m.promo) {
        s += '/';
        if (tolower(m.promo) == 'p') s += 'p';
        else s += toupper(m.promo);
    }

    // Check/checkmate
    Position tmp = pos;
    Undo u;
    makeMove(tmp, m, u);
    bool oppCheck = isCheck(tmp);
    bool oppHasMoves = hasLegalMoves(tmp);
    if (oppCheck) {
        if (!oppHasMoves) s += '#';
        else s += '+';
    }
    unmakeMove(tmp, m, u);

    return s;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    double timeoutSec;
    string fenLine;
    if (!(cin >> timeoutSec)) return 0;
    getline(cin, fenLine); // consume endline
    getline(cin, fenLine);

    Position pos;
    parseFEN(fenLine, pos);

    timeLimit = Clock::now() + chrono::milliseconds((int)((timeoutSec - 0.1) * 1000));
    timeUp = false;

    vector<Move> rootMoves;
    generateMoves(pos, rootMoves);

    if (rootMoves.empty()) {
        // No legal moves; output something reasonable (checkmate/stalemate)
        cout << "O-O" << '\n';
        return 0;
    }

    Move bestMove = rootMoves[0];
    double bestScore = -numeric_limits<double>::infinity();

    int maxDepth = 64; // practical search depth will be limited by time
    for (int depth = 1; depth <= maxDepth; ++depth) {
        if (Clock::now() >= timeLimit) break;
        double alpha = -numeric_limits<double>::infinity();
        double beta = numeric_limits<double>::infinity();
        Move iterBest = rootMoves[0];
        double iterBestScore = -numeric_limits<double>::infinity();

        // simple reordering: put previous best first
        auto it = find_if(rootMoves.begin(), rootMoves.end(), [&](const Move &m) {
            return m.from == bestMove.from && m.to == bestMove.to &&
                   m.promo == bestMove.promo && m.isCastle == bestMove.isCastle &&
                   m.isEnPassant == bestMove.isEnPassant;
        });
        if (it != rootMoves.end()) {
            rotate(rootMoves.begin(), it, it + 1);
        }

        for (const Move &m : rootMoves) {
            if (Clock::now() >= timeLimit) { timeUp = true; break; }
            Undo u;
            makeMove(pos, m, u);
            double score = -negamax(pos, depth - 1, -beta, -alpha);
            unmakeMove(pos, m, u);
            if (timeUp) break;
            if (score > iterBestScore) {
                iterBestScore = score;
                iterBest = m;
            }
            if (score > alpha) alpha = score;
        }

        if (!timeUp) {
            bestMove = iterBest;
            bestScore = iterBestScore;
        } else {
            break;
        }
    }

    string out = moveToAlgebraic(pos, bestMove);
    cout << out << '\n';
    return 0;
}
