#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

int timer;
vector<int> disc, low;
vector<vector<int>> adj;
vector<pair<int, int>> bridges;

void dfs(int u, int par) {
    disc[u] = low[u] = timer++;
    for (int v : adj[u]) {
        if (v == par) continue;
        if (disc[v] == -1) {
            dfs(v, u);
            low[u] = min(low[u], low[v]);
            if (low[v] > disc[u]) {
                int a = min(u, v);
                int b = max(u, v);
                bridges.emplace_back(a, b);
            }
        } else {
            low[u] = min(low[u], disc[v]);
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    
    int N, M;
    cin >> N >> M;
    
    if (N <= 1) {
        cout << 0 << endl;
        return 0;
    }
    
    adj.assign(N, vector<int>());
    for (int i = 0; i < M; i++) {
        int u, v;
        cin >> u >> v;
        if (u != v) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }
    
    disc.assign(N, -1);
    low.assign(N, -1);
    timer = 0;
    bridges.clear();
    
    dfs(0, -1);
    
    if (!bridges.empty()) {
        cout << 1 << endl;
        auto [u, v] = bridges[0];
        cout << u << " " << v << endl;
        return 0;
    }
    
    int min_deg = INT_MAX;
    int best_v = -1;
    for (int i = 0; i < N; i++) {
        if ((int)adj[i].size() < min_deg) {
            min_deg = adj[i].size();
            best_v = i;
        }
    }
    
    if (best_v == -1 || min_deg == 0) {
        cout << 0 << endl;
        return 0;
    }
    
    cout << min_deg << endl;
    for (int nei : adj[best_v]) {
        cout << best_v << " " << nei << endl;
    }
    
    return 0;
}
