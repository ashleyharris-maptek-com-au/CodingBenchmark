using System;
using System.Collections.Generic;

public class QAPSolver
{
    public static void Main()
    {
        // Read N
        string line = Console.ReadLine();
        if (line == null) return;
        int N = int.Parse(line.Trim());

        // Read flow matrix
        long[][] flow = new long[N][];
        for (int i = 0; i < N; i++)
        {
            line = Console.ReadLine();
            if (line == null) return;
            string[] parts = line.Trim().Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            flow[i] = new long[N];
            for (int j = 0; j < N; j++)
            {
                flow[i][j] = long.Parse(parts[j]);
            }
        }

        // Read dist matrix
        long[][] dist = new long[N][];
        for (int i = 0; i < N; i++)
        {
            line = Console.ReadLine();
            if (line == null) return;
            string[] parts = line.Trim().Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            dist[i] = new long[N];
            for (int j = 0; j < N; j++)
            {
                dist[i][j] = long.Parse(parts[j]);
            }
        }

        // Initialize random permutation
        int[] perm = new int[N];
        for (int i = 0; i < N; i++) perm[i] = i;

        Random rnd = new Random(42); // Fixed seed for reproducibility, can remove for true random
        for (int i = N - 1; i > 0; i--)
        {
            int j = rnd.Next(i + 1);
            int tmp = perm[i];
            perm[i] = perm[j];
            perm[j] = tmp;
        }

        // Compute initial cost
        long currentCost = 0;
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                currentCost += flow[i][j] * dist[perm[i]][perm[j]];
            }
        }

        // Local search with random improving swaps
        int maxNoImprove = Math.Max(50000, N * 25);
        int noImproveCount = 0;
        int maxIterations = 2000000; // Safety cap
        int iter = 0;

        while (noImproveCount < maxNoImprove && iter < maxIterations)
        {
            iter++;

            int x = rnd.Next(N);
            int y = rnd.Next(N);
            if (x == y) continue;

            long delta = ComputeDelta(flow, dist, perm, x, y);

            if (delta < 0)
            {
                // Accept swap
                int tmp = perm[x];
                perm[x] = perm[y];
                perm[y] = tmp;

                currentCost += delta;
                noImproveCount = 0;
            }
            else
            {
                noImproveCount++;
            }
        }

        // Output result
        Console.WriteLine(currentCost);
        for (int i = 0; i < N; i++)
        {
            if (i > 0) Console.Write(" ");
            Console.Write(perm[i]);
        }
        Console.WriteLine();
    }

    private static long ComputeDelta(long[][] flow, long[][] dist, int[] perm, int x, int y)
    {
        long delta = 0;
        int px = perm[x];
        int py = perm[y];

        for (int k = 0; k < flow.Length; k++)
        {
            if (k == x || k == y) continue;

            // Contributions for pairs (x,k), (k,x), (y,k), (k,y)
            delta += flow[x][k] * (dist[py][perm[k]] - dist[px][perm[k]]);
            delta += flow[k][x] * (dist[perm[k]][py] - dist[perm[k]][px]);

            delta += flow[y][k] * (dist[px][perm[k]] - dist[py][perm[k]]);
            delta += flow[k][y] * (dist[perm[k]][px] - dist[perm[k]][py]);
        }

        // Cross terms (x,y) and (y,x)
        delta += flow[x][y] * (dist[py][px] - dist[px][py]);
        delta += flow[y][x] * (dist[px][py] - dist[py][px]);

        // Diagonal/self terms (x,x) and (y,y)
        delta += flow[x][x] * (dist[py][py] - dist[px][px]);
        delta += flow[y][y] * (dist[px][px] - dist[py][py]);

        return delta;
    }
}
