```javascript
const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

const machineCount = data.machineCount || 1;
const tasks = data.tasks || [];

if (tasks.length === 0) {
  console.log(JSON.stringify({ schedule: [] }));
  process.exit(0);
}

const taskMap = new Map();
const successors = new Map();
const indegree = new Map();
const releaseTime = new Map();

for (const task of tasks) {
  const id = task.id;
  taskMap.set(id, task);
  successors.set(id, []);
  indegree.set(id, 0);
  releaseTime.set(id, 0);
}

for (const task of tasks) {
  const id = task.id;
  const deps = task.deps || [];
  indegree.set(id, deps.length);
  for (const depId of deps) {
    if (successors.has(depId)) {
      successors.get(depId).push(id);
    }
  }
}

let ready = [];
for (const id of taskMap.keys()) {
  if (indegree.get(id) === 0) {
    ready.push(id);
  }
}

const machineFree = new Array(machineCount).fill(0);
const schedule = [];

function getPriority(id) {
  const t = taskMap.get(id);
  return {
    deadline: (t.deadline !== undefined ? t.deadline : Infinity),
    weight: (t.weight !== undefined ? t.weight : 0)
  };
}

while (ready.length > 0) {
  let bestIdx = 0;
  let bestP = getPriority(ready[0]);

  for (let i = 1; i < ready.length; i++) {
    const p = getPriority(ready[i]);
    if (p.deadline < bestP.deadline ||
        (p.deadline === bestP.deadline && p.weight > bestP.weight)) {
      bestIdx = i;
      bestP = p;
    }
  }

  const taskId = ready.splice(bestIdx, 1)[0];
  const task = taskMap.get(taskId);
  const dur = task.duration || 0;

  let minFree = machineFree[0];
  let minMach = 0;
  for (let m = 1; m < machineCount; m++) {
    if (machineFree[m] < minFree) {
      minFree = machineFree[m];
      minMach = m;
    }
  }

  const start = Math.max(minFree, releaseTime.get(taskId) || 0);
  const finish = start + dur;

  schedule.push({
    id: taskId,
    start: start,
    machine: minMach
  });

  machineFree[minMach] = finish;

  const succs = successors.get(taskId) || [];
  for (const succId of succs) {
    const currRel = releaseTime.get(succId) || 0;
    releaseTime.set(succId, Math.max(currRel, finish));
    const newInd = indegree.get(succId) - 1;
    indegree.set(succId, newInd);
    if (newInd === 0) {
      ready.push(succId);
    }
  }
}

const output = { schedule };
console.log(JSON.stringify(output));
```
