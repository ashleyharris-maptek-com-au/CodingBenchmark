```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    
    vector<long long> c(n);
    for (int i = 0; i < n; i++) {
        cin >> c[i];
    }
    
    vector<long long> adj(n, 0);
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue; // ignore self-loops if any
        adj[u] |= (1LL << v);
        adj[v] |= (1LL << u);
    }
    
    if (n > 20) {
        // Fallback: for larger n we still try but will be slow/invalid in practice
        // In real usage n <= 20 is expected for this exact solver
        cerr << "Warning: n > 20, exact solution may not finish in time\n";
    }
    
    long long best_val = LLONG_MIN / 2;
    vector<int> best_x(n, 0);
    
    int max_mask = (n <= 20 ? (1 << n) : 0);
    
    for (int s = 0; s < max_mask; s++) {
        // Quick validity check using bitmasks
        bool valid = true;
        long long val = 0;
        
        for (int i = 0; i < n && valid; i++) {
            if (s & (1 << i)) {
                val += c[i];
                if (adj[i] & s) {
                    valid = false;
                }
            }
        }
        
        if (valid && val > best_val) {
            best_val = val;
            for (int i = 0; i < n; i++) {
                best_x[i] = (s >> i) & 1;
            }
        }
    }
    
    // If n > 20 or no solution found (should not happen), fall back to greedy
    // but for correctness we assume n<=20
    if (best_val == LLONG_MIN / 2) {
        // All-zero is always feasible
        best_val = 0;
        fill(best_x.begin(), best_x.end(), 0);
        
        // Simple greedy for larger n (not guaranteed optimal)
        if (n > 20) {
            vector<int> order(n);
            iota(order.begin(), order.end(), 0);
            sort(order.begin(), order.end(), [&](int a, int b) {
                return c[a] > c[b];
            });
            
            vector<bool> taken(n, false);
            long long greedy_val = 0;
            for (int i : order) {
                if (c[i] <= 0) continue;
                bool can = true;
                for (int j = 0; j < n; j++) {
                    if (taken[j] && ((adj[i] >> j) & 1)) {
                        can = false;
                        break;
                    }
                }
                if (can) {
                    taken[i] = true;
                    greedy_val += c[i];
                }
            }
            if (greedy_val > best_val) {
                best_val = greedy_val;
                for (int i = 0; i < n; i++) best_x[i] = taken[i];
            }
        }
    }
    
    cout << best_val << '\n';
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << " ";
        cout << best_x[i];
    }
    cout << '\n';
    
    return 0;
}
```
