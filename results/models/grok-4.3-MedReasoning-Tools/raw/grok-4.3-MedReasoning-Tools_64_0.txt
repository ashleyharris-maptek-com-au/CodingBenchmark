```typescript
declare const require: any;

const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const probability: number[][] = data.probability;
const drones: { id: number; x: number; y: number }[] = data.drones;
const maxStepsPerDrone: number = data.maxStepsPerDrone;

const rows: number = probability.length;
const cols: number = probability[0].length;

const dirs: number[][] = [[0, 0], [0, 1], [0, -1], [1, 0], [-1, 0]];

function isValid(x: number, y: number): boolean {
  return x >= 0 && x < cols && y >= 0 && y < rows && probability[y][x] !== -1;
}

const covered: boolean[][] = Array.from({ length: rows }, () => Array(cols).fill(false));
const paths: number[][][] = [];

for (const drone of drones) {
  let x: number = drone.x;
  let y: number = drone.y;
  const path: number[][] = [[x, y]];

  if (probability[y][x] > 0) {
    covered[y][x] = true;
  }

  for (let step = 0; step < maxStepsPerDrone; step++) {
    let bestScore: number = -Infinity;
    let bestX: number = x;
    let bestY: number = y;

    for (const [dx, dy] of dirs) {
      const nx: number = x + dx;
      const ny: number = y + dy;

      if (!isValid(nx, ny)) continue;

      let score: number = probability[ny][nx];
      if (covered[ny][nx]) {
        score = 0;
      }

      if (score > bestScore) {
        bestScore = score;
        bestX = nx;
        bestY = ny;
      }
    }

    x = bestX;
    y = bestY;
    path.push([x, y]);

    if (probability[y][x] > 0 && !covered[y][x]) {
      covered[y][x] = true;
    }
  }

  paths.push(path);
}

const output = { paths: paths };
console.log(JSON.stringify(output));
```
