```cpp
#include <iostream>
#include <vector>
#include <array>
#include <iomanip>
#include <cstdint>

struct Vec3 {
    double x, y, z;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    std::cout << std::fixed << std::setprecision(17);

    int VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) {
        return 1;
    }

    std::vector<Vec3> vertsA(VA);
    for (int i = 0; i < VA; ++i) {
        std::cin >> vertsA[i].x >> vertsA[i].y >> vertsA[i].z;
    }

    std::vector<std::array<int, 3>> facesA(FA);
    for (int i = 0; i < FA; ++i) {
        std::cin >> facesA[i][0] >> facesA[i][1] >> facesA[i][2];
    }

    std::vector<Vec3> vertsB(VB);
    for (int i = 0; i < VB; ++i) {
        std::cin >> vertsB[i].x >> vertsB[i].y >> vertsB[i].z;
    }

    std::vector<std::array<int, 3>> facesB(FB);
    for (int i = 0; i < FB; ++i) {
        std::cin >> facesB[i][0] >> facesB[i][1] >> facesB[i][2];
    }

    // Simple union by concatenation (correct when meshes do not intersect)
    // For intersecting cases a full intersection-resolution algorithm is required.
    const size_t offset = vertsA.size();
    std::vector<Vec3> outVerts;
    outVerts.reserve(vertsA.size() + vertsB.size());
    outVerts.insert(outVerts.end(), vertsA.begin(), vertsA.end());
    outVerts.insert(outVerts.end(), vertsB.begin(), vertsB.end());

    std::vector<std::array<int, 3>> outFaces;
    outFaces.reserve(facesA.size() + facesB.size());
    outFaces.insert(outFaces.end(), facesA.begin(), facesA.end());

    for (const auto& f : facesB) {
        outFaces.push_back({static_cast<int>(f[0] + offset),
                            static_cast<int>(f[1] + offset),
                            static_cast<int>(f[2] + offset)});
    }

    std::cout << outVerts.size() << " " << outFaces.size() << "\n";

    for (const auto& v : outVerts) {
        std::cout << v.x << " " << v.y << " " << v.z << "\n";
    }

    for (const auto& f : outFaces) {
        std::cout << f[0] << " " << f[1] << " " << f[2] << "\n";
    }

    return 0;
}
```
