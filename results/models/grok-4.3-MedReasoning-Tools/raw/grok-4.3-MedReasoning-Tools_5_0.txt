```rust
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let w: usize = parts[0];
    let h: usize = parts[1];
    let obs_count: usize = parts[2];
    let require_cycle: bool = parts[3] == 1;

    let mut is_block = vec![vec![false; w]; h];
    for _ in 0..obs_count {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<usize> = line
            .split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        let x = coords[0];
        let y = coords[1];
        if x < w && y < h {
            is_block[y][x] = true;
        }
    }

    if w == 0 || h == 0 {
        println!("0");
        return;
    }

    if is_block[0][0] {
        println!("0");
        return;
    }

    let mut total_free: usize = 0;
    for y in 0..h {
        for x in 0..w {
            if !is_block[y][x] {
                total_free += 1;
            }
        }
    }

    if total_free == 0 {
        println!("0");
        return;
    }

    let dirs: [(i32, i32); 4] = [(0, 1), (1, 0), (0, -1), (-1, 0)];

    let mut visited = is_block.clone();
    let mut path: Vec<(usize, usize)> = Vec::with_capacity(total_free);

    fn get_neighbors(
        x: usize,
        y: usize,
        w: usize,
        h: usize,
        visited: &Vec<Vec<bool>>,
        dirs: &[(i32, i32)],
    ) -> Vec<(usize, usize)> {
        let mut neigh = Vec::new();
        for &(dx, dy) in dirs {
            let nx = x as i32 + dx;
            let ny = y as i32 + dy;
            if nx >= 0 && nx < w as i32 && ny >= 0 && ny < h as i32 {
                let ux = nx as usize;
                let uy = ny as usize;
                if !visited[uy][ux] {
                    neigh.push((ux, uy));
                }
            }
        }
        neigh
    }

    fn count_unvisited_neighbors(
        x: usize,
        y: usize,
        w: usize,
        h: usize,
        visited: &Vec<Vec<bool>>,
        dirs: &[(i32, i32)],
    ) -> usize {
        let mut cnt = 0;
        for &(dx, dy) in dirs {
            let nx = x as i32 + dx;
            let ny = y as i32 + dy;
            if nx >= 0 && nx < w as i32 && ny >= 0 && ny < h as i32 {
                let ux = nx as usize;
                let uy = ny as usize;
                if !visited[uy][ux] {
                    cnt += 1;
                }
            }
        }
        cnt
    }

    fn dfs(
        x: usize,
        y: usize,
        path: &mut Vec<(usize, usize)>,
        visited: &mut Vec<Vec<bool>>,
        total_free: usize,
        w: usize,
        h: usize,
        require_cycle: bool,
        dirs: &[(i32, i32)],
    ) -> bool {
        path.push((x, y));
        visited[y][x] = true;

        if path.len() == total_free {
            let success = if require_cycle && total_free > 1 {
                let last = path.last().unwrap();
                let dx = (last.0 as i32).abs_diff(0);
                let dy = (last.1 as i32).abs_diff(0);
                dx + dy == 1
            } else {
                true
            };
            if success {
                return true;
            } else {
                visited[y][x] = false;
                path.pop();
                return false;
            }
        }

        let mut neigh = get_neighbors(x, y, w, h, visited, dirs);

        // Warnsdorff heuristic: sort by ascending number of unvisited neighbors
        let mut degrees: Vec<usize> = neigh
            .iter()
            .map(|&(nx, ny)| count_unvisited_neighbors(nx, ny, w, h, visited, dirs))
            .collect();

        let mut indices: Vec<usize> = (0..neigh.len()).collect();
        indices.sort_by_key(|&i| degrees[i]);

        let sorted_neigh: Vec<(usize, usize)> = indices.iter().map(|&i| neigh[i]).collect();
        neigh = sorted_neigh;

        for &(nx, ny) in &neigh {
            if dfs(nx, ny, path, visited, total_free, w, h, require_cycle, dirs) {
                return true;
            }
        }

        // backtrack
        visited[y][x] = false;
        path.pop();
        false
    }

    if dfs(0, 0, &mut path, &mut visited, total_free, w, h, require_cycle, &dirs) {
        println!("{}", path.len());
        for &(x, y) in &path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}
```
