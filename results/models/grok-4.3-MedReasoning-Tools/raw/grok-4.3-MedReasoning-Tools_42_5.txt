```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 200
OpCapability Geometry
OpExtension "SPV_KHR_vulkan_memory_model"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %out_outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %out_outColor Location 0

; Types
%void = OpTypeVoid
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_2
%arr_v3float_2 = OpTypeArray %v3float %uint_2

%ptr_Input_gl_PerVertex_arr = OpTypePointer Input %arr_gl_PerVertex_in
%ptr_Input_v3_arr = OpTypePointer Input %arr_v3float_2
%ptr_Output_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v3float = OpTypePointer Output %v3float
%ptr_v4float = OpTypePointer Function %v4float
%ptr_v3float = OpTypePointer Function %v3float

%fn_main = OpTypeFunction %void

; Constants
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_015 = OpConstant %float 0.015
%float_05 = OpConstant %float 0.05
%float_075 = OpConstant %float 0.75

; Variables
%gl_in = OpVariable %ptr_Input_gl_PerVertex_arr Input
%gl_out = OpVariable %ptr_Output_gl_PerVertex Output
%in_worldPos = OpVariable %ptr_Input_v3_arr Input
%in_normal = OpVariable %ptr_Input_v3_arr Input
%in_color = OpVariable %ptr_Input_v3_arr Input
%out_outColor = OpVariable %ptr_Output_v3float Output

; GLSL.std.450 import
%glsl_std450 = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %fn_main
%entry = OpLabel

; Load inputs
%ptr_worldPos0 = OpAccessChain %ptr_Input_v3 %in_worldPos %int_0
%p0 = OpLoad %v3float %ptr_worldPos0
%ptr_worldPos1 = OpAccessChain %ptr_Input_v3 %in_worldPos %int_1
%p1 = OpLoad %v3float %ptr_worldPos1

%ptr_color0 = OpAccessChain %ptr_Input_v3 %in_color %int_0
%c0 = OpLoad %v3float %ptr_color0
%ptr_color1 = OpAccessChain %ptr_Input_v3 %in_color %int_1
%c1 = OpLoad %v3float %ptr_color1

; dir = p1 - p0
%dir = OpFSub %v3float %p1 %p0

; Extract xy for 2D perpendicular calculation
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1

; perp = normalize(-dir.y, dir.x)
%neg_dir_y = OpFNegate %float %dir_y
%perp2 = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp_norm = OpExtInst %v2float %glsl_std450 Normalize %perp2

; Precompute scaled perps
%shaft_perp = OpVectorTimesScalar %v2float %perp_norm %float_015
%arrow_perp = OpVectorTimesScalar %v2float %perp_norm %float_05

; mid = p0 + 0.75 * (p1 - p0)
%delta_scaled = OpVectorTimesScalar %v3float %dir %float_075
%mid = OpFAdd %v3float %p0 %delta_scaled

; ===== SHAFT (4 vertices, triangle strip) =====
; Extract components for p0 and mid
%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p0z = OpCompositeExtract %float %p0 2

%midx = OpCompositeExtract %float %mid 0
%midy = OpCompositeExtract %float %mid 1
%midz = OpCompositeExtract %float %mid 2

; Shaft vertex 0: p0 - perp*0.015
%sp0x = OpFSub %float %p0x %shaft_perp_x
%sp0y = OpFSub %float %p0y %shaft_perp_y
%shaft0 = OpCompositeConstruct %v4float %sp0x %sp0y %p0z %float_1
%shaft0_color = OpCopyObject %v3float %c0

; Shaft vertex 1: p0 + perp*0.015
%sp1x = OpFAdd %float %p0x %shaft_perp_x
%sp1y = OpFAdd %float %p0y %shaft_perp_y
%shaft1 = OpCompositeConstruct %v4float %sp1x %sp1y %p0z %float_1
%shaft1_color = OpCopyObject %v3float %c0

; Shaft vertex 2: mid - perp*0.015
%sm0x = OpFSub %float %midx %shaft_perp_x
%sm0y = OpFSub %float %midy %shaft_perp_y
%shaft2 = OpCompositeConstruct %v4float %sm0x %sm0y %midz %float_1
%shaft2_color = OpCopyObject %v3float %c0

; Shaft vertex 3: mid + perp*0.015
%sm1x = OpFAdd %float %midx %shaft_perp_x
%sm1y = OpFAdd %float %midy %shaft_perp_y
%shaft3 = OpCompositeConstruct %v4float %sm1x %sm1y %midz %float_1
%shaft3_color = OpCopyObject %v3float %c0

; Emit shaft vertices
OpStore %out_outColor %shaft0_color
%pos0_ptr = OpAccessChain %ptr_v4float %gl_out %int_0
OpStore %pos0_ptr %shaft0
OpEmitVertex

OpStore %out_outColor %shaft1_color
%pos1_ptr = OpAccessChain %ptr_v4float %gl_out %int_0
OpStore %pos1_ptr %shaft1
OpEmitVertex

OpStore %out_outColor %shaft2_color
%pos2_ptr = OpAccessChain %ptr_v4float %gl_out %int_0
OpStore %pos2_ptr %shaft2
OpEmitVertex

OpStore %out_outColor %shaft3_color
%pos3_ptr = OpAccessChain %ptr_v4float %gl_out %int_0
OpStore %pos3_ptr %shaft3
OpEmitVertex

OpEndPrimitive

; ===== ARROWHEAD (3 vertices, triangle strip) =====
; Arrow base points use mid ± arrow_perp, tip = p1

%p1x = OpCompositeExtract %float %p1 0
%p1y = OpCompositeExtract %float %p1 1
%p1z = OpCompositeExtract %float %p1 2

; Arrow vertex 0: mid + perp*0.05
%ap0x = OpFAdd %float %midx %arrow_perp_x
%ap0y = OpFAdd %float %midy %arrow_perp_y
%arrow0 = OpCompositeConstruct %v4float %ap0x %ap0y %midz %float_1
%arrow0_color = OpCopyObject %v3float %c1

; Arrow vertex 1: mid - perp*0.05
%ap1x = OpFSub %float %midx %arrow_perp_x
%ap1y = OpFSub %float %midy %arrow_perp_y
%arrow1 = OpCompositeConstruct %v4float %ap1x %ap1y %midz %float_1
%arrow1_color = OpCopyObject %v3float %c1

; Arrow vertex 2: p1 (tip)
%arrow2 = OpCompositeConstruct %v4float %p1x %p1y %p1z %float_1
%arrow2_color = OpCopyObject %v3float %c1

; Emit arrowhead
OpStore %out_outColor %arrow0_color
%apos0_ptr = OpAccessChain %ptr_v4float %gl_out %int_0
OpStore %apos0_ptr %arrow0
OpEmitVertex

OpStore %out_outColor %arrow1_color
%apos1_ptr = OpAccessChain %ptr_v4float %gl_out %int_0
OpStore %apos1_ptr %arrow1
OpEmitVertex

OpStore %out_outColor %arrow2_color
%apos2_ptr = OpAccessChain %ptr_v4float %gl_out %int_0
OpStore %apos2_ptr %arrow2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
