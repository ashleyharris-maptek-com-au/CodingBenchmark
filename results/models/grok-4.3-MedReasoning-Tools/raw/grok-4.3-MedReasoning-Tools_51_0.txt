```python
import math

# Module-level persistent state for controllers and filters
_pid_state = {
    'alt_int': 0.0,
    'alt_prev_err': 0.0,
    'hdg_int': 0.0,
    'hdg_prev_err': 0.0,
    'spd_int': 0.0,
    'spd_prev_err': 0.0,
    'pitch_int': 0.0,
    'yaw_damp_int': 0.0,
    'trim': 0.0,
    'last_ias': 250.0,
    'last_alt': 30000.0,
    'last_hdg': 0.0,
    'phase': 'cruise',          # cruise / climb / descent / approach / takeoff
    'last_radar_alt': 10000.0,
    'time_since_ground': 60.0,
}

def _shortest_angle_diff(target, current):
    diff = (target - current + 180.0) % 360.0 - 180.0
    return diff

def _median(values):
    if not values:
        return 0.0
    s = sorted(values)
    n = len(s)
    mid = n // 2
    if n % 2 == 1:
        return s[mid]
    return (s[mid-1] + s[mid]) / 2.0

def _select_best_alt(state):
    # Prefer radar below ~2500 ft, baro otherwise. Use median of valid sensors.
    radar_alts = []
    for i in [1, 2, 3]:
        if state.get(f'radar_alt_{i}_valid', False):
            val = state.get(f'radar_alt_{i}_ft', 0.0)
            if 0 < val < 8000:
                radar_alts.append(val)
    if radar_alts and _pid_state['last_radar_alt'] < 3000:
        return _median(radar_alts)

    baro_alts = []
    for i in [1, 2, 3]:
        if state.get(f'baro_alt_{i}_valid', False):
            baro_alts.append(state.get(f'baro_alt_{i}_ft', 0.0))
    if len(baro_alts) >= 2:
        return _median(baro_alts)
    if baro_alts:
        return baro_alts[0]

    # Fallback to GPS
    gps_alts = []
    for i in [1, 2, 3]:
        if state.get(f'gps_{i}_valid', False):
            gps_alts.append(state.get(f'gps_{i}_alt_ft', 0.0))
    if gps_alts:
        return _median(gps_alts)

    return _pid_state['last_alt']

def _select_best_ias(state):
    ias_vals = []
    for i in [1, 2, 3]:
        if state.get(f'ias_{i}_valid', False):
            val = state.get(f'ias_{i}_kt', 0.0)
            if 50 < val < 450:
                ias_vals.append(val)
    if ias_vals:
        return _median(ias_vals)
    return _pid_state['last_ias']

def _select_best_hdg(state):
    candidates = []
    if state.get('irs_1_valid', False):
        candidates.append(state.get('irs_1_heading_deg', 0.0))
    if state.get('irs_2_valid', False):
        candidates.append(state.get('irs_2_heading_deg', 0.0))
    if state.get('irs_3_valid', False):
        candidates.append(state.get('irs_3_heading_deg', 0.0))
    if state.get('gps_1_valid', False):
        candidates.append(state.get('gps_1_track_deg', 0.0))
    if candidates:
        return _median(candidates)
    return state.get('magnetic_heading_deg', state.get('true_heading_deg', 0.0))

def _select_attitude(state):
    roll = pitch = 0.0
    valid_count = 0
    for i in [1, 2, 3]:
        if state.get(f'irs_{i}_valid', False):
            roll += state.get(f'irs_{i}_roll_deg', 0.0)
            pitch += state.get(f'irs_{i}_pitch_deg', 0.0)
            valid_count += 1
    if valid_count > 0:
        return roll / valid_count, pitch / valid_count

    # Gyro fallback (integrated attitude would be done by sim, use current)
    return state.get('irs_1_roll_deg', 0.0), state.get('irs_1_pitch_deg', 0.0)

def _select_rates(state):
    roll_rate = pitch_rate = yaw_rate = 0.0
    valid = 0
    for i in [1, 2, 3]:
        if state.get(f'gyro_{i}_valid', False):
            roll_rate += state.get(f'gyro_{i}_roll_rate_dps', 0.0)
            pitch_rate += state.get(f'gyro_{i}_pitch_rate_dps', 0.0)
            yaw_rate += state.get(f'gyro_{i}_yaw_rate_dps', 0.0)
            valid += 1
    if valid > 0:
        return roll_rate / valid, pitch_rate / valid, yaw_rate / valid
    return 0.0, 0.0, 0.0

def autopilot_step(state: dict, dt: float) -> dict:
    # === SENSOR FUSION ===
    alt = _select_best_alt(state)
    ias = _select_best_ias(state)
    hdg = _select_best_hdg(state)
    roll, pitch = _select_attitude(state)
    roll_rate, pitch_rate, yaw_rate = _select_rates(state)

    vs = state.get('vertical_speed_fpm', 0.0)
    radar_alt = state.get('radar_alt_1_ft', state.get('radar_alt_2_ft', alt * 0.3))
    aoa = max([state.get(f'aoa_{i}_deg', 0.0) for i in [1,2,3] if state.get(f'aoa_{i}_valid', False)] or [5.0])
    mach = max(state.get('mach_1', 0.0), state.get('mach_2', 0.0), state.get('mach_3', 0.0))

    # Update persistent filters
    _pid_state['last_alt'] = 0.95 * _pid_state['last_alt'] + 0.05 * alt
    _pid_state['last_ias'] = 0.9 * _pid_state['last_ias'] + 0.1 * ias
    _pid_state['last_hdg'] = hdg
    _pid_state['last_radar_alt'] = 0.7 * _pid_state['last_radar_alt'] + 0.3 * radar_alt

    # Phase detection (very lightweight)
    wow = any(state.get(k, False) for k in ['weight_on_wheels_left', 'weight_on_wheels_nose', 'weight_on_wheels_right'])
    if wow:
        _pid_state['phase'] = 'takeoff'
        _pid_state['time_since_ground'] = 0.0
    else:
        _pid_state['time_since_ground'] += dt
        if radar_alt < 800 and vs < -300:
            _pid_state['phase'] = 'approach'
        elif alt < 3000 and ias < 220:
            _pid_state['phase'] = 'approach'
        elif vs > 300 and alt < 18000:
            _pid_state['phase'] = 'climb'
        elif vs < -300:
            _pid_state['phase'] = 'descent'
        else:
            _pid_state['phase'] = 'cruise'

    # === TARGETS ===
    tgt_alt = state.get('target_alt_ft', alt)
    tgt_hdg = state.get('target_hdg_deg', hdg)
    tgt_ias = state.get('target_ias_kt', ias)
    ap_on = state.get('ap_engaged', True)
    at_on = state.get('at_engaged', True)

    # === LATERAL (Heading hold via aileron) ===
    hdg_err = _shortest_angle_diff(tgt_hdg, hdg)
    desired_roll = max(min(hdg_err * 1.8, 28.0), -28.0)
    roll_err = desired_roll - roll

    # PD + integral (anti-windup)
    _pid_state['hdg_int'] += hdg_err * dt * 0.02
    _pid_state['hdg_int'] = max(min(_pid_state['hdg_int'], 40.0), -40.0)
    aileron = (0.035 * roll_err +
               0.008 * roll_rate * -1.0 +
               0.0008 * _pid_state['hdg_int'])
    aileron = max(min(aileron, 1.0), -1.0)

    # === VERTICAL (Altitude / VS capture via elevator) ===
    alt_err = tgt_alt - alt
    # Desired VS proportional to altitude error, limited
    desired_vs = max(min(alt_err * 0.45, 2200.0), -2200.0)
    if _pid_state['phase'] == 'approach':
        desired_vs = max(min(desired_vs, 800), -1200)
    vs_err = desired_vs - vs

    desired_pitch = max(min(vs_err * 0.0035, 12.0), -8.0)
    pitch_err = desired_pitch - pitch

    _pid_state['alt_int'] += alt_err * dt * 0.00015
    _pid_state['alt_int'] = max(min(_pid_state['alt_int'], 800.0), -800.0)
    _pid_state['pitch_int'] += pitch_err * dt * 0.04
    _pid_state['pitch_int'] = max(min(_pid_state['pitch_int'], 5.0), -5.0)

    elevator = (0.065 * pitch_err +
                0.012 * pitch_rate * -1.0 +
                0.0008 * _pid_state['alt_int'] +
                0.008 * _pid_state['pitch_int'])
    elevator = max(min(elevator, 1.0), -1.0)

    # Stall / high AoA protection
    if aoa > 11.5:
        elevator = min(elevator, -0.25)
    if aoa > 14.0:
        elevator = min(elevator, -0.55)

    # === SPEED / AUTOTHROTTLE ===
    spd_err = tgt_ias - ias
    _pid_state['spd_int'] += spd_err * dt * 0.003
    _pid_state['spd_int'] = max(min(_pid_state['spd_int'], 120.0), -120.0)

    throttle_cmd = 0.018 * spd_err + 0.0009 * _pid_state['spd_int']
    # Phase-based limits
    if _pid_state['phase'] in ('approach', 'descent'):
        throttle_cmd = min(throttle_cmd, 0.75)
    if mach > 0.80 or ias > 355:
        throttle_cmd = min(throttle_cmd, 0.60)

    throttle = max(min(throttle_cmd, 1.0), 0.05)

    # Engine failure / fire handling (asymmetric thrust)
    eng1_ok = state.get('engine_1_running', True) and not state.get('engine_1_fire', False)
    eng2_ok = state.get('engine_2_running', True) and not state.get('engine_2_fire', False)
    n1_1 = state.get('engine_1_n1_pct', 80.0)
    n1_2 = state.get('engine_2_n1_pct', 80.0)

    if not eng1_ok or n1_1 < 25:
        throttle_1 = 0.0
    else:
        throttle_1 = throttle
    if not eng2_ok or n1_2 < 25:
        throttle_2 = 0.0
    else:
        throttle_2 = throttle

    # === RUDDER (Yaw damper + engine-out compensation) ===
    rudder = -0.18 * yaw_rate   # primary yaw damper

    # Asymmetric thrust compensation
    thrust_diff = (n1_1 - n1_2) * 0.0035
    if not eng1_ok:
        rudder += 0.25   # right rudder for left engine failure
    elif not eng2_ok:
        rudder -= 0.25   # left rudder for right engine failure
    else:
        rudder += thrust_diff

    rudder = max(min(rudder, 0.85), -0.85)

    # === TRIM (slowly relieves elevator) ===
    if abs(elevator) > 0.12 and abs(alt_err) < 800:
        trim_step = 0.008 * math.copysign(1.0, elevator) * dt
        _pid_state['trim'] += trim_step
    _pid_state['trim'] = max(min(_pid_state['trim'], 0.6), -0.6)
    elevator_trim = _pid_state['trim']

    # If we trimmed a lot, ease elevator command slightly
    if abs(_pid_state['trim']) > 0.25:
        elevator *= 0.85

    # === FLAPS, GEAR, SPEEDBRAKE ===
    flaps = 0.0
    speedbrake = 0.0
    gear = 0

    # Speedbrake / overspeed protection
    if ias > 355 or mach > 0.81:
        speedbrake = min(0.6 + (ias - 355) * 0.01, 1.0)

    # Flap schedule (conservative)
    if _pid_state['phase'] in ('takeoff', 'approach'):
        if ias < 145:
            flaps = 25
        elif ias < 175:
            flaps = 15
        elif ias < 210:
            flaps = 5
        else:
            flaps = 0
    else:
        if ias < 165:
            flaps = 10
        elif ias < 195:
            flaps = 5
        else:
            flaps = 0

    # Gear logic
    if _pid_state['phase'] == 'takeoff':
        gear = 0 if _pid_state['time_since_ground'] > 8.0 and radar_alt > 30 else 1
    elif _pid_state['phase'] == 'approach':
        if radar_alt < 2500 or alt < 2200:
            gear = 1
        else:
            gear = 0
    elif alt < 1800 and vs < 0:
        gear = 1
    else:
        gear = 0

    # Final sanity clamps
    elevator = max(min(elevator, 1.0), -1.0)
    aileron = max(min(aileron, 1.0), -1.0)
    rudder = max(min(rudder, 1.0), -1.0)
    throttle_1 = max(min(throttle_1, 1.0), 0.0)
    throttle_2 = max(min(throttle_2, 1.0), 0.0)
    flaps = max(min(flaps, 40.0), 0.0)
    speedbrake = max(min(speedbrake, 1.0), 0.0)
    gear = 1 if gear else 0

    return {
        'elevator': round(elevator, 4),
        'aileron': round(aileron, 4),
        'rudder': round(rudder, 4),
        'throttle_1': round(throttle_1, 4),
        'throttle_2': round(throttle_2, 4),
        'elevator_trim': round(elevator_trim, 4),
        'flaps': round(flaps, 1),
        'speedbrake': round(speedbrake, 3),
        'gear': int(gear)
    }
```
