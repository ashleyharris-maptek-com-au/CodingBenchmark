#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    cin >> J >> M;

    vector<vector<pair<int, int>>> job_tasks(J);
    long long total_tasks = 0;

    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        job_tasks[j].resize(T);
        total_tasks += T;
        for (int t = 0; t < T; ++t) {
            int m, d;
            cin >> m >> d;
            job_tasks[j][t] = {m, d};
        }
    }

    // Adaptive strategy selection based on problem scale
    bool is_small = (J <= 30 && total_tasks <= 800);
    bool is_medium = (total_tasks <= 8000);

    vector<int> job_order(J);
    iota(job_order.begin(), job_order.end(), 0);

    if (is_small) {
        // SPT (Shortest Processing Time) job ordering for better quality on small instances
        vector<long long> job_sums(J, 0);
        for (int j = 0; j < J; ++j) {
            for (auto& p : job_tasks[j]) {
                job_sums[j] += p.second;
            }
        }
        sort(job_order.begin(), job_order.end(), [&](int a, int b) {
            return job_sums[a] < job_sums[b];
        });
    } else if (is_medium) {
        // For medium scale: use input order (fast and stable)
    } else {
        // Large scale: input order only (minimal overhead)
    }

    vector<long long> machine_free(M, 0LL);
    vector<vector<pair<long long, int>>> schedule(J);

    for (int j : job_order) {
        long long job_free = 0LL;
        const auto& tasks = job_tasks[j];
        schedule[j].resize(tasks.size());

        for (size_t t = 0; t < tasks.size(); ++t) {
            int m = tasks[t].first;
            long long d = tasks[t].second;   // promote to long long
            long long st = max(job_free, machine_free[m]);
            schedule[j][t] = {st, m};
            machine_free[m] = st + d;
            job_free = st + d;
        }
    }

    // Output schedule in job order
    for (int j = 0; j < J; ++j) {
        const auto& sch = schedule[j];
        for (size_t t = 0; t < sch.size(); ++t) {
            if (t > 0) cout << ' ';
            cout << sch[t].first << ' ' << sch[t].second;
        }
        cout << '\n';
    }

    return 0;
}
