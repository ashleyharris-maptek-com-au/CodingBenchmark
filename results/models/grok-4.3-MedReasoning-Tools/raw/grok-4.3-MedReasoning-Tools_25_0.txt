using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public class Vector3
{
    public double X, Y, Z;

    public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
    public Vector3() : this(0, 0, 0) { }

    public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    public static Vector3 operator *(Vector3 a, double s) => new Vector3(a.X * s, a.Y * s, a.Z * s);
    public static Vector3 operator /(Vector3 a, double s) => new Vector3(a.X / s, a.Y / s, a.Z / s);

    public double Magnitude() => Math.Sqrt(X * X + Y * Y + Z * Z);
    public Vector3 Normalized()
    {
        double mag = Magnitude();
        return mag > 1e-12 ? this / mag : new Vector3();
    }

    public override string ToString() => $"{X:F12} {Y:F12} {Z:F12}";
}

public class Body
{
    public string Name;
    public double Mass;
    public Vector3 Pos;
    public Vector3 Vel;
}

public class AsteroidInterceptor
{
    static void Main()
    {
        var culture = CultureInfo.InvariantCulture;
        string line = Console.ReadLine();
        if (line == null) return;

        var parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int numBodies = int.Parse(parts[0]);
        double warningDays = double.Parse(parts[1], culture);
        double impactorMass = double.Parse(parts[2], culture);
        double dvBudget = double.Parse(parts[3], culture);
        double asteroidMass = double.Parse(parts[4], culture);
        double t0Unix = double.Parse(parts[5], culture);

        var bodies = new List<Body>();
        Body earth = null;
        Body asteroid = null;

        for (int i = 0; i < numBodies; i++)
        {
            line = Console.ReadLine();
            if (line == null) continue;
            var bp = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            var body = new Body
            {
                Name = bp[0],
                Mass = double.Parse(bp[1], culture),
                Pos = new Vector3(double.Parse(bp[2], culture), double.Parse(bp[3], culture), double.Parse(bp[4], culture)),
                Vel = new Vector3(double.Parse(bp[5], culture), double.Parse(bp[6], culture), double.Parse(bp[7], culture))
            };
            bodies.Add(body);
            if (body.Name.Equals("Earth", StringComparison.OrdinalIgnoreCase)) earth = body;
            if (body.Name.Equals("Asteroid", StringComparison.OrdinalIgnoreCase)) asteroid = body;
        }

        string bspPath = Console.ReadLine()?.Trim() ?? "de421.bsp";

        if (earth == null || asteroid == null)
        {
            Console.WriteLine("0");
            return;
        }

        // Simple heuristic solver for intercept trajectory
        // Uses linear prediction + crude Lambert-like direct intercept
        double G = 6.67430e-11;
        double earthRadius = 6.371e6;
        double safetyMargin = 1.5e9; // ~10 Earth radii margin for deflection

        // Choose impact time based on warning window and complexity heuristic
        double tImpact = Math.Min(warningDays * 0.85, warningDays - 5);
        if (warningDays < 730) tImpact = warningDays * 0.65; // short warning: earlier intercept
        if (warningDays > 10000) tImpact = warningDays * 0.92; // long warning: more time for assists

        // Linear prediction for target position at impact
        Vector3 asteroidFuture = asteroid.Pos + asteroid.Vel * (tImpact * 86400.0);
        Vector3 earthFuture = earth.Pos + earth.Vel * (tImpact * 86400.0);

        // Launch from near Earth (assume launch from Earth position at t=0)
        Vector3 launchPos = earth.Pos;
        Vector3 targetPos = asteroidFuture;

        // Required velocity for direct intercept (crude two-body approximation)
        double dtSec = tImpact * 86400.0;
        Vector3 deltaPos = targetPos - launchPos;
        Vector3 requiredVel = deltaPos / dtSec;

        // Current velocity at Earth
        Vector3 currentVel = earth.Vel;
        Vector3 launchDv = requiredVel - currentVel;

        // Scale down if exceeds budget (prioritize intercept)
        double launchMag = launchDv.Magnitude();
        double maxLaunchDv = Math.Min(dvBudget * 0.65, 12000.0);
        if (launchMag > maxLaunchDv)
        {
            launchDv = launchDv.Normalized() * maxLaunchDv;
        }

        // Mid-course correction (small)
        double tCorrection = tImpact * 0.45;
        Vector3 correctionDv = new Vector3(0, 0, 0);
        if (warningDays > 365 && dvBudget > 500)
        {
            // Small plane change / timing correction for long missions
            correctionDv = (asteroid.Vel - earth.Vel) * 0.03;
            double corrMag = correctionDv.Magnitude();
            if (corrMag > 150) correctionDv = correctionDv.Normalized() * 120;
        }

        // Final impact burn (kinetic deflection)
        // Calculate required deflection delta-v at impact
        double relSpeed = (asteroid.Vel - earth.Vel).Magnitude();
        double momentumTransferFactor = impactorMass / (asteroidMass + impactorMass);
        double neededDeflectionDv = Math.Max(0.05, safetyMargin / (tImpact * 86400.0 * 2.0)); // rough miss distance

        Vector3 impactDirection = (asteroid.Vel - earth.Vel).Normalized();
        Vector3 impactDv = impactDirection * (neededDeflectionDv / momentumTransferFactor);

        double impactMag = impactDv.Magnitude();
        double remainingDv = dvBudget - launchMag - correctionDv.Magnitude();
        if (impactMag > remainingDv * 0.8)
        {
            impactDv = impactDv.Normalized() * Math.Min(impactMag, remainingDv * 0.8);
        }

        // Output format
        int numBurns = (correctionDv.Magnitude() > 5.0) ? 3 : 2;

        Console.WriteLine(numBurns);

        // Burn 1: Launch
        Console.WriteLine($"0.0 {launchDv.X:F10} {launchDv.Y:F10} {launchDv.Z:F10}");

        // Burn 2: Mid-course (if used)
        if (numBurns == 3)
        {
            Console.WriteLine($"{tCorrection:F6} {correctionDv.X:F10} {correctionDv.Y:F10} {correctionDv.Z:F10}");
        }

        // Final impact burn
        Console.WriteLine($"{tImpact:F6} {impactDv.X:F10} {impactDv.Y:F10} {impactDv.Z:F10}");
    }
}
