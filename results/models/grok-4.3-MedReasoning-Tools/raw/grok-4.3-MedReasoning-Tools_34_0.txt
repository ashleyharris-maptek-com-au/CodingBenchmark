```python
import sys
import math

def main():
    # Read input
    line = sys.stdin.readline().strip()
    if not line:
        return
    parts = line.split()
    N = int(parts[0])
    K = int(parts[1])
    CAPACITY = float(parts[2])
    
    line = sys.stdin.readline().strip()
    dx, dy = map(float, line.split())
    depot = (dx, dy)
    
    customers = []
    for _ in range(N):
        line = sys.stdin.readline().strip()
        x, y, d = map(float, line.split())
        customers.append((x, y, d))
    
    if N == 0:
        print(0.0)
        print(0)
        return
    
    # Precompute distances
    dist_depot = []
    for i in range(N):
        cx, cy, _ = customers[i]
        dist_depot.append(math.hypot(dx - cx, dy - cy))
    
    dist_matrix = [[0.0] * N for _ in range(N)]
    for i in range(N):
        x1, y1, _ = customers[i]
        for j in range(i + 1, N):
            x2, y2, _ = customers[j]
            d = math.hypot(x1 - x2, y1 - y2)
            dist_matrix[i][j] = d
            dist_matrix[j][i] = d
    
    # Best-fit decreasing assignment to K routes (capacity bins)
    demands = [c[2] for c in customers]
    cust_indices = sorted(range(N), key=lambda i: -demands[i])
    
    route_assignments = [[] for _ in range(K)]
    route_loads = [0.0] * K
    
    for i in cust_indices:
        d = demands[i]
        best_r = -1
        min_remaining = float('inf')
        for r in range(K):
            if route_loads[r] + d <= CAPACITY:
                rem = CAPACITY - route_loads[r]
                if rem < min_remaining:
                    min_remaining = rem
                    best_r = r
        if best_r != -1:
            route_assignments[best_r].append(i)
            route_loads[best_r] += d
    
    # Build ordered routes using nearest-neighbor heuristic
    def get_ordered_route(cust_list):
        if not cust_list:
            return []
        # Start with customer closest to depot
        current = min(cust_list, key=lambda x: dist_depot[x])
        order = [current]
        remaining = set(cust_list)
        remaining.remove(current)
        
        while remaining:
            next_c = min(remaining, key=lambda x: dist_matrix[current][x])
            order.append(next_c)
            remaining.remove(next_c)
            current = next_c
        return order
    
    final_routes = []
    total_distance = 0.0
    
    for r in range(K):
        custs = route_assignments[r]
        if not custs:
            continue
        order = get_ordered_route(custs)
        
        # Compute route distance
        rd = dist_depot[order[0]]
        for i in range(len(order) - 1):
            rd += dist_matrix[order[i]][order[i + 1]]
        rd += dist_depot[order[-1]]
        
        total_distance += rd
        final_routes.append(order)
    
    # Output
    print(total_distance)
    print(len(final_routes))
    for route in final_routes:
        print(' '.join(map(str, route)))

if __name__ == "__main__":
    main()
```
