```rust
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read first line: num_vertices num_edges
    let first_line = match lines.next() {
        Some(Ok(line)) => line,
        _ => {
            eprintln!("Error reading input");
            return;
        }
    };

    let mut parts = first_line.split_whitespace();
    let n: usize = match parts.next().and_then(|s| s.parse().ok()) {
        Some(val) => val,
        None => {
            eprintln!("Invalid number of vertices");
            return;
        }
    };
    let m: usize = match parts.next().and_then(|s| s.parse().ok()) {
        Some(val) => val,
        None => 0,
    };

    // Build adjacency list
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];

    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let mut edge_parts = line.split_whitespace();
            if let (Some(u_str), Some(v_str)) = (edge_parts.next(), edge_parts.next()) {
                if let (Ok(u), Ok(v)) = (u_str.parse::<usize>(), v_str.parse::<usize>()) {
                    if u < n && v < n {
                        adj[u].push(v);
                        adj[v].push(u);
                    }
                }
            }
        }
    }

    // Deduplicate and sort adjacency lists
    for i in 0..n {
        adj[i].sort_unstable();
        adj[i].dedup();
    }

    // Compute dominating set using greedy heuristic
    let mut dominating_set = greedy_dominating_set(&adj);

    // Sort for deterministic output
    dominating_set.sort_unstable();

    // Output
    println!("{}", dominating_set.len());
    if !dominating_set.is_empty() {
        println!(
            "{}",
            dominating_set
                .iter()
                .map(|v| v.to_string())
                .collect::<Vec<String>>()
                .join(" ")
        );
    } else {
        println!();
    }
}

fn greedy_dominating_set(adj: &[Vec<usize>]) -> Vec<usize> {
    let n = adj.len();
    if n == 0 {
        return Vec::new();
    }

    let mut covered = vec![false; n];
    let mut selected = vec![false; n];
    let mut result = Vec::new();

    // Greedy selection: repeatedly pick vertex covering the most uncovered vertices
    loop {
        let mut max_cover_count = 0usize;
        let mut best_vertex: Option<usize> = None;

        for v in 0..n {
            if selected[v] {
                continue;
            }

            let mut cover_count = 0usize;

            // Count self if not covered
            if !covered[v] {
                cover_count += 1;
            }

            // Count uncovered neighbors
            for &neighbor in &adj[v] {
                if !covered[neighbor] {
                    cover_count += 1;
                }
            }

            if cover_count > max_cover_count {
                max_cover_count = cover_count;
                best_vertex = Some(v);
            }
        }

        if max_cover_count == 0 {
            break; // All vertices are covered
        }

        if let Some(v) = best_vertex {
            selected[v] = true;
            result.push(v);

            // Mark newly covered vertices
            if !covered[v] {
                covered[v] = true;
            }
            for &neighbor in &adj[v] {
                if !covered[neighbor] {
                    covered[neighbor] = true;
                }
            }
        }
    }

    result
}
```
