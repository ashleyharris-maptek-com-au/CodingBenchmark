```typescript
declare const require: any;

const fs = require('fs');

interface Package {
  id: any;
  x: number;
  y: number;
}

interface Depot {
  x: number;
  y: number;
}

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depotPoint: Depot = { x: data.depot[0], y: data.depot[1] };
const truckCount: number = data.truckCount;
const packages: Package[] = data.packages;

const k = Math.min(truckCount, packages.length || 0);

if (k === 0 || packages.length === 0) {
  console.log(JSON.stringify({ routes: [] }));
  process.exit(0);
}

// Attach polar angle and sort (sweep algorithm)
const pkgsWithMeta = packages.map(pkg => {
  const dx = pkg.x - depotPoint.x;
  const dy = pkg.y - depotPoint.y;
  const angle = Math.atan2(dy, dx);
  const dist = Math.hypot(dx, dy);
  return { ...pkg, angle, dist };
});

pkgsWithMeta.sort((a, b) => {
  if (a.angle !== b.angle) return a.angle - b.angle;
  return a.dist - b.dist;
});

// Divide into k contiguous angular sectors (balanced by count)
const groups: Package[][] = Array.from({ length: k }, () => []);
const baseSize = Math.floor(pkgsWithMeta.length / k);
let remainder = pkgsWithMeta.length % k;
let currentIndex = 0;

for (let g = 0; g < k; g++) {
  const size = baseSize + (remainder > 0 ? 1 : 0);
  if (remainder > 0) remainder--;

  for (let i = 0; i < size && currentIndex < pkgsWithMeta.length; i++) {
    const meta = pkgsWithMeta[currentIndex];
    groups[g].push({ id: meta.id, x: meta.x, y: meta.y });
    currentIndex++;
  }
}

// Nearest-neighbor heuristic within each cluster (starting from depot)
function nearestNeighborRoute(group: Package[], depot: Depot): any[] {
  if (group.length === 0) return [];

  const remaining: Package[] = [...group];
  const route: any[] = [];
  let current: Depot = { ...depot };

  while (remaining.length > 0) {
    let bestIdx = 0;
    let minDist = Infinity;

    for (let i = 0; i < remaining.length; i++) {
      const d = Math.hypot(
        remaining[i].x - current.x,
        remaining[i].y - current.y
      );
      if (d < minDist) {
        minDist = d;
        bestIdx = i;
      }
    }

    const nextPkg = remaining.splice(bestIdx, 1)[0];
    route.push(nextPkg.id);
    current = { x: nextPkg.x, y: nextPkg.y };
  }

  return route;
}

const routes: any[][] = groups
  .map(group => nearestNeighborRoute(group, depotPoint))
  .filter(route => route.length > 0);

const output = { routes };
console.log(JSON.stringify(output));
```
