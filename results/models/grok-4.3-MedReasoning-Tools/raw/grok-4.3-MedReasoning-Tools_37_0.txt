```csharp
using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static List<int>[] adj;
    static int n;
    static bool[] removedGlobal;

    static void Main()
    {
        string input = Console.In.ReadToEnd();
        string[] tokens = input.Split(new char[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        n = int.Parse(tokens[idx++]);
        int m = int.Parse(tokens[idx++]);

        adj = new List<int>[n];
        for (int i = 0; i < n; i++)
            adj[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            int u = int.Parse(tokens[idx++]);
            int v = int.Parse(tokens[idx++]);
            if (u >= 0 && u < n && v >= 0 && v < n)
                adj[u].Add(v);
        }

        // Remove duplicate edges
        for (int i = 0; i < n; i++)
        {
            adj[i] = adj[i].Distinct().ToList();
        }

        // Handle self-loops: must remove these vertices
        List<int> fvs = new List<int>();
        bool[] hasSelfLoop = new bool[n];
        for (int u = 0; u < n; u++)
        {
            if (adj[u].Contains(u))
            {
                hasSelfLoop[u] = true;
                fvs.Add(u);
            }
        }

        // Candidates = vertices without self-loops
        List<int> candidates = new List<int>();
        for (int i = 0; i < n; i++)
        {
            if (!hasSelfLoop[i])
                candidates.Add(i);
        }

        // If no candidates, we are done
        if (candidates.Count == 0)
        {
            OutputResult(fvs);
            return;
        }

        // Heuristic ordering: sort candidates by total degree (descending)
        int[] degree = new int[n];
        for (int u = 0; u < n; u++)
        {
            degree[u] = adj[u].Count + 0; // can add indegree if wanted
        }
        for (int u = 0; u < n; u++)
        {
            foreach (var v in adj[u])
                degree[v]++; // rough indegree contribution
        }

        candidates = candidates.OrderByDescending(x => degree[x]).ToList();

        removedGlobal = new bool[n];
        foreach (int v in fvs)
            removedGlobal[v] = true;

        // Try increasing additional removals
        List<int> bestAdditional = null;
        for (int addK = 0; addK <= candidates.Count; addK++)
        {
            List<int> current = new List<int>();
            if (TryFindSet(0, addK, current, candidates))
            {
                bestAdditional = new List<int>(current);
                break;
            }
        }

        if (bestAdditional != null)
        {
            fvs.AddRange(bestAdditional);
        }

        OutputResult(fvs);
    }

    static void OutputResult(List<int> fvs)
    {
        fvs = fvs.Distinct().OrderBy(x => x).ToList();
        Console.WriteLine(fvs.Count);
        if (fvs.Count > 0)
        {
            Console.WriteLine(string.Join(" ", fvs));
        }
        else
        {
            Console.WriteLine();
        }
    }

    static bool TryFindSet(int startIdx, int remain, List<int> current, List<int> candidates)
    {
        if (remain == 0)
        {
            return IsAcyclic(current);
        }

        for (int i = startIdx; i < candidates.Count; i++)
        {
            int v = candidates[i];
            current.Add(v);
            removedGlobal[v] = true;

            if (TryFindSet(i + 1, remain - 1, current, candidates))
                return true;

            removedGlobal[v] = false;
            current.RemoveAt(current.Count - 1);
        }
        return false;
    }

    static bool IsAcyclic(List<int> additionalRemoved)
    {
        // Temporarily mark additional removed
        List<int> tempMarked = new List<int>();
        foreach (int v in additionalRemoved)
        {
            if (!removedGlobal[v])
            {
                removedGlobal[v] = true;
                tempMarked.Add(v);
            }
        }

        bool result = HasTopoOrder();

        // Restore
        foreach (int v in tempMarked)
            removedGlobal[v] = false;

        return result;
    }

    static bool HasTopoOrder()
    {
        int[] indeg = new int[n];
        int remainingCount = 0;

        for (int u = 0; u < n; u++)
        {
            if (removedGlobal[u]) continue;
            remainingCount++;
            foreach (int v in adj[u])
            {
                if (!removedGlobal[v])
                    indeg[v]++;
            }
        }

        if (remainingCount == 0) return true;

        Queue<int> q = new Queue<int>();
        for (int u = 0; u < n; u++)
        {
            if (!removedGlobal[u] && indeg[u] == 0)
                q.Enqueue(u);
        }

        int processed = 0;
        while (q.Count > 0)
        {
            int u = q.Dequeue();
            processed++;
            foreach (int v in adj[u])
            {
                if (!removedGlobal[v])
                {
                    indeg[v]--;
                    if (indeg[v] == 0)
                        q.Enqueue(v);
                }
            }
        }

        return processed == remainingCount;
    }
}
```
