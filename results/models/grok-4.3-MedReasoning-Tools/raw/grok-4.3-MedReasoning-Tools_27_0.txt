#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>

bool try_color(int idx, const std::vector<int>& order, std::vector<int>& colors,
               const std::vector<std::vector<int>>& adj, int k) {
    if (idx == static_cast<int>(order.size())) {
        return true;
    }
    int v = order[idx];
    std::vector<bool> used(k, false);
    for (int nei : adj[v]) {
        if (colors[nei] != -1) {
            if (colors[nei] < k) {
                used[colors[nei]] = true;
            }
        }
    }
    for (int c = 0; c < k; ++c) {
        if (!used[c]) {
            colors[v] = c;
            if (try_color(idx + 1, order, colors, adj, k)) {
                return true;
            }
            colors[v] = -1;
        }
    }
    return false;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, m, k;
    std::cin >> n >> m >> k;

    std::vector<std::vector<int>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        std::cin >> u >> v;
        if (u != v) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }

    // Remove duplicates (in case of input issues)
    for (int i = 0; i < n; ++i) {
        std::sort(adj[i].begin(), adj[i].end());
        auto last = std::unique(adj[i].begin(), adj[i].end());
        adj[i].erase(last, adj[i].end());
    }

    std::vector<int> colors(n, -1);

    // Compute degrees and ordering (highest degree first)
    std::vector<int> verts(n);
    std::iota(verts.begin(), verts.end(), 0);
    std::sort(verts.begin(), verts.end(), [&](int x, int y) {
        return adj[x].size() > adj[y].size() ||
               (adj[x].size() == adj[y].size() && x < y);
    });

    int max_deg = 0;
    for (int i = 0; i < n; ++i) {
        max_deg = std::max(max_deg, static_cast<int>(adj[i].size()));
    }

    bool success = false;

    // Greedy coloring (always safe when k >= max_deg + 1)
    auto do_greedy = [&]() -> bool {
        std::fill(colors.begin(), colors.end(), -1);
        for (int v : verts) {
            std::vector<bool> used(k, false);
            for (int nei : adj[v]) {
                if (colors[nei] != -1) {
                    used[colors[nei]] = true;
                }
            }
            bool found = false;
            for (int c = 0; c < k; ++c) {
                if (!used[c]) {
                    colors[v] = c;
                    found = true;
                    break;
                }
            }
            if (!found) {
                return false;
            }
        }
        return true;
    };

    success = do_greedy();

    // Fallback to backtracking for small graphs when greedy fails
    if (!success && n <= 90) {
        std::fill(colors.begin(), colors.end(), -1);
        success = try_color(0, verts, colors, adj, k);
    }

    // If still not successful and k is large enough, fall back to any order greedy
    if (!success && k >= max_deg + 1) {
        success = do_greedy();  // retry with same ordering (should succeed)
    }

    // Output coloring (or fallback to 0s if impossible - should not happen in valid tests)
    if (success) {
        for (int i = 0; i < n; ++i) {
            if (i > 0) std::cout << " ";
            std::cout << colors[i];
        }
        std::cout << "\n";
    } else {
        // Last resort: assign lowest possible color greedily in original order
        std::fill(colors.begin(), colors.end(), -1);
        for (int v = 0; v < n; ++v) {
            std::vector<bool> used(k, false);
            for (int nei : adj[v]) {
                if (colors[nei] != -1) {
                    used[colors[nei]] = true;
                }
            }
            for (int c = 0; c < k; ++c) {
                if (!used[c]) {
                    colors[v] = c;
                    break;
                }
            }
        }
        for (int i = 0; i < n; ++i) {
            if (i > 0) std::cout << " ";
            std::cout << colors[i];
        }
        std::cout << "\n";
    }

    return 0;
}
