```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<vector<int>> adj(n);
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        if (u != v) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }

    for (int i = 0; i < n; i++) {
        sort(adj[i].begin(), adj[i].end());
        auto last = unique(adj[i].begin(), adj[i].end());
        adj[i].erase(last, adj[i].end());
    }

    if (n == 0) {
        cout << 0 << '\n' << '\n';
        return 0;
    }

    // Initial greedy solution for lower bound (low degree first)
    vector<pair<int, int>> deg_order(n);
    for (int i = 0; i < n; i++) {
        deg_order[i] = {(int)adj[i].size(), i};
    }
    sort(deg_order.begin(), deg_order.end());

    vector<int> greedy_set;
    vector<char> used(n, 0);
    for (auto& p : deg_order) {
        int v = p.second;
        if (used[v]) continue;
        greedy_set.push_back(v);
        used[v] = 1;
        for (int u : adj[v]) {
            used[u] = 1;
        }
    }

    vector<int> best_set = greedy_set;
    int best_size = (int)greedy_set.size();

    // Ordering: high degree first for branching decisions
    vector<pair<int, int>> branch_order(n);
    for (int i = 0; i < n; i++) {
        branch_order[i] = {-(int)adj[i].size(), i};
    }
    sort(branch_order.begin(), branch_order.end());

    vector<int> order(n);
    for (int i = 0; i < n; i++) order[i] = branch_order[i].second;

    vector<char> in_set(n, 0);
    vector<int> current;

    function<void(int)> dfs = [&](int pos) {
        if (current.size() + (n - pos) <= (size_t)best_size) return;

        if (pos == n) {
            if ((int)current.size() > best_size) {
                best_size = (int)current.size();
                best_set = current;
            }
            return;
        }

        int v = order[pos];

        // Try include first (to find good solutions quickly)
        bool can = true;
        for (int u : adj[v]) {
            if (in_set[u]) {
                can = false;
                break;
            }
        }
        if (can) {
            in_set[v] = 1;
            current.push_back(v);
            dfs(pos + 1);
            current.pop_back();
            in_set[v] = 0;
        }

        // Skip
        dfs(pos + 1);
    };

    dfs(0);

    // Output
    cout << best_size << '\n';
    for (size_t i = 0; i < best_set.size(); ++i) {
        if (i > 0) cout << ' ';
        cout << best_set[i];
    }
    cout << '\n';

    return 0;
}
```
