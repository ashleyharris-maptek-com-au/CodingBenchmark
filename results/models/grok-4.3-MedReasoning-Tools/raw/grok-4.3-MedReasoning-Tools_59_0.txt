```javascript
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);
const prob = data.probability;
const drones = data.drones || [];
const maxSteps = data.maxStepsPerDrone || 0;
const H = prob.length;
const W = H > 0 ? prob[0].length : 0;
const dirs = [[0, 0], [0, 1], [0, -1], [1, 0], [-1, 0]];
const covered = Array.from({ length: H }, () => Array(W).fill(false));
const paths = [];
const currents = [];
for (let d = 0; d < drones.length; d++) {
  const drone = drones[d];
  const sx = drone.x;
  const sy = drone.y;
  paths.push([[sx, sy]]);
  currents.push({ x: sx, y: sy });
  if (sx >= 0 && sx < W && sy >= 0 && sy < H && prob[sy][sx] !== -1) {
    covered[sy][sx] = true;
  }
}
for (let step = 0; step < maxSteps; step++) {
  for (let d = 0; d < drones.length; d++) {
    const curr = currents[d];
    let bestScore = -Infinity;
    let bestNext = null;
    for (let [dx, dy] of dirs) {
      const nx = curr.x + dx;
      const ny = curr.y + dy;
      if (nx >= 0 && nx < W && ny >= 0 && ny < H && prob[ny][nx] !== -1) {
        let score = prob[ny][nx];
        if (covered[ny][nx]) {
          score -= 10000;
        }
        if (score > bestScore) {
          bestScore = score;
          bestNext = { x: nx, y: ny };
        }
      }
    }
    if (bestNext) {
      paths[d].push([bestNext.x, bestNext.y]);
      covered[bestNext.y][bestNext.x] = true;
      currents[d] = bestNext;
    } else {
      paths[d].push([curr.x, curr.y]);
    }
  }
}
const output = { paths: paths };
console.log(JSON.stringify(output));
```
