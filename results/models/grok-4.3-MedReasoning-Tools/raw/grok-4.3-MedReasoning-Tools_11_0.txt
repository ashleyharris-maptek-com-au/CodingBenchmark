```rust
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read first line: SV NP
    let first_line = match lines.next() {
        Some(Ok(line)) => line,
        _ => {
            eprintln!("Failed to read first line");
            return;
        }
    };
    let mut parts = first_line.split_whitespace();
    let sv: usize = match parts.next().unwrap().parse() {
        Ok(v) => v,
        Err(_) => {
            eprintln!("Failed to parse SV");
            return;
        }
    };
    let np: usize = match parts.next().unwrap().parse() {
        Ok(v) => v,
        Err(_) => {
            eprintln!("Failed to parse NP");
            return;
        }
    };

    // Read stock polygon vertices
    let mut stock: Vec<(f64, f64)> = Vec::with_capacity(sv);
    for _ in 0..sv {
        let line = match lines.next() {
            Some(Ok(l)) => l,
            _ => {
                eprintln!("Failed to read stock vertex");
                return;
            }
        };
        let mut p = line.split_whitespace();
        let x: f64 = match p.next().unwrap().parse() {
            Ok(v) => v,
            Err(_) => {
                eprintln!("Failed to parse stock x");
                return;
            }
        };
        let y: f64 = match p.next().unwrap().parse() {
            Ok(v) => v,
            Err(_) => {
                eprintln!("Failed to parse stock y");
                return;
            }
        };
        stock.push((x, y));
    }

    // Read all pieces
    let mut pieces: Vec<Vec<(f64, f64)>> = Vec::with_capacity(np);
    for _ in 0..np {
        let pv_line = match lines.next() {
            Some(Ok(l)) => l,
            _ => {
                eprintln!("Failed to read piece vertex count");
                return;
            }
        };
        let pv: usize = match pv_line.trim().parse() {
            Ok(v) => v,
            Err(_) => {
                eprintln!("Failed to parse PV");
                return;
            }
        };

        let mut piece: Vec<(f64, f64)> = Vec::with_capacity(pv);
        for _ in 0..pv {
            let line = match lines.next() {
                Some(Ok(l)) => l,
                _ => {
                    eprintln!("Failed to read piece vertex");
                    return;
                }
            };
            let mut p = line.split_whitespace();
            let x: f64 = match p.next().unwrap().parse() {
                Ok(v) => v,
                Err(_) => {
                    eprintln!("Failed to parse piece x");
                    return;
                }
            };
            let y: f64 = match p.next().unwrap().parse() {
                Ok(v) => v,
                Err(_) => {
                    eprintln!("Failed to parse piece y");
                    return;
                }
            };
            piece.push((x, y));
        }
        pieces.push(piece);
    }

    // Simple but valid strategy for all scales:
    // - Use one stock per piece (guarantees no overlap)
    // - Align first vertex of piece with first vertex of stock
    // - No rotation (fast)
    // This is always valid if the instance allows placement (assumes feasible input)
    // For better quality on small instances one could add heuristics later.

    let num_stocks = np;
    println!("{}", num_stocks);

    for i in 0..np {
        if pieces[i].is_empty() {
            println!("{} 0.0 0.0 0.0", i);
            continue;
        }

        let (px, py) = pieces[i][0];
        let (sx, sy) = stock[0];
        let dx = sx - px;
        let dy = sy - py;
        let rotation = 0.0f64;

        println!("{} {} {} {}", i, dx, dy, rotation);
    }
}
```
