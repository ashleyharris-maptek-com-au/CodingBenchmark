```cpp
#include <iostream>
#include <vector>
#include <deque>
#include <set>
#include <map>
#include <queue>
#include <algorithm>
#include <string>

using namespace std;

using Pos = vector<int>;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int D, max_turns;
    cin >> D >> max_turns;

    vector<int> bounds(D);
    for (int i = 0; i < D; i++) {
        cin >> bounds[i];
    }

    int snake_len;
    Pos head(D);
    cin >> snake_len;
    for (int i = 0; i < D; i++) {
        cin >> head[i];
    }

    int F;
    cin >> F;
    set<Pos> foods;
    for (int i = 0; i < F; i++) {
        Pos p(D);
        for (int j = 0; j < D; j++) {
            cin >> p[j];
        }
        foods.insert(p);
    }

    int O;
    cin >> O;
    set<Pos> obstacles;
    for (int i = 0; i < O; i++) {
        Pos p(D);
        for (int j = 0; j < D; j++) {
            cin >> p[j];
        }
        obstacles.insert(p);
    }

    deque<Pos> snake;
    snake.push_front(head);
    set<Pos> body;
    body.insert(head);

    // If initial length > 1, we assume tests start with length 1 or body grows on eats
    // (full body positions not provided in input)

    Pos current_head = head;
    int turns = 0;

    auto in_bounds = [&](const Pos& p) -> bool {
        for (int d = 0; d < D; d++) {
            if (p[d] < 0 || p[d] >= bounds[d]) return false;
        }
        return true;
    };

    while (turns < max_turns && !foods.empty()) {
        // BFS from current_head to any food
        set<Pos> visited;
        map<Pos, Pos> came_from;
        queue<Pos> q;

        q.push(current_head);
        visited.insert(current_head);
        came_from[current_head] = current_head;

        Pos target_found = {};
        bool path_found = false;

        while (!q.empty() && !path_found) {
            Pos curr = q.front();
            q.pop();

            if (foods.count(curr) && curr != current_head) {
                target_found = curr;
                path_found = true;
                break;
            }

            for (int a = 0; a < D; a++) {
                for (int di : {-1, 1}) {
                    Pos next = curr;
                    next[a] += di;

                    if (!in_bounds(next)) continue;
                    if (obstacles.count(next)) continue;
                    if (body.count(next)) continue;
                    if (visited.count(next)) continue;

                    visited.insert(next);
                    q.push(next);
                    came_from[next] = curr;
                }
            }
        }

        Pos next_head;
        bool made_move = false;

        if (path_found) {
            // Reconstruct first step
            Pos step = target_found;
            while (came_from[step] != current_head) {
                step = came_from[step];
            }
            next_head = step;
            made_move = true;
        } else {
            // Fallback: any valid move (prefer toward a food heuristically)
            int best_dist = INT_MAX;
            Pos best_next = {};

            for (int a = 0; a < D; a++) {
                for (int di : {-1, 1}) {
                    Pos cand = current_head;
                    cand[a] += di;

                    if (!in_bounds(cand)) continue;
                    if (obstacles.count(cand)) continue;
                    if (body.count(cand)) continue;

                    // Manhattan distance to closest food
                    int min_md = INT_MAX;
                    for (const auto& f : foods) {
                        int md = 0;
                        for (int d = 0; d < D; d++) {
                            md += abs(f[d] - cand[d]);
                        }
                        if (md < min_md) min_md = md;
                    }
                    if (min_md < best_dist) {
                        best_dist = min_md;
                        best_next = cand;
                    }
                }
            }

            if (best_dist != INT_MAX) {
                next_head = best_next;
                made_move = true;
            }
        }

        if (!made_move) {
            // No safe move, try to stay alive as long as possible
            bool emergency_move = false;
            for (int a = 0; a < D && !emergency_move; a++) {
                for (int di : {-1, 1}) {
                    Pos cand = current_head;
                    cand[a] += di;
                    if (in_bounds(cand) && !obstacles.count(cand) && !body.count(cand)) {
                        next_head = cand;
                        emergency_move = true;
                        made_move = true;
                        break;
                    }
                }
            }
            if (!emergency_move) {
                break; // Stuck
            }
        }

        if (!made_move) break;

        // Output the move
        int axis = -1, dir = 0;
        for (int a = 0; a < D; a++) {
            if (next_head[a] != current_head[a]) {
                axis = a;
                dir = next_head[a] - current_head[a];
                break;
            }
        }
        if (axis == -1) break; // Should not happen

        cout << axis << " " << dir << endl;
        cout.flush();

        // Simulate move
        bool eating = foods.count(next_head) > 0;
        if (eating) {
            foods.erase(next_head);
        }

        snake.push_front(next_head);
        body.insert(next_head);

        if (!eating) {
            if (!snake.empty()) {
                Pos tail = snake.back();
                snake.pop_back();
                body.erase(tail);
            }
        }

        current_head = next_head;
        turns++;
    }

    return 0;
}
```
