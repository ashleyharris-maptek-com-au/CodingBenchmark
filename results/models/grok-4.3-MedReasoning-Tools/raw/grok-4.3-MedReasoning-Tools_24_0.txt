#include <iostream>
#include <fstream>
#include <string>
#include <vector>
#include <thread>
#include <atomic>
#include <cmath>
#include <algorithm>
#include <chrono>
#include <cstdint>
#include <limits>

using namespace std;

static constexpr double PI = 3.14159265358979323846;

double normalize_angle(double a) {
    a = fmod(a + PI, 2.0 * PI);
    if (a < 0.0) a += 2.0 * PI;
    return a - PI;
}

void parse_terrain(const string& path, double width, double depth) {
    ifstream file(path, ios::binary);
    if (!file.is_open()) {
        return;
    }
    uint64_t num_cells = static_cast<uint64_t>(width) * static_cast<uint64_t>(depth);
    uint64_t total_bytes = num_cells * 6ULL;
    
    // Read in large chunks for speed on huge files
    constexpr size_t CHUNK = 1024 * 1024;
    vector<char> buffer(CHUNK);
    
    uint64_t bytes_read = 0;
    while (bytes_read < total_bytes) {
        size_t to_read = min(CHUNK, static_cast<size_t>(total_bytes - bytes_read));
        file.read(buffer.data(), to_read);
        if (file.gcount() != static_cast<streamsize>(to_read)) {
            break;
        }
        bytes_read += to_read;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.setf(ios::fixed);
    cout.precision(6);
    
    double width, depth, height, gravity, max_thrust, initial_fuel;
    cin >> width >> depth >> height >> gravity >> max_thrust >> initial_fuel;
    
    double start_x, start_y, start_z, target_x, target_y, target_z;
    cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z;
    
    string terrain_path;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');
    getline(cin, terrain_path);
    
    // Trim whitespace from path if any
    terrain_path.erase(0, terrain_path.find_first_not_of(" \t\r\n"));
    terrain_path.erase(terrain_path.find_last_not_of(" \t\r\n") + 1);
    
    atomic<bool> terrain_done{false};
    thread terrain_thread([&]() {
        parse_terrain(terrain_path, width, depth);
        terrain_done = true;
    });
    
    // Give terrain thread a head start but don't block forever
    this_thread::sleep_for(chrono::milliseconds(50));
    
    double prev_time = 0.0;
    double last_x = 0, last_y = 0, last_z = 0;
    bool first_state = true;
    
    double target_altitude = 25.0; // meters above pad for approach
    
    while (true) {
        double x, y, z, vx, vy, vz;
        double pitch, yaw, roll, pitch_rate, yaw_rate, roll_rate, fuel;
        
        if (!(cin >> x >> y >> z >> vx >> vy >> vz >> 
              pitch >> yaw >> roll >> pitch_rate >> yaw_rate >> roll_rate >> fuel)) {
            break;
        }
        
        // Compute errors
        double dx = target_x - x;
        double dy = target_y - y;
        double dz = target_z + target_altitude - z;
        
        double dist_h = sqrt(dx * dx + dy * dy);
        double dist_3d = sqrt(dist_h * dist_h + dz * dz);
        
        // Phase detection
        bool landing_phase = (dist_h < 8.0 && z < target_z + 12.0);
        bool final_descent = (dist_h < 3.0 && z < target_z + 4.0);
        
        // Desired accelerations (world frame)
        double Kp_h = 0.08;
        double Kd_h = 0.6;
        double ax = Kp_h * dx - Kd_h * vx;
        double ay = Kp_h * dy - Kd_h * vy;
        
        double az;
        if (final_descent) {
            // Very gentle final touchdown
            az = 1.62 - 3.5 * (vz + 0.3);
        } else if (landing_phase) {
            az = 1.62 - 2.2 * (vz + 0.8);
        } else {
            // Approach / hover above target
            double vz_des = (dist_h > 30.0) ? -0.3 : -1.2;
            az = 1.62 - 1.8 * (vz - vz_des);
        }
        
        // Limit horizontal acceleration request
        double max_h_accel = 6.5;
        double h_mag = sqrt(ax * ax + ay * ay);
        if (h_mag > max_h_accel) {
            ax = ax * max_h_accel / h_mag;
            ay = ay * max_h_accel / h_mag;
        }
        
        // Total desired acceleration
        double total_ax = ax;
        double total_ay = ay;
        double total_az = az;
        
        double accel_mag = sqrt(total_ax * total_ax + total_ay * total_ay + total_az * total_az);
        
        // Thrust direction
        double Dx = total_ax;
        double Dy = total_ay;
        double Dz = total_az;
        
        if (accel_mag < 0.01) {
            Dx = 0.0;
            Dy = 0.0;
            Dz = 1.0;
        } else {
            Dx /= accel_mag;
            Dy /= accel_mag;
            Dz /= accel_mag;
        }
        
        // Convert direction to pitch/yaw
        double horiz = sqrt(Dx * Dx + Dy * Dy);
        double pitch_des = 0.0;
        if (horiz > 1e-6) {
            pitch_des = atan2(horiz, max(Dz, 0.05));
        }
        pitch_des = clamp(pitch_des, -PI / 2.0 + 0.05, PI / 2.0 - 0.05);
        
        double yaw_des = atan2(Dy, Dx) + PI;  // offset for +pitch -> -X convention
        yaw_des = normalize_angle(yaw_des);
        
        double roll_des = 0.0;
        
        // Angle errors
        double pitch_err = normalize_angle(pitch_des - pitch);
        double yaw_err = normalize_angle(yaw_des - yaw);
        double roll_err = normalize_angle(roll_des - roll);
        
        // Angular commands (angular accel command)
        double ang_kp = 3.8;
        double ang_kd = 1.4;
        
        double pitch_cmd = ang_kp * pitch_err - ang_kd * pitch_rate;
        double yaw_cmd = ang_kp * yaw_err - ang_kd * yaw_rate;
        double roll_cmd = ang_kp * roll_err - ang_kd * roll_rate * 0.6;
        
        pitch_cmd = clamp(pitch_cmd, -1.0, 1.0);
        yaw_cmd = clamp(yaw_cmd, -1.0, 1.0);
        roll_cmd = clamp(roll_cmd, -1.0, 1.0);
        
        // Thrust magnitude
        double thrust = 0.0;
        
        if (final_descent) {
            thrust = clamp(0.22 + 0.45 * (1.0 - fabs(vz + 0.4)), 0.0, 0.85);
        } else if (landing_phase) {
            thrust = clamp(0.35 + 0.55 * (1.0 - fabs(vz + 0.9)), 0.0, 0.95);
        } else {
            // Normal flight
            thrust = clamp(accel_mag / 9.5, 0.15, 1.0);
            
            // Reduce thrust when very high and far
            if (dist_h > 150.0 && z > target_z + 80.0) {
                thrust *= 0.7;
            }
        }
        
        // Fuel conservation
        if (fuel < 8.0) {
            thrust = min(thrust, 0.45);
        }
        
        // Emergency hover if fuel critical
        if (fuel < 3.0 && z < target_z + 30.0) {
            thrust = min(thrust, 0.25);
        }
        
        // Output command
        cout << thrust << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n";
        cout.flush();
        
        first_state = false;
    }
    
    if (terrain_thread.joinable()) {
        terrain_thread.join();
    }
    
    return 0;
}
