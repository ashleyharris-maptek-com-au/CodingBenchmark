```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

class Program
{
    static int n;
    static HashSet<int>[] neighbors;
    static int bestSize = 0;
    static List<int> bestClique = new List<int>();
    static object lockObj = new object();
    static CancellationTokenSource cts = new CancellationTokenSource();
    static DateTime startTime;

    static void Main()
    {
        startTime = DateTime.Now;

        string[] firstLine = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        n = int.Parse(firstLine[0]);
        int m = int.Parse(firstLine[1]);

        neighbors = new HashSet<int>[n];
        for (int i = 0; i < n; i++)
            neighbors[i] = new HashSet<int>();

        for (int i = 0; i < m; i++)
        {
            string[] parts = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length < 2) continue;
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            if (u != v && u >= 0 && u < n && v >= 0 && v < n)
            {
                neighbors[u].Add(v);
                neighbors[v].Add(u);
            }
        }

        // Always run fast greedy heuristic (good lower bound quickly)
        RunGreedyHeuristic();

        // For small graphs, run exact solver with timeout
        if (n <= 80)
        {
            // Run exact in background with time limit
            Task.Run(() => RunExactWithTimeout(240000)); // ~4 minutes
        }
        else if (n <= 300)
        {
            // Medium graphs: try limited backtracking
            Task.Run(() => RunLimitedBacktrack(120000));
        }

        // Wait a bit for improvements, then output best found
        Thread.Sleep(100); // Allow some parallel work to start

        // Ensure we have at least 1 vertex if graph empty
        if (bestSize == 0 && n > 0)
        {
            bestSize = 1;
            bestClique = new List<int> { 0 };
        }

        // Output
        Console.WriteLine(bestSize);
        if (bestClique.Count > 0)
        {
            var sorted = bestClique.OrderBy(x => x).ToList();
            Console.WriteLine(string.Join(" ", sorted));
        }
        else
        {
            Console.WriteLine();
        }
    }

    static void RunGreedyHeuristic()
    {
        var order = Enumerable.Range(0, n)
            .OrderByDescending(v => neighbors[v].Count)
            .ToList();

        List<int> clique = new List<int>(Math.Min(100, n / 2));

        foreach (int v in order)
        {
            bool canAdd = true;
            foreach (int u in clique)
            {
                if (!neighbors[u].Contains(v))
                {
                    canAdd = false;
                    break;
                }
            }
            if (canAdd)
            {
                clique.Add(v);
                if (clique.Count > bestSize)
                {
                    lock (lockObj)
                    {
                        if (clique.Count > bestSize)
                        {
                            bestSize = clique.Count;
                            bestClique = new List<int>(clique);
                        }
                    }
                }
            }
        }

        // Try reverse degree order as well
        order = Enumerable.Range(0, n)
            .OrderBy(v => neighbors[v].Count)
            .ToList();

        clique.Clear();
        foreach (int v in order)
        {
            bool canAdd = true;
            foreach (int u in clique)
            {
                if (!neighbors[u].Contains(v))
                {
                    canAdd = false;
                    break;
                }
            }
            if (canAdd)
            {
                clique.Add(v);
                if (clique.Count > bestSize)
                {
                    lock (lockObj)
                    {
                        if (clique.Count > bestSize)
                        {
                            bestSize = clique.Count;
                            bestClique = new List<int>(clique);
                        }
                    }
                }
            }
        }
    }

    static void RunExactWithTimeout(int timeoutMs)
    {
        try
        {
            var token = cts.Token;
            var R = new HashSet<int>();
            var P = new HashSet<int>(Enumerable.Range(0, n));
            var X = new HashSet<int>();

            BronKerbosch(R, P, X, token, timeoutMs);
        }
        catch { }
    }

    static void BronKerbosch(HashSet<int> R, HashSet<int> P, HashSet<int> X, CancellationToken token, int timeoutMs)
    {
        if (token.IsCancellationRequested) return;

        if ((DateTime.Now - startTime).TotalMilliseconds > timeoutMs)
        {
            cts.Cancel();
            return;
        }

        if (P.Count == 0 && X.Count == 0)
        {
            if (R.Count > bestSize)
            {
                lock (lockObj)
                {
                    if (R.Count > bestSize)
                    {
                        bestSize = R.Count;
                        bestClique = new List<int>(R);
                    }
                }
            }
            return;
        }

        if (R.Count + P.Count <= bestSize) return;

        // Pivoting
        HashSet<int> union = new HashSet<int>(P);
        union.UnionWith(X);
        if (union.Count == 0) return;

        int pivot = union.First();
        var candidates = P.Except(neighbors[pivot]).ToList();

        foreach (int v in candidates)
        {
            if (token.IsCancellationRequested) return;
            if ((DateTime.Now - startTime).TotalMilliseconds > timeoutMs) return;

            var newR = new HashSet<int>(R) { v };
            var newP = new HashSet<int>(P.Intersect(neighbors[v]));
            var newX = new HashSet<int>(X.Intersect(neighbors[v]));

            BronKerbosch(newR, newP, newX, token, timeoutMs);

            P.Remove(v);
            X.Add(v);
        }
    }

    static void RunLimitedBacktrack(int timeoutMs)
    {
        try
        {
            var order = Enumerable.Range(0, n)
                .OrderByDescending(v => neighbors[v].Count)
                .ToList();

            Backtrack(new List<int>(), 0, order, timeoutMs);
        }
        catch { }
    }

    static void Backtrack(List<int> current, int startIdx, List<int> order, int timeoutMs)
    {
        if ((DateTime.Now - startTime).TotalMilliseconds > timeoutMs) return;

        if (current.Count > bestSize)
        {
            lock (lockObj)
            {
                if (current.Count > bestSize)
                {
                    bestSize = current.Count;
                    bestClique = new List<int>(current);
                }
            }
        }

        if (current.Count + (order.Count - startIdx) <= bestSize) return;

        for (int i = startIdx; i < order.Count; i++)
        {
            if ((DateTime.Now - startTime).TotalMilliseconds > timeoutMs) return;

            int v = order[i];
            bool canAdd = true;
            foreach (int u in current)
            {
                if (!neighbors[u].Contains(v))
                {
                    canAdd = false;
                    break;
                }
            }

            if (canAdd)
            {
                current.Add(v);
                Backtrack(current, i + 1, order, timeoutMs);
                current.RemoveAt(current.Count - 1);
            }
        }
    }
}
```
