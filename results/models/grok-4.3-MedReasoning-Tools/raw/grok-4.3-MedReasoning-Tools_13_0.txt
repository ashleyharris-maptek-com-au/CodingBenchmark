#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <limits>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    std::string line;
    if (!std::getline(std::cin, line)) {
        std::cout << std::endl;
        return 0;
    }
    int N = 0;
    try {
        N = std::stoi(line);
    } catch (...) {
        std::cout << std::endl;
        return 0;
    }

    std::vector<std::string> strs(N);
    size_t min_len = std::numeric_limits<size_t>::max();
    for (int i = 0; i < N; ++i) {
        if (!std::getline(std::cin, strs[i])) {
            strs[i] = "";
        }
        if (strs[i].size() < min_len) {
            min_len = strs[i].size();
        }
    }

    if (N == 0 || min_len == 0) {
        std::cout << std::endl;
        return 0;
    }

    // Move shortest string to index 0 for efficiency
    size_t shortest_idx = 0;
    for (size_t i = 1; i < static_cast<size_t>(N); ++i) {
        if (strs[i].size() < strs[shortest_idx].size()) {
            shortest_idx = i;
        }
    }
    if (shortest_idx != 0) {
        std::swap(strs[0], strs[shortest_idx]);
    }

    auto find_common = [&](int L, std::string& res) -> bool {
        if (L <= 0) {
            res = "";
            return true;
        }
        const std::string& base = strs[0];
        if (base.size() < static_cast<size_t>(L)) return false;
        for (size_t i = 0; i + static_cast<size_t>(L) <= base.size(); ++i) {
            bool common = true;
            for (size_t j = 1; j < static_cast<size_t>(N); ++j) {
                if (strs[j].find(base.data() + i, 0, static_cast<size_t>(L)) == std::string::npos) {
                    common = false;
                    break;
                }
            }
            if (common) {
                res = base.substr(i, static_cast<size_t>(L));
                return true;
            }
        }
        return false;
    };

    std::string best = "";
    int low = 0;
    int high = static_cast<int>(min_len);
    while (low <= high) {
        int mid = low + (high - low) / 2;
        std::string cand;
        if (find_common(mid, cand)) {
            best = std::move(cand);
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }

    std::cout << best << std::endl;
    return 0;
}
