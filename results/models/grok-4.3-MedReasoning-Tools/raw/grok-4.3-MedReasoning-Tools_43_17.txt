```hlsl
[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    float2 scale = float2(8.0, 16.0);
    float2 uv = input.uv * scale;

    float row = floor(uv.y);
    float stagger = fmod(row, 2.0) * 0.5;
    uv.x += stagger;

    float2 f = frac(uv);

    float mx = 0.05;
    float my = 0.1;

    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor = float3(0.7, 0.2, 0.1);

    float3 baseColor;
    if (f.x < mx || f.x > 1.0 - mx || f.y < my || f.y > 1.0 - my) {
        baseColor = mortarColor;
    } else {
        baseColor = brickColor;
    }

    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(0.0, dot(N, L));

    float3 shaded = baseColor * (0.4 + 0.6 * diffuse);

    return float4(shaded, 1.0);
}
```
