```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Types
%void = OpTypeVoid
%fn_main = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

; Constants
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_1 = OpConstant %float 1
%float_thick = OpConstant %float 0.025
%float_nthick = OpConstant %float -0.025

; Input gl_PerVertex block (array of 4)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_glpv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_glpv = OpTypePointer Input %arr_glpv_in
%gl_in = OpVariable %ptr_in_arr_glpv Input

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_glpv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_glpv Output

; Input color array (location 2)
%arr_color_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_color = OpTypePointer Input %arr_color_in
%in_color = OpVariable %ptr_in_arr_color Input
OpDecorate %in_color Location 2

; Output color (location 0)
%ptr_out_color = OpTypePointer Output %v3float
%out_color = OpVariable %ptr_out_color Output
OpDecorate %out_color Location 0

; Pointer types
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %fn_main
%entry = OpLabel

; Load 4 input positions (gl_in[i].gl_Position)
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %p2_ptr
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%pos3 = OpLoad %v4float %p3_ptr

; Extract XY components for 2D tangent computation
%pos0_xy = OpVectorShuffle %v2float %pos0 %pos0 0 1
%pos1_xy = OpVectorShuffle %v2float %pos1 %pos1 0 1
%pos2_xy = OpVectorShuffle %v2float %pos2 %pos2 0 1
%pos3_xy = OpVectorShuffle %v2float %pos3 %pos3 0 1

; t0 = normalize(pos2_xy - pos0_xy)
%diff_t0 = OpFSub %v2float %pos2_xy %pos0_xy
%len_t0 = OpExtInst %float %1 Length %diff_t0
%inv_len_t0 = OpFDiv %float %float_1 %len_t0
%t0 = OpVectorTimesScalar %v2float %diff_t0 %inv_len_t0

; t1 = normalize(pos3_xy - pos1_xy)
%diff_t1 = OpFSub %v2float %pos3_xy %pos1_xy
%len_t1 = OpExtInst %float %1 Length %diff_t1
%inv_len_t1 = OpFDiv %float %float_1 %len_t1
%t1 = OpVectorTimesScalar %v2float %diff_t1 %inv_len_t1

; perp0 = vec2(-t0.y, t0.x)
%t0x = OpCompositeExtract %float %t0 0
%t0y = OpCompositeExtract %float %t0 1
%neg_t0y = OpFNegate %float %t0y
%perp0 = OpCompositeConstruct %v2float %neg_t0y %t0x

; perp1 = vec2(-t1.y, t1.x)
%t1x = OpCompositeExtract %float %t1 0
%t1y = OpCompositeExtract %float %t1 1
%neg_t1y = OpFNegate %float %t1y
%perp1 = OpCompositeConstruct %v2float %neg_t1y %t1x

; Offsets
%off0p = OpVectorTimesScalar %v2float %perp0 %float_thick
%off0m = OpVectorTimesScalar %v2float %perp0 %float_nthick
%off1p = OpVectorTimesScalar %v2float %perp1 %float_thick
%off1m = OpVectorTimesScalar %v2float %perp1 %float_nthick

; Start and end XY offsets
%start_p = OpFAdd %v2float %pos1_xy %off0p
%start_m = OpFAdd %v2float %pos1_xy %off0m
%end_p   = OpFAdd %v2float %pos2_xy %off1p
%end_m   = OpFAdd %v2float %pos2_xy %off1m

; Reconstruct full vec4 positions (preserve ZW)
%zw1 = OpVectorShuffle %v2float %pos1 %pos1 2 3
%zw2 = OpVectorShuffle %v2float %pos2 %pos2 2 3

%v4_0_x = OpCompositeExtract %float %start_p 0
%v4_0_y = OpCompositeExtract %float %start_p 1
%v4_0_z = OpCompositeExtract %float %zw1 0
%v4_0_w = OpCompositeExtract %float %zw1 1
%v4_0 = OpCompositeConstruct %v4float %v4_0_x %v4_0_y %v4_0_z %v4_0_w

%v4_1_x = OpCompositeExtract %float %start_m 0
%v4_1_y = OpCompositeExtract %float %start_m 1
%v4_1_z = OpCompositeExtract %float %zw1 0
%v4_1_w = OpCompositeExtract %float %zw1 1
%v4_1 = OpCompositeConstruct %v4float %v4_1_x %v4_1_y %v4_1_z %v4_1_w

%v4_2_x = OpCompositeExtract %float %end_p 0
%v4_2_y = OpCompositeExtract %float %end_p 1
%v4_2_z = OpCompositeExtract %float %zw2 0
%v4_2_w = OpCompositeExtract %float %zw2 1
%v4_2 = OpCompositeConstruct %v4float %v4_2_x %v4_2_y %v4_2_z %v4_2_w

%v4_3_x = OpCompositeExtract %float %end_m 0
%v4_3_y = OpCompositeExtract %float %end_m 1
%v4_3_z = OpCompositeExtract %float %zw2 0
%v4_3_w = OpCompositeExtract %float %zw2 1
%v4_3 = OpCompositeConstruct %v4float %v4_3_x %v4_3_y %v4_3_z %v4_3_w

; Load colors
%c1_ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
%c1 = OpLoad %v3float %c1_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %in_color %int_2
%c2 = OpLoad %v3float %c2_ptr

; Emit vertex 0: start + perp0
%pos_out_ptr0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_out_ptr0 %v4_0
OpStore %out_color %c1
OpEmitVertex

; Emit vertex 1: start - perp0
%pos_out_ptr1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_out_ptr1 %v4_1
OpStore %out_color %c1
OpEmitVertex

; Emit vertex 2: end + perp1
%pos_out_ptr2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_out_ptr2 %v4_2
OpStore %out_color %c2
OpEmitVertex

; Emit vertex 3: end - perp1
%pos_out_ptr3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_out_ptr3 %v4_3
OpStore %out_color %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
