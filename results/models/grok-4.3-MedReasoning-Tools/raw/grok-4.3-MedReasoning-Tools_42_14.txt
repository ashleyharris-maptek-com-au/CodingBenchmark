```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 200
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

; Types
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%void = OpTypeVoid
%func = OpTypeFunction %void

; gl_PerVertex input struct (size 6 for adjacency)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; gl_PerVertex output
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; User input arrays
%arr_v3_6 = OpTypeArray %v3float %uint_6
%ptr_in_v3_arr = OpTypePointer Input %arr_v3_6
%in_worldPos = OpVariable %ptr_in_v3_arr Input
%in_normal = OpVariable %ptr_in_v3_arr Input
%in_color = OpVariable %ptr_in_v3_arr Input

; Output color
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Pointer types
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; Constants
%uint_6 = OpConstant %uint 6
%uint_0 = OpConstant %uint 0
%uint_2 = OpConstant %uint 2
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_1_3 = OpConstant %float 0.333333333

; GLSL.std.450
%glsl450 = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %func
%entry = OpLabel

; Load positions (p0, p2, p4)
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_0 %int_0
%p0v4 = OpLoad %v4float %p0_ptr
%p0 = OpVectorShuffle %v3float %p0v4 %p0v4 0 1 2

%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_2 %int_0
%p2v4 = OpLoad %v4float %p2_ptr
%p2 = OpVectorShuffle %v3float %p2v4 %p2v4 0 1 2

%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_4 %int_0
%p4v4 = OpLoad %v4float %p4_ptr
%p4 = OpVectorShuffle %v3float %p4v4 %p4v4 0 1 2

; Edge vectors
%e1 = OpFSub %v3float %p2 %p0
%e2 = OpFSub %v3float %p4 %p0

; Manual cross product
%e1x = OpCompositeExtract %float %e1 0
%e1y = OpCompositeExtract %float %e1 1
%e1z = OpCompositeExtract %float %e1 2
%e2x = OpCompositeExtract %float %e2 0
%e2y = OpCompositeExtract %float %e2 1
%e2z = OpCompositeExtract %float %e2 2

%cx1 = OpFMul %float %e1y %e2z
%cx2 = OpFMul %float %e1z %e2y
%cx  = OpFSub %float %cx1 %cx2

%cy1 = OpFMul %float %e1z %e2x
%cy2 = OpFMul %float %e1x %e2z
%cy  = OpFSub %float %cy1 %cy2

%cz1 = OpFMul %float %e1x %e2y
%cz2 = OpFMul %float %e1y %e2x
%cz  = OpFSub %float %cz1 %cz2

%N = OpCompositeConstruct %v3float %cx %cy %cz

; Normalize face normal
%N_norm = OpExtInst %v3float %glsl450 69 %N

; Light direction
%L = OpConstantComposite %v3float %float_1 %float_1 %float_1
%L_norm = OpExtInst %v3float %glsl450 69 %L

; Diffuse
%dotNL = OpDot %float %N_norm %L_norm
%diffuse = OpExtInst %float %glsl450 40 %dotNL %float_0

; Load colors[0,2,4]
%c0_ptr = OpAccessChain %ptr_in_v3 %in_color %uint_0
%c0 = OpLoad %v3float %c0_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %in_color %uint_2
%c2 = OpLoad %v3float %c2_ptr
%c4_ptr = OpAccessChain %ptr_in_v3 %in_color %uint_4
%c4 = OpLoad %v3float %c4_ptr

; Average base color
%sum12 = OpFAdd %v3float %c0 %c2
%sum   = OpFAdd %v3float %sum12 %c4
%base_color = OpVectorTimesScalar %v3float %sum %float_1_3

; Final shading factor
%diff_scaled = OpFMul %float %diffuse %float_0_8
%factor = OpFAdd %float %float_0_2 %diff_scaled
%final_color = OpVectorTimesScalar %v3float %base_color %factor

; Emit vertex 0
%out_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos0 %p0v4
OpStore %outColor %final_color
OpEmitVertex

; Emit vertex 2
%out_pos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos2 %p2v4
OpStore %outColor %final_color
OpEmitVertex

; Emit vertex 4
%out_pos4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos4 %p4v4
OpStore %outColor %final_color
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
