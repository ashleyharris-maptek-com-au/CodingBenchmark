```rust
use std::collections::HashMap;
use std::io::{self, BufRead, Write};

#[derive(Clone)]
struct Sample {
    depth: f64,
    values: Vec<f64>,
}

#[derive(Clone)]
struct Hole {
    id: String,
    start: [f64; 3],
    dir: [f64; 3],
    length: f64,
    samples: Vec<Sample>,
}

#[derive(Clone)]
struct SampleData {
    hole_id: String,
    idx: usize,
    pos: [f64; 3],
    values: Vec<f64>,
}

fn normalize(v: [f64; 3]) -> [f64; 3] {
    let norm = (v[0] * v[0] + v[1] * v[1] + v[2] * v[2]).sqrt();
    if norm < 1e-12 {
        return [0.0, 0.0, 0.0];
    }
    [v[0] / norm, v[1] / norm, v[2] / norm]
}

fn compute_position(start: [f64; 3], dir_unit: [f64; 3], depth: f64) -> [f64; 3] {
    [
        start[0] + dir_unit[0] * depth,
        start[1] + dir_unit[1] * depth,
        start[2] + dir_unit[2] * depth,
    ]
}

fn mean_and_std(values: &[f64]) -> (f64, f64) {
    if values.is_empty() {
        return (0.0, 1e-9);
    }
    let n = values.len() as f64;
    let sum: f64 = values.iter().sum();
    let mean = sum / n;
    let variance: f64 = values.iter().map(|&x| (x - mean).powi(2)).sum::<f64>() / n.max(1.0);
    let std = variance.sqrt().max(1e-9);
    (mean, std)
}

fn main() {
    let stdin = io::stdin();
    let reader = stdin.lock();
    let mut lines = reader.lines();

    // Parse header
    let first_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => {
            println!("0");
            return;
        }
    };

    let header: Vec<&str> = first_line.trim().split_whitespace().collect();
    if header.len() < 2 {
        println!("0");
        return;
    }

    let num_holes: usize = header[0].parse().unwrap_or(0);
    let num_props: usize = header[1].parse().unwrap_or(0);

    if num_props == 0 || num_holes == 0 {
        println!("0");
        return;
    }

    let prop_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => {
            println!("0");
            return;
        }
    };
    let property_names: Vec<String> = prop_line
        .trim()
        .split_whitespace()
        .take(num_props)
        .map(|s| s.to_string())
        .collect();

    if property_names.len() != num_props {
        println!("0");
        return;
    }

    let mut holes: Vec<Hole> = Vec::with_capacity(num_holes);
    let mut all_samples: Vec<SampleData> = Vec::new();

    for _ in 0..num_holes {
        let hole_line = match lines.next() {
            Some(Ok(l)) => l,
            _ => break,
        };

        let parts: Vec<&str> = hole_line.trim().split_whitespace().collect();
        if parts.len() < 9 {
            continue;
        }

        let hole_id = parts[0].to_string();
        let start_x: f64 = parts[1].parse().unwrap_or(0.0);
        let start_y: f64 = parts[2].parse().unwrap_or(0.0);
        let start_z: f64 = parts[3].parse().unwrap_or(0.0);
        let dir_x: f64 = parts[4].parse().unwrap_or(0.0);
        let dir_y: f64 = parts[5].parse().unwrap_or(0.0);
        let dir_z: f64 = parts[6].parse().unwrap_or(0.0);
        let length: f64 = parts[7].parse().unwrap_or(0.0);
        let num_samples: usize = parts[8].parse().unwrap_or(0);

        let mut samples = Vec::with_capacity(num_samples);

        for _ in 0..num_samples {
            let sample_line = match lines.next() {
                Some(Ok(l)) => l,
                _ => break,
            };
            let s_parts: Vec<&str> = sample_line.trim().split_whitespace().collect();
            if s_parts.len() < num_props + 1 {
                continue;
            }

            let depth: f64 = s_parts[0].parse().unwrap_or(0.0);
            let mut values = Vec::with_capacity(num_props);
            for i in 1..=num_props {
                let v: f64 = s_parts[i].parse().unwrap_or(0.0);
                values.push(v);
            }

            samples.push(Sample { depth, values });
        }

        let hole = Hole {
            id: hole_id,
            start: [start_x, start_y, start_z],
            dir: [dir_x, dir_y, dir_z],
            length,
            samples,
        };

        holes.push(hole);
    }

    // Compute positions and build all_samples
    let dir_units: Vec<[f64; 3]> = holes
        .iter()
        .map(|h| normalize(h.dir))
        .collect();

    for (h_idx, hole) in holes.iter().enumerate() {
        let dir_unit = dir_units[h_idx];
        for (s_idx, sample) in hole.samples.iter().enumerate() {
            let pos = compute_position(hole.start, dir_unit, sample.depth);
            all_samples.push(SampleData {
                hole_id: hole.id.clone(),
                idx: s_idx,
                pos,
                values: sample.values.clone(),
            });
        }
    }

    let total_samples = all_samples.len();
    if total_samples == 0 {
        println!("0");
        return;
    }

    // Adaptive strategy selection
    let use_brute = total_samples <= 4000;
    let cell_size = if total_samples < 5000 {
        25.0
    } else if total_samples < 20000 {
        60.0
    } else if total_samples < 100000 {
        150.0
    } else {
        300.0
    };

    let mut suspects: Vec<(f64, String, usize, String)> = Vec::new();

    if use_brute {
        // Brute-force KNN for small datasets
        for (i, sample) in all_samples.iter().enumerate() {
            let mut neighbors: Vec<(f64, Vec<f64>)> = Vec::new();
            for (j, other) in all_samples.iter().enumerate() {
                if i == j {
                    continue;
                }
                let dx = sample.pos[0] - other.pos[0];
                let dy = sample.pos[1] - other.pos[1];
                let dz = sample.pos[2] - other.pos[2];
                let dist = (dx * dx + dy * dy + dz * dz).sqrt();
                if dist < 500.0 {
                    neighbors.push((dist, other.values.clone()));
                }
            }

            if neighbors.len() < 3 {
                continue;
            }

            // Take up to 30 closest
            neighbors.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap());
            let k = neighbors.len().min(30);
            let neighbor_values: Vec<Vec<f64>> = neighbors.iter().take(k).map(|n| n.1.clone()).collect();

            for p in 0..num_props {
                let mut local_vals: Vec<f64> = neighbor_values.iter().map(|v| v[p]).collect();
                let (mean, std) = mean_and_std(&local_vals);
                let val = sample.values[p];
                let anomaly = ((val - mean) / std).abs();

                if anomaly > 2.8 {
                    let conf = anomaly.min(12.0);
                    suspects.push((conf, sample.hole_id.clone(), sample.idx, property_names[p].clone()));
                }
            }
        }
    } else {
        // Spatial binning for large datasets
        let mut grid: HashMap<(i64, i64, i64), Vec<usize>> = HashMap::new();

        for (idx, sample) in all_samples.iter().enumerate() {
            let key = (
                (sample.pos[0] / cell_size).floor() as i64,
                (sample.pos[1] / cell_size).floor() as i64,
                (sample.pos[2] / cell_size).floor() as i64,
            );
            grid.entry(key).or_default().push(idx);
        }

        let neighbor_offsets: Vec<(i64, i64, i64)> = (-1..=1)
            .flat_map(|dx| (-1..=1).flat_map(move |dy| (-1..=1).map(move |dz| (dx, dy, dz))))
            .collect();

        for (i, sample) in all_samples.iter().enumerate() {
            let key = (
                (sample.pos[0] / cell_size).floor() as i64,
                (sample.pos[1] / cell_size).floor() as i64,
                (sample.pos[2] / cell_size).floor() as i64,
            );

            let mut local_indices: Vec<usize> = Vec::new();
            for &(dx, dy, dz) in &neighbor_offsets {
                let nkey = (key.0 + dx, key.1 + dy, key.2 + dz);
                if let Some(indices) = grid.get(&nkey) {
                    local_indices.extend(indices);
                }
            }

            // Remove self and limit size for performance
            local_indices.retain(|&j| j != i);
            if local_indices.len() > 250 {
                local_indices.truncate(250);
            }

            if local_indices.len() < 4 {
                continue;
            }

            for p in 0..num_props {
                let mut local_vals: Vec<f64> = local_indices
                    .iter()
                    .map(|&j| all_samples[j].values[p])
                    .collect();

                let (mean, std) = mean_and_std(&local_vals);
                let val = sample.values[p];
                let anomaly = ((val - mean) / std).abs();

                if anomaly > 3.0 {
                    let conf = (anomaly * 1.15).min(15.0);
                    suspects.push((conf, sample.hole_id.clone(), sample.idx, property_names[p].clone()));
                }
            }
        }
    }

    // Remove duplicates (keep highest confidence per sample+property)
    let mut unique_suspects: HashMap<(String, usize, String), f64> = HashMap::new();
    for (conf, hid, idx, pname) in suspects {
        let key = (hid, idx, pname);
        unique_suspects
            .entry(key)
            .and_modify(|e| *e = e.max(conf))
            .or_insert(conf);
    }

    let mut final_suspects: Vec<(f64, String, usize, String)> = unique_suspects
        .into_iter()
        .map(|((hid, idx, pname), conf)| (conf, hid, idx, pname))
        .collect();

    final_suspects.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap());

    // Output
    println!("{}", final_suspects.len());

    let stdout = io::stdout();
    let mut handle = stdout.lock();

    for (conf, hole_id, sample_idx, prop_name) in final_suspects {
        writeln!(
            handle,
            "{} {} {} {:.4}",
            hole_id, sample_idx, prop_name, conf
        )
        .ok();
    }
}
```
