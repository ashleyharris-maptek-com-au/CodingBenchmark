use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let nv_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => {
            println!("0");
            return;
        }
    };
    let nv: usize = match nv_line.trim().parse() {
        Ok(n) => n,
        _ => {
            println!("0");
            return;
        }
    };

    let mut poly: Vec<(f64, f64)> = Vec::with_capacity(nv);
    for _ in 0..nv {
        if let Some(Ok(l)) = lines.next() {
            let parts: Vec<&str> = l.split_whitespace().collect();
            if parts.len() >= 2 {
                if let (Ok(x), Ok(y)) = (parts[0].parse::<f64>(), parts[1].parse::<f64>()) {
                    poly.push((x, y));
                }
            }
        }
    }

    let sun_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => String::new(),
    };
    let sun_parts: Vec<&str> = sun_line.split_whitespace().collect();
    let _sx: f64 = sun_parts.get(0).and_then(|s| s.parse().ok()).unwrap_or(0.0);
    let _sy: f64 = sun_parts.get(1).and_then(|s| s.parse().ok()).unwrap_or(0.0);
    let _sz: f64 = sun_parts.get(2).and_then(|s| s.parse().ok()).unwrap_or(1.0);

    if poly.is_empty() {
        println!("0");
        return;
    }

    let n = poly.len() as f64;
    let cx: f64 = poly.iter().map(|p| p.0).sum::<f64>() / n;
    let cy: f64 = poly.iter().map(|p| p.1).sum::<f64>() / n;

    let min_x = poly.iter().map(|p| p.0).fold(f64::INFINITY, |a, b| a.min(b));
    let max_x = poly.iter().map(|p| p.0).fold(f64::NEG_INFINITY, |a, b| a.max(b));
    let min_y = poly.iter().map(|p| p.1).fold(f64::INFINITY, |a, b| a.min(b));
    let max_y = poly.iter().map(|p| p.1).fold(f64::NEG_INFINITY, |a, b| a.max(b));

    let width = (max_x - min_x).max(1.0);
    let height = (max_y - min_y).max(1.0);
    let area = width * height;

    // Adaptive strategy based on scale (NV and bounding box size)
    let mut outputs: Vec<String> = Vec::new();

    if nv <= 4 || area < 25.0 {
        // Simple case: single large tetrahedron
        let scale = (area.sqrt() * 0.6).min(10.0).max(1.5);
        let pz = 2.5 + (_sz.abs() * 0.5);
        outputs.push(format!(
            "{:.6} {:.6} {:.6} 1.000000 0.000000 0.000000 0.000000 {:.6}",
            cx, cy, pz, scale
        ));
    } else if nv <= 12 || area < 200.0 {
        // Medium: 4 tetrahedrons in a grid
        let base_scale = (area.sqrt() * 0.25).min(8.0).max(2.0);
        let offset = (width.max(height) * 0.25).min(5.0);
        let pz_base = 3.0;

        for i in 0..2 {
            for j in 0..2 {
                let px = cx + (i as f64 - 0.5) * offset;
                let py = cy + (j as f64 - 0.5) * offset;
                let pz = pz_base + (i as f64 + j as f64) * 0.8;
                let scale = base_scale;
                outputs.push(format!(
                    "{:.6} {:.6} {:.6} 1.000000 0.000000 0.000000 0.000000 {:.6}",
                    px, py, pz, scale
                ));
            }
        }
    } else {
        // Large/complex: grid of tetrahedrons
        let base_scale = 4.0_f64.min(10.0);
        let step = base_scale * 1.8;
        let cols = ((width / step).ceil() as usize).max(2).min(12);
        let rows = ((height / step).ceil() as usize).max(2).min(12);

        let mut pz = 2.5;
        for i in 0..cols {
            for j in 0..rows {
                let px = cx + (i as f64 - (cols as f64 - 1.0) / 2.0) * step;
                let py = cy + (j as f64 - (rows as f64 - 1.0) / 2.0) * step;
                outputs.push(format!(
                    "{:.6} {:.6} {:.6} 1.000000 0.000000 0.000000 0.000000 {:.6}",
                    px, py, pz, base_scale
                ));
                pz += 0.35; // Slight z-offset to avoid intersections
            }
        }
    }

    // Ensure max scale
    for out in &mut outputs {
        let parts: Vec<&str> = out.split_whitespace().collect();
        if parts.len() >= 8 {
            if let Ok(mut s) = parts[7].parse::<f64>() {
                if s > 10.0 {
                    s = 10.0;
                    *out = format!(
                        "{} {} {} {} {} {} {} {}",
                        parts[0], parts[1], parts[2],
                        parts[3], parts[4], parts[5], parts[6], s
                    );
                }
            }
        }
    }

    println!("{}", outputs.len());
    for line in outputs {
        println!("{}", line);
    }
}
