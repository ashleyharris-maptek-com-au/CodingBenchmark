#include <iostream>
#include <vector>
#include <cmath>
#include <iomanip>
#include <algorithm>

struct Vec3 {
    double x, y, z;
};

Vec3 operator+(const Vec3& a, const Vec3& b) {
    return {a.x + b.x, a.y + b.y, a.z + b.z};
}

struct Placement {
    double cx, cy, cz;
    double rx, ry, rz;
};

int main() {
    int V;
    double e;
    std::cin >> V >> e;

    std::vector<Vec3> verts(V);
    for (int i = 0; i < V; ++i) {
        std::cin >> verts[i].x >> verts[i].y >> verts[i].z;
    }

    if (V == 0 || e <= 0) {
        std::cout << 0 << std::endl;
        return 0;
    }

    // Compute AABB
    double minx = verts[0].x, maxx = verts[0].x;
    double miny = verts[0].y, maxy = verts[0].y;
    double minz = verts[0].z, maxz = verts[0].z;
    for (const auto& p : verts) {
        minx = std::min(minx, p.x);
        maxx = std::max(maxx, p.x);
        miny = std::min(miny, p.y);
        maxy = std::max(maxy, p.y);
        minz = std::min(minz, p.z);
        maxz = std::max(maxz, p.z);
    }

    double sqrt2 = std::sqrt(2.0);
    double s = e / (2.0 * sqrt2);  // scaling factor for unit tetra coords

    // Relative tetra vertices (edge length e, center at origin)
    Vec3 tet_rel[4] = {
        { s,  s,  s},
        { s, -s, -s},
        {-s,  s, -s},
        {-s, -s,  s}
    };

    // Safe grid step (conservative for no overlap via circumspheres)
    double R = e * std::sqrt(3.0 / 8.0);
    double step = 2.0 * R * 1.05;  // ~1.29 * e spacing

    std::vector<Placement> placements;

    // Grid over possible centers (AABB shrunk by tetra extent)
    double extent = s;
    for (double x = minx + extent; x <= maxx - extent + 1e-9; x += step) {
        for (double y = miny + extent; y <= maxy - extent + 1e-9; y += step) {
            for (double z = minz + extent; z <= maxz - extent + 1e-9; z += step) {
                Vec3 C = {x, y, z};

                bool valid = true;
                for (int i = 0; i < 4; ++i) {
                    Vec3 v = C + tet_rel[i];
                    if (v.x < minx || v.x > maxx ||
                        v.y < miny || v.y > maxy ||
                        v.z < minz || v.z > maxz) {
                        valid = false;
                        break;
                    }
                }

                if (valid) {
                    placements.push_back({x, y, z, 0.0, 0.0, 0.0});
                }
            }
        }
    }

    // Output
    std::cout << placements.size() << std::endl;
    std::cout << std::fixed << std::setprecision(10);
    for (const auto& p : placements) {
        std::cout << p.cx << " " << p.cy << " " << p.cz << " "
                  << p.rx << " " << p.ry << " " << p.rz << std::endl;
    }

    return 0;
}
