use std::io::{self, BufRead};
use std::collections::HashMap;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = match lines.next() {
        Some(Ok(line)) => line,
        _ => {
            println!("0");
            return;
        }
    };

    let mut parts = first_line.split_whitespace();
    let n: usize = match parts.next().and_then(|s| s.parse().ok()) {
        Some(val) => val,
        None => {
            println!("0");
            return;
        }
    };
    let m: usize = match parts.next().and_then(|s| s.parse().ok()) {
        Some(val) => val,
        None => 0,
    };

    let mut adj: Vec<Vec<usize>> = vec![vec![]; n];

    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let mut ep = line.split_whitespace();
            if let (Some(u_str), Some(v_str)) = (ep.next(), ep.next()) {
                if let (Ok(u), Ok(v)) = (u_str.parse::<usize>(), v_str.parse::<usize>()) {
                    if u < n && v < n && u != v {
                        adj[u].push(v);
                        adj[v].push(u);
                    }
                }
            }
        }
    }

    // Deduplicate adjacency lists
    for i in 0..n {
        adj[i].sort_unstable();
        adj[i].dedup();
    }

    let comps = find_components(n, &adj);
    let positions = layout_all(n, &comps, &adj);

    println!("{}", n);
    for (x, y) in positions {
        println!("{:.6} {:.6}", x, y);
    }
}

fn find_components(n: usize, adj: &[Vec<usize>]) -> Vec<Vec<usize>> {
    let mut visited = vec![false; n];
    let mut comps: Vec<Vec<usize>> = Vec::new();

    for i in 0..n {
        if !visited[i] {
            let mut comp = Vec::new();
            let mut stack = vec![i];
            visited[i] = true;

            while let Some(u) = stack.pop() {
                comp.push(u);
                for &v in &adj[u] {
                    if !visited[v] {
                        visited[v] = true;
                        stack.push(v);
                    }
                }
            }
            comps.push(comp);
        }
    }
    comps
}

fn layout_all(n: usize, comps: &[Vec<usize>], adj: &[Vec<usize>]) -> Vec<(f64, f64)> {
    let mut global_pos: Vec<(f64, f64)> = vec![(0.0, 0.0); n];

    let mut x_offset: f64 = 0.0;
    let mut y_offset: f64 = 0.0;
    let mut row_max_height: f64 = 0.0;
    let padding: f64 = 80.0;
    let max_row_width: f64 = 12000.0;

    for comp_nodes in comps {
        let s = comp_nodes.len();
        if s == 0 {
            continue;
        }

        let local_pos = if s == 1 {
            vec![(0.0, 0.0)]
        } else if s <= 120 {
            let local_adj = build_local_adj(comp_nodes, adj);
            force_directed(&local_adj, 450)
        } else {
            circular_layout(s)
        };

        // Estimate size of this component
        let comp_diam: f64 = (s as f64).sqrt() * 35.0 + 50.0;

        if x_offset + comp_diam > max_row_width && comp_nodes.len() > 1 {
            x_offset = 0.0;
            y_offset += row_max_height + padding;
            row_max_height = 0.0;
        }

        // Apply local positions with scaling and offset
        let scale: f64 = 1.3;
        for (i, &node) in comp_nodes.iter().enumerate() {
            if i < local_pos.len() {
                let (lx, ly) = local_pos[i];
                global_pos[node] = (
                    x_offset + lx * scale,
                    y_offset + ly * scale,
                );
            }
        }

        row_max_height = row_max_height.max(comp_diam);
        x_offset += comp_diam + padding * 0.6;
    }

    global_pos
}

fn build_local_adj(nodes: &[usize], adj: &[Vec<usize>]) -> Vec<Vec<usize>> {
    let s = nodes.len();
    let mut node_to_local: HashMap<usize, usize> = HashMap::with_capacity(s);
    for (i, &v) in nodes.iter().enumerate() {
        node_to_local.insert(v, i);
    }

    let mut local_adj: Vec<Vec<usize>> = vec![vec![]; s];

    for (li, &u) in nodes.iter().enumerate() {
        for &v in &adj[u] {
            if let Some(&lv) = node_to_local.get(&v) {
                local_adj[li].push(lv);
            }
        }
    }
    local_adj
}

fn circular_layout(s: usize) -> Vec<(f64, f64)> {
    if s == 0 {
        return vec![];
    }
    let r = (s as f64 * 4.5).max(25.0);
    (0..s)
        .map(|i| {
            let angle = 2.0 * std::f64::consts::PI * (i as f64) / (s as f64);
            (r * angle.cos(), r * angle.sin())
        })
        .collect()
}

fn force_directed(local_adj: &[Vec<usize>], iterations: usize) -> Vec<(f64, f64)> {
    let s = local_adj.len();
    if s == 0 {
        return vec![];
    }

    // Grid initialization
    let cols = ((s as f64).sqrt().ceil() as usize).max(1);
    let mut pos: Vec<(f64, f64)> = (0..s)
        .map(|i| {
            let row = i / cols;
            let col = i % cols;
            (col as f64 * 28.0 - 200.0, row as f64 * 28.0 - 150.0)
        })
        .collect();

    let k: f64 = 32.0;
    let mut temp: f64 = 180.0;

    for _ in 0..iterations {
        let mut disp: Vec<(f64, f64)> = vec![(0.0, 0.0); s];

        // Repulsive forces (all pairs)
        for i in 0..s {
            for j in (i + 1)..s {
                let dx = pos[i].0 - pos[j].0;
                let dy = pos[i].1 - pos[j].1;
                let dist2 = dx * dx + dy * dy + 1e-8;
                let dist = dist2.sqrt();
                let force = k * k / dist;
                let fx = dx / dist * force;
                let fy = dy / dist * force;
                disp[i].0 += fx;
                disp[i].1 += fy;
                disp[j].0 -= fx;
                disp[j].1 -= fy;
            }
        }

        // Attractive forces (edges, avoid double counting)
        for u in 0..s {
            for &v in &local_adj[u] {
                if v > u {
                    let dx = pos[u].0 - pos[v].0;
                    let dy = pos[u].1 - pos[v].1;
                    let dist = (dx * dx + dy * dy).sqrt() + 1e-8;
                    let force = dist * dist / k;
                    let fx = dx / dist * force;
                    let fy = dy / dist * force;
                    disp[u].0 -= fx;
                    disp[u].1 -= fy;
                    disp[v].0 += fx;
                    disp[v].1 += fy;
                }
            }
        }

        // Apply displacements with temperature limit
        for i in 0..s {
            let dlen = (disp[i].0 * disp[i].0 + disp[i].1 * disp[i].1).sqrt() + 1e-8;
            let limited = temp.min(dlen);
            pos[i].0 += disp[i].0 / dlen * limited;
            pos[i].1 += disp[i].1 / dlen * limited;
        }

        temp *= 0.92;
        if temp < 0.5 {
            break;
        }
    }

    // Center at origin
    let mut cx = 0.0;
    let mut cy = 0.0;
    for &(x, y) in &pos {
        cx += x;
        cy += y;
    }
    cx /= s as f64;
    cy /= s as f64;

    for p in &mut pos {
        p.0 -= cx;
        p.1 -= cy;
    }

    pos
}
