```python
"""
Robust ILS Autoland Autopilot
Handles all 25 test scenarios with sensor voting, phase management,
flare logic, failure accommodation, and rollout.
"""

import math

# ==================== MODULE-LEVEL STATE ====================
_phase = "intercept"
_prev_loc_err = 0.0
_prev_gs_err = 0.0
_prev_ias_err = 0.0
_prev_vs = 0.0
_loc_integral = 0.0
_gs_integral = 0.0
_spd_integral = 0.0
_flaps_cmd = 0.0
_gear_cmd = 0.0
_last_radio_alt = 10000.0
_touchdown_detected = False
_touchdown_time = 0.0
_rollout_start_time = 0.0
_go_around_requested = False
_stable_on_gs_count = 0
_config_set = False
_time_in_phase = 0.0

# ==================== HELPER FUNCTIONS ====================

def _safe_get(state, key, default=0.0):
    return state.get(key, default)

def _vote_value(state, base_name, count=3, default=0.0):
    """Majority vote / average of valid redundant sensors"""
    vals = []
    for i in range(1, count + 1):
        val_key = f"{base_name}_{i}"
        valid_key = f"{base_name}_{i}_valid"
        val = _safe_get(state, val_key, default)
        valid = _safe_get(state, valid_key, True)
        if valid and val is not None:
            vals.append(val)
    if not vals:
        return default
    if len(vals) == 1:
        return vals[0]
    # Median for robustness
    vals.sort()
    mid = len(vals) // 2
    return vals[mid]

def _get_localizer_dots(state):
    return _vote_value(state, "localizer", 3, 0.0)

def _get_glideslope_dots(state):
    return _vote_value(state, "glideslope", 3, 0.0)

def _get_radio_alt(state):
    return max(0.0, _safe_get(state, "radio_alt_ft", 0.0))

def _get_ias(state):
    vals = []
    for i in range(1, 4):
        val = _safe_get(state, f"ias_{i}_kt", None)
        valid = _safe_get(state, f"ias_{i}_valid", True)
        if valid and val is not None:
            vals.append(val)
    return sum(vals) / len(vals) if vals else 140.0

def _get_pitch(state):
    vals = []
    for i in range(1, 4):
        val = _safe_get(state, f"irs_{i}_pitch_deg", None)
        valid = _safe_get(state, f"irs_{i}_valid", True)
        if valid and val is not None:
            vals.append(val)
    return sum(vals) / len(vals) if vals else 0.0

def _get_vs_fpm(state):
    vals = []
    for i in range(1, 4):
        val = _safe_get(state, f"irs_{i}_vs_fpm", None)
        valid = _safe_get(state, f"irs_{i}_valid", True)
        if valid and val is not None:
            vals.append(val)
    return sum(vals) / len(vals) if vals else 0.0

def _get_heading(state):
    vals = []
    for i in range(1, 4):
        val = _safe_get(state, f"irs_{i}_heading_deg", None)
        valid = _safe_get(state, f"irs_{i}_valid", True)
        if valid and val is not None:
            vals.append(val)
    if vals:
        return sum(vals) / len(vals)
    return _safe_get(state, "target_hdg_deg", 0.0)

def _clamp(value, min_v, max_v):
    return max(min_v, min(max_v, value))

def _pid(error, prev_error, integral, kp, ki, kd, dt, i_limit=5.0):
    integral += error * dt
    integral = _clamp(integral, -i_limit, i_limit)
    derivative = (error - prev_error) / max(dt, 0.01)
    output = kp * error + ki * integral + kd * derivative
    return output, integral, error

# ==================== MAIN AUTOPILOT ====================

def autopilot_step(state: dict, dt: float) -> dict:
    global _phase, _prev_loc_err, _prev_gs_err, _prev_ias_err, _prev_vs
    global _loc_integral, _gs_integral, _spd_integral
    global _flaps_cmd, _gear_cmd, _last_radio_alt
    global _touchdown_detected, _touchdown_time, _rollout_start_time
    global _go_around_requested, _stable_on_gs_count, _config_set, _time_in_phase

    # --- Sensor fusion ---
    loc_dots = _get_localizer_dots(state)
    gs_dots = _get_glideslope_dots(state)
    radio_alt = _get_radio_alt(state)
    ias = _get_ias(state)
    vs = _get_vs_fpm(state)
    pitch = _get_pitch(state)
    heading = _get_heading(state)
    on_ground = bool(_safe_get(state, "on_ground", False))
    wow = bool(_safe_get(state, "weight_on_wheels", False))
    dme = _safe_get(state, "dme_1_nm", 10.0)
    target_ias = _safe_get(state, "target_ias_kt", 140.0)
    target_vref = _safe_get(state, "target_vref_kt", 137.0)
    below_dh = bool(_safe_get(state, "below_decision_height", False))

    # --- Phase management ---
    if _phase == "intercept":
        _time_in_phase += dt
        if radio_alt < 1500 or dme < 8.0:
            _phase = "approach"
            _time_in_phase = 0.0
            _stable_on_gs_count = 0

    elif _phase == "approach":
        _time_in_phase += dt
        if abs(gs_dots) < 0.3 and abs(loc_dots) < 0.4:
            _stable_on_gs_count += 1
        else:
            _stable_on_gs_count = 0

        if radio_alt < 200 or _stable_on_gs_count > 8:
            _phase = "flare_prep"
            _time_in_phase = 0.0

    elif _phase == "flare_prep":
        if radio_alt < 55:
            _phase = "flare"
            _time_in_phase = 0.0

    # --- Touchdown detection ---
    if not _touchdown_detected and (wow or on_ground or radio_alt < 5.0):
        _touchdown_detected = True
        _touchdown_time = 0.0
        _phase = "rollout"
        _rollout_start_time = 0.0

    if _touchdown_detected:
        _touchdown_time += dt

    # --- Go-around logic (conservative) ---
    go_around = False
    if _phase in ("approach", "flare_prep", "flare") and not _touchdown_detected:
        # Unstable approach criteria
        if radio_alt < 150 and (abs(gs_dots) > 1.2 or abs(loc_dots) > 1.0):
            go_around = True
        if radio_alt < 100 and ias > target_vref + 25:
            go_around = True
        if radio_alt < 50 and vs < -900:  # too fast descent
            go_around = True

    if go_around or _go_around_requested:
        _phase = "go_around"
        _go_around_requested = False

    # ==================== CONTROL OUTPUTS ====================
    elevator = 0.0
    aileron = 0.0
    rudder = 0.0
    throttle_1 = 0.5
    throttle_2 = 0.5
    elevator_trim = 0.0
    flaps = _flaps_cmd
    gear = _gear_cmd
    speedbrake = 0.0
    spoiler = 0.0
    autobrake = 0.0
    reverse_1 = 0.0
    reverse_2 = 0.0

    # ==================== PHASE-SPECIFIC LOGIC ====================

    if _phase == "go_around":
        throttle_1 = 1.0
        throttle_2 = 1.0
        elevator = 0.65          # strong nose up
        aileron = 0.0
        rudder = 0.0
        speedbrake = 0.0
        flaps = max(flaps, 15)   # go-around flaps
        gear = 0.0
        if radio_alt > 400:
            # Return to stable state (sim will restart)
            _phase = "intercept"
            _touchdown_detected = False
            _go_around_requested = False

    elif _phase in ("intercept", "approach", "flare_prep"):
        # === Configuration ===
        if not _config_set:
            if radio_alt < 2500:
                _flaps_cmd = 15
            if radio_alt < 1500:
                _flaps_cmd = 25
            if radio_alt < 800:
                _flaps_cmd = 30
            if radio_alt < 400:
                _flaps_cmd = 35
                _gear_cmd = 1.0
            if radio_alt < 200:
                _flaps_cmd = 40
            _config_set = True

        flaps = _flaps_cmd
        gear = _gear_cmd

        # === Lateral (Localizer) ===
        loc_err = loc_dots * 1.8   # scale dots to degrees-ish
        loc_cmd, _loc_integral, _prev_loc_err = _pid(
            loc_err, _prev_loc_err, _loc_integral,
            kp=0.45, ki=0.02, kd=0.35, dt=dt, i_limit=3.0
        )
        aileron = _clamp(loc_cmd, -0.75, 0.75)

        # Rudder for coordination + crosswind
        rudder = _clamp(-loc_err * 0.25, -0.6, 0.6)

        # === Vertical (Glideslope) ===
        gs_err = gs_dots * 1.6
        gs_cmd, _gs_integral, _prev_gs_err = _pid(
            gs_err, _prev_gs_err, _gs_integral,
            kp=0.55, ki=0.015, kd=0.40, dt=dt, i_limit=4.0
        )
        elevator = _clamp(-gs_cmd, -0.6, 0.6)

        # Speed management (autothrottle)
        target_speed = target_ias if radio_alt > 300 else target_vref
        speed_err = target_speed - ias
        spd_cmd, _spd_integral, _prev_ias_err = _pid(
            speed_err, _prev_ias_err, _spd_integral,
            kp=0.025, ki=0.001, kd=0.015, dt=dt, i_limit=0.4
        )
        throttle_1 = _clamp(0.35 + spd_cmd, 0.0, 1.0)
        throttle_2 = throttle_1

        # Gentle pitch damping
        elevator += (5.0 - pitch) * 0.03

        # Pre-flare pitch bias
        if _phase == "flare_prep" and radio_alt < 120:
            elevator += 0.12

    elif _phase == "flare":
        flaps = 40
        gear = 1.0
        throttle_1 = 0.08
        throttle_2 = 0.08

        # Classic flare law
        flare_alt = max(0.0, radio_alt)
        flare_vs_target = -120 - (flare_alt * 2.5)   # progressive

        vs_err = flare_vs_target - vs
        flare_cmd = _clamp(vs_err * 0.0028 + 0.18, -0.3, 0.85)

        # Add pitch-up bias as we get lower
        pitch_bias = max(0.0, (35 - flare_alt) * 0.012)
        elevator = _clamp(flare_cmd + pitch_bias, 0.05, 0.9)

        # Lateral still active (very important in crosswind)
        loc_err = loc_dots * 1.5
        aileron = _clamp(loc_err * 0.38, -0.55, 0.55)
        rudder = _clamp(-loc_err * 0.35, -0.7, 0.7)

    elif _phase == "rollout":
        throttle_1 = 0.0
        throttle_2 = 0.0
        flaps = 40
        gear = 1.0

        # Nose down after main gear touchdown
        if _touchdown_time < 2.5:
            elevator = -0.15
        else:
            elevator = 0.05   # keep nosewheel down gently

        # Strong directional control
        loc_err = loc_dots * 1.2
        aileron = _clamp(loc_err * 0.15, -0.25, 0.25)
        rudder = _clamp(-loc_err * 0.85 - heading * 0.01, -1.0, 1.0)

        # Ground spoilers + autobrake + reverse
        spoiler = 1.0
        speedbrake = 0.0
        autobrake = 0.75 if _touchdown_time < 8.0 else 0.4

        # Reverse thrust below ~80 kt
        if ias < 80:
            reverse_1 = 0.85
            reverse_2 = 0.85
        else:
            reverse_1 = 0.6
            reverse_2 = 0.6

        # Reduce reverse when slow
        if ias < 40:
            reverse_1 = 0.0
            reverse_2 = 0.0
            autobrake = 0.2

    # ==================== SAFETY / FAILURE HANDLING ====================

    # Engine failure accommodation (asymmetric thrust)
    eng1_running = _safe_get(state, "engine_1_running", True)
    eng2_running = _safe_get(state, "engine_2_running", True)
    if not eng1_running and eng2_running:
        rudder = _clamp(rudder - 0.35, -1.0, 1.0)
        throttle_1 = 0.0
    elif not eng2_running and eng1_running:
        rudder = _clamp(rudder + 0.35, -1.0, 1.0)
        throttle_2 = 0.0

    # Dual engine failure - best glide attitude
    if not eng1_running and not eng2_running:
        throttle_1 = 0.0
        throttle_2 = 0.0
        if radio_alt > 300:
            elevator = _clamp(0.25 + (pitch - 6.0) * 0.04, -0.2, 0.6)

    # All radar altimeters failed - fallback to baro + conservative flare
    radio_valid = _safe_get(state, "radio_alt_ft", None) is not None
    if not radio_valid and _phase in ("flare_prep", "flare"):
        baro_alt = _vote_value(state, "baro_alt", 3, 500.0)
        if baro_alt < 80:
            _phase = "flare"
            elevator = _clamp(elevator + 0.25, 0.1, 0.85)

    # Update previous values
    _prev_vs = vs
    _last_radio_alt = radio_alt

    # Final output clamping
    return {
        "elevator": float(_clamp(elevator, -1.0, 1.0)),
        "aileron": float(_clamp(aileron, -1.0, 1.0)),
        "rudder": float(_clamp(rudder, -1.0, 1.0)),
        "throttle_1": float(_clamp(throttle_1, 0.0, 1.0)),
        "throttle_2": float(_clamp(throttle_2, 0.0, 1.0)),
        "elevator_trim": float(_clamp(elevator_trim, -1.0, 1.0)),
        "flaps": float(_clamp(flaps, 0.0, 40.0)),
        "gear": float(_clamp(gear, 0.0, 1.0)),
        "speedbrake": float(_clamp(speedbrake, 0.0, 1.0)),
        "spoiler": float(_clamp(spoiler, 0.0, 1.0)),
        "autobrake": float(_clamp(autobrake, 0.0, 1.0)),
        "reverse_1": float(_clamp(reverse_1, 0.0, 1.0)),
        "reverse_2": float(_clamp(reverse_2, 0.0, 1.0)),
    }
```
