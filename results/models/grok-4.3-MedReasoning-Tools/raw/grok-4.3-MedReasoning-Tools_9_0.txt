```csharp
using System;
using System.Collections.Generic;
using System.IO;

public class Program
{
    public static void Main()
    {
        // Fast input handling
        string line1 = Console.ReadLine();
        if (line1 == null) return;
        var parts = line1.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(parts[0]);
        int stockLength = int.Parse(parts[1]);

        string line2 = Console.ReadLine();
        if (line2 == null) return;
        var cutStrs = line2.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);

        var items = new List<(int length, int index)>(N);
        for (int i = 0; i < N; i++)
        {
            int len = int.Parse(cutStrs[i]);
            if (len <= stockLength)
            {
                items.Add((len, i));
            }
        }

        // Sort descending by length (First Fit Decreasing)
        items.Sort((a, b) => b.length.CompareTo(a.length));

        var bins = new List<List<int>>();
        var remains = new List<int>();

        foreach (var item in items)
        {
            bool placed = false;
            for (int b = 0; b < bins.Count; b++)
            {
                if (remains[b] >= item.length)
                {
                    bins[b].Add(item.index);
                    remains[b] -= item.length;
                    placed = true;
                    break;
                }
            }
            if (!placed)
            {
                bins.Add(new List<int> { item.index });
                remains.Add(stockLength - item.length);
            }
        }

        // Output
        Console.WriteLine(bins.Count);
        foreach (var bin in bins)
        {
            Console.WriteLine(string.Join(" ", bin));
        }
    }
}
```
