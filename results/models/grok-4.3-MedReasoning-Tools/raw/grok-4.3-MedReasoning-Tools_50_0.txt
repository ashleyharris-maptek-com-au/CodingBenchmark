```glsl
#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 radii;
    vec4 weights;
    vec4 bounds;
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    if (i >= N) {
        return;
    }

    // Read current boid
    vec3 pos = inData[i * 2].xyz;
    vec3 vel = inData[i * 2 + 1].xyz;

    // Radii and weights
    float dt         = radii.x;
    float sepRadius  = radii.y;
    float aliRadius  = radii.z;
    float cohRadius  = radii.w;

    float sepWeight  = weights.x;
    float aliWeight  = weights.y;
    float cohWeight  = weights.z;
    float maxSpeed   = weights.w;

    float boundSize  = bounds.x;
    float boundForce = bounds.y;

    // Accumulators
    vec3 separation = vec3(0.0);
    vec3 alignment  = vec3(0.0);
    vec3 cohesion   = vec3(0.0);

    int countSep = 0;
    int countAli = 0;
    int countCoh = 0;

    // Brute-force neighbor loop
    for (uint j = 0u; j < N; ++j) {
        if (i == j) continue;

        vec3 posj = inData[j * 2].xyz;
        vec3 velj = inData[j * 2 + 1].xyz;

        vec3 diff = pos - posj;
        float dist = length(diff);

        if (dist < 1e-5) continue; // skip zero-distance neighbors

        // Separation
        if (dist < sepRadius) {
            separation += normalize(diff);
            countSep++;
        }

        // Alignment
        if (dist < aliRadius) {
            alignment += velj;
            countAli++;
        }

        // Cohesion
        if (dist < cohRadius) {
            cohesion += posj;
            countCoh++;
        }
    }

    // Average contributions
    if (countSep > 0) {
        separation /= float(countSep);
    }
    if (countAli > 0) {
        alignment /= float(countAli);
    }
    if (countCoh > 0) {
        cohesion = (cohesion / float(countCoh)) - pos;
    }

    // Compute acceleration (steering forces)
    vec3 accel = vec3(0.0);
    accel += separation * sepWeight;
    accel += (alignment - vel) * aliWeight;
    accel += cohesion * cohWeight;

    // Boundary force (push toward center)
    float thresh = boundSize * 0.4;
    float halfBound = boundSize * 0.5;

    if (abs(pos.x) > thresh) {
        accel.x += -sign(pos.x) * boundForce * (halfBound - abs(pos.x)) / halfBound;
    }
    if (abs(pos.y) > thresh) {
        accel.y += -sign(pos.y) * boundForce * (halfBound - abs(pos.y)) / halfBound;
    }
    if (abs(pos.z) > thresh) {
        accel.z += -sign(pos.z) * boundForce * (halfBound - abs(pos.z)) / halfBound;
    }

    // Update velocity
    vec3 newVel = vel + accel * dt;

    // Clamp speed
    float speed = length(newVel);
    if (speed > maxSpeed && speed > 1e-5) {
        newVel = newVel * (maxSpeed / speed);
    }

    // Update position
    vec3 newPos = pos + newVel * dt;

    // Write output
    outData[i * 2]     = vec4(newPos, 0.0);
    outData[i * 2 + 1] = vec4(newVel, 0.0);
}
```
