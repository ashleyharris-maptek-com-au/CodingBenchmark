```glsl
#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    if (i >= N) {
        return;
    }

    float dt        = physics.x;
    float softening = physics.y;
    float G         = physics.z;

    // Read current state of body i
    vec4  posMass   = inData[i * 2];
    vec3  pos_i     = posMass.xyz;
    float mass_i    = posMass.w;

    vec3  vel_i     = inData[i * 2 + 1].xyz;

    // Compute acceleration
    vec3 acc = vec3(0.0);

    for (uint j = 0u; j < N; ++j) {
        if (j == i) {
            continue;
        }

        vec4  posMass_j = inData[j * 2];
        vec3  pos_j     = posMass_j.xyz;
        float mass_j    = posMass_j.w;

        vec3  diff      = pos_j - pos_i;
        float distSqr   = dot(diff, diff) + softening * softening;
        float invDist   = inversesqrt(distSqr);
        float invDist3  = invDist * invDist * invDist;

        acc += G * mass_j * diff * invDist3;
    }

    // Semi-implicit Euler / leapfrog-style integration
    vec3 new_vel = vel_i + acc * dt;
    vec3 new_pos = pos_i + new_vel * dt;

    // Write updated state
    outData[i * 2]     = vec4(new_pos, mass_i);
    outData[i * 2 + 1] = vec4(new_vel, 0.0);
}
```
