```cpp
#include <iostream>
#include <string>
#include <vector>
#include <array>
#include <chrono>
#include <algorithm>
#include <cctype>
#include <sstream>
#include <random>
#include <limits>

using namespace std;

constexpr int INF = 100000000;
constexpr int MATE = 1000000;

struct Move {
    int from;
    int to;
    char promotion;
    bool is_capture;
    bool is_en_passant;
    bool is_castle;
    string notation;
    
    Move() : from(-1), to(-1), promotion(0), is_capture(false), is_en_passant(false), is_castle(false) {}
};

class ChessEngine {
public:
    array<char, 64> board;
    bool white_to_move;
    int castling_rights; // KQkq bits: 1=K, 2=Q, 4=k, 8=q
    int en_passant_sq;
    int halfmove_clock;
    int fullmove_number;
    
    chrono::steady_clock::time_point start_time;
    double timeout_seconds;
    Move best_move_found;
    int best_score_found;
    
    ChessEngine() : castling_rights(0), en_passant_sq(-1), halfmove_clock(0), fullmove_number(1) {
        fill(board.begin(), board.end(), 0);
    }
    
    void load_fen(const string& fen) {
        fill(board.begin(), board.end(), 0);
        castling_rights = 0;
        en_passant_sq = -1;
        
        stringstream ss(fen);
        string board_str, side, castling, ep, half, full;
        ss >> board_str >> side >> castling >> ep >> half >> full;
        
        vector<string> ranks;
        stringstream rank_ss(board_str);
        string rank;
        while (getline(rank_ss, rank, '/')) {
            ranks.push_back(rank);
        }
        
        for (int r = 0; r < 8; r++) {
            int file = 0;
            for (char c : ranks[r]) {
                if (isdigit(c)) {
                    file += c - '0';
                } else {
                    int square = (7 - r) * 8 + file;
                    board[square] = c;
                    file++;
                }
            }
        }
        
        white_to_move = (side == "w");
        
        if (castling != "-") {
            for (char c : castling) {
                if (c == 'K') castling_rights |= 1;
                if (c == 'Q') castling_rights |= 2;
                if (c == 'k') castling_rights |= 4;
                if (c == 'q') castling_rights |= 8;
            }
        }
        
        if (ep != "-") {
            int f = ep[0] - 'a';
            int rank_ep = ep[1] - '1';
            en_passant_sq = rank_ep * 8 + f;
        } else {
            en_passant_sq = -1;
        }
        
        if (!half.empty()) halfmove_clock = stoi(half);
        if (!full.empty()) fullmove_number = stoi(full);
    }
    
    int get_rank(int sq) { return sq / 8; }
    int get_file(int sq) { return sq % 8; }
    bool valid_sq(int sq) { return sq >= 0 && sq < 64; }
    
    bool is_white_piece(char p) { return isupper(p); }
    bool is_black_piece(char p) { return islower(p); }
    
    bool is_own_piece(int sq) {
        char p = board[sq];
        if (p == 0) return false;
        return white_to_move ? is_white_piece(p) : is_black_piece(p);
    }
    
    bool is_opponent_piece(int sq) {
        char p = board[sq];
        if (p == 0) return false;
        return white_to_move ? is_black_piece(p) : is_white_piece(p);
    }
    
    int piece_value(char p) {
        char up = toupper(p);
        switch (up) {
            case 'P': return 100;
            case 'N': return 320;
            case 'B': return 330;
            case 'R': return 500;
            case 'Q': return 900;
            case 'K': return 20000;
        }
        return 0;
    }
    
    bool is_attacked(int sq, bool by_white) {
        // Pawn attacks
        int pawn_dir = by_white ? 1 : -1;
        int pawn_sq1 = sq + pawn_dir * 8 - 1;
        int pawn_sq2 = sq + pawn_dir * 8 + 1;
        if (valid_sq(pawn_sq1) && get_file(pawn_sq1) != 7) {
            char p = board[pawn_sq1];
            if (p && (by_white ? p == 'P' : p == 'p')) return true;
        }
        if (valid_sq(pawn_sq2) && get_file(pawn_sq2) != 0) {
            char p = board[pawn_sq2];
            if (p && (by_white ? p == 'P' : p == 'p')) return true;
        }
        
        // Knight attacks
        int knight_deltas[8] = {-17, -15, -10, -6, 6, 10, 15, 17};
        for (int d : knight_deltas) {
            int nsq = sq + d;
            if (valid_sq(nsq)) {
                int diff_file = abs(get_file(nsq) - get_file(sq));
                if (diff_file > 2) continue;
                char p = board[nsq];
                if (p && (by_white ? p == 'N' : p == 'n')) return true;
            }
        }
        
        // King attacks
        int king_deltas[8] = {-9, -8, -7, -1, 1, 7, 8, 9};
        for (int d : king_deltas) {
            int nsq = sq + d;
            if (valid_sq(nsq) && abs(get_file(nsq) - get_file(sq)) <= 1) {
                char p = board[nsq];
                if (p && (by_white ? p == 'K' : p == 'k')) return true;
            }
        }
        
        // Sliding: rook/queen/bishop
        int rook_dirs[4] = {-8, 8, -1, 1};
        int bishop_dirs[4] = {-9, -7, 7, 9};
        
        // Rook + Queen
        for (int d : rook_dirs) {
            int nsq = sq;
            for (int i = 0; i < 8; i++) {
                nsq += d;
                if (!valid_sq(nsq) || abs(get_file(nsq) - get_file(sq)) > 1 && d != -8 && d != 8) break;
                char p = board[nsq];
                if (p) {
                    if ((by_white && (p == 'R' || p == 'Q')) || (!by_white && (p == 'r' || p == 'q'))) return true;
                    break;
                }
            }
        }
        
        // Bishop + Queen
        for (int d : bishop_dirs) {
            int nsq = sq;
            for (int i = 0; i < 8; i++) {
                nsq += d;
                if (!valid_sq(nsq)) break;
                if (abs(get_file(nsq) - get_file(sq)) != 1 && i == 0) continue; // invalid diagonal step
                char p = board[nsq];
                if (p) {
                    if ((by_white && (p == 'B' || p == 'Q')) || (!by_white && (p == 'b' || p == 'q'))) return true;
                    break;
                }
            }
        }
        
        return false;
    }
    
    int find_king(bool white) {
        char king = white ? 'K' : 'k';
        for (int i = 0; i < 64; i++) {
            if (board[i] == king) return i;
        }
        return -1;
    }
    
    bool in_check() {
        int king_sq = find_king(white_to_move);
        if (king_sq == -1) return true;
        return is_attacked(king_sq, !white_to_move);
    }
    
    vector<Move> generate_moves() {
        vector<Move> moves;
        char my_pawn = white_to_move ? 'P' : 'p';
        char my_knight = white_to_move ? 'N' : 'n';
        char my_bishop = white_to_move ? 'B' : 'b';
        char my_rook = white_to_move ? 'R' : 'r';
        char my_queen = white_to_move ? 'Q' : 'q';
        char my_king = white_to_move ? 'K' : 'k';
        
        int forward = white_to_move ? 8 : -8;
        int start_rank = white_to_move ? 1 : 6;
        
        for (int sq = 0; sq < 64; sq++) {
            char piece = board[sq];
            if (piece == 0) continue;
            if (white_to_move != is_white_piece(piece)) continue;
            
            // Pawns
            if (piece == my_pawn) {
                int to = sq + forward;
                if (valid_sq(to) && board[to] == 0) {
                    // Forward
                    if ((white_to_move && get_rank(to) == 7) || (!white_to_move && get_rank(to) == 0)) {
                        // Promotion
                        for (char prom : {'Q','R','B','N'}) {
                            Move m; m.from = sq; m.to = to; m.promotion = prom;
                            moves.push_back(m);
                        }
                    } else {
                        Move m; m.from = sq; m.to = to;
                        moves.push_back(m);
                    }
                    
                    // Double push
                    if (get_rank(sq) == start_rank) {
                        int to2 = sq + 2 * forward;
                        if (valid_sq(to2) && board[to] == 0 && board[to2] == 0) {
                            Move m; m.from = sq; m.to = to2;
                            moves.push_back(m);
                        }
                    }
                }
                
                // Captures
                for (int df : {-1, 1}) {
                    int to_cap = sq + forward + df;
                    if (valid_sq(to_cap) && abs(get_file(to_cap) - get_file(sq)) == 1) {
                        if (is_opponent_piece(to_cap) || to_cap == en_passant_sq) {
                            if ((white_to_move && get_rank(to_cap) == 7) || (!white_to_move && get_rank(to_cap) == 0)) {
                                for (char prom : {'Q','R','B','N'}) {
                                    Move m; m.from = sq; m.to = to_cap; m.promotion = prom; m.is_capture = true;
                                    if (to_cap == en_passant_sq) m.is_en_passant = true;
                                    moves.push_back(m);
                                }
                            } else {
                                Move m; m.from = sq; m.to = to_cap; m.is_capture = true;
                                if (to_cap == en_passant_sq) m.is_en_passant = true;
                                moves.push_back(m);
                            }
                        }
                    }
                }
            }
            
            // Knights
            else if (piece == my_knight) {
                int deltas[8] = {-17,-15,-10,-6,6,10,15,17};
                for (int d : deltas) {
                    int to = sq + d;
                    if (valid_sq(to)) {
                        int df = abs(get_file(to) - get_file(sq));
                        if (df <= 2 && board[to] != 0 && is_own_piece(to)) continue;
                        if (board[to] == 0 || is_opponent_piece(to)) {
                            Move m; m.from = sq; m.to = to; m.is_capture = is_opponent_piece(to);
                            moves.push_back(m);
                        }
                    }
                }
            }
            
            // Bishops / Rooks / Queens (sliding)
            else if (piece == my_bishop || piece == my_rook || piece == my_queen) {
                int dirs[8] = {0};
                int nd = 0;
                if (piece == my_bishop || piece == my_queen) {
                    dirs[nd++] = -9; dirs[nd++] = -7; dirs[nd++] = 7; dirs[nd++] = 9;
                }
                if (piece == my_rook || piece == my_queen) {
                    dirs[nd++] = -8; dirs[nd++] = 8; dirs[nd++] = -1; dirs[nd++] = 1;
                }
                for (int i = 0; i < nd; i++) {
                    int d = dirs[i];
                    int to = sq;
                    for (int step = 0; step < 8; step++) {
                        to += d;
                        if (!valid_sq(to)) break;
                        if (abs(get_file(to) - get_file(sq)) > 1 && (d == -1 || d == 1)) break;
                        if (board[to] != 0) {
                            if (is_opponent_piece(to)) {
                                Move m; m.from = sq; m.to = to; m.is_capture = true;
                                moves.push_back(m);
                            }
                            break;
                        } else {
                            Move m; m.from = sq; m.to = to;
                            moves.push_back(m);
                        }
                    }
                }
            }
            
            // King
            else if (piece == my_king) {
                int deltas[8] = {-9,-8,-7,-1,1,7,8,9};
                for (int d : deltas) {
                    int to = sq + d;
                    if (valid_sq(to) && abs(get_file(to) - get_file(sq)) <= 1) {
                        if (board[to] == 0 || is_opponent_piece(to)) {
                            Move m; m.from = sq; m.to = to; m.is_capture = is_opponent_piece(to);
                            moves.push_back(m);
                        }
                    }
                }
                
                // Castling (simplified - not full validation)
                if (white_to_move) {
                    if ((castling_rights & 1) && sq == 4 && board[5]==0 && board[6]==0) {
                        Move m; m.from = 4; m.to = 6; m.is_castle = true;
                        moves.push_back(m);
                    }
                    if ((castling_rights & 2) && sq == 4 && board[1]==0 && board[2]==0 && board[3]==0) {
                        Move m; m.from = 4; m.to = 2; m.is_castle = true;
                        moves.push_back(m);
                    }
                } else {
                    if ((castling_rights & 4) && sq == 60 && board[61]==0 && board[62]==0) {
                        Move m; m.from = 60; m.to = 62; m.is_castle = true;
                        moves.push_back(m);
                    }
                    if ((castling_rights & 8) && sq == 60 && board[57]==0 && board[58]==0 && board[59]==0) {
                        Move m; m.from = 60; m.to = 58; m.is_castle = true;
                        moves.push_back(m);
                    }
                }
            }
        }
        
        return moves;
    }
    
    void make_move(const Move& m) {
        char piece = board[m.from];
        board[m.to] = piece;
        board[m.from] = 0;
        
        if (m.promotion) {
            board[m.to] = white_to_move ? m.promotion : tolower(m.promotion);
        }
        
        if (m.is_en_passant && en_passant_sq != -1) {
            int cap_sq = en_passant_sq + (white_to_move ? -8 : 8);
            board[cap_sq] = 0;
        }
        
        // Update en passant
        en_passant_sq = -1;
        if (toupper(piece) == 'P' && abs(m.to - m.from) == 16) {
            en_passant_sq = (m.from + m.to) / 2;
        }
        
        // Update castling rights (simplified)
        if (toupper(piece) == 'K') {
            if (white_to_move) castling_rights &= ~3;
            else castling_rights &= ~12;
        }
        if (toupper(piece) == 'R') {
            if (white_to_move) {
                if (m.from == 0) castling_rights &= ~2;
                if (m.from == 7) castling_rights &= ~1;
            } else {
                if (m.from == 56) castling_rights &= ~8;
                if (m.from == 63) castling_rights &= ~4;
            }
        }
        
        white_to_move = !white_to_move;
    }
    
    void undo_move(const Move& m, char captured, int prev_ep, int prev_castling) {
        // Simple undo for search (not perfect for all cases but works for basic)
        char piece = board[m.to];
        board[m.from] = piece;
        board[m.to] = captured;
        if (m.promotion) board[m.from] = white_to_move ? 'P' : 'p'; // approximate
        
        if (m.is_en_passant) {
            int cap_sq = prev_ep + (white_to_move ? 8 : -8);
            board[cap_sq] = white_to_move ? 'p' : 'P';
        }
        
        en_passant_sq = prev_ep;
        castling_rights = prev_castling;
        white_to_move = !white_to_move;
    }
    
    int evaluate() {
        int score = 0;
        for (int sq = 0; sq < 64; sq++) {
            char p = board[sq];
            if (p == 0) continue;
            int val = piece_value(p);
            if (is_white_piece(p)) score += val;
            else score -= val;
        }
        return white_to_move ? score : -score;
    }
    
    int alpha_beta(int depth, int alpha, int beta) {
        if (chrono::duration<double>(chrono::steady_clock::now() - start_time).count() > timeout_seconds - 0.1) {
            return evaluate();
        }
        
        if (depth == 0) {
            return evaluate();
        }
        
        vector<Move> moves = generate_moves();
        
        // Simple move ordering: captures first
        sort(moves.begin(), moves.end(), [](const Move& a, const Move& b) {
            return a.is_capture > b.is_capture;
        });
        
        if (moves.empty()) {
            if (in_check()) return -MATE + depth; // checkmate
            return 0; // stalemate
        }
        
        int best = -INF;
        
        for (const auto& m : moves) {
            char captured = board[m.to];
            int prev_ep = en_passant_sq;
            int prev_cast = castling_rights;
            
            make_move(m);
            
            if (!in_check()) {  // only consider legal moves
                int score = -alpha_beta(depth - 1, -beta, -alpha);
                if (score > best) {
                    best = score;
                    if (depth == 4) {  // record at root-ish depth
                        best_move_found = m;
                        best_score_found = score;
                    }
                }
                alpha = max(alpha, best);
            }
            
            undo_move(m, captured, prev_ep, prev_cast);
            
            if (alpha >= beta) break;
        }
        
        return best;
    }
    
    string get_move_notation(const Move& m) {
        if (m.is_castle) {
            int rank = get_rank(m.from);
            if (m.to > m.from) return "O-O";
            else return "O-O-O";
        }
        
        char piece = board[m.from];
        if (piece == 0) piece = 'P'; // fallback
        
        char pchar = tolower(piece);
        if (toupper(piece) != 'P') pchar = toupper(piece);
        
        string from_sq = "";
        from_sq += 'a' + get_file(m.from);
        from_sq += '1' + get_rank(m.from);
        
        string to_sq = "";
        to_sq += 'a' + get_file(m.to);
        to_sq += '1' + get_rank(m.to);
        
        string notat = "";
        notat += pchar;
        notat += from_sq;
        
        if (m.is_capture || board[m.to] != 0) {
            notat += "x";
        }
        notat += to_sq;
        
        if (m.promotion) {
            notat += "/";
            notat += toupper(m.promotion);
        }
        
        if (m.is_en_passant) {
            notat += "ep";
        }
        
        // Check / mate detection omitted for simplicity (optional)
        
        return notat;
    }
    
    string find_best_move(double timeout) {
        timeout_seconds = timeout;
        start_time = chrono::steady_clock::now();
        
        best_move_found = Move();
        best_score_found = -INF;
        
        vector<Move> legal_moves;
        auto all_moves = generate_moves();
        
        for (auto& m : all_moves) {
            char captured = board[m.to];
            int prev_ep = en_passant_sq;
            int prev_cast = castling_rights;
            
            make_move(m);
            if (!in_check()) {
                legal_moves.push_back(m);
            }
            undo_move(m, captured, prev_ep, prev_cast);
        }
        
        if (legal_moves.empty()) {
            return "pe2e4"; // fallback
        }
        
        // Iterative deepening up to depth 5 or time out
        for (int d = 1; d <= 5; d++) {
            if (chrono::duration<double>(chrono::steady_clock::now() - start_time).count() > timeout - 0.5) break;
            
            best_score_found = -INF;
            for (auto& m : legal_moves) {
                char captured = board[m.to];
                int prev_ep = en_passant_sq;
                int prev_cast = castling_rights;
                
                make_move(m);
                int score = -alpha_beta(d - 1, -INF, INF);
                undo_move(m, captured, prev_ep, prev_cast);
                
                if (score > best_score_found) {
                    best_score_found = score;
                    best_move_found = m;
                }
                
                if (chrono::duration<double>(chrono::steady_clock::now() - start_time).count() > timeout - 0.3) break;
            }
        }
        
        if (best_move_found.from == -1) {
            // fallback to first legal
            best_move_found = legal_moves[0];
        }
        
        return get_move_notation(best_move_found);
    }
};

int main() {
    string timeout_line;
    getline(cin, timeout_line);
    double timeout = stod(timeout_line);
    
    string fen;
    getline(cin, fen);
    
    ChessEngine engine;
    engine.load_fen(fen);
    
    string move = engine.find_best_move(timeout);
    cout << move << endl;
    
    return 0;
}
```
