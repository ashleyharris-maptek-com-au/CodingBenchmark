```cpp
#include <vector>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <numeric>
#include <iomanip>

struct Point {
    double x, y;
    int id;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    
    int N;
    std::cin >> N;
    
    if (N == 0) {
        return 0;
    }
    
    std::vector<Point> pts(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> pts[i].x >> pts[i].y;
        pts[i].id = i;
    }
    
    std::vector<int> order;
    order.reserve(N);
    
    if (N <= 10000) {
        // Nearest Neighbor heuristic (good quality for medium sizes)
        std::vector<bool> visited(N, false);
        int current = 0;
        order.push_back(0);
        visited[0] = true;
        
        for (int i = 1; i < N; ++i) {
            int next = -1;
            double min_dist = 1e30;
            for (int j = 0; j < N; ++j) {
                if (!visited[j]) {
                    double dx = pts[j].x - pts[current].x;
                    double dy = pts[j].y - pts[current].y;
                    double dist = dx * dx + dy * dy;
                    if (dist < min_dist) {
                        min_dist = dist;
                        next = j;
                    }
                }
            }
            order.push_back(next);
            visited[next] = true;
            current = next;
        }
    } else {
        // Angular sort around centroid (fast O(N log N) for huge N)
        double cx = 0.0, cy = 0.0;
        for (const auto& p : pts) {
            cx += p.x;
            cy += p.y;
        }
        cx /= N;
        cy /= N;
        
        std::vector<Point> sorted_pts = pts;
        std::sort(sorted_pts.begin(), sorted_pts.end(),
            [cx, cy](const Point& a, const Point& b) {
                double ang_a = std::atan2(a.y - cy, a.x - cx);
                double ang_b = std::atan2(b.y - cy, b.x - cx);
                return ang_a < ang_b;
            });
        
        // Find position of city 0 and rotate
        size_t zero_pos = 0;
        for (size_t i = 0; i < sorted_pts.size(); ++i) {
            if (sorted_pts[i].id == 0) {
                zero_pos = i;
                break;
            }
        }
        
        order.resize(N);
        for (int i = 0; i < N; ++i) {
            order[i] = sorted_pts[(zero_pos + i) % N].id;
        }
    }
    
    // Output the tour (starts with 0)
    for (int i = 0; i < N; ++i) {
        if (i > 0) std::cout << ' ';
        std::cout << order[i];
    }
    std::cout << '\n';
    
    return 0;
}
```
