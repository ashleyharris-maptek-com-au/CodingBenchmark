```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <map>
#include <numeric>
#include <set>

using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    cin >> n >> m >> k;

    vector<vector<pair<int, ll>>> g(n);
    map<pair<int, int>, ll> edge_w;

    for (int i = 0; i < m; ++i) {
        int u, v;
        ll w;
        cin >> u >> v >> w;
        g[u].emplace_back(v, w);
        g[v].emplace_back(u, w);
        int a = min(u, v), b = max(u, v);
        if (edge_w.find({a, b}) == edge_w.end() || w < edge_w[{a, b}]) {
            edge_w[{a, b}] = w;
        }
    }

    vector<int> term(k);
    for (int i = 0; i < k; ++i) {
        cin >> term[i];
    }

    if (k <= 1) {
        cout << "0 0\n";
        return 0;
    }

    const ll INF = 1LL << 60;
    vector<vector<ll>> dist(k, vector<ll>(n, INF));
    vector<vector<int>> prevv(k, vector<int>(n, -1));

    for (int i = 0; i < k; ++i) {
        int src = term[i];
        vector<ll> d(n, INF);
        d[src] = 0;
        vector<int> pr(n, -1);
        priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
        pq.emplace(0, src);

        while (!pq.empty()) {
            auto [cost, u] = pq.top();
            pq.pop();
            if (cost > d[u]) continue;
            for (auto [v, w] : g[u]) {
                if (d[v] > d[u] + w) {
                    d[v] = d[u] + w;
                    pr[v] = u;
                    pq.emplace(d[v], v);
                }
            }
        }
        dist[i] = d;
        prevv[i] = pr;
    }

    struct TEdge {
        int u, v;
        ll w;
    };
    vector<TEdge> tedges;
    for (int i = 0; i < k; ++i) {
        for (int j = i + 1; j < k; ++j) {
            ll dd = dist[i][term[j]];
            if (dd < INF) {
                tedges.push_back({i, j, dd});
            }
        }
    }

    sort(tedges.begin(), tedges.end(), [](const TEdge& a, const TEdge& b) {
        return a.w < b.w;
    });

    vector<int> parent(k);
    iota(parent.begin(), parent.end(), 0);
    auto find_term = [&](auto& self, int x) -> int {
        return parent[x] == x ? x : parent[x] = self(self, parent[x]);
    };

    vector<pair<int, int>> mst_pairs;
    for (auto& e : tedges) {
        int pu = find_term(find_term, e.u);
        int pv = find_term(find_term, e.v);
        if (pu != pv) {
            parent[pu] = pv;
            mst_pairs.emplace_back(e.u, e.v);
        }
    }

    if (mst_pairs.size() != static_cast<size_t>(k - 1)) {
        // Not connected, fallback: output nothing or minimal
        cout << "0 0\n";
        return 0;
    }

    set<pair<int, int>> used_edges;
    for (auto [idx1, idx2] : mst_pairs) {
        int target = term[idx2];
        int cur = target;
        int src = term[idx1];
        while (cur != src) {
            int p = prevv[idx1][cur];
            if (p == -1) break;
            int a = min(p, cur);
            int b = max(p, cur);
            used_edges.emplace(a, b);
            cur = p;
        }
    }

    vector<tuple<ll, int, int>> union_edges;
    for (auto [u, v] : used_edges) {
        auto it = edge_w.find({u, v});
        if (it != edge_w.end()) {
            union_edges.emplace_back(it->second, u, v);
        }
    }

    sort(union_edges.begin(), union_edges.end());

    vector<int> par(n);
    iota(par.begin(), par.end(), 0);
    auto find = [&](auto& self, int x) -> int {
        return par[x] == x ? x : par[x] = self(self, par[x]);
    };

    vector<pair<int, int>> tree_edges;
    ll total_weight = 0;

    for (auto [w, u, v] : union_edges) {
        int pu = find(find, u);
        int pv = find(find, v);
        if (pu != pv) {
            par[pu] = pv;
            tree_edges.emplace_back(u, v);
            total_weight += w;
        }
    }

    // Verify terminals are connected (optional check, assume success)
    int root = find(find, term[0]);
    bool connected = true;
    for (int t : term) {
        if (find(find, t) != root) {
            connected = false;
            break;
        }
    }

    if (!connected || tree_edges.empty() && k > 1) {
        // Fallback: output 0 if failed
        cout << "0 0\n";
        return 0;
    }

    cout << total_weight << " " << tree_edges.size() << "\n";
    for (auto [u, v] : tree_edges) {
        cout << u << " " << v << "\n";
    }

    return 0;
}
```
