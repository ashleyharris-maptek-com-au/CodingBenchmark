#include <iostream>
#include <vector>
#include <algorithm>
#include <tuple>
#include <cstdint>

struct Rectangle {
    int w;
    int h;
    int id;
};

struct Level {
    int y;
    int height;
    int width_used;
};

struct Bin {
    std::vector<Level> levels;
};

struct Placement {
    int board;
    int x;
    int y;
    bool rotated;
};

bool try_place(Bin& bin, const Rectangle& rect, int board_w, int board_h, Placement& placement) {
    int orig_w = rect.w;
    int orig_h = rect.h;

    // Try existing levels - prefer non-rotated first
    for (size_t i = 0; i < bin.levels.size(); ++i) {
        Level& lvl = bin.levels[i];
        int remaining = board_w - lvl.width_used;

        // Try non-rotated
        if (orig_w <= remaining && orig_h <= lvl.height) {
            placement.x = lvl.width_used;
            placement.y = lvl.y;
            placement.rotated = false;
            lvl.width_used += orig_w;
            return true;
        }
        // Try rotated
        if (orig_h <= remaining && orig_w <= lvl.height) {
            placement.x = lvl.width_used;
            placement.y = lvl.y;
            placement.rotated = true;
            lvl.width_used += orig_h;
            return true;
        }
    }

    // Try to create a new level
    int new_y = 0;
    if (!bin.levels.empty()) {
        new_y = bin.levels.back().y + bin.levels.back().height;
    }

    // Try both orientations for new level
    for (int rot = 0; rot < 2; ++rot) {
        int rw = (rot == 0) ? orig_w : orig_h;
        int rh = (rot == 0) ? orig_h : orig_w;

        if (rw > board_w) continue;
        if (rh > board_h - new_y) continue;

        // Create new level
        Level new_level;
        new_level.y = new_y;
        new_level.height = rh;
        new_level.width_used = rw;

        placement.x = 0;
        placement.y = new_y;
        placement.rotated = (rot == 1);

        bin.levels.push_back(new_level);
        return true;
    }

    return false;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N, board_w, board_h;
    std::cin >> N >> board_w >> board_h;

    std::vector<Rectangle> rects(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> rects[i].w >> rects[i].h;
        rects[i].id = i;
    }

    // Sort by decreasing max dimension (largest side first)
    std::sort(rects.begin(), rects.end(), [](const Rectangle& a, const Rectangle& b) {
        return std::max(a.w, a.h) > std::max(b.w, b.h);
    });

    std::vector<Bin> bins;
    std::vector<Placement> placements(N);

    for (const auto& rect : rects) {
        bool placed = false;

        for (size_t b = 0; b < bins.size(); ++b) {
            if (try_place(bins[b], rect, board_w, board_h, placements[rect.id])) {
                placements[rect.id].board = static_cast<int>(b);
                placed = true;
                break;
            }
        }

        if (!placed) {
            // Open new bin
            Bin new_bin;
            // Place first rectangle in new bin
            int orig_w = rect.w;
            int orig_h = rect.h;
            bool success = false;

            // Try non-rotated first
            if (orig_w <= board_w && orig_h <= board_h) {
                Level first_level;
                first_level.y = 0;
                first_level.height = orig_h;
                first_level.width_used = orig_w;

                new_bin.levels.push_back(first_level);

                placements[rect.id].board = static_cast<int>(bins.size());
                placements[rect.id].x = 0;
                placements[rect.id].y = 0;
                placements[rect.id].rotated = false;
                success = true;
            } else if (orig_h <= board_w && orig_w <= board_h) {
                // Rotated
                Level first_level;
                first_level.y = 0;
                first_level.height = orig_w;
                first_level.width_used = orig_h;

                new_bin.levels.push_back(first_level);

                placements[rect.id].board = static_cast<int>(bins.size());
                placements[rect.id].x = 0;
                placements[rect.id].y = 0;
                placements[rect.id].rotated = true;
                success = true;
            }

            if (success) {
                bins.push_back(std::move(new_bin));
            } else {
                // Should not happen if input is valid
                std::cerr << "Error: rectangle too large for board\n";
                return 1;
            }
        }
    }

    // Output
    std::cout << bins.size() << '\n';

    for (int i = 0; i < N; ++i) {
        const auto& p = placements[i];
        std::cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << (p.rotated ? 1 : 0) << '\n';
    }

    return 0;
}
