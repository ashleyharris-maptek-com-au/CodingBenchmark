```rust
use std::io::{self, BufRead, BufReader, Write};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let mut reader = BufReader::with_capacity(1 << 20, stdin.lock());
    
    let mut line = String::new();
    if reader.read_line(&mut line).is_err() {
        println!("0");
        return;
    }
    
    let mut parts = line.trim().split_whitespace();
    let h: usize = match parts.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let w: usize = match parts.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    
    if h == 0 || w == 0 {
        println!("0");
        return;
    }
    
    let mut grid = vec![0u8; h * w];
    let mut start: Option<(usize, usize)> = None;
    let mut goal: Option<(usize, usize)> = None;
    
    for y in 0..h {
        line.clear();
        if reader.read_line(&mut line).is_err() {
            println!("0");
            return;
        }
        let bytes = line.trim_end().as_bytes();
        if bytes.len() < w {
            println!("0");
            return;
        }
        for x in 0..w {
            let idx = y * w + x;
            let c = bytes[x];
            grid[idx] = c;
            if c == b'A' {
                start = Some((y, x));
            } else if c == b'B' {
                goal = Some((y, x));
            }
        }
    }
    
    let start = match start {
        Some(s) => s,
        None => {
            println!("0");
            return;
        }
    };
    let goal = match goal {
        Some(g) => g,
        None => {
            println!("0");
            return;
        }
    };
    
    if start == goal {
        println!("1");
        println!("{} {}", start.1, start.0);
        return;
    }
    
    let total = h * w;
    let mut visited = vec![false; total];
    let mut parent = vec![usize::MAX; total];
    let mut q: VecDeque<(usize, usize)> = VecDeque::new();
    
    let start_idx = start.0 * w + start.1;
    let goal_idx = goal.0 * w + goal.1;
    
    visited[start_idx] = true;
    parent[start_idx] = start_idx;
    q.push_back(start);
    
    let dirs: [(isize, isize); 4] = [(-1, 0), (1, 0), (0, -1), (0, 1)];
    let mut found = false;
    
    while let Some((y, x)) = q.pop_front() {
        if (y, x) == goal {
            found = true;
            break;
        }
        
        for &(dy, dx) in &dirs {
            let ny = y.wrapping_add_signed(dy);
            let nx = x.wrapping_add_signed(dx);
            
            if ny < h && nx < w {
                let nidx = ny * w + nx;
                if !visited[nidx] && grid[nidx] != b'#' {
                    visited[nidx] = true;
                    parent[nidx] = y * w + x;
                    q.push_back((ny, nx));
                }
            }
        }
    }
    
    if !found && !visited[goal_idx] {
        println!("0");
        return;
    }
    
    // Reconstruct path
    let mut path = Vec::new();
    let mut curr = goal_idx;
    
    loop {
        let y = curr / w;
        let x = curr % w;
        path.push((x, y)); // (x, y) where x=col, y=row
        if curr == start_idx {
            break;
        }
        curr = parent[curr];
        if curr == usize::MAX {
            println!("0");
            return;
        }
    }
    
    path.reverse();
    
    let stdout = io::stdout();
    let mut out = io::BufWriter::with_capacity(1 << 20, stdout.lock());
    
    writeln!(out, "{}", path.len()).unwrap();
    for (x, y) in path {
        writeln!(out, "{} {}", x, y).unwrap();
    }
}
```
