```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 100
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_color Location 2
OpDecorate %out_color Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%float_h = OpConstant %float 0.05

; gl_PerVertex input block (array size 1 for InputPoints)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_Input_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in Input

; User input: color (location 2)
%arr_v3float = OpTypeArray %v3float %uint_1
%ptr_Input_arr_v3float = OpTypePointer Input %arr_v3float
%in_color = OpVariable %ptr_Input_arr_v3float Input

; gl_PerVertex output block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output

; Output color
%ptr_Output_v3float = OpTypePointer Output %v3float
%out_color = OpVariable %ptr_Output_v3float Output

; Pointer types for access
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float

; Function
%func_type = OpTypeFunction %void
%main = OpFunction %void None %func_type
%entry = OpLabel

; Load input gl_Position (gl_in[0].gl_Position)
%gl_in_pos_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%pos_in = OpLoad %v4float %gl_in_pos_ptr

; Extract components (keep z and w)
%pos_x = OpCompositeExtract %float %pos_in 0
%pos_y = OpCompositeExtract %float %pos_in 1
%pos_z = OpCompositeExtract %float %pos_in 2
%pos_w = OpCompositeExtract %float %pos_in 3

; Load input color
%color_ptr = OpAccessChain %ptr_Input_v3float %in_color %int_0
%vertex_color = OpLoad %v3float %color_ptr

; Precompute gl_out position pointer
%gl_out_pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0

; ========== Vertex 0: Top (x, y+h) ==========
%top_y = OpFAdd %float %pos_y %float_h
%pos_top = OpCompositeConstruct %v4float %pos_x %top_y %pos_z %pos_w
OpStore %gl_out_pos_ptr %pos_top
OpStore %out_color %vertex_color
OpEmitVertex

; ========== Vertex 1: Right (x+h, y) ==========
%right_x = OpFAdd %float %pos_x %float_h
%pos_right = OpCompositeConstruct %v4float %right_x %pos_y %pos_z %pos_w
OpStore %gl_out_pos_ptr %pos_right
OpStore %out_color %vertex_color
OpEmitVertex

; ========== Vertex 2: Left (x-h, y) ==========
%left_x = OpFSub %float %pos_x %float_h
%pos_left = OpCompositeConstruct %v4float %left_x %pos_y %pos_z %pos_w
OpStore %gl_out_pos_ptr %pos_left
OpStore %out_color %vertex_color
OpEmitVertex

; ========== Vertex 3: Bottom (x, y-h) ==========
%bot_y = OpFSub %float %pos_y %float_h
%pos_bot = OpCompositeConstruct %v4float %pos_x %bot_y %pos_z %pos_w
OpStore %gl_out_pos_ptr %pos_bot
OpStore %out_color %vertex_color
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
