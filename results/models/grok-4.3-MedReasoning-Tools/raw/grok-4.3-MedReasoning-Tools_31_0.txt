use std::io::{self, Read};

fn generate_sums(arr: &[i64]) -> Vec<(i64, u32)> {
    let m = arr.len();
    let mut sums = Vec::with_capacity(1 << m);
    sums.push((0i64, 0u32));
    for (i, &val) in arr.iter().enumerate() {
        let len = sums.len();
        for j in 0..len {
            let (s, mask) = sums[j];
            sums.push((s + val, mask | (1u32 << i)));
        }
    }
    sums
}

fn find_subset(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    if target == 0 {
        return Some(vec![]);
    }
    for (i, &x) in nums.iter().enumerate() {
        if x == target {
            return Some(vec![i]);
        }
    }
    if n <= 40 {
        let mid = n / 2;
        let (left, right) = nums.split_at(mid);
        let mut left_sums = generate_sums(left);
        let mut right_sums = generate_sums(right);
        left_sums.sort_unstable_by_key(|k| k.0);
        right_sums.sort_unstable_by_key(|k| k.0);
        for &(lsum, lmask) in &left_sums {
            let needed = target - lsum;
            if let Ok(idx) = right_sums.binary_search_by_key(&needed, |&(s, _)| s) {
                let (_, rmask) = right_sums[idx];
                let mut subset = Vec::new();
                for i in 0..mid {
                    if (lmask & (1 << i)) != 0 {
                        subset.push(i);
                    }
                }
                for i in 0..(n - mid) {
                    if (rmask & (1 << i)) != 0 {
                        subset.push(mid + i);
                    }
                }
                return Some(subset);
            }
        }
        None
    } else {
        // Fallback for large N: check if total sum matches target
        let mut total: i64 = 0;
        let mut all_indices: Vec<usize> = Vec::with_capacity(n);
        for (i, &x) in nums.iter().enumerate() {
            total += x;
            all_indices.push(i);
        }
        if total == target {
            return Some(all_indices);
        }
        None
    }
}

fn main() {
    let mut input = String::new();
    io::stdin()
        .read_to_string(&mut input)
        .expect("Failed to read input");
    let mut iter = input.split_whitespace();
    let n: usize = match iter.next() {
        Some(s) => s.parse().expect("Failed to parse n"),
        None => {
            println!("NO");
            return;
        }
    };
    let target: i64 = match iter.next() {
        Some(s) => s.parse().expect("Failed to parse target"),
        None => {
            println!("NO");
            return;
        }
    };
    let nums: Vec<i64> = iter
        .map(|s| s.parse().expect("Failed to parse number"))
        .collect();
    if nums.len() != n {
        println!("NO");
        return;
    }
    match find_subset(&nums, target) {
        Some(idxs) => {
            println!("YES");
            if !idxs.is_empty() {
                let line: Vec<String> = idxs.iter().map(|x| x.to_string()).collect();
                println!("{}", line.join(" "));
            }
        }
        None => {
            println!("NO");
        }
    }
}
