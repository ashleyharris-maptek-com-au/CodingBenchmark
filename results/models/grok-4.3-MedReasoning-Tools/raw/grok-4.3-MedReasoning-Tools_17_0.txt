```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public class Program
{
    struct Box
    {
        public int Index;
        public long W, H, D;
    }

    struct Placement
    {
        public int Index;
        public long X, Y, Z, W, H, D;
    }

    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split();
        int N = int.Parse(firstLine[0]);
        long W = long.Parse(firstLine[1]);
        long H = long.Parse(firstLine[2]);
        long D = long.Parse(firstLine[3]);

        List<Box> boxes = new List<Box>(N);
        for (int i = 0; i < N; i++)
        {
            string[] parts = Console.ReadLine().Split();
            long w = long.Parse(parts[0]);
            long h = long.Parse(parts[1]);
            long d = long.Parse(parts[2]);
            boxes.Add(new Box { Index = i, W = w, H = h, D = d });
        }

        // Sort by largest dimension descending (safe, no overflow)
        boxes.Sort((a, b) =>
        {
            long ma = Math.Max(Math.Max(a.W, a.H), a.D);
            long mb = Math.Max(Math.Max(b.W, b.H), b.D);
            return mb.CompareTo(ma);
        });

        List<Placement> packed = new List<Placement>();
        List<(long x, long y, long z)> candidates = new List<(long x, long y, long z)>();
        HashSet<(long, long, long)> candSet = new HashSet<(long, long, long)>();
        candidates.Add((0, 0, 0));
        candSet.Add((0, 0, 0));

        foreach (var box in boxes)
        {
            long bestX = -1, bestY = -1, bestZ = -1;
            long bestW = 0, bestH = 0, bestD = 0;
            long bestScoreZ = long.MaxValue;
            long bestScoreY = long.MaxValue;
            long bestScoreX = long.MaxValue;

            long[] dims = { box.W, box.H, box.D };

            // All 6 orientations
            for (int o = 0; o < 6; o++)
            {
                long ow, oh, od;
                if (o == 0) { ow = dims[0]; oh = dims[1]; od = dims[2]; }
                else if (o == 1) { ow = dims[0]; oh = dims[2]; od = dims[1]; }
                else if (o == 2) { ow = dims[1]; oh = dims[0]; od = dims[2]; }
                else if (o == 3) { ow = dims[1]; oh = dims[2]; od = dims[0]; }
                else if (o == 4) { ow = dims[2]; oh = dims[0]; od = dims[1]; }
                else { ow = dims[2]; oh = dims[1]; od = dims[0]; }

                if (ow > W || oh > H || od > D) continue;

                foreach (var cand in candidates)
                {
                    long x = cand.x;
                    long y = cand.y;
                    long z = cand.z;

                    if (x < 0 || y < 0 || z < 0) continue;
                    if (x + ow > W || y + oh > H || z + od > D) continue;

                    bool collision = false;
                    foreach (var p in packed)
                    {
                        if (!(p.X + p.W <= x || x + ow <= p.X ||
                              p.Y + p.H <= y || y + oh <= p.Y ||
                              p.Z + p.D <= z || z + od <= p.Z))
                        {
                            collision = true;
                            break;
                        }
                    }

                    if (!collision)
                    {
                        // Prefer lowest Z, then lowest Y, then lowest X
                        if (z < bestScoreZ ||
                            (z == bestScoreZ && y < bestScoreY) ||
                            (z == bestScoreZ && y == bestScoreY && x < bestScoreX))
                        {
                            bestScoreZ = z;
                            bestScoreY = y;
                            bestScoreX = x;
                            bestX = x;
                            bestY = y;
                            bestZ = z;
                            bestW = ow;
                            bestH = oh;
                            bestD = od;
                        }
                    }
                }
            }

            if (bestX >= 0)
            {
                Placement place = new Placement
                {
                    Index = box.Index,
                    X = bestX,
                    Y = bestY,
                    Z = bestZ,
                    W = bestW,
                    H = bestH,
                    D = bestD
                };
                packed.Add(place);

                // Add new extreme points
                var np1 = (bestX + bestW, bestY, bestZ);
                var np2 = (bestX, bestY + bestH, bestZ);
                var np3 = (bestX, bestY, bestZ + bestD);

                if (np1.Item1 <= W && np1.Item2 <= H && np1.Item3 <= D && candSet.Add(np1))
                    candidates.Add(np1);

                if (np2.Item1 <= W && np2.Item2 <= H && np2.Item3 <= D && candSet.Add(np2))
                    candidates.Add(np2);

                if (np3.Item1 <= W && np3.Item2 <= H && np3.Item3 <= D && candSet.Add(np3))
                    candidates.Add(np3);
            }
        }

        // Output
        Console.WriteLine(packed.Count);
        foreach (var p in packed)
        {
            Console.WriteLine($"{p.Index} {p.X} {p.Y} {p.Z}");
        }
    }
}
```
