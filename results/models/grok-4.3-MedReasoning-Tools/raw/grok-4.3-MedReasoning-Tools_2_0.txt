using System;
using System.Collections.Generic;
using System.Text;

class Program
{
    static void Main()
    {
        string line = Console.ReadLine();
        if (line == null) return;
        string[] parts = line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(parts[0]);
        int M = int.Parse(parts[1]);

        List<List<int>> adj = new List<List<int>>(N);
        List<List<int>> graph = new List<List<int>>(N);
        for (int i = 0; i < N; i++)
        {
            adj.Add(new List<int>());
            graph.Add(new List<int>());
        }

        for (int i = 0; i < M; i++)
        {
            line = Console.ReadLine();
            if (line == null) break;
            parts = line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length < 2) continue;
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            adj[u].Add(v);
            adj[v].Add(u);
            graph[u].Add(v);
            graph[v].Add(u);
        }

        // Compute initial degrees
        int[] degree = new int[N];
        for (int i = 0; i < N; i++)
        {
            degree[i] = adj[i].Count;
        }

        List<int> odds = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if (degree[i] % 2 == 1)
                odds.Add(i);
        }

        // Make Eulerian by adding duplicate paths from odds to 0 using spanning tree from 0
        if (odds.Count > 0)
        {
            int[] parent = new int[N];
            Array.Fill(parent, -1);
            bool[] visited = new bool[N];
            Queue<int> q = new Queue<int>();
            q.Enqueue(0);
            visited[0] = true;
            parent[0] = -2;

            while (q.Count > 0)
            {
                int u = q.Dequeue();
                foreach (int v in graph[u])
                {
                    if (!visited[v])
                    {
                        visited[v] = true;
                        parent[v] = u;
                        q.Enqueue(v);
                    }
                }
            }

            foreach (int o in odds)
            {
                if (o == 0 || !visited[o]) continue;
                int cur = o;
                while (cur != 0)
                {
                    int p = parent[cur];
                    if (p == -1) break;
                    adj[p].Add(cur);
                    adj[cur].Add(p);
                    cur = p;
                }
            }
        }

        // Hierholzer's algorithm
        int[] pos = new int[N];
        List<int> circuit = new List<int>();
        Stack<int> st = new Stack<int>();
        st.Push(0);

        while (st.Count > 0)
        {
            int u = st.Peek();
            if (pos[u] < adj[u].Count)
            {
                int v = adj[u][pos[u]];
                pos[u]++;
                st.Push(v);
            }
            else
            {
                circuit.Add(st.Pop());
            }
        }

        circuit.Reverse();

        // Output the route
        if (circuit.Count == 0)
        {
            Console.WriteLine("0");
            return;
        }

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < circuit.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(circuit[i]);
        }
        Console.WriteLine(sb.ToString());
    }
}
