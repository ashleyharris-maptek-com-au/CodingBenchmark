using System;
using System.Collections.Generic;
using System.Linq;

public struct Vector3
{
    public double X, Y, Z;

    public Vector3(double x, double y, double z)
    {
        X = x; Y = y; Z = z;
    }

    public static Vector3 operator -(Vector3 a, Vector3 b)
    {
        return new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    }

    public static Vector3 operator +(Vector3 a, Vector3 b)
    {
        return new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    }

    public double Magnitude()
    {
        return Math.Sqrt(X * X + Y * Y + Z * Z);
    }
}

public class State
{
    public Vector3 Pos;
    public Vector3 Vel;
}

public class OrbitalTSP
{
    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(firstLine[0]);
        double mu = double.Parse(firstLine[1]);

        State start = ReadState();
        List<State> stations = new List<State>();
        for (int i = 0; i < N; i++)
        {
            stations.Add(ReadState());
        }

        if (N == 0)
        {
            Console.WriteLine();
            return;
        }

        double[] costFromStart = new double[N];
        double[,] dist = new double[N, N];

        for (int i = 0; i < N; i++)
        {
            costFromStart[i] = (start.Pos - stations[i].Pos).Magnitude();
        }

        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                if (i != j)
                    dist[i, j] = (stations[i].Pos - stations[j].Pos).Magnitude();
                else
                    dist[i, j] = 0;
            }
        }

        List<int> bestPath;

        if (N <= 9)
        {
            bestPath = SolveExact(N, costFromStart, dist);
        }
        else
        {
            bestPath = SolveHeuristic(N, costFromStart, dist);
        }

        Console.WriteLine(string.Join(" ", bestPath));
    }

    static State ReadState()
    {
        string[] parts = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        return new State
        {
            Pos = new Vector3(double.Parse(parts[0]), double.Parse(parts[1]), double.Parse(parts[2])),
            Vel = new Vector3(double.Parse(parts[3]), double.Parse(parts[4]), double.Parse(parts[5]))
        };
    }

    // Exact solver using backtracking (pruned)
    static List<int> SolveExact(int N, double[] costFromStart, double[,] dist)
    {
        double bestCost = double.MaxValue;
        List<int> bestPath = null;

        bool[] used = new bool[N];
        List<int> current = new List<int>(N);

        void Search(int depth, double costSoFar, int last)
        {
            if (costSoFar >= bestCost) return;

            if (depth == N)
            {
                if (costSoFar < bestCost)
                {
                    bestCost = costSoFar;
                    bestPath = new List<int>(current);
                }
                return;
            }

            for (int i = 0; i < N; i++)
            {
                if (!used[i])
                {
                    double addCost = (depth == 0) ? costFromStart[i] : dist[last, i];
                    double newCost = costSoFar + addCost;

                    if (newCost < bestCost)
                    {
                        used[i] = true;
                        current.Add(i);
                        Search(depth + 1, newCost, i);
                        current.RemoveAt(current.Count - 1);
                        used[i] = false;
                    }
                }
            }
        }

        Search(0, 0.0, -1);
        return bestPath ?? Enumerable.Range(0, N).ToList();
    }

    // Heuristic: Nearest Neighbor + 2-Opt
    static List<int> SolveHeuristic(int N, double[] costFromStart, double[,] dist)
    {
        // Nearest Neighbor
        List<int> path = new List<int>(N);
        bool[] visited = new bool[N];
        int current = -1;

        for (int k = 0; k < N; k++)
        {
            int next = -1;
            double minD = double.MaxValue;

            for (int i = 0; i < N; i++)
            {
                if (!visited[i])
                {
                    double d = (current == -1) ? costFromStart[i] : dist[current, i];
                    if (d < minD)
                    {
                        minD = d;
                        next = i;
                    }
                }
            }

            if (next == -1) break;

            path.Add(next);
            visited[next] = true;
            current = next;
        }

        // 2-Opt improvement
        double currentCost = ComputePathCost(path, costFromStart, dist);
        bool improved = true;
        int maxIterations = 200;

        while (improved && maxIterations-- > 0)
        {
            improved = false;

            for (int i = 0; i < N - 1; i++)
            {
                for (int j = i + 2; j < N; j++)
                {
                    // Try reversing segment [i..j]
                    List<int> newPath = new List<int>(path);
                    newPath.Reverse(i, j - i + 1);

                    double newCost = ComputePathCost(newPath, costFromStart, dist);

                    if (newCost < currentCost)
                    {
                        path = newPath;
                        currentCost = newCost;
                        improved = true;
                    }
                }
            }
        }

        return path;
    }

    static double ComputePathCost(List<int> path, double[] costFromStart, double[,] dist)
    {
        if (path.Count == 0) return 0;

        double cost = costFromStart[path[0]];

        for (int i = 0; i < path.Count - 1; i++)
        {
            cost += dist[path[i], path[i + 1]];
        }

        return cost;
    }
}
