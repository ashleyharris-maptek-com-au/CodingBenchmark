```python
import sys

sys.setrecursionlimit(100000)

def main():
    input = sys.stdin.read
    data = input().split()
    it = iter(data)
    try:
        N = int(next(it))
        M = int(next(it))
    except:
        print("SAT")
        print("")
        return

    clauses = []
    for _ in range(M):
        try:
            a = int(next(it))
            b = int(next(it))
            c = int(next(it))
            clauses.append((a, b, c))
        except:
            break

    if N == 0:
        print("SAT")
        print("")
        return

    assign = [None] * (N + 1)

    def check_conflict():
        for clause in clauses:
            all_assigned_false = True
            for lit in clause:
                v = abs(lit)
                if assign[v] is None:
                    all_assigned_false = False
                    break
                if (lit > 0 and assign[v]) or (lit < 0 and not assign[v]):
                    all_assigned_false = False
                    break
            if all_assigned_false:
                return True
        return False

    def solve(pos):
        if pos == N + 1:
            return True

        for val in (False, True):
            assign[pos] = val
            if not check_conflict():
                if solve(pos + 1):
                    return True
            assign[pos] = None
        return False

    if solve(1):
        print("SAT")
        result = []
        for i in range(1, N + 1):
            result.append("1" if assign[i] else "0")
        print(" ".join(result))
    else:
        # Should not happen as per problem guarantee
        print("SAT")
        print(" ".join(["0"] * N))

if __name__ == "__main__":
    main()
```
