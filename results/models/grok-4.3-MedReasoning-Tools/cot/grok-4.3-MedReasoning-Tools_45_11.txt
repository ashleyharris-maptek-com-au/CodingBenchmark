; Minimal SPIR-V for polka dot fragment

. capability Shader

. extinst import "GLSL.std.450" as glsl

. memory Model Logical GLSL450

. entryPoint Fragment %main " " %out %uv %normal %worldPos

. executionMode %main OriginUpperLeft

; decorations

. decorate %out Location 0

. decorate %uv Location 2

. decorate %normal Location 1

. decorate %worldPos Location 0

. decorate %ubo DescriptorSet 0 Binding 0 Block

. memberDecorate %ubo 0 Offset 0

. memberDecorate %ubo 1 Offset 64

. memberDecorate %ubo 2 Offset 128
