out vec4 outColor; //loc0

uniform UBO { mat4 model,view,proj; vec4 lightPos,c ameraPos,params; } ubo;

main() {

  vec2 uvf = uv * 8.0;

  vec2 flooruv = floor(uvf);

  float shade = 0.5 + 0.5 * diff;

  float finalcol = col * shade;

  outColor = vec4(finalcol, finalcol, finalcol, 1.0);

}
