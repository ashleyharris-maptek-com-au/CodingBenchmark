import struct

words = []

def add_word(w):

    words.append(w)

def add_string(s):

    # add bytes of string padded to multiple of 4

    b = s.encode('utf-8') + b'\x00'

    while len(b) % 4 != 0:

        b += b'\x00'

    for i in range(0, len(b), 4):

        chunk = b[i:i+4]

        w = struct.unpack('<I', chunk)[0]

        add_word(w)
