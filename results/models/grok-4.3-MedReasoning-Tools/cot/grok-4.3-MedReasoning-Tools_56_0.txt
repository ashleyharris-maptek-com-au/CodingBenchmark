function getCentroid(group) {

  if (group.length === 0) return {x:0, y:0};

  let sumX = 0, sumY = 0;

  for (let p of group) {

    sumX += p.x;

    sumY += p.y;

  }

  return {x: sumX / group.length, y: sumY / group.length};

}
