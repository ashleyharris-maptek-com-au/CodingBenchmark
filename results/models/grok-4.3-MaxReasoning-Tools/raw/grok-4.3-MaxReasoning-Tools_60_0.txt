        • Multi-machine list scheduling with release times<br>
        • Critical path + earliest deadline + weight priority<br>
        • Handles machine idle time correctly<br>

<pre><code>const fs = require('fs');

// Read input from stdin
const input = fs.readFileSync(0, 'utf8').trim();
let data;
try {
    data = JSON.parse(input);
} catch (e) {
    console.error("Invalid JSON input");
    process.exit(1);
}

const machineCount = data.machineCount || 1;
const tasks = data.tasks || [];

if (tasks.length === 0) {
    console.log(JSON.stringify({ schedule: [] }));
    process.exit(0);
}

// Data structures
const taskMap = new Map();
const durationMap = new Map();
const deadlineMap = new Map();
const weightMap = new Map();
const successors = new Map();
const indegree = new Map();
const releaseTime = new Map();

for (let task of tasks) {
    const id = task.id;
    taskMap.set(id, task);
    durationMap.set(id, task.duration || 0);
    deadlineMap.set(id, task.deadline !== undefined ? task.deadline : Infinity);
    weightMap.set(id, task.weight || 1);
    releaseTime.set(id, 0);
    indegree.set(id, (task.deps && Array.isArray(task.deps)) ? task.deps.length : 0);
    successors.set(id, []);
}

// Build successors graph
for (let task of tasks) {
    const id = task.id;
    if (task.deps && Array.isArray(task.deps)) {
        for (let dep of task.deps) {
            if (!successors.has(dep)) {
                successors.set(dep, []);
            }
            successors.get(dep).push(id);
        }
    }
}

// Compute critical path (longest path sum of durations)
const critical = new Map();

function getCritical(id) {
    if (critical.has(id)) return critical.get(id);
    let maxLen = durationMap.get(id) || 0;
    const succs = successors.get(id) || [];
    for (let s of succs) {
        maxLen = Math.max(maxLen, (durationMap.get(id) || 0) + getCritical(s));
    }
    critical.set(id, maxLen);
    return maxLen;
}

for (let task of tasks) {
    getCritical(task.id);
}

// Initialize ready tasks
let ready = [];
for (let [id, deg] of indegree) {
    if (deg === 0) {
        ready.push(id);
    }
}

// Machine availability times
const machineFree = new Array(machineCount).fill(0);
const schedule = [];

// Helper: select best task from a list
function selectBestFromList(list) {
    if (!list || list.length === 0) return null;
    
    let bestId = list[0];
    let bestCrit = critical.get(bestId) || 0;
    let bestDl = deadlineMap.get(bestId) || Infinity;
    let bestW = weightMap.get(bestId) || 0;

    for (let i = 1; i &lt; list.length; i++) {
        const id = list[i];
        const crit = critical.get(id) || 0;
        const dl = deadlineMap.get(id) || Infinity;
        const w = weightMap.get(id) || 0;

        let better = false;
        if (crit &gt; bestCrit) {
            better = true;
        } else if (crit === bestCrit) {
            if (dl &lt; bestDl) {
                better = true;
            } else if (dl === bestDl &amp;&amp; w &gt; bestW) {
                better = true;
            }
        }
        if (better) {
            bestId = id;
            bestCrit = crit;
            bestDl = dl;
            bestW = w;
        }
    }
    return bestId;
}

// Main scheduling loop
let scheduledCount = 0;
const totalTasks = tasks.length;

while (ready.length &gt; 0 &amp;&amp; scheduledCount &lt; totalTasks) {
    // Find machine with earliest availability
    let minM = 0;
    let avail = machineFree[0];
    for (let m = 1; m &lt; machineCount; m++) {
        if (machineFree[m] &lt; avail) {
            avail = machineFree[m];
            minM = m;
        }
    }

    // Find candidates that can start at or before avail
    let candidates = [];
    let minR = Infinity;
    
    for (let id of ready) {
        const r = releaseTime.get(id) || 0;
        if (r &lt;= avail) {
            candidates.push(id);
        } else if (r &lt; minR) {
            minR = r;
        }
    }

    let chosenId;
    let start;

    if (candidates.length &gt; 0) {
        // Schedule a ready task now
        chosenId = selectBestFromList(candidates);
        start = avail;
    } else {
        // All ready tasks have future release times — wait for the earliest
        minR = Infinity;
        for (let id of ready) {
            const r = releaseTime.get(id) || 0;
            if (r &lt; minR) minR = r;
        }
        
        candidates = [];
        for (let id of ready) {
            if ((releaseTime.get(id) || 0) &lt;= minR) {
                candidates.push(id);
            }
        }
        chosenId = selectBestFromList(candidates);
        start = minR;
    }

    if (!chosenId) break;

    // Remove from ready
    const idx = ready.indexOf(chosenId);
    if (idx !== -1) ready.splice(idx, 1);

    const dur = durationMap.get(chosenId) || 0;
    const finish = start + dur;

    // Record schedule
    schedule.push({
        id: chosenId,
        start: start,
        machine: minM
    });

    // Update machine availability
    machineFree[minM] = finish;

    // Update successors
    const succs = successors.get(chosenId) || [];
    for (let s of succs) {
        const currRel = releaseTime.get(s) || 0;
        if (finish &gt; currRel) {
            releaseTime.set(s, finish);
        }
        let deg = indegree.get(s) || 0;
        deg--;
        if (deg &lt; 0) deg = 0;
        indegree.set(s, deg);
        
        if (deg === 0) {
            ready.push(s);
        }
    }

    scheduledCount++;
}

// Output
const output = {
    schedule: schedule
};

console.log(JSON.stringify(output));
