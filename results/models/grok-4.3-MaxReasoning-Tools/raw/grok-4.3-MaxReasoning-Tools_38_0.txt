import sys
from collections import defaultdict

def find_exact_cover(sets, U):
    n = len(sets)
    elem_to_sets = defaultdict(list)
    for i in range(n):
        for e in sets[i]:
            elem_to_sets[e].append(i)
    
    def backtrack(remaining, chosen, used_sets):
        if not remaining:
            return chosen[:]
        
        # MRV heuristic: choose element with fewest options
        best_elem = None
        min_options = float('inf')
        for e in remaining:
            options = 0
            for si in elem_to_sets[e]:
                if si in used_sets:
                    continue
                s = sets[si]
                if all(ee in remaining for ee in s):
                    options += 1
            if options < min_options:
                min_options = options
                best_elem = e
                if min_options == 0:
                    break
        
        if min_options == 0:
            return None
        
        elem = best_elem
        for set_idx in elem_to_sets[elem]:
            if set_idx in used_sets:
                continue
            s = sets[set_idx]
            if all(e in remaining for e in s):
                new_remaining = remaining - set(s)
                used_sets.add(set_idx)
                chosen.append(set_idx)
                res = backtrack(new_remaining, chosen, used_sets)
                if res is not None:
                    return res
                chosen.pop()
                used_sets.remove(set_idx)
        return None
    
    remaining = set(range(U))
    chosen = []
    used_sets = set()
    solution = backtrack(remaining, chosen, used_sets)
    if solution is not None:
        solution.sort()
    return solution

def main():
    data = sys.stdin.read().strip().splitlines()
    if not data:
        print("NO SOLUTION")
        return
    
    first_line = data[0].strip().split()
    if len(first_line) < 2:
        print("NO SOLUTION")
        return
    
    try:
        U = int(first_line[0])
        S = int(first_line[1])
    except ValueError:
        print("NO SOLUTION")
        return
    
    sets = []
    for i in range(S):
        line_idx = 1 + i
        if line_idx < len(data):
            line = data[line_idx].strip()
            els = []
            seen = set()
            for token in line.split():
                try:
                    e = int(token)
                    if 0 <= e < U and e not in seen:
                        seen.add(e)
                        els.append(e)
                except ValueError:
                    continue
            sets.append(els)
        else:
            sets.append([])
    
    solution = find_exact_cover(sets, U)
    
    if solution is not None:
        print("SOLUTION")
        print(" ".join(map(str, solution)))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    main()
