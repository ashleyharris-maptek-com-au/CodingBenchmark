```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
#include <random>
#include <chrono>
#include <cstdint>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, m, k;
    std::cin >> n >> m >> k;

    std::vector<std::vector<int>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        std::cin >> u >> v;
        if (u != v && u >= 0 && u < n && v >= 0 && v < n) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }

    // Deduplicate and sort adjacency lists
    for (auto& lis : adj) {
        std::sort(lis.begin(), lis.end());
        auto last = std::unique(lis.begin(), lis.end());
        lis.erase(last, lis.end());
    }

    std::vector<int> col(n, -1);

    // Strategy selector based on graph complexity
    auto try_coloring = [&](const std::vector<int>& ord) -> bool {
        std::fill(col.begin(), col.end(), -1);
        for (int idx = 0; idx < n; ++idx) {
            int v = ord[idx];
            std::vector<char> used(k, 0);
            for (int nei : adj[v]) {
                if (col[nei] != -1) {
                    int c = col[nei];
                    if (c >= 0 && c < k) used[c] = 1;
                }
            }
            int chosen = -1;
            for (int c = 0; c < k; ++c) {
                if (!used[c]) {
                    chosen = c;
                    break;
                }
            }
            if (chosen == -1) {
                return false;
            }
            col[v] = chosen;
        }
        return true;
    };

    // Build degree-descending order (Welsh-Powell / largest-first greedy)
    std::vector<int> deg_order(n);
    std::iota(deg_order.begin(), deg_order.end(), 0);
    std::sort(deg_order.begin(), deg_order.end(), [&](int a, int b) {
        size_t da = adj[a].size();
        size_t db = adj[b].size();
        if (da != db) return da > db;
        return a < b;
    });

    bool success = try_coloring(deg_order);

    // Adaptive fallback: use random orderings for harder / smaller graphs
    if (!success) {
        int max_trials = 1;
        if (n <= 5000) {
            max_trials = 30;
        } else if (n <= 50000) {
            max_trials = 8;
        } else {
            max_trials = 3;
        }

        std::mt19937 rng(42); // deterministic seed for reproducibility
        std::vector<int> rand_order(n);
        std::iota(rand_order.begin(), rand_order.end(), 0);

        for (int trial = 0; trial < max_trials && !success; ++trial) {
            std::shuffle(rand_order.begin(), rand_order.end(), rng);
            if (try_coloring(rand_order)) {
                success = true;
                break;
            }
        }
    }

    // Additional adaptive strategy: try natural ordering (by vertex id) if previous failed
    if (!success) {
        std::vector<int> id_order(n);
        std::iota(id_order.begin(), id_order.end(), 0);
        success = try_coloring(id_order);
    }

    // Final safety: if still no valid assignment found (extremely rare),
    // fall back to assigning lowest possible colors greedily in current order
    // (this may fail the validation but ensures output is complete)
    if (!success) {
        std::fill(col.begin(), col.end(), 0);
        // Re-run a final greedy pass on deg_order to minimize conflicts
        for (int v : deg_order) {
            std::vector<char> used(k, 0);
            for (int nei : adj[v]) {
                if (col[nei] >= 0 && col[nei] < k) used[col[nei]] = 1;
            }
            for (int c = 0; c < k; ++c) {
                if (!used[c]) {
                    col[v] = c;
                    break;
                }
            }
        }
    }

    // Output the coloring
    for (int i = 0; i < n; ++i) {
        if (i > 0) std::cout << ' ';
        std::cout << col[i];
    }
    std::cout << '\n';

    return 0;
}
```
