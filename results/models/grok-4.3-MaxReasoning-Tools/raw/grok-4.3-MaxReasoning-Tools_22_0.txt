```csharp
using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        string line = Console.ReadLine();
        if (line == null) return;
        string[] parts = line.Trim().Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 3) return;

        int width = int.Parse(parts[0]);
        int height = int.Parse(parts[1]);
        int num_pieces = int.Parse(parts[2]);

        bool[][] board = new bool[height][];
        for (int i = 0; i < height; i++)
        {
            board[i] = new bool[width];
        }

        int piecesProcessed = 0;

        while (piecesProcessed < num_pieces)
        {
            line = Console.ReadLine();
            if (line == null) break;
            line = line.Trim();
            if (line.Length == 0) continue;

            char pieceType = line[0];
            if (!"IOTSZJL".Contains(pieceType))
            {
                continue;
            }

            // Adaptive weights based on current fill ratio
            int currMaxH = 0;
            for (int c = 0; c < width; c++)
            {
                for (int r = height - 1; r >= 0; r--)
                {
                    if (board[r][c])
                    {
                        currMaxH = Math.Max(currMaxH, r + 1);
                        break;
                    }
                }
            }
            double fillRatio = height > 0 ? (double)currMaxH / height : 0;

            int linesWeight = 2000;
            int heightPenalty = 5;
            int holesPenalty = 20;
            int bumpPenalty = 3;

            if (fillRatio > 0.65)
            {
                linesWeight = 3500;
                heightPenalty = 12;
                holesPenalty = 45;
                bumpPenalty = 5;
            }
            else if (fillRatio > 0.4)
            {
                linesWeight = 2500;
                heightPenalty = 8;
                holesPenalty = 30;
            }

            // Find best placement
            int bestScore = int.MinValue;
            int bestRot = 0;
            int bestCol = 0;

            for (int rot = 0; rot < 4; rot++)
            {
                var shape = GetPieceShape(pieceType, rot);
                if (shape.Count == 0) continue;

                int maxDy = 0;
                int maxDx = 0;
                foreach (var (dx, dy) in shape)
                {
                    maxDx = Math.Max(maxDx, dx);
                    maxDy = Math.Max(maxDy, dy);
                }
                int pWidth = maxDx + 1;

                for (int col = 0; col <= width - pWidth; col++)
                {
                    // Find landing row (bottom of piece)
                    int startR = height - maxDy - 1;
                    if (startR < 0) startR = 0;
                    int landingR = startR;

                    while (true)
                    {
                        bool canFall = true;
                        foreach (var (dx, dy) in shape)
                        {
                            int cc = col + dx;
                            int rr = landingR - 1 + dy;
                            if (rr < 0 || (rr < height && board[rr][cc]))
                            {
                                canFall = false;
                                break;
                            }
                        }
                        if (canFall)
                        {
                            landingR--;
                        }
                        else
                        {
                            break;
                        }
                    }

                    // Validate placement
                    bool valid = true;
                    List<(int r, int c)> placed = new List<(int r, int c)>();
                    foreach (var (dx, dy) in shape)
                    {
                        int cc = col + dx;
                        int rr = landingR + dy;
                        if (rr >= height || cc >= width || cc < 0 || board[rr][cc])
                        {
                            valid = false;
                            break;
                        }
                        placed.Add((rr, cc));
                    }
                    if (!valid || landingR < 0) continue;

                    // Temp place
                    foreach (var (rr, cc) in placed)
                    {
                        board[rr][cc] = true;
                    }

                    // Find full rows
                    List<int> fullRows = new List<int>();
                    for (int r = 0; r < height; r++)
                    {
                        bool isFull = true;
                        for (int c = 0; c < width; c++)
                        {
                            if (!board[r][c])
                            {
                                isFull = false;
                                break;
                            }
                        }
                        if (isFull) fullRows.Add(r);
                    }

                    int linesCleared = fullRows.Count;

                    // Compute after-clear stats
                    bool[] isFullArr = new bool[height];
                    foreach (int fr in fullRows) isFullArr[fr] = true;

                    List<int> keptRows = new List<int>();
                    for (int r = 0; r < height; r++)
                    {
                        if (!isFullArr[r]) keptRows.Add(r);
                    }

                    int aggHeight = 0;
                    int holesCount = 0;
                    int[] newColHeights = new int[width];

                    for (int c = 0; c < width; c++)
                    {
                        int colFilled = 0;
                        bool hasFilledAbove = false;
                        int localHoles = 0;

                        for (int i = keptRows.Count - 1; i >= 0; i--)
                        {
                            int r = keptRows[i];
                            if (board[r][c])
                            {
                                hasFilledAbove = true;
                                colFilled++;
                            }
                            else if (hasFilledAbove)
                            {
                                localHoles++;
                            }
                        }

                        holesCount += localHoles;
                        newColHeights[c] = colFilled;
                        aggHeight += colFilled;
                    }

                    int bumpiness = 0;
                    for (int c = 0; c < width - 1; c++)
                    {
                        bumpiness += Math.Abs(newColHeights[c] - newColHeights[c + 1]);
                    }

                    int score = linesCleared * linesWeight 
                                - aggHeight * heightPenalty 
                                - holesCount * holesPenalty 
                                - bumpiness * bumpPenalty;

                    // Undo placement
                    foreach (var (rr, cc) in placed)
                    {
                        board[rr][cc] = false;
                    }

                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestRot = rot;
                        bestCol = col;
                    }
                }
            }

            // Output placement
            Console.WriteLine($"{bestRot} {bestCol}");
            Console.Out.Flush();

            // Read result
            string result = Console.ReadLine();
            if (result == null) break;
            result = result.Trim().ToLower();

            if (result.Contains("gameover"))
            {
                break;
            }

            // Apply placement to real board (recompute landing for best)
            var bestShape = GetPieceShape(pieceType, bestRot);
            int bMaxDy = 0;
            foreach (var (_, dy) in bestShape) bMaxDy = Math.Max(bMaxDy, dy);

            int bStartR = height - bMaxDy - 1;
            if (bStartR < 0) bStartR = 0;
            int bLandingR = bStartR;

            while (true)
            {
                bool canFall = true;
                foreach (var (dx, dy) in bestShape)
                {
                    int cc = bestCol + dx;
                    int rr = bLandingR - 1 + dy;
                    if (rr < 0 || (rr < height && board[rr][cc]))
                    {
                        canFall = false;
                        break;
                    }
                }
                if (canFall)
                {
                    bLandingR--;
                }
                else
                {
                    break;
                }
            }

            // Place permanently
            List<(int r, int c)> finalPlaced = new List<(int r, int c)>();
            foreach (var (dx, dy) in bestShape)
            {
                int rr = bLandingR + dy;
                int cc = bestCol + dx;
                if (rr < height && cc < width && cc >= 0)
                {
                    board[rr][cc] = true;
                    finalPlaced.Add((rr, cc));
                }
            }

            // Clear lines
            List<int> fullToClear = new List<int>();
            for (int r = 0; r < height; r++)
            {
                bool full = true;
                for (int c = 0; c < width; c++)
                {
                    if (!board[r][c])
                    {
                        full = false;
                        break;
                    }
                }
                if (full) fullToClear.Add(r);
            }

            bool[] clearMask = new bool[height];
            foreach (int fr in fullToClear) clearMask[fr] = true;

            int writePos = 0;
            for (int r = 0; r < height; r++)
            {
                if (!clearMask[r])
                {
                    if (writePos != r)
                    {
                        for (int c = 0; c < width; c++)
                        {
                            board[writePos][c] = board[r][c];
                        }
                    }
                    writePos++;
                }
            }

            // Clear remaining upper rows
            for (int r = writePos; r < height; r++)
            {
                for (int c = 0; c < width; c++)
                {
                    board[r][c] = false;
                }
            }

            piecesProcessed++;
        }
    }

    static List<(int x, int y)> GetPieceShape(char piece, int rotation)
    {
        List<(int x, int y)> shape = new List<(int x, int y)>();
        rotation %= 4;
        if (rotation < 0) rotation += 4;

        switch (piece)
        {
            case 'I':
                if (rotation % 2 == 0)
                    shape.AddRange(new[] { (0, 0), (1, 0), (2, 0), (3, 0) });
                else
                    shape.AddRange(new[] { (0, 0), (0, 1), (0, 2), (0, 3) });
                break;

            case 'O':
                shape.AddRange(new[] { (0, 0), (1, 0), (0, 1), (1, 1) });
                break;

            case 'T':
                if (rotation == 0)
                    shape.AddRange(new[] { (0, 0), (1, 0), (2, 0), (1, 1) });
                else if (rotation == 1)
                    shape.AddRange(new[] { (0, 0), (0, 1), (0, 2), (1, 1) });
                else if (rotation == 2)
                    shape.AddRange(new[] { (0, 1), (1, 0), (1, 1), (2, 1) });
                else
                    shape.AddRange(new[] { (0, 1), (1, 0), (1, 1), (1, 2) });
                break;

            case 'S':
                if (rotation % 2 == 0)
                    shape.AddRange(new[] { (1, 0), (2, 0), (0, 1), (1, 1) });
                else
                    shape.AddRange(new[] { (0, 1), (1, 0), (1, 1), (0, 2) });
                break;

            case 'Z':
                if (rotation % 2 == 0)
                    shape.AddRange(new[] { (0, 0), (1, 0), (1, 1), (2, 1) });
                else
                    shape.AddRange(new[] { (0, 1), (1, 0), (1, 1), (0, 2) });
                break;

            case 'J':
                if (rotation == 0)
                    shape.AddRange(new[] { (0, 0), (0, 1), (1, 1), (2, 1) });
                else if (rotation == 1)
                    shape.AddRange(new[] { (0, 0), (1, 0), (0, 1), (0, 2) });
                else if (rotation == 2)
                    shape.AddRange(new[] { (0, 0), (1, 0), (2, 0), (2, 1) });
                else
                    shape.AddRange(new[] { (1, 0), (1, 1), (1, 2), (0, 2) });
                break;

            case 'L':
                if (rotation == 0)
                    shape.AddRange(new[] { (0, 0), (1, 0), (2, 0), (2, 1) });
                else if (rotation == 1)
                    shape.AddRange(new[] { (0, 0), (0, 1), (0, 2), (1, 2) });
                else if (rotation == 2)
                    shape.AddRange(new[] { (0, 1), (1, 1), (2, 0), (2, 1) });
                else
                    shape.AddRange(new[] { (0, 0), (1, 0), (1, 1), (1, 2) });
                break;
        }
        return shape;
    }
}
```
