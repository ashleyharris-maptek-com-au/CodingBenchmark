use std::io::{self, BufRead, BufReader, Write, BufWriter};
use std::f64::consts::PI;

fn main() {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin);
    let mut stdout = BufWriter::new(io::stdout());
    let mut line = String::new();

    // Read initial line: width height gravity max_thrust fuel
    if reader.read_line(&mut line).is_err() {
        return;
    }
    let nums: Vec<f64> = line
        .trim()
        .split_whitespace()
        .map(|s| s.parse::<f64>().unwrap_or(0.0))
        .collect();
    if nums.len() < 5 {
        return;
    }
    let width = nums[0] as usize;
    let height = nums[1] as usize;
    let gravity = nums[2];
    let max_thrust = nums[3];
    let _initial_fuel = nums[4];

    line.clear();

    // Read second line: start_x start_y target_x target_y
    if reader.read_line(&mut line).is_err() {
        return;
    }
    let pos: Vec<f64> = line
        .trim()
        .split_whitespace()
        .map(|s| s.parse::<f64>().unwrap_or(0.0))
        .collect();
    if pos.len() < 4 {
        return;
    }
    let target_x = pos[2];
    let target_y = pos[3];

    // Consume map rows (one line per row)
    for _ in 0..height {
        line.clear();
        let _ = reader.read_line(&mut line);
    }

    // Main control loop
    loop {
        line.clear();
        match reader.read_line(&mut line) {
            Ok(0) => break,
            Ok(_) => {}
            Err(_) => break,
        }

        let state_str = line.trim();
        if state_str.is_empty() {
            continue;
        }

        let state: Vec<f64> = state_str
            .split_whitespace()
            .map(|s| s.parse::<f64>().unwrap_or(0.0))
            .collect();

        if state.len() < 7 {
            continue;
        }

        let x = state[0];
        let y = state[1];
        let vx = state[2];
        let vy = state[3];
        let angle = state[4];
        let _angular_vel = state[5];
        let _fuel = state[6];

        let (thrust, turn) = compute_thrust_turn(
            x,
            y,
            vx,
            vy,
            angle,
            target_x,
            target_y,
            gravity,
            max_thrust,
        );

        let _ = writeln!(&mut stdout, "{:.4} {:.4}", thrust, turn);
        let _ = stdout.flush();
    }
}

fn compute_thrust_turn(
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    target_x: f64,
    target_y: f64,
    gravity: f64,
    max_thrust: f64,
) -> (f64, f64) {
    // Tuned gains for stable real-time control
    let kp_pos_x: f64 = 0.38;
    let max_vx: f64 = 14.0;
    let kp_vel_x: f64 = 2.8;

    let kp_pos_y: f64 = 0.11;
    let max_vy: f64 = 9.0;
    let kp_vel_y: f64 = 4.2;

    let kp_angle: f64 = 5.5;
    let max_tilt: f64 = PI / 2.8;

    // Horizontal control
    let error_x = target_x - x;
    let mut desired_vx = error_x * kp_pos_x;
    desired_vx = desired_vx.clamp(-max_vx, max_vx);

    let vel_error_x = desired_vx - vx;
    let desired_ax = vel_error_x * kp_vel_x;

    // Vertical control
    let error_y = target_y - y;
    let mut desired_vy = error_y * kp_pos_y;
    desired_vy = desired_vy.clamp(0.4, max_vy);

    // Slow descent when close horizontally
    let horiz_dist = error_x.abs();
    if horiz_dist < 18.0 && error_y < 40.0 {
        desired_vy = (desired_vy * 0.55).max(0.9);
    }

    // Near landing zone: very soft approach
    let dist = ((target_x - x).powi(2) + (target_y - y).powi(2)).sqrt();
    if dist < 12.0 {
        desired_vy = (desired_vy * 0.45).max(0.6);
    }

    let vel_error_y = vy - desired_vy;
    let desired_ay = gravity - kp_vel_y * vel_error_y;

    // Required thrust acceleration components
    let needed_ax = desired_ax;
    let needed_ay = desired_ay - gravity;

    let thrust_mag = (needed_ax * needed_ax + needed_ay * needed_ay).sqrt();
    let mut thrust = (thrust_mag / max_thrust).clamp(0.0, 1.0);

    // Desired angle from thrust vector
    let mut desired_angle = if thrust_mag > 0.05 {
        needed_ax.atan2(-needed_ay)
    } else {
        0.0
    };

    // Clamp desired tilt for stability
    desired_angle = desired_angle.clamp(-max_tilt, max_tilt);

    // Angle error with normalization
    let mut angle_error = desired_angle - angle;
    let two_pi = 2.0 * PI;
    angle_error = ((angle_error + PI) % two_pi) - PI;

    let mut turn = kp_angle * angle_error;
    turn = turn.clamp(-1.0, 1.0);

    // Soft landing mode when very close and slow
    if dist < 6.0 && vy.abs() < 4.5 && vx.abs() < 4.5 {
        thrust = thrust.min(0.75);
        if vy > 2.8 {
            thrust = thrust.max(0.65);
        }
    }

    // Emergency brake near ground
    if error_y < 8.0 && vy > 4.5 {
        thrust = 1.0;
    }

    // Prevent excessive thrust when almost landed
    if error_y < 3.0 && vy < 2.0 {
        thrust = thrust.min(0.45);
    }

    (thrust, turn)
}
