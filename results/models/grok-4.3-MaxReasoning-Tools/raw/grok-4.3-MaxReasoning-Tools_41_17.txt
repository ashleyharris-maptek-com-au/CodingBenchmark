; SPIR-V
; Version: 1.0
OpCapability Shader
%glsl_std450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %uv %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_UBO = OpTypePointer Uniform %UBO
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%func_type = OpTypeFunction %void
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_8 = OpConstant %float 8
%float_16 = OpConstant %float 16
%float_0_05 = OpConstant %float 0.05
%float_0_1 = OpConstant %float 0.1
%float_0_5 = OpConstant %float 0.5
%float_0_95 = OpConstant %float 0.95
%float_0_9 = OpConstant %float 0.9
%float_2 = OpConstant %float 2
%float_0_7 = OpConstant %float 0.7
%float_0_2 = OpConstant %float 0.2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%vec3_mortar = OpConstantComposite %vec3 %float_0_7 %float_0_7 %float_0_7
%vec3_brick = OpConstantComposite %vec3 %float_0_7 %float_0_2 %float_0_1
%outColor = OpVariable %ptr_Output_vec4 Output
%worldPos = OpVariable %ptr_Input_vec3 Input
%normal = OpVariable %ptr_Input_vec3 Input
%uv = OpVariable %ptr_Input_vec2 Input
%ubo = OpVariable %ptr_UBO Uniform
%main = OpFunction %void None %func_type
%entry = OpLabel
%worldPos_val = OpLoad %vec3 %worldPos
%normal_val = OpLoad %vec3 %normal
%uv_val = OpLoad %vec2 %uv
%lightPos_ptr = OpAccessChain %ptr_Uniform_vec4 %ubo %int_3
%lightPos_val = OpLoad %vec4 %lightPos_ptr
%light_xyz = OpVectorShuffle %vec3 %lightPos_val %lightPos_val 0 1 2
%L_vec = OpFSub %vec3 %light_xyz %worldPos_val
%L = OpExtInst %vec3 %glsl_std450 69 %L_vec
%N = OpExtInst %vec3 %glsl_std450 69 %normal_val
%dotNL = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl_std450 40 %dotNL %float_0
%uv_x = OpCompositeExtract %float %uv_val 0
%uv_y = OpCompositeExtract %float %uv_val 1
%scaled_x = OpFMul %float %uv_x %float_8
%scaled_y = OpFMul %float %uv_y %float_16
%row = OpExtInst %float %glsl_std450 8 %scaled_y
%row_mod = OpFMod %float %row %float_2
%half = OpFOrdGreaterThan %bool %row_mod %float_0_5
%stagger = OpSelect %float %half %float_0_5 %float_0
%px = OpFAdd %float %scaled_x %stagger
%fx = OpExtInst %float %glsl_std450 10 %px
%fy = OpExtInst %float %glsl_std450 10 %scaled_y
%mortar_x1 = OpFOrdLessThan %bool %fx %float_0_05
%mortar_x2 = OpFOrdGreaterThan %bool %fx %float_0_95
%mortar_y1 = OpFOrdLessThan %bool %fy %float_0_1
%mortar_y2 = OpFOrdGreaterThan %bool %fy %float_0_9
%mortar_x = OpLogicalOr %bool %mortar_x1 %mortar_x2
%mortar_y = OpLogicalOr %bool %mortar_y1 %mortar_y2
%is_mortar = OpLogicalOr %bool %mortar_x %mortar_y
%base_color = OpSelect %vec3 %is_mortar %vec3_mortar %vec3_brick
%temp = OpFMul %float %float_0_6 %diffuse
%factor = OpFAdd %float %float_0_4 %temp
%shaded = OpVectorTimesScalar %vec3 %base_color %factor
%out_rgba = OpCompositeConstruct %vec4 %shaded %float_1
OpStore %outColor %out_rgba
OpReturn
OpFunctionEnd
