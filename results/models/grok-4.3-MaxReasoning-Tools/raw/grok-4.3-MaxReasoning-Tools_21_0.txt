#include <iostream>
#include <vector>
#include <deque>
#include <set>
#include <climits>
#include <algorithm>

using Pos = std::vector<int>;

bool in_bounds(const Pos& p, const std::vector<int>& bounds) {
    for (size_t i = 0; i < p.size(); ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    return true;
}

int manhattan(const Pos& a, const Pos& b) {
    int dist = 0;
    for (size_t i = 0; i < a.size(); ++i) {
        dist += std::abs(a[i] - b[i]);
    }
    return dist;
}

Pos find_closest_food(const Pos& head, const std::set<Pos>& foods) {
    if (foods.empty()) return head;
    Pos best = *foods.begin();
    int min_d = manhattan(head, best);
    for (const auto& f : foods) {
        int d = manhattan(head, f);
        if (d < min_d) {
            min_d = d;
            best = f;
        }
    }
    return best;
}

bool is_valid_move(const Pos& new_head, const std::deque<Pos>& snake,
                   const std::set<Pos>& obstacles, const std::set<Pos>& foods,
                   const std::vector<int>& bounds) {
    if (!in_bounds(new_head, bounds)) return false;
    if (obstacles.count(new_head)) return false;
    
    bool eating = foods.count(new_head) > 0;
    
    for (size_t i = 0; i < snake.size(); ++i) {
        if (snake[i] == new_head) {
            if (eating || i != snake.size() - 1) {
                return false;
            }
        }
    }
    return true;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    
    int D, max_turns;
    std::cin >> D >> max_turns;
    
    std::vector<int> bounds(D);
    for (int i = 0; i < D; ++i) {
        std::cin >> bounds[i];
    }
    
    int snake_len;
    std::cin >> snake_len;
    
    std::deque<Pos> snake;
    Pos head(D);
    for (int i = 0; i < D; ++i) {
        std::cin >> head[i];
    }
    snake.push_front(head);
    
    // If snake_len > 1, we treat as length 1 initially since only head given in spec
    // (additional body positions would be needed otherwise)
    
    int F;
    std::cin >> F;
    std::set<Pos> food_set;
    for (int i = 0; i < F; ++i) {
        Pos f(D);
        for (int j = 0; j < D; ++j) {
            std::cin >> f[j];
        }
        food_set.insert(f);
    }
    
    int O;
    std::cin >> O;
    std::set<Pos> obstacle_set;
    for (int i = 0; i < O; ++i) {
        Pos o(D);
        for (int j = 0; j < D; ++j) {
            std::cin >> o[j];
        }
        obstacle_set.insert(o);
    }
    
    int turns = 0;
    while (turns < max_turns && !food_set.empty() && !snake.empty()) {
        Pos current_head = snake.front();
        
        Pos target = find_closest_food(current_head, food_set);
        
        int best_dist = INT_MAX;
        std::pair<int, int> chosen = {-1, 0};
        
        for (int d = 0; d < D; ++d) {
            for (int dir : {-1, 1}) {
                Pos np = current_head;
                np[d] += dir;
                
                if (is_valid_move(np, snake, obstacle_set, food_set, bounds)) {
                    int dist = manhattan(np, target);
                    if (dist < best_dist) {
                        best_dist = dist;
                        chosen = {d, dir};
                    }
                }
            }
        }
        
        if (chosen.first == -1) {
            // Fallback: any valid move
            bool found = false;
            for (int d = 0; d < D && !found; ++d) {
                for (int dir : {-1, 1}) {
                    Pos np = current_head;
                    np[d] += dir;
                    if (is_valid_move(np, snake, obstacle_set, food_set, bounds)) {
                        chosen = {d, dir};
                        found = true;
                        break;
                    }
                }
            }
            if (!found) {
                break; // No valid moves
            }
        }
        
        // Output move
        std::cout << chosen.first << " " << chosen.second << "\n";
        
        // Simulate
        Pos new_head = current_head;
        new_head[chosen.first] += chosen.second;
        
        bool eating = food_set.count(new_head) > 0;
        if (eating) {
            food_set.erase(new_head);
        }
        
        snake.push_front(new_head);
        if (!eating) {
            snake.pop_back();
        }
        
        ++turns;
    }
    
    return 0;
}
