        function createGridVisual(data, cutsSet) {
            const container = document.getElementById('grid-visual');
            container.innerHTML = '';
            container.style.gridTemplateColumns = `repeat(${data.width}, 22px)`;
            
            const ignSet = new Set(data.ignitions.map(i => `${i[0]},${i[1]}`));
            const townSet = new Set(data.towns.map(t => `${t.x},${t.y}`));
            
            for (let y = 0; y < data.height; y++) {
                for (let x = 0; x < data.width; x++) {
                    const cell = document.createElement('div');
                    cell.className = 'cell';
                    
                    const key = `${x},${y}`;
                    const isCut = cutsSet && cutsSet.has(key);
                    const isIgn = ignSet.has(key);
                    const isTown = townSet.has(key);
                    const isForest = data.grid[y] && data.grid[y][x] === '.';
                    
                    if (isCut) {
                        cell.classList.add('cut');
                        cell.textContent = '✂';
                    } else if (isIgn) {
                        cell.classList.add('ignition');
                        cell.textContent = '🔥';
                    } else if (isTown) {
                        cell.classList.add('town');
                        cell.textContent = '🏠';
                    } else if (isForest) {
                        cell.classList.add('forest');
                    } else {
                        cell.classList.add('water');
                        cell.style.background = '#0e7490';
                    }
                    
                    container.appendChild(cell);
                }
            }
        }
        
        function simulateFire(cutGrid, grid, ignitions, width, height) {
            const visited = Array.from({length: height}, () => Array(width).fill(false));
            const queue = [];
            let qStart = 0;
            
            for (const [x, y] of ignitions) {
                if (x >= 0 && x < width && y >= 0 && y < height && 
                    grid[y][x] === '.' && !cutGrid[y][x]) {
                    if (!visited[y][x]) {
                        visited[y][x] = true;
                        queue.push([x, y]);
                    }
                }
            }
            
            const dirs = [[-1,0],[1,0],[0,-1],[0,1]];
            
            while (qStart < queue.length) {
                const [x, y] = queue[qStart++];
                for (const [dx, dy] of dirs) {
                    const nx = x + dx;
                    const ny = y + dy;
                    if (nx >= 0 && nx < width && ny >= 0 && ny < height &&
                        grid[ny][nx] === '.' && !cutGrid[ny][nx] && !visited[ny][nx]) {
                        visited[ny][nx] = true;
                        queue.push([nx, ny]);
                    }
                }
            }
            return visited;
        }
        
        function findPathToTarget(cutGrid, grid, ignitions, targetX, targetY, width, height) {
            const visited = Array.from({length: height}, () => Array(width).fill(false));
            const parent = Array.from({length: height}, () => Array(width).fill(null));
            const queue = [];
            let qStart = 0;
            let reachedTarget = false;
            
            const dirs = [[-1,0],[1,0],[0,-1],[0,1]];
            
            for (const [sx, sy] of ignitions) {
                if (sx >= 0 && sx < width && sy >= 0 && sy < height &&
                    grid[sy][sx] === '.' && !cutGrid[sy][sx]) {
                    visited[sy][sx] = true;
                    parent[sy][sx] = [-1, -1];
                    queue.push([sx, sy]);
                    if (sx === targetX && sy === targetY) reachedTarget = true;
                }
            }
            
            if (reachedTarget) {
                return [[targetX, targetY]];
            }
            
            while (qStart < queue.length && !reachedTarget) {
                const [x, y] = queue[qStart++];
                for (const [dx, dy] of dirs) {
                    const nx = x + dx;
                    const ny = y + dy;
                    if (nx >= 0 && nx < width && ny >= 0 && ny < height &&
                        grid[ny][nx] === '.' && !cutGrid[ny][nx] && !visited[ny][nx]) {
                        visited[ny][nx] = true;
                        parent[ny][nx] = [x, y];
                        queue.push([nx, ny]);
                        if (nx === targetX && ny === targetY) {
                            reachedTarget = true;
                            break;
                        }
                    }
                }
            }
            
            if (!reachedTarget || !parent[targetY] || parent[targetY][targetX] === null) {
                return null;
            }
            
            // Reconstruct path
            const path = [];
            let curX = targetX;
            let curY = targetY;
            
            while (true) {
                path.unshift([curX, curY]);
                const p = parent[curY][curX];
                if (!p || (p[0] === -1 && p[1] === -1)) {
                    break;
                }
                curX = p[0];
                curY = p[1];
            }
            
            return path;
        }
        
        function runAlgorithm() {
            const inputText = document.getElementById('input-json').value.trim();
            let data;
            
            try {
                data = JSON.parse(inputText);
            } catch(e) {
                alert("Invalid JSON input");
                return;
            }
            
            const { width, height, grid, ignitions, towns, cutBudget } = data;
            
            if (!grid || !ignitions || !towns) {
                alert("Missing required fields in input");
                return;
            }
            
            // Initialize cutGrid
            const cutGrid = Array.from({length: height}, () => new Array(width).fill(false));
            const cuts = [];
            
            // Main greedy loop
            for (let i = 0; i < cutBudget; i++) {
                // Simulate current fire
                const visited = simulateFire(cutGrid, grid, ignitions, width, height);
                
                // Find highest value threatened town
                let maxValue = 0;
                let target = null;
                
                for (const town of towns) {
                    const tx = town.x;
                    const ty = town.y;
                    if (tx >= 0 && tx < width && ty >= 0 && ty < height && 
                        visited[ty][tx] && town.value > maxValue) {
                        maxValue = town.value;
                        target = town;
                    }
                }
                
                if (!target) break;
                
                // Find path to target
                const path = findPathToTarget(cutGrid, grid, ignitions, target.x, target.y, width, height);
                
                if (!path || path.length === 0) break;
                
                // Select cut position: prefer cell before the town
                let cutIdx;
                if (path.length === 1) {
                    cutIdx = 0;
                } else {
                    cutIdx = path.length - 2;
                }
                
                const [cutX, cutY] = path[cutIdx];
                
                // Skip if already cut (shouldn't happen)
                if (cutGrid[cutY][cutX]) continue;
                
                // Place cut
                cutGrid[cutY][cutX] = true;
                cuts.push([cutX, cutY]);
            }
            
            // Prepare output
            const output = {
                firebreaks: cuts
            };
            
            // Display results
            const outputEl = document.getElementById('output-json');
            outputEl.textContent = JSON.stringify(output, null, 2);
            
            // Update visual
            const cutsSet = new Set(cuts.map(c => `${c[0]},${c[1]}`));
            createGridVisual(data, cutsSet);
            
            // Update budget display
            document.getElementById('budget-val').textContent = cutBudget;
            
            // Show a quick summary
            console.log('%c[Firebreak] Algorithm completed', 'color:#64748b', {
                cuts: cuts.length,
                budgetUsed: cuts.length,
                remainingBudget: cutBudget - cuts.length
            });
        }
        
        // Boot with demo
        window.onload = function() {
            const demoInput = document.getElementById('input-json');
            
            // Run demo automatically on load
            setTimeout(() => {
                runAlgorithm();
            }, 400);
            
            // Keyboard support
            demoInput.addEventListener('keydown', function(e) {
                if (e.key === 'Enter' && (e.ctrlKey || e.metaKey)) {
                    e.preventDefault();
                    runAlgorithm();
                }
            });
        }
        
        // Expose for external testing
        window.runFirebreak = runAlgorithm;
    </script>
