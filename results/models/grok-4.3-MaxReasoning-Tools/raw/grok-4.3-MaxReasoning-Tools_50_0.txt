```glsl
#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 radii;
    vec4 weights;
    vec4 bounds;
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) return;

    vec3 pos = inData[i * 2].xyz;
    vec3 vel = inData[i * 2 + 1].xyz;

    vec3 sep = vec3(0.0);
    vec3 ali = vec3(0.0);
    vec3 coh = vec3(0.0);

    int sepCount = 0;
    int aliCount = 0;
    int cohCount = 0;

    for (uint j = 0u; j < N; ++j) {
        if (j == i) continue;

        vec3 pos_j = inData[j * 2].xyz;
        vec3 vel_j = inData[j * 2 + 1].xyz;

        vec3 diff = pos - pos_j;
        float dist = length(diff);

        if (dist < 0.001) continue;

        // Separation
        if (dist < radii.y) {
            sep += normalize(diff);
            sepCount++;
        }

        // Alignment
        if (dist < radii.z) {
            ali += vel_j;
            aliCount++;
        }

        // Cohesion
        if (dist < radii.w) {
            coh += pos_j;
            cohCount++;
        }
    }

    vec3 acceleration = vec3(0.0);

    if (sepCount > 0) {
        sep /= float(sepCount);
        acceleration += sep * weights.x;
    }

    if (aliCount > 0) {
        ali /= float(aliCount);
        acceleration += ali * weights.y;
    }

    if (cohCount > 0) {
        coh /= float(cohCount);
        acceleration += (coh - pos) * weights.z;
    }

    // Boundary force
    float boundSize = bounds.x;
    float boundForce = bounds.y;
    float thresh = boundSize * 0.4;
    float scale = boundSize * 0.5;

    vec3 boundary = vec3(0.0);

    // X
    if (pos.x > thresh) {
        boundary.x = -(pos.x / scale) * boundForce;
    } else if (pos.x < -thresh) {
        boundary.x = -(pos.x / scale) * boundForce;
    }

    // Y
    if (pos.y > thresh) {
        boundary.y = -(pos.y / scale) * boundForce;
    } else if (pos.y < -thresh) {
        boundary.y = -(pos.y / scale) * boundForce;
    }

    // Z
    if (pos.z > thresh) {
        boundary.z = -(pos.z / scale) * boundForce;
    } else if (pos.z < -thresh) {
        boundary.z = -(pos.z / scale) * boundForce;
    }

    acceleration += boundary;

    // Update velocity
    float dt = radii.x;
    vel += acceleration * dt;

    // Clamp speed
    float maxSpeed = weights.w;
    float speed = length(vel);
    if (speed > maxSpeed && speed > 0.0) {
        vel = (vel / speed) * maxSpeed;
    }

    // Update position
    pos += vel * dt;

    // Write results
    outData[i * 2] = vec4(pos, 0.0);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
```
