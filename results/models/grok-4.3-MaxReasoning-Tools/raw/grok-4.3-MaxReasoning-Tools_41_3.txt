```spirv
; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inColor %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0
%void = OpTypeVoid
%funcType = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%ptrInputV3 = OpTypePointer Input %v3float
%ptrOutputV4 = OpTypePointer Output %v4float
%inColor = OpVariable %ptrInputV3 Input
%outColor = OpVariable %ptrOutputV4 Output
%float_1 = OpConstant %float 1.0
%main = OpFunction %void None %funcType
%entry = OpLabel
%color = OpLoad %v3float %inColor
%result = OpCompositeConstruct %v4float %color %float_1
OpStore %outColor %result
OpReturn
OpFunctionEnd
```
