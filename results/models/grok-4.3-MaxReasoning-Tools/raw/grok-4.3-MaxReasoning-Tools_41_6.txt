```spirv
; SPIR-V
; Version: 1.0

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor %ubo
OpExecutionMode %main OriginUpperLeft

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%v2float = OpTypeVector %float 2
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%int = OpTypeInt 32 1

%inWorldPos = OpVariable %_ptr_Input_v3float Input
%inNormal = OpVariable %_ptr_Input_v3float Input
%inUV = OpVariable %_ptr_Input_v2float Input
%inTangent = OpVariable %_ptr_Input_v3float Input
%inColor = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%func = OpTypeFunction %void

%float0 = OpConstant %float 0
%float1 = OpConstant %float 1
%float0_1 = OpConstant %float 0.1
%float0_6 = OpConstant %float 0.6
%float0_8 = OpConstant %float 0.8
%float64 = OpConstant %float 64

%int0 = OpConstant %int 0
%int1 = OpConstant %int 1
%int2 = OpConstant %int 2
%int3 = OpConstant %int 3
%int4 = OpConstant %int 4
%int5 = OpConstant %int 5

%main = OpFunction %void None %func
%entry = OpLabel

%worldPos = OpLoad %v3float %inWorldPos
%normal = OpLoad %v3float %inNormal

%lightPosPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int3
%lightPos = OpLoad %v4float %lightPosPtr
%cameraPosPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int4
%cameraPos = OpLoad %v4float %cameraPosPtr

%lightXYZ = OpVectorShuffle %v3float %lightPos %lightPos 0 1 2
%Lvec = OpFSub %v3float %lightXYZ %worldPos
%L = OpNormalize %v3float %Lvec

%camXYZ = OpVectorShuffle %v3float %cameraPos %cameraPos 0 1 2
%Vvec = OpFSub %v3float %camXYZ %worldPos
%V = OpNormalize %v3float %Vvec

%LV = OpFAdd %v3float %L %V
%H = OpNormalize %v3float %LV

%N = OpNormalize %v3float %normal

%dotNL = OpDot %float %N %L
%diffuse = OpFMax %float %dotNL %float0

%dotNH = OpDot %float %N %H
%maxNH = OpFMax %float %dotNH %float0
%spec = OpPow %float %maxNH %float64

%diffTerm = OpFMul %float %float0_6 %diffuse
%specTerm = OpFMul %float %float0_8 %spec
%tmp = OpFAdd %float %float0_1 %diffTerm
%intensity = OpFAdd %float %tmp %specTerm

%finalColor = OpCompositeConstruct %v4float %intensity %intensity %intensity %float1
OpStore %outColor %finalColor

OpReturn
OpFunctionEnd
```
