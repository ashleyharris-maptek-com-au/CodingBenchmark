```python
import math

# Module-level persistent state
_ALT_I = 0.0
_HDG_I = 0.0
_IAS_I = 0.0
_ELEV_TRIM = 0.0
_PREV_ROLL = 0.0
_PREV_PITCH = 0.0
_LAST_ALT = 0.0
_LAST_IAS = 250.0
_TIME_ACCUM = 0.0
_ENG_FAIL_TIME = 0.0

def _clamp(val, lo, hi):
    return max(lo, min(hi, val))

def _normalize_hdg(hdg):
    return (hdg + 180.0) % 360.0 - 180.0

def _median(vals):
    if not vals:
        return 0.0
    s = sorted(vals)
    n = len(s)
    return s[n // 2] if n % 2 == 1 else (s[n // 2 - 1] + s[n // 2]) / 2.0

def _get_best_alt(state):
    baros = []
    for i in [1, 2, 3]:
        if state.get(f"baro_alt_{i}_valid", True):
            baros.append(state.get(f"baro_alt_{i}_ft", 0.0))
    if baros:
        return _median(baros)
    # Fallbacks
    gps = [state.get(f"gps_{i}_alt_ft", 0.0) for i in [1, 2, 3] if state.get(f"gps_{i}_valid", True)]
    if gps:
        return _median(gps)
    radar = [state.get(f"radar_alt_{i}_ft", 0.0) for i in [1, 2, 3] if state.get(f"radar_alt_{i}_valid", True)]
    return _median(radar) if radar else 10000.0

def _get_best_ias(state):
    iass = []
    for i in [1, 2, 3]:
        if state.get(f"ias_{i}_valid", True):
            iass.append(state.get(f"ias_{i}_kt", 250.0))
    return _median(iass) if iass else 250.0

def _get_best_hdg(state):
    hdgs = []
    for i in [1, 2, 3]:
        if state.get(f"irs_{i}_valid", True):
            hdgs.append(state.get(f"irs_{i}_heading_deg", 0.0))
    if not hdgs:
        hdgs = [state.get("magnetic_heading_deg", 0.0)]
    return _median(hdgs)

def _get_best_pitch_roll(state):
    pitches = []
    rolls = []
    for i in [1, 2, 3]:
        if state.get(f"irs_{i}_valid", True):
            pitches.append(state.get(f"irs_{i}_pitch_deg", 0.0))
            rolls.append(state.get(f"irs_{i}_roll_deg", 0.0))
    pitch = _median(pitches) if pitches else state.get("irs_1_pitch_deg", 0.0)
    roll = _median(rolls) if rolls else state.get("irs_1_roll_deg", 0.0)
    return pitch, roll

def _get_vs(state):
    vss = []
    for i in [1, 2, 3]:
        if state.get(f"irs_{i}_valid", True):
            vss.append(state.get(f"irs_{i}_vs_fpm", 0.0))
    if vss:
        return _median(vss)
    return state.get("vertical_speed_fpm", 0.0)

def _get_aoa(state):
    aoas = []
    for i in [1, 2, 3]:
        if state.get(f"aoa_{i}_valid", True):
            aoas.append(state.get(f"aoa_{i}_deg", 0.0))
    return _median(aoas) if aoas else 4.0

def _get_yaw_rate(state):
    rates = []
    for i in [1, 2, 3]:
        if state.get(f"gyro_{i}_valid", True):
            rates.append(state.get(f"gyro_{i}_yaw_rate_dps", 0.0))
    return _median(rates) if rates else 0.0

def _get_roll_rate(state):
    rates = []
    for i in [1, 2, 3]:
        if state.get(f"gyro_{i}_valid", True):
            rates.append(state.get(f"gyro_{i}_roll_rate_dps", 0.0))
    return _median(rates) if rates else 0.0

def _get_pitch_rate(state):
    rates = []
    for i in [1, 2, 3]:
        if state.get(f"gyro_{i}_valid", True):
            rates.append(state.get(f"gyro_{i}_pitch_rate_dps", 0.0))
    return _median(rates) if rates else 0.0

def autopilot_step(state: dict, dt: float) -> dict:
    global _ALT_I, _HDG_I, _IAS_I, _ELEV_TRIM
    global _PREV_ROLL, _PREV_PITCH, _LAST_ALT, _LAST_IAS, _TIME_ACCUM, _ENG_FAIL_TIME

    if dt <= 0:
        dt = 0.05

    _TIME_ACCUM += dt

    # === SENSOR FUSION ===
    alt = _get_best_alt(state)
    ias = _get_best_ias(state)
    hdg = _get_best_hdg(state)
    hdg = _normalize_hdg(hdg)
    pitch, roll = _get_best_pitch_roll(state)
    vs = _get_vs(state)
    aoa = _get_aoa(state)
    yaw_rate = _get_yaw_rate(state)
    roll_rate = _get_roll_rate(state)
    pitch_rate = _get_pitch_rate(state)

    # Mach / speed protection
    mach = state.get("mach_1", state.get("mach_2", 0.0))
    mach = max(mach, state.get("mach_3", 0.0)) if mach == 0 else mach

    # Targets
    tgt_alt = state.get("target_alt_ft", alt)
    tgt_hdg = state.get("target_hdg_deg", hdg)
    tgt_ias = state.get("target_ias_kt", ias)
    ap_engaged = state.get("ap_engaged", True)
    at_engaged = state.get("at_engaged", True)

    # Engine status
    eng1_run = state.get("engine_1_running", True)
    eng2_run = state.get("engine_2_running", True)
    thrust1 = state.get("engine_1_thrust_pct", 80.0)
    thrust2 = state.get("engine_2_thrust_pct", 80.0)

    # === STALL / HIGH AOA PROTECTION ===
    stall_protect = False
    if aoa > 11.0 or (ias < 140 and alt < 8000):
        stall_protect = True

    # === ASYMMETRIC THRUST / ENGINE FAILURE DETECTION ===
    eng_fail = False
    rudder_bias = 0.0
    if not eng1_run or not eng2_run:
        eng_fail = True
        _ENG_FAIL_TIME += dt
        thrust_diff = thrust1 - thrust2
        rudder_bias = _clamp(thrust_diff * 0.0045, -0.45, 0.45)
    else:
        _ENG_FAIL_TIME = max(0.0, _ENG_FAIL_TIME - dt * 0.5)

    # === LATERAL (HEADING / ROLL) ===
    hdg_err = _normalize_hdg(tgt_hdg - hdg)

    # Desired bank angle (proportional + integral)
    des_bank = _clamp(hdg_err * 1.8, -28.0, 28.0)
    if abs(hdg_err) < 3.0:
        des_bank *= 0.6   # reduce bank near target

    bank_err = des_bank - roll

    # Aileron: PD + integral on heading
    _HDG_I += hdg_err * dt * 0.03
    _HDG_I = _clamp(_HDG_I, -12.0, 12.0)

    aileron = bank_err * 0.055
    aileron += _HDG_I * 0.012
    aileron += roll_rate * -0.035
    aileron = _clamp(aileron, -0.95, 0.95)

    # Yaw damper + engine failure compensation
    rudder = yaw_rate * -0.065
    rudder += rudder_bias
    if eng_fail:
        rudder = _clamp(rudder, -0.65, 0.65)
    else:
        rudder = _clamp(rudder, -0.35, 0.35)

    # === VERTICAL (ALTITUDE / PITCH) ===
    alt_err = tgt_alt - alt
    _LAST_ALT = alt

    # Target vertical speed
    vs_tgt = _clamp(alt_err * 0.65, -2200.0, 2200.0)
    if stall_protect:
        vs_tgt = _clamp(vs_tgt, -800.0, 300.0)

    vs_err = vs_tgt - vs

    # Pitch target from VS
    pitch_tgt = vs_err * 0.0028
    pitch_tgt += alt_err * 0.00018
    pitch_tgt = _clamp(pitch_tgt, -12.0, 14.0)

    if stall_protect:
        pitch_tgt = _clamp(pitch_tgt, -6.0, 4.0)

    pitch_err = pitch_tgt - pitch

    # Elevator: PD controller
    _ALT_I += alt_err * dt * 0.0008
    _ALT_I = _clamp(_ALT_I, -25.0, 25.0)

    elevator = pitch_err * 0.11
    elevator += _ALT_I * 0.0009
    elevator += pitch_rate * -0.028
    elevator = _clamp(elevator, -0.92, 0.92)

    # === ELEVATOR TRIM (slow auto-trim) ===
    trim_adjust = 0.0
    if abs(elevator) > 0.08 and abs(vs) < 800:
        trim_adjust = -elevator * 0.035 * dt
    _ELEV_TRIM += trim_adjust
    _ELEV_TRIM = _clamp(_ELEV_TRIM, -0.95, 0.95)

    # === AUTOTHROTTLE / SPEED ===
    ias_err = tgt_ias - ias
    _LAST_IAS = ias

    # Mach / Vne protection
    speed_protect = False
    if ias > 360 or mach > 0.81:
        speed_protect = True
        ias_err -= 15.0

    _IAS_I += ias_err * dt * 0.004
    _IAS_I = _clamp(_IAS_I, -8.0, 8.0)

    throttle_base = 0.48 + ias_err * 0.009 + _IAS_I * 0.0015

    if speed_protect:
        throttle_base -= 0.25
    if stall_protect:
        throttle_base += 0.18

    throttle = _clamp(throttle_base, 0.0, 1.0)

    if not at_engaged:
        throttle = state.get("throttle_1_pos", throttle)

    # Asymmetric thrust handling for throttles
    if eng_fail:
        if not eng1_run:
            throttle1 = min(throttle * 1.15, 1.0)
            throttle2 = throttle * 0.65
        elif not eng2_run:
            throttle2 = min(throttle * 1.15, 1.0)
            throttle1 = throttle * 0.65
        else:
            throttle1 = throttle2 = throttle
    else:
        throttle1 = throttle2 = throttle

    # === FLIGHT PHASE LOGIC (flaps / gear / speedbrake) ===
    radar_alt = min(state.get("radar_alt_1_ft", 5000),
                    state.get("radar_alt_2_ft", 5000),
                    state.get("radar_alt_3_ft", 5000))

    # Gear logic
    gear_cmd = None
    current_gear = state.get("gear_handle_down", 0)
    if radar_alt < 800 and alt < 3500 and not current_gear:
        gear_cmd = 1
    elif radar_alt > 1200 and alt > 4500:
        gear_cmd = 0

    # Flaps logic (conservative)
    flaps_cmd = None
    current_flaps = state.get("flap_pos_left_deg", 0)
    if radar_alt < 2500 and ias < 210:
        if current_flaps < 15:
            flaps_cmd = 15
    elif radar_alt > 4000 and ias > 220:
        if current_flaps > 5:
            flaps_cmd = 5
    elif ias > 280 or alt > 12000:
        flaps_cmd = 0

    # Speedbrake (only when needed)
    speedbrake_cmd = 0.0
    if alt_err < -1200 and vs < -1200 and ias > 240:
        speedbrake_cmd = 0.25
    if speed_protect:
        speedbrake_cmd = max(speedbrake_cmd, 0.18)

    # === BUILD OUTPUT ===
    out = {
        "elevator": elevator,
        "aileron": aileron,
        "rudder": rudder,
        "throttle_1": throttle1,
        "throttle_2": throttle2,
        "elevator_trim": _ELEV_TRIM,
    }

    if flaps_cmd is not None:
        out["flaps"] = float(flaps_cmd)
    if gear_cmd is not None:
        out["gear"] = float(gear_cmd)
    if speedbrake_cmd > 0.05:
        out["speedbrake"] = speedbrake_cmd

    # Very conservative fallback if AP disengaged
    if not ap_engaged:
        out = {
            "elevator": 0.0,
            "aileron": 0.0,
            "rudder": rudder * 0.3,
            "throttle_1": state.get("throttle_1_pos", 0.5),
            "throttle_2": state.get("throttle_2_pos", 0.5),
        }

    return out
```
