```rust
use std::io::{self, BufRead};

const DIRECTIONS: [(isize, isize); 4] = [(0, 1), (1, 0), (0, -1), (-1, 0)];

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read first line: width height obstacle_count require_cycle
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let w: usize = parts.next().unwrap().parse().unwrap();
    let h: usize = parts.next().unwrap().parse().unwrap();
    let obs_count: usize = parts.next().unwrap().parse().unwrap();
    let require_cycle: bool = parts.next().unwrap().parse::<u8>().unwrap() == 1;

    // Read obstacles
    let mut blocked = vec![vec![false; h]; w];
    for _ in 0..obs_count {
        let line = lines.next().unwrap().unwrap();
        let mut xy = line.split_whitespace();
        let x: usize = xy.next().unwrap().parse().unwrap();
        let y: usize = xy.next().unwrap().parse().unwrap();
        if x < w && y < h {
            blocked[x][y] = true;
        }
    }

    // Check starting cell
    if w == 0 || h == 0 || blocked[0][0] {
        println!("0");
        return;
    }

    // Count free cells
    let mut total_free: usize = 0;
    for x in 0..w {
        for y in 0..h {
            if !blocked[x][y] {
                total_free += 1;
            }
        }
    }

    if total_free == 0 {
        println!("0");
        return;
    }

    // Visited and path
    let mut visited = vec![vec![false; h]; w];
    visited[0][0] = true;
    let mut path: Vec<(usize, usize)> = Vec::with_capacity(total_free);
    path.push((0, 0));

    // Recursive solver
    fn solve(
        x: usize,
        y: usize,
        path: &mut Vec<(usize, usize)>,
        visited: &mut Vec<Vec<bool>>,
        blocked: &Vec<Vec<bool>>,
        w: usize,
        h: usize,
        total_free: usize,
        require_cycle: bool,
    ) -> bool {
        if path.len() == total_free {
            if !require_cycle {
                return true;
            }
            // Check if last cell is adjacent to (0,0)
            let &(lx, ly) = path.last().unwrap();
            for &(dx, dy) in &DIRECTIONS {
                let nx = lx as isize + dx;
                let ny = ly as isize + dy;
                if nx == 0 && ny == 0 {
                    return true;
                }
            }
            return false;
        }

        // Collect candidates
        let mut candidates: Vec<(usize, usize, usize)> = Vec::new();

        for &(dx, dy) in &DIRECTIONS {
            let nx = x as isize + dx;
            let ny = y as isize + dy;
            if nx >= 0 && ny >= 0 {
                let nx = nx as usize;
                let ny = ny as usize;
                if nx < w && ny < h && !blocked[nx][ny] && !visited[nx][ny] {
                    // Compute degree for heuristic (number of unvisited neighbors)
                    let mut degree: usize = 0;
                    for &(dx2, dy2) in &DIRECTIONS {
                        let nnx = nx as isize + dx2;
                        let nny = ny as isize + dy2;
                        if nnx >= 0 && nny >= 0 {
                            let nnx = nnx as usize;
                            let nny = nny as usize;
                            if nnx < w && nny < h && !blocked[nnx][nny] && !visited[nnx][nny] {
                                degree += 1;
                            }
                        }
                    }
                    candidates.push((nx, ny, degree));
                }
            }
        }

        // Sort by ascending degree (Warnsdorff heuristic)
        candidates.sort_by_key(|&(_, _, deg)| deg);

        for &(nx, ny, _) in &candidates {
            visited[nx][ny] = true;
            path.push((nx, ny));

            if solve(nx, ny, path, visited, blocked, w, h, total_free, require_cycle) {
                return true;
            }

            path.pop();
            visited[nx][ny] = false;
        }

        false
    }

    let found = solve(0, 0, &mut path, &mut visited, &blocked, w, h, total_free, require_cycle);

    if found {
        println!("{}", path.len());
        for &(x, y) in &path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}
```
