```ts
declare const require: any;

const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Package {
    id: any;
    x: number;
    y: number;
}

interface InputData {
    depot: [number, number];
    truckCount: number;
    packages: Package[];
}

const inputData: InputData = data;

const depotPoint = { x: inputData.depot[0], y: inputData.depot[1] };
const K = Math.max(1, Math.min(inputData.truckCount, inputData.packages.length || 0));
const packages: Package[] = inputData.packages || [];

function distance(a: { x: number; y: number }, b: { x: number; y: number }): number {
    return Math.hypot(a.x - b.x, a.y - b.y);
}

function getAngle(p: { x: number; y: number }, center: { x: number; y: number }): number {
    return Math.atan2(p.y - center.y, p.x - center.x);
}

function orderRoute(cluster: Package[], depot: { x: number; y: number }): any[] {
    if (cluster.length === 0) return [];
    
    const remaining = [...cluster];
    const route: any[] = [];
    let current = depot;
    
    while (remaining.length > 0) {
        let minDist = Infinity;
        let bestIdx = -1;
        
        for (let i = 0; i < remaining.length; i++) {
            const d = distance(current, remaining[i]);
            if (d < minDist) {
                minDist = d;
                bestIdx = i;
            }
        }
        
        const nextPkg = remaining[bestIdx];
        route.push(nextPkg.id);
        current = nextPkg;
        remaining.splice(bestIdx, 1);
    }
    
    return route;
}

// Sweep clustering by polar angle
const sortedPackages = [...packages].sort((a, b) => {
    const angleA = getAngle(a, depotPoint);
    const angleB = getAngle(b, depotPoint);
    if (angleA !== angleB) return angleA - angleB;
    // Tie-break by distance from depot
    return distance(a, depotPoint) - distance(b, depotPoint);
});

const n = sortedPackages.length;
const routes: any[][] = [];

if (n > 0 && K > 0) {
    const baseSize = Math.floor(n / K);
    const remainder = n % K;
    let start = 0;
    
    for (let i = 0; i < K; i++) {
        const size = baseSize + (i < remainder ? 1 : 0);
        if (size > 0 && start < n) {
            const cluster = sortedPackages.slice(start, start + size);
            const route = orderRoute(cluster, depotPoint);
            if (route.length > 0) {
                routes.push(route);
            }
            start += size;
        }
    }
}

// Ensure all packages are covered (fallback for edge cases)
if (routes.length === 0 && n > 0) {
    // Fallback: all in one route
    const route = orderRoute(sortedPackages, depotPoint);
    if (route.length > 0) routes.push(route);
} else if (n > 0) {
    // Verify coverage and fix if needed (rare)
    const used = new Set<any>();
    for (const r of routes) {
        for (const id of r) used.add(id);
    }
    const missing: Package[] = [];
    for (const p of packages) {
        if (!used.has(p.id)) missing.push(p);
    }
    if (missing.length > 0) {
        const fallbackRoute = orderRoute(missing, depotPoint);
        if (fallbackRoute.length > 0) routes.push(fallbackRoute);
    }
}

const output = { routes };
console.log(JSON.stringify(output));
```
