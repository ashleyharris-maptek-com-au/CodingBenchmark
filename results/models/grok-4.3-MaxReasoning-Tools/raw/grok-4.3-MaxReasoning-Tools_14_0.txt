```csharp
using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        string line = Console.ReadLine();
        if (string.IsNullOrEmpty(line)) return;

        string[] parts = line.Trim().Split(new char[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 2) return;

        int H = int.Parse(parts[0]);
        int W = int.Parse(parts[1]);

        string[] board = new string[H];
        for (int i = 0; i < H; i++)
        {
            board[i] = Console.ReadLine();
            if (board[i] == null) board[i] = "";
        }

        long totalCells = (long)H * W;

        // Adaptive strategy selection based on board complexity
        bool useOptimized = totalCells > 50000; // Large boards: prioritize speed & memory
        bool useThorough = totalCells <= 5000;   // Small boards: thorough local scans

        int[][] dirs = new int[8][]
        {
            new int[] { -1, -1 }, new int[] { -1, 0 }, new int[] { -1, 1 },
            new int[] { 0, -1 },                       new int[] { 0, 1 },
            new int[] { 1, -1 },  new int[] { 1, 0 },  new int[] { 1, 1 }
        };

        HashSet<(int x, int y)> safeSet = new HashSet<(int, int)>();

        // Collect revealed cells
        List<(int r, int c, int num)> revealed = new List<(int r, int c, int num)>();
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                char ch = board[r][c];
                if (ch >= '0' && ch <= '8')
                {
                    revealed.Add((r, c, ch - '0'));
                }
            }
        }

        // Reusable list for neighbors
        List<(int x, int y)> unknowns = new List<(int x, int y)>(8);

        // Strategy: Local constraint deduction (core for all sizes)
        // For very large boards, we process revealed efficiently
        // For small boards, we perform extra passes for potential chaining opportunities (though limited)
        int iterations = useThorough ? 3 : 1;

        for (int iter = 0; iter < iterations; iter++)
        {
            bool changed = false;

            foreach (var (r, c, num) in revealed)
            {
                int flagged = 0;
                unknowns.Clear();

                for (int d = 0; d < 8; d++)
                {
                    int nr = r + dirs[d][0];
                    int nc = c + dirs[d][1];

                    if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                    {
                        char ch = board[nr][nc];
                        if (ch == '*')
                        {
                            flagged++;
                        }
                        else if (ch == ' ')
                        {
                            unknowns.Add((nc, nr)); // (x=col, y=row)
                        }
                    }
                }

                int remaining = num - flagged;

                if (remaining == 0 && unknowns.Count > 0)
                {
                    foreach (var pos in unknowns)
                    {
                        if (safeSet.Add(pos))
                        {
                            changed = true;
                        }
                    }
                }
                // Note: If remaining == unknowns.Count, all are mines (ignored here)
            }

            if (!changed && !useThorough) break;
        }

        // For small boards (low complexity), we could expand here with
        // additional pair-wise or frontier analysis if needed.
        // For extreme boards, the above O(revealed * 8) is highly efficient.

        // Output results
        Console.WriteLine(safeSet.Count);
        foreach (var (x, y) in safeSet)
        {
            Console.WriteLine($"{x} {y}");
        }
    }
}
```
