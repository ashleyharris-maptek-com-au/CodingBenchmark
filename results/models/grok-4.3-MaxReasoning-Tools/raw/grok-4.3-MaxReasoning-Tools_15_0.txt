use std::io::{self, BufRead};

fn point_in_polygon(px: f64, py: f64, poly: &[(f64, f64)]) -> bool {
    if poly.is_empty() {
        return false;
    }
    let mut inside = false;
    let n = poly.len();
    let mut j = n - 1;
    for i in 0..n {
        let (xi, yi) = poly[i];
        let (xj, yj) = poly[j];
        if ((yi > py) != (yj > py)) && (px < (xj - xi) * (py - yi) / (yj - yi) + xi) {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let nv_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => return,
    };
    let nv: usize = match nv_line.trim().parse() {
        Ok(v) => v,
        _ => return,
    };

    let mut poly: Vec<(f64, f64)> = Vec::with_capacity(nv);
    for _ in 0..nv {
        let line = match lines.next() {
            Some(Ok(l)) => l,
            _ => return,
        };
        let mut parts = line.trim().split_whitespace();
        let x: f64 = match parts.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            _ => return,
        };
        let y: f64 = match parts.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            _ => return,
        };
        poly.push((x, y));
    }

    let sun_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => return,
    };
    let mut parts = sun_line.trim().split_whitespace();
    let sx: f64 = match parts.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        _ => return,
    };
    let sy: f64 = match parts.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        _ => return,
    };
    let sz: f64 = match parts.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        _ => return,
    };

    if poly.is_empty() {
        println!("0");
        return;
    }

    // Compute bounding box
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for &(x, y) in &poly {
        if x < min_x {
            min_x = x;
        }
        if x > max_x {
            max_x = x;
        }
        if y < min_y {
            min_y = y;
        }
        if y > max_y {
            max_y = y;
        }
    }

    let width = max_x - min_x;
    let height = max_y - min_y;
    let bbox_size = width.max(height).max(1.0);

    // Adaptive logic based on problem scale (bbox_size)
    let (tetra_scale, step, num_layers, layer_height): (f64, f64, usize, f64) = if bbox_size < 8.0 {
        (3.0, 2.0, 1, 0.0)
    } else if bbox_size < 25.0 {
        (2.5, 2.2, 2, 5.0)
    } else if bbox_size < 80.0 {
        (2.0, 2.0, 2, 7.0)
    } else {
        (1.8, 1.9, 3, 8.0)
    };

    let mut tetras: Vec<(f64, f64, f64, f64, f64, f64, f64, f64)> = Vec::new();

    // Generate grid points over bbox with margin
    let margin = step * 1.5;
    let mut x = min_x - margin;
    let mut idx = 0usize;
    while x <= max_x + margin {
        let mut y = min_y - margin;
        while y <= max_y + margin {
            let inside = point_in_polygon(x, y, &poly);
            if inside {
                let layer = idx % num_layers;
                let pz = layer as f64 * layer_height;

                // Compensate for sun direction to center shadow at (x, y)
                let t = if sz.abs() > 1e-9 {
                    -pz / sz
                } else {
                    0.0
                };
                let cx = x - t * sx;
                let cy = y - t * sy;

                let qw = 1.0;
                let qx = 0.0;
                let qy = 0.0;
                let qz = 0.0;
                tetras.push((cx, cy, pz, qw, qx, qy, qz, tetra_scale));
                idx += 1;
            }
            y += step;
        }
        x += step;
    }

    // Ensure at least one tetra for very small/edge cases
    if tetras.is_empty() {
        let cx = (min_x + max_x) / 2.0;
        let cy = (min_y + max_y) / 2.0;
        let pz = 0.0;
        let qw = 1.0;
        let qx = 0.0;
        let qy = 0.0;
        let qz = 0.0;
        tetras.push((cx, cy, pz, qw, qx, qy, qz, tetra_scale.min(4.0)));
    }

    // Output
    println!("{}", tetras.len());
    for t in &tetras {
        println!(
            "{:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6}",
            t.0, t.1, t.2, t.3, t.4, t.5, t.6, t.7
        );
    }
}
