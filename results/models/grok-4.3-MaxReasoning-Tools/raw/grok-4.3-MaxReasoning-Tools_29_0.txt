```csharp
using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static int n;
    static List<int>[] adj;
    static bool[,] adjMat;
    static int maxSize = 0;
    static List<int> bestClique;

    static void Main()
    {
        string input = Console.In.ReadToEnd();
        string[] tokens = input.Split(new char[] { ' ', '\r', '\n', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int index = 0;
        n = int.Parse(tokens[index++]);
        int m = int.Parse(tokens[index++]);

        adj = new List<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            int u = int.Parse(tokens[index++]);
            int v = int.Parse(tokens[index++]);
            if (u != v)
            {
                if (!adj[u].Contains(v))
                {
                    adj[u].Add(v);
                    adj[v].Add(u);
                }
            }
        }

        for (int i = 0; i < n; i++)
        {
            adj[i] = adj[i].Distinct().ToList();
            adj[i].Sort();
        }

        if (n <= 90)
        {
            adjMat = new bool[n, n];
            for (int i = 0; i < n; i++)
            {
                foreach (int nei in adj[i])
                {
                    adjMat[i, nei] = true;
                }
            }
            FindExact();
        }
        else
        {
            FindHeuristic();
        }

        Console.WriteLine(maxSize);
        if (bestClique != null && bestClique.Count > 0)
        {
            Console.WriteLine(string.Join(" ", bestClique));
        }
        else
        {
            Console.WriteLine(1);
            Console.WriteLine("0");
        }
    }

    static bool IsConnected(int u, int v)
    {
        if (adjMat != null) return adjMat[u, v];
        return adj[u].BinarySearch(v) >= 0;
    }

    static void FindExact()
    {
        maxSize = 0;
        bestClique = null;

        GreedyClique();

        int[] degrees = new int[n];
        for (int i = 0; i < n; i++) degrees[i] = adj[i].Count;

        List<int> ordered = Enumerable.Range(0, n)
            .OrderByDescending(i => degrees[i])
            .ToList();

        Search(new List<int>(), ordered);
    }

    static void GreedyClique()
    {
        int[] degrees = new int[n];
        for (int i = 0; i < n; i++) degrees[i] = adj[i].Count;

        List<int> orderedVerts = Enumerable.Range(0, n)
            .OrderByDescending(v => degrees[v])
            .ToList();

        List<int> clique = new List<int>();
        foreach (int v in orderedVerts)
        {
            bool canAdd = true;
            foreach (int u in clique)
            {
                if (!IsConnected(u, v))
                {
                    canAdd = false;
                    break;
                }
            }
            if (canAdd)
            {
                clique.Add(v);
            }
        }

        if (clique.Count > maxSize)
        {
            maxSize = clique.Count;
            bestClique = new List<int>(clique);
        }
    }

    static void Search(List<int> clique, List<int> candidates)
    {
        if (clique.Count > maxSize)
        {
            maxSize = clique.Count;
            bestClique = new List<int>(clique);
        }

        if (clique.Count + candidates.Count <= maxSize) return;

        for (int i = 0; i < candidates.Count; i++)
        {
            int v = candidates[i];

            List<int> newCands = new List<int>();
            for (int j = i + 1; j < candidates.Count; j++)
            {
                int u = candidates[j];
                if (IsConnected(v, u))
                {
                    newCands.Add(u);
                }
            }

            if (clique.Count + 1 + newCands.Count <= maxSize) continue;

            clique.Add(v);
            Search(clique, newCands);
            clique.RemoveAt(clique.Count - 1);
        }
    }

    static void FindHeuristic()
    {
        GreedyClique();
    }
}
```
