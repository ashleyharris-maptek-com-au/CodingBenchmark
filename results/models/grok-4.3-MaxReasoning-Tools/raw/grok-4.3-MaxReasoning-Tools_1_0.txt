#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <limits>
#include <numeric>
#include <cstdint>

struct Point {
    double x, y;
};

double dist(const std::vector<Point>& pts, int i, int j) {
    double dx = pts[i].x - pts[j].x;
    double dy = pts[i].y - pts[j].y;
    return std::hypot(dx, dy);
}

std::vector<int> held_karp(const std::vector<Point>& pts) {
    int n = pts.size();
    if (n == 0) return {};
    if (n == 1) return {0};
    const double INF = 1e30;
    int max_mask = 1 << n;
    std::vector<std::vector<double>> dp(max_mask, std::vector<double>(n, INF));
    std::vector<std::vector<int>> predecessor(max_mask, std::vector<int>(n, -1));
    int start = 0;
    dp[1 << start][start] = 0.0;
    for (int mask = 0; mask < max_mask; ++mask) {
        for (int u = 0; u < n; ++u) {
            if ((mask & (1 << u)) == 0 || dp[mask][u] >= INF) continue;
            for (int v = 0; v < n; ++v) {
                if ((mask & (1 << v)) != 0) continue;
                double cost = dp[mask][u] + dist(pts, u, v);
                int nmask = mask | (1 << v);
                if (cost < dp[nmask][v]) {
                    dp[nmask][v] = cost;
                    predecessor[nmask][v] = u;
                }
            }
        }
    }
    int full_mask = (1 << n) - 1;
    double min_cost = INF;
    int last = -1;
    for (int u = 0; u < n; ++u) {
        if (dp[full_mask][u] < min_cost) {
            min_cost = dp[full_mask][u];
            last = u;
        }
    }
    std::vector<int> path;
    int mask = full_mask;
    int curr = last;
    while (curr != -1) {
        path.push_back(curr);
        int prev_u = predecessor[mask][curr];
        mask &= ~(1 << curr);
        curr = prev_u;
    }
    std::reverse(path.begin(), path.end());
    return path;
}

std::vector<int> nearest_neighbor(const std::vector<Point>& pts) {
    int n = pts.size();
    if (n == 0) return {};
    std::vector<int> tour;
    tour.reserve(n);
    std::vector<char> visited(n, 0);
    int start = 0;
    visited[start] = 1;
    tour.push_back(start);
    for (int step = 1; step < n; ++step) {
        int current = tour.back();
        int next = -1;
        double min_dist = std::numeric_limits<double>::max();
        for (int j = 0; j < n; ++j) {
            if (!visited[j]) {
                double d = dist(pts, current, j);
                if (d < min_dist) {
                    min_dist = d;
                    next = j;
                }
            }
        }
        tour.push_back(next);
        visited[next] = 1;
    }
    return tour;
}

std::vector<int> two_opt_improve(std::vector<int> tour, const std::vector<Point>& pts) {
    int n = tour.size();
    if (n <= 3) return tour;
    bool improved = true;
    int max_passes = std::min(30, n / 10 + 5);
    int passes = 0;
    while (improved && passes < max_passes) {
        improved = false;
        passes++;
        for (int i = 0; i < n - 1; ++i) {
            for (int j = i + 2; j < n - 1; ++j) {  // avoid wrap issues
                double old1 = dist(pts, tour[i], tour[i + 1]);
                double old2 = dist(pts, tour[j], tour[j + 1]);
                double new1 = dist(pts, tour[i], tour[j]);
                double new2 = dist(pts, tour[i + 1], tour[j + 1]);
                if (new1 + new2 < old1 + old2 - 1e-9) {
                    std::reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
    return tour;
}

std::vector<int> angular_sort_tour(const std::vector<Point>& pts) {
    int n = pts.size();
    if (n == 0) return {};
    if (n == 1) return {0};
    double cx = 0.0, cy = 0.0;
    for (const auto& p : pts) {
        cx += p.x;
        cy += p.y;
    }
    cx /= n;
    cy /= n;
    std::vector<int> idx(n);
    std::iota(idx.begin(), idx.end(), 0);
    std::sort(idx.begin(), idx.end(), [&](int a, int b) {
        double da = std::atan2(pts[a].y - cy, pts[a].x - cx);
        double db = std::atan2(pts[b].y - cy, pts[b].x - cx);
        if (std::abs(da - db) < 1e-9) {
            return dist(pts, a, 0) < dist(pts, b, 0);
        }
        return da < db;
    });
    auto it = std::find(idx.begin(), idx.end(), 0);
    if (it != idx.end()) {
        std::rotate(idx.begin(), it, idx.end());
    }
    return idx;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    int N;
    std::cin >> N;
    std::vector<Point> pts(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> pts[i].x >> pts[i].y;
    }
    std::vector<int> tour;
    if (N <= 20) {
        tour = held_karp(pts);
    } else if (N <= 3000) {
        tour = nearest_neighbor(pts);
        if (N <= 800) {
            tour = two_opt_improve(tour, pts);
        }
    } else {
        tour = angular_sort_tour(pts);
    }
    // Ensure starts with 0 and rotate if necessary
    if (!tour.empty() && tour[0] != 0) {
        auto pos = std::find(tour.begin(), tour.end(), 0);
        if (pos != tour.end()) {
            std::rotate(tour.begin(), pos, tour.end());
        }
    }
    // Output
    for (size_t i = 0; i < tour.size(); ++i) {
        if (i > 0) std::cout << " ";
        std::cout << tour[i];
    }
    std::cout << "\n";
    return 0;
}
