import sys

def main():
    sys.setrecursionlimit(100000)
    data = sys.stdin.read().split()
    it = iter(data)
    try:
        N = int(next(it))
        M = int(next(it))
    except StopIteration:
        print("SAT")
        print("")
        return

    clauses = []
    for _ in range(M):
        try:
            l1 = int(next(it))
            l2 = int(next(it))
            l3 = int(next(it))
            clauses.append((l1, l2, l3))
        except StopIteration:
            break

    if not clauses:
        print("SAT")
        print(" ".join(["0"] * N) if N > 0 else "")
        return

    assignment = [0] * (N + 1)

    # Precompute max variable for each clause
    max_var_per_clause = []
    for clause in clauses:
        mv = max(abs(lit) for lit in clause)
        max_var_per_clause.append(mv)

    # Group clauses by their max variable
    clauses_by_max = [[] for _ in range(N + 1)]
    for i in range(len(clauses)):
        mv = max_var_per_clause[i]
        if mv <= N:
            clauses_by_max[mv].append(i)

    def is_clause_satisfied(clause):
        for lit in clause:
            v = abs(lit)
            val = assignment[v]
            if (lit > 0 and val == 1) or (lit < 0 and val == 0):
                return True
        return False

    def bt(var):
        if var > N:
            return True

        # Try True (1)
        assignment[var] = 1
        good = True
        for cidx in clauses_by_max[var]:
            if not is_clause_satisfied(clauses[cidx]):
                good = False
                break
        if good and bt(var + 1):
            return True

        # Try False (0)
        assignment[var] = 0
        good = True
        for cidx in clauses_by_max[var]:
            if not is_clause_satisfied(clauses[cidx]):
                good = False
                break
        if good and bt(var + 1):
            return True

        return False

    success = bt(1)

    if success:
        print("SAT")
        print(" ".join(str(assignment[i]) for i in range(1, N + 1)))
    else:
        # Fallback (should not happen as guaranteed satisfiable)
        print("SAT")
        print(" ".join(["1"] * N))

if __name__ == "__main__":
    main()
