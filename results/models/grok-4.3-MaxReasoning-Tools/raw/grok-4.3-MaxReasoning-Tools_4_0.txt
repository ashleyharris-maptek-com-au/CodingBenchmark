#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <iomanip>

struct Point3D {
    double x, y, z;
    Point3D(double x = 0, double y = 0, double z = 0) : x(x), y(y), z(z) {}
};

Point3D operator+(const Point3D& a, const Point3D& b) {
    return {a.x + b.x, a.y + b.y, a.z + b.z};
}

Point3D operator-(const Point3D& a, const Point3D& b) {
    return {a.x - b.x, a.y - b.y, a.z - b.z};
}

Point3D operator*(const Point3D& a, double s) {
    return {a.x * s, a.y * s, a.z * s};
}

double dot(const Point3D& a, const Point3D& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

double norm(const Point3D& a) {
    return std::sqrt(dot(a, a));
}

Point3D cross(const Point3D& a, const Point3D& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

struct Plane {
    Point3D point;
    Point3D normal;
};

struct PlacedTetra {
    Point3D center;
    double rx, ry, rz;
};

struct Orient {
    double rx, ry, rz;
};

std::vector<Point3D> getTetraVertices(const Point3D& center, double rx, double ry, double rz, double edge) {
    double s = edge / (2.0 * std::sqrt(2.0));
    std::vector<Point3D> base = {
        {s, s, s},
        {s, -s, -s},
        {-s, s, -s},
        {-s, -s, s}
    };
    std::vector<Point3D> verts;
    verts.reserve(4);
    for (auto p : base) {
        // Rotate X
        double y1 = p.y * std::cos(rx) - p.z * std::sin(rx);
        double z1 = p.y * std::sin(rx) + p.z * std::cos(rx);
        p.y = y1;
        p.z = z1;
        // Rotate Y
        double x2 = p.x * std::cos(ry) + p.z * std::sin(ry);
        double z2 = -p.x * std::sin(ry) + p.z * std::cos(ry);
        p.x = x2;
        p.z = z2;
        // Rotate Z
        double x3 = p.x * std::cos(rz) - p.y * std::sin(rz);
        double y3 = p.x * std::sin(rz) + p.y * std::cos(rz);
        p.x = x3;
        p.y = y3;
        verts.push_back({p.x + center.x, p.y + center.y, p.z + center.z});
    }
    return verts;
}

bool isInsideAABB(const Point3D& p, double minx, double maxx, double miny, double maxy, double minz, double maxz) {
    return p.x >= minx && p.x <= maxx &&
           p.y >= miny && p.y <= maxy &&
           p.z >= minz && p.z <= maxz;
}

bool isInsideConvex(const Point3D& p, const std::vector<Plane>& planes) {
    for (const auto& pl : planes) {
        double d = dot(pl.normal, p - pl.point);
        if (d > 1e-5) return false;
    }
    return true;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    int V;
    double edge;
    std::cin >> V >> edge;
    if (V <= 0 || edge <= 0) {
        std::cout << 0 << "\n";
        return 0;
    }
    std::vector<Point3D> container(V);
    double minx = 1e18, maxx = -1e18;
    double miny = 1e18, maxy = -1e18;
    double minz = 1e18, maxz = -1e18;
    for (int i = 0; i < V; i++) {
        std::cin >> container[i].x >> container[i].y >> container[i].z;
        minx = std::min(minx, container[i].x);
        maxx = std::max(maxx, container[i].x);
        miny = std::min(miny, container[i].y);
        maxy = std::max(maxy, container[i].y);
        minz = std::min(minz, container[i].z);
        maxz = std::max(maxz, container[i].z);
    }
    Point3D hull_center{0, 0, 0};
    for (const auto& p : container) {
        hull_center.x += p.x;
        hull_center.y += p.y;
        hull_center.z += p.z;
    }
    if (V > 0) {
        hull_center.x /= V;
        hull_center.y /= V;
        hull_center.z /= V;
    }
    std::vector<Plane> hull_planes;
    if (V >= 4) {
        for (int i = 0; i < V; i++) {
            for (int j = i + 1; j < V; j++) {
                for (int k = j + 1; k < V; k++) {
                    Point3D a = container[i];
                    Point3D b = container[j];
                    Point3D c = container[k];
                    Point3D ab = b - a;
                    Point3D ac = c - a;
                    Point3D n = cross(ab, ac);
                    double nlen = norm(n);
                    if (nlen < 1e-8) continue;
                    n = n * (1.0 / nlen);
                    bool on_one_side = true;
                    int first_side = 0;
                    for (int m = 0; m < V && on_one_side; m++) {
                        if (m == i || m == j || m == k) continue;
                        double d = dot(n, container[m] - a);
                        if (std::abs(d) < 1e-7) continue;
                        int s = (d > 0 ? 1 : -1);
                        if (first_side == 0) first_side = s;
                        else if (s != first_side) {
                            on_one_side = false;
                        }
                    }
                    if (on_one_side) {
                        double dcent = dot(n, hull_center - a);
                        if (dcent > 0) {
                            n = n * (-1.0);
                        }
                        hull_planes.push_back({a, n});
                    }
                }
            }
        }
    }
    bool use_convex = !hull_planes.empty();
    const double PI = std::acos(-1.0);
    std::vector<Orient> orients = {
        {0.0, 0.0, 0.0},
        {PI, 0.0, 0.0},
        {0.0, PI, 0.0},
        {0.0, 0.0, PI}
    };
    std::vector<PlacedTetra> placed;
    double R = std::sqrt(6.0) / 4.0 * edge;
    double base_step = edge * 0.92;
    double Lx = maxx - minx;
    double Ly = maxy - miny;
    double Lz = maxz - minz;
    double maxL = std::max({Lx, Ly, Lz});
    int max_per_dim = 140;
    double step = std::max(base_step, maxL / max_per_dim);
    double min_center_dist = edge * 1.12;
    double start_margin = R * 1.05;
    const size_t max_recent_checks = 1200;
    for (double cx = minx + start_margin; cx <= maxx - start_margin + 1e-9; cx += step) {
        for (double cy = miny + start_margin; cy <= maxy - start_margin + 1e-9; cy += step) {
            for (double cz = minz + start_margin; cz <= maxz - start_margin + 1e-9; cz += step) {
                bool placed_any = false;
                for (const auto& o : orients) {
                    auto verts = getTetraVertices({cx, cy, cz}, o.rx, o.ry, o.rz, edge);
                    bool inside = true;
                    for (const auto& v : verts) {
                        if (use_convex) {
                            if (!isInsideConvex(v, hull_planes)) {
                                inside = false;
                                break;
                            }
                        } else {
                            if (!isInsideAABB(v, minx, maxx, miny, maxy, minz, maxz)) {
                                inside = false;
                                break;
                            }
                        }
                    }
                    if (!inside) continue;
                    bool overlap = false;
                    size_t start_j = (placed.size() > max_recent_checks ? placed.size() - max_recent_checks : 0);
                    for (size_t j = start_j; j < placed.size(); ++j) {
                        Point3D diff = placed[j].center - Point3D{cx, cy, cz};
                        if (norm(diff) < min_center_dist) {
                            overlap = true;
                            break;
                        }
                    }
                    if (!overlap) {
                        placed.push_back({{cx, cy, cz}, o.rx, o.ry, o.rz});
                        placed_any = true;
                        break;
                    }
                }
                if (placed.size() > 500000) break; // safety for output size
            }
            if (placed.size() > 500000) break;
        }
        if (placed.size() > 500000) break;
    }
    std::cout << placed.size() << "\n";
    std::cout << std::fixed << std::setprecision(10);
    for (const auto& p : placed) {
        std::cout << p.center.x << " " << p.center.y << " " << p.center.z << " "
                  << p.rx << " " << p.ry << " " << p.rz << "\n";
    }
    return 0;
}
