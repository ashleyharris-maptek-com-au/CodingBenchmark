#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <iomanip>
#include <limits>
#include <array>

using namespace std;

struct Vec3 {
    double x, y, z;
};

Vec3 add(const Vec3& a, const Vec3& b) {
    return {a.x + b.x, a.y + b.y, a.z + b.z};
}

double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

double norm(const Vec3& v) {
    return sqrt(dot(v, v));
}

Vec3 rotate(Vec3 p, double rx, double ry, double rz) {
    // Rotate around X
    {
        double c = cos(rx);
        double s = sin(rx);
        double ny = p.y * c - p.z * s;
        double nz = p.y * s + p.z * c;
        p.y = ny;
        p.z = nz;
    }
    // Rotate around Y
    {
        double c = cos(ry);
        double s = sin(ry);
        double nx = p.x * c + p.z * s;
        double nz = -p.x * s + p.z * c;
        p.x = nx;
        p.z = nz;
    }
    // Rotate around Z
    {
        double c = cos(rz);
        double s = sin(rz);
        double nx = p.x * c - p.y * s;
        double ny = p.x * s + p.y * c;
        p.x = nx;
        p.y = ny;
    }
    return p;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    
    int V;
    double e;
    cin >> V >> e;
    
    vector<Vec3> vertices(V);
    double minx = numeric_limits<double>::max();
    double miny = numeric_limits<double>::max();
    double minz = numeric_limits<double>::max();
    double maxx = numeric_limits<double>::lowest();
    double maxy = numeric_limits<double>::lowest();
    double maxz = numeric_limits<double>::lowest();
    
    for (int i = 0; i < V; i++) {
        cin >> vertices[i].x >> vertices[i].y >> vertices[i].z;
        minx = min(minx, vertices[i].x);
        maxx = max(maxx, vertices[i].x);
        miny = min(miny, vertices[i].y);
        maxy = max(maxy, vertices[i].y);
        minz = min(minz, vertices[i].z);
        maxz = max(maxz, vertices[i].z);
    }
    
    if (V == 0 || e <= 0) {
        cout << 0 << endl;
        return 0;
    }
    
    double s = e / (2.0 * sqrt(2.0));
    vector<Vec3> default_verts = {
        {s, s, s},
        {s, -s, -s},
        {-s, s, -s},
        {-s, -s, s}
    };
    
    double R = e * sqrt(3.0 / 8.0);
    double min_center_dist = 2.0 * R * 1.12;
    double base_spacing = min_center_dist;
    
    double dx = maxx - minx;
    double dy = maxy - miny;
    double dz = maxz - minz;
    
    long long est_steps = 1;
    if (dx > 0 && dy > 0 && dz > 0) {
        double nx = dx / base_spacing + 2;
        double ny = dy / base_spacing + 2;
        double nz = dz / base_spacing + 2;
        if (nx > 1e9 || ny > 1e9 || nz > 1e9) {
            est_steps = 10000000LL;
        } else {
            est_steps = (long long)(nx * ny * nz);
        }
    }
    
    double spacing = base_spacing;
    const long long MAX_CANDIDATES = 2000000LL;
    if (est_steps > MAX_CANDIDATES) {
        double scale = cbrt(static_cast<double>(est_steps) / MAX_CANDIDATES);
        spacing = base_spacing * max(scale, 1.0);
    }
    
    double offset = spacing * 0.5;
    
    vector<pair<Vec3, array<double, 3>>> placements;
    
    // Grid generation
    for (double x = minx + offset; x <= maxx - offset + 1e-9; x += spacing) {
        for (double y = miny + offset; y <= maxy - offset + 1e-9; y += spacing) {
            for (double z = minz + offset; z <= maxz - offset + 1e-9; z += spacing) {
                Vec3 c = {x, y, z};
                
                bool inside = true;
                for (const auto& dv : default_verts) {
                    Vec3 v = add(rotate(dv, 0.0, 0.0, 0.0), c);
                    if (v.x < minx || v.x > maxx ||
                        v.y < miny || v.y > maxy ||
                        v.z < minz || v.z > maxz) {
                        inside = false;
                        break;
                    }
                }
                
                if (inside) {
                    placements.emplace_back(c, array<double, 3>{0.0, 0.0, 0.0});
                }
            }
        }
    }
    
    // Output
    cout << placements.size() << endl;
    cout << fixed << setprecision(8);
    
    for (const auto& p : placements) {
        const Vec3& c = p.first;
        const auto& angles = p.second;
        cout << c.x << " " << c.y << " " << c.z << " "
             << angles[0] << " " << angles[1] << " " << angles[2] << "\n";
    }
    
    return 0;
}
