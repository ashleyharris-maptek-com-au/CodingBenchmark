```rust
use std::collections::VecDeque;
use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut lines = input.lines();

    let first_line = lines.next().unwrap();
    let mut parts = first_line.split_whitespace();
    let h: usize = parts.next().unwrap().parse().unwrap();
    let w: usize = parts.next().unwrap().parse().unwrap();

    let mut maze: Vec<Vec<u8>> = Vec::with_capacity(h);
    let mut start_r: usize = 0;
    let mut start_c: usize = 0;
    let mut goal_r: usize = 0;
    let mut goal_c: usize = 0;

    for r in 0..h {
        let line = lines.next().unwrap();
        let bytes = line.as_bytes();
        let mut row = vec![0u8; w];
        for c in 0..w {
            if c < bytes.len() {
                row[c] = bytes[c];
                if bytes[c] == b'A' {
                    start_r = r;
                    start_c = c;
                } else if bytes[c] == b'B' {
                    goal_r = r;
                    goal_c = c;
                }
            }
        }
        maze.push(row);
    }

    let start = start_r * w + start_c;
    let goal = goal_r * w + goal_c;

    if start == goal {
        println!("1");
        println!("{} {}", start_c, start_r);
        return;
    }

    let total = h * w;
    let mut vis: Vec<u8> = vec![0u8; total];
    let mut parent: Vec<u32> = vec![u32::MAX; total];
    let mut q: VecDeque<usize> = VecDeque::new();

    vis[start] = 1;
    parent[start] = start as u32;
    q.push_back(start);

    let dirs: [(isize, isize); 4] = [(-1, 0), (1, 0), (0, -1), (0, 1)];

    let mut found = false;

    while let Some(curr) = q.pop_front() {
        if curr == goal {
            found = true;
            break;
        }
        let r = curr / w;
        let c = curr % w;
        for &(dr, dc) in &dirs {
            let nr = r as isize + dr;
            let nc = c as isize + dc;
            if nr >= 0 && nr < h as isize && nc >= 0 && nc < w as isize {
                let nr_usize = nr as usize;
                let nc_usize = nc as usize;
                let ni = nr_usize * w + nc_usize;
                if vis[ni] == 0 && maze[nr_usize][nc_usize] != b'#' {
                    vis[ni] = 1;
                    parent[ni] = curr as u32;
                    q.push_back(ni);
                }
            }
        }
    }

    if !found {
        println!("0");
        return;
    }

    let mut path: Vec<usize> = Vec::new();
    let mut current = goal;
    while current != start {
        path.push(current);
        current = parent[current] as usize;
    }
    path.push(start);
    path.reverse();

    let n = path.len();
    println!("{}", n);
    for &p in &path {
        let r = p / w;
        let c = p % w;
        println!("{} {}", c, r);
    }
}
```
