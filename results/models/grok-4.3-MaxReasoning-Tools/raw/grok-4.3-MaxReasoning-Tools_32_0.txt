```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#include <map>
#include <set>
#include <tuple>
#include <limits>
#include <numeric>

using namespace std;
using ll = long long;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n, m, k;
  cin >> n >> m >> k;

  vector<vector<pair<int, ll>>> g(n);
  map<pair<int, int>, ll> edge_weight;

  for (int i = 0; i < m; i++) {
    int u, v;
    ll w;
    cin >> u >> v >> w;
    int a = min(u, v);
    int b = max(u, v);
    auto key = make_pair(a, b);
    if (edge_weight.count(key)) {
      edge_weight[key] = min(edge_weight[key], w);
    } else {
      edge_weight[key] = w;
    }
    g[u].emplace_back(v, w);
    g[v].emplace_back(u, w);
  }

  vector<int> terminals(k);
  for (int i = 0; i < k; i++) {
    cin >> terminals[i];
  }

  if (k <= 1) {
    cout << 0 << " " << 0 << "\n";
    return 0;
  }

  const ll INF = 4e18;
  vector<vector<ll>> term_dist(k, vector<ll>(k, INF));
  vector<vector<int>> prevs(k, vector<int>(n, -1));

  for (int s = 0; s < k; s++) {
    int src = terminals[s];
    vector<ll> d(n, INF);
    vector<int> prev(n, -1);
    priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
    d[src] = 0;
    pq.emplace(0, src);
    while (!pq.empty()) {
      auto [cost, u] = pq.top();
      pq.pop();
      if (cost > d[u]) continue;
      for (auto& p : g[u]) {
        int v = p.first;
        ll w = p.second;
        if (d[v] > d[u] + w) {
          d[v] = d[u] + w;
          prev[v] = u;
          pq.emplace(d[v], v);
        }
      }
    }
    prevs[s] = prev;
    for (int t = 0; t < k; t++) {
      term_dist[s][t] = d[terminals[t]];
    }
  }

  struct TermEdge {
    int i, j;
    ll w;
  };
  vector<TermEdge> term_edges;
  for (int i = 0; i < k; i++) {
    for (int j = i + 1; j < k; j++) {
      if (term_dist[i][j] < INF) {
        term_edges.push_back({i, j, term_dist[i][j]});
      }
    }
  }
  sort(term_edges.begin(), term_edges.end(), [](const TermEdge& a, const TermEdge& b) {
    return a.w < b.w;
  });

  vector<int> tparent(k), trank(k, 0);
  iota(tparent.begin(), tparent.end(), 0);
  auto tfind = [&](auto&& self, int x) -> int {
    return tparent[x] == x ? x : tparent[x] = self(self, tparent[x]);
  };
  auto tunite = [&](int x, int y) -> bool {
    x = tfind(tfind, x);
    y = tfind(tfind, y);
    if (x == y) return false;
    if (trank[x] < trank[y]) swap(x, y);
    tparent[y] = x;
    if (trank[x] == trank[y]) trank[x]++;
    return true;
  };

  vector<pair<int, int>> mst_term_pairs;
  for (auto& e : term_edges) {
    if (tunite(e.i, e.j)) {
      mst_term_pairs.emplace_back(e.i, e.j);
      if ((int)mst_term_pairs.size() == k - 1) break;
    }
  }

  set<pair<int, int>> used_edge_set;
  for (auto [si, ti] : mst_term_pairs) {
    int src = terminals[si];
    int tgt = terminals[ti];
    if (term_dist[si][ti] >= INF) continue;
    int cur = tgt;
    while (cur != src) {
      int p = prevs[si][cur];
      if (p == -1) break;
      int x = min(cur, p);
      int y = max(cur, p);
      used_edge_set.insert({x, y});
      cur = p;
    }
  }

  vector<tuple<ll, int, int>> cand_edges;
  for (auto& p : used_edge_set) {
    ll w = edge_weight[p];
    cand_edges.emplace_back(w, p.first, p.second);
  }
  sort(cand_edges.begin(), cand_edges.end());

  vector<int> uf_parent(n), rankk(n, 0);
  iota(uf_parent.begin(), uf_parent.end(), 0);
  auto mfind = [&](auto&& self, int x) -> int {
    return uf_parent[x] == x ? x : uf_parent[x] = self(self, uf_parent[x]);
  };
  auto munite = [&](int x, int y) -> bool {
    int px = mfind(mfind, x);
    int py = mfind(mfind, y);
    if (px == py) return false;
    if (rankk[px] < rankk[py]) swap(px, py);
    uf_parent[py] = px;
    if (rankk[px] == rankk[py]) rankk[px]++;
    return true;
  };

  ll total_weight = 0;
  vector<pair<int, int>> final_edges;
  for (auto& [w, u, v] : cand_edges) {
    if (munite(u, v)) {
      total_weight += w;
      final_edges.emplace_back(u, v);
      int root = mfind(mfind, terminals[0]);
      bool conn = true;
      for (int j = 1; j < k; j++) {
        if (mfind(mfind, terminals[j]) != root) {
          conn = false;
          break;
        }
      }
      if (conn) break;
    }
  }

  cout << total_weight << " " << final_edges.size() << "\n";
  for (auto [u, v] : final_edges) {
    if (u > v) swap(u, v);
    cout << u << " " << v << "\n";
  }

  return 0;
}
```
