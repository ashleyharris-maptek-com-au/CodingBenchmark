use std::io::{self, BufRead};

fn is_dominating_set(n: usize, adj: &[Vec<usize>], s: &[usize]) -> bool {
    if s.is_empty() && n > 0 {
        return false;
    }
    let mut covered = vec![false; n];
    for &v in s {
        if v < n {
            covered[v] = true;
            for &u in &adj[v] {
                if u < n {
                    covered[u] = true;
                }
            }
        }
    }
    covered.iter().all(|&b| b)
}

fn compute_dominating_set(n: usize, adj: &[Vec<usize>]) -> Vec<usize> {
    if n == 0 {
        return Vec::new();
    }

    let mut current_cover: Vec<usize> = (0..n).map(|i| adj[i].len() + 1).collect();
    let mut is_dominated: Vec<bool> = vec![false; n];
    let mut selected: Vec<usize> = Vec::new();
    let mut num_undominated = n;

    while num_undominated > 0 {
        // Find vertex with maximum current cover
        let mut best_v = 0;
        let mut best_cover = 0usize;
        for v in 0..n {
            if current_cover[v] > best_cover {
                best_cover = current_cover[v];
                best_v = v;
            }
        }

        if best_cover == 0 {
            // Fallback: pick any undominated vertex
            let mut found = false;
            for v in 0..n {
                if !is_dominated[v] {
                    best_v = v;
                    found = true;
                    break;
                }
            }
            if !found {
                break;
            }
        }

        selected.push(best_v);

        // Collect newly dominated vertices
        let mut new_dominated: Vec<usize> = Vec::new();
        if !is_dominated[best_v] {
            new_dominated.push(best_v);
        }
        for &u in &adj[best_v] {
            if !is_dominated[u] {
                new_dominated.push(u);
            }
        }

        // Mark as dominated and update count
        for &u in &new_dominated {
            if !is_dominated[u] {
                is_dominated[u] = true;
                num_undominated -= 1;
            }
        }

        // Decrement cover counts for all dominators of newly dominated vertices
        for &u in &new_dominated {
            let mut dominators = vec![u];
            dominators.extend_from_slice(&adj[u]);
            for &w in &dominators {
                if w < n && current_cover[w] > 0 {
                    current_cover[w] -= 1;
                }
            }
        }
    }

    selected
}

fn improve_dominating_set(n: usize, adj: &[Vec<usize>], ds: &mut Vec<usize>) {
    if ds.is_empty() {
        return;
    }

    let mut result = ds.clone();
    let max_passes = result.len() + 1;

    for _ in 0..max_passes {
        let mut removed_any = false;
        let current_len = result.len();
        for i in (0..current_len).rev() {
            if i >= result.len() {
                continue;
            }
            let mut temp = result.clone();
            temp.remove(i);
            if is_dominating_set(n, adj, &temp) {
                result = temp;
                removed_any = true;
                break;
            }
        }
        if !removed_any {
            break;
        }
    }

    *ds = result;
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = match lines.next() {
        Some(Ok(line)) => line,
        _ => {
            println!("0");
            println!();
            return;
        }
    };

    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap_or("0").parse().unwrap_or(0);
    let m: usize = parts.next().unwrap_or("0").parse().unwrap_or(0);

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];

    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let mut p = line.split_whitespace();
            let u: Option<usize> = p.next().and_then(|s| s.parse().ok());
            let v: Option<usize> = p.next().and_then(|s| s.parse().ok());
            if let (Some(u), Some(v)) = (u, v) {
                if u < n && v < n && u != v {
                    adj[u].push(v);
                    adj[v].push(u);
                }
            }
        } else {
            break;
        }
    }

    for i in 0..n {
        adj[i].sort_unstable();
        adj[i].dedup();
    }

    let mut ds = compute_dominating_set(n, &adj);
    improve_dominating_set(n, &adj, &mut ds);

    // Ensure validity (fallback if somehow invalid)
    if !is_dominating_set(n, &adj, &ds) {
        // Very rare fallback: add all undominated
        let mut covered = vec![false; n];
        for &v in &ds {
            if v < n {
                covered[v] = true;
                for &u in &adj[v] {
                    if u < n {
                        covered[u] = true;
                    }
                }
            }
        }
        for i in 0..n {
            if !covered[i] {
                ds.push(i);
            }
        }
    }

    // Output
    println!("{}", ds.len());
    for (i, &v) in ds.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
