using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        string line = Console.ReadLine().Trim();
        string[] tokens = line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(tokens[0]);
        double mu = double.Parse(tokens[1]);

        line = Console.ReadLine().Trim();
        tokens = line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        double[] startPos = new double[3];
        for (int i = 0; i < 3; i++) startPos[i] = double.Parse(tokens[i]);

        List<double[]> positions = new List<double[]>();
        for (int i = 0; i < N; i++)
        {
            line = Console.ReadLine().Trim();
            tokens = line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            double[] pos = new double[3];
            for (int j = 0; j < 3; j++) pos[j] = double.Parse(tokens[j]);
            positions.Add(pos);
        }

        List<int> order;
        if (N <= 9)
        {
            order = SolveExact(N, positions, startPos);
        }
        else
        {
            order = SolveHeuristic(N, positions, startPos);
        }

        Console.WriteLine(string.Join(" ", order));
    }

    static double Distance(double[] a, double[] b)
    {
        double dx = a[0] - b[0];
        double dy = a[1] - b[1];
        double dz = a[2] - b[2];
        return Math.Sqrt(dx * dx + dy * dy + dz * dz);
    }

    static List<int> SolveExact(int N, List<double[]> positions, double[] startPos)
    {
        double bestCost = double.MaxValue;
        List<int> bestOrder = null;
        bool[] visited = new bool[N];
        List<int> path = new List<int>();

        void Backtrack(double costSoFar)
        {
            if (path.Count == N)
            {
                if (costSoFar < bestCost)
                {
                    bestCost = costSoFar;
                    bestOrder = new List<int>(path);
                }
                return;
            }

            int last = path.Count > 0 ? path[path.Count - 1] : -1;
            for (int i = 0; i < N; i++)
            {
                if (!visited[i])
                {
                    double add = (path.Count == 0)
                        ? Distance(startPos, positions[i])
                        : Distance(positions[last], positions[i]);

                    if (costSoFar + add < bestCost)
                    {
                        visited[i] = true;
                        path.Add(i);
                        Backtrack(costSoFar + add);
                        path.RemoveAt(path.Count - 1);
                        visited[i] = false;
                    }
                }
            }
        }

        Backtrack(0.0);
        return bestOrder ?? Enumerable.Range(0, N).ToList();
    }

    static double ComputePathCost(List<int> order, List<double[]> positions, double[] startPos)
    {
        if (order.Count == 0) return 0.0;
        double cost = Distance(startPos, positions[order[0]]);
        for (int i = 1; i < order.Count; i++)
        {
            cost += Distance(positions[order[i - 1]], positions[order[i]]);
        }
        return cost;
    }

    static List<int> SolveHeuristic(int N, List<double[]> positions, double[] startPos)
    {
        List<int> bestOrder = null;
        double bestCost = double.MaxValue;

        for (int startIdx = 0; startIdx < N; startIdx++)
        {
            List<int> order = new List<int>();
            bool[] visited = new bool[N];

            order.Add(startIdx);
            visited[startIdx] = true;
            int current = startIdx;

            for (int k = 1; k < N; k++)
            {
                double minD = double.MaxValue;
                int next = -1;
                for (int i = 0; i < N; i++)
                {
                    if (!visited[i])
                    {
                        double d = Distance(positions[current], positions[i]);
                        if (d < minD)
                        {
                            minD = d;
                            next = i;
                        }
                    }
                }
                order.Add(next);
                visited[next] = true;
                current = next;
            }

            double c = ComputePathCost(order, positions, startPos);
            if (c < bestCost)
            {
                bestCost = c;
                bestOrder = new List<int>(order);
            }
        }

        // 2-opt improvement
        if (bestOrder == null) bestOrder = Enumerable.Range(0, N).ToList();
        double currentCost = bestCost;
        bool improved = true;
        int iterations = 0;
        int maxIterations = 1000;

        while (improved && iterations < maxIterations)
        {
            improved = false;
            iterations++;

            for (int i = 0; i < N - 2; i++)
            {
                for (int j = i + 2; j < N; j++)
                {
                    List<int> newOrder = new List<int>(bestOrder);
                    int left = i + 1;
                    int right = j;
                    if (left >= right) continue;

                    while (left < right)
                    {
                        int tmp = newOrder[left];
                        newOrder[left] = newOrder[right];
                        newOrder[right] = tmp;
                        left++;
                        right--;
                    }

                    double newC = ComputePathCost(newOrder, positions, startPos);
                    if (newC < currentCost)
                    {
                        bestOrder = newOrder;
                        currentCost = newC;
                        improved = true;
                    }
                }
            }
        }

        return bestOrder;
    }
}
