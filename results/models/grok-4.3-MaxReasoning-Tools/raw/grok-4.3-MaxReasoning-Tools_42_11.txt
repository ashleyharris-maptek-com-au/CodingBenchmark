; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 280
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_gl_PerVertex_in
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%arr_v3float_4 = OpTypeArray %v3float %uint_4
%ptr_in_v3arr = OpTypePointer Input %arr_v3float_4
%ptr_out_v3 = OpTypePointer Output %v3float

%ptr_v4_in = OpTypePointer Input %v4float
%ptr_v3_in = OpTypePointer Input %v3float
%ptr_v4_out = OpTypePointer Output %v4float

%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_out = OpVariable %ptr_out_pv Output
%in_worldPos = OpVariable %ptr_in_v3arr Input
%in_normal = OpVariable %ptr_in_v3arr Input
%in_color = OpVariable %ptr_in_v3arr Input
%outColor = OpVariable %ptr_out_v3 Output

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_third = OpConstant %float 0.333333
%float_two_third = OpConstant %float 0.666667
%float_offset = OpConstant %float 0.02

%func_type = OpTypeFunction %void

%main = OpFunction %void None %func_type
%entry = OpLabel

; Load gl_in positions (clip space)
%ptr_p0 = OpAccessChain %ptr_v4_in %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_p0
%ptr_p1 = OpAccessChain %ptr_v4_in %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr_p1
%ptr_p2 = OpAccessChain %ptr_v4_in %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %ptr_p2
%ptr_p3 = OpAccessChain %ptr_v4_in %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %ptr_p3

; Load colors (color[1] and color[2])
%ptr_c1 = OpAccessChain %ptr_v3_in %in_color %int_1
%c1 = OpLoad %v3float %ptr_c1
%ptr_c2 = OpAccessChain %ptr_v3_in %in_color %int_2
%c2 = OpLoad %v3float %ptr_c2

; ==================== QUAD 1: t=0.0 to t=1/3 ====================
%A1 = OpExtInst %v4float %glsl 47 %p1 %p2 %float_0
%B1 = OpExtInst %v4float %glsl 47 %p1 %p2 %float_third

; color at start and end of quad 1
%col_start1 = OpCopyObject %v3float %c1
%col_end1 = OpExtInst %v3float %glsl 47 %c1 %c2 %float_third

; --- Compute dir / perp for quad 1 (A1 -> B1) using NDC ---
%wA1 = OpCompositeExtract %float %A1 3
%inv_wA1 = OpFDiv %float %float_1 %wA1
%A1x = OpCompositeExtract %float %A1 0
%A1y = OpCompositeExtract %float %A1 1
%ndcA1x = OpFMul %float %A1x %inv_wA1
%ndcA1y = OpFMul %float %A1y %inv_wA1

%wB1 = OpCompositeExtract %float %B1 3
%inv_wB1 = OpFDiv %float %float_1 %wB1
%B1x = OpCompositeExtract %float %B1 0
%B1y = OpCompositeExtract %float %B1 1
%ndcB1x = OpFMul %float %B1x %inv_wB1
%ndcB1y = OpFMul %float %B1y %inv_wB1

%dx1 = OpFSub %float %ndcB1x %ndcA1x
%dy1 = OpFSub %float %ndcB1y %ndcA1y
%delta1 = OpCompositeConstruct %v2float %dx1 %dy1
%dir1 = OpExtInst %v2float %glsl 69 %delta1

%dir1y = OpCompositeExtract %float %dir1 1
%neg_dir1y = OpFNegate %float %dir1y
%dir1x = OpCompositeExtract %float %dir1 0
%perp1 = OpCompositeConstruct %v2float %neg_dir1y %dir1x

%perp_off1 = OpVectorTimesScalar %v2float %perp1 %float_offset
%off1x = OpCompositeExtract %float %perp_off1 0
%off1y = OpCompositeExtract %float %perp_off1 1

; leftA1 / rightA1
%A1z = OpCompositeExtract %float %A1 2
%off_clip_x1 = OpFMul %float %off1x %wA1
%off_clip_y1 = OpFMul %float %off1y %wA1
%leftA1 = OpCompositeConstruct %v4float %off_clip_x1 %off_clip_y1 %A1z %wA1
%neg_off_clip_x1 = OpFNegate %float %off_clip_x1
%neg_off_clip_y1 = OpFNegate %float %off_clip_y1
%rightA1 = OpCompositeConstruct %v4float %neg_off_clip_x1 %neg_off_clip_y1 %A1z %wA1

; leftB1 / rightB1
%B1z = OpCompositeExtract %float %B1 2
%off_clip_xB1 = OpFMul %float %off1x %wB1
%off_clip_yB1 = OpFMul %float %off1y %wB1
%leftB1 = OpCompositeConstruct %v4float %off_clip_xB1 %off_clip_yB1 %B1z %wB1
%neg_off_clip_xB1 = OpFNegate %float %off_clip_xB1
%neg_off_clip_yB1 = OpFNegate %float %off_clip_yB1
%rightB1 = OpCompositeConstruct %v4float %neg_off_clip_xB1 %neg_off_clip_yB1 %B1z %wB1

; Emit quad 1 (triangle strip)
%pos_ptr1 = OpAccessChain %ptr_v4_out %gl_out %int_0
OpStore %pos_ptr1 %leftA1
OpStore %outColor %col_start1
OpEmitVertex

OpStore %pos_ptr1 %rightA1
OpStore %outColor %col_start1
OpEmitVertex

OpStore %pos_ptr1 %leftB1
OpStore %outColor %col_end1
OpEmitVertex

OpStore %pos_ptr1 %rightB1
OpStore %outColor %col_end1
OpEmitVertex

OpEndPrimitive

; ==================== QUAD 2: t=2/3 to t=1.0 ====================
%A2 = OpExtInst %v4float %glsl 47 %p1 %p2 %float_two_third
%B2 = OpExtInst %v4float %glsl 47 %p1 %p2 %float_1

; colors
%col_start2 = OpExtInst %v3float %glsl 47 %c1 %c2 %float_two_third
%col_end2 = OpCopyObject %v3float %c2

; --- Compute dir / perp for quad 2 (A2 -> B2) ---
%wA2 = OpCompositeExtract %float %A2 3
%inv_wA2 = OpFDiv %float %float_1 %wA2
%A2x = OpCompositeExtract %float %A2 0
%A2y = OpCompositeExtract %float %A2 1
%ndcA2x = OpFMul %float %A2x %inv_wA2
%ndcA2y = OpFMul %float %A2y %inv_wA2

%wB2 = OpCompositeExtract %float %B2 3
%inv_wB2 = OpFDiv %float %float_1 %wB2
%B2x = OpCompositeExtract %float %B2 0
%B2y = OpCompositeExtract %float %B2 1
%ndcB2x = OpFMul %float %B2x %inv_wB2
%ndcB2y = OpFMul %float %B2y %inv_wB2

%dx2 = OpFSub %float %ndcB2x %ndcA2x
%dy2 = OpFSub %float %ndcB2y %ndcA2y
%delta2 = OpCompositeConstruct %v2float %dx2 %dy2
%dir2 = OpExtInst %v2float %glsl 69 %delta2

%dir2y = OpCompositeExtract %float %dir2 1
%neg_dir2y = OpFNegate %float %dir2y
%dir2x = OpCompositeExtract %float %dir2 0
%perp2 = OpCompositeConstruct %v2float %neg_dir2y %dir2x

%perp_off2 = OpVectorTimesScalar %v2float %perp2 %float_offset
%off2x = OpCompositeExtract %float %perp_off2 0
%off2y = OpCompositeExtract %float %perp_off2 1

; leftA2 / rightA2
%A2z = OpCompositeExtract %float %A2 2
%off_clip_x2 = OpFMul %float %off2x %wA2
%off_clip_y2 = OpFMul %float %off2y %wA2
%leftA2 = OpCompositeConstruct %v4float %off_clip_x2 %off_clip_y2 %A2z %wA2
%neg_off_clip_x2 = OpFNegate %float %off_clip_x2
%neg_off_clip_y2 = OpFNegate %float %off_clip_y2
%rightA2 = OpCompositeConstruct %v4float %neg_off_clip_x2 %neg_off_clip_y2 %A2z %wA2

; leftB2 / rightB2
%B2z = OpCompositeExtract %float %B2 2
%off_clip_xB2 = OpFMul %float %off2x %wB2
%off_clip_yB2 = OpFMul %float %off2y %wB2
%leftB2 = OpCompositeConstruct %v4float %off_clip_xB2 %off_clip_yB2 %B2z %wB2
%neg_off_clip_xB2 = OpFNegate %float %off_clip_xB2
%neg_off_clip_yB2 = OpFNegate %float %off_clip_yB2
%rightB2 = OpCompositeConstruct %v4float %neg_off_clip_xB2 %neg_off_clip_yB2 %B2z %wB2

; Emit quad 2 (triangle strip)
OpStore %pos_ptr1 %leftA2
OpStore %outColor %col_start2
OpEmitVertex

OpStore %pos_ptr1 %rightA2
OpStore %outColor %col_start2
OpEmitVertex

OpStore %pos_ptr1 %leftB2
OpStore %outColor %col_end2
OpEmitVertex

OpStore %pos_ptr1 %rightB2
OpStore %outColor %col_end2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
