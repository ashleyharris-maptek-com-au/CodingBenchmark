declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data: any = JSON.parse(input);

const width: number = data.width;
const height: number = data.height;
const grid: string[] = data.grid;
const ignitions: number[][] = data.ignitions;
const towns: { id: number; x: number; y: number; value: number }[] = data.towns;
const cutBudget: number = data.cutBudget;

const dirs: number[][] = [[0, 1], [1, 0], [0, -1], [-1, 0]];

function getReachableCells(
  grid: string[],
  width: number,
  height: number,
  ignitions: number[][],
  cuts: boolean[][]
): boolean[][] {
  const reachable: boolean[][] = Array.from({ length: height }, () => Array(width).fill(false));
  const q: [number, number][] = [];
  let head = 0;

  for (const [x, y] of ignitions) {
    if (y >= 0 && y < height && x >= 0 && x < width && grid[y][x] === '.' && !cuts[y][x]) {
      reachable[y][x] = true;
      q.push([x, y]);
    }
  }

  while (head < q.length) {
    const [x, y] = q[head++];
    for (const [dx, dy] of dirs) {
      const nx = x + dx;
      const ny = y + dy;
      if (
        nx >= 0 && nx < width && ny >= 0 && ny < height &&
        grid[ny][nx] === '.' && !reachable[ny][nx] && !cuts[ny][nx]
      ) {
        reachable[ny][nx] = true;
        q.push([nx, ny]);
      }
    }
  }
  return reachable;
}

function computeDistFromIgn(
  grid: string[],
  width: number,
  height: number,
  ignitions: number[][],
  cuts: boolean[][]
): number[][] {
  const dist: number[][] = Array.from({ length: height }, () => Array(width).fill(Infinity));
  const q: [number, number][] = [];
  let head = 0;

  for (const [x, y] of ignitions) {
    if (y >= 0 && y < height && x >= 0 && x < width && grid[y][x] === '.' && !cuts[y][x]) {
      dist[y][x] = 0;
      q.push([x, y]);
    }
  }

  while (head < q.length) {
    const [x, y] = q[head++];
    for (const [dx, dy] of dirs) {
      const nx = x + dx;
      const ny = y + dy;
      if (
        nx >= 0 && nx < width && ny >= 0 && ny < height &&
        grid[ny][nx] === '.' && dist[ny][nx] === Infinity && !cuts[ny][nx]
      ) {
        dist[ny][nx] = dist[y][x] + 1;
        q.push([nx, ny]);
      }
    }
  }
  return dist;
}

function findCellToCut(
  town: { x: number; y: number; value: number },
  ignitions: number[][],
  cuts: boolean[][],
  grid: string[],
  width: number,
  height: number,
  distFromIgn: number[][]
): [number, number] | null {
  const visited: boolean[][] = Array.from({ length: height }, () => Array(width).fill(false));
  const parent: ([number, number] | null)[][] = Array.from({ length: height }, () => Array(width).fill(null));
  const q: [number, number][] = [];
  let head = 0;

  const tx = town.x;
  const ty = town.y;

  if (cuts[ty][tx]) return null;

  visited[ty][tx] = true;
  parent[ty][tx] = null;
  q.push([tx, ty]);

  let reached: [number, number] | null = null;

  while (head < q.length) {
    const [x, y] = q[head++];

    const isIgn = ignitions.some(([ix, iy]) => ix === x && iy === y);
    if (isIgn && !cuts[y][x]) {
      reached = [x, y];
      break;
    }

    for (const [dx, dy] of dirs) {
      const nx = x + dx;
      const ny = y + dy;
      if (
        nx >= 0 && nx < width && ny >= 0 && ny < height &&
        grid[ny][nx] === '.' && !visited[ny][nx] && !cuts[ny][nx]
      ) {
        visited[ny][nx] = true;
        parent[ny][nx] = [x, y];
        q.push([nx, ny]);
      }
    }
  }

  if (!reached) return null;

  // Reconstruct path from ignition back to town
  const path: [number, number][] = [];
  let curr: [number, number] | null = reached;
  while (curr !== null) {
    path.push(curr);
    const p = parent[curr[1]][curr[0]];
    if (p === null) break;
    curr = p;
  }

  // path[0] = ignition, path[path.length-1] = town
  if (path.length <= 1) return null;

  // Choose best cell: prefer lowest distFromIgn, skip ignition and town
  let bestCell: [number, number] | null = null;
  let bestD = Infinity;

  for (let i = 1; i < path.length - 1; i++) {
    const [cx, cy] = path[i];
    const d = distFromIgn[cy][cx];
    if (d < bestD && d < Infinity) {
      bestD = d;
      bestCell = [cx, cy];
    }
  }

  if (bestCell) return bestCell;

  // Fallback: pick cell next to ignition (path[1])
  if (path.length >= 2) {
    const [cx, cy] = path[1];
    const isTown = cx === tx && cy === ty;
    if (!isTown) {
      return [cx, cy];
    }
  }

  return null;
}

const initialCuts: boolean[][] = Array.from({ length: height }, () => Array(width).fill(false));
const initialReach = getReachableCells(grid, width, height, ignitions, initialCuts);

const threatenedTowns = towns
  .filter((town) => {
    const { x, y } = town;
    return x >= 0 && x < width && y >= 0 && y < height && initialReach[y][x];
  })
  .sort((a, b) => b.value - a.value);

const cuts: boolean[][] = Array.from({ length: height }, () => Array(width).fill(false));
let cutsMade = 0;
const distFromIgn = computeDistFromIgn(grid, width, height, ignitions, cuts);

const maxIter = Math.min(300, cutBudget * 2 + 10);
let iterations = 0;
let changed = true;

while (changed && cutsMade < cutBudget && iterations < maxIter) {
  changed = false;
  iterations++;

  for (const town of threatenedTowns) {
    if (cutsMade >= cutBudget) break;

    const currReach = getReachableCells(grid, width, height, ignitions, cuts);
    if (!currReach[town.y][town.x]) continue;

    const cell = findCellToCut(town, ignitions, cuts, grid, width, height, distFromIgn);
    if (cell) {
      const [cx, cy] = cell;
      if (!cuts[cy][cx] && grid[cy][cx] === '.') {
        const isTownPos = towns.some((t) => t.x === cx && t.y === cy);
        if (!isTownPos) {
          cuts[cy][cx] = true;
          cutsMade++;
          changed = true;
        }
      }
    }
  }
}

// Collect firebreaks
const firebreaks: number[][] = [];
for (let y = 0; y < height; y++) {
  for (let x = 0; x < width; x++) {
    if (cuts[y][x]) {
      firebreaks.push([x, y]);
    }
  }
}
