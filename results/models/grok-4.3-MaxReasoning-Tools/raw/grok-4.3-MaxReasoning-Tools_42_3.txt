```spirv
; SPIR-V
; Version: 1.0
; Generator: Khronos SPIR-V Tools Assembler
; Bound: 128
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_2 = OpConstant %uint 2
%arr2_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%arr2_v3_in = OpTypeArray %v3float %uint_2

%ptr_arr_pv_in = OpTypePointer Input %arr2_pv_in
%ptr_arr_v3_in = OpTypePointer Input %arr2_v3_in
%ptr_pv_out = OpTypePointer Output %gl_PerVertex_out
%ptr_v3_out = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_arr_pv_in Input
%color_in = OpVariable %ptr_arr_v3_in Input
%gl_out = OpVariable %ptr_pv_out Output
%outColor = OpVariable %ptr_v3_out Output

%ptr_v4_in = OpTypePointer Input %v4float
%ptr_v3_in = OpTypePointer Input %v3float
%ptr_v4_out = OpTypePointer Output %v4float

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_thick = OpConstant %float 0.03

%main = OpFunction %void None
%entry = OpLabel

; Load input positions
%gl_in_0_ptr = OpAccessChain %ptr_v4_in %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %gl_in_0_ptr
%gl_in_1_ptr = OpAccessChain %ptr_v4_in %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %gl_in_1_ptr

; Load input colors
%color_0_ptr = OpAccessChain %ptr_v3_in %color_in %int_0
%c0 = OpLoad %v3float %color_0_ptr
%color_1_ptr = OpAccessChain %ptr_v3_in %color_in %int_1
%c1 = OpLoad %v3float %color_1_ptr

; Compute direction
%dir = OpFSub %v4float %p1 %p0
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1

; Compute perpendicular ( -dir.y, dir.x )
%neg_dir_y = OpFNegate %float %dir_y
%perp_x = %neg_dir_y
%perp_y = %dir_x

; Compute length for normalization
%px_sq = OpFMul %float %perp_x %perp_x
%py_sq = OpFMul %float %perp_y %perp_y
%len_sq = OpFAdd %float %px_sq %py_sq
%len = OpSqrt %float %len_sq

; Normalize and scale by thickness
%inv_len = OpFDiv %float %float_1 %len
%norm_perp_x = OpFMul %float %perp_x %inv_len
%norm_perp_y = OpFMul %float %perp_y %inv_len
%scaled_perp_x = OpFMul %float %norm_perp_x %float_thick
%scaled_perp_y = OpFMul %float %norm_perp_y %float_thick
%perp_scaled = OpCompositeConstruct %v2float %scaled_perp_x %scaled_perp_y

; Extract z and w from p0 (shared for line segment)
%p0_z = OpCompositeExtract %float %p0 2
%p0_w = OpCompositeExtract %float %p0 3

; Extract xy components
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1

; Compute offset positions (xy)
%p0_plus_xy = OpFAdd %v2float %p0_xy %perp_scaled
%p0_minus_xy = OpFSub %v2float %p0_xy %perp_scaled
%p1_plus_xy = OpFAdd %v2float %p1_xy %perp_scaled
%p1_minus_xy = OpFSub %v2float %p1_xy %perp_scaled

; Construct full vec4 positions (keep z/w from p0)
%v0_x = OpCompositeExtract %float %p0_plus_xy 0
%v0_y = OpCompositeExtract %float %p0_plus_xy 1
%pos0 = OpCompositeConstruct %v4float %v0_x %v0_y %p0_z %p0_w

%v1_x = OpCompositeExtract %float %p0_minus_xy 0
%v1_y = OpCompositeExtract %float %p0_minus_xy 1
%pos1 = OpCompositeConstruct %v4float %v1_x %v1_y %p0_z %p0_w

%v2_x = OpCompositeExtract %float %p1_plus_xy 0
%v2_y = OpCompositeExtract %float %p1_plus_xy 1
%pos2 = OpCompositeConstruct %v4float %v2_x %v2_y %p0_z %p0_w

%v3_x = OpCompositeExtract %float %p1_minus_xy 0
%v3_y = OpCompositeExtract %float %p1_minus_xy 1
%pos3 = OpCompositeConstruct %v4float %v3_x %v3_y %p0_z %p0_w

; Prepare output position pointer (reused)
%out_pos_ptr = OpAccessChain %ptr_v4_out %gl_out %int_0

; Emit vertex 0: p0 + perp, color c0
OpStore %out_pos_ptr %pos0
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 1: p0 - perp, color c0
OpStore %out_pos_ptr %pos1
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 2: p1 + perp, color c1
OpStore %out_pos_ptr %pos2
OpStore %outColor %c1
OpEmitVertex

; Emit vertex 3: p1 - perp, color c1
OpStore %out_pos_ptr %pos3
OpStore %outColor %c1
OpEmitVertex

; Finish the triangle strip
OpEndPrimitive

OpReturn
OpFunctionEnd
```
