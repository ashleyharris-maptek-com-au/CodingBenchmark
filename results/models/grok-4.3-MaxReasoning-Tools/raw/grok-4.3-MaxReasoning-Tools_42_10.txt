```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 200
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %out_color
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%uint = OpTypeInt 32 0
%uint_4 = OpConstant %uint 4
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_Input_arr_gl_PerVertex = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex Input

%ptr_Output_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex Output

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%arr_v3float = OpTypeArray %v3float %uint_4
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3float

%in_worldPos = OpVariable %ptr_Input_arr_v3 Input
OpDecorate %in_worldPos Location 0

%in_normal = OpVariable %ptr_Input_arr_v3 Input
OpDecorate %in_normal Location 1

%in_color = OpVariable %ptr_Input_arr_v3 Input
OpDecorate %in_color Location 2

%ptr_Output_v3float = OpTypePointer Output %v3float
%out_color = OpVariable %ptr_Output_v3float Output
OpDecorate %out_color Location 0

%void = OpTypeVoid
%v4float = OpTypeVector %float 4
%v2float = OpTypeVector %float 2
%int = OpTypeInt 32 1
%bool = OpTypeBool
%func = OpTypeFunction %void

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_5 = OpConstant %int 5

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_4 = OpConstant %float 4
%float_0_02 = OpConstant %float 0.02

%ptr_Input_gl_PerVertex = OpTypePointer Input %gl_PerVertex_in
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float

%main = OpFunction %void None %func
%entry = OpLabel

; Load gl_in positions (P0, P1, P2, P3)
%gl_in0 = OpAccessChain %ptr_Input_gl_PerVertex %gl_in %int_0
%pos_ptr0 = OpAccessChain %ptr_Input_v4float %gl_in0 %int_0
%P0 = OpLoad %v4float %pos_ptr0

%gl_in1 = OpAccessChain %ptr_Input_gl_PerVertex %gl_in %int_1
%pos_ptr1 = OpAccessChain %ptr_Input_v4float %gl_in1 %int_0
%P1 = OpLoad %v4float %pos_ptr1

%gl_in2 = OpAccessChain %ptr_Input_gl_PerVertex %gl_in %int_2
%pos_ptr2 = OpAccessChain %ptr_Input_v4float %gl_in2 %int_0
%P2 = OpLoad %v4float %pos_ptr2

%gl_in3 = OpAccessChain %ptr_Input_gl_PerVertex %gl_in %int_3
%pos_ptr3 = OpAccessChain %ptr_Input_v4float %gl_in3 %int_0
%P3 = OpLoad %v4float %pos_ptr3

; Load colors[1] and colors[2]
%col_ptr1 = OpAccessChain %ptr_Input_v3float %in_color %int_1
%C1 = OpLoad %v3float %col_ptr1

%col_ptr2 = OpAccessChain %ptr_Input_v3float %in_color %int_2
%C2 = OpLoad %v3float %col_ptr2

; Tangent directions
%tang_start = OpFSub %v4float %P2 %P0
%tang_end = OpFSub %v4float %P3 %P1

; Branch to loop header
OpBranch %loop_header

%loop_header = OpLabel
%i = OpPhi %int %int_0 %entry %next_i %loop_latch

; t = float(i) / 4.0
%fi = OpConvertSToF %float %i
%t = OpFDiv %float %fi %float_4
%omt = OpFSub %float %float_1 %t

; Interpolate position: (1-t)*P1 + t*P2
%P1s = OpVectorTimesScalar %v4float %P1 %omt
%P2s = OpVectorTimesScalar %v4float %P2 %t
%pos = OpFAdd %v4float %P1s %P2s

; Interpolate tangent
%ts_s = OpVectorTimesScalar %v4float %tang_start %omt
%te_s = OpVectorTimesScalar %v4float %tang_end %t
%tang = OpFAdd %v4float %ts_s %te_s

; 2D tangent
%tang2 = OpVectorShuffle %v2float %tang %tang 0 1

; Normalize tangent
%dot2 = OpDot %float %tang2 %tang2
%len = OpSqrt %float %dot2
%inv_len = OpFDiv %float %float_1 %len
%norm_tang2 = OpVectorTimesScalar %v2float %tang2 %inv_len

; Compute perpendicular
%nt_x = OpCompositeExtract %float %norm_tang2 0
%nt_y = OpCompositeExtract %float %norm_tang2 1
%px = OpFNegate %float %nt_y
%py = %nt_x
%perp = OpCompositeConstruct %v2float %px %py

; Offset vector
%offset = OpVectorTimesScalar %v2float %perp %float_0_02

; Extract xy from pos
%pos_xy = OpVectorShuffle %v2float %pos %pos 0 1

; + offset vertex
%posp_xy = OpFAdd %v2float %pos_xy %offset
%pos_plus = OpVectorShuffle %v4float %pos %posp_xy 4 5 2 3

; - offset vertex
%offsetn = OpFNegate %v2float %offset
%posm_xy = OpFAdd %v2float %pos_xy %offsetn
%pos_minus = OpVectorShuffle %v4float %pos %posm_xy 4 5 2 3

; Interpolate color
%col1s = OpVectorTimesScalar %v3float %C1 %omt
%col2s = OpVectorTimesScalar %v3float %C2 %t
%col = OpFAdd %v3float %col1s %col2s

; Emit +perp vertex
%out_pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr %pos_plus
OpStore %out_color %col
OpEmitVertex

; Emit -perp vertex
OpStore %out_pos_ptr %pos_minus
OpStore %out_color %col
OpEmitVertex

; Increment counter
%next_i = OpIAdd %int %i %int_1
%cond = OpSLessThan %bool %next_i %int_5

OpLoopMerge %loop_merge %loop_latch None
OpBranchConditional %cond %loop_latch %loop_merge

%loop_latch = OpLabel
OpBranch %loop_header

%loop_merge = OpLabel

; End primitive
OpEndPrimitive
OpReturn
OpFunctionEnd
```
