```cpp
#include <iostream>
#include <vector>
#include <string>
#include <unordered_set>
#include <unordered_map>
#include <algorithm>
#include <cstdint>
#include <climits>
#include <utility>

struct HashVal {
    uint64_t h1;
    uint64_t h2;
};

struct HashHasher {
    size_t operator()(const HashVal& hv) const {
        size_t h = std::hash<uint64_t>{}(hv.h1);
        size_t h2v = std::hash<uint64_t>{}(hv.h2);
        h ^= h2v + 0x9e3779b9 + (h << 6) + (h >> 2);
        return h;
    }
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N;
    std::cin >> N;
    std::string dummy;
    std::getline(std::cin, dummy);

    std::vector<std::string> strs(N);
    for (int i = 0; i < N; ++i) {
        std::getline(std::cin, strs[i]);
    }

    if (N == 0) {
        std::cout << std::endl;
        return 0;
    }

    int shortest_idx = 0;
    for (int i = 1; i < N; ++i) {
        if (strs[i].size() < strs[shortest_idx].size()) {
            shortest_idx = i;
        }
    }

    int min_l = strs[shortest_idx].size();
    if (min_l == 0) {
        std::cout << std::endl;
        return 0;
    }

    const uint64_t BASE1 = 131ULL;
    const uint64_t BASE2 = 137ULL;

    std::vector<uint64_t> power1(min_l + 1, 1ULL);
    std::vector<uint64_t> power2(min_l + 1, 1ULL);
    for (int i = 1; i <= min_l; ++i) {
        power1[i] = power1[i - 1] * BASE1;
        power2[i] = power2[i - 1] * BASE2;
    }

    int NN = N;
    std::vector<std::vector<uint64_t>> pref1(NN), pref2(NN);
    for (int id = 0; id < NN; ++id) {
        int len = strs[id].size();
        pref1[id].assign(len + 1, 0ULL);
        pref2[id].assign(len + 1, 0ULL);
        for (int j = 1; j <= len; ++j) {
            uint64_t c = static_cast<uint64_t>(static_cast<unsigned char>(strs[id][j - 1]));
            pref1[id][j] = pref1[id][j - 1] * BASE1 + c;
            pref2[id][j] = pref2[id][j - 1] * BASE2 + c;
        }
    }

    auto get_h1 = [&](int id, int pos, int L) -> uint64_t {
        return pref1[id][pos + L] - pref1[id][pos] * power1[L];
    };

    auto get_h2 = [&](int id, int pos, int L) -> uint64_t {
        return pref2[id][pos + L] - pref2[id][pos] * power2[L];
    };

    using CandSet = std::unordered_set<HashVal, HashHasher>;

    auto hasCommon = [&](int L, HashVal* out = nullptr) -> bool {
        if (L == 0) {
            if (out) *out = {0ULL, 0ULL};
            return true;
        }
        int n0 = strs[shortest_idx].size();
        if (L > n0) return false;
        CandSet candidates;
        int cand_size = n0 - L + 1;
        if (cand_size > 0) candidates.reserve(cand_size);
        for (int i = 0; i <= n0 - L; ++i) {
            HashVal hv{get_h1(shortest_idx, i, L), get_h2(shortest_idx, i, L)};
            candidates.insert(hv);
        }
        for (int j = 0; j < NN; ++j) {
            if (j == shortest_idx) continue;
            int nj = strs[j].size();
            if (L > nj) return false;
            CandSet this_set;
            int this_size = nj - L + 1;
            if (this_size > 0) this_set.reserve(this_size);
            for (int i = 0; i <= nj - L; ++i) {
                HashVal hv{get_h1(j, i, L), get_h2(j, i, L)};
                this_set.insert(hv);
            }
            CandSet new_set;
            new_set.reserve(candidates.size());
            for (const auto& hv : candidates) {
                if (this_set.count(hv)) {
                    new_set.insert(hv);
                }
            }
            candidates = std::move(new_set);
            if (candidates.empty()) return false;
        }
        if (out && !candidates.empty()) {
            *out = *candidates.begin();
        }
        return !candidates.empty();
    };

    int left = 0, right = min_l;
    while (left < right) {
        int mid = left + (right - left + 1) / 2;
        if (hasCommon(mid)) {
            left = mid;
        } else {
            right = mid - 1;
        }
    }
    int maxL = left;

    if (maxL == 0) {
        std::cout << std::endl;
        return 0;
    }

    HashVal good_h;
    hasCommon(maxL, &good_h);

    int n0 = strs[shortest_idx].size();
    int pos = -1;
    for (int i = 0; i <= n0 - maxL; ++i) {
        if (get_h1(shortest_idx, i, maxL) == good_h.h1 &&
            get_h2(shortest_idx, i, maxL) == good_h.h2) {
            pos = i;
            break;
        }
    }

    if (pos != -1) {
        std::string ans = strs[shortest_idx].substr(pos, maxL);
        bool verified = true;
        for (int j = 0; j < NN && verified; ++j) {
            if (j != shortest_idx && strs[j].find(ans) == std::string::npos) {
                verified = false;
            }
        }
        if (verified) {
            std::cout << ans << std::endl;
        } else {
            std::cout << std::endl;
        }
    } else {
        std::cout << std::endl;
    }

    return 0;
}
```
