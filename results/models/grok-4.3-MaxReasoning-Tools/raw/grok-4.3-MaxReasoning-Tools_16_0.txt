```cpp
#include <bits/stdc++.h>
using namespace std;

using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    cin >> J >> M;

    vector<vector<pair<int, int>>> jobs(J);
    vector<int> num_tasks(J, 0);
    ll total_tasks = 0;

    for (int j = 0; j < J; j++) {
        int T;
        cin >> T;
        num_tasks[j] = T;
        jobs[j].resize(T);
        for (int t = 0; t < T; t++) {
            int m, d;
            cin >> m >> d;
            jobs[j][t] = {m, d};
        }
        total_tasks += T;
    }

    // Adaptive strategy selection based on scale
    bool use_simple_scan = (total_tasks <= 800 && J <= 80);

    vector<ll> task_starts_flat; // not needed, use per job
    vector<vector<ll>> task_starts(J);

    vector<int> curr_task(J, 0);
    vector<ll> job_rdy(J, 0LL);
    vector<ll> mach_rdy(M, 0LL);

    if (use_simple_scan) {
        // Simple O(J * tasks) scan for small instances
        while (true) {
            ll best_est = LLONG_MAX / 2;
            int best_j = -1;
            int best_dur = INT_MAX;
            for (int j = 0; j < J; j++) {
                if (curr_task[j] >= num_tasks[j]) continue;
                auto [m, dur] = jobs[j][curr_task[j]];
                ll est = max(job_rdy[j], mach_rdy[m]);
                bool better = false;
                if (est < best_est) better = true;
                else if (est == best_est && dur < best_dur) better = true;
                if (better) {
                    best_est = est;
                    best_j = j;
                    best_dur = dur;
                }
            }
            if (best_j == -1) break;

            int t_idx = curr_task[best_j];
            auto [m, d] = jobs[best_j][t_idx];
            ll start = best_est;

            task_starts[best_j].push_back(start);
            mach_rdy[m] = start + d;
            job_rdy[best_j] = start + d;
            curr_task[best_j]++;
        }
    } else {
        // Efficient heap-based dispatcher for medium/large scale
        // Min-heap per machine: {job_rdy, job_id}
        vector<priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>>> waiters(M);

        // Initialize
        for (int j = 0; j < J; j++) {
            if (num_tasks[j] > 0) {
                int m = jobs[j][0].first;
                waiters[m].emplace(0LL, j);
            }
        }

        while (true) {
            ll best_est = LLONG_MAX / 2;
            int best_m = -1;
            int best_j = -1;
            int best_dur = INT_MAX;

            for (int m = 0; m < M; m++) {
                if (waiters[m].empty()) continue;

                auto [jrdy, j] = waiters[m].top();

                // Robustness check
                if (curr_task[j] >= num_tasks[j] || jobs[j][curr_task[j]].first != m) {
                    waiters[m].pop();
                    continue;
                }

                ll est = max(jrdy, mach_rdy[m]);
                int dur = jobs[j][curr_task[j]].second;

                bool better = false;
                if (est < best_est) better = true;
                else if (est == best_est && dur < best_dur) better = true;

                if (better) {
                    best_est = est;
                    best_m = m;
                    best_j = j;
                    best_dur = dur;
                }
            }

            if (best_m == -1) break;

            // Pop the chosen
            waiters[best_m].pop();

            int j = best_j;
            int t_idx = curr_task[j];
            auto [m, d] = jobs[j][t_idx];
            ll start = best_est;

            task_starts[j].push_back(start);
            mach_rdy[m] = start + d;
            job_rdy[j] = start + d;
            curr_task[j]++;

            if (curr_task[j] < num_tasks[j]) {
                int next_m = jobs[j][curr_task[j]].first;
                waiters[next_m].emplace(job_rdy[j], j);
            }
        }
    }

    // Output
    for (int j = 0; j < J; j++) {
        for (size_t t = 0; t < task_starts[j].size(); t++) {
            if (t > 0) cout << " ";
            cout << task_starts[j][t] << " " << jobs[j][t].first;
        }
        cout << "\n";
    }

    return 0;
}
```
