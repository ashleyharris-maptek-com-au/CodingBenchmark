    <pre><code>const fs = require('fs');

// Read input from stdin
const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

const grid = data.grid || [];
const robots = data.robots || [];
const jobs = data.jobs || [];

const H = grid.length;
const W = H > 0 ? grid[0].length : 0;
const R = robots.length;
const J = jobs.length;

// Directions for BFS
const DIRS = [[0, 1], [0, -1], [1, 0], [-1, 0]];

// BFS to compute distance grid from a starting position
function computeDist(sx, sy) {
    if (sx < 0 || sx >= W || sy < 0 || sy >= H || grid[sy][sx] !== '.') {
        return null;
    }
    
    const dist = Array.from({ length: H }, () => Array(W).fill(-1));
    const queue = [];
    
    dist[sy][sx] = 0;
    queue.push([sy, sx]);
    
    let head = 0;
    while (head < queue.length) {
        const [y, x] = queue[head++];
        
        for (let [dy, dx] of DIRS) {
            const ny = y + dy;
            const nx = x + dx;
            
            if (ny >= 0 && ny < H && nx >= 0 && nx < W && 
                grid[ny][nx] === '.' && dist[ny][nx] === -1) {
                dist[ny][nx] = dist[y][x] + 1;
                queue.push([ny, nx]);
            }
        }
    }
    
    return dist;
}

// Precompute distance maps
const robotDistMaps = [];
for (let r = 0; r < R; r++) {
    const rx = robots[r].x;
    const ry = robots[r].y;
    robotDistMaps[r] = computeDist(rx, ry);
}

const jobDistMaps = [];
for (let j = 0; j < J; j++) {
    const jx = jobs[j].x;
    const jy = jobs[j].y;
    jobDistMaps[j] = computeDist(jx, jy);
}

// Build candidates: all (robot, job) pairs with distance
let candidates = [];
for (let r = 0; r < R; r++) {
    if (!robotDistMaps[r]) continue;
    for (let j = 0; j < J; j++) {
        const jy = jobs[j].y;
        const jx = jobs[j].x;
        const d = robotDistMaps[r][jy][jx];
        if (d >= 0) {
            candidates.push({ r, j, d });
        }
    }
}

// Sort candidates: smallest distance first, then by job id for determinism
candidates.sort((a, b) => {
    if (a.d !== b.d) return a.d - b.d;
    const idA = jobs[a.j].id;
    const idB = jobs[b.j].id;
    if (idA < idB) return -1;
    if (idA > idB) return 1;
    return 0;
});

// Assign jobs greedily
const assignedJobs = new Array(J).fill(false);
const robotAssignedJobs = Array.from({ length: R }, () => []);

for (let cand of candidates) {
    if (!assignedJobs[cand.j]) {
        assignedJobs[cand.j] = true;
        robotAssignedJobs[cand.r].push(cand.j);
    }
}

// Assign any remaining jobs (fallback)
for (let j = 0; j < J; j++) {
    if (!assignedJobs[j]) {
        // Assign to first robot that can reach it, or robot 0
        let assigned = false;
        for (let r = 0; r < R; r++) {
            if (robotDistMaps[r]) {
                const d = robotDistMaps[r][jobs[j].y][jobs[j].x];
                if (d >= 0) {
                    robotAssignedJobs[r].push(j);
                    assignedJobs[j] = true;
                    assigned = true;
                    break;
                }
            }
        }
        if (!assigned && R > 0) {
            robotAssignedJobs[0].push(j);
            assignedJobs[j] = true;
        }
    }
}

// Order jobs for each robot using nearest neighbor
const assignments = [];

for (let r = 0; r < R; r++) {
    const assignedJobIdxs = robotAssignedJobs[r];
    if (assignedJobIdxs.length === 0) {
        assignments.push([]);
        continue;
    }
    
    const ordered = [];
    const remaining = new Set(assignedJobIdxs);
    let lastJobIdx = null; // null means from robot position
    
    while (remaining.size > 0) {
        let minD = Infinity;
        let best = -1;
        
        for (let j of remaining) {
            let d = -1;
            
            const jy = jobs[j].y;
            const jx = jobs[j].x;
            
            if (lastJobIdx === null) {
                // From robot
                if (robotDistMaps[r]) {
                    d = robotDistMaps[r][jy][jx];
                }
            } else {
                // From previous job
                if (jobDistMaps[lastJobIdx]) {
                    d = jobDistMaps[lastJobIdx][jy][jx];
                }
            }
            
            if (d < 0) d = Infinity;
            
            let better = false;
            if (d < minD) {
                better = true;
            } else if (d === minD) {
                const idCurr = jobs[j].id;
                const idBest = jobs[best].id;
                if (idCurr < idBest) better = true;
            }
            
            if (better) {
                minD = d;
                best = j;
            }
        }
        
        if (best === -1) break;
        
        ordered.push(best);
        remaining.delete(best);
        lastJobIdx = best;
    }
    
    // Convert to job IDs
    const jobIds = ordered.map(jidx => jobs[jidx].id);
    assignments.push(jobIds);
}

// Output result
const output = { assignments };
console.log(JSON.stringify(output));
