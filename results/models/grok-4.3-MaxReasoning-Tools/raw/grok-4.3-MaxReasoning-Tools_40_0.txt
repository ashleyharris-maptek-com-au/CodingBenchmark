```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    int n, m;
    cin >> n >> m;
    vector<long long> c(n);
    for(int i = 0; i < n; i++) cin >> c[i];
    vector<vector<int>> adj(n);
    set<pair<int, int>> edge_set;
    for(int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        if(u > v) swap(u, v);
        if(u != v && edge_set.find({u, v}) == edge_set.end()) {
            edge_set.insert({u, v});
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }
    vector<int> active;
    for(int i = 0; i < n; i++) {
        if(c[i] > 0) active.push_back(i);
    }
    int k = active.size();
    if(k == 0) {
        cout << 0 << '\n';
        for(int i = 0; i < n; i++) {
            if(i > 0) cout << ' ';
            cout << 0;
        }
        cout << '\n';
        return 0;
    }
    vector<int> order = active;
    sort(order.begin(), order.end(), [&](int a, int b) {
        if(c[a] != c[b]) return c[a] > c[b];
        return a < b;
    });
    vector<int> pos(n, -1);
    for(int i = 0; i < k; i++) {
        pos[order[i]] = i;
    }
    vector<vector<int>> proc_low(n);
    for(int ii = 0; ii < k; ii++) {
        int v = order[ii];
        for(int u : adj[v]) {
            if(pos[u] != -1 && pos[u] < pos[v]) {
                proc_low[v].push_back(u);
            }
        }
    }
    vector<long long> suffix_max(k + 1, 0);
    for(int i = k - 1; i >= 0; i--) {
        suffix_max[i] = suffix_max[i + 1] + c[order[i]];
    }
    // Greedy initial solution
    vector<int> best(n, 0);
    long long ans = 0;
    vector<bool> g_taken(n, false);
    for(int v : order) {
        bool conflict = false;
        for(int u : adj[v]) {
            if(g_taken[u]) {
                conflict = true;
                break;
            }
        }
        if(!conflict) {
            g_taken[v] = true;
            ans += c[v];
        }
    }
    for(int i = 0; i < n; i++) {
        if(g_taken[i]) best[i] = 1;
    }
    vector<int> cur(n, 0);
    auto dfs = [&](auto&& self, int i, long long sum) -> void {
        if(i == k) {
            if(sum > ans) {
                ans = sum;
                best = cur;
            }
            return;
        }
        if(sum + suffix_max[i] <= ans) return;
        // skip
        self(self, i + 1, sum);
        // take
        int v = order[i];
        bool can = true;
        for(int u : proc_low[v]) {
            if(cur[u]) {
                can = false;
                break;
            }
        }
        if(can) {
            cur[v] = 1;
            self(self, i + 1, sum + c[v]);
            cur[v] = 0;
        }
    };
    dfs(dfs, 0, 0LL);
    cout << ans << '\n';
    for(int i = 0; i < n; i++) {
        if(i > 0) cout << ' ';
        cout << best[i];
    }
    cout << '\n';
    return 0;
}
```
