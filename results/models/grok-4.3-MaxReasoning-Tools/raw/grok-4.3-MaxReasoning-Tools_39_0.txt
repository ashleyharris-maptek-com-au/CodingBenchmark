use std::io::{self, BufRead};

fn compute_cut(partition: &[usize], adj: &[Vec<usize>], n: usize) -> usize {
    let mut in_s = vec![false; n];
    for &v in partition {
        in_s[v] = true;
    }
    let mut cut = 0usize;
    for &v in partition {
        for &u in &adj[v] {
            if !in_s[u] {
                cut += 1;
            }
        }
    }
    cut
}

fn search(
    current: &mut Vec<usize>,
    start: usize,
    remain: usize,
    adj: &[Vec<usize>],
    n: usize,
    min_cut: &mut usize,
    best: &mut Vec<usize>,
) {
    if remain == 0 {
        let cut = compute_cut(current, adj, n);
        if cut < *min_cut {
            *min_cut = cut;
            best.clear();
            best.extend_from_slice(current);
        }
        return;
    }
    for i in start..n {
        current.push(i);
        search(current, i + 1, remain - 1, adj, n, min_cut, best);
        current.pop();
    }
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut nums = first_line.split_whitespace();
    let n: usize = nums.next().unwrap().parse().unwrap();
    let m: usize = nums.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut parts = line.split_whitespace();
        if let (Some(u_str), Some(v_str)) = (parts.next(), parts.next()) {
            if let (Ok(u), Ok(v)) = (u_str.parse::<usize>(), v_str.parse::<usize>()) {
                if u != v && u < n && v < n {
                    adj[u].push(v);
                    adj[v].push(u);
                }
            }
        }
    }

    for i in 0..n {
        adj[i].sort_unstable();
        adj[i].dedup();
    }

    let k = n / 2;
    let mut min_cut = usize::MAX;
    let mut best: Vec<usize> = Vec::new();
    let mut current: Vec<usize> = Vec::new();
    search(&mut current, 0, k, &adj, n, &mut min_cut, &mut best);

    println!("{}", min_cut);
    for (i, &v) in best.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
