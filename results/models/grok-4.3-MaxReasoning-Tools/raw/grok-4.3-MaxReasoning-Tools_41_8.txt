; SPIR-V
; Version: 1.0
; Bound: 200
; Schema: 0

OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %in_worldPos %in_normal %in_color
OpExecutionMode %main OriginUpperLeft

OpDecorate %outColor Location 0
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 3
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBOStruct Block
OpMemberDecorate %UBOStruct 0 Offset 0
OpMemberDecorate %UBOStruct 1 Offset 64
OpMemberDecorate %UBOStruct 2 Offset 128
OpMemberDecorate %UBOStruct 3 Offset 192
OpMemberDecorate %UBOStruct 4 Offset 208
OpMemberDecorate %UBOStruct 5 Offset 224

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBOStruct = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %UBOStruct
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%bool = OpTypeBool
%int = OpTypeInt 32 1

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%float_0 = OpConstant %float 0
%float_025 = OpConstant %float 0.25
%float_05 = OpConstant %float 0.5
%float_075 = OpConstant %float 0.75
%float_1 = OpConstant %float 1
%float_02 = OpConstant %float 0.2
%float_04 = OpConstant %float 0.4
%float_07 = OpConstant %float 0.7

%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%in_color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_ubo Uniform

%main = OpFunction %void None %func
%entry = OpLabel

%worldPos = OpLoad %vec3 %in_worldPos
%normal = OpLoad %vec3 %in_normal
%color = OpLoad %vec3 %in_color

%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%lightDir = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2

%L_vec = OpFSub %vec3 %lightDir %worldPos
%L = OpExtInst %vec3 %glsl Normalize %L_vec

%N = OpExtInst %vec3 %glsl Normalize %normal

%dot_prod = OpDot %float %N %L
%d = OpExtInst %float %glsl FMax %dot_prod %float_0

%gt075 = OpFOrdGreaterThan %bool %d %float_075
%gt05 = OpFOrdGreaterThan %bool %d %float_05
%gt025 = OpFOrdGreaterThan %bool %d %float_025

%band1 = OpSelect %float %gt025 %float_04 %float_02
%band2 = OpSelect %float %gt05 %float_07 %band1
%band = OpSelect %float %gt075 %float_1 %band2

%diffuse_color = OpVectorTimesScalar %vec3 %color %band
%final_color = OpCompositeConstruct %vec4 %diffuse_color %float_1

OpStore %outColor %final_color
OpReturn
OpFunctionEnd
