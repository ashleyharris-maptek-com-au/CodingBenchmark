```spirv
; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
%GLSL450 = OpExtInstImport "GLSL.std.450"
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Input_vec2 = OpTypePointer Input %vec2
%_ptr_Output_vec4 = OpTypePointer Output %vec4
%_ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%bool = OpTypeBool
%int = OpTypeInt 32 1
%main_type = OpTypeFunction %void
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_16 = OpConstant %float 16
%pi = OpConstant %float 3.14159
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_0_9 = OpConstant %float 0.9
%float_0_1 = OpConstant %float 0.1
%red = OpConstantComposite %vec3 %float_0_9 %float_0_1 %float_0_1
%blue = OpConstantComposite %vec3 %float_0_1 %float_0_1 %float_0_9
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%in_worldPos = OpVariable %_ptr_Input_vec3 Input
%in_normal = OpVariable %_ptr_Input_vec3 Input
%in_uv = OpVariable %_ptr_Input_vec2 Input
%outColor = OpVariable %_ptr_Output_vec4 Output
%main = OpFunction %void None %main_type
%entry = OpLabel
%worldPos = OpLoad %vec3 %in_worldPos
%normal = OpLoad %vec3 %in_normal
%uv = OpLoad %vec2 %in_uv
%lightPos_ptr = OpAccessChain %_ptr_Uniform_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%lightPos_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%light_dir_vec = OpFSub %vec3 %lightPos_xyz %worldPos
%L = OpExtInst %vec3 %GLSL450 Normalize %light_dir_vec
%N = OpExtInst %vec3 %GLSL450 Normalize %normal
%dotNL = OpDot %float %N %L
%diffuse = OpExtInst %float %GLSL450 FMax %dotNL %float_0
%temp1 = OpFMul %float %float_0_6 %diffuse
%shading = OpFAdd %float %float_0_4 %temp1
%uv_y = OpCompositeExtract %float %uv 1
%scaled = OpFMul %float %uv_y %float_16
%angle = OpFMul %float %scaled %pi
%sin_val = OpExtInst %float %GLSL450 Sin %angle
%is_positive = OpFOrdGreaterThan %bool %sin_val %float_0
%stripe = OpSelect %float %is_positive %float_1 %float_0
%baseColor = OpExtInst %vec3 %GLSL450 FMix %blue %red %stripe
%shading_vec = OpCompositeConstruct %vec3 %shading %shading %shading
%final_rgb = OpFMul %vec3 %baseColor %shading_vec
%final_r = OpCompositeExtract %float %final_rgb 0
%final_g = OpCompositeExtract %float %final_rgb 1
%final_b = OpCompositeExtract %float %final_rgb 2
%out_val = OpCompositeConstruct %vec4 %final_r %final_g %final_b %float_1
OpStore %outColor %out_val
OpReturn
OpFunctionEnd
```
