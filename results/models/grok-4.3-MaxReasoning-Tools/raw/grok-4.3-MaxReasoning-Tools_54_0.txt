```python
import math

# Module-level persistent state (reset between scenarios and go-arounds)
_phase = "intercept"
_flaps_setting = 5.0
_gear_down = False
_loc_integral = 0.0
_gs_integral = 0.0
_speed_integral = 0.0
_elev_trim = 0.0
_touchdown_time = 0.0
_time_since_flare = 0.0
_last_radio_alt = 1000.0
_ga_triggered = False


def autopilot_step(state: dict, dt: float) -> dict:
    global _phase, _flaps_setting, _gear_down
    global _loc_integral, _gs_integral, _speed_integral
    global _elev_trim, _touchdown_time, _time_since_flare
    global _last_radio_alt, _ga_triggered

    def get(key, default=0.0):
        val = state.get(key)
        if val is None:
            return default
        return float(val)

    def get_valid(key_base, idx):
        valid = get(f"{key_base}_{idx}_valid", False)
        val = get(f"{key_base}_{idx}_dots" if "dots" in key_base else f"{key_base}_{idx}", 0.0)
        return val if valid else None

    # === SENSOR FUSION (median / average of valid) ===
    radio_alt = get("radio_alt_ft", 500.0)
    baro_alt = 0.0
    baro_cnt = 0
    for i in [1, 2, 3]:
        if get(f"baro_alt_{i}_valid", False):
            baro_alt += get(f"baro_alt_{i}_ft", radio_alt)
            baro_cnt += 1
    if baro_cnt > 0:
        baro_alt /= baro_cnt
    else:
        baro_alt = radio_alt

    # ILS Localizer
    loc_vals = []
    for i in [1, 2, 3]:
        v = get_valid("localizer", i)
        if v is not None:
            loc_vals.append(v)
    loc_dots = sum(loc_vals) / len(loc_vals) if loc_vals else 0.0

    # ILS Glideslope
    gs_vals = []
    for i in [1, 2, 3]:
        v = get_valid("glideslope", i)
        if v is not None:
            gs_vals.append(v)
    gs_dots = sum(gs_vals) / len(gs_vals) if gs_vals else 0.0

    # Airspeed
    ias_vals = []
    for i in [1, 2, 3]:
        if get(f"ias_{i}_valid", False):
            ias_vals.append(get(f"ias_{i}_kt", 140.0))
    ias = sum(ias_vals) / len(ias_vals) if ias_vals else get("ias_1_kt", 140.0)

    # Attitude
    pitch = get("irs_1_pitch_deg", 0.0)
    roll = get("irs_1_roll_deg", 0.0)
    heading = get("irs_1_heading_deg", 0.0)
    vs_fpm = get("irs_1_vs_fpm", 0.0)

    # Other
    target_ias = get("target_ias_kt", 140.0)
    target_vref = get("target_vref_kt", 137.0)
    on_ground = bool(get("weight_on_wheels", False) or get("on_ground", False))
    dme = get("dme_1_nm", 5.0)

    # Update last altitude for rate
    alt_rate = (radio_alt - _last_radio_alt) / max(dt, 0.01)
    _last_radio_alt = radio_alt

    # === PHASE MANAGEMENT ===
    if on_ground:
        _phase = "rollout"
    elif radio_alt < 8.0:
        _phase = "rollout"
    elif radio_alt < 45.0:
        _phase = "flare"
    elif radio_alt < 250.0:
        _phase = "approach"
    else:
        _phase = "intercept"

    # === INTEGRATORS (anti-windup) ===
    _loc_integral += loc_dots * dt
    _loc_integral = max(min(_loc_integral, 4.0), -4.0)

    _gs_integral += gs_dots * dt
    _gs_integral = max(min(_gs_integral, 3.5), -3.5)

    speed_err = target_ias - ias
    _speed_integral += speed_err * dt
    _speed_integral = max(min(_speed_integral, 15.0), -15.0)

    # === DEFAULT CONTROLS ===
    elevator = 0.0
    aileron = 0.0
    rudder = 0.0
    throttle_1 = 0.25
    throttle_2 = 0.25
    elevator_trim = _elev_trim
    flaps = _flaps_setting
    gear = 1.0 if _gear_down else 0.0
    speedbrake = 0.0
    spoiler = 0.0
    autobrake = 0.0
    reverse_1 = 0.0
    reverse_2 = 0.0

    # === CONFIGURATION (progressive) ===
    if radio_alt < 1800 and _flaps_setting < 10:
        _flaps_setting = 10.0
    if radio_alt < 1200 and _flaps_setting < 25:
        _flaps_setting = 25.0
    if radio_alt < 900 and not _gear_down:
        _gear_down = True
    if radio_alt < 600 and _flaps_setting < 35:
        _flaps_setting = 35.0
    if radio_alt < 350 and _flaps_setting < 40:
        _flaps_setting = 40.0

    flaps = _flaps_setting
    gear = 1.0 if _gear_down else 0.0

    # === GO-AROUND DETECTION (conservative) ===
    unstable = False
    if 80 < radio_alt < 450:
        if abs(gs_dots) > 1.35 or abs(loc_dots) > 1.35 or abs(speed_err) > 22:
            unstable = True
    if unstable and not _ga_triggered:
        _ga_triggered = True
        _phase = "intercept"
        _flaps_setting = 15.0
        _gear_down = False
        _loc_integral = 0.0
        _gs_integral = 0.0

    # === CONTROL LAWS ===
    if _phase in ["intercept", "approach"]:
        # Lateral (localizer tracking)
        loc_kp = 0.42
        loc_ki = 0.07
        aileron_cmd = -(loc_kp * loc_dots + loc_ki * _loc_integral)
        aileron = max(min(aileron_cmd, 0.95), -0.95)

        # Rudder for coordination / yaw
        rudder = max(min(-0.18 * loc_dots - 0.08 * roll, 0.6), -0.6)

        # Vertical (glideslope)
        gs_kp = 0.48
        gs_ki = 0.09
        pitch_cmd = -(gs_kp * gs_dots + gs_ki * _gs_integral)

        # Attitude blending
        pitch_err = pitch_cmd - pitch
        elevator = max(min(0.75 * pitch_err + 0.15 * pitch_cmd, 0.92), -0.92)

        # Speed (throttle)
        thr_kp = 0.028
        thr_ki = 0.006
        throttle_cmd = 0.42 + thr_kp * speed_err + thr_ki * _speed_integral
        throttle_1 = throttle_2 = max(0.0, min(1.0, throttle_cmd))

        # Gentle trim update
        _elev_trim = max(min(_elev_trim + 0.015 * dt * elevator, 0.4), -0.4)

    elif _phase == "flare":
        _time_since_flare += dt

        # Flare: gradual nose-up to reduce sink rate
        flare_height = max(0.0, 42.0 - radio_alt)
        flare_factor = min(flare_height / 42.0, 1.0)

        # Target pitch ~ 4-7 degrees nose up at touchdown
        target_pitch = 2.0 + 5.5 * flare_factor

        pitch_err = target_pitch - pitch
        elevator = max(min(0.85 * pitch_err, 0.95), -0.4)

        # Idle thrust
        throttle_1 = throttle_2 = 0.08

        # Keep aligned
        aileron = max(min(-0.25 * loc_dots, 0.7), -0.7)
        rudder = max(min(-0.22 * loc_dots, 0.65), -0.65)

        # Slight speedbrake if too fast
        if ias > target_vref + 12:
            speedbrake = 0.35

    elif _phase == "rollout":
        _touchdown_time += dt

        # Nose down after touchdown
        if _touchdown_time < 1.8:
            elevator = -0.25
        else:
            elevator = max(-0.55, -0.15 - 0.12 * (_touchdown_time - 2.0))

        # Ground spoilers + autobrake
        spoiler = 1.0
        autobrake = 0.85 if _touchdown_time > 1.5 else 0.55

        # Reverse thrust (full after nose down)
        if _touchdown_time > 1.2:
            reverse_1 = 0.85
            reverse_2 = 0.85

        # Throttles idle
        throttle_1 = 0.0
        throttle_2 = 0.0

        # Directional control via rudder (localizer still useful on rollout)
        rudder = max(min(-0.55 * loc_dots, 0.75), -0.75)

        # Reduce flaps slowly on rollout (not critical)
        if _touchdown_time > 8.0 and _flaps_setting > 10:
            _flaps_setting = max(10.0, _flaps_setting - 4.0 * dt)

        # Disarm reverse when slow
        if ias < 55.0:
            reverse_1 = 0.0
            reverse_2 = 0.0
            autobrake = 0.4

    # === FAILURE / EDGE HANDLING ===
    # Single engine failure handling (rudder compensation)
    eng1_running = get("engine_1_running", True)
    eng2_running = get("engine_2_running", True)
    if not eng1_running and eng2_running:
        rudder = max(min(rudder + 0.22, 0.85), -0.85)
    elif not eng2_running and eng1_running:
        rudder = max(min(rudder - 0.22, 0.85), -0.85)

    # Very high sink rate protection (flare early)
    if radio_alt < 120 and vs_fpm < -900 and _phase != "rollout":
        elevator = max(elevator, 0.55)

    # Clamp all outputs
    elevator = max(min(elevator, 1.0), -1.0)
    aileron = max(min(aileron, 1.0), -1.0)
    rudder = max(min(rudder, 1.0), -1.0)
    throttle_1 = max(min(throttle_1, 1.0), 0.0)
    throttle_2 = max(min(throttle_2, 1.0), 0.0)
    elevator_trim = max(min(elevator_trim, 0.5), -0.5)
    flaps = max(min(flaps, 40.0), 0.0)
    gear = max(min(gear, 1.0), 0.0)
    speedbrake = max(min(speedbrake, 1.0), 0.0)
    spoiler = max(min(spoiler, 1.0), 0.0)
    autobrake = max(min(autobrake, 1.0), 0.0)
    reverse_1 = max(min(reverse_1, 1.0), 0.0)
    reverse_2 = max(min(reverse_2, 1.0), 0.0)

    # Return control dict
    return {
        "elevator": float(elevator),
        "aileron": float(aileron),
        "rudder": float(rudder),
        "throttle_1": float(throttle_1),
        "throttle_2": float(throttle_2),
        "elevator_trim": float(elevator_trim),
        "flaps": float(flaps),
        "gear": float(gear),
        "speedbrake": float(speedbrake),
        "spoiler": float(spoiler),
        "autobrake": float(autobrake),
        "reverse_1": float(reverse_1),
        "reverse_2": float(reverse_2),
    }
```
