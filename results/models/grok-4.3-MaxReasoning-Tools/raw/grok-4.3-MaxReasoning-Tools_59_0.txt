        <pre id="code"><code>const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

const prob = data.probability;
const drones = data.drones || [];
const maxSteps = data.maxStepsPerDrone || 0;

const rows = prob.length;
const cols = rows > 0 ? prob[0].length : 0;

function isValid(x, y) {
    return x >= 0 && x < cols && y >= 0 && y < rows && prob[y][x] !== -1;
}

const moveDirs = [[-1, 0], [1, 0], [0, -1], [0, 1]];

const covered = Array.from({ length: rows }, () => Array(cols).fill(false));
const paths = [];

for (let d = 0; d < drones.length; d++) {
    const drone = drones[d];
    let cx = drone.x;
    let cy = drone.y;
    
    let path = [[cx, cy]];
    
    if (prob[cy] && prob[cy][cx] > 0) {
        covered[cy][cx] = true;
    }
    
    let stepsLeft = maxSteps;
    
    while (stepsLeft > 0) {
        // BFS to find best target cell
        const queue = [];
        const visited = Array.from({ length: rows }, () => Array(cols).fill(false));
        const prevX = Array.from({ length: rows }, () => Array(cols).fill(-1));
        const prevY = Array.from({ length: rows }, () => Array(cols).fill(-1));
        
        queue.push({ x: cx, y: cy, dist: 0 });
        visited[cy][cx] = true;
        
        let bestScore = -1;
        let bestDist = Infinity;
        let targetX = cx;
        let targetY = cy;
        
        while (queue.length > 0) {
            const curr = queue.shift();
            const curx = curr.x;
            const cury = curr.y;
            const dist = curr.dist;
            
            if (dist > stepsLeft) continue;
            
            const cellVal = prob[cury] && prob[cury][curx] !== undefined ? prob[cury][curx] : -1;
            let curScore = (covered[cury][curx] || cellVal <= 0) ? -1 : cellVal;
            
            if (curScore > bestScore || (curScore === bestScore && dist < bestDist)) {
                bestScore = curScore;
                bestDist = dist;
                targetX = curx;
                targetY = cury;
            }
            
            for (const [dx, dy] of moveDirs) {
                const nx = curx + dx;
                const ny = cury + dy;
                
                if (isValid(nx, ny) && !visited[ny][nx] && dist + 1 <= stepsLeft) {
                    visited[ny][nx] = true;
                    queue.push({ x: nx, y: ny, dist: dist + 1 });
                    prevX[ny][nx] = curx;
                    prevY[ny][nx] = cury;
                }
            }
        }
        
        // If no valuable uncovered cell reachable
        if (bestScore <= 0) {
            for (let k = 0; k < stepsLeft; k++) {
                path.push([cx, cy]);
            }
            break;
        }
        
        // Reconstruct shortest path to target
        let segPath = [];
        let tx = targetX;
        let ty = targetY;
        
        while (!(tx === cx && ty === cy)) {
            segPath.unshift([tx, ty]);
            const px = prevX[ty][tx];
            const py = prevY[ty][tx];
            
            if (px === -1) break;
            
            tx = px;
            ty = py;
        }
        
        // Append path segments
        let moved = false;
        for (let i = 0; i < segPath.length && stepsLeft > 0; i++) {
            const [nx, ny] = segPath[i];
            path.push([nx, ny]);
            cx = nx;
            cy = ny;
            
            if (prob[cy] && prob[cy][cx] > 0 && !covered[cy][cx]) {
                covered[cy][cx] = true;
            }
            
            stepsLeft--;
            moved = true;
        }
        
        // Safety: if no movement happened, stay once and decrement
        if (!moved && stepsLeft > 0) {
            path.push([cx, cy]);
            stepsLeft--;
        }
    }
    
    // Ensure path has exactly maxSteps + 1 positions (pad with stays if needed)
    while (path.length < maxSteps + 1) {
        path.push([cx, cy]);
    }
    
    // Trim if somehow too long (shouldn't happen)
    if (path.length > maxSteps + 1) {
        path = path.slice(0, maxSteps + 1);
    }
    
    paths.push(path);
}

const output = { paths: paths };
console.log(JSON.stringify(output));</code></pre>
        
        <div class="section">
            <p><strong>How it works:</strong></p>
            <ul>
                <li>Reads JSON from stdin</li>
                <li>For each drone, uses BFS from current position to find the highest-value uncovered cell within remaining steps</li>
                <li>Moves along shortest valid path to that cell</li>
                <li>Updates global coverage to maximize unique probability</li>
                <li>Fills remaining steps with stays if needed</li>
                <li>Outputs valid JSON with paths array</li>
            </ul>
            <p>Scales to reasonably sized grids (e.g. 100x100) and multiple drones.</p>
