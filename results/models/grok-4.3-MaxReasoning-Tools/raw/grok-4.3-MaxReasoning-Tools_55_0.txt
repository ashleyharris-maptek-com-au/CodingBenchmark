#include <iostream>
#include <vector>
#include <string>
#include <sstream>
#include <chrono>
#include <algorithm>
#include <map>
#include <cctype>

using namespace std;

struct Move {
    int from;
    int to;
    char promotion;
    bool capture;
};

struct UndoInfo {
    char captured;
    string old_castling;
    int old_ep;
    int old_half;
    char original_piece;
    bool castling;
    int rook_from;
    int rook_to;
    bool was_en_passant;
    int ep_capture_sq;
};

char board[64];
char current_side;
string castling_rights;
int ep_square;
int halfmove_clock;
int fullmove_number;

map<char, int> piece_value = {
    {'P', 100}, {'N', 320}, {'B', 330}, {'R', 500}, {'Q', 900}, {'K', 20000},
    {'p', -100}, {'n', -320}, {'b', -330}, {'r', -500}, {'q', -900}, {'k', -20000}
};

string sq_to_str(int sq) {
    if (sq < 0 || sq > 63) return "";
    char file = 'a' + (sq % 8);
    char rank = '1' + (sq / 8);
    return string(1, file) + rank;
}

int str_to_sq(const string& s) {
    if (s == "-" || s.empty()) return -1;
    return (s[0] - 'a') + 8 * (s[1] - '1');
}

void load_fen(const string& fen) {
    stringstream ss(fen);
    string board_part, active, castle, enp, half, full;
    ss >> board_part >> active >> castle >> enp >> half >> full;
    int index = 0;
    for (char c : board_part) {
        if (c == '/') continue;
        if (isdigit(c)) {
            int skip = c - '0';
            for (int i = 0; i < skip; i++) {
                if (index < 64) board[index++] = '.';
            }
        } else if (isalpha(c)) {
            if (index < 64) board[index++] = c;
        }
    }
    current_side = active[0];
    castling_rights = castle;
    ep_square = str_to_sq(enp);
    halfmove_clock = stoi(half);
    fullmove_number = stoi(full);
}

int find_king(char side) {
    char king = (side == 'w') ? 'K' : 'k';
    for (int i = 0; i < 64; i++) {
        if (board[i] == king) return i;
    }
    return -1;
}

bool is_attacked(int sq, char by_side) {
    if (sq < 0 || sq > 63) return false;
    int f = sq % 8;
    int r = sq / 8;

    // Pawns
    if (by_side == 'w') {
        if (r > 0) {
            if (f < 7 && board[(r-1)*8 + (f+1)] == 'P') return true;
            if (f > 0 && board[(r-1)*8 + (f-1)] == 'P') return true;
        }
    } else {
        if (r < 7) {
            if (f < 7 && board[(r+1)*8 + (f+1)] == 'p') return true;
            if (f > 0 && board[(r+1)*8 + (f-1)] == 'p') return true;
        }
    }

    // Knights
    int knight_deltas[8] = {-17, -15, -10, -6, 6, 10, 15, 17};
    for (int d : knight_deltas) {
        int nsq = sq + d;
        if (nsq >= 0 && nsq < 64) {
            int fd = abs((nsq % 8) - (sq % 8));
            int rd = abs((nsq / 8) - (sq / 8));
            if ((fd == 1 && rd == 2) || (fd == 2 && rd == 1)) {
                char p = board[nsq];
                if ((by_side == 'w' && p == 'N') || (by_side == 'b' && p == 'n')) return true;
            }
        }
    }

    // King
    int king_deltas[8] = {-9, -8, -7, -1, 1, 7, 8, 9};
    for (int d : king_deltas) {
        int nsq = sq + d;
        if (nsq >= 0 && nsq < 64) {
            int fd = abs((nsq % 8) - (sq % 8));
            int rd = abs((nsq / 8) - (sq / 8));
            if (fd <= 1 && rd <= 1) {
                char p = board[nsq];
                if ((by_side == 'w' && p == 'K') || (by_side == 'b' && p == 'k')) return true;
            }
        }
    }

    // Sliding - Rook/Queen orthogonal
    int rook_dfs[4] = {-1, 1, 0, 0};
    int rook_drs[4] = {0, 0, -1, 1};
    for (int i = 0; i < 4; i++) {
        int df = rook_dfs[i];
        int dr = rook_drs[i];
        int cf = f;
        int cr = r;
        while (true) {
            cf += df;
            cr += dr;
            if (cf < 0 || cf > 7 || cr < 0 || cr > 7) break;
            int nsq = cr * 8 + cf;
            char p = board[nsq];
            if (p != '.') {
                if ((by_side == 'w' && (p == 'R' || p == 'Q')) || (by_side == 'b' && (p == 'r' || p == 'q'))) return true;
                break;
            }
        }
    }

    // Sliding - Bishop/Queen diagonal
    int bishop_dfs[4] = {-1, 1, -1, 1};
    int bishop_drs[4] = {-1, -1, 1, 1};
    for (int i = 0; i < 4; i++) {
        int df = bishop_dfs[i];
        int dr = bishop_drs[i];
        int cf = f;
        int cr = r;
        while (true) {
            cf += df;
            cr += dr;
            if (cf < 0 || cf > 7 || cr < 0 || cr > 7) break;
            int nsq = cr * 8 + cf;
            char p = board[nsq];
            if (p != '.') {
                if ((by_side == 'w' && (p == 'B' || p == 'Q')) || (by_side == 'b' && (p == 'b' || p == 'q'))) return true;
                break;
            }
        }
    }

    return false;
}

bool is_in_check(char side) {
    int ksq = find_king(side);
    if (ksq == -1) return true;
    char opp = (side == 'w' ? 'b' : 'w');
    return is_attacked(ksq, opp);
}

UndoInfo make_move(Move m) {
    UndoInfo u;
    u.captured = board[m.to];
    u.old_castling = castling_rights;
    u.old_ep = ep_square;
    u.old_half = halfmove_clock;
    u.original_piece = board[m.from];
    u.castling = false;
    u.rook_from = -1;
    u.rook_to = -1;
    u.was_en_passant = false;
    u.ep_capture_sq = -1;

    char piece = board[m.from];
    bool is_pawn = (tolower(piece) == 'p');
    bool is_king = (tolower(piece) == 'k');
    bool is_rook_move = (tolower(piece) == 'r');

    board[m.to] = piece;
    board[m.from] = '.';

    if (m.promotion != 0) {
        board[m.to] = m.promotion;
    }

    // En passant
    if (is_pawn && ep_square != -1 && m.to == ep_square) {
        u.was_en_passant = true;
        u.ep_capture_sq = (current_side == 'w' ? m.to - 8 : m.to + 8);
        u.captured = board[u.ep_capture_sq];
        board[u.ep_capture_sq] = '.';
    }

    // Castling
    if (is_king && abs(m.to - m.from) == 2) {
        u.castling = true;
        if (m.to > m.from) { // kingside
            u.rook_from = (current_side == 'w') ? 7 : 63;
            u.rook_to = (current_side == 'w') ? 5 : 61;
        } else { // queenside
            u.rook_from = (current_side == 'w') ? 0 : 56;
            u.rook_to = (current_side == 'w') ? 3 : 59;
        }
        board[u.rook_to] = board[u.rook_from];
        board[u.rook_from] = '.';
    }

    // Update castling rights
    if (is_king) {
        string newc = "";
        for (char c : castling_rights) {
            if (current_side == 'w') {
                if (c != 'K' && c != 'Q') newc += c;
            } else {
                if (c != 'k' && c != 'q') newc += c;
            }
        }
        castling_rights = newc;
    }
    if (is_rook_move) {
        if (current_side == 'w') {
            if (m.from == 7) castling_rights.erase(remove(castling_rights.begin(), castling_rights.end(), 'K'), castling_rights.end());
            if (m.from == 0) castling_rights.erase(remove(castling_rights.begin(), castling_rights.end(), 'Q'), castling_rights.end());
        } else {
            if (m.from == 63) castling_rights.erase(remove(castling_rights.begin(), castling_rights.end(), 'k'), castling_rights.end());
            if (m.from == 56) castling_rights.erase(remove(castling_rights.begin(), castling_rights.end(), 'q'), castling_rights.end());
        }
    }

    // EP
    ep_square = -1;
    if (is_pawn && abs(m.to - m.from) == 16) {
        ep_square = (m.from + m.to) / 2;
    }

    // Halfmove
    if (is_pawn || u.captured != '.') {
        halfmove_clock = 0;
    } else {
        halfmove_clock++;
    }

    // Switch side
    current_side = (current_side == 'w' ? 'b' : 'w');

    return u;
}

void unmake_move(Move m, UndoInfo u) {
    board[m.from] = u.original_piece;

    if (u.castling) {
        board[m.to] = '.';
        board[u.rook_from] = board[u.rook_to];
        board[u.rook_to] = '.';
    } else if (u.was_en_passant) {
        board[m.to] = '.';
        board[u.ep_capture_sq] = u.captured;
    } else {
        board[m.to] = u.captured;
    }

    castling_rights = u.old_castling;
    ep_square = u.old_ep;
    halfmove_clock = u.old_half;

    current_side = (current_side == 'w' ? 'b' : 'w');
}

vector<Move> generate_pseudo_moves() {
    vector<Move> moves;
    char my_pawn = (current_side == 'w' ? 'P' : 'p');
    for (int sq = 0; sq < 64; sq++) {
        char piece = board[sq];
        if (piece == '.') continue;
        bool white_piece = isupper(piece);
        if ((current_side == 'w' && !white_piece) || (current_side == 'b' && white_piece)) continue;

        if (tolower(piece) == 'p') {
            int dir = (current_side == 'w' ? 8 : -8);
            int start_r = (current_side == 'w' ? 1 : 6);
            int promo_r = (current_side == 'w' ? 7 : 0);
            // forward
            int to = sq + dir;
            if (to >= 0 && to < 64 && board[to] == '.') {
                if ((sq / 8) == promo_r) {
                    vector<char> promos = {'Q', 'R', 'B', 'N'};
                    for (char pr : promos) {
                        char promo_p = (current_side == 'w' ? pr : tolower(pr));
                        moves.push_back({sq, to, promo_p, false});
                    }
                } else {
                    moves.push_back({sq, to, 0, false});
                }
                if ((sq / 8) == start_r) {
                    int to2 = sq + 2 * dir;
                    if (to2 >= 0 && to2 < 64 && board[to2] == '.') {
                        moves.push_back({sq, to2, 0, false});
                    }
                }
            }
            // captures
            int cap_d[2] = {dir - 1, dir + 1};
            for (int d : cap_d) {
                int to_c = sq + d;
                if (to_c >= 0 && to_c < 64 && abs((to_c % 8) - (sq % 8)) == 1) {
                    char targ = board[to_c];
                    if (targ != '.' && ((current_side == 'w' && islower(targ)) || (current_side == 'b' && isupper(targ)))) {
                        if ((sq / 8) == promo_r) {
                            vector<char> promos = {'Q', 'R', 'B', 'N'};
                            for (char pr : promos) {
                                char promo_p = (current_side == 'w' ? pr : tolower(pr));
                                moves.push_back({sq, to_c, promo_p, true});
                            }
                        } else {
                            moves.push_back({sq, to_c, 0, true});
                        }
                    }
                }
            }
            // en passant
            if (ep_square != -1) {
                for (int d : cap_d) {
                    int to_c = sq + d;
                    if (to_c == ep_square && abs((to_c % 8) - (sq % 8)) == 1) {
                        moves.push_back({sq, to_c, 0, true});
                    }
                }
            }
        } else if (tolower(piece) == 'n') {
            int deltas[8] = {-17,-15,-10,-6,6,10,15,17};
            for (int d : deltas) {
                int to = sq + d;
                if (to >= 0 && to < 64) {
                    int fd = abs((to % 8) - (sq % 8));
                    int rd = abs((to / 8) - (sq / 8));
                    if ((fd == 1 && rd == 2) || (fd == 2 && rd == 1)) {
                        char targ = board[to];
                        if (targ == '.' || (current_side == 'w' && islower(targ)) || (current_side == 'b' && isupper(targ))) {
                            moves.push_back({sq, to, 0, (targ != '.')});
                        }
                    }
                }
            }
        } else if (tolower(piece) == 'b' || tolower(piece) == 'r' || tolower(piece) == 'q') {
            vector<pair<int, int>> dirs;
            if (tolower(piece) == 'b' || tolower(piece) == 'q') {
                dirs.emplace_back(-1, -1); dirs.emplace_back(-1, 1);
                dirs.emplace_back(1, -1); dirs.emplace_back(1, 1);
            }
            if (tolower(piece) == 'r' || tolower(piece) == 'q') {
                dirs.emplace_back(-1, 0); dirs.emplace_back(1, 0);
                dirs.emplace_back(0, -1); dirs.emplace_back(0, 1);
            }
            for (auto [df, dr] : dirs) {
                int cf = sq % 8;
                int cr = sq / 8;
                while (true) {
                    cf += df;
                    cr += dr;
                    if (cf < 0 || cf > 7 || cr < 0 || cr > 7) break;
                    int to = cr * 8 + cf;
                    char targ = board[to];
                    if (targ == '.') {
                        moves.push_back({sq, to, 0, false});
                    } else {
                        if ((current_side == 'w' && islower(targ)) || (current_side == 'b' && isupper(targ))) {
                            moves.push_back({sq, to, 0, true});
                        }
                        break;
                    }
                }
            }
        } else if (tolower(piece) == 'k') {
            int deltas[8] = {-9,-8,-7,-1,1,7,8,9};
            for (int d : deltas) {
                int to = sq + d;
                if (to >= 0 && to < 64) {
                    int fd = abs((to % 8) - (sq % 8));
                    int rd = abs((to / 8) - (sq / 8));
                    if (fd <= 1 && rd <= 1) {
                        char targ = board[to];
                        if (targ == '.' || (current_side == 'w' && islower(targ)) || (current_side == 'b' && isupper(targ))) {
                            moves.push_back({sq, to, 0, (targ != '.')});
                        }
                    }
                }
            }
            // Castling pseudo
            if (current_side == 'w') {
                if (castling_rights.find('K') != string::npos && board[5] == '.' && board[6] == '.') {
                    moves.push_back({sq, 6, 0, false});
                }
                if (castling_rights.find('Q') != string::npos && board[1] == '.' && board[2] == '.' && board[3] == '.') {
                    moves.push_back({sq, 2, 0, false});
                }
            } else {
                if (castling_rights.find('k') != string::npos && board[61] == '.' && board[62] == '.') {
                    moves.push_back({sq, 62, 0, false});
                }
                if (castling_rights.find('q') != string::npos && board[57] == '.' && board[58] == '.' && board[59] == '.') {
                    moves.push_back({sq, 58, 0, false});
                }
            }
        }
    }
    return moves;
}

vector<Move> generate_legal_moves() {
    vector<Move> pseudo = generate_pseudo_moves();
    vector<Move> legal;
    for (auto m : pseudo) {
        if (tolower(board[m.from]) == 'k' && abs(m.to - m.from) == 2) {
            if (is_in_check(current_side)) continue;
            // Basic path check for castling
            if (current_side == 'w') {
                if (m.to == 6) { // O-O
                    if (is_attacked(5, 'b') || is_attacked(6, 'b')) continue;
                } else if (m.to == 2) { // O-O-O
                    if (is_attacked(3, 'b') || is_attacked(2, 'b') || is_attacked(1, 'b')) continue;
                }
            } else {
                if (m.to == 62) {
                    if (is_attacked(61, 'w') || is_attacked(62, 'w')) continue;
                } else if (m.to == 58) {
                    if (is_attacked(59, 'w') || is_attacked(58, 'w') || is_attacked(57, 'w')) continue;
                }
            }
        }
        UndoInfo u = make_move(m);
        char mover_side = (current_side == 'w' ? 'b' : 'w');
        if (!is_in_check(mover_side)) {
            legal.push_back(m);
        }
        unmake_move(m, u);
    }
    return legal;
}

int evaluate() {
    int score = 0;
    for (int i = 0; i < 64; i++) {
        if (board[i] != '.') {
            auto it = piece_value.find(board[i]);
            if (it != piece_value.end()) score += it->second;
        }
    }
    return score;
}

int alpha_beta(int depth, int alpha, int beta, chrono::steady_clock::time_point start_time, double timeout) {
    auto now = chrono::steady_clock::now();
    double elapsed = chrono::duration<double>(now - start_time).count();
    if (elapsed > timeout - 0.3) return evaluate();

    if (depth == 0) return evaluate();

    vector<Move> moves = generate_legal_moves();
    if (moves.empty()) {
        if (is_in_check(current_side)) {
            return (current_side == 'w' ? -20000 : 20000);
        }
        return 0;
    }

    for (auto& m : moves) {
        now = chrono::steady_clock::now();
        elapsed = chrono::duration<double>(now - start_time).count();
        if (elapsed > timeout - 0.3) break;

        UndoInfo u = make_move(m);
        int score = -alpha_beta(depth - 1, -beta, -alpha, start_time, timeout);
        unmake_move(m, u);
        if (score >= beta) return beta;
        if (score > alpha) alpha = score;
    }
    return alpha;
}

Move find_best_move(double timeout_sec) {
    auto start_time = chrono::steady_clock::now();
    vector<Move> moves = generate_legal_moves();
    if (moves.empty()) {
        return {-1, -1, 0, false};
    }

    Move best = moves[0];
    int best_score = -10000000;

    // Iterative deepening limited by time
    for (int d = 1; d <= 6; d++) {
        auto now = chrono::steady_clock::now();
        double elapsed = chrono::duration<double>(now - start_time).count();
        if (elapsed > timeout_sec * 0.85) break;

        int alpha = -10000000;
        int beta = 10000000;
        Move local_best = moves[0];
        int local_best_score = -10000000;

        for (auto& m : moves) {
            now = chrono::steady_clock::now();
            elapsed = chrono::duration<double>(now - start_time).count();
            if (elapsed > timeout_sec * 0.85) break;

            UndoInfo u = make_move(m);
            int score = -alpha_beta(d - 1, -beta, -alpha, start_time, timeout_sec);
            unmake_move(m, u);

            if (score > local_best_score) {
                local_best_score = score;
                local_best = m;
            }
        }

        if (local_best_score > best_score) {
            best_score = local_best_score;
            best = local_best;
        }
    }

    return best;
}

string get_move_notation(Move m) {
    if (m.from < 0) return "O-O";
    char piece = board[m.from];
    string notation = "";
    if (tolower(piece) == 'p') {
        notation += "p";
    } else {
        notation += toupper(piece);
    }
    notation += sq_to_str(m.from);
    if (m.capture) notation += "x";
    notation += sq_to_str(m.to);
    if (m.promotion != 0) {
        notation += "/";
        notation += toupper(m.promotion);
    }
    if (tolower(piece) == 'k' && abs(m.to - m.from) == 2) {
        notation = (m.to > m.from ? "O-O" : "O-O-O");
    }
    return notation;
}

int main() {
    string timeout_str;
    getline(cin, timeout_str);
    double timeout = stod(timeout_str);
    string fen;
    getline(cin, fen);

    load_fen(fen);

    Move best = find_best_move(timeout);
    string move_str = get_move_notation(best);
    cout << move_str << endl;

    return 0;
}
