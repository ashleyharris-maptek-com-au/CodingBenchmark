#include <iostream>
#include <vector>
#include <algorithm>

struct Item {
    int w, h, id;
};

struct Level {
    int y, h, used_w;
};

struct BinState {
    std::vector<Level> levels;
};

struct Answer {
    int board_idx, x, y, rotated;
};

bool place_in_existing(BinState& bin, int rw, int rh, int bin_idx, int item_id, bool is_rot, std::vector<Answer>& answers, int board_w, int board_h) {
    if (rw > board_w || rh > board_h) return false;
    for (auto& lev : bin.levels) {
        if (rh <= lev.h && lev.used_w + rw <= board_w) {
            answers[item_id] = {bin_idx, lev.used_w, lev.y, is_rot ? 1 : 0};
            lev.used_w += rw;
            return true;
        }
    }
    int new_y = 0;
    if (!bin.levels.empty()) {
        const Level& last = bin.levels.back();
        new_y = last.y + last.h;
    }
    if (new_y + rh <= board_h && rw <= board_w) {
        bin.levels.push_back({new_y, rh, rw});
        answers[item_id] = {bin_idx, 0, new_y, is_rot ? 1 : 0};
        return true;
    }
    return false;
}

void place_new_bin(BinState& bin, int rw, int rh, int bin_idx, int item_id, bool is_rot, std::vector<Answer>& answers) {
    bin.levels.push_back({0, rh, rw});
    answers[item_id] = {bin_idx, 0, 0, is_rot ? 1 : 0};
}

int main() {
    int N, BOARD_W, BOARD_H;
    std::cin >> N >> BOARD_W >> BOARD_H;
    std::vector<Item> items(N);
    for (int i = 0; i < N; i++) {
        std::cin >> items[i].w >> items[i].h;
        items[i].id = i;
    }
    std::sort(items.begin(), items.end(), [](const Item& a, const Item& b) {
        int ma = std::max(a.w, a.h);
        int mb = std::max(b.w, b.h);
        if (ma != mb) return ma > mb;
        return a.w * 1LL * a.h > b.w * 1LL * b.h;
    });
    std::vector<BinState> bins;
    std::vector<Answer> answers(N);
    for (const auto& item : items) {
        bool placed = false;
        for (size_t bi = 0; bi < bins.size() && !placed; ++bi) {
            if (place_in_existing(bins[bi], item.w, item.h, (int)bi, item.id, false, answers, BOARD_W, BOARD_H)) {
                placed = true;
            } else if (place_in_existing(bins[bi], item.h, item.w, (int)bi, item.id, true, answers, BOARD_W, BOARD_H)) {
                placed = true;
            }
        }
        if (!placed) {
            BinState new_bin;
            bins.push_back(new_bin);
            int new_bi = (int)bins.size() - 1;
            if (item.w <= BOARD_W && item.h <= BOARD_H) {
                place_new_bin(bins.back(), item.w, item.h, new_bi, item.id, false, answers);
            } else if (item.h <= BOARD_W && item.w <= BOARD_H) {
                place_new_bin(bins.back(), item.h, item.w, new_bi, item.id, true, answers);
            }
        }
    }
    std::cout << bins.size() << '\n';
    for (int i = 0; i < N; i++) {
        const Answer& a = answers[i];
        std::cout << a.board_idx << " " << a.x << " " << a.y << " " << a.rotated << '\n';
    }
    return 0;
}
