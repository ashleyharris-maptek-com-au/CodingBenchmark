```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>
#include <cstdint>

struct Item {
    int w;
    int h;
    int id;
};

struct Placement {
    int item_id;
    int x;
    int y;
    bool rotated;
    int placed_w;
    int placed_h;
};

struct Shelf {
    int y;
    int height;
    int used_width;
};

class Bin {
public:
    int W;
    int H;
    std::vector<Shelf> shelves;
    std::vector<Placement> placements;
    int max_y;

    Bin(int w, int h) : W(w), H(h), max_y(0) {}

    bool try_place(const Item& item) {
        for (int r = 0; r < 2; ++r) {
            int pw = (r == 1 ? item.h : item.w);
            int ph = (r == 1 ? item.w : item.h);

            if (pw > W || ph > H) continue;

            // Best-fit shelf: tightest remaining width that fits
            int best_idx = -1;
            int best_rem = INT_MAX;

            for (size_t i = 0; i < shelves.size(); ++i) {
                int rem = W - shelves[i].used_width;
                if (ph <= shelves[i].height && pw <= rem && rem < best_rem) {
                    best_rem = rem;
                    best_idx = static_cast<int>(i);
                }
            }

            if (best_idx != -1) {
                Shelf& sh = shelves[best_idx];
                Placement p;
                p.item_id = item.id;
                p.x = sh.used_width;
                p.y = sh.y;
                p.rotated = (r == 1);
                p.placed_w = pw;
                p.placed_h = ph;
                placements.push_back(p);
                sh.used_width += pw;
                return true;
            }

            // Try creating a new shelf at the top
            if (max_y + ph <= H && pw <= W) {
                Shelf new_shelf;
                new_shelf.y = max_y;
                new_shelf.height = ph;
                new_shelf.used_width = pw;
                shelves.push_back(new_shelf);
                max_y += ph;

                Placement p;
                p.item_id = item.id;
                p.x = 0;
                p.y = new_shelf.y;
                p.rotated = (r == 1);
                p.placed_w = pw;
                p.placed_h = ph;
                placements.push_back(p);
                return true;
            }
        }
        return false;
    }
};

struct OutputInfo {
    int board_idx;
    int x;
    int y;
    int rotated;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N, W, H;
    std::cin >> N >> W >> H;

    std::vector<Item> items(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> items[i].w >> items[i].h;
        items[i].id = i;
    }

    // Sort by decreasing area (primary), then width (tiebreaker)
    std::sort(items.begin(), items.end(), [](const Item& a, const Item& b) {
        long long area_a = static_cast<long long>(a.w) * a.h;
        long long area_b = static_cast<long long>(b.w) * b.h;
        if (area_a != area_b) return area_a > area_b;
        return a.w > b.w;
    });

    std::vector<Bin> bins;
    bins.reserve(64); // heuristic reserve

    for (const auto& item : items) {
        bool placed = false;
        for (auto& bin : bins) {
            if (bin.try_place(item)) {
                placed = true;
                break;
            }
        }
        if (!placed) {
            Bin new_bin(W, H);
            if (new_bin.try_place(item)) {
                bins.push_back(std::move(new_bin));
            } else {
                // Fallback: try again with original orientation only (rare case)
                // If still fails, place at (0,0) anyway (assumes input valid)
                Bin fallback(W, H);
                Placement p;
                p.item_id = item.id;
                p.x = 0;
                p.y = 0;
                p.rotated = 0;
                p.placed_w = item.w;
                p.placed_h = item.h;
                if (item.w <= W && item.h <= H) {
                    fallback.placements.push_back(p);
                    fallback.shelves.push_back({0, item.h, item.w});
                    fallback.max_y = item.h;
                } else if (item.h <= W && item.w <= H) {
                    p.rotated = 1;
                    p.placed_w = item.h;
                    p.placed_h = item.w;
                    fallback.placements.push_back(p);
                    fallback.shelves.push_back({0, item.w, item.h});
                    fallback.max_y = item.w;
                }
                bins.push_back(std::move(fallback));
            }
        }
    }

    // Build output mapping
    std::vector<OutputInfo> results(N, {0, 0, 0, 0});
    for (size_t b = 0; b < bins.size(); ++b) {
        for (const auto& p : bins[b].placements) {
            results[p.item_id].board_idx = static_cast<int>(b);
            results[p.item_id].x = p.x;
            results[p.item_id].y = p.y;
            results[p.item_id].rotated = p.rotated ? 1 : 0;
        }
    }

    // Output
    std::cout << bins.size() << '\n';
    for (int i = 0; i < N; ++i) {
        const auto& r = results[i];
        std::cout << r.board_idx << ' ' << r.x << ' ' << r.y << ' ' << r.rotated << '\n';
    }

    return 0;
}
```
