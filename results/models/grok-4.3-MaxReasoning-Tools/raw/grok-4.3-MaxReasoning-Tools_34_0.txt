```python
import sys
import math

def main():
    input_lines = sys.stdin.read().strip().splitlines()
    if not input_lines:
        return
    
    header = input_lines[0].split()
    N = int(header[0])
    K = int(header[1])
    CAPACITY = float(header[2])
    
    depot_line = input_lines[1].split()
    depot = (float(depot_line[0]), float(depot_line[1]))
    
    customers = []
    demands = []
    for i in range(N):
        line = input_lines[2 + i].split()
        x = float(line[0])
        y = float(line[1])
        d = float(line[2])
        customers.append((x, y))
        demands.append(d)
    
    if N == 0:
        print(0.0)
        print(0)
        return
    
    def euclid(p1, p2):
        return math.hypot(p1[0] - p2[0], p1[1] - p2[1])
    
    dist_depot = [euclid(depot, customers[i]) for i in range(N)]
    
    # Order customers: largest demand first, then farthest from depot
    cust_order = sorted(range(N), key=lambda i: (-demands[i], -dist_depot[i]))
    
    routes = []
    loads = []
    
    for cust in cust_order:
        best_r = -1
        best_cost = float('inf')
        
        for r in range(len(routes)):
            if loads[r] + demands[cust] <= CAPACITY + 1e-9:
                if not routes[r]:
                    cost = dist_depot[cust]
                else:
                    min_dist = min(euclid(customers[cust], customers[j]) for j in routes[r])
                    cost = min_dist
                if cost < best_cost:
                    best_cost = cost
                    best_r = r
        
        if best_r == -1 and len(routes) < K:
            routes.append([])
            loads.append(0.0)
            best_r = len(routes) - 1
        
        if best_r != -1:
            routes[best_r].append(cust)
            loads[best_r] += demands[cust]
    
    # Build ordered routes using nearest neighbor
    def build_ordered_route(cust_list):
        if not cust_list:
            return []
        unvisited = set(cust_list)
        route = []
        current = depot
        while unvisited:
            best = min(unvisited, key=lambda c: euclid(current, customers[c]))
            route.append(best)
            current = customers[best]
            unvisited.remove(best)
        return route
    
    ordered_routes = []
    for r_list in routes:
        if r_list:
            ordered_routes.append(build_ordered_route(r_list))
    
    routes = ordered_routes
    
    # Calculate route distance
    def calc_route_dist(route):
        if not route:
            return 0.0
        total = euclid(depot, customers[route[0]])
        for i in range(len(route) - 1):
            total += euclid(customers[route[i]], customers[route[i + 1]])
        total += euclid(customers[route[-1]], depot)
        return total
    
    # 2-opt improvement
    def two_opt(route):
        if len(route) < 3:
            return route
        
        best = route[:]
        best_dist = calc_route_dist(best)
        improved = True
        max_passes = 30
        
        for _ in range(max_passes):
            if not improved:
                break
            improved = False
            for i in range(len(route)):
                for j in range(i + 2, len(route) + (1 if i > 0 else 0)):
                    # Create new route by reversing segment [i:j]
                    if j >= len(route):
                        new_route = route[i:] + route[:j - len(route)]
                        new_route = new_route[::-1]  # but careful with cycle, simpler path reverse
                    else:
                        new_route = route[:i] + route[i:j][::-1] + route[j:]
                    
                    # For path optimization (depot fixed)
                    new_dist = calc_route_dist(new_route)
                    if new_dist < best_dist - 1e-9:
                        best = new_route
                        best_dist = new_dist
                        route = new_route[:]
                        improved = True
                        break
                if improved:
                    break
            if not improved:
                break
        
        return best
    
    # Apply 2-opt to each route
    for i in range(len(routes)):
        routes[i] = two_opt(routes[i])
    
    # Filter any empty routes (should not happen)
    routes = [r for r in routes if r]
    
    # Compute total distance
    total_distance = sum(calc_route_dist(r) for r in routes)
    
    # Output
    print(total_distance)
    print(len(routes))
    for r in routes:
        print(' '.join(map(str, r)))

if __name__ == "__main__":
    main()
```
