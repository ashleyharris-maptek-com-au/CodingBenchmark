using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public class Program
{
    public static void Main(string[] args)
    {
        string headerLine = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(headerLine))
        {
            Console.WriteLine("0");
            return;
        }

        string[] hparts = headerLine.Trim().Split(new char[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        if (hparts.Length < 6)
        {
            Console.WriteLine("0");
            return;
        }

        int XC = int.Parse(hparts[0]);
        int YC = int.Parse(hparts[1]);
        int ZC = int.Parse(hparts[2]);
        double XB = double.Parse(hparts[3]);
        double YB = double.Parse(hparts[4]);
        double ZB = double.Parse(hparts[5]);

        long totalVoxels = (long)XC * YC * ZC;

        var stdinStream = Console.OpenStandardInput();
        BinaryReader br = new BinaryReader(stdinStream, Encoding.ASCII, true);

        var valuables = new List<(int x, int y, int z, int mass, int value)>();

        int maxValuables = 2000;
        bool isLarge = totalVoxels > 5_000_000;

        try
        {
            for (int x = 0; x < XC; x++)
            {
                for (int y = 0; y < YC; y++)
                {
                    for (int z = 0; z < ZC; z++)
                    {
                        if (!isLarge && valuables.Count >= maxValuables * 2) break;

                        byte d;
                        try
                        {
                            d = br.ReadByte();
                        }
                        catch
                        {
                            break;
                        }

                        int mass = d % 16;
                        int val = d / 16;

                        if (val > 0 && mass > 0)
                        {
                            valuables.Add((x, y, z, mass, val));
                            if (valuables.Count >= maxValuables && isLarge)
                            {
                                break;
                            }
                        }
                    }
                    if (valuables.Count >= maxValuables && isLarge) break;
                }
                if (valuables.Count >= maxValuables && isLarge) break;
            }
        }
        catch (EndOfStreamException)
        {
            // Partial read ok
        }

        // Adaptive: sort by potential profit (value * mass)
        valuables.Sort((a, b) =>
        {
            long profitA = (long)a.value * a.mass;
            long profitB = (long)b.value * b.mass;
            return profitB.CompareTo(profitA);
        });

        // Limit operations for large scale to stay under time
        int maxOps = isLarge ? 800 : 5000;
        if (valuables.Count > maxOps)
        {
            valuables = valuables.GetRange(0, maxOps);
        }

        // Generate operations
        var operations = new List<string>();

        foreach (var v in valuables)
        {
            int x = v.x;
            int y = v.y;
            int z = v.z;
            int mass = v.mass;
            int val = v.value;

            // Generate 4-connected Manhattan path to (0,0): X first then Y
            var path = new List<(int px, int py)>();
            int cx = x;
            int cy = y;

            // Move X
            while (cx != 0)
            {
                cx += (cx > 0 ? -1 : 1);
                path.Add((cx, cy));
            }

            // Move Y
            while (cy != 0)
            {
                cy += (cy > 0 ? -1 : 1);
                path.Add((cx, cy));
            }

            // Build operation string
            StringBuilder opBuilder = new StringBuilder();
            opBuilder.Append(path.Count);
            opBuilder.Append($" {x} {y} {z}");

            foreach (var p in path)
            {
                opBuilder.Append($" {p.px} {p.py}");
            }

            // Final destination: processing plant (0,0) - valuable ore disappears
            opBuilder.Append(" 0 0");

            operations.Add(opBuilder.ToString());
        }

        // Output
        Console.WriteLine(operations.Count);
        foreach (string op in operations)
        {
            Console.WriteLine(op);
        }
    }
}
