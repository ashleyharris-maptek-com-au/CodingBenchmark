```csharp
using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(firstLine[0]);
        int M = int.Parse(firstLine[1]);

        var adj = new List<List<(int to, int eid)>>(N);
        for (int i = 0; i < N; i++) adj.Add(new List<(int, int)>());

        var neighborSet = new HashSet<int>[N];
        for (int i = 0; i < N; i++) neighborSet[i] = new HashSet<int>();

        for (int i = 0; i < M; i++)
        {
            string[] parts = Console.ReadLine().Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            // int w = int.Parse(parts[2]); // weights ignored for route

            if (u != v)
            {
                adj[u].Add((v, i));
                adj[v].Add((u, i));
                neighborSet[u].Add(v);
                neighborSet[v].Add(u);
            }
            else
            {
                // self-loop: degree +2, handled automatically
                adj[u].Add((v, i));
                adj[v].Add((u, i));
            }
        }

        var g = new List<List<int>>(N);
        for (int i = 0; i < N; i++)
        {
            g.Add(new List<int>(neighborSet[i]));
        }

        // Collect odd degree nodes
        var odds = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if (adj[i].Count % 2 == 1)
                odds.Add(i);
        }

        // List<bool> for used edges (extendable)
        var edgeUsed = new List<bool>(M);
        for (int i = 0; i < M; i++) edgeUsed.Add(false);

        // Duplicate paths to pair odds
        for (int i = 0; i < odds.Count; i += 2)
        {
            if (i + 1 >= odds.Count) break;
            int a = odds[i];
            int b = odds[i + 1];
            if (a == b) continue;

            var path = FindPath(a, b, g, N);
            if (path != null && path.Count > 1)
            {
                DuplicatePath(path, adj, edgeUsed);
            }
        }

        // Now run Hierholzer's algorithm
        int[] nextEdgeIdx = new int[N];
        var circuit = new List<int>();
        var stack = new Stack<int>();
        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            bool advanced = false;

            while (nextEdgeIdx[u] < adj[u].Count)
            {
                var (v, eid) = adj[u][nextEdgeIdx[u]];
                if (eid < edgeUsed.Count && !edgeUsed[eid])
                {
                    edgeUsed[eid] = true;
                    nextEdgeIdx[u]++;
                    stack.Push(v);
                    advanced = true;
                    break;
                }
                else
                {
                    nextEdgeIdx[u]++;
                }
            }

            if (!advanced)
            {
                circuit.Add(u);
                stack.Pop();
            }
        }

        circuit.Reverse();

        // Output the route (starts and ends at 0 if successful)
        if (circuit.Count > 0)
        {
            // Ensure starts at 0
            if (circuit[0] != 0 && circuit.Count > 1)
            {
                // Fallback: if not starting at 0, try to rotate (rare)
                int idx = circuit.IndexOf(0);
                if (idx >= 0)
                {
                    var newCircuit = new List<int>();
                    for (int i = idx; i < circuit.Count; i++) newCircuit.Add(circuit[i]);
                    for (int i = 0; i < idx; i++) newCircuit.Add(circuit[i]);
                    circuit = newCircuit;
                }
            }

            Console.WriteLine(string.Join(" ", circuit));
        }
        else
        {
            Console.WriteLine("0");
        }
    }

    static List<int> FindPath(int src, int dest, List<List<int>> g, int N)
    {
        if (src == dest) return new List<int> { src };

        bool[] visited = new bool[N];
        int[] parent = new int[N];
        Array.Fill(parent, -1);

        var queue = new Queue<int>();
        queue.Enqueue(src);
        visited[src] = true;

        while (queue.Count > 0)
        {
            int u = queue.Dequeue();
            foreach (int v in g[u])
            {
                if (!visited[v])
                {
                    visited[v] = true;
                    parent[v] = u;
                    queue.Enqueue(v);
                    if (v == dest)
                    {
                        var path = new List<int>();
                        int current = dest;
                        while (current != -1)
                        {
                            path.Add(current);
                            current = parent[current];
                        }
                        path.Reverse();
                        return path;
                    }
                }
            }
        }
        return null;
    }

    static void DuplicatePath(List<int> path, List<List<(int, int)>> adj, List<bool> edgeUsed)
    {
        for (int j = 0; j < path.Count - 1; j++)
        {
            int x = path[j];
            int y = path[j + 1];
            int newEid = edgeUsed.Count;
            adj[x].Add((y, newEid));
            adj[y].Add((x, newEid));
            edgeUsed.Add(false);
        }
    }
}
```
