```csharp
using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        string input = Console.In.ReadToEnd();
        string[] tokens = input.Split(new char[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);

        int N = int.Parse(tokens[0]);
        int L = int.Parse(tokens[1]);

        int[] lengths = new int[N];
        for (int i = 0; i < N; i++)
        {
            lengths[i] = int.Parse(tokens[2 + i]);
        }

        // Create index array and sort by length descending (FFD)
        int[] indices = new int[N];
        for (int i = 0; i < N; i++)
        {
            indices[i] = i;
        }

        Array.Sort(indices, (a, b) => lengths[b].CompareTo(lengths[a]));

        List<int> binRemaining = new List<int>();
        List<List<int>> binItems = new List<List<int>>();

        foreach (int idx in indices)
        {
            int size = lengths[idx];
            bool placed = false;

            for (int b = 0; b < binRemaining.Count; b++)
            {
                if (binRemaining[b] >= size)
                {
                    binRemaining[b] -= size;
                    binItems[b].Add(idx);
                    placed = true;
                    break;
                }
            }

            if (!placed)
            {
                binRemaining.Add(L - size);
                binItems.Add(new List<int> { idx });
            }
        }

        // Output
        int numStocks = binItems.Count;
        Console.WriteLine(numStocks);

        for (int b = 0; b < numStocks; b++)
        {
            binItems[b].Sort(); // sort indices ascending for readability
            Console.WriteLine(string.Join(" ", binItems[b]));
        }
    }
}
```
