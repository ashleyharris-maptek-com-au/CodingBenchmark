```csharp
using System;
using System.Collections.Generic;

class Program
{
    class RemComparer : IComparer<(int rem, int binId)>
    {
        public int Compare((int rem, int binId) x, (int rem, int binId) y)
        {
            if (x.rem != y.rem)
                return y.rem.CompareTo(x.rem);
            return x.binId.CompareTo(y.binId);
        }
    }

    static void Main()
    {
        string line1 = Console.ReadLine();
        string[] parts = line1.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(parts[0]);
        int stock = int.Parse(parts[1]);

        string line2 = Console.ReadLine();
        string[] cutParts = line2.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);

        var items = new List<(int length, int index)>(N);
        for (int i = 0; i < N; i++)
        {
            int len = int.Parse(cutParts[i]);
            items.Add((len, i));
        }

        items.Sort((a, b) => b.length.CompareTo(a.length));

        var binAssignments = new List<List<int>>();
        var binRemain = new List<int>();
        var remSet = new SortedSet<(int rem, int binId)>(new RemComparer());

        for (int i = 0; i < items.Count; i++)
        {
            var (len, idx) = items[i];
            bool placed = false;

            if (remSet.Count > 0)
            {
                var maxEntry = remSet.Max;
                if (maxEntry.rem >= len)
                {
                    for (int b = 0; b < binRemain.Count; b++)
                    {
                        if (binRemain[b] >= len)
                        {
                            int oldRem = binRemain[b];
                            remSet.Remove((oldRem, b));
                            binRemain[b] -= len;
                            binAssignments[b].Add(idx);
                            remSet.Add((binRemain[b], b));
                            placed = true;
                            break;
                        }
                    }
                }
            }

            if (!placed)
            {
                int b = binAssignments.Count;
                binAssignments.Add(new List<int> { idx });
                binRemain.Add(stock - len);
                remSet.Add((stock - len, b));
            }
        }

        Console.WriteLine(binAssignments.Count);
        for (int b = 0; b < binAssignments.Count; b++)
        {
            Console.WriteLine(string.Join(" ", binAssignments[b]));
        }
    }
}
```
