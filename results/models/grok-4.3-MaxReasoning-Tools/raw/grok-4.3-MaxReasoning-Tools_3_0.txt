use std::io::{self, BufRead};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let reader = stdin.lock();
    let mut lines = reader.lines();

    let header = match lines.next() {
        Some(Ok(line)) => line,
        _ => {
            println!("0");
            return;
        }
    };

    let mut iter = header.split_whitespace();
    let n: usize = match iter.next() {
        Some(s) => match s.parse() {
            Ok(val) => val,
        Err(_) => 0,
        },
        None => 0,
    };
    let m: usize = match iter.next() {
        Some(s) => match s.parse() {
            Ok(val) => val,
            Err(_) => 0,
        },
        None => 0,
    };

    if n == 0 {
        println!("0");
        return;
    }

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];

    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let mut parts = line.split_whitespace();
            if let (Some(u_str), Some(v_str)) = (parts.next(), parts.next()) {
                if let (Ok(u), Ok(v)) = (u_str.parse::<usize>(), v_str.parse::<usize>()) {
                    if u < n && v < n && u != v {
                        adj[u].push(v);
                        adj[v].push(u);
                    }
                }
            }
        }
    }

    let positions = compute_layout(n, adj);

    println!("{}", n);
    for (x, y) in positions {
        println!("{:.6} {:.6}", x, y);
    }
}

fn compute_layout(n: usize, mut adj: Vec<Vec<usize>>) -> Vec<(f64, f64)> {
    if n == 0 {
        return vec![];
    }

    // Deduplicate adjacency lists to avoid inflated degrees
    for i in 0..n {
        adj[i].sort_unstable();
        adj[i].dedup();
    }

    // Compute degrees
    let mut degrees: Vec<usize> = adj.iter().map(|neighbors| neighbors.len()).collect();

    // Find start node: highest degree
    let mut start = 0usize;
    for i in 1..n {
        if degrees[i] > degrees[start] {
            start = i;
        }
    }

    // BFS traversal order covering all components
    let mut order: Vec<usize> = Vec::with_capacity(n);
    let mut visited = vec![false; n];
    let mut queue: VecDeque<usize> = VecDeque::new();

    // Start BFS from highest-degree node
    queue.push_back(start);
    visited[start] = true;

    while let Some(u) = queue.pop_front() {
        order.push(u);
        for &v in &adj[u] {
            if !visited[v] {
                visited[v] = true;
                queue.push_back(v);
            }
        }
    }

    // Handle remaining disconnected components
    for i in 0..n {
        if !visited[i] {
            queue.push_back(i);
            visited[i] = true;
            while let Some(u) = queue.pop_front() {
                order.push(u);
                for &v in &adj[u] {
                    if !visited[v] {
                        visited[v] = true;
                        queue.push_back(v);
                    }
                }
            }
        }
    }

    // Grid layout: nodes close in BFS order are placed near each other
    let cols = ((n as f64).sqrt().ceil() as usize).max(1);
    let spacing: f64 = 12.0;

    let mut positions = vec![(0.0f64, 0.0f64); n];

    for (idx, &node) in order.iter().enumerate() {
        let row = idx / cols;
        let col = idx % cols;
        let x = col as f64 * spacing;
        let y = row as f64 * spacing;
        positions[node] = (x, y);
    }

    // Optional centering around origin (helps visualization)
    if n > 1 {
        let mut sum_x = 0.0;
        let mut sum_y = 0.0;
        for &(x, y) in &positions {
            sum_x += x;
            sum_y += y;
        }
        let cx = sum_x / n as f64;
        let cy = sum_y / n as f64;

        for pos in &mut positions {
            pos.0 -= cx;
            pos.1 -= cy;
        }
    }

    positions
}
