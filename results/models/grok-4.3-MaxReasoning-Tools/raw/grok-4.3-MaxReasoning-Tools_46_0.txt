```glsl
#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 physics;
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    if (i >= N) {
        return;
    }

    // Read current state of body i
    vec4 posMass = inData[i * 2 + 0];
    vec4 velPad  = inData[i * 2 + 1];

    vec3 pos   = posMass.xyz;
    float mass = posMass.w;
    vec3 vel   = velPad.xyz;

    // Extract simulation parameters
    float dt         = physics.x;
    float softening  = physics.y;
    float G          = physics.z;

    // Compute acceleration due to all other bodies
    vec3 acc = vec3(0.0);

    for (uint j = 0; j < N; ++j) {
        if (j == i) {
            continue;
        }

        vec4 posMassJ = inData[j * 2 + 0];
        vec3 posJ = posMassJ.xyz;
        float massJ = posMassJ.w;

        vec3 delta = posJ - pos;
        float distSq = dot(delta, delta) + softening * softening;

        float invDist = inversesqrt(distSq);
        float invDist3 = invDist * invDist * invDist;

        acc += G * massJ * delta * invDist3;
    }

    // Integrate: Euler / leapfrog style
    vec3 newPos = pos + vel * dt;
    vec3 newVel = vel + acc * dt;

    // Write updated state to output buffer
    outData[i * 2 + 0] = vec4(newPos, mass);
    outData[i * 2 + 1] = vec4(newVel, 0.0);
}
```
