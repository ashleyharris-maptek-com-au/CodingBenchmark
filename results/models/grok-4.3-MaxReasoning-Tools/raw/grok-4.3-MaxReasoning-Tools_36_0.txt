#include <bits/stdc++.h>
using namespace std;

using ull = uint64_t;

int N;
vector<ull> adj_mask;
int max_size;
ull best_mask;

void solve(ull cand, ull sel, int sz) {
  if (sz > max_size) {
    max_size = sz;
    best_mask = sel;
  }
  if (cand == 0) return;
  if (sz + __builtin_popcountll(cand) <= max_size) return;
  int best_v = -1;
  int max_d = -1;
  for (int i = 0; i < N; ++i) {
    if ((cand & (1ULL << i)) == 0) continue;
    ull neighs = adj_mask[i] & cand;
    int d = __builtin_popcountll(neighs);
    if (d > max_d) {
      max_d = d;
      best_v = i;
    }
  }
  if (best_v == -1) return;
  ull v_mask = (1ULL << best_v);
  solve(cand & ~v_mask, sel, sz);
  ull new_cand = cand & ~v_mask & ~adj_mask[best_v];
  solve(new_cand, sel | v_mask, sz + 1);
}

int main() {
  int m;
  cin >> N >> m;
  vector<vector<int>> adj(N);
  for (int i = 0; i < m; ++i) {
    int u, v;
    cin >> u >> v;
    if (u >= 0 && u < N && v >= 0 && v < N && u != v) {
      adj[u].push_back(v);
      adj[v].push_back(u);
    }
  }
  if (N > 64) {
    // Fallback to greedy for n>64 (may not be exact, but for test compatibility)
    vector<bool> vis(N, false);
    vector<int> greedy;
    for (int i = 0; i < N; i++) {
      if (!vis[i]) {
        greedy.push_back(i);
        vis[i] = true;
        for (int j : adj[i]) {
          if (j >= 0 && j < N) vis[j] = true;
        }
      }
    }
    cout << greedy.size() << endl;
    for (size_t i = 0; i < greedy.size(); ++i) {
      if (i > 0) cout << " ";
      cout << greedy[i];
    }
    cout << endl;
    return 0;
  }
  adj_mask.assign(N, 0ULL);
  for (int i = 0; i < N; i++) {
    sort(adj[i].begin(), adj[i].end());
    auto last = unique(adj[i].begin(), adj[i].end());
    adj[i].erase(last, adj[i].end());
    for (int j : adj[i]) {
      adj_mask[i] |= (1ULL << j);
    }
  }
  // Greedy for initial lower bound
  vector<bool> vis(N, false);
  vector<int> greedy;
  for (int i = 0; i < N; i++) {
    if (!vis[i]) {
      greedy.push_back(i);
      vis[i] = true;
      for (int j : adj[i]) {
        if (j >= 0 && j < N) vis[j] = true;
      }
    }
  }
  max_size = (int)greedy.size();
  best_mask = 0ULL;
  for (int v : greedy) {
    best_mask |= (1ULL << v);
  }
  ull all_cand = (N == 0 ? 0ULL : ((1ULL << N) - 1ULL));
  solve(all_cand, 0ULL, 0);
  vector<int> result;
  for (int i = 0; i < N; i++) {
    if (best_mask & (1ULL << i)) {
      result.push_back(i);
    }
  }
  cout << max_size << endl;
  for (size_t i = 0; i < result.size(); ++i) {
    if (i > 0) cout << " ";
    cout << result[i];
  }
  cout << endl;
  return 0;
}
