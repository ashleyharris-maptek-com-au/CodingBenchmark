```cpp
#include <iostream>
#include <iomanip>
#include <string>
#include <vector>
#include <thread>
#include <mutex>
#include <atomic>
#include <chrono>
#include <cmath>
#include <algorithm>
#include <fstream>
#include <numbers>

struct State {
    double x = 0, y = 0, z = 0;
    double vx = 0, vy = 0, vz = 0;
    double pitch = 0, yaw = 0, roll = 0;
    double prate = 0, yrate = 0, rrate = 0;
    double fuel = 0;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    std::cout.tie(nullptr);
    std::cout << std::fixed << std::setprecision(4);

    double width, depth, height, gravity, max_thrust, initial_fuel;
    double start_x, start_y, start_z, target_x, target_y, target_z;
    std::string terrain_path;

    std::cin >> width >> depth >> height >> gravity >> max_thrust >> initial_fuel;
    std::cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z;

    std::string line;
    std::getline(std::cin, line); // consume rest of line
    std::getline(std::cin, terrain_path);

    // Skip possible "STATE" marker
    std::string marker;
    if (std::cin >> marker) {
        if (marker != "STATE" && marker != "state") {
            // If not STATE, it might be first state value - but unlikely; proceed
        }
    }

    // Optional: background terrain parse (grace period)
    std::thread terrain_thread([terrain_path, width, depth]() {
        std::ifstream file(terrain_path, std::ios::binary);
        if (file) {
            file.seekg(0, std::ios::end);
            auto fsize = file.tellg();
            // Read some to validate, but don't block long
            if (fsize > 0) {
                file.seekg(0);
                uint16_t dummy;
                for (int i = 0; i < 100 && file; ++i) {
                    file.read(reinterpret_cast<char*>(&dummy), 2);
                }
            }
        }
    });

    State current_state;
    bool has_new_state = false;
    std::mutex state_mutex;
    std::atomic<bool> running{true};

    auto reader = [&]() {
        while (running.load()) {
            double vals[13];
            bool success = true;
            for (int i = 0; i < 13; ++i) {
                if (!(std::cin >> vals[i])) {
                    success = false;
                    break;
                }
            }
            if (!success) {
                running = false;
                break;
            }
            {
                std::lock_guard<std::mutex> lock(state_mutex);
                current_state.x = vals[0];
                current_state.y = vals[1];
                current_state.z = vals[2];
                current_state.vx = vals[3];
                current_state.vy = vals[4];
                current_state.vz = vals[5];
                current_state.pitch = vals[6];
                current_state.yaw = vals[7];
                current_state.roll = vals[8];
                current_state.prate = vals[9];
                current_state.yrate = vals[10];
                current_state.rrate = vals[11];
                current_state.fuel = vals[12];
                has_new_state = true;
            }
        }
    };

    std::thread reader_thread(reader);

    // Persistent controller state
    double i_vz = 0.0;
    double last_dist_h = 10000.0;
    const double PI = std::numbers::pi;

    auto compute_command = [&](const State& s) -> std::tuple<double, double, double, double> {
        double ex = target_x - s.x;
        double ey = target_y - s.y;
        double dist_h = std::sqrt(ex * ex + ey * ey);

        // Horizontal velocity targets (scale aware)
        double max_h_speed = std::min(80.0, dist_h * 0.15 + 5.0);
        double k_pos = 0.04;
        double vx_des = k_pos * ex;
        double vy_des = k_pos * ey;
        vx_des = std::clamp(vx_des, -max_h_speed, max_h_speed);
        vy_des = std::clamp(vy_des, -max_h_speed, max_h_speed);

        double k_vel_h = 0.9;
        double ax_des = k_vel_h * (vx_des - s.vx);
        double ay_des = k_vel_h * (vy_des - s.vy);
        ax_des = std::clamp(ax_des, -8.0, 8.0);
        ay_des = std::clamp(ay_des, -8.0, 8.0);

        // Vertical target
        double target_alt = target_z + 12.0;
        if (dist_h < 12.0 && std::abs(s.vx) < 3.0 && std::abs(s.vy) < 3.0) {
            target_alt = target_z + 3.5;
        }
        if (dist_h < 4.5 && std::abs(s.vx) < 1.8 && std::abs(s.vy) < 1.8 && s.z < target_z + 8.0) {
            target_alt = target_z + 0.8;
        }
        if (dist_h < 2.0 && std::abs(s.vx) < 1.0 && std::abs(s.vy) < 1.0 && s.z < target_z + 4.0) {
            target_alt = target_z - 0.3;
        }

        double ez = target_alt - s.z;
        double vz_des = 0.75 * ez;
        vz_des = std::clamp(vz_des, -4.5, 3.0);

        double evz = vz_des - s.vz;
        i_vz += evz * 0.08;
        i_vz = std::clamp(i_vz, -3.0, 3.0);

        double az_des = gravity + 0.85 * evz + 0.25 * ez;
        az_des = std::clamp(az_des, -3.0, 12.0);

        // Compute desired thrust direction from desired accel
        double tdx = ax_des;
        double tdy = ay_des;
        double tdz = az_des;
        double total = std::sqrt(tdx*tdx + tdy*tdy + tdz*tdz);
        double thrust = 0.0;
        double pitch_des = 0.0;
        double yaw_des = s.yaw;

        if (total > 0.05) {
            tdx /= total;
            tdy /= total;
            tdz /= total;

            yaw_des = std::atan2(tdy, tdx);

            // Compute sin(pitch)
            double cp = std::cos(yaw_des);
            double sin_p_val;
            if (std::abs(cp) > 0.15) {
                sin_p_val = -tdx / cp;
            } else {
                double sp = std::sin(yaw_des);
                sin_p_val = (std::abs(sp) > 0.01) ? -tdy / sp : 0.0;
            }
            sin_p_val = std::clamp(sin_p_val, -1.0, 1.0);
            pitch_des = std::asin(sin_p_val);

            // Thrust magnitude estimate
            double vert_comp = tdz;
            if (vert_comp < 0.3) vert_comp = 0.3;
            thrust = (az_des / vert_comp) * 0.55; // empirical scaling for mass/thrust
        } else {
            pitch_des = 0.0;
            yaw_des = s.yaw;
            thrust = 0.45 + 0.6 * evz;
        }

        // Refine thrust with vertical PID + integral
        thrust = 0.38 + 0.72 * evz + 0.18 * i_vz;
        if (dist_h < 8.0) {
            thrust *= 0.92;
        }

        // Tilt compensation
        double tilt = std::abs(pitch_des);
        double cos_t = std::cos(tilt);
        if (cos_t > 0.2) thrust /= cos_t;
        thrust = std::clamp(thrust, 0.05, 1.0);

        // Attitude control (PD on angle + rate)
        double pitch_err = pitch_des - s.pitch;
        double yaw_err = yaw_des - s.yaw;

        // Normalize yaw error
        while (yaw_err > PI) yaw_err -= 2 * PI;
        while (yaw_err < -PI) yaw_err += 2 * PI;

        double p_rate_des = 2.8 * pitch_err;
        double y_rate_des = 2.8 * yaw_err;
        p_rate_des = std::clamp(p_rate_des, -2.5, 2.5);
        y_rate_des = std::clamp(y_rate_des, -2.5, 2.5);

        double p_cmd = 1.8 * (p_rate_des - s.prate) + 1.2 * pitch_err;
        double y_cmd = 1.8 * (y_rate_des - s.yrate) + 1.2 * yaw_err;
        double r_cmd = -2.8 * s.rrate;

        p_cmd = std::clamp(p_cmd, -1.0, 1.0);
        y_cmd = std::clamp(y_cmd, -1.0, 1.0);
        r_cmd = std::clamp(r_cmd, -1.0, 1.0);

        return {thrust, p_cmd, y_cmd, r_cmd};
    };

    // Main control loop
    State local_state;
    auto last_output = std::chrono::steady_clock::now();

    while (running.load()) {
        bool got_state = false;
        {
            std::lock_guard<std::mutex> lock(state_mutex);
            if (has_new_state) {
                local_state = current_state;
                has_new_state = false;
                got_state = true;
            }
        }

        if (got_state) {
            auto [thrust, p_cmd, y_cmd, r_cmd] = compute_command(local_state);
            std::cout << thrust << " " << p_cmd << " " << y_cmd << " " << r_cmd << "\n";
            std::cout.flush();
            last_output = std::chrono::steady_clock::now();
        } else {
            // Output last command or neutral to keep alive
            auto now = std::chrono::steady_clock::now();
            if (std::chrono::duration_cast<std::chrono::milliseconds>(now - last_output).count() > 80) {
                std::cout << "0.35 0.0 0.0 0.0\n";
                std::cout.flush();
                last_output = now;
            }
            std::this_thread::sleep_for(std::chrono::milliseconds(12));
        }

        // Safety: prevent busy loop
        std::this_thread::sleep_for(std::chrono::milliseconds(5));
    }

    if (terrain_thread.joinable()) terrain_thread.join();
    if (reader_thread.joinable()) reader_thread.join();

    return 0;
}
```
