fn dfs(current: &mut Vec<usize>, start: usize, remain: usize, adj: &[Vec<usize>], n: usize, min : &mut usize, best: &mut Vec<usize>) {

    if remain == 0 {

        let cut = compute_cut(current, adj, n);

        if cut < *min {

            *min = cut;

            best.clear();

            best.extend(current.iter().cloned());

        }

        return;

    }

    for i in start..n {

        current.push(i);

        dfs(current, i+1, remain-1, adj, n, min, best);

        current.pop();

    }

}

fn compute_cut(partition: &[usize], adj: &[Vec<usize>], n: usize) -> usize {

    let mut in_s = vec![false; n];

     for &v in partition {

         in_s[v] = true;

     }

     let mut cut = 0;

     for &v in partition {

         for &u in &adj[v] {

             if !in_s[u] {

                 cut += 1;

             }

         }

     }

     cut

}
