vector<int> held_karp(const vector<Point>& pts) {

  int n = pts.size();

  if(n == 1) return {0};

  int maxm = 1 << n;

  vector<vector<double>> dp(maxm, vector<double>(n, numeric_limits<double >::infinity()));

  vector<vector<int>> prev(maxm, vector<int>(n, -1));

  dp[1][0] = 0;

  for(int mask = 0; mask < maxm; mask++) {

    for(int u = 0; u < n; u++) {

      if ((mask & (1 << u)) == 0) continue;

      if(dp[mask][u] == numeric_limits<double>::infinity()) continue;

      for(int v = 0; v < n; v++) {

        if(mask & (1 << v)) continue;

        double cost = dp[mask][u ] + dist(pts, u, v);

        int nmask = mask | (1 << v);

        if(cost < dp[nmask][v]) {

          dp[nmask][v] = cost;

          prev[nmask][v] = u;

        }

      }

    }

  }

 int full = (1 << n) - 1;

  double min_cost = numeric_limits<double>::infinity();

  int end_city = -1;

  for(int u = 0; u < n; u++) {

    if(dp[full][u] < min_cost) {

      min_cost = dp [full][u];

      end_city = u;

    }

  }

  vector<int> path;

  int mask = full;

  int cur = end_city;

  while(cur != -1) {

    path.push_back(cur);

    if(mask == 1) break; // only 0

    int p = prev[mask][cur];

    mask &= ~(1 << cur);

    cur = p;

  }

  reverse(path.begin(), path.end());

  return path;
