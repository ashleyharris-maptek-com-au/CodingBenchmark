function getBurnedValue(cuts: boolean[][], towns: any[], ignitions: number[][], width: number, height: number): number {

  // multi source BFS from ignitions

  const dirs = [[0,1],[ 1,0],[0,-1],[-1,0]];

  const visited = new Array(height).fill(null).map(() => new Array(width).fill(false));

  const queue: [number,number][] = [];

  for (const [ix,iy] of ignitions) {

    if (grid[i y][ix] === '.' && !cuts[iy][ix]) {

      queue.push([ix,iy]);

      visited[iy][ix] = true;

    }

  }

  while (queue.length >0) {

    const [cx,cy] = queue.shift()!;

    for (const [ dx,dy] of dirs) {

      const nx = cx + dx, ny = cy + dy;

      if (nx>=0 && nx<width && ny>=0 && ny<height && grid[ny][nx] === '.' && !cuts[ny][nx] && !visited[ny][nx ]) {

        visited[ny][nx] = true;

        queue.push([nx,ny]);

      }

    }

  }

  let value = 0;

  for (const town of towns) {

    const {x,y,value: val} = town;

    if (visited[y][x]) value += val;

  }

  return value;

}
