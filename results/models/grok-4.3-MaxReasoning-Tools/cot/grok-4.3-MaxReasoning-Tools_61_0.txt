Then function dist(p1, p 2) {

return Math.hypot(p1.x-p2.x, p1.y-p2.y);

}

function closestToCurrent(current, remaining) {

let minD = Infinity;

let best = null;

for (const p of remaining) {

const d = dist(current, p);

if (d < minD && p) {

minD = d;

best = p;

}

}

return best;

}

function buildRoute(cluster, depot) {

if (cluster.length === 0) return [];

const route = [];

let current = depot;

const remaining = [...cluster];

while (remaining.length > 0) {

const next = closestToCurrent(current, remaining);

if (!next) break;

route.push(next.id);

current = next;

remaining = remaining.filter(p => p.id !== next.id); // since id may not unique? but assume unique

}

return route;

}
