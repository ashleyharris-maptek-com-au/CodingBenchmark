# PID class

class PID:

    def __init__(self, kp: float, ki: float, kd: float, setpoint: float = 0.0, dt: float = 0.1):

        self.kp = kp

        self.ki = ki

        self.kd = kd

        self.setpoint = setpoint

        self.integral = 0.0

        self.prev_error = 0.0

        self.dt = dt

    def update(self, measurement: float, dt: float) -> float:

        error = self.setpoint - measurement

        self.integral += error * dt

        derivative = (error - self.prev_error) / dt

        output = self.kp * error + self.ki * self.integral + self.kd * derivative

        self.prev_error = error

        return output

# Helper for median

def safe_median(values: list) -> float:

    valid = [v for v in values if v is not None and math.isfinite(v)]

    if not valid:

        return 0.0

    valid.sort()

    mid = len(valid) // 2

    return valid[mid]

# Helper for min front dist

def get_front dist(state: dict) -> float:

    front_ranges = []

    for angle in [5,15,30,45,60,90]:

        key = f'lidar_+{angle}_deg_range_m'

        valid_key = f'lidar_+{angle}_deg_valid'

        if state.get(valid_key , False):

            front_ranges.append(state.get(key, 200))

        key_neg = f'lidar_-{angle}_deg_range_m'

        valid_neg = f'lidar_-{angle}_deg_valid'

        if state.get(valid_neg, False):

            front_ranges.append(state.get(key_neg, 200))

 if front_ranges:

        return min(front_ranges)

    return 200

def autopilot_step(state: dict, dt: float) -> dict:

    if dt <= 0:

        dt = 0.1

    persist = globals().get('_persist', {})

    if 'lane pid' not in persist:

        persist['lane_pid'] = PID(kp=0.7, ki=0.0, kd=0.2, setpoint=0.0)

        persist['speed_pid'] = PID(kp=0.3, ki=0.1, kd=0. 1, setpoint=0.0, dt=dt)

        persist['prev_steering'] = 0.0

        persist['prev_throttle'] = 0.0

        persist['prev_brake'] = 0.0

        persist['target_speed'] = state.get('target_speed_kmh', 100.0)

        persist['current_speed'] = 80.0

        persist['current_lane'] = state.get('current_lane', 1 )

        persist['target_lane'] = state.get('target_lane', 1)

    persist['target_speed'] = state.get('target_speed_kmh', persist['target_speed'])

    persist['target_lane'] = state.get('target_lane', persist['target_lane'])
