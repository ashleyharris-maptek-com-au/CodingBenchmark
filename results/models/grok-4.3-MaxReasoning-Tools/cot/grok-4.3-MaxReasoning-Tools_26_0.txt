For argmin_dist:

fn argmin_dist(p: &[f32;3], cents: &[ [f32 ;3] ]) -> usize {

    let mut min_d = f32::MAX;

    let mut idx = 0;

    for (i, c) in cents.iter().enumerate() {

        let d = (p[0]-c[0]).powi(2) + (p[1]- c[1]).powi(2) + (p[2]-c[2]).powi(2);

        if d < min_d {

            min_d = d;

            idx = i;

        }

    }

    idx

}

fn init_centroids(points: & Vec<[f32;3], k: usize) -> Vec<[f32;3]> {

    let n = points.len();

    let k = k.min(n);

    let mut cents = vec![[0.0;3];k];

    for i in 0..k {

        let step = n / k ;

        cents[i] = points[i * step];

    }

    cents

}
