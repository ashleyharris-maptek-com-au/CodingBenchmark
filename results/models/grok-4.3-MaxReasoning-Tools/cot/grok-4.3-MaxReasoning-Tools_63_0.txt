function shortestDist(grid: string[], sx: number, sy: number, ex: number, ey: number): number {

  const rows = grid.length;

  const cols = grid[ 0 ].length;

  if (sx === ex && sy === ey) return 0;

  const visited = Array(rows).fill(0).map(() => Array(cols).fill(false));

  const queue: [number, number, number][] = [[sx, sy, 0]]; // x,y, dist

  visited[sy][sx] = true;

  const dirs = [[0,1],[1,0],[0,-1],[-1,0]];

  while (queue.length > 0) {

    const [x, y, dist] = queue.shift()!;

    for (const [dx, dy] of dirs) {

      const nx = x + dx;

      const ny = y + dy;

      if (nx >=0 && nx < cols && ny >=0 && ny < rows && !visited[ny][nx] && grid[ny][nx] !== '#') {

        if (nx === ex && ny === ey) return dist + 1;

        visited[ny][nx] = true;

        queue.push([nx, ny, dist + 1]);

      }

    }

  }

  return -1; // or throw, but assume not

}
