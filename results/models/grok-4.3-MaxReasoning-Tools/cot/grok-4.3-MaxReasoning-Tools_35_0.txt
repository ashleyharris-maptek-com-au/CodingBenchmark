fn greedy(n: usize, adj: &Vec<Vec<usize>>) -> Vec<usize> {

    let mut is_dominated = vec![false; n];

    let mut selected = Vec::new();

    while selected.len() < n { // safety

        let mut best_v = None;

        let mut best_cover = 0;

        for v in 0..n {

            let mut cover = 0 as usize;

            if !is_dominated[v] {

                cover += 1;

            }

            for &nei in &adj[v] {

                if !is_dominated[nei] {

                    cover += 1;

                }

            }

            if cover > best_cover {

                best_cover = cover;

                best_v = Some(v);

            }

        }

        if let Some(v) = best_v {

            if best_cover == 0 {

                break;

            }

            selected.push(v);

            // mark dominated

            if !is_dominated[v] {

                is_dominated[v] = true;

            }

            for &nei in &adj[v] {

                if !is_dominated[nei] {

                    is_dominated[nei] = true;

                }

            }

        } else {

            break;

        }

    }

    selected

}

fn greedy(n: usize, adj: &Vec<Vec<usize>>) -> Vec<usize> {

    let mut current_cover = vec![0; n];

    for v in 0..n {

        current_cover[v] = adj[v].len() + 1; // self + neighbors

    }

    let mut is_dominated = vec![false; n];

    let mut selected = Vec::new();

    for _ in 0..n {

        let mut best_v = 0;

        let mut best_cover = 0;

        for v in 0..n {

            if current_cover[v] > best_cover {

                best_cover = current_cover[v];

                best_v = v;

            }

        }

        if best_cover <= 0 {

            break;

        }

        selected.push(best_v);

        // now find new dominated

        let mut new_list = Vec::new();

        if !is_dominated[best_v] {

            new_list.push(best_v);

        }

        for &u in &adj[best_v] {

            if !is_dominated[u] {

                new_list.push(u);

            }

        }

        // mark

        for &u in &new_list {

            is_dominated[u] = true;

        }

        // update covers

        for &u in &new_list {

            let mut doms = vec![u];

            for &nei in &adj[u] {

                doms.push(nei);

            }

            doms.sort();

            doms.dedup(); // optional

            for &w in &doms {

                if current_cover[w] > 0 {

                    current_cover[w] -= 1;

                }

            }

        }

    }

    selected

}
