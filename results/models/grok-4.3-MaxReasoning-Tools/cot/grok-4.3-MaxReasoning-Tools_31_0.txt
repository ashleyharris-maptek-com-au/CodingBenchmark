fn get_prev_map(arr: &[i64], base: usize) -> HashMap<i64, (i64, usize)> {

    let mut prev_map = HashMap::new();

    prev_map.insert(0, (0, usize ::MAX));

    let mut reachable = HashSet::new();

    reachable.insert(0i64);

    for (i, &val) in arr.iter().enumerate() {

        let idx = base + i;

        let mut new_sums = Vec::new();

        for &s in &reachable {

            let ns = s + val;

             if !reachable.contains(&ns) {

                new_sums.push(ns);

                prev_map.insert(ns, (s, idx));

            }

        }

        for ns in new_sums {

            reachable.insert(ns);

        }

    }

    prev_map

}

Then, fn reconstruct (prev_map: &HashMap<i64, (i64, usize)>, target: i64) -> Vec<usize> {

    let mut indices = Vec::new();

    let mut curr = target;

    while curr != 0 {

        if let Some(&(p, idx)) = prev_map.get(& curr) {

            if idx != usize::MAX {

                indices.push(idx);

            }

            curr = p;

        } else {

            return vec![]; // error

        }

    }

    indices.sort_unstable();

    indices

}
