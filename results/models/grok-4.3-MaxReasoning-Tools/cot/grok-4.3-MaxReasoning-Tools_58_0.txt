const H = grid.length;

const W = grid[0].length;

const dirs = [[-1,0],[1,0],[0,-1],[0,1]] // dy,dx for y,x

function getPos(point) { return {x: point.x, y: point.y} }

Then, to create points:

const robotPoints = robots.map(r => ({x: r.x, y: r.y, id: r.id, type: 'robot'}));

const jobPoints = jobs.map(j => ({x: j.x, y: j.y, id: j.id, type: 'job', priority: j.priority}));

const allPoints = [...robotPoints, ...jobPoints];

const N = allPoints.length; // R + J

const pointDist = Array.from({ length: N }, () => Array(N).fill(-1));

 // wait, better compute on fly? No, precompute.

for(let s=0; s<N; s++){

  const {x: sx, y: sy} = allPoints[s];

  const distGrid = Array.from({length: H}, () => Array(W).fill(-1));

  const queue = [];

  distGrid[sy][sx] = 0;

  queue.push([sy, sx]);

  while(queue.length){

    const [y,x] = queue.shift();

    for(let [dy,dx] of dirs){

      const ny = y+dy, nx = x+dx;

      if(ny>=0 && ny<H && nx>=0 && nx<W && grid[ny][nx] === '.' && distGrid[ny][nx] === -1){

        distGrid[ny][nx] = distGrid[y][x] +1;

        queue.push([ny, nx]);

      }

    }

  }

  // now for each target t=0 to N-1

  for(let t=0; t<N; t++){

    const {y: ty, x: tx} = allPoints[t];

    pointDist[s ][t] = distGrid[ty][tx];

  }

}
