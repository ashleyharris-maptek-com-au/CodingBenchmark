using System;

using System.IO;

using System.Collections.Generic;

using System.Linq;

class Program {

    static void Main() {

 // Read N mu

  string[] lines = File.ReadAllLines(Console.OpenStandardInput()); wait, better use Console.ReadLine

  string line = Console.ReadLine();

  var tokens = line.Split(new char[] {' '}, StringSplitOptions.RemoveEmptyEntries);

  int N = int.Parse(tokens[0]);

 double mu = double.Parse(tokens[1]); // not used

  // Read start

  line = Console.ReadLine();

  tokens = line.Split(new char[] {' '}, StringSplitOptions.RemoveEmptyEntries);

  double[] start_pos = new double[3];

  double[] start_vel = new double[3 ];

  for(int k=0; k<3; k++) start_pos[k] = double.Parse(tokens[k]);

  for(int k=0; k<3; k++) start_vel[k] = double.Parse(tokens[3+k]);

  // Read stations

  List<double[]> station_pos = new List<double []>();

  List<double[]> station_vel = new List<double []>();

  for(int i=0; i<N; i++) {

    line = Console.ReadLine();

    tokens = line.Split(new char[] {' '}, StringSplitOptions.RemoveEmptyEntries);

    double[] p = new double[3];

    double [] v = new double[3];

    for(int k=0; k<3; k++) p[k] = double.Parse(tokens[k]);

    for(int k=0; k<3; k++) v[k] = double.Parse(tokens[3+k]);

    station_pos.Add(p);

    station_vel.Add (v);

  }

  // Now, to find order

  // Precompute distances from start to all stations

  double[,] dist = new double[N, N];

  for(int i=0; i<N; i++) {

    dist[i,i] = 0;

    for(int j=0; j<N; j++) if(i!=j) {

      dist[i,j] = Distance(station_pos[i], station_pos[j]);

    }

  }

  // To find best path

  // For simplicity, use NN with best first

  List<int> best_order = null;

  double best_cost = double .MaxValue;

  for(int first =0; first<N; first++) {

    // Run NN from first

    List<int> order = new List<int>();

    bool[] vis = new bool[N];

    int curr = first;

    order.Add(curr);

    vis[curr] = true;

    double cost = Distance(start_pos, station_pos[first]);

    for(int step=1; step<N; step++) {

      double min_d = double.MaxValue;

      int next = -1;

      for(int i=0; i<N; i++) if(!vis[i]) {

        double d = dist[curr, i ];

        if(d < min_d) {

          min_d = d;

          next = i;

        }

      }

      order.Add(next);

      vis[next] = true;

      cost += min_d;

      curr = next;

    }

    if(cost < best_cost) {

      best_cost = cost;

      best_order = new List<int>(order);

    }

  }

  // Output

  Console.WriteLine(string.Join(" ", best_order));

  }

  static double Distance(double[] a, double[] b) {

    double dx = a[0]-b[0];

    double dy = a[1]-b [1];

    double dz = a[2]-b[2];

    return Math.Sqrt(dx*dx + dy*dy + dz*dz);

  }

}
