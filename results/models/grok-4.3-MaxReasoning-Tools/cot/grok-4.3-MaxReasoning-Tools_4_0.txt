vector<Point> get_vertices(const Point& center, double rx, double ry, double rz, double e) {

  double s = e / (2.0 * sqrt(2.0));

  vector<Point> base = {

    Point(s, s, s),

    Point(s, -s, -s),

    Point(-s, s, -s),

    Point(-s, -s, s)

  };

  vector<Point> verts;

  for(auto p : base) {

    // apply rotations X then Y then Z

    // X

    double cy = cos(rx), sy = sin(rx);

    double y = p.y * cy - p.z * sy;

    double z = p.y * sy + p.z * cy;

    p.y = y;

    p.z = z;

    // Y

    double cy2 = cos(ry), sy2 = sin(ry);

    double x = p.x * cy2 + p.z * sy2;

    z = -p.x * sy2 + p.z * cy2;

    p.x = x;

    p.z = z;

    // Z

    double cy3 = cos(rz), sy3 = sin(rz);

    x = p.x * cy3 - p.y * sy3;

    y = p.x * sy3 + p.y * cy3;

    p.x = x;

    p.y = y;

    verts.push_back (center + p);

  }

  return verts;

}
