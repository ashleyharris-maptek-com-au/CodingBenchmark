OpCapability Shader

OpMemoryModel Logical GLSL450

OpEntryPoint Fragment %main "main" %in_normal %out_color

OpDecorate %out_color Location 0

OpDecorate %in_normal Location 1

; Types

 %void = OpTypeVoid

  %float = OpTypeFloat 32

  %v3 = OpTypeVector %float 3

  %v4 = OpTypeVector %float 4

  %ptr_in = OpTypePointer Input %v3

  %ptr_out = OpTypePointer Output %v4

  %func = OpTypeFunction %void

; Variables

 %in_normal = OpVariable %ptr_in Input

 %out_color = OpVariable %ptr_out Output

; Constants

 %one = OpConstant %float 1.0

 %half = OpConstant %float 0 5

 %v3_one = OpConstantComposite %v3 %one %one %one

 %v3_half = OpConstantComposite %v3 %half %half %half

; Function

 %main = OpFunction %void None %func

 %label = OpLabel

 %normal = OpLoad % v3 %in_normal

 %add = OpFAdd %v3 %normal %v3_one

 %mul = OpFMul %v3 %add %v3_half

 %r = OpCompositeExtract %float %mul 0

 %g = OpCompositeExtract %float %mul 1

 %b = OpCompositeExtract %float %mul 2

 %alpha = OpCopyObject %float %one   # or OpConstant

 %outv = OpCompositeConstruct %v4 %r %g %b %alpha

 OpStore %out_color %outv

 OpReturn

 OpFunctionEnd
