use std::cmp::Ordering;
use std::io::{self, Read};

struct GraphBisection {
    n: usize,
    half: usize,
    adj: Vec<Vec<usize>>,
    assign: Vec<i8>,     // -1 unassigned, 0 -> A, 1 -> B
    best_cut: i32,
    best_assign: Vec<i8>,
}

impl GraphBisection {
    fn new(n: usize, adj: Vec<Vec<usize>>) -> Self {
        let mut assign = vec![-1i8; n];
        // Symmetry breaking: put vertex 0 into A
        if n > 0 {
            assign[0] = 0;
        }
        GraphBisection {
            n,
            half: n / 2,
            adj,
            assign,
            best_cut: i32::MAX / 4,
            best_assign: vec![-1i8; n],
        }
    }

    // Heuristic initial solution to get a good upper bound
    fn initial_greedy(&mut self) {
        if self.n == 0 {
            return;
        }
        // Start from current assignment, try to greedily fill others
        let mut cur = self.assign.clone();
        let mut ca: usize = if cur[0] == 0 { 1 } else { 0 };
        let mut cb: usize = 0;
        if cur[0] == 1 {
            cb = 1;
        }

        // Order vertices by decreasing degree (excluding 0)
        let mut order: Vec<usize> = (0..self.n).collect();
        order.sort_by(|&a, &b| {
            let da = self.adj[a].len();
            let db = self.adj[b].len();
            // descending degree
            db.cmp(&da)
        });

        for &u in &order {
            if u == 0 { continue; }
            // Skip if already assigned (shouldn't be)
            if cur[u] != -1 { continue; }

            // Try both sides if feasible, pick the one with smaller incremental cut
            let mut best_side: i8 = -1;
            let mut best_inc: i32 = i32::MAX;

            // Determine feasible sides
            for s in 0..=1 {
                if s == 0 && ca == self.half { continue; }
                if s == 1 && cb == self.half { continue; }

                let mut a_nei = 0usize; // edges to B (for placing in A)
                let mut b_nei = 0usize; // edges to A (for placing in B)
                for &v in &self.adj[u] {
                    if cur[v] == 1 { a_nei += 1; }
                    else if cur[v] == 0 { b_nei += 1; }
                }
                let inc = if s == 0 { a_nei as i32 } else { b_nei as i32 };
                if inc < best_inc {
                    best_inc = inc;
                    best_side = s;
                }
            }

            if best_side != -1 {
                cur[u] = best_side;
                if best_side == 0 { ca += 1; } else { cb += 1; }
            }
        }

        // Compute final cut for this partition
        if ca == self.half && cb == self.half {
            let mut cut: i32 = 0;
            for u in 0..self.n {
                for &v in &self.adj[u] {
                    if v > u && cur[u] != cur[v] {
                        cut += 1;
                    }
                }
            }
            if cut < self.best_cut {
                self.best_cut = cut;
                self.best_assign = cur.clone();
            }
        }

        // If we didn't get a valid complete partition, we still keep best_cut as is.
    }

    // Entry point to start exact search
    fn search(&mut self) {
        // Start from current state (0 is in A)
        // Ensure best_assign is initialized with current assignment if not yet set
        self.dfs(1, 0, 0);
        if self.best_assign.iter().all(|&x| x == -1) {
            // Fall back to current assignment if nothing found
            self.best_assign = self.assign.clone();
        }
    }

    // Depth-first search with branch-and-bound and LB
    fn dfs(&mut self, count_a: usize, count_b: usize, cut_fixed: i32) {
        // Prune impossible counts
        if count_a > self.half || count_b > self.half {
            return;
        }

        // Lower bound based on assigned/unassigned optimizations
        if let Some(lb) = self.lb(count_a, count_b, cut_fixed) {
            if lb > self.best_cut {
                return;
            }
        } else {
            // infeasible due to counts
            return;
        }

        // Check if all vertices are assigned
        if self.assign.iter().all(|&x| x != -1) {
            if count_a == self.half && count_b == self.half {
                // Final cut is exactly cut_fixed
                if cut_fixed < self.best_cut {
                    self.best_cut = cut_fixed;
                    self.best_assign = self.assign.clone();
                }
            }
            return;
        }

        // Pick a vertex to branch on: choose one with most assigned neighbors
        let mut u_opt: Option<usize> = None;
        let mut best_deg_assigned: usize = 0;
        for i in 0..self.n {
            if self.assign[i] == -1 {
                let mut deg_ass = 0usize;
                for &nb in &self.adj[i] {
                    if self.assign[nb] != -1 {
                        deg_ass += 1;
                    }
                }
                if deg_ass > best_deg_assigned {
                    best_deg_assigned = deg_ass;
                    u_opt = Some(i);
                }
            }
        }
        let u = match u_opt {
            Some(x) => x,
            None => return,
        };

        // Compute intersection counts with assigned neighbors for branching order
        let mut a_nei = 0usize;
        let mut b_nei = 0usize;
        for &nb in &self.adj[u] {
            match self.assign[nb] {
                1 => a_nei += 1,
                0 => b_nei += 1,
                _ => {}
            }
        }

        // Build two options (A or B) with their incremental cuts
        let mut opts: Vec<(i8, i32)> = Vec::new();
        if count_a < self.half {
            opts.push((0i8, a_nei as i32)); // placing u in A adds a_nei to cut
        }
        if count_b < self.half {
            opts.push((1i8, b_nei as i32)); // placing u in B adds b_nei to cut
        }
        // Order by smaller incremental cut first
        opts.sort_by(|a, b| a.1.cmp(&b.1));

        for (side, _inc) in opts {
            // Apply assignment
            self.assign[u] = side;
            if side == 0 {
                let new_cut = cut_fixed + a_nei as i32;
                self.dfs(count_a + 1, count_b, new_cut);
            } else {
                let new_cut = cut_fixed + b_nei as i32;
                self.dfs(count_a, count_b + 1, new_cut);
            }
            self.assign[u] = -1;
        }
    }

    // Compute a lower bound for the final cut given current partial assignment
    // Returns None if counts infeasible, else Some(lb)
    fn lb(&self, count_a: usize, count_b: usize, cut_fixed: i32) -> Option<i32> {
        // Gather for unassigned vertices:
        // For each unassigned i:
        //   a_i = number of neighbors assigned to B (1)
        //   b_i = number of neighbors assigned to A (0)
        // Base cost = sum b_i
        // delta_i = a_i - b_i
        // We must assign exactly kA_remain of the unassigned to A
        // minimal extra cost from unassigned-to-assigned edges is
        // base + sum of smallest kA_remain deltas.
        let mut unassigned = 0usize;
        let mut base_cost: i32 = 0;
        let mut deltas: Vec<i32> = Vec::new();

        for i in 0..self.n {
            if self.assign[i] == -1 {
                let mut a_i = 0usize;
                let mut b_i = 0usize;
                for &nb in &self.adj[i] {
                    match self.assign[nb] {
                        1 => a_i += 1,
                        0 => b_i += 1,
                        _ => {}
                    }
                }
                base_cost += b_i as i32;
                deltas.push(a_i as i32 - b_i as i32);
                unassigned += 1;
            }
        }

        // Feasibility: remaining A to place
        let k_a_remain = self.half as isize - count_a as isize;
        if k_a_remain < 0 || k_a_remain > unassigned as isize {
            return None;
        }

        deltas.sort_by(|x, y| x.cmp(y)); // ascending
        let k = k_a_remain as usize;
        let mut sum_k: i32 = 0;
        for i in 0..k {
            sum_k += deltas[i];
        }

        let lb_unassigned = base_cost + sum_k;
        Some(cut_fixed + lb_unassigned)
    }
}

fn main() {
    // Read all input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
    }

    let mut solver = GraphBisection::new(n, adj);
    // Initial greedy solution to get a reasonable upper bound
    solver.initial_greedy();
    // Run exact search
    solver.search();

    // Output
    println!("{}", solver.best_cut);

    // Collect A-partition vertices from best_assign if available, else from current assign
    let mut a_list: Vec<usize> = Vec::new();
    let assign_to_use = if solver.best_assign.iter().any(|&x| x != -1) {
        &solver.best_assign
    } else {
        &solver.assign
    };
    for i in 0..n {
        if assign_to_use[i] == 0 {
            a_list.push(i);
        }
    }

    // Ensure exactly n/2 vertices in A
    // If not enough (due to some unforeseen issue), fill with remaining indices.
    if a_list.len() != n / 2 {
        let mut seen = vec![false; n];
        for &x in &a_list { seen[x] = true; }
        for i in 0..n {
            if a_list.len() == n / 2 { break; }
            if !seen[i] {
                a_list.push(i);
            }
        }
        a_list.truncate(n / 2);
    }

    // Print second line
    let line2: String = a_list.iter().map(|x| x.to_string()).collect::<Vec<_>>().join(" ");
    println!("{}", line2);
}
