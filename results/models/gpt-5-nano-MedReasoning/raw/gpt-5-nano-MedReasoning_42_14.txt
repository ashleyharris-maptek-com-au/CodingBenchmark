; SPIR-V Geometry Shader
; Fixed vertex/point/triangle strip output with per-face lighting (flat shading)
; Input primitive: InputTrianglesAdjacency (6 verts per primitive)
; Output primitive: OutputTriangleStrip, 3 vertices
; Invocations: 1
; Emit three vertices from gl_in indices 0, 2, 4
; Flat color output (outColor) to fragment shader, via flat-interpolated output

OpCapability Geometry
OpMemoryModel Logical GLSL450

%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint  = OpTypeInt 32 0

; Constants
%uint_6 = OpConstant %uint 6
%uint_0 = OpConstant %uint 0
%uint_2 = OpConstant %uint 2
%uint_4 = OpConstant %uint 4
%float_0  = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_inv3 = OpConstant %float 0.3333333333
%float_inv_sqrt3_x = OpConstant %float 0.57735026919
%float_0_577 = OpConstant %float 0.57735026919

; gl_PerVertex_in (vertex input from vertex shader)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; gl_PerVertex_out (per-vertex output to fragment shader / next stage)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output per-vertex color
%ptr_out_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3float Output
OpDecorate %outColor Location 0
OpDecorate %outColor Flat

; Inputs from Vertex Shader (per-vertex arrays)
%arr_worldPos_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_worldPos = OpTypePointer Input %arr_worldPos_in
%worldPos = OpVariable %ptr_in_arr_worldPos Input

%arr_normal_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_normal = OpTypePointer Input %arr_normal_in
%normal = OpVariable %ptr_in_arr_normal Input

%arr_color_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_color = OpTypePointer Input %arr_color_in
%color = OpVariable %ptr_in_arr_color Input

; Minimal UBO (set=0, binding=0) to satisfy interface requirement (layout not used here)
%mat4 = OpTypeMatrix %v4float 4
%ubo_struct = OpTypeStruct %mat4 %mat4
%ptr_ubo_uniform = OpTypePointer Uniform %ubo_struct
%ubo = OpVariable %ptr_ubo_uniform Uniform
OpDecorate %ubo Block

; Pointers/couples for later use
%ptr_gl_per_vertex_in = OpTypePointer Input %gl_PerVertex_in
%ptr_v4float_in = OpTypePointer Input %v4float
%ptr_v3float_in = OpTypePointer Input %v3float
%ptr_v3float_out = OpTypePointer Output %v3float
%ptr_v4float_out = OpTypePointer Output %v4float

; Fixed indices
%i0 = OpConstant %uint 0
%i2 = OpConstant %uint 2
%i4 = OpConstant %uint 4
%i0m = OpConstant %uint 0

; Entry point
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Helper: load gl_in[i].gl_Position (vec4)
; i0
%gl_in_i0_ptr = OpAccessChain %ptr_gl_per_vertex_in %gl_in %i0
%gl_in_i0_pos_ptr = OpAccessChain %ptr_v4float_in %gl_in_i0_ptr %i0m
%p0 = OpLoad %v4float %gl_in_i0_pos_ptr
; i2
%gl_in_i2_ptr = OpAccessChain %ptr_gl_per_vertex_in %gl_in %i2
%gl_in_i2_pos_ptr = OpAccessChain %ptr_v4float_in %gl_in_i2_ptr %i0m
%p2 = OpLoad %v4float %gl_in_i2_pos_ptr
; i4
%gl_in_i4_ptr = OpAccessChain %ptr_gl_per_vertex_in %gl_in %i4
%gl_in_i4_pos_ptr = OpAccessChain %ptr_v4float_in %gl_in_i4_ptr %i0m
%p4 = OpLoad %v4float %gl_in_i4_pos_ptr

; Convert to vec3 (xyz)
%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p0z = OpCompositeExtract %float %p0 2
%p0_xyz = OpCompositeConstruct %v3float %p0x %p0y %p0z

%p2x = OpCompositeExtract %float %p2 0
%p2y = OpCompositeExtract %float %p2 1
%p2z = OpCompositeExtract %float %p2 2
%p2_xyz = OpCompositeConstruct %v3float %p2x %p2y %p2z

%p4x = OpCompositeExtract %float %p4 0
%p4y = OpCompositeExtract %float %p4 1
%p4z = OpCompositeExtract %float %p4 2
%p4_xyz = OpCompositeConstruct %v3float %p4x %p4y %p4z

; a = p2 - p0
%ax = OpFSub %float %p2x %p0x
%ay = OpFSub %float %p2y %p0y
%az = OpFSub %float %p2z %p0z
%a = OpCompositeConstruct %v3float %ax %ay %az

; b = p4 - p0
%bx = OpFSub %float %p4x %p0x
%by = OpFSub %float %p4y %p0y
%bz = OpFSub %float %p4z %p0z
%b = OpCompositeConstruct %v3float %bx %by %bz

; Cross product a x b
%ay2 = OpCompositeExtract %float %a 1
%az2 = OpCompositeExtract %float %a 2
%ax2 = OpCompositeExtract %float %a 0
%by2 = OpCompositeExtract %float %b 1
%bz2 = OpCompositeExtract %float %b 2
* Cross components
%term1 = OpFMul %float %ay2 %bz2
%term2 = OpFMul %float %az2 %by2
%cx = OpFSub %float %term1 % %term2

%term3 = OpFMul %float %az2 %bx
%term4 = OpFMul %float %ax2 %bz2
%cy = OpFSub %float %term3 % %term4

%term5 = OpFMul %float %ax2 %by2
%term6 = OpFMul %float %ay2 %bx
%cz = OpFSub %float %term5 % %term6

%cross = OpCompositeConstruct %v3float %cx %cy %cz

; Normalize N
%N = OpExtInst %v3float cross GLSLstd450 69

; L = normalize(vec3(1,1,1)) -> use 0.57735026919
%L = OpCompositeConstruct %v3float %float_inv_sqrt3 %float_inv_sqrt3 %float_inv_sqrt3

; Dot = N · L
%Nx = OpCompositeExtract %float %N 0
%Ny = OpCompositeExtract %float %N 1
%Nz = OpCompositeExtract %float %N 2
%Lx = OpCompositeExtract %float %L 0
%Ly = OpCompositeExtract %float %L 1
%Lz = OpCompositeExtract %float %L 2
%t1 = OpFMul %float %Nx %Lx
%t2 = OpFMul %float %Ny %Ly
%t3 = OpFMul %float %Nz %Lz
%d12 = OpFAdd %float %t1 %t2
%dot = OpFAdd %float %d12 %t3

; diffuse = max(dot, 0.0) using GLSL std 450 FMax(40)
%zero = OpConstant %float 0.0
%diffuse = OpExtInst %float %dot GLSLstd450 40 %dot %zero

; scale = 0.2 + 0.8 * diffuse
%mul2 = OpFMul %float %diffuse %float_0_8
%scale = OpFAdd %float %mul2 %float_0_2

; Colors: color[0], color[2], color[4]
%col0_ptr = OpAccessChain %ptr_in_arr_color %color %%idx0
%col0 = OpLoad %v3float %col0_ptr
%col2_ptr = OpAccessChain %ptr_in_arr_color %color %i2
%col2 = OpLoad %v3float %col2_ptr
%col4_ptr = OpAccessChain %ptr_in_arr_color %color %i4
%col4 = OpLoad %v3float %col4_ptr

; baseColor = (col0 + col2 + col4) / 3
%c0x = OpCompositeExtract %float %col0 0
%c0y = OpCompositeExtract %float %col0 1
%c0z = OpCompositeExtract %float %col0 2

%c2x = OpCompositeExtract %float %col2 0
%c2y = OpCompositeExtract %float %col2 1
%c2z = OpCompositeExtract %float %col2 2

%c4x = OpCompositeExtract %float %col4 0
%c4y = OpCompositeExtract %float %col4 1
%c4z = OpCompositeExtract %float %col4 2

%sx = OpFAdd %float %c0x %c2x
$s1 = OpFAdd %float %sx %c4x
$base_x = OpFMul %float %s1 %%float_inv3  ; divide by 3

$sy = OpFAdd %float %c0y %c2y
$s2 = OpFAdd %float %sy %c4y
$base_y = OpFMul %float %s2 %%float_inv3

$sz = OpFAdd %float %c0z %c2z
$s3 = OpFAdd %float %sz %c4z
$base_z = OpFMul %float %s3 %%float_inv3

;baseColor vec3
%baseColor = OpCompositeConstruct %v3float %base_x %base_y %base_z

; finalColor = baseColor * scale
%fcx = OpCompositeExtract %float %baseColor 0
%fcy = OpCompositeExtract %float %baseColor 1
%fcz = OpCompositeExtract %float %baseColor 2
%fout_x = OpFMul %float %fcx %scale
%fout_y = OpFMul %float %fcy %scale
%fout_z = OpFMul %float %fcz %scale
%finalColor = OpCompositeConstruct %v3float %fout_x %fout_y %fout_z

; Emit 3 vertices (0, 2, 4) with finalColor
; Vertex 0
OpStore %outColor %finalColor
; gl_Position for vertex 0
%gl_out_pos0_ptr = OpAccessChain %ptr_v4float_out %gl_out %i0
OpStore %gl_out_pos0_ptr %p0
OpEmitVertex

; Vertex 2
OpStore %outColor %finalColor
%gl_out_pos2_ptr = OpAccessChain %ptr_v4float_out %gl_out %i2
OpStore %gl_out_pos2_ptr %p2
OpEmitVertex

; Vertex 4
OpStore %outColor %finalColor
%gl_out_pos4_ptr = OpAccessChain %ptr_v4float_out %gl_out %i4
OpStore %gl_out_pos4_ptr %p4
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd

; Execution mode and entry point
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Entry point
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %outColor

; Function prototype
%void_fn = OpFunction %void None %void_fn
%void_fn = OpFunction %void None %void_fn
; (The function body above ends with OpFunctionEnd)
