; SPIR-V geometry shader implementing 8 radial lines expanded into tapered ribbons
; Vertex input primitive: InputLines (2 vertices)
; Output primitive: OutputTriangleStrip with 4 vertices
; Thickness: start 0.04 at p0, end 0.005 at p1
; Colors: color[0] at p0 vertices, color[1] at p1 vertices

; SPIR-V header
OpCapability Geometry
OpMemoryModel Logical GLSL450
%GLSL = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%func_t = OpTypeFunction %void

%float = OpTypeFloat 32
$v2f = OpTypeVector %float 2
$v3f = OpTypeVector %float 3
$v4f = OpTypeVector %float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; Constants
%const_0 = OpConstant %int 0
%const_1 = OpConstant %int 1
%const_2 = OpConstant %int 2

%float_0   = OpConstant %float 0.0
%float_04  = OpConstant %float 0.04
%float_005 = OpConstant %float 0.005
%one = OpConstant %float 1.0

; gl_PerVertex_in (input from vertex shader)
%gl_PerVertex_in = OpTypeStruct %v4f
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; User-defined inputs from vertex shader
%arr_vec3 = OpTypeArray %v3f %uint_2
%ptr_in_vec3 = OpTypePointer Input %arr_vec3
%in_worldPos = OpVariable %ptr_in_vec3 Input
%in_color    = OpVariable %ptr_in_vec3 Input

; gl_PerVertex_out (output to fragment stage)
%gl_PerVertex_out = OpTypeStruct %v4f
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color per-vertex for fragment shader
%outColor_type = OpTypeVector %float 3
%ptr_out_color = OpTypePointer Output %outColor_type
%outColor = OpVariable %ptr_out_color Output

; Entry point interface
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_color %gl_out %outColor

; Execution modes
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Function declaration
%void_t = %void
%void = OpTypeVoid
%main_type = OpTypeFunction %void

; Function body
%main = OpFunction %void None %main_type
%entry = OpLabel

; Load endpoints from gl_in
%ptr_in_v4 = OpTypePointer Input %v4f
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %const_0 %const_0
%p0 = OpLoad %v4f %p0_ptr

%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %const_1 %const_0
%p1 = OpLoad %v4f %p1_ptr

; Decompose endpoints
%p0_x = OpCompositeExtract %float %p0 0
%p0_y = OpCompositeExtract %float %p0 1
%p0_z = OpCompositeExtract %float %p0 2
%p0_w = OpCompositeExtract %float %p0 3

%p1_x = OpCompositeExtract %float %p1 0
%p1_y = OpCompositeExtract %float %p1 1
%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3

; Direction in screen-space
%dx = OpFSub %float %p1_x %p0_x
%dy = OpFSub %float %p1_y %p0_y
; Perpendicular vector (normalized)
%neg_dy = OpFNegate %float %dy
%nx = %neg_dy
%ny = %dx

; Length (for normalization)
%dx2 = OpFMul %float %dx %dx
%dy2 = OpFMul %float %dy %dy
%sum = OpFAdd %float %dx2 %dy2
%len = OpExtInst %float %GLSL Sqrt %sum
%one_over_len = OpFDiv %float %one %len

; Perp components (unit length in XY plane)
%perp_x = OpFMul %float %nx %one_over_len
%perp_y = OpFMul %float %ny %one_over_len

; Start/end thickness
%start_x = OpFMul %float %perp_x %float_04
%start_y = OpFMul %float %perp_y %float_04
%end_x   = OpFMul %float %perp_x %float_005
%end_y   = OpFMul %float %perp_y %float_005

; p0 +/- thick
%p0_plus_x = OpFAdd %float %p0_x % %start_x
%p0_plus_y = OpFAdd %float %p0_y % %start_y
%p0_plus = OpCompositeConstruct %v4f %p0_plus_x %p0_plus_y %p0_z %p0_w

%p0_minus_x = OpFSub %float %p0_x % %start_x
%p0_minus_y = OpFSub %float %p0_y % %start_y
%p0_minus = OpCompositeConstruct %v4f %p0_minus_x %p0_minus_y %p0_z %p0_w

; p1 +/- thick
%p1_plus_x = OpFAdd %float %p1_x % %end_x
%p1_plus_y = OpFAdd %float %p1_y % %end_y
%p1_plus = OpCompositeConstruct %v4f %p1_plus_x %p1_plus_y %p1_z %p1_w

%p1_minus_x = OpFSub %float %p1_x % %end_x
%p1_minus_y = OpFSub %float %p1_y % %end_y
%p1_minus = OpCompositeConstruct %v4f %p1_minus_x %p1_minus_y %p1_z %p1_w

; Read per-vertex colors
%ptr_color_v3 = OpTypePointer Input %v3f
%col0_ptr = OpAccessChain %ptr_color_v3 %in_color %const_0
%col0 = OpLoad %v3f %col0_ptr

%col1_ptr = OpAccessChain %ptr_color_v3 %in_color %const_1
%col1 = OpLoad %v3f %col1_ptr

; Output position pointer
%out_pos_ptr = OpTypePointer Output %v4f
%gl_out_ptr_zero = OpAccessChain %out_pos_ptr %gl_out %const_0

; Emit vertices in order: p0_plus, p0_minus, p1_plus, p1_minus
OpStore %gl_out_ptr_zero %p0_plus
OpStore %outColor %col0
OpEmitVertex

OpStore %gl_out_ptr_zero %p0_minus
OpStore %outColor %col0
OpEmitVertex

OpStore %gl_out_ptr_zero %p1_plus
OpStore %outColor %col1
OpEmitVertex

OpStore %gl_out_ptr_zero %p1_minus
OpStore %outColor %col1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd

This SPIR-V includes:
- OpCapability Geometry
- OpMemoryModel Logical GLSL450
- OpEntryPoint Geometry %main "main" with interface variables gl_in, worldPos_in, color_in, gl_out, outColor
- OpExecutionMode: InputLines, OutputTriangleStrip, OutputVertices 4, Invocations 1
- Input and output interface variables for per-vertex data
- Geometry shader main function that reads endpoints and per-vertex colors, computes a 2D perpendicular for a ribbon, expands to four vertices (p0+; p0-; p1+; p1-), assigns colors (col0 for start vertices, col1 for end vertices), and emits a 4-vertex triangle strip with 4 vertices.
