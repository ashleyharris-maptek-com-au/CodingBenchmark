```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    // Segment tree to support O(log n) placement decisions.
    // Stores the maximum remaining length in each segment.
    class SegmentTree
    {
        private readonly int n;
        private readonly int[] tree;

        public SegmentTree(int n)
        {
            this.n = n;
            tree = new int[n * 4];
        }

        public void Update(int pos, int value)
        {
            Update(1, 0, n - 1, pos, value);
        }

        private void Update(int node, int l, int r, int pos, int value)
        {
            if (l == r)
            {
                tree[node] = value;
                return;
            }
            int mid = (l + r) >> 1;
            if (pos <= mid) Update(node << 1, l, mid, pos, value);
            else Update((node << 1) | 1, mid + 1, r, pos, value);
            tree[node] = Math.Max(tree[node << 1], tree[(node << 1) | 1]);
        }

        // Find the leftmost bin index in [0, limit] whose remaining >= threshold.
        // Returns -1 if none exists.
        public int FindFirst(int threshold, int limit)
        {
            return FindFirst(1, 0, n - 1, threshold, limit);
        }

        private int FindFirst(int node, int l, int r, int threshold, int limit)
        {
            if (l > limit) return -1;
            if (tree[node] < threshold) return -1;
            if (l == r) return l;
            int mid = (l + r) >> 1;
            int left = FindFirst(node << 1, l, mid, threshold, limit);
            if (left != -1) return left;
            return FindFirst((node << 1) | 1, mid + 1, r, threshold, limit);
        }
    }

    struct Item
    {
        public int Length;
        public int Index;
    }

    static void Main()
    {
        // Read entire input and split by whitespace
        string all;
        using (var sr = new StreamReader(Console.OpenStandardInput()))
            all = sr.ReadToEnd();

        var parts = all.Split(new[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int pos = 0;
        if (parts.Length < 2)
            return;

        int n = int.Parse(parts[pos++]);
        int stockLength = int.Parse(parts[pos++]);

        int[] cuts = new int[Math.Max(0, n)];
        for (int i = 0; i < n; i++)
        {
            if (pos < parts.Length)
                cuts[i] = int.Parse(parts[pos++]);
            else
                cuts[i] = 0;
        }

        // Prepare items sorted by decreasing length (First Fit Decreasing heuristic)
        Item[] items = new Item[n];
        for (int i = 0; i < n; i++)
        {
            items[i] = new Item { Length = cuts[i], Index = i };
        }
        Array.Sort(items, (a, b) =>
        {
            int c = b.Length.CompareTo(a.Length);
            if (c != 0) return c;
            return a.Index.CompareTo(b.Index);
        });

        // Data structures for packing
        int[] remaining = new int[n]; // remaining capacity per bin (binCount bins used)
        var bins = new List<List<int>>(); // per-bin list of cut indices
        var seg = new SegmentTree(n);

        int binCount = 0;

        foreach (var it in items)
        {
            int L = it.Length;
            int posBin = seg.FindFirst(L, binCount - 1);
            if (posBin == -1)
            {
                // Create new bin
                var list = new List<int>();
                list.Add(it.Index);
                bins.Add(list);

                int rem = stockLength - L;
                remaining[binCount] = rem;
                seg.Update(binCount, rem);
                binCount++;
            }
            else
            {
                // Place into existing bin
                bins[posBin].Add(it.Index);
                remaining[posBin] -= L;
                seg.Update(posBin, remaining[posBin]);
            }
        }

        // Output
        var sb = new StringBuilder();
        sb.Append(binCount).AppendLine();
        for (int i = 0; i < binCount; i++)
        {
            if (bins[i].Count == 0)
            {
                sb.AppendLine();
            }
            else
            {
                sb.Append(string.Join(" ", bins[i]));
                sb.AppendLine();
            }
        }

        Console.Write(sb.ToString());
    }
}
```
