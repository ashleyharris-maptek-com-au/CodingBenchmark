; SPIR-V
; Version: 1.0
; Generator: Mandelbrot
; Bound: 999
; Schema: 0

OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%bool = OpTypeBool
%vec2 = OpTypeVector %float 2
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; Uniform block (DescriptorSet 0, Binding 0)
%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%_ptr_Uniform_Ubo = OpTypePointer Uniform %ubo_struct
%ubo = OpVariable %_ptr_Uniform_Ubo Uniform
OpDecorate %ubo Block
OpMemberDecorate %ubo_struct 0 Offset 0
OpMemberDecorate %ubo_struct 1 Offset 64
OpMemberDecorate %ubo_struct 2 Offset 128
OpMemberDecorate %ubo_struct 3 Offset 192
OpMemberDecorate %ubo_struct 4 Offset 208
OpMemberDecorate %ubo_struct 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Input/output
%_ptr_Input_Vec2 = OpTypePointer Input %vec2
%uv = OpVariable %_ptr_Input_Vec2 Input
OpDecorate %uv Location 2

%_ptr_Output_Vec4 = OpTypePointer Output %vec4
$outColor = OpVariable %_ptr_Output_Vec4 Output
OpDecorate %outColor Location 0

; Function types
%_func_void = OpTypeFunction %void
; Entry point function
%main = OpFunction %void None %_func_void
%_entry = OpLabel

; Mandelbrot parameters from UV
%_const_3 = OpConstant %float 3.0
%_const_2 = OpConstant %float 2.0
%_const_1 = OpConstant %float 1.0
%_const_m2 = OpConstant %float -2.0
%_zero = OpConstant %float 0.0
%_const_20 = OpConstant %float 20.0
%_const_4 = OpConstant %float 4.0
%_const_0_5 = OpConstant %float 0.5
%_one = OpConstant %float 1.0

; Load UV
%uvVal = OpLoad %vec2 %uv
%uvx = OpCompositeExtract %float %uvVal 0
%uvy = OpCompositeExtract %float %uvVal 1

; c = (uv.x*3 - 2, uv.y*2 - 1)
%tmpx = OpFMul %float %uvx %_const_3
%cr = OpFSub %float %tmpx %_const_m2
%tmpy = OpFMul %float %uvy %_const_2
%ci = OpFSub %float %tmpy %%_const_1

; Local loop variables
%_ptr_Function_float = OpTypePointer Function %float
%zr = OpVariable %_ptr_Function_float Function
%zi = OpVariable %_ptr_Function_float Function
%crVar = OpVariable %_ptr_Function_float Function
%ciVar = OpVariable %_ptr_Function_float Function
%iVar = OpVariable %_ptr_Function_float Function
%mag2 = OpVariable %_ptr_Function_float Function

; Initialize values
OpStore %zr 0.0
OpStore %zi 0.0
OpStore %crVar %cr
OpStore %ciVar %ci
OpStore %iVar 0.0
OpStore %mag2 0.0

; Loop header (evaluate condition, branch to Body or Merge)
%LoopLabel = OpLabel
OpLoopMerge %MergeLabel %BodyLabel None
; cond: i < 20.0  AND  mag2 <= 4.0
% iVal = OpLoad %float %iVar
% mag2Val = OpLoad %float %mag2
% iterOk = OpFOrdLessThan %bool %iVal %_const_20
% magEsc = OpFOrdGreaterThan %bool %mag2Val %_const_4
% notEsc = OpLogicalNot %bool %magEsc
% goOn = OpLogicalAnd %bool %iterOk %notEsc
OpBranchConditional %goOn %BodyLabel %MergeLabel

; Body: perform one Mandelbrot iteration
%BodyLabel = OpLabel
%zrVal = OpLoad %float %zr
%ziVal = OpLoad %float %zi
%crVal = OpLoad %float %crVar
%ciVal = OpLoad %float %ciVar

; compute z^2 + c
%zr2 = OpFMul %float %zrVal %zrVal
%zi2 = OpFMul %float %ziVal %ziVal
%temp = OpFSub %float %zr2 %zi2
%newzr = OpFAdd %float %temp %crVal

%ziZi = OpFMul %float %zrVal %ziVal
%twoZrZi = OpFMul %float 2.0 %ziZi
%newzi = OpFAdd %float %twoZrZi % ciVal

OpStore %zr %newzr
OpStore %zi %newzi

% iNext = OpFAdd %float %iVal 1.0
OpStore %iVar %iNext

% mag2Next = newzr*newzr + newzi*newzi
%mag2Next = OpFMul %float %newzr %newzr
%mag2tmp = OpFMul %float %newzi %newzi
%mag2sum = OpFAdd %float %mag2Next %mag2tmp
OpStore %mag2 %mag2sum

OpBranch %LoopLabel

; Merge: compute final color and write to output
%MergeLabel = OpLabel
%ifI = OpLoad %float %iVar
%t = OpFDiv %float %ifI %_const_20
%r = %t
%g = OpFMul %float %t %_const_0_5
%b = OpFSub %float %_one % %t
%color = OpCompositeConstruct %vec4 %r %g %b %_one
OpStore %outColor %color

OpReturn
OpFunctionEnd

; Entry point and execution mode
OpEntryPoint Fragment %main "main" %uv %outColor
OpExecutionMode %main OriginUpperLeft

; (UV and OUT are already decorated above)
