```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <numeric>
#include <limits>
#include <random>

struct City {
    int id;
    double x, y;
};

static inline double dist(const City& a, const City& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return std::sqrt(dx*dx + dy*dy);
}

// Exact TSP for very small N (N <= 15)
static std::vector<int> solveExact(const std::vector<City>& cities) {
    int N = (int)cities.size();
    if (N <= 1) return std::vector<int>{0};

    int n = N - 1;
    int full = (1 << n) - 1;
    const double INF = std::numeric_limits<double>::infinity();

    // dp[mask][j] = min distance starting at 0, visiting set 'mask' (cities 1..N-1),
    // ending at city (j+1)
    std::vector<std::vector<double>> dp(1 << n, std::vector<double>(n, INF));
    std::vector<std::vector<int>> par(1 << n, std::vector<int>(n, -1));

    // base
    for (int j = 0; j < n; ++j) {
        dp[1 << j][j] = dist(cities[0], cities[j + 1]);
    }

    // transitions
    for (int mask = 1; mask <= full; ++mask) {
        for (int j = 0; j < n; ++j) {
            if (!(mask & (1 << j))) continue;
            int prevMask = mask ^ (1 << j);
            if (prevMask == 0) continue;

            double best = INF;
            int bestI = -1;
            for (int i = 0; i < n; ++i) {
                if (!(prevMask & (1 << i))) continue;
                double val = dp[prevMask][i] + dist(cities[i + 1], cities[j + 1]);
                if (val < best) {
                    best = val;
                    bestI = i;
                }
            }
            dp[mask][j] = best;
            par[mask][j] = bestI;
        }
    }

    // close the tour
    double best = INF;
    int last = -1;
    for (int j = 0; j < n; ++j) {
        double val = dp[full][j] + dist(cities[j + 1], cities[0]);
        if (val < best) {
            best = val;
            last = j;
        }
    }

    // reconstruct path
    std::vector<int> seq;
    int mask = full;
    int cur = last;
    for (int t = n; t >= 1; --t) {
        seq.push_back(cur + 1);
        int p = par[mask][cur];
        mask ^= (1 << cur);
        cur = p;
    }
    std::reverse(seq.begin(), seq.end());

    std::vector<int> tour;
    tour.reserve(N);
    tour.push_back(0);
    for (int v : seq) tour.push_back(v);
    return tour;
}

// Nearest-neighbor + 2-opt (for moderate N)
static std::vector<int> solveNN2Opt(const std::vector<City>& cities) {
    int N = (int)cities.size();
    if (N <= 2) {
        std::vector<int> trivial;
        trivial.reserve(N);
        for (int i = 0; i < N; ++i) trivial.push_back(i);
        return trivial;
    }

    // NN tour starting at 0
    std::vector<int> tour;
    tour.reserve(N);
    std::vector<char> used(N, 0);
    int cur = 0;
    tour.push_back(cur);
    used[cur] = 1;
    for (int k = 1; k < N; ++k) {
        int best = -1;
        double bestD = std::numeric_limits<double>::infinity();
        for (int i = 0; i < N; ++i) {
            if (used[i]) continue;
            double d = dist(cities[cur], cities[i]);
            if (d < bestD) {
                bestD = d;
                best = i;
            }
        }
        if (best == -1) break;
        tour.push_back(best);
        used[best] = 1;
        cur = best;
    }

    // 2-opt on the cycle
    bool improved = true;
    const double EPS = 1e-12;
    while (improved) {
        improved = false;
        for (int i = 0; i < N - 1 && !improved; ++i) {
            int i1 = i + 1;
            for (int j = i + 2; j < N; ++j) {
                int a = tour[i];
                int b = tour[i1];
                int c = tour[j];
                int d = tour[(j + 1) % N];
                double delta = dist(cities[a], cities[c]) + dist(cities[b], cities[d])
                             - dist(cities[a], cities[b]) - dist(cities[c], cities[d]);
                if (delta < -EPS) {
                    std::reverse(tour.begin() + i1, tour.begin() + j + 1);
                    improved = true;
                    break;
                }
            }
        }
    }

    return tour;
}

// Initial tour by sorting by x (and y as tie-breaker)
static std::vector<int> initialTourSortByX(const std::vector<City>& cities) {
    int N = (int)cities.size();
    std::vector<int> order(N);
    std::iota(order.begin(), order.end(), 0);
    std::sort(order.begin(), order.end(), [&](int a, int b) {
        if (cities[a].x != cities[b].x) return cities[a].x < cities[b].x;
        if (cities[a].y != cities[b].y) return cities[a].y < cities[b].y;
        return a < b;
    });

    // rotate to start at 0
    int pos = 0;
    for (int i = 0; i < N; ++i) if (order[i] == 0) { pos = i; break; }

    std::vector<int> tour;
    tour.reserve(N);
    for (int i = 0; i < N; ++i) tour.push_back(order[(pos + i) % N]);
    return tour;
}

// Limited randomized 2-opt to improve larger instances without spending too much time
static void limited2opt(std::vector<int>& tour,
                      const std::vector<City>& cities,
                      int iterations) {
    int N = (int)tour.size();
    if (N < 4) return;

    std::mt19937 rng(123456789);
    std::uniform_int_distribution<int> distN(0, N - 1);
    const double EPS = 1e-12;

    for (int it = 0; it < iterations; ++it) {
        int i = distN(rng);
        int j = distN(rng);
        if (i == j) continue;
        int ii = std::min(i, j);
        int jj = std::max(i, j);
        if (jj == ii + 1) continue; // adjacent, no improvement by reversal
        // optional skip full wrap
        // compute delta for reversing [ii+1 .. jj]
        int a = tour[ii];
        int b = tour[ii + 1];
        int c = tour[jj];
        int d = tour[(jj + 1) % N];
        double delta = dist(cities[a], cities[c]) + dist(cities[b], cities[d])
                     - dist(cities[a], cities[b]) - dist(cities[c], cities[d]);
        if (delta < -EPS) {
            std::reverse(tour.begin() + ii + 1, tour.begin() + jj + 1);
        }
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N;
    if (!(std::cin >> N)) return 0;
    std::vector<City> cities;
    cities.reserve(N);
    for (int i = 0; i < N; ++i) {
        double x, y;
        std::cin >> x >> y;
        cities.push_back({i, x, y});
    }

    std::vector<int> tour;

    // Adaptive strategy based on N
    if (N <= 15) {
        tour = solveExact(cities);
    } else if (N <= 2000) {
        tour = solveNN2Opt(cities);
    } else {
        // Large-scale: fast, scalable heuristic
        tour = initialTourSortByX(cities);
        // Optional lightweight improvement for moderately large N
        if (N <= 100000) {
            limited2opt(tour, cities, 20000); // deterministic seed in function
        }
    }

    // Ensure tour starts at city 0
    int pos0 = -1;
    for (int i = 0; i < (int)tour.size(); ++i) {
        if (tour[i] == 0) { pos0 = i; break; }
    }
    if (pos0 <= 0) {
        // If 0 is already first or not found (shouldn't happen), rotate to make first
        // If pos0 == -1, just start from 0
        if (pos0 == -1) pos0 = 0;
    }

    // Rotate so that 0 is first
    std::vector<int> output;
    output.reserve(N);
    for (int i = 0; i < N; ++i) {
        output.push_back(tour[(pos0 + i) % N]);
    }

    // Print single line: N integers
    for (int i = 0; i < N; ++i) {
        if (i) std::cout << ' ';
        std::cout << output[i];
    }
    std::cout << '\n';
    return 0;
}
``` 

Explanation of adaptive strategy:
- For very small N (N <= 15), we solve exactly with Held-Karp DP to guarantee optimality.
- For medium N (up to 2000), we use a robust NN construction followed by 2-opt local improvement on a cycle to enhance route quality.
- For larger N, we switch to a fast heuristic: initial tour by sorting by x-coordinate (with y tie-breaker) and apply a light, bounded 2-opt like improvement (random pair reversals) to improve quality without blowing up runtime. The code always starts the tour at city 0 (rotation step).

The output is a valid permutation of 0..N-1 with 0 as the first city, as required. The implementation uses standard C++17/C++20 features supported by MSVC 19.50.
