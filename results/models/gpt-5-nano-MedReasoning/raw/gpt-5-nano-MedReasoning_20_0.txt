```rust
use std::cmp::Ordering;
use std::collections::HashMap;
use std::f64;
use std::io::{self, Read};

// A small, self-contained Rust program to detect suspect drillhole samples.
// It reads from stdin, writes to stdout, and uses only the standard library.
//
// Idea (adaptive and scalable):
// - Each sample has 3D position (computed from hole start and direction) and a vector of property values.
// - For each sample and for every property, we compute an anomaly score (z-score) comparing the sample
//   value to the local neighborhood (nearby samples by 3D distance).
// - If neighborhood data is insufficient, we fall back to the global distribution for that property.
// - We output one line per suspect (hole_id, sample_idx, property_name, confidence) sorted by
//   confidence descending. Confidence is roughly proportional to |z| with a cap.
// - An adaptive strategy is implemented by tuning a grid-based neighbor search to scale from thousands
//   to hundreds of thousands of samples.
//
// Input format (stdin):
// num_holes num_properties
// property_name_1 property_name_2 ...
// hole_id start_x start_y start_z dir_x dir_y dir_z length num_samples
// depth prop1 prop2 ...
// depth prop1 prop2 ...
// ... (repeat per hole)
//
 // Output format (stdout):
// M
// hole_id sample_idx property_name confidence
// ...

// A tiny fast input scanner for whitespace-delimited tokens.
struct Scanner {
    data: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self {
            data: input.into_bytes(),
            idx: 0,
        }
    }

    fn skip_whitespace(&mut self) {
        while self.idx < self.data.len() {
            let b = self.data[self.idx];
            if b == b' ' || b == b'\n' || b == b'\r' || b == b'\t' {
                self.idx += 1;
            } else {
                break;
            }
        }
    }

    fn next<T: std::str::FromStr>(&mut self) -> T {
        self.skip_whitespace();
        let start = self.idx;
        while self.idx < self.data.len() {
            let b = self.data[self.idx];
            if b == b' ' || b == b'\n' || b == b'\r' || b == b'\t' {
                break;
            }
            self.idx += 1;
        }
        let s = std::str::from_utf8(&self.data[start..self.idx]).unwrap();
        s.parse::<T>().ok().unwrap()
    }

    fn next_string(&mut self) -> String {
        self.skip_whitespace();
        let start = self.idx;
        while self.idx < self.data.len() {
            let b = self.data[self.idx];
            if b == b' ' || b == b'\n' || b == b'\r' || b == b'\t' {
                break;
            }
            self.idx += 1;
        }
        std::str::from_utf8(&self.data[start..self.idx])
            .unwrap()
            .to_string()
    }
}

struct Hole {
    id: String,
    samples: Vec<Sample>,
    // keep original position and direction for possible debugging; not strictly needed later
    start: [f64; 3],
    dir_norm: [f64; 3],
}

struct Sample {
    depth: f64,
    values: Vec<f64>,
    pos: [f64; 3],
    sample_idx: usize, // index within this hole
}

struct GlobalMeanStd {
    mean: Vec<f64>,
    std: Vec<f64>,
    count: Vec<usize>,
}

struct Candidate {
    hole_idx: usize,
    sample_idx: usize,
    property_idx: usize,
    property_name: String,
    confidence: f64,
}

fn main() {
    // 1. Parse input
    let mut sc = Scanner::new();

    let num_holes: usize = sc.next();
    let num_properties: usize = sc.next();

    // Property names
    let mut property_names: Vec<String> = Vec::with_capacity(num_properties);
    for _ in 0..num_properties {
        property_names.push(sc.next_string());
    }

    let mut holes: Vec<Hole> = Vec::with_capacity(num_holes);

    for _ in 0..num_holes {
        // hole header
        let hole_id = sc.next_string();

        let start_x: f64 = sc.next();
        let start_y: f64 = sc.next();
        let start_z: f64 = sc.next();

        let dir_x: f64 = sc.next();
        let dir_y: f64 = sc.next();
        let dir_z: f64 = sc.next();

        let length: f64 = sc.next(); // not directly used except for semantics

        let num_samples: usize = sc.next();

        // compute normalized direction
        let mut nx = dir_x;
        let mut ny = dir_y;
        let mut nz = dir_z;
        let len = (dir_x * dir_x + dir_y * dir_y + dir_z * dir_z).sqrt();
        if len > 0.0 {
            nx = dir_x / len;
            ny = dir_y / len;
            nz = dir_z / len;
        } else {
            nx = 0.0;
            ny = 0.0;
            nz = 0.0;
        }

        let mut samples: Vec<Sample> = Vec::with_capacity(num_samples);
        for si in 0..num_samples {
            let depth = sc.next::<f64>();
            let mut vals: Vec<f64> = Vec::with_capacity(num_properties);
            for _ in 0..num_properties {
                vals.push(sc.next::<f64>());
            }
            let pos = [
                start_x + nx * depth,
                start_y + ny * depth,
                start_z + nz * depth,
            ];
            samples.push(Sample {
                depth,
                values: vals,
                pos,
                sample_idx: si,
            });
        }

        holes.push(Hole {
            id: hole_id,
            samples,
            start: [start_x, start_y, start_z],
            dir_norm: [nx, ny, nz],
        });
    }

    // Flatten all samples for processing
    let total_samples: usize = holes.iter().map(|h| h.samples.len()).sum();
    if total_samples == 0 {
        // Nothing to process
        println!("0");
        return;
    }

    // Build a flat vector of all samples with minimal per-sample metadata
    // We will reuse positions and property vectors by cloning values as needed.
    #[derive(Clone)]
    struct FlatSample {
        hole_idx: usize,
        sample_idx: usize,
        pos: [f64; 3],
        values: Vec<f64>,
    }

    let mut all_samples: Vec<FlatSample> = Vec::with_capacity(total_samples);
    for (hi, hole) in holes.iter().enumerate() {
        for s in hole.samples.iter() {
            all_samples.push(FlatSample {
                hole_idx: hi,
                sample_idx: s.sample_idx,
                pos: s.pos,
                values: s.values.clone(),
            });
        }
    }

    // Precompute global per-property mean/std (with non-NaN values only)
    let mut global_mean: Vec<f64> = vec![0.0; num_properties];
    let mut global_std: Vec<f64> = vec![0.0; num_properties];
    let mut global_count: Vec<usize> = vec![0; num_properties];

    // First pass: sums
    for p in 0..num_properties {
        let mut sum = 0.0;
        let mut sumsq = 0.0;
        let mut cnt = 0usize;
        for s in all_samples.iter() {
            let v = s.values[p];
            if v.is_finite() {
                sum += v;
                sumsq += v * v;
                cnt += 1;
            }
        }
        global_count[p] = cnt;
        if cnt > 0 {
            let mean = sum / (cnt as f64);
            // unbiased variance estimate
            let var = if cnt > 1 {
                let v = sumsq - sum * sum / (cnt as f64);
                (v / (cnt as f64 - 1.0)).max(0.0)
            } else {
                0.0
            };
            global_mean[p] = mean;
            global_std[p] = var.sqrt();
        } else {
            global_mean[p] = f64::NAN;
            global_std[p] = f64::NAN;
        }
    }

    // Adaptive grid-based neighbor search
    // Choose a cell size based on dataset size (scales reasonably)
    let cell_size: f64 = {
        // Simple heuristic: scale cell size with dataset size
        if all_samples.len() > 200_000 {
            80.0
        } else if all_samples.len() > 100_000 {
            64.0
        } else if all_samples.len() > 40_000 {
            32.0
        } else {
            16.0
        }
    };

    // Radius to consider neighbors
    let radius: f64 = if all_samples.len() > 100_000 { 60.0 } else { 40.0 };

    // Build grid: cell -> indices
    let mut grid: HashMap<(i32, i32, i32), Vec<usize>> = HashMap::new();
    let mut cell_of: Vec<(i32, i32, i32)> = Vec::with_capacity(all_samples.len());

    // Helper to compute cell coordinate
    let to_cell = |v: f64| -> i32 { (v / cell_size).floor() as i32 };

    for (idx, s) in all_samples.iter().enumerate() {
        let (cx, cy, cz) = (to_cell(s.pos[0]), to_cell(s.pos[1]), to_cell(s.pos[2]));
        cell_of.push((cx, cy, cz));
        grid.entry((cx, cy, cz))
            .or_insert_with(Vec::new)
            .push(idx);
    }

    // Collect candidates
    let min_neighbors: usize = 3; // require at least this many neighbors for local statistics

    let mut candidates: Vec<Candidate> = Vec::new();

    // For each sample, accumulate neighbor sums for each property
    // We'll reuse a few vectors to avoid repeated allocations
    let mut sum_p: Vec<f64> = vec![0.0; num_properties];
    let mut sumsq_p: Vec<f64> = vec![0.0; num_properties];
    let mut cnt_p: Vec<usize> = vec![0; num_properties];

    for i in 0..all_samples.len() {
        // reset accumulators
        for p in 0..num_properties {
            sum_p[p] = 0.0;
            sumsq_p[p] = 0.0;
            cnt_p[p] = 0;
        }

        // gather neighbors from 27 neighboring cells
        let (ix, iy, iz) = cell_of[i];
        for dx in -1..=1 {
            for dy in -1..=1 {
                for dz in -1..=1 {
                    let key = (ix + dx, iy + dy, iz + dz);
                    if let Some(indices) = grid.get(&key) {
                        for &j in indices {
                            if j == i {
                                continue;
                            }
                            // distance check to ensure near neighbors (saves some work)
                            // We only consider those within the specified radius.
                            let dxp = all_samples[i].pos[0] - all_samples[j].pos[0];
                            let dyp = all_samples[i].pos[1] - all_samples[j].pos[1];
                            let dzp = all_samples[i].pos[2] - all_samples[j].pos[2];
                            let dist = (dxp * dxp + dyp * dyp + dzp * dzp).sqrt();
                            if dist <= radius {
                                // accumulate per-property values for neighbors
                                for p in 0..num_properties {
                                    let v = all_samples[j].values[p];
                                    if v.is_finite() {
                                        sum_p[p] += v;
                                        sumsq_p[p] += v * v;
                                        cnt_p[p] += 1;
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }

        // For this sample i, compute per-property z-scores where possible
        for p in 0..num_properties {
            let v = all_samples[i].values[p];
            if !v.is_finite() {
                continue;
            }

            // compute neighbor-based statistics if we had enough neighbors
            let mut z: Option<f64> = None;
            if cnt_p[p] >= min_neighbors && cnt_p[p] > 1 {
                let mean = sum_p[p] / (cnt_p[p] as f64);
                let var = {
                    let m = mean;
                    // sample variance estimate
                    let s = sumsq_p[p] - (sum_p[p] * sum_p[p]) / (cnt_p[p] as f64);
                    if cnt_p[p] > 1 {
                        (s / (cnt_p[p] as f64 - 1.0)).max(0.0)
                    } else {
                        0.0
                    }
                };
                let std = var.sqrt();
                if std > 0.0 {
                    z = Some((v - mean) / std);
                }
            }

            // If neighbor-based z not available, fall back to global distribution
            if z.is_none() {
                if global_count[p] >= 2 && global_std[p] > 0.0 && global_std[p].is_finite() && global_mean[p].is_finite() {
                    z = Some((v - global_mean[p]) / global_std[p]);
                }
            }

            if let Some(zv) = z {
                // Confidence is a simple function of |z|
                // Cap at 1.0 using a threshold (roughly akin to z=3 -> 1.0)
                let conf = (zv.abs() / 3.0).min(1.0);
                if conf > 0.0 {
                    candidates.push(Candidate {
                        hole_idx: all_samples[i].hole_idx,
                        sample_idx: all_samples[i].sample_idx,
                        property_idx: p,
                        property_name: property_names[p].clone(),
                        confidence: conf,
                    });
                }
            }
        }
    }

    // Sort candidates by confidence descending
    candidates.sort_by(|a, b| {
        // Descending by confidence
        b.confidence.partial_cmp(&a.confidence).unwrap_or(Ordering::Equal)
    });

    // Output
    // First line: M
    let m = candidates.len();
    let mut out = String::new();
    out.push_str(&format!("{}", m));
    out.push('\n');
    for c in candidates.iter() {
        let hole_id = &holes[c.hole_idx].id;
        // sample_idx is 0-based per hole
        out.push_str(&format!(
            "{} {} {} {:.6}\n",
            hole_id, c.sample_idx, c.property_name, c.confidence
        ));
        // Note: per problem statement, each line is: hole_id sample_idx property_name confidence
    }

    print!("{}", out);
}
```


Explanation of key design choices and adaptive strategy:

- Input handling: A compact, no-allocating scanner reads tokens directly from stdin. It tolerates large datasets by streaming tokens into memory-as-needed. All data uses f64 for numerical stability.

- Data model:
  - Hole: stores id, start point, direction (normalized for 3D sampling), and its samples.
  - Sample: stores depth, property values, and position in 3D space (computed as start + unit_dir * depth). This enables spatial neighborhood queries.

- Outlier scoring:
  - For each sample and each property, we compute a z-score that expresses how far the sample’s value is from the local neighborhood mean, scaled by local neighborhood standard deviation.
  - If insufficient local neighborhood data exists (neighbor count below min_neighbors), we fall back to the global distribution for that property (mean and std across all samples for that property).
  - If even global stats are unavailable, the property is skipped for that sample.
  - A candidate is produced for every (hole, sample, property) triple that yields a finite confidence. Confidence is a normalized value between 0 and 1, roughly representing how anomalous the value is (|z| / 3.0 clamped to [0,1]).

- Spatial indexing and scale adaptation:
  - To ensure the program scales from thousands to tens of thousands of samples (and beyond), a grid-based neighbor search is used.
  - The grid is constructed from sample positions using a cell_size that adapts to the dataset size:
    - tiny data: cell_size 16.0
    - medium: 32.0
    - large: 64.0–80.0
  - Radius for neighbor search is radius = 40–60 depending on dataset size. The algorithm checks neighboring cells (a 3x3x3 cube of cells) to find nearby samples and accumulate per-property sums and sums of squares.

- Robustness considerations:
  - Missing data (NaNs) in property values are skipped for both neighbor statistics and global statistics.
  - If a sample has NaN for a property, that property is simply not considered for anomaly when computing candidates.
  - The code does not deduplicate samples (to preserve original sample indexing). Duplicates will naturally yield suspect signals as they create nearly identical positions and values, which may appear as outliers for one of the duplicates due to neighborhood statistics.

- Output:
  - The program outputs M on the first line, followed by M lines of:
    hole_id sample_idx property_name confidence
  - Lines are sorted by confidence descending to emphasize the most obvious anomalies first.
  - sample_idx is 0-based within that hole, aligning with common indexing conventions in code-oriented tasks. If you need 1-based indexing, adjust the print accordingly.

- Edge cases handled:
  - No samples: prints 0.
  - Missing property values: properties with missing data are skipped in anomaly reporting.
  - Edge-case geometry (zero-length direction): positions default to the hole start if the direction vector is zero-length.

- Performance notes:
  - The grid-based neighbor search avoids O(n^2) comparisons and scales well with large datasets.
  - Global statistics are computed once and reused for fallback scoring.
  - The program uses only the Rust standard library and should compile with rustc 1.92.0 or newer.
