```glsl
#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) writeonly buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    if (i >= N) {
        return;
    }

    // Load current boid
    vec3 pos = inData[i * 2 + 0].xyz;
    vec3 vel = inData[i * 2 + 1].xyz;

    float dt     = radii.x;
    float sepR   = radii.y;
    float alignR = radii.z;
    float cohR   = radii.w;

    vec3 sepAcc   = vec3(0.0);
    vec3 alignAcc = vec3(0.0);
    vec3 cohAcc   = vec3(0.0);

    int nSep   = 0;
    int nAlign = 0;
    int nCoh   = 0;

    // Simple O(N) neighborhood search (per step). No self, no zero-distance.
    for (uint j = 0u; j < N; ++j) {
        if (j == i) continue;

        vec3 nPos = inData[j * 2 + 0].xyz;
        vec3 nVel = inData[j * 2 + 1].xyz;

        vec3 delta = nPos - pos;
        float dist = length(delta);

        // Separation: steer away from too-close neighbors
        if (sepR > 0.0 && dist > 0.0 && dist < sepR) {
            vec3 away = pos - nPos;
            away = away / max(dist, 1e-6);
            sepAcc += away;
            nSep += 1;
        }

        // Alignment: align velocity with neighbors
        if (alignR > 0.0 && dist > 0.0 && dist < alignR) {
            alignAcc += nVel;
            nAlign += 1;
        }

        // Cohesion: steer toward center of mass (mean position)
        if (cohR > 0.0 && dist > 0.0 && dist < cohR) {
            cohAcc += nPos;
            nCoh += 1;
        }
    }

    vec3 accel = vec3(0.0);

    // Apply averaged contributions with weights
    if (nSep > 0) {
        accel += (sepAcc / float(nSep)) * weights.x;
    }
    if (nAlign > 0) {
        accel += (alignAcc / float(nAlign)) * weights.y;
    }
    if (nCoh > 0) {
        vec3 meanPos = cohAcc / float(nCoh);
        accel += (meanPos - pos) * weights.z;
    }

    // Update velocity with acceleration
    vel += accel * dt;

    // Clamp speed to maxSpeed
    float maxSpeed = weights.w;
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = normalize(vel) * maxSpeed;
    }

    // Boundary handling: push toward center when beyond 0.4*boundSize,
    // with strength scaling up to 0.5*boundSize using boundForce.
    float boundSize   = bounds.x;
    float boundHalf   = boundSize * 0.5;
    float threshold   = boundSize * 0.4;

    // X axis
    if (abs(pos.x) > threshold) {
        float t = clamp((abs(pos.x) - threshold) / (boundHalf - threshold), 0.0, 1.0);
        vel.x += -sign(pos.x) * bounds.y * t;
    }
    // Y axis
    if (abs(pos.y) > threshold) {
        float t = clamp((abs(pos.y) - threshold) / (boundHalf - threshold), 0.0, 1.0);
        vel.y += -sign(pos.y) * bounds.y * t;
    }
    // Z axis
    if (abs(pos.z) > threshold) {
        float t = clamp((abs(pos.z) - threshold) / (boundHalf - threshold), 0.0, 1.0);
        vel.z += -sign(pos.z) * bounds.y * t;
    }

    // Update position
    pos += vel * dt;

    // Write results
    outData[i * 2 + 0] = vec4(pos, 0.0);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
```
