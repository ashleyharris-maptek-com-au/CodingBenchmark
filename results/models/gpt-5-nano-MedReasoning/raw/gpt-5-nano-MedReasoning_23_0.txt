use std::io::{self, Read, Write};

fn clamp(v: f64, min: f64, max: f64) -> f64 {
    if v < min { min } else if v > max { max } else { v }
}

/// Simple helper: sign of angle (-1, 0, 1)
fn sign(v: f64) -> f64 {
    if v > 0.0 { 1.0 } else if v < 0.0 { -1.0 } else { 0.0 }
}

/// Check if a map cell at world coordinates (x, y) is an obstacle.
/// The map is a Vec<Vec<char>> with rows from top (y=0) to bottom.
fn is_obstacle_at(x: f64, y: f64, map: &Vec<Vec<char>>) -> bool {
    let h = map.len();
    if h == 0 { return true; }
    let w = map[0].len();

    if x < 0.0 || y < 0.0 { return true; }
    let ix = x.floor() as isize;
    let iy = y.floor() as isize;
    if ix < 0 || iy < 0 { return true; }
    if (ix as usize) >= w || (iy as usize) >= h { return true; }

    match map[iy as usize][ix as usize] {
        '#' => true,
        _ => false,
    }
}

fn main() {
    // Read entire stdin once to simplify initial parsing.
    // The environment provides a stream: initial map block, then state updates.
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.lines();

    // Parse initial map header (width height gravity max_thrust fuel)
    let line1 = loop {
        if let Some(l) = it.next() {
            let s = l.trim();
            if !s.is_empty() {
                break s.to_string();
            }
        } else {
            // No data
            return;
        }
    };
    let mut t1 = line1.split_whitespace();
    let width: usize = t1.next().unwrap().parse().unwrap();
    let height: usize = t1.next().unwrap().parse().unwrap();
    let gravity: f64 = t1.next().unwrap().parse().unwrap();
    let max_thrust: f64 = t1.next().unwrap().parse().unwrap();
    let _fuel_max: f64 = t1.next().unwrap().parse().unwrap(); // total fuel capacity (not strictly needed)

    // Parse start and target line
    let line2 = loop {
        if let Some(l) = it.next() {
            let s = l.trim();
            if !s.is_empty() {
                break s.to_string();
            }
        } else {
            // Not enough data
            return;
        }
    };
    let mut t2 = line2.split_whitespace();
    let start_x: f64 = t2.next().unwrap().parse().unwrap();
    let start_y: f64 = t2.next().unwrap().parse().unwrap();
    let target_x: f64 = t2.next().unwrap().parse().unwrap();
    let target_y: f64 = t2.next().unwrap().parse().unwrap();

    // Parse map rows
    // Map uses height lines following the header
    let mut map: Vec<Vec<char>> = Vec::with_capacity(height);
    for _ in 0..height {
        let line = loop {
            if let Some(l) = it.next() {
                // Keep line even if it has trailing spaces
                let s = l;
                // If the line is non-empty, use it
                // (Line may contain characters like ".#STL" possibly with spaces)
                // We don't require pure length here; we'll filter valid chars.
                break s.to_string();
            } else {
                // If input ends early, bail out
                return;
            }
        };
        // Filter only valid map characters
        let row: Vec<char> = line
            .chars()
            .filter(|&c| matches!(c, '.' | '#' | 'S' | 'T' | 'L'))
            .collect();
        // If the row is shorter than width, pad with empty spaces (treated as obstacle)
        let mut r = row;
        if r.len() < width {
            r.resize(width, '#'); // be safe
        }
        map.push(r);
    }

    // We will now process streaming state updates from the remaining data.
    // The rest of the input is expected to be lines with:
    // x y vx vy angle angular_vel fuel_remaining
    // (7 numbers)
    // Since we already read the entire stdin into a string, we need to parse
    // remaining tokens from the leftover lines (if any).
    // Build a vector of all remaining tokens.
    let mut remaining_tokens: Vec<f64> = Vec::new();
    for line in it {
        let mut itok = line.split_whitespace();
        for tok in itok.by_ref() {
            if let Ok(v) = tok.parse::<f64>() {
                remaining_tokens.push(v);
            }
        }
    }

    // If there are no updates yet, we can't output anything.
    // We'll process tokens in chunks of 7.
    // We also need a reasonable time step. The problem statement says updates come every 0.1s.
    // We'll assume dt = 0.1 for our internal heuristics.
    let dt: f64 = 0.1;

    // Prepare a simple output handle to flush frequently
    let stdout = io::stdout();
    let mut handle = io::BufWriter::new(stdout.lock());

    // If there is an initial state provided by remaining_tokens, we can start producing commands.
    // We'll process as many complete states as possible.
    let mut idx = 0usize;
    while idx + 6 < remaining_tokens.len() {
        // Parse one state update (7 numbers)
        let x = remaining_tokens[idx + 0];
        let y = remaining_tokens[idx + 1];
        let vx = remaining_tokens[idx + 2];
        let vy = remaining_tokens[idx + 3];
        let angle = remaining_tokens[idx + 4];
        let angular_vel = remaining_tokens[idx + 5];
        let fuel_remaining = remaining_tokens[idx + 6];

        idx += 7;

        // Basic safety: if we run out of fuel, zero thrust.
        // We'll compute simple thrust/turn commands.

        // Horizontal decision: aim toward the target in x
        let dx = target_x - x;
        // Desired lean angle to push toward target (smoothed)
        // Limit to +/- 0.35 rad (~20 degrees)
        let desired_angle = clamp(dx * 0.01, -0.35, 0.35);

        // Turn command to move angle toward desired_angle
        // We use a simple proportional controller on angle error
        let mut turn = (desired_angle - angle) * 2.0; // gain
        turn = clamp(turn, -1.0, 1.0);

        // Vertical thrust heuristic (gravity is downward; thrust acts along lander's facing)
        // Use a simple rule: counter gravity enough to slow descent, but not waste fuel.
        let mut thrust = 0.0f64;
        if fuel_remaining > 0.0 {
            // If descending fast, burn more
            if vy > 0.5 {
                thrust = 0.9;
            } else if vy > -0.2 {
                thrust = 0.6;
            } else {
                thrust = 0.25;
            }

            // When approaching target vertically, reduce thrust for a soft landing
            let dy_to_target = target_y - y;
            if dx.abs() < 1.0 && dy_to_target.abs() < 3.0 {
                thrust = 0.15;
            }

            // If near ground bounds or predicted collision, reduce thrust to avoid wall
            let next_x = x + vx * dt;
            let next_y = y + vy * dt;
            if is_obstacle_at(next_x, next_y, &map) {
                thrust = 0.0;
                // Try to gently adjust by turning toward target
                let away = -sign(angle);
                turn = clamp(away * 0.6, -1.0, 1.0);
            }
        }

        // Clamp thrust to [0, 1]
        thrust = clamp(thrust, 0.0, 1.0);

        // Print command: "thrust turn" with two decimals
        writeln!(handle, "{:.2} {:.2}", thrust, turn).unwrap();
        handle.flush().unwrap();
    }

    // If there were fewer than 7 tokens, the program would just exit gracefully.
}
