; SPIR-V Geometry shader implementing a thick ribbon from InputLines (2 vertices)
; Produces a quad strip (4 vertices) for a ribbon around the input thin line
; Reads per-vertex positions and colors, outputs gl_Position and outColor

OpCapability Geometry
OpMemoryModel Logical GLSL450
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%int32 = OpTypeInt 32 1
%uint32 = OpTypeInt 32 0

; Size for input primitive (2 for InputLines)
%size2 = OpConstant %uint32 2

; gl_PerVertex_in (input from vertex shader)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; gl_in as array of per-vertex input structures
%arr_pv_in = OpTypeArray %gl_PerVertex_in %size2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
; Access: OpAccessChain %ptr_in_v4 %gl_in %int_i %int_0  -> gl_in[i].gl_Position

; Input arrays from vertex shader (worldPos, normal, color)
%arr_worldPos = OpTypeArray %v3float %size2
%ptr_in_arr_worldPos = OpTypePointer Input %arr_worldPos
%in_worldPos = OpVariable %ptr_in_arr_worldPos Input
OpDecorate %in_worldPos Location 0

%arr_normal = OpTypeArray %v3float %size2
%ptr_in_arr_normal = OpTypePointer Input %arr_normal
%in_normal = OpVariable %ptr_in_arr_normal Input
OpDecorate %in_normal Location 1

%arr_color = OpTypeArray %v3float %size2
%ptr_in_arr_color = OpTypePointer Input %arr_color
%in_color = OpVariable %ptr_in_arr_color Input
OpDecorate %in_color Location 2

; Geometry shader output: gl_PerVertex_out (contains gl_Position)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Out color to fragment shader
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output
OpDecorate %outColor Location 0

; Entry point
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor

; Execution modes
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Function declarations
%void = OpTypeVoid
%func_type = OpTypeFunction %void

; Pointers and constants
%ptr_in_pv_v4 = OpTypePointer Input %v4float
%ptr_color_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%zero_vec3 = OpConstant %v3float 0.0 0.0 0.0  ; not used, placeholder

; 0 and 1 indices for accessing arrays
%int32 = OpTypeInt 32 1
%idx0 = OpConstant %int32 0
%idx1 = OpConstant %int32 1
%zero0 = OpConstant %int32 0

; Normal float constants
%float0 = OpConstant %float 0.0
%float1 = OpConstant %float 1.0
%float_thickness = OpConstant %float 0.03

; Main function
%main = OpFunction %void None %func_type
%entry = OpLabel

; Load endpoints from gl_in
%p0_ptr = OpAccessChain %ptr_in_pv_v4 %gl_in %idx0 %zero0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_pv_v4 %gl_in %idx1 %zero0
%p1 = OpLoad %v4float %p1_ptr

; Load colors from input color[] (c0 and c1)
%col_ptr0 = OpAccessChain %ptr_color_v3 %in_color %idx0
%col0 = OpLoad %v3float %col_ptr0
%col_ptr1 = OpAccessChain %ptr_color_v3 %in_color %idx1
%col1 = OpLoad %v3float %col_ptr1

; Extract x,y components for end points
%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p1x = OpCompositeExtract %float %p1 0
%p1y = OpCompositeExtract %float %p1 1

; dir = p1 - p0
%dir_x = OpFSub %float %p1x %p0x
%dir_y = OpFSub %float %p1y %p0y

; perp = normalize(-dir.y, dir.x) * thickness
%neg_dir_y = OpFNegate %float %dir_y
%dir_x2 = OpFMul %float %dir_x %dir_x
%dir_y2 = OpFMul %float %dir_y %dir_y
%len2 = OpFAdd %float %dir_x2 %dir_y2
%len = OpExtInst %float %GLSLstd450 Sqrt %len2
%inv_len = OpFDiv %float %float1 %len

%tmp1 = OpFMul %float %neg_dir_y %inv_len
%perp_x = OpFMul %float %tmp1 %float_thickness
%tmp2 = OpFMul %float %dir_x %inv_len
%perp_y = OpFMul %float %tmp2 %float_thickness

; Offsets for thick line
%off = OpCompositeConstruct %v4float %perp_x %perp_y %float0 %float0
%neg_perp_x = OpFNegate %float %perp_x
%neg_perp_y = OpFNegate %float %perp_y
%neg_off = OpCompositeConstruct %v4float %neg_perp_x %neg_perp_y %float0 %float0

; Four positions for the ribbon quad (p0+perp, p0-perp, p1+perp, p1-perp)
%pos0p = OpFAdd %v4float %p0 %off
%pos0m = OpFAdd %v4float %p0 %neg_off
%pos1p = OpFAdd %v4float %p1 %off
%pos1m = OpFAdd %v4float %p1 %neg_off

; Pointer to gl_Position inside gl_out
%pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %zero0

; Emit Vertex 0: p0 + perp with color c0
OpStore %pos_ptr %pos0p
OpStore %outColor %col0
OpEmitVertex

; Emit Vertex 1: p0 - perp with color c0
OpStore %pos_ptr %pos0m
OpStore %outColor %col0
OpEmitVertex

; Emit Vertex 2: p1 + perp with color c1
OpStore %pos_ptr %pos1p
OpStore %outColor %col1
OpEmitVertex

; Emit Vertex 3: p1 - perp with color c1
OpStore %pos_ptr %pos1m
OpStore %outColor %col1
OpEmitVertex

; End primitive and finish
OpEndPrimitive
OpReturn
OpFunctionEnd

; End of shader
