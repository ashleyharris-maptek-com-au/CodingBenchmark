; SPIR-V Geometry Shader implementing a dashed thick curve with adjacency
; Geometry shader using InputLinesAdjacency, OutputTriangleStrip, 8 vertices total
; Caps: Geometry, Logical GLSL450
; Entry point: main, with input/output interface variables listed
; Assumes thickness = 0.01 (half thickness of 0.02)
; Splits segment AB into 3 parts, draws AB and CD as two quads (8 vertices total)

/* Header and capabilities */
OpCapability Geometry
OpMemoryModel Logical GLSL450

; Import GLSL std450 for math (sqrt, etc.)
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Type declarations
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0

; Basic constants
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_third = OpConstant %float 0.33333334
%float_twothird = OpConstant %float 0.66666669
%float_half = OpConstant %float 0.01
%float_epsilon = OpConstant %float 0.0

; Vector/matrix placeholders (UBO is declared for interface completeness)
%mat4 = OpTypeMatrix %v4float 4
; UBO layout placeholder (not used in computations here)
%UBOType = OpTypeStruct
%UBOPtr = OpTypePointer Uniform %UBOType
%UBO = OpVariable %UBOPtr Uniform

; gl_PerVertex_in (input from Vertex shader)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Per-vertex inputs from Vertex shader (arrays)
; worldPos[]; normal[]; color[]
%arr_v3float = OpTypeArray %v3float %uint_4

%ptr_in_worldPos = OpTypePointer Input %arr_v3float
%worldPos = OpVariable %ptr_in_worldPos Input

%ptr_in_normal = OpTypePointer Input %arr_v3float
%normal = OpVariable %ptr_in_normal Input

%ptr_in_color = OpTypePointer Input %arr_v3float
%color = OpVariable %ptr_in_color Input

; gl_PerVertex_out (output to Fragment Shader)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color (per-vertex to fragment shader)
%outColorType = OpTypeVector %float 3
%ptr_out_color = OpTypePointer Output %outColorType
%outColor = OpVariable %ptr_out_color Output  ; layout(location = 0)

; Offsets and helpers
%ptr_out_v4 = OpTypePointer Output %v4float

; Function and entry point
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Per-vertex color interpolation helpers
; (Load color[1] and color[2] once per invocation)
%idx_1 = %uint_1
%idx_2 = %uint_2

; Load color[1]
%color1_ptr = OpAccessChain %ptr_v3float %color %idx_1
%color1 = OpLoad %v3float %color1_ptr
; Load color[2]
%color2_ptr = OpAccessChain %ptr_v3float %color %idx_2
%color2 = OpLoad %v3float %color2_ptr

; Load gl_Position for AB endpoints (start = gl_in[1], end = gl_in[2])
%start_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_1 %uint_0
%start = OpLoad %v4float %start_ptr

%end_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_2 %uint_0
%end = OpLoad %v4float %end_ptr

; Load previous and next vertices for adjacency tangents
%prev_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_0 %uint_0
%prev = OpLoad %v4float %prev_ptr

%next_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_3 %uint_0
%next = OpLoad %v4float %next_ptr

; Extract 2D components (x,y) for tangent computation
%start_x = OpCompositeExtract %float %start %uint_0
%start_y = OpCompositeExtract %float %start %uint_1
%start_z = OpCompositeExtract %float %start %uint_2
%start_w = OpCompositeExtract %float %start %uint_3

%end_x = OpCompositeExtract %float %end %uint_0
%end_y = OpCompositeExtract %float %end %uint_1
%end_z = OpCompositeExtract %float %end %uint_2
%end_w = OpCompositeExtract %float %end %uint_3

%prev_x = OpCompositeExtract %float %prev %uint_0
%prev_y = OpCompositeExtract %float %prev %uint_1
%prev_z = OpCompositeExtract %float %prev %uint_2
%prev_w = OpCompositeExtract %float %prev %uint_3

%next_x = OpCompositeExtract %float %next %uint_0
%next_y = OpCompositeExtract %float %next %uint_1
%next_z = OpCompositeExtract %float %next %uint_2
%next_w = OpCompositeExtract %float %next %uint_3

; Compute 3D segment difference for CD and AB
%dx4 = OpFSub %float %end_x %start_x
%dy4 = OpFSub %float %end_y %start_y
%dz4 = OpFSub %float %end_z %start_z
%dw4 = OpFSub %float %end_w %start_w

; 1/3 and 2/3 factors
; Compute B = start + (end - start)/3 and C = start + 2*(end - start)/3
%oneThirdDx = OpFMul %float %dx4 %float_third
%oneThirdDy = OpFMul %float %dy4 %float_third
%oneThirdDz = OpFMul %float %dz4 %float_third
%oneThirdDw = OpFMul %float %dw4 %float_third

%twoThirdDx = OpFMul %float %dx4 %float_twothird
%twoThirdDy = OpFMul %float %dy4 %float_twothird
%twoThirdDz = OpFMul %float %dz4 %float_twothird
%twoThirdDw = OpFMul %float %dw4 %float_twothird

%Bx = OpFAdd %float start_x %oneThirdDx
By = OpFAdd %float start_y %oneThirdDy
%Bz = OpFAdd %float start_z %oneThirdDz
%Bw = OpFAdd %float start_w %oneThirdDw

%Bcx = OpFAdd %float start_x %twoThirdDx
%Bcy = OpFAdd %float start_y %twoThirdDy
%Bcz = OpFAdd %float start_z %twoThirdDz
%Bcw = OpFAdd %float start_w %twoThirdDw

%Dx = End_x
%Dy = End_y
%Dz = End_z
/Dw = End_w

; Tangent and normal computation (2D: x/y)
%seg_dx = %dx4
%seg_dy = %dy4

; seg_len = sqrt(dx^2 + dy^2)
%seg_dx2 = OpFMul %float %seg_dx %seg_dx
%seg_dy2 = OpFMul %float %seg_dy %seg_dy
%seg_len2 = OpFAdd %float %seg_dx2 %seg_dy2
%seg_len = OpExtInst %float %GLSLstd450 Sqrt %seg_len2
%one_over_len = OpDiv %float %float_1 %seg_len

; t = normalize((end-start))
%tx = OpFMul %float %seg_dx %%one_over_len
; t.y
%ty = OpFMul %float %seg_dy %%one_over_len

; Normal n = vec2(-t.y, t.x)
%n_x = OpFNeg %float %ty
%n_y = %tx

; Offset in clip space: offset = n * halfThickness
%offX = OpFMul %float %n_x %float_half
%offY = OpFMul %float %n_y %float_half

; Build 3D start and end points for AB (A=start, B=B)
; A_plus = (start_x + offX, start_y + offY, start_z, start_w)
; A_minus = (start_x - offX, start_y - offY, start_z, start_w)
%Ax_plus = OpFAdd %float start_x %offX
%Ay_plus = OpFAdd %float start_y %offY
%Az_plus = start_z
%Aw_plus = start_w

%A_plus = OpCompositeConstruct %v4float %Ax_plus %Ay_plus %Az_plus %Aw_plus

%Ax_minus = OpFSub %float start_x %offX
%Ay_minus = OpFSub %float start_y %offY
%Az_minus = start_z
%Aw_minus = start_w
%A_minus = OpCompositeConstruct %v4float %Ax_minus %Ay_minus %Az_minus %Aw_minus

; B_plus / B_minus derived from B (3 partitions)
%B_plus_x = Bx + offX
%B_plus_y = By + offY
%B_plus_z = Bz
%B_plus_w = Bw
%B_plus = OpCompositeConstruct %v4float %B_plus_x %B_plus_y %B_plus_z %B_plus_w

%B_minus_x = Bx - offX
%B_minus_y = By - offY
%B_minus_z = Bz
%B_minus_w = Bw
%B_minus = OpCompositeConstruct %v4float %B_minus_x %B_minus_y %B_minus_z %B_minus_w

; C_plus / C_minus from C
%C_plus = (Cx + offX, Cy + offY, Cz + offZ, Cw + offW)
%C_plus_x = Bcx + offX
%C_plus_y = Bcy + offY
%C_plus_z = Bcz
%C_plus_w = Bcw
%C_plus = OpCompositeConstruct %v4float %Cx_plus %Cy_plus %Cz_plus %Cw_plus

; C_minus
%C_minus_x = Cx - offX
%C_minus_y = Cy - offY
%C_minus_z = Cz
%C_minus_w = Cw
%C_minus = OpCompositeConstruct %v4float %Cx_minus %Cy_minus %C_minus_z %C_minus_w

; D_plus / D_minus from D
%D_plus_x = Dx + offX
%D_plus_y = Dy + offY
%D_plus_z = Dz
%D_plus_w = Dw
%D_plus = OpCompositeConstruct %v4float %D_plus_x %D_plus_y %Dz %Dw ; Dz,Dw already

%D_minus_x = Dx - offX
%D_minus_y = Dy - offY
%D_minus_z = Dz
%D_minus_w = Dw
%D_minus = OpCompositeConstruct %v4float %D_minus_x %D_minus_y %D_minus_z %D_minus_w

; Emit first quad: A_plus, B_plus, A_minus, B_minus
; Set color1 for A_plus and A_minus, color2 for B_plus and B_minus
; gl_Position and outColor assignments for each vertex

; Vertex 1: A_plus (A+)
OpStore %outColor %color1
%gl_pos1 = %A_plus
OpStore %gl_out_v4_ptr = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_out_v4_ptr %gl_pos1
OpEmitVertex

; Vertex 2: B_plus
OpStore %outColor %color2
%gl_pos2 = %B_plus
OpStore %gl_out_v4_ptr %gl_pos2
OpEmitVertex

; Vertex 3: A_minus
OpStore %outColor %color1
%gl_pos3 = %A_minus
OpStore %gl_out_v4_ptr %gl_pos3
OpEmitVertex

; Vertex 4: B_minus
OpStore %outColor %color2
%gl_pos4 = %B_minus
OpStore %gl_out_v4_ptr %gl_pos4
OpEmitVertex

; End first primitive
OpEndPrimitive

; Emit second quad: C_plus, D_plus, C_minus, D_minus
; Vertex 5: C_plus
OpStore %outColor %color1
%gl_pos5 = %C_plus
OpStore %gl_out_v4_ptr %gl_pos5
OpEmitVertex

; Vertex 6: D_plus
OpStore %outColor %color2
%gl_pos6 = %D_plus
OpStore %gl_out_v4_ptr %gl_pos6
OpEmitVertex

; Vertex 7: C_minus
OpStore %outColor %color1
%gl_pos7 = %C_minus
OpStore %gl_out_v4_ptr %gl_pos7
OpEmitVertex

; Vertex 8: D_minus
OpStore %outColor %color2
%gl_pos8 = %D_minus
OpStore %gl_out_v4_ptr %gl_pos8
OpEmitVertex

; End second primitive
OpEndPrimitive

; Finish
OpReturn
OpFunctionEnd

/***************
Notes:
- This shader reads gl_Position for the 2 end points (indices 1 and 2) and uses the adjacency vertices (0 and 3) to compute a smoothed tangent direction in screen space.
- A 2D normal to the tangent is used to offset the segment by a fixed half-thickness (0.01) to achieve a thickness of 0.02 in clip space.
- The segment AB is split into AB and BC parts with AB corresponding to t in [0, 1/3], and CD to [2/3, 1], and the middle third is skipped, producing a dash effect.
- Two quads (each as a strip of 4 vertices) are emitted, totaling 8 vertices, with EndPrimitive called after each quad.
- Colors are taken from color[1] and color[2], and applied as vertex colors to produce an interpolated gradient across the quads.
- A minimal UBO structure is declared to satisfy the requirement of a set/binding layout, though it is not used in the math here.
- The interface list in OpEntryPoint includes the per-vertex inputs (worldPos, normal, color) and the per-vertex color output (outColor).
********/
