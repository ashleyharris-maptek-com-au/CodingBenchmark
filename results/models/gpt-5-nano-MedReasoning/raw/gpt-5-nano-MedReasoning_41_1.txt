; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; Uniform Block (UBO)
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%_ptr_Uniform_Ubo = OpTypePointer Uniform %UBO

; Input / Output pointer types
%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Input_vec2 = OpTypePointer Input %vec2
%_ptr_Output_vec4 = OpTypePointer Output %vec4

; Variables
%in_worldPos = OpVariable %_ptr_Input_vec3 Input
%in_normal   = OpVariable %_ptr_Input_vec3 Input
%in_uv       = OpVariable %_ptr_Input_vec2 Input
%in_tangent  = OpVariable %_ptr_Input_vec3 Input
%in_color    = OpVariable %_ptr_Input_vec3 Input
$outColor    = OpVariable %_ptr_Output_vec4 Output
%ubo         = OpVariable %_ptr_Uniform_Ubo Uniform

; Decorations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block

; Types for function
%voidFn = OpTypeFunction %void
%void = OpTypeVoid
%funcType = OpTypeFunction %void

; Constants
%one = OpConstant %float 1.0
%half = OpConstant %float 0.5

; Main function
%main = OpFunction %void None %funcType
%entry = OpLabel

; Load and compute normalized normal
%normal = OpLoad %vec3 %in_normal
%dot = OpDot %float %normal %normal
%len = OpExtInst %float %GLSLstd450 Sqrt %dot
%invLen = OpFDiv %float %one %len
%normed = OpVectorTimesScalar %vec3 %normal %invLen
%tmp = OpVectorTimesScalar %vec3 %normed %half
%offset = OpCompositeConstruct %vec3 %half %half %half
%colorVec3 = OpFAdd %vec3 %tmp %offset
%colorVec4 = OpCompositeConstruct %vec4 %colorVec3 %one

; Store and finish
OpStore %outColor %colorVec4
OpReturn
OpFunctionEnd

; Entry point and execution mode
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft
