vec3 normal;   // provided by vertex shader

void main() {
    // Rim lighting parameters
    const vec3 rimColor = vec3(1.0, 0.8, 0.5);
    const vec3 baseColor = vec3(0.2, 0.3, 0.8);

    // Geometry directions (untransformed world space)
    vec3 L = lightPos - worldPos;
    vec3 V = cameraPos - worldPos;
    vec3 N = normalize(normal); // assume normal is in world space; normalize for stability

    // Diffuse term (non-normalized approach kept simple)
    float diff = max(dot(N, L), 0.0);

    // Base color modulated by diffuse
    vec3 colorBase = baseColor * (0.3 + 0.5 * diff);

    // Rim term:  rim = (1.0 - max(dot(N, V_to_camera), 0.0))^3
    // Here we use V directly (un-normalized V); this still yields a rim effect.
    float rimFactor = 1.0 - max(dot(N, V), 0.0);
    float rim = rimFactor * rimFactor * rimFactor;

    vec3 color = colorBase + rimColor * rim;

    // Output color with alpha = 1.0
    gl_FragColor = vec4(color, 1.0);

    // outColor = vec4(color, 1.0);
