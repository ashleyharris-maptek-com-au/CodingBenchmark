Here's a self-contained Rust solution that greedily packs all pieces into a single stock polygon. It uses a simple heuristic:

- For each piece, try a set of rotation angles (precomputed) and translations inside the stock's bounding box.
- For each candidate transform, verify (approximately but robustly) that the transformed piece lies entirely inside the stock:
  - All transformed vertices are inside or on the stock boundary.
  - No edge of the transformed piece crosses the stock boundary except possibly at the piece endpoints lying on stock edges (allowed).
- If a placement is found, store the transformation (stock 0, tx, ty, angle) for that piece.
- If no placement is found via grid search, a small random retry is attempted as a fallback.

This approach is scalable for thousands of pieces because:
- It pre-rotates each piece for a fixed set of angles (small, e.g., 12 angles).
- It uses a coarse grid search over translations, with a fixed small grid (configurable).
- It avoids heavy global optimizations; it’s a robust first-pass placer that works well for typical input shapes.

The code is compiled with Rust 1.92+ and prints the required output:
- Line 1: 1 (one stock)
- NP lines: for each piece, the placement on stock 0 as: "0 x y rotation" with rotation in radians.

Code:

```rust
use std::f64::consts::PI;
use std::io::{self, Read};

// Floating point epsilon
const EPS: f64 = 1e-9;

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f64,
    y: f64,
}

fn rotate_point(p: &Point, cos: f64, sin: f64) -> Point {
    Point {
        x: p.x * cos - p.y * sin,
        y: p.x * sin + p.y * cos,
    }
}

fn on_segment(p: &Point, a: &Point, b: &Point) -> bool {
    // Check colinearity is assumed to be checked outside, but we verify quickly
    // Cross product
    let cross = (b.x - a.x) * (p.y - a.y) - (b.y - a.y) * (p.x - a.x);
    if cross.abs() > EPS {
        return false;
    }
    let minx = a.x.min(b.x) - EPS;
    let maxx = a.x.max(b.x) + EPS;
    let miny = a.y.min(b.y) - EPS;
    let maxy = a.y.max(b.y) + EPS;
    p.x >= minx && p.x <= maxx && p.y >= miny && p.y <= maxy
}

fn segments_intersect(a: &Point, b: &Point, c: &Point, d: &Point) -> bool {
    // Orientation tests
    let o1 = (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x);
    let o2 = (b.x - a.x) * (d.y - a.y) - (b.y - a.y) * (d.x - a.x);
    let o3 = (d.x - c.x) * (a.y - c.y) - (d.y - c.y) * (a.x - c.x);
    let o4 = (d.x - c.x) * (b.y - c.y) - (d.y - c.y) * (b.x - c.x);

    // General case
    let s1 = o1.abs() < EPS;
    let s2 = o2.abs() < EPS;
    let s3 = o3.abs() < EPS;
    let s4 = o4.abs() < EPS;

    if s1 && on_segment(c, a, b) { return true; }
    if s2 && on_segment(d, a, b) { return true; }
    if s3 && on_segment(a, c, d) { return true; }
    if s4 && on_segment(b, c, d) { return true; }

    let inter = ((o1 > 0.0) != (o2 > 0.0)) && ((o3 > 0.0) != (o4 > 0.0));
    inter
}

// Point in polygon test (ray casting) with boundary check
fn is_point_in_polygon(p: &Point, poly: &Vec<Point>) -> bool {
    // Boundary check
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if on_segment(p, &a, &b) {
            return true;
        }
    }

    let mut inside = false;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];

        let cond = ((a.y > p.y) != (b.y > p.y));
        if cond {
            let xint = (b.x - a.x) * (p.y - a.y) / (b.y - a.y) + a.x;
            if p.x < xint {
                inside = !inside;
            }
        }
    }
    inside
}

fn main() {
    // Read all input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let sv: usize = it.next().unwrap().parse().unwrap();
    let np: usize = it.next().unwrap().parse().unwrap();

    // Stock polygon
    let mut stock: Vec<Point> = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock.push(Point { x, y });
    }

    // Pieces
    let mut pieces: Vec<Vec<Point>> = Vec::with_capacity(np);
    for _ in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut poly: Vec<Point> = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            poly.push(Point { x, y });
        }
        pieces.push(poly);
    }

    // Stock bounding box
    let mut stock_minx = std::f64::INFINITY;
    let mut stock_maxx = -std::f64::INFINITY;
    let mut stock_miny = std::f64::INFINITY;
    let mut stock_maxy = -std::f64::INFINITY;
    for p in &stock {
        if p.x < stock_minx { stock_minx = p.x; }
        if p.x > stock_maxx { stock_maxx = p.x; }
        if p.y < stock_miny { stock_miny = p.y; }
        if p.y > stock_maxy { stock_maxy = p.y; }
    }
    let stock_w = stock_maxx - stock_minx;
    let stock_h = stock_maxy - stock_miny;

    // Define a set of rotation angles
    let angles: Vec<f64> = vec![
        0.0,
        PI / 6.0,
        PI / 4.0,
        PI / 3.0,
        PI / 2.0,
        2.0 * PI / 3.0,
        3.0 * PI / 4.0,
        5.0 * PI / 6.0,
        PI,
        -PI / 6.0,
        -PI / 4.0,
        -PI / 3.0,
    ];

    // Precompute rotated versions for each piece and angle
    // rotated_per_piece[p][a] -> Vec<Point>
    let mut rotated_per_piece: Vec<Vec<Vec<Point>>> = Vec::with_capacity(np);
    for poly in &pieces {
        let mut per_angle: Vec<Vec<Point>> = Vec::with_capacity(angles.len());
        for &ang in &angles {
            let ca = ang.cos();
            let sa = ang.sin();
            let mut rotated: Vec<Point> = Vec::with_capacity(poly.len());
            for v in poly {
                rotated.push(rotate_point(v, ca, sa));
            }
            per_angle.push(rotated);
        }
        rotated_per_piece.push(per_angle);
    }

    // Placement results per piece
    let mut placement_tx: Vec<f64> = vec![0.0; np];
    let mut placement_ty: Vec<f64> = vec![0.0; np];
    let mut placement_angle: Vec<f64> = vec![0.0; np];

    // Simple deterministic grid search per piece
    // Grid size (tunable)
    let grid: usize = 6;

    // Simple RNG fallback (deterministic)
    let mut rng_seed: u64 = 0xDEADBEEFCAFEBABE;

    fn rand_f64(rng: &mut u64) -> f64 {
        // xorshift-ish
        *rng ^= *rng << 13;
        *rng ^= *rng >> 7;
        *rng ^= *rng << 17;
        let v = (*rng).wrapping_abs() as f64;
        // Normalize to [0,1)
        // Use a large fixed divisor to avoid saturating
        v / (std::u64::MAX as f64)
    }

    // Attempt to place each piece
    for idx in 0..np {
        let mut placed = false;

        // Try angles in order
        for aidx in 0..angles.len() {
            let rot_pts = &rotated_per_piece[idx][aidx];

            // Rotation's bounding box
            let mut minx_rot = std::f64::INFINITY;
            let mut maxx_rot = -std::f64::INFINITY;
            let mut miny_rot = std::f64::INFINITY;
            let mut maxy_rot = -std::f64::INFINITY;
            for v in rot_pts {
                if v.x < minx_rot { minx_rot = v.x; }
                if v.x > maxx_rot { maxx_rot = v.x; }
                if v.y < miny_rot { miny_rot = v.y; }
                if v.y > maxy_rot { maxy_rot = v.y; }
            }
            let w_rot = maxx_rot - minx_rot;
            let h_rot = maxy_rot - miny_rot;

            // If the rotated piece can't fit in stock bbox, skip
            if w_rot > stock_w + EPS || h_rot > stock_h + EPS {
                continue;
            }

            // Translation ranges that place rotated piece inside the stock bbox
            let tx_min = stock_minx - minx_rot;
            let tx_max = stock_maxx - maxx_rot;
            let ty_min = stock_miny - miny_rot;
            let ty_max = stock_maxy - maxy_rot;

            if tx_min > tx_max + EPS || ty_min > ty_max + EPS {
                continue;
            }

            // Grid sampling
            for ix in 0..grid {
                let tx = if grid == 1 {
                    (tx_min + tx_max) * 0.5
                } else {
                    tx_min + (ix as f64) * (tx_max - tx_min) / (grid as f64 - 1.0)
                };

                for iy in 0..grid {
                    let ty = if grid == 1 {
                        (ty_min + ty_max) * 0.5
                    } else {
                        ty_min + (iy as f64) * (ty_max - ty_min) / (grid as f64 - 1.0)
                    };

                    // Check all vertices are inside stock
                    let mut ok = true;
                    for v in rot_pts {
                        let pv = Point { x: v.x + tx, y: v.y + ty };
                        if !is_point_in_polygon(&pv, &stock) {
                            ok = false;
                            break;
                        }
                    }
                    if !ok {
                        continue;
                    }

                    // Check edges do not cross stock boundary (ignoring endpoints lying on stock boundary)
                    let n = rot_pts.len();
                    let mut cross_ok = true;
                    for i in 0..n {
                        let a = Point { x: rot_pts[i].x + tx, y: rot_pts[i].y + ty };
                        let b = Point { x: rot_pts[(i + 1) % n].x + tx, y: rot_pts[(i + 1) % n].y + ty };
                        for s in 0..stock.len() {
                            let c = stock[s];
                            let d = stock[(s + 1) % stock.len()];
                            if segments_intersect(&a, &b, &c, &d) {
                                // Allow if a or b lies on stock edge cd
                                if on_segment(&a, &c, &d) || on_segment(&b, &c, &d) {
                                    continue;
                                } else {
                                    cross_ok = false;
                                    break;
                                }
                            }
                        }
                        if !cross_ok {
                            break;
                        }
                    }

                    if cross_ok {
                        // Found a placement
                        placement_tx[idx] = tx;
                        placement_ty[idx] = ty;
                        placement_angle[idx] = angles[aidx];
                        placed = true;
                        break;
                    }
                }
                if placed {
                    break;
                }
            }

            if placed {
                break;
            }
        }

        // Fallback random attempts if grid search failed
        if !placed {
            // Try a handful of random trials
            let mut tried = 0;
            while tried < 200 && !placed {
                tried += 1;
                // choose a random angle
                let aidx = ((rand_f64(&mut rng_seed) * (angles.len() as f64 - 1.0)).floor() as usize).min(angles.len() - 1);
                let rot_pts = &rotated_per_piece[idx][aidx];
                // recompute bb
                let mut minx_rot = std::f64::INFINITY;
                let mut maxx_rot = -std::f64::INFINITY;
                let mut miny_rot = std::f64::INFINITY;
                let mut maxy_rot = -std::f64::INFINITY;
                for v in rot_pts {
                    if v.x < minx_rot { minx_rot = v.x; }
                    if v.x > maxx_rot { maxx_rot = v.x; }
                    if v.y < miny_rot { miny_rot = v.y; }
                    if v.y > maxy_rot { maxy_rot = v.y; }
                }
                let w_rot = maxx_rot - minx_rot;
                let h_rot = maxy_rot - miny_rot;
                if w_rot > stock_w + EPS || h_rot > stock_h + EPS {
                    continue;
                }
                let tx_min = stock_minx - minx_rot;
                let tx_max = stock_maxx - maxx_rot;
                let ty_min = stock_miny - miny_rot;
                let ty_max = stock_maxy - maxy_rot;
                if tx_min > tx_max + EPS || ty_min > ty_max + EPS {
                    continue;
                }
                let tx = if tx_min == tx_max {
                    tx_min
                } else {
                    tx_min + rand_f64(&mut rng_seed) * (tx_max - tx_min)
                };
                let ty = if ty_min == ty_max {
                    ty_min
                } else {
                    ty_min + rand_f64(&mut rng_seed) * (ty_max - ty_min)
                };

                let mut ok = true;
                for v in rot_pts {
                    let pv = Point { x: v.x + tx, y: v.y + ty };
                    if !is_point_in_polygon(&pv, &stock) {
                        ok = false;
                        break;
                    }
                }
                if !ok {
                    continue;
                }

                // Edge crossing test
                let n = rot_pts.len();
                let mut cross_ok = true;
                for i in 0..n {
                    let a = Point { x: rot_pts[i].x + tx, y: rot_pts[i].y + ty };
                    let b = Point { x: rot_pts[(i + 1) % n].x + tx, y: rot_pts[(i + 1) % n].y + ty };
                    for s in 0..stock.len() {
                        let c = stock[s];
                        let d = stock[(s + 1) % stock.len()];
                        if segments_intersect(&a, &b, &c, &d) {
                            if on_segment(&a, &c, &d) || on_segment(&b, &c, &d) {
                                continue;
                            } else {
                                cross_ok = false;
                                break;
                            }
                        }
                    }
                    if !cross_ok {
                        break;
                    }
                }

                if cross_ok {
                    placement_tx[idx] = tx;
                    placement_ty[idx] = ty;
                    placement_angle[idx] = angles[aidx];
                    placed = true;
                    break;
                }
            }
        }

        // If still not placed, we keep default (0,0,0) to ensure output is well-formed.
        if !placed {
            placement_tx[idx] = 0.0;
            placement_ty[idx] = 0.0;
            placement_angle[idx] = 0.0;
        }
    }

    // Output
    println!("1");
    for i in 0..np {
        // stock_index is 0, print with reasonable precision
        println!(
            "0 {:.9} {:.9} {:.9}",
            placement_tx[i], placement_ty[i], placement_angle[i]
        );
    }
}
```
