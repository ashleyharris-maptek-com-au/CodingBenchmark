; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; GLSL std450 extended instruction set
%GLSL = OpExtInstImport "GLSL.std.450"

; Uniform block (std140 layout)
%UBO_t = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptrUBO = OpTypePointer Uniform %UBO_t

; Pointers for inputs/outputs
%ptrInputVec3 = OpTypePointer Input %vec3
%ptrOutputVec4 = OpTypePointer Output %vec4

; Pointer to uniform block
%ptrUBOPtr = %ptrUBO  ; alias for clarity (not used directly)

; Block variable
%ubo = OpVariable %ptrUBO Uniform
; Input/output variables
%in_worldPos = OpVariable %ptrInputVec3 Input
%in_normal = OpVariable %ptrInputVec3 Input
%outColor = OpVariable %ptrOutputVec4 Output

; Decorations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO_t Block
; Offsets inside the std140 UBO
OpMemberDecorate %UBO_t 0 Offset 0
OpMemberDecorate %UBO_t 1 Offset 64
OpMemberDecorate %UBO_t 2 Offset 128
OpMemberDecorate %UBO_t 3 Offset 192
OpMemberDecorate %UBO_t 4 Offset 208
OpMemberDecorate %UBO_t 5 Offset 224

; Basic types for convenience
%ptrDB = OpTypePointer Uniform %float
%uint = OpTypeInt 32 0
%zero = OpConstant %float 0.0
%one  = OpConstant %float 1.0
%three = OpConstant %float 3.0
%onePointFive = OpConstant %float 1.5
%floatTwo = OpConstant %float 2.0
%float0Dot7 = OpConstant %float 0.7
%float0Dot3 = OpConstant %float 0.3
%float0Dot8 = OpConstant %float 0.8
%float0Dot2 = OpConstant %float 0.2
%floatDist = OpConstant %float 3.0
%uint3 = OpConstant %uint 3
%uint4 = OpConstant %uint 4

; Entry point and function type
%void = OpTypeVoid
%funcType = OpTypeFunction %void
%main = OpFunction %void None %funcType
%entry = OpLabel

; Load worldPos and normal
%worldPos = OpLoad %vec3 %in_worldPos
%N = OpLoad %vec3 %in_normal
; Normalize normal
%N_norm = OpExtInst %vec3 %GLSL Normalize %N

; Load lightPos from UBO (member 3)
%lightPos_ptr = OpAccessChain %ptrDB %ubo %uint3
%lightPos = OpLoad %vec4 %lightPos_ptr
%lp_x = OpCompositeExtract %float %lightPos 0
%lp_y = OpCompositeExtract %float %lightPos 1
%lp_z = OpCompositeExtract %float %lightPos 2
%lightPos_xyz = OpCompositeConstruct %vec3 %lp_x %lp_y %lp_z

; Compute L = normalize(lightPos.xyz - worldPos)
%toLight = OpVectorSub %vec3 %lightPos_xyz %worldPos
%L = OpExtInst %vec3 %GLSL Normalize %toLight

; Diffuse term: dot(N, L) clamped to >= 0
%dotNL = OpExtInst %float %GLSL Dot %N_norm %L
%diffuse = OpExtInst %float %GLSL FMax %dotNL %zero

; Lambertian factor: 0.3 + 0.7 * diffuse
%mul7 = OpExtInst %float %GLSL FMul %diffuse %float0Dot7
%lambert = OpFAdd %float %mul7 %float0Dot3

; Base color (0.8, 0.2, 0.2) * lambert
%baseColor3 = OpCompositeConstruct %vec3 %float0Dot8 %float0Dot2 %float0Dot2
%baseColor = OpVectorTimesScalar %vec3 %baseColor3 %lambert

; Fog parameters
; cameraPos from UBO (member 4)
%camPos_ptr = OpAccessChain %ptrDB %ubo %uint4
%camPos4 = OpLoad %vec4 %camPos_ptr
%cam_x = OpCompositeExtract %float %camPos4 0
%cam_y = OpCompositeExtract %float %camPos4 1
%cam_z = OpCompositeExtract %float %camPos4 2
%camPos_xyz = OpCompositeConstruct %vec3 %cam_x %cam_y %cam_z

; Distances
%diffVec = OpVectorSub %vec3 %camPos_xyz %worldPos
%dist = OpExtInst %float %GLSL Length %diffVec

; fogFactor = clamp((dist - 1.5)/3.0, 0.0, 1.0)
%distMinus = OpFSub %float %dist %onePointFive
%fogRaw = OpFDiv %float %distMinus %floatDist
%fogFactor = OpExtInst %float %GLSL FClamp %fogRaw %zero %one

; Fog color (0.7, 0.7, 0.8)
%fogColor3 = OpCompositeConstruct %vec3 %float0Dot7 %float0Dot7 %float0Dot8

; Mix: finalColor3 = (1 - fogFactor) * baseColor + fogFactor * fogColor
%invFog = OpFSub %float %one %fogFactor
%termA = OpVectorTimesScalar %vec3 %baseColor %invFog
%termB = OpVectorTimesScalar %vec3 %fogColor3 %fogFactor
%finalColor3 = OpVectorAdd %vec3 %termA %termB
%finalColor4 = OpCompositeConstruct %vec4 %finalColor3 %one

; Store to output
OpStore %outColor %finalColor4

OpReturn
OpFunctionEnd

; Entry point and execution mode
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor
OpExecutionMode %main OriginUpperLeft
