Fragment shader code (GLSL):

#version 450
layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;

layout(location = 0) out vec4 outColor;

const vec3 brickColor = vec3(0.7, 0.2, 0.1);
const vec3 mortarColor = vec3(0.7, 0.7, 0.7);
const vec3 lightDir = normalize(vec3(0.5, 0.8, 0.4)); // simple directional light

void main() {
    // Brick pattern in UV space
    // Scale UV to tile the 8x16 brick layout
    vec2 scale = vec2(8.0, 16.0);
    vec2 uvScaled = uv * scale;

    // Determine brick offset for staggered rows
    float row = floor(uvScaled.y);
    float offset = mod(row, 2.0) * 0.5; // half-brick offset for odd rows
    vec2 cellPos = vec2(uvScaled.x + offset, uvScaled.y);
    vec2 cellIndex = floor(cellPos);
    vec2 fracPos = fract(cellPos);

    // Mortar thickness in UV units (tuned for a visual brick look)
    // These values produce visible mortar lines between bricks.
    float mortarX = 0.05;
    float mortarY = 0.10;

    // Compute mortar mask: if we are within mortar lines around brick borders
    // For vertical mortar lines, check x within mortar region of each brick cell
    float inVerticalMortar = step(fracPos.x, mortarX) + step(1.0 - mortarX, fracPos.x);
    // For horizontal mortar lines, check y within mortar region between rows
    float inHorizontalMortar = step(fracPos.y, mortarY) + step(1.0 - mortarY, fracPos.y);

    // Combine to decide color region
    float isMortar = max(inVerticalMortar, inHorizontalMortar);

    // Brick color region
    vec3 brickRegion = brickColor;

    // Final color before lighting: mortar or brick
    vec3 baseColor = mix(brickRegion, mortarColor, isMortar);

    // Lambertian shading using world-space normal and a fixed light direction
    vec3 N = normalize(normal);
    vec3 L = normalize(lightDir);
    float NdotL = max(dot(N, L), 0.0);

    vec3 litColor = baseColor * (0.25 + 0.75 * NdotL); // ambient + diffuse

    outColor = vec4(litColor, 1.0);
}
