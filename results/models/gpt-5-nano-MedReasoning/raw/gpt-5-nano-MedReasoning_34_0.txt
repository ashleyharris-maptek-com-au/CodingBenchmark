```python
import sys
import math

def solve():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    try:
        N = int(next(it))
        K = int(next(it))
        CAP = float(next(it))
    except StopIteration:
        return

    depot_x = float(next(it))
    depot_y = float(next(it))

    customers = []
    for i in range(N):
        x = float(next(it))
        y = float(next(it))
        d = float(next(it))
        customers.append({'id': i, 'x': x, 'y': y, 'demand': d})

    # Helper: Euclidean distance
    def dist(a_x, a_y, b_x, b_y):
        return math.hypot(a_x - b_x, a_y - b_y)

    # Distance from depot to customer
    def depot_dist(cid):
        c = customers[cid]
        return dist(depot_x, depot_y, c['x'], c['y'])

    # Distance between two customers
    def cust_dist(a, b):
        ca = customers[a]
        cb = customers[b]
        return dist(ca['x'], ca['y'], cb['x'], cb['y'])

    # Order of visiting a given route by simple nearest-neighbor starting from depot
    def order_route_by_nearest(route_ids):
        if not route_ids:
            return []
        unvisited = set(route_ids)
        order = []
        curx, cury = depot_x, depot_y
        while unvisited:
            best = None
            best_d = float('inf')
            for cid in unvisited:
                c = customers[cid]
                d = dist(curx, cury, c['x'], c['y'])
                if d < best_d:
                    best_d = d
                    best = cid
            order.append(best)
            unvisited.remove(best)
            curx, cury = customers[best]['x'], customers[best]['y']
        return order

    # Distance of a route given a visiting order
    def route_distance(route_order):
        if not route_order:
            return 0.0
        total = 0.0
        # from depot to first
        first = customers[route_order[0]]
        total += dist(depot_x, depot_y, first['x'], first['y'])
        # between customers
        for i in range(len(route_order) - 1):
            a = route_order[i]
            b = route_order[i+1]
            total += dist(customers[a]['x'], customers[a]['y'], customers[b]['x'], customers[b]['y'])
        # last back to depot
        last = customers[route_order[-1]]
        total += dist(last['x'], last['y'], depot_x, depot_y)
        return total

    # Compute total distance for a set of routes (each route is a list of customer IDs)
    def compute_total(routes):
        total = 0.0
        orders = []
        for r in routes:
            ord_r = order_route_by_nearest(r)
            orders.append(ord_r)
            total += route_distance(ord_r)
        return total, orders

    # Partition a given order of IDs into routes respecting capacity
    def partition_by_capacity(order_ids):
        routes = []
        cur = []
        cur_cap = 0.0
        for cid in order_ids:
            d = customers[cid]['demand']
            if cur_cap + d <= CAP + 1e-9:
                cur.append(cid)
                cur_cap += d
            else:
                if cur:
                    routes.append(cur)
                cur = [cid]
                cur_cap = d
        if cur:
            routes.append(cur)
        return routes

    # Merge adjacent routes if their combined demand fits into capacity
    def merge_adjacent_routes(routes):
        if not routes:
            return []
        # First pass: try to merge adjacent pairs
        merged = []
        i = 0
        while i < len(routes):
            if i + 1 < len(routes):
                dsum = sum(customers[c]['demand'] for c in routes[i]) + sum(customers[c]['demand'] for c in routes[i+1])
                if dsum <= CAP + 1e-9:
                    merged.append(routes[i] + routes[i+1])
                    i += 2
                    continue
            merged.append(routes[i])
            i += 1
        # Repeat until no more merges possible
        changed = True
        while changed:
            changed = False
            if len(merged) <= 1:
                break
            newr = []
            i = 0
            while i < len(merged):
                if i + 1 < len(merged):
                    dsum = sum(customers[c]['demand'] for c in merged[i] + merged[i+1])
                    if dsum <= CAP + 1e-9:
                        newr.append(merged[i] + merged[i+1])
                        i += 2
                        changed = True
                        continue
                newr.append(merged[i])
                i += 1
            merged = newr
        return merged

    # Pack all IDs into exactly K bins if possible (greedy)
    def pack_to_k_bins(ids_order, Kbins):
        rem = [CAP] * Kbins
        bins = [[] for _ in range(Kbins)]
        for cid in ids_order:
            d = customers[cid]['demand']
            placed = False
            # choose the bin with enough room (prefer larger remaining)
            best_bin = None
            best_rem = -1.0
            for b in range(Kbins):
                if rem[b] + 1e-9 >= d:
                    if rem[b] > best_rem:
                        best_rem = rem[b]
                        best_bin = b
            if best_bin is not None:
                bins[best_bin].append(cid)
                rem[best_bin] -= d
                placed = True
            else:
                return None
        # remove empties
        return [b for b in bins if b]

    # If N == 0
    if N == 0:
        print("0")
        print("0")
        return

    # Prepare several strategy orders
    ids = [c['id'] for c in customers]
    strategies = []

    def add_strategy(order_ids, label):
        strategies.append((order_ids, label))

    add_strategy(ids, "id_order")

    add_strategy(sorted(ids, key=lambda i: customers[i]['demand'], reverse=True), "demand_desc")

    add_strategy(sorted(ids, key=lambda i: depot_dist(i)), "dist_to_depot_asc")

    add_strategy(sorted(ids, key=lambda i: math.atan2(customers[i]['y'] - depot_y, customers[i]['x'] - depot_x)), "angleAround")

    best_routes_final = None
    best_total = float('inf')

    for order_ids, label in strategies:
        # Partition by capacity
        initial_routes = partition_by_capacity(order_ids)
        # Try to merge to reduce route count
        merged = merge_adjacent_routes(initial_routes)
        candidate_routes = None

        if merged and len(merged) <= K:
            candidate_routes = merged
        else:
            # Try packing into K bins
            pack = pack_to_k_bins(order_ids, max(1, K))
            if pack is not None and len(pack) <= K and len(pack) > 0:
                candidate_routes = pack

        if candidate_routes is None:
            continue

        total, orders = compute_total(candidate_routes)
        if total < best_total:
            best_total = total
            best_routes_final = candidate_routes

    # Fallback if nothing found
    if best_routes_final is None or len(best_routes_final) == 0:
        # Try a simple greedy distribution into K bins (if K>0)
        if K > 0:
            rem = [CAP] * K
            bins = [[] for _ in range(K)]
            # simple order: by id
            for cid in ids:
                d = customers[cid]['demand']
                placed = False
                # try to fit in a bin with max remaining
                candidates = [(ri, rem[ri]) for ri in range(K) if rem[ri] >= d - 1e-9]
                if candidates:
                    ri = max(candidates, key=lambda t: t[1])[0]
                    bins[ri].append(cid)
                    rem[ri] -= d
                    placed = True
                # else: cannot place; skip (should not happen if feasible)
            routes_from_bins = [b for b in bins if b]
            if routes_from_bins:
                best_routes_final = routes_from_bins
                best_total, _ = compute_total(best_routes_final)

    if best_routes_final is None:
        # If still nothing, output zero
        print("0")
        print("0")
        return

    # Prepare final output
    total_distance = best_total
    final_routes = best_routes_final

    # Ensure we do not exceed K routes (safety)
    if len(final_routes) > max(0, K):
        # If somehow more, try to merge adjacent ones that fit
        final_routes = merge_adjacent_routes(final_routes)
        if len(final_routes) > max(0, K):
            final_routes = final_routes[:max(0, K)]

    print("{:.6f}".format(total_distance))
    print(len(final_routes))

    for route in final_routes:
        ordered = order_route_by_nearest(route)
        if not ordered:
            print()
        else:
            print(" ".join(str(cid) for cid in ordered))

if __name__ == "__main__":
    solve()
```
