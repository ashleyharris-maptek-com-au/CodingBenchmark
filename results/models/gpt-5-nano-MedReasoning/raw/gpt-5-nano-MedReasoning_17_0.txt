using System;
using System.Collections.Generic;
using System.Linq;

namespace BinPacking3D
{
    class Program
    {
        // Represents a box with original index and dimensions
        class Box
        {
            public int Id;
            public int W, H, D;
            public Box(int w, int h, int d, int id)
            {
                W = w; H = h; D = d; Id = id;
            }

            // Generate possible orientations that fit within container W x D base and height <= containerH
            public List<Orient> GetOrientations(int containerW, int containerD, int containerH)
            {
                var list = new List<Orient>();
                var seen = new HashSet<string>();

                void Add(int height, int bw, int bd)
                {
                    if (height <= containerH && bw <= containerW && bd <= containerD)
                    {
                        string key = height + ":" + bw + ":" + bd;
                        if (!seen.Contains(key))
                        {
                            seen.Add(key);
                            list.Add(new Orient { Height = height, Bw = bw, Bd = bd });
                        }
                    }
                }

                int w = W, h = H, d = D;
                // All 6 permutations compressed to height and two base dims
                Add(height: w, Bw: h, Bd: d);
                Add(height: w, Bw: d, Bd: h);

                Add(height: h, Bw: w, Bd: d);
                Add(height: h, Bw: d, Bd: w);

                Add(height: d, Bw: w, Bd: h);
                Add(height: d, Bw: h, Bd: w);

                // Sort by height descending, then by base area descending
                list.Sort((a, b) =>
                {
                    int dh = b.Height - a.Height;
                    if (dh != 0) return dh;
                    int areaA = a.Bw * a.Bd;
                    int areaB = b.Bw * b.Bd;
                    return areaB.CompareTo(areaA);
                });

                return list;
            }
        }

        // Orientation option for a box
        class Orient
        {
            public int Height;
            public int Bw;
            public int Bd;
        }

        // Free rectangle in a shelf (guillotine-style packing)
        class Rect
        {
            public int X;
            public int Z;
            public int W;
            public int D;
        }

        // Shelf along Y axis
        class Shelf
        {
            public int Y;
            public int Height;
            private int Wcap;
            private int Dcap;
            public List<Rect> Free;

            public Shelf(int y, int height, int W, int D)
            {
                Y = y;
                Height = height;
                Wcap = W;
                Dcap = D;
                Free = new List<Rect> { new Rect { X = 0, Z = 0, W = Wcap, D = Dcap } };
            }

            // Try to place a rectangle of size (bw x bd) on this shelf
            // Returns coordinates and the actual used footprint (which may be rotated)
            public bool TryPlace(int bw, int bd, out int x, out int z, out int usedW, out int usedD)
            {
                x = z = usedW = usedD = 0;
                for (int i = 0; i < Free.Count; i++)
                {
                    var r = Free[i];
                    // Try normal orientation
                    if (r.W >= bw && r.D >= bd)
                    {
                        x = r.X; z = r.Z;
                        usedW = bw; usedD = bd;
                        Split(i, r.X, r.Z, bw, bd);
                        return true;
                    }
                    // Try rotated in 2D plane
                    if (r.W >= bd && r.D >= bw)
                    {
                        x = r.X; z = r.Z;
                        usedW = bd; usedD = bw;
                        Split(i, r.X, r.Z, bd, bw);
                        return true;
                    }
                }
                return false;
            }

            // Split the free rectangle after placing at (x,z) with footprint (bw x bd)
            // Using right and top residuals to avoid overlap
            private void Split(int idx, int x, int z, int bw, int bd)
            {
                var rect = Free[idx];
                Free.RemoveAt(idx);
                // Right region to the right of placed rectangle
                if (rect.W - bw > 0 && bd > 0)
                {
                    Free.Add(new Rect { X = x + bw, Z = z, W = rect.W - bw, D = bd });
                }
                // Top region above the placed rectangle
                if (rect.D - bd > 0)
                {
                    Free.Add(new Rect { X = x, Z = z + bd, W = bw, D = rect.D - bd });
                }
                Prune();
            }

            // Remove any free rectangles that are contained within another
            private void Prune()
            {
                // Simple but effective pruning
                for (int i = 0; i < Free.Count; i++)
                {
                    var A = Free[i];
                    bool removed = false;
                    for (int j = Free.Count - 1; j > i; j--)
                    {
                        var B = Free[j];
                        if (Contains(A, B))
                        {
                            // A inside B -> remove A
                            Free.RemoveAt(i);
                            i--;
                            removed = true;
                            break;
                        }
                        else if (Contains(B, A))
                        {
                            // B inside A -> remove B
                            Free.RemoveAt(j);
                        }
                    }
                    if (removed) { /* continue with next i */ }
                }
            }

            private bool Contains(Rect inner, Rect outer)
            {
                return inner.X >= outer.X && inner.Z >= outer.Z
                    && inner.X + inner.W <= outer.X + outer.W
                    && inner.Z + inner.D <= outer.Z + outer.D;
            }
        }

        // A placed box result
        class Placement
        {
            public int BoxIndex;
            public int X;
            public int Y;
            public int Z;
        }

        static void Main(string[] args)
        {
            // Read input
            var first = ReadIntsLine();
            if (first == null || first.Length < 4)
                return;

            int N = first[0];
            int W = first[1];
            int H = first[2];
            int D = first[3];

            var boxes = new List<Box>();
            for (int i = 0; i < N; i++)
            {
                var line = ReadIntsLine();
                if (line == null || line.Length < 3) break;
                int w = line[0], h = line[1], d = line[2];
                boxes.Add(new Box(w, h, d, i));
            }

            // Sort boxes by overall size (max dimension) descending to place tall/large first
            boxes.Sort((a, b) =>
            {
                int ma = Math.Max(a.W, Math.Max(a.H, a.D));
                int mb = Math.Max(b.W, Math.Max(b.H, b.D));
                return mb.CompareTo(ma);
            });

            // Shelf-based packing
            var shelves = new List<Shelf>();
            int usedHeight = 0;
            var placements = new List<Placement>();

            foreach (var box in boxes)
            {
                // Generate orientations that fit container bounds
                var orients = box.GetOrientations(W, D, H);
                if (orients.Count == 0)
                    continue; // cannot place at all

                bool placedForThisBox = false;

                // Try orientations in order
                foreach (var o in orients)
                {
                    // Try to place on existing shelves
                    foreach (var s in shelves)
                    {
                        if (o.Height <= s.Height)
                        {
                            int x, z, uw, ud;
                            if (s.TryPlace(o.Bw, o.Bd, out x, out z, out uw, out ud))
                            {
                                placements.Add(new Placement { BoxIndex = box.Id, X = x, Y = s.Y, Z = z });
                                placedForThisBox = true;
                                break;
                            }
                        }
                    }
                    if (placedForThisBox) break;

                    // If not placed yet, try to place on a new shelf
                    if (usedHeight + o.Height <= H)
                    {
                        var newShelf = new Shelf(usedHeight, o.Height, W, D);
                        int x, z, uw, ud;
                        if (newShelf.TryPlace(o.Bw, o.Bd, out x, out z, out uw, out ud))
                        {
                            shelves.Add(newShelf);
                            usedHeight += newShelf.Height;
                            placements.Add(new Placement { BoxIndex = box.Id, X = x, Y = newShelf.Y, Z = z });
                            placedForThisBox = true;
                            break;
                        }
                    }

                    // If not placed in this orientation, continue to next orientation
                }

                // If not placed in any orientation, skip this box
            }

            // Output results
            Console.WriteLine(placements.Count);
            foreach (var p in placements)
            {
                Console.WriteLine($"{p.BoxIndex} {p.X} {p.Y} {p.Z}");
            }
        }

        // Helpers
        static int[] ReadIntsLine()
        {
            var line = Console.ReadLine();
            if (line == null) return null;
            line = line.Trim();
            if (line.Length == 0) return null;
            return line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries)
                       .Select(s => int.Parse(s))
                       .ToArray();
        }

        // Read line and parse integers (alternative method if needed)
        static int[] ReadIntsLineNullable(string s)
        {
            if (string.IsNullOrWhiteSpace(s)) return null;
            return s.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries)
                    .Select(str => int.Parse(str))
                    .ToArray();
        }
    }
}
