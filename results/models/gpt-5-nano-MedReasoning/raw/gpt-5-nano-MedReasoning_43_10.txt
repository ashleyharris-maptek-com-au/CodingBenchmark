struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize world-space normal
    float3 N = normalize(input.normal);

    // Compute light direction from world position to light position
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Lambertian diffuse
    float diffuse = max(dot(N, L), 0.0);

    // Lambertian shading factor
    float lambert = 0.4 + 0.6 * diffuse;

    // Horizontal stripes driven by sin(uv.y * pi * 16)
    float s = sin(input.uv.y * 3.14159f * 16.0f);
    float3 red  = float3(0.9, 0.1, 0.1);
    float3 blue = float3(0.1, 0.1, 0.9);
    float3 stripeColor = (s >= 0.0) ? red : blue;

    // Apply Lambert shading
    float3 color = stripeColor * lambert;

    return float4(color, 1.0);
}
