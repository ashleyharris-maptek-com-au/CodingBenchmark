import sys

def solve():
    data = sys.stdin.read().splitlines()
    if not data:
        print("NO SOLUTION")
        return

    first = data[0].strip().split()
    if len(first) < 2:
        print("NO SOLUTION")
        return

    try:
        U = int(first[0])
        S = int(first[1])
    except ValueError:
        print("NO SOLUTION")
        return

    sets = []
    # Read up to S lines of sets
    for i in range(1, 1 + S):
        if i < len(data):
            line = data[i].strip()
        else:
            line = ""
        if line == "":
            elements = []
        else:
            elements = line.split()
        uniq = []
        seen = set()
        for t in elements:
            try:
                e = int(t)
            except ValueError:
                continue
            if 0 <= e < U and e not in seen:
                seen.add(e)
                uniq.append(e)
        sets.append(uniq)

    # If fewer lines than S were provided, pad with empty sets
    while len(sets) < S:
        sets.append([])

    # Build element -> sets mapping
    elem_to_sets = [[] for _ in range(U)]
    for s, arr in enumerate(sets):
        for e in arr:
            if 0 <= e < U:
                elem_to_sets[e].append(s)

    covered = [False] * U
    chosen = []
    sys.setrecursionlimit(1000000)

    def search(covered_count):
        if covered_count == U:
            return True

        best_e = -1
        best_candidates = None
        min_len = None

        for e in range(U):
            if not covered[e]:
                cand = []
                for s in elem_to_sets[e]:
                    ok = True
                    for x in sets[s]:
                        if covered[x]:
                            ok = False
                            break
                    if ok:
                        cand.append(s)
                l = len(cand)
                if l == 0:
                    return False
                if min_len is None or l < min_len or (l == min_len and e < best_e):
                    min_len = l
                    best_e = e
                    best_candidates = cand
                    if l == 1:
                        break

        if best_candidates is None:
            return False

        for s in best_candidates:
            chosen.append(s)
            changed = []
            for x in sets[s]:
                if not covered[x]:
                    covered[x] = True
                    changed.append(x)
            if search(covered_count + len(changed)):
                return True
            for x in changed:
                covered[x] = False
            chosen.pop()

        return False

    ok = search(0)

    if ok:
        # Output solution
        print("SOLUTION")
        print(" ".join(str(idx) for idx in chosen))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    solve()
