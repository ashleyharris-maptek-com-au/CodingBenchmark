; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 999
; Schema: 0

OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%f32 = OpTypeFloat 32
%u32 = OpTypeInt 32 0
%v2f = OpTypeVector %f32 2
%v3f = OpTypeVector %f32 3
%v4f = OpTypeVector %f32 4
%mat4 = OpTypeMatrix %v4f 4

; UBO struct
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
%UBOPtr = OpTypePointer Uniform %UBO
; Member pointer type for accessing UBO fields (vec4)
%v4f_ptr_uniform = OpTypePointer Uniform %v4f

; Input/Output types
%v3f_ptr_in = OpTypePointer Input %v3f
%v2f_ptr_in = OpTypePointer Input %v2f
%v4f_ptr_out = OpTypePointer Output %v4f

; Uniform pointer to block
; Base UBO variable will be declared below as %ubo
; Access chains will use %v4f_ptr_uniform with indices
; Also need a pointer to the UBO base for OpAccessChain
%Zero = OpConstant %f32 0.0

; Variables
; Uniform block
%ubo = OpVariable %UBOPtr Uniform

; Inputs
%worldPos = OpVariable %v3f_ptr_in Input
%normalIn = OpVariable %v3f_ptr_in Input
%uvIn = OpVariable %v2f_ptr_in Input
%tangentIn = OpVariable %v3f_ptr_in Input
%colorIn = OpVariable %v3f_ptr_in Input

; Output
$outColor = OpVariable %v4f_ptr_out Output

; Offsets and Decorations for UBO
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Decorations for inputs/outputs
OpDecorate %worldPos Location 0
OpDecorate %normalIn Location 1
OpDecorate %uvIn Location 2
OpDecorate %tangentIn Location 3
OpDecorate %colorIn Location 4
OpDecorate %outColor Location 0
OpName %main "main"

; Entry point declaration
OpEntryPoint Fragment %main "main" %worldPos %normalIn %uvIn %tangentIn %colorIn %outColor
OpExecutionMode %main OriginUpperLeft

; Anticipated main function type
%func = OpTypeFunction %void

; Helper: constants
%c01 = OpConstant %f32 0.1
%c06 = OpConstant %f32 0.6
%c08 = OpConstant %f32 0.8
%zeroF = OpConstant %f32 0.0
%oneF = OpConstant %f32 1.0
%powExp = OpConstant %f32 64.0

; Indices for OpAccessChain
%idx0 = OpConstant %u32 0
%idx3 = OpConstant %u32 3
%idx4 = OpConstant %u32 4

; Preamble for uniforms and needed pointers
%ubo_ptr_in = OpTypePointer Uniform %UBO
%lightPos_ptr = OpTypePointer Uniform %v4f
%camPos_ptr   = OpTypePointer Uniform %v4f

; Begin main
%main = OpFunction %void None %func
%entry = OpLabel

; Load world position P
%P = OpLoad %v3f %worldPos

; Load and normalize N
%N = OpLoad %v3f %normalIn
%NdotN = OpDot %f32 %N %N
%Nlen = OpSqrt %f32 %NdotN
%Nnorm = OpVectorTimesScalar %v3f %N %Nlen

; Load L = normalize(lightPos - P)
%lightPos_p = OpAccessChain %lightPos_ptr %ubo %idx3
%lightPos4 = OpLoad %v4f %lightPos_p
%Lx = OpCompositeExtract %f32 %lightPos4 0
%Ly = OpCompositeExtract %f32 %lightPos4 1
%Lz = OpCompositeExtract %f32 %lightPos4 2
%L = OpCompositeConstruct %v3f %Lx %Ly %Lz
%Ldiff = OpVectorSub %v3f %L %P
%Ldiff_len2 = OpDot %f32 %Ldiff %Ldiff
%Llen = OpSqrt %f32 %Ldiff_len2
%Lvec = OpVectorTimesScalar %v3f %Ldiff %Llen

; Load V = normalize(camPos - P)
%camPos_p = OpAccessChain %lightPos_ptr %ubo %idx4
%camPos4 = OpLoad %v4f %camPos_p
%Cpx = OpCompositeExtract %f32 %camPos4 0
%Cpy = OpCompositeExtract %f32 %camPos4 1
%Cpz = OpCompositeExtract %f32 %camPos4 2
%Cpos = OpCompositeConstruct %v3f %Cpx %Cpy %Cpz
%Vdiff = OpVectorSub %v3f %Cpos %P
%Vdiff_len2 = OpDot %f32 %Vdiff %Vdiff
%Vlen = OpSqrt %f32 %Vdiff_len2
%Vvec = OpVectorTimesScalar %v3f %Vdiff %Vlen

; H = normalize(L + V)
%LplusV = OpVectorAdd %v3f %Lvec %Vvec
%Hlen2 = OpDot %f32 %LplusV %LplusV
%Hlen = OpSqrt %f32 %Hlen2
%H = OpVectorTimesScalar %v3f %LplusV %Hlen

; Diffuse term: max(dot(N, L), 0)
%NdotL = OpDot %f32 %Nnorm %Lvec
%condNL = OpFOrdGreaterThan %bool %NdotL %zeroF
%diffuse = OpSelect %f32 %condNL %NdotL %zeroF

; Specular term: pow(max(dot(N, H), 0), 64)
%NH = OpDot %f32 %Nnorm %H
%condNH = OpFOrdGreaterThan %bool %NH %zeroF
%NH_clamped = OpSelect %f32 %condNH %NH %zeroF
%spec = OpPow %f32 %NH_clamped %powExp

; Intensity = 0.1 + 0.6*diffuse + 0.8*spec
%diffTerm = OpFMul %f32 %diffuse %c06
$specTerm = OpFMul %f32 %spec %c08
%sum = OpFAdd %f32 %diffTerm %specTerm
%intensity = OpFAdd %f32 %sum %c01

; Output color: grayscale with alpha 1.0
%outCol3 = OpCompositeConstruct %v3f %intensity %intensity %intensity
%outCol4 = OpCompositeConstruct %v4f %intensity %intensity %intensity %oneF
OpStore %outColor %outCol4

OpReturn
OpFunctionEnd
