using System;
using System.IO;
using System.Text;
using System.Diagnostics;

class FastScanner
{
    private readonly Stream _stream;
    private readonly byte[] _buffer = new byte[1 << 16];
    private int _len = 0, _ptr = 0;

    public FastScanner(Stream s = null)
    {
        _stream = s ?? Console.OpenStandardInput();
    }

    private int Read()
    {
        if (_ptr >= _len)
        {
            _len = _stream.Read(_buffer, 0, _buffer.Length);
            _ptr = 0;
            if (_len <= 0) return -1;
        }
        return _buffer[_ptr++];
    }

    public long NextLong()
    {
        int c;
        do
        {
            c = Read();
        } while (c <= ' ' && c != -1);

        bool neg = false;
        if (c == '-')
        {
            neg = true;
            c = Read();
        }

        long val = 0;
        while (c > ' ')
        {
            val = val * 10 + (c - '0');
            c = Read();
        }
        return neg ? -val : val;
    }

    public int NextInt()
    {
        return (int)NextLong();
    }
}

class Program
{
    static void Main(string[] args)
    {
        var scanner = new FastScanner();
        int N = scanner.NextInt();
        long[,] F = new long[N, N];
        long[,] D = new long[N, N];

        // Read flow matrix
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                F[i, j] = scanner.NextLong();

        // Read distance matrix
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                D[i, j] = scanner.NextLong();

        // Heuristic initial permutation
        long[] rowSumF = new long[N];
        long[] rowSumD = new long[N];
        for (int i = 0; i < N; i++)
        {
            long s = 0;
            for (int j = 0; j < N; j++) s += F[i, j];
            rowSumF[i] = s;
        }
        for (int i = 0; i < N; i++)
        {
            long s = 0;
            for (int j = 0; j < N; j++) s += D[i, j];
            rowSumD[i] = s;
        }

        int[] facOrder = new int[N];
        for (int i = 0; i < N; i++) facOrder[i] = i;
        Array.Sort(facOrder, (a, b) => rowSumF[b].CompareTo(rowSumF[a])); // descending

        int[] locOrder = new int[N];
        for (int i = 0; i < N; i++) locOrder[i] = i;
        Array.Sort(locOrder, (a, b) => rowSumD[a].CompareTo(rowSumD[b])); // ascending

        int[] P = new int[N];
        for (int t = 0; t < N; t++) P[facOrder[t]] = locOrder[t];

        // Compute initial cost
        long cost = 0;
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                cost += F[i, j] * D[P[i], P[j]];
            }
        }

        // Randomized local improvement with time bound
        Random rng = new Random(123456);
        Stopwatch sw = Stopwatch.StartNew();
        const int TimeLimitMs = 4 * 60 * 1000; // ~4 minutes
        int m = Math.Min(32, Math.Max(0, N - 2)); // number of samples per test

        long bestCost = cost;
        int[] bestP = (int[])P.Clone();

        // If N is small, still allow some improvements
        while (sw.ElapsedMilliseconds < TimeLimitMs)
        {
            int a = rng.Next(N);
            int b = rng.Next(N);
            if (a == b) continue;
            int la = P[a], lb = P[b];

            long deltaPartial = 0;
            if (m > 0)
            {
                for (int s = 0; s < m; s++)
                {
                    int k = rng.Next(N);
                    if (k == a || k == b) { s--; continue; }

                    deltaPartial += F[a, k] * (D[lb, P[k]] - D[la, P[k]]);
                    deltaPartial += F[k, a] * (D[P[k], lb] - D[P[k], la]);
                    deltaPartial += F[b, k] * (D[la, P[k]] - D[lb, P[k]]);
                    deltaPartial += F[k, b] * (D[P[k], la] - D[P[k], lb]);
                }
            }

            // delta for the pair (a,b)
            deltaPartial += F[a, b] * (D[lb, la] - D[la, lb]);
            deltaPartial += F[b, a] * (D[la, lb] - D[lb, la]);

            if (deltaPartial < 0)
            {
                // Compute exact delta
                long deltaFull = 0;
                for (int k = 0; k < N; k++)
                {
                    deltaFull += F[a, k] * (D[lb, P[k]] - D[la, P[k]]);
                    deltaFull += F[k, a] * (D[P[k], lb] - D[P[k], la]);
                    deltaFull += F[b, k] * (D[la, P[k]] - D[lb, P[k]]);
                    deltaFull += F[k, b] * (D[P[k], la] - D[P[k], lb]);
                }
                deltaFull += F[a, b] * (D[lb, la] - D[la, lb]);
                deltaFull += F[b, a] * (D[la, lb] - D[lb, la]);

                if (deltaFull < 0)
                {
                    // Apply swap
                    int tmp = P[a];
                    P[a] = P[b];
                    P[b] = tmp;
                    cost += deltaFull;

                    if (cost < bestCost)
                    {
                        bestCost = cost;
                        bestP = (int[])P.Clone();
                    }
                }
            }
        }

        // Output best found
        // If no improvement found, bestP remains initial
        cost = bestCost;
        int[] outputP = bestP;

        Console.WriteLine(cost.ToString());
        var sb = new StringBuilder();
        for (int i = 0; i < N; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(outputP[i]);
        }
        Console.WriteLine(sb.ToString());
    }
}
