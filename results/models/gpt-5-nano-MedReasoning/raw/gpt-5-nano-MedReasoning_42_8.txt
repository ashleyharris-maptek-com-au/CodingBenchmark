; SPIR-V assembly for a geometry shader that shrinks each input triangle toward its centroid
; and emits three vertices (triangle strip with 3 vertices).
; Geometry shader requirements satisfied:
; - OpCapability Geometry
; - OpMemoryModel Logical GLSL450
; - OpEntryPoint Geometry %main "main" listing ALL interface variables
; - OpExecutionMode %main triangles (InputTriangles)
; - OpExecutionMode %main OutputTriangleStrip
; - OpExecutionMode %main OutputVertices 3
; - OpExecutionMode %main Invocations 1

OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3

; Structures for gl_PerVertex_in/out
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Basic integer/array types for indices and array sizes
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3

; Convenience constants
%float_33333334 = OpConstant %float 0.33333334
%float_7 = OpConstant %float 0.7

; Input: per-vertex gl_Position via gl_in
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
; Access: OpAccessChain %ptr_in_v4 %gl_in %int_i %int_0

; Input arrays (may be unused for the computation but must be declared and listed in interface)
%arr_worldPos_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_worldPos = OpTypePointer Input %arr_worldPos_in
%worldPos = OpVariable %ptr_in_arr_worldPos Input

%arr_normal_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_normal = OpTypePointer Input %arr_normal_in
%normal = OpVariable %ptr_in_arr_normal Input

; Input array of colors (used per-vertex)
%arr_color_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_color = OpTypePointer Input %arr_color_in
%color = OpVariable %ptr_in_arr_color Input
; Access: OpAccessChain %ptr_in_color_vec3 %color %int_i

; Output: per-vertex gl_Position via gl_out
%gl_out = OpVariable (OpTypePointer Output %gl_PerVertex_out) Output
%ptr_out_v4 = OpTypePointer Output %v4float
; Output: per-vertex color (to fragment shader)
%outColor = OpVariable (OpTypePointer Output %v3float) Output

; Uniform Block (UBO) - declared for accessibility (layout matches test 41 as requested)
%mat4float = OpTypeMatrix %v4float 4
%ubo_block_type = OpTypeStruct %mat4float
%ptr_ubo = OpTypePointer Uniform %ubo_block_type
%ubo = OpVariable %ptr_ubo Uniform

; Entry points and interface
%func_type = OpTypeFunction %void
%main = OpFunction %void None %func_type
%entry = OpLabel

; Geometry shader will shrink each vertex position toward centroid of the triangle
; Load the 3 input positions from gl_in
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr

; Centroid C = (p0 + p1 + p2) / 3
%sum01 = OpFAdd %v4float %p0 %p1
%sum = OpFAdd %v4float %sum01 %p2
%centroid = OpVectorTimesScalar %v4float %sum %float_33333334

; Shrink factor s = 0.7
; new_i = centroid + s * (p_i - centroid)
%delta0 = OpFSub %v4float %p0 %centroid
%scaled0 = OpVectorTimesScalar %v4float %delta0 %float_7
%new0 = OpFAdd %v4float %centroid %scaled0

%delta1 = OpFSub %v4float %p1 %centroid
%scaled1 = OpVectorTimesScalar %v4float %delta1 %float_7
%new1 = OpFAdd %v4float %centroid %scaled1

%delta2 = OpFSub %v4float %p2 %centroid
%scaled2 = OpVectorTimesScalar %v4float %delta2 %float_7
%new2 = OpFAdd %v4float %centroid %scaled2

; Emit vertex 0
%gl_out_ptr0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_ptr0 %new0

; Emit vertex 0 color
%color0_ptr = OpAccessChain %ptr_in_color_vec3 %color %int_0
%color0 = OpLoad %v3float %color0_ptr
OpStore %outColor %color0
OpEmitVertex

; Emit vertex 1
%gl_out_ptr1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_ptr1 %new1

%color1_ptr = OpAccessChain %ptr_in_color_vec3 %color %int_1
%color1 = OpLoad %v3float %color1_ptr
OpStore %outColor %color1
OpEmitVertex

; Emit vertex 2
%gl_out_ptr2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_ptr2 %new2

%color2_ptr = OpAccessChain %ptr_in_color_vec3 %color %int_2
%color2 = OpLoad %v3float %color2_ptr
OpStore %outColor %color2
OpEmitVertex

; End primitive
OpEndPrimitive

OpReturn
OpFunctionEnd

; Interfaces for entry point (ALL used by the geometry shader)
OpEntryPoint Geometry %main "main"
  %gl_in
  %worldPos
  %normal
  %color
  %gl_out
  %outColor
