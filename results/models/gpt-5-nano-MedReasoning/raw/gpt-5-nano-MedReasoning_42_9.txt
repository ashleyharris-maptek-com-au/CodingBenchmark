; SPIR-V assembly for a geometry shader implementing a thick sine-curve using adjacency
; Target: Vulkan 1.0, GLSL450 style, fixed vertex/fragment pipeline
; Geometry shader requirements met:
;  - OpCapability Geometry
;  - OpMemoryModel Logical GLSL450
;  - OpEntryPoint Geometry %main "main" listing ALL interface variables
;  - OpExecutionMode %main InputLinesAdjacency
;  - OpExecutionMode %main OutputPrimitive TriangleStrip
;  - OpExecutionMode %main OutputVertices 4
;  - OpExecutionMode %main Invocations 1

               OpCapability Geometry
               OpCapability Shader
               OpMemoryModel Logical GLSL450

; Types
%void      = OpTypeVoid
%float     = OpTypeFloat 32
%v2float   = OpTypeVector %float 2
%v3float   = OpTypeVector %float 3
%v4float   = OpTypeVector %float 4
%uint      = OpTypeInt 32 0

; Extensions (for GLSL std450 normalize)
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Constants
%uint_0    = OpConstant %uint 0
%uint_1    = OpConstant %uint 1
%uint_2    = OpConstant %uint 2
%uint_3    = OpConstant %uint 3
%uint_4    = OpConstant %uint 4

%float_0   = OpConstant %float 0.0
%float_1   = OpConstant %float 1.0
%float_025 = OpConstant %float 0.025

; gl_PerVertex_in (vertex shader input block)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; gl_PerVertex_out (geometry shader output block)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

; Per-vertex output color (to fragment shader)
%outColor_ptr = OpTypePointer Output %v3float
%outColor = OpVariable %outColor_ptr Output
; Interface: outColor will be sent to fragment shader as layout(location = 0) out vec3 outColor

; Input arrays from vertex shader (per-vertex attributes, same length as gl_in)
; worldPos[]  (unused in logic, kept for compatibility)
%arr_vec3 = OpTypeArray %v3float %uint_4
%ptr_in_worldPos = OpTypePointer Input %arr_vec3
%in_worldPos = OpVariable %ptr_in_worldPos Input
; normal[]
%ptr_in_normal = OpTypePointer Input %arr_vec3
%in_normal = OpVariable %ptr_in_normal Input
; color[]
%ptr_in_color = OpTypePointer Input %arr_vec3
%in_color = OpVariable %ptr_in_color Input
; Pointer type for color element
%ptr_in_vec3 = OpTypePointer Input %v3float

; Interface declarations for entry point
; Function and entry point declaration
%void_fn = OpTypeFunction %void

; main entry point
%main = OpFunction %void None %void_fn
%lab  = OpLabel

; Load the four gl_in positions (gl_in[i].gl_Position, i = 0..3)
%pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_0 %uint_0
%ptr_in_v4 = OpTypePointer Input %v4float
%pos0 = OpLoad %v4float %pos0_ptr

%pos1_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_1 %uint_0
%pos1 = OpLoad %v4float %pos1_ptr

%pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_2 %uint_0
%pos2 = OpLoad %v4float %pos2_ptr

%pos3_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_3 %uint_0
%pos3 = OpLoad %v4float %pos3_ptr

; Extract xyz components
%pos0_xyz = OpCompositeExtract %v3float %pos0 0 1 2
%pos1_xyz = OpCompositeExtract %v3float %pos1 0 1 2
%pos2_xyz = OpCompositeExtract %v3float %pos2 0 1 2
%pos3_xyz = OpCompositeExtract %v3float %pos3 0 1 2

; Tangents
%delta0 = OpFSub %v3float %pos2_xyz %pos0_xyz
%t0 = OpExtInst %v3float %GLSLstd450 Normalize %delta0

%delta1 = OpFSub %v3float %pos3_xyz %pos1_xyz
%t1 = OpExtInst %v3float %GLSLstd450 Normalize %delta1

; Perpendiculars in XY plane
;t0 components
%t0_x = OpCompositeExtract %float %t0 0
%t0_y = OpCompositeExtract %float %t0 1
;%perp0 = vec3(-t0.y, t0.x, 0)
%neg_t0_y = OpFNegate %float %t0_y
%zero_0 = %float_0
%perp0 = OpCompositeConstruct %v3float %neg_t0_y %t0_x %zero_0

; t1 components
%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1
%neg_t1_y = OpFNegate %float %t1_y
%perp1 = OpCompositeConstruct %v3float %neg_t1_y %t1_x %zero_0

; Offset by thickness
%offset0 = OpVectorTimesScalar %v3float %perp0 %float_025
%offset1 = OpVectorTimesScalar %v3float %perp1 %float_025

; Start and end positions in xyz
%start_xyz = %pos1_xyz
%end_xyz   = %pos2_xyz

; Start/end +/- offsets
%start_plus = OpFAdd %v3float %start_xyz %offset0
%start_minus= OpFSub %v3float %start_xyz %offset0
%end_plus  = OpFAdd %v3float %end_xyz %offset1
%end_minus = OpFSub %v3float %end_xyz %offset1

; Convert to vec4 with w = 1.0
%start_plus_x = OpCompositeExtract %float %start_plus 0
%start_plus_y = OpCompositeExtract %float %start_plus 1
%start_plus_z = OpCompositeExtract %float %start_plus 2
%one = OpConstant %float 1.0
%start_plus_v4 = OpCompositeConstruct %v4float %start_plus_x %start_plus_y %start_plus_z %one

%start_minus_x = OpCompositeExtract %float %start_minus 0
%start_minus_y = OpCompositeExtract %float %start_minus 1
%start_minus_z = OpCompositeExtract %float %start_minus 2
%start_minus_v4 = OpCompositeConstruct %v4float %start_minus_x %start_minus_y %start_minus_z %one

%end_plus_x = OpCompositeExtract %float %end_plus 0
%end_plus_y = OpCompositeExtract %float %end_plus 1
%end_plus_z = OpCompositeExtract %float %end_plus 2
%end_plus_v4 = OpCompositeConstruct %v4float %end_plus_x %end_plus_y %end_plus_z %one

%end_minus_x = OpCompositeExtract %float %end_minus 0
%end_minus_y = OpCompositeExtract %float %end_minus 1
%end_minus_z = OpCompositeExtract %float %end_minus 2
%end_minus_v4 = OpCompositeConstruct %v4float %end_minus_x %end_minus_y %end_minus_z %one

; Load colors for start and end vertices
%color1_ptr = OpAccessChain %ptr_in_vec3 %in_color %uint_1
%color1 = OpLoad %v3float %color1_ptr

%color2_ptr = OpAccessChain %ptr_in_vec3 %in_color %uint_2
%color2 = OpLoad %v3float %color2_ptr

; Emit four vertices in a triangle strip:
; Vertex 0: start_plus, color1
OpStore %outColor %color1
%gl_out_pos_ptr0 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_out_pos_ptr0 %start_plus_v4
OpEmitVertex

; Vertex 1: start_minus, color1
OpStore %outColor %color1
%gl_out_pos_ptr1 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_out_pos_ptr1 %start_minus_v4
OpEmitVertex

; Vertex 2: end_plus, color2
OpStore %outColor %color2
%gl_out_pos_ptr2 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_out_pos_ptr2 %end_plus_v4
OpEmitVertex

; Vertex 3: end_minus, color2
OpStore %outColor %color2
%gl_out_pos_ptr3 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_out_pos_ptr3 %end_minus_v4
OpEmitVertex

; End primitive
OpEndPrimitive

; End function
OpReturn
OpFunctionEnd

; Execution mode and entry point
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

;  - gl_in (per-vertex input array)
;  - worldPos[], normal[], color[] (input arrays)
;  - gl_out (per-vertex output block with gl_Position)
;  - outColor (per-vertex color to fragment shader)
;  If your tool requires explicit interface listing, include the IDs above in the OpEntryPoint line accordingly.)
