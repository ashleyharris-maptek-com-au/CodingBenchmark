; SPIR-V Geometry Shader (Vulkan 1.0 compatible)
; Geometry shader that renders a thick Catmull-Rom-inspired sine-like curve
; Input primitive: InputLinesAdjacency (4 control points: gl_in[0..3])
; Output: a single triangle strip (10 vertices) with color interpolation
; Thickness: 0.02 (in clip-space units)
; Catmull-Rom tangents approximated by (P2-P0)/2 and (P3-P1)/2

OpCapability Geometry
OpMemoryModel Logical GLSL450

%GLSL = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0

; Constants
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_5 = OpConstant %float 0.5
%float_0_25 = OpConstant %float 0.25
%float_0_75 = OpConstant %float 0.75
%float_25 = OpConstant %float 0.25
%float_5 = OpConstant %float 0.5
%float_75 = OpConstant %float 0.75
%float_02 = OpConstant %float 0.02
%float_1_0 = OpConstant %float 1.0

; Types (aliases used for brevity)
%mat4 = OpTypeMatrix %v4float 4

; gl_in (per-vertex input block)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
; Access: OpAccessChain %ptr_in_v4 %gl_in %uint_i %uint_0

%ptr_in_v4 = OpTypePointer Input %v4float

; User inputs (per-vertex arrays)
%arr_vec3_4 = OpTypeArray %v3float %uint_4

%ptr_world_in = OpTypePointer Input %arr_vec3_4
%worldPos_in = OpVariable %ptr_world_in Input
OpDecorate %worldPos_in Location 0

%ptr_normal_in = OpTypePointer Input %arr_vec3_4
%normal_in = OpVariable %ptr_normal_in Input
OpDecorate %normal_in Location 1

%ptr_color_in = OpTypePointer Input %arr_vec3_4
%color_in = OpVariable %ptr_color_in Input
OpDecorate %color_in Location 2

; gl_PerVertex_out (per-vertex output block)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
; Member 0 of gl_PerVertex_out is BuiltIn Position (handled by the struct)

; Output variable: per-vertex color to fragment stage
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output
OpDecorate %outColor Location 0

; UBO (set=0, binding=0) - minimal layout to satisfy accessibility
%v3float_0 = OpConstant %v3float 0.0 0.0 0.0
%v4float_0 = OpConstant %v4float 0.0 0.0 0.0 0.0
; Mat4 matrix (MVP), vec3 lightPos, vec3 cameraPos, vec4 params
%UBOData = OpTypeStruct %mat4 %v3float %v3float %v4float
%ptr_UBO = OpTypePointer Uniform %UBOData
%UBO = OpVariable %ptr_UBO Uniform
OpDecorate %UBO DescriptorSet 0
OpDecorate %UBO Binding 0

; Pipeline entry point and interface
%void = OpTypeVoid
%func_type = OpTypeFunction %void

; Entry point
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %outColor

; Execution modes
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

; Function main
%main = OpFunction %void None %func_type
%label = OpLabel

; Load control points P0..P3 from gl_in
; P0
%p0x_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_0 %uint_0
%p0x4 = OpLoad %v4float %p0x_ptr
%x0 = OpCompositeExtract %float %p0x4 0
%y0 = OpCompositeExtract %float %p0x4 1
%z0 = OpCompositeExtract %float %p0x4 2
%P0 = OpCompositeConstruct %v3float %x0 %y0 %z0

; P1
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_1 %uint_0
%p1v4 = OpLoad %v4float %p1_ptr
%x1 = OpCompositeExtract %float %p1v4 0
%y1 = OpCompositeExtract %float %p1v4 1
%z1 = OpCompositeExtract %float %p1v4 2
%P1 = OpCompositeConstruct %v3float %x1 %y1 %z1

; P2
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_2 %uint_0
%p2v4 = OpLoad %v4float %p2_ptr
%x2 = OpCompositeExtract %float %p2v4 0
%y2 = OpCompositeExtract %float %p2v4 1
%z2 = OpCompositeExtract %float %p2v4 2
%P2 = OpCompositeConstruct %v3float %x2 %y2 %z2

; P3
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_3 %uint_0
%p3v4 = OpLoad %v4float %p3_ptr
%x3 = OpCompositeExtract %float %p3v4 0
%y3 = OpCompositeExtract %float %p3v4 1
%z3 = OpCompositeExtract %float %p3v4 2
%P3 = OpCompositeConstruct %v3float %x3 %y3 %z3

; Compute tangent = 0.5*(P2-P0) + 0.5*(P3-P1); then normalize
%diff20 = OpFSub %v3float %P2 %P0
%diff31 = OpFSub %v3float %P3 %P1
%half = %float_0_5
%half_diff20 = OpVectorTimesScalar %v3float %diff20 %half
%half_diff31 = OpVectorTimesScalar %v3float %diff31 %half
%tan_raw = OpFAdd %v3float %half_diff20 %half_diff31
%tan = OpExtInst %v3float %GLSL Normalize %tan_raw

; Up vector (0,0,1)
%up = OpConstantComposite %v3float %float_0 %float_0 %float_1

; Perpendicular direction = normalize(cross(tan, up))
%tan_cross_up = OpExtInst %v3float %GLSL Cross %tan %up
%perp = OpExtInst %v3float %GLSL Normalize %tan_cross_up
; Thickness
%thick = %float_0_02
%offset = OpVectorTimesScalar %v3float %perp %thick

; Preload colors
%col1_ptr = OpAccessChain %ptr_color_elem_in %color_in %uint_1
%color1 = OpLoad %v3float %col1_ptr
%col2_ptr = OpAccessChain %ptr_color_elem_in %color_in %uint_2
%color2 = OpLoad %v3float %col2_ptr

; t values for 5 samples
%t0 = %float_0
%t1 = %float_0_25
%t2 = %float_0_5
%t3 = %float_0_75
%t4 = %float_1_0
%one = %float_1
; 1 - t values
%omt0 = OpFSub %float %one %t0
%omt1 = OpFSub %float %one %t1
%omt2 = OpFSub %float %one %t2
%omt3 = OpFSub %float %one %t3
%omt4 = OpFSub %float %one %t4

; Pt0 = P1 (t=0)
%Pt0 = %P1
; Pt1
%tmpA1 = OpVectorTimesScalar %v3float %P1 %omt1
%tmpB1 = OpVectorTimesScalar %v3float %P2 %t1
%Pt1 = OpFAdd %v3float %tmpA1 %tmpB1
; Pt2
%tmpA2 = OpVectorTimesScalar %v3float %P1 %omt2
%tmpB2 = OpVectorTimesScalar %v3float %P2 %t2
%Pt2 = OpFAdd %v3float %tmpA2 %tmpB2
; Pt3
%tmpA3 = OpVectorTimesScalar %v3float %P1 %omt3
%tmpB3 = OpVectorTimesScalar %v3float %P2 %t3
%Pt3 = OpFAdd %v3float %tmpA3 %tmpB3
; Pt4
%Pt4 = %P2

; Helpers: function to emit a vertex (plus or minus direction)
; We'll define a small block of repeated code for each vertex
; We'll keep the same local approach: compute gl_Position from PtX +/- offset, assign color from tX, emit vertex

; Emit Pt0_plus
%Pt0_plus = OpFAdd %v3float %Pt0 %offset
%x0p = OpCompositeExtract %float %Pt0_plus 0
%y0p = OpCompositeExtract %float %Pt0_plus 1
%z0p = OpCompositeExtract %float %Pt0_plus 2
%glpos0p = OpCompositeConstruct %v4float %x0p %y0p %z0p %float_1
%glpos0p_ptr = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %glpos0p_ptr %glpos0p
; Color t0 = color1*(1-0) + color2*0
%col0_t = %color1
OpStore %outColor %col0_t
OpEmitVertex

; Emit Pt0_minus
%Pt0_minus = OpFSub %v3float %Pt0 %offset
%x0m = OpCompositeExtract %float %Pt0_minus 0
%y0m = OpCompositeExtract %float %Pt0_minus 1
%z0m = OpCompositeExtract %float %Pt0_minus 2
%glpos0m = OpCompositeConstruct %v4float %x0m %y0m %z0m %float_1
%glpos0m_ptr = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %glpos0m_ptr %glpos0m
OpStore %outColor %col0_t
OpEmitVertex

; Emit Pt1_plus
%Pt1_plus = OpFAdd %v3float %Pt1 %offset
%x1p = OpCompositeExtract %float %Pt1_plus 0
%y1p = OpCompositeExtract %float %Pt1_plus 1
%z1p = OpCompositeExtract %float %Pt1_plus 2
%glpos1p = OpCompositeConstruct %v4float %x1p %y1p %z1p %float_1
%glpos1p_ptr = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %glpos1p_ptr %glpos1p
; Color t1 = color1*omt1 + color2*t1
%col1_t1a = OpVectorTimesScalar %v3float %color1 % %omt1
%col1_t1b = OpVectorTimesScalar %v3float %color2 % %t1
%col1_t1 = OpFAdd %v3float %col1_t1a % %col1_t1b
OpStore %outColor %col1_t1
OpEmitVertex

; Pt1_minus
%Pt1_minus = OpFSub %v3float %Pt1 %offset
%x1m = OpCompositeExtract %float %Pt1_minus 0
%y1m = OpCompositeExtract %float %Pt1_minus 1
%z1m = OpCompositeExtract %float %Pt1_minus 2
%glpos1m = OpCompositeConstruct %v4float %x1m %y1m %z1m %float_1
%glpos1m_ptr = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %glpos1m_ptr %glpos1m
OpStore %outColor %col1_t1
OpEmitVertex

; Pt2_plus
%Pt2_plus = OpFAdd %v3float %Pt2 %offset
%x2p = OpCompositeExtract %float %Pt2_plus 0
%y2p = OpCompositeExtract %float %Pt2_plus 1
%z2p = OpCompositeExtract %float %Pt2_plus 2
%glpos2p = OpCompositeConstruct %v4float %x2p %y2p %z2p %float_1
%glpos2p_ptr = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %glpos2p_ptr %glpos2p
; Color t2
%col1_t2a = OpVectorTimesScalar %v3float %color1 % %omt2
%col1_t2b = OpVectorTimesScalar %v3float %color2 % %t2
%col2_t2 = OpFAdd %v3float %col1_t2a % %col1_t2b
OpStore %outColor %col2_t2
OpEmitVertex

; Pt2_minus
%Pt2_minus = OpFSub %v3float %Pt2 %offset
%x2m = OpCompositeExtract %float %Pt2_minus 0
%y2m = OpCompositeExtract %float %Pt2_minus 1
%z2m = OpCompositeExtract %float %Pt2_minus 2
%glpos2m = OpCompositeConstruct %v4float %x2m %y2m %z2m %float_1
%glpos2m_ptr = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %glpos2m_ptr %glpos2m
OpStore %outColor %col2_t2
OpEmitVertex

; Pt3_plus
%Pt3_plus = OpFAdd %v3float %Pt3 %offset
%x3p = OpCompositeExtract %float %Pt3_plus 0
%y3p = OpCompositeExtract %float %Pt3_plus 1
%z3p = OpCompositeExtract %float %Pt3_plus 2
%glpos3p = OpCompositeConstruct %v4float %x3p %y3p %z3p %float_1
%glpos3p_ptr = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %glpos3p_ptr %glpos3p
; Color t3
%col1_t3a = OpVectorTimesScalar %v3float %color1 % %omt3
%col1_t3b = OpVectorTimesScalar %v3float %color2 % %t3
%col3_t = OpFAdd %v3float %col1_t3a % %col1_t3b
OpStore %outColor %col3_t
OpEmitVertex

; Pt3_minus
%Pt3_minus = OpFSub %v3float %Pt3 %offset
%x3m = OpCompositeExtract %float %Pt3_minus 0
%y3m = OpCompositeExtract %float %Pt3_minus 1
%z3m = OpCompositeExtract %float %Pt3_minus 2
%glpos3m = OpCompositeConstruct %v4float %x3m %y3m %z3m %float_1
%glpos3m_ptr = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %glpos3m_ptr %glpos3m
OpStore %outColor %col3_t
OpEmitVertex

; Pt4_plus
%Pt4_plus = OpFAdd %v3float %Pt4 %offset
%x4p = OpCompositeExtract %float %Pt4_plus 0
%y4p = OpCompositeExtract %float %Pt4_plus 1
%z4p = OpCompositeExtract %float %Pt4_plus 2
%glpos4p = OpCompositeConstruct %v4float %x4p %y4p %z4p %float_1
%glpos4p_ptr = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %glpos4p_ptr %glpos4p
; Color t4
%col1_t4a = OpVectorTimesScalar %v3float %color1 % %omt4
%col1_t4b = OpVectorTimesScalar %v3float %color2 % %t4
%col4_t = OpFAdd %v3float %col1_t4a % %col1_t4b
OpStore %outColor %col4_t
OpEmitVertex

; Pt4_minus
%Pt4_minus = OpFSub %v3float %Pt4 %offset
%x4m = OpCompositeExtract %float %Pt4_minus 0
%y4m = OpCompositeExtract %float %Pt4_minus 1
%z4m = OpCompositeExtract %float %Pt4_minus 2
%glpos4m = OpCompositeConstruct %v4float %x4m %y4m %z4m %float_1
%glpos4m_ptr = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %glpos4m_ptr %glpos4m
OpStore %outColor %col4_t
OpEmitVertex

; End the primitive
OpEndPrimitive

OpReturn
OpFunctionEnd

; End of shader
