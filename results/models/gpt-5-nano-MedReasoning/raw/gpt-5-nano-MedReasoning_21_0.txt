```cpp
#include <iostream>
#include <vector>
#include <deque>
#include <unordered_set>
#include <limits>
#include <cmath>
#include <algorithm>

struct VecHash {
    std::size_t operator()(const std::vector<int>& v) const noexcept {
        // A simple but reasonably good hash for a small dimensional vector
        std::size_t h = 1469598103934665603ULL;
        for (int x : v) {
            // mix in x
            h ^= static_cast<std::size_t>(x) + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
            h *= 1099511628211ULL;
        }
        return h;
    }
};

using Vec = std::vector<int>;
using VecSet = std::unordered_set<Vec, VecHash>;

int main() {
    // Fast IO (not strictly necessary, but helps in large runs)
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int D;
    long long max_turns;
    if (!(std::cin >> D >> max_turns)) {
        return 0;
    }

    std::vector<int> bounds(D);
    for (int i = 0; i < D; ++i) std::cin >> bounds[i];

    int snake_length;
    std::cin >> snake_length;

    Vec head_start(D);
    for (int i = 0; i < D; ++i) std::cin >> head_start[i];

    int F;
    std::cin >> F;
    VecSet foods;
    for (int i = 0; i < F; ++i) {
        Vec pos(D);
        for (int j = 0; j < D; ++j) std::cin >> pos[j];
        foods.insert(pos);
    }

    int O;
    std::cin >> O;
    VecSet obstacles;
    for (int i = 0; i < O; ++i) {
        Vec pos(D);
        for (int j = 0; j < D; ++j) std::cin >> pos[j];
        obstacles.insert(pos);
    }

    // Initialize snake body: treat head at front, rest along -X (dimension 0)
    std::deque<Vec> body;
    Vec head = head_start;
    body.push_front(head);
    for (int i = 1; i < snake_length; ++i) {
        Vec seg = head;
        seg[0] -= i; // move along dimension 0 negative
        body.push_back(seg);
    }

    // Helper lambdas
    auto inBounds = [&](const Vec& v) -> bool {
        for (int i = 0; i < D; ++i) {
            if (v[i] < 0 || v[i] >= bounds[i]) return false;
        }
        return true;
    };

    auto manhattan = [&](const Vec& a, const Vec& b) -> int {
        int s = 0;
        for (int i = 0; i < D; ++i) s += std::abs(a[i] - b[i]);
        return s;
    };

    // Quick exit: if there is no food at all, we already won
    if (foods.empty()) {
        return 0;
    }

    // Main loop
    long long turn = 0;
    while (turn < max_turns) {
        // Check win condition
        if (foods.empty()) {
            // All food eaten: win
            return 0;
        }

        // Current head
        head = body.front();

        // Choose the best safe move towards a food (greedy)
        int best_dim = -1;
        int best_sign = 0;
        int best_dist = std::numeric_limits<int>::max();

        // Prepare a function-like lambda to test a candidate move
        auto isSafeMove = [&](const Vec& nh, bool will_grow) -> bool {
            // Bounds
            if (!inBounds(nh)) return false;
            // Obstacle
            if (obstacles.find(nh) != obstacles.end()) return false;
            // Self-collision: if not growing, tail will move away, so allow moving into tail
            for (size_t i = 0; i < body.size(); ++i) {
                const Vec& seg = body[i];
                if (seg == nh) {
                    if (will_grow) return false;
                    else {
                        // If it's the tail, it's allowed (will vacate)
                        if (i == body.size() - 1) {
                            continue;
                        } else {
                            return false;
                        }
                    }
                }
            }
            return true;
        };

        // Enumerate all 2*D possible moves
        for (int dim = 0; dim < D; ++dim) {
            for (int sgn = -1; sgn <= 1; sgn += 2) {
                Vec nh = head;
                nh[dim] += sgn;
                if (!inBounds(nh)) continue;
                bool will_grow = (foods.find(nh) != foods.end());
                // Quick obstacle check and self-collision
                if (!isSafeMove(nh, will_grow)) continue;
                // Distance to nearest food from this candidate head
                int dist = 0;
                if (!foods.empty()) {
                    dist = std::numeric_limits<int>::max();
                    for (const Vec& f : foods) {
                        dist = std::min(dist, manhattan(nh, f));
                        if (dist == 0) break;
                    }
                } else {
                    dist = 0;
                }

                // Choose best by smallest distance to food; tie-breaker: smaller dim
                if (dist < best_dist || (dist == best_dist && dim < best_dim) || (dist == best_dist && dim == best_dim && sgn < best_sign)) {
                    best_dist = dist;
                    best_dim = dim;
                    best_sign = sgn;
                }
            }
        }

        Vec next_head;
        bool will_grow = false;
        if (best_dim != -1) {
            next_head = head;
            next_head[best_dim] += best_sign;
            will_grow = (foods.find(next_head) != foods.end());
        } else {
            // Fallback: pick any legal move ignoring long-term safety (to ensure termination)
            bool found = false;
            for (int dim = 0; dim < D && !found; ++dim) {
                for (int sgn = -1; sgn <= 1; sgn += 2) {
                    Vec nh = head;
                    nh[dim] += sgn;
                    if (!inBounds(nh)) continue;
                    if (obstacles.find(nh) != obstacles.end()) continue;
                    // We don't check self-collision here to keep a move possible
                    next_head = nh;
                    will_grow = (foods.find(nh) != foods.end());
                    best_dim = dim;
                    best_sign = sgn;
                    found = true;
                    break;
                }
            }
            if (!found) {
                // No possible moves (surrounded). Terminate.
                break;
            }
        }

        // Apply the move
        // If next_head not prepared (edge case), break
        if (best_dim == -1) break;

        // Print the move
        std::cout << best_dim << " " << best_sign << "\n";
        std::cout.flush();

        // Update snake state
        body.push_front(next_head);
        if (will_grow) {
            // Eat food: do not pop tail, remove food
            foods.erase(next_head);
        } else {
            // Normal move: pop tail
            body.pop_back();
        }

        // Increment turn
        ++turn;

        // Win check after move
        if (foods.empty()) {
            // All food eaten: win
            return 0;
        }
    }

    // Exiting due to max turns or no moves left
    return 0;
}
```
