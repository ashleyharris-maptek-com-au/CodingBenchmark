#include <bits/stdc++.h>
using namespace std;

/*
  Longest Common Substring across N strings (substr, not subsequence)
  Adaptive solver:
    - For small total input size: use Suffix Array on a concatenated array with separators.
    - For larger inputs: use a hash-based intersection with binary search on length.

  Hash-based approach uses 64-bit rolling hashes with two independent simple (mod 2^64) streams
  to avoid dependency on 128-bit intrinsics (MSVC compatibility).

  This solver prints one of the longest common substrings (or empty line if none).
*/

static const uint64_t BASE1 = 1315423911ULL;
static const uint64_t BASE2 = 11400714819323198485ULL; // large odd
static const size_t SA_THRESHOLD = 3 * 1000 * 1000; // 3 million total chars threshold for SA path

// ---------- Suffix Array path (for small sums) ----------

struct SA_Int {
    vector<int> s;      // integer-coded string with separators
    vector<int> owner;    // string id for each position (or -1 for separator)
    vector<int> startPos; // start position of each original string in s
    vector<int> sa;       // suffix array
    vector<int> lcp;      // lcp array
};

// Build suffix array for vector<int> (doubling algorithm with struct-based sort)
static void build_sa(const vector<int>& s, vector<int>& sa, vector<int>& lcp, vector<int>& owner) {
    int n = (int)s.size();
    sa.resize(n);
    vector<int> rank(n), tmp(n);

    for (int i = 0; i < n; ++i) {
        sa[i] = i;
        rank[i] = s[i];
    }

    // temporary structure for sorting by (a, b)
    struct Node { int idx; int a; int b; };
    vector<Node> arr;
    arr.reserve(n);

    for (int k = 1; k < n; k <<= 1) {
        arr.clear();
        for (int i = 0; i < n; ++i) {
            int a = rank[i];
            int b = (i + k < n) ? rank[i + k] : -1;
            arr.push_back({i, a, b});
        }
        sort(arr.begin(), arr.end(), [](const Node& x, const Node& y){
            if (x.a != y.a) return x.a < y.a;
            return x.b < y.b;
        });
        for (int i = 0; i < n; ++i) sa[i] = arr[i].idx;

        tmp[sa[0]] = 0;
        for (int i = 1; i < n; ++i) {
            int a1 = arr[i-1].a;
            int b1 = arr[i-1].b;
            int a2 = arr[i].a;
            int b2 = arr[i].b;
            bool diff = (a1 != a2) || (b1 != b2);
            tmp[sa[i]] = tmp[sa[i-1]] + (diff ? 1 : 0);
        }
        for (int i = 0; i < n; ++i) rank[i] = tmp[i];
        if (rank[sa[n-1]] == n - 1) break;
    }

    // Build LCP (Kasai)
    lcp.assign(n, 0);
    vector<int> inv_sa(n);
    for (int i = 0; i < n; ++i) inv_sa[sa[i]] = i;
    int h = 0;
    for (int i = 0; i < n; ++i) {
        int r = inv_sa[i];
        if (r == 0) continue;
        int j = sa[r - 1];
        while (i + h < n && j + h < n && s[i + h] == s[j + h]) ++h;
        lcp[r] = h;
        if (h > 0) --h;
    }
}

// ---------- Suffix Array solution entry (for small total length) ----------
static string solve_by_sa(const vector<string>& S) {
    int N = (int)S.size();
    if (N == 0) return "";
    if (N == 1) return S[0];

    // Build concatenated integer sequence with separators (-1, -2, ...)
    vector<int> codes;
    vector<int> owner;
    vector<int> startPos(N);

    for (int i = 0; i < N; ++i) {
        startPos[i] = (int)codes.size();
        for (unsigned char c : S[i]) {
            codes.push_back((int)c);
            owner.push_back(i);
        }
        // separator unique per string (negative)
        codes.push_back(-(i + 1));
        owner.push_back(-1);
    }

    // If total length is zero or something degenerate
    int n = (int)codes.size();
    if (n == 0) return "";

    vector<int> sa, lcp, own = owner;
    build_sa(codes, sa, lcp, own);

    // Sliding window to cover all N strings
    vector<int> cnt(N, 0);
    int unique = 0;
    deque<pair<int,int>> dq; // (index, lcp value)
    int bestLen = 0;
    int bestLeft = 0;
    int L = 0;

    for (int r = 0; r < n; ++r) {
        int sid = own[sa[r]];
        if (sid >= 0) {
            cnt[sid]++;
            if (cnt[sid] == 1) ++unique;
        }
        if (r > 0) {
            int val = lcp[r];
            while (!dq.empty() && dq.back().second >= val) dq.pop_back();
            dq.emplace_back(r, val);
        }

        while (unique == N) {
            if (sa[L] >= 0 && L < r) {
                int curMin = dq.empty() ? 0 : dq.front().second;
                if (curMin > bestLen) {
                    bestLen = curMin;
                    bestLeft = L;
                }
            }
            int sidL = own[sa[L]];
            if (sidL >= 0) {
                cnt[sidL]--;
                if (cnt[sidL] == 0) --unique;
            }
            ++L;
            while (!dq.empty() && dq.front().first <= L) dq.pop_front();
        }
    }

    if (bestLen <= 0) return "";

    // Reconstruct: take substring from the corresponding original string
    int pos = sa[bestLeft];
    int pid = own[pos];
    if (pid < 0) {
        // Fallback: find non-separator near bestLeft
        for (int i = bestLeft+1; i <= min(bestLeft+5, n-1); ++i) {
            if (own[sa[i]] >= 0) { pid = own[sa[i]]; pos = sa[i]; break; }
        }
        if (pid < 0) return "";
    }

    int start = startPos[pid];
    int idxInside = pos - start;
    if (idxInside < 0) idxInside = 0;
    return S[pid].substr(idxInside, bestLen);
}

// ---------- Hash-based (for large inputs) ----------

struct StringInfo {
    string s;
    vector<uint64_t> H1;
    vector<uint64_t> H2;
};
static inline uint64_t mul64(uint64_t a, uint64_t b) {
    // 64-bit overflow multiplication is defined for unsigned integers modulo 2^64
    return a * b;
}
static inline uint64_t sub64(uint64_t a, uint64_t b) {
    return a - b; // wrap around modulo 2^64
}

static string solve_by_hash(const vector<string>& S) {
    int N = (int)S.size();
    if (N == 0) return "";
    if (N == 1) return S[0];
    if (S.empty()) return "";

    // Precompute max length and per-string rolling-hash data
    size_t maxLen = 0;
    int minLen = INT_MAX;
    for (auto &ss : S) {
        maxLen = max(maxLen, ss.size());
        minLen = min(minLen, (int)ss.size());
    }
    if (minLen == 0) return "";

    // Build per-string hash data (64-bit, two bases)
    // We'll use two independent 64-bit rolling hash streams (mod 2^64)
    vector<StringInfo> infos(N);
    // Precompute powers up to maxLen
    vector<uint64_t> pow1(maxLen + 1), pow2(maxLen + 1);
    pow1[0] = 1;
    pow2[0] = 1;
    for (size_t i = 1; i <= maxLen; ++i) {
        pow1[i] = mul64(pow1[i-1], BASE1);
        pow2[i] = mul64(pow2[i-1], BASE2);
    }

    // Fill per-string H arrays
    for (int i = 0; i < N; ++i) {
        const string &t = S[i];
        infos[i].s = t;
        size_t n = t.size();
        infos[i].H1.resize(n + 1);
        infos[i].H2.resize(n + 1);
        uint64_t h1 = 0, h2 = 0;
        infos[i].H1[0] = h1;
        infos[i].H2[0] = h2;
        for (size_t j = 0; j < n; ++j) {
            unsigned int ch = (unsigned char) t[j];
            h1 = mul64(h1, BASE1) + (uint64_t)(ch + 1);
            h2 = mul64(h2, BASE2) + (uint64_t)(ch + 1);
            infos[i].H1[j+1] = h1;
            infos[i].H2[j+1] = h2;
        }
    }

    auto getHash1 = [&](int idx, int l, int len) -> uint64_t {
        // substring infos[idx].H1[l..l+len)
        const auto &H1 = infos[idx].H1;
        uint64_t a = H1[l];
        uint64_t b = H1[l+len];
        return sub64(b, mul64(a, pow1[len]));
    };
    auto getHash2 = [&](int idx, int l, int len) -> uint64_t {
        const auto &H2 = infos[idx].H2;
        uint64_t a = H2[l];
        uint64_t b = H2[l+len];
        return sub64(b, mul64(a, pow2[len]));
    };

    auto exists_with_example = [&](int L, string &out) -> bool {
        if (L <= 0) { out.clear(); return true; }
        // Build set of keys from first string
        struct Key { uint64_t a; uint64_t b; };
        struct KeyHash {
            size_t operator()(const Key& k) const {
                return (size_t)(k.a ^ (k.b << 1));
            }
        };
        struct KeyEq {
            bool operator()(const Key& x, const Key& y) const {
                return x.a == y.a && x.b == y.b;
            }
        };

        unordered_set<Key, KeyHash, KeyEq> common;
        int n0 = (int)infos[0].s.size();
        common.reserve((size_t)max(1, n0 - L + 1) * 2);

        for (int i = 0; i + L <= n0; ++i) {
            uint64_t h1 = getHash1(0, i, L);
            uint64_t h2 = getHash2(0, i, L);
            common.insert({h1, h2});
        }

        // Intersect with each subsequent string
        for (int k = 1; k < N; ++k) {
            int nk = (int)infos[k].s.size();
            unordered_set<Key, KeyHash, KeyEq> next;
            next.reserve(min((size_t)common.size(), (size_t)max(0, nk - L + 1)) * 2);
            for (int i = 0; i + L <= nk; ++i) {
                uint64_t h1 = getHash1(k, i, L);
                uint64_t h2 = getHash2(k, i, L);
                Key key{h1, h2};
                if (common.find(key) != common.end()) {
                    next.insert(key);
                }
            }
            if (next.empty()) return false;
            common.swap(next);
        }

        if (common.empty()) return false;

        // Reconstruct: pick any key from common and find actual substring in first string
        auto it = common.begin();
        uint64_t a = it->a, b = it->b;
        // locate in first string
        int n0 = (int)infos[0].s.size();
        for (int i = 0; i + L <= n0; ++i) {
            uint64_t h1 = getHash1(0, i, L);
            uint64_t h2 = getHash2(0, i, L);
            if (h1 == a && h2 == b) {
                out = infos[0].s.substr(i, L);
                return true;
            }
        }
        // Fallback (shouldn't happen if no hash collision aggressively)
        return false;
    };

    // Binary search on length
    int lo = 0, hi = minLen;
    string ans;
    while (lo < hi) {
        int mid = lo + (hi - lo + 1) / 2;
        string tmp;
        bool ok = exists_with_example(mid, tmp);
        if (ok) {
            lo = mid;
            ans = tmp;
        } else {
            hi = mid - 1;
        }
    }
    if (lo == 0) return ""; // no common substring
    // If we already computed ans during binary search, ensure we return the final best
    if ((int)ans.size() != lo) {
        // compute once more to get the actual substring
        string tmp;
        exists_with_example(lo, tmp);
        ans = tmp;
    }
    return ans;
}

// ---------- Main solve (adaptive) ----------

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    string dummy;
    getline(cin, dummy); // consume endline

    vector<string> S(N);
    for (int i = 0; i < N; ++i) {
        getline(cin, S[i]);
        // If there might be stray CR in Windows line endings
        if (!S[i].empty() && S[i].back() == '\r') S[i].pop_back();
    }

    if (N == 0) {
        cout << "\n";
        return 0;
    }
    if (N == 1) {
        cout << S[0] << "\n";
        return 0;
    }

    // Edge: if any string is empty, result is empty
    bool anyEmpty = false;
    size_t totalLen = 0;
    int minLen = INT_MAX;
    for (auto &t : S) {
        totalLen += t.size();
        minLen = min(minLen, (int)t.size());
        if (t.empty()) anyEmpty = true;
    }
    if (anyEmpty) {
        cout << "\n";
        return 0;
    }

    // Adaptive strategy
    string answer;
    if (totalLen <= (size_t)SA_THRESHOLD) {
        answer = solve_by_sa(S);
    } else {
        answer = solve_by_hash(S);
    }

    cout << answer << "\n";
    return 0;
}
