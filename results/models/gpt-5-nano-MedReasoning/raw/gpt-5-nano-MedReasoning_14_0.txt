```csharp
using System;
using System.Collections.Generic;

namespace MinesweeperSolver
{
    // Represents a numeric cell constraint.
    // value: the revealed digit (0-8)
    // flagged: number of adjacent cells already known to be mines ('*')
    // neighbors: list of adjacent unknown cells (variables)
    class DigitInfo
    {
        public int value;
        public int flagged;
        public List<int> neighbors;

        public DigitInfo(int value, int flagged)
        {
            this.value = value;
            this.flagged = flagged;
            this.neighbors = new List<int>();
        }
    }

    class Program
    {
        // Entry point
        static void Main()
        {
            // Read first line: H W
            string first = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(first))
            {
                Console.WriteLine("0");
                return;
            }
            string[] parts = first.Trim().Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length < 2)
            {
                Console.WriteLine("0");
                return;
            }
            int H = int.Parse(parts[0]);
            int W = int.Parse(parts[1]);

            // Read grid lines
            char[][] grid = new char[H][];
            for (int r = 0; r < H; r++)
            {
                string line = Console.ReadLine();
                if (line == null)
                    line = "";
                // Ensure line length at least W by padding spaces if needed
                if (line.Length < W)
                {
                    line = line + new string(' ', W - line.Length);
                }
                grid[r] = line.ToCharArray();
            }

            int Ncells = H * W;
            int[] varIndex = new int[Ncells];
            for (int i = 0; i < Ncells; i++) varIndex[i] = -1;

            List<DigitInfo> digits = new List<DigitInfo>();
            List<int> varR = new List<int>();
            List<int> varC = new List<int>();

            int[] dr = new int[] { -1, -1, -1, 0, 0, 1, 1, 1 };
            int[] dc = new int[] { -1, 0, 1, -1, 1, -1, 0, 1 };

            // Build constraints: for every digit cell, collect adjacent unknowns
            for (int r = 0; r < H; r++)
            {
                for (int c = 0; c < W; c++)
                {
                    char ch = grid[r][c];
                    if (ch >= '0' && ch <= '8')
                    {
                        int val = ch - '0';
                        int flagged = 0;
                        DigitInfo di = new DigitInfo(val, 0); // we'll set flagged below as we scan
                        // Count flagged around and collect unknown neighbors
                        for (int k = 0; k < 8; k++)
                        {
                            int nr = r + dr[k];
                            int nc = c + dc[k];
                            if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;
                            char nb = grid[nr][nc];
                            if (nb == '*')
                                flagged++;
                        }
                        // Second pass: collect unknown neighbors and assign var indices
                        for (int k = 0; k < 8; k++)
                        {
                            int nr = r + dr[k];
                            int nc = c + dc[k];
                            if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;
                            char nb = grid[nr][nc];
                            if (nb == ' ')
                            {
                                int id = varIndex[nr * W + nc];
                                if (id == -1)
                                {
                                    id = varR.Count;
                                    varIndex[nr * W + nc] = id;
                                    varR.Add(nr);
                                    varC.Add(nc);
                                }
                                di.neighbors.Add(id);
                            }
                        }
                        di.flagged = flagged;
                        digits.Add(di);
                    }
                }
            }

            int varCount = varR.Count;

            // If there are no unknowns near digits, we can simply output nothing
            // But we still need to handle cases with unknowns not adjacent to digits (these can never be proven safe)

            // Build per-variable to digits mapping
            List<List<int>> varToDigits = new List<List<int>>(varCount);
            for (int i = 0; i < varCount; i++) varToDigits.Add(new List<int>());

            for (int d = 0; d < digits.Count; d++)
            {
                foreach (int v in digits[d].neighbors)
                {
                    varToDigits[v].Add(d);
                }
            }

            // If there are many variables, we will switch strategies adaptively
            // Strategy A (adaptive): full constraint propagation using local deductions across digits
            // Strategy B (fallback): only per-digit immediate deductions (no cross-propagation)

            const int FULL_PROP_THRESHOLD = 200000; // heuristic threshold

            List<(int x, int y)> result = new List<(int x, int y)>();

            if (varCount <= FULL_PROP_THRESHOLD)
            {
                // Full propagation
                int D = digits.Count;
                int[] digitRequired = new int[D];
                int[] digitUnknown = new int[D];

                for (int d = 0; d < D; d++)
                {
                    digitRequired[d] = digits[d].value - digits[d].flagged;
                    digitUnknown[d] = digits[d].neighbors.Count;
                }

                // Variable status: -1 unknown, 0 safe, 1 mine
                sbyte[] varStatus = new sbyte[varCount];
                for (int i = 0; i < varCount; i++) varStatus[i] = -1;

                Queue<(int varId, int val)> assignQ = new Queue<(int, int)>();

                // Initial propagation: any digit with 0 or all unknowns as mines
                for (int d = 0; d < D; d++)
                {
                    if (digitUnknown[d] == 0) continue;
                    int req = digitRequired[d];
                    int unk = digitUnknown[d];
                    if (req == 0)
                    {
                        foreach (int v in digits[d].neighbors)
                        {
                            if (varStatus[v] == -1)
                                assignQ.Enqueue((v, 0));
                        }
                    }
                    else if (req == unk)
                    {
                        foreach (int v in digits[d].neighbors)
                        {
                            if (varStatus[v] == -1)
                                assignQ.Enqueue((v, 1));
                        }
                    }
                }

                // Helper to assign a variable safely and propagate
                while (assignQ.Count > 0)
                {
                    var item = assignQ.Dequeue();
                    int vid = item.varId;
                    int val = item.val;

                    if (vid < 0 || vid >= varCount) continue;
                    if (varStatus[vid] != -1)
                    {
                        // Check for consistency; if conflict, ignore (board should be consistent)
                        if (varStatus[vid] != val) continue;
                        else continue;
                    }

                    varStatus[vid] = (sbyte)val;

                    // Propagate to all affected digits
                    foreach (int did in varToDigits[vid])
                    {
                        if (val == 1)
                        {
                            digitUnknown[did]--;
                            digitRequired[did]--;
                        }
                        else // val == 0
                        {
                            digitUnknown[did]--;
                        }

                        if (digitUnknown[did] > 0)
                        {
                            int req = digitRequired[did];
                            int unk = digitUnknown[did];

                            if (req == 0)
                            {
                                foreach (int v2 in digits[did].neighbors)
                                {
                                    if (varStatus[v2] == -1)
                                        assignQ.Enqueue((v2, 0));
                                }
                            }
                            else if (req == unk)
                            {
                                foreach (int v2 in digits[did].neighbors)
                                {
                                    if (varStatus[v2] == -1)
                                        assignQ.Enqueue((v2, 1));
                                }
                            }
                        }
                    }
                }

                // Collect safe cells deduced
                for (int vid = 0; vid < varCount; vid++)
                {
                    if (varStatus[vid] == 0)
                    {
                        result.Add((varC[vid], varR[vid]));
                    }
                }
            }
            else
            {
                // Strategy B: fallback per-digit quick deductions (no cross-propagation)
                // This catches obvious safes like 0 around a digit or digits that force all unknown neighbors to be mines
                // We'll mark safes in a boolean array to avoid duplicates
                bool[] safeSimple = new bool[varCount];

                for (int d = 0; d < digits.Count; d++)
                {
                    int req = digits[d].value - digits[d].flagged;
                    int unk = digits[d].neighbors.Count;
                    if (unk <= 0) continue;

                    if (req == 0)
                    {
                        foreach (int v in digits[d].neighbors)
                        {
                            safeSimple[v] = true;
                        }
                    }
                    // We intentionally skip the req == unk case here to avoid guessing mines;
                    // However, marking safes is safe.
                }

                for (int vid = 0; vid < varCount; vid++)
                {
                    if (safeSimple[vid])
                    {
                        result.Add((varC[vid], varR[vid]));
                    }
                }
            }

            // Output
            Console.WriteLine(result.Count);
            foreach (var p in result)
            {
                Console.WriteLine($"{p.x} {p.y}");
            }
        }
    }
}
```
