; SPIR-V geometry shader (Vulkan 1.0 compatible)
; Expands each input point into a 6-triangle hexagon (circle approximation)
; Uses per-vertex emission via separate 3-vertex triangle strips (center, edge i, edge i+1)
; Output primitive: Triangle strip concept is realized by six 3-vertex strips (EndPrimitive between each)

; Basic capabilities and memory model
OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

; Integer types for indices
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

; gl_PerVertex_in structure (Block with BuiltIn Position)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; gl_in: input array of per-vertex data (size 1 for InputPoints)
%uint1 = OpConstant %uint 1
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Input vertex attributes from vertex shader
; worldPos[] (location = 0)
%ptr_vec3_in = OpTypePointer Input %v3float
worldPos
%worldPos = OpVariable %ptr_vec3_in Input
OpDecorate %worldPos Location 0

; normal[] (location = 1) - not used in this task, but declared for completeness
%normal_ptr = OpTypePointer Input %v3float
%normal = OpVariable %normal_ptr Input
OpDecorate %normal Location 1

; color[] (location = 2) - per-vertex color
%color_ptr = OpTypePointer Input %v3float
%color = OpVariable %color_ptr Input
OpDecorate %color Location 2

; Output: per-vertex color for fragment shader (location 0)
%ptr_vec3_out = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_vec3_out Output
OpDecorate %outColor Location 0

; Output gl_PerVertex_out structure for gl_Position
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
; Access to gl_Position is through member 0 of gl_out

; Uniform block (set=0, binding=0) present but not strictly used in this implementation
%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float
OpDecorate %ubo_struct Block
%ptr_ubo = OpTypePointer Uniform %ubo_struct
%ubo = OpVariable %ptr_ubo Uniform

; Entry point declaration
%void = OpTypeVoid
%func_type_void = OpTypeFunction %void

; Begin main function
%main = OpFunction %void None %func_type_void
%entry = OpLabel

; Constants for hexagon geometry
%R = OpConstant %float 0.04          ; hexagon radius
%halfR = OpConstant %float 0.02       ; R/2
%dy = OpConstant %float 0.034641026  ; R * sin(60) = 0.04 * 0.8660254
%negR = OpConstant %float -0.04
%negHalfR = OpConstant %float -0.02
%negDy = OpConstant %float -0.034641026
%scale075 = OpConstant %float 0.75
%one = OpConstant %float 1.0

; Load center position (worldPos[0]) as vec3
%center_vec3_ptr = OpAccessChain %ptr_vec3_in %worldPos %int_0
%center = OpLoad %v3float %center_vec3_ptr
; Extract center components
%Cx = OpCompositeExtract %float %center 0
%Cy = OpCompositeExtract %float %center 1
%Cz = OpCompositeExtract %float %center 2
%Cpp = OpConstant %float 1.0

; Center color
%centerColor = OpLoad %v3float %color

; Edge color (scaled)
%edgeColor = OpVectorTimesScalar %v3float %centerColor %scale075

; Compute hexagon edge positions (7 edges, with E6 == E0 to close)
; E0 = (Cx + R, Cy)
%E0x = OpFAdd %float %Cx %R
%E0y = OpCopyObject %float %Cy

; E1 = (Cx + R/2, Cy + dy)
%E1x = OpFAdd %float %Cx %halfR
%E1y = OpFAdd %float %Cy %dy

; E2 = (Cx - R/2, Cy + dy)
%E2x = OpFAdd %float %Cx %negHalfR
%E2y = OpFAdd %float %Cy %dy

; E3 = (Cx - R, Cy)
%E3x = OpFAdd %float %Cx %negR
%E3y = OpCopyObject %float %Cy

; E4 = (Cx - R/2, Cy - dy)
%E4x = OpFAdd %float %Cx %negHalfR
%E4y = OpFAdd %float %Cy %negDy

; E5 = (Cx + R/2, Cy - dy)
%E5x = OpFAdd %float %Cx %halfR
%E5y = OpFAdd %float %Cy %negDy

; E6 = (Cx + R, Cy)  (duplicate of E0)
%E6x = %E0x
%E6y = %E0y

; Build actual edge vectors as vec4 with same z and w as center (we know Cz and w = 1.0)
%E0 = OpCompositeConstruct %v4float %E0x %E0y %Cz %Cpp
%E1 = OpCompositeConstruct %v4float %E1x %E1y %Cz %Cpp
%E2 = OpCompositeConstruct %v4float %E2x %E2y %Cz %Cpp
%E3 = OpCompositeConstruct %v4float %E3x %E3y %Cz %Cpp
%E4 = OpCompositeConstruct %v4float %E4x %E4y %Cz %Cpp
%E5 = OpCompositeConstruct %v4float %E5x %E5y %Cz %Cpp
%E6 = OpCompositeConstruct %v4float %E6x %E6y %Cz %Cpp

; Prepare to emit 6 triangles (center, Ei, E(i+1)) with EndPrimitive between each

; Helper: write a single vertex (pos, color) then EmitVertex
; Vertex 1: center
%gl_pos_ptr = OpAccessChain %ptr_out_pv_v4 %gl_out %int_0
OpStore %gl_pos_ptr %centerPos
OpStore %outColor %centerColor
OpEmitVertex

; Vertex 2: E0
%centerPosCreated0 = OpCompositeConstruct %v4float %E0x %E0y %Cz %Cpp
OpStore %gl_pos_ptr %centerPosCreated0
OpStore %outColor %edgeColor
OpEmitVertex

; Vertex 3: E1
%edgePos0 = %E1
OpStore %gl_pos_ptr %edgePos0
OpStore %outColor %edgeColor
OpEmitVertex

; End primitive for triangle 0
OpEndPrimitive

; Triangle 1: center, E1, E2
; Vertex 1: center
%centerPosCreated1 = OpCompositeConstruct %v4float %Cx %Cy %Cz %Cpp
OpStore %gl_pos_ptr %centerPosCreated1
OpStore %outColor %centerColor
OpEmitVertex

; Vertex 2: E1
OpStore %gl_pos_ptr %E1
OpStore %outColor %edgeColor
OpEmitVertex

; Vertex 3: E2
OpStore %gl_pos_ptr %E2
OpStore %outColor %edgeColor
OpEmitVertex

OpEndPrimitive

; Triangle 2: center, E2, E3
; Vertex 1: center
OpStore %gl_pos_ptr %centerPosCreated1
OpStore %outColor %centerColor
OpEmitVertex

; Vertex 2: E2
OpStore %gl_pos_ptr %E2
OpStore %outColor %edgeColor
OpEmitVertex

; Vertex 3: E3
OpStore %gl_pos_ptr %E3
OpStore %outColor %edgeColor
OpEmitVertex

OpEndPrimitive

; Triangle 3: center, E3, E4
; Vertex 1: center
OpStore %gl_pos_ptr %centerPosCreated1
OpStore %outColor %centerColor
OpEmitVertex

; Vertex 2: E3
OpStore %gl_pos_ptr %E3
OpStore %outColor %edgeColor
OpEmitVertex

; Vertex 3: E4
OpStore %gl_pos_ptr %E4
OpStore %outColor %edgeColor
OpEmitVertex

OpEndPrimitive

; Triangle 4: center, E4, E5
; Vertex 1: center
OpStore %gl_pos_ptr %centerPosCreated1
OpStore %outColor %centerColor
OpEmitVertex

; Vertex 2: E4
OpStore %gl_pos_ptr %E4
OpStore %outColor %edgeColor
OpEmitVertex

; Vertex 3: E5
OpStore %gl_pos_ptr %E5
OpStore %outColor %edgeColor
OpEmitVertex

OpEndPrimitive

; Triangle 5: center, E5, E6
; Vertex 1: center
OpStore %gl_pos_ptr %centerPosCreated1
OpStore %outColor %centerColor
OpEmitVertex

; Vertex 2: E5
OpStore %gl_pos_ptr %E5
OpStore %outColor %edgeColor
OpEmitVertex

; Vertex 3: E6
OpStore %gl_pos_ptr %E6
OpStore %outColor %edgeColor
OpEmitVertex

OpEndPrimitive

; End of main
OpReturn
OpFunctionEnd

; Execution mode and entry point
OpEntryPoint Geometry %main "main" %worldPos %normal %color %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

; Note:
; This shader reads worldPos (location 0) for the center position and color (location 2) for the center color.
; It emits six separate 3-vertex triangle strips (center, edge_i, edge_(i+1)) to approximate a hexagon
; with radius 0.04 in clip-space. The color for edge vertices uses a 0.75 intensity of the center color.
