```rust
use std::collections::{HashMap, VecDeque};
use std::io::{self, Read};

fn main() {
    // Read entire input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut lines = input.lines();

    // Read H and W
    let first_line = lines.next().unwrap_or("0 0");
    let mut wh = first_line.split_whitespace();
    let h: usize = wh.next().unwrap_or("0").parse().unwrap_or(0);
    let w: usize = wh.next().unwrap_or("0").parse().unwrap_or(0);

    // Read maze
    let mut grid: Vec<Vec<u8>> = Vec::with_capacity(h);
    let mut start: Option<(usize, usize)> = None;
    let mut goal: Option<(usize, usize)> = None;

    for i in 0..h {
        // Read the row line (preserve spaces)
        let mut line = lines.next().unwrap_or("").to_string();
        // Remove potential Windows CR
        if line.ends_with('\r') {
            line.pop();
        }
        // If the line is shorter than W (malformed input edge), pad with spaces
        if line.len() < w {
            line.push_str(&" ".repeat(w - line.len()));
        }
        let row = line.as_bytes().to_vec();

        // Locate A and B
        for (j, &cell) in row.iter().enumerate() {
            if cell == b'A' {
                start = Some((i, j));
            } else if cell == b'B' {
                goal = Some((i, j));
            }
        }

        grid.push(row);
    }

    let s = match start {
        Some(pos) => pos,
        None => {
            // No start found
            println!("0");
            return;
        }
    };

    let t = match goal {
        Some(pos) => pos,
        None => {
            // No goal found
            println!("0");
            return;
        }
    };

    // 4-directional BFS with parent tracing
    let mut parent: HashMap<(usize, usize), (usize, usize)> = HashMap::new();
    let mut q: VecDeque<(usize, usize)> = VecDeque::new();

    q.push_back(s);
    parent.insert(s, s); // root's parent points to itself

    let directions = [(1isize, 0isize), (-1, 0), (0, 1), (0, -1)];

    while let Some((x, y)) = q.pop_front() {
        // Early exit if we've reached the goal
        if (x, y) == t {
            break;
        }

        for (dx, dy) in &directions {
            let nx = x as isize + dx;
            let ny = y as isize + dy;

            if nx < 0 || ny < 0 {
                continue;
            }

            let nx_u = nx as usize;
            let ny_u = ny as usize;

            if nx_u >= h || ny_u >= w {
                continue;
            }

            // Check wall
            if grid[nx_u][ny_u] == b'#' {
                continue;
            }

            // If not visited
            if !parent.contains_key(&(nx_u, ny_u)) {
                parent.insert((nx_u, ny_u), (x, y));
                q.push_back((nx_u, ny_u));
            }
        }
    }

    // If goal wasn't reached
    if !parent.contains_key(&t) {
        println!("0");
        return;
    }

    // Reconstruct path from goal to start
    let mut path: Vec<(usize, usize)> = Vec::new();
    let mut cur = t;
    path.push(cur);
    while cur != s {
        // Safe unwrap because we know a path exists
        cur = *parent.get(&cur).unwrap();
        path.push(cur);
    }
    path.reverse();

    // Output
    println!("{}", path.len());
    for (x, y) in path {
        println!("{} {}", x, y);
    }
}
```
