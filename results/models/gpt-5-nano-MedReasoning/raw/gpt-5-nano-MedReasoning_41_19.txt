               OpCapability Shader
               OpMemoryModel Logical GLSL450
               %GLSLstd450 = OpExtInstImport "GLSL.std.450"

               %void = OpTypeVoid
               %float = OpTypeFloat 32
               %vec2 = OpTypeVector %float 2
               %vec3 = OpTypeVector %float 3
               %vec4 = OpTypeVector %float 4
               %mat4 = OpTypeMatrix %vec4 4

               %UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
               %_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
               %_ptr_Uniform_Vec4 = OpTypePointer Uniform %vec4
               %_ptr_Uniform_Vec3 = OpTypePointer Uniform %vec3

               %_ptr_Input_vec3 = OpTypePointer Input %vec3
               %_ptr_Input_vec2 = OpTypePointer Input %vec2
               %_ptr_Output_vec4 = OpTypePointer Output %vec4

               %inWorldPos = OpVariable %_ptr_Input_vec3 Input
               %inNormal   = OpVariable %_ptr_Input_vec3 Input
               %inColor    = OpVariable %_ptr_Input_vec3 Input
               %outColor   = OpVariable %_ptr_Output_vec4 Output
               %ubo        = OpVariable %_ptr_Uniform_UBO Uniform

               ; Decorations
               OpDecorate %inWorldPos Location 0
               OpDecorate %inNormal   Location 1
               OpDecorate %inColor    Location 4
               OpDecorate %outColor   Location 0
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0
               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224

               %uint = OpTypeInt 32 0
               %idx3 = OpConstant %uint 3
               %idx4 = OpConstant %uint 4

               %f0 = OpConstant %float 0.0
               %f1 = OpConstant %float 1.0
               %f32 = OpConstant %float 32.0
               %f0_1 = OpConstant %float 0.1
               %f0_05 = OpConstant %float 0.05

               %vec4Zero = OpConstant %vec4 (0.0, 0.0, 0.0, 0.0)

               %main = OpFunction %void None %void_fn
               %entry = OpLabel

               ; Load inputs
               %worldPos = OpLoad %vec3 %inWorldPos
               %worldNormalRaw = OpLoad %vec3 %inNormal
               %color = OpLoad %vec3 %inColor

               ; Normalize world normal
               %N = OpExtInst %vec3 %GLSLstd450 GLSLstd450 normalize %worldNormalRaw

               ; Load lightPos (vec4) from UBO and take xyz
               %lightPosPtr = OpAccessChain %_ptr_Uniform_Vec4 %ubo %idx3
               %lightPosW = OpLoad %vec4 %lightPosPtr
               %Lpos = OpCompositeExtract %vec3 %lightPosW 0 1 2

               ; Compute L = normalize(lightPos.xyz - worldPos)
               %Ldir = OpFSub %vec3 %Lpos %worldPos
               %L = OpExtInst %vec3 %GLSLstd450 GLSLstd450 normalize %Ldir

               ; Ambient term: ambientColor * color
               %colorR = OpCompositeExtract %float %color 0
               %colorG = OpCompositeExtract %float %color 1
               %colorB = OpCompositeExtract %float %color 2

               %ambientR = OpFMul %float %colorR %f0_1
               %ambientG = OpFMul %float %colorG %f0_05
               %ambientB = OpFMul %float %colorB %f0_05
               %ambient = OpCompositeConstruct %vec3 %ambientR %ambientG %ambientB

               ; Diffuse term: max(dot(N,L),0) * color
               %dotNL = OpExtInst %float %GLSLstd450 GLSLstd450 dot %N %L
               %diffFactor = OpExtInst %float %GLSLstd450 GLSLstd450 max %dotNL %f0
               %diffR = OpFMul %float %diffFactor %colorR
               %diffG = OpFMul %float %diffFactor %colorG
               %diffB = OpFMul %float %diffFactor %colorB
               %diffuse = OpCompositeConstruct %vec3 %diffR %diffG %diffB

               ; Specular term: pow(max(dot(R,V),0), 32) with R = reflect(-L,N)
               %minusL = OpFNeg %vec3 %L
               %R = OpExtInst %vec3 %GLSLstd450 GLSLstd450 Reflect %minusL %N
               %Rnorm = OpExtInst %vec3 %GLSLstd450 GLSLstd450 normalize %R
               ; V = normalize(cameraPos - worldPos)
               %cameraPosPtr = OpAccessChain %_ptr_Uniform_Vec4 %ubo %idx4
               %cameraPosW = OpLoad %vec4 %cameraPosPtr
               %cameraPos = OpCompositeExtract %vec3 %cameraPosW 0 1 2
               %Vdir = OpFSub %vec3 %cameraPos %worldPos
               %V = OpExtInst %vec3 %GLSLstd450 GLSLstd450 normalize %Vdir
               %dotRV = OpExtInst %float %GLSLstd450 GLSLstd450 dot %Rnorm %V
               %clampRV = OpExtInst %float %GLSLstd450 GLSLstd450 max %dotRV %f0
               %specPow = OpExtInst %float %GLSLstd450 GLSLstd450 Pow %clampRV %f32
               %specular = OpCompositeConstruct %vec3 %specPow %specPow %specPow

               ; Sum ambient + diffuse + specular
               %ambPlusDiff = OpFAdd %vec3 %ambient %diffuse
               %colorSum = OpFAdd %vec3 %ambPlusDiff %specular

               ; Clamp to [0,1]
               %zeroVec3 = OpCompositeConstruct %vec3 %f0 %f0 %f0
               %oneVec3  = OpCompositeConstruct %vec3 %f1 %f1 %f1
               %clamped = OpExtInst %vec3 %GLSLstd450 GLSLstd450 clamp %colorSum %zeroVec3 %oneVec3

               ; Output vec4(clamped, 1.0)
               %outColorVec4 = OpCompositeConstruct %vec4 %clamped %f1
               OpStore %outColor %outColorVec4

               OpReturn
               OpFunctionEnd

               ; Function declarations (types)
               %void_fn = OpTypeFunction %void
               %void = OpTypeVoid
