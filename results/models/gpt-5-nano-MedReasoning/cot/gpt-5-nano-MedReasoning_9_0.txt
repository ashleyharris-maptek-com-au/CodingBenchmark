```csharp
using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

class Program {
    static void Main() {
        var sc = new FastScanner();
        int n = sc.NextInt();
        int stockLen = sc.NextInt();
        int[] cuts = new int[n];
        for (int i = 0; i < n; i++) 
            cuts[i] = sc.NextInt();
        
        // Now compute
        if (n == 0) {
            Console.WriteLine(0);
            return;
        }

        var items = new Item[n];
        for (int i = 0; i < n; i++)
            items[i] = new Item { length = cuts[i], index = i };
        Array.Sort(items, (a, b) => b.length.CompareTo(a.length));

        var seg = new SegmentTree(n);
        int[] remaining = new int[n];
        int binCount = 0;
        List<List<int>> bins = new List<List<int>>();

        foreach (var item in items) {
            int L = item.length;
            int pos = seg.FindFirst(L, binCount - 1);
            if (pos == -1) {
                var list = new List<int> { item.index };
                bins.Add(list);
                int rem = stockLen - L;
                remaining[binCount] = rem;
                seg.Update(binCount, rem);
                binCount++;
            } else {
                bins[pos].Add(item.index);
                remaining[pos] -= L;
                seg.Update(pos, remaining[pos]);
            }
        }

        var sb = new StringBuilder();
        sb.Append(binCount).AppendLine();
        for (int i = 0; i < binCount; i++) {
            sb.Append(string.Join(" ", bins[i]));
            sb.AppendLine();
        }
        Console.Write(sb.ToString());
    }
    
    struct Item {
        public int length;
        public int index;
    }
    
    class FastScanner {
        private readonly string[] tokens;
        private int idx;

        public FastScanner() {
            var input = new StreamReader(Console.OpenStandardInput());
            string all = input.ReadToEnd();
            tokens = all.Split(new char[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            idx = 0;
        }

        public int NextInt() => int.Parse(tokens[idx++]);
    }

    class SegmentTree {
        private int n;
        private int[] tree;

        public SegmentTree(int n) {
            this.n = n;
            tree = new int[n * 4];
        }

        public void Update(int pos, int value) {
            Update(1, 0, n - 1, pos, value);
        }

        private void Update(int node, int l, int r, int pos, int val) {
            if (l == r) {
                tree[node] = val;
                return;
            }
            int mid = (l + r) / 2;
            if (pos <= mid) 
                Update(node * 2, l, mid, pos, val);
            else 
                Update(node * 2 + 1, mid + 1, r, pos, val);
            tree[node] = Math.Max(tree[node * 2], tree[node * 2 + 1]);
        }

        public int FindFirst(int threshold, int limit) {
            return FindFirst(1, 0, n - 1, threshold, limit);
        }

        private int FindFirst(int node, int l, int r, int threshold, int limit) {
            if (l > limit) return -1;
            if (tree[node] < threshold) return -1;
            if (l == r) return l;
            int mid = (l + r) >> 1;
            int left = FindFirst(node * 2, l, mid, threshold, limit);
            if (left != -1) return left;
            return FindFirst(node * 2 + 1, mid + 1, r, threshold, limit);
        }
    }
}
```
