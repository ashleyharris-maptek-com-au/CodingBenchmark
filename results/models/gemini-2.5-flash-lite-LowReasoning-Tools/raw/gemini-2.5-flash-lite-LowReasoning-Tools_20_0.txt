use std::io::{self, BufRead};
use std::collections::HashMap;

struct Sample {
    hole_id: String,
    sample_idx: usize,
    depth: f64,
    measurements: Vec<f64>,
    position: [f64; 3],
}

struct Drillhole {
    id: String,
    start: [f64; 3],
    direction: [f64; 3],
    length: f64,
    samples: Vec<Sample>,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let num_holes: usize = parts[0].parse().unwrap();
    let num_properties: usize = parts[1].parse().unwrap();

    let second_line = lines.next().unwrap().unwrap();
    let property_names: Vec<String> = second_line.split_whitespace().map(|s| s.to_string()).collect();

    let mut drillholes = Vec::with_capacity(num_holes);

    for _ in 0..num_holes {
        let hole_line = lines.next().unwrap().unwrap();
        let p: Vec<&str> = hole_line.split_whitespace().collect();
        let id = p[0].to_string();
        let start = [p[1].parse().unwrap(), p[2].parse().unwrap(), p[3].parse().unwrap()];
        let direction = [p[4].parse().unwrap(), p[5].parse().unwrap(), p[6].parse().unwrap()];
        let length: f64 = p[7].parse().unwrap();
        let num_samples: usize = p[8].parse().unwrap();

        let mut samples = Vec::with_capacity(num_samples);
        for i in 0..num_samples {
            let sample_line = lines.next().unwrap().unwrap();
            let s: Vec<&str> = sample_line.split_whitespace().collect();
            let depth: f64 = s[0].parse().unwrap();
            
            let position = [
                start[0] + direction[0] * depth,
                start[1] + direction[1] * depth,
                start[2] + direction[2] * depth,
            ];

            let mut measurements = Vec::with_capacity(num_properties);
            for j in 0..num_properties {
                measurements.push(s[j + 1].parse().unwrap());
            }

            let sample = Sample { 
                hole_id: id.clone(), 
                sample_idx: i, 
                depth, 
                measurements, 
                position 
            };
            samples.push(sample);
        }

        drillholes.push(Drillhole { id, start, direction, length, samples });
    }

    // Grid-based anomaly detection
    let grid_size = 20.0;
    let mut grid: HashMap<(i32, i32, i32), Vec<&Sample>> = HashMap::new();
    
    for hole in &drillholes {
        for sample in &hole.samples {
            let cell = (
                (sample.position[0] / grid_size).floor() as i32,
                (sample.position[1] / grid_size).floor() as i32,
                (sample.position[2] / grid_size).floor() as i32,
            );
            grid.entry(cell).or_insert(Vec::new()).push(sample);
        }
    }

    struct Anomaly {
        hole_id: String,
        sample_idx: usize,
        property_name: String,
        confidence: f64,
    }
    
    let mut anomalies = Vec::new();

    for hole in &drillholes {
        for sample in &hole.samples {
            let cell = (
                (sample.position[0] / grid_size).floor() as i32,
                (sample.position[1] / grid_size).floor() as i32,
                (sample.position[2] / grid_size).floor() as i32,
            );

            let mut neighbor_measurements = vec![Vec::new(); num_properties];
            let mut count = 0;

            for dx in -1..=1 {
                for dy in -1..=1 {
                    for dz in -1..=1 {
                        if let Some(neighbors) = grid.get(&(cell.0 + dx, cell.1 + dy, cell.2 + dz)) {
                            for neighbor in neighbors {
                                if neighbor.hole_id == sample.hole_id && (neighbor.sample_idx as i32 - sample.sample_idx as i32).abs() < 2 {
                                    continue; // Skip same hole neighbors that are too close in sequence
                                }
                                count += 1;
                                for i in 0..num_properties {
                                    neighbor_measurements[i].push(neighbor.measurements[i]);
                                }
                            }
                        }
                    }
                }
            }

            if count > 5 {
                for i in 0..num_properties {
                    let vals = &neighbor_measurements[i];
                    let mean = vals.iter().sum::<f64>() / vals.len() as f64;
                    let std_dev = (vals.iter().map(|&x| (x - mean).powi(2)).sum::<f64>() / vals.len() as f64).sqrt();
                    
                    let diff = (sample.measurements[i] - mean).abs();
                    if std_dev > 0.0 && diff > 3.0 * std_dev {
                        anomalies.push(Anomaly {
                            hole_id: sample.hole_id.clone(),
                            sample_idx: sample.sample_idx,
                            property_name: property_names[i].clone(),
                            confidence: diff / std_dev,
                        });
                    }
                }
            }
        }
    }

    anomalies.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap());

    println!("{}", anomalies.len());
    for a in anomalies {
        println!("{} {} {} {:.2}", a.hole_id, a.sample_idx, a.property_name, a.confidence);
    }
}
