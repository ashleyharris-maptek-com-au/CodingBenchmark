#include "game.hpp"
#include <random>

Game::Game(int dims, std::vector<int> size) : dimensions(dims), grid_size(size), snake([&](){ 
    Point start(dims); 
    for(int i=0; i<dims; ++i) start[i] = size[i]/2; 
    return start; }()), score(0), game_over(false) {
    food = generate_food();
}

void Game::update(Point direction) {
    if (game_over) return;

    Point next_head = snake.get_body().front();
    for (int i = 0; i < dimensions; ++i) {
        next_head[i] += direction[i];
        
        // Wall collision / Wrapping
        if (next_head[i] < 0 || next_head[i] >= grid_size[i]) {
            game_over = true;
            return;
        }
    }

    if (next_head == food) {
        snake.grow(direction);
        score++;
        food = generate_food();
    } else {
        if (snake.contains(next_head)) {
            game_over = true;
            return;
        }
        snake.move(direction);
    }
}

const Snake& Game::get_snake() const { return snake; }
const Point& Game::get_food() const { return food; }
int Game::get_score() const { return score; }
bool Game::is_game_over() const { return game_over; }

Point Game::generate_food() {
    static std::mt19937 rng(std::random_device{}());
    Point new_food(dimensions);
    do {
        for (int i = 0; i < dimensions; ++i) {
            std::uniform_int_distribution<int> dist(0, grid_size[i] - 1);
            new_food[i] = dist(rng);
        }
    } while (snake.contains(new_food));
    return new_food;
}
