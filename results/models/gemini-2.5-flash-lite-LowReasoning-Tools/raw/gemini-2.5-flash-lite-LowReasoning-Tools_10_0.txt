#include <iostream>
#include <vector>
#include <algorithm>

struct InputRect {
    int id;
    int w, h;
};

struct PackedRect {
    int id;
    int board_index;
    int x, y;
    bool rotated;
};

struct FreeRect {
    int x, y, w, h;
};

class Bin {
public:
    int board_width, board_height;
    std::vector<FreeRect> free_rects;

    Bin(int w, int h) : board_width(w), board_height(h) {
        free_rects.push_back({0, 0, w, h});
    }

    bool try_pack(int w, int h, int& out_x, int& out_y, bool& out_rotated) {
        // Simple Best-Fit based on free_rects
        int best_idx = -1;
        int min_area = 2000000000;
        bool best_rotated = false;

        for (size_t i = 0; i < free_rects.size(); ++i) {
            auto& fr = free_rects[i];
            
            // Try not rotated
            if (fr.w >= w && fr.h >= h) {
                int area = fr.w * fr.h; // Actually, maybe minimize wasted space?
                if (area < min_area) {
                    min_area = area;
                    best_idx = i;
                    best_rotated = false;
                }
            }
            // Try rotated
            if (fr.w >= h && fr.h >= w) {
                int area = fr.w * fr.h;
                if (area < min_area) {
                    min_area = area;
                    best_idx = i;
                    best_rotated = true;
                }
            }
        }

        if (best_idx == -1) return false;

        FreeRect fr = free_rects[best_idx];
        free_rects.erase(free_rects.begin() + best_idx);

        int rw = best_rotated ? h : w;
        int rh = best_rotated ? w : h;

        out_x = fr.x;
        out_y = fr.y;
        out_rotated = best_rotated;

        // Split remaining space (simple guillotine split)
        // If fr.w - rw > fr.h - rh, split horizontally, else vertically
        if (fr.w - rw > fr.h - rh) {
            if (fr.h - rh > 0)
                free_rects.push_back({fr.x, fr.y + rh, rw, fr.h - rh});
            if (fr.w - rw > 0)
                free_rects.push_back({fr.x + rw, fr.y, fr.w - rw, fr.h});
        } else {
            if (fr.w - rw > 0)
                free_rects.push_back({fr.x + rw, fr.y, fr.w - rw, rh});
            if (fr.h - rh > 0)
                free_rects.push_back({fr.x, fr.y + rh, fr.w, fr.h - rh});
        }

        return true;
    }
};

int main() {
    std::ios::sync_with_stdio(0);
    std::cin.tie(0);
    int n, board_w, board_h;
    if (!(std::cin >> n >> board_w >> board_h)) return 0;

    std::vector<InputRect> rects(n);
    for (int i = 0; i < n; ++i) {
        rects[i].id = i;
        std::cin >> rects[i].w >> rects[i].h;
    }

    // Sort by area descending
    std::sort(rects.begin(), rects.end(), [](const InputRect& a, const InputRect& b) {
        return (long long)a.w * a.h > (long long)b.w * b.h;
    });

    std::vector<Bin> bins;
    std::vector<PackedRect> packed;
    std::vector<int> active_bins;

    for (const auto& r : rects) {
        bool packed_in_existing = false;
        for (int i = 0; i < (int)active_bins.size(); ++i) {
            int bin_idx = active_bins[i];
            int x, y;
            bool rotated;
            if (bins[bin_idx].try_pack(r.w, r.h, x, y, rotated)) {
                packed.push_back({r.id, bin_idx, x, y, rotated});
                packed_in_existing = true;
                break;
            } else {
                // If try_pack fails, this bin might be full. 
                // For simplicity, just continue checking others.
            }
        }
        if (!packed_in_existing) {
            bins.emplace_back(board_w, board_h);
            int new_bin_idx = (int)bins.size() - 1;
            active_bins.push_back(new_bin_idx);
            int x, y;
            bool rotated;
            bins.back().try_pack(r.w, r.h, x, y, rotated);
            packed.push_back({r.id, new_bin_idx, x, y, rotated});
        }
    }

    std::cout << bins.size() << "\n";
    std::sort(packed.begin(), packed.end(), [](const PackedRect& a, const PackedRect& b) {
        return a.id < b.id;
    });
    for (const auto& p : packed) {
        std::cout << p.board_index << " " << p.x << " " << p.y << " " << (p.rotated ? 1 : 0) << "\n";
    }

    return 0;
}
