def solve_3sat(variables, clauses):
    """
    Solves 3-SAT problem using backtracking.
    
    variables: list of variable names
    clauses: list of clauses, where each clause is a list of (variable_name, is_positive)
    
    Returns a dictionary mapping variable names to boolean values if satisfiable, else None.
    """
    assignment = {}

    def is_consistent(clause, assignment):
        # A clause is satisfied if at least one literal is True.
        # A clause might be incomplete (some variables unassigned).
        # We check if it can still be satisfied.
        has_unassigned = False
        for var, is_positive in clause:
            if var not in assignment:
                has_unassigned = True
            else:
                val = assignment[var]
                if is_positive == val:
                    return True # Clause satisfied
        
        # If we reach here, no literal is True yet.
        # If there are unassigned variables, it might still be satisfied later.
        if has_unassigned:
            return None # Unknown
        
        return False # Clause unsatisfied

    def backtrack(index):
        if index == len(variables):
            # All variables assigned, check if all clauses are satisfied
            for clause in clauses:
                if not is_consistent(clause, assignment):
                    return False
            return True
        
        var = variables[index]
        for val in [True, False]:
            assignment[var] = val
            
            # Pruning: check if any clause is already violated
            possible = True
            for clause in clauses:
                if is_consistent(clause, assignment) == False:
                    possible = False
                    break
            
            if possible:
                if backtrack(index + 1):
                    return True
            
            del assignment[var]
            
        return False

    if backtrack(0):
        return assignment
    return None

if __name__ == "__main__":
    # Example: (x1 OR x2) AND (NOT x1 OR x3) AND (NOT x2 OR NOT x3)
    # Variables: x1, x2, x3
    vars = ['x1', 'x2', 'x3']
    clauses = [
        [('x1', True), ('x2', True)],
        [('x1', False), ('x3', True)],
        [('x2', False), ('x3', False)]
    ]
    
    result = solve_3sat(vars, clauses)
    print(f"Satisfiable: {result is not None}")
    if result:
        print(f"Assignment: {result}")
