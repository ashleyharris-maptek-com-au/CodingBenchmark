using System;
using System.Collections.Generic;
using System.Linq;

namespace TetrisAI
{
    class Program
    {
        static void Main(string[] args)
        {
            string[] setup = Console.ReadLine().Split(' ');
            int width = int.Parse(setup[0]);
            int height = int.Parse(setup[1]);
            int numPieces = int.Parse(setup[2]);

            Board board = new Board(width, height);

            for (int i = 0; i < numPieces; i++)
            {
                string pieceType = Console.ReadLine();
                Move bestMove = board.FindBestMove(pieceType);
                Console.WriteLine($"{bestMove.Rotation} {bestMove.Column}");
                Console.Out.Flush();

                string result = Console.ReadLine();
                if (result == "gameover") break;
                board.ApplyMove(pieceType, bestMove);
            }
        }
    }

    class Move
    {
        public int Rotation { get; set; }
        public int Column { get; set; }
        public double Score { get; set; }
    }

    class Board
    {
        private int width;
        private int height;
        private int[,] grid;

        public Board(int width, int height)
        {
            this.width = width;
            this.height = height;
            grid = new int[height, width];
        }

        public Move FindBestMove(string pieceType)
        {
            // Simple heuristic-based search
            Move bestMove = null;
            double bestScore = double.MinValue;

            var piece = PieceDefinitions.GetPiece(pieceType);

            for (int rot = 0; rot < 4; rot++)
            {
                var rotatedPiece = piece.Rotate(rot);
                for (int col = 0; col <= width - rotatedPiece.Width; col++)
                {
                    int row = GetDropRow(rotatedPiece, col);
                    if (row != -1)
                    {
                        double score = EvaluatePlacement(rotatedPiece, col, row);
                        if (bestMove == null || score > bestScore)
                        {
                            bestScore = score;
                            bestMove = new Move { Rotation = rot, Column = col, Score = score };
                        }
                    }
                }
            }
            return bestMove;
        }

        private int GetDropRow(Piece piece, int col)
        {
            for (int r = height - piece.Height; r >= 0; r--)
            {
                if (IsValidPlacement(piece, col, r))
                {
                    return r;
                }
            }
            return -1;
        }

        private bool IsValidPlacement(Piece piece, int col, int row)
        {
            for (int r = 0; r < piece.Height; r++)
            {
                for (int c = 0; c < piece.Width; c++)
                {
                    if (piece.Shape[r, c] != 0)
                    {
                        int boardR = row + r;
                        int boardC = col + c;
                        if (boardR >= height || boardC < 0 || boardC >= width || grid[boardR, boardC] != 0)
                            return false;
                    }
                }
            }
            return true;
        }

        private double EvaluatePlacement(Piece piece, int col, int row)
        {
            int[,] tempGrid = (int[,])grid.Clone();
            PlacePiece(tempGrid, piece, col, row);

            int linesCleared = 0;
            for (int r = 0; r < height; r++)
            {
                bool full = true;
                for (int c = 0; c < width; c++) if (tempGrid[r, c] == 0) full = false;
                if (full) linesCleared++;
            }

            int aggregateHeight = 0;
            int holes = 0;
            int bumpiness = 0;
            int[] columnHeights = new int[width];

            for (int c = 0; c < width; c++)
            {
                int h = 0;
                for (int r = 0; r < height; r++)
                {
                    if (tempGrid[r, c] != 0)
                    {
                        h = height - r;
                        break;
                    }
                }
                columnHeights[c] = h;
                aggregateHeight += h;
                
                bool foundBlock = false;
                for (int r = 0; r < height; r++)
                {
                    if (tempGrid[r, c] != 0) foundBlock = true;
                    else if (foundBlock) holes++;
                }
            }

            for (int c = 0; c < width - 1; c++)
                bumpiness += Math.Abs(columnHeights[c] - columnHeights[c + 1]);

            return (linesCleared * 10) - (aggregateHeight * 0.5) - (holes * 5) - (bumpiness * 0.2);
        }

        private void PlacePiece(int[,] targetGrid, Piece piece, int col, int row)
        {
            for (int r = 0; r < piece.Height; r++)
            {
                for (int c = 0; c < piece.Width; c++)
                {
                    if (piece.Shape[r, c] != 0)
                    {
                        targetGrid[row + r, col + c] = 1;
                    }
                }
            }
        }

        public void ApplyMove(string pieceType, Move move)
        {
            var piece = PieceDefinitions.GetPiece(pieceType).Rotate(move.Rotation);
            int row = GetDropRow(piece, move.Column);
            PlacePiece(grid, piece, move.Column, row);

            for (int r = 0; r < height; r++)
            {
                bool full = true;
                for (int c = 0; c < width; c++) if (grid[r, c] == 0) full = false;
                if (full)
                {
                    for (int r2 = r; r2 > 0; r2--)
                        for (int c = 0; c < width; c++)
                            grid[r2, c] = grid[r2 - 1, c];
                    for (int c = 0; c < width; c++) grid[0, c] = 0;
                }
            }
        }
    }

    class Piece
    {
        public int[,] Shape { get; set; }
        public int Width { get; set; }
        public int Height { get; set; }

        public Piece(int[,] shape)
        {
            Shape = shape;
            Height = shape.GetLength(0);
            Width = shape.GetLength(1);
        }

        public Piece Rotate(int times)
        {
            int[,] current = Shape;
            for (int i = 0; i < times % 4; i++)
            {
                int h = current.GetLength(0);
                int w = current.GetLength(1);
                int[,] next = new int[w, h];
                for (int r = 0; r < h; r++)
                    for (int c = 0; c < w; c++)
                        next[c, h - 1 - r] = current[r, c];
                current = next;
            }
            return new Piece(current);
        }
    }

    static class PieceDefinitions
    {
        private static Dictionary<string, int[,]> shapes = new Dictionary<string, int[,]>
        {
            {"I", new int[,] {{1, 1, 1, 1}}},
            {"O", new int[,] {{1, 1}, {1, 1}}},
            {"T", new int[,] {{0, 1, 0}, {1, 1, 1}}},
            {"S", new int[,] {{0, 1, 1}, {1, 1, 0}}},
            {"Z", new int[,] {{1, 1, 0}, {0, 1, 1}}},
            {"J", new int[,] {{1, 0, 0}, {1, 1, 1}}},
            {"L", new int[,] {{0, 0, 1}, {1, 1, 1}}}
        };

        public static Piece GetPiece(string type)
        {
            return new Piece(shapes[type]);
        }
    }
}
