using System;
using System.Collections.Generic;
using System.Linq;

namespace BinPacking3D
{
    public struct Point
    {
        public double X, Y, Z;
        public Point(double x, double y, double z) { X = x; Y = y; Z = z; }
    }

    public class Polyhedron
    {
        public List<Point> Vertices;
        public List<int[]> Faces;
        public int BoundingWidth, BoundingHeight, BoundingDepth;

        public Polyhedron(List<Point> vertices, List<int[]> faces)
        {
            Vertices = vertices;
            Faces = faces;
            CalculateBoundingBox();
        }

        private void CalculateBoundingBox()
        {
            int minX = (int)Math.Floor(Vertices.Min(v => v.X));
            int maxX = (int)Math.Ceiling(Vertices.Max(v => v.X));
            int minY = (int)Math.Floor(Vertices.Min(v => v.Y));
            int maxY = (int)Math.Ceiling(Vertices.Max(v => v.Y));
            int minZ = (int)Math.Floor(Vertices.Min(v => v.Z));
            int maxZ = (int)Math.Ceiling(Vertices.Max(v => v.Z));
            BoundingWidth = maxX - minX;
            BoundingHeight = maxY - minY;
            BoundingDepth = maxZ - minZ;
        }
    }

    class Program
    {
        static void Main(string[] args)
        {
            // Read input
            string[] line1 = Console.ReadLine().Split(' ');
            int V = int.Parse(line1[0]);
            int F = int.Parse(line1[1]);

            List<Point> vertices = new List<Point>();
            for (int i = 0; i < V; i++)
            {
                string[] v = Console.ReadLine().Split(' ');
                vertices.Add(new Point(double.Parse(v[0]), double.Parse(v[1]), double.Parse(v[2])));
            }

            List<int[]> faces = new List<int[]>();
            for (int i = 0; i < F; i++)
            {
                string[] f = Console.ReadLine().Split(' ');
                faces.Add(new int[] { int.Parse(f[0]), int.Parse(f[1]), int.Parse(f[2]) });
            }

            string[] containerLine = Console.ReadLine().Split(' ');
            int W = (int)double.Parse(containerLine[0]);
            int H = (int)double.Parse(containerLine[1]);
            int D = (int)double.Parse(containerLine[2]);

            var polyhedron = new Polyhedron(vertices, faces);
            
            // Occupancy grid
            bool[,,] grid = new bool[W, H, D];
            List<string> results = new List<string>();
            int count = 0;

            // Greedy packing
            for (int z = 0; z <= D - polyhedron.BoundingDepth; z++)
            {
                for (int y = 0; y <= H - polyhedron.BoundingHeight; y++)
                {
                    for (int x = 0; x <= W - polyhedron.BoundingWidth; x++)
                    {
                        if (CanFit(grid, x, y, z, polyhedron.BoundingWidth, polyhedron.BoundingHeight, polyhedron.BoundingDepth))
                        {
                            Place(grid, x, y, z, polyhedron.BoundingWidth, polyhedron.BoundingHeight, polyhedron.BoundingDepth);
                            results.Add($"{x} {y} {z} 1 0 0 0");
                            count++;
                        }
                    }
                }
            }

            Console.WriteLine(count);
            foreach (var res in results)
            {
                Console.WriteLine(res);
            }
        }

        static bool CanFit(bool[,,] grid, int x, int y, int z, int w, int h, int d)
        {
            for (int i = z; i < z + d; i++)
            {
                for (int j = y; j < y + h; j++)
                {
                    for (int k = x; k < x + w; k++)
                    {
                        if (grid[k, j, i]) return false;
                    }
                }
            }
            return true;
        }

        static void Place(bool[,,] grid, int x, int y, int z, int w, int h, int d)
        {
            for (int i = z; i < z + d; i++)
            {
                for (int j = y; j < y + h; j++)
                {
                    for (int k = x; k < x + w; k++)
                    {
                        grid[k, j, i] = true;
                    }
                }
            }
        }
    }
}
