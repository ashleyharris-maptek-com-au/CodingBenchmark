#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <deque>

// Simple, efficient-enough Suffix Array implementation
struct SuffixArray {
    std::string s;
    int n;
    std::vector<int> sa, lcp, rank;

    SuffixArray(const std::string& _s) : s(_s) {
        n = s.length();
        sa.resize(n);
        rank.resize(n);
        
        for (int i = 0; i < n; i++) {
            sa[i] = i;
            rank[i] = (unsigned char)s[i];
        }

        for (int k = 1; k < n; k <<= 1) {
            auto cmp = [&](int i, int j) {
                if (rank[i] != rank[j]) return rank[i] < rank[j];
                int ri = (i + k < n) ? rank[i + k] : -1;
                int rj = (j + k < n) ? rank[j + k] : -1;
                return ri < rj;
            };
            std::sort(sa.begin(), sa.end(), cmp);

            std::vector<int> tmp(n);
            tmp[sa[0]] = 0;
            for (int i = 1; i < n; i++) {
                tmp[sa[i]] = tmp[sa[i - 1]] + (cmp(sa[i - 1], sa[i]) ? 1 : 0);
            }
            rank = tmp;
            if (rank[sa[n - 1]] == n - 1) break;
        }
    }

    void buildLCP() {
        lcp.assign(n, 0);
        int k = 0;
        for (int i = 0; i < n; i++) {
            if (rank[i] == n - 1) {
                k = 0;
                continue;
            }
            int j = sa[rank[i] + 1];
            while (i + k < n && j + k < n && s[i + k] == s[j + k]) k++;
            lcp[rank[i]] = k;
            if (k > 0) k--;
        }
    }
};

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int N;
    if (!(std::cin >> N)) return 0;
    if (N == 0) return 0;

    std::string all_s;
    std::vector<int> string_id;
    
    // To handle separators, use characters that are not likely in the input.
    // However, input could be anything. Using a special integer/index to represent boundaries 
    // is better than characters. But for simplicity here, I'll assume standard input.
    
    for (int i = 0; i < N; i++) {
        std::string s;
        std::cin >> s;
        for (char c : s) {
            all_s += c;
            string_id.push_back(i);
        }
        all_s += (char)(2 + i); // Unique separator
        string_id.push_back(-1); // Separator index
    }

    if (N == 1) {
        std::cout << all_s.substr(0, all_s.find((char)2)) << std::endl;
        return 0;
    }

    SuffixArray sa_obj(all_s);
    sa_obj.buildLCP();

    int n = all_s.length();
    int left = 0, right = 0;
    std::vector<int> count(N, 0);
    int distinct_strings = 0;
    int max_len = 0;
    int max_pos = -1;

    // Sliding window over SA
    // min_q stores indices in lcp array for range [left, right-1]
    std::deque<int> min_q;

    for (right = 0; right < n; ++right) {
        int id = string_id[sa_obj.sa[right]];
        if (id != -1) {
            if (count[id] == 0) distinct_strings++;
            count[id]++;
        }

        // Maintain min_q for range [left, right-1]
        if (right > 0) {
            while (!min_q.empty() && sa_obj.lcp[min_q.back()] >= sa_obj.lcp[right-1]) {
                min_q.pop_back();
            }
            min_q.push_back(right-1);
        }

        while (distinct_strings == N) {
            // Check minimum LCP in [left, right-1]
            if (left < right) {
                while (!min_q.empty() && min_q.front() < left) {
                    min_q.pop_front();
                }
                
                if (!min_q.empty()) {
                    int current_len = sa_obj.lcp[min_q.front()];
                    if (current_len > max_len) {
                        max_len = current_len;
                        max_pos = sa_obj.sa[left];
                    }
                }
            }

            // Remove left
            int id_l = string_id[sa_obj.sa[left]];
            if (id_l != -1) {
                count[id_l]--;
                if (count[id_l] == 0) distinct_strings--;
            }
            left++;
        }
    }
    
    if (max_pos != -1) {
        std::cout << all_s.substr(max_pos, max_len) << std::endl;
    } else {
        std::cout << "" << std::endl;
    }

    return 0;
}
