#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <random>

struct Point {
    double x, y, z;
    Point operator+(const Point& other) const { return {x + other.x, y + other.y, z + other.z}; }
    Point operator-(const Point& other) const { return {x - other.x, y - other.y, z - other.z}; }
    Point operator*(double scalar) const { return {x * scalar, y * scalar, z * scalar}; }
};

double dot(Point a, Point b) { return a.x * b.x + a.y * b.y + a.z * b.z; }
Point cross(Point a, Point b) {
    return {a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}

// Tetrahedron geometry
struct Tetrahedron {
    Point center;
    double rx, ry, rz; // Euler angles
    double edge_length;

    std::vector<Point> get_vertices() const {
        // Regular tetrahedron vertices relative to center
        // For edge length e, distance from center to vertex is R = e * sqrt(3/8)
        // This is a simplified approximation for rotation
        double R = edge_length * std::sqrt(3.0 / 8.0);
        
        // Base vertices
        std::vector<Point> v = {
            {R, 0, -R / std::sqrt(2.0)},
            {-R / 2.0, R * std::sqrt(3.0) / 2.0, -R / std::sqrt(2.0)},
            {-R / 2.0, -R * std::sqrt(3.0) / 2.0, -R / std::sqrt(2.0)},
            {0, 0, R * std::sqrt(2.0)}
        };

        // Apply rotation (simplified Euler)
        auto rotate = [&](Point p) {
            // Rotate around X, Y, Z
            double x = p.x, y = p.y, z = p.z;
            // X
            double y1 = y * std::cos(rx) - z * std::sin(rx);
            double z1 = y * std::sin(rx) + z * std::cos(rx);
            y = y1; z = z1;
            // Y
            double x2 = x * std::cos(ry) + z * std::sin(ry);
            double z2 = -x * std::sin(ry) + z * std::cos(ry);
            x = x2; z = z2;
            // Z
            double x3 = x * std::cos(rz) - y * std::sin(rz);
            double y3 = x * std::sin(rz) + y * std::cos(rz);
            x = x3; y = y3;
            return Point{x + center.x, y + center.y, z + center.z};
        };

        for (auto& p : v) p = rotate(p);
        return v;
    }
};

int main() {
    int V;
    double edge_length;
    if (!(std::cin >> V >> edge_length)) return 0;

    std::vector<Point> container_vertices(V);
    for (int i = 0; i < V; ++i) {
        std::cin >> container_vertices[i].x >> container_vertices[i].y >> container_vertices[i].z;
    }

    // Simple greedy packing
    // 1. Find bounding box
    double min_x = 1e18, max_x = -1e18;
    double min_y = 1e18, max_y = -1e18;
    double min_z = 1e18, max_z = -1e18;
    for (const auto& p : container_vertices) {
        min_x = std::min(min_x, p.x); max_x = std::max(max_x, p.x);
        min_y = std::min(min_y, p.y); max_y = std::max(max_y, p.y);
        min_z = std::min(min_z, p.z); max_z = std::max(max_z, p.z);
    }

    std::vector<Tetrahedron> placed;
    std::mt19937 rng(42);
    std::uniform_real_distribution<double> dist_x(min_x, max_x);
    std::uniform_real_distribution<double> dist_y(min_y, max_y);
    std::uniform_real_distribution<double> dist_z(min_z, max_z);
    std::uniform_real_distribution<double> dist_angle(0, 2 * 3.1415926535);

    // Try to place tetrahedrons
    int attempts = 100000;
    for (int i = 0; i < attempts; ++i) {
        Tetrahedron t = {
            {dist_x(rng), dist_y(rng), dist_z(rng)},
            dist_angle(rng), dist_angle(rng), dist_angle(rng),
            edge_length
        };

        // Check if inside (simplified: check if center is inside bounding box)
        // A better check would be point-in-polyhedron, but for now let's just check if vertices are within bounds
        auto verts = t.get_vertices();
        bool inside = true;
        for (const auto& v : verts) {
            if (v.x < min_x || v.x > max_x || v.y < min_y || v.y > max_y || v.z < min_z || v.z > max_z) {
                inside = false;
                break;
            }
        }
        if (!inside) continue;

        // Check overlap
        bool overlap = false;
        for (const auto& p : placed) {
            // Simple distance check
            double dx = t.center.x - p.center.x;
            double dy = t.center.y - p.center.y;
            double dz = t.center.z - p.center.z;
            if (std::sqrt(dx*dx + dy*dy + dz*dz) < edge_length * 0.8) {
                overlap = true;
                break;
            }
        }
        if (!overlap) {
            placed.push_back(t);
        }
    }

    std::cout << placed.size() << std::endl;
    for (const auto& t : placed) {
        std::cout << t.center.x << " " << t.center.y << " " << t.center.z << " " 
                  << t.rx << " " << t.ry << " " << t.rz << std::endl;
    }

    return 0;
}
