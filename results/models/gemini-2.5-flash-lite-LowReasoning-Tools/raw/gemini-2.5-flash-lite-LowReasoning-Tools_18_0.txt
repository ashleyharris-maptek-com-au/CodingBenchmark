#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <climits>

using namespace std;

// Edmonds-Karp algorithm to find the maximum flow
long long bfs(int s, int t, vector<int>& parent, const vector<vector<int>>& capacity, int n) {
    fill(parent.begin(), parent.end(), -1);
    parent[s] = s;
    queue<pair<int, long long>> q;
    q.push({s, LLONG_MAX});

    while (!q.empty()) {
        int cur = q.front().first;
        long long flow = q.front().second;
        q.pop();

        for (int next = 0; next < n; next++) {
            if (parent[next] == -1 && capacity[cur][next] > 0) {
                parent[next] = cur;
                long long new_flow = min(flow, (long long)capacity[cur][next]);
                if (next == t) return new_flow;
                q.push({next, new_flow});
            }
        }
    }
    return 0;
}

long long max_flow(int s, int t, int n, vector<vector<int>>& capacity) {
    long long flow = 0;
    vector<int> parent(n);
    long long new_flow;

    while (new_flow = bfs(s, t, parent, capacity, n)) {
        flow += new_flow;
        int cur = t;
        while (cur != s) {
            int prev = parent[cur];
            capacity[prev][cur] -= new_flow;
            capacity[cur][prev] += new_flow;
            cur = prev;
        }
    }
    return flow;
}

int main() {
    // Example graph:
    // 0 -> 1 (cap 10)
    // 0 -> 2 (cap 5)
    // 1 -> 2 (cap 15)
    // 1 -> 3 (cap 5)
    // 2 -> 3 (cap 10)
    int n = 4; // number of nodes
    int s = 0; // source
    int t = 3; // sink
    vector<vector<int>> capacity(n, vector<int>(n, 0));
    capacity[0][1] = 10;
    capacity[0][2] = 5;
    capacity[1][2] = 15;
    capacity[1][3] = 5;
    capacity[2][3] = 10;

    cout << "Maximum flow (which equals minimum cut capacity) is: " << max_flow(s, t, n, capacity) << endl;

    return 0;
}
