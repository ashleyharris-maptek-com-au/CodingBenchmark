struct Vector3 {
    double x, y, z;
    Vector3 operator+(const Vector3& v) const { return {x + v.x, y + v.y, z + v.z}; }
    Vector3 operator-(const Vector3& v) const { return {x - v.x, y - v.y, z - v.z}; }
    Vector3 operator*(double s) const { return {x * s, y * s, z * s}; }
    double dot(const Vector3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vector3 cross(const Vector3& v) const {
        return {y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x};
    }
    double lengthSq() const { return x * x + y * y + z * z; }
};

struct Triangle {
    int a, b, c;
    Vector3 normal(const std::vector<Vector3>& verts) const {
        Vector3 v1 = verts[b] - verts[a];
        Vector3 v2 = verts[c] - verts[a];
        return v1.cross(v2);
    }
};

struct Mesh {
    std::vector<Vector3> vertices;
    std::vector<Triangle> faces;
};

// Simple AABB for BVH
struct AABB {
    Vector3 min, max;
    void expand(const Vector3& p) {
        min.x = std::min(min.x, p.x); min.y = std::min(min.y, p.y); min.z = std::min(min.z, p.z);
        max.x = std::max(max.x, p.x); max.y = std::max(max.y, p.y); max.z = std::max(max.z, p.z);
    }
    bool intersects(const AABB& other) const {
        return (min.x <= other.max.x && max.x >= other.min.x) &&
               (min.y <= other.max.y && max.y >= other.min.y) &&
               (min.z <= other.max.z && max.z >= other.min.z);
    }
};

// BVH Node
struct BVHNode {
    AABB bounds;
    int left = -1, right = -1;
    int triIndex = -1; // -1 if internal node
};

class BVH {
public:
    std::vector<BVHNode> nodes;
    std::vector<int> triIndices;
    const Mesh& mesh;

    BVH(const Mesh& m) : mesh(m) {
        triIndices.resize(mesh.faces.size());
        for(size_t i = 0; i < triIndices.size(); ++i) triIndices[i] = i;
        build(0, triIndices.size());
    }

    void build(int start, int end) {
        BVHNode node;
        node.bounds = { {std::numeric_limits<double>::max(), std::numeric_limits<double>::max(), std::numeric_limits<double>::max()},
                        {std::numeric_limits<double>::lowest(), std::numeric_limits<double>::lowest(), std::numeric_limits<double>::lowest()} };
        
        for(int i = start; i < end; ++i) {
            const auto& tri = mesh.faces[triIndices[i]];
            node.bounds.expand(mesh.vertices[tri.a]);
            node.bounds.expand(mesh.vertices[tri.b]);
            node.bounds.expand(mesh.vertices[tri.c]);
        }

        if (end - start <= 1) {
            node.triIndex = triIndices[start];
            nodes.push_back(node);
            return;
        }

        int axis = 0;
        if ((node.bounds.max.y - node.bounds.min.y) > (node.bounds.max.x - node.bounds.min.x)) axis = 1;
        if ((node.bounds.max.z - node.bounds.min.z) > (node.bounds.max.y - node.bounds.min.y)) axis = 2;

        int mid = (start + end) / 2;
        std::nth_element(triIndices.begin() + start, triIndices.begin() + mid, triIndices.begin() + end, [&](int i1, int i2) {
            auto center = [&](int idx) {
                const auto& t = mesh.faces[idx];
                return (mesh.vertices[t.a].x + mesh.vertices[t.b].x + mesh.vertices[t.c].x) / 3.0;
            };
            if (axis == 1) center = [&](int idx) { const auto& t = mesh.faces[idx]; return (mesh.vertices[t.a].y + mesh.vertices[t.b].y + mesh.vertices[t.c].y) / 3.0; };
            if (axis == 2) center = [&](int idx) { const auto& t = mesh.faces[idx]; return (mesh.vertices[t.a].z + mesh.vertices[t.b].z + mesh.vertices[t.c].z) / 3.0; };
            return center(i1) < center(i2);
        });

        int nodeIdx = nodes.size();
        nodes.push_back(node);
        
        int leftChild = nodes.size();
        build(start, mid);
        nodes[nodeIdx].left = leftChild;
        
        int rightChild = nodes.size();
        build(mid, end);
        nodes[nodeIdx].right = rightChild;
    }
};
