#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>

using namespace std;

const long long INF = numeric_limits<long long>::max() / 2;

struct Edge {
    int to;
    long long weight;
};

int main() {
    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<vector<Edge>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) {
        cin >> terminals[i];
    }

    // dp[mask][v]
    vector<vector<long long>> dp(1 << k, vector<long long>(n, INF));

    for (int i = 0; i < k; ++i) {
        dp[1 << i][terminals[i]] = 0;
    }

    for (int mask = 1; mask < (1 << k); ++mask) {
        // Submask transition
        for (int submask = (mask - 1) & mask; submask > 0; submask = (submask - 1) & mask) {
            for (int v = 0; v < n; ++v) {
                dp[mask][v] = min(dp[mask][v], dp[submask][v] + dp[mask ^ submask][v]);
            }
        }

        // Dijkstra relaxation
        priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
        for (int v = 0; v < n; ++v) {
            if (dp[mask][v] != INF) {
                pq.push({dp[mask][v], v});
            }
        }

        while (!pq.empty()) {
            long long d = pq.top().first;
            int u = pq.top().second;
            pq.pop();

            if (d > dp[mask][u]) continue;

            for (auto& edge : adj[u]) {
                if (dp[mask][edge.to] > dp[mask][u] + edge.weight) {
                    dp[mask][edge.to] = dp[mask][u] + edge.weight;
                    pq.push({dp[mask][edge.to], edge.to});
                }
            }
        }
    }

    // Result
    int full_mask = (1 << k) - 1;
    long long min_weight = INF;
    int root = -1;
    for (int i = 0; i < n; ++i) {
        if (dp[full_mask][i] < min_weight) {
            min_weight = dp[full_mask][i];
            root = i;
        }
    }

    // Reconstruct
    vector<pair<int, int>> result_edges;
    auto reconstruct = [&](auto self, int mask, int v) -> void {
        if (mask == 0 || (mask & (mask - 1)) == 0) return; // Base case: single terminal

        // Check if transition was submask
        for (int submask = (mask - 1) & mask; submask > 0; submask = (submask - 1) & mask) {
            if (dp[mask][v] == dp[submask][v] + dp[mask ^ submask][v]) {
                self(self, submask, v);
                self(self, mask ^ submask, v);
                return;
            }
        }

        // Must be edge transition
        for (auto& edge : adj[v]) {
            if (dp[mask][v] == dp[mask][edge.to] + edge.weight) {
                result_edges.push_back({v, edge.to});
                self(self, mask, edge.to);
                return;
            }
        }
    };

    reconstruct(reconstruct, full_mask, root);

    // Print
    cout << min_weight << " " << result_edges.size() << endl;
    for (auto& edge : result_edges) {
        cout << edge.first << " " << edge.second << endl;
    }

    return 0;
}
