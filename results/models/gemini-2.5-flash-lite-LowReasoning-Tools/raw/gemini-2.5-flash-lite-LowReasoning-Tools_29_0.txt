using System;
using System.Collections.Generic;
using System.Linq;

public class MaximumCliqueSolver
{
    private readonly int _numVertices;
    private readonly bool[,] _adjacencyMatrix;

    public MaximumCliqueSolver(int numVertices, bool[,] adjacencyMatrix)
    {
        _numVertices = numVertices;
        _adjacencyMatrix = adjacencyMatrix;
    }

    public List<int> FindMaximumClique()
    {
        var maxClique = new List<int>();
        BronKerbosch(new HashSet<int>(), new HashSet<int>(Enumerable.Range(0, _numVertices)), new HashSet<int>(), ref maxClique);
        return maxClique;
    }

    private void BronKerbosch(HashSet<int> R, HashSet<int> P, HashSet<int> X, ref List<int> maxClique)
    {
        if (P.Count == 0 && X.Count == 0)
        {
            if (R.Count > maxClique.Count)
            {
                maxClique = new List<int>(R);
            }
            return;
        }

        // Pivot selection to optimize
        var PUnionX = P.Union(X);
        if (!PUnionX.Any()) return;
        
        int pivot = PUnionX.First();
        var neighborsPivot = GetNeighbors(pivot);
        
        var Pcopy = new HashSet<int>(P);
        foreach (var v in Pcopy.Except(neighborsPivot))
        {
            var neighborsV = GetNeighbors(v);
            BronKerbosch(
                new HashSet<int>(R) { v },
                new HashSet<int>(P.Intersect(neighborsV)),
                new HashSet<int>(X.Intersect(neighborsV)),
                ref maxClique
            );
            P.Remove(v);
            X.Add(v);
        }
    }

    private HashSet<int> GetNeighbors(int v)
    {
        var neighbors = new HashSet<int>();
        for (int i = 0; i < _numVertices; i++)
        {
            if (_adjacencyMatrix[v, i])
            {
                neighbors.Add(i);
            }
        }
        return neighbors;
    }
}

public class Program
{
    public static void Main()
    {
        // Example Graph (5 vertices)
        // 0-1, 0-2, 1-2, 1-3, 3-4
        int n = 5;
        bool[,] adj = new bool[n, n];
        adj[0, 1] = adj[1, 0] = true;
        adj[0, 2] = adj[2, 0] = true;
        adj[1, 2] = adj[2, 1] = true;
        adj[1, 3] = adj[3, 1] = true;
        adj[3, 4] = adj[4, 3] = true;

        var solver = new MaximumCliqueSolver(n, adj);
        var clique = solver.FindMaximumClique();

        Console.WriteLine("Maximum Clique: " + string.Join(", ", clique));
    }
}
