#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstdint>
#include <iomanip>

struct City {
    float x, y;
    int id;
};

// Simple Morton code for Z-order curve
uint64_t morton_code(float x, float y, float min_x, float min_y, float range) {
    uint32_t ix = static_cast<uint32_t>((x - min_x) / range * 65535.0f);
    uint32_t iy = static_cast<uint32_t>((y - min_y) / range * 65535.0f);
    
    uint64_t code = 0;
    for (int i = 0; i < 16; ++i) {
        code |= ((uint64_t)(ix & (1 << i)) << i) | ((uint64_t)(iy & (1 << i)) << (i + 1));
    }
    return code;
}

void solve_small(int N, std::vector<City>& cities) {
    // Nearest Neighbor
    std::vector<int> tour;
    tour.reserve(N);
    std::vector<bool> visited(N, false);
    
    int current = 0;
    tour.push_back(0);
    visited[0] = true;
    
    for (int i = 1; i < N; ++i) {
        int next = -1;
        float min_dist = 1e18f;
        for (int j = 0; j < N; ++j) {
            if (!visited[j]) {
                float dx = cities[current].x - cities[j].x;
                float dy = cities[current].y - cities[j].y;
                float dist = dx * dx + dy * dy;
                if (dist < min_dist) {
                    min_dist = dist;
                    next = j;
                }
            }
        }
        tour.push_back(next);
        visited[next] = true;
        current = next;
    }
    
    // Output
    for (int i = 0; i < N; ++i) {
        std::cout << tour[i] << (i == N - 1 ? "" : " ");
    }
    std::cout << std::endl;
}

void solve_large(int N, std::vector<City>& cities) {
    float min_x = cities[0].x, max_x = cities[0].x;
    float min_y = cities[0].y, max_y = cities[0].y;
    
    for (const auto& c : cities) {
        if (c.x < min_x) min_x = c.x;
        if (c.x > max_x) max_x = c.x;
        if (c.y < min_y) min_y = c.y;
        if (c.y > max_y) max_y = c.y;
    }
    
    float range = std::max(max_x - min_x, max_y - min_y);
    if (range == 0) range = 1.0f;
    
    struct IndexedCity {
        uint64_t code;
        int id;
    };
    
    std::vector<IndexedCity> indexed_cities(N);
    for (int i = 0; i < N; ++i) {
        indexed_cities[i] = {morton_code(cities[i].x, cities[i].y, min_x, min_y, range), cities[i].id};
    }
    
    std::sort(indexed_cities.begin(), indexed_cities.end(), [](const IndexedCity& a, const IndexedCity& b) {
        return a.code < b.code;
    });
    
    // Ensure city 0 is first
    int start_idx = 0;
    for (int i = 0; i < N; ++i) {
        if (indexed_cities[i].id == 0) {
            start_idx = i;
            break;
        }
    }
    
    for (int i = 0; i < N; ++i) {
        std::cout << indexed_cities[(start_idx + i) % N].id << (i == N - 1 ? "" : " ");
    }
    std::cout << std::endl;
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);
    
    int N;
    if (!(std::cin >> N)) return 0;
    
    std::vector<City> cities(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> cities[i].x >> cities[i].y;
        cities[i].id = i;
    }
    
    if (N <= 1000) {
        solve_small(N, cities);
    } else {
        solve_large(N, cities);
    }
    
    return 0;
}
