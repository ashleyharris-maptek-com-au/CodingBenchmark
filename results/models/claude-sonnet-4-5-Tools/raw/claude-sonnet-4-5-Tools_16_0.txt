#include <iostream>
#include <vector>
#include <algorithm>
#include <limits>

using namespace std;

struct Task {
    int job_id;
    int task_id;
    int machine;
    int duration;
    int start_time;
};

struct Job {
    vector<Task> tasks;
    int next_task_idx;
    int earliest_start;
    int total_duration;
};

class Scheduler {
private:
    int J, M;
    vector<Job> jobs;
    vector<Job> best_jobs;
    int total_tasks;

    // Schedule using different dispatch rules:
    // 0 = FIFO (First job first)
    // 1 = MWR (Most Work Remaining)
    // 2 = SPT (Shortest Processing Time)
    int schedule_with_rule(int rule) {
        vector<Job> temp_jobs = jobs;
        vector<int> machine_available(M, 0);
        vector<int> remaining_work(J);

        for (int j = 0; j < J; j++) {
            temp_jobs[j].next_task_idx = 0;
            temp_jobs[j].earliest_start = 0;
            remaining_work[j] = temp_jobs[j].total_duration;
        }

        int scheduled = 0;
        int makespan = 0;

        while (scheduled < total_tasks) {
            int best_job = -1;
            int best_start = numeric_limits<int>::max();
            int best_priority = (rule == 1) ? numeric_limits<int>::min() : numeric_limits<int>::max();

            // Find the next task to schedule based on the rule
            for (int j = 0; j < J; j++) {
                if (temp_jobs[j].next_task_idx < temp_jobs[j].tasks.size()) {
                    Task& task = temp_jobs[j].tasks[temp_jobs[j].next_task_idx];
                    int earliest = max(temp_jobs[j].earliest_start, machine_available[task.machine]);

                    int priority;
                    if (rule == 0) { // FIFO
                        priority = j;
                    } else if (rule == 1) { // MWR
                        priority = remaining_work[j];
                    } else { // SPT
                        priority = task.duration;
                    }

                    bool better = false;
                    if (earliest < best_start) {
                        better = true;
                    } else if (earliest == best_start) {
                        if (rule == 1) { // MWR - maximize priority
                            better = (priority > best_priority) ||
                                     (priority == best_priority && j < best_job);
                        } else { // FIFO, SPT - minimize priority
                            better = (priority < best_priority) ||
                                     (priority == best_priority && j < best_job);
                        }
                    }

                    if (better) {
                        best_start = earliest;
                        best_job = j;
                        best_priority = priority;
                    }
                }
            }

            // Schedule the selected task
            Task& task = temp_jobs[best_job].tasks[temp_jobs[best_job].next_task_idx];
            task.start_time = best_start;
            machine_available[task.machine] = best_start + task.duration;
            temp_jobs[best_job].earliest_start = best_start + task.duration;
            remaining_work[best_job] -= task.duration;
            temp_jobs[best_job].next_task_idx++;
            makespan = max(makespan, machine_available[task.machine]);
            scheduled++;
        }

        best_jobs = temp_jobs;
        return makespan;
    }

public:
    Scheduler(int j, int m) : J(j), M(m), total_tasks(0) {
        jobs.resize(J);
    }

    void read_input() {
        for (int j = 0; j < J; j++) {
            int T;
            cin >> T;
            jobs[j].tasks.resize(T);
            jobs[j].total_duration = 0;

            for (int t = 0; t < T; t++) {
                jobs[j].tasks[t].job_id = j;
                jobs[j].tasks[t].task_id = t;
                cin >> jobs[j].tasks[t].machine >> jobs[j].tasks[t].duration;
                jobs[j].tasks[t].start_time = -1;
                jobs[j].total_duration += jobs[j].tasks[t].duration;
            }
            total_tasks += T;
        }
    }

    void solve() {
        // Adaptive strategy based on problem size
        if (total_tasks <= 5000) {
            // For smaller problems: try multiple dispatch rules and pick the best
            int best_makespan = numeric_limits<int>::max();

            for (int rule = 0; rule < 3; rule++) {
                int makespan = schedule_with_rule(rule);
                if (makespan < best_makespan) {
                    best_makespan = makespan;
                    jobs = best_jobs;
                }
            }
        } else {
            // For large problems: use MWR heuristic (good balance of quality and speed)
            schedule_with_rule(1);
            jobs = best_jobs;
        }
    }

    void output() {
        for (int j = 0; j < J; j++) {
            for (size_t t = 0; t < jobs[j].tasks.size(); t++) {
                if (t > 0) cout << " ";
                cout << jobs[j].tasks[t].start_time << " " << jobs[j].tasks[t].machine;
            }
            cout << "\n";
        }
    }
};

int main() {
    // Fast I/O for large inputs
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    cin >> J >> M;

    Scheduler scheduler(J, M);
    scheduler.read_input();
    scheduler.solve();
    scheduler.output();

    return 0;
}
