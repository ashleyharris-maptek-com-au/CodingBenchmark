#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <chrono>
#include <set>

using namespace std;

struct Vec3 {
    double x, y, z;
    Vec3(double x = 0, double y = 0, double z = 0) : x(x), y(y), z(z) {}
    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
    Vec3 operator/(double s) const { return Vec3(x / s, y / s, z / s); }
    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const {
        return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x);
    }
    double length() const { return sqrt(x * x + y * y + z * z); }
    double lengthSquared() const { return x * x + y * y + z * z; }
    Vec3 normalize() const {
        double len = length();
        return len > 1e-10 ? (*this) / len : Vec3(0, 0, 0);
    }
};

struct Tetrahedron {
    Vec3 center;
    Vec3 rotation;

    Tetrahedron(Vec3 c, Vec3 r) : center(c), rotation(r) {}
};

// Rotate point by Euler angles (ZYX convention)
Vec3 rotatePoint(const Vec3& p, const Vec3& angles) {
    double cx = cos(angles.x), sx = sin(angles.x);
    double cy = cos(angles.y), sy = sin(angles.y);
    double cz = cos(angles.z), sz = sin(angles.z);

    Vec3 result;
    result.x = p.x * (cy * cz) + p.y * (sx * sy * cz - cx * sz) + p.z * (cx * sy * cz + sx * sz);
    result.y = p.x * (cy * sz) + p.y * (sx * sy * sz + cx * cz) + p.z * (cx * sy * sz - sx * cz);
    result.z = p.x * (-sy) + p.y * (sx * cy) + p.z * (cx * cy);

    return result;
}

// Get standard tetrahedron vertices (centered at origin)
vector<Vec3> getStandardTetrahedronVertices(double edge) {
    vector<Vec3> verts;
    double h = edge * sqrt(2.0 / 3.0);
    double r = edge / sqrt(3.0);

    // Vertices of regular tetrahedron centered at origin
    verts.push_back(Vec3(r, 0, -h / 4.0));
    verts.push_back(Vec3(-r / 2.0, edge / 2.0, -h / 4.0));
    verts.push_back(Vec3(-r / 2.0, -edge / 2.0, -h / 4.0));
    verts.push_back(Vec3(0, 0, 3.0 * h / 4.0));

    return verts;
}

// Transform tetrahedron vertices by rotation and translation
vector<Vec3> getTransformedVertices(const vector<Vec3>& stdVerts, const Vec3& center, const Vec3& rotation) {
    vector<Vec3> transformed;
    for (const auto& v : stdVerts) {
        transformed.push_back(rotatePoint(v, rotation) + center);
    }
    return transformed;
}

// Check if point is inside or on boundary of tetrahedron
bool pointInTetrahedron(const Vec3& p, const vector<Vec3>& tet) {
    // Use barycentric coordinates
    Vec3 v0 = tet[1] - tet[0];
    Vec3 v1 = tet[2] - tet[0];
    Vec3 v2 = tet[3] - tet[0];
    Vec3 vp = p - tet[0];

    double d00 = v0.dot(v0);
    double d01 = v0.dot(v1);
    double d02 = v0.dot(v2);
    double d11 = v1.dot(v1);
    double d12 = v1.dot(v2);
    double d22 = v2.dot(v2);

    double d0p = v0.dot(vp);
    double d1p = v1.dot(vp);
    double d2p = v2.dot(vp);

    double denom = d00 * d11 * d22 + 2.0 * d01 * d02 * d12 - d00 * d12 * d12 - d11 * d02 * d02 - d22 * d01 * d01;

    if (fabs(denom) < 1e-10) return false;

    double invDenom = 1.0 / denom;

    double u = (d11 * d22 - d12 * d12) * d0p + (d02 * d12 - d01 * d22) * d1p + (d01 * d12 - d02 * d11) * d2p;
    u *= invDenom;

    double v = (d02 * d12 - d01 * d22) * d0p + (d00 * d22 - d02 * d02) * d1p + (d01 * d02 - d00 * d12) * d2p;
    v *= invDenom;

    double w = (d01 * d12 - d02 * d11) * d0p + (d01 * d02 - d00 * d12) * d1p + (d00 * d11 - d01 * d01) * d2p;
    w *= invDenom;

    return u >= -1e-6 && v >= -1e-6 && w >= -1e-6 && u + v + w <= 1.0 + 1e-6;
}

// Simple convex hull point test using signed volumes
bool pointInConvexHull(const Vec3& p, const vector<Vec3>& hull) {
    if (hull.size() < 4) return false;

    // Bounding box test first
    Vec3 minB = hull[0], maxB = hull[0];
    for (const auto& v : hull) {
        minB.x = min(minB.x, v.x);
        minB.y = min(minB.y, v.y);
        minB.z = min(minB.z, v.z);
        maxB.x = max(maxB.x, v.x);
        maxB.y = max(maxB.y, v.y);
        maxB.z = max(maxB.z, v.z);
    }

    double margin = 0.05;
    return p.x >= minB.x + margin && p.x <= maxB.x - margin &&
           p.y >= minB.y + margin && p.y <= maxB.y - margin &&
           p.z >= minB.z + margin && p.z <= maxB.z - margin;
}

// Check if tetrahedron is fully inside container
bool tetrahedronInContainer(const vector<Vec3>& tetVerts, const vector<Vec3>& container) {
    for (const auto& v : tetVerts) {
        if (!pointInConvexHull(v, container)) {
            return false;
        }
    }
    return true;
}

// Check overlap between two tetrahedrons using distance heuristic
bool tetrahedronsOverlap(const Tetrahedron& t1, const Tetrahedron& t2, double edge) {
    double dist = (t1.center - t2.center).length();
    double R = edge * sqrt(3.0 / 8.0); // circumradius
    return dist < 1.5 * R; // Conservative overlap check
}

int main() {
    auto startTime = chrono::high_resolution_clock::now();

    int V;
    double edge;
    cin >> V >> edge;

    vector<Vec3> container(V);
    for (int i = 0; i < V; i++) {
        cin >> container[i].x >> container[i].y >> container[i].z;
    }

    // Compute bounding box
    Vec3 minB = container[0], maxB = container[0];
    for (const auto& v : container) {
        minB.x = min(minB.x, v.x);
        minB.y = min(minB.y, v.y);
        minB.z = min(minB.z, v.z);
        maxB.x = max(maxB.x, v.x);
        maxB.y = max(maxB.y, v.y);
        maxB.z = max(maxB.z, v.z);
    }

    Vec3 size = maxB - minB;
    double volume = size.x * size.y * size.z;

    // Get standard tetrahedron
    vector<Vec3> stdTet = getStandardTetrahedronVertices(edge);

    vector<Tetrahedron> placed;

    // Adaptive parameters based on container size
    double spacing;
    int maxOrientations;

    if (volume < 100) {
        spacing = edge * 0.65;
        maxOrientations = 6;
    } else if (volume < 10000) {
        spacing = edge * 0.7;
        maxOrientations = 4;
    } else {
        spacing = edge * 0.75;
        maxOrientations = 2;
    }

    // Orientation candidates
    vector<Vec3> orientations;
    orientations.push_back(Vec3(0, 0, 0));
    if (maxOrientations >= 2) orientations.push_back(Vec3(M_PI, 0, 0));
    if (maxOrientations >= 4) {
        orientations.push_back(Vec3(0, M_PI, 0));
        orientations.push_back(Vec3(0, 0, M_PI));
    }
    if (maxOrientations >= 6) {
        orientations.push_back(Vec3(M_PI/2, 0, 0));
        orientations.push_back(Vec3(0, M_PI/2, 0));
    }

    // Grid packing
    double margin = edge * 0.5;
    for (double x = minB.x + margin; x < maxB.x - margin; x += spacing) {
        for (double y = minB.y + margin; y < maxB.y - margin; y += spacing) {
            for (double z = minB.z + margin; z < maxB.z - margin; z += spacing) {
                Vec3 center(x, y, z);

                // Quick check if center is in container
                if (!pointInConvexHull(center, container)) continue;

                // Try orientations
                for (const auto& rot : orientations) {
                    Tetrahedron candidate(center, rot);

                    // Get transformed vertices
                    vector<Vec3> tetVerts = getTransformedVertices(stdTet, center, rot);

                    // Check if tetrahedron is in container
                    if (!tetrahedronInContainer(tetVerts, container)) continue;

                    // Check overlap with placed tetrahedrons
                    bool overlaps = false;
                    for (const auto& t : placed) {
                        if (tetrahedronsOverlap(candidate, t, edge)) {
                            overlaps = true;
                            break;
                        }
                    }

                    if (!overlaps) {
                        placed.push_back(candidate);
                        break; // Found valid orientation, move to next position
                    }
                }

                // Check time limit (keep 5 second buffer)
                auto elapsed = chrono::duration_cast<chrono::seconds>(
                    chrono::high_resolution_clock::now() - startTime);
                if (elapsed.count() > 25) goto done;
            }
        }
    }

done:
    // Output results
    cout << placed.size() << endl;
    for (const auto& t : placed) {
        cout << t.center.x << " " << t.center.y << " " << t.center.z << " "
             << t.rotation.x << " " << t.rotation.y << " " << t.rotation.z << endl;
    }

    return 0;
}
