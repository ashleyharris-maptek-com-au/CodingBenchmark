using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        // Read input
        var firstLine = Console.ReadLine().Split();
        int N = int.Parse(firstLine[0]);
        int W = int.Parse(firstLine[1]);
        int H = int.Parse(firstLine[2]);
        int D = int.Parse(firstLine[3]);

        var boxes = new List<Box>();
        for (int i = 0; i < N; i++)
        {
            var dims = Console.ReadLine().Split();
            boxes.Add(new Box
            {
                Index = i,
                W = int.Parse(dims[0]),
                H = int.Parse(dims[1]),
                D = int.Parse(dims[2])
            });
        }

        // Pack boxes
        var packer = new BinPacker(W, H, D);
        var packed = packer.Pack(boxes);

        // Output results
        Console.WriteLine(packed.Count);
        foreach (var placement in packed)
        {
            Console.WriteLine($"{placement.BoxIndex} {placement.X} {placement.Y} {placement.Z}");
        }
    }
}

class Box
{
    public int Index;
    public int W, H, D;
    public long Volume => (long)W * H * D;
}

class Placement
{
    public int BoxIndex;
    public int X, Y, Z;
    public int W, H, D;
}

class BinPacker
{
    private int containerW, containerH, containerD;
    private List<Placement> placements = new List<Placement>();

    public BinPacker(int w, int h, int d)
    {
        containerW = w;
        containerH = h;
        containerD = d;
    }

    public List<Placement> Pack(List<Box> boxes)
    {
        // Sort by volume descending (largest first heuristic)
        var sortedBoxes = boxes.OrderByDescending(b => b.Volume).ThenBy(b => b.Index).ToList();

        foreach (var box in sortedBoxes)
        {
            TryPlaceBox(box);
        }

        return placements;
    }

    private bool TryPlaceBox(Box box)
    {
        // Try all unique orientations of the box
        var orientations = GetOrientations(box);

        foreach (var (w, h, d) in orientations)
        {
            var pos = FindPosition(w, h, d);
            if (pos.HasValue)
            {
                placements.Add(new Placement
                {
                    BoxIndex = box.Index,
                    X = pos.Value.x,
                    Y = pos.Value.y,
                    Z = pos.Value.z,
                    W = w,
                    H = h,
                    D = d
                });
                return true;
            }
        }

        return false;
    }

    private List<(int w, int h, int d)> GetOrientations(Box box)
    {
        // Generate all 6 unique orientations
        var orientations = new HashSet<(int, int, int)>
        {
            (box.W, box.H, box.D),
            (box.W, box.D, box.H),
            (box.H, box.W, box.D),
            (box.H, box.D, box.W),
            (box.D, box.W, box.H),
            (box.D, box.H, box.W)
        };
        return orientations.ToList();
    }

    private (int x, int y, int z)? FindPosition(int w, int h, int d)
    {
        // Check if box fits in container at all
        if (w > containerW || h > containerH || d > containerD)
            return null;

        // Generate candidate positions from corners of existing boxes (corner-point heuristic)
        var candidateX = new SortedSet<int> { 0 };
        var candidateY = new SortedSet<int> { 0 };
        var candidateZ = new SortedSet<int> { 0 };

        foreach (var p in placements)
        {
            candidateX.Add(p.X + p.W);
            candidateY.Add(p.Y + p.H);
            candidateZ.Add(p.Z + p.D);
        }

        // Try positions in order (prefer lower coordinates for tighter packing)
        foreach (var x in candidateX)
        {
            if (x + w > containerW) break;
            foreach (var y in candidateY)
            {
                if (y + h > containerH) break;
                foreach (var z in candidateZ)
                {
                    if (z + d > containerD) break;

                    if (!HasCollision(x, y, z, w, h, d))
                    {
                        return (x, y, z);
                    }
                }
            }
        }

        return null;
    }

    private bool HasCollision(int x, int y, int z, int w, int h, int d)
    {
        int x2 = x + w;
        int y2 = y + h;
        int z2 = z + d;

        // Check AABB (Axis-Aligned Bounding Box) collision with all placed boxes
        foreach (var p in placements)
        {
            int px2 = p.X + p.W;
            int py2 = p.Y + p.H;
            int pz2 = p.Z + p.D;

            // Boxes collide if they overlap on all three axes
            if (!(x >= px2 || x2 <= p.X ||
                  y >= py2 || y2 <= p.Y ||
                  z >= pz2 || z2 <= p.Z))
            {
                return true;
            }
        }
        return false;
    }
}
