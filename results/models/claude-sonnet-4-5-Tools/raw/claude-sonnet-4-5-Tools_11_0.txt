use std::io::{self, BufRead};
use std::f64::consts::PI;

#[derive(Debug, Clone)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn new(x: f64, y: f64) -> Self {
        Point { x, y }
    }

    fn rotate(&self, angle: f64) -> Point {
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        Point {
            x: self.x * cos_a - self.y * sin_a,
            y: self.x * sin_a + self.y * cos_a,
        }
    }

    fn translate(&self, dx: f64, dy: f64) -> Point {
        Point {
            x: self.x + dx,
            y: self.y + dy,
        }
    }
}

#[derive(Debug, Clone)]
struct Polygon {
    vertices: Vec<Point>,
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        Polygon { vertices }
    }

    fn bounds(&self) -> (Point, Point) {
        let mut min_x = f64::INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for v in &self.vertices {
            min_x = min_x.min(v.x);
            min_y = min_y.min(v.y);
            max_x = max_x.max(v.x);
            max_y = max_y.max(v.y);
        }

        (Point::new(min_x, min_y), Point::new(max_x, max_y))
    }

    fn area(&self) -> f64 {
        let mut area = 0.0;
        let n = self.vertices.len();
        for i in 0..n {
            let j = (i + 1) % n;
            area += self.vertices[i].x * self.vertices[j].y;
            area -= self.vertices[j].x * self.vertices[i].y;
        }
        area.abs() / 2.0
    }

    fn transform(&self, translation: &Point, rotation: f64) -> Polygon {
        let rotated: Vec<Point> = self.vertices.iter()
            .map(|p| p.rotate(rotation))
            .collect();

        let translated: Vec<Point> = rotated.iter()
            .map(|p| p.translate(translation.x, translation.y))
            .collect();

        Polygon::new(translated)
    }

    fn contains_point(&self, point: &Point) -> bool {
        let mut inside = false;
        let n = self.vertices.len();

        for i in 0..n {
            let j = (i + 1) % n;
            let vi = &self.vertices[i];
            let vj = &self.vertices[j];

            if ((vi.y > point.y) != (vj.y > point.y)) &&
               (point.x < (vj.x - vi.x) * (point.y - vi.y) / (vj.y - vi.y) + vi.x) {
                inside = !inside;
            }
        }
        inside
    }

    fn contains_polygon(&self, other: &Polygon) -> bool {
        // Check if all vertices of other polygon are inside this polygon
        for vertex in &other.vertices {
            if !self.contains_point(vertex) {
                return false;
            }
        }

        // Check for edge intersections
        if self.intersects_polygon(other) {
            return false;
        }

        true
    }

    fn intersects_polygon(&self, other: &Polygon) -> bool {
        // Check if any edges intersect
        for i in 0..self.vertices.len() {
            let j = (i + 1) % self.vertices.len();
            let a1 = &self.vertices[i];
            let a2 = &self.vertices[j];

            for k in 0..other.vertices.len() {
                let l = (k + 1) % other.vertices.len();
                let b1 = &other.vertices[k];
                let b2 = &other.vertices[l];

                if segments_intersect(a1, a2, b1, b2) {
                    return true;
                }
            }
        }
        false
    }
}

fn segments_intersect(a1: &Point, a2: &Point, b1: &Point, b2: &Point) -> bool {
    let d1 = direction(b1, b2, a1);
    let d2 = direction(b1, b2, a2);
    let d3 = direction(a1, a2, b1);
    let d4 = direction(a1, a2, b2);

    if ((d1 > 0.0 && d2 < 0.0) || (d1 < 0.0 && d2 > 0.0)) &&
       ((d3 > 0.0 && d4 < 0.0) || (d3 < 0.0 && d4 > 0.0)) {
        return true;
    }

    if d1.abs() < 1e-9 && on_segment(b1, a1, b2) { return true; }
    if d2.abs() < 1e-9 && on_segment(b1, a2, b2) { return true; }
    if d3.abs() < 1e-9 && on_segment(a1, b1, a2) { return true; }
    if d4.abs() < 1e-9 && on_segment(a1, b2, a2) { return true; }

    false
}

fn direction(p1: &Point, p2: &Point, p3: &Point) -> f64 {
    (p3.x - p1.x) * (p2.y - p1.y) - (p2.x - p1.x) * (p3.y - p1.y)
}

fn on_segment(p1: &Point, p2: &Point, p3: &Point) -> bool {
    p2.x >= p1.x.min(p3.x) && p2.x <= p1.x.max(p3.x) &&
    p2.y >= p1.y.min(p3.y) && p2.y <= p1.y.max(p3.y)
}

#[derive(Debug)]
struct PlacedPiece {
    polygon: Polygon,
    stock_index: usize,
    translation: Point,
    rotation: f64,
}

#[derive(Debug)]
struct Stock {
    polygon: Polygon,
    placed_pieces: Vec<Polygon>,
}

impl Stock {
    fn new(polygon: Polygon) -> Self {
        Stock {
            polygon,
            placed_pieces: Vec::new(),
        }
    }

    fn can_place(&self, piece: &Polygon) -> bool {
        // Check if piece fits in stock
        if !self.polygon.contains_polygon(piece) {
            return false;
        }

        // Check for overlaps with placed pieces
        for placed in &self.placed_pieces {
            if piece.intersects_polygon(placed) || placed.intersects_polygon(piece) {
                return false;
            }

            // Additional check: if any vertex of one is inside the other
            for v in &piece.vertices {
                if placed.contains_point(v) {
                    return false;
                }
            }
            for v in &placed.vertices {
                if piece.contains_point(v) {
                    return false;
                }
            }
        }

        true
    }

    fn place(&mut self, piece: Polygon) {
        self.placed_pieces.push(piece);
    }
}

fn try_place_in_stock(
    stock: &Stock,
    piece: &Polygon,
    rotation: f64,
    grid_step: f64,
) -> Option<Point> {
    let rotated = piece.transform(&Point::new(0.0, 0.0), rotation);
    let (min_bound, _) = rotated.bounds();

    // Normalize so piece starts at origin
    let normalized: Vec<Point> = rotated.vertices.iter()
        .map(|p| Point::new(p.x - min_bound.x, p.y - min_bound.y))
        .collect();
    let normalized_piece = Polygon::new(normalized);

    let (stock_min, stock_max) = stock.polygon.bounds();
    let (piece_min, piece_max) = normalized_piece.bounds();
    let piece_width = piece_max.x - piece_min.x;
    let piece_height = piece_max.y - piece_min.y;

    // Try positions with adaptive grid
    let x_range = (stock_max.x - stock_min.x - piece_width).max(0.0);
    let y_range = (stock_max.y - stock_min.y - piece_height).max(0.0);

    let x_steps = ((x_range / grid_step).ceil() as usize + 1).min(50);
    let y_steps = ((y_range / grid_step).ceil() as usize + 1).min(50);

    for y_idx in 0..y_steps {
        let y = stock_min.y + (y_idx as f64) * (y_range / y_steps.max(1) as f64);

        for x_idx in 0..x_steps {
            let x = stock_min.x + (x_idx as f64) * (x_range / x_steps.max(1) as f64);

            let translation = Point::new(x, y);
            let transformed = normalized_piece.transform(&translation, 0.0);

            if stock.can_place(&transformed) {
                return Some(translation);
            }
        }
    }

    None
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read stock polygon
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let (sv, np) = (parts[0], parts[1]);

    let mut stock_vertices = Vec::new();
    for _ in 0..sv {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<f64> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        stock_vertices.push(Point::new(coords[0], coords[1]));
    }
    let stock_polygon = Polygon::new(stock_vertices);

    // Read pieces
    let mut pieces = Vec::new();
    for _ in 0..np {
        let line = lines.next().unwrap().unwrap();
        let pv: usize = line.trim().parse().unwrap();

        let mut piece_vertices = Vec::new();
        for _ in 0..pv {
            let line = lines.next().unwrap().unwrap();
            let coords: Vec<f64> = line.split_whitespace()
                .map(|s| s.parse().unwrap())
                .collect();
            piece_vertices.push(Point::new(coords[0], coords[1]));
        }
        pieces.push(Polygon::new(piece_vertices));
    }

    // Sort pieces by area (largest first)
    let mut piece_indices: Vec<usize> = (0..pieces.len()).collect();
    piece_indices.sort_by(|&a, &b| {
        pieces[b].area().partial_cmp(&pieces[a].area()).unwrap()
    });

    // Determine rotation granularity based on problem size
    let rotations: Vec<f64> = if np <= 50 {
        // Fine granularity for small problems
        (0..24).map(|i| (i as f64) * PI / 12.0).collect()
    } else if np <= 500 {
        // Medium granularity
        (0..8).map(|i| (i as f64) * PI / 4.0).collect()
    } else {
        // Coarse granularity for large problems
        vec![0.0, PI / 2.0, PI, 3.0 * PI / 2.0]
    };

    // Grid step for placement attempts
    let grid_step = if np <= 50 { 5.0 } else if np <= 500 { 10.0 } else { 20.0 };

    let mut stocks = vec![Stock::new(stock_polygon.clone())];
    let mut placements = vec![None; pieces.len()];

    // Place each piece
    for &piece_idx in &piece_indices {
        let piece = &pieces[piece_idx];
        let mut placed = false;

        // Try to place in existing stocks
        for stock_idx in 0..stocks.len() {
            if placed {
                break;
            }

            for &rotation in &rotations {
                if let Some(translation) = try_place_in_stock(&stocks[stock_idx], piece, rotation, grid_step) {
                    let rotated = piece.transform(&Point::new(0.0, 0.0), rotation);
                    let (min_bound, _) = rotated.bounds();
                    let normalized: Vec<Point> = rotated.vertices.iter()
                        .map(|p| Point::new(p.x - min_bound.x, p.y - min_bound.y))
                        .collect();
                    let normalized_piece = Polygon::new(normalized);
                    let final_piece = normalized_piece.transform(&translation, 0.0);

                    stocks[stock_idx].place(final_piece);
                    placements[piece_idx] = Some((stock_idx, translation, rotation));
                    placed = true;
                    break;
                }
            }
        }

        // If not placed, create new stock
        if !placed {
            let new_stock_idx = stocks.len();
            stocks.push(Stock::new(stock_polygon.clone()));

            // Try to place in new stock
            for &rotation in &rotations {
                if let Some(translation) = try_place_in_stock(&stocks[new_stock_idx], piece, rotation, grid_step) {
                    let rotated = piece.transform(&Point::new(0.0, 0.0), rotation);
                    let (min_bound, _) = rotated.bounds();
                    let normalized: Vec<Point> = rotated.vertices.iter()
                        .map(|p| Point::new(p.x - min_bound.x, p.y - min_bound.y))
                        .collect();
                    let normalized_piece = Polygon::new(normalized);
                    let final_piece = normalized_piece.transform(&translation, 0.0);

                    stocks[new_stock_idx].place(final_piece);
                    placements[piece_idx] = Some((new_stock_idx, translation, rotation));
                    break;
                }
            }
        }
    }

    // Output results
    println!("{}", stocks.len());
    for placement in &placements {
        if let Some((stock_idx, translation, rotation)) = placement {
            println!("{} {} {} {}", stock_idx, translation.x, translation.y, rotation);
        } else {
            println!("0 0.0 0.0 0.0");
        }
    }
}
