use std::io::{self, BufRead, Write};
use std::sync::mpsc::{self, Receiver, Sender, TryRecvError};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Debug, Clone)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel_remaining: f64,
}

#[derive(Debug)]
struct Environment {
    width: f64,
    height: f64,
    gravity: f64,
    max_thrust: f64,
    fuel: f64,
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
}

fn main() {
    let (state_tx, state_rx): (Sender<State>, Receiver<State>) = mpsc::channel();
    let (cmd_tx, cmd_rx): (Sender<(f64, f64)>, Receiver<(f64, f64)>) = mpsc::channel();

    let env = read_map();

    thread::spawn(move || {
        input_thread(state_tx);
    });

    thread::spawn(move || {
        output_thread(cmd_rx);
    });

    control_loop(state_rx, cmd_tx, env);
}

fn read_map() -> Environment {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let line1 = lines.next().unwrap().unwrap();
    let parts: Vec<f64> = line1.split_whitespace().map(|s| s.parse().unwrap()).collect();
    let (width, height, gravity, max_thrust, fuel) = (parts[0], parts[1], parts[2], parts[3], parts[4]);

    let line2 = lines.next().unwrap().unwrap();
    let parts: Vec<f64> = line2.split_whitespace().map(|s| s.parse().unwrap()).collect();
    let (start_x, start_y, target_x, target_y) = (parts[0], parts[1], parts[2], parts[3]);

    for _ in 0..(height as usize) {
        lines.next();
    }

    Environment {
        width,
        height,
        gravity,
        max_thrust,
        fuel,
        start_x,
        start_y,
        target_x,
        target_y,
    }
}

fn input_thread(state_tx: Sender<State>) {
    let stdin = io::stdin();
    let reader = stdin.lock();

    for line in reader.lines() {
        if let Ok(line) = line {
            let parts: Vec<f64> = line.split_whitespace()
                .filter_map(|s| s.parse().ok())
                .collect();

            if parts.len() >= 7 {
                let state = State {
                    x: parts[0],
                    y: parts[1],
                    vx: parts[2],
                    vy: parts[3],
                    angle: parts[4],
                    angular_vel: parts[5],
                    fuel_remaining: parts[6],
                };
                if state_tx.send(state).is_err() {
                    break;
                }
            }
        }
    }
}

fn output_thread(cmd_rx: Receiver<(f64, f64)>) {
    let stdout = io::stdout();
    let mut handle = stdout.lock();

    loop {
        match cmd_rx.recv() {
            Ok((thrust, turn)) => {
                if writeln!(handle, "{} {}", thrust, turn).is_err() {
                    break;
                }
                let _ = handle.flush();
            }
            Err(_) => break,
        }
    }
}

fn control_loop(state_rx: Receiver<State>, cmd_tx: Sender<(f64, f64)>, env: Environment) {
    let mut last_state: Option<State> = None;
    let target_update_interval = Duration::from_millis(50);
    let mut last_update = Instant::now();

    loop {
        match state_rx.try_recv() {
            Ok(state) => {
                last_state = Some(state);
            }
            Err(TryRecvError::Empty) => {}
            Err(TryRecvError::Disconnected) => break,
        }

        if last_update.elapsed() >= target_update_interval {
            if let Some(ref state) = last_state {
                let (thrust, turn) = compute_control(state, &env);
                if cmd_tx.send((thrust, turn)).is_err() {
                    break;
                }
            } else {
                if cmd_tx.send((0.0, 0.0)).is_err() {
                    break;
                }
            }
            last_update = Instant::now();
        }

        thread::sleep(Duration::from_millis(1));
    }
}

fn compute_control(state: &State, env: &Environment) -> (f64, f64) {
    let dx = env.target_x - state.x;
    let dy = env.target_y - state.y;
    let distance = (dx * dx + dy * dy).sqrt();

    let target_angle = if distance > 5.0 {
        let angle_to_target = dy.atan2(dx);
        let desired_angle = angle_to_target - std::f64::consts::FRAC_PI_2;
        let hover_component = -0.3;
        desired_angle * 0.7 + hover_component * 0.3
    } else {
        0.0
    };

    let angle_error = target_angle - state.angle;
    let angle_error = ((angle_error + std::f64::consts::PI) % (2.0 * std::f64::consts::PI)) - std::f64::consts::PI;

    let turn_kp = 2.0;
    let turn_kd = 1.0;
    let turn = (angle_error * turn_kp - state.angular_vel * turn_kd).clamp(-1.0, 1.0);

    let vertical_speed_target = if distance > 10.0 {
        dy * 0.3
    } else if distance > 3.0 {
        dy * 0.2
    } else {
        -0.5
    };

    let vy_error = state.vy - vertical_speed_target;

    let cos_angle = state.angle.cos();
    let effective_thrust = if cos_angle.abs() > 0.1 {
        (env.gravity + vy_error * 0.8) / (cos_angle * env.max_thrust)
    } else {
        1.0
    };

    let horizontal_damping = if distance < 5.0 {
        -state.vx * 0.3
    } else {
        0.0
    };

    let sin_angle = state.angle.sin();
    let horizontal_correction = if sin_angle.abs() > 0.01 {
        horizontal_damping / sin_angle * 0.1
    } else {
        0.0
    };

    let thrust = (effective_thrust + horizontal_correction).clamp(0.0, 1.0);

    let final_thrust = if state.fuel_remaining > 1.0 {
        thrust
    } else if state.fuel_remaining > 0.1 {
        thrust * 0.5
    } else {
        0.0
    };

    (final_thrust, turn)
}
