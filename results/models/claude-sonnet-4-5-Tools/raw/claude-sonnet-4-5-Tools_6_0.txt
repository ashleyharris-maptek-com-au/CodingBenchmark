using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    const double MU = 398600.4418; // km^3/s^2
    const double EARTH_RADIUS = 6371.0; // km

    static void Main()
    {
        // Read input
        var firstLine = Console.ReadLine().Split();
        int N = int.Parse(firstLine[0]);
        double mu = double.Parse(firstLine[1]);

        // Read start orbit
        var startLine = Console.ReadLine().Split();
        var startOrbit = new Orbit
        {
            Position = new Vector3(
                double.Parse(startLine[0]),
                double.Parse(startLine[1]),
                double.Parse(startLine[2])
            ),
            Velocity = new Vector3(
                double.Parse(startLine[3]),
                double.Parse(startLine[4]),
                double.Parse(startLine[5])
            )
        };

        // Read station orbits
        var stations = new Orbit[N];
        for (int i = 0; i < N; i++)
        {
            var line = Console.ReadLine().Split();
            stations[i] = new Orbit
            {
                Position = new Vector3(
                    double.Parse(line[0]),
                    double.Parse(line[1]),
                    double.Parse(line[2])
                ),
                Velocity = new Vector3(
                    double.Parse(line[3]),
                    double.Parse(line[4]),
                    double.Parse(line[5])
                )
            };
        }

        // Solve TSP
        int[] solution = SolveTSP(startOrbit, stations, N);

        // Output
        Console.WriteLine(string.Join(" ", solution));
    }

    static int[] SolveTSP(Orbit start, Orbit[] stations, int N)
    {
        if (N == 1)
        {
            return new int[] { 0 };
        }

        if (N == 2)
        {
            // Try both orders
            double cost1 = EstimateDeltaV(start, stations[0]) + EstimateDeltaV(stations[0], stations[1]);
            double cost2 = EstimateDeltaV(start, stations[1]) + EstimateDeltaV(stations[1], stations[0]);
            return cost1 <= cost2 ? new int[] { 0, 1 } : new int[] { 1, 0 };
        }

        // Use adaptive strategy based on N
        if (N <= 10)
        {
            // Small: More intensive optimization
            return SolveWithHeuristic(start, stations, N, iterations: 2000);
        }
        else if (N <= 20)
        {
            // Medium: Moderate optimization
            return SolveWithHeuristic(start, stations, N, iterations: 1000);
        }
        else if (N <= 50)
        {
            // Large: Fast heuristic
            return SolveWithHeuristic(start, stations, N, iterations: 500);
        }
        else
        {
            // Extreme: Minimal optimization
            return SolveWithHeuristic(start, stations, N, iterations: 200);
        }
    }

    static int[] SolveWithHeuristic(Orbit start, Orbit[] stations, int N, int iterations)
    {
        // Try multiple greedy starts and pick best
        int[] bestTour = null;
        double bestCost = double.MaxValue;

        // Nearest neighbor from start
        var tour1 = NearestNeighbor(start, stations, N);
        var improved1 = TwoOpt(start, stations, tour1, iterations);
        double cost1 = CalculateTourCost(start, stations, improved1);
        if (cost1 < bestCost)
        {
            bestCost = cost1;
            bestTour = improved1;
        }

        // Try a few random starts for smaller problems
        if (N <= 30)
        {
            var random = new Random(42);
            for (int attempt = 0; attempt < Math.Min(3, iterations / 100); attempt++)
            {
                var tour = RandomTour(N, random);
                var improved = TwoOpt(start, stations, tour, iterations / 3);
                double cost = CalculateTourCost(start, stations, improved);
                if (cost < bestCost)
                {
                    bestCost = cost;
                    bestTour = improved;
                }
            }
        }

        return bestTour;
    }

    static int[] NearestNeighbor(Orbit start, Orbit[] stations, int N)
    {
        var tour = new List<int>();
        var unvisited = new HashSet<int>(Enumerable.Range(0, N));

        Orbit current = start;

        while (unvisited.Count > 0)
        {
            int nearest = -1;
            double minCost = double.MaxValue;

            foreach (int i in unvisited)
            {
                double cost = EstimateDeltaV(current, stations[i]);
                if (cost < minCost)
                {
                    minCost = cost;
                    nearest = i;
                }
            }

            tour.Add(nearest);
            unvisited.Remove(nearest);
            current = stations[nearest];
        }

        return tour.ToArray();
    }

    static int[] RandomTour(int N, Random random)
    {
        var tour = Enumerable.Range(0, N).ToArray();
        // Fisher-Yates shuffle
        for (int i = N - 1; i > 0; i--)
        {
            int j = random.Next(i + 1);
            int temp = tour[i];
            tour[i] = tour[j];
            tour[j] = temp;
        }
        return tour;
    }

    static int[] TwoOpt(Orbit start, Orbit[] stations, int[] tour, int maxIterations)
    {
        int N = tour.Length;
        var best = (int[])tour.Clone();
        double bestCost = CalculateTourCost(start, stations, best);

        bool improved = true;
        int iteration = 0;

        while (improved && iteration < maxIterations)
        {
            improved = false;
            iteration++;

            for (int i = 0; i < N - 1; i++)
            {
                for (int j = i + 1; j < N; j++)
                {
                    // Calculate cost change without full recalculation
                    double oldCost = 0;
                    double newCost = 0;

                    // Cost of edges being removed
                    if (i == 0)
                    {
                        oldCost += EstimateDeltaV(start, stations[best[i]]);
                        newCost += EstimateDeltaV(start, stations[best[j]]);
                    }
                    else
                    {
                        oldCost += EstimateDeltaV(stations[best[i - 1]], stations[best[i]]);
                        newCost += EstimateDeltaV(stations[best[i - 1]], stations[best[j]]);
                    }

                    if (j < N - 1)
                    {
                        oldCost += EstimateDeltaV(stations[best[j]], stations[best[j + 1]]);
                        newCost += EstimateDeltaV(stations[best[i]], stations[best[j + 1]]);
                    }

                    if (newCost < oldCost)
                    {
                        // Perform 2-opt swap
                        Array.Reverse(best, i, j - i + 1);
                        bestCost = CalculateTourCost(start, stations, best);
                        improved = true;
                    }
                }
            }
        }

        return best;
    }

    static double CalculateTourCost(Orbit start, Orbit[] stations, int[] tour)
    {
        double totalCost = 0;
        Orbit current = start;

        foreach (int i in tour)
        {
            totalCost += EstimateDeltaV(current, stations[i]);
            current = stations[i];
        }

        return totalCost;
    }

    static double EstimateDeltaV(Orbit from, Orbit to)
    {
        // Estimate delta-V using orbital mechanics principles

        // Method 1: Velocity difference (simple but effective for similar orbits)
        double dvVelocity = Vector3.Distance(from.Velocity, to.Velocity);

        // Method 2: Consider orbital energy difference
        double r1 = from.Position.Magnitude();
        double r2 = to.Position.Magnitude();
        double v1 = from.Velocity.Magnitude();
        double v2 = to.Velocity.Magnitude();

        // Specific orbital energy
        double energy1 = v1 * v1 / 2.0 - MU / r1;
        double energy2 = v2 * v2 / 2.0 - MU / r2;
        double energyDiff = Math.Abs(energy2 - energy1);

        // Combine metrics
        // Velocity difference is primary, energy difference is secondary
        return dvVelocity + energyDiff * 0.0001;
    }

    struct Vector3
    {
        public double X, Y, Z;

        public Vector3(double x, double y, double z)
        {
            X = x; Y = y; Z = z;
        }

        public static double Distance(Vector3 a, Vector3 b)
        {
            double dx = a.X - b.X;
            double dy = a.Y - b.Y;
            double dz = a.Z - b.Z;
            return Math.Sqrt(dx * dx + dy * dy + dz * dz);
        }

        public double Magnitude()
        {
            return Math.Sqrt(X * X + Y * Y + Z * Z);
        }

        public static Vector3 operator -(Vector3 a, Vector3 b)
        {
            return new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        }

        public static double Dot(Vector3 a, Vector3 b)
        {
            return a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        }
    }

    class Orbit
    {
        public Vector3 Position;
        public Vector3 Velocity;
    }
}
