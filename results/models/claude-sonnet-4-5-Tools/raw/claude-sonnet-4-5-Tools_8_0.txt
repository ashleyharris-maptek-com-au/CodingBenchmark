use std::collections::VecDeque;
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read dimensions
    let first_line = lines.next().unwrap().unwrap();
    let dims: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let (h, w) = (dims[0], dims[1]);

    // Read maze as flat array for memory efficiency
    let mut maze = Vec::with_capacity(h * w);
    let mut start = 0;
    let mut end = 0;

    for i in 0..h {
        let line = lines.next().unwrap().unwrap();
        let chars: Vec<char> = line.chars().collect();
        for j in 0..w {
            let c = if j < chars.len() { chars[j] } else { ' ' };
            let idx = i * w + j;
            if c == 'A' {
                start = idx;
            } else if c == 'B' {
                end = idx;
            }
            maze.push(c);
        }
    }

    // BFS pathfinding
    let mut queue = VecDeque::new();
    let mut visited = vec![false; h * w];
    let mut parent = vec![usize::MAX; h * w];

    queue.push_back(start);
    visited[start] = true;

    let mut found = false;

    while let Some(pos) = queue.pop_front() {
        if pos == end {
            found = true;
            break;
        }

        let y = pos / w;
        let x = pos % w;

        // Check all 4 directions (up, down, left, right)
        // Up
        if y > 0 {
            let npos = pos - w;
            if !visited[npos] && maze[npos] != '#' {
                visited[npos] = true;
                parent[npos] = pos;
                queue.push_back(npos);
            }
        }
        // Down
        if y + 1 < h {
            let npos = pos + w;
            if !visited[npos] && maze[npos] != '#' {
                visited[npos] = true;
                parent[npos] = pos;
                queue.push_back(npos);
            }
        }
        // Left
        if x > 0 {
            let npos = pos - 1;
            if !visited[npos] && maze[npos] != '#' {
                visited[npos] = true;
                parent[npos] = pos;
                queue.push_back(npos);
            }
        }
        // Right
        if x + 1 < w {
            let npos = pos + 1;
            if !visited[npos] && maze[npos] != '#' {
                visited[npos] = true;
                parent[npos] = pos;
                queue.push_back(npos);
            }
        }
    }

    if !found {
        println!("0");
        return;
    }

    // Reconstruct path from end to start
    let mut path = Vec::new();
    let mut current = end;

    loop {
        path.push(current);
        if current == start {
            break;
        }
        current = parent[current];
    }

    path.reverse();

    // Output path (x=column, y=row, 0-indexed)
    println!("{}", path.len());
    for pos in path {
        let y = pos / w;
        let x = pos % w;
        println!("{} {}", x, y);
    }
}
