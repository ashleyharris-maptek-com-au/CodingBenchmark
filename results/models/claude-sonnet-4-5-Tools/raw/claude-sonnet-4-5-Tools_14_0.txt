using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        var input = Console.In.ReadToEnd().Split('\n');
        var firstLine = input[0].Trim().Split(' ');
        int H = int.Parse(firstLine[0]);
        int W = int.Parse(firstLine[1]);

        char[][] board = new char[H][];
        for (int i = 0; i < H; i++)
        {
            string line = i + 1 < input.Length ? input[i + 1] : "";
            board[i] = new char[W];
            for (int j = 0; j < W; j++)
            {
                board[i][j] = j < line.Length ? line[j] : ' ';
            }
        }

        var solver = new MinesweeperSolver(board, H, W);
        var safeCells = solver.FindSafeCells();

        Console.WriteLine(safeCells.Count);
        foreach (var cell in safeCells)
        {
            Console.WriteLine($"{cell.x} {cell.y}");
        }
    }
}

class MinesweeperSolver
{
    private char[][] board;
    private int H, W;
    private HashSet<(int x, int y)> knownMines;
    private HashSet<(int x, int y)> knownSafe;
    private Queue<(int x, int y)> workQueue;

    public MinesweeperSolver(char[][] board, int H, int W)
    {
        this.board = board;
        this.H = H;
        this.W = W;
        this.knownMines = new HashSet<(int x, int y)>();
        this.knownSafe = new HashSet<(int x, int y)>();
        this.workQueue = new Queue<(int x, int y)>();

        // Initialize: mark all flagged mines and queue all numbered cells
        for (int y = 0; y < H; y++)
        {
            for (int x = 0; x < W; x++)
            {
                if (board[y][x] == '*')
                {
                    knownMines.Add((x, y));
                }
                else if (board[y][x] >= '0' && board[y][x] <= '8')
                {
                    workQueue.Enqueue((x, y));
                }
            }
        }
    }

    public List<(int x, int y)> FindSafeCells()
    {
        // Use adaptive strategy based on board size
        bool useOptimizedApproach = H * W > 100000;

        if (useOptimizedApproach)
        {
            SolveWithWorkQueue();
        }
        else
        {
            SolveWithFullScan();
        }

        // Return only cells that are currently unknown on the board
        var result = new List<(int x, int y)>();
        foreach (var (x, y) in knownSafe)
        {
            if (board[y][x] == ' ')
            {
                result.Add((x, y));
            }
        }

        result.Sort();
        return result;
    }

    private void SolveWithWorkQueue()
    {
        var processed = new HashSet<(int x, int y)>();

        while (workQueue.Count > 0)
        {
            var (x, y) = workQueue.Dequeue();

            if (ProcessCell(x, y))
            {
                // If we found new information, re-queue neighbors
                foreach (var (nx, ny) in GetNeighbors(x, y))
                {
                    if (board[ny][nx] >= '0' && board[ny][nx] <= '8' && !processed.Contains((nx, ny)))
                    {
                        workQueue.Enqueue((nx, ny));
                        processed.Remove((nx, ny));
                    }
                }
            }
            processed.Add((x, y));
        }
    }

    private void SolveWithFullScan()
    {
        bool changed = true;
        int iterations = 0;
        int maxIterations = 100;

        while (changed && iterations < maxIterations)
        {
            changed = false;
            iterations++;

            for (int y = 0; y < H; y++)
            {
                for (int x = 0; x < W; x++)
                {
                    char cell = board[y][x];
                    if (cell >= '0' && cell <= '8')
                    {
                        if (ProcessCell(x, y))
                            changed = true;
                    }
                }
            }
        }
    }

    private bool ProcessCell(int x, int y)
    {
        int mineCount = board[y][x] - '0';
        var neighbors = GetNeighbors(x, y);

        var unknownNeighbors = new List<(int x, int y)>();
        int adjacentMines = 0;

        foreach (var (nx, ny) in neighbors)
        {
            if (knownMines.Contains((nx, ny)) || board[ny][nx] == '*')
            {
                adjacentMines++;
            }
            else if (board[ny][nx] == ' ' && !knownSafe.Contains((nx, ny)))
            {
                unknownNeighbors.Add((nx, ny));
            }
        }

        bool changed = false;

        // If we've found all mines, remaining unknowns are safe
        if (adjacentMines == mineCount && unknownNeighbors.Count > 0)
        {
            foreach (var unknown in unknownNeighbors)
            {
                if (knownSafe.Add(unknown))
                    changed = true;
            }
        }

        // If remaining unknowns equal remaining mines, they're all mines
        int remainingMines = mineCount - adjacentMines;
        if (remainingMines > 0 && unknownNeighbors.Count == remainingMines)
        {
            foreach (var unknown in unknownNeighbors)
            {
                if (knownMines.Add(unknown))
                    changed = true;
            }
        }

        return changed;
    }

    private List<(int x, int y)> GetNeighbors(int x, int y)
    {
        var neighbors = new List<(int x, int y)>(8);
        for (int dy = -1; dy <= 1; dy++)
        {
            for (int dx = -1; dx <= 1; dx++)
            {
                if (dx == 0 && dy == 0) continue;
                int nx = x + dx;
                int ny = y + dy;
                if (nx >= 0 && nx < W && ny >= 0 && ny < H)
                    neighbors.Add((nx, ny));
            }
        }
        return neighbors;
    }
}
