using System;
using System.Collections.Generic;
using System.Linq;
using System.Numerics;

class Program
{
    static void Main()
    {
        // Parse input
        var line = Console.ReadLine().Split();
        int V = int.Parse(line[0]);
        int F = int.Parse(line[1]);

        Vector3[] vertices = new Vector3[V];
        for (int i = 0; i < V; i++)
        {
            var parts = Console.ReadLine().Split();
            vertices[i] = new Vector3(float.Parse(parts[0]), float.Parse(parts[1]), float.Parse(parts[2]));
        }

        int[][] faces = new int[F][];
        for (int i = 0; i < F; i++)
        {
            var parts = Console.ReadLine().Split();
            faces[i] = new int[] { int.Parse(parts[0]), int.Parse(parts[1]), int.Parse(parts[2]) };
        }

        var lastLine = Console.ReadLine().Split();
        float W = float.Parse(lastLine[0]);
        float H = float.Parse(lastLine[1]);
        float D = float.Parse(lastLine[2]);

        // Solve
        var solver = new BinPackingSolver(vertices, faces, W, H, D);
        var placements = solver.Solve();

        // Output
        Console.WriteLine(placements.Count);
        foreach (var p in placements)
        {
            Console.WriteLine($"{p.Translation.X} {p.Translation.Y} {p.Translation.Z} {p.Rotation.W} {p.Rotation.X} {p.Rotation.Y} {p.Rotation.Z}");
        }
    }
}

struct Placement
{
    public Vector3 Translation;
    public Quaternion Rotation;
}

class BinPackingSolver
{
    private Vector3[] originalVertices;
    private int[][] faces;
    private float containerW, containerH, containerD;
    private List<Placement> placements = new List<Placement>();

    public BinPackingSolver(Vector3[] vertices, int[][] faces, float w, float h, float d)
    {
        this.originalVertices = vertices;
        this.faces = faces;
        this.containerW = w;
        this.containerH = h;
        this.containerD = d;
    }

    public List<Placement> Solve()
    {
        // Calculate bounding box of original polyhedron
        var bounds = GetBounds(originalVertices);
        Vector3 size = bounds.Max - bounds.Min;

        // Determine scale and choose strategy
        float containerVolume = containerW * containerH * containerD;
        float polyVolume = size.X * size.Y * size.Z;

        // Generate rotation candidates
        var rotations = GenerateRotations();

        // Find best rotation (smallest bounding box)
        Quaternion bestRotation = Quaternion.Identity;
        Vector3 bestSize = size;
        float minVolume = polyVolume;

        foreach (var rot in rotations)
        {
            var rotated = ApplyRotation(originalVertices, rot);
            var rotBounds = GetBounds(rotated);
            Vector3 rotSize = rotBounds.Max - rotBounds.Min;
            float rotVolume = rotSize.X * rotSize.Y * rotSize.Z;
            if (rotVolume < minVolume)
            {
                minVolume = rotVolume;
                bestRotation = rot;
                bestSize = rotSize;
            }
        }

        // Adaptive strategy based on container size
        if (containerVolume > 1000000) // Large/Extreme scale
        {
            return SolveGridBased(bestRotation, bestSize);
        }
        else
        {
            return SolveGridBased(bestRotation, bestSize);
        }
    }

    private List<Quaternion> GenerateRotations()
    {
        var rotations = new List<Quaternion>();

        // Identity
        rotations.Add(Quaternion.Identity);

        // 90° rotations around each axis
        rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitX, MathF.PI / 2));
        rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitX, MathF.PI));
        rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitX, 3 * MathF.PI / 2));

        rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitY, MathF.PI / 2));
        rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitY, MathF.PI));
        rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitY, 3 * MathF.PI / 2));

        rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitZ, MathF.PI / 2));
        rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitZ, MathF.PI));
        rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitZ, 3 * MathF.PI / 2));

        return rotations;
    }

    private List<Placement> SolveGridBased(Quaternion rotation, Vector3 size)
    {
        var placements = new List<Placement>();

        // Add small margin to avoid floating point issues
        float margin = 0.01f;
        float sizeX = size.X + margin;
        float sizeY = size.Y + margin;
        float sizeZ = size.Z + margin;

        // Get rotated vertices to calculate proper offset
        var rotatedVerts = ApplyRotation(originalVertices, rotation);
        var rotBounds = GetBounds(rotatedVerts);
        Vector3 offset = -rotBounds.Min; // Offset to move min corner to origin

        // Grid-based placement
        int nx = (int)(containerW / sizeX);
        int ny = (int)(containerH / sizeY);
        int nz = (int)(containerD / sizeZ);

        for (int iz = 0; iz < nz; iz++)
        {
            for (int iy = 0; iy < ny; iy++)
            {
                for (int ix = 0; ix < nx; ix++)
                {
                    Vector3 pos = new Vector3(
                        ix * sizeX + offset.X,
                        iy * sizeY + offset.Y,
                        iz * sizeZ + offset.Z
                    );

                    // Verify placement is within bounds
                    var testVerts = ApplyTransformation(originalVertices, pos, rotation);
                    if (IsInsideContainer(testVerts))
                    {
                        placements.Add(new Placement
                        {
                            Translation = pos,
                            Rotation = rotation
                        });
                    }
                }
            }
        }

        return placements;
    }

    private Vector3[] ApplyRotation(Vector3[] verts, Quaternion rotation)
    {
        var result = new Vector3[verts.Length];
        for (int i = 0; i < verts.Length; i++)
        {
            result[i] = Vector3.Transform(verts[i], rotation);
        }
        return result;
    }

    private Vector3[] ApplyTransformation(Vector3[] verts, Vector3 translation, Quaternion rotation)
    {
        var result = new Vector3[verts.Length];
        for (int i = 0; i < verts.Length; i++)
        {
            result[i] = Vector3.Transform(verts[i], rotation) + translation;
        }
        return result;
    }

    private (Vector3 Min, Vector3 Max) GetBounds(Vector3[] verts)
    {
        if (verts.Length == 0)
            return (Vector3.Zero, Vector3.Zero);

        Vector3 min = verts[0];
        Vector3 max = verts[0];

        foreach (var v in verts)
        {
            min = Vector3.Min(min, v);
            max = Vector3.Max(max, v);
        }

        return (min, max);
    }

    private bool IsInsideContainer(Vector3[] verts)
    {
        foreach (var v in verts)
        {
            if (v.X < 0 || v.X > containerW ||
                v.Y < 0 || v.Y > containerH ||
                v.Z < 0 || v.Z > containerD)
            {
                return false;
            }
        }
        return true;
    }
}
