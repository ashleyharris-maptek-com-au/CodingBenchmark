#include <iostream>
#include <vector>
#include <deque>
#include <queue>
#include <set>
#include <map>
#include <algorithm>
#include <cmath>
#include <limits>

using namespace std;

struct Position {
    vector<int> coords;

    Position() {}
    Position(int dims) : coords(dims, 0) {}
    Position(const vector<int>& c) : coords(c) {}

    bool operator<(const Position& other) const {
        return coords < other.coords;
    }

    bool operator==(const Position& other) const {
        return coords == other.coords;
    }

    Position move(int axis, int dir) const {
        Position result = *this;
        result.coords[axis] += dir;
        return result;
    }

    int manhattanDist(const Position& other) const {
        int dist = 0;
        for (size_t i = 0; i < coords.size(); i++) {
            dist += abs(coords[i] - other.coords[i]);
        }
        return dist;
    }
};

class SnakeGame {
private:
    int dimensions;
    int maxTurns;
    vector<int> bounds;
    deque<Position> snake;
    set<Position> food;
    set<Position> obstacles;
    set<Position> snakeSet;
    int turnCount;

    bool isValid(const Position& pos) const {
        for (int i = 0; i < dimensions; i++) {
            if (pos.coords[i] < 0 || pos.coords[i] >= bounds[i]) {
                return false;
            }
        }
        return true;
    }

    bool isOccupied(const Position& pos) const {
        return obstacles.count(pos) > 0 || snakeSet.count(pos) > 0;
    }

    struct State {
        Position pos;
        deque<Position> snake;
        set<Position> snakeSet;
        int dist;

        bool operator>(const State& other) const {
            return dist > other.dist;
        }
    };

    // BFS to find path to target, considering snake body changes
    vector<pair<int, int>> findPath(const Position& target) {
        map<Position, pair<int, int>> parent;
        queue<Position> q;
        set<Position> visited;

        Position head = snake.front();
        q.push(head);
        visited.insert(head);

        while (!q.empty()) {
            Position current = q.front();
            q.pop();

            if (current == target) {
                // Reconstruct path
                vector<pair<int, int>> path;
                Position pos = target;
                while (parent.count(pos)) {
                    path.push_back(parent[pos]);
                    pos = pos.move(parent[pos].first, -parent[pos].second);
                }
                reverse(path.begin(), path.end());
                return path;
            }

            // Try all possible moves
            for (int axis = 0; axis < dimensions; axis++) {
                for (int dir : {-1, 1}) {
                    Position next = current.move(axis, dir);

                    if (!isValid(next) || visited.count(next)) {
                        continue;
                    }

                    // Check if this move would be safe
                    // We need to simulate the snake movement
                    bool wouldCollide = false;

                    // Check obstacles
                    if (obstacles.count(next)) {
                        wouldCollide = true;
                    }

                    // Check snake body (simplified - actual snake body will shrink as we move)
                    // For now, allow moving to positions that will be vacated
                    if (!wouldCollide && snakeSet.count(next)) {
                        // This is part of current snake, might be ok if it's the tail
                        if (next != snake.back() || food.count(target)) {
                            wouldCollide = true;
                        }
                    }

                    if (!wouldCollide) {
                        visited.insert(next);
                        parent[next] = {axis, dir};
                        q.push(next);
                    }
                }
            }
        }

        return {}; // No path found
    }

    // Find nearest food using Manhattan distance
    Position findNearestFood() {
        Position head = snake.front();
        Position nearest;
        int minDist = numeric_limits<int>::max();

        for (const Position& f : food) {
            int dist = head.manhattanDist(f);
            if (dist < minDist) {
                minDist = dist;
                nearest = f;
            }
        }

        return nearest;
    }

    // Greedy move towards target
    pair<int, int> greedyMove(const Position& target) {
        Position head = snake.front();
        int bestAxis = -1;
        int bestDir = 0;
        int bestDist = head.manhattanDist(target);

        // Try each possible move
        for (int axis = 0; axis < dimensions; axis++) {
            for (int dir : {-1, 1}) {
                Position next = head.move(axis, dir);

                if (!isValid(next) || isOccupied(next)) {
                    continue;
                }

                int dist = next.manhattanDist(target);
                if (dist < bestDist) {
                    bestDist = dist;
                    bestAxis = axis;
                    bestDir = dir;
                }
            }
        }

        // If no improving move, try any valid move
        if (bestAxis == -1) {
            for (int axis = 0; axis < dimensions; axis++) {
                for (int dir : {-1, 1}) {
                    Position next = head.move(axis, dir);

                    if (isValid(next) && !isOccupied(next)) {
                        return {axis, dir};
                    }
                }
            }
        }

        return {bestAxis, bestDir};
    }

public:
    SnakeGame() : turnCount(0) {}

    void readInput() {
        cin >> dimensions >> maxTurns;

        bounds.resize(dimensions);
        for (int i = 0; i < dimensions; i++) {
            cin >> bounds[i];
        }

        int snakeLength;
        cin >> snakeLength;

        Position head(dimensions);
        for (int i = 0; i < dimensions; i++) {
            cin >> head.coords[i];
        }

        // Initialize snake
        snake.push_back(head);
        snakeSet.insert(head);

        // For initial length > 1, extend backwards arbitrarily
        for (int i = 1; i < snakeLength; i++) {
            Position tail = snake.back();
            bool placed = false;

            for (int axis = 0; axis < dimensions && !placed; axis++) {
                for (int dir : {-1, 1}) {
                    Position next = tail.move(axis, dir);
                    if (isValid(next) && snakeSet.count(next) == 0) {
                        snake.push_back(next);
                        snakeSet.insert(next);
                        placed = true;
                        break;
                    }
                }
            }
        }

        int F;
        cin >> F;
        for (int i = 0; i < F; i++) {
            Position f(dimensions);
            for (int j = 0; j < dimensions; j++) {
                cin >> f.coords[j];
            }
            food.insert(f);
        }

        int O;
        cin >> O;
        for (int i = 0; i < O; i++) {
            Position o(dimensions);
            for (int j = 0; j < dimensions; j++) {
                cin >> o.coords[j];
            }
            obstacles.insert(o);
        }
    }

    bool makeMove(int axis, int dir) {
        Position head = snake.front();
        Position newHead = head.move(axis, dir);

        // Check validity
        if (!isValid(newHead)) {
            return false;
        }

        // Check collision with obstacle
        if (obstacles.count(newHead)) {
            return false;
        }

        // Check collision with self
        if (snakeSet.count(newHead) && newHead != snake.back()) {
            return false;
        }

        // Move snake
        snake.push_front(newHead);
        snakeSet.insert(newHead);

        // Check if we ate food
        if (food.count(newHead)) {
            food.erase(newHead);
        } else {
            // Remove tail
            Position tail = snake.back();
            snake.pop_back();
            snakeSet.erase(tail);
        }

        turnCount++;
        return true;
    }

    void play() {
        while (!food.empty() && turnCount < maxTurns) {
            Position target = findNearestFood();

            // Try to find a path
            vector<pair<int, int>> path = findPath(target);

            pair<int, int> move;
            if (!path.empty()) {
                move = path[0];
            } else {
                // Fallback to greedy move
                move = greedyMove(target);
            }

            if (move.first == -1) {
                // No valid move, game over
                break;
            }

            cout << move.first << " " << move.second << endl;

            if (!makeMove(move.first, move.second)) {
                // Invalid move, game over
                break;
            }
        }
    }
};

int main() {
    SnakeGame game;
    game.readInput();
    game.play();

    return 0;
}
