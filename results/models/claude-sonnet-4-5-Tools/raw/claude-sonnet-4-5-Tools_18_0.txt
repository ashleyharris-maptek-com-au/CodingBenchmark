#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>

using namespace std;

const int INF = 2000000000;

struct Edge {
    int to, cap, flow;
    size_t rev;
};

class MaxFlow {
    vector<vector<Edge>> graph;
    vector<int> level, iter;
    int n;

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q;
        level[s] = 0;
        q.push(s);

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (const auto& e : graph[v]) {
                if (level[e.to] < 0 && e.flow < e.cap) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }
        return level[t] >= 0;
    }

    int dfs(int v, int t, int pushed) {
        if (v == t) return pushed;

        for (int& cid = iter[v]; cid < (int)graph[v].size(); cid++) {
            Edge& e = graph[v][cid];
            if (level[v] + 1 != level[e.to] || e.cap <= e.flow)
                continue;

            int tr = dfs(e.to, t, min(pushed, e.cap - e.flow));
            if (tr > 0) {
                e.flow += tr;
                graph[e.to][e.rev].flow -= tr;
                return tr;
            }
        }
        return 0;
    }

public:
    MaxFlow(int n) : n(n), graph(n), level(n), iter(n) {}

    void addEdge(int from, int to, int cap) {
        graph[from].push_back({to, cap, 0, graph[to].size()});
        graph[to].push_back({from, cap, 0, graph[from].size() - 1});
    }

    int maxflow(int s, int t) {
        for (auto& edges : graph) {
            for (auto& e : edges) {
                e.flow = 0;
            }
        }

        int flow = 0;
        while (bfs(s, t)) {
            fill(iter.begin(), iter.end(), 0);
            while (int pushed = dfs(s, t, INF)) {
                flow += pushed;
            }
        }
        return flow;
    }

    vector<bool> minCut(int s) {
        vector<bool> reachable(n);
        queue<int> q;
        q.push(s);
        reachable[s] = true;

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (const auto& e : graph[v]) {
                if (!reachable[e.to] && e.flow < e.cap) {
                    reachable[e.to] = true;
                    q.push(e.to);
                }
            }
        }
        return reachable;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<pair<int, int>> edges(m);
    MaxFlow flow(n);

    for (int i = 0; i < m; i++) {
        cin >> edges[i].first >> edges[i].second;
        flow.addEdge(edges[i].first, edges[i].second, 1);
    }

    int min_cut = INF;
    int best_sink = -1;

    for (int t = 1; t < n; t++) {
        int cut_val = flow.maxflow(0, t);
        if (cut_val < min_cut) {
            min_cut = cut_val;
            best_sink = t;
        }
    }

    flow.maxflow(0, best_sink);
    auto side = flow.minCut(0);

    vector<pair<int, int>> result;
    for (const auto& e : edges) {
        if (side[e.first] != side[e.second]) {
            result.push_back(e);
        }
    }

    cout << result.size() << "\n";
    for (const auto& e : result) {
        cout << e.first << " " << e.second << "\n";
    }

    return 0;
}
