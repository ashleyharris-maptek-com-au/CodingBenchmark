using System;
using System.Linq;

class TetrisPlayer
{
    static void Main()
    {
        var game = new TetrisGame();
        game.Play();
    }
}

class TetrisGame
{
    private int width;
    private int height;
    private int numPieces;
    private int[,] board;

    public void Play()
    {
        // Read game configuration
        var firstLine = Console.ReadLine().Split();
        width = int.Parse(firstLine[0]);
        height = int.Parse(firstLine[1]);
        numPieces = int.Parse(firstLine[2]);

        board = new int[height, width];

        // Play each piece
        for (int i = 0; i < numPieces; i++)
        {
            var pieceType = Console.ReadLine().Trim();
            var (rotation, column) = FindBestPlacement(pieceType);

            Console.WriteLine($"{rotation} {column}");
            Console.Out.Flush();

            // Read result and update board
            var result = Console.ReadLine().Trim();
            PlacePiece(pieceType, rotation, column);

            if (result.StartsWith("cleared_"))
            {
                int linesCleared = int.Parse(result.Substring(8));
                ClearLines(linesCleared);
            }
            else if (result == "gameover")
            {
                break;
            }
        }
    }

    private (int rotation, int column) FindBestPlacement(string pieceType)
    {
        int bestRotation = 0;
        int bestColumn = 0;
        double bestScore = double.MinValue;

        var piece = GetPiece(pieceType);

        // Try all rotations
        for (int rotation = 0; rotation < 4; rotation++)
        {
            var rotatedPiece = RotatePiece(piece, rotation);

            // Try all columns
            for (int column = 0; column <= width - rotatedPiece.GetLength(1); column++)
            {
                if (CanPlace(rotatedPiece, column))
                {
                    // Simulate placement
                    var testBoard = CloneBoard(board);
                    int dropRow = SimulateDrop(rotatedPiece, column, testBoard);

                    if (dropRow >= 0)
                    {
                        PlacePieceOnBoard(rotatedPiece, dropRow, column, testBoard);
                        int linesCleared = CountCompletedLines(testBoard);

                        double score = EvaluateBoard(testBoard, linesCleared);

                        if (score > bestScore)
                        {
                            bestScore = score;
                            bestRotation = rotation;
                            bestColumn = column;
                        }
                    }
                }
            }
        }

        return (bestRotation, bestColumn);
    }

    private int[,] CloneBoard(int[,] original)
    {
        int[,] clone = new int[height, width];
        for (int r = 0; r < height; r++)
        {
            for (int c = 0; c < width; c++)
            {
                clone[r, c] = original[r, c];
            }
        }
        return clone;
    }

    private double EvaluateBoard(int[,] testBoard, int linesCleared)
    {
        double score = 0;

        // Reward lines cleared
        score += linesCleared * 1000;

        // Calculate column heights
        int[] columnHeights = new int[width];
        for (int col = 0; col < width; col++)
        {
            for (int row = 0; row < height; row++)
            {
                if (testBoard[row, col] == 1)
                {
                    columnHeights[col] = height - row;
                    break;
                }
            }
        }

        // Calculate aggregate height
        int aggregateHeight = columnHeights.Sum();
        score -= aggregateHeight * 0.5;

        // Calculate holes
        int holes = 0;
        for (int col = 0; col < width; col++)
        {
            bool foundBlock = false;
            for (int row = 0; row < height; row++)
            {
                if (testBoard[row, col] == 1)
                {
                    foundBlock = true;
                }
                else if (foundBlock)
                {
                    holes++;
                }
            }
        }
        score -= holes * 35;

        // Calculate bumpiness
        int bumpiness = 0;
        for (int col = 0; col < width - 1; col++)
        {
            bumpiness += Math.Abs(columnHeights[col] - columnHeights[col + 1]);
        }
        score -= bumpiness * 2;

        // Penalize maximum height
        int maxHeight = columnHeights.Max();
        score -= maxHeight * maxHeight * 0.1;

        return score;
    }

    private int CountCompletedLines(int[,] testBoard)
    {
        int count = 0;
        for (int row = 0; row < height; row++)
        {
            bool complete = true;
            for (int col = 0; col < width; col++)
            {
                if (testBoard[row, col] == 0)
                {
                    complete = false;
                    break;
                }
            }
            if (complete) count++;
        }
        return count;
    }

    private bool CanPlace(int[,] piece, int column)
    {
        return column >= 0 && column + piece.GetLength(1) <= width;
    }

    private int SimulateDrop(int[,] piece, int column, int[,] testBoard)
    {
        int pieceHeight = piece.GetLength(0);
        int pieceWidth = piece.GetLength(1);

        for (int row = 0; row < height; row++)
        {
            // Check if piece would collide
            bool collision = false;

            if (row + pieceHeight > height)
            {
                return row - 1;
            }

            for (int pr = 0; pr < pieceHeight; pr++)
            {
                for (int pc = 0; pc < pieceWidth; pc++)
                {
                    if (piece[pr, pc] == 1)
                    {
                        int boardRow = row + pr;
                        int boardCol = column + pc;

                        if (boardRow >= height || testBoard[boardRow, boardCol] == 1)
                        {
                            collision = true;
                            break;
                        }
                    }
                }
                if (collision) break;
            }

            if (collision)
            {
                return row - 1;
            }
        }

        return height - pieceHeight;
    }

    private void PlacePieceOnBoard(int[,] piece, int row, int column, int[,] targetBoard)
    {
        int pieceHeight = piece.GetLength(0);
        int pieceWidth = piece.GetLength(1);

        for (int pr = 0; pr < pieceHeight; pr++)
        {
            for (int pc = 0; pc < pieceWidth; pc++)
            {
                if (piece[pr, pc] == 1)
                {
                    int boardRow = row + pr;
                    int boardCol = column + pc;

                    if (boardRow >= 0 && boardRow < height && boardCol >= 0 && boardCol < width)
                    {
                        targetBoard[boardRow, boardCol] = 1;
                    }
                }
            }
        }
    }

    private void PlacePiece(string pieceType, int rotation, int column)
    {
        var piece = GetPiece(pieceType);
        var rotatedPiece = RotatePiece(piece, rotation);
        int dropRow = SimulateDrop(rotatedPiece, column, board);

        if (dropRow >= 0)
        {
            PlacePieceOnBoard(rotatedPiece, dropRow, column, board);
        }
    }

    private void ClearLines(int linesCleared)
    {
        // Find and clear completed lines from bottom up
        int cleared = 0;
        for (int row = height - 1; row >= 0 && cleared < linesCleared; row--)
        {
            bool complete = true;
            for (int col = 0; col < width; col++)
            {
                if (board[row, col] == 0)
                {
                    complete = false;
                    break;
                }
            }

            if (complete)
            {
                // Shift rows down
                for (int r = row; r > 0; r--)
                {
                    for (int col = 0; col < width; col++)
                    {
                        board[r, col] = board[r - 1, col];
                    }
                }

                // Clear top row
                for (int col = 0; col < width; col++)
                {
                    board[0, col] = 0;
                }

                cleared++;
                row++; // Check this row again after shift
            }
        }
    }

    private int[,] GetPiece(string pieceType)
    {
        switch (pieceType)
        {
            case "I":
                return new int[,] { { 1, 1, 1, 1 } };

            case "O":
                return new int[,] {
                    { 1, 1 },
                    { 1, 1 }
                };

            case "T":
                return new int[,] {
                    { 0, 1, 0 },
                    { 1, 1, 1 }
                };

            case "S":
                return new int[,] {
                    { 0, 1, 1 },
                    { 1, 1, 0 }
                };

            case "Z":
                return new int[,] {
                    { 1, 1, 0 },
                    { 0, 1, 1 }
                };

            case "J":
                return new int[,] {
                    { 1, 0, 0 },
                    { 1, 1, 1 }
                };

            case "L":
                return new int[,] {
                    { 0, 0, 1 },
                    { 1, 1, 1 }
                };

            default:
                return new int[,] { { 1 } };
        }
    }

    private int[,] RotatePiece(int[,] piece, int rotation)
    {
        rotation = rotation % 4;
        var result = piece;

        for (int i = 0; i < rotation; i++)
        {
            result = RotateOnce(result);
        }

        return result;
    }

    private int[,] RotateOnce(int[,] piece)
    {
        int rows = piece.GetLength(0);
        int cols = piece.GetLength(1);
        int[,] rotated = new int[cols, rows];

        for (int r = 0; r < rows; r++)
        {
            for (int c = 0; c < cols; c++)
            {
                rotated[c, rows - 1 - r] = piece[r, c];
            }
        }

        return rotated;
    }
}
