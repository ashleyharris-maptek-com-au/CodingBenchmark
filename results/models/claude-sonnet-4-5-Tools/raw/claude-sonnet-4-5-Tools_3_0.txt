use std::io::{self, BufRead};
use std::f64::consts::PI;

const OPTIMAL_DISTANCE: f64 = 100.0;
const AREA_MULTIPLIER: f64 = 10000.0;

#[derive(Clone, Debug)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    fn new(x: f64, y: f64) -> Self {
        Vec2 { x, y }
    }

    fn zero() -> Self {
        Vec2 { x: 0.0, y: 0.0 }
    }

    fn length(&self) -> f64 {
        (self.x * self.x + self.y * self.y).sqrt()
    }

    fn add(&self, other: &Vec2) -> Vec2 {
        Vec2::new(self.x + other.x, self.y + other.y)
    }

    fn sub(&self, other: &Vec2) -> Vec2 {
        Vec2::new(self.x - other.x, self.y - other.y)
    }

    fn mul(&self, scalar: f64) -> Vec2 {
        Vec2::new(self.x * scalar, self.y * scalar)
    }
}

struct Graph {
    n: usize,
    edges: Vec<(usize, usize)>,
    adj: Vec<Vec<usize>>,
}

impl Graph {
    fn new(n: usize, edges: Vec<(usize, usize)>) -> Self {
        let mut adj = vec![Vec::new(); n];
        for &(u, v) in &edges {
            if u != v && u < n && v < n {
                adj[u].push(v);
                adj[v].push(u);
            }
        }
        Graph { n, edges, adj }
    }

    fn get_components(&self) -> Vec<Vec<usize>> {
        let mut visited = vec![false; self.n];
        let mut components = Vec::new();

        for start in 0..self.n {
            if !visited[start] {
                let mut component = Vec::new();
                let mut stack = vec![start];

                while let Some(node) = stack.pop() {
                    if visited[node] {
                        continue;
                    }
                    visited[node] = true;
                    component.push(node);

                    for &neighbor in &self.adj[node] {
                        if !visited[neighbor] {
                            stack.push(neighbor);
                        }
                    }
                }

                components.push(component);
            }
        }

        components
    }
}

fn initialize_positions(n: usize, components: &[Vec<usize>]) -> Vec<Vec2> {
    let mut positions = vec![Vec2::zero(); n];
    let area = (n as f64) * AREA_MULTIPLIER;
    let side = area.sqrt();

    let mut offset_x = 0.0;
    let comp_spacing = side / (components.len() as f64).sqrt();

    for component in components {
        let comp_size = component.len();
        let radius = (comp_size as f64 * 50.0).min(side / 4.0);

        for (i, &node) in component.iter().enumerate() {
            let angle = 2.0 * PI * (i as f64) / (comp_size as f64);
            positions[node] = Vec2::new(
                offset_x + radius * angle.cos(),
                radius * angle.sin(),
            );
        }

        offset_x += comp_spacing;
    }

    positions
}

fn layout_small(graph: &Graph, iterations: usize) -> Vec<Vec2> {
    let components = graph.get_components();
    let mut positions = initialize_positions(graph.n, &components);

    let area = (graph.n as f64) * AREA_MULTIPLIER;
    let k = (area / (graph.n as f64)).sqrt();

    for iter in 0..iterations {
        let temp = 100.0 * (1.0 - (iter as f64) / (iterations as f64));
        let mut forces = vec![Vec2::zero(); graph.n];

        // Repulsive forces (all pairs)
        for i in 0..graph.n {
            for j in (i + 1)..graph.n {
                let delta = positions[i].sub(&positions[j]);
                let dist = delta.length().max(0.01);
                let repulsion = (k * k) / dist;
                let force = delta.mul(repulsion / dist);
                forces[i] = forces[i].add(&force);
                forces[j] = forces[j].sub(&force);
            }
        }

        // Attractive forces (edges)
        for &(u, v) in &graph.edges {
            if u < graph.n && v < graph.n && u != v {
                let delta = positions[v].sub(&positions[u]);
                let dist = delta.length().max(0.01);
                let attraction = (dist * dist) / k;
                let force = delta.mul(attraction / dist);
                forces[u] = forces[u].add(&force);
                forces[v] = forces[v].sub(&force);
            }
        }

        // Apply forces with temperature cooling
        for i in 0..graph.n {
            let force_len = forces[i].length();
            if force_len > 0.0 {
                let displacement = forces[i].mul(temp.min(force_len) / force_len);
                positions[i] = positions[i].add(&displacement);
            }
        }
    }

    positions
}

fn layout_medium(graph: &Graph) -> Vec<Vec2> {
    layout_small(graph, 100)
}

fn layout_large(graph: &Graph) -> Vec<Vec2> {
    let components = graph.get_components();
    let mut positions = initialize_positions(graph.n, &components);

    let area = (graph.n as f64) * AREA_MULTIPLIER;
    let k = (area / (graph.n as f64)).sqrt() * 0.5;
    let iterations = 50;

    for iter in 0..iterations {
        let temp = 50.0 * (1.0 - (iter as f64) / (iterations as f64));
        let mut forces = vec![Vec2::zero(); graph.n];

        // Attractive forces only (too expensive to compute all repulsive)
        for &(u, v) in &graph.edges {
            if u < graph.n && v < graph.n && u != v {
                let delta = positions[v].sub(&positions[u]);
                let dist = delta.length().max(0.1);
                let attraction = (dist * dist) / (k * 10.0);
                let force = delta.mul(attraction / dist);
                forces[u] = forces[u].add(&force);
                forces[v] = forces[v].sub(&force);
            }
        }

        // Sparse repulsive sampling
        let sample_size = (graph.n / 10).max(100).min(500);
        for i in 0..graph.n {
            for _ in 0..sample_size.min(graph.n) {
                let j = (i * 7919 + iter * 1009) % graph.n;
                if i != j {
                    let delta = positions[i].sub(&positions[j]);
                    let dist = delta.length().max(0.1);
                    let repulsion = (k * k * 0.1) / dist;
                    let force = delta.mul(repulsion / dist);
                    forces[i] = forces[i].add(&force);
                }
            }
        }

        // Apply forces
        for i in 0..graph.n {
            let force_len = forces[i].length();
            if force_len > 0.0 {
                let displacement = forces[i].mul(temp.min(force_len) / force_len);
                positions[i] = positions[i].add(&displacement);
            }
        }
    }

    positions
}

fn layout_extreme(graph: &Graph) -> Vec<Vec2> {
    // For very large graphs, use a simple grid-based layout with local adjustments
    let components = graph.get_components();
    let mut positions = vec![Vec2::zero(); graph.n];

    let grid_size = (graph.n as f64).sqrt().ceil() as usize;
    let spacing = 100.0;

    for (idx, &node) in components.iter().flat_map(|c| c.iter()).enumerate() {
        let row = idx / grid_size;
        let col = idx % grid_size;
        positions[node] = Vec2::new(col as f64 * spacing, row as f64 * spacing);
    }

    // Quick refinement: only move connected nodes closer
    for _ in 0..10 {
        let mut new_positions = positions.clone();

        for &(u, v) in &graph.edges {
            if u < graph.n && v < graph.n && u != v {
                let delta = positions[v].sub(&positions[u]);
                let adjustment = delta.mul(0.1);
                new_positions[u] = new_positions[u].add(&adjustment);
                new_positions[v] = new_positions[v].sub(&adjustment);
            }
        }

        positions = new_positions;
    }

    positions
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let n = parts[0];
    let m = parts[1];

    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let parts: Vec<usize> = line.split_whitespace()
                .map(|s| s.parse().unwrap())
                .collect();
            if parts.len() >= 2 {
                edges.push((parts[0], parts[1]));
            }
        }
    }

    let graph = Graph::new(n, edges);

    // Choose algorithm based on graph size
    let positions = if n <= 100 {
        layout_small(&graph, 200)
    } else if n <= 1000 {
        layout_medium(&graph)
    } else if n <= 10000 {
        layout_large(&graph)
    } else {
        layout_extreme(&graph)
    };

    // Output
    println!("{}", n);
    for pos in positions {
        println!("{} {}", pos.x, pos.y);
    }
}
