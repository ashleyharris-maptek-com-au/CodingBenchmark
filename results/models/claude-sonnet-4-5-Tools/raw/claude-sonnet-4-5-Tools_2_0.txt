using System;
using System.Collections.Generic;
using System.Linq;

class ChinesePostman
{
    static void Main()
    {
        var line = Console.ReadLine().Split();
        int n = int.Parse(line[0]);
        int m = int.Parse(line[1]);

        var graph = new List<(int, int)>[n];
        for (int i = 0; i < n; i++)
            graph[i] = new List<(int, int)>();

        var edges = new List<(int, int, int)>();
        for (int i = 0; i < m; i++)
        {
            var parts = Console.ReadLine().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            int w = int.Parse(parts[2]);
            graph[u].Add((v, i));
            graph[v].Add((u, i));
            edges.Add((u, v, w));
        }

        // Find odd degree vertices
        var oddVertices = new List<int>();
        for (int i = 0; i < n; i++)
        {
            if (graph[i].Count % 2 == 1)
                oddVertices.Add(i);
        }

        // If there are odd vertices, we need to add edges
        if (oddVertices.Count > 0)
        {
            // Compute shortest paths between all odd vertices
            var dist = new int[oddVertices.Count, oddVertices.Count];
            var parent = new int[oddVertices.Count, oddVertices.Count];

            for (int i = 0; i < oddVertices.Count; i++)
            {
                var distances = Dijkstra(graph, edges, oddVertices[i], n);
                var parents = new int[n];
                Array.Fill(parents, -1);

                // Reconstruct parent info for shortest paths
                var d2 = new int[n];
                var p2 = new int[n];
                Array.Fill(d2, int.MaxValue);
                Array.Fill(p2, -1);
                d2[oddVertices[i]] = 0;

                var pq = new PriorityQueue<(int node, int dist), int>();
                pq.Enqueue((oddVertices[i], 0), 0);

                while (pq.Count > 0)
                {
                    var (node, nodeDist) = pq.Dequeue();
                    if (nodeDist > d2[node]) continue;

                    foreach (var (neighbor, edgeIdx) in graph[node])
                    {
                        int newDist = nodeDist + edges[edgeIdx].Item3;
                        if (newDist < d2[neighbor])
                        {
                            d2[neighbor] = newDist;
                            p2[neighbor] = node;
                            pq.Enqueue((neighbor, newDist), newDist);
                        }
                    }
                }

                for (int j = 0; j < oddVertices.Count; j++)
                {
                    dist[i, j] = d2[oddVertices[j]];
                    parent[i, j] = p2[oddVertices[j]];
                }
            }

            // Find minimum weight perfect matching on odd vertices
            int k = oddVertices.Count;
            var matching = MinWeightMatching(dist, k);

            // Add matched edges to graph
            var edgeCount = new Dictionary<(int, int), int>();
            for (int i = 0; i < edges.Count; i++)
            {
                var key = edges[i].Item1 < edges[i].Item2
                    ? (edges[i].Item1, edges[i].Item2)
                    : (edges[i].Item2, edges[i].Item1);
                edgeCount[key] = edgeCount.GetValueOrDefault(key, 0) + 1;
            }

            for (int i = 0; i < matching.Length; i += 2)
            {
                int u = oddVertices[matching[i]];
                int v = oddVertices[matching[i + 1]];

                // Add edges along shortest path from u to v
                var path = ReconstructPath(graph, edges, u, v, n);
                for (int j = 0; j < path.Count - 1; j++)
                {
                    var key = path[j] < path[j + 1]
                        ? (path[j], path[j + 1])
                        : (path[j + 1], path[j]);
                    edgeCount[key] = edgeCount.GetValueOrDefault(key, 0) + 1;
                }
            }

            // Rebuild graph with duplicated edges
            var newGraph = new List<(int, int)>[n];
            for (int i = 0; i < n; i++)
                newGraph[i] = new List<(int, int)>();

            int edgeId = 0;
            var newEdges = new List<(int, int, int)>();
            foreach (var kvp in edgeCount)
            {
                for (int i = 0; i < kvp.Value; i++)
                {
                    newGraph[kvp.Key.Item1].Add((kvp.Key.Item2, edgeId));
                    newGraph[kvp.Key.Item2].Add((kvp.Key.Item1, edgeId));
                    newEdges.Add((kvp.Key.Item1, kvp.Key.Item2, 1));
                    edgeId++;
                }
            }

            graph = newGraph;
            edges = newEdges;
        }

        // Find Eulerian circuit
        var circuit = FindEulerianCircuit(graph, edges, 0);
        Console.WriteLine(string.Join(" ", circuit));
    }

    static int[] Dijkstra(List<(int, int)>[] graph, List<(int, int, int)> edges, int start, int n)
    {
        var dist = new int[n];
        Array.Fill(dist, int.MaxValue);
        dist[start] = 0;

        var pq = new PriorityQueue<(int node, int dist), int>();
        pq.Enqueue((start, 0), 0);

        while (pq.Count > 0)
        {
            var (node, nodeDist) = pq.Dequeue();
            if (nodeDist > dist[node]) continue;

            foreach (var (neighbor, edgeIdx) in graph[node])
            {
                int newDist = nodeDist + edges[edgeIdx].Item3;
                if (newDist < dist[neighbor])
                {
                    dist[neighbor] = newDist;
                    pq.Enqueue((neighbor, newDist), newDist);
                }
            }
        }

        return dist;
    }

    static List<int> ReconstructPath(List<(int, int)>[] graph, List<(int, int, int)> edges, int start, int end, int n)
    {
        var parent = new int[n];
        Array.Fill(parent, -1);
        var dist = new int[n];
        Array.Fill(dist, int.MaxValue);
        dist[start] = 0;

        var pq = new PriorityQueue<(int node, int dist), int>();
        pq.Enqueue((start, 0), 0);

        while (pq.Count > 0)
        {
            var (node, nodeDist) = pq.Dequeue();
            if (nodeDist > dist[node]) continue;
            if (node == end) break;

            foreach (var (neighbor, edgeIdx) in graph[node])
            {
                int newDist = nodeDist + edges[edgeIdx].Item3;
                if (newDist < dist[neighbor])
                {
                    dist[neighbor] = newDist;
                    parent[neighbor] = node;
                    pq.Enqueue((neighbor, newDist), newDist);
                }
            }
        }

        var path = new List<int>();
        int curr = end;
        while (curr != -1)
        {
            path.Add(curr);
            curr = parent[curr];
        }
        path.Reverse();
        return path;
    }

    static int[] MinWeightMatching(int[,] dist, int k)
    {
        // For small k, use exact DP; for large k, use greedy
        if (k <= 16)
        {
            return MinWeightMatchingDP(dist, k);
        }
        else
        {
            return MinWeightMatchingGreedy(dist, k);
        }
    }

    static int[] MinWeightMatchingDP(int[,] dist, int k)
    {
        int n = 1 << k;
        var dp = new int[n];
        var parent = new int[n];
        Array.Fill(dp, int.MaxValue);
        dp[0] = 0;

        for (int mask = 0; mask < n; mask++)
        {
            if (dp[mask] == int.MaxValue) continue;

            int first = -1;
            for (int i = 0; i < k; i++)
            {
                if ((mask & (1 << i)) == 0)
                {
                    first = i;
                    break;
                }
            }

            if (first == -1) continue;

            for (int second = first + 1; second < k; second++)
            {
                if ((mask & (1 << second)) == 0)
                {
                    int newMask = mask | (1 << first) | (1 << second);
                    int newCost = dp[mask] + dist[first, second];
                    if (newCost < dp[newMask])
                    {
                        dp[newMask] = newCost;
                        parent[newMask] = mask;
                    }
                }
            }
        }

        // Reconstruct matching
        var matching = new List<int>();
        int currentMask = n - 1;
        while (currentMask != 0)
        {
            int prevMask = parent[currentMask];
            int diff = currentMask ^ prevMask;

            for (int i = 0; i < k; i++)
            {
                if ((diff & (1 << i)) != 0)
                {
                    matching.Add(i);
                }
            }

            currentMask = prevMask;
        }

        return matching.ToArray();
    }

    static int[] MinWeightMatchingGreedy(int[,] dist, int k)
    {
        var matching = new List<int>();
        var used = new bool[k];

        while (matching.Count < k)
        {
            int bestI = -1, bestJ = -1, bestDist = int.MaxValue;

            for (int i = 0; i < k; i++)
            {
                if (used[i]) continue;
                for (int j = i + 1; j < k; j++)
                {
                    if (used[j]) continue;
                    if (dist[i, j] < bestDist)
                    {
                        bestDist = dist[i, j];
                        bestI = i;
                        bestJ = j;
                    }
                }
            }

            matching.Add(bestI);
            matching.Add(bestJ);
            used[bestI] = true;
            used[bestJ] = true;
        }

        return matching.ToArray();
    }

    static List<int> FindEulerianCircuit(List<(int, int)>[] graph, List<(int, int, int)> edges, int start)
    {
        var edgeUsed = new bool[edges.Count];
        var stack = new Stack<int>();
        var circuit = new List<int>();

        stack.Push(start);

        while (stack.Count > 0)
        {
            int v = stack.Peek();
            bool foundEdge = false;

            foreach (var (neighbor, edgeIdx) in graph[v])
            {
                if (!edgeUsed[edgeIdx])
                {
                    edgeUsed[edgeIdx] = true;
                    stack.Push(neighbor);
                    foundEdge = true;
                    break;
                }
            }

            if (!foundEdge)
            {
                circuit.Add(stack.Pop());
            }
        }

        circuit.Reverse();
        return circuit;
    }
}
