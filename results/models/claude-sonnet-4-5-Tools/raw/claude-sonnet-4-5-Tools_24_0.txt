#include <iostream>
#include <fstream>
#include <sstream>
#include <string>
#include <vector>
#include <cmath>
#include <thread>
#include <atomic>
#include <mutex>
#include <chrono>
#include <algorithm>
#include <iomanip>

using namespace std;

// State structure
struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
    bool valid = false;
};

// Global shared state
atomic<bool> running(true);
mutex state_mutex;
State current_state;

// Configuration
struct Config {
    int width, depth, height;
    double gravity, max_thrust, fuel;
    double start_x, start_y, start_z;
    double target_x, target_y, target_z;
    string terrain_path;
};

Config config;

// Read state updates from stdin in a separate thread
void read_state_thread() {
    string line;
    while (running && getline(cin, line)) {
        if (line.empty()) continue;

        istringstream iss(line);
        State new_state;

        if (iss >> new_state.x >> new_state.y >> new_state.z
               >> new_state.vx >> new_state.vy >> new_state.vz
               >> new_state.pitch >> new_state.yaw >> new_state.roll
               >> new_state.pitch_rate >> new_state.yaw_rate >> new_state.roll_rate
               >> new_state.fuel) {
            new_state.valid = true;

            lock_guard<mutex> lock(state_mutex);
            current_state = new_state;
        }
    }
    running = false;
}

// Clamp function
double clamp(double val, double min_val, double max_val) {
    return max(min_val, min(max_val, val));
}

// PID controller
struct PIDController {
    double kp, ki, kd;
    double integral = 0;
    double prev_error = 0;
    double integral_limit = 100.0;

    PIDController(double p, double i, double d) : kp(p), ki(i), kd(d) {}

    double update(double error, double dt) {
        integral += error * dt;
        integral = clamp(integral, -integral_limit, integral_limit);

        double derivative = (error - prev_error) / dt;
        prev_error = error;

        return kp * error + ki * integral + kd * derivative;
    }

    void reset() {
        integral = 0;
        prev_error = 0;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    // Read initial configuration
    cin >> config.width >> config.depth >> config.height
        >> config.gravity >> config.max_thrust >> config.fuel;
    cin >> config.start_x >> config.start_y >> config.start_z
        >> config.target_x >> config.target_y >> config.target_z;
    cin.ignore(); // consume newline

    getline(cin, config.terrain_path);

    string state_marker;
    getline(cin, state_marker); // "STATE"

    // Start input thread
    thread input_thread(read_state_thread);

    // Wait for first state
    while (running) {
        {
            lock_guard<mutex> lock(state_mutex);
            if (current_state.valid) break;
        }
        this_thread::sleep_for(chrono::milliseconds(1));
    }

    // PID controllers for position
    PIDController pid_x(0.15, 0.001, 0.8);
    PIDController pid_y(0.15, 0.001, 0.8);
    PIDController pid_z(0.25, 0.005, 1.2);

    // PID controllers for attitude
    PIDController pid_pitch(2.0, 0.01, 0.5);
    PIDController pid_yaw(2.0, 0.01, 0.5);

    auto last_time = chrono::steady_clock::now();
    const double dt = 0.1;

    while (running) {
        auto now = chrono::steady_clock::now();

        State state;
        {
            lock_guard<mutex> lock(state_mutex);
            state = current_state;
        }

        if (!state.valid) {
            this_thread::sleep_for(chrono::milliseconds(10));
            continue;
        }

        // Calculate errors
        double dx = config.target_x - state.x;
        double dy = config.target_y - state.y;
        double dz = config.target_z - state.z;

        double horizontal_dist = sqrt(dx*dx + dy*dy);
        double total_dist = sqrt(dx*dx + dy*dy + dz*dz);

        // Velocity magnitude
        double speed = sqrt(state.vx*state.vx + state.vy*state.vy + state.vz*state.vz);

        // Determine phase: approach or landing
        bool landing_phase = (horizontal_dist < 20.0 && dz < 50.0) || total_dist < 30.0;

        double thrust = 0.0;
        double target_pitch = 0.0;
        double target_yaw = 0.0;

        if (landing_phase) {
            // Landing phase: prioritize gentle vertical descent
            double target_vz = clamp(-0.5 * dz, -2.0, 0.5);
            double vz_error = target_vz - state.vz;

            // Vertical thrust to counteract gravity and reach target velocity
            double base_thrust = config.gravity / config.max_thrust;
            double z_correction = pid_z.update(vz_error, dt) / config.max_thrust;
            thrust = base_thrust + z_correction;

            // Small horizontal corrections if close to target
            if (horizontal_dist > 0.5 && dz > 2.0) {
                double angle_to_target = atan2(dy, dx);
                target_yaw = angle_to_target;

                // Gentle tilt toward target
                double tilt_amount = clamp(horizontal_dist * 0.05, 0.0, 0.15);
                target_pitch = tilt_amount;
            } else {
                target_pitch = 0.0;
                target_yaw = state.yaw; // hold current yaw
            }
        } else {
            // Approach phase: move toward target
            double angle_to_target = atan2(dy, dx);
            target_yaw = angle_to_target;

            // Calculate desired horizontal velocity
            double target_vh = clamp(horizontal_dist * 0.3, 0.0, 20.0);
            double current_vh = sqrt(state.vx*state.vx + state.vy*state.vy);

            // Calculate desired vertical velocity
            double target_vz;
            if (dz > 100.0) {
                target_vz = clamp(dz * 0.1, -5.0, 2.0);
            } else if (dz > 20.0) {
                target_vz = clamp(dz * 0.05, -3.0, 0.0);
            } else {
                target_vz = clamp(dz * 0.1, -2.0, 0.0);
            }

            double vz_error = target_vz - state.vz;

            // Thrust to maintain altitude while approaching
            double base_thrust = config.gravity / config.max_thrust;
            double z_correction = pid_z.update(vz_error, dt) / config.max_thrust;
            thrust = base_thrust + z_correction;

            // Tilt to accelerate horizontally
            if (horizontal_dist > 10.0) {
                double tilt_amount = clamp(horizontal_dist * 0.01, 0.0, 0.5);
                target_pitch = tilt_amount;

                // Additional thrust when tilted
                thrust += tilt_amount * 0.3;
            } else {
                target_pitch = clamp(horizontal_dist * 0.05, 0.0, 0.2);
            }
        }

        // Clamp thrust
        thrust = clamp(thrust, 0.0, 1.0);

        // If low on fuel, reduce thrust but stay above hover
        if (state.fuel < 50.0) {
            thrust = clamp(thrust, 0.0, 0.8);
        }

        // Attitude control
        double pitch_error = target_pitch - state.pitch;
        double yaw_error = target_yaw - state.yaw;

        // Normalize yaw error to [-pi, pi]
        while (yaw_error > M_PI) yaw_error -= 2*M_PI;
        while (yaw_error < -M_PI) yaw_error += 2*M_PI;

        double pitch_cmd = pid_pitch.update(pitch_error, dt) - state.pitch_rate * 0.3;
        double yaw_cmd = pid_yaw.update(yaw_error, dt) - state.yaw_rate * 0.3;
        double roll_cmd = -state.roll * 0.5 - state.roll_rate * 0.3;

        // Clamp commands
        pitch_cmd = clamp(pitch_cmd, -1.0, 1.0);
        yaw_cmd = clamp(yaw_cmd, -1.0, 1.0);
        roll_cmd = clamp(roll_cmd, -1.0, 1.0);

        // Output commands
        cout << fixed << setprecision(6);
        cout << thrust << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n";
        cout.flush();

        // Sleep to maintain ~10Hz output
        this_thread::sleep_for(chrono::milliseconds(95));
    }

    // Wait for input thread
    if (input_thread.joinable()) {
        input_thread.join();
    }

    return 0;
}
