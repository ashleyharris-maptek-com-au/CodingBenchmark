using namespace std;

struct Vec3 {
    double x, y, z;
    Vec3(double x = 0, double y = 0, double z = 0) : x(x), y(y), z(z) {}
    Vec3 operator-(const Vec3& o) const { return Vec3(x - o.x, y - o.y, z - o.z); }
    Vec3 operator+(const Vec3& o) const { return Vec3(x + o.x, y + o.y, z + o.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
    double dot(const Vec3& o) const { return x * o.x + y * o.y + z * o.z; }
    Vec3 cross(const Vec3& o) const {
        return Vec3(y * o.z - z * o.y, z * o.x - x * o.z, x * o.y - y * o.x);
    }
    double length() const { return sqrt(x * x + y * y + z * z); }
    Vec3 normalize() const { double l = length(); return l > 0 ? Vec3(x / l, y / l, z / l) : *this; }
};

struct Triangle {
    int a, b, c;
    Triangle(int a = 0, int b = 0, int c = 0) : a(a), b(b), c(c) {}
};

struct Mesh {
    vector<Vec3> vertices;
    vector<Triangle> faces;
};

// Compute triangle normal
Vec3 computeNormal(const Vec3& a, const Vec3& b, const Vec3& c) {
    return (b - a).cross(c - a).normalize();
}

// Check if point is inside mesh using ray casting
bool isPointInsideMesh(const Vec3& point, const Mesh& mesh) {
    if (mesh.faces.empty()) return false;

    // Cast ray in positive X direction
    Vec3 rayDir(1, 0, 0);
    int intersections = 0;

    for (const auto& face : mesh.faces) {
        const Vec3& v0 = mesh.vertices[face.a];
        const Vec3& v1 = mesh.vertices[face.b];
        const Vec3& v2 = mesh.vertices[face.c];

        Vec3 e1 = v1 - v0;
        Vec3 e2 = v2 - v0;
        Vec3 h = rayDir.cross(e2);
        double a = e1.dot(h);

        if (abs(a) < 1e-10) continue;

        double f = 1.0 / a;
        Vec3 s = point - v0;
        double u = f * s.dot(h);

        if (u < 0.0 || u > 1.0) continue;

        Vec3 q = s.cross(e1);
        double v = f * rayDir.dot(q);

        if (v < 0.0 || u + v > 1.0) continue;

        double t = f * e2.dot(q);

        if (t > 1e-10) {
            intersections++;
        }
    }

    return (intersections % 2) == 1;
}

// Compute centroid of a triangle
Vec3 triangleCentroid(const Vec3& a, const Vec3& b, const Vec3& c) {
    return Vec3((a.x + b.x + c.x) / 3.0, (a.y + b.y + c.y) / 3.0, (a.z + b.z + c.z) / 3.0);
}

// Simple CSG Union using face filtering
Mesh computeUnion(const Mesh& meshA, const Mesh& meshB) {
    Mesh result;

    // For small meshes, use geometric approach
    // For CSG union: keep faces of A that are outside B, and faces of B that are outside A

    // Add vertices from mesh A
    int offsetA = 0;
    for (const auto& v : meshA.vertices) {
        result.vertices.push_back(v);
    }

    // Add vertices from mesh B
    int offsetB = meshA.vertices.size();
    for (const auto& v : meshB.vertices) {
        result.vertices.push_back(v);
    }

    // Filter faces from mesh A (keep if outside mesh B)
    for (const auto& face : meshA.faces) {
        const Vec3& v0 = meshA.vertices[face.a];
        const Vec3& v1 = meshA.vertices[face.b];
        const Vec3& v2 = meshA.vertices[face.c];
        Vec3 centroid = triangleCentroid(v0, v1, v2);
        Vec3 normal = computeNormal(v0, v1, v2);

        // Sample point slightly outside the triangle
        Vec3 testPoint = centroid + normal * 1e-6;

        if (!isPointInsideMesh(testPoint, meshB)) {
            result.faces.push_back(Triangle(face.a + offsetA, face.b + offsetA, face.c + offsetA));
        }
    }

    // Filter faces from mesh B (keep if outside mesh A)
    for (const auto& face : meshB.faces) {
        const Vec3& v0 = meshB.vertices[face.a];
        const Vec3& v1 = meshB.vertices[face.b];
        const Vec3& v2 = meshB.vertices[face.c];
        Vec3 centroid = triangleCentroid(v0, v1, v2);
        Vec3 normal = computeNormal(v0, v1, v2);

        // Sample point slightly outside the triangle
        Vec3 testPoint = centroid + normal * 1e-6;

        if (!isPointInsideMesh(testPoint, meshA)) {
            result.faces.push_back(Triangle(face.a + offsetB, face.b + offsetB, face.c + offsetB));
        }
    }

    return result;
}

// Remove unused vertices and reindex faces
Mesh cleanupMesh(const Mesh& mesh) {
    Mesh result;
    map<int, int> oldToNew;

    // Find used vertices
    set<int> usedVertices;
    for (const auto& face : mesh.faces) {
        usedVertices.insert(face.a);
        usedVertices.insert(face.b);
        usedVertices.insert(face.c);
    }

    // Create mapping and add used vertices
    int newIndex = 0;
    for (int oldIdx : usedVertices) {
        oldToNew[oldIdx] = newIndex++;
        result.vertices.push_back(mesh.vertices[oldIdx]);
    }

    // Reindex faces
    for (const auto& face : mesh.faces) {
        result.faces.push_back(Triangle(
            oldToNew[face.a],
            oldToNew[face.b],
            oldToNew[face.c]
        ));
    }

    return result;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA, FA, VB, FB;
    cin >> VA >> FA >> VB >> FB;

    Mesh meshA, meshB;

    // Read mesh A vertices
    meshA.vertices.resize(VA);
    for (int i = 0; i < VA; i++) {
        cin >> meshA.vertices[i].x >> meshA.vertices[i].y >> meshA.vertices[i].z;
    }

    // Read mesh A faces
    meshA.faces.resize(FA);
    for (int i = 0; i < FA; i++) {
        cin >> meshA.faces[i].a >> meshA.faces[i].b >> meshA.faces[i].c;
    }

    // Read mesh B vertices
    meshB.vertices.resize(VB);
    for (int i = 0; i < VB; i++) {
        cin >> meshB.vertices[i].x >> meshB.vertices[i].y >> meshB.vertices[i].z;
    }

    // Read mesh B faces
    meshB.faces.resize(FB);
    for (int i = 0; i < FB; i++) {
        cin >> meshB.faces[i].a >> meshB.faces[i].b >> meshB.faces[i].c;
    }

    // Compute union
    Mesh result = computeUnion(meshA, meshB);

    // Cleanup unused vertices
    result = cleanupMesh(result);

    // Output result
    cout << result.vertices.size() << " " << result.faces.size() << "\n";

    for (const auto& v : result.vertices) {
        cout << v.x << " " << v.y << " " << v.z << "\n";
    }

    for (const auto& f : result.faces) {
        cout << f.a << " " << f.b << " " << f.c << "\n";
    }

    return 0;
}
