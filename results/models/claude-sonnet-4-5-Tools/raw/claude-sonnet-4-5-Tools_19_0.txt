using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class Program
{
    static void Main()
    {
        // Read header
        string header = Console.ReadLine();
        string[] parts = header.Split(' ');
        int xCount = int.Parse(parts[0]);
        int yCount = int.Parse(parts[1]);
        int zCount = int.Parse(parts[2]);

        // Read binary voxel data
        byte[,,] grid = new byte[xCount, yCount, zCount];
        using (var stdin = Console.OpenStandardInput())
        {
            for (int x = 0; x < xCount; x++)
            {
                for (int y = 0; y < yCount; y++)
                {
                    for (int z = 0; z < zCount; z++)
                    {
                        int b = stdin.ReadByte();
                        if (b == -1) b = 0;
                        grid[x, y, z] = (byte)b;
                    }
                }
            }
        }

        // Build height map and find valuable ore
        int[,] heightMap = new int[xCount, yCount];
        List<Voxel> valuableOre = new List<Voxel>();

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                heightMap[x, y] = 0;
                for (int z = zCount - 1; z >= 0; z--)
                {
                    if (grid[x, y, z] != 0)
                    {
                        heightMap[x, y] = z + 1;
                        break;
                    }
                }

                // Find valuable ore in this column
                for (int z = 0; z < heightMap[x, y]; z++)
                {
                    byte voxel = grid[x, y, z];
                    if (voxel != 0)
                    {
                        int mass = voxel % 16;
                        int value = voxel / 16;
                        if (value > 0 && mass > 0)
                        {
                            valuableOre.Add(new Voxel
                            {
                                X = x,
                                Y = y,
                                Z = z,
                                Mass = mass,
                                Value = value
                            });
                        }
                    }
                }
            }
        }

        // Sort by profit potential (value * mass) descending
        valuableOre.Sort((a, b) => (b.Value * b.Mass).CompareTo(a.Value * a.Mass));

        List<Operation> operations = new List<Operation>();

        // Strategy: prioritize high-value ore close to the plant
        foreach (var ore in valuableOre)
        {
            // Check if this voxel is still mineable (exposed from top)
            if (heightMap[ore.X, ore.Y] != ore.Z + 1)
                continue;

            // Check if digging above this voxel is necessary
            bool needsToDig = false;
            for (int z = ore.Z + 1; z < heightMap[ore.X, ore.Y]; z++)
            {
                if (grid[ore.X, ore.Y, z] != 0)
                {
                    needsToDig = true;
                    break;
                }
            }

            // Dig overburden first
            if (needsToDig)
            {
                for (int z = heightMap[ore.X, ore.Y] - 1; z > ore.Z; z--)
                {
                    byte voxel = grid[ore.X, ore.Y, z];
                    if (voxel != 0)
                    {
                        int mass = voxel % 16;
                        int value = voxel / 16;

                        // Find dump location for overburden
                        var dumpLoc = FindDumpLocation(ore.X, ore.Y, xCount, yCount, heightMap);
                        var path = FindPath(ore.X, ore.Y, z, dumpLoc.X, dumpLoc.Y, heightMap);

                        if (path != null)
                        {
                            operations.Add(new Operation
                            {
                                DigX = ore.X,
                                DigY = ore.Y,
                                DigZ = z,
                                Path = path,
                                DestX = dumpLoc.X,
                                DestY = dumpLoc.Y
                            });

                            grid[ore.X, ore.Y, z] = 0;
                            heightMap[dumpLoc.X, dumpLoc.Y]++;
                        }
                    }
                }
                heightMap[ore.X, ore.Y] = ore.Z + 1;
            }

            // Now dig the valuable ore and take it to plant
            var orePath = FindPath(ore.X, ore.Y, ore.Z, 0, 0, heightMap);
            if (orePath != null)
            {
                operations.Add(new Operation
                {
                    DigX = ore.X,
                    DigY = ore.Y,
                    DigZ = ore.Z,
                    Path = orePath,
                    DestX = 0,
                    DestY = 0
                });

                grid[ore.X, ore.Y, ore.Z] = 0;
                heightMap[ore.X, ore.Y]--;
            }
        }

        // Output
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.Write(op.Path.Count);
            Console.Write($" {op.DigX} {op.DigY} {op.DigZ}");
            foreach (var step in op.Path)
            {
                Console.Write($" {step.X} {step.Y}");
            }
            Console.WriteLine($" {op.DestX} {op.DestY}");
        }
    }

    static (int X, int Y) FindDumpLocation(int fromX, int fromY, int xCount, int yCount, int[,] heightMap)
    {
        // Find nearest location with low height
        int bestX = fromX, bestY = fromY;
        int minHeight = int.MaxValue;

        for (int radius = 1; radius < Math.Max(xCount, yCount); radius++)
        {
            for (int dx = -radius; dx <= radius; dx++)
            {
                for (int dy = -radius; dy <= radius; dy++)
                {
                    if (Math.Abs(dx) != radius && Math.Abs(dy) != radius)
                        continue;

                    int nx = fromX + dx;
                    int ny = fromY + dy;

                    if (nx >= 0 && nx < xCount && ny >= 0 && ny < yCount)
                    {
                        if (heightMap[nx, ny] < minHeight)
                        {
                            minHeight = heightMap[nx, ny];
                            bestX = nx;
                            bestY = ny;
                        }
                    }
                }
            }
            if (minHeight < int.MaxValue)
                break;
        }

        return (bestX, bestY);
    }

    static List<(int X, int Y)> FindPath(int fromX, int fromY, int fromZ, int toX, int toY, int[,] heightMap)
    {
        List<(int X, int Y)> path = new List<(int, int)>();

        int curX = fromX, curY = fromY;
        int curZ = fromZ;

        while (curX != toX || curY != toY)
        {
            int dx = 0, dy = 0;

            if (curX < toX) dx = 1;
            else if (curX > toX) dx = -1;
            else if (curY < toY) dy = 1;
            else if (curY > toY) dy = -1;

            int nextX = curX + dx;
            int nextY = curY + dy;

            // Check slope constraint
            int groundHeight = heightMap[nextX, nextY];
            if (Math.Abs(groundHeight - curZ) > 1)
            {
                // Try alternative direction
                if (dx != 0)
                {
                    dx = 0;
                    dy = (curY < toY) ? 1 : -1;
                }
                else
                {
                    dy = 0;
                    dx = (curX < toX) ? 1 : -1;
                }
                nextX = curX + dx;
                nextY = curY + dy;
                groundHeight = heightMap[nextX, nextY];
            }

            path.Add((nextX, nextY));
            curX = nextX;
            curY = nextY;
            curZ = groundHeight;
        }

        return path;
    }
}

class Voxel
{
    public int X, Y, Z;
    public int Mass, Value;
}

class Operation
{
    public int DigX, DigY, DigZ;
    public List<(int X, int Y)> Path;
    public int DestX, DestY;
}
