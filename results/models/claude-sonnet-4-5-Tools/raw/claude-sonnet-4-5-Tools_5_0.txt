use std::collections::HashSet;
use std::io::{self, BufRead};

#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
struct Cell {
    x: i32,
    y: i32,
}

impl Cell {
    fn new(x: i32, y: i32) -> Self {
        Cell { x, y }
    }

    fn neighbors(&self) -> [(i32, i32); 4] {
        [
            (self.x, self.y + 1),
            (self.x + 1, self.y),
            (self.x, self.y - 1),
            (self.x - 1, self.y),
        ]
    }

    fn is_adjacent(&self, other: &Cell) -> bool {
        (self.x - other.x).abs() + (self.y - other.y).abs() == 1
    }
}

struct Grid {
    width: i32,
    height: i32,
    obstacles: HashSet<Cell>,
    require_cycle: bool,
    total_cells: usize,
}

impl Grid {
    fn new(width: i32, height: i32, obstacles: HashSet<Cell>, require_cycle: bool) -> Self {
        let total_cells = (width * height) as usize - obstacles.len();
        Grid {
            width,
            height,
            obstacles,
            require_cycle,
            total_cells,
        }
    }

    fn is_valid(&self, x: i32, y: i32) -> bool {
        x >= 0 && x < self.width
            && y >= 0 && y < self.height
            && !self.obstacles.contains(&Cell::new(x, y))
    }

    fn solve(&self) -> Option<Vec<Cell>> {
        let start = Cell::new(0, 0);

        if !self.is_valid(start.x, start.y) {
            return None;
        }

        let mut path = vec![start];
        let mut visited = vec![vec![false; self.height as usize]; self.width as usize];
        visited[0][0] = true;

        // Initialize degree count for each cell
        let mut degree = vec![vec![0u8; self.height as usize]; self.width as usize];
        for x in 0..self.width {
            for y in 0..self.height {
                if self.is_valid(x, y) {
                    let cell = Cell::new(x, y);
                    let mut count = 0;
                    for (nx, ny) in cell.neighbors().iter() {
                        if self.is_valid(*nx, *ny) {
                            count += 1;
                        }
                    }
                    degree[x as usize][y as usize] = count;
                }
            }
        }

        if self.backtrack(&mut path, &mut visited, &mut degree) {
            Some(path)
        } else {
            None
        }
    }

    fn backtrack(&self, path: &mut Vec<Cell>, visited: &mut Vec<Vec<bool>>, degree: &mut Vec<Vec<u8>>) -> bool {
        // Check if path is complete
        if path.len() == self.total_cells {
            if self.require_cycle {
                let start = Cell::new(0, 0);
                return path.last().unwrap().is_adjacent(&start);
            }
            return true;
        }

        let current = *path.last().unwrap();

        // Collect unvisited neighbors
        let mut neighbors = Vec::new();
        for (nx, ny) in current.neighbors().iter() {
            if self.is_valid(*nx, *ny) && !visited[*nx as usize][*ny as usize] {
                neighbors.push((*nx, *ny));
            }
        }

        // Sort by degree (Warnsdorff's heuristic - visit cells with fewer options first)
        neighbors.sort_by_key(|(x, y)| degree[*x as usize][*y as usize]);

        for (nx, ny) in neighbors {
            // Try this neighbor
            visited[nx as usize][ny as usize] = true;
            path.push(Cell::new(nx, ny));

            // Update degrees of neighbors of the new cell
            let mut degree_changes = Vec::new();
            for (nnx, nny) in Cell::new(nx, ny).neighbors().iter() {
                if self.is_valid(*nnx, *nny) && !visited[*nnx as usize][*nny as usize] {
                    degree[*nnx as usize][*nny as usize] -= 1;
                    degree_changes.push((*nnx, *nny));
                }
            }

            // Pruning: check if any unvisited cell becomes isolated (degree 0)
            // This means we've created an unreachable cell
            let mut has_isolated = false;
            if path.len() < self.total_cells {
                for (cx, cy) in &degree_changes {
                    if degree[*cx as usize][*cy as usize] == 0 {
                        has_isolated = true;
                        break;
                    }
                }
            }

            // Continue search if no isolated cells
            if !has_isolated && self.backtrack(path, visited, degree) {
                return true;
            }

            // Backtrack: restore state
            path.pop();
            visited[nx as usize][ny as usize] = false;
            for (nnx, nny) in degree_changes {
                degree[nnx as usize][nny as usize] += 1;
            }
        }

        false
    }
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Parse first line
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<i32> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    let width = parts[0];
    let height = parts[1];
    let obstacle_count = parts[2] as usize;
    let require_cycle = parts[3] == 1;

    // Parse obstacles
    let mut obstacles = HashSet::new();
    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<i32> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        obstacles.insert(Cell::new(coords[0], coords[1]));
    }

    // Solve and output
    let grid = Grid::new(width, height, obstacles, require_cycle);

    if let Some(path) = grid.solve() {
        println!("{}", path.len());
        for cell in path {
            println!("{} {}", cell.x, cell.y);
        }
    } else {
        println!("0");
    }
}
