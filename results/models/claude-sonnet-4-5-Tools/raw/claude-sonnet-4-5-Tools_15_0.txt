use std::io::{self, BufRead};

#[derive(Clone, Copy, Debug)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Self { x, y, z }
    }

    fn dot(&self, other: &Vec3) -> f64 {
        self.x * other.x + self.y * other.y + self.z * other.z
    }

    fn cross(&self, other: &Vec3) -> Vec3 {
        Vec3::new(
            self.y * other.z - self.z * other.y,
            self.z * other.x - self.x * other.z,
            self.x * other.y - self.y * other.x,
        )
    }

    fn length(&self) -> f64 {
        (self.x * self.x + self.y * self.y + self.z * self.z).sqrt()
    }

    fn normalize(&self) -> Vec3 {
        let len = self.length();
        Vec3::new(self.x / len, self.y / len, self.z / len)
    }

    fn add(&self, other: &Vec3) -> Vec3 {
        Vec3::new(self.x + other.x, self.y + other.y, self.z + other.z)
    }

    fn sub(&self, other: &Vec3) -> Vec3 {
        Vec3::new(self.x - other.x, self.y - other.y, self.z - other.z)
    }

    fn scale(&self, s: f64) -> Vec3 {
        Vec3::new(self.x * s, self.y * s, self.z * s)
    }
}

#[derive(Clone, Copy, Debug)]
struct Quaternion {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

impl Quaternion {
    fn identity() -> Self {
        Self { w: 1.0, x: 0.0, y: 0.0, z: 0.0 }
    }

    fn from_axis_angle(axis: Vec3, angle: f64) -> Self {
        let half_angle = angle / 2.0;
        let s = half_angle.sin();
        Self {
            w: half_angle.cos(),
            x: axis.x * s,
            y: axis.y * s,
            z: axis.z * s,
        }
    }

    fn from_rotation_between(from: Vec3, to: Vec3) -> Self {
        let from = from.normalize();
        let to = to.normalize();
        let dot = from.dot(&to);

        if dot > 0.9999 {
            return Quaternion::identity();
        }

        if dot < -0.9999 {
            let axis = Vec3::new(1.0, 0.0, 0.0).cross(&from);
            if axis.length() < 0.0001 {
                let axis = Vec3::new(0.0, 1.0, 0.0).cross(&from);
                return Quaternion::from_axis_angle(axis.normalize(), std::f64::consts::PI);
            }
            return Quaternion::from_axis_angle(axis.normalize(), std::f64::consts::PI);
        }

        let axis = from.cross(&to);
        let w = 1.0 + dot;
        let q = Quaternion {
            w,
            x: axis.x,
            y: axis.y,
            z: axis.z,
        };

        let len = (q.w * q.w + q.x * q.x + q.y * q.y + q.z * q.z).sqrt();
        Quaternion {
            w: q.w / len,
            x: q.x / len,
            y: q.y / len,
            z: q.z / len,
        }
    }
}

#[derive(Clone, Copy, Debug)]
struct Tetrahedron {
    position: Vec3,
    rotation: Quaternion,
    scale: f64,
}

fn project_point_to_plane(point: Vec3, sun_dir: Vec3) -> (f64, f64) {
    if sun_dir.z.abs() < 0.0001 {
        return (point.x, point.y);
    }

    let t = -point.z / sun_dir.z;
    let projected_x = point.x + t * sun_dir.x;
    let projected_y = point.y + t * sun_dir.y;
    (projected_x, projected_y)
}

fn compute_tetrahedron_shadow(tet: &Tetrahedron, sun_dir: Vec3) -> Vec<(f64, f64)> {
    let base_vertices = vec![
        Vec3::new(1.0, 0.0, -0.707),
        Vec3::new(-1.0, 0.0, -0.707),
        Vec3::new(0.0, 1.0, 0.707),
        Vec3::new(0.0, -1.0, 0.707),
    ];

    let mut shadow_points = Vec::new();

    for base_vert in base_vertices {
        let scaled = base_vert.scale(tet.scale);

        let qw = tet.rotation.w;
        let qx = tet.rotation.x;
        let qy = tet.rotation.y;
        let qz = tet.rotation.z;

        let rotated = Vec3::new(
            scaled.x * (1.0 - 2.0 * (qy * qy + qz * qz)) +
            scaled.y * (2.0 * (qx * qy - qw * qz)) +
            scaled.z * (2.0 * (qx * qz + qw * qy)),

            scaled.x * (2.0 * (qx * qy + qw * qz)) +
            scaled.y * (1.0 - 2.0 * (qx * qx + qz * qz)) +
            scaled.z * (2.0 * (qy * qz - qw * qx)),

            scaled.x * (2.0 * (qx * qz - qw * qy)) +
            scaled.y * (2.0 * (qy * qz + qw * qx)) +
            scaled.z * (1.0 - 2.0 * (qx * qx + qy * qy)),
        );

        let world_pos = rotated.add(&tet.position);
        shadow_points.push(project_point_to_plane(world_pos, sun_dir));
    }

    shadow_points
}

fn point_in_polygon(point: (f64, f64), polygon: &[(f64, f64)]) -> bool {
    let (x, y) = point;
    let n = polygon.len();
    let mut inside = false;

    let mut j = n - 1;
    for i in 0..n {
        let (xi, yi) = polygon[i];
        let (xj, yj) = polygon[j];

        if ((yi > y) != (yj > y)) && (x < (xj - xi) * (y - yi) / (yj - yi) + xi) {
            inside = !inside;
        }
        j = i;
    }

    inside
}

fn polygon_bounds(polygon: &[(f64, f64)]) -> (f64, f64, f64, f64) {
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;

    for &(x, y) in polygon {
        min_x = min_x.min(x);
        max_x = max_x.max(x);
        min_y = min_y.min(y);
        max_y = max_y.max(y);
    }

    (min_x, max_x, min_y, max_y)
}

fn estimate_coverage(tetrahedra: &[Tetrahedron], polygon: &[(f64, f64)], sun_dir: Vec3) -> f64 {
    let (min_x, max_x, min_y, max_y) = polygon_bounds(polygon);
    let samples = 50;
    let dx = (max_x - min_x) / samples as f64;
    let dy = (max_y - min_y) / samples as f64;

    let mut covered = 0;
    let mut total = 0;

    for i in 0..samples {
        for j in 0..samples {
            let x = min_x + (i as f64 + 0.5) * dx;
            let y = min_y + (j as f64 + 0.5) * dy;

            if point_in_polygon((x, y), polygon) {
                total += 1;

                for tet in tetrahedra {
                    let shadow = compute_tetrahedron_shadow(tet, sun_dir);
                    if point_in_polygon((x, y), &shadow) {
                        covered += 1;
                        break;
                    }
                }
            }
        }
    }

    if total == 0 {
        0.0
    } else {
        covered as f64 / total as f64
    }
}

fn solve_shadow_covering(polygon: Vec<(f64, f64)>, sun_dir: Vec3) -> Vec<Tetrahedron> {
    let (min_x, max_x, min_y, max_y) = polygon_bounds(&polygon);
    let center_x = (min_x + max_x) / 2.0;
    let center_y = (min_y + max_y) / 2.0;
    let width = max_x - min_x;
    let height = max_y - min_y;
    let max_dim = width.max(height);

    let sun_normalized = sun_dir.normalize();
    let default_up = Vec3::new(0.0, 0.0, -1.0);
    let rotation = Quaternion::from_rotation_between(default_up, sun_normalized.scale(-1.0));

    let num_vertices = polygon.len();
    let complexity = num_vertices as f64 * max_dim;

    let mut tetrahedra = Vec::new();

    if complexity < 100.0 {
        let scale = (max_dim / 2.0).min(10.0);
        let height_above = scale * 2.0;

        tetrahedra.push(Tetrahedron {
            position: Vec3::new(center_x, center_y, height_above),
            rotation,
            scale,
        });
    } else if complexity < 1000.0 {
        let grid_size = 2;
        let scale = (max_dim / (grid_size as f64 * 1.5)).min(10.0);
        let height_above = scale * 2.0;

        for i in 0..grid_size {
            for j in 0..grid_size {
                let x = min_x + (i as f64 + 0.5) * width / grid_size as f64;
                let y = min_y + (j as f64 + 0.5) * height / grid_size as f64;

                tetrahedra.push(Tetrahedron {
                    position: Vec3::new(x, y, height_above),
                    rotation,
                    scale,
                });
            }
        }
    } else {
        let grid_size = ((complexity / 500.0).sqrt() as usize).max(3).min(10);
        let scale = (max_dim / (grid_size as f64 * 1.5)).min(10.0);
        let height_above = scale * 2.0;

        for i in 0..grid_size {
            for j in 0..grid_size {
                let x = min_x + (i as f64 + 0.5) * width / grid_size as f64;
                let y = min_y + (j as f64 + 0.5) * height / grid_size as f64;

                tetrahedra.push(Tetrahedron {
                    position: Vec3::new(x, y, height_above),
                    rotation,
                    scale,
                });
            }
        }
    }

    tetrahedra
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let nv: usize = lines.next().unwrap().unwrap().trim().parse().unwrap();

    let mut polygon = Vec::new();
    for _ in 0..nv {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<f64> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        polygon.push((parts[0], parts[1]));
    }

    let sun_line = lines.next().unwrap().unwrap();
    let sun_parts: Vec<f64> = sun_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let sun_dir = Vec3::new(sun_parts[0], sun_parts[1], sun_parts[2]);

    let tetrahedra = solve_shadow_covering(polygon, sun_dir);

    println!("{}", tetrahedra.len());
    for tet in tetrahedra {
        println!("{} {} {} {} {} {} {} {}",
            tet.position.x, tet.position.y, tet.position.z,
            tet.rotation.w, tet.rotation.x, tet.rotation.y, tet.rotation.z,
            tet.scale
        );
    }
}
