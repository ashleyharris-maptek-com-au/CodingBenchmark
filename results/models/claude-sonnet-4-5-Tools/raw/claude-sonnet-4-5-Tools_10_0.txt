#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdint>

using namespace std;

struct Rectangle {
    int w, h;
    int original_index;
};

struct Placement {
    int board_index;
    int x, y;
    bool rotated;
};

struct SkylineSegment {
    int x, y, width;
};

class SkylinePacker {
private:
    int board_width, board_height;
    vector<SkylineSegment> skyline;

public:
    SkylinePacker(int w, int h) : board_width(w), board_height(h) {
        skyline.push_back({0, 0, w});
    }

    bool canPlace(int w, int h, int& out_x, int& out_y) {
        int best_y = board_height + 1;
        int best_idx = -1;
        int best_x = 0;

        for (int i = 0; i < skyline.size(); i++) {
            int x = skyline[i].x;
            int y = skyline[i].y;

            // Check if rectangle fits starting at this segment
            if (x + w > board_width) continue;

            int max_y = y;
            int width_covered = 0;

            for (int j = i; j < skyline.size() && width_covered < w; j++) {
                max_y = max(max_y, skyline[j].y);
                width_covered += skyline[j].width;
            }

            if (width_covered >= w && max_y + h <= board_height) {
                if (max_y < best_y || (max_y == best_y && x < best_x)) {
                    best_y = max_y;
                    best_idx = i;
                    best_x = x;
                }
            }
        }

        if (best_idx == -1) return false;

        out_x = best_x;
        out_y = best_y;
        return true;
    }

    void place(int x, int y, int w, int h) {
        // Add new segment for placed rectangle
        vector<SkylineSegment> new_skyline;
        bool segment_added = false;

        for (int i = 0; i < skyline.size(); i++) {
            SkylineSegment seg = skyline[i];

            // Segment is completely before the rectangle
            if (seg.x + seg.width <= x) {
                new_skyline.push_back(seg);
            }
            // Segment is completely after the rectangle
            else if (seg.x >= x + w) {
                if (!segment_added) {
                    new_skyline.push_back({x, y + h, w});
                    segment_added = true;
                }
                new_skyline.push_back(seg);
            }
            // Segment overlaps with rectangle
            else {
                if (!segment_added) {
                    new_skyline.push_back({x, y + h, w});
                    segment_added = true;
                }

                // Left part before rectangle
                if (seg.x < x) {
                    new_skyline.push_back({seg.x, seg.y, x - seg.x});
                }

                // Right part after rectangle
                if (seg.x + seg.width > x + w) {
                    new_skyline.push_back({x + w, seg.y, seg.x + seg.width - (x + w)});
                }
            }
        }

        if (!segment_added) {
            new_skyline.push_back({x, y + h, w});
        }

        // Merge adjacent segments with same height
        skyline.clear();
        for (int i = 0; i < new_skyline.size(); i++) {
            if (skyline.empty() || skyline.back().y != new_skyline[i].y) {
                skyline.push_back(new_skyline[i]);
            } else {
                skyline.back().width += new_skyline[i].width;
            }
        }
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, board_width, board_height;
    cin >> n >> board_width >> board_height;

    vector<Rectangle> rectangles(n);
    for (int i = 0; i < n; i++) {
        cin >> rectangles[i].w >> rectangles[i].h;
        rectangles[i].original_index = i;
    }

    // Sort by area (descending) for better packing
    vector<int> sorted_indices(n);
    for (int i = 0; i < n; i++) sorted_indices[i] = i;

    sort(sorted_indices.begin(), sorted_indices.end(), [&](int a, int b) {
        long long area_a = (long long)rectangles[a].w * rectangles[a].h;
        long long area_b = (long long)rectangles[b].w * rectangles[b].h;
        if (area_a != area_b) return area_a > area_b;
        return max(rectangles[a].w, rectangles[a].h) > max(rectangles[b].w, rectangles[b].h);
    });

    vector<SkylinePacker> boards;
    vector<Placement> placements(n);

    // Place each rectangle
    for (int idx : sorted_indices) {
        Rectangle& rect = rectangles[idx];
        bool placed = false;

        // Try to place on existing boards
        for (int b = 0; b < boards.size() && !placed; b++) {
            int x, y;

            // Try normal orientation
            if (boards[b].canPlace(rect.w, rect.h, x, y)) {
                boards[b].place(x, y, rect.w, rect.h);
                placements[idx] = {b, x, y, false};
                placed = true;
            }
            // Try rotated orientation
            else if (boards[b].canPlace(rect.h, rect.w, x, y)) {
                boards[b].place(x, y, rect.h, rect.w);
                placements[idx] = {b, x, y, true};
                placed = true;
            }
        }

        // Create new board if needed
        if (!placed) {
            boards.emplace_back(board_width, board_height);
            int b = boards.size() - 1;
            int x, y;

            // Try normal orientation
            if (boards[b].canPlace(rect.w, rect.h, x, y)) {
                boards[b].place(x, y, rect.w, rect.h);
                placements[idx] = {b, x, y, false};
            }
            // Try rotated orientation
            else if (boards[b].canPlace(rect.h, rect.w, x, y)) {
                boards[b].place(x, y, rect.h, rect.w);
                placements[idx] = {b, x, y, true};
            }
        }
    }

    // Output results
    cout << boards.size() << "\n";
    for (int i = 0; i < n; i++) {
        cout << placements[i].board_index << " "
             << placements[i].x << " "
             << placements[i].y << " "
             << placements[i].rotated << "\n";
    }

    return 0;
}
