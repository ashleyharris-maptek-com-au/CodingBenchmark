using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        // Read input
        string[] firstLine = Console.ReadLine().Split();
        int n = int.Parse(firstLine[0]);
        int stockLength = int.Parse(firstLine[1]);

        string[] cutStrings = Console.ReadLine().Split();
        int[] cuts = new int[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = int.Parse(cutStrings[i]);
        }

        // Sort indices by cut length descending (First Fit Decreasing)
        int[] indices = new int[n];
        for (int i = 0; i < n; i++)
        {
            indices[i] = i;
        }
        Array.Sort(indices, (a, b) => cuts[b].CompareTo(cuts[a]));

        // Bin packing using First Fit Decreasing
        List<List<int>> bins = new List<List<int>>();
        List<int> binSpace = new List<int>();

        foreach (int idx in indices)
        {
            int cutLength = cuts[idx];

            // Find first bin with enough space
            int binIdx = -1;
            for (int i = 0; i < bins.Count; i++)
            {
                if (binSpace[i] >= cutLength)
                {
                    binIdx = i;
                    break;
                }
            }

            if (binIdx == -1)
            {
                // Create new bin
                bins.Add(new List<int> { idx });
                binSpace.Add(stockLength - cutLength);
            }
            else
            {
                // Add to existing bin
                bins[binIdx].Add(idx);
                binSpace[binIdx] -= cutLength;
            }
        }

        // Output
        Console.WriteLine(bins.Count);
        foreach (var bin in bins)
        {
            Console.WriteLine(string.Join(" ", bin));
        }
    }
}
