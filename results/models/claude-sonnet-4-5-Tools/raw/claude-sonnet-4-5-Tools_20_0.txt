use std::io::{self, BufRead};
use std::collections::HashMap;

#[derive(Debug, Clone)]
struct Sample {
    hole_id: usize,
    sample_idx: usize,
    position: (f64, f64, f64),
    properties: Vec<f64>,
}

#[derive(Debug)]
struct Drillhole {
    id: usize,
    start: (f64, f64, f64),
    direction: (f64, f64, f64),
    samples: Vec<Vec<f64>>,
}

#[derive(Debug)]
struct Outlier {
    hole_id: usize,
    sample_idx: usize,
    property_idx: usize,
    confidence: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines().filter_map(Result::ok);

    // Read header
    let first_line = lines.next().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let _num_holes: usize = parts[0].parse().unwrap();
    let num_properties: usize = parts[1].parse().unwrap();

    let property_names: Vec<String> = lines.next().unwrap()
        .split_whitespace()
        .map(|s| s.to_string())
        .collect();

    // Read drillholes and build flat sample list
    let mut all_samples = Vec::new();

    while let Some(line) = lines.next() {
        let parts: Vec<&str> = line.split_whitespace().collect();
        if parts.len() < 9 {
            continue;
        }

        let hole_id: usize = parts[0].parse().unwrap_or(0);
        let start = (
            parts[1].parse().unwrap_or(0.0),
            parts[2].parse().unwrap_or(0.0),
            parts[3].parse().unwrap_or(0.0),
        );
        let direction = (
            parts[4].parse().unwrap_or(0.0),
            parts[5].parse().unwrap_or(0.0),
            parts[6].parse().unwrap_or(0.0),
        );
        let num_samples: usize = parts[8].parse().unwrap_or(0);

        for sample_idx in 0..num_samples {
            if let Some(sample_line) = lines.next() {
                let sample_parts: Vec<&str> = sample_line.split_whitespace().collect();
                if sample_parts.is_empty() {
                    continue;
                }

                let depth: f64 = sample_parts[0].parse().unwrap_or(0.0);
                let properties: Vec<f64> = sample_parts[1..]
                    .iter()
                    .take(num_properties)
                    .filter_map(|s| s.parse().ok())
                    .collect();

                if properties.len() != num_properties {
                    continue;
                }

                let position = (
                    start.0 + direction.0 * depth,
                    start.1 + direction.1 * depth,
                    start.2 + direction.2 * depth,
                );

                all_samples.push(Sample {
                    hole_id,
                    sample_idx,
                    position,
                    properties,
                });
            }
        }
    }

    if all_samples.is_empty() {
        println!("0");
        return;
    }

    // Adaptive outlier detection
    let outliers = if all_samples.len() < 5000 {
        detect_outliers_direct(&all_samples, num_properties)
    } else {
        detect_outliers_gridded(&all_samples, num_properties)
    };

    // Output results
    println!("{}", outliers.len());
    for outlier in outliers {
        println!("{} {} {} {:.6}",
            outlier.hole_id,
            outlier.sample_idx,
            property_names.get(outlier.property_idx).map(|s| s.as_str()).unwrap_or("unknown"),
            outlier.confidence
        );
    }
}

fn distance_squared(p1: (f64, f64, f64), p2: (f64, f64, f64)) -> f64 {
    let dx = p1.0 - p2.0;
    let dy = p1.1 - p2.1;
    let dz = p1.2 - p2.2;
    dx * dx + dy * dy + dz * dz
}

fn detect_outliers_direct(samples: &[Sample], num_properties: usize) -> Vec<Outlier> {
    let mut outliers = Vec::new();
    let k = 15.min(samples.len().saturating_sub(1));

    if k < 3 {
        return outliers;
    }

    for (i, sample) in samples.iter().enumerate() {
        // Find K nearest neighbors
        let mut distances: Vec<(usize, f64)> = samples.iter()
            .enumerate()
            .filter(|(j, _)| *j != i)
            .map(|(j, other)| (j, distance_squared(sample.position, other.position)))
            .collect();

        distances.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap_or(std::cmp::Ordering::Equal));

        let neighbor_indices: Vec<usize> = distances.iter()
            .take(k)
            .map(|(idx, _)| *idx)
            .collect();

        // Check each property for outliers
        for prop_idx in 0..num_properties {
            if prop_idx >= sample.properties.len() {
                continue;
            }

            let value = sample.properties[prop_idx];
            if !value.is_finite() {
                continue;
            }

            let neighbor_values: Vec<f64> = neighbor_indices.iter()
                .filter_map(|&j| {
                    samples.get(j)
                        .and_then(|s| s.properties.get(prop_idx))
                        .copied()
                        .filter(|v| v.is_finite())
                })
                .collect();

            if neighbor_values.len() < 3 {
                continue;
            }

            // Calculate statistics
            let mean = neighbor_values.iter().sum::<f64>() / neighbor_values.len() as f64;
            let variance = neighbor_values.iter()
                .map(|v| (v - mean).powi(2))
                .sum::<f64>() / neighbor_values.len() as f64;
            let std_dev = variance.sqrt();

            // Z-score outlier detection
            if std_dev > 1e-9 {
                let z_score = ((value - mean) / std_dev).abs();
                if z_score > 3.5 {
                    outliers.push(Outlier {
                        hole_id: sample.hole_id,
                        sample_idx: sample.sample_idx,
                        property_idx: prop_idx,
                        confidence: z_score,
                    });
                }
            }
        }
    }

    // Sort by confidence descending
    outliers.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap_or(std::cmp::Ordering::Equal));
    outliers
}

fn detect_outliers_gridded(samples: &[Sample], num_properties: usize) -> Vec<Outlier> {
    // Build spatial grid for efficient neighbor search
    let grid = build_grid(samples);
    let cell_size = estimate_cell_size(samples);
    let mut outliers = Vec::new();
    let k = 15.min(samples.len().saturating_sub(1));

    if k < 3 {
        return outliers;
    }

    for (i, sample) in samples.iter().enumerate() {
        // Get nearby samples from grid
        let nearby = get_nearby_samples(&grid, sample.position, cell_size, samples);

        if nearby.len() < k + 1 {
            continue;
        }

        // Find K nearest neighbors
        let mut distances: Vec<(usize, f64)> = nearby.iter()
            .filter(|&&j| j != i)
            .map(|&j| (j, distance_squared(sample.position, samples[j].position)))
            .collect();

        distances.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap_or(std::cmp::Ordering::Equal));

        let neighbor_indices: Vec<usize> = distances.iter()
            .take(k)
            .map(|(idx, _)| *idx)
            .collect();

        if neighbor_indices.len() < 3 {
            continue;
        }

        // Check each property
        for prop_idx in 0..num_properties {
            if prop_idx >= sample.properties.len() {
                continue;
            }

            let value = sample.properties[prop_idx];
            if !value.is_finite() {
                continue;
            }

            let neighbor_values: Vec<f64> = neighbor_indices.iter()
                .filter_map(|&j| {
                    samples.get(j)
                        .and_then(|s| s.properties.get(prop_idx))
                        .copied()
                        .filter(|v| v.is_finite())
                })
                .collect();

            if neighbor_values.len() < 3 {
                continue;
            }

            let mean = neighbor_values.iter().sum::<f64>() / neighbor_values.len() as f64;
            let variance = neighbor_values.iter()
                .map(|v| (v - mean).powi(2))
                .sum::<f64>() / neighbor_values.len() as f64;
            let std_dev = variance.sqrt();

            if std_dev > 1e-9 {
                let z_score = ((value - mean) / std_dev).abs();
                if z_score > 3.5 {
                    outliers.push(Outlier {
                        hole_id: sample.hole_id,
                        sample_idx: sample.sample_idx,
                        property_idx: prop_idx,
                        confidence: z_score,
                    });
                }
            }
        }
    }

    outliers.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap_or(std::cmp::Ordering::Equal));
    outliers
}

type Grid = HashMap<(i32, i32, i32), Vec<usize>>;

fn estimate_cell_size(samples: &[Sample]) -> f64 {
    if samples.is_empty() {
        return 100.0;
    }

    let first = samples[0].position;
    let mut min = first;
    let mut max = first;

    for sample in samples.iter().skip(1) {
        let p = sample.position;
        min.0 = min.0.min(p.0);
        min.1 = min.1.min(p.1);
        min.2 = min.2.min(p.2);
        max.0 = max.0.max(p.0);
        max.1 = max.1.max(p.1);
        max.2 = max.2.max(p.2);
    }

    let extent_x = (max.0 - min.0).abs().max(1.0);
    let extent_y = (max.1 - min.1).abs().max(1.0);
    let extent_z = (max.2 - min.2).abs().max(1.0);
    let avg_extent = (extent_x + extent_y + extent_z) / 3.0;

    // Target ~20 samples per cell
    let target_cells = (samples.len() as f64 / 20.0).max(8.0);
    (avg_extent / target_cells.cbrt()).max(1.0)
}

fn grid_key(pos: (f64, f64, f64), cell_size: f64) -> (i32, i32, i32) {
    (
        (pos.0 / cell_size).floor() as i32,
        (pos.1 / cell_size).floor() as i32,
        (pos.2 / cell_size).floor() as i32,
    )
}

fn build_grid(samples: &[Sample]) -> Grid {
    let cell_size = estimate_cell_size(samples);
    let mut grid = HashMap::new();

    for (i, sample) in samples.iter().enumerate() {
        let key = grid_key(sample.position, cell_size);
        grid.entry(key).or_insert_with(Vec::new).push(i);
    }

    grid
}

fn get_nearby_samples(grid: &Grid, pos: (f64, f64, f64), cell_size: f64, _samples: &[Sample]) -> Vec<usize> {
    let key = grid_key(pos, cell_size);
    let mut nearby = Vec::new();

    // Search 3x3x3 neighborhood
    for dx in -1..=1 {
        for dy in -1..=1 {
            for dz in -1..=1 {
                let neighbor_key = (key.0 + dx, key.1 + dy, key.2 + dz);
                if let Some(indices) = grid.get(&neighbor_key) {
                    nearby.extend_from_slice(indices);
                }
            }
        }
    }

    nearby
}
