#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <unordered_set>
#include <unordered_map>

using namespace std;

const long long MOD1 = 1000000007LL;
const long long MOD2 = 1000000009LL;
const long long BASE = 256;

// Naive approach for small inputs
string naiveLCS(const vector<string>& strings) {
    if (strings.empty()) return "";

    // Find shortest string
    int minIdx = 0;
    for (int i = 1; i < strings.size(); i++) {
        if (strings[i].length() < strings[minIdx].length()) {
            minIdx = i;
        }
    }

    const string& shortest = strings[minIdx];
    string result = "";

    // Check all substrings of shortest string, longest first
    for (int len = shortest.length(); len > result.length(); len--) {
        for (int start = 0; start + len <= shortest.length(); start++) {
            string candidate = shortest.substr(start, len);
            bool found = true;

            for (int s = 0; s < strings.size(); s++) {
                if (s == minIdx) continue;
                if (strings[s].find(candidate) == string::npos) {
                    found = false;
                    break;
                }
            }

            if (found) {
                return candidate;
            }
        }
    }

    return result;
}

// Rolling hash based approach for larger inputs
class RollingHash {
public:
    vector<long long> pow1, pow2;

    void init(int maxLen) {
        pow1.resize(maxLen + 1);
        pow2.resize(maxLen + 1);
        pow1[0] = pow2[0] = 1;
        for (int i = 1; i <= maxLen; i++) {
            pow1[i] = (pow1[i-1] * BASE) % MOD1;
            pow2[i] = (pow2[i-1] * BASE) % MOD2;
        }
    }

    void getAllHashes(const string& s, int len, unordered_set<long long>& hashes) {
        if ((int)s.length() < len) return;

        long long h1 = 0, h2 = 0;
        for (int i = 0; i < len; i++) {
            h1 = (h1 * BASE + (unsigned char)s[i]) % MOD1;
            h2 = (h2 * BASE + (unsigned char)s[i]) % MOD2;
        }
        hashes.insert(h1 * MOD2 + h2);

        for (size_t i = len; i < s.length(); i++) {
            h1 = ((h1 - (unsigned char)s[i - len] * pow1[len - 1]) % MOD1 + MOD1) % MOD1;
            h1 = (h1 * BASE + (unsigned char)s[i]) % MOD1;

            h2 = ((h2 - (unsigned char)s[i - len] * pow2[len - 1]) % MOD2 + MOD2) % MOD2;
            h2 = (h2 * BASE + (unsigned char)s[i]) % MOD2;

            hashes.insert(h1 * MOD2 + h2);
        }
    }

    void getAllHashesWithStrings(const string& s, int len,
                                  unordered_map<long long, string>& hashToStr) {
        if ((int)s.length() < len) return;

        long long h1 = 0, h2 = 0;
        for (int i = 0; i < len; i++) {
            h1 = (h1 * BASE + (unsigned char)s[i]) % MOD1;
            h2 = (h2 * BASE + (unsigned char)s[i]) % MOD2;
        }
        long long combined = h1 * MOD2 + h2;
        hashToStr[combined] = s.substr(0, len);

        for (size_t i = len; i < s.length(); i++) {
            h1 = ((h1 - (unsigned char)s[i - len] * pow1[len - 1]) % MOD1 + MOD1) % MOD1;
            h1 = (h1 * BASE + (unsigned char)s[i]) % MOD1;

            h2 = ((h2 - (unsigned char)s[i - len] * pow2[len - 1]) % MOD2 + MOD2) % MOD2;
            h2 = (h2 * BASE + (unsigned char)s[i]) % MOD2;

            combined = h1 * MOD2 + h2;
            if (hashToStr.find(combined) == hashToStr.end()) {
                hashToStr[combined] = s.substr(i - len + 1, len);
            }
        }
    }
};

bool hasCommonSubstring(const vector<string>& strings, int len, RollingHash& rh) {
    if (len == 0) return true;

    unordered_set<long long> hashes;
    rh.getAllHashes(strings[0], len, hashes);

    if (hashes.empty()) return false;

    for (size_t s = 1; s < strings.size(); s++) {
        unordered_set<long long> currentHashes;
        rh.getAllHashes(strings[s], len, currentHashes);

        unordered_set<long long> intersection;
        for (const auto& h : hashes) {
            if (currentHashes.count(h)) {
                intersection.insert(h);
            }
        }

        hashes = move(intersection);
        if (hashes.empty()) return false;
    }

    return true;
}

string hashBasedLCS(const vector<string>& strings) {
    if (strings.empty()) return "";
    if (strings.size() == 1) return strings[0];

    int minLen = strings[0].length();
    for (const auto& s : strings) {
        minLen = min(minLen, (int)s.length());
        if (minLen == 0) return "";
    }

    RollingHash rh;
    rh.init(minLen);

    // Binary search on length
    int left = 0, right = minLen;
    int maxLen = 0;

    while (left <= right) {
        int mid = (left + right) / 2;
        if (hasCommonSubstring(strings, mid, rh)) {
            maxLen = mid;
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }

    if (maxLen == 0) return "";

    // Find the actual substring
    unordered_map<long long, string> hashToStr;
    rh.getAllHashesWithStrings(strings[0], maxLen, hashToStr);

    unordered_set<long long> hashes;
    for (const auto& [h, _] : hashToStr) {
        hashes.insert(h);
    }

    for (size_t s = 1; s < strings.size(); s++) {
        unordered_set<long long> currentHashes;
        rh.getAllHashes(strings[s], maxLen, currentHashes);

        unordered_set<long long> intersection;
        for (const auto& h : hashes) {
            if (currentHashes.count(h)) {
                intersection.insert(h);
            }
        }

        hashes = move(intersection);
    }

    if (!hashes.empty()) {
        return hashToStr[*hashes.begin()];
    }

    return "";
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    cin >> n;
    cin.ignore();

    vector<string> strings(n);
    for (int i = 0; i < n; i++) {
        getline(cin, strings[i]);
    }

    // Adaptive strategy
    long long totalSize = 0;
    int minLen = strings.empty() ? 0 : strings[0].length();

    for (const auto& s : strings) {
        totalSize += s.length();
        minLen = min(minLen, (int)s.length());
    }

    string result;

    // Use naive approach for small inputs
    if (totalSize < 10000 || (n <= 3 && minLen < 1000)) {
        result = naiveLCS(strings);
    } else {
        result = hashBasedLCS(strings);
    }

    cout << result << endl;

    return 0;
}
