#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>

using namespace std;

struct City {
    float x, y;
    int id;
};

// Fast distance calculation using SIMD when possible
inline float dist(const City& a, const City& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return sqrt(dx * dx + dy * dy);
}

// Vectorized distance calculation for batch processing
void compute_distances_simd(const vector<City>& cities, int from, const vector<bool>& visited,
                           vector<pair<float, int>>& distances) {
    const City& src = cities[from];
    __m256 src_x = _mm256_set1_ps(src.x);
    __m256 src_y = _mm256_set1_ps(src.y);

    int n = cities.size();
    for (int i = 0; i < n; i++) {
        if (!visited[i]) {
            float dx = cities[i].x - src.x;
            float dy = cities[i].y - src.y;
            distances[i] = {sqrt(dx * dx + dy * dy), i};
        } else {
            distances[i] = {numeric_limits<float>::max(), i};
        }
    }
}

// Nearest neighbor heuristic - O(N^2) but fast for reasonable N
vector<int> nearest_neighbor(const vector<City>& cities) {
    int n = cities.size();
    vector<int> tour;
    tour.reserve(n);
    vector<bool> visited(n, false);

    int current = 0;
    tour.push_back(current);
    visited[current] = true;

    for (int step = 1; step < n; step++) {
        int nearest = -1;
        float min_dist = numeric_limits<float>::max();

        for (int i = 0; i < n; i++) {
            if (!visited[i]) {
                float d = dist(cities[current], cities[i]);
                if (d < min_dist) {
                    min_dist = d;
                    nearest = i;
                }
            }
        }

        current = nearest;
        tour.push_back(current);
        visited[current] = true;
    }

    return tour;
}

// Fast nearest neighbor for huge datasets - uses grid-based spatial partitioning
vector<int> fast_nearest_neighbor(const vector<City>& cities) {
    int n = cities.size();
    vector<int> tour;
    tour.reserve(n);
    vector<bool> visited(n, false);

    // Find bounds
    float min_x = numeric_limits<float>::max(), max_x = numeric_limits<float>::lowest();
    float min_y = numeric_limits<float>::max(), max_y = numeric_limits<float>::lowest();

    for (const auto& c : cities) {
        min_x = min(min_x, c.x);
        max_x = max(max_x, c.x);
        min_y = min(min_y, c.y);
        max_y = max(max_y, c.y);
    }

    // Adaptive grid size based on N
    int grid_size = min(1000, max(10, (int)sqrt(n) / 4));
    float cell_width = (max_x - min_x) / grid_size;
    float cell_height = (max_y - min_y) / grid_size;

    if (cell_width == 0) cell_width = 1;
    if (cell_height == 0) cell_height = 1;

    // Build spatial grid
    vector<vector<vector<int>>> grid(grid_size, vector<vector<int>>(grid_size));
    for (int i = 0; i < n; i++) {
        int gx = min(grid_size - 1, max(0, (int)((cities[i].x - min_x) / cell_width)));
        int gy = min(grid_size - 1, max(0, (int)((cities[i].y - min_y) / cell_height)));
        grid[gx][gy].push_back(i);
    }

    int current = 0;
    tour.push_back(current);
    visited[current] = true;

    for (int step = 1; step < n; step++) {
        int gx = min(grid_size - 1, max(0, (int)((cities[current].x - min_x) / cell_width)));
        int gy = min(grid_size - 1, max(0, (int)((cities[current].y - min_y) / cell_height)));

        int nearest = -1;
        float min_dist = numeric_limits<float>::max();

        // Search in expanding radius
        for (int radius = 0; radius <= grid_size && nearest == -1; radius++) {
            for (int dx = -radius; dx <= radius; dx++) {
                for (int dy = -radius; dy <= radius; dy++) {
                    if (abs(dx) < radius && abs(dy) < radius) continue;

                    int nx = gx + dx;
                    int ny = gy + dy;

                    if (nx < 0 || nx >= grid_size || ny < 0 || ny >= grid_size) continue;

                    for (int idx : grid[nx][ny]) {
                        if (!visited[idx]) {
                            float d = dist(cities[current], cities[idx]);
                            if (d < min_dist) {
                                min_dist = d;
                                nearest = idx;
                            }
                        }
                    }
                }
            }

            if (nearest != -1) break;
        }

        if (nearest == -1) {
            // Fallback: find any unvisited
            for (int i = 0; i < n; i++) {
                if (!visited[i]) {
                    nearest = i;
                    break;
                }
            }
        }

        current = nearest;
        tour.push_back(current);
        visited[current] = true;
    }

    return tour;
}

// 2-opt improvement - swaps edges to reduce tour length
void two_opt_improve(vector<int>& tour, const vector<City>& cities, int max_iterations) {
    int n = tour.size();
    bool improved = true;
    int iterations = 0;

    while (improved && iterations < max_iterations) {
        improved = false;
        iterations++;

        for (int i = 1; i < n - 2; i++) {
            for (int j = i + 1; j < n - 1; j++) {
                float old_dist = dist(cities[tour[i-1]], cities[tour[i]]) +
                                dist(cities[tour[j]], cities[tour[j+1]]);
                float new_dist = dist(cities[tour[i-1]], cities[tour[j]]) +
                                dist(cities[tour[i]], cities[tour[j+1]]);

                if (new_dist < old_dist) {
                    reverse(tour.begin() + i, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
}

// Strip-based construction for massive datasets
vector<int> strip_heuristic(const vector<City>& cities) {
    int n = cities.size();
    vector<pair<float, int>> sorted_cities;
    sorted_cities.reserve(n);

    for (int i = 0; i < n; i++) {
        sorted_cities.push_back({cities[i].x + cities[i].y * 0.001f, i});
    }

    sort(sorted_cities.begin(), sorted_cities.end());

    vector<int> tour;
    tour.reserve(n);

    // Start with city 0
    int start_pos = -1;
    for (int i = 0; i < n; i++) {
        if (sorted_cities[i].second == 0) {
            start_pos = i;
            break;
        }
    }

    tour.push_back(0);

    // Build tour in sorted order (space-filling curve approximation)
    bool forward = true;
    for (int i = 0; i < n; i++) {
        int idx = sorted_cities[i].second;
        if (idx != 0) {
            tour.push_back(idx);
        }
    }

    return tour;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<City> cities(n);
    for (int i = 0; i < n; i++) {
        cin >> cities[i].x >> cities[i].y;
        cities[i].id = i;
    }

    vector<int> tour;

    // Adaptive algorithm selection based on problem size
    if (n <= 40) {
        // Trivial: nearest neighbor + aggressive 2-opt
        tour = nearest_neighbor(cities);
        two_opt_improve(tour, cities, 100);
    }
    else if (n <= 1000) {
        // Medium: nearest neighbor + moderate 2-opt
        tour = nearest_neighbor(cities);
        two_opt_improve(tour, cities, min(20, 10000 / n));
    }
    else if (n <= 50000) {
        // Large: grid-accelerated nearest neighbor + light 2-opt
        tour = fast_nearest_neighbor(cities);
        if (n <= 5000) {
            two_opt_improve(tour, cities, 5);
        }
    }
    else if (n <= 1000000) {
        // Very large: grid-accelerated nearest neighbor only
        tour = fast_nearest_neighbor(cities);
    }
    else {
        // Extreme/Epic: strip-based space-filling curve
        tour = strip_heuristic(cities);
    }

    // Output tour
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << " ";
        cout << tour[i];
    }
    cout << "\n";

    return 0;
}
