const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

const grid = data.probability;
const drones = data.drones || [];
const maxSteps = data.maxStepsPerDrone || 0;

const h = grid.length;
const w = h > 0 ? grid[0].length : 0;

const covered = Array.from({ length: h }, () => Array(w).fill(false));

function inBounds(x, y) {
  return x >= 0 && y >= 0 && x < w && y < h;
}

const paths = [];

for (let d = 0; d < drones.length; d++) {
  const drone = drones[d];
  let x = drone.x;
  let y = drone.y;

  const path = [];
  if (inBounds(x, y) && grid[y][x] !== -1) {
    path.push([x, y]);
    if (!covered[y][x]) covered[y][x] = true;
  } else {
    // If starting cell invalid, still output as given
    path.push([x, y]);
  }

  for (let step = 0; step < maxSteps; step++) {
    let bestScore = -Infinity;
    let bestPos = [x, y];

    const candidates = [
      [x, y],
      [x + 1, y],
      [x - 1, y],
      [x, y + 1],
      [x, y - 1],
    ];

    for (const [nx, ny] of candidates) {
      if (!inBounds(nx, ny)) continue;
      if (grid[ny][nx] === -1) continue;
      const score = covered[ny][nx] ? 0 : grid[ny][nx];
      if (score > bestScore) {
        bestScore = score;
        bestPos = [nx, ny];
      }
    }

    x = bestPos[0];
    y = bestPos[1];
    path.push([x, y]);
    if (inBounds(x, y) && grid[y][x] !== -1) {
      if (!covered[y][x]) covered[y][x] = true;
    }
  }

  paths.push(path);
}

process.stdout.write(JSON.stringify({ paths }));
