```typescript
declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number; priority: number };

const grid: string[] = data.grid;
const robotsInput: Robot[] = data.robots;
const jobsInput: Job[] = data.jobs;

const H = grid.length;
const W = H > 0 ? grid[0].length : 0;
const size = W * H;

// Build open cell map
const open = new Uint8Array(size);
for (let y = 0; y < H; y++) {
  const row = grid[y];
  for (let x = 0; x < W; x++) {
    open[y * W + x] = row.charAt(x) === '.' ? 1 : 0;
  }
}

// Prepare robots sorted by id
let maxRobotId = -1;
for (const r of robotsInput) if (r.id > maxRobotId) maxRobotId = r.id;
const robots: Robot[] = robotsInput.slice().sort((a, b) => a.id - b.id);
const R = robots.length;

// Prepare jobs with indices
const J = jobsInput.length;
const jobs: Job[] = jobsInput.slice();

// Job positions
const jobCellIdx = new Int32Array(J);
for (let j = 0; j < J; j++) {
  jobCellIdx[j] = jobs[j].y * W + jobs[j].x;
}

// Points list: robots first, then jobs
const P = R + J;
const pointCellIdx = new Int32Array(P);
for (let i = 0; i < R; i++) {
  pointCellIdx[i] = robots[i].y * W + robots[i].x;
}
for (let j = 0; j < J; j++) {
  pointCellIdx[R + j] = jobCellIdx[j];
}

// BFS from a point to all jobs
function bfsToJobs(startIdx: number): Int32Array {
  const dist = new Int32Array(size);
  dist.fill(-1);
  const queue = new Int32Array(size);
  let head = 0, tail = 0;

  dist[startIdx] = 0;
  queue[tail++] = startIdx;

  while (head < tail) {
    const idx = queue[head++];
    const d = dist[idx] + 1;
    const x = idx % W;
    const y = (idx / W) | 0;

    let n: number;
    if (x > 0) {
      n = idx - 1;
      if (open[n] && dist[n] === -1) { dist[n] = d; queue[tail++] = n; }
    }
    if (x + 1 < W) {
      n = idx + 1;
      if (open[n] && dist[n] === -1) { dist[n] = d; queue[tail++] = n; }
    }
    if (y > 0) {
      n = idx - W;
      if (open[n] && dist[n] === -1) { dist[n] = d; queue[tail++] = n; }
    }
    if (y + 1 < H) {
      n = idx + W;
      if (open[n] && dist[n] === -1) { dist[n] = d; queue[tail++] = n; }
    }
  }

  const res = new Int32Array(J);
  for (let j = 0; j < J; j++) {
    const d = dist[jobCellIdx[j]];
    res[j] = d >= 0 ? d : 1000000000;
  }
  return res;
}

// Precompute distances from all points to all jobs
const distToJobs: Int32Array[] = new Array(P);
for (let i = 0; i < P; i++) {
  distToJobs[i] = bfsToJobs(pointCellIdx[i]);
}

// Sort jobs by priority descending, tie by id ascending
const jobOrder = Array.from({ length: J }, (_, i) => i);
jobOrder.sort((a, b) => {
  const pa = jobs[a].priority ?? 0;
  const pb = jobs[b].priority ?? 0;
  if (pa !== pb) return pb - pa;
  return jobs[a].id - jobs[b].id;
});

// Assignments output array indexed by robot id
const assignments: number[][] = Array.from({ length: maxRobotId + 1 }, () => []);

// Current position index for each robot in robots[] list
const robotPosIndex = new Int32Array(R);
for (let i = 0; i < R; i++) robotPosIndex[i] = i;

// Greedy assignment
for (const jobIdx of jobOrder) {
  let bestRobotIdx = -1;
  let bestDist = Infinity;
  let bestRobotId = Infinity;

  for (let i = 0; i < R; i++) {
    const dist = distToJobs[robotPosIndex[i]][jobIdx];
    const rid = robots[i].id;
    if (dist < bestDist || (dist === bestDist && rid < bestRobotId)) {
      bestDist = dist;
      bestRobotIdx = i;
      bestRobotId = rid;
    }
  }

  if (bestRobotIdx >= 0) {
    assignments[bestRobotId].push(jobs[jobIdx].id);
    robotPosIndex[bestRobotIdx] = R + jobIdx;
  }
}

// Output
const output = JSON.stringify({ assignments });
process.stdout.write(output);
```
