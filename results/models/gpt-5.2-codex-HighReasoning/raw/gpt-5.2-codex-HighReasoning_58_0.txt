const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input ? JSON.parse(input) : {};

const grid = data.grid || [];
const robots = data.robots || [];
const jobs = data.jobs || [];

const H = grid.length;
const W = H > 0 ? grid[0].length : 0;
const N = H * W;

// Precompute open cells
const open = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = grid[y];
  for (let x = 0; x < W; x++) {
    if (row[x] === '.') open[y * W + x] = 1;
  }
}

// Jobs with index
const jobsArr = jobs.map((j, idx) => ({
  id: j.id,
  x: j.x,
  y: j.y,
  priority: j.priority || 0,
  index: idx
}));

// Robots list
const R = robots.length;
const J = jobsArr.length;

// Determine max robot id for output
let maxRobotId = -1;
for (const r of robots) if (r.id > maxRobotId) maxRobotId = r.id;
if (maxRobotId < 0) maxRobotId = -1;

// Sort jobs by priority desc then id asc
jobsArr.sort((a, b) => {
  if (b.priority !== a.priority) return b.priority - a.priority;
  return a.id - b.id;
});

// BFS function
function bfs(startX, startY) {
  const dist = new Int32Array(N);
  dist.fill(-1);
  if (startX < 0 || startX >= W || startY < 0 || startY >= H) return dist;
  const startIdx = startY * W + startX;
  if (!open[startIdx]) return dist;

  const queue = new Int32Array(N);
  let head = 0, tail = 0;
  queue[tail++] = startIdx;
  dist[startIdx] = 0;

  while (head < tail) {
    const idx = queue[head++];
    const d = dist[idx];
    const x = idx % W;
    const y = (idx / W) | 0;

    let nidx;

    if (y > 0) {
      nidx = idx - W;
      if (open[nidx] && dist[nidx] === -1) {
        dist[nidx] = d + 1;
        queue[tail++] = nidx;
      }
    }
    if (y + 1 < H) {
      nidx = idx + W;
      if (open[nidx] && dist[nidx] === -1) {
        dist[nidx] = d + 1;
        queue[tail++] = nidx;
      }
    }
    if (x > 0) {
      nidx = idx - 1;
      if (open[nidx] && dist[nidx] === -1) {
        dist[nidx] = d + 1;
        queue[tail++] = nidx;
      }
    }
    if (x + 1 < W) {
      nidx = idx + 1;
      if (open[nidx] && dist[nidx] === -1) {
        dist[nidx] = d + 1;
        queue[tail++] = nidx;
      }
    }
  }
  return dist;
}

// Precompute distances from each robot to each job
const robotJobDists = new Array(R);
for (let i = 0; i < R; i++) {
  const r = robots[i];
  const distMap = bfs(r.x, r.y);
  const dists = new Int32Array(J);
  for (let j = 0; j < J; j++) {
    const job = jobsArr[j];
    const idx = job.y * W + job.x;
    let d = (idx >= 0 && idx < N) ? distMap[idx] : -1;
    if (d < 0) d = 1e9;
    dists[job.index] = d;
  }
  robotJobDists[i] = dists;
}

// Assignments per robot index
const assigned = new Array(R);
for (let i = 0; i < R; i++) assigned[i] = [];

// Greedy assignment: closest robot based on start distance
for (const job of jobsArr) {
  let bestIdx = 0;
  let bestDist = robotJobDists[0][job.index];
  let bestRobotId = robots[0] ? robots[0].id : 0;
  for (let i = 1; i < R; i++) {
    const d = robotJobDists[i][job.index];
    const rid = robots[i].id;
    if (d < bestDist || (d === bestDist && rid < bestRobotId)) {
      bestDist = d;
      bestIdx = i;
      bestRobotId = rid;
    }
  }
  if (R > 0) assigned[bestIdx].push(job.id);
}

// Output assignments by robot id
const output = new Array(maxRobotId + 1);
for (let i = 0; i <= maxRobotId; i++) output[i] = [];
for (let i = 0; i < R; i++) {
  const rid = robots[i].id;
  output[rid] = assigned[i];
}

const result = { assignments: output };
console.log(JSON.stringify(result));
