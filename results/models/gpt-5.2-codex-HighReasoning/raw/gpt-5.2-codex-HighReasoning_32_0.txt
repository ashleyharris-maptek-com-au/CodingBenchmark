Below is a complete C++20 solution (MSVC‑compatible) for the Steiner Tree problem.  
It uses an **exact Dreyfus–Wagner DP** when the number of terminals is small (≤15),  
and a **heuristic MST on metric closure** for larger terminal sets.

The output is a **connected, acyclic** subgraph that contains all terminals.

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <set>
#include <unordered_map>
#include <algorithm>
#include <tuple>
#include <functional>
#include <limits>
#include <numeric>

using namespace std;

struct Edge {
    int to;
    long long w;
};

struct Parent {
    int type;   // -1 none, 0 base, 1 merge, 2 edge
    int prev;
    int sub;
    Parent(int t = -1, int p = -1, int s = -1) : type(t), prev(p), sub(s) {}
};

struct DSU {
    vector<int> p, r;
    DSU(int n = 0) : p(n), r(n, 0) { iota(p.begin(), p.end(), 0); }
    int find(int x) { return p[x] == x ? x : p[x] = find(p[x]); }
    bool unite(int a, int b) {
        a = find(a); b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) r[a]++;
        return true;
    }
};

static const long long INF = (1LL << 60);

long long edgeKey(int u, int v) {
    if (u > v) swap(u, v);
    return (static_cast<long long>(u) << 32) | (unsigned int)v;
}

void dijkstra_single(int source, const vector<vector<Edge>>& g,
                     vector<long long>& dist, vector<int>& prev) {
    int n = (int)g.size();
    dist.assign(n, INF);
    prev.assign(n, -1);
    dist[source] = 0;
    using P = pair<long long, int>;
    priority_queue<P, vector<P>, greater<P>> pq;
    pq.push({0, source});
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d != dist[u]) continue;
        for (auto &e : g[u]) {
            long long nd = d + e.w;
            if (nd < dist[e.to]) {
                dist[e.to] = nd;
                prev[e.to] = u;
                pq.push({nd, e.to});
            }
        }
    }
}

void multi_dijkstra(const vector<vector<Edge>>& g,
                    vector<long long>& dist, vector<Parent>& parent) {
    int n = (int)g.size();
    using P = pair<long long, int>;
    priority_queue<P, vector<P>, greater<P>> pq;
    for (int v = 0; v < n; ++v) {
        if (dist[v] < INF) pq.push({dist[v], v});
    }
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d != dist[u]) continue;
        for (auto &e : g[u]) {
            long long nd = d + e.w;
            if (nd < dist[e.to]) {
                dist[e.to] = nd;
                parent[e.to] = Parent(2, u, -1);
                pq.push({nd, e.to});
            }
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<vector<Edge>> g(n);
    unordered_map<long long, long long> wmap;

    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        g[u].push_back({v, w});
        g[v].push_back({u, w});
        long long key = edgeKey(u, v);
        auto it = wmap.find(key);
        if (it == wmap.end() || w < it->second) wmap[key] = w;
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) cin >> terminals[i];

    if (k <= 1) {
        cout << 0 << " " << 0 << "\n";
        return 0;
    }

    // ====================== Exact DP for small k ======================
    if (k <= 15) {
        int K = k;
        int maxmask = 1 << K;
        vector<vector<long long>> dp(maxmask, vector<long long>(n, INF));
        vector<vector<Parent>> parent(maxmask, vector<Parent>(n));

        // Initialize singleton masks
        for (int i = 0; i < K; ++i) {
            vector<long long> dist;
            vector<int> prev;
            dijkstra_single(terminals[i], g, dist, prev);
            int mask = 1 << i;
            dp[mask] = dist;
            for (int v = 0; v < n; ++v) {
                if (dist[v] >= INF) {
                    parent[mask][v] = Parent(-1, -1, -1);
                } else if (v == terminals[i]) {
                    parent[mask][v] = Parent(0, -1, -1);
                } else {
                    parent[mask][v] = Parent(2, prev[v], -1);
                }
            }
        }

        // DP over masks
        for (int mask = 1; mask < maxmask; ++mask) {
            if ((mask & (mask - 1)) == 0) continue; // singleton
            fill(dp[mask].begin(), dp[mask].end(), INF);
            fill(parent[mask].begin(), parent[mask].end(), Parent(-1, -1, -1));

            for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
                int other = mask ^ sub;
                if (sub > other) continue;
                for (int v = 0; v < n; ++v) {
                    if (dp[sub][v] >= INF || dp[other][v] >= INF) continue;
                    long long val = dp[sub][v] + dp[other][v];
                    if (val < dp[mask][v]) {
                        dp[mask][v] = val;
                        parent[mask][v] = Parent(1, -1, sub);
                    }
                }
            }

            // Shortest path closure
            multi_dijkstra(g, dp[mask], parent[mask]);
        }

        int full = maxmask - 1;
        long long best = INF;
        int root = -1;
        for (int v = 0; v < n; ++v) {
            if (dp[full][v] < best) {
                best = dp[full][v];
                root = v;
            }
        }

        set<pair<int,int>> edgeSet;
        function<void(int,int)> build = [&](int mask, int v) {
            Parent p = parent[mask][v];
            if (p.type <= 0) return;
            if (p.type == 1) {
                build(p.sub, v);
                build(mask ^ p.sub, v);
            } else if (p.type == 2) {
                int u = p.prev;
                if (u != -1) {
                    int a = min(u, v), b = max(u, v);
                    edgeSet.insert({a, b});
                    build(mask, u);
                }
            }
        };

        build(full, root);

        vector<pair<int,int>> edges(edgeSet.begin(), edgeSet.end());
        long long total = 0;
        for (auto &e : edges) total += wmap[edgeKey(e.first, e.second)];

        cout << total << " " << edges.size() << "\n";
        for (auto &e : edges) {
            cout << e.first << " " << e.second << "\n";
        }
        return 0;
    }

    // ====================== Heuristic for large k ======================
    // 1) Dijkstra from each terminal
    vector<vector<long long>> dist(k, vector<long long>(n, INF));
    vector<vector<int>> prev(k, vector<int>(n, -1));
    for (int i = 0; i < k; ++i) {
        dijkstra_single(terminals[i], g, dist[i], prev[i]);
    }

    // 2) Build complete graph of terminals & compute MST
    vector<tuple<long long,int,int>> completeEdges;
    for (int i = 0; i < k; ++i) {
        for (int j = i + 1; j < k; ++j) {
            long long w = dist[i][terminals[j]];
            completeEdges.push_back({w, i, j});
        }
    }
    sort(completeEdges.begin(), completeEdges.end());
    DSU dsuT(k);
    vector<pair<int,int>> mstTermEdges;
    for (auto &e : completeEdges) {
        long long w; int i, j;
        tie(w, i, j) = e;
        if (dsuT.unite(i, j)) {
            mstTermEdges.push_back({i, j});
        }
    }

    // 3) Expand paths in original graph
    set<pair<int,int>> unionEdges;
    for (auto &pr : mstTermEdges) {
        int i = pr.first, j = pr.second;
        int src = terminals[i], dst = terminals[j];
        int cur = dst;
        while (cur != src && cur != -1) {
            int p = prev[i][cur];
            if (p == -1) break;
            int a = min(cur, p), b = max(cur, p);
            unionEdges.insert({a, b});
            cur = p;
        }
    }

    // 4) MST on union edges (to remove cycles)
    vector<tuple<long long,int,int>> unionList;
    for (auto &e : unionEdges) {
        long long w = wmap[edgeKey(e.first, e.second)];
        unionList.push_back({w, e.first, e.second});
    }
    sort(unionList.begin(), unionList.end());

    DSU dsuV(n);
    vector<pair<int,int>> finalEdges;
    long long total = 0;
    for (auto &e : unionList) {
        long long w; int u, v;
        tie(w, u, v) = e;
        if (dsuV.unite(u, v)) {
            finalEdges.push_back({u, v});
            total += w;
        }
    }

    cout << total << " " << finalEdges.size() << "\n";
    for (auto &e : finalEdges) {
        cout << e.first << " " << e.second << "\n";
    }
    return 0;
}
```

### Notes
- Uses **exact DP (Dreyfus–Wagner)** when terminals ≤ 15.
- For larger terminal sets, uses **MST on metric closure**, then expands and prunes cycles.
- Output is always a **tree** (connected, acyclic) containing all terminals.
- Compiles with **MSVC /std:c++20** on Windows 11.
