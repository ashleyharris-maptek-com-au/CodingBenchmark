const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

const width = data.width;
const height = data.height;
const grid = data.grid;
const ignitions = data.ignitions || [];
const towns = data.towns || [];
const cutBudget = data.cutBudget || 0;

// BFS to find cells that burn without any cuts
const visited = Array.from({ length: height }, () => Array(width).fill(false));
const queueX = [];
const queueY = [];
let qh = 0;

function inBounds(x, y) {
  return x >= 0 && y >= 0 && x < width && y < height;
}

for (const [x, y] of ignitions) {
  if (inBounds(x, y) && grid[y][x] === '.' && !visited[y][x]) {
    visited[y][x] = true;
    queueX.push(x);
    queueY.push(y);
  }
}

const dirs = [[1,0],[-1,0],[0,1],[0,-1]];
while (qh < queueX.length) {
  const x = queueX[qh];
  const y = queueY[qh];
  qh++;
  for (const [dx, dy] of dirs) {
    const nx = x + dx, ny = y + dy;
    if (inBounds(nx, ny) && !visited[ny][nx] && grid[ny][nx] === '.') {
      visited[ny][nx] = true;
      queueX.push(nx);
      queueY.push(ny);
    }
  }
}

// Collect towns that would burn
const burnableTowns = [];
for (const t of towns) {
  const x = t.x, y = t.y;
  if (!inBounds(x, y)) continue;
  if (grid[y][x] !== '.') continue; // already safe (water/rock)
  if (visited[y][x]) {
    burnableTowns.push(t);
  }
}

// Sort by value descending
burnableTowns.sort((a, b) => b.value - a.value);

const firebreaks = [];
const used = new Set();
let remaining = cutBudget;

for (const t of burnableTowns) {
  if (remaining <= 0) break;
  const key = t.x + ',' + t.y;
  if (used.has(key)) continue;
  used.add(key);
  firebreaks.push([t.x, t.y]);
  remaining--;
}

const output = { firebreaks };
process.stdout.write(JSON.stringify(output));
