Below is a complete, compilable C++20 solution that reads the entire game description from stdin, plans moves offline, and outputs a sequence of moves.  
It supports arbitrary dimensions (2D–20D+) and uses:

- **A\*** (with Manhattan heuristic) to find a path to the nearest food when possible  
- A **greedy fallback** if A\* cannot find a path quickly  
- Safe collision handling with obstacles + self body  
- Works across all dimensional sizes without non‑standard headers  

```cpp
#include <iostream>
#include <vector>
#include <array>
#include <deque>
#include <unordered_set>
#include <unordered_map>
#include <queue>
#include <algorithm>
#include <random>
#include <chrono>
#include <limits>
#include <cmath>

using namespace std;

static int D = 0;
static const int MAXD = 25;

struct Pos {
    array<int, MAXD> c;
    Pos() { c.fill(0); }
};

struct PosHash {
    size_t operator()(Pos const& p) const noexcept {
        size_t h = 0;
        for (int i = 0; i < D; ++i) {
            h ^= std::hash<int>()(p.c[i] + 0x9e3779b9 + (h << 6) + (h >> 2));
        }
        return h;
    }
};

struct PosEq {
    bool operator()(Pos const& a, Pos const& b) const noexcept {
        for (int i = 0; i < D; ++i) {
            if (a.c[i] != b.c[i]) return false;
        }
        return true;
    }
};

bool posEqual(const Pos& a, const Pos& b) {
    for (int i = 0; i < D; ++i) {
        if (a.c[i] != b.c[i]) return false;
    }
    return true;
}

int manhattan(const Pos& a, const Pos& b) {
    int dist = 0;
    for (int i = 0; i < D; ++i) dist += std::abs(a.c[i] - b.c[i]);
    return dist;
}

bool inBounds(const Pos& p, const vector<int>& bounds) {
    for (int i = 0; i < D; ++i) {
        if (p.c[i] < 0 || p.c[i] >= bounds[i]) return false;
    }
    return true;
}

struct Node {
    Pos pos;
    int g;
    int f;
};
struct NodeCmp {
    bool operator()(const Node& a, const Node& b) const {
        return a.f > b.f; // min-heap
    }
};

vector<Pos> astar(const Pos& start, const Pos& goal,
                  const unordered_set<Pos, PosHash, PosEq>& obstacles,
                  const unordered_set<Pos, PosHash, PosEq>& body,
                  const vector<int>& bounds,
                  size_t maxNodes)
{
    priority_queue<Node, vector<Node>, NodeCmp> open;
    unordered_map<Pos, int, PosHash, PosEq> gscore;
    unordered_map<Pos, Pos, PosHash, PosEq> parent;
    unordered_set<Pos, PosHash, PosEq> closed;

    gscore[start] = 0;
    open.push({start, 0, manhattan(start, goal)});
    size_t expanded = 0;

    while (!open.empty()) {
        Node cur = open.top(); open.pop();

        if (closed.find(cur.pos) != closed.end()) continue;
        closed.insert(cur.pos);

        if (posEqual(cur.pos, goal)) {
            // reconstruct path
            vector<Pos> path;
            Pos p = goal;
            path.push_back(p);
            while (!posEqual(p, start)) {
                auto it = parent.find(p);
                if (it == parent.end()) break;
                p = it->second;
                path.push_back(p);
            }
            reverse(path.begin(), path.end());
            if (!path.empty() && posEqual(path.front(), start))
                return path;
            break;
        }

        if (++expanded > maxNodes) break;

        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                Pos nxt = cur.pos;
                nxt.c[axis] += dir;
                if (!inBounds(nxt, bounds)) continue;
                if (obstacles.find(nxt) != obstacles.end()) continue;
                if (body.find(nxt) != body.end() && !posEqual(nxt, goal)) continue;

                int tentative_g = cur.g + 1;
                auto it = gscore.find(nxt);
                if (it == gscore.end() || tentative_g < it->second) {
                    gscore[nxt] = tentative_g;
                    parent[nxt] = cur.pos;
                    int f = tentative_g + manhattan(nxt, goal);
                    open.push({nxt, tentative_g, f});
                }
            }
        }
    }

    return {};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int max_turns;
    if (!(cin >> D >> max_turns)) return 0;

    vector<int> bounds(D);
    for (int i = 0; i < D; ++i) cin >> bounds[i];

    int snake_length;
    Pos head;
    cin >> snake_length;
    for (int i = 0; i < D; ++i) cin >> head.c[i];

    deque<Pos> body;
    body.push_front(head);

    // Attempt to extend body along -axis 0 if length > 1
    for (int k = 1; k < snake_length; ++k) {
        Pos p = head;
        p.c[0] -= k;
        if (p.c[0] < 0) {
            p.c[0] = head.c[0] + k;
            if (p.c[0] >= bounds[0]) break;
        }
        body.push_back(p);
    }

    unordered_set<Pos, PosHash, PosEq> occupied;
    for (auto& p : body) occupied.insert(p);

    int F;
    cin >> F;
    vector<Pos> foods;
    foods.reserve(F);
    unordered_set<Pos, PosHash, PosEq> food_set;
    for (int i = 0; i < F; ++i) {
        Pos f;
        for (int d = 0; d < D; ++d) cin >> f.c[d];
        foods.push_back(f);
        food_set.insert(f);
    }

    int O;
    cin >> O;
    unordered_set<Pos, PosHash, PosEq> obstacles;
    for (int i = 0; i < O; ++i) {
        Pos o;
        for (int d = 0; d < D; ++d) cin >> o.c[d];
        obstacles.insert(o);
    }

    vector<pair<int,int>> output_moves;
    std::mt19937 rng((unsigned)chrono::steady_clock::now().time_since_epoch().count());

    for (int turn = 0; turn < max_turns && !food_set.empty(); ++turn) {
        // choose target food
        vector<pair<int, Pos>> candidates;
        for (auto& f : foods) {
            if (food_set.find(f) == food_set.end()) continue;
            candidates.push_back({manhattan(head, f), f});
        }
        if (candidates.empty()) break;

        sort(candidates.begin(), candidates.end(),
             [](auto& a, auto& b){ return a.first < b.first; });

        Pos next_pos;
        bool found_path = false;

        // try A* for nearest few foods
        size_t maxNodes = 200000;
        int tryCount = min(5, (int)candidates.size());
        for (int i = 0; i < tryCount; ++i) {
            Pos target = candidates[i].second;
            vector<Pos> path = astar(head, target, obstacles, occupied, bounds, maxNodes);
            if (path.size() >= 2) {
                next_pos = path[1];
                found_path = true;
                break;
            }
        }

        auto isFood = [&](const Pos& p)->bool {
            return food_set.find(p) != food_set.end();
        };

        auto isSafe = [&](const Pos& p, bool grow)->bool {
            if (!inBounds(p, bounds)) return false;
            if (obstacles.find(p) != obstacles.end()) return false;
            if (occupied.find(p) != occupied.end()) {
                // Allow moving into tail only if not growing
                Pos tail = body.back();
                if (!grow && posEqual(p, tail)) return true;
                return false;
            }
            return true;
        };

        int bestAxis = -1, bestDir = 0;

        if (!found_path) {
            // Greedy fallback
            int bestScore = std::numeric_limits<int>::max();
            int bestDegree = -1;

            for (int axis = 0; axis < D; ++axis) {
                for (int dir : {-1, 1}) {
                    Pos np = head;
                    np.c[axis] += dir;
                    bool grow = isFood(np);
                    if (!isSafe(np, grow)) continue;

                    int score = std::numeric_limits<int>::max();
                    for (auto& f : foods) {
                        if (food_set.find(f) == food_set.end()) continue;
                        score = min(score, manhattan(np, f));
                    }
                    if (score == std::numeric_limits<int>::max()) score = 0;

                    int degree = 0;
                    for (int ax2 = 0; ax2 < D; ++ax2) {
                        for (int d2 : {-1, 1}) {
                            Pos np2 = np;
                            np2.c[ax2] += d2;
                            if (inBounds(np2, bounds) && obstacles.find(np2) == obstacles.end())
                                degree++;
                        }
                    }

                    if (score < bestScore || (score == bestScore && degree > bestDegree)) {
                        bestScore = score;
                        bestDegree = degree;
                        bestAxis = axis;
                        bestDir = dir;
                    }
                }
            }

            if (bestAxis < 0) break; // no safe move

            next_pos = head;
            next_pos.c[bestAxis] += bestDir;
        } else {
            for (int axis = 0; axis < D; ++axis) {
                int diff = next_pos.c[axis] - head.c[axis];
                if (diff != 0) {
                    bestAxis = axis;
                    bestDir = (diff > 0) ? 1 : -1;
                    break;
                }
            }
        }

        if (bestAxis < 0) break;

        // execute move
        Pos new_head = head;
        new_head.c[bestAxis] += bestDir;
        bool grow = isFood(new_head);

        if (!isSafe(new_head, grow)) break;

        body.push_front(new_head);
        occupied.insert(new_head);
        head = new_head;

        if (grow) {
            food_set.erase(new_head);
        } else {
            Pos tail = body.back();
            body.pop_back();
            occupied.erase(tail);
        }

        output_moves.push_back({bestAxis, bestDir});
    }

    // Print outputs
    for (auto& mv : output_moves) {
        cout << mv.first << " " << mv.second << "\n";
    }

    return 0;
}
```

### Notes
- Works for **any dimension count** (2D–20D+).  
- Uses only standard C++20 headers (compatible with MSVC).  
- A* search is bounded to avoid explosion in high-dimensional grids.  
- Greedy fallback ensures it still plays safely even if A* fails.
