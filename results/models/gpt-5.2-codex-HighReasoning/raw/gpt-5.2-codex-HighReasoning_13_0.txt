```cpp
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <cstdint>

using namespace std;

// Suffix Automaton with sparse transitions (linked list)
class SuffixAutomaton {
public:
    struct State {
        int link;
        int len;
        int head;
        int first_pos;
    };
    struct Edge {
        int to;
        int next;
        unsigned char ch;
    };

    vector<State> st;
    vector<Edge> edges;
    int last;

    SuffixAutomaton(int maxLen = 0) {
        st.reserve(2 * maxLen + 1);
        edges.reserve(4 * maxLen + 1);
        st.push_back({-1, 0, -1, -1}); // root
        last = 0;
    }

    inline int get_trans(int v, unsigned char c) const {
        for (int e = st[v].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) return edges[e].to;
        }
        return -1;
    }

    inline void set_trans(int v, unsigned char c, int to) {
        for (int e = st[v].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) {
                edges[e].to = to;
                return;
            }
        }
        edges.push_back({to, st[v].head, c});
        st[v].head = (int)edges.size() - 1;
    }

    inline void copy_transitions(int src, int dst) {
        for (int e = st[src].head; e != -1; e = edges[e].next) {
            edges.push_back({edges[e].to, st[dst].head, edges[e].ch});
            st[dst].head = (int)edges.size() - 1;
        }
    }

    void extend(unsigned char c, int pos) {
        int cur = (int)st.size();
        st.push_back({0, st[last].len + 1, -1, pos});

        int p = last;
        while (p != -1 && get_trans(p, c) == -1) {
            set_trans(p, c, cur);
            p = st[p].link;
        }
        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = get_trans(p, c);
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = (int)st.size();
                st.push_back({st[q].link, st[p].len + 1, -1, st[q].first_pos});
                copy_transitions(q, clone);
                while (p != -1 && get_trans(p, c) == q) {
                    set_trans(p, c, clone);
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }
};

// Small-case O(n*m) DP for 2 strings
string lcs_dp(const string& a, const string& b) {
    const string *s1 = &a, *s2 = &b;
    if (s1->size() > s2->size()) swap(s1, s2);

    vector<int> dp(s2->size() + 1, 0);
    int best = 0, endPos = 0;

    for (size_t i = 1; i <= s1->size(); ++i) {
        for (size_t j = s2->size(); j > 0; --j) {
            if ((*s1)[i - 1] == (*s2)[j - 1])
                dp[j] = dp[j - 1] + 1;
            else
                dp[j] = 0;

            if (dp[j] > best) {
                best = dp[j];
                endPos = (int)i;
            }
        }
    }
    return s1->substr(endPos - best, best);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    string line;
    getline(cin, line); // consume endline

    vector<string> strs;
    strs.reserve(N);
    for (int i = 0; i < N; ++i) {
        getline(cin, line);
        strs.push_back(line);
    }
    if (N == 0) return 0;

    // If any empty string -> answer is empty
    for (auto &s : strs) {
        if (s.empty()) {
            cout << "\n";
            return 0;
        }
    }

    if (N == 1) {
        cout << strs[0] << "\n";
        return 0;
    }

    // Choose shortest string as base
    int idx = 0;
    for (int i = 1; i < N; ++i) {
        if (strs[i].size() < strs[idx].size()) idx = i;
    }
    if (idx != 0) swap(strs[0], strs[idx]);
    const string& base = strs[0];

    // Adaptive strategy: for small 2-string case use DP
    const long long DP_THRESHOLD = 40000000LL;
    if (N == 2 && (long long)strs[0].size() * (long long)strs[1].size() <= DP_THRESHOLD) {
        cout << lcs_dp(strs[0], strs[1]) << "\n";
        return 0;
    }

    // Build SAM on shortest string
    SuffixAutomaton sam((int)base.size());
    for (int i = 0; i < (int)base.size(); ++i) {
        sam.extend((unsigned char)base[i], i);
    }

    int sz = (int)sam.st.size();

    // Build order by length (counting sort)
    int maxLen = 0;
    for (const auto& st : sam.st) maxLen = max(maxLen, st.len);

    vector<int> cnt(maxLen + 1, 0);
    for (const auto& st : sam.st) cnt[st.len]++;
    for (int i = 1; i <= maxLen; ++i) cnt[i] += cnt[i - 1];

    vector<int> order(sz);
    for (int v = 0; v < sz; ++v) {
        order[--cnt[sam.st[v].len]] = v;
    }

    // Arrays for multiple-string LCS
    vector<int> mn(sz), mx(sz);
    for (int v = 0; v < sz; ++v) mn[v] = sam.st[v].len;

    // Process each other string
    for (int si = 1; si < N; ++si) {
        const string& s = strs[si];
        int v = 0, l = 0;

        for (char ch : s) {
            unsigned char c = (unsigned char)ch;
            int next = sam.get_trans(v, c);

            if (next == -1) {
                while (v != -1 && (next = sam.get_trans(v, c)) == -1) {
                    v = sam.st[v].link;
                    if (v != -1) l = sam.st[v].len;
                }
                if (v == -1) {
                    v = 0; l = 0;
                    continue;
                }
            }
            v = next;
            l++;
            if (l > mx[v]) mx[v] = l;
        }

        // Propagate max matches along suffix links
        for (int i = sz - 1; i > 0; --i) {
            int state = order[i];
            int p = sam.st[state].link;
            if (p != -1) {
                int val = min(mx[state], sam.st[p].len);
                if (val > mx[p]) mx[p] = val;
            }
        }

        // Update global min and reset mx
        for (int v = 0; v < sz; ++v) {
            if (mx[v] < mn[v]) mn[v] = mx[v];
            mx[v] = 0;
        }
    }

    // Find best state
    int bestLen = 0;
    int bestState = 0;
    for (int v = 0; v < sz; ++v) {
        if (mn[v] > bestLen) {
            bestLen = mn[v];
            bestState = v;
        }
    }

    if (bestLen == 0) {
        cout << "\n";
        return 0;
    }

    int endPos = sam.st[bestState].first_pos;
    string ans = base.substr(endPos - bestLen + 1, bestLen);
    cout << ans << "\n";
    return 0;
}
```

**Highlights:**
- **Adaptive logic**: uses O(n·m) DP for tiny 2-string cases, SAM for everything else.
- **Scalable**: SAM built only from shortest string; all others are streamed through it.
- **Correct**: deterministic automaton, no hashing collisions.
- **Handles extreme input sizes** efficiently within memory limits.
