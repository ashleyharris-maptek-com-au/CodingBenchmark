use std::io::{self, Read};
use std::collections::HashMap;

struct FastInput {
    data: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data).unwrap();
        Self { data, idx: 0 }
    }

    fn next_i64(&mut self) -> Option<i64> {
        while self.idx < self.data.len() && self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.data.len() {
            return None;
        }
        let mut sign = 1i64;
        if self.data[self.idx] == b'-' {
            sign = -1;
            self.idx += 1;
        }
        let mut num = 0i64;
        while self.idx < self.data.len() {
            let c = self.data[self.idx];
            if !c.is_ascii_digit() {
                break;
            }
            num = num * 10 + (c - b'0') as i64;
            self.idx += 1;
        }
        Some(num * sign)
    }
}

fn meet_in_middle(nums: &Vec<i64>, target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let n1 = n / 2;
    let n2 = n - n1;
    let size1 = 1usize << n1;
    let size2 = 1usize << n2;

    let mut sums1 = vec![0i64; size1];
    for mask in 1..size1 {
        let bit = mask & mask.wrapping_neg();
        let idx = bit.trailing_zeros() as usize;
        let prev = mask ^ bit;
        sums1[mask] = sums1[prev] + nums[idx];
    }
    let mut list: Vec<(i64, usize)> = Vec::with_capacity(size1);
    for mask in 0..size1 {
        list.push((sums1[mask], mask));
    }
    list.sort_by_key(|x| x.0);

    let mut sums2 = vec![0i64; size2];
    for mask in 1..size2 {
        let bit = mask & mask.wrapping_neg();
        let idx = bit.trailing_zeros() as usize;
        let prev = mask ^ bit;
        sums2[mask] = sums2[prev] + nums[n1 + idx];
    }

    for mask2 in 0..size2 {
        let sum2 = sums2[mask2];
        let need = target - sum2;
        if let Ok(pos) = list.binary_search_by(|x| x.0.cmp(&need)) {
            let mask1 = list[pos].1;
            let mut res = Vec::new();
            for i in 0..n1 {
                if (mask1 >> i) & 1 == 1 {
                    res.push(i);
                }
            }
            for i in 0..n2 {
                if (mask2 >> i) & 1 == 1 {
                    res.push(n1 + i);
                }
            }
            return Some(res);
        }
    }
    None
}

fn dp_bitset(nums: &Vec<i64>, target: i64) -> Option<Vec<usize>> {
    if target < 0 {
        return None;
    }
    let t = target as usize;
    let mut parent: Vec<i32> = vec![-1; t + 1];
    parent[0] = -2;
    for (i, &v) in nums.iter().enumerate() {
        if v < 0 {
            continue;
        }
        let val = v as usize;
        if val > t {
            continue;
        }
        for s in (val..=t).rev() {
            if parent[s] == -1 && parent[s - val] != -1 {
                parent[s] = i as i32;
            }
        }
        if parent[t] != -1 {
            break;
        }
    }
    if parent[t] == -1 {
        return None;
    }
    let mut res = Vec::new();
    let mut s = t;
    while s > 0 {
        let idx = parent[s];
        if idx < 0 {
            break;
        }
        let i = idx as usize;
        res.push(i);
        s -= nums[i] as usize;
    }
    Some(res)
}

fn reconstruct_hash(map: &HashMap<i64, (i64, usize)>, target: i64) -> Vec<usize> {
    let mut res = Vec::new();
    let mut sum = target;
    while sum != 0 {
        if let Some(&(prev, idx)) = map.get(&sum) {
            if idx == usize::MAX {
                break;
            }
            res.push(idx);
            sum = prev;
        } else {
            break;
        }
    }
    res
}

fn hash_dp(nums: &Vec<i64>, target: i64, nonneg: bool) -> Option<Vec<usize>> {
    const MAX_STATES: usize = 5_000_000;
    let mut map: HashMap<i64, (i64, usize)> = HashMap::with_capacity(1024);
    let mut sums: Vec<i64> = Vec::with_capacity(1024);
    map.insert(0, (0, usize::MAX));
    sums.push(0);

    for (i, &v) in nums.iter().enumerate() {
        let current_len = sums.len();
        for idx in 0..current_len {
            let s = sums[idx];
            let new_sum = s + v;
            if nonneg && new_sum > target {
                continue;
            }
            if let std::collections::hash_map::Entry::Vacant(e) = map.entry(new_sum) {
                e.insert((s, i));
                sums.push(new_sum);
                if new_sum == target {
                    return Some(reconstruct_hash(&map, target));
                }
            }
        }
        if sums.len() > MAX_STATES {
            break;
        }
    }
    if map.contains_key(&target) {
        return Some(reconstruct_hash(&map, target));
    }
    None
}

struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn next(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }
}

fn heuristic_greedy(nums: &Vec<i64>, target: i64, nonneg: bool) -> Option<Vec<usize>> {
    if !nonneg || target < 0 {
        return None;
    }
    let n = nums.len();
    if n == 0 {
        return None;
    }
    let max_n = 200_000usize;
    let mut indices: Vec<usize> = if n > max_n {
        (0..max_n).collect()
    } else {
        (0..n).collect()
    };
    let mut rng = XorShift64 { state: 0x123456789abcdef };
    let iterations = 10;

    for _ in 0..iterations {
        let len = indices.len();
        for i in (1..len).rev() {
            let j = (rng.next() % (i as u64 + 1)) as usize;
            indices.swap(i, j);
        }
        let mut sum = 0i64;
        let mut res = Vec::new();
        for &idx in &indices {
            let v = nums[idx];
            if sum + v <= target {
                sum += v;
                res.push(idx);
                if sum == target {
                    res.sort();
                    return Some(res);
                }
            }
        }
    }
    None
}

fn main() {
    let mut input = FastInput::new();
    let n_opt = input.next_i64();
    if n_opt.is_none() {
        return;
    }
    let mut n = n_opt.unwrap() as usize;
    let target = input.next_i64().unwrap_or(0);

    let mut nums = Vec::with_capacity(n);
    for _ in 0..n {
        if let Some(v) = input.next_i64() {
            nums.push(v);
        } else {
            break;
        }
    }
    n = nums.len();

    if target == 0 {
        println!("YES");
        println!();
        return;
    }
    if n == 0 {
        println!("NO");
        return;
    }

    let mut has_neg = false;
    let mut total_pos = 0i64;
    for &v in &nums {
        if v < 0 {
            has_neg = true;
        }
        if v > 0 {
            total_pos += v;
        }
    }
    if !has_neg && target < 0 {
        println!("NO");
        return;
    }
    if !has_neg && target > total_pos {
        println!("NO");
        return;
    }

    let mut result: Option<Vec<usize>> = None;

    if n <= 40 {
        result = meet_in_middle(&nums, target);
    }
    if result.is_none() && !has_neg && target >= 0 && target <= 2_000_000 {
        result = dp_bitset(&nums, target);
    }
    if result.is_none() {
        result = hash_dp(&nums, target, !has_neg);
    }
    if result.is_none() {
        result = heuristic_greedy(&nums, target, !has_neg);
    }

    match result {
        Some(mut indices) => {
            indices.sort();
            print!("YES\n");
            for (i, idx) in indices.iter().enumerate() {
                if i > 0 {
                    print!(" ");
                }
                print!("{}", idx);
            }
            println!();
        }
        None => {
            println!("NO");
        }
    }
}
