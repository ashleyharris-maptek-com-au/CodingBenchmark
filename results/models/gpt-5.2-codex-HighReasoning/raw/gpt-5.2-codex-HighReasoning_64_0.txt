// Plan grid paths for search drones - Node.js TypeScript program
declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

type Drone = { id: any; x: number; y: number };

const grid: number[][] = data.probability;
const drones: Drone[] = data.drones;
const maxSteps: number = data.maxStepsPerDrone;

const rows = grid.length;
const cols = rows > 0 ? grid[0].length : 0;

const visited = new Set<number>();

function inBounds(x: number, y: number): boolean {
  return y >= 0 && y < rows && x >= 0 && x < cols;
}

function key(x: number, y: number): number {
  return y * cols + x;
}

const paths: number[][][] = [];
const positions: { x: number; y: number }[] = [];

for (let i = 0; i < drones.length; i++) {
  const d = drones[i];
  const x = d.x;
  const y = d.y;
  positions.push({ x, y });
  paths.push([[x, y]]);
  if (inBounds(x, y) && grid[y][x] > 0) {
    visited.add(key(x, y));
  }
}

const moves = [
  { dx: 0, dy: 0 },  // stay
  { dx: 1, dy: 0 },
  { dx: -1, dy: 0 },
  { dx: 0, dy: 1 },
  { dx: 0, dy: -1 },
];

for (let step = 0; step < maxSteps; step++) {
  for (let i = 0; i < drones.length; i++) {
    const pos = positions[i];
    let bestX = pos.x;
    let bestY = pos.y;
    let bestPrimary = -1;
    let bestSecondary = -1;

    for (const m of moves) {
      const nx = pos.x + m.dx;
      const ny = pos.y + m.dy;
      if (!inBounds(nx, ny)) continue;
      if (grid[ny][nx] === -1) continue;

      const k = key(nx, ny);
      const prob = grid[ny][nx] > 0 ? grid[ny][nx] : 0;
      const unvisited = !visited.has(k);
      const primary = unvisited ? prob : 0;
      const secondary = prob;

      if (
        primary > bestPrimary ||
        (primary === bestPrimary && secondary > bestSecondary)
      ) {
        bestPrimary = primary;
        bestSecondary = secondary;
        bestX = nx;
        bestY = ny;
      }
    }

    positions[i] = { x: bestX, y: bestY };
    paths[i].push([bestX, bestY]);
    if (inBounds(bestX, bestY) && grid[bestY][bestX] > 0) {
      visited.add(key(bestX, bestY));
    }
  }
}

const output = { paths };
process.stdout.write(JSON.stringify(output));
