Below is a complete, compilable Rust program (edition 2024) that reads drillhole assay data, computes spatial outliers, and outputs suspect entries sorted by confidence.  
It adapts strategy based on dataset size:

- **Small datasets** → brute force neighbors  
- **Medium datasets** → spatial grid hashing  
- **Large datasets** → along‑hole neighbors only  

The confidence is the **z‑score** against local neighbors (with global fallback).

```rust
use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read};

#[derive(Clone)]
struct Sample {
    hole_idx: usize,
    sample_idx: usize,
    pos: [f64; 3],
    val_offset: usize,
}

#[derive(Clone)]
struct Outlier {
    conf: f64,
    hole_idx: usize,
    sample_idx: usize,
    prop_idx: usize,
}

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).unwrap();
        Self { buf: input, idx: 0 }
    }

    fn next_token(&mut self) -> Option<&str> {
        while self.idx < self.buf.len()
            && self.buf[self.idx].is_ascii_whitespace()
        {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let start = self.idx;
        while self.idx < self.buf.len()
            && !self.buf[self.idx].is_ascii_whitespace()
        {
            self.idx += 1;
        }
        std::str::from_utf8(&self.buf[start..self.idx]).ok()
    }

    fn next_usize(&mut self) -> Option<usize> {
        self.next_token()?.parse().ok()
    }

    fn next_f64(&mut self) -> Option<f64> {
        self.next_token()?.parse().ok()
    }

    fn next_string(&mut self) -> Option<String> {
        self.next_token().map(|s| s.to_string())
    }
}

fn main() {
    let mut sc = Scanner::new();

    let num_holes = match sc.next_usize() {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let num_props = match sc.next_usize() {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut prop_names = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        if let Some(name) = sc.next_string() {
            prop_names.push(name);
        } else {
            prop_names.push(format!("P{}", prop_names.len()));
        }
    }

    let mut hole_ids: Vec<String> = Vec::with_capacity(num_holes);
    let mut hole_samples: Vec<Vec<usize>> = Vec::with_capacity(num_holes);
    let mut samples: Vec<Sample> = Vec::new();
    let mut values: Vec<f64> = Vec::new();

    let mut min_xyz = [f64::INFINITY; 3];
    let mut max_xyz = [f64::NEG_INFINITY; 3];

    let mut global_sum = vec![0.0f64; num_props];
    let mut global_sum_sq = vec![0.0f64; num_props];
    let mut global_count = vec![0u64; num_props];

    for _ in 0..num_holes {
        let hole_id = match sc.next_string() {
            Some(v) => v,
            None => break,
        };
        let sx = sc.next_f64().unwrap_or(0.0);
        let sy = sc.next_f64().unwrap_or(0.0);
        let sz = sc.next_f64().unwrap_or(0.0);
        let dx = sc.next_f64().unwrap_or(0.0);
        let dy = sc.next_f64().unwrap_or(0.0);
        let dz = sc.next_f64().unwrap_or(0.0);
        let _length = sc.next_f64().unwrap_or(0.0);
        let ns = sc.next_usize().unwrap_or(0);

        let mut dir = [dx, dy, dz];
        let norm = (dx * dx + dy * dy + dz * dz).sqrt();
        if norm > 1e-12 {
            dir[0] /= norm;
            dir[1] /= norm;
            dir[2] /= norm;
        } else {
            dir = [0.0, 0.0, 1.0];
        }

        let hole_idx = hole_ids.len();
        hole_ids.push(hole_id);
        hole_samples.push(Vec::with_capacity(ns));

        for sidx in 0..ns {
            let depth = match sc.next_f64() {
                Some(v) => v,
                None => break,
            };

            let pos = [
                sx + dir[0] * depth,
                sy + dir[1] * depth,
                sz + dir[2] * depth,
            ];

            for d in 0..3 {
                if pos[d] < min_xyz[d] { min_xyz[d] = pos[d]; }
                if pos[d] > max_xyz[d] { max_xyz[d] = pos[d]; }
            }

            let val_offset = values.len();
            for p in 0..num_props {
                let v = sc.next_f64().unwrap_or(f64::NAN);
                values.push(v);
                if v.is_finite() {
                    global_sum[p] += v;
                    global_sum_sq[p] += v * v;
                    global_count[p] += 1;
                }
            }

            let sample_idx = samples.len();
            samples.push(Sample {
                hole_idx,
                sample_idx: sidx,
                pos,
                val_offset,
            });
            hole_samples[hole_idx].push(sample_idx);
        }
    }

    let total_samples = samples.len();
    if total_samples == 0 {
        println!("0");
        return;
    }

    // Global mean/std for fallback
    let mut global_mean = vec![0.0; num_props];
    let mut global_std = vec![1.0; num_props];
    for p in 0..num_props {
        if global_count[p] > 0 {
            let c = global_count[p] as f64;
            let mean = global_sum[p] / c;
            let var = (global_sum_sq[p] / c) - mean * mean;
            global_mean[p] = mean;
            global_std[p] = var.max(1e-12).sqrt();
        }
    }

    // Decide strategy
    let strategy = if total_samples <= 5000 {
        0 // brute
    } else if total_samples <= 1_000_000 {
        1 // grid
    } else {
        2 // along-hole
    };

    let max_neighbors = if total_samples < 10_000 { 64 }
        else if total_samples < 100_000 { 48 }
        else if total_samples < 1_000_000 { 32 }
        else { 16 };

    let threshold = if total_samples < 10_000 { 2.5 }
        else if total_samples < 1_000_000 { 3.0 }
        else { 3.5 };

    // Build grid if needed
    let mut grid: HashMap<(i32, i32, i32), Vec<usize>> = HashMap::new();
    let mut cell_size = 1.0;
    let mut inv_cell = 1.0;
    let mut radius2 = 1.0;

    if strategy == 1 {
        let dx = (max_xyz[0] - min_xyz[0]).max(1.0);
        let dy = (max_xyz[1] - min_xyz[1]).max(1.0);
        let dz = (max_xyz[2] - min_xyz[2]).max(1.0);
        let volume = dx * dy * dz;
        let spacing = (volume / total_samples as f64).cbrt().max(1e-6);
        cell_size = spacing * 2.0;
        inv_cell = 1.0 / cell_size;
        radius2 = (cell_size * 1.5).powi(2);

        for (i, s) in samples.iter().enumerate() {
            let cx = ((s.pos[0] - min_xyz[0]) * inv_cell).floor() as i32;
            let cy = ((s.pos[1] - min_xyz[1]) * inv_cell).floor() as i32;
            let cz = ((s.pos[2] - min_xyz[2]) * inv_cell).floor() as i32;
            grid.entry((cx, cy, cz)).or_insert_with(Vec::new).push(i);
        }
    }

    let mut outliers: Vec<Outlier> = Vec::new();

    let mut sum = vec![0.0f64; num_props];
    let mut sum_sq = vec![0.0f64; num_props];
    let mut count = vec![0u32; num_props];

    for (idx, s) in samples.iter().enumerate() {
        let mut neighbors: Vec<(f64, usize)> = Vec::new();

        match strategy {
            0 => {
                for (j, other) in samples.iter().enumerate() {
                    if j == idx { continue; }
                    let dx = s.pos[0] - other.pos[0];
                    let dy = s.pos[1] - other.pos[1];
                    let dz = s.pos[2] - other.pos[2];
                    let dist2 = dx*dx + dy*dy + dz*dz;
                    neighbors.push((dist2, j));
                }
                neighbors.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(Ordering::Equal));
                neighbors.truncate(max_neighbors);
            }
            1 => {
                let cx = ((s.pos[0] - min_xyz[0]) * inv_cell).floor() as i32;
                let cy = ((s.pos[1] - min_xyz[1]) * inv_cell).floor() as i32;
                let cz = ((s.pos[2] - min_xyz[2]) * inv_cell).floor() as i32;
                for dx in -1..=1 {
                    for dy in -1..=1 {
                        for dz in -1..=1 {
                            if let Some(list) = grid.get(&(cx + dx, cy + dy, cz + dz)) {
                                for &j in list {
                                    if j == idx { continue; }
                                    let o = &samples[j];
                                    let dx = s.pos[0] - o.pos[0];
                                    let dy = s.pos[1] - o.pos[1];
                                    let dz = s.pos[2] - o.pos[2];
                                    let dist2 = dx*dx + dy*dy + dz*dz;
                                    if dist2 <= radius2 {
                                        neighbors.push((dist2, j));
                                    }
                                }
                            }
                        }
                    }
                }
                if neighbors.len() > max_neighbors {
                    neighbors.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(Ordering::Equal));
                    neighbors.truncate(max_neighbors);
                }
            }
            _ => {
                // along-hole neighbors only
                let hlist = &hole_samples[s.hole_idx];
                let win = 2usize;
                let start = s.sample_idx.saturating_sub(win);
                let end = (s.sample_idx + win + 1).min(hlist.len());
                for j in start..end {
                    let sid = hlist[j];
                    if sid != idx {
                        neighbors.push((0.0, sid));
                    }
                }
            }
        }

        if neighbors.len() < 3 {
            // add along-hole neighbors for fallback
            let hlist = &hole_samples[s.hole_idx];
            let win = 2usize;
            let start = s.sample_idx.saturating_sub(win);
            let end = (s.sample_idx + win + 1).min(hlist.len());
            for j in start..end {
                let sid = hlist[j];
                if sid != idx && !neighbors.iter().any(|&(_, id)| id == sid) {
                    neighbors.push((0.0, sid));
                }
            }
        }

        sum.fill(0.0);
        sum_sq.fill(0.0);
        count.fill(0);

        for &(_, nid) in neighbors.iter() {
            let base = samples[nid].val_offset;
            for p in 0..num_props {
                let v = values[base + p];
                if v.is_finite() {
                    sum[p] += v;
                    sum_sq[p] += v * v;
                    count[p] += 1;
                }
            }
        }

        let base = s.val_offset;
        for p in 0..num_props {
            let v = values[base + p];
            if !v.is_finite() { continue; }

            let (mean, std) = if count[p] >= 3 {
                let c = count[p] as f64;
                let m = sum[p] / c;
                let var = (sum_sq[p] / c) - m * m;
                (m, var.max(1e-12).sqrt())
            } else {
                (global_mean[p], global_std[p])
            };

            let z = (v - mean).abs() / (std + 1e-12);
            if z > threshold {
                outliers.push(Outlier {
                    conf: z,
                    hole_idx: s.hole_idx,
                    sample_idx: s.sample_idx,
                    prop_idx: p,
                });
            }
        }
    }

    outliers.sort_by(|a, b| b.conf.partial_cmp(&a.conf).unwrap_or(Ordering::Equal));

    println!("{}", outliers.len());
    for o in outliers {
        println!(
            "{} {} {} {:.3}",
            hole_ids[o.hole_idx],
            o.sample_idx,
            prop_names[o.prop_idx],
            o.conf
        );
    }
}
```

### Notes
- Handles missing or malformed values as `NaN`.
- Uses global stats as fallback if too few neighbors.
- Scales strategy to dataset size for performance.
- Works entirely with the Rust standard library.

You can compile with:

```bash
rustc main.rs -O
```

This will run efficiently across small and large datasets while reporting the most obvious outliers.
