```typescript
declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8').trim();
if (input.length === 0) {
  process.exit(0);
}
const data = JSON.parse(input);

interface Task {
  id: any;
  duration: number;
  deps: any[];
  deadline: number;
  weight: number;
  index: number;
}

class BinaryHeap<T> {
  private data: T[] = [];
  constructor(private higher: (a: T, b: T) => boolean) {}
  size(): number { return this.data.length; }
  peek(): T | undefined { return this.data[0]; }
  push(item: T): void {
    const d = this.data;
    d.push(item);
    let i = d.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.higher(d[i], d[p])) {
        [d[i], d[p]] = [d[p], d[i]];
        i = p;
      } else break;
    }
  }
  pop(): T | undefined {
    const d = this.data;
    if (d.length === 0) return undefined;
    const top = d[0];
    const last = d.pop()!;
    if (d.length > 0) {
      d[0] = last;
      let i = 0;
      while (true) {
        let l = i * 2 + 1;
        let r = l + 1;
        let best = i;
        if (l < d.length && this.higher(d[l], d[best])) best = l;
        if (r < d.length && this.higher(d[r], d[best])) best = r;
        if (best !== i) {
          [d[i], d[best]] = [d[best], d[i]];
          i = best;
        } else break;
      }
    }
    return top;
  }
}

const machineCount: number = data.machineCount ?? 1;
const tasksData: any[] = data.tasks ?? [];
const n = tasksData.length;

const tasks: Task[] = new Array(n);
const idToIndex = new Map<any, number>();

for (let i = 0; i < n; i++) {
  const t = tasksData[i] || {};
  const task: Task = {
    id: t.id,
    duration: Number(t.duration ?? 0),
    deps: Array.isArray(t.deps) ? t.deps : [],
    deadline: Number(t.deadline ?? Number.MAX_SAFE_INTEGER),
    weight: Number(t.weight ?? 1),
    index: i
  };
  tasks[i] = task;
  idToIndex.set(task.id, i);
}

const dependents: number[][] = new Array(n);
const indegree: number[] = new Array(n).fill(0);
for (let i = 0; i < n; i++) dependents[i] = [];

for (let i = 0; i < n; i++) {
  const task = tasks[i];
  for (const depId of task.deps) {
    const j = idToIndex.get(depId);
    if (j !== undefined) {
      dependents[j].push(i);
      indegree[i]++;
    }
  }
}

// Priority: higher weight, earlier deadline, shorter duration
const readyHeap = new BinaryHeap<Task>((a, b) => {
  if (a.weight !== b.weight) return a.weight > b.weight;
  if (a.deadline !== b.deadline) return a.deadline < b.deadline;
  if (a.duration !== b.duration) return a.duration < b.duration;
  return a.index < b.index;
});

const futureHeap = new BinaryHeap<{task: Task; release: number}>((a, b) => a.release < b.release);
const completionHeap = new BinaryHeap<{time: number; task: Task}>((a, b) => a.time < b.time);
const machineHeap = new BinaryHeap<{time: number; machine: number}>((a, b) => a.time < b.time);

for (let m = 0; m < machineCount; m++) {
  machineHeap.push({time: 0, machine: m});
}

for (let i = 0; i < n; i++) {
  if (indegree[i] === 0) readyHeap.push(tasks[i]);
}

const schedule: {id: any; start: number; machine: number}[] = [];
let scheduled = 0;

while (scheduled < n) {
  const mach = machineHeap.pop();
  if (!mach) break;
  let current = mach.time;

  // Process completions up to current time
  while (completionHeap.size() > 0 && completionHeap.peek()!.time <= current) {
    const event = completionHeap.pop()!;
    const t = event.task;
    for (const depIdx of dependents[t.index]) {
      indegree[depIdx]--;
      if (indegree[depIdx] === 0) {
        const release = event.time;
        if (release <= current) readyHeap.push(tasks[depIdx]);
        else futureHeap.push({task: tasks[depIdx], release});
      }
    }
  }

  // Move future tasks that are now ready
  while (futureHeap.size() > 0 && futureHeap.peek()!.release <= current) {
    readyHeap.push(futureHeap.pop()!.task);
  }

  if (readyHeap.size() === 0) {
    let next = Infinity;
    if (completionHeap.size() > 0) next = Math.min(next, completionHeap.peek()!.time);
    if (futureHeap.size() > 0) next = Math.min(next, futureHeap.peek()!.release);
    if (next === Infinity) {
      break; // likely a cycle, stop
    }
    mach.time = Math.max(mach.time, next);
    machineHeap.push(mach);
    continue;
  }

  const task = readyHeap.pop()!;
  const start = current;
  const finish = start + task.duration;

  schedule.push({id: task.id, start: Math.trunc(start), machine: mach.machine});
  scheduled++;

  completionHeap.push({time: finish, task});
  mach.time = finish;
  machineHeap.push(mach);
}

console.log(JSON.stringify({schedule}));
```
