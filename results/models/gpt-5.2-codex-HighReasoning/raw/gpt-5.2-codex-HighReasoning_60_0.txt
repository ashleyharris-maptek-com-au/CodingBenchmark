const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

class Heap {
  constructor(compare) {
    this.data = [];
    this.compare = compare;
  }
  size() { return this.data.length; }
  peek() { return this.data[0]; }
  push(item) {
    const a = this.data;
    a.push(item);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.compare(a[i], a[p])) {
        [a[i], a[p]] = [a[p], a[i]];
        i = p;
      } else break;
    }
  }
  pop() {
    const a = this.data;
    if (a.length === 0) return undefined;
    const res = a[0];
    const last = a.pop();
    if (a.length > 0) {
      a[0] = last;
      let i = 0;
      const n = a.length;
      while (true) {
        let l = i * 2 + 1;
        let r = l + 1;
        let smallest = i;
        if (l < n && this.compare(a[l], a[smallest])) smallest = l;
        if (r < n && this.compare(a[r], a[smallest])) smallest = r;
        if (smallest !== i) {
          [a[i], a[smallest]] = [a[smallest], a[i]];
          i = smallest;
        } else break;
      }
    }
    return res;
  }
}

const tasks = data.tasks;
const n = tasks.length;
const idToIndex = new Map();
for (let i = 0; i < n; i++) idToIndex.set(tasks[i].id, i);

const indeg = new Array(n).fill(0);
const succ = Array.from({ length: n }, () => []);
for (let i = 0; i < n; i++) {
  const deps = tasks[i].deps || [];
  indeg[i] = deps.length;
  for (const d of deps) {
    const j = idToIndex.get(d);
    succ[j].push(i);
  }
}

const maxDepFinish = new Array(n).fill(0);
const finishTime = new Array(n).fill(0);

const futureReady = new Heap((a, b) => a.readyTime < b.readyTime);
const available = new Heap((a, b) => {
  if (a.deadline !== b.deadline) return a.deadline < b.deadline;
  if (a.weight !== b.weight) return a.weight > b.weight;
  if (a.duration !== b.duration) return a.duration < b.duration;
  return a.id < b.id;
});

for (let i = 0; i < n; i++) {
  if (indeg[i] === 0) {
    futureReady.push({ idx: i, readyTime: 0 });
  }
}

const machineCount = data.machineCount;
const machines = new Heap((a, b) => {
  if (a.time !== b.time) return a.time < b.time;
  return a.index < b.index;
});
for (let i = 0; i < machineCount; i++) {
  machines.push({ time: 0, index: i });
}

const schedule = [];

function moveReady(currentTime) {
  while (futureReady.size() > 0 && futureReady.peek().readyTime <= currentTime) {
    const r = futureReady.pop();
    const t = tasks[r.idx];
    available.push({
      idx: r.idx,
      deadline: t.deadline,
      weight: t.weight,
      duration: t.duration,
      id: t.id
    });
  }
}

let scheduled = 0;
while (scheduled < n) {
  let machine = machines.pop();
  let time = machine.time;
  moveReady(time);
  if (available.size() === 0) {
    if (futureReady.size() === 0) break;
    const nextReady = futureReady.peek().readyTime;
    time = Math.max(time, nextReady);
    moveReady(time);
  }
  const taskNode = available.pop();
  const idx = taskNode.idx;
  const task = tasks[idx];
  const start = time;
  const finish = start + task.duration;
  machine.time = finish;
  machines.push(machine);

  schedule.push({ id: task.id, start: start, machine: machine.index });
  finishTime[idx] = finish;

  for (const s of succ[idx]) {
    if (finish > maxDepFinish[s]) maxDepFinish[s] = finish;
    indeg[s]--;
    if (indeg[s] === 0) {
      futureReady.push({ idx: s, readyTime: maxDepFinish[s] });
    }
  }
  scheduled++;
}

const output = { schedule };
process.stdout.write(JSON.stringify(output));
