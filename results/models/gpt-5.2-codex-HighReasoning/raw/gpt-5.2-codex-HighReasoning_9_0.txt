```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public class Program
{
    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        if (n == int.MinValue) return;
        int stockLength = fs.NextInt();

        int[] lengths = new int[n];
        int[] indices = new int[n];
        for (int i = 0; i < n; i++)
        {
            lengths[i] = fs.NextInt();
            indices[i] = i;
        }

        // Sort cuts by length descending (First-Fit Decreasing)
        Array.Sort(lengths, indices, new DescComparer());

        var bins = new List<List<int>>();
        var rems = new List<int>();

        var remSet = new SortedSet<int>();
        var remToBins = new Dictionary<int, Stack<int>>();

        for (int i = 0; i < n; i++)
        {
            int cutLen = lengths[i];
            int idx = indices[i];

            int rem = FindCeiling(remSet, cutLen);
            if (rem >= 0)
            {
                var stack = remToBins[rem];
                int bin = stack.Pop();
                if (stack.Count == 0)
                {
                    remToBins.Remove(rem);
                    remSet.Remove(rem);
                }

                bins[bin].Add(idx);
                int newRem = rem - cutLen;
                rems[bin] = newRem;

                if (!remToBins.TryGetValue(newRem, out var stack2))
                {
                    stack2 = new Stack<int>();
                    remToBins[newRem] = stack2;
                    remSet.Add(newRem);
                }
                stack2.Push(bin);
            }
            else
            {
                int bin = bins.Count;
                var list = new List<int>(4);
                list.Add(idx);
                bins.Add(list);

                int newRem = stockLength - cutLen;
                rems.Add(newRem);

                if (!remToBins.TryGetValue(newRem, out var stack2))
                {
                    stack2 = new Stack<int>();
                    remToBins[newRem] = stack2;
                    remSet.Add(newRem);
                }
                stack2.Push(bin);
            }
        }

        var sw = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 16);
        sw.WriteLine(bins.Count);
        var sb = new StringBuilder();

        for (int b = 0; b < bins.Count; b++)
        {
            sb.Clear();
            var list = bins[b];
            for (int j = 0; j < list.Count; j++)
            {
                if (j > 0) sb.Append(' ');
                sb.Append(list[j]);
            }
            sw.WriteLine(sb.ToString());
        }
        sw.Flush();
    }

    private static int FindCeiling(SortedSet<int> set, int value)
    {
        if (set.Count == 0) return -1;
        var view = set.GetViewBetween(value, int.MaxValue);
        if (view.Count == 0) return -1;
        return view.Min;
    }

    private class DescComparer : IComparer<int>
    {
        public int Compare(int a, int b) => b.CompareTo(a);
    }

    private class FastScanner
    {
        private readonly Stream _s = Console.OpenStandardInput();
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len = 0, _ptr = 0;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32 && c != -1)
            {
                val = val * 10 + (c - 48);
                c = Read();
            }
            return val * sign;
        }
    }
}
```
