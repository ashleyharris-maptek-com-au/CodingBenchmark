Below is a complete, compilable C# solution that reads the QAP instance, builds a good initial assignment by sorting total flow vs. total distance, and then improves it using a fast local search (swap-based hill climbing).  
For small N (≤150) it performs a thorough improvement; for larger N it uses randomized swaps.

```csharp
using System;
using System.IO;
using System.Text;

public class Program
{
    // Fast input reader
    class FastScanner
    {
        private readonly Stream _input = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len = 0, _ptr = 0;

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _input.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return int.MinValue;
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return val * sign;
        }
    }

    static long ComputeCost(int[][] flow, int[][] dist, int[] perm)
    {
        int n = perm.Length;
        long cost = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = perm[i];
            for (int j = 0; j < n; j++)
            {
                cost += (long)flow[i][j] * dist[pi][perm[j]];
            }
        }
        return cost;
    }

    static long DeltaSwap(int a, int b, int[][] flow, int[][] dist, int[] perm)
    {
        int n = perm.Length;
        int pa = perm[a];
        int pb = perm[b];
        long delta = 0;

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;
            int pk = perm[k];

            delta += (long)flow[a][k] * (dist[pb][pk] - dist[pa][pk]);
            delta += (long)flow[b][k] * (dist[pa][pk] - dist[pb][pk]);
            delta += (long)flow[k][a] * (dist[pk][pb] - dist[pk][pa]);
            delta += (long)flow[k][b] * (dist[pk][pa] - dist[pk][pb]);
        }

        // terms where i,j are in {a,b}
        delta += (long)flow[a][a] * (dist[pb][pb] - dist[pa][pa]);
        delta += (long)flow[a][b] * (dist[pb][pa] - dist[pa][pb]);
        delta += (long)flow[b][a] * (dist[pa][pb] - dist[pb][pa]);
        delta += (long)flow[b][b] * (dist[pa][pa] - dist[pb][pb]);

        return delta;
    }

    public static void Main()
    {
        FastScanner fs = new FastScanner();
        int N = fs.NextInt();
        if (N == int.MinValue) return;

        int[][] flow = new int[N][];
        int[][] dist = new int[N][];

        for (int i = 0; i < N; i++)
        {
            flow[i] = new int[N];
            for (int j = 0; j < N; j++) flow[i][j] = fs.NextInt();
        }

        for (int i = 0; i < N; i++)
        {
            dist[i] = new int[N];
            for (int j = 0; j < N; j++) dist[i][j] = fs.NextInt();
        }

        // Compute total flow and distance scores for heuristic ordering
        long[] flowSum = new long[N];
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                int v = flow[i][j];
                flowSum[i] += v;
                flowSum[j] += v;
            }
        }

        long[] distSum = new long[N];
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                int v = dist[i][j];
                distSum[i] += v;
                distSum[j] += v;
            }
        }

        int[] facIdx = new int[N];
        int[] locIdx = new int[N];
        for (int i = 0; i < N; i++)
        {
            facIdx[i] = i;
            locIdx[i] = i;
        }

        Array.Sort(facIdx, (a, b) => flowSum[b].CompareTo(flowSum[a])); // descending
        Array.Sort(locIdx, (a, b) => distSum[a].CompareTo(distSum[b])); // ascending

        int[] perm = new int[N];
        for (int i = 0; i < N; i++)
            perm[facIdx[i]] = locIdx[i];

        long cost = ComputeCost(flow, dist, perm);

        // Local search improvement
        if (N <= 150)
        {
            bool improved = true;
            while (improved)
            {
                improved = false;
                for (int i = 0; i < N; i++)
                {
                    for (int j = i + 1; j < N; j++)
                    {
                        long delta = DeltaSwap(i, j, flow, dist, perm);
                        if (delta < 0)
                        {
                            int tmp = perm[i];
                            perm[i] = perm[j];
                            perm[j] = tmp;
                            cost += delta;
                            improved = true;
                        }
                    }
                }
            }
        }
        else
        {
            Random rng = new Random(1);
            int maxIter = Math.Max(10000, N * 30);
            for (int iter = 0; iter < maxIter; iter++)
            {
                int a = rng.Next(N);
                int b = rng.Next(N - 1);
                if (b >= a) b++;
                long delta = DeltaSwap(a, b, flow, dist, perm);
                if (delta < 0)
                {
                    int tmp = perm[a];
                    perm[a] = perm[b];
                    perm[b] = tmp;
                    cost += delta;
                }
            }
        }

        // Output
        StringBuilder sb = new StringBuilder();
        sb.Append(cost).Append('\n');
        for (int i = 0; i < N; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(perm[i]);
        }
        Console.Write(sb.ToString());
    }
}
```

**Notes:**
- Uses only standard libraries.
- Outputs a valid permutation of 0..N-1.
- For large N, runs fast and produces a reasonable heuristic solution.
- For small N, performs stronger improvement.
