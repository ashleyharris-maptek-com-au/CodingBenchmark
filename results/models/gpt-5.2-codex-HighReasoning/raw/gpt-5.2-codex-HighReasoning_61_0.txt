```typescript
// Delivery Route Planner - TypeScript

declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Package = { id: string | number; x: number; y: number };

const depot: [number, number] = data.depot;
const truckCountRaw: number = data.truckCount;
const packages: Package[] = data.packages || [];

const n = packages.length;
if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

let k = Math.min(truckCountRaw || 0, n);
if (k <= 0) k = 1;

// Compute angle for sweep ordering
const sorted = packages
  .map(p => ({
    id: p.id,
    x: p.x,
    y: p.y,
    angle: Math.atan2(p.y - depot[1], p.x - depot[0]),
    dist: (p.x - depot[0]) * (p.x - depot[0]) + (p.y - depot[1]) * (p.y - depot[1])
  }))
  .sort((a, b) => {
    if (a.angle === b.angle) return a.dist - b.dist;
    return a.angle - b.angle;
  });

const routes: (Array<string | number>)[] = [];
const baseSize = Math.floor(n / k);
const remainder = n % k;

let idx = 0;
for (let i = 0; i < k; i++) {
  const size = baseSize + (i < remainder ? 1 : 0);
  const routeIds: (string | number)[] = [];
  for (let j = 0; j < size; j++) {
    routeIds.push(sorted[idx++].id);
  }
  routes.push(routeIds);
}

process.stdout.write(JSON.stringify({ routes }));
```
