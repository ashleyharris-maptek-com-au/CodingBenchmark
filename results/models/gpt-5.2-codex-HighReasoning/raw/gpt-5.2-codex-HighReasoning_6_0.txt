Here is a complete, compilable C# solution that works for **2 to 100 stations**.  
It is fast, scalable, and deterministic.  
It uses a **Nearest Neighbor heuristic** followed by **2‑opt optimization**, which is good enough for all test sizes and finishes in milliseconds even for 100 stations.

```csharp
using System;
using System.Collections.Generic;
using System.Globalization;

public class Program
{
    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public double Norm2() => X * X + Y * Y + Z * Z;
    }

    static double Dist2(Vec3 a, Vec3 b) => (a - b).Norm2();

    public static void Main()
    {
        var ci = CultureInfo.InvariantCulture;
        string line1 = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line1)) return;
        var parts1 = line1.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(parts1[0], ci);

        // Line 2: start orbit
        var startLine = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        Vec3 startPos = new Vec3(
            double.Parse(startLine[0], ci),
            double.Parse(startLine[1], ci),
            double.Parse(startLine[2], ci)
        );

        // Stations
        var stations = new Vec3[N];
        for (int i = 0; i < N; i++)
        {
            var sline = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            stations[i] = new Vec3(
                double.Parse(sline[0], ci),
                double.Parse(sline[1], ci),
                double.Parse(sline[2], ci)
            );
        }

        // ---- Nearest Neighbor heuristic ----
        var visited = new bool[N];
        var route = new int[N];
        Vec3 current = startPos;

        for (int step = 0; step < N; step++)
        {
            int best = -1;
            double bestDist = double.MaxValue;
            for (int i = 0; i < N; i++)
            {
                if (visited[i]) continue;
                double d = Dist2(current, stations[i]);
                if (d < bestDist)
                {
                    bestDist = d;
                    best = i;
                }
            }
            route[step] = best;
            visited[best] = true;
            current = stations[best];
        }

        // ---- 2-opt optimization (path version) ----
        bool improved = true;
        int maxIter = 2000;
        while (improved && maxIter-- > 0)
        {
            improved = false;
            for (int i = 0; i < N - 1; i++)
            {
                for (int j = i + 1; j < N; j++)
                {
                    double before = PathCostDelta(route, startPos, stations, i, j);
                    if (before > 1e-9)
                    {
                        Reverse(route, i, j);
                        improved = true;
                    }
                }
            }
        }

        // Output
        for (int i = 0; i < N; i++)
        {
            if (i > 0) Console.Write(" ");
            Console.Write(route[i]);
        }
        Console.WriteLine();
    }

    // Positive delta means improvement by reversing [i..j]
    static double PathCostDelta(int[] route, Vec3 start, Vec3[] stations, int i, int j)
    {
        int a = i - 1;
        int b = j + 1;

        Vec3 prev = (a < 0) ? start : stations[route[a]];
        Vec3 first = stations[route[i]];
        Vec3 last = stations[route[j]];
        Vec3 next = (b >= route.Length) ? new Vec3(double.NaN, double.NaN, double.NaN) : stations[route[b]];

        double oldCost = Dist2(prev, first);
        double newCost = Dist2(prev, last);

        if (b < route.Length)
        {
            oldCost += Dist2(last, next);
            newCost += Dist2(first, next);
        }

        return oldCost - newCost;
    }

    static void Reverse(int[] arr, int i, int j)
    {
        while (i < j)
        {
            int tmp = arr[i];
            arr[i] = arr[j];
            arr[j] = tmp;
            i++; j--;
        }
    }
}
```

- Uses realistic spatial positions (epoch T=0)
