OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %ubo_struct Block
OpMemberDecorate %ubo_struct 0 Offset 0
OpMemberDecorate %ubo_struct 0 MatrixStride 16
OpMemberDecorate %ubo_struct 0 ColMajor
OpMemberDecorate %ubo_struct 1 Offset 64
OpMemberDecorate %ubo_struct 1 MatrixStride 16
OpMemberDecorate %ubo_struct 1 ColMajor
OpMemberDecorate %ubo_struct 2 Offset 128
OpMemberDecorate %ubo_struct 2 MatrixStride 16
OpMemberDecorate %ubo_struct 2 ColMajor
OpMemberDecorate %ubo_struct 3 Offset 192
OpMemberDecorate %ubo_struct 4 Offset 208
OpMemberDecorate %ubo_struct 5 Offset 224

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%fn = OpTypeFunction %void

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_thick = OpConstant %float 0.02
%float_1_3 = OpConstant %float 0.33333334
%float_2_3 = OpConstant %float 0.66666667

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_uniform_ubo = OpTypePointer Uniform %ubo_struct

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output
%ubo = OpVariable %ptr_uniform_ubo Uniform

%main = OpFunction %void None %fn
%entry = OpLabel
%outPosPtr = OpAccessChain %ptr_out_v4 %gl_out %int_0

%p0ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0ptr
%p1ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1ptr
%p2ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2ptr
%p3ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3ptr

%p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p2xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p3xy = OpVectorShuffle %v2float %p3 %p3 0 1

%dirS = OpFSub %v2float %p2xy %p0xy
%dotS = OpDot %float %dirS %dirS
%lenS = OpExtInst %float %glsl Sqrt %dotS
%normS = OpFDiv %v2float %dirS %lenS
%nsx = OpCompositeExtract %float %normS 0
%nsy = OpCompositeExtract %float %normS 1
%neg_nsy = OpFNegate %float %nsy
%perpS = OpCompositeConstruct %v2float %neg_nsy %nsx

%dirE = OpFSub %v2float %p3xy %p1xy
%dotE = OpDot %float %dirE %dirE
%lenE = OpExtInst %float %glsl Sqrt %dotE
%normE = OpFDiv %v2float %dirE %lenE
%nex = OpCompositeExtract %float %normE 0
%ney = OpCompositeExtract %float %normE 1
%neg_ney = OpFNegate %float %ney
%perpE = OpCompositeConstruct %v2float %neg_ney %nex

%deltaPos = OpFSub %v4float %p2 %p1
%deltaPerp = OpFSub %v2float %perpE %perpS

%c1ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %c1ptr
%c2ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %c2ptr
%deltaColor = OpFSub %v3float %c2 %c1

%mulPos0 = OpVectorTimesScalar %v4float %deltaPos %float_0
%pos0 = OpFAdd %v4float %p1 %mulPos0
%mulPos1 = OpVectorTimesScalar %v4float %deltaPos %float_1_3
%pos1 = OpFAdd %v4float %p1 %mulPos1

%mulPerp0 = OpVectorTimesScalar %v2float %deltaPerp %float_0
%perp0 = OpFAdd %v2float %perpS %mulPerp0
%mulPerp1 = OpVectorTimesScalar %v2float %deltaPerp %float_1_3
%perp1 = OpFAdd %v2float %perpS %mulPerp1

%off0 = OpVectorTimesScalar %v2float %perp0 %float_thick
%off1 = OpVectorTimesScalar %v2float %perp1 %float_thick

%mulCol0 = OpVectorTimesScalar %v3float %deltaColor %float_0
%col0 = OpFAdd %v3float %c1 %mulCol0
%mulCol1 = OpVectorTimesScalar %v3float %deltaColor %float_1_3
%col1 = OpFAdd %v3float %c1 %mulCol1

%off0x = OpCompositeExtract %float %off0 0
%off0y = OpCompositeExtract %float %off0 1
%off0_4 = OpCompositeConstruct %v4float %off0x %off0y %float_0 %float_0
%off1x = OpCompositeExtract %float %off1 0
%off1y = OpCompositeExtract %float %off1 1
%off1_4 = OpCompositeConstruct %v4float %off1x %off1y %float_0 %float_0

%pos0_plus = OpFAdd %v4float %pos0 %off0_4
%pos0_minus = OpFSub %v4float %pos0 %off0_4
%pos1_plus = OpFAdd %v4float %pos1 %off1_4
%pos1_minus = OpFSub %v4float %pos1 %off1_4

OpStore %outPosPtr %pos0_plus
OpStore %outColor %col0
OpEmitVertex
OpStore %outPosPtr %pos0_minus
OpStore %outColor %col0
OpEmitVertex
OpStore %outPosPtr %pos1_plus
OpStore %outColor %col1
OpEmitVertex
OpStore %outPosPtr %pos1_minus
OpStore %outColor %col1
OpEmitVertex
OpEndPrimitive

%mulPos2 = OpVectorTimesScalar %v4float %deltaPos %float_2_3
%pos2 = OpFAdd %v4float %p1 %mulPos2
%mulPos3 = OpVectorTimesScalar %v4float %deltaPos %float_1
%pos3 = OpFAdd %v4float %p1 %mulPos3

%mulPerp2 = OpVectorTimesScalar %v2float %deltaPerp %float_2_3
%perp2 = OpFAdd %v2float %perpS %mulPerp2
%mulPerp3 = OpVectorTimesScalar %v2float %deltaPerp %float_1
%perp3 = OpFAdd %v2float %perpS %mulPerp3

%off2 = OpVectorTimesScalar %v2float %perp2 %float_thick
%off3 = OpVectorTimesScalar %v2float %perp3 %float_thick

%mulCol2 = OpVectorTimesScalar %v3float %deltaColor %float_2_3
%col2 = OpFAdd %v3float %c1 %mulCol2
%mulCol3 = OpVectorTimesScalar %v3float %deltaColor %float_1
%col3 = OpFAdd %v3float %c1 %mulCol3

%off2x = OpCompositeExtract %float %off2 0
%off2y = OpCompositeExtract %float %off2 1
%off2_4 = OpCompositeConstruct %v4float %off2x %off2y %float_0 %float_0
%off3x = OpCompositeExtract %float %off3 0
%off3y = OpCompositeExtract %float %off3 1
%off3_4 = OpCompositeConstruct %v4float %off3x %off3y %float_0 %float_0

%pos2_plus = OpFAdd %v4float %pos2 %off2_4
%pos2_minus = OpFSub %v4float %pos2 %off2_4
%pos3_plus = OpFAdd %v4float %pos3 %off3_4
%pos3_minus = OpFSub %v4float %pos3 %off3_4

OpStore %outPosPtr %pos2_plus
OpStore %outColor %col2
OpEmitVertex
OpStore %outPosPtr %pos2_minus
OpStore %outColor %col2
OpEmitVertex
OpStore %outPosPtr %pos3_plus
OpStore %outColor %col3
OpEmitVertex
OpStore %outPosPtr %pos3_minus
OpStore %outColor %col3
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
