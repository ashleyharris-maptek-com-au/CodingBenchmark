Below is a complete, compilable Rust program (edition 2024) that solves mazes of any size using BFS.  
It is designed to be memory‑efficient while still handling very large inputs (up to 50,000 × 50,000).

It:

- Reads the maze line by line (does **not** store full text input).
- Stores walls in a compact bitset (1 bit per cell).
- Stores BFS parent direction in a `Vec<u8>` (1 byte per cell).
- Produces a valid path if one exists.

```rust
use std::io::{self, BufRead, BufReader, Write};

fn set_wall(bits: &mut [u64], idx: usize) {
    let word = idx >> 6;
    let bit = idx & 63;
    bits[word] |= 1u64 << bit;
}

fn is_wall(bits: &[u64], idx: usize) -> bool {
    let word = idx >> 6;
    let bit = idx & 63;
    ((bits[word] >> bit) & 1) != 0
}

fn main() {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());

    // Read first line: H W
    let mut first_line = String::new();
    if reader.read_line(&mut first_line).unwrap() == 0 {
        return;
    }
    let mut it = first_line.split_whitespace();
    let h: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let w: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    if h == 0 || w == 0 {
        println!("0");
        return;
    }

    let cells = h.checked_mul(w).unwrap();
    let wall_bits_len = (cells + 63) / 64;
    let mut walls = vec![0u64; wall_bits_len];
    let mut parent = vec![0u8; cells];

    let mut start: Option<usize> = None;
    let mut goal: Option<usize> = None;

    // Read maze row by row
    let mut line_buf: Vec<u8> = Vec::with_capacity(w + 2);
    for r in 0..h {
        line_buf.clear();
        reader.read_until(b'\n', &mut line_buf).unwrap();

        if line_buf.last() == Some(&b'\n') {
            line_buf.pop();
        }
        if line_buf.last() == Some(&b'\r') {
            line_buf.pop();
        }

        if line_buf.len() < w {
            line_buf.resize(w, b' ');
        }

        for c in 0..w {
            let idx = r * w + c;
            match line_buf[c] {
                b'#' => set_wall(&mut walls, idx),
                b'A' => start = Some(idx),
                b'B' => goal = Some(idx),
                _ => {}
            }
        }
    }

    let start = match start {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let goal = match goal {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    if start == goal {
        println!("1\n0 0");
        return;
    }

    // BFS
    let mut queue: Vec<usize> = Vec::new();
    queue.push(start);
    parent[start] = 255; // mark start

    let mut head = 0;
    while head < queue.len() {
        let idx = queue[head];
        head += 1;

        if idx == goal {
            break;
        }

        let r = idx / w;
        let c = idx % w;

        // Up
        if r > 0 {
            let n = idx - w;
            if parent[n] == 0 && !is_wall(&walls, n) {
                parent[n] = 2; // came from down
                queue.push(n);
            }
        }
        // Down
        if r + 1 < h {
            let n = idx + w;
            if parent[n] == 0 && !is_wall(&walls, n) {
                parent[n] = 1; // came from up
                queue.push(n);
            }
        }
        // Left
        if c > 0 {
            let n = idx - 1;
            if parent[n] == 0 && !is_wall(&walls, n) {
                parent[n] = 4; // came from right
                queue.push(n);
            }
        }
        // Right
        if c + 1 < w {
            let n = idx + 1;
            if parent[n] == 0 && !is_wall(&walls, n) {
                parent[n] = 3; // came from left
                queue.push(n);
            }
        }
    }

    if parent[goal] == 0 {
        println!("0");
        return;
    }

    // Reconstruct path
    let mut path: Vec<usize> = Vec::new();
    let mut cur = goal;
    path.push(cur);
    while cur != start {
        let dir = parent[cur];
        cur = match dir {
            1 => cur - w, // previous is up
            2 => cur + w, // previous is down
            3 => cur - 1, // previous is left
            4 => cur + 1, // previous is right
            _ => break,
        };
        path.push(cur);
    }
    path.reverse();

    // Output
    let mut out = io::BufWriter::new(io::stdout().lock());
    writeln!(out, "{}", path.len()).unwrap();
    for idx in path {
        let r = idx / w;
        let c = idx % w;
        writeln!(out, "{} {}", r, c).unwrap();
    }
}
```
