Here is a complete, runnable Python solution that reads a CVRP instance from **stdin**, constructs feasible routes using a simple **sweep heuristic**, respects vehicle capacity, and outputs a valid routing plan:

```python
import sys
import math

def read_input():
    data = sys.stdin.read().strip().split()
    if not data:
        return None
    it = iter(data)
    N = int(next(it))
    K = int(next(it))
    CAP = float(next(it))
    depot_x = float(next(it))
    depot_y = float(next(it))
    customers = []
    for i in range(N):
        x = float(next(it))
        y = float(next(it))
        d = float(next(it))
        customers.append((x, y, d))
    return N, K, CAP, (depot_x, depot_y), customers

def distance(p1, p2):
    return math.hypot(p1[0]-p2[0], p1[1]-p2[1])

def route_distance(route, customers, depot):
    if not route:
        return 0.0
    dist = 0.0
    prev = depot
    for idx in route:
        dist += distance(prev, customers[idx])
        prev = customers[idx]
    dist += distance(prev, depot)
    return dist

def main():
    parsed = read_input()
    if not parsed:
        return
    N, K, CAP, depot, customers = parsed

    # Compute polar angle for sweep heuristic
    angles = []
    for i, (x, y, d) in enumerate(customers):
        ang = math.atan2(y - depot[1], x - depot[0])
        angles.append((ang, i))
    angles.sort()

    # Build initial routes by sweeping
    routes = []
    current_route = []
    current_load = 0.0
    for _, idx in angles:
        demand = customers[idx][2]
        if current_load + demand <= CAP or not current_route:
            current_route.append(idx)
            current_load += demand
        else:
            routes.append(current_route)
            current_route = [idx]
            current_load = demand
    if current_route:
        routes.append(current_route)

    # Try to merge routes if we exceed K
    def route_load(r):
        return sum(customers[i][2] for i in r)

    while len(routes) > K:
        # Sort by load ascending
        routes.sort(key=route_load)
        merged = False
        r = routes.pop(0)
        r_load = route_load(r)
        for j in range(len(routes)):
            if r_load + route_load(routes[j]) <= CAP:
                routes[j] = routes[j] + r  # concatenate
                merged = True
                break
        if not merged:
            # cannot merge anymore
            routes.insert(0, r)
            break

    total_dist = sum(route_distance(r, customers, depot) for r in routes)

    # Output
    print(f"{total_dist:.3f}")
    print(len(routes))
    for r in routes:
        print(" ".join(map(str, r)))

if __name__ == "__main__":
    main()
```

### ✅ Features:
- Reads from `stdin`, writes to `stdout`.
- Produces valid routes covering each customer exactly once.
- Respects vehicle capacity.
- Uses a sweep heuristic (angle-sorted clustering).
- Attempts to merge routes to respect the K vehicle limit.
- Outputs total distance and route list.
