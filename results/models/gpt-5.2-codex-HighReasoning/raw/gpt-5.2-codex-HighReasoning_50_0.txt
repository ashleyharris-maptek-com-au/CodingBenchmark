```glsl
#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main()
{
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) return;

    vec3 pos = inData[i * 2].xyz;
    vec3 vel = inData[i * 2 + 1].xyz;

    float dt        = radii.x;
    float sepRadius = radii.y;
    float alignRad  = radii.z;
    float cohRad    = radii.w;

    float sepWeight   = weights.x;
    float alignWeight = weights.y;
    float cohWeight   = weights.z;
    float maxSpeed    = weights.w;

    vec3 sepAcc  = vec3(0.0);
    vec3 alignAcc = vec3(0.0);
    vec3 cohAcc  = vec3(0.0);
    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;

    for (uint j = 0; j < N; ++j)
    {
        if (j == i) continue;

        vec3 posj = inData[j * 2].xyz;
        vec3 velj = inData[j * 2 + 1].xyz;

        vec3 diff = pos - posj;
        float dist = length(diff);
        if (dist <= 0.00001) continue;

        if (dist <= sepRadius) {
            sepAcc += diff / dist;  // weighted by inverse distance
            sepCount++;
        }

        if (dist <= alignRad) {
            alignAcc += velj;
            alignCount++;
        }

        if (dist <= cohRad) {
            cohAcc += posj;
            cohCount++;
        }
    }

    if (sepCount > 0) {
        sepAcc /= float(sepCount);
    }
    if (alignCount > 0) {
        alignAcc = (alignAcc / float(alignCount)) - vel;
    }
    if (cohCount > 0) {
        cohAcc = (cohAcc / float(cohCount)) - pos;
    }

    vec3 accel = sepAcc * sepWeight + alignAcc * alignWeight + cohAcc * cohWeight;

    // Boundary force
    float boundSize  = bounds.x;
    float boundForce = bounds.y;
    float limit = boundSize * 0.4;
    float scale = boundSize * 0.5;

    // X
    if (pos.x > limit) {
        accel.x -= boundForce * (pos.x - limit) / scale;
    } else if (pos.x < -limit) {
        accel.x += boundForce * (-limit - pos.x) / scale;
    }
    // Y
    if (pos.y > limit) {
        accel.y -= boundForce * (pos.y - limit) / scale;
    } else if (pos.y < -limit) {
        accel.y += boundForce * (-limit - pos.y) / scale;
    }
    // Z
    if (pos.z > limit) {
        accel.z -= boundForce * (pos.z - limit) / scale;
    } else if (pos.z < -limit) {
        accel.z += boundForce * (-limit - pos.z) / scale;
    }

    // Integrate velocity and position
    vel += accel * dt;

    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = normalize(vel) * maxSpeed;
    }

    pos += vel * dt;

    outData[i * 2]     = vec4(pos, 0.0);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
```
