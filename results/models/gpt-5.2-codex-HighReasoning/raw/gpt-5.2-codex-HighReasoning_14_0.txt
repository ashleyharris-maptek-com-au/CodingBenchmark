Below is a complete, compilable C# Minesweeper solver that scales from tiny to massive boards.  
It uses **fast local inference** everywhere, and **adaptive constraint solving** for small/medium boards or small components.

✅ **Never guesses**  
✅ **Outputs only provably safe cells**  
✅ **Handles huge boards efficiently**

---

```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Numerics;

public class Program
{
    // Grid dimensions
    static int H, W;

    // Cell states: 0 = unknown, 1 = mine, 2 = safe-known, 3 = revealed number
    static byte[] state;
    static byte[] num;

    // Queue for number processing
    static List<int> queue;
    static int qHead;
    static bool[] inQueue;

    // Result safe cells
    static List<int> safeCells;

    // Adaptive flags
    static bool allowAdvanced = false;
    static int maxVars = 20;
    static int maxEqs = 30;

    static readonly int[] dr = { -1, -1, -1, 0, 0, 1, 1, 1 };
    static readonly int[] dc = { -1, 0, 1, -1, 1, -1, 0, 1 };

    public static void Main()
    {
        var reader = new StreamReader(Console.OpenStandardInput(), System.Text.Encoding.ASCII, false, 1 << 20);
        string first = reader.ReadLine();
        if (string.IsNullOrEmpty(first)) return;

        var parts = first.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        H = int.Parse(parts[0]);
        W = int.Parse(parts[1]);

        int N = H * W;
        state = new byte[N];
        num = new byte[N];
        safeCells = new List<int>();

        var numbers = new List<int>();

        for (int r = 0; r < H; r++)
        {
            string line = reader.ReadLine() ?? "";
            if (line.Length < W) line = line.PadRight(W, ' ');

            for (int c = 0; c < W; c++)
            {
                char ch = (c < line.Length) ? line[c] : ' ';
                int idx = r * W + c;

                if (ch == '*')
                {
                    state[idx] = 1; // mine
                }
                else if (ch >= '0' && ch <= '8')
                {
                    state[idx] = 3; // number
                    num[idx] = (byte)(ch - '0');
                    numbers.Add(idx);
                }
                else
                {
                    state[idx] = 0; // unknown
                }
            }
        }

        // Initialize queue with all numbers
        queue = new List<int>(numbers.Count + 8);
        inQueue = new bool[N];
        foreach (int idx in numbers)
        {
            queue.Add(idx);
            inQueue[idx] = true;
        }
        qHead = 0;

        long size = (long)H * W;
        allowAdvanced = (size <= 2_000_000 && numbers.Count <= 500_000);

        if (size <= 200_000) maxVars = 20;
        else if (size <= 1_000_000) maxVars = 18;
        else maxVars = 15;
        maxEqs = maxVars * 2;

        bool changed;
        do
        {
            changed = ApplyBasic();
            if (allowAdvanced)
            {
                bool adv = ApplyAdvanced();
                if (adv) changed = true;
            }
        } while (changed);

        // Output
        var writer = new StreamWriter(Console.OpenStandardOutput());
        writer.WriteLine(safeCells.Count);
        foreach (int idx in safeCells)
        {
            int r = idx / W;
            int c = idx - r * W;
            writer.WriteLine($"{c} {r}");
        }
        writer.Flush();
    }

    // Basic rule-based deductions
    static bool ApplyBasic()
    {
        bool changed = false;

        while (qHead < queue.Count)
        {
            int idx = queue[qHead++];
            inQueue[idx] = false;
            if (state[idx] != 3) continue;

            int r = idx / W;
            int c = idx - r * W;

            int mineCount = 0;
            int unknownCount = 0;
            Span<int> unknowns = stackalloc int[8];

            for (int k = 0; k < 8; k++)
            {
                int nr = r + dr[k];
                int nc = c + dc[k];
                if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;

                int nIdx = nr * W + nc;
                byte st = state[nIdx];
                if (st == 1) mineCount++;
                else if (st == 0) unknowns[unknownCount++] = nIdx;
            }

            if (unknownCount == 0) continue;

            int required = num[idx];

            if (required == mineCount)
            {
                for (int i = 0; i < unknownCount; i++)
                    if (MarkSafe(unknowns[i])) changed = true;
            }
            else if (required == mineCount + unknownCount)
            {
                for (int i = 0; i < unknownCount; i++)
                    if (MarkMine(unknowns[i])) changed = true;
            }
        }

        return changed;
    }

    // Advanced component solving using exhaustive CSP for small components
    static bool ApplyAdvanced()
    {
        int N = state.Length;
        bool[] visitedUnknown = new bool[N];
        bool[] visitedNumber = new bool[N];

        bool changed = false;

        for (int idx = 0; idx < N; idx++)
        {
            if (state[idx] != 0 || visitedUnknown[idx] || !HasAdjacentNumber(idx))
                continue;

            // BFS to build component
            List<int> compVars = new List<int>();
            List<int> compNums = new List<int>();
            var uQueue = new List<int>();
            var nQueue = new List<int>();
            int uHead = 0, nHead = 0;

            visitedUnknown[idx] = true;
            uQueue.Add(idx);

            while (uHead < uQueue.Count || nHead < nQueue.Count)
            {
                while (uHead < uQueue.Count)
                {
                    int u = uQueue[uHead++];
                    compVars.Add(u);

                    int r = u / W;
                    int c = u - r * W;

                    for (int k = 0; k < 8; k++)
                    {
                        int nr = r + dr[k];
                        int nc = c + dc[k];
                        if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;

                        int nIdx = nr * W + nc;
                        if (state[nIdx] == 3 && !visitedNumber[nIdx])
                        {
                            visitedNumber[nIdx] = true;
                            nQueue.Add(nIdx);
                        }
                    }
                }

                while (nHead < nQueue.Count)
                {
                    int n = nQueue[nHead++];
                    compNums.Add(n);

                    int r = n / W;
                    int c = n - r * W;

                    for (int k = 0; k < 8; k++)
                    {
                        int nr = r + dr[k];
                        int nc = c + dc[k];
                        if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;

                        int nIdx = nr * W + nc;
                        if (state[nIdx] == 0 && !visitedUnknown[nIdx])
                        {
                            visitedUnknown[nIdx] = true;
                            uQueue.Add(nIdx);
                        }
                    }
                }
            }

            if (compVars.Count == 0 || compNums.Count == 0) continue;
            if (compVars.Count > maxVars || compNums.Count > maxEqs) continue;

            // Build mapping
            var map = new Dictionary<int, int>(compVars.Count);
            for (int i = 0; i < compVars.Count; i++) map[compVars[i]] = i;

            var masks = new List<ulong>();
            var reqs = new List<int>();

            foreach (int nIdx in compNums)
            {
                int r = nIdx / W;
                int c = nIdx - r * W;

                int required = num[nIdx];
                int mines = 0;
                ulong mask = 0;

                for (int k = 0; k < 8; k++)
                {
                    int nr = r + dr[k];
                    int nc = c + dc[k];
                    if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;

                    int t = nr * W + nc;
                    if (state[t] == 1) mines++;
                    else if (state[t] == 0 && map.TryGetValue(t, out int vi))
                        mask |= 1UL << vi;
                }

                required -= mines;
                int count = BitOperations.PopCount(mask);

                if (count == 0)
                {
                    if (required != 0) { masks.Clear(); break; }
                    else continue;
                }
                if (required < 0 || required > count) { masks.Clear(); break; }

                masks.Add(mask);
                reqs.Add(required);
            }

            if (masks.Count == 0) continue;

            int nVars = compVars.Count;
            int eqCount = masks.Count;

            var varEqs = new List<int>[nVars];
            for (int i = 0; i < nVars; i++) varEqs[i] = new List<int>();

            for (int ei = 0; ei < eqCount; ei++)
            {
                ulong mask = masks[ei];
                for (int v = 0; v < nVars; v++)
                    if ((mask & (1UL << v)) != 0)
                        varEqs[v].Add(ei);
            }

            int[] eqVarCount = new int[eqCount];
            for (int ei = 0; ei < eqCount; ei++)
                eqVarCount[ei] = BitOperations.PopCount(masks[ei]);

            int[] eqAssigned = new int[eqCount];
            int[] eqMine = new int[eqCount];
            int[] reqArr = reqs.ToArray();

            ulong allBits = (nVars == 64) ? ulong.MaxValue : ((1UL << nVars) - 1);
            ulong alwaysMine = allBits;
            ulong alwaysSafe = allBits;
            long solutions = 0;

            void DFS(int vIdx, ulong minesMask)
            {
                if (vIdx == nVars)
                {
                    solutions++;
                    alwaysMine &= minesMask;
                    alwaysSafe &= (~minesMask) & allBits;
                    return;
                }

                for (int val = 0; val <= 1; val++)
                {
                    bool ok = true;
                    foreach (int eq in varEqs[vIdx])
                    {
                        eqAssigned[eq]++;
                        if (val == 1) eqMine[eq]++;

                        int mine = eqMine[eq];
                        int assigned = eqAssigned[eq];

                        if (mine > reqArr[eq] ||
                            mine + (eqVarCount[eq] - assigned) < reqArr[eq])
                            ok = false;
                    }

                    if (ok)
                        DFS(vIdx + 1, val == 1 ? minesMask | (1UL << vIdx) : minesMask);

                    foreach (int eq in varEqs[vIdx])
                    {
                        if (val == 1) eqMine[eq]--;
                        eqAssigned[eq]--;
                    }
                }
            }

            DFS(0, 0);

            if (solutions == 0) continue;

            ulong safeMask = alwaysSafe;
            ulong mineMask = alwaysMine;

            for (int v = 0; v < nVars; v++)
            {
                ulong bit = 1UL << v;
                if ((safeMask & bit) != 0)
                {
                    if (MarkSafe(compVars[v])) changed = true;
                }
                else if ((mineMask & bit) != 0)
                {
                    if (MarkMine(compVars[v])) changed = true;
                }
            }
        }

        return changed;
    }

    static bool HasAdjacentNumber(int idx)
    {
        int r = idx / W;
        int c = idx - r * W;

        for (int k = 0; k < 8; k++)
        {
            int nr = r + dr[k];
            int nc = c + dc[k];
            if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;

            int nIdx = nr * W + nc;
            if (state[nIdx] == 3) return true;
        }
        return false;
    }

    static void EnqueueNumber(int idx)
    {
        if (state[idx] == 3 && !inQueue[idx])
        {
            inQueue[idx] = true;
            queue.Add(idx);
        }
    }

    static void EnqueueAdjacentNumbers(int idx)
    {
        int r = idx / W;
        int c = idx - r * W;

        for (int k = 0; k < 8; k++)
        {
            int nr = r + dr[k];
            int nc = c + dc[k];
            if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;

            int nIdx = nr * W + nc;
            if (state[nIdx] == 3) EnqueueNumber(nIdx);
        }
    }

    static bool MarkSafe(int idx)
    {
        if (state[idx] != 0) return false;
        state[idx] = 2;
        safeCells.Add(idx);
        EnqueueAdjacentNumbers(idx);
        return true;
    }

    static bool MarkMine(int idx)
    {
        if (state[idx] != 0) return false;
        state[idx] = 1;
        EnqueueAdjacentNumbers(idx);
        return true;
    }
}
```
