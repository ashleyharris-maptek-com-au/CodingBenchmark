OpCapability Shader
OpCapability Geometry
%ext = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor %ubo
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal   Location 1
OpDecorate %color    Location 2
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void      = OpTypeVoid
%func      = OpTypeFunction %void
%bool      = OpTypeBool
%int       = OpTypeInt 32 1
%uint      = OpTypeInt 32 0
%float     = OpTypeFloat 32
%v2float   = OpTypeVector %float 2
%v3float   = OpTypeVector %float 3
%v4float   = OpTypeVector %float 4
%mat4      = OpTypeMatrix %v4float 4

%gl_PerVertex_in  = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_6 = OpConstant %uint 6
%int_0  = OpConstant %int 0
%int_1  = OpConstant %int 1
%int_2  = OpConstant %int 2
%int_3  = OpConstant %int 3
%int_4  = OpConstant %int 4
%int_5  = OpConstant %int 5

%float_0   = OpConstant %float 0.0
%float_1   = OpConstant %float 1.0
%float_0_3 = OpConstant %float 0.3
%float_thick = OpConstant %float 0.015

%v3_white = OpConstantComposite %v3float %float_1 %float_1 %float_1

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%arr_v3_in = OpTypeArray %v3float %uint_6

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4     = OpTypePointer Input %v4float
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3     = OpTypePointer Input %v3float

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO

%gl_in    = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal   = OpVariable %ptr_in_arr_v3 Input
%color    = OpVariable %ptr_in_arr_v3 Input

%gl_out   = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%ubo      = OpVariable %ptr_ubo Uniform

%main = OpFunction %void None %func
%entry = OpLabel

%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; ---- Emit filled triangle (vertices 0,2,4) ----
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0     = OpLoad %v4float %p0_ptr
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0     = OpLoad %v3float %c0_ptr
%c0_dim = OpVectorTimesScalar %v3float %c0 %float_0_3
OpStore %out_pos_ptr %p0
OpStore %outColor %c0_dim
OpEmitVertex

%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2     = OpLoad %v4float %p2_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c2     = OpLoad %v3float %c2_ptr
%c2_dim = OpVectorTimesScalar %v3float %c2 %float_0_3
OpStore %out_pos_ptr %p2
OpStore %outColor %c2_dim
OpEmitVertex

%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p4     = OpLoad %v4float %p4_ptr
%c4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
%c4     = OpLoad %v3float %c4_ptr
%c4_dim = OpVectorTimesScalar %v3float %c4 %float_0_3
OpStore %out_pos_ptr %p4
OpStore %outColor %c4_dim
OpEmitVertex

OpEndPrimitive

; ---- Edge 0-2 ----
%p0e = OpLoad %v4float %p0_ptr
%p2e = OpLoad %v4float %p2_ptr
%p0_xy = OpVectorShuffle %v2float %p0e %p0e 0 1
%p2_xy = OpVectorShuffle %v2float %p2e %p2e 0 1
%delta = OpFSub %v2float %p2_xy %p0_xy
%dir   = OpExtInst %v2float %ext Normalize %delta
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%offset2 = OpVectorTimesScalar %v2float %perp %float_thick
%off_x = OpCompositeExtract %float %offset2 0
%off_y = OpCompositeExtract %float %offset2 1
%offset4 = OpCompositeConstruct %v4float %off_x %off_y %float_0 %float_0
%p0_plus  = OpFAdd %v4float %p0e %offset4
%p0_minus = OpFSub %v4float %p0e %offset4
%p2_plus  = OpFAdd %v4float %p2e %offset4
%p2_minus = OpFSub %v4float %p2e %offset4
OpStore %out_pos_ptr %p0_plus
OpStore %outColor %v3_white
OpEmitVertex
OpStore %out_pos_ptr %p0_minus
OpStore %outColor %v3_white
OpEmitVertex
OpStore %out_pos_ptr %p2_plus
OpStore %outColor %v3_white
OpEmitVertex
OpStore %out_pos_ptr %p2_minus
OpStore %outColor %v3_white
OpEmitVertex
OpEndPrimitive

; ---- Edge 2-4 ----
%p2e2 = OpLoad %v4float %p2_ptr
%p4e  = OpLoad %v4float %p4_ptr
%p2_xy2 = OpVectorShuffle %v2float %p2e2 %p2e2 0 1
%p4_xy  = OpVectorShuffle %v2float %p4e %p4e 0 1
%delta2 = OpFSub %v2float %p4_xy %p2_xy2
%dir2   = OpExtInst %v2float %ext Normalize %delta2
%dir2_x = OpCompositeExtract %float %dir2 0
%dir2_y = OpCompositeExtract %float %dir2 1
%neg_dir2_y = OpFNegate %float %dir2_y
%perp2 = OpCompositeConstruct %v2float %neg_dir2_y %dir2_x
%offset2b = OpVectorTimesScalar %v2float %perp2 %float_thick
%off2_x = OpCompositeExtract %float %offset2b 0
%off2_y = OpCompositeExtract %float %offset2b 1
%offset4b = OpCompositeConstruct %v4float %off2_x %off2_y %float_0 %float_0
%p2_plus2  = OpFAdd %v4float %p2e2 %offset4b
%p2_minus2 = OpFSub %v4float %p2e2 %offset4b
%p4_plus   = OpFAdd %v4float %p4e %offset4b
%p4_minus  = OpFSub %v4float %p4e %offset4b
OpStore %out_pos_ptr %p2_plus2
OpStore %outColor %v3_white
OpEmitVertex
OpStore %out_pos_ptr %p2_minus2
OpStore %outColor %v3_white
OpEmitVertex
OpStore %out_pos_ptr %p4_plus
OpStore %outColor %v3_white
OpEmitVertex
OpStore %out_pos_ptr %p4_minus
OpStore %outColor %v3_white
OpEmitVertex
OpEndPrimitive

; ---- Edge 4-0 ----
%p4e2 = OpLoad %v4float %p4_ptr
%p0e2 = OpLoad %v4float %p0_ptr
%p4_xy2 = OpVectorShuffle %v2float %p4e2 %p4e2 0 1
%p0_xy2 = OpVectorShuffle %v2float %p0e2 %p0e2 0 1
%delta3 = OpFSub %v2float %p0_xy2 %p4_xy2
%dir3   = OpExtInst %v2float %ext Normalize %delta3
%dir3_x = OpCompositeExtract %float %dir3 0
%dir3_y = OpCompositeExtract %float %dir3 1
%neg_dir3_y = OpFNegate %float %dir3_y
%perp3 = OpCompositeConstruct %v2float %neg_dir3_y %dir3_x
%offset3 = OpVectorTimesScalar %v2float %perp3 %float_thick
%off3_x = OpCompositeExtract %float %offset3 0
%off3_y = OpCompositeExtract %float %offset3 1
%offset4c = OpCompositeConstruct %v4float %off3_x %off3_y %float_0 %float_0
%p4_plus2  = OpFAdd %v4float %p4e2 %offset4c
%p4_minus2 = OpFSub %v4float %p4e2 %offset4c
%p0_plus2  = OpFAdd %v4float %p0e2 %offset4c
%p0_minus2 = OpFSub %v4float %p0e2 %offset4c
OpStore %out_pos_ptr %p4_plus2
OpStore %outColor %v3_white
OpEmitVertex
OpStore %out_pos_ptr %p4_minus2
OpStore %outColor %v3_white
OpEmitVertex
OpStore %out_pos_ptr %p0_plus2
OpStore %outColor %v3_white
OpEmitVertex
OpStore %out_pos_ptr %p0_minus2
OpStore %outColor %v3_white
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
