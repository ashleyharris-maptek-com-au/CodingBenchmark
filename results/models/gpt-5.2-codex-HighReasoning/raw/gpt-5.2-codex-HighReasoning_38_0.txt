import sys

def solve_exact_cover(U, sets):
    # Initial uncovered elements and available set indices
    uncovered = set(range(U))
    available = list(range(len(sets)))

    # Recursive Algorithm X style solver
    def backtrack(uncovered, available):
        if not uncovered:
            return []

        # Choose element with fewest covering sets
        best_e = None
        best_cands = None
        min_count = None

        for e in uncovered:
            cands = [i for i in available if e in sets[i]]
            if not cands:
                return None
            if min_count is None or len(cands) < min_count:
                min_count = len(cands)
                best_e = e
                best_cands = cands
                if min_count == 1:
                    break

        # Try each candidate set
        for i in best_cands:
            chosen_set = sets[i]
            new_uncovered = uncovered - chosen_set
            new_available = [j for j in available if j != i and sets[j].isdisjoint(chosen_set)]
            res = backtrack(new_uncovered, new_available)
            if res is not None:
                return [i] + res
        return None

    return backtrack(uncovered, available)

def main():
    data = sys.stdin.read().splitlines()
    if not data:
        return

    first = data[0].split()
    if len(first) < 2:
        return
    U = int(first[0])
    S = int(first[1])

    sets = []
    for i in range(S):
        line = data[i + 1].strip() if i + 1 < len(data) else ""
        if line == "":
            elems = []
        else:
            elems = list(map(int, line.split()))
        sets.append(set(elems))

    solution = solve_exact_cover(U, sets)

    if solution is None:
        print("NO SOLUTION")
    else:
        solution_sorted = sorted(solution)
        print("SOLUTION")
        if solution_sorted:
            print(" ".join(map(str, solution_sorted)))
        else:
            print()

if __name__ == "__main__":
    sys.setrecursionlimit(10000)
    main()
