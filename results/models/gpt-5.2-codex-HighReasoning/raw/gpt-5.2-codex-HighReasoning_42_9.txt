OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%glsl450 = OpExtInstImport "GLSL450"

%void = OpTypeVoid
%fn = OpTypeFunction %void

%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%ptr_in_v3 = OpTypePointer Input %v3float

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%float_0_025 = OpConstant %float 0.025

%main = OpFunction %void None %fn
%label = OpLabel

%ptr_prev = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%prev4 = OpLoad %v4float %ptr_prev
%ptr_start = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%start4 = OpLoad %v4float %ptr_start
%ptr_end = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%end4 = OpLoad %v4float %ptr_end
%ptr_next = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%next4 = OpLoad %v4float %ptr_next

%prev_xy = OpVectorShuffle %v2float %prev4 %prev4 0 1
%start_xy = OpVectorShuffle %v2float %start4 %start4 0 1
%end_xy = OpVectorShuffle %v2float %end4 %end4 0 1
%next_xy = OpVectorShuffle %v2float %next4 %next4 0 1

%dir0 = OpFSub %v2float %end_xy %prev_xy
%t0 = OpExtInst %v2float %glsl450 Normalize %dir0
%dir1 = OpFSub %v2float %next_xy %start_xy
%t1 = OpExtInst %v2float %glsl450 Normalize %dir1

%t0x = OpCompositeExtract %float %t0 0
%t0y = OpCompositeExtract %float %t0 1
%neg_t0y = OpFNegate %float %t0y
%perp0 = OpCompositeConstruct %v2float %neg_t0y %t0x

%t1x = OpCompositeExtract %float %t1 0
%t1y = OpCompositeExtract %float %t1 1
%neg_t1y = OpFNegate %float %t1y
%perp1 = OpCompositeConstruct %v2float %neg_t1y %t1x

%perp0s = OpVectorTimesScalar %v2float %perp0 %float_0_025
%perp1s = OpVectorTimesScalar %v2float %perp1 %float_0_025

%start_plus = OpFAdd %v2float %start_xy %perp0s
%start_minus = OpFSub %v2float %start_xy %perp0s
%end_plus = OpFAdd %v2float %end_xy %perp1s
%end_minus = OpFSub %v2float %end_xy %perp1s

%start_z = OpCompositeExtract %float %start4 2
%start_w = OpCompositeExtract %float %start4 3
%end_z = OpCompositeExtract %float %end4 2
%end_w = OpCompositeExtract %float %end4 3

%spx = OpCompositeExtract %float %start_plus 0
%spy = OpCompositeExtract %float %start_plus 1
%smx = OpCompositeExtract %float %start_minus 0
%smy = OpCompositeExtract %float %start_minus 1
%epx = OpCompositeExtract %float %end_plus 0
%epy = OpCompositeExtract %float %end_plus 1
%emx = OpCompositeExtract %float %end_minus 0
%emy = OpCompositeExtract %float %end_minus 1

%pos0 = OpCompositeConstruct %v4float %spx %spy %start_z %start_w
%pos1 = OpCompositeConstruct %v4float %smx %smy %start_z %start_w
%pos2 = OpCompositeConstruct %v4float %epx %epy %end_z %end_w
%pos3 = OpCompositeConstruct %v4float %emx %emy %end_z %end_w

%ptr_col1 = OpAccessChain %ptr_in_v3 %color %int_1
%col1 = OpLoad %v3float %ptr_col1
%ptr_col2 = OpAccessChain %ptr_in_v3 %color %int_2
%col2 = OpLoad %v3float %ptr_col2

%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos %pos0
OpStore %outColor %col1
OpEmitVertex

OpStore %ptr_out_pos %pos1
OpStore %outColor %col1
OpEmitVertex

OpStore %ptr_out_pos %pos2
OpStore %outColor %col2
OpEmitVertex

OpStore %ptr_out_pos %pos3
OpStore %outColor %col2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
