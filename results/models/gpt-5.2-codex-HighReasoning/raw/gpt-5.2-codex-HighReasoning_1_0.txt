Here is a complete, standard C++20 TSP solver that **adapts to scale**.  
It uses different strategies based on the number of cities:

- **Small (≤2000)**: Nearest‐Neighbor (O(N²))
- **Medium (≤100000)**: Sort by X/Y (O(N log N))
- **Large/Extreme/Epic (>100000)**: Identity order (O(N))

This ensures the program runs within time even for millions of cities, and always outputs a valid permutation starting at city **0**.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <limits>
#include <cstdint>

using namespace std;

struct Point {
    float x, y;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long N;
    if (!(cin >> N)) return 0;
    if (N <= 0) return 0;

    const long long NN_THRESHOLD = 2000;
    const long long SORT_THRESHOLD = 100000;

    // HUGE: identity order, no storage required
    if (N > SORT_THRESHOLD) {
        // read and ignore coordinates, output identity permutation
        float x, y;
        cin >> x >> y; // city 0
        cout << 0;
        for (long long i = 1; i < N; i++) {
            cin >> x >> y;
            cout << ' ' << i;
        }
        cout << '\n';
        return 0;
    }

    // Otherwise we store coordinates
    vector<Point> pts(N);
    for (long long i = 0; i < N; i++) {
        cin >> pts[i].x >> pts[i].y;
    }

    // SMALL: nearest neighbor
    if (N <= NN_THRESHOLD) {
        vector<char> visited(N, 0);
        vector<int> path;
        path.reserve(N);

        int cur = 0;
        visited[cur] = 1;
        path.push_back(cur);

        for (long long step = 1; step < N; step++) {
            int best = -1;
            double bestDist = numeric_limits<double>::infinity();
            for (int j = 0; j < N; j++) {
                if (!visited[j]) {
                    double dx = pts[cur].x - pts[j].x;
                    double dy = pts[cur].y - pts[j].y;
                    double d = dx*dx + dy*dy;
                    if (d < bestDist) {
                        bestDist = d;
                        best = j;
                    }
                }
            }
            visited[best] = 1;
            cur = best;
            path.push_back(cur);
        }

        for (size_t i = 0; i < path.size(); i++) {
            if (i) cout << ' ';
            cout << path[i];
        }
        cout << '\n';
        return 0;
    }

    // MEDIUM: sort by X then Y
    vector<int> idx(N);
    for (int i = 0; i < N; i++) idx[i] = i;

    sort(idx.begin(), idx.end(), [&](int a, int b) {
        if (pts[a].x != pts[b].x) return pts[a].x < pts[b].x;
        return pts[a].y < pts[b].y;
    });

    // Rotate so that city 0 is first
    auto it = find(idx.begin(), idx.end(), 0);
    if (it != idx.end() && it != idx.begin()) {
        rotate(idx.begin(), it, idx.end());
    }

    for (size_t i = 0; i < idx.size(); i++) {
        if (i) cout << ' ';
        cout << idx[i];
    }
    cout << '\n';

    return 0;
}
```

✅ **Properties:**
- Always outputs a valid permutation 0..N-1
- Always starts at city 0
- Handles *all* scales efficiently
- Uses only standard C++20 and supported headers
