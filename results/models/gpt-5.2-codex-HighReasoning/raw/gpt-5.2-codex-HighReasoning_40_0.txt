Here is a complete C++17 solution that exactly solves the given 0/1 ILP (maximum weight independent set) by reducing it to a **maximum weight clique** in the complement graph and using a **branch-and-bound** algorithm with greedy coloring bounds:

#include <bits/stdc++.h>
using namespace std;

struct Bitset {
    int n, words;
    vector<uint64_t> bits;
    Bitset() : n(0), words(0) {}
    Bitset(int n_, bool full=false) { init(n_, full); }
    void init(int n_, bool full=false) {
        n = n_;
        words = (n + 63) / 64;
        bits.assign(words, full ? ~0ULL : 0ULL);
        if (full && (n % 64)) bits.back() = (1ULL << (n % 64)) - 1;
    }
    bool any() const {
        for (uint64_t w : bits) if (w) return true;
        return false;
    }
    void set(int i) { bits[i>>6] |= 1ULL << (i & 63); }
    void reset(int i) { bits[i>>6] &= ~(1ULL << (i & 63)); }
    bool test(int i) const { return bits[i>>6] & (1ULL << (i & 63)); }
    int pop_lsb() {
        for (int i = 0; i < words; ++i) {
            uint64_t w = bits[i];
            if (w) {
                int b = __builtin_ctzll(w);
                bits[i] = w & (w - 1);
                return i * 64 + b;
            }
        }
        return -1;
    }
};

int Np;
vector<long long> w;
vector<Bitset> adjH;
long long bestWeight = 0;
vector<int> bestClique;

void colorSort(const Bitset& P, vector<int>& order, vector<long long>& bounds) {
    order.clear(); bounds.clear();
    vector<vector<int>> colorVerts;
    vector<long long> colorMax;
    Bitset uncolored = P;

    while (uncolored.any()) {
        Bitset available = uncolored;
        vector<int> cls;
        long long mx = 0;

        while (available.any()) {
            int v = available.pop_lsb();
            cls.push_back(v);
            if (w[v] > mx) mx = w[v];
            uncolored.reset(v);

            for (int i = 0; i < available.words; ++i) {
                available.bits[i] &= ~adjH[v].bits[i];
            }
        }

        sort(cls.begin(), cls.end(),
             [&](int a, int b){ return w[a] < w[b]; });
        colorVerts.push_back(cls);
        colorMax.push_back(mx);
    }

    long long sum = 0;
    for (size_t c = 0; c < colorVerts.size(); ++c) {
        sum += colorMax[c];
        for (int v : colorVerts[c]) {
            order.push_back(v);
            bounds.push_back(sum);
        }
    }
}

void expand(Bitset P, long long currWeight, vector<int>& currClique) {
    if (!P.any()) {
        if (currWeight > bestWeight) {
            bestWeight = currWeight;
            bestClique = currClique;
        }
        return;
    }

    vector<int> order;
    vector<long long> bounds;
    colorSort(P, order, bounds);

    for (int i = (int)order.size() - 1; i >= 0; --i) {
        int v = order[i];
        if (currWeight + bounds[i] <= bestWeight) return;

        currClique.push_back(v);
        Bitset P2 = P;
        for (int k = 0; k < P2.words; ++k)
            P2.bits[k] &= adjH[v].bits[k];

        expand(P2, currWeight + w[v], currClique);
        currClique.pop_back();
        P.reset(v);
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;
    vector<long long> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    vector<pair<int,int>> edges(m);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        edges[i] = {u, v};
    }

    vector<int> newIndex(n, -1), oldIndex;
    for (int i = 0; i < n; ++i) {
        if (c[i] > 0) {
            newIndex[i] = (int)oldIndex.size();
            oldIndex.push_back(i);
            w.push_back(c[i]);
        }
    }

    Np = (int)w.size();
    if (Np == 0) {
        cout << 0 << "\n";
        for (int i = 0; i < n; ++i) {
            if (i) cout << " ";
            cout << 0;
        }
        cout << "\n";
        return 0;
    }

    vector<Bitset> adjG(Np, Bitset(Np));
    for (auto [u, v] : edges) {
        if (newIndex[u] != -1 && newIndex[v] != -1) {
            int a = newIndex[u], b = newIndex[v];
            adjG[a].set(b);
            adjG[b].set(a);
        }
    }

    adjH.assign(Np, Bitset(Np));
    Bitset full(Np, true);
    uint64_t mask = (Np % 64) ? ((1ULL << (Np % 64)) - 1) : ~0ULL;

    for (int i = 0; i < Np; ++i) {
        adjH[i] = full;
        adjH[i].reset(i);
        for (int k = 0; k < adjH[i].words; ++k)
            adjH[i].bits[k] &= ~adjG[i].bits[k];
        if (Np % 64) adjH[i].bits.back() &= mask;
    }

    // Greedy lower bound
    vector<int> order(Np);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b){ return w[a] > w[b]; });
    vector<int> greedy;
    long long gsum = 0;
    for (int v : order) {
        bool ok = true;
        for (int u : greedy) {
            if (!adjH[v].test(u)) { ok = false; break; }
        }
        if (ok) {
            greedy.push_back(v);
            gsum += w[v];
        }
    }
    bestWeight = gsum;
    bestClique = greedy;

    Bitset P(Np);
    for (int i = 0; i < Np; ++i) P.set(i);

    vector<int> currClique;
    expand(P, 0, currClique);

    vector<int> x(n, 0);
    for (int v : bestClique) {
        int orig = oldIndex[v];
        x[orig] = 1;
    }

    long long objective = 0;
    for (int i = 0; i < n; ++i)
        if (x[i]) objective += c[i];

    cout << objective << "\n";
    for (int i = 0; i < n; ++i) {
        if (i) cout << " ";
        cout << x[i];
    }
    cout << "\n";
    return 0;
}
