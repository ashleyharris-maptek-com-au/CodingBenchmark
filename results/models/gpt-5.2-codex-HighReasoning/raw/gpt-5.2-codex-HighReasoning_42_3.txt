OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

%ext = OpExtInstImport "GLSL.std.450"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0_03 = OpConstant %float 0.03

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4float = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3float = OpTypePointer Input %v3float

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_out_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3float Output

%main = OpFunction %void None %func
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

%c0_ptr = OpAccessChain %ptr_in_v3float %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3float %color %int_1
%c1 = OpLoad %v3float %c1_ptr

%p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir = OpFSub %v2float %p1xy %p0xy

%dirx = OpCompositeExtract %float %dir 0
%diry = OpCompositeExtract %float %dir 1
%negdiry = OpFNegate %float %diry
%perpdir = OpCompositeConstruct %v2float %negdiry %dirx
%perpnorm = OpExtInst %v2float %ext Normalize %perpdir
%perp = OpVectorTimesScalar %v2float %perpnorm %float_0_03

%p0_plus_xy = OpFAdd %v2float %p0xy %perp
%p0_minus_xy = OpFSub %v2float %p0xy %perp
%p1_plus_xy = OpFAdd %v2float %p1xy %perp
%p1_minus_xy = OpFSub %v2float %p1xy %perp

%p0z = OpCompositeExtract %float %p0 2
%p0w = OpCompositeExtract %float %p0 3
%p1z = OpCompositeExtract %float %p1 2
%p1w = OpCompositeExtract %float %p1 3

%p0px = OpCompositeExtract %float %p0_plus_xy 0
%p0py = OpCompositeExtract %float %p0_plus_xy 1
%pos0plus = OpCompositeConstruct %v4float %p0px %p0py %p0z %p0w

%out_pos_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %out_pos_ptr %pos0plus
OpStore %outColor %c0
OpEmitVertex

%p0mx = OpCompositeExtract %float %p0_minus_xy 0
%p0my = OpCompositeExtract %float %p0_minus_xy 1
%pos0minus = OpCompositeConstruct %v4float %p0mx %p0my %p0z %p0w
OpStore %out_pos_ptr %pos0minus
OpStore %outColor %c0
OpEmitVertex

%p1px = OpCompositeExtract %float %p1_plus_xy 0
%p1py = OpCompositeExtract %float %p1_plus_xy 1
%pos1plus = OpCompositeConstruct %v4float %p1px %p1py %p1z %p1w
OpStore %out_pos_ptr %pos1plus
OpStore %outColor %c1
OpEmitVertex

%p1mx = OpCompositeExtract %float %p1_minus_xy 0
%p1my = OpCompositeExtract %float %p1_minus_xy 1
%pos1minus = OpCompositeConstruct %v4float %p1mx %p1my %p1z %p1w
OpStore %out_pos_ptr %pos1minus
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
