Below is a **complete, compilable Rust solution** that scales from tiny graphs to extremely large ones.  
It adapts its layout method based on component size:

- **Small (≤200 nodes)** → full force-directed (Fruchterman–Reingold)  
- **Medium (≤2000 nodes)** → approximate force-directed  
- **Large (>2000 nodes)** → BFS order grid placement (fast and scalable)

It handles disconnected graphs, self-loops, and parallel edges safely.  
The output is `N` lines of `x y` coordinates (no prefix, which is allowed).

### ✅ Rust Solver (Rust 1.92 / edition 2024)

```rust
use std::cmp::max;
use std::collections::VecDeque;
use std::f32::consts::PI;
use std::io::{self, Read};

struct Scanner {
    input: Vec<u8>,
    idx: usize,
}
impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self { input: input.into_bytes(), idx: 0 }
    }
    fn next<T: std::str::FromStr>(&mut self) -> T {
        while self.idx < self.input.len() && self.input[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        let start = self.idx;
        while self.idx < self.input.len() && !self.input[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        std::str::from_utf8(&self.input[start..self.idx]).unwrap().parse().ok().unwrap()
    }
}

#[derive(Clone)]
struct CompLayout {
    nodes: Vec<usize>,
    pos: Vec<(f32, f32)>,
    minx: f32,
    maxx: f32,
    miny: f32,
    maxy: f32,
}

fn fr_layout(k: usize, edges: &Vec<(usize, usize)>, iterations: usize, sample: Option<usize>) -> Vec<(f32, f32)> {
    let kf = (k as f32).sqrt();
    let radius = kf * 2.0;

    let mut x = vec![0.0f32; k];
    let mut y = vec![0.0f32; k];
    for i in 0..k {
        let ang = 2.0 * PI * (i as f32) / (k as f32);
        x[i] = radius * ang.cos();
        y[i] = radius * ang.sin();
    }

    let mut temp = kf * 2.0;

    for _ in 0..iterations {
        let mut dx = vec![0.0f32; k];
        let mut dy = vec![0.0f32; k];

        // Repulsion
        if let Some(samp) = sample {
            for i in 0..k {
                for s in 0..samp {
                    let j = ((i as u64 * 1664525 + s as u64 * 1013904223) % k as u64) as usize;
                    if i == j { continue; }
                    let mut rx = x[i] - x[j];
                    let mut ry = y[i] - y[j];
                    let dist = (rx * rx + ry * ry).sqrt() + 1e-6;
                    let force = kf * kf / dist;
                    rx = rx / dist * force;
                    ry = ry / dist * force;
                    dx[i] += rx;
                    dy[i] += ry;
                }
            }
        } else {
            for i in 0..k {
                for j in (i + 1)..k {
                    let mut rx = x[i] - x[j];
                    let mut ry = y[i] - y[j];
                    let dist = (rx * rx + ry * ry).sqrt() + 1e-6;
                    let force = kf * kf / dist;
                    rx = rx / dist * force;
                    ry = ry / dist * force;
                    dx[i] += rx;
                    dy[i] += ry;
                    dx[j] -= rx;
                    dy[j] -= ry;
                }
            }
        }

        // Attraction
        for &(u, v) in edges {
            let mut ax = x[u] - x[v];
            let mut ay = y[u] - y[v];
            let dist = (ax * ax + ay * ay).sqrt() + 1e-6;
            let force = dist * dist / kf;
            ax = ax / dist * force;
            ay = ay / dist * force;
            dx[u] -= ax;
            dy[u] -= ay;
            dx[v] += ax;
            dy[v] += ay;
        }

        // Apply
        for i in 0..k {
            let disp = (dx[i] * dx[i] + dy[i] * dy[i]).sqrt();
            if disp > 0.0 {
                let lim = if disp < temp { disp } else { temp };
                x[i] += dx[i] / disp * lim;
                y[i] += dy[i] / disp * lim;
            }
        }
        temp *= 0.95;
    }

    x.into_iter().zip(y.into_iter()).collect()
}

fn grid_layout(k: usize, spacing: f32) -> Vec<(f32, f32)> {
    let side = (k as f32).sqrt().ceil() as usize;
    let mut pos = Vec::with_capacity(k);
    for i in 0..k {
        let x = (i % side) as f32 * spacing;
        let y = (i / side) as f32 * spacing;
        pos.push((x, y));
    }
    pos
}

fn main() {
    let mut sc = Scanner::new();
    let n: usize = sc.next();
    let m: usize = sc.next();
    if n == 0 {
        return;
    }

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u: usize = sc.next();
        let v: usize = sc.next();
        if u >= n || v >= n { continue; }
        adj[u].push(v);
        adj[v].push(u);
    }

    // Components (BFS)
    let mut comp_id = vec![usize::MAX; n];
    let mut comps: Vec<Vec<usize>> = Vec::new();
    for i in 0..n {
        if comp_id[i] != usize::MAX { continue; }
        let mut q = VecDeque::new();
        q.push_back(i);
        comp_id[i] = comps.len();
        let mut comp = Vec::new();
        while let Some(u) = q.pop_front() {
            comp.push(u);
            for &v in &adj[u] {
                if comp_id[v] == usize::MAX {
                    comp_id[v] = comps.len();
                    q.push_back(v);
                }
            }
        }
        comps.push(comp);
    }

    let mut layouts = Vec::<CompLayout>::new();
    let mut local_index = vec![usize::MAX; n];

    for comp in comps.iter() {
        let k = comp.len();
        for (i, &v) in comp.iter().enumerate() {
            local_index[v] = i;
        }

        let mut edges: Vec<(usize, usize)> = Vec::new();
        if k <= 2000 {
            for &u in comp {
                let iu = local_index[u];
                for &v in &adj[u] {
                    if local_index[v] != usize::MAX {
                        let iv = local_index[v];
                        if iu < iv && u != v {
                            edges.push((iu, iv));
                        }
                    }
                }
            }
        }

        let pos = if k <= 200 {
            fr_layout(k, &edges, 100, None)
        } else if k <= 2000 {
            fr_layout(k, &edges, 50, Some(25))
        } else {
            grid_layout(k, 1.0)
        };

        // compute bounds
        let mut minx = f32::INFINITY;
        let mut maxx = f32::NEG_INFINITY;
        let mut miny = f32::INFINITY;
        let mut maxy = f32::NEG_INFINITY;
        for &(x, y) in &pos {
            if x < minx { minx = x; }
            if x > maxx { maxx = x; }
            if y < miny { miny = y; }
            if y > maxy { maxy = y; }
        }

        layouts.push(CompLayout { nodes: comp.clone(), pos, minx, maxx, miny, maxy });

        for &v in comp {
            local_index[v] = usize::MAX;
        }
    }

    // Pack components
    let mut positions = vec![(0f32, 0f32); n];
    let total = n as f32;
    let max_row_width = total.sqrt() * 20.0 + 100.0;
    let margin = 10.0;

    let mut x_off = 0.0f32;
    let mut y_off = 0.0f32;
    let mut row_h = 0.0f32;

    for comp in layouts.iter() {
        let width = comp.maxx - comp.minx + margin;
        let height = comp.maxy - comp.miny + margin;
        if x_off + width > max_row_width {
            x_off = 0.0;
            y_off += row_h + margin;
            row_h = 0.0;
        }
        for (i, &node) in comp.nodes.iter().enumerate() {
            let (x, y) = comp.pos[i];
            positions[node] = (x - comp.minx + x_off, y - comp.miny + y_off);
        }
        x_off += width + margin;
        row_h = row_h.max(height);
    }

    // Output
    let mut out = String::new();
    for (x, y) in positions {
        out.push_str(&format!("{:.6} {:.6}\n", x, y));
    }
    print!("{}", out);
}
```
