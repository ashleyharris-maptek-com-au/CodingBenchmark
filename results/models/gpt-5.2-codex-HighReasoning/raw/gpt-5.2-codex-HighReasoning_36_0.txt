Below is a complete C++20 solution that computes an **exact maximum independent set** using a branch‑and‑bound **maximum clique** algorithm on the **complement graph** (independent set in G = clique in complement).  
It is exact and works well for typical competitive programming constraints (n up to ~100).  

The algorithm uses greedy coloring for strong pruning.

```cpp
#include <bits/stdc++.h>
using namespace std;

using ull = unsigned long long;

struct Bitset {
    int n = 0;
    int blocks = 0;
    vector<ull> b;
    Bitset() {}
    Bitset(int n_, bool fill = false) : n(n_), blocks((n_ + 63) / 64), b(blocks, fill ? ~0ULL : 0ULL) {
        if (fill) trim();
    }
    void trim() {
        if (n == 0) return;
        int rem = n % 64;
        if (rem) b[blocks - 1] &= ((1ULL << rem) - 1);
    }
    void set(int pos) { b[pos >> 6] |= 1ULL << (pos & 63); }
    void reset(int pos) { b[pos >> 6] &= ~(1ULL << (pos & 63)); }
    bool test(int pos) const { return (b[pos >> 6] >> (pos & 63)) & 1ULL; }
    bool any() const {
        for (ull x : b) if (x) return true;
        return false;
    }
    int first() const {
        for (int i = 0; i < blocks; ++i) {
            ull x = b[i];
            if (x) {
                return i * 64 + std::countr_zero(x);
            }
        }
        return -1;
    }
    Bitset intersect(const Bitset& other) const {
        Bitset r(n);
        for (int i = 0; i < blocks; ++i) r.b[i] = b[i] & other.b[i];
        return r;
    }
};

int N;
vector<Bitset> adj; // adjacency of complement graph
vector<int> best;

void colorSort(const Bitset& P, vector<int>& order, vector<int>& colors) {
    order.clear();
    colors.clear();
    Bitset remaining = P;
    int color = 0;
    while (remaining.any()) {
        ++color;
        Bitset available = remaining;
        while (available.any()) {
            int v = available.first();
            order.push_back(v);
            colors.push_back(color);
            remaining.reset(v);
            available.reset(v);
            // remove neighbors from available
            for (int i = 0; i < available.blocks; ++i) {
                available.b[i] &= ~adj[v].b[i];
            }
        }
    }
}

void expand(Bitset P, vector<int>& current) {
    if (!P.any()) {
        if (current.size() > best.size()) best = current;
        return;
    }
    vector<int> order, colors;
    colorSort(P, order, colors);

    for (int i = (int)order.size() - 1; i >= 0; --i) {
        if ((int)current.size() + colors[i] <= (int)best.size()) return;
        int v = order[i];

        current.push_back(v);
        Bitset P2 = P.intersect(adj[v]);
        expand(P2, current);
        current.pop_back();

        P.reset(v);
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int m;
    if (!(cin >> N >> m)) return 0;

    if (N == 0) {
        cout << 0 << "\n\n";
        return 0;
    }

    vector<Bitset> g(N, Bitset(N));
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue;
        g[u].set(v);
        g[v].set(u);
    }

    Bitset full(N, true);
    adj.assign(N, Bitset(N));
    for (int i = 0; i < N; ++i) {
        adj[i] = full;
        adj[i].reset(i);
        for (int k = 0; k < adj[i].blocks; ++k) {
            adj[i].b[k] &= ~g[i].b[k];
        }
    }

    vector<int> current;
    expand(full, current);

    sort(best.begin(), best.end());
    cout << best.size() << "\n";
    for (size_t i = 0; i < best.size(); ++i) {
        if (i) cout << ' ';
        cout << best[i];
    }
    cout << "\n";
    return 0;
}
```

This program:
- Reads the graph.
- Builds the complement graph.
- Finds the maximum clique (exact), which is the maximum independent set in the original graph.
- Prints the size and vertices.
