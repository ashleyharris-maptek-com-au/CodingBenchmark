const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
if (input.length === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  return;
}
const data = JSON.parse(input);

const depot = data.depot || [0, 0];
const truckCount = Math.max(0, data.truckCount | 0);
const packages = Array.isArray(data.packages) ? data.packages : [];

const n = packages.length;
if (n === 0 || truckCount === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  return;
}

const k = Math.min(truckCount, n);

// compute angle and distance
const withAngle = packages.map(p => {
  const dx = p.x - depot[0];
  const dy = p.y - depot[1];
  return {
    id: p.id,
    angle: Math.atan2(dy, dx),
    dist: Math.hypot(dx, dy)
  };
});

// sort by angle then distance
withAngle.sort((a, b) => {
  if (a.angle !== b.angle) return a.angle - b.angle;
  return a.dist - b.dist;
});

// partition into k routes (consecutive chunks)
const routes = [];
const size = Math.ceil(n / k);
for (let i = 0; i < k; i++) {
  const start = i * size;
  if (start >= n) break;
  const end = Math.min(start + size, n);
  const route = [];
  for (let j = start; j < end; j++) {
    route.push(withAngle[j].id);
  }
  routes.push(route);
}

process.stdout.write(JSON.stringify({ routes }));
