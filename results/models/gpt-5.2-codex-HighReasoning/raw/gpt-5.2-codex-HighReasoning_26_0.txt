use std::cmp::min;
use std::io::{self, Read, Write, BufWriter};

#[derive(Clone, Copy)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

impl Point {
    fn zero() -> Self {
        Point { x: 0.0, y: 0.0, z: 0.0 }
    }
}

fn dist2(a: &Point, b: &Point) -> f32 {
    let dx = a.x - b.x;
    let dy = a.y - b.y;
    let dz = a.z - b.z;
    dx * dx + dy * dy + dz * dz
}

struct Input<R: Read> {
    reader: R,
    buf: [u8; 1 << 20],
    pos: usize,
    cap: usize,
}

impl<R: Read> Input<R> {
    fn new(reader: R) -> Self {
        Input {
            reader,
            buf: [0; 1 << 20],
            pos: 0,
            cap: 0,
        }
    }

    fn fill(&mut self) -> bool {
        self.cap = self.reader.read(&mut self.buf).unwrap_or(0);
        self.pos = 0;
        self.cap > 0
    }

    fn read_byte(&mut self) -> Option<u8> {
        if self.pos >= self.cap {
            if !self.fill() {
                return None;
            }
        }
        let b = self.buf[self.pos];
        self.pos += 1;
        Some(b)
    }

    fn read_u64(&mut self) -> Option<u64> {
        let mut b;
        loop {
            b = self.read_byte()?;
            if !b.is_ascii_whitespace() {
                break;
            }
        }
        let mut num = 0u64;
        while !b.is_ascii_whitespace() {
            if b < b'0' || b > b'9' {
                return None;
            }
            num = num * 10 + (b - b'0') as u64;
            match self.read_byte() {
                Some(nb) => b = nb,
                None => return Some(num),
            }
        }
        Some(num)
    }

    fn read_exact(&mut self, out: &mut [u8]) -> bool {
        let mut filled = 0;
        while filled < out.len() {
            if self.pos >= self.cap {
                self.cap = self.reader.read(&mut self.buf).unwrap_or(0);
                self.pos = 0;
                if self.cap == 0 {
                    return false;
                }
            }
            let avail = self.cap - self.pos;
            let to_copy = min(avail, out.len() - filled);
            out[filled..filled + to_copy]
                .copy_from_slice(&self.buf[self.pos..self.pos + to_copy]);
            self.pos += to_copy;
            filled += to_copy;
        }
        true
    }

    fn read_point(&mut self) -> Option<Point> {
        let mut bytes = [0u8; 12];
        if !self.read_exact(&mut bytes) {
            return None;
        }
        let x = f32::from_le_bytes([bytes[0], bytes[1], bytes[2], bytes[3]]);
        let y = f32::from_le_bytes([bytes[4], bytes[5], bytes[6], bytes[7]]);
        let z = f32::from_le_bytes([bytes[8], bytes[9], bytes[10], bytes[11]]);
        Some(Point { x, y, z })
    }
}

fn kmeans(points: &Vec<Point>, k: usize, max_iter: usize) -> Vec<u32> {
    let n = points.len();
    if n == 0 || k == 0 {
        return Vec::new();
    }

    let mut centroids = Vec::with_capacity(k);
    for i in 0..k {
        centroids.push(points[i % n]);
    }

    let mut assignments = vec![0u32; n];

    for _ in 0..max_iter {
        let mut changed = false;
        let mut sums = vec![Point::zero(); k];
        let mut counts = vec![0u32; k];

        for (idx, p) in points.iter().enumerate() {
            let mut best = 0usize;
            let mut best_dist = dist2(p, &centroids[0]);
            for c in 1..k {
                let d = dist2(p, &centroids[c]);
                if d < best_dist {
                    best_dist = d;
                    best = c;
                }
            }
            if assignments[idx] != best as u32 {
                changed = true;
                assignments[idx] = best as u32;
            }
            let s = &mut sums[best];
            s.x += p.x;
            s.y += p.y;
            s.z += p.z;
            counts[best] += 1;
        }

        for c in 0..k {
            if counts[c] > 0 {
                let inv = 1.0 / counts[c] as f32;
                centroids[c].x = sums[c].x * inv;
                centroids[c].y = sums[c].y * inv;
                centroids[c].z = sums[c].z * inv;
            } else {
                centroids[c] = points[c % n];
            }
        }

        if !changed {
            break;
        }
    }

    assignments
}

fn streaming_kmeans<R: Read, W: Write>(
    input: &mut Input<R>,
    n_points: u64,
    k: usize,
    writer: &mut W,
) {
    if k == 0 || n_points == 0 {
        return;
    }

    let batch_size = if n_points <= 100_000_000 {
        100_000
    } else if n_points <= 1_000_000_000 {
        50_000
    } else {
        20_000
    };

    let mut centroids = Vec::with_capacity(k);
    let mut counts = vec![0u64; k];

    let init = min(k as u64, n_points) as usize;
    let mut out_buf: Vec<u8> = Vec::with_capacity(init * 4);

    for i in 0..init {
        if let Some(p) = input.read_point() {
            centroids.push(p);
            counts[i] = 1;
            out_buf.extend_from_slice(&(i as u32).to_le_bytes());
        } else {
            break;
        }
    }

    if !out_buf.is_empty() {
        writer.write_all(&out_buf).unwrap();
        out_buf.clear();
    }

    if centroids.is_empty() {
        return;
    }
    while centroids.len() < k {
        let idx = centroids.len() % init;
        centroids.push(centroids[idx]);
    }

    let mut processed = init as u64;
    let mut sums = vec![Point::zero(); k];
    let mut counts_batch = vec![0u32; k];
    let mut assignments: Vec<u32> = Vec::with_capacity(batch_size);
    let mut batch_points: Vec<Point> = Vec::with_capacity(batch_size);

    while processed < n_points {
        let to_read = min(batch_size as u64, n_points - processed) as usize;
        batch_points.clear();

        for _ in 0..to_read {
            if let Some(p) = input.read_point() {
                batch_points.push(p);
            } else {
                break;
            }
        }

        if batch_points.is_empty() {
            break;
        }

        sums.fill(Point::zero());
        counts_batch.fill(0);
        assignments.clear();

        for p in &batch_points {
            let mut best = 0usize;
            let mut best_dist = dist2(p, &centroids[0]);
            for i in 1..k {
                let d = dist2(p, &centroids[i]);
                if d < best_dist {
                    best_dist = d;
                    best = i;
                }
            }
            assignments.push(best as u32);

            let s = &mut sums[best];
            s.x += p.x;
            s.y += p.y;
            s.z += p.z;
            counts_batch[best] += 1;
        }

        for i in 0..k {
            let cb = counts_batch[i] as u64;
            if cb > 0 {
                let total = counts[i] + cb;
                let inv_total = 1.0 / total as f32;
                let c = &mut centroids[i];
                c.x = (c.x * counts[i] as f32 + sums[i].x) * inv_total;
                c.y = (c.y * counts[i] as f32 + sums[i].y) * inv_total;
                c.z = (c.z * counts[i] as f32 + sums[i].z) * inv_total;
                counts[i] = total;
            }
        }

        out_buf.clear();
        out_buf.reserve(assignments.len() * 4);
        for &a in &assignments {
            out_buf.extend_from_slice(&a.to_le_bytes());
        }
        writer.write_all(&out_buf).unwrap();

        processed += batch_points.len() as u64;
    }
}

fn main() {
    const SMALL_THRESHOLD: u64 = 10_000_000;

    let stdin = io::stdin();
    let mut input = Input::new(stdin.lock());

    let n_points = match input.read_u64() {
        Some(v) => v,
        None => return,
    };
    let k = match input.read_u64() {
        Some(v) => v as usize,
        None => return,
    };

    if n_points == 0 || k == 0 {
        return;
    }

    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());

    if n_points <= SMALL_THRESHOLD {
        let n = n_points as usize;
        let mut points = Vec::with_capacity(n);
        for _ in 0..n {
            if let Some(p) = input.read_point() {
                points.push(p);
            } else {
                break;
            }
        }
        let assignments = kmeans(&points, k, 20);
        let mut out_buf = Vec::with_capacity(assignments.len() * 4);
        for a in assignments {
            out_buf.extend_from_slice(&(a as u32).to_le_bytes());
        }
        writer.write_all(&out_buf).unwrap();
    } else {
        streaming_kmeans(&mut input, n_points, k, &mut writer);
    }

    writer.flush().unwrap();
}
