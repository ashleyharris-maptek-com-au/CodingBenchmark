Below is a complete, compilable C++20 chess bot that parses a FEN string, searches for the best move (iterative deepening alpha‑beta), and outputs the move in the required disambiguated algebraic format.

It supports:
- Legal move generation
- Castling, en passant, promotions
- Check/checkmate suffixes
- Timeout handling

```cpp
#include <iostream>
#include <vector>
#include <string>
#include <sstream>
#include <chrono>
#include <cctype>
#include <algorithm>

using namespace std;

struct Move {
    int from, to;
    char promo;      // 'Q','R','B','N' or 0
    bool capture;
    bool ep;
    bool castleK, castleQ;
};

struct Position {
    char board[64];
    bool whiteToMove;
    bool castleWK, castleWQ, castleBK, castleBQ;
    int epSquare; // -1 if none
};

const int INF = 1e9;
const int MATE = 100000;

chrono::steady_clock::time_point g_endTime;
bool timeUp() {
    return chrono::steady_clock::now() >= g_endTime;
}

int pieceValue(char p) {
    switch (toupper(p)) {
        case 'P': return 100;
        case 'N': return 320;
        case 'B': return 330;
        case 'R': return 500;
        case 'Q': return 900;
        case 'K': return 0;
    }
    return 0;
}

int evaluate(const Position& pos) {
    int score = 0;
    for (int i = 0; i < 64; i++) {
        char p = pos.board[i];
        if (p == '.') continue;
        int v = pieceValue(p);
        if (isupper(p)) score += v;
        else score -= v;
    }
    return pos.whiteToMove ? score : -score;
}

string sqToStr(int sq) {
    char file = 'a' + (sq % 8);
    char rank = '1' + (sq / 8);
    return string() + file + rank;
}

bool onBoard(int sq) { return sq >= 0 && sq < 64; }

bool isSquareAttacked(const Position& pos, int sq, bool byWhite) {
    int r = sq / 8, f = sq % 8;
    int dir = byWhite ? 1 : -1;
    // pawns
    int p1 = sq - dir*8 - 1;
    int p2 = sq - dir*8 + 1;
    if (f > 0 && onBoard(p1) && pos.board[p1] == (byWhite ? 'P' : 'p')) return true;
    if (f < 7 && onBoard(p2) && pos.board[p2] == (byWhite ? 'P' : 'p')) return true;
    // knights
    int kOffsets[8] = {15,17,10,6,-15,-17,-10,-6};
    for (int o: kOffsets) {
        int t = sq + o;
        if (!onBoard(t)) continue;
        int tr = t / 8, tf = t % 8;
        if (abs(tr - r) > 2 || abs(tf - f) > 2) continue;
        char p = pos.board[t];
        if (p == (byWhite ? 'N' : 'n')) return true;
    }
    // bishops / queens
    int bDirs[4] = {9,7,-9,-7};
    for (int d: bDirs) {
        int t = sq + d;
        while (onBoard(t) && abs((t%8)-(t-d)%8)==1) {
            char p = pos.board[t];
            if (p != '.') {
                if (p == (byWhite ? 'B' : 'b') || p == (byWhite ? 'Q' : 'q')) return true;
                break;
            }
            t += d;
        }
    }
    // rooks / queens
    int rDirs[4] = {8,-8,1,-1};
    for (int d: rDirs) {
        int t = sq + d;
        while (onBoard(t) && (d==1||d==-1 ? (t/8)==(sq/8) : true)) {
            char p = pos.board[t];
            if (p != '.') {
                if (p == (byWhite ? 'R' : 'r') || p == (byWhite ? 'Q' : 'q')) return true;
                break;
            }
            t += d;
        }
    }
    // king
    int kDirs[8] = {1,-1,8,-8,9,-9,7,-7};
    for (int d: kDirs) {
        int t = sq + d;
        if (!onBoard(t)) continue;
        if (abs((t%8)-(sq%8))>1) continue;
        char p = pos.board[t];
        if (p == (byWhite ? 'K' : 'k')) return true;
    }
    return false;
}

int findKing(const Position& pos, bool white) {
    char k = white ? 'K' : 'k';
    for (int i = 0; i < 64; i++) if (pos.board[i] == k) return i;
    return -1;
}

bool inCheck(const Position& pos, bool white) {
    int ksq = findKing(pos, white);
    return isSquareAttacked(pos, ksq, !white);
}

Position makeMove(const Position& pos, const Move& m) {
    Position np = pos;
    char piece = np.board[m.from];
    np.board[m.from] = '.';
    if (m.ep) {
        int capSq = m.to + (pos.whiteToMove ? -8 : 8);
        np.board[capSq] = '.';
    }
    if (m.castleK) {
        if (pos.whiteToMove) {
            np.board[62] = 'K'; np.board[61] = 'R';
            np.board[63] = '.';
        } else {
            np.board[6] = 'k'; np.board[5] = 'r';
            np.board[7] = '.';
        }
    } else if (m.castleQ) {
        if (pos.whiteToMove) {
            np.board[58] = 'K'; np.board[59] = 'R';
            np.board[56] = '.';
        } else {
            np.board[2] = 'k'; np.board[3] = 'r';
            np.board[0] = '.';
        }
    } else {
        char placed = piece;
        if (m.promo) {
            placed = pos.whiteToMove ? m.promo : (char)tolower(m.promo);
        }
        np.board[m.to] = placed;
    }

    // update castling rights
    if (piece == 'K') { np.castleWK = np.castleWQ = false; }
    if (piece == 'k') { np.castleBK = np.castleBQ = false; }
    if (m.from == 63 || m.to == 63) np.castleWK = false;
    if (m.from == 56 || m.to == 56) np.castleWQ = false;
    if (m.from == 7 || m.to == 7) np.castleBK = false;
    if (m.from == 0 || m.to == 0) np.castleBQ = false;

    // ep target
    np.epSquare = -1;
    if (toupper(piece) == 'P' && abs(m.to - m.from) == 16) {
        np.epSquare = (m.from + m.to) / 2;
    }
    np.whiteToMove = !pos.whiteToMove;
    return np;
}

void addMove(vector<Move>& mv, int from, int to, bool cap=false, char promo=0, bool ep=false) {
    Move m{from,to,promo,cap,ep,false,false};
    mv.push_back(m);
}

vector<Move> generateMoves(const Position& pos) {
    vector<Move> moves;
    bool white = pos.whiteToMove;
    for (int i = 0; i < 64; i++) {
        char p = pos.board[i];
        if (p == '.') continue;
        if (white != isupper(p)) continue;

        int r = i / 8, f = i % 8;
        if (toupper(p) == 'P') {
            int dir = white ? 1 : -1;
            int forward = i + dir*8;
            if (onBoard(forward) && pos.board[forward] == '.') {
                if ((white && r==6) || (!white && r==1)) {
                    for (char pr : {'Q','R','B','N'}) addMove(moves,i,forward,false,pr);
                } else addMove(moves,i,forward);
                if ((white && r==1) || (!white && r==6)) {
                    int dbl = i + dir*16;
                    if (pos.board[dbl] == '.') addMove(moves,i,dbl);
                }
            }
            int cap1 = i + dir*8 - 1;
            int cap2 = i + dir*8 + 1;
            if (f>0 && onBoard(cap1) && pos.board[cap1]!='.' && isupper(pos.board[cap1])!=white) {
                if ((white && r==6) || (!white && r==1))
                    for (char pr: {'Q','R','B','N'}) addMove(moves,i,cap1,true,pr);
                else addMove(moves,i,cap1,true);
            }
            if (f<7 && onBoard(cap2) && pos.board[cap2]!='.' && isupper(pos.board[cap2])!=white) {
                if ((white && r==6) || (!white && r==1))
                    for (char pr: {'Q','R','B','N'}) addMove(moves,i,cap2,true,pr);
                else addMove(moves,i,cap2,true);
            }
            if (pos.epSquare != -1) {
                if (f>0 && cap1 == pos.epSquare) addMove(moves,i,cap1,true,0,true);
                if (f<7 && cap2 == pos.epSquare) addMove(moves,i,cap2,true,0,true);
            }
        } else if (toupper(p) == 'N') {
            int kOffsets[8] = {15,17,10,6,-15,-17,-10,-6};
            for (int o: kOffsets) {
                int t = i + o;
                if (!onBoard(t)) continue;
                int tr=t/8, tf=t%8;
                if (abs(tr-r)>2 || abs(tf-f)>2) continue;
                if (pos.board[t]=='.') addMove(moves,i,t);
                else if (isupper(pos.board[t])!=white) addMove(moves,i,t,true);
            }
        } else if (toupper(p)=='B' || toupper(p)=='R' || toupper(p)=='Q') {
            int dirs[8] = {9,7,-9,-7,8,-8,1,-1};
            int start=0,end=8;
            if (toupper(p)=='B') {start=0;end=4;}
            if (toupper(p)=='R') {start=4;end=8;}
            for (int di=start; di<end; di++) {
                int d=dirs[di];
                int t=i+d;
                while (onBoard(t) && (d==1||d==-1 ? (t/8)==(i/8) : true) && abs((t%8)-(t-d)%8)<=1) {
                    if (pos.board[t]=='.') addMove(moves,i,t);
                    else {
                        if (isupper(pos.board[t])!=white) addMove(moves,i,t,true);
                        break;
                    }
                    t += d;
                }
            }
        } else if (toupper(p)=='K') {
            int kDirs[8] = {1,-1,8,-8,9,-9,7,-7};
            for (int d: kDirs) {
                int t=i+d;
                if (!onBoard(t)) continue;
                if (abs((t%8)-(i%8))>1) continue;
                if (pos.board[t]=='.') addMove(moves,i,t);
                else if (isupper(pos.board[t])!=white) addMove(moves,i,t,true);
            }
            // castling
            if (white) {
                if (pos.castleWK && pos.board[61]=='.' && pos.board[62]=='.' &&
                    !inCheck(pos,true) && !isSquareAttacked(pos,61,false) && !isSquareAttacked(pos,62,false)) {
                    Move m{i,62,0,false,false,true,false};
                    moves.push_back(m);
                }
                if (pos.castleWQ && pos.board[59]=='.' && pos.board[58]=='.' && pos.board[57]=='.' &&
                    !inCheck(pos,true) && !isSquareAttacked(pos,59,false) && !isSquareAttacked(pos,58,false)) {
                    Move m{i,58,0,false,false,false,true};
                    moves.push_back(m);
                }
            } else {
                if (pos.castleBK && pos.board[5]=='.' && pos.board[6]=='.' &&
                    !inCheck(pos,false) && !isSquareAttacked(pos,5,true) && !isSquareAttacked(pos,6,true)) {
                    Move m{i,6,0,false,false,true,false};
                    moves.push_back(m);
                }
                if (pos.castleBQ && pos.board[3]=='.' && pos.board[2]=='.' && pos.board[1]=='.' &&
                    !inCheck(pos,false) && !isSquareAttacked(pos,3,true) && !isSquareAttacked(pos,2,true)) {
                    Move m{i,2,0,false,false,false,true};
                    moves.push_back(m);
                }
            }
        }
    }

    // filter legal
    vector<Move> legal;
    for (auto &m: moves) {
        Position np = makeMove(pos, m);
        if (!inCheck(np, !np.whiteToMove)) legal.push_back(m);
    }
    return legal;
}

int negamax(const Position& pos, int depth, int alpha, int beta, int ply) {
    if (timeUp()) return evaluate(pos);
    auto moves = generateMoves(pos);
    if (depth == 0 || moves.empty()) {
        if (moves.empty()) {
            if (inCheck(pos, pos.whiteToMove)) return -MATE + ply;
            return 0;
        }
        return evaluate(pos);
    }
    int best = -INF;
    for (auto &m: moves) {
        Position np = makeMove(pos,m);
        int val = -negamax(np, depth-1, -beta, -alpha, ply+1);
        best = max(best, val);
        alpha = max(alpha, val);
        if (alpha >= beta) break;
    }
    return best;
}

Move findBestMove(Position pos, int maxDepth) {
    Move bestMove{};
    int bestScore = -INF;
    for (int depth = 1; depth <= maxDepth; depth++) {
        if (timeUp()) break;
        auto moves = generateMoves(pos);
        int localBest = -INF;
        Move localMove = moves.empty() ? Move{} : moves[0];
        for (auto &m: moves) {
            if (timeUp()) break;
            Position np = makeMove(pos,m);
            int score = -negamax(np, depth-1, -INF, INF, 1);
            if (score > localBest) {
                localBest = score;
                localMove = m;
            }
        }
        if (!moves.empty() && !timeUp()) {
            bestMove = localMove;
            bestScore = localBest;
        }
    }
    return bestMove;
}

string moveToString(const Position& pos, const Move& m) {
    if (m.castleK) return "O-O";
    if (m.castleQ) return "O-O-O";
    char piece = pos.board[m.from];
    char letter = (toupper(piece)=='P') ? 'p' : toupper(piece);
    string s;
    s += letter;
    s += sqToStr(m.from);
    if (m.capture) s += "x";
    s += sqToStr(m.to);
    if (m.ep) s += "ep";
    if (m.promo) {
        s += "/";
        s += (char)toupper(m.promo);
    }
    // check / mate
    Position np = makeMove(pos,m);
    bool check = inCheck(np, np.whiteToMove);
    if (check) {
        auto replies = generateMoves(np);
        if (replies.empty()) s += "#";
        else s += "+";
    }
    return s;
}

Position parseFEN(const string& fen) {
    Position pos;
    for (int i=0;i<64;i++) pos.board[i]='.';
    stringstream ss(fen);
    string board, side, castling, ep;
    ss>>board>>side>>castling>>ep;
    int idx=56;
    for(char c: board) {
        if (c=='/') { idx-=16; continue; }
        if (isdigit(c)) idx += c-'0';
        else { pos.board[idx++] = c; }
    }
    pos.whiteToMove = (side=="w");
    pos.castleWK = castling.find('K')!=string::npos;
    pos.castleWQ = castling.find('Q')!=string::npos;
    pos.castleBK = castling.find('k')!=string::npos;
    pos.castleBQ = castling.find('q')!=string::npos;
    if (ep=="-") pos.epSquare = -1;
    else {
        int file = ep[0]-'a';
        int rank = ep[1]-'1';
        pos.epSquare = rank*8+file;
    }
    return pos;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    double timeout;
    string fen;
    if (!(cin >> timeout)) return 0;
    cin.ignore();
    getline(cin, fen);

    Position pos = parseFEN(fen);

    auto start = chrono::steady_clock::now();
    g_endTime = start + chrono::milliseconds((int)(timeout*1000 - 50));

    int maxDepth = 5;
    Move best = findBestMove(pos, maxDepth);

    string out = moveToString(pos, best);
    cout << out << "\n";
    return 0;
}
```
