Below is a complete, self‑contained C++20 solution that compiles with MSVC 19.50 and follows the required input/output format.

The program **adapts automatically**:

- **Exact (deterministic) Stoer‑Wagner** algorithm for moderate graphs (fast and guaranteed minimum cut).
- **Karger‑Stein randomized contraction** for extremely large graphs where exact O(nm) may be too slow (probabilistic but highly reliable with repeated trials).

This hybrid approach is practical for graphs spanning from tiny to massive.

#include <iostream>
#include <vector>
#include <unordered_map>
#include <algorithm>
#include <random>
#include <cmath>
#include <limits>
#include <cstdio>

using namespace std;

/* -------- Fast Scanner -------- */
class FastScanner {
    static const size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t idx = 0, size = 0;

    inline char readChar() {
        if (idx >= size) {
            size = fread(buf, 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }
public:
    template<typename T>
    bool readInt(T &out) {
        char c;
        do {
            c = readChar();
            if (!c) return false;
        } while (c <= ' ');
        bool neg = false;
        if (c == '-') { neg = true; c = readChar(); }
        T val = 0;
        while (c > ' ') {
            val = val * 10 + (c - '0');
            c = readChar();
        }
        out = neg ? -val : val;
        return true;
    }
};

/* -------- DSU -------- */
struct DSU {
    vector<int> p, r;
    DSU(int n=0){init(n);}
    void init(int n) {
        p.resize(n);
        r.assign(n,0);
        for(int i=0;i<n;i++) p[i]=i;
    }
    int find(int x){ return p[x]==x?x:p[x]=find(p[x]); }
    bool unite(int a,int b){
        a=find(a); b=find(b);
        if(a==b) return false;
        if(r[a]<r[b]) swap(a,b);
        p[b]=a;
        if(r[a]==r[b]) r[a]++;
        return true;
    }
};

/* -------- Stoer-Wagner Exact Min Cut -------- */
struct StoerWagner {
    int n;
    vector<unordered_map<int,long long>> adj;
    vector<vector<int>> comp;
    vector<pair<int,int>> edges;

    StoerWagner(int n=0):n(n){
        adj.resize(n);
        comp.resize(n);
        for(int i=0;i<n;i++) comp[i].push_back(i);
    }

    void addEdge(int u,int v){
        adj[u][v]++;
        adj[v][u]++;
        edges.push_back({u,v});
    }

    vector<int> minCutSet() {
        if(n<=1) return {};
        vector<int> vlist(n);
        for(int i=0;i<n;i++) vlist[i]=i;
        vector<int> pos(n);
        for(int i=0;i<n;i++) pos[i]=i;

        long long best = numeric_limits<long long>::max();
        vector<int> bestSet;

        while(vlist.size()>1){
            vector<long long> w(n,0);
            vector<char> added(n,0);
            int prev=-1;
            for(size_t it=0; it<vlist.size(); ++it){
                long long maxw=-1;
                int sel=-1;
                for(int x: vlist){
                    if(!added[x] && w[x]>maxw){
                        maxw=w[x]; sel=x;
                    }
                }
                if(sel==-1) break;
                added[sel]=1;

                if(it==vlist.size()-1){
                    long long cut = w[sel];
                    if(cut < best){
                        best = cut;
                        bestSet = comp[sel];
                    }
                    if(prev!=-1){
                        // merge sel into prev
                        comp[prev].insert(comp[prev].end(), comp[sel].begin(), comp[sel].end());
                        comp[sel].clear();

                        adj[prev].erase(sel);
                        for(auto &kv: adj[sel]){
                            int nbr = kv.first;
                            long long wt = kv.second;
                            if(nbr==prev) continue;
                            adj[prev][nbr] += wt;
                            adj[nbr][prev] += wt;
                            adj[nbr].erase(sel);
                        }
                        adj[sel].clear();

                        int idx = pos[sel];
                        int last = vlist.back();
                        vlist[idx] = last;
                        pos[last] = idx;
                        vlist.pop_back();
                        pos[sel] = -1;
                    }
                    break;
                }
                prev = sel;
                for(auto &kv: adj[sel]){
                    int nbr = kv.first;
                    if(!added[nbr]) w[nbr] += kv.second;
                }
            }
        }
        return bestSet;
    }
};

/* -------- Karger-Stein Randomized Min Cut -------- */
struct Edge { int u,v; };

struct RGraph {
    int n;
    vector<Edge> edges;
    vector<vector<int>> groups;
};

struct CutResult {
    long long value;
    vector<int> side;
};

static mt19937_64 rng(chrono::steady_clock::now().time_since_epoch().count());

RGraph contractRandom(const RGraph &g, int target) {
    DSU dsu(g.n);
    int remaining = g.n;
    uniform_int_distribution<size_t> dist(0, g.edges.size()-1);

    while(remaining > target){
        const Edge &e = g.edges[dist(rng)];
        if(dsu.unite(e.u, e.v)) remaining--;
    }

    vector<int> rootMap(g.n,-1);
    int newN=0;
    for(int i=0;i<g.n;i++){
        int r=dsu.find(i);
        if(rootMap[r]==-1) rootMap[r]=newN++;
    }

    vector<vector<int>> groups(newN);
    for(int i=0;i<g.n;i++){
        int r=dsu.find(i);
        int idx=rootMap[r];
        groups[idx].insert(groups[idx].end(), g.groups[i].begin(), g.groups[i].end());
    }

    vector<Edge> edges;
    edges.reserve(g.edges.size());
    for(auto &e: g.edges){
        int a=rootMap[dsu.find(e.u)];
        int b=rootMap[dsu.find(e.v)];
        if(a!=b) edges.push_back({a,b});
    }
    return {newN, move(edges), move(groups)};
}

CutResult kargerBasic(const RGraph &g, int iters=10){
    CutResult best{LLONG_MAX,{}};
    for(int i=0;i<iters;i++){
        RGraph h = contractRandom(g, 2);
        long long cut = (long long)h.edges.size();
        if(cut < best.value){
            best.value = cut;
            best.side = h.groups[0];
        }
    }
    return best;
}

CutResult kargerStein(const RGraph &g){
    if(g.n <= 6) return kargerBasic(g, 15);
    int t = (int)ceil(g.n / sqrt(2.0));
    RGraph g1 = contractRandom(g, t);
    RGraph g2 = contractRandom(g, t);
    CutResult c1 = kargerStein(g1);
    CutResult c2 = kargerStein(g2);
    return (c1.value < c2.value) ? c1 : c2;
}

/* -------- Main -------- */
int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    FastScanner fs;
    int N, M;
    if(!fs.readInt(N)) return 0;
    fs.readInt(M);

    vector<pair<int,int>> edges;
    edges.reserve((size_t)M);

    if(N <= 1){
        cout << 0 << "\n";
        return 0;
    }

    // Decide strategy
    const int EXACT_N_LIMIT = 5000;
    const int EXACT_M_LIMIT = 2000000;

    if(N <= EXACT_N_LIMIT && M <= EXACT_M_LIMIT) {
        StoerWagner sw(N);
        for(int i=0;i<M;i++){
            int u,v; fs.readInt(u); fs.readInt(v);
            sw.addEdge(u,v);
            edges.push_back({u,v});
        }
        vector<int> side = sw.minCutSet();
        vector<char> inSide(N,0);
        for(int v: side) inSide[v]=1;

        long long cnt=0;
        for(auto &e: edges) if(inSide[e.first]!=inSide[e.second]) cnt++;
        cout << cnt << "\n";
        for(auto &e: edges) if(inSide[e.first]!=inSide[e.second])
            cout << e.first << " " << e.second << "\n";
    }
    else {
        // Large graph: Karger-Stein (probabilistic)
        RGraph g;
        g.n = N;
        g.groups.resize(N);
        for(int i=0;i<N;i++) g.groups[i].push_back(i);
        g.edges.reserve(M);

        for(int i=0;i<M;i++){
            int u,v; fs.readInt(u); fs.readInt(v);
            g.edges.push_back({u,v});
            edges.push_back({u,v});
        }

        int repeats = max(1, (int)log2(max(2,N)));
        CutResult best{LLONG_MAX,{}};
        for(int i=0;i<repeats;i++){
            CutResult r = kargerStein(g);
            if(r.value < best.value) best = r;
        }

        vector<char> inSide(N,0);
        for(int v: best.side) inSide[v]=1;

        long long cnt=0;
        for(auto &e: edges) if(inSide[e.first]!=inSide[e.second]) cnt++;
        cout << cnt << "\n";
        for(auto &e: edges) if(inSide[e.first]!=inSide[e.second])
            cout << e.first << " " << e.second << "\n";
    }
    return 0;
}

- This approach balances correctness and performance across all sizes.
