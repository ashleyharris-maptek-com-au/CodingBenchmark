Below is a complete, compilable C# solution that uses:

- **Exact maximum clique search** with bitset + Tomita‑style coloring for **small/medium graphs**  
- **Exact degeneracy‑based backtracking** for **large sparse graphs**  
- **Fast input**, adjacency lists, sorting, greedy lower bound  
- Works across sparse to dense graphs, with automatic strategy selection  

> The algorithm is exact. For huge graphs, it relies on sparsity (low degeneracy) to stay fast.

```csharp
using System;
using System.IO;
using System.Collections.Generic;
using System.Linq;

public class Program
{
    const int BITSET_THRESHOLD = 5000; // switch to bitset-based algorithm

    static int n, m;
    static int[][] adjList;
    static int[] degree;

    // Bitset graph for small N
    static ulong[][] adjBits;
    static int bitWords;

    // Best clique found
    static int bestSize = 0;
    static List<int> bestClique = new List<int>();

    public static void Main()
    {
        var fs = new FastScanner();
        int nn = fs.NextInt();
        if (nn == int.MinValue) return;
        int mm = fs.NextInt();
        n = nn; m = mm;

        if (n <= 0)
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }

        // Read edges
        int[] eu = new int[m];
        int[] ev = new int[m];
        degree = new int[n];

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if (u == v) { i--; continue; } // ignore self-loops
            eu[i] = u;
            ev[i] = v;
            degree[u]++;
            degree[v]++;
        }

        // Build adjacency lists
        adjList = new int[n][];
        int[] idx = new int[n];
        for (int i = 0; i < n; i++)
            adjList[i] = new int[degree[i]];

        for (int i = 0; i < m; i++)
        {
            int u = eu[i];
            int v = ev[i];
            adjList[u][idx[u]++] = v;
            adjList[v][idx[v]++] = u;
        }
        for (int i = 0; i < n; i++)
            Array.Sort(adjList[i]);

        // Greedy lower bound
        List<int> greedy = GreedyClique();
        bestSize = greedy.Count;
        bestClique = new List<int>(greedy);

        if (bestSize == n)
        {
            Output();
            return;
        }

        if (n <= BITSET_THRESHOLD)
        {
            BuildBitsets();
            var all = new List<int>(n);
            for (int i = 0; i < n; i++) all.Add(i);
            var R = new List<int>();
            ExpandBit(all, R);
        }
        else
        {
            // Large sparse graph -> degeneracy ordering + exact backtracking
            int[] order = DegeneracyOrder();
            int[] pos = new int[n];
            for (int i = 0; i < n; i++) pos[order[i]] = i;

            for (int i = 0; i < n; i++)
            {
                int v = order[i];
                var candidates = new List<int>();
                foreach (int u in adjList[v])
                {
                    if (pos[u] > i) candidates.Add(u);
                }
                if (candidates.Count + 1 <= bestSize) continue;
                var Q = new List<int> { v };
                ExpandList(candidates, Q);
            }
        }

        Output();
    }

    // ------- Output ----------
    static void Output()
    {
        Console.WriteLine(bestSize);
        if (bestSize == 0)
        {
            Console.WriteLine();
            return;
        }
        Console.WriteLine(string.Join(" ", bestClique));
    }

    // ------- Greedy clique (for initial lower bound) ----------
    static List<int> GreedyClique()
    {
        var verts = new List<int>(n);
        for (int i = 0; i < n; i++) verts.Add(i);
        verts.Sort((a, b) => degree[b].CompareTo(degree[a]));

        var clique = new List<int>();
        var candidates = verts;

        while (candidates.Count > 0)
        {
            int v = candidates[0];
            clique.Add(v);
            var next = new List<int>();
            for (int i = 1; i < candidates.Count; i++)
            {
                int u = candidates[i];
                if (IsAdjacentList(v, u)) next.Add(u);
            }
            candidates = next;
        }
        return clique;
    }

    // ------- Bitset build ----------
    static void BuildBitsets()
    {
        bitWords = (n + 63) >> 6;
        adjBits = new ulong[n][];
        for (int i = 0; i < n; i++)
            adjBits[i] = new ulong[bitWords];

        for (int v = 0; v < n; v++)
        {
            foreach (int u in adjList[v])
            {
                adjBits[v][u >> 6] |= 1UL << (u & 63);
            }
        }
    }

    // ------- Bitset algorithm (Tomita + coloring) ----------
    static void ExpandBit(List<int> P, List<int> R)
    {
        if (P.Count == 0)
        {
            if (R.Count > bestSize)
            {
                bestSize = R.Count;
                bestClique = new List<int>(R);
            }
            return;
        }

        ColorSort(P, out int[] order, out int[] colors);

        for (int i = P.Count - 1; i >= 0; i--)
        {
            if (R.Count + colors[i] <= bestSize) return;

            int v = order[i];
            var P2 = new List<int>();
            for (int j = 0; j < i; j++)
            {
                int u = order[j];
                if (IsAdjacentBit(v, u)) P2.Add(u);
            }

            R.Add(v);
            if (P2.Count == 0)
            {
                if (R.Count > bestSize)
                {
                    bestSize = R.Count;
                    bestClique = new List<int>(R);
                }
            }
            else
            {
                ExpandBit(P2, R);
            }
            R.RemoveAt(R.Count - 1);
        }
    }

    static void ColorSort(List<int> P, out int[] order, out int[] colors)
    {
        P.Sort((a, b) => degree[b].CompareTo(degree[a]));
        int m = P.Count;
        order = new int[m];
        colors = new int[m];

        var colorSets = new List<ulong[]>();

        for (int i = 0; i < m; i++)
        {
            int v = P[i];
            int c;
            for (c = 0; c < colorSets.Count; c++)
            {
                if (!Intersects(adjBits[v], colorSets[c])) break;
            }
            if (c == colorSets.Count)
                colorSets.Add(new ulong[bitWords]);

            colorSets[c][v >> 6] |= 1UL << (v & 63);
            order[i] = v;
            colors[i] = c + 1;
        }
    }

    static bool Intersects(ulong[] a, ulong[] b)
    {
        for (int i = 0; i < a.Length; i++)
            if ((a[i] & b[i]) != 0) return true;
        return false;
    }

    static bool IsAdjacentBit(int u, int v)
    {
        return (adjBits[u][v >> 6] & (1UL << (v & 63))) != 0;
    }

    // ------- Large sparse exact (degeneracy ordering + backtracking) ----------
    static void ExpandList(List<int> C, List<int> Q)
    {
        int ccount = C.Count;
        for (int i = 0; i < ccount; i++)
        {
            if (Q.Count + (ccount - i) <= bestSize) return;

            int v = C[i];

            var Cprime = new List<int>();
            int[] neigh = adjList[v];
            int p1 = i + 1;
            int p2 = 0;

            while (p1 < ccount && p2 < neigh.Length)
            {
                int a = C[p1];
                int b = neigh[p2];
                if (a == b)
                {
                    Cprime.Add(a);
                    p1++; p2++;
                }
                else if (a < b) p1++;
                else p2++;
            }

            Q.Add(v);
            if (Cprime.Count == 0)
            {
                if (Q.Count > bestSize)
                {
                    bestSize = Q.Count;
                    bestClique = new List<int>(Q);
                }
            }
            else
            {
                if (Q.Count + Cprime.Count > bestSize)
                    ExpandList(Cprime, Q);
            }
            Q.RemoveAt(Q.Count - 1);
        }
    }

    static int[] DegeneracyOrder()
    {
        int maxdeg = 0;
        for (int i = 0; i < n; i++) if (degree[i] > maxdeg) maxdeg = degree[i];

        int[] deg = (int[])degree.Clone();
        int[] bin = new int[maxdeg + 1];
        for (int i = 0; i < n; i++) bin[deg[i]]++;

        int start = 0;
        for (int d = 0; d <= maxdeg; d++)
        {
            int num = bin[d];
            bin[d] = start;
            start += num;
        }

        int[] pos = new int[n];
        int[] vert = new int[n];
        for (int v = 0; v < n; v++)
        {
            pos[v] = bin[deg[v]];
            vert[pos[v]] = v;
            bin[deg[v]]++;
        }

        for (int d = maxdeg; d >= 1; d--) bin[d] = bin[d - 1];
        bin[0] = 0;

        int[] order = new int[n];
        for (int i = 0; i < n; i++)
        {
            int v = vert[i];
            order[i] = v;
            foreach (int u in adjList[v])
            {
                if (deg[u] > 0)
                {
                    int du = deg[u];
                    int pu = pos[u];
                    int pw = bin[du];
                    int w = vert[pw];
                    if (u != w)
                    {
                        vert[pu] = w; vert[pw] = u;
                        pos[u] = pw; pos[w] = pu;
                    }
                    bin[du]++;
                    deg[u]--;
                }
            }
            deg[v] = 0;
        }
        return order;
    }

    static bool IsAdjacentList(int u, int v)
    {
        return Array.BinarySearch(adjList[u], v) >= 0;
    }

    // ------- FastScanner ----------
    public class FastScanner
    {
        private readonly Stream stream = Console.OpenStandardInput();
        private readonly byte[] buffer = new byte[1 << 16];
        private int len, ptr;

        private int Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return int.MinValue;
            } while (c <= 32);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > 32 && c != -1)
            {
                val = val * 10 + (c - 48);
                c = Read();
            }
            return val * sign;
        }
    }
}
```

**Notes:**
- Produces an **exact maximum clique**.
- Uses **bitset + coloring** for small/dense graphs.
- Uses **degeneracy ordering** for massive sparse graphs.
- Handles up to **1 million vertices** if graph remains sparse.

You can compile with:
```
csc Program.cs
```

And run:
```
Program.exe < input.txt
```
