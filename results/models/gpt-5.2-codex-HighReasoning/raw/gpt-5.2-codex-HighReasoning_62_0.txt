declare const require: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
if (!input || input.trim().length === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}
const data = JSON.parse(input);

const width: number = data.width | 0;
const height: number = data.height | 0;
const grid: string[] = data.grid;
const ignitions: number[][] = data.ignitions || [];
const towns: { id: any; x: number; y: number; value: number }[] = data.towns || [];
const cutBudget: number = data.cutBudget | 0;

const n = width * height;
if (n <= 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const passable = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row = grid[y];
  const base = y * width;
  for (let x = 0; x < width; x++) {
    if (row.charAt(x) === '.') passable[base + x] = 1;
  }
}

const townValue = new Float64Array(n);
const isTown = new Uint8Array(n);
for (const t of towns) {
  const x = t.x | 0;
  const y = t.y | 0;
  if (x >= 0 && x < width && y >= 0 && y < height) {
    const idx = y * width + x;
    townValue[idx] += +t.value;
    isTown[idx] = 1;
  }
}

const dist = new Int32Array(n);
dist.fill(-1);
const parent = new Int32Array(n);
parent.fill(-1);
const isIgnition = new Uint8Array(n);

const queue = new Int32Array(n);
let qh = 0, qt = 0;
const order = new Int32Array(n);
let orderLen = 0;
const roots: number[] = [];

for (const ig of ignitions) {
  const x = ig[0] | 0;
  const y = ig[1] | 0;
  if (x >= 0 && x < width && y >= 0 && y < height) {
    const idx = y * width + x;
    if (passable[idx] && dist[idx] === -1) {
      dist[idx] = 0;
      parent[idx] = -1;
      isIgnition[idx] = 1;
      queue[qt++] = idx;
      order[orderLen++] = idx;
      roots.push(idx);
    }
  }
}

// Multi-source BFS
while (qh < qt) {
  const idx = queue[qh++];
  const x = idx % width;
  const y = (idx / width) | 0;
  let nidx: number;

  if (x > 0) {
    nidx = idx - 1;
    if (passable[nidx] && dist[nidx] === -1) {
      dist[nidx] = dist[idx] + 1;
      parent[nidx] = idx;
      queue[qt++] = nidx;
      order[orderLen++] = nidx;
    }
  }
  if (x + 1 < width) {
    nidx = idx + 1;
    if (passable[nidx] && dist[nidx] === -1) {
      dist[nidx] = dist[idx] + 1;
      parent[nidx] = idx;
      queue[qt++] = nidx;
      order[orderLen++] = nidx;
    }
  }
  if (y > 0) {
    nidx = idx - width;
    if (passable[nidx] && dist[nidx] === -1) {
      dist[nidx] = dist[idx] + 1;
      parent[nidx] = idx;
      queue[qt++] = nidx;
      order[orderLen++] = nidx;
    }
  }
  if (y + 1 < height) {
    nidx = idx + width;
    if (passable[nidx] && dist[nidx] === -1) {
      dist[nidx] = dist[idx] + 1;
      parent[nidx] = idx;
      queue[qt++] = nidx;
      order[orderLen++] = nidx;
    }
  }
}

const sum = new Float64Array(n);
sum.set(townValue);

// Compute subtree sums using BFS order reversed
for (let i = orderLen - 1; i >= 0; i--) {
  const idx = order[i];
  const p = parent[idx];
  if (p >= 0) sum[p] += sum[idx];
}

// Build child lists
const headChild = new Int32Array(n);
headChild.fill(-1);
const nextSibling = new Int32Array(n);
for (let i = 0; i < orderLen; i++) {
  const idx = order[i];
  const p = parent[idx];
  if (p >= 0) {
    nextSibling[idx] = headChild[p];
    headChild[p] = idx;
  }
}

// Euler tour for subtree intervals
const entry = new Int32Array(n);
entry.fill(-1);
const exit = new Int32Array(n);
exit.fill(-1);
let time = 0;
const stackN: number[] = [];
const stackS: number[] = [];

for (const root of roots) {
  stackN.push(root);
  stackS.push(0);
  while (stackN.length > 0) {
    const node = stackN.pop()!;
    const state = stackS.pop()!;
    if (state === 0) {
      entry[node] = time++;
      stackN.push(node);
      stackS.push(1);
      for (let ch = headChild[node]; ch !== -1; ch = nextSibling[ch]) {
        stackN.push(ch);
        stackS.push(0);
      }
    } else {
      exit[node] = time - 1;
    }
  }
}

class Fenwick {
  size: number;
  tree: Int32Array;
  constructor(n: number) {
    this.size = n + 2;
    this.tree = new Int32Array(this.size);
  }
  update(i: number, delta: number) {
    for (i += 1; i < this.size; i += i & -i) this.tree[i] += delta;
  }
  query(i: number) {
    let res = 0;
    for (i += 1; i > 0; i -= i & -i) res += this.tree[i];
    return res;
  }
  rangeAdd(l: number, r: number, delta: number) {
    if (l > r) return;
    this.update(l, delta);
    this.update(r + 1, -delta);
  }
}

// Candidate cells
const candidates: number[] = [];
for (let i = 0; i < orderLen; i++) {
  const idx = order[i];
  if (sum[idx] > 0 && passable[idx] && !isIgnition[idx] && !isTown[idx]) {
    candidates.push(idx);
  }
}

candidates.sort((a, b) => sum[b] - sum[a]);

const selected: number[] = [];
if (cutBudget > 0 && candidates.length > 0 && time > 0) {
  const bit = new Fenwick(time);
  for (const idx of candidates) {
    if (selected.length >= cutBudget) break;
    const e = entry[idx];
    if (e < 0) continue;
    if (bit.query(e) > 0) continue; // blocked by ancestor cut
    selected.push(idx);
    bit.rangeAdd(e, exit[idx], 1);
  }
}

const firebreaks = selected.map((idx) => [idx % width, Math.floor(idx / width)]);

process.stdout.write(JSON.stringify({ firebreaks }));
